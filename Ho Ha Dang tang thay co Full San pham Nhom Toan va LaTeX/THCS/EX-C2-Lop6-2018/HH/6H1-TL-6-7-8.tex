	\subsection{Tóm tắt lý thuyết}
		\begin{enumerate}
			\item Đoạn thẳng $ AB $ là hình gồm điểm $ A $, điểm $ B $ và tất cả các điểm nằm giữa $ A $ và $ B $.
			\item Khi hai đoạn thẳng hoặc một đoạn thẳng với một đường thẳng hoặc một đoạn thẳng với một tia chỉ có một điểm chung, ta nói chúng cắt nhau.
			\item Nhận xét: Mỗi đoạn thẳng có một độ dài. Độ dài đoạn thẳng là một số lớn hơn $ 0 $.
			\item So sánh hai đoạn thẳng
				\begin{itemize} 
				\item Nếu độ dài của hai đoạn thẳng $ AB $ và $ CD $ bằng nhau thì $ AB = CD $.
				\item Nếu độ dài đoạn thẳng $ AB $ lớn hơn độ dài đoạn thẳng $ CD $ thì $ AB > CD $ hay $ CD < AB $.
				\end{itemize}
		\end{enumerate}

		\begin{enumEX}{1}
			\item Nếu $ n = 14 $ thì có bao nhiêu đoạn thẳng tạo thành?
			\item Nếu có $ 210 $ đoạn thẳng tạo thành thì $ n $ là bao nhiêu?

		\begin{enumEX}{1}

		\end{enumEX}
	}
	\end{vd}

\section{KHI NÀO THÌ $AM + MB = AB$?}
	\subsection{Tóm tắt lý thuyết.}
	\begin{enumerate}
		\item Nếu điểm $ M $ nằm giữa hai điểm $ A $ và $ B $ thì $ AM + MB = AB $.
		\immini{Ngược lại, nếu $ AM + MB = AB $ thì điểm $ M $ nằm giữa hai điểm $ A $ và $ B $.}
		{
			\begin{tikzpicture}
				\tkzDefPoints{0/0/A,2/0/M,6/0/B}
				\tkzDrawPoints(A,B,M)
				\tkzDrawSegments(A,B)
				\tkzLabelPoints[below](A,M,B)
			\end{tikzpicture}
		}
	\item \textbf{\textit{Chú ý}}
	\begin{itemize} 
		\item \immini{Ta có thể dùng mệnh đề sau tương đương với tính chất trên:\\
		Nếu $ AM + MB \neq AB $ thì điểm $ M $ không nằm giữa $ A $ và $ B $.}
		{
			\begin{tikzpicture}
			\tkzDefPoints{0/0/A,2/0/M,4/0/N,6/0/B}
			\tkzDrawPoints(A,B,M,N)
			\tkzDrawSegments(A,B)
			\tkzLabelPoints[below](A,M,B,N)
			\end{tikzpicture}
		}
		\item Cộng liên tiếp.\\
		Nếu $ M $ nằm giữa $ A $ và $ B $; $ N $ nằm giữa $ M $ và $ B $ thì $ AM + MN + NB = AB $.
	\end{itemize}
	\end{enumerate}

	\subsection{Các dạng bài tập và phương pháp giải}
	\begin{dang}{Tính độ dài đoạn thẳng}
		Vận dụng: Nếu điểm $ M $ nằm giữa $ A $ và $ B $ thì $ AM + MB = AB $.
	\end{dang}
	\begin{vd}
		Cho điểm $ M $ nằm giữa hai điểm $ A $ và $ B $. Biết $ AB = 6 $ cm; $ AM = 2 $ cm. Tính độ dài $ BM $.
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
				\tkzDefPoints{0/0/A,2/0/M,6/0/B}
				\tkzDrawPoints(A,B,M)
				\tkzDrawSegments(A,B)
				\tkzLabelPoints[below](A,M,B)
			\end{tikzpicture}
		\end{center}
		$ M $ nằm giữa $ A $ và $ B $, ta có:
		\begin{align*}
		& AM + MB = AB \\
		& 2 + MB = 6 \\
		& MB = 4 \ (\text{cm}).
		\end{align*}
	}
	\end{vd}

	\begin{vd}
		Cho điểm $ I $ nằm giữa $ C $ và $ D $. Biết rằng $ IC - ID = 1 $cm và $ CD = 5 $cm. Tính độ dài $ IC $; $ ID $.
		\loigiai{
		\begin{center}
			\begin{tikzpicture}
				\draw (0,0) node [below] {$C$} circle(1pt) -- (3,0) node [below] {$I$} circle(1pt) -- (5,0) node [below] {$D$} circle(1pt) circle(1pt) ;
			\end{tikzpicture}
		\end{center}
		$ I $ nằm giữa $ C $ và $ D $, ta có: 
		\begin{align*}
		& IC + ID = CD \\
		& IC + ID = 5 .
		\end{align*}
		Mà $ IC - ID = 1 $.\\
		Do đó 
		\begin{align*}
			IC = (5 + 1):2 = 3 \ (\text{cm})\\
			ID = (5 - 1):2 = 2 \ (\text{cm}).
		\end{align*}
	}
	\end{vd}

	\begin{dang}{Xác định điểm nằm giữa hai điểm khác}
		Nếu $ AM + MB = AB $ thì điểm $ M $ nằm giữa hai điểm $ A $ và $ B $.
	\end{dang}
	
	\begin{vd}
		Cho ba điểm $ A $, $ B $, $ C $, biết $ AC = 2 $ cm, $ BC = 3 $ cm, $ AB = 5 $ cm. Điểm nào nằm giữa hai điểm còn lại?
		\loigiai{
			Ta có: $ AC + BC = 2 + 3 = 5 $.\\
			Suy ra: $ AC + BC = AB $.\\
			Do đó điểm $ C $ nằm giữa $ A $ và $ B $.\\
			\textit{\textbf{Nhận xét:}} Bạn nên thực hiện tổng hai số đo đoạn thẳng nhỏ $ AC $, $ BC $ sau đó so sánh kết quả với số đo đoạn lớn nhất $ AB $. Nếu bằng nhau thì điểm $ C $ nằm giữa hai điểm $ A $ và $ B $.
		}
	\end{vd}

	\begin{vd}
		Cho ba điểm $ A $, $ B $, $ C $, biết $ AB = 3{,}1 $ cm, $ AC = 1{,}9 $ cm, $ BC = 4 $ cm.
		\begin{enumEX}{1}
			\item Chứng tỏ trong ba điểm $ A $, $ B $, $ C $ không có điểm nào nằm giữa hai điểm còn lại.
			\item Chứng tỏ ba điểm $ A $, $ B $, $ C $ không thẳng hàng.
		\end{enumEX}
		\loigiai{
			\begin{enumEX}{1}
				\item Ta có: $ AC + AB = 1{,}9 + 3{,}1 = 5 $ cm.\\
				Suy ra $ AC + AB \neq BC $ nên $ A $ không nằm giữa $ B $ và $ C $.\\
				Ta có: $ AC + BC = 1{,}9 + 4 = 5{,}9 $ cm.\\
				Suy ra $ AC + BC \neq AB $ nên $ C $ không nằm giữa $ A $ và $ B $.\\
				Ta có: $ AB + BC = 3{,}1 + 4 = 7{,}1 $ cm.\\
				Suy ra $ AB + BC \neq AC $ nên $ B $ không nằm giữa $ A $ và $ C $.
				\item $ A $, $ B $, $ C $ không có điểm nào nằm giữa hai điểm còn lại. Do vậy ba điểm $ A $, $ B $, $ C $ không thẳng hàng.
			\end{enumEX}
		Nhận xét: Trong ba điểm, nếu không có điểm nào nằm giữa hai điểm còn lại thì ba điểm đó không thẳng hàng.
		}
	\end{vd}

	\subsection{Bài tập tự luyện}
	\begin{bt}%[6H1B8]%[Lục Hán Tường, dự án EX-C2-L6]%
		Cho điểm $ M $ thuộc đoạn thẳng $ AB $. Biết $ AM = 2 $ cm; $ AB = 5 $ cm. Tính độ dài đoạn $ MB $.
		\loigiai{
			\begin{center}
				\begin{tikzpicture}
					\tkzDefPoints{0/0/A,2/0/M,5/0/B}
					\tkzDrawPoints(A,B,M)
					\tkzDrawSegments(A,B)
					\tkzLabelPoints[below](A,M,B)
				\end{tikzpicture}
			\end{center}
			$ M $ nằm giữa $ A $ và $ B $, ta có:
			\begin{align*}
				& AM + MB = AB \\
				& 2 + MB = 5 \\
				& MB = 3 \ (\text{cm}).
			\end{align*}
		}
	\end{bt}

	\begin{bt}%[6H1B8]%[Lục Hán Tường, dự án EX-C2-L6]%
		Cho $ I $ là một điểm thuộc đoạn thẳng $ EF $. Biết $ EI = 5 $ cm; $ IF = 4 $ cm. Tính độ dài đoạn $ EF $.
		\loigiai{
			\begin{center}
				\begin{tikzpicture}
				\tkzDefPoints{0/0/E,9/0/F,5/0/I}
				\tkzDrawPoints(E,I,F)
				\tkzDrawSegments(E,F)
				\tkzLabelPoints[below](E,I,F)
				\end{tikzpicture}
			\end{center}
			$ I $ nằm giữa $ E $ và $ F $, ta có:
			\begin{align*}
			EF = EI + IF = 5 + 4 = 9 \ (\text{cm}).
			\end{align*}
		}
	\end{bt}

	\begin{bt}%[6H1B8]%[Lục Hán Tường, dự án EX-C2-L6]%
		Ba điểm $ A $, $ B $, $ C $ có thẳng hàng không? Nếu biết
		\begin{enumEX}{1}
			\item $ AB = 3{,}6 $ cm; $ AC = 2{,}6 $ cm; $ BC = 6 $ cm.
			\item $ AB = 4 $ cm; $ AC = 7 $ cm; $ BC = 3 $ cm.
		\end{enumEX}
		\loigiai{
		\begin{enumEX}{1}
			\item Ta có: $ AC + AB = 2{,}6 + 3{,}6 = 6{,}2 $ cm.\\
			Suy ra $ AC + AB \neq BC $ nên $ A $ không nằm giữa $ B $ và $ C $.\\
			Ta có: $ AC + BC = 2{,}6 + 6 = 8{,}6 $ cm.\\
			Suy ra $ AC + BC \neq AB $ nên $ C $ không nằm giữa $ A $ và $ B $.\\
			Ta có: $ AB + BC = 3{,}6 + 6 = 9{,}6 $ cm.\\
			Suy ra $ AB + BC \neq AC $ nên $ B $ không nằm giữa $ A $ và $ C $.\\
			Do $ A $, $ B $, $ C $ không có điểm nào nằm giữa hai điểm còn lại. Do vậy ba điểm $ A $, $ B $, $ C $ không thẳng hàng.
			
			\item Ta có: $ AB + BC = 4 + 3 = 7 $ cm.\\
			Suy ra $ AB + BC = AC $ nên $ B $ nằm giữa $ A $ và $ C $ hay ba điểm $ A $, $ B $, $ C $ thẳng hàng.
		\end{enumEX}
		}
	\end{bt}

	\begin{bt}%[6H1K8]%[Lục Hán Tường, dự án EX-C2-L6]%
		Cho hai điểm $ A $, $ B $ sao cho $ AB = 8 $ cm. Trên đường thẳng $ AB $ lấy điểm $ M $ sao cho $ MB = 3 MA $. Tính độ dài $ MA $.
		\loigiai{
			\begin{enumEX}[Trường hợp 1:]{1}
				\item Điểm $ M $ nằm giữa $ A $ và $ B $.
				\begin{center}
					\begin{tikzpicture}
						\tkzDefPoints{0/0/A,2/0/M,8/0/B}
						\tkzDrawPoints(A,B,M)
						\tkzDrawSegments(A,B)
						\tkzLabelPoints[below](A,M,B)
					\end{tikzpicture}
				\end{center}
				Ta có
				\begin{align*}
					& MA + BM = AB \\
					& MA + 3MA = AB \\
					& 4MA = 8 \\
					& MA = 2 \ (\text{cm}).
				\end{align*}
				
				\item Điểm $ A $ nằm giữa $ M $  và $ B $.
				\begin{center}
					\begin{tikzpicture}
						\tkzDefPoints{0/0/A,-4/0/M,8/0/B}
						\tkzDrawPoints(A,B,M)
						\tkzDrawSegments(M,B)
						\tkzLabelPoints[below](A,M,B)
					\end{tikzpicture}
				\end{center}
				Ta có
				\begin{align*}
					& MA + AB = BM \\
					& MB - MA = AB \\
					&  3MA - MA = AB \\ 
					& 2MA = 8 \\
					& MA = 4 \ (\text{cm}).
				\end{align*}
				
				\item Điểm $ B $ nằm giữa $ M $ và $ A $.\\
				Ta có $ AM > BM $ mà $ BM = 3AM > AM \Rightarrow $ vô lý. TRường hợp này không xảy ra.
			\end{enumEX}
		}
	\end{bt}

	\begin{bt}%[6H1B8]%[Lục Hán Tường, dự án EX-C2-L6]%
		Cho điểm $ C $ nằm giữa $ A $ và $ B $; điểm $ D $ nằm giữa $ C $ và $ B $. Biết rằng $ AB = 7 $ cm, $ AC = 3 $ cm, $ BD = 1 $ cm. Tính $ CD $.
		\loigiai{
			\begin{center}
				\begin{tikzpicture}
				\tkzDefPoints{0/0/A,3/0/C,6/0/D,7/0/B}
				\tkzDrawPoints(A,B,C,D)
				\tkzDrawSegments(A,B)
				\tkzLabelPoints[below](A,B,C,D)
				\end{tikzpicture}
			\end{center}
			Ta có $ C $ nằm giữa $ A $ và $ D $, $ D $ nằm giữa $ C $ và $ B $ nên
			\begin{align*}
				& AB = AC + CD + DB \\ & 7 = 3 + CD + 1 \\ & CD = 3 \ (\text{cm}).
			\end{align*}
		}
	\end{bt}
