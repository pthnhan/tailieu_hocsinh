\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Cần Giờ, Hồ Chí Minh]%[Phạm Tuấn Hùng-9DT19]%[9D1B9]
	Thu gọn các biểu thức sau:
	\begin{enumerate}
		\item $A=\sqrt{8}-\dfrac{1}{3}\sqrt{18}+\sqrt{50}$.
		\item $B=\sqrt{16-6\sqrt{7}}+\sqrt{(2-\sqrt{7})^2}$.
		\item $C=\sqrt{10}+\sqrt{2}\cdot \sqrt{3-\sqrt{5}}$.
		\item $D=\dfrac{2}{\sqrt{5}+1}+\sqrt{\dfrac{2}{3-\sqrt{5}}}$.
	\end{enumerate}

	\loigiai{
		\begin{enumerate}
			\item $A=\sqrt{8}-\dfrac{1}{3}\sqrt{18}+\sqrt{50}=2\sqrt{2}-\dfrac{1}{3}\cdot 3\sqrt{2}+5\sqrt{2}=2\sqrt{2}-\sqrt{2}+5\sqrt{2}=6\sqrt{2}$.
			\item $B=\sqrt{16-6\sqrt{7}}+\sqrt{(2-\sqrt{7})^2}=\sqrt{(3-\sqrt{7})^2}+\sqrt{(2-\sqrt{7})^2}=3-\sqrt{7}+\sqrt{7}-2=1$.
			\item $C=\sqrt{10}+\sqrt{2}\cdot \sqrt{3-\sqrt{5}}=\sqrt{10}+\sqrt{6-2\sqrt{5}}=\sqrt{10}+\sqrt{(\sqrt{5}-1)^2}=\sqrt{10}+\sqrt{5}-1$.
			\item
			 \begin{eqnarray*}
			 	D&=&\dfrac{2}{\sqrt{5}+1}+\sqrt{\dfrac{2}{3-\sqrt{5}}}\\
			 	&=&\dfrac{2(\sqrt{5}-1)}{4}+\sqrt{\dfrac{2(3+\sqrt{5})}{4}}\\
			 	&=&\dfrac{\sqrt{5}-1+\sqrt{(\sqrt{5}+1)^2}}{2}\\
			 	&=&\sqrt{5}.
			 \end{eqnarray*}
			
		\end{enumerate}
	
	}
\end{bt}

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Cần Giờ, Hồ Chí Minh]%[Phạm Tuấn Hùng-9DT19]%[9D2B5]
	Cho hàm số $y=x-2$ có đồ thị là $(d_1)$ và hàm số $y=-2x+1$ có đồ thị là $(d_2)$.
	\begin{enumerate}
		\item Vẽ $(d_1)$ và $(d_2)$ trên cùng một mặt phẳng tọa độ $Oxy$.
		\item Xác định các hệ số $a$, $b$ của đường thẳng $(d_3):y=ax+b$. Biết $(d_3)$ song song $(d_1)$ và $(d_3)$ cắt $(d_2)$ tại điểm có tung độ bằng $3$.
	\end{enumerate}
	
	\loigiai{
		\begin{enumerate}
			\item Xét đồ thị hàm số $y=x-2$ ta có $a=1>0$ nên đồ thị hàm số đồng biến trên $\mathbb{R}$. Ta có bảng giá trị
			\begin{center}
				\begin{tabular}{|c|c|c|}
					\hline $x$ & $2$ & $0$ \\ 
					\hline $y=x-2$ & $0$ & $-2$ \\ 
					\hline
				\end{tabular}
			\end{center}
			Vậy đồ thị hàm số $y=x-2$ là đường thẳng nối hai điểm có tọa độ $(1;-1)$ và $(0;-2)$. \\
			
			Xét đồ thị hàm số $y=-2x+1$ ta có $a=-2<0$ nên đồ thị hàm số nghịch biến trên $\mathbb{R}$. Ta có bảng giá trị
			\begin{center}
				\begin{tabular}{|c|c|c|}
					\hline $x$ & $0$ & $2$ \\ 
					\hline $y=-2x+1$ & $1$ & $-3$ \\ 
					\hline
				\end{tabular}
			\end{center}
			Vậy đồ thị hàm số $y=x-2$ là đường thẳng nối hai điểm có tọa độ $(1;-1)$ và $(2;0)$. \\
		
	
			\begin{center}
				\begin{tikzpicture}[scale=1, line join=round, line cap=round, >=stealth]
				%Phan Khai Bao Mac Dinh
				\tkzInit[ymin=-4.5,ymax=4.5,xmin=-4.5,xmax=4.5]
				\tkzAxeXY
				%\tkzGrid
				\tkzClip
				\tkzSetUpPoint[color=black,fill=black,size=6]
				%Dinh Hinh
				\tkzDefPoints{2/0/A,0/-2/B,0/1/C,2/-3/D}
				\tkzDrawLines[add=5 and 5](A,B C,D)
				\tkzDrawPoints(A,B,C,D)
				\tkzLabelLine[pos=2,sloped,above](A,B){$y=x-2$}
				\tkzLabelLine[pos=-0.5,sloped,below](C,D){$y=-2x+1$}
				\end{tikzpicture}
			\end{center}
			\item Ta có đồ thị hàm số $y=ax+b$ song song với $(d_1)$ nên có hệ số góc bằng với hàm số $y=x-2$. Vậy hàm số của đường thẳng $(d_3)$ có dạng $y=x+b$. Ta đi xác định hệ số $b$.\\
			Gọi $A(x_A;3)$ là điểm có tung độ bằng $3$ và thuộc đồ thị $(d_2)$. Ta có:
			$$3=-2x_A+1\Leftrightarrow x_A=\dfrac{3-1}{-2}=-1$$
			Vậy $A(-1;3)$ cũng thuộc đồ thị $(d_3)$. Ta có
			$$3=-1+b\Leftrightarrow b=4$$
			Vậy hàm số của đồ thị $(d_3)$ là $y=x+4$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Cần Giờ, Hồ Chí Minh]%[Phạm Tuấn Hùng-9DT19]%[7D1B2]
	Nhân dịp Tết Dương lịch $2019$, siêu thị Nguyễn Kim bán hàng khuyến mãi giá một chiếc tivi hiệu TOSHIBA $42$ inch là $7\ 200\ 000$ đồng. Lần $1$ giảm $10\%$, lần $2$ giảm thêm $5\%$ trên giá trị còn lại lần $1$. Hỏi sau $2$ lần giảm giá thì giá của $1$ chiếc tivi là bao nhiêu?
	\loigiai{
		Giá tiền của chiếc tivi sau khi giảm giá hai lần là:
		$$7\ 200\ 000\cdot (100\%-10\%)\cdot(100\%-5\%)=7\ 200\ 000\cdot90\%\cdot95\%=6\ 156\ 000 \text{ (đồng)}$$
	}
\end{bt}

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Cần Giờ, Hồ Chí Minh]%[Phạm Tuấn Hùng-9DT19]%[9H1K2]
	Từ một ngon hải đăng $AB$ cao $75$ m, người ta quan sát $2$ lần thấy một chiếc thuyền đang tiến về ngọn hải đăng theo một đường thẳng với các góc hạ lần lượt là $30^\circ$ (góc $\widehat{XAD}$) và $60^\circ$ (góc $\widehat{XAC}$). Hỏi chiếc thuyền đi được bao nhiêu mét giữa $2$ lần quan sát. 
	\begin{center}

	\loigiai{
		\begin{itemize}
			\item Ta có $\widehat{BAD}=90^\circ-\widehat{XAD}=60^\circ$.
			\item Xét $\triangle ABD$ vuông tại $B$ ta có $\tan 60^\circ=\dfrac{BD}{AB}$. Vậy
			$$BD=AB\cdot \tan 60^\circ=75\sqrt{3}.$$
			\item Ta có $\widehat{BAC}=90^\circ-\widehat{XAC}=30^\circ$.
			\item Xét $\triangle ABC$ vuông tại $B$ ta có $\tan 30^\circ=\dfrac{BC}{AB}$. Vậy
			$$BC=AB\cdot \tan 30^\circ=75\dfrac{\sqrt{3}}{3}=25\sqrt{3}.$$
		\end{itemize}
		Vậy quãng đưỡng thuyền di chuyển được là $$CD=BD-BC=75\sqrt{3}-25\sqrt{3}=50\sqrt{3}.$$
	}

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Cần Giờ, Hồ Chí Minh]%[Phạm Tuấn Hùng-9DT19]%[9H2K6]
	Từ điểm $A$ bên ngoài đường tròn $(O)$ vẽ hai tiếp tuyến $AB$ và $AC$ với $(O)$ ($B$, $C$ là tiếp điểm). Gọi $H$ là giao điểm của $AO$ và $BC$.
	\begin{enumerate}
		\item Chứng minh $AO$ là đường trung trực của $BC$.
		\item Gọi $D$ là điểm trên cung nhỏ $BC$ của đường tròn $(O)$. Tiếp tuyến tại $D$ của đường tròn $(O)$ cắt $AB$, $AC$ lần lượt tại $M$, $N$. Chứng minh rằng chu vi của tam giác $\triangle AMN$ bằng $AB+AC$.
		\item Đường thẳng $AD$ cắt $(O)$ tại điểm thứ hai $E$. Chứng minh $\widehat{AHD}=\widehat{AEO}$.
	\end{enumerate}
\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=1, line join=round, line cap=round, >=stealth]
			%Phan Khai Bao Mac Dinh
			\tkzInit[ymin=-0.5,ymax=6.5,xmin=-0.5,xmax=10.5]
			%\tkzAxeXY
			%\tkzGrid
			\tkzClip
			\tkzSetUpPoint[color=black,fill=black,size=6]
			%Dinh Hinh

			\tkzTangent[from=A](O,a)\tkzGetPoints{B}{C}
			\tkzInterLL(O,A)(B,C)\tkzGetPoint{H}
			\tkzDefPointBy[rotation= center O angle -47](B) \tkzGetPoint{D}
			\tkzDefLine[orthogonal=through D](O,D) \tkzGetPoint{d}
			\tkzInterLL(D,d)(A,B)\tkzGetPoint{M}
			\tkzInterLL(D,d)(A,C)\tkzGetPoint{N}
			\tkzInterLC(A,D)(O,a)\tkzGetPoints{E}{}
			\tkzDrawCircle[radius](O,a)
			
			\tkzMarkRightAngles[fill= gray!20](O,B,A B,H,A)
			
			
			\tkzDrawPoints(O,A,B,C,H,D,M,N,E)
			\tkzDrawSegments(O,A A,B A,C B,C M,N A,E B,D E,O H,D O,B E,B)
			
			\tkzLabelPoints[above](E)
			\tkzLabelPoints[below](A,C,N,H)

			\tkzLabelPoints[above right](B,M)
			\tkzLabelPoints[above left]()
			\tkzLabelPoints[below right]()

			\end{tikzpicture}

	\begin{enumerate}
		\item Ta có
				\begin{itemize}
					\item $OB=OC$ (bán kính đường tròn tâm $O$).
					\item $AB=AC$ (tính chất hai tiếp tuyến cắt nhau tại $A$).
				\end{itemize}
			Vậy $OA$ là đường trung trực của đoạn thẳng $BC$ (Tập hợp tất cả các điểm cách đều hai đầu mút của đoạn thẳng $BC$).
		\item Ta có 
				\begin{itemize}
					\item $MD=MB$ (tính chất hai tiếp tuyến cắt nhau tại $M$).
					\item $NC=ND$ (tính chất hai tiếp tuyến cắt nhau tại $N$).
				\end{itemize}
			Vậy $P_{\triangle AMN}=AM+AN+MN=AM+AN+MD+ND=AM+MN+BM+NC=AB+AC$.
		\item Ta có
			\begin{itemize}
				\item Xét $\triangle OBA$ vuông tại $B$ có đường cao $BH$. Theo hệ thức lượng ta có $AH\cdot AO=AB^2$.
				\item Xét $\triangle ABD \sim \triangle AEB$ (góc-góc) ta có $AD\cdot AE=AB^2$.
			\end{itemize}
			Vậy $AH\cdot AO=AD\cdot AE \Rightarrow \dfrac{AH}{AE}=\dfrac{AD}{AO}$. Vậy $\triangle AHD\sim \triangle AEO$ (cạnh-góc-cạnh). Vậy $\widehat{AHD}=\widehat{AEO}$.
	\end{enumerate}
	}
