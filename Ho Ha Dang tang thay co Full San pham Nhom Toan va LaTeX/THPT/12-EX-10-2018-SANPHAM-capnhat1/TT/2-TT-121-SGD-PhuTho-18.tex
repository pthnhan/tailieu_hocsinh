\begin{ex}%[Đề thi thử THPTQG, 2018, SGD Phú Thọ]%[Trần Bá Huy, 12-EX-10]%[2D1Y4-1]
Đồ thị hàm số nào dưới đây \textbf{không} có tiệm cận đứng?
\choice
{$y=\dfrac{x^2+1}{x+1}$}
{\True $y=\dfrac{x^2+3x+2}{x+1}$}
{$y=\dfrac{x-1}{x+1}$}
{$y=\dfrac{2}{x+1}$}
\loigiai{
Ta có\\
$\displaystyle\lim\limits_{x\to (-1)^{+}}\dfrac{x^2+1}{x+1}=+\infty$, suy ra $x=-1$ là tiệm cận đứng.\\
$\displaystyle\lim\limits_{x\to (-1)^{+}}\dfrac{x-1}{x+1}=-\infty$, suy ra $x=-1$ là tiệm cận đứng.\\
$\displaystyle\lim\limits_{x\to (-1)^{+}}\dfrac{2}{x+1}=+\infty$, suy ra $x=-1$ là tiệm cận đứng.\\
$\displaystyle\lim\limits_{x\to -1}\dfrac{x^2+3x+2}{x+1}=\displaystyle\lim\limits_{x\to -1}\dfrac{(x+1)(x+2)}{x+1}=\displaystyle\lim\limits_{x\to -1}(x+2)=1$, suy ra đồ thị của hàm số $y=\dfrac{x^2+3x+2}{x+1}$ không có tiệm cận đứng.
}
\end{ex}

\begin{ex}%[Đề thi thử THPTQG, 2018, SGD Phú Thọ]%[Trần Bá Huy, 12-EX-10]%[2D3B2-3]
Cho hàm số $y=f(x)$ liên tục trên đoạn $[1;2]$ và $\displaystyle\int\limits_{1}^{2}(x-1)f'(x)\mathrm{\,d}x=a$. Tính $\displaystyle\int\limits_{1}^{2}f(x)\mathrm{\, d}x$ theo $a$ và $b=f(2)$.
\choice
{$a-b$}
{$a+b$}
{\True $b-a$}
{$-b-a$}
\loigiai{
Áp dụng công thức $\displaystyle\int\limits_{\alpha}^{\beta}u\mathrm{\,d}v=uv\Big|_{\alpha}^{\beta}-\displaystyle\int\limits_{\alpha}^{\beta}v\mathrm{\, d}u$, ta có
\begin{align*}
a=\ &\displaystyle\int\limits_{1}^{2}(x-1)f'(x)\mathrm{\,d}x=\displaystyle\int\limits_{1}^{2}(x-1)\mathrm{\,d}\left(f(x)\right)=(x-1)f(x)\Big|_{1}^{2}-\displaystyle\int\limits_{1}^{2}f(x)\mathrm{\,d}x\\
=\ & f(2)-\displaystyle\int\limits_{1}^{2}f(x)\mathrm{\,d}x=b-\displaystyle\int\limits_{1}^{2}f(x)\mathrm{\,d}x.
\end{align*}
Từ đó suy ra $\displaystyle\int\limits_{1}^{2}f(x)\mathrm{\,d}x=b-a$.
}
\end{ex}
