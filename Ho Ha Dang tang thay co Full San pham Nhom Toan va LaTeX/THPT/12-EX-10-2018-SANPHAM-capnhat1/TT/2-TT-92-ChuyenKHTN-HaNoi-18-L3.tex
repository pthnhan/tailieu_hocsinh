\begin{ex}%[TT, THPT Chuyên KHTN, Hà Nội]%[Bùi Quốc Hoàn, dự án EX10]%[1D1K3-2]
Phương trình $4\sin^{2}2x  -  3\sin 2x\cos 2x - \cos^{2}2x = 0$ có bao nhiêu nghiệm trong khoảng $\left(0;\pi\right)$?
	\choice
	{$1$}
	{$2$}
	{$3$}
	{\True $4$}
	\loigiai{Ta có 
			\begin{align*}
	   { } & 4\sin^{2}2x  -  3\sin 2x\cos 2x - \cos^{2}2x = 0\\
	   \Leftrightarrow &  4\sin^{2}2x  -  4\sin 2x\cos 2x + \sin 2x\cos 2x - \cos^{2}2x = 0\\
	   \Leftrightarrow & 4\sin 2x\left(\sin 2x - \cos 2x  \right) + \cos 2x\left(\sin 2x - \cos 2x  \right) = 0\\
	    \Leftrightarrow & \left(\sin 2x - \cos 2x  \right)\left(4\sin 2x + \cos 2x  \right) = 0\\
	     \Leftrightarrow &\hoac{&\sin 2x - \cos 2x = 0\\& 4\sin 2x + \cos 2x = 0}
		\end{align*}
	- Khi $\sin 2x - \cos 2x = 0$, dễ thấy $\cos 2x = 0$ không là nghiệm nên
	$$\sin 2x - \cos 2x = 0\Leftrightarrow \sin 2x = \cos 2x\Leftrightarrow \tan 2x = 1\Leftrightarrow  2x = \dfrac{\pi}{4} + k\pi\Leftrightarrow x = \dfrac{\pi}{8} + k\dfrac{\pi}{2}\ \left(k\in\mathbb{Z}\right).$$
	Do $x\in \left(0;\pi\right)$ suy ra $k = 0$; $k= 1$.\\
	- Khi $4\sin 2x + \cos 2x = 0$, dễ thấy $\cos 2x = 0$ không là nghiệm nên
		\begin{align*}
	{ }&4\sin 2x + \cos 2x = 0\Leftrightarrow 4\sin 2x = - \cos 2x\\
	\Leftrightarrow &\tan 2x = - \dfrac{1}{4}\Leftrightarrow  2x = \arctan \left(- \dfrac{1}{4}\right) + k\pi\\
	\Leftrightarrow &x = \dfrac{1}{2}\arctan \left(- \dfrac{1}{4}\right) + k\dfrac{\pi}{2}\ \left(k\in\mathbb{Z}\right).
	\end{align*}
	Do $x\in \left(0;\pi\right)$ suy ra $k = 0$; $k= 1$.\\
	Vậy phương trình có $4$ nghiệm thuộc $\left(0;\pi\right)$.
	}
\end{ex}

	$$2\left(\mathrm{C}^{0}_{2n} + \mathrm{C}^{2}_{2n}x^2 + \cdots + \mathrm{C}^{2n - 2}_{2n}x^{2n - 2} + \mathrm{C}^{2n}_{2n}x^{2n}\right) =  \left(1 + x\right)^{2n} + \left(1 - x\right)^{2n}\quad (*)$$
Lấy tích phân hai vế của $(*)$ ta có 
		$$2\displaystyle\int\limits_{0}^{1}\left(\mathrm{C}^{0}_{2n} + \mathrm{C}^{2}_{2n}x^2 + \cdots + \mathrm{C}^{2n - 2}_{2n}x^{2n - 2} + \mathrm{C}^{2n}_{2n}x^{2n}\right)\, \mathrm{d}x  =  \displaystyle\int\limits_{0}^{1}\left[\left(1 + x\right)^{2n} + \left(1 - x\right)^{2n}\right]\, \mathrm{d}x\quad (**)$$

	\begin{align*}
{ }&2\displaystyle\int\limits_{0}^{1}\left(\mathrm{C}^{0}_{2n} + \mathrm{C}^{2}_{2n}x^2 + \cdots + \mathrm{C}^{2n - 2}_{2n}x^{2n - 2} + \mathrm{C}^{2n}_{2n}x^{2n}\right)\, \mathrm{d}x \\

=  &2\left(\mathrm{C}^{0}_{2n} + \dfrac{\mathrm{C}^{2}_{2n}}{3} + \dfrac{\mathrm{C}^{4}_{2n}}{5} + \dfrac{\mathrm{C}^{6}_{2n}}{7} + \cdots + \dfrac{\mathrm{C}^{2n - 2}_{2n}}{2n - 1} + \dfrac{\mathrm{C}^{2n}_{2n}}{2n + 1}\right)
\end{align*}
Mặt khác 
$$\displaystyle\int\limits_{0}^{1}\left[\left(1 + x\right)^{2n} + \left(1 - x\right)^{2n}\right]\, \mathrm{d}x = \left[\dfrac{\left(1 + x\right)^{2n+ 1}}{2n + 1} - \dfrac{\left(1 - x\right)^{2n + 1}}{2n + 1}\right] \Bigg\vert_{0}^{1} = \dfrac{2^{2n + 1}}{2n + 1}.$$
Từ $(**)$ ta suy ra 
	\begin{align*}
{ }&2\left(\mathrm{C}^{0}_{2n} + \dfrac{\mathrm{C}^{2}_{2n}}{3} + \dfrac{\mathrm{C}^{4}_{2n}}{5} + \dfrac{\mathrm{C}^{6}_{2n}}{7} + \cdots + \dfrac{\mathrm{C}^{2n - 2}_{2n}}{2n - 1} + \dfrac{\mathrm{C}^{2n}_{2n}}{2n + 1}\right) = \dfrac{2^{2n + 1}}{2n + 1}\\

\end{align*}
Do đó $\dfrac{2^{2n}}{2n + 1} = \dfrac{8192}{15}\Leftrightarrow \dfrac{2^{2n}}{2n + 1} = \dfrac{2^{13}}{15}\Leftrightarrow 15\cdot 2^{2n - 13} = 2n + 1$.\\
