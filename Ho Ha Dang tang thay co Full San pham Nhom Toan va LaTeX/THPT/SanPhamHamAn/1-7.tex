\begin{ex}%[WTT-218cau-ham-an-2D1, Nguyễn Kim Đông]%[2D1G1-2]
\immini{	Cho hàm số $f(x)$ xác định trên $\mathbb{R}$ và có đồ thị hàm số $f'(x)$ là đường cong trong hình bên. Mệnh đề nào dưới đây đúng?
	\choice
	{Hàm số $f(x)$ nghịch biến trên khoảng $(-1;1)$}
	{Hàm số $f(x)$ đồng biến trên khoảng $(1; 2)$}
	{Hàm số $f(x)$ đồng biến trên khoảng $(-2;1)$}
	{\True Hàm số $f(x)$ nghịch biến trên khoảng $(0; 2)$}}
{\begin{tikzpicture}[>=stealth,line cap=round,line join=round,x=1cm,y=1cm,scale=0.6]
	\draw[->] (-3,0)--(3,0) node[below] {$x$};
	\draw[->] (0,-4)--(0,4) node[right] {$y$};
	\draw[thick,samples=1000,domain=-2.25:2.25] plot(\x,{(\x)^3-4*(\x)});
	\draw (0.3,0)node[above] {$O$};
	\draw (-2,0)node[above left] {$-2$};
	\draw (2,0)node[below right] {$2$};
	\draw[fill=black] (-2,0) circle (1.5pt);
	\draw[fill=black] (2,0) circle (1.5pt);
	\end{tikzpicture}}
	\loigiai{
		\textbf{Cách 1}: sử dụng bảng biến thiên. Từ đồ thị của hàm số $y=f'(x)$ ta có bảng biến thiên như sau:
				\begin{center}
			\begin{tikzpicture}[>=stealth]
			\tkzTabInit[lgt=1.5,espcl=2]
			{$x$/.7,$g'$/.7,$g$/2}
			{$-\infty$,$-2$,$0$,$2$,$+\infty$}
			\tkzTabLine{,-,z,+,z,-,z,+,} 
			\tkzTabVar{+/,-/,+/,-/,+/}
			\end{tikzpicture}
		\end{center}
		\textbf{Cách 2}: Quan sát đồ thị hàm số $y=f'(x)$ \\
		Nếu trong khoảng $K$ đồ thị hàm số $f'(x)$ nằm trên trục hoành (có thể tiếp xúc) thì $f(x)$ đồng biến trên $K$.\\
		Nếu trong khoảng $K$ đồ thị hàm số $f'(x)$ nằm dưới trục hoành (có thể tiếp xúc) thì $f(x)$ nghịch biến trên $K$.\\
		Nếu trong khoảng $K$ đồ thị hàm số $f'(x)$ vừa có phần nằm dưới trục hoành vừa có phần nằm trên trục hoành thì loại phương án đó.\\
		Trên khoảng $(0; 2)$ ta thấy đồ thị hàm số $y=f'(x)$ nằm bên dưới trục hoành.}
\end{ex}
