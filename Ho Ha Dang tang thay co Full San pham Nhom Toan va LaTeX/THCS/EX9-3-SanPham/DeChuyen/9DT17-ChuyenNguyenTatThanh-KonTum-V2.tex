		\item[b)] ĐKXĐ: $ x\ge 0; y\ge 0 $.

	\end{enumerate}
}

\begin{ex}%[Đề thi vào 10 chuyên Nguyễn Tất Thành, Kon Tum 2017]%[9D1K7]%[9D5G1]
	Cho biểu thức $M=\dfrac{y}{\sqrt{xy}-x}+\dfrac{x}{\sqrt{xy}+y}-\dfrac{x+y}{\sqrt{xy}}$ $\left(x>y>0 \right). $
	\begin{enumerate}
		\item Rút gọn $M.$
		\item Tìm giá trị nhỏ nhất của $N=x^2-\dfrac{M}{x+y},$ với $x>y>0.$
	\end{enumerate}
\loigiai
{\begin{enumerate}
		\item[a)] Với $ x>y>0 $,
		\\$ M=\cfrac{y\sqrt{xy}+y^2+x\sqrt{xy}-x^2}{(y-x)\sqrt{xy}}-\cfrac{x+y}{\sqrt{xy}}=\cfrac{(x+y)(\sqrt{xy}+y-x)}{(y-x)\sqrt{xy}}-\cfrac{(x+y)(y-x)}{(y-x)\sqrt{xy}}=\cfrac{x+y}{y-x}. $
		\item[b)] Với $ x>y>0 $,
		\\ $ N=x^2-\cfrac{1}{y(y-x)}=x^2+\cfrac{1}{y(x-y)}\ge x^2+\cfrac{1}{\cfrac{(y+x-y)^2}{4}}=x^2+\cfrac{4}{x^2} \ge 2\sqrt{x^2.\cfrac{4}{x^2}}=4 .$
		\\	Dấu $ ``=" $ xảy ra khi và chỉ khi $ \heva{&y=x-y\\&x^2=\cfrac{4}{x^2}\\&x>y>0} \Leftrightarrow \heva{&x=\sqrt{2} \\&y=\cfrac{\sqrt{2}}{2}}. $
\end{enumerate}}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Nguyễn Tất Thành, Kon Tum 2017]%[9D5G5]
	Xét 2020 số thực $x_1$, $x_2$, $...$, $x_{2020}$ chỉ nhận một trong hai giá trị $2-\sqrt{3}$ và $2+\sqrt{3}$. Hỏi biểu thức $\sum\limits_{k = 1}^{1010} {{x_{2k - 1}}{x_{2k}}} $ nhận bao nhiêu giá trị nguyên khác nhau?
	\loigiai{Vì các số $ x_i $ chỉ nhận một trong hai giá trị $ 2-\sqrt{3} $ và $ 2+\sqrt{3} $ \\
		$	\Rightarrow \hoac{&x_{2k-1}.x_{2k}= (2-\sqrt{3})(2+\sqrt{3})=1 \\&x_{2k-1}.x_{2k}=(2-\sqrt{3})^2=7-4\sqrt{3} \\&x_{2k-1}.x_{2k}=(2+\sqrt{3})^2=7+4\sqrt{3} .}  $
		\\ Do đó để $P=\sum\limits_{k = 1}^{1010} {x_{2k-1}.x_{2k}}   $ nhận giá trị nguyên thì chỉ xảy ra 2 trường hợp sau:
		\begin{itemize}
		\item TH1: $ x_{2k-1}.x_{2k}= (2-\sqrt{3})(2+\sqrt{3}) $ với $ \forall  k $.
		\\ Khi đó $ P=1010 $.
		\item TH2: $ \heva{& m  \text{ giá trị }  k  \text{ thỏa mãn } x_{2k-1}.x_{2k}=(2-\sqrt{3})^2 \\& m  \text{ giá trị }  k  \text{ thỏa mãn } x_{2k-1}.x_{2k}=(2+\sqrt{3})^2  \\& 1010-2m  \text{ giá trị }  k  \text{ thỏa mãn } x_{2k-1}.x_{2k}=(2-\sqrt{3})(2+\sqrt{3})} $ với $ 1\le m \le 505 $.
		\\ Khi đó $ P=1010+12m $ với $ 1\le m \le 505 $.
		\\ Vậy $ P $ nhận $ 506 $ giá trị nguyên khác nhau.
	\end{itemize}
	}

\begin{ex}%[Đề thi vào 10 chuyên Nguyễn Tất Thành, Kon Tum 2017]%[9D4G6]
	Cho hàm số $y=f(x)=x^2+mx+m-13.$ Gọi $x_1$, $x_2$ là hai nghiệm phân biệt của phương trình $f(x)=0$ với $m>13$. Tìm $m$ thỏa mãn điều kiện
	\begin{center}
		$ \left| {{x_1}} \right|.f({x_2} - m) + \left| {{x_2}} \right|.f({x_1} - m) = 104. $
	\end{center}
\loigiai{
	$ f(x)=x^2+mx+m-13 $ (với $ m>13 $) .
	\\$ x_1,x_2 $ là 2 nghiệm phân biệt của đa thức $ f(x)=0 $ $ \Rightarrow \heva{&x_1+x_2=-m\\&x_1x_2=m-13.} $
	\\ Vì $ m>13 $ và $x_1+x_2 >0$ nên $ x_1<0; x_2<0 $.
	\\Có $ f(x-m)=(x-m)^2+m(x-m)+m-13=x^2-mx+m-13=f(x)-2mx $.
	\\$ \Rightarrow \heva{& f(x_1-m)=-2mx_1\\& f(x_2-m)=-2mx_2 .}$
	\\Theo giả thiết $ P= |x_1|f(x_2-m)+|x_2|f(x_1-m)=104$.
	\\$\Leftrightarrow |x_1|.(-2mx_2)+|x_2|.(-2mx_1)=104 $.
	\\$ \Leftrightarrow -2mx_2.(-x_1)-2mx_1.(-x_2)=104$.
	\\$ \Leftrightarrow mx_1x_2=26 \Leftrightarrow m(m-13)=26 \Leftrightarrow m=\cfrac{13+\sqrt{273}}{2}$ (vì $ m>13 $).
}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Nguyễn Tất Thành, Kon Tum 2017]%[9D5G1]
	Với $a$, $b$, $c$ là các số thực dương thỏa mãn $a+b+c=1$. Tìm giá trị lớn nhất của biểu thức 
	\[ P=6(ab+bc+ca)+a(a-b)^2 +b(b-c)^2+c(c-a)^2.\]
	\loigiai{
		Có $ 0<a,b,c<1 $.
		\\Do $ 2ab=a^2+b^2-(a-b)^2 $
		\\ $
		\begin{aligned}
		\Rightarrow 6(ab+bc+ca)&=2(a^2+b^2+c^2)-\left[(a-b)^2+(b-c)^2+(c-a)^2\right]+2(2ab+2bc+2ca)
		\\&=2(a+b+c)^2-\left[(a-b)^2+(b-c)^2+(c-a)^2\right] .
		\end{aligned} $
		\\Do đó $ P=2+(a-1)(a-b)^2+(b-1)(b-c)^2+(c-1)(c-a)^2 $.
		\\Lại có $ a<1 $ suy ra $ (a-1)(a-b)^2 \le 0 $.
		\\Dấu $ ``="  $ xảy ra khi và chỉ khi $ a=b $.
		\\Tương tự với $ (b-1)(b-c)^2 $ và $ (c-1)(c-a)^2 $.
		\\ Vậy $ P \le 2 $.
		\\Dấu $ ``="  $ xảy ra khi và chỉ khi $ \heva{&a=b=c\\&a+b+c=1} $ hay $ a=b=c=\cfrac{1}{3} $.
	}
\end{ex}
