\section{PHƯƠNG TRÌNH ĐƯỜNG THẲNG TRONG KHÔNG GIAN}
\subsection{Tóm tắt lí thuyết}
\begin{itemize}
	\item
Phương trình tham số của đường thẳng $d$ qua $M(x_0;y_0;z_0)$ và có véc-tơ chỉ phương $\vec{u}=(a;b;c)$ là $ d: \begin{cases}
x=x_0+at,\\
y=y_0+bt,\\
z=z_0+ct.
\end{cases}$
\item 
Phương trình chính tắc của đường thẳng $d$ qua $M(x_0;y_0;z_0)$ và có véc-tơ chỉ phương $\vec{u}=(a;b;c)$ là $d: \dfrac{x-x_0}{a}=\dfrac{y-y_0}{b}=\dfrac{z-z_0}{c}$ với $abc\ne 0$.
\end{itemize}

\subsection{Các dạng toán}

\begin{bt}%[Mai Hà Lan]%[2H3K3]
    Viết phương trình đường thẳng $d$. Biết $d$ đi qua giao điểm của hai đường thẳng $\Delta: \dfrac{x - 1}{2} = \dfrac{y + 2}{-3} = \dfrac{z}{1}$ và $\Delta': \dfrac{x - 3}{5} = \dfrac{y + 5}{1} = \dfrac{z-1}{7}$. Và song song với hai mặt phẳng \begin{align*}
            (P): &  7x - 10y + 5z + 1 = 0  & (Q): & 3x + 6y - 2z - 2018 = 0
        \end{align*}
        \loigiai{
        Ta có $(P)$, $(Q)$ có véc tơ pháp tuyến lần lượt là $\vec{n_1} = (7;-10;5)$, $\vec{n_2} = (3;6;-2)$.

	\loigiai{Nhận xét, do $\overrightarrow{n_P}\perp \overrightarrow{u_d'}$ nên $d'\parallel (P)$, do đó ta chỉ cần tìm hình chiếu vuông góc của điểm $A(1;1;1)$ lên $(P)$ là điểm $A'(\dfrac{1}{3};\dfrac{1}{3};\dfrac{1}{3})$, sau đó viết phương trình $d$ qua $A'$ nhận $\overrightarrow{u_d'}$ làm $vec-tơ$ chỉ phương $d:\dfrac{x-\frac{1}{3}}{2}=\dfrac{y-\frac{1}{3}}{-1}=\dfrac{z-\frac{1}{3}}{-1}$          
}
\end{vd}

\begin{bt}%[Vũ Văn Trường]%[2H3Y3]
Trong  không gian $Oxyz$, viết phương trình  hình chiếu vuông  góc  của  đường thẳng  $d:\dfrac{x+1}{2}=\dfrac{y-2}{3}=\dfrac{z+3}{1}$ trên mặt phẳng  tọa độ $(Oxy)$.
 \loigiai{
Trên đường thẳng  $d$ lấy hai điểm $A(-1;2;-3)$  và  $B(1;5;-2)$.\\
Gọi $A', B'$ lần lượt là  hình chiếu của  $A,B$  xuống  mặt phẳng  $(Oxy)$ suy ra  $A'(-1;2;0)$ và  $B'(1;5;0)$.\\
Khi đó hình  chiếu $d'$  của  $d$  xuống  $(Oxy)$  qua  hai điểm $A',B'$.\\
Đường thẳng $A',B'$ có phương trình tham số $\heva{&x=-1+2t\\&y=2+3t\\&z=0}$
 }
\end{bt}
\begin{bt}%[Vũ Văn Trường]%[2H3Y3]
Trong không gian với hệ tọa độ $Oxyz$, cho hai điểm $A(1;2;-3), B(2;5;7)$. Viết phương trình hình chiếu vuông góc của $d$ trên mặt phẳng $(Oxz)$.
\loigiai{Gọi $A', B'$ lần lượt là hình chiếu vuông góc của $A, B$ trên mặt phẳng $(Oxz)$. Dễ thấy $A'(1;0;-3), B'(2;0;7)$ và $\overrightarrow{A'B'}=(1;0;10)$.\\
Đường thẳng $A'B'$ chính là hình chiếu vuông góc của đường thẳng $d$ trên mặt phẳng $(Oxy)$. Đường thẳng $A'B'$ có phương trình $\heva{&x=1+t\\&y=0\\&z=-3+10t.}$
}
\end{bt}
\begin{bt}%[Vũ Văn Trường]%[2H3B3]
Trong không gian với hệ tọa độ $Oxyz$, cho hai đường thẳng $d$ và mặt phẳng $(P)$ có phương trình $d:  \heva{&x=4+2t \\ &y=2+3t \\ &z=-1-2t} (t \in \mathbb{R})$, $(P): 3x+y-z-4=0$. Viết phương trình hình chiếu vuông góc của $d$ trên $(P)$.
\loigiai{
Dễ thấy $d$ cắt mặt phẳng $(P)$ tại điểm $A(2;-1;1)$ và $B(4;2;-1)\in d$.\\
Gọi $B'$ là hình chiếu vuông góc của $B$ trên $(P)$, khi đó ta có $B'(1;1;0)$.\\
Đường thẳng $AB'$ chính là hình chiếu vuông góc của đường thẳng $d$ trên mặt phẳng $(P)$. Đường thẳng $AB'$ có phương trình $\dfrac{x-1}{1}=\dfrac{y-1}{-2}=\dfrac{z}{1}$.
}
\end{bt}
\begin{bt}%[Vũ Văn Trường]%[2H3B3]
Trong không gian với hệ tọa độ $Oxyz$, cho đường thẳng $d: \dfrac{x-1}{2}=\dfrac{y-3}{-1}=\dfrac{z}{2}$. Viết phương trình hình chiếu vuông góc $d'$ của $d$ trên mặt phẳng $(P): 3x-y+z-9=0$.
\loigiai{Gọi $(Q)$ là mặt phẳng đi qua $d$ và vuông góc với $(P)$, khi đó $d'=(P)\cap (Q)$.
Ta có $\overrightarrow{u}=(2;-1;2)$ là véc-tơ chỉ phương của $d$; $\overrightarrow{n}=(3;-1;1)$ là véc-tơ pháp tuyến của $(P)$.\\
Do đó $\overrightarrow{n'}=\left[\overrightarrow{u},\overrightarrow{n}\right]=(1;4;1)$ là véc-tơ pháp tuyến của $(Q)$.\\
Từ đó suy ra $\overrightarrow{u'}=\left[\overrightarrow{n'},\overrightarrow{n}\right]=(5;2;-13)$ là véc-tơ chỉ phương của $d'$.\\
Ta có $M(3;2;2)$ là giao điểm của $d$ và $(P)$, do đó $d')$ có phương trình: $\dfrac{x-3}{5}=\dfrac{y-2}{2}=\dfrac{z-2}{-13}$.

}
\end{bt}
\begin{bt}%[Vũ Văn Trường]%[2H3K3]
Trong không gian với hệ tọa độ $Oxyz$, cho hình chóp $S.ABCD$ có đáy là hình bình hành tâm $I$. Biết $A(1;2;1$, $B(2;3;0)$, $D(-2;1;2)$ và $S(0;4;3)$. Gọi $M$ là trung điểm $SB$ và $G$ là trọng tâm tam giác $SBD$. Viết phương trình hình chiếu vuông góc của đường thẳng $MG$ trên mặt phẳng $(ABCD)$.
\loigiai{
Ta có $I(0;2;1), \overrightarrow{AB}=(1;1;-1), \overrightarrow{AD}=(-3;-1;1)$ và $\overrightarrow{IS}=(0;2;2)$.\\
Dễ thấy $\overrightarrow{IS}\cdot \overrightarrow{AB}=\overrightarrow{IS}\cdot \overrightarrow{AD}=0$ nên $SI\perp (ABCD)$.\\
Từ đó suy ra $BD$ chính là hình chiếu vuông góc của $MG$ trên mặt phẳng $(ABCD)$.\\
Từ đó suy ra hình chiếu vuông góc của đường thẳng $MG$ trên mặt phẳng $(ABCD)$ có phương trình $\dfrac{x-2}{2}=\dfrac{y-3}{1}=\dfrac{z}{-1}$.
}
\end{bt}
