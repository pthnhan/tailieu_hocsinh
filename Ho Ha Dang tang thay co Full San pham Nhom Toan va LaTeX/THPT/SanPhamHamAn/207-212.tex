\begin{ex}%[2D1G5-2]%[2D1G5-2]%[Dự án WTT-Nguyễn Anh Quốc]
	Cho hàm số $y=f(x)$ có đạo hàm $f'(x)=x(x-1)^2(x^2+mx+9)$ với mọi $x\in \mathbb{R}$ và $m$ là tham số. Có bao nhiêu số nguyên dương $m$ để hàm số $g(x)=f(3-x)$ đồng biến trên khoảng $(3;+\infty)$?	
	\choice
	{$5$}
	{\True $6$}
	{$7$}
	{$8$}
	\loigiai{
		Từ giả thiết suy ra $f'(3-x)=(3-x)(2-x)^2\left[(3-x)^2+m(3-x)+9 \right]$.\\
		Ta có $g'(x)=-f'(3-x)$.\\
		Do đó hàm số $g(x)$ đồng biến trên $(3;+\infty)$ khi và chỉ khi $g'(x)\ge 0,\ \forall x\in (3;+\infty)$\\
		$\Leftrightarrow f'(3-x)\ge 0,\ \forall x\in (3;+\infty)\\ \Leftrightarrow (3-x)(2-x)^2\left[(3-x)^2+m(3-x)+9 \right] \ge 0,\ \forall x\in (3;+\infty)\\ \Leftrightarrow m\le \dfrac{(x-3)^2+9}{x-3}, \ \forall x\in (3;+\infty) \Leftrightarrow m\le \min\limits_{(3;+\infty)} h(x)$ với $h(x)=\dfrac{(x-3)^2+9}{x-3}$.\\
		Theo bất đẳng thức Cô-si ta có	$h(x)=x-3+\dfrac{9}{x-3}\ge 2\sqrt{(x-3) \cdot \dfrac{9}{x-3}}=6$, $\forall x\in(3;+\infty)$\\
		Suy ra $\min\limits_{(3;+\infty)} h(x)=6$ khi $x=6$. Do đó $m \le 6\Rightarrow m\in \left\lbrace 1,2,3,4,5,6 \right\rbrace$.
	}
\end{ex}
\begin{ex}%[2D1G5-2]%[2D1G5-2]%[Dự án WTT-Nguyễn Anh Quốc]
	Cho hàm số $y=f(x)$ có đạo hàm $f'(x)=x^2(x-1)(x^2+mx+5)$, $\forall x\in \mathbb{R}$ và $m$ là tham số. Có bao nhiêu số nguyên âm $m$ để hàm số $g(x)=f(x^2)$ đồng biến trên $(1;+\infty)$?	
	\choice
	{$3$}
	{\True $4$}
	{$5$}
	{$7$}
	\loigiai{
		Từ giả thiết suy ra $f'(x^2)=x^4(x^2-1)(x^4+mx^2+5)$.\\
		Ta có $g'(x)=2xf'(x^2)$.\\
		Để hàm số $g(x)$ đồng biến trên khoảng $(1;+\infty)$ khi và chỉ khi $g'(x)\ge 0$, $\forall x\in (1;+\infty) \\\Leftrightarrow 2xf'(x^2)\ge 0,\  \forall x\in (1;+\infty) \Leftrightarrow 2x\cdot x^4 (x^2-1)(x^4+mx^2+5)\ge 0,\ \forall x\in (1;+\infty) \\\Leftrightarrow x^4+mx^2+5\ge 0, \ \forall x\in (1;+\infty) \Leftrightarrow m \ge \dfrac{-x^4-5}{x^2},\ \forall x\in (1;+\infty) \\\Leftrightarrow m\ge \max\limits_{(1;+\infty)} h(x)$ với $h(x)=\dfrac{-x^4-5}{x^2}$.\\
		Khảo sát hàm số $h(x)=\dfrac{-x^4-5}{x^2}$ trên $(1;+\infty)$ ta được $\max\limits_{(1;+\infty)} h(x)=-2\sqrt{5}$.\\ Suy ra $m\ge -2\sqrt{5}\Rightarrow m\in \left\lbrace -4,-3,-2,-1 \right\rbrace$.
	}
\end{ex}
\begin{ex}%[2D1G5-2]%[2D1G5-2]%[Dự án WTT-Nguyễn Anh Quốc]
	Cho hàm số $y=f(x)$ có đạo hàm $f'(x)=x(x-1)^2(3x^4+mx^3+1)$, với mọi $x\in \mathbb{R}$ và $m$ là tham số. Có bao nhiêu số nguyên âm để hàm số $g(x)=f(x^2)$ đồng biến trên khoảng $(0;+\infty)$.	
	\choice
	{$3$}
	{\True $4$}
	{$5$}
	{$6$}
	\loigiai{
		Từ giả thiết suy ra $f'(x^2)=x^2(x^2-1)^2(3x^8+mx^6+1)$.\\
		Ta có $g'(x)=2xf'(x^2)$.\\
		Để hàm số $g(x)$ đồng biến trên khoảng $(0;+\infty)$ khi và chỉ khi $g'(x)\ge 0$, $\forall x\in (0;+\infty)$ $\Leftrightarrow 2xf'(x^2)\ge0$, $\forall x\in (0;+\infty)$ $\Leftrightarrow 2x\cdot x^2 (x^2-1)^2(3x^8+mx^6+1)\ge 0$, $\forall x\in (0;+\infty)$ $\Leftrightarrow 3x^8+mx^6+1\ge 0$, $\forall x\in (0;+\infty)\\\Leftrightarrow m\ge \dfrac{-3x^8-1}{x^6} \Leftrightarrow m\ge \max\limits_{(0;+\infty)} h(x)$ với $h(x)=\dfrac{-3x^8-1}{x^6}$.\\
		Khảo sát hàm $h(x)=\dfrac{-3x^8+1}{x^6}$ trên $(0;+\infty)$ ta được $\max\limits_{(0;+\infty)} h(x)=-4$. Suy ra $m\ge -4$. Vậy $m\in \left\lbrace -4;-3;-2;-1\right\rbrace$.
	}
\end{ex}
\begin{ex}%[2D1G5-2]%[2D1G5-2]%[Dự án WTT-Nguyễn Anh Quốc]
	Cho hàm số $f(x)$ có đạo hàm $f'(x)=(x-1)^2(x^2-2x)$ với mọi $x\in \mathbb{R}$ và $m$ là tham số. Có bao nhiêu số nguyên $m<100$ để hàm số $g(x)=f(x^2-8x+m)$ đồng biến trên khoảng $(4;+\infty)$?	
	\choice
	{$18$}
	{\True $82$}
	{$83$}
	{$84$}
	\loigiai{
		Ta có $f'(x)=(x-1)^2(x^2-2x)>0 \Leftrightarrow \hoac{&x<0\\&x>2}$.\\
		Xét hàm số $g'(x)=(2x-8)f'(x^2-8x+m)$.\\
		Để hàm số $g(x)$ đồng biến trên khoảng $(4;+\infty)$ khi và chỉ khi $g'(x)\ge 0$, $ \forall x>4$ \\$\Leftrightarrow (2x-8)f'(x^2-8x+m)\ge 0$, $x>4 \Leftrightarrow f'(x^2-8x+m)\ge 0,\ \forall x>4 \\\Leftrightarrow \hoac{&x^2-8x+m\le 0,\ \forall x\in (0;+\infty)\\&x^2-8x+m\ge 2,\ \forall x\in (0;+\infty)}$ $\Leftrightarrow x\ge 18$.\\
		Vậy $18\le m <100$.
	}
\end{ex}
