\begin{bt}%[HK2, Sở giáo dục, Bắc Ninh, 2017]%[1D4B1-3]%[Nguyễn Vân Trường, 11-HK2]
Tính  giới hạn 
 $\underset{}{\lim}\dfrac{n+1}{3n+2}.$

	\loigiai{
$\underset{}{\lim}\dfrac{n+1}{3n+2}=\underset{}{\lim}\dfrac{1+\dfrac{1}{n}}{3+\dfrac{2}{n}}=\dfrac{1}{3}.$

}
\end{bt}


\begin{bt}%[HK2, Sở giáo dục, Bắc Ninh, 2017]%[1D4B1-4]%[Nguyễn Vân Trường, 11-HK2]
Tính các giới hạn 
 $\underset{}{\lim}\left(\sqrt{4n^2+n+1}-2n\right).$

	\loigiai{
$\underset{}{\lim}\left(\sqrt{4n^2+n+1}-2n\right)=\underset{}{\lim}\dfrac{n+1}{\sqrt{4n^2+n+1}+2n}=\underset{}{\lim}\dfrac{1+\dfrac{1}{n}}{\sqrt{4+\dfrac{1}{n}+\dfrac{1}{n^2}}+2}=\dfrac{1}{4}.$

}
\end{bt}


\begin{bt}%[HK2, Sở giáo dục, Bắc Ninh, 2017]%[1D5K2-2]%[Nguyễn Vân Trường, 11-HK2]
	Cho hàm số $y=x^3-3x+1$ có đồ thị $(C)$. Viết phương trình tiếp tuyến của đồ thị $(C)$ tại điểm $M$ thuộc $(C)$ biết khoảng cách từ $M$ đến trục $Oy$ bằng $1$.
	\loigiai{
Tập xác định là $\mathbb{R}$, $y' = 3x^2-3$. 
Gọi $M(x_0;y_0)$ là tiếp điểm của tiếp tuyến cần tìm. 
Theo giả thiết $\mathrm{d}(M;Oy) =1 \Leftrightarrow |x_0| =1 \Leftrightarrow x_0 = \pm 1.$ \\
{\bf TH1}: $x_0=1 \Rightarrow y_0=-1;y'(x_0) = 0$. Phương trình tiếp tuyến cần tìm là $d: y=-1$.\\
{\bf TH2}: $x_0=-1 \Rightarrow y_0=3;y'(x_0) = 0$. Phương trình tiếp tuyến cần tìm là $d: y=3$.
}
\end{bt}

\begin{bt}%[HK2, Sở giáo dục, Bắc Ninh, 2017]%[1D5B2-1]%[Nguyễn Vân Trường, 11-HK2]
Cho hàm số $y=x^4-2x^2-3$. Giải phương trình $y'=0$.
	\loigiai{

 Ta có $y'=4x^3-4x$. Khi đó $y'=0 \Leftrightarrow 4x^3-4x=0 \Leftrightarrow \hoac{ & x = 0 \\ & x =-1 \\ & x =1.}$ \\
Vậy phương trình có các nghiệm $x=0,x=1$ và $x=-1$.
}
\end{bt}

\begin{bt}%[HK2, Sở giáo dục, Bắc Ninh, 2017]%[1D4K3-3]%[Nguyễn Vân Trường, 11-HK2]

 Tìm $a$ để hàm số $f(x) = \heva{&\dfrac{\sqrt{3x+1}-\sqrt[3]{2x+6}}{x-1} & \text{ nếu } x \ne 1 \\ & a & \text{ nếu } x=1}$ liên tục tại điểm $x_0=1.$

	\loigiai{
	
 \begin{align*}
\underset{x \rightarrow 1}{\lim} f(x) & = \underset{x \rightarrow 1}{\lim} \dfrac{\sqrt{3x+1}-\sqrt[3]{2x+6}}{x-1}=  \underset{x \rightarrow 1}{\lim} \left[\dfrac{\sqrt{3x+1}-2}{x-1}-\dfrac{\sqrt[3]{2x+6}-2}{x-1}\right]\\
&=  \underset{x \rightarrow 1}{\lim} \left[\dfrac{3}{\sqrt{3x+1}+2}-\dfrac{2}{\sqrt[3]{(2x+6)^2}+2\sqrt[3]{2x+6}+4}\right] = \dfrac{7}{12}
\end{align*}
và $f(1) = a$. Để hàm số liên tục tại $x=1$ thì $ \underset{x \rightarrow 1}{\lim}  f(x) = f(1) \Leftrightarrow a = \dfrac{7}{12}.$ Vậy $a = \dfrac{7}{12}.$
}
\end{bt}




\begin{bt}%[HK2, Sở giáo dục, Bắc Ninh, 2017]%[1H3B3-2]%[1H3K5-3]%[1H3K4-3]%[Nguyễn Vân Trường, 11-HK2]
	Cho hình chóp $S{.}ABCD$ có đáy $ABCD$ là hình vuông tâm $O$, cạnh $a$, $SA = SB = SC = SD = \dfrac{a\sqrt{5}}{2}.$
\begin{enumerate}
\item Chứng minh đường thẳng $SO$ vuông góc với mặt phẳng $(ABCD)$.
\item  Tính khoảng cách từ điểm $O$ đến mặt phẳng $(SBC)$.
\item Tính côsin của góc giữa hai mặt phẳng $(SAB)$ và $(SBC)$.
\end{enumerate}
	\loigiai{
\immini{
\begin{enumerate}
\item Ta có $SA=SC \Rightarrow \triangle SAC$ cân $ SO \perp AC.$ Chứng minh tương tự ta có $SO \perp BD$. Do đó $SO \perp (ABCD)$.
\item Gọi $K$ là trung điểm của $BC$. \\
Khi đó, $OK \perp BC, SO\perp BC \Rightarrow BC \perp (SOK).$\\
Kẻ $OH \perp SK (H \in SK) \Rightarrow OH \perp (SBC)$\\
$ \Rightarrow \mathrm{d}(O,(SBC)) =OH$.\\
Ta có $OK = \dfrac{a}{2}, BD = a\sqrt{2}$\\
$ \Rightarrow SO =\sqrt{SB^2-OB^2} = \dfrac{a\sqrt{3}}{2}$.\\
Mặt khác $\dfrac{1}{OH^2}=\dfrac{1}{SO^2}+\dfrac{1}{KO^2} \Rightarrow OH = \dfrac{a \sqrt{3}}{4}.$
\end{enumerate}

\begin{enumerate}
\item[c)] Ta có $(SAB)\cap (SBC) = SB.$ Ta cũng có  $AC \perp BD$ và $AC \perp SO \Rightarrow$ $AC \perp (SBD)$.\\
Kẻ  $AM \perp SB (M \in SB) \Rightarrow SB \perp (AMC)$. Do đó góc giữa mặt phẳng $(SAB)$ và $(SBC)$ bằng góc giữa $AM$ và $CM$. Ta có
$AM=CM=\dfrac{SK\cdot BC}{SB} = \dfrac{2a\sqrt{5}}{5}$.\\
Áp dụng định lý côsin ta có $\cos \widehat{AMC} =\dfrac{AM^2+CM^2-AC^2}{2AM \cdot CM}=-\dfrac{1}{4}.$\\

\begin{bt}%[HK2, Sở giáo dục, Bắc Ninh, 2017]%[1D3G4-5]%[Nguyễn Vân Trường, 11-HK2]
	Cho $n$ là số nguyên dương và $n$ tam giác $A_1B_1C_1,A_2B_2C_2,...,A_nB_nC_n$, trong đó các điểm $A_{i+1},B_{i+1},C_{i+1}$ lần lượt nằm trên các cạnh $B_iC_i,A_iC_i,A_iB_i$ $(i = \overline{1,n-1})$ sao cho $A_{i+1}C_i = 2A_{i+1}B_i,B_{i+1}A_i = 2B_{i+1}C_i, C_{i+1}B_i=2C_{i+1}A_i$. Gọi $S$ là tổng tất cả diện tích của $n$ tam giác trên, biết rằng tam giác $A_1B_1C_1$ có diện tích bằng $2$. Hãy tính $S$ theo $n$ và tìm số nguyên dương $n$ sao cho $S=3\left(1-\dfrac{1}{3^{2017}}\right)$.
	\loigiai{ 
\immini{
Gọi $S_i \left(i=\overline{1,n}\right)$ là diện tích của $\triangle A_iB_iC_i$. \\
Ta có $\dfrac{S_{A_1B_2C_2}}{S_{A_1B_1C_1}}= \dfrac{A_1B_2}{A_1C_1}\cdot \dfrac{A_1C_2}{A_1B_1}=\dfrac{1}{3}\cdot \dfrac{2}{3}=\dfrac{2}{9}.$\\
Tương tự ta có $\dfrac{S_{ A_2B_1C_2}}{S_{A_1B_1C_1}}= \dfrac{S_{A_2B_2C_1}}{S_{ A_1B_1C_1}}=\dfrac{2}{9}.$ \\
Do đó $\dfrac{S_{A_2B_2C_2}}{S_{A_1B_1C_1}}=1-3\cdot\dfrac{2}{9}=\dfrac{1}{3} \Rightarrow S_2=\dfrac{1}{3}S_1.$}{

\noindent   Tương tự ta có $S_{i+1}= \dfrac{1}{3}S_i$ với mọi $\overline{1,n-1}$.  Khi đó 
\begin{center}
$S=S_1\left( 1+\dfrac{1}{3}+\cdots+\dfrac{1}{3^{n-1}}\right)  = 2\cdot \dfrac{1-\dfrac{1}{3^n}}{1-\dfrac{1}{3}}=3\left(1-\dfrac{1}{3^n}\right).$ 
\end{center}
Theo giả thiết ta có $3\left(1-\dfrac{1}{3^n}\right)=3\left(1-\dfrac{1}{3^{2017}}\right) \Leftrightarrow n = 2017.$
}
