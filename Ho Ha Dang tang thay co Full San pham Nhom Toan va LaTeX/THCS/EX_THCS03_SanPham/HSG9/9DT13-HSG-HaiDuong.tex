\begin{ex}%[9D1K8]
	Cho biểu thức $A=\dfrac{\sqrt{x+4\sqrt{x-4}}+\sqrt{x-4\sqrt{x-4}}}{\sqrt{1-\dfrac{8}{x}+\dfrac{16}{x^2}}}$. 
	\begin{enumerate}
	\item Rút gọn $A$. 
	\item Tìm các giá trị nguyên của $x$ để $A$ có giá trị nguyên.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item Điều kiện $x>4$.\\
	Ta có $A=\dfrac{\sqrt{x-4}+2+\left|\sqrt{x-4}-2\right|}{\dfrac{x-4}{x}}$.\\
	Nếu $x>8$ thì $A=\dfrac{2x}{\sqrt{x-4}}$.\\
	Nếu $4<x\le 8$ thì $A=\dfrac{4x}{x-4}$.
	\item Với $A=\dfrac{4x}{x-4}4+\dfrac{16}{x-4}$. $A\in \Bbb Z\Leftrightarrow 16 \,\, \vdots \,\,(x-4)$. Do $x\in \Bbb Z, 4 <x \le 8$ nên $x\in \{5,6,8\}$.\\
	Với $A=\dfrac{2x}{\sqrt{x-4}}\in \Bbb Z$ thì do $x\in \Bbb Z$ nên $\sqrt{x-4}=a\in \Bbb Z \Rightarrow A=\dfrac{2(a^2+4)}{a}=2a+\dfrac{8}{a}\Rightarrow 8 \,\, \vdots \,\,a$.\\
	Lại có $x>8\Rightarrow a>2$, do đó $a=4$ hoặc $a=8$. Từ đó suy ra $x=20$ hoặc $x=68$.\\
	Vậy $x\in \{5,6,8,20,68\}$.
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[9D4G7]
Giải các phương trình
	\begin{enumerate}
		\item $\sqrt{x^2-3x+2}+\sqrt{x+3}=\sqrt{x-2}+\sqrt{x^2+2x-3}$.
		\item $(4x+2)\sqrt{x+8}=3x^2+7x+8$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
		\item Điều kiện $x\ge 2$. Phương trình đã cho tương đương với
		\allowdisplaybreaks{
		\begin{align*}
		&\sqrt{(x-1)(x-2)}+\sqrt{x+3}=\sqrt{x-2}+\sqrt{(x-1)(x+3)}\\
		\Leftrightarrow &\left(\sqrt{x-2}-\sqrt{x+3}\right)\left(\sqrt{x-1}-1\right)=0\\
		\Leftrightarrow & \sqrt{x-2}=\sqrt{x+3}\,\, \text{hoặc}\,\,\sqrt{x-1}=1
		\end{align*}}
		Giải từng trường hợp ta được nghiệm là $x=2$.\\
		Vậy phương trình có nghiệm $x=2$.
		\item Điều kiện $x\ge -8$. Phương trình đã cho tương đương với 
		\allowdisplaybreaks{
		\begin{align*}
		&\left(\sqrt{x+8}-x-2\right)\left(\sqrt{x+8}-3x\right)=0\\
		\Leftrightarrow & \sqrt{x+8}=x+2\,\, \text{hoặc}\,\,\sqrt{x+8}=3x
		\end{align*}}
		\begin{itemize}
		\item Trường hợp 1 $\sqrt{x+8}=x+2\Leftrightarrow \begin{cases} x\ge -2 \\ x^2+3x-4=0\end{cases} \Leftrightarrow x =1.$
		\item Trường hợp 2 $\sqrt{x+8}=3x\Leftrightarrow \begin{cases} x\ge 0 \\ 9x^2-x-8=0\end{cases} \Leftrightarrow x =1.$
		\end{itemize}
		Vậy phương trình có nghiệm $x=1$.
		\end{enumerate}}
\end{ex}

\begin{ex}%[9D1G9]%[9D5G3]
\hfill
\begin{enumerate}
\item Cho $f(x)=(x^3+12x-31)^{2013}$. Tính $f(a)$ với $a=\sqrt[3]{16-8\sqrt{5}}+\sqrt[3]{16+8\sqrt{5}}$.
\item Tìm các số nguyên $x,y$ thỏa mãn $y^2+2xy-3x-2=0$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $a^3=32+3\cdot (-4)\cdot a\Leftrightarrow a^3+12a-31=1\Rightarrow f(a)=1^{2013}=1.$
\item Ta có 
\allowdisplaybreaks{
		\begin{align*}
		&y^2+2xy-3x-2=0\\
		\Leftrightarrow & x^2+2xy +y^2=x^2+3x+2\\
		\Leftrightarrow & (x+y)^2=(x+1)(x+2)\hfill (1)
		\end{align*}}
		Với $x,y$ nguyên thì vế trái của $(1)$ là số chính phương, vế phải của $(1)$ là tích hai số nguyên liên tiếp nên $(x+y)^2=(x+1)(x+2)=0$. Suy ra $(x;y)=(-1;1)$ hoặc $(x;y)=(-2;2)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[9D5G5]%[9D5G1]
\hfill
\begin{enumerate}
\item Cho $a,b,c$ là ba số hữu tỉ thỏa mãn $abc=1$ và $\dfrac{a}{b^2}+\dfrac{b}{c^2}+\dfrac{c}{a^2}=\dfrac{a^2}{c}+\dfrac{b^2}{a}+\dfrac{c^2}{b}$. Chứng minh rằng ít nhất một trong ba số $a,b,c$ là bình phương của một số hữu tỉ.
\item Cho $a,b,c$ là ba số dương có tổng bằng $3$. Chứng minh rằng $\dfrac{a}{1+b^2}+\dfrac{b}{1+c^2}+\dfrac{c}{1+a^2} \ge \dfrac{3}{2}$.
\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item Vì $abc=1$ nên 
	\allowdisplaybreaks{
	\begin{align*}
	&\dfrac{a}{b^2}+\dfrac{b}{c^2}+\dfrac{c}{a^2}=\dfrac{a^2}{c}+\dfrac{b^2}{a}+\dfrac{c^2}{b}\\
	\Leftrightarrow & a^3c^2+b^3a^2+c^3a^2=a^3b^2+b^3c^2+c^3a^2\\
	\Leftrightarrow & (a^2b^2c^2-a^3c^2)-(b^3a^2-a^3b^2)-(c^3b^2-c^3a)-(b^3c-abc)=0\\
	\Leftrightarrow & (b^2-a)\left((a^2c^2-ba^2)-(c^3-bc)\right)=0\\
	\Leftrightarrow & (b^2-a)(c^2-b)(a^2-c)=0
	\end{align*}}
	Vậy ta được điều phải chứng minh.
	\item Vì $a,b$ dương và $1+b^2\ge 2b$ với mọi $b$ nên $\dfrac{a}{1+b^2}=a-\dfrac{ab^2}{1+b^2\ge a-\dfrac{ab^2}{2b}}=a-\dfrac{ab}{2}$.\\
	Tương tự ta cũng có$\dfrac{b}{1+c^2}\ge b-\dfrac{bc}{2}; \dfrac{c}{1+a^2}\ge c-\dfrac{ca}{2}$. Mà $a+b+c=3$ nên cộng các bất đẳng thức trên với nhau ta được
	\[\dfrac{a}{1+b^2}+\dfrac{b}{1+c^2}+\dfrac{c}{1+a^2} \ge 3-\dfrac{ab+bc+ca}{2} \tag {1}\]
	Mặt khác cũng do $a+b+c=3$ suy ra $(a+b+c)^2=9\Leftrightarrow a^2+b^2+c^2+2(ab+bc+ca)=9$. Mà $a^2+b^2+c^2\ge ab+bc+ca$ nên $ab+bc+ca \le 3. \hfill (2)$
	Từ $(1)$ và $(2)$ suy ra điều phải chứng minh.\\
	Đẳng thức xảy ra khi và chỉ khi $a=b=c=1.$
	
	\end{enumerate}
		}
\end{ex}

\begin{ex}%[9H3G5]
Cho đường tròn $(O;R)$ và hai đường kính $AB, CD$ sao cho tiếp tuyến tại $A$ của $(O;R)$ cắt các đường thẳng $BC, BD$ tại hai điểm tương ứng là $E,F$. Gọi $P,Q$ lần lượt là trung điểm của các đoạn thẳng $AE,AF$. 
\begin{enumerate}
\item Chứng minh rằng trực tâm $H$ của tam giác $BPQ$ là trung điểm của đoạn thẳng $OA$.
\item Hai đường kính $AB, CD$ thỏa mãn điều kiện gì thì tam giác $BPQ$ có diện tích nhỏ nhất.
\item Chứng minh rằng $CE.DF.EF=CD^3$ và $\dfrac{BE^3}{BF^3}=\dfrac{CE}{DF}$.
\item Biết rằng tam giác $BEF$có một hình vuông $BMKN$ nội tiếp, $K\in EF$, $M\in BE$, $N\in BF$ sao cho tỉ số giữa cạnh của hình vuông với bán kính đường tròn nội tiếp tam giác $BEF$ là $\dfrac{2+\sqrt{2}}{2}$. Tính các góc nhọn của tam giác $BEF$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1.5]
	\tkzInit[xmin=-1.5,xmax=5,ymin=-0.5,ymax=2.5] 
    \tkzClip
	\tkzDefPoints{0/0/A, 0/2/B, 0/1/O}
	\tkzDefPointBy[rotation = center  O angle -60](A)\tkzGetPoint{C}
	\tkzDefPointBy[rotation = center  O angle -60](B)\tkzGetPoint{D}
	\tkzDrawCircle[radius](O,A)	
	\tkzTangent[at=A](O) \tkzGetPoint{a}
	\tkzInterLL(B,C)(A,a)\tkzGetPoint{E}
	\tkzInterLL(B,D)(A,a)\tkzGetPoint{F}
	\tkzDefMidPoint(A,E)\tkzGetPoint{P}
	\tkzDefMidPoint(A,F)\tkzGetPoint{Q}
	\tkzDefMidPoint(A,O)\tkzGetPoint{H}
	\tkzDrawSegments(A,B C,D E,F B,E B,F B,P B,Q O,E P,H)
	\tkzDrawPoints[fill=white](O,A,B,C,D,E,F,P,Q,H)
   	\tkzLabelPoints[left](C,E)
   	\tkzLabelPoints[right](F,H)
   	\tkzLabelPoints[below](A,P,Q)
   	\tkzLabelPoints[above right](D)
   	\tkzLabelPoints[below right](O)
   	\tkzLabelPoints[above](B)
\end{tikzpicture}
\end{center}
}
\loigiai{
\begin{enumerate}
\item $BA$ là đường cao của tam giác $\triangle BPQ \Rightarrow H \in BA. \hfill (1)$\\
Ta có $AB^2=AE.AF \Rightarrow \dfrac{AE}{AB}=\dfrac{AB}{AF}\Rightarrow \dfrac{AE}{AO}=\dfrac{AB}{AQ}$.\\
Lại có $\triangle AEO$ đồng dạng với $\triangle ABQ\Rightarrow \widehat{AEO}=\widehat{ABQ}$.\\
Mặt khác $\widehat{ABQ}=\widehat{APH}\Rightarrow \widehat{AEO}=\widehat{APH}$ nên $PH\parallel EO \hfill (2)$\\
Trong tam giác $AEO$ có $P$ là trung điểm của $AE$, kết hợp với $(1),(2)$ suy ra $H$ là trung điểm của $OA.$
\item Ta có $S_{BPQ}=\dfrac{AB\cdot PQ}{2}=R\cdot PQ =R(AP+AQ)=\dfrac{R(AE+AF)}{2}\ge R\sqrt{AE\cdot AF}=2R^2$.\\
Suy ra $S_{BPQ}=2R^2\Leftrightarrow AE=AF\Leftrightarrow \triangle BEF$ vuông cân tại $B \Leftrightarrow \triangle BCD$ vuông cân tại $B\Leftrightarrow AB \perp CD$.\\
Vậy tam giác $BPQ$ có diện tích nhỏ nhất khi $AB\perp CD$.
\item Ta có $AC\perp BE$; $AD\perp BF$ do đó $CD^4=AB^4=(AE\cdot AF)^2 =AE^2\cdot AF^2 =(EC\cdot EB)(FD\cdot FB)= (EC\cdot FD)(EB\cdot FB)=(EC\cdot FD)(AB\cdot EF)=CE\cdot DF \cdot CD\cdot EF\Rightarrow CD^3=CE\cdot DF\cdot EF$.\\
Ta cũng có $\dfrac{BE^2}{BF^2}=\dfrac{EA\cdot EF}{FA\cdot FE}=\dfrac{EA}{FA}$.\\
$\Rightarrow \dfrac{BE^4}{BF^4}=\dfrac{EA^2}{FA^2}=\dfrac{EC\cdot EB}{FD\cdot FB}\Rightarrow \dfrac{BE^3}{BF^3}=\dfrac{CE}{DF}$.
\item Gọi $I$ là tâm đường tròn nội tiếp tam giác $EBF$. Do tứ giác $BMKN$ là hình vuông nên $BK$ là phân giác của $\widehat{EBF}$, do đó $I\in BK$. Kẻ $IJ\perp BE, IS\perp BF$, ($J\in BE,S\in BF$).\\
Ta thấy $IS\parallel KN \Rightarrow \dfrac{BK}{BI}=\dfrac{KN}{IS}=\dfrac{2+\sqrt{2}}{2}=1+\dfrac{\sqrt{2}}{2}$.\\
Mà$\dfrac{BK}{BI}=\dfrac{BI+IK}{BI}=1+\dfrac{IK}{IB}\Rightarrow \dfrac{IK}{IB}=\dfrac{\sqrt{2}}{2}$.\\
Mặt khác $FI$ là phân giác của góc $\widehat{BFE}$ nên $\dfrac{FK}{FB}=\dfrac{IK}{IB}=\dfrac{\sqrt{2}}{2}\Rightarrow FB^2=2FK^2$.\\
Tương tự ta có $EB^2=2EK^2\Rightarrow EF^2=EB^2+FB^2=2(EK^2+FK^2)\Rightarrow (EK+KF)^2=2(EK^2+FK^2)\Rightarrow (EK-FK)^2=0\Rightarrow EK=FK$.\\
Suy ra tam giác $EBF$ vuông cân tại $B\Rightarrow \widehat{BEF}=\widehat{BFE}=45^\circ$. 
\end{enumerate}}
\end{ex}
