\begin{ex}%[Đề thi vào 10, Sở giáo dục Thanh Hóa, 2016]%[9D2B2]%[9D3B3]
	\hfill
	\begin{enumerate}[1)]
		\item Giải các phương trình sau:
		
		a) $x-6=0$.
		
		b) $x^2-5x+4=0$.
		
		\item Giải hệ phương trình: $\begin{cases}
		2x-y=3\\3x+y=2.
		\end{cases}$
		
	\end{enumerate}
	\loigiai{
	\begin{enumerate}[1)]
		\item 
		\hfill
		\begin{enumerate}[a)]
			\item $x-6=0 \Leftrightarrow x=6$.  Vậy tập nghiệm của phương trình là $\{6\}$.
			\item $x^2-5x+4=0$. Có $a+b+c=1-5+4=0$ nên phương trình có hai nghiệm $x=1,x=4$.
			
			Vậy tập nghiệm của phương trình là $\{1;4\}$.
		\end{enumerate}
		\item $\begin{cases}
		2x-y=3\\3x+y=2.
		\end{cases} \Leftrightarrow \begin{cases}
		5x=5 \\ 3x+y=2
		\end{cases} \Leftrightarrow \begin{cases}
		x=1 \\ 3.1+y=2
		\end{cases} \Leftrightarrow \begin{cases}
		x=1\\y=-1.
		\end{cases}$
		
		Hệ có nghiệm duy nhất $(1;-1)$.
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Thanh Hóa, 2016]%[9D2B2]%[9D3B3]
	Cho biểu thức $A = \left( \displaystyle \frac{y\sqrt{y}-1}{y- \sqrt{y}}- \frac{y \sqrt{y}+1}{y+\sqrt{y}}\right):\displaystyle \frac{2\left(y-2\sqrt{y}+1\right)}{y-1}$ (với $y >0$ và $y \ne 1$). 
	
	\begin{enumerate}[1)]
		\item Rút gọn biểu thức $B$.
		
		\item Tìm các số nguyên $y$ để biểu thức $B$ có giá trị nguyên.
	\end{enumerate}
\loigiai
	{
	\begin{enumerate}[1)]
		\item Có
		
		$B= \left[
		\displaystyle \frac{\left(\sqrt{y}-1\right) \left( y + \sqrt{y} +1\right)}{\sqrt{y}\left(\sqrt{y}-1\right)}- 
		\frac{\left(\sqrt{y}+1\right) \left( y - \sqrt{y} +1\right)}{\sqrt{y}\left(\sqrt{y}+1\right)}
		\right]:
		\displaystyle \frac{2 \left(\sqrt{y}-1\right)^2}{\left(\sqrt{y}-1\right) \left( \sqrt{y}+1\right)}$
		
		$= \left(
		\displaystyle \frac{ y + \sqrt{y} +1}{\sqrt{y}}- 
		\frac{ y - \sqrt{y} +1}{\sqrt{y}}
		\right):
		\displaystyle \frac{2 \left(\sqrt{y}-1\right)}{\left( \sqrt{y}+1\right)}$
		
		$\displaystyle =\frac{2 \sqrt{y}}{\sqrt{y}} \cdot \frac{\sqrt{y}+1}{2\left(\sqrt{y}-1\right)}$
		
		$\displaystyle =\frac{\sqrt{y}+1}{\sqrt{y}-1}$.
		
		\item $B = \displaystyle \frac{\sqrt{y}-1+2}{\sqrt{y}-1} =1 + \frac{2}{\sqrt{y}-1}$.
		
		Vì $y$ nguyên nên ta có $B$ nguyên $\Leftrightarrow \displaystyle \frac{2}{\sqrt{y}-1}$ nguyên $\Leftrightarrow \sqrt{y}-1$ là ước của $2$.
		
		Mặt khác $y>0, y \ne 1$ nên $\sqrt{y}-1 > -1$. Do đó 
		
		Vậy $y=4$ hoặc $y=9$ thỏa mãn đề bài.
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Thanh Hóa, 2016]%[9D2B2]%[9D3B3]
	Trong mặt phẳng tọa độ $Oxy$, cho đường thẳng $(d): y =nx+1$ và parabol $(P): y=2x^2$.
	\begin{enumerate}[1)]
		\item Tìm $n$ để đường thẳng $(d)$ đi qua điểm $B(1;3)$.
		
		\item Chứng minh rằng đường thẳng $(d)$ luôn cắt parabol $(P)$ tại hai điểm phân biệt $M(x_1;y_1), N(x_2;y_2)$. Hãy tính giá trị của biểu thức $S=x_1x_2+y_1y_2.$
	\end{enumerate}
\loigiai{
	\begin{enumerate}[1)]
		\item Đường thẳng $(d)$ đi qua điểm $B(1;3) \Leftrightarrow 3=n . 1+1 \Leftrightarrow n=2.$
		
		Vậy $n=2$.
		
		\item Xét phương trình hoàng độ giao điểm của $(d)$ và $(P)$:		
		$$2x^2=nx+1 \Leftrightarrow 2x^2-nx-1=0\,\quad (1)$$  
		Có $\Delta = n^2-4.2.(-1)=n^2+8$. Vì $n^2 \ge 0, \forall n \Rightarrow n^2+8 \ge 8 >0, \forall n$.
		
		Suy ra phương trình (1) luôn có $2$ nghiệm phân biệt với mọi $n$ $\Rightarrow$ $(d)$ luôn cắt $(P)$ tại $2$ điểm phân biệt $M(x_1;y_1), N(x_2;y_2) \forall n$, trong đó $x_1,x_2$ là hai nghiệm của (1) và $y_1=2x_1^2, y_2=2x_2^2$.
		
		Theo định lý Viét ta có: $x_1x_2= \displaystyle -\frac{1}{2} \Rightarrow S =x_1x_2 + 2x_1^2.2x_2^2 = x_1x_2+4(x_1x_2)^2=\frac{1}{2}$
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Thanh Hóa, 2016]%[9D2B2]%[9D3B3]
	Cho tứ giác $MNPQ$ nội tiếp đường tròn đường kính $MQ$. Hai đường chéo $MP$ và $NQ$ cắt nhau tại $E$. Gọi $F$ là điểm thuộc đường thẳng $MQ$ sao cho $EF$ vuông góc với $MQ$. Đường thẳng $PF$ cắt đường tròn đường kính $MQ$ tại điểm thứ hai là $M$. Gọi $N$ là giao điểm của $NQ$ và $PF$. Chứng minh rằng:
	\begin{enumerate}[1)]
		\item Tứ giác $PEFQ$ nội tiếp đường tròn.
		\item $FM$ là đường phân giác của góc $\widehat{NFM}$.
		\item $NQ \cdot NE = NE \cdot NQ$.
	\end{enumerate}
		\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\draw (0,0) circle  (4cm);
			\tkzDefPoints{0/0/O, -4/0/M, 4/0/Q, -2/3.464/N, 1/3.873/P}
			\coordinate (E) at ($(M)!.685!(P)$);
			\coordinate (F) at ($(O)!.15!(M)$);
			\coordinate (H) at ($(F)!.55!(P)$);
			\coordinate (K) at ($(O)!.3!(Q)$);
			\tkzDrawSegments[thick](M,Q M,N N,Q P,Q N,P M,P E,F N,F P,F H,K)
			\tkzMarkRightAngle(E,F,O)
			\tkzMarkRightAngle[size=.1](M,P,Q)
			\tkzMarkAngles[size=5mm](H,K,O)
			\tkzLabelPoints[below](O)
			\tkzLabelPoints[below left](M)
			\tkzLabelPoints[above left](N)
			\tkzLabelPoints[above right](P)
			\tkzLabelPoints[below right](Q)
			\tkzLabelPoints[above](E)
			\tkzLabelPoints[below left](F)
			\tkzLabelPoints[above right](H)
			\tkzLabelPoints[below](K)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}[1)]
			\item Có $\widehat{MPQ}=90^{\circ}$ (góc nội tiếp chắn nửa đường tròn), hay $\widehat{EPQ}=90^{\circ}$.
			
			Mặt khác, $EF \perp MQ$ nên $\widehat{EFQ}=90^{\circ}$.
			
			Suy ra $\widehat{EPQ}+\widehat{EFQ}=180^{\circ}$, do đó $PEFQ$ là tứ giác nội tiếp.
			
			\item Vì $PEFQ$ là tứ giác nội tiếp (cmt) nên $\widehat{PFQ}=\widehat{PEQ}$ (do hai góc nội tiếp cùng chắn cung $PQ$) (1).
			
			Chứng minh tương tự có tứ giác $MNEF$ nội tiếp, suy ra $\widehat{NFM}=\widehat{NEM}$ (do hai góc nội tiếp cùng chắn cung $MN$) (2).
			
			Có $\widehat{NEM}=\widehat{PEQ}$, $\widehat{MFM}=\widehat{PFQ}$ (3)
			
			Từ (1), (2) và (3) suy ra $\widehat{NFM}=\widehat{MFM}$, do đó $FM$ là tia phân giác của $\widehat{NFM}$.
			
			\item Vẽ $NK \parallel NF (K \in MQ)$.
			
			Ta có $\widehat{NKF}=\widehat{NFM}$ (đồng vị); $\widehat{NFM}=\widehat{NFK} \Rightarrow \widehat{NKF}=\widehat{NFK} \Rightarrow \triangle NFQ \text{ cân tại } N \Rightarrow NK=NF$.
			
			Vì $NK \parallel NF$ nên $\dfrac{NK}{NF}=\dfrac{QN}{QN} \Rrightarrow \dfrac{NF}{NF}=\dfrac{QN}{QN}$ (4)
			
			Vì $\widehat{NFM}=\widehat{NFK}$ nên $\widehat{EFN}=\widehat{EFN}$ (cùng phụ với hai góc bằng nhau).
			
			Suy ra $FE$ là phân giác $\widehat{NFN}$ của $\triangle NFN$. Theo định lí đường phân giác ta có $\dfrac{NF}{NF}=\dfrac{NE}{NE}$ (5)
			
			Từ (4) và (5) $\Rightarrow \dfrac{QN}{QN}=\dfrac{NE}{NE} \Rightarrow NQ.NE=NE.QN$ (đpcm)
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Thanh Hóa, 2016]%[9D2B2]%[9D3B3]
	Cho $m,n,p$ là các số thực dương thỏa mãn: $m^2+2n^2 \le 3p^2$. Chứng minh rằng: $ \displaystyle \frac{1}{m}+ \frac{2}{n} \ge \frac{3}{p}.$
	\loigiai{
		
		\noindent Áp dụng bất đẳng thức Bunhiacopxki cho hai bộ số, ta có		
		$$(m+2n)^2=(1.m+\sqrt{2}.\sqrt{2}n)^2 \leq (1+2)(m^2+2n^2) \leq 3.3p^2=9p^2 \Rightarrow m+2n \leq 3p.$$		
		Với mọi $x, y, z >0$, áp dụng bất đẳng thức Côsi cho 3 số dương ta có		
		$$(x+y+z)\left(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\right) \geq 3 \sqrt[3]{xyz}.3 \sqrt[3]{\dfrac{1}{xyz}}=9 \Rightarrow \dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z} \geq \dfrac{9}{x+y+z}.$$
		
		Áp dụng bất đẳng thức trên, ta có 
		$\dfrac{1}{m}+\dfrac{2}{n}=\dfrac{1}{m}+\dfrac{1}{n}+\dfrac{1}{n} \geq \dfrac{9}{m+n+p}=\dfrac{9}{m+2n} \geq \dfrac{9}{3p}=\dfrac{3}{p}$ (đpcm).
		
		Dấu "=" xảy ra $\Leftrightarrow m=n=p$.
	}
\end{ex}
