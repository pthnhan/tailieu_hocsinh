\begin{bt}%[Đề Học Kỳ 1 lớp 11 trường THPT Nguyễn Hữu Huân, Tp Hồ Chí Minh, 2017-2018]%[Lục Hán Tường EX-11HK1-18]%[1D1B3-1]%[1D1B3-3]
	Giải phương trình sau
	\begin{enumerate}
		\item $ \cos x + \cos 2x - 2 = 0 $.
		\item $ \sqrt{3}\sin 2x + \cos 2x = 2 $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item 
			\begin{align*}
				&\cos x + \cos 2x - 2 = 0
				\Leftrightarrow  2\cos^2x - 1 + \cos x - 2 = 0 \\
				\Leftrightarrow & \left(\cos x - 1\right)\left(2\cos x + 3\right) = 0
				\Leftrightarrow  \hoac{& \cos x = 1\\& \cos x = -\dfrac{3}{2} \ \textrm{(Vô nghiệm)}}\\
				\Leftrightarrow & \cos x = 1
				\Leftrightarrow  x = k2\pi \qquad (k \in \mathbb{Z}).
			\end{align*}
			
			\item 
			\begin{align*}
			& \sqrt{3}\sin 2x + \cos 2x = 2
			\Leftrightarrow  \dfrac{\sqrt{3}}{2}\sin 2x + \dfrac{1}{2}\cos 2x = 1\\
			\Leftrightarrow & \sin\left(2x + \dfrac{\pi}{6}\right) = 1
			\Leftrightarrow  2x + \dfrac{\pi}{6} = \dfrac{\pi}{2} + k2\pi\\
			\Leftrightarrow & x = \dfrac{\pi}{6} + k\pi \qquad (k \in \mathbb{Z}).
			\end{align*}
		\end{enumerate}
	}
	
\end{bt}
\begin{bt}%[Đề Học Kỳ 1 lớp 11 trường THPT Nguyễn Hữu Huân, Tp Hồ Chí Minh, 2017-2018]%[Lục Hán Tường EX-11HK1-18]%[1D2B3-2]
	Tìm số hạng chứa $ x^{10} $ trong khai triển nhị thức $ \left(x^2 - \dfrac{1}{x}\right)^{18} $.
	\loigiai{
	Số hạng tổng quát của khai triển nhị thức $ \left(x^2 - \dfrac{1}{x}\right)^{18} $.
	\begin{align*}
	u_k = \mathrm{C}^k_{18}\left(x^2\right)^{18-k}\left(-\dfrac{1}{x}\right)^k = \left(-1\right)^k\mathrm{C}^k_{18}x^{36 - 3k} \qquad (k\in \mathbb{Z}_+).
	\end{align*}
	$ u_k $ chứa $ x^{10} $ khi $ 36 - 3k = 10 \Leftrightarrow k = \dfrac{26}{3} $ (Vô lý).\\
	Vậy không có số hạng chứa $ x^{10} $ trong khai triển nhị thức $ \left(x^2 - \dfrac{1}{x}\right)^{18} $.
	}
\end{bt}

\begin{bt}%[Đề Học Kỳ 1 lớp 11 trường THPT Nguyễn Hữu Huân, Tp Hồ Chí Minh, 2017-2018]%[Lục Hán Tường EX-11HK1-18]%[1D2B5-4]
	Một nhóm nhảy flashmob có $5$ bạn mặc áo đen, $8$ bạn áo hồng và $6$ bạn áo đỏ đạt giải quán quân. Ban tổ chức yêu cầu cử $3$ thành viên lên nhận giải. Tính xác suất để $3$ bạn ấy mặc áo đủ ba màu?
	\loigiai{
	Số cách chọn nhóm $ 3 $ bạn bất kỳ: $ \mathrm{C}^3_{19} $.\\
	Số cách chọn $ 1 $ bạn áo đen: $ \mathrm{C}^1_{5} $.\\
	Số cách chọn $ 1 $ bạn áo hồng: $ \mathrm{C}^1_{8} $.\\
	Số cách chọn $ 1 $ bạn áo đỏ: $ \mathrm{C}^1_{6} $.\\
	Số cách chọn nhóm $ 3 $ bạn mặc đủ ba màu: $ \mathrm{C}^1_{5} \cdot \mathrm{C}^1_{8} \cdot \mathrm{C}^1_{6} $.\\
	Suy ra, xác suất để chọn được $3$ bạn mặc áo đủ ba màu
	\begin{align*}
	\mathrm{P} = \dfrac{\mathrm{C}^1_{5} \cdot \mathrm{C}^1_{8} \cdot \mathrm{C}^1_{6}}{\mathrm{C}^3_{19}} = \dfrac{240}{969} = \dfrac{80}{323}.
	\end{align*}
	}
\end{bt}

\begin{bt}%[Đề Học Kỳ 1 lớp 11 trường THPT Nguyễn Hữu Huân, Tp Hồ Chí Minh, 2017-2018]%[Lục Hán Tường EX-11HK1-18]%[1D3B3-2]
	Xác định số hạng đầu $ u_1 $ và công sai $ d $ của cấp số cộng $ (u_n) $ biết $ \heva{& u_3 + u_{10} = -31\\& 2u_{4} - u_9 = 7} $.
	\loigiai{
	Ta có 
	\begin{align*}
	\heva{& u_3 + u_{10} = -31\\& 2u_{4} - u_9 = 7} \Leftrightarrow \heva{& u_1 + 2d + u_{1} + 9d = -31\\& 2\left( u_{1} + 3d \right) - \left( u_1 + 8d \right) = 7} \Leftrightarrow \heva{& 2u_1 + 11d = -31\\& u_{1} - 2d = 7} \Leftrightarrow \heva{& u_1 = 1\\& d = -3}.
	\end{align*}
	}
\end{bt}

\begin{bt}%[Đề Học Kỳ 1 lớp 11 trường THPT Nguyễn Hữu Huân, Tp Hồ Chí Minh, 2017-2018]%[Lục Hán Tường EX-11HK1-18]%[1D3K4-5]
	Cho cấp số nhân $ (u_n) $ biết: $ \heva{& u_1 + u_{5} = 51\\& u_{2} + u_6 = 102} $. Hỏi tổng của bao nhiêu số hạng đầu tiên của cấp số nhân sẽ bằng $ 3069 $?
	\loigiai{
	Gọi $ q $ là công bội của cấp số nhân. Ta có
	\begin{align*}
	\heva{& u_1 + u_{5} = 51\\& u_{2} + u_6 = 102} \Leftrightarrow \heva{& u_1 + u_{5} = 51\\&q\left( u_{1} + u_5\right) = 102}
	\Leftrightarrow \heva{& u_1 + u_{1}q^4 = 51\\& q = 2} \Leftrightarrow \heva{& 17 u_1 = 51\\& q = 2} \Leftrightarrow \heva{& u_1 = 3\\& q = 2}.
	\end{align*}
	$ \Rightarrow $ Tổng $ n $ số hạng đầu của cấp số nhân.
	\begin{align*}
	S_n = u_1 + u_2 + \ldots + u_n = u_1\dfrac{q^{n} - 1}{q - 1} = 3\left(2^{n} - 1\right).
	\end{align*}
	Suy ra 
	\begin{align*}
	S_n = 3069 \Leftrightarrow 3\left(2^{n} - 1\right) = 3069 \Leftrightarrow 2^{n} - 1 = 1023 \Leftrightarrow 2^{n} = 1024 \Leftrightarrow n = 10.
	\end{align*}
	Vậy tổng của $ 10 $ số hạng đầu tiên của cấp số nhân sẽ bằng $ 3069 $.
	}
\end{bt}

\begin{bt}%[Đề Học Kỳ 1 lớp 11 trường THPT Nguyễn Hữu Huân, Tp Hồ Chí Minh, 2017-2018]%[Lục Hán Tường EX-11HK1-18]%[1D2K5-3]
	Đội thanh niên tình nguyện của trường có 18 học sinh gồm 5 học sinh lớp 12, 6 học sinh lớp 11 và 7 học sinh lớp 10. Trường cần chọn 7 học sinh để làm nhiệm vụ. Hỏi có bao nhiêu cách chọn 7 học sinh đủ cả ba khối lớp?
	\loigiai{
		Số cách chọn đội có các học sinh bất kỳ:
		$
		\mathrm{C}_{18}^7 = 31824
		$.\\
		Số cách chọn đội không có học sinh lớp $ 10 $:
		$
		\mathrm{C}_{11}^7 = 330
		$.\\
		Số cách chọn đội không có học sinh lớp $ 11 $:
		$
		\mathrm{C}_{12}^7 = 792
		$.\\
		Số cách chọn đội không có học sinh lớp $ 12 $:
		$
		\mathrm{C}_{13}^7 = 1716
		$.\\
		Suy ra, số cách chọn đội có đủ học sinh của ba khối
		\begin{align*}
		\mathrm{C}_{18}^7 - \mathrm{C}_{11}^7 - \mathrm{C}_{12}^7 - \mathrm{C}_{13}^7 = 31824 - 330 - 792 - 1716 = 28986.
		\end{align*}
		Xác suất chọn được đội có đủ học sinh của ba khối
		\begin{align*}
		\mathrm{P} = \dfrac{28986}{31824} = \dfrac{4831}{5304}.
		\end{align*}
	}

	Cho hình chóp $ S.ABCD $ có đáy là hình bình hành tâm $ O $. Gọi $ M, N $ lần lượt là trung điểm của $ CD $ và $ SB $.
	\begin{enumerate}
		\item Chứng minh: $ (OMN) \parallel (SAD) $.
		\item Tìm giao điểm $ I $ của $ SC $ và mặt phẳng $ (OMN) $.
		\item Tìm thiết diện của hình chóp $ S.ABCD $ với mặt phẳng $ (OMN) $.
	\end{enumerate}
	\loigiai{

	\loigiai{ 
	\begin{enumerate}
		\item Ta có $ M $, $N$, $ O $ là trung điểm của $ CD $, $ SB $, $ BD $ nên
		\begin{align*}
		\heva{& MO \parallel AD \Rightarrow MO \parallel (SAD)\\& NO \parallel SD \Rightarrow NO \parallel (SAD) \\& MO \cap NO = O} \Rightarrow (OMN) \parallel (SAD).
		\end{align*}
		
		\item Giả sử $ d $ là giao tuyến của $ (OMN) $ và $ (SBC) $. Ta có
		\begin{align*}
		\heva{& BC \parallel OM \Rightarrow BC \parallel(OMN)\\
		& (OMN) \cap (SBC) = d\\
		& N \in (SBC)\\
		& N \in (OMN)}\Rightarrow \heva{& d \parallel BC\\& N \in d}.
		\end{align*}
		Gọi $ I = d \cap SC \Rightarrow I = SC \cap (OMN) $.
		
		\item Trong $ (ABCD) $ kéo dài $ OM $ cắt $ AB $ tại $ E $. Suy ra $ (OMN) \cap (ABCD) = ME $.\\

		\heva{NE \subset (OMN)\\NE \subset (SAB)} \Rightarrow (OMN) \cap (SAB) = EN.
		\end{align*}
		Theo chứng minh trên $ (OMN) \cap (SBC) = NI $.\\

		\begin{align*}
		\heva{IM \subset (OMN)\\IM \subset (SCD)} \Rightarrow (OMN) \cap (SCD) = IM.
		\end{align*}
		Do đó thiết diện của hình chop $ S.ABCD $ với mặt phẳng $ (OMN) $ là tứ giác $ MENI $.
	\end{enumerate}

	Tìm công thức tính số hạng tổng quát của dãy số $ (u_n) $ biết $ \heva{& u_1 = 9\\& u_{n + 1} = 3u_n + 2\cdot 5^n - 4 \ , \ n \geqslant 1} $.
	\loigiai{ 
	Ta có
	\begin{align*}
	u_{n + 1} = 3u_n + 2\cdot 5^n - 4 \Leftrightarrow u_{n + 1} - 5^{n + 1} - 2 = 3\left(u_{n} - 5^{n} - 2\right).
	\end{align*}
	Đặt $ v_n = u_{n} - 5^{n} - 2 \Rightarrow v_{n + 1} = 3v_n \Rightarrow (v_n) $ là một cấp số nhân với công bội bằng $ q = 3 $.\\
	Mặt khác $ v_1 = u_1 - 5^1 - 2 = 9 - 5 - 2 = 2 $.
	\begin{align*}
	& \Rightarrow v_n = v_1\cdot q^{n - 1} \Leftrightarrow v_n = 2\cdot 3^{n - 1}\\
	& \Leftrightarrow u_{n} - 5^{n} - 2 = 2\cdot 3^{n - 1}\\
	& \Leftrightarrow u_n = 2\cdot 3^{n - 1} + 5^{n} + 2
\end{align*}	
Vậy dãy $ (u_n) $ có số hạng tổng quát $ u_n = 2\cdot 3^{n - 1} + 5^{n} + 2 $.
	}
\end{bt}
