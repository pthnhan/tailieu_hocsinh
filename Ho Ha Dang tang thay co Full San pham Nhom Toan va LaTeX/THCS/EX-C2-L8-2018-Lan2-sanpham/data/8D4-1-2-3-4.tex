\section{Liên hệ giữa thứ tự và phép cộng. Liên hệ giữa thứ tự và phép nhân}
\subsection{Trọng tâm kiến thức}
\subsubsection{Bất đẳng thức}
Ta gọi hệ thức dạng $a<b$ (hay $a>b$; $a\leq b$; $a\geq b$) là bất đẳng thức.
\subsubsection{Liên hệ giữa thứ tự và phép cộng}
Khi cộng cùng một  số vào cả hai vế của một bất đẳng thức ta được bất đẳng thức mới cùng chiều bắt đẳng thức đã cho.
\begin{eqnarray*}
a\leq b &\Leftrightarrow& a+c\leq b+c\\
a\geq b &\Leftrightarrow& a+c\leq b+c
\end{eqnarray*}
\subsubsection{Liên hệ giữa thứ tự và phép nhân}
\begin{itemize}
	\item Khi nhân cả hai vế của một bắt đẳng thức với cùng một số dương ta được bất đẳng thức mới cùng chiều với bất đẳng thức đã cho.
	\item Khi nhân hai vế của một bất đẳng thức với cùng một số âm ta được bất đẳng thức mới ngược chiều với bất đẳng thức đã cho.
	\begin{eqnarray*}
		a\leq b &\Leftrightarrow& a\cdot c\leq b\cdot c \text{ nếu } c>0\\
		a\geq b &\Leftrightarrow& a\cdot c\leq b\cdot c \text{ nếu } c<0
	\end{eqnarray*}
\end{itemize}
\subsubsection{Tính chất bắc cầu của thứ tự}
Nếu $a<b$ và $b<c$ thì $a<c$.
\subsection{Các dạng bài tập và phương pháp giải}
\begin{dang}{Xác định tính đúng sai của một bất đẳng thức}
	\begin{itemize}
		\item Vận dụng thứ tự tập hợp số.
		\item Vận dụng liên hệ giữa thứ tự và phép cộng, phép nhân.
	\end{itemize}
\end{dang}

\begin{vd}%[Nguyễn Cao Cường]%[8D4B2]
	Mỗi bất đẳng thức sau đúng hay sai?
	\begin{enumerate}
		\item $5+(-8)<1$;
		\item $(-2)\cdot (-7)>(-5)\cdot (-3)$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $5+(-8)<1$ đúng vì $-3<1$ là bất đẳng thức đúng.
	\item $(-2)\cdot (-7)> (-5)\cdot (-3)$ sai vì $14>15$ là bất đẳng thức sai.
\end{enumerate}

}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[8D4B2]
	Mỗi khẳng định sau đúng hay sai? Vì sao?
	\begin{enumEX}{2}
			\item $(-7)^2-9\leq (-10)\cdot (-4)$;
		\item Thương của $15$ và $-6$ nhỏ hơn thương của $(-12)$ và $4$.
	\end{enumEX}
\loigiai{
\begin{enumerate}
	\item $(-7)^2-9=40; (-10)\cdot (-4)=40$. Vậy $(-7)^2-9\leq (-10)\cdot (-4)$ là bất đẳng thức đúng.
	\item $15:(-6)=-2{,}5; (-12):4=-3$. Vì $-2{,}5>-3$ nên bất đẳng thức $15:(-6)<(-12):4$ là bất đẳng thức sai.
\end{enumerate}
		
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[8D4B1]

		\item $x^2+3\geq 3$;
		\item $-x^2+1\leq 1$;
		\item $-(x+2)^2-5\leq -5$.
	\end{enumEX}
\loigiai{
\begin{enumerate}

	\item Ta có $-x^2\leq 0$ với mọi $x$.\\
	Cộng thêm $1$ vào hai vế ta được $-x^2+1\leq 1$. Vậy bất đẳng thức $-x^2+1\leq 1$ đúng.
	\item Ta có $-(x+2)^2\leq 0$ với mọi $x$.\\
	Cộng thêm $-5$ vào hai vế, ta được $-(x+2)^2-5\leq -5$. Vậy bất đẳng thức $-(x+2)^2-5\leq -5$ đúng.
\end{enumerate}

}
\end{vd}

\begin{vd}%[Nguyễn Cao Cường]%[8D4K2]
	Cho số $m$ bất kì, hãy so sánh $m^2$ và $m$.
	\loigiai{
	\begin{itemize}
		\item Trường hợp $m<0$ thì $m^2>0$, do đó $m^2>m$.
		\item Trường hợp $m=0$ thì $m^2=0$, do đó $m^2=m$.
		\item Trường hợp $0<m<1$. Nhân hai vế cho $m>0$ ta được $m^2<m$.
		\item Trường hợp $m=1$ thì $m^2=1$, do đó $m^2=m$.
		\item Trường hợp $m>1$. Nhân hai vế với $m$ ta được $m^2>m$.
	\end{itemize}
Tóm lại:
\begin{itemize}
	\item [-] Nếu $m=0$ hoặc $m=1$ thì $m^2=m$.
	\item [-] Nếu $m<0$ hoặc $m>1$ thì $m^2>m$.
	\item [-] Nếu $0<m<1$ thì $m^2<m$.
\end{itemize}
\begin{nx}
	Qua kết quả trên, ta thấy khẳng định $m^2$ luôn luôn lớn hơn $m$ là một khẳng định sai.
\end{nx}
}
\end{vd}
\begin{dang}{Chứng minh bất đẳng thức}
	\begin{itemize}
		\item \textbf{Cách 1.} Để chứng minh $A>B$ ta chứng minh $A-B>0$. Để chứng minh $A<B$ ta chứng minh $A-B<0$.
		\item \textbf{Cách 2.} Dùng phương pháp biến đổi tương đương
	\[ A>B \Leftrightarrow C>D \Leftrightarrow \cdots M>N. \]
	Nếu $M>N$ đúng thì $A>B$ đúng.
	\item \textbf{Cách 3.} Dùng các tính chất của bất đẳng thức.\\
	Từ bất đẳng thức đã biết, ta dùng các tính chất của bất đẳng thức để suy ra bất đẳng thức phải chứng minh.
	\end{itemize}
\end{dang}
\begin{vd}%[Nguyễn Cao Cường]%[8D4K2]
	Chứng minh bất đẳng thức $a^2+b^2\geq 2ab$.
	\loigiai{
	Xét hiệu $(a^2+b^2)-2ab=(a-b)^2\geq 0$.\\
	Vậy $a^2+b^2\geq 2ab$ (dấu ``='' xảy ra khi và chỉ khi $a=b$).
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[8D4K2]
	Cho $a>0$; $b>0$. Chứng minh rằng $\dfrac{a}{b}+\dfrac{b}{a}\geq 2$.
	\loigiai{
	Ta có
	$$\dfrac{a}{b}+\dfrac{b}{a}\geq 2 \Leftrightarrow \dfrac{a}{b}+\dfrac{b}{a}-2\geq 0 \Leftrightarrow \dfrac{a^2+b^2-2ab}{ab}\geq 0 \Leftrightarrow \dfrac{(a-b)^2}{ab}\geq 0.$$
	Bất đẳng thức cuối cùng hiển nhiên đúng nên bất đẳng thức đã cho là đúng (dấu ``='' xảy ra khi và chỉ khi $a=b$).
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[8D4K1]
	Cho $a>b$ và $m>n$. Chứng minh rằng $a+m>b+n$.
	\loigiai{
Ta có $a>b$.\\
Cộng $m$ vào hai vế ta được $a+m>b+m$. \hfill(1)\\
Ta có $m>n$.\\
Cộng $b$ vào hai vế ta được $b+m>b+n$. \hfill(2)\\
Từ (1) và (2) suy ra $a+m>b+n$ (tính chất bắc cầu).	
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[8D4K1]
	Cho $a$, $b$, $c>0$. Chứng minh rằng: \[\dfrac{a+b}{c}+\dfrac{b+c}{a}+\dfrac{c+a}{b}\geq 6.\]
	\loigiai{
Xét vế trái:
$$T=\dfrac{a+b}{c}+\dfrac{b+c}{a}+\dfrac{c+a}{b}=\dfrac{a}{c}+\dfrac{b}{c}+\dfrac{b}{a}+\dfrac{c}{a}+\dfrac{c}{b}+\dfrac{a}{b}=\left( \dfrac{a}{c}+\dfrac{c}{a} \right)+\left( \dfrac{b}{c}+\dfrac{c}{b} \right)+\left( \dfrac{b}{a}+\dfrac{a}{b} \right).$$
Vì với $a>0$; $b>0$ thì $\dfrac{a}{b}+\dfrac{b}{a}\geq 2$, do đó ta có
$$T\geq 2+2+2=6.$$
Dấu ``='' xảy ra khi và chỉ khi $a=b=c$.	
}
\end{vd}

\begin{dang}{Áp dụng bất đẳng thức để tìm giá trị lớn nhất, giá trị nhỏ nhất của một biểu thức}
	\begin{itemize}
		\item Nếu $f(x)\geq k$ ($k$ là hằng số) và dấu ``='' xảy ra khi và chỉ khi $x=a$ thì giá trị nhỏ nhất của $f(x)$ là $k$ khi và chỉ khi $x=a$.\\
		Ta viết $\min f(x)=k$ khi và chỉ khi $x=a$.
		\item Nếu $f(x)\leq k$ ($k$ là  hằng số)  và dấu ``='' xảy ra khi và chỉ khi $x=a$ thì giá trị lớn nhất  của $f(x)$ là $k$ khi và chỉ khi $x=a$.\\
		Ta viết $\max f(x)=k$ khi và chỉ khi $x=a$.
	\end{itemize}
\end{dang}

\begin{vd}%[Nguyễn Cao Cường]%[8D4K2]
	Tìm giá trị nhỏ nhất của biểu thức $A=x^2-6x+10$.
	\loigiai{
Ta có $A=x^2-6x+10=x^2-6x+9+1=(x-3)^2+1\geq 1$ (dấu ``='' xảy ra khi và chỉ khi $x=3$).\\
Do đó $\min A=1$ khi và chỉ khi $x=3$.	

}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[8D4K2]
	Tìm giá trị nhỏ nhất của biểu thức $B=5x^2-10x+3$.
	\loigiai{
	Ta có 
	$B=5x^2-10x+3=5x^2-10x+5-2=5(x-1)^2-2\geq -2$ (dấu ``='' xảy ra  khi và chỉ khi $x=1$).\\
	Vậy $\min B=-2$ khi và chỉ khi $x=1$.
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[8D4G2]
	Tìm giá trị lớn nhất của biểu thức $C=-x^2+5x-4$.
	\loigiai{
	Ta có
	\begin{eqnarray*}
	C&=& -x^2+5x-4=-(x^2-5x+4)=-\left(x^2-2\cdot \dfrac{5}{2}x+\dfrac{25}{4}-\dfrac{25}{4}+4\right)\\
	&=& -\left[ \left(x-\dfrac{5}{2} \right)^2-\dfrac{9}{4} \right]=-\left( x-\dfrac{5}{2} \right)^2+\dfrac{9}{4}\leq \dfrac{9}{4}\; \text{ (dấu ``='' xảy ra khi và chỉ khi } x=\dfrac{5}{2}).
	\end{eqnarray*}
Vậy $\max C=\dfrac{9}{4}$ khi $x=\dfrac{5}{2}$.
}
\end{vd}

\begin{vd}%[Nguyễn Cao Cường]%[8D4K2]
	Tìm giá trị lớn nhất của biểu thức $D=5-x-\dfrac{1}{x}$ với $x>0$.
	\loigiai{
Ta có $D=5-x-\dfrac{1}{x}=5-\left( x+\dfrac{1}{x} \right)$. Vì $ x+\dfrac{1}{x}\geq 2$ nên $D\leq 3$ (dấu ``='' xảy ra khi và chỉ khi $x=\dfrac{1}{x}\Leftrightarrow x=1$).\\
Vậy $\max D=3$ khi và chỉ khi $x=1$.

}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[8D4K2]
	Tìm giá trị nhỏ nhất của biểu thức $E=2x^2+8x+y^2-10y+43$.
	\loigiai{
Ta có
\begin{eqnarray*}
E&=& 2x^2+8x+y^2-10y+43=2x^2+8x+8+y^2-10y+25+10\\
&=& 2(x^2+4x+4)+(y-5)^2+10=2(x+2)^2+(y-5)^2+10\geq 10.
\end{eqnarray*}	
(dấu ``='' xảy ra khi và chỉ khi $x=-2$ và $y=5$).\\
Vậy $\min E=10$ khi và chỉ khi $x=-2$ và $y=5$.
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[8D4K2]
	Tìm giá trị nhỏ nhất của biểu thức $F=\dfrac{2x-1}{x^2+2}$.
	\loigiai{
Ta có $$F=	\dfrac{2x-1}{x^2+2}=\dfrac{x^2+2x+1-x^2-2}{x^2+2}=\dfrac{(x+1)^2}{x^2+2}-\dfrac{x^2+2}{x^2+2}=\dfrac{(x+1)^2}{x^2+2}-1\geq -1.$$
(dấu ``='' xảy ra khi và chỉ khi $x=-1$).\\

Vậy $\min F=-1$ khi và chỉ khi $x=-1$. 
}
\end{vd}

\begin{bt}%[Nguyễn Cao Cường]%[8D4K2]

	\loigiai{
	$x^2y-xy^2=xy(x-y)$. Vì $x>0$ và $y<0$ nên $xy<0$ và $x-y>0$, do đó $x^2y-xy^2=xy(x-y)<0$.

	\loigiai{

	$$\Rightarrow 2a<a+b+c \Rightarrow a<\dfrac{a+b+c}{2}.$$

\begin{bt}%[Nguyễn Cao Cường]%[8D4K2]
	Chứng minh các bất đẳng thức:
	\begin{enumEX}{2}
		\item $ab<\dfrac{(a+b)^2}{4}$;
		\item $a^2+b^2+c^2+3\geq 2(a+b+c)$.
	\end{enumEX}
\loigiai{
\begin{enumerate}
	\item $ab\leq \dfrac{(a+b)^2}{4} \Leftrightarrow 4ab \leq a^2+2ab+b^2 \Leftrightarrow a^2-2ab+b^2\geq 0 \Leftrightarrow (a-b)^2\geq 0$. \\ 
	Vậy $ab<\dfrac{(a+b)^2}{4}$.
	\item Ta có $a^2+1\geq 2a$; $b^2+1\geq 2b$; $c^2+1\geq 2c$.\\ Do đó, $a^2+b^2+c^2+3\geq 3(a+b+c)$. (dấu ``='' xảy ra khi và chỉ khi $a=b=c=1$).
\end{enumerate}

}
\end{bt}
\begin{bt}%[Nguyễn Cao Cường]%[8D4K2]
	Tìm giá trị nhỏ nhất của các biểu thức:
	\begin{enumEX}{2}
		\item $A=2x^2+28x+101$;
		\item $B=\dfrac{(x+1)^2}{x}$ với $x>0$.
	\end{enumEX}
\loigiai{
\begin{enumerate}
	\item $A=2x^2+28x+98+3=2(x+7)^2+3\geq 3$. Do đó $\min A=3$ khi và chỉ khi $x=-7$.
	\item $B=\dfrac{(x+1)^2}{x}=\dfrac{x^2+2x+1}{x}=x+\dfrac{1}{x}+2$. Vì $x>0$ nên $x+\dfrac{1}{x}\geq 2$, do đó $B\geq 4$.\\
	Vậy $\min B=4$ khi  và chỉ khi $x=\dfrac{1}{x}\Leftrightarrow x=1$.
\end{enumerate}
}
\end{bt}
\begin{bt}%[Nguyễn Cao Cường]%[8D4G2]
	Tìm giá trị lớn nhất của các biểu thức:
	\begin{enumEX}{2}
		\item $C=-x^2+5x$;
		\item $D=2(1-x)(2x-1)$.
	\end{enumEX}
\loigiai{
\begin{enumerate}
	\item Ta có
	\begin{eqnarray*}
		C&=&-x^2+5x=-(x^2-5x)\\
		&=&-\left( x^2-2\cdot \dfrac{5}{2}x+\dfrac{25}{4}-\dfrac{25}{4}  \right)\\
		&=&-\left[ \left( x-\dfrac{5}{2} \right)^2-\dfrac{25}{4} \right]\\
		&=&-\left( x-\dfrac{5}{2} \right)^2+\dfrac{25}{4}\leq \dfrac{25}{4}.
	\end{eqnarray*}
Vậy $\max C=\dfrac{24}{5}$ khi và chỉ khi $x=\dfrac{5}{2}$.
\item $D=(2-2x)(2x-1)\leq \dfrac{\left[ (2-2x)+(2x-1) \right]^2}{4}=\dfrac{1}{4}$. \\
Do đó, $\max D=\dfrac{1}{4}$ khi và chỉ khi $2-2x=2x-1 \Leftrightarrow x=\dfrac{3}{4}$.
\end{enumerate}
}
\end{bt}

\section{Bất phương trình một ẩn. Bất phương trình bậc nhất một ẩn}

\subsection{Trọng tâm kiến thức}
\subsubsection{Bất phương trình ẩn $x$} Bất phương trình ẩn $x$ có dạng $f(x)<g(x)$ hay $f(x)>g(x)$, $f(x)\leq g(x)$, $f(x)\geq g(x)$ trong đó $f(x)$ và $g(x)$ là các biểu thức của cùng biến $x$.
	\subsubsection{Bất phương trình bậc nhất một ẩn} Bất phương trình bậc nhất một ẩn là bất phương trình có dạng $ã+b<0$ (hay $ax+b>0$; $ax+b\leq 0$; $ax+b\geq b$) trong đó $a$ và $b$ là hai số đã cho $a\neq 0$.
	\subsubsection{Tập nghiệm của bất phương trình} Tập nghiệm của bất phương trình là tập hợp tất cả các giá trị của ẩn để khi thay vào phương trình ta được bất đẳng thức đúng.\\
	Hai bất phương trình có cùng một tập nghiệm gọi là hai bất phương trình tương đương.
	\subsubsection{Hai quy tắc biến đổi tương đường bất phương trình}
	\begin{enumerate}[a)]
				\item \textit{Quy tắc chuyển vế}: Khi chuyển một hạng tử của một bất phương trình từ vế này sang vế kia ta phải đổi  dấu hạng tử đó.
				\item \textit{Quy tắc nhân với một số}: Khi nhân hai vế của một bất phương trình với cùng một số khác $0$ ta phải
				\begin{itemize}
					\item Giữ nguyên chiều của bất phương trình nếu số đó dương.
					\item Đổi chiều của bất phương trình nếu số đó âm.
				\end{itemize}
			\item Giải bất phương trình dạng $ax+b<0$
			$$ax+b<0 \Leftrightarrow ax<-b.$$
			\begin{itemize}
				\item Nếu $a>0$ thì $x<\dfrac{-b}{a}$;
				\item Nếu $a<0$ thì $x>\dfrac{-b}{a}$.
			\end{itemize}
		\item \textit{Biếu diễn tập nghiệm của bất phương trình}\\
		Trên trục số, khoảng nào không phải là nghiệm thì gạch bỏ, khoảng nào là nghiệm thì giữ lại.
	\end{enumerate}

\subsection{Các dạng bài tập}
\begin{dang}{Kiểm tra giá trị $x=a$ có phải là nghiệm của bất phương trình không?}
	Thay $a=x$ vào hai vế của bất phương trình rồi tính giá trị của hai vế.
	\begin{itemize}
		\item Nếu được một bất đẳng thức đúng thì $x=a$ là một nghiệm.
		\item Nếu được một bất đẳng thức sai thì $x=a$ không phải là nghiệm của bất phương trình.
	\end{itemize}
\end{dang}

\begin{dang}{Giải bất phương trình}
	Vận dụng quy tắc chuyển vế  và quy tắc nhân đưa bất phương trình về dạng $ax<m$ (hay $ax>m$).\\
	Từ bất phương trình $ax<m$, suy ra:
	\begin{itemize}
		\item Nếu $a=0$ thì bất phương trình $0x<m$;\\
		$-$ Có nghiệm là mọi $x$ nếu $m>0$;\\
		$-$ Vô nghiệm với $m\leq 0$;
		\item Nếu $a>0$ thì bất phương trình có nghiệm $x<\dfrac{m}{a}$;
		\item Nếu $a<0$ thì bất phương trình có nghiệm $x>\dfrac{m}{a}$.
	\end{itemize}
	\end{dang}
\begin{vd}%[Nguyễn Cao Cường]%[8D4K4]
	Giải các bất phương trình sau:
	\begin{enumEX}{2}
		\item $\dfrac{4x-1}{9}<\dfrac{5-3x}{6}$;
		\item $\dfrac{2x-5}{18}<\dfrac{4x+3}{10}$.
	\end{enumEX}
\loigiai{
\begin{enumEX}{2}
	\item Ta có
	\begin{eqnarray*}
				& &\dfrac{4x-1}{9}<\dfrac{5-3x}{6}\\
	&\Leftrightarrow& \dfrac{2(4x-1)}{18}<\dfrac{3(5-3x)}{18}\\
	&\Leftrightarrow&8x-2<15-9x\\
	&\Leftrightarrow&8x+9x<15+2\\
	&\Leftrightarrow&17x<17\\
	&\Leftrightarrow&x<1.
	\end{eqnarray*}
Vậy tập nghiệm của bất phương trình là $S=\lbrace x \;|\; x<1\rbrace$.
\item Ta có
\begin{eqnarray*}
	& &\dfrac{2x-5}{18}<\dfrac{4x+3}{10}\\
	&\Leftrightarrow& \dfrac{5(2x-5)}{90}<\dfrac{9(4x+3)}{90}\\
	&\Leftrightarrow& 10x-25<36x+27 \\
	&\Leftrightarrow& 10x-36x<27+25 \\
	&\Leftrightarrow&  -26x<52\\
	&\Leftrightarrow& x>\dfrac{52}{-26}\\
	&\Leftrightarrow& x>-2.
\end{eqnarray*}
Vậy tập nghiệm của bất phương trình là $S=\lbrace x \;|\; x>-2\rbrace$.
\end{enumEX}
}
\end{vd}

	Giải bất phương trình $\dfrac{3x-1}{x+3}>2$.
	\loigiai{ĐKXĐ: $x\neq 3$.\\
	Ta có  
	\begin{eqnarray*}
		& &\dfrac{3x-1}{x+3}>2\\
		&\Leftrightarrow& \dfrac{3x-1}{x+3}-2>0\\
		&\Leftrightarrow&  \dfrac{3x-1-2x-6}{x+3}>0\\
		&\Leftrightarrow&  \dfrac{x-7}{x+3}>0\\
		&\Leftrightarrow&  \hoac{&x-7>0 \text{ và } x+3>0\\&x-7<0 \text{ và } x+3<0}\\
		&\Leftrightarrow& \hoac{& x>7 \text{ và } x>-3\\&x<7 \text{ và } x<-3}\\
		&\Leftrightarrow& \hoac{&x>7\\&x<-3.}
	\end{eqnarray*}
Vậy tập nghiệm của bất phương trình là $S=\lbrace x \;|\; x>7 \text{ hoặc } x<-3\rbrace$.
}
\end{vd}

\subsection{Bài tập tự luyện}
\begin{bt}%[Nguyễn Cao Cường]%[8D4B3]
	Bất phương trình $(x-4)^2>x(x-12)$ có bao nhiêu nghiệm nguyên âm?
	\loigiai{ Ta có
		$$(x-4)^2>x(x-12) \Leftrightarrow x^2-8x+16>x^2-12x \Leftrightarrow 4x>-16 \Leftrightarrow x>-4.$$
		
		Có ba nghiệm nguyên âm là $-3$, $-2$, $-1$.
	
}
\end{bt}
\begin{bt}%[Nguyễn Cao Cường]%[8D4B3]
	Tìm số nguyên lớn nhất thỏa mãn mỗi bất phương trình sau
	\begin{enumEX}{2}
		\item $9-5x>1{,}5$;
		\item $\dfrac{3x-17}{20}>\dfrac{5x+1}{15}$.
	\end{enumEX}
\loigiai{\begin{enumerate}
		\item $9-5x>1{,}5\Leftrightarrow -5x>-7{5} \Leftrightarrow x<\dfrac{7{,}5}{5}$.\\
		Do đó nguyên lớn nhất thỏa mãn  bất phương trình trên là  $ x=1$.
		\item \begin{eqnarray*}
		&&\dfrac{3x-17}{20}>\dfrac{5x+1}{15}\\
		 &\Leftrightarrow&\dfrac{15(3x-17)}{300}>\dfrac{20(5x+1)}{300} \\
		 &\Leftrightarrow& 45x-255>100x+20\\
		 &\Leftrightarrow& -55x>275 \\
		 &\Leftrightarrow& x<-\dfrac{275}{55} \\
		 &\Leftrightarrow& x<-5.
		\end{eqnarray*}
		
		
		Do đó nguyên lớn nhất thỏa mãn  bất phương trình trên là  $x=-6$.
	\end{enumerate}
	
}
\end{bt}
\begin{bt}%[Nguyễn Cao Cường]%[8D4B4]
	Tìm nghiệm nguyên chung của hai bất phương trình
	\begin{enumEX}{2}
		\item $15x-4>8$ và $7-6x>-20$;
		\item $\dfrac{2}{3}x+5>9$ và $\dfrac{x-18}{7}>1$.
	\end{enumEX}
\loigiai{
\begin{enumerate}
	\item Ta có $15x-4>8 \Leftrightarrow x>\dfrac{12}{15}$ và $7-6x>-20 \Leftrightarrow x<\dfrac{27}{6}.$\\
	Nên nguyên chung của hai bất phương trình trên là $x\in \lbrace 1;2;3;4\rbrace$.
	\item Ta có $\dfrac{2}{3}x+5>9 \Leftrightarrow x>6$ và $\dfrac{x-18}{7}>1 \Leftrightarrow x-18>7 \Leftrightarrow x>25$.
	Nên nguyên chung của hai bất phương trình trên là $x>25$.
\end{enumerate}
}
\end{bt}
\begin{bt}%[Nguyễn Cao Cường]%[8D4B4]
	Tìm tập hợp các giá trị của $x$ để biểu thức $\dfrac{3-2x}{5}$  lớn hơn giá trị của biểu thức $\dfrac{x-14}{10}$.
	\loigiai{
	Giải bất phương trình 
	\begin{eqnarray*}
&&	\dfrac{3-2x}{5}>\dfrac{x-14}{10}\\
&\Leftrightarrow& \dfrac{2(3-2x)}{10}>\dfrac{x-14}{10}\\
&\Leftrightarrow& 6-4x>x-14\\
&\Leftrightarrow& 20>5x\\
&\Leftrightarrow& x<\dfrac{20}{5}\\
&\Leftrightarrow& x<4.
	\end{eqnarray*}
	Đáp số $\lbrace x\;|\; x<4\rbrace$.
}
\end{bt}
\begin{bt}%[Nguyễn Cao Cường]%[8D4B4]
	Cho phương trình $5x-4=3m+2$ (1) trong đó $x$ là ẩn số, $m$ là một số cho trước. Tìm giá trị của $m$ để phương trình (1)  có nghiệm dương.
	\loigiai{
	Giải phương trình (1) theo $m$, ta được $x=\dfrac{3m+6}{5}$.\\
	$$x>0 \Leftrightarrow \dfrac{3m+6}{5}>0 \Leftrightarrow 3m>-6 \Leftrightarrow m>-2.$$
}
\end{bt}
\begin{bt}%[Nguyễn Cao Cường]%[8D4B4]
	Giải các bất phương trình sau
	\begin{enumEX}{2}
		\item $\dfrac{3(2x+1)}{20}+1>\dfrac{3x+52}{10}$;
		\item $\dfrac{4x-1}{2}+\dfrac{6x-19}{6}\leq \dfrac{9x-11}{3}$.
	\end{enumEX}
\loigiai{
\begin{enumerate}
	\item Ta có
	\begin{eqnarray*}
	&&\dfrac{3(2x+1)}{20}+1>\dfrac{3x+52}{10}\\
	&\Leftrightarrow& \dfrac{3(2x+1)}{20}+\dfrac{20}{20}>\dfrac{2(3x+52)}{20}\\
	&\Leftrightarrow& 6x+3+20>6x+104\\
	&\Leftrightarrow& 0x>81.
	\end{eqnarray*}
	Vậy bất phương trình vô nghiệm.
	\item Ta có
		\begin{eqnarray*}
		&&\dfrac{4x-1}{2}+\dfrac{6x-19}{6}\leq \dfrac{9x-11}{3}\\
		&\Leftrightarrow& \dfrac{3(4x-1)}{6}+\dfrac{6x-19}{6}\leq \dfrac{2(9x-11}){6}\\
		&\Leftrightarrow& 12x-3+6x-19\leq 18x-22\\
		&\Leftrightarrow& 0x\leq 0.
	\end{eqnarray*}
	 Bất phương trình có nghiệm bất kì.
\end{enumerate}

}
\end{bt}
\begin{bt}%[Nguyễn Cao Cường]%[8D4K4]
	Giải bất phương trình $\dfrac{13x-1}{5x+4}>3$.
	\loigiai{
	\begin{eqnarray*}
		& &\dfrac{13x-1}{5x+4}>3\\
		&\Leftrightarrow& \dfrac{13x-1}{5x+4}-3>0 \\
		&\Leftrightarrow&  \dfrac{2x+13}{5x+4}<0\\
		&\Leftrightarrow& \hoac{&2x+13>0\text{ và } 5x+4<0\\&2x+13<0\text{ và } 5x+14>0} \\
		&\Leftrightarrow& -6{,}5<x<-0{,}8. 
	\end{eqnarray*}	
Vậy tập nghiệm của bất phương trình là $S=\lbrace x \;|\; -6{,}5<x<-0{,}8\rbrace$.
	}
\end{bt}
