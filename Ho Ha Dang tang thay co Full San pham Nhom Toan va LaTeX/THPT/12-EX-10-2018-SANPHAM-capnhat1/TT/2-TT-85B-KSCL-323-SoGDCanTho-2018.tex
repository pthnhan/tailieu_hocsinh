\begin{ex}%[KSCL, Sở GD Cần Thơ, 2018 - mã đề 232]%[Dương Phước Sang, Ex-10-2018]%[2H1G3-4]%[1H3G5-4]
	Cho hình chóp $S.ABCD$ có đáy là hình vuông cạnh $2a$, tam giác $SAB$ đều, góc giữa $(SCD)$ và $(ABCD)$ bằng $60^{\circ}$. Gọi $M$ là trung điểm của cạnh $AB$. 	\choice
	{\True $\dfrac{a\sqrt{5}}{5}$}
	{$\dfrac{a\sqrt{5}}{10}$}
	{$\dfrac{3a\sqrt{5}}{10}$}
	{$\dfrac{5a\sqrt{3}}{3}$}
	\loigiai{
		\immini{Gọi $I$ là trung điểm cạnh $CD$, khi đó $\left\{\begin{aligned}
			&AB\perp SM \\
			&AB\perp MI
			\end{aligned}\right. \Rightarrow AB\perp (SMI)$. \\
			Do $CD \parallel AB$ nên $CD\perp (SMI) \Rightarrow ((SCD),(ABCD))=\widehat{SIM}$. \\

		Gọi $N$ là trung điểm cạnh $BC$ ta có $AC \parallel MN$ \\

		Vậy $\mathrm{d}(AC,SM)=\dfrac{2}{3}\dfrac{SH\cdot HK}{\sqrt{SH^2+HK^2}}=\dfrac{a\sqrt{5}}{5}$.}
\end{ex}
