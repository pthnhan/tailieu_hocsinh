	Cho phương trình $x^2-(2m+3)x+3m+1=0$, $m$ là tham số.
%\begin{enumerate}
\begin{listEX}
	\item Tìm tất cả các số thực $m$ để phương trình đã cho có hai nghiệm $x_1$, $x_2$ thỏa mãn điều kiện $x_1^2+x_2^2-x_1x_2=7.$
	\item Tìm tất cả các số nguyên $m$ để phương trình đã cho có nghiệm nguyên.
\end{listEX}
%\end{enumerate}
\loigiai{
\begin{enumerate}
%\begin{listEX}
	\item Ta có $\Delta=(2m+3)^2-4(3m+1)=4m^2+5>0$ với mọi $m$. Do đó, phương trình luôn có hai nghiệm $x_1$, $x_2$.\\
	Theo định lý Vi-et, ta có $\heva{&x_1+x_2=2m+3\\&x_1x_2=3m+1}$.\\
	Khi đó 
	$$\begin{aligned}
	x_1^2+x_2^2-x_1x_2=7 &\Leftrightarrow \left(x_1+x_2\right)^2-3x_1x_2=7\\ &\Leftrightarrow \left(2m+3\right)^2-3\left(3m+1\right)=7\\
	&\Leftrightarrow 4m^2+3-1=0 \Leftrightarrow \hoac{&m=-1\\&m=\dfrac{1}{4}.}
	\end{aligned}$$
	Vậy $m=-1; m=\dfrac{1}{4}.$
	\item Để phương trình có nghiệm nguyên thì $\Delta=4m^2+5$ phải là một số chính phương. Đặt $4m^2+5=k^2$,\ $k \in \mathbb{N}.$\\
	Khi đó $(k-2m)(k+2m)=5=1\cdot5=(-1)\cdot(-5).$ Suy ra các trường hợp. 
	\begin{enumerate}[\textbf{TH1.}]
		\item $\heva{&k-2m=1\\&k+2m=5}\Rightarrow m=1.$
		\item $\heva{&k-2m=5\\&k+2m=1} \Rightarrow m=-1.$
		\item $\heva{&k-2m=-1\\&k+2m=-5} \Rightarrow m=-1.$
		\item $\heva{&k-2m=-5\\&k+2m=-1} \Rightarrow m=1.$
	\end{enumerate}
	Thử lại với $m=1$, $m=-1$ thì phương trình đã cho đều có nghiệm nguyên. Vậy $m=1$, $m=-1.$
%\end{listEX}
\end{enumerate}
}
\end{ex}

	Giải phương trình $\sqrt{x}+\sqrt{x+3}=\sqrt{2x^2+4x+3}$.	
	\loigiai{
	Điều kiện: $x\ge0.$\\
	Ta có 
	$$\begin{aligned}
	&\sqrt{x}+\sqrt{x+3}=\sqrt{2x^2+4x+3}\\
	&\Leftrightarrow x+x+3+2\sqrt{x(x+3)}=2x^2+4x+3\\
	&\Leftrightarrow 2\sqrt{x(x+3)}=2x^2+2x \Leftrightarrow \sqrt{x(x+3)}=x^2+x\\
	&\Leftrightarrow x(x+3)=x^2(x+1)^2 \Leftrightarrow \hoac{&x=0\\&x+3=x(x^2+2x+1)} \\
	&\Leftrightarrow \hoac{&x=0\\&x^3+2x^2-3=0}
	\Leftrightarrow \hoac{&x=0\\&(x-1)(x^2+3x+3)}=0 \Leftrightarrow \hoac{&x=0\\&x=1.}
	\end{aligned}$$
 Đối chiếu điều kiện ta có nghiệm là $x=0$, $x=1.$}	
\end{ex}
\begin{ex}%[Đề thi vào 10 chuyên Toán V2, chuyên Đại Học Vinh, tỉnh Nghệ An, năm 2018]%[Nguyễn Văn Vũ, Dự án EX-TS10C-18]%[9D0G2]
	Giải hệ phương trình $\heva{&x+\dfrac{1}{x}+y+\dfrac{1}{y}=3\\&x^2+\dfrac{1}{x^2}+y^2+\dfrac{1}{y^2}=5.}$
	\loigiai{
	Điều kiện: $x,y\ne0.$ Ta có hệ phương trình đã cho tương đương với
	$$\heva{&\left(x+\dfrac{1}{x}\right)^2+\left(y-\dfrac{1}{y}\right)^2=5\\&x+\dfrac{1}{x}+y-\dfrac{1}{y}=3.}$$
	Đặt $a=x+\dfrac{1}{x}$, $b=y-\dfrac{1}{y}$ ta được hệ phương trình $\heva{&a^2+b^2=5\\&a+b=3} \Leftrightarrow \hoac{&a=2,\ b=1\\&a=1,\ b=2.}$
	\begin{enumerate}[\textbf{TH1.}]
		\item Với $a=2, b=1$ ta có $\heva{&x+\dfrac{1}{x}=2\\&y-\dfrac{1}{y}=1}\Leftrightarrow \heva{&x^2-2x+1=0\\&y^2-y-1=0} \Leftrightarrow \heva{&x=1\\&y=\pm\dfrac{1\pm\sqrt{5}}{2},}$
		\item Với $a=1, b=2$ ta có $\heva{&x+\dfrac{1}{x}=1\\&y-\dfrac{1}{y}=2}\Leftrightarrow \heva{&x^2-x+1=0\\&y^2-2y-1=0}$ (vô nghiệm).
	\end{enumerate}
	Vậy nghiệm $(x,y)$ là $\left(1;\dfrac{1+\sqrt{5}}{2}\right), \left(1;\dfrac{1-\sqrt{5}}{2}\right).$
	}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Toán V2, chuyên Đại Học Vinh, tỉnh Nghệ An, năm 2018]%[Nguyễn Văn Vũ, Dự án EX-TS10C-18]%[9D0G4]
    Cho số tự nhiên $n$ ($n\ge2$) và số nguyên tố $p$ thỏa mãn $p-1$ chia hết cho $n$ đồng thời $n^3-1$ chia hết cho $p$. Chứng minh rằng $n+p$ là một số chính phương.
    \loigiai{
	Ta có $n^3-1=(n-1)(n^2+n+1)\ \vdots \ p.$ \hfill$(1)$\\
	Do $p-1\ \vdots \ n$ nên $p-1 \ge n$ hay $p\ge n+1$. Do đó $n-1<p$ nên từ $(1)$ suy ra $n^2+n+1\ \vdots \ p.$  \\
	Đặt $p-1=an,$ $a \in \mathbb{N}, a\ge1.$ Khi đó $p=an+1$ và $n^2+n+1\ \vdots \ an+1$. Suy ra $a\le n+1$ (vì nếu $a>n+1$ thì $an+1>(n+1)n+1$, mâu thuẫn).\hfill$(2)$\\
	Mặt khác $a(n^2+n+1)-n(an+1)\ \vdots \ an+1$ hay $(a-n)n+a \ \vdots \ an+1.$ \hfill$(3)$\\
	Do $a\ge1$ nên $(a-1)n+a>0$, do đó từ $(3)$ suy ra $(a-1)n+a\ge an+1$ hay $a\ge n+1$. \hfill$(4)$\\
	Từ $(2)$ và $(4)$ suy ra $a=n+1$. Do đó $p=(n+1)n+1=n^2+n+1.$ Suy ra $n+p=n^2+2n+1=(n+1)^2$ là số chính phương, đpcm. 
}
\end{ex} 

\begin{ex}%[Đề thi vào 10 chuyên Toán V2, chuyên Đại Học Vinh, tỉnh Nghệ An, năm 2018]%[Nguyễn Văn Vũ, Dự án EX-TS10C-18]%[9D0G7]
	Cho các số thực không âm $a$, $b$ thỏa mãn $(a-b)^2=a+b+2.$ Chứng minh rằng \begin{center}
	$\left(1+\dfrac{a^3}{(b+1)^3}\right)\cdot\left(1+\dfrac{b^3}{(a+1)^3}\right)\le9.$\end{center}
\loigiai{
	Ta có
	$$\begin{aligned}
	&(a-b)^2=a+b+2 \Leftrightarrow a^2-2ab+b^2=a+b+2\\
	&a(a+1)+b(b+1)=2(a+1)(b+1) \Leftrightarrow \dfrac{a}{b+1}+\dfrac{b}{a+1}=2.
	\end{aligned}$$
Đặt $x=\dfrac{a}{b+1}$, $y=\dfrac{b}{a+1}$. Khi đó $x,y\ge0$ và $x+y=2.$\\ Bất đẳng thức cần chứng minh trở thành $(1+x^3)(1+y^3)\le9$
	$$\begin{aligned}
	&\Leftrightarrow 1+x^3y^3+x^3+y^3\le9\\
	&\Leftrightarrow x^3y^3+(x+y)\left((x+y)^2-3xy\right)\le8\\
	&\Leftrightarrow x^3y^3+2(4-3xy)\le 8\\
	&\Leftrightarrow xy(x^2y^2-6)\le0, \texttt{đúng vì}\ 0\le xy\le\dfrac{(x+y)^2}{4}=1. 
	\end{aligned}$$ 
	Dấu đẳng thức xảy ra khi và chỉ khi $\heva{&xy=0\\&x+y=2}$ hay $\hoac{&a=0,\ b=2\\&a=2,\ b=0.}$
}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Toán V2, chuyên Đại Học Vinh, tỉnh Nghệ An, năm 2018]%[Nguyễn Văn Vũ, Dự án EX-TS10C-18]%[9D0G8]
Cho $17$ số tự nhiên mà các chữ số của mỗi số được lấy từ tập hợp $\{0;1;2;3;4\}$. Chứng minh rằng ta có thể chọn được $5$ số trong $17$ số đã cho sao cho tổng của $5$ số này chia hết cho $5$. 
\loigiai{
Kí hiệu $T_0$, $T_1$, $T_2$, $T_3$, $T_4$ lần lượt là tập hợp các số có chữ số tận cùng là $0, 1, 2, 3, 4$.\\
- Nếu mỗi tập hợp trên đều khác rỗng thì ta chọn từ mỗi tập một phần tử,. Khi đó tổng của $5$ số được chọn có tận cùng bằng $0$ nên chia hết ch $5$.\\
- Nếu có một tập rỗng thì khi đó theo nguyên lí Đirichlê, trong $4$ tập còn lại luôn có một tập có ít nhất $5$ phần tử. Ta chọn $5$ số từ tập này, khi đó tổng của $5$ số được chọn cũng chia hết cho $5$.\\
Vậy trong mọi trường hợp ta luôn chọn được $5$ số có tổng chia hết cho $5$.
}
\end{ex}
