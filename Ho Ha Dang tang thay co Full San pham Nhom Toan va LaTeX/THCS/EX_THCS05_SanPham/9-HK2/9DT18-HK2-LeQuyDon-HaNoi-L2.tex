\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Lê Quý Đôn, Hà Nội]%[Trần Bá Huy, EX-THCS05]%[9D1B2]%[9D1B8]%[9D1K8]
Cho hai biểu thức $A=\dfrac{x+\sqrt{x}+1}{x+1}$ và $B=\dfrac{1}{\sqrt{x}-1}-\dfrac{2\sqrt{x}}{x\sqrt{x}+\sqrt{x}-x-1}$ với $x\geq 0$, $x\neq 1$.
\begin{enumerate}
\item Tính giá trị biểu thức $A$ khi $x=\sqrt{\left(5+\sqrt{13}\right)^2}+\sqrt{\left(\sqrt{13}-4\right)^2}$.
\item Rút gọn biểu thức $B$.
\item Tìm giá trị nhỏ nhất của biểu thức $P=A\div B$ khi $x>1$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có\\
$\begin{aligned}
x& =\sqrt{\left(5+\sqrt{13}\right)^2}+\sqrt{\left(\sqrt{13}-4\right)^2}=\left|5+\sqrt{13}\right|+\left|\sqrt{13}-4\right|=5+\sqrt{13}+4-\sqrt{13}=9.\\
\Rightarrow A& =\dfrac{x+\sqrt{x}+1}{x+1}=\dfrac{9+\sqrt{9}+1}{9+1}=\dfrac{9+3+1}{10}=\dfrac{13}{10}.
\end{aligned}$
\item Ta có\\
$\begin{aligned}
B& =\dfrac{1}{\sqrt{x}-1}-\dfrac{2\sqrt{x}}{x\left(\sqrt{x}-1\right)+\left(\sqrt{x}-1\right)}=\dfrac{1}{\sqrt{x}-1}-\dfrac{2\sqrt{x}}{\left(\sqrt{x}-1\right)\left(x+1\right)}\\
&= \dfrac{x+1-2\sqrt{x}}{\left(\sqrt{x}-1\right)\left(x+1\right)}=\dfrac{\left(\sqrt{x}-1\right)^2}{\left(\sqrt{x}-1\right)\left(x+1\right)}=\dfrac{\sqrt{x}-1}{x+1}.
\end{aligned}$
\item Ta có\\
$\begin{aligned}
P& =A\div B=\dfrac{x+\sqrt{x}+1}{x+1}\div\dfrac{\sqrt{x}-1}{x+1}=\dfrac{x+\sqrt{x}+1}{x+1}\cdot \dfrac{x+1}{\sqrt{x}-1}=\dfrac{x+\sqrt{x}+1}{\sqrt{x}-1}\\
& = \dfrac{(x-1)+\left(\sqrt{x}-1\right)+3}{\sqrt{x}-1}=\sqrt{x}+1+1+\dfrac{3}{\sqrt{x}-1}=\left(\sqrt{x}-1\right)+\dfrac{3}{\sqrt{x}-1}+3.
\end{aligned}$\\
Áp dụng bất đẳng thức $a+b\geq 2\sqrt{ab}$ với $a, b>0$, đẳng thức xảy ra khi $a=b$, ta được\\
$$\left(\sqrt{x}-1\right)+\dfrac{3}{\sqrt{x}-1}\geq 2\sqrt{\left(\sqrt{x}-1\right)\cdot\dfrac{3}{\sqrt{x}-1}}=2\sqrt{3}.$$
Đẳng thức xảy ra khi
$$\sqrt{x}-1=\dfrac{3}{\sqrt{x}-1}\Leftrightarrow \left(\sqrt{x}-1\right)^2=3\Leftrightarrow \sqrt{x}-1=\pm \sqrt{3}\Rightarrow x=1+\sqrt{3}\ (\text{do }x>1).$$
Từ đó suy ra $P\geq 2\sqrt{3}+3$ và $\min P=2\sqrt{3}+3\Leftrightarrow x=1+\sqrt{3}$.
\end{enumerate}
}
\end{ex}



\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Lê Quý Đôn, Hà Nội]%[Trần Bá Huy, EX-THCS05]%[9D3K6]%[9D4K8]
\textit{Giải bài toán sau bằng cách lập phương trình hoặc hệ phương trình.}\\
Hai người cùng làm chung một công việc thì sau $3$ giờ $36$ phút làm xong. Nếu làm một mình thì người thứ nhất hoàn thành công việc sớm hơn người thứ hai là $3$ giờ. Hỏi nếu mỗi người làm một mình thì sau bao lâu xong công việc.
\loigiai{
Ta có $3$ giờ $36$ phút = $3{,}6$ giờ.\\
Gọi $x$ và $y$ là thời gian mà mỗi người làm một mình xong công việc ($x,y>3{,}6$).\\
Trong $1$ giờ,
\begin{itemize}
\item người thứ nhất làm được $\dfrac{1}{x}$ công việc;
\item người thứ hai làm được $\dfrac{1}{y}$ công việc;
\item cả hai người làm được $\dfrac{1}{3{,}6}$ công việc.
\end{itemize}
Suy ra $\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{1}{3{,}6}$.\\
Ngoài ra, theo giả thiết, nếu làm một mình thì người thứ nhất hoàn thành xong công việc sớm hơn người thứ hai $3$ giờ nên có phương trình $y-x=3$.\\
Giải hệ $\begin{cases}\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{1}{3{,}6}\\ y-x=3\end{cases}\Leftrightarrow \begin{cases}\dfrac{1}{x}+\dfrac{1}{x+3}=\dfrac{5}{18}\\ y=x+3\end{cases}\Leftrightarrow \begin{cases}5x^2-21x-54=0\\ y=x+3\end{cases}\Leftrightarrow \begin{cases}x=6\\ y=9.\end{cases}$\\
Vậy người thứ nhất làm một mình xong công việc trong $6$ giờ, người thứ hai làm một mình xong công việc trong $9$ giờ.
}
\end{ex}



\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Lê Quý Đôn, Hà Nội]%[Trần Bá Huy, EX-THCS05]%[9D3K4]
Giải hệ phương trình: $\begin{cases} 2\sqrt{x-2}+\dfrac{y}{y+3}=1\\ 4\sqrt{x-2}-\dfrac{3y}{y+3}=7.\end{cases}$
\loigiai{
Điều kiện $x\geq 2,\ y\neq -3$. Đặt $a=\sqrt{x-2},\ b=\dfrac{y}{y+3}$. Thu được\\
$\begin{aligned}
& \begin{cases}2a+b=1\\ 4a-3b=7\end{cases}\Leftrightarrow \begin{cases}4a+2b=2\\ 4a-3b=7\end{cases}\Leftrightarrow \begin{cases}5b=-5\\ 2a+b=1\end{cases}\Leftrightarrow \begin{cases}b=-1\\ a=1.\end{cases}
\end{aligned}$\\
Từ đó suy ra\\
$\begin{cases}\sqrt{x-2}=1\\ \dfrac{y}{y+3}=-1\end{cases}\Leftrightarrow \begin{cases}x-2=1\\ y=-y-3\end{cases}\Leftrightarrow \begin{cases}x=3\\ y=-\dfrac{3}{2}.\end{cases}$\\
Vậy hệ có nghiệm $(x;y)=\left(3;-\dfrac{3}{2}\right)$.
}
\end{ex}



\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Lê Quý Đôn, Hà Nội]%[Trần Bá Huy, EX-THCS05]%[9D4B4]%[9D4K6]
Cho phương trình: $x^2-(4m-1)x+3m^2-2m=0$ với $x$ là ẩn.
\begin{enumerate}
\item Giải phương trình khi $m=1$.
\item Tìm $m$ để phương trình có hai nghiệm $x_1,\ x_2$ thỏa mãn $x_1^2+x_2^2=7$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Khi $m=1$, phương trình trở thành $x^2-3x+1=0$. Ta có
$$\Delta =(-3)^2-4\cdot 1\cdot 1=5>0\Rightarrow \sqrt{\Delta}=\sqrt{5}.$$
Phương trình có hai nghiệm
$$x_1=\dfrac{3-\sqrt{5}}{2};\quad x_2=\dfrac{3+\sqrt{5}}{2}.$$
\item Ta có $\Delta=(4m-1)^2-4(3m^2-2m)=16m^2-8m+1-12m^2+8m=4m^2+1>0\ \forall m\in\mathbb{R}$ nên phương trình luôn có hai nghiệm phân biệt.\\
Theo định lí Vi-ét, ta có $x_1+x_2=-\dfrac{b}{a}=4m-1,\ x_1x_2=\dfrac{c}{a}=3m^2-2m$.\\
Theo giả thiết thì
{\allowdisplaybreaks
\begin{align*}
& x_1^2+x_2^2=7\Leftrightarrow (x_1+x_2)^2-2x_1x_2=7\\
\Leftrightarrow\ & (4m-1)^2-2(3m^2-2m)=7\\
\Leftrightarrow\ & 16m^2-8m+1-6m^2+4m=7\\
\Leftrightarrow\ & 10m^2-4m-6=0\Leftrightarrow \left[\begin{aligned}
& m=1\\ & m=-\dfrac{3}{5}.
\end{aligned}\right.
\end{align*}}
Vậy có hai giá trị thỏa mãn là $m=1$ và $m=-\dfrac{3}{5}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Lê Quý Đôn, Hà Nội]%[Trần Bá Huy, EX-THCS05]%[9D4G7]
Giải phương trình: $2\sqrt{x+5}+9=2\sqrt{2x+1}+\sqrt{2x^2+11x+5}$.
\loigiai{
Phương trình đã cho được viết lại
$$2\sqrt{x+5}+9=2\sqrt{2x+1}+\sqrt{(x+5)(2x+1)}.\quad (1)$$
Điều kiện $\begin{cases}x+5\geq 0\\ 2x+1\geq 0\end{cases}\Leftrightarrow \begin{cases}x\geq -5\\ x\geq -\dfrac{1}{2}\end{cases}\Leftrightarrow x\geq -\dfrac{1}{2}$.\\
Đặt $a=\sqrt{x+5}$, $b=\sqrt{2x+1}$. Điều kiện $a,b\geq 0$. Ta có
$$a^2=x+5,\ b^2=2x+1\Rightarrow 2a^2-b^2=9.\quad (2)$$
Phương trình $(1)$ trở thành
$$2a+9=2b+ab\Leftrightarrow 2a+(2a^2-b^2)=2b+ab.\quad (3)$$
Tiếp tục biến đổi phương trình $(3)$
$$(3)\Leftrightarrow (a^2-b^2)+(2a-2b)+(a^2-ab)=0\Leftrightarrow (a-b)(a+b+2+a)=0\Leftrightarrow \left[\begin{aligned}
& a=b\\ & 2a+b+2=0.
\end{aligned}\right.$$
Nhưng đẳng thức $2a+b+2=0$ không xảy ra do $a,b\geq 0$. Xét $a=b$, khi đó
$$\sqrt{x+5}=\sqrt{2x+1}\Leftrightarrow x+5=2x+1\Leftrightarrow x=4\ (\text{nhận}).$$
Vậy phương trình có một nghiệm $x=4$.
}
\end{ex}
