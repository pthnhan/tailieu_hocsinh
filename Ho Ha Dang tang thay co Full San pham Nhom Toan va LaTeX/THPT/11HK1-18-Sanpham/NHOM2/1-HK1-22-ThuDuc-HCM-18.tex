\begin{bt}%[HK1 THPT Thủ Đức, TP. HCM][Nguyễn Phú Thạch, DA 11HK1-18]%[1D2B2-1]
	\begin{enumerate}
	\item Có bao nhiêu số tự nhiên chẵn gồm $ 3 $ chữ số khác nhau được lập từ các chữ số $1,2,5,7,8$.
	\item Một lớp học có $40$ học sinh trong đó có $22$ học sinh nam và $18$ học sinh nữa. Hỏi có bao nhiêu cách lập ra một đội văn nghệ của lớp có $4$ học sinh sao cho đội văn nghệ đó có cả nam và nữ?
\end{enumerate}
	\loigiai{
	\begin{enumerate}
		 \item {Gọi $x=\overline{abc}$ là số thỏa mãn yêu cầu bài toán.\\
			Do $x \vdots 2 \Rightarrow c \in \{2;8\}\Rightarrow$ có hai cách chọn.\\
			$a$ có $ 4 $ cách chọn, $b$ có $3$ cách chọn. Suy ra có $4\cdot3\cdot2=24$ cách chọn.}
		\item {\begin{enumerate}[TH 1.]
				\item $1$ nam và $3$ nữ, có $\mathrm{C}_{22}^1\cdot \mathrm{C}_{18}^3$ cách chọn.
				\item $2$ nam và $2$ nữ, có $\mathrm{C}_{22}^2\cdot \mathrm{C}_{18}^2$ cách chọn.
				\item $3$ nam và $1$ nữ, có $\mathrm{C}_{22}^3\cdot \mathrm{C}_{18}^1$ cách chọn.
			\end{enumerate}
		Vậy có $\mathrm{C}_{22}^1\cdot \mathrm{C}_{18}^3+\mathrm{C}_{22}^2\cdot \mathrm{C}_{18}^2+\mathrm{C}_{22}^2\cdot \mathrm{C}_{18}^2=81015$ cách lập ra một đội văn nghệ của lớp có $4$ học sinh sao cho đội văn nghệ có cả nam và nữ.
		}
	\end{enumerate}
		
	}

\begin{bt}%[HK1 THPT Thủ Đức, TP. HCM][Nguyễn Phú Thạch, DA 11HK1-18]%[1D2K3-2]
Tìm số hạng chứa $x^7$ trong khai triển thành đa thức của $P(x)=(2x+1)(1-x)^{11}$.

\loigiai{
	Ta có $P(x)=2x(1-x)^{11}+(1-x)^{11}$.\\
	Xét khai triển $(1-x)^{11}$. Số hạng tổng quát thứ $k+1$ là\\ $T_{k+1}=\mathrm{C}_{11}^k1^k(-x)^{11-k}=\mathrm{C}_{11}^k(-1)^{11-k}x^{11-k}\quad (k\leq11,k\in \mathbb{N})$.\\
	Số hạng chứa $x^7$ ứng với $k$ thỏa mãn: $11-k=7 \Leftrightarrow k=4.$\\
	Xét khai triển  $2x(1-x)^{11}$. Số hạng chứa $x^7$ ứng với $k$ thỏa mãn: $11-k=6 \Leftrightarrow k=5$.\\
	Vậy số hạng chứa $x^7$ trong khai triển của $P(x)$ là $\left[2\mathrm{C}_{11}^5(-1)^6+\mathrm{C}_{11}^4(-1)^7\right]x^7$.
}
\end{bt}

\begin{bt}%[HK1 THPT Thủ Đức, TP. HCM][Nguyễn Phú Thạch, DA 11HK1-18]%[1D2K5-2]
	Xét bài toán: Một bao hạt giống gồm đậu xanh và đâu đỏ trong đó có $\dfrac{3}{5}$ hạt giống là đậu xanh, $\dfrac{2}{5}$ hạt giống là đậu đỏ. Do bao hạt giống này bị lỗi nên chỉ có $\dfrac{2}{5}$ hạt giống đậu xanh nảy mầm và $\dfrac{3}{4}$ hạt giống đậu đỏ nảy mầm. Lấy ngẫu nhiên trong bao một hạt giống và gieo thì thấy nó nảy mầm thành $1$ cậy đậu. Tính xác suất để cây đậu đó là cây đậu xanh.\\
	Một học sinh giải bài toán trên như sau:
	\begin{itemize}
		\item \textit{Bước 1:} Gọi số hạt giống trong bao là $5n \Rightarrow $ số hạt giống đậu xanh là $3n$, số hạt giống đậu đỏ là $2n$. Số hạt giống đậu xanh nảy mầm là $2n$, số hạt giống đậu đỏ nảy mầm là $\dfrac{3n}{2}$.
		\item \textit{Bước 2:} Số phần tử của không gian mẫu là $n(\Omega)=5n$.
		\item \textit{Bước 3:} Gọi $A$ là biến cố thỏa yêu cầu đề bài $\Rightarrow n(A)=2n$.
		\item \textit{Bước 4:} Xác suất cần tìm là $P(A)=\dfrac{n(A)}{n(\Omega)}=\dfrac{2}{5}$.
	\end{itemize}
Học sinh trên giải \textbf{đúng} hay \textbf{sai}? Nếu \textbf{sai} thì \textbf{sai} từ bước nào, lúc đó hãy giải lại cho đúng bài toán trên.
\loigiai{
	Học sinh trên giải sai (bước 2).\\
	Số phần tử của không gian mẫu $n(\Omega)=2n+\dfrac{3n}{2}=\dfrac{7n}{2}$.\\
	Xác suất cần tìm $P(A)=\dfrac{n(A)}{n(\Omega)}=\dfrac{2n}{\dfrac{7n}{2}}=\dfrac{4}{7}$.
}
\end{bt}

\begin{bt}%[HK1 THPT Thủ Đức, TP. HCM][Nguyễn Phú Thạch, DA 11HK1-18]%[1D3K3-6]
	Để chuẩn bị vốn từ vựng cho kì thi IELTS sắp tới, ngày đầu tiên, ban An học $3$ từ vựng mới và ngay sau học nhiều hơn ngày trước đó $2$ từ mới. Hỏi để có vốn từ mới tối thiểu là $2000$ từ, bạn An cần phải học ít nhất bao nhiêu này?
	\loigiai{
		Vốn từ vựng mà bạn an cần học là một cấp số cộng $(u_n)$ với $u_1=3$, công sai $d=2$.\\
		Ta có \begin{align*}
		S_n\geq 2000 \Leftrightarrow &\dfrac{n}{2}[2\cdot 3+(n-1)2]\geq 2000 \Leftrightarrow n^2+2n-2000 \geq 0\\
		\Leftrightarrow &\hoac{&n\geq -1+2\sqrt{501}\\&n\leq -1-2\sqrt{501}}\Rightarrow n \geq 44.
		\end{align*}
		Vậy bạn An phải học tối thiểu $44$ ngày để được số từ vựng là $2000$ từ.
	}
\end{bt}

\begin{bt}%[HK1 THPT Thủ Đức, TP. HCM][Nguyễn Phú Thạch, DA 11HK1-18]%[1D3B4-5]
	Cho cấp số nhân $(u_n)$ có công bội $q>1$ thỏa mãn $\heva{&u_2=10\\&u_1+2u_3=45}$.
	Tìm số hạng đầu $u_1$, công bội $q$ và tổng của $12$ số hạng đầu tiên của cấp số nhân $(u_n)$.
	\loigiai{
		Ta có $\heva{&u_2=10\\&u_1+2u_3=45}\Leftrightarrow \heva{&u_1q=10\\&u_1+2u_1q^2=45}\Rightarrow \dfrac{q}{1+2q^2}=\dfrac{10}{45} \Leftrightarrow 4q^2-9q+2=0 \Leftrightarrow \hoac{&q=\dfrac{1}{4}\\&q=2.}$\\
		So điều kiện ta có $q=2 \Rightarrow u_1=5$. Vậy $S_{12}=\dfrac{u_1(1-q^{11})}{1-q}=10235$.
	}
\end{bt}
