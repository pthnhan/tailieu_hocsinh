		\end{align*}

\begin{ex}%[Đề thi tuyển sinh 10 năm học 2018-2019 tỉnh Tây Ninh]%[9D4B6]
	Tìm giá trị của $m$ để phương trình $2x^2-5x+2m-1=0$ có hai nghiệm phân biệt $x_1$ và $x_2$ thỏa mãn  $\dfrac{1}{x_1}+\dfrac{1}{x_2}=\dfrac{5}{2}.$\
	\loigiai{
	Phương trình có hai nghiệm phân biệt $x_1,\,x_2$ khi và chỉ khi	
	\begin{align*}
&	\begin{cases} a\neq 0\\ \Delta >0	\end{cases} \Leftrightarrow \begin{cases}
	2\neq 0\\ (-5)^2-4\cdot 2(2m-1)>0	\end{cases}\\
\Leftrightarrow&	25-16m+8>0 \Leftrightarrow 33-16m >0 \Leftrightarrow m <\dfrac{33}{16}.\quad \quad \quad (1)
	\end{align*}
	Theo công thức Viète ta có $\begin{cases}
	x_1+x_2=\dfrac{5}{2}\\ x_1x_2=\dfrac{2m-1}{2}	\end{cases}$.
$$	\dfrac{1}{x_1}+\dfrac{1}{x_2}=\dfrac{5}{2}\Leftrightarrow \dfrac{x_1+x_2}{x_1x_2}=\dfrac{5}{2}
	\Leftrightarrow \dfrac{\dfrac{5}{2}}{\dfrac{2m-1}{2}}=\dfrac{5}{2}\Leftrightarrow \dfrac{2}{2m-1}=1.	$$
Giải phương trình trên, ta được nghiệm $m=\dfrac{3}{2}$, nghiệm này không thỏa mãn điều kiện $(1)$. Vậy không tồn tại $m$ thỏa mãn yêu cầu bài toán.
	}
	\end{ex}
\begin{ex}%[Đề thi tuyển sinh 10 năm học 2018-2019 tỉnh Tây Ninh]%[9D4K8]
	Một đội máy xúc được thuê đào $20000$ m$^3$ đất để mở rộng hồ Dầu Tiếng. Ban đầu đội dự định đào mỗi ngày đào một lượng đất nhất định để hoàn thành công việc, nhưng khi đào được $5000$ m$^3$ đất thì đội tăng cường thêm một số máy xúc nên mỗi ngày đào thêm được $100$ m$^3$, do đó hoàn thành công việc trong $35$ ngày. Hỏi ban đầu đội dự định mỗi ngày đào bao nhiêu m$^3$ đất?
	\loigiai{
		Gọi $x$ (m$^3$) là lượng đất đội dự định đào trong một ngày, $(x>0)$.\\
		Thời gian đội đào $5000$ m$^3$ đất là $\dfrac{5000}{x}$ (ngày).\\
	Thời gian đội đào phần đất còn lại sau khi tăng số máy là $\dfrac{20000-5000}{x+100}=\dfrac{15000}{x+100}$ (ngày).\\
	Theo đề bài ta có phương trình
\begin{center}
$	\begin{array}{ll}
&	\dfrac{5000}{x}+\dfrac{15000}{x+100} =35\\
\Leftrightarrow &5000(x+100)+15000x=35x(x+100)\\
\Leftrightarrow &1000(x+100)+3000x=7x(x+100)\\
\Leftrightarrow &7x^2-3300x-100000=0.	
\end{array}$
\end{center}
		Xét phương trình bậc hai $x^2-3300x-100000=0$ có $a=7,\,b=-3300,\,c=-100000$.\\
		$\Delta = b^2-4ac=(-3300)^2-4\cdot 7\cdot (-100000)=13690000>0.$\\
		Suy ra phương trình trên có hai nghiệm phân biệt\\
		 $$x_1=\dfrac{-b+\sqrt{\Delta}}{2a}= 500,\, x_2=\dfrac{-b-\sqrt{\Delta}}{2a}=-\dfrac{200}{7}.$$
		Đối chiếu điều kiện, ta nhận nghiệm $x=500$.\\
		 Vậy mỗi ngày đội đào được $500$ m$^3$ đất.
}
	\end{ex}
\begin{ex}%[Đề thi tuyển sinh 10 năm học 2018-2019 tỉnh Tây Ninh]%[9H2K5]
	Cho tam giác $ABC$ có ba góc nhọn $(AB<AC)$ và có đường cao $AH$ ($H$ thuộc cạnh $BC$). Gọi $D,\,E$ lần lượt là trung điểm của $AB$ và $AC$. Chứng minh $DE$ là tiếp tuyến chung của hai đường tròn lần lượt ngoại tiếp tam giác $DBH$ và tam giác $ECH$.
	\loigiai{
		\begin{center}
				\begin{tikzpicture}
			\tkzInit[ymin=-3.8,ymax=6,xmin=-1,xmax=8.5]
			\tkzClip
		\coordinate(a)at(2,5);
		\coordinate(b)at(0,0);
		\coordinate(c)at(8,0);
		\tkzDrawPoints(a,b,c);
		\draw(b)node[left]{$B$}--(a)node[above]{$A$}--(c)node[right]{$C$}--(b);
			\tkzDrawAltitude(b,c)(a)\tkzGetPoint{H}\tkzLabelPoints[below right](H)
			\tkzDefMidPoint(a,b)
			 \tkzGetPoint{D}\tkzLabelPoints[above left](D)
			\tkzDefMidPoint(a,c) \tkzGetPoint{E}\tkzLabelPoints[above right](E)
			\draw(D)--(E)--(H)--(D);
			\tkzDefCircle[circum](D,b,H) \tkzGetPoint{I} \tkzGetLength{r}
			\tkzLabelPoints[right](I);
			\tkzDrawCircle(I,b);
				\tkzDefCircle[circum](E,c,H) \tkzGetPoint{J} \tkzGetLength{s}
			\tkzLabelPoints(J);
			\tkzDrawCircle(J,c);
			\draw(D)--(I);	\draw(b)--(I);	\draw(H)--(I);
			\draw(E)--(J);	\draw(c)--(J);	\draw(H)--(J);
				\tkzMarkRightAngles (a,H,b)
		\end{tikzpicture}
		\end{center}
		Gọi $I,\,J$ lần lượt là tâm đường tròn ngoại tiếp $\Delta BDH$ và $\Delta HEC$.\\
		$\Delta ABH$ vuông tại $H$ có $HD$ là trung tuyến nên $HD=\dfrac{1}{2}AB=BD$.\\
	Ta có
	\begin{align*}
	&\begin{cases}
&IB=IH \quad  \text{ ($I$ là tâm đường tròn ngoại tiếp $\Delta BDH$)}\\ & HD=BD \quad \text{(Chứng minh trên)}
	 \end{cases}\\
	 \Rightarrow & DI \text{ là đường trung trực của } BH\\
	 \Rightarrow & DI \perp BH. \quad (1)
	\end{align*}
Xét tam giác $ABC$ có
\begin{align*}
&\begin{cases}
&D \text{ là trung điểm } AB\\& E\text{ là trung điểm } AC
\end{cases}\\
\Rightarrow & DE \text{ là đường trung bình } \Delta ABC\\
\Rightarrow & DE \parallel BH.\quad (2)
\end{align*}
Từ $(1)$ và $(2)$ suy ra  $DE \perp DI$.\\
Chứng minh tương tự ta cũng được $DE\perp EJ$, vậy $DE$ là tiếp tuyến chung  của hai đường tròn lần lượt ngoại tiếp tam giác $DBH$ và tam giác $ECH$.
	}
\end{ex}
\begin{ex}%[Đề thi tuyển sinh 10 năm học 2018-2019 tỉnh Tây Ninh]%[9H1G2]
	Cho đường tròn tâm $O$ bán kính $2R$ (kí hiệu là $(O,2R)$) và đường tròn tâm $O'$ bán kính $R$ (kí hiệu là $(O';R)$) tiếp xúc ngoài tại điểm A. Lấy điểm $B$ trên đường tròn $(O;2R)$ sao cho $\widehat{BAO}=30^\circ$, tia $BA$ cắt đường tròn $(O';R)$ tại điểm $C$ (khác điểm $A$). Tiếp tuyến của đường tròn $(O';R)$ tại điểm $C$ cắt đường thẳng $BO$ tại điểm $E$. Tính theo $R$ diện tích tam giác $ABE$.
	\loigiai{
	\begin{center}
			\begin{tikzpicture}
		\tkzInit[ymin=-4.2,ymax=6,xmin=-5,xmax=10]
		\tkzClip
			\coordinate(a)at(4,0);
		\coordinate(i)at(0,0);
		\coordinate(j)at(6,0);
			\draw(i)node[left]{$O$}--(a)node[above left]{$A$}--(j)node[right]{$O'$};
		\tkzDrawCircle(i,a);
		\tkzDrawCircle(j,a);
		\tkzDefPointBy[rotation = center i angle -120](a)    \tkzGetPoint{b};
	\tkzInterLC(a,b)(j,a)    \tkzGetPoints{a}{c};
\tkzTangent[at=c](j)    \tkzGetPoint{t}
\tkzInterLL(b,i)(c,t) \tkzGetPoint{e}
\tkzDrawPoints(a,i,j,b,c,e);
\draw(b)node[below left]{$B$}--(a)--(c)node[above right]{$C$}--(e)node[above]{$E$}--(b);
\draw(j)--(c);
\draw(a)--(e);
\tkzDrawAltitude(a,b)(i)\tkzGetPoint{H}\tkzLabelPoints(H)
\tkzDrawAltitude(a,c)(e)\tkzGetPoint{K}\tkzLabelPoints[xshift=0.1cm, yshift=0.2cm](K)
	\tkzMarkRightAngles (e,c,j b,e,c i,H,b c,K,e)
	\end{tikzpicture}
	\end{center}
		Ta có \\
	$\begin{cases}
\widehat{O'AC}=\widehat{OAB} \quad&\text{(hai góc đối đỉnh)}\\
\widehat{O'AC}=\widehat{ACO'} &\text{($\Delta AO'C$ cân tại $O'$)}\\
\widehat{OAB}=\widehat{OBA}&\text{($\Delta AOB$ cân tại $O$)}
	\end{cases}$ 
		$\Rightarrow \widehat{ACO'}=\widehat{ABO} \Rightarrow O'C \parallel BE.$\\
		$$\Rightarrow \dfrac{AB}{AC}=\dfrac{OA}{O'A}=\dfrac{2R}{R}=2. \quad \text{(Định lý Ta-lét)}$$
		Dựng $OH \perp AB$ tại $H$, ta có
		$$	\begin{array}{rl}
BH=OB\cdot\cos B&=2R\cdot 30^\circ =2R\cdot \dfrac{\sqrt{3}}{2}=R\sqrt{3}\\
\Rightarrow  AB&=2BH=2R\sqrt{3}\\
\Rightarrow  AC&=\dfrac{1}{2}AB=R\sqrt{3}\\
\Rightarrow BC&=AB+AC=3R\sqrt{3}.
\end{array}$$
Ta có $$\left\{ \begin{array}{ll}
O'C \parallel BE\quad& \text{(chứng minh trên)}\\
O'C \perp CE &\text{(tính chất tiếp tuyến)}
\end{array}\right. \Rightarrow BE \perp EC.$$
Xét tam giác $BEC$ vuông tại $E$
\begin{align*}
&\begin{cases}
&EB=BC\cdot \cos B =3R\sqrt{3}\cdot \cos 30^\circ=\dfrac{9R}{2}\\
&EC=BC\cdot \sin B=3R\sqrt{3} \cdot \sin 30^\circ=\dfrac{3\sqrt{3}R}{2}.
\end{cases}\\
&\Rightarrow S_{BEC}=\dfrac{1}{2}BE\cdot EC=\dfrac{1}{2}\cdot \dfrac{9R}{2} \cdot \dfrac{3\sqrt{3}R}{2}=\dfrac{27\sqrt{3}}{8}R^2.
\end{align*}
Dựng $EK \perp AC $ tại $K$. Ta có
$$\dfrac{S_{AEB}}{S_{BEC}}=\dfrac{\dfrac{1}{2}EK\cdot AB}{\dfrac{1}{2}EK\cdot BC}=\dfrac{AB}{BC}=\dfrac{2}{3}.$$
$$\Rightarrow S_{AEB}=\dfrac{2}{3}S_{BEC}=\dfrac{2}{3}\cdot \dfrac{27\sqrt{3}}{8}R^2=\dfrac{9\sqrt{3}}{4}R^2.$$
	}
	\end{ex}
