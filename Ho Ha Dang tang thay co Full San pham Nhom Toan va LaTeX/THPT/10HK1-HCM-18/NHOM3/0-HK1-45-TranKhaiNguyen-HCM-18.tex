\begin{bt}%[Đề HK1, Toán 10, THPT Trần Khai Nguyên, Hồ Chí Minh 2017]%[Kiều Ngân]%[0D2B1-2]
	Tìm tập xác định của hàm số $y=\dfrac{2+3x}{\left( |x|-1 \right)(2-x)}$.
	\loigiai{
		Hàm số $y$ xác định khi và chỉ khi
		$\heva{&|x|-1 \ne 0\\& 2-x \ne 0} \Leftrightarrow \heva{&x \ne 1\\&x \ne -1\\&x \ne 2}$.\\
		Vậy tập xác định là $\mathscr{D}= \mathbb{R} \setminus \{-1;1;2\}$. 
	}
\end{bt}

\begin{bt}%[Đề HK1, Toán 10, THPT Trần Khai Nguyên, Hồ Chí Minh 2017]%[Kiều Ngân]%[0D2B1-3]
	Xét sự biến thiên của hàm số $f(x)=\dfrac{3}{2x-1}$ trên khoảng $\left( -\infty;\dfrac{1}{2} \right)$.
	\loigiai{
		Hàm số $y$ xác định trên khoảng $\left( -\infty;\dfrac{1}{2} \right)$.\\
		$\forall x_1, x_2 \in \left(-\infty;\dfrac{1}{2} \right), x_1 \ne x_2$, ta có
		\begin{itemize}
			\item $f(x_1)-f(x_2)=\dfrac{3}{2x_1 -1} -\dfrac{3}{2x_2 -1}
			=\dfrac{3(2x_2-1)-3(2x_1-1)}{(2x_1 -1)(2x_2 -1)}
			=\dfrac{-6(x_1-x_2)}{(2x_1 -1)(2x_2 -1)}$.
			\item $K=\dfrac{f(x_1)-f(x_2)}{x_1 -x_2} =\dfrac{-6}{(2x_1 -1)(2x_2 -1)}$.
			\item Vì $x_1, x_2 \in \left( -\infty;\dfrac{1}{2} \right)$ nên $\heva{&x_1<\dfrac{1}{2}\\&x_2<\dfrac{1}{2}} \Rightarrow \heva{&2x_1 -1<0\\&2x_2 -1<0} \Rightarrow (2x_1 -1)(2x_2 -1)>0 \Rightarrow K<0$.
		\end{itemize}
		Vậy hàm số $y$ nghịch biến trên khoảng $\left( -\infty;\dfrac{1}{2} \right)$. 
	}
\end{bt}

\begin{bt}%[Đề HK1, Toán 10, THPT Trần Khai Nguyên, Hồ Chí Minh 2017]%[Kiều Ngân]%[0D2B3-2]
	Viết phương trình của parabol $(P) \colon y=ax^2+bx+2$ biết $(P)$ qua điểm $A(1;-2)$ và cắt trục hoành tại điểm $B$ có hoành độ là $\dfrac{1}{3}$.
	\loigiai{
		Ta có $\heva{&A(1;-2) \in (P)\\& B\left(\dfrac{1}{3}; 0\right) \in (P)} 
		\Leftrightarrow \heva{&a+b+2=-2\\&\dfrac{1}{9}a+\dfrac{1}{3}b+2=0}
		\Leftrightarrow \heva{&a+b=-4\\&\dfrac{1}{9}a+\dfrac{1}{3}b=-2}
		\Leftrightarrow \heva{&a=3\\&b=-7}$.\\
		Vậy $(P) \colon y=3x^2-7x+2$. 
	}
\end{bt}

\begin{bt}%[Đề HK1, Toán 10, THPT Trần Khai Nguyên, Hồ Chí Minh 2017]%[Kiều Ngân]%[0D3K2-4]
	Giải các phương trình sau:
	\begin{enumerate}
		\item $\sqrt{x^2-3x+11}=5-x$.
		\item $\sqrt{x^2+\dfrac{1}{x^2}+\dfrac{19}{4}}=x+\dfrac{1}{x}+\dfrac{1}{2}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $\sqrt{x^2-3x+11}=5-x
				\Leftrightarrow \heva{&5-x \ge 0\\& x^2-3x+11=(5-x)^2}
				\Leftrightarrow \heva{&x \le 5\\&x^2-3x+11=25-10x+x^2}\\
				\Leftrightarrow \heva{&x \le 5\\&7x=14}
				\Leftrightarrow \heva{&x \le 5\\&x=2 \text{ (nhận)}}$.
			Vậy phương trình có nghiệm $x=2$.
			\item $\sqrt{x^2+\dfrac{1}{x^2}+\dfrac{19}{4}}=x+\dfrac{1}{x}+\dfrac{1}{2}$.\\
			Đặt $t=x+\dfrac{1}{x} \Rightarrow t^2=x^2+\dfrac{1}{x^2}+2 \Rightarrow t^2-2=x^2+\dfrac{1}{x^2}$.\\
			Phương trình trở thành:\\
			$\sqrt{t^2-2+\dfrac{19}{4}}=t+\dfrac{1}{2}
			\Leftrightarrow \heva{&t +\dfrac{1}{2} \ge 0\\& t^2 +\dfrac{11}{4} =\left( t +\dfrac{1}{2} \right)^2}
			\Leftrightarrow \heva{&t \ge -\dfrac{1}{2}\\& t^2 +\dfrac{11}{4} = t^2+t +\dfrac{1}{4}}
			\Leftrightarrow \heva{&t \ge -\dfrac{1}{2}\\& t = \dfrac{5}{2} \text{ (nhận)}}$.\\
				$t=\dfrac{5}{2} \Leftrightarrow x +\dfrac{1}{x} =\dfrac{5}{2} \Leftrightarrow 2x^2-5x+2=0
				\Leftrightarrow \hoac{&x=2\\&x=\dfrac{1}{2}}$.
				Vậy tập nghiệm $S=\left\{ 2; \dfrac{1}{2} \right\}$.
		\end{enumerate} 
	}
\end{bt}

\begin{bt}%[Đề HK1, Toán 10, THPT Trần Khai Nguyên, Hồ Chí Minh 2017]%[Kiều Ngân]%[0D3B2-6]
	Tìm tham số $m$ để phương trình $mx^2-2(m-3)x+m-5=0$ có nghiệm.
	\loigiai{
		Ta xét các trường hợp
		\begin{itemize}
			\item $a=0 \Leftrightarrow m=0$\\
			Phương trình trở thành $6x-5=0 \Leftrightarrow x=\dfrac{5}{6}$.\\
			Do đó nhận $m=0$.
			\item $a \ne 0 \Leftrightarrow m \ne 0$\\
			Phương trình có nghiệm khi và chỉ khi $\Delta' \ge 0 \Leftrightarrow -m+9 \ge 0 \Leftrightarrow m \le 9$.
		\end{itemize}
		Vậy $m \le 9$ thỏa yêu cầu bài toán. 
	}
\end{bt}

\begin{bt}%[Đề HK1, Toán 10, THPT Trần Khai Nguyên, Hồ Chí Minh 2017]%[Kiều Ngân]%[0D3B3-3]
	Giải hệ phương trình $\heva{&\dfrac{2x-6}{y+2} -\dfrac{x}{y-1} =-4\\&\dfrac{x-3}{y+2} +\dfrac{3x}{y-1} = \dfrac{31}{2}}.$
	\loigiai{
		Điều kiện: $\heva{&y \ne -2\\&y \ne 1}$.\\
		Đặt $\heva{&u=\dfrac{x-3}{y+2}\\&v=\dfrac{x}{y-1}}$.
		Hệ trở thành $\heva{&2u-v=-4\\&u+3v=\dfrac{31}{2}}
		\Leftrightarrow \heva{&u=\dfrac{1}{2}\\&v=5}$.\\
		Khi đó $\heva{&\dfrac{x-3}{y+2} =\dfrac{1}{2}\\&\dfrac{x}{y-1} =5}
		\Leftrightarrow \heva{&2x-y=8\\&x-5y=-5}
		\Leftrightarrow \heva{&x=5\\&y=2}$.\\  
		Vậy nghiệm của hệ phương trình là $(5;2)$. 
	}
\end{bt}

\begin{bt}%[Đề HK1, Toán 10, THPT Trần Khai Nguyên, Hồ Chí Minh 2017]%[Kiều Ngân]%[0H2B2-1][0H2B2-1][0H2B2-3][0H2B2-4]
	Trong mặt phẳng tọa độ $Oxy$ cho ba điểm $A(1;3)$, $B(3;5)$, $C(7;5)$.
	\begin{enumerate}
		\item Chứng minh $3$ điểm $A$, $B$, $C$ tạo thành một tam giác. Tính số đo góc $\widehat {ABC}$.
		\item Tìm tọa độ điểm $D$ để tứ giác $ABCD$ là hình bình hành. Gọi $I$ là tâm hình bình hành, tính $\vv{IA}\cdot \vv{IB}$.
		\item Tìm tọa độ điểm $E$ để tứ giác $ABCE$ là hình thang vuông có cạnh góc vuông là $AE$.
		\item Trong tam giác $ABC$, đường phân giác xuất phát từ đỉnh $A$ cắt cạnh $BC$ tại $K$, tìm tọa độ điểm $K$. 
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Chứng minh $3$ điểm $A$, $B$, $C$ tạo thành một tam giác. Tính số đo góc $\widehat {ABC}$.
				\begin{itemize}
				\item $\vv{AB}=(2;2)$, $\vv{AC}=(6;2)$.\\
				Ta có $\dfrac{2}{6} \ne \dfrac{2}{2}$ nên $\vv{AB}$ và $\vv{AC}$ không cùng phương.\\
				Vậy $A$, $B$, $C$ tạo thành một tam giác.
				\item $\vv{BA}=(-2;-2) \Rightarrow BA = 2\sqrt{2}$, $\vv{BC}=(4;0) \Rightarrow BC = 4$.\\
				$\cos \widehat{ABC} = \dfrac{\vv{BA} \cdot \vv{BC}}{BA \cdot BC}
				= -\dfrac{\sqrt{2}}{2} \Rightarrow \widehat{ABC} =135^\circ.$ 
				\end{itemize}
			\item Tìm tọa độ điểm $D$ để tứ giác $ABCD$ là hình bình hành. Gọi $I$ là tâm hình bình hành, tính $\vv{IA}\cdot \vv{IB}$.
				\begin{itemize}
				\item Gọi $D(x;y)$. $ABCD$ là hình bình hành nên $\vv{AD}=\vv{BC}
				\Leftrightarrow \heva{&x-1=4\\&y-3=0} \Leftrightarrow \heva{&x=5\\&y=3.}$\\
				Vậy $D(5;3)$.
				\immini{
				\item $I$ là tâm của hình bình hành nên $I$ là trung điểm $AC$\\ $\Rightarrow \heva{&x_I =\dfrac{x_A +x_C}{2}\\&y_I =\dfrac{y_A +y_C}{2}}
				\Leftrightarrow \heva{&x_I=4\\&y_I=4}
				\Rightarrow I(4;4)$.\\
				$\vv{IA}=(-3;-1)$, $\vv{IB}=(-1;1)$.\\
				Vậy $\vv{IA} \cdot \vv{IB} =(-3) \cdot (-1) + (-1) \cdot 1 =2$.
				}{
				\begin{tikzpicture}[>=stealth, line join=round, line cap=round, scale=.8]
				\tkzDefPoints{0/0/A, 4/0/D, -1/-2/B}
				\tkzDefPointsBy[translation=from A to D](B){C}
				\tkzDrawSegments(A,B A,D B,C C,D A,C B,D)
				\tkzInterLL(A,C)(B,D)\tkzGetPoint{I}
				\tkzDrawPoints[fill=black](A,B,C,D,I)
				\tkzLabelPoints[left](A,B)
				\tkzLabelPoints[right](C,D)
				\tkzLabelPoints[below](I)
				\end{tikzpicture}
				}		
				\end{itemize}
			\item Tìm tọa độ điểm $E$ để tứ giác $ABCE$ là hình thang vuông có cạnh góc vuông là $AE$.
				\begin{itemize}
				\item Gọi $E(x;y)$
				\item $\vv{AB}=(2;2)$, $\vv{EC}=(7-x;5-y)$, $\vv{AE}=(x-1;y-3)$.
				\immini{
				\item $ABCE$ là hình thang vuông có cạnh góc vuông là $AE$ nên $\vv{AB}$, $\vv{EC}$ cùng hướng và $\vv{AB} \perp \vv{AE}$.\\
				Suy ra $\heva{&\dfrac{7-x}{2} = \dfrac{5-y}{2}>0\\&2(x-1)+2(y-3)=0}
				\Leftrightarrow \heva{&x=3\\&y=1}$ (nhận).\\
				Vậy $E(3;1)$.
			}{
				\begin{tikzpicture}[>=stealth, line join=round, line cap=round, scale=.8]
				\tkzDefPoints{0/0/A, 3/0/B, 4/-2.5/C}
				\coordinate (E) at ($(A)+(0,-2.5)$);
				\tkzDrawSegments(A,B B,C C,E E,A)
				\tkzDrawPoints[fill=black](A,B,C,E)
				\tkzLabelPoints[left](A,E)
				\tkzLabelPoints[right](B,C)
				\tkzMarkRightAngles(B,A,E C,E,A)
				\end{tikzpicture}
			}
				\end{itemize}
			\item Trong tam giác $ABC$, đường phân giác xuất phát từ đỉnh $A$ cắt cạnh $BC$ tại $K$, tìm tọa độ điểm $K$.
				\begin{itemize}
				\item $\vv{AB}=(2;2) \Rightarrow AB=2\sqrt{2}$.
				\item $\vv{AC}=(6;2) \Rightarrow AC=2\sqrt{10}$.
				\item Theo tính chất đường phân giác, ta có
				\immini{
				$\dfrac{KC}{KB}=\dfrac{AC}{AB}=\sqrt{5}$
			$\Rightarrow KC=\sqrt{5}KB$\\
			$\Rightarrow \vv{KC}=-\sqrt{5}\vv{KB}$
			$\Leftrightarrow \heva{&7-x=-\sqrt{5}(3-x)\\&5-y=-\sqrt{5}(5-y)}$\\
			$\Leftrightarrow \heva{&x=\dfrac{7+3\sqrt{5}}{1+\sqrt{5}}\\&y=5}$
			$\Leftrightarrow \heva{&x=2+\sqrt{5}\\&y=5.}$\\
			Vậy $K(2+\sqrt{5};5)$.
			}{\begin{tikzpicture}
				\tkzDefPoints{0/0/A, 3/-2/C, -1/-2/B}
				\tkzDefLine[bisector](B,A,C)\tkzGetPoint{d}
				\tkzInterLL(A,d)(B,C)\tkzGetPoint{K}
				\tkzDrawSegments(A,B B,C C,A A,K)
				\tkzDrawPoints[fill=black](A,B,C,K)
				\tkzLabelPoints[above](A)
				\tkzLabelPoints[left](B)
				\tkzLabelPoints[right](C)
				\tkzLabelPoints[below](K)
				\tkzMarkAngles[size=0.5cm](B,A,K)
				\tkzMarkAngles[size=0.7cm](K,A,C)
				\end{tikzpicture}
			}
				\end{itemize}
		\end{enumerate}
	}
\end{bt}
