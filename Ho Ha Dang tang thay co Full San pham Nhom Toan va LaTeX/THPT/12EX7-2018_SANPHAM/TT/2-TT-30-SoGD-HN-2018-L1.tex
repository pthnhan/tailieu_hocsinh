		&2x^2+m=0. \\
		\end{aligned} \right.$

		Gọi $H$, $I$ lần lượt là trung điểm của $AB$, $AC'$. Suy ra $HI\parallel BC'$. Trong mặt phẳng $(ABB'A')$, tia $A'H$ cắt tia $B'B$ tại $S$, gọi $K$ là hình chiếu của $B$ trên $SH$. Dễ thấy $BK\perp (SCH)$. Gọi $M$ là hình chiếu của $K$ trên $A'C$, chú ý rằng $CH=HA'$ nên $HI\perp A'C$, do đó $KM\parallel HI\parallel BC'$. Trong mặt phẳng $(BC'MK)$ lấy điểm $N$ trên $BC'$ sao cho $BKMN$ là hình bình hành. Khi đó $MN$ là đoạn vuông góc chung cần tìm. Ta có
		\begin{align*}
			\dfrac{NB}{BC'}=\dfrac{MK}{2HI}=\dfrac{1}{2} \left( 1+ \dfrac{HK}{A'H} \right)=\dfrac{1}{2}\left( 1+ \dfrac{HK}{HS} \right)=\dfrac{1}{2}\left( 1+ \dfrac{HB^2}{HS^2} \right).
		\end{align*}
		Do $2HB= SB$ nên
		\begin{align*}
			\dfrac{NB}{BC'}= \dfrac{1}{2}\left( 1+ \dfrac{HB^2}{HB^2+SB^2} \right)=\dfrac{1}{2}\left( 1+\dfrac{HB^2}{HB^2+4HB^2} \right)=\dfrac{3}{5}.
		\end{align*}
		Vậy $\dfrac{NB}{NC'}=\dfrac{3}{2}$.
	}
\end{ex}

	Trong không gian $Oxyz$, cho hai điểm $A(1;2;1)$, $B(2;-1;3)$. Tìm điểm $M$ trên mặt phẳng $(Oxy)$ sao cho $MA^2-2MB^2$ lớn nhất.

	{\True $M(3;-4;0)$}
	{$M\left( \dfrac{3}{2}; \dfrac{1}{2};0 \right)$}
	{$M(0;0;5)$}

\begin{ex}%[Đề khảo sát học sinh THPT lớp 12, 2017 - 2018, SGD Hà Nội]%[1D4G3]
	Phương trình $\sqrt{x-512}+ \sqrt{1024-x}=16+ 4\sqrt[8]{(x-512)(1024-x)}$ có bao nhiêu nghiệm?
	\choice
	{$2$ nghiệm}
	{$8$ nghiệm}
	{$4$ nghiệm}
	{\True $3$ nghiệm}
	\loigiai{
		Điều kiện $512\le x\le 1024$. Đặt $u=\sqrt[8]{x-512}\ge 0$, $v=\sqrt[8]{1024-x}\ge 0$. Suy ra ta có hệ phương trình
		\begin{align*}
			\begin{cases}
				u^4+v^4=16+4uv& (1)\\ u^8+v^8=512 &(2)
			\end{cases}.
		\end{align*}
		Từ phương trình $(1)$ ta có $16+4uv=u^4+v^4\ge 2u^2v^2$, suy ra $0\le uv\le 4$. Do hai vế của phương trình $(1)$ không âm nên bình phương hai vế của $(1)$ ta được phương trình tương đương
		\begin{align*}
			u^8+2u^4v^4+v^8=16u^2v^2+128uv+ 256.\tag{3}
		\end{align*}
		Thay $(2)$ vào $(3)$ ta được
		\begin{align*}
			&(uv)^4-8(uv)^2-64uv+128=0\\
			\Leftrightarrow &(uv-4)\left[ (uv)^3+4(uv)^2+8uv-32 \right]=0.
		\end{align*}
		Với $uv=4$ ta suy ra $x=768$. Xét $uv<4$. Chú ý rằng hàm $f(t)=t^3+4t^2+8t-32$, $t\in [0;4]$ có $f'(t)=3t^2+4t+8>0$, $\forall t\in [0;4]$  và $f(0)f(4)<0$ nên tồn tại duy nhất $P\in (0;4)$ để $f(P)=0$. Do đó $uv=P$. Ta có hệ phương trình
		\begin{align*}
			\heva{&u^8+v^8=512\\ &u^8v^8=P^8.}
		\end{align*}
		Ta có $512^2-4P^8=4^9-4P^8>0$, do đó theo định lí Vi\` ete đảo, tồn tại hai bộ số $(u_1;v_1)$ và $(u_2;v_2)$ ($u_1,v_1,u_2,v_2>0$) là nghiệm của hệ. Mỗi bộ số đó sinh ra một nghiệm của phương trình đã cho.\\
		Vậy phương trình ban đầu có $3$ nghiệm.
	}
\end{ex}
