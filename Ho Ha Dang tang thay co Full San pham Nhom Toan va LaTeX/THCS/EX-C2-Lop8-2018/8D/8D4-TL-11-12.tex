\chapter{Bất phương trình bậc nhất một ẩn}
\section{Liên hệ giữa thứ tự và phép cộng, phép nhân}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Bất đẳng thức}
Ta gọi hệ thức dạng $a>b$ (hoặc $a<b$, $a\ge b$, $a\le b$) là một bất đẳng thức.\\
Để chứng minh bất đẳng thức $a>b$, ta thường xét hiệu $a-b$ và chứng minh hiệu đó là số dương.\\
Các cách khác chứng minh bất đẳng thức được nêu trong chuyên đề \textit{Bất đẳng thức}.
\subsubsection{Một số tính chất}
\begin{tc}
Tính bắc cầu: \[a>b;b>c\Rightarrow a>c.\]
\end{tc}
\begin{tc}
Cộng hai vế của bất đẳng thức với cùng một số: \[a>b\Rightarrow a+c>b+c.\]
\end{tc}
\begin{tc}
Nhân hai vế của bất đẳng thức với cùng một số: 
\begin{align*}
&a>b;c>0\Rightarrow ac>bc\\ 
&a>b;c<0\Rightarrow ac<bc.
\end{align*}
\end{tc}
Các tính chất khác được nêu trong chuyên đề \textit{Bất đẳng thức}.
\subsection{Một số ví dụ}
\begin{vd}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K2]
Chứng minh các bất đẳng thức:
\[x^2+y^2\ge \dfrac{\left(x+y\right)^2}{2}\ge 2xy.\]
\loigiai{
\begin{itemize}
\item Xét hiệu
\begin{align*}
x^2+y^2-\dfrac{\left(x+y\right)^2}{2}&=\dfrac{2x^2+2y^2-x^2-2xy-y^2}{2}\\ 
&=\dfrac{x^2-2xy+y^2}{2}=\dfrac{\left(x-y\right)^2}{2}\ge 0.
\end{align*}
Vậy $x^2+y^2\ge \dfrac{\left(x+y\right)^2}{2}$. Xảy ra đẳng thức khi và chỉ khi $x=y$.
\item Xét hiệu:
\[\dfrac{\left(x+y\right)^2}{2}-2xy=\dfrac{x^2+2xy+y^2-4xy}{2}=\dfrac{\left(x-y\right)^2}{2}\ge 0.\]
Vậy $\dfrac{\left(x+y\right)^2}{2}\ge 2xy$. Xảy ra đẳng thức khi và chỉ khi $x=y$.
\end{itemize}
}
\end{vd}

\begin{nx} \quad
	\begin{itemize}
		\item Bất đẳng thức $x^2+y^2\ge  \dfrac{\left(x+y\right)^2}{2}$ cho liên hệ giữa tổng bình phương của hai số $x$, $y$ và bình phương của tổng hai số đó.
		\item Bất đẳng thức $\dfrac{\left(x+y\right)^2}{2}\ge 2xy$ hay $\left(\dfrac{x+y}{2}\right)^2\ge xy$ cho liên hệ  giữa tổng hai số $x$, $y$ và tích của hai số đó. Với $x$, $y$ không âm, bất đẳng thức này được viết dưới dạng $\dfrac{x+y}{2}\ge $ (trung bình công hai số lớn hơn hoặc bằng trung bình nhân của chúng), đó là \textit{bất đẳng thức Cô-si} với hai số không âm.
	\end{itemize}
\end{nx}
\begin{vd}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K2]
Chứng minh các bất đẳng thức:
\begin{enumerate}
\item $x+\dfrac{1}{x}\ge 2$ với $x>0$;
\item $\dfrac{1}{a}+\dfrac{1}{b}\ge \dfrac{4}{a+b}$ với $a>0$, $b>0$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Xét hiệu:
\[x+\dfrac{1}{x}-2=\dfrac{x^2+1-2x}{2}=\dfrac{\left(x-1\right)^2}{2}\ge 0,\;\text{vì $(x-1)^2\ge 0$, $x>0$}.\]
Xảy ra đẳng thức khi và chỉ khi $x=1$.
\item Xét hiệu:
\begin{align*}
\dfrac{1}{a}+\dfrac{1}{b}- \dfrac{4}{a+b}&=\dfrac{b(a+b)+a(a+b)-4ab}{ab(a+b)}\\ 
&=\dfrac{a^2-2ab+b^2}{ab(a+b)}\\
&=\dfrac{\left(a-b\right)^2}{ab(a+b)}\ge 0,\;\text{vì $a,b>0$}.
\end{align*}
Xảy ra đẳng thức khi và chỉ khi $a=b$.
\end{enumerate}
\begin{nx}
\quad
\begin{itemize}
\item Bất đẳng thức $x+\dfrac{1}{x}\ge 2$ (với $x>0$) cho liên hệ giữa một số dương với nghịch đảo của nó.
\item Bất đẳng thức $\dfrac{1}{a}+\dfrac{1}{b}\ge \dfrac{4}{a+b}$ với $a>0$, $b>0$ cho liên hệ giữa tổng các nghịch đảo của hai số dương và nghịch đảo của tổng hai số đó.
\end{itemize}

\end{nx}
}
\end{vd}

\subsubsection{Bài tập tự luyện}
\begin{bt}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K2]
Chứng minh các bất đẳng thức:
\begin{listEX}[3]
\item $4x^2+4x+5>0$;
\item $x^2-x+1>0$;
\item $x^2+ab+b^2\ge 0$.
\end{listEX}
\loigiai{
\begin{enumerate}
\item $4x^2+4x+5=(2x+1)^2+4>0$.
\item $x^2-x+1=\left(x-\dfrac{1}{2}\right)^2+\dfrac{3}{4}>0$.
\item $a^2+ab+b^2=\left(a+\dfrac{b}{2}\right)^2+\dfrac{3b^2}{4}\ge 0$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K2]
Chứng minh bất đẳng thức $\dfrac{x-x^2+1}{x-x^2-1}<1.$
\loigiai{
Xét hiệu
\[\dfrac{x-x^2+1}{x-x^2-1}-1=\dfrac{-2}{x^2-x+1}<0,\]
(vì $x^2-x+1=\left(x-\dfrac{1}{2}\right)^2+\dfrac{3}{4}>0$). Do đó $\dfrac{x-x^2+1}{x-x^2-1}<1.$
}
\end{bt}

\begin{bt}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K2]
Rút gọn rồi chứng minh rằng biểu thức sau không âm với mọi giá trị của $x$:
\[\dfrac{x^4+x^3+x+1}{x^4-x^3+2x^2-x+1}.\]
\loigiai{
Ta có
\[\dfrac{x^4+x^3+x+1}{x^4-x^3+2x^2-x+1}=\dfrac{(x+1)^2(x^2-x+1)}{(x^2+1)(x^2-x+1)}=\dfrac{(x+1)^2}{x^2+1}\ge 0.\]
Đẳng thức xảy ra khi và chỉ khi $x=-1$.
}
\end{bt}

\begin{bt}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K2]
Chứng minh các bất đẳng thức:
\begin{enumerate}
\item $a^3+b^3\ge ab(a+b)$ với $a,b>0$;
\item $a^4+b^4\ge ab\left(a^2+b^2\right)$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Xét hiệu
\[a^3+b^3-ab(a+b)=(a+b)(a-b)^2\ge 0\;\text{(do $a,b>0$)}.\]
Do đó $a^3+b^3\ge ab(a+b)$ với $a,b>0$. Đẳng thức xảy ra khi và chỉ khi $a=b$.
\item Xét hiệu
\[a^4+b^4-ab(a^2+b^2)=a^3(a-b)-b^3(a-b)=(a-b)^2(a^2+ab+b^2)\ge 0.\]
Do đó $a^4+b^4\ge ab\left(a^2+b^2\right)$. Đẳng thức xảy ra khi và chỉ khi $a=b$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K2]
Chứng minh các bất đẳng thức:
\begin{enumerate}
\item $\left(a^2+b^2\right)\left(x^2+y^2\right)\ge \left(ax+by\right)^2$\\
(bất đẳng thức Bu-nhi-a-cốp-xki với hai cặp số $a,b$ và $x,y$).
\item $\left(a^2+b^2+c^2\right)\left(x^2+y^2+z^2\right)\ge \left(ax+by+cz\right)^2$\\
(bất đẳng thức Bu-nhi-a-cốp-xki với hai bộ ba số số $a,b,c$ và $x,y,z$).
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có
\begin{align*}
&\left(a^2+b^2\right)\left(x^2+y^2\right)-\left(ax+by\right)^2\\ 
=\;&a^2x^2+a^2y^2+b^2x^2+b^2y^2-a^2x^2-2abxy-b^2y^2\\
=\;&(ay-bx)^2\ge 0.
\end{align*}
Xảy ra đẳng thức khi và chỉ khi $ay=bx$.
\item Ta có
\begin{align*}
&\left(a^2+b^2+c^2\right)\left(x^2+y^2+z^2\right)\ge \left(ax+by+cz\right)^2\\ 
=\;&(ay-bx)^2+(az-cx)^2+(bz-cy)^2\ge 0.
\end{align*}
Xảy ra đẳng thức khi và chỉ khi $ay=bx$, $az=cx$, $bz=cy$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K2]
Cho $a$ và $b$ cùng dấu. Chứng minh rằng 
\begin{listEX}[2]
\item Nếu $a>b$ thì $\dfrac{1}{a}<\dfrac{1}{b}$;
\item $\dfrac{a}{b}+\dfrac{b}{a}\ge 2$.
\end{listEX}
\loigiai{
\begin{enumerate}
\item Ta có \[\dfrac{1}{a}-\dfrac{1}{b}=\dfrac{b-a}{ab}<0\] (vì $b-a<0$, $ab>0$). Do đó $\dfrac{1}{a}<\dfrac{1}{b}$.
\item Ta có \[\dfrac{a}{b}+\dfrac{b}{a}-2=\dfrac{a^2+b^2-2ab}{ab}=\dfrac{\left(a-b\right)^2}{ab}\ge 0\]
(vì $ab>0$). Do đó $\dfrac{a}{b}+\dfrac{b}{a}\ge 2$ với $ab>0$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K2]
Gọi $\dfrac{2}{\dfrac{1}{a}+\dfrac{1}{b}}$ là trung bình điều hòa của $a$ và $b$. Chứng minh rằng trung bình điều hòa của hai số dương $a$ và $b$ nhỏ hơn hoặc bằng trung bình cộng của hai số ấy.
\loigiai{
Ta có
\[\dfrac{a+b}{2}-\dfrac{2}{\dfrac{1}{a}+\dfrac{1}{b}}=\dfrac{a+b}{2}-\dfrac{2ab}{a+b}=\dfrac{(a+b)^2-4ab}{2(a+b)}=\dfrac{(a-b)^2}{2(a+b)}\ge 0\]
(vì $a,b>0$). Do đó $\dfrac{2}{\dfrac{1}{a}+\dfrac{1}{b}}\le \dfrac{a+b}{2}$. Đẳng thức xảy ra khi và chỉ khi $a=b$.
}
\end{bt}

\begin{bt}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K2]
Chứng minh các bất đẳng thức:
\begin{enumerate}
\item $2\left(a^2+b^2\right)\ge \left(a+b\right)^2$;
\item $3\left(a^2+b^2+c^2\right)\ge \left(a+b+c\right)^2\ge 3(ab+bc+ca)$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có
\[2\left(a^2+b^2\right)-\left(a+b\right)^2=a^2-2ab+b^2=(a-b)^2\ge 0.\]
Do đó $2\left(a^2+b^2\right)\ge \left(a+b\right)^2$. Đẳng thức xảy ra khi và chỉ khi $a=b$.
\item Ta có
\begin{align*}
&3\left(a^2+b^2+c^2\right)-\left(a+b+c\right)^2\\ 
=\;&3\left(a^2+b^2+c^2\right)-\left(a^2+b^2+c^2+2ab+2bc+2ca\right)\\
=\;&(a-b)^2+(b-c)^2+(c-a)^2\ge 0\\
\Rightarrow\;&3\left(a^2+b^2+c^2\right)\ge \left(a+b+c\right)^2.\tag{1}
\end{align*}
Ta có
\begin{align*}
&\left(a+b+c\right)^2-3(ab+bc+ca)\\ 
=\;&\left(a^2+b^2+c^2+2ab+2bc+2ca\right)-3(ab+bc+ca)\\
=\;&(a-b)^2+(b-c)^2+(c-a)^2\ge 0\\
\Rightarrow\;&\left(a+b+c\right)^2\ge 3(ab+bc+ca).\tag{2}
\end{align*}
Từ $(1)$ và $(2)$ ta có $3\left(a^2+b^2+c^2\right)\ge \left(a+b+c\right)^2\ge 3(ab+bc+ca)$. Đẳng thức xảy ra khi và chỉ khi $a=b=c$.
\end{enumerate}
}
\end{bt}


\begin{bt}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K2]
Chứng minh các bất đẳng thức:
\begin{listEX}[2]
\item $\dfrac{a^2+b^2}{2}\ge \left(\dfrac{a+b}{2}\right)^2$;
\item $\dfrac{a^4+b^4}{2}\ge \left(\dfrac{a+b}{2}\right)^4$;
\item $\dfrac{a^2+b^2+c^2}{3}\ge \left(\dfrac{a+b+c}{3}\right)^2$.
\end{listEX}
\loigiai{
\begin{enumerate}
\item Ta có
\[\dfrac{a^2+b^2}{2}-\left(\dfrac{a+b}{2}\right)^2=\dfrac{2(a^2+b^2)-(a+b)^2}{4}=\dfrac{(a-b)^2}{4}\ge 0.\]
Do đó $\dfrac{a^2+b^2}{2}\ge \left(\dfrac{a+b}{2}\right)^2$. Đẳng thức xảy ra khi và chỉ khi $a=b$.
\item Áp dụng câu a) hai lần ta có
\[\dfrac{a^4+b^4}{2}\ge \left(\dfrac{a^2+b^2}{2}\right)^2\ge \left(\dfrac{a+b}{2}\right)^4.\]
Đẳng thức xảy ra khi và chỉ khi $a=b$.
\item Ta có
\begin{align*}
&\dfrac{a^2+b^2+c^2}{3}-\left(\dfrac{a+b+c}{3}\right)^2\\ 
=\;&\dfrac{3(a^2+b^2+c^2)-(a+b+c)^2}{9}\\
=\;&\dfrac{(a-b)^2+(b-c)^2+(c-a)^2}{9}\ge 0.
\end{align*}
Do đó $\dfrac{a^2+b^2+c^2}{3}\ge \left(\dfrac{a+b+c}{3}\right)^2$. Đẳng thức xảy ra khi và chỉ khi $a=b=c$.
\end{enumerate}
}
\end{bt}

\section{Bất phương trình bậc nhất một ẩn}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Khái niệm về bất phương trình bậc nhất một ẩn}
Bất phương trình bậc nhất một ẩn là bất phương trình có dạng $ax+b>0$ (hoặc $ax+b<0$, $ax+b\ge 0$, $ax+b\le 0$), trong đó $x$ là ẩn, $a$ và $b$ là các số đã cho, $a\ne 0$.
\subsubsection{Hai bất phương trình tương đương}
Hai bất phương trình được gọ là tương đương nếu chúng có cùng một tập nghiệm.
\subsubsection{Một số chú ý khi giải một bất phương trình}
Khi giải một bất phương trình, ta có thể:
\begin{itemize}
\item Chuyển một hạng tử từ vế này sang vế kia và đổi dấu hạng tử đó.
\item Nhân (hoặc chia) cả hai vế với cùng một số dương.
\item Nhân (hoặc chia) cả hai vế với cùng một số âm và đổi chiều của bất phương trình.
\end{itemize}
\subsection{Các dạng toán}
\begin{vd}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K4]
Giải bất phương trình với $m$ là hằng:
\[mx+1\ge m^2+x.\]
\loigiai{
Biến đổi tương đương
\[mx+1\ge m^2+x\Leftrightarrow mx-x\ge m^2-1\Leftrightarrow (m-1)x\ge m^2-1.\tag{1}\]
\begin{itemize}
\item Nếu $m>1$ thì nghiệm của bất phương trình là $x\ge m+1$.
\item Nếu $m<1$ thì nghiệm của bất phương trình là $x\le m+1$.
\item Nếu $m=1$ thì $(1)$ có dạng $0x\ge 0$: nghiệm của bất phương trình là mọi $x$.
\end{itemize}
}
\end{vd}

\begin{vd}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K4]
Giải bất phương trình với $a$ là hằng:
\[\dfrac{x+1}{a}+ax>\dfrac{x+2}{a}-2x.\]
\loigiai{
Điều kiện xác định của bất phương trình là $a\ne 0$. Biến đổi bất phương trình:
\begin{align*}
\dfrac{x+1}{a}+ax>\dfrac{x+2}{a}-2x&\Leftrightarrow \dfrac{x}{a}+\dfrac{1}{a}+ax>\dfrac{x}{a}+\dfrac{2}{a}-2x\\ 
&\Leftrightarrow ax+2x>\dfrac{2}{a}-\dfrac{1}{a}\\
&\Leftrightarrow (a+2)x>\dfrac{1}{a}.\tag{1}
\end{align*}
\begin{itemize}
\item Nếu $a>-2$, $a\ne 0$ thì nghiệm của bất phương trình là $x>\dfrac{1}{a(a+2)}$.
\item Nếu $a<-2$ thì nghiệm của bất phương trình là $x<\dfrac{1}{a(a+2)}$.
\item Nếu $a=-2$ thì $(1)$ có dạng $0x>-\dfrac{1}{2}$, nghiệm đúng với mọi $x$.
\end{itemize}
}
\end{vd}

\begin{vd}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4G4]
Kí hiệu $\left[a\right]$ (phần nguyên của $a$) là số nguyên lớn nhất không vượt quá $a$. Tìm $x$ biết rằng \[\left[\dfrac{3x-5}{7}\right]=x.\]
\loigiai{
Theo đề bài, $x$ là số nguyên lớn nhất không vượt quá $\dfrac{3x-5}{7}$. Do đó
\[\left[\dfrac{3x-5}{7}\right]=x\Leftrightarrow \left\{\begin{aligned}&0\le \dfrac{3x-5}{7}-x< 1,&\qquad (1)\\
&x\in\mathbb{Z}.&\qquad (2)\end{aligned}\right.\]
Giải bất phương trình $(1)$:
\begin{align*}
0\le \dfrac{3x-5}{7}-x\le 1&\Leftrightarrow 0\le \dfrac{-4x-5}{7}<1\Leftrightarrow 0\le -4x-5<7\\ 
&\Leftrightarrow 5\le -4x<12\Leftrightarrow -\dfrac{5}{4}\ge x>-3.
\end{align*}
Theo $(2)$, $x\in\mathbb{Z}$, do đó $x=-2$.
}
\end{vd}
\subsubsection{Bài tập tự luyện}
\begin{bt}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K4]
Tìm giá trị của $x$ thỏa mãn cả hai bất phương trình:
\[\dfrac{2x}{5}+\dfrac{3-2x}{3}\ge \dfrac{3x+2}{2}\;\text{và}\;\dfrac{x}{2}+\dfrac{3-2x}{5}\ge \dfrac{3x-5}{6}.\]
\loigiai{
Ta có
\[\dfrac{2x}{5}+\dfrac{3-2x}{3}\ge \dfrac{3x+2}{2}\Leftrightarrow 12x+10(3-2x)\ge 15(3x+2)\Leftrightarrow x\le 0.\tag{1}\]
Mặt khác
\[\dfrac{x}{2}+\dfrac{3-2x}{5}\ge \dfrac{3x-5}{6}\Leftrightarrow 15x+6(3-2x)\ge 5(3x-5)\Leftrightarrow x\le \dfrac{43}{12}.\tag{2}\]
Từ $(1)$ và $(2)$ ta có $x\le 0$.
}
\end{bt}

\begin{bt}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K4]
Tìm số nguyên $x$ thỏa mãn cả hai bất phương trình:
\[\dfrac{3x-2}{5}\ge \dfrac{x}{2}+0{,}8\;\text{và}\;1-\dfrac{2x-5}{6}> \dfrac{3-x}{4}.\]
\loigiai{
Ta có
\[\dfrac{3x-2}{5}\ge \dfrac{x}{2}+0{,}8\Leftrightarrow 2(3x-2)\ge 5x+8\Leftrightarrow x\ge 12.\tag{1}\]
Mặt khác
\[1-\dfrac{2x-5}{6}> \dfrac{3-x}{4}\Leftrightarrow 12-2(2x-5)>3(3-x)\Leftrightarrow x<13.\tag{2}\]
Từ $(1)$ và $(2)$ ta có $-12\le x<13$. Vì $x\in\mathbb{Z}$ nên $x=12$.
}
\end{bt}

\begin{bt}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K4]
Giải các bất phương trình với $a$ là hằng:
\begin{enumerate}
\item $2(x+2)<a(a+x)$;
\item $a(x-a)\le x-1$;
\item $\dfrac{2x}{a^2-a+1}-\dfrac{1}{2a+2}<\dfrac{4x-1}{2a^2-2a+2}+\dfrac{a-2ax}{1+a^3}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có
\[2(x+2)<a(a+x)\Leftrightarrow (a+2)x<a^2-4.\]
		\begin{itemize}
		\item Nếu $a>-2$ thì $x<a-2$.
		\item Nếu $a<-2$ thì $x>a-2$.
		\item Nếu $a=-2$: Vô nghiệm.
		\end{itemize}
\item Ta có
\[a(x-a)\le x-1\Leftrightarrow (a-1)x\le a^2-1.\]
		\begin{itemize}
		\item Nếu $a>1$ thì $x\le a+1$.
		\item Nếu $a<1$ thì $x\ge a+1$.
		\item Nếu $a=1$ thì $0x\le 0$, nghiệm đúng với mọi $x$.
		\end{itemize}
\item Điều kiện: $a\ne -1$. Ta có
\begin{align*}
&\dfrac{2x}{a^2-a+1}-\dfrac{1}{2a+2}<\dfrac{4x-1}{2a^2-2a+2}+\dfrac{a-2ax}{1+a^3}\\ 
\Leftrightarrow\;& \dfrac{a}{2(1+a^3)}\left(4x-a\right)<0\Leftrightarrow \dfrac{a}{a+1}\left(4x-a\right)<0.
\end{align*}
		\begin{itemize}
		\item Nếu $a=0$ thì bất phương trình vô nghiệm.
		\item Nếu $\dfrac{a}{a+1}>0$ (tức là $a<-1$ hoặc $a>0$) thì $x<\dfrac{a}{4}$.
		\item Nếu $\dfrac{a}{a+1}<0$ (tức là $-1<a<0$) thì $x>\dfrac{a}{4}$.
		\end{itemize}
\end{enumerate}
}
\end{bt}

\begin{bt}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4K4]
Tìm giá trị của $m$ để nghiệm của phương trình sau là số dương:

Tìm giá trị của $m$ để nghiệm của phương trình $4mx>x+1$ là

\item $x>9$;
\item $x<-5$.

Có bao nhiêu số tự nhiên $n$ nằm giữa $1$ và $2000$ sao cho phân số $\dfrac{n^2+7}{n+4}$ không phải là phân số tối giản?
\loigiai{
Ta có \[A=\dfrac{n^2+7}{n+4}=\dfrac{(n+4)(n-4)+23}{n+4}=n-4+\dfrac{23}{n+4}.\]
$A$ rút gọn được $\Leftrightarrow 23$ và $n+4$ có ước chung khác $\pm 1\Leftrightarrow n+4\,\vdots\, 23,$ hay nói cách khác $n=23k-4$ với $k\in\mathbb{N}^\ast$.\\
Ta có $1<n<2000\Leftrightarrow 1<23k-4<2000\Leftrightarrow \dfrac{5}{23}<k<87\dfrac{3}{23}$. Do $k\in\mathbb{N}^\ast$ nên $k$ nhận $87$ giá trị (là $1,2,3,\ldots,87$). Vậy có $87$ số tự nhiên $n$ phải tìm.
}
\end{bt}

\begin{bt}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 8]%[8D4G4]
Cho một dãy số tự nhiên bắt đầu từ $1$. Người ta xóa đi một số thì trung bình cộng của các số còn lại bằng $35\dfrac{7}{17}$. Tìm số bị xóa.
\loigiai{
Giả sử ta có $n$ số tự nhiên liên tiếp từ $1$ đến $n$.\\
Nếu xóa số $1$ thì trung bình cộng của các số còn lại là
\[\dfrac{2+3+\cdots+n}{n-1}=\dfrac{\left(2+n\right)\left(n-1\right)}{2\left(n-1\right)}=\dfrac{2+n}{2}.\]
Nếu xóa số $n$ thì trung bình cộng của các số còn lại là
\[\dfrac{1+2+\cdots+(n-1)}{n-1}=\dfrac{n(n-1)}{2\left(n-1\right)}=\dfrac{n}{2}.\]
Ta có \[\dfrac{n}{2}\le 35\dfrac{7}{17}\le \dfrac{n+2}{2}\Leftrightarrow n\le 70\dfrac{14}{17}\le n+2\Leftrightarrow 68\dfrac{14}{17}\le n\le 70\dfrac{14}{17}.\]
Do $n\in\mathbb{N}$ nên $n=69$ hoặc $n=70$.
\begin{itemize}
\item Với $n=70$, tổng của $69$ số còn lại là: $35\dfrac{7}{17}\cdot 69\notin \mathbb{N}$, loại.
\item Với tổng của $69$ số còn lại là: $35\dfrac{7}{17}\cdot 68=2408$.
\end{itemize}
Số bị xóa là số: $(1+2+\cdots+69)-2048=2415-2408=7.$
}
\end{bt}
