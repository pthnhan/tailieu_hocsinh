\begin{ex}%[Đề thi vào 10, Chuyên sư phạm Hà Nội-Vòng 1, 2016]%[9D1B8]
Cho biểu thức:
$$P=\left(\dfrac{\sqrt{1+a}}{\sqrt{1+a}-\sqrt{1-a}}+\dfrac{1-a}{\sqrt{1-a^2}-1+a}\right)\left(\sqrt{\dfrac{1}{a^2}-1}-\dfrac{1}{a}\right)$$ với $0<a<1$. Chứng minh rằng $P=-1$.
\loigiai{Với $0<a<1$ ta có:
\begin{align*}
P& = \left(\dfrac{\sqrt{1+a}}{\sqrt{1+a}-\sqrt{1-a}}+\dfrac{\left(\sqrt{1-a}\right)^2}{\sqrt{(1-a)(1+a)}-\left(\sqrt{1-a}\right)^2}\right)\left(\sqrt{\dfrac{1-a^2}{a^2}}-\dfrac{1}{a}\right)\\
& = \left(\dfrac{\sqrt{1+a}}{\sqrt{1+a}-\sqrt{1-a}}+\dfrac{\sqrt{1-a}}{\sqrt{1+a}-\sqrt{1-a}}\right)\left(\dfrac{\sqrt{1-a}.\sqrt{1+a}}{a}-\dfrac{1}{a}\right)\\
& = \dfrac{\sqrt{1+a}+\sqrt{1-a}}{\sqrt{1+a}-\sqrt{1-a}}.\dfrac{2\sqrt{1-a}.\sqrt{1+a}-(1-a)-(1+a)}{2a}\\
& = \dfrac{\sqrt{1+a}+\sqrt{1-a}}{\sqrt{1+a}-\sqrt{1-a}}.\dfrac{\left[-\left(\sqrt{1+a}-\sqrt{1-a}\right)^2\right]}{2a}\\
& = -\dfrac{\left(\sqrt{1+a}+\sqrt{1-a}\right)\left(\sqrt{1+a}-\sqrt{1-a}\right)}{2a}\\
& = -\dfrac{\left(1+a-(1-a)\right)}{2a}\\
&=-1.
\end{align*}
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên sư phạm Hà Nội-Vòng 1, 2016]%[9D4B3]%[9D4K6]
Cho parabol $(P): y = -x^2$ và đường thẳng $d: y= 2mx -1$ với $m$ là tham số.
    \begin{enumerate}
        \item Tìm tọa độ giao điểm của $d$ và $P$ khi $m=1$.
        \item Chứng minh rằng với mọi $m$, $d$ luôn cắt $(P)$ tại hai điểm phân biệt $A, B$. Gọi $y_1, y_2$ là tung độ của $A$ và $B$. Tìm $m$ sao cho $|y_1^2-y_2^2|=3\sqrt{5}$.
    \end{enumerate}
\loigiai{
    \begin{enumerate}
        \item Khi $m=1$ ta có $d: y=2x-1$ và $(P):y=-x^2$. Phương trình hoành độ giao điểm của $d$ và $(P)$ là:
        $$-x^2=2x-1\Leftrightarrow x^2+2x-1=0\Leftrightarrow \hoac{&x=-1-\sqrt{2}\\&x=-1+\sqrt{2}}$$
        Với $x=-1-\sqrt{2}\Rightarrow y=-3-2\sqrt{2}$;\\
        Với $x=-1+\sqrt{2}\Rightarrow y=-3+2\sqrt{2}.$\\
        Vậy các giao điểm của $d$ và $(P)$ là $\left(-1-\sqrt{2},-3-2\sqrt{2}\right)$; $\left(-1+\sqrt{2},-3+2\sqrt{2}\right).$
        \item Phương trình hoành độ giao điểm của $d$ và $(P)$ là: $-x^2=2mx-1 \Leftrightarrow x^2+2mx-1=0$ \hfill (1)\\
        Ta có $\Delta' =m^2+1>0\  \forall m$ nên $(1)$ luôn có hai nghiệm phân biệt $x_1, x_2$ phân biệt với mọi $m$ hay $d$ luôn cắt $(P)$ tại hai điểm phân biệt $A(x_1;y_1), B(x_2;y_2)$ với $\heva{&y_1=2mx_1-1\\&y_2=2mx_2-1}.$\\
        Từ đó suy ra
        \begin{align*}
        |y_1^2-y_2^2|&=|(y_1-y_2)(y_1+y_2)| \\
         &=|(2mx_1-1-2mx_2+1)(2mx_1-1+2mx_2-1)|\\
         &=|4m(x_1-x_2)\left[m(x_1+x_2)-1\right]|
        \end{align*}
        Theo định lý Vi-ét ta có $\heva{&x_1+x_2=-2m\\&x_1x_2=-1}$.\\
        Ta có $|x_1-x_2|=\sqrt{(x_1-x_2)^2}=\sqrt{(x_1+x_2)^2-4x_1x_2}=\sqrt{4m^2+4}=2\sqrt{m^2+1}$.\\
        Vậy $|y_1^2-y_2^2|=4|m|(2m^2+1).2\sqrt{m^2+1}$, do đó
        $$|y_1^2-y_2^2|=3\sqrt{5}\Leftrightarrow 64m^2(2m^2+1)^2(m^2+1)=45\Leftrightarrow 64(4m^4+4m^2+1)(m^4+m^2)=45$$
        Đặt $t=m^4+m^2, t\geq 0$ ta được phương trình $64t(4t+1)=45\Leftrightarrow 256t^2+64t-45=0.$\hfill (2)\\
        Giải $(2)$, kết hợp với điều kiện $t\geq 0$ ta được $t=\dfrac{5}{16}$\\
        Từ đó suy ra $m^4+m^2=\dfrac{5}{16}\Leftrightarrow 16m^4+16m^2-5=0\Leftrightarrow \hoac{&m^2=\dfrac{1}{4}\\&m^2=-\dfrac{5}{4}}\Leftrightarrow \hoac{&m=\dfrac{1}{2}\\&m=-\dfrac{1}{2}}$.
    \end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên sư phạm Hà Nội-Vòng 1, 2016]%[9D3B5]
Một người đi xe máy từ địa điểm $A$ đến địa điểm $B$ cách nhau $120\mathrm{\, km}$. Vận
tốc trên $\dfrac{3}{4}$ quãng đường $AB$ đầu không đổi, vận tốc trên $\dfrac{1}{4}$ quãng đường $AB$ còn lại bằng $\dfrac{1}{2}$ vận tốc trên $\dfrac{3}{4}$ quãng đường $AB$ đầu. Khi đến $B$ người đó nghỉ lại $30$ phút rồi trở lại $A$ với vận tốc lớn hơn vận tốc trên $\dfrac{3}{4}$ quãng đường $AB$ đầu tiên lúc đi là $10\mathrm{\, km/h}$. Thời gian kể từ lúc xuất phát tại $A$ đến khi xe trở về $A$ là $8,5$ giờ. Tính vận tốc của xe máy trên quãng đường người đó đi từ $B$ về $A$.
\loigiai{Gọi vận tốc của xe máy trên $\dfrac{3}{4}$ quãng đường $AB$ đầu là $x (\mathrm{\, km/h}), x>0.$\\
Vận tốc của xe máy trên $\dfrac{1}{4}$ quãng đường sau là $0,5x (\mathrm{\, km/h}).$\\
Vận tốc của xe máy khi đi quay trở lại từ $B$ về $A$ là $x+10 (\mathrm{\, km/h})$.\\
Tổng thời gian của chuyến đi là $\dfrac{90}{x}+\dfrac{30}{0,5x}+\dfrac{120}{x+10}+\dfrac{1}{2}=8,5$\hfill (1)\\
\begin{align*}
  (1) &\Leftrightarrow \dfrac{90}{x}+\dfrac{60}{x}+\dfrac{120}{x+10}=8\\
  &\Leftrightarrow \dfrac{150}{x}+\dfrac{120}{x+10}=8\\
  &\Leftrightarrow 75(x+10)+60x=4x(x+10) \\
  &\Leftrightarrow 4x^2-95x-750=0\\
  &\Leftrightarrow x=30 \text{ (do } x>0).
\end{align*}
Vậy vận tốc của xe máy trên quãng đường đi từ $B$ về $A$ là $30+10=40 (\mathrm{\, km/h}).$
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên sư phạm Hà Nội-Vòng 1, 2016]%[9H3B7]%[9H3K7]%[9H3K3]
Cho ba điểm $A, M, B$ phân biệt, thẳng hàng và $M$ nằm giữa $A$ và $B$. Trên cùng một nửa mặt phẳng bờ là đường thẳng $AB$ dựng hai tam giác đều $AMC$ và $BMD$. Gọi $P$ là giao điểm của $AD$ và $BC$.
    \begin{enumerate}
        \item Chứng minh $AMPC$ và $BMPD$ là các tứ giác nội tiếp.
        \item Chứng minh $\sqrt{CP.CB}+\sqrt{DP.DA}=AB$.
        \item Đường nối tâm của hai đường tròn ngoại tiếp các tứ giác $AMPC$ và
$BMPD$ cắt $PA$ và $PB$ tại $E$ và $F$. Chứng minh $CDEF$ là hình thang.
    \end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[line cap=round, line join=round]
\tkzDefPoints{0/0/A, 8/0/B}
\coordinate (M) at ($(A)!0.35!(B)$);
\tkzDefPointBy[rotation = center A angle 60](M)
\tkzGetPoint{C}
\tkzDefPointBy[rotation = center M angle 60](B)
\tkzGetPoint{D}
\tkzInterLL(A,D)(C,B)
\tkzGetPoint{P}
\tkzDrawCircle[circum](A,M,C)
\tkzDrawCircle[circum](M,B,D)
\tkzCircumCenter(A,M,C)
\tkzGetPoint{I}
\tkzCircumCenter(M,B,D)
\tkzGetPoint{J}
\tkzInterLL(A,D)(I,J)
\tkzGetPoint{E}
\tkzInterLL(C,B)(I,J)
\tkzGetPoint{F}
\tkzDrawPoints(I,J)
\tkzDrawSegments(A,B A,D B,C C,A C,M D,M D,B M,P E,F)
\tkzLabelPoints[below](M,E,F)
\tkzLabelPoints[left](A)
\tkzLabelPoints[right](B)
\tkzLabelPoints[above](C,D,P)
\end{tikzpicture}
\end{center}
\begin{enumerate}
        \item Vì $\widehat{CMA}=\widehat{DMB}=60^{\circ}$ nên  $\widehat{CMB}=\widehat{DMA}=120^{\circ}.$ Xét $\Delta CMB$ và $\Delta AMD$ ta có\\
            $\heva{&CM=AM\\&\widehat{CMB}=\widehat{DMA}\\&MB=MD}\Rightarrow \Delta CMB=\Delta AMD$\\
            Từ đó suy ra $\heva{&\widehat{MCB}=\widehat{MAD}\\&\widehat{MBC}=\widehat{MDA}}$.\\
            Do đó $AMPC$ và $BMPD$ là các tứ giác nội tiếp.
        \item Vì $AMPC$ là tứ giác nội tiếp nên $\widehat{CPM}=180^{\circ}-\widehat{CAM}=120^{\circ}=\widehat{CMB}.$\\
        Từ đó suy ra $\Delta CPM\backsim \Delta CMB \Rightarrow \dfrac{CP}{CM}=\dfrac{CM}{CB} \Rightarrow CP.CB=CM^2 \Rightarrow CM=\sqrt{CP.CB}$.\\
        Tương tự ta chứng minh được $DM=\sqrt{DP.DA}$.\\
        Vậy $\sqrt{CP.CB}+\sqrt{DP.DA}=CM+DM=AM+BM=AB.$
        \item Ta có $EF$ là đường trung trực của $PM$ suy ra $EP=EM$, do đó $\Delta EPM$ cân tại $E$.\\
        Mặt khác $\widehat{EPM}=\widehat{ACM}=60^{\circ}$ (do $AMPC$ là tứ giác nội tiếp). Từ đó suy $\Delta EPM$ đều, do đó $PE=PM$. Tương tự chứng minh được $PF=PM$.\\
        Ta có $CM\parallel DB$ nên $\widehat{PCM}=\widehat{PBD}$.\\
        Mà $BMPD$ là tứ giác nội tiếp nên $\widehat{PBD}=\widehat{PMD} \Rightarrow \widehat{PCM}=\widehat{PMD}.$\\
        Ta lại có $\widehat{CPM}=\widehat{DPM}=120^{\circ} \Rightarrow \Delta CPM\backsim \Delta MPD \Rightarrow \dfrac{CP}{MP}=\dfrac{PM}{PD} \Rightarrow \dfrac{CP}{PF}=\dfrac{PE}{PD}$.\\
        Theo định lý Ta-lét đảo ta có $CE\parallel DF\Rightarrow CDEF$ là hình thang.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên sư phạm Hà Nội-Vòng 1, 2016]%[9D3G8]
Cho $a, b, c$ là ba số thực không âm và thỏa mãn $a+b+c=1$. Chứng minh rằng:
$$\sqrt{5a+4}+\sqrt{5b+4}+\sqrt{5c+4}\geq 7.$$
\loigiai{Ta có $\heva{&a,b,c\geq 0\\&a+b+c=1}\Rightarrow \heva{&a(1-a)\geq 0\\&b(1-b)\geq 0\\&c(1-c)\geq 0}\Rightarrow \heva{&a\geq a^2\\&b\geq b^2\\&c\geq c^2}.$\\
Suy ra $\sqrt{5a+4}\geq \sqrt{a^2+4a+4}=\sqrt{(a+2)^2}\Rightarrow \sqrt{5a+4}\geq a+2.$\hfill (1)\\
Tương tự ta có $\heva{&\sqrt{5b+4}\geq b+2\\&\sqrt{5c+4}\geq c+2}.$\hfill (2)\\
Từ (1) và (2) ta suy ra \begin{align*}
                          \sqrt{5a+4}+\sqrt{5b+4}+\sqrt{5c+4} & \geq a+2+b+2+c+2 \\
                          & \geq a+b+c+6\\
                          & \geq 7.
                        \end{align*}
}
\end{ex}
