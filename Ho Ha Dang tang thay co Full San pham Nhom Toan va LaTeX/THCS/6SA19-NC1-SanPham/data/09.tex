
\section{So sánh phân số}

\subsection{KIẾN THỨC CƠ BẢN}
\begin{enumerate}
	\item Muốn qui đồng mẫu nhiều phân số có mẫu dương, ta làm như sau:\\
	Tìm một BC của các mẫu (thường là BCNN) để làm mẫu chung.\\
	Tìm thừa số phụ của mỗi mẫu (bằng cách chia mẫu chung cho từng mẫu).\\
	Nhân tử và mẫu của mỗi phân số với thừa số phụ tương ứng.
	\item Trong hai phân số có cùng một mẫu dương, phân số nào có tử lớn hơn thì phân số đó lớn hơn.
	\item Muốn so sánh hai phân số không cùng mẫu, ta viết chúng dưới dạng hai phân số có cùng một mẫu dương rồi so sánh các tử với nhau: phân số nào có tử lớn hơn thì phân số đó lớn hơn.
\end{enumerate}

\subsection{KIẾN THỨC NÂNG CAO}
\begin{enumerate}
	\item Trong hai phân số có cùng một tử, tử và mẫu đều dương, phân số nào có mẫu nhỏ hơn thì phân số đó lớn hơn.
	\item Nếu cộng cả tử và mẫu cuả một phân số nhỏ hơn 1, tử và mẫu đều dương, với cùng một số nguyên dương thì giá trị của phân số đó tăng lên\\
	Xem chứng minh ở ví dụ 9.2. 
	\item Với kiến thức về trừ phân số, ta có khái niệm phân số bù đến đơn vị của một phân số: Cho phân số $\dfrac{a}{b}$ nhỏ hơn 1, ta gọi \textit{phần bù đến đơn vị} của phân số $\dfrac{a}{b}$ là hiệu $1-\dfrac{a}{b}$, tức là $\dfrac{b-a}{b}$.\\
	Trong hai phân số có phần bù đến đơn vị khác nhau, phân số nào có phần bù nhỏ hơn thì phân số đó lớn hơn.
\end{enumerate}
 	
 \subsection{VÍ DỤ MINH HỌA}
	
	\begin{vd}%[9.1]
		\begin{enumEX}{1}
			\item Trong hai phân số có cùng một tử, tử và mẫu đều dương, phân số nào có mẫu nhỏ hơn thì phân số đó lớn hơn..
			\item Áp dụng tính chất để so sánh các phân số: \\
			 $ \dfrac{10}{11}, \dfrac{12}{13} $ và $\dfrac{15}{16}$ ; \hspace{2cm} $\dfrac{n+1}{n+5}$ và $\dfrac{n+2}{n+3} \left( n \in \mathbb{N}\right) $
		\end{enumEX}
		\loigiai{\fontsize{12pt}{28pt}\selectfont
			\begin{enumerate}
				\item So sánh hai phân số $\dfrac{a}{b}$ và $\dfrac{a}{c}$ với $a, b, c \in \mathbb{N}^{*}$ và $b<c$.\\
			Ta có $\dfrac{a}{b} = \dfrac{ac}{bc}$, $\dfrac{a}{c} = \dfrac{ab}{bc}$.\\
			Do $b<c$ và $a>0$ nên $ab<ac$ \\
			Suy ra $\dfrac{ac}{bc} > \dfrac{ab}{bc}$ tức $\dfrac{a}{b} > \dfrac{a}{c}$
				\item Quy đồng tử:
				\fontsize{12pt}{32pt}\selectfont
				\[ \dfrac{10}{11} = \dfrac{60}{66}; \dfrac{12}{13} = \dfrac{60}{65}; \dfrac{15}{16} = \dfrac{60}{64} \]
				Ta có $\dfrac{60}{66}<\dfrac{60}{65}<\dfrac{60}{64}$ nên $\dfrac{10}{11}<\dfrac{12}{13}<\dfrac{15}{16}.$\\
			* Để so sánh $\dfrac{n+1}{n+5}$ và $\dfrac{n+2}{n+3} $, ta dùng phân số trung gian  $\dfrac{n+1}{n+3}$. Ta có: \\
			\hspace{2cm} $\dfrac{n+1}{n+5} < \dfrac{n+1}{n+3} $ ( So sánh hai phân số cùng tử ) \\
			\hspace{2cm} $\dfrac{n+1}{n+3} <\dfrac{n+2}{n+3} $ ( So sánh hai phân số cùng tử ) \\
			Vậy $\dfrac{n+1}{n+5} <\dfrac{n+2}{n+3}. $
			\end{enumerate}
		}
	\end{vd}

	\begin{vd}%[9.2]
	Chứng minh rằng nếu cộng cả tử và mẫu của một phân số nhỏ hơn 1, tử và mẫu đều dương, với cùng một số nguyên dương thì giá trị của phân số đó tăng thêm.
	\loigiai{
		Gọi phân số đã cho là $\dfrac{a}{b}$ với $a, b \in \mathbb{N}^{*}$ và $a<b$ \\
		Cộng số nguyên dương n vào tử và mẫu, ta được $\dfrac{a+n}{b+n}.$ \\
		Cần chứng minh $\dfrac{a}{b}<\dfrac{a+n}{b+n}.$\\
		Qui đồng mẫu của hai phân số trên: \\
		\[ \dfrac{a}{b} = \dfrac{a\left(b+n \right) }{b\left(b+n \right)} = \dfrac{ab+an}{b \left(b+n \right) } \hspace{6cm} (1) \]
		\[\dfrac{a+n}{b+n} = \dfrac{\left(a+n \right)b }{\left(b+n \right)b} = \dfrac{ab+bn}{b \left(b+n \right) } \hspace{5.6cm} (2) \]
		\fontsize{12pt}{28pt}\selectfont
		Do $a<b$ và $n>0$ nên $an<bn.\hspace{8cm} (3)$ \\
		Từ (1), (2) và (3) suy ra $\dfrac{a}{b}<\dfrac{a+n}{b+n}.$
	}
\end{vd}
	
	\begin{vd}%[9.3]
	So sánh hai phân số $A= \dfrac{13579}{34567}$ và $B= \dfrac{13580}{34569}$
	\loigiai{
		Đặt $13579 = a$ và $34567 = b$ thì $A = \dfrac{a}{b}$ và $B = \dfrac{a+1}{b+2}$
		
		\[A= \dfrac{a}{b} = \dfrac{a\left(b+2 \right) }{b\left(b+2 \right)} = \dfrac{ab+2a}{b \left(b+2 \right) } \hspace{5.5cm} (1) \]
		\[B= \dfrac{a+1}{b+2} = \dfrac{\left(a+1 \right)b }{\left(b+2 \right)b} = \dfrac{ab+b}{b \left(b+2 \right) } \hspace{5cm} (2)\]
		Ta thấy $13579.2 < 34567$ tức là $2a<b$ . \hspace{6.5cm} (3)\\
		Từ (1), (2) và (3) suy ra $A < B.$
	}
\end{vd}

	\begin{vd}%[9.4]
	So sánh hai phân số $A= \dfrac{10^{8} +1}{10^{9} +1}$ và $B= \dfrac{10^{9} +1}{10^{10} +1}$
	\loigiai{
\textit{Cách 1.} B là phân số nhỏ hơn 1. Nếu cộng cùng một số nguyên dương vào tử và mẫu của B thì giá trị của B tăng thêm. Do đó, \\
		
		\[B= \dfrac{10^{9}+1}{10^{10}+1} < \dfrac{10^{9}+1+9}{10^{10}+1+9} = \dfrac{10^{9}+10}{10^{10}+10} 
		=\dfrac{10\left( 10^{8}+1\right) }{10\left( 10^{9}+1\right)} =\dfrac{10^{8}+1}{10^{9}+1} =A \]
		Vậy $ B < A.$
		
\textit{Cách 2.} (sau khi học phép nhân phân số) \\
\[10A= \dfrac{10\left( 10^{8}+1\right) }{10^{9}+1} = \dfrac{10^{9}+10}{10^{9}+1} =1+\dfrac{9}{10^{9}+1}  \]
\[10B= \dfrac{10\left( 10^{9}+1\right) }{10^{10}+1} = \dfrac{10^{10}+10}{10^{10}+1} =1+\dfrac{9}{10^{10}+1}  \]
Ta thấy $\dfrac{9}{10^{9}+1} > \dfrac{9}{10^{10}+1}$ (so sánh hai phân số cùng tử) nên $10A >10B.$\\
 Do đó $A>B$.
	}
\end{vd}

\subsection{Bài tập}
\begin{bt}%[9.1] 
Xếp các phân số sau theo thứ tự từ nhỏ đến lớn : $\dfrac{10}{19}, \dfrac{12}{23},\dfrac{15}{17},\dfrac{20}{29},\dfrac{60}{71}.$
	\loigiai{
		\fontsize{12pt}{30pt}\selectfont
		Qui đồng tử ta được $\dfrac{60}{114}, \dfrac{60}{115},\dfrac{60}{68},\dfrac{60}{87},\dfrac{60}{71}.$ \\
		Ta thấy $\dfrac{60}{115}<\dfrac{60}{114}<\dfrac{60}{87}<\dfrac{60}{71}<\dfrac{60}{68}$ nên \\ $\dfrac{12}{23}<\dfrac{10}{19}<\dfrac{20}{29}<\dfrac{60}{71}<\dfrac{15}{17}.$
	}
\end{bt}
\begin{bt}%[9.2]
	Tìm phân số tối giản $\dfrac{a}{b}$ nhỏ hơn $1$ có tích $ab=80$
	\loigiai{
		Phân tích ra thừa số nguyên tố $80 = 2^4.5$ nên		 $a.b = 80 = 1.80 = 2.40 = 4.20 = 8.10 = 16.5$ \\
		Để phân số $\dfrac{a}{b}$ là phân số tối giản và nhỏ hơn $1$ thì $a<b$ và $UCLN\left(a,b \right) = 1 $. Khi đó\\
		$\dfrac{a}{b} \in \left\lbrace  \dfrac{1}{80};\dfrac{5}{16}\right\rbrace $.
	}
\end{bt}

\begin{bt}%[9.3]
	Tìm số tự nhiên x sao cho
	\begin{enumEX}{2}
		\item $\dfrac{1}{5}< \dfrac{x}{30}<\dfrac{1}{4}$; 
		\item $\dfrac{5}{8}< \dfrac{4}{x}<\dfrac{5}{7}$.
	\end{enumEX}
	\loigiai{
		\fontsize{12pt}{28pt}\selectfont
	\begin{enumerate}
		\item $\dfrac{1}{5}< \dfrac{x}{30}<\dfrac{1}{4}$ nên $\dfrac{12}{60}< \dfrac{2x}{60}<\dfrac{15}{60}$. \\
		Suy ra $12<2x<15$, do đó $2x=14$. Vậy $x=7.$
		\item $\dfrac{5}{8}< \dfrac{4}{x}<\dfrac{5}{7}$ nên $\dfrac{20}{32}< \dfrac{40}{5x}<\dfrac{20}{28}$.\\
		Suy ra $28<5x<32$, do đó $5x=30$. Vậy $x=6.$
	\end{enumerate}
	}
\end{bt}

\begin{bt}%[9.4]
	So sánh các phân số sau mà không qui đồng mẫu hoặc tử
	\begin{enumEX}{2}
		\item $\dfrac{7}{15}$ và $ \dfrac{20}{39}$; 
		\item $\dfrac{14}{41}$ và $ \dfrac{17}{54}$.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item $\dfrac{7}{15} = \dfrac{14}{30}<\dfrac{15}{30} = \dfrac{1}{2} = \dfrac{20}{40}<\dfrac{20}{39}$ nên $\dfrac{7}{15} < \dfrac{20}{39}$. \\
			
			\item $\dfrac{14}{41}>\dfrac{14}{42}=\dfrac{1}{3}=\dfrac{18}{54}>\dfrac{17}{54} $
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[9.5]
	So sánh các phân số sau ($n$ là số tự nhiên)
	\begin{enumEX}{2}
		\item $\dfrac{n}{2n+3}$ và $ \dfrac{n+2}{2n+1}$; 
		\item $\dfrac{n}{3n+1}$ và $ \dfrac{2n}{6n+1}$.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item $\dfrac{n}{2n+3}<\dfrac{n+2}{2n+3}<\dfrac{n+2}{2n+1}$. \\
			
			\item $\dfrac{n}{3n+1}=\dfrac{2n}{6n+2}<\dfrac{2n}{6n+1} $
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[9.6]
	So sánh các phân số sau 
	\begin{enumEX}{2}
		\item $A=\dfrac{35420}{35423}$ và $B= \dfrac{25343}{25345}$; 
		\item $C=\dfrac{5^{12}+1}{5^{13}+1}$ và $ D=\dfrac{5^{11}+1}{5^{12}+1}$.
	\end{enumEX}
	\loigiai{
		
		\begin{enumerate}
		\item Đặt $35420 = a$ và $25343 = b$ thì $A = \dfrac{a}{a+3}$ và $B = \dfrac{b}{b+2}$
			\[A= \dfrac{a}{a+3} = \dfrac{a\left(b+2 \right) }{\left(a+3 \right) \left(b+2 \right)} = \dfrac{ab+2a}{\left(a+3 \right)  \left(b+2 \right) } \]
			\[B= \dfrac{b}{b+2} = \dfrac{\left(a+3 \right)b }{\left(b+2 \right)\left(a+3 \right)} = \dfrac{ab+3b}{\left(a+3 \right) \left(b+2 \right) } \]
			Hãy so sánh $2a$ và $3b$.\\
			\textit{Đáp số} $A<B$.
		\item Giải tương tự Ví dụ 9.4.\\
			\textit{Đáp số} $C<D$ 
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[9.7]
	Cho các số tự nhiên $x, y$ thỏa mãn $1\leq y<x\leq 30$. 
	\begin{enumEX}{1}
		\item Tìm giá trị lớn nhất của phân số $\dfrac{x+y}{x-y}$.
		\item Tìm giá trị lớn nhất của phân số $\dfrac{xy}{x-y}$.
	\end{enumEX}
	\loigiai{
		
		\begin{enumerate}
			\item Phân số $\dfrac{x+y}{x-y}$ lớn nhất khi tử lớn nhất (bằng $30+29=59$) và mẫu nhỏ nhất (bằng $1$). Chọn $x=30, y=29$, khi đó $\dfrac{x+y}{x-y} = 59$
			
			\item Phân số $\dfrac{x.y}{x-y}$ lớn nhất khi tử lớn nhất (bằng $30.29=870$) và mẫu nhỏ nhất (bằng $1$). Chọn $x=30, y=29$, khi đó $\dfrac{x+y}{x-y} = 870$
			
		\end{enumerate}
	}
\end{bt}
