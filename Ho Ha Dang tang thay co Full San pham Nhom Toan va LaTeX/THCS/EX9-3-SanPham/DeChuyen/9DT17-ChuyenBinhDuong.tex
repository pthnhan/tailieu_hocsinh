	\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Bình Dương, 2017]%[9D5G1]%[9D5G5]
		\textbf{($3$ điểm )}
		\begin{enumerate}[a/] 
			\item Giải phương trình: $x^{2}-3x+1=-\dfrac{\sqrt{3}}{3}\sqrt{x^{4}+x^{2}+1}$ 
			\item Giải hệ phương trình:$\begin{cases}
			x^{2}-\rvert x \rvert= \rvert yz\rvert\\
			y^{2}-\rvert y \rvert= \rvert xz\rvert\\
			z^{2}-\rvert z \rvert= \rvert yx\rvert
			\end{cases}$ 
		\end{enumerate}
	\loigiai{\begin{enumerate}[a/]
			\item Ta có: $x^{2}-3x+1=\dfrac{-\sqrt{3}}{3}\sqrt{x^{4}+x^{2}+1} \Leftrightarrow  (x^{2}-x+1)-2x=\dfrac{-1}{\sqrt{3}}\sqrt{(x^{2}-x+1)(x^{2}+x+1)}$\\
			Đặt $a=x^{2}-x+1,b=x^{2}+x+1 \Leftrightarrow 2a-b=\dfrac{-1}{\sqrt{3}}\sqrt{ab}\Rightarrow(2a-b)^{2}=\dfrac{ab}{3}\Rightarrow12a^{2}-13ab+3b^{2}=0\Rightarrow(4a-3b)(3a-b)=0\Rightarrow \left[\begin{array}{l}
			4a=3b\\3a=b
			\end{array}\right.\Rightarrow\left[\begin{array}{l}
			x^{2}-7x+1=0\\x^{2}-2x+1=0
			\end{array}\right.\Rightarrow\left[\begin{array}{l}
			x=\dfrac{7+3\sqrt{5}}{2}\\x=\dfrac{7-3\sqrt{5}}{2}\\x=1
			\end{array}\right.$
			\item $\begin{cases}
			x^{2}-\rvert x \rvert= \rvert yz\rvert\\
			y^{2}-\rvert y \rvert= \rvert xz\rvert\\
			z^{2}-\rvert z \rvert= \rvert yx\rvert
			\end{cases}\Leftrightarrow \begin{cases}
			\rvert x\rvert(\rvert x\rvert-1)=\rvert yz\rvert \\\rvert y \rvert(\rvert y\rvert-1)=\rvert xz\rvert\\\rvert z\rvert(\rvert z\rvert-1 )=\rvert yx\rvert    
			\end{cases}\Rightarrow \rvert xyz \rvert  =(\rvert  x \rvert  -1)(\rvert  y \rvert  -1)(\rvert  z \rvert -1 )  (xyz\neq 0)$(vô lý).\\Lại có $(x,y,z)=(0,0,0)$ là một nghiệm của hệ. Với mọi $(x,y,z)=(0,a,b)$ hoặc $(x,y,z)=(0,0,b)$ đều không thoả mãn hệ; vai trò của x,y,z như nhau. \\
			Kết luận: hệ có nghiệm duy nhất $(x,y,z)=(0,0,0)$.
		\end{enumerate}
	}
	\end{ex} 

   \begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Bình Dương, 2017]%[9D5G2]
	\textbf{($1,5$ điểm )}
	\\ Với $x,y$ là các số dương thoả mãn diều kiện $x\geq 2y$, tìm giá trị nhỏ nhất của biểu thức $M=\dfrac{x^{2}+y^{2}}{xy}$ 
	\loigiai{$M=\dfrac{x^{2}+4y^{2}-3y^{2}}{xy}\geq \dfrac{4xy}{xy}-\dfrac{3y}{x}\neq 4-\dfrac{3}{2}=\dfrac{5}{2}$\\
		Kết luận: giá trị nhỏ nhất của $M=\dfrac{5}{2}$   
	} 
   \end{ex} 

   \begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Bình Dương, 2017]%[9D5G2]5[9D5G3]%[9D5G1]
	\textbf{($2$ điểm )}
	\begin{enumerate}[a/]
		\item Cho $a,b,c,d$ là các số thực thoả mãn: $b+d\neq 0; \dfrac{ac}{b+d}\geq 2$. Chứng minh phương trình $(x^{2}+ax+b)(x^{2}+cx+d)=0$ ẩn $x$ luôn có nghiệm.  
		\item Tìm các cặp số nguyên $(x,y)$  thoả mãn: $x^{2}-y^{2}=xy+8$ .  
	\end{enumerate}\loigiai{\begin{enumerate}
		\item Giả sử phương trình đã cho vô nghiệm $\Rightarrow\begin{cases}
		a^{2}-4b< 0\\c^{2}-4d< 0		
		\end{cases}\Rightarrow \dfrac{a^{2}+c^{2}}{b+d}<4$ mà $\dfrac{a^{2}+c^{2}}{b+d}\geq 4$. Suy ra phương trình đã cho luôn có nghiệm.
		\item $x^{2}-y^{2}=xy+8\Rightarrow x^{2}-xy-(y^{2}+6)=0\Rightarrow \Delta = y^{2}+4(y^{2}+8)=5y^{2}+32$. Để $(x,y) $ là cặp số ngyên suy ra $\Delta=a^{2}$. $a^{2}$ là số chính phương nên chia cho $5$ dư $0,1,4$  mà theo $\Delta, a^{2}$ chia $5$ dư $2$. Vậy không có cặp số nào thoả mãn.          
	\end{enumerate}
     }
   \end{ex}

		\item Tính số đo góc $\widehat{BIF}$.
		\item Gọi $H$ là giao điểm của $BM$ và $EF$. Chứng minh rằng nếu $AM=AB $ thì tứ giác $ABHI$ là tứ giác nội tiếp.
