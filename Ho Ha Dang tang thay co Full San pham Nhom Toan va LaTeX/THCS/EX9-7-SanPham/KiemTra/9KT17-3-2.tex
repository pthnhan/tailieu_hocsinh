\begin{ex}%[9D1B8]
	Cho biểu thức: $A=\left( \dfrac{x^3-1}{x-1} +x\right) \left( \dfrac{x^3+1}{x+1} -x\right) : \dfrac{x{\left( 1-x^2\right) }^2}{x^2-2}$
	\begin{enumerate}    
		\item Rút gọn biểu thức $A$
		\item Tính giá trị của biểu thức khi cho $x=\sqrt{6+2\sqrt{2}}$
		\item Tìm giá trị của $x$ để $A=-1$.
	\end{enumerate}
	\loigiai{Điều kiện xác định: $x \neq \pm 1, x \neq \pm\sqrt{2} $
		\begin{enumerate}    
			
			\item $A=\left( \dfrac{\left( x-1\right) \left( x^2+x+1\right) }{x-1} +x \right) \left( \dfrac{\left( x+1\right) \left( x^2-x+1\right) }{x+1} -x\right) \cdot \dfrac{x^2-2}{x{\left( 1-x^2\right) }^2} \\
			=\left( x^2+2x+1\right) \left( x^2-2x+1\right) \cdot \dfrac{x^2-2}{x{\left[ \left( 1+x\right) \left( 1-x\right) \right] }^2}\\
			={\left( x+1\right) }^2{\left( x-1 \right) }^2 \cdot \dfrac{x^2-2}{x{\left( x+1\right) }^2{\left( x-1\right) }^2}\\
			=\dfrac{x^2-2}{x}.$
			\item Với $x=\sqrt{3+2\sqrt{2}}=\sqrt{{\left( \sqrt{2} +1\right) }^2} = \sqrt{2} +1$ ta có:\\
			$A=\dfrac{{\left(  \sqrt{2} +1\right) }^2-2}{ \sqrt{2} +1}=\dfrac{1+2\sqrt{2}}{ \sqrt{2} +1}\\
			=\dfrac{\left( 1+2\sqrt{2} \right) \left( \sqrt{2} -1 \right) }{ \left( \sqrt{2} +1\right) \left( \sqrt{2} -1\right) }\\
			=\sqrt{2} -1+4-2\sqrt{2} =3-\sqrt{2}.$
			\item $A=\dfrac{x^2-2}{x} =-1 \Rightarrow x^2+x-2=0 \\
			\Rightarrow x^2+2x-x-2=0 \\
			\Rightarrow x\left( x+2\right) -\left( x+2\right) =0\\
			\Rightarrow \left( x-1\right) \left( x+2\right) =0\\
			\Rightarrow \hoac{&x=1\\&x=-2}$\\
			Đối chiếu điều kiện suy ra giá trị cần tìm là $x=-2$
			
	\end{enumerate}}
\end{ex}

	\loigiai{Gọi quãng đường BC là $x$ với $x>0$, AB là $15+x$.\\
		 Ta có $4$ giờ $15$ phút tương ứng với $\dfrac{17}{4}$ giờ.\\
		 Tổng thời gian đi quãng đường  AB và BC chính là thời gian đi từ A đến C nên ta có phương trình sau:\\
		$t_{AB}+t_{BC}=4\dfrac{1}{4}\\
		\Rightarrow \dfrac{x+15}{40}+\dfrac{x}{30}=\dfrac{17}{4}\\
		\Leftrightarrow \dfrac{17}{120}x=\dfrac{31}{8}\\
		\Leftrightarrow x=\dfrac{465}{7}$\\
		Vậy $BC=\dfrac{465}{7}$ km, $AB=\dfrac{570}{7}$ km
	}

\begin{ex}%[9H3K4]%[9H3K7]
	Từ điểm $A$ ở bên ngoài đường tròn $(O)$ kẻ hai tiếp tuyến $AB$ và $AC$ ($B, C$ là các tiếp điểm). $M$ là điểm bất kì trên cung nhỏ $BC$.  Kẻ $MI \perp AB$,  $MH \perp BC$,  $MK \perp AC$   ($I , H , K$ là chân các đường vuông góc).
	\begin{enumerate}
	
	 	\item Chứng minh tứ giác $BIMH$ nội tiếp.
		\item Chứng minh $MH^2 = MI.MK$.
		\item Gọi $P$ là giao điểm của $IH$ và $MB$, $Q$ là giao điểm của $KH$ và  $MC$. Chứng minh tứ giác $MPHQ$ nội tiếp.
	 
\end{enumerate}
	\loigiai{
		
		\begin{center}
			{\begin{tikzpicture}[scale=0.7]
				\tkzDefPoint[label=180:O](0,0){O}
				\tkzDefPoint[label=60:B](2,4){B}
				\tkzDefCircle[radius](O,B)
				\tkzDrawCircle[radius](O,B)
				\tkzDefPoint[label=-60:C](2,-4){C}
				\tkzDrawPolygon(C,B,O)
				\tkzTangent[at=B](O)
				\tkzGetPoint{b}
				\tkzDrawLines[add=12 and 1](B,b)
				\tkzTangent[at=C](O)
				\tkzGetPoint{c}
				\tkzDrawLines[add=1 and 12](C,c)
				\tkzInterLL(B,b)(C,c){A}
				\tkzGetPoint{A}
				\tkzLabelPoints[below](A)
				\tkzDefPoint[label=0:M](4,2){M}
				\tkzDrawAltitude(A,B)(M)
				\tkzGetPoint{I}
				\tkzMarkRightAngle[size=.3](A,I,M)
				\tkzLabelPoints[above](I)
				\tkzDrawAltitude(C,B)(M)
				\tkzGetPoint{H}
				\tkzMarkRightAngle[size=.3](B,H,M)
				\tkzLabelPoints[left](H)
				\tkzDrawAltitude(A,C)(M)
				\tkzGetPoint{K}
				\tkzLabelPoints[below](K)
				\tkzDrawSegment(M,C)
				\tkzDrawSegment(H,K)
				\tkzDrawSegment(M,B)
				\tkzDrawSegment(H,I)
				\tkzMarkRightAngle[size=.3](M,K,A)
				\tkzInterLL(I,H)(M,B){P}
				\tkzGetPoint{P}
				\tkzLabelPoints[left](P)
				\tkzInterLL(K,H)(M,C){Q}
				\tkzGetPoint{Q}
				\tkzLabelPoints[left](Q)

				\end{tikzpicture}
				
		}\end{center}
	\begin{enumerate}
		\item Do $MI \perp AB$,  $MH \perp BC$ nên $\widehat{BMH}=\widehat{BMI}=90^{\circ}$ suy ra $\widehat{BMH}+\widehat{BMI}=180^{\circ}$.
		\\ Vậy tứ giác $BIMH$ nội tiếp.
		\item Ta sẽ chứng minh $ \triangle MHK \sim \triangle MIH $.
		\\ Ta có tứ giác $MHCK$ nội tiếp do $\widehat{MHC}+\widehat{MKC}=180^{\circ}$ ($\widehat{MHC}=\widehat{MKC}=90^{\circ}$).
		\\ Suy ra $\widehat{MKH}=\widehat{MCH}, \widehat{MHK}=\widehat{MCK}\ (1)$\\
		Do tứ giác $BIMH$ nội tiếp nên $\widehat{MHI}=\widehat{MBI}, \widehat{MIH}=\widehat{MBH}\ (2)$\\
		Lại có $AC,AB$ là các tiếp tuyến của $(O)$ nên $\widehat{MCH}=\widehat{MBI}, \widehat{MCK}=\widehat{MBH}\ (3)$\\
		Từ $(1), (2), (3)$ suy ra $\widehat{MKH}=\widehat{MHI}, \widehat{MHK}=\widehat{MIH}.$\\
		Do đó $ \triangle MHK \sim \triangle MIH $ suy ra $\dfrac{MH}{MI}=\dfrac{MK}{MH}\Leftrightarrow MH^2 = MI.MK$
		\item Ta sẽ chứng minh $\widehat{PHQ}+\widehat{PMQ}=180^{\circ}$. 
		\\ Ta có hai tam giác vuông $\triangle MHC$ và $\triangle MIB$ đồng dạng (do $\widehat{MCH}=\widehat{MBI}$) nên $\widehat{HMC}=\widehat{IMB}$.
		\\ Mà $\widehat{IMB}=\widehat{IMH}$ nên $\widehat{HMC}=\widehat{IHB}$.
		\\ Lại có hai tam giác vuông $\triangle MKC$ và $\triangle MHB$ đồng dạng (do $\widehat{MCK}=\widehat{MBH}$) nên $\widehat{KMC}=\widehat{HMB}$.
		\\ Mà $\widehat{KHC}=\widehat{KMC}$ (tứ giác $MHCK$ nội tiếp) nên $\widehat{KHC}=\widehat{HMB}$.
		\\ Mặt khác $$\widehat{PHQ}+\widehat{PMQ}=\widehat{PHQ}+\widehat{HMB}+\widehat{HMC}=\widehat{PHQ}+\widehat{KHC}+\widehat{IHB}=180^{\circ}.$$
		Vậy tứ giác $MPHQ$ nội tiếp.
			\end{enumerate}
	}
\end{ex}
\begin{ex}%[9D5G1]
	Tìm giá trị nhỏ nhất của biểu thức:\\
	$P=\left( \sqrt{x}+\dfrac{a}{\sqrt{x}}\right) \left( \sqrt{x}+\dfrac{b}{\sqrt{x}} \right) $\\
	Với $x>0$, $a$ và $b$ là các hằng số dương cho trước.
	\loigiai{$P=\left( \sqrt{x}+\dfrac{a}{\sqrt{x}}\right) \left( \sqrt{x}+\dfrac{b}{\sqrt{x}} \right)\\
		=x+a+b+\dfrac{ab}{x}\\
		=x-2\sqrt{x}.\sqrt{\dfrac{ab}{x}}+\dfrac{ab}{x}+2\sqrt{ab}+a+b\\
		=\left( \sqrt{x}-\sqrt{\dfrac{ab}{x}}\right) ^2+a+b+2\sqrt{ab}\geqslant a+b+2\sqrt{ab} $\\
		Vậy giá trị nhỏ nhất của P là: $a+b+2\sqrt{ab} $\\
		Dấu bằng xảy ra khi: $x=\sqrt{ab}$
	
	}
	\end{ex}
