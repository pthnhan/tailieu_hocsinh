\begin{ex}%[Đề thi vào 10-Chuyên Khoa học tự nhiên 2015 - Vòng 1]%[8D2G9]
Giả sử $a,b$ là hai số thực phân biệt thỏa mãn $a^2+3a=b^2+3b=2$
\begin{enumerate}
	\item Chứng minh rằng $a+b=-3$.
	\item Chứng minh rằng $a^3+b^3=-45$.
\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item $\heva{&a^2+3a=2\\&b^2+3b=2}\Rightarrow a^2-b^2+3(a-b)=0\Leftrightarrow (a-b)(a+b)+3(a-b)=0$\\
		$\Leftrightarrow (a-b)(a+b+3)=0\Leftrightarrow \hoac{&a-b=0\mbox{ (loại vì $a\neq b$)}\\&a+b=-3}$.
		\item Từ kết quả câu $a)$ ta suy ra $(a+b)^3=-27\Leftrightarrow a^3+b^3+3ab(a+b)=-27\Leftrightarrow a^3+b^3-9ab=-27$.\\
		Từ giả thiêt, ta có $a^2+3a+b^2+3b=4\Leftrightarrow (a+b)^2-2ab+3(a+b)=4\Leftrightarrow ab=-2$.\\
		Vậy $a^3+b^3=-45$.
	\end{enumerate} 

}
\end{ex}
\begin{ex}%[Đề thi vào 10-Chuyên Khoa học tự nhiên 2015 - Vòng 1]%[9D3G4]
Giải hệ phương trình $\heva{&2x+3y=5xy\\&4x^2+y^2=5xy^2}$
\loigiai{\begin{itemize}
		\item Dễ thấy $x=y=0$ là một nghiệm của hệ phương trình.
		\item Nếu $y\neq 0$, nhân hai vế của phương trình $2x+3y=5xy$ với $y$, ta được: \\
		$\heva{&2xy+3y^2=5xy^2\\&4x^2+y^2=5xy^2}\Leftrightarrow \heva{&2x+3y=5xy\\&2x^2-xy-y^2=0}\Leftrightarrow \heva{&2x+3y=5xy\\&(x-y)(2x+y)=0}$\\
		$\Leftrightarrow \hoac{&\heva{&2x+3y=5xy\\&x-y=0}\\&\heva{&2x+3y=5xy\\&2x+y=0}}\Leftrightarrow \hoac{&\heva{&x=1\\&y=1}\\&\heva{&x=\dfrac{2}{5}\\&y=\dfrac{4}{5}}}$.
	\end{itemize}
Vậy hệ đã cho có ba nghiệm $(0;0)$, $(1;1)$ và $\left(\dfrac{2}{5};-\dfrac{4}{5}\right)$.
}
\end{ex}
\begin{ex}%[Đề thi vào 10-Chuyên Khoa học tự nhiên 2015 - Vòng 1]%[9D5G2]
Tìm các số $x,y$ không nhỏ hơn $2$ sao cho $xy-1$ chia hết cho $(x-1)(y-1)$.
\loigiai{Ta có $xy-1\;\vdots\; (x-1)(y-1)\Leftrightarrow xy-1\vdots\; xy+1-x-y$.\\
Suy ra $(x+1)+(y-1)\;\vdots\; (x-1)(y-1)$ $\Rightarrow x-1\;\vdots\; y-1$ và $y-1\;\vdots\; x-1\Rightarrow x=y$\\
Từ giả thiết suy ra $x^2-1\;\vdots\; (x-1)^2\Rightarrow x+1\;\vdots\; x-1\Rightarrow 2\;\vdots\; x-1\Rightarrow \hoac{&x=2\\&x=3}$.\\
Vậy $x=y=3$ hoặc $x=y=3$.
}
\end{ex}
\begin{ex}%[Đề thi vào 10-Chuyên Khoa học tự nhiên 2015 - Vòng 1]%[9D5G1]
Với $x,y$ là những số thực thỏa mãn đẳng thức $x^2y^2+2y+1=0$. Tìm giá trị lớn nhất và nhỏ nhất của biểu thức $P=\dfrac{xy}{3y+1}$.
\loigiai{$x^2y^2+2y+1=0\Leftrightarrow 2y=-x^2y^2-1\Leftrightarrow y=\dfrac{-x^2y^2-1}{2}$.\\
	$P=\dfrac{2xy}{3\left(x^2y^2-1\right)+2}=\dfrac{2xy}{-3x^2y^2-1}\Leftrightarrow 3Px^2y^2+2xy+P=0$ $(1)$, $\Delta=4-12P^2$.\\
	Phương trình $(1)$ có nghiệm khi $\Delta \geq 0\Rightarrow 4-12P^2\geq 0\Rightarrow P^2\leq \dfrac{1}{3}\Leftrightarrow -\dfrac{1}{\sqrt{3}}\leq P\leq \dfrac{1}{3}$.\\
	Vậy $\min P=-\dfrac{1}{\sqrt{3}}\Leftrightarrow x=-\dfrac{\sqrt{3}}{2},y=-\dfrac{2}{3}$; $\max P=\dfrac{1}{\sqrt{3}}\Leftrightarrow x=\dfrac{\sqrt{3}}{2},y=\dfrac{2}{3}$.
}
\end{ex}

\begin{ex}%[Đề thi vào 10-Chuyên Khoa học tự nhiên 2015 - Vòng 1]%[9D5G5]
Cho bảng ô vuông $2015\time 2015$. Kí hiệu ô $\left(i;j\right)$ là ô ở hàng thứ $i$, cột $j$. Ta viết các số nguyên dương từ $1$ đến $2015$ theo qui tắc sau:
\immini{
\begin{enumerate}[i)]
	\item Số $1$ được viết ở ô $(1;1)$.
	\item Nếu số $k$ được viết ở ô $(i;j), i>1$ thì $k+1$ được viết vào ô $(i-1;j+1)$.
	\item Nếu $k$ được viết vào ô $(1;j)$ thì $k+1$ được viết vào ô $(j+1;1)$.
	\end{enumerate}
(xem hình vẽ)\\
Khi đó $2015$ được viết vào ô $(m;n)$. Hãy xác định $m,n$.}{\begin{tabular}{|c|c|c|c|c|}
	\hline 
	1 & 3 & 6 &10 & \dots\\ 
	\hline 
	2 & 5 & 9 &\dots & \\  
	\hline 
	4 & 8 & \dots & & \\  
	\hline 
	7 & \dots & & & \\  
	\hline 
	\dots &  & & & \\  
	\hline 
\end{tabular}}
\loigiai{
	Theo đề bài, các số nguyên dương được sắp xếp theo từng hàng chéo của bảng: hàng chéo thứ nhất có $1$ số, hàng chéo thứ hai có $2$ số, hàng chéo thứ ba có $3$ số, $\ldots$\\
	Giả sử số $x$ nằm ở hàng chéo thứ $k$ ta có:\\
	$\dfrac{k(k-1)}{2}<x\leq \dfrac{k(k+1)}{2}\Rightarrow \dfrac{-1+\sqrt{1+8x}}{2}\leq k<\dfrac{1+\sqrt{1+8x}}{2}\Rightarrow k=\left[\dfrac{-1+\sqrt{1+8x}}{2}\right]$\\
	Áp dụng với $x=2015$ ta có $k=\left[\dfrac{-1+\sqrt{1+8\cdot 2015}}{2}\right]=63$\\
	Số đầu tiên ở hàng chéo thứ $k=63$ là $\dfrac{k(k+1)}{2}+1=1954$\\
	Như vậy số $2015$ nằm ở vị trí thứ $2015-1954+1=62$ của hàng chéo thứ $63$ (vị trí áp chót).\\
	Vậy tọa độ của nó là $(2;62)$. 
}
\end{ex}
\begin{ex}%[Đề thi vào 10-Chuyên Khoa học tự nhiên 2015 - Vòng 1]%[9D5G1]
Giả sử $a,b,c$ là các số thực dương thỏa mãn $ab+bc+ac+abc\leq 4$. Chứng minh rằng $a^2+b^2+c^2+a+b+c\geq 2(ab+bc+ac)$.
\loigiai{Áp dụng bất đẳng thức Cô-si cho bốn số $ab,ac,bc,abc$ ta có:\\
	$4\geq ab+bc+ac+abc\geq 4\sqrt[4]{a^3b^3c^3}\Rightarrow abc\leq 1\Rightarrow a+b+c\geq 3\sqrt[3]{abc}\geq 3\sqrt[3]{a^2b^2c^2}$.\\
	Bất đẳng thức cần chứng minh tương đương với $a^2+b^2+c^2+3\sqrt[3]{a^2b^2c^2}\geq 2(ab+ac+bc)$ $(1)$.\\
	Đặt $\sqrt[3]{a^2}=x$, $\sqrt[3]{b^2}=y$, $\sqrt[3]{c^2}=z$($x,y,z>0$)\\
	$(1)\Leftrightarrow x^3+y^3+z^3+3xyz\geq 2\sqrt{x^3y^3}+2\sqrt{y^3z^3}+2\sqrt{x^3z^3}$\\
	Áp dụng bất đẳng thức Schur bậc ba, ta có:\\
	$x^3+y^3+z^3+3xyz\geq xy(x+y)+yz(y+z)+xz(x+z)$\\
	$\Leftrightarrow x(x-y)(x-z)+y(y-x)(y-z)+z(z-x)(z-y)\geq 0$ với mọi số thực không âm $x,y,z$.\\
	Thật vậy, không làm mất tính tổng quát, giả sử $x\geq y\geq z\Rightarrow z(z-x)(z-y)\geq 0$\\
	Ta được $x(x-z)-y(y-z)=x^2-xz+yz-y^2=(x-y)(x+y-z)\geq 0$\\
	$\Rightarrow x(x-y)(x-z)+y(x-y)(y-z)\geq 0\Leftrightarrow x(x-z)(x-y)+y(y-z)(y-x)\geq 0$\\
	$\Rightarrow x(x-y)(x-z)+y(y-x)(y-z)+z(z-x)(z-y)\geq 0$.\\
	Từ đó, ta có $x^3+y^3+z^3+3xyz\geq xy(x+y)+yz(y+z)+xz(x+z)\geq 2\sqrt{x^3y^3}+2\sqrt{y^3z^3}+2\sqrt{x^3z^3}$.\\
	Dấu "=" xảy ra khi $\hoac{&x=y=z\\&x=y,z=0}\Rightarrow a=b=c=1$.
}
\end{ex}
