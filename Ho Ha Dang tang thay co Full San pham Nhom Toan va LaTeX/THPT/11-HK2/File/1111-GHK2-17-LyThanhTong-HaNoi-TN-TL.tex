{\True $m= 4 $}
{$m=3 $}
{$m=1 $}
\loigiai{
\begin{itemize}
\item Xét trên khoảng $ (-\infty;3) $ và $ (3;+\infty)\Rightarrow f(x)=\dfrac{x^2-2x-3}{x-3}$ là hàm phân thức hữu tỉ nên liên tục trên $(-\infty;3) $ và $ (3;+\infty). $
\item Xét tại điểm $ x=3. $\\

Hàm số liên tục trên $ \Bbb{R} \Leftrightarrow $ hàm số liên tục tại $ x=3$ \\
 $\Leftrightarrow \lim\limits_{x\to 3} f(x)=f(3)\Leftrightarrow 4=12-2m\Leftrightarrow m=4. $
\end{itemize}

}
