\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 3, TP Hồ Chí Minh]%[Đoàn Ngọc Lê, dự án 8DT18 - HK1]%[8D1B5]
Thực hiện các phép tính, rút gọn
\begin{listEX}[2]
\item $4x(3-x)+(2x-3)^2.$
\item $15x^2y^2:3xy^2+\left( 4x^2-y^2 \right):(2x+y).$
\item $\dfrac{x}{x-3}+\dfrac{2x}{x-3}-\dfrac{9}{x-3}.$
\item $\dfrac{3}{b-1}-\dfrac{3+6b}{b^3-1}+\dfrac{1-2b}{b^2+b+1}.$
\end{listEX}
\loigiai{
\begin{listEX}
\item $4x(3-x)+(2x-3)^2=12x-4x^2+4x^2-12x+9=9.$
\item $15x^2y^2:3xy^2+\left( 4x^2-y^2 \right):(2x+y)=5x+(2x-y)(2x+y):(2x+y)=5x+2x-y=7x-y.$
\item Điều kiện: $x\ne 3$ $$\dfrac{x}{x-3}+\dfrac{2x}{x-3}-\dfrac{9}{x-3}=\dfrac{x+2x-9}{x-3}=\dfrac{3(x-3)}{x-3}=3.$$\vspace*{-1cm}
\item Điều kiện $b\ne 1$
\begin{align*}
&\dfrac{3}{b-1}-\dfrac{3+6b}{b^3-1}+\dfrac{1-2b}{b^2+b+1}\\
=& \dfrac{3\left( b^2+b+1 \right)-(3+6b)+(1-2b)(b-1)}{(b-1)\left( b^2+b+1 \right)}\\
=&\dfrac{3b^2+3b+3-3-6b+b-1-2b^2+2b}{(b-1)\left( b^2+b+1 \right)}\\
=&\dfrac{b^2-1}{(b-1)\left( b^2+b+1 \right)}\\
=&\dfrac{b+1}{b^2+b+1}.
\end{align*}
\end{listEX}\vspace*{-1cm}
}
\end{bt}

Phân tích các đa thức sau thành nhân tử
\begin{listEX}[2]
\item $4x^3+4x^2+x.$
\item $ab-7b+3a-21.$
\item $5x(x-3)+6-2x.$
\item $2x^2+3x^2-1.$
\end{listEX}
\loigiai{
\begin{listEX}
\item $4x^3+4x^2+x=x\left( 4x^2+4x+1 \right)=x\left( 2x+1 \right)^2.$
\item $ab-7b+3a-21=b(a-7)+3(a-7)=(a-7)(b+3).$
\item $5x(x-3)+6-2x=5x(x-3)-2(x-3)=(x-3)(5x-2).$
\item \begin{align*}
2x^3+3x^2-1=&2x^3+2x^2+x^2-1=2x^2(x+1)+(x+1)(x-1)=(x+1)(2x^2+x-1)\\
=&(x+1)(2x^2-x+2x-1)=(x+1)(x+1)(2x-1)=(x+1)^2(2x-1).
\end{align*}
\end{listEX}
}
\end{bt}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 3, TP Hồ Chí Minh]%[Đoàn Ngọc Lê, dự án 8DT18 - HK1]%[8H2B2]
Một nền nhà hình chữ nhật có chiều dài $20$m và chiều rộng bằng $\dfrac{1}{4}$ chiều dài. Người ta muốn lát gạch hình vuông cạnh $5$dm lên nền nhà đó nên đã mua gạch với tổng số tiền là $36800000$ đồng (giả sử khoảng cách giữa hai viên gạch kề nhau là không đáng kể). Hỏi giá của một viên gạch là bao nhiêu?
\loigiai{
Diện tích của nền nhà là
$$20\cdot 20\cdot\dfrac{1}{4}=100\ (\text{m}^2).$$
Diện tích một viên gạch là
$$\left( \dfrac{5}{10} \right)^2=0{,}25\ (\text{m}^2).$$
Số viên gạch cần dùng là
$$100:0{,}25=400\ (\text{viên}).$$
Số tiền một viên gạch là
$$36800000:400=92000\ (\text{đồng}).$$
}
\end{bt}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 3, TP Hồ Chí Minh]%[Đoàn Ngọc Lê, dự án 8DT18 - HK1]%[6D1B5]%[6D1B5]
Theo quyết định của bộ công thương ban hành, giá bán lẻ điện sinh hoạt từ ngày $01/12/2017$ sẽ dao động trong khoảng từ $1549$ đến $2701$ đồng mỗi kWh tùy bậc thang. Dưới đây là bảng so sánh giá điện trước và sau khi điều chỉnh:
\begin{center}
\begin{tabular}{|l|c|c|}
\hline
\textbf{MỨC SỬ DỤNG TRONG THÁNG (kWh)} & \textbf{GIÁ CŨ} & \textbf{GIÁ MỚI}\\
\hline
Bậc 1: Cho kWh từ \quad $0-50$\quad\ \  \quad ($50$ kWh) & $1484$ & $1549$\\
\hline
Bậc 2: Cho kWh từ \quad $51-100$\ \  \quad ($50$ kWh) & $1533$ & $1600$\\
\hline
Bậc 3: Cho kWh từ \quad $101-200$ \quad ($100$ kWh) & $1786$ & $1858$\\
\hline
Bậc 4: Cho kWh từ \quad $201-300$ \quad ($100$ kWh) & $2242$ & $2340$\\
\hline
Bậc 5: Cho kWh từ \quad $301-400$ \quad ($100$ kWh) & $2503$ & $2615$\\
\hline
Bậc 6: Cho kWh từ \quad $401$ trở lên & $2587$ & $2701$\\
\hline
\end{tabular}
\end{center}
Khi không tính thuế VAT
\begin{enumerate}
\item Trong tháng $11/2017$, hộ A tiêu thụ $540$ kWh. Hỏi hộ A phải trả bao nhiêu tiền?
\item Nếu tháng $12/2017$, hộ A vẫn tiêu thụ $540$ kWh thì theo giá mới số tiền phải trả tăng lên bao nhiêu?
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Số tiền hộ A phải trả trong tháng $11/2017$ là
$$50\cdot 1484+50\cdot 1533+100\cdot 1786+100\cdot 2242+100\cdot 2503+140\cdot 2587=1166130\ (\text{đồng}).$$
\item Số tiền hộ A phải trả trong tháng $12/2017$ là
$$50\cdot 1549+50\cdot 1600+100\cdot 1858+100\cdot 2340+100\cdot 2615+140\cdot 2701=1216890\ (\text{đồng}).$$
Số tiền hộ A phải trả thêm là $$1216890-1166130=50760\ (\text{đồng}).$$
\end{enumerate}\vspace*{-0.7cm}
}
\end{bt}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 3, TP Hồ Chí Minh]%[Đoàn Ngọc Lê, dự án 8DT18 - HK1]%[8H1K9]%[8H2K3]
Cho $\triangle ABC$ vuông tại $A$ với $AB < AC$ và có đường cao $AH$. Gọi $M$ là trung điểm của đoạn $AB$, $E$ là điểm đối xứng với điểm $H$ qua điểm $M$.
\begin{enumerate}
\item Chứng minh tứ giác $AHBE$ là hình chữ nhật.
\item Lấy điểm $D$ trên đoạn $HC$ sao cho $H$ là trung điểm của đoạn $DB$. Qua điểm $D$ vẽ đường thẳng vuông góc với $AC$ và cắt các đường thẳng $AC$, $MH$ lần lượt tại $K$, $F$. Gọi $N$ là
trung điểm của đoạn $AK$. Chứng minh tứ giác $AMFD$ là hình bình hành và $HN\perp AK$.
\item Vẽ $KV$ vuông góc $HK$ ($V$ thuộc $CD$). Chứng minh $S_{KDC}=2S_{KDV}$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[line cap=round, line join=round,font=\footnotesize, scale=1.5]
\tikzset{label style/.style={font=\footnotesize}}
	\clip (-2,-0.5) rectangle (5.5,3.5);
	\tkzDefPoints{0/0/A, 5/0/C, 0/3/B}
	\tkzDrawAltitude(B,C)(A)\tkzGetPoint{H}
	\tkzDefMidPoint(A,B)\tkzGetPoint{M}
	\tkzDefBarycentricPoint(M=-2,H=1)\tkzGetPoint{E}
	\tkzDefBarycentricPoint(H=-2,B=1)\tkzGetPoint{D}
	\tkzDrawAltitude(A,C)(D)\tkzGetPoint{K}
	\tkzDefMidPoint(A,K)\tkzGetPoint{N}
	\tkzInterLL(H,E)(D,K)\tkzGetPoint{F}
	\tkzDefMidPoint(D,C)\tkzGetPoint{V}
	\tkzDrawPolygon(A,B,C)
	\tkzDrawPolygon(A,E,B,C,D,F,E,H,N,H,K,V,C,D)
	\tkzMarkRightAngles(H,K,V B,A,C D,K,C A,H,B H,N,A)
	
	\tkzLabelPoints[below](A,C)
	\tkzLabelPoints[above](B,F,H)
	\tkzLabelPoints[below left](E)
	\tkzLabelPoints[above left](M)
	\tkzLabelPoints[below](K,N)
	\tkzLabelPoints[above right](D,V)
	\tkzDrawPoints[fill=black](A,B,C,D,E,F,H,K,V,M)
\end{tikzpicture}
\end{center}

\begin{enumerate}
\item Tứ giác $AHBE$ có hai đường chéo $AB$ và $HE$ cắt nhau tại trung điểm mỗi đường $\widehat{AHB}=90^\circ$ nên tứ giác $AHBE$ là hình chữ nhật.
\item $\heva{MA\perp AC\\ FD\perp AC}\Rightarrow MA\parallel FD$.\tagEX{1}
Tứ giác $AEHD$ có $AE\parallel HD$ (vì tứ giác $AHBE$ là hình chữ nhật), $AE=HD(=BH)$ nên tứ giác $AEHD$  là hình bình hành. Suy ra $AD\parallel MF$.\tagEX{2}
Từ $(1)$ và $(2)$ suy ra tứ giác $AMFD$ là hình bình hành.\\
Xét tứ giác $ABDK$ có $AB\parallel DK$ nên tứ giác $ABDK$ là hình thang. Lại có $H,N$ lần lượt là trung điểm của $BD$ và $AK$ nên $HN$ là đường trung bình của hình thang $ABDK$. Suy ra $HN\parallel AB$. Mà $AB\perp AC$ nên $HN\perp AC$, hay $HN\perp AK$.
\item $\triangle HAK$ có $HN$ vừa là đường cao, vừa là đường trung tuyến nên $\triangle HAK$ cân tại $H$. \\Suy ra $\widehat{HAK}=\widehat{HKA}$.\\
$\heva{&\widehat{HAK}+\widehat{VKC}=90^\circ\\&\widehat{HKA}+\widehat{KCD}=90^\circ \\&\widehat{HAK}=\widehat{HKA}}\Rightarrow \widehat{VKC}=\widehat{VCK}\Rightarrow VC=VK$\tagEX{3}
Từ đó ta chứng minh được $\widehat{VKD}=\widehat{VDK}$. Suy ra $VD=VK$.\tagEX{4}
Từ $(3)$ và $(4)$ suy ra $V$ là trung điểm của $DC$. Nên $S_{KDC}=2S_{KDV}$.
\end{enumerate}
}
