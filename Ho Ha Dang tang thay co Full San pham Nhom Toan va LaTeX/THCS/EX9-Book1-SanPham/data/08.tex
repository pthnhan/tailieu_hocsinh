\begin{vd}
    Tìm nghiệm nguyên của phương trình:
    \[x^3+x^2+x+1=y^3.\tag{1}\]
    \loigiai{
        Ta thấy $x^2+x+1=\left(x+\dfrac{1}{2}\right)^2+\dfrac{3}{4}>0$ nên từ (1) suy ra $x^3<y^3$. Do đó $x<y$.\\
        Xét hai trường hợp:
        \begin{enumerate}
            \item Trường hợp $y=x+1$. Thay vào (1) được:\\
            $x^3+x^2+x+1=(x+1)^3\\
            \Leftrightarrow x^3+x^2+x+1=x^3+3x^2+3x+1\\
            \Leftrightarrow 2x^2+2x=0 \Leftrightarrow 2x(x+1)=0$\\
            Vậy $x_1=0;x_2=-1$
            \item Trường hợp $y>x+1$. Ta có $y^3>(x+1)^3$ nên từ (1) ta có:\\
            $x^3+x^2+x+1>(x+1)^3\\
            \Leftrightarrow x^3+x^2+x+1>x^3+3x^2+3x+1\\
            \Leftrightarrow 2x^2+2x<0 \Leftrightarrow 2x(x+1)<0\\
            \Leftrightarrow -1<x<0$, loại do $x\in\mathbb{Z}$\\
            Nghiệm $(x;y)$ là $(0;1),(-1;0)$
        \end{enumerate}  
    }
\end{vd}

\begin{vd}
    Tính nghiệm nguyên của các phương trình:
    \begin{multicols}{2}
        \begin{enumerate}
            \item $x^3-y^3=xy+8$
            \item $x^3+y^3=3xy+3$
        \end{enumerate}
    \end{multicols}
    \loigiai{
        \begin{enumerate}
            \item \textit{Cách 1:} $x^3-y^3=xy+8\\
            \Leftrightarrow (x-y)^3+3xy(x-y)=xy+8$\\
            Đặt $x-y=a,xy=b$ ($a,b$ là các số nguyên), ta có:\\
            $a^3+3ab=b+8 \Leftrightarrow a^3-8=-b(3a-1)$\\
            Suy ra $(a^3-8)\, \vdots \,(3a-1) \Rightarrow 27(a^3-8)\, \vdots \,
            (3a-1) \Rightarrow (27a^3-1-215)\, \vdots \, (3a-1)$\\
            Do $(27a^3-1)\, \vdots \,(3a-1)$ nên $215\, \vdots \, (3a-1)$\\
            Phân tích ra thừa số nguyên tố: $215=5.43$\\
            Do đó $3a-1 \in \{\pm 1\ ;\ \pm 5\ ;\ \pm 43\ ;\ \pm 215\}$\\
            Do $3a-1$ chia cho 3 dư 2 nên $3a-1 \in \{-1\ ;\ 5\ ;\ -43\ ;\ 215\}$\\
            Ta có bảng giá trị sau:
            \begin{center}
                \begin{tabular}{|c|c|c|c|c|}
                    \hline
                    $3a-1$ & $-1$ & 5 & $-43$ & 215 \\
                    \hline
                    $a$ & 0 & 2 & $-14$ & 72 \\
                    \hline
                    $b=\dfrac{a^3-8}{1-3a}$ & $-8$ & 0 & $-64$ & $-1736$ \\
                    \hline
                \end{tabular}
            \end{center}
            $\Leftrightarrow (x+y)^3-3xy(x+y)=3xy+3$\\
            Đặt $x+y=a\ ,\ xy=b$ ($a,b$ là các số nguyên) ta có:\\
            $a^3-3ab=3b+3 \Leftrightarrow a^3-3=3b(a+1)$ nên $(a^3-3)\, \vdots \, (a+1)\\ \Leftrightarrow (a^3+1-4)\, \vdots \,(a+1) \Leftrightarrow 4 \, \vdots \, (a+1)$\\
            Do $a^3-3 \, \vdots \, 3$ nên $a\, \vdots \, 3$, suy ra $a+1$ chia cho 3 dư 1\\
            Ta có bảng giá trị sau:
            \begin{center}
                \begin{tabular}{|c|c|c|c|}
                    \hline 
                    $a+1$ & 1 & $-2$ & 4 \\
                    \hline
                    $a$ & 0 & $-3$ & 3 \\
                    \hline
                    $b=\dfrac{a^3-3}{3(a+1)}$ & $-1$ & 5 & 2 \\
                    \hline
                \end{tabular}
            \end{center}
            \begin{itemize}
                \item Trường hợp $a=0\ ;\ b=-1$ cho $(x;y)$ bằng $(1;-1)$ và hoán vị của nó.
                \item Trường hợp $a=-3\ ;\ b=5$ không cho nghiệm
                \item Trường hợp $a=3\ ;\ b=2$ cho $(x;y)$ bằng $(1;2)$ và hoán vị của nó.
            \end{itemize}
            Vậy nghiệm $(x;y)$ là $(1;-1)\ ,\ (-1;1)\ ,\ (1;2)\ ,\ (2;1)$
            \item \textit{Cách 2:} Đặt $x+y=a (a\in\mathbb{Z})$ . Ta có\\
            $x^3+(a-x)^3=3x(a-x)+3\\
            \Leftrightarrow a^3-3a^2+3ax^2=3ax-3x^2+3\\
            \Leftrightarrow 3(a+1)x^2-3a(a+1)x+(a^3-3)=0$
            \begin{itemize}
                \item Với $a=-1$ thì phương trình (1) trở thành $0x=4$, vô nghiệm
                \item Với $a\ne -1$ ta có:\\
                $\Delta = 9a^2(a+1)^2-12(a+1)(a^3-3)\\
                = 3(a+1)[3a^2(a+1)-4(a^3-3)]\\
                = 3(a+1)(3a^2-a^3+12)$\\
                Để phương trình (1) có nghiệm, ta phải có $\Delta \geq 0$\\
                Xét dấu $3a^2-a^3+12$, tức là $a^2(3-a)+12$ ta thấy:\\
                - Với $a \leq 3$ thì $a^2(3-a)+12>0$\\
                - Với $a \geq 4$ thì $a^2(3-a)+12<0$\\ Xét dấu $a+1$ ta thấy $a+1 \geq 0 \Leftrightarrow a\geq -1$.\\
                Ta có bảng dấu của $\Delta$
                \begin{center}
                    \begin{tabular}{|c|c|c|c|c|c|c|c|c|}
                        \hline
                        $a$ & ... & $-2$ & 0 & 1 & 2 & 3 & 4 & ... \\
                        \hline
                        $3a^2-a^3+12$ & + & + & + & + & + & + & - & - \\
                        \hline
                        $a+1$ & - & - & + & + & + & + & + & + \\
                        \hline
                        $\Delta $ & - & - & + & + & + & + & - & - \\
                        \hline
                    \end{tabular}
                \end{center}
                Để $\Delta \geq 0$ ta chỉ xét $a\in \{0;1;2;3\}$\\
                Lần lượt xét $a$ bằng 0;1;2;3 ta tìm được nghiệm $(x;y)$ là $(1;-1)\ ,\ (-1;1)\ ,\ (1;2)\ ,\ (2;1)$
            \end{itemize}
            \textit{Kinh nghiệm giải toán:} Ở câu a, do hệ số của $xy$ là 1 nên có thể giải theo cách 2\\
            Ở các phương trình chứa biểu thức $x^3+y^3$ hoặc $x^3-y^3$ nên vận dụng các hằng đẳng thức đáng nhớ có liên quan. Cũng có thể đặt ẩn phụ cho các biểu thức $x+y$ (hoặc $(x-y)$) và $xy$.
\end{enumerate}    
    }
\end{vd}

\begin{vd}
    Tìm nghiệm nguyên dương của phương trình:
    $$x^3-y^3=13(x^2+y^2)$$
    \loigiai{
    Đặt Ư$CLN(x,y)=d \geq 1$ thì $x=ad\ ;\ y=bd$ với $a>b\geq 1$ và $(a,b)=1$ \\
    Ta có $a^3d^3-b^3d^3=13(a^2d^2+b^2d^2)\\
    \Leftrightarrow d(a-b)(a^2+ab+b^2)=13(a^2+b^2) \hfill (1)$\\
    Suy ra $13(a^2+b^2) \, \vdots \, (a^2+ab+b^2)  \hfill (2)$\\
    Ta chứng minh được $(a^2+b^2\ ,\ a^2+ab+b^2)=1$. Thật vậy, nếu $a^2+b^2$ và $a^2+ab+b^2$ cùng có ước nguyên tố $p$ thì $ab\, \vdots \, p$. Suy ra tồn tại $a\, \vdots \, p$ hoặc $b \, \vdots \, p$. Giả sử $a \, \vdots \, p$ thì $b^2 \, \vdots \, p$ nên $b \, \vdots \, p$, trái với giả thiết $(a,b)=1$\\
    Do đó từ (2) suy ra $13 \, \vdots \, a^2+ab+b^2$. Do $a \geq 2$ và $b \geq 1$ nên: $a^2+ab+b^2=13 (3)$
    Do $a>b$ nên $a^2+ab+b^2 > 3b^2$. Từ $3b^2 <13$ ta có $b=1$ hoặc $b=2$
     \begin{itemize}
        \item Thay $b=1$ vào (3)
 được $a^2+a=12 \Rightarrow a(a+1)=12 \Rightarrow a=3$. Thay $a=3,b=1$ vào (1) được $d=5$. Từ đó $x=15, y=5$
         \item Thay $b=2$ vào (3) được $a^2+2a=9 \Rightarrow a(a+2)=9$ (loại)
     \end{itemize}   
     Vậy nghiệm $(x;y)$ là $(15;5)$  
    }
\end{vd}

\begin{bt}
    Tìm số nguyên $x$ để $x^3+3x$ là lập phương của một số nguyên
    \loigiai{
        Giả sử $x^3+3x=y^3 \,\, (y\in\mathbb{Z})$.\hfill(1)\\
        Ta có $x^3-3x^2+3x-1<x^3+3x<x^3+3x^2+3x+1$ nên $(x-1)^3<y^3<(x+1)^3$\\
        Suy ra $y^3=x^3$.\hfill$(2)$\\
        Từ (1) và (2) suy ra $x^3+3x=x^3 \Leftrightarrow 3x=0 \Leftrightarrow x=0$\\
        Với $x=0$ thì $y=0$   
    }  
\end{bt}

\begin{bt}
    Cho phương trình sau với nghiệm nguyên:
    $$x^3+2x^2+3x+1=y^3$$
    \begin{enumerate}
        \item Chứng minh rằng $y \leq x+1$
        \item Chứng minh rằng $y>x-1$
        \item Giải phương trình trên
    \end{enumerate}
    \loigiai{
        \begin{enumerate}
            \item Ta có $y^3=(x^3+3x^2+3x+1)-x^2=(x+1)^3-x^2 \leq (x+1)^3$ suy ra $y\leq x+1$
            \item Ta có $y^3=x^3+2x^2+3x+1=(x^3-3x^2+3x-1)+5x^2+2=(x-1)^3+5x^2+2 > (x-1)^3$ suy ra $y > x-1$
            \item Từ câu a và b suy ra $x-1<y\leq x+1$\\
            Do $x,y$ là các số nguyên nên $y=x$ hoặc $y=x+1$
            \begin{itemize}
                \item Trường hợp $y=x$, ta có $x^3=x^3+2x^2+3x+1 \Leftrightarrow 2x^2+3x+1=0$\\
                suy ra $x_1=-1\ ,\ x_2=-\dfrac{1}{2}$ loại
                \item Trường hợp $y=x+1$, ta có $(x+1)^3=x^3+2x^2+3x+1 \Leftrightarrow x^2=0 \Leftrightarrow x=0$.\\
                Suy ra $y=1$. Nghiệm $(x;y)$ là $(-1;-1),\ (0;1)$
            \end{itemize}
        \end{enumerate}    
    }
\end{bt}

\begin{bt}
    Tìm nghiệm nguyên của mỗi phương trình sau:
    \begin{multicols}{2}\begin{enumerate}
        \item $x^3-y^3=xy+25$
        \item $x^3-y^3=2xy+13$
        \item $x^3-y^3=3xy+1$
        \item $x^3+y^3=3xy-1$
\end{enumerate}\end{multicols}
\loigiai{
    \begin{enumerate}
        \item Giải tương tự như cách 1 của ví dụ 2a\\
        $x^3-y^3=xy+25 \Leftrightarrow (x-y)^3+3xy(x-y)=xy+25$\\
        Đặt $x-y=a\ ,\ xy=b$ ($a,b$ là các số nguyên), ta có:\\
        $a^3+3ab=b+25 \Leftrightarrow a^3-25=b(1-3a)$\\
        Suy ra $(a^3-25)\, \vdots \, (3a-1)$, nên $27(a^3-25)\, \vdots \, (3a-1)$, từ đó $3a-1$ là ước của 674 và chọn được $3a-1$ lấy giá trị thuộc $\{-1\ ;\ 2\ ;\ -337\ ;\ 674\}$\\
        Tương ứng với các giá trị trên tìm được:
        \begin{multicols}{4}
            \begin{itemize}
                \item $\heva{&a=0\\&b=-25}$
                \item $\heva{&a=1\\&b=12}$
                \item $\heva{&a=-112\\&b=-4169}$
                \item $\heva{&a=225\\&b=-16900}$
            \end{itemize}
        \end{multicols}
        Do phải có $a^2+4b \geq 0$ nên chỉ có một trường hợp $a=1,b=12$.\\
        Nghiệm $(x;y)$ là $(4;3),\ (-3;-4)$.\\
       \noindent{\bf Lưu ý.} Nếu yêu cầu tìm nghiệm nguyên dương của phương trình $x^3-y^3=xy+25$, ta có cách giải sau:\\
        Từ (1) suy ra $x>y$, do đó $x-y \geq 1$\\
        $(1) \Leftrightarrow (x-y)(x^2+xy+y^2)=xy+25$\\
        Do $x-y \geq 1$ và $x^2+xy+y^2 >0$ nên $x^2+xy+y^2 \leq xy+25$\\
        Suy ra $x^2+y^2 \leq 25$ nên $x \leq 4$\\
        Mặt khác do $x>y$ nên $xy \leq 2$. Từ (1) suy ra \\
        $x^3-y^3 \geq 27 \Rightarrow x^3 >27 \Rightarrow x >3$\\
        Từ (2) và (3) suy ra $x=4$. Do $y<x$ nên $y\in\{1;2;3\}$\\
        Chỉ có $y=3$ thỏa mãn (1)\\
        Nghiệm $(x;y)$ là $(4;3)$
        \item $x^3-y^3=2xy+13 \Leftrightarrow (x-y)^3+3xy(x-y)=2xy+13$\\
        Đặt $x-y=a,xy=b$ ($a,b$ là các số nguyên), ta có:\\
        $a^3+3ab=2b+13 \Leftrightarrow a^3-13=b(2-3a)$\\
        Suy ra $(a^3-13)\, \vdots \, (3a-2)$ nên $27(a^3-13)\, \vdots \, (3a-2)$, từ đó $3a-2$ là ước của 343 và chọn được $3a-2$ lấy giá trị thuộc $\{1;7;49;343\}$\\
        Tương ứng với các giá trị trên tìm được 
        \begin{multicols}{4}
            \begin{itemize}
                \item $\heva{&a=1\\&b=12}$
                \item $\heva{&a=3\\&b=-2}$
                \item $\heva{&a=17\\&b=-100}$
                \item $\heva{&a=115\\&b=-4434}$
            \end{itemize}
        \end{multicols}
        Do phải có $a^2 +4b \geq 0$ nên chỉ có hai trường hợp đầu thỏa mãn.\\
        Vậy nghiệm $(x;y)$ là $(4;3),\ (-3;-4),\ (1;-2),  (2;-1)$.
        \item Áp dụng hằng đẳng thức \\
        $$a^3+b^3+c^3-3abc=\dfrac{1}{2}(a+b+c)[(a-b)^2+(b-c)^2+(c-a)^2]$$ 
        với $a=x\ ,\ b=-y\ ,\ c=-1$ ta có: $$x^3-y^3-1-3xy=\dfrac{1}{2}(x-y-1)[(x+y)^2+(-y+1)^2+(-1-x)^2].$$
        Theo đề bài $x^3-y^3-1-3xy=0$\\
        Từ đó suy ra $(x-y-1)[(x+y)^2+(1-y)^2+(x+1)^2]=0$\\
        Nghiệm $(x;y)$ là $(-1;1)$ và $(t;t-1)$ với $t\in\mathbb{Z}$
        \item Nghiệm $(x;y)$ là $(1;1)$ và $(t;-t-1)$ với $t\in\mathbb{Z}$
\end{enumerate}
}
\end{bt}

\begin{bt}

    \begin{multicols}{2}\begin{enumerate}

        \item $x^3+y^3=(x+y)^2$
        \item $x^3+y^3=6xy-1$
        \item $x^3+y^3=2xy+11$

            \item Nếu $x+y \ne 0$ thì $x^2-xy+y^2=x+y\\

                \item Thay $y=0$ vào (1) được $x^2-x=0$, ta có $x_5=0\ ,\ x_6=1$\\

            \end{itemize}
            \item $x^3+y^3=6xy-1 \Leftrightarrow (x+y)^3-3xy(x+y)=6xy-1$\\

            \item $x^3+y^3=2xy+11 \Leftrightarrow (x+y)^3-3xy(x+y)=2xy+11$\\

\begin{bt}
    Giải mỗi phương trình sau:
    \begin{enumerate}
        \item $x^2y-2xy+y=125x$ với nghiệm nguyên dương
        \item $x^3-x^2y+2x-y=2$ với nghiệm nguyên
        \item $x^2+y^3-3y^2+3y=6$ với nghiệm tự nhiên
        \item $x^3-y^3=2(x^2+y^2)+3xy+17$ với nghiệm tự nhiên
        \item $x^3+y^3=3(x^2+y^2)$ với nghiệm nguyên dương
\end{enumerate}
\loigiai{
    \begin{enumerate}
        \item $x^2y-2xy+y-125x=0 \Leftrightarrow y(x^2-2x+1)=125x$
        \begin{itemize}
            \item $x=1$ không thỏa mãn phương trình
            \item Với $x\ne 1$ ta có $y=\dfrac{125x}{(x-1)^2}$\\
            Do $x$ và $x-1$ là nguyên tố cùng nhau nên $5^3\, \vdots \,(x-1)^2$. Từ đó $(x-1)^2$ lấy giá trị thuộc $\{1;25\}$\\
            Vì $x,y$ nguyên dương nên ta tìm được các nghiệm $(x;y)$ là $(2;250)\ ,\ (6;30)$
        \end{itemize}
        \item $x^3-x^2y+2x-y=2 \Leftrightarrow (x^2+1)y=x^3+2x-2\\
        \Leftrightarrow (x^2+1)y=x(x^2+1)+(x-2)$\\
        Suy ra $(x-2)\, \vdots \, (x^2+1) \Rightarrow (x^2-4)\, \vdots \,(x^2+1) \Rightarrow 5\, \vdots \,(x^2+1) \Rightarrow x^2+1 \in\{1;5\}$\\
        Do đó $x\in\{0;2;-2\}$\\
        Với $x=0$ thì $y=-2$\\
        Với $x=2$ thì $y=2$\\
        Với $x=-2$ thì $y$ không là số nguyên\\
        Nghiệm $(x;y)$ là $(0;-2)\ ,\ (2;2)$
        \item $x^2+y^3-3y^2+3y=6 \Leftrightarrow x^2+(y-1)^3=5$\\
        Do $x^2 \geq 0$ nên $(y-1)^3 \leq 5$. Suy ra $y-1 \leq 1$, do đó $y \in\{0;1;2\}$\\
        Nghiệm $(x;y)$ là $(2;2)$
        \item $x^3-y^3=2(x^2+y^2)+3xy+17\\
        \Leftrightarrow (x-y)(x^2+xy+y^2)=xy+17$.\hfill$(1)$\\
        Do $x,y$ là các số tự nhiên nên $xy+17 >0$\\
        Vì $x=y=0$ không thỏa mãn (1) nên $x^2+xy+y^2>0$.\\ 
        Suy ra $x-y-2>0$ do đó $x>y+2$, vậy $x\geq 3$.\hfill$(2)$\\
        Do $x-y-2 \geq 1$ nên từ (1) suy ra $x^2+xy+y^2 \leq xy+17$ từ đó $x^2+y^2 \leq 17$\hfill$(3)$\\
        Từ (2) và (3) suy ra $x\in\{3;4\}$.\\
        Thử các cặp số $(x;y)=(3;0),\ (4;0),\ (4;1)$ vào phương trình đã cho ta được một nghiệm $(x;y)$ là $(4;1)$.
        \item Giải tương tự ví dụ 4, được $3\, \vdots \, a^2-ab+b^2$ nên $a^2-ab+b^2 \in\{1;3\}$. Giả sử $a\geq b$ thì $b=1$. Bạn đọc tự giải tiếp để được nghiệm $(x;y)$ là $(3;3)$.
\end{enumerate}
}
\end{bt}

\begin{bt}
    Tìm nghiệm nguyên của mỗi phương trình sau:
    \begin{multicols}{3}
        \begin{enumerate}
            \item $9x^2-6x=y^3$
            \item $9x^3+6=y^3$
            \item $54x^3-1=y^3$
        \end{enumerate}
    \end{multicols}
    \loigiai{
        \begin{enumerate}
            \item Ta có:
            {\allowdisplaybreaks
            \begin{align*}
            &9x^2-6x=y^3\tag{1}\\
            \Leftrightarrow&(3x-1)^2=y^3+1.\tag{2}
           \end{align*}}Từ (2) suy ra $y^3+1 \geq 0$ nên $y^3 \geq -1$, do đó $y \geq -1$
            \begin{itemize}
                \item Thay $y=-1$ vào (2) được $(3x-1)^2=0$ nên $x=\dfrac{1}{3}$ loại
                \item Thay $y=0$ vào (1) đươc $3x(3x-2)=0$ nên $x=0$
                \item Thay $y=1$ vào (2) được $(3x-1)^2=2$ loại
                \item Xét $y \geq 2$. Ta có $(3x-1)^2=(y+1)(y^2-y+1)$\\
                Đặt $d=$ ƯCLN$(y+1,y^2-y+1)$\\
                Do $y^2-y+1=y(y+1)-2(y+1)+3$ nên $3\, \vdots \, d$\\
                Do $(3x-1)^2$ không chia hết cho 3 nên từ (3) suy ra $d\ne 3$\\
                Vậy $d=1$\\
                Các số nguyên dương $y+1$ và $y^2-y+1$ nguyên tố cùng nhau có tích là một số chính phương nên mỗi số đều là số chính phương, nhưng $(y-1)^2<y^2-y+1<y^2$ (do $y\geq 2$) nên $(y^2-y+1)$ không thể là số chính phương.\\
                Nghiệm $(x;y)$ là $(0;0)$ 
            \end{itemize}
            \item $9x^3+6=y^3$. Vế trái chia cho 9 có số dư là 6, còn vế phải chia cho 9 có số dư là 0,1,8. Phương trình không có nghiệm nguyên.
            \item $54x^3-1=y^3$.\hfill$(1)$\\
            \begin{itemize}
                \item Với $x=0$ thì $y=-1$
                \item Xét $x\ne 0$. Nhân hai vế của (1) với $216x^3$ được:\\
                $216x^3(54x^3-1)=216x^3y^3 \Leftrightarrow 216x^3.54x^3-216x^3=(6xy)^3\\
                \Leftrightarrow (6.18.x^3)^2-2(6.18x^3)=(6xy)^3\\
                \Leftrightarrow (6.18x^3-1)^2=(6xy)^3+1$\\
                Đặt $a=18x^3,b=6xy$ ta có $(6a-1)^2=b^3+1$\\
                Suy ra $b^3 \geq -1$ nên $b \geq -1$\\
                Lần lượt xét $b=-1\ ,\ b=0\ ,\ b=1\ ,\ b \geq 2$ và giải theo cách ở câu a đều loại\\
                Nghiệm $(x;y)$ là $(0;-1)$
            \end{itemize}
\end{enumerate}    
    }
