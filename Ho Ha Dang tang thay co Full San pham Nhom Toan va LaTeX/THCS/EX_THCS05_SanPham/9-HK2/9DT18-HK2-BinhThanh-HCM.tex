\begin{ex}%[Đề thi HK2 Toán 9 năm học 2017-2018, Q.Bình Thạnh, TPHCM]%[9D4B4]%[9D4B7]
	Giải các phương trình sau
	\begin{enumerate}
		\item $3(x^2-5)=4x$.
		\item $4x^4+3x^2-1=0$.
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Phương trình đã cho tương đương $3x^2-4x-15=0$.\\
		Hệ số $a=3, b=-4, c=-15$; $\Delta'=(-2)^2-3\cdot(-15)=49>0 \Rightarrow \sqrt{\Delta'}=7$.\\
		Suy ra phương trình có hai nghiệm phân biệt
		$\left[\begin{aligned} &x_1=\dfrac{2+7}{3}=3 \\ &x_2=\dfrac{2-7}{3}=-\dfrac{5}{3} \end{aligned}\right.$.\\
		Vậy phương trình đã cho có tập nghiệm $S=\left\lbrace -\dfrac{5}{3};3 \right\rbrace$.
		\item Đặt $t=x^2$, điều kiện $t\geqslant0$. Phương trình đã cho trở thành
		\[4t^2+3t-1=0 \tag{2}\]
		Vì $a-b+c=4-3+(-1)=0$ nên phương trình $(2)$ có $2$ nghiệm
		$\left[\begin{aligned} & t_1=-1 \,\text{(loại)}\\& t_2=\dfrac{1}{4} \end{aligned}\right.$.\\
		Từ $t_2=\dfrac{1}{4}$, suy ra $x^2=\dfrac{1}{4} \Leftrightarrow x=\pm \dfrac{1}{2}$.\\
		Vậy phương trình đã cho có tập nghiệm $S=\left\lbrace -\dfrac{1}{2};\dfrac{1}{2}\right\rbrace$.
	\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi HK2 Toán 9 năm học 2017-2018, Q.Bình Thạnh, TPHCM]%[9D4Y2]%[9D4B5]
	Cho hàm số $y=-\dfrac{1}{2}x^2$ có đồ thị $(P)$.
	\begin{enumerate}
		\item Vẽ $(P)$.
		\item Tìm các tọa độ giao điểm của $(P)$ và đường thẳng $(D)\colon y=x-4$ bằng phép toán.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
		\immini{
			\item
			\begin{itemize}
					\item Tập xác định: $\mathscr{D}=\mathbb{R}.$
					\item Bảng giá trị của hàm số $y=-\dfrac{1}{2}x^2$. 
					\begin{center}
						\begin{tabular}[c]{|c|c|c|c|c|c|}
							\hline
							$x$ & $-3$ & $-2$ & $0$ & $2$ & $3$\\
							\hline
							$y$ & $-\dfrac{9}{2}$ & $-2$ & $0$ & $-2$ & $-\dfrac{9}{2}$\\
							\hline 
						\end{tabular}
					\end{center}
					\item Vì hệ số $a=-\dfrac{1}{2}<0$ nên hàm số đồng biến trên $(-\infty;0)$ và nghịch biến trên $(0;+\infty)$.
			\end{itemize}
	}
{
\begin{tikzpicture}[>=stealth,x=1cm,y=1cm,scale=0.8]
\draw[->,line width = 1pt] (-4,0)--(0,0) node[above left]{$O$}--(4,0) node[above]{$x$};
\draw[->,line width = 1pt] (0,-5.5) --(0,2) node[right]{$y$};
\foreach \x in {-3,-2,-1,1,2,3}{\draw[-] (\x,-0.1)--(\x,0.1) node[shift={(0,0.2)}]{$\x$};}
\foreach \y in {-4,-3,-2,-1,1}{\draw[-] (-0.1,\y)--(0.1,\y) node[shift={(-0.5,0.2)}]{$\y$};}

\draw[dashed] (-2,0)--(-2,-2)--(2,-2)--(2,0);
\draw[dashed] (-3,0)--(-3,-4.25)--(3,-4.25)--(3,0);
\end{tikzpicture}
}
	\item
	Phương trình hoành độ giao điểm của $(D)$ và $(P)$ là
	\[-\dfrac{1}{2}x^2=x-4 \Leftrightarrow x^2+2x-8=0 \tag{1}\]
	Phương trình $(1)$ có $\Delta'=(-1)^2-1\cdot(-8)=9>0$.\\
	Suy ra phương trình $(1)$ có hai nghiệm phân biệt $\left[ \begin{aligned} &x=2 \\&x=-4 \end{aligned}\right.$.\\
	Ta có $x=2 \Rightarrow y=-2$; $x=-4 \Rightarrow y=-8$.\\
	Vậy $(D)$ và $(P)$ cắt nhau tại hai điểm có tọa độ là $(2;-2)$ và $(-4;-8)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi HK2 Toán 9 năm học 2017-2018, Q.Bình Thạnh, TPHCM]%[9D3K5]
	Một khu vườn hình chữ nhật có chu vi $70\text{\,m}$. Tính diện tích khu vườn biết $2$ lần chiều dài bé hơn $3$ lần chiều rộng $5\text{\,m}$.
	\loigiai{
		Gọi $x$, $y$ lần lượt là chiều dài và chiều rộng của hình chữ nhật. Điều kiện $x\geqslant y>0$.
		\begin{itemize}
			\item Nửa chu vi khu vườn là $\dfrac{70}{2}=35$ $\Rightarrow x+y=35 \quad (1)$.
			\item $2$ lần chiều dài bé hơn $3$ lần chiều rộng $5\text{\,m}$ $\Rightarrow 2x=3y-5 \quad (2)$.
		\end{itemize}
	Từ $(1)$, $(2)$ ta có hệ phương trình
	\begin{center}
			$\begin{cases}x+y=35 \\ 2x-3y=-5\end{cases} \Leftrightarrow \begin{cases}2x+2y=70 \\ 2x-3y=-5\end{cases} \Leftrightarrow \begin{cases}x+y=35 \\ 5y=75\end{cases} \Leftrightarrow \begin{cases}x=20 \\ y=15.\end{cases}$
	\end{center}
Suy ra, diện tích khu vườn là $xy=20\cdot 15 = 300\text{\,m}^2$.
	}
\end{ex}

\begin{ex}%[Đề thi HK2 Toán 9 năm học 2017-2018, Q.Bình Thạnh, TPHCM]%[9D3B5]
	Một xí nghiệp may cần thanh lý $1410$ bộ quần áo. Biết mỗi ngày xí nghiệp đó bán được $30$ bộ quần áo. Gọi $x$ là số ngày đã bán, $y$ là số bộ quần áo còn lại sau $x$ ngày bán.
	\begin{enumerate}
		\item Hãy lập công thức tính $y$ theo $x$.
		\item Xí nghiệp cần bao nhiêu ngày để bán hết số bộ quần áo cần thanh lý?
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Sau $x$ ngày, xí nghiệp bán được $30\cdot x$ bộ quần áo; do đó, số bộ quần áo còn lại là $y=1410-30x$.
			\item Xí nghiệp cần $\dfrac{1410}{30}=47$ ngày để bán hết số bộ quần áo cần thanh lý.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi HK2 Toán 9 năm học 2017-2018, Q.Bình Thạnh, TPHCM]%[9H3K10]
	\immini{
	Cho $A$ là điểm thuộc nữa đường tròn $(O)$ đường kính $BC=6$ cm và $\widehat {ACB}=30^{\circ}$. Tính $AB$, $AC$ và diện tích phần gạch chéo.}
{
	\begin{tikzpicture}[>=stealth, line join=round, line cap = round]
	\tkzDefPoints{0/0/O, -3/0/B, 3/0/C}
	\tkzDefPointBy[rotation=center O angle -60](B) \tkzGetPoint{A}
	\draw[pattern = north east lines] (3,0) arc (0:180:3) --(A)--cycle;	
	\tkzDrawPoints(O,A,B,C)
	\tkzDrawSegments(A,B B,C C,A)
	\tkzLabelPoints[right](C)	\tkzLabelPoints[left](B)	
	\tkzLabelPoints[below](O)	\tkzLabelPoints[above](A)
	\tkzMarkRightAngles(B,A,C);
	\tkzMarkAngle[label=$30^\circ$, size=0.7cm](A,C,B)
	\end{tikzpicture}
}
	\loigiai{
	\begin{itemize}
		\item Vì $A$ thuộc nữa đường tròn đường kính $BC$ nên tam giác $ABC$ vuông tại $A$. Ta có
		\begin{align*}
		&\sin \widehat {ACB}=\dfrac{AB}{BC} \Rightarrow AB=BC\cdot \sin 30^{\circ}=6\cdot \dfrac{1}{2}= 3\text{\,cm}.\\
		&\cos \widehat {ACB}=\dfrac{AC}{BC} \Rightarrow AC=BC\cdot \cos 30^{\circ}=6\cdot \dfrac{\sqrt{3}}{2}= 3\sqrt{3}\text{\,cm}.
		\end{align*}
		\item Ta có 
		\begin{align*}
		S_\text{phần gạch chéo}&=S_\text{nửa hình tròn} - S_{ABC}\\
		&=\dfrac{1}{2}\cdot \pi OB^2 - \dfrac{1}{2}\cdot AB\cdot AC\\
		&=\dfrac{1}{2}\cdot \pi \cdot 3^2 - \dfrac{1}{2}\cdot 3\cdot 3\sqrt{3}
		=\dfrac{9(\pi-\sqrt{3})}{2} \text{\,cm}^2.
		\end{align*}
	\end{itemize}

	}
\end{ex}

\begin{ex}%[Đề thi HK2 Toán 9 năm học 2017-2018, Q.Bình Thạnh, TPHCM]%[9D4B6]%[9D4K6]
	Cho phương trình $x^2-2x+m-3=0$ ($x$ là ẩn).
	\begin{enumerate}
		\item Tìm $m$ để phương trình có hai nghiệm $x_1$ và $x_2$.
		\item Gọi $x_1$ và $x_2$ là hai nghiệm của phương trình trên. Tìm $m$ để $x_1^2+x_2^2+x_1^2\cdot x_2^2=4$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Phương trình có hai nghiệm $x_1$, $x_2$ khi và chỉ khi
			\begin{center}
				$\Delta'\geqslant 0 \,\Leftrightarrow\, 1^2 - 1\cdot (m-3) \geqslant 0 
				\,\Leftrightarrow\, m \leqslant 4. \quad (*)$
			\end{center}
			\item Hệ thức Vi-ét: $\left\lbrace \begin{aligned}& x_1+x_2=2 \\ & x_1x_2=m-3 \end{aligned}\right.$.
			Theo đề ta có
			\begin{eqnarray*}
				& & x_1^2+x_2^2+x_1^2\cdot x_2^2=4\\
				&\Leftrightarrow & (x_1+x_2)^2-2x_1x_2+x_1^2\cdot x_2^2=4\\
				&\Leftrightarrow & 2^2-2(m-3)+(m-3)^2=4\\
				&\Leftrightarrow & m^2-8m+15=0 \Leftrightarrow
				\left[\begin{aligned} & m=5 \,\, \text{(không thỏa (*))}\\& m=3.\end{aligned}\right.
			\end{eqnarray*}
		Vậy $m=3$ là giá trị cần tìm.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi HK2 Toán 9 năm học 2017-2018, Q.Bình Thạnh, TPHCM]%[9H3B7]%[9H3K4]%[9H3G7]
	Cho tam giác $ABC$ nhọn ($AB<AC$) nội tiếp đường tròn $(O)$, các đường cao $BE$, $CF$ cắt nhau tại $H$.
	\begin{enumerate}
		\item Chứng minh tứ giác $AEHF$ và $BCEF$ nội tiếp.
		\item Hai đường thẳng $EF$ và $BC$ cắt nhau tại $I$. Vẽ tiếp tuyến $ID$ với đường tròn $(O)$ ($D$ là tiếp điểm, $D$ thuộc cung nhỏ $BC$). Chứng minh $ID^2=IB\cdot IC$.
		\item $DE$ và $DF$ cắt $(O)$ tại $M$, $N$. Chứng minh $NM \parallel EF$.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=.8]
			\tkzDefPoints{-1.8/4.1/A, -4/-2/B, 4/-2/C}
			\tkzCircumCenter(A,B,C)\tkzGetPoint{O};
			\tkzDefPointBy[projection=onto A--C](B)\tkzGetPoint{E}
			\tkzDefPointBy[projection=onto A--B](C)\tkzGetPoint{F}
			\tkzInterLL(B,E)(C,F)	\tkzGetPoint{H}
			\tkzInterLL(E,F)(C,B)	\tkzGetPoint{I}
			\tkzDrawCircle(O,A)
			\tkzDefMidPoint(I,O)\tkzGetPoint{K}
			\tkzInterCC(O,A)(K,O)\tkzGetFirstPoint{D}
			\tkzInterLC(D,E)(O,A)\tkzGetSecondPoint{M};
			\tkzInterLC(D,F)(O,A)\tkzGetSecondPoint{N};
			
			\tkzDrawPoints(O,A,B,C,E,F,H,D,I,M,N)
			\tkzDrawSegments(A,B I,C C,A B,E C,F I,D O,D E,I D,B D,C D,M D,N M,N A,M A,N)
			\tkzLabelPoints[right](C,E,O)	\tkzLabelPoints[left](I,N)	\tkzLabelPoints[shift={(-0.6,0.36)}](F)
			\tkzLabelPoints[below](H,D)	\tkzLabelPoints[shift={(-0.5,0)}](B)	\tkzLabelPoints[above](A,M)
			\tkzMarkRightAngles(B,F,C B,E,C I,D,O);
			\tkzMarkAngles[arc=ll,size=0.8cm](B,D,I I,C,D)
			\tkzMarkAngles(F,D,I N,M,D)	
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item 
			\begin{itemize}
				\item Ta có $\widehat {AEH} = \widehat {AFH}=90^\circ$ (do $BE$, $CF$ là đường cao)\\ 
				$\Rightarrow \widehat {AEH} + \widehat {AFH}=180^\circ \Rightarrow$ tứ giác $AEHF$ nội tiếp.
				\item Ta có $\widehat {BEC} = \widehat {BFC}=90^\circ$. Do đó, tứ giác $BCEF$ có đỉnh $E$, $F$ cùng nhìn cạnh $BC$ dưới một góc vuông $\Rightarrow$ tứ giác $BCEF$ nội tiếp.
			\end{itemize}
			\item Xét hai tam giác $IBD$ và $IDC$ có\\
			$\widehat I$ là góc chung\\
			$\widehat{IDB}=\widehat{ICD}$ (góc tạo bởi tiếp tuyến $ID$ và dây cung $BD$ bằng góc nội tiếp chắn $\wideparen{BD}$)
			$\Rightarrow \triangle {IBD}\backsim \triangle {IDC}$
			$\Rightarrow \dfrac{ID}{IC}=\dfrac{IB}{ID} \Rightarrow ID^2=IB\cdot IC$. $\quad (1)$
			\item 
			\begin{itemize}
				\item Xét hai tam giác $IBE$ và $IFC$ có\\
				$\widehat I$ là góc chung\\
				$\widehat{IEB}=\widehat{ICF}$ (cùng chắn $\wideparen{BF}$)\\
				$\Rightarrow \triangle {IBE}\backsim \triangle {IFC}$ $\Rightarrow \dfrac{IB}{IF}=\dfrac{IE}{IC} \Rightarrow IB\cdot IC = IE\cdot IF$. $\quad (2)$\\
				Từ $(1)$ và $(2)$, suy ra $ID^2=IE\cdot IF \Rightarrow \dfrac{ID}{IE}=\dfrac{IF}{ID}$.
				\item Xét hai tam giác $IDF$ và $IED$ có $\widehat I$ là góc chung và
				$\dfrac{ID}{IE}=\dfrac{IF}{ID}$\\
				$\Rightarrow \triangle {IDF}\backsim \triangle {IED} 
				\Rightarrow \widehat{IDF}=\widehat{IED}$.\\
				mà $\widehat{IDF}=\widehat{DMN}$ (góc tạo bởi tiếp tuyến $ID$ và dây $DN$ bằng góc nội tiếp chắn $\wideparen{DN}$).
				Suy ra $\widehat{IED}=\widehat{DMN} \Rightarrow EF \parallel MN$.			
			\end{itemize}			
		\end{enumerate}
	}
\end{ex}
