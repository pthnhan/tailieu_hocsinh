\begin{bt}[Đề thi vào chuyên 10, Sở giáo dục Hà Nội, 2017]%[9D5G3]%[Triệu Minh Hà]
	Tìm tất cả các cặp số nguyên dương $(x;y)$ thỏa mãn $x^3+y^3-9xy=0$.
	\loigiai
	{Ta có 
		{\allowdisplaybreaks
		\begin{align*}
		x^3+y^3-9xy=0 &\Leftrightarrow x^3+y^3+27-9xy=27\\ &\Leftrightarrow (x+y+3)(x^2+y^2+9-3x-3y-xy)=27 
		\\ & \Leftrightarrow (x+y+3)\left[ (x+y)^2-3(x+y) -3xy+9 \right]=27.\tag{*}
		\end{align*}}Vì vậy $x+y+3 \mid 27$. Mà $x,y \in \mathbb{N}^*$ nên $x+y+3 \geq 5$.
		\\ Vì vậy ta có $\hoac{x+y+3&=9 \\ x+y+3&=27} \Leftrightarrow \hoac{x+y&=6 \\ x+y&=24}$
		\\Trường hợp 1: Nếu $x+y=6$ thay vào $(*)$ suy ra $xy=8$.
		\\ Do đó $x,y$ là nghiệm của phương trình $t^2-6t+8=0 \Rightarrow \hoac{(x,y)=(2,4) \\ (x,y)=(4,2).}$
		\\ Trường hợp 2: Nếu $x+y=24$ thay vào $(*)$ ta có $xy=\dfrac{512}{3}$ (loại).
	}
\end{bt}

Ta có: 
{\allowdisplaybreaks

		&\Leftrightarrow 2(x+y)(x^2+x)-(x^2+x)=10 \\ 
		&\Leftrightarrow (x^2+x)\left[2(x+y)-1\right]=10.

		+) $x^2+x=(x+\dfrac{1}{2})^2-\dfrac{1}{4}>-1\Rightarrow x^2+x\ge 0$. \\
		Từ các nhận xét trên ta thấy chỉ có các trường hợp (TH) sau: \\
		$\begin{cases} &x^2+x=10 \\ &2(x+y)-1=1 \end{cases}$

				& x=-2}\\
			& x+y=3 }\Leftrightarrow \hoac{& \heva{& x=1 \\  & y=2}\\
			& \heva{ & x=-2 \\  & y=5}}$.\\

	}

\begin{bt}[Đề thi vào 10 chuyên, Sở giáo dục Hưng Yên, 2017]%[9D5G3]%[Triệu Minh Hà]
	Tìm tất cả các nghiệm nguyên của phương trình $ y^3 - 2x -2 = x(x+1)^2 $.
	\loigiai{
		Ta có $ y^3 = x^3 + 2x^2 + 3x +2 $. Rõ ràng $ 2x^2 + 3x + 2 = 2\left(x + \dfrac{3}{4}\right)^2 + \dfrac{7}{8} \ge \dfrac{7}{8} > 0  $. Do đó $ y^3 > x^3 $.\\
		Mặt khác, xét $ |x| > 1 $. \\
		Khi đó $ y^3 = (x+1)^3 + 1-x^2 \le (x+1)^3 $, mẫu thuẫn vì $ x^3 < y^3 < (x+1)^3 $.\\
		Vậy $ |x| \le 1 $. Khi đó $ x = \pm 1 \text{ hoặc } x = 0 $.
		\begin{enumerate}[TH1.]
			\item $ x = -1 $. Thay vào pt được $ y = 1 $.
			\item $ x = 1 $. Thay vào pt được $ y = 2 $.
			\item $ x = 0 $. Thay vào pt được $ y = \sqrt[3]{2} \notin \mathbb{Z} $, loại.
		\end{enumerate}
		Vậy phương trình có 2 nghiệm $ (x;y) $ là $ (-1;1), (1;2) $.
	}
\end{bt}

\begin{bt}[Thi vào 10, Chuyên Lam Sơn, Thanh Hóa, 2017]%[9D5G3]%[Triệu Minh Hà]
	Cho biểu thức $$A=\left(1-\dfrac{\sqrt{x}}{\sqrt{x}+1}\right):\left(\dfrac{\sqrt{x}+3}{\sqrt{x}-2}-\dfrac{\sqrt{x}+3}{\sqrt{x}-2}+\dfrac{\sqrt{x}+2}{x-5\sqrt{x}+6}\right)$$
	\begin{enumerate}[a)]
		\item Rút gọn biểu thức $A$.
		\item Tìm tất cả các giá trị nguyên của $x$ để biểu thức $A$ nhận giá trị nguyên.
	\end{enumerate}
	\loigiai
	{\hfill
		\begin{enumerate}
			\item Rút gọn biểu thức $A$.\\
			$\begin{aligned}
			A&=\left(1-\dfrac{\sqrt{x}}{\sqrt{x}+1}\right):\left(\dfrac{\sqrt{x}+3}{\sqrt{x}-2}-\dfrac{\sqrt{x}+3}{\sqrt{x}-2}+\dfrac{\sqrt{x}+2}{x-5\sqrt{x}+6}\right)\\
			&=\dfrac{1}{\sqrt{x}+1}:\dfrac{(\sqrt{x}+3)(\sqrt{x}-3)-(\sqrt{x}+2)(\sqrt{x}-2)+\sqrt{x}+2}{(\sqrt{x}-2)(\sqrt{x}-3)}\\
			&=\dfrac{1}{\sqrt{x}+1}:\dfrac{x-9-x+4+\sqrt{x}+2}{(\sqrt{x}-2)(\sqrt{x}-3)}\\
			&=\dfrac{1}{\sqrt{x}+1}:\dfrac{\sqrt{x}-3}{(\sqrt{x}-2)(\sqrt{x}+3)}\\
			&=\dfrac{1}{\sqrt{x}+1}:\dfrac{1}{\sqrt{x}-2}\\
			&=\dfrac{\sqrt{x}-2}{\sqrt{x}+1}.
			\end{aligned}$
			\item $A=\dfrac{\sqrt{x}-2}{\sqrt{x}+1}=1-\dfrac{-3}{\sqrt{x}+1}$. Để $A$ nhận giá trị nguyên khi $\dfrac{-3}{\sqrt{x}+1}$ đạt giá trị nguyên. Hay $-3\vdots(\sqrt{x}+1)\Leftrightarrow \sqrt{x}+1$ là ước của $-3$.\\
			Nên $\sqrt{x}+1=1\Leftrightarrow \sqrt{x}=0\Leftrightarrow x=0$ (thỏa mãn).\\
			$\sqrt{x}+1=-1\Leftrightarrow \sqrt{x}=-2$ (không thỏa mãn).\\
			$\sqrt{x}+1=3\Leftrightarrow \sqrt{x}=2\Leftrightarrow x=4$ (thỏa mãn).\\
			$\sqrt{x}+1=-3\Leftrightarrow \sqrt{x}=-4<0$ (không thỏa mãn).\\
			Vậy $x=0$ hoặc $x=4$ thì $A$ nhận giá trị nguyên.
		\end{enumerate}
	}
\end{bt}

}

\begin{bt}[Thi vào 10, Chuyên Bạc Liêu, 2017]%[9D5G3]%[Triệu Minh Hà]
Tìm các số nguyên $x, y$ thỏa mãn: 
$$x^2+y^2+5x^2y^2+60=37xy.$$
	\loigiai{
		\textbf{Cách 1.} Ta có:
		{\allowdisplaybreaks
		\begin{align*}
		&x^2+y^2+5x^2y^2+60=37xy \\
		\Leftrightarrow&x^2-2xy+y^2=-5x^2y^2+35xy-60\\
		\Leftrightarrow&(x-y)^2=5(-x^2y^2+7xy-12).\tag{*}
		\end{align*}}Vì $(x-y)^2 \geq 0,  \forall x, y \in \mathbb{Z}$ nên $-x^2y^2+7xy-12 \geq 0, \forall x, y \in \mathbb{Z}$.\hfill$(1)$\\
		Đặt $t=xy \,\, (t \in \mathbb{Z})$. Từ $(1)$ ta có $-t^2+7t-12 \geq 0 \Leftrightarrow 3 \leq t \leq 4$.\\
		Mà $t \in \mathbb{Z}$ nên ta có $t=3 \vee t=4$ hay $xy=3 \vee xy=4$.\\
		Khi đó $5(-x^2y^2+7xy-12)=0$ nên từ $(*)$ ta cũng có $(x-y)^2=0 $ hay $x=y$.
		\begin{itemize}
			\item $\begin{cases}xy=3\\
			x=y
			\end{cases} \Leftrightarrow \begin{cases}
			x^2=3\hspace{1cm} \text{(không tồn tại $x\in \mathbb{N})$}\\
			x=y
			\end{cases}$.
			\item $\begin{cases}
			xy=4\\
			x=y
			\end{cases} \Leftrightarrow$ $\begin{cases}
			x=2\\
			y=2
			\end{cases}$
			$\vee$
			$\begin{cases}
			x=-2\\
			y=-2
			\end{cases}$.
		\end{itemize}
		Vậy cặp số nguyên $(x;y)$ thỏa mãn phương trình trên là: $(2;2), (-2;-2)$.\\
		\textbf{Cách 2.}
		Xét trường hợp $x, y \geq 0$.\\
		Áp dụng bất đẳng thức AM-GM cho hai số không âm $x^2, y^2$ ta có:
		\begin{center}
			$\begin{gathered}
			\,\,\,\,\,\,\,37xy \geqslant 2xy + 5{x^2}{y^2} + 60 \hfill \\
			\Leftrightarrow 35xy \geqslant 5{x^2}{y^2} + 60 \hfill \\
			\Leftrightarrow 7xy \geqslant {x^2}{y^2} + 12 \hfill \\
			\Rightarrow 8xy \geqslant {x^2}{y^2} + 12 \hfill \\
			\Leftrightarrow 4 \geqslant {\left( {xy - 4} \right)^2} \hfill \\
			\Leftrightarrow 2 \geqslant xy - 4 \geqslant  - 2 \hfill \\
			\Leftrightarrow 6 \geqslant xy \geqslant 2. \hfill \\ 
			\end{gathered} $
		\end{center}
		Suy ra, $xy \in \left\{ {2;3;4;5;6} \right\}$.\\
		Mặt khác, phương trình ban đầu tương đương với:
		\[(x+y)^2=39xy-5x^2y^2-60.\tag{*}\]
		Ta xét các trường hợp:
		\begin{itemize}
			\item TH1: $xy=2$. Từ $(*)$ suy ra $(x+y)^2=-6$ (vô lí).
			\item TH2: $xy=3$. Từ $(*)$ suy ra $(x+y)^2=12$ (loại vì $(x+y)^2$ là số chính phương).
			\item TH3: $xy=4$. Từ $(*)$ suy ra $(x+y)^2=16$. Giải ra ta được $x=y=2$.
			\item TH4: $xy=5$. Từ $(*)$ suy ra $(x+y)^2=10$ (loại vì $(x+y)^2$ là số chính phương).
			\item TH5: $xy=6$. Từ $(*)$ suy ra $(x+y)^2=-6$ (vô lí).
		\end{itemize}
		Thử lại ta thấy $(x;y)=(2;2)$ là nghiệm của phương trình.\\
		Do $xy$ và $(-x).(-y)$ có vai trò như nhau trong phương trình ban đầu nên nghiệm của phương trình là 
		$(2;2), (-2;-2)$.
}
\end{bt}

\begin{bt}[Đề thi vào 10 Chuyên, Sở giáo dục Bình Phước, 2017]%[9D5G3]%[Triệu Minh Hà]

	$\begin{aligned}
	S&=n(n^4+5n^3+5n^2-5n-6)\\
	&=n[(n^2-1)(n^2+6)+5n(n^2-1)]\\

\end{bt}

\begin{bt}[Đề thi vào 10, Chuyên Đắk Lắk, 2017]%[9D5G3]%[Triệu Minh Hà]
\hfill
\begin{enumerate}
	\item[1.] Tìm các số nguyên tố $p$ sao cho $13p+1$ là lập phương của một số tự nhiên.
	\item[2.] Tìm hai số $x,y$ nguyên dương sao cho $\left(x+2\right)^2-6\left(y-1\right)^2+xy=24$.
\end{enumerate}
\loigiai
{
	\begin{enumerate}
		\item[1.] Giả sử $13p+1=a^3\ (a\in\mathbb{N}, a\ge 3)$.\\
		Phương trình $\Leftrightarrow 13p=\left(a-1\right)\left(a^2+a+1\right)$.\\

\section{Trong các đề thi học sinh giỏi quốc gia và quốc tế.}
\begin{bt}[Olympic Châu Á Thái Bình Dương lần thứ nhất, 1989]
	Chứng minh rằng phương trình $5n^2=36a^2+18b^2+6c^2$ ($4$ ẩn) không có nghiệm nguyên, ngoại trừ $a=b=c=n=0$.
	\loigiai{
		Giả sử phương trình đã cho có nghiệm nguyên $(a;b;c;n)$ khác $(0;0;0;0)$.
	Dễ thấy vế phải của phương trình 
		\begin{align*}
		5n^2=36a^2+18b^2+6c^2\tag{1}
		\end{align*}
		là bội của $3$, do đó $n$ chia hết cho $3$. Từ đó, $5n^2-36a^2-18b^2$ chia hết cho $9$, và do $$6c^2=5n^2-36a^2-18b^2$$
		nên $c$ chia hết cho $3$.
Như thế có thể chia hai vế của phương trình cho $9$ và được:
		\begin{align*}
		5m^2=4a^2+2b^2+6d^2.\tag{2}
		\end{align*}
Vì phương trình $(1)$ có nghiệm nên phương trình $(2)$ phải có nghiệm. Nếu $(2)$ có nghiệm thì sẽ có nghiệm với $a,b,d,m\geq 0$. Gọi $(a;b;d;m)$ là một nghiệm nguyên không âm của phương trình với giá trị $m$ là số tự nhiên nhỏ nhất. Ta xét số dư modulo $16$.
Do một số bình phương có dư $0,\ 1,\ 4$, hay $9$ modulo $16$ và từ phương trình $(2)$, suy ra $m$ là số chẵn nên $5m^2\equiv 0$ hay $4$ (mod $16$). Tương tự, $4a^2\equiv 0$ hay $4$ (mod $16$). Suy ra 		\begin{align*}
		2b^2+6d^2\equiv 0,\ 4 \text{ hay } 12\ (\text{mod } 16).\tag{3}
		\end{align*}		
		Mà $2b^2\equiv 0,\ 2$ hay $8$ (mod $16$) và $6d^2\equiv 0,\ 6$ hay $8$ (mod $16$). Do đó 
		\begin{align*}
		2b^2+6d^2 \equiv 0,\ 2,\ 6,\ 8,\ 10 \text{ hay } 14\ (\text{mod } 16). \tag{4}
		\end{align*}
		Từ $(3)$ và $(4)$ suy ra
		\begin{align*}
		2b^2+6d^2 \equiv 0\ (\text{mod } 16).\tag{5}
		\end{align*}
		Suy ra $b$ và $d$ đều là những số chẵn. Do vậy số $a$ không thể là số chẵn vì nếu ngược lại thì $$\dfrac{m}{2},\ \dfrac{a}{2},\ \dfrac{b}{2},\ \dfrac{d}{2}$$ sẽ là nghiệm của phương trình $(2)$ với $\dfrac{m}{2}<m$, vô lí.
	Như vậy ta có thể chia hai vế cho $4$ và được phương trình
		\begin{align*}
		5k^2=a^2+2e^2+6f^2 \tag{6}
		\end{align*}
		với $a$ là số lẻ. Suy ra $k$ cũng là số lẻ. Do đó $5k^2-a^2\equiv 4$ hay $12$ (mod $16$). Từ phương trình $(6)$ suy ra, $2e^2+6f^2\equiv  4$ hay $12$ (mod $16$). Điều này không thể xảy ra do $(5)$.
	Vậy phương trình $4$ ẩn đã cho không có nghiệm nguyên, ngoại trừ $a=b=c=n=0$.
	}
\end{bt}


\begin{bt}[Olympic Châu Á Thái Bình Dương lần thứ năm, 1993]
	Tìm tất cả các số nguyên dương $n$ để phương trình $$x^n+(x+2)^n+(2-x)^n=0$$ có nghiệm nguyên.
	\loigiai{
		Hiển nhiên phương trình vô nghiệm nếu $n$ chẵn vì khi đó tất cả các số hạng ở vế trái đều không âm và có ít nhất một số hạng dương. Ngoài ra, bằng cách kiểm tra trực tiếp ta thấy khi $n=1$, phương trình có nghiệm là $x=-4$.\\
		Xét $n$ lẻ và $n>3$. 
		
		Nếu $x$ dương, ta có
		$$x^n+(x+2)^n>(x+2)^n>(x-2)^n \Rightarrow x^n+(x+2)^n+(2-x)^n>0$$
		nên phương trình vô nghiệm. Từ đó, nghiệm $x$ nếu có phải là số âm. 
		
		Đặt $y=-x$. Rõ ràng $x=-1$ không phải là một nghiệm với mọi $n$, do vậy, nếu $x=-y$ là một nghiệm của phương trình thì $(x+2)=-(y-2)\leq 0$. Ta có
		\begin{align*}
		(y+2)^n-(y-2)^n=y^n.\tag{1}
		\end{align*}
		
		Bây giờ, vì $4=\left[(y+2)-(y-2)\right]$ chia hết $(y+2)^n-(y-2)^n$ nên từ phương trình $(1)$ suy ra $2$ chia hết $y$. Đặt $y=2z$, ta được
		\begin{align*}
		&(2z+2)^n=(2z)^n+(2z-2)^n\\
		\Leftrightarrow\ & (z+1)^n-(z-1)^n=z^n.\tag{2}
		\end{align*}
		Ta có $2$ chia hết $(z+1)^n-(z-1)^n$ nên từ phương trình $(2)$ suy ra $2$ chia hết $z$, do đó $z+1$ và $z-1$ là hai số lẻ. Ta lại có
		$$a^n-b^n=(a-b)(a^{n-1}+a^{n-2}b+a^{n-3}b^2+\cdots+b^{n-1}).$$
		Nếu cả $a$ lẫn $b$ đều là các số lẻ thì mỗi số hạng trong tổng 
		$$(a^{n-1}+a^{n-2}b+a^{n-3}b^2+\cdots+b^{n-1})$$
		là một số lẻ và do $n$ lẻ nên các số hạng là số lẻ, suy ra 
		$$(a^{n-1}+a^{n-2}b+a^{n-3}b^2+\cdots+b^{n-1})$$
		là một số lẻ. Từ đây, đặt $a=z+1,\ b=z-1$, ta thấy rằng số
		$$(z+1)^n-(z-1)^n=2(a^{n-1}+a^{n-2}b+a^{n-3}b^2+\cdots+b^{n-1})$$
		là số không chia hết cho $4$, tức là vế trái của $(2)$ không chia hết cho $4$. Nhưng vế phải của $(2)$ là $z^n$ chia hết cho $4$ vì $z$ là số chẵn. Do đó phương trình vô nghiệm.
		
		Tóm lại, chỉ có $n=1$ thỏa mãn yêu cầu đề bài.
	}
\end{bt}

\begin{bt}[Đề thi chọn đội tuyển Quốc gia Singapore, 1998 - 1999] 
	Tìm tất cả các số nguyên $m$ sao cho phương trình $$x^3-mx^2+mx-(m^2+1)=0$$
	có nghiệm nguyên.
	\loigiai{
		Giả sử $p$ là một số nguyên sao cho
		\begin{align*}
		&p^3-mp^2+mp-(m^2+1)=0\\
		\Leftrightarrow\ & (p^2+m)(p-m)=1.
		\end{align*}
		Do $p$ và $m$ là các số nguyên, một trong hai trường hợp sau phải xảy ra
		\begin{align*}
		(1)\qquad & \begin{cases} p^2+m=-1\\ p-m=-1\end{cases},\\
		(2)\qquad & \begin{cases} p^2+m=-1\\p-m=1\end{cases}.
		\end{align*}
		Trong trường hợp $(1)$, ta có $m=p+1$, suy ra $p^2+p+1=-1$ hay $p^2+p+2=0$, vô nghiệm.\\
		Ở trường hợp $(2)$, ta có $m=p-1$, suy ra $p^2+p-1=1$, suy ra $p^2+p-2=0$, cho ta $p=-2$ và $p=1$. Ta được hai cặp $(m=-3;\ p=-2)$ và $(m=0;\ p=1)$.\\
		Vậy với $m=-3$ hoặc $m=0$ thì phương trình đã cho có nghiệm nguyên.
	}
\end{bt}

\begin{bt}[Đề thi vô địch Quốc gia Hàn Quốc, 1997]
	Tìm tất cả các số nguyên $x,\ y,\ z$ thỏa mãn điều kiện $$x^2+y^2+z^2-2xyz=0.$$
	\loigiai{
		Ta chứng minh phương trình có nghiệm nguyên duy nhất là $x=y=z=0$. 		
		Trước hết, để ý rằng các số $x,\ y,\ z$ không thể đồng thời là ba lẻ, vì nếu ngược lại thì $$x^2+y^2+z^2-2xyz$$ sẽ là số lẻ, do đó không thể bằng $0$ được. 		
		Vậy phải có ít nhất một số chẵn, khi đó $xyz$ chia hết cho $2$. Suy ra $x^2+y^2+z^2$ chia hết cho 4. 		
		%Vì $x^2\equiv 0,\ 1$ (mod $4$), $y^2\equiv 0,\ 1$ (mod $4$), $z^2\equiv 0,\ 1$ (mod $4$) nên
		%$$x^2+y^2+z^2\equiv 0,\ 1,\ 2,\ 3\ (\text{mod } 4)$$
		%mà theo trên $x^2+y^2+z^2\equiv 0$ (mod $4$) nên cả ba số $x,y,z$ phải chẵn.
	Vì tất cả các số bình phương đều đồng dư với $0$ hoặc $1$ (mod $4$) nên từ $x^2+y^2+z^2\ \vdots\ 4$, suy ra $x,\ y,\ z$ phải đồng thời là những số chẵn. Ta viết $x=2x_1,\ y=2y_1,\ z=2z_1$, suy ra
	{\allowdisplaybreaks
		\begin{align*}
		&4x_1^2+4y_1^2+4z_1^2 =16x_1y_1z_1\\
		\Leftrightarrow&x_1^2+y_1^2+z_1^2 =4x_1y_1z_1.
		\end{align*}}Do vế phải chia hết cho $4$ nên lí luận tương tự như trên, ta được $x_1,\ y_1,\ z_1$ phải là các số chẵn, như vậy ta có thể viết $x_1=2x_2,\ y_1=2y_2,\ z_1=2z_2$, từ đó ta được
		$$x_2^2+y_2^2+z_2^2=8x_2y_2z_2.$$
Tổng quát, nếu $n\geq 1$ thì $x_n^2+y_n^2+z_n^2=2^{n+1}x_ny_nz_n$, suy ra các số $x_n,\ y_n,\ z_n$ đều là các số chẵn. Như vậy ta có thể viết
		$$x_n=2x_{n+1},\ y_n=2y_{n+1},\ z_n=z_{n+1},$$
		và lại có
		$$x_{n+1}^2+y_{n+1}^2+z_{n+1}^2=2^{n+2}x_{n+1}y_{n+1}z_{n+1}.$$
Lặp lại lí luận này nhiều lần, ta được dãy vô hạn các số nguyên $(x_1,\ x_2,\ \cdots)$ thỏa mãn $x_i=2x_{i+1}$. Lúc đó, $x=2^nx_n$, nên $2^n$ chia hết $x$ với mọi $n\geq 1$, cho nên $x=0$. Tương tự cũng có $y=z=0$.
	}
\end{bt}

\begin{bt}[Đề thi vô địch Quốc gia Đài Loan, 1999]
	Xác định tất cả các số nguyên dương $(x;y;z)$ sao cho $$(x+1)^{y+1}+1=(x+2)^{z+1}.$$
	\loigiai{
		Đặt $a=x+1,\ b=y+1,\ c=z+1$. Ta có $a,\ b,\ c\geq 2$ và 
		\begin{align*}
		& a^b+1=(a+1)^c\tag{1}\\
		\Leftrightarrow\ & \left[(a+1)-1\right]^b+1=(a+1)^c.\tag{2}
		\end{align*}
		Từ hai phương trình trên suy ra $(-1)^b+1\equiv 0$ (mod $a+1$), suy ra $b$ là số lẻ. Ta có 
		$$\left[(a+1)-1\right]^b=(a+1)^b-C_b^{b-1}(a+1)^{b-1}+C_b^{b-2}(a+1)^{b-2}+\cdots+C_b^1(a+1)-1$$
		Đưa phương trình $(2)$ về phương trình đồng dư (mod $(a+1)^2)$ ta được
		$$C_b^1(a+1)-1+1\equiv 0\ (\text{mod } (a+1)^2)\Leftrightarrow b(a+1)\equiv 0\ (\text{mod } (a+1)^2)$$
		suy ra $(a+1)$ chia hết $b$ và do đó $a$ là số chẵn. Mặt khác, 
		$$(a+1)^c=a^c+C_c^{c-1}a^{c-1}+C_c^{c-2}a^{c-2}+\cdots+C_c^1a+1$$
		Đưa phương trình $(1)$ về phương trình đồng dư (mod $a^2$) ta được 
		$$1\equiv C_c^1a+1\ (\text{mod } a^2)\Leftrightarrow ca\equiv 0\ (\text{mod } a^2)$$ 
		từ đó $c$ chia hết cho $a$, suy ra $c$ là số chẵn. Đặt $a=2a_1,\ c=2c_1$, ta có
		\begin{align*}
		2^ba_1^b=a^b=(a+1)^c-1=\Big[(a+1)^{c_1}-1\Big]\Big[(a+1)^{c_1}+1\Big].\tag{3}
		\end{align*}
		Chú ý rằng UCLN$\Big((a+1)^{c_1}-1,(a+1)^{c_1}+1\Big)=$ UCLN$\Big((a+1)^{c_1}+1,2\Big)\leq 2$ nên từ $(3)$
		suy ra $$\text{UCLN} \Big((a+1)^{c_1}-1,(a+1)^{c_1}+1\Big)=2$$
		Từ đó, do $2a_1$ là một ước số của $(a+1)^{c_1}-1$ (theo $(1)$), nên ta có thể kết luận:
		$$(a+1)^{c_1}-1=2a_1^b,\quad (a+1)^{c_1}+1=2^{b-1}.$$
		Ta phải có $2^{b-1}>2a_1^b$ nên suy ra $a_1=1$. Từ đó, các phương trình trên cho ta $c_1=1$ và $b=3$.\\
		Nghiệm duy nhất của bài toán là $(x;y;z)=(1;2;1)$.
	}
\end{bt}

\begin{bt}[Đề dự tuyển IMO lần thứ 32, năm 1991]
	Tìm nghiệm nguyên dương $x,\ y,\ z$ của phương trình $$3^x+4^y=5^z.$$
	\loigiai{
		Ta có $4^n\equiv 1$ (mod $3$) với mọi $n$ nguyên dương nên từ $3^x+4^y=5^z$, suy ra 
		$$5^z\equiv 1\ (\text{mod } 3).$$
		Mà $5^z\equiv 2^z\ (\text{mod } 3)$ nên
		$$2^z\equiv 1\ (\text{mod } 3)$$
		suy ra $z$ chẵn. Đặt $z=2z_1$ thì phương trình trở thành
		$$3^x+4^y=5^{2z_1}\Leftrightarrow (5^{z_1}-2^y)(5^{z_1}+2^y)=3^x.$$
		Vì $$\text{UCLN}\Big(5^{z_1}-2^y,5^{z_1}+2^y\Big)=\text{UCLN}\Big(2\cdot 2^y,5^{z_1}+2^y\Big)\not\vdots\ 3$$
		Và $$5^{z_1}-2^y<5^{z_1}+2^y$$
		Nên từ phương trình trên suy ra
		\begin{align*}
		5^{z_1}+2^y=3^x\quad \text{và}\quad 5^{z_1}-2^y=1\tag{*}
		\end{align*}
		Chú ý rằng $5^{z_1}\equiv (-1)^{z_1}$ (mod $3$) và $2^y\equiv (-1)^y$ (mod $3$) nên từ $(*)$ suy ra
		$$(-1)^{z_1}+(-1)^y\equiv 0\ (\text{mod } 3)\ \text{và } (-1)^{z_1}-(-1)^y\equiv 1\ (\text{mod } 3)$$
		nên $z_1$ lẻ và $y$ chẵn. Do đó $y\geq 2$. Ta có
		$$2^y\equiv 0,\ 3^x\equiv (-1)^x,\ 5^x\equiv 1\ (\text{mod } 4)$$
		từ đó phương trình $5^{z_1}+2^y=3^x$ có thể đưa được về đồng dư thức $1\equiv (-1)^x$ (mod $4$), suy ra $x$ cũng chẵn.\\
		Bây giờ, giả sử $y\geq 4$. Chú ý rằng $z_1$ lẻ, $x$ chẵn nên
		$$5^{z_1}\equiv 5,\ 2^y\equiv 0,\ 3^x\equiv 1\ (\text{mod } 8)$$
		Do đó có thể đưa phương trình $5^{z_1}+2^y=3^x$ về dạng $5\equiv 1$ (mod $8$), điều này mâu thuẫn. Suy ra $y=2,\ z_1=1,\ x=2$.\\
		Vậy $x=y=z=2$ là nghiệm duy nhất của phương trình.
	}
\end{bt}

\begin{bt}[Đề dự tuyển IMO thứ 38, năm 1997]
	Gọi $a,\ b,\ c$ là các số nguyên dương sao cho $a$ và $b$ là các số nguyên tố cùng nhau và $c$ nguyên tố cùng nhau với $a$ hay với $b$. Chứng minh rằng tồn tại vô hạn bộ ba $(x;y;z)$ các số nguyên dương phân biệt $x,\ y,\ z$ sao cho $$x^a+y^b=z^c.$$
	\loigiai{Cho $P\geq 3$ là một số nguyên dương, thế thì $Q=P^c-1>1$.\\
		Ta tìm một nghiệm có dạng
		$$x=Q^m;\ y=Q^n; z=PQ^k.$$
		Vì $x^a+y^b=Q^{ma}+Q^{nb};\ z^c=P^cQ^{kc}=Q^{kc-1}-Q^{kc}$ nên một nghiệm như thế sẽ tìm được nếu một trong các hệ sau có nghiệm
		$$\begin{cases}ma=kc+1\\ nb=kc \end{cases},\qquad \begin{cases}nb=kc+1\\ ma=kc \end{cases}$$
		Từ các điều kiện, suy ra rằng hoặc UCLN$(a,\ bc)=1$, hoặc UCLN$(b,\ ac)=1$. Giả sử rằng UCLN$(a,\ bc)=1$. Ta sẽ chứng minh rằng hệ thứ nhất có một nghiệm.
		Đặt $k=bt$ và $n=ct$ thỏa mãn phương trình thứ hai của hệ này. Thay $k=bt$ vào phương trình đầu của hệ ta được $ma=tbc+1$. Do UCLN$(a,\ bc)=1$ nên các số nguyên dương như $m$ và $t$ có thể tìm được, điều đó nghĩa là hệ nói trên có nghiệm. Rõ ràng là ta có $m\neq n$, vì UCLN$(kc,\ kc+1)=1$. Do đó $x\neq y$. Số $z$ khác $x$ và $y$ vì nó nguyên tố cùng nhau với cả hai số ấy.
		Vì $P$ bất kì nên ta thu được vô số nghiệm.
	}
\end{bt}
