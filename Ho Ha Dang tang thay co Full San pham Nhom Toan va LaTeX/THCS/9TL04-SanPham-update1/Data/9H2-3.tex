\begin{vd}
	Cho đường tròn tâm $(O)$ bán kính $5\text{\,cm}$, dây $AB$ bằng $8\text{\,cm}$.
	\begin{enumerate}
		\item Tính khoảng cách từ tâm $O$ đến dây $AB$.
		\item Gọi $I$ là điểm thuộc dây $AB$ sao cho $AI=1\text{\,cm}$. Kẻ dây $CD$ qua $I$ và vuông góc với $AB$. Chứng minh rằng $CD=AB$.
	\end{enumerate}
	\loigiai{
	\immini{\vspace*{-20pt}
	\begin{enumerate}
		\item Gọi $H$ là trung điểm của $AB$, suy ra $OH\perp AB$.\\
		Khoảng cách từ $O$ đến dây $AB$ là
		\[OH=\sqrt{OA^2-HA^2}=\sqrt{5^2-4^2}=3\text{\,cm}.\]\vspace*{-24pt}
		\item Kẻ $OK\perp CD$ tại $K$. Suy ra $OKIH$ là hình chữ nhật.\\
		mà $IH=AH-AI=3\text{\,cm}$ $\Rightarrow IH=OH$.\\
		suy ra $OKIH$ là hình vuông $\Rightarrow OK=OH$.\\
		Do đó khoảng cách từ tâm $O$ đến hai dây $AB$ và $CD$ bằng nhau, suy ra $AB=CD$.
	\end{enumerate}

\begin{vd}
	Cho đường tròn tâm $(O)$ các dây $MN$ và $PQ$ bằng nhau, các tia $MN$ và $PQ$ cắt nhau tại điểm $A$ nằm bên ngoài đường tròn. Gọi $E$ và $F$ theo thứ tự là trung điểm của $MN$ và $PQ$. Chứng minh rằng:
	\begin{listEX}[2]
		\item $AE=AF$.
		\item $AN=AQ$.
	\end{listEX}
	\loigiai{
		\begin{enumerate}
		\item Chứng minh $AE=AF$.
		\immini{
			Vì $E$, $F$ lần lượt là trung điểm của $MN$, $PQ$ nên
			\begin{center}
				$OE\perp MN$ và $OF\perp PQ$.
			\end{center}
			Mặt khác, $MN=PQ \,\Rightarrow\, OE=OF.$\\
			Suy ra
			$$AE=\sqrt{OA^2-OE^2}=\sqrt{OA^2-OF^2}=AF.$$
		}{
			\begin{tikzpicture}[scale=.9]
			\tkzDefPoints{0/0/O}
			\tkzDefShiftPoint[O](0:2){N}
			\tkzDefShiftPoint[O](-120:2){M}
			\tkzDefShiftPoint[O](30:2){Q}
			\tkzDefShiftPoint[O](150:2){P}

			\tkzInterLL(P,Q)(M,N)\tkzGetPoint{A}
			\tkzDefMidPoint(M,N)\tkzGetPoint{E}
			\tkzDefMidPoint(P,Q)\tkzGetPoint{F}
			\tkzDrawSegments(A,M A,P O,E O,F O,A)
			\tkzDrawPoints[fill=black](O,A,M,N,P,Q,E,F)		
			\tkzLabelPoints[above](Q,F)
			\tkzLabelPoints[right](A)
			\tkzLabelPoints[left](O,P)
			\tkzLabelPoints[below](M)
			\tkzLabelPoints[shift={(-0.1,0.1)}](N,E)
			\tkzMarkRightAngles(O,E,A O,F,A)
			\end{tikzpicture}
		}\vspace*{-1.2cm}
			\item Chứng minh $AN=AQ$.\\
			Ta có
			\[AN=AE-NE \text{\;và\;} AQ=AF-FQ\]
			mà
			\[NE=\dfrac{1}{2}MN=\dfrac{1}{2}PQ=QF \text{\;và\;} AE=AF\]
			Suy ra $AN=AQ$.
		\end{enumerate}
	}
\end{vd}

\begin{bt}
	Cho đường tròn tâm $O$ bán kính $OA=11\text{\,cm}$. Điểm $M$ thuộc bán kính $OA$ và cách $O$ là $7\text{\,cm}$. Qua $M$ kẻ dây $CD$ có độ dài $18\text{\,cm}$, $MC<MD$. Tính các độ dài $MC$, $MD$.
	\loigiai{
		\immini{
			Kẻ $OI\perp CD$ tại $I$, suy ra $I$ là trung điểm $CD$. Ta có
			\[OI=\sqrt{OC^2-CI^2}=\sqrt{11^2-9^2}=2\sqrt{10}.\]
			và 
			\[IM=\sqrt{OM^2-OI^2}=\sqrt{7^2-\left(2\sqrt{10}\right)^2}=\sqrt{9}=3.\]
			Suy ra
			\[CM=CI-IM=9-3=6\text{\,(cm)}\]
			và \[DM=CD-CM=18-6=12\text{\,(cm)}.\]
