\begin{bt}%[HKI, THPT Nguyễn Du, TP.HCM,2018]%[Dương Phước Sang, dự án 11HKI-18]%[1D1B3-1]
	Giải phương trình $\sin^2x+\dfrac{5}{2}\cos x-2=0$.
	\loigiai{
		$\begin{aligned}\text{Ta có } \sin^2x+\dfrac{5}{2}\cos x-2=0 &\Leftrightarrow 1-\cos^2x+\dfrac{5}{2}\cos x-2=0 \Leftrightarrow -\cos^2x+\dfrac{5}{2}\cos x-1=0\\
		&\Leftrightarrow \left[\begin{aligned}
		&\cos x=2 \\
		&\cos x=\dfrac{1}{2}
		\end{aligned}\right. \Leftrightarrow \cos x=\cos \dfrac{\pi}{3} \Leftrightarrow x=\pm \dfrac{\pi}{3}+k2\pi ~(k\in \mathbb{Z}).
		\end{aligned}$}
\end{bt}
\begin{bt}%[HKI, THPT Nguyễn Du, TP.HCM,2018]%[Dương Phước Sang, dự án 11HKI-18]%[1D1K3-1]
	Giải phương trình $\dfrac{\cos x}{1-\sin x}=1+\sin x$.
	\loigiai{
		Điều kiện: $\sin x\ne 1 \Leftrightarrow x\ne \dfrac{\pi}{2}+k2\pi ~(k\in \mathbb{Z})$. \\
		$\begin{aligned}\text{Ta có } \dfrac{\cos x}{1-\sin x}=1+\sin x &\Rightarrow \cos x=1-\sin^2x \Leftrightarrow \cos x=\cos^2x\\
		&\Leftrightarrow \cos x\left(1-\cos x\right)=0 \Leftrightarrow \left[\begin{aligned}
		&\cos x=0 \\
		&\cos x=1
		\end{aligned}\right. \Leftrightarrow \left[\begin{aligned}
		&x=\dfrac{\pi}{2}+k\pi \\
		&x=k2\pi
		\end{aligned}\right. ~(k\in \mathbb{Z}).
		\end{aligned}$ \\
		So với điều kiện ta nhận $x=-\dfrac{\pi}{2}+k2\pi $ và $x=k2\pi ~(k\in \mathbb{Z})$ làm các nghiệm của phương trình.}
\end{bt}
\begin{bt}%[HKI, THPT Nguyễn Du, TP.HCM,2018]%[Dương Phước Sang, dự án 11HKI-18]%[1D2B2-6]
	Tìm số nguyên dương $n$ thoả mãn $\mathrm{A}_n^1+\mathrm{C}_n^2=78$.
	\loigiai{
		Điều kiện: $n\geqslant 2~(n\in \mathbb{N})$. \\
		Ta có $\mathrm{A}_n^1+\mathrm{C}_n^2=78 \Leftrightarrow n+\dfrac{n(n-1)}{2}=78 \Leftrightarrow n^2+n-156=0 \Leftrightarrow n=12$ (do $n \geqslant 2,\,n \in \mathbb{N}$).}
\end{bt}
\begin{bt}%[HKI, THPT Nguyễn Du, TP.HCM,2018]%[Dương Phước Sang, dự án 11HKI-18]%[1D2B2-1]
	Cho tập hợp $A=\{0;1;2;3;4;5;6\}$. Hỏi có bao nhiêu số tự nhiên có $4$ chữ số đôi một khác nhau lấy từ tập hợp $A$?
	\loigiai{
		Ta có $\mathrm{A}_7^4$ cách xếp thứ tự $4$ phần tử khác nhau của $A$ thành một dãy gồm $4$ chữ số khác nhau. \\
		Tuy nhiên nếu chữ số $0$ đặt ở bìa trái thì số cách lập dãy $4$ chữ số khác nhau sẽ là $\mathrm{A}_6^3$. \\
		Vậy có $\mathrm{A}_7^4-\mathrm{A}_6^3=720$ số thoả mãn yêu cầu bài toán.}
\end{bt}
\begin{bt}%[HKI, THPT Nguyễn Du, TP.HCM,2018]%[Dương Phước Sang, dự án 11HKI-18]%[1D2K2-1]
	Cho tập hợp $A=\{0;1;2;3;4;5;6\}$. Hỏi có bao nhiêu số tự nhiên có $4$ chữ số lấy từ tập hợp $A$ sao cho chữ số phía sau lớn hơn chữ số liền trước nó?
	\loigiai{
		Ta có $\mathrm{C}_7^4$ cách xếp thứ tự $4$ phần tử khác nhau của $A$ thành một dãy theo thứ tự tăng dần. \\
		Tuy nhiên nếu chữ số $0$ đặt ở bìa trái thì số cách lập dãy $4$ chữ số tăng dần sẽ là $\mathrm{C}_6^3$. \\
		Vậy có $\mathrm{C}_7^4-\mathrm{C}_6^3=15$ số thoả mãn yêu cầu bài toán.}
\end{bt}
\begin{bt}%[HKI, THPT Nguyễn Du, TP.HCM,2018]%[Dương Phước Sang, dự án 11HKI-18]%[1D2B3-2]
	Tìm hệ số của $x^3$ trong khai triển biểu thức ${\left(2x-\dfrac{1}{x^2}\right)\!}^{12}$.
	\loigiai{
		Ta có ${\left(2x-\dfrac{1}{x^2}\right)\!}^{12}=\displaystyle\sum_{k=0}^{12} \mathrm{C}_{12}^k(2x)^{12-k}{\left(-\dfrac{1}{x^2}\right)}^k=\displaystyle\sum_{k=0}^{12} \mathrm{C}_{12}^k2^{12-k}(-1)^kx^{12-3k}$. \\
		Số hạng chứa $x^3$ trong khai triển trên thoả mãn điều kiện $12-3k=3 \Leftrightarrow k=3$. \\
		Vậy hệ số cần tìm là $\mathrm{C}_{12}^32^9(-1)^3=-112640$.}
\end{bt}
\begin{bt}%[HKI, THPT Nguyễn Du, TP.HCM,2018]%[Dương Phước Sang, dự án 11HKI-18]%[1D2B5-2]
	Trong chuyến đi học tập tham quan ngoại khóa ở Đà lạt của Trường THPT Nguyễn Du, bạn An vào một cửa hàng bán hoa để mua hoa tặng mẹ. Bạn An chọn ngẫu nhiên 5 bông hoa hồng từ một chậu hoa có 10 bông hồng nhung và 7 bông hồng trắng. Tính xác suất để mẹ bạn An nhận được:
	\begin{multicols}{2}\begin{enumerate}
			\item 5 bông hoa cùng màu.
			\item 5 bông hoa có đủ hai màu.
	\end{enumerate}\end{multicols}
	\loigiai{
		Phép thử “chọn 5 bông hoa từ 10 bông hồng nhung và 7 bông hồng trắng” có $n( \Omega )=\mathrm{C}_{17}^5$. \\
		Đặt A: “có 5 bông hoa cùng màu” có $n(A)=\mathrm{C}_{10}^5+\mathrm{C}_7^5$. \\
		Vậy $\mathrm{P}(A)=\dfrac{n(A)}{n( \Omega )}=\dfrac{\mathrm{C}_{10}^5+\mathrm{C}_7^5}{\mathrm{C}_{17}^5}=\dfrac{3}{68}$. \\
		Đặt B: “5 bông hoa có đủ 2 màu” thì $\overline{B}=A$ do đó $\mathrm{P}(B)=1-\mathrm{P}(A)=\dfrac{65}{68}$.}
\end{bt}
\begin{bt}%[HKI, THPT Nguyễn Du, TP.HCM,2018]%[Dương Phước Sang, dự án 11HKI-18]%[1H2G2-1]%[1H2K2-3]%[1H2B1-2]
	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình thang đáy lớn $AB$ và $AB=3a$, $CD=2a$. Gọi $O$ là giao điểm của $AC$ và $BD$, $M$ là điểm trên cạnh $SA$ sao cho $AM=\dfrac{3}{5}SA$.
	\begin{enumerate}
		\item Tìm giao tuyến của hai mặt phẳng $(SAD)$ và $(SBC)$.
		\item Tìm giao điểm $N$ của mặt phẳng $(MCD)$ và $SB$. Chứng minh $ON \parallel (SAD)$
		\item Gọi $I=SO\cap MC$. Tính $\dfrac{OI}{SI}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $S \in (SAD) \cap (SBC)$. \qquad (1)\\
			Gọi $E=AD \cap BC$ thì $\left\{\begin{aligned}
			&E \in AD \subset (SAD)\\
			&E \in BC \subset (SBC)\\
			\end{aligned}\right.$\\
			$\Rightarrow  E \in (SAD) \cap (SBC)$.\qquad (2)\\
			Từ (1) và (2) ta có $SE=(SAD)\cap (SBC)$.
			\immini{
				\item Gọi $d=(MCD)\cap (SAB)$.\\
				Ta có $\left\{\begin{aligned}
				&M \in (MCD)\\
				&M \in SA \subset (SAB)\\
				\end{aligned}\right.$\\
				$\Rightarrow M \in (MCD)\cap(SAB) \Rightarrow M \in d$.\\
				Do $AB \parallel CD$ nên $d \parallel AB$.\\
				Gọi $N=d\cap SB$ ta có $\left\{\begin{aligned}
				&N \in d \subset (MCD)\\
				&N \in SB\\
				\end{aligned}\right.$\\
				$\Rightarrow N=SB\cap (MCD)$.\\
				Do $MN \parallel AB$ nên $\dfrac{SN}{SB}=\dfrac{SM}{SA}=\dfrac{2}{5}$.\\
				Do $AB\parallel CD$ nên $\dfrac{OD}{OB}=\dfrac{CD}{AB}=\dfrac{2}{3} \Rightarrow \dfrac{OD}{BD}=\dfrac{2}{5}$.\\
				Như vậy $\dfrac{SN}{SB}=\dfrac{OD}{BD}=\dfrac{2}{5}\Rightarrow ON \parallel SD \subset (SAD)$.\\
				Ngoài ra $ON \not\subset (SAD)$ nên $ON \parallel (SAD)$.}
			{\begin{tikzpicture}[scale=0.8, line join=round, line cap=round]
				\tkzDefPoints{0/0/A,5/0/B,0.4/-2/D}
				\tkzDefPointBy[translation=from A to B](D)\tkzGetPoint{D'}

			\item Lấy $F$ trên đoạn $AM$ sao cho $OF \parallel MC$. Khi đó $\dfrac{MF}{MA}=\dfrac{OC}{AC}=\dfrac{2}{5}$.\\
			Suy ra $\dfrac{MF}{SM}=\dfrac{MF}{MA}\cdot\dfrac{MA}{SM}=\dfrac{2}{5}\cdot\dfrac{3}{2}=\dfrac{3}{5}$.
	\end{enumerate}}
