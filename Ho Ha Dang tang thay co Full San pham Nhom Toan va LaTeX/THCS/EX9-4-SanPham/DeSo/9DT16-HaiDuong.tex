	\item $\heva{&2x+y-3=0\\&\dfrac{x}{4}=\dfrac{y}{3}-1}$

   \begin{enumerate}
   	\item Tìm $a$ và $b$ biết đồ thị hàm số $y = ax + b$ đi qua điểm $A(-1; 5)$ và song song với đường thẳng $y=3x + 1$.
   	\item Một đội xe phải chuyên chở $36$ tấn hàng. Trước khi làm việc, đội xe đó được bổ sung thêm $3$ xe nữa nên mỗi xe chở ít hơn $1$ tấn so với dự định. Hỏi đội xe lúc đầu có bao nhiêu xe? Biết rằng số hàng chở trên tất cả các xe có khối lượng bằng nhau.
   \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item Từ giả thiết ta có $a=3$, $b=8$. Vậy hàm số cần tìm là $y=3x+8$.
        \item Gọi $x$ là số xe ban đầu ($x>0$). Theo bài ra ta có phương trình $\dfrac{36}{x}-\dfrac{36}{x+3}=1\Leftrightarrow x=9$. Vậy ban đầu có 9 xe.
    \end{enumerate}

\begin{ex}%[Đề thi vào 10, Sở Giáo Dục Hải Dương, 2016]%[9H3K7]%[9H3G8]
Cho nửa đường tròn $(O)$ đường kính $AB$. Gọi $C$ là điểm cố định thuộc đoạn thẳng $OB$ ($C$ khác $O$ và $B$). Dựng đường thẳng $d$ vuông góc với $AB$ tại điểm $C$, cắt nửa đường tròn $(O)$ tại điểm $M$. Trên cung nhỏ $MB$ lấy điểm $N$ bất kỳ ($N$ khác $M$ và $B$), tia $AN$ cắt đường thẳng $d$ tại điểm $F$, tia $BN$ cắt đường thẳng $d$ tại điểm $E$. Đường thẳng $AE$ cắt nửa đường tròn $(O)$ tại điểm $D$ ($D$ khác $A$).
\begin{enumerate}
	\item Chứng minh: $AD\cdot AE = AC\cdot AB$.
	\item Chứng minh ba điểm $B$, $F$, $D$ thẳng hàng và $F$ là tâm đường tròn nội tiếp tam giác $CDN$.
	\item Gọi $I$ là tâm đường tròn ngoại tiếp tam giác $AEF$. Chứng minh rằng điểm $I$ luôn nằm trên một đường thẳng cố định khi điểm $N$ di chuyển trên cung nhỏ $MB$.
\end{enumerate}
	\loigiai{ 
		\begin{center}
			\begin{tikzpicture}[scale=1]
			\tkzInit[xmin=-4,xmax=6 ,ymin=-2 ,ymax=6] 
			\tkzClip[space=.5]
			\tkzDefPoints{-3/0/A, 3/0/B, 0/0/O, 2/0/C, 4/2/x}
			\tkzDrawArc(O,B)(A)
			\tkzDefLine[perpendicular=through C,K=1](A,B)\tkzGetPoint{d}
			\tkzInterLC(C,d)(O,A)\tkzGetPoints{M}{M'}
			\tkzInterLC(O,x)(O,A)\tkzGetPoints{N'}{N}
			
			\tkzInterLL(A,N)(C,d)\tkzGetPoint{F}
			\tkzInterLL(B,N)(C,d)\tkzGetPoint{E}
			\tkzInterLC(A,E)(O,A)\tkzGetPoints{D'}{D}
			\tkzCircumCenter(A,E,F)\tkzGetPoint{I}
			\tkzDrawCircle(I,A)
			\tkzInterLC(A,B)(I,A)\tkzGetPoints{K}{K'}
			\tkzDrawSegments(A,B C,d A,N B,N A,E N,E B,D C,D D,N N,C)
			\tkzDrawPoints[fill=black](O,I,K)
			\tkzLabelPoints[above](D,I)
			\tkzLabelPoints[right](N,E)
			\tkzLabelPoints[left](F)
			\tkzLabelPoints[below](A,O,B,C,K)
			\tkzMarkRightAngle(B,D,E)
			\tkzMarkRightAngle(E,C,B)

		\end{center}
	\begin{enumerate}
		\item Theo giả thiết, $\angle{BDE}=\angle{ECB}=90^\circ$ nên tứ giác $BCDE$ nội tiếp đường tròn đường kính $BE$. Do đó, $AD\cdot AE=AC\cdot AB$.
		\item Xét tam giác $ABE$. Ta có: $EC$ và $AN$ là hai đường cao và cắt nhau tại $F$ nên $F$ là trực tâm của tam giác $ABE$. Mặt khác, $BD\perp AE$ nên $BD$ đi qua $F$ hay $B$, $F$, $D$ thẳng hàng.
		
		Xét tam giác $CDN$. Ta có: $\angle{CDF}=\angle{CAF}=\angle{BAN}=\angle{BDN}$ (tứ giác $ADFC$ và $ADNB$ nội tiếp). Suy ra $DF$ là phân giác của góc $CDN$. Tương tự, ta cũng có $NF$ là phân giác góc $CND$ hay $F$ chính là tâm đường tròn nội tiếp tam giác $CDN$.
		\item Gọi $K$ là giao điểm của $(I)$ và $AB$. Ta có tứ giác $AKFE$ nội tiếp nên $CK\cdot CA=CF\cdot CE$. Ngoài ra, tam giác $AFC$ đồng dạng với tam giác $EBC$ nên $CF\cdot CE=BC\cdot AC$. Từ đó suy ra $BC=CK$ hay $C$ là trung điểm của$BK$ do đó $K$ cố định. Hơn nữa, $IA=IK$ nên $I$ nằm trên đường trung trực của $AK$ (cố định).
	\end{enumerate}
}

\begin{ex}%[Đề thi vào 10, Sở Giáo Dục Hải Dương, 2016]%[9D5G1]
    Cho $a$, $b$, $c$ là ba số thực dương thỏa mãn $abc = 1$. Tìm giá trị lớn nhất của biểu thức $$P=\dfrac{ab}{a^5+b^5+ab}+\dfrac{bc}{b^5+c^5+bc}+\dfrac{ca}{c^5+a^5+ca}.$$
\loigiai{ Áp dụng AM-GM ta có: $a^5+a^5+a^5+b^5+b^5\geq 5a^3b^2$. 
	
	Do đó, $5(a^5+b^5)\geq 5(a^3b^2+a^2b^3)\Rightarrow a^5+b^5\geq a^2b^2(a+b)$.
	
	Từ đó ta có: $\dfrac{ab}{a^5+b^5+ab}\leq\dfrac{1}{ab(a+b)+1}=\dfrac{c}{a+b+c}$. Vì vậy, $$P\leq 1.$$
	Dấu \lq\lq=\rq\rq xảy ra khi và chỉ khi $a=b=c=1.$
}
\end{ex}
