	Trong các hàm số sau, hàm số nào là hàm số chẵn?
	\choice
	{\True $y=\dfrac{|3+x|-|3-x|}{x}$}
	{$y=\sqrt{4-2x}-\sqrt{4+2x}$}
	{$y=2x^4+x^3-5x$}
	{$y=x^5-3x^3+2x$}
	\loigiai{
		\begin{itemize}
			\item Hàm số $y=\dfrac{|3+x|-|3-x|}{x}$ có tập xác định là $\mathscr{D}=\mathbb{R} \setminus \{0\}$.\\
			Với $x \in \mathscr{D}$ ta có $-x \in \mathscr{D}$.\\
			Ta có $y(-x)=\dfrac{|3-x|-|3+x|}{-x}=\dfrac{|3+x| - |3-x|}{x} =y(x)$.\\
			Do đó hàm số $y=\dfrac{|3+x|-|3-x|}{x}$ là hàm số chẵn.
			\item Hàm số $y=\sqrt{4-2x}-\sqrt{4+2x}$ có tập xác định là $\mathscr{D}=[-2;2]$.\\
			Với $x \in \mathscr{D}$ ta có $-x \in \mathscr{D}$.\\
			Ta có $y(-x)=\sqrt{4+2x}-\sqrt{4-2x} = -\left( \sqrt{4-2x} -\sqrt{4+2x}\right) = -y(x)$.\\
			Do đó hàm số $y=\sqrt{4-2x}-\sqrt{4+2x}$ là hàm số lẻ.
			\item Hàm số $y=2x^4+x^3-5x$ có tập xác định là $\mathscr{D}=\mathbb{R}$.\\
			Với $x \in \mathbb{R}$ ta có $-x \in \mathbb{R}$.\\
			Ta có $y(-x)=2(-x)^4+(-x)^3-5(-x)=2x^4-x^3+5x \ne y(x)$.\\
			Do đó hàm số $y=2x^4+x^3-5x$ không phải là hàm số chẵn.
			\item Hàm số $y=x^5-3x^3+2x$ có tập xác định $\mathscr{D}=\mathbb{R}$.\\
			Với $x \in \mathbb{R}$ ta có $-x \in \mathbb{R}$.\\
			Ta có $y(-x)=(-x)^5-3(-x)^3+2(-x)=-x^5+3x^3-2x=-y(x)$.\\
			Do đó hàm số $y=x^5-3x^3+2x$ là hàm số lẻ.
		\end{itemize}
	}
\end{ex}

\begin{ex}%[Thi GHK, THPT Thủ Thiêm - Hồ Chí Minh, 2018-2019]%[Nguyễn Thị Chúc, EX-DT-10&11]%[0D2B1-4]
	Trong các hàm số dưới đây, hàm số nào là hàm số lẻ?
	\choice
	{\True $y=\dfrac{|5x-1|-|5x+1|}{x^2}$}
	{$y=5x^4+3x^2+1$}
	{$y=2x^3-3x^2+1$}
	{$y=\sqrt{2-3x}+\sqrt{2+3x}$}
	\loigiai{
		\begin{itemize}
			\item Hàm số $y=\dfrac{|5x-1|-|5x+1|}{x^2}$ có tập xác định là $\mathscr{D}=\mathbb{R} \setminus \{1\}$.\\
			Với $x \in \mathscr{D}$ ta có $-x \in \mathscr{D}$.\\
			Ta có $y(-x)=\dfrac{|5(-x)-1|-|5(-x)+1|}{(-x)^2}=\dfrac{|5x+1|-|5x-1|}{x^2}=-y(x)$.\\
			Do đó hàm số $y=\dfrac{|5x-1|-|5x+1|}{x^2}$ là hàm số lẻ.
			\item Hàm số $y=5x^4+3x^2+1$ có tập xác định là $\mathscr{D}=\mathbb{R}$.\\
			Với $x \in \mathbb{R}$ ta có $-x \in \mathbb{R}$.\\
			Ta có $y(-x)=5(-x)^4+3(-x)^2+1=5x^4+3x^2+1=y(x)$.\\
			Do đó hàm số $y=5x^4+3x^2+1$ là hàm số chẵn.
			\item Hàm số $y=2x^3-3x^2+1$ có tập xác định là $\mathscr{D}=\mathbb{R}$.\\
			Với $x \in \mathbb{R}$ ta có $-x \in \mathbb{R}$.\\
			Ta có $y(-x)=2(-x)^3-3(-x)^2+1=-2x^3-3x^2+1 \ne -y(x)$.\\
			Do đó hàm số $y=2x^3-3x^2+1$ không là hàm số lẻ.
			\item Hàm số $y=\sqrt{2-3x}+\sqrt{2+3x}$ có tập xác định là $\mathscr{D}=\left[ -\dfrac{2}{3};\dfrac{2}{3}\right]$.\\
			Với $x \in \mathscr{D}$ ta có $-x \in \mathscr{D}$.\\
			Ta có $y(-x)=\sqrt{2+3x}+\sqrt{2-3x}=y(x)$.\\
			Do đó hàm số $y=\sqrt{2-3x}+\sqrt{2+3x}$ là hàm số chẵn.
		\end{itemize}
		
	}
\end{ex}
