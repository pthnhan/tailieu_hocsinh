\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Bình Chánh, Hồ Chí Minh]%[Đỗ Đức Thảo, dự án 9DT19-HK1]%[9D1B8]
	Rút gọn các biểu thức sau
	\begin{enumEX}{1}
		\item $ \sqrt{11-6\sqrt{2}}-\sqrt{6-4\sqrt{2}} $.
		\item $ \dfrac{1}{2+\sqrt{3}}-\dfrac{2}{3+\sqrt{3}}+\dfrac{\sqrt{2}}{\sqrt{6}} $.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item \mbox{}\\ $\
			\begin{array}{lcl}
			\sqrt{11-6\sqrt{2}}-\sqrt{6-4\sqrt{2}}&=&\sqrt{(3-\sqrt{2})^2}-\sqrt{(2-\sqrt{2})^2}\\
			&=&|3-\sqrt{2}|-|2-\sqrt{2}|\\
			&=&3-\sqrt{2}-(2-\sqrt{2})\\
			&=&1.
			\end{array}
			$
			
			\item \mbox{}\\
			$\begin{array}{lcl}
			\dfrac{1}{2+\sqrt{3}}-\dfrac{2}{3+\sqrt{3}}+\dfrac{\sqrt{2}}{\sqrt{6}} &=&\dfrac{2-\sqrt{3}}{2^2-(\sqrt{3})^2}-\dfrac{2(3-\sqrt{3})}{3^2-(\sqrt{3})^2}+\dfrac{1}{\sqrt{3}} \\
			&=&2-\sqrt{3}-\dfrac{1}{3}(3-\sqrt{3})+\dfrac{1}{\sqrt{3}}\\
			&=&1-\dfrac{1}{\sqrt{3}}.
			\end{array}$
		\end{enumEX}
	}
\end{bt}

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Bình Chánh, Hồ Chí Minh]%[Đỗ Đức Thảo, dự án 9DT19-HK1]%[9D1B7]
	Giải phương trình $\sqrt{4x+20}+\sqrt{x+5}-\dfrac{1}{3}\sqrt{9x+45}=4$.
	\loigiai{
		$$\begin{array}{lccl}
		&\sqrt{4x+20}+\sqrt{x+5}-\dfrac{1}{3}\sqrt{9x+45}&=&4\\
		\Leftrightarrow & 2\sqrt{x+5}+\sqrt{x+5}-\dfrac{1}{3}\cdot 3\sqrt{x+5}
		&=&4\\
		\Leftrightarrow & 2\sqrt{x+5}&=&4\\
		\Leftrightarrow & \sqrt{x+5}&=&2 \ (\star).
		\end{array}$$
		Điều kiện của phương trình $ (\star) $ là $ x+5 \ge 0 \Leftrightarrow x \ge -5 $.\\ 
		Khi đó
		$$ 	(\star) \Leftrightarrow x+5=4 \Leftrightarrow x=-1   \text{ (nhận). } $$
		Vậy phương trình đã cho có nghiệm là $ x=-1 $.
	}
\end{bt}

	Thực hiện chương trình khuyến mãi: \lq\lq Ngày Chủ Nhật Vàng \rq\rq, một cửa hàng điện máy giảm giá $20\%$ trên $1$ tivi cho lô hàng tivi gồm có $50$ cái với giá bán lẻ trước đó là $6\ 200\ 000$ đồng/cái. Đến trưa cùng ngày thì cửa hàng đã bán được $20$ cái và cửa hàng quyết định giảm thêm $10\%$ nữa (so với giá đã giảm lần $1$) cho số tivi còn lại.
	\begin{enumerate}
		\item Tính số tiền mà cửa hàng thu được khi bán hết lô hàng tivi đó.
		\item Biết rằng giá vốn là $4\ 000\ 000$ đồng$/1$ cái tivi. Hỏi cửa hàng lời hay lỗ khi bán hết lô hàng tivi đó và số tiền lời hay lỗ là bao nhiêu?
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Số tiền bán $20$ cái tivi ban đầu là 
			$$20\cdot 6\ 200\ 000 \cdot 80\% = 99\ 200\ 000 \text{(đồng)}.$$
			Số tiền bán $30$ cái tivi còn lại là
			$$30 \cdot 6\ 200\ 000 \cdot 80\% \cdot 90\% =133\ 920\ 000 \text{(đồng)}.$$
			Vậy tổng số tiền cửa hàng thu được khi bán hết số tivi đó là
			$$S = 99\ 200\ 000 + 133\ 920\ 000 =233\ 120\ 000 \text{(đồng)}.$$
			\item Số tiền cửa hàng bỏ ra mua số tivi đó là
			$$50\cdot 4\ 000\ 000 = 200\ 000\ 000 \text{(đồng)}.$$
			Vậy cửa hàng đã lời và lời số tiền là
			$$233\ 120\ 000 - 200\ 000\ 000 =33\ 120\ 000 \text{(đồng)}.$$ 
		\end{enumerate}
	}
\end{bt}

	Cho đường tròn $(O;R)$, đường kính $AB$, dây cung $BC=R$.
	\begin{enumerate}
		\item Tính các cạnh và các góc chưa biết của $\Delta ABC$ theo $R$.
		\item Đường thẳng qua $O$ vuông góc với $AC$ cắt tiếp tuyến tại $A$ của đường tròn $(O)$ ở $D$. Chứng minh $DC$ là tiếp tuyến của đường tròn $(O)$.
		\item Đường thẳng $OD$ cắt đường tròn $(O)$ tại $I$. Chứng minh $I$ là tâm đường tròn nội tiếp tam giác $ADC$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item 
			\immini
			{
				Ta có $OB=OC=BC=R$ nên tam giác $OBC$ là tam giác đều, do đó $\widehat{OBC}=60^\circ$.\\
				Trong tam giác $ABC$ có $OC=R=\dfrac{2R}{2}=\dfrac{AB}{2}$ và $O$ là trung điểm $AB$, do đó tam giác $ACB$ vuông tại $C$. Suy ra $\widehat{BAC}=30^\circ$.\\
				Tam giác vuông $ABC$ có $AC = AB \cdot \cos 30^\circ = \dfrac{\sqrt{3}}{2} \cdot 2R = R \sqrt{3}$.
			}
			{
				\begin{tikzpicture}[line join = round, line cap = round, >=stealth, scale=0.7]

			\item Ta có $OD$ vuông góc với dây cung $AC$ nên 
			$OD$ là đường trung trực của $AC$. Suy ra $DA = DC$.\\
			Do đó $\triangle OAD = \triangle OCD$ (c.c.c).\\
