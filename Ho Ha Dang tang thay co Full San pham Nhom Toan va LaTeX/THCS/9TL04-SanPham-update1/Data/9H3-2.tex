\begin{vd}
	Cho hai đường tròn bằng nhau $(O)$ và $(O')$ cắt nhau tại hai điểm $A$ và $B$. Kẻ các đường kính $AOC$, $AO'D$. Gọi $E$ là giao điểm thứ hai của $AC$ với đường tròn $(O')$.
	\begin{enumerate}
		\item So sánh các cung nhỏ $BC$, $BD$.
		\item Chứng minh rằng $B$ là điểm chính giữa của cung $\wideparen{EBD}$ (tức là điểm $B$ chia cung $\wideparen{EBD}$ thành hai cung bằng nhau $\wideparen{BE} = \wideparen{BD}$).
	\end{enumerate}
	\loigiai{
	\begin{center}

\pgfmathsetmacro{\so}{2} % lấy số

Vậy $\triangle{BAC} = \triangle{BAD}$. Suy ra $BC = BD$.\\
Mặt khác, hai đường tròn $(O)$ và $(O')$ bằng nhau nên hai dây bằng nhau sẽ căng hai dây bằng nhau. Vậy $\wideparen{BC} = \wideparen{BD}$.

Do $BC = BD$ (câu a) nên $EB$ là đường trung tuyến của tam giác vuông $ECD$ ($\widehat{E} = 90^\circ$). Suy ra $BE = BD$.\\
		Trong $(O')$ ta có, $BE = BD$ suy ra $\wideparen{BE} = \wideparen{BD}$ hay $B$ là điểm chính giữa của cung $EBD$.
\end{enumerate}
}
\end{vd}

		\item Không mất tính tổng quát giả sử $AC > AB$, đường tròn ngoại tiếp tam giác $ABC$ cắt đường phân giác $AD$ tại $I$ $\Rightarrow \wideparen{BI} = \wideparen{IC} \Rightarrow BI = IC$.\\
		Tam giác $ABC$ không cân, suy ra $H$, $D$, $M$ là ba điểm phân biệt. \\
		Mặt khác, $D$ nằm  giữa $A$ và $I$,	$AM$ là trung tuyến $\Rightarrow IM \perp BC$, $AH$ là đường cao $\Rightarrow AH \perp BC$. Do đó $D$ nằm giữa $H$ và $M$.
		\item Tam giác $ABC$ nhọn $\Rightarrow \widehat{BAC} < 90^\circ \Rightarrow \wideparen{BC}$ nhỏ hơn nửa đường tròn.\\
		$\Rightarrow M$ nằm giữa $O$ và $I$ $\Rightarrow AM$ nằm giữa hai tia $AI$ và $AO$.\\
		$\Rightarrow \heva{& \widehat{MAD} < \widehat{OAI} \\ & OA = OI} \Rightarrow \widehat{OAI} = \widehat{OIA}$.\\
		Mà $AH \parallel IM \Rightarrow \widehat{OIA} = \widehat{IAH}$. Vậy $\widehat{MAD} < \widehat{DAH}$.
	\end{enumerate}
}
\end{vd}

	Cho đường tròn tâm $O$. Trên nửa đường tròn đường kính $AB$ lấy hai điểm $C$, $D$. Từ $C$ kẻ $CH$ vuông góc với $AB$, nó cắt đường tròn tại điểm thứ hai là $E$. Từ $A$ kẻ $AK$ vuông góc với $DC$, nó cắt đường tròn tại điểm thứ hai là $F$. Chứng minh rằng:
	\begin{enumerate}
		\item Hai cung nhỏ $CF$ và $DB$ bằng nhau.
		\item Hai cung nhỏ $BF$ và $DE$ bằng nhau.
		\item $DE =BF$.
	\end{enumerate}
	\loigiai{
\immini{\begin{enumerate}
		\item $CD$ và $FB$ đều vuông góc với $AK$ nên $CD \parallel FB$. \\
		Suy ra $\wideparen{CF} = \wideparen{DB}$ (hai cung bị chắn giữa hai dây song song). \quad $(1)$
		\item Do tính chất đối xứng qua đường kính $AB$ ta có $\wideparen{BC} = \wideparen{BE} \quad (2)$\\
		Cộng từng vế của $(1)$ và $(2)$ ta được\\
		$\wideparen{BC} + \widehat{CF} = \wideparen{DB} + \widehat{BE}$ (tính chất cộng hai cung) hay $\wideparen{BF} = \wideparen{DE} \quad (3)$.
		\item Với $(3)$ ta suy ra $BF = BE$.
	\end{enumerate}}{\begin{tikzpicture}[>=stealth,scale=1, line join = round, line cap = round]
\tkzInit[xmin=-2.5,xmax=2.5,ymin=-2.5,ymax=2.5]
\tkzClip
\tikzset{label style/.style={font=\footnotesize}}
\tkzDefPoints{0/0/O,2/0/B,-2/0/A}
\tkzDrawCircle(O,A)
\tkzDefPointBy[rotation = center O angle 25](B)\tkzGetPoint{D}
\tkzDefPointBy[rotation = center O angle 120](B)\tkzGetPoint{C}
\tkzDefPointBy[projection=onto A--B](C)\tkzGetPoint{H}
\tkzInterLC(C,H)(O,A)\tkzGetFirstPoint{E}
\tkzDefPointBy[projection=onto D--C](A)\tkzGetPoint{K}
\tkzInterLC(A,K)(O,A)\tkzGetSecondPoint{F}
\tkzDrawPoints[fill=black](O,A,B,D,C,H,E,K,F)
\tkzLabelPoints[above](O,C,K)
\tkzLabelPoints[below](E)
\tkzLabelPoints[left](A,F)
\tkzLabelPoints[right](B,D)
\tkzLabelPoints[below right](H)
\tkzMarkRightAngles(C,H,B D,K,A B,F,A)
\tkzDrawSegments(A,B C,H D,K A,K H,E B,F)
\end{tikzpicture}}
}
\end{bt}

%%Bai tap 06
\begin{bt}
	Trên dây cung $AB$ của một đường tròn $O$, lấy hai điểm $C$ và $D$ chia dây này thành ba đoạn thẳng bằng nhau $AC = CD = DB$. Các bán kính qua $C$ và $D$ cắt cung nhỏ $AB$ lần lượt tại $E$ và $F$. Chứng minh rằng:
	\begin{enumEX}{2}
		\item $\wideparen{AE} = \widehat{FB}$.
		\item $\widehat{AE} < \wideparen{EF}$.
	\end{enumEX}
	\loigiai{
\immini{\begin{enumerate}
		\item Tam giác cân $AOB$ có $\widehat{OAB} = \widehat{OBA}$.\\
		Mặt khác, $\triangle{AOC} = \triangle{BOD}$ (c.g.c) vì có $OA = OB$, $\widehat{OAB} = \widehat{OBA}$, $AC = BD$. Từ đó suy ra $\widehat{AOC} = \widehat{BOD}$ suy ra $\wideparen{AE} = \wideparen{FB}$.
		\item Tam giác $OCD$ là tam giác cân ($OC = OD$ do $\triangle{AOC} = \triangle{BOD}$) nên $\widehat{ODC} < 90^\circ$, từ đó suy ra $\widehat{CDF} > 90^\circ$.\\
		Mặt khác, trong tam giác $CDF$ có $\widehat{CDF} > \widehat{CFD}$ suy ra $CF > CD$ hay $CF > CA$.\\
		Xét $\triangle{AOC}$ và $\triangle{COF}$ có $OA = OF$, $OC$ chung, nhưng $CF > AC$ suy ra $\widehat{COD} > \widehat{AOC}$. Từ đó suy ra $\wideparen{EF} > \wideparen{AE}$. 
	\end{enumerate}}{\begin{tikzpicture}[>=stealth,scale=1, line join = round, line cap = round]
\tkzInit[xmin=-2.5,xmax=2.5,ymin=-2.5,ymax=2.5]
\tkzClip
\tikzset{label style/.style={font=\footnotesize}}
\tkzDefPoints{0/0/O,-2/-1/A}
\tkzDrawCircle(O,A)
\tkzDefPointBy[rotation = center O angle 126](A)\tkzGetPoint{B}
\tkzDefBarycentricPoint(A=2,B=1)
\tkzGetPoint{C}
\tkzDefBarycentricPoint(A=1,B=2)
\tkzGetPoint{D}
\tkzInterLC(O,C)(O,A)\tkzGetSecondPoint{E}
\tkzInterLC(O,D)(O,A)\tkzGetSecondPoint{F}
\tkzMarkSegments[mark=s||](A,C C,D D,B)
\tkzDrawPoints[fill=black](O,A,B,C,D,E,F)
\tkzDrawSegments(O,A O,B A,B O,E O,F C,F)
\tkzLabelPoints[above](O,C,D)
\tkzLabelPoints[left](A)
\tkzLabelPoints[right](B)
\tkzLabelPoints[below](E,F)
\end{tikzpicture}}
}
\end{bt}
