\begin{ex}%[Đề thi vào 10, Sở giáo dục Bình Định, 2017]%[9D1Y1]%[9D1B3]%[9D5G2]
Cho hai biểu thức $A=\dfrac{\sqrt{x}}{\sqrt{x}-2};B=\dfrac{2}{\sqrt{x}+2}+\dfrac{4\sqrt{x}}{x-4}$ với $x\geq0,x\neq4$.
\begin{enumerate}[a)]
\item Tính giá trị biểu thức $A$ khi $x=9$.
\item Rút gọn biểu thức $T=A-B$.
\item Tìm $x$ để $T$ là số nguyên.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item Thay $x=9$ và biểu thức A ta được:
$$A=\dfrac{\sqrt{9}}{\sqrt{9}-2}=\dfrac{3}{1}=3.$$
\item Ta có:\\
$T=A-B=\dfrac{\sqrt{x}}{\sqrt{x}-2}-\dfrac{2}{\sqrt{x}+2}-\dfrac{4\sqrt{x}}{x-4}$.\\
$=\dfrac{\sqrt{x}(\sqrt{x}+2)}{x-4}-\dfrac{2(\sqrt{x}-2)}{x-4}-\dfrac{4\sqrt{x}}{x-4}$\\
$=\dfrac{x+2\sqrt{x}-2\sqrt{x}+4-4\sqrt{x}}{x-4}$\\
$=\dfrac{x-4\sqrt{x}+4}{x-4}=\dfrac{(\sqrt{x}-2)^2}{(\sqrt{x}-2)(\sqrt{x}+2)}=\dfrac{\sqrt{x}-2}{\sqrt{x}+2}$.
\item $T$ nguyên $\Leftrightarrow \dfrac{\sqrt{x}-2}{\sqrt{x}+2}\in\mathbb{Z}\Leftrightarrow 1-\dfrac{4}{\sqrt{x}+2}\in\mathbb{Z}\Leftrightarrow \dfrac{4}{\sqrt{x}+2}\in\mathbb{Z}$ hay $\sqrt{x}+2$ là ước của $4$.\\
Ta có $U(4)=\left\lbrace\pm1;\pm2;\pm4 \right\rbrace $.\\
Cho $\sqrt{x}+2$ bằng các giá trị trong tập $U(4)$ ta  được $\left[\begin{array}{l}
x=0\; \text{(nhận)} \\ 
x=4\; \text{(loại)}
\end{array}  \right.$. \\
Vậy $x=0$ thì $T$ là số nguyên.
\end{enumerate}
 }
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Bình Định, 2017]%[9D4B3]%[9D4K6]
	Cho phương trình : $x^2-2mx-6m-9=0$ ($m$ là tham số)
\begin{enumerate}[a)]
\item Giải phương trình khi $m=0$.
\item Tìm $m$ để phương trình có hai nghiệm $x_1,x_2$ trái dấu và thỏa mãn: $x_1^2+x_2^2=13$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item Với $m=0$ phương trình trở thành: $x^2-9=0\Leftrightarrow(x-3)(x+3)=0\Leftrightarrow\left[ \begin{aligned}x=3\\x=-3\end{aligned}\right.$.\\ Vậy $S=\left\lbrace -3;3\right\rbrace $.
\item Để phương trình có hai nghiệm phân biệt và trái dấu thì: $$a.c<0\Leftrightarrow -6m-9<0\Leftrightarrow m>-\dfrac{3}{2}$$
Mặt khác $x_1^2+x_2^2=13\Leftrightarrow (x_1+x_2)^2-2x_1x_2=13$ \hfill (1) \\
Áp dụng định lí Viét ta có:
$\left\lbrace\begin{array}{ll}
x_1+x_2&=2m\\
x_1x_2&=-6m-9
\end{array}\right.$\\
Thay vào (1) ta được: $4m^2-2(-6m-9)=13\Leftrightarrow4m^2+12m+6=0\Leftrightarrow\left[\begin{aligned}m=\dfrac{-3+\sqrt{3}}{2}\; \text{(nhận)}\\m=\dfrac{-3-\sqrt{3}}{2}\; \text{(loại)}\end{aligned}\right. $
\newline Vậy $m=\dfrac{-3+\sqrt{3}}{2}$ thì phương trình có hai nghiệm $x_1,x_2$ trái dấu và thỏa $x_1^2+x_2^2=13$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Bình Định, 2017]%[9D3B5]%[9D3B6]
Một đám đất hình chữ nhật có chu vi là $24$ m. Nếu tăng độ dài một cạnh lên $2$ m và giảm độ dài cạnh còn lại $1$ m thì diện tích đám đất sẽ tăng thêm $1$ m$^2$. Tính độ dài các cạnh lúc ban đầu của đám đất.
\loigiai{
Gọi $x,y$ (m) là độ dài các cạnh của đám đất hình chữ nhật lúc ban đầu ($x,y>0$).\\
Vì chu vi là $24$ m nên ta có:
 \begin{align*}
 2(x+y)=24 \Leftrightarrow x+y = 12. \tag{1}
 \end{align*} 
Diện tích đám đất ban đầu là: $S_1=xy$.\\
Nếu tăng độ dài một cạnh lên $2$ m và giảm độ dài cạnh còn lại $1$ m thì diện tích đám đất là: $$S_2 = (x+2)(y-1).$$
Vì tăng độ dài một cạnh lên $2$ m và giảm độ dài cạnh còn lại $1$ m thì diện tích đám đất sẽ tăng thêm $1$ m$^2$ nên ta có:
\begin{align*}
&S_2 = S_1 +1\\
\Leftrightarrow & (x+2)(y-1)=xy+1\\
\Leftrightarrow & x-2y+3=0. \tag{2}
\end{align*}
Từ (1) và (2) ta có hệ phương trình: $\begin{cases}
x+y=12 \\
x-2y=-3
\end{cases} \Leftrightarrow 
\begin{cases}
x= 7\\
y=5
\end{cases}$ (nhận).\\
Vậy độ dài các cạnh lúc ban đầu của đám đất lúc ban đầu là $5$ m và $7$ m.
 }
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Bình Định, 2017]%[9H3B7]%[9H1G2]
Cho tam giác $ABC$ $(AB<AC)$ nội tiếp trong đường tròn tâm $(O)$, $M$ là một điểm nằm trên cung $BC$ không chứa điểm $A$. Gọi $D,E,F$ lần lượt là hình chiếu vuông góc của $M$ trên các đường thẳng $BC,CA,AB$. Chứng minh rằng:
\begin{enumerate}[a)]
\item Bốn điểm $M,D,B,F$ thuộc một đường tròn và bốn điểm $M,D,E,C$ thuộc một đường tròn;
\item Ba điểm $D,E,F$ thẳng hàng;
\item $\dfrac{BC}{MD}=\dfrac{CA}{ME}+\dfrac{AB}{MF}$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\immini{
\item Ta có: $D$ và $F$ lần lượt là hình chiếu của $M$ lên $BC$ và $AB$.\\
$\Rightarrow \widehat{BDM}+\widehat{BFM}=90^\circ+90^\circ=180^\circ$\\
$\Rightarrow DMFB$ nội tiếp đường tròn.\\
$\Rightarrow M,D,B,F$ thuộc một đường tròn.\\
Ta có: $D$ và $E$ lần lượt là hình chiếu của $M$ lên $BC$ và $AC$.\\
$\Rightarrow \widehat{MDC}=90^\circ=\widehat{MEC}$\\
$\Rightarrow MDEC$ nội tiếp đường tròn.\\
$\Rightarrow M,D,E,C$ thuộc một đường tròn.
}{
\begin{tikzpicture}
%\draw[cyan,dashed] (-4,-4) grid (4,4);
\clip (-4,-4) rectangle (2.8,2.8);
\tkzDefPoints{-2/2/A, -3/-2/B, 2/-2/C}\tkzDrawPolygon(A,B,C)
\tkzCircumCenter(A,B,C) \tkzGetPoint{O}
\tkzDrawCircle(O,A)
\tkzDefPointBy[rotation= center O angle 40](B) \tkzGetPoint{M}
\tkzDefPointBy[projection=onto A--B](M) \tkzGetPoint{F}
\tkzDefPointBy[projection=onto A--C](M) \tkzGetPoint{E}
\tkzDefPointBy[projection=onto B--C](M) \tkzGetPoint{D}
\tkzDrawSegments(B,F M,F M,D M,E F,E M,B M,C A,M)
\tkzDrawPoints(A,B,C,O,M,F,E,D)
\tkzLabelPoints[above left](A)
\tkzLabelPoints[above](D,O)
\tkzLabelPoints[below left](B,M,F)
\tkzLabelPoints[below right](C)
\tkzLabelPoints[above right](E)
\tkzMarkAngles[arc=l,size=7mm](B,D,F B,M,F)
\tkzMarkAngles[arc=ll,size=8mm](D,E,M D,C,M)
\tkzMarkAngles[mark=|,arc=l,size=4mm](F,B,M A,C,M)
\tkzMarkRightAngles(M,F,B M,D,C M,E,C)
\end{tikzpicture}
}
\item Ta có: $ \widehat{FMB}=\widehat{FDB}$ (do $MDBF$ nội tiếp). \hfill (i)\\
$\widehat{CME}=\widehat{CDE}$ (do $DECM$ nội tiếp). \hfill (ii)\\
Mặt khác, tứ giác $ACMB$ nội tiếp $(O)$ nên $\widehat{FBM}=\widehat{ACM}$. \hfill (1)\\
Xét hai tam giác vuông $FBM$ và $ECM$, ta có:
$\begin{cases}
\widehat{FBM}+\widehat{FMB}=90^\circ\\
\widehat{CME}+\widehat{ECM}=90^\circ
\end{cases}$.  \hfill (2)\\
Từ (1) và (2) suy ra $\widehat{FMB}=\widehat{CME}$. \hfill (iii)\\
Từ (i), (ii) và (iii) suy ra $\widehat{FDB}=\widehat{CDE}$.\\
$\Rightarrow E,F,D$ thẳng hàng (đpcm).
\item Ta có:
\begin{align*}
\dfrac{AC}{ME}+\dfrac{AB}{MF}=&\dfrac{AE-EC}{ME}+\dfrac{AF+FB}{MF}\\
=&\dfrac{AE}{ME}-\dfrac{EC}{ME}+\dfrac{AF}{MF}+\dfrac{FB}{MF}\\
=&\tan \widehat{AME}-\tan \widehat{CME}+\tan \widehat{AMF}+\tan \widehat{FMB} \\
=&\tan \widehat{AME}+\tan \widehat{AMF}\; \text{(cmt câu b)}\\
=&\tan \widehat{BMD}+\tan \widehat{MDC}\; \text{(do tứ giác $ABMC$ nội tiếp)}\\
=&\dfrac{BD}{MD}+\dfrac{CD}{MD}&\\
=&\dfrac{BC}{MD}\; (\text{đpcm}).
\end{align*}
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Bình Định, 2017]%[9D5G1]
    Cho $a,b,c$ là các số thực dương. Chứng minh rằng:
    $$\dfrac{a^5}{bc}+\dfrac{b^5}{ca}+\dfrac{c^5}{ab}\ge a^3+b^3+c^3$$
\loigiai{
Áp dụng bất đẳng thức Cauchy cho các dãy $5$ số thực dương $\left\{\dfrac{a^5}{bc};\dfrac{a^5}{bc};\dfrac{a^5}{bc};b^3;c^3\right\}$; $\left\{\dfrac{b^5}{ac};\dfrac{b^5}{ac};\dfrac{b^5}{ac};a^3;c^3\right\}$ và $\left\{\dfrac{c^5}{ab};\dfrac{c^5}{ab};\dfrac{c^5}{ab};a^3;b^3\right\}$, ta có:
    $$\dfrac{a^5}{bc}+\dfrac{a^5}{bc}+\dfrac{a^5}{bc}+b^3+c^3\geq 5a^3$$
    $$\dfrac{b^5}{ac}+\dfrac{b^5}{ac}+\dfrac{b^5}{ac}+a^3+c^3\geq 5b^3$$
    $$\dfrac{c^5}{ab}+\dfrac{c^5}{ab}+\dfrac{c^5}{ab}+a^3+b^3\geq 5c^3$$
    Suy ra: $3\left(\dfrac{a^5}{bc}+\dfrac{b^5}{ca}+\dfrac{c^5}{ab}\right)+2\left(a^3+b^3+c^3\right)\geq 5\left(a^3+b^3+c^3\right)\Rightarrow \dfrac{a^5}{bc}+\dfrac{b^5}{ca}+\dfrac{c^5}{ab}\ge a^3+b^3+c^3$.
 }
\end{ex}
