\begin{ex}%[Đề thi HSG 9, thành phố Hà Nội, năm 2018]%[Nguyễn Hữu Nhân, dự án EX-9G-18]%[9D0B1]
	Cho các số thực $a$, $b$, $c$ thỏa mãn $a+b+c=2018$ và $\dfrac{1}{b+c}+\dfrac{1}{c+a}+\dfrac{1}{a+b}=\dfrac{2017}{2018}$\\
	Tính giá trị biểu thức $P=\dfrac{a}{b+c}+\dfrac{b}{c+a}+\dfrac{c}{a+b}.$
	\loigiai{
	Ta có \begin{align*}
	P+3&= \left( \dfrac{a}{b+c} +1 \right) + \left( \dfrac{b}{c+a} +1 \right) + \left( \dfrac{c}{a+b} +1 \right) \\
	&= \dfrac{a+b+c}{b+c} + \dfrac{a+b+c}{c+a} + \dfrac{a+b+c}{a+b} \\
	& = \left( a+b+c \right) \left( \dfrac{1}{b+c} + \dfrac{1}{c+a} + \dfrac{1}{a+b} \right) \\
	&= 2018 \cdot \dfrac{2017}{2018} \\
	&=2017.
	\end{align*}
Vậy $P=2017-3=2014$.}
\end{ex}

\begin{ex}%[Đề thi HSG 9, thành phố Hà Nội, năm 2018]%[Nguyễn Hữu Nhân, dự án EX-9G-18]%[9D0G5]
	Tìm tất cả các cặp số nguyên $(x,y)$ thỏa mãn $\dfrac{x-y}{x^2+xy+y^2}=\dfrac{7}{13}$.
	\loigiai{Điều kiện: $x^2+y^2 \neq 0$.\\
	Phương trình đã cho tương đương với 
	\begin{equation}\label{1814.1}
	13(x-y)=7(x^2+xy+y^2) \Leftrightarrow 7x^2 +(7y-13)x +7y^2+13y=0. 
	\end{equation}
	Ta coi phương trình $(1)$ là phương trình bậc hai ẩn $x$ với tham số $y$. 
	\\ Ta có $\Delta =(7y-13)^2- 4 \cdot 7 (7y^2+13y) =13^2 -147y^2 -546y = 26^2 -3(7y+13)^2$.
	\\ Để phương trình $(1)$ có nghiệm thì 
	\begin{align*}
	\Delta \geq 0 &\Leftrightarrow 26^2 -3(7y+13)^2 \geq 0 \\
	& \Leftrightarrow  26^2 \geq 3 (7y+13)^2 \\
	& \Leftrightarrow  \dfrac{26}{\sqrt{3}} \geq 7y+13 \geq -\dfrac{26}{\sqrt{3}} \\
	& \Leftrightarrow  \dfrac{1}{7} \left( \dfrac{26}{\sqrt{3}} -13 \right) \geq y \geq - \dfrac{1}{7} \left( \dfrac{26}{\sqrt{3}} +13\right).
	\end{align*}
Suy ra $0 \geq y \geq -4$.
\begin{enumerate}
	\item Nếu $y=0$ thì $\Delta =169$. Phương trình \eqref{1814.1} có nghiệm $x=0$ (loại) và $x=\dfrac{13}{7}$ (loại).
	\item Nếu $y=-1$ thì $\Delta =568$ không phải số chính phương nên phương trình $(1)$ không có nghiệm nguyên.
	\item Nếu $y=-2$ thì $\Delta =673$ không phải số chính phương nên phương trình $(1)$ không có nghiệm nguyên.
	\item Nếu $y=-3$ thì $\Delta =484$. Phương trình \eqref{1814.1} có nghiệm $x=4$ (nhận) và $x=\dfrac{6}{7}$ (loại).
	\item Nếu $y=-4$ thì $\Delta =1$. Phương trình \eqref{1814.1} có nghiệm $x=3$ (nhận) và $x=\dfrac{20}{7}$ (loại).
\end{enumerate}
Vậy phương trình đã cho có các nghiệm nguyên $(x;y)$ là $(-3;4)$ và $(-4;3)$.} 
\end{ex}

\begin{ex}%[Đề thi HSG 9, thành phố Hà Nội, năm 2018]%[Nguyễn Hữu Nhân, dự án EX-9G-18]%[9D0K2]
Giải phương trình $6x^2+2x+1=3x\sqrt{6x+3}.$
\loigiai{
Điều kiện: $6x+3 \geq 0 \Leftrightarrow x \geq -\dfrac{1}{2}$.
\\ Nếu $x = -\dfrac{1}{2}$ thì phương trình đã cho trở thành $6\left( -\dfrac{1}{2}\right)^2=0$ (vô lí).
\\ Nếu $x > -\dfrac{1}{2}$ thì $2x+1>0$. Chia cả hai vế của phương trình cho $2x+1$ ta có
\begin{eqnarray*}
	6\left( \dfrac{x}{\sqrt{2x+1}}\right)^2 +1= \dfrac{3\sqrt{3}x}{\sqrt{2x+1}} &\Leftrightarrow& 6\left( \dfrac{x}{\sqrt{2x+1}}\right)^2 - \dfrac{3\sqrt{3}x}{\sqrt{2x+1}}+1 =0 \\ &\Leftrightarrow & \hoac{& \dfrac{x}{\sqrt{2x+1}} = \dfrac{1}{\sqrt{3}} \\ & \dfrac{x}{\sqrt{2x+1}} = \dfrac{1}{2\sqrt{3}}.}
\end{eqnarray*}
TH1: Ta có $\dfrac{x}{\sqrt{2x+1}} =\dfrac{1}{\sqrt{3}} \Leftrightarrow \sqrt{3}x =\sqrt{2x+1} \Leftrightarrow \heva{&x >0 \\ &3x^2 =2x+1} \Leftrightarrow \heva{&x>0 \\ &\hoac{&x=1 \text{ (nhận)}\\ &x=-3 \text{ (loại).}}}$ \\
TH2: Ta có $\dfrac{x}{\sqrt{2x+1}} =\dfrac{1}{2\sqrt{3}} \Leftrightarrow 2\sqrt{3}x =\sqrt{2x+1} \Leftrightarrow \heva{ &x>0 \\ &12x^2=2x+1} \Leftrightarrow \heva{&x>0 \\ &\hoac{&x= \dfrac{1+\sqrt{13}}{12} \text{ (nhận)} \\ &x=\dfrac{1-\sqrt{13}}{12} \text{ (loại).}}} $
\\ Vậy phương trình có tập nghiệm $S= \left\{ 1; \dfrac{1+\sqrt{13}}{12} \right\}$.}
\end{ex}

\begin{ex}%[Đề thi HSG 9, thành phố Hà Nội, năm 2018]%[Nguyễn Hữu Nhân, dự án EX-9G-18]%[9D0G2]
Giải hệ phương trình $\heva{&x^3+x+2=y^3-3y^2+4y \\& 2\sqrt{x+2}=y+2.}$
\loigiai{Điều kiện: $\heva{&x+2 \geq 0 \\ &y+2\geq 0} \Leftrightarrow \heva{&x \geq -2 \\ &y \geq -2.}$\\
Phương trình đầu tiên tương đương với 
\begin{eqnarray*}
	x^3+x+2 = (y-1)^3 + (y-1)+2 &\Leftrightarrow& \left( x^3 - (y-1)^3 \right) +x - (y-1) =0 \\ &\Leftrightarrow & (x-y+1) \left( x^2 +x(y-1)+(y-1)^2+1 \right)=0 \\
	&\Leftrightarrow & x-y+1=0 \text{ (do } x^2 +x(y-1)+(y-1)^2+1 >0).
\end{eqnarray*}
Vậy $x=y-1$. Thay vào phương trình thứ hai ta có 
\begin{eqnarray*}
	2\sqrt{y+1} =y+2 \Leftrightarrow \heva{&y+2 \geq 0 \\ &4(y+1) =(y+2)^2}
	\Leftrightarrow  \heva{& y\geq -2 \\ &y^2 =0} 
	\Leftrightarrow \heva{&y\geq -2 \\ &y=0 \text{ (nhận)}.}
\end{eqnarray*}
Với $y=0$, ta có $x=0-1=-1$. Vậy hệ phương trình có nghiệm $(x;y)=(-1;0)$.}
\end{ex}

\begin{ex}%[Đề thi HSG 9, thành phố Hà Nội, năm 2018]%[Nguyễn Hữu Nhân, dự án EX-9G-18]%[9D0G4]
	Cho $p$ là số nguyên tố. Chứng minh rằng không tồn tại cặp số nguyên dương $(m,n)$ thỏa mãn $m^{2019}+n^{2019}=p^{2018}.$
	\loigiai{ \\
	\textbf{Bổ đề:} Cho $a,b$ nguyên dương và $\gcd (a,b) =1$. Khi đó $\gcd \left( a-b, \dfrac{a^n-b^n}{a-b}\right) =\gcd (a-b,n)$.
	\centerline{Chứng minh}
	\\ Ta có \begin{eqnarray*}
	\dfrac{a^n-b^n}{a-b} &=& a^{n-1} +a^{n-2}b +\ldots +ab^{n-2}+b^{n-1} \\
	&=& \left(a^{n-1} -b^{n-1}\right) +\left(a^{n-2}b - b^{n-1} \right) + \ldots + \left( ab^{n-2} -b^{n-1} \right) + nb^{n-1}.
	\end{eqnarray*}
	Chú ý $ \left( a^{k}-b^{k} \right) ~ \vdots ~ (a-b)$ với mọi $k \in \mathbb{N}^*$. Do đó ta có thể viết $\dfrac{a^n-b^n}{a-b}= (a-b)A +nb^{n-1}$.
	\\ Vậy $\gcd \left( a-b, \dfrac{a^n-b^n}{a-b}\right) = \gcd \left( a-b, (a-b)A +nb^{n-1} \right) =\gcd \left( a-b, nb^{n-1}\right) =\gcd \left( a-b, n\right).$
	\\ \textbf{Quay lại bài toán.}
	\\ Giả sử tồn tại $m,n$ nguyên dương $(m \geq n)$ sao cho $m^{2019} + n^{2019} =p^{2018}$.
	\\ Đặt $d = \gcd (m,n)$. Khi đó $p^{2018}$ chia hết cho $d$ nên $d=p^{k}$ với $k \in \mathbb{N}$.
	\\ Nếu $k \geq 1$ thì $m,n \geq p$ và do đó $m^{2019} + n^{2019} \geq 2p^{2019} >p^{2018}$ (vô lí). Vậy $d=p^0=1$.
	\\ Ta có $p^{2018} = m^{2019} + n^{2019} = (m+n) \left( \dfrac{m^{2019} + n^{2019}}{m+n}\right)$. 
	\\ Do mỗi số tự nhiên lớn hơn $1$ có duy nhất một cách phân tích thành tích các lũy thừa của các số nguyên tố nên $m+n =p^r$ với $r \in \mathbb{N}^*$.
	\\ TH1: Nếu $m+n \geq 2p$ thì $m \geq p$. Do đó $m^{2019} + n^{2019} \geq p^{2019} > p^{2018}$ (vô lí).
	\\ TH2: Nếu $m+n=p$ thì 
	\begin{align*}
		(m+n)^{2018} - n^{2018} = p^{2018} - n^{2018} = m^{2019} +n^{2019} -n^{2018}=m^{2019} + n^{2018}(n-1).
	\end{align*}
	Ta có $(m+n)^{2018} - n^{2018}$ và $m^{2019}$ chia hết cho $m$ nên $n^{2018}(n-1)$ cũng chia hết cho $m$.
	\\ Mặt khác $\gcd (m,n) =1$ nên $(n-1) ~ \vdots ~ m$. Mà $n-1<m$ nên $n=1$. Vậy $m=p-1$ và $n=1$. 
	\\ Nếu $p=2$ thì ta có $1^{2019}+1^{2019} =2^{2018}$ (vô lí). Do đó $p$ là số nguyên tố lớn hơn hoặc bằng $3$.
	\\ Khi đó \begin{align}
	(p-1)^{2019} + 1=p^{2018} &\Leftrightarrow \left( (p-1)^{3} \right) ^{673} +1 = p^{2018} \notag \\
	&\Leftrightarrow \left( (p-1)^3+ 1\right) \left( \dfrac{(p-1)^{2019} +1 }{(p-1)^3 +1}\right) =p^{2018} \notag \\
	&\Leftrightarrow  \left( p^3-3p^2+3p \right) \left( \dfrac{(p-1)^{2019} +1 }{(p-1)^3 +1}\right) =p^{2018}.\label{1814.2}
	\end{align}	
	Ta có $p^2 \leq p^3 -3p^2 +3p < p^3$ với mọi $ p \geq 3$. 
	\\Mặt khác $p^3-3p^2+3p$ có dạng $p^{s}$ ($s \in \mathbb{N}^{*}$) nên $p^3-3p^2+3p=p^2$ và do đó $p=3$.
	\\ Thay vào \eqref{1814.2} ta có $\dfrac{(p-1)^{2019} +1 }{(p-1)^3 +1} = p^{2016}$.
	\\ Suy ra $\gcd \left( p^3-3p^2+3p , \dfrac{(p-1)^{2019} +1 }{(p-1)^3 +1}\right) =\gcd \left( p^2, p^{2016}\right) =p^2. \refstepcounter{equation}\hfill (\theequation)\label{1814.3}$
	\\ Mặt khác, áp dụng bổ đề ta có 
	\begin{equation}\label{1814.4}
		\gcd \left( p^3-3p^2+3p , \dfrac{(p-1)^{2019} +1 }{(p-1)^3 +1}\right) =\gcd \left( p^3-3p^2+3p , 673\right)=\gcd (9,673)=1.  
	\end{equation}
	Từ \eqref{1814.3} và \eqref{1814.4}, ta có mâu thuẫn. Do đó không tồn tại $m,n$ nguyên dương thỏa mãn đề bài.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, thành phố Hà Nội, năm 2018]%[Nguyễn Hữu Nhân, dự án EX-9G-18]%[9D0G7]
	Xét ba số không âm $x$, $y$, $z$ thỏa mãn $x+y+z=3$ và $x \le y \le z$. Tìm giá trị nhỏ nhất của biểu thức $P=\dfrac{x}{y^3+16}+\dfrac{y}{z^3+16}+\dfrac{z}{x^3+16}.$
	\loigiai{Ta có \begin{align}
		3-16P &= \left( x- \dfrac{16x}{y^3+16}\right) +\left( y- \dfrac{16y}{z^3+16}\right) + \left( z-\dfrac{16z}{x^3+16}\right) \notag \\
		& = \dfrac{xy^3}{y^3+16} + \dfrac{yz^3}{z^3+16} + \dfrac{zx^3}{x^3+16}.  \label{1814.5}
		\end{align}
	Ta có $y^3+16=(y+4)(y-2)^2+12y \geq 12y$ nên $\dfrac{y}{y^3+16} \leq \dfrac{1}{12}$. Do đó $\dfrac{xy^3}{y^3+16} \leq \dfrac{xy^2}{12}$.
	\\ Tương tự ta có: $\dfrac{yz^3}{z^3+16} \leq \dfrac{yz^2}{12}$, $\dfrac{zx^3}{x^3+16} \leq \dfrac{zx^2}{12}$.
	\\ Suy ra 
	\begin{equation}
	\dfrac{xy^3}{y^3+16} + \dfrac{yz^3}{z^3+16} + \dfrac{zx^3}{x^3+16} \leq \dfrac{xy^2+yz^2+zx^2}{12}. \label{1814.6}
	\end{equation}
	Do $x\leq y \leq z$ nên ta có $(y-z)(y-x) \leq 0 \Leftrightarrow y^2+zx \leq xy+yz \Rightarrow xy^2 +zx^2 \leq x^2y +xyz.$
	Từ đó ta có \begin{align}
	xy^2+yz^2+zx^2 &\leq x^2y +yz^2+xyz = y \left( x^2+z^2+xz\right) \notag \\
	&\leq  y(x+z)^2 = y(3-y)^2 \notag \\
	&\leq  \dfrac{1}{2} \left( \dfrac{2y+3-y+3-y}{3}\right)^3 =4. \label{1814.7}
	\end{align}
Từ \eqref{1814.5}, \eqref{1814.6} và \eqref{1814.7} ta có $3-16P \leq \dfrac{4}{12} \Leftrightarrow P \geq \dfrac{1}{6}.$
\\ Dấu đẳng thức xảy ra khi $x=0,y=1,z=2$. Vậy $\min P=\dfrac{1}{6}$.
	}
\end{ex}

	\item Ta đánh số các đỉnh của đa giác đều $20$ cạnh tăng dần từ $1$ đến $2n$, theo chiều kim đồng hồ.
	\\ Với mỗi đoạn thẳng tạo thành từ hai đỉnh $i$ và $j$ $(i<j)$, ta quy ước một số đại diện cho độ dài của đoạn thẳng đó như sau 
	$$\mathrm{d}(i,j) = \heva{&j-i \text{ nếu } 0<j-i\leq 10 \\ & 20 - (j-i) \text{ nếu }10< j-i <20. } $$
	Khi đó, hai đoạn thẳng có độ dài hình học thông thường bằng nhau khi và chỉ khi độ dài theo quy ước trên bằng nhau.
	\\ Giả sử tồn tại một cách chia $20$ đỉnh thành $10$ cặp điểm $(a_1,b_1)$, $(a_2,b_2)$,..., $(a_{10},b_{10})$ (với $a_i<b_i$) thỏa mãn đề bài và trong số $10$ đoạn thẳng được tạo ra không có hai đoạn thẳng nào bằng nhau.
	\\ Không mất tổng quát, ta giả sử các cặp điểm $(a_1,b_1), \ldots , (a_l,b_l)$ thỏa mãn $\mathrm{d}(a_i,b_i)=b_i-a_i$ và các cặp điểm $(a_{l+1},b_{l+1}), \ldots , (a_{10},b_{10})$ thỏa mãn $\mathrm{d}(a_i,b_i)=20-(b_i-a_i)$.
	\\ Do các đoạn thẳng được tạo thành có độ dài đôi một khác nhau nên ta có \begin{eqnarray*}
	\mathrm{d}(a_1,b_1) + \mathrm{d}(a_2,b_2) + \ldots +\mathrm{d}(a_{10},b_{10}) = 1+2+\ldots 10=55.
	\end{eqnarray*}
	Ta có 
	\begin{eqnarray*}
	\mathrm{d}(a_1,b_1)+\ldots +\mathrm{d}(a_l,b_l) &=&(b_1+\ldots +b_l) - (a_1+\ldots +a_l) \\\mathrm{d}(a_{l+1}+b_{l+1}) +\ldots +\mathrm{d}(a_{10}+b_{10}) &=& 20(10-l) + (a_{l+1}+\ldots +a_{10}) -(b_{l+1}+\ldots +b_{10}).
	\end{eqnarray*}
Suy ra \begin{eqnarray*}
55 &=& 20(10-l) +\left(b_1+\ldots+b_l +a_{l+1}+\ldots+a_{10}\right) - \left( a_1+\ldots +a_l+ b_{l+1}+\ldots +b_{10}\right) \\
&=&20(10-l) + \left(a_1+\ldots +a_{10}+b_1 +\ldots +b_{10}\right) -2 \left( a_1+\ldots +a_l+ b_{l+1}+\ldots +b_{10}\right) \\
&=& 20(10-l) + \left(1+2+\ldots +20\right) -2 \left( a_1+\ldots +a_l+ b_{l+1}+\ldots +b_{10}\right) \\
&=& 20(10-l) +210 -2 \left( a_1+\ldots +a_l+ b_{l+1}+\ldots +b_{10}\right).
\end{eqnarray*}
Ta thấy vế phải là số chẵn trong khi $55$ là số lẻ nên giả sử ban đầu sai. Do đó trong mười đoạn thẳng được tạo ra luôn tồn tại hai đoạn thẳng bằng nhau.
\end{enumerate}}
