\begin{bt}%[Thi GHK, Lương Văn Chánh - Phú Yen, 2018]%[Huỳnh Xuân Tín, EX-DT-10&11]%[1D1B1-1]
	Tìm tập xác định của các hàm số sau
	\begin{enumerate}[a)]
\item $y=\sqrt{2-\sin 2x}$.
\item $y=\dfrac{1+\cot x}{\cos x}$.
\end{enumerate}
	\loigiai{
	\begin{enumerate}[a)]
		\item $y=\sqrt{2-\sin 2x}$.\\
	Vì	$-1\le \sin 2x\le 1$ nên $ 2-\sin 2x>0,\forall x\in \mathbb{R}.$\\
	Vậy hàm số có tập xác định là $\mathscr{D}=\mathbb{R}$.
		\item $y=\dfrac{1+\cot x}{\cos x}$.\\
		Điều kiện xác định của hàm số trên $\heva{\sin x&\not=&0\\ \cos x&\not=&0}\Leftrightarrow x\not= k\dfrac{\pi}{2}$.\\
		Vậy tập xác định của hàm số là $\mathscr{D}=\mathbb{R}\setminus \left\lbrace k\dfrac{\pi}{2}, k\in \mathbb{Z} \right\rbrace $.
	\end{enumerate}	
	}
\end{bt}

\begin{bt}%[Thi GHK, Lương Văn Chánh - Phú Yen, 2018]%[Huỳnh Xuân Tín, EX-DT-10&11]%[1D1B1-5]%[1D1K1-5]
	Tìm giá trị lớn nhất, giá trị nhỏ nhất của các hàm số sau
	\begin{enumerate}[a)]
		\item $y=1+2\cos x$.
		\item $y=2\sin^2\dfrac{x}{2}+2\sin\dfrac{x}{2}\cos\dfrac{x}{2}$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}[a)]
		\item $y=1+2\cos x$ có tập xác định là $\mathscr{D}=\mathbb{R}$.\\
		Từ $-1\le \cos x\le 1$ hay  $-2\le 2\cos x\le 2$ suy ra $-1\le 1+2\cos x\le 3$.\\
	Vậy giá trị nhỏ nhất của hàm số là $-1$ khi $\cos x=-1$ hay $x=\pi+k2\pi$.\\ Giá trị lớn nhất của hàm số là $3$ khi $\cos x=1$ hay $x=k2\pi$	.
		\item $y=2\sin^2\dfrac{x}{2}+2\sin\dfrac{x}{2}\cos\dfrac{x}{2}$, có tập xác định là $\mathscr{D}=\mathbb{R}$.\\
	Ta có $y=2\sin^2\dfrac{x}{2}+2\sin\dfrac{x}{2}\cos\dfrac{x}{2}=\sin x-\cos x+1=\sqrt{2}\sin \left( x-\dfrac{\pi }{4} \right)+1$.\\
	Ta có $-1\le \sin\left( x-\dfrac{\pi }{4} \right)\le 1\Rightarrow -\sqrt{2}\le \sqrt{2}\sin\left( x-\dfrac{\pi }{4} \right)\le \sqrt{2}\Rightarrow 1-\sqrt{2}\le y\le 1+\sqrt{2}$.\\
	Vậy giá trị nhỏ nhất của hàm số là $1-\sqrt{2}$ khi $\sin\left( x-\dfrac{\pi }{4} \right)=-1$ hay $x=-\dfrac{\pi }{4}+k2\pi$\\ 
	Giá trị lớn nhất của hàm số là $1+\sqrt{2}$ khi $\sin\left( x-\dfrac{\pi }{4} \right)=1$ hay  $x=\dfrac{3\pi }{4}+k2\pi$.
	\end{enumerate}	
	}
\end{bt}

\begin{bt}%[Thi GHK, Lương Văn Chánh - Phú Yen, 2018]%[Huỳnh Xuân Tín, EX-DT-10&11]%[1D1K3-3]%[1D1K3-2]
	Giải các phương trình sau
	\begin{enumerate}[a)]
	\item $\sin2x+\sqrt{3}\cos2x=2\sin x$.
	\item $\sin^2x-2\sin x\cos x- 3\cos^2x=0$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}[a)]
		\item Ta có
		\begin{eqnarray*}
			& & \sin2x+\sqrt{3}\cos2x=2\sin x\\
			&\Leftrightarrow & \dfrac{1}{2}\sin 2x+\dfrac{\sqrt{3}}{2}\cos 2x=\sin x\\
			&\Leftrightarrow & \sin \left( 2x+\dfrac{\pi }{3} \right)=\sin x\\
			&\Leftrightarrow & \hoac{2x+\dfrac{\pi }{3}&=&x+k2\pi\\ 2x+\dfrac{\pi }{3}&=&\pi -x+k2\pi} \,\,\,\left( k\in \mathbb{Z} \right)\\
			&\Leftrightarrow & \hoac{x&=&-\dfrac{\pi }{3}+k2\pi \\ x&=&\dfrac{2\pi }{9}+k\dfrac{2\pi }{3}}\,\,\,\left( k\in \mathbb{Z} \right).
		\end{eqnarray*}
	Vậy nghiệm của phương trình là $x=-\dfrac{\pi }{3}+k2\pi$, $x=\dfrac{2\pi }{9}+k\dfrac{2\pi }{3}$ với $k\in \mathbb{Z}$.
		\item Ta xét hai trường hợp
		
		\begin{enumerate}[TH 1.]
\item Với $\cos x=0$ không thỏa mãn phương trình, vậy $cos x\not=0$.
\item Với $cos x\not=0$. Chia hai vế phương trình cho $\cos^2x$ ta được	
\begin{eqnarray*}
	& & \tan ^2x-2\tan x-3=0\\
	&\Leftrightarrow & \hoac{\tan x&=&-1\\ \tan x&=&3}\\
	&\Leftrightarrow & \hoac{x&=&-\dfrac{\pi }{4}+k\pi \\ x&=&\arctan 3+k\pi} \,\,\,\,\left( k\in \mathbb{Z} \right).
\end{eqnarray*}\end{enumerate}	
Vậy nghiệm của phương trình là $x=-\dfrac{\pi }{4}+k\pi$, $x=\arctan 3+k\pi$ với $k\in \mathbb{Z}$.	\end{enumerate}	}
\end{bt}


\begin{bt}%[Thi GHK, Lương Văn Chánh - Phú Yen, 2018]%[Huỳnh Xuân Tín, EX-DT-10&11]%[1D1B3-1]%[1D1K3-7]
	Cho phương trình $\cos^2x-(m-2)\cos x-2m=0. \quad (1)$
	\begin{enumerate}[a)]
		\item Giải phương trình (1) với $m=1$.
		\item Tìm $m$ để phương trình (1) có nghiệm $x\in \left(\dfrac{\pi}{3};\dfrac{2\pi}{3}\right) $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
		\item Với $m=1$, Phương trình đã cho trở thành
		\begin{eqnarray*}
		& & \cos ^2x+\cos x-2=0\\\
		&\Leftrightarrow & \hoac{\cos x&=&1\\ \cos x&=&-2}
	\end{eqnarray*}
	Vì $-1\le\cos x\le 1$ nên $\cos x=1$ hay $x=k2\pi,\,\,\,\,\,\,\left( k\in \mathbb{Z} \right)$.
		\item Ta có $\cos^2x-(m-2)\cos x-2m=0\Leftrightarrow \hoac{&\cos x&=m\\&\cos x&=-2\,\,\, \left( \text{loại}\right) .}$\\
		Từ $x\in \left(\dfrac{\pi}{3};\dfrac{2\pi}{3}\right)\Rightarrow -\dfrac{1}{2}<\cos x <\dfrac{1}{2}\Rightarrow -\dfrac{1}{2}<m <\dfrac{1}{2}$.\\
		Vậy giá trị $m$ cần tìm là $-\dfrac{1}{2}<m <\dfrac{1}{2}$.
	\end{enumerate}	
	}
\end{bt}
