	Tìm hệ số chứa $x^{10}$ trong khai triển $f(x)=\left(\dfrac{1}{4}x^2+x+1\right)^2\left(x+2\right)^{3n}$ với $n$ là số tự nhiên thỏa mãn hệ thức $\mathrm{A}_n^3+\mathrm{C}_n^{n-2}=14n$.
	\choice
	{\True $2^5\mathrm{C}_{19}^{10}$}
	{$2^5\mathrm{C}_{19}^{10}x^{10}$}
	{$2^9\mathrm{C}_{19}^{10} $}
	{$2^9\mathrm{C}_{19}^{10}x^{10} $}
	\loigiai{
		Ta có phương trình
		\begin{align*}
		&\mathrm{A}_n^3+\mathrm{C}_n^{n-2}=14n\\
		\Leftrightarrow &\dfrac{n!}{\left(n-3\right)!}+\dfrac{n!}{2!\left(n-2\right)!}=14n\\
		\Leftrightarrow &n\left(n-1\right)\left(n-2\right)+\dfrac{{n\left(n-1\right)}}{2}=14n\\
		\Leftrightarrow &n\left(2n^2-5n-25\right)=0\\
		\Leftrightarrow &\hoac{&n=5\\&n=0 \text{ (loại)}\\&n=-\dfrac{5}{2}  \text{ (loại)}.}
		\end{align*}
