\textbf{ĐỀ CHẴN}

\begin{bt}%[Thi GHK, Đoàn Thượng - Hải Dương, 2018]%[Đỗ Đường Hiếu, EX-DT-10&11]%[0D1B1-2]%[0D1B1-3]
	Xét tính đúng, sai và lập mệnh đề phủ định của mỗi mệnh đề sau:
	\begin{enumEX}{1}
		\item $\forall n\in\mathbb{N}\colon 4n^2$ chia hết cho $n$.
		\item $\forall x\in\mathbb{R}\colon x^2-6x+10>0$.
		\item $\exists x\in\mathbb{Q}\colon x^2-7\ne 7x$.
		\item Tổng ba góc trong của một tam giác bằng $180^\circ$.
		\item $\sqrt{9}$ là số vô tỉ.
		\item Paris là thủ đô của nước Pháp.
	\end{enumEX}
	\loigiai{
	\begin{enumEX}{1}
		\item Ta có $4n^2:n=4n$ nên mệnh đề \lq\lq$\forall n\in\mathbb{N}\colon 4n^2$ chia hết cho $n$\rq\rq\, là mệnh đề đúng.\\
		Mệnh đề phủ định của nó là: \lq\lq$\exists n\in\mathbb{N}\colon 4n^2$ không chia hết cho $n$\rq\rq.
		\item Ta có $x^2-6x+10=(x-3)^2+1>0, \forall x\in\mathbb{R}$ nên mệnh đề \lq\lq$\forall x\in\mathbb{R}\colon x^2-6x+10>0$\rq\rq\, là mệnh đề đúng.\\
		Mệnh đề phủ định của nó là: \lq\lq$\exists  x\in\mathbb{R}\colon x^2-6x+10\le 0$\rq\rq.
		\item Với $x=0\in\mathbb{Q}$ ta có $0^2-7\ne 7\cdot 0$ nên mệnh đề \lq\lq$\exists x\in\mathbb{Q}\colon x^2-7\ne 7x$\rq\rq\, là mệnh đề đúng.\\
		Mệnh đề phủ định của nó là:  \lq\lq$\forall x\in\mathbb{Q}\colon x^2-7= 7x$\rq\rq.
		\item Mệnh đề \lq\lq Tổng ba góc trong của một tam giác bằng $180^\circ$\rq\rq \, là mệnh đề đúng.\\
		Mệnh đề phủ định của nó là: \lq\lq Tổng ba góc trong của một tam giác không bằng $180^\circ$\rq\rq.
		\item Mệnh đề \lq\lq $\sqrt{9}$ là số vô tỉ\rq\rq \, là mệnh đề sai.\\
		Mệnh đề phủ định của nó là: \lq\lq $\sqrt{9}$ là số hữu tỉ\rq\rq.
		\item Mệnh đề \lq\lq Paris là thủ đô của nước Pháp\rq\rq \, là mệnh đề đúng.\\
		Mệnh đề phủ định của nó là: \lq\lq Paris không phải là thủ đô của nước Pháp\rq\rq.
	\end{enumEX}}
\end{bt}



\begin{bt}%[Thi GHK, Đoàn Thượng - Hải Dương, 2018]%[Đỗ Đường Hiếu, EX-DT-10&11]%[0D1B3-1]%[0D1B3-2]
	Cho các tập hợp\\
	 $A=\{-3;5;6\}$; $B=\{x\in\mathbb{R}\colon x^2-4x-5=0\}$;
	$C=\{x\in\mathbb{N}\colon(x-2)(x^2+5x-6)=0\}$.
	\begin{listEX}[1]
		\item Viết tập hợp $B$ và $C$ dưới dạng liệt kê các phần tử. Tìm $A\cap B$; $A\cup C$.
		\item Tìm $(A\cup B)\setminus C$; $(A\setminus B)\cap C$.
	\end{listEX}
	\loigiai{
	\begin{itemize}
		\item Ta có
		\begin{itemize}
			\item $x^2-4x-5=0\Leftrightarrow \hoac{&x=-1\\&x=5}$. Do đó $B=\{-1;5\}$.
			\item $(x-2)(x^2+5x-6)=0\Leftrightarrow \hoac{&x=2\in\mathbb{N}\\&x=1\in\mathbb{N}\\&x=-6\notin\mathbb{N}}$. Do đó $C=\{1;2\}$.
		\end{itemize}
		Từ đó ta có $A\cap B=\{5\}$; $A\cup C=\{-3;1;2;5;6\}$.
		\item 
		\begin{itemize}
			\item Ta có $A\cup B=\{-3; -1; 5; 6\}$ nên $(A\cup B)\setminus C=\{-3; -1; 5; 6\}$.
			\item Ta có $A\setminus B=\{-3; 6\}$ nên  $(A\setminus B)\cap C=\varnothing$.
		\end{itemize}
	\end{itemize}}
\end{bt}

\begin{bt}%[Thi GHK, Đoàn Thượng - Hải Dương, 2018]%[Đỗ Đường Hiếu, EX-DT-10&11]%[0D1K4-1]
	Cho hai tập hợp $A=\left[a;a+1\right]$; $B=\left[b;b+2\right]$. Các số $a$ và $b$ thỏa mãn điều kiện gì để $A\cap B\ne\varnothing$?
	\loigiai{
	Ta có $A\cap B=\varnothing\Leftrightarrow \hoac{&a+1<b\\&b+2<a}\Leftrightarrow \hoac{&a<b-1\\&a>b+2}$.\\
	Do đó $A\cap B\ne\varnothing\Leftrightarrow \heva{&a\ge b-1\\&a\le b+2}$. Hay $b-1\le a\le b+2$.
	}
\end{bt}
