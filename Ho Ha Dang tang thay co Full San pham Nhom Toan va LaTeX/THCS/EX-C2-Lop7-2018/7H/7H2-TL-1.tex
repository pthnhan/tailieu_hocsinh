\section{TỔNG BA GÓC CỦA MỘT TAM GIÁC}
\subsection{Tóm tắt lý thuyết}
\subsubsection{Tổng ba góc của một tam giác}
Tổng ba góc của một tam giác bằng $180^\circ$.
\subsubsection{Áp dụng vào tam giác vuông}
\immini{\textbf{Tam giác vuông} là tam giác có một góc vuông.\\
Với $\triangle ABC$ có $\widehat{A}=90^\circ$, ta nói:
\begin{itemize}
	\item $\triangle ABC$ vuông tại $A$.
	\item $AB$, $AC$ gọi là các cạnh góc vuông và $BC$ là cạnh huyền.
	\item Các góc $\widehat{B}$, $\widehat{C}$ là hai góc nhọn của tam giác.
\end{itemize}}{\begin{tikzpicture}[scale=0.7]
\tkzDefPoints{0/0/A,3/0/B,0/5/C}
\tkzDrawSegments(A,C C,B B,A)
\tkzDrawPoints[fill=black](A,B,C)
\tkzLabelPoints[left](A,C)\tkzLabelPoints[right](B)
\tkzMarkRightAngle(C,A,B)
\end{tikzpicture}}
\begin{nx}
	Như vậy, nếu $\triangle ABC$ vuông tại $A$ ta sẽ nhận được $\widehat{A}=90^\circ$. Mặt khác, ta có: 
	\begin{eqnarray*}
		& &\widehat{A}+\widehat{B}+\widehat{C}=180^\circ\Leftrightarrow \widehat{B}+\widehat{C}=180^\circ-\widehat{A}=90^\circ\\
		& &\Leftrightarrow \widehat{B}\text{ và }\widehat{C}\text{ phụ nhau.}
	\end{eqnarray*}
	Vậy, ta có kết quả ``Trong tam giác vuông, hai góc nhọn phụ nhau''.
\end{nx}
\subsubsection{Góc ngoài của tam giác}
\textbf{Góc ngoài của một tam giác} là góc kề bù với một góc của tam giác ấy.\\
\begin{nx}
	\begin{itemize}
		\item Ta có: $$\widehat{xAB}+\widehat{A}=180^\circ\Leftrightarrow \widehat{xAB}=180^\circ-\widehat{A}.$$
		\immini{Mặt khác, ta có:
			$$\widehat{A}+\widehat{B}+\widehat{C}=180^\circ\Leftrightarrow \widehat{A}=180^\circ-\widehat{B}-\widehat{C}.$$
			Từ đó, suy ra: 
			$$\widehat{xAB}=180^\circ-(180^\circ-\widehat{B}-\widehat{C})=\widehat{B}+\widehat{C}.$$}{\begin{tikzpicture}[scale=0.6]
			\tkzDefPoints{0/0/A,5/0/C,2/2/B}
			\coordinate (x) at ($(C)!3/2!(A)$);
			\tkzDrawPoints[fill=black](A,B,C)
			\tkzDrawSegments(A,B B,C C,x)
			\tkzLabelPoints[above](x,B)\tkzLabelPoints[below](A,C)
			\end{tikzpicture}}
		\textit{Mỗi góc ngoài của một tam giác bằng tổng của hai góc trong không kề với nó.}
		\item Ta có: $$\widehat{xAB}=\widehat{B}+\widehat{C}\Rightarrow\widehat{xAB}>\widehat{B}\text{ và }\widetilde{xAB}>\widehat{C}.$$
		Từ đây, ta thu được nhận xét quan trọng ``\textit{Góc ngoài của một tam giác lớn hơn mỗi góc trong không kề với nó}''
	\end{itemize}
\end{nx}
\subsubsection{Cặp góc có cạnh tương ứng vuông góc}
Nếu hai góc có cạnh tương ứng vuông góc thì chúng bằng nhau hoặc bù nhau, cụ thể:
\begin{enumerate}	
	\item Chúng bằng nhau nếu hai góc cùng nhọn hoặc cùng tù.
	\item Chúng bù nhau nếu góc này nhọn và góc kia tù.
	\item Nếu một góc vuông thì góc còn lại cũng vuông.
\end{enumerate}

\begin{listEX}
	\item Ta có $\widehat{ADC}-\widehat{ADB}=\widehat{B}-\widehat{C}=2\widehat{C}$.\\
	Lại có $\widehat{ADC}+\widehat{ADB}=180^\circ$.\\
	Do đó $\widehat{ADC}=90^\circ+\widehat{C}$ suy ra $\widehat{ADE}=180^\circ-\widehat{ADC}=90^\circ-\widehat{C}$.\\
	Mà $\widehat{AED}=90^\circ-\widehat{ADE}=\widehat{C}$. Vậy tam giác $AEC$ là tam giá cân tại $A$.
	\item Ta có $\widehat{ADB}=90^\circ-\widehat{C}$, nên $\widehat{BAD}=180^\circ-\widehat{B}-\widehat{ADB}=90^\circ-2\widehat{C}$.\\
	Suy ra $\widehat{EAB}=90^\circ-\widehat{ADB}=2\widehat{C}$.\\
	Ta có $\widehat{EAC}=180^\circ-2\widehat{AEC}-\widehat{C}=180^\circ-2\widehat{C}$ suy ra $\widehat{yAE}=\dfrac{1}{2}\widehat{xAE}=\widehat{C}$.\\
	Vậy $\widehat{yAB}=3\widehat{C}=\widehat{B}$. Do đó $Ay\parallel BE$.\\
\end{listEX}
