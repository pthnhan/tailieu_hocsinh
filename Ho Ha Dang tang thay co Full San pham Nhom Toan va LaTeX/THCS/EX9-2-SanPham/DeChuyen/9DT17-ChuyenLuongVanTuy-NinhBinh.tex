\begin{ex}%[Đề thi vào 10, Chuyên Lương Văn Tuỵ, Ninh Bình, 2017]%[9D1B8]%[9D1B9]
Cho $P=\dfrac{3\sqrt{a}}{\sqrt{a}+2}+\dfrac{\sqrt{a}+1}{\sqrt{a}-2}+\dfrac{5\sqrt{a}+2}{4-a}\quad (a\ge 0,\,a\ne4)$.
\begin{enumerate}[a)]
\item Rút gọn biểu thức $P$;
\item Tính giá trị biểu thức $P$ khi $a=\sqrt[3]{1+\dfrac{\sqrt{84}}{9}}+\sqrt[3]{1-\dfrac{\sqrt{84}}{9}}$.
\end{enumerate}
\loigiai{\begin{enumerate}[a)]
\item $P=\dfrac{3a-6\sqrt{a}+a+3\sqrt{a}+2-5\sqrt{a}-2}{\left(\sqrt{a}+2\right)\left(\sqrt{a}-2\right)}=\dfrac{4a-8\sqrt{a}}{\left(\sqrt{a}+2\right)\left(\sqrt{a}-2\right)}=\dfrac{4\sqrt{a}}{\sqrt{a}+2}$.
\item Đặt $b=\sqrt[3]{1+\dfrac{\sqrt{84}}{9}}$ và $c=\sqrt[3]{1-\dfrac{\sqrt{84}}{9}}$. Ta có $a^3=b^3+c^3+3bc(b+c)$. Chú ý $b^3+c^3=2$ và $bc=-\dfrac{1}{3}$, ta suy ra $a^3=2-a$ hay $(a-1)(a^2+a+2)=0$. Do đó $a=1$. Thay vào biểu thức rút gọn của $P$ ta được $P=\dfrac{4}{3}$ khi $a=\sqrt[3]{1+\dfrac{\sqrt{84}}{9}}+\sqrt[3]{1-\dfrac{\sqrt{84}}{9}}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Lương Văn Tuỵ, Ninh Bình, 2017]%[9D5B5]%[9D5K5]
\begin{enumerate}[a)]
\item Giải phương trình $\left(\sqrt{x+4}-\sqrt{x-1}\right)\left(\sqrt{x^2+3x-4}+1\right)=5$.
\item Giải hệ phương trình $\begin{cases} x^3-3x=y^3+y&\qquad (1)\\ x^2=y^2+3 &\qquad (2)\end{cases}$.
\end{enumerate}
\loigiai{\begin{enumerate}[a)]
\item Điều kiện $x\ge 1$. Đặt $u=\sqrt{x+4}>0$ và $v=\sqrt{x-1}\ge 0$. Khi đó phương trình trở thành
\begin{equation*}
\begin{cases}(u-v)(uv+1)=5&\qquad (1) \\ u^2-v^2=5&\qquad (2) \end{cases}.
\end{equation*}
Thay $(2)$ vào $(1)$ với chú ý $u>v$ ta được $uv+1=u+v$ hay $u=1$ hoặc $v=1$. Tuy nhiên từ $(2)$, ta thấy $u^2>5>1$ nên $v=1$. Thay vào $(2)$ ta được $u=\sqrt{6}$. Do đó ta có
$$\heva{&\sqrt{x+4}=\sqrt{6}\\ &\sqrt{x-1}=1} \Leftrightarrow  x=2.$$
Vậy tập nghiệm của phương trình đã cho là $S=\{2\}$.

\item Dễ thấy với $y=0$ khi và chỉ khi $x^2-3=0$ hay $x=\pm \sqrt{3}$. Xét $y\ne 0$. Thay $x^2-3=y^2$ vào $(1)$ ta được $xy=y^2+1$ hay $x-y=\dfrac{1}{y}\quad (3)$. Thay $(3)$ vào $(2)$ với chú ý $(2)$ tương đương $(x-y)(x+y)=3$, ta được $x=2y$. Kết hợp $(3)$ ta được $y=\pm 1$. Do đó $x=\pm 2$.\\
Vậy hệ đã cho có bốn nghiệm $\left(\sqrt{3};0\right)$, $\left(-\sqrt{3};0\right)$, $\left(2;1\right)$ và $(-2;-1)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Lương Văn Tuỵ, Ninh Bình, 2017]%[9D1B8]%[9D5K2]
\begin{enumerate}[a)]
\item Cho các số hữu tỉ $a$, $b$, $c$ thoả mãn $ab+bc+ca=2017$. Chứng minh rằng
$$\sqrt{\left(a^2+2017\right)\left(b^2+2017\right)\left(c^2+2017\right)}$$
là một số hữu tỉ.
\item Tìm $x$, $y$ nguyên dương thoả mãn phương trình $7x^2+3y^2=714$.
\end{enumerate}

\loigiai{\begin{enumerate}[a)]
\item Ta có $a^2+2017=a^2+ab+bc+ca=(a+b)(a+c)$. Tương tự $b^2+2017=(b+c)(b+a)$ và $c^2+2017=(c+a)(c+b)$. Từ đó
$$\sqrt{\left(a^2+2017\right) \left(b^2+2017\right) \left(c^2+2017\right)} = \sqrt{(a+b)^2(b+c)^2(c+a)^2} = |a+b||b+c||c+a|$$
là một số hữu tỉ.

\item Giả sử $x$ và $y$ là hai số nguyên dương thoả mãn đề bài. Do $7\,|\,714$ và $7\,|\,7x^2$ nên $7\,|\,3y^2$. Chú ý $(3,7)=1$ nên $7\,|\,y^2$ hay $7\,|\,y$. Lại có $3y^2<714$ nên $y \le 15$. Do đó $y=7$ hoặc $y=14$.\\
Nếu $y=7$ thì $7x^2=714-3\cdot 7^2=567$, suy ra $x=9$.\\
Nếu $y=14$ thì $7x^2=714-3\cdot 14^2=126$, suy ra $x^2=18$ (vô lí).\\
Vậy có duy nhất một cặp số thoả mãn bài toán là $(x;y)=(9;7)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Lương Văn Tuỵ, Ninh Bình, 2017]%[9H3K4]%[9H3B7]%[9H3G4]
Cho hai đường tròn $(O)$ và $(O')$ cắt nhau tại hai điểm $A$, $B$ ($O$, $O'$ thuộc hai nửa mặt phẳng bờ $AB$). Tiếp tuyến chung gần $B$ của hai đường tròn lần lượt tiếp xúc với $(O)$ và $(O')$ tại $C$ và $D$. Qua $A$ kẻ đường thẳng song song với $CD$ lần lượt cắt $(O)$ và $(O')$ tại $M$ và $N$ ($M$, $N$ khác $A$). Các đường thẳng $CM$ và $DN$ cắt nhau tại $E$. Gọi $P$ và $Q$ lần lượt là giao của hai đường thẳng $MN$ với đường thẳng $BC$ và đường thẳng $BD$. Chứng minh rằng:
\begin{enumerate}[a)]
\item Đường thẳng $AE$ vuông góc đường thẳng $CD$.
\item Tứ giác $BCED$ nội tiếp.
\item Tam giác $EPQ$ là tam giác cân.
\end{enumerate}

\begin{enumerate}[a)]
\item Do $AM \parallel CD$ nên $\widehat{AMC}=\widehat{DCE}$ và $\widehat{MAC}=\widehat{ACD}$. \begin{ex}%[Đề thi vào 10, Chuyên Lương Văn Tuỵ, Ninh Bình, 2017]%[9D5G1]
Cho các số thực dương $a$, $b$, $c$ thoả mãn điều kiện $a+b+c=2018$. Tìm giá trị lớn nhất của biểu thức $P=\dfrac{a}{a+\sqrt{2018a+bc}}+\dfrac{b}{b+\sqrt{2018b+ca}}+\dfrac{c}{c+\sqrt{2018c+ab}}$.

\loigiai{Từ giả thiết và bất đẳng thức Cauchy - Schwartz, ta có
\begin{align*}
&2018a+bc=(a+b+c)a+bc=(a+b)(c+a)\ge \left(\sqrt{ab}+\sqrt{ac}\right)^2\\
\Rightarrow &\dfrac{a}{a+\sqrt{2018a+bc}}\le \dfrac{a}{a+\sqrt{ab}+\sqrt{ac}}=\dfrac{\sqrt{a}}{\sqrt{a}+\sqrt{b}+\sqrt{c}}
\end{align*}
Hoàn toàn tương tự ta có $\dfrac{b}{b+\sqrt{2018b+ca}}\le \dfrac{\sqrt{b}}{\sqrt{a}+\sqrt{b}+\sqrt{c}}
$, $\dfrac{c}{c+\sqrt{2018c+ab}} \le \dfrac{\sqrt{c}}{\sqrt{a}+\sqrt{b}+\sqrt{c}}
$.
Do đó $$P \le \dfrac{\sqrt{a}}{\sqrt{a}+\sqrt{b}+\sqrt{c}}
  +\dfrac{\sqrt{b}}{\sqrt{a}+\sqrt{b}+\sqrt{c}} + \le \dfrac{\sqrt{c}}{\sqrt{a}+\sqrt{b}+\sqrt{c}}=1.$$
Dấu bằng xảy ra khi $a=b=c$. Vậy $\max P=1$.
}
\end{ex}
