	Tìm $ x $, biết: $ \sqrt{16x + 32} + 2 = \sqrt{4x + 8} $.
	\loigiai{
		Điều kiện $ x \geqslant -2 $.\\
		Ta có
		\begin{eqnarray*}
			& &\sqrt{16x + 32} + 2 = \sqrt{4x + 8}\\
			& \Leftrightarrow & 4\sqrt{x + 2} + 2 = 2 \sqrt{x + 2}\\
			& \Leftrightarrow & 2\sqrt{x + 2} = - 2 \text{ (Vô lý).}
		\end{eqnarray*}
		Vậy phương trình đã cho vô nghiệm.
	}
\end{bt}

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Phú Nhuận, Thành phố Hồ Chí Minh]%[Lục Hán Tường, dự án 9DT19-HK1]%[9D2T2]
	Rừng ngập mặn Cần Giờ (còn gọi là Rừng Sác), trong chiến tranh bom đạn và chất độc hóa học đã làm nơi đây trở thành "vùng đất chết"; được trồng lại từ năm 1979, nay đã trở thành ``lá phổi xanh'' cho Thành phố Hồ Chí Minh, được UNESCO công nhận là khu dự trữ sinh quyển của thế giới đầu tiên ở Việt Nam vào ngày 20/01/2000. Diện tích rừng phủ xanh được cho bởi hàm số $ S = 0{,}05t + 3{,}14 $, trong đó $ S $ tính bằng nghìn héc-ta, $ t $ tính bằng số năm kể từ năm 2000.
	\begin{enumerate}
		\item Tính diện tích Rừng Sác được phủ xanh vào năm 2000.
		\item Diện tích Rừng Sác được phủ xanh đạt $ 4{,}04 $ nghìn héc-ta vào năm nào?
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Vào năm 2000, ta có $ t = 0 $. Suy ra diện tích rừng được phủ xanh
			\begin{align*}
				S_1 = 0{,}05 \cdot 0 + 3{,}14 = 3{,}14.
			\end{align*}
			Vậy vào năm 2000 thì diện tích Rừng Sác được phủ xanh là $ 3{,}14 $ nghìn héc-ta.
			\item Diện tích Rừng Sác được phủ xanh đạt $ 4{,}04 $ nghìn héc-ta thì
			\begin{eqnarray*}
				& & 0{,}05t + 3{,}14 = 4{,}04\\
				& \Leftrightarrow & 0{,}05t = 0{,}9\\
				& \Leftrightarrow & t = 18.
			\end{eqnarray*}
			Vậy đến năm 2018 thì diện tích Rừng Sác được phủ xanh đạt $ 4{,}04 $ nghìn héc-ta.
		\end{enumerate}
	}
\end{bt}
