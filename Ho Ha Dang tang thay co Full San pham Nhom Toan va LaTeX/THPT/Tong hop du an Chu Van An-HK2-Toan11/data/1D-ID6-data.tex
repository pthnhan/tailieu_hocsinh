\section{Phương pháp quy nạp}
	\dang{Các dạng toán áp dụng trực tiếp phương pháp quy nạp}
\begin{ex}%[1D3G1-1][Đề cương khối 11 HK2, Chu Văn An, HN, 2018][Nguyễn Đức Lợi]\\
Tính $S=1+2\cdot 2 + 3\cdot 2^2 + \cdots + 2018\cdot 2^{2017}$.
\choice
{\True $2017\cdot 2^{2018}+1$}
{$2017\cdot 2^{2018}-1$}
{$2017\cdot 2^{2018}-1$}
{$2^{2018}+1$}
\loigiai{
Ta chứng minh công thức tổng quát bằng phương pháp quy nạp
\[S_n=1+2\cdot 2 + 3 \cdot 2^2 + \cdots + (n+1) \cdot 2^n=n \cdot 2^{n+1}+1, \forall n \in \Bbb N \tag{$*$}\]
Dễ thấy $(*)$ đúng khi $n=1$.\\
Giả sử $(*)$ đúng khi $n=k$, tức là $S_k=1+2\cdot 2 + 3 \cdot 2^2 + \cdots +(k+1) \cdot 2^k=k \cdot 2^{k+1}+1, k\in \Bbb N^*$. \\
Ta chứng minh $(*)$ đúng khi $n=k+1$. Thật vậy
$$S_{k+1}=S_k+(k+2)2^{k+1}=k\cdot  2^{k+1}+1+(k+2)\cdot 2^{k+1}=(2k+2)\cdot 2^{k+1}+1=(k+1)\cdot 2^{k+2}+1$$
Vậy đẳng thức $(*)$ được chứng minh.\\
Áp dụng vào bài toán với $n=2017$ ta được $S=2017\cdot 2^{2018}+1$.
}
\end{ex}
\section{Dãy số}
	\dang{Biểu diễn dãy số, tìm công thức tổng quát dãy số}
\begin{ex}%[1D3K2-1][Đề cương khối 11 HK2, Chu Văn An, HN, 2018][Trần Phong]\\
Cho dãy số $(u_n)$ được xác định bởi $\heva{&u_1=5\\& u_{n+1}=u_n+n}$. Tìm số hạng tổng quát của dãy số.
\choice
{\True $u_n=5+\dfrac{(n-1)n}{2}$}
{$u_n=\dfrac{(n-1)n}{2}$}
{$u_n=5+\dfrac{(n+1)n}{2}$}
{$u_n=\dfrac{5+(n+1)(n+2)}{2}$}
\loigiai{
Ta có
$$\begin{array}{lll}
&u_{n+1}-u_n &=n\\
&u_n-u_{n-1} &=n-1\\
&u_{n-1}-u_{n-2}&=n-2\\
&\quad \ldots & \quad \ldots \\
&u_2-u_{1}&=1\\
\end{array}$$
Cộng vế theo vế ta được: $u_{n+1}-u_1=n+(n-1)+(n-2)+\cdots+1=\dfrac{n(n+1)}{2}$.\\
Hay $u_{n+1}=5+\dfrac{n(n+1)}{2}$. Suy ra $u_n=5+\dfrac{(n-1)n}{2},\forall n\ge1$.
}
\end{ex}
	\dang{Tìm hạng tử trong dãy số}
\begin{ex}%[Đề cương HK2, Chu Văn An, Hà Nội,Phan Minh Tâm][1D3B2-2]\\
	Cho dãy số $ (u_n) $ với $ u_n=\dfrac{an^2}{n+1} $ ($ a $ là hằng số). Số hạng $ u_{n+1} $ là số hạng nào sau đây?
	\choice
	{$ u_{n+1}=\dfrac{an^2}{n+2} $}
	{$ u_{n+1}=\dfrac{an^2+1}{n+1} $}
	{\True $ u_{n+1}=\dfrac{a(n+1)^2}{n+2} $}
	{$ u_{n+1}=\dfrac{a(n+1)^2}{n+1} $}
	\loigiai{
		Từ công thức số hạng tổng quát $ u_n=\dfrac{an^2}{n+1} $, ta có $ u_{n+1}=\dfrac{a(n+1)^2}{(n+1)+1}=\dfrac{a(n+1)^2}{n+2} $.
	}
\end{ex}
\begin{ex}%[1D3B2-2][Đề cương Toán 11, HK2, Chu Văn An, Hà Nội][Nguyễn Thị Kiều Ngân]\\
Dãy số $(u_n)$ được xác định bởi $u_n=\dfrac{n^2+3n+7}{n+1}$ có bao nhiêu số hạng nhận giá trị nguyên?
\choice
{$2$}
{Không có}
{\True $1$}
{$4$}
\loigiai{
	Ta có $u_n=\dfrac{n^2+3n+7}{n+1}=n+2+\dfrac{5}{n+1}$, $\forall n \in \mathbb{N^*}$.\\
	$u_n$ là số nguyên 
	$\Leftrightarrow$ $\dfrac{5}{n+1}$  là số nguyên $\Leftrightarrow$ $5$ chia hết cho $(n+1)$ $\Leftrightarrow \hoac{&n+1=1\\&n+1=5\\&n+1=-1 \quad \text{(loại)}\\&n+1=-5 \quad \text{(loại)}}\\ \Leftrightarrow \hoac{&n=0 \quad \text{(loại)}\\&n=4 \quad \text{(nhận).}}$\\
	Vậy có $1$ số hạng nhận giá trị nguyên là $u_4=7$.
}
\end{ex}
\begin{ex}%[Đề cương Chu Văn An Toán 11][1D3G2-2][NguyenThanhKhoa]\\
Cho dãy số có $4$ số hạng đầu là: $-1$, $3$, $19$, $53$. Tìm số hạng thứ $10$ của dãy.
\choice
{$u_{10}=97$}
{$u_{10}=71$}
{$u_{10}=1414$}
{\True$u_{10}=971$}
\loigiai{
	Công thức số hạng tổng quát của dãy số: $u_n=n^3-3n+1$.\\
Vậy $u_{10}=10^3-3.10+1=971$.}
\end{ex}
	\dang{Dãy số tăng, dãy số giảm}
\begin{ex}%[Đề cương Chu Văn An Toán 11][1D3B2-3][NguyenThanhKhoa]\\
Xét tính tăng giảm của dãy số $u_n=\dfrac{3n^2-2n+1}{n+1}$.
\choice
{Dãy số giảm}
{\True Dãy số tăng}
{Dãy số không tăng không giảm}
{ Cả A, B, C đều sai}
\loigiai{
Ta có: $u_{n+1}=\dfrac{3n^2+4n+2}{n+2}$.\\
Xét $u_{n+1}-u_n=\dfrac{3n^2+9n}{(n+2)(n+1)}>0$.\\
Vậy dãy số $(u_n)$ là dãy số tăng.	}
\end{ex} 
\begin{ex}%[1D3K2-3][Đề cương Toán 11, HK2, Chu Văn An, Hà Nội][Nguyễn Thị Kiều Ngân]\\
	Xét tính tăng giảm của dãy số $u_n=n-\sqrt{n^2-1}$.
	\choice
	{\True Dãy số giảm}
	{Dãy số không tăng không giảm}
	{Dãy số tăng}
	{Ba phương án đều sai}
	\loigiai{
		Ta có $u_{n+1}=n+1-\sqrt{(n+1)^2-1}=n+1-\sqrt{n^2+2n}$.\\
		$u_{n+1}-u_n=1+\sqrt{n^2-1}-\sqrt{n^2+2n}$.\\
		Dãy số $(u_n)$ là dãy số giảm nếu $u_{n+1}-u_n<0$, $\forall n \in \mathbb{N^*}$.\\
		Dãy số $(u_n)$ là dãy số tăng nếu $u_{n+1}-u_n>0$, $\forall n \in \mathbb{N^*}$.\\
		Ta có
		\begin{eqnarray*}
			& & u_{n+1}-u_n<0 \\
			&\Leftrightarrow & 1+\sqrt{n^2-1}-\sqrt{n^2+2n}<0\\
			& \Leftrightarrow & 1+\sqrt{n^2-1}<\sqrt{n^2+2n}\\
			& \Leftrightarrow & 1+n^2-1+2\sqrt{n^2-1} < n^2+2n\\
			& \Leftrightarrow & \sqrt{n^2-1} < n\\
			& \Leftrightarrow & n^2-1 < n^2
			 \quad \text{(luôn đúng).}
		\end{eqnarray*}
		Vậy dãy số giảm.
	}
\end{ex}
	\dang{Dãy số bị chặn trên, bị chặn dưới}
\begin{ex}%[1D3K2-4][Đề cương khối 11 HK2, Chu Văn An, HN, 2018][Trần Phong]\\
Xét tính tăng, giảm và bị chặn của dãy số $(u_n)$, biết $u_n=1+\dfrac{1}{2^2}+\dfrac{1}{3^2}+\cdots+\dfrac{1}{n^2}$.
\choice
{\True Dãy số tăng, bị chặn}
{Dãy số tăng, bị chặn dưới}
{Dãy số giảm, bị chặn trên}
{Ba phương án đều sai}
\loigiai{
Ta có $u_{n+1}-u_n=\dfrac{1}{(n+1)^2}>0\Rightarrow (u_n)$ là dãy số tăng.\\
Lại có $$u_n<1+\dfrac{1}{1\cdot 2}+\dfrac{1}{2\cdot 3}+\cdots+\dfrac{1}{(n-1)n}=1+1-\dfrac{1}{2}+\dfrac{1}{2}-\dfrac{1}{3}+\cdots+\dfrac{1}{n-1}-\dfrac{1}{n}=2-\dfrac{1}{n}<2,\forall n\in\mathbb N^*.$$
Suy ra dãy $(u_n)$ bị chặn trên bởi $2$.\\
Lại có $u_n>0,\forall n\in\mathbb N^*$, suy ra dãy $(u_n)$ bị chặn dưới bởi $0$.\\
Vậy dãy $(u_n)$ là dãy số tăng và bị chặn.
}
\end{ex}
\begin{ex}%[Đề cương Chu Văn An Toán 11][1D3K2-4][NguyenThanhKhoa]\\
Xét tính bị chặn của dãy số $u_n=\dfrac{1}{1.3}+\dfrac{1}{3.5}+...+\dfrac{1}{(2n-1)(2n+1)}$
.
\choice
{\True Bị chặn}
{Bị chặn trên}
{Bị chặn dưới}
{Không bị chặn}
\loigiai{
	Ta có: $2u_n=\dfrac{2}{1.3}+\dfrac{2}{3.5}+...+\dfrac{2}{(2n-1)(2n+1)}=\left(1-\dfrac{1}{3}\right)+\left(\dfrac{1}{3}-\dfrac{1}{5}\right)+...+\left(\dfrac{1}{2n-1}-\dfrac{1}{2n+1}\right)\\=1-\dfrac{1}{2n+1}$
$\Rightarrow u_n=\dfrac{1}{2}-\dfrac{1}{2(2n+1)}<\dfrac{1}{2}$.\\
Lại có $u_n>0$.\\
Vậy $0<u_n<\dfrac{1}{2}$ nên dãy số $(u_n)$ bị chặn.}
\end{ex}
\section{Cấp số cộng}
	\dang{Nhận diện cấp số cộng}
\begin{ex}%[1D3Y3-1][Đề cương khối 11 Hk2, Chu Văn An, HN, 2018][Lê Thị Thúy Hằng]\\
Cho một cấp số cộng có $u_1 = -\dfrac{1}{2}$; $d= \dfrac{1}{2}$. Trong các kết quả dưới đây, kết quả nào đúng?
\choice
{Cấp số cộng có dạng khai triển là $-\dfrac{1}{2}; 0; 1; \dfrac{1}{2}; 1; \ldots $}
{\True Cấp số cộng có dạng khai triển là $-\dfrac{1}{2}; 0; \dfrac{1}{2}; 1; \dfrac{3}{2}; \ldots $}
{Cấp số cộng có dạng khai triển là $-\dfrac{1}{2}; 0; \dfrac{1}{2}; 0; \dfrac{1}{2}; \ldots $}
{Cấp số cộng có dạng khai triển là $\dfrac{1}{2}; 1; \dfrac{3}{2}; 2; \dfrac{5}{2}; \ldots $}
\loigiai{
Cấp số cộng có $u_1 = -\dfrac{1}{2}$; $d= \dfrac{1}{2}$ nên nó có dạng khai triển là  $-\dfrac{1}{2}; 0; \dfrac{1}{2}; 1; \dfrac{3}{2}; \ldots $.
}
\end{ex} 
\begin{ex}%[1D3Y3-1][Đề cương Toán 11, HK2, Chu Văn An, Hà Nội][Nguyễn Thị Kiều Ngân]\\
	Dãy số $(u_n)$ có phải cấp số cộng không? Nếu phải hãy xác định công sai? Biết $u_n=2n+3$.
	\choice
	{$d=-2$}
	{\True $d=2$}
	{$d=5$}
	{$d=3$}
	\loigiai{
		Ta có $u_{n+1}-u_n=2(n+1)+3-(2n+3)=2$, $\forall n \in \mathbb{N^*}$.\\
		Suy ra dãy số $(u_n)$ là cấp số cộng có công sai $d=2$.
	}
\end{ex}
\begin{ex}%[1D3B3-1][Đề cương Toán 11, HK2, Chu Văn An, Hà Nội][Nguyễn Thị Kiều Ngân]\\
	Cho $a$, $b$, $c$ theo thứ tự lập thành cấp số cộng, đẳng thức nào sau đây đúng?
	\choice
	{$a^2-c^2=ab-bc$}
	{$a^2+c^2=2ab+2bc$}
	{$a^2+c^2=2ab-2bc$}
	{\True $a^2-c^2=2ab-2bc$}
	\loigiai{
		Vì $a$, $b$, $c$ theo thứ tự lập thành cấp số cộng nên $a+c=2b$.\\
		Ta có $a^2-c^2=(a+c)(a-c)=2b(a-c)=2ab-2bc$.
	}
\end{ex}
	\dang{Tìm công thức của cấp số cộng}
\begin{ex}%[1D3B3-2][Đề cương khối 11 HK2, Chu Văn An, HN, 2018][Trần Phong]\\
Cho một cấp số cộng có $u_1=-3$, $u_6=27$. Tìm công sai $d$.
\choice
{$d=5$}
{$d=7$}
{\True $d=6$}
{$d=8$}
\loigiai{
Ta có $u_6=u_1+5d\Leftrightarrow 27=-3+5d\Leftrightarrow d=6$.
}
\end{ex}
\begin{ex}%[Đề cương HK2, Chu Văn An, Hà Nội,Phan Minh Tâm][1D3K3-2]\\
	Cho dãy số có các số hạng đầu là $ 8,15,22,29,36,... $ Viết công thức số hạng tổng quát $ u_n $ của dãy số trên?
	\choice
	{$ u_n=7n+7 $}
	{$ u_n=7n $}
	{Dãy số không viết được dưới dạng công thức tổng quát}
	{\True $ u_n=7n+1 $}
	\loigiai{
		Từ các số hạng đầu tiên ta suy ra dãy số là cấp số cộng với $ \heva{&u_1=8\\&d=u_2-u_1=7} $. Do đó, công thức số hạng tổng quát của dãy số là $ u_n=u_1+(n-1)d=8+(n-1)\cdot 7=7n+1 $.
	}
\end{ex}
\begin{ex}%[Đề cương HK2, Chu Văn An, Hà Nội,Phan Minh Tâm][1D3K3-2]\\
	Cho dãy số có các số hạng đầu là $ 8,15,22,29,36,... $ Viết công thức số hạng tổng quát $ u_n $ của dãy số trên?
	\choice
	{$ u_n=7n+7 $}
	{$ u_n=7n $}
	{Dãy số không viết được dưới dạng công thức tổng quát}
	{\True $ u_n=7n+1 $}
	\loigiai{
		Từ các số hạng đầu tiên ta suy ra dãy số là cấp số cộng với $ \heva{&u_1=8\\&d=u_2-u_1=7} $. Do đó, công thức số hạng tổng quát của dãy số là $ u_n=u_1+(n-1)d=8+(n-1)\cdot 7=7n+1 $.
	}
\end{ex}
	\dang{Tìm điều kiện để một dãy số là cấp số cộng}
\begin{ex}%[1D3K3-4][Đề cương khối 11 Hk2, Chu Văn An, HN, 2018][Lê Thị Thúy Hằng]\\
Xác định $a, b$ để phương trình $x^3 +ax +b=0$ có ba nghiệm phân biệt lập thành cấp số cộng.
\choice
{$b=0, a>0$}
{$b>0, a<0$}
{\True $b=0, a<0$}
{$b=0, a=1$}
\loigiai{
Phương trình $x^3 +ax +b=0$ có ba nghiệm phân biệt lập thành cấp số cộng nên ta gọi $u-d$, $u$ và $u+d$ theo thứ tự là ba nghiệm của phương trình. Khi đó, ta có
$$\allowdisplaybreaks 
(x-u)(x-u+d)(x-u-d)=0 
\Leftrightarrow x^3 -3ux^2  +(3u^2 -d)x -u^3 +d^2u =0,$$
\text{ suy ra } $x^3 -3ux^2  +(3u^2 -d)x -u^3 +d^2u = x^3 +ax +b$. \hfill(1) \\
Đồng nhất hai vế $(1)$ ta có 
$\left\{ \begin{aligned}
& -3u=0 \\
& 3u^2-d^2=a \\
& -u^3 +d^2u=b 
\end{aligned}
\right. 
\Leftrightarrow \left\{ \begin{aligned}
& u=0 \\
& d^2=-a \\
& b=0.
\end{aligned}
\right. $ \hfill(2) \\
Từ hệ $(2)$, phương trình $x^3 +ax +b=0$ có ba nghiệm phân biệt lập thành cấp số cộng khi và chỉ khi phương trình $d^2=-a$ ($d$ là ẩn) có hai nghiệm phân biệt hay $a<0$. \\
Vậy $a<0$, $b=0$ thỏa yêu cầu bài toán.
}
\end{ex}
\begin{ex}%[1D3G3-4][Đề cương khối 11 HK2, Chu Văn An, HN, 2018][Nguyễn Đức Lợi]\\
Tính tổng tất cả các giá trị thực của tham số $m$ để phương trình $x^4-20x^2+(m-1)^2=0$ có bốn nghiệm phân biệt lập thành một cấp số cộng.
\choicefix
{$7$}
{$-2$}
{\True $2$}
{Đáp án khác}
\loigiai{
Đặt $t=x^2, t \ge 0$. Phương trình đã cho trở thành $t^2-20t+(m-1)^2=0$. \hfill $(1)$\\
Phương trình ban đầu có 4 nghiệm khi và chỉ khi phương trình $(1)$ có hai nghiệm dương phân biệt
\[\Leftrightarrow \heva{& \Delta'=100-(m-1)^2>0\\ & t_1+t_2=10 >0 \\& t_1t_2 =(m-1)^2 >0} \Leftrightarrow \heva{&-9<m<11\\& m\ne 1} \tag{$*$}\]
Với điều kiện $(*)$ thì $(1)$ có hai nghiệm là $t_1>t_2>0$, như vậy phương trình ban đầu có 4 nghiệm là $-\sqrt{t_1} < -\sqrt{t_2}<\sqrt{t_2}<\sqrt{t_1}$. Để 4 nghiệm này lập thành một cấp số cộng thì điều kiện cần và đủ là 
$$\sqrt{t_1}-\sqrt{t_2}=2\sqrt{t_2}\Leftrightarrow t_1=9t_2$$
Mặt khác theo định lí Viet thì $\heva{ & t_1+t_2=10 \\& t_1t_2 =(m-1)^2}$.\\
Do đó ta được $t_2=1$; $t_1=9$ và $(m-1)^2=9 \Leftrightarrow \hoac{& m=4\\ & m=-2}$\\
Vậy tổng các giá trị thỏa mãn yêu cầu bài toán bằng $4+(-2)=2$.
}
\end{ex}
	\dang{Tính tổng của dãy nhiều số hạng liên quan đến cấp số cộng, tổng các hạng tử của cấp số cộng}
\begin{ex}%[1D3K3-5][Đề cương khối 11 Hk2, Chu Văn An, HN, 2018][Lê Thị Thúy Hằng]\\
Cho cấp số cộng $(u_n)$ thỏa mãn 
$\left\{ \begin{aligned}
& u_5 + 3 u_3 - u_2 = -21 \\
& 3 u_7 - 2 u_4 = -34
\end{aligned}
\right.$.
Tính $S= u_4 + u_5 + \cdots + u_{30}$.
\choice
{\True $-1242$}
{$-1222$}
{$-1276$}
{$-1286$}
\loigiai{
Gọi cấp số cộng $(u_n)$ có số hạng đầu là $u_1$ và công sai là $d$. \\
Ta có \\ 
$$ \allowdisplaybreaks \begin{aligned}
& \left\{ \begin{aligned}
& u_5 + 3 u_3 - u_2 = -21 \\
& 3 u_7 - 2 u_4 = -34
\end{aligned}
\right.
&& \Leftrightarrow \left\{ \begin{aligned}
& u_1 + 4d + 3 (u_1 + 2d) - u_1 -d = -21 \\
& 3 (u_1 +6d) - 2 (u_1 + 3d) = -34
\end{aligned}
\right. \\
\Leftrightarrow & \left\{ \begin{aligned}
	& u_1 + 3d = -7 \\
	& u_1 + 12d = -34
\end{aligned}
\right.
&& \Leftrightarrow \left\{ \begin{aligned}
	& u_1 = 2 \\
	& d = -3.
\end{aligned}
\right.
\end{aligned}$$
Vậy $S= u_4 + u_5 + \cdots + u_{30} 
= \dfrac{(u_4 + u_{30}) \cdot 27}{2}
= \dfrac{(2 u_1 + 32d) \cdot 27}{2}
= \dfrac{(2 \cdot 2 + 32 \cdot (-3)) \cdot 27}{2} = -1242$.
}
\end{ex}
	\dang{Các bài toán thực tế}
\begin{ex}%[Đề cương HK2, Chu Văn An, Hà Nội,Phan Minh Tâm][1D3K3-6]\\
	Tam giác $ ABC $ có ba góc $ A $, $ B $, $ C $ theo thứ tự lập thành cấp số cộng và $ C=5A $. Xác định số đo các góc $ A $, $ B $, $ C $.
	\choice
	{$ 10^{\circ}, 120^{\circ}, 50^{\circ} $}
	{$ 15^{\circ}, 105^{\circ},  60^{\circ} $}
	{$ 5^{\circ}, 60^{\circ}, 25^{\circ} $}
	{\True $ 20^{\circ}, 60^{\circ}, 100^{\circ} $}
	\loigiai{

 Cho hàm số $y=2x^3 +3x^2-4x+5$ có đồ thị $(C)$. Trong số các tiếp tuyến của $(C)$, có một tiếp tuyến có hệ số góc nhỏ nhất. Hệ số góc của tiếp tuyến này bằng
 \choice
 {$-7{,}5 $}
 {$ -9{,}5$}
 {$ -3{,}5$}
 {\True $ -{5,5}$}
 \loigiai{
 Ta có $y'=6x^2+6x-4$, $\Delta =132$.\\
 Giả sử tiếp tuyến của $(C)$ có hệ số góc là $k$. Khi đó $\min k=\min y'=\dfrac{-\Delta}{4a}=-5{,}5$.
 }
 \end{ex}
\begin{ex}%[1D5G2-5][Đề cương khối 11 HK2, Chu Văn An, HN, 2018][Phan Tấn Phú]\\
Cho hàm số $y=x^3-3x+1$	có đồ thị $(C)$. Gọi $A(x_A;y_A)$, $B(x_B;y_B)$ với $x_A>x_B$ là các điểm thuộc $(C)$ sao cho các tiếp tuyến tại $A, B$ song song với nhau và $AB=6\sqrt{37}$. Tính $S=2x_A-3x_B$.
\choice
{\True $S=15$}
{$S=90$}
{$S=-15$}
{$S=-90$}
\loigiai{
Ta có $y'=3x^2-3$. Vì tiếp tuyến tại $A$ và $B$ song song với nhau nên
$$y'(x_A)=y'(x_B) \Leftrightarrow 3x_A^2-3=3x_B^2-3 \Leftrightarrow (x_A-x_B)(x_A+x_B)=0 \Leftrightarrow x_A=\pm x_B.$$
Vì $x_A>x_B$ nên chỉ được $x_A=-x_B$. Vì $AB=6\sqrt{37}$ nên
$$
\begin{aligned}
&\sqrt{(x_B-x_A)^2+(y_B-y_A)^2}=6\sqrt{37}\\
\Leftrightarrow &(x_B-x_A)^2 + \left[(x_B^3-3x_B+1)-(x_A^3-3x_A+1)\right]^2 = 1332\\
\Leftrightarrow &(x_B-x_A)^2+\left[(x_B^3-x_A^3)-3(x_B-x_A)\right]^2 = 1332\\
\Leftrightarrow & (2x_B)^2+\left(2x_B^3-6x_B\right)^2=1332\\
\Leftrightarrow & x_B^2+\left[x_B\cdot (x_B^2-3)\right]^2=333
\end{aligned}
$$
Đặt $t=x_B^2$ ta được
$$
\begin{aligned}
&t+t(t-3)^2=333\\
\Leftrightarrow & (t-9)(t^2+3t+37)=0\\
\Leftrightarrow & t=9.
\end{aligned}
$$
Ta được $x_B^2=9 \Leftrightarrow x_B=\pm 3$.\\
Khi $x_B=3$ thì $x_A=-3$ (loại vì $x_A>x_B$).\\
Khi $x_B=-3$ thì $x_A=3$ (nhận).\\
Vậy $2x_A-3x_B=15$.
}
\end{ex}
	\dang{Bài toán quãng đường, vận tốc, gia tốc}
\begin{ex}%[1D5B2-6][Đề cương khối 11 Hk2, Chu Văn An, HN, 2018][Đỗ Vũ Minh Thắng]\\
Một chất điểm chuyển động theo phương trình $s(t)=t^2$, trong đó $t$ tính bằng giây và $s(t)$ tính bằng mét. Tính vận tốc của chất điểm tại thời điểm $t=2$ giây.
\choice{$3$m/s}{$2$m/s}{\True $4$m/s}{$5$m/s}
\loigiai{
$v(t)=s'(t)=2t$.\\
Vận tốc của chất điểm tại thời điểm $t=2$ giây là $2\cdot 2=4$ m/s.
}
\end{ex}
\section{Đạo hàm của các hàm số lượng giác}
	\dang{Giới hạn hàm số lượng giác}
\begin{ex}%[Đề cương khối 11 HK2, Chu Văn An, Hà Nội,2018][1D5G3-2][Cao Thành Thái]\\
 Cho $a$ và $b$ là các số thực khác $0$. Tính $\lim\limits_{x \to 0} \dfrac{1-\sqrt{ax+1}}{\sin bx}$.
 \choice
  {$-\dfrac{2a}{b}$}
  {$\dfrac{a}{2b}$}
  {\True $-\dfrac{a}{2b}$}
  {$\dfrac{2a}{b}$}
 \loigiai
  {
  Với $a \neq 0$, $b \neq 0$, $x \neq 0$ ta có
  \begin{eqnarray*}
   \dfrac{1-\sqrt{ax+1}}{\sin bx} &=& \dfrac{-ax}{\left(1+\sqrt{ax+1}\right)\sin bx} = \dfrac{-a}{b\left(1+\sqrt{ax+1}\right)} \cdot \dfrac{bx}{\sin bx}
   = \dfrac{-a}{b\left(1+\sqrt{ax+1}\right)} \cdot \dfrac{1}{\dfrac{\sin bx}{bx}}.
  \end{eqnarray*}
  Vậy $\lim\limits_{x \to 0} \dfrac{1-\sqrt{ax+1}}{\sin bx} = \lim\limits_{x \to 0} \left[ \dfrac{-a}{b\left(1+\sqrt{ax+1}\right)} \cdot \dfrac{1}{\dfrac{\sin bx}{bx}} \right] = -\dfrac{a}{2b} \cdot 1 = -\dfrac{a}{2b}$.
  }
\end{ex}
