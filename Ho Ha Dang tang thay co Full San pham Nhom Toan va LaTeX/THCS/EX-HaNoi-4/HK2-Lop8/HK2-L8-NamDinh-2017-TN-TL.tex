\noindent\textbf{I. PHẦN TRẮC NGHIỆM}
\setcounter{ex}{0}
\begin{ex}%[8D3B4]
Giá trị biểu thức $A=3x-4$ tại số thực $x$ thỏa mãn $(2-x)(x^2+1)=0$ là
\choice
{$-7$}
{$-1$}
{\True $2$}
{$-10$}
\loigiai{
Ta có $(2-x)(x^2+1)=0\Leftrightarrow x=2$ (vì $x^2+1>0$) nên $A=2$.
}
\end{ex}

\begin{ex}%[8D3B4]
Phương trình $(x-1)(2x+1)=(x-1)(x+3)$ có tập nghiệm là 
\choice
{$\{-4\}$}
{$\{2\}$}
{$\{-2;1\}$}
{\True $\{1;2\}$}
\loigiai{
Phương trình tương đương với $(x-1)(x-2)=0\Leftrightarrow \hoac{& x=1\\ & x=2}$.
}
\end{ex}

\begin{ex}%[8D3B5]
Điều kiện xác định của phương trình $\dfrac{x+1}{1-x}+\dfrac{x^2}{x^2-2x+1}=0$ là
\choice
{\True $x\neq 1$}
{$x\neq \pm 1$}
{$x<1$}
{$x>1$}
\loigiai{
Ta có $x^2-2x+1=(x-1)^2$ nên điều kiện xác định là $x-1\neq 0\Leftrightarrow x\neq 1$.
}
\end{ex}

\begin{ex}%[8D4Y3]
Số $1$ thuộc tập nghiệm của bất phương trình
\choice
{$x^3<1$}
{\True $x(1-x)\le 0$}
{$-x^2+2x-1>0$}
{$3-x>2$}
\loigiai{
Ta có $1(1-1)=0$ thỏa mãn $x(1-x)\le 0$.
}
\end{ex}

\begin{ex}%[8D4B2]
Với các số thực $a,b,c$ bất kỳ. Khẳng định nào sau đây \textbf{sai}?
\choice
{Nếu $a>b$ và $b\ge c$ thì $a>c$}
{Nếu $a\ge b$ và $c>0$ thì $ac\ge bc$}
{\True Nếu $a<b$ và $c<0$ thì $-ac>-bc$}
{Nếu $a<b$ và $c<0$ thì $ac>bc$}
\loigiai{
Ta có $a<b\Leftrightarrow ac>bc\Leftrightarrow -ac<-bc$ (Với $c<0$).
}
\end{ex}

\begin{ex}%[8H3B2]
Cho $\triangle ABC$. Nếu điểm $D$ thuộc cạnh $AB$ sao cho $AB=3BD$ và điểm $E$ thuộc cạnh $AC$ sao cho $AE=2EC$ thì
\choice
{$BC=2DE$}
{$BC=3DE$}
{$BC=\dfrac{2}{3}DE$}
{\True $BC=\dfrac{3}{2}DE$}
\loigiai{
Ta có $AB=3BD\Leftrightarrow \dfrac{AB}{BD}=3$ và $AE=2EC\Leftrightarrow AE+EC=3EC\Leftrightarrow AC=3EC\Leftrightarrow \dfrac{AC}{EC}=3$. Tức là $\dfrac{AB}{EC}=\dfrac{AC}{EC}=3$, suy ra $DE\parallel BC$ và $\dfrac{BC}{DE}=\dfrac{AB}{AD}=\dfrac{3}{2}\Leftrightarrow BC=\dfrac{3}{2}DE$.
}
\end{ex}

\begin{ex}%[8H3Y3]
Cho $\triangle ABC$, tia $AD$ là tia phân giác của $\widehat{BAC}$ ($D\in BC$). Cho độ dài đoạn thẳng $AB=14$ cm, $AC=21$ cm, $BD=8$ cm. Độ dài đoạn thẳng $BC$ là
\choice
{\True $20$ cm}
{$15$ cm}
{$\dfrac{40}{3}$ cm}
{$12$ cm}
\loigiai{
Ta có $AD$ là tia phân giác nên $\dfrac{DB}{DC}=\dfrac{AB}{AC}\Leftrightarrow \dfrac{8}{DC}=\dfrac{14}{21}\Leftrightarrow DC=12\Rightarrow BC=20$ cm.
}
\end{ex}

\begin{ex}%[8H4B6]
Cho hình lăng trụ đứng tam giác có chiều cao là $6$ cm và mặt đáy là tam giác có các cạnh lần lượt là $3$ cm, $4$ cm và $5$ cm. Thể tích của hình lăng trụ đã cho bằng
\choice
{$12$ cm$^3$}
{\True $36$ cm$^3$}
{$72$ cm$^3$}
{$84$ cm$^3$}
\loigiai{
Ta có $3^2+4^2=5^2$ nên đáy lăng trụ là tam giác vuông cạnh huyền bằng $5$ cm. Do đó diện tích đáy là $S=\dfrac{1}{2}\cdot 3\cdot 4=6$ cm$^2$.\\
Thể tích lăng trụ là $V=h\cdot S=6\cdot 6=36$ cm$^3$.
}
\end{ex}
\noindent\textbf{II. PHẦN TỰ LUẬN}
\setcounter{ex}{0}
\begin{ex}%[8D4B4]
Giải bất phương trình $\dfrac{x+7}{5}+\dfrac{4x+5}{3}\ge 0$.
\loigiai{
Bất phương trình tương đương với
$$\dfrac{3(x+7)+5(4x+5)}{15}\ge 0\Leftrightarrow 23x>-46\Leftrightarrow x\ge -2.$$
Vậy $x\ge -2$.
}
\end{ex}

\begin{ex}%[8D3Y4]
Giải các phương trình $(3x-2)(2x+5)=0$.
\loigiai{
Phương trình tương đương với $\hoac{& 3x-2=0\\ & 2x+5=0}\Leftrightarrow\hoac{& x=\dfrac{2}{3}\\ & x=-\dfrac{5}{2}}$.\\
Vậy nghiệm phương trình là $x=\dfrac{2}{3}$, $x=-\dfrac{5}{2}$.
}
\end{ex}

\begin{ex}%[8D3B5]
Giải phương trình $\dfrac{x-1}{x+1}+\dfrac{4}{1-x^2}=\dfrac{2(x+1)}{x-1}$.
\loigiai{
Điều kiện $x\neq \pm 1$.\\
Phương trình tương đương với
\begin{eqnarray*}
&& \dfrac{(x-1)^2-4}{(x-1)(x+1)}=\dfrac{2(x+1)^2}{(x-1)(x+1)}\\
& \Rightarrow & (x-1)^2-2^2=2(x+1)^2\Leftrightarrow (x-3)(x+1)-2(x+1)^2=0\\
& \Leftrightarrow & (x+1)\left[x-3-2(x+1)\right]=0\Leftrightarrow (x+1)(-x-5)=0\\
& \Leftrightarrow & \hoac{& x=-1\quad \text{(loại)}\\ & x=-5\quad\text{(thỏa mãn)}.}
\end{eqnarray*}
Vậy nghiệm phương trình là $x=-5$.
}
\end{ex}

\begin{ex}%[8D4G5]
Giải phương trình $|x+1|+\left|x^2+x-2\right|=x^3-1$.
\loigiai{
Ta có $|x-1|+\left|x^2+x-2\right|\ge 0$ và $x^2+x+1>0$. Suy ra
$$x^3-1\ge 0\Leftrightarrow (x-1)(x^2+x+1)\ge 0\Leftrightarrow x-1\ge 0\Leftrightarrow x\ge 1.$$
Khi đó $|x-1|=x-1$.\\
Lại có $x^2+x-2=(x-1)(x+2)\ge 0$, với $x\ge 1$ nên $|x^2+x-2|=x^2+x-2$. Do đó phương trình trở thành
\begin{eqnarray*}
&& x+1+x^2+x-2=x^3-1\\
& \Leftrightarrow & x^3-x^2-2x=0\\
& \Leftrightarrow & x(x+1)(x-2)=0\Leftrightarrow \hoac{& x=0\quad \text{(loại)}\\ & x=-1\quad \text{(loại)}\\ & x=2\quad \text{(thỏa mãn)}.}
\end{eqnarray*}
Vậy nghiệm phương trình là $x=2$.
}
\end{ex}

\begin{ex}%[8D4G2]
Xét hai số thực dương $a,b$ thỏa mãn $a^2+b^2\le 2$. Chứng minh rằng
$$\dfrac{1}{1+a^2}+\dfrac{1}{1+b^2}\le \dfrac{2}{1+ab}.$$
\loigiai{
Ta có $a^2+b^2\le 2$. Mà $a^2+b^2\ge 2ab\Rightarrow 2ab\le 2\Leftrightarrow ab\le 1\Leftrightarrow ab-1\le 0$.\hfill (1)\\
Xét hiệu
\[H=\dfrac{1}{1+a^2}+\dfrac{1}{1+b^2}-\dfrac{2}{1+ab}=\dfrac{(1+b^2)(1+ab)+(1+a^2)(1+ab)-2(1+a^2)(1+b^2)}{(1+a^2)(1+b^2)(1+ab)}.\tag{2}\]
Khai triển rút gọn tử thức ta được
\begin{align*}
& \,\, (1+b^2)(1+ab)+(1+a^2)(1+ab)-2(1+a^2)(1+b^2)\\
= & \,\, 2ab+ab^3+a^3b-a^2-b^2-2a^2b^2\\
= & \,\, 2ab(1-ab)+a^2(ab-1)+b^2(ab-1)\\
= & \,\, (ab-1)(a^2+b^2-2ab)\\
= & \,\, (ab-1)(a-b)^2.\tag{3}
\end{align*}
Từ (1), (2) và (3) suy ra $H\le 0$ hay
$$\dfrac{1}{1+a^2}+\dfrac{1}{1+b^2}-\dfrac{2}{1+ab}\Leftrightarrow \dfrac{1}{1+a^2}+\dfrac{1}{1+b^2}\le \dfrac{2}{1+ab}.$$
}
\end{ex}
