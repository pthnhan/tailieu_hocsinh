\begin{ex}%[HSG Croatia năm 2018, lớp 11]%[ Lê Thanh Hải, Dự án EX-Nat.MO-18]%[1D0K7]
	Chứng minh đẳng thức sau $$\cos^3  \dfrac{x}{3}+\cos^3 \dfrac{x+2\pi}{3}+\cos^3\dfrac{x+4\pi}{3}=\dfrac{3}{4}\cos x$$ đúng với mọi số thực $x$.
	
	\loigiai{
		Ta có
		\begin{eqnarray*}
			& &\cos^3 \dfrac{x}{3}+\cos^3 \dfrac{x+2\pi}{3}+\cos^3\dfrac{x+4\pi}{3}=\dfrac{3}{4}\cos x
			\\& \Leftrightarrow & \dfrac{\cos x +3\cos\dfrac{x}{3}}{4}+\dfrac{\cos (x+2\pi)+3\cos\dfrac{x+2\pi}{3}}{4}+\dfrac{\cos(x+4\pi)+3\cos\dfrac{x+4\pi}{3}}{4}=\dfrac{3}{4}\cos x
			\\& \Leftrightarrow & \dfrac{3}{4}\cos x+ \dfrac{3}{4}\left(\cos\dfrac{x}{3}+\cos\dfrac{x+2\pi}{3}+ \cos\dfrac{x+4\pi}{3} \right)=\dfrac{3}{4}\cos x	
			\\& \Leftrightarrow &  \dfrac{3}{4}\left(\left( \cos\dfrac{x}{3}+ \cos\dfrac{x+4\pi}{3}\right)  +\cos\dfrac{x+2\pi}{3} \right)= 0	
			\\ &\Leftrightarrow & \dfrac{3}{4}\left(2\cos \dfrac{x+2\pi}{3}\cos \dfrac{2\pi}{3}+\cos \dfrac{x+2\pi}{3} \right)= 0
			\\ & \Leftrightarrow & \dfrac{3}{4}\cos \dfrac{x+2\pi}{3}\left(2\cos\dfrac{2\pi}{3}+1 \right)= 0
			\\ & \Leftrightarrow & 0 = 0 \text{(luôn đúng)}.
		\end{eqnarray*}	
		Vậy đẳng thức được chứng minh.
	}
\end{ex}


\begin{ex}%[HSG Croatia năm 2018, lớp 11]%[Lê Thanh Hải, Dự án EX-Nat.MO-18]%[8D0G3]
	Đặt $S= \lbrace 0,95 \rbrace$. Trong mỗi bước, Lucija mở rộng tập $S$ theo cách sau. Cô ấy chọn một đa thức có hệ số nằm trong tập $S$, khác $0$ và mở rộng tập $S$ với tất cả nghiệm nguyên của đa thức đã chọn. Cô ấy lặp lại quá trình trên bằng cách chọn đa thức khác với hệ số từ tập $S$ đã được mở rộng miễn là cô ấy có được những nghiệm mới.\\
	Chứng minh rằng Lucija có thể mở rộng tập $S$ lên đến một tập không thể mở rộng được nữa trong hữu hạn bước lặp. Tập $S$ thu được cuối cùng có bao nhiêu phần tử?
	\loigiai{
		\begin{nx}
			Nếu một đa thức có hệ số nguyên thì các nghiệm của nó là ước của hệ số tự do.
		\end{nx}
		\textbf{TH1:} Hệ số tự do là $0$, khi đó giả sử đa thức dạng $$95x^{n_1}+95x^{n_2}+\cdots+95x^{n_k}$$ với $n_1>n_2>\cdots n_k$, viết lại thành $95x^{n_k}\cdot Q(x)$ trong đó $Q(x)$ có hạng tử tự do là $1$ nên nghiệm nguyên nếu có sẽ là ước của $1$, tức $\pm 1$. Lúc này ta thêm $\pm 1$ vào tập $S$.\\
		\textbf{TH2:} Hệ số tự do của đa thức đã chọn là $95$.\\
		$95$ chỉ có hữu hạn ước thuộc $\left\lbrace \pm 1, \pm 5, \pm 19, \pm 95 \right\rbrace $.\\
		$5, 19$ đều là các số nguyên tố nên khi để $5, -5, 19, -19$ làm hệ số tự do trong một đa thức mới thì đa thức mới sẽ không sinh ra nghiệm nguyên mới.\\ Do đó, tập $S$ chỉ có thể mở rộng một cách hữu hạn. Chúng ta sẽ chứng minh rằng Lucija có thể thêm tất cả ước của $95$ vào trong $S$.\\
		Bây giờ ta có thể mở rộng $S$ với $-95$ vì nó là nghiệm của $x+95$.\\
		Đa thức $-x^{3}+x^{2}+x+95$ với hệ số trong $S$ có một nghiệm là $5$, vì thế $5$ trở thành một phần tử của $S$. Bây giờ ta thêm $-5$ vào $S$ vì $-5$ là nghiệm của $x+5$.\\
		Cuối cùng, $-19,19 \in S$ vì mỗi số lần lượt là ước của các đa thức $5x+95$, $5x-95$. 
		Vậy tập $S$ thu được sau cùng có $9$ phần tử là $0, \pm 1, \pm 5, \pm 19, \pm 95$.
	}
\end{ex}


\begin{ex}%[HSG Croatia năm 2018, lớp 11]%[Lê Thanh Hải, Dự án EX-Nat.MO-18]%[8D0B5]%[7D0B5]
	Xác định tất cả các cặp số nguyên dương $ (a,b) $ thỏa $ b^2+3a $ chia hết cho $ a^2b $.
	\loigiai{ 
		Do $ b^2+3a $ chia hết cho $ a^2b $ nên tồn tại số nguyên dương $ k $ sao cho
		\[ a^2b\cdot k=b^2+3a.\]
		Phương trình trên tương đương với $ a^2b\cdot k-3a=b^2 $, từ đó suy ra $ b^2 \vdots a  $ và $ 3a \vdots b $.\\ Do đó, các số
		 $\dfrac{b^2}{a} , \dfrac{3a}{b} \text{ và } \dfrac{b^2+3a}{a^2b} = \dfrac{b}{a^2} + \dfrac{3}{ab}$
		là các số nguyên. Nhân số thứ hai và thứ ba lại với nhau ta được $\dfrac{3}{a}+\dfrac{9}{b^2}  $ là một số nguyên, suy ra $\dfrac{9a}{b^2}$ là số nguyên, do đó $ 9a \vdots b^2 $. Như vậy, ta đã có
		\[ b^2m=9a , an =b^2.\]
		với $ m,n $ là các số nguyên dương. Từ đó suy ra $ mn=9 $. Do đó $ n \in \{1,3,9\} $ nên $ b^2 \in \{a,3a,9a\} $. Điều kiện ban đầu giờ trở thành $ (n+3) \vdots ab $.\\
		Ta xét ba khả năng có thể có của $ n $ như sau\\
		Nếu $ n=1 $ thì $ b^2=a $, do đó điều kiện ban đầu trở thành $ 4 \vdots ab=b^3 $. Khi đó chỉ có một phương án đó là $ a=b=1 $.\\
		Nếu $ n=3 $ thì $ b^2=3a $, do đó $ 6 \vdots ab=\dfrac{b^3}{3}$, nghĩa là $ 18 \vdots b^3 $. Do $ b $ là bội của $ 3 $ nên không có phương án nào cho trường hợp này.\\
		Nếu $ n=9 $ thì $ b^2=9a $, do đó $ 12 \vdots ab=\dfrac{b^3}{9} $, nghĩa là $ 108 \vdots b^3 $. Do $ b $ là bội của $ 3 $ nên có một phương án là $ (a,b)=(1,3) $.\\
		Vậy có hai cặp số thỏa yêu cầu bài toán đó là $ (1,1) $ và $ (1,3) $.
		
	}
\end{ex}




\begin{ex}%[HSG Croatia năm 2018, lớp 11]%[Lê Thanh Hải, Dự án EX-Nat.MO-18]%[9H0G3]
	Cho tam giác $ABC$ cân tại $A$. Gọi $M$, $N$ lần lượt là trung điểm của $AB$ và $BC$. Đường tròn ngoại tiếp tam giác $AMC$ cắt $AN$ tại điểm $P$ khác $A$.Đường thẳng đi qua $P$ song song với BC cắt đường tròn ngoại tiếp tam giác $ABC$ tại $B_1$ và $C_1$. Chứng minh tam giác $AB_1C_1$ là tam giác đều.
	\loigiai{ 
		\begin{center}
			\begin{tikzpicture}[line join = round, line cap = round, >=stealth, scale=1]
			\tkzInit[xmin=-4,xmax=4,ymin=-5,ymax=5]
			\tkzClip
			%	\tikzset{label style/.style={font=\footnotesize}}
			%\draw[cyan,dashed] (-5,-5) grid (5,5);
			\coordinate[label={above}:{\footnotesize $A$}] (A) at (0,4);
			\coordinate[label={above left}:{\footnotesize $B$}] (B) at (-3.5,0);
			\coordinate[label={above right}:{\footnotesize $C$}] (C) at (3.5,0);
			\coordinate[label={above left}:{\footnotesize $M$}] (M) at ($(A)!0.5!(B)$);
			\coordinate[label={below right}:{\footnotesize $N$}] (N) at ($(B)!0.5!(C)$);
			\coordinate[label={above}:{\footnotesize $Q$}] (Q) at ($(B)!0.5!(M)$);
			\tkzCircumCenter(A,B,C) \tkzGetPoint{O}
			\tkzLabelPoints[left](O) \tkzDrawCircle[radius](O,A)
			\tkzCircumCenter(A,M,C) \tkzGetPoint{O'} 
			\tkzDrawCircle[radius](O',A)
			\tkzInterLC(A,N)(O',A)    \tkzGetPoints{P}{A}
			\tkzLabelPoints[above right](P)
			\tkzDefLine[parallel=through P](C,B) \tkzGetPoint{c}
			\tkzInterLC(c,P)(O,A)    \tkzGetPoints{C_1}{B_1}
			\tkzLabelPoints[right](C_1)
			\tkzLabelPoints[left](B_1)
			\tkzDrawSegments(A,B B,C C,A M,P B,P P,C)
			\tkzDrawSegments[color = gray,style=dashed](Q,P M,O A,P A,B_1 A,C_1 B_1,C_1)
			\tkzMarkAngles[size=0.5cm,arc=l,mark=|](B,A,N N,A,C)
			\tkzMarkRightAngles[size=0.3](A,M,O A,Q,P A,N,C)
			\tkzMarkSegments[mark=|, size=1.5mm](M,P P,B P,C)
			\end{tikzpicture}
		\end{center}
		Ta có các điểm $A,M,P$ và $C$ nội tiếp đường tròn và $\widehat{MAP} = \widehat{PAC}$. Suy ra $MP = PC$, vì góc chắn 2 cung này bằng nhau.\\
		Vì $P$ thuộc trung tuyến của $BC$ nên $BP=CP$. Vì $MP=BP$ nên $P$ thuộc trung tuyến của $BM$.\\
		Gọi $Q$ là trung điểm của BM. O là tâm của đường tròn ngoại tiếp tam giác $ABC$.\\
		Khi đó $PQ \perp AB$, $OM \perp AB$, suy ra $PQ$ song song $OM$. Vì $M$ là trung điểm của $AB$, $Q$ là trung điểm của $MB$ nên $MQ = \dfrac{1}{3} AQ$. Suy ra $OP = \dfrac{1}{3}AP$.\\
		Xét tam giác $AB_1C_1$. Ta có $O$ là tâm đường tròn ngoại tiếp tam giác $AB_1C_1$. $P$ là chân đường cao kẻ từ đỉnh $A$. Mà $A, O, P$ thẳng hàng nên tam giác $AB_1C_1$ cân.\\
		Do đó $AP$ là trung tuyến của tam giác $AB_1C_1$. Do $OP=\dfrac{1}{3}AP$ nên $O$ là trọng tâm của $AB_1C_1$. Mà trọng tâm trùng vs tâm đường tròn ngoại tiếp nên tam giác $AB_1C_1$ đều.	
	}

\begin{ex}%[HSG Croatia năm 2018, lớp 11]%[Le Thanh Hai, Dự án EX-Nat.MO-18]%[1D0G6]
	Hai người chơi thay phiên viết mỗi lần một chữ số, từ trái sang phải. Một người được coi là thua nếu sau lượt của người đó, có một dãy các chữ số
	$$a_1,a_2, \ldots ,a_n$$ sao cho tồn tại một số nguyên $k$ để $\overline{a_ka_{k+1}\ldots a_n}$ là bội của $11$.\\
	Hỏi người nào có thể thắng và cách chơi của anh ấy như thế nào?
	\loigiai{
		Ta sẽ chứng minh người chơi thứ hai có thể thắng bất kể người chơi thứ nhất viết như thế nào. Tất nhiên rằng không ai trong hai người chơi sẽ viết số $0$ vào bất kì lượt nào.\\
		Chú ý rằng $10^r \equiv (-1)^r$ (mod $11$), do đó
		$$\overline{a_ka_{k+1}\ldots a_n}\equiv a_n-a_{n-1}+a_{n-2}-\cdots+(-1)^{k-1}a_k\quad (\mod 11)$$
		Gọi $N_k$ là phần dư khi chia $\overline{a_ka_{k+1}\cdots a_n}$ cho $11$, với $k=1,\ldots,n$. Ta có sau $n$ lượt, các số $N_1,\ldots,N_n$ là các số phân biệt, và trong lượt tiếp theo ta có các số $$a_{n+1}-N_1,\ldots,a_{n+1}-N_n,a_{n+1} \,  (\mod 11)$$ là những số phân biệt kể từ $a_{n+1}\ne 0$. Theo quy nạp, ta kết luận rằng sau mỗi lượt của trò chơi (với $n\le 10$), các số $N_1,\ldots,N_n$ là phân biệt. Bởi vì cách thức mà $N_1,\ldots,N_n$ biến đổi mỗi lượt, ta kết luận được có $n+1$ chữ số, nếu người chơi viết chúng sẽ dẫn anh ấy đến việc thua cuộc.\\
		Giả sử trò chơi kết thúc sau $9$ lượt chơi. Người chơi thứ hai thua cuộc nếu và chỉ nếu sau $9$ lượt, bộ số $\{N_1,N_2,\ldots,N_9\}$ là $\{1,2,\ldots,9\}$. Người chơi thứ hai thắng nếu và chỉ nếu $10$ là một trong các số $N_1,N_2,\ldots,N_9$.\\
		Nếu sau $8$ lượt chơi, $\{N_1,\ldots,N_8\}$ thiếu hai số không liên tiếp nằm giữa $1$ và $10$, khi đó bất kể người chơi đầu tiên chọn trong lượt thứ $9$, một trong các số $N_1,\ldots,N_9$ phải là $10$. Thật vậy, nếu người chơi thứ nhất chọn $X$, khi đó tồn tại $k$ thỏa $N_k=X+1$ (trong lượt thứ $9$). Do đó, sau lượt thứ $9$ ta có 
		$$N_k\equiv X-(X-1)\equiv -1 \equiv 10 (\mod 11).$$
		Ta kết luận được người chơi thứ hai chắc chắn rằng sau $8$ lượt chơi, giữa các số $N_1,\ldots,N_8$ không có hai số liên tiếp. Và người chơi thứ hai có thể đảm bảo trò chơi kéo dài ít nhất $7$ lượt chơi để
		$${N_1,\ldots,N_7}={1,2,\ldots,10}\setminus \{X,Y,Z\}.$$
		Nếu giữa các số $X,Y,Z$ không có hai số liên tiếp, thì người chơi thứ hai có thể chọn bất kì trong chúng.\\
		Nếu $Y=X+1$, thì người chơi thứ hai có thể viết một trong các số $X$ hoặc $X+1$.\\
		Vì vậy, sau lượt thứ $8$, có hai số liên tiếp còn thiếu từ $N_1,\ldots,N_8$. Thật vậy, nếu $Z=X-1$, người chơi thứ hai viết $X$, và nếu $Z\ne X-1$, người chơi thứ hai viết $X+1$.
	}
\end{ex}
