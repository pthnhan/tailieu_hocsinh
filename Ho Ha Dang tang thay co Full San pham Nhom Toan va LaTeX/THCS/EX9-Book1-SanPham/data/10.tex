\begin{vd}
   Tìm nghiệm nguyên của phương trình $$6x+15y+10z=3.$$
    \loigiai{
\begin{itemize}
	\item Ta thấy $10z\, \vdots \,3 $ nên $z \, \vdots \, 3$. Đặt $z=3k$ với $k \in \mathbf{Z}$ ta được
	$$ 6x+15y+30k=3 \Leftrightarrow 2x+5y+10k=1.$$ 
	\item Đưa về giải phương trình với hai ẩn là $x$, $y$ với $k$ là tham số $2x+5y=1-10k$ nên $$x=\dfrac{1-10k-5y}{2}=-5k-2y+\dfrac{1-y}{2}$$.
	Đặt $t=\dfrac{1-y}{2}$ với $t \in \mathbf{Z}$. Ta có $y=1-2t$, $x=-5k+5t-2$ và $z=3k$.
	\item Vậy nghiệm $(x,y,z)$ của phương trình là $(5t-5k-2; 1-2t; 3k)$ với $t$, $k$ là những số nguyên tùy ý.
\end{itemize}  
    }
\end{vd}	
\begin{note}
	\textbf{Kinh nghiệm giải toán}\\
	Trong cách giải trên, ta đã biến đổi phương trình đã cho về dạng $2x+5y=1-10k$, ta có các hệ số của $x$ và $y$ là hai số nguyên tố cùng nhau. Sau đó ta giải phương trình bậc nhất với hai ẩn là $x$ và $y$. ( Xem $\S 2$ Phương trình bậc nhất hai ẩn.)	
\end{note}
\begin{vd}
	Chứng minh rằng phương trình sau không có nghiệm nguyên $$x^2+y^2+z^2=2015.$$
	\loigiai{
		Ta biết rằng số chính phương chẵn thì chia hết cho $4$, còn số chính phương lẻ thì chia cho $4$ dư $1$ và chía cho $8$ cũng dư $1$.\\
		Tổng $x^2+y^2+z^2$ là số lẻ nên trong ba số $x^2$, $y^2$, $z^2$ phải có một số lẻ và hai số chẵn, hoặc ba số đều lẻ.
		\begin{itemize}
			\item Trường hợp 1: Có một số lẻ và hai số chẵn thì vế trái của phương trình $x^2+y^2+z^2=2015$ chia cho $4$ dư $1$, còn vế phải ( là $2015$) chia cho $4$ dư $3$, loại.
			\item Trường hợp 2: Cả ba số đều lẻ thì vế trái của phương trình $x^2+y^2+z^2=2015$ chia cho $8$ dư $3$, còn vế phải ( là $2015$) chia cho $8$ dư $7$, loại.
		\end{itemize}  
	Vậy phương trình đã cho không có nghiệm nguyên.
	}
\end{vd}
\begin{vd}
	Tìm nghiệm nguyên dương của phương trình $$2xyz=x+y+z+16.$$
	\loigiai{
	Do vai trò bình đẳng của $x$, $y$, $z$ ta giả sử $1\leq x \leq y \leq z$. Khi đó $2xyz \leq 3z+16$. Do $z$ nguyên dương nên $2xy \leq 3+\frac{16}{z} \leq 19.$ Từ $2x^2 \leq 2xy \leq 19 \Rightarrow x^2 \leq 9$. Do $x$ nguyên dương nên $$x \in \{1;2;3\}.$$
		\begin{itemize}
			\item Trường hợp $x=1$: thay vào phương trình $2xyz=x+y+z+16$ ta được\\ $2yz-y-z=17\Leftrightarrow 4yz-2y-2z=34\Leftrightarrow (2y-1)(2z-1)=35$. Ta tìm được $(y;z)$ bằng $(1;18)$ và $(3;4)$.
			\item Trường hợp $x=2$: thay vào phương trình $2xyz=x+y+z+16$ ta được\\ $4yz-y-z=18\Leftrightarrow 16yz-4y-4z=72\Leftrightarrow (4y-1)(4z-1)=73$. Ta tìm được $4y-1=1$ nên $4y=2$, loại.
			\item Trường hợp $x=3$: thay vào phương trình $2xyz=x+y+z+16$ ta được\\ $6yz-y-z=19\Leftrightarrow 36yz-6y-6z=114\Leftrightarrow (6y-1)(6z-1)=115$. Ta tìm được $y=1$, $z=4$, loại vì trái giả thiết $x \leq y$.
		\end{itemize}  
		Vậy nghiệm $(x;y;z)$ của phương trình là $(1;1;18), (1;3;4)$ và các hoán vị của nó.
	}
\end{vd}
\begin{note}
	\textbf{Kinh nghiệm giải toán}\\
	Do vài trò bình đẳng của $x$, $y$, $z$ trong phương trình $2xyz=x+y+z+16$, ta đã giả sử $1 \leq x \leq y \leq z$, nhờ đó đã chặn trên được $x \leq 3$.\\
	Ta còn có thể đạt được ước lượng mạnh hơn đó là $x \leq 2$ bằng cách làm như sau:\\
	Với giả sử $z \geq y \geq x \geq 1 $, nếu $x \geq 3$ thì $xy \geq 9$ nên $ 2xyz \geq 18z =3z +15z$.\hfill$(1)$\\
	Ta lại có $3z \geq x+y+z$ và $15z \geq x+y+z +45$.\hfill$(2)$\\
	Từ (1) và (2) suy ra $2xyz \geq x+y+z+45$ trái với đề bài. Vậy $x \leq 2$.
\end{note}
\begin{vd}
	Tìm nghiệm nguyên dương của phương trình $$4xyz=x+2y+4z+3.$$
	\loigiai{
	
		\begin{itemize}
			\item Xét $x=1$, thay vào phương trình  $4xyz=x+2y+4z+3$ ta được $$4yz-2y-4z=4\Leftrightarrow2yz -y-2z=2\Leftrightarrow(y-1)(2z-1)=3.$$
			Từ đó ta có $\heva{&y-1=1\\&2z-1=3}$ hoặc $\heva{&y-1=3\\&2z-1=1.}$
			Suy ra $(y,z)$ bằng $(2;2)$ hoặc $(4;1)$.
			\item Xét $x \geq 2$. Từ phương trình  $4xyz=x+2y+4z+3$ ta có $$2y+4z+3=x(4yz-1) \geq 2(4yz-1)=8yz-2$$
			nên $8yz -2y-4z \leq 5 \Leftrightarrow (2y-1)(4z-1) \leq 6$. Do $4z-1\geq 3$ nên $2y-1 \leq 2$ hay $2y\leq 3$. Do $y$ nguyên dương nên $y=1$.\\
			Thay vào phương trình $4xyz=x+2y+4z+3$ được $4xz-x-4z=5 \Leftrightarrow (x-1)(4z-1)=6$. \\
			Do $4z-1$ là ước lẻ của $6$ và $4z-1 \geq 3$ nên
			$\heva{&4z-1=3\\&x-1=2} \Leftrightarrow \heva{&z=1\\&x=3.}$
		\end{itemize}  
		Vậy nghiệm $(x;y;z)$ của phương trình là $(1;2;2), (1;4;1),(3;1;1)$.
	}
\end{vd}
\begin{note}
	\textbf{Kinh nghiệm giải toán}
	\begin{itemize}
		\item Do vai trò của $x,y,z$ không bình đẳng nên ta không thể giả sử $1\leq x \leq y \leq z$. Trong ví dụ trên, do $x$ nguyên dương nên trước hết ta xét $x=1$, sau đó xét $x\geq 2$. Nhờ $x \geq 2$ mà ta giới hạn được $2y \leq 3$.\\
		Với bài toán tìm nghiệm nguyên dương của phương trình nhiều ẩn, ta thường xét một hoặc vài giá trị của một ẩn, rồi xét tiếp trường hợp còn lại.
		\item Ta cũng có thể đặt $t=2y$ và $u=4z$ đưa phương trình về dạng $$xtu=2x+2t+2u+6$$
		để sắp xếp $1 \leq x \leq t \leq u$ (chú ý rằng có điều kiện $t \, \vdots \, 2 $ và $u \, \vdots \, 4$).
	\end{itemize}
\end{note}

\begin{bt}
    Tìm nghiệm nguyên của mỗi phương trình sau
    \begin{enumerate}[a)]
    	\item $2x+5y-z=4$.
    	\item $2x-5y-6z=4$.
    \end{enumerate}
    \loigiai{
    \begin{enumerate}[a)]
    	\item Ta có $2x+5y-z=4\Leftrightarrow 5y -z =4-2x $, đặt $x=m$, $y=n$ với $m,n \in \mathbf{Z}$, thay vào ta được $z= 2m+5n-4$. Vậy nghiệm $(x;y;z)$ của phương trình là $(m;n; 2m+5n-4)$  với $m,n \in \mathbf{Z}$.
    	\item Xét phương trình $2x-5y-6z=4$, dễ thấy các hệ số của $x,z$ và hệ số tự do $4$ chia hết cho $2$, suy ra $5y  \, \vdots \, 2 \Rightarrow y  \, \vdots \, 2$. Đặt $y=2k$ với $k \in \mathbf{Z}$. Ta có $$x-5k-3z=2\Leftrightarrow x=5k+3z+2$$ 
    	Đặt $z=t$ với $t \in \mathbf{Z}$. Khi đó nghiệm của hệ là $\heva{&x=5k+3t+2\\&y=2k\\&z=t}$ với $k, t \in \mathbf{Z}$.
    \end{enumerate}
    }
\end{bt}

 	}

 \begin{bt} % Bai 74
 	Tìm các số nguyên dương $x,y$ để biểu thức sau là số chính phương.
 	\begin{center}
 		$(x+y+1)^2-2x+2y$.
 	\end{center}
 	\loigiai{
 		Đặt $(x+y+1)^2-2x+2y=z^2$.\hfill$(1)$\\
 		Khi đó:
 	$$z^2=(x+y)^2+2(x+y)+1-2x+2y
 		\Leftrightarrow z^2=(x+y)^2+4y+1.$$
 		Do $y\geq 1$ nên $z^2>(x+y)^2$.\hfill$(2)$\\
 		Do $x \geq 1$ nên $z^2=(x+y)^2+4y+1<(x+y)^2+4(x+y)+4=(x+y+2)^2.$\\
 		Từ đó và (2) suy ra 
 		$(x+y)^2<z^2<(x+y+2)^2$.\\
 		Do đó $z^2=(x+y+1)^2$. Kết hợp với (1) được \\
 		\quad $(x+y+1)^2-2x+2y=(x+y+1)^2\Leftrightarrow 2x=2y \Leftrightarrow x=y$.\\
 		Vậy  $x=y=m$ với $m$ là số nguyên dương tùy ý.
 	}
 \end{bt} 
\begin{bt}
	Chứng minh rằng phương trình sau không có nghiệm nguyên
	$$ x^3+y^3+z^3=2003.$$
	\loigiai{
		Với $a\in\mathbb{Z}$   ta thấy số dư trong phép chia của $a^3$  cho $9$ chỉ có thể là $0,1,8.$ Do đó số dư trong phép chia $x^3+y^3+z^3$ cho $9$ chỉ có thể là $0,1,2,3,6,7,8.$ Trong khi đó $2003$ chia cho $9$ dư $5$ nên phương trình đã cho vô nghiệm (vì hai vế phương trình có số dư khác nhau trong phép chia cho $9$).
	}
\end{bt}

\begin{bt}
	Tìm nghiệm nguyên dương của mỗi phương trình sau
	\begin{enumerate}
		\item [a)] $x^2+y^2+z^2+xyz=13;$
		\item [b)] $xyz=3(x+y+z);$
		\item [c)] $xyz=10(x+y+z).$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item [a)] Giả sử $1\leq x \leq y \leq z.$ Ta có 
			$$ 13=x^2+y^2+z^2+xyz\geq 3x^2+x^3\geq 4x^2 $$ nên $x^2=1$ hay $x=1.$ Khi đó
			$$ 12=y^2+z^2+yz\geq 3y^2 $$ nên $y^2\leq 4$ suy ra $y=1 \vee y=2.$ Thử lại trực tiếp chỉ có $y=2$ ta được $z=2.$
			Vậy phương trình có nghiệm $(1;2;2)$ và các hoán vị.
			\item [b)] Giả sử $1\leq x \leq y \leq z.$ Ta có $xyz\leq 3\cdot 3z=9z$ nên $xy\leq 9.$ Suy ra $x^2\leq 9,$ do đó $x\in \left\{1;2;3\right\}.$
			\begin{itemize}
				\item Với $x=1,$ ta có $yz-3y-3z=3 \Leftrightarrow (y-3)(z-3)=12,$ trong đó $y-3\leq z-3.$ Ta có bảng giá trị 
				\begin{center}
					\begin{tabular}{|c|c|c|c|c|}
						\hline 
						$y-3$ & 1 & 2 & 3  \\ 
						\hline 
						$z-3$ & 12 & 6 & 4  \\ 
						\hline 
					\end{tabular} 
				\end{center}
				nên có nghiệm $(x;y;z)$ là $(1;4;15),(1;5;9),(1;6;7).$
				\item Với $x=2,$ tương tự có nghiệm $(x;y;z)$ là $(2;2;12),(2;3;5).$
				\item Với $x=3,$ tương tự có nghiệm $(x;y;z)$ là $(3;3;3).$
			\end{itemize}
			Vậy nghiệm $(x;y;z)$ là $(1;4;15),(1;5;9),(1;6;7),(2;2;12),(2;3;5),(3;3;3)$ và các hoán vị.
			\item [c)] Giả sử $1\leq x \leq y \leq z.$ Ta có $xyz=10(x+y+z)\leq 30z \Rightarrow xy\leq 30,$ do đó $x^2\leq xy\leq 30.$ Vậy $x\in \left\{1;2;3;4;5\right\}.$
			\begin{itemize}
				\item Với $x=1,$ ta có $yz-10y-10z=10 \Leftrightarrow (y-10)(z-10)=110.$ Ta có bảng giá trị 
				\begin{center}
					\begin{tabular}{|c|c|c|c|c|c|}
						\hline 
						$y-10$ & 1 & 2 & 5 & 10 \\ 
						\hline 
						$z-10$ & 110 & 55 & 22 & 11 \\ 
						\hline 
					\end{tabular} 
				\end{center}
				nên có nghiệm $(x;y;z)$ là $(1;11;120),(1;12;65),(1;15;32),(1;20;21).$
				\item Với $x=2,$ tương tự có nghiệm $(x;y;z)$ là $(2;6;40),(2;10;12).$
				\item Với $x=3,$ tương tự có nghiệm $(x;y;z)$ là $(3;4;35),(3;5;16).$
				\item Với $x=4,$ tương tự có nghiệm $(x;y;z)$ là $(4;5;9).$
				\item Với $x=5,$ phương trình vô nghiệm.
			\end{itemize}
			Vậy nghiệm $(x;y;z)$ là $(1;11;120),(1;12;65),(1;15;32),(1;20;21),(2;6;40),(2;10;12),$ \\ $  (3;4;35),(3;5;16),(4;5;9)$ và các hoán vị.
			
			
		\end{enumerate}        
	}
\end{bt}





\begin{bt}
	Tìm nghiệm nguyên dương của mỗi phương trình sau:
	\begin{enumerate}
		\item [a)] $3xyz=x+y+3z;$
		\item [b)] $5xyz=x+5y-4z+31;$
		\item [c)] $3xyz=x+2y+3z+7;$
		\item [d)] $3xyz=x+3y+4z+5.$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item [a)] Đặt $t=3z,t\in\mathbb{N^*}, t\, \vdots \, 3$ đưa về phương trình $xyt=x+y+t.$ Tìm được $t=3$ (xem thí dụ 7). Nghiệm $(x;y;z)$ là $(1;2;1),(2;1;1).$
			\item [b)] 
			\begin{itemize}
				\item Xét $x=1,$ thay vào phương trình ta được 
				$$ 5yz-5y+4z=32\Leftrightarrow (5y+4)(z-1)=28. $$
				Ta có $5y+4\geq 9$ và $5y+4$ chia $5$ dư $4$ nên 
				$$ \heva {& 5y+4=14\\ & z-1=2}\Leftrightarrow \heva {&y=2\\ & z=3}. $$
				\item Xét $x\geq 2$ ta có 
				$$ 5y-4z+31=x(5yz-1)\geq 2(5yz-1)\Leftrightarrow 10yz-5y+4z\leq 33 $$ hay $(5y+2)(2z-1)\leq 31.$ Vì $5y+2\geq 7$ nên $2z-1\leq 4.$ Do đó $z\in\left\{1;2\right\}.$
				\begin{itemize}
					\item Thay $z=1$ vào phương trình ta được $5xy-x-5y=27\Leftrightarrow (5y-1)(x-1)=28.$ Ta thấy $5y-1$ chia cho $5$ dư $4$ nên 
					\begin{center}
						\begin{tabular}{|c|c|c|c|}
							\hline 
							$5y-1$ & 4 & 14   \\ 
							\hline 
							$x-1$ & 7 & 2   \\ 
							\hline 
						\end{tabular} 
					\end{center}
					từ đó nghiệm $(x;y;z)$ là $(8;1;1),(3;3;1).$
					\item Thay $z=2$ vào phương trình ta được $10xy-x-5y=23\Leftrightarrow (2x-1)(10y-1)=47.$ Không có nghiệm nguyên.
				\end{itemize}
			\end{itemize}
			Vậy nghiệm $(x;y;z)$ là $(8;1;1),(3;3;1).$
			\item [c)] 
			\begin{itemize}
				\item Xét $x=1,$ thay vào phương trình ta được 
				$$ 3yz-2y-3z=8\Leftrightarrow (y-1)(3z-2)=10. $$
				Ta có  $3x-2$ chia $3$ dư $1$ nên 
				$$ \heva {& y-1=1\\ & 3z-2=10} \mbox{hoặc} \heva {& y-1=10\\ & 3z-2=1}$$
				nên có nghiệm $(x;y;z)$ là $(1;2;4),(1;11;1).$
				\item Xét $x\geq 2$ ta có 
				$$ 2y+3z+7=x(3yz-1)\geq  2(3yz-1)\Leftrightarrow 6yz-2y-3z\leq 9 $$ hay $(2y-1)(3z-1)\leq 10.$ Vì $3z-1\geq 2$ nên $2y-1\leq 5.$ Do đó $y\in\left\{1;2;3\right\}.$
				\begin{itemize}
					\item Thay $y=1$ vào phương trình ta được $3xz-x-3z=9\Leftrightarrow (x-1)(3z-1)=10.$ Ta thấy $3z-1$ chia cho $3$ dư $2$ nên 
					\begin{center}
						\begin{tabular}{|c|c|c|c|}
							\hline 
							$x-1$ & 2 & 5   \\ 
							\hline 
							$3z-1$ & 5 & 2   \\ 
							\hline 
						\end{tabular} 
					\end{center}
					từ đó nghiệm $(x;y;z)$ là $(3;1;2),(6;1;1).$
					\item Thay $y=2$ và $y=3$ vào phương trình ta được nghiệm $(x;y;z)$ là $(2;3;1).$
				\end{itemize}
			\end{itemize}
			Vậy nghiệm $(x;y;z)$ là $(1;2;4),(1;11;1),(3;1;2),(6;1;1),(2;3;1) .$
			\item [d)] Xét lần lượt $x=1$ và $x\geq 2$ ta được nghiệm $(x;y;z)$ là $(1;2;6),(1;3;3),(2;1;5),(6;1;1),(3;2;1).$
		\end{enumerate}    
	}
\end{bt}

\begin{bt}
	Tìm nghiệm nguyên của phương trình
	$$ x^3+y^3+z^3=3xyz+1 $$
	\loigiai{
		Ta có $$ x^3+y^3+z^3-3xyz=1\Leftrightarrow (x+y+z)(x^2+y^2+z^2-xy-yz-zx)=1. $$
		Dễ thấy $x^2+y^2+z^2-xy-yz-zx\geq 0$ nên chỉ có thể 
		$$\heva{& x+y+z=1 \\ & x^2+y^2+z^2-xy-yz-zx=1}$$
		Từ đó suy ra $$ \heva {& x^2+y^2+z^2+2xy+2yz+2zx=1 \\& x^2+y^2+z^2-xy-yz-zx=1} $$
		nên $xy+yz+zx=0$ do đó $x^2+y^2+z^2=1.$ Giả sử $x^2\geq y^2 \geq z^2$ suy ra $z=0,y=0, x=1$ hoặc $x=-1.$ Thử lại phương trình có nghiệm $(x;y;z)$ là $(1;0;0)$ và các hoán vị.
	}
\end{bt}


\begin{bt}
	Tìm nghiệm nguyên của mỗi phương trình sau
	\begin{enumerate}
		\item [a)] $x^2+y^2+z^2=x^2y^2;$
		\item [b)] $x^4+y^4+z^4+t^4=2015.$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item [a)] 
			Xét phương trình $x^2+y^2+z^2=x^2y^2(*).$
			Ta có $x^2y^2$ là số chính phương nên khi chia cho $4$ chỉ có số dư là $0,1.$ Nếu $x^2y^2$ chia cho $4$ dư $1$ thì $x,y$ đều là số lẻ. Khi đó $x^2+y^2+z^2$ chia cho $4$ có thể dư $2$ hoặc $3.$ Điều này mâu thuẫn nên $x^2y^2$ chia hết cho $4$ từ đó $x,y$ không thể cùng là số lẻ. 
			\begin{itemize}
				\item Trong hai số $x,y$ có một sô chẵn, một số lẻ. Khi đó $x^2+y^2+z^2$ chia cho $4$ có số dư là $1$ hoặc $2.$ Điều này mâu thuẫn vì hai vế của $(*)$ có số dư khi chia cho $4$ khác nhau.
				\item Nếu $x,y$ đều là số chẵn suy ra $z$ chẵn. Đặt $x=2x_1,y=2y_1,z=2z_1$ với $x_1,y_1,z_1 \in \mathbb{Z}.$ thay vào phương trình ta được $x_1^2+y_1^2+z_1^2=4x_1^2y_1^2.$ Lập luận tương tự và sử dụng phương pháp lùi vô hạn phương trình có nghiệm duy nhất là $(0;0;0).$
			\end{itemize}
			Vậy phương trình có nghiệm duy nhất là $(0;0;0).$
			\item [b)] Với $a\in\mathbb{Z}$ ta thấy số dự trong phép chia của $a^4$   cho $16$ chỉ có thể là $0$  hoặc $1$ do đó số dư trong phép chia $x^4+y^4+z^4+t^4$ cho $16$ chỉ có thể là $0,1,2,3,4.$ Trong khi đó $2015$ chia cho $16$ dư $15$ nên phương trình đã cho vô nghiệm.
		\end{enumerate}        
	}
\end{bt}
