Giải phương trình $x^4-22x^2+25=0$.
\loigiai{Đặt $t=x^2,\ t \geq 0$, ta có phương trình $$t^2-22t+25=0 \Leftrightarrow t= 11\pm 4\sqrt{6}=\left(2\sqrt{2} \pm \sqrt{3}\right)^2.$$
	Từ đó, ta có các nghiệm của phương trình là $x= \pm \left(2\sqrt{2}-\sqrt{3}\right). $}

\begin{ex}%[Đề thi vào 10, chuyên Lương Thế Vinh, tỉnh Đồng Nai, năm 2018]%[Nguyễn Tất Thu, Dự án EX-TS10C-18]%[9D0B1]%[9D0B7]
	Cho biểu thức $P=\left(\dfrac{a}{\sqrt{a}+2}+\dfrac{a+\sqrt{a}}{a+3\sqrt{a}+2}\right)\dfrac{4-a}{\sqrt{a}}$ (với $a$ là số thực dương).
	\begin{enumerate}
		\item Rút gọn biểu thức $P$.
		\item  Tìm $a$ để $P$ đạt giá trị lớn nhất.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Ta có $$\begin{aligned} P&=\left(\dfrac{a}{\sqrt{a}+2}+\dfrac{\sqrt{a}(\sqrt{a}+1)}{(\sqrt{a}+2)(\sqrt{a}+1)}\right)\dfrac{4-a}{\sqrt{a}} \\& =\left(\dfrac{a}{\sqrt{a}+2}+\dfrac{\sqrt{a}}{\sqrt{a}+2}\right)\dfrac{(2+\sqrt{a})(2-\sqrt{a})}{\sqrt{a}} \\&=(\sqrt{a}+1)(2-\sqrt{a})=2+\sqrt{a}-a. \end{aligned} $$
		\item  Ta có $P=\dfrac{9}{4}-\left(\dfrac{1}{2}-\sqrt{a}\right)^2 \leq \dfrac{9}{4}.$ Đẳng thức xảy ra khi $a=\dfrac{1}{4}.$ \\Vậy $a=\dfrac{1}{4}$ là giá trị cần tìm.
	\end{enumerate}
}
	
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Lương Thế Vinh, tỉnh Đồng Nai, năm 2018]%[Nguyễn Tất Thu, Dự án EX-TS10C-18]%[9D0B2]
	
	Giải hệ phương trình $\heva{&x^2-xy=6 \\& 3x^2+2xy-3y^2=30.}$
	\loigiai{
		Từ hệ, ta có 
		$$3x^2+2xy-3y^2=5(x^2-xy) \Leftrightarrow 2x^2-7xy+3y^2=0 \Leftrightarrow (x-3y)(2x-y)=0 \Leftrightarrow x=3y,\ y=2x.$$
		\begin{itemize}
			\item $x=3y$ thay vào hệ ta có $9y^2-3y-6=0 \Leftrightarrow \hoac{&y=1 \rightarrow x=3 \\ &y=-\dfrac{2}{3} \rightarrow x=-2.}$
			\item $y=2x$ thay vào hệ ta có $-x^2=6 $ (vô nghiệm).
		\end{itemize}
		Vậy hệ có hai nghiệm $(x;y)=(3;1),\ (-2;-\dfrac{2}{3})$.
	}
	
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Lương Thế Vinh, tỉnh Đồng Nai, năm 2018]%[Nguyễn Tất Thu, Dự án EX-TS10C-18]%[9D0K7]
		Tìm tất cả các tham số thực $m$ để phương trình $x^2-(m+1)x+2m=0$ có hai nghiệm phân biệt $x_1,\ x_2$ sao cho biểu thức $P=\dfrac{x_1+x_2-1}{(x_1+x_2)^2-3x_1x_2+3}$ đạt giá trị nhỏ nhất.
	\loigiai{
		Phương trình có hai nghiệm phân biệt khi và chỉ khi $$\Delta =(m+1)^2-8m>0 \Leftrightarrow (m-3)^2>8 \Leftrightarrow \hoac{&m>3+2\sqrt{2}\\& m<3-2\sqrt{2}.}$$
		Theo định lí Vi-ét ta có $x_1+x_2=m+1,\ x_1x_2=2m$. Khi đó
		$$P=\dfrac{m}{(m+1)^2-6m+3}=\dfrac{m}{m^2-4m+4} \Rightarrow P+\dfrac{1}{8} =\dfrac{m^2+4m+4}{(m-2)^2}=\dfrac{1}{8}\left(\dfrac{m+2}{m-2}\right)^2 \geq 0.$$
		Do đó $P \geq -\dfrac{1}{8}$. Đẳng thức xảy ra khi $m=-2.$ \\ Vậy $m=-2$ là giá trị cần tìm.	
	}
	
	
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Lương Thế Vinh, tỉnh Đồng Nai, năm 2018]%[Nguyễn Tất Thu, Dự án EX-TS10C-18]%[9D0K5]
	Tìm cặp số nguyên $(x;y)$ thỏa mãn $2x^2-4y^2-2xy-3x-3=0$.
	\loigiai{Ta xét phương trình $2x^2-(2y+3)x-4y^2-3=0.$ \hfill (1)
		
		Ta xem phương trình (1) như là phương trình bậc hai theo ẩn $x$, với tham số là $y$. Lúc này, điều kiện cần để phương trình (1) có nghiệm nguyên là $\Delta =(6y+1)^2+32$ là số chính phương, hay có số tự nhiên $a$ để $$(6y+1)^2+32=a^2 \Leftrightarrow (6y+1+a)(6y+1-a)=-32.$$
		Do $6y+1+a\geq 6y+1-a$ và hai số $6y+1+a,\ 6y+1-a$ là hai số cùng tính chẵn lẻ, nên ta có các trường hợp sau:
		\begin{itemize}
			\item $\heva{&6y+1+a=16 \\ &6y+1-a=-2} \Leftrightarrow \heva{& y=1 \\& a=9}$, thay vào phương trình ta được $x=-1$.
			\item $\heva{&6y+1+a=2 \\ &6y+1-a=-16} \Leftrightarrow \heva{& y=-\dfrac{4}{3} \\& a=9}$ (loại).
			\item $\heva{&6y+1+a=8 \\ &6y+1-a=-4} \Leftrightarrow \heva{& y=\dfrac{1}{6} \\& a=6}$, (loại).
			\item $\heva{&6y+1+a=4 \\ &6y+1-a=-8} \Leftrightarrow \heva{& y=-\dfrac{1}{2} \\& a=6}$, (loại).
		\end{itemize}
		Vậy phương trình đã cho có nghiệm $(x;y)=(-1;1).$}
	
\end{ex}

	$$\dfrac{a^3+b^3}{ab(a^2+b^2)}+\dfrac{b^3+c^3}{bc(b^2+c^2)}+\dfrac{c^3+a^3}{ca(c^2+a^2)} \geq \dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}.$$
	\loigiai{Ta có $$2(a^3+b^3)-(a+b)(a^2+b^2)=a^3+b^3-ab^2-a^2b=(a+b)(a-b)^2 \geq 0.$$
		Do đó $$a^3+b^3 \geq \dfrac{1}{2}(a+b)(a^2+b^2) \Rightarrow \dfrac{a^3+b^3}{ab(a^2+b^2)}\geq \dfrac{1}{2}\left(\dfrac{1}{a}+\dfrac{1}{b}\right).$$
		Tương tự ta cũng có $\dfrac{b^3+c^3}{bc(b^2+c^2)}\geq \dfrac{1}{2}\left(\dfrac{1}{b}+\dfrac{1}{c}\right),\ \dfrac{c^3+a^3}{ca(c^2+a^2)}\geq \dfrac{1}{2}\left(\dfrac{1}{c}+\dfrac{1}{a}\right). $\\
		Cộng các bất đẳng thức trên theo vế, ta có đpcm.}

\begin{ex}%[Đề thi vào 10, chuyên Lương Thế Vinh, tỉnh Đồng Nai, năm 2018]%[Nguyễn Tất Thu, Dự án EX-TS10C-18]%[9H0B1]%[9H0B2]%9H0K1]%[9H0G6]
	Cho đường tròn $(O)$ đường kính $AB$ cố định. Biết điểm $C$ thuộc đường tròn $(O)$, với $C$ khác $A$ và $B$. Vẽ đường kính $CD$ của đường tròn $(O)$. Tiếp tuyến tại $B$ của đường tròn $(O)$ cắt hai đường thẳng $AC$ và $AD$ lần lượt tại $E$ và $F$.
	\begin{enumerate}
		\item Chứng minh tứ giác $ECDF$ nội tiếp đường tròn.
		\item Gọi $H$ là trung điểm của đoạn thẳng $BF$. Chứng minh rằng $OE$ vuông góc với $AH$.
		\item Gọi $K$ là giao điểm của hai đường thẳng $OE$ và $AH$. Chứng minh rằng điểm $K$ thuộc đường tròn ngoại tiếp tứ giác $ECDF$.
		\item Gọi $I$ là tâm của đường tròn $(I)$ ngoại tiếp tứ giác $ECDF$. CHứng minh rằng điểm $I$ luôn thuộc một đường thẳng cố định và đường tròn $(I)$ luôn đi qua hai điểm cố định khi $C$ di động trên đường tròn $(O)$.
		
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\tkzInit[ymin=-3,ymax=7,xmin=-1,xmax=12]
			\tkzClip
			\tkzSetUpPoint[color=black,fill=white,size=6]
			\tkzDefPoints{0/0/A, 4/0/B,2/0/O, 0/4/X}
			\tkzDrawCircle[radius](O,A)
			\tkzInterLC(O,X)(O,A) \tkzGetPoints{Y}{C}
			\tkzDefLine[orthogonal=through B](A,B)\tkzGetPoint{b}
			\tkzDefPointBy[symmetry = center O](C)    \tkzGetPoint{D}
			\tkzInterLL(A,C)(B,b) \tkzGetPoint{E}
			\tkzInterLL(A,D)(B,b) \tkzGetPoint{F}
			\tkzDefMidPoint(B,F) \tkzGetPoint{H}
			\tkzInterLL(E,O)(A,H) \tkzGetPoint{K}
			\tkzDefCircle[circum](E,D,C) \tkzGetPoint{I}
			\tkzDrawCircle[circum](E,D,C)
			\tkzInterLC(A,B)(I,C) \tkzGetPoints{P}{Q}
			
			\tkzDrawSegments(A,B A,E A,F E,F C,D A,H E,K B,D B,P H,D B,C)
			\tkzDrawPoints(A,B,O,C,E,F,D,H,K,I,P,Q)
			\tkzLabelPoints[left](A,I)
			\tkzLabelPoints[right](H,P)
			\tkzLabelPoints[above](E)
			\tkzLabelPoints[above left](C,Q)			
			\tkzLabelPoints[below](D,F,K)
			\tkzLabelPoints[above right](B,O)			
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Ta có $\widehat{CDA}=\widehat{CBA}$ (góc nội tiếp chắn cung $AC$). Mà $\widehat{CBA}=\widehat{CEB}$ (cùng phụ với $\widehat{CAB}$), suy ra $\widehat{CEF}=\widehat{CDA}$. Do đó tứ giác $CDFE$ nội tiếp.
			\item  Ta có $\Delta BDF$ vuông tại $D$ và $H$ là trung điểm $BF$, nên ta có $HB=HD$. Mặt khác $OB=OD$, nên $OH \perp BD$, do đó $OH \perp AC$. Lại có $AO \perp EH$, nên suy ra $O$ là trực tâm tam giác $AHE$, suy ra $OE \perp AH.$
			\item  Ta có tứ giác $EAKB$ nội tiếp (do $\widehat{AKE}=\widehat{ABE}=90^\circ$), nên $\widehat{KAO}=\widehat{OEB}$, suy ra $\Delta AOK \backsim \Delta EOB$, nên ta có $OK\cdot OE =OA\cdot OB=OC\cdot OD \Rightarrow \dfrac{OK}{OC}=\dfrac{OD}{OE}.$\\ Suy ra $\Delta KOD \backsim \Delta COE \Rightarrow \widehat{ODK}= \widehat{OEC}$, hay là $\widehat{CDK}= \widehat{KEC}$. Từ đó, ta có $K$ nằm trên đường tròn ngoại tiếp tứ giác $CDFE.$
			\item  Gọi $P,\ Q$ lần lượt là giao điểm của đường tròn $(I)$ với đường thẳng $AB$.\\ Khi đó, ta có $OP\cdot OQ=OC\cdot OD=R^2$. \hfill (1)\\
			Mặt khác $AP\cdot AQ=AC\cdot AE =AB^2=4R^2$. Mà 
			 \begin{align*} AP \cdot AQ&=(AO-OQ)\cdot (AO+OP)=R^2+AO(OP-OQ)-OP\cdot OQ\\&=R(OP-OQ) \Rightarrow OP-OQ=4R. \tag{2} \end{align*}
			Từ (1) và (2) ta có $OP,\ OQ$ không đổi, hay $P,\ Q$ là các điểm cố định. Do đó, $I$ luôn nằm trên đường trung trực của đoạn $PQ$ là một đường thẳng cố định.
		\end{enumerate}
	}
	
\end{ex}
