	(1,5 điểm)
	\begin{enumerate}
	\item  Tìm $x$ để biểu thức $A=\sqrt{2x-1}$ có nghĩa.
	\item  Không sử dụng máy tính cầm tay, tính giá trị của biểu thức $B=\sqrt{3}\left(\sqrt{3^2\cdot 3}-2\sqrt{2^2\cdot 3}+\sqrt{4^2\cdot 3}\right) $. 
	\item  Rút gọn biểu thức $C=\left(\dfrac{\sqrt{a}}{\sqrt{a}-1}-\dfrac{\sqrt{a}}{a-\sqrt{a}}\right):\dfrac{\sqrt{a}+1}{a-1}$ với $a>0$ và $a\ne 1 $. 
\end{enumerate}
\loigiai{
	\begin{enumerate}
	\item 	$A=\sqrt{2x-1}$ có nghĩa $\Leftrightarrow 2x-1\geqslant 0\Leftrightarrow 2x\geqslant 1\Leftrightarrow x\geqslant \dfrac{1}{2} $. 
	\item 	$B=\sqrt{3}\left(\sqrt{3^2\cdot 3}-2\sqrt{2^2\cdot 3}+\sqrt{4^2\cdot 3}\right)=\sqrt{3}\left({3\sqrt{3}-4\sqrt{3}+4\sqrt{3}}\right)=3\cdot 3-4\cdot 3+4\cdot 3=9 $.
	\item 
	$C=\left(\dfrac{\sqrt{a}}{\sqrt{a}-1}-\dfrac{\sqrt{a}}{a-\sqrt{a}}\right): \dfrac{\sqrt{a}+1}{a-1} =\dfrac{a-\sqrt{a}}{\sqrt{a}\left(\sqrt{a}-1\right)}:\dfrac{\sqrt{a}+1}{(\sqrt{a}-1)(\sqrt{a}+1)}
	 =\sqrt{a}-1$.
\end{enumerate}}
	\end{ex}

		\begin{enumerate}
		\item  Giải phương trình $x^4+3x^2-4=0 $. 
		\item  Cho đường thẳng $d:y=(m-1)x+n $. Tìm các giá trị của $m$ và $n$ để đường thẳng $d$ đi qua điểm $A(1;-1)$ và có hệ số góc bằng $-3 $. 
\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item 	Đặt $t=x^2\geqslant 0$, phương trình đã cho trở thành $t^2+3t-4=0\Leftrightarrow \left[\begin{aligned}& t=1 \\
		& t=-4 \,\text{(L)} 
		\end{aligned}\right.$.\\
		Với $t=1$ ta có $x^2=1 \Leftrightarrow x=\pm 1 $. 
		\item 	Vì đường thẳng $d$ có hệ số góc bằng $-3$ nên ta có $m-1=-3\Leftrightarrow m=-2 $. \\
		Khi đó $d:y=-3x+n$ qua $A(1;-1)$ nên ta có $-1=-3\cdot 1+n\Leftrightarrow n=2 $. \\
		Vậy $m=-2,n=2 $.
	\end{enumerate} }

\begin{ex}%[Tuyển sinh lớp 10 Tỉnh Thừa Thiên Huế 2018-2019, Ex-THCS06]%[8D3B7]
			(1,0 điểm)\\	
			Để phục vụ cho Festval Huế 2018, một cơ sở sản xuất nón lá dự kiến làm ra 300 chiếc nón lá trong một thời gian đã dự định. Do được bổ sung thêm nhân công nên mỗi ngày cơ sở đó làm ra được nhiều hơn 5 chiếc nón lá so với dự kiến ban đầu, vì vậy cơ sở sản xuất đã hoàn thành 300 chiếc nón lá sớm hơn 3 ngày so với dự kiến đã định. Hỏi theo dự kiến ban đầu, mỗi ngày cơ sở đó làm ra bao nhiêu chiếc nón lá. Biết rằng số chiếc nón lá làm ra mỗi ngày là bằng nhau và nguyên chiếc.
		\loigiai{
			Gọi $x$ là số nón lá mà cơ sở đó dự kiến làm ra trong 1 ngày, $x>0$.\\
			Thời gian làm xong là $\dfrac{300}{x}$ ngày.\\
			Số nón làm trong thực tế là $x+5$ chiếc.\\
			Thời gian làm trong thực tê là $\dfrac{300}{x+5}$ ngày.\\
			Theo bài ra ta có 
			$\dfrac{300}{x}-\dfrac{300}{x+5}=3 \Leftrightarrow3x^2+15x-1500=0 \Leftrightarrow \left[\begin{aligned}&x=20\\&x=-25 \, \text{(L)}\end{aligned}\right.$.\\
			Vậy số nón dự định làm ban đầu là 20 chiếc.}
\end{ex}
\begin{ex}%[Tuyển sinh lớp 10 Tỉnh Thừa Thiên Huế 2018-2019, Ex-THCS06]%[9D4B6]
	(2,0 điểm)\\
	Cho phương trình $x^2+2mx+m^2+m=0$ (1) (với $x$ là ẩn số).
		\begin{enumerate}
			\item  Giải phương trình (1) với $m=-1$.				\item  Tìm giá trị của $m$ để phương trình (1) có hai nghiệm phân biệt.
			\item  Tìm giá trị của $m$ để phương trình (1) có hai nghiệm phân biệt $x_1,x_2$ thỏa mãn điều kiện:
				\centerline{$(x_1-x_2)(x_1^2-x_2^2)=32$}
		\end{enumerate}
\loigiai{	\begin{enumerate}
		\item	Với $m=-1$ phương trình (1) trở thành $x^2-2x=0$ (2).\\
				Phương trình
				(2)$\Leftrightarrow x(x-2)=0\Leftrightarrow \left[\begin{aligned}& x=0 \\
				& x=2 
				\end{aligned}\right.$.\\
				Vậy khi $m=-1$ phương trình (1) có tập nghiệm là $S=\left\{{0;2}\right\}$.
		\item 	Phương trình $x^2+2mx+m^2+m=0$ có:
				$\Delta =(2m)^2-4(m^2+m)=-4m$.\\
				Để phương trình (1) có hai nghiệm phân biệt khi và chỉ khi $\Delta >0\Leftrightarrow -4m>0\Leftrightarrow m<0$.\\
				Vậy $m<0$ thì phương trình (1) có hai nghiệm phân biệt.
		\item Ta có 
				\begin{align*}
		&(x_1-x_2)(x_1^2-x_2^2)=32\\
		\Leftrightarrow & (x_1-x_2)^2(x_1+x_2)=32\\
		 \Leftrightarrow& \left[(x_1+x_2)^2-4x_1x_2\right](x_1+x_2)=32 \, (3).
				 		\end{align*} \\
				Áp dụng định lý Viet: $\left\{\begin{aligned}& x_1+x_2=-2m \\
				& x_1x_2=m^2+m 
				\end{aligned}\right.$ \\
				Suy ra (3) trở thành:
				\begin{align*} &\left[(-2m)^2-4(m^2+m)\right](-2m)=32\\
				\Leftrightarrow &8m^2=32\\
				\Leftrightarrow &m=\pm 2
			\end{align*} 
				Vậy $m=-2$ (vì $m<0$).
					\end{enumerate}}
			\end{ex}
\begin{ex}%[Tuyển sinh lớp 10 Tỉnh Thừa Thiên Huế 2018-2019, Ex-THCS06]%[9H3K7] 
		(3,0 điểm)\\
	Cho tam giác $ABC$ cân tại $A. $ Gọi $M$ là điểm bất kỳ nằm trên cạnh $AC$ ($M$ không trùng $A$ và $C$). Một đường thẳng đi qua điểm $M$ cắt cạnh $BC$ tại $I$ và cắt đường thẳng $AB$ tại $N$ sao cho $I$ là trung điểm của đoạn thẳng $MN $. Đường phân giác trong của góc $\widehat{BAC}$ cắt đường tròn ngoại tiếp tam giác $AMN$ tại điểm $D$ ($D$ không trùng $A$). Chứng minh rằng:
		\begin{enumerate}
				\item  $DN=DM$ và $DI\perp MN $. 					\item  Tứ giác $BNDI$ nội tiếp.
				\item  Đường tròn ngoại tiếp tam giác $AMN$ luôn đi qua một điểm cố định (khác điểm $A)$ khi $M$ di chuyển trên cạnh $AC$. 
		\end{enumerate}
	\loigiai{\centerline{
\begin{tikzpicture}[>=stealth,scale=.6, line join = round, line cap = round]
\tkzInit[xmin=-6,xmax=6,ymin=-6,ymax=6]
\tkzDefPoints{-4/0/N,0/-2/D,4/0/M,3/7/A}
\tkzDefCircle[circum](M,D,N)     \tkzGetPoint{L}
\coordinate (I) at ($(N)!0.5!(M)$);
\tkzDrawCircle[circum](M,D,N)
\tkzDefPointBy[reflection = over A--D](I)    \tkzGetPoint{I'}
\tkzInterLL(I,I')(A,N)    \tkzGetPoint{B}
\tkzInterLL(I,I')(A,M)    \tkzGetPoint{C}
\tkzInterLL(A,D)(B,C)    \tkzGetPoint{H}
\tkzInterLL(A,D)(N,M)    \tkzGetPoint{J}
\tkzDrawSegments(A,N A,D A,C B,C M,N)
\tkzLabelPoints[left](B,N)
\tkzLabelPoints[right](A,M,C)
\tkzLabelPoints[below](I,D)
\tkzLabelPoints[below right](H)
\tkzLabelPoints[above right](J)
\tkzDrawPoints(A,B,C,D,M,N,I,H,J)
\end{tikzpicture}}
\begin{enumerate}
		\item 	Trong đường tròn ngoại tiếp tam giác $AMN$, ta có $\widehat{NAD}$ là góc nội tiếp chắn cung $\overset\frown{ND}, \widehat{DAM}$ là góc nội tiếp chắn cung $\overset\frown{MD}$.\\
		Mà $\widehat{NAD}=\widehat{DAM}$ (do $AD$ là tia phân giác góc $\widehat{BAC}$).\\
		Nên sđ $\overset\frown{ND}$ = sđ $\overset\frown{MD} \Rightarrow \overset\frown{ND}=\overset\frown{MD}\Rightarrow ND=MD$. \\
		Trong đường tròn ngoại tiếp $\Delta AMD, I$ là trung điểm của dây $MN$ $\Rightarrow IN=IM$ mà $DN=DM$.\\
		Suy ra $I,D$ thuộc đường trung trực của đoạn thẳng $MN $. \\
		Do đó $DI\perp MN $. 
	\item 	Gọi $H$ là giao điểm của $BC$ và $AD,J$ là giao điểm của $MN$ và $AD. $ \\
	Ta có $\Delta ABC$ cân tại $A$ có đường phân giác $AD$ đồng thời là đường cao nên $AD\perp BC$.\\ 
					Hai tam giác vuông $DIJ$ và $IHJ (\widehat{I}=\widehat{H}=90^\circ)$ có $\widehat{J}$ chung nên $\Delta DIJ\sim \Delta IHJ$ \\
					Suy ra $\widehat{IDJ}=\widehat{JIH}$ mà $\widehat{JIH}=\widehat{BIN}$ (đối đỉnh) nên $\widehat{IDJ}=\widehat{BIN}(1)$.\\
					Mặt khác\\
					 $\widehat{ANM}=\widehat{{ADM}}$ (góc nội tiếp cùng chắn cung $\overset\frown{AM})$ (2)\\
					Từ (1) và (2) suy ra\\ $\widehat{IDJ}+\widehat{ADM}=\widehat{BIN}+\widehat{ANH}\Rightarrow \widehat{IDM}=180^\circ-\widehat{NBI}\Rightarrow \widehat{IDM}+\widehat{NBI}=180^\circ$ \\
					Như đã chứng minh ở câu (a), $\Delta NDM$ cân tại $D$ nên đường cao $DI$ đồng thời là đường phân giác $\Rightarrow \widehat{NDI}=\widehat{IDM}$.\\
					 Do đó $\widehat{NDI}+\widehat{NBI}=180^\circ $. Vậy, tứ giác $NBID$ nội tiếp.
					\item 	Vì tứ giác $NBID$ nội tiếp mà $\widehat{NID}=90^\circ \Rightarrow \widehat{NID}$ chắn cung có số đo $180^\circ$ tức $ND$ là đường kính của đường tròn đó.\\
		 $\Rightarrow \widehat{NBD}=90^\circ$ hay $BD\perp AB$.\\
		$\Delta ABC$ cố định. $\Rightarrow $ đường phân giác trong $AD$ cố định và đường thẳng vuông góc với $AB$ tại $B$ cố định và $D$ là giao điểm của hai đường thẳng cố định này nên $D$ cố định.\\
		Vậy, đường tròn ngoại tiếp tam giác $AMN$ luôn đi qua điểm $D$ cố định.			
	\end{enumerate}}
\end{ex}
 \begin{ex}%[Tuyển sinh lớp 10 Tỉnh Thừa Thiên Huế 2018-2019, Ex-THCS06]%[9H4B2] 
	(1,0 điểm)\\
		Cho hình chữ nhật $ABCD$ với $AB=2a,BC=a $. Khi quay hình chữ nhật $ABCD$ quanh cạnh $AB$ một vòng thì được hình trụ có thể tích $V_1$ và khi quay hình chữ nhật $ABCD$ quanh cạnh $BC$ một vòng thì được hình trụ có thể tích $V_2 $. Tính tỉ số $\dfrac{V_1}{V_2} $. 			
\loigiai{					
		Khi quay hình chữ nhật $ABCD$ quanh cạnh $AB$ một vòng thì được hình trụ có chiều cao $h=AB=2a,$ bán kính đáy $R=BC=a$ nên có thể tích \\
		\centerline{$V_1=\pi R^2h=\pi a^2\cdot 2a=2\pi a^3$ (đvtt).}\\
		Khi quay hình chữ nhật $ABCD$ quanh cạnh $BC$ một vòng thì được hình trụ có chiều cao $h'=BC=a,$ bán kính đáy $R'=CD=2a$ nên có thể tích\\ \centerline{$V_2=\pi R'^2h'=\pi (2a)^2\cdot a=4\pi a^3$ (đvtt).}\\
		Vậy $\dfrac{V_1}{V_2}=\dfrac{2\pi a^3}{4\pi a^3}=\dfrac{1}{2} $.}
\end{ex}
