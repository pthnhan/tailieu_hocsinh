\begin{bt}%[HK2 (2015-2016), Sở GD \& ĐT Bắc Giang]%[0D2B1]
	Tìm tập xác định cùa hàm số $y=\sqrt{3x-4}+\sqrt{5-x}$.
	\loigiai{
Hàm số đã cho xác định khi $\heva{&3x-4\ge 0\\ &5-x\ge 0}\Leftrightarrow \dfrac{4}{3}\le x\le 5$.\\
Vậy tập xác định của hàm số đã cho là $\mathscr{D}=\left[\dfrac{4}{3};5\right]$.	
}
\end{bt}

\begin{bt}%[HK2 (2015-2016), Sở GD \& ĐT Bắc Giang]%[0D4B5]
	Giải bất phương trình $-x^2+3x+4\ge 0$.
	\loigiai{
Đặt $f(x)=-x^2+3x+4$. Ta có $f(x)=0\Leftrightarrow \hoac{&x=-1\\ &x=4.}$\\
Bảng xét dấu
\begin{center}
	\begin{tikzpicture}
	\tkzTabInit[nocadre,lgt=1.5,espcl=2.5,deltacl=.5]
	{$x$ /.7, $f(x)$ /.7}
	{$-\infty$ , $-1$ , $4$ , $+\infty$}
	\tkzTabLine{,-,z,+,z,-,}
	\end{tikzpicture}
\end{center}	
Vậy tập nghiệm của bất phương trình đã cho là $S=[-1;4]$.
}
\end{bt}

\begin{bt}%[HK2 (2015-2016), Sở GD \& ĐT Bắc Giang]%[0D6B3]
	Cho $\sin a+\cos a=\dfrac{1}{3}$. Tính $P=\sin a\cos a$.
	\loigiai{
\begin{align*}
\sin a+\cos a=\dfrac{1}{3}&\Rightarrow \left(\sin a+\cos a\right)^2=\dfrac{1}{9}\\
&\Rightarrow \sin^2a+\cos^2a+2\sin a\cos a=\dfrac{1}{9}\\
&\Rightarrow 1+2\sin a\cos a=\dfrac{1}{9}\\
&\Rightarrow \sin a\cos a=-\dfrac{4}{9}.
\end{align*}
Vậy $P=-\dfrac{4}{9}$.
}
\end{bt}

\begin{bt}%[HK2 (2015-2016), Sở GD \& ĐT Bắc Giang]%[0D4K5]
	Cho hàm số $f(x)=x^2-(2m+1)x+4$, với $m$ là tham số.
	\begin{enumerate}
		\item Tìm $m$ để $f(x)>0$ với mọi $x\in\mathbb{R}$.
		\item Tìm các giá trị của $m$ để phương trình $f(x)=0$ có hai nghiệm phân biệt $x_1$, $x_2$ sao cho $x_1^2+x_2^2=17$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $f(x)>0$ với mọi $x\in\mathbb{R}\Leftrightarrow\heva{&a=1>0\\ &\Delta=(2m+1)^2-16<0}\Leftrightarrow 4m^2+4m-15<0\Leftrightarrow -\dfrac{5}{2}<m<\dfrac{3}{2}$.
	\item PT đã cho có hai nghiệm phân biệt $\Leftrightarrow \Delta>0\Leftrightarrow \hoac{&m>\dfrac{3}{2}\\ &m<-\dfrac{5}{2}.}$\\
	Theo định lí Viet, ta có $\heva{&x_1+x_2=2m+1\\ &x_1x_2=4.}$\\
	Khi đó $x_1^2+x_2^2=17\Leftrightarrow \left(x_1+x_2\right)^2-2x_1x_2=17\Leftrightarrow (2m+1)^2=25\Leftrightarrow\hoac{&m=2\\ &m=-3.}$\\
	Kết hợp với điều kiện ta thấy $m=2$, $m=-3$ là các giá trị thỏa mãn bài toán.
\end{enumerate}
}
\end{bt}

\begin{bt}%[HK2 (2015-2016), Sở GD \& ĐT Bắc Giang]%[0H2B3]
	Cho tam giác $ABC$ có $AB=3\ \mathrm{cm}$, $BC=4\ \mathrm{cm}$ và $CA=6\ \mathrm{cm}$.
	\begin{enumerate}
		\item Tính cosin của góc $A$.
		\item Tính diện tích của tam giác $ABC$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có $\cos A=\dfrac{AB^2+CA^2-BC^2}{2\cdot AB\cdot CA}=\dfrac{9+36-16}{2\cdot 3\cdot 6}=\dfrac{29}{36}$.
	\item Ta có $p=\dfrac{AB+BC+CA}{2}=\dfrac{13}{2}$.\\
	Theo công thức Hê-rông ta được
	$$S=\sqrt{p(p-AB)(p-BC)(p-CA)}=\dfrac{\sqrt{455}}{4}.$$
\end{enumerate}
}
\end{bt}

\begin{bt}%[HK2 (2015-2016), Sở GD \& ĐT Bắc Giang]%[0H2K2]%[0H3K1]
	Trong mặt phẳng với hệ trục $Oxy$, cho hình vuông $ABCD$ có tâm là điểm $I$. Gọi $G$ và $K$ lần lượt là trọng tâm các tam giác $ACD$ và $ABI$.
	\begin{enumerate}
		\item Chứng minh rằng tam giác $AGK$ vuông cân tại $K$.
		\item Tìm tọa độ đỉnh $A$ biết rằng $G(1;-2)$, $K(3;1)$ và điểm $A$ có tung độ dương.
	\end{enumerate}
\loigiai{
\begin{center}
	\begin{tikzpicture}
	\coordinate (a) at (0,0);
	\coordinate (b) at (0,-4);
	\coordinate (c) at (4,-4);
	\coordinate (d) at (4,0);
	\coordinate (i) at ($(a)!.5!(c)$);
	\coordinate (g) at ($(d)!2/3!(i)$);
	\coordinate (x) at ($(b)!.5!(i)$);
	\coordinate (k) at ($(a)!2/3!(x)$);
	\draw (a)--(b)--(c)--(d)--cycle;
	\draw (a)--(c) (b)--(d) (a)--(k)--(g)--(a) (k)--(x);
	\draw (a) node[left] {\footnotesize$A$} (b) node[left] {\footnotesize $B$} (c) node[right] {\footnotesize $C$} (d) node[right] {\footnotesize $D$} (i) node[below] {\footnotesize $I$} (g) node[below] {\footnotesize $G$} (k) node[left] {\footnotesize $K$};
	\end{tikzpicture}
\end{center}
\begin{enumerate}
	\item Ta có\\
	$\vec{AG}=\dfrac{1}{3}\left(\vec{AD}+\vec{AC}\right)=\dfrac{2}{3}\vec{AD}+\dfrac{1}{3}\vec{AB}$.\\
	$\vec{AK}=\dfrac{1}{3}\left(\vec{AI}+\vec{AB}\right)=\dfrac{1}{2}\vec{AB}+\dfrac{1}{6}\vec{AD}\Rightarrow \vec{KG}=\vec{AG}-\vec{AK}=\dfrac{1}{2}\vec{AD}-\dfrac{1}{6}\vec{AB}$.\\
	Đặt $AB=a$, ta được
	\begin{itemize}
		\item $AK=\left|\vec{AK}\right|=\sqrt{\dfrac{1}{4}AB^2+\dfrac{1}{36}AD^2}=\dfrac{a\sqrt{10}}{6}$.
		\item $KG=\left|\vec{KG}\right|=\sqrt{\dfrac{1}{4}AD^2+\dfrac{1}{36}AB^2}=\dfrac{a\sqrt{10}}{6}$.
		\item $\vec{AK}\cdot\vec{KG}=\dfrac{1}{12}\vec{AD}^2-\dfrac{1}{12}\vec{AB}^2=0\Rightarrow \vec{AK}\perp \vec{KG}$.
	\end{itemize}
	Vậy tam giác $AGK$ vuông cân tại $K$.
	\item Ta có
	\begin{itemize}
		\item Phương trình đường thẳng $AK\colon 2x+3y-9=0$.
		\item Do điểm $A\in AK$ nên $A\left(a;\dfrac{9-2a}{3}\right)$.
		\item Vì $KA=KG$ nên
		$$(a-3)^2+\left(\dfrac{9-2a}{3}-1\right)^2=13\Leftrightarrow \hoac{&a=6\\ &a=0.}$$
	\end{itemize}
Từ đó ta được $A(6;-1)$ hoặc $A(0;3)$.\\
Mà điểm $A$ có tung độ dương nên $A(0;3)$ thỏa mãn.
\end{enumerate}
}
\end{bt}

\begin{bt}%[HK2 (2015-2016), Sở GD \& ĐT Bắc Giang]%[0D4G5]
	Giải bất phương trình $\left(x^2-4\right)\sqrt{x+5}+(x+1)\sqrt{x+2}+x^3+2>x^2+6x$.
	\loigiai{
Điều kiện xác định: $x\ge -2$. Ta có
\begin{align*}
\text{BPT}&\Leftrightarrow \left(x^2-4\right)\left(\sqrt{x+5}-2\right)+(x+1)\left(\sqrt{x+2}-2\right)+\left(x^3+x^2-4x-4\right)>0\\
&\Leftrightarrow (x+1)(x-2)\left[\dfrac{x+2}{\sqrt{x+5}+2}+\dfrac{1}{\sqrt{x+2}+2}+(x+2)\right]>0.
\end{align*}
Ta thấy với $x\ge -2$ thì $\dfrac{x+2}{\sqrt{x+5}+2}+\dfrac{1}{\sqrt{x+2}+2}+(x+2)>0$\\
nên BPT đã cho tương đương với $(x+1)(x-2)>0\Leftrightarrow \hoac{&x>2\\ &x<-1.}$\\
Kết hợp với điều kiện ta được tập nghiệm của bất phương trình đã cho là $S=[-2;-1)\cup (2;+\infty)$.
}
\end{bt}
