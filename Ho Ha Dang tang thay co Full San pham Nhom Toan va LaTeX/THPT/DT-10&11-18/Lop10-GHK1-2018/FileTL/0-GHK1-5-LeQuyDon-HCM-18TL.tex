	\begin{enumerate}
         \item Tập hợp điểm $E$ sao cho $2|\vec{EA}+\vec{EB}+\vec{ED}|=3|\vec{EC}+\vec{ED}|$.

         \end{enumerate}

	Cho $\triangle ABC$ vuông cân tại $B$ có phân giác trong $AD$ của góc $\widehat{BAC}$, $AB=a$. Tính $|\left(\sqrt{2}+1\right)\vec{AD}-\sqrt{2}\vec{AB}|$ theo $a$.
	\loigiai{
	\immini{Ta có $AC=\sqrt{BA^2+BC^2}=a\sqrt{2}$.\\
	Do $AD$ là phân giác trong của góc $BAC\\
	\Rightarrow \dfrac{DB}{DC}=\dfrac{AB}{AC}=\dfrac{a}{a\sqrt{2}}=\dfrac{1}{\sqrt{2}}\Rightarrow \dfrac{DB}{BC}=\dfrac{1}{1+\sqrt{2}}$.\\
$\vec{AD}=\vec{AB}+\vec{BD}=\vec{AB}+\dfrac{BD}{BC}\cdot \vec{BC}=\vec{AB}+\dfrac{1}{1+\sqrt{2}}\vec{BC}$}{\begin{tikzpicture}[scale=1,font=\footnotesize,line join = round, line cap = round, >= stealth]

Vậy $\left| \left(\sqrt{2}+1\right)\vec{AD}-\sqrt{2}\vec{AB} \right|=\left|\vec{AC}\right|=a\sqrt{2}$. }
