\begin{ex}%[Sở Nam Định 2016]%[9D1B6]
Cho biểu thức $P=\left (\dfrac{\sqrt{x}+1}{\sqrt{x}-2}-\dfrac{2}{x-4}\right ) \left ( \sqrt{x}-1 +\dfrac{\sqrt{x}-4}{\sqrt{x}}\right )$ (với $x>0$ và $x\neq 4$).
\begin{enumerate}
\item[1)]  Chứng minh $P=\sqrt{x}+3$.
\item[2)]  Tìm các giá trị của  $x$ sao cho $P=x+3$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[1)]  
\begin{align*}
P&=\left (\dfrac{\sqrt{x}+1}{\sqrt{x}-2}-\dfrac{2}{x-4}\right ) \left ( \sqrt{x}-1 +\dfrac{\sqrt{x}-4}{\sqrt{x}}\right ) \\
  &=\left (\dfrac{(\sqrt{x}+2)(\sqrt{x}+1)}{x-4}-\dfrac{2}{x-4}\right ) \left ( \dfrac{x-4}{\sqrt{x}}\right ) \\
  &=\left (\dfrac{x+3\sqrt{x}}{x-4}\right ) \left ( \dfrac{x-4}{\sqrt{x}}\right ) \\
  &=\sqrt{x}+3
\end{align*}
\item[2)]  $\sqrt{x}+3 = x+3 \iff x-\sqrt{x}=0 \iff \sqrt{x}=1$ hoặc $\sqrt{x}=0$ (loại) $\iff x=1$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Sở Nam Định 2016]%[9D4B6]
Cho phương trình $x^2-2(2m+1)x+4m^2-2m+3=0$ ($m$ là tham số).
\begin{enumerate}
\item[1)] Giải phương trình khi $m=2$.
\item[2)] Tìm các giá trị của tham số $m$ để phương trình có hai nghiệm phân biệt $x_1,x_2$ thỏa mãn 
\[(x_1-1)^2+(x_2-1)^2+2(x_1+x_2 -x_1x_2) =18\]
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[1)] $x^2-10x+15=0 \iff x= 5 \pm \sqrt{10}$.
\item[2)] $\Delta'= (2m+1)^2-(4m^2-2m+3) = 6m-2>0 \iff m > \dfrac{1}{3}$. Theo định lý vi-et ta có $x_1+x_2=2(2m+1)$ và $x_1x_2=-2m+3$, khi đó 
$(x_1-1)^2+(x_2-1)^2+2(x_1+x_2 -x_1x_2) =18 \iff (x_1+x_2)^2-4x_1x_2=16 \iff m=1 $ (thỏa mãn)
\end{enumerate}
}
\end{ex}
\begin{ex}%[Sở Nam Định 2016]%[9D3K4]
Giải hệ phương trình  $\heva{&\dfrac{5}{x-2} - \dfrac{2y-4}{y-3}=2 \\&\dfrac{x+2}{x-2} - \dfrac{2}{y-3} =4}$
\loigiai{
\begin{align*}
&\heva{&\dfrac{5}{x-2} - \dfrac{2y-4}{y-3}=2 \\&\dfrac{x+2}{x-2} - \dfrac{2}{y-3} =4} \iff \heva{&\dfrac{5}{x-2} - \dfrac{2}{y-3}=4 \\&\dfrac{4}{x-2} - \dfrac{2}{y-3} =3} 
\iff  \heva{&\dfrac{1}{x-2} =1 \\&\dfrac{4}{x-2} - \dfrac{2}{y-3} =3}   \iff \heva{&x=3 \\&y=5}
\end{align*}
}
\end{ex}

\begin{ex}%[Sở Nam Định 2016]%[9H3K3]%[9H3K4]%[9H3K7]
Cho tam giác $ABC$ có ba góc nhọn $(AB<AC)$ nội tiếp đường tròn tâm $I$. Gọi $H$ là trực tâm và $D,E,F$ lần lượt là chân các đường cao kẻ từ $A,B,C$ của tam giác $ABC$. Kẻ $DK$ vuông góc với đường thẳng $BE$ tại $K$. 
\begin{enumerate}
\item[1)] Chứng minh tứ giác $BECF$ là tứ giác  nội tiếp và tam giác $DKH$ đồng dạng với tam giác $BEC$.
\item[2)] Chứng  minh $\widehat{BED} = \widehat{BEF}$.
\item[3)] Gọi $G$ là tâm đường tròn ngoại tiếp tam giác $DKE$. Chứng minh $IA \perp KG$. 
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1.2]
\tkzDefPoints{0/2/A,-2/-2/B,4/-2/C}
%\tkzCentroid(A,B,C)\tkzGetPoint{G}
\tkzCircumCenter(A,B,C)\tkzGetPoint{I}
\tkzOrthoCenter(A,B,C)\tkzGetPoint{H}
\tkzInterLL(A,B)(C,H) \tkzGetPoint{F}
\tkzInterLL(C,A)(B,H) \tkzGetPoint{E}
\tkzInterLL(B,C)(A,H) \tkzGetPoint{D}
\tkzDefLine[orthogonal=through D](B,E) \tkzGetPoint{K'}
\tkzInterLL(B,E)(D,K') \tkzGetPoint{K}
\tkzDefMidPoint(D,E)\tkzGetPoint{G}
\tkzDefPointBy[translation = from E to F](A) \tkzGetPoint{A'}
\tkzDefPointBy[translation = from K to G](A) \tkzGetPoint{A''}
\tkzDrawCircle(I,A)
\tkzLabelPoints[above](A,K) \tkzLabelPoints[left](B)
\tkzLabelPoints[right](C)\tkzLabelPoints[below](D,I) 
\tkzLabelPoints[above right=-2pt](E) \tkzLabelPoints[above left=-2pt](F) 
\tkzLabelPoints[below right=-4pt](G) 
\tkzLabelPoints[shift={(-0.6,0.7)}](H) 
\tkzLabelPoint[above](A'){$t$}
\tkzMarkRightAngle(B,K,D) 
\tkzMarkRightAngle(C,D,A) 
\tkzMarkRightAngle(A',A,I)

\end{tikzpicture}

\begin{enumerate}
\item[1)] Ta có $\widehat{ABE} + \widehat{BAC} = \widehat{ACF} + \widehat{BAC} =90^{\circ} \Longrightarrow \widehat{ABE} =  \widehat{ACF} $ cho nên tứ giác $BECF$ nội tiếp. Hai tam giác vuông $DHK$ và $BEC$ có 
$\widehat{DHK} = \widehat{DAC} =\widehat{EBC}$ cho nên đồng dạng.
\item[2)] Dễ thấy các tứ giác $DHEC$, $AEHF$ nội tiếp  và $\widehat{BAD} =\widehat{BCF}$ cho nên
$\widehat{BED} = \widehat{BEF}$.
\item[3)] Tam giác $DEF$ vuông cho nên $G$ là trung điểm của $DE$, tam giác $KGE$ cân tại $G$ suy ra $\widehat{GKE} =\widehat{BED} = \widehat{BEF}$ cho nên $KG \parallel EF$.  Ta cần chứng minh $EF \perp IA$. Qua $A$ dựng tiếp tuyến $At$, ta có $\widehat{BAt} = \widehat{BCA} = \widehat{AFE}$ cho nên $At \parallel EF$
kéo theo $EF \perp IA$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Sở Nam Định 2016]%[9D5G2]
Giải phương trình $2(x+1)\sqrt{x} + \sqrt{3(2x^3+5x^2+4x+1)} = 5x^3-3x^2+8$.
\loigiai{Điều kiện xác định $x \geq 0$. \\
Áp dụng BĐT Cauchy cho hai số thực không âm ta có : \\
$2(x+1)\sqrt{x} \leq (x+1)^2$\\ $\sqrt{3(2x^3+5x^2+4x+1)} = \dfrac{1}{2} \sqrt{12(2x^3+5x^2+4x+1)} \leq$
$\dfrac{1}{4} ( 12+ 2x^3+5x^2+4x+1)$\\
Cho nên $VT \leq \dfrac{1}{4} (2x^3+9x^2+12x+17)$. \\
Xét hiệu $5x^3-3x^2+8 - \dfrac{1}{4} (2x^3+9x^2+12x+17) = \dfrac{3}{4} ( 6x^3-7x^2-4x+5) = \dfrac{3}{4} (x-1)^2(6x+5) \geq  0$ với mọi $x \geq 0$. Vậy $2(x+1)\sqrt{x} + \sqrt{3(2x^3+5x^2+4x+1)} \leq  5x^3-3x^2+8$, dấu bằng xảy ra khi $x=1$, cho nên phương trình có nghiệm duy nhất $x=1$. 
}
\end{ex}
