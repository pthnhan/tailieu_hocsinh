\begin{ex}%[THPT Chuyên Long An, 2017-2018]%[1D1Y1-3]
	Trong các hàm số sau đây, hàm số nào là hàm số \textbf{chẵn}?
	\choice
	{\True $y=\cos 2x$}
	{$y=\cot 2x$}
	{$y=\tan 2x$}
	{$y=\sin 2x $}
	\loigiai{	
	Hàm số $y=\cos 2x$.
	\begin{itemize}
		\item $\mathscr{D}=\mathbb{ R}$.
		\item $\forall x \in \mathscr{D} \Rightarrow -x \in  \mathscr{D}$ và $f(-x)= \cos (-2x) = \cos 2x = f(x)$.
	\end{itemize}
Vậy $y=\cos 2x$ là hàm số chẵn.

Hàm số $y=\cot 2x$.
\begin{itemize}
	\item $\mathscr{D}=\mathbb{ R} \backslash \left\lbrace \dfrac{\pi}{4}+k\dfrac{\pi}{2}\right\rbrace , k \in \mathbb{ Z}$.
		\item $\forall x \in \mathscr{D} \Rightarrow -x \in  \mathscr{D}$ và $f(-x)= \cot (-2x) = - \cot 2x = -f(x)$.
\end{itemize}
Vậy $y=\cot 2x$ là hàm số lẻ.

Hàm số $y=\tan 2x$.
\begin{itemize}
	\item $\mathscr{D}=\mathbb{ R} \backslash \left\lbrace k\dfrac{\pi}{2}\right\rbrace , k \in \mathbb{ Z}$.
	\item $\forall x \in \mathscr{D} \Rightarrow -x \in  \mathscr{D}$ và $f(-x)= \tan (-2x) = - \tan 2x = -f(x)$.
\end{itemize}
Vậy $y=\tan 2x$ là hàm số lẻ.

Hàm số $y=\sin 2x$.
\begin{itemize}
	\item $\mathscr{D}=\mathbb{ R}$.
	\item $\forall x \in \mathscr{D} \Rightarrow -x \in  \mathscr{D}$ và $f(-x)= \sin (-2x) =- \sin 2x = -f(x)$.
\end{itemize}
Vậy $y=\sin 2x$ là hàm số lẻ.
}
\end{ex}

	Giải phương trình:
	\begin{enumerate}
		\item $\sin x+\sqrt{3}\cos x=\sqrt{2}$.
		\item $\sin^2x-2\cos x+2=0$.
		\item $\sin x+\sin 2x+\sin 3x=\cos x+\cos 2x+\cos 3x$.
	\end{enumerate}
	\loigiai{
a) Ta có	
			\begin{eqnarray*}
			&& \sin x+\sqrt{3}\cos x=\sqrt{2}\\
			& \Leftrightarrow &\dfrac{1}{2}\sin x+\dfrac{\sqrt{3}}{2}\cos x=\dfrac{\sqrt{2}}{2}\\
			&\Leftrightarrow &\sin \left(x+\dfrac{\pi}{3}\right)=\dfrac{\sqrt{2}}{2}\\
		&	\Leftrightarrow &\hoac{&x=-\dfrac{\pi}{12}+k2\pi\\&x=\dfrac{5\pi}{12}+k2\pi}, k\in \mathbb{ Z}.
			\end{eqnarray*}
b) Ta có
\begin{eqnarray*}
	&&\sin^2x-2\cos x+2=0\\ &\Leftrightarrow &-\cos^2 x-2\cos x+3=0\\ &\Leftrightarrow &\hoac{&\cos x=1\\&\cos x=-3\text{ (vô nghiệm)}}\\ &\Leftrightarrow &x=k2\pi, k\in \mathbb{ Z}.
\end{eqnarray*}
c) Ta có
\begin{eqnarray*}
	&& \sin x+\sin 2x+\sin 3x=\cos x+\cos 2x+\cos 3x\\ &\Leftrightarrow &2\sin 2x\cos x+\sin 2x=2\cos 2x\cos x+\cos 2x\\ &\Leftrightarrow &(2\cos x+1)(\sin 2x-\cos 2x)=0\\ &\Leftrightarrow &\hoac{&\cos x=-\frac{1}{2}\\ 
		&\sin 2x=\cos 2x}\\ &\Leftrightarrow &\hoac{&x=\pm \frac{2\pi}{3}+k2\pi\\&x=\frac{\pi}{8}+k\dfrac{\pi}{2}},k\in \mathbb{ Z}.
\end{eqnarray*}		
	}
\end{bt}
