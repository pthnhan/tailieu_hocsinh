\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Quận 6 - Hồ Chí Minh]%[8D1Y1]%[8D2B5]
	Thự hiện phép tính
	\begin{enumerate}
		\item $x(x-2)-(x-3)^2$.
		\item $\dfrac{4}{x}+\dfrac{-12}{x(x+3)}+\dfrac{5}{x+3}$.
	\end{enumerate}	
\loigiai{
	\begin{enumerate}
	\item Ta có $x(x-2)-(x-3)^2=x^2-2x-x^2+6x-9=4x-9$.
	\item Ta có $\dfrac{4}{x}+\dfrac{-12}{x(x+3)}+\dfrac{5}{x+3}=\dfrac{4(x+3)-12+5x}{x(x+3)}=\dfrac{9x}{x(x+3)}=\dfrac{9}{x+3}$.
\end{enumerate}		
	}
\end{bt}
\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Quận 6 - Hồ Chí Minh]%[8D1Y7]%[8D1B8]
	Phân tích đa thức thành nhân tử
	\begin{enumerate}
		\item $4x^2+8x+4$.
		\item $x^2-xy+5x-5y$.
	\end{enumerate}	
\loigiai{
	\begin{enumerate}
		\item Ta có $4x^2+8x+4=4(x^2+2x+1)=4(x+1)^2$.
		\item Ta có $x^2-xy+5x-5y=x(x-y)+5(x-y)=(x-y)(x+5)$.
	\end{enumerate}	
	}
\end{bt}
\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Quận 6 - Hồ Chí Minh]%[8D2Y1]%[8D2Y1]%[8D2Y9]
	Một công ty may phải sản xuất $12 000$ sản phẩm trong $x$ ngày. Khi
	thực hiện đã làm xong sớm hơn một ngày.
	\begin{enumerate}
		\item Hãy biểu diễn qua $x$ số sản phẩm phải sản xuất trong một ngày theo kế hoạch.
		\item Hãy biểu diễn qua $x$ số sản phẩm thực tế đã làm trong một ngày.
		\item Tính số sản phẩm thực tế đã làm trong một ngày với $x = 21$.
	\end{enumerate}	
\loigiai{
	\begin{enumerate}
		\item Số sản phẩm phải sản xuất trong một ngày theo kế hoạch là $\dfrac{12000}{x}$.
		\item Số sản phẩm thực tế đã làm trong một ngày $\dfrac{12000}{x-1}$.
		\item Với $x = 21$, ta có số sản phẩm thực tế đã làm trong một ngày là $\dfrac{12000}{21-1}=600$.
	\end{enumerate}		
	}
\end{bt}
\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Quận 6 - Hồ Chí Minh]%[8D1K2]
	Tìm ba số tự nhiên lẻ liên tiếp, biết tích của hai số sau lớn hơn tích của
	hai số đầu là $196$.
\loigiai{
	Gọi $3$ số tự nhiên liên tiếp là $n$, $n+1$, $n+2$ ($n\in\mathbb{N}$).\\
	Theo bài ra ta có phương trình
	$$n(n+1)+196=(n+1)(n+2)\Leftrightarrow n^2+n+196=n^2+3n+2\Leftrightarrow 2n=194\Leftrightarrow n=97.$$	
	Vậy $3$ số tự nhiên liên tiếp cần tìm là $97$, $98$, $99$.
	}
\end{bt}

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Quận 6 - Hồ Chí Minh]%[8H1B9]
	Một gian phòng có nền hình chữ nhật với kích thước là $3{,}6$ m và
	$5{,}8$ m, có một cửa sổ hình chữ nhật kích thước là $0{,}8$ m và $1{,}2$ m và một cửa ra vào hình	chữ nhật kích thước $1{,}2$ m và $2$ m.
	\begin{enumerate}
	\item Tính diện tích nền nhà.
	\item Tính diện tích cửa sổ, diện tích cửa ra vào.
	\item Ta coi một gian phòng đạt mức chuẩn về ánh sáng nếu diện tích các cửa bằng $20\%$ diện tích nền nhà. Hỏi gian phòng trên có đạt mức chuẩn về ánh sáng hay không?
	\end{enumerate}
\loigiai{
	\begin{enumerate}
	\item Tính diện tích nền nhà là $3{,}6\cdot 5{,}8=20{,}88$ m$^2$.
	\item Tính diện tích cửa sổ và diện tích cửa ra vào là $0{,}8\cdot 1{,}2+1{,}2\cdot 2=3{,}36$ m$^2$.
	\item Tỉ lệ phần trăm diện tích các cửa và diện tích nền nhà là $\dfrac{3{,}36}{20{,}88}\approx 16\% < 20\%$. Vậy gian phòng trên không đạt mức chuẩn về ánh sáng.
	\end{enumerate}		
	}
\end{bt}
\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Quận 6 - Hồ Chí Minh]%[8D1G3]
	Cho $a-b = 12$ và $a^2+b^2=154$ . Tính $a^3+b^3$.
\loigiai{
	Ta có $\left\{\begin{aligned}&a-b=12\\&a^2+b^2=154\end{aligned}\right.\Rightarrow \left\{\begin{aligned}&a^2-2ab+b^2=144 \\&a^2+b^2=154\end{aligned}\right.\Rightarrow \left\{\begin{aligned}&ab=5\\&a^2+b^2=154\\&(a+b)^2=164& \end{aligned}\right.\Rightarrow \left\{\begin{aligned}&ab=5\\&a^2+b^2=154\\&a+b=2\sqrt{41}.\end{aligned}\right.$\\
	Do đó $a^3+b^3=(a+b)(a^2-ab+b^2)=2\sqrt{41}\cdot \left(154-5\right)=298\sqrt{41}$.	
	}
\end{bt}
