\begin{bt}%[Thi GHK, Nguyễn Tất Thành - Hà Nội, 2018]%[Lê Quốc Hiệp, EX-DT-10&11]%[0D2B1-2]
	Tìm tập xác định $\mathscr{D}$ của hàm số $f(x)=\dfrac{x^2-3}{\sqrt{2x-3}}+2\sqrt{2-x}$.
	\loigiai
	{
		Hàm số đã cho xác định khi
		\[\heva{&2x-3>0\\&2-x\ge0}\Leftrightarrow\heva{&x>\dfrac{3}{2}\\&x\le2}\Leftrightarrow \dfrac{3}{2}<x\le2.\]
		Vậy $\mathscr{D}=\left(\dfrac{3}{2};2\right]$.
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Tất Thành - Hà Nội, 2018]%[Lê Quốc Hiệp, EX-DT-10&11]%[0D1K3-1]
	Cho tập $A=(-1;3]$ và $B=[m-2;m+3]$. Tìm $m$ để $A\cap B=\varnothing$.
	\loigiai
	{
		Ta có $A\cap B=\varnothing$ khi
		\[\hoac{&m-2>3\\&m+3\le-1}\Leftrightarrow\hoac{&m>5\\&m\le-4.}\]
		Vậy $m\le-4$ hoặc $m>5$.
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Tất Thành - Hà Nội, 2018]%[Lê Quốc Hiệp, EX-DT-10&11]%[0D2B2-1]
	Tìm $m$ để hàm số $f(x)=4mx+3-(5+2m)x$ đồng biến trên $\mathbb{R}$.
	\loigiai
	{
		Ta có $f(x)=4mx+3-(5+2m)x=(2m-5)x+3$.\\
		Để hàm số $f(x)$ đồng biến trên $\mathbb{R}$ thì
		\[2m-5>0\Leftrightarrow m>\dfrac{5}{2}.\]
		Vậy $m>\dfrac{5}{2}$.
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Tất Thành - Hà Nội, 2018]%[Lê Quốc Hiệp, EX-DT-10&11]%[0D2B1-4]
	Xét tính chẵn, lẻ của hàm số $f(x)=2x^4-5x^2+3$.
	\loigiai
	{
		Hàm số đã cho có tập xác định $\mathscr{D}=\mathbb{R}$.\\
		Ta có
		\begin{itemize}
			\item $\forall x\in\mathscr{D}\Rightarrow -x\in\mathscr{D}$.
			\item $f(-x)=2(-x)^4-5(-x)^2+3=2x^4-5x^2+3=f(x)$.
		\end{itemize}
	Vậy hàm số $f(x)=2x^4-5x^2+3$ là hàm số chẵn.
	}
\end{bt}

		\item Tìm $m$ để phương trình $-x^2-4x=m+3$ có hai nghiệm âm phân biệt.

\begin{bt}%[Thi GHK, Nguyễn Tất Thành - Hà Nội, 2018]%[Lê Quốc Hiệp, EX-DT-10&11]%[0D2K3-2]
	Tìm $a,~b,~c$ để đồ thị hàm số $y=ax^2+bx+c$ là đường parabol có đỉnh $I(2;-2)$ và đi qua điểm $A(0;2)$.
	\loigiai
	{
		Xét parabol $(P)\colon y=ax^2+bx+c$.\\
		Ta có $A(0;2)\in(P)\Leftrightarrow a\cdot0+b\cdot0+c=2\Leftrightarrow c=2.\hfill(1)$\\
		Vì $I(2;-2)\in(P)\Leftrightarrow a\cdot4+b\cdot2+c=-2\Leftrightarrow 4a+2b+c=-2.\hfill(2)$\\
		Mặt khác, hoành độ đỉnh của $(P)$ là $-\dfrac{b}{2a}$ nên ta có $-\dfrac{b}{2a}=2\Leftrightarrow 4a+b=0.\hfill(3)$\\
		Từ (1), (2) và (3) ta có hệ phương trình 
		\[\heva{&c=2\\&4a+b=0\\&4a+2b+c=-2}\Leftrightarrow\heva{&a=1\\&b=-4\\&c=2.}\]
		Vậy $a=1,~b=-4$ và $c=2$.
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Tất Thành - Hà Nội, 2018]%[Lê Quốc Hiệp, EX-DT-10&11]%[0H1K3-5]
	Cho tam giác $ABC$ có trọng tâm $G$ và hai điểm $P,~Q$ thỏa mãn $\vec{PA}=2\vec{PB},~3\vec{QA}=-2\vec{QC}$. Chứng minh rằng ba điểm $P,~Q,~G$ thẳng hàng.
	\loigiai
	{
		\immini
		{
			Ta có
			\begin{align*}
			&\vec{PA}=2\vec{PB}\\
			 & \Leftrightarrow \vec{PG}+\vec{GA}=2\vec{PG}+2\vec{GB}\\
			& \Leftrightarrow \vec{PG}=\vec{GA}-2\vec{GB}.\tag{1}
			\end{align*}
			Mặt khác
		}
		{\begin{tikzpicture}[line cap=round,line join=round,font=\footnotesize,>=stealth,scale=1]
			\fill (0,0) coordinate [label=left:$A$] (A) circle(1pt)
			(0:3) coordinate [label=below:$B$] (B) circle(1pt)
			(80:4) coordinate [label=above:$C$] (C) circle(1pt)
			($2*(B)-(A)$) coordinate [label=above:$P$] (P) circle(1pt)
			($3/5*(A)+2/5*(C)$) coordinate [label=left:$Q$] (Q) circle(1pt);
			\tkzCentroid(A,B,C)\tkzGetPoint{G}
			\fill (G) node[above]{$G$} circle(1pt);
			\draw (A)--(B)--(C)--cycle (Q)--(P)--(B);
			\end{tikzpicture}}
		\begin{align*}
		&3\vec{QA}=-2\vec{QC}\\
		\Leftrightarrow & 3\vec{QG}+3\vec{GA}=-2\vec{QG}-2\vec{GC}\\
		\Leftrightarrow & 5\vec{QG}=-3\vec{GA}+2\vec{GA}+2\vec{GB}\\
		\Leftrightarrow & 5\vec{QG}=-\vec{GA}+2\vec{GB}\\
		\Leftrightarrow & \vec{QG}=-\dfrac{1}{5}\vec{GA}+\dfrac{2}{5}\vec{GB}.\tag{2}
		\end{align*}
		Từ (1) và (2) suy ra 
		\[\vec{QG}=-\dfrac{1}{5}\vec{PG}.\]
		Do đó ba điểm $P,~Q,~G$ thẳng hàng.
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Tất Thành - Hà Nội, 2018]%[Lê Quốc Hiệp, EX-DT-10&11]%[0H1G3-7]
	Cho tam giác đều $ABC$ cạnh $a$ nội tiếp đường tròn $(O)$. Điểm $M$ thuộc $(O)$. Tìm giá trị lớn nhất, nhỏ nhất của $\left|\vec{MA}+\vec{MB}-\vec{MC}\right|$.
	\loigiai
	{
		\immini
		{Gọi $I$ là đỉnh thứ tư của hình bình hành $ACBI$.\\
		Ta có
		\[\vec{IA}+\vec{IB}-\vec{IC}=\vec{0}.\]
		Với mọi điểm $M$ ta có
		\begin{eqnarray*}
			 \vec{MA}+\vec{MB}-\vec{MC}&=&\vec{MI}+\vec{IA}+\vec{MI}+\vec{IB}-\left(\vec{MI}+\vec{IC}\right) \\
			&=&\vec{MI}.
		\end{eqnarray*}
		Khi đó $\left|\vec{MA}+\vec{MB}-\vec{MC}\right|=\left|\vec{MI}\right|=MI$.	
		}
		{\begin{tikzpicture}[line cap=round,line join=round,font=\footnotesize,>=stealth,scale=1]
			\fill (-30:2) coordinate [label=below right:$A$] (A) circle(1pt)
			(90:2) coordinate [label=above:$B$] (B) circle(1pt)
			(210:2) coordinate [label=below left:$C$] (C) circle(1pt)
			($(A)+(B)-(C)$) coordinate [label=above:$I$] (I) circle(1pt)
			(0,0) coordinate [label=above:$O$] (O) circle(1pt);
			\tkzInterLC[R](O,I)(O,2 cm) \tkzGetPoints{x}{y}
			\tkzInterLL(I,C)(B,A)\tkzGetPoint{H}
			\fill (y) node[shift={(80:2ex)}]{$M$} circle(1pt)
			(H) node[shift={(80:2ex)}]{$H$} circle(1pt);
			\draw (0,0) circle (2) (A)--(B)--(C)--cycle (I)--(A) (C)--(I)--(B);
			\end{tikzpicture}}
		\noindent Như vậy $MI$ lớn nhất khi $M$ trùng với điểm $C$ và $MI$ nhỏ nhất khi $M$ là giao điểm của $CI$ với đường tròn $(O)$ ($M$ khác $C$).\\
		Gọi $H$ là tâm hình thoi $ACBI$, suy ra $CI=2CH=2\cdot\dfrac{a\sqrt{3}}{2}=a\sqrt{3}$.\\
		Mặt khác, $MI=OC=\dfrac{2}{3}CH=\dfrac{2}{3}\cdot\dfrac{a\sqrt{3}}{2}=\dfrac{a\sqrt{3}}{3}$.\\
		Vậy giá trị lớn nhất, nhỏ nhất của $\left|\vec{MA}+\vec{MB}-\vec{MC}\right|$ lần lượt bằng $a\sqrt{3},~\dfrac{a\sqrt{3}}{3}$.
		
	}
\end{bt}
