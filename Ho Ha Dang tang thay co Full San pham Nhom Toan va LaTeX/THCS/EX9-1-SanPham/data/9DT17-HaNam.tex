\begin{ex}%[Đề thi vào 10, Sở giáo dục Hà Nam, 2017]%[9D4B2]%[9D4K6]
Trong mặt phẳng tọa độ $Oxy$, cho parabol $y=\dfrac{-x^{2}}{2}\ (P)$ và đường thẳng $y=x+m\ (d)$.
\begin{enumerate}
\item Tìm tọa độ điểm $M$ thuộc parabol $(P)$ biết điểm $M$ có tung độ bằng $-8$.
\item Tìm $m$ để đường thẳng $(d)$ cắt parabol $(P)$ tại hai điểm $A(x_{1};y_{1})$, $B(x_{2};y_{2})$ phân biệt sao cho $(x_{1}+y_{1})(x_{2}+y_{2})=\dfrac{33}{4}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Vì $M$ thuộc $(P)$ có tung độ bằng $-8$ nên $-\dfrac{x^{2}}{2}=-8\Leftrightarrow x=\pm 4$. Vậy $M(-4;-8)$ hoặc $M(4;-8)$.
\item Phương trình hoành độ giao điểm $-\dfrac{x^{2}}{2}=x+m\Leftrightarrow x^{2}+2x+2m=0$. Để $(P)$ cắt $(d)$ tại hai điểm phân biệt thì $\heva{
&a\neq 0\\
&\Delta >0
}\Leftrightarrow m<\dfrac{1}{2}$.\\
Ta có $y_{1}=x_{1}+m$, $y_{2}=x_{2}+m$ nên\\
$(x_{1}+y_{1})(x_{2}+y_{2})=\dfrac{33}{4}\Leftrightarrow 4x_{1}x_{2}+2m(x_{1}+x_{2})+m^{2}=\dfrac{33}{4}\Leftrightarrow 4m^{2}+16m-33=0\Leftrightarrow \hoac{
&m=\dfrac{3}{2}(\mbox{loại})\\
&m=-\dfrac{11}{2}(\mbox{TM})
}$
\end{enumerate}
}
\end{ex}

\item Cho biểu thức $B=\left(\dfrac{1}{\sqrt{x}+1}+\dfrac{1}{\sqrt{x}-1}\right)\left(\dfrac{\sqrt{x}-1}{\sqrt{x}}\right)$ (với $x>0$, $x\neq 1$).\\

\begin{ex}%[Đề thi vào 10, Sở giáo dục Hà Nam, 2017]%[9H3B7]%[9H2B6]%[9H3K4]%[9H3G7]
Cho đường tròn $(O)$. Từ một điểm $M$ nằm ngoài đường tròn $(O)$, kẻ hai tiếp tuyến $MA$, $MB$ đến đường tròn $(O)$ (với $A$, $B$ là các tiếp điểm). Gọi $BE$ là đường kính của đường tròn $(O)$ và $F$ là giao điểm thứ hai của đường thẳng $ME$ với đường tròn $(O)$. Đường thẳng $AF$ cắt $MO$ tại điểm $N$. Gọi $H$ là giao điểm của $MO$ và $AB$.
\begin{multicols}{2}		
\columnsep=2pt 		
\columnseprule=0pt 		
\begin{enumerate}
\item Chứng minh $MAOB$ là tứ giác nội tiếp.
\item Chứng minh $AE\parallel MO$.
\item Chứng minh $MN^{2}=NF.NA$.
\item Chứng minh $MN=NH$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzInit[xmin=-4.5,xmax=8,ymin=-4.5,ymax=4.5]
\tkzClip
\tkzDefPoints{-4/0/M, 5/0/O,2/0/x}
\tkzTangent[from=M](O,x)\tkzGetPoints{B}{A}
\tkzInterLC(B,O)(O,x)\tkzGetPoints{E'}{E}
\tkzInterLC(M,E)(O,x)\tkzGetPoints{F}{F'}
\tkzInterLL(A,F)(M,O)\tkzGetPoint{N}
\tkzInterLL(A,B)(M,O)\tkzGetPoint{H}
\tkzDrawCircle(O,x)
\tkzMarkRightAngles(M,A,O M,B,O O,H,A)
\tkzDrawSegments(M,O M,A M,B O,A O,B B,E M,E A,N A,B H,F B,F A,E)
\tkzDrawPoints(O,H,N,M,A,B,E,F)
\tkzLabelPoints[right](O)
\tkzLabelPoints[left](M)
\tkzLabelPoints[above](A,F,E)
\tkzLabelPoints[below](N,B)
\tkzLabelPoints[below right](H)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Vì $MA$, $MB$ là tiếp tuyến của $(O)$ nên $\widehat{MAO}=\widehat{MBO}=90^{\circ}$ nên $MAOB$ là tứ giác nội tiếp.
\item Vì $BE$ là đường kính của đường tròn $(O)$ nên $\widehat{BAE}=90^{\circ}\Rightarrow AE\perp AB$. Mà $MA$, $MB$ là tiếp tuyến của $(O)$ nên $AB\perp MO\Rightarrow AE\parallel MO$.
\item Vì $AE\parallel MO$ và $MA$ là tiếp tuyến của đường tròn $(O)$ nên $\widehat{NMF}=\widehat{AEF}=\widehat{NAM}\Rightarrow \Delta NMA\backsim \Delta NFM\Rightarrow \dfrac{NM}{NF}=\dfrac{NA}{NM}\Rightarrow MN^{2}=NF.NA$.
\item Ta có $\widehat{MFB}=\widehat{MHB}=90^{\circ}$ nên $MFHB$ là tứ giác nội tiếp $\Rightarrow \widehat{FHN}=\widehat{FBM}=\widehat{NAH}\Rightarrow \Delta NAH\backsim \Delta NHF\Rightarrow \dfrac{NA}{NH}=\dfrac{NH}{NF}\Rightarrow NH^{2}=NA.NF\Rightarrow NH=NM$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Hà Nam, 2017]%[9D5G1]
Cho $a$, $b$, $c$ là các số thực không âm thỏa mãn $ab+bc+ca=3$ và $a\geq c$. Tìm giá trị nhỏ nhất của biểu thức $P=\dfrac{1}{(a+1)^{2}}+\dfrac{2}{(b+1)^{2}}+\dfrac{3}{(c+1)^{2}}$.
\loigiai{
\begin{enumerate}
\item (Cách 1) Sử dụng $x^{2}+y^{2}+z^{2}\geq xy+yz+zx$ và $(x+y+z)^{2}\geq 3(xy+yz+zx)$.\\
Đặt $x=\dfrac{1}{a+1}$, $y=\dfrac{1}{b+1}$, $z=\dfrac{1}{c+1}$ ($x,y,z>0$). Vì $a\geq c$ nên $z\geq x$.\\
Ta có $P\geq 2(x^{2}+y^{2}+z^{2})\geq 2(xy+yz+zx)$.\\
Mà $(a+b+c)^{2}\geq 3(ab+bc+ca)=9\Rightarrow a+b+c\geq 3$.\\
Lại có $3=ab+bc+ca\geq 3\sqrt[3]{a^{2}b^{2}c^{2}}\Rightarrow abc\leq 1\Rightarrow 3\geq 3abc\Rightarrow a+b+c\geq 3abc$.\\
Xét $xy+yz+zx=\dfrac{3(a+b+c+3)}{3abc+3(a+b+c)+12}\geq \dfrac{3(a+b+c+3)}{4(a+b+c+3)}=\dfrac{3}{4}\Rightarrow P\geq \dfrac{3}{2}$.\\
Dấu bằng xảy ra khi $a=b=c=1$.
\item (Cách 2) Sử dụng $(x+y+z)^{2}\geq 3(xy+yz+zx)$.\\
Đặt $x=a+1$, $y=b+1$, $z=c+1$ ($x,y,z>0$). Vì $a\geq c$ nên $x\geq z\Rightarrow \dfrac{1}{z^{2}}\geq \dfrac{1}{x^{2}}$.\\
Ta có $P\geq 2\left(\dfrac{1}{x^{2}}+\dfrac{1}{y^{2}}+\dfrac{1}{z^{2}}\right)\geq \dfrac{2}{3}\left(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\right)^{2}$.\\
Từ giả thiết ta có $(x-1)(y-1)+(y-1)(z-1)+(z-1)(x-1)=3\Leftrightarrow \dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=2\left(\dfrac{1}{xy}+\dfrac{1}{yz}+\dfrac{1}{zx}\right)$.\\
Lại có $\left(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\right)^{2}\geq 3\left(\dfrac{1}{xy}+\dfrac{1}{yz}+\dfrac{1}{zx}\right)=\dfrac{3}{2}\left(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\right)\Rightarrow \dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\geq \dfrac{3}{2}\Rightarrow P\geq \dfrac{3}{2}$.
\\
Dấu bằng xảy ra khi $a=b=c=1$.
\end{enumerate}
}
\end{ex}
