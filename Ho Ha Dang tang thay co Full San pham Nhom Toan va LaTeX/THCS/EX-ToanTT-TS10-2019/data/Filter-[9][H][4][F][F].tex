\begin{bt}%[Bùi Quốc Hoàn]%[EX-TS10-ToanTT-2019]%[9H4T3]
\immini
{Theo quy định trong thể thao, bán kính quả bóng rổ của nữ nhỏ hơn quả bóng rổ của nam. Bán kính của quả bóng rổ được cho bởi công thức: $r = \sqrt[3]{\dfrac{3V}{4\pi}}$. 
	Trong đó: $r$ là bán kính của quả bóng rổ, tính bằng $\mathrm{ inch}$ ($1\,  \mathrm{inch} = 2{,}54\, \mathrm{cm}$), $V$ là thể tích không khí được chứa trong quả bóng, tính bằng $\mathrm{inch}^3$.
}
{\begin{tikzpicture}[scale=1, font=\footnotesize,line join=round, line cap=round,>=stealth]
	\tkzDefPoints{0/0/A,5/0/B,7/1.5/C,2/1.5/D}
	\tkzDrawPolygon[pattern=dots](A,B,C,D)
	\shade[ball color=orange!70] (3.5,2) circle(40pt);
\end{tikzpicture}
}

\begin{enumerate}
	\item Tính bán kính quả bóng rổ của nữ, biết nó chứa được $413\, \mathrm{inch}^3$ không khí.
	\dapso{Bán kính bằng $4{,}64\, \mathrm{inch}$}
	\item Biết tỉ lệ thể tích không khí trong quả bóng rổ của nữ và quả bóng rổ của nam là $\dfrac{59}{65}$. Tính bán kính quả bóng rổ của nam.
	\dapso{Bán kính bằng $4{,}77\, \mathrm{inch}$}
\end{enumerate}
	\loigiai{\begin{enumerate}
			\item Gọi $r$, $V$ lần lượt là bán kính và thể tích quả bóng rổ nữ.\\
			 Áp dụng công thức $r = \sqrt[3]{\dfrac{3V}{4\pi}}$ với  $V = 413\, \mathrm{inch}^3$ suy ra $r = \sqrt[3]{\dfrac{3\cdot 413}{4\pi}}\approx 4{,}62\, \left(\mathrm{inch}\right)$.
			\item Gọi $r_{1}$, $V_{1}$ lần lượt là bán kính  và thể tích quả bóng rổ nữ. Khi đó $r_{1} = \sqrt[3]{\dfrac{3V_{1}}{4\pi}}$ suy ra 
			$$\dfrac{r_{1}}{r}  = \dfrac{\sqrt[3]{\dfrac{3V_{1}}{4\pi}} }{\sqrt[3]{\dfrac{3V}{4\pi}}} = \sqrt[3]{\dfrac{3V_{1}}{4\pi}\cdot \dfrac{4\pi}{3V}} = \sqrt[3]{\dfrac{V_{1}}{V}}.$$
			Do giả thiết, ta có $\dfrac{V}{V_{1}} = \dfrac{59}{65}$ nên $\dfrac{r_{1}}{r} =\sqrt[3]{\dfrac{V_{1}}{V}} =  \sqrt[3]{\dfrac{65}{59}}$ suy ra 
			$$r_{1} = \sqrt[3]{\dfrac{65}{59}}\cdot r = \sqrt[3]{\dfrac{65}{59}}\cdot \sqrt[3]{\dfrac{3\cdot 413}{4\pi}}= \sqrt[3]{\dfrac{1365}{4\pi}}\approx 4{,}77\, \left(\mathrm{inch}\right).$$
			\end{enumerate}	}
\end{bt}
