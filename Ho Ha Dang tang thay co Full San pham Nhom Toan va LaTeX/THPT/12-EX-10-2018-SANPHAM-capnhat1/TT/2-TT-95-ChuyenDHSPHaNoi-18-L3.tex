	Cho số phức $z=\cos 2\alpha+(\sin \alpha-\cos \alpha)i$ với $\alpha\in\mathbb{R}$. Giá trị lớn nhất của $|z|$ là
	\choice
	{$\dfrac{4}{3}$}
	{\True $\dfrac{3}{2}$}
	{$\sqrt{2}$}
	{$2$}
	\loigiai{
		Ta có
		\begin{eqnarray*}
			|z| &=& \sqrt{\cos^2 2\alpha+(\sin \alpha-\cos \alpha)^2} \\
			&=& \sqrt{1-\sin^2 2\alpha+1-2\sin \alpha \cos \alpha} \\
			&=& \sqrt{2-\sin^2 2\alpha-\sin 2\alpha} \\
			&=& \sqrt{\dfrac{9}{4}-\left(\sin 2\alpha +\dfrac{1}{2}\right)^2}\le \dfrac{3}{2}.
		\end{eqnarray*}
		Dấu bằng xảy ra khi và chỉ khi $\sin 2\alpha=-\dfrac{1}{2}$. Vậy giá trị lớn nhất của $|z|$ là $\dfrac{3}{2}$.
	}

	\choice

	Nghiệm của phương trình $\log_{2018} x+\log_{\sqrt{2018}} x+\log_{\sqrt[3]{2018}} x+\cdots+\log_{\sqrt[2018]{2018}} x=\dfrac{2019}{2}$ là
	\choice
	{$\sqrt[2019]{2018}$}
	{$1$}
	{$2018$}
	{\True $\sqrt[2018]{2018}$}
	\loigiai{
		Điều kiện: $x>0$. Phương trình đã cho tương đương với
		\begin{eqnarray*}
			& &\log_{2018} x+2\log_{2018} x+3\log_{2018} x+\cdots+2018\log_{2018} x=\dfrac{2019}{2} \\
			&\Leftrightarrow& \left(1+2+3+\cdots+2018\right)\log_{2018} x=\dfrac{2019}{2} \\
			&\Leftrightarrow& \dfrac{2018\cdot 2019}{2}\log_{2018} x=\dfrac{2019}{2} \\
			&\Leftrightarrow& \log_{2018} x=\dfrac{1}{2018} \Leftrightarrow x=\sqrt[2018]{2018}.
		\end{eqnarray*}
		Vậy phương trình có nghiệm duy nhất $x=\sqrt[2018]{2018}$.
	}
\end{ex}
