\begin{bt}%[Đề HK1, Bùi Thị Xuân, HCM 2017]%[1D1B3-4]
Giải phương trình $4(\sin x+\cos x)-3\sin x\cos x-4=0$.
\loigiai{
Đặt $\sin x+\cos x=t\Rightarrow \sin x\cos x=\dfrac{t^2-1}{2}$. Ta được
\begin{eqnarray*}
&& 4t-\dfrac{3}{2}(t^2-1)-4=0\\
& \Leftrightarrow & 3t^2-8t+5=0\Leftrightarrow \hoac{& t=1\\ & t=\dfrac{5}{3}}.
\end{eqnarray*}
\begin{itemize}
\item Với $t=1\Rightarrow\sin x+\cos x=1\Leftrightarrow\cos\left(x-\dfrac{\pi}{4}\right)=\dfrac{\sqrt{2}}{2}\Leftrightarrow\hoac{& x=\dfrac{\pi}{2}+k2\pi\\ & x=k2\pi}$, $k\in\mathbb{Z}$.
\item Với $t=\dfrac{5}{3}\Rightarrow \sin x+\cos x=\dfrac{5}{3}$. Ta có $1^2+1^2-\left(\dfrac{5}{3}\right)^2=-\dfrac{7}{9}<0$ nên phương trình vô nghiệm.
\end{itemize}
Vậy nghiệm phương trình là $x\in\left\{\left.\dfrac{\pi}{2}+k2\pi;k2\pi\right|k\in\mathbb{Z}\right\}$.
}
\end{bt}

\begin{bt}%[Đề HK1, Bùi Thị Xuân, HCM 2017]%[Lê Minh An, Dự án 11-HK1-2018]%[1D1B3-5]
Giải phương trình $\cos x-\sqrt{3}\cos 3x+\cos 5x=0$.
\loigiai{
Ta có $\cos x+\cos 5x=2\cos 2x\cos 3x$ nên phương trình tương đương với
\begin{eqnarray*}
&& 2\cos 2x\cos 3x-\sqrt{3}\cos 3x=0\\
& \Leftrightarrow & \cos 3x\left(2\cos 2x-\sqrt{3}\right)=0\\
& \Leftrightarrow & \hoac{& \cos 3x=0\\ & \cos 2x=\dfrac{\sqrt{3}}{2}}
\Leftrightarrow\hoac{& x=\dfrac{\pi}{6}+\dfrac{k\pi}{3}\\ & x=\dfrac{\pi}{12}+k\pi\\ & x=-\dfrac{\pi}{12}+k\pi}.
\end{eqnarray*}
Vậy nghiệm phương trình là $x\in\left\{\left.\dfrac{\pi}{6}+\dfrac{k\pi}{3};\dfrac{\pi}{12}+k\pi; -\dfrac{\pi}{12}+k\pi\right|k\in\mathbb{Z}\right\}$.
}
\end{bt}

\begin{bt}%[Đề HK1, Bùi Thị Xuân, HCM 2017]%[Lê Minh An, Dự án 11-HK1-2018]%[1D2K2-1]
Có $5$ nhà Toán học nam, $3$ nhà Toán học nữ và $4$ nhà Vật lí nam. Cần lập một đoàn công tác gồm $3$ người, trong đó cần có cả nam và nữ, đồng thời cần có cả nhà Toán học và nhà Vật lí. Có bao nhiêu cách chọn?
\loigiai{
Vì các nhà Vật lí đều là nam nên cần chọn ít nhất một nhà Toán học nữ.
\begin{itemize}
\item Chọn $1$ nhà Toán học nữ, $1$ nhà Toán học nam và $1$ nhà Vật lí nam. Có $\mathrm{C}_{3}^{1}\cdot\mathrm{C}_{5}^{1}\cdot\mathrm{C}_{4}^{1}=60$ cách chọn.
\item Chọn $1$ nhà Toán học nữ, $2$ nhà Vật lí nam. Có $\mathrm{C}_{3}^{1}\cdot\mathrm{C}_{4}^{2}=18$ cách chọn.
\item Chọn $2$ nhà Toán học nữ, $1$ nhà Vật lí nam. Có $\mathrm{C}_{3}^{2}\cdot\mathrm{C}_{4}^{1}=12$ cách chọn.
\end{itemize}
Vậy có $60+18+12=90$ cách chọn thỏa mãn đề bài.
}
\end{bt}

\begin{bt}%[Đề HK1, Bùi Thị Xuân, HCM 2017]%[Lê Minh An, Dự án 11-HK1-2018]%[1D2K5-5]
Ba học sinh cùng làm bài kiểm tra độc lập với nhau. Xác suất làm được bài của học sinh thứ nhất là $0{,}8$, của học sinh thứ hai là $0{,}95$, của học sinh thứ ba là $0{,}6$. Tính xác suất để có đúng $2$ học sinh làm được bài.
\loigiai{
\begin{itemize}
\item Gọi $A$ là biến cố: ``có đúng $2$ học sinh làm được bài''.
\item Gọi $A_i$ là biến cố: `` học sinh thứ $i$ làm được bài''.
Ta có $P\left(A_1\right)=0{,}8$, $P\left(A_2\right)=0{,}95$ và $P\left(A_3\right)=0{,}6$.
\item Lại có 
\begin{eqnarray*}
P(A) & = & P\left(A_1A_2\overline{A_3}\cup A_1\overline{A_2}A_3\cup \overline{A_1}A_2A_3\right)\\
& = & P\left(A_1A_2\overline{A_3}\right)+P\left(A_1\overline{A_2}A_3\right)+P\left(\overline{A_1}A_2A_3\right)\\
& = & P\left(A_1\right)\cdot P\left(A_2\right)\cdot P\left(\overline{A_3}\right)+P\left(A_1\right)\cdot P\left(\overline{A_2}\right)\cdot P\left(A_3\right)+P\left(\overline{A_1}\right)\cdot P\left(A_2\right)\cdot P\left(A_3\right)\\
& = & 0{,}8\cdot 0{,}95\cdot(1- 0{,}6)+0{,}8\cdot (1-0{,}95)\cdot 0{,}6+(1-0{,}8)\cdot 0{,}95\cdot 0{,}6\\
& = & 0{,}442.
\end{eqnarray*}
\end{itemize}
}
\end{bt}

\begin{bt}%[Đề HK1, Bùi Thị Xuân, HCM 2017]%[Lê Minh An, Dự án 11-HK1-2018]%[1D2K3-2]
Cho $n$ là số nguyên dương và $x$ là số thực dương. Tìm số hạng không chứa $x$ trong khai triển nhị thức Niu-tơn của $\left(\sqrt[3]{x}+\dfrac{2}{\sqrt{2}}\right)^n$, biết rằng $\mathrm{C}_{n}^{6}+3\mathrm{C}_{n}^{7}+3\mathrm{C}_{n}^{8}+\mathrm{C}_{n}^{9}=2\mathrm{C}_{n+2}^{8}$ (với $\sqrt[m]{k}=x^{\tfrac{k}{m}}$, $\forall m,k\in\mathbb{N}^{*}$, $m\ge 2$).
\loigiai{
Điều kiện $x\ge 9$. Trước hết ta có $\mathrm{C}_{n}^{k}+\mathrm{C}_{n}^{k+1}=\mathrm{C}_{n+1}^{k+1}$. Thật vậy,
\begin{eqnarray*}
\mathrm{C}_{n}^{k}+\mathrm{C}_{n}^{k+1} & = &
\dfrac{n!}{(n-k)!k!}+\dfrac{n!}{(n-k-1)!(k+1)!} \\
& = & \dfrac{n!\cdot (k+1)+n!\cdot (n-k)}{(n-k)!(k+1)!}
=\dfrac{n!\cdot (n+1)}{(n-k)!(k+1)!}\\
& = & \dfrac{(n+1)!}{(n-k)!(k+1)!}=\mathrm{C}_{n+1}^{k+1}.
\end{eqnarray*}
Áp dụng ta có
\begin{eqnarray*}
&& \mathrm{C}_{n}^{6}+3\mathrm{C}_{n}^{7}+3\mathrm{C}_{n}^{8}+\mathrm{C}_{n}^{9}\\
& = & \left(\mathrm{C}_{n}^{6}+\mathrm{C}_{n}^{7}\right)+2\left(\mathrm{C}_{n}^{7}+\mathrm{C}_{n}^{8}\right)+\mathrm{C}_{n}^{8}+\mathrm{C}_{n}^{9}\\
& = & \mathrm{C}_{n+1}^{7}+2\mathrm{C}_{n+1}^{8}+\mathrm{C}_{n+1}^{9}
=\mathrm{C}_{n+2}^{8}+\mathrm{C}_{n+2}^{9}=\mathrm{C}_{n+3}^{9}.
\end{eqnarray*}
Do đó
$\mathrm{C}_{n}^{6}+3\mathrm{C}_{n}^{7}+3\mathrm{C}_{n}^{8}+\mathrm{C}_{n}^{9}=2\mathrm{C}_{n+2}^{8}\Leftrightarrow \mathrm{C}_{n+3}^{9}=2\mathrm{C}_{n+2}^{8}\Leftrightarrow\dfrac{n+3}{9}=2\Leftrightarrow n=15$.\\
Khi đó, số hạng tổng quát của $\left(\sqrt[3]{x}+\dfrac{2}{\sqrt{2}}\right)^{15}$ là $\mathrm{C}_{15}^{k}\cdot\left(\sqrt[3]{x}\right)^{15-k}\cdot\left(\dfrac{2}{\sqrt{x}}\right)^k=\mathrm{C}_{15}^{k}\cdot 2^k\cdot x^{\tfrac{30-5k}{6}}$.\\
Số hạng không chứa $x$ ứng với $\dfrac{30-5k}{6}=0\Leftrightarrow k=6$. Vậy số hạng cần tìm là $\mathrm{C}_{15}^{6}\cdot 2^6=320320$.
}
\end{bt}

\begin{bt}%[Đề HK1, Bùi Thị Xuân, HCM 2017]%[Lê Minh An, Dự án 11-HK1-2018]%[1D3G1-1]
Dùng phương pháp quy nạp, hãy chứng minh $u_n=13^{n+1}+5n^3+n-7$ luôn chia hết cho $6$ với mọi số nguyên dương $n$.
\loigiai{
Ta có 
\[u_n=13^{n+1}+5n^3+n-7.\tag{1}\]
\begin{itemize}
\item Khi $n=1$ ta có $u_1=168\Rightarrow u_1\,\,\vdots\,\, 6$.
\item Giả sử khi $n=k$, $k\ge 1$ ta có $u_k=13^{k+1}+5k^3+k-7\,\,\vdots\,\, 6$.
\item Ta chứng minh khi $n=k+1$ thì $u_{k+1}$ chia hết cho $6$. Thật vậy, ta có
\begin{eqnarray*}
u_{k+1} & = & 13^{k+2}+5(k+1)^3+(k+1)-7\\
& = & 13\cdot 13^{k+1}+5\left(k^3+3k^2+3k+1\right)+k-6\\
& = & 13\cdot 13^{k+1}+5k^3+5\left(3k^2+3k\right)+k-1\\
& = & 13\left( 13^{k+1}+5k^3+k-7\right)+5\left[3k(k+1)\right]-6\left(10k^3+2k-15\right)\\
& = & 13u_k+5\left[3k(k+1)\right]-6\left( 10k^3+2k-15\right)
\end{eqnarray*}
Vì $k(k+1)\,\,\vdots\,\, 2$ nên $3k(k+1)\,\,\vdots\,\, 6$, $u_k\,\,\vdots\,\, 6$ và $6\left( 10k^3+2k-15\right)\,\,\vdots\,\, 6$ nên $u_{k+1}\,\,\vdots\,\, 6$.
\end{itemize}
Vậy $u_n\,\,\vdots\,\, 6$, $\forall n\in\mathbb{N}^{*}$.
}
\end{bt}

\begin{bt}%[Đề HK1, Bùi Thị Xuân, HCM 2017]%[Lê Minh An, Dự án 11-HK1-2018]%[1H2B3-5]%[1H2K4-2]%[1H2G4-6]
Cho hình chóp $S.ABCD$ có $ABCD$ là hình thang đáy lớn $AD$, biết $AD=2BC$, tam giác $SAD$ cân tại $S$. Gọi $H$, $K$ lần lượt là trung điểm của $AD$ và $SD$. Lấy các điểm $M$, $E$, $Q$ lần lượt trên các cạnh $SC$, cạnh $AB$ và cạnh $SB$ sao cho $SC=3SM$, $EB=2EA$, $SB=3SQ$.
\begin{enumerate}
\item Tìm giao điểm $N$ của $BM$ và mặt phẳng $(SAD)$.
\item Chứng minh $(CHK)\parallel (SAB)$.
\item Chứng minh $EM\parallel (SAD)$.
\item Mặt phẳng $(P)$ chứa $EM$ và song song với $BC$, cắt $CD$ tại $R$. Xác định tính chất thiết diện của hình chóp $S.ABCD$ khi cắt bởi mặt phẳng $(P)$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1]
\tkzDefPoints{0/0/A, 5/0/D, 1.5/-2/B, 2/3.5/S}
\coordinate (H) at ($(A)!.5!(D)$);
\tkzDefPointBy[translation = from A to H](B)\tkzGetPoint{C}
\coordinate (K) at ($(S)!.5!(D)$);
\coordinate (Q) at ($(S)!.35!(B)$);
\tkzDefLine[parallel=through Q](A,S)\tkzGetPoint{q}
\tkzInterLL(Q,q)(A,B)\tkzGetPoint{E}
\tkzDefLine[parallel=through Q](B,C)\tkzGetPoint{q'}
\tkzInterLL(Q,q')(S,C)\tkzGetPoint{M}
\tkzDefLine[parallel=through M](S,D)\tkzGetPoint{m}
\tkzInterLL(M,m)(C,D)\tkzGetPoint{R}
\tkzDefPointBy[translation = from H to D](S)\tkzGetPoint{d}
\tkzDefPointBy[translation = from H to A](S)\tkzGetPoint{d'}
\tkzInterLL(B,M)(S,d)\tkzGetPoint{N}
\draw (A)node[left]{$A$}--(B)node[below left]{$B$}--(C)node[below right]{$C$}--(D)node[right]{$D$}--(S)node[above]{$S$}--(A);
\draw (N)node[above]{$N$}--(B)--(S)--(C)--(K)node[above right]{$K$};
\draw (E)node[below left]{$E$}--(Q)node[above left]{$Q$}--(M)node[right]{$M$}--(R)node[below right]{$R$};
\draw (d)node[above]{$d$}--(d');
\draw[dashed] (A)--(D);
\draw[dashed] (E)--(R);
\draw[dashed] (C)--(H)node[above left]{$H$}--(K);

\tkzDrawPoints[fill=black](A,B,C,D,S,Q,E,M,K,R,N,H)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item
\begin{itemize}
\item Ta có $\heva{& S\in(SAD)\cap (SBC)\\ & AD\parallel BC\\ & AD\subset(SAD),\,\,BC\subset(SBC)}$.\\
Suy ra $(SAD)\cap (SBC)=d$, với $d$ là đường thẳng qua $S$ và $d\parallel AD\parallel BC$.
\item Trong $(SBC)$, gọi $N=BM\cap d$, suy ra
$$\heva{& N\in BM\\ & N\in d,\,\, d\subset (SAD)\Rightarrow N\in (SAD)}\Rightarrow N=BM\cap (SAD).$$
\end{itemize}
\item Ta có $AD\parallel BC$, $AD=2BC$ nên $AH\parallel BC$ và $AH=BC$.\\
Do đó $ABCH$ là hình bình hành.\\
Suy ra $CH\parallel AB$ và $HK\parallel SA$ ($HK$ là đường trung bình của $\triangle SAD$).\\
Lại có $SA\subset (SAB)$; $CH$, $HK\subset (CHK)$. Vậy $(CHK)\parallel(SAB)$.
\item Ta có $\dfrac{SQ}{SB}=\dfrac{SM}{SC}=\dfrac{AE}{AB}=\dfrac{1}{3}$, nên $MQ\parallel BC\parallel AD$ và $EQ\parallel SA$.\\
Lại có $MQ$, $EQ\subset (EMQ)$; $AD$, $SA\subset (SAD)$. Suy ra $(EMQ)\parallel (SAD)$.\\
Mà $EM\subset (EMQ)$. Vậy $EM\parallel (SAD)$.
\item Do $BC\parallel (P)$, $BC\subset (SBC)$, $(P)\cap (SBC)=MQ'$ nên $MQ'\parallel BC\Rightarrow Q\equiv Q'$.\hfill (1)\\
Lại có $BC\parallel (P)$, $BC\subset (ABCD)$, $(P)\cap (ABCD)=ER$ nên $ER\parallel BC$.\hfill (2)\\
Khi đó ta có 
$$\dfrac{SQ}{SB}=\dfrac{AE}{AB}=\dfrac{1}{3}\Rightarrow EQ=\dfrac{2}{3}SA;\quad \dfrac{SM}{SC}=\dfrac{DR}{DC}=\dfrac{1}{3}\Rightarrow MR=\dfrac{2}{3}SD.$$
Mà $SA=SD$ nên $EQ=MR$.\hfill (3)\\
Ngoài ra, $EQ\parallel SA$, $MR\parallel SD$ và $SA$ không song song với $SD$ nên $MR$ không song song với $EQ$.\hfill (4)\\
Từ (1), (2), (3) và (4) suy ra thiết diện $MQER$ là hình thang cân có hai đáy là $MQ$ và $ER$.
\end{enumerate}
}
\end{bt}
