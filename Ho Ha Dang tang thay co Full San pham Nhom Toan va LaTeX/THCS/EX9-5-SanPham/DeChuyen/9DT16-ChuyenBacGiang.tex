\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Bắc Giang, 2017]%[9D1B8]%[9D4K6]
\hfill
\begin{enumerate}[1)]
\item Cho biểu thức $A=\left(\dfrac{\sqrt{ab}+\sqrt{b}}{\sqrt{a}+\sqrt{b}}+\dfrac{\sqrt{ab}+\sqrt{a}}{\sqrt{b}-\sqrt{a}}+1\right):\left(\dfrac{\sqrt{ab}+\sqrt{b}}{\sqrt{a}+\sqrt{b}}+\dfrac{\sqrt{ab}+\sqrt{a}}{\sqrt{a}-\sqrt{b}}-1\right)$.
		\begin{enumerate}[a)]
		\item Tìm điều kiện của $a$, $b$ để biểu thức $A$ có nghĩa, từ đó hãy rút gọn biểu thức $A$.
		\item Cho $\sqrt{ab}+1=4\sqrt{b}$, tìm giá trị lớn nhất của biểu thức $A$.
		\end{enumerate}
\item Tìm giá trị của $m$ để phương trình $2x^2-4mx+2m^2-1=0$ ($x$ là ẩn, $m$ là tham số) có hai nghiệm phân biệt $x_1$, $x_2$ thỏa mãn $2x_1^2+4mx_2+2m^2<2017$.
\end{enumerate}
\loigiai{
\begin{enumerate}[1)]
\item  $A=\left(\dfrac{\sqrt{ab}+\sqrt{b}}{\sqrt{a}+\sqrt{b}}+\dfrac{\sqrt{ab}+\sqrt{a}}{\sqrt{b}-\sqrt{a}}+1\right):\left(\dfrac{\sqrt{ab}+\sqrt{b}}{\sqrt{a}+\sqrt{b}}+\dfrac{\sqrt{ab}+\sqrt{a}}{\sqrt{a}-\sqrt{b}}-1\right)$.\\
		\begin{enumerate}[a)]
		\item 
Đặt $B=\dfrac{\sqrt{ab}+\sqrt{b}}{\sqrt{a}+\sqrt{b}}+\dfrac{\sqrt{ab}+\sqrt{a}}{\sqrt{b}-\sqrt{a}}+1$; $C=\dfrac{\sqrt{ab}+\sqrt{b}}{\sqrt{a}+\sqrt{b}}+\dfrac{\sqrt{ab}+\sqrt{a}}{\sqrt{a}-\sqrt{b}}-1$.\\
Vậy $A=B: C$. Điều kiện của $B, C$ có nghĩa là $a\ge 0; b\ge 0; a\ne b$.\\
$B=\dfrac{\sqrt{ab}+\sqrt{b}}{\sqrt{a}+\sqrt{b}}+\dfrac{\sqrt{ab}+\sqrt{a}}{\sqrt{b}-\sqrt{a}}+1=\dfrac{-2b(\sqrt{a}+1)}{a-b}$.\\
$C=\dfrac{\sqrt{ab}+\sqrt{b}}{\sqrt{a}+\sqrt{b}}+\dfrac{\sqrt{ab}+\sqrt{a}}{\sqrt{a}-\sqrt{b}}-1=\dfrac{2\sqrt{ab}(\sqrt{a}+1)}{a-b}$.\\	
Điều kiện của $A$ có nghĩa là $a>0; b>0; a\ne b$.\\
Vậy $A=\dfrac{-2b(\sqrt{a}+1)}{a-b}: \dfrac{2\sqrt{ab}(\sqrt{a}+1)}{a-b}=\dfrac{-\sqrt{b}}{\sqrt{a}}$ với $a>0, b>0, a\ne b$.
		\item 
Ta có $\sqrt{ab}+1=4\sqrt{b}\Leftrightarrow \sqrt{a}+\dfrac{1}{\sqrt{b}}=4$.\\
Theo bất đẳng thức Cô si ta có\\
 $4=\sqrt{a}+\dfrac{1}{\sqrt{b}}\ge 2\sqrt{\sqrt{\dfrac{a}{b}}}\Rightarrow \sqrt{\dfrac{a}{b}}\le 4\Rightarrow \sqrt{\dfrac{b}{a}}\ge \dfrac{1}{4}$.\\
Vậy $A=\dfrac{-\sqrt{b}}{\sqrt{a}}\le \dfrac{-1}{4}$. Dấu bằng xảy ra khi $a=4; b=\dfrac{1}{4}$.\\	
Vậy giá trị lớn nhất của $A=\dfrac{-1}{4}$ khi $a=4; b=\dfrac{1}{4}$.
		\end{enumerate}
\item 
$2x^2-4mx+2m^2-1=0$  $(1)$\\
Ta có: $\Delta '=4m^2-2(2m^2-1)=2>0$.\\
Suy ra phương trình (1) luôn có hai nghiệm phân biệt $x_1$, $x_2$ với mọi $m$. \\
Theo hệ thức Vi-ét ta có: $x_1+x_2=2m$.\\
Do $x_1$ là nghiệm của phương trình $(1)$ nên $2x_1^2-4m x_1+2m^2-1=0 \Rightarrow 2x_1^2=4mx_1-2m^2+1$.\\
Ta có $2x_1^2+4mx_2+2m^2<2017$ $\Leftrightarrow 4mx_1-2m^2+1+4mx_2+2m^2<2017 \Leftrightarrow 4m(x_1+x_2)<2016$\\
$\Leftrightarrow 4m.2m<2016 \Leftrightarrow m^2<252 \Leftrightarrow  -6\sqrt{7}<m<6\sqrt{7}$.\\
Vậy $-6\sqrt{7}<m<6\sqrt{7}$ thỏa mãn yêu cầu đầu bài. 
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Bắc Giang, 2016]%[9D4K7]%[9D4G7]
 \hfill
\begin{enumerate}[1)]
\item Giải phương trình $x(2x^2+13x-6)=(x^2+8x-6)\sqrt{x^2+6x}$.
\item Giải hệ phương trình $\heva{& \sqrt{2x-1}+x^2-6y+10=0 \\  & x^2-6y^2+xy+2x+11y-3=0.}$
\end{enumerate}

\loigiai{
\begin{enumerate}[1)]
\item 
$x(2x^2+13x-6)=(x^2+8x-6)\sqrt{x^2+6x}\;\;(*)$\\
Điều kiện $x^2+6x\ge 0$.\\
Phương trình $(*) $ $\Leftrightarrow \left(\sqrt{x^2+6x}-2x\right)\left(\sqrt{x^2+6x}-x^2-6x+6\right)=0$\\
$\Leftrightarrow \hoac{& \sqrt{x^2+6x}-2x=0 \\ & \sqrt{x^2+6x}-x^2-6x+6=0.}$\\
Xét $\sqrt{x^2+6x}-2x=0\Leftrightarrow \sqrt{x^2+6x}=2x$ $\Leftrightarrow \heva{& x\ge 0 \\  & x^2+6x=4x^2 }\Leftrightarrow \hoac{& x=0 \\ & x=2.}$\\
Xét  $ \sqrt{x^2+6x}-x^2-6x+6=0\Leftrightarrow x^2+6x-\sqrt{x^2+6x}-6=0  \Leftrightarrow (\sqrt{x^2+6x}-3)(\sqrt{x^2+6x}+2)=0$
$ \Leftrightarrow \sqrt{x^2+6x}-3=0 \Leftrightarrow x^2+6x=9 \Leftrightarrow \hoac{& x=-3-3\sqrt{2} \\ & x=-3+3\sqrt{2}}$ (điều kiện thỏa mãn). \\
Vậy phương trình đã cho có tập nghiệm $\left\{-3-3\sqrt{2}; -3+3\sqrt{2}; 0; 2\right\}$.
\item 
$\heva{& \sqrt{2x-1}+x^2-6y+10=0 &(1) \\  & x^2-6y^2+xy+2x+11y-3=0 &(2)}$\\
ĐK: $x\ge \dfrac{1}{2}$.\\
Từ $(2)$ suy ra $\hoac{& x=2y-3 \\ & x=1-3y.}$\\
Với $x=1-3y \Leftrightarrow 3y=1-x$ thay vào (1) ta được $\sqrt{2x-1}+x^2-2(1-x)+10=0$\\
$ \Leftrightarrow \sqrt{2x-1}+x^2+2x+8=0$
$\Leftrightarrow \sqrt{2x-1}+{(x+1)}^2+7=0$ (vô nghiệm).\\
Với $x=2y-3 \Leftrightarrow 2y=x+3$ thay vào (1) ta được $\sqrt{2x-1}+x^2-3(x+3)+10=0$\\
$\Leftrightarrow \sqrt{2x-1}+x^2-3x+1=0 (3)$.\\
Đặt $t=\sqrt{2x-1}$ $(t\ge 0)$, phương trình (3) trở thành\\
 $t+x^2-x-t^2=0 \Leftrightarrow (t-x)(1-t-x)=0 \Leftrightarrow \hoac{& t=x \\ & t=1-x. }$\\
Với $t=x$ ta có $x=\sqrt{2x-1}\Leftrightarrow {(x-1)}^2=0\Leftrightarrow x=1\Rightarrow y=2.$\\
Với $t=1-x$ ta có $1-x=\sqrt{2x-1}\Leftrightarrow x=2-\sqrt{2}$.\\
Khi $x=2-\sqrt{2}\Rightarrow y=\dfrac{5-\sqrt{2}}{2}$.\\
Đối chiếu điều kiện và kết luận hệ có các nghiệm $(1; 2)$, $\left(2-\sqrt{2}; \dfrac{5-\sqrt{2}}{2}\right)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Bắc Giang, 2016]%[9D5G2]%[9D5G5]

\begin{enumerate}[1)]
\item Tìm số tự nhiên có bốn chữ số biết rằng khi chia số đó cho $120$ được số dư là $88$ và khi chia cho $61$ được số dư là $39$. 
\item Trong một khu vườn hình chữ nhật có chiều dài $8$ m và chiều rộng $6$ m người ta trồng $19$ cây. Chứng minh rằng trong mọi cách trồng $19$ cây đó, có ít nhất hai cây mà khoảng cách vị trí trồng giữa chúng không lớn hơn $\sqrt{5}$ m.
\end{enumerate}
\loigiai{
\begin{enumerate}[1)]
\item Gọi số tự nhiên có 4 chữ số phải tìm là $x, 1000\le x\le 9999.$\\ 
Theo giả thiết ta có $x=120n+88=61k+39$ với $n, k\in \mathbb{N}$.\\
Từ đó ta có $120n-60k=k-49$ suy ra $k-49$ chia hết cho 60.\\
Mặt khác ta có:\\ 
$ 1000\le x\le 9999 \Rightarrow 1000 \le 61k+39 \le 9999  \Rightarrow \dfrac{961}{61}\le k\le \dfrac{9960}{61}$\\
$\Rightarrow 15<k<164  \Rightarrow -34<k-49<115.$\\
Mà $k-49$ chia hết cho 60 nên $k-49$ chỉ có thể là $0; 60$.\\
Với $k=49 \Rightarrow x=3028$ (chia $120$ dư $28$).\\
Với $k=109 \Rightarrow x=6688$ (chia $120$ dư $88$).\\
Vậy số cần tìm là $6688$.

\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Bắc Giang, 2016]%[9D5G1]
Cho hai số thực $a\ne 0; b\ne 0$ thoả mãn $a(ab+1)=a^2b^2-ab+1$. Chứng minh rằng 
$$a^3b^3+1\le 16a^3.$$
\loigiai{
Đặt $\dfrac{1}{a}=c$ khi đó $a(ab+1)=a^2b^2-ab+1\Rightarrow b+c=b^2-bc+c^2 \;\;(4).$\\
Từ $a^3b^3+1\le 16a^3$ ta được  $b^3+c^3\le 16.$\\
Ta có $T=b^3+c^3=(b+c)(b^2-bc+c^2)={(b+c)}^2$.\\
Mặt khác từ $(4)$ ta có $b+c={(b+c)}^2-3bc\ge {(b+c)}^2-\dfrac{3}{4}{(b+c)}^2=\dfrac{1}{4}{(b+c)}^2\Rightarrow {(b+c)}^2\le 4(b+c).$ Khẳng định được $0\le b+c\le 4$.\\
Suy ra $T={(b+c)}^2\le 16$. Điều phải chứng minh.\\
Dấu bằng xảy ra khi $a=\dfrac{1}{2}; b=2.$
}
\end{ex}
