\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Thái Nguyên]%[9D3B3]
	Cho biểu thức $P$= $\left( \dfrac{\sqrt{x}-1}{\sqrt{x}+1}+ \dfrac{1}{\sqrt{x}}- \dfrac{2\sqrt{x}}{x+ \sqrt{x}}
	\right) : \dfrac{1}{\sqrt{x}+1}$; $(x > 0)$.
	\hfill
	\begin{enumerate}
		\item  Rút gọn biểu thức $P$.
		\item Tìm giá trị nhỏ nhất của biểu thức $P$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $P$ = $\left( \dfrac{\sqrt{x}-1}{\sqrt{x}+1}+ \dfrac{1}{\sqrt{x}}- \dfrac{2\sqrt{x}}{x+ \sqrt{x}}
			\right) : \dfrac{1}{\sqrt{x}+1}$.\\
			= ${\left[ \dfrac{ \sqrt{x}(\sqrt{x}-1)+ \sqrt{x}+1- 2\sqrt{x}}{\sqrt{x}(\sqrt{x}+1)}\right]} : \dfrac{1}{\sqrt{x}+1}$.\\
		     = $\dfrac{x- \sqrt{x}+ \sqrt{x}+ 1- 2\sqrt{x}}{\sqrt{x}(\sqrt{x}+1)}$.\\
		     = $\dfrac{x- \sqrt{x}+ \sqrt{x}+1- 2\sqrt{x}}{\sqrt{x}(\sqrt{x}+1)}. (\sqrt{x}+1)$.\\
		     = $\dfrac{(\sqrt{x}-1)^2}{\sqrt{x}}$.
			\item Do $(\sqrt{x}-1)^2 \geq 0$  và $\sqrt{x}>0$ với mọi $x>0$.\\
			Nên $P \geq 0$, với mọi $x>0$. Suy ra $MinP=0 \Leftrightarrow (\sqrt{x}-1)^2=0 \Leftrightarrow \sqrt{x}-1 =0 \Leftrightarrow x=1$.\\
			Vậy $MinP= 0 \Leftrightarrow x=1$.
	\end{enumerate}}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Thái Nguyên]%[9D3B3]
	Giải phương trình, hệ phương trình sau:
\hfill
\begin{enumerate}
	\item  $\sqrt{x^2+ 6x+ 9}= 5$
	\item $\begin{cases} x-2y=1\\ x+y=4.\end{cases}$
\end{enumerate}
	\loigiai{
		\begin{enumerate}
		\item Ta có phương trình tương đương với $|x+3|= 5 \Leftrightarrow \begin{cases} x+3=5\\ x+3=-5.\end{cases} \Leftrightarrow \begin{cases} x=2\\ x=-8.\end{cases}$\\
		Vậy phương trình có nghiệm $S=\{2; -8\}$.
		\item $\begin{cases} x-2y=1\\ x+y=4.\end{cases} \Leftrightarrow \begin{cases} -y=5\\ x+y=4.\end{cases} \Leftrightarrow \begin{cases} y=-5\\ x=9.\end{cases} $\\
		Vậy hệ phương trình có nghiệm duy nhất $(x; y)=(9; -5)$.
   \end{enumerate}}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Thái Nguyên]%[9D3B3]
	Cho hàm số bậc nhất $y=(m-1)x+ 3 (1)$ (với $m \ne 1$)
	\hfill
	\begin{enumerate}
		\item  Xác định $m$ để đồ thị hàm số $(1)$ song song với đường thẳng $y=-x+ 1$.
		\item Xác định $m$ để đường thẳng $y= 1-3x$, $y= -0,5x- 1,5$ và đồ thị hàm số $(1)$ cùng đi qua một điểm.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Để  đồ thị hàm số $(1)$ song song với đường thẳng $y=-x+ 1$ thì $\begin{cases} m-1= -1\\ 3 \ne1.\end{cases} \Leftrightarrow m=0.$\\
			Vậy với $m=0$ thì đồ thị hàm số $(1)$ song song với đường thẳng $y= -x+1$.
			\item Ta có giao của hai đường thẳng $y=1-3x$ và $y= -0,5-1,5$ là nghiệm của hệ phương trình:
		$\begin{cases} y=1-3x\\ y= -0,5x-1,5.\end{cases} \Leftrightarrow \begin{cases} 2,5x= 2,5\\ y= 1-3x.\end{cases} \Leftrightarrow \begin{cases} x=1\\ y=-2.\end{cases} $\\
		Để ba đường thẳng trên cùng đi qua một điểm thì đường thẳng $(1)$ phải đi qua giao điểm của hai đường thẳng kia. Tức là $-2=(m-1)(1)+3 \Leftrightarrow m+2=-2 \Leftrightarrow m=-4$.\\
		Vậy với $m=-4$ thì ba đường thẳng trên cùng đi qua một điểm.
	\end{enumerate}}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Thái Nguyên]%[9D3B3]
	Cho nửa đường tròn tâm $O$ đường kính $AB$. Từ một điểm $K$ trên nửa đường tròn vẽ tiếp tuyến $xy$ với nửa đường tròn. Vẽ $AD$ và $BC$ vuông góc với $xy$ ($D,C$ là chân đường vuông góc). 
		\hfill
	\begin{enumerate}
		\item Chứng minh rằng $KC= KD$.
		\item Chứng minh rằng đường tròn đường kính $CD$ tiếp xúc với ba đường thẳng $AD, BC$ và $AB$.
		\item Chứng minh rằng $AD+BC$ có giá trị không đổi khi điểm $K$ di động trên nửa đường tròn.
		\item Xác định vị trí điểm $K$ trên nửa đường tròn tâm $O$ để diện tích tứ giác $ABCD$ có giá trị lớn nhất.
	\end{enumerate}
	\loigiai{
	\begin{center}
	\begin{tikzpicture}[scale =0.6]
		\tkzDefPoints{-4/0/A,4/0/B}
		\tkzDefMidPoint(A,B)		\tkzGetPoint{O}
		\tkzDefPointBy[rotation = center O angle 115](B)	\tkzGetPoint{K}
		\tkzDrawArc(O,B)(A)
		\tkzDefLine[perpendicular=through K,K=0.5](O,K)		\tkzGetPoint{d}
		\tkzDrawLine[add=1.6 and 2.7](d,K)
		\tkzLabelLine[pos=3.5,above](d,K){$y$}
		\tkzLabelLine[pos=-1.5,above](d,K){$x$}
		\tkzDefPointBy[projection = onto d--K](A)		\tkzGetPoint{D}
		\tkzDefPointBy[projection = onto d--K](B)		\tkzGetPoint{C}
		\tkzDrawCircle[diameter](C,D)
		\tkzDefPointBy[projection = onto A--B](K)		\tkzGetPoint{H}
		\tkzMarkRightAngles[size=0.4](A,D,K O,K,C B,C,K K,H,O)
		\tkzMarkSegments[mark=x](K,C K,D)
		\tkzDrawSegments(A,D K,H O,K B,C A,B)
		\tkzDrawPoints(A,B,O,K,D,C,H)
		\tkzLabelPoints[above](K,C)
		\tkzLabelPoints[below](A,H,O,B)
		\tkzLabelPoints[left](D)
	\end{tikzpicture}
\end{center}
			\begin{enumerate}
		\item Dễ thấy tứ giác $ABCD$ là hình thang, có $O$ là trung điểm của $AB$ và $AD // OK // BC$ nên $K$ là trung điểm của $DC$. Suy ra $KC= KD$.
		\item Do $KC=KD$ và $AD \perp CD$ và $BC \perp CD$ nên hai đường thẳng $AD$ và $BC$ tiếp xúc với đường tròn tâm $K$ bán kính $KC=KD= \dfrac{CD}{2}$ $(1)$.\\
		Kẻ $KH \perp AB$, nối $K$ với $B$. Xét hai tam giác vuông $BKH$ và $BKC$ ta có:\\
		$KB$ chung,\\
		$\widehat{KBH}= \widehat{KBC}$ ( vì cùng bằng $ \widehat{OKB})$.\\
		Suy ra $\bigtriangleup BKH= \bigtriangleup BKC$ ( cạnh huyền, góc nhọn).\\
		Suy ra $KH= KC$. Hay $AB$ tiếp xúc với đường tròn tâm $K$ bán kính $KC=KD$ $(2)$.\\
		Từ $(1)$ và $(2)$, suy ra đường tròn đường kính $CD$ tiếp xúc với ba đường thẳng $AB, AD, BC$.
		\item Do $OK$ là đường trung bình của hình thang $ABCD$ nên $AD+ BC = 2OK= 2R$. Mà nửa đường tròn tâm $O$ cố định nên khi $K$ thay đổi thì tổng $AD+ BC$ luôn không đổi.
		\item Do $ABCD$ là hình thang vuông nên $S_{ABCD} = \dfrac{(AD+BC).CD}{2}= \dfrac{2R.CD}{2} \leq \dfrac{2R.AB}{2}= 2R^2$.\\
		Vậy để diện tích tứ giác $ABCD$ đạt giá trị lớn nhất thì $CD= AB$, tức là $K$ phải là điểm chính giữa của cung $AB$.
			\end{enumerate}
	}
\end{ex}

	$P= \dfrac{1}{\sqrt{1}- \sqrt{2}}- \dfrac{1}{\sqrt{2}- \sqrt{3}}+ \dfrac{1}{\sqrt{3}- \sqrt{4}}- \ldots + \dfrac{1}{\sqrt{119}- \sqrt{120}}- \dfrac{1}{\sqrt{120}- \sqrt{121}}$.

			$= -\sqrt{1}- \sqrt{2}+ \sqrt{2}+ \sqrt{3}- \sqrt{3}- \sqrt{4}+ \ldots - \sqrt{119}- \sqrt{120}+ \sqrt{120}+ \sqrt{121}$.\\
			$= \sqrt{121}- 1= 11- 1= 10$. 
			
	}
