\begin{ex}%[Tuyển sinh 10, 2018-2019, Bình Định]%[Trần Bá Huy, EX_THCS06]%[9D1B8]
Cho biểu thức $A=\left(\dfrac{1}{x+\sqrt{x}}-\dfrac{1}{\sqrt{x}+1}\right)\div \dfrac{\sqrt{x}}{x+2\sqrt{x}+1}$, với $x>0$.
\begin{enumerate}
\item Rút gọn biểu thức $A$.
\item Tìm các giá trị của $x$ để $A>\dfrac{1}{2}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Với $x>0$, ta có\\
{\allowdisplaybreaks
$\begin{aligned}
A=\ & \left(\dfrac{1}{\sqrt{x}\left(\sqrt{x}+1\right)}-\dfrac{1}{\sqrt{x}+1}\right)\cdot \dfrac{\left(\sqrt{x}\right)^2+2\sqrt{x}+1}{\sqrt{x}}\\
=\ &\dfrac{1-\sqrt{x}}{\sqrt{x}(\sqrt{x}+1)}\cdot\dfrac{\left(\sqrt{x}+1\right)^2}{\sqrt{x}}=\dfrac{(1-\sqrt{x})(\sqrt{x}+1)}{x}\\
=\ &\dfrac{1-x}{x}.
\end{aligned}$}
\item Với $x>0$, ta có
$$A>\dfrac{1}{2}\Leftrightarrow \dfrac{1-x}{x}>\dfrac{1}{2}\Leftrightarrow 2-2x>x\Leftrightarrow 3x<2\Leftrightarrow x<\dfrac{2}{3}.$$
Vậy các giá trị $x$ cần tìm là $0<x<\dfrac{2}{3}$.
\end{enumerate}
}
\end{ex}



\begin{ex}%[Tuyển sinh 10, 2018-2019, Bình Định]%[Trần Bá Huy, EX_THCS06]%[9D3Y4]
Không dùng máy tính, trình bày cách giải hệ phương trình $\begin{cases}2x-y=4\\ x+3y=-5\end{cases}$.
\loigiai{
Ta có
$$\begin{cases}2x-y=4\\ x+3y=-5\end{cases}\Leftrightarrow \begin{cases} 6x-3y=12\\ x+3y=-5\end{cases}\Leftrightarrow\begin{cases}7x=7\\ y=2x-4\end{cases}\Leftrightarrow\begin{cases}x=1\\ y=-2.\end{cases}$$
Vậy hệ có nghiệm $(x;y)=(1;2)$.
}
\end{ex}



\begin{ex}%[Tuyển sinh 10, 2018-2019, Bình Định]%[Trần Bá Huy, EX_THCS06]%[9D2B5]
Trong mặt phẳng tọa độ $Oxy$ đường thẳng $d$ có hệ số góc $k$ đi qua điểm $M(1;-3)$ cắt các trục $Ox$, $Oy$ lần lượt tại $A$ và $B$.
\begin{enumerate}
\item Xác định tọa độ các điểm $A$ và $B$ theo $k$.
\item Tính diện tích tam giác $OAB$ khi $k=2$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Do $d$ đi qua điểm $M(1;-3)$ và có hệ số góc $k$ nên có phương trình $$y=k(x-1)-3\Leftrightarrow y=kx-k-3,$$
ở đây $k\neq 0$ vì $d$ cắt hai trục tọa độ.\\
Cho $x=0$, thu được $y=-k-3$, suy ra $B(0;-k-3)$.\\
Cho $y=0$, thu được $x=\dfrac{k+3}{k}$, suy ra $A\left(\dfrac{k+3}{k};0\right)$.\\
Vậy $A\left(\dfrac{k+3}{k};0\right)$ và $B(0;-k-3)$.
\item 
\immini{
Khi $k=2$ thì $A\left(\dfrac{5}{2};0\right)$, $B(0;-5)$.\\
Tam giác $OAB$ vuông tại $O$ có $OA=\dfrac{5}{2}$ và $OB=5$, từ đó
$$S_\text{OAB}=\dfrac{1}{2}OA\cdot OB=\dfrac{1}{2}\cdot\dfrac{5}{2}\cdot 5=\dfrac{25}{4}.$$
}{
\begin{tikzpicture}[>=stealth,scale=0.45]
\draw[->] (-1.5,0) -- (4.5,0) node[above] {$x$};
\draw[->] (0,-6) -- (0,1.5) node[left] {$y$};
\tkzDefPoints{2.5/0/A,0/0/O,0/-5/B}
\draw (A)--(B);
\tkzDrawPoints(O,A,B)
\tkzLabelPoints[above](A)
\tkzLabelPoints[left](B)
\tkzLabelPoints[below left](O)
\node[below right] at (2.5,0) {\footnotesize $\dfrac{5}{2}$};
\node[below right] at (0,-5) {\footnotesize $-5$};
\end{tikzpicture}
}
\end{enumerate}
}
\end{ex}



\begin{ex}%[Tuyển sinh 10, 2018-2019, Bình Định]%[Trần Bá Huy, EX_THCS06]%[9D3K6]
Tìm một số có hai chữ số biết rằng hiệu của số ban đầu và số đảo ngược của nó bằng $18$ (số đảo ngược của một số là số thu được bằng cách viết các chữ số của số đó theo thứ tự ngược lại) và tổng của số ban đầu với bình phương số đảo ngược của nó bằng $618$.
\loigiai{
Gọi số cần tìm là $\overline{ab}$, khi đó số đảo ngược của nó là $\overline{ba}$ ($a,b\in\{1;2;\cdots;9\},\ a>b$).\\
Theo giả thiết bài toán ta có hệ
\begin{align*}
& \begin{cases}\overline{ab}-\overline{ba}=18\\  \overline{ab}+(\overline{ba})^2=618\end{cases}\Leftrightarrow \begin{cases}\overline{ab}=18+\overline{ba}\\ 18+\overline{ba}+(\overline{ba})^2=618\end{cases}\Leftrightarrow \begin{cases}\overline{ab}=18+\overline{ba}\\ \left(\overline{ba}\right)^2+\overline{ba}-600=0\end{cases}\\
\Leftrightarrow\ &\begin{cases}\overline{ab}=18+\overline{ba}\\ \left[\begin{array}{l}
\overline{ba}= 24\\ \overline{ba}=-25
\end{array}\right. \end{cases}\Leftrightarrow \left[\begin{array}{l}
\overline{ba}=24,\ \overline{ab}=42\quad \text{(thỏa mãn)}\\ \overline{ba}=-25,\ \overline{ab}=-7\quad \text{(loại)}.
\end{array}\right.
\end{align*}
Vậy số cần tìm là $42$.
}
\end{ex}



\begin{ex}%[Tuyển sinh 10, 2018-2019, Bình Định]%[Trần Bá Huy, EX_THCS06]%[9H3B7]%[9H2B2]%[9H1K4]
Cho tam giác đều $ABC$ có đường cao $AH$. Trên cạnh $BC$ lấy điểm $M$ tùy ý ($M$ không trùng với $B$, $C$, $H$). Gọi $P$, $Q$ lần lượt là hình chiếu vuông góc của $M$ lên $AB$ và $AC$.
\begin{enumerate}
\item Chứng minh tứ giác $APMQ$ nội tiếp được trong đường tròn và xác định tâm $O$ của đường tròn này.
\item Chứng minh $OH\perp PQ$.
\item Chứng minh $MP+MQ=AH$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzDefPoints{0/0/B, 7/0/C, 2/0/M}
\tkzDefTriangle[equilateral](B,C) \tkzGetPoint{A}
\coordinate (H) at ($(B)!(A)!(C)$);
\coordinate (P) at ($(A)!(M)!(B)$);
\coordinate (Q) at ($(A)!(M)!(C)$);
\coordinate (O) at ($(A)!0.5!(M)$);
\draw (A)--(B)--(C)--cycle (A)--(H) (Q)--(M)--(P) (A)--(M) (P)--(Q) (O)--(H) (O)--(P)--(H)--(Q)--(O);
\tkzDrawCircle(O,A)
\tkzDrawPoints(A,B,C,H,M,P,Q,O)
\tkzLabelPoints[below](B,M,H,C)
\tkzLabelPoints[above](A)
\tkzLabelPoints[right](Q)
\tkzLabelPoints[left](P,O)
\tkzMarkRightAngle[size=0.3](M,P,A)
\tkzMarkRightAngle[size=0.3](A,Q,M)
\tkzMarkRightAngle[size=0.3](C,H,A)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Chứng minh tứ giác $APMQ$ nội tiếp được và xác định tâm $O$ của đường tròn này.\\
Theo giả thiết bài toán, hai điểm $P$ và $Q$ cùng nhìn đoạn $AM$ dưới một góc vuông, suy ra chúng thuộc đường tròn đường kính $AM$.\\
Vậy tứ giác $APMQ$ nội tiếp được đường tròn, tâm $O$ của đường tròn này là trung điểm của đoạn $AM$.
\item Chứng minh $OH\perp PQ$.\\
Vì $H$ nhìn đoạn $AM$ dưới một góc vuông nên $H$ thuộc đường tròn $(O)$ đường kính $AM$.\\
Mà $\triangle ABC$ đều nên $\widehat{HAP}=\widehat{HAQ}$, suy ra hai dây cung $HP=HQ$.\\
Mặt khác $OP=OQ$, suy ra $OH$ là đường trung trực của $PQ$. Nói cách khác $OH\perp PQ$.
\item Chứng minh $MP+MQ=AH$.\\
Do $\triangle BMP$ vuông tại $P$ và có góc $\widehat{B}=60^\circ$ nên $MP=MB\sin 60^\circ=\dfrac{MB\sqrt{3}}{2}$.\\
Tương tự $MQ=\dfrac{MC\sqrt{3}}{2}$. Suy ra $MP+MQ=\dfrac{MB\sqrt{3}}{2}+\dfrac{MC\sqrt{3}}{2}=\dfrac{BC\sqrt{3}}{2}$.\\
Mà $\triangle ABC$ đều nên $AH=\dfrac{BC\sqrt{3}}{2}$. Từ đó $AH=MP+MQ$.
\end{enumerate}
}
\end{ex}



\begin{ex}%[Tuyển sinh 10, 2018-2019, Bình Định]%[Trần Bá Huy, EX_THCS06]%[9H1G4]
Cho tam giác đều $ABC$ có cạnh bằng $a$. Hai điểm $M$, $N$ lần lượt di động trên hai đoạn thẳng $AB$, $AC$ sao cho $\dfrac{AM}{MB}+\dfrac{AN}{NC}=1$. Đặt $AM=x$ và $AN=y$. Chứng minh $MN=a-x-y$.
\loigiai{
\immini{
Theo giả thiết ta có\\
$\begin{aligned}
& \dfrac{x}{a-x}+\dfrac{y}{a-y}=1\\
\Leftrightarrow\ & ax-xy+ay-xy=a^2-ax-ay+xy\\
\Leftrightarrow\ & a^2+x^2+y^2-2ax-2by+2xy=x^2+y^2-xy\\
\Leftrightarrow\ & (a-x-y)^2=x^2+y^2-xy.
\end{aligned}$\\
Không mất tính tổng quát, giả sử $x>y$.\\
Kẻ $MK\parallel BC\ (K\in AC)$, $NL\parallel BC\ (L\in AB)$.\\
Khi đó $\triangle AMK$ và $\triangle ALN$ là các tam giác đều (tam giác cân và có góc bằng $60^\circ$) và tứ giác $LNKM$ là hình thang cân. Kẻ $NH\perp MK$ tại $H$.\\
}{
\begin{tikzpicture}[scale=0.85]
\tkzDefPoints{0/0/B,8/0/C}
\tkzDefTriangle[equilateral](B,C) \tkzGetPoint{A}
\draw (A)--(B)--(C)--cycle;
\coordinate (M) at ($(A)!0.6!(B)$);
\coordinate (K) at ($(A)!0.6!(C)$);
\coordinate (N) at ($(A)!0.3!(C)$);
\coordinate (L) at ($(A)!0.3!(B)$);
\coordinate (H) at ($(M)!(N)!(K)$);
\tkzDrawPoints(A,B,C,M,N,K,L,H)
\draw (M)--(N) (M)--(K)--(L)--(N) (N)--(H);
\tkzLabelPoints[above](A)
\tkzLabelPoints[below](B,C,H)
\tkzLabelPoints[right](N,K)
\tkzLabelPoints[left](M,L)
\tkzMarkRightAngle[size=0.4](N,H,M)
\node[red,right] at ($(A)!0.5!(N)$) {$y$};
\node[red,left] at ($(L)!0.5!(M)$) {$x-y$};
\end{tikzpicture}
}
\noindent
Tam giác $NHK$ vuông tại $H$, có $NK=x-y$, $\widehat{HNK}=\dfrac{1}{2}\widehat{BAC}=30^\circ$. \\
Suy ra $HK=NK\cdot\sin30^\circ=\dfrac{x-y}{2}$, $NH=NK\cos30^\circ=\dfrac{(x-y)\sqrt{3}}{2}$.\\
Xét tam giác vuông $MNH$, có\\
$\begin{aligned}
& MH=x-\dfrac{x-y}{2}=\dfrac{x+y}{2}, \\
&MN=\sqrt{\left(\dfrac{x+y}{2}\right)^2+\left(\dfrac{(x-y)\sqrt{3}}{2}\right)^2}=\sqrt{x^2+y^2-xy}=\sqrt{(a-x-y)^2}=|a-x-y|.
\end{aligned}$\\
Cuối cùng, vì $\dfrac{x}{a-x}+\dfrac{y}{a-y}=1$ nên
$$\begin{cases}\dfrac{x}{a-x}<1\\ \dfrac{y}{a-y}<1\end{cases}\Leftrightarrow \begin{cases}x<a-x\\ y<a-y\end{cases}\Rightarrow x+y<2a-x-y\Rightarrow a-x-y>0.$$
Vậy $MN=a-x-y$.
}
\end{ex}
