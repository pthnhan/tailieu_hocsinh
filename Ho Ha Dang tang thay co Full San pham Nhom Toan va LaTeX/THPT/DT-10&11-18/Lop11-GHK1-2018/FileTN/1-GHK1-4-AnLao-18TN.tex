\begin{bt}%[Kiểm tra 45' Lượng giác - THPT An Lão - HP]%[Nguyễn Quốc Thịnh - EX-DT-10&11-2018]%[1D1B3-1]
	Giải phương trình $-2\sin^2 x-\sin x +3=0$.
	\loigiai{
		Ta có $-2\sin^2 x-\sin x +3=0 \Leftrightarrow \hoac{&\sin x = 1 \\ & \sin x = -\dfrac{3}{2} (\text{ PTVN})} \Leftrightarrow x =\dfrac{\pi}{2} + k 2\pi, k \in \mathbb{Z}$.
	}
\end{bt}
\begin{bt}%[Kiểm tra 45' Lượng giác - THPT An Lão - HP]%[Nguyễn Quốc Thịnh - EX-DT-10&11-2018]%[1D1B3-3]
	Giải phương trình $\sqrt{3}\sin x-\cos x=\sqrt{2}$.
	\loigiai{
		\begin{eqnarray*}
			&\sqrt{3}\sin x-\cos x=\sqrt{2}& \Leftrightarrow \sin x \cdot \dfrac{\sqrt{3}}{2} - \cos x \cdot \dfrac{1}{2} = \dfrac{\sqrt{2}}{2} \\
			& & \Leftrightarrow \sin \left(x - \dfrac{\pi}{6} \right) = \sin \dfrac{\pi}{4}\\
			& & \Leftrightarrow \hoac{&x - \dfrac{\pi}{6} = \dfrac{\pi}{4} + k 2\pi \\& x - \dfrac{\pi}{6} = \dfrac{3\pi}{4} + k \pi }\\
			& & \Leftrightarrow \hoac{& x = \dfrac{5 \pi}{12} + k 2 \pi \\ & x = \dfrac{11\pi}{12}+ k 2 \pi  },k\in \mathbb{Z}.
		\end{eqnarray*}
	}
\end{bt}
\begin{bt}%[Kiểm tra 45' Lượng giác - THPT An Lão - HP]%[Nguyễn Quốc Thịnh - EX-DT-10&11-2018]%[1D1K3-6]
	Giải phương trình $\dfrac{1-\cos 2x}{\cos x}=\dfrac{\sin 2x}{1+\cos 2x}$.
	\loigiai{
		Điều kiện $\heva{& \cos x \ne 0 \\ & 1 + \cos 2 x \ne 0} \Leftrightarrow \cos x \ne 0 \Leftrightarrow x \ne \dfrac{\pi}{2} + k \pi, k \in \mathbb{Z}.$\\
		Ta có
		\begin{eqnarray*}
			& \dfrac{1-\cos 2x}{\cos x}=\dfrac{\sin 2x}{1+\cos 2x} &\Leftrightarrow \dfrac{2 \sin^2x}{\cos x} = \dfrac{2 \sin x \cdot \cos x}{2 \cos^2x}\\
			& &\Leftrightarrow 2\sin^2x = \sin x \\
			& &\Leftrightarrow \hoac{&\sin x = 0 \\ & \sin x = \dfrac{1}{2}} \\
			& &\Leftrightarrow \hoac{& x = k\pi \\ & x = \dfrac{\pi}{6} + k2\pi \\& x = \dfrac{5\pi}{6} + k 2\pi} , k \in \mathbb{Z} \, (\text{TMĐK})
		\end{eqnarray*}
		Vậy phương trình đã cho có $3$ họ nghiệm là $x = k\pi$, $x = \dfrac{\pi}{6} + k2\pi$ và $x = \dfrac{5\pi}{6} + k2\pi$ $(k \in \mathbb{Z})$.
	}
\end{bt}
