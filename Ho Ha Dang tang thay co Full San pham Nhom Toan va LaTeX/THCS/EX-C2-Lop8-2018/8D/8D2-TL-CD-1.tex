\section{Một số phương pháp phân tích đa thức thành nhân tử}
\subsection{Phương pháp tách một hạng tử thành nhiều hạng tử}
\begin{vd}%[Triệu Minh Hà, dự án EX-C2-L8]%[8D1B9]
	Phân tích đa thức thành nhân tử:
	\begin{center}
		$3x^2-8x+4.$
	\end{center}
\loigiai{
Đa thức trên không chứa nhân tử chung, không có dạng một hằng đẳng thức đáng nhớ nào, cũng không thể nhóm các hạng tử. Ta biến đổi đa thức ấy thành đa thức có nhiều hạng tử hơn.
\begin{itemize}
	\item[Cách 1.] (Tách hạng tử thứ hai)\\
	$3x^2-8x+4=3x^2-6x-2x+4=3x(x-2)-2(x-2)=(x-2)(3x-2).$
	\item[Cách 2.] (Tách hạng tử thứ nhất)
	\begin{align*}
	&3x^2-8x+4=4x^2-8x+4-x^2=(2x-2)^2-x^2\\
	=&(2x-2+x)(2x-2-x)=(3x-2)(x-2).
	\end{align*}
\end{itemize}
	\begin{nx}
		Trong cách 1, hạng tử $-8x$ được tách thành hai hạng tử $-6x$ và $-2x$. Trong đa thức $3x^2-6x-2x+4$, hệ số của các hạng tử là $3$, $-6$, $-2$, $4$. Các hệ số thứ hai và thứ tư đều gấp $-2$ lần hệ số liền trước, nhờ đó mà xuất hiện nhân tử chung $x-2$.
	\end{nx}
	 Một cách tổng quát, để phân tích tam thức bậc hai $ax^2+bx+c$ thành nhân tử, ta tách hạng tử $bx$ thành $b_1x+b_2x$ sao cho $\dfrac{b_1}{a}=\dfrac{c}{b_2}$, tức là $b_1b_2=ac$.\\
	 Trong thực hành ta làm như sau:
	 \begin{itemize}
	 	\item[Bước 1:]  Tính tích $ac$.
	 	\item[Bước 2:] Phân tích $ac$ ra tích của hai thừa số nguyên bằng mọi cách.
	 	\item[Bước 3:] Chọn hai thừa số mà tổng bằng $b$.
	 \end{itemize}
	 Trong ví dụ trên, đa thức $3x^2-8x+4$ có $a=3$, $b=-8$, $c=4$. Tích $ac=3\cdot 4=12$. Phân tích $12$ ra tích của hai thừa số, hai thừa số này cùng dấu (vì tích của chúng bằng $12$), và cùng âm (để tổng của chúng bằng $-8$) : $(-1)(-12)$, $(-2)(-6)$, $(-3)(-4)$. Chọn hai thừa số mà tổng bằng $-8$, đó là $-2$ và $-6$.
}
\end{vd}
\begin{vd}%[Triệu Minh Hà, dự án EX-C2-L8]%[8D1B9]
	Phân tích đa thức thành nhân tử:
	\begin{center}
		$4x^2-4x-3.$
	\end{center}
	\loigiai{
		\begin{itemize}
			\item[Cách 1.] (Tách hạng tử thứ hai)\\
			$4x^2-4x-3=4x^2+2x-6x-3=2x(2x+1)-3(2x+1)=(2x+1)(2x-3).$
			\item[Cách 2.] (Tách hạng tử thứ ba)
			\begin{align*}
			4x^2-4x-3=4x^2-4x+1-4=(2x-1)^2-2^2=(2x+1)(2x-3)
			\end{align*}
		\end{itemize}
	\begin{nx}
		Qua hai ví dụ trên, ta thấy việc tách một hạng tử thành nhiều hạng tử khác thường nhằm mục đích:
		\begin{itemize}
			\item Làm xuất hiện các hệ số tỉ lệ, nhờ đó mà xuất hiện nhân tử chung (cách 1);
			\item Làm xuất hiện hiệu của hai bình phương (cách 2).
		\end{itemize} 
	\end{nx}
	Với các đa thức có bậc từ bậc ba trở lên, để dễ dàng làm xuất hiện các hệ tỉ lệ, người ta thường dùng cách \textit{tìm nghiệm của đa thức}.\\
Ta nhắc lại khái niệm nghiệm của đa thức: số $a$ được gọi là nghiệm của đa thức $f(x)$ nếu $f(a)=0$. Như vậy, nếu đa thức $f(x)$ có nghiệm $x=a$ thì nó chứa nhân tử $x-a$.\\
Ta chứng minh được rằng nghiệm của đa thức, nếu có, phải là ước của hệ số tự do.\\
Ta chứng minh được rằng nghiệm nguyên của đa thức, nếu có, phải là ước của hệ số tự do.\\
Thật vậy, giả sử đa thức $a_0x^n+a_1a^{n-1}+\cdots+a_{n-1}x+a_n$ với các hệ số $a_0$, $a_1$, $\ldots$, $a_n$ nguyên, có nghiệm $x=a(x\in\mathbb{Z})$. Thế thì
\begin{align*}
a_0x^n+a_1a^{n-1}+\cdots+a_{n-1}x+a_n=(x-a)(b_0x^{n-1}+b_1x^{n-2}+\cdots+b_{n-1}).
\end{align*}
trong đó $b_0$, $b_1$, $\ldots$, $b_{n-1}$ nguyên. Hạng tử có bậc thấp nhất của tích ở vế phải bằng $-ab_{n-1}$, hạng tử có bậc thấp nhất của vế trái bằng $a_n$. Do đó $-ab_{n-1}=a_n$, tức $a$ là ước của $a_n$.
	}
\end{vd}
\begin{vd}%[Triệu Minh Hà, dự án EX-C2-L8]%[8D1K9]
	Phân tích đa thức thành nhân tử:
	\begin{center}
		$f(x)=x^3-x^2-4.$
	\end{center}
	\loigiai{
		Lần lượt kiểm tra với $x=\pm 1,\pm 2,\pm 4$, ta thấy $f(2)=2^3-2^2-4=0$. Đa thức có nghiệm $x=2$, do đó chứa nhân tử $x-2$.\\
		Ta tách các hạng tử như sau:
		\begin{itemize}
			\item[Cách 1.]
		\begin{align*}
	&x^3-x^2-4=x^2-2x^2+x^2-2x+2x-4\\
	=&x^2(x-2)+x(x-2)+2(x-2)=(x-2)(x^2+x+2).
		\end{align*}
		
			\item[Cách 2.]
		\begin{eqnarray*}
	&	&x^3-x^2-4=x^3-8-x^2+4\\
	&	=&(x-2)(x^2+2x+4)-(x+2)(x-2)\\
	&	=&(x-2)(x^2+2x+4-x-2)=(x-2)(x^2+x+2).
		\end{eqnarray*}
		\end{itemize}
	}
\end{vd}
	\begin{nx}
		Khi xét nghiệm nguyên của đa thức, nên nhớ hai định lí sau:
		\begin{enumerate}
			\item Nếu đa thức $f(x)$ có tổng các hệ số bằng $0$ thì $1$ là nghiệm của đa thức, do đó đa thức chứa nhân tử $x-1$.\\
			Chẳng hạn, đa thức $x^2-5x^2+8x-4$ có $1-5+8-4=0$ nên $1$ là nghiệm của đa thức, đa thức chứa nhân tử $x-1$.
			\item Nếu đa thức $f(x)$ có tổng các hệ số của hạng tử bậc chẵn bằng tổng các hệ số của hạng tử bậc lẻ thì $-1$ là nghiệm của đa thưc, đa thức chưa nhân tử $x+1$.\\
			Chẳng hạng, đa thức $x^3-5x^2+3x+9$ có $9-5=3+1$ nên $-1$ là nghiệm của đa thức, đa thức chứa nhân tử $x+1$.
		\end{enumerate}
	\end{nx}
\begin{nx}
	Để nhanh chóng loại trừ các ước của hệ số tự do không là nghiệm của đa thức, có thể dùng nhận xét sau:\\Nếu $x$ là nghiệm nguyên của đa thức $f(x)$ và $f(1)$, $f(-1)$ khác $0$ thì $\dfrac{f(1)}{a-1}$ và $\dfrac{f(-1)}{a+1}$ đều là số nguyên.
	\loigiai{
Số $a$ là nghiệm của $f(x)$ nên
\begin{align*}
f(x)=(x-a)\cdot Q(x).\tag{1}
\end{align*}
Thay $x=1$ vào (1), ta có $f(1)=(1-a)Q(1).$\\
Do $f(1)\ne 0$ nên $a\ne 1$, vì thế $Q(1)=\dfrac{f(1)}{1-a}$, tức là $\dfrac{f(1)}{a-1}$ là số nguyên.
}
\end{nx}
Lấy một ví dụ: $f(x)=4x^3-13x^2+9x-18$.\\
Các ước của $18$ là $\pm 1$, $\pm 2$, $\pm 3$, $\pm 6$, $\pm 9$, $\pm 18$.
\begin{align*}
	&f(1)=4-13+9-18=-18, &f(-1)=-4-13-9-18=-44.
\end{align*}
Hiển nhiên $\pm 1$ không là nghiệm của $f(x)$. Ta thấy $\dfrac{-18}{-3-1}$, $\dfrac{-18}{\pm 6-1}$, $\dfrac{-18}{\pm 9-1}$, $\dfrac{-18}{\pm 18-1}$ không nguyên nên $-3$, $\pm 6$, $\pm 9$, $\pm 18$ không là nghiệm của $f(x)$.\\
Ta thấy $\dfrac{-44}{2+1}$ không nguyên nên $2$ không là nghiệm của $f(x)$. Chỉ còn $-2$ và $3$. Kiểm tra thấy $3$ là nghiệm của $f(x)$. Do đó, ta tách các hạng tử như sau:
\begin{eqnarray*}
& &4x^3-13x^2+9x-18=4x^3-12x^2-x^2+3x+6x-18\\
&=& 4x^2(x-3)-x(x-3)+6(x-3)=(x-3)(4x^2-x+6).
\end{eqnarray*}
\begin{vd}[Triệu Minh Hà, dự án EX-C2-L8][8D1K9]
	Phân tích đa thức sau thành nhân tử:
	\begin{center}
		$3x^2-7x^2+17x-5$.
	\end{center}
\loigiai{
Các số $\pm 1, \pm 5$ không là nghiệm của đa thức. Như vậy, đa thức không có nghiệm nguyên. Tuy vậy, đa thức có thể có nghiệm hữu tỉ khác. Ta chứng minh được rằng trong đa thức có các hệ số nguyên, nghiệm hữu tỉ (nếu có) phải có dạng $\dfrac{p}{q}$ trong đó $p$ là ước của hệ số tự do, $q$ là ước dương của hệ số cao nhất.\\
Xét các số $\pm \dfrac{1}{3}$, $\pm\dfrac{5}{3}$, ta thấy $\dfrac{1}{3}$ là nghiệm của đa thức, do đó đa thức chứa thừa số $3x-1$. Ta tách các hạng tử như sau:
\begin{eqnarray*}
& &3x^3-7x^2+17x-5=3x^2-x^2-6x^2+2x+15x-5\\
&=& x^2(3x-1)-2x(3x-1)+5(2x-1)=(3x-1)(x^2-2x+5).
\end{eqnarray*}
}
\end{vd}
\subsection{Phương pháp thêm và bớt cùng một hạng tử}
\subsubsection{Thêm và bớt cùng một hạng tử làm xuất hiện hiệu của hai bình phương}
\begin{vd}%[Triệu Minh Hà, dự án EX-C2-L8]%[8D1K9]
	Phân tích đa thức sau thành nhân tử:
	\begin{center}
		$4x^4+81$.
	\end{center}
\loigiai{
Thêm và bớt $36x^2$:
\begin{eqnarray*}
&4x^4+81&=4x^4+36x^2+81-36x^2\\
& &=(2x^2+9)^2-(6x)^2=(2x^2+9+6x)(2x^2+9-6x).
\end{eqnarray*}
}
\end{vd}
\begin{vd}%[Triệu Minh Hà, dự án EX-C2-L8]%[8D1K9]
	Phân tích đa thức sau thành nhân tử:
	\begin{center}
		$64x^4+y^4$.
	\end{center}
\loigiai{
Thêm và bớt $16x^2y^2$:
\begin{eqnarray*}
&64 x^4+y^4&=64x^4+16x^2y^2+y^4-16x^2yy^2=(8x^2+y^2)^2-(4xy)^2\\
& &=(8x^2+y^2+4xy)(8x^2+y^2-4xy).
\end{eqnarray*}}
\end{vd}
\subsubsection{Thêm và bớt cùng một hạng tử làm xuất hiện nhân tử chung}
\begin{vd}%[Triệu Minh Hà, dự án EX-C2-L8]%[8D1G9]
	Phân tích đa thức thành nhân tử:\begin{center}
		$x^5+x-1$.
	\end{center}
\loigiai{
\begin{itemize}
	\item[Cách 1.]
	\begin{eqnarray*}
	&x^5+x-1&=x^5-x^4+x^3-x^3+x^2-x^2+x-1\\
	& &=x^3(x^2-x+1)+x^2(x^2-x+1)-(x^2-x+1)\\
	& &=(x^2-x+1)(x^2+x+1).
	\end{eqnarray*}
\item[Cách 2] Thêm và bớt $x^2$:\begin{eqnarray*}
&x^5+x-1&=x^5+x^2-x^2+x-1=x^2(x^3+1)-(x^2-x+1)\\
& &=(x^2-x+1)[x^2(x+1)-1]=(x^2-x+1)(x^3+x^2-1).
\end{eqnarray*} 
\end{itemize}
}
\end{vd}
\begin{vd}%[Triệu Minh Hà, dự án EX-C2-L8]%[8D1G9]
	Phân tích đa thức thành nhân tử:
	\begin{center}
		$x(x+4)(x+6)(x+10)+128$.
	\end{center}
\loigiai{
Đặt $x^2+10x+12=y$, đa thức đã cho có dạng:
\begin{eqnarray*}
& &(y-12)(y+12)+128=y^2-16=(y+4)(y-4)\\
&\Rightarrow & x(x+4)(x+6)(x+10)+128=(x^2+10x+16)(x^2+10x+8)=(x+2)(x+8)(x^2+10x+8).
\end{eqnarray*}
}
\end{vd}
\begin{nx}
	Trong ví dụ trên, nhờ phương pháp đổi biến, ta đưa đa thức bậc bốn với $x$ thành đa thức bậc hai đối với $y$.
\end{nx}
\begin{vd}%[Triệu Minh Hà, dự án EX-C2-L8]%[8D1G9]
	Phân tích đa thức thành nhân tử:
	\begin{center}
		$A=x^4+6x^3+7x^2-6x+1$.
	\end{center}
\loigiai{
Giả sử $x\ne 0$. Ta viết đa thức dưới dạng:
\begin{align*}
A=x^2\left(x^2+6x+7-\dfrac{6}{x^2}+\dfrac{1}{x^2} \right)=x^2\left[\left(x^2+\dfrac{1}{x^2} \right)+6\left(x-\dfrac{1}{x} \right)+7   \right]  
\end{align*}
Đặt $x-\dfrac{1}{x}=y$ thì $x^2+\dfrac{1}{x^2}=y^2+2$. Do đó
\begin{eqnarray*}
&A&=x^2(y^2+2+6y+7)=x^2(y+3)^2=(xy+3x)^2\\
& &=\left[x\left(x-\dfrac{1}{x} \right)+3x  \right]^2=(x^2+3x-1)^2. 
\end{eqnarray*}
}
\end{vd}
\subsection{Phương pháp hệ số bất định}
\begin{vd}%[Triệu Minh Hà, dự án EX-C2-L8]%[8D1G9]
	Phân tích đa thức sau thành nhân tử:
	\begin{align*}
	x^4-6x^3+12x^2-14x+3.
	\end{align*}
	\loigiai{
	Các số $\pm 1$, $\pm 3$ không là nghiệm của đa thức, đa thức không có nghiệm nguyên, cũng không có nghiệm hữu tỉ. Như vậy nếu đa thức trên phân tích được thành nhân tử thì phải có dạng $(x^2+ax+b)(x^2+cx+d)$. Phép nhân này cho kết quả $x^4+(a+c)x^3+(ac+b+d)x^2+(ad+bc)x+bd$. Đồng nhất đa thức này với đa thức đã cho, ta được hệ điều kiện:
	\begin{align*}
	\heva{&a+c=-6\\&ac+b+d=12\\&ad+bc=-14\\&bd=3.}
	\end{align*}
	Xét $bd=3$ với $b,d\in\mathbb{Z}$, $b\in\{\pm 1,\pm 3\}$. Với $b=3$ thì $d=1$, hệ điều kiện trên trở thành:
	\begin{align*}
	\heva{&a+c=-6\\&ac=8\\&a+3c=-14.}
	\end{align*}
	Suy ra $2c=-14-(-6)=-8$. Do đó $c=-4$, $a=-2$.\\
	Vậy đa thức đã cho phân tích thành $(x^2-2x+3)(x^2-4x+1)$.
}
\end{vd}
\begin{note}
	Ta trình bày lời giải của ví dụ trên như sau:
	\begin{eqnarray*}
	& &x^4-6x^3+12x^2-14x+3\\
	&=&x^4-4x^3+x^2-2x^2+8x^2-2x+3x^2-12x+3\\
	&=& x^2(x^2-4x+1)-2x(x^2-4x+1)+3(x^2-4x+1)\\
	&=&(x^2-4x+1)(x^2-2x+3).
	\end{eqnarray*}
\end{note}
\subsection{Phương pháp xét giá trị riêng}
Trong phương pháp này, trước hết ta xác định các nhân tử chứa biến của đa thức, rồi gán cho các biến các giá trị cụ thể để xác định nhân tử còn lại.
\begin{vd}%[Triệu Minh Hà, dự án EX-C2-L8]%[8D1G9]
	Phân tích đa thức sau thành nhân tử:\begin{align*}
	P=x^2(y-z)+y^2(z-x)+z^2(x-y)
	\end{align*}
	\loigiai{
Thử thay $x$ bởi $y$ thì $P=y^2(y-z)+z^2(z-y)=0$. Như vậy $P$ chia hết cho $x-y$.\\
Ta lại thấy nếu thay $x$ bởi $y$, thay $y$ bởi $z$, thay $z$ vởi $x$ thì $P$ không đổi (ta nói đa thức $P$ có thể hoán vị vòng quanh $x\rightarrow y\rightarrow z\rightarrow x$). Do đó, nếu $P$ đã chia hết cho $x-y$ thì cũng chia hết cho $y-z$ và $z-x$. Vậy $P$ có dạng
\begin{align*}
k(x-y)(y-z)(z-x).
\end{align*}
Ta thấy $k$ phải là hằng số (không chứa biến) vì $P$ có bậc ba đối với tập hợp các biến $x,y,z$ còn tích $(x-y)(y-z)(z-x)$ cũng có bậc ba đối với các biến $x,y,z$.\\
Vì đẳng thức $x^2(y-z)+y^2(z-x)+z^2(x-y)=k(x-y)(y-z)(z-x)$ đúng với mọi $x,y,x$ nên ta gán cho các biến $x,y,z$ các giá trị riêng, chẳng hạn $x=2$, $y=1$, $z=0$, ta được:
\begin{align*}
4\cdot 1+1\cdot (-2)+0=k\cdot 1\cdot 1\cdot (-2)\Leftrightarrow k=-1.
\end{align*}	
Vậy $P=-(x-y)(y-z)(z-x)=(x-y)(y-z)(x-z)$.
}
\end{vd}
\subsection{Bài tập}
Phân tích đa thức sau thành nhân tử
\begin{bt}%[Triệu Minh Hà, dự án EX-C2-L8]%[8D1B9]
	\begin{enumerate}
		\item $6x^2-11x+3$.
		\item $2x^2+3x-27$.
		\item $2x^2-5xy-3y^2$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $(3x-1)(2x-3)$.
	\item $(2x+9)(x-3)$.
	\item $(x-3y)(2x+y)$.
\end{enumerate}
}
\end{bt}
\begin{bt}%[Triệu Minh Hà, dự án EX-C2-L8]%[8D1B9]
	\begin{enumerate}
		\item $x^3+2x-3$.
		\item $x^3-7x+6$.
		\item $x^3+5x^2+8x+4$.
		\item $x^3-9x^2+6x+16$.
		\item $x^3-x^2-x-2$.
