\chapter{Phép nhân và phép chia đa thức}
\section{Tính chất cơ bản của phân thức, rút gọn phân thức.}
\subsection{Tóm tắt lý thuyết}
	\textit{Phân thức đại số} là một biểu thức có dạng $\dfrac{A}{B}$, trong đó $A$ và $B$ là các đa thức, $B \ne 0$.\\
	Phân thức đại số có các tính chất cơ bản sau:\\
	- Nếu nhân cả tử thức và mẫu thức của một phân thức với cùng một đa thức khác $0$ thì được một phân thức bằng phân thức đã cho.\\
	- Nếu chia cả tử thức và mẫu thức của một phân thức cho cùng một nhân tử chung của chúng thì được một phân thức bằng phân thức đã cho.\\
	Muốn rút gọn một phân thức đại số, ta có thể:\\
	- Phân tích tử thức và mẫu thức thành nhân tử;\\
	- Chia cả tử thức và mẫu thức cho nhân tử chung.
\subsection{Ví dụ}
\begin{vd}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2B3]
	Cho phân thức $$M=\dfrac{\left(a^2+b^2+c^2\right)\left(a+b+c\right)^2+(ab+bc+ca)^2}{(a+b+c)^2-(ab+bc+ca)}.$$
	\begin{enumerate}
		\item Tìm các giá trị $a$, $b$, $c$ để phân thức được xác định (tức là để mẫu khác $0$).
		\item Rút gọn phân thức $M$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Ta có 
		\begin{eqnarray*}
			&(a+b+c)^2-(ab+bc+ca)=0& \Leftrightarrow a^2+b^2+c^2+ab+bc+ca=0 \\
			& &\Leftrightarrow 2a^2+2b^2+2c^2+2ab+2bc+2ca=0\\
			& &\Leftrightarrow(a+b)^2+(b+c)^2+(c+a)^2=0\\
			& &\Leftrightarrow a+b=b+c=c+a=0\\
			& &\Leftrightarrow a=b=c=0.
		\end{eqnarray*}
	Với điều kiện để phân thức $M$ được xác định là $a$, $b$, $c$ không đồng thời bằng $0$.
		\item Chú ý rằng $(a+b+c)^2=a^2+b^2+c^2+2(ab+bc+ac)$. Do đó, ta đặt $a^2+b^2+c^2=x$, $ab+bc+ca=y$. Khi đó $(a+b+c)^2=x+2y$. Ta có
		\begin{eqnarray*}
			&M&=\dfrac{x(x+2)+y^2}{x+2y-y}=\dfrac{x^2+2xy+y^2}{x+y}=\dfrac{(x+y)^2}{x+y}=x+y \\
			& &=a^2+b^2+c^2+ab+bc+ca. \\ 
		\end{eqnarray*}
	\end{enumerate}
}
\end{vd}
\begin{vd}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2B3]
	Rút gọn biểu thức
	$$A = \dfrac{(b-c)^3+(c-a)^3+(a-b)^3}{a^2(b-c)+b^2(c-a)+c^2(a-b)}.$$
	\loigiai{
	Phân tích mẫu thức thành nhân tử:
	\begin{eqnarray*}
		&a^2(b-c)+b^2(c-a)+c^2(a-b)&=a^2(b-c)+b^2c-ab^2+ac^2-bc^2 \\
		&&=a^2(b-c)+bc(b-c)-a(b^2-c^2)\\
		&&=(b-c)(a^2+bc-ab-ac)\\
		&&=(b-c)\left[a(a-b)-c(a-b)\right]\\
		&&=(b-c)(a-b)(a-c).
	\end{eqnarray*}	
	Do đó $A = \dfrac{(b-c)^3+(c-a)^3+(a-b)^3}{-(a-b)(b-c)(c-a)}$.\\
	Ta có nhận xét: Nếu $x+y+z=0$ thì $x^3+y^3+z^3=3xyz$. Đặt $b-c=x$, $c-a=y$, $a-b=z$ thì $x+y+z=0$. Theo nhận xét trên
	$$A=\dfrac{x^3+y^3+z^3}{-xyz}= \dfrac{3xyz}{-xyz}=-3.$$
}
\end{vd}
\begin{vd}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2K3]
	Chứng minh rằng với mọi số nguyên $n$ thì phân số $\dfrac{n^3+2n}{n^4+3n^2+1}$ là phân số tối giản.
	\loigiai{
	Để chứng minh phân số đã cho là tối giản, ta sẽ chứng tỏ rằng tử và mẫu chỉ có ước chung là $\pm 1$.\\
	Gọi $d$ là ước của $n^3+2n$ và $n^4+3n^2+1$. Ta có:\\
		$n^3 + 2n \, \vdots \, d \Rightarrow n\left(n^3+2n\right) \, \vdots \, d \Rightarrow n^4+2n^2 \, \vdots \, d, $ \hfill $(1)$\\
	$n^4+3n^2+1 - \left(n^4+2n^2\right) = n^2+1 \, \vdots \, d \Rightarrow \left(n^2+1\right)^2 =n^4+2n^2+1 \, \vdots \,d,$ \hfill $(2)$\\
	Từ (1) và (2) suy ra 
	$$ \left(n^4+2n^2+1\right) - \left(n^4+2n^2\right) \, \vdots \, d \Rightarrow 1 \, \vdots \, d \Rightarrow d = \pm 1.$$
	Vậy $\dfrac{n^3+2n}{n^4+3n^2+1}$ là phân số tối giản.
}
\end{vd}
\begin{vd}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2K3]
	Chứng minh rằng:
	$$1+x+x^2+x^3+...+x^{31}= (1+x)\left(1+x^2\right)\left(1+x^4\right)\left(1+x^8\right)\left(1+x^{16}\right).\quad (1)$$
	\loigiai{
	Gọi vế trái của đẳng thức (1) là $A$, vế phải là $B$. \\
	Ta có $(1-x)\cdot A = 1-x^{32}$ theo hằng đẳng thức $8$. \\
	$(1-x) \cdot B = (1-x)(1+x)\left(1+x^2\right)\left(1+x^4\right)\left(1+x^8\right)\left(1+x^16\right)=1-x^{32}$.	\\
	Nếu $x\ne1$ thì $A$ và $B$ đều bằng phân thức $\dfrac{1-x^{32}}{1-x}$. Do đó $A=B$.\\
	Nếu $x=1$ thì hai vế của (1) đều bằng $32$. Do đó $A=B$.\\
	Trong cả hai trường hợp, đằng thức (1) đều đúng.	
}
\end{vd}
\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2B3]
	Tìm giá trị của $x$ để các phân thức sau bằng $0$:
	\begin{enumEX}[a)]{2}
		\item $\dfrac{x^4+x^3+x+1}{x^4-x^3+2x^2-x+1}$;
		\item $\dfrac{x^4-5x^2+4}{x^4-10x^2+9}$.
	\end{enumEX}
	\loigiai{
	\begin{enumerate}
		\item Phân thức $\dfrac{x^4+x^3+x+1}{x^4-x^3+2x^2-x+1}$ bằng $0$ khi tử thức bằng $0$ và mẫu thức khác $0$.\\
		Ta có \begin{eqnarray*}
			&x^4+x^3+x+1&=0 \\
			&x^3(x+1)+(x+1)&=0\\
			&(x+1)\left(x^3+1\right)&=0\\
			&x&=-1.
		\end{eqnarray*}
	Thay $x=-1$ vào mẫu thức ta được $(-1)^4-(-1)^3+2(-1)^2-(-1)+1 \ne 0$.\\
	Vậy $x=-1$ thỏa yêu cầu bài.
		\item Phân thức $\dfrac{x^4-5x^2+4}{x^4-10x^2+9}$ bằng $0$ khi tử thức bằng $0$ và mẫu thức khác $0$.
		Ta có \begin{eqnarray*}
			&x^4-5x^2+4&=0\\
			&\left(x^2-1\right)\left(x^2-4\right)&=0.\\
				\end{eqnarray*}
		Với $x^2=1$ thay vào mẫu thức, ta được $1-10+9=0$ nên loại. \\
		Với $x^2=4$ thay vào mẫu thức, ta được $(4)^2-10 \cdot 4 + 9 \ne 0$ nên $x=\pm 2$ thỏa yêu cầu bài.
	\end{enumerate}
}
\end{bt}
\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2K3]
	Rút gọn các phân thức:
	\begin{enumEX}[a)]{2}
		\item $A = \dfrac{1235 \cdot 2469 - 1234}{1234 \cdot 2469}$;
		\item $B= \dfrac{4002}{1000 \cdot 1002 - 999 \cdot 1001}$.
	\end{enumEX}
	\loigiai{
	\begin{enumerate}
		\item Đặt $x = 1234$. Ta có 
		\begin{eqnarray*}
			&A&= \dfrac{(x+1)(2x+1)-x}{x(2x+1)+x+1} \\
			& &= \dfrac{2x^2+2x+1}{2x^2+2x+1} \\
			& &=1.
		\end{eqnarray*}
	
		\item Đặt $x=1000$. Ta có 
		\begin{eqnarray*}
			&B&=\dfrac{4x+2}{x(x+2)-(x-1)(x+1)} \\
			& &=\dfrac{2(2x+1)}{x^2+2x-\left(x^2-1\right)} \\
			& &=\dfrac{2(2x+1)}{2x+1}\\
			& &=2.
		\end{eqnarray*}
	\end{enumerate}
}
\end{bt}
\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2K3]
	Rút gọn các phân thức:
	\begin{enumEX}[a)]{2}
		\item $\dfrac{3x^3-7x^2+5x-1}{2x^3-x^2-4x+3}$;
		\item $\dfrac{(x-y)^3-3xy(x+y)+y^3}{x-6y}$;
		\item $\dfrac{x^2+y^2+z^2-2xy+2xz-2yz}{x^2-2xy+y^2-z^2}$.
	\end{enumEX}
	\loigiai{
	\begin{enumerate}
		\item $\dfrac{3x^3-7x^2+5x-1}{2x^3-x^2-4x+3} = \dfrac{(3x-1)(x-1)^2}{(2x+3)(x-1)^2} = \dfrac{3x-1}{2x+3}$;
		\item \begin{eqnarray*}
			&\dfrac{(x-y)^3-3xy(x+y)+y^3}{x-6y}&= \dfrac{x^3-3x^2y+3xy^2-y^3-3x^2y-3xy^2+y^3}{x-6y} \\
			& &= \dfrac{x^2\left(x-6y\right)}{x-6y} \\
			& &=x^2;
		\end{eqnarray*}
		\item $\dfrac{x^2+y^2+z^2-2xy+2xz-2yz}{x^2-2xy+y^2-z^2} = \dfrac{(x-y+z)^2}{(x-y+z)(x-y-z)}= \dfrac{x-y+z}{x-y+z}$.
	\end{enumerate}
}
\end{bt}
\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2K3]
	Rút gọn các phân thức với $n$ là số tự nhiên:
	\begin{enumEX}[a)]{3}
		\item $\dfrac{(n+1)!}{n!(n+2)}$;
		\item $\dfrac{n!}{(n+1)!-n!}$;
		\item $\dfrac{(n+1)!-(n+2)!}{(n+1)!+(n+2)!}$.
	\end{enumEX}
	\loigiai{
	\begin{enumerate}
		\item $\dfrac{(n+1)!}{n!(n+2)} =\dfrac{n!(n+1)}{n!(n+2)} =\dfrac{n+1}{n+2}$;
		\item $\dfrac{n!}{(n+1)!-n!} = \dfrac{n!}{n!(n+1-1)} = \dfrac{1}{n}$;
		\item $\dfrac{(n+1)!-(n+2)!}{(n+1)!+(n+2)!} = \dfrac{(n+1)!-(n+1)!(n+2)}{(n+1)!+(n+1)!(n+2)} =\dfrac{(n+1)!(-n-1)}{(n+1)!(n+3)} = \dfrac{-n-1}{n+3}$.
	\end{enumerate}
	
}
\end{bt}
\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2K3]
	Rút gọn các phân thức:
	\begin{enumEX}[a)]{2}
		\item $\dfrac{a^2(b-c)+b^2(c-a)+c^2(a-b)}{ab^2-ac^2-b^3+bc^2}$;
		\item $\dfrac{2x^3-7x^2-12x+45}{3x^3-19x^2+33x-9}$;
		\item $\dfrac{x^3-y^3+z^3+3xyz}{(x+y)^2+(y+z)^2+(z-x)^2}$;
		\item $\dfrac{x^3+y^3+z^3-3xyz}{(x-y)^2+(y-z)^2+(z-x)^2}$.
	\end{enumEX}
\loigiai{
	\begin{enumerate}
		\item $\dfrac{a^2(b-c)+b^2(c-a)+c^2(a-b)}{ab^2-ac^2-b^3+bc^2} = \dfrac{(a-b)(b-c)(a-c)}{(a-b)(b-c)(b+c)} = \dfrac{a-c}{b+c}$;
		\item $\dfrac{2x^3-7x^2-12x+45}{3x^3-19x^2+33x-9} = \dfrac{(x-3)^2(2x+5)}{(x-3)^2(3x-1)} = \dfrac{2x+5}{3x-1}$;
		\item $\dfrac{x^3-y^3+z^3+3xyz}{(x+y)^2+(y+z)^2+(z-x)^2} = \dfrac{1}{2}(x-y+z)$;
		\item $\dfrac{x^3+y^3+z^3-3xyz}{(x-y)^2+(y-z)^2+(z-x)^2} = \dfrac{1}{2}(x+y+z)$.
	\end{enumerate}
}
\end{bt}
\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2K3]
	Chứng minh rằng các phân số sau tối giản với mọi số tự nhiên $n$:
	\begin{enumEX}[a)]{4}
		\item $\dfrac{3n+1}{5n+2}$;
		\item $\dfrac{12n+1}{30n+2}$;
		\item $\dfrac{n^3+2n}{n^4+3n^2+1}$;
		\item $\dfrac{2n+1}{2n^2-1}$.
	\end{enumEX}
	\loigiai{
	\begin{enumerate}
		\item Giả sử $(3n+1,5n+2)=d$.\\
		Ta có $3(5n+2)-5(3n+1) \, \vdots \, d \Rightarrow 1 \, \vdots \, d \Rightarrow d = \pm 1$.\\
		Vậy phân số $\dfrac{3n+1}{5n+2}$ là phân số tối giản.
		\item Giả sử $(12n+1,30n+2) = d$.\\
		Ta có $5(12n+1) - 3(30n + 2)\, \vdots \, d \Rightarrow -1 \, \vdots \, d \Rightarrow d = \pm 1$.\\
		Vậy phân số $\dfrac{12n+1}{30n+2}$ là phân số tối giản.
		\item Giả sử $\left(n^4+3n^2+1,n^4+3n^2+1\right) =d$. \\
		Ta có $\left(n^4+3n^2+1\right) - n\left(n^3+2n\right) = n^2+1 \, \vdots \, d$.\\
		Do đó $\left(n^4+3n^2+1\right) - \left(n^2+1\right)^2=n^2 \, \vdots \, d$.\\
		Suy ra $1 \, \vdots \, d \Rightarrow d = \pm 1$.
		\item Giả sử $d \in \text(ƯC)\left(2n+1,2n^2-1\right) \Rightarrow n(2n+1)-\left(2n^2-1\right)=n+1 \, \vdots \, d \Rightarrow 2n + 2 \, \vdots \, d$\\ $\Rightarrow (2n+2)-(2n+1)=1 \, \vdots \, d \Rightarrow d = \pm 1$.
	\end{enumerate}
}
\end{bt}
\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2B3]
	Chứng minh rằng phân số $\dfrac{n^7+n^2+1}{n^8+n+1}$ không tối giản với mọi số nguyên dương $n$.
	\loigiai{
	Ta có $n^7+n^2+1 = \left(n^2+n+1\right) \left(n^5-n^4+n^2-n+1\right)$, $n^8+n+1 = \left(n^6-n^5+n^3-n^2+1\right) \left(n^2+n+1\right)$.\\
	Tử và mẫu cùng chứa thừ số $n^2+n+1$ lớn hơn $1$ n phân số $\dfrac{n^7+n^2+1}{n^8+n+1}$ không tối giản với mọi số nguyên dương $n$.
}
\end{bt}
\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2K3]
	Viết gọn biểu thức dưới dạng một phân thức: 
	$$\left(x^2-x+1\right) \left(x^4-x^2+1\right)\left(x^8-x^4+1\right)\left(x^{16}-x^8+1\right)\left(x^{32}-x^{16}+1\right).$$
	\loigiai{
	Đặt $A=\left(x^2-x+1\right) \left(x^4-x^2+1\right)\left(x^8-x^4+1\right)\left(x^{16}-x^8+1\right)\left(x^{32}-x^{16}+1\right)$.\\
	Nhân biểu thức $A$ với $x^2+x+1$, ta được 

		$$\left(x^2+x+1\right) \cdot A = \left(x^{64}+x^{32}+1\right).$$

	Do $x^2+x+1 \ne 0$ nên $A =\dfrac{x^{64}+x^{32}+1}{x^2+x+1}$.
}
\end{bt}
\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2G3]
	Cho biết $x$, $y$, $z$ khác $0$ và $\dfrac{(ax+by+cz)^2}{x^2+y^2+z^2} = a^2+b^2+c^2$. Chứng minh $\dfrac{a}{x}=\dfrac{b}{y}=\dfrac{c}{z}.$
	\loigiai{
	Ta có \begin{eqnarray*}
		&\dfrac{(ax+by+cz)^2}{x^2+y^2+z^2}&= a^2+b^2+c^2\\
		&(ax)^2+(by)^2+(cz)^2+2abxy+2acxz+2bcyz&=\left(a^2+b^2+c^2\right)\cdot \left(x^2+y^2+z^2\right) \\
		&(ay-bx)^2+(az-cx)^2+(bz-cy)^2&=0
	\end{eqnarray*}
	Do đó ta được: $\heva{&ay-bx=0\\&az-cx=0\\&bz-cy=0}$.\\ Suy ra $\dfrac{a}{x}=\dfrac{b}{y}=\dfrac{c}{z}$.
}
\end{bt}
\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2B3]
	Cho biết $ax+by+cz=0$. Rút gọn $A =\dfrac{bc(y-z)^2+ca(x-z)^2+ab(x-y)^2}{ax^2+by^2+cz^2}$.
	\loigiai{
	Ta có 
	\begin{eqnarray*}
		&B&=bc(y-z)^2+ca(z-x)^2+(bz-cy)^2=0 \\
		& &=bcy^2+bcz^2+cax^2+abx^2+aby^2-2(bcyz+acxz+abxy). \quad \quad \quad \quad \quad \quad \quad \quad \quad \quad \quad \quad(1)
	\end{eqnarray*}
	Từ giả thiết ta suy ra $a^2x^2+b^2y^2+c^2z^2+2(bcyz+acxz+abxy)=0$. \hfill (2)\\
	Từ (1) và (2) suy ra 
	\begin{eqnarray*}
		&B&=ax^2(b+c)+by^2(a+c)+cz^2(a+b)+a^2x^2+b^2y^2+c^2z^2\\
		& &=ax^2(a+b+c)+by^2(a+b+c)+cz^2(a+b+c)\\
		& &=(a+b+c)\left(ax^2+by^2+cz^2\right). 
	\end{eqnarray*}
	Do đó $A=\dfrac{B}{ax^2+by^2+cz^2} = a+b+c$.
}
\end{bt}
\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2B3]
	Rút gọn biểu thức $A= \dfrac{x^2+y^2+z^2}{(y-z)^2+(z-x)^2+(x-y)^2}$, biết rằng $x+y+z=0$.
	\loigiai{
	Ta có \begin{eqnarray*}
		&x+y+z&=0 \\
		&x^2+y^2+z^2+2xy+2xz+2yz &=0 \\
		&x^2+y^2+z^2&=-(2xy+2xz+2yz). 
	\end{eqnarray*}
	Do đó 
	\begin{eqnarray*}
		&A&= \dfrac{x^2+y^2+z^2}{(y-z)^2+(z-x)^2+(x-y)^2}\\
		& &= \dfrac{x^2+y^2+z^2}{2x^2+2y^2+2z^2-(2xy-2xz-2yz)}\\
		& &= \dfrac{x^2+y^2+z^2}{3\left(x^2+y^2+z^2\right)}\\
		& &= \dfrac{1}{3}.
	\end{eqnarray*}
}
\end{bt}
\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2B3]
	Tính giá trị của biểu thức $A=\dfrac{x-y}{x+y}$, biết $x^2-2y^2=xy$ $(y \ne 0;\, x+y \ne 0)$.
	\loigiai{
	Ta có \begin{eqnarray*}
		&x^2-xy-2y^2=0&\Leftrightarrow x^2+xy-2xy-2y^2=0 \\
		& &\Leftrightarrow x(x+y)-2y(x+y)=0 \\
		& &\Leftrightarrow (x+y)(x-2y)=0.  
	\end{eqnarray*}
	Do $x+y \ne 0$ nên $x=2y$. Vậy $A=\dfrac{2y-y}{2y+y}=\dfrac{y}{3y}=\dfrac{1}{3}$.
}

\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2B3]

\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2K3]
	Tìm số nguyên $x$ để phân thức sau có giá trị là số nguyên:
	\begin{enumEX}[a)]{3}
		\item $\dfrac{3}{2x-1}$;
		\item $\dfrac{5}{x^2+1}$;
		\item $\dfrac{7}{x^2-x+1}$;
		\item $\dfrac{x^2-59}{x+8}$;
		\item $\dfrac{x+2}{x^2+4}$.
	\end{enumEX}
	\loigiai{
	\begin{enumerate}
		\item Để $\dfrac{3}{2x-1}$ có giá trị là số nguyên thì $2x-1 \in \{-3;-1;1;3\}$. Do đó $x\in \{-1;0;1;2\}$.
		\item Để $\dfrac{5}{x^2+1}$ có giá trị là số nguyên thì $x^2+1 \in \{1;5\}$. Do đó $x \in \{-2;0;2\}$.
		\item Để $\dfrac{7}{x^2-x+1}$ có giá trị là số nguyên thì $x^2-x+1 \in \{-7;-1;1;7\}$. Do đó $x \in \{-2;0;1;3\}$.
		\item Để $\dfrac{x^2-59}{x+8}$ có giá trị là số nguyên thì $x^2-59 \,\vdots\, x+8 \Leftrightarrow x^2-64 + \, \vdots \, x + 8 \Leftrightarrow 5 \, \vdots \, x+8$.\\
		Do đó $x\in \{-13;-9;-7;-3\}$.
		\item Để $\dfrac{x+2}{x^+4}$ có giá trị là số nguyên thì $$x+2 \, \vdots \, x^2+4 \Rightarrow  (x+2)(x-2)\, \vdots \, x^2+4 \Rightarrow x^2 + 4 - 8 \, \vdots \, x^2 + 4 \Rightarrow 8 \, \vdots \,  x^2 + 4.$$
		$\bullet$ Xét $x^2+4 = 4 \Leftrightarrow x = 0$ (không thỏa).\\
		$\bullet$ Xét $x^2+4 = 8 \Leftrightarrow x = \pm 2$; thử lại ta thấy $x=-2$ thỏa yêu cầu bài toán.
	\end{enumerate}
}
\end{bt}

\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2G3]
	Chứng minh rằng nên các chữ số $a$, $b$, $c$ khác $0$ thỏa mãn điều kiện $\overline{ab} : c \overline{bc} = a : c$ thì $\overline{abbb}: \overline{bbbc} = a : c$.
	\loigiai{
	Ta có $\overline{abbb}: \cdot c = (1000a+111b)c = 1000ac+111bc=ac+111c(9a+b)$.
	$a \cdot \overline{bbbc} = a (1110b+c) = ac + 1110ab$.\\ 
	Ta cần chứng minh $111c(9a+b)=111ab$, tức là $c(9z+b)=10ab$.\\
	Theo giả thiết ta có 
	\begin{eqnarray*}
		&\overline{ab}\cdot c = \overline{bc} \cdot a&\Rightarrow (10a+b)c=(10b+c)a \\
		& &\Rightarrow 10ac+bc=10ab+ac\\
		& &\Rightarrow 9ac+bc=10ab\\
		& &\Rightarrow c(9a+b)=10ab. 
	\end{eqnarray*}
}

\begin{bt}%[Nguyễn Thành Sơn, dự án EX-C2-8-2019]%[8D2K3]
	Điểm trung bình môn Toán của các học sinh nam và nữ hai lớp $8$A và $8$B được thống kê ở bảng sau
	\begin{center}
		\begin{tabular}{|c|c|c|c|}
			\hline
			&Lớp $8$A& Lớp $8$B& Cả hai lớp $8A$ và $8$B\\
			\hline
			Nam&$7,1$&$8,1$&$7,9$\\
			\hline 
			Nữ&$7,6$&$9,0$& \\
			\hline
			Cả lớp &$7,4$&$8,4$&\\
			\hline
		\end{tabular}
	\end{center}
Tính điểm trung bình môn Toán của các học sinh cả hai lớp $8$A và $8$B.
\loigiai{
Gọi số học sinh nam và nữ của hai lớp $8$A theo thứ tự là $a$ và $b$, số học sinh nam và nữ của lớp $8$B theo thứ tự là $c$ và $d$. Ta cần tìm $\dfrac{7,6b+9d}{b+d}$. Ta có:\\
$\dfrac{7,1a+7,6b}{a+b} = 7,4$ \hfill (1)\\
$\dfrac{8,1c+9d}{c+d} = 8,4$ \hfill (2)\\
$\dfrac{7,1a+8,1c}{a+c}=7,9$ \hfill (3)\\
Từ $(1)$ suy ra $b=1,5a$. Từ $(3)$ suy ra $c=4a$. Từ $(2)$ suy ra $d=0,5c$, do đó $d=2a$. Ta được:\\
$$\dfrac{7,6b+9d}{b+d}=\dfrac{7,6 \cdot 1,5 a + 9 \cdot a}{1,5 a + 2a}= 8,4.$$
Vậy điểm trung bình phải tìm là $8,4$.
}
\end{bt}
