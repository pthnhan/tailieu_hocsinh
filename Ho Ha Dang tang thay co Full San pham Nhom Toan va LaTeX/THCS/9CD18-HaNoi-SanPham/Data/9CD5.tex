\begin{bt}%[Trần Chiến]
	Trong mặt phẳng tọa độ $Oxy$, cho parabol $(P):y=\dfrac{1}{2}x^2$ và đường thẳng $d:y=(m-1)x-m+\dfrac{3}{2}$ ($m$ là tham số).
	\begin{enumerate}
		\item Tìm tọa độ các giao điểm của $(P)$ và $d$ khi $m=1$.
		\item Với những giá trị nào của $m$ thì $(P)$ và $d$ cắt nhau tại hai điểm phân biệt $A(x_1;y_1)$ và $B(x_2;y_2)$ thỏa mãn hoành độ điểm này bằng bình phương hoành độ điểm kia.
		\item Với các dữ kiện ở trên, tìm $m$ để $x_1,x_2$ là độ dài hai cạnh góc vuông của tam giác vuông có độ dài cạnh huyền bằng $4$.
	\end{enumerate}
	\loigiai{
		a) Khi $m=1$ ta có phương trình hoành độ giao điểm là 
		$\dfrac{1}{2}x^2=\dfrac{1}{2}\Leftrightarrow x=\pm 1$.\\
		Vậy các giao điểm là $M\left(1;\dfrac{1}{2}\right)$ và $N\left(-1;\dfrac{1}{2}\right)$.\\
		b) Phương trình hoành độ giao điểm: $\dfrac{1}{2}x^2-(m-1)x+m-\dfrac{3}{2}=0.\quad (1)$ \\
		Để phương trình có hai nghiệm phân biệt thì $\Delta=m^2-4m+4>0\Leftrightarrow (m-2)^2>0 \Leftrightarrow m\neq 2$.\\
		Theo giả thiết ta có $x_1,x_2$ là hai nghiệm của phương trình trên. \\
		Giả sử $x_1=2x_2$, theo định lí Vi-ét ta có:
		\begin{eqnarray*}
			\heva{& x_1+x_2=2m-2 \\ & x_1x_2=2m-3} & \Leftrightarrow & \heva{& 3x_2=2m-2 \\ & 2x_2^2=2m-3} \\
			& \Leftrightarrow & \heva{& x_2=\dfrac{2m-2}{3} & \\ & 2\left(\dfrac{2m-2}{3}\right)^2=2m-3. & (2)} 
		\end{eqnarray*}
		Ta có $(2)\Leftrightarrow 8m^2-16m+8=18m-27 \Leftrightarrow 8m^2-34m+35=0 \Leftrightarrow \hoac{& m=\dfrac{5}{2}\\ & m=\dfrac{7}{4}.}$\\
		Vậy $m=\dfrac{5}{2}$ (khi đó $x_1=2,x_2=1$) hoặc $m=\dfrac{7}{4}$ (khi đó $x_1=1,x_2=\dfrac{1}{2}$).\\
		c) Theo giả thiết, phương trình $(1)$ có hai nghiệm $x_1,x_2$ thỏa mãn
		\allowdisplaybreaks{
			\begin{eqnarray*}
				\heva{& x_1>0\\ & x_2>0\\ & x_1^2+x_2^2=4^2=16} & \Leftrightarrow & \heva{& x_1+x_2>0\\ & x_1x_2>0\\ & (x_1+x_2)^2-2x_1x_2=16} \\
				& \Leftrightarrow & \heva{& 2m-2>0\\ & 2m-3>0\\ & (2m-2)^2-2(2m-3)=16} \\
				& \Leftrightarrow & \heva{& m>1\\ & m>\dfrac{3}{2}\\ & 4m^2-12m-6=0} \\
				& \Leftrightarrow & \heva{& m>\dfrac{3}{2}\\ & m=\dfrac{3\pm\sqrt{15}}{2}} \Rightarrow m=\dfrac{3+\sqrt{15}}{2}.
			\end{eqnarray*}
		}
		Vậy $m=\dfrac{3+\sqrt{15}}{2}$.
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Trong mặt phẳng tọa độ $Oxy$, cho parabol $(P):y=-x^2$ và đường thẳng $d:y=5x+m-2$ ($m$ là tham số).
	\begin{enumerate}
		\item Tìm tọa độ các giao điểm của $(P)$ và $d$ khi $m=-12$.
		\item Xác định tất cả các giá trị của $m$ để $d$ tiếp xúc với $(P)$. Tìm tọa độ tiếp điểm.
		\item Xác định các giá trị của $m$ sao cho $d$ cắt $(P)$ tại hai điểm phân biệt có hoành độ $x_1,x_2$ sao cho $\dfrac{1}{x_1-1}+\dfrac{1}{x_2-1}=2$.
	\end{enumerate}
	\loigiai{
		a) Phương trình hoành độ giao điểm: $x^2+5x+m-2=0. \quad (*)$\\
		Khi $m=-12$, phương trình trở thành $x^2+5x-14=0\Leftrightarrow \hoac{& x=-7\\ &x=2}$.\\
		Vậy các giao điểm là $A(-7;-49)$ và $B(2;-4)$.\\
		b) Để $d$ tiếp xúc với $(P)$ thì $\Delta=5^2-4(m-2)=33-4m=0\Leftrightarrow m=\dfrac{33}{4}$.\\
		Với $m=\dfrac{33}{4}$, thay vào $(*)$ ta được $x=-\dfrac{5}{2}$ suy ra $y=-\dfrac{25}{4}$. Vậy tiếp điểm $M\left(-\dfrac{5}{2};-\dfrac{25}{4}\right)$.\\
		c) Để $d$ cắt $(P)$ tại hai điểm phân biệt thì phương trình $(*)$ có hai nghiệm phân biệt khác $1$, khi đó
		$$\heva{&\Delta=5^2-4(m-2)=33-4m>0\\& 1^2+5\cdot 1+m-2\neq 0} \Leftrightarrow \heva{& m<\dfrac{33}{4}\\ &m\neq -4.}$$
		Ta có 
		\begin{eqnarray*}
			\dfrac{1}{x_1-1}+\dfrac{1}{x_2-1}=2 & \Leftrightarrow & x_1-1+x_2-1=2(x_1-1)(x_2-1)\\
			& \Leftrightarrow & x_1+x_2-2=2[x_1x_2-(x_1+x_2)-1]\\
			& \Leftrightarrow & -5-2=2(m-2+5+1)\\
			& \Leftrightarrow & m=-\dfrac{15}{2}\ \text{(thỏa mãn điều kiện).}
		\end{eqnarray*}
		Vậy $m=-\dfrac{15}{2}$.
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Trong mặt phẳng tọa độ $Oxy$, cho parabol $(P):y=-\dfrac{1}{2}x^2$ và đường thẳng $y=(m-1)x+\dfrac{1}{2}$.
	\begin{enumerate}
		\item Tìm giá trị của $m$ để $d$ tiếp xúc với $(P)$. Tìm tọa độ tiếp điểm.
		\item Tìm $m$ để $d$ cắt $(P)$ tại hai điểm phân biệt có hoành độ $x_1, x_2$ thỏa mãn $(x_1+1)(x_2+1)=12$.
	\end{enumerate}
	\loigiai{
		a) Phương trình hoành độ giao điểm: $\dfrac{1}{2}x^2+(m-1)x+\dfrac{1}{2}=0$. $\quad (*)$\\
		Để $d$ tiếp xúc với $(P)$ thì phương trình $(*)$ có nghiệm kép, khi đó $$\Delta=(m-1)^2-1=m^2-2m=0\Leftrightarrow \hoac{& m=0\\ & m=2.}$$
		Với $m=0$: thay vào $(*)$ ta tìm được $x=1$ suy ra $y=-\dfrac{1}{2}$. Tiếp điểm $A\left(1;-\dfrac{1}{2}\right)$.\\
		Với $m=2$: thay vào $(*)$ ta tìm được $x=-1$ suy ra $y=-\dfrac{1}{2}$. Tiếp điểm $B\left(-1;-\dfrac{1}{2}\right)$.\\
		b) Để $d$ cắt $(P)$ tại hai điểm phân biệt thì $\Delta>0 \Leftrightarrow \hoac{& m>2\\ & m<0}$. Ta có
		\begin{eqnarray*}
			(x_1+1)(x_2+1)=12 & \Leftrightarrow & x_1x_2+x_1+x_2-11=0\\
			& \Leftrightarrow & 1-2(m-1)-11=0\\
			& \Leftrightarrow & m=-4.
		\end{eqnarray*} 
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Trong mặt phẳng toa độ $Oxy$, cho parabol $(P):y=x^2$ và đường thẳng $d:y=(2m+1)x-m^2+1$.
	\begin{enumerate}
		\item Tìm tọa độ các giao điểm của $d$ và $(P)$ khi $m=-1$.
		\item Tìm $m$ để $d$ và $(P)$ tiếp xúc với nhau.
		\item Tìm $m$ để $d$ và $(P)$ có hai giao điểm nằm khác phía với trục tung.
	\end{enumerate}
	\loigiai{
		a) Phương trình hoành độ giao điểm: $x^2-(2m+1)x+m^2-1=0$. $\quad (*)$\\
		Khi $m=-1$, phương trình trở thành $x^2+x=0\Leftrightarrow \hoac{& x=0\\ & x=-1}$.\\
		Với $x=0$ thì $y=0$, giao điểm là $A(0;0)$.\\
		Với $x=-1$ thì $y=1$, giao điểm là $B(-1;1)$.\\
		b) Để $d$ và $(P)$ tiếp xúc nhau thì $\Delta=(2m+1)^2-4(m^2-1)=0 \Leftrightarrow 4m+5=0\Leftrightarrow m=-\dfrac{5}{4}$.\\
		c) Yêu cầu đề bài tương đương với phương trình $(*)$ có hai nghiệm trái dấu. \\
		Khi đó, $m^2-1<0\Leftrightarrow -1<m<1$.
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Trong mặt phẳng tọa độ $Oxy$, cho parabol $(P):y=x^2$ và đường thẳng $d:y=2mx-m^2+m-1=0$ ($m$ là tham số).
	\begin{enumerate}
		\item Tìm tọa độ các giao điểm của $(P)$ và $d$ khi $m=1$.
		\item Tìm $m$ để $d$ cắt $(P)$ tại hai điểm phân biệt có hoành độ $x_1,x_2$ thỏa mãn biểu thức $A=x_1x_2-x_1-x_2$ đạt giá trị nhỏ nhất.
	\end{enumerate}
	\loigiai{
		a) Phương trình hoành độ giao điểm $x^2-2mx+m^2-m+1=0.\quad (*)$\\
		Khi $m=1$, phương trình trở thành $x^2-2x+1=0\Leftrightarrow x=1$. Suy ra giao điểm là $A(1;1)$.\\
		b) Để phương trình có hai nghiệm phân biệt thì $$\Delta'=m^2-(m^2-m+1)>0 \Leftrightarrow m-1>0\Leftrightarrow m>1.$$
		Ta có $A=m^2-m+1-2m=m^2-3m+1=\left(m-\dfrac{3}{2}\right)^2-\dfrac{5}{4}\geq -\dfrac{5}{4}$.\\
		Dấu bằng xảy ra khi $m=\dfrac{3}{2}$ (thỏa mãn điều kiện).\\
		Vậy với $m=\dfrac{3}{2}$ thì biểu thức $A$ đạt giá trị nhỏ nhất là $-\dfrac{5}{4}$.
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Trong mặt phẳng tọa độ $Oxy$, cho đường thẳng $d:y=2(m+1)x-m^2-3$ và parabol $(P):y=x^2$.
	\begin{enumerate}
		\item Tìm tất cả các giá trị của $m$ để $d$ luôn cắt $(P)$ tại hai điểm phân biệt.
		\item Tìm tất cả các giá trị của $m$ để tổng khoảng cách từ hai giao điểm trên đến trục tung bằng $10$.
	\end{enumerate}
	\loigiai{
		a) Phương trình hoành độ giao điểm: $x^2-2(m+1)x+m^2+3=0.\quad (*)$\\
		Ta có $\Delta'=(m+1)^2-(m^2+3)=2m-2$.\\
		Để $d$ luôn cắt $(P)$ tại hai điểm phân biệt thì $\Delta'>0\Leftrightarrow m>2$.\\
		b) Giả sử phương trình $(*)$ có hai nghiệm phân biệt là $x_1$ và $x_2$, theo giả thiết ta có $$|x_1|+|x_2|=10\Leftrightarrow x_1^2+x_2^2+2|x_1x_2|=100\Leftrightarrow (x_1+x_2)^2-2x_1x_2+2|x_1x_2|=100.$$
		Lại có $x_1x_2=m^2+3>0\ \forall m$ nên $|x_1x_2|=x_1x_2=m^2+m$.\\
		Khi đó ta có $|x_1|+|x_2|=10\Leftrightarrow (x_1+x_2)^2=100\Leftrightarrow x_1+x_2=\pm 10$.
		\begin{itemize}
			\item TH1: $x_1+x_2=10$, kết hợp với $x_1+x_2=2(m+1)$ (định lí Vi-ét) ta có $2(m+1)=10\Leftrightarrow m=4$ (thỏa mãn).
			\item TH2: $x_1+x_2=-10$, kết hợp với $x_1+x_2=2(m+1)$ (định lí Vi-ét) ta có $2(m+1)=-10\Leftrightarrow m=-6$ (loại).
		\end{itemize}
		Vậy $m=4$ thỏa điều kiện bài toán.
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Trong mặt phẳng tọa độ $Oxy$, cho parabol $(P):y=x^2$ và đường thẳng $\Delta:y=x-m$.
	\begin{enumerate}
		\item Tìm tọa độ các giao điểm của $\Delta$ và $(P)$ khi $m=-6$.
		\item Tìm các giá trị của $m$ để $\Delta$ cắt $(P)$ tại hai điểm phân biệt $A(x_1;y_1)$ và $B(x_2;y_2)$ sao cho $(x_2-x_1)^4+(y_2-y_1)^4=18$.
	\end{enumerate}
	\loigiai{
		a) Phương trình hoành độ giao điểm $x^2-x+m=0$.\\
		Khi $m=-6$, phương trình trở thành $x^2-x-6=0\Leftrightarrow \hoac{& x=3\\ & x=-2}$.\\
		Vậy các giao điểm là $M(3;9)$ và $N(-2;4)$.	\\
		b) Để $\Delta$ cắt $(P)$ tại hai điểm phân biệt thì phương trình $(*)$ có hai nghiệm phân biệt. \\
		Khi đó, $\Delta=1-4m>0\Leftrightarrow m<\dfrac{1}{4}$. Ta có
		\allowdisplaybreaks{
			\begin{eqnarray*}
				(x_2-x_1)^4+(y_2-y_1)^4=18 & \Leftrightarrow & (x_2-x_1)^4+[(x_2-m)-(x_1-m)]^4=18\\
				& \Leftrightarrow & (x_2-x_1)^4+(x_2-x_1)^4=18\\
				& \Leftrightarrow & (x_2-x_1)]^4=9\\
				& \Leftrightarrow & [(x_2+x_1)^2-4x_1x_2]^2=9\\
				& \Leftrightarrow & (1-4m)^2=9\\
				& \Leftrightarrow & \hoac{& m=-\dfrac{1}{2}\\ & m=1.}
			\end{eqnarray*}
		}
		So với điều kiện, ta nhận $m=-\dfrac{1}{2}$.
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Cho đường thẳng $d:y=2(m-1)x-m$ và parabol $(P):y=(m-2)x^2$. 
	\begin{enumerate}
		\item Tìm tọa độ các giao điểm của $d$ và $(P)$ khi $m=3$.
		\item Biết $d$ cắt $(P)$ tại hai điểm có hoành độ lần lượt là độ dài hai cạnh góc vuông của một tam giác vuông. Tìm $m$ để độ dài đường cao ứng với cạnh huyền của tam giác vuông đó bằng $\dfrac{2}{\sqrt{5}}$.
	\end{enumerate}
	\loigiai{
		a) Phương trình hoành độ giao điểm: $(m-2)x^2-2(m-1)x+m=0.\quad (*)$\\
		Khi $m=3$, phương trình trở thành $x^2-4x+3=0\Leftrightarrow \hoac{& x=1\\ & x=3}$.\\
		Vậy các giao điểm là $A(1;1)$ và $B(3;9)$.\\
		b) Ta có $\Delta'=(m-1)^2-(m-2)m=1>0 $ nên phương trình $(*)$ có hai nghiệm phân biệt $x_1,x_2$, do đó $d$ luôn cắt $(P)$ tại hai điểm phân biệt. Theo giả thiết, ta có 
		\allowdisplaybreaks{
			\begin{eqnarray*}
				\dfrac{1}{x_1^2}+\dfrac{1}{x_2^2}=\dfrac{5}{4} & \Leftrightarrow & \dfrac{x_1^2+x_2^2}{x_1^2x_2^2}=\dfrac{5}{4}\\
				& \Leftrightarrow & 4(x_1^2+x_2^2)=5(x_1x_2)^2\\
				& \Leftrightarrow & 4[(x_1+x_2)^2-2x_1x_2]=5(x_1x_2)^2\\
				& \Leftrightarrow & 4\left[\left(\dfrac{2m-2}{m-2}\right)^2- \dfrac{2m}{m-2}\right]=5\left(\dfrac{m}{m-2}\right)^2\\
				& \Leftrightarrow & 4[(2m-2)^2-2m(m-2)]=5m^2\\
				& \Leftrightarrow & 3m^2-16m+16=0\\
				& \Leftrightarrow & \heva{& m=\dfrac{4}{3}\\ & m=4.}
			\end{eqnarray*}
		}
		Vậy $m=4$ hoặc $m=\dfrac{4}{3}$.
	}
\end{bt}

\begin{bt}%[9D4K6]
	Trong mặt phẳng tọa độ $Oxy$, cho đường thẳng $(d)\colon y=(m+2)x+3$ và parabol $(P) \colon y=x^2.$
	\begin{enumerate}
		\item Chứng minh $(d)$ luôn cắt $(P)$ tại hai điểm phân biệt.
		\item Chứng minh hai giao điểm của $(d)$ và $(P)$ luôn nằm khác phía nhau đối với trục tung.
		\item Tìm tất cả giá trị của $m$ để $(d)$ cắt $(P)$ tại hai điểm phân biệt có hoành độ là các số nguyên.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Phương trình hoành độ giao điểm $(d)$ và $(P)$ là
			$$(m+2)x+3=x^2\Leftrightarrow x^2-(m+2)x-3=0.$$
			Ta có $\Delta=(m+2)^2+12>0, \forall m\in\mathbb{R}$.\\
			Suy ra $(d)$ luôn cắt $(P)$ tại hai điểm phân biệt.
			\item Gọi $x_1, x_2$ là hoành độ hai giao điểm của $(d)$ và $(P)$. Theo vi-ét $$x_1x_2=-3<0.$$ Suy ra hai giao điểm luôn khác phía đối với trục tung.
			\item Gọi $x_1, x_2$ là hoành độ hai giao điểm của $(d)$ và $(P)$.\\
			Ta có $$\heva{x_1=\dfrac{m+2+\sqrt{\Delta}}{2}\\x_2=\dfrac{m+2-\sqrt{\Delta}}{2}}\Rightarrow\heva{x_1+x_2&=m+2\\x_1x_2&=-3.}$$
			Do $$\heva{&x_1\in\mathbb{Z}\\&x_2\in\mathbb{Z}} \Rightarrow \heva{&x_1 \in \mathbb{Z}\\&x_2+x_2\in\mathbb{Z}} \Rightarrow \heva{&m+\sqrt{\Delta}\in\mathbb{Z}\\&m+2\in\mathbb{Z}} \Rightarrow \heva{&m\in\mathbb{Z}\\ &\sqrt{\Delta}\in\mathbb{Z}} \Rightarrow \heva{&m\in\mathbb{Z}\\ &\sqrt{(m+2)^2+12}\in\mathbb{Z}.}$$
			$\Rightarrow\exists k\in\mathbb{Z}:(m+2)^2+12=k^2
			\Rightarrow k^2-(m+2)^2=12 \\
			\Rightarrow \heva{&(m+2)^2=4\\&k^2=16} 
			\Rightarrow (m+2)^2=4
			\Rightarrow \hoac{&m+2=2 \Rightarrow m=0\\&m+2=-2 \Rightarrow m=-4.}$\\
			Thử lại, ta thấy có hai giá trị của m thỏa mãn yêu cầu bài toán là $0$ và $-4$. 
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[9D4B4]%[9D4B6]
	Cho parabol $y=x^2$ và đường thẳng $d \colon y=mx-m+4\text{ ($x$ là ẩn số và $m$ là tham số)}$.
	\begin{enumerate}
		\item Tìm giao điểm của parabol và đường thẳng $d$ khi $m = 4$.
		\item Chứng minh rằng với mọi $m$, parabol và đường thẳng luôn giao nhau tại hai điểm có hoành độ phân biệt $x_1$ và $x_2$. Tìm tất cả các giá trị nguyên dương của $m$ để $(5x_1 - 1)(5x_2 - 1) < 0$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $m = 4\Rightarrow d \colon y=4x-4+4=4x$.\\
			Phương trình hoành độ giao điểm:
			$$4x=x^2\Leftrightarrow x^2-4x=0\Leftrightarrow \hoac{x&=0\Rightarrow y=0\\x&=4\Rightarrow y=16}.$$
			Vậy khi $m=4$, parabol và đường thẳng cắt nhau tại hai điểm $(0;0)$ và $(4;16)$.
			\item Ta có $\triangle = (-m)^2 - 4(m - 4) = m^2 - 4m + 16 = (m - 2)^2 + 12 > 0, \forall m \in \mathbb{R}$ nên parabol và $d$ luôn giao nhau tại hai điểm phân biệt có hoành độ $x_1, x_2$, với mọi giá trị của $m$.\\
			Theo định lý Vi - ét ta có $\begin{cases}
			x_1 + x_2 = m\\ x_1x_2 = m - 4
			\end{cases}$. Suy ra
			\begin{eqnarray*}
				&& (5x_1  - 1)(5x_2 - 1) < 0\\
				&\Leftrightarrow &25x_1x_2 - 5(x_1 + x_2) + 1 < 0\\
				&\Leftrightarrow &25(m - 4) - 5m + 1 < 0\\

\begin{bt}%[9D4B4]%[9D4G6]

			$$x^2 = 2(m+1)x - 6m + 4. $$
			Tương đương $x^2 - 2(m+1) + 6m - 4 = 0$.\\
			$\Delta' = (m+1)^2 - (6m-4) = m^2 - 4m + 5  = (m-2)^2 + 1 > 0$ với mọi $m$.\\

			\item  Theo định lý Vi-ét, ta có $x_1, x_2$ thỏa mãn $\heva{&x_1 + x_2 = 2m + 2\\ &x_1x_2 = 6m - 4.}$\\

				\Leftrightarrow & 4m^2 - 6m = 0\\
				\Leftrightarrow & \hoac{m = 0\\ m = \dfrac{3}{2}.}

	}

\begin{bt}%[9D4K5]%[9D4K6]

\begin{bt}%[9D4G6]
	Cho parabol $y=x^2$ và đường thẳng $d\colon y=mx-m+1$ ( với $m$ là tham số). Gọi $x_1$ và $x_2$ là hoành độ giao điểm của hai đường trên. Tìm giá trị nhỏ nhất của biểu thức $$P= \dfrac{2x_1x_2+3}{x_1^2+x_2^2+2(x_1x_2+1)}.$$
	\loigiai
	{ 
		Phương trình hoành độ giao điểm
		$$x^2 = mx - m + 1 .$$
		Tương đương $x^2 - mx + m - 1 = 0$.\\
		$\Delta = m^2 - 4(m-1) = m^2 - 4m + 4 = (m-2)^2$.\\
		Vậy phương trình luôn có nghiệm.\\
		Để phương trình có hai nghiệm phân biệt thì $\Delta > 0 \Leftrightarrow m \ne 2$.\\
		Nhận thấy $a + b + c = 0$ nên phương trình có một nghiệm bằng $1$, nghiệm còn lại bằng $\dfrac{c}{a} = m - 1$.\\
		Vì $P$ đối xứng theo $x_1$ và $x_2$ nên $P = \dfrac{2(m-1)+ 3}{1 + (m-1)^2 + 2m} = \dfrac{2m + 1}{m^2 + 2}$ (*).\\
		Từ (*) ta có $Pm^2 - 2m + 2P - 1 = 0$\\
		Coi đây là một phương trình bậc hai theo $m$, xét $\Delta' = 1 - P(2P-1)$\\
		$\Delta' \ge 0 \Leftrightarrow -2P^2 + P + 1 \ge 0 \Leftrightarrow (P-1)(2P+1) \le 0 \Leftrightarrow -\dfrac{1}{2} \le P \le 1$.\\
		Vậy để có giá trị $m$ thỏa (*) thì phải có $-\dfrac{1}{2} \le P \le 1$.\\
		Do đó $P \ge -\dfrac{1}{2}$.\\
		Ta có $\dfrac{2m + 1}{m^2 + 2} = -\dfrac{1}{2} \Leftrightarrow m^2 + 2 = -4m - 2 \Leftrightarrow m^2 + 4m + 4 = 0 \Leftrightarrow m = -2$.\\
		Kết luận: $\min P = -\dfrac{1}{2}$, đạt được khi $m = -2$.\\
	}
\end{bt}

			$M=\dfrac{-24}{(x_1+x_2)^2-8x_1x_2}=\dfrac{-24}{4m^2-8(m-2)}=\dfrac{-6}{m^2-2m+4}=\dfrac{-6}{(m-1)^2+3}\geq -2. $\\
			Vậy giá trị nhỏ nhất của $M$ là $-2$ khi $m=1$.
		\end{enumerate}	
	}
\end{bt}

	Trong mặt phẳng tọa độ $Oxy$ cho parabol $(P)\colon y=-x^2$ và đường thẳng $(d)\colon y=mx-m-2$.
	\begin{enumerate}
		\item
		Chứng minh rằng	với mọi giá trị của $m$ đường thẳng $(d)$ luôn cắt parabol $(P)$ tại hai điểm có hoành độ phân biệt $x_1$, $x_2$.
		\item
		Tìm $m$ để $\left |x_1-x_2\right |=\sqrt{20}.$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item
			Phương trình hoành độ giao điểm của $(d)$ và $(P)$ là
			\begin{align*}
			-x^2=mx-m-2\Leftrightarrow x^2+mx-m-2=0. \tag{1}
			\end{align*}
			Ta có $\Delta= (m+2)^2+4>0$ với mọi $m$ nên phương trình $(1)$ luôn có hai nghiệm phân biệt với mọi $m$. Do đó với mọi giá trị $m$ thì $(d)$ luôn cắt $(P)$ tại hai điểm phân biệt.
			\item 	
			Theo Vi-ét ta có $\heva{&x_1+x_2=-m\\&x_1x_2=-(m+2).}$\\
			Do đó $\left |x_1-x_2\right |=\sqrt{20}\Leftrightarrow (x_1+x_2)^2-4x_1x_2=20\Leftrightarrow m^2+4m-12=0\Leftrightarrow \hoac{&m=2\\&m=-6.}$
		\end{enumerate}		
	}

\begin{bt}%[Đề số 10]%[Thịnh Trần, dự án 9CD-HaNoi]
	Cho parabol $(P)\colon y=x^2$ và đường thẳng $(d)\colon y=mx$ với $m$ là tham số.
	\begin{enumerate}
		\item
		Tìm các giá trị của $m$ để $(P)$ và $(d)$ cắt nhau tại điểm có tung độ bằng $9$.
		\item
		Tìm các giá trị của $m$ để $(P)$ và $(d)$ cắt nhau tại hai điểm mà khoảng cách giữa hai điểm này bằng $\sqrt{6}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item 
			Vì $(P)$ và $(d)$ cắt nhau tại điểm có tung độ bằng $9$ nên $9=x^2\Leftrightarrow x=\pm 3\Rightarrow m=\pm 3$.
			\item
			Phương trình hoành độ giao điểm của $(d)$ và $(P)$ là
			\begin{align*}
			x^2=mx\Leftrightarrow \hoac{&x=0\\&x=m.}
			\end{align*}
			Từ đó tọa độ hai giao điểm là $A(0;0)$, $B(m;m^2)$ với $m\neq 0$.\\
			Ta có $AB=\sqrt{6}\Leftrightarrow m^2+m^4=6\Leftrightarrow m^2=2.$ Vậy $m=\pm \sqrt{2}.$
		\end{enumerate}		
	}
\end{bt}
