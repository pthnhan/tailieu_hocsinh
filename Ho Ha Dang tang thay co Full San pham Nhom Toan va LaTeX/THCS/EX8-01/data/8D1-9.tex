\section{CHIA ĐA THỨC CHO ĐƠN THỨC}
\subsection{Tóm tắt lí thuyết}
	\begin{enumerate}[\tickEX]
		\item Muốn chia đa thức $ A $ cho đơn thức $ B $ (trường hợp các hạng tử của đa thức $ A $ đều chia hết cho đơn thức $ B $), ta chia mỗi hạng tử của $ A $ cho $ B $ rồi cộng các kết quả với nhau. 
		\begin{eqnarray*}
			\left( A+B\right) : C = (A:C) + (B:C).
		\end{eqnarray*}
		\item Một đa thức muốn chia hết cho đơn thức thì tất cả các hạng tử của đa thức phải chia hết cho đơn thức.
		\item Quy tắc chia đa thức cho đơn thức:
\begin{itemize}
	\item $ x^m : x^n = x^{m-n} $ nếu $ m>n $.
	\item $ x^m : x^n = 1 $ nếu $ m=n $.
	\item $ \left(x^m \right)^n = x^{m\cdot n} $.
	\item $ x^0 = 1; 1^n = 1. $
	\item $ \left(-x \right)^n = x^n $ nếu $ n  $ chẵn.
	\item $ \left(-x \right)^n = -x^n $ nếu $ n  $ lẻ.
\end{itemize}		
	\end{enumerate}
\subsection{Các ví dụ}
\begin{dang}[Thực hiện phép tính chia đa thức cho đơn thức.]
Sử dụng các công thức sau:
\begin{itemize}
	\item $ x^m : x^n = x^{m-n} $ nếu $ m>n $.
	\item $ x^m : x^n = 1 $ nếu $ m=n $.
	\item $ \left(x^m \right)^n = x^{m\cdot n} $.
	\item $ x^0 = 1; 1^n = 1. $
	\item $ \left(-x \right)^n = x^n $ nếu $ n  $ chẵn.
	\item $ \left(-x \right)^n = -x^n $ nếu $ n  $ lẻ.
	\item $ (A+B):C=(A:C)+(B:C). $
\end{itemize}
\end{dang}
\begin{vd}
Thực hiện phép tính: $ \left(8x^5-3x^4+2x^3-4x \right):(-2x). $
\loigiai{
	\begin{eqnarray*}
	&&\quad \left(8x^5-3x^4+2x^3-4x \right):(-2x)\\&
	&= \left(8x^5 \right):	(-2x) + (-3x^4):(-2x) + \left( 2x^3\right):(-2x) + \left( -4x\right):(-2x)\\
	&&=-4x^4 +\dfrac{3}{2}x^3 - x^2 + 2x.
	\end{eqnarray*}
 }
\end{vd}
\begin{vd}
Thực hiện phép tính:	$ \left( 4x^7 -3x^3 - 4x^2\right) : \left( 7x^2\right)$.
\loigiai{
\begin{eqnarray*}
	&&\quad \left( 4x^7 -3x^3 - 4x^2\right) : \left( 7x^2\right)\\
	&&= \left( 4x^7 \right) : \left( 7x^2\right)+\left( -3x^3 \right) : \left( 7x^2\right)+\left( - 4x^2\right) : \left( 7x^2\right)\\
	&&= \dfrac{4}{7}x^5 - \dfrac{3}{7}x-\dfrac{4}{7}.
\end{eqnarray*}	
}
\end{vd}
\begin{vd}
Thực hiện phép tính:	$ \left(7y^4-3y^2 + 2y \right):\left( 2y\right) $.
\loigiai{
\begin{eqnarray*}
	&&\quad \left(7y^4-3y^2 + 2y \right):\left( 2y\right)\\
	&&= \left(7y^4\right):\left( 2y\right)+\left(-3y^2  \right):\left( 2y\right)+\left(2y \right):\left( 2y\right)\\
	&&= \dfrac{7}{2}y^3 -\dfrac{3}{2}y + 1.
\end{eqnarray*}	
}
\end{vd}
\begin{vd}
	Thực hiện phép tính: $ \left( 2x^5y^3 - 4x^3y^5 + 6xy^4\right):\left( 2xy \right) $.
	\loigiai{
	\begin{eqnarray*}
	&& \quad \left( 2x^5y^3 - 4x^3y^5 + 6xy^4\right):\left( 2xy \right)\\
	&& = (2xy)\cdot \left( x^4y^2 - 2x^2y^4 + 3y^3 \right):(2xy).\\
	&&= x^4y^2 - 2x^2y^4 + 3y^3.	
	\end{eqnarray*}	
	}
\end{vd}
\begin{vd}
	Thực hiện phép tính: $\left( 9x^3y^3 - 18x^2y^4-6x^3y^2\right) :\left( 3x^2y\right)$.
\loigiai{
\begin{eqnarray*}
	&&\quad \left( 9x^3y^3 - 18x^2y^4-6x^3y^2\right) :\left( 3x^2y\right)\\
	&&=\left( 3x^2y\right) \cdot  \left( 3xy^2 - 6y^3-2xy\right) :\left( 3x^2y\right)\\
	&&=3xy^2 - 6y^3-2xy.
\end{eqnarray*}	
}
\end{vd}
\begin{vd}
	Thực hiện phép tính: $\left( 4x^4 - 8x^2y+24x^2y^5\right):\left( -4x^2\right) $.
	\loigiai{
	\begin{eqnarray*}
		&&\quad \left( 4x^4 - 8x^2y+24x^2y^5\right):\left( -4x^2\right)\\
		&&= \left( -4x^2\right) \cdot \left(-x^2 + 2y-6y^5 \right)\\
		&&= -x^2 + 2y-6y^5.
	\end{eqnarray*}	
	}
\end{vd}
\begin{vd}
	Thực hiện phép tính: $ \left( x^3 - 2xy^2 + 6x^2y^3\right) :\left( -\dfrac{1}{2}x\right)$.
	\loigiai{
	\begin{eqnarray*}
	&& \quad \left( x^3 - 2xy^2 + 6x^2y^3\right) : \left( -\dfrac{1}{2}x\right)\\
	&& = \left( -\dfrac{1}{2}x\right) \cdot \left( -2x^2 + 4y^2 -12xy^3 \right): \left( -\dfrac{1}{2}x\right)\\
	&&= -2x^2 + 4y^2 -12xy^3.	
	\end{eqnarray*}	
	}
\end{vd}

	\item Thay giá trị của biến đã cho vào biểu thức rút gọn để tìm giá trị của biểu thức đã cho.

\end{dang}
\begin{vd}
Tính giá trị biểu thức $ A = 15a^4b^3c^2:5ab^2c^2$ tại $ a = 2 $, $ b =- 10 $, $ c =  2018.$
\loigiai{
Ta có \begin{eqnarray*}
A = 15a^4b^3c^2:5ab^2c^2 = 3a^3b.	
\end{eqnarray*}	
Thay $ a = 2$, $ b =- 10 $ ta được $ A = 3\left( 2\right)^3 \cdot \left(-10 \right) =-240.$}
\end{vd}
\begin{vd}
Tính giá trị biểu thức 
$ P = \left[3\left( x-y\right)^4+2\left( x-y\right)^3-5\left( x-y\right)^2  \right] :\left( x-y\right)^2 $
tại $ x = 1 $, $ y = -7 $.
\loigiai{
Đặt $ t = x - y $, ta có
\begin{eqnarray*}
	P &=& \left[3\left( x-y\right)^4+2\left( x-y\right)^3-5\left( x-y\right)^2  \right] :\left( x-y\right)^2\\
	&=& \left( 3t^4+2t^3-5t\right):t^2\\
	&=&3t^2+2t-5.
\end{eqnarray*}	
Với $ x = 1 $, $ y = -7 $ suy ra $ t  = 8$. \\
Do vậy $ P = 3\cdot 8^2 + 2\cdot 8 - 5 = 203. $
}
\end{vd}
\begin{dang}[Tìm điều kiện để phép chia đa thức là chia hết.]
\textbf{Phương pháp:} Tìm điều kiện để hạng tử có bậc thấp nhất trong đa thức chia hết cho đơn thức.
\end{dang}
\begin{vd}
	Tìm số tự nhiên $ n $ để phép chia sau là chia hết \begin{eqnarray*}
	\left( 3x^3 - 7x^2 + 2x \right):4x^n.	
	\end{eqnarray*}
	\loigiai{
	Để phép chia là 	chia hết khi $ x \ \vdots \ x^n $.\\
	Suy ra $ n = 0 $ hoặc $ n = 1. $
	}
\end{vd}
\begin{vd}

		\left( 13x^4y^3 - 7x^3y^4 + 2x^2y^3 \right):3x^n	y^n.
	\end{eqnarray*}
	\loigiai{
		Để phép chia là 	chia hết khi $ \heva{&x^2 \ \vdots \ x^n\\ &y^3 \ \vdots \ y^n. }$\\
		Do $ n $ là số tự nhiên nên   $n \in \left\{0;1;2\right\}.$
	}
\end{vd}

\begin{vn}

	\begin{eqnarray*}
	&& \quad \left( 30x^3y^4 - 15x^3y^2+3x^3y^5\right) : \left( 3x^2y^2\right)\\
	&& = \left( 3x^2y^2 \right) \cdot \left( 10xy^2 - 5x+xy^3\right) : \left( 3x^2y^2\right)	\\
	&&= 10xy^2 - 5x+xy^3.
	\end{eqnarray*}	
	}

\begin{vn}

	\begin{eqnarray*}
		&& \quad \left( 5xy^2 - 3xy + 6x^3y^2\right): \left( -xy\right)\\
		&& = \left( -xy\right) \cdot \left( -5y + 3 - 6x^2y\right): \left( -xy\right)\\

\begin{vn}
	Thực hiện phép tính: $ \left[2(a-b)^2+3(b-a)^3-4(a-b)^4\right] : \left( b-a\right)^2.$
	\loigiai{
	Nhận xét: $ (b-a)^2 = \left(-1 \cdot (a-b) \right)^2 = (a-b)^2. $	
	\begin{eqnarray*}
	&& \quad \left[2(a-b)^2+3(b-a)^3-4(a-b)^4\right] : \left( b-a\right)^2 \\
	&& = \left[2(b-a)^2+3(b-a)^3-4(b-a)^4\right] : \left( b-a\right)^2	\\
	&& = 2 + 3(b-a) - 4(b-a)^2.
	\end{eqnarray*}
	}
\end{vn}

\begin{vn}
	Thực hiện phép tính: $ \left[(x-y)^4 - 3(y-x)^3+2(x-y)^2\right] : \left( x^2 - 2xy+y^2\right)$.
	\loigiai{
\begin{nx}
\begin{itemize}
	\item $ x^2 - 2xy+y^2 = (x-y)^2 $.
	\item $ (y-x)^3 = (y-x)\cdot (x-y)^2. $
\end{itemize}
\end{nx} 	 
Ta có:
\begin{eqnarray*}
&&\quad \left[(x-y)^4 - 3(y-x)^3+2(x-y)^2\right] : \left( x^2 - 2xy+y^2\right)\\
&& = \left[(x-y)^4 - 3(y-x)^3+2(x-y)^2\right] : \left( x-y\right)	^2\\
&& = (x-y)^2 - 3(y-x) + 2.
\end{eqnarray*}
	}
\end{vn}
\begin{vn}
	Tính giá trị biểu thức
	\begin{enumerate}
		\item $ A = \left( 15x^4y^5-17x^3y^2\right):xy^2 $ tại $ x = 2 $, $ y = 3;$
		\item $ B = \left[5\left( x-y\right)^6+2\left( x-y\right)^5-7\left( x-y\right)^3\right] : \left( y-x\right)^2$ tại $ x = 7 $, $ y = 13 $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $$ A = \left( 15x^4y^5-17x^3y^2\right):xy^2 = 15x^3y^3  - 17x^2.$$
			Thay $ x = 2 $, $ y = 3 $, ta được
			$$A = 15 \cdot 2^3 \cdot 3^3 - 17 \cdot 2^2 = 1012.$$
			\item Đặt $t = x-y$ suy ra $(x-y)^2 = (y-x)^2 = t^2 $. Ta có 
			\begin{eqnarray*}
				B &=& \left[5\left( x-y\right)^6+2\left( x-y\right)^5-7\left( x-y\right)^3\right] : \left( y-x\right)^2\\
				&=& \left[5t^6 + 2t^5-7t^3\right]:t^2\\
				&=&	5t^4 +2t^3-7t.
			\end{eqnarray*}
			Theo giả thiết ta có $ x = 7 $, $ y = 13 $ nên $  t = -6 $.\\ Vậy $ B = 5 \left( -6\right)^4 + 2\left(-6 \right)^3-7\left(-6 \right) = 6090.$
		\end{enumerate}	
	}
\end{vn}
\begin{vn}
	Tính giá trị biểu thức $C =  \left( x^2+6\right)\left( x^2+4\right):\left( x^2+5\right) $ tại $ x = -2. $
	\loigiai{Đặt $ t = x^2+5 $. Ta có
		\begin{eqnarray*}
			&&\quad \left( x^2+6\right)\left( x^2+4\right):\left( x^2+5\right)	\\
			&& = \left( t+1\right)\left( t-1\right):t\\
			&& = \left( t^2-1\right):t\\
			&& = t - \dfrac{1}{t}.
		\end{eqnarray*}	
		Với $ x = -2 $ suy ra $ t = 9 $. Ta được
		$$C = 9 - \dfrac{1}{9} =\dfrac{80}{9}. $$
	}
\end{vn}
