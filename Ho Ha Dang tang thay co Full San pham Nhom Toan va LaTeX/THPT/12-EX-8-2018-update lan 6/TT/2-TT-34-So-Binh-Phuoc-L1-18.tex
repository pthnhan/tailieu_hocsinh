\begin{ex}%[Đề thi thử QG lần 1- Sở Bình Phước -2018]%[2D1K2-4]%[Trịnh Văn Xuân -Ex-8]

	\choice
	{$\left(-3;\dfrac{-5}{2}\right)$}
	{\True $\left(-5;\dfrac{-7}{2}\right)$}
	{$(-2;-1)$}
	{$\left(\dfrac{-7}{2};-3\right)$}
	\loigiai{
Ta có $y'=x^2-2ax-3a$. Để hàm số đã cho có hai điểm cực trị $x_1$, $x_2$ thì phương trình $y'=0$ có hai nghiệm phân biệt $\Leftrightarrow x^2-2ax-3=0\ (1)$ có hai nghiệm phân biệt. Phương trình $(1)$ có hai nghiệm phân biệt khi và chỉ khi $\Delta'=a^2+3a>0 \Leftrightarrow \hoac{&a>0\\ &a<-3.}$	\\
Áp dụng định lí Vi-et ta được $x_1+x_2=2a\ (2)$.\\
Vì $x_1$ là nghiệm của phương trình $(1)$ nên ta được
$$x_1^2-2ax_1-3a=0\Rightarrow x_1^2+2ax_2+9a=\left(x_1^2-2ax_1-3a\right)+2a\left(x_1+x_2\right)+12a=4a^2+12a.$$
Tương tự ta có $x_2^2+2ax_1+9a=4a^2+12a$. Khi đó
\begin{align*}
\dfrac{x_1^2+2ax_2+9a}{a^2}+\dfrac{a^2}{x_2^2+2ax_1+9a}=2&\Leftrightarrow \dfrac{4a^2+12a}{a^2}+\dfrac{a^2}{4a^2+12a}=2\\
&\Leftrightarrow \dfrac{4a+12}{a}+\dfrac{a}{4a+12}=2\\
&\Leftrightarrow (4a+12)^2+a^2-2a(4a+12)=0\\
&\Leftrightarrow (4a+12-a)^2=0\Leftrightarrow a=-4\in\left(-5;\dfrac{-7}{2}\right).
\end{align*}
}
\end{ex}

\begin{ex}%[Đề thi thử QG lần 1- Sở Bình Phước -2018]%[1H3G4-3]%[Trịnh Văn Xuân -Ex-8]
	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình vuông cạnh $a$, $SA$ vuông góc với mặt phẳng đáy và $SA=x$. Xác định $x$ để hai mặt phẳng $(SBC)$ và $(SCD)$ hợp với nhau góc $60^\circ$.
	\choice
	{$x=\dfrac{3a}{2}$}
	{$x=\dfrac{a}{2}$}
	{\True $x=a$}
	{$x=2a$}
	\loigiai{
\begin{center}
	\begin{tikzpicture}[scale=0.7]
	\coordinate (S) at (0,5);
	\coordinate (A) at (0,0);
	\coordinate (B) at (6,0);
	\coordinate (C) at (4,-2.5);
	\coordinate (D) at ($(A)+(C)-(B)$);
	\coordinate (I) at ($(S)!.45!(B)$);
	\coordinate (J) at ($(S)!.45!(D)$);
	\draw (S)--(D)--(C)--(B)--(S)--(C);
	\draw [dashed] (S)--(A)--(D) (B)--(A)--(I)--(J)--(A)--(C) (B)--(D);
	\tkzDrawPoints(S,A,B,C,D,I,J)
	\tkzLabelPoints[above](S)
	\tkzLabelPoints[left](A,J)
	\tkzLabelPoints[below](C,D)
	\tkzLabelPoints[right](B,I)
%	\tkzMarkAngles[size=.83, fill=green, opacity=0.4](I,A,J)
%	\tkzLabelAngle[pos=0.57,rotate=-12](I,A,J){\footnotesize $60^\circ$}
	\tkzMarkRightAngle(S,I,A)
	\tkzMarkRightAngle(A,J,S)
	\end{tikzpicture}
\end{center}	
Trong mặt phẳng $(SAB)$ dựng $AI\perp SB$, ta được $AI\perp (SBC)\quad (1)$.\\
Trong mặt phẳng $(SAD)$ dựng $AJ\perp SD$, ta được $AJ\perp (SCD)\quad (2)$.\\
Từ $(1)$ và $(2)$ suy ra góc $\big((SBC),(SCD)\big)=(AI,AJ)=\widehat{IAJ}$.\\
Mặt khác, ta có $\dfrac{1}{AI^2}=\dfrac{1}{AS^2}+\dfrac{1}{AB^2}$, $\dfrac{1}{AJ^2}=\dfrac{1}{AS^2}+\dfrac{1}{AD^2}$.\\
Suy ra $AI=AJ$. Do đó nếu góc $\widehat{IAJ}=60^\circ$ thì $\triangle AIJ$ đều $\Rightarrow AI=AJ=IJ$.\\
Xét $\triangle SAB$ vuông tại $A$ có $AI$ là đường cao $\Rightarrow AI\cdot SB=SA\cdot AB\Rightarrow AI=\dfrac{SA\cdot AB}{SB}\quad (3)$.\\
Và có $SA^2=SI\cdot SB\Rightarrow SI=\dfrac{SA^2}{SB}\quad (4)$; $SA^2=SJ\cdot SD\Rightarrow SJ=\dfrac{SA^2}{SD}\quad (4')$.\\
Suy ra $IJ\parallel BD\ (\text{vì }SB=SD)\Rightarrow \dfrac{IJ}{BD}=\dfrac{SI}{SB}\Rightarrow IJ=\dfrac{SI\cdot BD}{SB}=\dfrac{SA^2\cdot BD}{SB^2}\quad (5)$.\\
Thế $(3)$ và $(5)$ vào $AI=IJ$ suy ra 
$$AB=\dfrac{SA\cdot BD}{SB}\Leftrightarrow AB\cdot SB=SA\cdot BD\Leftrightarrow a\cdot \sqrt{x^2+a^2}=x\cdot a\cdot\sqrt 2\Leftrightarrow x^2+a^2=2x^2\Leftrightarrow x=a.$$
}
\end{ex}
