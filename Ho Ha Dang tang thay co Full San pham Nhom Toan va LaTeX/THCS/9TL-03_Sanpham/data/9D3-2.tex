
\section{PHƯƠNG PHÁP GIẢI HỆ PHƯƠNG TRÌNH}
\subsection{Nền tảng kiến thức}
\subsubsection{Phương pháp thế}
Để giải hệ phương trình bằng phương pháp thế ta thực hiện theo các bước sau:
\begin{enumerate}[\textbf{Bước} 1.]
	\item Biểu thị một ẩn (giả sử ẩn $x$) theo ẩn còn lại (ẩn $y$) từ một trong các phương trình của hệ.
	\item Thay biểu thức của $x$ vào phương trình còn lại rồi tìm giá trị của $y$.  
	\item Thay giá trị $y$ vừa tìm được vào biểu thức của $x$ để tìm giá trị của $x$.
	\item Kết luận nghiệm của hệ phương trình\end{enumerate}
\begin{vd}
	Giải các hệ phương trình sau bằng phương pháp thế
	\begin{enumEX}{2}
		\item $\heva{&4x+5y=3\\&x-3y=5.}$
		\item $\heva{&7x-2y=1\\&3x+y=6.}$
		\item $\heva{&5x+3y=1\\&2x+y=-1.}$
		\item $\heva{&x+y\sqrt{5}=0\\&x\sqrt{5}+3y=1-\sqrt{5}.}$
	\end{enumEX}
	\loigiai{ 
		\begin{enumEX}{1}
			\item $\heva{&4x+5y=3\\&x-3y=5} \Leftrightarrow \heva{&x=5+3y\\&4x+5y=3}\Leftrightarrow \heva{&x=5+3y\\&4(5+3y)+5y=3}\\\Leftrightarrow \heva{&x=5+3y\\&17y=-17} \Leftrightarrow \heva{&x=5+3y\\&y=-1}\Leftrightarrow \heva{&x=5+3\cdot(-1)=2\\&y=-1.}$ \\ 
			Vậy hệ phương trình đã cho có nghiệm là $(x;y)=(2;-1)$.
			\item $\heva{&7x-2y=1\\&3x+y=6}  \Leftrightarrow \heva{&y=6-3x\\&7x-2y=1} \Leftrightarrow \heva{&y=6-3x\\&7x-2(6-3x)=1} \Leftrightarrow \heva{&y=6-3x\\&13x=13} \Leftrightarrow \heva{&y=3\\&x=1.}$\\ 
			Vậy hệ phương trình đã cho có nghiệm là $(x;y)=(1;3).$
			\item $\heva{&5x+3y=1\\&2x+y=-1} \Leftrightarrow \heva{&y=-1-2x\\&5x+3y=1} \Leftrightarrow  \heva{&y=-1-2x\\&5x+3(-1-2x)=1}\\ \Leftrightarrow \heva{&y=-1-2x\\&x=-4} \Leftrightarrow \heva{&y=-1-2\cdot(-4)=7\\&x=-4.}$\\ 
			Vậy nghiệm của hệ phương trình đã cho là $(x;y)=(-4;7).$
			\item $\heva{&x+y\sqrt{5}=0\\&x\sqrt{5}+3y=1-\sqrt{5}} \Leftrightarrow \heva{&x=-y\sqrt{5}\\& x\sqrt{5}+3y=1-\sqrt{5}} \Leftrightarrow \heva{&x=-y\sqrt{5}\\ &-y\sqrt{5}\sqrt{5}+3y=1-\sqrt{5} } \\ \Leftrightarrow \heva{&x=-y\sqrt{5}\\&y=\dfrac{-1+\sqrt{5}}{2}} \Leftrightarrow \heva{&x=\dfrac{\sqrt{5}}{2} -\dfrac{5}{2}\\&y=\dfrac{-1+\sqrt{5}}{2}.}$\\ Vậy hệ phương trình đã cho có nghiệm là $(x;y)=\left(\dfrac{\sqrt{5}}{2}-\dfrac{5}{2}; \dfrac{-1+\sqrt{5}}{2}\right).$
		\end{enumEX}
		
	}
\end{vd}

\subsubsection{Phương pháp cộng đại số}
Để giải hệ phương trình bằng phương pháp cộng đại số ta thực hiện các bước sau:
\begin{enumerate}[\textbf{Bước} 1.]
	\item Nhân cả hai vế của các phương trình trong hệ với số thích hợp (nếu cần) để đưa hệ đã cho về hệ mới, trong đó các hệ số của một ẩn nào đó bằng nhau (hoặc đối nhau).
	\item Trừ ( hoặc cộng ) từng vế của các phương trình trong hệ mới để khử bớt một ẩn.
	\item Giải phương trình một ẩn vừa thu được.
	\item Thay giá trị tìm được của ẩn này vào một trong các phương trình của	hệ để tìm ẩn còn lại.
	\item Kết luận nghiệm của hệ phương trình.
\end{enumerate}
\begin{vd}
	Giải các hệ phương trình sau bằng phương pháp cộng đại số.
	\begin{enumEX}{2}
		\item $\heva{&x-3y=1\\&2x+3y=11.}$
		\item $\heva{&2x-y=3\\&x-y=1.}$
		\item $\heva{&3x+4y=18\\&4x-3y=-1.}$
		\item $\heva{&\sqrt{3}x-\sqrt{2}y=1\\&\sqrt{2}x+3\sqrt{3}y=4\sqrt{6}.}$
	\end{enumEX}
	\loigiai{\begin{enumEX}{1}
			\item $\heva{&x-3y=1\\&2x+3y=11}\Leftrightarrow \heva{&3x=12\\&x-3y=1} \Leftrightarrow \heva{&x=4\\&4-3y=1} \Leftrightarrow \heva{&x=4\\&y=1.}$ \\ Vậy hệ phương trình đã cho có nghiệm là $(x;y)=(4;1).$
			\item $\heva{&2x-y=3\\&x-y=1} \Leftrightarrow \heva{&x=2\\&x-y=1} \Leftrightarrow \heva{&x=2\\&2-y=1} \Leftrightarrow \heva{&x=2\\&y=1.}$\\ Vậy hệ phương trình đã cho có nghiệm là $(x;y)=(2;1).$
			\item $\heva{&3x+4y=18\\&4x-3y=-1} \Leftrightarrow \heva{&12x+16y=72\\&12x-9y=-3} \Leftrightarrow \heva{&25y=75\\&12x-9y=-9} \Leftrightarrow \heva{&y=3\\&12x-9\cdot3=-9} \Leftrightarrow \heva{&x=2\\&y=3.}$ \\ Vậy hệ phương trình đã cho có nghiệm là $(x;y)=(2;3).$
			\item $\heva{&\sqrt{3}x-\sqrt{2}y=1\\&\sqrt{2}x+3\sqrt{3}y=4\sqrt{6}} \Leftrightarrow \heva{&\sqrt{6}x-2y=\sqrt{2}\\ &\sqrt{6}x+9y=12\sqrt{2} } \Leftrightarrow \heva{&11y=11\sqrt{2}\\&\sqrt{6}x-2y=\sqrt{2}} \Leftrightarrow \heva{&y=\sqrt{2}\\&\sqrt{6}x -2\sqrt{2}=\sqrt{2}}\\ \Leftrightarrow \heva{&x=\sqrt{3}\\&y=\sqrt{2}.}$\\ Vậy hệ phương trình đã cho có nghiệm là $(x;y)=\left(\sqrt{3};\sqrt{2}\right)$.		
	\end{enumEX}}
\end{vd}

\subsubsection{Phương pháp đặt ẩn phụ}
Để giải hệ phương trình ta còn dùng phương pháp đặt ẩn phụ thông qua các ẩn đã cho.\\Với dạng này ta cần nhận biết được sự tương đồng của các ẩn từ đó chọn ẩn phụ đặt cho hợp lý để đưa về hệ phương trình bậc nhất hai ẩn rồi áp dụng phương pháp thế hoặc phương pháp cộng đại số để giải. Sau khi tìm được nghiệm theo ẩn mới, sau đó ta thay lại ẩn ban đầu để tìm nghiệm của hệ đã cho. 
\begin{vd}
	Giải các hệ phương trình.
	\begin{enumEX}{3}
		\item $\heva{&\dfrac{1}{x}-\dfrac{1}{y}=1\\&\dfrac{3}{x}+\dfrac{4}{y}=5.}$
		\item $\heva{&\dfrac{1}{x-2}-\dfrac{1}{y-1}=2\\&\dfrac{2}{x-2}+\dfrac{3}{y-1}=1.}$
		\item $\heva{&\dfrac{4}{3}\left(\dfrac{1}{x}+\dfrac{1}{y}\right)=1\\&\dfrac{1}{6x}+\dfrac{1}{5y}=\dfrac{2}{15}.}$
	\end{enumEX}
	\loigiai{\begin{enumEX}{1}
			\item Điều kiện xác định $x\ne 0, y\ne 0$.\\ Đặt $a=\dfrac{1}{x}, b=\dfrac{1}{y},$ hệ phương trình đã cho trở thành\\ $ \heva{&a-b=1\\&3a+4b=5} \Leftrightarrow \heva{&a=1+b\\&3a+4b=5}\Leftrightarrow \heva{&a=1+b\\&3(1+b)+4b=5} \Leftrightarrow \heva{&a=1+b\\&7b=2}\\ \Leftrightarrow \heva{&a=1+\dfrac{2}{7}\\& b=\dfrac{2}{7}}\Leftrightarrow \heva{&a=\dfrac{9}{7}\\&b=\dfrac{2}{7}} $. Khi đó ta có
			$ \heva{&\dfrac{1}{x}=\dfrac{9}{7}\\&\dfrac{1}{y}=\dfrac{2}{7}} \Leftrightarrow \heva{&x=\dfrac{7}{9}\\&y=\dfrac{7}{2}.} \text{ (nhận) }$\\ Vậy hệ phương trình đã cho có nghiệm là $(x;y)=\left(\dfrac{7}{9};\dfrac{7}{2}\right).$
			\item Điều kiện xác định $x\ne 2 , y\ne 1$.\\  
			Đặt $a=\dfrac{1}{x-2}, b=\dfrac{1}{y-1}$, hệ phương trình đã cho trở thành \\$\heva{&a-b=2\\&2a+3b=1} \Leftrightarrow \heva{&a=2+b\\&2(2+b)+3b=1} \Leftrightarrow \heva{&a=2+b\\&5b=-3} \Leftrightarrow \heva{&a=2+b\\&b=-\dfrac{3}{5}} \Leftrightarrow \heva{&a=\dfrac{7}{5}\\&b=-\dfrac{3}{5}.} 
			$\\
			Từ đó thay vào ta tìm được $ \heva{&x = \dfrac{1}{a} + 2 = \dfrac{19}{7}\\&y = \dfrac{1}{b}+1=\dfrac{2}{5}.} \text{ (nhận) } $\\
			Vậy hệ phương trình đã cho có nghiệm là $(x;y)=\left(\dfrac{19}{7};\dfrac{2}{5} \right) .$
			\item Điều kiện xác định $x\ne 0 , y\ne 0$.\\
			Đặt $a=\dfrac{1}{x}, b=\dfrac{1}{y}$, hệ phương trình đã trở thành \\ $\heva{&4a+4b=3\\& \dfrac{1}{6}a+\dfrac{1}{5}b=\dfrac{2}{15}} \Leftrightarrow \heva{&4a+4b=3\\&5a+6b=4} \Leftrightarrow \heva{&20a+20b=15\\&20a+24b=16} \Leftrightarrow \heva{&4b=1\\&20a+20b=15} \\
			\Leftrightarrow \heva{&b=\dfrac{1}{4}\\&20a +20\cdot \dfrac{1}{4}=15} \Leftrightarrow \heva{&b=\dfrac{1}{4}\\&a=\dfrac{1}{2}.} $ Khi đó ta có $ \heva{&\dfrac{1}{x}=\dfrac{1}{2}\\&\dfrac{1}{y}=\dfrac{1}{4}} \Leftrightarrow \heva{&x=2\\&y=4.} \text{ (nhận) }$ \\ 
			Vậy hệ phương trình đã cho có nghiệm là $ (x;y) = (2;4) $.
	\end{enumEX}}
\end{vd}


\subsection{Áp dụng giải toán}

\begin{dang}{Giải và biện luận hệ phương trình}
	\begin{itemize}
		\item Dùng phương pháp thế, biểu diễn $1$ ẩn theo ẩn còn lại sau đó đưa về phương trình bậc nhất $1$ ẩn.
		\item Giải và biện luận phương trình bậc nhất $1$ ẩn.
		\item Kết luận tập nghiệm của hệ phương trình.
	\end{itemize}
\end{dang}

\begin{vd}
	Cho hệ phương trình ($m$ là tham số)
	\[\heva{&x+my=1&(1)\\&(5m+2)x+3y=m-2. &(2)} \]
Giải và biện luận hệ phương trình theo $m$.
\loigiai{Từ phương trình $(1)$ suy ra $x=1-my$, thay vào phương trình $(2)$ ta được\\
	$$(5m+2)(1-my) +3y=m-2 \Leftrightarrow (5m^2+2m-3)y =4m+4\\ \Leftrightarrow (m+1)(5m-3)y=4(m+1). \quad (3)$$
	\begin{itemize}
		\item Nếu $m=-1 $ thì $(3) \Leftrightarrow 0\cdot y=0$ luôn đúng với mọi $ y \in \mathbb{R}$. Vậy phương trình có vô số nghiệm.
		\item Nếu $m=\dfrac{3}{5}$ thì phương trình $(3) \Leftrightarrow 0 \cdot y=\dfrac{-32}{5}$ (vô lý). Vậy hệ phương trình vô nghiệm.
		\item Nếu $m \ne -1$ và $m \ne \dfrac{3}{5}$ thì $(3) \Leftrightarrow y= \dfrac{4}{5m-3}$, do đó $x= \dfrac{m-3}{5m-3}$. Vậy hệ phương trình có nghiệm duy nhất $(x;y)=\left(\dfrac{m-3}{5m-3};\dfrac{4}{5m-3} \right)$.
	\end{itemize}
	}
\end{vd}
\begin{vd}
	Cho hệ phương trình ($m$ là tham số)
	\[\heva{&3x+my=2 & (1)\\&x+(3m-2)y=m. &(2)} \]
Giải và biện luận hệ phương trình theo $m$.
	\loigiai{Từ phương trình $(2)$ suy ra $x=m-(3m-2)y$, thay vào phương trình $(1)$ ta được\\
		$3m-3(3m-2)y+my=2 \Leftrightarrow (-8m+6)y=2-3m. \quad (3)$
	\begin{itemize}
		\item Nếu $m=\dfrac{3}{4} $ thì $(3) \Leftrightarrow 0\cdot y=-\dfrac{1}{4}$ (vô lý) nên hệ phương trình vô nghiệm.
		\item Nếu $m\ne \dfrac{3}{4}$  thì $(3) \Leftrightarrow y= \dfrac{2-3m}{-8m+6}$, do đó $x= \dfrac{m^2-6m+4}{-8m+6}$.\\
		Hệ phương trình có nghiệm duy nhất $(x;y)=\left( \dfrac{m^2-6m+4}{-8m+6};\dfrac{2-3m}{-8m+6} \right)$.
	\end{itemize}
	}
\end{vd}

\begin{dang}{Xác định tham số để hệ có nghiệm duy nhất}
	Xét hệ $\heva{&ax + by = c\\&a'x + b'y = c'\\}$ với các hệ số chứa tham số.
	\begin{itemize}
		\item Đầu tiên, ta lần lượt xét $a' = 0$, $b' = 0$, $c' = 0$ và kiểm tra các tham số ứng với từng trường hợp.
		\item Sau đó, ta xét $a', b', c' \ne 0$. Khi ấy, hệ có nghiệm duy nhất khi và chỉ khi $\dfrac{a}{a'} \ne \dfrac{b}{b'}$.
	\end{itemize}
\end{dang}

\begin{vd}
	Cho hệ phương trình $\heva{x + my = 2\\mx + y = 3}$. Tìm tất cả các giá trị của tham số $m$ để hệ có nghiệm duy nhất.
	\loigiai{Xét $m = 0$. Khi đó hệ trở thành $\heva{x=2\\y=3}$, rõ ràng có nghiệm duy nhất.\\
		Xét $m \ne 0$. Khi đó hệ có nghiệm duy nhất khi và chỉ khi $\dfrac{1}{m} \ne \dfrac{m}{1} \Leftrightarrow m \ne \pm 1$. \\
		Vậy tất cả các giá trị của tham số $m$ để hệ có nghiệm duy nhất là $m \ne \pm 1$.
	}
\end{vd}	

\begin{vd}
	Cho hệ phương trình $\heva{&ax - y = 2\\ &x + ay = 3.}$ \\
	Chứng minh rằng với mọi $a$ thì hệ có nghiệm duy nhất. Tìm nghiệm đó.
	\loigiai{Xét $a = 0$. Khi đó hệ trở thành $\heva{-y& = 2\\x & = 3,}$ rõ ràng có nghiệm duy nhất là $ (3;-2) $.\\
		Xét $a \ne 0$. Khi đó hệ có nghiệm duy nhất khi và chỉ khi $\dfrac{a}{1} \ne \dfrac{-1}{a} \Leftrightarrow a^2 + 1 \ne 0$.\\
		Ta thấy rằng $a^2 + 1 \ge 1 > 0$ với mọi $a$. Do đó rõ ràng $a^2 + 1 \ne 0$ với mọi $a$. Vậy hệ luôn có nghiệm duy nhất.\\
		Ta có hệ tương đương $$\heva{&ax - 2 = y\\ &x + ay = 3} \Leftrightarrow \heva{&y = ax - 2\\& x + a(ax - 2) = 3}\Leftrightarrow \heva{&y = ax - 2\\ &(a^2 + 1)x = 2a + 3} \Leftrightarrow \heva{&y = \dfrac{3a - 2}{a^2+1}\\& x = \dfrac{2a + 3}{a^2+1}.}$$
		Kết luận: Với mọi $a$, hệ có nghiệm duy nhất là $(x;y) = \left(\dfrac{2a+3}{a^2+1};\dfrac{3a-2}{a^2+1}\right)$.
	}
\end{vd}
\begin{dang}{Xác định tham số để hệ vô nghiệm}
	Xét hệ $\heva{&ax + by = c\\&a'x + b'y = c'\\}$ với các hệ số chứa tham số. Ta lần lượt xét $a' = 0$, $b' = 0$, $c' = 0$ và kiểm tra các tham số ứng với từng trường hợp. Sau đó, xét $a', b', c' \ne 0$. Khi ấy, hệ vô nghiệm khi và chỉ khi $\dfrac{a}{a'} = \dfrac{b}{b'} \ne \dfrac{c}{c'}$.
\end{dang}

\begin{vd}
	Cho hệ phương trình $\heva{&x + y = 1\\&ax + 2y = 0.}$\\
	Tìm tất cả các giá trị của tham số $a$ để hệ vô nghiệm.
	\loigiai{
		Hệ tương đương 	$\heva{&ax + 2y = 0 \\ &x + y = 1.\\}$\\
		Do đó hệ vô nghiệm khi và chỉ khi $\dfrac{a}{1} = \dfrac{2}{1} \ne \dfrac{0}{1} \Leftrightarrow a = 2$.
	}
\end{vd}

\begin{vd}
	Cho hệ phương trình $\heva{&mx + 3my = 2\\&2m^2x + 6m^2y = m.}$ \\
	Chứng minh rằng hệ vô nghiệm với mọi giá trị của tham số $m$.
	\loigiai{
		Xét $m = 0$. Khi đó hệ trở thành $\heva{0x + 0y = 2\\ 0x + 0y = 0}$, rõ ràng hệ vô nghiệm. \\
		Xét $m \ne 0$. Khi đó hệ vô nghiệm khi và chỉ khi $\dfrac{m}{2m^2} = \dfrac{3m}{6m^2} \ne \dfrac{2}{m} (*)$\\
		Ta thấy với mọi $m \ne 0$ thì $\dfrac{m}{2m^2} = \dfrac{3m}{6m^2} = \dfrac{1}{2m}$. \\
		Mặt khác $\dfrac{2}{m} - \dfrac{1}{2m} = \dfrac{3}{2m} \ne 0$ với mọi $m$ khác $0$. Do đó $\dfrac{2}{m} \ne \dfrac{1}{2m}$ với mọi $m$ khác $0$.\\
		Vậy $(*)$ đúng với mọi $m$ khác $0$. Kết hợp với trường hợp trên, ta suy ra hệ vô nghiệm với mọi giá trị của tham số $m$.
	}
\end{vd}

\begin{dang}{Xác định tham số để hệ có vô số nghiệm}
	Xét hệ $\heva{&ax + by = c\\&a'x + b'y = c'\\}$ với các hệ số chứa tham số. Ta lần lượt xét $a' = 0$, $b' = 0$, $c' = 0$ và kiểm tra các tham số ứng với từng trường hợp. Sau đó, xét $a', b', c' \ne 0$. Khi ấy, hệ có vô số nghiệm khi và chỉ khi $\dfrac{a}{a'} = \dfrac{b}{b'} = \dfrac{c}{c'}$.
	
\end{dang}

\begin{vd}
	Cho hệ phương trình $\heva{&2x - y = m\\ &mx + \sqrt{2}y = m.}$\\
	Tìm tất cả các giá trị của tham số $m$ để hệ có vô số nghiệm.
	\loigiai{
		Xét $m = 0$. Khi đó, hệ trở thành $\heva{&2x - y = 0\\ &\sqrt{2}y = 0} \Leftrightarrow \heva{&x=0\\ &y = 0.}$ \\
		Vậy khi $m = 0$ thì hệ có nghiệm duy nhất. Ta loại giá trị này. \\
		Xét $m \ne 0$. Khi đó hệ có vô số nghiệm khi và chỉ khi $\dfrac{2}{m} = \dfrac{-1}{\sqrt{2}} = \dfrac{m}{m}$\\
		Tương đương $\heva{&\dfrac{2}{m} = \dfrac{-1}{\sqrt{2}}\\&\dfrac{2}{m} = 1}$. \\
		Ta thấy hệ này vô nghiệm.\\
		Vậy không có giá trị nào của tham số $m$ để hệ có vô số nghiệm.
	}
\end{vd}

\begin{vd}
	Cho hệ phương trình $\heva{&3x + (m^2+1)y = 5m - 10 \\& -9x + (-3m^2-3)y = -15m+30.}$ \\
	Chứng minh rằng hệ có vô số nghiệm với mọi giá trị của tham số $m$.
	\loigiai{
		Xét $m = 2$. Khi đó hệ trở thành $\heva{&3x + 5y = 0\\&-9x - 15y = 0} \Leftrightarrow \heva{&3x + 5y = 0\\&3x + 5y = 0}$.\\
		Vậy hệ có vô số nghiệm khi $m = 2$. \\
		Xét $m \ne 2$. Khi đó hệ có vô số nghiệm khi và chỉ khi $\dfrac{3}{-9} = \dfrac{m^2 + 1}{-3m^2-3} = \dfrac{5m - 10}{-15m + 30} (*)$\\
		Ta thấy với mọi $m \ne 2$ thì $\dfrac{3}{-9} = \dfrac{m^2 + 1}{-3m^2-3} = \dfrac{5m - 10}{-15m + 30} = \dfrac{1}{3}$. \\
		Do đó $(*)$ đúng với mọi $m \ne 2$. Kết hợp với trường hợp trên, ta suy ra hệ có vô số nghiệm với mọi giá trị của tham số $m$.
	}
\end{vd}
\begin{dang}{Xác định tham số để hệ có nghiệm thỏa điều kiện khác}
	Xét hệ $\heva{&ax + by = c\\&a'x + b'y = c'}$ với các hệ số chứa tham số.\\
	
	Nếu yêu cầu của bài toán liên quan đến tính chất của nghiệm (chẳng hạn tìm $m$ để hệ có nghiệm nguyên), tiến hành rút thế hoặc cộng đại số để giải ra cụ thể giá trị của $x$, $y$ theo tham số. Sau đó, tùy vào yêu cầu của bài toán, tiến hành giải tiếp.
\end{dang}
\begin{vd}
	Cho hệ phương $\heva{&(a+1)x - y = a + 1 \\& x + (a-1)y = 2}$ với tham số $a$. Tìm tất cả các giá trị nguyên của tham số $a$ sao cho hệ có nghiệm nguyên.
	\loigiai{
		Xét $a = 1$. Hệ trở thành $\heva{&2x - y = 2\\ &x = 2} \Leftrightarrow \heva{&y = 2\\ &x = 2.}$ \\ Vậy ta nhận giá trị $a = 1$. \\
		Xét $a \ne 1$. Khi đó hệ có nghiệm duy nhất khi và chỉ khi $\dfrac{a + 1}{1} \ne \dfrac{-1}{a - 1} \Leftrightarrow a \ne 0$. \\
		Vậy xét $a \ne 0$. Khi đó, hệ tương đương
		\begin{align*}
		&\phantom{\Leftrightarrow}~\heva{&y = (a+1)x - (a+1) \\& x + (a-1)y = 2} \\
		&\Leftrightarrow \heva{&y = (a+1)x - (a+1) \\& x + (a-1)(a+1)(x-1) = 2} \\
		&\Leftrightarrow \heva{&y = (a+1)x - (a+1) \\& a^2x + -a^2 + 1 = 2}\\
		&\Leftrightarrow \heva{&y = \dfrac{a + 1}{a^2} \\& x = 1 + \dfrac{1}{a^2}.}
		\end{align*}
		Để hệ có nghiệm nguyên thì $x$ phải nguyên. Do đó $1 + \dfrac{1}{a^2}$ phải nguyên. Vậy $a^2$ phải là ước của $1$. Mặt khác, $a^2 \ge 0$. Vậy $a^2 = 1$. Suy ra $a$ chỉ có thể nhận giá trị $\pm 1$. \\
		Ta không xét $a = 1$. Xét $a = -1$, khi đó $y = 0, x = 2$ (thỏa)\\
		Kết luận: Hệ có nghiệm duy nhất, và nghiệm đó là nghiệm nguyên khi và chỉ khi $a = \pm 1$.
	}
\end{vd}

\begin{vd}
	Cho hệ phương $\heva{&2x + (a-2)y = a + 1\\ &(a+2)x - 2y = 3}$ với tham số $a$. Tìm tất cả các giá trị của tham số $a$ sao cho hệ có nghiệm duy nhất. Trong các giá trị đó, tìm giá trị của $a$ để tổng $x + y$ đạt giá trị lớn nhất.
	\loigiai{
		Xét $a = -2$. Khi đó hệ trở thành $\heva{&2x - 4y = -1\\& - 2y = 3} \Leftrightarrow \heva{&2x - 4y = -1\\& y = \dfrac{-3}{2}} \Leftrightarrow \heva{&x = \dfrac{-7}{2}\\ &y = \dfrac{-3}{2}.} $ \\
		Vậy khi $a = -2$, hệ có nghiệm duy nhất. Suy ra $ x+y = -5 $.\\
		Xét $a \ne -2$. Khi đó hệ có nghiệm duy nhất khi và chỉ khi $\dfrac{2}{a+2} \ne \dfrac{a-2}{-2} \Leftrightarrow a^2 \ne 0 \Leftrightarrow a \ne 0$. \\
		Vậy hệ có nghiệm duy nhất khi và chỉ khi $a \ne 0$.\\
		Xét $a \ne 0$. Hệ tương đương
		\begin{align*} 
		&\phantom{\Leftrightarrow}~\heva{&2x + (a-2)y = a + 1\\ &y = \dfrac{a+2}{2}x - \dfrac{3}{2}}\\
		&\Leftrightarrow \heva{&2x + \dfrac{a^2-4}{2}x - \dfrac{3(a-2)}{2} = a + 1\\ &y = \dfrac{a+2}{2}x - \dfrac{3}{2}} \\
		&\Leftrightarrow \heva{& a^2x = 5a -4\\ &y = \dfrac{a+2}{2}x - \dfrac{3}{2}}\\
		&\Leftrightarrow \heva{& x = \dfrac{5a -4}{a^2}\\ &y = \dfrac{a^2 + 3a - 4}{a^2}.} \end{align*}
		Ta có $x + y=\dfrac{a^2 + 8a - 8}{a^2} = -\dfrac{8}{a^2} + \dfrac{8}{a} + 1 =  -8\left(\dfrac{1}{a} - \dfrac{1}{2} \right)^2 - 1 \le -1 $.\\
		Dấu đẳng thức xảy ra khi và chỉ khi $\dfrac{1}{a} - \dfrac{1}{2} = 0$ hay $a = 2$.\\
		Mà khi $a = 2$ thì $ x+y = -5 < -1 $ nên $ a=2 $ là giá trị cần tìm.} 
\end{vd}

\begin{bt}
	Cho ba đường thẳng
	\begin{align*}
	&(d_1)\colon x-2y=-3;\\
	&(d_2)\colon \sqrt{2}x+y=\sqrt{2}+2;\\
	&(d_m)\colon mx-(1-2m)y=5-m.
	\end{align*}
	\begin{enumEX}{1}
		\item Xác định $m$ để ba đường thẳng $(d_1)$; $(d_2)$ và $(d_m)$ đồng quy.
		\item Chứng minh rằng $(d_m)$ luôn đi qua một điểm cố định với mọi $m$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Tọa độ giao điểm của $(d_1)$ và $(d_2)$ là nghiệm của hệ phương trình
			$$\heva{&x-2y=-3\\&\sqrt{2}x+y=\sqrt{2}+2}\Leftrightarrow \heva{&x=1\\&y=2.}$$
			Vậy tọa độ giao điểm của $(d_1)$ và $(d_2)$ là $M(1;2)$.\\
			Ba đường thẳng $(d_1)$; $(d_2)$ và $(d_m)$ đồng quy khi và chỉ khi 
			$$M\in (d_m)\Leftrightarrow m\cdot 1-(1-2m)\cdot 2=5-m\Leftrightarrow m=\dfrac{7}{6}.$$
			Vậy với $m=\dfrac{7}{6}$ thì ba đường thẳng $(d_1)$; $(d_2)$ và $(d_m)$ đồng quy.
			\item Đặt $E(x_0;y_0)$ là điểm cố định thuộc $(d_m)$. Khi đó
			\begin{eqnarray*}
				& & mx_0-(1-2m)y_0=5-m, \forall m\in \mathbb{R}\\
				&\Leftrightarrow & \left( x_0+2y_0+1 \right) m=y_0+5, \forall m\in \mathbb{R}\\
				&\Leftrightarrow & \heva{&x_0+2y_0+1=0\\ & y_0+5=0}\Leftrightarrow \heva{&x_0=9\\ & y_0=-5.}
			\end{eqnarray*}
			Vậy điểm cố định mà $(d_m)$ luôn đi qua là $E(9;-5)$.
		\end{enumEX}
	}
\end{bt}


\begin{bt}
	Cho hệ phương trình
	$$\heva{&(m-1)x-y=2\\ &mx+y=m.}$$
	\begin{enumEX}{1}
		\item Giải hệ phương trình khi $m=\sqrt{2}$.
		\item Xác định giá trị của $m$ để hệ có nghiệm $(x;y)$ duy nhất thỏa điều kiện $x+y>0$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Với $m=\sqrt{2}$ hệ phương trình đã cho trở thành $\heva{&\left( \sqrt{2}-1 \right) x-y=2\\ &\sqrt{2} x+y=\sqrt{2}.} \quad (I)$
			$$(I)\Leftrightarrow  \heva{&\left( 2\sqrt{2}-1 \right) x=2 +\sqrt{2}\\ &\sqrt{2} x+y=\sqrt{2}} \Leftrightarrow \heva{&x=\dfrac{2+\sqrt{2}}{2\sqrt{2}-1}\\ &y=\sqrt{2}-\sqrt{2} \cdot \dfrac{2+\sqrt{2}}{2\sqrt{2}-1}}
			\Leftrightarrow \heva{&x=\dfrac{6+5\sqrt{2}}{7}\\ &y=\dfrac{-10+\sqrt{2}}{7}.}$$
			Vậy tập nghiệm của hệ phương trình là $S=\left\{ \left( \dfrac{6+5\sqrt{2}}{7}; \dfrac{-10+\sqrt{2}}{7} \right) \right\}$.
			\item 
			Ta có $\heva{&(m-1)x-y=2\\ &m x+y=m} \Leftrightarrow  \heva{&(2m-1)x=2 +m\\ &m x+y=m.}$\\
			\textbf{Trường hợp 1.} $m=\dfrac{1}{2}$, hệ đã cho vô nghiệm.\\
			\textbf{Trường hợp 2.} $m \ne \dfrac{1}{2}$, hệ đã cho tương đương với
			$$ \heva{&x=\dfrac{m+2}{2m-1}\\ &y=m-m\cdot \dfrac{m+2}{2m-1}}
			\Leftrightarrow \heva{&x=\dfrac{m+2}{2m-1}\\ &y=\dfrac{m^2-3m}{2m-1}.}$$
			Suy ra $x+y = \dfrac{m+2}{2m-1}+\dfrac{m^2-3m}{2m-1} = \dfrac{m^2-2m+2}{2m-1} $.\\
			Ta cần có $x+y >0  \Leftrightarrow  \dfrac{m^2-2m+2}{2m-1}>0  \Leftrightarrow  2m-1 >0$ (do $m^2-2m+2>0$) $\Leftrightarrow  m > \dfrac{1}{2}$.\\
			Vậy $m>\dfrac{1}{2}$ thì hệ đã cho có nghiệm $(x;y)$ duy nhất thỏa điều kiện $x+y>0$.
		\end{enumEX}
	}
\end{bt}

\begin{bt}%[Đề thi chuyên Toán ĐHSP Hà Nội năm 2011 - 2012]
	Giải hệ phương trình sau
	$$\heva{&x^2+y^2+\dfrac{2xy}{x+y}=1&(1)\\&\sqrt{x+y}=x^2-y.&(2)}$$
	\loigiai{
		ĐKXĐ: $x+y>0$.
		\begin{align*}
		(1)&\Leftrightarrow (x+y)^2-1=2xy-\dfrac{2xy}{x+y}\\
		&\Leftrightarrow (x+y-1)(x+y+1)=2xy\left(1-\dfrac{1}{x+y)}\right)=\dfrac{2xy(x+y-1)}{x+y}\\
		&\Leftrightarrow \hoac{&x+y-1=0\\&x+y+1=\dfrac{2xy}{x+y}}\\
		&\Leftrightarrow \hoac{&y=1-x\\&x^2+y^2+(x+y)=0\quad\text{(vô nghiệm do $x+y>0$).}}
		\end{align*}
		Thay $y=1-x$ vào $(2)$ ta được 
		$$1=x^2+x-1\Leftrightarrow x^2+x-2=0 \Leftrightarrow (x-1)(x+2) = 0 \Leftrightarrow  \hoac{&x=1\\&x=-2.}$$
		\begin{itemize}
			\item $x=1\Rightarrow y=0$.
			\item $x=-2\Rightarrow y=3$.
		\end{itemize}
		Vậy tập nghiệm của hệ phương trình là $S=\left\{ (1;0);(-2;3)\right\}$.
	}
\end{bt}
