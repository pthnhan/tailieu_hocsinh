	Giải phương trình $\dfrac{\sin x-\sin 2x+\sin 3x}{\cos x-\cos 2x+\cos 3x}=-\dfrac{1}{\sqrt{3}}$.

\begin{bt}%[HK1, Chuyên Lê Khiết, Q-Ngãi]%[1D3B1]
	Cho dãy số $(u_n)$ xác định bởi $\heva{ u_1&=5 \\ u_{n+1}&=u_n+3n-2 \text{ với } n\ge 1}$. \\Bằng phương pháp quy nạp, chứng minh rằng $u_n=5+\dfrac{(n-1)(3n-4)}{2}$.
	\loigiai{
		Bằng phương pháp quy nạp, chứng minh rằng $u_n=5+\dfrac{(n-1)(3n-4)}{2}$. (1)
		\begin{itemize}
			\item Với $n=1$ thì $u_1=5+\dfrac{(1-1)(3\cdot1-4)}{2}=5$ hệ thức (1) đúng. 
			\item Giả sử (1) đúng với $n=k$, tức là: $u_k=5+\dfrac{(k-1)(3k-4)}{2}$. Ta phải chứng minh (1) cũng đúng với $n=k+1$, tức là $u_{k+1}=5+\dfrac{k(3k-1)}{2}$.\\
			Thật vậy, từ giả thiết ta có $$u_{k+1}=u_k+3k-2=5+\dfrac{(k-1)(3k-4)}{2}+3k-2=5+\dfrac{k(3k-1)}{2}.$$
			Vậy hệ thức (1) đúng với mọi $n\in\mathbb{N^*}$.
		\end{itemize}	
	}
\end{bt}

\begin{bt}%[HK1, Chuyên Lê Khiết, Q-Ngãi]%[1H2B4]%[1H2K4]
	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình bình hành. Điểm $M$ di động trên đoạn thẳng $BC$. Lấy các điểm $N, P, Q$ lần lượt thuộc các cạnh $SC, SD, AD$ sao cho $MN \parallel SB$; $NP \parallel CD$; $MQ \parallel CD$. Gọi $K$ là giao điểm của $MN$ và $PQ$. 
	\begin{enumerate}
		\item Chứng minh $PQ \parallel (SAB)$.
		\item Chứng minh rằng $K$ luôn chạy trên một đường thẳng cố định.
		\item Gọi $E$ là điểm nằm trên đoạn $AB$ ($E \not\equiv A$ và $E \not\equiv B$). Tìm giao điểm của $KE$ và $(SCD)$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate} 
			\item \textit{Chứng minh $PQ\parallel (SAB)$.} 
			\immini{
				Ta có:       $$MN\parallel SB\Rightarrow \dfrac{MC}{MB}=\dfrac{NC}{NS}$$
				$$NP\parallel CD\Rightarrow \dfrac{NC}{NS}=\dfrac{PD}{PS}$$ 
				$$MQ\parallel CD\Rightarrow \dfrac{MC}{MB}=\dfrac{QD}{QA}$$ 
				Suy ra:      $$\dfrac{PD}{PS}=\dfrac{QD}{QA}\Rightarrow PQ\parallel SA$$ 
				Mà:    $SA\subset (SAB)$  và  $PQ\not\subset (SAB)$ 
				nên:      $PQ\parallel (SAB)$.
			}{
				\begin{tikzpicture}[line cap=round, line join=round, scale=0.8]
				\tkzDefPoints{0/0/D,6/0/C,  3/2/A, 4/8/S}
				% C là điểm sao cho ABCD là hình thoi
				\tkzDefPointsBy[translation= from D to C](A){B}
				\coordinate(P) at ($(S)!0.6!(D)$);
				\coordinate(N) at ($(S)!0.6!(C)$);
				\coordinate(Q) at ($(D)!0.3!(A)$);
				\coordinate(M) at ($(C)!0.3!(B)$);
				\coordinate(E) at ($(A)!0.3!(B)$);
				\coordinate(F) at ($(D)!0.3!(C)$);
				\tkzInterLL(M,N)(P,Q)\tkzGetPoint{K}
				\tkzInterLL(K,E)(S,F)\tkzGetPoint{G}
				%\tkzDefMidPoint(S,A) \tkzGetPoint{M}
				%\tkzDefMidPoint(A,B) \tkzGetPoint{N}
				\tkzFillPolygon[white](S,A,B)
				\tkzFillPolygon[white](S,B,C)
				
				\tkzDrawPolygon(S,C,D)
				\tkzDrawSegments[dashed](S,A A,D A,B M,Q P,Q E,F K,E S,G)
				\tkzDrawSegments(S,C B,C S,B N,P M,K P,K G,F)
				\tkzDrawLines(S,K)
				
				\tkzLabelPoints[below](A, M, C,D,F,E)
				\tkzLabelPoints[below left](G)
				\tkzLabelPoints[right](B)
				\tkzLabelPoints[left](Q)
				\tkzLabelPoints[above right](N)
				\tkzLabelPoints[above](S,K)
				\tkzLabelPoints[above left](P)
				\end{tikzpicture}
			}
			\item \textit{Chứng minh $K$ luôn chạy trên một đường thẳng cố định.}\\
			
			Ta có: $BC\parallel AD$, $BC\subset (SBC)$ và $AD\subset (SAD)$.\\
			Gọi $d=(SBC)\cap (SAD)$ $\Rightarrow d\parallel BC\parallel AD$ và $S\in d$ $\Rightarrow d$ cố định (vì $S.ABCD$ cố định) \textbf{(1)}\\
			Lại có: $MN\subset (SBC)$; $PQ\subset (SAD)$ và $PQ\cap MN=K\Rightarrow K\in (SBC)\cap (SAD)$ hay $K\in d$. \textbf{(2)}\\
			Từ (1), (2) suy ra $K$ luôn chạy trên một đường thẳng cố định là giao tuyến của hai mặt phẳng $(SAC)$ và $(SBD)$.
			\item \textit{Tìm giao điểm của $KE$ và $(SCD)$.}\\
			Qua $E$ vẽ đường thẳng song song với $AD$ cắt $CD$ tại $F$. Suy ra $EF \parallel d$, suy ra các điểm $S$, $K$, $E$, $F$ cùng nằm trên mặt phẳng  $(EF;d)$.\\
			$(EF; d)\cap (SCD)=SF$. Gọi $G=KE \cap SF$, suy ra $G=KE \cap (SCD)$.
		\end{enumerate}
	}
\end{bt}
