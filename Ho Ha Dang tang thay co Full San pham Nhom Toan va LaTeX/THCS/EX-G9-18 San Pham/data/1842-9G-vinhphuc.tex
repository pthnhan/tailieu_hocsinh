	Rút gọn biểu thức $P=\left(\dfrac{\sqrt{a}+2018}{a+2\sqrt{a}+1}-\dfrac{\sqrt{a}-2018}{a-1}\right)\cdot\dfrac{\sqrt{a}+1}{2\sqrt{a}}$
	\loigiai{
	Ta có 
	\begin{eqnarray*}
	P&=&\dfrac{\sqrt{a}+2018}{\left(\sqrt{a}+1\right)^2}\cdot\dfrac{\sqrt{a}+1}{2\sqrt{a}}-\dfrac{\sqrt{a}-2018}{\left(\sqrt{a}-1\right)\left(\sqrt{a}+1\right)}\cdot\dfrac{\sqrt{a}+1}{2\sqrt{a}}\\

	&=&\dfrac{\left(\sqrt{a}+2018\right)\left(\sqrt{a}-1\right)-\left(\sqrt{a}-2018\right)\left(\sqrt{a}+1\right)}{2\sqrt{a}\left(\sqrt{a}+1\right)\left(\sqrt{a}-1\right)}\\
	&=&\dfrac{4034\sqrt{a}}{2\sqrt{a}\left(a-1\right)}=\dfrac{2017}{a-1}.
	\end{eqnarray*}
	
		}

\begin{ex}%[Đề thi HSG 9, tỉnh Vĩnh Phúc, năm 2018]%[Nguyễn Hiếu, Dự án EX-9G-18]%[9D0K1]
    Cho $x,y,z>0$ thoả mãn $x+y=\left(\sqrt{x}+\sqrt{y}-\sqrt{z}\right)^2$; $\sqrt{x}+\sqrt{y}\neq \sqrt{z}$ và $y\neq z$. Chứng minh rằng
$\dfrac{x+\left(\sqrt{x}-\sqrt{z}\right)^2}{y+\left(\sqrt{y}-\sqrt{z}\right)^2}=\dfrac{\sqrt{x}-\sqrt{z}}{\sqrt{y}-\sqrt{z}}$.
\loigiai
    {
    Ta có 
	\begin{eqnarray*}
	&&x+y=\left(\sqrt{x}+\sqrt{y}-\sqrt{z}\right)^2\\
	&\Leftrightarrow& x+y=\left(\sqrt{x}-\sqrt{z}\right)^2+2\sqrt{y}\left(\sqrt{x}-\sqrt{z}\right)+y\\
	&\Leftrightarrow & x+\left(\sqrt{x}-\sqrt{z}\right)^2=2\left(\sqrt{x}-\sqrt{z}\right)^2+2\sqrt{y}\left(\sqrt{x}-\sqrt{z}\right)^2\\
	&\Leftrightarrow & x+\left(\sqrt{x}-\sqrt{z}\right)^2 =2\left(\sqrt{x}-\sqrt{z}\right)\left(\sqrt{x}+\sqrt{y}-\sqrt{z}\right).
	\end{eqnarray*}	   
	Tương tự ta cũng có $y+\left(\sqrt{y}-\sqrt{z}\right)^2=2\left(\sqrt{y}-\sqrt{z}\right)\left(\sqrt{x}+\sqrt{y}-\sqrt{z}\right)$.\\
Suy ra 
$$\dfrac{x+\left(\sqrt{x}-\sqrt{z}\right)^2}{y+\left(\sqrt{y}-\sqrt{z}\right)}=\dfrac{\left(\sqrt{x}-\sqrt{z}\right)\left(\sqrt{x}+\sqrt{y}-\sqrt{z}\right)}{\left(\sqrt{y}-\sqrt{z}\right)\left(\sqrt{x}+\sqrt{y}-\sqrt{z}\right)}=\dfrac{\sqrt{x}-\sqrt{z}}{\sqrt{y}-\sqrt{z}}.$$
    }
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Vĩnh Phúc, năm 2018]%[Nguyễn Hiếu, Dự án EX-9G-18]%[9D0K4]
Tìm số tự nhiên $\overline{abcd}$ sao cho $\overline{abcd}+\overline{abc}+\overline{ab}+a=4321$.
\loigiai{
Ta có $\overline{abcd}+\overline{abc}+\overline{ab}+a=4321 \Rightarrow 1111a+111b+11c+d=4321$.\\
Vì $111b+11c+d \leq 999+99+9=1107$ nên $4321\leq 1107 +1111a \Rightarrow a>2$
$\Rightarrow a\geq 3$.\\
Mặt khác $1111a\leq 4321 \Rightarrow a<4 \Rightarrow a\leq 3$. Do đó $a=3$, thay vào $\overline{abcd}+\overline{abc}+\overline{ab}+a=4321$ được $111b+11c+d=988$.\\
Ta lại có $11c+d\leq 99+9=108$ $\Rightarrow 988\leq 111b+108 \Rightarrow b>7 \Rightarrow b\geq 8$ và $111b\leq 988 \Rightarrow b\leq 8$.\\
Suy ra $b=8$. Từ đó ta có $11c+d=100.$ Lập luận tương tự như trên, ta tìm được $c=9$, $d=1$.\\
Vậy số tự nhiên cần tìm là $\overline{abcd}=3891$.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Vĩnh Phúc, năm 2018]%[Nguyễn Hiếu, Dự án EX-9G-18]%[9D0B2]
Cho hệ phương trình $\heva{&(m-1)x+y=2\\ &x+2y=2}$ (với $m$ là tham số). Tìm tất cả các giá trị nguyên của $m$ để hệ phương trình có nghiệm $\left(x,y\right)$ trong đó $x,y$ là các số nguyên.
\loigiai{
Từ phương trình $x+2y=2 \Rightarrow x=2-2y$.\\
Thay vào phương trình $\left(m-1\right)x+y=2$ ta được $y\left(3-2m\right)=4-2m$.\\
Vì $m$ là số nguyên nên $3-2m\neq 0$. Do đó $y=\dfrac{4-2m}{3-2m}=1+\dfrac{1}{3-2m}$ là số nguyên khi $3-2m $ là ước của $1$ $\Rightarrow 3-2m \in \left \{ 1;-1 \right \}\Rightarrow  m\in \left \{ 1;2 \right \}$.\\
Lần lượt thay $m=1$, $m=2$ vào hệ phương trình ta thấy thoả mãn.\\
Vậy $m=1$, $m=2$.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Vĩnh Phúc, năm 2018]%[Nguyễn Hiếu, Dự án EX-9G-18]%[9D0K2]

\loigiai{
Điều kiện $-4\leq x\leq 1$. Khi đó phương trình đã cho tương đương với
\begin{eqnarray*}
&&5+2\sqrt{\left(1-x\right)\left(4+x\right)}=9\\
&\Leftrightarrow &\sqrt{4-3x-x^2}=2 \\
&\Leftrightarrow & x\left(x+3\right)=0\\
& \Leftrightarrow & \hoac{&x = 0\\&x =  - 3.}
\end{eqnarray*}
Đối chiếu điều kiện ta có tập nghiệm của phương trình là $S=\left\{-3;0\right\}$.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Vĩnh Phúc, năm 2018]%[Nguyễn Hiếu, Dự án EX-9G-18]%[9D0G7]
Cho $a,b,c>0$ thoả mãn $\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c} \leq 2$. Chứng minh rằng
$$\dfrac{1}{\sqrt{5a^2+2ab+2b^2}}+\dfrac{1}{\sqrt{5b^2+2bc+2c^2}}+\dfrac{1}{\sqrt{5c^2+2ca+2a^2}}\leq \dfrac{2}{3}.$$
\loigiai{
Ta có $\sqrt{5a^2+2ab+2b^2}=\sqrt{\left(2a+b\right)^2+\left(a-b\right)^2}\geq \sqrt{\left(2a+b\right)^2}=2a+b$\\
$\Rightarrow \dfrac{1}{\sqrt{5a^2+2ab+2b^2}}\leq \dfrac{1}{2a+b}.$\\
Áp dụng BĐT Cauchy ta có $\left(a+b+c\right)\left(\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}\right) \geq 3\sqrt[3]{a^2b}.3\sqrt[3]{\dfrac{1}{a^2b}}=9$ \\
$\Rightarrow \dfrac{1}{2a+b}\leq \dfrac{1}{9}\left(\dfrac{1}{a}+\dfrac{1}{a}+\dfrac{1}{b}\right)$
$\Rightarrow \dfrac{1}{\sqrt{5a^2+2ab+2b^2}}\leq \dfrac{1}{9}\left(\dfrac{1}{a}+\dfrac{1}{a}+\dfrac{1}{b}\right)$.\\
Chứng minh tương tự ta cũng có
$$\dfrac{1}{\sqrt{5b^2+2bc+2c^2}}\leq \dfrac{1}{9}\left(\dfrac{1}{b}+\dfrac{1}{b}+\dfrac{1}{c}\right)$$
$$ \dfrac{1}{\sqrt{5c^2+2ca+2a^2}}\leq \dfrac{1}{9}\left(\dfrac{1}{c}+\dfrac{1}{c}+\dfrac{1}{a}\right).$$
Cộng các BĐT này lại ta được
$$\dfrac{1}{\sqrt{5a^2+2ab+2b^2}}+\dfrac{1}{\sqrt{5b^2+2bc+2c^2}}+\dfrac{1}{\sqrt{5c^2+2ca+2a^2}}\leq \dfrac{1}{3}\left(\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}\right)\leq \dfrac{2}{3}.$$
Dấu $``="$ xảy ra khi và chỉ khi $a=b=c=\dfrac{3}{2}$.
}
\end{ex}
