\begin{ex}%[Đề thi HSG 9, tỉnh Gia Lai, năm 2018]%[Hoàng Mạnh Linh, Dự án EX-9G-18]%[0D0G1]
	Cho $A=\dfrac{1}{2^2}+\dfrac{1}{3^2}+\dfrac{1}{4^2}+\cdot \cdot \cdot +\dfrac{1}{{2018}^2}$. So sánh $A$ với $\dfrac{2017}{2018} $.
		\loigiai{
		Ta có: 
		{\allowdisplaybreaks
			\begin{align*}
			A&=\dfrac{1}{2^2}+\dfrac{1}{3^2}+\dfrac{1}{4^2}+\cdot \cdot \cdot +\dfrac{1}{{2018}^2}<\dfrac{1}{1.2}+\dfrac{1}{2.3}+\dfrac{1}{3.4}+\cdot \cdot \cdot +\dfrac{1}{2017.2018}\\
			&=\dfrac{1}{1}-\dfrac{1}{2}+\dfrac{1}{2}-\dfrac{1}{3}+\dfrac{1}{3}-\dfrac{1}{4}+\cdot \cdot \cdot +\dfrac{1}{2017}-\dfrac{1}{2018}\\
			&=1-\dfrac{1}{2018}=\dfrac{2017}{2018}.
			\end{align*}}Vậy A < $\dfrac{2017}{2018}$.
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Gia Lai, năm 2018]%[Hoàng Mạnh Linh, Dự án EX-9G-18]%[0D0G2]
    Giải hệ phương trình 
    $\left\{\begin{aligned}& x^2+y^2+4y=1 \\
    & 3x+xy+y=3.
    \end{aligned}\right.$
    \loigiai{
    	Nhân hai vế của phương trình thứ hai trong hệ cho $2$ rồi cộng vế theo vế với phương trình thứ nhất trong hệ ta được 
    	{\allowdisplaybreaks
    		\begin{align*}
    		&x^2+y^2+4y+6x+2xy+2y=7
    		\Leftrightarrow \left({x+y}\right)^2+6\left({x+y}\right)-7=0
    		\Leftrightarrow \hoac{&x+y=1\\&x+y=-7.} 
    		\end{align*}}
    	\begin{itemize}
    		\item Ta xét trường hợp $x+y=1\Leftrightarrow y=1-x $. 
    		Khi đó ta có phương trình tìm $x$ sau:
    		{\allowdisplaybreaks
    			\begin{align*}
    			3x+x\left({1-x}\right)+1-x=3 \Leftrightarrow x^2-3x+2=0 \Leftrightarrow
    			\hoac{&x=1\\&x=2.}
    			\end{align*}}Với $x=1$ thì $y=0$; với $x=2$ thì $y=-1 $.
    		\item  Ta xét trường hợp $x+y=-7\Leftrightarrow y=-7-x$ 
    		Khi đó phương trình xác định $x$ là:
    		$$3x+x\left({-7-x}\right)-7-x=3 \Leftrightarrow x^2+5x+10=0.$$
    		Phương trình xác định $x$  vô nghiệm nên trong trường hợp này hệ vô nghiệm.
    	\end{itemize}
    	Vậy hệ có hai nghiệm là: $\left( x; y \right) = \left( 1;0\right),\,\,\left( x; y \right) = \left( 2; - 1 \right)$.
    }
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Gia Lai, năm 2018]%[Hoàng Mạnh Linh, Dự án EX-9G-18]%[0D0G5]%[0D0G4]
	\begin{enumerate}
		\item [$1)$] Tìm các cặp số nguyên $\left(x;y\right)$ thỏa mãn $x^4+2x^3+3x^2+2x-y^2-y=0 $.
		\item [$2)$] Có bao nhiêu số nguyên dương có 6 chữ số dạng $\overline{{abcdef}}$ sao cho $$100\left({a-d}\right)+10\left({b-e}\right)+\left({c-f}\right)$$
		chia hết cho $1001$?
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item [$1)$] Ta có: 
		{\allowdisplaybreaks
			\begin{align*}
			x^4+2x^3+3x^2+2x-y^2-y=0 &\Leftrightarrow   \left(x^2+x\right)^2+2\left(x^2+x\right)=y^2+y\\
			&\Leftrightarrow \left(x^2+x+1\right)^2=y^2+y+1.\tag{*}
			\end{align*}}\begin{itemize}
			\item Nếu $y>0$ thì $y^2<y^2+y+1<(y+1)^2$, suy ra $y^2+y+1$ không là số chính phương nên không tồn tại số nguyên $x$, $y$ thỏa mãn $(*)$.
			\item Nếu $y<-1$ thì $(y+1)^2<y^2+y+1<y^2$, suy ra $y^2+y+1$ không là số chính phương nên không tồn tại số nguyên $x$, $y$ thỏa mãn $(*)$.
			\item Nếu $y=0$ hoặc $y=-1$ thì từ $(*)$ suy ra: 
			$x^2+x+1=\pm 1\Leftrightarrow \left[\begin{aligned}& x=0 \\
			& x=-1.
			\end{aligned}\right.$
		\end{itemize}
		Vậy các cặp số nguyên $(x; y)$  thỏa mãn yêu cầu bài toán là: $\left(0;0\right);\left(0;-1\right);\left(-1;0\right);\left(-1;-1\right)$.
		\item [$2)$] 
		Ta có 
		{\allowdisplaybreaks
			\begin{align*}
			\overline{abcdef}&=\overline{{abc}}.1000+\overline{{def}}\\&=\overline{{abc}}.1001-\overline{{abc}}+\overline{{def}}\\&=\overline{{abc}}.1001-\left({100\left({a-d}\right)+10\left({b-e}\right)+\left({c-f}\right)}\right)
			\end{align*}}nên $100\left({a-d}\right)+10\left({b-e}\right)+\left({c-f}\right)$ chia hết cho $1001$ khi và chỉ khi $\overline{{abcdef}}$ chia hết cho $1001$. 
		Do đó, số các số cần tìm chính là số các số nguyên dương có $6$ chữ số chia hết cho $1001$.
		Số nguyên dương bé nhất có $6$ chữ số chia hết cho $1001$ là $100100$. Số nguyên dương lớn nhất có $6$ chữ số chia hết cho $1001$ là $999999$. Vậy số các số nguyên dương có $6$ chữ số chia hết cho $1001$ là $\dfrac{999999-100100}{1001}+1=900 $. 
	\end{enumerate}
}	
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Gia Lai, năm 2018]%[Hoàng Mạnh Linh, Dự án EX-9G-18]%[0H0G1]
		Cho $\triangle ABC$ nội tiếp đường tròn $(O)$. $P$ di chuyển trên $\wideparen{BC}$ chứa $A$ của $(O)$. $I$ là tâm đường tròn nội tiếp $\triangle ABC$. Gọi  $Q$ là tâm đường tròn nội tiếp $\triangle PBC$.
	\begin{enumerate}
		\item [a)] Biểu diễn số đo góc $BIC$ theo số đo của góc $BAC. $
		\item [b)] Chứng minh rằng $B;I;Q;C$ cùng nằm trên một đường tròn.
		\item [c)] Trên tia $BQ;CQ$ lần lượt lấy các điểm $M;N$ sao cho $BM=BI$; $CN=CI$. Chứng minh rằng $MN$ luôn đi qua một điểm cố định khi $P$ di chuyển trên $\wideparen{BC}$ chứa $A$ của $(O) $.
	\end{enumerate}	
\loigiai{
	\begin{center}
		\begin{tikzpicture}[scale = 0.85]
		\tkzDefPoint(-3,0){B}\tkzDefPoint(6,0){C}
		\tkzDefPoint(3,6){A}
		\tkzDrawPolygon(A,B,C)
		\tkzDefCircle[circum](A,B,C) \tkzGetPoint{O}
		\tkzDefCircle[in](A,B,C) \tkzGetPoint{I}
		\tkzDrawCircle[circum](A,B,C)
		\tkzDrawCircle[in](A,B,C)
		\tkzDrawPoints(A,B,C,O,I)
		\tkzLabelPoint[above](A){\footnotesize $A$}
		\tkzLabelPoint[left](B){\footnotesize $B$}
		\tkzLabelPoint[right](C){\footnotesize $C$}
		\tkzLabelPoint[left](O){\footnotesize $O$}
		\tkzLabelPoint[above](I){\footnotesize $I$}
		\tkzDefPointBy[rotation= center O angle 60](A)
		\tkzGetPoint{P}
		\tkzDrawPoints(P)
		\tkzLabelPoints[above](P)
		\tkzDefCircle[in](P,B,C) \tkzGetPoint{Q}\tkzDrawPoints(Q)
		\tkzLabelPoints[above](Q)
		\tkzDefCircle[circum](I,B,C) \tkzGetPoint{T}
		\tkzDrawCircle[circum](I,B,C)
		\tkzDrawCircle(B,I)
		\tkzDrawCircle(C,I)
		\tkzInterLC(B,Q)(B,I) 
		\tkzGetPoints{M'}{M}
		\tkzInterLC(C,Q)(C,I) 
		\tkzGetPoints{N'}{N}
		\tkzDrawPoints[size=12](N,M)
		\tkzLabelPoints[above](M)
		\tkzLabelPoints[below left](N)
		\tkzInterCC(B,I)(C,I)
		\tkzGetPoints{I}{K}
		\tkzDrawPoints[size=12](I,K)
		\tkzLabelPoints[right](K)
		\tkzDrawSegments(P,B P,C I,B I,C Q,B Q,C Q,M M,K I,K Q,I)
		\tkzMarkAngle(I,B,Q)
		\tkzMarkAngle(I,C,Q)
		\end{tikzpicture}
	\end{center}
	\begin{enumerate}
		\item [a)] Ta có 
		$$\widehat{BIC}=180^\circ-\widehat{IBC} - \widehat{ICB}
		=180^\circ-\dfrac{\widehat{ABC}}{2}-\dfrac{\widehat{ACB}}{2}=180^\circ-\dfrac{{180}^\circ-\widehat{BAC}}{2}=90^\circ+\dfrac{\widehat{BAC}}{2}.$$
		\item [b)]
		Theo câu a), ta có $\widehat{BIC} = 90^\circ + \dfrac{\widehat{BAC}}{2}$.\\
		Tương tự $\widehat{BQC}=90^\circ+\dfrac{\widehat{BPC}}{2}$.
		Hơn nữa 
	\begin{align*}
	\widehat{BAC}=\widehat{BPC}\hspace{0.1cm} \text{(hai góc nội tiếp chắn cùng một cung của đường tròn $(O) $)}
	\end{align*} 
		suy ra 
		\begin{align*}
		\widehat{BQC}=\widehat{BIC}
		\end{align*}
		Vì hai điểm $Q$ và $I$ nằm cùng nữa mặt phẳng bờ $BC$ và cùng nhìn đoạn $BC$ dưới hai góc bằng nhau nên 4 điểm $B;I;Q;C$ cùng thuộc một đường tròn.
		\item [c)] Vì $ \heva{IB + IC > BC\\ \left| {IB - IC} \right| < BC}$ (bất đẳng thức trong tam giác) nên đường tròn $\left({B;BI}\right)$ cắt đường tròn $\left({C;CI}\right)$ tại hai điểm phân biệt. Gọi $K$ là giao điểm thứ hai của hai đường tròn $\left({B;BI}\right)$ và $\left({C;CI}\right)$ ($K$ cố định).\\
		Góc $\widehat{{IBM}}$ là góc ở tâm chắn cung $\wideparen{IM}$ và $\widehat{IKM}$ là góc nội tiếp chắn cung $\wideparen{IM}$ của đường tròn $\left({B;BI}\right)$, suy ra
		\begin{align}\label{1810.1}
		\widehat{IKM}=\dfrac{1}{2}\widehat{IBM}
		\end{align}
		Góc $\widehat{ICN}$ là góc ở tâm chắn cung $\wideparen{IN}$ và $\widehat{IKN}$ là góc nội tiếp chắn cung $\wideparen{IN}$ của đường tròn $\left({C;CI}\right)$, suy ra
		\begin{align}\label{1810.2}
		\widehat{IKN} = \dfrac{1}{2}\widehat{ICN}
		\end{align}
		Theo câu b), bốn điểm  $B;I;Q;C$ thuộc một đường tròn, suy ra 
		$\widehat{QBI} = \widehat{QCI}$ (hai góc nội tiếp cùng chắn cung $\wideparen{QI}$ của đường tròn đi qua bốn điểm  $Q, I, C, B$) hay 
		\begin{align}\label{1810.3}
		\widehat{IBM}=\widehat{ICN}
		\end{align}
		Từ \eqref{1810.1}, \eqref{1810.2} và \eqref{1810.3}, suy ra $\widehat{IKM}=\widehat{IKN}$.
		Ta có $\widehat{IKM}=\widehat{IKN}$ và hai tia $KM;KN$ cùng thuộc một nửa mặt phẳng bờ $IK$ nên $KM \equiv KN$.
		Vậy $MN$ đi qua $K$ cố định.
		
	\end{enumerate}	
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Gia Lai, năm 2018]%[Hoàng Mạnh Linh, Dự án EX-9G-18]%[0D0G8]
	Xét một thanh gỗ ngang có hai đầu. Một con kiến đi từ đầu này đến đầu kia của thanh thì mất $5$ phút. Khi đi đến một trong hai đầu thì kiến sẽ rơi xuống đất. Bây giờ giả sử trên thanh gỗ đó có $5$ con kiến và đi cùng với tốc độ như vậy nhưng về các hướng khác nhau. Nếu có hai con kiến nào đi ngược hướng và đụng đầu nhau thì chúng lập tức quay ngược lại và đi tiếp (giả sử rằng kích thước cũng như thời gian quay đầu của các con kiến không đáng kể).
	\begin{enumerate}
		\item [$a)$] Hãy lý luận để chứng tỏ rằng tất cả các con kiến thể nào cũng sẽ rơi hết xuống đất.
		\item [$b)$] Cần ít nhất bao nhiêu thời gian để chắc chắn rằng cả $5$ con kiến đều rơi xuống đất?
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item [$a)$]  Xét con kiến ở ngoài cùng bên phải, nếu có nhiều con như vậy thì chọn một con bất kỳ. 
		\begin{itemize}
			\item Nếu con kiến này đi về phía phải thì nó sẽ không đụng đầu bất cứ con kiến nào, cứ thế nó đi và sẽ bị rơi xuống đất.
			\item Xét trường hợp con kiến này đi về phía trái. Nếu nó đụng đầu với con kiến đi ngược chiều thì nó sẽ quay về bên phải để đi tiếp, tất nhiên nó sẽ không đụng đầu với con kiến nào nữa, cứ thế nó đi và sẽ rơi xuống đất. Trong trường hợp ngược lại, nó cứ đi về bên trái và rơi xuống đất. 
			\item Tiếp tục lý luận như vậy cho các con kiến còn lại trên thanh gỗ, rõ ràng đến một lúc nào đó thì cả thảy 5 con kiến trên thanh ngang đều rơi xuống đất.
		\end{itemize}
		\item [$b)$]  Cần ít nhất bao nhiêu thời gian để chắc chắn rằng cả $5$ con kiến đều rơi xuống đất?
		\begin{itemize}
			\item Trong trường hợp “xấu nhất”, nếu có một con kiến đi từ đầu này đến đầu kia của thanh gỗ mà không đụng đầu phải con kiến nào đi ngược chiều thì phải mất 5 phút con kiến này mới rơi xuống đất. Như vậy khoảng thời gian ít nhất để chắc chắn rằng không còn con kiến nào trên thanh ngang sẽ không nhỏ hơn 5 phút.
			\item Ta hãy quan sát các con kiến “từ xa”, không phân biệt các con kiến với nhau, thì chúng quay đầu đi ngược lại cũng không khác gì chúng cứ tiếp tục đi: nếu chẳng hạn con kiến A đụng đầu con kiến B rồi đi ngược lại, thì cũng giống như chúng cứ tiếp tục đi nhưng ta “đổi tên” hai con kiến này cho nhau thôi. Nếu chỉ nhìn tổng thể chứ không phân biệt các con kiến, thì việc có quay đầu lại hay không vẫn thế. Và tất nhiên nếu không cần quay đầu lại, thì thời gian tối đa cần thiết để chúng rơi xuống hết là $5$ phút.
		\end{itemize}
		Vậy khoảng thời gian cần tìm là $5$ phút.
	\end{enumerate}
}
\end{ex}
