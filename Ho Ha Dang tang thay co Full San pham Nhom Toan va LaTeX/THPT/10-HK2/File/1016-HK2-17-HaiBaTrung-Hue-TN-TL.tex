\begin{ex}%[HK2 (2016-2017), THPT Hai Bà Trưng, Huế]%[0D6K3]%[Nguyễn Văn Vũ]
Nếu $\sin (\alpha-\beta)=\dfrac{1}{3}\sin \beta$ thì $\tan (\alpha-\beta)$ bằng
		\choice
		{$\dfrac{\cos \alpha}{\sin\alpha+2}$}
		{\True $\dfrac{\sin \alpha}{\cos \alpha+3}$}
		{$\dfrac{\sin \alpha}{\cos \alpha+2}$}
		{$\dfrac{\cos \alpha}{\sin\alpha+3}$}
		\loigiai{
Ta có $\sin (\alpha-\beta)=\dfrac{1}{3}\sin \beta\Leftrightarrow \sin\alpha\cos\beta-\cos\alpha\sin\beta=\dfrac{1}{3}\sin \beta
\Leftrightarrow \cos\beta\sin\alpha=\sin\beta\left(\cos\alpha+\dfrac{1}{3}\right)\quad (*)$.\\
$\bullet$ Trường hợp $\sin\alpha\neq 0, \sin\beta\neq 0$.\\
Ta có $(*)\Leftrightarrow\dfrac{\cos\beta}{\sin\beta}=\dfrac{\cos\alpha+\dfrac{1}{3}}{\sin\alpha}$.\\
Khi đó
$$\begin{aligned}\tan (\alpha-\beta)&=\dfrac{\sin (\alpha-\beta)}{\cos (\alpha-\beta)}
=\dfrac{\dfrac{1}{3}\sin \beta}{\cos \alpha\cos \beta+\sin \alpha\sin \beta}=\dfrac{\dfrac{1}{3}}{\cos\alpha\dfrac{\cos\beta}{\sin\beta}+\sin \alpha }\\
&=\dfrac{\dfrac{1}{3}}{\cos\alpha\dfrac{\cos\alpha+\dfrac{1}{3}}{\sin\alpha}+\sin \alpha}
=\dfrac{\sin \alpha}{3\cos^2\alpha+\cos\alpha+3\sin^2\alpha}=\dfrac{\sin \alpha}{\cos \alpha+3}.
\end{aligned}$$
$\bullet$ Trường hợp $\sin\alpha=0\Rightarrow \cos\alpha=\pm 1$ và kết hợp với $(*)\Rightarrow\sin\beta=0\Rightarrow\cos\beta=\pm1$.
Khi đó
$$\tan (\alpha-\beta)=\dfrac{\tan\alpha-\tan\beta}{1+\tan\alpha\tan\beta}=0=\dfrac{\sin \alpha}{\cos \alpha+3}.$$
Tương tự, nếu $\sin\beta=0$ ta thu được $\tan (\alpha-\beta)=0=\dfrac{\sin \alpha}{\cos \alpha+3}$.
	}
\end{ex}

\begin{bt}%[HK2 (2016-2017), THPT Hai Bà Trưng, Huế]%[0D6B2]%[Nguyện Ngô]
Với điều kiện $x\neq k\pi$, chứng minh đẳng thức sau: $\sin x\left(\dfrac{\sin x}{1-\cos x}+\dfrac{1-\cos x}{\sin x} \right) =2$.
\loigiai
{Ta có
$$\begin{aligned}\sin x\left(\dfrac{\sin x}{1-\cos x}+\dfrac{1-\cos x}{\sin x} \right) &=\sin x\left(\dfrac{\sin^2x+1-2\cos x+\cos^2x}{(1-\cos x)\sin x} \right)\\
&=\sin x\cdot\dfrac{2-2\cos x}{(1-\cos x)\sin x}=\sin x\cdot\dfrac{2}{\sin x}=2.
\end{aligned}$$	
}
\end{bt}
