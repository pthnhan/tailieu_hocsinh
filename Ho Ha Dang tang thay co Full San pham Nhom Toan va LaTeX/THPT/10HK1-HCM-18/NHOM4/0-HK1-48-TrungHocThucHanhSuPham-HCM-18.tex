		\end{align*}
		Phương trình có nghiệm duy nhất khi và chỉ khi $-(m+4) \neq 0 \Leftrightarrow m \neq -4$.
	}

	Giải phương trình		
		 $\dfrac{3x^2+2x-1}{(2-x )(3x-1 )}=1$.		
\loigiai{ Điều kiện: $x \neq 2$ và $x\neq \dfrac{1}{3}$. Khi đó
		$$\begin{aligned}
		&\dfrac{3x^2+2x-1}{(2-x )(3x-1 )}=1\\
		\Leftrightarrow &\dfrac{(3x^2+2x-1) - (2-x)(3x-1)}{(2-x)(3x-1)}=0\\
		\Leftrightarrow &(3x^2+2x-1) - (2-x)(3x-1)=0\\
		\Leftrightarrow &6x^2-5x+1=0\\
		\Leftrightarrow &\hoac{&x=\dfrac{1}{2}\\
		&x=\dfrac{1}{3}}.
		\end{aligned}$$
		Nghiệm $x=\dfrac{1}{3}$ không thỏa mãn điều kiện nên phương trình có một nghiệm là $x=\dfrac{1}{2}$.
		}

\begin{bt}%[0-HK1-48-Trung Học Thực Hành Sư Phạm-HCM-18]%[Lê Ngọc Hiếu, HKI]%[0D3K2-4]
	Giải phương trình sau.
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\sqrt{8x^2-6x+1}-4x+1=0$.
			\item $x^2-4x+2\sqrt{-x^2+4x+5}=2$.
			\end{enumerate}
	\end{multicols}
	\loigiai{\begin{enumerate}
			\item ĐK: $8x^2-6x+1 \geq 0$\\
			Ta có
			$$\begin{aligned}
			&\sqrt{8x^2-6x+1}-4x+1=0\\
			&\Rightarrow \sqrt{8x^2-6x+1}=4x-1\\
			&\Rightarrow 8x^2-6x+1=(4x-1)^2\\
			&\Rightarrow 8x^2-6x+1=16x^2-8x+1\\
			&\Rightarrow 8x^2-2x=0\\
			&\Rightarrow \hoac{&x=0\\&x=\dfrac{1}{4}.}
			\end{aligned}$$
			Thử lại ta được phương trình có nghiệm $ x=\dfrac{1}{4}$.
			\item Đặt $t=\sqrt{-x^2+4x+5}$, $t \ge 0$. Khi đó phương trình trở thành
			\begin{align*}
			-t^2+2t+3=0
			\Leftrightarrow \hoac{&t=-1 \text{ (loại)}\\&t=3 \text{ (nhận)}}.
			\end{align*}\\
	Với $t=3 \Leftrightarrow -x^2+4x+5=9	\Leftrightarrow x=2$.
		\end{enumerate}}
\end{bt}

\begin{bt}%[0-HK1-48-Trung Học Thực Hành Sư Phạm-HCM-18]%[Lê Ngọc Hiếu, HKI]%[0D3K2-5]
	Cho phương trình $x^2+2(m-1)x+m^2=0$ ($m$ là tham số). Tìm $m$ để phương trình có $2$ nghiệm phân biệt $x_1,x_2$ sao cho $x_{1}^{2}+x_{2}^{2}-x_1-x_2=2$.
	\loigiai{Ta có $\Delta' = (m-1)^2-m^2=-2m+1$.\\
	Phương trình có hai nghiệm phân biệt khi và chỉ khi $\Delta'>0 \Leftrightarrow m<\dfrac{1}{2}$.\\
	Gọi $x_1, x_2$ là hai nghiệm của phương trình. Theo định lý Vi-et ta có:
	$$\heva{x_1+x_2&=-2(m-1)\\ x_1 x_2&=m^2.}$$
	Khi đó ta được
	$$\begin{aligned}
		&x_{1}^{2}+x_{2}^{2}-x_1-x_2=2\Leftrightarrow (x_1+x_2)^2-2x_1 x_2-(x_1+x_2)=2\\
		&\Leftrightarrow 4(m-1)^2-2m^2+2(m-1)=2	\Leftrightarrow 2m^2-6m=0
		\Leftrightarrow \hoac{&m=0\\&m=3.}
	\end{aligned}$$
	So điều kiện $\Delta'>0$ ta được $m=0$ thỏa ycbt.
	}
\end{bt}
