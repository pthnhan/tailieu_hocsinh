\begin{ex}%[Thi vào 10, tỉnh Bình Thuận, 2016]%[9D3B3]%[9D3B4]%[9D4B4]
Giải phương trình và hệ phương trình sau
\begin{enumerate}
\item $x^2+5x+6=0$.
\item $\heva{&x+7y=2\\&2x-y=4.}$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $\Delta=1$. Phương trình có hai nghiệm phân biệt $x_1=-2, x_2=-3.$
\item Ta có hệ phương trình tương đương với $\heva{&3x=6\\&2x-y=4}\Leftrightarrow \heva{x&=2\\y&=0.}$\\
Vậy hệ phương trình có nghiệm duy nhất $(x;y)=(2;0)$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Thi vào 10, tỉnh Bình Thuận, 2016]%[9D1B8]%[9D1K8]
\begin{enumerate}
\item[]
\item Tính giá trị của biểu thức: $A=\left(\sqrt{28}-2\sqrt{7}+\sqrt{7}\right).\sqrt{7}$.
\item Rút gọn biểu thức $B=\dfrac{a\sqrt{b}-b\sqrt{a}}{\sqrt{ab}}:\dfrac{1}{\sqrt{a}+\sqrt{b}}$, với $a>0; b>0$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có:\\
$
\begin{aligned}
A=\left(\sqrt{28}-2\sqrt{7}+\sqrt{7}\right).\sqrt{7}=\left(2\sqrt{7}-2\sqrt{7}+\sqrt{7}\right).\sqrt{7}=\sqrt{7}.\sqrt{7}=7.
\end{aligned}
$
\item Ta có:\\
$
\begin{aligned}
B&=\dfrac{a\sqrt{b}-b\sqrt{a}}{\sqrt{ab}}:\dfrac{1}{\sqrt{a}+\sqrt{b}}\\
&=\dfrac{\sqrt{ab}(\sqrt{a}-\sqrt{b})}{\sqrt{ab}}:\dfrac{1}{\sqrt{a}+\sqrt{b}}\\
&=(\sqrt{a}-\sqrt{b}).(\sqrt{a}+\sqrt{b})\\
&=(\sqrt{a})^2-(\sqrt{b})^2=a-b.
\end{aligned}
$
\end{enumerate}
}
\end{ex}

\begin{ex}%[Thi vào 10, tỉnh Bình Thuận, 2016]%[9D4B2]%[9D4K5]
\begin{enumerate}
\item[]
\item Cho hàm số $y=x^2$. Điền các giá trị $y$ tương ứng vào bảng sau:
\begin{center}
\begin{tabular}{|c|c|c|c|c|c|}
\hline
$x$&$-2$&$-1$&$0$&$1$&$2$\\
\hline
$y=x^2$&&&&&\\
\hline
\end{tabular}
\end{center}

\item Tìm tham số $m$ để đường thẳng $(d):y=2x-m$ cắt parabol $(P): y=x^2$ tại hai điểm phân biệt.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có bảng giá trị như sau
\begin{center}
\begin{tabular}{|c|c|c|c|c|c|}
\hline
$x$&$-2$&$-1$&$0$&$1$&$2$\\
\hline
$y=x^2$&$4$&$1$&$0$&$1$&$4$\\
\hline
\end{tabular}
\end{center}
\item Phương trình hoành độ giao điểm của $(P)$ và $(d)$ là:\\
$
\begin{aligned}
&x^2=2x-m
\Leftrightarrow &x^2-2x+m=0. (1)\\
\Delta'=1-m.\\
\end{aligned}
$\\
$(P)$ và $d$ cắt nhau tại hai điểm phân biệt\\
$\Leftrightarrow$ Phương trình $(1)$ có hai nghiệm phân biệt.\\
$\Leftrightarrow \Delta'>0 \Leftrightarrow 1-m>0 \Leftrightarrow m<1$. 
\end{enumerate}
}
\end{ex}
\begin{ex}%[Thi vào 10, tỉnh Bình Thuận, 2016]%[9H3B4]%[9H3K3]
Cho tam giác nhọn $ABC$ nội tiếp đường tròn tâm $O$ bán kính $R$. Hai tiếp tuyến tại $B$ và $C$ của đường tròn $O$ cắt nhau tại $M$.
\begin{enumerate}
\item Chứng minh tứ giác $OBMC$ nội tiếp.
\item Đường thẳng $MA$ cắt đường tròn $O$ tại điểm thứ hai là $D$.\\
Chứng minh $\triangle MAB \backsim \triangle MBD$.
\item Đường thẳng $BO$ cắt đường tròn $O$ tại $E$. Chứng minh: $MO$ song song với $EC$.
\item Giả sử $\widehat{BAC}=60^{\circ}$. Tính theo $R$ thể tích của hình sinh ra khi quay $\triangle BMC$ một vòng quanh cạnh $BC$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Xét tứ giác $OMBC$, ta có
$\widehat{OBM}=90^{\circ}$ $(BM$ là tiếp tuyến).\\
$\widehat{OCM}=90^{\circ}$ $(CM$ là tiếp tuyến).\\
$\Rightarrow \widehat{OBM}+\widehat{OCM}=90^{\circ}+90^{\circ}=180^{\circ}$.\\
$\Rightarrow $Tứ giác $OBMC$ nội tiếp đường tròn đường kính $OM$.
\item Xét $\triangle MAB$ và $\triangle MBD$.\\
Ta có: $\widehat{BMA}$ chung.\\
$\widehat{MAB}=\widehat{MBD}=\dfrac{1}{2}\text{sđ} \wideparen{BD} (\text{nhỏ})$.\\
$\Rightarrow \triangle MAB \backsim \triangle MBD (g.g)$.
\begin{center}
\begin{tikzpicture}[rotate=90]
\tkzInit[xmin=-0.5,ymin=-6.5,xmax=8,ymax=7]\tkzClip
\tkzDefPoints{1/1/B}

\tkzGetPoint{O_1}

\tkzInterLC(B,O)(O,A)

\tkzMarkRightAngles[size=.2](O,B,M O,C,M C,H,M)
\tkzLabelPoint[above](O_1){$O'$}
\end{tikzpicture}
\end{center}
\item Ta có $OB=OC=R$ và $MB=MC$ (tính chất $2$ tiếp tuyến cắt nhau).\\
$\Rightarrow MO$ là đường trung trực của $BC \Rightarrow MO \perp BC$ $(1)$.\\
Ta có $\widehat{BCE}=90^{\circ}$ (góc nội tiếp chắn nửa đường tròn) $\Rightarrow EC \perp BC$ $(2)$.\\
Từ $(1)$ và $(2)$ $\Rightarrow MO \parallel EC$.
\item Khi $\widehat{BAC}=60^{\circ}$ thì $\triangle BMC$ là tam giác đều (vì $MB=MC$ và $\widehat{MBC}=\widehat{BAC}=60^{\circ}$).\\
Gọi $H$ là giao điểm của $BC$ và $OM$.\\
Khi quay $\triangle BMC$ quanh cạnh $BC$ thì hình sinh ra là hai hình nón bằng nhau có chung mặt đáy bán kính là $HM$, đường cao là $BH$.\\
Ta có $OM$ là trung trực của $BC \Rightarrow OM \perp BC$ tại trung điểm $H$.\\
Lại có:\\
$\widehat{BAC}=\dfrac{1}{2}\widehat{BOC}$ (quan hệ giữa góc nội tiếp và góc ở tâm cùng chắn $1$ cung).\\
Mà $\widehat{BOM}=\dfrac{1}{2}\widehat{BOC}$ ($OM$ là phân giác của $\widehat{BOC}$)\\
$\Rightarrow \widehat{BOM}=60^{\circ}$.\\
Xét $\triangle OBH$, $BH=OB.\sin{\widehat{BOH}}=OB.\sin{60^{\circ}}=\dfrac{R\sqrt{3}}{2}$.\\
Trong $\triangle BMH$ vuông tại $H$ có:\\
$$MH=\dfrac{BH}{\tan{\widehat{BMH}}}=\dfrac{\dfrac{R\sqrt{3}}{2}}{\tan{30^{\circ}}}=\dfrac{3R}{2}.$$
Thể tích hai hình nón là:
$$V=2.\dfrac{1}{3}\pi MH^2. BH=2.\dfrac{1}{3}.\pi\left(\dfrac{3R}{2}\right)^2.\dfrac{R\sqrt{3}}{2}=\dfrac{3\sqrt{3}}{4}R^3.$$
\end{enumerate}
}
\end{ex}
