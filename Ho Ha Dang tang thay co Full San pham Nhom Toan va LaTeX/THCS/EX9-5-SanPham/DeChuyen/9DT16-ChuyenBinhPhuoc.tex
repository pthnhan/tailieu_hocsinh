\begin{ex}%[Đề thi vào 10 chuyên Bình Phước 2016]%[9D1K8]%[9D1K7]
	Cho biểu thức $P=\left(\dfrac{\sqrt{x}+2}{x+2\sqrt{x}+1}+\dfrac{2-\sqrt{x}}{x-1}\right)\cdot\dfrac{\sqrt{x}+1}{\sqrt{x}}$, với $x>0$, $x\ne 1$.
    \hfill
    \begin{enumerate}
        \item Rút gọn biểu thức $P$.
        \item Tìm giá trị của biểu thức $P$ tại $x=\sqrt{46-6\sqrt{5}}-3\left(\sqrt{5}-1\right)$.
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item Ta có:\\
        $\begin{aligned}
        P&=\left[\dfrac{\sqrt{x}+2}{(\sqrt{x}+1)^2}+\dfrac{2-\sqrt{x}}{(\sqrt{x}-1)(\sqrt{x}+1)}\cdot\dfrac{\sqrt{x}+1}{\sqrt{x}}\right]\\
        &=\dfrac{x+\sqrt{x}+2-x+\sqrt{x}+2}{(\sqrt{x}-1)(\sqrt{x}+1)^2}\cdot\dfrac{\sqrt{x}+1}{\sqrt{x}}=\dfrac{2\sqrt{x}}{(\sqrt{x}-1)(\sqrt{x}+1)^2}\cdot\dfrac{\sqrt{x}+1}{\sqrt{x}}\\
        &=\dfrac{x+\sqrt{x}+2-x+\sqrt{x}+2}{(\sqrt{x}-1)(\sqrt{x}+1)^2}\cdot\dfrac{\sqrt{x}+1}{\sqrt{x}}=\dfrac{2}{(\sqrt{x}-1)(\sqrt{x}+1)^2}=\dfrac{2}{x-1}
        \end{aligned}$
        \item Ta có:\\
         $\begin{aligned}
         x&=\sqrt{46-6\sqrt{5}}-3\left(\sqrt{5}-1\right)=\sqrt{45-2.3\sqrt{5}+1}-3\sqrt{5}+3\\
         &=\sqrt{\left(3\sqrt{5}-1\right)^2}-3\sqrt{5}+3=\left|3\sqrt{5}-1\right|-3\sqrt{5}-3=3\sqrt{5}-1-3\sqrt{5}+3=2.
         \end{aligned}$\\
         Suy ra $P=\dfrac{2}{2-1}=2$.
    \end{enumerate}
    }
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Bình Phước 2016]%[9D4G6]
	Cho phương trình $x^2-2mx+m^2-4m-3=0$ ($m$ là tham số). Tìm $m$ để phương trình có hai nghiệm $x_1$, $x_2$ sao cho biểu thức $T=x_1^2+x_2^2-x_1x_2$ đạt giá trị nhỏ nhất.
	\loigiai{Phương trình có hai nghiệm khi $\Delta \geq 0$.\\
	Do đó $(-m)^2-(m^2-4m-3)=4m+3\geq0\Leftrightarrow m\geq -\dfrac{3}{4}\quad(1)$\\
	Theo định lí Vi-ét thì $\heva{&x_1+x_2=2m\\x_1.x_2=m^2-4m+3}$.\\
	Ta có: $T=x_1^2+x_2^2-x_1x_2=(x_1+x_2)^2-3x_1x_2=4m^2-3(m^2-4m-3)=m^2+12m+9$.\\
	Xét hàm số $y=m^2+12m+9$ ta có $a=1$, $b=12$, $c=9$.\\
	Vì $a=1>0$ và $-\dfrac{b}{2a}=-\dfrac{12}{2.1}=-6$ nên hàm số đồng biến khi $m>-6$ và hàm số nghịch biến khi $m<-6$.\\
	Suy ra $m\geq-\dfrac{3}{4}$ thì $m^2+12m+9$ luôn đồng biến. Do đó $m^2+12m+9\geq\dfrac{9}{16}$.\\
	Hay giá trị nhỏ nhất của $T$ là $T=\dfrac{9}{16}$ đạt được khi $m=-\dfrac{3}{4}$.
}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Bình Phước 2016]%[9D4G7]
	Giải phương trình: $4\left(x^2+1\right)=3\sqrt{2x^2-7x+3}+14x$.
	\loigiai{Điều kiện: $2x^2-7x+3\geq0\Leftrightarrow x\leq\dfrac{1}{2}$ hoặc $x\geq 3$.\\
		Ta có $4\left(x^2+1\right)=3\sqrt{2x^2-7x+3}+14x$\\
		$\Leftrightarrow 2\left(2x^2-7x+3\right)-3\sqrt{2x^2-7x+3}-2=0$\quad(1).\\
		Đặt $t=\sqrt{2x^2-7x+3}\geq0$ khi đó $(1)\Leftrightarrow 2t^2-3t-2=0\Leftrightarrow \hoac{&t=2\\&t=-\dfrac{1}{2}}$.\\
		Với $t=-\dfrac{1}{2}/<0$ không thỏa mãn.\\
		Với $t=2\Rightarrow \sqrt{2x^2-7x+3}=2\Rightarrow 2x^2-7x-1=0\Leftrightarrow x=\dfrac{7\pm \sqrt{57}}{2}$.\\
		Đối chiếu với điều kiện ban đầu ta suy ra $x=\dfrac{7+\sqrt{57}}{4}$ thỏa mãn.
	}	
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Bình Phước 2016]%[9D5G1]
	Giải hệ phương trình: $\heva{&xy-y^2=\sqrt{3y-1}-\sqrt{x+2y-1}\quad(1)\\&x^3y-4xy^2+7xy-5x-y+2=0\quad(2)}$.
	\loigiai{Điều kiện: $\heva{&3y-1\geq0\\
		&x+2y-1\geq0}\Leftrightarrow\heva{&y\geq\dfrac{1}{3}\\
	&x+2y\geq1}$.\\
	Ta có: $xy-y^2=\sqrt{3y-1}-\sqrt{x+2y-1}$\\
	$\Leftrightarrow y(x-y)+\sqrt{x+2y-1}-\sqrt{3y-1}=0$\\
	$\Leftrightarrow y(x-y)+\dfrac{x-y}{\sqrt{x+2y-1}+\sqrt{3y-1}}=0$\\
	$\Leftrightarrow (x-y)\left(y+\dfrac{1}{\sqrt{x+2y-1}+\sqrt{3y-1}}\right)\quad(*)$\\
	Trường hợp 1: Với $y=\dfrac{1}{3}$ và $x+2y-1=0\Leftrightarrow x=\dfrac{1}{3}$ ta thấy không thỏa mãn $(2)$.\\
	Trường hợp 2: $y>\dfrac{1}{3}$ và $x\ne\dfrac{1}{3} \Rightarrow y+\dfrac{1}{\sqrt{x+2y-1}+\sqrt{3y-1}}>0$ đo đó $(*)$ có nghiệm khi $x=y$ thế vào $(2)$ ta được:\\
	$x^4-4x^3+7x^2-6x+2=0\Leftrightarrow (x-1)^2\left(x^2-2x+2\right)=0\Leftrightarrow \hoac{&x=1\\&x^2-2x+2=0}$.\\
	Với $x^2-2x+2=0$ ta có: $\Delta'=(-1)^2-2.1=-1<0$ (vô nghiệm).\\
	Với $x=1$ thì $y=1$ thỏa mãn.\\
	Vậy nghiệm của hệ phương trình là $(x;y)=(1;1)$.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Bình Phước 2016]%[9D5G2]
	Cho biểu thức $Q=a^4+2a^3-16a^2-2a+15$. Tìm tất cả các giá trị nguyên của $a$ để $Q$ chia hết cho $16$.	
	\loigiai{$Q=a^4+2a^3-16a^2-2a+15=(a^4+2a^3-2a-1)-16(a^2-1)=(a-1)(a+1)^3-16(a^2-1).$\\
		Với $a$ lẻ, $a=2k+1$, $k\in\mathbb{Z}$. Khi đó $(a-1)(a+1)^3=2k(2k+2)^3=16k(k+1)\vdots 16\Rightarrow Q\vdots16$\\
		Với $k$ chẵn, $a=2k$, $k\in\mathbb{Z}$. Khi đó $(a-1)(a+1)^3=(2k-1)(2k+1)^3$ là một số lẻ nên không chia hết cho $16$. Do đó $Q$ không chia hết cho $16$.\\
		Vậy khi $a$ là một số lẻ thì $Q$ chia hết cho 16.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Bình Phước 2016]%[9D5G1]
	Cho hai số thực $a$, $b$ đều lớn hơn 1. Chứng minh rằng:
	\[\dfrac{6}{a\sqrt{b-1}+b\sqrt{a-1}}+\sqrt{3ab+4}\geq\dfrac{11}{2}\]
	\loigiai{\textbf{Cách 1}: Áp dụng bất đẳng thức Cô-si ta có:
		\begin{equation}\label{bdt1}
		\sqrt{b-1}\leq\dfrac{b-1+1}{2}=\dfrac{b}{2}\Rightarrow a\sqrt{b-1}\leq\dfrac{ab}{2}
		\end{equation}
		Tương tự: \begin{equation}\label{bdt2}
		\sqrt{a-1}\leq\dfrac{a-1+1}{2}=\dfrac{a}{2}\Rightarrow b\sqrt{a-1}\leq\dfrac{ab}{2}
		\end{equation}
		Từ (\ref{bdt1}) và (\ref{bdt2}) ta được:
		\[\dfrac{6}{a\sqrt{b-1}+b\sqrt{a-1}}+\sqrt{3ab+4}\leq\dfrac{6}{ab}+\sqrt{3ab+4}=\dfrac{18}{3ab}+\sqrt{3ab+4}\]
		Đặt $t=\sqrt{3ab+4}\Rightarrow t^2-4=3ab\Leftrightarrow (t-2)(t+2)=3ab$. Khi đó:\\
		\[\begin{aligned}
		S&=\dfrac{18}{(t-2)(t+2)}+t=\dfrac{18}{(t-2)(t+2)}+\dfrac{3}{4}(t-2)+\dfrac{1}{4}(t+2)+1\\
		&\leq3\sqrt[3]{\dfrac{18}{(t-2)(t+2)}\cdot(t-2)\cdot(t+2)}+1=\dfrac{11}{2}
		\end{aligned}\]
		Dấu "$=$" xảy ra khi $t=4\Rightarrow a=b=2$.\\
		\textbf{Cách 2}: Ta có: 
		\begin{equation}\label{bdt3}
		a\sqrt{b-1}=\sqrt{a}.\sqrt{ab-a}\leq\dfrac{a+ab-a}{2}=\dfrac{ab}{2}
		\end{equation}
		Tương tự: 
		\begin{equation}\label{bdt4}
		b\sqrt{a-1}=\sqrt{b}.\sqrt{ab-b}\leq\dfrac{b+ab-b}{2}=\dfrac{ab}{2}
		\end{equation}
		Từ (\ref{bdt3}) và (\ref{bdt4}) suy ra $\dfrac{6}{a\sqrt{b-1}+b\sqrt{a-1}}\geq\dfrac{6}{ab}$\\
		Lại có: $\sqrt{3ab+4}=\dfrac{3ab+4}{\sqrt{3ab+4}}=\dfrac{4(3ab+4)}{4\sqrt{3ab+4}}\geq\dfrac{8(3ab+4)}{20+3ab}$.\\
		Ta chứng minh $\dfrac{6}{a\sqrt{b-1}+b\sqrt{a-1}}+\sqrt{3ab+4}-\dfrac{11}{2}\geq0$\\
		Thật vậy ta có: $\dfrac{6}{a\sqrt{b-1}+b\sqrt{a-1}}+\sqrt{3ab+4}-\dfrac{11}{2}\geq\dfrac{6}{ab}+\dfrac{8(3ab+4)}{20+3ab}-\dfrac{11}{2}$\\
		$\dfrac{6}{ab}+\dfrac{8(3ab+4)}{20+3ab}-\dfrac{11}{2}=\dfrac{12(20+3ab)+8(3ab+4)-11ab(20+3ab)}{2ab(20+3ab)}=\dfrac{15ab(ab-4)^2}{2ab(20+3ab)}\geq0$
		luôn đúng với mọi $a,b>1$. (đpcm)
	}

\end{ex}
