\loigiai{
Gọi $I$ là trung điểm của $AB$, khi đó $I=\left( \dfrac{a+1}{2}; \dfrac{b-1}{2} \right)$ và $\overrightarrow{AB}=(a-1;b+1)$. Gọi $\overrightarrow{u}(3;2)$ là véc-tơ chỉ phương của $d$. Điểm $B$ đối xứng với điểm $A$ qua đường thẳng $d$ khi và chỉ khi
\begin{align*}
\heva{&I\in d\\ &AB\perp d}\Leftrightarrow \heva{&I\in d\\ &\overrightarrow{AB}\cdot \overrightarrow{u}=0} \Leftrightarrow \heva{&2\cdot \dfrac{a+1}{2}-3\cdot \dfrac{b-1}{2}+1=0\\ &3(a-1)+2(b+1)=0}\Leftrightarrow \heva{&a=-\dfrac{11}{13}\\ &b=\dfrac{23}{13}.} 
\end{align*}

\end{align*}
