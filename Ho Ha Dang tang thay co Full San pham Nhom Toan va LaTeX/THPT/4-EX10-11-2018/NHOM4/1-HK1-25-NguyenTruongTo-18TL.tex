\begin{bt}%[1D1B3]
Giải phương trình sau: $3\cos^2x -2\sin x + 2 =0$.

\loigiai{
 Phương trình tương đương $3(1-\sin^2x) -2\sin x +2 =0$
\newline $\Leftrightarrow -3\sin^2x -2\sin x +5 =0\Leftrightarrow
 \hoac{&\sin x = 1\\&\sin x =-\dfrac{5}{3}}$.\\
 Với $\sin x = -\dfrac{5}{3}:\, \text{Phương trình vô nghiệm}$\\
 Với $\sin x = 1 \Leftrightarrow x = \dfrac{\pi}{2}+k2\pi, k \in \mathbb{Z}$.\\
 Vậy phương trình có nghiệm $x = \dfrac{\pi}{2}+k2\pi, k \in \mathbb{Z}$.
}
\end{bt}
\begin{bt}%[1D1K3]

\loigiai{
 Phương trình tương đương:\newline
$
\dfrac{1}{2}\sin 2x -\dfrac{\sqrt{3}}{2}\cos 2x =\cos x \newline
\Leftrightarrow \sin 2x\cos \dfrac{\pi}{3} -\cos 2x \sin \dfrac{\pi}{3}=\cos x
\Leftrightarrow \sin \left(2x-\dfrac{\pi}{3}\right) = \sin \left(\dfrac{\pi}{2}-x\right) \\ \Leftrightarrow \hoac{&2x-\dfrac{\pi}{3}=\dfrac{\pi}{2}-x+k2\pi\\&2x-\dfrac{\pi}{3}=\pi -\dfrac{\pi}{2}+x+k2\pi },k \in \mathbb{Z}\Leftrightarrow\hoac{&x=\dfrac{5\pi}{18}+k\dfrac{2\pi}{3}\\&x=\dfrac{5\pi}{6}+k2\pi},k \in \mathbb{Z}.$
}
\end{bt}
\begin{bt}%[1D2B5]%[1D2K5]
 Một hộp chứa $7$ bi xanh, $6$ bi vàng và $3$ bi đỏ. Lấy ngẫu nhiên $4$ viên bi. Tính xác suất sao cho
\begin{enumerate}
\item Lấy được đúng $2$ bi vàng.
\item Lấy được ít nhất một bi xanh.
\end{enumerate}
\loigiai{\begin{enumerate}

\item Số phần tử của không gian mẫu là $n(\Omega)=\mathrm{C}_{16}^4=1820$.\newline
Gọi $A$:  \lq\lq Lấy được đúng $2$ viên bi vàng\rq\rq $\Rightarrow n(A) = \mathrm{C}_6^2\times \mathrm{C}_{10}^2 =675$.
\newline
Vậy $\mathrm{P}(A) = \dfrac{n(A)}{n(\Omega)}=\dfrac{135}{364}$
\item Gọi $B$: \lq \lq Lấy được ít nhất một viên bi xanh\rq\rq. \newline
Suy ra biến cố đối của biến cố $B$ là $\overline{B}$: \lq \lq Không lấy được viên bi xanh nào\rq \rq.
\newline
$\Rightarrow n\left(\overline{B}\right) = \mathrm{C}_9^4=126$.
\newline
Ta có: $\mathrm{P}\left(\overline{B}\right) = \dfrac{n\left(\overline{B}\right)}{n(\Omega)}=\dfrac{9}{130}$.
\newline
Vậy $\mathrm{P}(B) = 1- \mathrm{P}\left(\overline{B}\right) = 1-\dfrac{9}{130}=\dfrac{121}{130}$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[1D2B3]
 Tìm số hạng chính giữa của khai triển $(1-3x)^{20}$.
\loigiai{
Số hạng tổng quát trong khai triển là: $T_{k+1}=\mathrm{C}_{20}^k1^{20-k}(-3x)^k = \mathrm{C}_{20}^k(-3)^kx^k$.\newline
Số hạng chính giữa trong khai triển là số hạng thứ $11$ ứng với $k =10$.\newline
Vậy số hạng cần tìm là: $T_{11}=\mathrm{C}_{20}^{10}(-3)^{10}x^{10}$.
}

\end{bt}

\begin{bt}%[1H1B2]
Trong mặt phẳng với hệ tọa độ $Oxy$, viết phương trình đường thẳng $d'$ là ảnh của đường thẳng $d$: $2x-3y+6=0$ qua phép tịnh tiến theo véc-tơ $\vec{v}=(2;-1)$.
\loigiai{
Gọi $\mathrm{T}_{\vec{v}}(d) = d' \Rightarrow \hoac{&d\parallel d'\\&d\equiv d'}$. Suy ra phương trình của $d': 2x-3y +c=0, c \in \mathbb{R}$.\newline
Gọi $M(0;2) \in d$, $M' =\mathrm{T}_{\vec{v}}(M) \Rightarrow M'(2;1) \in d'$.\newline
Suy ra $4-3+c=0\Leftrightarrow c =-1$.\newline
Vậy phương trình $d': 2x-3y-1=0$.
}

\end{bt}

\begin{bt}%[1D2G3]
Rút gọn biểu thức $A = \mathrm{C}_{2n}^0 + \mathrm{C}_{2n}^2 + \mathrm{C}_{2n}^4 + \mathrm{C}_{2n}^{2n}$ với $n \in \mathbb{N}^*$.
\loigiai{
Xét khai triển $(1+x)^{2n}=\mathrm{C}_{2n}^0 + \mathrm{C}_{2n}^1x + \mathrm{C}_{2n}^2x^2+...+\mathrm{C}_{2n}^{2n}x^{2n}$.
\newline
Cho $x=1$ ta có: $\mathrm{C}_{2n}^0 + \mathrm{C}_{2n}^1 + \mathrm{C}_{2n}^2 + ... + \mathrm{C}_{2n}^{2n}=2^{2n}$  $(1)$.
\newline
Cho $x =-1$ ta có: $\mathrm{C}_{2n}^0 - \mathrm{C}_{2n}^1 + \mathrm{C}_{2n}^2 - ... +\mathrm{C}_{2n}^{2n}=0$ $(2)$. \newline
Cộng $(1)$ và $(2)$ vế theo vế ta có: $2(\mathrm{C}_{2n}^0 + \mathrm{C}_{2n}^2 + \mathrm{C}_{2n}^4 + ... + \mathrm{C}_{2n}^{2n}) = 2^{2n}\Leftrightarrow 2A = 2^{2n} \Leftrightarrow A = 2^{2n-1}$.
}
\end{bt}
