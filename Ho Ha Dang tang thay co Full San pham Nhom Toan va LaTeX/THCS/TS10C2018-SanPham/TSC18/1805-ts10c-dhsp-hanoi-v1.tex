\begin{ex}%[Đề thi vào 10, chuyên sư phạm Hà Nội, Vòng 1, năm 2018]%[Nguyễn Duy Khánh, Dự án EX-TS10C-18]%[9D1G8]%[9D1K2]
	Cho biểu thức
	$$P = \dfrac{2}{(x+1)\sqrt{x+1}+(x-1)\sqrt{x-1}}\cdot\dfrac{\dfrac{2x}{\sqrt{x-1}}-\sqrt{x+1}}{\dfrac{1}{\sqrt{x-1}}-\dfrac{1}{\sqrt{x+1}}},\quad \mbox{với } x > 1.$$
	\begin{listEX}
	\item Rút gọn biểu thức P.
	\item Tìm $x$ để $P = x-1$.
	\end{listEX}
	
	\loigiai{
	\begin{enumerate}[\quad a)]
	\item Với $x > 1,$ ta có
	\begin{eqnarray*}
			P &= &\dfrac{2}{\sqrt[3]{(x+1)^3}+\sqrt[3]{(x-1)^3}}\cdot\dfrac{2x-\sqrt{x+1}.\sqrt{x-1}}{\sqrt{x-1}}\cdot\dfrac{\sqrt{x-1}.\sqrt{x+1}}{\sqrt{x+1}-\sqrt{x-1}}\\
	& =& \dfrac{2}{\left(\sqrt{x+1}+\sqrt{x-1}\right)\left(x+1-\sqrt{x+1}\cdot\sqrt{x-1}+x-1\right)}.\dfrac{\left(2x-\sqrt{x+1}\cdot\sqrt{x-1}\right)\sqrt{x+1}}{\sqrt{x+1}-\sqrt{x-1}}\\
	& = &\dfrac{2}{\left(\sqrt{x+1}+\sqrt{x-1}\right)\left(2x-\sqrt{x+1}\cdot\sqrt{x-1}\right)}.\dfrac{\left(2x-\sqrt{x+1}\cdot\sqrt{x-1}\right).\sqrt{x+1}}{\sqrt{x+1}-\sqrt{x-1}}\\
	& =& \dfrac{2}{\left(\sqrt{x+1}+\sqrt{x-1}\right)\left(\sqrt{x+1}-\sqrt{x-1}\right)}\\
	& = &\dfrac{2\sqrt{x+1}}{x+1-\left(x-1\right)} = \dfrac{2\sqrt{x+1}}{2}\\
	& = &\sqrt{x+1}.
\end{eqnarray*}
	\item Với $x > 1$ ta có
\begin{eqnarray*}
P = x-1 &\Leftrightarrow & \sqrt{x+1} = x-1\\
&\Leftrightarrow & x+1 = (x-1)^2\\
&\Leftrightarrow & x(x-3) = 0.\\
& \Leftrightarrow &  \hoac{&x = 0 \mbox{ (không thỏa mãn) }\\&x = 3\mbox{ (thỏa mãn).}}
\end{eqnarray*}


	Vậy với $x = 3$ thì $P = x-1$.
	\end{enumerate}		
	}
\end{ex}



\begin{ex}%[Đề thi vào 10, chuyên sư phạm Hà Nội, Vòng 1, năm 2018]%[Nguyễn Duy Khánh, Dự án EX-TS10C-18]%[9D4G8]
Một nhà máy chuyên sản xuất một loại sản phẩm. Năm 2015, nhà máy sản xuất được 5000 sản phẩm. Do ảnh hưởng của thị trường tiêu thụ nên sản lượng của nhà máy trong các năm 2016 và 2017 đều giảm. Cụ thể: số lượng sản phẩm nhà máy sản xuất được trong 2016 giảm $x\%$ so với số lượng sản phẩm nhà máy sản xuất được trong năm 2015, số lượng sản phẩm nhà máy sản xuất được trong năm 2017 cũng giảm $x\%$ so với số lượng sản phẩm nhà máy sản xuất được trong năm 2016. Biết rằng số lượng sản phẩm nhà máy sản xuất được trong năm 2017 giảm $51\%$ so với số lượng sản phẩm nhà máy sản xuất được trong 2015. Tìm $x$.
\loigiai{
Điều kiện $0 < x < 100$.\\
Số sản phẩm nhà máy đó sản xuất được trong năm 2016 là: $$5000-5000\cdot x\% = 5000-50x\mbox{ (sản phẩm)}.$$
Số sản phẩm nhà máy đó sản xuất được trong năm 2017 là:
\begin{eqnarray*}
5000-50x-(5000-50x)\cdot x\% & = & 5000-50x-\dfrac{(100-x)x}{2}\\
& = & \dfrac{x^2-200x+10000}{2}\mbox{ (sản phẩm).}
\end{eqnarray*}
Số sản phẩm năm 2017 giảm 51\% so với năm 2015 nên ta có phương trình:
\begin{eqnarray*}
& &\dfrac{x^2-200x+10000}{2} = 5000-5000\cdot 51\%\\
&\Leftrightarrow &  x^2-200x+10000 = 4900\\
& \Leftrightarrow & x^2-200x+5100 = 0\\
&\Leftrightarrow &\hoac{&x=30 \mbox{ (thỏa mãn)}\\&x=170\mbox{ (không thỏa mãn).}}
\end{eqnarray*}



Vậy $x = 30$.
}
    \end{ex}
    
    
\begin{ex}% [Đề thi vào 10, chuyên sư phạm Hà Nội, Vòng 1, năm 2018]%[Nguyễn Duy Khánh, Dự án EX-TS10C-18]%[9D0G1]%[9D0G2]
Cho phương trình $x^3 -x-1 = 0$. Giả sử $x_0$ là một nghiệm của phương trình đã cho.
\begin{enumerate}
\item Chứng minh $x_0 > 0.$
\item Tính giá trị của biểu thức
$$M = \dfrac{x_0^2-1}{x_0^3}\cdot\sqrt{2x_0^2+3x_0+2}.$$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta xét các trường hợp sau
\begin{itemize}
\item Nếu $x_0 < -1$ ta có $x_0^3-x_0=x_0(x_0^2-1)$.\\
Vì $x_0<-1$ nên $x_0^2-1>0$ suy ra $x_0^3-x_0-1<0$ (vô lý)
\item Nếu $-1\leq x_0 < 0$ ta có $x_0^3-x_0<-x_0\leq 1$ (vô lý)
\item Nếu $x_0 = 0$, ta thấy không phải là nghiệm của phương trình\\
Vậy $x_0 > 0$.

\end{itemize}
\item Ta đặt $x_0=a \; (a > 0)$ để tiện biến đổi
$$
M = \dfrac{a^2-1}{a^3}.\sqrt{2a^2+3a+2}
 = \dfrac{(a-1)(a+1)}{a^3}.\sqrt{2a^2+3a+2}
$$
Vì $a^3-a-1=0 \Rightarrow a^3=a+1 > 1 \;(\mbox{vì } a > 0)$, từ đó suy ra $a^3 > 1$ hay $a > 1$. Vậy $$M = (a-1)\sqrt{2a^2+3a+2}>0. \quad (*) $$

Ta có \begin{eqnarray*} M^2&=&(a-1)^2(2a^2+3a+2)\\
& =& (a^2-2a+1)(2a^2+3a+2)\\
& = & 2a^4+3a^3+2a^2-4a^3-6a^2-4a+2a^2+3a+2\\
&= & 2a^4-a^3-2a^2-a+2.
\end{eqnarray*}
mặt khác $a^3=a+1$ suy ra $a^4=a^2+a$\\
suy ra $$M^2 = 2(a^2+a)-a^3-2a^2-a+2 = -a^3+a+2=-(a^3-a-1)+1 = 1.$$
Mà $M > 0$ (chứng minh trên) suy ra $M = 1$.\\
Vậy $M = 1$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên sư phạm Hà Nội, Vòng 1, năm 2018]%[Nguyễn Duy Khánh, Dự án EX-TS10C-18]%[9D0G3]
Cho các số nguyên $x,\;x_1,x_2\;\ldots,x_9$ thỏa mãn
$$(1+x_1)(1+x_2)\ldots (1+x_9) = (1-x_1)(1-x_2)\ldots(1-x_9) = x$$
Tính $P = x.\:x_1.\:x_2\:\ldots\:x_9$.
\loigiai{
\begin{itemize}
\item Nếu tồn tại $i \; (i = \overline{1;9})$ mà $|x_i| = 1$ ta có $x = 0$ $\Rightarrow P = x.\:x_1.\:x_2\:\ldots\:x_9 = 0.$
\item Nếu tồn tại $i\; (i = \overline{1;9})$ mà $x_i = 0$ $\Rightarrow P = x.\:x_1.\:x_2\:\ldots\:x_9 = 0$
\item Xét trường hợp $x_i\ne 0$ và $x_i \ne 1; x_i\ne -1$,\\
Ta có 
\begin{eqnarray*}
 x^2 &=& (1+x_1)(1+x_2)\ldots (1+x_9).(1-x_1)(1-x_2)\ldots(1-x_9)\\
 x^2 & =& (1-x_1^2)(1-x_2^2)\ldots(1-x_9^2)
\end{eqnarray*}
Vì $x_i\ne 0; x_i \ne 1; x_i\ne -1$ và $x_i\in \mathbb{Z} \Rightarrow 1 - x_i^2 < 0$\\
$\Rightarrow (1-x_1^2)(1-x_2^2)\ldots (1-x_9^2)<0$\\
$\Rightarrow x^2 < 0$ (vô lý).\\
Vậy $P = 0.$
\end{itemize}
}
\end{ex}
