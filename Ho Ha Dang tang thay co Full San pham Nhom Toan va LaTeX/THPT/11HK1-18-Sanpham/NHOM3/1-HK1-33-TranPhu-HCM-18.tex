\begin{bt}%[Đề HK1 lớp 11, Trần Phú, Hồ Chí Minh 2018]%[Trần Tuấn Việt, Dự án 11-HK1-18]%[1D1B1-1]
	Tìm tập xác định của hàm số $y = \dfrac{\tan x + 2}{1 - \sin x}$.
	\loigiai{Điều kiện xác định: $\heva{&\cos x \neq 0\\ & \sin x \neq 1} \Leftrightarrow \heva{& x \neq \dfrac{\pi}{2} + k \pi\\& x \neq \dfrac{\pi}{2} + k2\pi} \Leftrightarrow x \neq \dfrac{\pi}{2} + k\pi, k \in \mathbb{Z}$.\medskip
	
\noindent Vậy tập xác định của hàm số là $\mathscr{D} = \mathbb{R} \setminus \left\lbrace \dfrac{\pi}{2} + k\pi,\, k \in \mathbb{Z}\right\rbrace$.}
\end{bt}

\begin{bt}%[Đề HK1 lớp 11, Trần Phú, Hồ Chí Minh 2018]%[Trần Tuấn Việt, Dự án 11-HK1-18]%[1D1B3-5]
Giải phương trình $1 + \cos x + \cos 2x + \cos 3x = 0$.
\loigiai{\begin{eqnarray*}
1 + \cos x + \cos 2x + \cos 3x = 0 & \Leftrightarrow &(\cos 2x + 1) + (\cos x + \cos 3x) = 0\\
&\Leftrightarrow & 2 \cos ^2x + 2\cos 2x \cos x = 0\\
& \Leftrightarrow & \cos x (\cos x + \cos 2x) = 0\\
&\Leftrightarrow & \hoac{&\cos x = 0\\&\cos x + \cos 2x = 0}\\
& \Leftrightarrow & \hoac{&x = \dfrac{\pi}{2} + k \pi\\ &x = \pi + k2\pi\\ &x = \dfrac{\pi}{3} + k\dfrac{2\pi}{3}}, k \in \mathbb{Z}.
\end{eqnarray*}}
\end{bt}

\begin{bt}%[Đề HK1 lớp 11, Trần Phú, Hồ Chí Minh 2018]%[Trần Tuấn Việt, Dự án 11-HK1-18]%[1D2B3-2]
Tìm số hạng có số mũ của $x$ bằng số mũ của $y$ trong khai triển Newton của nhị thức $\left(x^2 + \dfrac{2y}{x}\right)^{14}$, với $x \neq 0$.
	\loigiai{\begin{itemize}
	\item $\left(x^2 + \dfrac{2y}{x}\right)^{14} = \displaystyle \sum_{k = 0}^{14} \mathrm{C}_{14}^k (x^2)^{14 - k} \left( \dfrac{2y}{x}\right)^k =  \sum_{k = 0}^{14}\mathrm{C}_{14}^k 2^k x^{28 - 3k} y^k$.
	\item Yêu cầu bài toán tương đương $28 - 3k = k \Leftrightarrow k = 7.$
	\item Số hạng cần tìm là $\mathrm{C}_{14}^72^7x^7y^7 = 439296x^7y^7$. \end{itemize}}
\end{bt}

\begin{bt}%[Đề HK1 lớp 11, Trần Phú, Hồ Chí Minh 2018]%[Trần Tuấn Việt, Dự án 11-HK1-18]%[1D2B1-3]
Để cổ vũ giải bóng chuyền nữ chào mừng ngày $20 - 11$, một đội học sinh gồm có $5$ nam và $5$ nữ xếp hàng ngang kế bên sân thi đấu. Hỏi có bao nhiêu cách xếp sao cho nam, nữ đứng xen kẽ nhau.
	\loigiai{Ta xét hai trường hợp sau:
\begin{itemize}
\item TH1: nam ở vị trí lẻ, nữ ở vị trí chẵn thì số cách chọn là $5! 5!$ (cách).
\item TH2: nam ở vị trí chẵn, nữ ở vị trí lẻ, số cách chọn là $5! 5!$ (cách).
\end{itemize}
Vậy có $2\cdot 5! 5! = 28800$ cách xếp nam và nữ xen kẽ nhau.
}
\end{bt}


\begin{bt}%[Đề HK1 lớp 11, Trần Phú, Hồ Chí Minh 2018]%[Trần Tuấn Việt, Dự án 11-HK1-18]%[1D2K5-2]
Trong giải bóng bàn có $12$ giáo viên tham gia, trong đó có thầy Anh và thầy Tài. Ban tổ chức chia $12$ người ra làm hai bảng thi đấu $A$ và $B$, mỗi bảng có $6$ người. Biết rằng việc bốc thăm bảng thi đấu diễn ra một cách ngẫu nhiên, tính xác suất để thầy Anh và thầy Tài ở $2$ bảng đấu khác nhau.
	\loigiai{\begin{itemize}
	\item Số cách sắp xếp $12$ người xếp vào hai bảng đấu là $n(\Omega) = \mathrm{C}_{12}^6 \cdot \mathrm{C}_6^6 = 924$.
	\item Gọi $A$ là biến cố thầy Anh và thầy Tài ở hai bảng đấu khác nhau.
	\begin{enumerate}[-]
	\item Số cách xếp thầy Anh vào một trong hai bảng đấu có $2$ cách.
	\item Số cách chọn $5$ người (không có thầy Tài) vào bảng thầy Anh là $\mathrm{C}_{10}^5$.\\
	$\Rightarrow$ $n(A) = 2\mathrm{C}_{10}^5 = 504$.
	\end{enumerate}
	\item $P(A) = \dfrac{n(A)}{n(\Omega)} = \dfrac{6}{11}$.
	\end{itemize}
Vậy xác suất để thầy Anh và thầy Tài ở $2$ bảng đấu khác nhau là $P(A) = \dfrac{6}{11}$.}
\end{bt}	


\begin{bt}%[Đề HK1 lớp 11, Trần Phú, Hồ Chí Minh 2018]%[1D3B3-4]
Cho $a, b, c$ là $3$ số hạng liên tiếp của một cấp số cộng. Chứng minh rằng $$(a + b + c)^2 + 8(b - c)^2 = 4a^2 + b^2 + 4c^2.$$ \loigiai{\begin{itemize}
\item $a,b,c$ là $3$ số hạng liên tiếp của cấp số cộng nên ta giả sử $b = a + d, c = a + 2d$.
\item Ta có 
$VT = (3a + 3d)^2 + 8d^2 = 9a^2 + 9d^2 + 18ad + 8d^2 = 9a^2 + 17d^2 + 18ad. \qquad (1)$\\
$VP = 4a^2 + (a + d)^2 + (a + 2d)^2 = 9a^2 + 17d^2 + 18ad. \qquad (2)$
\item Từ $(1)$ và $(2)$, ta suy ra $(a + b + c)^2 + 8(b - c)^2 = 4a^2 + b^2 + 4c^2.$ (đpcm)
\end{itemize}
}
\end{bt}	



\begin{bt}%[Đề HK1 lớp 11, Trần Phú, Hồ Chí Minh 2018]%[1D3B4-2]
Tìm số hạng đầu $u_1$ và công bội $q$ của cấp số nhân $(u_n)$, biết $\heva{&u_5 - u_1 = 30\\&u_4 - u_2 = 12}$.
	\loigiai{Dễ có $q \neq 0, q \neq 1$. Khi đó ta có\\
	$\heva{&u_5 - u_1 = 30\\&u_4 - u_2 = 12} \Leftrightarrow \heva{&u_1 q^4 - u_1 = 30\\&u_1q^3 - u_1q = 12} \Leftrightarrow \heva{&u_1( q^4 - 1) = 30\\&u_1 \cdot q( q^2 - 1) = 12}$ $\Leftrightarrow$ $\dfrac{q^2 + 1}{q} = \dfrac{5}{2} \Leftrightarrow \hoac{& q = 2, \, u_1 = 2\\&q = \dfrac{1}{2},\, u_1 = -32.}$}
\end{bt}	


\begin{bt}%[Đề HK1 lớp 11, Trần Phú, Hồ Chí Minh 2018]%[1H2B3-3]%[1H2B2-4]%[1H2K3-2]
