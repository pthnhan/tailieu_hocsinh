\begin{bt}%[Đề HK1, Củ Chi, Hồ Chí Minh, 2017]%[Phùng Hoàng Em, dự án 10-EX-HCM]%[0D2K1-2][0D2B1-4]
	\item 
	\begin{enumerate}
		\item Tìm tập xác định của hàm số $f(x)=\dfrac{\sqrt{3-x}}{x^2-3x+2}$.
		\item Xét tính chẵn lẻ của hàm số $f(x)=3x^5+7x^3-5x$.
	\end{enumerate}
\loigiai{
		\begin{enumerate}
			\item Hàm số xác định khi $\heva{&3-x \ge 0\\&x^2-3x+2\ne 0}\Leftrightarrow \heva{& x \le 3\\&x\ne 1 \text{ và } x\ne 2.}$\\
			Suy ra, tập xác định của hàm số là $\mathscr{D}=\left(-\infty;3\right]  \setminus \left\{1; 2\right\}$.
			\item Tập xác định $\mathscr{D}=\mathbb{R}$.
			\begin{itemize}
				\item Ta có $\forall x\in \mathscr{D}$ thì $ - x\in \mathscr{D}$;
				\item Mặt khác,  $f\left(-x\right)=3{\left(-x\right)}^5+7{\left(-x\right)}^3-5\left(-x\right)=-\left(3x^5+7x^3-5x\right)=-f(x)$.
			\end{itemize}
			Suy ra hàm số $f(x)=3x^5+7x^3-5x$ là hàm số lẻ.
	\end{enumerate}
	}
\end{bt}
\begin{bt}%[Đề HK1, Củ Chi, Hồ Chí Minh, 2017]%[Phùng Hoàng Em, dự án 10-EX-HCM]%[0D2K3-2]
	Xác định hàm số $y=ax^2 + bx + c$ $\left(a\ne 0\right)$ có đồ thị $(P)$, biết $(P)$ qua $A\left(0; 5\right)$ và có đỉnh $I(3;-4)$.
\loigiai{
	$(P)$ qua $A\left(0; 5\right)$ và có đỉnh $I(3; - 4)$ nên ta có hệ phương trình
	$$\heva{&a\cdot 0^2+b\cdot 0 + c=5 \\ &-\dfrac{b}{2a}=3 \\& 9a + 3b + c= - 4} \Leftrightarrow \heva{& c=5 \\ &6a + b=0 \\ 
	& 9a + 3b + c= - 4} \Leftrightarrow \heva{& a=1 \\ & b= - 6 \\ 
	& c=5.}$$
	Vậy, hàm số cần tìm là $y=x^2 - 6x + 5$.
}
\end{bt}

\begin{bt}%[Đề HK1, Củ Chi, Hồ Chí Minh, 2017]%[Phùng Hoàng Em, dự án 10-EX-HCM]%[0D3B2-6]
	Cho phương trình $mx^2 - 2(m - 1)x + m + 7=0$, với $m$ là tham số. Tìm $m$ để phương trình có nghiệm kép. Tính nghiệm kép đó.
\loigiai{
	Phương trình có nghiệm kép khi và chỉ khi
	\begin{eqnarray*}
		\heva{& a\ne 0 \\  & \Delta =0} &\Leftrightarrow& \heva{& m\ne 0 \\  & 4{(m - 1)}^2 - 4m(m + 7)=0}\\
		&\Leftrightarrow& \heva{& m\ne 0 \\  & - 36m + 4=0} \Leftrightarrow \heva{& m\ne 0 \\  & m=\dfrac{1}{9}} \Leftrightarrow m=\dfrac{1}{9}.
	\end{eqnarray*}
	Nghiệm kép của phương trình là $x=-\dfrac{b}{2a}=\dfrac{2(m-1)}{2m}=\dfrac{2\left(\dfrac{1}{9}-1\right)}{2\cdot \dfrac{1}{9}}= - 8$.
}
\end{bt}

\begin{bt}%[Đề HK1, Củ Chi, Hồ Chí Minh, 2017]%[Phùng Hoàng Em, dự án 10-EX-HCM]%[0D4B5-5][0D4B5-6][0D4B5-2][0D4K5-3]
	Giải các phương trình, bất phương trình sau:
	\begin{multicols}{2}
	\begin{enumerate}
		\item $\left|5x^2 - 3x - 2\right|=\left|x^2 - 1\right|$;
		\item $9x + \sqrt{3x - 2}=10$;
		\item $\left(2x + 1\right)(3 - x)<0$;
		\item $\dfrac{x^2 - 3x}{(x + 2)\left(1 - 2x\right)}\le 0$.
	\end{enumerate}
	\end{multicols}
\loigiai{
	\begin{enumerate}
		\item Phương trình tương đương với 
		$$\hoac{& 5x^2 - 3x - 2=x^2 - 1\\& 5x^2 - 3x - 2= - \left(x^2 - 1\right)} \Leftrightarrow\hoac{	& 4x^2 - 3x - 1=0 \\ & 6x^2 - 3x - 3=0}\Leftrightarrow \hoac{& x=1; x=-\dfrac{1}{4} \\& x=1; x=-\dfrac{1}{2}.}$$
		Vậy tập nghiệm là $S=\left\{1;-\dfrac{1}{2};-\dfrac{1}{4}\right\}$.
		\item Phương trình tương đương với
		\begin{eqnarray*}
		\sqrt{3x - 2}=10 - 9x &\Leftrightarrow& \heva{& 10 - 9x\ge 0 \\  & 3x - 2=\left(10 - 9x\right)^2}
		\Leftrightarrow \heva{& x\le \dfrac{10}{9} \\  & 81x^2 - 183x + 102=0}\\
			&\Leftrightarrow& \heva{& x\le \dfrac{10}{9} \\  & x=1 \text{ hoặc }x=\dfrac{34}{27}} \Leftrightarrow x=1.
		\end{eqnarray*}
		Vậy tập nghiệm là $S=\left\{1\right\}$.
		\item $\left(2x + 1\right)(3 - x)<0\Leftrightarrow - 2x^2 + 5x + 3<0$.\\
		Xét $ - 2x^2 + 5x + 3=0\Leftrightarrow x=-\dfrac{1}{2}$ hoặc $x=3$.\\
       	Bảng xét dấu
            \begin{center}
                 \begin{tikzpicture}
              \tkzTabInit[nocadre=false, lgt=3.5, espcl=2.5]
              {$x$/1,$- 2x^2 + 5x + 3$/1}
              {$-\infty$,$-\dfrac{1}{2}$,$3$,$+\infty$}
              \tkzTabLine{,-,0,+,0,-,}
              \end{tikzpicture}
            \end{center}
      	Vậy tập nghiệm là $S=\left(- \infty;-\dfrac{1}{2}\right)\cup\left(3;+\infty\right)$.
		\item Đặt $f(x)=\dfrac{x^2 - 3x}{(x + 2)\left(1 - 2x\right)}=\dfrac{x^2 - 3x}{-2x^2-3x+2}$.\\
		Điều kiện xác định là $x \ne -2$ và $x \ne \dfrac{1}{2}$.
		\begin{itemize}
			\item $x^2 - 3x=0\Leftrightarrow x=0$ hoặc $x=3$.
			\item $- 2x^2 - 3x + 2=0\Leftrightarrow x= - 2$ hoặc $x=\dfrac{1}{2}$.
		\end{itemize}
		Bảng xét dấu
		\begin{center}
		     \begin{tikzpicture}
		     \tkzTabInit[deltacl=0.5,espcl=1.5,lgt=3.5]
		     {$x$/1,$x^2 - 3x$/1,$- 2x^2 - 3x + 2$/1,$f(x)$/1}
		     {$-\infty$,$-2$,$0$,$\dfrac{1}{2}$,$3$,$+\infty$}
		     \tkzTabLine{,+,t,+,0,-,t,-,0,+,}
		     \tkzTabLine{,-,0,+,t,+,0,-,t,-,}
		     \tkzTabLine{,-,d,+,0,-,d,+,0,-,}
		     \end{tikzpicture}
    	\end{center}
		Vậy tập nghiệm là $S=\left(- \infty; - 2\right)\cup \left[0; \dfrac{1}{2}\right)\cup \left[3; + \infty\right)$.
	\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Củ Chi, Hồ Chí Minh, 2017]%[Phùng Hoàng Em, dự án 10-EX-HCM]%[0H2B2-2][0H2K2-3]
    Trong mặt phẳng $Oxy$, cho tam giác $ABC$ với $A\left(-4;1\right), B\left(2;4\right), C\left(2;-2\right)$.
    \begin{enumerate}
        \item Chứng minh tam giác $ABC$ là tam giác cân. Tính chu vi, diện tích tam giác $ABC$.
        \item Tìm tọa độ $H$ là chân đường cao hạ từ $B$ của tam giác $ABC$.
        \item Tìm tọa độ tâm đường tròn nội tiếp tam giác $ABC$.
    \end{enumerate}
    \loigiai{
       	 \begin{center}
    		\begin{tikzpicture}[scale=0.8]
    		\tkzInit[xmin=-4.5,xmax=4 ,ymin=-1.4 ,ymax=4.2]
			\tkzClip[space=.4]
    		\tkzDefPoints{0/4/A,-4/-1/B,4/-1/C}
    		\tkzDrawAltitude(A,C)(B)\tkzGetPoint{H}
    		\tkzDefMidPoint(B,C)
    		\tkzGetPoint{M}
    		\tkzInCenter(A,B,C)\tkzGetPoint{I}
    		\tkzDrawPoints[fill=black](A,B,C,I,M,H)
    		\tkzDrawSegments(A,B B,C A,C A,M B,I)
    		\tkzMarkRightAngle(B,H,C)
    		\tkzMarkSegments[mark=||](M,B M,C)
    		\tkzLabelPoints[right](I,H)
    		\tkzLabelPoints[above](A)
    		\tkzLabelPoints[below](B,M,C)
    	 \end{tikzpicture}
    	\end{center}
   \begin{enumerate}
            \item Ta có $AB= 3\sqrt{5}$, $AC=3\sqrt{5}$, $BC=6$. Vậy $\triangle ABC$ cân tại $A$.\\
            Chu vi của $\triangle ABC$ là $2p=AB+BC+CA=6+6\sqrt{5}$.\\
            Ta có $p=3+3\sqrt{5}$. Diện tích của $\triangle ABC$ là $S=\sqrt{p\left(p-AB\right)\left(p-AC\right)\left(p-BC\right)}=18$.
            \item Gọi $H(a;b)$ là chân đường cao kẻ từ $B$ của $\triangle ABC$.\\
            Khi đó $\vec{BH}=(a-2;b-4)$, $\vec{AC}=(6;-3)$, $\vec{AH}=(a+4;b-1)$.\\
            Ta có $\heva{&\vec{BH} \cdot \vec{AC}=0\\&\vec{AH} \text{ cùng phương } \vec{AC}} \Leftrightarrow \heva{&(a-2)\cdot 6 +(b-4)\cdot (-3)=0 \\& \dfrac{a+4}{6}=\dfrac{b-1}{-3}}\Leftrightarrow \heva{&a=-\dfrac{2}{5}\\&b=-\dfrac{4}{5}.}$\\
            Vậy $H\left(-\dfrac{2}{5};-\dfrac{4}{5}\right)$.
            \item Gọi $I(m;n)$ là tâm đường tròn nội tiếp tam giác $ABC$; $M\left(2;1\right)$ là trung điểm của $BC$.\\
            Do $\Delta ABC$ cân tại $A$ nên $AM$ cũng là đường phân giác của $\triangle ABC$. Suy ra $I \in AM$.\\
            Xét $\triangle ABM$ có $BI$ là đường phân giác, suy ra: $\dfrac{IA}{IM}=\dfrac{BA}{BM}=\dfrac{3\sqrt{5}}{3}=\sqrt{5}$.\\
            Từ đây, ta có $\vec{IA}=-\sqrt{5}\cdot \vec{IM}\Leftrightarrow \heva{&-4-m=-\sqrt{5}\left(2-m\right)\\& 1-n=-\sqrt{5}\left(1-n\right)}\Leftrightarrow \heva{&m=\dfrac{2\sqrt{5}-4}{\sqrt{5}+1}=\dfrac{7-3\sqrt{5}}{2}\\&n=1.}$\\
            Vậy $I\left(\dfrac{7-3\sqrt{5}}{2};1\right)$.
        \end{enumerate}
        }
\end{bt}

\begin{bt}%[Đề HK1, Củ Chi, Hồ Chí Minh, 2017]%[Phùng Hoàng Em, dự án 10-EX-HCM]%[0H1K3-4]
    Cho tam giác $ABC$ có $M$ là điểm trên đoạn $BC$ sao cho $BM=3MC$.
    \begin{enumerate}
        \item Chứng minh rằng $\vec{AM}=\dfrac{1}{4}\vec{AB}+\dfrac{3}{4}\vec{AC}$.
        \item Gọi $I$ là trung điểm của $BC$ và $G$ là trọng tâm của $\triangle ABC$. Tính $\vec{AM}$ theo $\vec{AG}$ và $\vec{BC}$.
    \end{enumerate}
    \loigiai{
        \begin{center}
            \begin{tikzpicture}
            \tkzDefPoints{0/0/A,4/0/B,3/4/C}
            \tkzDefMidPoint(B,C)
            \tkzGetPoint{I}
            \tkzDefPointBy[homothety = center B ratio .75](C)
            \tkzGetPoint{M}
            \tkzCentroid(A,B,C)
            \tkzGetPoint{G}
            \tkzDrawPoints[fill=black](A,B,C,I,M,G)
            \tkzDrawSegments(A,B A,I)
            \tkzDrawSegments[->](A,M A,G B,C A,C)
            \tkzLabelPoints[right](M,I,B)
            \tkzLabelPoints[below](A,G)
            \tkzLabelPoints[above](C)
            \end{tikzpicture}
        \end{center} 
        
        \begin{enumerate}
            \item Ta có $\vec{AM}=\vec{AC}+\vec{CM}=\vec{AC}+\dfrac{1}{4}\vec{CB}=\dfrac{1}{4}\left(\vec{AB}-\vec{AC}\right)+\vec{AC}=\dfrac{1}{4}\vec{AB}+\dfrac{3}{4}\vec{AC}$.
            \item Từ kết quả của câu a), ta có:
            \begin{eqnarray*}
            \vec{AM}&=&\dfrac{1}{4}\vec{AB}+\dfrac{3}{4}\vec{AC}\\
            		&=&\dfrac{1}{2}\left(\vec{AB}+\vec{AC}\right)+\dfrac{1}{4}\left(\vec{AC}-\vec{AB}\right)\\
            		&=&\vec{AI}+\dfrac{1}{4}\vec{BC}=\dfrac{3}{2}\vec{AG}+\dfrac{1}{4}\vec{BC}.
            \end{eqnarray*}
        \end{enumerate}
    }
\end{bt}

\begin{bt}%[Đề HK1, Củ Chi, Hồ Chí Minh, 2017]%[Phùng Hoàng Em, dự án 10-EX-HCM]%[0D4G1-4]
Xét hai số thực $a$, $b$ thỏa mãn $|a| \le 1$ và $|b| \le 1$. Chứng minh $|a+b|\le |1+ab|$.
\loigiai{
Ta có: 
	\begin{eqnarray*}
	\left|a+b\right|\le \left|1+ab\right|&\Leftrightarrow& {(a+b)}^2\le {\left(1+ab\right)}^2\\
	&\Leftrightarrow& a^2+2ab + b^2\le 1+2ab+a^2b^2\\
	&\Leftrightarrow& 1-a^2+a^2b^2-b^2\ge 0\\
	&\Leftrightarrow& \left(1-a^2\right)-b^2\left(1-a^2\right)\ge 0\\
	&\Leftrightarrow& \left(1-a^2\right)\left(1-b^2\right)\ge 0 \quad (*)
	\end{eqnarray*}
	Do $\heva{& |a|\le 1 \\ & |b|\le 1} \Leftrightarrow \heva{& a^2\le 1 \\ & b^2\le 1} \Leftrightarrow \heva{& 1-a^2\ge 0 \\ & 1-b^2\ge 0} \Rightarrow \left(1-a^2\right)\left(1-b^2\right)\ge 0$, nên (*) đúng.\\ 
Vậy bất đẳng thức được chứng minh.}

\end{bt}
