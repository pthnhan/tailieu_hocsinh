\begin{ex}%[Đề thi vào 10, Sở giáo dục Hà Nam, 2016]%[9D1Y6]%[9D1B8]
    \hfill
    \begin{enumerate}
       \item Rút gọn biểu thức: $A=\sqrt{32}-\sqrt{72}+2\sqrt{3-2\sqrt{2}}.$
       \item Cho biểu thức $B=\left(\dfrac{1}{\sqrt{x}-2}-\dfrac{1}{\sqrt{x}+2}\right):\dfrac{\sqrt{x}}{x-4}$ với $0<x\neq 4;$ Rút gọn biểu thức $B$ và tìm $x$ để $B=12$.
     \end{enumerate}
\loigiai{
    \begin{enumerate}
        \item $A=\sqrt{32}-\sqrt{72}+2\sqrt{3-2\sqrt{2}}=4\sqrt{2}-6\sqrt{2}+2(\sqrt{2}-1)=-2.$
        \item Với $0<x\neq 4$ ta có:\\
        $B=\left(\dfrac{1}{\sqrt{x}-2}-\dfrac{1}{\sqrt{x}+2}\right):\dfrac{\sqrt{x}}{x-4}=\dfrac{\sqrt{x}+2-(\sqrt{x}-2)}{x-4}.\dfrac{x-4}{\sqrt{x}}=\dfrac{4}{\sqrt{x}}.$\\
        $B=12 \Leftrightarrow \dfrac{4}{\sqrt{x}}=12 \Leftrightarrow \sqrt{x}=\dfrac{1}{3} \Leftrightarrow x=\dfrac{1}{9}.$ (thỏa mãn điều kiện $0<x\neq 4$).
    \end{enumerate}
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Hà Nam, 2016]%[9D4Y4]%[9D3Y3]
\hfill
\begin{enumerate}
	\item Giải phương trình: $x^2-3x+2=0$.
	\item Giải hệ phương trình $\heva{&x-y=1\\&2x+3y=17}$.
\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Ta có $a+b+c=1-3+2=0$ nên phương trình đã cho có hai nghiệm phân biệt $x_1=1, x_2=\dfrac{c}{a}=2.$
		\item Ta có $\heva{&x-y=1\\&2x+3y=17}\Leftrightarrow \heva{&x=y+1\\&2(y+1)+3y=17}\Leftrightarrow \heva{&x=y+1\\&5y=15}\Leftrightarrow\heva{&x=4\\&y=3}.$\\
Vậy nghiệm của hệ phương trình đã cho là $\heva{&x=4\\&y=3}$.
	\end{enumerate}
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Hà Nam, 2016]%[9D4B3]%[9D4B6]
Trong mặt phẳng tọa độ $Oxy$ cho parabol $(P)$ có phương trình $y=2x^2$.
  \begin{enumerate}
  	\item Tìm tọa độ các giao điểm của đường thẳng $(d): y=3x+2$ và parabol $(P)$.
  	\item Chứng minh rằng đường thẳng $(d_m): y=mx+1$ luôn cắt parabol $(P)$ tại hai điểm phân biệt có hoành độ $x_1, x_2$. Tìm $m$ để $4(x_1^2+x_2^2)+(2x_1+1)(2x_2+1)=9.$
  \end{enumerate}
\loigiai{
    \begin{enumerate}
   	\item Xét phương trình hoành độ giao điểm của $(d)$ và $(P)$:
   $$2x^2=3x+2\Leftrightarrow 2x^2-3x-2=0$$
   Ta có $\Delta =25 \Rightarrow $ phương trình có hai nghiệm $\heva{&x=2\\&x=-\dfrac{1}{2}}$.\\
   Với $x=2 \Rightarrow y=8$;  với $x=-\dfrac{1}{2} \Rightarrow y=\dfrac{1}{2}.$\\
   Từ đó suy ra $d$ cắt $(P)$ tại hai điểm phân biệt $A(2;8), B\left(-\dfrac{1}{2};\dfrac{1}{2}\right)$.
   	\item Xét phương trình hoành độ giao điểm của $(d_m)$ và $(P)$:
   $$2x^2=mx+1 \Leftrightarrow 2x^2-mx-1=0$$
   Ta có $\Delta = m^2+8>0\ \forall m$ nên phương trình luôn có hai nghiệm phân biệt $x_1, x_2$ phân biệt với mọi $m$. Do đó $(d_m)$ luôn cắt parabol $(P)$ tại hai điểm phân biệt với mọi $m$.\\
   Theo định lý Vi-ét ta có $\heva{&x_1+x_2=\dfrac{m}{2}\\&x_1.x_2=-\dfrac{1}{2}}.$\\
   Ta có \begin{align*}
           4(x_1^2+x_2^2)+(2x_1+1)(2x_2+1)=9&\Leftrightarrow 4\left[(x_1+x_2)^2-2x_1x_2\right]+4x_1x_+2(x_1+x_2)-8=0 \\
           &\Leftrightarrow 4(x_1+x_2)^2-4x_1x_2+2(x_1+x_2)-8=0\\
           &\Leftrightarrow 4\left(\dfrac{m}{2}\right)^2-4\left(-\dfrac{1}{2}\right)+2.\dfrac{m}{2}-8=0 \\
           &\Leftrightarrow m^2+m-6=0\\
           &\Leftrightarrow \hoac{&m=-3\\&m=2}
         \end{align*}
   Vậy $(d_m): y=mx+1$ luôn cắt parabol $(P)$ tại hai điểm phân biệt có hoành độ $x_1, x_2$ và các hoành độ $x_1, x_2$ thỏa mãn điều kiện $4(x_1^2+x_2^2)+(2x_1+1)(2x_2+1)=9 \Leftrightarrow \hoac{&m=-3\\&m=2}.$
   \end{enumerate}}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Hà Nam, 2016]%[9H3Y7]%[9H3B3]%[9H3B7]%[9H3K4]
 Cho đường tròn tâm $O$ đường kính $AB$ cố định, điểm $D$ cố định thuộc đoạn $AO$ ($D$ không trùng với $A, O$). Kẻ dây $MN$ vuông góc với $AB$ tại $D$. Gọi $C$ là điểm tùy ý thuộc cung lớn $MN$ sao cho $C$ không trùng với $M, N$ và $B$. Gọi $E$ là giao điểm của $AC$ với $MN$.
 \begin{enumerate}
 	\item Chứng minh tứ giác $DECB$ nội tiếp.
 	\item Chứng minh $CA$ là tia phân giác của $\widehat{MCN}$.
 	\item Chứng minh $AB^2=AE.AC+BD.AB.$
 	\item Hãy xác định vị trí của điểm $C$ sao cho khoảng cách từ $N$ đến tâm đường tròn ngoại tiếp tam giác $CME$ là nhỏ nhất.
 \end{enumerate}
\loigiai{
    \begin{enumerate}
         \immini{\item Nối $CB$, ta có $\widehat{ACB}=90^{\circ}$ (góc nội tiếp chắn nửa đường tròn tâm $O$).\\
          Mặt khác theo giả thiết ta có $\widehat{EDB}=90^{\circ}$.\\
          Từ đó suy ra $\widehat{ACB}+\widehat{EDB}=180^{\circ}$, do đó tứ giác $DECB$ nội tiếp.
                   \item Nối $CM, CN, OM, ON$ ta có $OM=ON$ nên $\Delta OMN$ cân tại $O$, lại do $OD\perp MN$ nên $OD$ là phân giác của góc $\widehat{MON}$.\\
                       Từ đó suy ra $\wideparen{AM}=\wideparen{AN}$, do đó ta có $\widehat{MCA}=\widehat{NCA}$ nên $CA$ là tia phân giác của $\widehat{MCN}$.
         }
         {\begin{tikzpicture}
\tkzDefPoints{-3/0/B, 3/0/A, 1.5/0/D, -2.25/0/P}
\coordinate (O) at ($(A)!0.5!(B)$);
\tkzDrawCircle[radius](O,A)

\tkzInterLC(S,D)(O,A)

\tkzInterLC(S',P)(O,A)

\tkzInterLL(A,C)(M,N)

\tkzDrawPoints(A,B,C,D,O,E,M,N,O')
\tkzLabelPoints[right](A)

\tkzMarkRightAngle(B,C,A)

}
         \item Dễ thấy $\Delta ADE \backsim \Delta ACB$, từ đó suy ra $\dfrac{AD}{AC}=\dfrac{AE}{AB} \Leftrightarrow AE.AC=AD.AB$.\\
             Vậy $AE.AC+BD.AB=AD.AB+BD.AB=AB(AD+BD)=AB^2$.
         \item Gọi $O'$ là tâm đường tròn ngoại tiếp tam giác $CME$. Ta có $\widehat{AMN}=\widehat{MCA}$, từ đó suy ra $AM$ là tiếp tuyến của $(O')$ tại điểm $M$.\\
             Mặt khác $BM\perp AM \Rightarrow O'$ chạy trên đường thẳng $MB$. Vậy khoảng cách từ $N$ đến $O'$ nhỏ nhất khi $O'$ là hình chiếu của $N$ lên $MB.$\\
             Từ đó ta có cách xác định điểm $C$ như sau:\\
             - Xác định điểm $O'$ là hình chiếu vuông góc của điểm $N$ trên $MB$.\\
             - Vẽ đường tròn tâm $O'$, bán kính $O'M$.\\
             - Đường tròn $(O')$ cắt đường tròn $(O)$ tại điểm thứ hai khác điểm $M$ chính là điểm $C$ cần xác định.
     \end{enumerate}}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Hà Nam, 2016]%[9D5G1]
 Cho $a, b, c$ là các số thực dương thỏa mãn điều kiện $a+b+c=3$. Tìm giá trị lớn nhất của biểu thức $$P=\dfrac{bc}{\sqrt{3a+bc}}+\dfrac{ca}{\sqrt{3b+ca}}+\dfrac{ab}{\sqrt{3c+ab}}$$
 \loigiai{Ta có $3a+bc=(a+b+c)a+bc=a^2+ab+ac+bc=(a+b)(a+c)$.\\
 Tương tự ta có $\heva{&3b+ca=(b+c)(b+a)\\&3c+ab=(c+a)(c+b)}$.\\
 Áp dụng bất đẳng thức Cô si cho hai số dương $\dfrac{1}{a+b}$ và $\dfrac{1}{a+c}$ ta có $\dfrac{1}{a+b}+\dfrac{1}{a+c}\geq 2\dfrac{1}{\sqrt{(a+b)(a+c)}}.$\\
 Từ đó suy ra $\sqrt{(a+b)(a+c)}\geq \dfrac{2}{\frac{1}{a+b}+\frac{1}{a+c}}\Leftrightarrow \dfrac{bc}{\sqrt{3a+bc}}\leq \dfrac{bc}{2}\left(\dfrac{1}{a+b}+\dfrac{1}{a+c}\right)$.\\
 Tương tự ta có $\heva{&\dfrac{ca}{\sqrt{3b+ca}}\leq \dfrac{ca}{2}\left(\dfrac{1}{b+c}+\dfrac{1}{b+a}\right)\\&\dfrac{ab}{\sqrt{3c+ab}}\leq \dfrac{ab}{2}\left(\dfrac{1}{c+a}+\dfrac{1}{c+b}\right)}$.\\
 Cộng vế với vế các bất đẳng thức trên ta suy ra:
\begin{align*}
  P &\leq \dfrac{bc}{2}\left(\dfrac{1}{a+b}+\dfrac{1}{a+c}\right)+\dfrac{ca}{2}\left(\dfrac{1}{b+c}+\dfrac{1}{b+a}\right)+\dfrac{ab}{2}\left(\dfrac{1}{c+a}+\dfrac{1}{c+b}\right) \\
  &\leq \dfrac{c}{2(a+b)}(a+b)+\dfrac{b}{2(a+c)}(a+c)+\dfrac{a}{2(c+b)}(c+b) \\
  &\leq \dfrac{a+b+c}{2} \\
  & \leq \dfrac{3}{2}.
\end{align*}
Dấu đẳng thức xảy ra khi và chỉ khi $a=b=c=1$.\\
Vậy $\max P=\dfrac{3}{2}$, đạt được khi $a=b=c=1$.
}
\end{ex}
