\begin{ex}%[HK1 THPT Lương Thế Vinh - Hà Nội -2018]%[0D3B2][0D3K2]
	Giải các phương trình:
	\begin{enumerate}
		\item $|x-1|=|x^2+2x|$.
		\item $\sqrt{2(x+1)}-2=\sqrt{x-1}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $|x-1|=|x^2+2x|\Leftrightarrow \hoac{& x-1=x^{2}+2x\\ & x-1=-x^{2}-2x}\Leftrightarrow \hoac{& x^{2}+x+1=0\ \left(\mbox{Vô nghiệm}\right)\\ & x^{2}+3x-1=0}\Leftrightarrow x=\dfrac{-3\pm\sqrt{13}}{2}$.
			\item Điều kiện $x\geq 1$.\\
			$\sqrt{2(x+1)}-2=\sqrt{x-1}\Leftrightarrow \sqrt{2(x+1)}=2+\sqrt{x-1}\Leftrightarrow 2(x+1)=4+x-1+4\sqrt{x-1}$\\
			$\Leftrightarrow 4\sqrt{x-1}=x-1\Leftrightarrow \heva{&x-1\geq 0\\ & 16(x-1)=(x-1)^{2}}\Leftrightarrow \hoac{&x=1\\ & x=17}$
		\end{enumerate}
	}
\end{ex}
\begin{ex}%[HK1 THPT Lương Thế Vinh - Hà Nội -2018]%[0D3B3]
	Giải hệ phương trình $\heva{&3x-2y=\dfrac{4}{x}\\ &3y-2x=\dfrac{4}{y}}$.
	\loigiai{
		Điều kiện $x\neq 0$, $y\neq0$.\\
		$\heva{&3x-2y=\dfrac{4}{x}\\ &3y-2x=\dfrac{4}{y}}\Leftrightarrow \heva{&3x^{2}-2xy=4\\ &3y^{2}-2xy=4}\Rightarrow x^{2}-y^{2}=0\Rightarrow x=y$ hoặc $x=-y$.
		\begin{itemize}
			\item Với $x=y$ ta có $3x^{2}-2x^{2}=4\Rightarrow x^{2}=4\Rightarrow x=\pm 2$ (TM)\\
			$\Rightarrow$ nghiệm của hệ là $(2;2)$ hoặc $(-2;-2)$.
			\item Với $x=-y$ ta có $-3x^{2}+2x^{2}=4\Rightarrow -x^{2}=4$ (Vô lí)
		\end{itemize}
		Vậy nghiệm của hệ là $(2;2)$ hoặc $(-2;-2)$.
	}
\end{ex}
\begin{ex}%[HK1 THPT Lương Thế Vinh - Hà Nội -2018]%[0D3B2]%[0D3K2]
	Cho phương trình $(x-2)(2x^2-2x+3m-1)=0$ \ ($1$) \ với $m\in\mathbb{R}$ là tham số.
	\begin{enumerate}
		\item Tìm $m$ để phương trình (1) nhận $x_0=3$ là một nghiệm.
		\item Tìm $m$ để phương trình (1) có ba nghiệm phân biệt, trong đó có đúng một nghiệm âm.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Với $x=3$, thay vào $(1)$ ta có $18-6+3m-1=0\Rightarrow m=-\dfrac{11}{3}$.
			\item $(1)\Leftrightarrow \hoac{&x=2\\ & 2x^2-2x+3m-1=0\ (2)}$\\
			Để $(1)$ có ba nghiệm phân biệt thì $(2)$ phải có hai nghiệm phân biệt khác $2$\\
			$\Leftrightarrow \heva{&a\neq 0\\ & \Delta >0\\ & 3m+3\neq 0}\Leftrightarrow \heva{&2\neq 0\ (\mbox{LĐ})\\ & 1-2(3m-1)>0\\ & m\neq -1}\Leftrightarrow \heva{&m<\dfrac{1}{2}\\ &m\neq -1}$.\\
			Ta xét các trường hợp:
			\begin{itemize}
				\item $(2)$ có một nghiệm âm và một nghiệm bằng $0\Rightarrow 3m-1=0\Rightarrow m=\dfrac{1}{3}$.\\
				Với $m=\dfrac{1}{3}$, ta có $(2)\Leftrightarrow 2x^{2}-2x=0\Rightarrow x=0$ hoặc $x=1$ (loại).
				\item $(2)$ có hai nghiệm trái dấu $\Leftrightarrow 2(3m-1)<0\Leftrightarrow m<\dfrac{1}{3}$.
			\end{itemize}
			Kết luận: với $m<\dfrac{1}{3}$, $m\neq -1$ thì phương trình $(1)$ có ba nghiệm phân biệt trong đó có đúng một nghiệm âm.
		\end{enumerate}
	}
\end{ex}
\begin{ex}%[HK1 THPT Lương Thế Vinh - Hà Nội -2018]%[0H2B2]
	Trong mặt phẳng tọa độ $Oxy$ cho tam giác $ABC$ có $A(2;2)$, $B(5;3)$ và $C(4;-4)$. Chứng minh rằng tam giác $ABC$ vuông và tìm tọa độ điểm $D$ sao cho bốn điểm $A$, $B$, $C$, $D$ lập thành một hình chữ nhật.
	\loigiai{
		\immini{
			Ta có $\vec{AB}=(3;1)$, $\vec{AC}=(2;-6)\Rightarrow \vec{AB}.\vec{AC}=0\Rightarrow AB\perp AC$.\\
