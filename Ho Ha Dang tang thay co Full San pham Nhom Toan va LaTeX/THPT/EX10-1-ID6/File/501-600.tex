	Hàm số nào sau đây đồng biến trên $\mathbb{R}$?
	\choice
	{$y=3$}
	{$y=\dfrac{2}{x}+4$}
	{$y=-x+5$}
	{\True $y=\dfrac{1}{2}x+3$}
	\loigiai{
		\begin{itemize}
			\item $y=3$ là hàm hằng.
			\item $y=\dfrac{2}{x}+4$ là hàm nghịch biến trên từng khoảng xác định do\\
			+) Với mọi $x_1<x_2\in (0;+\infty)$ ta có $f(x_1)-f(x_2)=\dfrac{2}{x_1}-\dfrac{2}{x_2}=\dfrac{2(x_2-x_1)}{x_1x_2}>0\Rightarrow f(x_1)>f(x_2)$, từ đó suy ra hàm số nghịch biến trên khoảng $(0;+\infty)$.\\
			+) Với mọi $x_1<x_2\in (-\infty;0)$ ta có $f(x_1)-f(x_2)=\dfrac{2}{x_1}-\dfrac{2}{x_2}=\dfrac{2(x_2-x_1)}{x_1x_2}>0\Rightarrow f(x_1)>f(x_2)$, từ đó suy ra hàm số nghịch biến trên khoảng $(-\infty;0)$.
			\item $y=-x+5$ là hàm nghịch biến do hệ số $a=-1<0$.
			\item $y=\dfrac{1}{2}x+3$ là hàm đồng biến do hệ số $a=\dfrac{1}{2}>0$.
		\end{itemize}
	}
\end{ex}
