\begin{ex}%[Thi thử L2, THPT Bình Minh, Ninh Bình]%[2D1G5-4]%[Bùi Quốc Hoàn, dự án (12EX-8)]
Biết rằng đồ thị hàm số $f\left(x\right) = ax^4 + bx^3 + cx^2 + dx + e$, ($a, b,  c,  d, e \in\mathbb{R}$; $a\neq 0$, $b\neq 0$) cắt trục hoành $Ox$ tại $4$ điểm phân biệt. Khi đó đồ thị hàm số  
$$g\left(x\right) = \left(4ax^3 + 3bx^2 + 2cx + d\right)^2 - 2\left(6ax^2 + 3bx + c\right)\cdot\left(ax^4 + bx^3 + cx^2 + dx + e\right).$$
cắt trục hoành $Ox$ tại bao nhiêu điểm?	
	\choice
	{$6$}
	{\True $0$}
	{$4$}
	{$2$}
	\loigiai{Do giả thiết suy ra phương trình $f\left(x\right) = 0$ có $4$ nghiệm phân  biệt $x_1$, $x_2$, $x_3$, $x_4$. Khi đó 
	$$f\left(x\right) = a\left(x - x_1\right)\cdot\left(x - x_2\right)\cdot\left(x - x_3\right)\cdot\left(x - x_4\right).$$
	Suy ra 
	$$f'\left(x\right) = a\sum_{i = 1}^{4}\left(\prod_{j = 1, j\neq i}^{4}\left(x - x_j\right)\right).$$
	- Nếu $x = x_1$ là nghiệm của phương trình $g(x) = 0$ suy ra $g\left(x_1\right) = 0$. Khi đó 
	$$f''\left(x_1\right)\cdot f\left(x_1\right) - f'\left(x_1 \right)\cdot f'\left(x_1\right)  = 0\Leftrightarrow f'\left(x_1 \right) = 0.$$
	mâu thuẫn nếu $x_i\neq x_1,\ \forall i\neq 1$.\\
	Chứng minh tương tự suy ra $x = x_2$, $x = x_3$, $x= x_4$ không là nghiệm của $g\left(x\right) = 0$.\\
	- Nếu  $x\neq  x_i$ với $i = 1, 2, 3, 4$ ta có  
	$$\dfrac{f'\left(x \right)}{f\left(x \right)} = \displaystyle\sum_{i = 1}^{4}\dfrac{1}{x - x_i}\Rightarrow \dfrac{f''\left(x \right)\cdot f\left(x\right) - f'\left(x \right)\cdot f'\left(x\right)}{\left(f\left(x\right)\right)^2} =  - \displaystyle\sum_{i = 1}^{4}\dfrac{1}{\left(x - x_i\right)^2}.$$
Do đó 	$\dfrac{f''\left(x \right)\cdot f\left(x\right) - f'\left(x \right)\cdot f'\left(x\right)}{\left(f\left(x\right)\right)^2} <  0\ \forall x\neq x_i$ nên phương trình  $f''\left(x \right)\cdot f\left(x\right) - f'\left(x \right)\cdot f'\left(x\right)  = 0$ vô nghiệm. Hay phương trình $g\left(x\right) = 0$ vô nghiệm.
}
\end{ex}
