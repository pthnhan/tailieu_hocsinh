	Trong các hàm số sau, hàm số nào là hàm số chẵn?
	\choice
	{$y=\dfrac{\sqrt{x^2+1}}{x-1}$}
	{$y=x|x|$}
	{$y=|x-1|$}
	{\True $y=x^2+2|x|+2$}
	\loigiai{
	\begin{itemize}
		\item Xét hàm số $y=\dfrac{\sqrt{x^2+1}}{x-1}$.\\
		Tập xác định là $\mathscr{D}=\mathbb{R}\setminus \{1\}$, do đó hàm số đã cho không phải là hàm số chẵn.
		\item Xét hàm số $y=x|x|$.
		Tập xác định là $\mathscr{D}=\mathbb{R}$.\\
		$\forall x\in\mathscr{D} \Rightarrow -x\in\mathscr{D}$;\quad $f(-x)=-x|-x|=-x|x|=-f(x)$, do đó $y=x|x|$ là hàm số lẻ.
		\item Xét hàm số $y=|x-1|$.
		Tập xác định là $\mathscr{D}=\mathbb{R}$.\\
		$\forall x\in\mathscr{D} \Rightarrow -x\in\mathscr{D}$;\quad $f(-1)=|-1-1|=2\neq f(1)=|1-1|=0$, do đó hàm số đã cho không phải là hàm chẵn.
		\item Xét hàm số $y=x^2+2|x|+2$.
		Tập xác định là $\mathscr{D}=\mathbb{R}$.\\
		$\forall x\in\mathscr{D} \Rightarrow -x\in\mathscr{D}$;\quad $f(-x)=(-x)^2+2|-x|+2=x^2+2|x|+2 =f(x)$, do đó hàm số đã cho là hàm chẵn.
	\end{itemize}	
	}
\end{ex}

\begin{bt}%[Kiểm tra GHK1, Thực Hành Sư Phạm - Cần Thơ, 2019]%[Trần Thịnh, EX-DT-10&11]%[0D2B3-3]%[0D2B3-1]%[0D2K3-4]
	Cho hàm số $f(x)=\heva{&\dfrac{1}{4}x^2 &\text{khi }& x\leq 2\\ &3-x &\text{khi }& x\geq 2.}$
	\begin{enumerate}
		\item Vẽ đồ thị hàm số $f(x)$.
		\item Lập bảng biến thiên và tìm các khoảng đồng biến, nghịch biến của hàm số $f(x)$.
		\item Dựa vào đồ thị, tìm điều kiện của tham số $m$ để phương trình $f(x)=m$ có ít nhất hai nghiệm.
	\end{enumerate}	
\loigiai{
	\begin{enumerate}
	\item
		Đồ thị hàm số $y=f(x)$ gồm hai phần:
		\immini{
		\begin{itemize}
			\item Phần parabol $y=\dfrac{1}{4}x^2$ ứng với $x\leq 2$.
			\item Phần đường thẳng $y=3-x$ ứng với $x\geq 2$.
		\end{itemize}
		Bảng giá trị
		\begin{center}
			\begin{tabular}{|c|c|c|c|c|c|}
				\hline 
				$x$ & $-4$ & $-2$	& $0$ & $2$ & $3$ \\ 
				\hline 
				$y=f(x)$ & $4$ & $1$ & $0$ & $1$ & $0$\\ 
				\hline
			\end{tabular}
		\end{center}	
		}{
		\vspace*{-0.5cm}
		\begin{tikzpicture}[>=stealth, scale=0.7, line join = round, line cap = round, font=\footnotesize]
		%\draw [dotted, step=1, black, thin, help lines](-5,-2) grid (5,5);
