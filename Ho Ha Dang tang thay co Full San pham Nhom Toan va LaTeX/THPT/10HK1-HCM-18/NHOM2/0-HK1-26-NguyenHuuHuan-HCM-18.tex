\begin{bt}%[Đề HK1, Nguyễn Hữu Huân, TP.HCM, 17-18]%[Dương Phước Sang]%[0D3B2-6]
	Tìm tất cả các giá trị của tham số $m$ để phương trình $m(mx+1)=x-1$ nghiệm đúng với mọi $x\in \mathbb{R}$.
	\loigiai{
		Ta có $m(mx+1)=x-1 \Leftrightarrow m^2x+m=x-1 \Leftrightarrow (m^2-1)x=-m-1$. \qquad$(1)$ \\
		Phương trình $(1)$ nghiệm đúng với mọi $x\in \mathbb{R} \Leftrightarrow \left\{\begin{aligned}
		&m^2-1=0 \\
		&-m-1=0
		\end{aligned}\right. \Leftrightarrow \left\{\begin{aligned}
		&m=\pm 1 \\
		&m=-1
		\end{aligned}\right. \Leftrightarrow m=-1$. \\
		Vậy $m=-1$ là giá trị tham số cần tìm.}
\end{bt} 

\begin{bt}%[Đề HK1, Nguyễn Hữu Huân, TP.HCM, 17-18]%[Dương Phước Sang]%[0D3B2-6]%[0D3K2-5]
	Cho phương trình $mx^2-2(m+2)x+m+1=0$. \qquad$(1)$ \\
	a) Tìm tất cả các giá trị của tham số $m$ để $(1)$ có hai nghiệm phân biệt $x_1$, $x_2$. \\
	b) Tính giá trị của biểu thức $A=x_1+x_2-4x_1x_2$.
	\loigiai{
		\begin{enumerate}[\bf a)]
			\item Phương trình $mx^2-2(m+2)x+m+1=0$ có $2$ nghiệm phân biệt $x_1$ và $x_2$ khi và chỉ khi \\
			$\left\{\begin{aligned}
			&a=m\ne 0 \\
			&{\Delta}'=(m+2)^2-m(m+1)>0
			\end{aligned}\right. \Leftrightarrow \left\{\begin{aligned}
			&m\ne 0 \\
			&3m+4>0
			\end{aligned}\right. \Leftrightarrow \left\{\begin{aligned}
			&m\ne 0 \\
			&m>-\dfrac{4}{3}.
			\end{aligned}\right. $
			\item Với $\left\{\begin{aligned}
			&m\ne 0 \\
			&m>-\dfrac{4}{3}
			\end{aligned}\right. $ ta có $\left\{\begin{aligned}
			&x_1+x_2=\dfrac{2(m+2)}{m} \\
			&x_1x_2=\dfrac{m+1}{m}
			\end{aligned}\right. $ $\Rightarrow A=x_1+x_2-4x_1x_2=\dfrac{2(m+2)}{m}-\dfrac{4(m+1)}{m}=-2$.
		\end{enumerate}}
\end{bt} 

\begin{bt}%[Đề HK1, Nguyễn Hữu Huân, TP.HCM, 17-18]%[Dương Phước Sang]%[0D3B2-3]
	Giải phương trình $\dfrac{x-1}{x}-\dfrac{1}{x+1}=\dfrac{2x-1}{x^2+x}$.
	\loigiai{
		Với điều kiện $\left\{\begin{aligned}
		&x\ne 0 \\
		&x\ne -1
		\end{aligned}\right. $ thì \\
		$\dfrac{x-1}{x}-\dfrac{1}{x+1}=\dfrac{2x-1}{x^2+x} \Leftrightarrow (x-1)(x+1)-x=2x-1 \Leftrightarrow x^2-3x=0 \Leftrightarrow \left[\begin{aligned}
		&x=0 \\
		&x=3.
		\end{aligned}\right. $ \\
		So với điều kiện ta nhận $x=3$ làm nghiệm duy nhất của phương trình.}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Hữu Huân, TP.HCM, 17-18]%[Dương Phước Sang]%[0D3K2-4]
	Giải phương trình $\left(|x-1|-2\right)\left(\sqrt{2x-1}-3\right)=0$.
	\loigiai{
		Với điều kiện $2x\geqslant 1$ thì $\left(|x-1|-2\right)\left(\sqrt{2x-1}-3\right)=0$ \\
		$ \Leftrightarrow \left[\begin{aligned}
		&|x-1|-2=0 \\
		&\sqrt{2x-1}-3=0
		\end{aligned}\right. \Leftrightarrow \left[\begin{aligned}
		&|x-1|=2 \\
		&\sqrt{2x-1}=3
		\end{aligned}\right. \Leftrightarrow \left[\begin{aligned}
		&x-1=2 \\
		&x-1=-2 \\
		&2x-1=9
		\end{aligned}\right. \Leftrightarrow \left[\begin{aligned}
		&x=3 \\
		&x=-1 \\
		&x=5.
		\end{aligned}\right. $ \\
		So với điều kiện ta chỉ nhận $x=3$ và $x=5$ làm các nghiệm cho phương trình.}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Hữu Huân, TP.HCM, 17-18]%[Dương Phước Sang]%[0D3B3-4]
	Giải hệ phương trình $\left\{\begin{aligned}
	&x+y=1 \\
	&x^2-3xy+y=9.
	\end{aligned}\right. $
	\loigiai{
		Xét hệ phương trình 	$\left\{\begin{aligned}
		&x+y=1&(1) \\
		&x^2-3xy+y=9&(2)
		\end{aligned}\right. $. \\
		Từ $(1)$ ta suy ra $y=1-x$, thay vào $(2)$ ta được \\
		$x^2-3x(1-x)+(1-x)=9 \Leftrightarrow 4x^2-4x-8=0 \Leftrightarrow x=-1\vee x=2$. \\
		Vậy hệ phương trình đã cho có hai nghiệm $(x;y)=(-1;2)$ và $(x;y)=(2;-1)$.}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Hữu Huân, TP.HCM, 17-18]%[Dương Phước Sang]%[0D4K1-3]
	Chứng minh rằng $\sqrt{11-a}+\sqrt{7+a}\leqslant 6,\forall a\in [-7;11]$.
	\loigiai{
		Với $a\in [-7;11]$, áp dụng bất đẳng thức Bunyakovski ta được \\
		${\left(1\cdot \sqrt{11-a}+1\cdot \sqrt{7+a}\right)}^2\leqslant (1^2+1^2)\left[{\left(\sqrt{11-a}\right)}^2+{\left(\sqrt{7+a}\right)}^2\right]=2\cdot 18=36$. \\
		Suy ra $\sqrt{11-a}+\sqrt{7+a}\leqslant 6,\forall a\in [-7;11]$. \\
		Dấu “=” xảy ra khi và chỉ khi $\sqrt{11-a}=\sqrt{7+a} \Leftrightarrow 11-a=7+a \Leftrightarrow a=2\in [-7;11]$.}
\end{bt} 

\begin{bt}%[Đề HK1, Nguyễn Hữu Huân, TP.HCM, 17-18]%[Dương Phước Sang]%[0H2K2-4]
	Trong hệ trục toạ độ $Oxy$, cho hai điểm $A(1;3)$, $B(-2;5)$. Tìm điểm $M$ trên trục hoành sao cho $AB=AM$.
	\loigiai{
		Ta có $M\in Ox \Leftrightarrow M(a;0)$ với $a\in \mathbb{R}$. Từ đó $\left\{\begin{aligned}
		&\overrightarrow{AM}=(a-1;-3) \\
		&\overrightarrow{BM}=(a+2;-5).
		\end{aligned}\right. $ \\
		Theo giả thiết $AM=BM \Leftrightarrow (a-1)^2+(-3)^2=(a+2)^2+(-5)^2 \Leftrightarrow 6a=-19 \Leftrightarrow a=-\dfrac{19}{6}$. \\
		Vậy điểm cần tìm là $M\left(-\dfrac{19}{6};0\right)$.}
\end{bt} 

\begin{bt}%[Đề HK1, Nguyễn Hữu Huân, TP.HCM, 17-18]%[Dương Phước Sang]%[0H2K2-2]
	Cho hình chữ nhật $ABCD$ có $O$ là giao điểm của hai đường chéo $AC$ và $BD$, $M$ là điểm tuỳ ý. Chứng minh rằng $4\overrightarrow{MA}\cdot \overrightarrow{MC}+BD^2=4MO^2$.
	\loigiai{
		Do $O$ là trung điểm của $AC$ nên $\overrightarrow{MA}+\overrightarrow{MC}=2\overrightarrow{MO} \Rightarrow {\left(\overrightarrow{MA}+\overrightarrow{MC}\right)}^2={\left(2\overrightarrow{MO}\right)}^2$ \\
		$ \Rightarrow MA^2+MC^2+2\overrightarrow{MA}\cdot \overrightarrow{MC}=4MO^2$. \qquad$(1)$ \\
		Lại có $\overrightarrow{MC}-\overrightarrow{MA}=\overrightarrow{AC} \Rightarrow {\left(\overrightarrow{MC}-\overrightarrow{MA}\right)}^2={\left(\overrightarrow{AC}\right)}^2$\\
		$\Rightarrow MA^2+MC^2-2\overrightarrow{MA}\cdot \overrightarrow{MC}=AC^2$.\qquad $(2)$ \\
		Từ (1) và (2), trừ vế theo vế ta được: \\
		$4\overrightarrow{MA}\cdot \overrightarrow{MC}=4MO^2-AC^2 \Rightarrow 4\overrightarrow{MA}\cdot \overrightarrow{MC}+BD^2=4MO^2$ (do $AC^2=BD^2$ ).}
\end{bt} 

\begin{bt}%[Đề HK1, Nguyễn Hữu Huân, TP.HCM, 17-18]%[Dương Phước Sang]%[0H2B3-1]%[0H2G3-1]
	Cho tam giác $ABC$ có $A=120^{\circ}$, $BC=8$.
	\begin{enumerate}
		\item Tìm bán kính đường tròn ngoại tiếp tam giác $ABC$.
		\item Gọi $D$ là tâm đường tròn nội tiếp tam giác $ABC$ và $E$ là tâm đường tròn ngoại tiếp tam giác $BCD$. Tính bán kính đường tròn ngoại tiếp tam giác $BCE$.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[line cap=round,line join=round,>=stealth,scale=1]
				\tkzDefPoints{0/0/B, 6/0/C, 1.5/4/A}
				\clip (-1,4.5) rectangle (7,-2.5);
				\tkzDrawPolygon(A,B,C)
				\tkzInCenter(A,B,C)\tkzGetPoint{D}
				\tkzDrawCircle[in](A,B,C)
				\tkzCircumCenter(D,B,C)\tkzGetPoint{E}
				\tkzDrawCircle[circum](D,B,C)
				\tkzDrawPolygon(B,E,C,D)
				\tkzDrawSegment(D,E)
				\tkzDrawPoints[fill=black, size=3pt](A,B,C,D,E)
				\tkzLabelPoints[above](A,D)
				\tkzLabelPoints[left](B)
				\tkzLabelPoints[right](C)
				\tkzLabelPoints[below](E)
				\tkzMarkAngle[size=0.5](D,B,A)
				\tkzMarkAngle[size=0.55](C,B,D)
				\tkzMarkAngle[size=0.6](A,C,D)
				\tkzMarkAngle[size=0.65](A,C,D)
				\tkzMarkAngle[size=0.55](D,C,B)
				\tkzMarkAngle[size=0.61](D,C,B)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Gọi $ R$ là bán kính đường tròn ngoại tiếp tam giác $ABC$. \\
			Áp dụng định lý sin ta có $\dfrac{BC}{\sin A}=2R \Rightarrow R=\dfrac{BC}{2\sin A}=\dfrac{8}{2\sin 120^{\circ}}=\dfrac{8}{\sqrt{3}}$.
			\item \textbf{\underline{Cách 1}:}\\
			Từ giả thiết ta suy ra $\triangle EBD$ cân tại $E \Rightarrow \widehat{BDE}=\widehat{DBE}=\widehat{DBC}+\widehat{CBE}$. \\
			Mà $\widehat{ABD}=\widehat{DBC}$ nên $\widehat{ABD}+\widehat{CBE}=\widehat{BDE}$.\qquad\qquad $(1)$ \\
			Hoàn toàn tương tự ta có $\widehat{ACD}+\widehat{BCE}=\widehat{CDE}$.\qquad$(2)$ \\
			Ngoài ra 	$\widehat{ABE}+\widehat{ACE}=\left(\widehat{ABD}+\widehat{CBE}\right)+\left(\widehat{ACD}+\widehat{BCE}\right)+\widehat{DBC}+\widehat{DCB}$. \qquad $(3)$ \\
			Kết hợp $(1)$, $(2)$ và $(3)$ ta được $\widehat{ABE}+\widehat{ACE}=\widehat{BDE}+\widehat{CDE}+\widehat{DBC}+\widehat{DCB}=180^\circ$. \\
			\textbf{\underline{Cách 2}:}\\
			Xét đường tròn $(E)$, ta có $\left\{\begin{aligned}
			&\text{sđ}\overset\frown{BD}=2\widehat{BCD} \\
			&\text{sđ}\overset\frown{CD}=2\widehat{CBD}
			\end{aligned}\right. \Rightarrow \left\{\begin{aligned}
			&\widehat{BED}=\widehat{BCA} \\
			&\widehat{CED}=\widehat{ABC}
			\end{aligned}\right. \Rightarrow \widehat{BEC}=\widehat{BCA}+\widehat{ABC}$\\
			$\Rightarrow \widehat{BEC}+\widehat{BAC}=\widehat{BCA}+\widehat{ABC}+\widehat{BAC}=180^{\circ}$.\\
			Như vậy tứ giác $ABEC$ nội tiếp được đường tròn, điều đó cũng có nghĩa là đường tròn ngoại tiếp tam giác $BCE$ cũng chính là đường tròn ngoại tiếp tam giác $ABC$. Bán kính đường tròn đó là $R=\dfrac{8}{\sqrt{3}}$.
		\end{enumerate}}
\end{bt} 

\begin{bt}%[Đề HK1, Nguyễn Hữu Huân, TP.HCM, 17-18]%[Dương Phước Sang]%[0H2T2-1]
	Một chiếc xe chuyển động từ $A$ đến $B$ dưới tác dụng của hai lực ${\overrightarrow{F}}_1$ và ${\overrightarrow{F}}_2$. Lực ${\overrightarrow{F}}_1$ vuông góc với chiều chuyển động, lực ${\overrightarrow{F}}_2$ cùng hướng với chiều chuyển động. Gọi $\overrightarrow{F}$ là hợp lực của ${\overrightarrow{F}}_1$ và ${\overrightarrow{F}}_2$, $\overrightarrow{F}$ tạo với ${\overrightarrow{F}}_2$ một góc $\alpha~\left(0^{\circ}<\alpha <90^{\circ}\right)$ (như hình vẽ).
	\begin{center}
		\begin{tikzpicture}[line cap=round,line join=round,>=stealth,scale=1]
			\def \r{4pt}
			\tkzDefPoints{0/0/A, 6/0/B,-1.4/0/o}
			\coordinate (l) at ($(o)+(0,0.65)$);
			\coordinate (x) at ($(A)!0.65!(B)$);
			\coordinate (y) at ($(A)+(0,2)$);
			\coordinate (f) at ($(x)+(y)-(A)$);
			\coordinate (i) at ($(o)!0.25!(A)-(0,\r)$);
			\coordinate (j) at ($(A)!0.25!(o)-(0,\r)$);
			\draw (i) circle(\r);\draw (j) circle(\r);
			\draw[->] (x)--(B) (A)--(x);\draw[->] (A)--(y);\draw[->](A)--(f);\draw[->](A)--(B);
			\draw[dashed] (x)--(f)--(y);
			\draw (A) rectangle (l);
			\draw ($(i)-(2,\r)$)--($(B)-(-3,2*\r)$);
			\tkzMarkAngle[size=0.6](B,A,f)
			\node at ($(A)+(0.8,0.2)$) {$\alpha$};
			\node at ($(y)!0.2!(A)-(0.3,0)$) {$\overrightarrow{F}_1$};
			\node at ($(x)!0.2!(A)+(0,0.3)$) {$\overrightarrow{F}_2$};
			\node at ($(f)!0.4!(A)+(0,0.4)$) {$\overrightarrow{F}$};
			\node at ($(A)+(0.2,0.35)$) {$A$};
			\node at (B)[above] {$B$};
		\end{tikzpicture}
	\end{center}
	Hãy tính công của lực $\overrightarrow{F}$ khi xe di chuyển từ $A$ đến $B$ biết công của lực $\overrightarrow{F}$ được tính bằng $\mathrm{A}=\overrightarrow{F}\cdot \overrightarrow{AB}$ (jun). Cho biết $AB=1000$ m, độ lớn của lực ${\overrightarrow{F}}_1$ và ${\overrightarrow{F}}_2$ lần lượt là $50$ N và $70$ N.
	\loigiai{
		Công của lực $\overrightarrow{F}$ là $\mathrm{A}=\overrightarrow{F}\cdot \overrightarrow{AB}=\left|\overrightarrow{AB}\right|\cdot \left|\overrightarrow{F}\right|\cdot \cos \alpha =\left|\overrightarrow{AB}\right|\cdot \left|{\overrightarrow{F}}_2\right|=1000\cdot 70=70000$ jun.}
\end{bt}
