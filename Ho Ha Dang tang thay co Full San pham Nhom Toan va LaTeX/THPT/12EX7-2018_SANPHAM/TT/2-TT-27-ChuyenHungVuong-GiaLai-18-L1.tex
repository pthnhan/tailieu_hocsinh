\item [(III)] Nếu $f'(x) \le 0$, $\forall x \in I$ thì hàm số $f$ nghịch biến trên khoảng $I$.
\item [(IV)] Nếu  $f'(x) \le 0$, $\forall x \in I$ và $f'(x) = 0$  tại vô số điểm trên $I$  thì hàm số $f$  không thể nghịch biến trên khoảng $I$.
\end{itemize}
Trong các mệnh đề trên, mệnh đề nào đúng, mệnh đề nào sai?
\choice
{\True  (I) và (II)  đúng, còn (III)  và (IV)  sai}
{(I), (II)   và (III)  đúng, còn (IV)  sai}
{(I), (II)   và  (IV) đúng, còn  (III) sai}
{Cả  (I), (II), (III)  và  (IV) đúng}
\loigiai{Dễ thấy (I) và (II)  đúng, còn (III)  và (IV)  sai.}
\end{ex}
\begin{ex}%[Thi thử L1, chuyên Hùng Vương, Gia Lai 2018]%[2D1B2-1]%[Nguyễn Tài Chung, 12EX-7]
Cho hàm số $y=f(x)$  có đạo hàm trên $\mathbb{R}$. Xét tính đúng sai của các mệnh đề sau:
\begin{itemize}
\item [(I)] Nếu $f'(x) > 0$  trên khoảng $\left( x_0 - h;x_0 \right)$ và $f'(x) < 0$ trên khoảng $\left(x_0;x_0+h\right)$ (với $h>0$) thì hàm số đạt cực đại tại điểm $x_0$.
\item [(II)] Nếu hàm số đạt cực đại tại điểm $x_0$ thì tồn tại các khoảng $\left(x_0-h;x_0\right),\left(x_0;x_0+h\right)$ (với $h>0$) sao cho $f'(x)>0$ trên khoảng $\left(x_0-h;x_0\right)$ và $f'(x)<0$ trên khoảng $\left(x_0;x_0+h\right)$.
\end{itemize}
\choice
{Cả (I) và (II) cùng sai}
{\True Mệnh đề (I) đúng, mệnh đề (II) sai}
{Mệnh đề (I) sai, mệnh đề (II) đúng}
{Cả (I) và (II) cùng đúng}
\loigiai{Dễ thấy (I) đúng. Ta sẽ chứng minh đề (II) sai. Xét hàm số 
		$$f(x) = \left\{ \begin{array}{ll}
		2 - x^2\left( 2 + \sin \dfrac{1}{x} \right)&\text{khi}\,\, x \ne 0\\
		2&\text{khi}\,\, x = 0.
		\end{array} \right.$$
		Khi $x\ne 0$, ta có 
		$$f(x)=2-x^2\left(2+\sin \dfrac{1}{x}\right)\Rightarrow f'(x)=-4x-2x\sin \dfrac{1}{x}+\cos \dfrac{1}{x}.$$
		Ta có 
		$$\mathop {\lim }\limits_{x \to 0} \frac{f(x) - f(0)}{x - 0} = \mathop {\lim }\limits_{x \to 0} \dfrac{ - x^2\left( 2 + \sin \dfrac{1}{x} \right)}{x} =  - \mathop {\lim }\limits_{x \to 0} x\left( {2 + \sin \dfrac{1}{x}} \right).$$
		Mà $\mathop {\lim }\limits_{x \to 0} 2x = 0$  và $\left|{x\sin \dfrac{1}{x}}\right|\leqslant \left|x\right|$ nên $\mathop {\lim }\limits_{x \to 0} \left( {x\sin \dfrac{1}{x}} \right) = 0$. \\
		Do đó $\displaystyle \mathop {\lim }\limits_{x \to 0} \frac{f(x) - f(0)}{x - 0} = 0 \Rightarrow f'(0) = 0.$ 
		Tóm lại: 
		$$f'(x) = \left\{ \begin{array}{ll}
		- 4x - 2x\sin \dfrac{1}{x} + \cos \dfrac{1}{x}&\text{khi}\,\, x \ne 0\\
		0&\text{khi}\,\, x = 0.
		\end{array} \right.$$
		Ta có $f(x)\leqslant 2=f(0),\forall x\in \mathbb{R}$, dấu “=” xảy ra khi và chỉ khi $x=2$, do đó hàm số đạt cực đại tại điểm $x_0=0$.  Giả sử tồn tại $\varepsilon >0$ sao cho \[f'(x)>0,\forall x\in \left(-\varepsilon;0\right); \,\, f'(x)<0,\forall x\in \left(0;\varepsilon \right).\tag{*}\]
		Do $\displaystyle \mathop {\lim }\limits_{x \to 0} ( - 4x) = 0,\,\mathop {\lim }\limits_{x \to 0} \left(  - 2x\sin \frac{1}{x} \right) = 0 \Rightarrow \mathop {\lim }\limits_{x \to 0} \left(  - 4x - 2x\sin \frac{1}{x} \right) = 0$ nên với $\left|x\right|$ đủ nhỏ thì 
		$$\left|{-4x-2x\sin \dfrac{1}{x}}\right|<1.$$
		Với số nguyên dương $k$ đủ lớn ta có 
		$$x=-\dfrac{1}{(2k+1)\pi }\in \left(-\varepsilon ;0\right), \,\,  x=\dfrac{1}{2k\pi }\in \left({0;\varepsilon }\right)\,\,\text{($\left|x\right|$ đủ nhỏ).}$$
		Ta có $f'\left(-\dfrac{1}{(2k+1)\pi }\right)<0$ và $f'\left(\dfrac{1}{2k\pi }\right)>0$, mâu thuẫn với $(*)$, do đó không tồn tại số $\varepsilon >0$ như đã nói ở trên. Như vậy, tồn tại hàm số 
		$$f(x) = \left\{ \begin{array}{ll}
		2 - {x^2}\left( 2 + \sin \dfrac{1}{x} \right)&\text{khi}\,\, x \ne 0\\
		2&\text{khi}\,\, x = 0
		\end{array} \right.$$
		có đạo hàm trên $\mathbb{R}$, đạt cực đại tại $x_0=0$ nhưng không thể tồn tại  khoảng $\left(x_0-h;x_0\right)$, $\left(x_0;x_0+h\right)$ ($h>0$) sao cho $f'(x)>0$ trên khoảng $\left(x_0-h;x_0\right)$ và $f'(x)<0$ trên khoảng $\left(x_0;x_0+h\right)$.}
\end{ex}

\begin{ex}%[Thi thử L1, chuyên Hùng Vương, Gia Lai 2018]%[2D1G5-4]%[Nguyễn Tài Chung, 12EX-7]
Cho hàm số đa thức bậc ba $y=f(x)$ có đồ thị đi qua các điểm $A(2;4)$, $B(3;9)$, $C(4;16)$. Các đường thẳng $AB, AC, BC$ lại cắt đồ thị tại lần lượt tại các điểm $D$, $E$, $F$ ($D$ khác $A$ và $B$; $E$ khác $A$ và $C$; $F$ khác $B$ và $C$). Biết rằng tổng các hoành độ của $D$, $E$, $F$ bằng $24$. Tính $f(0)$.
\choice
{$-2$}
{$0$}
{$2$}
{\True $\dfrac{24}{5}$}
\loigiai{Giải sử $f(x)=a(x-2)(x-3)(x-4)+x^2$ ($a\ne 0$). Ta có
	$$AB:y=5x-6;AC:y=6x-8;BC:y=7x-12.$$
	Hoành độ điểm $D$ là nghiệm của phương trình
	{\allowdisplaybreaks
	\begin{align*}
	&a\left({x-2}\right)\left({x-3}\right)\left({x-4}\right)=-x^2+5x-6\\
	\Leftrightarrow& a\left({x-2}\right)\left({x-3}\right)\left({x-4}\right)=-\left({x-2}\right)\left({x-3}\right)\\
	\Rightarrow& a(x-4)=-1\Rightarrow x=-\dfrac{1}{a}+4.
\end{align*}}Hoành độ điểm $E$ là nghiệm của phương trình
{\allowdisplaybreaks
	\begin{align*}
	&a\left({x-2}\right)\left({x-3}\right)\left({x-4}\right)=-x^2+6x-8\\
	\Leftrightarrow& a\left({x-2}\right)\left({x-3}\right)\left({x-4}\right)=-\left({x-2}\right)\left({x-4}\right)\\
	\Rightarrow& a(x-3)=-1\Rightarrow x=-\dfrac{1}{a}+3.
	\end{align*}}Hoành độ điểm $F$ là nghiệm của phương trình:
	{\allowdisplaybreaks
		\begin{align*}
	&a\left({x-2}\right)\left({x-3}\right)\left({x-4}\right)=-x^2+7x-12\\
	\Leftrightarrow& a\left({x-2}\right)\left({x-3}\right)\left({x-4}\right)=-\left({x-3}\right)\left({x-4}\right)\\
	\Rightarrow& a(x-2)=-1\Rightarrow x=-\dfrac{1}{a}+2.
	\end{align*}}Theo giả thiết ta có $$-\dfrac{1}{a}+2-\dfrac{1}{a}+3+-\dfrac{1}{a}+4=24\Leftrightarrow -\dfrac{3}{a}=15\Leftrightarrow a=-\dfrac{1}{5}.$$ 
	Do đó $f(0)=a\left({-2}\right)\left({-3}\right)\left({-4}\right)=\dfrac{24}{5}$.}
\end{ex}
