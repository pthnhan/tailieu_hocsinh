\begin{bt}%[HK2 (2016-2017), Hà Nam]%[0D5K5]%[Trần Nhân Kiệt]
Cho bất phương trình $ mx^2-2(m-2)x+m-3>0. (*) $
		\begin{enumerate}
		\item Giải bất phương trình $ (*) $ với $ m=1. $
		\item Tìm tất cả các giá trị của tham số thực $ m $ để bất phương trình $ (*) $ nghiệm đúng với mọi $ x\in \Bbb{R}. $
		\end{enumerate}
	\loigiai{
		\begin{enumerate}
		\item Với $ m=1\Rightarrow  x^2+2x-2>0\Leftrightarrow x\in (-\infty;-1-\sqrt{3})\cup (-1+\sqrt{3};+\infty).$\\
Tập nghiệm của bất phương trình là $ S= (-\infty;-1-\sqrt{3})\cup (-1+\sqrt{3};+\infty).$
 		\item $ (*) $ đúng với mọi $ x\in \Bbb{R}\Leftrightarrow \heva{&m>0\\&\Delta'<0}\Leftrightarrow \heva{&m>0\\&(m-2)^2-m(m-3)<0} $\\
$ \Leftrightarrow \heva{&m>0\\&-m+4<0}\Leftrightarrow\heva{&m>0\\&m>4}\Leftrightarrow m>4. $\\
Vậy $ m\in (4;+\infty) $ thỏa mãi bài.
		\end{enumerate}
	}
\end{bt}

tam giác $ ABC $ theo $ r. $
	\loigiai{
Áp dụng bất đẳng thức Cô-si $ (p-a)(p-b)\le \left(\dfrac{p-a+p-b}{2}\right)^2\Leftrightarrow (p-a)(p-b)\le \dfrac{c^2}{4}. $\\
Chứng minh tương tự ta có $ (p-b)(p-c)\le \dfrac{a^2}{4} $ và $ (p-a)(p-c)\le \dfrac{b^2}{4}. $\\
Suy ra $ (p-a)^2(p-b)^2(p-c)^2\le \dfrac{a^2b^2c^2}{64}\Leftrightarrow (p-a)(p-b)(p-c)\le \dfrac{abc}{8}$\\
$ \Rightarrow p(p-a)(p-b)(p-c)\le p\cdot \dfrac{abc}{8} \,\,(1).$\\
Mặt khác áp dụng bất đẳng thức Cô-si $ abc\le \left(\dfrac{a+b+c}{3}\right)^3\Leftrightarrow abc\le \dfrac{8p^3}{27}\Leftrightarrow \dfrac{abc}{8}\le \dfrac{p^3}{27}. $\\
$(1) \Rightarrow p(p-a)(p-b)(p-c)\le p\cdot \dfrac{p^3}{27}$ \\
$\Leftrightarrow S^2\le \dfrac{1}{27}\cdot \dfrac{S^4}{r^4}$ (vì $ S=\sqrt{p(p-a)(p-b)(p-c)}=pr $) \\
 $\Leftrightarrow S\ge 3\sqrt{3}r^2. $\\
Vì bán kính $ r $ không đổi nên diện tích $ S $ nhỏ nhất bằng $ 3\sqrt{3}r^2. $ Dấu $ ``=" $ xảy ra khi $ a=b=c$  $\Leftrightarrow \triangle ABC $ là tam giác đều.


	}
