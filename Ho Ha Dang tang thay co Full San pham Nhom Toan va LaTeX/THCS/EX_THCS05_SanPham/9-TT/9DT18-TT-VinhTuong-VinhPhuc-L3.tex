$\left\lbrace \begin{aligned}&x - y = 2 \\ &mx - y = 3 \end{aligned}\right.$ $\Leftrightarrow$ $\left\lbrace \begin{aligned}&x = y + 2 \\ &(m - 1)y = - 2m + 3 \end{aligned}\right.$ $\Leftrightarrow$ $\left\lbrace \begin{aligned} &x = \dfrac{1}{m - 1} \\ &y = \dfrac{ - 2m + 3}{m - 1} \end{aligned}\right.$.\medskip

\noindent Để nghiệm thu được có biểu diễn trên trục tọa độ $Oxy$ thuộc góc phần tư thứ nhất thì \medskip

$\left\lbrace \begin{aligned}& x > 0\\& y > 0 \end{aligned}\right.$ $\Leftrightarrow$ $\left\lbrace \begin{aligned}&\dfrac{1}{m - 1} > 0 \\ &\dfrac{-2m + 3}{m - 1} > 0\end{aligned}\right.$ $\Leftrightarrow$ $\left\lbrace \begin{aligned}&m > 1 \\ &-2m + 3 > 0\end{aligned}\right.$ $\Rightarrow$ $1 < m < \dfrac{3}{2}$.

\begin{ex} %[Đề KSCL Lần 3 năm học 2017-2018, Vĩnh Tường, Vĩnh Phúc]%[9D1B7]%[9D1B8]
\begin{enumerate}\item Tính $A = \left(\sqrt{10} - \sqrt{2} \right) \sqrt{3 + \sqrt{5}}$.
\item Rút gọn biểu thức $B = \left( \dfrac{\sqrt{x}}{\sqrt{x} + 4} + \dfrac{4}{\sqrt{x} - 4}\right) \div \dfrac{x + 16}{\sqrt{x} + 2} $ (với $x \geq 0; x \neq 16$).
\end{enumerate}
\loigiai{\begin{enumerate}
\item Ta có\\
$\begin{aligned}
A &= \sqrt{2} \left( \sqrt{5} - 1\right) \sqrt{ 3 + \sqrt{5}}= \left( \sqrt{5} - 1\right) \sqrt{ 6 + 2\sqrt{5}} = \left(\sqrt{5} - 1\right)\sqrt{\left(\sqrt{5} + 1\right)^2}\\
&= \left(\sqrt{5} - 1\right) \left(\sqrt{5} + 1 \right) = 4. \end{aligned}$\smallskip

Vậy $A = 4$.
\item Với $x \geq 0; x \neq 16$, ta có\\
$\begin{aligned}
B &= \left( \dfrac{\sqrt{x}}{\sqrt{x} + 4} + \dfrac{4}{\sqrt{x} - 4}\right) \div \dfrac{x + 16}{\sqrt{x} + 2} = \dfrac{x - 4\sqrt{x} + 4\sqrt{x} + 16}{x - 16} \div \dfrac{x + 16}{\sqrt{x} + 2} \\
&= \dfrac{x + 16}{x - 16} \div \dfrac{x + 16}{\sqrt{x} + 2} = \dfrac{\sqrt{x} + 2}{x  - 16}.\end{aligned}$ \smallskip

Vậy $B = \dfrac{\sqrt{x} + 2}{x  - 16}$.
\end{enumerate}
}
\end{ex}


\begin{ex}%[Đề KSCL Lần 3 năm học 2017-2018, Vĩnh Tường, Vĩnh Phúc]%[9D3B2]%[9D3K3]
Cho hệ phương trình $\left\lbrace \begin{aligned}&3x - y = 2m - 1\\ &x + 2y = 3m + 2 \end{aligned}\right.$ \hspace{20pt} (I).
\begin{enumerate}
\item Giải hệ phương trình (I) khi $m = 1$. 
\item Tìm $m$ để hệ phương trình (I) có nghiệm $(x; y)$ thỏa mãn $x^2 + y^2 = 5$.
\end{enumerate}
\loigiai{\begin{enumerate}
\item Khi $m = 1$ ta có hệ (I) tương đương với \smallskip

$\left\lbrace \begin{aligned}&3x - y = 1\\ &x + 2y = 5 \end{aligned}\right.$ $\Leftrightarrow$ $\left\lbrace \begin{aligned}&3y = 3x - 1\\ &x + 6x - 2 = 5 \end{aligned}\right.$ $\Leftrightarrow  $ $\left\lbrace \begin{aligned}&y = 2\\ &x = 1 \end{aligned}\right.$. \smallskip
 
\noindent Vậy nghiệm của hệ khi $m = 1$ là $(x; y) = (1; 2)$.
 
\item $\left\lbrace \begin{aligned}&3x - y = 2m - 1\\ &x + 2y = 3m + 2 \end{aligned}\right.$ $\Leftrightarrow$ $\left\lbrace \begin{aligned}&y = 3x - 2m + 1\\ &x + 6x - 4m + 2 = 3m + 2 \end{aligned}\right.$ $\Leftrightarrow$ $\left\lbrace \begin{aligned}&y = 3x - 2m + 1\\ &7x = 7m \end{aligned}\right.$ $\Leftrightarrow$ $\left\lbrace \begin{aligned}&y = m + 1\\ &x = m \end{aligned}\right.$.\smallskip

Khi đó ta có $x^2 + y^2 = m^2 + (m  + 1)^2 = 2m^2 + 2m + 1$. Để $x^2 + y^2 = 5$ thì $2m^2 + 2m + 1 = 5$ $\Leftrightarrow$ $m^2 + m - 2 = 0$ $\Leftrightarrow$ $m = 1$ hoặc $m = -2$.\smallskip

Vậy với $m = 1$ hoặc $m = -2$ thì hệ (I) có nghiệm thỏa mãn $x^2 + y^2 = 5$.

\end{enumerate}

}
\end{ex}


\begin{ex}%[Đề KSCL Lần 3 năm học 2017-2018, Vĩnh Tường, Vĩnh Phúc]%[9D3B5]
Tháng giêng hai tổ sản xuất được $900$ chi tiết máy; tháng hai do cải tiến kỹ thuật tổ I vượt mức $15\%$ và tổ II vượt mức $10\%$ so với tháng giêng, vì vậy hai tổ đã sản xuất được $1010$ chi tiết máy. Hỏi tháng giêng mỗi tổ sản xuất được bao nhiêu chi tiết máy?
\loigiai{Gọi số sản phẩm tổ I sản xuất được trong tháng giêng là $x$ (sản phẩm) $\Rightarrow$ số sản phẩm tổ II sản xuất được trong tháng giêng là $900 - x$ (sản phẩm). \smallskip

\noindent Trong tháng hai tổ I sản xuất được $x + \dfrac{15}{100}x$ (chi tiết máy), tổ II sản xuất được $900 - x + \dfrac{10}{100}(900 - x)$ (chi tiết máy). Theo đề bài, trong tháng hai tổng số sản phẩm hai tổ sản xuất được là $1010$, do đó
$$ \begin{array}{lccl}
&x +  \dfrac{3}{20}x + 900 - x + \dfrac{1}{10} (900 - x) & = &1010\\
\Leftrightarrow & \dfrac{1}{20}x &= &20\\
\Leftrightarrow &x &= &400.
\end{array} $$
Vậy trong tháng giêng tổ I sản xuất được $400$ chi tiết máy, tổ II sản xuất được $500$ chi tiết máy.}
\end{ex}

$\left\lbrace \begin{aligned}&\text{góc } \widehat{A} \text{ chung}\\&\widehat{ANE} = \widehat{NCA} \text{ (cùng bằng số đo cung nhỏ } NE)  \end{aligned}\right. \Rightarrow \Delta AEN \backsim \Delta ANC \text{ (g.g)}.$

Cho $\left(x + \sqrt{1 + y^2} \right)\left(y + \sqrt{1 + x^2}\right) = 1$. Tính giá trị của biểu thức $A = x + y$.
\loigiai{Ta có\smallskip

$\left(x - \sqrt{1 + y^2}\right) \left(y - \sqrt{1 + x^2}\right) + \left(x + \sqrt{1 + y^2}\right) \left(y + \sqrt{1 + x^2}\right) = 2xy + 2\sqrt{\left(1 + x^2\right)\left(1 + y^2\right)}$. \smallskip

\noindent Suy ra, \begin{align*}\left(x - \sqrt{1 + y^2}\right) \left(y - \sqrt{1 + x^2}\right) & = 2xy + 2\sqrt{\left(1 + x^2 \right) \left(1 + y^2\right)} - \left(x + \sqrt{1 + y^2}\right)\left(y + \sqrt{1 + x^2}\right)\\
& = 2xy + 2\sqrt{\left(1 + x^2 \right)\left(1 + y^2\right)} - 1. \tag{1}  \end{align*}
\noindent Lại có $\left(x + \sqrt{1 + y^2} \right) \left(y + \sqrt{1 + x^2}\right) = 1$. \hspace{\fill} ($2$)\smallskip

\noindent Nhân ($1$) với ($2$), vế với vế, ta được :\\
$\begin{array}{lrcl} &\left(x^2 - 1 - y^2\right)\left(y^2 - 1 - x^2\right) & = &2xy + 2\sqrt{\left(1 + x^2 \right)\left(1 + y^2\right)} - 1\\
\Leftrightarrow & 1 - \left(x^2 - y^2\right)^2 & = &2xy + 2\sqrt{\left(1 + x^2 \right)\left( 1 + y^2\right)} - 1 \\
\Leftrightarrow & 2 \left(1 - xy\right) & = & 2\sqrt{\left(1 + x^2\right)\left(1 + y^2\right)} + \left( x^2 - y^2\right)^2 \\
\Leftrightarrow & 2 \left(1 - xy \right) & = & 2\sqrt{ \left(1 - xy\right)^2 + \left(x + y\right)^2} + \left( x^2 - y^2\right)^2 \geq 2\left(1 - xy\right).
\end{array} $\\
\noindent Dấu $``="$ xảy ra khi và chỉ khi $x + y = 0$. \smallskip

\noindent Vậy $A = 0$.  
}
\end{ex}
