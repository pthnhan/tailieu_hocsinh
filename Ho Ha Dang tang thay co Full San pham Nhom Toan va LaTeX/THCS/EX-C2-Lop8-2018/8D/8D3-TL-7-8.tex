\chapter{Phương trình bậc nhất một ẩn}
\section{Khái niệm về phương trình. Phương trình bậc nhất}
\begin{enumEX}{1}
\item Ta gọi hệ thức dạng $A(x)=B(x)$ là phương trình với ẩn $x$. Giải phương trình $A(x)=B(x)$ là tìm mọi giá trị của $x$ để các giá trị tương ứng của hai biểu thức $A(x)$ và $B(x)$ bằng nhau.\\
Tập hợp các giá trị đó gọi là tập nghiệm của phương trình đã cho, và thường được ký hiệu là $S$.
\item Hai phương trình được gọi là tương đương nếu chúng có cùng một tập nghiệm.
\item Khi giải một phương trình, ta có thể\\
$\bullet$ Chuyển một hạng tử từ vế này sang vế kia và đổi dấu hạng tử đó.\\
$\bullet$ Nhân (hoặc chia) cả hai vế với cùng một số khác $0$.\\
Khi đó phương trình mới tương đương với phương trình đã cho.
\item Phương trình bậc nhất là phương trình có dạng $ax+b=0$ trong đó $x$ là ẩn số, $a$ và $b$ là các số đã cho, $a\ne 0$.\\
Khi giải phương trình có hệ số chữ trong mục này, ta cũng xét các phương trình có dạng $ax+b=0$ trong đó $a=0$.
\end{enumEX}

\begin{vd}%[EX-C2-L8-2018, Lê Văn Thiện]%[8D3K1]
	Giải phương trình sau, với $a$ là hằng số (ta còn gọi $a$ là tham số)
	\[a(ax+1)=x(a+2)+2\]
	\loigiai{
	Biến đổi phương trình đã cho thành
	\begin{align*}
		& a^2x-ax-2x=2-x\\
		\Leftrightarrow &  x(a^2-s-2)=2-a\\
		\Leftrightarrow & (a+1)(a-2)x=2-a. \qquad (1)
	\end{align*}
	Ký hiệu $S$ là tập nghiệm của phương trình đã cho, ta có\\
	Nếu $a\ne -1$, $a\ne 2$ thì $S=\left\{-\dfrac{1}{a+1}\right\}$.\\
	Nếu $a=-1$ thì $(1)$ có dạng $0x=3$, vô nghiệm, $S=\varnothing$.\\
	Nếu $a=2$ thì $(1)$ có dạng $0x=0$, phương trình nghiệm đúng với mọi $x$, $S=R$.
}
\end{vd}


\begin{vd}%[EX-C2-L8-2018, Lê Văn Thiện]%[8D3K1]
Giải phương trình với $a$ là tham số
\begin{eqnarray*}
	\dfrac{x-a}{3}=\dfrac{a+3}{a}-2		\qquad \qquad (1)
\end{eqnarray*}
\loigiai{
Điều kiện xác định của phương trình là $a\ne 0$.\\
Biến đổi phương trình
\begin{align*}
	(1)&\Leftrightarrow a(x-a)=3(x+3)-6a\\
		&\Leftrightarrow ax-a^2=3x+9-6a\\
		&\Leftrightarrow ax-3x=a^2-6a+9\\
		&\Leftrightarrow (a-3)x =(a-3)^2 \qquad\qquad (2)
\end{align*}
Nếu $a\ne 3$, phương trình có nghiệm $x=a-3$.\\
Nếu $a=3$ thì $(2)$ có dạng $0x=0$, suy ra với mọi $x$ đều là nghiệm.\\
Kết luận\\
Nếu $a\ne 0$; $a\ne 3$ thì $(1)$ có một nghiệm $x=a-3$.\\
Nếu $a=3$ thì $(1)$ nghiệm đúng với mọi $x$.\\
Nếu $a=0$ thì $(1)$ vô nghiệm.
}
\end{vd}

\begin{vd}%[EX-C2-L8-2018, Lê Văn Thiện]%[8D3K1]
	Chứng minh rằng tồn tại các hằng số $a,b,c$ để phương trình sau có vô số nghiệm
	\begin{eqnarray*}
		\dfrac{x-ab}{a+b}+\dfrac{x-ac}{a+c}+\dfrac{x-bc}{b+c}=a+b+c	\qquad\qquad (1)
	\end{eqnarray*}
\loigiai{Điều kiện xác định của phương trình
	\begin{eqnarray*}
		a+b\ne 0; ~ a+c\ne 0;~b+c\ne 0
	\end{eqnarray*}
Khi đó,
\begin{align*}
	(1) &\Leftrightarrow \left(\dfrac{x-ab}{a+b}-c \right) +\left(\dfrac{x-ac}{a+c}-b \right) + \left( \dfrac{x-bc}{b+c}-a\right)=0\\
	 &\Leftrightarrow \dfrac{x-ab-ac-bc}{a+b}+\dfrac{x-ac-ab-bc}{a+c}+\dfrac{x-bc-ab-ac}{b+c}=0\\
	 &\Leftrightarrow (x-ab-ac-bc)\left(\dfrac{1}{a+b}+\dfrac{1}{a+c} +\dfrac{1}{b+c}\right) =0.
\end{align*}
$(1)$ có vô số nghiệm $\Leftrightarrow \dfrac{1}{a+b}+\dfrac{1}{a+c} +\dfrac{1}{b+c} =0.$ \qquad\qquad $(2)$.\\
Chẳng hạn ta chọn $a=1$; $b=1$, để $(2)$ xảy ra ta chọn $c$ sao cho 
\begin{eqnarray*}
	\dfrac{1}{2}+\dfrac{1}{1+c}+\dfrac{1}{1+c}=0 \Leftrightarrow \dfrac{2}{1+c}= \dfrac{-1}{2} \Rightarrow c=-5.
\end{eqnarray*}
Như vậy $(-1)$ có vô số nghiệm, chẳng hạn khi $a=1;b=1;c=-5$.
}
\end{vd}

\begin{vd}%[EX-C2-L8-2018, Lê Văn Thiện]%[8D3K1]
	Giải phương trình 
	\begin{eqnarray*}
		\dfrac{x-a}{a+b}+\dfrac{x-b}{a-b}= \dfrac{2ab}{b^2-a^2} ~~~(a,b \text{ hằng số}).
	\end{eqnarray*}
\loigiai{
Điều kiện xác định của phương trình là $a\ne \pm b.$\\
Biến đổi phương trình 
\begin{align*}
	&(x-a)(x-b)+(x-b)(a+b)=-2ab\\
	\Leftrightarrow & ax-bx -a^2+ab+ax+bx-ab-b^2=-2ab\\
	\Leftrightarrow & 2ax=a^2+b^2-2ab\\
	\Leftrightarrow & 2ax=(a-b)^2.
\end{align*}
Nếu $a\ne 0$ thì $x=\dfrac{(a-b)^2}{2a}$.\\
Nếu $a=0$ thì $(1)$ có dạng $0x=b^2$. Do $a\ne b$ nên $b\ne 0$, phương trình vô nghiệm.\\
Kết luận\\
Nếu $a\ne 0$, $a\ne \pm b$ thì $S=\left\lbrace \dfrac{(a-b)^2}{2a} \right\rbrace $.\\
Còn lại, $S=\varnothing$.
}
\end{vd}

}
\end{bt}


\begin{bt}%[EX-C2-L8-2018, Lê Văn Thiện]%[8D3K1]
	Giải các phương trình với tham số $a,b$
	\begin{enumerate}
		\item $\dfrac{x-a}{a+1}+\dfrac{x-1}{a-1}=\dfrac{2a}{1-a^2}$;
		\item $\dfrac{x+a-1}{a+2}+\dfrac{x-a}{a-2}+\dfrac{x-a}{4-a^2}=0$;
		\item $3x+\dfrac{x}{a}-\dfrac{3a}{a+1}= \dfrac{4ax}{(a+1)^2}+\dfrac{(2a+1)x}{a(a+1)^2}-\dfrac{3a^2}{(a+1)^3}.$
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $\dfrac{x-a}{a+1}+\dfrac{x-1}{a-1}=\dfrac{2a}{1-a^2}$;\\
	Điều kiện là $a\ne \pm 1$. Biến đổi phương trình thành $2ax=(a-1)^2.$\\
	Nếu $a\ne 0$ (và $a\ne \pm 1$) thì $S=\left\lbrace \dfrac{(a-1)^2}{2a} \right\rbrace .$\\
	Nếu $a=0$ thì $S=\varnothing$.
	\item $\dfrac{x+a-1}{a+2}+\dfrac{x-a}{a-2}+\dfrac{x-a}{4-a^2}=0$;\\
	Điều kiện là $a\ne \pm 2$. Biến đổi phương trình thành $(2a-1)x=2(2a-1).$
	Nếu $a\ne \dfrac{1}{2}$  thì $x=2.$\\
	Nếu $a=\dfrac{1}{2}$ thì $0x=0$, vô số nghiệm.
	\item $3x+\dfrac{x}{a}-\dfrac{3a}{a+1}= \dfrac{4ax}{(a+1)^2}+\dfrac{(2a+1)x}{a(a+1)^2}-\dfrac{3a^2}{(a+1)^3}.$\\
	Điều kiện là $a\ne 0;a\ne -1$. Biến đổi phương trình thành
	\begin{eqnarray*}
		\dfrac{3a(a^2-a+1)}{a(a+1)^2}x=\dfrac{3a(a^2-a+1)}{(a+1)^3}
	\end{eqnarray*}
Do $a\ne 0,$ $a\ne -1,$ $a^2-a+1\ne 0$ (chứng minh dễ dàng) nên $x=\dfrac{a}{a+1}.$
\end{enumerate}
}
\end{bt}


\begin{bt}%[EX-C2-L8-2018, Lê Văn Thiện]%[8D3K1]
	Giải phương trình với các tham số $a,b,c$
	\begin{enumerate}
		\item $\dfrac{x-a}{b+c}+\dfrac{x-b}{c+a}+\dfrac{x-c}{a+b}=3$;
		\item $\dfrac{x-a}{b+c}+\dfrac{x-b}{c+a}+\dfrac{x-c}{a+b}=\dfrac{3x}{a+b+c}$;
		\item $\dfrac{a+b-x}{c}+\dfrac{a+c-x}{b}+\dfrac{b+c-x}{a}= 1-\dfrac{4x}{a+b+c}$;
		\item $\dfrac{2a+b+c-3x}{a}+\dfrac{a+2b+c-3x}{b}+\dfrac{a+b+2c-3x}{c} = 6-\dfrac{9x}{a+b+c}$.
	\end{enumerate}

\loigiai{
\begin{enumerate}
	\item Ta có
	\begin{align*}
		& \dfrac{x-a}{b+c}+\dfrac{x-b}{c+a}+\dfrac{x-c}{a+b}=3\\
		\Leftrightarrow & \dfrac{x-a}{b+c}-1+\dfrac{x-b}{c+a}-1+\dfrac{x-c}{a+b}-1=0\\
		\Leftrightarrow & (x-a-b-c)\left(\dfrac{1}{b+c}+\dfrac{1}{a+c}+\dfrac{1}{a+b} \right)=0. 
	\end{align*}
	Nếu $\dfrac{1}{b+c}+\dfrac{1}{a+c}+\dfrac{1}{a+b}\ne 0$, phương trình có một nghiệm $x=a+b+c.$\\
	Nếu $\dfrac{1}{b+c}+\dfrac{1}{a+c}+\dfrac{1}{a+b}=0$, phương trình có tập nghiệm là $\mathbb{R}.$
	\item $\dfrac{x-a}{b+c}+\dfrac{x-b}{c+a}+\dfrac{x-c}{a+b}=\dfrac{3x}{a+b+c}$;\\
	Tương tự câu a) ta có $$(x-a-b-c)\left(\dfrac{1}{b+c}+\dfrac{1}{a+c}+\dfrac{1}{a+b} -\dfrac{3}{a+b+c}\right)=0.$$\\
	Nếu $\dfrac{1}{b+c}+\dfrac{1}{a+c}+\dfrac{1}{a+b} -\dfrac{3}{a+b+c}\ne 0$, phương trình có một nghiệm $x=a+b+c.$\\
	Nếu $\dfrac{1}{b+c}+\dfrac{1}{a+c}+\dfrac{1}{a+b} -\dfrac{3}{a+b+c}=0$, phương trình có tập nghiệm là $\mathbb{R}.$
	\item $\dfrac{a+b-x}{c}+\dfrac{a+c-x}{b}+\dfrac{b+c-x}{a}= 1-\dfrac{4x}{a+b+c}$;\\
	Làm tương tự câu a), ta có
	$$(a+b+c-x)\left(\dfrac{1}{c}+\dfrac{1}{b}+\dfrac{1}{a} -\dfrac{4}{a+b+c}\right)=0.$$\\
	\item $\dfrac{2a+b+c-3x}{a}+\dfrac{a+2b+c-3x}{b}+\dfrac{a+b+2c-3x}{c} = 6-\dfrac{9x}{a+b+c}$.\\
	Điều kiện $a;b;c;a+b+c\ne 0$. Khi đó
	\begin{align*}
		&\dfrac{2a+b+c-3x}{a}+\dfrac{a+2b+c-3x}{b}+\dfrac{a+b+2c-3x}{c} = 6-\dfrac{9x}{a+b+c}\\
		\Leftrightarrow & \dfrac{2a+b+c-3x}{a}-1+\dfrac{a+2b+c-3x}{b}-1+\dfrac{a+b+2c-3x}{c}-1 = 6-\dfrac{9x}{a+b+c}-3\\
		\Leftrightarrow & (a+b+c-3x)\left(\dfrac{1}{c}+\dfrac{1}{b}+\dfrac{1}{a} -\dfrac{3}{a+b+c} \right) =0
	\end{align*}
	Nếu $\dfrac{1}{c}+\dfrac{1}{b}+\dfrac{1}{a} -\dfrac{3}{a+b+c}\ne 0$, phương trình có một nghiệm $x=\dfrac{a+b+c}{3}.$\\
	Nếu $ \dfrac{1}{c}+\dfrac{1}{b}+\dfrac{1}{a} -\dfrac{3}{a+b+c}=0$, phương trình có tập nghiệm là $\mathbb{R}.$
\end{enumerate}


}
\end{bt}


\section{Phương trình tích}
Phương trình tích (một ẩn) là phương trình dạng có dạng
\begin{eqnarray*}
	A(x)B(x)\ldots =0			\qquad \qquad (1)
\end{eqnarray*}
trong đó $A(x), B(x),\ldots,$ là các đa thức.\\
Để giải $(1)$, ta chỉ cần giải từng phương trình $A(x)=0;B(x)=0,\ldots$ rồi lấy tất cả các nghiệm của chúng.\\
Các phương pháp phân tích đa thức thành nhân tử chung có vai trò quan trọng trong việc đưa một phương trình về dạng tích rồi phân tích. Cách đặt ẩn phụ cũng thường được sử dụng để trình bày lời giải được gọn gàng.\\

\begin{vd}%[EX-C2-L8-2018, Lê Văn Thiện]%[8D3K2]
	Giải phương trình
	\begin{eqnarray*}
		(x+3)^3-(x+1)^3=56
	\end{eqnarray*}
\loigiai{
\textit{Cách 1.}
\begin{align*}
	&(x+3)^3-(x+1)^3=56\\
	\Leftrightarrow & x^3+ 9x^2+27x+27-x^3-3x^2-3x-1=56\\
	\Leftrightarrow & 6x^2+24x+26=56\\
	\Leftrightarrow & 6(x^2+4x-5)=0\\
	\Leftrightarrow & 6(x^2-x+5x-5)=0\\
	\Leftrightarrow & x(x-1)+5(x-1)=0\\
	\Leftrightarrow & (x-1)(x+5)=0.
\end{align*}
Kết luận $S=\{1;-5\}$.\\
\textit{Cách 2.} Chú ý rằng $x+2$ là trung bình cộng của $x+3$ và $x+1$, ta đặt $x+2=y$, phương trình trở thành
\begin{align*}
					&(y+1)^3-(y-1)^3=56\\
	\Leftrightarrow & y^3+3y^2+3y+1-y^3+3y^2-3y+1=56\\
	\Leftrightarrow & 6y^2+2 =56 \Leftrightarrow y^2 = 9 \Leftrightarrow y=\pm 3.
\end{align*}
Với $y=3$ thì $x=1$. Với $y=-3$ thì $x=-5$.\\
Kết luận $S=\{1;-5\}$.
}
\end{vd}

\begin{vd}%[EX-C2-L8-2018, Lê Văn Thiện]%[8D3K1]
	Giải phương trình
	\begin{eqnarray*}
		x^3+(x-1)^3=(2x-1)^3.		\qquad\qquad (1)
	\end{eqnarray*}
\loigiai{
Ta thấy $x+(x-1)=2x-1$. Đặt $x-1=y$ thì $(1)$ có dạng
\begin{align*}
	& x^3+y^3=(x+y)^3\\
	\Leftrightarrow & x^3 + y^3 = x^3+y^3 +3xy(x+y)\\
	\Leftrightarrow & xy(x+y=0)\\
	\Leftrightarrow &\hoac{&x=0\\&y=0\\&x+y=0}\\
	\Leftrightarrow& \hoac{&x=0\\&x-1=0\\&2x-1=0} \\
	\Leftrightarrow &\hoac{&x=0\\&x=1\\&x=\dfrac{1}{2}}.
\end{align*}
$\Leftrightarrow \hoac{&x=0\\&y=0\\&x+y=0}\Leftrightarrow \hoac{&x=0\\&x-1=0\\&2x-1=0} \Leftrightarrow \hoac{&x=0\\&x=1\\&x=\dfrac{1}{2}}.$
Kết luận $S=\left\lbrace 0;\dfrac{1}{1};1\right\rbrace $
}
\end{vd}

\begin{vd}%[EX-C2-L8-2018, Lê Văn Thiện]%[8D3K1]
	Giải phương trình
	\begin{eqnarray}
		(x+1)^2(x+2)+(x-1)^2(x-2)=12.
	\end{eqnarray}
	\loigiai{
	Rút gọn vế trái của phương trình, ta được
	\begin{align*}
	2x^3+10x=12& \Leftrightarrow x^3+5x-6=0\\
				& \Leftrightarrow (x^3-1)+5(x-1)=0\\
				& \Leftrightarrow (x-1)(x^2+x+6)=0
	\end{align*}
Dễ dàng chứng minh được $x^2+x+6\ne 0$. Do đó $S={1}$.
}
\end{vd}

\begin{vd}%[EX-C2-L8-2018, Lê Văn Thiện]%[8D3K1]
	Giải phương trình
	\begin{eqnarray}
	(x^2-1)(x^2+4x+3)=192.
	\end{eqnarray}
	\loigiai{
		Biển đổi phương trình thành
		\[(x^2-1)(x+1)(x+3)=192\Leftrightarrow (x-1)(x+1)^2(x+3)=192.\]
		Đặt $x+1=y$, phương trình trở thành
		\[(y-2)y^2(y+2)=192\Leftrightarrow y^2(y^2-4)=192.\]
		Đặt $y^2-2=z$ thì $z+2\ge 0$, phương trình trở thành
		\[(z+2)(z-2)=192\Leftrightarrow z^2 =196 \Leftrightarrow z=\pm 14.\]
		Loại $z=-14$ vì trái với điều kiện $z+2\ge 0$.\\
		Với $z=14$ thì $y^2=16$, do đó $y=\pm 4$.\\
		Với $y=4$ thì $x+1=4$ nên $x=3$.\\
		Với $y-4$ thì $x+1=-4$ nên $x=-5$.

\begin{vd}%[EX-C2-L8-2018, Lê Văn Thiện]%[8D3K1]
	Giải các phương trình
	\begin{enumerate}
	\item $x^4+3x^3+4x^2+3x+1=0$;					\qquad\qquad $(1)$.
	\item $x^5-x^4+3x^3+3x62-x+1=0$			\qquad\qquad $(2)$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $x^4+3x^3+4x^2+3x+1=0$ \qquad $(1)$.\\
	Ta thấy $x=0$ không là nghiệm của phương trình $(1)$. Chia hai vế của $(1)$ cho $x^2\ne 0$, ta được
	\begin{align*}
	&x^2+3x+4+\dfrac{3}{x}+\dfrac{1}{x^2}=0\\
	\Leftrightarrow & \left( x^2+\dfrac{1}{x^2}\right) + 3\left(x+\dfrac{1}{x}  \right) +4=0.
	\end{align*}

	\begin{note}
		Cũng có thể giải phương trình $(1)$ bằng cách biến đổi vế trái thành
		\[(x+1)^2(x^2+x+1).\]
	\end{note}

	Ta thấy $x=-1$ là một ngiệm của phương trình $(2)$ vì tổng các hệ số của hạng tử bậc chẵn bằng tổng các hệ số của hạng tử bậc lẻ. Biến đổi phương trình $(2)$ thành
	\[(x+1)(x^4-2x^3+5x^2-2x+1)=0.\]

	\[x^4-2x^3+5x^2-2x+1=0	\qquad \qquad (3)\]	
	Ta thấy $x=0$ không là nghiệm không là nghiệm của $(3)$. Chia $2$ vế của $(3)$ cho $x^2\ne 0$, ta được 
	\begin{align*}
		&  x^2-2x+5-\dfrac{2}{5}+\dfrac{1}{x^2}=0\\
		\Leftrightarrow  & \left(x62+\dfrac{1}{x^2}\right) -2\left( x+\dfrac{1}{x}\right) +5=0. 
	\end{align*}
	Đặt $x+\dfrac{1}{x}=y$ thì $x^2+\dfrac{1}{x^2}=y^2-2$, ta được $y^2-2y+3=0$, vô nghiệm.
	Kết luận $S=\{-1\}$.
\end{enumerate}

\begin{note}
Ta gọi các phương trình $(1)$ và $(2)$ là \textit{phương trình đối xứng}: các hệ số của đa thức ở $2$ vế trái có tính chất đối xứng qua hạng tử đứng giữa.\\
Phương trình $(1)$ là phương trình đối xứng bậc chẵn, phương trình $(2)$ là phương trình đối xứng bậc lẻ.\\
Phương trình đối xứng bậc lẻ bao giờ cũng nhận $x=-1$ làm một nghiệm, do đó bằng cách chia hai vế cho $x+1$, ta thu được phương trình đối xứng bậc chẵn $2n$.\\
Phương trình đối xứng bậc chẵn $2n$ đối với $x$ đưa được về phương trình bậc $n$ đối với $y$ bằng cách đặt ẩn phụ $y=x+\dfrac{1}{x}$.\\
Ta có nhận xét sau để kiểm tra lại nghiệm của phương trình đối xứng: Nếu $a$ là nghiệm phương trình thì $\dfrac{1}{a}$ cũng là nghiệm của phương trình.
\end{note}

}
