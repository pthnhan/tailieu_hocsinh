\begin{ex}%[9DT18-TT-ArchimedesAcademy-HaNoi-L6]%[Đỗ Viết Lân, THCS05]%[9D1Y1]%[9D1K8]%[9D1G8]
Cho hai biểu thức\\
$$A=\dfrac{x+7}{\sqrt{x}} \text{ và } B=\dfrac{\sqrt{x}}{\sqrt{x}+3}+\dfrac{2\sqrt{x}-1}{\sqrt{x}-3}-\dfrac{2x-\sqrt{x}-3}{x-9} \text{ (với } x>0,x \ne 9)$$
\begin{enumerate}[1.]
\item Tính giá trị của biểu thức $A$ khi $x=16.$
\item Rút gọn biểu thức $B.$
\item Tính giá trị nhỏ nhất của biểu thức $P=A+\dfrac{1}{B}.$
\end{enumerate}
\loigiai{
\begin{enumerate}[1.]
\item Thay $x=16$ vào biểu thức $A$, ta có $A=\dfrac{16+7}{\sqrt{16}}=\dfrac{23}{4}.$
\item 
	\begin{eqnarray*}
	&B&=\dfrac{\sqrt{x}}{\sqrt{x}+3}+\dfrac{2\sqrt{x}-1}{\sqrt{x}-3}-\dfrac{2x-\sqrt{x}-3}{x-9}\\
	&&=\dfrac{\sqrt{x}(\sqrt{x}-3)}{(\sqrt{x}+3)(\sqrt{x}-3)}+\dfrac{(2\sqrt{x}-1)(\sqrt{x}+3)}{(\sqrt{x}-3)(\sqrt{x}+3)}-\dfrac{2x-\sqrt{x}-3}{(\sqrt{x}+3)(\sqrt{x}-3)}\\
	&&=\dfrac{x+3\sqrt{x}}{(\sqrt{x}+3)(\sqrt{x}-3)}\\
	&&=\dfrac{\sqrt{x}(\sqrt{x}+3)}{(\sqrt{x}+3)(\sqrt{x}-3)}\\
	&&=\dfrac{\sqrt{x}}{\sqrt{x}-3}.
	\end{eqnarray*}
\item Với $x>0,x \ne 9$ thì
\begin{eqnarray*}
&P&=A+\dfrac{1}{B}\\
&&=\dfrac{x+7}{\sqrt{x}}+\dfrac{\sqrt{x}-3}{\sqrt{x}}\\
&&=\sqrt{x}+\dfrac{7}{\sqrt{x}}+1-\dfrac{3}{\sqrt{x}}\\
&&=\sqrt{x}+\dfrac{4}{\sqrt{x}}+1\\
&&\ge 2\sqrt{\sqrt{x} \cdot \dfrac{4}{\sqrt{x}}}+1\quad \text{(Bất đẳng thức Cô-si)}\\
&&=2 \cdot 2 +1=5.
\end{eqnarray*}
Dấu ``='' xảy ra $\Leftrightarrow \sqrt{x} = \dfrac{4}{\sqrt{x}} \Leftrightarrow x=4.$\\
Vậy giá trị nhỏ nhất của $P$ là $5$ khi $x=4$.

\end{enumerate}
}
\end{ex}


\begin{ex}%[9DT18-TT-ArchimedesAcademy-HaNoi-L6]%[Đỗ Viết Lân, THCS05]%[9D4B8]
\textit{Giải bài toán bằng cách lập phương trình hoặc hệ phương trình}\\
Một ô tô đi từ A đến B cách nhau $260$ km, sau khi ô tô đi được $120$ km với vận tốc dự định thì tăng vận tốc thêm $10$ km/h trên quãng đường còn lại. Tính vận tốc dự định của ô tô, biết xe đến B sớm hơn thời gian dự định $20$ phút.

\loigiai{
Gọi vận tốc dự định của ô tô là $x$ (km/h, $x>0$).\\
Thời gian dự định đi hết quãng đường AB là $\dfrac{260}{x}$ (h).\\
Thời gian thực tế ô tô đi trên quãng đường dài $120$ km là $\dfrac{120}{x}$ (h).\\
Thời gian thực tế ô tô đi trên quãng đường còn lại là $\dfrac{140}{x+10}$ (h).\\
Vì xe đến B sớm hơn thời gian dự định $20$ phút = $\dfrac{1}{3}$ h nên ta có phương trình
	\begin{eqnarray*}
	&&\dfrac{12}{x}+\dfrac{140}{x+10}+\dfrac{1}{3}=\dfrac{260}{x}\\
	&\Leftrightarrow &360x+3600+420x+x^2+10x=780x+7800\\
	&\Leftrightarrow & x^2+10x-4200=0\\
	&\Leftrightarrow &\hoac{&x=-70&\text{(loại)}\\&x=60&\text{(chọn)}}.
	\end{eqnarray*}
Vậy vận tốc dự định của ô tô là $60$ km/h.

}
\end{ex}


\begin{ex}%[9DT18-TT-ArchimedesAcademy-HaNoi-L6]%[Đỗ Viết Lân, THCS05]%[9D5G2]
Cho hệ phương trình $\heva{&x+2y=3\\&x+my=1}$ ($m$ là tham số). Tìm giá trị nguyên của $m$ để hệ có nghiệm duy nhất $(x,y)$ sao cho $x,y$ là các số nguyên.
\loigiai{
	Ta có $x+2y=3\Leftrightarrow x = 3-2y$. Thay vào phương trình còn lại ta được
	$$x+my = 1 \Leftrightarrow (3-2y)+my = 1 \Leftrightarrow (m-2)y = -2 \hfill (*)$$
	Để hệ phương trình có nghiệm duy nhất thì phương trình $(*)$ có nghiệm duy nhất $\Leftrightarrow m \neq 2$.\\
	Khi đó ta có $y = \dfrac{-2}{m-2}$.\\
	Để $y$ nguyên thì $m-2$ là ước của $-2 \Rightarrow m - 2 \in \{\pm 1; \pm 2\}$. Do đó $m \in \{0;1;3;4\}$.
}
\end{ex}

\begin{ex}%[9DT18-TT-ArchimedesAcademy-HaNoi-L6][Đỗ Viết Lân, THCS05]%[9D4B4]%[9D4K7]
Cho parabol $(P):y=x^2$ và đường thẳng $(d):y=-2mx-4m$ ($m$ là tham số).
	\begin{enumerate}[a)]
	\item Tìm $m$ để $(d)$ cắt $(P)$ tại hai điểm phân biệt $A,B$.
	\item Giả sử $x_1,x_2$ là hoành độ của $A,B$. Tìm $m$ để $|x_1|+|x_2|=3.$
	\end{enumerate}
\loigiai{
	Xét phương trình hoành độ giao điểm của $(d)$ và $(P)$:
	$$x^2 = -2mx - 4m \Leftrightarrow x^2 + 2mx+4m = 0\quad (1).$$
	Ta có $\Delta' = m^2 - 4m = m(m-4).$
	\begin{enumerate}
	\item Để $(d)$ cắt $(P)$ tại hai điểm phân biệt thì phương trình $(1)$ có hai nghiệm phân biệt\\
	$\Leftrightarrow \Delta' > 0 \Leftrightarrow m >4$ hoặc $m<0$.
	\item Theo hệ thức Vi-ét ta có $\begin{cases}x_1 + x_2 = -2m\\x_1x_2 = 4m \end{cases}$.\\
	Theo đề ta có
	\begin{eqnarray*}
	&&|x_1| + |x_2| = 3\\
	&\Leftrightarrow &(|x_1| + |x_2|)^2=9\\
	&\Leftrightarrow &x_1^2 + x_2^2 + 2|x_1x_2| = 9\\
	&\Leftrightarrow &(x_1+x_2)^2 - 2x_1x_2 + 2|x_1x_2| - 9 = 0\\
	&\Leftrightarrow &4m^2 - 8m + 2|4m| -  9 = 0
	\end{eqnarray*}
	Theo điều kiện có nghiệm của phương trình ta xét hai trường hợp sau:\\
	+ Nếu $m < 0$ thì ta có phương trình $4m^2 - 16m-9=0 \Leftrightarrow m = -\dfrac{1}{2}$.\\
	+ Nếu $m > 4$ thì ta có phương trình $4m^2 - 9 =0 \Leftrightarrow m = \dfrac{3}{2}$ (loại vì không thỏa điều kiện có nghiệm).\\
	Vậy $m = -\dfrac{1}{2}$.
	\end{enumerate}
}
\end{ex}


\begin{ex}%[9DT18-TT-ArchimedesAcademy-HaNoi-L6]%[Đỗ Viết Lân, THCS05]%[9H3B7]%[9H3B9]%[9H3K7]%[9H3K4]
Cho tam giác $ABC$ nội tiếp đường tròn $(O;R)$, đường kính $BC\,(AB>AC).$ Từ $A$ kẻ tiếp tuyến với đường tròn $O$ cắt tia $BC$ tại $M$. Kẻ dây $AD$ vuông góc với $BC$ tại $H$.
\begin{enumerate}
\item Chứng minh tứ giác $AMDO$ nội tiếp.
\item Giả sử $\widehat{ABC}=30^{\circ}$. Tính diện tích viên phân giới hạn bởi dây $AC$ và cung $AC$ nhỏ theo $R$.
\item Kẻ $AN$ vuông góc với $BD$ ($N$ thuộc $BD$), gọi $E$ là trung điểm của $AN$, $F$ là giao điểm thứ hai của $BE$ với $(O)$, $P$ là giao điểm của $AN$ với $BC$, $Q$ là giao điểm của $AF$ với $BC$.
	\begin{enumerate}
	\item Chứng minh tứ giác $AEHF$ nội tiếp.
	\item Chứng minh $BH^2=BP \cdot BQ$.
	\end{enumerate}
\item Từ $F$ kẻ đường thẳng song song với $BC$ cắt $AD$ và $AM$ lần lượt tại  $I$ và $K$. Chứng minh rằng $F$ là trung điểm của $IK$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=0.8]

\begin{enumerate}
\item $\triangle OAM$ và $\triangle ODM$ có $OA = OD$, $OM$ chung.\\
Ngoài ra $\triangle OAD$ cân tại $O$ có $OM$ là đường cao nên cũng là đường phân giác $\Rightarrow \widehat{AOM} = \widehat{DOM}$.\\
Do đó $\triangle OAM = \triangle ODM$.\\
$\Rightarrow \widehat{ODM} = \widehat{OAM} = 90^\circ$.\\
$\Rightarrow \widehat{OAM} + \widehat{ODM} = 180^\circ \Rightarrow OAMD$ là tứ giác nội tiếp.
\item Ta có $\widehat{ABC} = 30^\circ \Rightarrow \widehat{ACB} = 60^\circ \Rightarrow \triangle AOC$ đều $\Rightarrow S_{\triangle AOC} = \dfrac{R^2\sqrt{3}}{4}$.\\
$S_{\text{quạt }AOC} = \dfrac{\pi R^2n}{360} = \dfrac{\pi R^2}{6}$.\\
Do đó diện tích viên phân cần tìm là $S = \dfrac{\pi R^2}{6} - \dfrac{R^2\sqrt{3}}{4} = \dfrac{R^2(2\pi - 3\sqrt{3})}{12}$.

Do đó $\dfrac{BE}{BQ} = \dfrac{BP}{BF} \Rightarrow BE \cdot BF = BP \cdot BQ$.\\
Vậy $BH^2=BP \cdot BQ$.
\end{enumerate}
\item Ta có $\widehat{HAM} = \widehat{NBA} = \dfrac{1}{2} \text{sđ}\,\wideparen{AD}$.\\

Suy ra $\triangle EBN \backsim \triangle QAH \Rightarrow \dfrac{BN}{AH} = \dfrac{EN}{QH}$.\\
Do đó $\dfrac{AN}{HM} = \dfrac{EN}{QH}$ mà $E$ là trung điểm của $AN$\\
$\Rightarrow EN = \dfrac{1}{2}AN \Rightarrow HQ = \dfrac{1}{2}HM \Rightarrow HQ = QM$.\\
Do $IK \parallel HM \Rightarrow \dfrac{IF}{HQ} = \dfrac{FK}{QM} \Rightarrow FI = FK \Rightarrow F$ là trung điểm $IK$.
\end{enumerate}
}

\begin{ex}%[9DT18-TT-ArchimedesAcademy-HaNoi-L6]%[Đỗ Viết Lân, THCS05]%[9D5G1]
Cho các số thực $a,b,c$ thỏa mãn $a \ge 2;b \ge 5; c \ge 5$ và $2a^2+b^2+c^2=69$.  Tính giá trị nhỏ nhất của $P=12a+13b+11c$.
\loigiai{
Đặt $\begin{cases} a = 2+x\\b=5+y\\c=5+z \end{cases}$\\
Khi đó từ giả thiết ta có $2x^2 + y^2 + z^2 + 8x + 10y + 10z = 11 \quad (*) $.\\
Giả sử $\max\{y,z\} > 1$. Nếu $x, y, z > 1$ thì $VT(*) > 11$\\
Suy ra $0 \leq y, z \leq 1$.\\
Từ $(*)$ ta có $x < 2$ vì nếu trái lại thì $VT(*) \geq 2.2^2 + 8.2 > 11$.\\
Khi đó ta có $\begin{cases}
4x \geq 2x^2\\3y \geq y^2\\z \geq z^2
\end{cases} \Rightarrow 4x + 3y + z \geq 2x^2 + y^2 + z^2$\\
$\Rightarrow 12x + 13y + 11z \geq 2x^2 + y^2 + z^2 + 8x + 10y + 10z = 11$.\\
Suy ra $P = 12a + 13b + 11c = 12x + 13y + 11z + 144 \geq 11 + 144 = 155$.\\
Vậy giá trị nhỏ nhất của $P$ là $155$ khi $\begin{cases}
a=2\\b=5\\c=6
\end{cases} $
}
\end{ex}
