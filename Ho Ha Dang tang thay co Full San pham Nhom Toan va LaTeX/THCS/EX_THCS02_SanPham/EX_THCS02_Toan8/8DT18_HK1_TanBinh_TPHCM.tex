	Phân tích đa thức sau thành nhân tử:
	\begin{enumerate}[\qquad 1)]
		\item $5xy^2+10y$.
		\item $4x^2-y^2+5(2x-y)$.
		\item $25-a^2+2ab-b^2$.
		\item $x^2+x-30$.
	\end{enumerate}
	\loigiai{
			\begin{enumerate}[\qquad 1)]
				\item $5xy^2+10y=5y(xy+2)$.
				\item $4x^2-y^2+5(2x-y)=(2x-y)(2x+y)+5(2x-y)=(2x-y)(2x+y+5)$.
				\item $25-a^2+2ab-b^2=5^2-(a^2-2ab+b^2)=5^2-(a-b)^2=(5-a+b)(5+a-b)$.
				\item $x^2+x-30=x^2-5x+6x-30=x(x-5)+6(x-5)=(x-5)(x+6)$.
			\end{enumerate}
			}
\end{ex}

	\begin{enumerate}[\qquad 1)]
		\item $(2x+3)^2-4x(x+3)$.
		\item $\dfrac{2}{x-2}+\dfrac{3}{x+2}- \dfrac{18-5x}{x^2-4}$.
	\end{enumerate}
	\loigiai{
			\begin{enumerate}[\qquad 1)]
				\item $(2x+3)^2-4x(x+3)=4x^2+12x+9-4x^2-12x=9$.
				\item $\dfrac{2}{x-2}+\dfrac{3}{x+2}- \dfrac{18-5x}{x^2-4}\\=\dfrac{2(x+2)+3(x-2)-(18-5x)}{x^2-4}\\= \dfrac{2x+4+3x-6-18+5x}{x^2-4} \\ = \dfrac{10x-20}{(x-2)(x+2)}\\=\dfrac{10(x-2)}{(x-2)(x+2)}\\=\dfrac{10}{x+2}$.
			\end{enumerate}
			}

\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Tân Bình, TPHCM]%[8D1K3]%[8D1G9]
	Tìm $x$ biết:
	\begin{enumerate}[\qquad \qquad 1)]
		\item $(x-5)^2-(x+3)(x-3)=14$.
		\item $(x+7)^2-3x-21=0$.
	\end{enumerate}
	\loigiai{
			\begin{enumerate}[\qquad \qquad 1)]
				\item $(x-5)^2-(x+3)(x-3)=14$ \\
					$\Leftrightarrow x^2-10x+25-(x^2-9)=14$\\ $\Leftrightarrow -10x =-20$\\ $\Leftrightarrow x=2$. 
				\item $(x+7)^2-3x-21=0$ \\
					$\Leftrightarrow (x+7)^2-3(x+7)=0$\\ $\Leftrightarrow (x+7)(x+7-3)=0$\\ $\Leftrightarrow (x+7)(x+4)=0$\\
					$\Leftrightarrow \left[ \begin{aligned} & x+7=0 \\ & x+4=0 \end{aligned} \right.$\\
					$\Leftrightarrow \left[ \begin{aligned} & x=-7 \\ & x=-4 \end{aligned} \right.$.
			\end{enumerate}
			}
\end{ex}

\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Tân Bình, TPHCM]%[8H1B9]%[8H1B4]%[8H1K9]%[8H1G9]
	Cho $\triangle ABC$ vuông tại $A$ ($AB<AC$), đường cao $AH$. Gọi $O$ là trung điểm của $BC$, $D$ là điểm đối xứng của $A$ qua $O$.
	\begin{enumerate}[\qquad 1)]
		\item Chứng minh tứ giác $ABDC$ là hình chữ nhật.
		\item Trên tia đối của tia $HA$ lấy điểm $E$ sao cho $HE=HA$. Chứng minh $ED=2HO$.
		\item Chứng minh tứ giác $BEDC$ là hình thang cân.
		\item Gọi $M$, $N$ lần lượt là hình chiếu của điểm $E$ lên hai đường thẳng $BD$ và $CD$. Chứng minh ba điểm $H$, $M$, $N$ thẳng hàng.
	\end{enumerate}
	\loigiai{ \\ 
		\begin{center}
			\begin{tikzpicture}[line cap=round,line join=round]  
			\tkzDefPoints{0/0/A, 8/0/C}
			\coordinate (B) at ($(A)!0.5!90:(C)$);
			\coordinate (H) at ($(B)!(A)!(C)$);
			\coordinate (O) at ($(B)!0.5!(C)$);
			\coordinate (D) at ($(A)!2!(O)$);
			\coordinate (E) at ($(A)!2!(H)$);
			\coordinate (M) at ($(B)!(E)!(D)$);
			\coordinate (N) at ($(C)!(E)!(D)$);
			\tkzDrawSegments(A,H A,D E,D H,E E,B D,N N,E E,M)
			\tkzDrawSegments[color=red](H,N)
			\tkzDrawPoints[fill](B,C,A,O)
			\tkzMarkSegments[size=3pt,mark=s|,color=red](O,C O,B)
			\tkzMarkSegments[size=3pt,mark=|,color=red](O,A O,D)
			\tkzMarkSegments[size=3pt,mark=||,color=red](H,A H,E)
			\tkzInterLL(E,D)(M,N)\tkzGetPoint{K}
			\tkzLabelPoints[above](E,O,H,N,K)
			\tkzLabelPoints[below](M)
			\tkzLabelPoints[below left](A)
			\tkzLabelPoints[below right](C)
			\tkzLabelPoints[left](B)
			\tkzLabelPoints[right](D)
			\tkzMarkRightAngles[size=0.2](B,A,C A,H,B E,N,D E,M,D)
			\tkzDrawPolygon(A,B,C) 	\tkzDrawPolygon(A,B,D,C)
			\tkzMarkAngle[size=.5,mark=|](D,C,B) \tkzMarkAngle[size=.7,mark=|](A,B,H) 	\tkzMarkAngle[size=.5,mark=|](C,B,E) 
			\tkzMarkAngle[size=.5,mark=|](N,D,K)
			\tkzMarkAngle[size=.5,mark=|](K,N,D) 
			\tkzMarkAngle[size=.5,mark=|](O,D,C)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}[\qquad 1)]
			\item Chứng minh tứ giác $ABDC$ là hình chữ nhật.\\
				Ta có $D$ đối xứng với $A$ qua $O$ nên $O$ là trung điểm $AD$.\\
				Tứ giác $ABDC$ có $O$ là giao điểm hai đường chéo $BC$ và $AD$ mà $O$ là trung điểm của mỗi đường nên $ABDC$ là hình bình hành.\\
				Hình bình hành $ABDC$ có $\widehat{BAC}=90^\circ$ nên $ABDC$ là hình chữ nhật.
				\item Chứng minh $ED=2HO$.\\
				Ta có $O$ là trung điểm $AD$, $H$ là trung điểm $AE$ nên $HO$ là đường trung bình của $\triangle ABC$. \\
				Do đó $ED \parallel HO$ và $ED=2HO$.
			\item Chứng minh tứ giác $BEDC$ là hình thang cân.\\
				Ta có $ED \parallel HO$ nên $ED \parallel BC$. Khi đó $BEDC$ là hình thang. (1)\\
				Xét $\triangle AHB$ và $\triangle EHB$ có: 
				$\left\{ \begin{aligned} & BH \text{ chung} \\ & \widehat{AHB} = \widehat{EHB}=90^\circ\\&HA=HE \text{ (giả thiết)}   \end{aligned} \right. \\\Rightarrow \triangle AHB =\triangle EHB$(cạnh - góc - cạnh).\\
				Suy ra $\widehat{ABH} = \widehat{EBH} = \widehat{EBC}$.\\
				Mà  $\widehat{ABH} = \widehat{DCB}$.\\
				Do đó $\widehat{EBC} = \widehat{DCB}$. (2)\\
				Từ (1) và (2) suy ra tứ giác $BEDC$ là hình thang cân.
			\item Chứng minh ba điểm $H$, $M$, $N$ thẳng hàng. \\
				Tứ giác $ENDM$ có $\widehat{EMD}=\widehat{MDN}=\widehat{END}=90^\circ$ nên $ENDM$ là hình chữ nhật.\\
				Gọi $K$ là giao điểm $ED$ và $MN$.\\
				Ta có $H$ là trung điểm $AE$, $K$ là trung điểm $ED$ nên $HK$ là đường trung bình tam giác $EAD$. Do đó $HK \parallel AD$. (3)\\
				Ta lại có $\widehat{KDN}=\widehat{OCD}$ (vì $ED \parallel BC$).\\
				Mà
				$\widehat{KDN}=\widehat{KND}$ (vì $\triangle KND$ cân tại $K$)
				và $\widehat{OCD}=\widehat{ODC}$ (vì $\triangle OCD$ cân tại $O$) 
				nên $\widehat{KND}=\widehat{ODC}$. Suy ra $KN \parallel OD$. (4)\\
				Từ (3) và (4) suy ra $H$, $K$, $N$ thẳng hàng.
				Mặt khác $M$, $K$, $N$ thẳng hàng.
				Vậy $H$, $M$, $N$ thẳng hàng.
		\end{enumerate} 
	}		
\end{ex}
\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Tân Bình, TPHCM]%[8H2K2]
	Gần Tết Bác An có một phòng cần thay đổi gạch lót sàn. Biết chiều dài cần $20$ viên gạch, chiều rộng cần $10$ viên gạch. Mỗi viên gạch có kích thước $40$ cm $\times$ $40$ cm với giá $65000$ đồng/viên gạch.
	\begin{enumerate}[\qquad 1)]
		\item Tính chiều dài, chiều rộng của căn phòng.
		\item Hỏi diện tích sàn nhà của căn phòng nhà bác An là bao nhiêu mét vuông?
		\item Hãy tính tiền bác An cần mua gạch để lót sàn?
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[\qquad 1)]
			\item 
			Chiều dài của căn phòng là: $40 \times 20=800$ (cm).\\
			Chiều rộng của căn phòng là: $40 \times 10=400$ (cm).
			\item Diện tích sàn nhà của căn phòng nhà bác An là: $800 \times 400=320000$ (cm$^2$). 
			\item 
			Số gạch cần để lót sàn là: $20 \times 10=200$ (viên).\\
			Số tiền bác An cần mua gạch để lót sàn là: $65000 \times 200=13000000$ (đồng).
		\end{enumerate}
			}
\end{ex}
