\begin{ex}%[Đề thi vào 10, Sở giáo dục TP HCM, 2017]%[9D4Y4]%[9D3K6]
\begin{enumerate}
	\item Giải phương trình: $x^2=(x-1)(3x-2)$
	\item Một miếng đất hình chữ nhật có chu vi $100$ m. Tính chiều dài và chiều rộng của miếng đất, biết rằng $ 5 $ lần chiều rộng hơn $ 2 $ lần chiều dài $40$ m.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Phương trình $x^2=(x-1)(3x-2)\Leftrightarrow 2x^2-5x+2=0$. Tính $\Delta=9$, phương trình có $ 2 $ nghiệm phân biệt $x_1=2$, $x_2=\dfrac{1}{2}$.
	\item Gọi $x,y$ (m) lần lượt là chiều rộng và chiều dài của miếng đất ($y>x>0$). Khi đó ta có hệ phương trình
	$$\heva{&2(x+y)=100\\& 5x-2y=40}\Leftrightarrow\heva{&x+y=50\\&7x=140}\Leftrightarrow \heva{&x=20\\&y=30} $$
Vậy, chiều rộng là $20$ m và chiều dài là $30$ m.
\end{enumerate}
}

\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục TP HCM, 2017]%[9D1K8]%[9H1K5]
\begin{enumerate}
\item Thu gọn biểu thức sau: $A=(\sqrt{3}+1) \sqrt{\dfrac{14-6\sqrt{3}}{5+\sqrt{3}}}$
\item Lúc $ 6 $ giờ sáng, bạn An đi xe đạp từ nhà (điểm $ A $) đến trường (điểm $ B $) phải leo lên và xuống  một con dốc (như hình vẽ bên dưới). Cho biết đoạn thẳng $ AB $ dài $762$ m, góc $A=6^{\circ}$, góc $B=4^{\circ}$

\begin{enumerate}
	\item Tính chiều cao $h$ của con dốc.
	\item Hỏi bạn An đến trường lúc mấy giờ? Biết rằng tốc độ trung bình lên dốc là $4$ km/h và tốc độ trung bình xuống dốc là $19$ km/h.
\end{enumerate}
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item 	Ta có  \begin{flalign*}
 A&=(\sqrt{3}+1) \sqrt{\dfrac{14-6\sqrt{3}}{5+\sqrt{3}}}=(\sqrt{3}+1) \sqrt{\dfrac{(14-6\sqrt{3})(5-\sqrt{3})}{25-3}}
	= (\sqrt{3}+1) \sqrt{\dfrac{44(2-\sqrt{3})}{22}}
\\	&=(\sqrt{3}+1) \sqrt{4-2\sqrt{3}}=(\sqrt{3}+1) \sqrt{(\sqrt{3}-1)^2}
		= (\sqrt{3}+1) (\sqrt{3}-1)=3-1=2.
	\end{flalign*}
	\item
	
	\begin{enumerate}
		\item Xét tam giác $ACH$ vuông tại $H$, ta có $\tan CAH=\dfrac{CH}{AH}\Rightarrow AH=\dfrac{CH}{\tan 6^{\circ}}\,(1)$. Tam giác $BCH$ vuông tại $H$, ta có $\tan CBH=\dfrac{CH}{BH}\Rightarrow BH=\dfrac{CH}{\tan 4^{\circ}}\,(2)$.
		
		Từ $(1)$ và $(2)$ ta có $AB=AH+BH=\dfrac{CH}{\tan 6^{\circ}}+\dfrac{CH}{\tan 4^{\circ}}\Leftrightarrow 762=CH\left(\dfrac{1}{\tan 6^{\circ}}+\dfrac{1}{\tan 4^{\circ}}\right)\Leftrightarrow CH=762: \left(\dfrac{1}{\tan 6^{\circ}}+\dfrac{1}{\tan 4^{\circ}}\right)\approx 32$ m. 
		\item $\circ$ Đoạn đường $AC=\dfrac{CH}{\sin 6^{\circ}}=\dfrac{32}{\sin 6^{\circ}}\approx 306 \text{ m }=0,306$ km, thời gian bạn An đi hết quãng đường $AC$ là $t_1=\dfrac{AC}{4}=\dfrac{0,306}{4}\approx 0,075$ h.
		
		$\circ$ Đoạn đường $BC=\dfrac{CH}{\sin 4^{\circ}}=\dfrac{32}{\sin 4^{\circ}}\approx 459\text{ m }=0,459$ km, thời gian bạn An đi hết quãng đường $BC$ là $t_2=\dfrac{BC}{19}=\dfrac{0,459}{19}\approx 0,024$ h. 
		
		$\circ$ Tổng thời gian bạn An đi là $t_1+t_2=0,075+0,024=0,099\approx 0,1$ h $=6$ phút. Vậy bạn An đến trường lúc $6$ giờ $6$ phút.
	\end{enumerate}
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục TP HCM, 2017]%[9D4B4]%[9D4K6]
	Cho phương trình: $x^2-(2m-1)x+m^2-1=0\quad(1)$  ($x$ là ẩn số)
\begin{enumerate}
	\item Tìm điều kiện của $m$ để phương trình $ (1) $ có $ 2 $ nghiệm phân biệt.
	\item Định $m$ để hai nghiệm $x_1,x_2$ của phương trình $ (1) $ thỏa mãn 
	$(x_1-x_2)^2=x_1-3x_2$.
\end{enumerate}
\loigiai{
Xét phương trình: $x^2-(2m-1)x+m^2-1=0\quad(1)$  ($x$ là ẩn số)
\begin{enumerate}
	\item 
	Ta tính được $\Delta = (2m-1)^2-4(m^2-1)=-4m+5$. Do $a=1\neq0$,  phương trình $ (1) $ có $ 2 $ nghiệm phân biệt khi và chỉ khi $\Delta>0\Leftrightarrow -4m+5>0\Leftrightarrow m<\dfrac{5}{4} \quad (*)$.
	\item Với điều kiện $(*)$, và theo Định lý Viète phương trình $(1)$ có hai nghiệm $x_1,x_2$ thỏa $\heva{&x_1+x_2=-\dfrac{b}{a}=2m-1\\&x_1x_2=\dfrac{c}{a}=m^2-1.}$
	
	Kết hợp với giả thiết, ta có
\begin{eqnarray*}
(x_1-x_2)^2=x_1-3x_2&\Leftrightarrow& (x_1+x_2)^2-4x_1x_2=x_1-3x_2\\&\Leftrightarrow& x_1-3x_2 = (2m-1)^2-4(m^2-1)=-4m+5.
\end{eqnarray*}
Khi đó, $\heva{&x_1+x_2=2m-1\\&x_1-3x_2=-4m+5}\Leftrightarrow \heva{&x_1+x_2=2m-1\\&4x_2=6(m-1)}\Leftrightarrow \heva{&x_1=\dfrac{m+1}{2}\\&x_2=\dfrac{3(m-1)}{2}}.$ 

Mặt khác $$x_1x_2=m^2-1\Leftrightarrow \dfrac{m+1}{2}.\dfrac{3(m-1)}{2}=m^2-1\Leftrightarrow m^2-1=0\Leftrightarrow m=\pm 1 \quad(\text{nhận}).$$
Vậy có $ 2 $ giá trị $m=\pm 1$ thỏa yêu cầu bài toán.
\end{enumerate}
}
\end{ex}

	Cho tam giác $ABC$ vuông tại $A$. Đường tròn tâm $O$ đường kính $AB$ cắt các đoạn $BC$ và $OC$ lần lượt là $D$ và $I$. Gọi $H$ là hình chiếu của $A$ lên $OC$; $AH$ cắt $BC$ tại $M$
\begin{enumerate}
	\item Chứng minh: Tứ giác $ACDH$ nội tiếp và $\widehat{CHD}=\widehat{ABC}$.
	\item Chứng minh: Hai tam giác $OHB$ và $OBC$ đồng dạng với nhau và $ HM $ là tia phân giác của góc $BHD$.            
	\item Gọi $ K $ là trung điểm của $ BD $. Chứng minh: $MD.BC=MB.CD$ và $MB.MD=MK.MC$.
	\item Gọi $E$ là giao điểm của $AM$ và $OK$; $J$ là giao điểm của $IM$ và $(O )$ ($ J $ khác $ I $).

	Chứng minh: Hai đường thẳng $OC$ và $EJ$ cắt nhau tại một điểm nằm trên $(O )$.
\end{enumerate}
	\loigiai{
 \begin{enumerate}

	Xét hai tam giác $OHB$ và $OBC$. Ta có
