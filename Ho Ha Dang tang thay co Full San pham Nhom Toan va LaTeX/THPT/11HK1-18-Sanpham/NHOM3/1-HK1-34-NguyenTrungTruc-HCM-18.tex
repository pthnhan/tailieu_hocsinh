\begin{bt}%[Đề thi Học Kì 1 T11, THPT Nguyễn Trung Trực, HCM 2018]%[1D1K3-3]%[1D1K3-4]%[1D2B2-6]
Giải các phương trình sau:
		\begin{enumerate}
			\item $\left(1 - \sqrt{3}\right)\sin x\cdot \cos x + \left(1 + \sqrt{3}\right)\cos^2x = \sqrt{3}$
		    \item $\sqrt{3}\left(\sin 2x  + \cos x\right) +  \cos 2x - 3\sin x = 2$
		    \item  $5\mathrm{A}^{3}_{n + 4} = \mathrm{C}^{5}_{n + 6}$\\
		    trong đó $\mathrm{A}^{k}_{n}$, $\mathrm{C}^{k}_{n}$ lần lượt là số chỉnh hợp và tổ hợp chập $k$ của $n$ phần tử, $n,\ k\in\mathbb{N}$.
	    \end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item  Xét phương trình $\left(1 - \sqrt{3}\right)\sin x\cdot \cos x + \left(1 + \sqrt{3}\right)\cos^2x = \sqrt{3}\quad (1)$
						
			- Nếu $\cos x = 0\Leftrightarrow x = \dfrac{\pi }{2} + k\pi,\ k\in \mathbb{ Z}$.  Thay vào $(1)$ suy ra $0 = \sqrt{3}$ vô lý.\\ 
			Do đó $x = \dfrac{\pi }{2} + k\pi,\ k\in \mathbb{ Z}$ không là nghiệm của $(1)$.
			
			- Nếu $\cos x\neq 0\Leftrightarrow x\neq\dfrac{\pi }{2} + k\pi,\ k\in \mathbb{ Z}$ thì 
		\begin{eqnarray*}
			(1) &\Leftrightarrow& \dfrac{\left(1 - \sqrt{3}\right)\sin x\cdot \cos x}{\cos^2x} + \dfrac{\left(1 + \sqrt{3}\right)\cos^2x}{\cos^2x} = \dfrac{\sqrt{3}}{\cos^2x}\\
			&\Leftrightarrow& \left(1 - \sqrt{3}\right)\tan x + \left(1 + \sqrt{3}\right) = \sqrt{3}\left(1 + \tan^2x\right)\\
			&\Leftrightarrow& \sqrt{3}\tan^2x - \left(1 - \sqrt{3}\right)\tan x - 1 = 0\Leftrightarrow \hoac{&\tan x = \dfrac{1}{\sqrt{3}}\\&\tan x = -1.}
		\end{eqnarray*}
			- Khi $\tan x = \dfrac{1}{\sqrt{3}}\Leftrightarrow x = \dfrac{\pi}{6} + m\pi\   \left(m \in \mathbb{Z}\right)$.
			
		 - Khi $\tan x = - 1\Leftrightarrow x = - \dfrac{3\pi}{4} + m\pi\  \left(m \in \mathbb{Z}\right)$.\\
		  Vậy nghiệm của phương trình là $x = \dfrac{\pi}{6} + m\pi$ và $x = - \dfrac{3\pi}{4} + m\pi\  \left(m \in \mathbb{Z}\right)$ .
			
		\item Ta có 
	\begin{eqnarray*}
			&{  }&\sqrt{3}\left(\sin 2x  + \cos x\right) +  \cos 2x - 3\sin x = 2\\ &\Leftrightarrow &\sqrt{3}\left(2\sin x\cdot\cos x + \cos x\right) + 1 - 2\sin^2x  - 3\sin x  - 2 = 0\\
			&\Leftrightarrow &\sqrt{3}\left(2\sin x\cdot\cos x + \cos x\right)  - \left(2\sin^2x  + 3\sin x  + 1\right) = 0\\
			&\Leftrightarrow &\sqrt{3}\cos x\left(2\sin x + 1\right)  - \left(2\sin x  + 1\right)\left(\sin x  + 1\right) = 0\\
			&\Leftrightarrow &\left(2\sin x + 1\right)\left(\sqrt{3}\cos x - \sin x - 1\right) = 0\\
		&\Leftrightarrow& \hoac{&2\sin x + 1 = 0\\&\sqrt{3}\cos x - \sin x - 1 = 0.}
				\end{eqnarray*}
					- Khi $2\sin x + 1 = 0\Leftrightarrow \sin x = - \dfrac{1}{2}\Leftrightarrow \hoac{&x = -\dfrac{\pi}{6} + k2\pi\\&x = \dfrac{7\pi}{6} + k2\pi}\  ,\ \left(k \in \mathbb{Z}\right)$.
					
					- Khi 
					\begin{eqnarray*}
					&{ }&\sqrt{3}\cos x - \sin x - 1 = 0 \Leftrightarrow \dfrac{\sqrt{3}}{2}\cos x - \dfrac{1}{2}\sin x =  \dfrac{1}{2}\\
					 &\Leftrightarrow &\cos\left(x + \dfrac{\pi}{6}\right) = \dfrac{1}{2}\Leftrightarrow \hoac{&x + \dfrac{\pi}{6} = \dfrac{\pi}{3} + k2\pi\\&x + \dfrac{\pi}{6} = - \dfrac{\pi}{3} + k2\pi}\Leftrightarrow \hoac{&x = \dfrac{\pi}{6} + k2\pi\\&x = - \dfrac{\pi}{2} + k2\pi}\  ,\ \left(k \in \mathbb{Z}\right).
					\end{eqnarray*}
		Hợp nghiệm suy ra phương trình có nghiệm là:\\
		 $x = \dfrac{\pi}{6} + m\pi$; $x = -\dfrac{\pi}{6} + k2\pi$ và $x = - \dfrac{\pi}{2} + k2\pi\  \left(m, k \in \mathbb{Z}\right)$.
		 \item  Điều kiện $\heva{&n\geq -1\\&n\in \mathbb{Z}}\quad (*)$.\\
		 Ta có 
		 	\begin{eqnarray*}
		 5\mathrm{A}^{3}_{n + 4} = \mathrm{C}^{5}_{n + 6}&\Leftrightarrow& 5\dfrac{\left(n + 4\right)!}{\left(n + 1\right)!} = \dfrac{\left(n + 6\right)!}{5!\left(n + 1\right)!}\\
		 &\Leftrightarrow& \left(n + 6\right)\left(n + 5\right) = 600\Leftrightarrow n^2 + 11n - 570 = 0\Leftrightarrow\hoac{&n = 19\\&n = - 30.}
		 	\end{eqnarray*}
		 So sánh với điều kiện $(*)$ suy ra $n = 19$ thỏa mãn bài toán.
		 
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Đề thi Học Kì 1 T11, THPT Nguyễn Trung Trực, HCM 2018]%[1D2K3-2]%
	Tìm số hạng chứa $x^{33}$ trong khai triển $\left(2x^3 - \dfrac{3}{x^2}\right)^{16}$.
	\loigiai{ Ta có 
				$$\left(2x^3 - \dfrac{3}{x^2}\right)^{16}= \sum\limits_{k = 1}^{16}\left(- 1\right)^{k}\cdot\mathrm{C}^{k}_{16}\cdot\left(2x^3\right)^{16 - k}\cdot\left(\dfrac{3}{x^2}\right)^{k} = \sum\limits_{k = 1}^{16}\left(- 1\right)^{k}\cdot 2^{16 - k}\cdot 3^{k}\cdot \mathrm{C}^{k}_{16}\cdot\left(x\right)^{48 - 5k}$$
	Suy ra số hạng thứ $k + 1$ có dạng 	$\left(- 1\right)^{k}\cdot 2^{16 - k}\cdot 3^{k}\cdot \mathrm{C}^{k}_{16}\cdot \left(x\right)^{48 - 5k}$.\\
	Để thỏa mãn bài toán suy ra $48 - 5k = 33\Leftrightarrow 5k = 15\Leftrightarrow k = 3$.\\
	Do đó hệ số của số hạng chứa $x^{33}$ là  $\left(- 1\right)^{3}\cdot 2^{13}\cdot 3^{3}\cdot \mathrm{C}^{3}_{16} = - 123863040$.
	}
\end{bt}

	Gieo ngẫu nhiên đồng thời một đồng tiền và một con xúc xắc (đều cân đối, đồng chất). Tính xác suất của biến cố con xúc xắc xuất hiện mặt có số chấm lẻ.
	\loigiai{Gọi $A$ là biến cố con xúc xắc suất hiện mặt có số chấm lẻ. Mà con xúc xắc có các mặt có số chấm lẻ là $1, 3, 5$. Suy ra $n\left(A\right) =  2\times 3 = 6$ và $n\left(\Omega\right) = 2\times 6 = 12$. Nên $P\left(A\right) = \dfrac{6}{12} = \dfrac{1}{2}$.\\
	Vậy xác suất của biến cố con xúc xắc xuất hiện mặt có số chấm lẻ bằng $\dfrac{1}{2}$.
		}

\begin{bt}%[Đề thi Học Kì 1 T11, THPT Nguyễn Trung Trực, HCM 2018]%[1D3K3-3]
	Tìm số hạng đầu và công sai của cấp số cộng $(u)_{n}$, biết $\heva{&u_{7} - u_{1} = 12\\&3u_{3} - 2u_{5} = 7.}$
	\loigiai{Gọi $d$ là công sai của cấp số cộng thỏa mãn bài toán. Ta có $u_{n} =  u_{1} + \left(n - 1\right)d$ với $\forall n\geq 2, n\in\mathbb{N}$.\\
Khi đó 
$$\heva{&u_{7} - u_{1} = 12\\&3u_{3} - 2u_{5} = 7}\Leftrightarrow \heva{&u_{1} + 6d  - u_{1} = 12\\&3u_{1} + 6d - 2u_{1} - 8d = 7}\Leftrightarrow \heva{&d = 2\\&u_{1} - 2d = 7}\Leftrightarrow \heva{&d = 2\\&u_{1} = 11}$$
Vậy số hạng đầu $u_1 = 11$ và công sai $d = 2$ thỏa mãn bài toán.		
	}
\end{bt}

\begin{bt}%[Đề thi Học Kì 1 T11, THPT Nguyễn Trung Trực, HCM 2018]%[1H1B2-2]
	Tìm ảnh của điểm $A\left(-2; 3\right)$ qua phép tịnh tiến $\mathrm{T}_{\overrightarrow{v}}$ với $\overrightarrow{v}\left(4;5\right)$.
	\loigiai{Gọi $B\left(x_{0}; y_{0}\right)$ là ảnh của điểm $A$ qua $\mathrm{T}_{\overrightarrow{v}}$. Do đó $\mathrm{T}_{\overrightarrow{v}}\left(A\right) = B$ suy ra $\overrightarrow{AB} = \overrightarrow{v}\quad (*)$.\\
	Ta có  $\overrightarrow{AB} = \left(x_{0} + 2; y_{0} - 3\right)$ từ $(*)$ suy ra $\heva{&x_{0} + 2 = 4\\&y_{0} - 3 = 5}\Leftrightarrow \heva{&x_{0} = 2\\&y_{0}  = 8.}$\\
	Vậy tọa độ ảnh của điểm $A$ là điểm $B\left(2;8\right)$. 
		
		
		
	}
\end{bt}
\begin{bt}%[Đề thi Học Kì 1 T11, THPT Nguyễn Trung Trực, HCM 2018]%[1H2B1-2]%[1H2K3-2]%[1H2K3-4]
Cho hình chóp  $S.ABCD$ với $ABCD$ là hình thang có $AD$ là đáy lớn. Gọi $M$, $N$, $K$ lần lượt là trung điểm của $SA$, $SD$, $CD$.
\begin{enumerate}
	\item Xác định giao tuyến của hai mặt phẳng $\left(SAB\right)$ và $\left(SCD\right)$.
	\item Chứng minh $MN\parallel \left(SBC\right)$.
	\item  Xác định thiết diện của hình chóp $S.ABCD$ khi cắt bởi mặt phẳng $\left(MNK\right)$. Thiết diện là hình gì? 	
\end{enumerate}

	\loigiai{
	\begin{enumerate}
		\item \immini{
			Do giả thiết, ta gọi $E$ là giao điểm của $AB$ và $CD$ suy ra\\ $E\in\left(SAB\right)\cap \left(SCD\right)$. Nên $SE$ là giao tuyến của hai mặt phẳng $\left(SAB\right)$ và $\left(SCD\right)$.
			\item  Dễ thấy $MN$ là đường trung bình của tam giác $SAD$ nên $MN\parallel AD\quad (1)$.\\
			Mặt khác $ABCD$ là hình thang nên $AD\parallel BC\quad (2)$.\\
			Từ $(1)$ và $(2)$ suy ra $MN\parallel BC$. Mà $BC\subset \left(SBC\right)$ nên $MN\parallel \left(SBC\right)$.
			\item Theo chứng minh trên $MN\parallel AD$ do đó mặt phẳng $\left(MNK\right)$ cắt mặt phẳng $\left(ABCD\right)$ theo giao tuyến $d$ đi qua $K$ và song song với $AD$. 
				}{\begin{tikzpicture}[scale=0.8]
		\tkzInit[xmin=- 8, xmax= 8, ymin =- 8, ymax= 8]
		\clip (-3.5,-3.5) rectangle (6.5,7);	
		\tkzDefPoints{0/0/A, 2/-1.5/B, 6/0/D, 5/-1.5/C, 2/5/S}
		\tkzDefMidPoint(S,A)\tkzGetPoint{M}
		\tkzDefMidPoint(C,D)\tkzGetPoint{K}
		\tkzDefMidPoint(S,D)\tkzGetPoint{N}

		}
	Trong mặt phẳng $\left(ABCD\right)$ qua $K$ kẻ $d$ song song với $AD$ và gọi $H$ là giao điểm của  $d$ và $AD$.\\
 Khi đó mặt phẳng $\left(MNK\right)$ cắt hình chóp $S.ABCD$ theo thiết diện là tứ giác $MNKH$.
	\end{enumerate}
}
