\begin{ex}%[Đề thi vào 10, Chuyên Hoàng Lê Kha Tây Ninh, 2017]%[9D4K6]
Tìm $m$ để phương trình $x^2-2(m-1)x+m^2-3m=0 $ có hai nghiệm	 phân biệt $x_1$~và~$x_2$ sao cho $T=x_1^2+x_2^2 -(m-1)(x_1+x_2)+m^2-3m$ đạt giá trị nhỏ nhất.
    
\loigiai
    { Phương trình có hai nghiệm phân biệt khi $\Delta > 0 \Leftrightarrow m>-1  $
    	
    Theo định lý Vi-ét  ta có $\heva{x_1+x_2&=2(m-1)\\x_1x_2&=m^2-3m}$
    
    $T=x_1^2+x_2^2 -(m-1)(x_1+x_2)+m^2-3m$
    
    $=(x_1+x_2)^2 -2x_1x_2 -(m-1)(x_1+x_2)+m^2-3m$
    
    $=4(m-1)^2-2(m^2-3m)-2(m-1)(m-1)+m^2-3m$
    
    $=m^2+m+2$
    
    $=\left(m+\dfrac{1}{2}\right)^2+\dfrac{7}{4} \geq \dfrac{7}{4}$
    
    Dấu $"="$ xảy ra khi $m=-\dfrac{1}{2}$
    
    Vậy khi $m=-\dfrac{1}{2}$ thì $T$ đạt giá trị nhỏ nhất. 
    
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Hoàng Lê Kha Tây Ninh, 2017] [9D3G6]
Giải hệ phương trình $\heva{&2x^2+4x+y^3+3=0\\&x^2y^3+y=2x}$.
	\loigiai
	{$\heva{&2x^2+4x+y^3+3=0\ (1)\\&x^2y^3+y=2x\ (2)}$. 
		
		Nhân hai vế của phương trình (2) rồi cộng vế theo vế vào phương trình (1) ta được phương trình
		
		$2x^2+2x^2y^3+2y+3=0$
		
		$\Leftrightarrow 2x^2(1+y)(y^2-y+1)+(y+1)(y^2-y+3)=0$
		
		$\Leftrightarrow (y+1)[2x^2(y^2-y+1)+(y^2-y+3)]=0$
		
		$\Leftrightarrow y=-1$
		
		Thay $y=-1$ vào (2) ta được
		
		$x^2+2x+1=0  \Leftrightarrow x=-1$
		
		Vậy nghiệm của hệ phương trình đã cho là $(-1;-1)$
	}
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên Hoàng Lê Kha Tây Ninh, 2017]%[9H5G4]
Cho $A$ là điểm cố định trên đường tròn $(O)$, bán kính $R$. Hai dây cung thay đổi $AB, AC$ của đường tròn $(O)$, thỏa; $AB.AC=2\sqrt{2}R^2$ ($B$ khác $C$), kẻ $AH$ vuông góc $BC$ ($H$ thuộc $BC$).
\begin{enumerate}[a)]
	\item Chứng minh $AH=R\sqrt{2}$.
	\item Gọi $D$ và $K$ lần lượt là hình chiếu vuông góc của $H$ trên $AB, AC$. Chứng minh diện tích tam giác $ABC$ bằng hai lần diện tích tam giác $ADK$.
\end{enumerate}
	\loigiai { 	
			\begin{center}
			\begin{tikzpicture}[scale=0.4]
			\tkzDefPoint(0,0){O} \tkzDefPoint(5,0){Q}
			\tkzDefPoint[label=below:,
			shift={(0,0)}](-30:5){C}
			\begin{scope}[shift=(O)]
			\tkzDefPoint(100:5){A}
			\tkzDefPoint(210:5){B}
			\end{scope}
			\tkzCalcLength[cm ](O,Q)\tkzGetLength{rOQ}
			\tkzDrawCircle[color=red](O,A)
			\tkzDrawSegments[blue](A,B 	B,C C,A)
			\tkzDrawAltitude[color=blue](B,C)(A)
			\tkzGetPoint{H}
			\tkzDrawAltitude[color=blue](A,B)(H)
			\tkzGetPoint{D}
			\tkzDrawAltitude[color=blue](A,C)(H)
			\tkzGetPoint{K}
			\tkzLabelPoints[below](B,C,H)
			\tkzLabelPoints[above](O,A,D,K)
			\tkzMarkRightAngle(A,D,H)
			\tkzMarkRightAngle(A,K,H)
			\end{tikzpicture}
		\end{center}
			
			
			\textbf{a)}  Ta có $\dfrac{AB}{\sin C}=\dfrac{AC}{\sin B}=2R$
			
			Theo giả thiết $AB.AC=2\sqrt{2}R^2$ suy ra $\sin B.\sin C=\dfrac{\sqrt{2}}{2}$
			
		$\Rightarrow \dfrac{AH}{AC} . \dfrac{AH}{AB}=\dfrac{\sqrt{2}}{2}$
		
		$\Rightarrow AH=R\sqrt{2} $ (đpcm)
		
			\textbf{b)}  Ta có $AD.AB=AH^2=AK.AC$ Suy ra $\bigtriangleup ADK \sim \bigtriangleup ACB$ với tỉ số
			 $k^2=\dfrac{AD}{AC}.\dfrac{AK}{AB} $
			 
			 $\Rightarrow k^2=\dfrac{AD}{AC}\dfrac{AK}{AB}=\dfrac{S_{ADK}}{S_{ABC}}$
			 
			 Mặt khác $AD.AB.AK.AC=AH^2 \Rightarrow \dfrac{AD.AK}{AC.AB}=\dfrac{AH^4}{(AC.AB)^2}=\dfrac{4R^2}{8R^2}=\dfrac{1}{2}=\dfrac{S_{ADK}}{S_{ABC}} $
			 
			 $\Rightarrow S_{ABC}=2S_{ADK} $ (đpcm)


	}
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên Hoàng Lê Kha Tây Ninh, 2017]%[9H5G4]
Cho tam giác $ABC$ nhọn ($AB<AC$) nội tiếp trong đường tròn $(O)$. Gọi $D$ là điểm chính giữa cung lớn $BC$. Gọi $E,F$ lần lượt là chân đường vuông góc kẻ từ $D$ đến đường phân giác trong góc $B$ và đường phân giác trong góc $C$ của tam giác $ABC$. Chứng minh trung điểm $H$ của $EF$ cách đều $B$ và $C$.
 	\loigiai
	{	\begin{center}
			\begin{tikzpicture}[scale=0.6]
			\tkzDefPoint(0,0){O} \tkzDefPoint(5,0){Q}
			\tkzDefPoint[label=below:,
			shift={(0,0)}](-30:5){C}
			\begin{scope}[shift=(O)]
			\tkzDefPoint(125:5){A}
			\tkzDefPoint(210:5){B}
						\tkzDefPoint(90:5){D}
									\tkzDefPoint(-90:5){D_1}
			
			\end{scope}

			\tkzDrawSegments[blue](A,B 	B,C C,A)
			
			
			
			
		
			\tkzDrawBisector[color=black](A,B,C)

				\tkzDrawBisector[color=black](A,C,B)

				\tkzDrawAltitude[color=blue](B,Q)(D)

				\tkzInterLL(C,M)(B,D)
			\tkzGetPoint{P}
			
			\tkzMarkRightAngle(D,E,Q)
		\tkzDrawAltitude[color=blue](C,P)(D)
		\tkzGetPoint{F}

	
		\tkzMarkRightAngle(D,F,C)
		\tkzDefMidPoint(E,F)
		\tkzGetPoint{H}
		\tkzDefMidPoint(B,C)
		\tkzGetPoint{K}
		\tkzInterLL(F,K)(A,C)
		\tkzGetPoint{J}
			\tkzDrawSegments[blue](A,B 	B,C C,A B,D J,K C,J D,D_1 E,K)
		\tkzDrawSegments[black](C,F E,F)
		\tkzDrawPoints(A,B,C,D,O,Q,E,P,F,H,K,J,D_1)
		\tkzLabelPoints[below](B,C,E,K,P,F,D_1)
		\tkzLabelPoints[above](A,D,J)
		\tkzLabelPoints[right](Q)
		\tkzLabelPoints[left](H)
		\tkzLabelPoints[right](O)

		 Gọi $P$ là giao điểm $BD$ và phân giác  trong góc $C$. $Q$ là giao điểm của $AC$ và phân giác trong của góc $B$, $K$ là trung điểm $BC$, $J$ là giao điểm của $FK$ và $AC$, $D_1$ là điểm đối xứng với $D$ qua $O$.
		
		Ta có $\widehat{BDD_1}=\dfrac{1}{2}\widehat{BAC}$.
		
		Mà $\dfrac{1}{2}(\widehat{BAC} +\widehat{ABC}+\widehat{ACB})=90^\circ$
		
		Mặt khác $\widehat{BDQ}+\widehat{BDD_1}+\widehat{FDK}=90^\circ \Leftrightarrow \widehat{BDQ}+\dfrac{1}{2}	\widehat{BAC} + \dfrac{1}{2} \widehat{ABC}=90^\circ$
		
		Suy ra $\widehat{DBQ}=\dfrac{1}{2}\widehat{ACB}$
		
		Mà $DFKB$ nội tiếp $\Rightarrow \widehat{DKF}=\widehat{DBQ}=\dfrac{1}{2} \widehat{BCA}$
		
		Ta có: $\widehat{DKF}+\widehat{JKC}=\widehat{KJC}+\widehat{JCP}  (=90^\circ)$
		
		Suy ra  $\widehat{JKC}=\widehat{KJC}$
		
		Suy ra tam giác $JKC$ cân tại $C$ có $CP$ là phân giác cũng là đường cao
		
	Vì $CP$ vuông góc với $FK$,  $CP$ vuông góc $DE$ 
	
	$\Rightarrow DE \parallel EK$
	
	Chứng minh tương tự  ta được $DF \parallel EK$ 
	
	Từ đó suy ra $DFKE$ là hình bình hành 
	
Mà $H$ là trung điểm $EF$

$\Rightarrow H$ là trung điểm $DK$


$\Rightarrow D, H, K$ thẳng hàng

Lại có $DK$ là đường trung trực đoạn $BC$ 

Nên $H$ cũng nằm trên trung trực đó

Vậy $H$ cách đều $B$ và $C$.

	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Hoàng Lê Kha Tây Ninh, 2017]%[9D5G1]
Cho $x,y$ là các số thực dương bé hơn $1$. Tìm giá trị lớn nhất của biểu thức
 $$Q=\dfrac{xy(1-x-y)}{(x+y)(1-x)(1-y)}$$.
\loigiai
{
	Xét $x+y\geq 1 \Rightarrow Q\leq 0$
	
	Xét $x+y<1$, đặt	$z=1-x-y \Rightarrow z>0$
	
Khi đó $Q=\dfrac{xyz}{(x+y)(y+z)(x+z)} \leq \dfrac{1}{8}$ (Theo AM-GM)

Dấu $"="$ xảy ra khi $x=y=\dfrac{1}{3}$
}
\end{ex}
