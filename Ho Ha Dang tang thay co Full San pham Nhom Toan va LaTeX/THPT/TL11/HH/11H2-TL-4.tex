\subsection{Tóm tắt lý thuyết}
\subsubsection{Định nghĩa}
\begin{dn}
	Hai mặt phẳng $(\alpha)$ và $(\beta)$ gọi là song song với nhau nếu chúng không có điểm chung, khi đó ta kí hiệu $(\alpha) \parallel (\beta)$.
\end{dn}

\begin{hq}
	Nếu đường thẳng $d$ song song với mặt phẳng $(\alpha)$ thì qua $d$ có duy nhất một mặt phẳng song song với $(\alpha)$.
\end{hq}

\begin{hq}
	Hai mặt phẳng phân biệt cùng song với mặt phẳng thứ ba thì song song với nhau.
\end{hq}

\begin{hq}
	Cho điểm $A$ không nằm trên mặt phẳng $(\alpha)$. Mọi đường thẳng đi qua $A$ và song song với $(\alpha)$ đều nằm trong mặt phẳng qua $A$ và song song với $(\alpha)$.
\end{hq}

\subsubsection{Hình chóp cụt}
\begin{dn}
	Cho hình chóp $S.A_1A_2\ldots A_n$. Một mặt phẳng $(P)$ song song với mặt đáy của hình chóp và không đi qua đỉnh lần lượt cắt các cạnh $SA_1$, $SA_2$, \ldots, $SA_n$ tại $A'_1,A'_2,\ldots,A'_n$. Hình tạo thành bởi hai đa giác $A'_1A'_2\ldots A'_n$, $A_1A_2\ldots A_n$ và các tứ giác $A_1A_2A'_2A'_1$, $A_2A_3A'_3A'_3$,\ldots, $A_nA_1A'_1A'_n$ gọi là \textit{hình chóp cụt}.

		\item Đáy $A_1A_2\ldots A_n$ của hình chóp gọi là đáy lớn của hình chóp cụt.

		\item Ta gọi tên hình chóp cụt theo đa giác đáy của nó (chóp cụt tam giác, tứ giác,\ldots).
	\end{itemize}
