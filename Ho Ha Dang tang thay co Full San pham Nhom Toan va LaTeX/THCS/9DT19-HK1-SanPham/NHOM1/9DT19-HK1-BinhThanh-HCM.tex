\begin{bt}%[Dự án 9DT19-HK1]%[Phạm Tuấn]%[9D1B8] 
Rút gọn biểu thức
$\dfrac{\sqrt{x}-1}{\sqrt{x}-3}-\dfrac{\sqrt{x}}{\sqrt{x}+3}-\dfrac{7\sqrt{x}+3}{x-9}$.
\loigiai{
Điều kiện xác định: $x\geqslant 0$ và $x\ne 9$.
\begin{align*}
\dfrac{\sqrt{x}-1}{\sqrt{x}-3}-\dfrac{\sqrt{x}}{\sqrt{x}+3}-\dfrac{7\sqrt{x}+3}{x-9}&=\dfrac{(\sqrt{x}-1)(\sqrt{x}+3)}{x-9}-\dfrac{\sqrt{x}(\sqrt{x}-3)}{x-9}-\dfrac{7\sqrt{x}+3}{x-9} \\
&=\dfrac{x+2\sqrt{x}-3-x+3\sqrt{x}-7\sqrt{x}-3}{x-9} \\
&=-\dfrac{2(\sqrt{x}+3)}{x-9} \\
&=-\dfrac{2}{\sqrt{x}-3}.
\end{align*}
}
\end{bt}

\begin{bt}%[Dự án 9DT19-HK1]%[Phạm Tuấn]%[9D1B8] 
Giải phương trình: $\sqrt{8x-4}-12\sqrt{\dfrac{2x-1}{9}}+\sqrt{18x-9}=3$.
\loigiai{
Điều kiện xác định: $x \geqslant \dfrac{1}{2}$. Ta có 
\begin{align*}
& \sqrt{8x-4}-12\sqrt{\dfrac{2x-1}{9}}+\sqrt{18x-9}=3 \\
\Leftrightarrow~ & 2\sqrt{2x-1}-4\sqrt{2x-1}+3\sqrt{2x-1}=3 \\
\Leftrightarrow~ & \sqrt{2x-1}=3 \\
\Leftrightarrow~ & 2x-1=9 \\
\Leftrightarrow~ & x=5 \textrm{ (thỏa mãn)}. 
\end{align*}
Vậy tập nghiệm của phương trình là $S=\{5\}$. 
}
\end{bt}

\begin{bt}%[Dự án 9DT19-HK1]%[Phạm Tuấn]%[9H2B6] 
Từ điểm $M$ ở ngoài đường tròn $(O;R)$, vẽ hai tiếp tuyến $MA$, $MB$ với đường tròn $(O)$ ($A,B$ là hai tiếp điểm) . $OM$ cắt $AB$ tại $H$.
\begin{enumerate}
\item Chứng minh $OM \perp AB$ và $OH \cdot OM=R^2$. 
\item Vẽ đường kính $AC$ của đường tròn $(O)$, $MC$ cắt $(O)$ tại $D$. Chứng minh $\Delta ACD$ vuông và $MH \cdot MO=MD \cdot MC$. 
\item Gọi $K$ là giao điểm của $MC$ và $AB$, $AD$ cắt $OM$ tại $I$. Chứng minh $KI \parallel AC$. 
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzInit[xmax=6,ymax=4,xmin=-4,ymin=-2]
\tkzClip
\tkzDefPoints{-3/1/M,2/1/O,4/3/E}
\tkzTangent[from=M](O,E) \tkzGetPoints{B}{A}
\tkzDefMidPoint(A,B) \tkzGetPoint{H}
\tkzDefPointBy[symmetry=center O](A) \tkzGetPoint{C}
\tkzInterLL(A,B)(M,C) \tkzGetPoint{K}
\tkzDefPointBy[projection=onto M--C](A) \tkzGetPoint{D}
\tkzInterLL(M,O)(A,D) \tkzGetPoint{I}
\tkzDrawCircle(O,A)
\tkzDrawSegments(M,A M,B M,O A,C A,D O,B I,K M,C A,B O,D)
\tkzDrawPoints[fill=black,size=6](A,B,C,D,I,K,O,M,H)
\tkzLabelPoints[above](A)
\tkzLabelPoints[below](B,C)
\tkzLabelPoints[left](M)
\tkzLabelPoints[right](O)
\tkzLabelPoints[above right](K,H)
\tkzLabelPoints[below left](D)
\tkzLabelPoints[above left](I)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Ta có $MA=MB$ và $OA=OB$ nên $AO$ là đường trung trực của $AB$, suy ra  $OM \perp AB$. Tam giác $MAO$ vuông tại $A$ và có $AH$ là đường cao suy ra $OH \cdot OM = OA^2=R^2$. 
\item  Tam giác $ACD$ có trung tuyến $OD=\dfrac{1}{2}AC$ nên $\widehat{ADC}=90^\circ$.  \\
Tam giác $MAO$ vuông tại $A$ và có $AH$ là đường cao suy ra $MH \cdot MO = MA^2 \quad (1)$. \\
Tam giác  $MAC$ vuông tại $A$ và có $AD$ là đường cao suy ra $MD \cdot MC = MA^2 \quad (2)$. \\
Từ $(1)$ và $(2)$ suy ra $MH \cdot MO=MD \cdot MC$. 
\item Tam giác $AMK$ có $MH$ và $AD$ là các đường cao nên $I$ là trực tâm của tam giác $AMK$, suy ra $KI \perp MA$, mà $AC \perp MA$, suy ra $KI\parallel AC$. 
\end{enumerate}
}
\end{bt}
