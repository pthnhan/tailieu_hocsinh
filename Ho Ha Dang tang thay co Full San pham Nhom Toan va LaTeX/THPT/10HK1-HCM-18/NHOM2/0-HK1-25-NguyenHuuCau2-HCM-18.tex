\begin{bt}%[HK1-Nguyễn Hữu Cầu 2-Hồ Chí Minh]%[Đỗ Viết Lân]%[0D3B2-2]
Giải phương trình $\sqrt{x^2 - 2x + 1} + x = 1$.
\loigiai{
	Ta có $\sqrt{x^2 - 2x + 1} + x = 1 \Leftrightarrow \sqrt{(x-1)^2} + x - 1 = 0 \Leftrightarrow |x - 1| + x - 1 = 0$.
	\begin{itemize}
	\item Nếu $x \geq 1$ thì $|x - 1| = x - 1$. Phương trình trở thành $x - 1 + x - 1 = 0 \Leftrightarrow x = 1$.
	\item Nếu $x < 1$ thì $|x - 1| = 1 - x$. Phương trình trở thành $1 - x + x - 1 = 0 \Leftrightarrow 0\cdot x = 0$. Phương trình này nghiệm đúng với mọi $x < 1$.
	\end{itemize}
	Vậy tập nghiệm của phương trình là $S = (-\infty;1]$.
}
\end{bt}

\begin{bt}%[HK1-Nguyễn Hữu Cầu 2-Hồ Chí Minh]%[Đỗ Viết Lân]%[0D3K3-4]
Giải hệ phương trình $\heva{&x^3 = 2y-1\quad (1)\\&y^3 = 2x-1\quad (2)}$.
\loigiai{
	Lấy $(1)$ trừ (2) theo vế ta có
	\begin{eqnarray*}
	&&x^3 - y^3 = 2y - 2x \\
	&\Leftrightarrow &(x-y)(x^2 + xy + y^2) + 2(x-y) = 0\\
	&\Leftrightarrow &(x-y)(x^2+xy+y^2+2) = 0.
	\end{eqnarray*}
	mà $x^2 + xy + y^2 + 2 = x^2 + 2\cdot x \cdot \dfrac{y}{2} + \dfrac{y^2}{4} + \dfrac{3y^2}{4} + 2 = \left( x + \dfrac{y}{2} \right)^2 + \dfrac{3y^2}{4} + 2 > 0$ với mọi $x,y$\\
	nên $x - y = 0 \Leftrightarrow x = y$.\\
	Thay $x = y$ vào $(1)$ ta có $x^3 = 2x - 1 \Leftrightarrow (x-1)(x^2 - x - 1) = 0 \Leftrightarrow \hoac{&x = 1\\&x = \dfrac{1 \pm \sqrt{5}}{2}}$.\\
	Vậy nghiệm hệ phương trình là $\heva{&x=1\\&y=1};\heva{&x=\dfrac{1 + \sqrt{5}}{2}\\&y=\dfrac{1 + \sqrt{5}}{2}};\heva{&x=\dfrac{1 - \sqrt{5}}{2}\\&y=\dfrac{1 - \sqrt{5}}{2}}$.
}
\end{bt}

\begin{bt}%[HK1-Nguyễn Hữu Cầu 2-Hồ Chí Minh]%[Đỗ Viết Lân]%[0D3B2-5]
Tìm tham số $m$ để phương trình $(m-1)x^2 - (2m+1)x + m + 1 = 0$ có hai nghiệm phân biệt $x_1,x_2$ sao cho khoảng cách giữa hai nghiệm đó trên trục số thực bằng $\sqrt{13}$, tức là $|x_1 - x_2| = \sqrt{13}$.
\loigiai{
	Để phương trình có hai nghiệm phân biệt $x_1$ và $x_2$ thì 
	$$\heva{&a \neq 0\\&\Delta > 0} \Leftrightarrow \heva{&m \neq 1\\&(2m+1)^2 - 4(m+1)(m-1)>0} \Leftrightarrow \heva{&m \neq 1\\&4m+5 > 0} \Leftrightarrow \heva{&m\neq 1\\& m > -\dfrac{5}{4}.}$$
	Theo định lý Vi-ét ta có $\heva{&x_1 + x_2 = \dfrac{2m+1}{m-1}\\&x_1 x_2 = \dfrac{m+1}{m-1}}$.\\
	Ta có
	\begin{eqnarray*}
	&&|x_1 - x_2| = \sqrt{13}\\
	&\Leftrightarrow &(x_1 - x_2)^2 = 13\\
	&\Leftrightarrow &(x_1 + x_2)^2 - 4x_1x_2 = 13\\
	&\Leftrightarrow &\left( \dfrac{2m+1}{m-1} \right)^2 - 4\cdot \dfrac{m+1}{m-1} = 13\\
	&\Leftrightarrow &(2m+1)^2 - 4(m-1)(m+1) = 13(m-1)^2\\
	&\Leftrightarrow &4m^2+4m+1-4m^2+4 = 13m^2-26m+13\\
	&\Leftrightarrow &13m^2 - 30m + 8 = 0\\
	&\Leftrightarrow &\hoac{&m = 2\\&m = \dfrac{4}{13}.} \text{ (thỏa mãn)}
	\end{eqnarray*}
	Vậy $m=2$ và $m = \dfrac{4}{13}$.
}
\end{bt}

\begin{bt}%[HK1-Nguyễn Hữu Cầu 2-Hồ Chí Minh]%[Đỗ Viết Lân]%[0D6B2-3]
Cho $\sin \left( \dfrac{33\pi}{2} + \alpha \right) = \dfrac{1}{5}$, với $-\dfrac{\pi}{2} < \alpha < 0$. Hãy tính giá trị của $\cot \left( \dfrac{11\pi}{2} - \alpha \right)$.
\loigiai{
	Ta có $\dfrac{1}{5} = \sin \left( \dfrac{33\pi}{2} + \alpha \right) = \sin \left( 16\pi + \dfrac{\pi}{2} + \alpha \right) = \sin \left( \dfrac{\pi}{2} + \alpha \right) = \cos \alpha$.\\
	Suy ra $1+ \tan^2 \alpha = \dfrac{1}{\cos^2 \alpha} = \dfrac{1}{25} \Rightarrow \tan^2\alpha = 24$.\\
	Vì $-\dfrac{\pi}{2} < \alpha < 0$ nên $\tan \alpha  < 0 \Rightarrow \tan\alpha = -2\sqrt{6}$.\\
	Do đó $\cot \left( \dfrac{11\pi}{2} - \alpha \right) = \cot \left( 5\pi + \dfrac{\pi}{2} - \alpha \right) = \cot \left( \dfrac{\pi}{2} - \alpha \right) = \tan \alpha = -2\sqrt{6}$.
}
\end{bt}

\begin{bt}%[HK1-Nguyễn Hữu Cầu 2-Hồ Chí Minh]%[Đỗ Viết Lân]%[0D6K3-2]
Không dùng máy tính, hãy tính giá trị chính xác của biểu thức $A = \sin^2 \left(\dfrac{\pi}{16}\right) + \cos^2 \left(\dfrac{\pi}{8}\right)$.
\loigiai{
	Ta có $\dfrac{\sqrt{2}}{2} = \cos \dfrac{\pi}{4} = 2 \cos^2 \dfrac{\pi}{8} - 1 \Rightarrow \cos^2 \dfrac{\pi}{8} = \dfrac{2+\sqrt{2}}{4}$.\\
	Do đó $2\sin^2 \left(\dfrac{\pi}{16}\right) = 1 - \cos\left(\dfrac{\pi}{8}\right) = 1 - \sqrt{\dfrac{2+\sqrt{2}}{4}} \Rightarrow \sin^2 \left(\dfrac{\pi}{16}\right) = \dfrac{2 - \sqrt{2+\sqrt{2}}}{4}$.\\
	Vậy $A = \dfrac{4+\sqrt{2}-\sqrt{2+\sqrt{2}}}{4}$.
}
\end{bt}

\begin{bt}%[HK1-Nguyễn Hữu Cầu 2-Hồ Chí Minh]%[Đỗ Viết Lân]%[0D6G3-4]
Cho tam giác $ABC$. Chứng minh rằng $\cos A - \cos B - \cos C = 1 - 4 \sin \dfrac{A}{2}\cos \dfrac{B}{2} \cos \dfrac{C}{2}$.
\loigiai{
	Ta có
	\begin{eqnarray*}
	&&\cos A - \cos B - \cos C\\
	&=&\cos A - (\cos B + \cos C)\\
	&=&1 - 2\sin^2\dfrac{A}{2} - 2\cos \dfrac{B+C}{2}\cos\dfrac{B-C}{2}\\
	&=&1 - 2\sin\dfrac{A}{2}\cos\dfrac{B+C}{2} - 2\sin\dfrac{A}{2}\cos\dfrac{B-C}{2}\\
	&=&1 - 2\sin\dfrac{A}{2}\left(\cos\dfrac{B+C}{2}+\cos\dfrac{B-C}{2}\right)\\
	&=&1 - 4\sin\dfrac{A}{2}\cos \dfrac{\dfrac{B+C}{2}+\dfrac{B-C}{2}}{2}\cos \dfrac{\dfrac{B+C}{2}-\dfrac{B-C}{2}}{2}\\
	&=&1 - 4 \sin \dfrac{A}{2}\cos \dfrac{B}{2} \cos \dfrac{C}{2}.
	\end{eqnarray*}
}
\end{bt}

\begin{bt}%[HK1-Nguyễn Hữu Cầu 2-Hồ Chí Minh]%[Đỗ Viết Lân]%[0H2K3-1]
Cho tam giác $ABC$ có $AB > AC$ và số đo góc $A$ bằng $120^\circ$. Gọi $M$ là trung điểm của $BC$. Biết rằng $BC = \sqrt{7}$ và $AM = \dfrac{\sqrt{3}}{2}$. Hãy tính độ dài các cạnh $AB$ và $AC$.
\loigiai{
	Theo công thức đường trung tuyến ta có $AM^2 = \dfrac{AB^2 + AC^2}{2} - \dfrac{BC^2}{4} \Leftrightarrow AB^2 + AC^2 = 5$.\\
	Theo định lý Cô-sin ta có $BC^2  = AB^2 + AC^2 - 2 \cdot AB\cdot AC \cdot \cos A = 5 + AB\cdot AC  \Leftrightarrow AB\cdot AC = 2$.\\
	Vậy $\heva{&AB^2 + AC^2 = 5\\&AB\cdot AC = 2} \Leftrightarrow\heva{&(AB+ AC)^2 = 9\\&AB\cdot AC = 2} \Leftrightarrow \heva{&AB+ AC = 3\\&AB\cdot AC = 2} \Leftrightarrow \heva{&AB = 2\\& AC = 1}$   (Vì $AB>AC$).
}
\end{bt}

\begin{bt}%[HK1-Nguyễn Hữu Cầu 2-Hồ Chí Minh]%[Đỗ Viết Lân]%[0H2K2-3]%[0H2K2-3]
Trong mặt phẳng với hệ tọa độ $Oxy$, cho tam giác $ABC$ có $A(2;1)$, $B(3;-1)$ và $C(-1;2)$.
\begin{enumerate}
\item Tìm tọa độ điểm $H$ là chân đường cao hạ từ đỉnh $A$ của tam giác $ABC$.
\item Tìm tọa độ điểm $I$ là giao điểm của hai đường thẳng $OA$ và $BC$.
\end{enumerate}
\loigiai{
	Ta có $\vec{BC} = (-4;3)$.
	\begin{enumerate}
	\item Gọi $H(a;b) \Rightarrow \vec{AH} = (a-2;b-1)$ và $\vec{BH} = (a-3;b+1)$. Do $H$ là chân đường cao từ $A$ nên ta có $\heva{&\vec{AH}\perp \vec{BC}\\&H \in BC}$.\\
	$\vec{AH}\perp \vec{BC} \Leftrightarrow \vec{AH} \cdot \vec{BC} = 0 \Leftrightarrow -4(a-2)+3(b-1) = 0 \Leftrightarrow -4a+3b = -5$.\\
	$H \in BC \Leftrightarrow \vec{BH} = k\cdot \vec{BC} \Leftrightarrow \heva{&a-3 = -4k\\&b+1 = 3k}$.\\
	Do đó ta có hệ $\heva{&-4a+3b = -5\\&a-3 = -4k\\&b+1 = 3k} \Leftrightarrow \heva{&a = \dfrac{7}{5}\\&b = \dfrac{1}{5}\\&k = \dfrac{2}{5}}$. Vậy $H\left(\dfrac{7}{5};\dfrac{1}{5}\right)$.
	\item Gọi $I(a;b)$. Khi đó $\vec{OI} = (a;b)$ và $\vec{BI} = (a-3;b+1)$. Do $I$ là giao điểm của $OA$ và $BC$ nên $\vec{OI}$ cùng phương với $\vec{OA}$ và $\vec{BI}$ cùng phương với $\vec{BC}$.\\
	Từ đó ta có $\heva{&\dfrac{a}{2} = \dfrac{b}{1}\\& \dfrac{a-3}{-4} = \dfrac{b+1}{3}} \Leftrightarrow \heva{&a = 2b\\&\dfrac{2b-3}{-4} = \dfrac{b+1}{3}} \Leftrightarrow \heva{&a = 1\\&b = \dfrac{1}{2}}$. Vậy $I\left(1;\dfrac{1}{2}\right)$.
	\end{enumerate}
}
\end{bt}

\begin{bt}%[HK1-Nguyễn Hữu Cầu 2-Hồ Chí Minh]%[Đỗ Viết Lân]%[0H2G2-5]
Cho hai véc-tơ $\vec{u}$ và $\vec{v}$. Chứng minh bất đẳng thức
$$|\vec{u}\cdot \vec{v}| \leq |\vec{u}|\cdot|\vec{v}|.$$
Hãy áp dụng bất đẳng thức trên để chứng minh mệnh đề sau đây đúng với mọi số thực $x$
$$0 \leq 3\cos x - 4 \sin x + 5 \leq 10.$$
\loigiai{
	Ta có $|\vec{u}\cdot \vec{v}| = \left| |\vec{u}|\cdot |\vec{v}|\cdot \cos\left(\vec{u},\vec{v}\right) \right| = |\vec{u}|\cdot |\vec{v}|\cdot \left|\cos\left(\vec{u},\vec{v}\right)\right| \leq |\vec{u}|\cdot|\vec{v}|.$\\
	Lấy $\vec{u} = (3;-4)$ và $\vec{v} = (\cos x; \sin x)$. Khi đó $\vec{u}\cdot\vec{v} = 3\cos x - 4 \sin x$, $|\vec{u}| = 5$, $|\vec{v}| = \sqrt{\cos^2 x + \sin^2 x} = 1$. Áp dụng bất đẳng thức trên ta có
	$$|3\cos x - 4 \sin x| \leq 5 \Leftrightarrow -5 \leq 3\cos x - 4 \sin x \leq 5 \Leftrightarrow 0 \leq 3\cos x - 4 \sin x + 5 \leq 10.$$
}
\end{bt}
