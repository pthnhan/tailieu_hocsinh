\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, tỉnh Vũng Tàu, Vòng 1, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-TS10C-18]%[9D0B1]%[9D0B2]%[9D0B2]
	\begin{listEX}
        \item Rút gọn biểu thức: $A = \dfrac{3}{\sqrt{7} - 2} - \dfrac{14}{\sqrt{7}} + \sqrt{\left(\sqrt{7} - 2\right)^2}$.
        \item Giải phương trình $5x^2 + 2\sqrt{5}x + 1 = 0$.
        \item Giải hệ phương trình $\heva{&3x - 2y = 16 \\ &x + 5y = -23.}$
    \end{listEX}
    \loigiai{
		\begin{listEX}
			\item Ta có 
					\begin{align*}
						A &= \dfrac{3}{\sqrt{7} - 2} - \dfrac{14}{\sqrt{7}} + \sqrt{\left(\sqrt{7} - 2\right)^2} = \dfrac{3\left(7 + \sqrt{2}\right)}{7 - 4} - \dfrac{14\sqrt{7}}{7} + \left|\sqrt{7} - 2\right|\\
						&= \sqrt{7} + 2 - 2\sqrt{7} + \sqrt{7} - 2 = 0.
					\end{align*}					
			\item Ta có $5x^2 + 2\sqrt{5}x + 1 = 0 \Leftrightarrow \left(\sqrt{5}x + 1\right)^2 = 0 \Leftrightarrow \sqrt{5}x + 1 = 0 \Leftrightarrow x = -\dfrac{\sqrt{5}}{5}$.\\
					Vậy phương trình đã cho có tập nghiệm $S = \left\lbrace-\dfrac{\sqrt{5}}{5} \right\rbrace$.
			\item Ta có $\heva{&3x - 2y = 16 \\ &x + 5y = -23} \Leftrightarrow \heva{&3x - 2y = 16 \\ &3x + 15y = -69} \Leftrightarrow \heva{&17y = -85 \\ &3x - 2y = 16} \Leftrightarrow \heva{&y = -5 \\ &3x + 10 = 16} \Leftrightarrow \heva{&x = 2 \\ &y = -5.}$\\
					Vậy hệ phương trình đã cho có tập nghiệm $S = \lbrace (2; -5)\rbrace$.
		\end{listEX}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, tỉnh Vũng Tàu, Vòng 1, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-TS10C-18]%[9D0B2]%[9D0K2]
	\begin{listEX}
        \item Tìm tất cả giá trị của hệ số $a$ để hàm số $y = ax + 2$ đồng biến và đồ thị của hàm số đi qua điểm $A(1; 3)$.
        \item Cho đường thẳng $(d)\colon y = (3 - 2m)x - m^2$ và parabol $(P)\colon y = x^2$. Tìm tất cả các giá trị của tham số $m$ để $(d)$ cắt $(P)$ tại hai điểm phân biệt có hoành độ $x_1$, $x_2$ và $x_1(x_2 - 1) + 2(x_1 - x_2) = 2x_1 - x_2$.
    \end{listEX}
    \loigiai{
		\begin{listEX}
        	\item Yêu cầu bài toán $\Leftrightarrow \heva{&a > 0 \\ &a\cdot 1 + 2 = 3} \Leftrightarrow a = 1$.
        	\item Phương trình hoành độ giao điểm của $(d)$ và $(P)$ là
        			$$x^2 = (3 - 2m)x - m^2 \Leftrightarrow x^2 - (3 - 2m)x + m^2 = 0.$$
        			$(d)$ cắt $(P)$ tại hai điểm phân biệt $\Leftrightarrow \Delta > 0 \Leftrightarrow (3 - 2m)^2 - 4m^2 > 0 \Leftrightarrow 9 - 12m > 0 \Leftrightarrow m < \dfrac{3}{4}$.\\
        			Theo hệ thức Vi-ét, ta có $\heva{&x_1 + x_2 = 3 - 2m \\ &x_1x_2 = m^2.}$\\
        			Ta có $x_1(x_2 - 1) + 2(x_1 - x_2) = 2x_1 - x_2 \Leftrightarrow x_1x_2 - (x_1 + x_2) = 0 \Leftrightarrow m^2 + 2m - 3 = 0 \Leftrightarrow \hoac{&m = 1 \\ &m = -3.}$\\
        			So sánh với điều kiện, ta được $m = -3$.
    	\end{listEX}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, tỉnh Vũng Tàu, Vòng 1, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-TS10C-18]%[9D0K2]%[9D0K2]
	\begin{listEX}
        \item Một mảnh vườn hình chữ nhật có chu vi bằng $174$ m. Nếu tăng chiều rộng $5$ m và giảm chiều dài $2$ m thì diện tích mảnh vườn đó tăng thêm $215$ m$^2$. Tính chiều rộng và chiều dài ban đầu của mảnh vườn.
        \item Giải phương trình $5x^4 - 2x^2 - 3x^2\sqrt{x^2 + 2} = 4$.
    \end{listEX}
	\loigiai{
		\begin{listEX}
       		\item Gọi $x$(m) và $y$(m) lần lượt là chiều dài và chiều rộng ban đầu của mảnh vườn $(x > 2, y > 5)$.\\
       				Chu vi mảnh vườn bằng $174$ m nên ta có $x + y = \dfrac{174}{2} = 87$. \hfill $(1)$\\
       				Khi tăng chiều rộng $5$ m và giảm chiều dài $2$ m thì diện tích mảnh vườn đó tăng thêm $215$ m$^2$ nên ta có phương trình $(x - 2)(y + 5) = xy + 215 \Leftrightarrow 5x - 2y = 225$. \hfill $(2)$\\
       				Từ $(1) \,\&\, (2)$, ta có hệ phương trình $\heva{&x + y = 87 \\ &5x - 2y = 225} \Leftrightarrow \heva{&2x + 2y = 174 \\ &5x - 2y = 225} \Leftrightarrow \heva{&x = 57 \\ &y = 30}$ (thỏa mãn điều kiện).\\
       				Vậy ban đầu chiều dài và chiều rộng mảnh vườn lần lượt là $57$ m và $30$ m.
        	\item Ta có $5x^4 - 2x^2 - 3x^2\sqrt{x^2 + 2} = 4 \Leftrightarrow 5x^4 - 3x^2\sqrt{x^2 + 2} - 2(x^2 + 2) = 0$.\\
        			Đặt $t = \sqrt{x^2 + 2}$ $(t \geq \sqrt{2})$, ta được phương trình
        			\begin{align*}
        				5x^4 - 3x^2t - 2t^2 = 0 \Leftrightarrow (x^2 - u)(5x^2 + 2u) = 0 \Leftrightarrow u = x^2 \,(\text{vì} \,5x^2 + 2u > 0).
        			\end{align*}        			
        			Khi đó, ta có $x^2 = \sqrt{x^2 + 2} \Leftrightarrow x^4 = x^2 + 2 \Leftrightarrow (x^2 - 2)(x^2 + 1) = 0 \Leftrightarrow x = \pm \sqrt{2}$.\\
        			Vậy phương trình đã cho có tập nghiệm $S = \left\lbrace -\sqrt{2}, \sqrt{2}\right\rbrace$.
    	\end{listEX}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, tỉnh Vũng Tàu, Vòng 1, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-TS10C-18]%[9H0B1]%[9H0B1]%[9H0K1]%[9H0G1]
	Cho đường tròn $(O)$ có $AB$ là dây cung không đi qua tâm và $I$ là trung điểm của dây $AB$. Trên tia đối của tia $AB$ lấy điểm $M$ khác điểm $A$. Vẽ hai tiếp tuyến $MC$ và $MD$ đến $(O)$ (tiếp điểm $C$ thuộc cung nhỏ $AB$, tiếp điểm $D$ thuộc cung lớn $AB$).
	\begin{listEX}
        \item Chứng minh tứ giác $OIMD$ nội tiếp được đường tròn.
        \item Chứng minh $MD^2 = MA \cdot MB$.
        \item Đường thẳng $OI$ cắt cung nhỏ $AB$ của $(O)$ tại điểm $N$, giao điểm của hai đường thẳng $DN$ và $MB$ là $E$. Chứng minh tam giác $MCE$ cân tại $M$.
        \item Đường thẳng $ON$ cắt đường thẳng $CD$ tại điểm $F$. Chứng minh $\dfrac{1}{OI \cdot OF} + \dfrac{1}{ME^2} = \dfrac{4}{CD^2}$.
    \end{listEX}
    \loigiai{
		\begin{center}
			\begin{tikzpicture}
				\clip (-10,-4.5) rectangle (5,9);
				\tkzDefPoints{0/0/O, 4/0/T, -9/0/M}
				\tkzDefPointBy[rotation = center O angle 40](T) \tkzGetPoint{B}
				\tkzInterLC(M,B)(O,T) \tkzGetPoints{A}{B}
				\tkzDefMidPoint(A,B) \tkzGetPoint{I}				
				\tkzDrawCircle(O,T)
				\tkzTangent[from=M](O,T) \tkzGetPoints{D}{C}				
				\tkzInterLC(O,I)(O,T) \tkzGetPoints{N'}{N}
				\tkzInterLL(O,N)(C,D) \tkzGetPoint{F}
				\tkzInterLL(D,N)(M,B) \tkzGetPoint{E}
				\tkzInterLL(O,M)(C,D) \tkzGetPoint{H}
				\tkzDrawSegments(M,C M,D M,B O,I O,D O,M O,C B,D A,D D,F O,F D,N C,E)   
				\tkzMarkRightAngles(O,C,M O,D,M O,I,B) 
				\tkzDrawPoints(O,M,A,B,C,D,I,H,F,E,N,H)
				\tkzLabelPoints[above](F)
				\tkzLabelPoints[below](D)
				\tkzLabelPoints[left](M)
				\tkzLabelPoints[right](O,B)
				\tkzLabelPoints[above left](C,A)
				\tkzLabelPoints[above right](N,I)
				\tkzLabelPoints[below left](E,H)
			\end{tikzpicture}
		\end{center}
		\begin{listEX}
       	 	\item Do $I$ là trung điểm của dây cung $AB$ của đường tròn $(O)$ nên $OI \perp AB \Rightarrow \widehat{MIO} = 90^\circ$.\\
       	 			Lại có $MD$ là tiếp tuyến của đường tròn $(O)$ tại $D$ nên $\widehat{MDO} = 90^\circ$.\\
       	 			Tứ giác $OIMD$ có $\widehat{MIO} + \widehat{MDO} = 90^\circ + 90^\circ = 180^\circ$ nên nội tiếp được đường tròn.
       	 	\item Hai tam giác $MAD$ và $MBD$ có $\widehat{BMD}$ chung và $\widehat{MDA} = \widehat{MBD}$ (cùng chắn cung $AD$).\\
       	 			$\Rightarrow \triangle MAD \backsim \triangle MBD \Rightarrow \dfrac{MA}{MD} = \dfrac{MD}{MB} \Rightarrow MD^2 = MA \cdot MB$.
        	\item Ta có $ON$ là đường trung trực của đoạn thẳng $AB$ nên $NA = NB \Rightarrow \wideparen{NA} = \wideparen{NB}$.\\
        			$\Rightarrow \widehat{MED} = \dfrac{1}{2}\left(\text{sđ}\wideparen{AD} + \text{sđ}\wideparen{NB}\right) = \dfrac{1}{2}\left(\text{sđ}\wideparen{AD} + \text{sđ}\wideparen{NA}\right) = \dfrac{1}{2}\text{sđ}\wideparen{ND} = \widehat{MDN}$.\\
        			$\Rightarrow$ Tam giác $MED$ cân tại $M$ $\Rightarrow ME = MD$.\\
        			Ta lại có $MC = MD$ (tính chất hai tiếp tuyến cắt nhau).
        			$\Rightarrow ME = MC \Rightarrow$ tam giác $MCE$ cân tại $M$.
        	\item Ta có $MC = MD$ và $OC = OD$ nên $MO$ là đường trung trực của đoạn thẳng $CD$ $\Rightarrow MO \perp CD$ tại trung điểm $H$ của $CD$.\\
        			Hai tam giác $OIM$ và $OHF$ có $\widehat{OIM} = \widehat{OHF} = 90^\circ$ và $\widehat{MOF}$ chung.\\
        			$\Rightarrow \triangle OIM \backsim \triangle OHF \Rightarrow \dfrac{OI}{OH} = \dfrac{OM}{OF} \Rightarrow OI \cdot OF = OM \cdot OH = OD^2$.\\
        			Do đó $\dfrac{1}{OI \cdot OF} + \dfrac{1}{ME^2} = \dfrac{1}{OD^2} + \dfrac{1}{MD^2} = \dfrac{1}{DH^2} = \dfrac{4}{CD^2}$.
    	\end{listEX}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, tỉnh Vũng Tàu, Vòng 1, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-TS10C-18]%[9D0G7]
	Cho $a > 0$, $b > 0$ và $a + b \leq 1$. Tính giá trị nhỏ nhất của biểu thức $S = \dfrac{a}{1 + b} + \dfrac{b}{1 + a} + \dfrac{1}{a + b}$.
	\loigiai{
		Áp dụng bất đẳng thức Cauchy, ta có:
		$$\dfrac{a}{1 + b} + \dfrac{4}{9}a(1 + b) \geq 2\sqrt{\dfrac{a}{1 + b} \cdot \dfrac{4a(1 + b)}{9}} = \dfrac{4a}{3} \Rightarrow \dfrac{a}{1 + b} \geq \dfrac{8}{9}a - \dfrac{4}{9}ab.$$
		Tương tự, ta có $\dfrac{b}{1 + a} \geq \dfrac{8}{9}b - \dfrac{4}{9}ab$.\\
		Do đó $S \geq \dfrac{1}{a + b} + \dfrac{8}{9}(a + b) - \dfrac{8}{9}ab = \left(\dfrac{1}{a + b} + a + b\right) - \dfrac{a + b + 8ab}{9} \geq 2 - \dfrac{a + b + 8ab}{9}$.\\
		Ta lại có $4ab \leq (a + b)^2 \leq 1 \Rightarrow a + b + 8ab \leq 3$.\\
		Do đó, ta có $S \geq 2 - \dfrac{3}{9} = \dfrac{5}{3}$. Đẳng thức xảy ra khi $a = b = \dfrac{1}{2}$.\\
		Vậy giá trị nhỏ nhất của $S$ là $\dfrac{5}{3}$, đạt được khi $a = b = \dfrac{1}{2}$.
	}
\end{ex}
