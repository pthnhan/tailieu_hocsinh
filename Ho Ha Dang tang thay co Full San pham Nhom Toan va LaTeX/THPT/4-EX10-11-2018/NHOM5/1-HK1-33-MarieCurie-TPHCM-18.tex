\begin{bt}%[HK1 Toán 11, 2017-2018, Marie Curie, HCM]%[1D1B2]%[1D1K2]%[1D1K2]
	Giải các phương trình

a)  $(\sin x - \cos x)^2 +2\sin x +\sin 2x = 0$.

b) $\cos^22x + 2\sin^2 x - 1 +\sin x \cos 2x = 0$.

c) $\sin x . \sin 3x = 3 \cos2x -3.$
\loigiai{
a) 

$
\begin{aligned}
 \text{PT} \Leftrightarrow  & \sin^2x+\cos^2x-2\sin x \cos x +2\sin x +\sin 2x = 0  \\
\Leftrightarrow &  1 -\sin2x +2\sin x +\sin 2x = 0 \\
\Leftrightarrow & \sin x = -\dfrac{1}{2} \\
\Leftrightarrow & \hoac{& x = -\dfrac{\pi}{6}+k2\pi \\ &x = \dfrac{7\pi}{6}+k2\pi} (k \in \mathbb{Z}). \\
\end{aligned}
$

b) 


$
\begin{aligned}
 \text{PT} \Leftrightarrow  &\cos^22x-\cos2x+\sin x \cos 2x = 0  \\
\Leftrightarrow & \cos2x (\cos2x-1+\sin x) =0 \\
\Leftrightarrow & \cos2x (-2\sin^2x+\sin x) =0 \\
\Leftrightarrow & \cos2x \sin x (1-2\sin x)=0 \\
\Leftrightarrow & \hoac{&  \cos2x = 0 \\ & \sin x =0 \\ & \sin x =\dfrac{1}{2} }\Leftrightarrow \hoac{& x = \dfrac{\pi}{4}+\dfrac{k\pi}{2}  \\ & x = k\pi \\ &   x = \dfrac{\pi}{6}+ k2\pi  \\ &x = \dfrac{5\pi}{6}+ k2\pi }  (k \in \mathbb{Z}). 
\end{aligned}
$

c) 

$
\begin{aligned}
 \text{PT} \Leftrightarrow  &-\dfrac{1}{2}(\cos 4x - \cos 2x) = 3\cos 2x -3  \\
\Leftrightarrow &\cos 4x - \cos 2x + 6\cos 2x  -6 =0 \\
\Leftrightarrow & 2\cos^22x-1 + 5 \cos 2x  -6 =0  \\
\Leftrightarrow & 2\cos^22x+ 5 \cos 2x-7 = 0 \\
\Leftrightarrow & \hoac{&  \cos2x = 1 \\ & \cos 2x = -\dfrac{7}{2 } (\text{ vô nghiệm})}\Leftrightarrow x = k\pi   (k \in \mathbb{Z}). 
\end{aligned}
$
}
\end{bt}

\begin{bt}%[HK1 Toán 11, 2017-2018, Marie Curie, HCM]%[1D2K2]
Một hội đồng thi tuyển học sinh giỏi có $10$ giáo viên nữ và $15$ giáo viên nam được ban tổ chức phân công làm nhiệm vụ coi thi. Ban tổ chức muốn chọn ra $10$ giáo viên làm giám thị   $\mathrm{I}$, $10$ giáo viên làm giám thị $\mathrm{II}$  và $5$ thầy làm nhiệm vụ giám thị hành lang. Hỏi ban tổ chức có bao nhiêu cách phân công như vậy?
\loigiai{
Số cách chọn $5$ thầy làm nhiệm vụ giám thị hành lang là $\mathrm{C}_{15}^5$.

\noindent Số cách chọn $10$ giáo viên làm  giám thị   $\mathrm{I}$  là $\mathrm{C}_{20}^{10}$.

\noindent Số cách chọn $10$ giáo viên làm  giám thị   $\mathrm{II}$ là $\mathrm{C}_{10}^{10}$.

Vậy số cách phân công nhiệm vụ là $\mathrm{C}_{15}^5\mathrm{C}_{20}^{10}\mathrm{C}_{10}^{10} = 554822268.$
}
\end{bt}

\begin{bt}%[HK1 Toán 11, 2017-2018, Marie Curie, HCM]%[1D2K5]
Có $20$ viên bi được đánh số từ $1$ đến $20$ chứa trong một cái hộp. Người ta chọn ngẫu nhiên $3$ viên bi từ $20$ viên bi nói trên. Tính xác suất để $3$ viên bi được chọn có tổng $3$ số ghi trên nó là một số chẵn.
\loigiai{
Số phần tử của không gian mẫu là $|\Omega| = \mathrm{C}_{20}^3 $.

\noindent Để tổng $3$ số là một số chẵn thì hoặc cả $3$ số đó là $3$ số chẵn hoặc trong $3$ số đó có $2$ số lẻ và $1$ số chẵn.

\noindent Số cách chọn ra $3$ viên bi đều ghi số chẵn là $\mathrm{C}_{10}^3$.

 \noindent Số cách chọn ra $3$ viên bi, trong đó có $2$ số lẻ và $1$ số chẵn là
$\mathrm{C}_{10}^2\mathrm{C}_{10}^1$.

\noindent Do đó số cách chọn $3$ viên bi có tổng $3$ số là một số chẵn là $\mathrm{C}_{10}^3+ \mathrm{C}_{10}^2\mathrm{C}_{10}^1$.

\noindent Vậy xác suất để $3$ viên bi được chọn có tổng $3$ số ghi trên nó là một số chẵn là 
\begin{center}
$P = \dfrac{\mathrm{C}_{10}^3+ \mathrm{C}_{10}^2\mathrm{C}_{10}^1}{\mathrm{C}_{20}^3} = \dfrac{1}{2}$.
\end{center}
}
\end{bt}

\begin{bt}%[HK1 Toán 11, 2017-2018, Marie Curie, HCM]%[1D2B2]
Giải phương trình $\mathrm{A}_n^2\cdot \mathrm{C}_n^{n-1} =48$.
\loigiai{
Điều kiện $n \in \mathbb{N}$  và $ n \ge 2$.

$\begin{aligned}
\text{PT} \Leftrightarrow & \dfrac{n!}{(n-2)!}\cdot \dfrac{n!}{(n-1)!} = 48\\ 
\Leftrightarrow  &  (n-1)\cdot n \cdot n = 48 \\
\Leftrightarrow  & (n-4)(n^2+3n+12) = 0\\
\Leftrightarrow  & \hoac{& n -4=0 \\ & n^2+3n+12 = 0 \text{ (vô nghiệm vì } n \ge 2)}  \\
\Leftrightarrow & n = 4.
\end{aligned}$

}
\end{bt}

\begin{bt}%[HK1 Toán 11, 2017-2018, Marie Curie, HCM]%[1D2K3]
Tìm số hạng không chứa $x$ trong khai triển $\left(x^2y+\dfrac{y^5}{x^3}\right)^{10},$ với $x,y \in \mathbb{R}$ và $x \ne 0$.
\loigiai{
Số hạng thứ $k+1$ là $T_{k+1} = \mathrm{C}_{10}^k (x^2y)^{10-k}\left(\dfrac{y^5}{x^3}\right)^k =  \mathrm{C}_{10}^k x^{20-5k}y^{10+4k}.$

 \noindent Số hạng không chứa $x$ tương ứng với $20-5k=0 \Leftrightarrow k=4$.

Vậy số hạng không chứa $x$ là $\mathrm{C}^4_{10}y^{26} = 201 y^{26}.$
}
\end{bt}
