\begin{ex}%[8D3B3][8D4B5][8D3K3]
	Giải các phương trình sau:
	\begin{enumerate}
		\item $x+2(x-3)=-9$.
		\item $|2x-3|=3x-1$.
		\item $\dfrac{2}{x^2-2x}+\dfrac{1}{x}=\dfrac{x+3}{x-2}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $x+2(x-3)=-9 \Leftrightarrow 3x=-3 \Leftrightarrow x=-1$. Vậy $x=-1$.
			\item $|2x-3|=3x-1 \Leftrightarrow \heva{&x \geq \dfrac{1}{3}\\&\hoac{&2x-3=3x-1\\&2x-3=1-3x}} \Leftrightarrow \heva{&x \geq \dfrac{1}{3}\\&\hoac{&x=-2 \, \text{(loại)}\\&x=\dfrac{4}{5}.}}$\\ Vậy $x=\dfrac{4}{5}$.
			\item $\dfrac{2}{x^2-2x}+\dfrac{1}{x}=\dfrac{x+3}{x-2}$.\\ Điều kiện xác định: $x \neq 0$ và $x \neq 2$.\\ Phương trình tương đương\\
			$\dfrac{2}{x^2-2x}+\dfrac{1}{x}-\dfrac{x+3}{x-2}=0 \Leftrightarrow \dfrac{2+x-2-(x+3)\cdot x}{x(x-2)}=0 \Leftrightarrow \dfrac{-x^2-2x}{x(x-2)}=0 \\ \Leftrightarrow \dfrac{-x(x+2)}{x(x-2)}=0 \Leftrightarrow \dfrac{x+2}{x-2}=0 \Leftrightarrow x+2=0 \Leftrightarrow x=-2$.\\ Vậy $x=-2$.
			
		\end{enumerate}
		
	}
\end{ex}

\begin{ex}%[8D3B7]
	Một miếng đất hình chữ nhật có chu vi là $120$m. Biết chiều dài gấp hai lần chiều rộng. Tính diện tích miếng đất hình chữ nhật.
	\loigiai{
		Gọi chiều dài của miếng đất là $x$(m) và chiều rộng của miếng đất	là $y$(m).\\ Theo đề bài chiều dài gấp hai lần chiều rộng nên ta có \[x=2y \tag{1}\].\\ Theo công thức tính chu vi hình chữ nhật, ta có \[2(x+y)=120 \tag{2}\]
		Thay $(1)$ vào $(2)$ ta được $y=20 \Rightarrow x=40$.\\ Vậy chiều dài miếng đất hình chữ nhật là $40$(m) và chiều rộng miếng đất đó là $20$(m).\\ Từ đây suy ra diện tích miếng đất hình chữ nhật là: $S=20 \times 40 =800$(m$^2$).
	}
\end{ex}

\begin{ex}%[8H4B3]
	Một bể chứa nước hình hộp chữ nhật có chiều dài $2{,}8$m, chiều rộng $1{,}4$m, chiều cao $1{,}5$m. Nước trong bể hiện chiếm $45\% $ thể tích của bể.
	\begin{enumerate}
		\item Hãy tính thể tích của bể chứa nước.
		\item Hỏi hiện trong bể chứa bao nhiêu lít nước ?
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Thể tích của bể chứa nước là: $V=2{,}8\times 1{,}4\times 1{,}5=5{,}88$ \, (m$^3$).
			\item  Số nước hiện có trong bể là: $5{,}88 \times 45\%=2{,}646$ \, (m$^3$).
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[8H3G2][8H3K6][8H3K7]
	Cho tam giác $ABC$ có $3$ góc nhọn $(AB<AC)$. Vẽ ba đường cao $AD, BE, CF$ cắt nhau tại $H$. 
	\begin{enumerate}
		\item Chứng minh rằng tam giác $ADB$ đồng dạng với tam giác $CFB$ và $BF\cdot BA=BD\cdot BC$.
		\item Chứng minh rằng tam giác $BFD$ đồng dạng với tam giác $BCA$.
		\item Qua $A$ vẽ đường thẳng $xy$ song song với $BC$. Tia $DF$ cắt đường thẳng $xy$ tại $M$. Gọi $I$ là giao điểm của $MC$ và $AD$. Chứng minh rằng $EI \parallel BC$.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[line cap=round,line join=round, >=stealth,scale=1]
			\tkzInit[xmin=-5,xmax=8.5,ymin=-1,ymax=7] 
			\tkzClip
			%	\tkzGrid 
			%	\tkzAxeXY
			\tkzDefPoints{0/0/B, 8/0/C, 2/6/A}
			\tkzDrawPolygon(C,B,A)
			\tkzDrawAltitude[draw =blue](B,C)(A) \tkzGetPoint{D} \tkzDrawAltitude[draw =blue](A,C)(B) \tkzGetPoint{E} \tkzDrawAltitude[draw =blue](B,A)(C) \tkzGetPoint{F} 
			\tkzInterLL(A,D)(B,E) \tkzGetPoint{H}
			\tkzLabelPoints[right](C,E) \tkzLabelPoints[above](A) \tkzLabelPoints[below](D) \tkzLabelPoints[left](F,B) \tkzLabelPoints[below right](H) 
			\tkzMarkRightAngle[size=0.3](A,D,C) \tkzMarkRightAngle[size=0.3](A,E,B) \tkzMarkRightAngle[size=0.3](A,F,C)  
			\tkzDefLine[parallel = through A](C,B) \tkzGetPoint{X}
			\tkzDrawLines[add=0.1 and 0.1](A,X D,F) \tkzInterLL(A,X)(D,F) \tkzGetPoint{M}  \tkzLabelPoints[above](M)
			\tkzDrawSegment(D,M)
			\tkzDrawSegments(M,C D,A) \tkzInterLL(M,C)(D,A) \tkzGetPoint{I} 
			\tkzDrawSegment(E,I)
			\tkzLabelPoint[above right](I) {$ I $}
			
			
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Xét $\triangle ADB$ và $\triangle CFB$ có\\
			$\heva{&\widehat{ADB}=\widehat{CFB}=90^{\circ}\\&\widehat{ABC} \, \text{là góc chung} } \Rightarrow \triangle ADB \backsim \triangle CFB \text{\,(G-G)} \Rightarrow \dfrac{DB}{BF}=\dfrac{BA}{BC} \Leftrightarrow BF\cdot BA=BD\cdot BC$.
			\item  Xét $\triangle BFD$ và $\triangle BCA$ có\\
			$\heva{&\dfrac{BF}{BD}=\dfrac{BC}{BA}\text{\, (do câu a)}\\&\widehat{ABC}\, \text{là góc chung}} \Rightarrow \triangle BFD \backsim \triangle BCA \text{\, (C-G-C)} $.
			\item Trước tiên dễ dàng có $\triangle ADC \backsim \triangle BEC\text{\, (G-G)} \Rightarrow\dfrac{AD}{EB}=\dfrac{DC}{EC}. \hfill{(1)} $\\ Lại có $MA \parallel BC \Rightarrow \widehat{AMD}=\widehat{MDB}\text{\,(so le trong)} \Rightarrow \widehat{AMD}=\widehat{BAE}\text{\,(do câu b)}$.\\ Do đó $\triangle MAD \backsim \triangle AEB\,\text(G-G) \Rightarrow \dfrac{MA}{AE}=\dfrac{AD}{EB}. \hfill{(2)}$ \\
			
			Từ $(1)$ và $(2)$ ta có $\dfrac{DC}{EC}=\dfrac{MA}{AE} \Leftrightarrow \dfrac{MA}{DC}=\dfrac{AE}{EC}$, lại có $MA \parallel DC \Rightarrow \dfrac{MA}{DC}=\dfrac{AI}{ID}$, \\do đó $\dfrac{AI}{ID}=\dfrac{AE}{EC} \Rightarrow EI \parallel DC \Rightarrow EI \parallel BC$.
		\end{enumerate}
	}
\end{ex}
