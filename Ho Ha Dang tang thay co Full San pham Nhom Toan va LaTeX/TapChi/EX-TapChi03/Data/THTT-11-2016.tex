	Tìm hai chữ số tận cùng của hiệu $ 2^{{9}^{2016}}-{2}^{{9}^{1945}}$.
	\loigiai{Ta thấy 
		$ \left(20k+1\right)\left(20h+1\right)=20\left(20kh+k+h\right)+1=20m+1$
		nên 
		$$ {\left(20k+1\right)}^{n}=20r+1, \,\, \text{với}\,\,  k, h, m, n, r\in \mathbb{N}^{*}.$$
		Do $ {9}^{2}=81=20k+1$
		nên $ {9}^{2016}={81}^{1008}={\left(20k+1\right)}^{1008}=20m+1$
		và 
		$$ {9}^{1945}={9.81}^{972}=9.{\left(20k+1\right)}^{972}=9\left(20n+1\right)=20r+9, \,\, \text{với}\,\,  m, n, r\in \mathbb{N}^{*}.$$
		Ta có
		{\allowdisplaybreaks
			\begin{align*}
			{2}^{20}&={2}^{10.2}={1024}^{2}=(1000+24)(1000+24)\\
			&=1000(1000+24+24)+576=100k+76.
			\end{align*}}Mà 
		$$ (100k+76)(100h+76)=100(100kh+76k+76h)+5776=100m+76$$
		nên $ {2}^{20n}={(100k+76)}^{n}=100r+76,$ với $k, h, m, n, r$ là các số nguyên dương.
		Từ đó:
		{\allowdisplaybreaks
			\begin{align*} {2}^{{9}^{2016}}&={2}^{20m+1}={2.2}^{20m}=2.\left(100t+76\right)=100.2t+152=100s+52\\
			{2}^{{9}^{1945}}&={2}^{20r+9}={2}^{9}{.2}^{20r}={2}^{9}\left(100x+76\right)=512\left(100x+76\right)\\
			& =\left(100.5+12\right)\left(100x+76\right)=100y+12.76=100z+12
			\end{align*}}với $m, t, s, r, x, y, z$ là các số nguyên dương.
		Từ kết quả trên suy ra
		$$ {2}^{{9}^{2016}}-{2}^{{9}^{1945}}=100s+52-(100z+12)=100(s-z)+40.$$
		Vậy hai chữ số tận cùng của hiệu $ {2}^{{9}^{2016}}-{2}^{{9}^{1945}}$ là $40$.
	}

	$ {n}^{4}+8n^3+19n^2-33n-90=0$.
	\loigiai{\noindent{\bf Cách 1.}  Viết lại đẳng thức đã cho thành
		{\allowdisplaybreaks
			\begin{align*}
			&{n}^{4}-2n^3+10n^3-20n^2+39n^2-78n+45n-90=0\\
			\Leftrightarrow& n^3\left(n-2\right)+10n^2\left(n-2\right)+39n\left(n-2\right)+45\left(n-2\right)=0\\
			\Leftrightarrow& \left(n-2\right)\left(n^3+10n^2+39n+45\right)=0.\tag{1}
			\end{align*}}Vì $n$ là số tự nhiên nên: $ n^3+10n^2+39n+45>0$. Do đó, từ 
		$(1)$ ta có $n-2=0$ hay $n=2$. Vậy giá trị $n$ cần tìm là $n = 2$.\\
		\noindent{\bf Cách 2.} Ta có
		$ {n}^{4}+8n^3+19n^2-33n-90=0$.

		$$ {n}^{4}+8n^3+19n^2-33n=90.$$
		Hay $ n\left(n^3+8n^2+19n-33\right)=90$.\hfill$(2)$\\
		Nếu $n$ là số lẻ thì $ n^3+8n^2+19n-33$ là số lẻ và vế trái của $(2)$ là số chẵn, không thỏa mãn.

		Với $ n>2$ ta có 
		$$ n^3+8n^2+19n-33>{2}^{3}+{8.2}^{2}+19.2-33=45.$$
		Suy ra $ n\left(n^3+8n^2+19n-33\right)>90$, không thỏa mãn $(2)$.
		Vậy $n = 2$ là giá trị duy nhất cần tìm.
	}

\begin{bt}[T3/473-Toán học \& Tuổi trẻ, tháng 11 năm 2016]%[Nguyễn Tài Chung, Dự án EX_TapChi03]%[7D0G5]
	Tìm tất cả các số nguyên dương $x, y$  sao cho 
	$ A=x^{2}+y^{2}+\dfrac{x^2y^2}{(x+y)^2}$ là số chính phương.
	\loigiai{Ta có 
		{\allowdisplaybreaks
			\begin{align*} A&=x^{2}+y^{2}+\dfrac{x^{2}y^{2}}{\left(x+y\right)^{2}}\\
			&={\left(x+y\right)}^{2}-2\left(x+y\right).\dfrac{xy}{x+y}+{\left(\dfrac{xy}{x+y}\right)}^{2}
			={\left(x+y-\dfrac{xy}{x+y}\right)}^{2}.
			\end{align*}}Để $A$ là số chính phương, ta cần tìm các số nguyên dương $x, y$ sao cho
		$ \dfrac{xy}{x+y}$ cũng là số nguyên dương. 
		Đặt $ \dfrac{xy}{x+y}=k$, $k\in \mathbb{N}^{*}$, hay
		{\allowdisplaybreaks
			\begin{align*}
			&\dfrac{xy}{x+y}=k\Leftrightarrow xy=k\left(x+y\right)\tag{1}\\
			\Leftrightarrow& \left(x-k\right)\left(y-k\right)=k^{2}.\tag{2}
			\end{align*}}Vì $x, y, k$ là các số nguyên dương nên từ (1) suy ra
		$x > k$, $y > k$, do đó $x - k$ và $y - k$ là các số nguyên dương.
		Gọi ƯCLN$ \left(x-k\right)\left(y-k\right)=d$ $ \left(d\in \mathbb{N}^{*}\right)$. Khi đó  
		$$\heva{
			&x-k\,\,\vdots\,\, d\\
			&y-k\,\,\vdots\,\, d\\
			&k^{2}\,\,\vdots\,\, {d}^{2}
		}
		\Rightarrow 
		\heva{
			&x\,\,\vdots\,\, d\\
			&y\,\,\vdots\,\, d\\
			&k\,\,\vdots\,\, d
		}$$
		Đặt $x = dx', \;y = dy', \;k = dk'$  $\left( x',\;y',\;k' \in \;\mathbb{N^*} \right)$.
		Thay vào (2) ta có: 
		\[\left(x'-k'\right)\left(y'-k'\right)={\left(k'\right)}^{2}.\tag{3}\]
		Ta có $ x-k=d\left(x'-k'\right),  y-k=d\left(y'-k'\right)$ và 
		$ \left(x-k, y-k\right)=d$ nên 
		$ \left(x'-k', y'-k'\right)=1$.
		Kết hợp với (3) suy ra $x' - k'$ và $y' - k'$  là các số chính phương.
		Do đó 
		$$\heva{
			&x'-k'=m^{2}\\
			&y'-k'=n^2\\
			&k'=mn.
		}
		\quad \left(m.n\in \mathbb{N}^*\right)\Leftrightarrow
		\heva{
			&x'=m^{2}+mn\\
			&y'=n^2+mn
		}
		$$
		Suy ra $x=dm^{2}+dmn$, $y=dn^2+dmn$
		$\left(m,n\in \mathbb{N}^{*}\right)$.\\
		Ngược lại với $x=dm^{2}+dmn$, $y=dn^2+dmn$
		$\left(m,n\in \mathbb{N}^*\right)$ thì
		$$ \dfrac{xy}{x+y}=\dfrac{\left(dm^2+dmn\right)\left(dn^2+dmn\right)}{d{\left(m+n\right)}^{2}}=dmn$$
		nên 
		$ A=\left(x+y-\dfrac{xy}{x+y}\right)^2
		=\left(d\left(m+n\right)^2-dmn\right)^2$ là số chính phương.\\
		Vậy với $x=dm^2+dmn$, $y=dn^2+dmn$
		$\left(m,n\in \mathbb{N}^*\right)$ thì $A$ là số chính phương.
	}
\end{bt}

\begin{bt}[T11/473-Toán học \& Tuổi trẻ, tháng 11 năm 2016]%[Nguyện Ngô, Dự án EX_TapChi03]%[1D0G6]
	Với hai số nguyên dương bất kì $n,p$, tìm số các hàm số
	\[f\colon \{1,2,3,\ldots,n\}\rightarrow\{-p,-p+1,-p+2,\ldots,p\}\]
	thỏa mãn tính chất $|f(i)-f(j)|\leq p$ với $i,j\in\{1,2,3,\ldots,n\}$.
	\loigiai
	{
		\textbf{Nhận xét:} Nếu tập hợp $A$ có $k$ phần tử, tập hợp $B$ có $l$ phần tử thì số cặp $(x; y)$ trong đó $x\in A, y\in B$ bằng $k\cdot l$.\\
		Với mỗi số  $m\in\{-p,-p+1,\ldots,p\}$ ta kí hiệu $t_m$ là số hàm số $f$ thỏa mãn bài toán và\\
		$\displaystyle\min_{1\leq i\leq n} f(i)=m$.\\
		Ta có
		{\allowdisplaybreaks
			\begin{align*}t_m=&\, \text{số hàm số}\, f\colon\{1,2,3,\ldots,n\}\rightarrow\{m,m+1,\ldots,a_m\}\\
			-&\text{số hàm số}\, f\colon\{1,2,3,\ldots,n\}\rightarrow\{m+1,\ldots,a_m\}\\
			=& (a_m-m+1)^n-(a_m-m)^n,
			\end{align*}
		}
		trong đó $a_m=\min\{m+p,p\}$.\\
		Chú ý $a_m=\heva{&m+p\quad&\text{nếu}&\quad m\leq 0\\&p\quad&\text{nếu}&\quad m>0.}$\\
		Do đó $t_m=\heva{&(p+1)^n-p^n \quad&\text{nếu}&\quad m\leq 0\\&(p-m+1)^n-(p-m)^n \quad&\text{nếu}&\quad m>0.}$\\
		Bởi vậy số các hàm số thỏa mãn bài toán là
		{\allowdisplaybreaks
			\begin{align*}
			t_{-p}+t_{-p+1}+\cdots+t_p&=(p+1)\left( (p+1)^n-p^n\right)+\displaystyle\sum_{m=1}^{p}\left((p-m+1)^n-(p-m)^n\right)\\
			&=(p+1)^{n+1}-(p+1)p^n+p^n=(p+1)^{n+1}-p^{n+1}.
			\end{align*}
		}
		\begin{nx}
			Đây là bài toán tính số phần tử của tập hợp dạng cơ bản, hay, nhưng không quá phức tạp.
		\end{nx}
