\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2012-2013, Tiền Giang]%[9D5G5]%[9D4G7]
\begin{enumerate}[\bf 1)]
	\item Giải hệ phương trình $\begin{cases} x^3 + 1 = 2\left(x^2 - x + y\right) \\ y^3 + 1 = 2\left(y^2 - y + x\right). \end{cases}$
	\item Cho phương trình $x^4 - 2mx^2 + 2m - 1 = 0$ $(1)$.
			\begin{enumerate}[a)]
				\item Tìm $m$ để $(1)$ có $4$ nghiệm $x_1$, $x_2$, $x_3$, $x_4$ thỏa $\begin{cases} x_1 < x_2 < x_3 < x_4 \\ x_4 - x_3 = x_3 - x_2 = x_2 - x_1. \end{cases}$
				\item Giải phương trình $(1)$ với $m$ vừa tìm được ở a).
			\end{enumerate}
\end{enumerate}
\loigiai{
\begin{enumerate}[\bf 1)]
	\item Xét hệ phương trình $\begin{cases} x^3 + 1 = 2\left(x^2 - x + y\right) &(1)\\ y^3 + 1 = 2\left(y^2 - y + x\right) &(2). \end{cases}$\\
			Trừ vế theo vế hai phương trình $(1)$ và $(2)$, ta được: $x^3 - y^3 = 2(x^2 - y^2) - 4(x - y)$\\
			\phantom{Trừ vế theo vế hai phương trình $(1)$ và $(2)$, ta được: }$\Leftrightarrow  \left[\begin{aligned} &x - y = 0 \\ &x^2 + xy + y^2 = 2(x + y) - 4\end{aligned}\right.$\\
			\begin{itemize}
				\item $x - y = 0 \Leftrightarrow x = y$, thay vào $(1)$, ta được phương trình:
						$$x^3 - 2x^2 + 1 = 0 \Leftrightarrow \left[\begin{aligned} &x = 1 \Rightarrow y = 1 \\ &x = \dfrac{1 + \sqrt{5}}{2} \Rightarrow y = \dfrac{1 + \sqrt{5}}{2} \\ &x = \dfrac{1 - \sqrt{5}}{2} \Rightarrow y = \dfrac{1 - \sqrt{5}}{2}\end{aligned}\right.$$
				\item $x^2 + xy + y^2 = 2(x + y) - 4 \Leftrightarrow (x + y)^2 - 2(x + y) - xy + 4 = 0$.\\
						Đặt $\begin{cases} S = x + y \\ P = xy \end{cases}$, ta được $\begin{cases} S^2 \geq 4P \\ S^2 - 2S - P + 4 = 0 \end{cases}$\\
						$\Rightarrow S^2 \geq 4(S^2 - 2S + 4) \Rightarrow 3\left(S - \dfrac{4}{3}\right)^2 + \dfrac{32}{3} \leq 0$ (không xảy ra). 
			\end{itemize}
			Vậy hệ phương trình đã cho có tập nghiệm $S = \left\lbrace (1; 1), \left(\dfrac{1 + \sqrt{5}}{2}; \dfrac{1 + \sqrt{5}}{2}\right), \left(\dfrac{1 - \sqrt{5}}{2}; \dfrac{1 - \sqrt{5}}{2}\right) \right\rbrace $.
	\item \begin{enumerate}[a)]
				\item Đặt $t = x^2$ $(t \geq 0)$, phương trình $(1)$ trở thành $t^2 - 2mt + 2m - 1 = 0 \Leftrightarrow \left[\begin{aligned} &t_1 = 1 \\ &t_2 = 2m - 1\end{aligned}\right.$\\
						Yêu cầu bài toán $\Leftrightarrow \begin{cases} t_2 > 0 \\ \left[\begin{aligned} &t_1 = 9t_2 \left(\mbox{do}\, \sqrt{t_1} - \sqrt{t_2} = \sqrt{t_2} - \left(-\sqrt{t_2}\right)\right) \\ &t_2 = 9t_1 \left(\mbox{do}\, \sqrt{t_2} - \sqrt{t_1} = \sqrt{t_1} - \left(-\sqrt{t_1}\right)\right)\end{aligned}\right. \end{cases}$\\
						\phantom{Yêu cầu bài toán }$\Leftrightarrow \left[\begin{aligned} &m = \dfrac{5}{9} \\ &m = 5\end{aligned}\right.$\\
						Vậy $m = 5$ hoặc $m = \dfrac{5}{9}$ là các giá trị cần tìm.
				\item Với $m = \dfrac{5}{9}$, phương trình $(1)$ trở thành: $$x^4 - \dfrac{10}{9}x^2 + \dfrac{1}{9} = 0 \Leftrightarrow \left[\begin{aligned} &x^2 = 1 \\ &x^2 = \dfrac{1}{9}\end{aligned}\right. \Leftrightarrow \left[\begin{aligned} &x_1 = -1 \\ &x_2 = -\dfrac{1}{3} \\ &x_3 = \dfrac{1}{3} \\ &x_4 = 1\end{aligned}\right.$$
						Vậy với $m = \dfrac{5}{9}$, phương trình đã cho có tập nghiệm $S = \left\lbrace -1; -\dfrac{1}{3}; \dfrac{1}{3}; 1\right\rbrace$.\\
						Với $m = 5$, phương trình $(1)$ trở thành: $$x^4 - 10x^2 + 9 = 0 \Leftrightarrow \left[\begin{aligned} &x^2 = 1 \\ &x^2 = 9\end{aligned}\right. \Leftrightarrow \left[\begin{aligned} &x_1 = -3 \\ &x_2 = -1 \\ &x_3 = 1 \\ &x_4 = 3\end{aligned}\right.$$
						Vậy với $m = 5$, phương trình đã cho có tập nghiệm $S = \left\lbrace -3; -1; 1; 3\right\rbrace$.
			\end{enumerate}
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2012-2013, Tiền Giang]%[9D4G6]
Cho parabol $(P): y = x^2$ và đường thẳng $(d): y = x + m$. Tìm $m$ để $(P)$ và $(d)$ cắt nhau tại hai điểm phân biệt $A$, $B$ sao cho tam giác $OAB$ là tam giác vuông.
\loigiai{
Phương trình hoành độ giao điểm của $(P)$ và $(d)$ là: $x^2 - x - m = 0$ $(1)$.\\
Phương trình $(1)$ có hai nghiệm phân biệt khác $0$ $\Leftrightarrow \begin{cases} m \neq 0 \\ \Delta = 1 + 4m > 0 \end{cases} \Leftrightarrow \begin{cases} m \neq 0 \\ m > -\dfrac{1}{4} \end{cases}$\\
Khi đó, ta có $(P)$ cắt $(d)$ tại hai điểm phân biệt $A\left(x_A; y_A\right)$ và $B\left(x_B; y_B\right)$ với $x_A$, $x_B$ là hai nghiệm của phương trình $(1)$.\\
Vì vai trò của $A$ và $B$ như nhau nên tam giác $OAB$ là tam giác vuông khi và chỉ khi tam giác $OAB$ là tam giác vuông tại $A$ hoặc vuông tại $O$.
\begin{enumerate}
	\item Tam giác $OAB$ vuông tại $A$ $\Leftrightarrow \begin{cases} OA \perp (d) \\ y_A = x_A^2 \\ y_A = x_A + m \end{cases} \Leftrightarrow \begin{cases} x_A = -1 \\ y_A = x_A^2 \\ y_A = x_A + m \end{cases} \Leftrightarrow \begin{cases} A(-1; 1) \\ B(2; 4) \\ m = 2. \end{cases}$
	\item Tam giác $OAB$ vuông tại $O$ $\Leftrightarrow \begin{cases} OA \perp OB \\ y_A = x_A^2; y_B = x_B^2 \\ x_A.x_B = - m \end{cases} \Leftrightarrow \begin{cases} x_A.x_B = -1 \\ y_A = x_A^2; y_B = x_B^2 \\ x_A.x_B = - m \end{cases}$\\
			\phantom{Tam giác $OAB$ vuông tại $O$ }$\Leftrightarrow \begin{cases} A\left(\dfrac{1 - \sqrt{5}}{2}; \dfrac{3 - \sqrt{5}}{2}\right)  \\ B\left(\dfrac{1 + \sqrt{5}}{2}; \dfrac{3 + \sqrt{5}}{2}\right) \\ m = 1. \end{cases}$
\end{enumerate}
Vậy $m = 2$ hoặc $m = 1$ là các giá trị cần tìm.
}
\end{ex}

\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2012-2013, Tiền Giang]%[9D5G1]%[9D5G5]
\begin{enumerate}
	\item Cho $4$ số $a$, $b$, $c$, $d$ thỏa mãn điều kiện $a + b + c + d = 2$. Chứng minh rằng $a^2 + b^2 + c^2 + d^2 \geq 1$.
	\item Cho $m \neq -1$ và $a^3 - 3a^2 + 3a(m + 1) - (m + 1)^2 = 0$. Hãy tìm giá trị nhỏ nhất của $a$.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có: $\left(a - \dfrac{1}{2}\right)^2 \geq 0$, $\left(b - \dfrac{1}{2}\right)^2 \geq 0$, $\left(c - \dfrac{1}{2}\right)^2 \geq 0$, $\left(d - \dfrac{1}{2}\right)^2 \geq 0$\\
			$\Rightarrow a^2 \geq a - \dfrac{1}{4}$, $b^2 \geq b - \dfrac{1}{4}$, $c^2 \geq c - \dfrac{1}{4}$, $d^2 \geq d - \dfrac{1}{4}$.\\
			$\Rightarrow a^2 + b^2 + c^2 + d^2 \geq (a + b + c + d) - 1$.\\
			$\Rightarrow a^2 + b^2 + c^2 + d^2 \geq 1$.
	\item Ta có: Phương trình $\Leftrightarrow (m + 1)a^3 - 3(m + 1)a^2 + 3(m + 1)^2a - (m + 1)^3 = 0$\\
			\phantom{Ta có: Phương trình }$\Leftrightarrow a^3 - 3(m + 1)a^2 + 3(m + 1)^2a - (m + 1)^3 = -ma^3$\\
			\phantom{Ta có: Phương trình }$\Leftrightarrow \left[a - (m + 1)\right]^3 =  -ma^3$\\
			\phantom{Ta có: Phương trình }$\Leftrightarrow a - (m + 1) = -\sqrt[3]{m}.a$\\
			\phantom{Ta có: Phương trình }$\Leftrightarrow a = \dfrac{m + 1}{\sqrt[3]{m} + 1}$\\
			\phantom{Ta có: Phương trình }$\Leftrightarrow a = \sqrt[3]{m^2} - \sqrt[3]{m} + 1$\\
			\phantom{Ta có: Phương trình }$\Leftrightarrow a = \left(\sqrt[3]{m} - \dfrac{1}{2}\right)^2 + \dfrac{3}{4} \geq \dfrac{3}{4}$.\\
			Vậy giá trị nhỏ nhất của $a$ là $\dfrac{3}{4}$ khi $m = \dfrac{1}{8}$. 
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2012-2013, Tiền Giang]%[9D5G5]
Chứng minh rằng $2^2 + 4^2 + 6^2 + \cdots + (2n)^2 = \dfrac{2n(n + 1)(2n + 1)}{3}$ với $n \in \mathbb{Z}$; $n \geq 1$.
\loigiai{
Đặt $g(n) = (2n)^2$; $f(n) = an^3 + bn^2 + cn + d$. Ta tìm các số $a$, $b$, $c$, $d$ sao cho $f(n) - f(n - 1) = g(n)$.\\
Ta có: $f(n) - f(n - 1) = g(n)$\\
\phantom{Ta có: }$\Leftrightarrow 3an^2 + (2b - 3a)n + a - b + c = 4n^2$\\
\phantom{Ta có: }$\Leftrightarrow \begin{cases} 3a = 4 \\ 2b - 3a = 0 \\ a - b + c = 0 \end{cases} \Leftrightarrow \begin{cases} a = \dfrac{4}{3} \\ b = 2 \\ c = \dfrac{2}{3} \\ d \in \mathbb{R} \end{cases}.$\\
Suy ra $f(n) = \dfrac{4}{3}n^3 + 2n^2 + \dfrac{2}{3}n + d$ ($d$ tùy ý).\\
Ta có: $\begin{cases} f(1) - f(0) = g(1) \\ f(2) - f(1) = g(2) \\ f(3) - f(2) = g(3) \\ \vdots \\ f(n) - f(n - 1) = g(n) \end{cases}$\\
Suy ra $f(n) - f(0) = 2^2 + 4^2 + 6^2 + \cdots + (2n)^2$\\
\phantom{Suy ra }$\Leftrightarrow \dfrac{4}{3}n^3 + 2n^2 + \dfrac{2}{3}n + d - d = 2^2 + 4^2 + 6^2 + \cdots + (2n)^2$\\
\phantom{Suy ra }$\Leftrightarrow 2^2 + 4^2 + 6^2 + \cdots + (2n)^2 = \dfrac{2n(n + 1)(2n + 1)}{3}$.
}
\end{ex}

\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2012-2013, Tiền Giang]%[9H1G2]%[9D5G5]
Cho tam giác $ABC$ có các phân giác trong của các góc nhọn $\widehat{BAC}$, $\widehat{ACB}$, $\widehat{CBA}$ theo thứ tự cắt các cạnh đối tại các điểm $M$, $N$, $P$. Đặt $a = BC$, $b = CA$, $c = AB$; $S_{\triangle MNP}$, $S_{\triangle ABC}$ theo thứ tự là diện tích của các tam giác $MNP$ và tam giác $ABC$.
\begin{enumerate}
	\item Chứng minh rằng $\dfrac{S_{\triangle MNP}}{S_{\triangle ABC}} = \dfrac{2abc}{(a + b)(b + c)(c + a)}$.
	\item Tìm giá trị lớn nhất của $\dfrac{S_{\triangle MNP}}{S_{\triangle ABC}}$.
\end{enumerate}
\loigiai{
\begin{center}
	 \begin{tikzpicture}   
	 	\tkzInit[xmin=-1,ymin=-1,xmax=9,ymax=7]
	 	\tkzClip
    	\tkzDefPoints{0/0/B, 8/0/C, 2/6/A}
    	\tkzDefCircle[in](A,B,C) \tkzGetPoint{I}
    	\tkzInterLL(A,I)(B,C) \tkzGetPoint{M}
    	\tkzInterLL(B,I)(A,C) \tkzGetPoint{N}
    	\tkzInterLL(C,I)(A,B) \tkzGetPoint{P}
    	\tkzDrawAltitude[draw =blue](A,C)(B) \tkzGetPoint{H}
    	\tkzDrawSegments(A,B B,C C,A  A,M B,N C,P M,N N,P P,M B,H)
    	\tkzDrawPoints(A,B,C,M,N,P,H,I)	
    	\tkzLabelPoints[above](A)
		\tkzLabelPoints[below](M)
		\tkzLabelPoints[left](P,B)
		\tkzLabelPoints[right](C,N,H)
		\tkzLabelPoints[above right](I)
		\tkzMarkRightAngles(B,H,A)
     \end{tikzpicture}
\end{center}
\begin{enumerate}
	\item Trong tam giác $ABC$, kẻ đường cao $AH$. Ta có: $S_{\triangle ABC} = \dfrac{1}{2}AC.BH = \dfrac{1}{2}AC.AB.\sin\widehat{BAC}$.\\
			Tương tự, ta có: $S_{\triangle ANP} = \dfrac{1}{2}AN.AP.\sin\widehat{BAC}$.\\
			Suy ra $\dfrac{S_{\triangle ANP}}{S_{\triangle ABC}} = \dfrac{AN.AP}{bc}$.\\
			Mà $\dfrac{AN}{b - AN} = \dfrac{c}{a} \Rightarrow AN = \dfrac{bc}{a + c}$.\\
			Tương tự, ta có: $AP = \dfrac{bc}{a + b}$.\\
			Vậy $\dfrac{S_{\triangle ANP}}{S_{\triangle ABC}} = \dfrac{bc}{(a + b)(a + c)}$.\\
			Tương tự, ta có: $\dfrac{S_{\triangle CMN}}{S_{\triangle ABC}} = \dfrac{ca}{(b + c)(b + a)}$, $\dfrac{S_{\triangle BMP}}{S_{\triangle ABC}} = \dfrac{ab}{(c + a)(c + b)}$.\\
			Suy ra $\dfrac{S_{\triangle MNP}}{S_{\triangle ABC}} = \dfrac{S_{\triangle ABC} - S_{\triangle ANP} - S_{\triangle CMN} - S_{\triangle BMP}}{\triangle ABC}$\\
			\phantom{Suy ra $\dfrac{\triangle MNP}{\triangle ABC}$ }$= 1 - \dfrac{S_{\triangle ANP}}{S_{\triangle ABC}} - \dfrac{S_{\triangle CMN}}{S_{\triangle ABC}} - \dfrac{S_{\triangle BMP}}{S_{\triangle ABC}}$\\
			\phantom{Suy ra $\dfrac{\triangle MNP}{\triangle ABC}$ }$= 1 - \dfrac{bc}{(a + b)(a + c)} - \dfrac{ca}{(b + c)(b + a)} - \dfrac{ab}{(c + a)(c + b)}$\\
			\phantom{Suy ra $\dfrac{\triangle MNP}{\triangle ABC}$ }$= \dfrac{2abc}{(a + b)(b + c)(c + a)}$.
	\item Ta có: $\left(\sqrt{a} - \sqrt{b}\right)^2 \geq 0 \Rightarrow a + b \geq 2\sqrt{ab}$.\\
			Tương tự, ta có: $b + c \geq 2\sqrt{bc}$; $c + a \geq 2\sqrt{ca}$.\\
			Suy ra $(a + b)(b + c)(c + a) \geq 8abc$.\\
			Vậy $\dfrac{S_{\triangle MNP}}{S_{\triangle ABC}} \leq \dfrac{1}{4}$.\\
			Suy ra giá trị lớn nhất của $\dfrac{S_{\triangle MNP}}{S_{\triangle ABC}}$ là $\dfrac{1}{4}$ khi tam giác $ABC$ là tam giác đều.
\end{enumerate}
}
\end{ex}
