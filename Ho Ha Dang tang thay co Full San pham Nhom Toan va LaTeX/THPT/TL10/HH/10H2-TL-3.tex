\section{HỆ THỨC LƯỢNG TRONG TAM GIÁC VÀ GIẢI TAM GIÁC}
\subsection{Tóm tắt lý thuyết}
Cho tam giác $ABC$, ta quy ước các kí hiệu sau.
\begin{itemize}
	\item $BC=a$, $CA=b$, $BC=a$.
	\item $p=\dfrac{a+b+c}{2}$ gọi là nửa chu vi của tam giác $ABC$.
	\item $m_a, m_b, m_c$ là độ dài đường trung tuyến tương ứng kẻ từ đỉnh $A, B, C$ của tam giác $ABC$.
	\item $h_a, b_b, h_c$ là độ dài đường cao tương ứng kẻ từ đỉnh $A, B, C$ của tam giác $ABC$.
	\item $l_a, l_b, l_c$ là độ dài đường phân giác trong tương ứng kẻ từ đỉnh $A, B, C$ của tam giác $ABC$.
	\item $R$ là bán kính đường tròn ngoại tiếp tam giác $ABC$.
	\item $r$ là bán kính đường tròn ngoại tiếp tam giác $ABC$.
\end{itemize}
\subsubsection{Hệ thức lượng trong tam giác vuông}
Cho tam giác \(ABC\) vuông tại \(A\), đường cao \(AH\).
\begin{center}
	\begin{tikzpicture}
	\coordinate (A) at (0,0);
	\coordinate (B) at (-1,-2);
	\coordinate (C) at (3,-2);
	\draw (A)node[above]{$A$}--(B)node[left]{$B$}--(C)node[right]{$C$}--cycle;
	\coordinate (H) at (0,-2);
	\draw [dashed] (A)--(H);
	\tkzMarkRightAngle(C,H,A)
	\tkzMarkRightAngle(B,A,C)
	\draw (H) node[below]{$H$};
	\end{tikzpicture}
\end{center}
\begin{itemize}
	\item Định lí Pitago:
	\begin{center}
		\fbox{\(BC^2=AB^2+AC^2\)}
	\end{center}
	\item Nếu biết 2 cạnh góc vuông thì có thể tính được đường cao \(AH\) bởi công thức:
	\begin{center}
		\fbox{\(\dfrac{1}{AH^2}=\dfrac{1}{AB^2}+\dfrac{1}{AC^2}\)}
	\end{center}
	\item Tích 2 cạnh góc vuông bằng tích cạnh huyền với đường cao tương ứng:
	\begin{center}
		\fbox{\(AB.AC=BC.AH\)}
	\end{center}
	\item Nếu biết 1 cạnh góc vuông và cạnh huyền thì có thể tính được hình chiếu của cạnh góc vuông đó lên cạnh huyền nhờ công thức:
	\begin{center}
		\fbox{\(AB^2=BH.BC; \quad AC^2=CH.BC\)}
	\end{center}
\end{itemize}
\subsubsection{Định lý hàm số cosin, công thức trung tuyến.}
Định lý hàm số cosin được phát minh bởi nhà toán học Al Kashi (1380 - 1429). Đây là một mở rộng của định lý Pythagore. Định lý hàm số cosin đưa ra một phương pháp giúp ta tìm được một cạnh của tam giác bất kì khi biết độ dài hai cạnh còn lại và số đo của góc xen giữa hai cạnh đó, từ đó cũng cho chúng ta tính được số đo của các góc còn lại của tam giác. Định lý được phát biểu như sau:
\begin{dl}
	Trong một tam giác bất kỳ, bình phương một cạnh bằng tổng bình phương của hai cạnh còn lại trừ đi hai lần tích của chúng với cosin của góc xen giữa hai cạnh đó.
\end{dl}
Nếu ký hiệu $a, b, c$ lần lượt là độ dài các cạnh $BC, CA, AB$ của tam giác $ABC$ thì ta có:
$$\heva{&a^2=b^2+c^2-2bc\cos A\\&b^2=c^2+a^2-2ca\cos B\\&c^2=a^2+b^2-2ab\cos C}$$
Từ định lý hàm số cosin ta cũng suy ra công thức tính cosin các góc của tam giác theo độ dài các cạnh của tam giác như sau:
$$\heva{&\cos A=\dfrac{b^2+c^2-a^2}{2bc}\\&\cos B=\dfrac{c^2+a^2-b^2}{2ca}\\&\cos C=\dfrac{a^2+b^2-c^2}{2ab}}$$
Mặt khác, sử dụng định lý hàm số cosin có thể giúp ta tìm được độ dài các đường trung tuyến theo ba cạnh của một tam giác. Cụ thể, nếu ký hiệu $m_a, m_b, m_c$ là độ dài các đường trung tuyến xuất phát từ các đỉnh $A, B, C$ thì:
$$\heva{&m_a^2=\dfrac{2\left(b^2+c^2\right)-a^2}{4}\\&m_b^2=\dfrac{2\left(c^2+a^2\right)-b^2}{4}\\&m_c^2=\dfrac{2\left(a^2+b^2\right)-c^2}{4}}$$
\subsubsection{Định lý sin}
\begin{dl}
Cho tam giác $ABC$, gọi $R$ là bán kính đường tròn ngoại tiếp tam giác $ABC$. Đặt $AB=c$, $BC=a$, $CA=b$. Ta có
$$\dfrac{a}{\sin A}=\dfrac{b}{\sin B}=\dfrac{c}{\sin C}=2R.$$
\end{dl}
\subsubsection{Các công thức diện tích tam giác}
Diện tích $S$ của tam giác $ABC$ được tính bởi một trong các công thức
\begin{align*}
S=&\dfrac{1}{2}a.h_a=\dfrac{1}{2}b.h_b=\dfrac{1}{2}c.h_c\\
=&\dfrac{1}{2}bc\sin A=\dfrac{1}{2}ca\sin B=\dfrac{1}{2}ab\sin C \\
=& \dfrac{abc}{4R} \\
=& pr \\
=& \sqrt{p(p-a)(p-b)(p-c)}
\end{align*}
\subsection{Các dạng toán}
\begin{dang}{Một số bài tập giúp nắm vững lý thuyết}
\textit{Mục này đưa ra một số bài tập mà việc giải quyết chỉ dùng đến các kiến thức về tích vô hướng của hai véc-tơ ở bài trước, chưa dùng đến các công thức về hệ thức lượng ở bài 3. Kết quả của các bài tập này sẽ dùng vào việc giới thiệu các công thức mới về hệ thức lượng trong tam giác.}
\end{dang}
\begin{vd}%[Phan Tấn Phú]%[0H2B3]
	Cho tam giác $ABC$
	\begin{enumerate}
		\item Tính $\vec{BC}$ theo $\vec{AB}$ và $\vec{AC}$.
		\item Tính $\vec{BC}.\vec{BC}$ từ đó tính tích vô hướng $\vec{AB}.\vec{AC}$ theo độ dài các cạnh của tam giác.
		\item Chứng minh rằng $\cos A=\dfrac{AB^2+AC^2-BC^2}{2.AB.AC}.$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $\vec{BC}=\vec{AC}-\vec{AB}$.
			\item $\vec{BC}.\vec{BC}=\left(\vec{AC}-\vec{AB}\right)^2=AB^2+AC^2-2\vec{AB}.\vec{AC}$. Từ đó suy ra $$\vec{AB}.\vec{AC}=\dfrac{AB^2+AC^2-BC^2}{2}.$$
			\item Theo câu trên ta có
			$$\begin{aligned}
			& AB.AC.\cos A=\dfrac{AB^2+AC^2-BC^2}{2}\\
			\Leftrightarrow & \cos A=\dfrac{AB^2+AC^2-BC^2}{2.AB.AC}
			\end{aligned}$$
		\end{enumerate}
	}
\end{vd}

\begin{vd}%[Phan Tấn Phú]%[0H2B3]
	Cho tam giác $ABC$ có $AM$ là trung tuyến.
	\begin{enumerate}
		\item Tính $\vec{BC}$ theo $\vec{AB}$ và $\vec{AC}$.
		\item Tính tích vô hướng $\vec{AB}.\vec{AC}$ theo độ dài các cạnh của tam giác.
		\item Tính $\vec{AM}$ theo $\vec{AB}$ và $\vec{AC}$.
		\item Chứng minh $AM^2=\dfrac{2AB^2+2AC^2-BC^2}{4}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Như bài trên.
			\item Như bài trên.
			\item $\vec{AM}=\dfrac{1}{2}\left(\vec{AB}+\vec{AC}\right)$
			\item Ta có
			$$\begin{aligned}
			AM^2&=\dfrac{AB^2+AC^2+2\vec{AB}.\vec{AC}}{4}\\
			&=\dfrac{AB^2+AC^2+AB^2+AC^2-BC^2}{4}\\
			&=\dfrac{2AB^2+2AC^2-BC^2}{4}
			\end{aligned}$$
		\end{enumerate}
	}
\end{vd}

\begin{vd}%[Phan Tấn Phú]%[0H2B3]
	Chứng minh rằng diện tích tam giác $ABC$ được tính bởi công thức $S=\dfrac{abc}{4R}$.
	\loigiai{
		Theo định lý sin, ta có $\dfrac{c}{\sin C}=2R$. Suy ra $\sin C=\dfrac{c}{2R}$. Từ đó
		$$S=\dfrac{1}{2}ab\sin C=\dfrac{1}{2}ab.\dfrac{c}{2R}=\dfrac{abc}{4R}.$$
	}
\end{vd}

\begin{vd}%[Phan Tấn Phú]%[0H2G3]
	Cho tam giác $ABC$ diện tích $S$. Chứng minh rằng 
	$$S=\dfrac{1}{2}\sqrt{AB^2.AC^2-\left(\vec{AB}.\vec{AC}\right)^2}$$
	\loigiai{
		$$\begin{aligned}
		S&=\frac{1}{2}AB.AC.\sin A \\
		&=\frac{1}{2}AB.AC.\sqrt{1-\cos^2A} \\
		&=\frac{1}{2}AB.AC.\sqrt{1-\left(\dfrac{\vec{AB}.\vec{AC}}{AB.AC}\right)^2} \\
		&=\dfrac{1}{2}\sqrt{AB^2.AC^2-\left(\vec{AB}.\vec{AC}\right)^2}
		\end{aligned}$$
	}
\end{vd}

\begin{vd}%[Phan Tấn Phú]%[0H2G3]
	Chứng minh công thức tính diện tích sau (công thức Hê-rông)
	$$S=\sqrt{p(p-a)(p-b)(p-c)}$$
	với $p$ là nửa chu vi tam giác, $a=BC, b=AC, c=AB$ là độ dài các cạnh.
	\loigiai{
		Ta có $\vec{AB}.\vec{AC}=\dfrac{AB^2+AC^2-BC^2}{2}=\dfrac{b^2+c^2-a^2}{2}$.\\
		Dùng kết quả bài trên, ta có
		$$\begin{aligned}
		S&=\dfrac{1}{2}\sqrt{AB^2.AC^2-\left(\vec{AB}.\vec{AC}\right)^2}\\
		&=\dfrac{1}{2}\sqrt{b^2c^2-\dfrac{1}{4}\left(b^2+c^2-a^2\right)^2}\\
		&=\dfrac{1}{2}\sqrt{\dfrac{1}{4}(2bc+b^2+c^2-a^2)(2bc-b^2-c^2+a^2)}\\
		&=\dfrac{1}{2}\sqrt{\dfrac{1}{4}[(b+c)^2-a^2][a^2-(b-c)^2]}\\
		&=\sqrt{\dfrac{b+c+a}{2}.\dfrac{b+c-a}{2}.\dfrac{a+b-c}{2}.\dfrac{a-b+c}{2}}\\
		&=\sqrt{p(p-a)(p-b)(p-c)}
		\end{aligned}$$
	}
\end{vd}

\begin{vd}%[Phan Tấn Phú]%[0H2K3]
	Trong mặt phẳng toạ độ $Oxy$, cho hai véc-tơ $a=(a_1;a_2)$, $b=(b_1;b_2)$. Chứng minh rằng
	$$Q=\sqrt{|\vec{a}|^2.|\vec{b}|^2-(\vec{a}.\vec{b})^2}=|a_1b_2-a_2b_1|$$
	\loigiai{
		$$\begin{aligned}
		Q^2&=\left(a_1^2+a_2^2\right)\left(b_1^2+b_2^2\right)-\left(a_1b_1+a_2b_2\right)^2\\
		&=a_1^2b_2^2+a_2^2b_1^2-2a_1b_2a_2b_1\\
		&=\left(a_1b_2-a_2b_1\right)^2
		\end{aligned}$$
		Suy ra $Q=\big|a_1b_2-a_2b_1\big|$.
	}
\end{vd}

\begin{vd}%[Phan Tấn Phú]%[0H2G3]
	Trong mặt phẳng toạ độ $Oxy$ cho tam giác $ABC$ có $A(x_A;y_A)$, $B(x_B;y_B)$, $C(x_C;y_C)$. Chứng minh rằng diện tích tam giác $ABC$ là
	$$S=\dfrac{1}{2}\Bigg|\left|
	\begin{array}{l l}
	(x_B-x_A) & (x_C-x_A) \\
	(y_B-y_A) & (y_C-y_A)
	\end{array}
	\right|\Bigg|=\dfrac{1}{2}
	\Big|(x_B-x_A)(y_C-y_A)-(x_C-x_A)(y_B-y_A)\Big|
	$$
	trong đó, người ta đặt $\left|\begin{array}{ll}a&b\\c&d\end{array}\right|=ad-bc$ (định thức cấp 2).
	\loigiai{
		Ta có
		$$\begin{aligned}
		\vec{AB}&=(x_B-x_A;y_B-y_A)\\
		\vec{AC}&=(x_C-x_A;y_C-y_A)
		\end{aligned}$$
		Ta đã biết $S=\dfrac{1}{2}\sqrt{AB^2.AC^2-\left(\vec{AB}.\vec{AC}\right)^2}$. Kết quả tính toán về toạ độ của bài trên ta được
		$$S=\dfrac{1}{2}
		\Big|(x_B-x_A)(y_C-y_A)-(x_C-x_A)(y_B-y_A)\Big|.$$
	}	
\end{vd}

\begin{vd}%[Phan Tấn Phú]%[0H2K3]
	Cho tam giác $ABC$, gọi $l_a$ là độ dài đường phân giác trong kẻ từ đỉnh $A$ của tam giác $ABC$. Chứng minh rằng $l_a=\dfrac{bc\sin A}{(b+c)\sin\frac{A}{2}}$.

\begin{vd}%[Phan Tấn Phú]%[0H2K3]
	Cho tam giác $ABC$ có $AB<AC$ (hay $c<b$), gọi $l'_a$ là độ dài đường phân \textbf{ngoài} kẻ từ đỉnh $A$ của tam giác $ABC$. Chứng minh rằng $l'_a=\dfrac{bc\sin A}{(b-c)\cos\frac{A}{2}}$.
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\clip (-8,-2.5) rectangle (6,1.5);

			\tkzInterLL(B,C)(A,I) \tkzGetPoint{D}

			\tkzInterLL(B,C)(A,x) \tkzGetPoint{E}
			\tkzDrawBisector(B,A,C)
			\tkzLabelPoints[below](E,B,D,C)
			\tkzMarkAngle[size=0.5](D,A,C)
			\tkzMarkAngle[size=0.6](B,A,D)
			\tkzMarkRightAngle[size=0.3](E,A,D)
			\draw (A)node[above]{$A$}--(E)--(C)--(A);
			\draw (A)--(B);
			\end{tikzpicture}
		\end{center}
		Gọi $E$ là chân đường phân giác ngoài kẻ từ đỉnh $A$ của tam giác $ABC$. Ta có $l'_a=AE$. Ta có
		$$\begin{aligned}
		&S_{ABC}=S_{ACE}-S_{ABE}\\
		\Leftrightarrow &\dfrac{1}{2}AB.AC.\sin A=\dfrac{1}{2}AC.AE.\sin\left(90^\circ+ \frac{A}{2}\right)-\dfrac{1}{2}AB.AE.\sin \left(90^\circ-\frac{A}{2}\right)\\
		\Leftrightarrow & cb\sin A=l'_a(b-c)\cos\frac{A}{2}\\
		\Leftrightarrow &l_a=\dfrac{bc\sin A}{(b-c)\cos\frac{A}{2}}

		Ta lại có: 
		\begin{align*}
		&\sin B\sin C=\dfrac{3}{4}\\
		\Leftrightarrow &\cos (B-C)-\cos (B+C)=\dfrac{3}{2}\\
		\Leftrightarrow &\cos (B-C)+\cos A=\dfrac{3}{2}\\
		\Leftrightarrow &\cos (B-C)=1 \\
		\Rightarrow     &B-C=0 \Leftrightarrow  B=C \Rightarrow \triangle ABC \text{  cân.} \quad (2) 	
		\end{align*}
		Từ $ (1) $ và $ (2) $ suy ra $ \triangle ABC $ đều.

	+\dfrac{b^2\cos \dfrac{C-A}{2}}{2\sin \dfrac{B}{2}}
	+\dfrac{c^2\cos \dfrac{A-B}{2}}{2\sin \dfrac{C}{2}}=a^2+b^2+c^2.
	$$
	Chứng minh rằng $ \triangle ABC $ đều.
	\loigiai{

		\begin{align*}
		\dfrac{a^2\cos \dfrac{B-C}{2}}{2\sin \dfrac{A}{2}} 
		&=\dfrac{a(2R\sin A)\cos \dfrac{B-C}{2}}{2\sin \dfrac{A}{2}} \\
		&=2aR\cos \dfrac{A}{2}\cos \dfrac{B-C}{2} \\
		&=2aR\sin \dfrac{B+C}{2} \cos \dfrac{B-C}{2}\\
		&=aR\left( \sin B+\sin C\right) = \dfrac{a(b+c)}{2}
		\end{align*}

		\dfrac{a^2\cos \dfrac{B-C}{2}}{2\sin \dfrac{A}{2}}

		&=\dfrac{a(b+c)}{2}+\dfrac{b(c+a)}{2}+\dfrac{c(a+b)}{2}=ab+bc+ca \\
		&\leq \dfrac{a^2+b^2}{2}+\dfrac{b^2+c^2}{2}+\dfrac{c^2+a^2}{2}=a^2+b^2+c^2
		\end{aligned}$$

	Một cuộc đua thuyền xuất phát từ điểm $A$ như hình vẽ bên và di chuyển theo hướng tây nam một góc $52^\circ$ tới điểm $B$, sau đó di chuyển theo hướng đông nam $40^\circ$ tới điểm $C$, cuối cùng quay trở lại điểm $A$. Điểm $C$ cách điểm $A$ một khoảng $8$km. Tính gần đúng tổng khoảng cách của đường đua.
	\begin{center}
		\begin{tikzpicture}[>=stealth,scale=1]
		\tkzDefPoints{0/0/B,0/3/o,3/2/A,3/-3/C}
		\tkzDefShiftPoint[B](-90:3){D}
		\tkzDefShiftPoint[o](0:.5){E}
		\tkzDefShiftPoint[o](90:.5){N}
		\tkzDefShiftPoint[o](180:.5){W}
		\tkzDefShiftPoint[o](-90:.5){S}
		\tkzDrawSegments[->](A,B B,C C,A o,N o,S o,E o,W)
		\tkzDrawSegments(B,D)
		\tkzDrawCircle[fill=black!50,R](o,.1cm)
		\tkzMarkAngles[mkpos=.2,size=.3](B,A,C D,B,C)
		\tkzLabelAngle[pos=.5](B,A,C){\footnotesize $52^\circ$}
		\tkzLabelAngle[pos=.6](D,B,C){\footnotesize $40^\circ$}
		\tkzDrawPoints(A,B,C,D)
		\tkzLabelPoints[left](B,D,W)
		\tkzLabelPoints[below](S)
		\tkzLabelPoints[above](N)
		\tkzLabelPoints[right](A,E,C)
		\end{tikzpicture}
	\end{center}
	\loigiai{
		Vì đường thẳng $BD$ và $AC$ song song, suy ra $\widehat{BCA}=\widehat{CBD}=40^\circ$. Do đó, $\widehat{ABC}=180^\circ-(52^\circ+40^\circ)=88^\circ$.\\
		Áp dụng định lý $\sin$ trong tam giác $ABC$, ta được: $\dfrac{a}{\sin 52^\circ}=\dfrac{b}{\sin88^\circ}=\dfrac{c}{\sin40^\circ}$.\\
		Vì $b=8$ theo đề, nên: $a=\dfrac{8\sin52^\circ}{\sin88^\circ}\approx 6,308$km và $c=\dfrac{8\sin40^\circ}{\sin88^\circ}\approx 5,145$km.\\
		Tổng khoảng cách của đươgf đua là: $8+6,308+5,145=19,453$km.
	}
\end{bt}

\begin{bt}%[Dũng Lê]%[0H3T3]
	Hai lực $\vec{f_1}$ và $\vec{f_2}$ cho trước cùng tác dụng lên một vật và tạo thành góc nhọn $\left(\vec{f_1},\vec{f_2}\right)=\alpha$. Hãy lập công thức tính cường độ của hợp lực $\vec{s}$.

			Vậy $\left|\vec{s}\right|=\left|\vec{AC}\right|=\left|\vec{f_1}+\vec{f_2}\right|$.\\

\begin{bt}%[Dũng Lê]%[0H3T3]
	\immini{
		Một hành khách ngồi trong một máy bay, bay ở độ cao $10$km nhìn xuống hai thị trấn dưới mặt đất. Góc hợp bởi phương ngang và hai thị trấn lần lượt là $28^\circ$ và $55^\circ$ (hình vẽ). Tính khoảng cách giữa hai thị trấn.

		\tkzDefPoints{0/0/O,0/3/A, 2/0/C,4/0/D,3/3/A'}
		\tkzDrawSegments[dashed](A,A' A,O A,C A,D O,C)
		\tkzDrawSegments(C,D)
		\tkzDrawPoints(A,O,C,D)

		\tkzLabelAngle[pos=.5](C,A,A'){\footnotesize $55^\circ$}
		\tkzLabelAngle[pos=1](D,A,A'){\footnotesize $28^\circ$}
		\tkzLabelLine[pos=.5,left](A,O){\footnotesize $10$km}
		\end{tikzpicture}
	}
	\loigiai{
		Ta có $\widehat{CAD}=55^\circ-28^\circ=27^\circ$.\\
		Xét tam giác $ACO$ có $\widehat{OAC}=90^\circ-55^\circ=35^\circ$ và $\cos\widehat{OAC}=\dfrac{AO}{AC}\Leftrightarrow AC=\dfrac{10}{\cos35^\circ}\approx 12,2$km.\\

		Áp dụng định lý sin trong tam giác $ACD$, ta có\\
		$\dfrac{CD}{\sin 27^\circ}=\dfrac{AC}{\sin 28^\circ}\Leftrightarrow CD=\dfrac{12,2\sin27^\circ}{\sin28^\circ}\approx 11,79$km.\\
		Vậy khoảng cách giữa hai thị trấn khoảng $11,9$km.
	}
