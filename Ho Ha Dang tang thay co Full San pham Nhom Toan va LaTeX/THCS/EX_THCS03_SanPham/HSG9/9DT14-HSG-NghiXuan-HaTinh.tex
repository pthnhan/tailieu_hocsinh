\begin{ex}%[HSG Toán 9 năm học 2013-2014, Nghi Xuân, Hà Tĩnh]%[9D1K3]
Tìm $x,y$ thỏa mãn $2x+y-2\sqrt{xy}-4\sqrt{x}+4=0$.
\loigiai{
Điều kiện: $\left[\begin{aligned}& x=0; y\in\mathbb{R}\\ & x>0; y\ge 0. \end{aligned}\right.$\\
\begin{itemize}
\item Với $x=0$, ta được $y=-4$ (thỏa mãn).
\item Với $x>0$, $y\geq 0$. Ta có
\begin{eqnarray*}
&& 2x+y-2\sqrt{xy}-4\sqrt{x}+4=0\\
& \Leftrightarrow & \left(x-2\sqrt{xy}+y\right)+\left(x-4\sqrt{x}+4\right)=0\\
& \Leftrightarrow & \left(\sqrt{x}-\sqrt{y}\right)^2+\left(\sqrt{x}-2\right)^2=0\\
& \Leftrightarrow & \left\{\begin{aligned}& \sqrt{x}-\sqrt{y}=0\\ &\sqrt{x}-2=0 \end{aligned}\right.\Leftrightarrow x=y=4.\quad\text{(thỏa mãn)}
\end{eqnarray*}
\end{itemize}
Vậy $(x;y)\in\{(0;-4);(4;4)\}$.
}
\end{ex}

\begin{ex}%[HSG Toán 9 năm học 2013-2014, Nghi Xuân, Hà Tĩnh]%[9D5G3]
Giải phương trình nghiệm nguyên $5x^4+y^2-4x^2y-85=0$.
\loigiai{
Phương trình đã cho tương đương với $x^4=85-\left(y-2x^2\right)^2$.\\
$\Rightarrow x^4\le 85<4^4$. Mà $x\in\mathbb{Z}$ nên $x^4\in \{0^4;1^4;2^4;3^4\}$.
\begin{itemize}
\item Với $x^4=0^4\Rightarrow y^2=85$ (loại).
\item Với $x^4=1^4\Rightarrow (y-2)^2=84$ (loại).
\item Với $x^4=2^4\Rightarrow (y-8)^2=69$ (loại).
\item Với $x^4=3^4\Rightarrow (y-18)^2=4\Leftrightarrow \left[\begin{aligned}& y-18=2\\ & y-18=-2 \end{aligned}\right.
\Leftrightarrow \left[\begin{aligned}& y=20\\ & y=16. \end{aligned}\right.$
\end{itemize}
Vậy phương trình có bốn nghiệm nguyên $(x;y)\in\{(3;20), (-3;20), (3;16), (-3;16)\}$.
}
\end{ex}
\begin{ex}%[HSG Toán 9 năm học 2013-2014, Nghi Xuân, Hà Tĩnh]%[9D5G2]
Cho $x,y,z$ là các số nguyên và 
$$\left\{\begin{aligned}& P=(x+2012)^5+(2y-2013)^5+(3z+2014)^5 \\ & S=x+2y+3z+2013. \end{aligned}\right.$$
Chứng minh rằng $P$ chia hết cho $30$ khi và chỉ khi $S$ chia hết cho $30$.
\loigiai{
Đặt $a=x+2012$, $b=2y-2013$, $c=3z+2014$. Ta được $\left\{\begin{aligned}& P=a^5+b^5+c^5\\ & S=a+b+c \end{aligned}\right.$ ($a,b,c\in\mathbb{Z}$).\\
Xét $P-S=(a^5-a)+(b^5-b)+(c^5-c)$.\\
Ta chứng minh được $m^5-m$ chia hết cho $30$. Thật vậy, ta có
$$m^5-m=m(m^4-1)=m(m-1)(m+1)(m^2+1)=5m(m-1)(m+1)+m(m-1)(m+1)(m-2)(m+2).$$
Mà $m(m-1)(m+1)$ là tích của ba số nguyên liên tiếp nên chia hết cho $2\cdot 3=6$, nghĩa là $5m(m-1)(m+1)$ chia hết cho $30$.\\
Lại có $m(m-1)(m+1)(m-2)(m+2)$ là tích của năm số nguyên liên tiếp nên chia hết cho $2\cdot 3\cdot 5=30$.\\
Do đó $P-S$ chia hết cho $30$. Hay $P$ chia hết cho $30$ khi vào chỉ khi $S$ chia hết cho $30$.
}
\end{ex}

\begin{ex}%[HSG Toán 9 năm học 2013-2014, Nghi Xuân, Hà Tĩnh]%[9D5G5]
Cho ba số $x,y,z$ khác $0$ thỏa mãn $\left\{\begin{aligned}& x+y+z=\dfrac{1}{2}\\ & \dfrac{1}{x^2}+\dfrac{1}{y^2}+\dfrac{1}{z^2}+\dfrac{1}{xyz}=4\\ & \dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}>0.\end{aligned}\right.$\\
Tính giá trị của biểu thức $P=\left(y^{2009}+z^{2009}\right)\left(z^{2011}+x^{2011}\right)\left(x^{2013}+y^{2013}\right)$.
\loigiai{
Từ  giả thiết ta có
\begin{eqnarray*}
4 & = & \dfrac{1}{x^2}+\dfrac{1}{y^2}+\dfrac{1}{z^2}+\dfrac{1}{xyz}=\dfrac{1}{x^2}+\dfrac{1}{y^2}+\dfrac{1}{z^2}+\dfrac{2(x+y+z)}{xyz}\\
& = & \dfrac{1}{x^2}+\dfrac{1}{y^2}+\dfrac{1}{z^2}+2\left( \dfrac{1}{xy}+\dfrac{1}{yz}+\dfrac{1}{zx} \right)={{\left( \dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z} \right)^2}}.
\end{eqnarray*}
Mà $\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}>0$, suy ra $\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=2$. Lại có $x+y+z=\dfrac{1}{2}$, suy ra $\dfrac{1}{x+y+z}=2$. \\
Do đó  $$\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=\dfrac{1}{x+y+z}\Leftrightarrow (x+y)(y+z)(z+x)=0\Leftrightarrow \left[\begin{aligned}& x=-y\\ & y=-z\\ & z=-x\end{aligned}\right.
\Leftrightarrow \left[\begin{aligned}& x^{2009}=-y^{2009}\\ & y^{2011}=-z^{2011}\\ & z^{2013}=-x^{2013}.\end{aligned}\right.$$
Vậy $P=0$.
}
\end{ex}

\begin{ex}%[HSG Toán 9 năm học 2013-2014, Nghi Xuân, Hà Tĩnh]%[8H3G7]
Cho $\triangle ABC$ nhọn có trực tâm $H$, trọng tâm $I$. Giao điểm của ba đường trung trực là $O$, trung điểm của $BC$ là $M$. Tính giá trị biểu thức $\sqrt{\dfrac{IO^2+OM^2}{IH^2+HA^2}}$.
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1.5]
\clip (-0.5,-1.9) rectangle (5.5,4);
\tkzDefPoints{1.5/3.5/A,0/0/B,5/0/C}
\tkzDefLine[perpendicular=through A](C,B)\tkzGetPoint{a}
\tkzInterLL(A,a)(B,C)\tkzGetPoint{h}
\tkzDefLine[perpendicular=through B](A,C)\tkzGetPoint{b}
\tkzInterLL(B,b)(A,C)\tkzGetPoint{k}
\tkzInterLL(A,a)(B,b)\tkzGetPoint{H}
\tkzDefMidPoint(C,B)\tkzGetPoint{M}
\tkzDefCircle[circum](A,B,C)\tkzGetPoint{O}
\tkzInterLL(A,M)(H,O)\tkzGetPoint{I}
\tkzMarkRightAngles(A,h,C A,k,B O,M,C)
\tkzDefPointBy[symmetry = center M](H)\tkzGetPoint{D}

\tkzDrawSegments(A,B B,C C,A A,h B,k A,M O,H O,M D,B D,C)
\tkzDrawSegments[dashed](H,D A,D)
\tkzDrawPoints[fill=white, size=8](A,B,C,M,O,H,I,D)
\tkzLabelPoints[below left](B,I,M)
\tkzLabelPoints[below](D)
\tkzLabelPoints[above](A)
\tkzLabelPoints[above right](O)
\tkzLabelPoints[above left](H)
\tkzLabelPoints[below right](C)
\end{tikzpicture}
\end{center}
Gọi $D$ là điểm đối xứng của $A$ qua $O$. Do $O$ là giao của ba đường trung trực của $\triangle ABC$ nên $OA=OC=OD$, suy ra $\triangle ACD$ vuông tại $C$ và $BH\parallel CD$. Tương tự cũng có $HC\parallel BD$, suy ra $BHCD$ là hình bình hành và $M$ là trung điểm $BC$.\\
Từ đó $MO$ là đường trung bình của $\triangle DAH$ hay $\dfrac{MO}{AH}=\dfrac{1}{2}$.\\
Lại có $I$ là trọng tâm $\triangle ABC$ nên $\dfrac{MI}{AI}=\dfrac{1}{2}$. Do đó $\dfrac{MO}{AH}=\dfrac{MI}{AI}$ và $\triangle MOI\backsim \triangle AHI$ (c.g.c).\\
Suy ra $\dfrac{IO}{IH}=\dfrac{1}{2}$. Khi đó
$$\dfrac{IO^2}{IH^2}=\dfrac{OM^2}{HA^2}=\dfrac{IO^2+OM^2}{IH^2+HA^2}=\dfrac{1}{4}\Rightarrow \sqrt{\dfrac{IO^2+OM^2}{IH^2+HA^2}}=\dfrac{1}{2}.$$
}
\end{ex}

\begin{ex}%[HSG Toán 9 năm học 2013-2014, Nghi Xuân, Hà Tĩnh]%[8H3G2]
Cho góc $\widehat{xOy}$ và một đường thẳng $d$ thay đổi luôn cắt các tia $Ox$, $Oy$ tại $M$ và $N$. Biết giá trị biểu thức $\dfrac{1}{OM}+\dfrac{1}{ON}$ không thay đổi khi đường thẳng $d$ thay đổi. Chứng minh rằng đường thẳng $d$ luôn đi qua một điểm cố định.
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1.5]
\clip (-0.5,-0.5) rectangle (7.5,3.3);
\tkzDefPoints{5/3/x,0/0/O,7/0/y}
\coordinate (M) at ($(O)!.8!(x)$);
\coordinate (N) at ($(O)!.7!(y)$);
\coordinate (I) at ($(M)!.47!(N)$);
\tkzDefLine[parallel=through I](O,y)\tkzGetPoint{e}
\tkzInterLL(I,e)(O,x)\tkzGetPoint{E}
\tkzDefLine[parallel=through I](O,x)\tkzGetPoint{d}
\tkzInterLL(I,d)(O,y)\tkzGetPoint{D}
\tkzDrawSegments(O,x O,y I,E I,D)
\tkzDrawLine(M,N)

\tkzDrawPoints[fill=white, size=8](O,M,N,I,D,E)
\tkzLabelPoints[below left](O,N)
\tkzLabelPoints[below](D,y)
\tkzLabelPoints[above right](I)
\tkzLabelPoints[above left](E,x,M)
\end{tikzpicture}
\end{center}
Giả sử $\dfrac{1}{OM}+\dfrac{1}{ON}=\dfrac{1}{a}$ ($a$ là một số dương không đổi). Lấy điểm $D$ trên $Oy$ sao cho $OD=a$. Khi đó $OD<ON$. Vẽ $DI\parallel Ox$ ($I\in MN$). Lấy $E$ trên $Ox$ sao cho $OE=ID$. Khi đó $OEID$ là hình bình hành.
Ta có $$\dfrac{OE}{OM}+\dfrac{OD}{ON}=\dfrac{NI}{NM}+\dfrac{EI}{ON}=\dfrac{NI}{NM}+\dfrac{MI}{MN}=1\Rightarrow \dfrac{1}{ON}+\dfrac{OE}{OD\cdot OM}=\dfrac{1}{OD}=\dfrac{1}{a}.$$
Khi đó
$$\dfrac{1}{OM}=\dfrac{OE}{OD\cdot OM}\Rightarrow \dfrac{OE}{OD}=1\Rightarrow OE=OD=a.$$
Mà $D\in Oy$, $E\in Ox$ nên $OD, OE$ cố định. Mặt khác $O$ cố định và $OEID$ là hình bình hành nên $I$ cố định. Vậy $d$ luôn đi qua điểm $I$ cố định.
}
\end{ex}

\begin{ex}%[HSG Toán 9 năm học 2013-2014, Nghi Xuân, Hà Tĩnh]%[9D5G1]
Cho các số $x,y,z$ không âm, không đồng thời bằng $0$ thỏa mãn
$$\dfrac{1}{x+1}+\dfrac{1}{y+2}+\dfrac{1}{z+3}\le 1.$$
Tìm giá trị nhỏ nhất của biểu thức $P=x+y+z+\dfrac{1}{x+y+z}$.
\loigiai{
Áp dụng BĐT Cauchy-Schwarz (Bunhiacopxki) ta có
$$(x+1+y+2+z+3)\left(\dfrac{1}{x+1}+\dfrac{1}{y+2}+\dfrac{1}{z+3}\right)\ge (1+1+1)^2\Leftrightarrow x+y+z+6\ge 9\Leftrightarrow x+y+z\ge 3.$$
Do đó $$P=\dfrac{8(x+y+z)}{9}+\dfrac{x+y+z}{9}+\dfrac{1}{x+y+z}\ge \dfrac{8}{3}+2\sqrt{\dfrac{x+y+z}{9}\cdot\dfrac{1}{x+y+z}}=\dfrac{10}{3}.$$
Đẳng thức xảy ra khi $(x;y;z)=(2;1;0)$. Vậy $\min P=\dfrac{10}{3}$ khi $(x;y;z)=(2;1;0)$.
}
\end{ex}

\begin{ex}%[HSG Toán 9 năm học 2013-2014, Nghi Xuân, Hà Tĩnh]%[9D5G1]
Cho các số dương $x,y,z$ thỏa mãn điều kiện $xy+yz+zx=671$. Chứng minh rằng
$$\dfrac{x}{x^2-yz+2013}+\dfrac{y}{y^2-zx+2013}+\dfrac{z}{z^2-xy+2013}\ge \dfrac{1}{x+y+z}.$$
\loigiai{
Ta có 
\begin{eqnarray*}
VT & = & \dfrac{x}{x^2-yz+2013}+\dfrac{y}{y^2-zx+2013}+\dfrac{z}{z^2-xy+2013}\\
& = & \dfrac{x^2}{x\left(x^2-yz+2013\right)}+\dfrac{y^2}{y\left(y^2-zx+2013\right)}+\dfrac{z^2}{z\left(z^2-xy+2013\right)}\\
& \ge & \dfrac{(x+y+z)^2}{x^3+y^3+z^3-3xyz+2013(x+y+z)}.
\end{eqnarray*}
Lại có $xy+yz+zx=671$ nên
\begin{eqnarray*}
&& x^3+y^3+z^3-3xyz+2013(x+y+z)\\
& = & (x+y+z)\left[(x+y+z)^2-3(xy+yz+zx)\right]+2013(x+y+z)\\
& = & (x+y+z)\left[(x+y+z)^2-3\cdot 671+2013\right]=(x+y+z)^3
\end{eqnarray*}
Từ đó
$$VT\ge\dfrac{(x+y+z)^2}{(x+y+z)^3}=\dfrac{1}{x+y+z}.$$
Đẳng thức xảy ra khi $x=x=z=\dfrac{\sqrt{2013}}{3}$.
}
\end{ex}
