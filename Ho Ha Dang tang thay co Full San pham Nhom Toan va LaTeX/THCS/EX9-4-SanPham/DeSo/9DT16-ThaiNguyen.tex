\begin{ex}%[Đề thi vào 10 Sở giáo dục Thái Nguyên 2016]%[9D2K2]
   Cho hàm số $y=\left( \sqrt{3}-\sqrt{2}\right)x +3 $ có đồ thị $(d)$. Hàm số đã cho là đồng biến hay nghịch biến trên 
   $\mathbb{R}$? Giải thích? Tìm tọa độ giao điểm của $(d)$ với trục tung.
   \loigiai{
   Hàm số $y=\left( \sqrt{3}-\sqrt{2}\right)x +3 $ đồng biến trên  $\mathbb{R}$ vì $a=\sqrt{3}-\sqrt{2} >0 $.\\
   Cho $x=0$ ta có $y=3$ suy ra tọa độ giao điểm của $(d)$ với trục tung là $(0;3).$
}
\end{ex}

	\end{align*} }

\begin{ex}%[Đề thi vào 10 Sở giáo dục Thái Nguyên 2016]%[9D1K8]
	Cho biếu thức $B= \left( \dfrac{\sqrt{x}-4}{x-2\sqrt{x}}+ \dfrac{3}{\sqrt{x}-2}\right) : \left( \dfrac{\sqrt{x}}{\sqrt{x}-2} - \dfrac{\sqrt{x}+2}{\sqrt{x}}\right)$ với $x>0, x \neq 4$. Hãy rút gọn $B$ và tính giá trị của $B$ khi $x=3+\sqrt{8}.$
	\loigiai{
	Ta có: \begin{align*}
	B&=\left(\dfrac{\sqrt{x}-4}{ \sqrt{x}(\sqrt{x}-2) } + \dfrac{3}{\sqrt{x}-2} \right) : \left( \dfrac{\sqrt{x}\sqrt{x}-\left(\sqrt{x}-2\right)\left(\sqrt{x}+2\right) }{ \sqrt{x}\left(\sqrt{x}-2\right) }\right) 
	\\ &=\dfrac{4\left(\sqrt{x}-1\right)}{ \sqrt{x} \left(\sqrt{x}-2\right) } . \dfrac{ \sqrt{x} \left( \sqrt{x}-2 \right) }{4}
	\\ &=\sqrt{x}-1
	\end{align*}}
Với $x=3+\sqrt{8}=\left(\sqrt{2}+1\right)^2$ ta có $\sqrt{x}=\sqrt{2}+1$. Do đó $B=\sqrt{2}+1-1=\sqrt{2}$.
\end{ex}

\begin{ex}%[Đề thi vào 10 Sở giáo dục Thái Nguyên 2016]%[9D3K3]
	Xác định các hệ số $a$ và $b$ biết hệ phương trình $\heva{ax+by&=4 \\bx-2y&=-2}$ có nghiệm $(x;y)=(2;-1).$
	\loigiai{
	Vì hệ có nghiệm $(2;-1)$ nên thế $x=2;y=-1$ vào hệ ta có:
	\begin{align*}
	\heva{2a+b(-1)=4 \\ 2b-2(-1)=-2} \Leftrightarrow \heva{2a-b=4 \\ b=-2} \Rightarrow \heva{a=1 \\ b=-2}
	\end{align*}}
	 Với $a=1;b=-2$ ta có hệ $\heva{x-2y=4 \\ -2x-2y=-2} \Leftrightarrow \heva{x=2\\ y=-1}$.
	 \\ Vậy hệ có nghiệm $(x;y)=(2;-1)$ thì $a=1$ và $b=-2$.
\end{ex}

\begin{ex}%[Đề thi vào 10 Sở giáo dục Thái Nguyên 2016]%[9D4K3]
	Không dùng máy tính cầm tay hãy giải phương trình:
	\[ x^2 + 6x -2016=0.\]
	\loigiai{
	Ta có $a=1$, $b=6$ ($b'=3$), $c=-2016$.
	\\ $\Delta'=3^2-1.(-2016)=2025>0$, $\sqrt{\Delta'}=45$.
	\\ Phương trình có hai nghiệm là $x_1=-3+45=42$; $x_2=-3-45=-48$.}
\end{ex}

\begin{ex}%[Đề thi vào 10 Sở giáo dục Thái Nguyên 2016]%[9D4K5]%[9D4K6]
	Cho phương trình $x^2-2mx+(m^2-4)=0 (1)$, $m$ là tham số.
	\begin{enumerate}
		\item Chứng minh phương trình $(1)$ luôn có hai nghiệm phân biệt với mọi giá trị của $m$.
		\item Gọi $x_1,x_2$ là hai nghiệm của phương trình $(1)$. Tìm $m$ để $x_1^2+x_2^2=26.$
	\end{enumerate}
	\loigiai{
	\begin{itemize}
		\item[a.] Ta có $\Delta=m^2-(m^2-4)=4>0, \forall m$. Suy ra phương trình luôn có hai nghiệm phân biệt với mọi giá trị $m$.
		\item[b.] Theo định lý Viet ta có: $\heva{x_1+x_2=2m \\ x_1x_2=m^2-4}$.
		\\ Mà $x_1^2+x_2^2=(x_1+x_2)^2-2x_1x_2=(2m)^2-2(m^2-4)=2m^2+8$.
		\\ Theo đề bài $x_1^2+x_2^2=26 \Rightarrow 2m^2+8=26 \Rightarrow m \pm 3$.
		\\ Thử lại: Với $m=3$, ta có phương trình $x^2-6x+5=0$ có hai nghiệm $x_1=1,x_2=5 \Rightarrow x_1^2+x_2^2=26$, thỏa mãn đề bài.
		\\ Tương tự với $m=-3$ cũng thỏa mãn đề bài. Vậy $m=\pm 3$.
\end{itemize}}
\end{ex}

\begin{ex}%[Đề thi vào 10 Sở giáo dục Thái Nguyên 2016]%[9H1G2]
	Không tính từng giá trị cụ thể, hãy sắp xếp các tỉ số lượng giác $\cos 20^{\circ}, \sin 38^{\circ}, \cos 55^{\circ}, \tan 48^{\circ}, \sin 88^{\circ}$ theo thứ tự tăng dần, giải thích?
	\loigiai{Ta có biến đổi sau: $\sin 38^\circ = \cos 52^\circ, \sin 88^\circ = \cos 2^\circ$.
	\\ Với $0<\alpha<90^\circ$, nếu $\alpha$ càng lớn thì $\cos \alpha$ càng nhỏ và $\cos \alpha<1$ nên $0 < \cos 55^\circ < \cos 52 ^\circ < \cos 20^\circ < \cos 2^\circ<1$.
	\\ Nếu $45^\circ \leq \beta <90^\circ$ thì $\tan \beta \geq 1 \Rightarrow \tan 48^\circ >1$.
	\\ Do đó ta sắp xếp theo thứ tự tăng dần như sau:
$$\cos 55^\circ, \sin 38^\circ,\cos 20^\circ, \sin 88^\circ, \tan 48^\circ$$}
\end{ex}

\begin{ex}%[Đề thi vào 10 Sở giáo dục Thái Nguyên 2016]%[9H1G2]
	Cho tam giác $ABC$ vuông tại $A$ có $\sin B=\frac{1}{3}$. Hãy tính các tỉ số lượng giác của góc $C$.
	\loigiai{\begin{center}
			\begin{tikzpicture}
		\tkzDefPoints{0/0/A, 0/3/C, 5/0/B}
		\tkzDrawPoints(A,B,C)
		\tkzLabelPoints[left](A,C)
		\tkzLabelPoints(B)
		\tkzDrawSegments(A,C B,C A,B)
		\tkzMarkRightAngle(C,A,B)
		\end{tikzpicture}
	\end{center}
Ta có $\sin B =\dfrac{1}{3} \Rightarrow \cos C =\dfrac{1}{3}$.
\\ Suy ra $\dfrac{AB}{BC}=\dfrac{1}{3} \Rightarrow \dfrac{AC^2}{BC^2}=\dfrac{1}{9} \Rightarrow \dfrac{AC^2}{AB^2}= \dfrac{1}{8} \Rightarrow \dfrac{AC}{AB} =\dfrac{1}{2\sqrt{2}}$
\\ hay $\cot C =\dfrac{1}{2\sqrt{2}}$, suy ra $\tan C =2\sqrt{2}$.
\\ $\sin C =\dfrac{AB}{BC}=\dfrac{\frac{AB}{AC}}{\frac{BC}{AC}} = \tan C. \cos C =2\sqrt{2} .\dfrac{1}{3}= \dfrac{2\sqrt{2}}{3}$}
\end{ex}

\begin{ex}%[Đề thi vào 10 Sở giáo dục Thái Nguyên 2016]%[9H3G7]
	Cho đường tròn tâm $ O $ và một điểm $ A $ nằm ngoài đường tròn. Kẻ hai tiếp tuyến $ AB, AC $ với đường tròn ($ B,C $ là hai tiếp điểm). Qua $ C $ kẻ một đường thẳng song song với $ OB $, cắt $ OA $ tại $ H $. Chứng minh rằng tứ giác $ ABOC $ nội tiếp được trong một đường tròn và $ H $ là trực tâm của tam giác $ ABC $.
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\tkzDefPoints{0/0/O, -1.8/2.4/B, -5/0/A, -3/0/I, -1.8/-2.4/C}
			\tkzDrawPoints(O,A)
			\tkzLabelPoints[left](A)
			\tkzLabelPoints(O)
			\tkzDrawPoints(B,C)
			\tkzLabelPoints[above](B)
			\tkzLabelPoints[below](C)
			\tkzDrawSegments(O,A O,B O,C A,B A,C B,C)
			\tkzDrawCircle(O,I)
			\tkzDrawAltitude(A,B)(C)
			\tkzDefPoints{-3.6/0/H}
			\tkzDrawPoints(H)
			\tkzLabelPoints[below](H)
			\tkzMarkRightAngle(A,B,O)
			\tkzMarkRightAngle(A,C,O)
			\end{tikzpicture}
		\end{center}
	Xét tứ giác $ABOC$ có $\widehat{ABO} =\widehat{ACO}=90^\circ$ (Theo tính chất tiếp tuyến).
	\\ Suy ra tứ giác $ABOC$ nội tiếp được trong một đường tròn.
	\\ Ta có $\heva{CH \parallel OB \\ OB \perp AB} \Rightarrow CH \perp AB$.
	\\ Trong tam giác $ABC$ có: $\heva{CH \perp AB \\ AH \perp BC} \Rightarrow$ $H$ là trực tâm của tam giác $ABC$.}
\end{ex}

\begin{ex}%[Đề thi vào 10 Sở giáo dục Thái Nguyên 2016]%[9H3G3]%[9H3G7]
	Cho tứ giác $ABCD$ nội tiếp trong đường tròn $(O;R)$ có hai đường chéo vuông góc với nhau và cắt nhau tại $I$. 	
	\begin{enumerate}
		\item Chứng minh rắng: $IA.DC=ID.AB$.
		\item Tính tống $AB^2+CD^2$ theo $R$.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\tkzDefPoints{0/0/I, 1/0/B, -3/0/D, 0/1/A, 0/-3/C}
			\tkzDrawPoints(A,B,C,D,I)
			\tkzLabelPoints[above](A)
			\tkzLabelPoints(B,C,I)
			\tkzLabelPoints[left](D)
			\tkzDrawSegments(A,C B,D A,B B,C C,D)
			\tkzCircumCenter(A,B,C)
			\tkzGetPoint{O}
			\tkzDrawPoints(O)
			\tkzLabelPoints(O)
			\tkzDrawCircle[radius](O,A)
			\tkzDefPointBy[symmetry = center O](C)
			\tkzGetPoint{E}
			\tkzDrawPoints(E)
			\tkzLabelPoints[above](E)
			\tkzDrawSegments(E,A E,D E,C)
			\tkzMarkRightAngle(A,I,B)
			\end{tikzpicture}
		\end{center}
	\begin{itemize}
		\item[a.] Xét hai tam giác $IAB$ và $IDC$ có: $\widehat{AIB}= \widehat{CID}$ (theo giả thiết $AC \perp BD$) và $\widehat{BAI} =\widehat{IDC}$ (góc nội tiếp cùng chắn cung $BC$).
		\\ Suy ra $\Delta IAB$ đồng dạng với $\Delta IDC$, do đó ta có $\dfrac{IA}{ID}= \dfrac{AB}{DC} \Rightarrow IA.DC=ID.AB$.
		\item[b.] Kẻ đường kính $CE$ của đường tròn $(O)$.
		\\ Ta có $\widehat{EAC} = \widehat{EDC} =90^\circ$ (góc nội tiếp chắn đường kính $EC$)
		\\ Suy ra $\heva{AC \perp BD \\ AC \perp AE} \Rightarrow AE \parallel BD \Rightarrow ABDE$ là hình thang cân (hình thang nội tiếp đường tròn) $\Rightarrow AB=DE$.
		\\ Do đó $AB^2+CD^2=DE^2+CD^2=EC^2=(2R)^2=4R^2$ (Do $\Delta DEC$ vuông tại $D$).
		\\ Vậy $AB^2+CD^2=4R^2$.
\end{itemize}}
\end{ex}
