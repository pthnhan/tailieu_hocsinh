\begin{ex}%[8D3B3]%[8D3B4]%[8D3B5]
    Giải các phương trình sau:
    \begin{enumerate}
        \item $4x+5=17-2x.$
        \item $(x-3)^2-15=(x-1)(x+3).$
        \item $2x(x-5)+3x-15=0.$
        \item $\dfrac{1}{x}=\dfrac{x+2}{x-2}-\dfrac{2}{x^2-2x}.$
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item $4x+5=17-2x \Leftrightarrow 6x=12 \Leftrightarrow x=2.$
        \item $(x-3)^2-15=(x-1)(x+3) \Leftrightarrow x^2-6x +9-15=x^2+2x-3 \Leftrightarrow 8x=-3 \Leftrightarrow x= -\dfrac{3}{8}.$
        \item $2x(x-5)+3x-15=0 \Leftrightarrow 2x(x-5)+3(x-5)=0 \Leftrightarrow (x-5)(2x+3)=0 \Leftrightarrow \hoac{&x=5\\&x=-\dfrac{3}{2}}.$
        \item
        \begin{eqnarray*}
        	&\dfrac{1}{x}&=\dfrac{x+2}{x-2}-\dfrac{2}{x^2-2x}\\
        	\Leftrightarrow&\dfrac{1}{x}&=\dfrac{x+2}{x-2}-\dfrac{2}{x(x-2)} \qquad (*)
      	\end{eqnarray*} 
        Điều kiện: $\heva{&x \ne 0\\& x \ne 2}$.   
        Với điều kiện trên, phương trình $(*)$ tương đương
        \begin{eqnarray*}
        	&x-2&=x(x+2)-2\\        	
        	\Leftrightarrow& x^2+x&=0\\        	
        	\Leftrightarrow& x(x+1)&=0\\        	
        	\Leftrightarrow& \hoac{&x=0 \quad\text{(loại)}\\&x=-1 \quad\text{(nhận)}}.&
        \end{eqnarray*}
    \end{enumerate}
    }
\end{ex}

\begin{ex}%[8D3K6][8D3T6]
    Giải bài toán bằng cách lập phương trình:\\
    Một cái sân hình chữ nhật có chiều dài lớn hơn chiều rộng $2$m và có chu vi là $20$m.
    \begin{enumerate}
    	\item Tính chiều rộng, chiều dài cái sân.
    	\item Người ta dùng loại gạch hình vuông có cạnh là $4$dm để lát hết cái sân đó. Biết giá tiền $1$ viên gạch là $20000$ đồng. Hỏi người ta phải trả bao nhiêu tiền gạch?
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item Gọi $x$(m) là chiều rộng của hình chữ nhật ($x\ge0$).
        
        Suy ra $x+2$(m) là chiều dài của hình chữ nhật.
        
        Theo đề bài ta có phương trình
        
        $2(x+2+x)=20 \Leftrightarrow 2x+2=10 \Leftrightarrow x=4 \quad \text{(nhận)}$.
        
        Chiều rộng của hình chữ nhật là $4$m.
        
        Chiều dài của hình chữ nhật là $6$m.
        \item Đổi $4$ dm=$0,4$ m.
        
        Diện tích của sân: $4 . 6 =24 $m$^2$.
        
        Số viên gạch cần dùng: $24:0,4=60$ (viên).
        
        Số tiền gạch phải trả: $60 . 20000=1200000$ (đồng).
    \end{enumerate}
    }
\end{ex}

\begin{ex}%[8H3T2]
	Bóng của một cột điện trên mặt đất dài $5$m. Cùng lúc đó một cây đèn giao thông cao $2,5$m có bóng dài $2$m. Tính chiều cao cột điện?
	\begin{center}
		\begin{tikzpicture}[scale=0.8]
	%		\tkzInit[xmax=10,ymax=7]
	%		\tkzAxeXY
	%		\tkzGrid
	\tkzDefPoints{0/0/A,5/0/B,5/6.25/C,2/2.5/N,2/0/M,0/-1/E,5/-1/F}
	\tkzDrawPoints(E,F)
	\tkzLabelPoints(B,M)
	\tkzDrawSegments(A,B B,C A,C M,N E,F)
	\tkzLabelPoints[above left](N,C)
	\tkzLabelPoints[below left](A)
	\tkzLabelSegment[right](M,N){$2,5$m}
	\tkzLabelSegment[right](B,C){$?$}
	\tkzLabelSegment[above](A,M){$2$m}
	\tkzLabelSegment[below](E,F){$5$m}
	\tkzMarkRightAngle(N,M,B) 
	\tkzMarkRightAngle(M,B,C)
	\end{tikzpicture}
\end{center}
	
\loigiai{
	
	Ta có $MN \parallel BC \Rightarrow \triangle AMN \backsim \triangle ABC$
	$\Rightarrow \dfrac{AM}{AB}=\dfrac{MN}{BC}$.
	
	$\Rightarrow BC=\dfrac{MN.AB}{AM}=\dfrac{2,5.5}{2}=6,25$m.
	
	Chiều cao của cây cột điện là $6,25$m.
	
}
\end{ex}
\begin{ex} %[8H3B7]%[8H3K3]%[8H3G6]
    Cho tam giác $ABC$ vuông tại $A$ có $AB=18$cm, $AC=24$cm với đường cao $AH$ ($H$ thuộc $BC$).
    \begin{enumerate}
        \item Chứng minh: $AB^2=BH.BC$.
        \item Kẻ đường phân giác $CD$ của tam giác $ABC$ ($D$ thuộc $AB$). Tính độ dài $DA$.
        \item Từ $B$ kẻ đường thẳng vuông góc với đường thẳng $CD$ tại $E$ và cắt đường thẳng $AH$ tại $F$. Trên đoạn thẳng $CD$ lấy điểm $G$ sao cho $BA=BG$.
        Chứng minh: $BG \bot FG$.
    \end{enumerate}
\loigiai
    {
    \immini{
    \begin{enumerate}
        \item Xét $\triangle ABC$ và $\triangle ABH$, ta có
        
       $\heva{&\widehat{BAC}=\widehat{AHB} \quad (=90^0)\\
       &\widehat{B} \quad \text{chung.}}$
        
        $\Rightarrow \triangle ABC \backsim \triangle HBA$ (g-g).
        
        $\Rightarrow \dfrac{AB}{BH}=\dfrac{BC}{AB}$.
        
        $\Rightarrow AB^2=BH.BC$.
    \end{enumerate}}{
        \begin{tikzpicture}[scale=0.8]   
        \tkzInit[xmax=5,ymax=5]
        %    \tkzAxeXY
        %	\tkzGrid
        \tkzDefPoints{0/0/A, 3/0/B, 0/5/C}
        \tkzDrawPolygon(A,B,C)
        \tkzDefPointBy[projection = onto B--C](A) \tkzGetPoint{H} 
        \tkzDefLine[bisector](A,C,B) \tkzGetPoint{x} 
        \tkzInterLL(A,B)(C,x) \tkzGetPoint{D} 
        \tkzDefLine[perpendicular=through B,K=-.5](C,D) \tkzGetPoint{c}
        \tkzInterLL(B,c)(A,H) \tkzGetPoint{F}
        \tkzInterLL(C,D)(B,c) \tkzGetPoint{E}
        \tkzInterLC(C,D)(B,A) \tkzGetPoints{G'}{G} 
        \tkzDrawSegments(C,E F,H F,B B,G F,G)
        \tkzMarkRightAngles(C,A,B F,H,B C,E,B) 
        \tkzMarkAngles[mark=|](A,C,D D,C,B) 
        \tkzMarkSegments[mark=||](A,B B,G)
        \tkzLabelPoints(A,B,F,E)
        \tkzLabelPoints[above right](C,D,G,H)
        %    \tkzDrawPoints(E,R,T)
        
        \end{tikzpicture}}
    \begin{enumerate}
        \item[b)] Xét tam giác vuông $ABC$, theo định lý Pytago, ta có
        
        $BC^2=AB^2+AC^2=18^2+24^2=900 \Rightarrow BC=30 $cm.
        
        Ta có $CD$ là phân giác của $\triangle ABC$ $\Rightarrow \dfrac{AD}{AC}=\dfrac{BD}{BC}=\dfrac{AB}{AC+BC}$.
        
        $\Rightarrow \dfrac{AD}{24}=\dfrac{18}{24+30}=\dfrac{1}{3} \Rightarrow AD= 8$cm.
        \item[c)] Xét $\triangle CBE$ và $\triangle FHB$ ta có
        
        $\heva{&\widehat{CEB}=\widehat{FHB} \quad(=90^0)\\
        	&\widehat{B}\quad \text{chung.}}$
		
		$\Rightarrow \triangle CEB \backsim \triangle FHB$ (g-g).
		
		$\Rightarrow \dfrac{BE}{BH}=\dfrac{BC}{BF}$.
		
		$\Rightarrow BE.BF=BH.BC$.
		
		Lại có, $\heva{&BH.BC=AB^2 \quad \text{(cmt)}\\
		&AB=BG\quad \text{(gt).}}$
		
		$\Rightarrow BE.BF=BG^2 \Rightarrow \dfrac{BE}{BG}=\dfrac{BG}{BF}$.
		
		Xét $\triangle BEG$ và $\triangle BGF$, ta có
		
		$\heva{&\widehat{B}\quad \text{chung}\\
		&\dfrac{BE}{BG}=\dfrac{BG}{BF}\quad \text{(cmt).}}$
		
		$\Rightarrow\triangle BEG \backsim \triangle BGF$ (c-g-c).
		
		$\Rightarrow\widehat{BEG}=\widehat{BGF}=90^0$.
		
		$\Rightarrow BG \perp FG$.
    \end{enumerate}
}
\end{ex}
