\begin{ex}%[Nguyễn Tài Tuệ]%[2D1K1-1]
	Cho hàm số $y=f(x)$ có đạo hàm $f^\prime(x)=x^2(x-1)(x-4)\cdot t(x)$ với mọi $ x\in \mathbb{R} $ và  $ t(x)>0 $   với mọi $ x\in \mathbb{R} $. Hàm số $g(x)=f\left(x^2\right)$ đồng biến trên khoảng nào trong các khoảng sau?
	\choice
	{$(-\infty-2)$}
	{\True $(-2;-1)$}
	{$(-1;1)$}
	{$(1;2)$}
	\loigiai{
	Ta có $g^\prime(x)=2x{f^\prime}\left(x^2\right)$.\newline
		Theo giả thiết $f^\prime(x)=x^2(x-1)(x-4)t(x)\Rightarrow{f^\prime}\left(x^2\right)=x^4\left(x^2-1\right)\left(x^2-4\right)\cdot t\left(x^2\right)$.\newline
		Từ đó suy ra $g^\prime(x)=2x^5\left(x^2-1\right)\left(x^2-4\right)\cdot\left(x^2\right)$.\newline
		Mà $ t(x)>0, \forall x\in \mathbb{R} $ nên dấu của $g^\prime(x)$ cùng dấu $2x^5\left(x^2-1\right)\left(x^2-4\right)$.\newline
		Bảng biến thiên
		\begin{center}
			\begin{tikzpicture}
			\tkzTabInit[nocadre=false,espcl=2.5,lgt=1.5]
			{$x$/0.7,$y'$/0.7,$y$/2.1}
			{$-\infty$,$-2$,$-1$,$0$,$1$,$2$,$+\infty$}
			\tkzTabLine{,-,0,+,0,-,0,+,-,0,+}
			\tkzTabVar{+/,-/,+/,-/,+/,-/,+/}
			\end{tikzpicture}
		\end{center}
	Dựa vào bảng biến thiên suy ra hàm số đồng biến trên  $(-2;-1)$.}
\end{ex}

\begin{ex}%[Nguyễn Tài Tuệ]%[2D1K1-1]
	Cho hàm số $f(x)$ có đạo hàm $f^\prime(x)=(x-1)^2\left(x^2-2x\right)$ với mọi $  x\in \mathbb{R} $. Hỏi số thực nào dưới đây thuộc khoảng đồng biến của hàm số $ g(x) =f(x^2-2x+2) $?
	\choice
	{$-2$}
	{\True $-1$}
	{$\dfrac{3}{2}$}
	{$3$}
	\loigiai{
		Ta có
		\begin{align*}
		g^\prime(x)&=2(x-1)\cdot {f^\prime}\left(x^2-2x+2\right)\\
		&=2(x-1)\left[\left(x^2-2x+2-1\right)^2\left(\left(x^2-2x+2\right)^2-2\left(x^2-2x+2\right)\right]\right]\\
		&=2(x-1)^5\left[(x-1)^4-1\right].
		\end{align*}
		Xét $2(x-1)^5\left[(x-1)^4-1\right]> 0\Leftrightarrow\left[\begin{array}{*{20}{l}}
		{0 < x < 1}\\
		{x > 2}
		\end{array}\right.$.\newline
		Suy ra hàm số đồng biến trên các khoảng $(0;1)$ và $(2;+\infty)$.\\
		 Vậy số $3$ thuộc khoảng đồng biến của hàm số $g(x)$.
		}
\end{ex}

\begin{ex}%[Nguyễn Tài Tuệ]%[2D1G1-1]
	Cho hàm số $y=f(x)$ có đạo hàm $f^\prime(x)=x{(x-1)^2}(x-2)$ với mọi $ x\in \mathbb{R} $. Hàm số $g(x)=f\left(\dfrac{5x}{x^2+4}\right)$ đồng biến trên khoảng nào trong các khoảng sau?
	\choice
	{$(-\infty;-2)$}
	{$(-2;1)$}
	{$(0;2)$}
	{\True $(2;4)$}
	\loigiai{
	 Ta có $f^\prime(x)=0\Leftrightarrow x{(x-1)^2}(x-2)=0\Leftrightarrow\left[\begin{array}{*{20}{l}}
		{x=0}\\
		{x=1}\\
		{x=2}
		\end{array}\right.$.\newline
		Xét $g'(x)=\dfrac{20-5x^2}{\left(x^2+4\right)^2} \cdot {f^\prime}\left(\dfrac{5x}{x^2+4}\right),$\\
				${g^\prime}(x)=0c\Leftrightarrow \left[\begin{array}{*{20}{c}}
		{20-5x^2=0}\\
		{\dfrac{5x}{x^2+4}=0}\\
		{\dfrac{5x}{x^2+4}=1}\\
		{\dfrac{5x}{x^2+4}=2}\\
		{\dfrac{5x}{x^2+4}=2}
		\end{array}\right. \Leftrightarrow \hoac{&x=\pm 2\\&x=0\\& x=1 (\text{ nghiệm bội chẵn})\\& x=4 (\text{ nghiệm bội chẵn}).}$ \newline
		Bảng biến thiên
		\begin{center}
			\begin{tikzpicture}[scale=1]
			\tkzTabInit[nocadre=false,lgt=1.2,espcl=2.5,deltacl=0.6]
			{$x$ /0.6,$g'(x)$ /0.6,$g(x)$ /2}
			{$-\infty$,$-2$,$0$,$1$,$2$,$4$,$ +\infty $}%
			\tkzTabLine{,-,0,+,0,-,0,-,0,+,0,+,}
			\tkzTabVar{+/,-/,+/,R/,-/,R/,+/}
					\end{tikzpicture}
		\end{center}
	
	
		Dựa vào bảng biến thiên và đối chiếu với các đáp án, suy ra hàm số đồng biến trên khoảng $(2;4)$.\newline
		\textbf{Chú ý:} Dấu của $g^\prime(x)$ được xác định như sau: Ví dụ xét trên khoảng $(4+\infty)$ ta chọn $x=5$\newline
	 $x=5\to\dfrac{20-5x^2}{\left(x^2+4\right)^2}< 0$ $(1)$\newline
	 $x=5\to\dfrac{5x}{x^2+4}=\dfrac{25}{29}\Rightarrow{f^\prime}\left(\dfrac{25}{29}\right)=\dfrac{25}{29}{\left(\dfrac{25}{29}-1\right)^2}\left(\dfrac{25}{29}-2\right) < 0$ $(2)$\newline
		Từ $(1)$ và $(2)$ suy ra $g^\prime(x) > 0$ trên khoảng $(4;+\infty).$}
			\end{ex}

\begin{ex}%[Nguyễn Tài Tuệ]%[2D1K2-1]
	Cho hàm số $y=f(x)$ có đạo hàm $f^\prime(x)=x^2(x-1){(x-4)^2}$ với mọi $ x\in \mathbb{R} $. Hàm số $g(x)=f\left(x^2\right)$ có bao nhiêu điểm cực trị?
	\choice
	{$2$}
	{\True $3$}
	{$4$}
	{$5$}
\loigiai{
 Ta có $g^\prime(x)=2x\cdot y^\prime\left(x^2\right)=2x^5\left(x^2-1\right){\left(x^2-4\right)^2}$.\\ $g^\prime(x)=0\Leftrightarrow 2x^5\left(x^2-1\right){\left(x^2-4\right)^2}=0\Leftrightarrow \hoac{&x=\pm 1\\&x=0\\&(x-2)^2(x+2)^2 =0.}$.\newline
	Ta thấy $x=\pm 1$ và $x=0$ là các nghiệm bội lẻ, do đó hàm số $g(x)$ có $3$ điểm cực trị. }
\end{ex}
