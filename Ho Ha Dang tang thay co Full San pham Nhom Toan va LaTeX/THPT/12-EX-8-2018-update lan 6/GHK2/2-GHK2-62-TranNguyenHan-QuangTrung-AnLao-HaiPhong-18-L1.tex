	\choice
	{Không có $m$ thỏa mãn}
	{$m\le -\dfrac{5}{2}$}
	{\True $m\ge 2$}
	{$m\ge-\dfrac{5}{2}$}
	\loigiai{
		Hàm số xác định khi $\sin^2x-4\cos x +2m\ge 0$ hay $-t^2-4t+2m+1\ge 0$, $\forall t\in \left[ -1;1\right]$ với $t=\cos x$.\\

\begin{ex}%[Thi thử L1 Trần Nguyên Hãn-Quang Trung-An Lão HP, 2018]%[2D2G4-5]%[Hoàng Trình, dự án(12EX-8)]
	Bạn Hùng trúng tuyển vào trường đại học Bách Khoa Hà Nội nhưng do không đủ tiền nộp học phí nên Hùng quyết định vay ngân hàng trong $4$ năm, mỗi năm vay $3.000.000$ đồng để nộp học phí với lãi suất $3\% $/năm. Sau khóa học 4 năm, Hùng phải trả nợ (cả gốc lẫn lãi) bằng hình thức trả góp hàng tháng, mỗi tháng trả T đồng (T không đổi), và phải trả hết nợ trong 5 năm. Tìm T (làm tròn đến hàng đơn vị), biết rằng từ khi bắt đầu trả nợ, ngân hàng tính lãi theo lãi suất mới là $0,25\%$/tháng.
	\choice
	{$253.982$}
	{\True $232.289$}
	{$345.821$}
	{$567.900$}
	\loigiai{
		Sau $1$ năm học số tiền Hùng nợ là $3+3r=3(1+r)$.\\
		Sau $2$ năm học số tiền Hùng nợ là $3(1+r)^2+3(1+r)$.\\
		Tương tự sau $4$ năm học số tiền Hùng nợ là $$3(1+r)^4+3(1+r)^3+3(1+r)^2+3(1+r)=12.927.407,43=A$$
		Số tiền nợ $A$ này bắt đầu được tính lãi và được trả góp trong $5$ năm.\\
		Sau $1$ tháng số tiền còn nợ là $A+Ar-T=A(1+r)-T$.\\
		Sau $2$ tháng số tiền còn nợ là $A(1+r)-T+\left(A(1+r)-T \right)r-T=A(1+r)^2-T(1+r)-T$.\\
		Tương tự, sau 60 tháng số tiền còn nợ là $A(1+r)^{60}-T(1+r)^{59}-T(1+r)^{58}-\cdots - T(1+r)-T$.\\
		Bạn Hùng trả hết nợ khi $A(1+r)^{60}-T(1+r)^{59}-T(1+r)^{58}-\cdots - T(1+r)-T=0$
		\begin{eqnarray*}
			%&&\text{Hàm số đồng biến trên} (1;3)\\
			&\Leftrightarrow& A(1+r)^{60}-T\left[ (1+r)^{59}-(1+r)^{58}-\cdots - (1+r)-1\right] =0\\
			&\Leftrightarrow& A(1+r)^{60}-T\dfrac{(1+r)^{60}-1}{1+r-1} =0\\
			&\Leftrightarrow& T=\dfrac{Ar(1+r)^{60}}{(1+r)^{60}-1}\\
		\end{eqnarray*}
		Suy ra $T=\dfrac{12.927.407,4\cdot (1+0,0025)^{60}\cdot 0,0025}{(1+0,0025)^{60}-1}\approx  232,289$ đồng.
		
	}
\end{ex}

	Quay hình vuông $ABCD$ cạnh $a$ quanh một cạnh của nó ta được một khối trụ. Tính thể tích của khối trụ đó.
	\choice
	{$2\pi a^3$}
	{$3\pi a^3$}
	{$\dfrac{1}{3}\pi a^3$}
	{\True $\pi a^3$}
	\loigiai{
		\immini{
			Theo đề ta có $l=r=a\Leftrightarrow V_{ABCD}=\pi \cdot r^2\cdot l=\pi a^3.  $	
		}
		{
			\begin{tikzpicture}[line cap=round,line join=round]
			\def\a{2}
			\def\b{0.5}
			\def\c{2}
			\coordinate[label=above:{$A$}] (A) at (0,0);
			\coordinate[label=below:{$D$}] (D) at ($(A)-(0,\c)$);
			\coordinate[label=above:{$B$}] (B) at (220:\a cm and \b cm);
			\tkzDefPointBy[rotation = center A angle 180](B)\tkzGetPoint{E}
			\coordinate[label=below:{$C$}] (C) at ($(B)-(0,\c)$);
			\coordinate[label=below:{$F$}] (F) at ($(E)-(0,\c)$);
			\draw (A) ellipse (\a cm and \b cm);
			\draw[dashed] (A)--(D) (E)--(F)--(C) ($(D)-(\a,0)$) arc (180:0:\a cm and \b cm);
			\draw($(A)+(\a,0)$)--($(D)+(\a,0)$) ($(A)-(\a,0)$)--($(D)-(\a,0)$) (B)--(A) node[midway,above]{$a$} (A)--(E)
			(B)--(C) node[midway,left]{$a$} ($(D)-(\a,0)$) arc (180:360:\a cm and \b cm);
			\tkzMarkRightAngle(E,B,C)
			\tkzMarkRightAngle(B,E,F)
			\tkzMarkRightAngle(E,F,C)
			\tkzMarkRightAngle(B,C,F)
			\tkzDrawPoints[fill=black](D,A)
			\tkzLabelPoints[above](E)
			\tkzDrawPoints(E, F, A, B, C, D);
			\end{tikzpicture}
		}
		
}\end{ex}
