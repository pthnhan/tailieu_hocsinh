\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[HK1, THPT Xuân Hòa, Vĩnh Phúc, 2018]%[Đức Nguyễn,12EX-5]%[2D1G5-6] 
	Chứng minh rằng với mọi $a$, đường thẳng $d:y=x+a$ luôn cắt đồ thị hàm số $y=\dfrac{-x+1}{2x-1}$ $(H)$ tại hai điểm phân biệt $A,B$. Gọi $k_1,k_2$ lần lượt là hệ số góc của các tiếp tuyến với $(H)$ tại $A$ và $B$. Tìm $a$ để tổng $k_1+k_2$ đạt giá trị lớn nhất.
	\loigiai{
		Phương trình hoành độ giao điểm của $d$ và $H$:\\
		$\dfrac{-x+1}{2x-1}=x+a\Leftrightarrow \heva{& x\ne\dfrac{1}{2} \\ & 2{x}^2+2ax-a-1=0}\left(*\right)$\\
		Đặt $gx=2x^2+2ax-a-1$. \\
		Vì $\heva{
			& {\Delta}'_g=a^2+2a+2>0,\forall a\\&g\left(\dfrac{1}{2}\right)=-\dfrac{1}{2}\ne 0,\forall a}$ nên $\left(*\right)$ có hai nghiệm phân biệt $x_1,x_2$ khác $\dfrac{1}{2}$ với mọi $a$. \\
		Vậy $d$ luôn cắt $H$ tại hai điểm phân biệt $A,B$ với mọi $a$.\\
		Gọi $A\left(x_1;y_1\right),B\left(x_2;y_2\right)$ với $x_1,x_2$ là hai nghiệm của $\left(*\right)$. Theo định lý Vi-ét ta có $x_1+x_2=-a,x_1x_2=\dfrac{-a-1}{2}$.\\
		Tiếp tuyến tại $A$ và $B$ có hệ số góc là $k_1=\dfrac{-1}{{\left(2x_1-1\right)}^2};k_2=\dfrac{-1}{{\left(2x_2-1\right)}^2}$.\\
		Ta có  \begin{align*}
		&k_1+k_2=\dfrac{-1}{{\left(2x_1-1\right)}^2}+\dfrac{-1}{{\left(2x_2-1\right)}^2}=-\left[\dfrac{{\left(2x_1-1\right)}^2+{\left(2x_2-1\right)}^2}{{\left(2x_1-1\right)}^2{\left(2x_2-1\right)}^2}\right]\\
		&=-\left[4{\left(x_1+x_2\right)}^2-8x_1x_2-4\left(x_1+x_2\right)+2\right]\text{(do }{\left(2x_1-1\right)}^2{\left(2x_2-1\right)}^2=1\text{)}\\
		&=-4{\left(a+1\right)}^2-2\le-2,\forall a
		\end{align*}
		Dấu bằng xảy ra $\Leftrightarrow a=-1$. Vậy $k_1+k_2$ đạt giá trị lớn nhất bằng $-2$ khi $a=-1$.
	}
\end{bt}
\begin{bt}%[HK1, THPT Xuân Hòa, Vĩnh Phúc, 2018]%[Đức Nguyễn,12EX-5]%[2H1K3-3] 
	Cho hình chóp $S.ABC$, có đáy $ABC$ là tam giác vuông tại $B$, $BC=a$, cạnh bên $SA=2a$, tam giác $SAC$ cân tại $S$ và nằm trong mặt phẳng vuông góc với đáy, góc giữa mặt bên $(SBC)$ và đáy bằng $60^\circ$. Tính thể tích khối chóp và khoảng cách giữa $SA$ và $BC$.
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\tkzDefPoints{0/0/H, -5/-1/A, 1.5/-2/B, 0/6/S}
			\tkzDefPointBy[symmetry = center H](A)
			\tkzGetPoint{C}
			\tkzDefPointBy[symmetry = center H](B)
			\tkzGetPoint{D}
			\tkzDefMidPoint(A,D)\tkzGetPoint{E}
			\tkzDefMidPoint(B,C)\tkzGetPoint{M}
			\tkzDefPointBy[projection= onto S--E](H)
			\tkzGetPoint{K}
			\tkzDrawSegments[dashed](A,D A,C D,C S,D S,H S,E E,M  K,H)
			\tkzDrawSegments(A,B B,C S,A S,B S,C)
			\tkzMarkRightAngles[size=.2](H,K,S C,B,A)
			\tkzLabelPoints[below](H,A,B)
			\tkzLabelPoints[above](S)
			\tkzLabelPoints[right](C,M)
			\tkzLabelPoints[left](E,K)
			\tkzLabelPoints[above right](D)
			\tkzDrawPoints[fill=black](S,A,B,C,D,K,H,E,M)
			\end{tikzpicture}
		\end{center}
		\begin{itemize}
			\item Tính $V_{SABC}$:\\
			Gọi $H$ là trung điểm của $AC\Rightarrow SH\bot AC$.\\
			Mà $(SAC)\bot (ABC)$ nên $SH\bot (ABC)$$\Rightarrow V_{SABC}=\dfrac{1}{3}.SH.{S}_{\Delta ABC}$.\\
			Gọi $M$ là trung điểm của $BC$, vì $BC\bot AC$ nên $HM\bot BC$.\\
			Do đó $((SBC),(ABC))=(SM,HM)=\widehat{SMH}=60^\circ$\\
			Ta có $SA=SC=2a$; $SM=\sqrt{SC^2-MC^2}=\dfrac{a\sqrt{15}}{2}$, 
			$SH=SM.\sin \widehat{SMH}=\dfrac{3a\sqrt{5}}{4}$, $AC=2HC=2\sqrt{SC^2-SH^2}=\dfrac{a\sqrt{19}}{2}$, 
			$AB=\sqrt{AC^2-BC^2}=\dfrac{a\sqrt{15}}{2}$.\\
			Vì tam giác $ABC$ vuông tại $B$ nên ${S}_{\Delta ABC}=\dfrac{1}{2}AB.BC=\dfrac{a^2\sqrt{15}}{4}$\\
			Vậy  $V_{SABC}=\dfrac{1}{3}.\dfrac{3a\sqrt{5}}{4}.\dfrac{a^2\sqrt{15}}{4}=\dfrac{5a^3\sqrt{3}}{16}$(đvtt)

			Vậy $d(BC;SA)=\dfrac{3a\sqrt{5}}{4}$.
		\end{itemize}
