\section{Đối xứng trục}
\subsection{Tóm tắt lí thuyết}

\begin{dn}
Hai điểm gọi là đối xứng nhau qua đường thẳng $d$ nếu $d$ là đường trung trực của đoạn thẳng nối hai điểm đó. Khi một điểm nằm trên đường thẳng $d$ thì điểm đối xứng với nó qua đường thẳng $d$ chính là điểm đó.
\end{dn}
\begin{dl}
Nếu hai đoạn thẳng (góc, tam giác) đối xứng nhau qua một đường thẳng thì chúng bằng nhau.
\end{dl}

\begin{dl}
Hình thang cân nhận đường thẳng đi qua hai trung điểm hai đáy làm trục đối xứng.
\end{dl}

\subsection{Các dạng toán}
\begin{vd}%[Đoàn Ngọc Lê, EX-C2-L8-2018]%[8H1K6]
Cho đường thẳng $d$ và hai điểm $A$, $B$ nằm cùng phía với đường thẳng $D$. Dựng điểm $C$ thuộc $d$ sao cho $CA+CB$ có độ dài ngắn nhất.
\loigiai{
\immini{
	\textit{Hướng suy nghĩ.} Bài toán trở nên đơn giản nếu cho $A,B$ nằm khác phía đối với $d$. Khi đó $C$ là giao điểm của $d$ với đoạn thẳng $AB$ (Hình 10a). Trong trường hợp $A,B$ nằm cùng phía với $d$, ta có thể tạo ra điểm $B'$ nằm khác phía với $A$ đối với $d$ mà độ dài $CB'$ luôn luôn bằng $CB$ khi $C$ thay đổi vị trí trên đường thẳng $d$. Điểm $B'$ chính là điểm đối xứng với $B$ qua $d$.
}{
	\begin{tikzpicture}[line cap=round, line join=round,scale=0.7]
	\tikzset{label style/.style={font=\footnotesize}}
	\tkzDefPoints{-2/0/a,2/0/d, -1/1.5/A, 1/-1.3/B}
	\tkzInterLL(A,B)(a,d)\tkzGetPoint{C}
	\draw (a)--(d)node[above]{\footnotesize $d$} (A)--(B);
	\tkzLabelPoints[above](A)
	\tkzLabelPoints[below left](C)
	\tkzLabelPoints[right](B)
	\tkzDrawPoints[fill=black](A,B,C)
	\end{tikzpicture}
}
\immini{
	\textit{Phân tích:} Vẽ $B'$ đối xứng với $B$ qua $d$. Gọi $M$ là điểm bất kì thuộc $d$. Ta có $MB'=MB$. Do đó
	$$AM+MB=AM+MB'\ge AB'\ \text{(hằng số)}.$$
	Vậy giá trị nhỏ nhất của $AM+MB$ bằng $AB'$ khi $M$ thuộc đoạn $AB'$.\\
	\textit{Cách dựng.} Dựng $B'$ đối xứng với $B$ qua $d$. Nối $A$ với $B'$, cắt $d$ ở $C$.\\
}{
	\begin{tikzpicture}[line cap=round, line join=round,scale=1]
	\tikzset{label style/.style={font=\footnotesize}}
	\clip (-2,-1.8) rectangle (3.2,2);
	\tkzDefPoints{-2/0/a,3/0/d, -1/1.5/A, 1/-1.3/B', 2/0/M}
	\tkzInterLL(A,B')(a,d)\tkzGetPoint{C}
	\tkzDefPointBy[reflection= over a--d](B')\tkzGetPoint{B}
	\tkzDefMidPoint(B,B')\tkzGetPoint{b}
	\tkzMarkRightAngle(B,b,d)
	\tkzDrawSegments[dashed](M,A M,B M,B' B,B')
	\tkzDrawSegments(A,B' a,d B,C)
	\tkzLabelPoints[above](A,d)
	\tkzLabelPoints[below left](C)
	\tkzLabelPoints[above right](B)
	\tkzLabelPoints[below right](M,B')
	\tkzDrawPoints[fill=black](A,B',B,C,M)
	\end{tikzpicture}
}
\textit{Chứng minh.} Gọi $M$ là điểm bất kì thuộc $d$. Ta có 
\begin{align*}
AM+MB&=AM+MB'\ge AB',\\
AC+CB&=AC+CB'=AB'.
\end{align*}
Vậy $AC+CB\le AM+MB$.\\
\textit{Biện luận.} Bài toán có một nghiệm hình.
}
\end{vd}

\item Trong cách dựng trên, do chú ý đến tính đối xứng trục, ta đã dựng $d$ là đường trung trực của $AB$, khi đó đường thẳng $d$ xác định, các điểm $C$ và $D$ đối xứng nhau qua $d$. \end{enumerate}
}

\subsection{Bài tập tự luyện}
\begin{bt}%[Đoàn Ngọc Lê, EX-C2-L8-2018]%[8H1K6]
Cho tam giác $ABC$ có $\widehat{A}=60^\circ$, các đường phân giác $BD$ và $CE$ cắt nhau tại $I$. Qua $E$ kẻ đường thẳng vuông góc với $BD$, cắt $BC$ ở $F$. Chứng minh rằng
\begin{enumerate}
\item $E$ và $F$ đối xứng với nhau qua $BD$.
\item $IF$ là tia phân giác của góc $BIC$.
\item $D$ và $F$ đối xứng nhau qua $IC$.
\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[line cap=round, line join=round,scale=1,font=\footnotesize]
				\tikzset{label style/.style={font=\footnotesize}}
				\clip (-1,-0.4) rectangle (5,3);
				\tkzDefPoints{0/0/A, 4/0/B}
				\tkzDefShiftPoint[A](60:3){C}
				\tkzDefLine[bisector](A,B,C)\tkzGetPoint{b}
				\tkzDefLine[bisector](A,C,B)\tkzGetPoint{c}
				\tkzInterLL(B,b)(C,c)\tkzGetPoint{I}
				\tkzInterLL(B,I)(C,A)\tkzGetPoint{D}
				\tkzInterLL(C,I)(B,A)\tkzGetPoint{E}
				\tkzDefLine[orthogonal=through E](B,D)\tkzGetPoint{e}
				\tkzInterLL(E,e)(B,C)\tkzGetPoint{F}
				\tkzMarkAngle[size=0.4](A,C,E)
				\tkzMarkAngle[size=0.48](E,C,B)
				\tkzMarkAngle[mark=|, size=0.6](C,B,D)
				\tkzMarkAngle[mark=|,size=0.55](D,B,A)
				\tkzInterLL(E,F)(B,D)\tkzGetPoint{H}
				\tkzMarkRightAngle(B,H,F)
				
				\tkzDrawSegments(A,B B,C C,A D,B E,C E,F I,F)
				\tkzDrawPoints[fill=black](A,B,C,D,E,F,H,I)
				\tkzLabelPoints[below left](A,I)
				\tkzLabelPoints[below right](B)
				\tkzLabelPoints[above](C)
				\tkzLabelPoints[above left](D)
				\tkzLabelPoints[below](E)
				\tkzLabelPoints[above right](F)
				\draw (H)+(0.1,-0.4) node{$H$};		
			\end{tikzpicture}
		\end{center}
	\begin{enumerate}
		\item Gọi $H$ là giao điểm của $EF$ và $BD$. Dễ thấy $\triangle BHE=\triangle BHF$ (cạnh góc vuông - góc nhọn). Do đó $HE=HF$ hay $H$ là trung điểm của $EF$. Vậy $E$ và $F$ đối xứng qua $BD$.
		\item Theo ý a), góc $\widehat{BIE}$ và góc $\widehat{BIF}$ đối xứng qua $BD$ nên $\widehat{BIE}=\widehat{BIF}$. Mặt khác
		\[\widehat{BIC}=180^\circ-\dfrac{\widehat{B}+\widehat{C}}{2}=180^\circ-\dfrac{180^\circ-\widehat{A}}{2}=120^\circ \]
		nên
		\[\widehat{BIF}=\widehat{BIE}=180^\circ-\widehat{BIC}=60^\circ.\]
		Suy ra $\widehat{CIF}=\widehat{BIF}=60^\circ$. Vậy $IF$ là tia phân giác góc $\widehat{BIC}$.
		\item Theo chứng minh trên ta có $\widehat{CID}=\widehat{BIE}=\widehat{CIF}$, do đó $\triangle CDI=\triangle CFI$ (g.c.g). Suy ra $CD=CF$ và $ID=IF$, hay $CI$ là đường trung trực của $DF$.\\
		Vậy $D$ và $F$ đối xứng nhau qua $CI$.
	\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đoàn Ngọc Lê, EX-C2-L8-2018]%[8H1G6]
