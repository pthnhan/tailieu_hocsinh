\section{Phương pháp nhân tử Largrange trong chứng minh bất đẳng thức}
%\begin{center}\textbf{\color{violet}{Trần Trung Kiên (Thành phố Hồ Chí Minh)}}\end{center}	
\noindent Phương pháp nhân tử \textit{Lagrange} (sẽ được học trong chương trình toán cao cấp của bậc đại học) khá hiệu quả trong những bài toán cực trị có điều kiện ràng buộc, ngoài ra còn có thể dùng để tìm điều kiện xảy ra dấu bằng của bất đẳng thức.

\subsection{Cơ sở lý thuyết}
\begin{dn}
	Cực trị $($cực đại hoặc cực tiểu$)$ có điều kiện của hàm hai biến $z=f\left ( x,y \right ) $ là cực trị của hàm này với điều kiện là các biến $x,y$ phải thỏa ràng buộc bởi phương trình $\varphi \left ( x,y \right )=0. $\\
	Để tìm cực trị có điều kiện của hàm $z=f\left ( x,y \right )$ khi hiện hữu phương trình ràng buộc $\varphi \left ( x,y \right )=0,$ người ta thiết lập một hàm bổ trợ là hàm lagrange $L\left ( x,\lambda y \right )=f\left ( x,y \right )+\lambda \varphi \left ( x,y \right ), $  trong đó $\lambda$ là một nhân tử hằng chưa xác định, gọi là nhân tử Lagrange.\\
	Điều kiện cần của cực trị là hệ ba phương trình 
	$$\left\{\begin{matrix}
	L_{x}^{'}\left ( x,y,\lambda  \right )=f_{x}^{'}\left ( x,y \right )+\lambda \varphi _{x}^{'}\left ( x,y \right )=0& \\ 
	L_{y}^{'} \left ( x,y,\lambda  \right )=f_{y}^{'}\left ( x,y \right )+\lambda \varphi _{y}^{'}\left ( x,y \right )=0& \\ 
	\varphi \left ( x,y \right ) =0& 
	\end{matrix}\right..  $$
\end{dn}
Giải hệ trên ta tìm được nghiệm là $x_0,y_0,\lambda _0 $. Vấn đề tồn tại và đặc tính của cực trị có điều kiện được giải bằng cách xét dấu vi phân cấp hai của hàm \textit{Lagrange} tại điểm $P_0\left ( x_0,y_0 \right )$ và $\lambda _{0} $- nghiệm của hệ phương trình trên. $P_0\left ( x_0,y_0 \right )$ là điểm dừng của hàm $L$.
$$d^{2}L=L_{xx}^{''}dx^2+2L_{xy}^{''}dxdy+L_{yy}^{''}dy^2  $$
Trong đó $dx,dy$ thỏa mãn ràng buộc biểu thị bằng phương trình 
$$\varphi _{x}^{'}dx+\varphi _{y}^{'}dy=0\left ( dx^2+dy^2\neq 0 \right ).  $$
Cụ thể xét hàm $f\left ( x,y \right ) $ đạt cực đại có điều kiện nếu $d^2L<0 $ và đạt cực tiểu có điều kiện nếu $d^2L>0 $ tại điểm dừng  $P_{\circ}\left ( x_\circ,y_\circ \right )$ và nhân tử $\lambda _{\circ} $.\\
\noindent{\bf Các bước cơ bản của phương pháp nhân tử Lagrange.} \\
1. Phát biểu bài toán dưới dạng mô hình toán học. Cực đại hoặc cực tiểu của hàm $z=f\left ( x,y \right ) $ với điều kiện rằng buộc $\varphi \left ( x,y \right )=0. $\\
2. Thiết lập hàm \textit{Lagrange} $ L\left ( x,y,\lambda  \right )=f\left ( x,y \right )+ \lambda \varphi \left ( x,y \right ). $\\
3. Tìm điểm dừng của $L$, tức là giải hệ phương trình $$\left\{\begin{matrix}
L_{x}^{'}\left ( x,y,\lambda  \right ) =0& \\ 
L_{y}^{'}\left ( x,y,\lambda  \right ) =0 & \\ 
L_{\lambda }^{'}\left ( x,y,\lambda  \right ) =0& 
\end{matrix}\right..  $$
4. Xét dấu $ d^{2}L $ tại điểm $ \left ( x_{\circ} ,y_{\circ}\right ) $ mà $\left ( x_{\circ} ,y_{\circ},\lambda _{\circ}\right ) $ là nghiệm của hệ phương trình ở bước $3.$
\begin{itemize}
	\item Nếu $d^{2}L\left ( x_{\circ},y_{\circ} ,\lambda _{\circ}\right )<0,\quad z_{\max}=f\left ( x_{\circ},y_{\circ} \right ) $;
	\item Nếu $d^{2}L\left ( x_{\circ},y_{\circ} ,\lambda _{\circ}\right )>0,\quad z_{\min}=f\left ( x_{\circ},y_{\circ} \right ) $.
\end{itemize}

\subsection{Một số ví dụ giải toán}
\begin{bt}\label{lgrltn1}%Bài tập 1, VMFPP-nhan-tu-Lagrange %[Hoang Trung Dung,Dự án EX_TapChi05]%[2D0K2]
	Cho hai số thực $x, y$ thỏa mãn $x+y=10.$ Tìm giá trị nhỏ nhất của biểu thức $$ f\left ( x,y \right )=x^2+y^2. $$
	\loigiai{
		\begin{itemize}
			\item Bước 1. Tìm cực tiểu đối với $ f\left ( x,y \right )=x^2+y^2 $ thỏa mãn điều kiện $\varphi \left ( x,y \right )=x+y-10=0  $
			\item Bước 2. $L\left ( x,y,\lambda  \right )=x^2+y^2+\lambda \left ( x+y-10 \right )  $
			\item Bước 3.
			\[\left\{ \begin{array}{l}
			\frac{{\partial f}}{{\partial x}} = 2x + \lambda \,\,\,\,\,\,\, = 0\\
			\frac{{\partial f}}{{\partial y}} = 2y + \lambda \,\,\,\,\,\,\, = 0\\
			\frac{{\partial f}}{{\partial \lambda }} = x + y - 10 = 0
			\end{array} \right. \Leftrightarrow \left\{ \begin{array}{l}
			x\,\; =  - \dfrac{\lambda }{2}\\
			y\,\; =  - \dfrac{\lambda }{2}\\
			\lambda \,\, =  - 10
			\end{array} \right.\]
			Điểm dừng $ \left ( 5,5,-10 \right ).$
			\item Bước 4. $ L_{xx}^{''}=2; L_{yy}^{''}=2; L_{xy}^{''}=0; d^2L\left ( 5,5,-10 \right )=2\left ( dx^2+dy^2 \right )>0 $
		\end{itemize}
		Do đó $f_{\min}=f\left ( 5,5 \right )=5^2+5^2=50.$\\
		\noindent{\bf Lưu ý.} Qua \textit{bài toán} \ref{lgrltn1} (ở trang \pageref{lgrltn1}), chúng ta đã phần nào nắm được ``tư tưởng'' phương pháp này. Để hiểu sâu hơn, ta tiếp tục  qua các bài toán khó hơn sau đây.
	}
\end{bt}

\begin{bt}%Bài tập 2, VMFPP-nhan-tu-Lagrange %[Hoang Trung Dung,Dự án EX_TapChi05]%[2D0K2]
	Nếu $a, b$ là các số thực dương thỏa mãn $a^{14}+b^{14}=2.$ Chứng minh rằng $$\frac{5a^2}{b}+\frac{3b^3}{a^2}\geq 8. $$
	\loigiai{Thiết lập hàm Lagrange $L\left ( a,b \right )=\dfrac{5a^2}{b}+\dfrac{3b^3}{a^2}-\lambda \left ( a^{14} +b^{14}-2\right ).$\\
		Điểm cực trị là nghiệm của hệ phương trình $$\left\{\begin{matrix}
		\dfrac{\partial L}{\partial a}=\dfrac{10a}{b} -\dfrac{6b^{3}}{a^{3}}-14a^{13}\lambda =0& \\ 
		\dfrac{\partial L}{\partial b}=\dfrac{-5a^{2}}{b^{2}}+9\dfrac{b^{2}}{a^{2}}-14b^{13}\lambda =0& \\ 
		a^{14}+b^{14}=2& 
		\end{matrix}\right..  $$
		Đặt $x=\dfrac{a}{b}$ ta quy ước mẫu bằng $0$ thì tử bằng $0$, đặt $P\left ( x \right )=5x^{18}-9x^{14}+10x^{4}-6=0. $\\
		$P\left ( x \right )=\left ( x^{2}-1\right )Q\left ( x \right ), $ với $Q(x)$ không có nghiệm thực.\\
		Vậy $a,b>0$ khi $x=1,$ điểm cực trị tại $a=b=1,$ ta có giá trị nhỏ nhất bằng $8.$
	}
\end{bt}

\begin{bt}[British Mathematical Olympiad 1986] %\textbf{(Bài 3 - Trang 2)} %Bài tập 3, VMFPP-nhan-tu-Lagrange %[Nguyễn Anh Tuấn,Dự án EX_TapChi05]%[2D0K2]
	Xét các số thực $a$, $b$, $c$ thoả mãn $a+b+c=0$ và $a^2+b^2+c^2=6$. 
	Tìm giá trị lớn nhất của biểu thức $A=a^2 b + b^2 c + c^2 a$.
	\loigiai{ 
		Bài toán này dấu bằng xảy ra khá đặc biệt khi $ x=2\cos \dfrac{2\pi}{9};y=2\cos \dfrac{4\pi}{9};z=2\cos \dfrac{8\pi}{9} $.\\
		Chính vì thế nên bài toán gây khó dễ cho các phương pháp ta đã biết, thậm chí là phương pháp mạnh như S.O.S	, UMV,... Trong quyển sử dụng phương pháp \textit{Cauchy-Schwarz} để chứng minh bất đẳng thức anh Cẩn có một lời giải khá độc đáo như sau:\\
		Xét hàm nhân tử \textit{Lagrange} như sau:
		$$ f(a;b;c)=a^2b+b^2c+c^2a+\lambda_1(a+b+c)+\lambda_2(a^2+b^2+c^2-6) $$
		Các điểm cực trị là nghiệm hệ phương trình:
		{\allowdisplaybreaks	
			$ \heva{&\dfrac{\partial f}{\partial a}=\dfrac{\partial f}{\partial b}=\dfrac{\partial f}{\partial c}=0\\&a+b+c=0\\&a^2+b^2+c^2=6 } $} $\Leftrightarrow$	{\allowdisplaybreaks	$ \heva{ & 2ab+c^2+\lambda_1 +2\lambda_2 a=0\\&2bc+a^2+\lambda_1 +2\lambda_2 c=0\\&2ca+b^2+\lambda_1 +2\lambda_2 c=0\\&a+b+c=0\\&a^2+b^2+c^2=6 } $}\\
		Cộng vế với vế của phương trình thứ nhất, thứ hai, thứ ba ta được:
		$$\left( {a + b + c} \right)^2  + 3\lambda _1  + 2\lambda _2 \left( {a + b + c} \right) = 0 \Rightarrow \lambda _1  = 0.$$
		Đến đây ta được
		$$\left\{ \begin{array}{l}
		2ab + c^2  + 2\lambda _2 a = 0 \\ 
		2bc + a^2  + 2\lambda _2 b = 0 \\ 
		2ca + b^2  + 2\lambda _2 c = 0 \\ 
		a + b + c = 0 \\ 
		a^2  + b^2  + c^2  = 0 \\ 
		\end{array} \right.$$
		Từ đây ta có
		$$\frac{{2ab + c^2 }}{a} = \frac{{2bc + a^2 }}{b} = \frac{{2ca + b^2 }}{c} =  - 2\lambda _2 .$$
		Đây là điều kiện của dấu đẳng thức trong bất đẳng thức 	\textit{Cauchy-Schwarz} nên ta sẽ chứng minh bài toán như sau:\\
		Theo bất đẳng thức 	\textit{Cauchy-Schwarz} ta có:
		$$ [a(2ab+c^2)+b(2bc+a^2)+c(2ac+b^2) ]^2 \le (a^2+b^2+c^2)[(2ab+c^2)^2+(2bc+a^2)^2+(2ac+b^2)^2]  $$
		Mặt khác ta có: $a(2ab+c^2)+b(2bc+a^2)+c(2ac+b^2)=3(a^2b+b^2c+c^2a)  $ và\\
		$ \sum (2ab+c^2)^2=2(ab+bc+ca)+(a^2+b^2+c^2)^2=54 $.\\	
		Nên do đó $ A= a^2 b + b^2 c + c^2 a \le 6$.\\
		Dấu $ = $ xảy ra khi $ x=2\cos \dfrac{2\pi}{9};y=2\cos \dfrac{4\pi}{9};z=2\cos \dfrac{8\pi}{9}  $			}
\end{bt}

\begin{bt}%Bài tập 4, VMFPP-nhan-tu-Lagrange %[Nguyễn Anh Tuấn,Dự án EX_TapChi05]%[2D0K2]
	Xét các số thực $x, y, z$ thỏa mãn $x+y+z=0$ và $ x^2+y^2+z^2=2 $. Tìm giá trị lớn nhất, giá trị nhỏ nhất của biểu thức $ P=x^3+y^3+z^3$.
	\loigiai{Đặt $ f(x;y;z)=x^3+y^3+z^3+\lambda_1 (x+y+z)+\lambda_2 (x^2+y^2+z^2-2) $.\\
		Điểm cực trị là nghiệm của hệ 
		{\allowdisplaybreaks	
			$ \heva{&\dfrac{\partial f}{\partial x}=\dfrac{\partial f}{\partial y}=\dfrac{\partial f}{\partial z}=0\\&x+y+z=0\\&x^2+y^2+z^2=2 } $} $\Leftrightarrow$	{\allowdisplaybreaks	$ \heva{ & 3x^2+\lambda_1 +2\lambda_2 x=0\\&3y^2+\lambda_1 +2\lambda_2 y=0\\&3z^2+\lambda_1 +2\lambda_2 z=0\\& x+y+z=0\\&x^2+y^2+z^2=2} $}\\
		Cộng lại ta có $$ 3(x^2+y^2+z^2)+3\lambda_1+2\lambda_2(x+y+z)=0 \Leftrightarrow \lambda_1=-2 .$$ Thay vào hệ trên ta có:
		{\allowdisplaybreaks	$$ \heva{ & 3x^2 +2\lambda_2 x-2=0\\&3y^2 +2\lambda_2 y-2=0\\&3z^2+2\lambda_2 z-2=0} $$}\\
		Ở đây ta quy ước nếu mẫu bằng 0 thì tử bằng 0\\
		$$\dfrac{3x^2-2}{x}=\dfrac{3y^2-2}{y}=\dfrac{3z^2-2}{z}=-2\lambda_2. $$	
		Áp dụng bất đẳng thức Cauchy-Schwarz ta có:
		{\allowdisplaybreaks
			\begin{align*}
			\left[x(3x^2-2)+y(3y^2-2)+z(3z^2-z)\right] ^2 & \le (x^2+y^2+z^2)\left[(3x^2-2)^2+(3y^2-2)^2+(3z^2-2)^2 \right]   \\&
			=2\cdot \left[9(x^4+y^4+z^4-12) \right]=12 
			\end{align*}		
		}
		Do $ x^4+y^4+z^4=(x^2+y^2+z^2)^2-2\left[(xy)^2+(yz)^2+(xz)^2 \right] =4-2\left[(xy+yz+xz)^2-2xyz(x+y+z) \right]  $.\\
		Mà $ xy+yz+zx=\dfrac{(x+y+z)^2-(x^2+y^2+z^2)}{2}=-1 $.\\ Do đó $ x^4+y^4+z^4 =2 $.
		Vậy $ -\dfrac{2\sqrt{3}}{3}\le P \le  \dfrac{2\sqrt{3}}{3}$
		
	}
\end{bt}

\begin{bt}%Bài tập 5, VMFPP-nhan-tu-Lagrange %[Huỳnh XuânTín,Dự án EX_TapChi05]%[2D0K2]
	Xét các số thực $a, b, c$ thỏa mãn $a^2+b^2+c^2+abc=4$. Tìm giá trị nhỏ nhất của biểu thức $D=a+b+c.$
	\loigiai{Đặt $f(a;b;c)=a+b+c;  g(a;b;c)=a^2+b^2+c^2+abc-4$.
		Ta có
		$$L=f- \lambda g=a+b+c-\lambda\left(a^2+b^2+c^2+abc-4 \right). $$
		Điểm cực trị là nghiệm của hệ
		$$\heva{\dfrac{\partial L}{\partial a}&=& 1-\lambda a-\lambda bc=0\\
			\dfrac{\partial L}{\partial b}&=& 1-\lambda b-\lambda ac=0\\
			\dfrac{\partial L}{\partial c}&=& 1-\lambda c-\lambda ba=0}\Rightarrow\heva{\lambda&=&\dfrac{1}{2a+bc}\\\lambda&=&\dfrac{1}{2b+ca}\\\lambda&=&\dfrac{1}{2c+ba}.}$$
		Do $\dfrac{1}{2a+bc}=\dfrac{1}{2b+ca}\Leftrightarrow (a-b)(2-c)=0$. Tương tự $(b-c)(2-a)$ và $(c-a)(2-b)=0$.\\
		Nếu $a=b=2$, từ $a^2+b^2+c^2+abc=4$ ta tìm được $c=-2$ và vì vậy $a+b+c=6$.\\
		Nếu $a=b=c\not= 2$ ta có $3a^2+a^3=4\Leftrightarrow (a-1)(a+2)^2=0\Leftrightarrow \hoac{a&=&1\\ a&=&-2}\Leftrightarrow\hoac{D&=&=3\\D&=&-2.}$\\
		Vậy giá trị nhỏ nhất của $D=a+b+c$ là $-6$.
		
	}
\end{bt}


\begin{bt}%Bài tập 6, VMFPP-nhan-tu-Lagrange %[Huỳnh XuânTín,Dự án EX_TapChi05]%[2D0K2]
	Xét các số thực dương  $a, b, c, d$ thỏa mãn $a+b+c+d=1$. Chứng minh rằng $$abc+bcd+cda+dab\le \dfrac{1}{27}+\dfrac{176}{27}abcd.$$
	\loigiai{Đặt $f=abc+bcd+cda+dab$ ta sẽ chứng minh $f\le \dfrac{1}{27}$.\\
		Đặt $g=a+b+c+d-1$. Ta thiết lập hàm Lagrange
		$$L=f-\lambda g=abc+bcd+cda+dab-\dfrac{176}{27}abcd- \lambda(a+b+c+d-1).$$
		Ta có 
		$$\heva{\dfrac{\partial L}{\partial a}&=& bc+cd+db-\dfrac{176}{27}bcd-\lambda =0\\
			\dfrac{\partial L}{\partial b}&=& ac+cd+da-\dfrac{176}{27}acd-\lambda =0\\
			\dfrac{\partial L}{\partial c}&=& ba+ad+db-\dfrac{176}{27}bad-\lambda =0\\
			\dfrac{\partial L}{\partial d}&=& bc+ca+ab-\dfrac{176}{27}bca-\lambda =0.}$$
		Từ hệ ta tìm được $\lambda=bc+cd+db-\dfrac{176}{27}bcd=ac+cd+da-\dfrac{176}{27}acd=ba+ad+db-\dfrac{176}{27}bad= bc+ca+ab-\dfrac{176}{27}bca.$\\
		Từ $bc+cd+db-\dfrac{176}{27}bcd=ac+cd+da-\dfrac{176}{27}acd$ ta có $(b-a)\left( c+d-\dfrac{176}{27}cd\right) =0$. Thiết lập tương tự $(b-c)\left( a+d-\dfrac{176}{27}cd\right) =0$, $(b-d)\left( a+c-\dfrac{176}{27}ac\right) =0$ và $(a-c)\left( b+d-\dfrac{176}{27}bd\right) =0$. Khi đó $a=b=c=d$ và từ $a+b+c+d=1$ nên $a=b=c=d=\dfrac{1}{4}$. Vậy $f\left(\dfrac{1}{4};\dfrac{1}{4};\dfrac{1}{4};\dfrac{1}{4} \right)=\dfrac{1}{27} $. Từ đây ta có điều cần chứng minh.
	}
	
\end{bt}

\begin{bt}%Bài tập 7, VMFPP-nhan-tu-Lagrange %[Huỳnh XuânTín,Dự án EX_TapChi05]%[2D0K2]
	Xét các số thực   $a, b, c$ thỏa mãn $a+b+c>0$. Chứng minh rằng $$a^3+b^3+c^3\le \left(a^2+b^2+c^2 \right)^{\frac{3}{2}}+3abc. $$
	\loigiai{Đặt $x=\dfrac{a}{\sqrt{a^2+b^2+c^2}}, y=\dfrac{b}{\sqrt{a^2+b^2+c^2}}, z=\dfrac{c}{\sqrt{a^2+b^2+c^2}}$. Bất đẳng thức cần chứng minh được viết lại thành
		$$x^3+y^3+z^3\le \left(x^2+y^2+z^2 \right)^{\frac{3}{2}}+3xyz \quad \mbox{với}\quad x^2+y^2+z^2=1.$$
		Đặt $f=a^3+b^3+c^3-3abc, g=a^2+b^2+c^2-1.$ Ta thiết lập hàm Lagrange
		$$L=f-\lambda g=a^3+b^3+c^3-3abc-\lambda\left(a^2+b^2+c^2-1 \right). $$
		Ta có hệ sau
		$$\heva{\dfrac{\partial L}{\partial a}&=& 3a^2-3bc-2\lambda a =0\\
			\dfrac{\partial L}{\partial b}&=& 3b^2-3ac-2\lambda b=0\\
			\dfrac{\partial L}{\partial c}&=& 3c^2-3ba-2\lambda c=0}\Rightarrow \lambda=\dfrac{3\left( a^2-bc\right) }{2a}=\dfrac{3\left( b^2-ac\right) }{2b}=\dfrac{3\left( c^2-ab\right) }{2c}.$$
		Từ trên ta có $\dfrac{3\left( a^2-bc\right) }{2a}=\dfrac{3\left( b^2-ac\right) }{2b}\Leftrightarrow (a-b)(ab+bcac)=0$.
		Tương tự ta có $(b-c)(ab+bca+ca)=0$ và $(c-a)(ab+bca+ca)=0$. Khi đó $a=b=c$ hoặc $ab+bca+ca=0.$
		\begin{itemize}
			\item Nếu $a=b=c$ thì $f(a;a;a)=0<1$.
			\item Nếu $ab+bc+ca=0$ thì $(a+b+c)^2=a^2+b^2+c^2+2(ab+bc+ca)=1$. Khi đó $a+b+c=1=1$ vì $a+b+c>0$.
		\end{itemize}Vì vậy $f(a;b;c)=a^3+b^3+c^3-3abc=(a+b+c)\left(a^2+b^2+c^2-ab-bc-ca \right)=1 $. Phép chứng minh hoàn tất.
	}
\end{bt}

	Cho $a, b, c, d$ là các số thực dương thỏa mãn $abcd=1$. Chứng minh rằng:
	$$\dfrac{1}{(1+a)^2}+\dfrac{1}{(1+b)^2}+\dfrac{1}{(1+c)^2}+\dfrac{1}{(1+d)^2}\geq 1.$$
	\loigiai{Đặt $f\left(a; b; c; d\right)=\dfrac{1}{(1+a)^2}+\dfrac{1}{(1+b)^2}+\dfrac{1}{(1+c)^2}+\dfrac{1}{(1+d)^2}$ và $g\left(a; b; c; d\right)=abcd-1$.\\Ta thiết lập hàm \textit{Lagrange}
		$$L=f-\lambda g =\dfrac{1}{(1+a)^2}+\dfrac{1}{(1+b)^2}+\dfrac{1}{(1+c)^2}+\dfrac{1}{(1+d)^2}-\lambda\left(abcd-1\right).$$
		Ta có hệ sau:
		$\heva{&\dfrac{\partial L}{\partial a}=\dfrac{-4}{(1+a)^2}-\dfrac{\lambda}{a}=0\\
			&\dfrac{\partial L}{\partial b}=\dfrac{-4}{(1+b)^2}-\dfrac{\lambda}{b}=0\\	
			&\dfrac{\partial L}{\partial c}=\dfrac{-4}{(1+c)^2}-\dfrac{\lambda}{c}=0\\
			&\dfrac{\partial L}{\partial d}=\dfrac{-4}{(1+d)^2}-\dfrac{\lambda}{d}=0}\Rightarrow \heva{&\lambda=\dfrac{-4a}{(1+a)^2}\\&\lambda=\dfrac{-4b}{(1+b)^2}\\&\lambda=\dfrac{-4c}{(1+c)^2}\\&\lambda=\dfrac{-4d}{(1+d)^2.}}$\\
		Từ hệ trên ta thu được:
		$$ \dfrac{-4a}{(1+a)^2}= \dfrac{-4b}{(1+b)^2}=\dfrac{-4c}{(1+c)^2}=\dfrac{-4d}{(1+d)^2}=\lambda.$$
		Ta có
		$$ (a-b)(1-ab)=0; (a-c)(1-ac)=0; (a-d)(1-ad)=0;$$
		$$ (b-c)(1-bc)=0; (b-d)(1-bd)=0; (c-d)(1-cd)=0. $$
		Giải phương trình ta thu được: $a=b=c=d$ và từ $abcd=1$ nên $a=b=c=d=1$ vì vậy ta có $$ f(1;1;1;1)=\dfrac{1}{4}+ \dfrac{1}{4}+\dfrac{1}{4}+\dfrac{1}{4}=1.$$
		Từ $f\left(1;1;\dfrac{1}{2};1\right)=\dfrac{1}{4}+\dfrac{1}{4}+\dfrac{1}{9}+\dfrac{4}{9}=\dfrac{1}{2}+\dfrac{5}{9}>1$\\
		Nên theo phương pháp nhân tử \textit{Lagrange} ta kết luận được $f\left(a;b;c;d\right)\geq 1$. Vậy bài toán được chứng minh.
	}
\end{bt}

\begin{bt}%Bài tập 9, VMFPP-nhan-tu-Lagrange %[Ngô Quang Anh,Dự án EX_TapChi05]%[2D0K2]
	Cho $a, b, c, d$ là những số thực dương thỏa $a+b+c+d=4$. Chứng minh rằng:
	$$ 27\left(\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}+\dfrac{1}{d}\right)\geq 9\left(a^3+b^3+c^3+d^3\right)+8.$$
	\loigiai{
		Ta thiết lập nhân tử \textit{Lagrange} lấy vế trái trừ vế phải, và đặt $g$ là $a+b+c+d-4=0$.\\Ta dễ dàng thấy $\dfrac{\partial (\dfrac{\partial f}{\partial a})}{\partial a}<0$, nên tồn tại giá trị cực tiểu.\\Nên nó tồn tại hằng số $\lambda$ thỏa mãn $\dfrac{\partial f}{\partial a}=\lambda\times\dfrac{\partial g}{\partial a}\Leftrightarrow -27\left(\dfrac{1}{a^2}+a^2\right)=\lambda$.\\
		Từ trên ta có $a, b, c, d=k$ hoặc $\dfrac{1}{k}$.\\
		1) Nếu $a, b, c, d=k$ thì $k=1$, ta tìm được $f>0$.\\
		2) Nếu $a, b, c=k$ và $d=\dfrac{1}{k}$ thì $k=\dfrac{1}{3}$ và $f=0$.\\
		3) Nếu $a, b=k$ và $c,d=\dfrac{1}{k}$ thì $k=1$ và $f>0$.\\
		Từ 1), 2), 3) ta có điều cần chứng minh.
	}
\end{bt}	

\begin{bt}%Bài tập 10, VMFPP-nhan-tu-Lagrange %[Ngô Quang Anh,Dự án EX_TapChi05]%[2D0K2]
	Cho $a, b, c, d$ là các số thực dương thỏa $a+b+c+d+abcd=5$. Chứng minh rằng:
	$$ \dfrac{1}{a}+ \dfrac{1}{b}+\dfrac{1}{c}+\dfrac{1}{d}\geq 4.$$
	\loigiai{
		Đặt $f(a;b;c;d)=\dfrac{1}{a}+ \dfrac{1}{b}+\dfrac{1}{c}+\dfrac{1}{d}$ và $g(a;b;c;d)=a+b+c+d+abcd-5$.\\Ta thiết lập nhân tử \textit{Lagrange}
		$$ L=f-\lambda g=\dfrac{1}{a}+ \dfrac{1}{b}+\dfrac{1}{c}+\dfrac{1}{d}-\lambda\left(a+b+c+d+abcd-5\right) $$
		Ta có hệ phương trình: 
		$\heva{&\dfrac{\partial L}{\partial a}=-\dfrac{1}{a^2}-\lambda\left(1+bcd\right)=0\\
			&\dfrac{\partial L}{\partial b}=-\dfrac{1}{b^2}-\lambda\left(1+acd\right)=0\\	
			&\dfrac{\partial L}{\partial c}=-\dfrac{1}{c^2}-\lambda\left(1+bad\right)=0\\
			&\dfrac{\partial L}{\partial d}=-\dfrac{1}{d^2}-\lambda\left(1+bca\right)=0.}$\\
		Từ hệ trên ta thu được:
		$$ \dfrac{-1}{a^2\left(1+bcd\right)}= \dfrac{-1}{b^2\left(1+acd\right)}=\dfrac{-1}{c^2\left(1+abd\right)}=\dfrac{-1}{d^2\left(1+acb\right)}=\lambda.$$
		Từ hai phương trình đầu ta có: $a^2\left(1+bcd\right)=b^2\left(1+acd\right)\Leftrightarrow (a-b)(a+b+abcd)=0$.\\
		Do $a+b+abcd>0$ nên $a=b$. Tương tự ta có $a=c=d$ nên $a=b=c=d$.\\Sử dụng giả thiết $a+b+c+d+abcd=5$ ta thu được $a^4+4a-5=0\Leftrightarrow a=1$.\\Do đó $a=b=c=d=1$ nên $f(1;1;1;1)=1+1+1+1=4$.\\Vậy ta có điều phải chứng minh.
		
		
	}
	
\end{bt}

\begin{bt}%Bài tập 11, VMFPP-nhan-tu-Lagrange %[Ngoc Dung Ho,Dự án EX_TapChi05]%[2D0K2]
	Cho $a\,\ b,\ c$ thực dương thỏa $a + b + c =1$. Chứng minh rằng:
	\begin{gather*}
	7\left(ab+bc+ca\right) \leq 9abc +2
	\end{gather*}
	%  $$7\left(ab+bc+ca\right) \leq 9abc +2$$
	\loigiai{Đặt $f(a; b; c)=7\left(ab+bc+ca\right)-9abc-2$; $g(a; b; c)=a + b + c-1$. Ta thiết lập hàm \textit{Lagrange}
		$$L=f-\lambda g=7\left(ab+bc+ca\right)-9abc-2-\lambda \left(a+b+c-1\right)$$ 
		\noindent Điểm cực trị là nghiệm của hệ 
		$$\heva{& L_a '=7\left(b+c\right) -9bc-\lambda =0\\ & L_b '=7\left(c+a\right) -9ac-\lambda=0\\ &L_c '=7\left(a+b\right) -9ab-\lambda=0} \Rightarrow \heva{& \lambda= 7\left(b+c\right) -9bc\\ &  \lambda= 7\left(c+a\right) -9ac\\& \lambda=7\left(a+b\right) -9ab}$$ Nên $$7\left(b+c\right) -9bc=7\left(c+a\right) -9ac=\lambda \Leftrightarrow \left(b-a\right)\left(7-9c\right)=0(1)$$
		Thiết lập tương tự ta cũng có $$\left(c-b\right)\left(7-9a\right)=0(2)$$ và $$\left(a-c\right)\left(7-9b\right)=0(3)$$ Giải hệ phương trình ta có $a=b=c=\dfrac{1}{3}$ và $$f(a; b; c)=7\left(ab+bc+ca\right)-9abc-2=\dfrac{21}{9}-\dfrac{9}{27}-2=0$$
		Từ $(1)$ nếu $a=b$ và $b \neq c$ thì từ phương trình trên ta có $a=b=\dfrac{7}{9}$ và $a+b=\dfrac{14}{9}>1$. Điều này mâu thuẫn do $a+b < a+b+c=1$.\\ Nếu $7-9c=0$ thì ta không thể có $7-9a=0$ hoặc $7-9b=0$ nên từ $(2)$; $(3)$ ta phải có $b=c$ và $a=c$ nên $a=b=c=\dfrac{7}{9}$.\\ Vậy giá trị nhỏ nhất $L=0$ hay $7\left(ab+bc+ca\right) \leq 9abc+2$
	}
\end{bt}

\begin{bt}%Bài tập 12, VMFPP-nhan-tu-Lagrange %[Ngoc Dung Ho,Dự án EX_TapChi05]%[2D0K2]
	Cho $a\,\ b,\ c,\ d$ thực  thỏa $a^{2} + b^{2} + c^{2} + d^{2} =1$. Chứng minh rằng:$$A=a^{3}+b^{3}+c^{3}+d^{3}+abc+bcd+cda+dab<1$$
	
	\loigiai{
		
		\noindent Ta thiết lập hàm \textit{Lagrange}
		$$L=a^{3}+b^{3}+c^{3}+d^{3}+abc+bcd+cda+dab-1-\lambda\left(a^{2} + b^{2} + c^{2}+d^{2} -1 \right)$$       
		Điểm cực trị là nghiệm của hệ $$\heva{
			& L_a '=3a^{2}+bc+cd+db-2\lambda a =0\\ 
			& L_b '=3b^{2}+ac+cd+da-2\lambda b =0\\
			& L_c '=3c^{2}+ab+bd+da-2\lambda c =0\\ 
			& L_d '=3d^{2}+bc+ca+ab-2\lambda d =0\\
			& a^{2} + b^{2} + c^{2}+d^{2} -1=0.
		}$$
		$$\Rightarrow \heva{
			& 2\lambda=\dfrac{3a^{2}+bc+cd+db}{a} \\ 
			& 2\lambda=\dfrac{3b^{2}+ac+cd+da}{2b}\\
			& 2\lambda=\dfrac{3c^{2}+ab+bd+da}{2c} \\ 
			& 2\lambda=\dfrac{3d^{2}+bc+ca+ab}{2d}.
		}$$
		Ta qui ước nếu mẫu bằng $0$ thì tử bằng $0$.\\ Từ đây ta có:
		\begin{align*}
		\dfrac{3a^{2}+bc+cd+db}{a}&=\dfrac{3b^{2}+ac+cd+da}{2b}\\
		=\dfrac{3c^{2}+ab+bd+da}{2c}&=\dfrac{3d^{2}+bc+ca+ab}{2d}= 2\lambda
		\end{align*}
		\noindent Theo bất đẳng thức \textit{Cauchy - Schwarz} ta có:
		$$9VT^{2}=\left[\displaystyle\sum a\left(3a^{2}+bc+cd+db\right)\right]^{2} \leq \left(a^{2}+b^{2}+c^{2}\right)\left[\displaystyle\sum\left(3a^{2}+bc+cd+db\right)^{2}\right]$$
		$=9\left(a^{2}+b^{2}+c^{2}+d^{2}\right)^{2}-2\left(a^{2}\, b^{2}+b^{2}\, c^{2}+c^{2}\, a^{2} + d^{2}\, a^{2} + d^{2}\, b^{2} + d^{2}\, c^{2}\right)$\\$-abc\left(a+b+c\right)-bcd\left(b+c+d\right)-cda\left(c+d+a\right)-dab\left(d+a+b\right) \leq 9$\\
		Thật vậy bất đẳng thức cuối tương đương:\\
		$$2\left(a^{2}\, b^{2}+b^{2}\, c^{2}+c^{2}\, a^{2} + d^{2}\, a^{2} + d^{2}\, b^{2} + d^{2}\, c^{2}\right.$$ 
		$$\geq abc\left(a+b+c\right)+bcd\left(b+c+d\right)+cda\left(c+d+a\right)+dab\left(d+a+b\right)$$
		Luôn đúng do \textit{AM - GM}. Bất đẳng thức được chứng minh.

	}

\end{bt}

\begin{bt}%Bài tập 13, VMFPP-nhan-tu-Lagrange %[Ngoc Dung Ho,Dự án EX_TapChi05]%[2D0K2]
	(IMO Shortlist 2007) Cho $a_{1};\,a_{2};\cdots;\,a_{100} \geq 0$ thỏa $a_{1}^{2}+a_{2}^{2}+ \cdots +a_{100}^{2}=1$. Chứng minh rằng:
	$$a_{1}^{2}\,a_{2}+a_{2}^{2}\, a_3+ \cdots +a_{100}^{2}\, a_{1} \leq \dfrac{1}{2}$$
	\loigiai{
		Để thuận tiện biến đổi ta đặt $a_{101}=a_{1};\, a_{102}=a-2$ và $S=a_{1}^{2}\,a_{2}+a_{2}^{2}\, a_3+ \cdots +a_{100}^{2}\, a_{1} $.\\ Áp dụng bất đẳng thức \textit{Cauchy - Schwarz}:
		\begin{align*}
		9S^{2}&=\left[\displaystyle\sum\limits_{k=1}^{100}a_{k+1}\,\left(a_{k}^{2}+2a_{k+1}\,a_{k+2}\right)\right]^{2} \leq \left(\displaystyle\sum\limits_{k=1}^{100}a_{k+1}^{2}\right)\left[\displaystyle\sum\limits_{k=1}^{100}\left(a_{k}^{2}+2a_{k+1}\,a_{k+2}\right)^{2}\right]\\
		&=\displaystyle\sum\limits_{1}^{100}\left(a_{k}^{2}+2a_{k+1}\,a_{k+2}\right)^{2} =\displaystyle\sum\limits_{1}^{100} \left( a_{k}^{4} + 4a_{k}^{2}\,a_{k+1}\,a_{k+2}+4a_{k+1}^{2}\,a_{k+2}^{2}\right)^{2}
		\end{align*}
		Mặt khác theo bất đẳng thức \textit{AM - GM} ta có $$4a_{k}^{2}\,a_{k+1}\,a_{k+2} \leq 2a_{k}^{2} \left(a_{k+1}^{2}+a_{k+2}^{2}\right)$$ Từ trên ta có $$9S^{2} \leq \displaystyle\sum\limits_{k=1}^{100} \left[a_{k}^{4}+2a_{k}^{2} \left(a_{k+1}^{2}a_{k+2}^{2}\right)+4a_{k+1}^{2}\,a_{k+2}^{2}\right]
		=\displaystyle\sum\limits_{k=1}^{100} \left[a_{k}^{4}+6a_{k+1}^{2}\, a_{k}^{2}+ 2a_{k}^{2}\,a_{k+2}^{2}\right]$$
		Sử dụng các đánh giá đơn giản
		\begin{align*}
		\displaystyle\sum\limits_{k=1}^{100} \left[a_{k}^{4}+2a_{k+1}^{2}\, a_{k}^{2}+ 2a_{k}^{2}\,a_{k+2}^{2}\right] &\leq \left(\displaystyle\sum\limits_{k=1}^{100}a_{k}^{2}\right)^{2}=1\\\displaystyle\sum\limits_{k=1}^{100}a_{k}^{2}\, a_{k+1}^{2} &\leq \left(\displaystyle\sum\limits_{i=1}^{50}a_{2i-1}^{2}\right)\;\left(\displaystyle\sum\limits_{j=1}^{50}a_{2j}^{2}\right)\\
		\text{Ta được }\ 9S^{2} \leq 1+4\left(\displaystyle\sum\limits_{i=1}^{50}a_{2i-1}^{2}\right)\;\left(\displaystyle\sum\limits_{j=1}^{50}a_{2j}^{2}\right)& \leq 1+\left(\displaystyle\sum\limits_{i=1}^{50}a_{2i-1}^{2}+\displaystyle\sum\limits_{j=1}^{50}a_{2j}^{2}\right)=2
		\end{align*} 
		Từ đây suy ra $S \leq \dfrac{\sqrt{2}}{3} < \dfrac{12}{25}$. Vậy bài toán được chứng minh.
	}
\end{bt}

\begin{bt}[Ví dụ 3.5 trang 123, Tạp chí Epsilon số 5] %[Hoang Trung Dung,Dự án EX_TapChi05]%[2D0G2]
	Tìm giá trị nhỏ nhất của hàm số $ f\left(x,y\right)= 5x^2+2xy+3y^2$ với điều kiện $$g \left(x,y\right)= 7x^2+2xy+4y^2-3=0.$$ 
	\loigiai{
		Xét $L=f\left(x,y\right)+\lambda g\left(x,y\right).$ Đạo hàm theo các biến $x$ và $y$ ta được 
		\[\left\{ \begin{array}{l}
		10x + 2y + \lambda \left( {14x + 2y} \right) = 0\\
		2x + 6y + \lambda \left( {2x + 8y} \right)\;\;\;\;\, = 0
		\end{array} \right.\] 
		Từ đây tính $\lambda$ từ các phương trình rồi cho bằng nhau, ta được 
		$$ \dfrac{10x+2y}{ 14x+2y} = \dfrac{6y+2x}{8y+2x}.$$
		Từ đó suy ra $\dfrac{y}{x} = \left\{ { - 1;2} \right\}.$ Thay vào phương trình $ g\left(x,y\right)=0,$
		ta được một số điểm "nghi vấn" của cực trị. Tính các giá trị của hàm số $f$ tại các điểm này, ta thu được giá trị nhỏ nhất đạt tại các điểm  
		\[\left( {x,y} \right) = \left\{ {\left( { - \frac{1}{3};\frac{1}{3}} \right);\left( {\frac{1}{3}; - \frac{1}{3}} \right)} \right\}.\]
		Từ đó dẫn đến kết luận của bài toán.
	}
\end{bt}

\begin{bt}[Ví dụ 3.6 trang 124 tạp chí Epsilon số 5- Chọn đội tuyển Việt Nam 2001]%[NGuyễn Anh Tuấn,Dự án EX_TapChi05]%[0D0K2]
	Cho $ x, y, z $ là các số thực dương thỏa mãn điều kiện $ 2x+4y+7z=2xyz $. Tìm giá trị nhỏ nhất của biểu thức $ x+y+z $.
	\loigiai{
		Xét $$L=x+y+z+\lambda\left(2x+4y+7z-2xyz\right) $$
		khi đó hệ phương trình $ L_x=L_y=L_z=0 $ có dạng
		$$\heva{1+\lambda (2-2yz)=0\\1+\lambda (4-2zx)=0\\1+\lambda (7-2xy)=0}$$
		Từ đây ta tìm được $$\heva{2yz=2+\dfrac{1}{\lambda}\\2zx=4+\dfrac{1}{\lambda}\\2xy=7+\dfrac{1}{\lambda}}$$
		Mặt khác, điều kiện $ x+4y+7z=2xyz $, có thể viết lại thành $$\dfrac{2}{2yz}+\dfrac{4}{2zx}+\dfrac{7}{2xy}=1$$
		Thay các biểu thức vừa tìm được ở trên vào, ta tìm được
		$$\dfrac{2\lambda}{2\lambda+1}+\dfrac{4\lambda}{4\lambda+1}+\dfrac{7\lambda}{7\lambda+1}=1.$$
		Biến đổi tương đương, ta được phương trình $$112\lambda^3+50\lambda^2-1=0$$
		Phương trình này có các nghiệm $ \lambda=\dfrac{1}{8} , \lambda=\dfrac{-4\pm \sqrt{2}}{14}$ (loại vì dẫn đến $ yz<0 $). Từ đó, ta có $$\heva{2yz=10\\2zx=12\\2xy=15}$$
		Từ đây, suy ra được điểm dừng là $ (x,y,z)=\left(3;\dfrac{5}{2};2\right) $. Sự tồn tại của giá trị nhỏ nhất được đảm bảo bởi bổ đề 3, do đó $ \left(3;\dfrac{5}{2};2\right) $ chính là điểm mà hàm số đạt giá trị nhỏ nhất. }
\end{bt}


\begin{bt}[Ví dụ 3.7 trang 125 tạp chí Epsilon số 5]%[Huỳnh Xuân Tín,Dự án EX_TapChi05]%[0D0K2]
	Tìm điểm $P$ nằm trong tam giác sao cho tổng các tỷ số độ dài các cạnh trên khoảng cách từ $P$ đến các cạnh này đạt giá trị nhỏ nhất. 	
	\loigiai{Gọi $a,b,c$ là độ dài các cạnh của tam giác và $x,y,z$ là khoảng cách từ $P$ đến các cạnh tương ứng. Ta cần tìm giá trị nhỏ nhất của hàm số $f(x,y,z)=\dfrac{a}{x}+\dfrac{b}{y}+\dfrac{c}{z}.$  Trong đó các đại lượng $x,y,z$ liên quan với nhau thông qua diện tích của tam giác. Nối $P$ với các đỉnh của tam giác, ta được ba tam giác con có tổng diện tích bằng diện tích $S$ của tam giác. Như vậy $$S=\dfrac{ax}{2}+\dfrac{by}{2}+\dfrac{cz}{2}.$$
		Ta có bài toán $$\heva{f(x,y,z)=\dfrac{a}{x}+\dfrac{b}{y}+\dfrac{c}{z}\rightarrow \min\\	ax+by+cz=2S.}$$
		Lập nhân tử Lagrrange $L\left(\lambda,x,y,z\right)= \dfrac{a}{x}+\dfrac{b}{y}+\dfrac{c}{z}+\lambda\left(ax+by+cz-2S. \right) $. Hệ phương trình tìm điểm dừng có dạng
		$$\heva{-\dfrac{a}{x^2}+\lambda a&=&0\\-\dfrac{b}{y^2}+\lambda b&=&0\\-\dfrac{c}{z^2}+\lambda c&=&0.}$$
		Từ đây suy ra ngay rằng $x=y=z$. Tức là $P$ cách đều các cạnh của tam giác. Suy ra $P$ là tâm đường tròn nội tiếp tam giác.
	}
\end{bt}

\begin{bt}[Ví dụ 3.8 trang 126 tạp chí Epsilon số 5-Bất đẳng thức Holder.]%[Huỳnh Xuân Tín,Dự án EX_TapChi05]%[0D0K2]
	Cho $x_1, x_2,...,x_n$	 là các số thực dương và $p,q$ là các số dương thỏa mãn điều kiện $\dfrac{1}{p}+\dfrac{1}{q}=1$. Chứng minh rằng ta có bất đẳng thức $$\sum\limits_{k=1}^{n}{x_ky_k}\le \left(\sum\limits_{k=1}^{n}{x_k^p} \right)^{\frac{1}{p}} \left(\sum\limits_{k=1}^{n}{y_k^q} \right)^{\frac{1}{q}}. $$
	\loigiai{Do tính thuần nhất, ta chỉ cần chứng minh rằng nếu \[\sum\limits_{k=1}^{n}{x_k^p}=1, \sum\limits_{k=1}^{n}{y_k^q}=1. \tag{1}\]
		thì $\sum\limits_{k=1}^{n}{x_ky_k}\le 1.$\\

	giải này, tại sao lại biết để bình phương các đại lượng $1 - Aa_i - Bb_i$ rồi cộng lại? Chúng ta sẽ
	lại phải đổ cho \textit{“kinh nghiệm, óc phán đoán”} hay \textit{“nhạy cảm toán học”} ... Toán cao cấp sẽ giúp
	chúng ta giải thích được lời giải độc đáo này:\\
	\noindent \textit{Xét bài toán tìm giá trị lớn nhất của hàm số}
	$$f(a, b) = \left(\displaystyle\sum_{i=1}^{n}{a_i}  \right)^2+\left(\displaystyle\sum_{i=1}^{n}{b_i}\right)^2 ,$$
	\textit{với các điều kiện ràng buộc} $\displaystyle\sum_{i=1}^{n}{a_i^2}=1, \; \displaystyle\sum_{i=1}^{n}{b_i^2}=1, \;\displaystyle\sum_{i=1}^{n}{a_i b_i}=0$.\\ 
	Lập nhân tử Langrange $$L=\left(\displaystyle\sum_{i=1}^{n}{a_i}\right)^2+ \left(\displaystyle\sum_{i=1}^{n}{b_i}\right)^2+ \lambda \left(\displaystyle\sum_{i=1}^{n}{a_i^2}-1\right)+\eta \left(\displaystyle\sum_{i=1}^{n}{b_i^2}\right)^2+\mu \left(\displaystyle\sum_{i=1}^{n}{a_i b_i}\right).$$
	Các phương trình tìm điểm dừng có dạng
	$$\left\{\begin{aligned}
	&2A+2\lambda a_i +\mu b_i=0, \; i=1, 2, \ldots, n\\
	&2B+2\eta a_i +\mu b_i=0, \; i=1, 2, \ldots, n\\
	\end{aligned}\right.$$

	$$\dfrac{A}{B}=\dfrac{2 \lambda}{\mu} = \dfrac{2\mu}{2 \eta}.$$

	dừng ta có các hệ thức $1 - Aa_i - Bb_i = 0$ và đây chính là cơ sở để ta \textit{“mạnh dạn”} thực hiện các
	phép bình phương nói trên.
}

		$$P=\dfrac{z}{z^4+12}+\dfrac{x}{x^2+2}+\dfrac{y}{y^3+6} \; (x, y, z \geq 0)$$
		$\left\{\begin{aligned}
		&P_x ' = -x^2+2=0\\
		&P_y ' = 6-2y^3=0\\
		&P_z ' = 12-3z^4=0\\
		\end{aligned}\right.$ $\Rightarrow \left\{\begin{aligned}
		&x=\sqrt{2}\\
		&y=\sqrt[3]{3}\\
		&z=\sqrt[4]{4}\\
		\end{aligned}\right.$\\
		$z^4+4+4+4 \geq 4\sqrt[4]{z^4.4^3}=8\sqrt[4]{4}.z.x^2 + 2 \geq 2\sqrt{2}.x.y^3+3+3 \geq 3\sqrt[3]{9}.y$\\
		$P \leq \dfrac{1}{8\sqrt[4]{4}}+\dfrac{1}{2\sqrt{2}}+\dfrac{1}{3\sqrt[3]{9}}=\dfrac{5}{8\sqrt{2}}+\dfrac{1}{3\sqrt[3]{9}}$\\
		$\text{GTLN } P = \dfrac{5}{8\sqrt{2}}+\dfrac{1}{3\sqrt[3]{9}} \Leftrightarrow \left\{\begin{aligned}
		&x=\sqrt{2}\\
		&y=\sqrt[3]{3}\\
		&z=\sqrt[4]{4}\\
		\end{aligned}\right. \Leftrightarrow \left\{\begin{aligned}
		&a=4\\
		&b=9\\
		&c=16\\
		\end{aligned}\right.$\\
		$\text{GTNN } P= 0$
	}
\end{bt}

\subsection{Bài tập tự luyện} 
\begin{bt}
Cho $x,\,y,\,z$ thực thuộc đoạn $\left[0;1\right]$, chứng minh rằng:$$x^{2}+y^{2}+z^{2} \leq 1+x^{2}\,y+y^{2}\,z+z^{2}\,x$$
\end{bt}
\begin{bt}
	Cho $x,\,y,\,z$ thực dương. Chứng minh rằng:$$\dfrac{x}{1+x+xy}+\dfrac{y}{1+y+yz}+\dfrac{z}{1+z+zx} \leq 1$$
\end{bt}

\begin{bt}
 Cho $a,\,b,\,c,\,d$ thực dương. Chứng minh rằng:$$25 \leq \left(a+b+c+d+e\right)\;\left(\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}+\dfrac{1}{d}+\dfrac{1}{e}\right) \leq 25+6\left(\sqrt{\dfrac{p}{q}}-\sqrt{\dfrac{q}{p}}\right)^{2}$$
	
\end{bt}
\begin{bt}(USAMO 2001) Cho $a,\,b,\,c$ thực dương thỏa $a^{2}+b^{2}+c^{2}+abc=4$. Chứng minh rằng:$$ab+bc+ca-abc \leq 2$$
\end{bt}
\begin{bt}(1999 Canada Math Olympiad) Cho $x,\,y,\,z$ là những số thực không âm thỏa $x+y+z=1$. Chứng minh rằng:$$x^{2}\,y+y^{2}\,z+z^{2}\,x \leq \dfrac{4}{27}$$
\end{bt}
\begin{bt}(Bất đẳng thức Schur) Cho $a,\,b,\,c \geq 0 $. Chứng minh rằng:$$a^{3}+b^{3}+c^{3}+3abc \geq ab\left(a+b\right)+bc\left(b+c\right)+ca\left(c+a\right)$$
\end{bt}
\begin{bt}
(Romanian IMO - BMO TST - 2007) Cho $a_{1}, \, a_{2}, \ldots, a_n\, ;\,b_{1}, \,b_{2}, \ldots ,b_n \left(n \geq 2 \right) $ thỏa mãn $a_{1}^{2}+a_{2}^{2}+\cdots +a_n^{2} = b_{1}^{2}+b_{2}^{2}+\cdots +b_n^{2} =1$; $a_{1}b_{1}+a_{2}b_{2}+\cdots +a_nb_n=0$. Chứng minh rằng:$$\left(a_{1}+a_{2}+ \cdots + a_n\right)^{2}+\left(b_{1}+b_{2}+ \cdots  +b_n\right)^{2} \leq n$$
\end{bt}
\begin{bt}
Cho $a_{1}, \, a_{2}, \ldots, a_n \geq 0$ và $a_{1}\ldotp a_{2} \ldots a_n = 1$. Chứng minh: $$\dfrac{1}{\left(1+a_{1}\right)^k}+\dfrac{1}{\left(1+a_{2}\right)^k}+ \cdots +\dfrac{1}{\left(1+a_n\right)^k} \geq min\left\lbrace1; \dfrac{n}{2^k} \right\rbrace \; \forall k \geq 0$$
\end{bt}
\begin{bt}
(VMEO II) Cho $a,\,b,\,c, \,x, \,y, \, z$ thỏa $ax+by+cz=xyz$. Chứng minh rằng:$$x+y+z \geq  \sqrt{4\left(a+b+c\right)+\sqrt{8\left(ab+bc+ca\right)}}$$	
\end{bt}

\begin{bt}
Cho $a,\,b,\,c > 0$ thỏa $a^{2}+b^{2}+c^{2}=3$. Chứng minh:$$2\left(a^{2}b+b^{2}c+c^{2}a\right)+15 \geq 3\left(a+b+c\right)+4\left(ab+bc+ca\right)$$	
\end{bt}
 
\begin{bt}
 Cho $a,\,b,\,c \in \mathbb{R}$ thỏa $2a^{3}+2b^{3}+c^{3}=4$. Tìm GTLN và GTNN của $c^{2}-2ab$	
\end{bt}

\begin{bt}
 Cho $a,\,b,\,c > 0, \; a+b+c=3$. Chứng minh:$$a^{4}+b^{4}+c^{4}+ab+bc+ca \geq 6$$	
\end{bt}
