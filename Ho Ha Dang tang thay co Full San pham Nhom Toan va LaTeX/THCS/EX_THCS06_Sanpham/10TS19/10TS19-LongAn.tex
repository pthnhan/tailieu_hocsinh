\begin{ex}%[Đề thi Tuyển sinh lớp 10 năm 2018, Long An]%[Thái Văn Đạt, EX_THCS06]%[9D1Y3]%[9D1B8]%[9D1Y2]
	\begin{enumerate}
		\item Rút gọn biểu thức $T=\sqrt{3}+\sqrt{27}-4\sqrt{3}.$
		\item Rút gọn biểu thức $A=\left(\dfrac{1}{\sqrt{x}-4}+\dfrac{1}{\sqrt{x}+4}\right): \dfrac{2\sqrt{x}}{x-16}$ với $x>0,x\ne 16$.
		\item Giải phương trình $\sqrt{x^2-8x+16}=2$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có $T=\sqrt{3}+3\sqrt{3}-4\sqrt{3}=0.$\\
	Vậy $T=0$.
	\item Tập xác định $\mathsf{D}=\mathbb{R^+}\setminus\lbrace16\rbrace$. Khi đó
	\begin{align*}
		A&=\left(\dfrac{1}{\sqrt{x}-4}+\dfrac{1}{\sqrt{x}+4}\right): \dfrac{2\sqrt{x}}{x-16}\\
		&=\dfrac{\sqrt{x}+4+\sqrt{x}-4}{(\sqrt{x}-4)(\sqrt{x}+4)}:\dfrac{2\sqrt{x}}{(\sqrt{x}-4)(\sqrt{x}+4)}\\
		&=\dfrac{2\sqrt{x}}{(\sqrt{x}-4)(\sqrt{x}+4)}\cdot\dfrac{(\sqrt{x}-4)(\sqrt{x}+4)}{2\sqrt{x}}\\
		&=1.
	\end{align*}
	Vậy $A=1$.
	\item Ta có $\sqrt{x^2-8x+16}=2\Leftrightarrow \sqrt{(x-4)^2}=2\Leftrightarrow |x-4|=2\Leftrightarrow x-4=\pm 2\Leftrightarrow x\in\lbrace2;6\rbrace$.\\
	Vậy $S=\lbrace2;6\rbrace$. 
\end{enumerate}
}
\end{ex}

	\begin{enumerate}

	\item Phương trình hoành độ giao điểm là

		\item Với $x=-3$ thì $y=(-3)^2=9$, ta được giao điểm $B(-3,9)$.

\begin{ex}%[Đề thi Tuyển sinh lớp 10 năm 2018, Long An]%[Thái Văn Đạt, EX_THCS06]%[9D4Y4]%[9D3Y3]%[9D4K6]
	\begin{enumerate}
		\item Giải phương trình $5x^2-7x-6=0$.
		\item Giải hệ phương trình $\left\lbrace \begin{aligned} &x+2y=6\\ &2x-2y=6. \end{aligned} \right.$
		\item Cho phương trình (ẩn $x$) $x^2-2(m-3)x+m^2+3=0.$ 
		\begin{enumerate}
			\item Tìm $m$ để phương trình trên có hai nghiệm phân biệt $x_1,x_2$.
			\item Tìm $m$ để phương trình trên có hai nghiệm phân biệt $x_1,x_2$ thỏa mãn $x_1^2+x_2^2=86$.
		\end{enumerate} 
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có $\Delta = (-7)^2-4\cdot5\cdot(-6)=169=13^2>0$.\\
	Suy ra phương trình có hai nghiệm phân biệt 
	$x_1=\dfrac{7+13}{2\cdot 5}=2,$
	$x_2=\dfrac{7-13}{2\cdot 5}=-\dfrac{3}{5}.$\\
	Vậy phương trình có hai nghiệm $x=2$ và $x=-\dfrac{3}{5}$.
	\item Ta có $$\left\lbrace \begin{aligned} &x+2y=6\\ &2x-2y=6 \end{aligned} \right. \Leftrightarrow \left\lbrace \begin{aligned} &x=6-2y\\ &x-y=3 \end{aligned} \right. \Leftrightarrow\left\lbrace\begin{aligned} &x=6-2y\\ &6-2y-y=3\end{aligned} \right. \Leftrightarrow\left\lbrace \begin{aligned} &x=6-2y\\ &y=1 \end{aligned} \right. \Leftrightarrow \left\lbrace \begin{aligned} &x=4\\ &y=1. \end{aligned} \right.$$
	Vậy hệ phương trình có nghiệm duy nhất $(x;y)=(4;1)$.
	\item $x^2-2(m-3)x+m^2+3=0$.
	\begin{enumerate}
	\item Phương trình có hai nghiệm phân biệt khi và chỉ khi 
	$$\Delta'>0\Leftrightarrow (m-3)^2-(m^2+3)>0\Leftrightarrow 6-6m>0\Leftrightarrow m<1.$$
	Vậy với $m<1$ thì phương trình có hai nghiệm phân biệt.
	\item Với $m<1$, phương trình có hai nghiệm phân biệt $x_1, x_2$ thỏa mãn $$\left\lbrace \begin{aligned} &x_1+x_2=2(m-3)=2m-6\\ &x_1x_2=m^2+3. \end{aligned} \right.$$
	Do $x_1^2+x_2^2=(x_1+x_2)^2-2x_1x_2$ nên từ $x_1^2+x_2^2=86$ ta thu được
	\begin{align*}
	&(2m-6)^2-2(m^2+3)=86\\
	\Leftrightarrow &4m^2-24m+36-2m^2-6=86\\
	\Leftrightarrow &m^2-12m-28=0\\
	\Leftrightarrow &(m+2)(m-14)=0\\
	\Leftrightarrow &\left[ \begin{aligned} &m=-2\ (\text{thỏa mãn})\\ &m=14\ (\text{loại}). \end{aligned} \right.
	\end{align*}
	Thử lại ta thấy $m=-2$ thỏa mãn.\\
	Vậy $m=-2$.
	\end{enumerate}
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi Tuyển sinh lớp 10 năm 2018, Long An]%[Thái Văn Đạt, EX_THCS06]%[9H1Y1]%[9H3K3]
	\begin{enumerate}
		\item Cho tam giác $ABC$ vuông tại $A$, đường cao $AH$, biết $AB=5$cm và $BC=13$cm. Từ $H$ kẻ $HK$ vuông góc với $AB$ $(K\in AB)$. Tính $AC$, $BH$ và $\cos HBK$.
		\item Cho tam giác $ABC$ có ba góc nhọn nội tiếp đường tròn tâm $O$; các đường cao $AD$, $BE$ cắt nhau tại $H$ và cắt đường tròn $(O)$ lần lượt tại $I$ và $K$ ($I$ khác $A$, $K$ khác $B$).
		\begin{enumerate}
			\item Chứng minh rằng tứ giác $CDHE$ nội tiếp.
			\item Chứng minh rằng tam giác $CIK$ cân.
			\item Kẻ đường kính $BF$ của đường tròn $(O)$. Gọi $P$ là trung điểm của $AC$. Chứng minh rằng ba điểm $H$, $P$, $F$ thẳng hàng.
		\end{enumerate}
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item 
	\immini
	{
		Áp dụng định lý Py-ta-go cho tam giác $ABC$ vuông tại $A$ ta có $$BC^2=AB^2+AC^2.$$
		$\Rightarrow$ $AC=\sqrt{BC^2-AB^2}=\sqrt{13^2-5^2}=12$cm.\\
		Áp dụng hệ thức lượng cho tam giác $ABC$ vuông tại $A$ có đường cao $AH$ ta được $BA^2=BH\cdot BC$ hay $BH=\dfrac{AB^2}{BC}=\dfrac{25}{13}$cm.\\
		Ta có: $\cos HBK=\cos ABC=\dfrac{AB}{BC}=\dfrac{5}{13}$.\\
		Vậy $AC=12$cm, $BH=\dfrac{25}{13}$cm và $\cos HBK=\dfrac{5}{13}$. 
	}
	{
		\begin{tikzpicture}[scale=0.7,x=1cm,y=1cm]
		\tkzInit[ymin=-1,ymax=7.5,xmin=-1,xmax=4]
		\tkzClip
		\tkzSetUpPoint[color=black,fill=white,size=6]
		\tkzDefPoints{0/0/A,3/0/B,0/7/C}
		\tkzDefPointBy[projection=onto B--C](A)\tkzGetPoint{H}
		\tkzDefPointBy[projection=onto B--A](H)\tkzGetPoint{K}

		\tkzDrawSegments(A,B B,C C,A A,H H,K)
		\tkzDrawPoints(A,B,C,H,K)
		\tkzLabelPoints[left](A,C)
		\tkzLabelPoints[right](B,H)
		\tkzLabelPoints[below](K)
		\tkzMarkRightAngles(B,A,C A,H,B H,K,B)
		\end{tikzpicture}
	} 
	\item 
	\immini
	{
		(a) Do $\widehat{HEC}=\widehat{HDC}=90^\circ$ nên $$\widehat{HEC}+\widehat{HDC}=180^\circ.$$
		Do đó tứ giác $HECD$ nội tiếp.\\
		(b) Trong đường tròn $(O)$ hai góc nội tiếp $\widehat{IBC}$ và $\widehat{IAC}$ cùng chắn cung $\wideparen{IC}$ nên $$\widehat{IBC}=\widehat{IAC}.$$
		Mà $\widehat{IAC}=90^\circ - \widehat{ACB}=\widehat{CBK}$ nên $$\widehat{IBC}=\widehat{KBC}.$$
		Từ đó $\wideparen{CI}=\wideparen{CK}$ hay $CI=CK$.\\
		Vậy tam giác $CIK$ cân tại $C$.
	}
	{
		\begin{tikzpicture}[scale=1,x=1cm,y=1cm]
		\tkzInit[ymin=-3,ymax=6,xmin=0,xmax=8]
		\tkzClip
		\tkzSetUpPoint[color=black,fill=white,size=6]
		\tkzDefPoints{3/5/A,1/0/B,7/0/C}
		\tkzCircumCenter(A,B,C)\tkzGetPoint{O}
		\tkzDrawCircle[circum](A,B,C)
		\tkzDefPointBy[projection=onto B--C](A)\tkzGetPoint{D}
		\tkzDefPointBy[projection=onto A--C](B)\tkzGetPoint{E}
		\tkzInterLC(A,D)(O,A)\tkzGetFirstPoint{I}
		\tkzInterLC(B,E)(O,A)\tkzGetSecondPoint{K}
		\tkzInterLL(A,D)(B,E)\tkzGetPoint{H}
		\tkzDefMidPoint(A,C)\tkzGetPoint{P}
		\tkzDefPointBy[symmetry=center O](B)\tkzGetPoint{F}
		
		\tkzDrawSegments(A,B B,C C,A A,I B,K B,F C,I C,K C,H C,F A,F F,H I,B)
		\tkzDrawPoints(A,B,C,O,D,E,I,K,H,P,F)
		\tkzLabelPoints[left](B,H)
		\tkzLabelPoints[right](C,K,F)
		\tkzLabelPoints[above](A,E,P)
		\tkzLabelPoints[below](I)
		\tkzLabelPoints[below right](O,D)
		\tkzMarkRightAngles(A,D,C B,E,C F,A,B F,C,B)
		\end{tikzpicture}
	}
	(c) Trong đường tròn $(O)$ có đường kính $BF$ nên $\widehat{BAF}=\widehat{BCF}=90^\circ$ hay $BA\perp AF$ và $BC\perp FC$.\\
	Trong tam giác $ABC$, hai đường cao $AD$ và $BE$ cắt nhau tại $H$ nên $H$ là trực tâm tam giác $ABC$ hay $AH\perp BC$ và $CH\perp AB$.\\
	Từ $BA\perp AF$ và $CH\perp AB$ ta thu được $CH\parallel AF$. Tương tự thì $AH\parallel CF$.\\
	Do đó tứ giác $AHCF$ là hình bình hành. Điều này kéo theo $AC$ và $FH$ cắt nhau tại trung điểm mỗi đường, tức là $F,P,H$ thẳng hàng. 
\end{enumerate}
}
