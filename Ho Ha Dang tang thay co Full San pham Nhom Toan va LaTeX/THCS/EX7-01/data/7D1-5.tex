\section{TỈ LỆ THỨC. TÍNH CHẤT CỦA DÃY TỈ SỐ BẰNG NHAU}
	Tỉ lệ thức có rất nhiều ứng dụng. Từ một tỉ lệ thức ta có thể chuyển thành một đẳng thức giữa hai tích. Trong một tỉ lệ thức nếu biết $3$ số hạng ta có thể tìm được số hạng thứ $4$. Trong chương II, khi học đại lượng tỉ lệ thuận, tỉ lệ nghịch ta sẽ thấy tỉ lệ thức là một phương tiện quan trọng giúp ta giải toán. Trong hình học, để học được định lí Ta-lét, tam giác đồng dạng (lớp 8) thì không thể thiếu kiến thức về tỉ lệ thức.
\subsection{Tóm tắt lí thuyết}
	\begin{enumerate}[\tickEX]
		\item $\dfrac{a}{b}=\dfrac{c}{d}$ hay $a:b=c:d$ $(b, d \ne 0)$ là một tỉ lệ thức. Trong đó $a$ và $d$ gọi là ngoại tỉ, $b$ và $c$ gọi là trung tỉ.
		\item Nếu $\dfrac{a}{b}=\dfrac{c}{d}$ thì $ad=bc.$
		\item Nếu $ad=bc$ và $a, b, c, d \ne 0$ thì $\dfrac{a}{b}=\dfrac{c}{d}$; $\dfrac{a}{c}=\dfrac{b}{d}$; $\dfrac{b}{a}=\dfrac{d}{c}$; $\dfrac{c}{a}=\dfrac{d}{b}.$
		\item Khi có nhiều tỉ số bằng nhau ghép lại, ta được một dãy tỉ số bằng nhau $\dfrac{a}{b}=\dfrac{c}{d}=\dfrac{e}{f}.$
		\item Khi có dãy tỉ số $\dfrac{a}{b}=\dfrac{c}{d}=\dfrac{e}{f}$, ta nói $a$, $c$, $e$ tỉ lệ với $b$, $d$, $f$ và viết $a:c:e=b:d:f.$
		\item Nếu $\dfrac{a}{b}=\dfrac{c}{d}=\dfrac{e}{f}$ thì $\dfrac{a}{b}=\dfrac{c}{d}=\dfrac{e}{f}=\dfrac{a+c+e}{b+d+f}=\dfrac{a-c+e}{b-d+f}$ (với các giả thiết đều có nghĩa).
	\end{enumerate}
\subsection{Các dạng toán}
\begin{dang}[Lập tỉ lệ thức]
	\begin{enumerate}[\tickEX]
		\item Nếu đề bài cho đẳng thức $ad=bc$ thì ta suy ra đươc $4$ tỉ lệ thức đã được học.
		\item Nếu đề bài cho tỉ lệ thức  $\dfrac{a}{b}=\dfrac{c}{d}$ thì ta suy ra được $3$ tỉ lệ thức còn lại.
		\item Để lập tỉ lệ thức từ các số đã cho ta cần xác định $4$ số $a$, $b$, $c$, $d$ sao cho $ad=bc$. Rồi áp dụng tính chất của tỉ lệ thức để viết được $4$ tỉ lệ thức.
	\end{enumerate}	
\end{dang}
\begin{vd}
	Lập tất cả các tỉ lệ thức có thể được từ đẳng thức $3\cdot4=2\cdot6$.
	\loigiai{
		Các tỉ lệ thức là $\dfrac{3}{2}=\dfrac{6}{4}$; $\dfrac{3}{6}=\dfrac{2}{4}$; $\dfrac{2}{3}=\dfrac{4}{6}$; $\dfrac{6}{3}=\dfrac{4}{2}.$
	}
\end{vd}
\begin{vd}
	Lập tất cả các tỉ lệ thức có thể được từ tỉ lệ thức $\dfrac{-4}{12}=\dfrac{-5}{15}.$
	\loigiai{
		Các tỉ lệ thức $\dfrac{-4}{-5}=\dfrac{12}{15}$; $\dfrac{12}{-4}=\dfrac{15}{-5}$; $\dfrac{-5}{-4}=\dfrac{15}{12}.$
	}
\end{vd}
\begin{vd}
	Lập tất cả các tỉ lệ thức có thể có được từ $4$ số $-25$; $-2$; $4$; $50.$
	\loigiai{
		Ta có $(-25)\cdot4=(-2)\cdot50=100.$\\
		Từ đó ta có thể viết được các tỉ lệ thức sau \[\dfrac{-25}{-2}=\dfrac{50}{4}; \dfrac{-25}{50}=\dfrac{-2}{4};\dfrac{-2}{-25}=\dfrac{4}{50}; \dfrac{50}{-25}=\dfrac{4}{-2}.\]
	}
\end{vd}
\begin{vd}
	Cho tập hợp các số $\mathrm{A}=\{4;8;16;32;64\}$. Hãy liệt kê tất cả các tỉ lệ thức mà các số hạng của nó là $4$ số khác nhau trong các số đã cho.
	\loigiai{Có 5 nhóm  gồm $4$ số thuộc $\mathrm{A}$ là
	$\{4;8;16;32\}$, $\{8;16;32;64\}$, $\{4;8;32;64\}$, $\{4;16;32;64\}$, $\{4;8;16;64\}.$
	\begin{itemize}
		\item Với bốn số $\{4;8;16;32\}$ ta có $8\cdot16=4\cdot32$. Từ đó được $4$ tỉ lệ thức \[\dfrac{8}{4}=\dfrac{32}{16}; \dfrac{8}{32}=\dfrac{4}{16};\dfrac{4}{8}=\dfrac{16}{32}; \dfrac{32}{8}=\dfrac{16}{4}.\]
		\item Với bốn số $\{8;16;32;64\}$ ta có $8\cdot64=16\cdot32$. Từ đó được $4$ tỉ lệ thức \[\dfrac{8}{16}=\dfrac{32}{64}; \dfrac{8}{32}=\dfrac{16}{64};\dfrac{16}{8}=\dfrac{64}{32}; \dfrac{32}{8}=\dfrac{64}{16}.\]
		\item Với bốn số $\{4;8;32;64\}$ ta có $4\cdot64=8\cdot32$. Từ đó được $4$ tỉ lệ thức \[\dfrac{4}{8}=\dfrac{32}{64}; \dfrac{4}{32}=\dfrac{8}{64};\dfrac{8}{4}=\dfrac{64}{32}; \dfrac{32}{4}=\dfrac{64}{8}.\]
		\item Với các nhóm $4$ số $\{4;16;32;64\}$ và $\{4;8;16;64\}$ không thành lập được tỉ lệ thức nào.
	\end{itemize}
	}
\end{vd}
\begin{dang}[Tìm số hạng chưa biết]
	\begin{enumerate}[\tickEX]
		\item Từ tỉ lệ thức $\dfrac{a}{b}=\dfrac{c}{d}$ suy ra $ad=bc$.
		\item Có thể vận dụng tính chất của dãy tỉ số bằng nhau để tìm số hạng của tỉ lệ thức.
	\end{enumerate}
\end{dang}
\begin{vd}
	Tìm $x$ trong tỉ lệ thức sau
	\begin{listEX}[3]
		\item $\dfrac{2x}{6}=\dfrac{5}{3}.$
		\item $\dfrac{x}{8}=\dfrac{5}{4}.$
		\item $\dfrac{x^2}{6}=\dfrac{24}{25}.$
	\end{listEX}
	\loigiai{
		\begin{enumerate}
			\item $\dfrac{2x}{6}=\dfrac{5}{3}$ $\Rightarrow (2x)\cdot 3=6 \cdot 5 \Rightarrow 6x=6\cdot 5 \Rightarrow x=5.$
			\item $\dfrac{x}{8}=\dfrac{5}{4}\Rightarrow 4x=8 \cdot 5 \Rightarrow x=2 \cdot 5 =10.$
			\item $\dfrac{x^2}{6}=\dfrac{24}{25}\Rightarrow x^2=\dfrac{6 \cdot 24}{25} \Rightarrow x^2=\dfrac{144}{25}\Rightarrow x= \pm \dfrac{12}{5}.$
		\end{enumerate}
	}
\end{vd}
\begin{vd}
	Tìm hai số $x$, $y$ biết $\dfrac{x}{3}=\dfrac{y}{5}$ và $x+y=16.$
	\loigiai{
		Ta có $\dfrac{x}{3}=\dfrac{y}{5}=\dfrac{x+y}{3+5}$ mà $x+y=16.$\\
		$\Rightarrow \dfrac{x}{3}=\dfrac{y}{5}=\dfrac{16}{8}=2.$\\
		$\Rightarrow x=6$ và $y=10.$
	}
\end{vd}
\begin{vd}
	Tìm hai số $x$, $y$ biết $x:2=y:(-5)$ và $x-y=(-7).$
	\loigiai{Ta có $\dfrac{x}{2}=\dfrac{y}{-5}=\dfrac{x-y}{2-(-5)}=\dfrac{x-y}{7}$ mà $x-y=-7.$\\
		$\Rightarrow \dfrac{x}{2}=\dfrac{y}{-5}= \dfrac{-7}{7}=-1.$\\
		$\Rightarrow x=-2$ và $y=5.$
	}
\end{vd}
\begin{vd}
	Tìm hai số $x$, $y$  biết $\dfrac{x}{2}=\dfrac{y}{5}$ và $xy=10.$
	\loigiai{Đặt $\dfrac{x}{2}=\dfrac{y}{5}=k \Rightarrow \heva{x&=2k\\y&=5k.}$\\
	Thế $x$, $y$ vào đẳng thức $xy=10$, ta được $10k^2=10 \Rightarrow k^2=1 \Rightarrow k=\pm 1.$
		\begin{itemize}	
			\item $k=1 \Rightarrow x=2$ và $y=5$.
			\item $k=-1 \Rightarrow x=-2$ và $y=-5.$
		\end{itemize}
	}
\end{vd}
\begin{vd}
	Tìm ba số $x$, $y$, $z$ biết $3x=2y$, $5y=4z$ và $x+y-z=10.$
	\loigiai{$3x=2y$, $5y=4z$ $\Rightarrow$ $15x=10y=8z$ 
		$\Rightarrow \dfrac{15x}{120}=\dfrac{10y}{120}=\dfrac{8z}{120}.$\\
		$\Rightarrow \dfrac{x}{8}=\dfrac{y}{12}=\dfrac{z}{15}.$\\
		Khi đó $\dfrac{x}{8}=\dfrac{y}{12}=\dfrac{z}{15}=\dfrac{x+y-z}{8+12-15}.$ \\
		mà $x+y-z=10.$\\
		$\Rightarrow  \dfrac{x}{8}=\dfrac{y}{12}=\dfrac{z}{15}=\dfrac{10}{5}=2$
		$\Rightarrow \heva{x&=16 \\ y&=24 \\z&=30.}$
	}
\end{vd}

	\loigiai{Gọi $a$, $b$ là chiều rộng và chiều dài hình chữ nhật ($a>b>0$).\\
		Theo đề bài ta có $\dfrac{a}{b}=\dfrac{3}{4}$ và $ab=300.$\\

	}

\begin{vd}
	Tìm các số $a$, $b$, $c$ biết $\dfrac{a}{2}=\dfrac{b}{3}=\dfrac{c}{4}$ và $a^2-b^2+2c^2=108.$
	\loigiai{Đặt  $\dfrac{a}{2}=\dfrac{b}{3}=\dfrac{c}{4}=k \Rightarrow \heva{a&=2k\\b&=3k\\c&=4k.}$\\
		Thay $a$, $b$, $c$ vào đẳng thức  $a^2-b^2+2c^2=108$, ta được \[4k^2-9k^2+32k^2=108 \Rightarrow k^2=4\Rightarrow k=\pm 2.\]
		\begin{itemize}
			\item Nếu $k=2$ thì $a=4$, $b=6$, $c=8$.
			\item Nếu $k=2$ thì $a=-4$, $b=-6$, $c=-8$.
		\end{itemize}
	}
\end{vd}
\begin{vd}
	Có 75 tờ giấy bạc loại $2000$ đồng, $5000$ đồng, $20000$ đồng. Trị giá mỗi loại tiền trên đều bằng nhau. Hỏi mỗi loại có mấy tờ.
	\loigiai{Gọi  $x$, $y$, $z$ thứ tự là số giấy bạc loại $2000$ đồng, $5000$ đồng và $20000$ đồng ($x$, $y$, $z$ $\in \mathbb{N}^*$).\\
		Theo đề bài ta có $x+y+z=75$ và $2000x=5000y=20000z.$\\
		Mà  $2000x=5000y=20000z\Rightarrow \dfrac{2000x}{20000}=\dfrac{5000y}{20000}=\dfrac{20000z}{20000}.\\
		 \Rightarrow \dfrac{x}{10}=\dfrac{y}{4}=\dfrac{z}{1}=\dfrac{75}{15}=5 \Rightarrow \heva{x&=50\\y&=20\\z&=5.}$\\
		 Vậy có 50 tờ giấy loại $2000$ đồng, 20 tờ giấy loại $5000$ đồng và 5 tờ giấy loại $20000$ đồng.
	}
\end{vd}
\begin{dang}[Chứng minh tỉ lệ thức]
	Có nhiều phương pháp chứng minh tỉ lệ thức. Ta bắt đầu bằng một số ví dụ đơn giản.
\end{dang}
\begin{vd}
	Cho tỉ lệ thức $\dfrac{a}{b}=\dfrac{c}{d}\ne 1$ với $a$, $b$, $c$, $d$ $\ne 0$. Chứng minh rằng $\dfrac{a-b}{a}=\dfrac{c-d}{c}.$
	\loigiai{
		\begin{description}
			\item [\emph{Cách 1:}]  $\dfrac{a}{b}=\dfrac{c}{d} \Rightarrow ad=bc.$\\
			Xét tích $(a-b)\cdot c = ac-bc=ac-ad=a \cdot (c-d).$\\
			Vậy $(a-b)\cdot c = a \cdot (c-d) \Rightarrow \dfrac{a-b}{a}=\dfrac{c-d}{c}.$
			\begin{note}
				Trong cách này, để chứng minh tỉ lệ thức $\dfrac{a-b}{a}=\dfrac{c-d}{c}$ ta chứng minh $(a-b)\cdot c = a \cdot (c-d).$
			\end{note}
			\item [\emph{Cách 2:}] Đặt $\dfrac{a}{b}=\dfrac{c}{d}=k \Rightarrow$ $a=kb$, $c=kd.$\\
			Thế thì 
				\begin{align}
				&\dfrac{a-b}{a}=\dfrac{kb-b}{kb}=\dfrac{b(k-1)}{kb}=\dfrac{k-1}{k} &\tag{1}\label{7D1-5:1}\\
				&\dfrac{c-d}{c}=\dfrac{kd-d}{kd}=\dfrac{d(k-1)}{kd}=\dfrac{k-1}{k} &\tag{2}\label{7D1-5:2}
				\end{align}
			Từ \eqref{7D1-5:1}, \eqref{7D1-5:2} suy ra $\dfrac{a-b}{a}=\dfrac{c-d}{c}.$
			\begin{note}
			Trong cách giải này, để chứng minh tỉ lệ thức $\dfrac{a-b}{a}=\dfrac{c-d}{c}$ ta chứng minh hai tỉ số ở hai vế cùng bằng một tỉ số thứ ba. Để làm được đều đó ta đặt giá trị chung của các tỉ số ở tỉ lệ thức đã cho là $k$, từ đó tính giá trị của mỗi tỉ số ở tỉ lệ thức phải chứng minh theo $k$.
			\end{note}
			\item [\emph{Cách 3:}] $\dfrac{a}{b}=\dfrac{c}{d} \Rightarrow \dfrac{a}{c}=\dfrac{b}{d}=\dfrac{a-b}{c-d}.$\\
			Vậy $\dfrac{a-b}{c-d}=\dfrac{a}{c} \Rightarrow \dfrac{a-b}{a}=\dfrac{c-d}{c}.$
			\begin{note}
				Trong cách giải này, sau khi hoán vị các trung tỉ của tỉ lệ thức đã cho, ta dùng tính chất của dãy tỉ số bằng nhau. Cuối cùng lại hoán vị các trung tỉ của tỉ lệ thức mới được tạo ra để đi đến tỉ lệ thức phải chứng minh.
			\end{note}
			\item [\emph{Cách 4:}] Vì $\dfrac{a}{b}=\dfrac{c}{d}$ nên $\dfrac{b}{a}=\dfrac{d}{c}.$\\
			Ta có $\dfrac{a-b}{a}=\dfrac{a}{a}-\dfrac{b}{a}=1-\dfrac{b}{a}=1-\dfrac{d}{c}=\dfrac{c-d}{c}.$\\
			Vậy $\dfrac{a-b}{a}=\dfrac{c-d}{c}.$
			\begin{note}
				Trong cách giải này, ta biến đổi tỉ số ở vế trái thành vế phải. Đó cũng là cách thường dùng để chứng minh một đẳng thức nói chung.
			\end{note}
			\item [\emph{Cách 5:}] Ta có $\dfrac{a}{b}=\dfrac{c}{d} \Leftrightarrow \dfrac{b}{a}=\dfrac{d}{c} \Leftrightarrow 1-\dfrac{b}{a}=1-\dfrac{d}{c} \Leftrightarrow \dfrac{a-b}{a}=\dfrac{c-d}{c}.$
			\begin{note}
				Trong cách giải này, từ tỉ lệ thức đã cho ta biến đổi dần thành tỉ lệ thức phải chứng minh bằng cách dùng các tính chất hoán vị, tính chất của đẳng thức $\dots$
			\end{note}
			\item [\emph{Cách 6:}] 
				\begin{eqnarray*}
					\dfrac{a-b}{a}=\dfrac{c-d}{c}
					& \Leftrightarrow & (a-b) \cdot c = (c-d) \cdot a\\
					& \Leftrightarrow & ac-bc=ca-da \\
					& \Leftrightarrow & bc=da\\
				    & \Leftrightarrow & \dfrac{a}{b}=\dfrac{c}{d}.
			    \end{eqnarray*}
		    Do $\dfrac{a}{b}=\dfrac{c}{d}$ là đẳng thức đúng nên $\dfrac{a-b}{a}=\dfrac{c-d}{c}$ là đẳng thức đúng.
		    \begin{note}
		    	Trong cách giải này, từ một đẳng thức cần chứng minh ta chứng minh tương đương với một đẳng thức đúng. Do đó đẳng thức cần chứng minh là đẳng thức đúng. 		    
	    	\end{note}
    	\item [\emph{Cách 7:}] $\dfrac{a}{b}=\dfrac{c}{d} \Rightarrow ad=bc.$\\
    	Ta có $\dfrac{a-b}{a}=\dfrac{(a-b)d}{ad}=\dfrac{ad-bd}{ad}=\dfrac{bc-bd}{bc}=\dfrac{b(c-d)}{bc}=\dfrac{c-d}{c}.$
		\end{description}
	}
\end{vd}
\begin{vd}
	Cho $\dfrac{a}{b}=\dfrac{c}{d}$ $\left(c \ne \pm \dfrac{3}{5}d\right)$. Chứng minh rằng $\dfrac{5a+3b}{5c+3d}=\dfrac{5a-3b}{5c-3d}.$
	\loigiai{$\dfrac{a}{b}=\dfrac{c}{d} \Rightarrow \dfrac{a}{c}=\dfrac{b}{d}=k.$\\
		$\Rightarrow k=\dfrac{5a}{5c}=\dfrac{3b}{3d}=\dfrac{5a+3b}{5c+3d}=\dfrac{5a-3d}{5c-3d}.$\\
			Vậy $\dfrac{5a+3b}{5c+3d}=\dfrac{5a-3b}{5c-3d}.$
	}
\end{vd}
\begin{vd}
	Cho tỉ lệ thức $\dfrac{a}{b}=\dfrac{c}{d}$. Chứng minh $\dfrac{a^2 +b^2}{c^2 +d^2}=\dfrac{ab}{cd}.$
	\loigiai{$\dfrac{a}{b}=\dfrac{c}{d} \Rightarrow \dfrac{a}{c}=\dfrac{b}{d}=k.$\\
		$\Rightarrow k^2 =\dfrac{a^2}{c^2}=\dfrac{b^2}{d^2}=\dfrac{ab}{cd}=\dfrac{a^2+b^2}{c^2+d^2}.$ \\
	Vậy $\dfrac{a^2 +b^2}{c^2 +d^2}=\dfrac{ab}{cd}.$
	}
\end{vd}
\begin{vd}
	Cho  $\dfrac{a}{b}=\dfrac{c}{d} \ne \pm 1$ và $c \ne 0$. Chứng minh rằng 
	\begin{listEX} [2]
		\item $\dfrac{(a-b)^2}{(c-d)^2}=\dfrac{ab}{cd}.$
		\item $\left(\dfrac{a+b}{c+d}\right)^3=\dfrac{a^3-b^3}{c^3-d^3}.$
	\end{listEX}
	\loigiai{
		\begin{enumerate}
			\item $\dfrac{a}{b}=\dfrac{c}{d} \Rightarrow \dfrac{a}{c}=\dfrac{b}{d}=\dfrac{a-b}{c-d}.$\\
			Suy ra $\dfrac{a}{c}\cdot\dfrac{b}{d}=\dfrac{a-b}{c-d}\cdot\dfrac{a-b}{c-d}.$\\
			Hay $\dfrac{ab}{cd}=\dfrac{(a-b)^2}{(c-d)^2}.$
			\item $\dfrac{a}{b}=\dfrac{c}{d} \Rightarrow \dfrac{a}{c}=\dfrac{b}{d}=\dfrac{a+b}{c+d}.$\\
			Suy ra $\left(\dfrac{a}{c}\right)^3=\left(\dfrac{b}{d}\right)^3=\left(\dfrac{a+b}{c+d}\right)^3.$\\
			Khi đó $\dfrac{a^3}{c^3}=\dfrac{b^3}{d^3}=\dfrac{a^3-b^3}{c^3-d^3}=\left(\dfrac{a+b}{c+d}\right)^3.$\\
			Vậy $\left(\dfrac{a+b}{c+d}\right)^3=\dfrac{a^3-b^3}{c^3-d^3}.$
		\end{enumerate}
	}
\end{vd}
\begin{vd}
	Cho $\dfrac{a^2+b^2}{c^2+d^2}=\dfrac{ab}{cd}$ với $a$, $b$, $c$, $d \ne 0$ và $c \ne \pm d$. Chứng minh rằng $\dfrac{a}{b}=\dfrac{c}{d}$ hoặc $\dfrac{a}{b}=\dfrac{d}{c}\cdot$
	\loigiai{Ta có
		\begin{align}
		&\dfrac{a^2+b^2}{c^2+d^2}=\dfrac{ab}{cd}=\dfrac{2ab}{2cd}=\dfrac{a^2+b^2+2ab}{c^2+d^2+2cd}=\dfrac{(a+b)^2}{(c+d)^2}=\left(\dfrac{a+b}{c+d}\right)^2 &\tag{1}\label{7D1-5:1}\\
		&\dfrac{a^2+b^2}{c^2+d^2}=\dfrac{2ab}{2cd}=\dfrac{a^2+b^2-2ab}{c^2+d^2-2cd}=\dfrac{(a-b)^2}{(c-d)^2}=\left(\dfrac{a-b}{c-d}\right)^2 &\tag{2}\label{7D1-5:2}
		\end{align}
		Từ \eqref{7D1-5:1}, \eqref{7D1-5:2} suy ra $\left(\dfrac{a+b}{c+d}\right)^2=\left(\dfrac{a-b}{c-d}\right)^2.$
		\begin{description}
			\item [\emph{Trường hợp 1:}]
			\begin{align}
			&\dfrac{a+b}{c+d}=\dfrac{a-b}{c-d}=\dfrac{a+b+a-b}{c+d+c-d}=\dfrac{2a}{2c}=\left(\dfrac{a}{c}\right)^2 &\tag{3}\label{7D1-5:3}\\
			&\dfrac{a+b}{c+d}=\dfrac{a-b}{c-d}=\dfrac{a+b-a+b}{c+d-c+d}=\dfrac{2b}{2d}=\left(\dfrac{b}{d}\right)^2 &\tag{4}\label{7D1-5:4}
			\end{align}
			Từ \eqref{7D1-5:3}, \eqref{7D1-5:4} suy ra $\dfrac{a}{c}=\dfrac{b}{d}$ hay $\dfrac{a}{b}=\dfrac{c}{d}.$
			\item [\emph{Trường hợp 2:}]
			\begin{align}
			&\dfrac{a+b}{c+d}=\dfrac{b-a}{c-d}=\dfrac{2b}{2c}=\dfrac{b}{c} &\tag{5}\label{7D1-5:5}\\
			&\dfrac{a+b}{c+d}=\dfrac{b-a}{c-d}=\dfrac{2a}{2d}=\dfrac{a}{d} &\tag{6}\label{7D1-5:6}
			\end{align}
