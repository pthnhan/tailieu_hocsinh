\section{CHIA ĐƠN THỨC CHO ĐƠN THỨC}
\subsection{Tóm tắt lí thuyết}
Cho $A$ và $B$ là hai đa thức, $B\ne 0$. Ta nói đa thức $A$ chia hết cho đa thức $B$ nếu tìm được một đa thức $Q$ sao cho $A=B\cdot Q$.\\
$A$ được gọi là \textit{đa thức bị chia}, $B$ được gọi là \textit{đa thức chia}, $Q$ được gọi là \textit{đa thức thương} (gọi tắt là \textit{thương}). Kí hiệu $Q=A:B$ hoặc $Q=\dfrac{A}{B}$.
\begin{note}
	Đơn thức $A$ chia hết cho đơn thức $B$ khi mỗi biến của $B$ đều là biến của $A$ với số mũ không lớn hơn số mũ của nó trong $A$.
\end{note}


\subsection{Các dạng toán}
\begin{dang}[Thực hiện phép chia hai đơn thức]
	Quy tắc: Muốn chia đơn thức $A$ cho đơn thức $B$ (trường hợp $A$ chia hết cho $B$) ta làm như sau:
	\begin{enumerate}[\tickEX]
		\item Chia hệ số của đơn thức $A$ cho hệ số của đơn thức $B$.
		\item Chia lũy thừa của từng biến trong $A$ cho lũy thừa của từng biến đó trong $B$.
		\item Nhân các kết quả tìm được với nhau.
	\end{enumerate}
\begin{note}
	Ở lớp 7 ta đã biết: Với mọi $x\ne0$, $m,n\in\mathbb{N}$, $m\geq n$ thì:
	\begin{itemize}
		\item $x^m:x^n=x^{m-n}$ nếu $m>n$.
		\item $x^m:x^n=1$ nếu $m=n$.
	\end{itemize}
\end{note}
\end{dang}
\begin{vd}
	Làm tính chia
\begin{listEX}[3]
	\item $5^3:(-5)^2$;
	\item $\left(\dfrac{3}{4}\right)^5:\left(\dfrac{3}{4}\right)^3$;
	\item $(-12)^3:8^3$.
\end{listEX}
\loigiai{
\begin{enumerate}[a)]
	\item $5^3:(-5)^2=5^3:5^2=5^{3-2}=5^1=5$.
	\item $\left(\dfrac{3}{4}\right)^5:\left(\dfrac{3}{4}\right)^3=\left(\dfrac{3}{4}\right)^{5-3}=\left(\dfrac{3}{4}\right)^2=\dfrac{9}{16}$.
	\item $(-12)^3:8^3=\left(\dfrac{-12}{8}\right)^3=\left(\dfrac{-3}{2}\right)^3=\dfrac{-27}{8}$.
\end{enumerate}

}

\end{vd}
\begin{vd}
	Làm tính chia
	\begin{listEX}[3]
		\item $x^2yz:\dfrac{1}{2}xyz$;
		\item $2x^3y^4:x^3y$;
		\item $18x^2y^2z:8xyz$;
		\item $5a^3b:(-2a^2b)$;
		\item $27x^4y^2z:9x^4y$.
		
	\end{listEX}
\loigiai{
\begin{enumerate}[a)]
	\item Để thực hiện phép chia $x^2yz:\dfrac{1}{2}xyz$ ta thực hiện như sau:
		\begin{enumerate}[\tickEX]
			\item Chia hai hệ số: $1$ cho $\dfrac{1}{2}$ ta được: $1:\dfrac{1}{2}=2$;
			\item Lấy $x^2$ chia $x$ ta được $x^2:x=x^{2-1}=x$;
			\item Lấy $y$ chia $y$ ta được $y:y=1$;
			\item Lấy $z$ chia $z$ ta được $z:z=1$;
			\item Nhân các kết quả lại với nhau. Ta có: $2\times x\times 1\times 1=2x$.
		\end{enumerate}
		Vậy $x^2yz:\dfrac{1}{2}xyz=2x$.
	\item Tương tự câu a), để thực hiên phép chia $2x^3y^4:x^3y$ ta cũng lần lượt làm theo từng bước:
		\begin{enumerate}[\tickEX]
			\item Lấy $2$ chia $1$ ta được $2:1=2$;
			\item Lấy $x^3$ chia $x^3$ ta được $x^3:x^3=1$;
			\item Lấy $y^4$ chia $y$ ta được $y^4:y=y^{4-1}=y^3$;
			\item Nhân các kết quả lại ta được: $2x^3y^4:x^3y=2\times 1\times y^3=2y^3$.
		\end{enumerate}
	Vậy $2x^3y^4:x^3y=2y^3$.
	\item Để thực hiện phép chia $18x^2y^2z:8xyz$, ta thực hiện các bước:
	\begin{enumerate}[\tickEX]
		\item Lấy $18$ chia $8$ ta được $18:8=\dfrac{18}{8}=\dfrac{9}{4}$;
		\item Lấy $x^2$ chia $x$ ta được $x^2:x=x^{2-1}=x$;
		\item Lấy $y^2$ chia $y$ ta được $y^2:y=y$;
		\item Lấy $z$ chia $z$ ta được $z:z=1$;
		\item Nhân các kết quả lại với nhau, ta được $18x^2y^2z:8xyz=\dfrac{9}{4}\times x\times y\times 1=\dfrac{9}{4}xy$.
	\end{enumerate}
	Vậy $18x^2y^2z:8xyz=\dfrac{9}{4}xy$.
	\item Để thực hiện phép chia $5a^3b:(-2a^2b)$, ta thực hiện các bước:
	\begin{enumerate}[\tickEX]
		\item Lấy $5$ chia $-2$ ta được $5:(-2)=\dfrac{-5}{2}$;
		\item Lấy $a^3$ chia $a^2$ ta được $a^3:a^2=a^{3-2}=a^1=a$;
		\item Lấy $b$ chia $b$ ta được $b:b=1$;
		\item Nhân các kết quả lại với nhau, ta được $5a^3b:(-2a^2b)=\dfrac{-5}{2}\times a\times 1=\dfrac{-5}{2}a$.
	\end{enumerate}
	Vậy $5a^3b:(-2a^2b)=\dfrac{-5}{2}a$.
	\item Để thực hiện phép chia $27x^4y^2z:9x^4y$, ta thực hiện các bước:
	\begin{enumerate}[\tickEX]
		\item Lấy $27$ chia $9$ ta được $27:9=\dfrac{27}{9}=3$;
		\item Lấy $x^4$ chia $x^4$ ta được $x^4:x^4=1$;
		\item Lấy $y^2$ chia $y$ ta được $y^2:y=y^{2-1}=y^1=y$;
		\item Lấy $z$ chia $1=z^0$ ta được $z:1=z$;
		\item Nhân các kết quả lại với nhau ta được $27x^4y^2z:9x^4y=3\times 1\times y\times z=3yz$.
	\end{enumerate}
	Vậy $27x^4y^2z:9x^4y=3yz$.
\end{enumerate}
}
\end{vd}

\begin{vd}
	Làm tính chia
	\begin{listEX}[3]
		\item $(x+y)^2:(x+y)$;
		\item $(x-y)^5:(y-x)^4$;
		\item $(x-y+z)^4:(x-y+z)^3$;
		\item $8(x+2y)^5:(2x+4y)$;
		\item $(-ab)^{10}:(-ab)^5$.
	\end{listEX}
\loigiai{
\begin{enumerate}[a)]
	\item Ta có $(x+y)^2:(x+y)=(x+y)^{2-1}=x+y$.
	\item $(x-y)^5:(y-x)^4=(x-y)^5:[-(x-y)]^4=(x-y)^5:(x-y)^4=(x-y)^{5-4}=(x-y)^1=x-y$.
	\item $(x-y+z)^4:(x-y+z)^3=(x-y+z)^{4-3}=x-y+z$.
	\item Để thực hiện phép chia $8(x+2y)^5:(2x+4y)$, ta thực hiện các bước:
	\begin{enumerate}[\tickEX]
		\item Biến đổi $8(x+2y)^5:(2x+4y)=8(x+2y)^5:2(x+2y)$;
		\item Lấy $8$ chia $2$ ta được $8:2=4$;
		\item Lấy $(x+2y)^5$ chia cho $(x+2y)$ ta được $(x+2y)^5:(x+2y)=(x+2y)^{5-1}=(x+2y)^4$;
		\item Nhân các kết quả lại với nhau ta được $8(x+2y)^5:(2x+4y)=2(x+2y)^4$.
	\end{enumerate}
\begin{note}
	Ta trình bày lời giải gọn hơn như sau:
	\begin{eqnarray*}
	8(x+2y)^5:(2x+4y)&=&8(x+2y)^5:2(x+2y)\\
	&=&4(x+2y)^4.
	\end{eqnarray*}
\end{note}
\item $(-ab)^{10}:(-ab)^5=(-ab)^{10-5}=(-ab)^5$.
\end{enumerate}}
\end{vd}
\begin{vd}

	A=(-15x^3y^6):(-5xy^2).
	\]
	\loigiai{
	Ta có: 	$A=(-15x^3y^6):(-5xy^2)=3x^2y^4\geq 0$ với mọi $x$, $y$.}
\end{vd}



\begin{dang}[Tìm giá trị của $n\in\mathbb{N}$ để phép chia hai đơn thức là phép chia hết]
\begin{note}
Cần chú ý rằng:	``đơn thức $A$ chia hết cho đơn thức $B$ khi mỗi biến của $B$ đều là biến của $A$ với số mũ không lớn hơn số mũ của nó trong $A$''.
\end{note}	
\end{dang}
\begin{vd}
	Tìm số tự nhiên $n$ để mỗi phép chia sau là phép chia hết:
	\begin{listEX}[2]
		\item $x^4:x^n$;
		\item $x^n:x^3$;
		\item $5x^ny^3:4x^2y^2$;
		\item $x^ny^{n+1}:x^2y^5$.
	\end{listEX}
\loigiai{
\begin{enumerate}[a)]
	\item Ta có phép chia $x^4:x^n$ là phép chia hết khi và chỉ khi số tự nhiên $n$ thỏa mãn\\
	$n\leq 4\Leftrightarrow n\in\{0;1;2;3;4\}$.
	\item Phép chia $x^n:x^3$ là phép chia hết khi và chỉ khi số tự nhiên $n$ thỏa mãn\\
	$n\geq 3\Leftrightarrow n\in\{3;4;5;\ldots\}$.
	\item Phép chia $5x^ny^3:4x^2y^2$ là phép chia hết khi và chỉ khi số tự nhiên $n$ thỏa mãn\\
	$n\geq 2\Leftrightarrow n\in\{2;3;\ldots\}$.
	\item Phép chia $x^ny^{n+1}:x^2y^5$ là phép chia hết khi và chỉ khi số tự nhiên $n$ thỏa mãn\\
	$\heva{&n\geq 2\\&n+1\geq 5}\Leftrightarrow n\geq 4\Leftrightarrow n\in\{4;5;6;\ldots\}$.
\end{enumerate}}
\end{vd}
\begin{vd}
	Tìm số tự nhiên $n$ để đơn thức $A$ chia hết cho đơn thức $B$, biết
	\[
	A=4x^{n+1}y^2;\quad B=3x^3y^{n-1}.
	\]
	\loigiai{
	Đơn thức $A$ chia hết cho đơn thức $B$ khi và chỉ khi số tự nhiên $n$ thỏa mãn\\
	$\heva{&n+1\geq 3\\&2\geq n-1}\Leftrightarrow \heva{&n\geq 2\\&n\leq 3}\Leftrightarrow n\in\{3;4\}$.}
\end{vd}

\begin{vd}
	Tìm $n\in\mathbb{N}$ để
	\begin{enumerate}[a)]
		\item đơn thức $18x^{n+2}$ chia hết cho đơn thức $-6x^5$;
		\item đơn thức $2x^ny^3$ chia hết cho đơn thức $5x^2y^{n-1}$.
	\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
	\item Đơn thức $18x^{n+2}$ chia hết cho đơn thức $-6x^5$ khi và chỉ khi\\ $n+2\geq 5\Leftrightarrow n\geq 3\Leftrightarrow n\in\{3;4;5;\ldots\}$.
	\item Đơn thức $2x^ny^3$ chia hết cho đơn thức $5x^2y^{n-1}$ khi và chỉ khi\\ $\heva{&n\geq 2\\&3\geq n-1}\Leftrightarrow\heva{&n\geq 2\\&n\leq 4}\Leftrightarrow n\in\{2;3;4\}$.
\end{enumerate}


}
\end{vd}


\begin{dang}[Tính giá trị của biểu thức là kết quả của phép chia hai đơn thức tại những điểm cụ thể]
	Ta (có thể) thực hiện phép tính theo các bước như sau.
	\begin{enumerate}[\tickEX]
		\item Bước 1: Thực hiện phép chia hai đơn thức;
		\item Bước 2: Thay các giá trị của biến vào kết quả nhận được ở bước 1.
	\end{enumerate}
\end{dang}
\begin{vd}
	Tính giá trị của biểu thức $P=(-x^2y^5)^2:(-x^2y^5)$ tại $x=\dfrac{1}{2}$ và $y=-1$.
	\loigiai{
	$P=(-x^2y^5)^2:(-x^2y^5)=(x^2y^5)^2:(-x^2y^5)=x^{4}y^{10}:(-x^2y^5)=-x^2y^5$.\\
	Với $x=\dfrac{1}{2}$, $y=-1$ ta có $P=-\left(\dfrac{1}{2}\right)^2\cdot(-1)^5=\dfrac{1}{4}$.

}
\end{vd}
\begin{vd}
	Tính giá trị của biểu thức $M=15x^4y^3z^2:5xy^2z^2$ tại $x=2$; $y=-10$ và $z=2004$.
	\loigiai{
	Ta có: $M=15x^4y^3z^2:5xy^2z^2=3x^3y$.\\
	Với $x=2$; $y=-10$ và $z=2004$ ta có $M=3\cdot2^3\cdot(-10)=-240$.}
\end{vd}



\begin{dang}[Chứng minh giá trị của một biểu thức không phụ thuộc vào một biến nào đó]
	\textbf{Phương pháp:} Rút gọn biểu thức, kết quả của biểu thức là hằng số hoặc một biểu thức không còn biến đó nữa.
\end{dang}
\begin{vd}
	Chứng minh rằng giá trị của biểu thức sau không phụ thuộc vào giá trị của biến $y$ ($x\ne0$; $y\ne 0$)
	\[
	B=\dfrac{2}{3}x^2y^3:\left(-\dfrac{1}{3}xy\right)+2x(y-1)(y+1).
	\]
	\loigiai{
	Ta có:
\begin{eqnarray*}
B&=&\dfrac{2}{3}x^2y^3:\left(-\dfrac{1}{3}xy\right)+2x(y-1)(y+1)\\
&=&-2xy^2+2x(y^2-1)\\
&=&-2xy^2+2xy^2-2x\\
&=&2x.
\end{eqnarray*} không phụ thuộc vào biến $y$.
}
\end{vd}
\begin{vd}
	Chứng minh rằng giá trị của biểu thức sau không  phụ thuộc vào biến $x$ ($x\ne 0$, $y\ne 0$).
	\[
	Q=\dfrac{1}{2}x^3y^2:\dfrac{1}{4}x^2y+xy-3y(x-1)-3.
	\]
	\loigiai{
	Ta có: 
	\begin{eqnarray*}
	Q&=&\dfrac{1}{2}x^3y^2:\dfrac{1}{4}x^2y+xy-3y(x-1)-3\\
	&=&2xy+xy-3xy+3y-3\\
	&=&3y-3.
	\end{eqnarray*}
không phụ thuộc vào $x$.}
\end{vd}
