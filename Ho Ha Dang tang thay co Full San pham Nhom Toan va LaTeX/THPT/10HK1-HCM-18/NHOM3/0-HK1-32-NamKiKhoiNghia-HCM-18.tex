\begin{bt}%[HK1, Trường THPT Nam Kỳ Khởi Nghĩa, HCM, 2017-2018]%[Đoàn Nhật Thiện]%[0D2B3-2]
	Xác định parabol $(P)\colon y=ax^2+bx+c$ biết $(P)$ có đỉnh $I(-2;-9)$ và đi qua điểm $A(1;0)$.
\loigiai{
	$(P)$ có đỉnh $I(-2;-9)\Rightarrow \left\{ \begin{aligned} & \dfrac{-b}{2a}=-2 &(1)\\
	& 4a-2b+c=-9 &(2)
	\end{aligned} \right.$.\\
	$(P)$ qua $A(1;0)\Rightarrow a+b+c=0\qquad(3)$.\\
	Từ $(1)$, $(2)$, $(3)\Rightarrow \left\{ \begin{aligned}& a=1 \\
	& b=4 \\
	& c=-5 
	\end{aligned}\right.\Rightarrow (P)\colon y=x^2+4x-5$.
}	
\end{bt}
\begin{bt}%[HK1, Trường THPT Nam Kỳ Khởi Nghĩa, HCM, 2017-2018]%[Đoàn Nhật Thiện]%[0D3B2-6]
	Giải và biện luận phương trình $(m+1)^2x+3-m=(7m-5)x$.
\loigiai{
	$(m+1)^2x+3-m=(7m-5)x \Leftrightarrow (m^2-5m+6)x=m-3\qquad (1)$.
	\begin{itemize}
		\item Trường hợp $a\ne 0\Leftrightarrow \left\{\begin{aligned}& m\ne 3 \\
		& m\ne 2 
		\end{aligned}\right.$. Phương trình $(1)$ có nghiệm $x=\dfrac{m-3}{m^2-5m+6}=\dfrac{1}{m-2}$. 
		\item Trường hợp $a=0\Leftrightarrow \left[\begin{aligned}& m=3 \\
		& m=2 
		\end{aligned}\right.$.
		\begin{itemize}
			\item Với $m=3\Rightarrow b=0$, phương trình $(1)$ vô số nghiệm.
			\item Với $m=2\Rightarrow b=-1$, phương trình $(1)$ vô nghiệm.
		\end{itemize}
	\end{itemize}
Kết luận, \begin{itemize}
	\item $\heva{&m\ne 3\\ &m\ne 2}\Rightarrow S=\left\{\dfrac{1}{m-2}\right\}$,
	\item $m=3\Rightarrow S=\mathbb{R}$,
	\item $m=2\Rightarrow S=\varnothing$.
\end{itemize}
}	
\end{bt}
\begin{bt}%[HK1, Trường THPT Nam Kỳ Khởi Nghĩa, HCM, 2017-2018]%[Đoàn Nhật Thiện]%[0D3B2-5]
	Cho phương trình $x^2+2(m+3)x+m^2-5m+2=0$ ($m$ là tham số). Tìm tất cả các giá trị của $m$ để phương trình
	\begin{enumerate}
		\item Có nghiệm kép và tìm nghiệm kép đó.
		\item Có 2 nghiệm phân biệt $x_1,x_2$ thỏa mãn $(x_1+3)(x_2+3)=95$.
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Điều kiện: $\left\{\begin{aligned}& a\ne 0 \\
		& \Delta =0 
		\end{aligned}\right.\Leftrightarrow \left\{\begin{aligned}& 1\ne 0 \text{ (đúng) } \\
		& 4(m+3)^2-4(m^2-5m+2)=0 
		\end{aligned}\right. \Leftrightarrow 44m+28=0\Leftrightarrow m=-\dfrac{7}{11}$.\\
		Nghiệm kép của phương trình là $x=-\dfrac{b}{2a}=\dfrac{-2(m+3)}{2}=-\dfrac{26}{11}$.
		\item Phương trình có 2 nghiệm phân biệt $\Leftrightarrow m>-\dfrac{7}{11}$.\\
		Áp dụng định lý Vi-ét: $S=x_1+x_2=\dfrac{-b}{a}=-2(m+3)$, $P=x_1x_2=\dfrac{c}{a}=m^2-5m+2$.\\
		Theo yêu cầu bài toán \begin{eqnarray*}
		& &(x_1+3)(x_2+3)=95\\
		&\Leftrightarrow& x_1x_2+3(x_1+x_2)+9=95\\
		&\Leftrightarrow& P+3S+9=95\\
		&\Leftrightarrow& m^2-11m-102=0\\
		&\Leftrightarrow& \left[\begin{aligned}& m=17 \text{ (nhận) } \\
				& m=-6 \text{ (loại)}. \end{aligned}\right.
		\end{eqnarray*}
	\end{enumerate}
Vậy $m=17$ thỏa yêu cầu bài toán.
}	
\end{bt}
\begin{bt}%[HK1, Trường THPT Nam Kỳ Khởi Nghĩa, HCM, 2017-2018]%[Đoàn Nhật Thiện]%[0D3B2-2]
	Giải phương trình	$\left|{7x^2-5x+2}\right|-x^2-3=0$.
	\loigiai{
		\begin{eqnarray*}
			& &\left|{7x^2-5x+2}\right|-x^2-3=0\\
			&\Leftrightarrow& \left|{7x^2-5x+2}\right|-x^2-3=0\\
			&\Leftrightarrow& \left|{7x^2-5x+2}\right|=x^2+3\\
			&\Leftrightarrow& \left\{\begin{aligned}&x^2+3 \ge 0 \qquad \text{(luôn đúng)}\\&\left[\begin{aligned}&7x^2-5x+2=x^2+3\\&7x^2-5x+2=-x^2-3\end{aligned}\right.\end{aligned}\right.\\
			&\Leftrightarrow& \left[\begin{aligned}&6x^2-5x-1=0\\&8x^2-5x+5=0 \text{ (vô nghiệm)}\end{aligned}\right.\\
			&\Leftrightarrow& \left[\begin{aligned}&x=1\\&x=-\dfrac{1}{6}.\end{aligned}\right.
		\end{eqnarray*}
	Vậy $S=\left\{1,-\dfrac{1}{6}\right\}$.
	}
\end{bt}
\begin{bt}%[HK1, Trường THPT Nam Kỳ Khởi Nghĩa, HCM, 2017-2018]%[Đoàn Nhật Thiện]%[0D3K2-4]
	Giải các phương trình
		\begin{multicols}{2}
			\begin{enumerate}
			\item $\sqrt{-x^2+6x-5}+2x=8$.
			\item $(x-2)(x+3)+\sqrt{x^2+x-2}=2$.
			\end{enumerate}
		\end{multicols}	
\loigiai{
\begin{enumerate}
	\item \begin{eqnarray*}
		& & \sqrt{-x^2+6x-5}+2x=8\\
		&\Leftrightarrow& \sqrt{-x^2+6x-5}=8-2x\\
		&\Leftrightarrow& \left\{\begin{aligned}&8-2x\ge 0\\&-x^2+6x-5=(8-2x)^2\end{aligned}\right.\\
		&\Leftrightarrow& \left\{\begin{aligned}&x\le 4\\&5x^2-38x+69=0\end{aligned}\right.\\
		&\Leftrightarrow& \left\{\begin{aligned}&x\le 4\\&\left[\begin{aligned}&x=\dfrac{23}{5}\text{ (loại)}\\&x=3\text{ (nhận)}.\end{aligned}\right.\end{aligned}\right.
	\end{eqnarray*}
Vậy $S=\{3\}$.
	\item 
	$(x-2)(x+3)+\sqrt{x^2+x-2}=2\Leftrightarrow x^2+x-6+\sqrt{x^2+x-2}=2\qquad (1)$.\\
	Đặt $t=\sqrt{x^2+x-2}$, $(t\ge  0)\Rightarrow t^2=x^2+x-2\Rightarrow x^2+x=t^2+2$.\\
	Phương trình $(1)$ trở thành $t^2+t-6=0\Leftrightarrow \left[\begin{aligned}&t=-3\text{ (loại)}\\&t=2\text{ (nhận)}.\end{aligned}\right.$\\
	Với $t=2\Leftrightarrow \sqrt{x^2+x-2}=2\Leftrightarrow x^2+x-6=0\Leftrightarrow \left[\begin{aligned}&x=-3\\&x=2\end{aligned}\right.$.
\end{enumerate}
Vậy $S=\{2,-3\}$.
}
\end{bt}		
\begin{bt}%[HK1, Trường THPT Nam Kỳ Khởi Nghĩa, HCM, 2017-2018]%[Đoàn Nhật Thiện]%[0D4K1-2]
	Chứng minh: $(a+b)(b+c)(c+a)\ge  8abc$, $\forall a$, $b$, $c\ge  0$.
	\loigiai{
		Áp dụng bất đẳng thức Cau-chy cho hai số dương, ta có:
		\begin{itemize}
			\item $a+b\ge  2\sqrt{ab}$; 
			\item $b+c\ge  2\sqrt{bc}$; 
			\item $c+a\ge  2\sqrt{ca}$.
		\end{itemize}
	Nhân vế theo vế, ta được
		\begin{eqnarray*}
			(a+b)(b+c)(c+a)&\ge&  2\sqrt{ab}\cdot 2\sqrt{bc}\cdot 2\sqrt{ca}\\
			&=&8\sqrt{ab\cdot bc\cdot ca}\\
			&=&8abc \text{ (điều phải chứng minh) }.
		\end{eqnarray*}
		Dấu ``$=$'' xảy ra khi $a=b=c$.
	}
\end{bt}
\begin{bt}%[HK1, Trường THPT Nam Kỳ Khởi Nghĩa, HCM, 2017-2018]%[Đoàn Nhật Thiện]%[0H2B2-3][0H1B4-3][0H2B2-2]
	Trong hệ trục tọa độ $Oxy$, cho ba điểm $A(-2;-3)$; $B(2;1)$; $C(-1;4)$.
	\begin{enumerate}
		\item Chứng minh $A$, $B$, $C$ tạo thành tam giác vuông. Tính chu vi và diện tích của $\Delta ABC$.
		\item Tìm tọa độ điểm $E$ sao cho $\vec{EA}-2\vec{AB}+3\vec{EC}=\vec{0}$.
		\item Tìm tọa độ điểm $F$ thuộc trục tung sao cho $\Delta AFC$ cân tại $F$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $\vec{AB}=(4;4)\Rightarrow AB=4\sqrt{2}$, $\vec{AC}=(1;7)\Rightarrow AC=5\sqrt{2}$, $\vec{BC}=(-3;3)\Rightarrow BC=3\sqrt{2}$.\\ 
			Ta có $\vec{AB}\cdot \vec{BC}=0$ nên $\Delta ABC$ vuông tại B.\\
			Chu vi tam giác $ABC$ là $P_{\Delta ABC}=AB+AC+BC=12\sqrt{2}$.\\
			Diện tích tam giác $ABC$ là $S_{\Delta ABC}=\dfrac{1}{2}AB\cdot BC=12$ (đơn vị diện tích).
			\item Gọi $E(x;y)$. Ta có 
			\begin{eqnarray*}
				& &\vec{EA}-2\vec{AB}+3\vec{EC}=\vec{0}\\
				&\Leftrightarrow& \left\{ \begin{aligned} &-2-x-2\cdot 4+3(-1-x)=0\\ &-3-y-2\cdot 4+3(4-y)=0\end{aligned} \right.	\\			
				&\Leftrightarrow& \left\{ \begin{aligned} &-13-4x=0\\ &1-4y=0 \end{aligned} \right.\\
				&\Leftrightarrow& \left\{\begin{aligned}& x=-\dfrac{13}{4} \\
					& y=\dfrac{1}{4}. 
				\end{aligned}\right.
			\end{eqnarray*}
			Vậy $E\left(-\dfrac{13}{4};\dfrac{1}{4}\right)$.
			\item Gọi $F(0;y)\in Oy$.\\
			Ta có	$\Delta AFC$ cân tại $F$ nên
			\begin{eqnarray*}
				& &FA=FC\\
				&\Leftrightarrow& (-2)^2+(-3-y)^2=(-1)^2+(4-y)^2\\
				&\Leftrightarrow& 4+9+6y+y^2=1+16-8y+y^2\\
				&\Leftrightarrow& 14y=4\\
				&\Leftrightarrow& y=\dfrac{2}{7}.
			\end{eqnarray*}
			Vậy $F\left(0;\dfrac{2}{7}\right)$.
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[HK1, Trường THPT Nam Kỳ Khởi Nghĩa, HCM, 2017-2018]%[Đoàn Nhật Thiện]%[0H2K2-1]
	Cho tam giác $ABC$ có $AC=2$, $BC=4$ và góc $\widehat{C}=120^\circ$. Tính $\vec{CB}\cdot \vec{CA}$ từ đó suy ra độ dài cạnh $AB$.
	\loigiai{
		Ta có $\vec{CB}\cdot\vec{CA}=4\cdot 2\cdot \cos 120^\circ=-4$.\\
		Mà \begin{eqnarray*}
			& &\vec{AB}=\vec{CB}-\vec{CA}\\
			&\Rightarrow& AB^2=(\vec{CB}-\vec{CA})^2\\
			&\Leftrightarrow& AB^2=\vec{CB}^2-2\vec{CB}\cdot \vec{CA}+\vec{CA}^2\\
			&\Leftrightarrow& AB^2=BC^2-2\vec{CB}\cdot \vec{CA}+AC^2\\
			&\Leftrightarrow& AB^2=4^2-2(-4)+2^2\\
			&\Leftrightarrow& AB^2=16+8+4\\
			&\Leftrightarrow& AB^2=28\\
			&\Rightarrow& AB=2\sqrt{7}.
		\end{eqnarray*}
}
\end{bt}
