
\section{PHƯƠNG TRÌNH BẬC HAI MỘT ẨN VÀ CÔNG THỨC NGHIỆM}

\subsection{Tóm tắt lí thuyết}
\subsubsection{Định nghĩa}
Phương trình bậc hai một ẩn là phương trình có dạng \[ax^2+bx+c=0,
\tag{1}\]
trong đó $x$ là ẩn; $a,b,c$ là các số cho trước gọi là hệ số và $a\ne 0$.\\
\noindent{\bf Nhận xét.} Phương trình $(1)$ tương đương với phương trình \[a\left[\left(x+\dfrac{b}{2a}\right)^2-\dfrac{b^2-4ac}{4a^2}\right]=0.\tag{2}\]
\subsubsection{Giải phương trình bậc hai}
Để tìm nghiệm của phương trình ($1$) ta dựa vào biệt số $\Delta=b^2-4ac$.
\begin{itemize}
\item Nếu $\Delta<0$ thì phương trình ($1$) vô nghiệm.
\item Nếu $\Delta=0$ thì phương trình ($1$) có nghiệm kép $x_1=x_2=-\dfrac{b}{2a}\cdot$
\item Nếu $\Delta>0$ thì phương trình ($1$) có hai nghiệm phân biệt $$x_1=\dfrac{-b+\sqrt{\Delta}}{2a};\; x_2=\dfrac{-b-\sqrt{\Delta}}{2a}\cdot$$
\end{itemize}
\textit{Đặc biệt}: Nếu $b=2b'$ thì ta có $\Delta'=\left(b'\right)^2-ac$. Khi đó
\begin{itemize}
\item Nếu $\Delta'<0$ thì phương trình ($1$) vô nghiệm.
\item Nếu $\Delta'=0$ thì phương trình ($1$) có nghiệm kép $x_1=x_2=-\dfrac{b'}{a}\cdot$
\item Nếu $\Delta'>0$ thì phương trình ($1$) có hai nghiệm phân biệt $$x_1=\dfrac{-b'+\sqrt{\Delta'}}{a};\; x_2=\dfrac{-b'-\sqrt{\Delta'}}{a}\cdot$$
\end{itemize}
\noindent{\bf Nhận xét.} Nếu $ac<0$ thì $\Delta>0$, do đó phương trình ($1$) có hai nghiệm phân biệt. Hơn nữa, hai nghiệm đó trái dấu.

\subsection{Áp dụng giải toán}
\begin{dang}{Giải phương trình bậc hai}
	Biến đổi phương trình về dạng $(1)$, xác định các hệ số $a$, $b$, $c$ rồi lập biệt số $\Delta=b^2-4ac$ (hoặc $\Delta'=\left(b'\right)^2-ac$). Từ đó tìm được nghiệm (nếu có) của phương trình.
\end{dang}

\noindent{\bf Chú ý.}  Trong một vài trường hợp cụ thể (ví dụ như hệ số $b=0$) thì ta thường sử dụng cách phân tích thành nhân tử hoặc sử dụng công thức $(2)$ để giải. Mời các bạn theo dõi ví dụ \ref{9D42VD1} và ví dụ \ref{9D42VD2} dưới đây.
\begin{vd}\label{9D42VD1}
	Giải các phương trình sau
	\begin{enumEX}{3}
		\item $4x^2-9=0$.
		\item $-2x^2+50=0$.
		\item $3x^2+11=0$.
	\end{enumEX}
\loigiai{
\begin{enumerate}
\item $4x^2-9=0\Leftrightarrow x^2=\left(\dfrac{3}{2}\right)^2\Leftrightarrow x=\pm \dfrac{3}{2}$.\\
Phương trình có tập nghiệm là $S=\left\{-\dfrac{3}{2};\dfrac{3}{2}\right\}$.
\item $-2x^2+50=0\Leftrightarrow x^2-25\Leftrightarrow (x-5)(x+5)=0\Leftrightarrow x=\pm 5$.\\
Phương trình có tập nghiệm là $S=\left\{-5;5\right\}$.
\item Vì $3x^2+11>0$ với mọi giá trị của $x$ nên phương trình đã cho vô nghiệm.
\end{enumerate}
}
\end{vd}

\begin{vd}\label{9D42VD2}
	Giải các phương trình sau
	\begin{enumEX}{2}
		\item $x(x-2)+4x-8=0$.
		\item $x^2-4x+2=0$.
		\item $2x^2+5x=1$.
	\end{enumEX}
\loigiai{
\begin{enumerate}
\item Ta có 
\begin{align*}
x(x-2)+4x-8=0&\Leftrightarrow x^2-2x+4x-8=0\Leftrightarrow (x^2+2x+1)-9=0\\ 
&\Leftrightarrow (x+1)^2-3^2=0\Leftrightarrow (x-2)(x+4)=0\Leftrightarrow \left[\begin{aligned}& x=2,\\ & x=-4.\end{aligned}\right.
\end{align*}
Phương trình có tập nghiệm là $S=\left\{-4;2\right\}$.
\item Ta có 
\begin{align*}
x^2-4x+2=0&\Leftrightarrow x^2-4x+4=2\\ 
&\Leftrightarrow (x-2)^2=4\Leftrightarrow \left[\begin{aligned}& x-2=\sqrt{2}\\ & x-2=-\sqrt{2}\end{aligned}\right.\Leftrightarrow \left[\begin{aligned}& x=2+\sqrt{2},\\ & x=2-\sqrt{2}.\end{aligned}\right.
\end{align*}
Phương trình có tập nghiệm là $S=\left\{2+\sqrt{2};2-\sqrt{2}\right\}$.
\item Ta có 
\begin{align*}
2x^2+5x=1&\Leftrightarrow x^2+\dfrac{5}{2}x=\dfrac{1}{2}\Leftrightarrow x^2+2\cdot x\cdot\dfrac{5}{4}+\dfrac{25}{16}=\dfrac{1}{2}+\dfrac{25}{16}\\ 
&\Leftrightarrow \left(x+\dfrac{5}{4}\right)^2=\dfrac{33}{16}\Leftrightarrow \left[\begin{aligned}& x+\dfrac{5}{4}=\dfrac{\sqrt{33}}{4}\\ & x+\dfrac{5}{4}=-\dfrac{\sqrt{33}}{4}\end{aligned}\right.\Leftrightarrow \left[\begin{aligned}& x=-\dfrac{5}{4}+\dfrac{\sqrt{33}}{4},\\ & x=\dfrac{5}{4}-\dfrac{\sqrt{33}}{4}.\end{aligned}\right.
\end{align*}
Phương trình có tập nghiệm là $S=\left\{-\dfrac{5-\sqrt{33}}{4};-\dfrac{5+\sqrt{33}}{4}\right\}$.
\end{enumerate}
}
\end{vd}

\begin{vd}
	Giải các phương trình sau
	\begin{enumEX}{2}
		\item $\left(2-\sqrt{5}\right)x^2-x+\left(\sqrt{5}-1\right)=0$.
		\item $\sqrt{2}x^2+4\sqrt{3}x-2\sqrt{2}=0$.
	\end{enumEX}
\loigiai{
\begin{enumerate}
\item Phương trình có dạng $ax^2+bx+c=0$ với $a=2-\sqrt{5}$; $b=-1$; $c=\sqrt{5}-1$.\\
Ta có $\Delta=b^2-4ac=(-1)^2-4\left(2-\sqrt{5}\right)\left(\sqrt{5}-1\right)=29-12\sqrt{5}=\left(2\sqrt{5}-3\right)^2>0$.\\
Phương trình có hai nghiệm phân biệt
\begin{align*}
&x_1=\dfrac{-b+\sqrt{\Delta}}{2a}=\dfrac{1+2\sqrt{5}-3}{2\left(2-\sqrt{5}\right)}=-3-\sqrt{5};\\ 
&x_2=\dfrac{-b-\sqrt{\Delta}}{2a}=\dfrac{1-2\sqrt{5}+3}{2\left(2-\sqrt{5}\right)}=1.
\end{align*}
\item Phương trình có dạng $ax^2+bx+c=0$ với $a=\sqrt{2}$; $b=4\sqrt{3}$ ($b'=2\sqrt{3}$); $c=-2\sqrt{2}$.\\
Ta có $\Delta'=\left(b'\right)^2-ac=\left(2\sqrt{3}\right)^2-\sqrt{2}\cdot\left(-2\sqrt{2}\right)=16>0$.\\
Phương trình có hai nghiệm phân biệt
\begin{align*}
&x_1=\dfrac{-b'+\sqrt{\Delta'}}{a}=\dfrac{-2\sqrt{3}+4}{\sqrt{2}}=-\sqrt{6}+2\sqrt{2};\\ 
&x_2=\dfrac{-b'-\sqrt{\Delta'}}{a}=\dfrac{-2\sqrt{3}+4}{\sqrt{2}}=-\sqrt{6}-2\sqrt{2}.
\end{align*}
\end{enumerate}
}
\end{vd}

\begin{vd}
	Giải các phương trình sau
	\begin{enumEX}{2}
		\item $\dfrac{x^2}{9}-\dfrac{8x}{3}+16=0$.
		\item $0,4x^2-7x+30=0$.
	\end{enumEX}
\loigiai{
\begin{enumerate}
\item Phương trình có dạng $ax^2+bx+c=0$ với $a=\dfrac{1}{9}$; $b=-\dfrac{8}{3}$ ($b'=-\dfrac{4}{3}$); $c=16$.\\
Ta có $\Delta'=b'^2-ac=\left(-\dfrac{4}{3}\right)^2-\dfrac{1}{9}\cdot 16=0$ nên phương trình có nghiệm kép
\[x_1=x_2=-\dfrac{b'}{a}=12.\]
\item Phương trình có dạng $ax^2+bx+c=0$ với $a=0,4$; $b=-7$; $c=30$.\\
Ta có $\Delta=b^2-4ac=(-7)^2-4\cdot 0,4\cdot 30=1>0$ nên phương trình có hai nghiệm phân biệt
\begin{align*}
&x_1=\dfrac{-b+\sqrt{\Delta}}{2a}=\dfrac{7+1}{2\cdot 0,4}=10;\\ 
&x_2=\dfrac{-b-\sqrt{\Delta}}{2a}=\dfrac{7-1}{2\cdot 0,4}=\dfrac{15}{2}.
\end{align*}
\end{enumerate}
}
\end{vd}

\begin{dang}{Giải và biện luận phương trình dạng $ax^2+bx+c=0$}
Khi gặp bài toán giải và biện luận phương trình dạng $ax^2+bx+c=0$, trong đó $a,b,c\in\mathbb{R}$ thì ta thực hiện các bước giải như sau
\begin{itemize}
\item Xét trường hợp $a=0$. Lúc đó phương trình đã cho trở thành $bx+c=0$. Đây là phương trình bậc nhất. Tiếp tục giải và biện luận phương trình bậc nhất này.
\item Xét trường hợp $a\ne 0$. Lúc đó ta tính biệt số $\Delta=b^2-4ac$ (hoặc $\Delta'=\left(b'\right)^2-ac$). Dựa vào các trường hợp $\Delta>0$, $\Delta=0$, $\Delta<0$ ($\Delta'>0$, $\Delta'=0$, $\Delta'<0$) ta sẽ kết luận được nghiệm của phương trình đã cho.
\item Nêu kết luận về các trường hợp đã giải quyết ở trên. Kết luận sẽ giúp ta phát hiện sự thiếu sót trong việc giải và biện luận một phương trình.
\end{itemize}
\end{dang}
\begin{vd}
	Giải và biện luận phương trình ($x$ là ẩn, $m$ là tham số) sau
\[x^2-2(m+3)x+(m^2-5)=0.\tag{1}\]
\begin{enumerate}
\item Giải phương trình với $m=2$.
\item Với giá trị nào của $m$ thì phương trình có hai nghiệm, nghiệm kép, vô nghiệm?
\item Tính hiệu của nghiệm lớn và nghiệm nhỏ trong trường hợp phương trình có hai nghiệm.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Với $m=2$ ta có phương trình \[x^2-10x-1=0.\]
Ta có $\Delta'=25+1=26>0$. Phương trình có hai nghiệm phân biệt \[x_1=5+\sqrt{26}\;\text{và}\;x_2=5-\sqrt{26}.\]
\item Phương trình $(1)$ có \[\Delta'=(m+3)^2-(m^2-5)=6m+14.\]
\begin{itemize}
\item Phương trình $(1)$ có hai nghiệm $\Leftrightarrow 6m+14>0\Leftrightarrow m>-\dfrac{7}{3}.$
\item Phương trình $(1)$ có nghiệm kép $\Leftrightarrow 6m+14=0\Leftrightarrow m=-\dfrac{7}{3}.$
\item Phương trình $(1)$ vô nghiệm $\Leftrightarrow 6m+14<0\Leftrightarrow m<-\dfrac{7}{3}.$
\end{itemize}
\item Khi $m>-\dfrac{7}{3}$, phương trình $(1)$ có hai nghiệm phân biệt \[x_1=(m+3)+\sqrt{6m+14};\; x_2=(m+3)-\sqrt{6m+14}.\]
Rõ ràng $x_1>x_2$. Khi ấy $x_1-x_2=2\sqrt{6m+14}$.
\end{enumerate}
}
\end{vd}
\begin{vd}\label{9D42VD6}
	Giải và biện luận phương trình ($x$ là ẩn, $m$ là tham số) sau
\[mx^2-2(m-1)x+m+1=0\tag{1}\]
\loigiai{
\begin{enumerate}
\item Với $m=0$. Phương trình $(1)$ trở thành: $2x+1=0\Leftrightarrow x=-\dfrac{1}{2}.$
\item Với $m\ne 0$. Ta có  $\Delta'=(m-1)^2-m(m+1)=-3m+1$.
		\begin{itemize}
			\item Nếu $\Delta'<0 \Leftrightarrow -3m+1<0 \Leftrightarrow m>\dfrac{1}{3}$ thì phương trình $(1)$ vô nghiệm.
			\item Nếu $\Delta'=0 \Leftrightarrow m=\dfrac{1}{3}$ thì phương trình $(1)$ có nghiệm kép $x_1=x_2=\dfrac{m-1}{m}=-2$.
			\item Nếu $\Delta'>0  \Leftrightarrow m<\dfrac{1}{3}$ thì phương trình $(1)$ có hai nghiệm phân biệt
		\[x_1=\dfrac{m-1+\sqrt{-3m+1}}{m};\; x_2=\dfrac{m-1-\sqrt{-3m+1}}{m}.\]
		\end{itemize}
\item[] \textit{Kết luận}:
\begin{itemize}
			\item $m=0$: phương trình $(1)$ có một nghiệm $x=-\dfrac{1}{2}$.
			\item $m<\dfrac{1}{3}$ và $m\ne 0$: phương trình $(1)$ có hai nghiệm phân biệt $x_{1,2}=\dfrac{m-1\pm\sqrt{-3m+1}}{m}.$
			\item $m=\dfrac{1}{3}$: phương trình $(1)$ có nghiệm kép $x=-2$.
			\item $m>\dfrac{1}{3}$: phương trình $(1)$ vô nghiệm.
		\end{itemize}
\end{enumerate}
}
\end{vd}
\noindent{\bf Chú ý.} Đôi khi ta gặp một bài toán giải và biện luận phương trình có mẫu số mà phương trình này đưa về được dạng phương trình bậc hai. Đây là bài toán khó hơn ví dụ \ref{9D42VD6}. Lúc ấy, nghiệm của phương trình bậc hai thu được không hẳn là nghiệm của phương trình đã cho mà ta phải so sánh chúng với điều kiện tồn tại của phương trình ban đầu. Ta xét ví dụ sau đây.
\begin{vd}
	Giải và biện luận phương trình ($m$ là tham số) sau
\[\dfrac{m}{x-1}+\dfrac{1}{x-m}=2.\tag{1}\]
\loigiai{
Điều kiện: $x\ne 1$ và $x\ne m$. Khi đó phương trình $(1)$ trở thành
\begin{align*}
n(x-m)+x-1=2(x-1)(x-m)&\Leftrightarrow (mx-m^2)+x-1=2(x^2-x-mx+m)\\ 
&\Leftrightarrow 2x^2-3(m+1)x+(m+1)^2=0.\tag{2}
\end{align*}
Phương trình $(2)$ có biệt số $\Delta=9(m+1)^2-8(m+1)^2=(m+1)^2\ge 0, \forall m.$ Khi đó
\begin{itemize}
\item Nếu $m=-1$ thì phương trình $(2)$ có nghiệm kép $x_0=\dfrac{3(m+1)}{4}=0$. Ta thấy nghiệm này thỏa mãn nên cũng là nghiệm của phương trình $(1)$.
\item Nếu $m\ne -1$ thì phương trình $(2)$ có  hai nghiệm phân biệt
\begin{align*}
&x_1=\dfrac{3(m+1)+(m+1)}{4}=m+1;\\ 
&x_2=\dfrac{3(m+1)-(m+1)}{4}=\dfrac{m+1}{2}.
\end{align*}
+ $x_1$ là nghiệm của $(1)$ khi $\begin{cases}m+1\ne 1\\ m+1\ne m\end{cases}\Leftrightarrow m\ne 0.$\\
+ $x_2$ là nghiệm của $(1)$ khi $\begin{cases}\dfrac{m+1}{2}\ne 1\\ \dfrac{m+1}{2}\ne m\end{cases}\Leftrightarrow m\ne 1.$
\end{itemize}
\textit{Kết luận}:
\begin{itemize}
\item $m=-1$: $(1)$ có một nghiệm $x_0=0$.
\item $m=0$: $(1)$ có một nghiệm $x_2=\dfrac{m+1}{2}=\dfrac{1}{2}$ (loại nghiệm $x_1$).
\item $m=1$: $(1)$ có một nghiệm $x_1=m+1=2$ (loại nghiệm $x_2$).
\item $m\ne 0$ và $m\ne \pm 1$: $(1)$ có hai nghiệm phân biệt $x_1=m+1$ và $x_2=\dfrac{m+1}{2}$.
\end{itemize}
}
\end{vd}

\subsection{Luyện tập}
\begin{bt}
Giải các phương trình
	\begin{enumEX}{2}  
		\item $5x^2-7x-6=0$.
		\item $x^2+2x-1=0$.
		\item $2x^2+5x+1=0$.
		\item $2x^2-\left(2\sqrt{6}+3\right)x+3\sqrt{6}=0$.
	\end{enumEX}
\loigiai{
\begin{enumerate}
\item Ta có $\Delta = (-7)^2-4\cdot5\cdot(-6)=169=13^2>0$.\\
	Do đó phương trình có hai nghiệm phân biệt 
	$x_1=\dfrac{7+13}{2\cdot 5}=2,$
	$x_2=\dfrac{7-13}{2\cdot 5}=-\dfrac{3}{5}.$\\
	Vậy phương trình có tập nghiệm $S=\left\{2;-\dfrac{3}{5}\right\}$.
\item Ta có $\Delta'=1+1=2$ nên phương trình có hai nghiệm phân biệt $x_1=-1+\sqrt{2}$; $x_2=-1-\sqrt{2}$.\\
	Vậy phương trình có tập nghiệm $S=\left\{-1+\sqrt{2};-1-\sqrt{2}\right\}$.
\item Ta có $\Delta=5^2-4\cdot 2\cdot 1=17$ nên phương trình có  hai nghiệm phân biệt
\[x_1=\dfrac{-5+\sqrt{17}}{2\cdot 2}=\dfrac{-5+\sqrt{17}}{4}; x_2=\dfrac{-5-\sqrt{17}}{2\cdot 2}=\dfrac{-5-\sqrt{17}}{4}.\]
	Vậy phương trình có tập nghiệm $S=\left\{\dfrac{-5-\sqrt{17}}{4};\dfrac{-5+\sqrt{17}}{4}\right\}$.
\item Ta có $\Delta=\left[-\left(2\sqrt{6}+3\right)\right]^2-4\cdot 2\cdot 3\sqrt{6}=33-12\sqrt{6}=\left(2\sqrt{6}-3\right)^2>0$ nên phương trình có hai nghiệm phân biệt
\begin{align*}
&x_1=\dfrac{\left(2\sqrt{6}+3\right)+\left(2\sqrt{6}-3\right)}{2\cdot 2}=\sqrt{6};\\ 
&x_2=\dfrac{\left(2\sqrt{6}+3\right)-\left(2\sqrt{6}-3\right)}{2\cdot 2}=\dfrac{3}{2}.
\end{align*}
Vậy phương trình có tập nghiệm $S=\left\{\sqrt{6};\dfrac{3}{2}\right\}$.
\end{enumerate}
}
\end{bt}

\begin{bt}
Cho phương trình ($m$ là tham số)
\[mx^2-(2m+1)x+(m+1)=0.\tag{1}\]
\begin{enumerate}
\item Giải phương trình $(1)$ với $m=-\dfrac{3}{5}$.
\item Chứng minh rằng phương trình $(1)$ luôn luôn có nghiệm với mọi giá trị của $m$.
\item Tìm giá trị của $m$ để phương trình $(1)$ có một nghiệm lớn hơn $2$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Với $m=-\dfrac{3}{5}$ thì phương trình $(1)$ trở thành
\[-\dfrac{3}{5}x^2+\dfrac{1}{5}x+\dfrac{2}{5}=0\Leftrightarrow 3x^2-x-2=0.\]
Ta có $\Delta=1-4\cdot 3\cdot (-2)=25>0$. Phương trình có hai nghiệm
\[x_1=\dfrac{1-5}{6}=-\dfrac{2}{3};
x_2=\dfrac{1+5}{6}=1.\]
\item Ta có
\begin{itemize}
\item Nếu $m=0$ thì phương trình $(1)$ trở thành $-x+1=0$. Phương trình này có nghiệm duy nhất $x=1$.
\item Nếu $m\ne 0$ thì $(1)$ là phương trình bậc hai có
\[\Delta=\left[-(2m+1)\right]^2-4m(m+1)=1>0.\]
Do đó phương trình có hai nghiệm phân biệt.
\end{itemize}
Tóm lại, với mọi giá trị của $m$ thì phương trình $(1)$ luôn có nghiệm.
\item Nếu $m\ne 0$ thì phương trình $(1)$ có hai nghiệm phân biệt là
\begin{align*}
&x_1=\dfrac{(2m+1)-1}{2m}=1;\\ 
&x_2=\dfrac{(2m+1)+1}{2m}=\dfrac{m+1}{m}.
\end{align*}
Vì nghiệm $x_1=1<2$ nên ta phải xét nghiệm $x_2>2$.
\[\dfrac{m+1}{m}>2\Leftrightarrow \dfrac{m+1}{m}-2>0\Leftrightarrow \dfrac{1-m}{m}>0\Leftrightarrow 0<m<1.\]
Vậy khi $0<m<1$ thì phương trình $(1)$ có một nghiệm lớn hơn $2$.
\end{enumerate}
}
\end{bt}

\begin{bt}
Giải và biện luận phương trình ($x$ là ẩn, $m$ là tham số) sau
\[(m-3)x^2-2mx+m-6=0.\tag{1}\]
\loigiai{
\begin{enumerate}
\item Với $m=3$. Phương trình $(1)$ trở thành: $-6x-3=0\Leftrightarrow x=-\dfrac{1}{2}.$
\item Với $m\ne 3$. Ta có  $\Delta'=m^2-(m-3)(m-6)=9m-18$.
		\begin{itemize}
			\item Nếu $\Delta'<0 \Leftrightarrow 9m-18<0 \Leftrightarrow m<2$ thì phương trình $(1)$ vô nghiệm.
			\item Nếu $\Delta'=0 \Leftrightarrow m=2$ thì phương trình $(1)$ có nghiệm kép $x_1=x_2=2$.
			\item Nếu $\Delta'>0  \Leftrightarrow m>2$ thì phương trình $(1)$ có hai nghiệm phân biệt
		\[x_1=\dfrac{m+3\sqrt{m-2}}{m-3};\; x_2=\dfrac{m-2\sqrt{m-2}}{m-3}.\]
		\end{itemize}
\item[] \textit{Kết luận}:
\begin{itemize}
			\item $m=3$: phương trình $(1)$ có một nghiệm $x=-\dfrac{1}{2}$.
			\item $m>2$ và $m\ne 3$: phương trình $(1)$ có hai nghiệm phân biệt $x_{1,2}=\dfrac{m\pm3\sqrt{m-2}}{m-3}.$
			\item $m=2$: phương trình $(1)$ có nghiệm kép $x=2$.
			\item $m<2$: phương trình $(1)$ vô nghiệm.
		\end{itemize}
\end{enumerate}
}
\end{bt}

\begin{bt}
Cho phương trình ($x$ là ẩn, $m$ là tham số) sau
\[(m^2-4)x^2+2(m+2)x+1=0\tag{1}\]
\begin{enumerate}
\item Tìm $m$ để phương trình $(1)$ có nghiệm.
\item Tìm $m$ để phương trình $(1)$ có nghiệm duy nhất.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có
	\begin{itemize}
	\item Khi $m^2-4=0\Leftrightarrow m=\pm 2$, thử trực tiếp ta thấy phương trình chỉ có nghiệm khi $m=2$.
	\item Khi $m^2-4\ne 0\Leftrightarrow m\ne\pm 2$.\hfill$(*)$\\
Ta có $\Delta'=4m+8$.\\
Để phương trình $(1)$ có nghiệm thì $\Delta'\ge 0\Leftrightarrow 4m+8\ge 0\Leftrightarrow m\ge -2$.\hfill$(**)$\\
Từ $(*)$ và $(**)$ suy ra $m>-2$ và $m\ne 2$.
	\end{itemize}
Vậy phương trình $(1)$ có nghiệm khi và chỉ khi $m>-2$.
\item Phương trình $(1)$ có nghiệm duy nhất trong hai trường hợp sau
\begin{itemize}
\item Trường hợp 1: $\begin{cases}m^2-4=0\\ 2(m+2)\ne 0\end{cases}\Leftrightarrow m=2$.
\item Trường hợp 2: $\begin{cases}m^2-4\ne 0\\ 4m+8=0\end{cases}\Leftrightarrow m\in\varnothing$.
\end{itemize}
Vậy với $m=2$ thì phương trình $(1)$ có nghiệm duy nhất.
\end{enumerate}
}
\end{bt}

\begin{bt}
	Cho $a^2+b^2>0$. Chứng minh rằng phương trình sau đây luôn có nghiệm
\[\dfrac{a^2}{x}+\dfrac{b^2}{x-1}=1.\tag{1}\]
\loigiai{
Điều kiện: $x\ne 0$ và $x\ne 1$. Khi đó phương trình $(1)$ trở thành
\[a^2(x-1)+b^2x=x(x-1)\Leftrightarrow x^2-(a^2+b^2+1)x+a^2=0.\tag{2}\]
Phương trình $(2)$ có $\Delta=(a^2+b^2+1)^2-4a^2=\left[(a-1)^2+b^2\right]\left[(a+1)^2+b^2\right]\ge 0,\forall a,b$. 
\begin{itemize}
\item $\Delta=0\Leftrightarrow \left[\begin{aligned}& (a-1)^2+b^2=0\\ & (a+1)^2+b^2=0\end{aligned}\right.\Leftrightarrow \left[\begin{aligned}& a=1,b=0;\\ & a=-1,b=0.\end{aligned}\right.$\\
+ Với $a=1$ và $b=0$ thì $(1)\Leftrightarrow \dfrac{1}{x}=1\Leftrightarrow x=1$ nên $(1)$ có nghiệm.\\
+ Với $a=-1$ và $b=0$ thì $(1)\Leftrightarrow \dfrac{1}{x}=1\Leftrightarrow x=1$ nên $(1)$ có nghiệm.
\item $\Delta>0$ khi $a\ne\pm 1$ hoặc $b\ne 0$, lúc đó phương trình $(2)$ có hai nghiệm phân biệt $x_1$, $x_2$. Ta đi chứng minh $x_1$ hoặc $x_2$ thỏa mãn $x\ne 0$ và $x\ne 1$.\\
Đặt $f(x)=x^2-(a^2+b^2+1)x+a^2$. Khi đó ta có $f(0)=a^2$; $f(1)=b^2$.\\
Do $a^2+b^2>0$ nên $f(0)$ và $f(1)$ không đồng thời bằng $0$, nên suy ra tồn tại $x_1$ hay $x_2$ khác $0$ và khác $-1$.
\end{itemize}
Vậy phương trình $(1)$ luôn có nghiệm.
}
\end{bt}

\begin{bt} 
Giải và biện luận theo tham số $m$ phương trình sau \[\dfrac{m-1}{mx-1}+\dfrac{2}{x^2-1}=\dfrac{m+5}{(1-mx)(x^2-1)}.\tag{1}\]
\loigiai{		
\begin{enumerate}
\item Nếu $m=0$ thì phương trình $(1)$ có dạng \[1+\dfrac{2}{x^2-1}=\dfrac{5}{x^2-1}\Leftrightarrow \dfrac{3}{x^2-1}=1\Leftrightarrow \begin{cases}x\ne\pm 1\\ x^2-1=3\end{cases}\Leftrightarrow x=\pm 2.\]
\item Nếu $m\ne 0$. Điều kiện $x\ne\pm 1$ và $x\ne \dfrac{1}{m}$.
\begin{align*}
(1)&\Leftrightarrow (m-1)(x^2-1)+2(mx-1)=-m-5\\ 
&\Leftrightarrow (m-1)x^2-m+1+2mx-2+m+5=0\\ 
&\Leftrightarrow (m-1)x^2+2mx+4=0.\tag{2}
\end{align*}
\begin{itemize}
\item Nếu $m=1$, phương trình $(2)$ có dạng $2x+4=0\Leftrightarrow x=-2$ (thỏa mãn điều kiện).
\item Nếu $m\ne 1$, phương trình $(2)$ có \[\Delta'=m^2-4(m-1)=(m-2)^2\ge 0,\forall m.\]
Phương trình $(2)$ có hai nghiệm
\begin{align*}
&x_1=\dfrac{-m+(m-2)}{m-1}=\dfrac{2}{1-m};\\ 
&x_2=\dfrac{-m-(m-2)}{m-1}=\dfrac{-2(m-1)}{m-1}=-2.
\end{align*}
Nghiệm $x_2$ là nghiệm của $(1)$ nếu $\dfrac{1}{m}\ne -2\Leftrightarrow m\ne -\dfrac{1}{2}$.\\
Nghiệm $x_1$ là nghiệm của $(1)$ nếu $\begin{cases}\dfrac{2}{1-m}\ne 1\\ \dfrac{2}{1-m}\ne -1\\ \dfrac{2}{1-m}\ne \dfrac{1}{m}\end{cases}\Leftrightarrow \begin{cases}m\ne -1, \\ m\ne 3, \\ m\ne \dfrac{1}{3}.\end{cases}$.
\end{itemize}
\textit{Kết luận}:
\begin{itemize}
\item Nếu $m=0$ thì phương trình $(1)$ có nghiệm $x=\pm 2$.
\item Nếu $m=1$ hoặc $m=-1$ hoặc $m=3$ hoặc $m=\dfrac{1}{3}$ thì phương trình $(1)$ có nghiệm $x=-2$.
\item Nếu $m=-\dfrac{1}{2}$ thì phương trình $(1)$ có nghiệm $x=\dfrac{4}{3}$.
\item Nếu $m\not\in\left\{-1;-\dfrac{1}{2};0;\dfrac{1}{3};1;3\right\}$ thì phương trình $(1)$ có nghiệm $x_1=-2$ và $x_2=\dfrac{2}{1-m}$.
\end{itemize}
\end{enumerate}
	}
\end{bt}

\begin{bt} 
Giải và biện luận phương trình ($m$ là tham số) \[(2m^2-3m-2)x^2+(m^2+7m+2)x-m^2-2m=0.\tag{1}\]
\loigiai{
\begin{enumerate}
\item Nếu $2m^2-3m-2=0\Leftrightarrow (2m+1)(m-2)=0\Leftrightarrow \left[\begin{aligned}& m=-\dfrac{1}{2},\\ & m=2.\end{aligned}\right.$
\begin{itemize}
\item Với $m=2$, phương trình $(1)\Leftrightarrow 20x-8=0\Leftrightarrow x=\dfrac{2}{5}.$
\item Với $m=-\dfrac{1}{2}$, phương trình $(1)\Leftrightarrow -5x+3=0\Leftrightarrow x=\dfrac{3}{5}.$
\end{itemize}
\item Nếu $m\ne 2$ và $m\ne -\dfrac{1}{2}$, phương trình $(1)$ là phương trình bậc hai có
\begin{align*}
\Delta&=(m^2+7m+2)^2+4(2m^2-3m-2)(m^2+2m)\\ 
&=9m^4+18m^3+21m^2+12m+4\\
&=9(m^4+2m^3+m^2)+12(m^2+m)+4\\
&9(m^2+m)^2+12(m^2+m)+4=(3m^2+3m+2)^2\ge 0,\forall m.
\end{align*}
Khi đó phương trình $(1)$ có hai nghiệm là
\begin{align*}
&x_1=\dfrac{-(m^2+7m+2)+(3m^2+3m+2)}{2(2m^2-3m-2)}=\dfrac{2m(m-2)}{2(2m+1)(m-2)}=\dfrac{m}{2m+1};\\ 
&x_2=\dfrac{-(m^2+7m+2)-(3m^2+3m+2)}{2(2m^2-3m-2)}=\dfrac{-2(2m^2+5m+2)}{2(2m+1)(m-2)}=\dfrac{m+2}{2-m}.
\end{align*}
\item[]\textit{Kết luận}:
\begin{itemize}
\item Nếu $m=2$ thì phương trình $(1)$ có nghiệm $x=\dfrac{2}{5}$.
\item Nếu $m=-\dfrac{1}{2}$ thì phương trình $(1)$ có nghiệm $x=\dfrac{3}{5}$.
\item Nếu $m\ne 2$ và $m\ne -\dfrac{1}{2}$ thì phương trình $(1)$ có nghiệm $x_1=\dfrac{m}{2m+1}$ và $x_2=\dfrac{m+2}{2-m}$.
\end{itemize}
\end{enumerate}
	}
\end{bt}

\begin{bt}
Cho hai phương trình
\begin{align*}
&2x^2+\left(3m+1\right)x-9=0,\tag{1}\\ 
&6x^2+\left(7m-1\right)x-19=0.\tag{2}
\end{align*}
Với giá trị nào của tham số $m$ thì hai phương trình $(1)$ và $(2)$ có nghiệm chung? Tìm các nghiệm chung đó.
\loigiai{
Giả sử $x_0$ là nghiệm chung của hai phương trình $(1)$ và $(2)$. Khi đó ta có
\begin{align*}
\begin{cases}2x_0^2+\left(3m+1\right)x_0-9=0\\6x_0^2+\left(7m-1\right)x_0-19=0\end{cases}&\Leftrightarrow \begin{cases}-6x_0^2-3\left(3m+1\right)x_0+27=0\\6x_0^2+\left(7m-1\right)x_0-19=0\end{cases}\\ 
&\Rightarrow -2(m+2)x_0+8=0 \Rightarrow (m+2)x_0=4.\tag{3}
\end{align*}
Do $x_0$ là nghiệm nên $m+2\ne 0\Leftrightarrow m\ne -2$. Khi đó từ $(3)$ suy ra $x_0=\dfrac{4}{m+2}$.\\
Thay $x_0=\dfrac{4}{m+2}$ vào $(1)$ ta được 
\[2\left(\dfrac{4}{m+2}\right)^2+(3m+1)\cdot\dfrac{4}{m+2}-9=0\Leftrightarrow 3m^2-8m+4=0\Leftrightarrow \left[\begin{aligned}& m=2,\\ & m=\dfrac{2}{3}.\end{aligned}\right.\]
Ngược lại:
\begin{itemize}
\item Với $m=2$ thì phương trình $(1)$ có nghiệm là $x=1$ và $x=-\dfrac{9}{2}$; phương trình $(2)$ có nghiệm là $x=1$ và $x=-\dfrac{19}{6}$. Do đó, hai phương trình $(1)$ và $(2)$ có nghiệm chung là $x=1$.
\item Với $m=\dfrac{2}{3}$ thì phương trình $(1)$ có nghiệm là $x=\dfrac{3}{2}$ và $x=-3$; phương trình $(2)$ có nghiệm là $x=\dfrac{3}{2}$ và $x=-\dfrac{19}{9}$. Do đó, hai phương trình $(1)$ và $(2)$ có nghiệm chung là $x=\dfrac{3}{2}$.
\end{itemize}
Vậy $m=2$ hoặc $m=\dfrac{2}{3}$ là giá trị cần tìm.
}
\end{bt}

\begin{bt}\label{9D42BT9}
Tìm giá trị nhỏ nhất của biểu thức $P=5x^2-4x+1$.
\loigiai{
Gọi $a$ là một giá trị của $P$. Biểu thức $P$ nhận giá trị $a$ khi và chỉ khi phương trình $5x^2-4x+1=a$ có nghiệm $\Leftrightarrow 5x^2-4x+1-a=0$ có nghiệm 
\[\Leftrightarrow \Delta'=5a-1\ge 0\Leftrightarrow a\ge \dfrac{1}{5}.\]
Vậy $P_{\min}=\dfrac{1}{5}$ khi và chỉ khi $x=\dfrac{2}{5}$.
}
\end{bt}
\noindent{\bf Chú ý.} Phương pháp giải ở bài tập \ref{9D42BT9} gọi là \textit{phương pháp miền giá trị của hàm số}. Để tìm được miền giá trị này, ta sử dụng điều kiện có nghiệm của phương trình bậc hai là $\Delta\ge 0$. Sau đây là một vài bài tập khác có thể sử dụng phương pháp miền giá trị của hàm số để giải.

\begin{bt}
Tìm giá trị nhỏ nhất, giá trị lớn nhất của biểu thức $Q=\dfrac{4x-3}{x^2+1}$.
\loigiai{
Điều kiện xác định: $x\in\mathbb{R}$.\\
Gọi $a$ là một giá trị của $Q$. Biểu thức $Q$ nhận giá trị $a$ khi và chỉ khi phương trình $\dfrac{4x-3}{x^2+1}=a$ có nghiệm \[\Leftrightarrow ax^2-4x+(a+3)=0\tag{1}\]có nghiệm.
\begin{itemize}
\item Nếu $a=0$ thì $(1)\Leftrightarrow -4x=-3\Leftrightarrow x=\dfrac{3}{4}$.
\item Nếu $a\ne 0$ thì $(1)$ (là phương trình bậc hai) có nghiệm
\[\Leftrightarrow \Delta'=2^2-a(a+3)\ge 0\Leftrightarrow -a^2-3a+4\ge 0\Leftrightarrow -4\le a\le 1.\]
\end{itemize}
Do đó ta có
\begin{itemize}
\item $Q_{\min}=-4$ khi và chỉ khi $x=-\dfrac{1}{2}$.
\item $Q_{\max}=1$ khi và chỉ khi $x=2$.
\end{itemize} 
}
\end{bt}

\begin{bt}
Tìm giá trị nhỏ nhất, giá trị lớn nhất của biểu thức $M=\dfrac{x^2+2x-1}{x^2-2x+3}$.
\loigiai{
Điều kiện xác định: $x\in\mathbb{R}$. 
Gọi $m$ là một giá trị của $M$. \\Biểu thức $M$ nhận giá trị $m$ khi và chỉ khi phương trình $\dfrac{x^2+2x-1}{x^2-2x+3}=m$ có nghiệm \[\Leftrightarrow (m-1)x^2-2(m+1)x+(3a+1)=0\tag{1}\]có nghiệm.

\end{itemize}
Do đó ta có
\begin{itemize}
\item $M_{\min}=1-\sqrt{2}$ khi và chỉ khi $x=1-\sqrt{2}$.
\item $M_{\max}=1+\sqrt{2}$ khi và chỉ khi $x=1+\sqrt{2}$.
\end{itemize} 
}
\end{bt}
