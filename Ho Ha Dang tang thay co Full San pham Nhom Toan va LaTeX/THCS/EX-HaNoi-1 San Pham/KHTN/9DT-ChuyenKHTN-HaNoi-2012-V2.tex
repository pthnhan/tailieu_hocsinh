\begin{ex}%[Đề thi vào 10, Chuyên Khoa học Tự nhiên, Hà Nội, 2012, Vòng 2]%[9D5K5]
Giải hệ phương trình $\heva{&xy(x+y)=2\\ &9xy(3x-y)+6=26x^3-2y^3}$.
\loigiai{
Hệ phương trình đã cho tương đương
\begin{align*}
\heva{& xy(x+y)=2\\ &(x+y)^3=(3x-y)^3} \Leftrightarrow \heva{&xy(x+y)=2\\ &x+y=3x-y} \Leftrightarrow \heva{&x=y\\ &xy(x+y)=2} \Leftrightarrow \heva{&x=1\\ &y=1}.
\end{align*}
Vậy hệ đã cho có nghiệm duy nhất $(1;1)$.
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Khoa học Tự nhiên, Hà Nội, 2012, Vòng 2]%[9D5G5]
Giải phương trình $\left( \sqrt{x+4}-2\right) \left( \sqrt{4-x}+2 \right)=2x$.
\loigiai{
Điều kiện của phương trình $-4\le x\le 4$.
Phương trình đã cho tương đương với
\begin{align*}
\dfrac{x}{\sqrt{x+4}+2} \left( \sqrt{4-x}+2 \right)=2x. \tag{1}
\end{align*}
Dễ thấy $x=0$ là nghiệm của phương trình. Xét $x\ne 0$, phương trình $(1)$ tương đương với
\begin{align*}
\sqrt{4-x}+2=2\left( \sqrt{x+4} +2\right).
\end{align*}
 Đặt $u=\sqrt{x+4}$, $v=\sqrt{4-x}$ ($u,v\ge 0$), hệ phương trình trở thành
\begin{align*}
\heva{&v=2u+2 \\ &u^2+v^2=8} \Leftrightarrow \heva{&u=\dfrac{2}{5}\\ &v=\dfrac{14}{5}}
\end{align*}
Từ đó ta tìm được $x=-\dfrac{96}{25}$. Vậy phương trình đã cho có hai nghiệm là $0$ và $-\dfrac{96}{25}$.
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Khoa học Tự nhiên, Hà Nội, 2012, Vòng 2]%[9D5G2]
Tìm hai chữ số cuối cùng của số $A=41^{106}+57^{2012}$.
\loigiai{
Ta có $41^2 = (40+1)^2=40^2+80+1 \equiv 81 \mod 100$. Suy ra $41^4 \equiv 81^2 \equiv 61 \mod{100}$ nên $41^5\equiv 61\cdot 41 \equiv 1 \mod{100}$. Do đó $41^{106}\equiv 41\left( 41^5 \right)^{21}\equiv 41 \mod{100}$.\\ Ta lại có $57^2 \equiv 49 \mod{100}$ nên $57^{4} \equiv 1 \mod{100}$, suy ra $57^{2012}= \left( 57^4 \right)^{503} \equiv 1 \mod{100}$.\\
Vậy hai chữ số tận cùng của $A$ là $41+1=42$.
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Khoa học Tự nhiên, Hà Nội, 2012, Vòng 2]%[9D5G1]
Tìm giá trị lớn nhất của hám số $y=3\sqrt{2x-1}+x\sqrt{5-4x^2}$, với $\dfrac{1}{2}\le x \le \dfrac{\sqrt{5}}{2}$.
\loigiai{
Với điều kiện $\dfrac{1}{2} \le x \le \dfrac{\sqrt{5}}{2}$, áp dụng bất đẳng thức Cauchy ta có
\begin{align*}
&x\sqrt{5-4x^2}\le \dfrac{x^2+5-4x^2}{2}=\dfrac{5-3x^2}{2},\\
&3\sqrt{2x-1}\le 3\cdot \dfrac{2x-1+1}{2}=3x \le \dfrac{3(x^2+1)}{2}.
\end{align*}
Do đó
\begin{align*}
y=3\sqrt{2x-1}+x\sqrt{5-4x^2} \le \dfrac{x^2+5-4x^2}{2} + \dfrac{3(x^2+1)}{2} =4.
\end{align*}
Đẳng thức xảy ra khi $x=1$. Vậy giá trị lớn nhất của $y$ là $4$.
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Khoa học Tự nhiên, Hà Nội, 2012, Vòng 2]%[9H3K7]
Cho tam giác nhọn $ABC$ ($AB>AC$), nội tiếp đường tròn $(O)$. Giả sử $M$, $N$ là hai điểm thuộc cung nhỏ $\wideparen{BC}$ sao cho $MN$ song song với $BC$ và tia $AN$ nằm giữa hai tia $AM$ và $AB$. Gọi $P$ là hình chiếu vuông góc của điểm $C$ trên $AN$ và $Q$ là hình chiếu vuông góc của điểm $M$ trên $AB$.
\begin{enumerate}
\item Giả sử $CP$ cắt $QM$ tại điểm $T$. Chứng minh rằng $T$ nằm trên đường tròn tâm $(O)$.
\item Gọi giao điểm của $NQ$ và $(O)$ là $R$ khác $N$. Giả sử $AM$ cắt $PQ$ tại $S$. Chứng minh rằng bốn điểm $A$, $R$, $Q$, $S$ cùng thuộc một đường tròn.
\end{enumerate}
\loigiai{
\immini{
\begin{enumerate}
\item Do $MN\parallel BC$ nên $\wideparen{CM}=\wideparen{BN}$ hay $\widehat{BAN}=\widehat{CAM}$. Mặt khác tứ giác $APQT$ nội tiếp đường trong đường kính $AT$ nên $\widehat{PTQ}=\widehat{PAQ}$. Suy ra $\widehat{CTM}=\widehat{CAM}$, do đó tứ giác $CATM$ nọi tiếp hay $T$ thuộc đường tròn $(O)$.
\item Do tứ giác $APQT$ nội tiếp nên $\widehat{TPQ}=\widehat{TAQ}$. Mặt khác do tứ giác $TACB$ nội tiếp nên $\widehat{TAB}=\widehat{TCB}$, suy ra $\widehat{TPQ}=\widehat{TCB}$. Từ đó $PQ\parallel BC$. Suy ra $\widehat{ASQ}=\widehat{AMN}$, mà $\widehat{AMN}+\widehat{ARQ}=180^\circ$ nên $\widehat{ASQ}+\widehat{ARQ}=180^\circ$, hay tứ giác $A$, $R$, $Q$, $S$ nội tiếp.
\end{enumerate}
}
{
\begin{tikzpicture}[>=stealth]
\clip (-3.5,-3.5) rectangle (3.5,3.5);
\tkzDefPoints{0/0/O}
\tkzDefShiftPoint[O](-35:3){C}
\tkzDefShiftPoint[O](215:3){B}
\tkzDefShiftPoint[O](55:3){A}
\tkzDefShiftPoint[O](-65:3){M}
\tkzDefMidPoint(B,C)\tkzGetPoint{a}
\tkzDefLine[parallel=through M](B,C)\tkzGetPoint{b}
\tkzInterLC(M,b)(O,A)\tkzGetSecondPoint{N}
\tkzDefPointBy[projection=onto A--N](C)\tkzGetPoint{P}
\tkzDefPointBy[projection=onto A--B](M)\tkzGetPoint{Q}
\tkzInterLL(C,P)(Q,M)\tkzGetPoint{T}
\tkzInterLC(N,Q)(O,A)\tkzGetSecondPoint{R}
\tkzInterLL(A,M)(P,Q)\tkzGetPoint{S}

\tkzDrawCircle(O,A)
\tkzDrawSegments(A,B B,C C,A M,N C,T N,R A,R A,M A,N Q,S M,T A,T)
\tkzLabelPoints[above right](A)
\tkzLabelPoints[above](R)
\tkzLabelPoints[above left](T)
\tkzLabelPoints[below](M,N,P)
\tkzLabelPoints[below left](B)
\tkzLabelPoints[below right](C)
\tkzLabelPoints[left](Q)
\tkzLabelPoints[right](S)
\tkzMarkRightAngles(A,P,T A,Q,T)
\tkzDrawPoints[fill=white, size=10](A,B,C,M,N,P,Q,R,S,T)

\end{tikzpicture}
}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Khoa học Tự nhiên, Hà Nội, 2012, Vòng 2]%[9D5G5]
Với mỗi số nguyên $n$ lớn hơn hoặc bằng $2$ cố định, xét các tập $n$ số thực đôi một khác nhau $X=\{x_1,x_2,\ldots, x_n\}$. Kí hiệu $C(X)$ là số các giá trị khác nhau của tổng $x_i+x_j$ ($1\le i<j\le n$). Tìm giá trị lớn nhất và nhỏ nhất của $C(X)$.
\loigiai{
\begin{itemize}
\item Giả sử các số của tập hợp $X$ được sắp xếp theo thứ tự $x_1<x_2<\cdots <x_n$. Ta có
\begin{align*}
x_1+x_2<x_1+x_3<\cdots <x_1+x_n<x_2+x_n<x_3+x_n<\cdots < x_{n-1}+x_n.
\end{align*}
Từ đó sẽ có ít nhất $2n-3$ giá trị phân biệt của tổng $x_i+x_j$. Suy ra $C(X)\ge 2n-3$. Xét tập $X_1=\{1,2,\ldots,n\}$. Khi đó với mọi $1\le i<j\le n$ thì $x_i+x_j=i+j\in\{3,4,\ldots, 2n-1\}$ và $C(X_1)=2n-3$. Vậy giá trị nhỏ nhất của $C(X)$ là $2n-3$.
\item Số các tổng $x_i+x_j$ là $\dfrac{n(n-1)}{2}$. Suy ra $C(X)\le \dfrac{n(n-1)}{2}$. Xét tập $X_2=\{2,2^2,\ldots, 2^n\}$. Giả sử tồn tại bốn chỉ số $i<r<s<j$ sao cho $x_i+x_j=x_r+x_s$. Khi đó $2^i+2^j=2^r+2^s$. Suy ra $1+2^{j-i}=2^{r-i}+2^{s-i}$, do đó $2 \mid 1$, vô lí. Điều đó khẳng định các tổng $x_i+x_j$ với $x_i, x_j\in X_2$ ($i<j$) đều khác nhau, suy ra $C(X_2)=\dfrac{n(n-1)}{2}$. Vậy giá trị lớn nhất của $C(X)$ là $\dfrac{n(n-1)}{2}$.
\end{itemize}
}
\end{ex}
