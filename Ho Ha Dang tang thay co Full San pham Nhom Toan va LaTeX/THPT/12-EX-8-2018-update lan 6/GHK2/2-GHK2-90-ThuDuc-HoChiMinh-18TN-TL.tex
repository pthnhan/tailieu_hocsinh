	\loigiai{
$I=\displaystyle\int\limits_{1}^{\mathrm{e}}  x\ln x\,\mathrm{\,d}x+ \displaystyle\int\limits_{1}^{\mathrm{e}} \dfrac{1}{x}\ln x\,\mathrm{\,d}x= J+K$.\\
Ta tính tích phân $J=\displaystyle\int\limits_{1}^{\mathrm{e}}  x\ln x\,\mathrm{\,d}x $\\
Với $\heva{u & =\ln x\\ \mathrm{\,d}v & =x\mathrm{\,d}x} \Leftrightarrow \heva{\mathrm{\,d}u & =\dfrac{1}{x}\mathrm{\,d}x\\ v & =\dfrac{x^2}{2}}.$\\
Khi đó $J=\dfrac{x^2}{2} \cdot \ln x\bigg|_{1}^{\mathrm{e}}- \displaystyle\int\limits_{1}^{\mathrm{e}}  \dfrac{x}{2} \,\mathrm{\,d}x= \dfrac{\mathrm{e}^2}{2}- \dfrac{x^2}{4}	\bigg|_{1}^{\mathrm{e}}=\dfrac{\mathrm{e}^2}{4}+\dfrac{1}{4}$.\\
Tính tích phân $K=\displaystyle\int\limits_{1}^{\mathrm{e}} \dfrac{1}{x}\ln x\,\mathrm{\,d}x= \displaystyle\int\limits_{1}^{\mathrm{e}} \left( \ln x\right)' \ln x\,\mathrm{\,d}x= \dfrac{\ln^2 x}{2}\bigg|_{1}^{\mathrm{e}}= \dfrac{1}{2}$.\\
Vậy $I=J+K=\dfrac{\mathrm{e}^2}{2}+\dfrac{1}{4}. $
	}
\end{bt}
