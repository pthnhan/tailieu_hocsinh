        \emph{Nhận xét:} Không cần xét trường hợp sau, vì trong các đáp án $1$ là giá trị nhỏ nhất.
}
\end{ex}

{\True $\dfrac{1}{2019\times2021}$}
{$\dfrac{1}{2018\times2021}$}
{$\dfrac{1}{2018\times2019}$}
{$\dfrac{1}{2021\times2022}$}
