\begin{bt}%[Đề HK1 Lớp 10, Nguyễn Công Trứ, HCM 2018]%[Trần Hòa]%[0D3B2-6]
	Cho phương trình $(m^2-1)x=m^2-4m+3$. Tìm các giá trị của $m$ sao cho phương trình có tập nghiệm là $\mathbb{R}$.
	\loigiai{
		Phương trình có tập nghiệm là $\mathbb{R}$ khi 
		$$\heva{&m^2-1=0\\&m^2-4m+3=0}\Leftrightarrow \heva{&m=\pm 1\\&{\hoac{&m=1\\&m=3}}}\Leftrightarrow m=1.$$
	}
\end{bt}
\begin{bt}%[Đề HK1 Lớp 10, Nguyễn Công Trứ, HCM 2018]%[Trần Hòa]%[0D3B2-5]
	Cho phương trình $mx^2-(2m+3)x+m+1=0$. Định $m$ để phương trình có $2$ nghiệm phân biệt $x_1$, $x_2$ thỏa $x_1^2 + x_2^2 = 1$.
	\loigiai{
		Phương trình có hai nghiệm phân biệt khi 
		$$\heva{&a\neq 0\\&\Delta >0}\Leftrightarrow \heva{&m\neq 0\\&\left[-(2m+3)\right]^2-4m(m+1)>0}\Leftrightarrow \heva{&m\neq 0\\&8m+9>0}\Leftrightarrow \heva{&m\neq 0\\&m >-\dfrac{9}{8}.}\quad (*)$$
		Với điều kiện $(*)$ phương trình có hai nghiệm $x_1$, $x_2$. Theo hệ thức Vi-ét có $\heva{&x_1+x_2=\dfrac{2m+3}{m}\\&x_1x_2 =\dfrac{m+1}{m}.}$\\
		\begin{eqnarray*}
                  x_1^2+x_2^2=1&\Leftrightarrow& (x_1+x_2)^2-2x_1x_2=1\\
                  &\Leftrightarrow& \left(\dfrac{2m+3}{m}\right)^2 -2\cdot \dfrac{m+1}{m}=1\\
                  &\Leftrightarrow& (2m+3)^2-2m(m+1)=m^2\\
                  &\Leftrightarrow& m^2+10m+9=0\\
                  &\Leftrightarrow& \hoac{&m=-1\,\,\text{(nhận)}\\&m=-9.\,\,\text{(loại)}}\\
                 \end{eqnarray*}
		Vậy $m=-1$ thỏa mãn yêu cầu bài toán.
	}
\end{bt}
\begin{bt}%[Đề HK1 Lớp 10, Nguyễn Công Trứ, HCM 2018]%[Trần Hòa]%[0D3B3-2]
	Định tham số $m$ để hệ phương trình $\heva{&mx +2y=3m+1\\& 3x + (m-1)y=4m-2}$ có nghiệm duy nhất. Tìm nghiệm duy nhất đó.
	\loigiai{
		Ta có\\
		$D=m(m-1)-6=m^2-m-6=(m-3)(m+2)$.\\
		$D_x=(3m+1)(m-1)-2(4m-2)=3m^2-10m+3=(m-3)(3m-1)$.\\
		$D_y=m(4m-2)-3(3m+1)=(m-3)(4m+1)$.\\
		$\bullet$ Hệ có nghiệm duy nhất khi và chỉ khi $D\neq 0\Leftrightarrow \heva{&m\neq 3\\&m\neq -2.}\quad (*)$\\
		$\bullet$ Với điều kiện $(*)$ nghiệm duy nhất của hệ là $\heva{&x=\dfrac{D_x}{D}=\dfrac{3m-1}{m+2}\\&y=\dfrac{D_y}{D}=\dfrac{4m+1}{m+2}.}$
	}
\end{bt}
\begin{bt}%[Đề HK1 Lớp 10, Nguyễn Công Trứ, HCM 2018]%[Trần Hòa]%[0D3B3-4]
	Giải hệ phương trình sau $\heva{&2x + 3y =2 \quad \quad \quad \quad (1)\\& 2x^2+xy-2y=0.\quad (2)}$
	\loigiai{
		$(1)\Leftrightarrow y=\dfrac{2-2x}{3}\quad (3)$, thay vào phương trình $(2)$ ta được
		$$2x^2+x\cdot \left(\dfrac{2-2x}{3}\right)-2\cdot \left(\dfrac{2-2x}{3}\right)=0\Leftrightarrow 4x^2+6x-4=0\Leftrightarrow \hoac{&x=\dfrac{1}{2}\\&x=-2.}$$
		$\bullet$ Với $x=\dfrac{1}{2}$ thay vào $(3)$ có $y=\dfrac{1}{3}$.\\
		$\bullet$ Với $x=-2$ thay vào $(3)$ có $y=2$.\\
		Vậy hệ có hai nghiệm $\left(\dfrac{1}{2};\dfrac{1}{3}\right)$ và $(-2;2)$.
	}
\end{bt}
\begin{bt}%[Đề HK1 Lớp 10, Nguyễn Công Trứ, HCM 2018]%[Trần Hòa]%[0D4K1-5]
	Tìm GTLN, GTNN của hàm số $y=4+\sqrt{(x-2)(6-x)}$ trên $[2;6]$.
	\loigiai{
		$\bullet$ Với $x\in [2;6]\Rightarrow (2-x)\ge 0, (6-x)\ge 0$ nên  $y=4+\sqrt{(x-2)(6-x)}\ge 4$.\\ Vậy GTNN của hàm số $y$ bằng $4$ khi $x=2$ hoặc $x=6$.\\
		$\bullet$ Áp dụng bất đẳng thức cô-si, có $y=4+\sqrt{(x-2)(6-x)}\le 4+\dfrac{x-2+6-x}{2}=6$.\\ Vậy GTLN của hàm số $y$ bằng $6$ khi $x-2=6-x$ hay $x=4$.
	}
\end{bt}
\begin{bt}%[Đề HK1 Lớp 10, Nguyễn Công Trứ, HCM 2018]%[Trần Hòa]%[0H2B2-1]%[0H2K2-1]
	Cho tam giác $ABC$ có $AB=5$, $AC=8$, góc $\widehat{BAC}$  bằng $120^\circ$. Gọi $E$ là điểm thỏa mãn $\vec{AC}=4\vec{AE}$.
	\begin{enumerate}
	    \item Tính $\vec{AB}\cdot \vec{AC}$ và $\vec{BC}\cdot \vec{AE}$.
	    \item Tính độ dài trung tuyến $AM$ của tam giác $ABC$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $\vec{AB}\cdot \vec{AC}=AB\cdot AC\cdot \cos \widehat{BAC}=5\cdot 8\cdot \left(-\dfrac{1}{2}\right)=-20$.
			\begin{eqnarray*}
                                \vec{BC}\cdot \vec{AE}
                                &=&\dfrac{1}{4}\left(\vec{AC}-\vec{AB}\right)\cdot \vec{AC}\\
                                &=&\dfrac{1}{4}\left(\vec{AC}\cdot \vec{AC}-\vec{AB}\cdot \vec{AC}\right)\\
                                &=&\dfrac{1}{4}\cdot AC^2-\dfrac{1}{4}\cdot AB\cdot AC\cdot \cos \widehat{BAC}\\
                                &=&1 6+5=21.
                           \end{eqnarray*}
			\item Vì $AM$ là trung tuyến của tam giác $ABC$ nên
			\begin{eqnarray*}
			\vec{AM}^2 &=&\dfrac{1}{4}\left(\vec{AB}+\vec{AC}\right)^2\\
			           &=&\dfrac{1}{4}\left(AB^2+2\vec{AB}\cdot \vec{AC}+AC^2\right)\\
			           &=&\dfrac{1}{4}(25-40+64)=\dfrac{49}{4}\\
			           &\Rightarrow& AM=\dfrac{7}{2}.
			\end{eqnarray*}
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Đề HK1 Lớp 10, Nguyễn Công Trứ, HCM 2018]%[Trần Hòa]%[0H1B4-3]%[0H2B2-1]
	Trong mặt phẳng tọa độ $Oxy$ cho ba điểm $A(1;-2)$, $B(0;1)$, $C(4;-1)$.
	\begin{enumerate}
	    \item Tìm tọa độ điểm $D$ sao cho $ABCD$ là hình bình hành. Xác định tọa độ tâm của hình bình hành $ABCD$.
	    \item Chứng minh tam giác $ABC$ vuông tại $A$. Từ đó tính diện tích tam giác $ABC$ và bán kính đường tròn ngoại tiếp tam giác $ABC$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Gọi $D(x;y)$. Có $\vec{AB}=(-1;3)$, $\vec{DC}=(4-x;-1-y)$.\\
			$ABCD$ là hình bình hành $\Leftrightarrow \vec{AB}=\vec{DC}\Leftrightarrow \heva{&-1=4-x\\&3=-1-y}\Leftrightarrow \heva{&x=5\\&y=-4}$.
			Vậy $D(5;-4)$.\\
			Gọi $I$ là tâm hình bình hành $ABCD$, suy ra $I$ là trung điểm của $AC$, vậy $I\left(\dfrac{5}{2};-\dfrac{3}{2}\right)$. 
			\item $\vec{AC}=(3;1)$. Có $\vec{AB}\cdot \vec{AC}=-1\cdot 3+3\cdot 1=0\Rightarrow \triangle ABC$ vuông tại $A$.\\
			Có $AB=\sqrt{(-1)^2+3^2}=\sqrt{10}$, $AC=\sqrt{3^2+1^2}=\sqrt{10}$, nên diện tích tam giác $ABC$ là $S_{\triangle ABC}=\dfrac{1}{2}AB\cdot AC=5$.\\
			Có $BC=\sqrt{AB^2+AC^2}=2\sqrt{5}$. Vì $BC$ là đường kính đường tròn ngoại tiếp tam giác $ABC$ nên bán kính là $R=\dfrac{BC}{2}=\sqrt{5}$.
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Đề HK1 Lớp 10, Nguyễn Công Trứ, HCM 2018]%[Trần Hòa]%[0D4G1-2]
	Cho các số thực không âm $x, y, z$. Chứng minh bất đẳng thức sau $$(1+x)(1+y)(1+z)=\left(1+\sqrt[3]{xyz}\right)^3.\quad (1)$$
	\loigiai{
		\begin{eqnarray*}
		(1)&\Leftrightarrow& 1+x+y+z+xy+yz+zx+xyz \ge 1+3\sqrt[3]{xyz}+3\sqrt[3]{(xyz)^2}+xyz\\
		   &\Leftrightarrow&  x+y+z+xy+yz+zx \ge 3\sqrt[3]{xyz}+3\sqrt[3]{(xyz)^2}.\quad (2)
		\end{eqnarray*}
		Áp dụng bất đẳng thức Cô-si cho ba số không âm ta có
		\begin{eqnarray*}
		x+y+z&\ge& 3\sqrt[3]{xyz}\quad \quad(3)\\
		xy+yz+zx&\ge& 3\sqrt[3]{(xyz)^2}.\quad    (4)
		\end{eqnarray*}
		Từ $(3)$ và $(4)$ suy ra
		$$x+y+z+xy+yz+zx \ge 3\sqrt[3]{xyz}+3\sqrt[3]{(xyz)^2}.$$
		Bất đẳng thức $(2)$ đúng, vậy bất đẳng thức $(1)$ được chứng minh.\\
		Dấu $``="$ xảy ra khi $x=y=z$.
	}
\end{bt}
