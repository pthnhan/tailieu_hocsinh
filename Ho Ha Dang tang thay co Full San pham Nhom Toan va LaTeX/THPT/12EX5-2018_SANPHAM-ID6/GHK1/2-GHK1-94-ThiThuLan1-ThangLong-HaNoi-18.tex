\begin{ex}%[Thi thử L1,Thăng Long, Hà Nội, 2018]%[Huỳnh Xuân Tín, dự án EX5]%[2D1B1-1]
	Hàm số nào dưới đây đồng biến trên $\mathbb{R}$?
	\choice
	{\True $y= x^3+3x+2$}
	{$y=x^4+2x^2+3$}
	{$y=2x^2$}
	{$y=\dfrac{x}{x+2}$}
	\loigiai{
		Ta đạo hàm lần lượt từng đáp án sẽ thu được kết quả.\\
		$y'= \left(x^3+3x+2\right)^{'} =3x^2+3>0, \forall x \in \mathbb{R}$.\\
		$y'= \left(x^4+2x^2+3\right)^{'} =4x^3+4x$ không âm khi $x\geq 0$.\\
		$y'= \left(2x^2\right)^{'} =4x$ không âm khi $x\geq 0$.\\
		$y'= \left(\dfrac{x}{x+2}\right)^{'} =\dfrac{2}{(x+2)^2}$ chỉ lớn hơn không  khi $x\not=-2$.\\
	}
\end{ex}

\begin{ex}%[Thi thử L1,Thăng Long, Hà Nội, 2018]%[Huỳnh Xuân Tín, dự án EX5]%[2D2K4-3]

	{Hàm số nghịch biến trên  $(1;\mathrm{e})$}
