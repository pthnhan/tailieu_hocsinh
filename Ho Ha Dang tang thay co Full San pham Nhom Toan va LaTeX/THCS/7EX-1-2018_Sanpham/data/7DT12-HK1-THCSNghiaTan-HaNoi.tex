\noindent\textbf{II. PHẦN TỰ LUẬN}
	\setcounter{ex}{0}
\begin{ex}%[Đề kiểm tra học kỳ I,THCSNghiaTan-HaNoi, 2011 - 2012]%[Lê Xuân Hòa, dự án 7EX-1-2018]%[7D1K2]
Tính hợp lý (nếu có thể)
\begin{enumerate}
\item $H =\dfrac{3}{29}-\dfrac{13}{47}-\left( -\dfrac{17}{29}\right) -\dfrac{34}{47} +\dfrac{9}{29}+\dfrac{1}{2}$.
\item $P=\left(\dfrac{2}{3}\right)^5 :\left(\dfrac{2}{3}\right) ^4 -\sqrt{4}+\dfrac{1}{3}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có \begin{align*}
H &=\dfrac{3}{29}-\dfrac{13}{47}-\left( -\dfrac{17}{29}\right) -\dfrac{34}{47} +\dfrac{9}{29}+\dfrac{1}{2}\\
	&= \dfrac{3}{29}-\dfrac{13}{47}+\dfrac{17}{29} -\dfrac{34}{47}+\dfrac{9}{29}+\dfrac{1}{2}\\
	&=\left(\dfrac{3}{29}+\dfrac{17}{29}+\dfrac{9}{29}\right) -\left(\dfrac{13}{47}+\dfrac{34}{47}\right) +\dfrac{1}{2}\\
	&=\dfrac{3+17+9}{29}-\dfrac{13+34}{47}+\dfrac{1}{2}\\
	&=\dfrac{29}{29}-\dfrac{47}{47}+\dfrac{1}{2}\\
	&=1-1+\dfrac{1}{2}=\dfrac{1}{2}.
\end{align*}
\item Ta có \begin{align*}
P&=\left(\dfrac{2}{3}\right)^5 :\left(\dfrac{2}{3}\right) ^4 -\sqrt{4}+\dfrac{1}{3}\\
	&=\left(\dfrac{2}{3}\right)^{5-4}-2+\dfrac{1}{3}\\
	&=\dfrac{2}{3}+\dfrac{1}{3}-2\\
	&=1-2=-1.
\end{align*}
\end{enumerate}
}
\end{ex}	

\begin{ex}%[Đề kiểm tra học kỳ I,THCSNghiaTan-HaNoi, 2011 - 2012]%[Lê Xuân Hòa, dự án 7EX-1-2018]%[7D1K2]
Tìm $x$, biết
\begin{enumerate}
\item $\dfrac{2}{5}-x=\dfrac{2}{3}$.
\item $\left(\dfrac{1}{2}+3x\right)\left( x^2-\dfrac{4}{9}\right) =0$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có 
\begin{align*}
 \dfrac{2}{5}-&x=\dfrac{2}{3} \\
 & x= \dfrac{2}{5} -\dfrac{2}{3}\\
 &x=\dfrac{2\cdot 3 -2\cdot 5}{15} =-\dfrac{4}{5}.
 \end{align*}
 Vậy $x=-\dfrac{4}{5}$.
\item Ta có 
\begin{align*}
&\left(\dfrac{1}{2}+3x\right)\left( x^2-\dfrac{4}{9}\right) =0\\
 \Leftrightarrow &\hoac{&x^2-\dfrac{4}{9}=0\\&\dfrac{1}{2}+3x =0}\\
 \Leftrightarrow &\hoac{&x=\pm\dfrac{2}{3}\\&x=-\dfrac{1}{6}}.
\end{align*}
Vậy $x=\pm\dfrac{2}{3}$, $x=-\dfrac{1}{6}$.
\end{enumerate}
}
\end{ex}	

\begin{ex}%[Đề kiểm tra học kỳ I,THCSNghiaTan-HaNoi, 2011 - 2012]%[Lê Xuân Hòa, dự án 7EX-1-2018]%[7D1K2]
Ba đội máy cày có $39$ máy (cùng năng suất) cày trên ba cánh đồng có diện tích như nhau. Đội thứ nhất hoàn thành công việc trong $6$ ngày, đội thứ hai hoàn thành công việc trong $4$ ngày, đội thứ ba hoàn thành công việc trong $8$ ngày. Hỏi mỗi đội có bao nhiêu máy cày?
\loigiai{
Gọi $a$, $b$, $c$ lần lượt là số máy cày của đội thứ nhất, thứ hai và thứ ba.\\ Suy ra $a+b+c =39. \qquad\qquad (1)$\\
Vì thời gian hoàn thành và số máy cày là hai đại lượng tỉ lệ nghịch nên ta có
$6a =4b =8c $ hay $a=\dfrac{2}{3}b$ và $c=\dfrac{1}{2}b$.\\
Thay vào $(1)$ ta có $b=18$ suy ra $a=12$, $c=9$.\\
Vậy số máy cày của đội thứ nhất là $18\cdot\dfrac{2}{3}=12$.\\
Số máy cày của đội thứ hai là $18$.\\
Số máy cày đội thứ ba là $9$.
}
\end{ex}	

\begin{ex}%[Đề kiểm tra học kỳ I,THCSNghiaTan-HaNoi, 2011 - 2012]%[Lê Xuân Hòa, dự án 7EX-1-2018]%[7D1K2]
Cho tam giác $ABC$. Gọi $M$ là trung điểm của cạnh $BC$. Trên tia đối của tia $MA$ lấy điểm $D$ sao cho $MA=MD$.
\begin{enumerate}
\item Chứng minh tam giác $ABM$ bằng tam giác $DCM$.
\item Chứng minh $AB = CD$ và $AB \parallel CD$.
\item Tính  $\widehat{ACD}$ nếu biết $\widehat{BAC}=70^\circ$.
\item Họi $H$ là trung điểm của $AB$, $K$ là trung điểm của $CD$. Chứng minh $M$ là trung điểm của $HK$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=.7]
\tkzInit[ymin=-3,ymax=6,xmin=-6,xmax=9]
\tkzClip
\tkzDefPoints{-2/0/A,6/0/B,0/5/C,8/5/D}
\tkzLabelPoints[above](C,D)
\tkzLabelPoints(A,B)
\tkzDefMidPoint(B,C)\tkzGetPoint{M}
\tkzDefMidPoint(B,A)\tkzGetPoint{H}
\tkzDefMidPoint(D,C)\tkzGetPoint{K}
\tkzDrawSegments(A,B B,C C,D D,A A,C B,D H,K)
\tkzLabelPoints(A,B, H,K,M)
\tkzDrawPoints(A,B,C,D,M,K,H)
\tkzMarkAngles[size=0.5cm, label=$70^\circ$](B,A,C)
\tkzMarkAngles[size=1cm,mark=|](C,D,A B,A,D)
\tkzMarkSegments[mark=|](M,B M,C)
\tkzMarkSegments[mark=x](M,A M,D)
\tkzMarkSegments[mark=||](A,H H,B K,C K,D)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Xét  $\Delta ABM$ và $\Delta DCM$, ta có 
\begin{list}{}{}
\item $AM=DM$
\item $\widehat{AMB}=\widehat{DMC}$ (đối đỉnh)
\item $BM = CM$
\end{list}
Do đó $\Delta ABM = \Delta DCM$ (c.g.c) 
\item Vì $\Delta ABM = \Delta DCM$ nên  $AB = CD$ và $\widehat{CDM}=\widehat{BAM}$.\\
Do hai góc $\widehat{CDM}=\widehat{BAM}$ bằng nhau trong trường hợp so le trong nên suy ra $AB \parallel CD$.
\item Ta có $AB \parallel CD \Rightarrow \widehat{BAC} +\widehat{ACD}=180^\circ$ (hai góc trong cùng phía). Suy ra $\widehat{ACD} =180^\circ -70^\circ =110^\circ$.
\item Ta có $\Delta HMA = \Delta KMD$ (c.g.c), suy ra $MH= MK$   \\
$\Rightarrow \widehat{KMD}=\widehat{HMA}$.
Ta có 
\begin{align*}
180^\circ = \widehat{AMC}+\widehat{CMK}+\widehat{KMD} = \widehat{AMC}+\widehat{CMK}+\widehat{HMA} =\widehat{HMK}.
\end{align*}
  Hay ba điểm $H$, $M$, $K$ thẳng hàng.
\end{enumerate}
}
\end{ex}	

\begin{ex}%[Đề kiểm tra học kỳ I,THCSNghiaTan-HaNoi, 2011 - 2012]%[Lê Xuân Hòa, dự án 7EX-1-2018]%[7D1K2]
Cho biết $\dfrac{a}{b}=\dfrac{b}{c}=\dfrac{c}{a}$ và $a+b+c \ne 0$. Tính $H=\dfrac{a^3b^2c^{1930}}{c^{1935}}$.
\loigiai{
Ta có $\dfrac{a}{b}=\dfrac{b}{c}=\dfrac{c}{a}=\dfrac{a+b+c}{a+b+c}=1$. Suy ra $a=b=c$\\
Khi đó $H =\dfrac{a^3a^2a^{1930}}{a^{1935}}=\dfrac{a^{3+2+1930}}{a^{1935}}=1$.\\
Vậy $H=1$.
}
\end{ex}
