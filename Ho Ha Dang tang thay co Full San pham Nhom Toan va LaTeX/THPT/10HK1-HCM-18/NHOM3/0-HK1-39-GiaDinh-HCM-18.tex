%Câu 1
\begin{bt}%[Đề thi HK1, Gia Định, TpHCM 2018]%[0D3B2-6]
	Tìm tất cả các giá trị của tham số $m$ để phương trình sau có nghiệm\vspace*{-6pt}
	\begin{center}
		$(m^2-m-6)x=m^5-3m^4+m^2-5m+6$.
	\end{center}
	\loigiai{
		\begin{itemize}
			\item \textbf{Trường hợp 1}. $m^2-m-6=0 \Leftrightarrow \hoac{&m=-2\\&m=3}$.\\
			$\star$ Với $m=-2$, khi đó $m^5-3m^4+m^2-5m+6=-60$ nên phương trình vô nghiệm.\\
			$\star$ Với $m=3$, khi đó $m^5-3m^4+m^2-5m+6=0$ nên phương trình có vô số nghiệm.
			\item \textbf{Trường hợp 2}. $m^2-m-6 \neq 0 \Leftrightarrow \heva{&m\neq -2\\&m\neq 3}$. Khi đó phương trình có nghiệm duy nhất
			\[x=\dfrac{m^5-3m^4+m^2-5m+6}{m^2-m-6}=\dfrac{m^4+m-2}{m+2}.\]
			Kết hợp hai trường hợp trên, ta được $m\neq -2$ là giá trị cần tìm.
		\end{itemize}
	}
\end{bt}
%Câu 2a
\begin{bt}%[Đề thi HK1, Gia Định, TpHCM 2018]%[0D3B2-4]
	Giải phương trình $\sqrt{2x^2-35x+17}=x-17$.
	\loigiai{
			Phương trình đã cho tương đương
			\[\heva{&x-17 \geqslant 0\\&2x^2-35x+17=(x-17)^2}
			\Leftrightarrow \heva{&x \geqslant 17\\&x^2-x-272=0}
			\Leftrightarrow \heva{&x \geqslant 17\\& \hoac{&x=-16\\&x=17}}
			\,\,\Leftrightarrow\,\, x=17.
			\]
			Vậy phương trình đã cho có tập nghiệm $S=\left\{17\right\}$.
		}
\end{bt}
%Câu 2b
\begin{bt}%[Đề thi HK1, Gia Định, TpHCM 2018]%[0D3K2-4]
	Giải phương trình $3\sqrt{x^2-3x+6}=3x-x^2+4$.
	\loigiai{
		Đặt $t=\sqrt{x^2-3x+6}$, điều kiện: $t\geqslant 0$. Phương trình đã cho trở thành
		\[t^2+3t-10=0 \,\Leftrightarrow\, \hoac{&t=2\\&t=-5 \,\, \text{(loại)}.}
		\]
		Với $t=2$, suy ra $\sqrt{x^2-3x+6}=2 \,\Leftrightarrow\, x^2-3x+6=4 \,\Leftrightarrow\, x^2-3x+2=0 \,\Leftrightarrow\, \hoac{&x=1\\&x=2}$.\\
		Vậy phương trình đã cho có tập nghiệm $S=\left\{1;2\right\}$.
	}
\end{bt}
%Câu 2c
\begin{bt}%[Đề thi HK1, Gia Định, TpHCM 2018]%[0D3K2-4]
	Giải phương trình $\sqrt{x-4}+\sqrt{x+4}=\sqrt{2x+6}$.
	\loigiai{
		Điều kiện: $\heva{&x-4\geqslant0\\&x+4\geqslant0\\&2x+6\geqslant0} \Leftrightarrow \heva{&x\geqslant4\\&x\geqslant-4\\&x\geqslant-3} \Leftrightarrow x\geqslant4$ $\,\,(*)$. Ta có
		\begin{eqnarray*}
			\sqrt{x-4}+\sqrt{x+4}=\sqrt{2x+6}
			& \Rightarrow & x-4+x+4+2\sqrt{(x-4)(x+4)}=2x+6 \\
			& \Leftrightarrow & \sqrt{x^2-16}=3 \,\Leftrightarrow\, x^2 = 25 
			\,\Leftrightarrow\, \hoac{&x=5\\&x=-5 \text{\,\,(không thỏa (*)).}}
		\end{eqnarray*}
		Vậy phương trình đã cho có tập nghiệm $S=\left\{5\right\}$.
	}
\end{bt}
%Câu 2d
\begin{bt}%[Đề thi HK1, Gia Định, TpHCM 2018]%[0D3K2-4]
	Giải phương trình $\sqrt{x+4}+\sqrt{5-x}=3+\sqrt{(x+4)(5-x)}$.
	\loigiai{
		Điều kiện: $\heva{&x+4\geqslant0\\&5-x\geqslant0} \Leftrightarrow  -4 \leqslant x\leqslant 5$ $\,\,(*)$.
		Đặt $t=\sqrt{x+4}+\sqrt{5-x}$, $t\geqslant 0$.\\
		Ta có $t^2=x+4+5-x+2\sqrt{(x+4)(5-x)} \,\Leftrightarrow\, \sqrt{(x+4)(5-x)} = \dfrac{t^2-9}{2}$.\\
		Phương trình đã cho trở thành
		\[t=3+ \dfrac{t^2-9}{2} \,\Leftrightarrow\, t^2-2t-3=0 \,\Leftrightarrow\, \hoac{&t=3\\&t=-1 \text{\,\,(loại)}.}
		\]
		Với $t=3$, suy ra $\sqrt{(x+4)(5-x)}=0 \,\Leftrightarrow\, \hoac{&x=-4\\&x=5} \text{\,\,(thỏa (*))}$.\\
		Vậy phương trình đã cho có tập nghiệm $S=\left\{-4;5\right\}$.
	}
\end{bt}

\begin{bt}%[Đề thi HK1, Gia Định, TpHCM 2018]%[0H2Y2-3][0H1B4-3]
	Trong mặt phẳng $Oxy$, cho tam giác $ABC$ có $A(-1;-2)$, $B(-4;2)$, $C(7;4)$.
	\begin{enumerate} 
			\item Chứng minh rằng tam giác $ABC$ vuông tại $A$.
			\item Tìm tọa độ tâm $K$ của đường tròn ngoại tiếp tam giác $ABC$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate} 
			\item Ta có $\vec{AB}=(-3;4)$, $\vec{AC}=(8;6)$.\\
			Vì $\vec{AB}\cdot \vec{AC}=-3\cdot 8 + 4\cdot 6 = 0$ nên $\vec{AB} \perp \vec{AC}$. Suy ra tam giác $ABC$ vuông tại $A$.
			\item Vì tam giác $ABC$ vuông tại $A$ nên tâm $K$ của đường tròn ngoại tiếp tam giác $ABC$ là trung điểm của $BC$. Do đó
			\[\heva{&x_K=\dfrac{x_B+x_C}{2}=\dfrac{3}{2}\\&
				y_K=\dfrac{y_B+y_C}{2}=3} \,\Rightarrow\, K\left(\dfrac{3}{2};3\right).\]
		\end{enumerate}
	}
\end{bt}
%Câu 4C
\begin{bt}%[Đề thi HK1, Gia Định, TpHCM 2018]%[0D3G2-5]
	Tìm tham số $m$ để phương trình $(m-16)x^2+2(m-19)x+m-14=0$ có hai nghiệm $x_1, x_2$ thỏa mãn $6(x_1^2+x_2^2)+2x_1^2\cdot x_2^2-16x_1-16x_2=14$.
	\loigiai{
	\begin{itemize}
		\item \textbf{Điều kiện 1}. Phương trình có hai nghiệm $x_1, x_2$ khi và chỉ khi
		\begin{eqnarray*}
			\heva{&a \neq 0\\& \Delta'\geqslant 0}
			& \Leftrightarrow & 
			\heva{&m-16 \neq 0\\& (m-19)^2-(m-16)(m-14)\geqslant 0} \\
			& \Leftrightarrow & 
			\heva{&m \neq 16\\& -8m+137 \geqslant 0}
			\,\Leftrightarrow\, \heva{&m \neq 16\\& m \leqslant \dfrac{137}{8}.} \text{\quad (1)}
		\end{eqnarray*}
		\item \textbf{Điều kiện 2}. Với điều kiện $(1)$, theo hệ thức Vi-ét, ta có $\heva{&x_1+x_2=\dfrac{-2m+38}{m-16}\\&x_1x_2=\dfrac{m-14}{m-16}}$.\\
		Yêu cầu\vspace*{-9pt}
		\begin{eqnarray*}
			& & 6(x_1^2+x_2^2)+2x_1^2\cdot x_2^2-16x_1-16x_2=14\\
			& \Leftrightarrow & 
			3(x_1+x_2)^2-6x_1x_2+(x_1x_2)^2-8(x_1+x_2)=7\\
			& \Leftrightarrow & 
			3\left(\dfrac{-2m+38}{m-16}\right)^2-6\left(\dfrac{m-14}{m-16}\right)+\left(\dfrac{m-14}{m-16}\right)^2-8\left(\dfrac{-2m+38}{m-16}\right)=7\\
			& \Leftrightarrow & 
			12(m-19)^2-6(m-14)(m-16)+(m-14)^2+16(m-19)(m-16)=7(m-16)^2\\
			& \Leftrightarrow & 
			16m^2-640m+6256=0 \,\Leftrightarrow\, m^2-40m+391=0
			\,\Leftrightarrow\,
			\hoac{&m=23\\&m=17.}
		\end{eqnarray*}
	Đối chiếu với điều kiện $(1)$, ta được $m=17$ là giá trị cần tìm.		
	\end{itemize}
	}
\end{bt}

\begin{bt}%[Đề thi HK1, Gia Định, TpHCM 2018]%[0D3K2-5]
	Tìm tham số $m$ để phương trình $(m-16)x^2+2(m-19)x+m-14=0$ có hai nghiệm $x_1$, $x_2$ thỏa mãn $6(x_1^2+x_2^2)+2x_1^2\cdot x_2^2=78$.
	\loigiai{
		\begin{itemize}
			\item \textbf{Điều kiện 1}. Phương trình có hai nghiệm $x_1, x_2$ khi và chỉ khi
			\begin{eqnarray*}
				\heva{&a \neq 0\\& \Delta'\geqslant 0}
				& \Leftrightarrow & 
				\heva{&m-16 \neq 0\\& (m-19)^2-(m-16)(m-14)\geqslant 0} \\
				& \Leftrightarrow & 
				\heva{&m \neq 16\\& -8m+137 \geqslant 0}
				\,\Leftrightarrow\, \heva{&m \neq 16\\& m \leqslant \dfrac{137}{8}.} \text{\quad (1)}
			\end{eqnarray*}
			\item \textbf{Điều kiện 2}. Với điều kiện $(1)$, theo hệ thức Vi-ét, ta có $\heva{&x_1+x_2=\dfrac{-2m+38}{m-16}\\&x_1x_2=\dfrac{m-14}{m-16}}$.\\
			Yêu cầu\vspace*{-9pt}
			\begin{eqnarray*}
				& & 6(x_1^2+x_2^2)+2x_1^2\cdot x_2^2=78\\
				& \Leftrightarrow & 
				3(x_1+x_2)^2-6x_1x_2+(x_1x_2)^2=39\\
				& \Leftrightarrow & 
				3\left(\dfrac{-2m+38}{m-16}\right)^2-6\left(\dfrac{m-14}{m-16}\right)+\left(\dfrac{m-14}{m-16}\right)^2=39\\
				& \Leftrightarrow & 
				12(m-19)^2-6(m-14)(m-16)+(m-14)^2=39(m-16)^2\\
				& \Leftrightarrow & 
				-32m^2+944m-6800=0 \,\Leftrightarrow\, \hoac{&m=\dfrac{25}{2}\\&m=17.}
			\end{eqnarray*}
			Đối chiếu với điều kiện $(1)$, ta được $m=\dfrac{25}{2}, m=17$ là giá trị cần tìm.		
		\end{itemize}
		
	}
\end{bt}

\begin{bt}%[Đề thi HK1, Gia Định, TpHCM 2018]%[0D3K2-5]
	Tìm tham số $m$ để phương trình $(m-16)x^2+2(m-19)x+m-14=0$ có hai nghiệm $x_1$, $x_2$ thỏa mãn $2(x_1^2+x_2^2)-x_1-x_2=16$.
	\loigiai{
		\begin{itemize}
			\item \textbf{Điều kiện 1}. Phương trình có hai nghiệm $x_1, x_2$ khi và chỉ khi
			\begin{eqnarray*}
				\heva{&a \neq 0\\& \Delta'\geqslant 0}
				& \Leftrightarrow & 
				\heva{&m-16 \neq 0\\& (m-19)^2-(m-16)(m-14)\geqslant 0} \\
				& \Leftrightarrow & 
				\heva{&m \neq 16\\& -8m+137 \geqslant 0}
				\,\Leftrightarrow\, \heva{&m \neq 16\\& m \leqslant \dfrac{137}{8}.} \text{\quad (1)}
			\end{eqnarray*}
			\item \textbf{Điều kiện 2}. Với điều kiện $(1)$, theo hệ thức Vi-ét, ta có $\heva{&x_1+x_2=\dfrac{-2m+38}{m-16}\\&x_1x_2=\dfrac{m-14}{m-16}}$.\\
			Yêu cầu\vspace*{-9pt}
			\begin{eqnarray*}
				& & 2(x_1^2+x_2^2)-x_1-x_2=16\\
				& \Leftrightarrow & 
				2(x_1+x_2)^2-4x_1x_2-(x_1+x_2)=16\\
				& \Leftrightarrow & 
				2\left(\dfrac{-2m+38}{m-16}\right)^2-4\left(\dfrac{m-14}{m-16}\right)-\left(\dfrac{-2m+38}{m-16}\right)=16\\
				& \Leftrightarrow & 
				8(m-19)^2-4(m-14)(m-16)+2(m-19)(m-16)=16(m-16)^2\\
				& \Leftrightarrow & 
				-10m^2+258m-1496=0 \,\Leftrightarrow\, \hoac{&m=\dfrac{44}{5}\\&m=17.}
			\end{eqnarray*}
			Đối chiếu với điều kiện $(1)$, ta được $m=\dfrac{44}{5}, m=17$ là giá trị cần tìm.		
		\end{itemize}		
	}
\end{bt}

\begin{bt}%[Đề thi HK1, Gia Định, TpHCM 2018]%[0H2K2-1][0H2G2-1]
	Cho tam giác $ABC$ có $AB=5$, $AC=8$, $BC=7$. Gọi $M,N,P$ là ba điểm thỏa mãn $2\overrightarrow{MA}+3\overrightarrow{MB}=\overrightarrow{0}$,
	$5\overrightarrow{NA}+3\overrightarrow{NC}=\overrightarrow{0}$,
	$4\overrightarrow{PB}+3\overrightarrow{PC}=\overrightarrow{0}$.
	\begin{enumerate}
		\item Tính $\vec{AB}\cdot \vec{AC}$, $\vec{BA}\cdot \vec{BC}$ và $\vec{AM}\cdot \vec{AN}$.
		\item Tính độ dài các đoạn $MN$, $AP$ và $NP$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Tính $\vec{AB}\cdot \vec{AC}$, $\vec{BA}\cdot \vec{BC}$ và $\vec{AM}\cdot \vec{AN}$.\\[1.5mm]
			$\bullet\,\,$ Ta có ${\vec{BC}}^2=\left(\vec{AC}-\vec{AB}\right)^2 
			\,\Leftrightarrow\, {\vec{BC}}^2={\vec{AC}}^2+{\vec{AB}}^2-2\vec{AC}\cdot \vec{AB}$.\\[1.5mm]
			Suy ra $\vec{AB}\cdot \vec{AC}=\dfrac{AB^2+AC^2-BC^2}{2}=\dfrac{5^2+8^2-7^2}{2}=20$.\\[1.5mm]			
			$\bullet\,\,$  Ta có $\vec{BA}\cdot \vec{BC}=
			\vec{BA}\cdot \left(\vec{AC}-\vec{AB}\right)
			={\vec{AB}}^2-\vec{AB}\cdot \vec{AC}=5^2-20=5$.\\[1.5mm]			
			$\bullet\,\,$  Ta có $2\vec{MA}+3\vec{MB}=\vec{0} \,\Leftrightarrow\, 2\vec{MA}+3\left(\vec{MA}+\vec{AB}\right)=\vec{0} 
			\,\Leftrightarrow\, \vec{AM}=\dfrac{3}{5}\vec{AB}$;\\[1.5mm]
			và $5\vec{NA}+3\vec{NC}=\vec{0} \,\Leftrightarrow\, 5\vec{NA}+3\left(\vec{NA}+\vec{AC}\right)=\vec{0} 
			\,\Leftrightarrow\, \vec{AN}=\dfrac{3}{8}\vec{AC}$.\\[1.5mm]
			Suy ra $\vec{AM}\cdot \vec{AN}=\dfrac{3}{5}\vec{AB} \cdot \dfrac{3}{8}\vec{AC} =\dfrac{9}{40} \cdot 20 = \dfrac{9}{2}$.\\[1.5mm]
			Vậy $\vec{AB}\cdot \vec{AC}=20$; $\vec{BA}\cdot \vec{BC}=5$ và $\vec{AM}\cdot \vec{AN}=\dfrac{9}{2}$.
						
			\item Tính độ dài các đoạn $MN$, $AP$ và $NP$.
			\immini{
				$\bullet\,\,$ Ta có 
				\begin{align*}
				MN^2&={\vec{MN}}^2=\left(\vec{AN}-\vec{AM}\right)^2\\
				&={\vec{AN}}^2+{\vec{AM}}^2-2\vec{AN}\cdot \vec{AM}\\
				&=\dfrac{9}{64}{\vec{AC}}^2+\dfrac{9}{25}{\vec{AB}}^2-2\vec{AN}\cdot \vec{AM}\\
				&=\dfrac{9}{64}\cdot 8^2 + \dfrac{9}{25}\cdot 5^2-2\cdot \dfrac{9}{2}=9.
				\end{align*}
				$\quad \Rightarrow MN = 3.$
			}{
				\begin{tikzpicture}[scale=.8,x=1cm,y=1cm]
				\tkzDefPoints{-3/0/B, -1/4.5/A, 4/0/C}
				\tkzDefMidPoint(A,B) \tkzGetPoint{H}
				\coordinate (M) at ($(A)!3/5!(B)$);
				\coordinate (N) at ($(A)!3/8!(C)$);
				\coordinate (P) at ($(B)!3/7!(C)$);
				\tkzDrawSegments(A,B B,C C,A M,N N,P P,M)
				\tkzDrawPoints(A,B,C,M,N,P)
				\tkzLabelPoints[above](A)
				\tkzLabelPoints[below](P)
				\tkzLabelPoints[left](B,M) 	
				\tkzLabelPoints[right](C,N)
				\end{tikzpicture}			
			}
			$\bullet\,\,$ Ta có $4\vec{PB}+3\vec{PC}=\vec{0} \,\Leftrightarrow\, 4\left(\vec{AB}-\vec{AP}\right)+3\left(\vec{AC}-\vec{AP}\right)=\vec{0}
			\,\Leftrightarrow\, \vec{AP}=\dfrac{4}{7}\vec{AB}+\dfrac{3}{7}\vec{AC}$.\\
			Suy ra ${\vec{AP}}^2=\left(\dfrac{4}{7}\vec{AB}+\dfrac{3}{7}\vec{AC}\right)^2
			=\left(\dfrac{4}{7}\right)^2\cdot 5^2
			+\left(\dfrac{3}{7}\right)^2\cdot 8^2
			+2\cdot\dfrac{4}{7}\cdot \dfrac{3}{7} \cdot 20= \dfrac{208}{7}$\\
			$\Rightarrow AP = \dfrac{4\sqrt{91}}{7}.$
			\\
			$\bullet\,\,$ Ta có $4\vec{PB}+3\vec{PC}=\vec{0} \,\Leftrightarrow\, \vec{BP}=\dfrac{3}{7}\vec{BC}$.\\
			Phân tích 
			$\vec{NP}=\vec{NA}+\vec{AB}+\vec{BP}=-\dfrac{3}{8}\vec{AC}+\vec{AB}+\dfrac{3}{7}\vec{BC}=\dfrac{3}{56}\vec{AC}+\dfrac{4}{7}\vec{AB}.$\\
			Suy ra ${\vec{NP}}^2=\left(\dfrac{3}{56}\vec{AC}+\dfrac{4}{7}\vec{AB}\right)^2
			=\left(\dfrac{3}{56}\right)^2\cdot 8^2
			+\left(\dfrac{4}{7}\right)^2\cdot 5^2
			+2\cdot\dfrac{3}{56}\cdot \dfrac{4}{7} \cdot 20= \dfrac{67}{7}$\\
			$\Rightarrow NP = \dfrac{\sqrt{469}}{7}.$\\[1.5mm]
			Vậy $MN = 3$; $AP=\dfrac{4\sqrt{91}}{7}$ và $NP=\dfrac{\sqrt{469}}{7}$.
		\end{enumerate}
	}
\end{bt}
