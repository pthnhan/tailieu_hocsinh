\begin{ex}%[2D2B5-1]%[2-TT-43-SoBacGiang]%[DoVuMinhThang]
Cho $x$, $y$ là các số thực dương thỏa mãn  $\dfrac{\log_2x}{\log_2(xy)+1}=\dfrac{\log_2y}{\log_2(xy)-1}=\log_2x+\log_2y$. Tính $x+y$.
\choice{$x+y=2$}
{\True $x+y=2$ hoặc $x+y= \sqrt[4]{8}+\dfrac{1}{\sqrt[4]{2}}$}
{$x+y=2+\dfrac{1}{\sqrt[4]{2}}$}
{$x+y=\dfrac{1}{2}$ hoặc $x+y=2$}
\loigiai{Điều kiện xác định $x>0, y>0, xy\ne \dfrac{1}{2}$ và $xy\ne 2$.
\begin{align*}
&\dfrac{\log_2x}{\log_2(xy)+1}=\log_2x+\log_2y \Leftrightarrow \log_2x=\log_2^2(xy)+\log_2(xy)\\
&\dfrac{\log_2y}{\log_2(xy)-1}=\log_2x+\log_2y \Leftrightarrow \log_2y=\log_2^2(xy)-\log_2(xy)
\end{align*}
Cộng hai biểu thức trên vế theo vế ta được 
\[
\log_2x+\log_2y=2\log_2^2(xy)\Leftrightarrow \log_2(xy)=2\log_2^2(xy)\Leftrightarrow \log_2(xy)=0 \vee \log_2(xy)=\dfrac{1}{2}.
\]
Với $\log_2(xy)=0\Leftrightarrow xy=1$, ta có $\log_2x=0\Leftrightarrow x=1 \Rightarrow y=1$.\\
Với $\log_2(xy)=\dfrac{1}{2} \Leftrightarrow xy=\sqrt{2}$, ta có $\log_2x=\dfrac{1}{4}+\dfrac{1}{2}=\dfrac{3}{4}\Leftrightarrow x=\sqrt[4]{8}\Rightarrow y=\dfrac{1}{\sqrt[4]{2}}$.
}
\end{ex}
