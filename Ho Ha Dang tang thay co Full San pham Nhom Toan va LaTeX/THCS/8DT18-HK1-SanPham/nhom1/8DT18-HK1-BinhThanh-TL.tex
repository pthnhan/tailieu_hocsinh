			\item $10x^2-x(x+2)+8x+1=0\Leftrightarrow 9x^2+6x+1=0\Leftrightarrow (3x+1)^2=0\Leftrightarrow 3x+1=0\Leftrightarrow x=-\dfrac{1}{3}.$
		\end{enumerate}
	}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Bình Thạnh, TP.Hồ Chí Minh]%[Bùi Mạnh Tiến, dự án 8DT18-HK1]%[8D2B5]%[8D2B6]
	Thực hiện phép tính sau
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\dfrac{x^2+6x-5}{x+3}+\dfrac{14}{x+3}.$
			\item $\dfrac{x}{2x-2}+\dfrac{x-2}{x^2-1}-\dfrac{5}{2x+2}.$
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item $\dfrac{x^2+6x-5}{x+3}+\dfrac{14}{x+3}=\dfrac{x^2+6x+9}{x+3}=\dfrac{(x+3)^2}{x+3}=x+3.$
			\item \begin{align*}
				\dfrac{x}{2x-2}+\dfrac{x-2}{x^2-1}-\dfrac{5}{2x+2}&=\dfrac{x}{2(x-1)}+\dfrac{x-2}{(x-1)(x+1)}-\dfrac{5}{2(x+1)}\\
				&=\dfrac{x(x+1)}{2(x-1)(x+1)}+\dfrac{2(x-2)}{2(x-1)(x+1)}-\dfrac{5(x-1)}{2(x-1)(x+1)}\\
				&=\dfrac{x(x+1)+2(x-2)-5(x-1)}{2(x-1)(x+1)} =\dfrac{x^2-2x+1}{2(x-1)(x+1)}\\
				&=\dfrac{(x-1)^2}{2(x-1)(x+1)}=\dfrac{x-1}{2(x+1)}.
			\end{align*}
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Bình Thạnh, TP.Hồ Chí Minh]%[Bùi Mạnh Tiến, dự án 8DT18-HK1]%[7H3B3]
	Bốn nhà máy được xây dựng tại bốn địa điểm $A$, $B$, $C$, $D$ là $4$ đỉnh của một tứ giác. Hãy tìm một điểm $E$ nằm trong tứ giác $ABCD$ để xây dựng trung tâm điều hành sao cho tổng chiều dài $EA$, $EB$, $EC$ và $ED$ là nhỏ nhất.
	\loigiai{
		\immini{
			Với $E$ là một điểm bất kì thì
			$$EA+EB+EC+ED=(EA+EC)+(EB+ED)\ge AC+BD.$$
			Dấu \lq\lq = \rq\rq\text{ }xảy ra khi $E$ nằm trên đoạn $AC$ và $E$ nằm trên đoạn $BD$ hay $E$ là giao điểm của $AC$ và $BD$.
		}

				\tkzDrawPoints[fill = black](A,B,C,D,F,E)

				\tkzDrawSegments(A,B B,C C,D D,A B,D A,C E,A E,B E,C E,D)
				\draw (A) node[above] {$A$} (B) node[left] {$B$} (C) node[below] {$C$} (D) node[right] {$D$} ($(E)+(0.2,-0.2)$) node {$E$};

		}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Bình Thạnh, TP.Hồ Chí Minh]%[Bùi Mạnh Tiến, dự án 8DT18-HK1]%[8H1B34]%[8H1B9]%[8H1B11]
	Cho tam giác $ABC$ vuông tại $A$ $(AB<AC)$. Gọi $M$ là trung điểm của $BC$, $E$ là trung điểm của $AC$.
	\begin{enumerate}
		\item Chứng minh tứ giác $ABME$ là hình thang vuông.
		\item Gọi $D$ là điểm đối xứng của $A$ qua $M$. Chứng minh tứ giác $ABDC$ là hình chữ nhật.
		\item Đường thẳng qua $B$ và song song với $AD$ cắt $EM$ tại $K$. Chứng minh tứ giác $BKDM$ là hình thoi.
		\item Gọi $I$ là trung điểm $MD$. Chứng minh $\triangle IKE$ cân. 
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale = 1.2]
				\clip (-0.5,-2.5) rectangle (5.5,2.5);
				\tkzDefPoints{0/0/B, 1/2/A, 5/0/C}
				\tkzDefPointWith[linear,K=0.5](C,B)\tkzGetPoint{M}
				\tkzDefPointWith[linear,K=0.5](A,C)\tkzGetPoint{E}
				\tkzDefPointWith[linear,K=2](A,M)\tkzGetPoint{D}
				\tkzDefPointWith[linear,K=0.5](M,D)\tkzGetPoint{I}
				\tkzDefLine[parallel=through B](A,D) \tkzGetPoint{b}
				\tkzInterLL(B,b)(E,M)    \tkzGetPoint{K}
				\tkzInterLL(B,D)(E,K)    \tkzGetPoint{O}
				

				
				\tkzDrawSegments(A,B B,C C,A E,K A,D D,B D,C D,K I,K B,K I,O I,E)
				\tkzDrawPoints[fill=black](A,B,C,D,E,M,I,O,K)
				\draw (A) node[above] {$A$} (B) node[left] {$B$} (C) node[right] {$C$} ($(E)+(0.2,0.2)$) node {$E$} (D) node[below] {$D$} (M) node[below left] {$M$} (I) node[right] {$I$} (K) node[below] {$K$} ($(O)+(0.07,0)$) node[below] {$O$};
				
				\tkzMarkRightAngle(B,A,C)
				\tkzMarkRightAngle(M,E,A)
				\tkzMarkRightAngle(M,O,D)
				\tkzMarkSegments[mark=|](A,E E,C)
				\tkzMarkSegments[mark=||](B,M M,C)
				\tkzMarkSegments[mark=||](A,M M,I)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Vì $M$, $E$ lần lượt là trung điểm của $BC$ và $AC$ nên $EM$ là đường trung bình của tam giác $ABC$, suy ra $EM\parallel AB$ mà $AB\perp AC\Rightarrow ME\perp AC$.\\
			Tứ giác $ABME$ có $AB\parallel ME$ nên $ABME$ là hình thang, đồng thời $\widehat{MEA}=\widehat{EAB}=90^\circ$ nên $ABME$ là hình thang vuông tại $A$ và $E$.
			\item Tứ giác $ABDC$ có hai đường chéo $BC$, $AD$ cắt nhau tại trung điểm của mỗi đường suy ra $ABDC$ là hình bình hành, mặt khác $\widehat{BAC}=90^\circ $ nên $ABDC$ là hình chữ nhật.
			\item Gọi $O$ là giao điểm của $ME$ và $BD$.\\
			Vì $\begin{cases}
				ME\perp AC\\
				AC\parallel BD
			\end{cases}\Rightarrow ME\perp BD\Rightarrow MO\perp BD$, mà $MB=MD\Rightarrow MK$ là trung trực của đoạn $BD\Rightarrow KB=KD$.\\
			Xét $\triangle BOK$ và $\triangle DOM$ có
			$$\begin{cases}
			\widehat{MOD}=\widehat{KOB}=90^\circ\\
			OB=OD\text{ }(O \text{ nằm trên trung trực của }BD)\\
			\widehat{ODM}=\widehat{OBK} \text{ (so le trong)}
			\end{cases}\Rightarrow \triangle BOK=\triangle DOM \text{ (g.c.g)}\Rightarrow MD=BK.$$
			Từ đó suy ra $MB=MD=KB=KD$ nên $BKDM$ là hình thoi.
			\item Tam giác $ODM$ vuông tại $O$ có $I$ là trung điểm của $MD$ nên $IO=IM\Rightarrow \triangle IMO$ cân tại $I$, suy ra $\widehat{IMO}=\widehat{IOM}\Rightarrow \widehat{IOK}=\widehat{IME}.$\\
			Vì $BKDM$ là hình thoi và $BDCA$ là hình chữ nhật nên $KO=OM=ME$.\\
			Xét $\triangle IOK$ và $\triangle IME$ có
			$$\begin{cases}
				IM=IO\\
				\widehat{IOK}=\widehat{IME}\\
				OK=ME
			\end{cases}\Rightarrow \triangle IOK=\triangle IME\text{ (c.g.c)}\Rightarrow IK=IE\Rightarrow \triangle IKE\text{ cân tại }I.$$
			
		\end{enumerate}
	}
\end{bt}
