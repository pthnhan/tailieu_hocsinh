\begin{bt}%[Thi GHK, Trần Khai Nguyên - Hồ Chí Minh, 2018]%[Kiều Ngân, EX-DT-10&11]%[1D1B2-1]
	Tìm giá trị của tham số $m$ để tập xác định của hàm số $y=\dfrac{\sin 2x}{2\cos 2x -m}$ là $\mathbb{R}$.
	\loigiai{
		Hàm số $y=\dfrac{\sin 2x}{2\cos 2x -m}$ có tập xác định là $\mathbb{R}$\\
		 $\Leftrightarrow 2\cos 2x -m \ne 0, \forall x\in \mathbb{R}$ $\Leftrightarrow \cos 2x \ne \dfrac{m}{2}, \forall x\in \mathbb{R}$\\
		 $\Leftrightarrow$ Phương trình $\cos 2x=\dfrac{m}{2}$ vô nghiệm $\Leftrightarrow \hoac{&\dfrac{m}{2}<-1\\&\dfrac{m}{2}>1} \Leftrightarrow \hoac{&m<-2\\&m>2.}$
	}
\end{bt}

\begin{bt}%[Thi GHK, Trần Khai Nguyên - Hồ Chí Minh, 2018]%[Kiều Ngân, EX-DT-10&11]%[1D1B2-1]
	Tìm tất cả các nghiệm thuộc nửa khoảng $\left(-\pi; \dfrac{3\pi}{2}\right]$ của phương trình $\sin \dfrac{2x}{3}-\cos x =0$.
	\loigiai{
		Ta có
		\begin{eqnarray*}
			& & \sin \dfrac{2x}{3}-\cos x =0
			\Leftrightarrow \sin \dfrac{2x}{3} =\cos x
			\Leftrightarrow \cos x =\cos \left(\dfrac{\pi}{2}-\dfrac{2x}{3}\right)\\
			& \Leftrightarrow & \hoac{&x=\dfrac{\pi}{2}-\dfrac{2x}{3}+k2\pi\\&x=-\dfrac{\pi}{2}+\dfrac{2x}{3}+k2\pi}
			\Leftrightarrow  \hoac{&x=\dfrac{3\pi}{10}+\dfrac{k6\pi}{5}\\&x=-\dfrac{3\pi}{2}+k6\pi} (k\in \mathbb{Z}).
		\end{eqnarray*}
		\begin{itemize}
			\item $x=\dfrac{3\pi}{10}+\dfrac{k6\pi}{5} \in \left(-\pi;\dfrac{3\pi}{2}\right] \Leftrightarrow k\in \{-1;0;1\} \Leftrightarrow x\in \left\{ -\dfrac{9\pi}{10};\dfrac{3\pi}{10}; \dfrac{3\pi}{2} \right\}$.
			\item $x=-\dfrac{3\pi}{2}+k6\pi \in \left(-\pi;\dfrac{3\pi}{2}\right] \Leftrightarrow k\in \varnothing$.
		\end{itemize}
		Vậy tập nghiệm $S=\left\{ -\dfrac{9\pi}{10};\dfrac{3\pi}{10}; \dfrac{3\pi}{2} \right\}$.
	}
\end{bt}

\begin{bt}%[Thi GHK, Trần Khai Nguyên - Hồ Chí Minh, 2018]%[Kiều Ngân, EX-DT-10&11]%[1D1B3-1]
	Giải phương trình $\cos^2 2x -8\sin^2 x +7=0$.
	\loigiai{
		Ta có
		\begin{eqnarray*}
			& & \cos^2 2x -8\sin^2 x +7=0
			\Leftrightarrow \cos^2 2x -4(1-\cos 2x) +7=0\\
			& \Leftrightarrow & \cos^2 2x +4\cos 2x +3=0
			\Leftrightarrow \hoac{&\cos 2x=-1\\&\cos 2x=-3 \text{ (vô nghiệm)}}\\
			& \Leftrightarrow & 2x=\pi+k2\pi 
			\Leftrightarrow x=\dfrac{\pi}{2}+k\pi \, (k\in \mathbb{Z}).
		\end{eqnarray*}
		Vậy tập nghiệm của phương trình là $S=\left\{\dfrac{\pi}{2}+k\pi \, (k\in \mathbb{Z}) \right\}$.
	}
\end{bt}

\begin{bt}%[Thi GHK, Trần Khai Nguyên - Hồ Chí Minh, 2018]%[Kiều Ngân, EX-DT-10&11]%[1D2B2-2]
	Với các chữ số $0$, $1$, $2$, $3$, $4$, $5$, $6$ có thể lập được bao nhiêu số tự nhiên thỏa yêu cầu sau
	\begin{enumerate}
		\item Gồm năm chữ số khác nhau, trong đó có mặt chữ số $5$.
		\item Gồm bốn chữ số khác nhau, sao cho mỗi chữ số đều lớn hơn chữ số đứng sau nó (trừ chữ số hàng đơn vị).
	\end{enumerate}
	\loigiai{
		Đặt $X=\{0;1;2;3;4;5;6\}$.
		\begin{enumerate}
			\item Gọi số gồm năm chữ số khác nhau là $\overline{abcde}$ với $\heva{&a, b, c, d, e \in X\\& a\ne 0\\& a, b, c, d, e \text{ phân biệt}}$.\\
			\textbf{Cách 1:}
			\begin{itemize}
				\item Chọn $a\in \{1;2;3;4;5;6\}$: có $6$ cách chọn.
				\item Chọn $4$ chữ số từ $6$ chữ số còn lại điền vào các vị trí $b$, $c$, $d$, $e$ có $\mathrm{A}_6^4=360$ cách.
			\end{itemize}
			Do đó có $6\cdot 360=2160$ số có năm chữ số khác nhau. Bây giờ ta đếm số lượng số có năm chữ số khác nhau không có mặt chữ số $5$.
			\begin{itemize}
				\item Chọn $a\in \{1;2;3;4;6\}$: có $5$ cách chọn.
				\item Chọn $4$ chữ số từ $5$ chữ số còn lại điền vào các vị trí $b$, $c$, $d$, $e$ có $\mathrm{A}_5^4=120$.\\
				Suy ra có $5\cdot 120=600$ số không có chữ số $5$.
			\end{itemize}
			Vậy có $2160-600=1560$ số thỏa yêu cầu bài toán.\\
			\textbf{Cách 2:}
				\begin{itemize}
					\item Trường hợp $a=5$ thì chọn $b,c,d,e \in \{0;1;2;3;4;6\}$ có $\mathrm{A}_6^4=360$ cách chọn.
					\item Trường hợp $a\ne 5$, chọn $a$ có $5$ cách, ta có $4$ cách chọn vị trí cho chữ số $5$. Chọn tiếp $3$ số trong $5$ số còn lại để điền vào $3$ vị trí còn lại ta có $\mathrm{A}_5^3=60$.\\
					Do đó có $360+5\cdot 4 \cdot 60=1560$ số thỏa yêu cầu bài toán. 
				\end{itemize}
			\item Chọn $4$ chữ số khác nhau từ $X$ ta có $\mathrm{C}_7^4$ cách.\\
			Mỗi cách chọn $4$ chữ số đó ta đều có duy nhất $1$ cách sắp xếp giảm dần (các chữ số đứng trước đều lớn hơn chữ số đứng sau nó).\\
			Vậy có $\mathrm{C}_7^4\cdot 1 =35$ số thỏa yêu cầu bài toán.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Thi GHK, Trần Khai Nguyên - Hồ Chí Minh, 2018]%[Kiều Ngân, EX-DT-10&11]%[1D2B2-1]
	Bạn An có $11$ quyển sách Toán, $7$ quyển sách Văn, $5$ quyển sách Anh. Tất cả các quyển sách đều khác nhau. Bạn An cần chọn một ít sách để tặng bạn Bình.
	\begin{enumerate}
		\item Hỏi bạn An có bao nhiêu cách chọn $4$ quyển sách để tặng?
		\item Hỏi bạn An có bao nhiêu cách chọn $4$ quyển sách để tặng nếu trong đó có ít nhất $2$ quyển sách Toán?
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Tổng số sách bạn An có là $11+7+5=23$ quyển. Số cách chọn $4$ quyển từ $23$ quyển để tặng là $\mathrm{C}_{23}^4=8855$ cách.
			\item Để chọn được $4$ quyển trong đó có ít nhất $2$ quyển sách Toán thì ta có các trường hợp sau
			\begin{itemize}
				\item TH1: $2$ quyển Toán và $2$ quyển Văn, Anh tùy ý có $\mathrm{C}_{11}^2 \cdot \mathrm{C}_{12}^2 =3630$ cách.
				\item TH2: $3$ quyển Toán và $1$ quyển Văn, Anh tùy ý có $\mathrm{C}_{11}^3\cdot \mathrm{C}_{12}^1=1980$ cách.
				\item TH3: $4$ quyển Toán có $\mathrm{C}_{11}^4=330$ cách.
			\end{itemize}
			Vậy có $3630+1980+330=5940$ cách thỏa yêu cầu bài toán.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Thi GHK, Trần Khai Nguyên - Hồ Chí Minh, 2018]%[Kiều Ngân, EX-DT-10&11]%[1H1B2-2]
	Trong mặt phẳng $Oxy$, cho đường thẳng $d\colon 2x-3y+4=0$. Viết phương trình đường thẳng $d'$ là ảnh của đường thẳng $d$ qua phép tịnh tiến theo $\overrightarrow{v}=(-1;2)$.
	\loigiai{
		Ta có $d'$ là ảnh của $d$ qua phép tịnh tiến theo $\overrightarrow{v}$ nên $d'\parallel d$ hay $d'\equiv d$. Suy ra phương trình $d'\colon 2x-3y+c=0$.\\
		Chọn $M(-2;0) \in d$.\\
		Gọi $M'(x_{M'};y_{M'})$ là ảnh của $M$ qua phép tịnh tiến $\mathrm{T}_{\overrightarrow{v}}$\\ $\Leftrightarrow \heva{&x_{M'}=x_M-1\\&y_{M'}=y_M+2} \Leftrightarrow \heva{&x_{M'}=-3\\&y_{M'}=2} \Rightarrow M'(-3;2)$.\\
		$M'(-3;2)\in d' \Rightarrow 2\cdot (-3)-3\cdot 2+c=0 \Leftrightarrow c=12$.\\
		Vậy $d'\colon 2x-3y+12=0$.
	}
\end{bt}

\begin{bt}%[Thi GHK, Trần Khai Nguyên - Hồ Chí Minh, 2018]%[Kiều Ngân, EX-DT-10&11]%[1H1B7-2]
	Trong mặt phẳng $Oxy$, cho đường tròn $(C) \colon (x-1)^2+(y-1)^2=4$. Tìm ảnh của đường tròn $(C)$ qua phép vị tự tâm $H(-1;2)$ tỉ số $k=3$.
	\loigiai{
		Đường tròn $(C)$ có tâm $I(1;1)$, bán kính $R=2$.\\
		Gọi $(C')$ là ảnh của $(C)$ qua phép vị tự $\mathrm{V}_{(H,3)}$ nên $(C')$ có tâm $I'$ với $I'=\mathrm{V}_{(H,3)}(I)$ và bán kính $R'=3\cdot R=6$.\\
		Ta có $\mathrm{V}_{(H,3)}(I)=I'$ $\Leftrightarrow \overrightarrow{HI'}=3\overrightarrow{HI}
		\Leftrightarrow \heva{&x_{I'}-x_H=3(x_I-x_H)\\&y_{I'}-y_H=3(y_I-y_H)}
		\Leftrightarrow \heva{&x_{I'}+1=3(1+1)\\&y_{I'}-2=3(1-2)}$\\
		$\Leftrightarrow \heva{&x_{I'}=5\\&y_{I'}=-1}
		\Rightarrow I'(5;-1)$.
		Vậy $(C')\colon (x-5)^2+(y+1)^2=36$. 
	}
\end{bt}
