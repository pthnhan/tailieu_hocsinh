\noindent\textbf{I. PHẦN TRẮC NGHIỆM}
\begin{ex}%[HK2 (2016-2017), THPT Nho Quan C, Ninh Bình][0D4B5]
	Nghiệm của bất phương trình $\vert 2x+1\vert>x+1$ là
	\choice
	{$x<-\dfrac{2}{3}$}
	{$-\dfrac{2}{3}<x<0$}
	{\True $x>0$ hoặc $x<-\dfrac{2}{3}$}
	{$x>0$}
	\loigiai{
	$\vert 2x+1\vert>x+1\Leftrightarrow \hoac{&x+1<0\\&\heva{&x+1\geq 0\\&(2x+1)^2>(x+1)^2}}\Leftrightarrow\hoac{&x<-1\\&\heva{&x\geq -1\\&3x^2+2x>0}}\Leftrightarrow\hoac{&x<-1\\&\heva{&x\geq -1\\&\hoac{&x>0\\&x<-\dfrac{2}{3}}}}\Leftrightarrow \hoac{&x>0\\&x<-\dfrac{2}{3}.}$\\
	Vậy nghiệm của bất phương trình đã cho là $x>0$ hoặc $x<-\dfrac{2}{3}$.
	}
\end{ex}

\begin{ex}%[HK2 (2016-2017), THPT Nho Quan C, Ninh Bình][0D4B5]
Cho biểu thức $f(x)=\dfrac{2x+3}{4x^2-2x-12}$. Mệnh đề nào dưới đây là \textbf{sai}?
\choice
{$f(x)>0,\forall x\in (2;+\infty)$}
{$f(x)\neq 0, \forall x\neq 2,x\neq -\dfrac{3}{2}$}
{$f(x)<0,\forall x<-\dfrac{3}{2}$}
{\True $f(x)<0,\forall x<2$}
\loigiai{
Ta có bảng xét dấu của $f(x)$ như sau:
\begin{center}
\begin{tikzpicture}
\tkzTabInit[lgt=3,espcl=3]{$x$/1,$2x+3$/1,$4x^2-2x-12$/1,$f(x)$/1}{$-\infty$,$-\dfrac{3}{2}$,$2$,$+\infty$}%
\tkzTabLine{,-,z,+,t,+,}
\tkzTabLine{,+,z,-,z,+,}
\tkzTabLine{,-,z,-,d,+,}
\end{tikzpicture}
\end{center}
Từ bảng xét dấu của $f(x)$ suy ra $f\left(-\dfrac{3}{2}\right)=0$.\\
Vậy trong $4$ khẳng định trên, khẳng định sai là $f(x)<0,\forall x<2$.
}
\end{ex}

\begin{ex}%[HK2 (2016-2017), THPT Nho Quan C, Ninh Bình][0D6B2]
Cho $\sin a=\dfrac{1}{3}$ với $\dfrac{\pi}{2}<a<\pi$. Tính $\cos a$.
\choice
{$\cos a=\dfrac{2\sqrt{2}}{3}$}
{\True $\cos a=-\dfrac{2\sqrt{2}}{3}$}
{$\cos a=\pm\dfrac{2\sqrt{2}}{3}$}
{$\cos a=\dfrac{8}{9}$}
\loigiai{
Ta có $\cos^2a=1-\sin^2a=1-\dfrac{1}{9}=\dfrac{8}{9}$.\\
Suy ra $\vert \cos a\vert=\dfrac{2\sqrt{2}}{3}$.\\
Vì $\dfrac{\pi}{2}<a<\pi$ nên $\cos a<0$.\\
Vậy $\cos a=-\dfrac{2\sqrt{2}}{3}$.
}
\end{ex}

\begin{ex}%[HK2 (2016-2017), THPT Nho Quan C, Ninh Bình][0H3Y1]
Cho đường thẳng $d: 3x-y+1=0$. Véc-tơ chỉ phương của đường thẳng $d$ là 
\choice
{\True $\vec{u}=(1;3)$}
{$\vec{u}=(3;1)$}
{$\vec{u}=(3;-1)$}
{$\vec{u}=(-1;3)$}
\loigiai{
Đường thẳng $d$ có phương trình tổng quát là $3x-y+1=0$ nên có véc-tơ pháp tuyến là $\vec{n}=(3;-1)$.\\
Suy ra $d$ có véc-tơ chỉ phương là $\vec{u}=(1;3)$.
}
\end{ex}

\begin{ex}%[HK2 (2016-2017), THPT Nho Quan C, Ninh Bình][0H3B1]
Phương trình tham số của đường thẳng đi qua điểm $I(-1;2)$ và vuông góc với đường thẳng có phương trình $2x-y+4=0$ là
\choice
{$\heva{&x=1+2t\\&y=2-t}$}
{$\heva{&x=t\\&y=4+2t}$}
{\True $\heva{&x=-1+2t\\&y=2-t}$}
{$\heva{&x=-1+2t\\&y=2+t}$}
\loigiai{
Gọi $\Delta:2x-y+4=0$. Ta có véc-tơ pháp tuyến của $\Delta$ là $\vec{n}_{\Delta}=(2;-1)$.\\
Gọi $d$ là đường thẳng đi qua $I(-1;2)$ và vuông góc với đường thẳng $\Delta$. Khi đó $d$ nhận véc-tơ $\vec{n}_{\Delta}=(2;-1)$ làm véc-tơ chỉ phương.\\
Vậy phương trình tham số của $d$ là $\heva{&x=-1+2t\\&y=2-t}$. 
}
\end{ex}

\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[HK2 (2016-2017), THPT Nho Quan C, Ninh Bình][0D4Y2][0D4K5]
Giải các bất phương trình sau:
\begin{multicols}{2}
		\begin{enumerate}
			\item $-2x-4>0$;
			\item $\sqrt{2x-1}+2>x$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item Ta có $-2x-4>0\Leftrightarrow x<-2$.\\
			Vậy tập nghiệm của bất phương trình là $(-\infty;-2)$.
			\item Điều kiện $x\geq\dfrac{1}{2}$.\\
			Với điều kiện trên, ta có
			\begin{align*}
			\sqrt{2x-1}+2>x&\Leftrightarrow\hoac{&x-2<0\\&\heva{&x-2\geq 0\\&2x-1>(x-2)^2}}\Leftrightarrow\hoac{&x<2\\&\heva{&x\geq 2\\&x^2-6x+5<0}}\\
			&\Leftrightarrow\hoac{&x<2\\&\heva{&x\geq 2\\&1<x<5}}\Leftrightarrow x<5.
			 \end{align*} 
			 Kết hợp với điều kiện (*), suy ra bất phương trình đã cho có tập nghiệm là $T=\left[\dfrac{1}{2};5\right)$.			
		\end{enumerate}
}
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Nho Quan C, Ninh Bình][0D6K3]
Cho $90^{\circ}<\alpha<180^{\circ}$ và $\sin\alpha=\dfrac{3}{4}$. Tính $\cos\alpha$, $\tan\alpha$, $\cot\alpha$, $\cos 3\alpha$ và $\tan 3\alpha$.
\loigiai{
Ta có
 $\cos^2\alpha=1-\sin^2\alpha=1-\dfrac{9}{16}=\dfrac{7}{16}\Rightarrow |\cos\alpha|=\dfrac{\sqrt{7}}{4}$.\\
Do $90^{\circ}<\alpha<180^{\circ}\Rightarrow \cos\alpha=-\dfrac{\sqrt{7}}{4}$.\\
$\tan\alpha=\dfrac{\sin\alpha}{\cos\alpha}=\dfrac{\dfrac{3}{4}}{-\dfrac{\sqrt{7}}{4}}=-\dfrac{3\sqrt{7}}{7},\cot\alpha=\dfrac{1}{\tan\alpha}=-\dfrac{7}{3\sqrt{7}}=-\dfrac{\sqrt{7}}{3}$.\\
$\cos 3\alpha=4\cos^3\alpha-3\cos\alpha=4\cdot\left(\dfrac{-\sqrt{7}}{4}\right)^3-3\cdot\left(\dfrac{-\sqrt{7}}{4}\right)=\dfrac{5\sqrt{7}}{16}$.\\
$\tan 3\alpha=\dfrac{\sin 3\alpha}{\cos 3\alpha}=\dfrac{3\sin\alpha-4\sin^3\alpha}{\dfrac{5\sqrt{7}}{16}}=\dfrac{3\cdot\dfrac{3}{4}-4\cdot\left(\dfrac{3}{4}\right)^3}{\dfrac{5\sqrt{7}}{16}}=\dfrac{9\sqrt{7}}{35}$.
}
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Nho Quan C, Ninh Bình][0H3B1][0H3K2]
Trong mặt phẳng tọa độ $Oxy$ cho $A(-1;2), B(3;1)$ và đường thẳng $\Delta:\heva{&x=1+t\\&y=2+t}, t\in\mathbb{R}$.
	\begin{enumerate}
		\item Viết phương trình tham số của đường thẳng $AB$.
		\item Viết phương trình đường tròn tâm $A$ và tiếp xúc với đường thẳng $\Delta$.
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Đường thẳng $AB$ đi qua $A(-1;2)$, có véc-tơ chỉ phương $\vec{AB}=(4;-1)$, có phương trình tham số là $\heva{&x=-1+4t\\&y=2-t}, t\in\mathbb{R}$.
		\item Đường thẳng $\Delta$ qua $M(1;2)$ có véc-tơ chỉ phương $\vec{u}=(1;1)$. Suy ra $\Delta$ có véc-tơ pháp tuyến $\vec{n}_{\Delta}=(1;-1)$.\\
		Do đó phương trình tổng quát của $\Delta$ là $1(x-1)-1(y-2)=0\Leftrightarrow x-y+1=0$.\\
		Đương tròn tâm $A$ tiếp xúc với $\Delta\Leftrightarrow$ bán kính của nó $R=d(A,\Delta)=\dfrac{|-1-2+1|}{\sqrt{1^2+(-1)^2}}=\sqrt{2}$.\\
		Vậy phương trình của đường tròn tâm $A$ và tiếp xúc với $\Delta$ là $(x+1)^2+(y-2)^2=2$.
	\end{enumerate}
}
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Nho Quan C, Ninh Bình][0D4K1]
Một công ty bất động sản có $50$ căn hộ cho thuê. Biết rằng cho thuê mỗi căn hộ với giá $2000000$ đồng một tháng thì mọi căn hộ đều có người thuê và cứ tăng thêm giá cho thuê mỗi căn hộ $100000$ đồng một tháng thì sẽ có hai căn hộ bị bỏ trống. Hỏi muốn có thu nhập cao nhất thì công ty đó phải cho thuê mỗi căn hộ với giá bao nhiêu một tháng?
\loigiai{
Gọi $x$ (đồng) là số tiền tăng thêm đối với mỗi căn hộ cho thuê.\\
Suy ra số căn hộ bị bỏ trống là $\dfrac{2x}{100000}$ (căn). Suy ra $0\leq x\leq 2500000$ (nếu $x>2500000$ thì tất cả các căn hộ sẽ không có người thuê).\\
Khi đó, số tiền thu nhập trong một tháng là
\begin{align*}
T&=\left(50-\dfrac{2x}{100000}\right)\left(2000000+x\right)=\dfrac{1}{50000}(2500000-x)(2000000+x)\\
&\leq \dfrac{1}{50000}\dfrac{(2500000+2000000)^2}{4}\quad\quad (\text{Theo BĐT Cô-si}).
\end{align*}
Dấu bằng xảy ra khi và chỉ khi $2500000-x=2000000+x\Leftrightarrow x=250000$.\\
Vậy muốn có thu nhập cao nhất thì công ty phải cho thuê mỗi căn hộ với giá $2250000$ đồng.
}
\end{bt}
