\subsection{Các dạng toán}
\begin{dang}[Chứng minh hai tam giác vuông bằng nhau]
\begin{enumerate}[\tickEX]
	\item Linh hoạt áp dụng các trường hợp bằng nhau của tam giác vuông ở trên.
\end{enumerate}
\end{dang}
\begin{vd}
Chứng minh rằng: Nếu cạnh huyền và một cạnh góc vuông của tam giác vuông này bằng cạnh huyền và một cạnh góc vuông của tam giác vuông kia thì hai tam giác vuông đó bằng nhau.
\loigiai
{
	\immini
	{
		Giả sử, có $\triangle ABC$ vuông tại $A$ và $\triangle A'B'C'$ vuông tại $A'$, biết $AB=A'B'$ và $BC=B'C'$. \\ Ta cần đi chứng minh rằng $\triangle ABC=\triangle A'B'C'$. \\ Thật vậy,
		trong $\triangle ABC$ vuông tại $A$, ta có
		$BC^2=AB^2+AC^2\Rightarrow AC^2=BC^2-AB^2=B'C'^2-A'B'^2\ (1)$.\\
		\noindent Trong $\triangle A'B'C'$ vuông tại $A'$, ta có
		$B'C'^2=A'B'^2+A'C'^2\Leftrightarrow A'C'^2=B'C'^2-A'B'^2\ (2)$.\\
		\noindent 	Từ (1) và (2) suy ra $AC^2=A'C'^2\Rightarrow AC=A'C'$.
		Khi đó, xét $\triangle ABC$ và $\triangle A'B'C'$ có
		$AB=A'B'$;
		$\widehat{A}=\widehat{A}'=90^\circ$;
		$AC=A'C'$.\\
		Vậy $\triangle ABC=\triangle A'B'C'$ (c.g.c).
	}
