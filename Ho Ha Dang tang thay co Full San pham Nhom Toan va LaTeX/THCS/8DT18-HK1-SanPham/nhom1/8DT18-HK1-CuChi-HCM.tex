Thực hiện các phép tính
\begin{enumEX}{2}
	\item $-5x^2(x-2)$.
	\item $(x-1)(x^2+x+1)$.
\end{enumEX}
\loigiai{
\begin{enumEX}{2}
	\item $-5x^2(x-2)=-5x^3+10x^2$.
	\item $(x-1)(x^2+x+1)=x^3-1$.
\end{enumEX}
}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Củ Chi, HCM]%[Hoàng Hải, dự án 8DT18]%[8D1B6]%[8D1K7]
Phân tích đa thức thành nhân tử
\begin{enumEX}{3}
	\item $3(x-y)+2(y-x)$.
	\item $7x^2-42x+63$.
	\item $x^2-8x+15$.
\end{enumEX}
\loigiai{
\begin{enumEX}{3}
	\item $\begin{aligned}[t]
	&3(x-y)+2(y-x) \\
	&= 3(x-y)-2(x-y) \\
	&= x-y.
	\end{aligned}$
	\item $\begin{aligned}[t]
	&7x^2-42x+63 \\
	&=7(x^2-6x+9) \\
	&=7(x-3)^2.
	\end{aligned}$
	\item $\begin{aligned}[t]
	&x^2-8x+15 \\
	&=(x-4)^2-1 \\
	&=(x-3)(x-5).
	\end{aligned}$
\end{enumEX}
}
\end{bt}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Củ Chi, HCM]%[Hoàng Hải, dự án 8DT18]%%[8D1B9]
Thực hiện các phép tính
\begin{enumEX}{2}
	\item $\dfrac{9x-10}{8}-\dfrac{6+x}{8}$.
	\item $\dfrac{2x-4}{x-1}+\dfrac{2x+2}{x^2-1}$.
\end{enumEX}
\loigiai{
\begin{enumEX}{2}
	\item $\begin{aligned}[t]
	&\dfrac{9x-10}{8}-\dfrac{6+x}{8} \\
	&= \dfrac{9x-10-6-x}{8} \\
	&= \dfrac{8x-16}{8} \\
	&= x-2.
	\end{aligned}$
	\item $\begin{aligned}[t]
	&\dfrac{2x-4}{x-1}+\dfrac{2x+2}{x^2-1} \\
	&= \dfrac{2x-4}{x-1}+\dfrac{2(x+1)}{(x-1)(x+1)} \\
	&= \dfrac{2x-4}{x-1}+\dfrac{2}{x-1} \\
	&= \dfrac{2(x-1)}{x-1}=2.
	\end{aligned}$
\end{enumEX}
}
\end{bt}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Củ Chi, HCM]%[Hoàng Hải, dự án 8DT18]%%[8D1T1]
Cách tính tiền nước sinh hoạt ở Thành phố Hồ Chí Minh cho 1 người dùng như sau: mức 1 giá 7000 đ/1m$^3$ cho 4 m$^3$ nước đầu tiên; mức 2 giá 10000
đ/1m$^3$ cho 3 m$^3$ nước tiếp theo; mức 3 giá 12500 đ/1m$^3$ cho số m$^3$ nước còn lại. Tháng 10/2017 gia đình bà Bê có 2 người đã dùng tổng cộng hết 20 m$^3$ nước. Hỏi gia đình bà Bê phải trả bao nhiêu tiền?
\loigiai{
Số tiền mà gia đình bà Bê phải trả là
$$ 7000\cdot4+10000\cdot3+12500\cdot(20-4-3)=220500\text{ đ}. $$
}
\end{bt}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Củ Chi, HCM]%[Hoàng Hải, dự án 8DT18]%%[8D1G9]
Tính giá trị của biểu thức
$$ A=(1-x)(1+x)(1+x^2)(1+x^4)(1+x^8)(1+x^{16})+x^{32}. $$
\loigiai{
Ta có
\begin{eqnarray*}
     A &=& (1-x)(1+x)(1+x^2)(1+x^4)(1+x^8)(1+x^{16})+x^{32} \\
     A &=& (1-x^2)(1+x^2)(1+x^4)(1+x^8)(1+x^{16})+x^{32} \\
     A &=& (1-x^{16})(1+x^{16})+x^{32} \\
     A &=& 1-x^{32}+x^{32} \\
     A &=& 1.
\end{eqnarray*}
}
\end{bt}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Củ Chi, HCM]%[Hoàng Hải, dự án 8DT18]%[8H1K9]
Cho tam giác $ABC$ vuông tại $A$ $(AB < AC)$ có đường cao $AH$. Từ $H$ kẻ $HM$ vuông góc với $AB$ $(M \in AB)$, kẻ $HN$ vuông góc với $AC$ $(N \in AC)$.
\begin{enumerate}
	\item Chứng minh tứ giác AMHN là hình chữ nhật.
	\item Gọi $I$ là trung điểm của $HC$, $K$ là điểm đối xứng với $A$ qua $I$. Chứng minh $AC$ song song $HK$.
	\item $MN$ cắt $AH$ tại $O$, $OC$ cắt $AK$ tại $D$. Chứng minh $AK = 3AD$.
\end{enumerate}
\loigiai{
\immini{
\begin{enumerate}
	\item Xét tứ giác $AMHN$ có $\widehat{MAN}=\widehat{AMH}=\widehat{ANH}=90^\circ$.\\
	Do đó $AMHN$ là hình chữ nhật (tứ giác có $3$ góc vuông).
	\item Xét tứ giác $AHKC$ có
	\begin{itemize}
		\item $I$ trung điểm $HC$ (giả thiết).
		\item $I$ trung điểm $AK$ ($K$ đối xứng $A$ qua $I$).
	\end{itemize}
    Vậy $AHKC$ là hình bình hành (hai đường chéo cắt nhau trung điểm mỗi đường).\\
    Suy ra $AC \parallel HK$.
    \item Xét $\triangle AHC$ có
    \begin{itemize}
    	\item $I$ trung điểm $HC$.
    	\item $O$ trung điểm $AH$ ($AMHN$ là hình chữ nhật).
    	\item $D$ là giao điểm của $AI$ và $CO$.
    \end{itemize}
    Do đó $D$ là trọng tâm $\triangle AHC$ suy ra $AD=\dfrac{2}{3}AI$.\\
    Mặt khác $AI=\dfrac{1}{2}AK$ nên $AD=\dfrac{2}{3}\cdot\dfrac{1}{2}AK=\dfrac{1}{3}AK$.\\
    Vậy $AK=3AD$.
\end{enumerate}
}{

\tkzInit[ymin=-1,ymax=7,xmin=-1,xmax=4]\tkzClip

\tkzDefPointBy[projection=onto B--C](A)\tkzGetPoint{H}
\tkzDefPointBy[projection=onto B--A](H)\tkzGetPoint{M}
\tkzDefPointBy[projection=onto A--C](H)\tkzGetPoint{N}
\tkzDefMidPoint(H,C)\tkzGetPoint{I}
\tkzDefPointBy[symmetry=center I](A)\tkzGetPoint{K}
\tkzDefMidPoint(A,H)\tkzGetPoint{O}
\tkzInterLL(A,K)(C,O)\tkzGetPoint{D}

\tkzDrawSegments(A,B B,C C,A A,H H,M H,N A,K C,K K,H M,N O,C)
\tkzDrawPoints[fill=black](A,B,C,H,M,N,I,K,O,D)
\tkzLabelPoints[left](N,I,D)
\tkzLabelPoints[right](B)
\tkzLabelPoints[above](C,K)
\tkzLabelPoints[above left]()
\tkzLabelPoints[above right](H)
\tkzLabelPoints[below](M,O)
\tkzLabelPoints[below left](A)
\tkzLabelPoints[below right]()
\tkzMarkRightAngles(B,A,C H,M,A H,N,A A,H,C)
\end{tikzpicture}
}
}
