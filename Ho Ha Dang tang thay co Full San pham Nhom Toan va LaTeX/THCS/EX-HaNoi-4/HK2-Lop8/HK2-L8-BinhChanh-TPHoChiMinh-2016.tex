\begin{ex}%[8D1Y3]]%[8D3K5]%[8D4B4]
	Giải các phương trình và bất phương trình sau:
	\begin{multicols}{2}
		\begin{enumerate}
			\item  $ \dfrac{5x-2}{3} = \dfrac{5-3x}{2} $.
			\item $ \dfrac{5x}{2x+2} +1 = \dfrac{6}{x+1} $.
			\item $ (x+2)(3-4x) = x^2+4x+4 $.
			\item $ \dfrac{x-1}{9} + \dfrac{x-2}{8} = \dfrac{x-3}{7} + \dfrac{x-4}{6} $.
			\item $ \dfrac{3x-1}{4} - \dfrac{3(x-2)}{8} -1 > \dfrac{5+3x}{2} $.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item $ \dfrac{5x-2}{3} = \dfrac{5-3x}{2} \Leftrightarrow \dfrac{10x-4}{6} = \dfrac{15-9x}{6} \Leftrightarrow 19x = 19 \Leftrightarrow x=1 $. Vậy $ S = \{1\} $.
			\item Điều kiện $ x\ne -1 $. Khi đó,
			\[  \dfrac{5x}{2x+2} +1 = \dfrac{6}{x+1} \Leftrightarrow  \dfrac{5x}{2(x+1)} +\dfrac{2x+2}{2(x+1)} = \dfrac{12}{2(x+1)} \Rightarrow 7x = 10 \Leftrightarrow x = \dfrac{10}{7} \text{ (nhận)}. \]
			Vậy $ S = \left\{\dfrac{10}{7} \right\}  $.
			\item Ta có biến đổi sau:
			\begin{align*}
			&\phantom{\Leftrightarrow } (x+2)(3-4x) = x^2+4x+4 \Leftrightarrow (x+2)(3-4x) = (x+2)^2 \\
			&\Leftrightarrow (x+2)(3-4x-x-2) = 0\Leftrightarrow (x+2)(1-5x) = 0\\
			&\Leftrightarrow x +2 = 0 \text{ hay } 1-5x = 0 \Leftrightarrow x = -2 \text{ hay } x = \dfrac{1}{5}.
			\end{align*}
			Vậy $ S = \left\{ -2; \dfrac{1}{5} \right\} $. 
			\item Ta có biến đổi sau:
			\begin{align*}
			&\phantom{\Leftrightarrow } \dfrac{x-1}{9} + \dfrac{x-2}{8} = \dfrac{x-3}{7} + \dfrac{x-4}{6} \Leftrightarrow \dfrac{x-1}{9} -1 + \dfrac{x-2}{8} - 1 = \dfrac{x-3}{7} -1 + \dfrac{x-4}{6} -1\\
			&\Leftrightarrow \dfrac{x-10}{9} + \dfrac{x-10}{8} = \dfrac{x-10}{7} + \dfrac{x-10}{6} \Leftrightarrow (x-10)\left(\dfrac{1}{9} + \dfrac{1}{8} - \dfrac{1}{7} - \dfrac{1}{6} \right) = 0\\
			&\Leftrightarrow x-10 = 0 \Leftrightarrow x = 10. 
			\end{align*}
			Vậy $ S = \{10\} $.
			\item $ \dfrac{3x-1}{4} - \dfrac{3(x-2)}{8} -1 > \dfrac{5+3x}{2} \Leftrightarrow 6x-2 - 3x+6 - 8 > 20+12x \Leftrightarrow -9x > 24 \Leftrightarrow x < \dfrac{-24}{9} $.\\
			Vậy $ S = \left\{x|x < \dfrac{-24}{9} \right\}  $.
		\end{enumerate}	
	}
\end{ex}
\begin{ex}%[8D3K6]
	Thùng dầu thứ nhất chứa số dầu gấp đôi số dầu trong thùng thứ hai. Nếu lấy từ thùng thứ nhất $ 30 $ lít cho vào thùng thứ hai thì số dầu ở cả hai thùng bằng nhau. Tính số dầu ở mỗi thùng lúc đầu.
	\loigiai{
		Gọi số dầu trong thùng thứ hai lúc ban đầu là $ x $ (lít), $ x > 0 $. Khi đó, số dầu trong thùng thứ nhất lúc đầu là $ 2x $ (lít).\\	
		Số dầu trong thùng thứ nhất lúc sau là $ 2x - 30 $ (lít). Số dầu trong thùng thứ hai lúc sau là $ x+30 $ (lít).\\	
		Theo đề bài thì ta có phương trình	$ 2x - 30 = x+30 \Leftrightarrow x= 60 \text{ (thỏa điều kiện)} $.\\	
		Vậy số dầu lúc đầu của thùng thứ nhất và thứ hai lần lượt là $ 120 $ lít và $ 60 $ lít.
	}
\end{ex}
\begin{ex}%[8H3Y7]%[8H3B7]%[8H3G2]
	Cho hình chữ nhật $ ABCD $ có $ AB = 8\mathrm{\ cm}, BC = 6\mathrm{\ cm} $. Vẽ đường cao $ AH $ của tam giác $ ABD $.
	\begin{enumerate}
		\item Chứng minh $ \triangle AHB $ đồng dạng với $ \triangle DAB $.
		\item Tính độ dài $ BD, AH $.
		\item Qua $ B $ kẻ đường thẳng $ a \perp BD $, $ a $ cắt $ DC $ kéo dài tại $ E $. Tính tỉ số diện tích $ \triangle BCE $ và $ \triangle BHA $.
		\item Kẻ đường cao $ CF $ của $ \triangle BCE $. Chứng minh $ AC.EF = EB. CF $.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\tkzInit[xmin = -0.5, xmax=6,ymin = -0.5,ymax=3]
			\tkzClip
			\tkzDefPoints{0/0/D, 4/0/C, 4/2.5/B, 0/2.5/A}
			\tkzDefLine [orthogonal=through A](B,D) \tkzGetPoint{h}
			\tkzInterLL(A,h)(B,D) \tkzGetPoint{H}
			\tkzDefLine [orthogonal=through B](B,D) \tkzGetPoint{e}
			\tkzInterLL(B,e)(C,D) \tkzGetPoint{E}
			\tkzDefLine [orthogonal=through C](B,E) \tkzGetPoint{f}
			\tkzInterLL(E,B)(C,f) \tkzGetPoint{F}
			\tkzDrawPolygon(A,B,C,D)
			\tkzDrawSegments(A,H B,D B,E C,F C,E A,C)
			\tkzDrawPoints(A,B,C,D,E,F,H)
			\tkzMarkRightAngle(A,H,D) \tkzMarkRightAngle(D,B,E) \tkzMarkRightAngle(C,F,E)
			\tkzLabelPoints[above right](B,F)
			\tkzLabelPoints[below right](E)
			\tkzLabelPoints[below](C,H)
			\tkzLabelPoints[below left](D)
			\tkzLabelPoints[above left](A)
			\tkzLabelSegment[above](A,B){$8\mathrm{\ cm}$}
			\tkzLabelSegment[rotate = 90](B,C){$6\mathrm{\ cm}$}
			\end{tikzpicture}
		\end{center}
		\begin{enumerate} 
			\item \textbf{Chứng minh $ \triangle AHB $ đồng dạng với $ \triangle DAB $.}\\
			Xét $ \triangle AHB $ và $ \triangle DAB $ có:
			\begin{itemize}
				\item $ \widehat{B} $ chung.
				\item $ \widehat{AHB} = \widehat{DAB} (=90^\circ) $ (vì $ AH $ là đường cao $ \triangle ABD $, $ ABCD $ là hình chữ nhật)
			\end{itemize}
			Vậy $ \triangle AHB \backsim \triangle DAB $ (g-g).
			\item \textbf{Tính độ dài $ BD, AH $.}\\
			Xét $ \triangle DAB $ vuông tại $ A $, theo định lý Pytago ta có:
			\[ BD^2 = AB^2  +AD^2 \Leftrightarrow BD^2 = 8^2  + 6^2 = 100 \Leftrightarrow BD = 10\mathrm{\ cm} \]
			Ta lại có $ \triangle AHB \backsim \triangle DAB  $ (cmt), suy ra $ \dfrac{AH}{AD} = \dfrac{AB}{BD} \Rightarrow AH = AD\cdot\dfrac{AB}{BD} = 4{,}8\mathrm{\ cm} $.
			\item \textbf{Tính tỉ số diện tích $ \triangle BCE $ và $ \triangle BHA $.}\\
			Ta có $ \triangle AHB \backsim \triangle DAB $ suy ra $ \dfrac{HB}{AB} = \dfrac{AB}{BD} \Rightarrow HB = \dfrac{8{,}8}{10} = 6{,}4\mathrm{\ cm} $.\\
			Xét $ \triangle BCE $ và $ \triangle BHA $ có:
			\begin{itemize}
				\item $ \widehat{AHB} = \widehat{BCE} = 90^\circ $.
				\item $ \widehat{ABH} = \widehat{CBH} $ (cùng phụ $ \widehat{CBD} $).
			\end{itemize}
			Vậy $ \triangle BCE \backsim \triangle BHA $ (g-g). Suy ra $ \dfrac{S_{\triangle CBE}}{S_{\triangle BHA}} = \dfrac{BC^2}{BH^2} = \dfrac{225}{256} $.
			\item \textbf{Chứng minh $ AC\cdot EF = EB\cdot CF $.}\\
			Xét $ \triangle BED $ có $ CF \parallel BD $ (vì cùng $ \perp BE $), suy ra $ \dfrac{EF}{EB} = \dfrac{CF}{BD} \Rightarrow BD\cdot EF = EB\cdot CF $.\\
			Mà $ AC = BD $ (vì $ ABCD $ là hình chữ nhật) nên $ AC\cdot EF = EB\cdot CF $.
		\end{enumerate}
	}
\end{ex}
