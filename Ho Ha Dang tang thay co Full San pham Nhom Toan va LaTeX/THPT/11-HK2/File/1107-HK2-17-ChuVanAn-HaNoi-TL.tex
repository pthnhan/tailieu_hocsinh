\begin{bt}%[HK2, Chu Văn An, Hà Nội, 2016-2017]%[1D4B2-7]%[Vũ Nguyễn Hoàng Anh, 11HK2]
Tính giới hạn sau $\lim\limits_{x\to +\infty} \dfrac{2x^3+3x^2-1}{x^3-x+2}$.
\loigiai{
Ta có
\begin{align*}
\lim\limits_{x\to +\infty} \dfrac{2x^3+3x^2-1}{x^3-x+2}=\lim \limits_{x\to +\infty} \dfrac{2+\dfrac{3}{x^2}-\dfrac{1}{x^3}}{1-\dfrac{1}{x^2}+\dfrac{2}{x^3}},
\end{align*}
mà $\lim\limits_{x\to +\infty}\left( 2+\dfrac{3}{x^2}-\dfrac{1}{x^3} \right)=2$ và $\lim\limits_{x\to +\infty}\left( 1-\dfrac{1}{x^2}+\dfrac{2}{x^3} \right)=1$ nên
\begin{align*}
\lim\limits_{x\to +\infty} \dfrac{2x^3+3x^2-1}{x^3-x+2}=\dfrac{2}{1}=2.
\end{align*}
}
\end{bt}

\begin{bt}%[HK2, Chu Văn An, Hà Nội, 2016-2017]%[1D4K2-4]%[Vũ Nguyễn Hoàng Anh, 11HK2]
Tính giới hạn sau $\lim\limits_{x\to -\infty}\left( \sqrt{x^2-x} -\sqrt{x^2-3x+1} \right)$.
\loigiai{
Ta có
\begin{align*}
\lim\limits_{x\to -\infty} \left( \sqrt{x^2-x} -\sqrt{x^2-3x+1} \right)&=\lim\limits_{x\to -\infty} \dfrac{2x-1}{\sqrt{x^2-x}+\sqrt{x^2-3x+1}}\\
&=\lim\limits_{x\to -\infty} \dfrac{2-\dfrac{1}{x}}{-\sqrt{1-\dfrac{1}{x}}-\sqrt{1-\dfrac{3}{x}+\dfrac{1}{x^2}}}.
\end{align*}
Mặt khác $\lim\limits_{x\to -\infty}\left( 2-\dfrac{1}{x} \right)=2$ và $\lim\limits_{x\to -\infty}\left( -\sqrt{1-\dfrac{1}{x}}-\sqrt{1-\dfrac{3}{x}+\dfrac{1}{x^2}} \right)=-2$ nên
\begin{align*}
\lim\limits_{x\to -\infty}\left( \sqrt{x^2-x} -\sqrt{x^2-3x+1} \right)=\dfrac{2}{-2}=-1.
\end{align*}
}
\end{bt}

\begin{bt}%[HK2, Chu Văn An, Hà Nội, 2016-2017]%[1D4K3-3]%[Vũ Nguyễn Hoàng Anh, 11HK2]
Cho hàm số $f(x)=\begin{cases} \dfrac{x^2-3x+2}{\sqrt{x+3}-2} & \text{khi}\ x>1\\ 3x-7 &\text{khi}\ x\le 1 \end{cases}$. Xét tính liên tục của hàm số $f$ tại điểm $x=1$.
\loigiai{
Ta có
\begin{align*}
\lim\limits_{x\to 1^+}f(x)&=\lim\limits_{x\to 1^+} \dfrac{x^2-3x+2}{\sqrt{x+3}-2}\\
&=\lim\limits_{x\to 1^+} \dfrac{(x-1)(x-2)\left( \sqrt{x+3}+2 \right)}{x-1}\\
&=\lim\limits_{x\to 1^+} \left[(x-2)\left( \sqrt{x+3}+2 \right)\right]\\
&=-4.
\end{align*}
Ta lại có
\begin{align*}
\lim\limits_{x\to 1^-}f(x)=\lim\limits_{x\to 1^-} (3x-7)=-4.
\end{align*}
Suy ra $\lim\limits_{x\to 1^+}f(x)=\lim\limits_{x\to 1^-}f(x)=f(1)$ nên hàm số $f(x)$ liên tục tại $x=1$.
}
\end{bt}

\begin{bt}%[HK2, Chu Văn An, Hà Nội, 2016-2017]%[1D5B2-1]%[1D5B2-2]%[Vũ Nguyễn Hoàng Anh, 11HK2]
Cho hàm số $y=\dfrac{1}{\sqrt{x^2+4}}$ có đồ thị là $(C)$.
\begin{enumerate}
\item Tìm đạo hàm $y'$ của hàm số đã cho.
\item Viết phương trình tiếp tuyến của đồ thị $(C)$ tại điểm có hoành độ bằng $0$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có
\begin{align*}
y'=-\dfrac{\left( \sqrt{x^2+4} \right)'}{x^2+4}=-\dfrac{x}{(x^2+4)\sqrt{x^2+4}}.
\end{align*}
\item Điểm $M\left( 0;\dfrac{1}{2} \right)\in (C)$, có hoành độ bằng $0$. Tiếp tuyến của đồ thị $(C)$ tại $M$ có phương trình
\begin{align*}
y=y'(0)(x-0)+\dfrac{1}{2}
\end{align*}
hay $y=\dfrac{1}{2}$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[HK2, Chu Văn An, Hà Nội, 2016-2017]%[1H3B3-2]%[1H3B3-3]%[1H3K4-3]%[1H3K5-3]%[Vũ Nguyễn Hoàng Anh, 11HK2]
Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình vuông cạnh $a$. Gọi $M$, $N$ lần lượt là trung điểm của $AB$ và $CD$. Đường thẳng $SM$ vuông góc với mặt phẳng $(ABCD)$ và $SM=a\sqrt{5}$.
\begin{enumerate}
\item Chứng minh rằng đường thẳng $CD$ vuông góc với mặt phẳng $(SMN)$.
\item Tính số đo của góc giữa đường thẳng $SC$ và mặt phẳng $(ABCD)$.
\item Tính số đo góc giữa mặt phẳng $(SAB)$ và mặt phẳng $(SCD)$.
\item Gọi $I$ là giao điểm của hai đường thẳng $BD$ và $CM$. Tính theo $a$ khoảng cách từ điểm $I$ đến mặt phẳng $(SCD)$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[>=stealth, scale=1.1]
\tkzDefPoints{0/0/M, -1.2/-0.6/B, 0/4.5/S}
\tkzDefBarycentricPoint(M=2,B=-1)\tkzGetPoint{A}
\coordinate (C) at($(B)+(4,0)$);
\tkzDefBarycentricPoint(A=-1,C=-1,B=1)\tkzGetPoint{D}
\tkzDefMidPoint(C,D)\tkzGetPoint{N}
\tkzInterLL(B,D)(C,M)\tkzGetPoint{I}
\tkzInterLL(A,C)(B,D)\tkzGetPoint{O}
\tkzDrawAltitude[dashed](S,N)(M)\tkzGetPoint{H}

\tkzLabelPoints[above](S,O)
\tkzLabelPoints[below left](B)
\tkzLabelPoints[above right](D,H,A)
\tkzLabelPoints[above left](M)
\tkzLabelPoints[below right](C,N)
\tkzLabelPoints[below](I)

\tkzDrawSegments[dashed](B,D A,C S,M A,B S,A A,D M,N C,M)
\tkzDrawSegments(S,B S,D S,C S,N B,C D,C)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Theo giả thiết $SM\perp (ABCD)$ nên $SM\perp CD$. Mặt khác do $M$, $N$ lần lượt là trung điểm $AB$, $CD$ nên $MN\perp CD$. Vậy $CD\perp (SMN)$.
\item Do $SM\perp (ABCD)$ nên $\left( (SC, (ABCD) \right)=(SC,MC)=\widehat{SCM}$. Ta có
\begin{align*}
\tan \widehat{SCM}=\dfrac{SM}{MC}=\dfrac{SM}{\sqrt{BM^2+CM^2}}=\dfrac{a\sqrt{5}}{\sqrt{\dfrac{a^2}{4}+a^2}}=2.
\end{align*}
Vậy $\left( SC,(ABCD) \right)=\arctan 2$.
\item Gọi $H$ là hình chiếu của $M$ trên $SN$. Khi đó $MH\perp SN$, mà $CD\perp (SMN)$ nên $MH\perp CD$. Suy ra $MH\perp (SCD)$. 
Ta có $MN\perp AB$ mà $SM\perp MN$ (do $SM\perp (ABCD)$) nên $MN\perp (SAB)$. Suy ra
\begin{align*}
\left( (SAB), (SCD) \right)= (MN,MH)=\widehat{HMN}=\widehat{MSN}.
\end{align*}
Ta có 
\begin{align*}
\tan \widehat{MSN}=\dfrac{MN}{SM}=\dfrac{a}{a\sqrt{5}}=\dfrac{\sqrt{5}}{5}.
\end{align*}
Vậy $\left( (SAB), (SCD) \right)=\arctan \dfrac{\sqrt{5}}{5}$.
\item Gọi $O$ là tâm của hình vuông $(ABCD)$. Dễ thấy $I$ là trọng tâm tam giác $ABC$ nên $\dfrac{IC}{MC}=\dfrac{2}{3}$. Suy ra
\begin{align*}
d(I,(SCD))&=\dfrac{2}{3}d(M,(SCD))=\dfrac{2}{3}MH\\
&=\dfrac{2}{3}\cdot \dfrac{SM\cdot MN}{SN}=\dfrac{2}{3}\cdot\dfrac{SM\cdot MN}{\sqrt{SM^2+MN^2}}\\
&=\dfrac{2}{3}\cdot \dfrac{a^2\sqrt{5}}{a\sqrt{6}}=\dfrac{a\sqrt{30}}{9}.
\end{align*}
Vậy $d(I,(SCD))=\dfrac{a\sqrt{30}}{9}$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[HK2, Chu Văn An, Hà Nội, 2016-2017]%[1D4G3-6]%[Vũ Nguyễn Hoàng Anh, 11HK2]
Chứng minh rằng phương trình 
$$|x|^3-3mx^2+  9(m^2+3)|x|-54=0$$
luôn có ít nhất hai nghiệm phân biệt với mọi giá trị thực của tham số $m$.
\loigiai{
Đặt $f(x)=|x|^3-3mx^2+  9(m^2+3)|x|-54$, dễ thấy $f(x)$ liên tục trên $\mathbb{R}$. Ta có
\begin{align*}
&f(0)=-54<0,\\
&f(2)=f(-2)=18m^2-12m+8=2\left[ (3m-1)^2+3 \right]>0,\ \forall m\in \mathbb{R}.
\end{align*}
Suy ra $f(-2)f(0)<0$ và $f(0)f(2)<0$. Theo định lí giá trị trung gian, tồn tại $x_1\in(-2,0)$ và $x_2\in (0,2)$ sao cho $f\left( x_1 \right)=f\left( x_2 \right)=0$. Ta có điều phải chứng minh.
}
\end{bt}
