\section{KHÁI NIỆM VỀ BIỂU THỨC ĐẠI SỐ}
\subsection{Tóm tắt lý thuyết}
\subsubsection{Biểu thức số}
\begin{dn}
Các số được nối với nhau bởi các phép tính (cộng, trừ, nhân, chia, nâng lên lũy thừa) được gọi là các biểu thức số.
\end{dn}

\subsubsection{Biểu thức đại số}
\begin{dn}
	Các biến được nối với nhau bởi các phép tính (cộng, trừ, nhân, chia, nâng lên lũy thừa) được gọi là các biểu thức đại số.
\end{dn}

\begin{note}\textbf{Chú ý}
	\begin{enumEX}{1}
		\item Trong biểu thức đại số, người ta cũng dùng các dấu ngoặc để chỉ thứ tự thực hiện các phép tính.
		\item Trong biểu thức đại số, vì các chữ đại diện cho số nên khi thực hiện các phép toán trên các chữ, ta có thể áp dụng các tính chất, quy tắc phép toán như trên các số.
	\begin{vd} Một vài tính chất\\
			$\bullet$ $x+y=y+x$; $xy=yx$ - Tính chất giao hoán.\\
			$\bullet$ $(x+y)+z=x+(y+z)$; $(xy)z=x(yz)$ - Tính chất kết hợp.\\
			$\bullet$ $(x+y)z=xz+yz$ - Tính chất phân phối.\\
			$\bullet$ $xx=x^2$; $xxx=x^3$.\\
			$\bullet$ $-(x-y+z)=-x+y-z$ - Quy tắc đổi dấu.	
	\end{vd}
		
	\end{enumEX}
\end{note}

\subsection{Phương pháp giải toán}
\begin{vd}%[Lê Văn Thiện, dự án(EX-C2-L7-2-2018)]%[7D4Y1]
	Viết biểu thức đại số để biểu đạt các ý sau
	\item[a)] Tổng của $a$ và $b$.
	\item[b)] Tích của $a$ và $b$.
	\item[c)] Tổng của $a$ và $b$ lập phương.
	\item[d)] Tổng các lập phương của $a$ và $b$.
	\item[e)] Lập phương của tổng của $a$ và $b$.
	\item[f)] Tích của tổng $a$ và $b$ với hiệu của $a$ và $b$.
	\loigiai{
	\begin{itemize}
	\item[a)] Tổng của $a$ và $b$ là $a+b$.
	\item[b)] Tích của $a$ và $b$ là $a\cdot b$.
	\item[c)] Tổng của $a$ và $b$ lập phương là $a+b^3$.
	\item[d)] Tổng các lập phương của $a$ và $b$ là $a^3+b^3$.
	\item[e)] Lập phương của tổng của $a$ và $b$ là $(a+b)^3$.
	\item[f)] Tích của tổng $a$ và $b$ với hiệu của $a$ và $b$ là $(a+b)(a-b)$.
	\end{itemize}
}
\end{vd}

\begin{vd}%[Lê Văn Thiện, dự án(EX-C2-L7-2-2018)]%[7D4B1]
Cho hình chữ nhật có hai cạnh liên tiếp bằng $a$ và $b$.
\begin{itemize}
	\item[a)] Viết biểu thức tính chu vi của hình chữ nhật.
	\item[b)] Viết biểu thức tính diện tích của hình chữ nhật.
\end{itemize}
\loigiai{
\begin{itemize}
	\item[a)] Biểu thức tính chu vi của hình chữ nhật là $CV=2\cdot (a+b)$.
	\item[b)] Biểu thức tính diện tích của hình chữ nhật là $S=a\cdot b$.
\end{itemize}
}
\end{vd}

\begin{vd}%[Lê Văn Thiện, dự án(EX-C2-L7-2-2018)]%[7D4B1]
	Cho hình chữ nhật có chiều dài hơn chiều rộng $2cm$. Viết biểu thức tính diện tích của hình chữ nhật.
	\loigiai{
	\textbf{Cách 1} Giả sử hình chữ nhật có chiều rộng bằng $x$, suy ra chiều dài bằng $x+2$. Khi đó, diện tích hình chữ nhật $(S)$ được cho bởi $x(x+2).$\\
	\textbf{Cách 2} Giả sử hình chữ nhật có chiều dài bằng $x$, suy ra chiều rộng bằng $x-2$. Khi đó, diện tích hình chữ nhật $(S)$ được cho bởi $x(x-2).$
}
\end{vd}

\begin{vd}%[Lê Văn Thiện, dự án(EX-C2-L7-2-2018)]%[7D4Y1]
	 (Bài $2$/tr $26$-SGK)
	Viết biểu thức đại số biểu thị diện tích hình thang có đáy lớn là $a$, đáy nhỏ là $b$, đường cao là $h$ ($a,b$ và $h$ có cùng đơn vị).
	\loigiai{
	Diện tích hình thang được viết như sau $S=\dfrac{1}{2}(a+b)\cdot h$.\\
	Công thức trên được đọc là \lq\lq\textit{Diện tích hình thang bằng đáy lớn cộng đáy nhỏ nhân chiều cao chia đôi.}\rq\rq
}
\end{vd}

		\item $9x^3$.
		\item $(x-1)(x+1)$.
	\end{enumEX}
\loigiai{
\begin{enumEX}{1}
	\item Ta đọc $x^2+8$ là tổng của $x$ bình phương và $8$.
	\item Ta đọc $9x^3$ là tích của $9$ và $x$ lập phương.
	\item Ta đọc $(x-1)(x+1)$ là tích của hiệu số $x$ và $1$ với tổng của chúng.
\end{enumEX}
}
\end{vd}

	\begin{enumEX}{1}
		\item Số giờ đi hết quãng đường $AB$ dài $x$ (km) của một người đi xe máy với vận tốc $30$km/h.
		\item Tổng quãng đường đi của một người, biết\\

		$\bullet$ Người đó đi xe đạp trong $y$ giờ với vận tốc $12$km/h.\\
		$\bullet$ Người đó đi xe máy trong $z$ giờ với vận tốc $30$km/h.	
	\end{enumEX}

\begin{enumEX}{1}
	\item Người đi xe máy đi quãng đường $AB$ dài $x$ km với vận tốc $30$km/h mất khoảng thời gian bằng $\dfrac{x}{30}$ (giờ).
	\item Ta có\\
		$\bullet$ Quãng đường người đó đi bộ trong $x$ giờ với vận tốc $5$km/h là $5x$ (km).\\
		$\bullet$ Quãng đường người đó đi xe đạp trong $y$ giờ với vận tốc $12$km/h là $12y$ (km).\\
		$\bullet$ Quãng đường người đó đi xe máy trong $z$ giờ với vận tốc $30$km/h là $30z$ (km).\\	
Vậy tổng quãng đường đi của người đó đi là $5x+12y+30z$ (km).
\end{enumEX}
}
\end{vd}

\begin{bt}%[Lê Văn Thiện, dự án(EX-C2-L7-2-2018)]%[7D4Y1]

	\begin{enumEX}{1}
		\item Tổng của $a$ bình phương và $b$ lập phương.
		\item Hiệu các lập phương của $a$ và $b$.
		\item Lập phương của hiệu $a$ và $b$.
	\end{enumEX}
\loigiai{
\begin{enumEX}{1}
	\item Tổng của $a$ bình phương và $b$ lập phương là $a^2+b^3$.
	\item Hiệu các lập phương của $a$ và $b$ là $a^3-b^3$.
	\item Lập phương của hiệu $a$ và $b$ là $(a-b)^3$.
\end{enumEX}
}
\end{bt}

\begin{bt}%[Lê Văn Thiện, dự án(EX-C2-L7-2-2018)]%[7D4Y1]
	Sử dụng các thuật ngữ đã học để đọc các biểu thức sau
	\begin{enumEX}{3}
		\item $x^3-1$.
		\item $5:x^2$.
		\item $(x+8)(x-2)$.
	\end{enumEX}
\loigiai{
\begin{enumEX}{1}
	\item Ta đọc $x^3-1$ là hiệu của $x$ lập phương và $1$. 
	\item Ta đọc $5:x^2$ là thương của $5$ và $x$ bình phương.
	\item Ta đọc $(x+8)(x-2)$ là tích của tổng hai số $x$ và $8$ với hiệu hai số $x$ và $2$.
\end{enumEX}
}
\end{bt}

		\textbf{Cách 2} Giả sử hình chữ nhật có chiều dài bằng $x$, suy ra chiều rộng bằng $x-2$. Khi đó, diện tích hình chữ nhật $(S)$ được cho bởi $x(x-2).$
	}
\end{bt}

\begin{bt}%[Lê Văn Thiện, dự án(EX-C2-L7-2-2018)]%[7D4Y1]
Viết biểu thức tính diện tích của hình thang có đáy lớn bằng hai đáy nhỏ và đường cao là $h$.
\loigiai{
	\textbf{Cách 1} Giả sử hình có đáy nhỏ  bằng $a$, suy ra đáy lớn bằng $2a$. Khi đó, diện tích hình thang $(S)$ được cho bởi $\dfrac{(a+2a)h}{2}=\dfrac{3}{2}ah.$\\
	\textbf{Cách 2} Giả sử hình có đáy lớn  bằng $a$, suy ra đáy nhỏ bằng $\dfrac{a}{2}$. Khi đó, diện tích hình thang $(S)$ được cho bởi $\dfrac{(a+\dfrac{a}{2})h}{2}=\dfrac{3}{4}ah.$}
\end{bt}

\begin{bt}%[Lê Văn Thiện, dự án(EX-C2-L7-2-2018)]%[7D4B1]

		\item Một số khi chia cho $8$ được thương là $a$ và dư $5$. Hiệu của số đó với $9$ thì chia cho $11$ được thương là $b$ và dư $3$.

\end{enumEX}

\section{GIÁ TRỊ CỦA MỘT BIỂU THỨC ĐẠI SỐ}
\subsection{Tóm tắt lý thuyết}
Để tính giá trị của một biểu thức đại số tại những giá trị cho trước của các biến, ta thay các giá trị cho trước đó vào biểu thức rồi thực hiện các phép tính.
\begin{note}{Chú ý}
	Để phát triển bài toán tính giá trị của biểu thức đại số, người ta thường xuyên yêu cầu \lq\lq\textit{Xây dựng biểu thức rồi tính giá trị của biểu thức trong một hay nhiều trường hợp cụ thể}.\rq\rq
\end{note}

\subsection{Phương pháp giải toán}
\begin{vd}%[Lê Văn Thiện, dự án(EX-C2-L7-2-2018)]%[7D4Y2]
	Tính giá trị biểu thức $x^2+3x-12$ tại $x=7$ và tại $x=8$.
	\loigiai{
	\begin{enumEX}{1}
		\item Thay $x=7$ vào biểu thức đã cho, ta được $7^2+3\cdot 7-12=58.$ \\
		Vậy giá trị của biểu thức tại $x=7$ bằng $58$. 
		\item Thay $x=8$ vào biểu thức đã cho, ta được $8^2+3\cdot 8-12=76.$ \\
		Vậy giá trị của biểu thức tại $x=8$ bằng $76$.
	\end{enumEX}
}
\end{vd}

\begin{note}{Chú ý}
	Sử dụng máy tính CASIO Fx - $570$MS tính giá trị của biểu thức trên tại các giá trị $x$ khác nhau chúng ta thực hiện như sau
	\begin{itemize}
		\item[1)] Nhập biểu thức $x^2+3x-12$ vào máy, bằng cách ấn
		\begin{center}
		\fbox{ALPHA} \fbox{X} \fbox{$x^2$} \fbox{$+$} $3$\fbox{ALPHA} \fbox{X} \fbox{$-$} $12$
		\end{center}
		\item[2)] Lưu trữ biểu thức vào bộ nhớ $CACL$, bằng cách ấn 
		\begin{center}
			\fbox{CACL}.
		\end{center}
		\item[3)] Để nhận được giá trị của hàm số với $X=7$, ta ấn 
		\begin{center}
			\fbox{7} \fbox{=} \qquad \qquad \qquad \fbox{\qquad 58}
		\end{center}
	\item[4)] Để nhận được giá trị của hàm số với $X=8$, ta ấn 
	\begin{center}

	\end{itemize}
Trong đó
\begin{itemize}
	\item[1)] Dấu \fbox{=} được nhập vào bằng phím màu đỏ trên bàn phím của máy tính.
	\begin{center}
		\fbox{ALPHA} \fbox{X} \fbox{$x^2$} \fbox{$+$} $3$\fbox{ALPHA} \fbox{X} \fbox{$-$} $12$
	\end{center}
	\item[2)] Biểu thức lưu trữ trong bộ nhớ $CACL$ bị xóa khi ta
	\begin{itemize}
		\item Thực hiện một phéo toán khác.
		\item Thay đổi Mode khác.
		\item Tắt máy tính.
	\end{itemize}
\end{itemize}
\end{note}

\begin{vd}%[Lê Văn Thiện, dự án(EX-C2-L7-2-2018)]%[7D4Y2]
	Tính giá trị biểu thức $x^2y+xy^2$ tại $x=1$ và $y=\dfrac{1}{2}$.
	\loigiai{
		Thay $x=1$ và $y=\dfrac{1}{2}$ vào biểu thức đã cho, ta được\\
		$$1^2\cdot\dfrac{1}{2}+1\cdot\left( \dfrac{1}{2}\right)^2=\dfrac{1}{2} +\dfrac{1}{4}=\dfrac{2+1}{4}=\dfrac{3}{4}.$$
		Vậy giá trị biểu thức $x^2y+xy^2$ tại $x=1$ và $y=\dfrac{1}{2}$ bằng $\dfrac{3}{4}.$
	}
\end{vd}
