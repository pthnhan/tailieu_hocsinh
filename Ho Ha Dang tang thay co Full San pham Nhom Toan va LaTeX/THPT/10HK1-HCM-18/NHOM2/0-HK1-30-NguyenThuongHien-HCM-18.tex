\begin{bt}%[Đề HK1, Nguyễn Thượng Hiền, Hồ Chí Minh 17-18]%[Lục Hán Tường]%[0D2B3-2]
Xác định Parabol $ y = ax^2 + bx + c $, biết rằng parabol đó đi qua gốc tọa độ $ O $ và có đỉnh $ I\left(1;-4\right) $.
\loigiai{
Ta có Parabol $ y = ax^2 + bx + c $ đi qua gốc $ O\left(0;0\right) $ nên
\begin{eqnarray*}
a\cdot 0^2 + b\cdot 0 + c = 0 \Leftrightarrow c = 0.
\end{eqnarray*}
Lại có tọa độ đỉnh $ I\left(1;-4\right) $ nên
\begin{eqnarray*}
\heva{&\dfrac{-b}{2a} = 1\\ &a\cdot 1^2 + b\cdot 1 + c = -4}
\Leftrightarrow \heva{& 2a + b = 0\\ &a + b = -4}
\Leftrightarrow \heva{&a = 4 \\ &b = -8.}
\end{eqnarray*}
Vậy Parabol cần tìm là $ y = 4x^2 - 8x $.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Thượng Hiền, Hồ Chí Minh 17-18]%[Lục Hán Tường]%[0D3K2-5]
Cho phương trình $ x^2 - 3x + 2m - 2 = 0 $. Tìm $ m $ để phương trình có hai nghiệm phân biệt $x_1$, $x_2$ thỏa mãn $ 4\left(x_1^2 + x_2^2\right) = 17\left|x_1x_2\right| $.
\loigiai{
Để phương trình đã cho có hai nghiệm phân biệt
\begin{eqnarray*}
 \Delta > 0 
 \Leftrightarrow  9 - 4\left(2m - 2\right) > 0 
 \Leftrightarrow  17 - 8m > 0 
 \Leftrightarrow  m < \dfrac{17}{8}. \quad \quad (*)
\end{eqnarray*}
$ \forall m $ thỏa $ (*) $ thì phương trình có hai nghiêm phân biệt $x_1$, $x_2$ thỏa $ \heva{&x_1 + x_2 = 3\\&x_1x_2 = 2m - 2.}$\\
Do đó 
\begin{eqnarray*}
&& 4\left(x_1^2 + x_2^2\right) = 17\left|x_1x_2\right|\\
& \Leftrightarrow & 4\left(x_1 + x_2 \right)^2 - 8x_1x_2 = 17\left|x_1x_2\right|\\
& \Leftrightarrow & 36 - 8\left(2m - 2 \right) = 17\left|2m - 2\right|\\
& \Leftrightarrow & 52 - 16m = 17\left|2m - 2\right|\\
& \Leftrightarrow &\hoac{&52 - 16m = 17\left(2m - 2\right)\\&16m - 52 = 17\left(2m - 2\right)}
 \Leftrightarrow \hoac{& m = \dfrac{43}{25} \\& m = -1 } \textrm{(thỏa điều kiện).}
\end{eqnarray*}
Vậy có $ 2 $ giá trị $ m $ thỏa yêu cầu bài toán: $ m = -1 $; $ m = \dfrac{43}{25} $.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Thượng Hiền, Hồ Chí Minh 17-18]%[Lục Hán Tường]%[0D3B2-4]
Giải phương trình $\sqrt{x^2 - 2x + 5} = x^2 - 2x + 3 $.
\loigiai{
$\sqrt{x^2 - 2x + 5} = x^2 - 2x + 3$.\hfill $(1)$\\
	Ta có $ x^2 - 2x + 5 = \left(x - 1\right)^2 + 4 \geqslant 4, \, \forall x $ nên phương trình $ (1) $ luôn có nghĩa.\\
	Đặt $ t = \sqrt{x^2 - 2x + 5} \quad \left(t \geqslant 2 \right) $. Ta được
	\begin{eqnarray*}
	t = t^2 - 2 \Leftrightarrow t^2 - t - 2 = 0 \Leftrightarrow \hoac{&t = -1 \textrm{\,(loại)}\\&t = 2} \Leftrightarrow t = 2 \Leftrightarrow \sqrt{x^2 - 2x + 5} = 2 \Leftrightarrow x = 1.
	\end{eqnarray*}
	Vậy phương trình có nghiệm $ x = 1 $.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Thượng Hiền, Hồ Chí Minh 17-18]%[Lục Hán Tường]%[0D3K2-4]
Giải phương trình $ \sqrt{2x - 7} + x^3 - \sqrt{x} = 343 $.
\loigiai{
$\sqrt{2x - 7} + x^3 - \sqrt{x} = 343$.\hfill $(1)$\\
	Điều kiện $ \heva{&2x - 7 \geqslant 0 \\& x \geqslant 0} \Leftrightarrow \heva{& x \geqslant \dfrac{7}{2}\\& x \geqslant 0} \Leftrightarrow x \geqslant \dfrac{7}{2}$.\hfill $(*)$
	\begin{eqnarray*}
	(1) &  \Leftrightarrow & \sqrt{2x - 7} - \sqrt{x} + x^3 - 343 = 0\\
	& \Leftrightarrow & \dfrac{2x - 7 - x}{\sqrt{2x - 7} + \sqrt{x}} + \left(x - 7\right)\left(x^2 + 7x + 49\right) = 0\\
	& \Leftrightarrow & \left(x - 7\right)\left(\dfrac{1}{\sqrt{2x - 7} + \sqrt{x}} + x^2 + 7x + 49\right) = 0\\
	& \Leftrightarrow & \hoac{&x = 7\\&\dfrac{1}{\sqrt{2x - 7} + \sqrt{x}} + x^2 + 7x + 49 = 0. \quad \quad(2)}
\end{eqnarray*}
	Ta có
	\begin{eqnarray*}
	\heva{& \sqrt{2x - 7} + \sqrt{x} > 0 \Rightarrow \dfrac{1}{\sqrt{2x - 7} + \sqrt{x}} > 0\\
	& x^2 + 7x + 49 = \left(x + \dfrac{7}{2}\right)^2 + \dfrac{147}{4} > 0} \Rightarrow \dfrac{1}{\sqrt{2x - 7} + \sqrt{x}} + x^2 + 7x + 49 > 0.
	\end{eqnarray*}
	Suy ra phương trình $ (2) $ vô nghiệm. Vậy phương trình đã cho có nghiệm $ x = 7 $.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Thượng Hiền, Hồ Chí Minh 17-18]%[Lục Hán Tường]%[0D4B1-5]
Tìm giá trị nhỏ nhất của hàm số $ y = \dfrac{x}{2} + \dfrac{8}{x - 1} $ với $ x > 1 $.
\loigiai{
Do $ x > 1 \Rightarrow x - 1 > 0 $. Áp dụng bất đẳng thức Cauchy cho $ 2 $ số $ \dfrac{x - 1}{2} $ và $ \dfrac{8}{x - 1} $ ta được
\begin{eqnarray*}
&&\dfrac{x - 1}{2} + \dfrac{8}{x - 1} \geqslant 2\sqrt{\dfrac{x - 1}{2} \cdot \dfrac{8}{x - 1}}\\
& \Leftrightarrow & \dfrac{x}{2} - \dfrac{1}{2} + \dfrac{8}{x - 1}  \geqslant 4\\
& \Leftrightarrow & y - \dfrac{1}{2}  \geqslant 4\\
& \Leftrightarrow & y  \geqslant \dfrac{9}{2}.
\end{eqnarray*}
Dấu \lq\lq=\rq\rq\, xảy ra khi và chỉ khi $\dfrac{x - 1}{2} = \dfrac{8}{x - 1} \Leftrightarrow \left(x - 1\right)^2 = 16 \Leftrightarrow x = 5 $ (loại nghiệm $ x = -3 $ vì $ x > 1 $).\\
Vậy $ \min y = \dfrac{9}{2} $ khi $ x = 5 $.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Thượng Hiền, Hồ Chí Minh 17-18]%[Lục Hán Tường]%[0H2Y2-3]%[0H2B2-3]
Trong mặt phẳng tọa độ $ Oxy $, cho tam giác $ ABC $ biết $ A\left(0;4\right) $; $ B\left(-6;1\right) $; $ C\left(-2;8\right) $.
\begin{enumerate}
	\item Chứng minh tam giác $ ABC $ là tam giác vuông. Tìm tọa độ tâm và tính bán kính đường tròn ngoại tiếp tam giác $ ABC $.
	\item Tìm tọa độ điểm $ M $ thuộc $ Ox $ sao cho tam giác $ MAB $ vuông tại $ M $.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có
	\begin{eqnarray*}
	& \vec{AB} & =  \left(-6;-3\right) \Rightarrow AB = \sqrt{\left(-6\right)^2 + \left(-3\right)^2} = \sqrt{45},\\
	& \vec{BC} & =  \left(4;7\right) \Rightarrow BC = \sqrt{\left(4\right)^2 + \left(7\right)^2} = \sqrt{65},\\
	& \vec{AC} & =  \left(-2;4\right) \Rightarrow AC = \sqrt{\left(-2\right)^2 + \left(4\right)^2} = \sqrt{20}.
	\end{eqnarray*}
	Do đó
	\begin{eqnarray*}
	AB^2 + AC^2 = 45 + 20 = 65 = BC^2.
	\end{eqnarray*}
	Suy ra tam giác $ ABC $ vuông tại $ A $ $\Rightarrow $ đường tròn ngoại tiếp tam giác $ ABC $ có tâm là trung điểm $ BC $ và bán kính bằng $ R = \dfrac{BC}{2} $, suy ra $I\left(-4;\dfrac{9}{2}\right)$ và $R = \dfrac{\sqrt{65}}{2}$.
	
	\item Gọi $ M(m;0) $ là điểm thuộc $ Ox $. Ta có
	\begin{eqnarray*}
	&\vec{AM} & = \left(m;-4\right),\\
	&\vec{BM} & = \left(m + 6;-1\right).
	\end{eqnarray*}
	Tam giác $ AMB $ vuông tại $ M $ khi
	\begin{eqnarray*}
	\vec{AM} \perp \vec{BM} \Leftrightarrow \vec{AM} \cdot \vec{BM} = 0 \Leftrightarrow m\left(m + 6\right) + 4 = 0 \Leftrightarrow \hoac{&m=-3 + \sqrt{5}\\&m=-3 - \sqrt{5}.}
	\end{eqnarray*}
	Vậy có $ 2 $ điểm $ M $ thỏa bài toán $ M_1\left(-3 + \sqrt{5};0\right) $ và $ M_2\left(-3 - \sqrt{5};0\right) $.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Thượng Hiền, Hồ Chí Minh 17-18]%[Lục Hán Tường]%[0H2Y3-1]%[0H2K3-1]%[0H2G2-4]
Cho tam giác $ ABC $ có $ AB = 2 $; $ AC = 2\sqrt{7} $ và $ BC = 4 $.
\begin{enumerate}
	\item Tính góc $ B $; bán kính đường tròn ngoại tiếp tam giác $ ABC $; diện tích tam giác $ ABC $.
	\item Tính độ dài đường phân giác trong của góc $ B $ của tam giác $ ABC $.
	\item Tìm tập hợp các điểm $ M $ thỏa $ \left(\vec{MA} + 2\vec{MB}\right)\left(\vec{AB} - \vec{AC}\right) = 0 $.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Áp dụng định lý cosin trong tam giác $ ABC $
	\begin{eqnarray*}
	\cos B = \dfrac{AB^2 + BC^2 - AC^2}{2\cdot AB \cdot BC}  =\dfrac{4 + 16 - 28}{2\cdot 2\cdot 4} = \dfrac{-1}{2} \Rightarrow B = 120^\circ.
	\end{eqnarray*}
	Áp dụng định lý sin trong tam giác $ ABC $
	\begin{eqnarray*}
	\dfrac{AC}{\sin B} = 2R \Rightarrow R = \dfrac{AC}{2\sin B} = \dfrac{2\sqrt{7}}{2\sin 120^\circ} = \dfrac{2\sqrt{7}}{2\cdot \dfrac{\sqrt{3}}{2}} = \dfrac{2\sqrt{21}}{3}.
	\end{eqnarray*}
	Diện tích tam giác $ ABC $
	\begin{eqnarray*}
	\mathrm{S}_{\triangle ABC} = \dfrac{AB\cdot BC \cdot AC}{4R} = \dfrac{2 \cdot 2\sqrt{7} \cdot 4}{4\cdot \dfrac{2\sqrt{21}}{3}} = 2\sqrt{3}.
	\end{eqnarray*}
	
	\item Gọi $ D $ là chân đường phân giác góc trong $ B $. Ta có
	\begin{eqnarray*}
	&&\mathrm{S}_{\triangle ABC} = \mathrm{S}_{\triangle BDA} + \mathrm{S}_{\triangle BDC} \\
	& \Leftrightarrow & \mathrm{S}_{\triangle ABC} = \dfrac{1}{2}BC\cdot BD \sin \widehat{CBD} + \dfrac{1}{2}BA\cdot BD \sin \widehat{ABD} \\
	& \Leftrightarrow & 2\sqrt{3} = \dfrac{1}{2}\cdot 4\cdot BD \sin 60^\circ + \dfrac{1}{2}\cdot 2 \cdot BD \sin 60^\circ \\
	& \Leftrightarrow & 2\sqrt{3} = \dfrac{3BD\sqrt{3}}{2}\\
	& \Leftrightarrow & BD = \dfrac{4}{3}.
	\end{eqnarray*}
	
	\item Gọi $ I $ là điểm thỏa $ \vec{IA} + 2\vec{IB} = \vec{0} \Leftrightarrow 3\vec{BI} = \vec{BA} \Rightarrow I $ là điểm cố định.\\
	Ta có
	\begin{eqnarray*}
	&&\left(\vec{MA} + 2\vec{MB}\right)\left(\vec{AB} - \vec{AC}\right) = 0\\
	& \Leftrightarrow & \left(3\vec{MI} + \vec{IA} + 2\vec{IB}\right)\vec{CB} = 0\\
	& \Leftrightarrow & 3\vec{MI} \cdot \vec{CB} = 0\\
	& \Leftrightarrow & \vec{MI} \perp \vec{CB}.
	\end{eqnarray*}
	Vậy tập hợp điểm $ M $ là đường thẳng qua $ I $ và vuông góc với $ BC $.
\end{enumerate}
}
\end{bt}
