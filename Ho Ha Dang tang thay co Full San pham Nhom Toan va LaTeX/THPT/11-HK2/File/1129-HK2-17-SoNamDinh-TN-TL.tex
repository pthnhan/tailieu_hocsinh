	Giải phương trình $\cos 2x+2\cos x-3=0$.
	\choice
	{$x=\pi+k2\pi,k\in\mathbb{Z}$}
	{\True $x=k2\pi,k\in\mathbb{Z}$}
	{$x=-\dfrac{\pi}{2}+k2\pi,k\in\mathbb{Z}$}

		$\cos 2x+2\cos x-3=0\Leftrightarrow 2\cos^2x+2\cos x-4=0\Leftrightarrow \hoac{&\cos x=1\\& \cos x=-2\;\text{(loại)}}\Leftrightarrow x=k2\pi,k\in\mathbb{Z}$.

\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[HK2 (2016-2017), Sở GD-ĐT Nam Định]%[1D4K2-7]%%[1D4K2-3]%[Nguyễn Khắc Hưởng]
	Tính các giới hạn sau:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\displaystyle\lim_{x\to +\infty}\dfrac{(x+1)(x^2-2)}{2x^3+x+1}$.
			\item $\displaystyle\lim_{x\to 1}\dfrac{\sqrt{x+3}-3x+1}{x^2+x-2}$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item $\displaystyle\lim_{x\to +\infty}\dfrac{(x+1)(x^2-2)}{2x^3+x+1}=\displaystyle\lim_{x\to +\infty}\dfrac{\left(1+\dfrac{1}{x}\right)\left(1-\dfrac{2}{x^2}\right)}{2+\dfrac{1}{x^2}+\dfrac{1}{3}}=\dfrac{1}{2}$.
			\item $\displaystyle\lim_{x\to 1}\dfrac{\sqrt{x+3}-3x+1}{x^2+x-2}=\displaystyle\lim_{x\to 1}\dfrac{\sqrt{x+3}-2-3(x-1)}{(x-1)(x+2)}=\displaystyle\lim_{x\to 1}\left[\dfrac{\sqrt{x+3}-2}{(x-1)(x+2)}-\dfrac{3}{x+2}\right]$\\
			$=\displaystyle\lim_{x\to 1}\left[\dfrac{(x-1)}{(x-1)(x+2)(\sqrt{x+3}+2)}-\dfrac{3}{x+2}\right]=\displaystyle\lim_{x\to 1}\left[\dfrac{1}{(x+2)(\sqrt{x+3}+2)}-\dfrac{3}{x+2}\right]=-\dfrac{11}{12}$.
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[HK2 (2016-2017), Sở GD-ĐT Nam Định]%[1D4B3-3]%[Nguyễn Khắc Hưởng]
	Cho hàm số $f(x)=\heva{&\dfrac{3x^3-x-2}{x-1}&\text{nếu}\; x\neq 1\\ &m-2x &\text{nếu } x=1}.$ Tìm tất cả các giá trị của tham số $m$ để hàm số đã cho liên tục tại $x=1$.
	\loigiai{
	\begin{itemize}
	\item Hàm số xác định tại $x=1$ và $f(1)=m-2$.
	\item $\displaystyle\lim_{x\to 1}f(x)=\displaystyle\lim_{x\to 1}\dfrac{(x-1)(3x^2+3x+2)}{x-1}=\displaystyle\lim_{x\to 1}(3x^2+3x+2)=8$.
	\item Hàm số liên tục tại $x=1$ khi và chỉ khi $\displaystyle\lim_{x\to 1}f(x)=f(1)\Leftrightarrow 8=m-2\Leftrightarrow m=10$.
	\end{itemize}
	}
\end{bt}

\begin{bt}%[HK2 (2016-2017), Sở GD-ĐT Nam Định]%[1D5K3-1]%[Nguyễn Khắc Hưởng]
	Cho hàm số $f(x)=\sin 2x+\sqrt{3}\cos 2x+12\sin \left(x+\dfrac{\pi}{6}\right)$. Giải phương trình $f'(x)+4=0$.
	\loigiai{
	\begin{itemize}
	\item $f'(x)=2\cos 2x-2\sqrt{3}\sin 2x+12\cos\left(x+\dfrac{\pi}{6}\right).$
	\item $f'(x)+4=0\Leftrightarrow \dfrac{1}{2}\cos 2x-\dfrac{\sqrt{3}}{2}\sin 2x+3\cos \left(x+\dfrac{\pi}{6}\right)+1=0$\\
	$\Leftrightarrow \cos\left(2x+\dfrac{\pi}{3}\right)+3\cos\left(x+\dfrac{\pi}{6}\right)+1=0\Leftrightarrow 2\cos^2\left(x+\dfrac{\pi}{6}\right)+3\cos\left(x+\dfrac{\pi}{6}\right)=0$\\
	$\Leftrightarrow \cos\left(x+\dfrac{\pi}{6}\right)=0$ (vì $\cos\left(x+\dfrac{\pi}{6}\right)\in [-1;1]$).\\
	$\Leftrightarrow x+\dfrac{\pi}{6}=\dfrac{\pi}{2}+k\pi\Leftrightarrow x=\dfrac{\pi}{3}+k\pi, k\in\mathbb{Z}$.
	\end{itemize}
	}
\end{bt}

\begin{bt}%[HK2 (2016-2017), Sở GD-ĐT Nam Định]%[1D5K2-3]%[Nguyễn Khắc Hưởng]
	Viết phương trình tiếp tuyến của đồ thị hàm số $y=x^3+3x+2,$ biết tiếp tuyến đó vuông góc với đường thẳng $\Delta: x+6y+6=0$.
	\loigiai{
	\begin{itemize}
	\item $y'=3x^2+3$
	\item Đường thẳng $\Delta: y=-\dfrac{1}{6}x-1$ có hệ số góc $k=-\dfrac{1}{6}$. Gọi $M(x_0;y_0)$ là tiếp điểm. Ta có $y'(x_0)\cdot \left(-\dfrac{1}{6}\right)=-1\Leftrightarrow (3x_0^2+3)\left(-\dfrac{1}{6}\right)=-1\Leftrightarrow \hoac{&x_0=1\\&x_0=-1}$.
	\item Với $x_0=1\Rightarrow y_0=6\Rightarrow M(1;6).$ Tiếp tuyến tại $M$ là: $\Delta_1:y=6x.$
	\item Với $x_0=-1\Rightarrow y_0=-2\Rightarrow M(-1;-2).$ Tiếp tuyến tại $M$ là: $\Delta_2:y=6x+4.$
	\end{itemize}
	}
\end{bt}

\begin{bt}%[HK2 (2016-2017), Sở GD-ĐT Nam Định]%[1H3B3-2][1H3G3-3]%[Nguyễn Khắc Hưởng]
	Cho hình chóp $S.ABCD$ có đáy là hình vuông có cạnh bằng $a\sqrt{2}$; $SA\perp (ABCD)$ và $SA=2a$. Gọi $E$ là hình chiếu vuông góc của $A$ trên cạnh $SB$.
		\begin{enumerate}
			\item Chứng minh $BD\perp (SAC).$
			\item Chứng minh $BC\perp (SAB)$ và $(AEC)\perp (SBC)$.
			\item Gọi $G$ và $K$ lần lượt là trọng tâm của các tam giác $SAD$ và $ACD$. Tính góc giữa đường thẳng $GK$ và mặt phẳng $(SAB)$.
		\end{enumerate}
	\loigiai{
		\immini{
		a) Chứng minh $BD\perp (SAC).$\\
$\heva{BD\perp AC\\BD\perp SA}\Rightarrow BD\perp (SAC)$.\\
		b) Chứng minh $BC\perp (SAB)$ và $(AEC)\perp (SBC)$.\\
			$\heva{BC\perp AB\\BC\perp SA}\Rightarrow BC\perp (SAB)$.\\
			$\heva{AE\perp SB\\AE\perp BC}\Rightarrow AE\perp (SBC)$.\\
			 Lại có $AE\subset (ACE)$. Do đó $(AEC)\perp (SBC)$.
	}

	c) Tính góc giữa đường thẳng $GK$ và mặt phẳng $(SAB)$.\\
	Gọi $I$ là trung điểm $AD$. Ta có $\dfrac{IG}{IS}=\dfrac{IK}{IC}=\dfrac{1}{3}\Rightarrow GK \parallel SC$. Suy ra $\widehat{GK,(SAB)}=\widehat{SC,(SAB)}$.\\
	Lại có $SB$ là hình chiếu của $SC$ trên $(SAB)$ nên $\widehat{SC,(SAB)}=\widehat{SC,SB}=\widehat{BSC}$ (do $\Delta SBC$ vuông tại $B$ nên $\widehat{BSC}$ nhọn).\\
	Ta có $AC=2a$, $\Delta SAC$ vuông tại $A\Rightarrow SC=\sqrt{SA^2+AC^2}=2\sqrt{2}a$.\\
	Lại có $\Delta SAB$ vuông tại $A\Rightarrow SB=\sqrt{SA^2+AB^2}=a\sqrt{6}$.\\ $\Delta SBC$ vuông tại $B$	, ta có $\cos\widehat{BSC}=\dfrac{SB}{SC}=\dfrac{\sqrt{3}}{2}\Rightarrow \widehat{BSC}=30^\circ$.\\
	Vậy góc giữa đường thẳng $GK$ và mặt phẳng $(SAB)$ bằng $30^\circ$.
