\subsubsection{Khoảng cách từ một điểm đến mặt phẳng}
\begin{dang}{Tính khoảng cách từ chân đường cao đến mặt bên của hình chóp}
	\textbf{Bài toán.} Cho hình chóp $S.ABC$ có cạnh bên $SA$ vuông góc với đáy. Tính khoảng cách từ $A$ đến $(SBC)$.\\
	\textbf{Lời giải}
	\immini{
		Kẻ $AK \perp BC$ tại $K$. Khi đó $BC \perp (SAK)$, suy ra $(SBC) \perp (SAK)$.\\
		Lại có $(SBC) \cap (SAK)=SK$. Kẻ $AH \perp SK$ tại $H$. Suy ra $AH \perp (SBC)$.\\
		Vậy $\mathrm{d}\left(A,(SBC)\right) = AH =d$.\\
		Công thức tính \fbox{$\dfrac{1}{d^2} = \dfrac{1}{SA^2}+\dfrac{1}{AK^2}$}\\
		* \textbf{Trường hợp đặc biệt:} Nếu $AS$, $AB$, $AC$ đôi một vuông góc thì ta có công thức tính
		\begin{center}
			\fbox{$\dfrac{1}{d^2}=\dfrac{1}{SA^2}+\dfrac{1}{AB^2}+\dfrac{1}{AC^2}$}
		\end{center}
	}
	{
		\begin{tikzpicture}
		\tkzDefPoints{0/0/A, 1.5/-2/B, 5/0/C}
		\tkzDefShiftPoint[A](90:3.5){S}
		\tkzDefBarycentricPoint(B=3,C=1)\tkzGetPoint{K}
		\tkzDefBarycentricPoint(S=2,K=3)\tkzGetPoint{H}
		\tkzDrawPoints[fill=black](A,B,C,H,K,S)
		\tkzDrawSegments(S,A A,B B,C S,B S,C S,K)
		\tkzDrawSegments[dashed](A,C A,K A,H)
		\tkzMarkRightAngles(A,H,K A,K,B)
		\tkzLabelPoints[above](S)
		\tkzLabelPoints[below](B)
		\tkzLabelPoints[left](A)
		\tkzLabelPoints[right](H,K,C)
		\end{tikzpicture}
	}
\end{dang}

\begin{vd}%[Cao Thành Thái, dự án EX-CTGN]%[1H3B5-3]
	Cho hình chóp $S.ABC$ có đáy $ABC$ là tam giác vuông cân tại $A$, $AB=SA=a$. Cạnh bên $SA$ vuông góc với đáy. Khoảng cách từ $A$ đến $(SBC)$ bằng
	\choice
	{\True $\dfrac{a\sqrt{3}}{3}$}
	{$\dfrac{a\sqrt{3}}{2}$}
	{$\dfrac{a\sqrt{2}}{3}$}
	{$\dfrac{a\sqrt{2}}{2}$}

	{
		\immini{
			Gọi $d$ là khoảng cách từ $A$ đến $(SBC)$. Do $SA$, $AB$, $AC$ đôi một vuông góc nên ta áp dụng công thức
			\begin{eqnarray*}
				&& \dfrac{1}{d^2}=\dfrac{1}{SA^2}+\dfrac{1}{AB^2}+\dfrac{1}{AC^2} = \dfrac{1}{a^2}+\dfrac{1}{a^2}+\dfrac{1}{a^2}=\dfrac{3}{a^2}\\
				&\Rightarrow & d=\sqrt{\dfrac{a^2}{3}}=\dfrac{a\sqrt{3}}{3}.
			\end{eqnarray*}
		}

\begin{dang}{Tính khoảng cách từ điểm đến mặt phẳng bằng tỉ số khoảng cách}
	\textbf{Công thức tỉ số khoảng cách:}\\
	Tính khoảng cách từ điểm $M$ đến mặt phẳng $(\alpha)$ khi biết $\mathrm{d}\left(A,(\alpha)\right)$.
	\begin{itemize}
		\item Nếu $A$, $M$ cùng nằm trên một đường thẳng song song với $(\alpha)$ thì $$\mathrm{d}\left(M,(\alpha)\right)=\mathrm{d}\left(A,(\alpha)\right).$$
		\item Nếu $A$, $M$ cùng nằm trên một đường thẳng cắt $(\alpha)$ tại $I$ thì $$\dfrac{\mathrm{d}\left(M,(\alpha)\right)}{\mathrm{d}\left(A,(\alpha)\right)} = \dfrac{IM}{IA}.$$
	\end{itemize}

		Kẻ tia $Cx\parallel BD$; kẻ $AH\perp Cx$ tại $H$. Suy ra $BD\parallel(SCH)$.\\
		Khi đó $\mathrm{d}(BD,SC)=\mathrm{d}(BD,(SCH))=\mathrm{d}(O,(SCH))=\dfrac{1}{2}\mathrm{d}(A,(SCH))$.\\
		Kẻ $AK\perp SH$ tại $K$. Suy ra $AK\perp (SHC)$. Do đó ${\rm d}\left(A,(SCH)\right)=AK$.\\
		Áp dụng định lý sin trong $\triangle AOB$: $\dfrac{AB}{\sin \widehat{AOB}}=\dfrac{AO}{\sin \widehat{ABO}}$
		$\Rightarrow \sin \widehat{AOB}=\dfrac{AB}{AO}\cdot \dfrac{AD}{BD}=\dfrac{4}{5}$.\\
		Ta có $AH=AC\cdot \sin \widehat{ACH}=AC\cdot \sin \widehat{AOB}=\dfrac{4a\sqrt{5}}{5}$.\\
		$\triangle SAH$ vuông tại $A$ có $AK$ là đường cao 
		$\Rightarrow AK=\dfrac{SA \cdot AH}{\sqrt{SA^2+AH^2}}=\dfrac{4a\sqrt{7}}{7}$.\\
		Vậy ${\rm d}(BD,SC)=\dfrac{1}{2}AK=\dfrac{2a\sqrt{7}}{7}$.
	}
