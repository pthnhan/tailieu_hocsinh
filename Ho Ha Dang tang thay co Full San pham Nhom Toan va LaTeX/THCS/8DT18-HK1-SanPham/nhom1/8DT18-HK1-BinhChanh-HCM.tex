\begin{bt}%[Đề kiểm tra Học kỳ 1 môn Toán 8 năm học 2017-2018, Bình Chánh, Tp. Hồ Chí Minh]%[Nguyễn Ngọc Hóa, dự án 8DT18]%[8D1Y1]
	Thực hiện phép tính: $(x+2)^2-x(x+5)$.
	\loigiai{
	$(x+2)^2-x(x+5)=x^2+4x+4-x^2-5x=-x+4$.
}
\end{bt}

\begin{bt}%[Đề kiểm tra Học kỳ 1 môn Toán 8 năm học 2017-2018, Bình Chánh, Tp. Hồ Chí Minh]%[Nguyễn Ngọc Hóa, dự án 8DT18]%[8D2B6]
	Thực hiện phép tính: $\dfrac{2}{x+3}-\dfrac{3}{3-x}+\dfrac{2-5x}{x^2-9}$.
	\loigiai{
		Điều kiện của phân thức: $x^2-9\ne 0\Leftrightarrow x\ne \pm 3$.\\
		Với $x\ne \pm 3$, ta có
		\begin{eqnarray*}
			\dfrac{2}{x+3}-\dfrac{3}{3-x}+\dfrac{2-5x}{x^2-9}&= &\dfrac{2}{x+3}+\dfrac{3}{x-3}+\dfrac{2-5x}{(x-3)(x+3)}\notag\\
			&= &\dfrac{2(x-3)+3(x+3)+2-5x}{(x-3)(x+3)}\notag\\
			&= &\dfrac{2x-6+3x+9+2-5x}{(x-3)(x+3)}\notag\\
			&= &\dfrac{5}{(x-3)(x+3)}.\notag
		\end{eqnarray*}	
	}
\end{bt}

\begin{bt}%[Đề kiểm tra Học kỳ 1 môn Toán 8 năm học 2017-2018, Bình Chánh, Tp. Hồ Chí Minh]%[Nguyễn Ngọc Hóa, dự án 8DT18]%[8D1B7]
	Phân tích các đa thức sau thành nhân tử
	\begin{enumerate}
		\begin{multicols}{2}
		\item $x(2x-3)-2(3-2x)$;
				\item $x^2-4y^2-2x+4y$.
		\end{multicols}


	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $x(2x-3)-2(3-2x)=x(2x-3)+2(2x-3)=(2x-3)(x+2)$.
	\item $x^2-4y^2-2x+4y=[x^2-(2y)^2]-(2x-4y)=(x-2y)(x+2y)-2(x-2y)=(x-2y)(x+2y-2)$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề kiểm tra Học kỳ 1 môn Toán 8 năm học 2017-2018, Bình Chánh, Tp. Hồ Chí Minh]%[Nguyễn Ngọc Hóa, dự án 8DT18]%[8D1B5]
	Tìm $x$, biết: $(x+3)^2-(x+2)(x-2)=0$.
	\loigiai{
	\begin{eqnarray*}
	(x+3)^2-(x+2)(x-2)=0&\Leftrightarrow &x^2+6x+9-x^2+4=0\notag\\
	&\Leftrightarrow &6x+13=0\notag\\
	&\Leftrightarrow &6x=-13\notag\\
	&\Leftrightarrow &x=\dfrac{-13}{6}.\notag\\
	\end{eqnarray*}
Vậy $x=\dfrac{-13}{6}$.
}
\end{bt}

\begin{bt}%[Đề kiểm tra Học kỳ 1 môn Toán 8 năm học 2017-2018, Bình Chánh, Tp. Hồ Chí Minh]%[Nguyễn Ngọc Hóa, dự án 8DT18]%[8D1G5]
	Tìm giá trị nhỏ nhất của biểu thức $A=x^2+2xy+2y^2-4y+3$.
	\loigiai{
	Ta có $A=x^2+2xy+2y^2-4y+3=(x^2+2xy+y^2)+(y^2-4y+4)-1=(x+y)^2+(y-2)^2-1$.\\
	Vì $(x+y)^2\ge 0$ và $(y-2)^2\ge 0$ nên $A\ge 0+0-1$ hay $A\ge -1$.\\
	Dấu ``$=$'' xảy ra khi $x+y=0$ và $y-2=0$ hay $x=-2$, $y=2$.\\
	Vậy $A$ đạt giá trị nhỏ nhất bằng $-1$ khi $x=-2$, $y=2$.
}
\end{bt}

\begin{bt}%[Đề kiểm tra Học kỳ 1 môn Toán 8 năm học 2017-2018, Bình Chánh, Tp. Hồ Chí Minh]%[Nguyễn Ngọc Hóa, dự án 8DT18]%[8D2B1]
	Một quyển vở mua lẻ có giá là $x$ đồng nhưng khi mua sỉ (mua từ $10$ quyển trở lên) thì mỗi quyển có giá rẻ hơn khi mua lẻ là $500$ đồng. Cô Nga dùng $810000$ đồng để mua vở khen thưởng cho học sinh lớp mình chủ nhiệm.
	\begin{enumerate}
		\item Hãy viết biểu thức theo $x$ số quyển vở cô Nga mua được theo giá sỉ.
		\item Với số tiền trên nếu mua theo giá sỉ thì sẽ thu lợi hơn mua theo giá bán lẻ bao nhiêu quyển vở? Biết rằng giá một quyển vở khi mua lẻ là $5000$ đồng.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Giá một quyển vở khi mua theo giá sỉ là $x-500$ (đồng).\\
	Số quyển vở cô Nga mua được theo giá sỉ là $\dfrac{810000}{x-500}$.
	\item Số quyển vở khi cô Nga mua theo giá bán lẻ là
	\begin{center}
		$810000:5000=162$ (quyển)
	\end{center}
Số quyển vở khi cô Nga mua theo giá bán sỉ là
\begin{center}
	$810000:4500=180$ (quyển)
\end{center}
Vậy khi mua theo giá sỉ, cô Nga thu lợi được $180-162=18$ quyển vở.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề kiểm tra Học kỳ 1 môn Toán 8 năm học 2017-2018, Bình Chánh, Tp. Hồ Chí Minh]%[Nguyễn Ngọc Hóa, dự án 8DT18]%[8H1B9]%[8H1B1]%[8H1K9]%[8H1G9]
	Cho tam giác $ABC$ vuông tại $A$ có $AB=6$ cm, $AC=8$ cm, $AM$ là đường trung tuyến.
	\begin{enumerate}
		\item Tính độ dài đoạn thẳng $AM$.
		\item Từ $M$ vẽ $MK$ vuông góc với $AB$ ($K\in AB$), $MN$ vuông góc với $AC$ ($N\in AC$). Chứng minh: $AKMN$ là hình chữ nhật.
		\item Chứng minh $KMCN$ là hình bình hành.
		\item Vẽ $AH$ vuông góc với $BC$. Chứng minh $KHMN$ là hình thang cân.
	\end{enumerate}
\loigiai{
\immini
{
\begin{enumerate}
	\item Theo định lý Py-ta-go ta có
	\begin{eqnarray*}
		&BC^2&=AB^2+AC^2=6^2+8^2=100\notag\\
		\Rightarrow &BC &=10 \text{ (cm)}\notag
	\end{eqnarray*}
	Do tam giác $ABC$ vuông tại $A$ có $AM$ là đường trung tuyến nên $AM=\dfrac{1}{2}BC$.\\
	$\Rightarrow AM=\dfrac{1}{2}\cdot 10=5$ (cm).
	\item Xét tứ giác $AKMN$ có $\widehat{K}=\widehat{A}=\widehat{N}=90^\circ$ nên $AKMN$ là hình chữ nhật.
\end{enumerate}	
}
{
	\begin{tikzpicture}[>=stealth,scale=1, line join = round, line cap = round]
	\tkzDefPoints{0/0/A,4/0/C,0/3/B}
	\coordinate (M) at ($(B)!0.5!(C)$);
	\tkzDrawAltitude(A,B)(M) \tkzGetPoint{K}
	\tkzDrawAltitude(A,C)(M) \tkzGetPoint{N}
	\tkzDrawAltitude(B,C)(A) \tkzGetPoint{H}
	\tkzMarkRightAngles[size=0.2,fill=gray!50](A,H,C B,A,C M,K,A M,N,C)
	\tkzDrawSegments(A,B B,C C,A A,M M,K M,N K,H K,N)
	\tkzDrawPoints[fill=black](A,B,C,M,K,N,H)
	\tkzLabelPoints[left](A,K)
	\tkzLabelPoints[above](B)
	\tkzLabelPoints[above right](M,H)
	\tkzLabelPoints[right](C)
	\tkzLabelPoints[below](N)
	\end{tikzpicture}
}
\begin{enumerate}[c)]
	\item Trong $\triangle ABC$ ta có
	\begin{itemize}
		\item $M$ là trung điểm của $BC$ (giả thiết);
		\item $MK\parallel AC$ ($AKMN$ là hình chữ nhật)
	\end{itemize}
	Do đó $K$ là trung điểm của $AB$ (định lý đường trung bình trong tam giác).\\
	$\Rightarrow KM=\dfrac{1}{2}AC$ \hfill $(1)$\\
	Hoàn toàn tương tự, $N$ cũng là trung điểm của $AC$ nên $NC=\dfrac{1}{2}AC$. \hfill $(2)$\\
	Từ (1) và (2) suy ra $KM=NC$ (cùng bằng $\dfrac{1}{2}AC$) và $KM\parallel NC$ (chứng minh trên).\\
	Vậy tứ giác $KMCN$ là hình bình hành.
	\item[d)] Vì $KMCN$ là hình bình hành nên $KN\parallel MC$ hay $KN\parallel MH$.\\ Do đó tứ giác $KHMN$ là hình thang.\hfill $(3)$\\
	Mặt khác, $\triangle AHC$ vuông tại $H$ có $HN$ là đường trung tuyến nên \begin{center}
		$HN=\dfrac{1}{2}AC$.
	\end{center}
Lại có $KM=AN=\dfrac{1}{2}AC$ ($AKMN$ là hình chữ nhật)\\
Suy ra $KM=HN$ (cùng $=\dfrac{1}{2}AC$) \hfill $(4)$\\
Từ (3) và (4) suy ra $KHMN$ là hình thang cân.
\end{enumerate}	
}
