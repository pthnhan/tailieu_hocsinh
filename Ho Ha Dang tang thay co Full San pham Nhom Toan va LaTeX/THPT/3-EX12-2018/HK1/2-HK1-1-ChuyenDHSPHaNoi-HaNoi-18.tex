\begin{ex}%[HK1, Chuyên ĐHSP Hà Nội, 2018]%[2H1B3]

\dfrac{1}{AH^2}&=\dfrac{1}{AB^2}+\dfrac{1}{AC^2}\\
\Rightarrow AH&=\dfrac{2a\sqrt 2}{3}
\end{align*}

\end{align*}
