\immini{

\begin{tikzpicture}[line join=round, line cap=round,>=stealth,scale=1]
	\def\xmin{-1}\def\xmax{3}\def\ymin{-2}\def\ymax{2}
	\draw[thick,->] (\xmin-0.6,0)--(\xmax+0.8,0) node[below] {\footnotesize $x$};
	\draw[thick,->] (0,\ymin-0.6)--(0,\ymax+0.6) node[right] {\footnotesize $y$};
	\draw (0,0) node [below left] {\footnotesize $O$};
	\foreach \x in {\xmin,...,\xmax}{\ifthenelse{\not \x=0}{
			\draw (\x,0.1)--(\x,-0.1)  ;}{}}
	\foreach \y in {\ymin,...,\ymax} {
		\ifthenelse{\not \y=0}{\draw (0.1,\y)--(-0.1,\y)  ;}{} }
	\clip (\xmin-0.5,\ymin-0.5) rectangle (\xmax+0.5,\ymax+0.5);
	\draw[smooth,samples=200,thick,domain=0:4,color=blue] plot (\x,{1/(5*((\x)-2)^(2)+1)-1/21});
	\draw[smooth,samples=200,thick,domain=0:4,color=red] plot (\x,{(-(10*(\x)-20))/(5*((\x)-2)^(2)+1)^(2)});
	\draw[smooth,samples=200,thick,domain=0:4] plot (\x,{(1/sqrt(5))*rad(atan((\x-2)*sqrt(5)))-\x/21-(1/sqrt(5))*rad(atan(-2*sqrt(5))))});
	
	%Ghi chú
	\node [above] at (1.7,1.35) {$ (a) $};
	\node [above] at (3.2,0.2) {$ (b) $};
	\node [above] at (3.2,1) {$ (c) $};
	\end{tikzpicture}
}
\choice
{$(b), (c), (a)$}
{$(c), (a), (b)$}
{$(a), (c), (b)$}
{\True $(c), (b), (a)$}
\loigiai{Ta thấy 
\begin{itemize}  
\item Do $(c)$ luôn nằm phía trên $Ox$, hai đường $(b)$, $(c)$ có cả đi lên, đi xuống nên $(c)$ không thể là đồ thị của $f'(x)$ hay $f"(x)$. Do đó $(c)$ là của $f(x)$.
\item Đường $b$ cũng nằm trên $Ox$ nên $(b)$, $(a)$ có lúc lên lúc xuống nên $b$ phải là của $f'(x)$.
Do vậy theo thứ tự đồ thị của $f(x),f'(x),f"(x)$ là $(c)$, $(b)$, $(a)$.
\end{itemize}
\begin{note}
{\bf Lưu ý:} Cũng có thể chú ý: hoành độ giao điểm của $a$ với $Ox$ là điểm cực trị của hàm số có đồ thị  $b$ nên hàm số có đồ thị $a$ là đạo hàm của hàm số có đồ thị $b$. Do tính lồi lõm của $a,b$ nên $c$ không thể là $f"(x)$.
\end{note}}
\end{ex}

\immini{
Cho đồ thị của hàm số $y=f(x)$ và $y=f'(x)$ như hình vẽ bên dưới. Biết hàm số $y=f(x)$ có đạo hàm cấp hai trên $\mathbb{R}$. Khẳng định nào sau \textbf{đúng}?}

\def\xmin{-2}\def\xmax{2}\def\ymin{-2}\def\ymax{2}

	\ifthenelse{\not \y=0}{\draw (-0.1,\y)--(0.1,\y) node [right] {\footnotesize $\y$};}{} }

\draw[smooth,samples=200,thick,domain=0:3] plot (-\x,{3*\x^4-6*\x^2+1});

{\True $f'(-1)<f"(1)$}
{$f'(-1)>f"(1)$}
{$f'(-1)=f"(1)$}
{$f"(0)\ne f"(1)$}
\loigiai{
Trước hết ta phân biệt đường $y=f(x)$ và $y=f'(x)$. Dễ thấy đường $y=f'(x)$ phải là đường phía trên (có đồ thị giống hàm trùng phương).  Theo đó $B$, $C$ sai. Còn lại $D$ sai do $f"(0)=f"(1)=0$ vì $x=0,x=1$ là các điểm cực trị của $f'(x)$.
}
