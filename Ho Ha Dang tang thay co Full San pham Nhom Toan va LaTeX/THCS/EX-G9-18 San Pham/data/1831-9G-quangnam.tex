\begin{ex}%[Đề HSG9, Quảng Nam, 2017-2018]%[Phan Hoàng Anh]%[9D0K1]
	Cho biểu thức $A=\dfrac{x+8}{x\sqrt{x}+8}+\dfrac{1}{x-2\sqrt{x}+4}+\dfrac{\sqrt{x+4-4\sqrt{x}}}{x-4}$, với $0\le x\le4$.\\
	Rút gọn biểu thức $A$. Tìm các số nguyên $x$ để $A$ là số nguyên.
	\loigiai{Ta có $\begin{aligned}[t]
	A&=\dfrac{x+8}{x\sqrt{x}+8}+\dfrac{1}{x-2\sqrt{x}+4}+\dfrac{\sqrt{x+4-4\sqrt{x}}}{x-4}\\
	&=\dfrac{x+8}{\left(\sqrt{x}+2\right)\left(x-2\sqrt{x}+4\right)}+\dfrac{1}{x-2\sqrt{x}+4}+\dfrac{\sqrt{\left(\sqrt{x}-2\right)^2}}{\left(\sqrt{x}-2\right)\left(\sqrt{x}+2\right)}\\
	&=\dfrac{x+8+\sqrt{x}+2}{\left(\sqrt{x}+2\right)\left(x-2\sqrt{x}+4\right)}+\dfrac{2-\sqrt{x}}{\left(\sqrt{x}-2\right)\left(\sqrt{x}+2\right)}\\
	&=\dfrac{x+\sqrt{x}+10-\left(x-2\sqrt{x}+4\right)}{\left(\sqrt{x}+2\right)\left(x-2\sqrt{x}+4\right)}=\dfrac{3\left(\sqrt{x}+2\right)}{\left(\sqrt{x}+2\right)\left(x-2\sqrt{x}+4\right)}=\dfrac{3}{x-2\sqrt{x}+4}
		\end{aligned}$.\\
	Để $A$ là số nguyên thì $x-2\sqrt{x}+4$ là ước của $3$, mặt khác $x-2\sqrt{x}+4=\left(\sqrt{x}-1\right)^2+3\ge3$ nên suy ra $x-2\sqrt{x}+4=3\Leftrightarrow\sqrt{x}=1\Leftrightarrow x=1$.}
\end{ex}
\begin{ex}%[Đề HSG9, Quảng Nam, 2017-2018]%[Phan Hoàng Anh]%[9D0K7]
Cho ba số thực $a$, $b$, $c$ sao cho $1\le a\le2$, $1\le b\le2$, $1\le c\le2$. Chứng minh bất đẳng thức $\dfrac{a}{b}+\dfrac{b}{c}+\dfrac{c}{a}+\dfrac{a}{c}+\dfrac{c}{b}+\dfrac{b}{a}\le7$.
	\loigiai{Không mất tính tổng quát, giả sử $a\ge b\ge c$, ta có 
	$$\left(\dfrac{a}{b}-1\right)\left(\dfrac{b}{c}-1\right)\ge0\Rightarrow\dfrac{a}{c}+1\ge\dfrac{a}{b}+\dfrac{b}{c}$$
	$$\left(\dfrac{b}{a}-1\right)\left(\dfrac{c}{b}-1\right)\ge0\Rightarrow\dfrac{c}{a}+1\ge\dfrac{b}{a}+\dfrac{c}{b}$$
Vậy VT $\le2\left(\dfrac{a}{c}+\dfrac{c}{a}\right)+2$.\\
Mặt khác ta có $1\le a\le2$, $1\le c\le2$, suy ra $\dfrac{1}{2}\le\dfrac{a}{c}\le2$ nên $$\left(\dfrac{a}{c}-2\right)\left(2\dfrac{a}{c}-1\right)\le0\Leftrightarrow2\left(\dfrac{a}{c}+\dfrac{c}{a}\right)\le5.$$
Vậy từ đó dẫn đến VT $\le5+2=7$, suy ra điều phải chứng minh.}
\end{ex}
\begin{ex}%[Đề HSG9, Quảng Nam, 2017-2018]%[Phan Hoàng Anh]%[9D0K2]
Cho phương trình $x^2-2x+3-2m=0$. Tìm $m$ để phương trình có hai nghiệm phân biệt $x_1$, $x_2$, trong đó có một nghiệm bằng bình phương nghiệm còn lại.
	\loigiai{Phương trình có hai nghiệm phân biệt khi $\Delta'>0\Leftrightarrow1-3+2m>0\Leftrightarrow m>1$.\\
	Theo định lý Vi-et ta có $\heva{&x_1+x_2=-\dfrac{b}{a}=2\\&x_1x_2=\dfrac{c}{a}=3-2m.}$\\
Yêu cầu bài toán $\begin{aligned}[t]
&\Leftrightarrow\hoac{&x_1=x_2^2\\&x_2=x_1^2}\Leftrightarrow\left(x_1-x_2^2\right)\left(x_2-x_1^2\right)=0\Leftrightarrow x_1x_2+x_1^2x_2^2-\left(x_1^3+x_2^3\right)=0\\
&\Leftrightarrow x_1x_2+x_1^2x_2^2-\left(x_1+x_2\right)^3+3x_1x_2\left(x_1+x_2\right)=0\\
&\Leftrightarrow 3-2m+(3-2m)^2-8+6(3-2m)=0\\
&\Leftrightarrow4m^2-26m+22=0\Leftrightarrow\hoac{&m=1\\&m=\dfrac{11}{2}.}
\end{aligned}$}
\end{ex}
\begin{ex}%[Đề HSG9, Quảng Nam, 2017-2018]%[Phan Hoàng Anh]%[9D0K2]
Giải phương trình $2\sqrt{1-x}+\sqrt{1-x^2}=3-x$.
	\loigiai{Điều kiện: $-1\le x\le1$.\\
	Ta có: $\begin{aligned}[t]
	&2\sqrt{1-x}+\sqrt{1-x^2}=3-x\\
	\Leftrightarrow \,& 1-x-2\sqrt{1-x}+1+1-\sqrt{1-x^2}=0\\
	\Leftrightarrow \,& \left(\sqrt{1-x}-1\right)^2+\dfrac{x^2}{1+\sqrt{1-x^2}}=0\\
	\Leftrightarrow\,&\dfrac{x^2}{\left(\sqrt{1-x}+1\right)^2}+\dfrac{x^2}{1+\sqrt{1-x^2}}=0\Leftrightarrow\hoac{&x=0\\&\dfrac{1}{\left(\sqrt{1-x}+1\right)^2}+\dfrac{1}{1+\sqrt{1-x^2}}=0\quad(\text{vô nghiệm})}
	\end{aligned}$\\
Vậy phương trình có nghiệm duy nhất $x=0$.}
\end{ex}
\begin{ex}%[Đề HSG9, Quảng Nam, 2017-2018]%[Phan Hoàng Anh]%[9D0K4]
Chứng minh rằng với mọi số tự nhiên $n\ge1$ thì $(n+2)(n+1)(n+8)$ không thể là lập phương của một số tự nhiên.
	\loigiai{Xét $x=(n+2)(n+1)(n+8)=n^3+11n^2+26n+16$.
	Do $n$ là số tự nhiên nên ta có $$n^3+6n^2+12n+8<n^3+11n^2+26n+16<n^3+12n^2+48n+64.$$
$\Rightarrow(n+2)^3<x<(n+4)^3$, vì thế nên nếu $x$ là lập phương của một số thì $x=(n+3)^3$ (vô lý).\\
Vậy $x$ không phải là lập phương của một số, với mọi số tự nhiên $n$.}
\end{ex}
\begin{ex}%[Đề HSG9, Quảng Nam, 2017-2018]%[Phan Hoàng Anh]%[9D0K4]
Cho số nguyên tố $p$, $(p>3)$ và hai số nguyên dương $a$, $b$ sao cho $p^2+a^2=b^2$. Chứng minh $a$ chia hết cho $12$ và $2\left(p+a+1\right)$ là số chính phương.
	\loigiai{Ta có $b^2-a^2=p^2\Leftrightarrow p^2=(b-a)(b+a)$.\\
	Suy ra $b-$ và $b+a$ đều là ước của $p^2$.\\
Do $b+a>b-a>0$ nên $\heva{&b-a=1\\&b+a=p^2}\Leftrightarrow\heva{&b=a+1\\&2a+1=p^2}$\\
$\Rightarrow 2\left(p+a+1\right)=p^2+2p+1=\left(p+1\right)^2$ là số chính phương.\\
Ta lại có $\heva{&b=a+1\\&p^2=a+b=2a+1}\Rightarrow 2a=p^2-1$.\\
Do $p$ là số nguyên tố lớn hơn $3$ nên $p^2-1\,\vdots\,3\Rightarrow 2a\,vdots\,3\Rightarrow a\,\vdots\,3$.
Mặt khác, cũng do $p$ là số nguyên tố nên ta xét hai trường hợp sau:
\begin{itemize}
	\item Nếu $p=4k+1$ thì $a=8k(k+1)\,\vdots\,4$.
	\item Nếu $p=4k+3$ thì $a=8(k+1)(k+2)\,\vdots\,4$.
\end{itemize}
Vậy ta suy ra $a$ chia hết cho $12$.}
\end{ex}
\begin{ex}%[Đề HSG9, Quảng Nam, 2017-2018]%[Phan Hoàng Anh]%[9H0K3]
Cho hình vuông $ABCD$ có cạnh bằng $4$ cm, $E$ là điểm nằm trên cạnh $BC$ ($E$ khác $B$ và $C$). Đường thẳng qua $B$, vuông góc với đường thẳng $DE$ tại $H$ và cắt đường thẳng $CD$ tại $F$. Gọi $K$ là giao điểm của $AH$ và $BD$.
\begin{enumerate}
	\item Chứng minh tứ giác $KDCE$ nội tiếp trong đường tròn và ba điểm $E$, $K$, $F$ thẳng hàng.
	\item Khi $E$ là trung điểm cạnh $BC$, tính diện tích tứ giác $BKEH$.
\end{enumerate}
	\loigiai{\begin{center}
					\begin{tikzpicture}
			\clip (-1,-1) rectangle (7,5);
			\tkzDefPoints{0/0/D, 4/0/C, 0/4/A, 4/4/B}
			\coordinate (E) at ($(B)!0.5!(C)$);
			\tkzDefPointBy[projection = onto D--E](B)	\tkzGetPoint{H}
			\tkzInterLL(A,H)(B,D)	\tkzGetPoint{K}
			\tkzInterLL(B,H)(C,D)	\tkzGetPoint{F}
			\tkzDrawPoints(A,B,C,D,E,F,H,K)
			\tkzDrawSegments(A,B B,C B,F D,F A,H K,F D,H A,D B,D H,C C,K A,C)
			\tkzLabelPoints[below left](D)
			\tkzLabelPoints[above left](A)		
			\tkzLabelPoints[above right](B,H)		
			\tkzLabelPoints[below right](C,F)		
			\tkzLabelPoints[above](K)
			\tkzMarkRightAngle(B,H,D)
			\tkzMarkRightAngle(B,C,D)
			\tkzMarkRightAngle(D,A,B)
			\tkzMarkRightAngle(E,K,B)
			\node at (3.8,1.6) {$E$};
			\end{tikzpicture}
		\end{center}
	\begin{enumerate}
		\item Ta có $\heva{&\widehat{BHD}=90^\circ\\&\widehat{BAD}=90^\circ}\Rightarrow$ tứ giác $BHDA$ là tứ giác nội tiếp $\Rightarrow\widehat{ABD}=\widehat{AHD}$.\\
		Mặt khác, do $ABCD$ là hình vuông nên $\widehat{ABD}=\widehat{KBE}$, suy ra $\widehat{KBE}=\widehat{KHE}$.\\
		$\Rightarrow$ tứ giác $BHEK$ là tứ giác nội tiếp, suy  ra $\widehat{BKE}=\widehat{BHE}=90^\circ$.\\
		$\Rightarrow\widehat{DKE}=\widehat{DCE}=90^\circ$, vậy tứ giác $DKEC$ là tứ giác nội tiếp.\\
		Do $DH$ và $BC$ là hai đường cao của tam giác $BCF$ nên $E$ là trực tâm của tam giác $BDF$, suy ra $FE\perp BD$. Mặt khác, có $EK\perp BD$ nên $E$, $F$, $K$ thẳng hàng.
		\item Có $BE=CE=\dfrac{1}{2}BC=2$ cm $\Rightarrow DE=\sqrt{CD^2+CE^2}=2\sqrt{5}$ cm.\\
		Có $\heva{&\widehat{BHE}=\widehat{ECD}\\&\widehat{BEH}=\widehat{DEC}}\Rightarrow\triangle BEH\backsim\triangle DEC\Rightarrow\dfrac{EH}{EC}=\dfrac{BH}{CD}=\dfrac{BE}{DE}=\dfrac{2}{2\sqrt{5}}\Rightarrow\heva{&BH=\dfrac{4}{\sqrt{5}}\\&EH=\dfrac{2}{\sqrt{5}}.}$\\
		Có $\heva{&\widehat{EBK}=45^\circ\\&\widehat{BKE}=90^\circ}\Rightarrow\triangle BKE$ vuông cân tại $K$, suy ra $BK=KE=\dfrac{BE\sqrt{2}}{2}=\sqrt{2}$.\\
		Diện tích tứ giác $BHKE$ là $$S_{BHKE}=S_{BHE}+S_{BKE}=\dfrac{1}{2}\times BH\times HE+\dfrac{1}{2}\times BK\times EK=\dfrac{9}{5}\text{ cm}^2.$$
\end{enumerate}}
\end{ex}
\begin{ex}%[Đề HSG9, Quảng Nam, 2017-2018]%[Phan Hoàng Anh]%[9H0K1]
Cho hai đường tròn $(C_1)$, $(C_2)$ cắt nhau tại hai điểm $A$, $B$. Tiếp tuyến tại $A$ của $(C_2)$ cắt $(C_1)$ tại $M$ ($M$ khác $A$). Tiếp tuyến tại $A$ của $(C_1)$ cắt $(C_2)$ tại $N$ ($N$ khác $A$). Đường thẳng $MB$ cắt $(C_2)$ tại $P$ ($P$ khác $B$). Đường thẳng $NB$ cắt $(C_1)$ tại $Q$ ($Q$ khác $B$).
\begin{enumerate}
	\item Chứng minh các tam giác $AMP$ và $AQN$ đồng dạng.
	\item Chứng minh $MB\cdot NA^2=NB\cdot MA^2$.
\end{enumerate}
\loigiai{\begin{center}
		\begin{tikzpicture}
		\tkzDefPoints{-2/0/C_1, 1/0/X, 5/0/C_2, 0/0/O}
		\tkzDrawCircle(C_1,X)
		\tkzDrawCircle(C_2,O)
		\tkzInterCC(C_1,X)(C_2,O)	\tkzGetPoints{A}{B}
		\tkzTangent[at=A](C_1)	\tkzGetPoint{n}
		\tkzTangent[at=A](C_2)	\tkzGetPoint{m}
		\tkzInterLC(A,n)(C_2,O)	\tkzGetFirstPoint{N}
		\tkzInterLC(A,m)(C_1,X)	\tkzGetSecondPoint{M}
		\tkzInterLC(M,B)(C_2,O)	\tkzGetSecondPoint{P}
		\tkzInterLC(N,B)(C_1,X)	\tkzGetSecondPoint{Q}
		\tkzDrawSegments(N,Q M,P A,M A,N A,P A,Q C_1,A C_2,A A,B)
		\tkzDrawPoints(C_1,C_2,A,B,M,N,P,Q)
		\tkzLabelPoints[below](N)
		\tkzLabelPoints[above left](Q,C_1)
		\tkzLabelPoints[below left](M)
		\tkzLabelPoints[above right](P,C_2)
		\node at (0.3,2.3) {$A$};
		\node at (0.3,-2.3) {$B$};
		\end{tikzpicture}
\end{center}
\begin{enumerate}
	\item Ta có $\widehat{AQN}=\widehat{AQB}=\widehat{AMP}=\widehat{AMB}=\dfrac{1}{2}\text{ sđ}\wideparen{AB}$ (góc nội tiếp chắn cung $\wideparen{AB}$ của đường tròn $\left(C_1\right)$)\\
	Ta có $\widehat{ANQ}=\widehat{ANB}=\widehat{APM}=\widehat{APB}=\dfrac{1}{2}\text{ sđ}\wideparen{AB}$ (góc nội tiếp chắn cung $\wideparen{AB}$ của đường tròn $\left(C_2\right)$)\\
	Vậy ta suy ra tam giác $AMP$ đồng dạng với tam giác $AQN$.
	\item Ta có $\heva{&\widehat{AMB}=\dfrac{1}{2}\text{ sđ}\wideparen{AB}\text{ (góc nội tiếp)}\\&\widehat{BAN}=\dfrac{1}{2}\text{ sđ}\wideparen{AB}\text{ (góc chắn bởi tiếp tuyến và dây cung)}}\Rightarrow\widehat{AMB}=\widehat{BAN}$.\\
	Ta có $\heva{&\widehat{ANB}=\dfrac{1}{2}\text{ sđ}\wideparen{AB}\text{ (góc nội tiếp)}\\&\widehat{BAM}=\dfrac{1}{2}\text{ sđ}\wideparen{AB}\text{ (góc chắn bởi tiếp tuyến và dây cung)}}\Rightarrow\widehat{ANB}=\widehat{BAM}$.\\
	Vậy ta suy ra $\triangle AMB\backsim\triangle NAB\Rightarrow\heva{&\dfrac{AM}{AN}=\dfrac{AB}{BN}\quad&(1)\\&\dfrac{AM}{AN}=\dfrac{MB}{AB}\quad&(2)}$\\
	Lấy $(1)$ nhân với $(2)$, ta suy ra $\dfrac{AM^2}{AN^2}=\dfrac{AB}{BN}\cdot\dfrac{MB}{AB}=\dfrac{BM}{BN}\Rightarrow MB\cdot AN^2=NB\cdot AM^2$.
\end{enumerate}}
\end{ex}
