\subsection {Số phức}
\begin{ex}%[Nguyễn Anh Tuấn, dự án EX-LT1]%[2D4Y1-3]
	Số phức liên hợp $\overline{z}$ của số phức $z=a+bi$ với $a,~b \in \mathbb{R}$ là
	\choice
	{$\overline{z}=-a+bi$}
	{$\overline{z}=b-ai$}
	{$\overline{z}=-a-bi$}
	{\True $\overline{z}=a-bi$}
	\loigiai{
		Với $z=a+bi$ suy ra số phức liên hợp là $\overline{z}=a-bi$.}
\end{ex}

\begin{ex}%[Nguyễn Anh Tuấn, dự án EX-LT1]%[2D4B1-3]
	Cho số phức $z\ne 0$ và $ z $ là một số thuần ảo. Mệnh đề nào sau đây là đúng?
	\choice
	{$z=i \cdot \overline{z}$}
	{$z=-i \cdot \overline{z}$}
	{$z=\overline{z}$}
	{\True $z=-\overline{z}$}
	\loigiai{
		Theo bài ra, ta đặt $z=ki\left(k\ne 0\right)$, suy ra $\overline{z}=-ki=-z\Leftrightarrow z=-\overline{z}$.}
\end{ex}

\begin{ex}%[Trần Nhân Kiệt, dự án EX-LT1]%[2D4Y1-1]
	Trong các kết luận sau, kết luận nào là {\bf sai}?
	\choice
	{Mô-đun của số phức $z$ là một số thực}
	{Mô-đun của số phức $z$ là một số phức}
	{\True Mô-đun của số phức $z$ là một số thực dương}
	{Mô-đun của số phức $z$ là một số thực không âm}
	\loigiai{
		Gọi $z=a+bi$ với $a, b\in \Bbb{R}.$ Mô-đun của $z$ là $|z|=\sqrt{a^2+b^2}.$\\
		Suy ra mô-đun của số phức $z$ là một số thực không âm.\\
		Kết luận, mô-đun của số phức $z$ là một số thực dương là {\bf sai} vì khi $z=0$ thì $|z|=0.$
	}
\end{ex}

\begin{ex}%[Trần Nhân Kiệt, dự án EX-LT1]%[2D4B1-2]
	Mệnh đề nào sau đây {\bf sai}?
	\choice
	{Hai số phức $z_1$ và $z_2$ có $\left|z_1\right|=\left|z_2\right|\neq 0$ thì các điểm biểu diễn $z_1$ và $z_2$ trên mặt phẳng tọa độ cùng nằm trên đường tròn có tâm là gốc tọa độ}
	{Số phức $z$ có phần thực và phần ảo bằng nhau thì điểm biểu diễn của số phức $z$ nằm trên đường phân giác của góc phần tư thứ nhất và thứ ba}
	{\True Hai số phức có mô-đun bằng nhau thì bằng nhau}
	{Số phức $z$ và số phức liên hợp $\overline{z}$ có phần thực bằng nhau}
	\loigiai{
		\begin{itemize}
			\item Giả sử $\left|z_1\right|=\left|z_2\right|=R$ và $M$, $N$ lần lượt là các điểm biểu diễn của $z_1$, $z_2$.\\
			Khi đó, $M$, $N$ thuộc đường tròn tâm là gốc tọa độ $O$, bán kính $R.$ 
			\item Số phức $z=a+bi$, $\left(a, b\in \Bbb{R}\right)$ có điểm biểu diễn là $M(a;b)$. Nếu phần thực bằng phần ảo, tức là $a=b$ thì $M(a;a)\in d\colon y=x\Rightarrow M$ nằm trên đường phân giác của góc phần tư thứ nhất và thứ ba.
			\item Hai số phức $z$ và $w$ có mô-đun bằng nhau thì có thể bằng nhau hoặc khác nhau.\\
			Thật vậy, $z=1+2i\Rightarrow |z|=\sqrt{5}$, $w=2+i\Rightarrow |w|=\sqrt{5}.$
			\item Giả sử $z=a+bi$, $\left(a, b\in \Bbb{R}\right)\Rightarrow $ số phức liên hợp của $z$ là $ \overline{z}=a-bi.$\\
			Vậy $z$ và $\overline{z}$ có phần thực bằng nhau. 
		\end{itemize}
	}
\end{ex}

\begin{ex}%[Trần Nhân Kiệt, dự án EX-LT1]%[2D4Y1-1]
	Gọi $M$ là điểm biểu diễn của số phức $z=a+bi$ $\left(a, b\in \Bbb{R}\right)$ trong mặt phẳng tọa độ. Mệnh đề nào sau đây đúng?

	}

\begin{ex}%[Trần Nhân Kiệt, dự án EX-LT1]%[2D4B1-1]
	Cho số phức $z$ thỏa mãn $|z|=z.$ Mệnh đề nào sau đây đúng?
	\choice
	{\True $z$ là số thực không âm}
	{$z$ là số thực âm}
	{$z$ là số thuần ảo có phần ảo dương}
	{$z$ là số thuần ảo có phần ảo âm}
	\loigiai{
		Gọi $z=a+bi$ với $a, b\in \Bbb{R}\Rightarrow |z|=\sqrt{a^2+b^2}.$\\
		Ta có
		\begin{align*}
		|z|=z&\Leftrightarrow \sqrt{a^2+b^2}=a+bi\Leftrightarrow \heva{&\sqrt{a^2+b^2}=a\\&b=0}\\
		&\Leftrightarrow \heva{&\sqrt{a^2}=a\\&b=0}\Leftrightarrow \heva{&|a|=a\\&b=0}\Leftrightarrow \heva{&a\ge 0\\&b=0.}
		\end{align*}
		Vậy $z=a\ge 0\Rightarrow z$ là số thực không âm.
	}
\end{ex}

\subsection {Cộng, trừ và nhân số phức}
\begin{ex}%[Phạm Đức, dự án EX-LT1]%[2D4Y2-1]
	Cho số phức $z$. Phép tính nào dưới đây luôn cho kết quả là một số thực?
	\choice
	{$z-\overline{z}$}
	{\True $z\cdot \overline{z}$}
	{$z^2$}
	{$z+2\overline{z}$}
	\loigiai{
		Ta có $z\cdot \overline{z}=|z|^2$ luôn là một số thực.}
\end{ex}

\begin{ex}%[Phạm Đức, dự án EX-LT1]%[2D4Y2-2]  
	Cho các số phức $z_1=a+bi$, $z_2=c+di$, $z_3=m+ni$ với $a\,,b\,,c\,,d\,,m\,,n\in \Bbb{R}
	$. Phần thực của số phức $(z_1+z_2)z_3$ bằng
	\choice
	{\True $(a+c)m-(b+d)n$}
	{$(a+c)n-(b+d)m$}
	{$(a+c)m+(b+d)n$}
	{$(a+c)n+(b+d)m$}
	\loigiai{
		Ta có
		\begin{align*}
		(z_1+z_2)z_3&=[(a+bi)+(c+di)](m+ni)\\
		&=[(a+c)+(b+d)i](m+ni)\\
		&=(a+c)m-(b+d)n+[(a+c)n+(b+d)m]i
		\end{align*}
		Nên phần thực của  $(z_1+z_2)z_3$ là $(a+c)m-(b+d)n$.
	}
\end{ex}

\begin{ex}%[Phạm Đức, dự án EX-LT1]%[2D4Y2-1]
	Cho $z_1$ và $z_2$ là hai số thuần ảo khác $0$. Mệnh đề nào dưới đây \textbf{sai}? 
	\choice
	{$z_1+z_2$ là số thuần ảo}
	{$z_1-z_2$ là số thuần ảo}
	{\True $z_1\cdot z_2$ là số thuần ảo}
	{$\overline{z_1}$ là số thuần ảo}
	\loigiai{
		Giả sử $z_1=mi$, $z_2=ni$ với $m\,,n\ne 0$. Khi đó $z_1\cdot z_2=mni^2=-mn$ là một số thực khác $0$.
	}
\end{ex}

\begin{ex}%[Phạm Đức, dự án EX-LT1]%[2D4B2-1]
	Cho hai số phức $z_1$ và $z_2$. Mệnh đề nào dưới đây \textbf{sai}?
	\choice
	{$\overline{z_1+z_2}=\overline{z_1}+\overline{z_2}$}
	{$\overline{z_1-z_2}=\overline{z_1}-\overline{z_2}$}
	{$\overline{z_1\cdot z_2}=\overline{z_1}\cdot \overline{z_2}$}
	{\True $z_1\cdot \overline{z_2}=\overline{z_1}\cdot z_2$}
	\loigiai{
		Giả sử $z_1=a+bi$, $z_2=c+di$ với $a,\,b,\,c,\,d\in \Bbb{R}$. Khi đó
		\begin{itemize}
			\item $z_1\cdot \overline{z_2}=(a+bi)(c-di)=ac+bd+(bc-ad)i$.
			\item $\overline{z_1}\cdot z_2=(a-bi)(c+di)=ac+bd+(ad-bc)i$.
		\end{itemize}
		Nên $z_1\cdot \overline{z_2}\ne \overline{z_1}\cdot z_2$\\
		Các đáp án khác được kiểm tra tương tự.
	}
\end{ex}

\begin{ex}%[Phạm Đức, dự án EX-LT1]%[2D4B2-4]
	Trong mặt phẳng tọa độ $Oxy$, gọi $M$ và $N$ lần lượt là điểm biểu diễn các số phức $z_1$ và $z_2$. Mệnh đề nào dưới đây đúng?
	\choice
	{$MN=|z_1+z_2|$}
	{\True $MN=|z_1-z_2|$}
	{$MN=|z_1\cdot z_2|$}
	{$MN=|z_1|-|z_2|$}
	\loigiai{
		Giả sử $z_1=x_1+y_1i$, $z_2=x_2+y_2i$ thì $M(x_1;y_1)$ và $N(x_2;y_2)$.\\
		Khi đó $|z_1-z_2|=\sqrt{(x_1-x_2)^2+(y_1-y_2)^2}=MN$.
	}
\end{ex}

\begin{ex}%[Lê Hồng Phi, dự án EX-LT1]%[2D4Y3-5]

	\choice
	{\True $\dfrac{1}{z}$}
	{$1-z$}
	{$-z$}

	\loigiai{

	}
\end{ex}

\begin{ex}%[Lê Hồng Phi, dự án EX-LT1]%[2D4B3-5]
	Cho số phức $ z=a+bi\neq 0,\ (a, b\in\mathbb{R}) $. Tính $ \left|\dfrac{1}{z}\right| $?
	\choice
	{$ \left|\dfrac{1}{z}\right| =1-\sqrt{a^2+b^2}$}
	{\True $ \left|\dfrac{1}{z}\right| =\dfrac{1}{\sqrt{a^2+b^2}}$}
	{$ \left|\dfrac{1}{z}\right| =\dfrac{1}{a^2+b^2}$}
	{$ \left|\dfrac{1}{z}\right| =\dfrac{a-b}{a^2+b^2}$}
	\loigiai{
		Ta có $ \left|\dfrac{1}{z}\right| =\dfrac{|1|}{|z|}=\dfrac{1}{\sqrt{a^2+b^2}}$.		
	}
\end{ex}

\begin{ex}%[Lê Hồng Phi, dự án EX-LT1]%[2D4B3-5]
	Cho 3 số phức $ z, z', w $ thỏa mãn $ w=\dfrac{z'}{z} $, trong đó $ z\neq 0 $. Đẳng thức nào dưới đây là {\bf sai}?
	\choice
	{$ z'=\dfrac{|z|^2w}{\bar{z}} $}
	{$ \left|z'\right|=\left|w\right|\cdot\left|\bar{z}\right| $}
	{\True $ z'\bar{z}=z^2w $}
	{$ z'=\dfrac{\left|\bar{z}\right|^2w}{\bar{z}} $}
	\loigiai{
		Ta có $ w=\dfrac{z'}{z}=\dfrac{z'\bar{z}}{\left|z\right|^2} $. Từ đây, ta có các đẳng thức đúng là
		\begin{enumerate}[$ \bullet $]
			\item $ z'=\dfrac{|z|^2w}{\bar{z}} $,
			\item  $z'=wz\Rightarrow \left|z'\right|=\left|wz\right|=\left|w\right|\cdot\left|z\right|\Rightarrow  \left|z'\right|=\left|w\right|\cdot\left|\bar{z}\right| $ (vì $ \left|z\right|=\left|\bar{z}\right| $),
			\item $ z'=\dfrac{|z|^2w}{\bar{z}}=\dfrac{\left|\bar{z}\right|^2w}{\bar{z}} $  (vì $ \left|z\right|=\left|\bar{z}\right| $).
		\end{enumerate}		
	}
\end{ex}

\begin{ex}%[Lê Hồng Phi, dự án EX-LT1]%[2D4Y3-5]
	Cho $ a, b, c, d\in\mathbb{R} $ và $ a^2+b^2\neq 0 $. Cách làm nào sau đây dùng để tính thương $ \dfrac{c+di}{a+bi} $?	
	\choice
	{Cộng cả tử và mẫu với số phức liên hợp của $ a+bi $}
	{Cộng cả tử và mẫu với số phức  $-a+bi $}
	{Nhân cả tử và mẫu với mô-đun của số phức  $ a+bi $}
	{\True Nhân cả tử và mẫu với số phức liên hợp của $ a+bi $}
	\loigiai{
		Trong số các cách làm ở 4 phương án, chỉ có cách ``Nhân cả tử và mẫu với số phức liên hợp của $ a+bi $'' là dùng để tính thương 	 $ \dfrac{c+di}{a+bi} $.	
	}
\end{ex}

\begin{ex}%[Lê Hồng Phi, dự án EX-LT1]%[2D4K3-5]
	Cho $ z $ là số phức thỏa mãn $ \left|\dfrac{z}{a+bi}+i\right|=1 $, trong đó $ a, b\in\mathbb{R} $ và $ a^2+b^2\neq 0 $. Tìm tập hợp các điểm biểu diễn số phức $ z $ trong mặt phẳng phức.	
	\choice
	{\True Tập hợp các điểm biểu diễn số phức $ z $ là đường tròn tâm $ I(b; -a) $ bán kính $ \sqrt{a^2+b^2} $}
	{Tập hợp các điểm biểu diễn số phức $ z $ là đường tròn tâm $ I(-b; a) $ bán kính $ \sqrt{a^2+b^2} $}
	{Tập hợp các điểm biểu diễn số phức $ z $ là đường tròn tâm $ I(-a; -b) $ bán kính $ 1 $}
	{Tập hợp các điểm biểu diễn số phức $ z $ là đường tròn tâm $ I(a; b) $ bán kính $ a^2+b^2 $}
	\loigiai{
		Gọi $ z=x+yi,\ (x, y\in\mathbb{R}) $ là số phức thỏa mãn bài toán.\\
		Khi đó, trong mặt phẳng phức, điểm $ M(x; y) $ biểu diễn cho số phức $ z $.\\
		Ta có $\begin{aligned}[t]
		\left|\dfrac{z}{a+bi}+i\right|=1&\Leftrightarrow \left|\dfrac{z-b+ai}{a+bi}\right|=1\\
		&\Leftrightarrow\dfrac{\left|z-b+ai\right|}{\left|a+bi\right|}=1\\
		&\Leftrightarrow\left|z-b+ai\right|=\left|a+bi\right|\\
		&\Leftrightarrow\left|(x-b)+(y+a)i\right|=\sqrt{a^2+b^2}\\
		&\Leftrightarrow (x-b)^2+(y+a)^2=a^2+b^2.
		\end{aligned} $	\\
		Vậy, tập hợp các điểm biểu diễn số phức $ z $ là đường tròn tâm $ I(b; -a) $ bán kính $ \sqrt{a^2+b^2} $.	
	}
\end{ex}

\begin{ex}%[Nguyễn Anh Tuấn, dự án EX-LT1]%[2D4Y4-4]

	\begin{enumerate}[(I)]
		\item Nếu $ \Delta $ là số thực âm thì phương trình $ (*) $ vô nghiệm.
		\item Nếu $ \Delta \ne 0$ thì phương trình $ (*) $ có hai nghiệm phân biệt.
		\item  Nếu $ \Delta=0 $ thì phương trình $ (*) $ một nghiệm duy nhất. 
	\end{enumerate}
	Số mệnh đề đúng là
	\choice
	{$ 0 $}
	{$ 1 $}
	{\True $ 2 $}
	{$ 3 $}
	\loigiai{
		Mệnh đề (I) sai, mệnh đề (II), (III) đúng.}
\end{ex}

\begin{ex}%[Nguyễn Anh Tuấn, dự án EX-LT1]%[2D4B4-4]

\begin{ex}%[Nguyễn Anh Tuấn, dự án EX-LT1]%[2D4B4-4]
	Cho $ P(z) $ là một đa thức với hệ số thực. Nếu số phức $ z $ thỏa mãn $ P(z)=0 $ thì 
	\choice
	{$ P(|z|)=0 $}
	{$ P\left(\dfrac{1}{z}\right)=0 $}
	{$ P\left(\dfrac{1}{\overline{z}}\right)=0 $}
	{\True $ P(\overline{z}) =0$}
	\loigiai{
		Với mọi số phức $ z, z' $ ta có $ \overline{z\cdot z'}=\overline{z} \cdot \overline{z'} $; $ \overline{z + z'}=\overline{z} + \overline{z'} $; $ \overline{z - z'}=\overline{z} - \overline{z'} $.\par
		\noindent  Từ $ P(z)=0 \Rightarrow \overline{P(z)}=0 \Rightarrow P(\overline{z})=0$.
	}
\end{ex}

\begin{ex}%[Lê Quốc Hiệp dự án EX-LT1]%[2D4B4-4]
	Cho phương trình bậc hai $ax^2+bx+c=0$ với $a,~b,~c\in \mathbb{R}$ có hai nghiệm phức là $x_1$ và $x_2$. Khẳng định nào dưới đây \textbf{sai?}
	\choice
	{$x_1+x_2=-\dfrac{b}{a}$}
	{\True $\bar{x}_1=\bar{x}_2$}
	{$x_1 x_2=\dfrac{c}{a}$}
	{$|x_1|=|x_2|$}
	\loigiai
	{
		Vì hai nghiệm phức của phương trình bậc hai đã cho là hai số phức liên hợp nhau nên khẳng định $\bar{x}_1=\bar{x}_2$ là khẳng định \textbf{sai}.
	}
\end{ex}

\begin{ex}%[Lê Quốc Hiệp dự án EX-LT1]%[2D4B4-4]
	Cho phương trình bậc hai $x^2+bx+c=0$ với $b,~c\in\mathbb{R}$ có một nghiệm phức là $x=2-i$. Trong các khẳng định dưới đây, khẳng định nào đúng?
	\choice
	{\True $(2+i)^2+b(2+i)+c=0$}
	{$(-2+i)^2+b(-2+i)+c=0$}
	{$(2+i)^2+b(2+i)-c=0$}
	{$(2+i)^2-b(2+i)+c=0$}
	\loigiai
	{
		Phương trình đã cho có một nghiệm phức là $x=2-i$ suy ra nghiệm phức còn lại là $2+i$. Do đó
		\[(2+i)^2+b(2+i)+c=0.\]
		
	}
\end{ex}

\begin{ex}%[Lê Quốc Hiệp dự án EX-LT1]%[2D4B4-4]
	Cho phương trình bậc hai $az^2+bz+c=0$ với $a,~b,~c\in \mathbb{R}$ có hai nghiệm phức là $z_1$ và $z_2$. Tính $P=\dfrac{z_1}{\bar{z}_2}+\dfrac{\bar{z}_1}{z_2}$.
	\choice
	{$P=0$}
	{$P=1$}
	{\True $P=2$}
	{$P=3$}
	\loigiai
	{
		Ta có $z_1=\bar{z}_2$ và  $z_2=\bar{z}_1$ nên
		\[P=\dfrac{z_1}{\bar{z}_2}+\dfrac{\bar{z}_1}{z_2}=\dfrac{\bar{z}_2}{\bar{z}_2}+\dfrac{z_2}{z_2}=2.\]
	}
\end{ex}

\begin{ex}%[Lê Quốc Hiệp dự án EX-LT1]%[2D4K4-4]
	Cho phương trình bậc hai $ax^2+bx+c=0$ với $a,~b,~c\in \mathbb{R}$ có hai nghiệm phức là $x_1$ và $x_2$. Biết rằng tập hợp điểm biểu diễn số phức $x_1$ là đường tròn có tâm $I(2;1)$, bán kính $R=2$. Tìm tập hợp điểm biểu diễn số phức $x_2$.
	\choice
	{$(x-2)^2+(y-1)^2=4$}
	{\True $(x-2)^2+(y+1)^2=4$}
	{$(x-2)^2+(y+1)^2=2$}
	{$(x-2)^2+(y-1)^2=2$}
	\loigiai
	{
		Gọi $M$ là điểm biểu diễn của số phức $z_1$ và $N$ là điểm biểu diễn của số phức $z_2$.\\
		Ta có $M$ thuộc đường tròn $(C_1)$ có phương trình
		\[(C_1)\colon(x-2)^2+(y-1)^2=4.\qquad(1)\]
		Vì $M$ và $N$ đối xứng nhau qua trục tọa độ $Ox$ nên $N$ thuộc đường tròn $(C_2)$ đối xứng với đường tròn $(C_1)$ qua trục tọa độ $Ox$.\\
		Từ (1) suy ra đường tròn $(C_2)$ có phương trình
		\[(C_2)\colon(x-2)^2+(y+1)^2=4.\]
		
	}
\end{ex}
