\subsection{Tóm tắt lí thuyết}
\subsubsection{Các tập hợp số đã học}
\begin{dn}
Tập hợp các số tự nhiên $\mathbb{N}=\left\{0,1,2,3,\ldots\right\}$ và $\mathbb{N}^*=\left\{1,2,3\ldots\right\}$.
\end{dn}

\begin{dn}
Tập hợp các số nguyên $\mathbb{Z}=\left\{\ldots,-3,-2,-1,0,1,2,3,\ldots\right\}$.
\end{dn}

\begin{dn}
Tập hợp các số hữu tỉ kí hiệu $\mathbb{Q}$, là số viết được dưới dạng phân số $\dfrac{a}{b}$ với $a$, $b\in\mathbb{Z}$, $b\neq 0$.
\end{dn}

\begin{dn}
Tập hợp các số thực kí hiệu $\mathbb{R}$, gồm các số thập phân hữu hạn, vô hạn tuần hoàn và vô hạn không tuần hoàn. Các số thập phân vô hạn không tuần hoàn gọi là số vô tỉ.
\end{dn}

\subsection{Các dạng toán}

\begin{dang}{Xác định giao - hợp của hai tập hợp}
\begin{enumerate}
\item Xác định giao của hai tập hợp ta làm như sau
\begin{itemize}
\item Biểu diễn các tập hợp lên trục số.
\item Dùng định nghĩa giao để xác định các phần tử của tập hợp.
\end{itemize}
\item Cho hai tập con của tập số thực $A$ và $B$. Tìm $A\cup B$ ta làm như sau
\begin{itemize}
\item Biểu diễn tập $A$ trên trục số, gạch chéo phần không thuộc $A$.
\item Làm tương tự đối với tập $B$.
\item Phần không gạch chéo trên hình là $A\cup B$.
\end{itemize}
\item Đối với hai tập $A$ và $B$ khác để tìm $A\cup B$ ta nhớ rằng $x\in A\cup B\Leftrightarrow\hoac{& x\in A\\& x\in B}$ 
\end{enumerate}
\end{dang}

\begin{vd}%[Hồ Thúy]%[0D1G4]
Cho các tập hợp $A=\{x\in \mathbb{R}| |x+2|<2\}$, $B=\{x\in \mathbb{R}| |x+4|\geq 3\}$, $C=[-5;3)$. Tìm các tập hợp
\begin{itemize}
	\item [a)] $A \cap B$.
	\item [b)] $B \cap C$.
	\item [c)] $A \cap B\cap C$.
	\item [d)] $A \cup B$.
	\item [e)] $A \cap B \cup C$.
	\item [f)] $(A \cup B)\cap(B \cup C)$.
\end{itemize}
\loigiai{
$|x+2|<2\Leftrightarrow -2<x+2<2\Leftrightarrow -4<x<0$. Do đó $A=(-4;0)$.\\
$|x+4|\geq 3\Leftrightarrow \hoac{&x+4\leq -3\\&x+4\geq 3}\Leftrightarrow \hoac{&x\leq -7\\&x\geq -1}$. Do đó $B=(-\infty;-7]\cup [-1;+\infty)$.\\
Biểu diễn tập $A$ trên trục số:\\ \begin{tikzpicture}[xscale=0.5,thick,>=stealth']
\draw[->](-9,0)->(6,0);
\IntervalLR{-9}{-4}
\IntervalGRF{}{}{(}{-4}
\IntervalLR{0}{5.7}
\IntervalGRF{)}{0}{}{}
\end{tikzpicture}\\
Biểu diễn tập $B$ trên trục số:\\ \begin{tikzpicture}[xscale=0.5,thick,>=stealth']
\draw[->](-9,0)->(6,0);
\IntervalLR{-7}{-1}
\IntervalGRF{]}{-7}{[}{-1}
\end{tikzpicture}\\
Biểu diễn tập $C$ trên trục số:\\ \begin{tikzpicture}[xscale=0.5,thick,>=stealth']
\draw[->](-9,0)->(6,0);
\IntervalLR{-9}{-5}
\IntervalGRF{}{}{[}{-5}
\IntervalLR{3}{5.7}
\IntervalGRF{)}{3}{}{}
\end{tikzpicture}

\begin{itemize}
	\item [a)] $A \cap B = [-1;0)$.
	\item [b)] $B \cap C = (-\infty;-7]\cup [-5;+\infty)$.
	\item [c)] $A \cap B\cap C= [-1;0)$.
	\item [d)] $A \cup B = (-\infty;-7]\cup (-4;+\infty)$.
	\item [e)] $A \cap B \cup C = [-5;3)$.
	\item [f)] $(A \cup B)\cap(B \cup C)$.\\
	$(A \cup B) = (-\infty;-7]\cup (-4;+\infty)$.\\
	$(B \cup C) = (-\infty;-7]\cup [-5;+\infty)$.\\
	Do đó $(A \cup B)\cap(B \cup C) = (-\infty;-7]\cup (-4;+\infty)$.
\end{itemize}
}
\end{vd}

		\item [b)] $(A \cup B)\cap C$.

		\item [d)] $A \cap (B \cup C)$.
	\end{itemize}

	\item $\dfrac{x+1}{x-1}\geq 0 \Leftrightarrow \hoac{&x\leq -1\\&x>1}$. Do đó $A=(-\infty;-1]\cup (1;+\infty)$.
	\item  $9-x^2\leq 0 \Leftrightarrow \hoac{&x\leq -3\\&x\geq 3}$. Do đó $B=(-\infty;-3]\cup [3;+\infty)$.

Do đó $C=[-2;+\infty)$.
\end{itemize}

	\item [c)] $(A \cup C) = \mathbb{R}.\\(A \cup C)\cap B = B = (-\infty;-3]\cup [3;+\infty)$.
