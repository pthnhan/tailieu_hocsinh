\begin{bt}%[Đề KT HK1, Lương Thế Vinh, TPHCM 2018]%[Hồ Minh Hòa, 11HKI-18]%[1D1B3-1]%[1D1K3-3]
Giải các phương trình sau:
\begin{multicols}{2}
\begin{enumerate}[a.]
 \item $2\cos^2x+9\cos x-5=0$
 \item $\sin 9x-2\sin 8x\cos x=\sqrt{3}\left(\cos 7x-\dfrac{2\sqrt{3}}{3}\right)$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}[a.]
	\item Ta có $2\cos^2x+9\cos x-5=0\Leftrightarrow \hoac{&\cos x=\dfrac{1}{2}\\ &\cos x=-5\textrm{ (vô nghiệm)}}\Leftrightarrow x=\pm\dfrac{\pi}{3}+k2\pi,\ k\in\mathbb{Z}$.
	\item Phương trình đã cho tương đương với
	\begin{align*}
	&\sin 9x-\sin 9x-\sin 7x-\sqrt{3}\cos 7x=-2\\
	\Leftrightarrow &\sin 7x+\sqrt{3}\cos 7x=2\\
	\Leftrightarrow &\sin\left(7x+\dfrac{\pi}{3}\right)=1\\
	\Leftrightarrow &7x+\dfrac{\pi}{3}=\dfrac{\pi}{2}+k2\pi\Leftrightarrow x=\dfrac{\pi}{42}+\dfrac{k2\pi}{7},\ k\in\mathbb{Z}.
	\end{align*}
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề KT HK1, Lương Thế Vinh, TPHCM 2018]%[Hồ Minh Hòa, 11HKI-18]%[1D2B5-2]%[1D2K5-2]
Một hộp đựng $10$ viên bi xanh được đánh số từ $1$ đến $10$, $8$ viên bi đỏ được đánh số từ $1$ đến $8$ và $7$ viên bi vàng được đánh số từ $1$ đến $7$. Chọn ngẫu nhiên $4$ viên bi. Tính xác suất để:
\begin{enumerate}[a.]
\item $4$ viên bi được chọn không có bi đỏ.
\item $4$ viên bi được chọn có đủ $3$ màu và tích các số trên $4$ viên bi đó là một số lẻ.
\end{enumerate}
\loigiai{
\begin{enumerate}[a.]
	\item Ta có $n(\Omega)=\mathrm{C}_{25}^4=2380$. Gọi $A$ là biến cố ``$4$ viên bi được chọn không có bi đỏ''.\\
	Khi đó $n(A)=\mathrm{C}_{17}^4=12650$. Vậy $\mathrm{P}(A)=\dfrac{n(A)}{n(\Omega)}=\dfrac{2380}{12650}=\dfrac{238}{1265}$.
	\item Gọi $B$ là biến cố ``$4$ viên bi được chọn có đủ $3$ màu và tích các số trên $4$ viên bi đó là một số lẻ''. Để tích các số trên các thẻ là một số lẻ thì các thẻ được chọn đều phải mang số lẻ. Số cách chọn $4$ thẻ đều lẻ và đủ ba màu tức là một trong các trường hợp:
	$2$ thẻ vàng, $1$ thẻ xanh, $1$ thẻ đỏ; $1$ thẻ vàng, $2$ thẻ xanh, $1$ thẻ đỏ; $1$ thẻ vàng, $1$ thẻ xanh, $2$ thẻ đỏ.\\
	Ta có $n(B)=\mathrm{C}_5^2\cdot\mathrm{C}_4^1\cdot\mathrm{C}_4^1+\mathrm{C}_5^1\cdot\mathrm{C}_4^2\cdot\mathrm{C}_4^1+\mathrm{C}_5^1\cdot\mathrm{C}_4^1\cdot\mathrm{C}_4^2=400$.\\
	Vì vậy, $\mathrm{P}(B)=\dfrac{n(B)}{n(\Omega)}=\dfrac{400}{12650}=\dfrac{8}{253}$.
\end{enumerate}
}
\end{bt}
\begin{bt}%[Đề KT HK1, Lương Thế Vinh, TPHCM 2018]%[Hồ Minh Hòa, 11HKI-18]%[1D2B3-2]
	Tìm số hạng chứa $x^5$ trong khai triển của biểu thức $\left( 2x^2 -\dfrac{3}{x^3}\right)^{10}$ với $x\ne0$.
\loigiai{
Số hạng tổng quát trong khai triển đã cho là $\mathrm{C}_{10}^k\cdot \left( 2x^2\right)^{10-k}\cdot\left(- \dfrac{3}{x^3}\right)^k=\mathrm{C}_{10}^k\cdot 2^{10-k}\cdot(-3)^k \cdot x^{20-5k}$.\\
Số hạng đó chứa $x^5$, suy ra $20-5k=5 \Leftrightarrow k=3.$\\
Vậy số hạng chứa $x^5$ trong khai triển đã cho là $\mathrm{C}_{10}^3\cdot 2^7\cdot(-3)^3 \cdot x^5=-414720x^5$.
}
\end{bt}
\begin{bt}%[Đề KT HK1, Lương Thế Vinh, TPHCM 2018]%[Hồ Minh Hòa, 11HKI-18]%[1D3B3-3]
	Tìm số hạng đầu $u_1$ và công sai $d$ của cấp số cộng $(u_n)$ thỏa mãn $\heva{&u_1+3u_5=28\\&S_6-u_3=29.}$
\loigiai{
Theo tính chất của cấp số cộng ta có $u_n=u_1+(n-1)d$ và $S_n=nu_1+\dfrac{n(n-1)}{2}d$.\\
Do đó $\heva{&u_1+3u_5=28\\&S_6-u_3=29} \Leftrightarrow \heva{&u_1+3(u_1+4d)=28\\&(6u_1+15d)-(u_1+2d)=29}\Leftrightarrow \heva{&u_1+3d=7\\&5u_1+13d=29} \Leftrightarrow \heva{&u_1=-2\\&d=3.}$
}
\end{bt}
\begin{bt}%[Đề KT HK1, Lương Thế Vinh, TPHCM 2018]%[Hồ Minh Hòa, 11HKI-18]%[1H2Y1-2]%[1H2B2-3]%[1H2K3-2]%[1H2K2-4]
	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình bình hành, $O$ là giao điểm của $AC$ và $BD$. Gọi $M, N$ lần lượt là trung điểm của $AB, SD$.
\begin{enumerate}[a.]
	\item Tìm giao tuyến của $(SCM)$ và $(SBD)$.
	\item Tìm giao điểm $K$ của $SA$ và $(OMN)$.
	\item Chứng minh rằng $MN \parallel (SBC)$.
	\item Tìm thiết diện của $(OMN)$ và hình chóp $S.ACD$. Thiết diện là hình gì?
\end{enumerate}
\loigiai{
\begin{center}
	\begin{tikzpicture}[scale=.8]
	\tkzDefPoints{0/0/A, -3/-3/B, 3/-3/C}
	\coordinate (D) at ($(A)+(C)-(B)$);
	\coordinate (O) at ($(A)!.5!(C)$);
	\coordinate (S) at ($(O)+(-1,5)$);
	\tkzDefMidPoint(A,B)\tkzGetPoint{M}
	\tkzDefMidPoint(S,D)\tkzGetPoint{N}
	\tkzDefMidPoint(A,S)\tkzGetPoint{K}
	\tkzDefMidPoint(C,D)\tkzGetPoint{P}
	\tkzInterLL(D,B)(C,M)\tkzGetPoint{I}
	\tkzDefMidPoint(C,D)\tkzGetPoint{J}
	\tkzDefMidPoint(C,S)\tkzGetPoint{Q}
	\tkzDrawSegments[dashed](S,A A,B A,D A,C B,D O,N O,M M,C K,N S,I O,P M,N O,K)
	\tkzDrawPolygon(S,C,D)
	\tkzDrawSegments(S,B B,C N,J B,Q Q,N)
	\tkzDrawPoints[color=black,fill=white,size=6](S,A,B,C,D,M,N,P,I,J,O,K,Q)
	\tkzLabelPoints[left](A,B,K,M)
	\tkzLabelPoints[right](C,D,J)
	\tkzLabelPoints[above](S,N)
	\tkzLabelPoints[below](O,I,Q)
	\end{tikzpicture}
\end{center}
\begin{enumerate}[a.]
	\item Gọi $I$ là giao điểm của $CM$ và $BD$, suy ra $I\in (SCM)\cap (SDB)$. Mặt khác $S \in (SCM)\cap (SBD)$.\\
	Vậy giao tuyến của hai mặt phẳng $(SCM)$ và $(SBD)$ là đường thẳng $SI$.
	\item $O, M$ lần lượt là trung điểm của $BD, BA$ nên $OM \parallel AD$. Mặt khác $N \in (OMN) \cap (SAD)$ nên giao tuyến của hai mặt phẳng $(OMN)$ và $(SAD)$ là đường thẳng $d$ qua $N$ và song song với $OM,AD$.\\
	$d$ cắt $SA$ tại $K$, suy ra $K$ là giao điểm của $SA$ và $(OMN)$.
	\item Gọi $Q$ là trung điểm của $SC$ ta có $\heva{&BM\parallel QN\;(\text{vì cùng song song với } CD)\\&BM=QN=\dfrac{1}{2}CD.}$\\
	Suy ra $BMNQ$ là hình bình hành nên $MN \parallel BQ$, do đó $MN \parallel (SBC)$.
	\item Ta có $OK=(OMN)\cap (SAC)$ và $KN=(OMN)\cap (SAD)$.\\
	Do $KN \parallel AD$ nên giao tuyến của $(OMN)$ và $(ACD)$ là đường thẳng $d_1$ qua $O$ và song song với $KN, AD$. Gọi $J=d_1 \cap CD$, suy ra $(OMN) \cap (ACD) = OJ$ và $(OMN) \cap (SCD) = NJ$.\\
	Vậy thiết diện của $(OMN)$ và hình chóp $S.ACD$ là tứ giác $OJNK$.\\
	Do $\heva{&OJ\parallel KN\;(\text{vì cùng song song với } AD)\\&OJ=KN=\dfrac{1}{2}AD.}$ nên thiết diện $OJNK$ là hình bình hành.
\end{enumerate}
}
