\begin{bt}%[HK1, Phan Đăng  Lưu, HCM 2017-2018]%[1D1B2-1]
	Giải phương trình $2\sin \left(2x-\dfrac{\pi}{3}\right)-\sqrt{3}=0$.
	\loigiai{
		Ta có 
		\begin{eqnarray*}
		&&2\sin \left(2x-\dfrac{\pi}{3}\right)-\sqrt{3}=0 \Leftrightarrow \sin \left(2x-\dfrac{\pi}{3}\right)=\dfrac{\sqrt{3}}{2}\\
		&\Leftrightarrow & \hoac{& 2x-\dfrac{\pi}{3}=\dfrac{\pi}{3}+k2\pi\\& 2x-\dfrac{\pi}{3}=\dfrac{2\pi}{3}+k2\pi}\Leftrightarrow \hoac{&x=\dfrac{\pi}{3}+k\pi\\&x=\dfrac{\pi}{2}+k\pi}, k\in \Bbb Z.
		\end{eqnarray*}
		Vậy phương trình có nghiệm là $x=\dfrac{\pi}{3}+k\pi$, $x=\dfrac{\pi}{2}+k\pi$ với $k\in \Bbb Z.$				
	
	}
\end{bt}
\begin{bt}%[HK1, Phan Đăng  Lưu, HCM 2017-2018]%[1D1B3-3]
	Giải phương trình $\sin \dfrac{x}{2}-\sqrt{3}\cos \dfrac{x}{2}=\sqrt{3}$.
		\loigiai{
		Ta có 
		\begin{eqnarray*}
		&& \sin \dfrac{x}{2}-\sqrt{3}\cos \dfrac{x}{2}=\sqrt{3} \Leftrightarrow  \dfrac{1}{2}\sin \dfrac{x}{2}-\dfrac{\sqrt{3}}{2}\cos \dfrac{x}{2}=\dfrac{\sqrt{3}}{2} \Leftrightarrow  \sin \left(\dfrac{x}{2}-\dfrac{\pi}{3}\right)=\dfrac{\sqrt{3}}{2}\\
		& \Leftrightarrow & \hoac{&\dfrac{x}{2}-\dfrac{\pi}{3}=\dfrac{\pi}{3}+k2\pi\\&\dfrac{x}{2}-\dfrac{\pi}{3}=\dfrac{2\pi}{3}+k2\pi}\Leftrightarrow \hoac{&x=\dfrac{4\pi}{3}+k4\pi\\&x=2\pi+k4\pi}, k\in \Bbb Z.
		\end{eqnarray*}
		Vậy phương trình có nghiệm là $x=\dfrac{4\pi}{3}+k4\pi$, $x=2\pi+k4\pi$ với $k\in \Bbb Z.$				
	
		}
\end{bt}

\begin{bt}%[HK1, Phan Đăng  Lưu, HCM 2017-2018]%[1D1B3-1]
	Giải phương trình $2\sin^2 3x -(\sqrt{3}-2)\cos 3x +\sqrt{3}-2=0$.
		\loigiai{
		Ta có 
		\begin{eqnarray*}
		& & 2\sin^2 3x -(\sqrt{3}-2)\cos 3x +\sqrt{3}-2=0\\
		& \Leftrightarrow & -2\cos^2 3x -(\sqrt{3}-2)\cos 3x + \sqrt{3}=0\\
		& \Leftrightarrow & \hoac{&\cos 3x =1 \\& \cos 3x = -\dfrac{\sqrt{3}}{2}& (\text{loại})}\\
		& \Leftrightarrow &\cos 3x =1  \Leftrightarrow  3x = k2\pi \Leftrightarrow x = \dfrac{k2\pi}{3}, k \in \Bbb Z.
		\end{eqnarray*}
		Vậy phương trình có nghiệm $x=\dfrac{k2\pi}{3}, k \in \Bbb Z.$
		}
\end{bt}

\begin{bt}%[HK1, Phan Đăng  Lưu, HCM 2017-2018]%[1D1K3-5]
	Giải phương trình $\sin 2x +\cos x=1+2\sin x$.
	\loigiai{
		Ta có 
		\begin{eqnarray*}
		& & \sin 2x +\cos x=1+2\sin x\\
		& \Leftrightarrow & 2\sin x\cos x -2\sin x + \cos x-1=0\\
		& \Leftrightarrow & (2\sin x -1)(\cos x-1) =0\\
		& \Leftrightarrow & \hoac{& \sin x = \dfrac{1}{2}\\ &\cos x =1}\\
		& \Leftrightarrow & \hoac{& x=\dfrac{\pi}{6}+k2\pi\\ & x=\dfrac{5\pi}{6}+k2\pi \\& x = k2\pi}, k\in \Bbb Z.
		\end{eqnarray*}
		Vậy phương trình có nghiệm $\hoac{&x=\dfrac{\pi}{6}+k2\pi\\& x=\dfrac{5\pi}{6}+k2\pi \\& x = k2\pi}, k\in \Bbb Z.$
	}
\end{bt}

\begin{bt}%[HK1, Phan Đăng  Lưu, HCM 2017-2018]%[1D2B2-1]
	Từ các số $0, 1, 2, \ldots, 9$ có thể lập được bao nhiêu số tự nhiên có năm chữ số khác nhau và chia hết cho 2.
	\loigiai{
	Giả sử số cần lập có dạng $\overline{abcde}$. \\
	Do $\overline{abcde} ~\vdots~ 2$ nên $e ~\vdots ~ 2$.
	\begin{itemize}
	\item Nếu $e=0$ thì có  $\mathrm{A}_9^4$ số $\overline{abcd}$ nên có  $\mathrm{A}_9^4$ số $\overline{abcde}$.
	\item Nếu $e\ne 0$ thì $e$ có 4 cách chọn, $a$ có 8 cách chọn, còn $\overline{bcd}$ có $\mathrm{A}_8^3$ cách. Suy ra có $4\cdot8\cdot \mathrm{A}_8^3$ số $\overline{abcde}$.
\end{itemize}		
Vậy có tổng cộng $\mathrm{A}_9^4 +4\cdot8\cdot \mathrm{A}_8^3= 13776$ số thỏa mãn yêu cầu bài toán.
	}
\end{bt}

\begin{bt}%[HK1, Phan Đăng  Lưu, HCM 2017-2018]%[1D2B3-2]
	Tìm hệ số của số hạng chứa $x^{37}$ trong khai triển $\left(\dfrac{1}{x^4}+x^7\right)^{10}$, với $x\ne 0$.
	\loigiai{
	Ta có $\left(\dfrac{1}{x^4}+x^7\right)^{10} = \displaystyle \sum_{k=0}^{10} \mathrm{C}_{10}^k \cdot x^{-4k}\cdot x^{70-7k} = \sum_{k=0}^{10} \mathrm{C}_{10}^k \cdot x^{70-11k}$.\\
	Chọn $k \in \Bbb N^*$ thỏa mãn $70-11k=37 \Leftrightarrow k = 3$ (thỏa mãn).\\
	Vậy hệ số của số hạng chứa $x^{37}$ là $\mathrm{C}_{10}^3=120$.
	}
\end{bt}

\begin{bt}%[HK1, Phan Đăng  Lưu, HCM 2017-2018]%[1D2B5-2]
	Nhân dịp năm mới, Hiệu trưởng của một trường THPT có 20 bao lì xì loại 20 ngàn đồng cho mỗi bao lì xì và 10 bao lì xì loại 10 ngàn đồng cho mỗi bao lì xì. Một giáo viên được chọn ngẫu nhiên 2 bao lì xì, tính xác suất để 
	\begin{enumerate}
	\item được 2 bao lì xì loại 20 ngàn đồng.
	\item được ít nhất một bao lì xì loại 20 ngàn đồng.
	\end{enumerate}
	\loigiai{
	Tổng số bao lì xì là $30$.\\
	Không gian mẫu có số phần tử $n(\Omega) = \mathrm{C}_{30}^2$.
	\begin{enumerate}
	\item Số cách để được 2 bao lì xì loại 20 ngàn là $\mathrm{C}_{20}^2$.\\
	Xác suất cần tính bằng $\dfrac{\mathrm{C}_{20}^2}{\mathrm{C}_{30}^2}=\dfrac{38}{87}$.
	\item Để được ít nhất một bao lì xì loại 20 ngàn thì người đó có thể nhận được 2 bao lì xì loại 20 ngàn hoặc một bao loại 10 ngàn, một bao loại 20 ngàn nên số cách thỏa mãn là $\mathrm{C}_{20}^2 + \mathrm{C}_{20}^1\cdot \mathrm{C}_{10}^1$.\\
	Vậy ta có xác suất cần tính bằng $\dfrac{\mathrm{C}_{20}^2+\mathrm{C}_{20}^1\cdot \mathrm{C}_{10}^1}{\mathrm{C}_{30}^2}=\dfrac{26}{29}$.
\end{enumerate}		
	}
\end{bt}

\begin{bt}%[HK1, Phan Đăng  Lưu, HCM 2017-2018]%[1H2B8-2]
	Trong mặt phẳng tọa độ $Oxy$, cho điểm $A(1;5)$. Tìm tọa độ điểm $A'$ là ảnh của điểm $A$ qua phép đồng dạng có được từ việc thực hiện liên tiếp phép tịnh tiến theo véc-tơ $\overrightarrow{v}=(3;2)$ và phép vị tự tâm $O$ tỉ số $k=0{,}5$.
	\loigiai{
	Gọi $M =\mathrm{T}_{\overrightarrow{v}} (A)$ $ \Leftrightarrow\overrightarrow{AM}=\overrightarrow{v} \Rightarrow \heva{&x_M =x_A +3=4\\&y_M=y_A+2=7}\Rightarrow M(4;7)$.\\
	Khi đó $A'$ là ảnh của $M(4;7)$ qua phép vị tự tâm $O$ tỉ số $k=\dfrac{1}{2}$.\\
	Suy ra $\overrightarrow{OA'}=\dfrac{1}{2}\overrightarrow{OM}=(4;7)\Rightarrow A'\left(2;\dfrac{7}{2}\right).$
	}
\end{bt}

\begin{bt}%[HK1, Phan Đăng  Lưu, HCM 2017-2018]%[1H2B1-2]%[1H2B3-2]%[1H2K4-2]
	Cho hình chóp $S.ABCD$ có $ABCD$ là hình bình hành tâm $O$. Gọi $M,N$ lần lượt là trung điểm của các đoạn $AB, SD$.
	\begin{enumerate}
	\item Tìm giao tuyến của hai mặt phẳng $(SAC)$ và $(SBD)$; $(OMN)$ và $(SAD)$.
	\item Chứng minh đường thẳng $SB$ song song với mặt phẳng $(ANC)$.
	\item Chứng minh đường thẳng $MN$ song song với mặt phẳng $(SBC)$.
	\end{enumerate}
	\loigiai{
\begin{center}
		\begin{tikzpicture}[scale=1, line join=round, line cap=round]
	\tkzDefPoints{0/0/A,-2/-1.6/B,1.6/-1.6/C}
	\coordinate (D) at ($(A)+(C)-(B)$);
	\coordinate (O) at ($(A)!1/2!(C)$);
	\coordinate (S) at ($(O)+(-1.5,4)$);
	\coordinate (M) at ($(A)!0.5!(B)$);
	\coordinate (N) at ($(S)!0.5!(D)$);	
	\coordinate (x) at ($(M)+(N)-(O)$);
	\coordinate (y) at ($2*(N)-(x)$);
	\tkzDrawPolygon(S,B,C,D)
	\tkzDrawSegments(S,C N,C N,y)
	\tkzDrawSegments[dashed](A,S A,B A,D A,C B,D S,O O,M O,N M,N N,x)
	\tkzDrawPoints[fill=black,size=4](D,C,A,B,S,M,N)
	\tkzLabelPoints[above](S)
	\tkzLabelPoints[below](B,C,O)
	\tkzLabelPoints[right](D)
	\tkzLabelPoints[left](x)
	\tkzLabelPoints[above left](A,M)
	\tkzLabelPoints[above right](N)
	\end{tikzpicture}
\end{center}	
\begin{enumerate}
\item Trong mặt phẳng $(ABCD)$ có $AC\cap BD =O \Rightarrow \heva{&S,O\in (SAC)\\& S,O\in (SBD)}\Rightarrow (SAC) \cap (SBD) = SO$.\\
Xét hai mặt phẳng $(OMN)$ và $(SAD)$ có $N$ là điểm chung.\\
Trong tam giác $ABD$ có $OM$ là đường trung bình nên $OM \parallel AD$. \\
Do đó, giao tuyến của hai mặt phẳng $(OMN)$ và $(SAD)$ là đường thẳng $Nx$ đi qua $N$ và song song với $AD$.
\item Ta có $O,N$ lần lượt là trung điểm của $BD$ và $SD$ nên $ON$ là đường trung bình của tam giác $SBD$. Suy ra $ON \parallel SB$. \\
Lại có $ON$ nằm trong mặt phẳng $(ANC)$. Vậy $SB \parallel (ANC)$.
\item Xét hai mặt phẳng $(SBC)$ và $(OMN)$ có $ON \parallel SB$ và $OM\parallel BC$. Hơn nữa các cặp đường thẳng $OM,ON$ và $SB,BC$ đôi một cắt nhau nên $(OMN) \parallel (SBC)$.\\
Suy ra $MN \parallel (SBC)$.
\end{enumerate} 
	}
\end{bt}
