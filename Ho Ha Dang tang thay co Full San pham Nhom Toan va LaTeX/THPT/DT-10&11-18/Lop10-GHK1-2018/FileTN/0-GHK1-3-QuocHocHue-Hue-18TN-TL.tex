\begin{ex}%[Thi GHKI, Quốc Học Huế, 2018]%[Đỗ Đức Thảo, EX-DT-10&11]%[0D2B3-3]
	Cho parabol $(P): y=ax^2+bx+c$ có đồ thị như hình vẽ dưới đây. Hãy tìm nhận xét đúng.\\
	\begin{center}
		\begin{tikzpicture}
		\draw[>=stealth,->] (-2,0) -- (5,0) node[below]{\scriptsize $x$};
		\draw[>=stealth,->] (0,-2) -- (0,4) node[left]{\scriptsize $y$};
		\draw (0,0) node[below left]{\scriptsize $O$};
		\foreach \i in {-1,1,2,3,4}
		\draw (\i,0.1)--(\i,-0.1) node [below] {\footnotesize \i};
		\foreach \i in {1,2,3}
		\draw (0.1,\i)--(-0.1,\i) node [left] {\footnotesize \i};
		\draw [thick,smooth,samples=100,domain=-0.2:4.1] plot (\x, {-(\x)^2+4*\x-1});
		\end{tikzpicture}
	\end{center}
	
	\choice
	{$ a<0, c<0, b \le 0 $}
	{$a>0, c<0, b \ge 0 $}
	{$  a>0, c>0, b>0 $}
	{\True $a<0, c<0, b>0$}
	\loigiai{
		Đồ thị hàm số parabol quay xuống nên $ a<0$. Khi $x=0$ thì $y=c<0$. Dựa vào tọa độ của đỉnh ta suy ra được $ b>0 $. Vậy $ a<0,c<0, b>0 $.
	}
\end{ex}

\begin{ex}%[Thi GHKI, Quốc Học Huế, 2018]%[Đỗ Đức Thảo, EX-DT-10&11]%[0D2B3-2]
	Cho hàm số $y=f(x)=ax^2+bx+6$, $(a\ne 0$) có đồ thị là parabol đỉnh $I(-2;4)$. Tìm tổng $S=a+b+6$.
	\choice
	{$\dfrac{5}{2}$}
	{\True $\dfrac{17}{2}$}
	{$\dfrac{15}{2}$}
	{$\dfrac{7}{2}$}
	\loigiai{
		Ta có hoành độ đỉnh $-2=\dfrac{-b}{2a} \Leftrightarrow 4a-b=0$.\\
		Mà parabol có đỉnh là $I(-2;4)$ nên $a\cdot(-2)^2 +b\cdot (-2) +6 =4 \Leftrightarrow 4a-2b=-2$.\\
		Suy ra $a=\dfrac{1}{2}, b=2$. Vậy $a+b+6 = \dfrac{17}{2}$.
	}
\end{ex}

\begin{ex}%[Thi GHKI, Quốc Học Huế, 2018]%[Đỗ Đức Thảo, EX-DT-10&11]%[0D2B1-4]
	Đồ thị hàm số nào sau đây nhận trục $Oy$ làm trục đối xứng?
	\choice
	{$y=x^3-|x|$}
	{\True $x^2-|x|$}
	{$x^2-x$}
	{$x^3-x$}
	\loigiai{
		Xét hàm số $y=x^2-|x|$.\\
		Tập xác định $\mathbb{R}$.\\
		Ta có $\forall x\in \mathbb{R} \Rightarrow -x \in \mathbb{R}$.\\
		Xét $f(-x) =(-x)^2 -|-x| = x^2 -|x| =f(x)$.\\
		Vậy $f$ là hàm chẵn nên đồ thị nhận trục $Oy$ làm trục đối xứng.
	}
\end{ex}

\begin{ex}%[Thi GHKI, Quốc Học Huế, 2018]%[Đỗ Đức Thảo, EX-DT-10&11]%[0D3K3-3]
	Cho đường thẳng $(d_1)\colon y=x+7$, đường thẳng $(d_2)\colon y=ax-4$ và đường thẳng $(d_3)\colon y=2x+8$. Biết rằng $(d_1),(d_2)$ và $(d_3)$ đồng quy tại $M(b;c)$, tính tổng $S=a+b+c$.
	\choice
	{\True $-5$}
	{$6$}
	{$-1$}
	{$-10$}
	\loigiai{
		Tọa độ giao điểm của $(d_1)$ và $(d_3)$ thỏa hệ $$\heva{& y=x+7 \\& y=2x+8 }\Leftrightarrow \heva{& x=-1 \\& y=6.}$$
		$(d_1),(d_2)$ và $(d_3)$ đồng quy tại $M(b;c)$ suy ra $M(-1;6)$.
		\\ Ta có $M\in (d_2)$ nên $6=-a-4\Leftrightarrow a=-10$.
		\\ Vậy $S=-10-1+6=-5.$	
	}
\end{ex}

\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[Thi GHKI, Quốc Học Huế, 2018]%[Đỗ Đức Thảo, EX-DT-10&11]%[0D2B3-3]%[0D2B3-4]
	Cho hàm số $y=3x^{2}-6x+2\quad  (1)$ có đồ thị $(P)$ và đường thẳng $d:y=m+x$ ($m$ là tham số).
	\begin{enumerate}
		\item Lập bảng biến thiên và vẽ đồ thị của hàm số $(1)$.
		\item Tìm các giá trị của $m$ để đường thẳng $d$ cắt $(P)$ tại hai điểm phân biệt và hai điểm đó nằm về hai phía đối với trục tung.
	\end{enumerate}	 
	\loigiai{
		\begin{enumerate}
			\item Bảng biến thiên
			\begin{center}
				\begin{tikzpicture}
			\tkzTabInit[nocadre=false,lgt=1.2,espcl=2.5,deltacl=0.6] % tùy chọn
				{$x$/1.2, $y$/2.5} % cột đầu tiên
				{$-\infty$, $1$, $+\infty$} % hàng 1 cột 2
				%\tkzTabLine{,-,0,+,} % hàng 2 cột 2
				\tkzTabVar{+/ $+\infty$, -/ $-1$ , +/ $+\infty$} % hàng 3 cột 2
				\end{tikzpicture}
			\end{center}
			
			Bảng giá trị
			\begin{center}
				\begin{tabular}{|m{1cm}|m{1cm}|m{1cm}|m{1cm}|m{1cm}|m{1cm}|} 
					\hline
					$x$ & $-1$ &$0$ & $1$ & $2$ & $3$ \\ 
					\hline
					$y$ & $11$ & $2$ & $-1$ &$2$ &$11$ \\ 
					\hline
				\end{tabular}
			\end{center}
			Tọa độ đỉnh $I(1;-1)$.\\
			Đồ thị
			
			\begin{center}
			\begin{tikzpicture}
			[scale=1, font=\footnotesize, line join=round, line cap=round, >=stealth]
				\draw [->, thick] (-3.2,0) -- (4.8,0) node [below]{$ x $};
				\foreach \x in {-3,-2,-1,1,2,3,4}
				\draw[shift={(\x,0)},color=black] (0pt,2pt) -- (0pt,-2pt) node[below] { $\x$};
				\draw [->, thick] (0,-2.2) -- (0,5.7) node [left]{$ y $};
				\foreach \y in {-2,-1,1,2,3,4,5}
				\draw[shift={(0,\y)},color=black] (2pt,0pt) -- (-2pt,0pt) node[right] { $\y$};
				\node (0,0) [below left](O){$ O $};
				\draw [thick, domain=-0.43:2.43] plot (\x, {3*(\x)^2-6*(\x)+2}) node[right]{$y=3x^{2}-6x+2$};
				\tkzDefPoints{1/-1/I}
				\tkzDrawPoint[fill=black](I)
				\tkzLabelPoint[below right](I) {$ I $}
				\draw [help lines, dashed] (0,-1) grid (1,0);
				\end{tikzpicture}
			\end{center}
			
			\item Phương trình hoành độ giao điểm của $d$ và $(P)$
			$$3x^{2}-6x+2=m+x \Leftrightarrow 3x^{2}-7x+2-m=0.$$
			$d$ cắt $(P)$ tại hai điểm phân biệt và hai điểm đó nằm về hai phía đối với trục tung khi và chỉ khi phương trình hoành độ giao điểm có hai nghiệm phân biệt trái dấu.\\
			$\Leftrightarrow \heva{& \Delta >0\\ &P=x_{1}x_{2}<0} \Leftrightarrow \heva{& (-7)^{2}-4 \cdot 3(2-m)>0\\ & \dfrac{2-m}{3} <0} \Leftrightarrow \heva{& m>- \dfrac{25}{12}\\ & m>2} \Leftrightarrow m>2$.\\
			Vậy với $m>2$ thì $d$ cắt $(P)$ tại hai điểm phân biệt và hai điểm đó nằm về hai phía đối với trục tung.
		\end{enumerate}		
	}
