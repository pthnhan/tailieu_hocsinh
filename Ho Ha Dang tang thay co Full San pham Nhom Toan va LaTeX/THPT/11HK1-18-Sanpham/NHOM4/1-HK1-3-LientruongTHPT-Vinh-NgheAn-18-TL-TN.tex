\begin{ex}%[HKI, Liên trường TP Vinh, Nghệ An, 2018]%[Ngọc Diệp, dự án 11HKI-18]%[1D1K3-7]
	Phương trình $5 \cos ^2 x +8(m +1)\sin x \cdot \cos x =  4m + \sin ^2 x$ (với $m$ là tham số) có nghiệm khi và chỉ khi
	\choice
	{\True $m \ge -\dfrac{21}{48}$}
	{$\forall m \in \mathbb{R}$}
	{$- \dfrac{21}{48} \le m \le \dfrac{21}{48}$}
	{$m \le \dfrac{21}{48}$}
	\loigiai{
		\begin{itemize}
		\item Trường hợp 1. $\cos x = 0 \Leftrightarrow x= \dfrac{\pi}{2} + k \pi $, $(k \in \mathbb{Z})$. Khi đó phương trình đã cho tương đương với $0=4m+1 \Leftrightarrow m = - \dfrac{1}{4}$. Vậy $m = - \dfrac{1}{4}$ thỏa mãn.
		\item Trường hợp 2. $\cos x \ne 0 \Leftrightarrow x \ne \dfrac{\pi}{2} + k \pi $, $(k \in \mathbb{Z})$. Khi đó, chia cả hai vế của phương trình trên cho $\cos x$ ta được
		\allowdisplaybreaks {\begin{align*}
		&5 +8(m +1) \dfrac{\sin x}{ \cos x} =  4m\cdot \dfrac{1}{\cos ^2 x} +\dfrac{\sin ^2 x}{\cos ^2 x} \\
		& \Leftrightarrow 5 +8(m +1) \tan x =  4m(1+ \tan ^2 x) +\tan ^2 x \\
		& \Leftrightarrow (4m+1)\tan ^2 x - 8(m+1) \tan x +4m-5 =0.	\tag{1}
		\end{align*}   }
	Với $m \ne \dfrac{-1}{4}$, phương trình $(1)$ là phương trình bậc hai ẩn $\tan x$. Phương trình $(1)$ có nghiệm khi và chỉ khi phương trình $ (4m+1)t^2 - 8(m+1) t +4m-5 =0$ có nghiệm, tức là $$ \Delta' \ge 0 \Leftrightarrow (4m+4)^2 - (4m+1)(4m-5) \ge 0 \Leftrightarrow 48m+21 \ge 0 \Leftrightarrow m \ge \dfrac{-21}{48}. $$
		\end{itemize}
	Từ hai trường hợp trên ta kết luận $m \ge -\dfrac{21}{48}$ thỏa mãn yêu cầu bài toán.
	}
\end{ex}

	điểm $N$ thuộc cạnh $SA$ sao cho $SN = 3AN $. Đường thẳng $MN$ cắt mặt phẳng
	$(ABCD)$ tại $P $, đường thẳng $PC$ cắt cạnh $AB$ tại $K $. Trình bày cách xác định điểm
	$K$ và tính tỉ số $\dfrac{KA}{KB}$.
	\loigiai{
\immini {
	Trong $(SAD)$ gọi $P =  MN \cap AD$. Ta có $$\heva{&P \in MN \\ & P \in AD \subset (ABCD)} \Rightarrow P = MN \cap (ABCD).$$
	Trong $(ABCD)$ gọi $K =  PC \cap AB$, khi đó điểm $K$ là điểm cần dựng.
	}
	{
		\begin{tikzpicture}[line cap=round, line join=round]%[Black Mild] %[line cap=round,line join=round,>=triangle 45,x=1.0cm,y=1.0cm]
		\tkzDefPoints{0/0/A, 5/0/B,  2/2/D, 3/6/S, 0.75/1.5/N}% C là điểm sao cho ABCD là hình thoi
		\tkzDefPointsBy[translation= from A to B](D){C}
		\tkzFillPolygon[white](S,A,B)
		\tkzFillPolygon[white](S,B,C)
		\tkzDrawPolygon(S,A,B)
		\tkzInterLL(A,C)(B,D) \tkzGetPoint{O}
		\tkzDrawSegments(S,C B,C)
		\tkzLabelPoints[right](C)
		\tkzLabelPoints[above](S)
		\tkzDefMidPoint(S,D)\tkzGetPoint{M}
				\tkzDefMidPoint(A,D)\tkzGetPoint{E}	
		\tkzInterLL(M,N)(A,D)\tkzGetPoint{P}
			\tkzInterLL(P,C)(A,B)\tkzGetPoint{K}
		\tkzDrawSegments(N,P P,K P,A)
		\tkzLabelPoints[below](A,B,P,K,E,D)
			\tkzLabelPoints[above left](N)
				\tkzLabelPoints[above right](M)
		\tkzDrawSegments[dashed](S,D C,D A,D M,N K,C M,E)
		\tkzDrawPoints(S,A,B,C,D,M,N,P,K,E)
		\end{tikzpicture}}
Từ  $SN = 3AN $ suy ra  $AN = \dfrac{1}{4}SA $.\\
Gọi $E$ là trung điểm $AD$. Ta có $AN \parallel ME$, theo định lí Talet suy ra
$$\dfrac{PA}{PE} = \dfrac{AN}{ME} = \dfrac{\dfrac{1}{4} SA}{\dfrac{1}{2} SA} = \dfrac{1}{2} \Rightarrow \dfrac{PA}{PD} = \dfrac{1}{3} .$$
Trong mặt phẳng $(ABCD)$, có $AK \parallel CD$ nên ta có $\dfrac{AK}{CD} = \dfrac{PA}{PD} = \dfrac{1}{3}$. Suy ra $\dfrac{AK}{BK} =\dfrac{1}{2}$.
}
\end{ex}
