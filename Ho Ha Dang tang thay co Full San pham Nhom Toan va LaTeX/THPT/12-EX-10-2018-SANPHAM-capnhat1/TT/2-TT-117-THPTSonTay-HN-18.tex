	{$\dfrac{10\pi - 15\sqrt{3}}{6}$}}
	{\begin{tikzpicture}[>=stealth] 
	\fill[color=gray, smooth,samples=100,domain=0:3, variable=\x](0,0) --  plot(\x,{sqrt(4*(\x)-(\x)^(2))})-- (3,1.732)--cycle;
	\fill[color=white, smooth,samples=100,domain=0:3, variable=\x](0,0) --  plot(\x, {sqrt{(\x)}}) -- (3,0)--cycle;
	\draw[->] (-1,0) --(5,0) node[below]{$x$};
	\draw[->] (0,-1) --(0,3) node[right]{$y$};
	\draw (0,0) node[below left]{$O$} circle (1pt);
	\draw (2,0) node[below]{$2$} circle (1pt);
	\draw (3,0) node[below]{$3$} circle (1pt);
	\draw (4,0) node[below]{$4$} circle (1pt);
	\draw[smooth,samples=100,domain=0:4] plot(\x,{sqrt{(\x)}});
	\draw[smooth,samples=100,domain=0:4] plot(\x,{sqrt(4*(\x)-(\x)^(2))});
	\end{tikzpicture}
	} 
	\loigiai{

	Vậy diện tích cần tính là
	$$S=\displaystyle \int \limits_0^3 \left(\sqrt{4x-x^2}-\sqrt{x}\right)\mathrm{\,d}x = \int \limits_0^3 \sqrt{4x-x^2} \mathrm{\,d}x - \int \limits_0^3 \sqrt{x}\mathrm{\,d}x = \int \limits_0^3 \sqrt{4x-x^2} \mathrm{\,d}x -2\sqrt{3}.$$

	$$\displaystyle \int \limits_0^3 \sqrt{4x-x^2} \mathrm{\,d}x  = \int \limits_{-\tfrac{\pi}{2}}^{\tfrac{\pi}{6}} 2\sqrt{1-\sin^2 t} \cos t \mathrm{\,d}t = \int \limits_{-\tfrac{\pi}{2}}^{\tfrac{\pi}{6}} 2(1+\cos 2t)\mathrm{\,d}t = (2t+\sin 2t)\bigg|_{-\tfrac{\pi}{2}}^{\tfrac{\pi}{6}}=\dfrac{8\pi + 3\sqrt{3}}{6}.$$
Vậy $S=\dfrac{8\pi + 3\sqrt{3}}{6} - 2\sqrt{3}=\dfrac{8\pi - 9\sqrt{3}}{6}$.
	}
