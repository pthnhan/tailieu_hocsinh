			\item Tìm tất cả các giá trị thực của tham số $m$ để phương trình $-x^3+3x^2+m+1=0$ có ba nghiệm thực phân biệt.
		\end{enumerate}

	\loigiai{
		\begin{enumerate}
			\item TXĐ: $D=\mathbb{R}$. Ta có: $y'=3x^2-6x=0\Leftrightarrow\hoac{x=0\\x=2}.$
