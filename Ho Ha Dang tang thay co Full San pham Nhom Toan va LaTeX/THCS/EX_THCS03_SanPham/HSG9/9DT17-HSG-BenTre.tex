\begin{enumerate}

\item Cho biểu thức $B=\sqrt{\dfrac{\left(x^2-3\right)^2+12x^2}{x^2}}+\sqrt{\left(x+2\right)^2-8x}$.\\
Rút gọn biểu thức $B$ và tìm các giá trị nguyên của $x$ để $B$ có giá trị nguyên.
\item Tìm tất cả các nghiệm nguyên của phương trình: $2y^2x+x+y+1=x^2+2y^2+xy$.

	\item Ta có $A=n^8+4n^7+6n^6+4n^5+n^4=n^4\left(n^4+4n^3+6n^2+4n+1\right)=\left[n\left(n+1\right)\right]^4$.\\
		Vì $n(n+1)$ là tích hai số nguyên liên tiếp nên $n(n+1)\vdots\ 2\Rightarrow \left[n\left(n+1\right)\right]^4\vdots\ 2^4=16$.\\
		Do đó: $A\vdots\ 16$ với mọi $n\in\mathbb{Z}$.
	\item $B=\sqrt{\dfrac{\left(x^2-3\right)^2+12x^2}{x^2}}+\sqrt{\left(x+2\right)^2-8x}=\sqrt{\dfrac{\left(x^2+3\right)^2}{x^2}}+\sqrt{\left(x-2\right)^2}=\dfrac{x^2+3}{\left|x\right|}+\left|x-2\right|$.\\
	\begin{itemize}
	\item Nếu $x<0\Rightarrow B=\dfrac{-x^2-3}{x}-x+2=\dfrac{-2x^2+2x-3}{x}=-2x+2-\dfrac{3}{x}$.\\
	$B$ có giá trị nguyên khi $\dfrac{3}{x}\in\mathbb{Z}\Leftrightarrow x\in\text{Ư}(3)$ và $x<0\Leftrightarrow \hoac{&x=-1\\&x=-3}$.
	\item Nếu $0<x\le 2\Rightarrow B=\dfrac{x^2+3}{x}-x+2=\dfrac{2x+3}{x}=2+\dfrac{3}{x}$.\\
	$B$ có giá trị nguyên khi $\dfrac{3}{x}\in\mathbb{Z}\Leftrightarrow x\in\text{Ư}(3)$ và $0<x\le 2\Leftrightarrow x=1$.
	\item Nếu $x>2\Rightarrow B=\dfrac{x^2+3}{x}+x-2=\dfrac{2x^2-2x+3}{x}=2x-2+\dfrac{3}{x}$.\\
	$B$ có giá trị nguyên khi $\dfrac{3}{x}\in\mathbb{Z}\Leftrightarrow x\in\text{Ư}(3)$ và $x>2\Leftrightarrow x=3$.
	\end{itemize}
	Tập các giá trị của $x$ để $B$ nguyên là $x\in\left\{\pm 1;\pm 3\right\}$.
	\item $2y^2x+x+y+1=x^2+2y^2+xy\Leftrightarrow (x-1)\left(x-2y^2+y\right)=1$.\\
	$\Leftrightarrow \hoac{&\heva{&x-1=1\\&x-2y^2+y=1}\\&\heva{&x-1=-1\\&x-2y^2+y=-1}}\Leftrightarrow \hoac{&\heva{&x=2\\&2y^2-y-1=0}\\&\heva{&x=0\\&2y^2-y-1=0}}\Leftrightarrow \hoac{&\heva{&x=2\\&y=1}\\&\heva{&x=0\\&y=1}}$.\\
	Vậy phương trình có hai nghiệm nguyên là $(2;1)$ và $(0;1)$.
	\end{enumerate}		
	}

\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2016-2017, Sở GD Bến Tre]%[9D5G5]
Cho $x,y,z$ là các số thực thỏa mãn $\heva{&x^2+xy+\dfrac{y^2}{3}=2017\\& z^2+\dfrac{y^2}{3}=1009\\&x^2+xz+z^2=1008} (x\ne 0,z\ne 0, x\ne -z)$.\\
Chứng minh rằng: $\dfrac{2z}{x}=\dfrac{y+z}{x+z}$.
	\loigiai{
	$\heva{&x^2+xy+\dfrac{y^2}{3}=2017\ (1)\\& z^2+\dfrac{y^2}{3}=1009\hspace{1.2cm} (2)\\&x^2+xz+z^2=1008\ \ (3)} (x\ne 0,z\ne 0, x\ne -z)$.\\
	Trừ $(1)$ và $(2)$ theo vế, ta có: $x^2-xy-z^2=1008\ (4)$.\\
	Trừ $(3)$ và $(4)$ theo vế, ta có: $xz-xy+2z^2=0$\\
	$\Leftrightarrow xz+2z^2=xy\Leftrightarrow 2xz+2z^2=xy+xz$.\\
	$\Leftrightarrow 2z(x+z)=x(y+z)$.\\
	$\Leftrightarrow \dfrac{2z}{x}=\dfrac{y+z}{x+z}$.
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2016-2017, Sở GD Bến Tre]%[9H3K7]
Cho đoạn thẳng $AB$ và điểm $E$ nằm giữa điểm $A$ và điểm $B$ sao cho $AE<BE$. Vẽ đường tròn $(O_1)$ đường kính $AE$ và đường tròn $(O_2)$ đường kính $BE$. Vẽ tiếp tuyến chung ngoài $MN$ của hai đường tròn $(O_1)$ và $(O_2)$ với $M$ là tiếp điểm thuộc $(O_1)$ và $N$ là tiếp điểm thuộc $(O_2)$.
\begin{enumerate}
\item Gọi $F$ là giao điểm của các đường thẳng $AM$ và $BN$. Chứng minh rằng đường thẳng $EF$ vuông góc với đường thẳng $AB$.
\item Với $AB=18$ cm và $AE=6$ cm, vẽ đường tròn $(O)$ đường kính $AB$. Đường thẳng $MN$ cắt đường tròn $(O)$ tại $C$ và $D$ sao cho điểm $C$ thuộc cung nhỏ $AD$. Tính độ dài đoạn thẳng $CD$.

	\begin{enumerate}
	\item $MN$ là tiếp tuyến chung của $(O_1)$ và $(O_2)$ nên $\heva{&MN\perp O_1M\\&MN\perp O_2N}\Rightarrow O_1M\parallel O_2N$.\\
	$\Rightarrow \widehat{MO_1E}+\widehat{NO_2E}=180^\circ$.\\
	$\Delta O_1AM$ cân tại $O_1\Rightarrow \widehat{MO_1E} =2\widehat{O_1AM}$.\\
	$\Delta O_2BN$ cân tại $O_2\Rightarrow \widehat{MO_2E} =2\widehat{O_2BN}$.\\
	$\Rightarrow \widehat{O_1AM}+\widehat{O_2BN}=90^\circ\Rightarrow \widehat{MFN}=90^\circ$.\\
	Mặt khác $\widehat{AME}=\widehat{BNE}=90^\circ$ (góc nội tiếp chắn nửa đường tròn).\\
	$\Rightarrow MENF$ là hình chữ nhật.\\

	Vì $OF\perp CD$ tại $I\Rightarrow CD=2ID=4\sqrt{14}$ cm.
