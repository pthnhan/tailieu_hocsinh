\begin{bt}%[Đề tuyển sinh 10 năm học 2018 - 2019 Bến Tre]%[Nguyễn Cao Cường, dự án EXTHCS-06]%[9D1K8]
	Rút gọn  các biểu thức
	\begin{enumerate}
		\item $A=\sqrt{12}+\sqrt{27}-\sqrt{48}$;
		\item $B=\left(\dfrac{1}{\sqrt{x}-1}-\dfrac{1}{\sqrt{x}+1}\right)\div \dfrac{x+1}{x-1}$, với $x\geq 0$ và $x\neq \pm 1$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Ta có
		$$A=\sqrt{12}+\sqrt{27}-\sqrt{48}=2\sqrt{3}+3\sqrt{3}-4\sqrt{3}=\sqrt{3}.$$
		\item Ta có
		$$B=\left(\dfrac{1}{\sqrt{x}-1}-\dfrac{1}{\sqrt{x}+1}\right)\div \dfrac{x+1}{x-1}=\dfrac{\sqrt{x}+1-(\sqrt{x}-1)}{(\sqrt{x}+1)(\sqrt{x}-1)}\cdot \dfrac{x-1}{x+1}=\dfrac{2}{x-1}\cdot \dfrac{x-1}{x+1}=\dfrac{2}{x+1}.$$
	\end{enumerate}	
	}
\end{bt}


\begin{bt}%[Đề tuyển sinh 10 năm học 2018 - 2019 Bến Tre]%[Nguyễn Cao Cường, dự án EXTHCS-06]%[9D3B2]
	Giải hệ phương trình 
	$\begin{cases}
	x+2y=12\\3x-y=1.
	\end{cases}$
	\loigiai{
		Ta có
		$$\begin{cases}
		x+2y=12\\3x-y=1.
		\end{cases}
		\Leftrightarrow 
			\begin{cases}
			3x+6y=36\\3x-y=1
			\end{cases}
			\Leftrightarrow 
			\begin{cases}
			7y=35\\3x-y=1
			\end{cases}
			\Leftrightarrow
			\begin{cases}
			x=2\\y=5.
			\end{cases}$$
			Vậy nghiệm của hệ phương trình là $(2; 5)$.
	}
\end{bt}



\begin{bt}%[Đề tuyển sinh 10 năm học 2018 - 2019 Bến Tre]%[Nguyễn Cao Cường, dự án EXTHCS-06]%[9D4B4]%[9D4K6]
	Cho phương trình $x^2+5x+m=0$\;\;\;$(*)$, ($m$ là tham số).
	\begin{enumerate}
		\item Giải phương trình $(*)$ khi $m=-3$.
		\item Tìm $m$ để phương trình $(*)$ có hai nghiệm $x_1, x_2$ thỏa mãn $9x_1+2x_2=18$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Khi $m=-3$ ta được phương trình $x^2+5x-3=0$. Ta có
			$\Delta=5^2-4(-3)=37>0$
			nên phương trình có hai nghiệm phân biệt 
			$\left[\begin{aligned} & x=\dfrac{-5+\sqrt{37}}{2} \\ & x=\dfrac{-5-\sqrt{37}}{2}.\end{aligned}\right.$
		\item Phương trình có hai nghiệm phân biệt khi $\Delta=25-4m\geq 0 \Leftrightarrow m\leq\dfrac{25}{4}$.\\
		Khi đó, ta có hệ
		$$\begin{cases} x_1+x_2=-5\\9x_1+2x_2=18\\ x_1\cdot x_2=m \end{cases}
		\Leftrightarrow \begin{cases} x_1=4\\x_2=-9\\ x_1\cdot x_2=m \end{cases}.$$
		Thay $x_1$, $x_2$ vào $x_1\cdot x_2=m$ ta được $4\cdot (-9)=m \Leftrightarrow m=-36$ (thỏa mãn).
		\end{enumerate}	
	}
\end{bt}

\begin{bt}%[Đề tuyển sinh 10 năm học 2018 - 2019 Bến Tre]%[Nguyễn Cao Cường, dự án EXTHCS-06]%[9H3B7]%[8H3K8]%[9H4G6]%[7H2K6]
	Cho đường tròn $(O;R)$ có đường kính $AB$ vuông góc với dây cung $MN$ tại $H$ ($H$ nằm giữa $O$ và $B$). Trên tia $MN$ lấy điểm $C$ nằm ngoài đường tròn $(O;R)$ sao cho đoạn thẳng $AC$ cắt đường tròn $(O;R)$ tại điểm $K$ ($K$ khác $A$), hai dây $MN$ và $BK$ cắt nhau tại $E$.
	\begin{enumerate}
		\item Chứng minh rằng tứ giác $AHEK$ là tứ giác nội tiếp.
		\item Chứng minh $CA\cdot CK=CE\cdot CH$.
		\item Qua điểm $N$ kẻ đường thẳng $d$ vuông góc với $AC$, $d$ cắt tia $MK$ tại $F$. Chứng minh tam giác $NFK$ cân.
		\item Khi $KE=KC$, chứng minh rằng $OK\parallel MN$.
	\end{enumerate}
	\loigiai{
		
		\begin{center}
			\begin{tikzpicture}
			\tkzInit[xmin=-1, xmax=7.5, ymin=-3, ymax=6.5]
			\tkzClip
			\tkzDefPoints{1/1/A, 7/1/B, 4/1/O, 6/1/H, 6/6/C}
			\tkzInterLC(C,H)(O,A)    \tkzGetPoints{N}{M}
			\tkzInterLC(C,A)(O,A)    \tkzGetPoints{T}{K}
			\tkzInterLL(M,N)(B,K)\tkzGetPoint{E}
			\tkzDefPointBy[projection = onto A--C](N)    \tkzGetPoint{U}
			\tkzInterLL(N,U)(M,K)    \tkzGetPoint{F}
			\tkzDrawPoints(A, B, O, C, H, M, N, K, E, F)
			\tkzDrawCircle(O,A)
			\tkzLabelPoints(H)
			\tkzLabelPoints[above](C, F)
			\tkzLabelPoints[above=10pt](K)
			\tkzLabelPoints[right=10pt, above](E)
			\tkzLabelPoints[left](A)
			\tkzLabelPoints[right](N, B)
			\tkzLabelPoints[below](O, M)
			\tkzDrawSegments(A,B A,C C,M K,B M,F K,O N,F K,N)
			\tkzMarkRightAngles(A,K,B K,U,N E,H,A)
			\end{tikzpicture}
		\end{center}
	\begin{enumerate}
		\item Xét tứ giác $AHEK$, ta có $\widehat{AKE}=\widehat{AHE}=90^{\circ} \Rightarrow \widehat{AKE}+\widehat{AHE}=180^{\circ}$. Suy ra tứ giác $AHEK$ nội tiếp trong đường tròn đường kính $AE$.
	
	\item Xét hai tam giác $HAC$ và $KEC$, ta có
		\begin{align*}
		&\widehat{C}\;\text{chung.}\\
		&\widehat{AHC}=\widehat{EKC}=90^{\circ}.
		\end{align*}
		Suy ra hai tam giác $HAC$ và $KEC$ đồng dạng.\\
		$\Rightarrow \dfrac{CA}{CE}=\dfrac{CH}{CK} \Rightarrow CA\cdot CK=CE\cdot CH$.
		\item Ta có $NF \parallel KB$ (cùng vuông góc với $AC$). \\
		Suy ra $\widehat{FNK}=\widehat{NKB}$ (so le trong) và $\widehat{NFK}=\widehat{MKB}$ (đồng vị). \hfill (1)\\
		Mặt khác, ta có
		\begin{align*}
		&\widehat{NKB} \; \text{chắn cung } NB.\\
		&\widehat{MKB} \; \text{chắn cung } MB.\\
		&\text{ và }  \wideparen{MB}=\wideparen{NB}.
		\end{align*}
		Nên $\widehat{NKB}=\widehat{MKB}$ (hai góc nội tiếp chắn hai cung bằng nhau). \hfill (2)\\
		Từ (1) và (2) suy ra $\widehat{FNK}=\widehat{NFK}$.\\
		Vậy tam giác $NFK$ cân tại $K$.
		\item Khi $KE=KC$ ta có tam giác $KEC$ vuông cân tại $K$. \\
		Suy ra $\widehat{KCE}=\widehat{KEC}=45^{\circ}$.\\
		Từ đó ta có $\widehat{KAB}=\widehat{KBA}=45^{\circ}$ nên tam giác $KAB$ vuông cân tại $K$. Suy ra $KO\perp AB$.\\
		Ta lại có $MN\perp AB$ (giả thuyết). Vậy $KO \parallel MN$.
	\end{enumerate}	

	}
\end{bt}
