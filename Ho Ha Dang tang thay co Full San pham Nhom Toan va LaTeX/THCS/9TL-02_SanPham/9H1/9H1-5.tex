Không dùng máy tính bỏ túi, hãy sắp xếp giá trị các tỉ số lượng giác sau theo thứ tự tăng dần.
$$\sin 20^{\circ}, \cos 20^{\circ}, \sin 35^{\circ}, \cos 40^{\circ}.$$
\loigiai{
Ta có
$$\cos 20^{\circ} = \sin 70^{\circ}, \cos 40^{\circ}= \sin 50^{\circ}.$$
Vì khi góc $\alpha$ tăng từ $0^{\circ}$ đến $90^{\circ}$ thì $\sin \alpha$ tăng, do đó
$$\sin 20^{\circ} < \sin 35^{\circ} < \sin 50^{\circ} < \sin 70^{\circ}.$$

$$\sin 20^{\circ} < \sin 35^{\circ} < \cos 40^{\circ} < \cos 20^{\circ}.$$
}
\end{bt}


\begin{bt}%[9H1-5-DeKiemTra-1B]%[Mai Sương 9TL-02]%[9H1B1]
Cho tam giác $MNP$ vuông tại $M$, đường cao $MH$. Biết rằng $MN=36$ cm, $MP=48$ cm. Tính độ dài các đoạn thẳng $HM,HN,HP$.
\loigiai{
\begin{center}
\begin{tikzpicture}%tam giác
\tkzDefPoints{0/0/M, 0/3/N, 4/0/P}
\tkzDefPointBy[projection=onto N--P](M) \tkzGetPoint{H}
%\tkzLabelPoints(M,N,P,H)
\tkzDrawPoints(M,N,P,H)
\tkzDrawSegments(M,N M,P N,P M,H)
\tkzMarkRightAngles(M,H,P N,M,P)
\tkzLabelPoints[below right](P)
\tkzLabelPoints[above left](N)
\tkzLabelPoints[below left](M)
\tkzLabelPoints[above right](H)
\tkzLabelSegments[below](M,P){$48$ cm}
\tkzLabelSegments[pos=0.7,left=0.2,rotate=90](M,N){$36$ cm}
\end{tikzpicture}
\end{center}
Xét tam giác $MNP$ vuông tại $M$, đường cao $MH$ có
\begin{itemize}
\item 	Tính $NP$ \begin{eqnarray*}
		&NP^2&=MN^2 + MP^2 \quad \text{ (Định lí Py-ta-go)}\\
		&&=36^2 + 48^2=3600 \\
		&\Rightarrow NP &= 60 \text{ (cm).}
		\end{eqnarray*} 
\item 	Tính $HM$ \begin{eqnarray*}
		&&MH \cdot NP = MN \cdot MP \quad \text{ (Hệ thức về cạnh và đường cao trong tam giác vuông)}\\
		&\Leftrightarrow &MH \cdot 60 = 36 \cdot 48 \\
		&\Leftrightarrow &MH = \dfrac{36 \cdot 48}{60} = 28{,}8 \text{ (cm).}
		\end{eqnarray*} 
\item 	Tính $HN$ \begin{eqnarray*}
		&&MN^2=NH \cdot NP \quad \text{ (Hệ thức về cạnh và đường cao trong tam giác vuông)}\\
		&\Leftrightarrow & 36^2= NH \cdot 60\\
		&\Leftrightarrow & NH = \dfrac{36^2}{60}=21{,}6\text{ (cm).}
		\end{eqnarray*}
\item 	Tính $HM$ $$HP=NP-NH=60-21{,}6=38{,}4 \text{ (cm)}.$$
	Vậy $$HM=28{,}8 \text{ cm}	,HN=21{,}6\text{ cm},HP=38{,}4 \text{ cm}.$$
\end{itemize}
}

\end{bt}





\begin{bt}%[9H1-5-DeKiemTra-1B]%[Mai Sương 9TL-02]%[9H1K2]
Cho tam giác $ABC$ có $AB=6$ cm, $AC=4{,}5$ cm, $BC=7{,}5$ cm.
\begin{enumerate}[a)]
\item Chứng minh tam giác $ABC$ vuông tại $A$.
\item Tính $\widehat{B}, \widehat{C}$ và đường cao $AH$.
\item Lấy một điểm $M$ bất kì trên cạnh $BC$ ($M$ khác $B,C$). Gọi hình chiếu của $M$ trên $AB,AC$ lần lượt là $P$ và $Q$. Chứng minh $PQ=AM$.
\item Xác định vị trí của điểm $M$ để $PQ$ có độ dài nhỏ nhất.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}%tam giác
\tkzInit[xmin=-0.5,xmax=5,ymin=-0.5,ymax=6.5] \tkzClip
\tkzDefPoints{0/0/A, 0/6/B, 4.5/0/C}
\tkzDefPointBy[projection=onto B--C](A)\tkzGetPoint{H}
\tkzDefPointBy[homothety=center B ratio 0.35](C) \tkzGetPoint{M}
\tkzDefPointBy[projection=onto A--B](M)\tkzGetPoint{P}
\tkzDefPointBy[projection=onto A--C](M)\tkzGetPoint{Q}
%\tkzLabelPoints(A,B,C,M,H,P,Q)
\tkzDrawPoints(A,B,C,M,H,P,Q)
\tkzDrawSegments(A,B A,C B,C A,H A,M M,P M,Q)
\tkzMarkRightAngles(B,A,C B,P,M M,Q,C A,H,C)
\tkzLabelPoints[left](P)
\tkzLabelPoints[below right](C)
\tkzLabelPoints[above left](B)
\tkzLabelPoints[below left](A)
\tkzLabelPoints[above right](M,H)
\tkzLabelPoints[below](Q)
\end{tikzpicture}
\end{center}
\begin{enumerate}[a)]
\item Ta có $\begin{cases} BC^2 =7{,}5^2=56{,}25\\
 AB^2+AC^2 =6^2 +4{,}5^2=56{,}25 \end{cases} \Rightarrow BC^2 =AB^2+AC^2.$\\
Theo định lý đảo của định lý Py-ta-go, suy ra tam giác $ABC$ vuông tại $A$.
\item Xét tam giác vuông $ABC$, đường cao $AH$, ta có
\begin{itemize}
\item $\tan B=\dfrac{AC}{AB}=\dfrac{4{,}5}{6}=0{,}75$ \quad(Tỉ số lượng giác của góc nhọn trong tam giác vuông)\\
$\Rightarrow \widehat{B}\approx 36^{\circ}52'.$
\item $\widehat{C}= 90^{\circ}-\widehat{B}\approx 90^{\circ} - 36^{\circ}52'=53^{\circ}8'.$
\item $AH \cdot BC=AB \cdot AC$ \quad(Hệ thức về cạnh và đường cao trong tam giác vuông)\\
$\Rightarrow AH= \dfrac{AB \cdot AC}{AH}=\dfrac{6 \cdot 4{,}5}{7{,}5}=3{,}6$ (cm)
\end{itemize}
Vậy $$\widehat{B}\approx 36^{\circ}52',\widehat{C}\approx 53^{\circ}8',AH=3{,}6 \text{ cm}.$$
\item Xét tứ giác $APMQ$ có $\widehat{P}=\widehat{A}=\widehat{Q}=90^{\circ}$\\
Suy ra tứ giác $APMQ$ là hình chữ nhật.\\
Vậy $AM=PQ$ (Tính chất hai đường chéo của hình chữ nhật).
\item $PQ$ có độ dài nhỏ nhất
$\Leftrightarrow AM$ có độ dài nhỏ nhất\\
$\Leftrightarrow AM \perp BC$\\
$\Leftrightarrow M \equiv H$.\\
Vậy $PQ$ có độ dài nhỏ nhất khi $M$ trùng $H$.
\end{enumerate}
}

\end{bt}

	\subsection{Đề số 3B (Tự luận dành cho học sinh giỏi)}
	\setcounter{bt}{0}
	\begin{bt}
		Cho tam giác $ ABC $ vuông tại $ A $, đường cao $ AH $. Kẻ $ HE $, $ HF $ lần lượt vuông góc với $ AB $, $ AC $. Chứng minh rằng
		\begin{enumerate}
			\item $ \dfrac{EB}{FC} = \left(\dfrac{AB}{AC}\right)^3 $.
			\item $ BC \cdot BE \cdot CF = AH^3 $.
		\end{enumerate}
		\loigiai{
		\begin{center}
		\begin{tikzpicture}[scale=.8]
			\tkzInit[ymin=-1,ymax=5,xmin=-1,xmax=11]
			\tkzClip
			\tkzDefPoints{0/0/B,2.5/0/H,2.5/4/A}
			%\def\x{50}
			
			\tkzDefLine[orthogonal=through A](B,A) \tkzGetPoint{d}
			\tkzInterLL(A,d)(H,B)\tkzGetPoint{C}
			\tkzDrawAltitude(A,B)(H)\tkzGetPoint{E}
			\tkzDrawAltitude(C,A)(H)\tkzGetPoint{F}
			\tkzDrawPoints(A,B,H,C,E,F)
			\tkzLabelPoints[above](A)
			\tkzLabelPoints[left](B,E)
			\tkzLabelPoints[right](C,F)
			\tkzLabelPoints[below](H)
			\tkzDrawSegments(C,B B,A C,A H,E H,F A,H)
			\tkzMarkRightAngles(A,H,B B,E,H C,F,H B,A,C)
		\end{tikzpicture}
		\end{center}
		\begin{enumerate}
		\item
		Xét tam giác vuông $ AHB $, đường cao $ HE $ có
		\begin{align}
			BH^2 = BA \cdot BE \label{eq:6.1.1}
		\end{align}
		Xét tam giác vuông $ AHC $, đường cao $ HF $ có
		\begin{align}
			CH^2 = CA \cdot CF \label{eq:6.1.2}
		\end{align}
		Từ \eqref{eq:6.1.1} và \eqref{eq:6.1.2} suy ra
		\begin{align}
			\left(\dfrac{BH}{CH}\right)^2 = \dfrac{AB}{AC}\cdot \dfrac{BE}{CF} \label{eq:6.1.3}	
		\end{align}
		Lại xét tam giác vuông $ ABC $, đường cao $ AH $ có
		\begin{align}
			\heva{& AB^2 = BH \cdot BC\\& AC^2 = CH \cdot BC} \Rightarrow \dfrac{BH}{CH} = \left(\dfrac{AB}{AC}\right)^2 \label{eq:6.1.4}
		\end{align}
		Do đó, từ \eqref{eq:6.1.3} và \eqref{eq:6.1.4} suy ra
		\begin{align*}
			\dfrac{AB}{AC}\cdot \dfrac{BE}{CF} = \left(\dfrac{AB}{AC}\right)^4 \Leftrightarrow \dfrac{BE}{CF} = \left(\dfrac{AB}{AC}\right)^3.
		\end{align*}
		
		\item
		Ta có $ \triangle ABC \backsim \triangle EBH \Rightarrow \dfrac{BE}{BA} = \dfrac{BH}{BC} \Leftrightarrow BE = \dfrac{BH\cdot BA}{BC} $.\\
		Mà $ BH \cdot BC = AB^2 \Rightarrow BH = \dfrac{AB^2}{BC} \Rightarrow BE = \dfrac{AB^3}{BC^2} $.
		Tương tự ta có $ CF = \dfrac{AC^3}{BC^2} $. \\
		Mà $ AB \cdot AC = AH \cdot BC \Rightarrow AH = \dfrac{AB \cdot AC}{BC} $. Suy ra
		\begin{align*}
			BC \cdot BE \cdot CF = \dfrac{AB^3}{BC^2} \cdot \dfrac{AC^3}{BC^2} \cdot BC = \left(\dfrac{AB \cdot AC}{BC}\right)^3 = AH^3.
		\end{align*}
		\end{enumerate}
		}
	\end{bt}
	
	\begin{bt}

		\end{align}
		Mặt khác, do tam giác $ DAC $ vuông tại $ A $ nên
		\begin{align}
			AD^2 = DH \cdot DC = 10\left(\dfrac{10 - x}{2}\right) = 5\left(10 - x\right). \label{eq:6.2.2}
		\end{align}

		\begin{align*}
			\left(\dfrac{10 - x}{2}\right)^2 + x^2 = 5\left(10 - x\right) \Leftrightarrow 5x^2 = 100 \Leftrightarrow x = 2\sqrt{5}.
		\end{align*}
		Vậy đường cao $ AH $ có độ dài $ 2\sqrt{5} $.
		}
