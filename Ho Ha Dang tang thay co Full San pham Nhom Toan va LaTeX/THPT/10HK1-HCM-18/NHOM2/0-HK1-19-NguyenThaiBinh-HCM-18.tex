\begin{bt}%[Đề HK1, Nguyễn Thái Bình, Hồ Chí Minh 17-18][Nguyễn Ngọc Thanh Sơn][0D2B1-2][0D2B1-2]
	Tìm tập xác định của các hàm số sau:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $y=\dfrac{2x}{{x^3}-1}$.
			\item $y=\dfrac{2}{({x^2}-1)\sqrt{x+2}}$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item  $y=\dfrac{2x}{{x^3}-1}$ xác định $\Leftrightarrow x^{3}-1 \ne 0$ $\Leftrightarrow {x} \ne 1$.\\
			Vậy $\mathscr{D}=\mathbb{R}\setminus \left\{1\right\}$.
			\item $y=\dfrac{2}{({x^2}-1)\sqrt{x+2}}$ xác định $\Leftrightarrow \heva{&x+2>0\\&{x^2}-1\ne 0} \Leftrightarrow \heva{&x>-2\\&x\ne -1\\&x \ne 1.}$\\
			Vậy $\mathscr{D}=(-2;+\infty)\setminus \left\{-1;1\right\}$.
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Đề HK1, Nguyễn Thái Bình, Hồ Chí Minh 17-18][Nguyễn Ngọc Thanh Sơn][0D2B3-4] 
	Cho $(P)\colon y=x^2-6x+4$ và $(d)\colon y=a(x-2)$. Chứng minh rằng $(P)$ và $(d)$ luôn có điểm chung với mọi $a$.
	\loigiai{
		Phương trình hoành độ giao điểm của $(P)$ và $(d)$:
		\begin{eqnarray*}
			& & x^2-6x+4=a(x-2)\\
			&\Leftrightarrow & x^2-(a+6)x+4+2a=0.\\
			& & \Delta = a^2+4a+20\\
			&\Leftrightarrow & \Delta=(a+2)^2+16>0 \quad \forall {a}.
		\end{eqnarray*}
		Vậy $(P)$ và $(d)$ luôn có điểm chung với mọi $a$.
	}
\end{bt}
\begin{bt}%[Đề HK1, Nguyễn Thái Bình, Hồ Chí Minh 17-18][Nguyễn Ngọc Thanh Sơn][0D3K1-2][0D3K2-5]
	\begin{enumerate}
		\item Định $m$ để phương trình: $m(mx+3)=4x+m^2+2$ vô nghiệm.
		\item Định $m$ để phương trình: $(m-1)x^2-2(m+1)x+m+2=0$ có hai nghiệm phân biệt $x_1$, $x_2$ thỏa ${x_1}^2={x_2}^2$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $m(mx+3)=4x+m^2+2 \Leftrightarrow (m^2-4)x=m^2-3m+2$.\\
			Để phương trình vô nghiệm
					$\Leftrightarrow \heva{&m^2-4=0\\&m^2-3m+2 \ne 0}
					\Leftrightarrow \heva{&\hoac{&m=2\\&m=-2}\\&\heva{&m \ne {1}\\&m \ne {2}}}
					\Leftrightarrow m=-2.$\\
			 Vậy $m=-2$ phương trình vô nghiệm.
			\item Để phương trình có hai nghiệm phân biệt 
					$\Leftrightarrow \heva{&m-1 \ne 0\\& \Delta' > 0}
					\Leftrightarrow \heva{&m \ne 1\\& m+3 > 0}
					\Leftrightarrow \heva{&m \ne 1\\& m> -3}$.
			Theo định lý Vi-ét:$\heva{&x_1+x_2=-\dfrac{b}{a}=\dfrac{2(m+1)}{m-1}\\& x_1 x_2=\dfrac{c}{a}=\dfrac{m+2}{m-1}}.$\\
			Ta có:
					${x_1}^2={x_2}^2
					\Leftrightarrow \hoac{&x_1=x_2 \text{(loại)}\\&x_1+x_2=0}
					\Leftrightarrow \dfrac{2(m+1)}{m-1}=0
					\Leftrightarrow m=-1 $ (nhận).\\
			Vậy $m=-1$ phương trình có hai nghiệm phân biệt.
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Đề HK1, Nguyễn Thái Bình, Hồ Chí Minh 17-18][Nguyễn Ngọc Thanh Sơn][0D3K2-4][0D3K2-2] 
	Giải các phương trình 
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\sqrt{-x^2-4x-3}=2x-5$.
			\item $|x^2-x+2|=2-3x$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item  			
					$\sqrt{-x^2-4x-3}=2x-5\\
					\Leftrightarrow \heva{&2x-5 \ge 0\\& -x^2-4x-3=(2x-5)^2}\Leftrightarrow \heva{&x \ge \dfrac{5}{2}\\& 5x^2-24x+28=0}\Leftrightarrow \heva{&x \ge \dfrac{5}{2}\\&\hoac{&x=2\\&x=\dfrac{14}{5}}}\Leftrightarrow x=\dfrac{14}{5}$.\\
					Vậy $S=\left\{\dfrac{14}{5}\right\}$.
			\item 
					$|x^2-x+2|=2-3x\\
					\Leftrightarrow \heva{&2-3x \ge 0\\ &\hoac{&x^2-x+2=2-3x\\&x^2-x+2=-2+3x}}
					\Leftrightarrow \heva{&x \le \dfrac{2}{3}\\&\hoac{&x^2+2x=0\\&x^2-4x+4=0}}
					\Leftrightarrow \heva{&x \le \dfrac{2}{3}\\&\hoac{&x=2\\&x=0\\&x=-2}}
					\Leftrightarrow \hoac{&x=0\\&x=-2.}$\\
					Vậy $S=\left\{-2; 0 \right\}$.
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Đề HK1, Nguyễn Thái Bình, Hồ Chí Minh 17-18][Nguyễn Ngọc Thanh Sơn][0D3T3-5] 
	Tuổi của bố hiện nay bằng $\dfrac{12}{5}$ tuổi của con. Cách đây $5$ năm thì tuổi của bố bằng $\dfrac{43}{15}$ tuổi của con. Hỏi tuổi của bố và con hiện nay?
	\loigiai{
		Gọi $x$ là tuổi của bố và y là tuổi của con.\\
		Theo đề bài, ta có:
			$\heva{&x=\dfrac{12}{5}y\\&x-5=\dfrac{43}{15}(y-5)} \Leftrightarrow \heva{&x-\dfrac{12}{5}y=0\\&x-\dfrac{43}{15}y=-\dfrac{28}{3}} \Leftrightarrow \heva{&x=48\\&y=20}$.\\
		Vậy tuổi của bố hiện nay là $48$ và tuổi của con là $20$.
	}
\end{bt}
\begin{bt}%[Đề HK1, Nguyễn Thái Bình, Hồ Chí Minh 17-18][Nguyễn Ngọc Thanh Sơn][0H2K2-1]
	Cho tam giác $ABC$ đều cạnh $a$. Tính $\vec{AB}\left(3\vec{AC}-\vec{AB}\right)$.
	\loigiai{
		Ta có: $\vec{AB}\left(3\vec{AC}-\vec{AB}\right)=3\vec{AB}\cdot \vec{AC}-\vec{AB}^2.$\\
		Mà $\vec{AB}\cdot \vec{AC}=AB.AC\cdot \cos A=a\cdot a \cdot \cos 60^{\circ}=\dfrac{a^2}{2}$ và $(\vec{AB})^2=a\cdot a=a^2$.\\
		Vậy $\vec{AB}\left(3\vec{AC}-\vec{AB}\right)=\dfrac{3a^2}{2}-a^2=\dfrac{a^2}{2}$.
	}
\end{bt}
\begin{bt}%[Đề HK1, Nguyễn Thái Bình, Hồ Chí Minh 17-18][Nguyễn Ngọc Thanh Sơn] [0H2B3-3][0H2K2-3]
	Trong mặt phẳng $Oxy$, cho tam giác $ABC$ có $A(1;2)$, $B(0;-2)$ và $C(-4;-1)$.
	\begin{enumerate}
		\item Tam giác $ABC$ là tam giác gì?
		\item Tìm tọa độ điểm $M$ trên đường thẳng $d\colon y=x+4$ sao cho tứ giác $ABCM$ là hình thang vuông tại $A$ và $B$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Tam giác $ABC$ là tam giác gì?
			\immini{
				Ta có: $\vec{AB}=(-1;-4)\Rightarrow AB=\sqrt{17}$.\\
				$\vec{BC}=(-4;1)\Rightarrow BC=\sqrt{17}$.\\
				và $\vec{AC}=(-5;-3)\Rightarrow AC=\sqrt{34}$.\\
				$\Rightarrow \heva{&AC^2=AB^2+BC^2\\&BC=AB.}$\\
				$\Rightarrow \triangle ABC$ vuông cân tại $B$.
			}{
				\begin{tikzpicture}[scale=0.7]
				\tkzInit[ymin=-2,ymax=2,xmin=-5,xmax=2]
				\clip (-7,-3) rectangle (3,3);
				\tkzDefPoints{1/2/A,0/-2/B,-4/-1/C}
				\tkzLabelPoint[left](C){$C(-4;-1)$}
				\tkzLabelPoint[right](B){$B(0;-2)$}
				\tkzLabelPoint[above](A){$A(1;2)$}
				\tkzDrawPoints(A,B,C)
				\tkzDrawSegments(A,B B,C C,A)
				\end{tikzpicture}
			}
			\item Gọi $M(x;y)$, $M \in d \Rightarrow M(x;x+4)$.\\
			Để $ABCM$ là hình thang vuông tại $A$ và $B$ $\Leftrightarrow \heva{&\vec{AM},\vec{BC} \text{cùng hướng}\\&AM \perp AB} \Leftrightarrow \heva{&\vec{AM},\vec{BC} \text{cùng hướng}\\&\vec{AM} \cdot \vec{AB}=0}$\\
			$\Rightarrow -(x-1)-4(x+2)=0 \Leftrightarrow x=-\dfrac{7}{5}$.\\
			$\Rightarrow M \left(-\dfrac{7}{5};\dfrac{13}{5}\right) \Rightarrow \vec{AM} = \left(-\dfrac{12}{5};\dfrac{3}{5}\right)\Rightarrow \vec{AM} =\dfrac{3}{5} \vec{BC}$.\\
			$\Rightarrow \vec{AM},\vec{BC} \text{cùng hướng}$.\\
			Vậy $M \left(-\dfrac{7}{5};\dfrac{13}{5} \right)$ thỏa mãn yêu cầu đề bài.
		\end{enumerate}
	}
\end{bt}
