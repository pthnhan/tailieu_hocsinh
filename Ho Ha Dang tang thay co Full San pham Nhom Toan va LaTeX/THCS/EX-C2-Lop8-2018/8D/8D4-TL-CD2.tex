\section{Tìm giá trị nhỏ nhất, giá trị lớn nhất của một biểu thức}
\subsection{Giá trị nhỏ nhất, giá trị lớn nhất của một biểu thức}
\begin{enumerate}[1.]
	\item Cho biểu thức $ f(x,y\ldots) $\\
	Ta nói $ M $ là giá trị lớn nhất(GTLN) của biểu thức $ f(x,y\ldots) $, kí hiệu $ \max f=M $ nếu hai điều kiện sau thỏa mãn:
	\begin{itemize}
		\item[-] Với mọi $ x,y,\ldots $ để  $ f(x,y\ldots) $ xác định thì \[f(x,y\ldots)\le M\,(M\text{ là hằng số})\tag{1}\]
		\item[-] Tồn tại $ x_0,y_0,\ldots $ sao cho \[f(x_0,y_0\ldots)=M\tag{2}\]
	\end{itemize}
	\item Cho biểu thức $ f(x,y\ldots) $\\
	Ta nói $ m $ là giá trị nhỏ nhất(GTNN) của biểu thức $ f(x,y\ldots) $, kí hiệu $ \min f=m $ nếu hai điều kiện sau thỏa mãn:
	\begin{itemize}
		\item[-] Với mọi $ x,y,\ldots $ để  $ f(x,y\ldots) $ xác định thì \[f(x,y\ldots)\ge m\,(m\text{ là hằng số})\tag{1'}\]
		\item[-] Tồn tại $ x_0,y_0,\ldots $ sao cho \[f(x_0,y_0\ldots)=m\tag{2'}\]
	\end{itemize}
	\item Chú ý rằng nếu chỉ có điều kiện (1) hay (1')
 thì chưa thể nói gì về cực trị của một biểu thức.\\
 Chẳng hạn, xét biểu thức \[A=(x-1)^2+(x-3)^2.\]
 Mặc dù ta có $A\ge 0$, nhưng chưa thể kết luận được $\min A=0$ vì không tồn tại giá trị nào của $x$ để $A=0$.
 \end{enumerate}
%\setcounter{vd}{102}
\begin{vd}%[Nguyễn Thế Út, dự án EX-C2 lớp 8 - 2018]%[8D0G8]
	\label{103}
Tìm giá trị nhỏ nhất của biểu thức $$A=(x-1)^2+(x-3)^2.$$
\loigiai{
Ta có \begin{align*}
A&=x^2-2x+1+x^2-6x+9=2(x^2-4x+5)\\
 &=2(x-2)^2+2\ge 2.
\end{align*}
$A=2\Leftrightarrow x-2=0\Leftrightarrow x=2.$\\ Vậy $\min A=2$ khi và chỉ khi $x=2$.
}
\end{vd}
\subsection{Tìm giá trị nhỏ nhất, giá trị lớn nhất của biểu thức chứa một biến}
\subsubsection{Tam thức bậc hai}
\begin{vd}%[Nguyễn Thế Út, dự án EX-C2 lớp 8 - 2018]%[8D0G8]
\begin{enumerate}
\item Tìm GTNN của $A=2x^2-8x+1$.
\item Tìm GTLN của $B=-5x^2-4x+1$.
\item Cho tam thức bậc hai $P=ax^2+bx+c$.\\ Tìm GTNN của $P$ nếu $a>0$.\\ Tìm GTLN của $P$ nếu $a<0$.
 \end{enumerate}
\loigiai{
\begin{enumerate}
	\item $ A=2x^2-8x+1=2(x^2-4x+4)-7=2(x-2)^2-7\ge -7. $\\ $ \min A=-7 $ khi và chỉ khỉ $ x=2 $.
	\item $ B=-5x^2-4x+1=-5\left(x^2+\dfrac{4}{5}x+\dfrac{4}{25}\right)+\dfrac{9}{5}=-5\left(x+\dfrac{2}{5}\right)^2+\dfrac{9}{5}\le \dfrac{9}{5} $.\\ $ \max B=\dfrac{9}{5}$ khi và chỉ khi $ x=-\dfrac{2}{5} $.
	\item $ P=ax^2+bx+c=a\left(x^2+\dfrac{b}{a}x\right)+c=a\left(x+\dfrac{b}{2a}\right)^2+c-\dfrac{b^2}{4a} $.\\ Đặt $ c-\dfrac{b^2}{4a}=k $. Do $\left(x+\dfrac{b}{2a}\right)^2\ge 0  $ nên 
	\begin{itemize}
		\item Nếu $ a>0 $ thì $ a\left(x+\dfrac{b}{2a}\right)^2\ge 0 $, do đó $ P\ge k $;\\ $ \min P=k $ khi và chỉ khi $ x=-\dfrac{b}{2a} $.
		\item Nếu $ a<0 $ thì $ a\left(x+\dfrac{b}{2a}\right)^2\le 0 $, do đó $ P\le k $;\\ $ \max P=k $ khi và chỉ khi $ x=-\dfrac{b}{2a} $.
	\end{itemize}
\end{enumerate}
}
\end{vd}
\subsubsection{Đa thức bậc cao hơn hai}
\begin{vd}%[Nguyễn Thế Út, dự án EX-C2 lớp 8 - 2018]%[8D0G8]
Tìm GTNN của $$A=x(x-3)(x-4)(x-7).$$
\loigiai{
\[A=(x^2-7x)(x^2-7x+12).\] Đặt $ x^2-7x+6=y $ thì \[A=(y-6)(y+6)=y^2-36\ge -36.\]
$ \min A=-36 \Leftrightarrow y=0\Leftrightarrow x^2-7x+6=0\Leftrightarrow \hoac{&x=1\\&x=6.} $
}
\end{vd}
\subsubsection{Phân thức có tử là hằng số, mẫu là tam thức bậc hai}
\begin{vd}%[Nguyễn Thế Út, dự án EX-C2 lớp 8 - 2018]%[8D0G8]
Tìm GTNN của $$A=\dfrac{2}{6x-5-9x^2}.$$
\loigiai{
\[A=\dfrac{-2}{9x^2-6x+5}=\dfrac{-2}{(3x-1)^2+4}.\] Ta thấy $ (3x-1)^2\ge 0 $ nên $ (3x-1)^2+4\ge 4 $.\\ Để ý tính chất $ a\ge b $ thì $ \dfrac{1}{a}\le \dfrac{1}{b} $ với $ a $ và $ b $ cùng dấu. Từ đó ta có \[\dfrac{1}{(3x-1)^2+4}\le \dfrac{1}{4}\Rightarrow \dfrac{-2}{(3x-1)^2+4}\ge \dfrac{-2}{4}\Rightarrow A\ge -\dfrac{1}{2}. \]
$ \min A=-\dfrac{1}{2}\Leftrightarrow 3x-1=0\Leftrightarrow x=\dfrac{1}{3} $.
}
\end{vd}
\subsubsection{Phân thức có mẫu là bình phương của một nhị thức}
\begin{vd}%[Nguyễn Thế Út, dự án EX-C2 lớp 8 - 2018]%[8D0G8]
	\label{107}
Tìm GTNN của $$A=\dfrac{3x^2-8x+6}{x^2-2x+1}.$$
\loigiai{
\begin{enumerate}[\bf Cách 1.]
	\item Đặt $ x-1=y $ thì $ x=y+1 $. Ta có \[A=\dfrac{3(y+1)^2-8(y+1)+6}{y^2}=\dfrac{3y^2-2y+1}{y^2}=3-\dfrac{2}{y}+\dfrac{1}{y^2}.\] Lại đặt $ \dfrac{1}{y}=z $ thì \[A=3-2z+z^2=(z-1)^2+2\ge 2.\] $ \min A=2\Leftrightarrow z=1\Leftrightarrow y=1\Leftrightarrow x-1=1\Leftrightarrow x=2 $.
	\item Viết $ A $ dưới dạng tổng của $ 2 $ với một biểu thức không âm \[A=\dfrac{(2x^2-4x+2)+(x^2-4x+4)}{x^2-2x+1}=2+\dfrac{(x-2)^2}{(x-1)^2}\ge 2.\] $ \min A=2 $ khi và chỉ khi $ x=2 $.
\end{enumerate}
}
\end{vd}
\subsubsection{Các phân thức dạng khác}
\begin{vd}%[Nguyễn Thế Út, dự án EX-C2 lớp 8 - 2018]%[8D0G8]
Tìm GTNN và GTLN của $$A=\dfrac{3-4x}{x^2+1}.$$
\loigiai{
Để tìm GTNN, viết $ A $ dưới dạng \[A=\dfrac{x^2-4x+4-x^2-1}{x^2+1}=\dfrac{(x-2)^2}{x^2+1}-1\ge -1.\] $ \min A=-1 $ khi và chỉ khi $ x=2 $.\\
Để tìm GTLN, viết $ A $ dưới dạng 
\[A=\dfrac{4x^2+4-4x^2-4x-1}{x^2+1}=4-\dfrac{(2x+1)^2}{x^2+1}\le 4.\] $ \max A=4 $ khi và chỉ khi $ x=-\dfrac{1}{2} $.
}
\end{vd}
\subsection{Tìm giá trị nhỏ nhất, giá trị lớn nhất của biểu thức có quan hệ ràng buộc giữa các biến}
\begin{vd}%[Nguyễn Thế Út, dự án EX-C2 lớp 8 - 2018]%[8D0G8]
	Tìm GTNN của $ A=x^3+y^3+xy $ biết rằng $ x+y=1 $.
	\loigiai{
Sử dụng điều kiện đã cho để rút gọn biểu thức $ A $: \[A=(x+y)(x^2-xy+y^2)+xy=x^2-xy+y^2+xy=x^2+y^2.\]Đến đây có nhiều cách giải
\begin{enumerate}[\bf Cách 1.]
	\item Biểu thị $ y $ theo $ x $ rồi đưa về tam thức bậc hai theo $ x $.\\
	Thay $ y=1-x $ vào biểu thức $ A $
	\[A=x^2+(1-x)^2=2(x^2-x)+1=2\left(x-\dfrac{1}{2}\right)^2+\dfrac{1}{2}\ge \dfrac{1}{2}.\]
	$ \min A=\dfrac{1}{2} $ khi và chỉ khi $ x=\dfrac{1}{2};\,y=\dfrac{1}{2} $.
	\item Sử dụng điều kiện đã cho làm xuất hiện một biểu thức mới có chứa $ A $
	\[x+y=1\Rightarrow x^2+2xy+y^2=1\tag{1}\]
	Mặt khác \[(x-y)^2\ge 0\Rightarrow x^2-2xy+y^2\ge 0\tag{2} \] 
	Cộng $ (1) $ và $ (2) $, ta được 
	\[2(x^2+y^2)\ge 1\Rightarrow x^2+y^2\ge \dfrac{1}{2}. \] $ \min A=\dfrac{1}{2} $ khi và chỉ khi $ x=y=\dfrac{1}{2} $.
	\item Sử dụng điều kiện đã cho để đưa vào một biến mới.\\ Đặt $ x=\dfrac{1}{2}+a $ thì $ y=\dfrac{1}{2}-a $. Biểu thị $ x^2+y^2 $ theo $ a $, ta được \[x^2+y^2=\left(\dfrac{1}{2}+a\right)^2+\left(\dfrac{1}{2}-a\right)^2=\dfrac{1}{2}+2a^2\ge \dfrac{1}{2}.\] $ \min A=\dfrac{1}{2} \Leftrightarrow a=0\Leftrightarrow x=y=\dfrac{1}{2}$.
\end{enumerate}	
}
\end{vd}

\begin{vd}%[Nguyễn Thế Út, dự án EX-C2 lớp 8 - 2018]%[8D0G8]
	Cho $ x+y+z=3 $.
	\begin{enumerate}
		\item Tìm GTNN của $ A=x^2+y^2+z^2 $.
		\item Tìm GTLN của $ B=xy+yz+zx $.
		\item Tìm GTNN của $ A+B $.
	\end{enumerate}
	\loigiai{
		Bình phương hai vế của đẳng thức $ x+y+z=3 $, ta được \[x^2+y^2+z^2+2(xy+yz+zx)=9\tag{1}\] tức là $ A+2B=9 $.\\ Dễ dàng chứng minh được \[A\ge B\tag{2}\] xảy ra đẳng thức khi và chỉ khi $ x=y=z $.
	\begin{enumerate}
		\item Từ $ (1) $ và $ (2) $ suy ra $ 3A\ge A+2B=9 $, nên $ A\ge 3 $.\\ Do đó $ \min A=3 $ khi và chỉ khi $ x=y=z=1 $.
		\begin{note}
			Có thể giải câu a bằng cách đổi biến $ x=1+a $, $ y=1+b $, $ z=1+c $ rồi xét $x^2+y^2+z^2  $.
		\end{note}
	\item Từ $ (1) $ và $ (2) $ suy ra $ 3B\le A+2B=9 $, nên $ B\le 3 $. Do đó $ \max B=3 $ khi và chỉ khi $ x=y=z=1 $.
	\begin{note}
		Có thể giải câu b dựa vào câu a: Vì $ A+2B=9 $ nên $ B $ lớn nhất khi và chỉ khi $ A $ nhỏ nhất.
	\end{note}
\item Ta có $ A+2B=9 $ mà $ B\le 3 $ (câu b) nên $ A+B\ge 6 $.\\ Do đó $ \min (A+B)=6 $ khi và chỉ khi $ x=y=z=1 $.
	\end{enumerate}
}
\end{vd}

\begin{vd}%[Nguyễn Thế Út, dự án EX-C2 lớp 8 - 2018]%[8D0G8]
	Tìm GTNN và GTLN của
	\begin{enumerate}
		\item Biểu thức $ A $, biết rằng $ A(A-1)\le 2 $;
		\item Biểu thức $ A =2-x-y-z $, biết rằng \[(2-x-y-z)^2=4-x^2-y^2-z^2.\]
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta sử dụng phương pháp xét dấu để tìm GTNN, GTLN của $ A $. Ta biến đổi
	\[A(A-1)\le 2\Leftrightarrow A^2-A-2\le 0\Leftrightarrow (A+1)(A-2)\le 0. \]
	Lập bảng xét dấu
\begin{center}

\tkzTabInit[lgt=3,espcl=2]
{$A$ /1, $A+1$ /1, $A-2$ /1, $(A+1)(A-2)$ /1.3}

	Do đó $ -1\le A\le 2 $ nên $ \min A=-1 $ và $ \max A=2 $.
	\item Từ giả thiết ta có \begin{align*}
		&x+y+z=2-A\tag{1}\\
		&x^2+y^2+z^2=4-A^2\tag{2}
	\end{align*}
Ta đưa ra một bất đẳng thức trong đó chứa $ x+y+z $ và $ x^2+y^2+z^2 $.\\ Ta có 
\[(x+y+z)^2=x^2+y^2+z^2+2(xy+yz+xz)\tag{3}.\] Mặt khác, dễ dàng chứng minh được 
\[xy+yz+zx\le x^2+y^2+z^2\tag{4}.\] Từ $ (3) $ và $ (4) $ suy ra $ (x+y+z)^2\le 3(x^2+y^2+z^2) $; xảy ra đẳng thức khi và chỉ khi $ x=y=z $.\\ Thay các biểu thức $ (1) $, $ (2) $ vào bất đẳng thức trên, ta có $$ (2-A)^2\le 3(4-A^2) \Leftrightarrow A^2-A-2\le 0.$$ Giải tiếp như câu a ta được \begin{itemize}
	\item $ \min A=-1\Leftrightarrow \heva{&x+y+z=3\\&x=y=z}\Leftrightarrow x=y=z=1 $.
	\item $ \max A=2\Leftrightarrow \heva{&x+y+z=0\\&x=y=z}\Leftrightarrow x=y=z=0 $.
\end{itemize}
\end{enumerate}
}
\end{vd}
\subsection{Các chú ý khi tìm giá trị nhỏ nhất, giá trị lớn nhất của một biểu thức}
\begin{note}
	Khi tìm cực trị(GTNN hay GTLN) của một biểu thức, ta có thể đổi biến.\\ 
	Chẳng hạn, ở ví dụ \ref{103} ta có thể đặt $ x-2=y $, khi đó 
	\[A=(y+1)^2+(y-1)^2=2y^2+2\ge 2.\] 
	Suy ra $ \min A=2 $ khi $ y=0 \Leftrightarrow x=2$.
\end{note}
\begin{note}
	Khi tìm cực trị của biểu thức, nhiều khi ta thay điều kiện để biểu thức đạt cực trị bởi điều kiện tương đương là biểu thức khác đạt cực trị.\\ 
	Chẳng hạn:
	\begin{itemize}
		\item $ -A $ lớn nhất $ \Leftrightarrow A $ nhỏ nhất.
		\item $ \dfrac{1}{B} $ lớn nhất $ \Leftrightarrow B $ nhỏ nhất với $ B>0 $.
		\item $ C $ lớn nhất $ \Leftrightarrow C^2 $ lớn nhất với $ C>0 $.
	\end{itemize}
\end{note}

\begin{vd}%[Nguyễn Thế Út, dự án EX-C2 lớp 8 - 2018]%[8D0G8]
	Tìm GTNN của $ A=\dfrac{x^4+1}{(x^2+1)^2} $.
	\loigiai{Chú ý rằng $ A>0 $ nên $ A $ lớn nhất $ \Leftrightarrow $ $ \dfrac{1}{A} $ nhỏ nhất và $ A $ nhỏ nhất $ \Leftrightarrow $ $ \dfrac{1}{A} $ lớn nhất. Ta có
		\[\dfrac{1}{A}=\dfrac{(x^2+1)^2}{x^4+1}=\dfrac{x^4+2x^2+1}{x^4+1}=1+\dfrac{2x^2}{x^4+1}.\]
		Tìm GTLN của $ A $: Ta có $ 2x^2\ge 0 $, $ x^4+1>0 $ nên $ \dfrac{2x^2}{x^4+1}\ge 0 $. Suy ra $ \dfrac{1}{A}\ge 1+0=1. $\\
		$ \min \dfrac{1}{A}=1 $ khi và chỉ khi $ x=0 $. Do đó $ \max A=1 $ khi và chỉ khi $ x=0 $.\\
		Tìm GTNN của $ A $: Ta có $ 2x^2\le x^4+1 $ (dễ chứng minh, dấu \lq\lq $ = $ \rq\rq xảy ra khi và chỉ khi $ x^2=1 $) mà $ x^4+1>0 $ nên $\dfrac{2x^2}{x^4+1}\le 1$. Suy ra $ \dfrac{1}{A}\le 1+1=2 $.\\
		$ \max \dfrac{1}{A}=2 $ khi và chỉ khi $ x^2=1 $. Do đó $ \min A=\dfrac{1}{2} $ khi và chỉ khi $ x=\pm 1 $.
	\begin{note}
		\begin{enumerate}[1.]
			\item Cách khác tìm GTLN của $ A $
			     \[A=\dfrac{(x^2+1)^2-2x^2}{(x^2+1)^2}=1-\dfrac{2x^2}{(x^2+1)^2}\le 1.\] $ \max A=1 $ khi và chỉ khi $ x=0 $.
			\item Cách khác tìm GTNN của $ A $
			\begin{enumerate}[\bf Cách 1.]
				\item Đặt $ \dfrac{1}{x^2+1}=y $ như {\bf Ví dụ \ref{107}}.
				\item $$A=\dfrac{2x^4+2}{(x^2+1)^2}=\dfrac{(x^2+1)+(x^2-1)^2}{2(x^2+1)^2}=\dfrac{1}{2}+\dfrac{(x^2-1)^2}{2(x^2+1)^2}\ge \dfrac{1}{2}. $$
				$ \min A=\dfrac{1}{2} $ khi và chỉ khi $ x=\pm 1 $.
			\end{enumerate}
		\end{enumerate}
	\end{note}
}
\end{vd}
\begin{note}
	Khi giải toán cực trị, nhiều khi ta cần xét nhiều khoảng giá trị của biến, sau đó so sánh các giá trị của biểu thức trong các khoảng ấy để tìm GTNN, GTLN.
\end{note}
\begin{vd}%[8D0G8]
	Tìm GTLN và GTNN của \[A=(3x-1)^2-4|3x-1|+5.\]
	\loigiai{Đặt $ |3x-1|=y $ thì \[A=y^2-4y+5=(y-2)^2+1\ge 1.\]
		$ \min A=1\Leftrightarrow y=2\Leftrightarrow |3x-1|=2\Leftrightarrow\hoac{&x=1\\&x=-\dfrac{1}{3}.}$
	
}
\end{vd}

\begin{vd}%[Nguyễn Thế Út, dự án EX-C2 lớp 8 - 2018]%[8D0G8]
	Tìm GTNN của \[B=|x-2|+|x-3|.\]
	\loigiai{
\begin{enumerate}[\bf Cách 1.]
	\item 
	\begin{itemize}
		\item Xét khoảng $ x<2 $ thì $ B=2-x+3-x=5-2x $.\\ Do $ x<2 $ nên $ -2x>-4 $, do đó $ B>1 $\hfill (1)
		\item Xét $ 2\le x\le 3 $ thì $ B=x-2+3-x=1 $\hfill (2)
		\item Xét $ x>3 $ thì $ B=x-2+x-3=2x-5 $.\\ Do $ x>3 $ nên $ 2x>6 $, do đó $ B>1 $\hfill (3)
	\end{itemize}
So sánh $ (1) $, $ (2) $ và $ (3) $ ta được $ \min B=1 $ khi và chỉ khi $ 2\le x\le 3 $.
\item Do giá trị tuyệt đối của một số lớn hơn hoặc bằng số đó nên 
    \[B=|x-2|+|x-3|=|x-2|+|3-x|\ge (x-2)+(3-x)=1.\]
    Do đó $ \min B=1\Leftrightarrow\heva{&x-2\ge 0\\&3-x\ge 0}\Leftrightarrow 2\le x\le 3 $.
\end{enumerate}	
}
\end{vd}

\begin{vd}%[Nguyễn Thế Út, dự án EX-C2 lớp 8 - 2018]%[8D0G8]
	Tìm GTNN của \[A=|x-1|+|x-7|+|x-9|.\]
	\loigiai{
	Do giá trị tuyệt đối của một số lớn hơn hoặc bằng số đó nên 
	\[|x-1|+|x-9|=|x-1|+|9-x|\ge x-1+9-x=8.\tag{1}\]
	Ta lại có $ |x-7|\ge 0 $\hfill (2)\\ Từ $ (1) $ và $ (2) $ suy ra $ A\ge 8 $.\\ Do đó $ \min A=8\Leftrightarrow\heva{&x-1\ge 0\\&9-x\ge 0\\&x=7}\Leftrightarrow x=7 $.
}
\end{vd}

\begin{note}

	\begin{itemize}
		\item Nếu hai số có tổng không đổi thì tích của chúng có giá trị lớn nhất khi và chỉ khi hai số đó bằng  nhau.
		\item Nếu hai số dương có tích không đổi thì tổng của chúng nhỏ nhất khi và chỉ khi hai số đó bằng nhau.
	\end{itemize}
Để chứng minh hai mệnh đề trên, ta dùng bất đẳng thức $ (a+b)^2\ge 4ab $:
   \begin{itemize}
   	\item Nếu hai số $ a $ và $ b $ có $ a+b=k $ (hằng số) thì từ $ (a+b)^2\ge 4ab $ ta có $ ab\le \dfrac{k^2}{4} $, do đó $ \max (ab)=\dfrac{k^2}{4} $ khi và chỉ khi $ a=b $.
   	\item Nếu hai số dương $ a $ và $ b $ có $ ab=p $ (hằng số) thì $ a+b $ nhỏ nhất $ \Leftrightarrow (a+b)^2 $ nhỏ nhất, do đó $ \min (a+b)^2 =4p$ khi và chỉ khi $ a=b $.
   \end{itemize}
\end{note}

\end{enumerate}

				\heva{&x-1\geq 0\\&3-x\geq 0\\&x-2=0\\&4-x\geq 0}\Leftrightarrow\heva{&1\leq x\leq 3\\&2\leq x\leq 4}\Leftrightarrow 2\leq x \leq 3.
			\end{align*}
			Vậy $\min B=4$ đạt được khi $2\leq x\leq 3$.	
		\end{enumerate}	
	}

	\begin{enumerate}

			\begin{align}
				x^4+y^4 \ \vdots \ 5 \tag{2}
			\end{align}		 
			Ta sẽ chứng minh $x\ \vdots \ 5$.\\
			Giả sử $x$ không chia hết cho $5$. Thế thì $x^4$ không chia hết chho $5$, $y^4$ không chia hết cho $5$, $y$ không chia hết cho $5$. Do $x$ và $y$ không chia hết cho $5$ nên $x^4$ và $y^4$ chia $5$ dư $1$, suy ra $x^4+y^4$ chia $5$ dư $2$, trái với $(2)$. Vậy $x\ \vdots \ 5$.\\

\begin{bt}%[Nguyễn Phú Thạch, dự án EX-C2 lớp 8 - 2018]%[8D0G8]
	Tìm số chính phương lớn nhất biết rằng nếu xóa hai chữ số tận cùng của nó (hai chữ số này không cùng bằng $0$), ta lại được một số chính phương.
	\loigiai{
		Gọi số chính phương phải tìm là $n^2$, ta có $n^2=100A+b$, ($A$ là số trăm, $1\leq b\leq 99$). Theo đề bài, $100A$ là số chính phương nên $A$ là số chính phương.\\
		Đặt $A=a^2$, ($a\in \mathbb{N}$). Ta cần tìm giá trị lớn nhất của $a$.
		Ta có
		{\allowdisplaybreaks
			\begin{eqnarray*}
				&&	n^2>100a^2 \Rightarrow n>10a \Rightarrow n \geq 10a+1\\
				&\Rightarrow& n^2\geq (10a+1)^2\Rightarrow 100a^2+b \geq 100a^2+20a+1\Rightarrow b\geq 20a+1.
		\end{eqnarray*}}
		Do $b\leq 99$ nên $20a+1 \geq 99 \Rightarrow a \leq 4$.\\
		Ta có $n^2=100a^2+b\leq 1600+99=1699$. Kiểm tra $42^2=1764,\,41^2=1681$. Số chính phương lớn nhất phải tìm là $1681=41^2$.
	}
\end{bt}
