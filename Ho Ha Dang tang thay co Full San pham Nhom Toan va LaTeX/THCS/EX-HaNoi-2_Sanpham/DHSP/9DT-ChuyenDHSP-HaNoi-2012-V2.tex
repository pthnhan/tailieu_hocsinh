\begin{ex}%[Đề thi vào 10, Chuyên ĐHSP Hà Nội năm 2012, vòng 2]%[9D5G5]
   Giải phương trình $\sqrt{x^2+2x+2\sqrt{x^2+2x-1}}+2x^2+4x-4=0$.
    \loigiai{ Điều kiện $x^2+2x-1\ge 0$.\\
        	Đặt $x^2+2x=a(a\ge 1)$. Phương trình trở thành:\\
    		 $\sqrt{a+2\sqrt{a-1}}+2a-4=0 \Leftrightarrow \sqrt{(\sqrt{a-1}+1)^2}=4-2a\Leftrightarrow\left|\sqrt{a-1}+1\right|=4-2a\\
    	 \Leftrightarrow\sqrt{a-1}=3-2a\Leftrightarrow \begin{cases}
    	 a&\le \dfrac{3}{2} \\
    	 a-1&=4a^2-12a+9
    	 \end{cases}\Leftrightarrow \begin{cases}
    	 a\le \dfrac{3}{2} \\
    	 (a-2)(4a-5)=0
    	 \end{cases}\Leftrightarrow a=\dfrac{5}{4}$. 
    	 
    	 Với $a=\dfrac{5}{4}$ ta có $4x^2+8x=5 \Leftrightarrow \hoac{&x=\dfrac{1}{2}\\&x=\dfrac{-5}{2}}$.
    	 Thay vào thỏa mãn điều kiện. Vậy tập nghiệm của phương trình $S=\left\{ \dfrac{1}{2}; \dfrac{-5}{2} \right \}$.
    }
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên ĐHSP Hà Nội năm 2012, vòng 2]%[9D5G5]
	Cho các số $a$, $b$, $c$ đôi một phân biệt thỏa mãn $a^2(b+c)=b^2(a+c)=2012$. Tính giá trị của biểu thức $M=c^2(a+c)$.
	\loigiai{Ta có  $\begin{cases}
			a^2(b+c)&= 2012 \\
		b^2(a+c)&= 2012
		\end{cases} \Leftrightarrow \begin{cases}
		a^2(b+c)-b^2(a+c)= 0 &(1)\\
		a^2(b+c)+b^2(a+c)= 4024 & (2)
		\end{cases} $
		
		
	$(1)\Leftrightarrow ab(a-b)+c(a+b)(a-b)=0\Leftrightarrow (a-b)(ab+bc+ca)=0 \Leftrightarrow ab+bc+ca=0 \ (\text{vì} \ a\ne b)$.
	
	$(2)\Leftrightarrow ab(a+b)+c(a+b)^2-2abc=4024\Leftrightarrow (a+b)(ab+bc+ca)-2abc=4024 \Leftrightarrow abc=-2012 $.
	
	Từ đây ta có $c(a+b)=-ab \Rightarrow c^2(a+b)=-abc=2012 $.}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên ĐHSP Hà Nội năm 2012, vòng 2]%[9D5G1]
Cho các số thực $x_1$, $x_2$,$\cdot\cdot\cdot$ ,$x_n (n\ge 3)$. Kí hiệu $\max\{x_1,x_2,\cdot\cdot\cdot x_n\}$ là số lớn nhất trong các số $x_1$, $x_2$,$\cdots$,$x_n$. Chứng minh rằng $$\max\{x_1,x_2,\cdot\cdot\cdot x_n\}\ge \dfrac{x_1+x_2+\cdot\cdot\cdot+x_n}{n}+\dfrac{|x_1-x_2|+|x_2-x_3|+\cdot\cdot\cdot+|x_{n-1}-x_n|+|x_n-x_1|}{2n}.$$
	\loigiai{Đặt  $x_m=\text{Max}\{x_1,x_2,\cdots, x_n\}$, ta chứng minh $x_i+x_j+|x_i-x_j|\le 2x_m$. Thật vậy
		
		$x_i+x_j+|x_i-x_j|\le 2x_m\Leftrightarrow (2x_m-x_i-x_j)\ge |x_i-x_j|\Leftrightarrow (x_m-x_i) (x_m-x_j)\ge 0$ luôn đúng.
		
		Ta có:
		
		 $\dfrac{x_1+x_2+\cdots +x_n}{n}+\dfrac{|x_1-x_2|+|x_2-x_3|+\cdots+|x_{n-1}-x_n|+|x_n-x_1|}{2n}$
		
		$=\dfrac{x_1+x_2+|x_1-x_2|+x_2+x_3+|x_2-x_3|+\cdots+x_n+x_1+|x_n-x_1|}{2n}$
		
		$\le \dfrac{2x_m+2_m+\cdots+2x_m}{2n}=x_m$
		
			}
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên ĐHSP Hà Nội năm 2012, vòng 2]%[9D5G4]%[9D5G5]
	Trong lớp học có 36 bàn học cá nhân, được xếp thành 4 hàng và 9 cột (các hàng đánh số từ 1 đến 4, các cột đánh số từ 1 đến 9). Sĩ số học sinh của lớp là 35. Sau một học kỳ cô giáo chủ nhiệm xếp lại chỗ ngồi các bạn học sinh trong lớp. Đối với mỗi học sinh của lớp, giả sử trước khi chuyển chỗ, bạn ngồi ở bàn thuộc hàng thứ $m$, cột thứ $n$ và sau khi chuyển chỗ, bạn ngồi ở bàn thuộc hàng thứ $a_m$, cột thứ $a_n$, ta gắn cho bạn đó số nguyên $(a_m+a_n)-(m+n)$. Chứng minh tổng của 35 số nguyên  gắn với 35 bạn học sinh không vượt quá 11.
	\loigiai{
		
		Mượn một bạn X ở lớp khác ngồi vào chỗ trống còn lại. Khi đó tổng các số nguyên gắn cho 36 học sinh trước và sau khi đổi chỗ bằng nhau.
		Do đó tổng của 35 số nguyên ứng với 35 học sinh bằng $-x$. Trong đó $x$ là số nguyên ứng với X.
		
		Vậy tổng 35 số không vượt quá $(4+9)-(1+1)=11$.
		
		
	}
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên ĐHSP Hà Nội năm 2012, vòng 2]%[9H3G7]
Cho hình vuông $ABCD$ nội tiếp đường tròn $(O)$. Điểm $M$ thuộc cung nhỏ $CD$ của $(O)$ ($M$ khác $C$ và $D$). $MA$ cắt $DB$, $DC$ theo thứ tự $X$, $Z$. $MB$ cắt $CA$, $CD$ tại $Y$, $T$. $CX$ cắt $DY$ tại $K$.
\begin{enumerate}
	\item Chứng minh rằng $\widehat{MXT}=\widehat{TXC}$, $\widehat{MYZ}=\widehat{ZYD}$ và $\widehat{CKD}=135^\circ$.
	\item Chứng minh rằng $\dfrac{KX}{MX}+\dfrac{KY}{MY}+\dfrac{ZT}{CD}=1$.
	\item Gọi $I$ là giao điểm của $MK$ và $CD$. Chứng minh rằng $XT$, $YZ$, $OI$ cùng đi qua tâm đường tròn ngoại tiếp tam giác $KZT$.
\end{enumerate}

	\loigiai{
		\begin{center}
			\begin{tikzpicture}
		\tkzDefPoints{0/0/D, 0/6/A, 6/6/B, 6/0/C, 4/-1.1231056/M}
		\tkzDrawSegments(D,B A,C A,B A,D B,C C,D)
		\tkzLabelPoints[below](C, D) \tkzLabelPoints[left](A), \tkzLabelPoints[right](B)
		\tkzDrawCircle[circum](A,B,C)

		\tkzInterLL(M,A)(D,B)\tkzGetPoint{X}\tkzLabelPoints[left](X)
		\tkzInterLL(M,B)(C,A)\tkzGetPoint{Y}\tkzLabelPoints[right](Y)
		\tkzInterLL(M,A)(C,D)\tkzGetPoint{Z}\tkzLabelPoints[below left](Z)
		\tkzInterLL(B,M)(D,C)\tkzGetPoint{T}\tkzLabelPoints[below right](T)
		\tkzDrawSegments(D,Y X,C)
		\tkzInterLL(C,X)(D,Y)\tkzGetPoint{K}\tkzLabelPoints[above](K)
		\tkzInterLL(C,D)(M,K)\tkzGetPoint{I}\tkzLabelPoints[below left](I)
			\tkzInterLL(X,T)(Y,Z)\tkzGetPoint{H}\tkzLabelPoints[right](H)
		\tkzDrawSegments(Z,Y X,T M,K H,K K,Z O,I)
		\end{tikzpicture}

\begin{enumerate}
\item $\widehat{MXD}=\dfrac{1}{2}\text{sđ}(\wideparen{AB}+\wideparen{MD})=\dfrac{1}{2}\text{sđ}(\wideparen{BC}+\wideparen{MD})=\widehat{MTD}$. Do đó tứ giác $DXTM$ nội tiếp suy ra $\widehat{MXT}=\widehat{MDC}$ và $\widehat{XTD}=\widehat{XMD}=45^\circ=\widehat{XBC}$ nên giác $XBCT$ nội tiếp, suy ra $\widehat{TXC}=\widehat{MBC}=\widehat{MDC}$. Vậy $\widehat{MXT}=\widehat{TXC}$.

Tương tự các tứ giác $MZYC$ và $AYZD$ nội tiếp, suy ra $\widehat{MYZ}=\widehat{ZYD}$.

Mặt khác vì $BXTC$ nội tiếp nên $\widehat{KCD}=\widehat{DBM}=\widehat{DCM}$, tứ giác $ADZY$ nội tiếp nên $\widehat{KDC}=\widehat{CAM}=\widehat{CDM}$. Vậy $\triangle DKC=\triangle DMC\Rightarrow \widehat{DKC}=\widehat{DMC}=135^\circ$.

\item Ta có  $\widehat{XKD}=180^\circ-\widehat{DKC}=45^\circ=\widehat{XMD}$ nên tứ giác $DXKM$ nội tiếp, suy ra $\widehat{KMH}=\widehat{KDX}$.
Mặt khác $\widehat{KCZ}=\widehat{XBT}=\widehat{KDX}$. Từ đó ta có $\widehat{KMZ}=\widehat{KCZ}$. Do đó tứ giác $KCMZ$ nội tiếp, suy ra $XK\cdot XC=XZ\cdot XM\Rightarrow \dfrac{XK}{XM}=\dfrac{XZ}{XC}=\dfrac{XZ}{XA}=\dfrac{DZ}{AB}=\dfrac{DZ}{DC}$.

Tương tự $\dfrac{YK}{YM}=\dfrac{CT}{DC}$. Từ đây ta có: $$\dfrac{KX}{MX}+\dfrac{KY}{MY}+\dfrac{ZT}{CD}=\dfrac{DZ}{CD}+\dfrac{CT}{DC}+\dfrac{ZT}{DC}=1$$

\item Gọi $H$ là giao của $XT$ và $YZ$.
Ta có tứ giác $YZMC$ nội tiếp nên $\widehat{YZC}=\widehat{YMC}=45^\circ\Rightarrow YZ\parallel BD$. Tương tự $XT\parallel AC$ suy ra $YZ\perp XT\Rightarrow \widehat{XHZ}=90^\circ$.

Mặt khác tứ giác $ZKCM$ nội tiếp nên $\widehat{ZKC}=\widehat{ZMC}=90^\circ$. Từ đây ta có $XKHZ$ nội tiếp, suy ra $\widehat{HKZ}=\widehat{HXZ}$, $\widehat{HZK}=\widehat{HXK}$. Theo câu a. ta có $\widehat{HXK}=\widehat{HXZ}$ nên $\widehat{HKZ}=\widehat{HZK}$. Vậy $HK=HZ$. Tương tự $HK=HT$, hay $H$ là tâm đường tròn ngoại tiếp $\Delta KZT$.
 
 Để ý rằng tam giác $\triangle HZT$ đồng dạng $\triangle ODC$, suy ra $\dfrac{HZ}{OD}=\dfrac{ZT}{DC}$ (1). Lại có $MI\parallel AD\parallel BC$ (vì cùng vuông góc $CD$) nên
 $\dfrac{IZ}{ID}=\dfrac{IM}{IM+AD}=\dfrac{IM}{IM+BC}=\dfrac{IT}{IC}=\dfrac{IZ+IT}{ID+IC}=\dfrac{ZT}{DC}$ (2).
 
 Từ (1) và (2) ta có $\dfrac{HZ}{OD}=\dfrac{IZ}{ID}$. Do đó $H$, $I$, $O$ thẳng hàng.
\end{enumerate}}
\end{ex}
