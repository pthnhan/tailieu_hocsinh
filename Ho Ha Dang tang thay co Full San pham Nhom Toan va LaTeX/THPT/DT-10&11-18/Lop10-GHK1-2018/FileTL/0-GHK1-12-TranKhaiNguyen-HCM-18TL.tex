\begin{bt}%[Thi GHK, Trần Khai Nguyên - HCM, 2018]%[Lê Đạt, EX-DT-10&11]%[0D1B3-2]
	Cho $ A=\{x\in\mathbb{R}:2<x\le6\} $, $ B=\{x\in\mathbb{R}:x^2\le9\} $,\\ $ C=\{x\in\mathbb{R}:x\le0 \text{ hay } 3\le x<8\} $.
	\begin{enumerate}
		\item Viết lại các tập hợp $ A $, $ B $, $ C $ dưới dạng đoạn, khoảng, nữa khoảng.
		\item Tìm $ (A\cup B)\setminus(A\cap B) $, $ \mathrm{C}_\mathbb{R} (A\cup B\cup C) $
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
		\item $ A=\left(2;6\right] $, $ B=\left[-3;3\right] $, $ C=\left(-\infty;0 \right]\cup \left[3;8\right) $.
		\item 
		\begin{itemize}
			\item	$ (A\cup B)\setminus(A\cap B)=\left[-3;6\right]\setminus\left(2;3\right]=\left[-3;2\right]\cup \left(3;6\right] $.
			\item Tìm  $ \mathrm{C}_\mathbb{R} (A\cup B\cup C) $.\\
			Ta có $ (A\cup B\cup C)=\left(-\infty;8\right) $.\\
			Nên suy ra $ \mathrm{C}_\mathbb{R} (A\cup B\cup C)=\left[8;+\infty \right) $.
		\end{itemize}
		\end{enumerate}}
\end{bt}


\begin{bt}%[Thi GHK, Trần Khai Nguyên - HCM, 2018]%[Lê Đạt, EX-DT-10&11]%[0D2B1-2]
	Tìm tập xác định của hàm số $ y=\dfrac{x-2}{\sqrt{x+4}}+\sqrt{-x+3} $.
	\loigiai{
	Điều kiện xác định của hàm số là
		 \begin{align*}
		 \heva{x+4>0\\-x+3\ge 0}   \Leftrightarrow   \heva{x&>-4\\x&\le3}\Leftrightarrow -4<x\le3
		 \end{align*}
		Vậy tập xác định của hàm số là $ \mathscr{D}=\left(-4;3\right] $.
	
		
	}
\end{bt}


\begin{bt}%[Thi GHK, Trần Khai Nguyên - HCM, 2018]%[Lê Đạt, EX-DT-10&11]%[0D2B1-4]
	Xét tính chẵn, lẻ của hàm số $ y=\dfrac{\sqrt{2018-x}-\sqrt{2018+x}}{x^3-x} $.
	\loigiai{
	Đặt $ y=f(x)=\dfrac{\sqrt{2018-x}-\sqrt{2018+x}}{x^3-x} $.\\
	Tập xác định $ \mathscr{D}=\left[-2018;2018\right]\setminus \left\lbrace -1;0;1\right\rbrace $.\\
	$\forall x\in\mathscr{D}\Rightarrow -x\in\mathscr{D} $.\\
	Ta có\\ $$ f(-x)=\dfrac{\sqrt{2018+x}-\sqrt{2018-x}}{-x^3+x}=\dfrac{\sqrt{2018-x}-\sqrt{2018+x}}{x^3-x}=f(x) .$$
	Suy ra $ y=f(x) $ là hàm số chẵn.
	}
\end{bt}



\begin{bt}%[Thi GHK, Trần Khai Nguyên - HCM, 2018]%[Lê Đạt, EX-DT-10&11]%[0D2B1-3]
	Xét sự biến thiên của hàm số $ y=\dfrac{2x+1}{x-1} $ trên $ (-\infty;1) $, từ đó suy ra sự biến thiên của hàm số trên khoảng $ (-4;0) $.
	\loigiai{
		Đặt $ y=f(x)=\dfrac{2x+1}{x-1} $ .\\
	$ \forall x_1,x_2\in(-\infty;1): x_1\ne x_2 $\\
	Ta được 
	\begin{align*}
	\dfrac{f(x_1)-f(x_2)}{x_1-x_2}
	&=\dfrac{\dfrac{2x_1+1}{x_1-1}-\dfrac{2x_2+1}{x_2-1}}{x_1-x_2}\\
	&=\dfrac{-3}{(x_1-1)(x_2-1)}
	\end{align*}
	Vì $ x_1,x_2\in(-\infty;1)$ nên $ x_1-1<0  $ và $ x_2-1<0 $.\\
	Do đó $ \dfrac{f(x_1)-f(x_2)}{x_1-x_2}=\dfrac{-3}{(x_1-1)(x_2-1)}<0 $.\\
	Suy ra hàm số $ y=f(x) $ nghịch biến trên $ (-\infty;1) $\\
	Vì hàm số nghịch biến trên $ (-\infty;1) $ nên cũng nghịch biến trên $ (-4;0) $.
}
\end{bt}




\begin{bt}%[Thi GHK, Trần Khai Nguyên - HCM, 2018]%[Lê Đạt, EX-DT-10&11]%[0D2B3-5]
	Một quả bóng được ném lên cao với vận tốc ban đầu là $ v_0=48 $ m/s và giả sử rằng sau $ t $ giây nó sẽ đạt độ cao là $ h(t)=v_0 t-4,8t^2 $.
	\begin{enumerate}
		\item Hỏi đến giây thứ $ 4 $, quả bóng ở độ cao bao nhiêu mét?
		\item Sau bao nhiêu giây thì quả bóng ở vị trí cao nhất?
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Thay t=4 và $ v_0=48 $ vào hàm số $ h(t)=v_0 t-4,8t^2 $ ta được $ h(4)=115{,}2 $ m.
		\item Quả bóng ở vị trí cao nhất khi $ h $ lớn nhất.\\

		Xét hàm số $ h(t)=48 t-4{,}8t^2 $ $ (t>0) $.\\
		Ta có tọa độ đỉnh $ I(5;120) $.\\

		\end{center}

\begin{bt}%[Thi GHK, Trần Khai Nguyên - HCM, 2018]%[Lê Đạt, EX-DT-10&11]%[0H1B2-5]
	Cho tam giác $ ABC $ vuông cân tại $ A $, cạnh $ BC=3a\sqrt{2} $. Tính $ \left|  \overrightarrow{AB}+\overrightarrow{AC} \right|  $, $ \left| \overrightarrow{BC}-\overrightarrow{BA} \right|  $.
	\loigiai{
	\begin{itemize}
		\item Tính $ \left|  \overrightarrow{AB}+\overrightarrow{AC} \right|  $.\\
			Vì tam giác $ ABC $ vuông cân tại $ A $ nên $ AB=AC=\dfrac{BC}{\sqrt{2}}=3a $.\\
		Gọi $ I $ là trung điểm $ BC $ suy ra $ AI=\dfrac{BC}{2}=\dfrac{3a\sqrt{2}}{2} $.\\
		Ta có
		$ \left|  \overrightarrow{AB}+\overrightarrow{AC} \right| =\left| 2\overrightarrow{AI}\right|=2AI=3a\sqrt{2}  $.
		\item Tính $ \left| \overrightarrow{BC}-\overrightarrow{BA} \right|  $.\\
		Ta có $ \left| \overrightarrow{BC}-\overrightarrow{BA} \right|=\left| \overrightarrow{AC}\right| =AC=3a  $.
	\end{itemize}
	}
\end{bt}

\begin{bt}%[Thi GHK, Trần Khai Nguyên - HCM, 2018]%[Lê Đạt, EX-DT-10&11]%[0H1K3-5]

Chứng minh $ 3 $ điểm $ B $, $ K $, $ E $ thẳng hàng.
