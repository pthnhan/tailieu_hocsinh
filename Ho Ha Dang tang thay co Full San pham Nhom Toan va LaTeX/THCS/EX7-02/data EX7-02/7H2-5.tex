\section{TAM GIÁC CÂN}
\subsection{Tóm tắt lí thuyết}
\begin{enumerate}[\tickEX]
	\item Tam giác cân.
		\immini{
		\begin{enumerate}
			\item Định nghĩa: Tam giác cân là tam giác có hai cạnh bằng nhau.\\
			$\triangle ABC$ cân tại $A$ $\Leftrightarrow$ $\heva{&\triangle ABC\\&AB=AC.}$
			\item Tính chất: Trong tam giác cân, hai góc ở đáy bằng nhau.\\
			$\triangle ABC$ cân tại $A$ $\Rightarrow$ $\widehat{B}=\widehat{C}.$
			\item  Dấu hiệu nhận biết:
		\end{enumerate}}
		{\begin{tikzpicture}[>=stealth,scale=1, line join = round, line cap = round]
		\tkzDefPoints{0/0/B,1.5/3/A,3/0/C}
		\tkzMarkSegments[mark=|](C,A A,B)
		\tkzDrawSegments(A,B B,C C,A)
		\tkzDrawPoints[fill=black](A,B,C)
		\tkzLabelPoints[left](B)
		\tkzLabelPoints[above](A)
		\tkzLabelPoints[right](C)
		\end{tikzpicture}
		}
			\begin{itemize}
				\item Theo định nghĩa.
				\item Nếu một tam giác có hai góc bằng nhau thì tam giác đó là tam giác cân.
			\end{itemize} 

	\item Tam giác vuông cân.
	\immini{
		\begin{enumerate}
			\item Định nghĩa: Tam giác vuông cân là tam giác vuông có hai cạnh góc vuông bằng nhau.\\
			$\triangle ABC$ vuông cân tại $A$ $\Leftrightarrow$ $\heva{&\triangle ABC\\& \widehat{A}=90^\circ\\ &AB=AC.}$
			\item Tính chất: Mỗi góc nhọn của tam giác vuông cân bằng $45^\circ$.\\
			$\triangle ABC$ vuông cân tại $A$ $\Rightarrow$ $\widehat{B}=\widehat{C}=45^\circ.$
	\end{enumerate}}
	{\begin{tikzpicture}[>=stealth,scale=1, line join = round, line cap = round]
		\tkzDefPoints{0/0/A,0/3/B,3/0/C}
		\tkzMarkRightAngles[size=0.3,fill=gray!50](C,A,B)
		\tkzMarkSegments[mark=|](C,A A,B)
		\tkzDrawSegments(A,B B,C C,A)
		\tkzDrawPoints[fill=black](A,B,C)
		\tkzLabelPoints[left](A)
		\tkzLabelPoints[above](B)
		\tkzLabelPoints[right](C)
		\end{tikzpicture}
	}
	\item Tam giác đều.
	\immini{
		\begin{enumerate}
			\item Định nghĩa: Tam giác đều là tam giác có ba cạnh bằng nhau.\\
			$\triangle ABC$ đều $\Leftrightarrow$ $\heva{&\triangle ABC\\&AB=BC=CA.}$
			\item Tính chất: Trong tam giác đều, mỗi góc bằng $60^\circ$.\\
			 $\widehat{A}=\widehat{B}=\widehat{C}=60^\circ.$
			 \item Dấu hiệu nhận biết:
	\end{enumerate}}
	{\begin{tikzpicture}[>=stealth,scale=1, line join = round, line cap = round]
		\tkzInit[xmin=-0.5,ymin=-0.5,xmax=3.5,ymax=3.25]
		\tkzClip

		\tkzDefPoints{0/0/B,3/0/A}
		\tkzDefCircle[radius](A,B)
		\tkzDefCircle[radius](B,A)
		\tkzInterCC(B,A)(A,B) \tkzGetPoints{C}{D}
		\tkzMarkSegments[mark=|](A,B A,C B,C)
		\tkzDrawSegments(A,B B,C C,A)
		\tkzDrawPoints[fill=black](A,B,C)
		\tkzLabelPoints[left](B)
		\tkzLabelPoints[above](C)
		\tkzLabelPoints[right](A)
		\end{tikzpicture}
	}
		\begin{itemize}
			\item Theo định nghĩa.
			\item Nếu một tam giác có ba góc bằng nhau thì tam giác đó là tam giác đều.
			\item Nếu một tam giác cân có một góc bằng $60^\circ$ thì tam giác đó là tam giác đều.
		\end{itemize} 
\end{enumerate}
\subsection{Các dạng toán}
\begin{dang}[Sử dụng tính chất của tam giác cân, tam giác đều, tam giác vuông cân để tính số đo góc]
Sử dụng tính chất về hai góc ở đáy tam giác cân bằng nhau, mỗi góc của tam giác đều bằng $60^\circ$, góc nhọn của tam giác vuông cân là $45^\circ$.
\end{dang}

\begin{vd}
Cho tam giác $ABC$ cân tại $A$ có $\widehat{B}=40^\circ$. Tính số đo góc $C$, góc $A$.
\loigiai
{
Tam giác $ABC$ cân tại $A$  thì $\widehat{B}=\widehat{C}$ nên $\widehat{C}=40^\circ$.\\
$\triangle ABC$ có $\widehat{A}+\widehat{B}+\widehat{C}=180^\circ$.\\
Nên $\widehat{A}+40^\circ+40^\circ=180^\circ$.\\
Suy ra $\widehat{A}=100^\circ$. 
}
\end{vd}
\begin{vd}
	Cho tam giác $ABC$ cân tại $A$ có $\widehat{A}=120^\circ$. Tính số đo góc $B$, góc $C$.
	\loigiai
	{
		Do tam giác $ABC$ cân tại $A$.\\  
		Nên $\widehat{B}=\widehat{C}=\dfrac{180^\circ-\widehat{A}}{2}=\dfrac{180^\circ-120^\circ}{2}=30^\circ$.\\
	Vậy $\widehat{B}=30^\circ$, $\widehat{C}=30^\circ$.
	}
\end{vd}
\begin{vd}[Đề thi HSG toán các vùng của Mĩ năm 1985]

	Nên $\widehat{E_1}+\widehat{E_2}=180^\circ-\dfrac{\widehat{A}+\widehat{C}}{2}=180^\circ-\dfrac{90^\circ}{2}=135^\circ$.\\
