\begin{bt}%[Đề HK1, Nguyễn Tất Thành, Hồ Chí Minh 17-18]%[Đào-V- Thủy]%[0D2B1-2]
	Tìm tập xác định của hàm số $y=\dfrac{\sqrt{3x-1}}{x^2-4}+\sqrt{3-x}$.
	\loigiai{
Hàm số xác định khi 
$$\heva{&3x-1\ge 0\\ &3-x\ge 0\\ &x^2-4\neq 0}\Leftrightarrow \heva{&x\ge\dfrac{1}{3}\\ &x\le 3\\ &x\neq\pm 2}\Leftrightarrow x\in \left[\dfrac{1}{3};3\right]\setminus \{2\}.$$	
Vậy tập xác định của hàm số đã cho là $\mathscr{D}=\left[\dfrac{1}{3};3\right]\setminus \{2\}$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Tất Thành, Hồ Chí Minh 17-18]%[Đào-V- Thủy]%[0D2K3-2]
	Tìm các hệ số $a$, $b$, $c$ của hàm số $y=ax^2+bx+c$, biết đồ thị hàm số là một đường parabol đi qua điểm $M(1;0)$ và có đỉnh $I\left(\dfrac{3}{4};-\dfrac{1}{8}\right)$.
	\loigiai{
	Vì parabol đi qua điểm $M(1;0)$ và có đỉnh $I\left(\dfrac{3}{4};-\dfrac{1}{8}\right)$ nên ta có
	$$\heva{&a+b+c=0\\ &-\dfrac{b}{2a}=\dfrac{3}{4}\\&\dfrac{9}{16}a+\dfrac{3}{4}b+c=-\dfrac{1}{8}}\Leftrightarrow \heva{&a+b+c=0\\ &6a+4b=0\\ &9a+12b+16c=-2}\Leftrightarrow \heva{&a=2\\ &b=-3\\ &c=1.}$$
	Vậy $a=2$, $b=-3$, $c=1$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Tất Thành, Hồ Chí Minh 17-18]%[Đào-V- Thủy]%[0D3B2-4]
	Giải phương trình $\sqrt{2x^2+x-11}=9-x$.
	\loigiai{
Ta có 
\begin{align*}
	\sqrt{2x^2+x-11}=9-x&\Leftrightarrow \heva{&9-x\ge 0\\ &2x^2+x-11=(9-x)^2}\\
	&\Leftrightarrow \heva{&x\le 9\\ &x^2+19x-22=0}\\
	&\Leftrightarrow \heva{&x\le 9\\ &x=4 \text{ hoặc }x=-23}\Leftrightarrow \hoac{&x=4\\ &x=-23.}
\end{align*}
Vậy phương trình đã cho có các nghiệm là $x=4$; $x=-23$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Tất Thành, Hồ Chí Minh 17-18]%[Đào-V- Thủy]%[0D3B3-3]
	Giải hệ phương trình $\heva{&3x+2y+\dfrac{8}{2x+3y}=6\\ &9x+6y-\dfrac{4}{2x+3y}=\dfrac{31}{2}.}$
	\loigiai{
Điều kiện $2x+3y\neq 0$.\\
 Đặt $a=3x+2y$, $b=\dfrac{1}{2x+3y}$. Hệ đã cho trở thành 
 $$\heva{&a+8b=6\\ &3a-4b=\dfrac{31}{2}}\Leftrightarrow \heva{&a=\dfrac{37}{7}\\ &b=\dfrac{5}{56}.}$$
 Khi đó ta được 
 $$\heva{&3x+2y=\dfrac{37}{7}\\ &\dfrac{1}{2x+3y}=\dfrac{31}{2}}\Leftrightarrow \heva{&3x+2y=\dfrac{37}{7}\\ &2x+3y=\dfrac{2}{31}}\Leftrightarrow \heva{&x=-\dfrac{229}{175}\\ &y=\dfrac{806}{175}.}$$
 Vậy hệ phương trình đã cho có nghiệm duy nhất $\left(-\dfrac{229}{175};\dfrac{806}{175}\right)$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Tất Thành, Hồ Chí Minh 17-18]%[Đào-V- Thủy]%[0D4K1-2]
	Cho hai số dương $a$, $b$ thỏa mãn $a+b=1$. Chứng minh $\dfrac{1}{1+a+2b}+\dfrac{1}{1+2a+b}\ge \dfrac{4}{5}$.
	\loigiai{
	Trước hết, ta chứng minh
	\[\text{Với }x,y>0, \text{ ta có } \dfrac{1}{x}+\dfrac{1}{y}\ge \dfrac{4}{x+y}\tag{1}\]
	Thật vậy, 
	\begin{align*}
	(1)\Leftrightarrow \dfrac{x+y}{xy}\ge \dfrac{4}{x+y}&\Leftrightarrow (x+y)^2\ge 4xy\\
	&\Leftrightarrow (x-y)^2\ge 0\quad (\text{luôn đúng})
	\end{align*}
	Khi đó
	$$\dfrac{1}{1+a+2b}+\dfrac{1}{1+2a+b}\ge \dfrac{4}{2+3(a+b)}=\dfrac{4}{5}.$$
	Dấu \lq\lq=\rq\rq\ đạt được khi $1+a+2b=1+2a+b\Leftrightarrow a=b=\dfrac{1}{2}$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Tất Thành, Hồ Chí Minh 17-18]%[Đào-V- Thủy]%[0H2B2-1]
	Cho hình vuông $ABCD$ có cạnh $a$, $N$ là trung điểm của $BC$. Tính $\vec{AD}\cdot \vec{AC}$; $\vec{AN}\cdot \vec{AD}$.
	\loigiai{
\begin{center}
	\begin{tikzpicture}
	\coordinate (A) at (0,0);
	\coordinate (B) at (4,0);
	\coordinate (C) at (4,4);
	\coordinate (D) at (0,4);
	\coordinate (N) at ($(B)!.5!(C)$);
	\draw (A)--(B)--(C)--(D)--cycle;
	\draw (A)--(N) (A)--(C);
	\tkzDrawPoints[fill=black](A,B,C,D,N)
	\tkzLabelPoints[left](A,D)
	\tkzLabelPoints[right](B,C,N)
	\end{tikzpicture}
\end{center}
\begin{itemize}
	\item $\vec{AD}\cdot \vec{AC}=\vec{AD}\cdot \left(\vec{AB}+\vec{AD}\right)=\vec{AD}^2=a^2$.
	\item $\vec{AN}\cdot \vec{AD}=\dfrac{1}{2}\left(\vec{AB}+\vec{AC}\right)\cdot \vec{AD}=\dfrac{1}{2}\vec{AD}\cdot \vec{AB}+\dfrac{1}{2}\vec{AD} \cdot\vec{AC}=\dfrac{1}{2}a^2$.
\end{itemize}
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Tất Thành, Hồ Chí Minh 17-18]%[Đào-V- Thủy]%[0H2K2-4]
	Trong mặt phẳng tọa độ $Oxy$ cho $\triangle ABC$ với $A(2;1)$, $B(-3;-4)$, $C(1;4)$. Tìm tọa độ tâm đường tròn ngoại tiếp tam giác $ABC$.
	\loigiai{
	Gọi $I(x;y)$ là tâm đường tròn ngoại tiếp $\triangle ABC$. Ta có
	\begin{align*}
	\heva{&IA=IB\\ &IA=IC}&\Leftrightarrow \heva{&(x-2)^2+(y-1)^2=(x+3)^2+(y+4)^2\\ &(x-2)^2+(y-1)^2=(x-1)^2+(y-4)^2}\\
	&\Leftrightarrow \heva{&x+y=-2\\ &x-3y=-6}\Leftrightarrow \heva{&x=-3\\ &y=1.}
	\end{align*}	
	Vậy tọa độ tâm $I(-3;1)$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Tất Thành, Hồ Chí Minh 17-18]%[Đào-V- Thủy]%[0HK2-2]
	Từ một điểm $M$ ở ngoài đường tròn tâm $O$ cho trước, kẻ đường thẳng cắt $(O)$ tại hai điểm $A$, $B$. Chứng minh $MA \cdot MB = MO^2 - OA^2$.
	\loigiai{
	\begin{center}
		\begin{tikzpicture}
		\tkzDefPoint(0,0){O}
		\tkzDefPoint(0,3){X}
		\tkzDefPoint(-7,0){M}
		\tkzDefPoint(0,2){Y}
		\tkzInterLC(M,Y)(O,X) \tkzGetPoints{A}{B}
		\pgfresetboundingbox
		\tkzDefMidPoint(A,B) \tkzGetPoint{I}
		\tkzDrawCircle(O,X)
		\tkzDrawSegments(M,B M,O O,A O,B O,I)
		\tkzMarkRightAngle(M,I,O)
		\tkzDrawPoints[fill=black](M,A,B,O,I)
		\tkzLabelPoints[below](O)
		\tkzLabelPoints[left](M)
		\tkzLabelPoints[above](I)
		\tkzLabelPoints[above left](A)
		\tkzLabelPoints[above right](B)
		\end{tikzpicture}
	\end{center}
Gọi $I$ là trung điểm của $AB$. Ta có
\begin{align*}
MO^2-OA^2&=\vec{MO}^2-\vec{OA}^2\\
&=\left(\vec{MO}+\vec{OA}\right)\cdot \left(\vec{MO}-\vec{OA}\right)\\
&=\vec{MA}\cdot \left(\vec{MO}+\vec{OB}-\vec{OA}-\vec{OB}\right)\\
&=\vec{MA}\cdot \left(\vec{MB}-2\vec{OI}\right)\\
&=\vec{MA}\cdot \vec{MB}-2\vec{MA}\cdot\vec{OI}\\
&=MA\cdot MB.
\end{align*}
Vì $\vec{MA}\cdot \vec{MB}=\left|\vec{MA}\right|\cdot \left|\vec{MB}\right|\cdot \cos\left(\vec{MA},\vec{MB}\right)=MA\cdot MB$.\\
và $\vec{MA}\cdot \vec{OI}=0$ do $MA\perp OI$. Vậy ta có đpcm.
}
\end{bt}
