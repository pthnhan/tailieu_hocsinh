\begin{ex}%[Đề thi vào 10, Sở giáo dục Yên Bái, 2016]%[8D3Y2]%[9D3Y4]%[9D4B5]%[9D4B8]
	\hfill
	\begin{enumerate} 
		\item Giải phương trình: $ 5x+6=3x. $
		\item Giải hệ phương trình: $ \heva{3x-2y=3 \\ x+2y=17}. $
		\item Tìm $ m $ để phương trình: $x^2-2(m+3)x+m^2+4m-7=0  $ có hai nghiệm phân biệt.
		\item Hàng ngày, bạn An đi học từ nhà trường trên quãng đường dài $ 8 $ km bằng xe máy điện với vận tốc không đổi. Hôm nay, vẫn trên đoạn đường đó, $ 2 $ km đầu bạn An đi với vận tốc như mọi khi, sau đó vì xe bị non hơi nên bạn đã dừng $ 1 $ phút để bơm. Để đến trường đúng giờ như mọi ngày, bạn An phải tăng vận tốc lên thêm $ 4 $ km/h. Tính vận tốc xe máy điện của bạn An khi tăng tốc. Với vận tốc đó bạn A có vi phạm luật giao thông hay không? Tại sao? Biết rằng đoạn đường bạn An đi là trong khu vực đông dân cư. 
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item $ 5x+6=3x \Leftrightarrow 2x=-6 \Leftrightarrow  x=-3. $ Vậy phương trình có tập nghiệm là $ S=\{-3\} $
		\item Ta có: 
		$ \heva{3x-2y=3 \\ x+2y=17} \Leftrightarrow  \heva{&4x=20 \\ &x+2y=17} \Leftrightarrow \heva{&x=5 \\&x+2y=17} \Leftrightarrow \heva{&x=5 \\&y=6}.  $\\
		Hệ phương trình đã cho có nghiệm duy nhất là: $ (5;6) $
		\item Phương trình đã cho có hai nghiệm phân biệt $ \Leftrightarrow  \Delta'=(m+3)^2-(m^2+4m-7)>0$\\
		$ \Leftrightarrow  2m+16>0 \Leftrightarrow m>-8. $\\
		Vậy với $ m>-8 $ thì phương trình đã cho có hai nghiệm phân biệt.
		\item Gọi vận tốc xe máy điện của An lúc bình thường là $ x $ (km/h) $ (x>0). $\\
		Vận tốc xe máy điện của An khi tăng tốc là $ x+4 $ (km/h).\\
		Thời gian An đi từ nhà đến trường lúc bình thường là: $ \dfrac{8}{x} $ (h).\\
		Đổi $ 1 $ phút $ =\dfrac{1}{60} $h. Thời gian An đi từ nhà đến trường hôm nay là: $ \dfrac{2}{x}+\dfrac{1}{60}+\dfrac{6}{x+4} $(h).\\
		Ta có: $\dfrac{8}{x}= \dfrac{2}{x}+\dfrac{1}{60}+\dfrac{6}{x+4}\Leftrightarrow \dfrac{6}{x}-\dfrac{6}{x+4}=\dfrac{1}{60}\Leftrightarrow \dfrac{24}{x(x+4)}=\dfrac{1}{60} $\\
		$ x^2+4x-1440=0\Leftrightarrow \hoac{&x=-40 \quad(\text{loại})\\ &x=36 \quad (\text{thỏa mãn})} $\\
		Vậy vận tóc xe máy điện của An khi tăng tốc là $ 36+4=40 $ (km/h).\\
		Vận tốc này không vi phạm luật giao thông vì trong khu vực đông dân cư, vận tốc tối đa của xe máy điện là $ 40 $ (km/h).
	\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Yên Bái, 2016]%[9D5G1]
	Cho $ 2 $ số dương $ a,b $ thỏa mãn $ (a+b)(a+b-1)=a^2+b^2. $ Tìm giá trị lớn nhất của biểu thức $$Q=\dfrac{1}{a^4+b^2+2ab^2} +\dfrac{1}{b^4+a^2+2ba^2}.$$
	\loigiai{
	Từ điều kiện đề bài suy ra $ (a+b)^2-(a+b)=a^2+b^2 \Leftrightarrow a+b=2ab. $\\
	Áp dụng bất đẳng thức Côsi ta có: $ a+b =2ab \leq \dfrac{(a+b)^2}{2} \Rightarrow (a+b)^2 \geq 2(a+b) \Rightarrow a+b \geq 2. $\\
	Ta lại có: $ a^4+b^2\geq 2a^2b; \quad b^4+a^2 \geq 2b^2a.$
	$$ \Rightarrow Q\leq \dfrac{1}{2a^2b+2b^2a}+\dfrac{1}{2b^2a+2a^2b}=\dfrac{2}{2ab(a+b)}=\dfrac{1}{ab(a+b)}$$
	Mà $ a+b \geq 2 \Rightarrow ab \geq 1 \Rightarrow ab(a+b) \geq 2 \Rightarrow \dfrac{1}{ab(a+b)}\leq \dfrac{1}{2} \Rightarrow Q \leq \dfrac{1}{2}$	\\
	Dấu ``='' xảy ra khi $ a=b=1. $ \\
	Vậy $ Q_{max} = \dfrac{1}{2} $ khi $ a=b=1. $
	
}
\end{ex}
