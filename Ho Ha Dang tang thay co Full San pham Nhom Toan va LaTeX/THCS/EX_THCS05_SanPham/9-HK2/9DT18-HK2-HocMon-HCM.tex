		\end{enumerate}
	}

\begin{ex}%[Toán 9, Kỳ 2, 2018, Hóc-Môn, TpHCM]%[9D4B6]%[9D4K6]
	Cho phương trình $x^2-2mx+m^2+2m-6=0$ (1) ($m$ là tham số).
	\begin{enumerate}
		\item Tìm $m$ để phương trình $(1)$ có nghiệm.
		\item Với $x_1, x_2$ là hai nghiệm của phương trình $(1)$. Tính $x_1+x_2$ và $x_1x_2$ theo $m$.
		\item Xác định $m$ để $x_1 x_2=3x_1+3x_2-1$.
	\end{enumerate}
 	\loigiai{
		\begin{enumerate}
			\item Phương trình có nghiệm $\Leftrightarrow \Delta'=-2m+6 \geq0 \Leftrightarrow m\leq 3$.\\
			\item Khi đó, áp dụng định lý Vi-ét, ta có $\begin{cases}
				x_1+x_2=2m\\ x_1x_2=m^2+2m-6.
				\end{cases}$
			\item Ta có $x_1 x_2=3x_1+3x_2-1 \Leftrightarrow m^2+2m-6=3\cdot (2m)-1 \Leftrightarrow m=-1$ hoặc $m=5$.\\
			Đối chiếu điều kiện, ta có $m=-1$ là giá trị cần tìm.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Toán 9, Kỳ 2, 2018, Hóc-Môn, TpHCM]%[9D3B5]
	Để thanh lý số tivi còn tồn kho gồm $50$ cái, một cửa hàng điện máy giảm giá $40\%$ trên một tivi với giá bán lẻ trước đó là $700.000$ đồng/cái. Sau ngày đầu tiên, cửa hàng bán được $30$ cái. Để thanh lý nhanh lô hàng, ngày hôm sau cửa hàng giảm giá thêm $10\%$ nữa (so với giá đã giảm lần $1$) nên đã bán hết số tivi còn lại. Tính số tiền mà cửa hàng thu được khi bán hết lô hàng tivi.
	\loigiai{
		Đặt $x=700.000$ đồng.\\
		Giá một cái tivi sau lần giảm thứ nhất là $0{,}6x$.\\
		Giá một cái tivi sau lần giảm thứ hai là $0{,}9\cdot 0{,}6x=0{,}54x$.\\
		Tổng số tiền mà cửa hàng thu được sau khi bán hết lô tivi là $$S=0{,}6x\cdot 30+0{,}54x\cdot 20=28{,}8x=20.160.000 \ \text{đồng}.$$
	}
\end{ex}

\begin{ex}%[Toán 9, Kỳ 2, 2018, Hóc-Môn, TpHCM]%[9D3K5]
	Bạn Thư được mẹ giao đi siêu thị mua hàng. Mẹ đưa Thư $29$ tờ tiền gồm hai loại là $20$ nghìn đồng và $50$ nghìn đồng. Sau khi mua hàng với trị giá hàng hóa là $970$ nghìn đồng, Thư còn được thối lại $30$ nghìn đồng. Hỏi mẹ đã đưa cho Thư bao nhiêu tờ tiền mỗi loại?
	\loigiai{
	Gọi $x$ là số tờ tiền loại $20$ nghìn đồng. Lúc đó, số tờ tiền loại $50$ nghìn đồng là $29-x$, với $0\leq x\leq 29$ và $x\in\mathbb{N}$.\\
	Tổng trị giá số tiền mà bạn Thư có được là $20\cdot x+50\cdot(29-x)=970+30 \Leftrightarrow x=15$.\\
	Vậy, mẹ bạn Thư đã đưa cho bạn ấy $15$ tờ $20$ nghìn đồng và $14$ tờ $50$ nghìn đồng.
	}
\end{ex}

\begin{ex}%[Toán 9, Kỳ 2, 2018, Hóc-Môn, TpHCM]%[9H3B7]%[9H3K4]%[9H3G7]
	Cho tam giác $ABC$ nhọn, $AB<AC$, nội tiếp đường tròn $(O;R)$. Gọi $H$ là giao điểm ba đường cao $AD, BE$ và $CF$ của tam giác $ABC$.
	\begin{enumerate}
		\item Chứng minh rằng tứ giác $BFEC$ nội tiếp đường tròn.
		\item Vẽ hình bình hành $BHCK$. Tính $AK$, biết $R=6$ cm.
		\item Gọi $S$ là giao điểm của $AK$ và $EF$. Đường thẳng qua $D$ và song song $HS$ cắt $AK$ tại điểm $Q$. Chứng minh rằng $AD$ là tiếp tuyến của đường tròn ngoại tiếp tam giác $DQK$. 
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=1.4]
			\tkzDefPoint(2,4){A}
			\tkzDefPoint(6,-1.5){C}
			\tkzDefPoint(0,-1.5){B}
			\tkzDefPoint(2,-1.5){D}
			%\tkzClip{xmax=7, ymax=5, xmin=-5, ymin=-5}
			%\tkzDefCircle[in](A,B,C)
			%\tkzGetPoint{I} \tkzGetLength{rIN}

			\tkzCircumCenter(A,B,C) \tkzGetPoint{O}
			\tkzDrawPoints(O)
			\tkzDrawCircle(O,A)
			\tkzTangent[at=A](O) \tkzGetPoint{t}
			%\tkzDrawCircle[R,blue](I,\rIN pt)
			\tkzLabelPoints[below](B,C)
			\tkzLabelPoints[above left](A)

			\tkzDefPointBy[projection=onto A--B](C) \tkzGetPoint{F}
			\tkzDefPointBy[projection=onto A--C](B) \tkzGetPoint{E}
			\tkzInterLL(A,D)(B,E) \tkzGetPoint{H}
			\tkzInterLC(A,O)(O,A) \tkzGetSecondPoint{K}
			\tkzInterLL(A,K)(E,F) \tkzGetPoint{S}
			\tkzDefLine[parallel=through D](H,S) \tkzGetPoint{D'}
			\tkzInterLL(D,D')(A,K) \tkzGetPoint{Q}
			\tkzDefCircle[circum](D,K,Q)
			\tkzGetPoint{P} \tkzGetLength{rPD}
			%\tkzDrawCircle[R,red](P,\rPD pt)

			\tkzInterLL(A,t)(K,E) \tkzGetPoint{T'}

			\tkzLabelPoints[above left](D,F,H,S)
			\tkzLabelPoints[below right](Q,O,K,E)
			\tkzLabelPoint[above](T) {$t$}
			\tkzDrawSegments(A,D B,E C,F C,K B,K A,K E,F D,Q D,K H,S T',T)
			\tkzMarkRightAngles[size=0.2](A,D,C B,F,C B,E,C A,S,E A,C,K A,B,K t,A,S)

	\begin{enumerate}
		\item Hai điểm $E$ và $F$ cùng nhìn $BC$ một góc $90^{\circ}$ nên tứ giác $BFEC$ nội tiếp đường tròn đường kính $BC$.
		\item Vì $BHCK$ là hình bình hành nên $KB\parallel HC$. Mà $HC \perp AB$ nên $\widehat{ABK}=90^{\circ}$, do đó $AK$ là đường kính của đường tròn $(O;R)$, suy ra $AK=2R=12$ cm.
		\item Gọi $At$ là tiếp tuyến của $(O;R)$ tại $A$. Ta có $\widehat{tAB}=\widehat{ACB}$ (cùng chắn cung $AB$) và $\widehat{ACB}=\widehat{AFE}$ (tứ giác $EFBC$ nội tiếp) nên $\widehat{tAB}=\widehat{AFE}$, đồng thời ở vị trí so le trong, do đó $At \parallel EF$. \\
		Mà $At\perp AO$ nên $EF\perp AO$.\\
		Hai tam giác vuông $ASE$ và $ACK$ đồng dạng nên $\dfrac{AE}{AK}=\dfrac{AS}{AC} \Rightarrow AE\cdot AC=AS\cdot AK$.\\
		Hơn nữa, tứ giác $DHEC$  nội tiếp nên $AH\cdot AD=AE\cdot AC$.\\
		Từ đó $AH\cdot AD=AS\cdot AK$, suy ra hai tam giác $AHS$ và $AKD$ đồng dạng, do đó $\widehat{ADQ}=\widehat{AHS}=\widehat{AKD}$.\\
		Như vậy, $AD$ là tiếp tuyến của đường tròn ngoại tiếp tam giác $DQK$.
	\end{enumerate}
	}
