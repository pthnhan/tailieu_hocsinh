\section{TẬP HỢP $\mathbb{Q}$ CÁC SỐ HỮU TỈ}
\subsection{Tóm tắt lí thuyết}
\begin{enumerate}[\tickEX]
	\item Số hữu tỉ là số viết được dưới dạng phân số $\dfrac{a}{b}$ với $a, b\in\mathbb{Z}, b\neq 0$. Tập hợp các số hữu tỉ kí hiệu là $\mathbb{Q}$.
	\item Ta có thể biểu diễn mọi số hữu tỉ trên trục số.\\
	Trên trục số, điểm biểu diễn số hữu tỉ $x$ được gọi là điểm $x$, nếu $x<y$ thì điểm biểu diễn $x$ nằm bên trái điểm biểu diễn $y$.
\begin{center}
\begin{tikzpicture}[>=stealth,line width=1pt]
\draw[->,line width = 1pt] (-5,0)--(5,0)node[below]{$x$};
\foreach \x in {-3,-2,-1,0,1,2,3}
{\draw (\x,.1)--(\x,-.1) node[below,blue]{$\x$};}
\end{tikzpicture}
\end{center}
	\item Với hai số hữu tỉ bất kì $x, y$ ta luôn có hoặc $x=y$ hoặc $x<y$ hoặc $x>y$.\\
Để so sánh hai số hữu tỉ, ta viết hai số hữu tỉ đó dưới dạng hai phân số, rồi so sánh hai phân số đó.
	\begin{itemize}
		\item Nếu $x<y$, thì trên trục số, điểm $x$ ở bên trái điểm $y$;
		\item Số hữu tỉ lớn hơn $0$ được gọi là số hữu tỉ dương;
		\item Số hữu tỉ nhỏ hơn $0$ được gọi là số hữu tỉ âm;		
		\item Số hữu tỉ $0$ không phải là số hữu tỉ dương cũng không phải là số hữu tỉ âm.
		\item Số hữu tỉ $\dfrac{a}{b}$ là số hữu tỉ dương nếu $a$ và $b$ cùng dấu, là số hữu tỉ âm nếu $a$ và $b$ khác dấu, bằng $0$ nếu $a=0.$
	\end{itemize}
\end{enumerate}
\subsection{Các dạng toán}
\begin{dang}[Nhận biết một số hữu tỉ, quan hệ tập hợp]
\begin{enumerate}[\tickEX]
	\item Muốn xem một số hữu tỉ hay không, ta hãy biến đổi xem số đó có dạng $\dfrac{a}{b}$ với $a,\ b\in\mathbb{Z}$; $b\neq 0$ hay không.
	\item Dựa vào trục số ta có thể xác định được số hữu tỉ.
\end{enumerate}
\end{dang}

\begin{vd}
Các số $0,3$; $-1,35$; $1\dfrac{3}{4}$; $0$; $-2$; $100$ có là số hữu tỉ không? Vì sao?
\loigiai
{
Ta có các số trên đều là số hữu tỉ vì:
 \begin{center}
 	$0,3=\dfrac{3}{10}$; $-1,35=\dfrac{-135}{100}=\dfrac{-27}{20}$; $1\dfrac{3}{4}=\dfrac{7}{4}$; $0=\dfrac{0}{1}$; $-2=\dfrac{-2}{1}$; $100=\dfrac{100}{1}$.
 \end{center} 

\begin{note}
Số nguyên $a$ cũng được xem là số hữu tỉ, vì $a=\dfrac{a}{1}$. Do đó, ta có $\mathbb{Z} \subset \mathbb{Q}$.
\end{note}
}
\end{vd}

\begin{vd} Chứng minh rằng không có số hữu tỉ $x$ thỏa mãn $x^2=2$. 
	\loigiai
	{
		Giả sử $x$ là số hữu tỉ, thì $x=\dfrac{a}{b}$, với $a, b\in\mathbb{Z}$; $b \neq 0$; $(a,b)=1$. Theo đề bài, ta có $a^2=2b^2$ suy ra $a^2$ chia hết cho $2$, do đó $a=2m, m\in\mathbb{Z}$. Lúc này, $b^2=2m^2$ nên $b$ cũng chia hết cho $2$. Như vậy, $a$ và $b$ có ước là $2$ trái với giả thiết $(a,b)=1$. Vậy, $x$ không phải là số hữu tỉ.
	}
\end{vd}

\begin{dang}[So sánh hai số hữu tỉ $x$ và $y$]
Viết $x, y$ dưới dạng hai phân số cùng mẫu dương $x=\dfrac{a}{m}, y=\dfrac{b}{m}$ ($m>0$).
\begin{enumerate}[\tickEX]
	\item Nếu $a<b$ thì $x<y$;
	\item Nếu $a>b$ thì $x>y$;
	\item Nếu $a=b$ thì $x=y$.
\end{enumerate}
Tổng quát $\dfrac{a}{b}<\dfrac{c}{d}\Leftrightarrow ad<bc$ ($a,\ b,\ c,\ d \in\mathbb{Z}$; $b,\ d>0$).
\end{dang}

\begin{vd}
Gỉa sử $x=\dfrac{a}{m}$, $y=\dfrac{b}{m}$ ($a,\ b,\ m \in \mathbb{Z},\ m>0$) và $x<y$. Chứng tỏ rằng nếu ta chọn $z=\dfrac{a+b}{2m}$ thì $x<z<y.$
	\loigiai{
		Vì $x<y$ nên ta có $a<b$. Do đó $$a=\dfrac{a+a}{2}<\dfrac{a+b}{2}<\dfrac{b+b}{2}=b.$$
		Suy ra $$\dfrac{a}{m}<\dfrac{a+b}{2m}<\dfrac{b}{m},$$
		hay $x<z<y.$
	}
\end{vd}

\begin{vd}
Cho số nguyên $a$ khác $0$ và $b,\ n$ là các số nguyên dương. Hãy so sánh $\dfrac{a}{b}$ và $\dfrac{a+n}{b+n}$?
\loigiai
{Ta xét các trường hợp sau
	\begin{enumerate}
		\item Nếu $a=b$ thì $\dfrac{a}{b}=\dfrac{a+n}{b+n}=1.$
		\item Nếu $a>b$ thì $a(b+n)-b(a+n)=n(a-b)>0$, nên $\dfrac{a}{b}>\dfrac{a+n}{b+n}$.
		\item Nếu $a<b$ thì $a(b+n)-b(a+n)=n(a-b)<0$, nên $\dfrac{a}{b}<\dfrac{a+n}{b+n}$.
	\end{enumerate}

}
\end{vd}

\begin{vd}
	Cho $6$ số nguyên dương $a<b<c<d<m<n$. Chứng minh rằng $$\dfrac{a+c+m+1}{a+b+c+d+m+n}<\dfrac{1}{2}.$$
	\loigiai{
	Ta có $a,\ b,\ c,\ d,\ m,\ n$ là các số nguyên dương và $a<b<c<d<m<n$, nên ta có $$b\geq a+1,\ d \geq c+1,\ n \geq m+1.$$ Suy ra $$a+b+c+d+m+n \geq 2a+2c+2m+3>2a+2c+2m+2. \eqno (1)$$
	Mặt khác $$\dfrac{1}{2}=\dfrac{a+c+m+1}{2(a+c+m+1)}=\dfrac{a+c+m+1}{2a+2c+2m+2}. \eqno (2) $$
	Từ (1) và (2) ta có 
	$$\dfrac{a+c+m+1}{a+b+c+d+m+n}<\dfrac{a+c+m+1}{2a+2c+2m+2}=\dfrac{1}{2}.$$
	Vậy bài toán được chứng minh.
}
\end{vd}
\begin{vd}
	Cho các số nguyên dương $a$ và $b$ thỏa mãn $8a^2-51b^2 \geq 1$ và $\dfrac{b}{a-2b}$ là số nguyên. Chứng minh rằng $a$ chia hết cho $b$.
	\loigiai{
		Ta có $8a^2-51b^2 \geq 1$, suy ra $8a^2 \geq 51b^2+1>51b^2$. Do đó
		$$(3a)^2=9a^2>8a^2>51b^2>49b^2=(7b)^2 \Rightarrow 3a>7b.$$
		Suy ra $3(a-2b)=3a-6b>7b-6b=b$, nên $\dfrac{b}{a-2b}<3$.\\
		Mà $\dfrac{b}{a-2b}$ là số nguyên dương nên $\dfrac{b}{a-2b} \in \{1;\ 2\}$.
		\begin{itemize}
			\item Với $\dfrac{b}{a-2b}=1$, ta có $b=a-2b$, hay $a=3b$, thỏa điều kiện $8a^2-51b^2 \geq 1$.
			\item Với $\dfrac{b}{a-2b}=2$, ta có $b=2(a-2b)=2a-4b$, hay $2a=5b$. Suy ra $b$ là số chẵn, hay $b=2m$ với $m$ là số nguyên dương. Khi đó $a=5m,$ nên $8a^2-51b^2=200m^2-2014m^2=-4m^2<0$, do đó trường hợp này ta loại.
		
		\end{itemize}
	Vậy ta có $a=3b$, hay $a$ chỉa hết cho $b$.
	}
\end{vd}

\begin{bt}

	}
\end{bt}
\begin{bt}
	\begin{enumerate}
		\item Viết ba số hữu tỉ có mẫu số khác nhau, lớn hơn $\dfrac{-1}{3}$ nhưng nhỏ hơn $\dfrac{4}{5}$.
		\item Tìm các số hữu tỉ có dạng $\dfrac{7}{a}$ biết rằng giá trị của số đó lớn hơn $\dfrac{-9}{11}$ và nhỏ hơn $\dfrac{-9}{13}$
		\item Có bao nhiêu phân số có tử số bằng $9$, lớn hơn $\dfrac{-3}{5}$ và nhỏ hơn $\dfrac{-4}{9}$?
		
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có $\dfrac{-1}{3}<\dfrac{4}{5}$, nên $\dfrac{-1}{3}<\dfrac{-1+4}{3+5}=\dfrac{3}{8}<\dfrac{4}{5}$, suy ra $\dfrac{-1}{5}<\dfrac{-1+3}{3+8}=\dfrac{2}{11}$\\ và $\dfrac{3}{8}<\dfrac{3+4}{8+5}=\dfrac{7}{13}<\dfrac{4}{5}$.\\
	Vậy ba số hữu tỉ cần tìm là $\dfrac{3}{8}$, $\dfrac{2}{11} $, $\dfrac{7}{13}$.
	\item Ta có $\dfrac{-9}{11}=\dfrac{-63}{77}$, $\dfrac{-9}{13}=\dfrac{-63}{91}$, $\dfrac{7}{a}=\dfrac{-63}{-9a}$.\\ Do đó $\dfrac{-9}{11}<\dfrac{7}{a}<\dfrac{-9}{13}$, trở thành $\dfrac{-63}{77}<\dfrac{-63}{-9a}<\dfrac{-63}{91}$. Do $-63<0$ nên ta có $77<-9a<91$, hay $\dfrac{91}{-9}<a<\dfrac{77}{-9}$, suy ra $a\in \{-10,\ -9,\ -8\}$.\\
	Vậy các phân số cần tìm là $\dfrac{7}{-10}$, $\dfrac{7}{-9}$ và $\dfrac{7}{-8}$.
	\item Xét phân số có dạng $\dfrac{9}{a}=\dfrac{-36}{-4a}$ và $\dfrac{-3}{5}=\dfrac{-36}{60}$, $\dfrac{-4}{9}=\dfrac{-36}{81}$. Do đó 
	$$\dfrac{-3}{5}<\dfrac{9}{9}<\dfrac{-4}{9}\ \text{trở thành}\ \dfrac{-36}{60}<\dfrac{-36}{-4a}<\dfrac{-36}{81},$$
	suy ra $60<-4a<81$, hay $\dfrac{81}{-4}<a<-15$, dẫn tới $a \in \{-20,\ -19,\ -18,\ -17,\ -16\}$.\\
	Vậy có $5$ phân số thỏa yêu cầu bài toán.
	\end{enumerate}
}
\end{bt}

\begin{bt}
Cho $x$ là số thỏa mãn $x^2=21$. Chứng minh rằng $x$ không là số hữu tỉ. 

\loigiai
{
Gải sử $x$ là số hữu tỉ, với $x=\dfrac{a}{b}$, $a$, $b$ $\in \mathbb{Z}$, $b \neq 0$; $(a,b)=1$. Vì $x^2=21$ nên ta có $\dfrac{a^2}{b^2}=21$, hay $a^2=21b^2$. Suy ra $a^2 \ \vdots\ 3$ hay $a\ \vdots\ 3$, do đó tồn tại số nguyên $m$ để $a=3m.$ Khi đó, ta có $9m^2=21b^2$, hay $3m^2=7b^2$. Suy ra $b^2\ \vdots \ 3$, hay là $b \ \vdots\ 3$. Lúc này, cả $a$ và $b$ chia hết cho $3$, điều này trái giả thiết $(a,b)=1$. Vậy điều giả sử ban đầu là sai, hay bài toán được chứng minh.
}
\end{bt}
