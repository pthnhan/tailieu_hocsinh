	\end{cases}$ liên tục trên $\mathbb{R}$.

			\item $f(2)=a+4$.
			\item $\displaystyle\lim_{x\rightarrow 2^+} \dfrac{\sqrt[3]{2x+4}-2}{x-2}=\displaystyle\lim_{x\rightarrow 2^+} \dfrac{2x-4}{(x-2)\left(\sqrt[3]{(2x+4)^2}+2\sqrt[3]{2x+4}+4\right)}=\dfrac{1}{6}.$

		\end{itemize}
