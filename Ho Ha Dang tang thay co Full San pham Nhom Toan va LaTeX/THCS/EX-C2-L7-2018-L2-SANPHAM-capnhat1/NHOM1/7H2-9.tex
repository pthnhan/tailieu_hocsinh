\section{Các trường hợp bằng nhau của tam giác vuông}

\subsection{Hỏi đáp nhanh}
\begin{ex}%[7H2Y8]
	Hai tam giác $ABC$ và $A'B'C'$ vuông ở $A$ và $A'$. Biết thêm $AB=A'C'$, $\widehat{B}=\widehat{C'}$. Khi đó hai tam giác này \sh{bằng nhau}, và ta viết  \sh{$\triangle ABC =\triangle A'C'B'$}.
\end{ex}
\begin{ex}%[7H2Y8]
	Đúng điền Đ, sai điền S.
	\begin{enumerate}
		\item \TF{Đ} Hai tam giác vuông có một góc nhọn bằng nhau và một cạnh góc vuông bằng nhau thì bằng nhau.
		\item \TF{Đ} Hai tam giác vuông có một góc nhọn bằng nhau và đường cao ứng với cạnh huyền bằng nhau thì bằng nhau.
		\item \TF{S} Hai tam giác vuông có các góc nhọn tương ứng bằng nhau thì bằng nhau.
		\item \TF{Đ} Hai tam giác vuông có một góc nhọn bằng nhau và cạnh huyền bằng nhau thì bằng nhau.
	\end{enumerate}	
\end{ex}

	\immini{
	Vẽ $BN \perp EM$ tại $N$.\\
