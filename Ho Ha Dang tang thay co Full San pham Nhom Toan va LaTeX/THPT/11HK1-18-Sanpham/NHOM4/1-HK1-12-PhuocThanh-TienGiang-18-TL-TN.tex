\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[Học Kì 1 lớp 11 trường THPT Phước Thạnh Tiền Giang, 2018]%[Phan Thanh Tâm, dự án(11HK1-18)]%[1D1B3-1]%[1D1K3-3]
	Giải các phương trình
	\begin{enumerate}
		\item $ 2\cos^27x-5\cos7x-7=0 $.
		\item $ \left(\sin2x+\cos2x\right)\cos x+2\cos 2x-\sin x=0 $.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Ta có  $ 2\cos^27x-5\cos7x-7=0 \Leftrightarrow \hoac{&\cos7x=-1\\&\cos7x=\dfrac{7}{2}}\Leftrightarrow x=\dfrac{\pi}{7}+k\dfrac{2\pi}{7},k\in\mathbb{Z}$.
		\item Ta có \begin{eqnarray*}
		& & \left(\sin2x+\cos2x\right)\cos x+2\cos 2x-\sin x=0\\
		&\Leftrightarrow & 2\sin x\cos^2x + \cos 2x.\cos x+2\cos 2x - \sin x=0\\
		&\Leftrightarrow & \sin x\cos2x + \cos 2x.\cos x+2\cos 2x =0\\
		&\Leftrightarrow & \cos 2x(\sin x+\cos x+2) =0\\		
		&\Leftrightarrow & \hoac{&\cos 2x=0\\&\sin x+\cos x+2=0} \\
		&\Leftrightarrow & x=\dfrac{\pi}{4}+k\dfrac{\pi}{2}, k\in \mathbb{Z}.
		\end{eqnarray*}
	\end{enumerate}
	}
\end{bt}
\begin{bt}%[Học Kì 1 lớp 11 trường THPT Phước Thạnh Tiền Giang, 2018]%[Phan Thanh Tâm, dự án(11HK1-18)]%[1D2B3-2]%[1D2B5-2]
	\begin{enumerate}
		\item Tìm số hạng chứa $ x^4 $ trong khai triển $ \left(2x-\dfrac{1}{x^2}\right)^{13} $.
		\item  Một hộp chứa 3 bi đỏ, 5 bi vàng, 6 bi xanh. Chọn ngẫu nhiên 3 bi từ hộp này. Tính xác suất sao cho 3 bi có đủ 3 màu.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Ta có $ \displaystyle\left(2x-\dfrac{1}{x^2}\right)^{13}=\sum\limits_{k=0}^{13}\mathrm{C}_{13}^k(2x)^{13-k}\left(-\dfrac{1}{x^2}\right)^{k} = \sum\limits_{k=0}^{13}\mathrm{C}_{13}^k2^{13-k}(-1)^kx^{13-3k}$.\\
		Số hạng chứa $ x^4 $ tương ứng $ 13-3k=4\Leftrightarrow k=3$.\\
		Vậy số hạng chứa $ x^4 $ là $\mathrm{C}_{13}^32^{10}(-1)^3x^{4}=-292864x^4 $.
		\item Ta có $ n(\Omega)=\mathrm{C}_{14}^3=364 $.\\
		$ n(A)=\mathrm{C}_3^1\cdot\mathrm{C}_5^1\cdot\mathrm{C}_6^1=90$.\\
		Vậy xác suất để 3 bi có đủ 3 màu là $ p(A) = \dfrac{90}{364}=\dfrac{45}{182} $.
	\end{enumerate}
	}
