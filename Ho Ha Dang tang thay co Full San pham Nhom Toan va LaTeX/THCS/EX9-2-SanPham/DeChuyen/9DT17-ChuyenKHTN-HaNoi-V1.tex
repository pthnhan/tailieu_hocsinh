		\hspace{.01mm}
		\begin{enumerate}[1)] 			
			\item Giải hệ phương trình:$$ \heva{&x^2+y^2-xy=1\\&x+x^2y=2y^3.} $$								
			\item Giải phương trình: $$2(x+1)\sqrt{x+1}=\left( \sqrt{x+1}+\sqrt{1-x}\right) \left(2-\sqrt{1-x^2}\right).$$
		\end{enumerate}	
	\loigiai{
		\begin{enumerate}[1)]
			\item 	Hệ đã cho tương đương với:
			$$
				\heva{x^2+y^2-xy=1 \qquad (1)\\x.1+x^2y=2y^3. \qquad (2)}
			$$
			Từ $ (1) $ và $ (2) $ suy ra:\\
			$ x(x^2+y^2-xy)+x^2y=2y^3 \Leftrightarrow x^3-2y^3+xy^2=0. $\\
			Với $ y=0 \Rightarrow x=0: $ không thỏa mãn hệ.\\
			 Với $ y \neq 0 $, chia cả hai vế cho $ y^3 $ ta được $ \left(\dfrac{x}{y}\right)^3-2+\dfrac{x}{y}=0. $\\
			 Đặt $ t=\dfrac{x}{y} $ ta có: $ t^3+t-2=0 \Leftrightarrow (t-1)(t^2+t+2)=0 \Leftrightarrow t=1  \Rightarrow x=y. $ \\
			 Do đó ta có: $ \heva{&x=y\\&x^2+y^2-xy=1} \Leftrightarrow \heva{x=y\\x^2=1} \Leftrightarrow x=y=\pm 1. $
			 
			 \item ĐKXĐ: $ -1 \leq x \leq 1. $ \\
			 Đặt $ \heva{u=\sqrt{x+1 }\\v=\sqrt{1-x}} \Rightarrow \heva{&u^2+v^2=2\\&2u^3=(u+v)(2-uv)} \Leftrightarrow \heva{&u^2+v^2=2\\&2u^3=(u+v)(u^2+v^2-uv)}$ \\
			  $ \Rightarrow 2u^3=u^3+v^3 \Leftrightarrow u = v$\\
			  $ \Rightarrow \sqrt{x+1}=\sqrt{1-x} \Leftrightarrow x=0 $(thỏa mãn).
		\end{enumerate}
			
	}

		\hspace{.01mm}
		\begin{enumerate}[1)]
		\item Chứng minh rằng không tồn tại các số nguyên $ x, y $ thỏa mãn đẳng thức: $$12x^2+26xy+15y^2=4617.$$
		\item Với $ a, b $ là các số thực dương, tìm giá trị lớn nhất của biểu thức: $$ M=(a+b)\left(\dfrac{1}{a^3+b}+\dfrac{1}{b^3+a} \right)-\dfrac{1}{ab}.$$ 
		\end{enumerate}
	\loigiai{
	\begin{enumerate}[1)]
		\item Đẳng thức đã cho được viết lại:\\
		  $x^2+4y^2+4xy+11x^2+11y^2+22xy=4617$ \\ 
		$\Leftrightarrow (x+2y)^2+11(x+y)^2=4617 \qquad (1) $\\
		
		\vspace*{-1cm}
		Ta có $ VT(1)\equiv0,1,3,5,9$(mod 11) \\
		mà $ 4617 \equiv 8$(mod 11)  \\
		Vậy không có số nguyên $ x, y $ nào thỏa mãn đẳng thức đã cho.
		\item Theo BĐT Bunhiacopsky ta có: \\
		 $ \left(\sqrt{a^3}\sqrt{\dfrac{1}{a}}+\sqrt{b}\sqrt{b}\right)^2 \leq (a^3+b)\left(\dfrac{1}{a}+b\right)$\\
		 $ (a+b)^2 \leq (a^3+b)(\dfrac{1}{a}+b) \Rightarrow \dfrac{a+b}{a^3+b} \leq \dfrac{\dfrac{1}{a}+b}{a+b}$\\
		 Tương tự ta cũng có: \\
		 	$ \dfrac{a+b}{a+b^3} \leq \dfrac{\dfrac{1}{b}+a}{a+b} $\\
		 	$\Rightarrow (a+b)\left(\dfrac{1}{a^3+b}+\dfrac{1}{a+b^3} \right) \leq \dfrac{1}{a+b}\left(\dfrac{1}{a}+b+\dfrac{1}{b}+a\right) $\\
			 $\Rightarrow (a+b)\left(\dfrac{1}{a^3+b}+\dfrac{1}{a+b^3} \right) \leq 1+\dfrac{1}{ab} $\\
			 $ \Rightarrow M \leq 1. $\\
			 Vậy $ Max M =1 $ khi $ a=b=1. $
	\end{enumerate}	
}

	\begin{ex}%[Đề thi vào 10, Chuyên KHTN Hà Nội, vòng 1, 2017]%[9D5G5]
		Tìm hợp số nguyên dương $ n $ sao cho tồn tại một cách sắp xếp các số $ 1, 2,\dots,n $ thành $ a_1, a_2,\dots, a_n $ mà khi chia các số $a_1,a_1a_2,a_1a_2a_3,\dots ,a_1a_2 \dots a_n$ cho $ n $ ta được các số dư đôi một khác nhau.
	\loigiai{
		Trước hết ta chứng minh bổ đề sau: Với $ n $ là hợp số và $ n>4 $ thì $ (n-1)!\vdots n $.\\
		Thật vậy, ta có: Với $ n $ là hợp số và $ n>4 $ thì $ n=a.b $ với $ a,b $ là các số nguyên khác 1 và $ n $. \\Vậy với $ 2\leq a;b \leq n-1$ thì $  (n-1)!\vdots n $.\\
		Từ giả thiết ta có $ a_n $ phải bằng $ n $, vì nếu $ a \neq n;a_i=n (i\in [1;n-1])$ thì: $$\heva{a_1a_2\dots a_i \vdots n \\ a_1a_2\dots a_n \vdots n}$$
		điều này trái với giả thiết.\\
		Do đó $ a_n=n $, nếu n là một số lớn hơn 4 và $ n $ là hợp số, theo bổ đề trên ta có: $$ a_1a_2\dots a_{n-1}=(n-1)!\vdots n. $$
		Mà $ a_1a_2\dots a_n \vdots n $ do đó hai số này chia cho $ n $ có cùng số dư là 0, điều này mâu thuẫn với giả thiết.\\
		Như vậy $ n\leq 4 \Rightarrow n=4 $ (vì $ n $ là hợp số).\\
		Xét với $ n=4 $ thì tồn tại dãy số:$ 1;3;2;4 $ có $ 1;1.3;1.3.2;1.3.2.4 $ khi chia cho 4 có số dư lần lượt là $ 1;3;2;0 $ thỏa mãn đề bài. \\
		Vậy $ n=4. $	
	}
	\end{ex}
