\begin{ex}%[Đề thi vào 10, Chuyên An Giang, 2017]% [9D1K2] 
	
	Cho $ x = \dfrac{13}{\sqrt{19+8\sqrt{3}}} $. Tính giá trị của biểu thức $ A=x^2-8x+15. $
	\loigiai{
		Ta có $19+8\sqrt{3}=16+2.4\sqrt{3}+3=(4+\sqrt{3})^2$. \\
		$ \Rightarrow x=\dfrac{13}{\sqrt{19+8\sqrt{3}}}=\dfrac{13}{\sqrt{(4+\sqrt{3})^2}}=\dfrac{13}{4+\sqrt{3}}=\dfrac{13(4-\sqrt{3})}{16-3}=4-\sqrt{3}. $\\
		$ \Rightarrow x^2=(4-\sqrt{3})^2=19-8\sqrt{3}. $\\
		Thay vào $ A $ ta được:\\
		$ A=19-8\sqrt{3}-8(4-\sqrt{3})+15 \Rightarrow A=2$.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên An Giang, 2017] %[9D2K4] 
	Cho hàm số $ y=ax+b \quad (a \neq 0) $ có đồ thị là đường thẳng $ (d) $ trên mặt phẳng tọa độ $ Oxy $. Viết theo $ a $ và $ b $ phương trình đường thẳng $ (d') $. Biết rằng $ (d) $ và $ (d') $ vuông góc với nhau đồng thời cắt nhau tại một điểm thuộc trục hoành.
	\loigiai{
		Ta có: $(d): y=ax+b \quad (a \neq 0).  $	\\
		Giả sử $ (d'): y=a'x+b' $\\
		Do $ (d) \perp (d') \Rightarrow a.a'=-1 \Rightarrow a'=\dfrac{1}{a} \quad (a \neq 0)$.\\
		Ta lại có: $ (d) $ cắt $ Ox $ tại điểm có hoành độ $ x=-\dfrac{b}{a} $ và $ (d') $ cắt $ Ox $ tại điểm có hoành độ $ x=-\dfrac{b'}{a'} $.\\
		Theo đề bài $ (d) $ và $ (d') $ cắt nhau tại cùng một điểm thuộc $ Ox $ nên:$$ -\dfrac{b}{a}=-\dfrac{b'}{a'} \Leftrightarrow b'=\dfrac{a'b}{a}=-\dfrac{b}{a^2}.$$
		Vậy đường thẳng $ (d') $ cần tìm có phương trình là: $ y=-\dfrac{1}{a}x-\dfrac{b}{a^2}. $		
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên An Giang, 2017] %[9D3K4] 
	Tìm $ x,y,z $ biết: $$ \heva{&x^2+(y-z+1)^2= 0\\&5y-3z-9 =0}.$$
	\loigiai{
		Do $ x^2 \geq 0; (y-z+1)^2 \geq 0 $ nên hệ phương trình trở thành: \\
		$$ \heva{&x=0 \\& y-z+1=0 \\&5y-3z-9=0} $$. \\
		Xét hệ $ \heva{&y-z+1=0 \\ &5y-3z-9=0} \Leftrightarrow \heva{&3y-3z+3=0 \\ &5y-3z -9=0} \Leftrightarrow \heva{&y-z+1=0 \\ &-2y+12=0} \Leftrightarrow \heva{&6-z+1=0 \\& y=6} \Leftrightarrow \heva{&y=6 \\&z=7}$\\
		Vậy nghiệm của hệ phương trình là: $ x=0; y=6; z=7. $
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên An Giang, 2017]% [9D4Y4]%[9D4K4] 
	Cho hai phương trình bậc hai ($ m $ là tham số): 
		$$ 2x^2+(m-1)x-3=0; \quad 4x^2 - (m-7)x-9=0.$$
		\begin{enumerate}[a.]
			\item Tìm $ m $ để cả hai phương trình đều có nghiệm.
			\item Tìm $ m $ để hai phương trình có ít nhất một nghiệm chung.
		\end{enumerate}
			\loigiai{
				\begin{enumerate}[a.]
					\item Dễ thấy cả hai phương trình đều có các hệ số $ a, c $ trái dấu. Do đó cả hai phương trình có nghiệm với mọi giá trị của $ m. $
					\item Giả sử cả hai phương trình có nghiệm chung là $ x_0 $ khi đó: $$ \heva{&2x_0^2 +(m-1)x_0 - 3=0\\& 4x_0^2 - (m-7)x_0 - 9 =0} \Rightarrow 6x_0^2 +6x_0 -12=0 \Rightarrow x_0=1; x_0=-2.$$
					Với $ x_0=1 \Rightarrow 2+(m-1)-3=0 \Rightarrow m =2. $\\	
					$ x_0=-2 \Rightarrow 8-2(m-1)-3=0 \Rightarrow m=\dfrac{7}{2}. $\\
					Vậy $ m=2;m=\dfrac{7}{2} $ thỏa mãn yêu cầu bài toán.
				\end{enumerate}
			}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên An Giang, 2017]% [9H3B3]%[9H3K7]%[9H3G7]  
	Cho tam giác $ ABC $ nội tiếp đường tròn $ (O) $. Biết $ \widehat{A}=60^{\circ}; \widehat{B} $ và $ \widehat{C} $ là hai góc nhọn có số đo khác nhau. Vẽ các đường cao $ BE,CF $ của tam giác $ ABC $ ($ E,F $ lần lượt thuộc $ AC,AB $).
	\begin{enumerate}[a.]
		\item Chứng minh rằng $ \widehat{BCF}=\widehat{BEF}. $
		\item Gọi $ I $ là trung điểm của $ BC $. Chứng minh tam giác $ IEF $ là tam giác đều.
		\item Gọi $ K $ là trung điểm của $ EF $. Chứng minh rằng $ IK $ song song $ OA. $
	\end{enumerate}
	\loigiai{
		\immini{
			
				a. Ta có $ BE $ là đường cao. $ \Rightarrow BE \perp EC \Rightarrow \widehat{BEC} =90^{\circ} $\\
				$ CF $ là đường cao $ \Rightarrow BF \perp FC \Rightarrow \widehat{BFC}=90^2.$\\
				$ \Rightarrow  $ Tứ giác $ BCEF $ nội tiếp đường tròn đường kính $ BC $.\\ 
				$ \Rightarrow \widehat{BCF}=\widehat{BEF} $ (góc nội tiếp cùng chắn một cung).\\
				b.  Ta có tứ giác $ BCEF $ nội tiếp đường tròn tâm $ I $ đường kính $ BC $ (theo câu $ a $). \\			
				$ \Rightarrow \widehat{EIF}=2\widehat{ECF}=60^{\circ} $ (góc nội tiếp và góc ở tâm cùng chắn cung).\\
	}{	
		\hspace*{.2cm}
		\begin{tikzpicture}
			\tkzInit[xmin=-3, ymin=-6.5,xmax=4,ymax=0.5]
			\tkzClip
			\tkzDefPoint(0,0){A}
			\tkzDefShiftPoint[A](240:5cm){B}
			\tkzDefShiftPoint[A](-60:6cm){C}
			\tkzDefCircle[circum](A,B,C) \tkzGetPoint{O}
			\tkzDrawCircle(O,A)
			\tkzDefPointBy[projection = onto A--C](B) \tkzGetPoint{E}
			\tkzDefPointBy[projection = onto A--B](C) \tkzGetPoint{F}
			\tkzDefMidPoint(B,C) \tkzGetPoint{I}	
			\tkzDefMidPoint(E,F) \tkzGetPoint{K}		
			\tkzTangent[at=A](O) \tkzGetPoint{x}
			\tkzMarkRightAngle[fill=gray!20](A,E,B)
			\tkzMarkRightAngle[fill=gray!20](A,F,C)
			\tkzMarkRightAngle[fill=gray!20](O,A,x)
			\tkzDrawLine[add=1 and .5](A,x)			
			\tkzDrawSegments(A,B B,C C,A O,A I,K)
			\tkzDrawSegments[color=blue](B,E C,F)
			\tkzDrawSegments[color=red](I,E E,F F,I)		
			\tkzMarkAngle[size=.5](B,A,C)		
			\tkzLabelAngle[pos=.8](B,A,C){$60^{\circ}$}
			\tkzDrawPoints(A,B,C,O,E,F,K)
			\node (A) at ($(A) + (120:3mm)$){$\footnotesize A$};
			\node (B) at ($(B) + (200:3mm)$){$\footnotesize B$};
			\node (C) at ($(C) + (-10:3mm)$){$\footnotesize C$};
			\node (E) at ($(E) + (45:5mm)$){$\footnotesize E$};
			\node (F) at ($(F) + (120:5mm)$){$\footnotesize F$};
			\node (O) at ($(O) + (240:3mm)$){$\footnotesize O$};
			\node (I) at ($(I) + (-90:3mm)$){$\footnotesize I$};
			\node (A) at ($(A) + (5:13mm)$){$\footnotesize x'$};
			\node (x) at ($(x) + (120:3mm)$){$\footnotesize x$};
			\node (K) at ($(K) + (145:5mm)$){$\footnotesize K$};
		\end{tikzpicture}
	}
		
		Mặt khác ta có $ IE =IF =\dfrac{BC}{2} \Rightarrow $ Tam giác IEF đều. \\
		c. Kẻ tiếp tuyến $ xAx' $ của đường tròn $ (O) $ tại $ A. $\\
		$ \Rightarrow xx' \perp OA \quad (1)$\\
		Ta lại có: $ \heva{&\widehat{AFE}+\widehat{BFE} =180^{\circ} \text{ (kề bù)} \\& \widehat{ACB}+\widehat{BFE}=180^{\circ} \text{ (Tứ giác }  BFEC \text{ nội tiếp)} } \Rightarrow \widehat{ACB}=\widehat{AFE}.$\\
		Mà $ \widehat{BAx}=\widehat{ACB} $ (góc nội tiếp và góc tạo bởi tiếp tuyến cùng chắn cung $ AB $).\\
		$ \Rightarrow \widehat{BAx} = \widehat{AFE} \Rightarrow xx' \parallel EF. \quad (2)$ \\
		Từ $ (1) $ và $ (2) $ suy ra: $ OA \perp EF.  $\\
		Mặt khác $ IK \perp EF $ ($ \triangle ABC$ đều).\\
		$ \Rightarrow IK \parallel OA. $		
}	
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên An Giang, 2017] %[9H3K9]
	\immini{
		Trong một hình vành khăn với các bán kính đường tròn là 10R và 8R. Xếp các hình tròn bán kính R tiếp xúc với cả hai đường tròn này không chồng lấn nhau. Hỏi xếp được nhiều nhất bao nhiêu hình tròn như thế?
}{
		\hspace*{1cm}
		\begin{tikzpicture}[scale=.7]
			\tkzDefPoint(0,0){O}
			\tkzDefShiftPoint[O](70:3cm){A}
			\tkzDefShiftPoint[O](70:2cm){B}
			\tkzDefMidPoint(A,B) \tkzGetPoint{I}	
			\tkzInterCC(O,I)(I,A) \tkzGetPoints{M}{}
			\tkzInterCC(O,I)(M,I) \tkzGetPoints{N}{}		
			\tkzDrawCircle[fill=gray!20](O,A)
			\tkzDrawCircle[fill=white](O,B)
			\tkzDrawCircle[fill=yellow!30](I,A)	
			\tkzDrawCircle[fill=yellow!30](N,M)
			\tkzDrawPoints(O)
		\end{tikzpicture}
	}
	\loigiai{			
			\immini{
			Xét hình tròn tâm $ I $ tiếp xúc với hai đường tròn tâm $ O $ của hình vành khăn.\\
			Từ $ O $ kẻ hai tiếp tuyến $ OT $ và $ OT' $ tiếp xúc với đường tròn tâm $ I $.\\
			Ta có $ IT=R; OI = 9R $. \\ Tam giác $ OIT $ vuông tại $ T $\\
			$ \Rightarrow \sin IOT =\dfrac{IT}{OI}=\dfrac{R}{9R}=\dfrac{1}{9} \Rightarrow \widehat{IOT} \approx 6^{\circ}23'$\\
			Vậy $\widehat{TOT'} =2\widehat{IOT} \approx 12^{\circ}46' $\\
			Do đó hình tròn xếp được trong hình vành khăn là: $ 360^{\circ}:12^{\circ}46' \approx 28,1984 $\\
			Vậy có thể xếp nhiều nhất $ 28 $ hình tròn bán kính $ R $ tiếp xúc với hai đường tròn của hình vành khăn.
			
		}{
			\hspace*{1cm}
			\begin{tikzpicture}[scale=1]
			\tkzDefPoint(0,0){O}
			\tkzDefShiftPoint[O](70:3cm){A}
			\tkzDefShiftPoint[O](70:2cm){B}
			\tkzDefMidPoint(A,B) \tkzGetPoint{I}
			\tkzInterCC(I,A)(O,I) \tkzGetPoints{T}{T'}			
			\tkzDrawCircle[fill=gray!20](O,A)
			\tkzDrawCircle[fill=white](O,B)
			\tkzDrawCircle[fill=yellow!30](I,A)	
			\tkzDrawSegments(O,A O,T O,T' I,T)
			\tkzDrawPoints(O,T,T',I)
			\node (O) at ($(O) + (145:5mm)$){$\footnotesize O$};
			\node (T) at ($(T) + (10:3mm)$){$\footnotesize T$};
			\node (T') at ($(T') + (120:2mm)$){$\footnotesize T'$};
			\node (I) at ($(I) + (120:2mm)$){$\footnotesize I$};
			\end{tikzpicture}
		}
	}
\end{ex}
