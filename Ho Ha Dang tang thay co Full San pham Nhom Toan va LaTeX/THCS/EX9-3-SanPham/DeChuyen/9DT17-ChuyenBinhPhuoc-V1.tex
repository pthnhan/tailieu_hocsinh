\begin{ex}%[Đề thi vào 10 Chuyên, Sở giáo dục Bình Phước, 2017]%[9D1K8]%[9D5G1]
    Cho biểu thức: $P=\dfrac{\sqrt{x}}{\sqrt{x}+2}+\dfrac{-x+x\sqrt{x}+6}{x+\sqrt{x}-2}-\dfrac{\sqrt{x}+1}{\sqrt{x}-1}$, với $x\geq0$, $x\ne1$.
    \begin{enumerate}
        \item Rút gọn biểu thức $P$.
        \item Cho biểu thức $Q=\dfrac{(x+27).P}{(\sqrt{x}+3)(\sqrt{x}-2)}$, với $x\geq0$, $x\ne1$, $x\ne4$. Chứng minh $Q\geq6$.
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item Ta có:\\
        $\begin{aligned}
        P&=\dfrac{\sqrt{x}}{\sqrt{x}+2}+\dfrac{-x+x\sqrt{x}+6}{x+\sqrt{x}-2}-\dfrac{\sqrt{x}+1}{\sqrt{x}-1}\\
        &=\dfrac{\sqrt{x}(\sqrt{x}+1)-x+x\sqrt{x}+6-(\sqrt{x}+1)(\sqrt{x}+2)}{(\sqrt{x}-1)(\sqrt{x}+2)}\\
        &=\dfrac{x-\sqrt{x}-x+x\sqrt{x}+6-x-3\sqrt{x}-2}{(\sqrt{x}-1)(\sqrt{x}+2)}\\
        &=\dfrac{-x+x\sqrt{x}-4\sqrt{x}+4}{(\sqrt{x}-1)(\sqrt{x}+2)}\\
        &=\dfrac{(\sqrt{x}-1)(x-4)}{(\sqrt{x}-1)(\sqrt{x}+2)}\\
        &=\sqrt{x}-2.
        \end{aligned}$
        \item Với $x\geq0$, $x\ne1$, $x\ne4$, ta có:\\
        $\begin{aligned}
		Q&=\dfrac{(x+27).P}{(\sqrt{x}+3)(\sqrt{x}-2)}=\dfrac{x+27}{\sqrt{x}+3}=\dfrac{x-9+36}{\sqrt{x}+3}\\
		&=\sqrt{x}-3+\dfrac{36}{\sqrt{x}+3}=-6+(\sqrt{x}+3)+\dfrac{36}{\sqrt{x}+3}\geq-6+12=6
        \end{aligned}$
    \end{enumerate}
    }
\end{ex}

\begin{ex}%[Đề thi vào 10 Chuyên, Sở giáo dục Bình Phước, 2017]%[9D4G6]
	Cho phương trình: $x^2-2(m-1)x+m^2-3=0$ ($x$ là ẩn, $m$ là tham số). Tìm $m$ để phương trình có hai nghiệm $x_1$, $x_2$ sao cho $x_1^2+4x_1+2x_2-2mx_1=1$.
	\loigiai{Phương trình đã cho có hai nghiệm khi và chỉ khi $\Delta'\geq0$\\
		$\Leftrightarrow -2m+4\geq0\Leftrightarrow m\leq2$\quad(1).\\
	Theo định lí Vi-ét: $\heva{x_1+x_2&=2(m-1)\\x_1.x_2&=m^2-3}$\\
	Mà $x_1^2+4x_1+2x_2-2mx_1=1$\\
	 $\Leftrightarrow x_1(x_1-2m+2)+2(x_1+x_2)=1$\\
	 $\Leftrightarrow -x_1x_2+2(x_1+x_2)=1$\\
	 $\Leftrightarrow -m^2+3+4(m-1)=1$\\
	 $\Leftrightarrow m^2-4m+2=0$\\
	 $\Leftrightarrow \hoac{m&=2+\sqrt{2}\\m&=2-\sqrt{2}}$\quad(2)\\
	 Từ (1) và (2) suy ra $m=2-\sqrt{2}$.
	 }
\end{ex}

\begin{ex}%[Đề thi vào 10 Chuyên, Sở giáo dục Bình Phước, 2017]%[9D5G5]
	\hfill
	\begin{enumerate}
		\item Giải phương trình: $x+2\sqrt{7-x}=2\sqrt{x-1}+\sqrt{-x^2+8x-7}+1$.
		\item Giải hệ phương trình: $\heva{4\sqrt{x+1}-xy\sqrt{y^2+4}&=0\\
		\sqrt{x^2-xy^2+1}+3\sqrt{x-1}&=xy^2}$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Điều kiện $1\leq x\leq7$.\\
		Ta có: \\
		$\begin{aligned}
		&x+2\sqrt{7-x}=2\sqrt{x-1}+\sqrt{-x^2+8x-7}+1\\
		\Leftrightarrow&2(\sqrt{7-x}-\sqrt{x-1})+(x-1)+\sqrt{(x-1)(7-x)}=0\\
		\Leftrightarrow&2(\sqrt{7-x}-\sqrt{x-1})+\sqrt{x-1}(\sqrt{x-1}-\sqrt{7-x})=0\\
		\Leftrightarrow&(\sqrt{7-x}-\sqrt{x-1})(2-\sqrt{x-1})=0\\
		\Leftrightarrow&\hoac{&\sqrt{x-1}=2\\&\sqrt{x-1}=\sqrt{7-x}}\Leftrightarrow\hoac{x=5\\x=4}(\text{thỏa mãn điều kiện}).
		\end{aligned}$\\
		Vậy phương trình có hai nghiệm $x=4$, $x=5$.
		\item Điều kiện $\heva{&x\geq1\\&x^2-xy^2+1\geq0}$, kết hợp với phương trình $(1)$, ta có: $y>0$.\\
		Từ $(1)$, ta có:\\
		$4\sqrt{x+1}-xy\sqrt{y^2+4}=0\Leftrightarrow 4\sqrt{x+1}=xy\sqrt{y^2+4}$\\
		$\Leftrightarrow 16(x+1)=x^2y^2(y^2+4)\Leftrightarrow (y^4+4y^2)x^2-16x-16=0$.\\
		Giải phương trình theo ẩn $x$ ta được $x=\dfrac{4}{y^2}$ hoặc $x=\dfrac{-4}{y^2+4}<0$ (loại).\\
		Với $x=\dfrac{4}{y^2}\Leftrightarrow xy^2=4$ thế vào phương trình $(2)$, ta được: $\sqrt{x^2-3}+3\sqrt{x-1}=4$.\\
		Điều kiện $x\geq\sqrt{3}$, ta có:\\
		$\begin{aligned}
		&\sqrt{x^2-3}+3\sqrt{x-1}=4\\
		\Leftrightarrow&(\sqrt{x^2-3}-1)+3(\sqrt{x-1}-1)=0\\
		\Leftrightarrow&\dfrac{x^2-4}{\sqrt{x^2-3}+1}+\dfrac{3(x-2)}{\sqrt{x-1}+1}=0\\
		\Leftrightarrow&(x-2)\left(\dfrac{x+2}{\sqrt{x^2-3}+1}+\dfrac{3}{\sqrt{x-1}+1}\right)=0\\
		\Leftrightarrow&x-2=0 \left(\text{vì} \dfrac{x+2}{\sqrt{x^2-3}+1}+\dfrac{3}{\sqrt{x-1}+1}>0\right)\Leftrightarrow x=2.
		\end{aligned}$\\
		Với $x=2$ ta có $\heva{y^2=2\\y>0}\Leftrightarrow y=\sqrt{2}$. Kết hợp với điều kiện trên, hệ phương trình có nghiệm $(2,\sqrt{2})$.
\end{enumerate}}
\end{ex}

\begin{ex}%[Đề thi vào 10 Chuyên, Sở giáo dục Bình Phước, 2017]%[9D5G3]
	Chứng minh biểu thức $S=n^2(n+2)^2+(n+1)(n^3-5n+1)-2n-1$ chia hết cho $120$, với $n$ là số nguyên.
	\loigiai{Ta có:\\
	$\begin{aligned}
	S&=n(n^4+5n^3+5n^2-5n-6)\\
	&=n[(n^2-1)(n^2+6)+5n(n^2-1)]\\
	&=n(n^2-1)(n^2+5n+6)\\
	&=n(n-1)(n+1)(n+2)(n+3)\\
	&=(n-1)n(n+1)(n+2)(n+3)
	\end{aligned}$}\\
Ta có $S$ là tích của $5$ số tự nhiên liên tiếp chia cho $5!$ nên chia hết cho $120$.
\end{ex}

\begin{ex}%[Đề thi vào 10 Chuyên, Sở giáo dục Bình Phước, 2017]%[9D5G1]
	\hfill
	\begin{enumerate}
		\item Cho ba số $a$, $b$, $c$ thỏa mãn $a+b+c=0$ và $|a|\leq1$, $|b|\leq1$, $|c|\leq1$. Chứng minh rằng $a^4+b^6+c^8\leq2$.
		\item Tìm giá trị nhỏ nhất của biểu thức $T=\dfrac{(x^3+y^3)-(x^2+y^2)}{(x-1)(y-1)}$ với $x$, $y$ là các số thực lớn hơn $1$.
	\end{enumerate}
\loigiai{\hfill
\begin{enumerate}
	\item Từ giả thuyết $|a|\leq1$, $|b|\leq1$, $|c|\leq1$, ta có $a^4\leq a^2$, $b^6\leq b^2$, $c^8\leq c^2$.\\
	Từ đó $a^4+b^6+c^8\leq a^2+b^2+c^2$.\\
	Lại có: $(a-1)(b+1)(c+1)-(a-1)(b-1)(c-1)\geq0$\\ nên $(a+1)(b+1)(c+1)-(a-1)(b-1)(c-1)\geq0$\\
	$\Leftrightarrow 2ab+2bc+2ca+2\geq0\Leftrightarrow -2(ab+bc+ca)\leq2$.\\
	Hơn nữa $a+b+c=0\Leftrightarrow a^2+b^2+c^2=-(ab+bc+ca)\leq2$. Vậy $a^4+b^6+c^8\leq2$.
	\item Ta có: $T=\dfrac{(x^3+y^3)-(x^2+y^2)}{(x-1)(y-1)}=\dfrac{x^2(x-1)+y^2(y-1)}{(x-1)(y-1)}=\dfrac{x^2}{y-1}+\dfrac{y^2}{x-1}$.\\
	Do $x>1$, $y>1$ nên $x-1>0$, $y-1>0$.\\
	Áp dụng bất đẳng thức Cô-si cho hai số dương $\dfrac{x^2}{y-1}$, $\dfrac{y^2}{x-1}$, ta có:\\
	$(x-1)+1\geq2\sqrt{x-1}\Leftrightarrow(\sqrt{x-1}-1)^2\geq0\Leftrightarrow x-2\sqrt{x-1}\geq0\Leftrightarrow \dfrac{x}{\sqrt{x-1}}\geq2$\\
	$(y-1)+1\geq2\sqrt{y-1}\Leftrightarrow(\sqrt{y-1}-1)^2\geq0\Leftrightarrow y-2\sqrt{y-1}\geq0\Leftrightarrow \dfrac{y}{\sqrt{y-1}}\geq2$\\
	Do đó: $T=\dfrac{x^2}{y-1}+\dfrac{y^2}{x-1}\geq\dfrac{2xy}{\sqrt{x-1}.\sqrt{y-1}}\geq8$.\\
	Dấu "$=$" xảy ra khi $\heva{&\dfrac{x^2}{y-1}=\dfrac{y^2}{x-1}\\
	&x-1=1\\
&y-1=1}\Leftrightarrow\heva{x=2\\y=2}$ (thỏa mãn điều kiện).\\
Vậy giá trị nhỏ nhất của biểu thức $T=8$ khi $x=y=2$.
\end{enumerate}}
\end{ex}
