\begin{bt}%[Đề HK1 T11, Thạch Thành 1, Thanh Hóa 2017]%[1D1K3]
	Giải các phương trình
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\sin \left(2x+\dfrac{\pi}{6}\right)=\dfrac{1}{2}$
			\item $\sqrt 3 \cdot \sin x+\cos x=-2$
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item $\sin \left(2x+\dfrac{\pi}{6}\right)=\dfrac{1}{2} \Leftrightarrow \sin \left(2x+\dfrac{\pi}{6}\right)=\sin \dfrac{\pi}{6} \Leftrightarrow \hoac{&2x+\dfrac{\pi}{6}=\dfrac{\pi}{6}+k 2\pi\\& 2x+\dfrac{\pi}{6}=\pi-\dfrac{\pi}{6}+k 2\pi }\Leftrightarrow \hoac{&x=k\pi\\& x=\dfrac{\pi}{3}+k\pi}$\\
			Vậy phương trình có hai họ nghiệm $x=k\pi;x=\dfrac{\pi}{3}+k\pi$ với $k \in \mathbb Z$.
			\item $\sqrt 3 \cdot \sin x+\cos x=-2 \Leftrightarrow \sin x \cdot  \dfrac{\sqrt 3}{2}+\dfrac{1}{2} \cdot \cos x=-1 \Leftrightarrow \sin x \cdot \cos \dfrac{\pi}{6}+\sin \dfrac{\pi}{6} \cdot \cos x=-1 \\ \Leftrightarrow \sin\left(x+\dfrac{\pi}{6}\right)=-1 \Leftrightarrow x+\dfrac{\pi}{6}=-\dfrac{\pi}{2}+k 2\pi \Leftrightarrow x=-\dfrac{2\pi}{3}+k2\pi, k\in \mathbb Z$.
		\end{enumerate}
	}
\end{bt}
