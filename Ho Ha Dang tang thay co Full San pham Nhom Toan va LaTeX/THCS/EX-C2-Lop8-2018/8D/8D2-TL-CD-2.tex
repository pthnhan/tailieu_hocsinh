\section{Tính chia hết của số nguyên}
Tính chia hết đối với số nguyên đã được trình bày ở cuốn \textit{Nâng cao và phát triển Toán 6, Nâng cao và phát triển Toán 7}. Nhờ sử dụng các hằng đẳng thức đáng nhớ và phân tích đa thức thành nhân tử ở lớp 8 ta có khả năng giải quyết nhiều bài toán về chia hết phức tạp hơn ở các lớp dưới.

\subsection{Chứng minh quan hệ chia hết}
Gọi $A(n)$ là một biểu thức phụ thuộc vào $n$ ($n\in \mathbb{N}$ hoặc $n\in \mathbb{Z}$).
\begin{note}
Để chứng minh biểu thức $A(n)$ chia hết cho một số $m$, ta thường phân tích biểu thức $A(n)$ thành thừa số, trong đó có một thừa số là $m$. Nếu $m$ là hợp số, ta phân tích nó thành một tích các thừa số đôi một nguyên tố cùng nhau, rồi chứng minh $A(n)$ chia hết cho tất cả các số đó. Nên lưu ý nhận xét: Trong $k$ số nguyên liên tiếp, bao giờ cũng tồn tại một bội số của $k$.
\end{note}

\begin{vd}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K9]
Chứng minh rằng $A=n^3\left(n^2-7\right)^2-36n$ chia hết cho $5040$ với mọi số tự nhiên $n$.
	\loigiai{
Phân tích ra thừa số nguyên tố: $5040=2^4\cdot 3^2\cdot 5\cdot 7$.\\
Phân tích 
\begin{eqnarray*}
	A&=&n\left[n^2\left(n^2-7\right)^2-36\right]=n\left[\left(n^3-7n\right)^2-6^2\right]\\
	&=&n\left(n^3-7n-6\right)\left(n^3-7n+6\right).
\end{eqnarray*}
Ta lại có
\begin{eqnarray*}
	n^3-7n-6&=&(n+1)(n+2)(n-3),\\
	n^3-7n+6&=&(n-1)(n-2)(n+3).
\end{eqnarray*}
Do đó $A=(n-3)(n-2)(n-1)n(n+1)(n+2)(n+3)$.\\
Đây là tích của bảy số nguyên liên tiếp. Trong bảy số nguyên liên tiếp:
\begin{itemize}
	\item Tồn tại một bội số của $5$ (nên $A$ chia hết cho $5$);
	\item Tồn tại một bội số của $7$ (nên $A$ chia hết cho $7$);
	\item Tồn tại một bội số của $3$ (nên $A$ chia hết cho $9$);
	\item Tồn tại một bội số của $2$, trong đó có một bội số của $4$ (nên $A$ chia hết cho $16$).
\end{itemize}
$A$ chia hết cho các số $5, 7, 9, 16$ đôi một nguyên tố cùng nhau nên $A$ chia hết cho $5\cdot 7\cdot 9\cdot 16=5040$.
\begin{note}
	Khi chứng minh $A(n)$ chia hết cho $m$, ta có thể xét mọi trường hợp về số dư khi chia $n$ cho $m$.
\end{note}
	}
\end{vd}

\begin{vd}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K9]
Chứng minh rằng với mọi số nguyên $a$ thì
\begin{listEX}[2]
	\item $a^2-a$ chia hết cho $2$;
	\item $a^3-a$ chia hết cho $3$;
	\item $a^5-a$ chia hết cho $5$;
	\item $a^7-a$ chia hết cho $7$.
\end{listEX}
	\loigiai{
\begin{enumerate}
	\item $a^2-a=a(a-1)$, chia hết cho $2$.
	\item $a^3-a=a\left(a^2-1\right)=(a-1)a(a+1)$, tích này chia hết cho $3$ vì tồn tại một bội của $3$.
	\item $A=a^5-a=a\left(a^2+1\right)\left(a^2-1\right)$.\\
	Nếu $a=5k$ ($k\in \mathbb{Z}$) thì $a$ chia hết cho $5$.\\
	Nếu $a=5k\pm 1$ ($k\in \mathbb{Z}$) thì $a^2-1$ chia hết cho $5$.\\
	Nếu $a=5k\pm 2$ ($k\in \mathbb{Z}$) thì $a^2+1$ chia hết cho $5$.\\
	Trường hợp nào cũng có một thừa số của $A$ chia hết cho $5$.\\
	Cách 2: Phân tích $a^5-a$ thành một tổng của hai số hạng chia hết cho $5$: Một số hạng là tích của năm số nguyên liên tiếp, một số hạng chứa thừa số $5$.
	\begin{eqnarray*}
		a^5-a&=&a\left(a^2-1\right)\left(a^2+1\right)\\
		&=&a\left(a^2-1\right)\left(a^2-4+5\right)\\
		&=&a\left(a^2-1\right)\left(a^2-4\right)+5a\left(a^2-1\right)\\
		&=&(a-2)(a-1)a(a+1)(a+2)+5a\left(a^2-1\right).
	\end{eqnarray*}
Số hạng thứ nhất là tích của năm số nguyên liên tiếp nên chia hết cho $5$, số hạng thứ hai cũng chia hết cho $5$. Do đó $a^5-a$ chia hết cho $5$.\\
Cách 3: Giải tương tự như cách 2: Xét hiệu giữa $a^5-a$ và tích năm số nguyên liên tiếp $(a-2)(a-1)a(a+1)(a+2)$, được $5a\left(a^2-1\right)$. Do đó $a^5-a$ chia hết cho $5$.
	\item $A=a^7-a=a\left(a^3+1\right)\left(a^3-1\right)$.\\
Nếu $a=7k$ ($k\in \mathbb{Z}$) thì $a$ chia hết cho $7$.\\
Nếu $a=7k+ 1$ ($k\in \mathbb{Z}$) thì $a^3-1$ chia hết cho $7$.\\
Nếu $a=7k- 1$ ($k\in \mathbb{Z}$) thì $a^3+1$ chia hết cho $7$.\\
Nếu $a=7k+2$ ($k\in \mathbb{Z}$) thì $a^3-1$ chia hết cho $7$.\\
Nếu $a=7k-2$ ($k\in \mathbb{Z}$) thì $a^3+1$ chia hết cho $7$.\\
Nếu $a=7k+3$ ($k\in \mathbb{Z}$) thì $a^3+1$ chia hết cho $7$.\\
Nếu $a=7k-3$ ($k\in \mathbb{Z}$) thì $a^3-1$ chia hết cho $7$.\\
Trường hợp nào cũng có một thừa số của $A$ chia hết cho $7$.\\
Vậy $a^7-a$ chia hết cho $7$.
\end{enumerate}	
	}
\end{vd}

\begin{vd}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K3]
Chứng minh rằng một số chính phương chia cho $3$ chỉ có thể có số dư bằng $0$ hoặc $1$.
	\loigiai{
Gọi $A$ là số chính phương $A=n^2$ ($n\in \mathbb{N}$). Xét các trường hợp
\begin{itemize}
	\item $n=3k$ ($k\in \mathbb{N}$)$\Rightarrow A=9k^2$, chia hết cho $3$.
	\item $n=3k\pm 1$ ($k\in \mathbb{N}$)$\Rightarrow A=9k^2\pm 6k+1$, chia cho $3$ dư $1$.
\end{itemize}
Vậy số chính phương chia cho $3$ chỉ có thể có số dư bằng $0$ hoặc $1$.
	}
\end{vd}

\begin{vd}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K3]
Chứng minh rằng một số chính phương chia cho $4$ chỉ có thể có số dư bằng $0$ hoặc $1$.
	\loigiai{
Gọi $A$ là số chính phương $A=n^2$ ($n\in \mathbb{N}$). Xét các trường hợp
\begin{itemize}
	\item $n=2k$ ($k\in \mathbb{N}$)$\Rightarrow A=4k^2$, chia hết cho $4$.
	\item $n=2k+ 1$ ($k\in \mathbb{N}$)$\Rightarrow A=4k^2+4k+1=4k(k+1)+1$, chia cho $4$ dư $1$.
\end{itemize}
Vậy số chính phương chia cho $4$ chỉ có thể có số dư bằng $0$ hoặc $1$.
\begin{note}
	Từ bài toán trên ta thấy
	\begin{itemize}
		\item Số chính phương chẵn thì chia hết cho $4$.
		\item Số chính phương lẻ thì chia cho $4$ dư $1$ (hơn nữa, chia cho $8$ cũng dư $1$).
	\end{itemize}
\end{note}
	}
\end{vd}

\begin{vd}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K9]
Các số sau có là số chính phương không?
\begin{eqnarray*}
	M&=&1992^2+1993^2+1994^2;\\
	N&=&1992^2+1993^2+1994^2+1995^2;\\
	P&=&1+9^{100}+94^{100}+1994^{100}\\
\end{eqnarray*}
	\loigiai{
Các số $1993^2, 1994^2$ là các số chính phương không chia hết cho $3$ nên chia cho $3$ dư $1$, còn $1992^2$ chia hết cho $3$. Số $M$ là số chia cho $3$ dư $2$, không là số chính phương.\\
Các số $1992^2, 1994^2$ là số chính phương chẵn nên chia hết cho $4$. Các số $1993^2, 1995^2$ là các số chính phương lẻ nên chia cho $4$ dư $1$. Số $N$ là số chia cho $4$ dư $2$, không là số chính phương.\\
Các số $94^{100}, 1994^{100}$ là số chính phương chẵn nên chia hết cho $4$. Còn $9^{100}$ là số chính phương lẻ nên chia cho $4$ dư $1$. Số $P$ là số chia cho $4$ dư $2$, không là số chính phương.
	}
\end{vd}

Cách 2: $A=16^n-1=(17-1)^n-1=BS 17+(-1)^n-1$ (theo công thức Niu-tơn).\\
Nếu $n$ chẵn thì $A=BS 17+1-1=BS 17$.\\
Nếu $n$ lẻ thì $A=BS 17-1-1$ không chia hết cho $17$.

$a_1=2004$\\
$a_2=20042004$\\

\subsection{Tìm số dư}
\begin{vd}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1G3]
Tìm số dư khi chia $2^{100}$:
\begin{listEX}[3]
	\item Cho $9$;
	\item Cho $25$;
	\item Cho $125$.
\end{listEX}
	\loigiai{
\begin{enumerate}
	\item Lũy thừa của $2$ sát với bội số của $9$ là $2^3=8=9-1$.\\
	Ta có $2^{100}=2\left(2^3\right)^{33}=2\left(9-1\right)^{33}=2\left(BS \;9-1\right)=BS \;9-2=BS \;9+7$.\\
	Số dư khi chia $2^{100}$ cho $9$ là $7$.
	\item Lũy thừa của $2$ sát với bội số của $25$ là $2^{10}=1024=BS \;25-1$.\\
	Ta có $2^{100}=\left(2^{10}\right)^{10}=\left(BS \;25-1\right)^{10}=BS \;25+1$.\\
	Số dư khi chia $2^{100}$ cho $25$ là $1$.
	\item Dùng công thức Niu-tơn:
	\[2^{100}=(5-1)^{50}=5^{50}-50\cdot 5^{49}+\cdots+\dfrac{50\cdot 49}{2}\cdot 5^2-50\cdot 5+1.\]
	Không kể phần hệ số của khai triển Niu-tơn thì $48$ số hạng đầu đã chứa lũy thừa của $5$ với số mũ lớn hơn hoặc bằng $3$ nên chia hết cho $125$. Hai số hạng tiếp theo cũng chia hết cho $125$, số hạng cuối cùng là $1$. Vậy $2^{100}=BS\;125+1$.
\end{enumerate}
\begin{note}
	Tổng quát hơn, ta chứng minh được rằng nếu một số tự nhiên $n$ không chia hết cho $5$ thì chia $n^{100}$ cho $125$ ta được số dư là $1$.\\
	Thật vậy, $n$ có dạng $5k\pm 1$ hoặc $5k\pm 2$. Ta có
	\[(5k\pm 1)^{100}=(5k)^{100}\pm\cdots +\dfrac{100\cdot 99}{2}(5k)^2\pm 100\cdot 5k+1=BS\;125+1.\]
	\[(5k\pm 2)^{100}=(5k)^{100}\pm\cdots +\dfrac{100\cdot 99}{2}(5k)^2\cdot 2^{98}\pm 100\cdot 5k\cdot 2^{99}+2^{100}=BS\;125+2^{100}.\]
	Ta lại có $2^{100}=BS\;125+1$ (câu c). Do đó $(5k\pm 2)^{100}=BS\;125+1$.
\end{note}
	}
\end{vd}

\begin{vd}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1G3]
Tìm ba chữ số tận cùng của $2^{100}$ khi viết trong hệ thập phân.
	\loigiai{
Tìm ba chữ số tận cùng của $2^{100}$ là tìm số dư khi chia $2^{100}$ cho $1000$. Trước hết tìm số dư khi chia $2^{100}$ cho $125$. Theo ví dụ trên ta có $2^{100}=BS\;125+1$, mà $2^{100}$ là số chẵn, nên ba chữ số tận cùng của nó chỉ có thể là $126, 376, 626$ hoặc $876$.\\
Hiển nhiên $2^{100}$ chia hết cho $8$ nên ba chữ số tận cùng của nó phải chia hết cho $8$. Trong bốn số trên chỉ có $376$ thỏa mãn điều kiện này.\\
Vậy ba chữ số tận cùng của $2^{100}$ là $376$.
\begin{note}
	Bạn đọc tự chứng minh rằng nếu $n$ là số chẵn không chia hết cho $5$ thì ba chữ số tận cùng của $n^{100}$ là $376$.
\end{note}
	}
\end{vd}

\begin{vd}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1G3]
Tìm bốn chữ số tận cùng của $5^{1994}$ khi viết trong hệ thập phân.
	\loigiai{
$5^4=625$. Ta thấy số tận cùng bằng $0$$625$ nâng lên lũy thừa nguyên dương bất kì vẫn tận cùng bằng $0$$625$ (chỉ cần kiểm tra: $\ldots$ $0$$625\times \ldots 0$$625=\ldots 0625$).\\
Do đó
\[5^{1994}=5^{4k+2}=25\left(5^4\right)^k=25(0625)^k=25(\ldots0625)=\ldots 5625.\]
Cách khác: Tìm số dư khi chia $5^{1994}$ cho $10000=2^4\cdot 5^4$.\\
\textit{Nhận xét:} $5^{4k}-1$ chia hết cho $5^4-1=\left(5^2+1\right)\left(5^2-1\right)$ nên chia hết cho $16$.\\
Ta có $5^{1994}=5^6\left(5^{1988}-1\right)+5^6$.\\
Do $5^6$ chia hết cho $5^4$, còn $5^{1988}-1$ chia heest cho $16$ (theo nhận xét trên) nên $5^6\left(5^{1988}-1\right)$ chia hết cho $10000$. Tính $5^6$, ta được $15625$. Vậy bốn chữ số tận cùng của $5^{1994}$ là $5625$.
\begin{note}
	Nếu viết $5^{1994}=5^2\left(5^{1992}-1\right)+5^2$ thì ta có $5^{1992}-1$ chia hết cho $16$, nhưng $5^2$ không chia hết cho $5^4$.\\
	Như thế trong bài toán này, ta cần viết $5^{1994}$ dưới dạng $5^n\left(5^{1994-n}-1\right)+5^n$ sao cho $n\ge 4$ và $1994-n$ chia hết cho $4$.
\end{note}
	}
\end{vd}

\subsection{Tìm điều kiện để chia hết}
\begin{vd}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K12]
Tìm số nguyên $x$ để giá trị của biểu thức $A$ chia hết cho giá trị của biểu thức $B$:
\[A=x^3+2x^2-3x+2,\qquad B=x^2-x.\]
	\loigiai{
Đặt tính chia:
\polylongdiv[style=D]{x^3+2x^2-3x+2}{x^2-x}\\
Muốn chia hết, ta phải có $2$ chia hết cho $n(n-1)$, do đó $2$ chia hết cho $n$.\\
Ta có
\begin{center}
	\begin{tabular}{ |c|c|c|c|c| } 
		\hline
		$n$ & $1$ & $-1$ &$2$ &$-2$ \\
		\hline
		$n-1$ & $0$ & $-2$ &$1$ &$-3$ \\
		\hline
		$n(n-1)$ & $0$ & $2$ &$2$ &$6$ \\
		\hline
		 &loại& & & loại\\
		\hline
	\end{tabular}
\end{center}
Vậy $n=-1; n=2$.
\begin{note}Chú ý:
\begin{enumerate}
	\item Không thể nói đa thức $A$ chia hết cho đa thức $B$. Ở đây chỉ tồn tại những giá trị nguyên của $n$ để giá trị của biểu thức $A$ chia hết cho giá trị của biểu thức $B$.
	\item Có thể thay việc đặt phép chia bằng cách biến đổi:
	\[x^3+2x^2-3x+2=x\left(x^2-x\right)+3(x^2-x)+2\]
\end{enumerate}	
\end{note}
	}
\end{vd}

\begin{vd}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K3]
Tìm số nguyên dương $n$ để $n^5+1$ chia hết cho $n^3+1$.
	\loigiai{
Biến đổi 
\begin{eqnarray*}
n^5+1\;\vdots\; n^3+1 &\Leftrightarrow& n^2\left(n^3+1\right)-\left(n^2-1\right) \;\vdots\;n^3+1\\
&\Leftrightarrow& (n+1)(n-1)\;\vdots\; (n+1)\left(n^2-n+1\right)\\
&\Leftrightarrow& n-1\;\vdots\;n^2-n+1 (\text{vì}\;n+1\ne 0).
\end{eqnarray*}
Nếu $n=1$ thì ta được $0$ chia hết cho $1$.\\
Nếu $n>1$ thì $n-1<n(n-1)+1=n^2-n+1$, do đó $n-1$ không thể chia hết cho $n^2-n+1$.\\
Vậy giá trị duy nhất của $n$ tìm được là $1$.
	}

\end{vd}

\begin{vd}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K3]
	Tìm số nguyên $n$ để $n^5+1$ chia hết cho $n^3+1$.
	\loigiai{
		Biến đổi 
		\begin{eqnarray*}
			n^5+1\;\vdots\; n^3+1 &\Leftrightarrow& n^2\left(n^3+1\right)-\left(n^2-1\right) \;\vdots\;n^3+1\\
			&\Leftrightarrow& (n+1)(n-1)\;\vdots\; (n+1)\left(n^2-n+1\right)\\
			&\Leftrightarrow& n-1\;\vdots\;n^2-n+1.
		\end{eqnarray*}
		$n-1\;\vdots\;n^2-n+1 \Rightarrow n(n-1)\;\vdots\;n^2-n+1 \Rightarrow n^2-n\;\vdots\;n^2-n+1$\\
		$\Rightarrow (n^2-n+1)-1 \;\vdots\;n^2-n+1 \Rightarrow 1\;\vdots\;n^2-n+1$.\\
		Có hai trường hợp:
		\begin{itemize}
			\item $n^2-n+1=1\Leftrightarrow n(n-1)=0 \Leftrightarrow n=0; n=1$. Các giá trị thỏa mãn đề bài.
			\item $n^2-n+1=-1 \Leftrightarrow n^2-n+2=0$, vô nghiệm.
		\end{itemize}
		Vậy $n=0; n=1$ là hai số phải tìm.
		\begin{note}
			Từ $n-1\;\vdots\;n^2-n+1$ suy ra $n(n-1) \;\vdots\;n^2-n+1$ là phép kéo theo chứ không là phép biến đổi tương đương. Do đó sau khi tìm được $n=0, n=1$, ta phải thử lại.
		\end{note}
	}
	
\end{vd}

\begin{vd}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1G5]
Tìm số tự nhiên $n$ sao cho $2^n-1$ chia hết cho $7$.
	\loigiai{
Nếu $n=3k$ ($k\in \mathbb{N}$) thì $2^n-1=2^{3k}-1=8^k-1$ chia hết cho $7$.\\
Nếu $n=3k+1$ ($k\in \mathbb{N}$) thì $2^n-1=2^{3k+1}-1=2\left(2^{3k}-1\right)+1=BS\;7+1$.\\
Nếu $n=3k+2$ ($k\in \mathbb{N}$) thì $2^n-1=2^{3k+2}-1=4\left(2^{3k}-1\right)+3=BS\;7+3$.\\
Vậy $2^n-1$ chia hết cho $7$ $\Leftrightarrow n=3k$ ($k\in \mathbb{N}$).
	}
\end{vd}


\subsection{Bài tập}
\begin{bt}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K9]
	Chứng minh rằng với mọi số nguyên $n$ ta có:
	\begin{enumerate}
		\item $n^3+3n^2+2n$ chia hết cho $6$.
		\item ${\left( n^2 + n - 1 \right)}^2-1$ chia hết cho $24$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Biến đổi $n^3+3n^2+2n=n(n+1)(n+2)$, đây là tích của ba số nguyên liên tiếp.\\
			Do đó $n^3+3n^2+2n$ chia hết cho $6$ với mọi nguyên $n$.\\
			Suy ra điều phải chứng minh.
			\item Biến đổi: ${\left( n^2 + n - 1 \right)}^2-1 = (n-1)n(n+1)(n+2)$.\\
			Đây là tích của $4$ số nguyên liên tiếp, do đó: ${\left( n^2 + n - 1 \right)}^2-1$ chia hết cho $24$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K9]
	Chứng minh rằng:
	\begin{enumerate}
		\item $n^3+6n^2+8n$ chia hết cho $48$ với mọi số chẵn $n$.
		\item $n^4-10n^2+9$ chia hết cho $384$ với mọi số lẻ $n$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Biến đổi $n^3+6n^2+8n=n(n+2)(n+4)$.\\
			Vì $n$ là số chẵn nên ta thay $n=2k$ với $k \in \mathbb{Z}$, ta có:\\
			$n^3+6n^2+8n=2k(2k+2)(2k+4)=8k(k+1)(k+2)$ chia hết cho $48$.\\
			Suy ra điều phải chứng minh.
			\item Đặt $A=n^4-10n^2+9$.\\
			$\Rightarrow A = n^4-10n^2+9=(n^2-1)(n^2-9)=(n-1)(n+1)(n-3)(n+3)$.\\
			Vì $n$ là số lẻ suy ra $n=2k+1$ với $k \in \mathbb{Z}$, thay vào biểu thức ta có:
			\begin{eqnarray*}
				& A & = (n-1)(n+1)(n-3)(n+3) = \\
				& & = (2k+1-1)(2k+1+1)(2k+1-3)(2k+1+3)\\
				& & = (2k-2)2k(2k+2)(2k+4) \\
				& & = 2(k-1)\cdot 2k\cdot 2(k+1) \cdot 2(k+2)\\
				& & = 8(k-1)k(k+1)(k+2)
			\end{eqnarray*}
			Vì $(k-1)k(k+1)(k+2)$ là tích của $4$ số nguyên liên tiếp nên $(k-1)k(k+1)(k+2)$ chia hết cho $24$\\
			Suy ra $A$ chia hết cho $384$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K9]
	Chứng minh rằng $n^6+n^4-2n^2$ chia hết cho $72$ với mọi số nguyên $n$.
	\loigiai{
		Đặt $A = n^6+n^4-2n^2$, ta có:
		\begin{eqnarray*}
			& A & = n^6+n^4-2n^2 \\
			& & = n^2(n^4+n^2-2)\\
			& & = n^2(n^2-1)(n^2+2) \\
			& & = n^2(n-1)(n+1)(n^2-4)+6n^2(n-1)(n+1)\\
			& & = n\cdot n \cdot (n-1)(n+1)(n-2)(n+2) +6n\cdot n(n-1)(n+1)\\
			& & = (n-2)(n-1)n\cdot n(n+1)(n+2) + 6n\cdot n(n-1)(n+1)
		\end{eqnarray*}
		$(n-2)(n-1)n$ là tích của ba số tự nhiên liên tiếp suy ra $(n-2)(n-1)n$ chia hết cho $3$.\\
		$n(n+1)(n+2)$ là tích của ba số tự nhiên liên tiếp suy ra $n(n+1)(n+2)$ chia hết cho $3$.\\
		$\Rightarrow (n-2)(n-1)n\cdot n(n+1)(n+2)$ chia hết cho $9$.\\
		Mà trong 5 số tự nhiên liên tiếp có ít nhất 1 số chia hết cho $2$ và 1 số chia hết cho $4$ suy ra: $(n-2)(n-1)n\cdot n(n+1)(n+2)$ chia hết cho $8$.\\
		Suy ra $(n-2)(n-1)n\cdot n(n+1)(n+2)$ chia hết cho $72$.\\
		Dễ dàng chứng minh được $6n\cdot n(n-1)(n+1)$ chia hết cho $72$.\\
		Suy ra $A$ chia hết cho $72$.
	}
\end{bt}

\begin{bt}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K9]
	Chứng minh rằng $3^{2n}-9$ chia hết cho $72$ với mọi số nguyên dương $n$.
	\loigiai{
		Biến đổi $B=3^{2n}-9=9^n-9$, nên $B \; \vdots \; 9$.\\
		Đề chứng minh $B \; \vdots \; 8$ ta viết $B$ dưới dạng:
		\begin{center}
			$ B = (3^n)^2-1-8=(3^n-1)(3^n+1)-8$
		\end{center}
		Vì $3^n-1$ và $3^n+1$ là $2$ số chẵn liên tiếp nên $(3^n-1)(3^n+1)$ chia hết cho $8$.\\
		Suy ra: $B$ chia hết cho $8$.\\
		Vậy $3^{2n}-9$ chia hết cho $72$ với mọi số nguyên dương $n$.
	}
\end{bt}

\begin{bt}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K9]
	Chứng minh rằng với mọi số tự nhiên $a$ và $n$ thì:
	\begin{enumerate}
		\item $7^n$ và $7^{n+4}$ có hai chữ số tận cùng như nhau.
		\item $a$ và $a^5$ có chữ số tận cùng như nhau.
		\item $a^n$ và $a^{n+4}$ có chữ số tận cùng như nhau $(n\ge 1)$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Xét hiệu $A = 7^{n+4} - 7^n$ ta có:
			\begin{eqnarray*}
				& A & = 7^{n+4} - 7^n \\
				& & = 7^n(7^4-1)\\
				& & = 7^n \cdot 2400 \\
				& \Rightarrow & A \; \vdots \; 100
			\end{eqnarray*}
			Vậy $2$ số $7^n$ và $7^{n+4}$ có hai chữ số tận cùng như nhau.
			\item Xét hiệu $B = a^5 - a = a(a^2+1)(a^2-1)$.\\
			Dễ dàng chứng minh $B \; \vdots \; 2$ và $B \; \vdots \; 5$ suy ra $B \; \vdots \; 10 $.\\
			Vậy $a$ và $a^5$ có chữ số tận cùng như nhau.
			\item Xét hiệu $C = a^{n+4} - a^n = a^n(a^2+1)(a^2-1)$.\\
			Dễ dàng chứng minh $C \; \vdots \; 2$ và $C \; \vdots \; 5$ suy ra $C \; \vdots \; 10 $.\\
			Vậy $a^n$ và $a^{n+4}$ có chữ số tận cùng như nhau.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K9]
	Tìm điều kiện của số tự nhiên $a$ để $a^2+3a+2$ chia hết cho $6$.
	\loigiai{
		$a^2+3a+2=(a+1)(a+2)$ luôn chia hết cho $2$.\\
		$a^2+3a+2 \; \vdots \; 3 \Leftrightarrow a^2$ chia $3$ dư $1$ $\Leftrightarrow a$ không chia hết cho $3$.\\
		Vậy nếu $a$ không chia hết cho $3$ thì $a^2+3a+2$ chia hết cho $6$.
	}
\end{bt}

\begin{bt}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K5]
	\begin{enumerate}
		\item Cho $a$ là số nguyên tố lớn hơn $3$. Chứng minh rằng $a^2-1$ chia hết cho $24$.
		\item Chứng minh rằng nếu $a$ và $b$ là các số nguyên tố lớn hơn $3$ thì $a^2-b^2$ chia hết cho $24$.
		\item Tìm điều kiện của số tự nhiên $a$ để $a^4-1$ chia hết cho $240$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $a^2$ là số chính phương lẻ nên chia cho $8$ dư $1$, $a^2$ là số chính phương không chia hết cho $3$ nên chia cho $3$ dư $1$.\\
			Suy ra $a^2-1$ chia hết cho $8$, chia hết cho $3$, do đó chia hết cho $24$.
			\item Áp dụng ý a) ta có nếu $a$ và $b$ là các số nguyên tố lớn hơn $3$ thì:\\
			$a^2-1$ chia hết cho $3$ và $b^2-1$ chia hết cho $3$.\\
			$\Leftrightarrow (a^2-1)-(b^2-1) \; \vdots \; 3 \Leftrightarrow a^2-b^2 \; \vdots \; 3$.\\
			Suy ra điều phải chứng minh.
			\item Ta có: $240=2^4\cdot 3 \cdot5 $.\\
			Vì $A \; \vdots \; 240 \Rightarrow a^4 - 1 \; \vdots \; 2 $ và $ a^4 - 1 \; \vdots \; 3 $; $ a^4 - 1 \; \vdots \; 5 $.\\
			Dễ thấy $a^4 - 1 \; \vdots \; 2 \Leftrightarrow a $ là số lẻ.\\
			Dễ thấy $a^4 - 1 \; \vdots \; 3 \Leftrightarrow a $ không chia hết cho $3$.\\
			Dễ thấy $a^4 - 1 \; \vdots \; 5 \Leftrightarrow a $ không chia hết cho $5$.\\
			Ta sẽ chứng minh nếu $a$ không chia hết cho $2$, $3$ và $5$ thì $a^4-1$ chia hết cho $240$.\\
			Thật vậy:
			\begin{eqnarray*}
				& A & = a^4 - 1 = (a^2-1)(a^2+1)\\
				& & = (a-1)(a+1)(a^2+1)
			\end{eqnarray*}
			Nếu $a$ lẻ thì $a=2k+1$ với ($k \in \mathbb{N}$).\\
			Khi đó ta có:
			\begin{eqnarray*}
				& A & = (a-1)(a+1)(a^2+1) \\
				& & = (2k+1-1)(2k+1+1) \cdot \left[ {\left( 2k+1 \right)}^2 + 1 \right] \\
				& & = 2k(2k+2)(4k^2+4k+2) \\
				& & = 8k(k+1)(2k^2+2k+1) \\
				& \Rightarrow & A \; \vdots \; 16
			\end{eqnarray*}
			Nếu $a$ không chia hết cho $3$ thì $a = 3k \pm 1 $ với ($k \in \mathbb{N}$).\\
			Khi đó ta có $a^2-1 \; \vdots \; 3 \Rightarrow A \; \vdots \; 3$.\\
			Nếu $a$ không chia hết cho $5$ thì $a = 5k \pm 1 $ hoặc $a = 5k \pm 2 $ với ($k \in \mathbb{N}$).\\
			Suy ra $a^2-1 \; \vdots \; 5 $ hoặc $a^2+1 \; \vdots \; 5 $
			$\Rightarrow A \; \vdots \; 5 $.\\
			$\Rightarrow A \; \vdots \; 240 $.\\
			Vậy nếu $a$ không chia hết cho $2$, $3$ và $5$ thì $a^4-1$ chia hết cho $240$.
		\end{enumerate}	
	}
\end{bt}

\begin{bt}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K5]
	Tìm ba số nguyên tố liên tiếp $a$, $b$, $c$ sao cho $a^2+b^2+c^2$ cũng là một số nguyên tố.
	\loigiai{
		Xét $2$ trường hợp:
		\begin{enumerate}
			\item Trong $a$, $b$, $c$ có một số bằng $3$.\\
			Khi đó: $2^2+3^2+5^2=38$, là hợp số, loại.\\
			Còn $3^2+5^2+7^2=83$ là một số nguyên tố, thỏa mãn.
			\item Cả $a$, $b$, $c$ đều lớn hơn $3$.\\
			Khi đó $a^2$, $b^2$, $c^2$ đều chia cho $3$ dư $1$ nên $a^2+b^2+c^2$ chia hết cho $3$, là hợp số, loại.
		\end{enumerate}
		Vậy ba số nguyên tố liên tiếp $a$, $b$, $c$ là $3$, $5$, $7$ thì $a^2+b^2+c^2$ cũng là một số nguyên tố.
	}
\end{bt}

\begin{bt}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K5]
	Cho bốn số nguyên dương $a$, $b$, $c$, $d$ thỏa mãn $a^2+b^2=c^2+d^2$. Chứng minh rằng $a+b+c+d$ là hợp số.
	\loigiai{
		Ta có: $a^2+b^2=c^2+d^2 \Rightarrow a^2+b^2+c^2+d^2 = 2(c^2+d^2) \; \vdots \; 2$.\\
		Xét:
		\begin{eqnarray*}
			& A &  = (a^2 + b^2 +c^2 +d^2) - (a + b + c + d)\\
			& & = (a^2-a)+(b^2-b)+(c^2-c)+(d^2-d) \\
			& & = a(a-1)+b(b-1)+c(c-1)+d(d-1)
		\end{eqnarray*}
		Vì $a(a-1) \; \vdots \; 2$; $b(b-1) \; \vdots \; 2$; $c(c-1) \; \vdots \; 2$; $d(d-1) \; \vdots \; 2$.\\
		$\Rightarrow A \; \vdots \; 2 \Leftrightarrow a + b + c + d \; \vdots \; 2 $.\\
		Mà $a$, $b$, $c$, $d$ là các số nguyên dương suy ra $a + b + c + d >2 \Rightarrow a + b + c + d$ là hợp số (điều phải chứng minh).
	}
\end{bt}

\begin{bt}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K9]
	Cho bốn số nguyên dương $a$, $b$, $c$, $d$ thỏa mãn $ab=cd$. Chứng minh rằng $a^5+b^5+c^5+d^5$ là hợp số.
	\loigiai{
		Gọi $\text{ƯCLN}(a,c)=k$, ta có: $a=ka_1$, $c=kc_1$ và $(a_1,c_1)=1$.\\
		Thay vào $ab=cd$ được $ka_1b=kc_1d$ nên: $a_1b=c_1d$
	}
\end{bt}

\begin{bt}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1K2]
	Cho các số tự nhiên $a$ và $b$. Chứng minh rằng:
	\begin{enumerate}
		\item Nếu $a^2+b^2$ chia hết cho $3$ thì $a$ và $b$ chia hết cho $3$.
		\item Nếu $a^2+b^2$ chia hết cho $7$ thì $a$ và $b$ chia hết cho $7$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Một số chính phương chia cho $3$ chỉ có thể dư $0$ hoặc $1$.\\
			Xét các trường hợp tổng của 2 số dư: $0+0$; $0+1$; $1+1$ thì chỉ có $0+0$ là chia hết cho $3$.\\
			Do đó để $a^2+b^2$ chia hết cho $3$ thì $a^2 \; \vdots \; 3$ và $b^2 \; \vdots \; 3$.\\

			\item $A={1994}^7+7$ chia $4$ dư $3$ nên không là số chính phương.
			\item $B=144 \cdots 4$ (có $99$ chữ số $4$) $\Rightarrow B = 4\cdot 36\underbrace {11 \cdots 1}_{97}$.\\
			$36\underbrace {11 \cdots 1}_{97}$ chia $4$ dư $3$ nên $36\underbrace {11 \cdots 1}_{97}$ không phải là số chính phương.\\
			Do đó, $\Rightarrow B = 4\cdot 36\underbrace {11 \cdots 1}_{97}$ không phải là số chính phương.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Tuấn Nguyễn, dự án EX-C2-L8-2018]%[8D1G5]
	Chứng minh rằng:
	\begin{enumerate}
		\item $A=1^2+2^2+3^2+4^2+\cdots +{100}^2$ không là số chính phương.
		\item $B=1^2+2^2+3^2+4^2+\cdots +{56}^2$ không là số chính phương.
		\item $C=1+3+5+7+\cdots +n$ là số chính phương ($n$ lẻ).
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $A=1^2+2^2+3^2+4^2+\cdots +{100}^2$ gồm $50$ số chính phương chẵn, $50$ số chính phương lẻ.\\
			Mỗi số chính phương chẵn đều chia hết cho $4$ nên tổng của $50$ số chính phương chẵn chia hết cho $4$.\\
			Mỗi số chính phương lẻ đều chia cho $4$ dư $1$ nên tổng của $50$ số chính phương lẻ chia cho $4$ dư $2$.\\
			Suy ra $A$ chia $4$ dư $2$. Suy ra $A$ không là số chính phương.
			\item $B=1^2+2^2+3^2+4^2+\cdots +{56}^2$ có thể viết dưới dạng tổng của $57$ số chính phương liên tiếp là\\
			$B = 0^2 + 1^2 + 2^2 + 3^2 + 4^2 +\cdots + {56}^2$.\\
			Xét tổng của $3$ số chính phương liên tiếp $(n-1)^2+n^2+(n+1)^2=3n^2+2$ chia $3$ dư $2$.\\
			Suy ra tổng của $3$ số chính phương liên tiếp chia $3$ dư $2$.
			Ta có:
			\begin{eqnarray*}
				& B & = 0^2 + 1^2+2^2+3^2+4^2+\cdots +{56}^2 \\
				& & = (0^2+1^2+2^2) + (3^2+4^2+5^2) + \cdots + ({54}^2 + {55}^2 + {56}^2) \\
				& & = 19 \cdot (\text{BS}3+2) \\
				& & = \text{BS}3+38 \\
				& & = \text{BS}3+2
			\end{eqnarray*}
			Một số chính phương chia $3$ chỉ có dư là $0$ hoặc $1$ nên $B$ không là số chính phương.
			\item Đặt $n=2k-1$ với $k \in \mathbb{N}$.\\
			Ta có:
			\begin{eqnarray*}
				& C & = 1 + 3 + 5 + \cdots +(2k-1) \\
				& & = \dfrac{1+(2k-1)}{2} \cdot k \\
				& & = k^2 \\
				& \Rightarrow & C \text{ là số chính phương.}
			\end{eqnarray*}
			Vậy với $n$ lẻ thì $C=1+3+5+7+\cdots +n$ là số chính phương.
		\end{enumerate}	
	}
\end{bt}
