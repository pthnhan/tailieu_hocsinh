\section{Phương pháp đưa về phương trình tích}

\begin{bt}%[Cao Đình Tới]
$x^2+x-1=(x+2) \sqrt{x^2-2x+2}$
\loigiai{
Điều kiện $x \in \mathbb{R}$.	\\
Phương trình đã cho tương đương với
\begin{align*}
&x^2-2x-7+3(x+2) -(x+2) \sqrt{x^2-2x+2}=0  \\
\Leftrightarrow & x^2-2x-7+(x+2) (3-\sqrt{x^2-2x+2})=0 \\
\Leftrightarrow & x^2-2x-7+(x+2) \dfrac{-x^2+2x+7}{3+\sqrt{x^2-2x+2}}=0\\
\Leftrightarrow & (x^2-2x-7)\left(1-\dfrac{x+2}{3+\sqrt{x^2-2x+2}} \right) =0
\end{align*}
\begin{enumerate}[$\bullet$]
\item Trường hợp 1:\\ $x^2-2x-7=0 \Leftrightarrow \left\{\begin{array}{l}x=1-2 \sqrt{2} \\ x=1+2 \sqrt{2} \end{array}\right.$ (thỏa mãn điều kiện).
\item 	Trường hợp 2:
\begin{align*}
&1-\dfrac{x+2}{3+\sqrt{x^2-2x+2}}=0  \\
\Leftrightarrow & 3+\sqrt{x^2-2x+2}= x+2 \\
\Leftrightarrow & \sqrt{x^2-2x+2}= x-1 \\
\Leftrightarrow & x^2-2x+2= x^2-2x+1 \, ( \text{với}\, x \ge 1)\\ 
\Leftrightarrow & 2= 1 \, \text{(vô lý)}
\end{align*}
\end{enumerate} 
Vậy $x=1-2 \sqrt{2}$ và $x=1+2 \sqrt{2}$ là nghiệm của phương trình.
}
\end{bt}

\begin{bt}%[Cao Đình Tới]
Giải phương các trình:
\begin{enumerate}
\item $\sqrt{x^2+10x+21}=3 \sqrt{x+3}+2\sqrt{x+7}-6$
\item $\sqrt{x^3-2x^2-3x+6}-4\sqrt{x^2-3}-\sqrt{x-2}=-4$
\item $\sqrt{-x^2+4x+5}+4=2\sqrt{x+1}+2\sqrt{5-x}$
\end{enumerate}
	\loigiai{
\begin{enumerate}
\item Điều kiện xác định: $x \ge -3$.\\
Phương trình đã cho tương đương với
\begin{align*}
&\sqrt{(x+3)(x+7)}-3 \sqrt{x+3}-2\sqrt{x+7}+6=0\\
\Leftrightarrow &\sqrt{x+3}\left(\sqrt{x+7}-3 \right)-2\left( \sqrt{x+7}-3\right)=0\\
\Leftrightarrow &\left(\sqrt{x+3}-2 \right) \left(\sqrt{x+7}-3 \right)=0\\
\Leftrightarrow &\left[\begin{array}{l} \sqrt{x+3}=2\\ \sqrt{x+7}=3\end{array}\right.\\
\Leftrightarrow   &\left[\begin{array}{l} x=1\\ x=2 \end{array}\right. \qquad \text{(thỏa mãn điều kiện xác định).}
\end{align*}
Vậy $x=1$ và $x=2$ là nghiệm của phương trình.
\item Điều kiện xác định: $x \ge 2$.\\
Phương trình đã cho tương đương với
\begin{align*}
&\sqrt{(x-2)\left(x^2-3 \right) }-\sqrt{x-2}-4\sqrt{x^2-3}+4=0\\
\Leftrightarrow &\sqrt{x-2}\left(\sqrt{x^2-3}-1 \right)-4\left( \sqrt{x^2-3}-1\right)=0\\
\Leftrightarrow &\left(\sqrt{x^2-3}-1 \right) \left(\sqrt{x-2}-4 \right)=0\\
\Leftrightarrow &\left[\begin{array}{l} \sqrt{x^2-3}-1=0\\ \sqrt{x-2}-4=0\end{array}\right.\\
\Leftrightarrow &\left[\begin{array}{l} x^2=4\\ x=18\end{array}\right.\\
\Leftrightarrow   &\left[\begin{array}{l} x=\pm 2\\ x=18 \end{array}\right.
\end{align*}
Nghiệm $x=-2$ không thỏa mãn điều kiện xác định. Vậy $x=2$ và $x=18$ là nghiệm của phương trình.
\item Điều kiện xác định: $-1 \le x \le 5$.\\
Phương trình đã cho tương đương với
\begin{align*}
&\sqrt{-x^2+4x+5}+4-2\sqrt{x+1}-2\sqrt{5-x}=0\\
\Leftrightarrow &\sqrt{(x+1)(5-x)}-2\sqrt{x+1}-2\sqrt{5-x}+4=0\\
\Leftrightarrow
 &\sqrt{x+1}\left(\sqrt{5-x}-2 \right)-2\left( \sqrt{5-x}-2\right)=0\\
\Leftrightarrow &\left(\sqrt{x+1}-2 \right) \left(\sqrt{5-x}-2 \right)=0\\
\Leftrightarrow &\left[\begin{array}{l} \sqrt{x+1}-2=0\\ \sqrt{5-x}-2=0\end{array}\right.\\
\Leftrightarrow &\left[\begin{array}{l} x=3\\ x=1\end{array}\right. \qquad \text{(thỏa mãn điều kiện xác định).}
\end{align*}
Vậy $x=3$ và $x=1$ là nghiệm của phương trình.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Cao Đình Tới]
Giải phương trình: $\sqrt{x+6} \cdot \sqrt[3]{x-9}-x+\sqrt{x+6}-x\sqrt[3]{x-9}=0$
	\loigiai{
Điều kiện xác định: $ x \ge -6$.\\
Phương trình đã cho tương đương với
\begin{align*}
&\sqrt{x+6} \cdot \sqrt[3]{x-9}+\sqrt{x+6}-x-x\sqrt[3]{x-9}=0\\
\Leftrightarrow &\sqrt{x+6}\left(\sqrt[3]{x-9}+1 \right) -x\left(\sqrt[3]{x-9}+1 \right)=0\\
\Leftrightarrow
 &\left(\sqrt{x+6}-x \right) \left(\sqrt[3]{x-9}+1 \right)=0\\
\Leftrightarrow &\left[\begin{array}{l} \sqrt{x+6}=x\\ \sqrt[3]{x-9}=-1\end{array}\right.\\
\Leftrightarrow   &\left[\begin{array}{l} \sqrt{x+6}=x  \qquad (1)\\ x=8 \qquad \qquad \text{(thỏa mãn điều kiện xác định).} \end{array}\right.
\end{align*}
Giải (1): 
\begin{align*} \sqrt{x+6}=x & \Leftrightarrow \left\{\begin{array}{l} x \ge 0\\ x+6=x^2 \end{array}\right. \\
&\Leftrightarrow \left\{\begin{array}{l} x\ge 0\\ x^2-x-6=0 \end{array}\right.\\
& \Leftrightarrow x=3
 \end{align*}
Vậy $x=8$ và $x=3$ là nghiệm của phương trình.
}
\end{bt}

\begin{bt}%[Cao Đình Tới]
Giải phương trình: $3\sqrt{x+1} -3 \sqrt{x-1}= \sqrt{x^2+5x+4} - \sqrt{x^2+3x-4} $
	\loigiai{
Điều kiện xác định: $ x \ge 1$.\\
Phương trình đã cho tương đương với
\begin{align*}
&3 \left(\sqrt{x+1}-\sqrt{x-1}\right)= \sqrt{(x+4)(x+1)}-\sqrt{(x+4)(x-1)} \\
\Leftrightarrow & 3 \left(\sqrt{x+1}-\sqrt{x-1}\right)=\sqrt{x+4} \left(\sqrt{x+1}-\sqrt{x-1}\right)\\
\Leftrightarrow
 &\left(\sqrt{x+1}-\sqrt{x-1}\right) \left(3-\sqrt{x+4}\right)=0
\end{align*}
\begin{enumerate}[$\bullet$]
\item 	Trường hợp 1:
\begin{align*}
&\sqrt{x+1}-\sqrt{x-1}=0 \\
\Leftrightarrow & \sqrt{x+1}=\sqrt{x-1} \\
\Leftrightarrow & x+1=x-1 \\
\Leftrightarrow & 1=-1 \, \text{(vô lý)}.
\end{align*}
\item 	Trường hợp 2:
\begin{align*}
&3-\sqrt{x+4}=0 \\
\Leftrightarrow & \sqrt{x+4}=3 \\
\Leftrightarrow & x=5. \qquad \text{(thỏa mãn điều kiện xác định).}
\end{align*}
\end{enumerate} 
Vậy phương trình có nghiệm $x=5$.
}
\end{bt}

\begin{bt}%[Cao Đình Tới]
	Giải phương trình: $x^2+8x=\sqrt{2x+6}$.
	\loigiai{
		Điều kiện xác định: $ x \ge -3$.\\
		Cộng $2x+6\dfrac{1}{4}$ vào hai vế thì phương trình đã cho tương đương với
		\begin{align*}
		&x^2+10x+6\dfrac{1}{4}=2x+6+\sqrt{2x+6}+\dfrac{1}{4}\\
		\Leftrightarrow & \left(2x+\dfrac{5}{2} \right)^2=\left( \sqrt{2x+6} +\dfrac{1}{2} \right)^2
		\end{align*}
	\begin{enumerate}[$\bullet$]
		\item 	Trường hợp 1:
		\begin{align*}
		&2x+\dfrac{5}{2}=\sqrt{2x+6} +\dfrac{1}{2} \\
		\Leftrightarrow & 2x+2=\sqrt{2x+6} \\
		\Leftrightarrow & \left\{\begin{array}{l}  x \ge -1\\ 2x^2+3x-1= 0 \end{array}\right. \\
		\Leftrightarrow & x=\dfrac{-3+\sqrt{17}}{4} \, \text{(thỏa mãn điều kiện xác định).}
		\end{align*}
		\item 	Trường hợp 2:
		\begin{align*}
		&2x+\dfrac{5}{2}=-\sqrt{2x+6} -\dfrac{1}{2} \\
		\Leftrightarrow & -2x-3=\sqrt{2x+6}\\
		\Leftrightarrow & \left\{\begin{array}{l}  x \le -\dfrac{3}{2}\\ 4x^2+10x+3= 0 \end{array}\right. \\
		\Leftrightarrow & x=\dfrac{-5-\sqrt{13}}{4} \,  \text{(thỏa mãn điều kiện xác định).}
		\end{align*}
	\end{enumerate}
		Vậy $x=\dfrac{-3+\sqrt{17}}{4}$ và $x=\dfrac{-5-\sqrt{13}}{4}$ là nghiệm của phương trình.
	}
\end{bt}

\begin{bt}%[Cao Đình Tới]
	Giải phương trình: $x^2-10x-12=4\sqrt{2x+3}$.
	\loigiai{
		Điều kiện xác định: $ x \ge -\dfrac{3}{2}$.\\
		Cộng $8x+13$ vào hai vế thì phương trình đã cho tương đương với
		\begin{align*}
		&x^2-2x+1=4(2x+3)+4\sqrt{2x+3}+1\\
		\Leftrightarrow & \left(x-1 \right)^2=\left( 2\sqrt{2x+3} +1 \right)^2
		\end{align*}
		\begin{enumerate}[$\bullet$]
			\item 	Trường hợp 1:
			\begin{align*}
			&x-1=2\sqrt{2x+3} +1 \\
			\Leftrightarrow & x-2=2\sqrt{2x+3} \\
			\Leftrightarrow & \left\{\begin{array}{l}  x \ge 2\\ x^2-12x-8= 0 \end{array}\right. \\
			\Leftrightarrow & x=6+2\sqrt{11}\, \text{(thỏa mãn điều kiện xác định).}
			\end{align*}
			\item 	Trường hợp 2:
			\begin{align*}
			&-x+1=2\sqrt{2x+3}+1 \\
			\Leftrightarrow & -x=2\sqrt{2x+3}\\
			\Leftrightarrow & \left\{\begin{array}{l}  x \le 0\\ x^2-8x-12= 0 \end{array}\right. \\
			\Leftrightarrow & x=4-2\sqrt{7} \,  \text{(thỏa mãn điều kiện xác định).}
			\end{align*}
		\end{enumerate}
		Vậy $x=6+2\sqrt{11}$ và $x=4-2\sqrt{7} $ là nghiệm của phương trình.
	}
\end{bt}

\begin{bt}%[Cao Đình Tới]
	Giải phương trình: $2x^2+x-3=2\sqrt{2x+3}$.
	\loigiai{
		Điều kiện xác định: $ x \ge -\dfrac{3}{2}$.\\
		Phương trình đã cho tương đương với
		\begin{align*}
		&4x^2+2x-6=4\sqrt{2x+3}\\
		&4x^2+2x-6+2x+7=4\sqrt{2x+3}+2x+7\\
		&4x^2+4x+1=2x+3+4\sqrt{2x+3}+4\\
		\Leftrightarrow & \left(2x+1 \right)^2=\left( \sqrt{2x+3} +2 \right)^2
		\end{align*}
		\begin{enumerate}[$\bullet$]
			\item 	Trường hợp 1:
			\begin{align*}
			&2x+1=\sqrt{2x+3} +2 \\
			\Leftrightarrow & 2x-1=\sqrt{2x+3} \\
			\Leftrightarrow & \left\{\begin{array}{l}  x \ge \dfrac{1}{2}\\ 2x^2-3x-1= 0 \end{array}\right. \\
			\Leftrightarrow & x=\dfrac{3+\sqrt{17}}{4}\, \text{(thỏa mãn điều kiện xác định).}
			\end{align*}
			\item 	Trường hợp 2:
			\begin{align*}
			&-2x-1=\sqrt{2x+3}+2 \\
			\Leftrightarrow & -2x-3=\sqrt{2x+3}\\
			\Leftrightarrow & \left\{\begin{array}{l}  x \le -\dfrac{3}{2}\\ 2x^2+5x+3= 0 \end{array}\right. \\
			\Leftrightarrow & x=-\dfrac{3}{2} \,  \text{(thỏa mãn điều kiện xác định).}
			\end{align*}
		\end{enumerate}
		Vậy $x=\dfrac{3+\sqrt{17}}{4}$ và $x=-\dfrac{3}{2} $ là nghiệm của phương trình.
	}
\end{bt}

\begin{bt}%[Cao Đình Tới]
	Giải phương trình: $4x+1-x^2=2\sqrt{2x+1}$.
	\loigiai{
		Điều kiện xác định: $ x \ge -\dfrac{1}{2}$.\\
		Phương trình đã cho tương đương với
		\begin{align*}
		&4x+1-x^2 +2x+2=2\sqrt{2x+1}+2x+2\\
		&x^2-2x+1=2x+1-2\sqrt{2x+1}+1\\
		\Leftrightarrow & \left(x-1 \right)^2=\left( \sqrt{2x+1} -1 \right)^2
		\end{align*}
		\begin{enumerate}[$\bullet$]
			\item 	Trường hợp 1:
			\begin{align*}
			&x-1=\sqrt{2x+1} -1 \\
			\Leftrightarrow & x=\sqrt{2x+1} \\
			\Leftrightarrow & \left\{\begin{array}{l}  x \ge 0\\ x^2-2x-1= 0 \end{array}\right. \\
			\Leftrightarrow & x=1+\sqrt{2}\, \text{(thỏa mãn điều kiện xác định).}
			\end{align*}
			\item 	Trường hợp 2:
			\begin{align*}
			&-x+1=\sqrt{2x+1}-1 \\
			\Leftrightarrow & -x+2=\sqrt{2x+1}\\
			\Leftrightarrow & \left\{\begin{array}{l}  x \le 2\\ x^2-6x+3= 0 \end{array}\right. \\
			\Leftrightarrow & x=3-\sqrt{6} \,  \text{(thỏa mãn điều kiện xác định).}
			\end{align*}
		\end{enumerate}
		Vậy $x=1+\sqrt{2}$ và $x=3-\sqrt{6} $ là nghiệm của phương trình.
	}
\end{bt}

\begin{bt}%[Cao Đình Tới]
	Giải phương trình: $x^2-x-2=2\sqrt{16x+1}$.
	\loigiai{
		Điều kiện xác định: $ x \ge -\dfrac{1}{16}$.\\
		Phương trình đã cho tương đương với
		\begin{align*}
		&4x^2-4x-8 =8\sqrt{16x+1}\\
		&4x^2-4x-8 +16x+17 =8\sqrt{16x+1}+16x+17\\
		&4x^2+12x+9=16x+1+8\sqrt{2x+1}+16\\
		\Leftrightarrow & \left(2x+3 \right)^2=\left( \sqrt{16x+1} +4 \right)^2
		\end{align*}
		\begin{enumerate}[$\bullet$]
			\item 	Trường hợp 1:
			\begin{align*}
			&2x+3 = \sqrt{16x+1} +4 \\
			\Leftrightarrow & 2x-1=\sqrt{16x+1} \\
			\Leftrightarrow & \left\{\begin{array}{l}  x \ge \dfrac{1}{2} \\ 2x^2-20x= 0 \end{array}\right. \\
			\Leftrightarrow & x=5 \, \text{(thỏa mãn điều kiện xác định).}
			\end{align*}
			\item 	Trường hợp 2:
			\begin{align*}
			&-2x-3=\sqrt{16x+1} +4 \\
			\Leftrightarrow & -2x-7=\sqrt{16x+1}\\
			\Leftrightarrow & \left\{\begin{array}{l}  x \le -\dfrac{7}{2}\\ x^2+3x+12= 0 \end{array}\right.  \text{vô nghiệm}.
			\end{align*}
		\end{enumerate}
		Vậy $x=5$ là nghiệm của phương trình.
	}
\end{bt}

\begin{bt}%[Cao Đình Tới]
	Giải phương trình: $2x^2-3x+2=x\sqrt{3x-2}$.
	\loigiai{
		Điều kiện xác định: $ x \ge \dfrac{2}{3}$.\\
		Phương trình đã cho tương đương với
		\begin{align*}
		&2x^2=3x-2+x\sqrt{3x-2}\\
		&2x^2+ \dfrac{x^2}{4}=3x-2+x\sqrt{3x-2} + \dfrac{x^2}{4}\\
		&\left( \dfrac{3x}{2} \right)^2=\left(\sqrt{3x-2}+\dfrac{x}{2} \right)^2
		\end{align*}
		\begin{enumerate}[$\bullet$]
			\item 	Trường hợp 1:
			\begin{align*}
			&\dfrac{3x}{2} = \sqrt{3x-2}+\dfrac{x}{2}\\
			\Leftrightarrow & x=\sqrt{3x-2} \\
			\Leftrightarrow & \left\{\begin{array}{l}  x \ge 0 \\ x^2-3x+2= 0 \end{array}\right. \\
			\Leftrightarrow & \left[\begin{array}{l}  x =1 \\ x=2 \end{array}\right. \,  \text{(thỏa mãn điều kiện xác định). }
			\end{align*}
			\item 	Trường hợp 2:
			\begin{align*}
			&\dfrac{3x}{2}=\sqrt{3x-2}+\dfrac{x}{2} \\
			\Leftrightarrow & -2x=\sqrt{3x-2} \\ &  \text{Không xảy ra với }  x \ge \dfrac{2}{3}
			\end{align*}
		\end{enumerate}
		Vậy $x=1$ và $x=2$ là nghiệm của phương trình.
	}
\end{bt}
