\begin{ex}%[Đề thi vào 10, Sở giáo dục Thái Bình, 2016]%[9D2B3]%[9D4K6]
Cho parabol $(P):y=x^{2}$ và đường thẳng $d:y=2(m-1)x+m^{2}+2m$ ($m$ là tham số).
\begin{enumerate}
\item Tìm $m$ để đường thẳng $d$ đi qua điểm $I(1;3)$.
\item Chứng minh rằng parabol $(P)$ luôn cắt đường thẳng $d$ tại hai điểm phân biệt $A$, $B$. Gọi $x_{1}$, $x_{2}$ là hoành độ của hai điểm $A$, $B$. Tìm $m$ để $x_{1}^{2}+x_{2}^{2}+6x_{1}x_{2}>2016$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Đường thẳng $d$ đi qua điểm $I(1;3)\Leftrightarrow 2(m-1)+m^{2}+2m=3\Leftrightarrow m^{2}+4m-5=0\Leftrightarrow \hoac{&m=1\\ &m=-5}$
\item Xét phương trình hoành độ giao điểm của $(P)$ và $d$:
\[
x^{2}=2(m-1)x+m^{2}+2m\Leftrightarrow x^{2}-2(m-1)x-m^{2}-2m=0
\]
Ta có $\Delta'=2m^{2}+1>0$ với $\forall m$ nên $(P)$ luôn cắt $d$ tại hai điểm phân biệt.\\
Theo Viet ta có $\heva{&x_{1}+x_{2}=2(m-1)\\ &x_{1}x_{2}=-m^{2}-2m}$. Theo đề bài
\begin{align*}
&x_{1}^{2}+x_{2}^{2}+6x_{1}x_{2}>2016\Leftrightarrow (x_{1}+x_{2})^{2}+4x_{1}x_{2}>2016\Leftrightarrow 4(m-1)^{2}-4m^{2}-8m-2016>0\\ 
&\Leftrightarrow m<-\dfrac{503}{4}
\end{align*}
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Thái Bình, 2016]%[9D3Y4]%[9D4B8]
\hfill\\
\begin{enumerate}
\item Giải hệ phương trình $\heva{&2x-y=1\\ &3x-4y=-6}$
\item Cho tam giác vuông có độ dài cạnh huyền bằng $15$ cm. Hai cạnh góc vuông có độ dài hơn kém nhau $3$ cm. Tìm độ dài hai cạnh góc vuông của tam giác vuông đó.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $\heva{&2x-y=1\\ &3x-4y=-6}\Leftrightarrow \heva{&8x-4y=4\\ &3x-4y=-6}\Leftrightarrow \heva{&5x=10\\ &3x-4y=-6}\Leftrightarrow \heva{&x=2\\ &y=3}$
\item Gọi độ dài cạnh góc vuông nhỏ nhất là $x$ (cm, $0<x<12$)\\
Độ dài cạnh góc vuông còn lại là $x+3$ cm.\\
Theo định lí Pitago $x^{2}+(x+3)^{2}=15^{2}\Leftrightarrow x^{2}+3x-108=0\Leftrightarrow \hoac{&x=9\ (TM)\\ &x=-12\ (\mbox{Loại})}$\\
Vậy độ dài hai cạnh góc vuông lần lượt là $9$ cm và $12$ cm.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Thái Bình, 2016]%[9H3Y7]%[9H2B5]%[9H3K4]%[9H3K6]
Cho đường tròn tâm $O$ và điểm $A$ nằm ngoài đường tròn. Từ $A$ kẻ hai tiếp tuyến $AB$ và $AC$ ($B$, $C$ là các tiếp điểm).
\begin{enumerate}
\item Chứng minh tứ giác $ABOC$ là tứ giác nội tiếp.
\item Gọi $H$ là trực tâm tam giác $ABC$. Chứng minh tứ giác $BOCH$ là hình thoi.
\item Gọi $I$ là giao điểm của đoạn $OA$ với đường tròn $(O)$. Chứng minh $I$ là tâm đường tròn nội tiếp tam giác $ABC$.
\item Cho $OB=3$ cm, $OA=5$ cm. Tính diện tích tam giác $ABC$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1]
\tkzInit[xmax=12,ymax=6]
\tkzDefPoint(0,0){O}
\tkzDefShiftPoint[O](0:9){A}
\tkzTangent[from with R=A](O,3 cm)\tkzGetPoints{B}{C}
\tkzInterLC(A,O)(O,B)\tkzGetPoints{I}{I'}
\tkzDrawAltitude(A,B)(C)\tkzGetPoint{F}
\tkzDrawAltitude(A,C)(B)\tkzGetPoint{E}
\tkzInterLL(A,O)(E,B)\tkzGetPoint{H}
\tkzInterLL(A,O)(C,B)\tkzGetPoint{K}
\tkzDrawCircle[radius](O,B)
\tkzDrawSegments(A,B A,C A,O B,C B,E B,O B,I C,F C,O C,I)
\pgfresetboundingbox
\tkzMarkAngle[size=.5](I,B,A)
\tkzMarkAngle[size=.5](A,C,I)
\tkzMarkAngle[size=.6](C,B,I)
\tkzMarkAngle[size=.6](I,C,B)
\tkzMarkRightAngle(A,B,O)
\tkzMarkRightAngle(A,C,O)
\tkzMarkRightAngle(O,K,B)
\tkzMarkRightAngle(B,E,A)
\tkzMarkRightAngle(C,F,A)
\tkzLabelPoints(A,C,K,H,I)
\tkzLabelPoints[above](B)
\tkzLabelPoints[below left](O)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Vì $AB$, $AC$ là các tiếp tuyến của $(O)$ (tại $B$, $C$) nên $\widehat{ABO}=\widehat{ACO}=90^{\circ}\Rightarrow ABOC$ là tứ giác nội tiếp đường tròn đường kính $AO$.
\item Vì $\heva{& OB\perp AB\\ & CH\perp AB}\Rightarrow OB\parallel CH$ (1). Tương tự $OC\parallel BH$ (2).\\
Từ (1) và (2) ta có $BOCH$ là hình bình hành. Mà $OB=OC$ nên $BOCH$ là hình thoi.
\item Vì $AB$, $AC$ là các tiếp tuyến của $(O)$ nên $AO$ là tia phân giác $\widehat{BAC}$. Vì $I$ là giao điểm của đoạn $AO$ với $(O)$ nên $I$ là điểm chính giữa của cung (nhỏ) $\wideparen{BC}\Rightarrow $ $\widehat{IBC}=\dfrac{1}{2}\mbox{sđ}\wideparen{BC}$ (3) (tính chất góc nội tiếp).\\
Vì $I$ là điểm chính giữa cung $\wideparen{BC}$ và $AB$ là tiếp tuyến của $(O)$ nên $\widehat{ABI}=\dfrac{1}{2}\mbox{sđ}\wideparen{BC}$ (4).\\
Từ (3) và (4) ta suy ra $BI$ là tia phân giác $\widehat{ABC}$, do vậy $I$ là tâm đường tròn nội tiếp tam giác $ABC$. 
\item Gọi $K$ là giao điểm của $OA$ và $BC\Rightarrow K$ là trung điểm của $BC$ và $BK\perp AO$.\\
Áp dụng định lý Pitago cho tam giác $AOB$ vuông tại $B$: $AB=\sqrt{AO^{2}-OB^{2}}=4$ cm.\\
Áp dụng hệ thức lượng cho tam giác $AOB$ vuông tại $B$:\\
$\dfrac{1}{BK^{2}}=\dfrac{1}{AB^{2}}+\dfrac{1}{OB^{2}}=\dfrac{25}{144}\Rightarrow BK=\dfrac{12}{5}\Rightarrow AK=\sqrt{AB^{2}-BK^{2}}=\dfrac{16}{5}$ cm.\\
Diện tích tam giác $ABC$ là $S(\Delta ABC)=\dfrac{1}{2}.AK.BC=\dfrac{192}{25}$ cm$^{2}$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Thái Bình, 2016]%[9H3B7]%[9D5G5]
Giải phương trình $x^{3}+(3x^{2}-4x-4)\sqrt{x+1}=0$.
\loigiai{
Điều kiện $x\geq -1$.\\
Đặt $y=\sqrt{x+1}$ ($y\geq 0$) $\Rightarrow x=y^{2}-1$. Khi đó phương trình trở thành $x^{3}+3x^{2}y-4y^{3}=0$.
\begin{itemize}
\item Nếu $y=0$ thì $x=0\Rightarrow$ phương trình vô nghiệm.
\item Nếu $y\neq 0$, chia hai vế cho $y^{3}$ ta được $\left(\dfrac{x}{y}\right)^{3}+3\left(\dfrac{x}{y}\right)^{2}-4=0\Leftrightarrow \hoac{&x=y\\ &x=-2y}$.
\begin{itemize}
\item Với $x=y\Rightarrow y^{2}-y-1=0\Rightarrow y=\dfrac{1+\sqrt{5}}{2}$ (vì $y\geq 0$) $\Rightarrow x=\dfrac{1+\sqrt{5}}{2}$.
\item Với $x=-2y\Rightarrow y^{2}+2y-1=0\Rightarrow y=\sqrt{2}-1$ (vì $y\geq 0$) $\Rightarrow x=2-2\sqrt{2}$.
\end{itemize}
\end{itemize}
Vậy tập nghiệm của phương trình là $\left\{\dfrac{1+\sqrt{5}}{2};2-2\sqrt{2}\right\}$.
}
\end{ex}
