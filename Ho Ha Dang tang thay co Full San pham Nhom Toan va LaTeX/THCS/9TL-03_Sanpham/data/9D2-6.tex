\subsection{Đề số 1 (Dành cho học sinh đại trà)}
\begin{ex}
Cho đường thẳng $d:y=2x-1$.
\begin{enumerate}
	\item Tìm tọa độ điểm $A$ thuộc $d$, biết rằng $A$ có hoành độ bằng $-1$.
	\item Tìm tọa độ điểm $B$ thuộc $d$, biết $B$ có tung độ bằng $5$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có $x_A=-1$, nên $y_A=2\cdot (-1)-1=-3$. Vậy $A(-1;-3)$.
	\item Ta có $y_B=5$, nên $2x_B-1=5 \Leftrightarrow x_B=3.$ Vậy $B(3;5).$
\end{enumerate}
}
\end{ex}
\begin{ex}
Cho hàm số $ y=(m-2)x+2m-2$.
\begin{enumerate}
	\item Tìm $m$ để hàm số đã cho đồng biến.
	\item Tìm $m$ để đồ thị hàm số đã cho đi qua điểm $A(2;-8)$.
	\item Vẽ đồ thị hàm số đã cho và tính góc giữa đường thẳng với trục $Ox$ khi $m=1$.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Hàm số đã cho đồng biến khi $m-2>0 \Leftrightarrow m>2.$
	\item  Đồ thị hàm số đã cho đi qua điểm $A(2;-8)$ khi 
	$$-8=(m-2)2+2m-2 \Leftrightarrow -8=4m-6 \Leftrightarrow m=-\dfrac{1}{2}.$$
	\item \immini{Với $m=0$ ta có $y=-x$.\\
	Do $a=-1<0$ nên hàm số nghịch biến.\\
	Đồ thị hàm số là đường thẳng đi qua hai điểm $(0;0)$ và $(-1;1)$.\\
Ta có $\tan \widehat{AOB}=\dfrac{AB}{OA}=1$, suy ra $\widehat{AOB}=45^\circ$.\\ Do đó $\alpha=180^\circ-45^\circ=135^\circ$.
}

\subsection{Đề số 2 (Dành cho học sinh khá, giỏi)}
\begin{ex}
	Cho ba đường thẳng $d_1:y=3x-1$,  $d_2:y=5-3x$ và $d_{3}:y=\left(2m-1\right)x+4-m$. 
	\begin{enumerate}
		\item Vẽ đường thẳng $d_1$.
		\item Tìm $m$ để $ d_2 \parallel d_3$.
		\item Tìm $m$ để hai đường thẳng $d_{1}$ và $d_{3}$ cắt nhau tại một điểm nằm trên trục $Oy$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item \immini{Đường thẳng $d_1$ đi qua hai điểm $M(0;-1)$,\ $N(1;2)$}
			{\begin{tikzpicture}[scale=0.7]
				\tkzInit[ymin=-2,ymax=2.7,xmin=-1,xmax=2]
				\tkzClip
				\draw[->] (-2.5,0)--(1.5,0) node[below]{$x$};
				\draw[->] (0,-2.5)--(0,2.5) node[right]{$y$};
				\draw[domain=-2:2] plot(\x,{3*(\x)-1});
				\node at (0,0) [below left]{$O$};
				\node at (1,0) [below]{$1$};
				\node at (0,2) [left]{$2$};
				\node at (0,-1) [left]{$-1$};
				\draw (1,0)  circle (1pt);
				\draw  (0,2) circle (1pt);
				\draw (0,0)circle (1pt);
				\draw (0,-1)circle (1pt);
				\draw[dashed] (1,0)--(1,2)--(0,2);
							
				\end{tikzpicture}}
			\item Ta có $$d_2 \parallel d_3 \Leftrightarrow \begin{cases} 2m-1=-3 \\ 4-m \neq 5	\end{cases} \Leftrightarrow \begin{cases} m=-1\\ m\neq -1 \end{cases}\ \text{vô nghiệm}.$$
			Vậy không tồn tại  $m$ để $d_2 \parallel d_3$.
			\item Gọi $B$ là giao điểm của hai đường thẳng $d_{1}$ và $d_{3}$.
			Vì $B$ nằm trên $Oy$ nên $B\left({0;y}\right)$, mà $B\in d_{1}$ nên $y=-1\Rightarrow B\left({0;-1}\right)$.\\
			Lại có $B\in d_{3}$ nên suy ra $-1=4-m\Leftrightarrow m=5.$ \\
			Vậy $m=5$ là giá trị cần tìm.
			
		\end{enumerate}
	}
\end{ex}
\begin{ex}
	Cho đường thẳng $d:y=(m-1)x+2m+1$.
	\begin{enumerate}
		\item Tìm $m$ để $y(2017^3)>y(2019^4)$.
		\item Tìm $m$ để đường thẳng $d$ cắt $Ox$, $Oy$ tại hai điểm $A$, $B$ sao cho tam giác $AOB$ cân.
	\end{enumerate}
		\loigiai{ 
		\begin{enumerate}
			\item Ta thấy hàm số đã cho có dạng $y=ax+b$.\\ Ta có $2017^3<2019^4$, nên $y\left(2017^3\right)>y \left(2019^4 \right) \Leftrightarrow$ Hàm số đã cho nghịch biến, hay $$m-1<0 \Leftrightarrow m<1.$$
			\item  Vì tam giác $AOB$ vuông cân tại $O$, nên đường thẳng $AB$ (tức là đường thẳng $d$) vuông góc với phân giác của góc $\widehat{AOB}$, tức là vuông góc với đường thẳng $y=x$ hoặc $y=-x$. Từ đó suy ra 
			$$\hoac{&m-1=1 \\ & m-1=-1} \Leftrightarrow \hoac{&m=2 \\ &m=0.}$$
			Thử lại, ta thấy cả hai giá trị này thỏa mãn.\\
			Vậy $m=0,\ m=2$ là các giá trị cần tìm. 
		\end{enumerate}
	}
\end{ex}
