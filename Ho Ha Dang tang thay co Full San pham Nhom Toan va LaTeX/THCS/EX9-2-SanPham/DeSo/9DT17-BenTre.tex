\begin{ex}%[Thi vào 10, Sở GD Bến Tre, 2017]%[9D1B7]%[9D3B3]
Không sử dụng máy tính cầm tay hãy 
\begin{enumerate}
\item Tính $\sqrt{18}-2\sqrt{2}+\dfrac{5}{\sqrt{2}}$
\item Giải hệ phương trình $\heva{& 3x-y=1\\ & x+2y=5}$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $\sqrt{18}-2\sqrt{2}+\dfrac{5}{\sqrt{2}}=3\sqrt{2}-2\sqrt{2}+\dfrac{5\sqrt{2}}{2}=\left(3-2+\dfrac{5}{2}\right)\sqrt{2}=\dfrac{7\sqrt{2}}{2}$.
\item $\heva{& 3x-y=1\\ & x+2y=5}\Leftrightarrow\heva{& 6x-2y=2\\& x+2y=5}\Leftrightarrow\heva{& 7x=7\\& x+2y=5}\Leftrightarrow\heva{& x=1\\& y=2}$\\
Vậy hệ phương trình có nghiệm $(x;y)=(1;2)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Thi vào 10, Sở GD Bến Tre, 2017]%[9D4K2]%[9D4B4]
Trong mặt phẳng tọa độ $Oxy$ cho parabol $(P):$ $y=-2x^2$ và đường thẳng $(d):$ $y=2x-4$.
\begin{enumerate}
\item Vẽ đồ thị của $(P)$ và $(d)$ trên cùng mặt phẳng tọa độ.
\item Bằng phương pháp đại số, hãy tìm tọa độ giao điểm của $(P)$ và $(d)$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Vẽ $(P)$ ta lập bảng giá trị của $(P):$ $y=-2x^2$\\
\begin{center}
\begin{tabular}{m{2cm}|m{1cm}|m{1cm}|m{1cm}|m{1cm}|m{1cm}}
$x$ & $-2$ & $-1$ & $0$ & $1$ & $2$\\\hline
$y=-2x^2$ & $-8$ & $-2$ & $0$ & $-2$ & $-8$\\
\end{tabular}
\end{center}
Vẽ $(d):$ $y=2x-4$\\
\hspace*{1cm} Cho $x=0\Rightarrow y=-4\Rightarrow (0;-4)$.\\
\hspace*{1cm} Cho $y=0\Rightarrow x=2\Rightarrow (2;0)$.\\
Khi đó ta vẽ $(d)$ đi qua $(0;-4)$ và $(2;0)$.
\begin{center}
	
	\begin{tikzpicture}[>=stealth,scale=.7,thick]
	\draw [->] (-3,0)--(3,0)node[above]{\footnotesize $x$};
	\foreach \x in {-2,-1,1,2}
	\draw [thin,above,shift={(\x,0)}] (0pt,-2pt)--(0pt,2pt)node[above]{\footnotesize \x};
	\draw[->] (0,-9)--(0,2)node[left]{\footnotesize $y$};
	\foreach \y in {-2,-4,-8}
	\draw [thin,left,shift={(0,\y)}] (-2pt,0pt)--(2pt,0pt)node[above left]{\footnotesize \y};
	\clip  (-3,-9) rectangle (3,2);
	\draw [thick,smooth,samples=100,domain=-2.5:2.5] plot(\x,{-2*(\x)^2});
	\draw [thick,smooth,samples=100,domain=-2.5:2.5] plot(\x,{2*(\x)-4});
	\draw [thin,dashed] (-2,0)--(-2,-8)--(2,-8)node[below left]{$(P)$}--(2,0);
	\draw [thin,dashed] (-1,0)--(-1,-2)--(1,-2)--(1,0);
	\draw (-0.3,0.3)node{\footnotesize $O$};
	\draw (2.2,1.3)node{$(d)$};
	\end{tikzpicture}
	
\end{center}
\item Phương trình hoành độ giao điểm của $(P)$ và $(d)$ là: $-2x^2=2x-4\Leftrightarrow 2x^2+2x-4=0\Leftrightarrow\hoac{& x=1\Rightarrow y=-2\\& x=-2\Rightarrow y=-8}$.\\
Vậy tọa độ giao điểm của $(P)$ và $(d)$ là: $(1;-2)$ và $(-2;-8)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Thi vào 10, Sở GD Bến Tre, 2017]%[9D4B4]%[9D4K6]
Cho phương trình $x^2-2(m-1)x-(2m+1)=0$\hspace{1cm} $(1)$\hspace{0.5cm} ($m$ là tham số).
\begin{enumerate}
\item Giải phương trình $(1)$ với $m=2$.
\item Chứng minh rằng phương trình $(1)$ luôn có hai nghiệm phân biệt với mọi $m$.
\item Tìm $m$ để phương trình $(1)$ luôn có hai nghiệm bằng nhau về giá trị tuyệt đối và trái dấu nhau.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Với $m=2$ phương trình trở thành $x^2-2x-3=0$ Phương trình có: $a-b+c=0\Rightarrow$ phương trình có hai nghiệm $\hoac{& x_1=-1\\& x_2=3}$.\\
Vậy với $m=2$ phương trình đã cho có hai nghiệm phân biệt $x_1=-1$; $x_2=3$.
\item Phương trình $(1)$ có $\Delta'=m^2+2>0$, $\forall m$. \\
Vậy phương trình $(1)$ luôn có hai nghiệm phân biệt với mọi $m$.
\item Theo hệ thức vi-ét ta có $\heva{& S=x_1+x_2=2m-2\\& P=x_1.x_2=-(2m+1)}$\\
Theo đề bài ta cần $x_1$, $x_2$ là hai nghiệm đối nhau của phương trình $(1)$ nên:\\
$\heva{& S=0\\& P<0}\Leftrightarrow\heva{& 2m-2=0\\& -(2m+1)<0}\Leftrightarrow\heva{& m=1\\& m>-\dfrac{1}{2}}\Leftrightarrow m=1$\\
Vậy khi $m=1$ thì phương trình $(1)$ có hai nghiệm bằng nhau về giá trị tuyệt đối và trái dấu nhau.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Thi vào 10, Sở GD Bến Tre, 2017]%[9H3K7]%[9H1K1]%[9D3G3]
Cho đường tròn $O$, đường kính $AB$. Trên tiếp tuyến của đường tròn $(O)$ tại $A$ lấy điểm $M$ ($M$ khác $A$). Từ $M$ vẽ tiếp tuyến thứ hai $MC$ với đường tròn $(O)$ ($C$ là tiếp điểm). Kẻ $CH\perp AB$ ($H\in AB$), $MB$ cắt đường tròn $(O)$ tại điểm thứ hai là $K$ và cắt $CH$ tại $N$. Chứng minh rằng
\begin{enumerate}
\item Tứ giác $AKNH$ nội tiếp trong một đường tròn.
\item $AM^2=MK.MB$.
\item $\widehat{KAC}=\widehat{OMB}$.
\item $N$ là trung điểm của $CK$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1.5]
\tkzInit[xmin=-2,xmax=5,ymin=-2,ymax=5]
\tkzClip
\tkzDefPoints{0/0/A, 4/0/B, 0/3/M}
\tkzDefMidPoint(A,B) \tkzGetPoint{O}
\tkzDrawCircle[radius](O,B)
\tkzTangent[from=M](O,B)\tkzGetPoints{A}{C}
\tkzDrawAltitude(A,B)(C)\tkzGetPoint{H}
\tkzInterLC(M,B)(O,B)\tkzGetPoints{K}{B}
\tkzInterLL(M,B)(C,H)\tkzGetPoint{N}
\tkzInterLL(M,O)(A,C)\tkzGetPoint{I}
\tkzMarkRightAngles(M,A,O C,H,B A,K,B)
\tkzDrawPoints(O,A,B,C,M,H,K,N,I)
\tkzDrawSegments(A,B A,M C,M M,B C,O A,C M,O A,K)
\tkzLabelPoints[right](B,N)
\tkzLabelPoints[above](M,C,K)
\tkzLabelPoints[below](O,H,I)
\tkzLabelPoints[left](A)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Có $\widehat{AKB}=90^\circ$ (góc nội tiếp chắn nửa đường tròn), $\widehat{AHN}=90^\circ$ (vì $CH\perp AB$)\\
$\Rightarrow \widehat{AKB}+\widehat{AHN}=90^\circ$. Vậy tứ giác $AKNH$ nội tiếp đường tròn.
\item Vì tam giác $ABM$ vuông tại $A$ có $AK\perp MB$ nên theo hệ thức lượng trong tam giác vuông ta có $AM^2=MK.MB$.
\item Gọi $I$ là giao điểm của $AC$ và $OM$. Có $MA=MC$ (tính chất hai tiếp tuyến cắt nhau) và $OA=OC=R$.\\
$\Rightarrow OM$ là đường trung trực của $AC\Rightarrow OM\perp AC$.\\
Khi đó tứ giác $AMKI$ nội tiếp đường tròn đường kính $MA$. Trong đường tròn đường kính $MA$ có $\widehat{KAI}=\widehat{KMI}$ (nội tiếp cùng chắn cung $\wideparen{IK}$) do đó $\widehat{KAC}=\widehat{OMB}$.
\item Có $\widehat{ACB}=90^\circ$ (góc nội tiếp chắn nửa đường tròn) $\Rightarrow BC\perp AC$ mà $OM\perp AC\Rightarrow OM\parallel BC\Rightarrow \widehat{AOM}=\widehat{HBC}$ (so le trong).\\
Xét tam giác $ \triangle AOM$ và $\triangle HBC$ có $\widehat{AOM}=\widehat{HBC}$ và $\widehat{OAM}=\widehat{BHC}=90^\circ$\\ 
$\Rightarrow \triangle AOM\backsim\triangle HBC$ (g.g)
$\Rightarrow \dfrac{AM}{HC}=\dfrac{OA}{BH}\Rightarrow HC=\dfrac{AM.HB}{OA}=2.\dfrac{AM.BH}{AB}$ $(1)$\\
Có $MA\perp AB$ và $CH\perp AB\Rightarrow CH\parallel MA$.\\
Tam giác $ABM$ có $CH\parallel MA\Rightarrow \dfrac{BH}{BA}=\dfrac{HN}{AM}$ (hệ quả của định lý ta-lét)\\ $\Rightarrow HN=\dfrac{AM.BH}{AB}$ $(2)$\\
Từ $(1)$ và $(2)\Rightarrow HC=2.HN\Rightarrow HN=\dfrac{HC}{2}$\\
$\Rightarrow N$ là trung điểm của $CH$.
\end{enumerate}
}
\end{ex}
