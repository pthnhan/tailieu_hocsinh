\begin{ex}%[Thi GHK1, Nguyễn Huệ - Hà Nội, 2018]%[Trần Tuấn Việt, EX-DT-10&11]
%[0D1B1-2]
Chọn mệnh đề đúng trong các mệnh đề sau:
	\choice
	{$``\forall x \in \mathbb{N}: x^2 - 2 = 0"$}
	{$``\exists x \in \mathbb{Q}: \left| x \right| = \sqrt{2}"$}
	{\True $``\forall a, b \in \mathbb{N}: ab(a + b) \vdots  2"$}
	{$``\exists x \in \mathbb{N}: x^2 < x"$}
	\loigiai{Xét mệnh đề $``\forall a, b \in \mathbb{N}: ab(a + b) \vdots  2"$ trong các trường hợp sau:
	\begin{itemize}
	\item TH1: Nếu $a, b$ cùng chẵn thì $ab(a + b) \vdots  2$.
	\item TH2: Nếu $a$ chẵn, $b$ lẻ thì $ab \vdots 2$. Do đó $ab(a + b) \vdots  2$.
	\item TH3: Nếu $a, b$ cùng lẻ thì $(a + b) \vdots 2$. Do đó $ab(a + b) \vdots  2$.
\end{itemize}
Suy ra, $``\forall a, b \in \mathbb{N}: ab(a + b) \vdots  2"$ là mệnh đề đúng.
	}
\end{ex}

\begin{multicols}{2}
\begin{enumerate}

\item $y = \dfrac{2}{x^2 - 4} - \sqrt{x + 1}$

\loigiai{
\begin{enumerate}

\item Điều kiện xác định: $\heva{& x^2 - 4 \neq 0\\& x + 1 \geq 0} \Leftrightarrow \heva{& x \neq 2\\& x \neq -2 \\& x \geq -1} \Leftrightarrow \heva{& x \geq -1\\& x \neq 2}$.\\
Suy ra, tập xác định của hàm số là $\mathscr{D} = \left[ -1; +\infty\right) \setminus \left\lbrace  2 \right\rbrace$.
\end{enumerate}}
\end{ex}

Cho hàm số $y = ax^2 + bx + 3$.
\begin{enumerate}
\item Lập bảng biến thiên và vẽ đồ thị của hàm số khi $a = -1, b = -2$.
\item Tìm $a, b$ để đồ thị của hàm số đi qua điểm $A(-2; 3), B(1; 2)$.
\end{enumerate}
\loigiai{\begin{enumerate}
\item Hàm số $y = -x^2 - 2x + 3$ đồng biến trên $\left( -\infty; -1\right)$, nghịch biến trên $\left( -1; + \infty\right)$ và đạt GTLN tại điểm $(-1; 4)$.\\
Bên dưới là bảng biến thiên của hàm số $y = -x^2 - 2x + 3$.

Cho tam giác $ABC$ có trung tuyến $AM$, $I$ là trung điểm của $AM$, $G$ là trọng tâm của tam giác.
\begin{enumerate}
\item Biểu diễn véc-tơ $\overrightarrow{GM}$ thông qua véc-tơ $\overrightarrow{BA}$ và $\overrightarrow{CA}$.
\item Chứng minh rằng $2\overrightarrow{IA} + \overrightarrow{IB} + \overrightarrow{IC} = \overrightarrow{0}$.
\end{enumerate}
\loigiai{\begin{enumerate}
\item Ta có: $\overrightarrow{GM} = \dfrac{1}{3} \overrightarrow{AM}$ và $2\overrightarrow{AM} = \overrightarrow{AB} + \overrightarrow{AC}.$\\
Suy ra $$6\overrightarrow{GM} = \overrightarrow{AB} + \overrightarrow{AC}.$$
Vậy $\overrightarrow{GM} = -\dfrac{1}{6} \overrightarrow{BA} - \dfrac{1}{6} \overrightarrow{CA}$.
\item Ta có $\overrightarrow{IB} + \overrightarrow{IC} = 2\overrightarrow{IM}$. Lại có $I$ là trung điểm của $AM$ nên $\overrightarrow{IM} = -\overrightarrow{IA}$. \\
Suy ra, ta có hệ thức $\overrightarrow{IB} + \overrightarrow{IC} = -2\overrightarrow{IA}$ hay $2\overrightarrow{IA} + \overrightarrow{IB} + \overrightarrow{IC} = \overrightarrow{0}$.
\end{enumerate}
}
\end{ex}
