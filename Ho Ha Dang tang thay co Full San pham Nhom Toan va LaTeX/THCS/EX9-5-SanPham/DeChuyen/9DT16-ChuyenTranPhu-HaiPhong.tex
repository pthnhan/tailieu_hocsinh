\begin{ex}%[Đề thi vào 10, Chuyên Trần Phú, Hải Phòng, 2016]%[9D1B8]%[9D4K6]
\hfill
\begin{enumerate}
\item Cho biểu thức $P=\dfrac{\sqrt{a^3}-\sqrt{b^3}}{a-b}-\dfrac{a}{\sqrt{a}+\sqrt{b}}-\dfrac{b}{\sqrt{b}-\sqrt{a}}$ với $a$, $b$ là các số dương khác nhau. Thu gọn rồi tính giá trị của biểu thức $P$ biết $(a-1)(b-1)+2\sqrt{ab}=1$.
\item Cho phương trình $x^2-x+b=0$ có các nghiệm là $x_1$, $x_2$ và phương trình $x^2-97x+a=0$ có các nghiệm là $x_1^4$, $x_2^4$. Tìm giá trị của $a$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có
\begin{align*}
P&=\dfrac{\sqrt{a^3}-\sqrt{b^3}}{a-b}+\dfrac{-a\sqrt{b}+a\sqrt{a}-b\sqrt{a}-b\sqrt{b}}{b-a}\\
&=\dfrac{\sqrt{a^3}-\sqrt{b^3}}{a-b}+\dfrac{\sqrt{a^3}-\sqrt{b^3}}{b-a}+\dfrac{a\sqrt{b}+b\sqrt{a}}{a-b}\\
&=\dfrac{\sqrt{ab}\left(\sqrt{a} + \sqrt{b} \right)}{\left( \sqrt{a}-\sqrt{b}\right)(\sqrt{a} +\sqrt{b})}= \dfrac{\sqrt{ab}}{\sqrt{a}-\sqrt{b}}.
\end{align*}
Theo giả thiết ta $ab-a+2\sqrt{ab}-b=0 \Leftrightarrow ab=\left( \sqrt{a}- \sqrt{b}\right)^2 \Leftrightarrow \sqrt{ab}=\left|
\sqrt{a} -\sqrt{b} \right|$.\\
Từ đó $P=\begin{cases} 1 &\text{nếu}\ a>b\\ -1 &\text{nếu}\ a<b\end{cases}$.

\item Theo định lí Vi\`ete ta có
\begin{align*}
\heva{&x_1+x_2=1\\ &x_1x_2=b}\qquad\text{và}\qquad \heva{&x_1^4+x_2^4=97\\ &x_1^4x_2^4=a}
\end{align*}
Ta có
\begin{align*}
&97=x_1^4+x_2^4=(x_1^2+x_2^2)^2-2x_1^2x_2^2\\
=&\left[(x_1+x_2)^2-2x_1x_2\right]^2-2x_1^2 x_2^2=(2b-1)^2-2b^2\\
=&2b^2-4b+1.
\end{align*}
Suy ra $b=8$ hoặc $b=-6$.\\
Với $b=8$ thì phương trình $x^2-x+b=0$ vô nghiệm.\\
Với $b=-6$ thì phương trình $x^2-x+b=0$ có hai nghiệm phân biệt, khi đó, $a=b^4=1296$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Trần Phú, Hải Phòng, 2016]%[9D4K7]%[9D3G4]
\hfill
\begin{enumerate}
\item Giải phương trình $(9x^2-18x+5)(3x^2-4x)-7=0$.
\item Giải hệ phương trình $\heva{&\sqrt{2x+3y}+\sqrt{2x-3y}=3\sqrt{2y}\\ &2\sqrt{2x+3y}-\sqrt{2x-3y}=6}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Phương trình đã cho tương đương
\begin{align*}
&3x(3x-5)(3x-1)(3x-4)-21=0\\
\Leftrightarrow &(9x^2-15x)(9x^2-15x+4)-21=0.
\end{align*}
Đặt $t=9x^2-15x+2$ phương trình trở thành $t^2-4-21=0$ hay $t=\pm 5$.\\
Với $t=5$ ta có $9x^2-15x-3=0\Leftrightarrow x=\dfrac{5\pm \sqrt{37}}{3}$.\\
Với $t=-5$, ta có $9x^2-15x+7=0$, phương trình vô nghiệm.\\
Vậy phương trình đã cho có hai nghiệm $x_{1,2}=\dfrac{5\pm \sqrt{37}}{3}$.
\item Điều kiện của phương trình là $\heva{&2x+3y\ge 0\\ &2x-3y\ge 0\\ &y\ge 0}$.\\
Đặt $a=\sqrt{2x+3y}\ge 0$, $\sqrt{2x-3y}\ge 0$. Khi đó ta có
\begin{align*}
\begin{cases}a+b=3\sqrt{2y}&(1)\\ 2a-b=6 &(2)\end{cases}.
\end{align*}
Ta có $(a-b)(a+b)=a^2-b^2=6y$, suy ta $3\sqrt{2y}(a-b)=6y$. Từ hệ phương trình ta thấy $y\ne 0$, suy ra $a-b=\sqrt{2y}$, suy ra $a-b=\dfrac{1}{3}(a+b)$ hay $a=2b$. Thay vào $(2)$ ta được $b=2$, suy ra $a=4$. Từ đó
\begin{align*}
\heva{&2x+3y=16\\ &2x-3y=4}\Leftrightarrow \heva{&x=5\\ &y=2}.
\end{align*}
Vậy hệ phương trình đã cho có nghiệm là $(5;2)$. 
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Trần Phú, Hải Phòng, 2016]%[9H3K7]%[9H3G7]
Cho tam giác nhọn $ABC$ nội tiếp đường tròn $(O)$, $AB<AC$, các đường cao $BD$, $CE$ cắt nhau tại $H$ ($D$ thuộc $AC$, $E$ thuộc $AB$). Gọi $M$ là trung điểm của $BC$, tia $MH$ cắt đường tròn $(O)$ tại $N$.
\begin{enumerate}
\item Chứng minh rằng năm điểm $A$, $D$, $H$, $E$, $N$ cùng nằm trên một đường tròn.
\item Lấy điểm $P$ trên đoạn $BC$ sao cho $\widehat{BHP}=\widehat{CHM}$, $Q$ là hình chiếu vuông góc của $A$ trên đường thẳng $HP$. Chứng minh rằng tứ giác $DENQ$ là hình thang cân.\
\item Chứng minh rằng đường tròn ngoại tiếp tam giác $MPQ$ tiếp xúc với đường tròn $(O)$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1.5]
\tkzDefPoints{0/4/A, -1/0/B, 4/0/C}
\tkzDrawAltitude(A,C)(B)\tkzGetPoint{D}
\tkzDrawAltitude(A,B)(C)\tkzGetPoint{E}
\tkzInterLL(B,D)(C,E)\tkzGetPoint{H}
\tkzDefMidPoint(B,C)\tkzGetPoint{M}
\tkzDefCircle[circum](A,B,C)\tkzGetPoint{O}
\tkzInterLC(M,H)(O,A)\tkzGetPoints{I}{N}
\tkzFindAngle(M,H,C)\tkzGetAngle{ang}
\tkzDefPointBy[rotation=center H angle \ang](B)\tkzGetPoint{x}
\tkzInterLL(H,x)(B,C) \tkzGetPoint{P}
\tkzDrawAltitude(P,H)(A)\tkzGetPoint{Q}
\tkzDrawCircle[circum, dashed](M,P,Q)\tkzGetPoint{J}
\tkzLabelPoints[above](A,J)
\tkzLabelPoints[below left](B,E)
\tkzLabelPoints[below right](C,I)
\tkzLabelPoints[left](N)
%\tkzLabelPoints[right](H)
\tkzLabelPoints[above right](D,O,Q)
\tkzLabelPoints[below](M,P,H)


\tkzMarkRightAngles(B,D,C C,E,B A,Q,P)
\tkzMarkAngles[size=0.91cm](B,H,P M,H,C)

\tkzDrawCircle(O,A)

\tkzDrawSegments(A,B B,C C,A B,D C,E N,I A,I A,H N,E A,N B,I C,I D,Q N,Q D,E P,Q O,N M,J)

\tkzDrawPoints[fill=white, size=8](A,B,C,D,E,O,H,I,N,M,P,Q,J)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Gọi $I$ là điểm đối xứng với $A$ qua $O$. Khi đó $\heva{&IB\parallel CE\ (\text{cùng vuông góc với}\ AB)\\  &IC\parallel BD\  \text{(cùng vuông góc với}\ AC)}$, suy ra $HBIC$ là hình bình hành, do đó $H$, $I$, $M$ thẳng hàng, kéo theo $\widehat{ANH}=90^\circ$. Mặt khác $\widehat{AEH}=\widehat{ADH}=90^\circ$ nên năm điểm $A$, $D$, $E$, $H$, $N$ cùng nằm trên đường tròn đường kính $AH$.
\item Dễ thấy $Q$ cũng thuộc đường tròn đường kính $AH$ và $\widehat{QHD}=\widehat{BHP}=\widehat{MHC}=\widehat{NHE}$ nên $NE=QD$. Hai dây cung $NE$ và $QD$ bằng nhau nên tứ giác $NQDE$ là hình thang cân.

\item Gọi $J$ là tâm đường tròn ngoại tiếp tam giác $MPQ$.\\
Hai góc $\widehat{NAH}$ và $\widehat{NMP}$ có các cạnh đôi một vuông góc nên $\widehat{NMP}=\widehat{NAH}$. Mà $\widehat{NAH}=\widehat{NQH}$ nên $\widehat{NMP}=\widehat{NQP}$ hay tứ giác $MPNQ$ nội tiếp đường tròn $(J)$.\\
Ta có $\triangle MEN=\triangle MDQ$ (c.g.c) nên $MN=MQ$, do đó $MJ$ là đường trung trực của $NQ$ mà $NQ\parallel DE$ nên $MJ\perp DE$.\\
Lại có $\widehat{BAI}+\widehat{DEA}=\widehat{BCI}+\widehat{BCA}=90^\circ$ nên $AO\perp DE$. Suy ra $MJ\parallel OI$. Do đó $\widehat{JNM}=\widehat{JMN}=\widehat{OIN}=\widehat{ONI}$. Từ đó $N$, $I$, $O$ thẳng hàng.\\
Vậy đường tròn $(O)$ và đường tròn $(J)$ tiếp xúc với nhau tại $N$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Trần Phú, Hải Phòng, 2016]%[9D5G1]
Cho $a$, $b$, $c >0$, $a+b+c\ge 9$, tìm giá trị nhỏ nhất của biểu thức $$A=2\sqrt{a^2+\dfrac{b^2}{3}+\dfrac{c^2}{5}}+3\sqrt{\dfrac{1}{a}+\dfrac{9}{b}+\dfrac{25}{c}}.$$
\loigiai{
Áp dụng bất đẳng thức Cauchy-Schwarz
\begin{align*}
&a^2+\dfrac{b^2}{3}+\dfrac{c^2}{5}\ge \dfrac{(a+b+c)^2}{9}\\
&\dfrac{1}{a}+\dfrac{9}{b}+\dfrac{25}{c}\ge \dfrac{(1+3+5)^2}{a+b+c}=\dfrac{81}{a+b+c}.
\end{align*}
Suy ra
\begin{align*}
A&\ge 2\cdot\dfrac{a+b+c}{3}+3\cdot\dfrac{9}{\sqrt{a+b+c}}\\
&\ge \dfrac{a+b+c}{3}+\dfrac{a+b+c}{3}+\dfrac{9}{\sqrt{a+b+c}}+\dfrac{9}{\sqrt{a+b+c}}+\dfrac{9}{\sqrt{a+b+c}}\\
&\ge 5\sqrt[5]{\dfrac{a+b+c}{3}\cdot\dfrac{a+b+c}{3}\cdot \dfrac{9}{\sqrt{a+b+c}}\cdot \dfrac{9}{\sqrt{a+b+c}}\cdot \dfrac{9}{\sqrt{a+b+c}}}\\
&\ge 5\sqrt[5]{81\sqrt{a+b+c}}\\
&\ge 15.
\end{align*}
Đẳng thức xảy ra khi $a=1$, $b=3$, $c=5$.\\
Vậy $\min A=15$.
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Trần Phú, Hải Phòng, 2016]%[9D5G2]%[9D5G5]
\hfill
\begin{enumerate}
\item Tìm tất cả các số nguyên $m \ge n \ge 0$ sao cho $(m+2n)^3$ là ước của $9n(m^2+mn+n^2)+16$.
\item Trong dãy $2016$ số thực $a_1$, $a_2$, $a_3$, \ldots, $a_{2016}$, ta đánh dấu tất cả các số dương và số mà có ít nhất một tổng của nó với một số các số liên tiếp ngay sau nó là một số dương (ví dụ trong dãy $-6$, $5$, $-3$, $3$, $1$, $-1$, $-2$, $-3$, \ldots, $-2011$ ta đánh dấu các số $a_2=5$, $a_3=-3$, $a_4=3$, $a_5=1$). Chứng minh rằng nếu trong dãy đã cho có ít nhất một số dương thì tổng tất cả các số được đánh dấu là một số dương.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Từ giả thiết suy ra
\begin{align*}
0\le 9n(m^2+mn+n^2)+16-(m+2n)^3=16-(m-n)^3.
\end{align*}
Đặt $A=16-(m-n)^3$. Do $m\ge n$ nên ta có ba trường hợp sau:\\
\textit{Trường hợp 1.} $m=n$.\\
Khi đó $A\,\vdots\, (m+2n)^3$, suy ra $16\,\vdots\, 27n^3$. Không tồn tại $n$ thoả mãn.\\
\textit{Trường hợp 2.} $m=n+1$.\\
Khi đó $A\,\vdots\, (m+2n)^3$, suy ra $15\,\vdots\, (3n+1)^3$. Từ đó, ta có $n=0$ và $m=1$.\\
\textit{Trường hợp 3.} $m=n+2$.\\
Khi đó $8\,\vdots\,(3n+2)^3$, suy ra $n=0$ và $m=2$.\\
Vậy các cặp $(m,n)$ cần tìm là $(1,0)$ và $(2,0)$.
\item Nếu các số được đánh dấu gồm toàn số không âm thì bài toán hiển nhiên. Giả sử có ít nhất một số được đánh dấu là số âm, kí hiệu là $a_i$. Gọi $j$ là chỉ số nhỏ nhất ($j>i$) sao cho $A=a_i+a_{i+1}+a_{i+2}+\cdots +a_j>0$ và $a_j>0$. Khi đó với mọi $i\le k<j$ ta đều có $A_k=a_i+a_{i+1}+\cdots +a_k\le 0$.\\
Suy ra $a_{k+1}+a_{k+2}+\cdots + a_j=A-A_k>0$ hay $a_{k+1}$ cũng là số được đánh dấu. Do đó các số $a_i$, $a_{i+1}$, \ldots, $a_{j}$ đều được đánh dấu.\\
Vậy tổng tất cả các số được đánh dấu phải là một số dương.
\end{enumerate}
}
\end{ex}
