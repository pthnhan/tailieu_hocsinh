	\begin{eqnarray*}
	\dfrac{f'(x)}{f(x)}=\tan x	&\Rightarrow & \displaystyle\int{\dfrac{f'(x)}{f(x)}}\mathrm{\,d}x=\displaystyle\int\tan x\mathrm{\,d}x\\
	&\Rightarrow & \displaystyle\int{\dfrac{f'(x)}{f(x)}}\mathrm{\,d}x=\displaystyle\int\dfrac{\sin x}{\cos x}\mathrm{\,d}x\\
	&\Rightarrow & \ln f(x)=-\ln\left(\cos x\right)+C.
