\item Hệ phương trình bậc cao hai ẩn

\end{itemize}

\item Phương pháp đặt ẩn phụ

\end{itemize}

\textit{Bài toán}


Anh Việt đi từ $A$ và đã đến $B$ gặp bạn đúng hẹn. Anh nói với bạn rằng:

Nếu tôi đi với vận tốc ít hơn vận tốc đã đi $6$km/h thì sẽ đến $B$ trước hện $2$ giờ.

Bạn hãy tính thời gian anh Việt đã đi quãng đường
$AB$.

\textbf{Giải bài toán bằng cách lập hệ phương trình}

Gọi vận tốc anh Việt đã đi quãng đường $AB$ là $v$(km/h), thời gian đã đi quãng đường $AB$ là $t$(giờ).

Trong trường hợp thứ nhất, vận tốc là $v-6$(km/h), thời gian là $t+2$(giờ).

Ta có phương trình : $(v-6)(t+2)=vt$.

Trong trường hợp thứ hai, vận tốc là $v+10$(km/h), thời gian là $t-2$(giờ)

Ta có phương trình: $(v+10)(t-2)=vt$.

Giải hệ phương trình
\begin{align*}
\heva{(v-6)(t+2)&=vt\\(v+10)(t-2)&=vt} \Leftrightarrow \heva{vt+2v-6t-12&=vt\\vt-2v+10t-20&=vt} \Leftrightarrow \heva{2v-6t&=12\\-2v+10t&=20}\Leftrightarrow \heva{4t&=32\\v-3t&=6}\Leftrightarrow \heva{v&=30\\t&=8}
\end{align*}

Thời gian anh Việt đi quãng đường $AB$ là $8$ giờ.

\subsection{Hệ phương trình bậc nhất hai ẩn}
\begin{vd}
Cho hệ phương trình $\heva{x+my =1&&(1)\\ 2mx+m(m-1)y=3.&&(2)}$

Tìm giá trị của $m$ đề hệ phương trình
\begin{enumerate}
\item Có nghiệm duy nhất
\item Vô nghiệm
\end{enumerate}
    \loigiai{
        \begin{enumerate}
        \item Với $m=0$ thì $(2)$ là $0x+0y=3$, vô nghiệm.

        Với $m \neq 0$, điều kiện để hệ có nghiệm duy nhất là $$\dfrac{2m}{1} \neq \dfrac{m(m-1)}{m} \Leftrightarrow 2m \neq m-1 \Leftrightarrow m \neq -1.$$

        Vậy giá trị của $m$ để hệ có nghiệm duy nhất là $m \neq 0$ và $m \neq -1$.
        \item Với $m=0$ thì hệ phương trình vô nghiệm

        Với $m \neq$, điều kiện để hệ vô nghiệm là $$\dfrac{2m}{1}=\dfrac{m(m-1)}{m} \neq \dfrac{1}{3} \Leftrightarrow 2m = m-1 \neq \dfrac{1}{3} \Leftrightarrow m = -1.$$

        Vậy giá trị của $m$ để hệ vô nghiệm là $m=0$ hoặc $m=-1$.
\end{enumerate}
Lưu ý: Có thể giải bằng cách rút $x$ từ $(1)$ rồi thay vào $(2)$ và rút gọn được $m(m+1)y=2m-3.$

Với $m \neq 0$ và $n \neq -1$ thì hệ có nghiệm duy nhất.

Với $m=0$ hoặc $m=-1$ thì hệ vô nghiệm
    }
\end{vd}
\begin{vd}
Cho hệ phương trình $\heva{2x+y&=3m&&(1)\\ mx-(m+1)y&=2m+2.&&(2) }$

Tìm giá trị của $m$ để hệ phương trình có nghiệm $(x;y)$ duy nhất thỏa mãn tích $xy$ nhỏ nhất.
\loigiai{

Rút $y$ từ $(1)$ được $y=3m-2x$. Thay vào $(2)$ được
\begin{align*}
&mx-(m+1)(3m-2x)=2m+2\\
\Leftrightarrow & mx -3m(m+1)+(2m+2)x=2m+2\\
\Leftrightarrow & (3m+2)x = 2m +2 + 3m^2+3m\\
\Leftrightarrow & (3m+2)x=(3m+2)(m+1)
\end{align*}

Nếu $m \neq - \dfrac{2}{3}$ thì hệ phương trình có nghiệm duy nhất $\heva{x&=m+1\\y&=m-2}$

Khi đó $xy=(m+1)(m-2) = m^2-m-2=\left(m-\dfrac{1}{2}	 \right)^2- \dfrac{9}{4} \geq 0.$

$\min (xy)= - \dfrac{9}{4}$ tại $m=\dfrac{1}{2}$ (thỏa mãn $m \neq - \dfrac{2}{3}$).
}
\end{vd}
\subsection{Hệ phương trình bậc cao hai ẩn}

Hệ phương trình bậc cao hai ẩn không được học chính thức trong chương trình Đại số $9$, nhưng với kiến thức về hệ phương trình (bậc nhất hai ẩn) và phương trình bậc hai một ẩn, ta có thể giải được hệ phương trình bậc cao hai ẩn.

Các phương pháp thường dùng để giải hệ phương trình bậc cao hai ẩn là phương pháp thế, phương pháp cộng đại số, phương pháp đặt ẩn phụ, phương pháp dùng hằng đẳng thức.
\subsubsection{Phương pháp thế}

Trong phương pháp thế, từ một phương trình ta biểu thị một ẩn theo ẩn kia (hoặc tìm giá trị của một ẩn) rồi thế vào phương trình còn lại.
\begin{vd}
Giải hệ phương trình $\heva{xy-3x-y&=1&& (1)\\ x^3-y^3+xy^2-x^2y&=0.&& (2)}$
\loigiai{
$(2)\Leftrightarrow (x-y)(x^2+y^2)=0\Leftrightarrow \hoac{x&=y\\x&=y=0.}$

Loại $x=y=0$ vì không thỏa mãn $(1)$.

Thay $y=x$ vào $(1)$ được $x^2-4x-1=0 \Leftrightarrow x= 2 \pm \sqrt{5}.$

Đáp số: Nghiệm $(x;y)$ của hệ là $(2+\sqrt{5};2+\sqrt{5}),(2-\sqrt{5};2-\sqrt{5}) $
}
\end{vd}
\begin{vd}
Giải hệ phương trình $\heva{x+y&=1 \ \ &(1) \\ x^4 - y^4 &=7x-y \ \ &(2)}$
\loigiai{
Thay $x+y=1$ vào $(2)$, ta được
\begin{align*}
&(x-y)(x^2+y^2)=7x-y\\
\Leftrightarrow & (x-y)(1-2xy)=7x-y\\
\Leftrightarrow & 2x(y^2-xy-3)=0\\
\Leftrightarrow & \hoac{x&=0 \\ y^2-xy-3&=0 \ \ (3)}
\end{align*}
Với $x=0$ thì $y=1.$\\
Với $x \neq 0$, thay $y$ bởi $1-x$ vào $(3)$ được $$2x^2-3x-2=0 \Leftrightarrow \hoac{x&=2 \\ x&= - \dfrac{1}{2}.}$$
Đáp số: Nghiệm $(x;y)$ của hệ là $(0;1)$, $(2;-1)$, $\left( - \dfrac{1}{2} ; \dfrac{3}{2}\right).$
}
\end{vd}
\subsubsection{Phương pháp cộng}

Trong phương pháp cộng, ta cộng hoặc trừ từng vế hai phương trình để khử một ẩn hoặc tìm mối quan hệ giữa hai ẩn.
\begin{vd}
Giải hệ phương trình $\heva{x^3-1&=2y \ \ &(1)\\ y^3 - 1 &= 2x \ \ &(2)}$
\loigiai{

Lấy $(1)$ trừ $(2)$ được $x^3-y^3=2(y-x)\Leftrightarrow (x-y)(x^2+xy+y^2+2)=0.$

Do $x^2+xy+y^2= \left( x+ \dfrac{y}{2}\right)^2+\dfrac{3y^2}{4}+2 >0$ nên $x=y.$

Thay $y=x$ vào $(1)$ được
\begin{align*}
&x^3-1=2x\\
\Leftrightarrow & x^3 + 1 -2x-2=0\\
\Leftrightarrow & (x+1)(x^2-x-1)=0\\
\Leftrightarrow & \hoac{x+1&=0\\x^2-x-1&=0}\\
\Leftrightarrow & \hoac{x&=-1\\x&=\dfrac{1 \pm \sqrt{5}}{2}}
\end{align*}
Đáp số: Nghiệm $(x;y)$ của hệ là $(-1;-1)$, $\left(\dfrac{1+\sqrt{5}}{2}; (\dfrac{1+\sqrt{5}}{2} \right)$, $\left( \dfrac{1-\sqrt{5}}{2};\dfrac{1-\sqrt{5}}{2}\right).$
}
\end{vd}

Lưu ý: Trong hệ phương trình trên, khi thay $x$ bởi $y$, thay $y$ bởi $x$ thì phương trình $(1)$ trở thành phương trình $(2)$,phương trình $(2)$ trở thành phương trình $(1)$. Ta gọi hệ phương trình trên là \textit{hệ đối xứng loại II}.

Để giải hệ phương trình đối xứng loại II, ta thường trừ từng vế hai phương trình và nhận được phương trình tích.

\subsubsection{Phương pháp đặt ẩn phụ}
\begin{vd}
Giải hệ phương trình $\heva{&x+y-xy=-2\\&x^2+y^2=20}$
\loigiai{
Đặt $x+y=a, xy=b$ ta có\\
$\heva{&(x+y)-xy=-2\\&(x+y)^2-2xy=20}\Leftrightarrow \heva{&a-b=-2&(1)\\&a^2-2b=20&(2)}$\\
Từ $(1)$ và $(2)$ suy ra: $a^2-2(a+2)=20 \Leftrightarrow a^2-2a-24=0 \Leftrightarrow \hoac{&a=6\\&a=-4}$.\\
\begin{itemize}
  \item Với $a=6$ thì $b=8$. Ta có $x$ và $y$ là nghiệm của phương trình $X^2-6X+8=0$.\\
  Giải phương trình ta được $X\in\{2;4\}$, khi đó $(x,y)$ là $(2;4), (4;2)$.
  \item Với $a=-4$ thì $b=-2$. Ta có $x$ và $y$ là nghiệm của phương trình $X^2+4X-2=0$.\\
  Giải phương trình ta được $X=-2\pm\sqrt{6}$, khi đó $(x,y)$ là $(-2+\sqrt{6};-2-\sqrt{6}), (-2-\sqrt{6};-2+\sqrt{6})$.
\end{itemize}
\emph{Đáp số}: Nghiệm $(x;y)$ là $(2;4), (4;2)$, $(-2+\sqrt{6};-2-\sqrt{6}), (-2-\sqrt{6};-2+\sqrt{6})$.
}
\end{vd}
\begin{note}
Trong hệ phương trình trên, khi thay $x$ bởi $y$ và thay $y$ bởi $x$ thì mỗi phương trình của hệ không đổi. Ta gọi hệ phương trình trên là \emph{hệ đối xứng loại 1}.\\
Để giải hệ phương trình đối xứng loại I, ta thường đặt ẩn phụ $x+y=a$ và $xy=b$ (giữa $a$ và $b$ có mối quan hệ $a^2\geq 4b$).
\end{note}
\begin{vd}
Giải hệ phương trình $\heva{&x^2+y^2=1\\&x^3+y^3=1}$
\loigiai{
\emph{Cách 1}:
Đặt $x+y=a, xy=b$ ta có\\
$$\heva{&x^2+y^2=1\\&x^3+y^3=1}\Leftrightarrow \heva{&(x+y)^2-2xy=1\\&(x+y)^3-3xy(x+y)=1}\Leftrightarrow \heva{&a^2-2b=1&(1)\\&a^3-3ab=1&(2)}$$
Từ (1) ta có $b=\dfrac{a^2-1}{2}$. Thay vào $(2)$ được\\
$a^3-\dfrac{3a(a^2-1)}{2}=1\Leftrightarrow a^3-3a+2=0 \Leftrightarrow (a+2)(a-1)^2=0 \Leftrightarrow \hoac{&a=-2\\&a=1}.$
\begin{itemize}
  \item Với $a=1$ thì $b=0$. Ta có $x$ và $y$ là nghiệm của phương trình $X^2-X=0 \Leftrightarrow \hoac{&X=0\\&X=1}$. Khi đó $(x;y)$ là $(0;1), (1;0)$.
  \item Với $a=-2$ thì $b=\dfrac{3}{2}$. Ta có $x$ và $y$ là nghiệm của phương trình $X^2+2X+\dfrac{3}{2}=0$. Phương trình vô nghiệm (có thể loại $a=-2$ và $b=\dfrac{3}{2}$ vì $a^2<4b$.\\
\end{itemize}
\emph{Đáp số}: Nghiệm $(x;y)$ là $(0;1), (1;0)$.\\
\emph{Cách 2}: Trừ từng vế hai phương trình đã cho ta được\\
$x^3+y^3-x^2-y^2=0 \Leftrightarrow x^2(x-1)+y^2(y-1)=0$\hfill (3)\\
Do $x^2+y^2=1$ nên $x^2\leq 1$, suy ra $x\leq 1$. Tương tự, $y\leq 1$.\\
Từ đó suy ra $x^2(x-1)\leq 0; y^2(y-1)\leq 0$.\\
Kết hợp với $(3)$ suy ra $x^2(x-1)=y^2(y-1)= 0$, do đó $x=0, y=1$ hoặc $x=1;y=0$.
}
\end{vd}
\begin{vd}
Giải hệ phương trình $\heva{&\dfrac{1}{y}-\dfrac{1}{x}=\dfrac{5}{12}\\&x^2+y^2=1}$
\loigiai{
$$\heva{&\dfrac{1}{y}-\dfrac{1}{x}=\dfrac{5}{12}\\&x^2+y^2=1}\Leftrightarrow \heva{&\dfrac{x-y}{xy}=\dfrac{5}{12}&(1)\\&(x-y)^2+2xy=1&(2)}$$
Đặt $xy=a$, từ $(1)$ ta có $x-y=\dfrac{5a}{12}$.\\
Thay vào $(2)$ ta được $\left(\dfrac{5a}{12}\right)^2+2a=1 \Leftrightarrow 25a^2+288a-144=0 \Leftrightarrow \hoac{&a=\dfrac{12}{25}\\&a=-12}$.
\begin{itemize}
  \item Với $a=\dfrac{12}{25}$ thì $x-y=\dfrac{5}{12}\cdot \dfrac{12}{25}=\dfrac{1}{5}$.\\
   Ta có $(x+y)^2=x^2+y^2+2xy=1+2a=1+\dfrac{24}{25}=\dfrac{49}{25}=\left(\dfrac{7}{5}\right)^2$.\\
   Từ $\heva{&x+y=\dfrac{7}{5}\\&x-y=\dfrac{1}{5}}$ ta được $\heva{&x=\dfrac{4}{5}\\&y=\dfrac{3}{5}}$\\
   Từ $\heva{&x+y=-\dfrac{7}{5}\\&x-y=\dfrac{1}{5}}$ ta được $\heva{&x=-\dfrac{3}{5}\\&y=-\dfrac{4}{5}}$\\
  \item Với $a=-12$ thì $(x+y)^2=x^2+y^2+2xy=1-24<0$, loại.\\
\end{itemize}
\emph{Đáp số}: Nghiệm $(x;y)$ là $\left(\dfrac{4}{5}; \dfrac{3}{5}\right), \left(-\dfrac{3}{5}; -\dfrac{4}{5}\right)$.
}
\end{vd}

\begin{vd}
Giải hệ phương trình $\heva{&x^2-xy+y^2=3&(1)\\&2x^2-xy+3y^2=12&(2)}$
\loigiai{Nhân $(1)$ với $4$ rồi trừ đi $(2)$ theo từng vế ta được
\[4(x^2-xy+y^2)-(2x^2-xy+3y^2)=0\Leftrightarrow 2x^2-3xy+y^2=0.\tag{3}\]
Chia hai vế của $(3)$ cho $y^2\neq 0$ được $2\left(\dfrac{x}{y}\right)^2-3\dfrac{x}{y}+1=0$.\\
Đặt $\dfrac{x}{y}=k$, ta được $2k^2-3k+1=0 \Leftrightarrow \hoac{&k=1\\&k=\dfrac{1}{2}}$.
\begin{itemize}
  \item Với $k=1$ thì $y=x$, thay vào $(1)$ ta được $x^2-x^2+x^2=3\Rightarrow x^2=3 \Rightarrow x=\pm \sqrt{3}$.\\
  Khi đó $(x;y)$ là $(\sqrt{3};\sqrt{3}), (-\sqrt{3};-\sqrt{3})$.
  \item Với $k=\dfrac{1}{2}$ thì $y=2x$, thay vào $(1)$ ta được $x^2-2x^2+4x^2=3 \Leftrightarrow x^2=1 \Leftrightarrow x=\pm 1$.\\
  Khi đó $(x;y)$ là $(1;2), (-1;-2)$.
\end{itemize}
\emph{Đáp số}: Nghiệm $(x;y)$ là $(\sqrt{3};\sqrt{3}), (-\sqrt{3};-\sqrt{3}), (1;2), (-1;-2)$.
}
\end{vd}

\begin{note}
Trong hệ phương trình trên, mỗi phương trình đều có vế trái là các hạng tử bậc hai, vế phải là hằng số. Ta gọi hệ phương trình trên là hệ đẳng cấp bậc hai.
Để giải hệ phương trình đẳng cấp bậc hai, ta thường khử hẳng số rồi chia hai vế của phương trình cho $y^2\neq 0$ và đặt ẩn phụ $\dfrac{x}{y}=k$. Cũng có thể biến đổi $(3)$ thành $(x-y)(2x-y)=0$.
\end{note}


\subsubsection{Phương pháp dùng bất đẳng thức}
\begin{vd}
Giải hệ phương trình sau với $x, y$ không âm $\heva{&x^2+xy(x+y)=3&(1)\\&y^2+x+y-6xy=-3&(2)}$
\loigiai{
Cộng $(1)$ với $(2)$ ta được
\begin{align*}
x^2+y^2+xy(x+y)+(x+y)-6xy=0&\Leftrightarrow (x+y)^2+(x+y)(xy+1)=8xy\\
&\Leftrightarrow(x+y)(x+y+xy+1)=8xy \\
&\Leftrightarrow (x+y)(x+1)(y+1)=8xy\ \ \ \  (3)
\end{align*}
Do $x, y$ không âm nên $x+y\geq 2\sqrt{xy}, x+1\geq 2\sqrt{x}, y+1\geq 2\sqrt{y}$.\\
Từ đó ta có $(x+y)(x+1)(y+1)\geq 8xy$ \hfill (4)\\
Do $(3)$ nên ở $(4)$, phải xảy ra dấu đẳng thức, tức là $x=y=1$.\\
Nghiệm $(x;y)$ là $(1;1)$.
}
\end{vd}

\subsection{Hệ phương trình ba ẩn}

\begin{vd}
Giải hệ phương trình $\heva{&x^2=2x-y\\&y^2=2y-z\\&z^2=2z-x}$
\loigiai{
Ta có $\heva{&x^2=2x-y\\&y^2=2y-z\\&z^2=2z-x}\Leftrightarrow \heva{&(x-1)^2=1-y\\&(y-1)^2=1-z\\&(z-1)^2=1-x}$.\\
Đặt $1-x=a, 1-y=b, 1-z=c$, ta có\\
$\heva{&a^2=b\\&b^2=c\\&c^2=a}$ nên $a^8=b^4=c^2=a$, do đó $a^8-a=0 \Leftrightarrow a(a^7-1)=0 \Leftrightarrow \hoac{&a=0\\&a=1}$.\\
Vậy $a=b=c=0$ hoặc $a=b=c=1$.\\
Nghiệm $(x;y;z)$ là $(0;0;0)$ và $(1;1;1)$.
}
\end{vd}
\begin{vd}
Giải hệ phương trình $\heva{&(x+y)^2=4z-1&(1)\\&(y+z)^2=4x-1&(2)\\&(z+x)^2=4y-1&(3)}$
\loigiai{
Từ $(1)$ suy ra $4z-1\geq 0$ nên $z\geq \dfrac{1}{4}$. Tương tự $x\geq \dfrac{1}{4}$ và $y\geq \dfrac{1}{4}$.\\
Do $x,y,z\geq \dfrac{1}{4}$ nên từ hệ đã cho ta có
$$\heva{&x+y=\sqrt{4z-1}\\&y+z=\sqrt{4x-1}\\&z+x=\sqrt{4y-1}}$$
Cộng từng vế các phương trình ta được
\begin{align*}
&2x+2y+2z=\sqrt{4z-1}+\sqrt{4x-1}+\sqrt{4y-1}\\
\Leftrightarrow & 4x+4y+4z-2\sqrt{4z-1}-2\sqrt{4x-1}-2\sqrt{4y-1}=0\\
\Leftrightarrow & (1-\sqrt{4x-1})^2+(1-\sqrt{4y-1})^2+(1-\sqrt{4z-1})^2=0 \\
\Leftrightarrow & \sqrt{4x-1}=\sqrt{4y-1}=\sqrt{4z-1}=1\\
\Leftrightarrow & x=y=z=\dfrac{1}{2}.
\end{align*}
Nghiệm $(x;y;z)$ là $\left(\dfrac{1}{2};\dfrac{1}{2};\dfrac{1}{2}\right)$.
}
\end{vd}
\begin{vd}
Giải hệ phương trình $\heva{&2x^2=y(x^2+1)&(1)\\&2y^2=z(y^2+1)&(2)\\&2z^2=x(z^2+1)&(3)}$
\loigiai{
\emph{Cách 1}: Từ các phương trình đã cho, ta thấy $x, y, z\geq 0$. Nếu một trong ba số bằng $0$ thì hai số kia bằng $0$. Xét trường hợp $x, y, z>0$\\
$(1)\Leftrightarrow \dfrac{x^2+1}{x^2}=\dfrac{2}{y} \Leftrightarrow 1+\dfrac{1}{x^2}-\dfrac{2}{y}=0$\hfill (4)\\
Tương tự ta có $(2)\Leftrightarrow 1+\dfrac{1}{y^2}-\dfrac{2}{z}=0$\hfill (5)\\
$(3)\Leftrightarrow 1+\dfrac{1}{z^2}-\dfrac{2}{x}=0$\hfill (6)\\
Từ $(4), (5), (6)$ suy ra
\begin{align*}
&\left(1-\dfrac{1}{x}\right)^2+\left(1-\dfrac{1}{y}\right)^2+\left(1-\dfrac{1}{z}\right)^2=0\\
\Leftrightarrow & 1-\dfrac{1}{x}=1-\dfrac{1}{y}=1-\dfrac{1}{z}=0\\
\Leftrightarrow & x=y=z=1
\end{align*}
Thử vào hệ thấy thỏa mãn. Vậy nghiệm $(x;y;z)$ là $\left(0;0;0\right)$ và $\left(1;1;1\right)$.\\
\emph{Cách 2}: Từ các phương trình đã cho ta thấy $x,y,z\geq 0$.\\
Nếu một trong ba số bằng $0$ thì hai số kia bằng $0$. Xét trường hợp $x, y, z>0$, nhân $(1), (2), (3)$ theo từng vế ta được\\
$8x^2y^2z^2=xyz(x^2+1)(y^2+1)(z^2+1) \Leftrightarrow (x^2+1)(y^2+1)(z^2+1)=8xyz$\hfill (4)\\
Ta lại có $\heva{&x^2+1\geq 2x>0\\&y^2+1\geq 2y>0\\&z^2+1\geq 2z>0}$ nên $(x^2+1)(y^2+1)(z^2+1)\geq 8xyz$\hfill (5)\\
Từ $(4)$ và $(5)$ suy ra $x=y=z=1$ (thỏa mãn hệ).\\
Nghiệm $(x;y;z)$ là $\left(0;0;0\right)$ và $\left(1;1;1\right)$.\\
}
\end{vd}
\begin{vd}
Cho hệ phương trình $\heva{&x^3=3y^2-3y+1&(1)\\&y^3=3z^2-3z+1&(2)\\&z^3=3x^2-3x+1&(3)}$\\
Chứng minh rằng:\\
\begin{enumerate}
  \item $x, y, z$ đều dương;\\
  \item $x=y=z=1$.
\end{enumerate}
\loigiai{
\begin{enumerate}
  \item Từ $(1)$ ta có $x^3=3y^2-3y+1=3\left(y-\dfrac{1}{2}\right)^2+\dfrac{1}{4}>0$, suy ra $x^3>0 \Rightarrow x>0$.\\
      Tương tự từ $(2)$ và $(3)$ suy ra $y>0, z>0$.
  \item Cộng $(1), (2), (3)$ theo vế ta được $(x-1)^3+(y-1)^3+(z-1)^3=0$\hfill (4)
\begin{itemize}
  \item Giả sử $x<1$ thì $x^3<1\Rightarrow 3y^2-3y+1<1\Rightarrow 3y(y-1)<0$\\
      Do $y>0$ nên $y-1<0 \Rightarrow y<1$\\
      Do $y<1$ nên $y^3<1 \Rightarrow  3z^2-3z+1<1\Rightarrow 3z(z-1)<0$\\
      Do $z>0$ nên $z<1$\\
      Do $x<1, y<1, z<1$ nên $(4)$ không xảy ra, loại.
  \item Giả sử $x>1$, ta cũng suy ra $y>1, z>1$ nên $(4)$ không thể xảy ra, loại.
  \item Vậy $x=1$. Từ $(1)$ suy ra $y=1$, từ $(3)$ suy ra $z=1$.\\
  Do đó $x=y=z=1$
\end{itemize}
\end{enumerate}
}
\end{vd}

\begin{bt}
Cho hệ phương trình $$\heva{&x+(m+1)y=2\\&(m+1)x-y=m+1}$$
Tìm giá trị của $m$ để nghiệm $(x;y)$ của hệ phương trình thỏa mãn $x-2y$ có giá trị lớn nhất.
\loigiai{
Rút $y$ từ $(m+1)x-y=m+1$ được $y=(m+1)x-(m+1)$. Thay vào $x+(m+1)y=2$ được $x+(m^2+1)x-(m^2-1)=2 \Leftrightarrow m^2x=m^2+1\ (1).$ \\
Với $m=0$ thì $(1)$ vô nghiệm. \\
Với $m\neq 0$ thì $x=\dfrac{m^2+1}{m^2}$; $y=\dfrac{m+1}{m^2}$. Khi đó \\
$x-2y=\dfrac{m^2+1-2(m+1)}{m^2} = \dfrac{m^2-2m-1}{m^2}=1-\dfrac{2}{m}-\dfrac{1}{m^2}.$\\
Đặt $\dfrac{1}{m}=z$ thì $x-2y=1-2z-z^2=2-(z+1)^2 \leq 2.$ \\
$\max (x-2y)=2 \Leftrightarrow z=-1 \Leftrightarrow m=-1.$ 
}
\end{bt}

\begin{bt}
Một người mang một số tiền đi mua táo. Nếu giá táo giảm đi $2$ nghìn đồng một quả thì số táo mua tăng thêm được $6$ quả. Nếu giá táo tăng thêm $2$ nghìn đồng một quả thì số táo giảm đi $4$ quả. Tính giá một quả táo. 
\loigiai{
Gọi số quả táo là $x$ (quả), giá một quả táo là $y$ (nghìn đồng). \\
Ta có hệ phương trình $\heva{&(x+6)(y-2)=xy\\&(x-4)(y+2)=xy} \Leftrightarrow \heva{&x=24\\&y=10}$\\
Giá táo là $10$ nghìn đồng một quả.
}
\end{bt}

\begin{bt}
Giải các hệ phương trình
\begin{enumerate}
\item[a)] $\heva{&x^2+y^2+x=3\\&x+2y+2xy=5}$
\item[b)] $\heva{&x^3-2x^2y+3y=0\\&2xy-y^2=3}$
\end{enumerate} 
\loigiai{
\begin{enumerate}
\item[a)] $\heva{&x^2+y^2+x=3 && (1)\\&x+2y+2xy=5 && (2)}$. \\
Cộng $(1)$ với $(2)$ được $(x+y)^2+2(x+y)-8=0$. Ta được $x+y=2$ hoặc $x+y=-4$. \\
Thay $y=2-x$ vào $(1)$ ta được $2x^2-3x+1=0 \Leftrightarrow \hoac{&x=1\\&x=\dfrac{1}{2}}$\\
Thay $y=-x-4$ vào $(1)$ ta được $2x^2+9x+13=0$, vô nghiệm. \\
Đáp số: $(1;1)$, $\left (\dfrac{1}{2};\dfrac{3}{2}\right ).$
\item[b)] $\heva{&x^3-2x^2y+3y=0 & (1)\\&2xy-y^2=3 & (2)}$ \\
Thay $3$ ở $(1)$ bởi $2xy-y^2$ ta được $x^3-2x^2y+y(2xy-y^2)=0 \Leftrightarrow x^3-2x^2y+2xy^2-y^3=0\ (3)$. \\
Do $(2)$ nên $y \neq 0$. Chia cả hai vế của $(3)$ cho $y^3$ được $\left (\dfrac{x}{y}\right )^3 -2 \left (\dfrac{x}{y}\right )^2 + 2 \cdot \dfrac{x}{y}-1=0$. \\
Đặt $\dfrac{x}{y}=a$. Ta có $a^3-2a^2+2a-1=0 \Leftrightarrow (a-1)(a^2-a+1)=0 \Leftrightarrow a=1$. \\
Suy ra $x=y$. Thay vào $(2)$ được $x= \pm \sqrt{3}$.  \\
Đáp số: $(\sqrt{3};\sqrt{3})$, $(-\sqrt{3};-\sqrt{3}).$
\end{enumerate} 
}
\end{bt}

\end{enumerate}

Đáp số: $(-1;-1)$, $(2;2)$.  
\item[b)] Đây là hệ đẳng cấp bậc hai. Ta thấy $y=0$ không thỏa mãn hệ phương trình. \\

Với $k=2$, ta có $x=2y$ và  được $x^2=6$. \\
Đáp số: $(2;2)$, $(-2;-2)$, $\left (\sqrt{6};\dfrac{\sqrt{6}}{2}\right )$, $\left (-\sqrt{6};-\dfrac{\sqrt{6}}{2}\right )$.
\end{enumerate}

\end{bt}


\begin{bt} Giải các hệ phương trình

\item[b)] $\heva{&x+y+\dfrac{1}{x}+\dfrac{1}{y}=6\\&x^2+y^2+\dfrac{1}{x^2}+\dfrac{1}{y^2}=16}$
\end{enumerate}
\loigiai{

Với $a=2,b=4$ ta được $x=1$, $y=2 \pm \sqrt{3}$. \\
Với $a=4,b=2$ ta được $x=2 \pm \sqrt{3}$, $y=1$.

}

\begin{bt}

\item[b)] $\heva{&x^2=y^3+my\\&y^2=x^3+mx}$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a)] Điều kiện cần. Giả sử $(x_0;y_0)$ là nghiệm của hệ phương trình thì $(x_0;y_0)$ cũng là nghiệm của hệ. \\

Với $m=32$ ta có $\heva{&2x+2y+xy=32\\&x^2+y^2=32}$ \\

Với $a=-12,b=56$ (loại). \\
Đáp số: $m=0$ hoặc $m=32$. 
\item[b)] $\heva{&x^2=y^3+my\\&y^2=x^3+mx}$ \\
Cũng lập luận để có $x=y$. Khi đó $x^2=x^3+mx \Leftrightarrow x(x^2-x+m)=0$. \\
Do $x=0$ là một nghiệm của phương trình nên $(0;0)$ là một nghiệm của hệ. Để nghiệm đó duy nhất thì phương trình $x^2-x+m=0$ vô nghiệm, tức là $m>\dfrac{1}{4}$. 
\end{enumerate}
}
\end{bt}

\begin{bt} 
Tìm các giá trị của $m$ để phương trình có nghiệm
\begin{enumerate}
\item[a)] $\heva{&x^2+y^2+x+y=2\\&xy(x+1)(y+1)=m}$
\item[b)] $\heva{&xy(x+2)(y+2)=m\\&x^2+y^2+2(x+y)=m+1}$

\loigiai{
\begin{enumerate}

 $$\heva{&\Delta' \geq 0\\&\left ( X_1+\dfrac{1}{4}\right ) \left ( X_2+\dfrac{1}{4}\right ) \geq 0 \\& \left ( X_1+\dfrac{1}{4}\right ) +\left ( X_2+\dfrac{1}{4}\right ) \geq 0 } \Leftrightarrow \heva{&m \leq 1\\& m \geq -\dfrac{9}{16}}$$

$a,b$ là nghiệm của phương trình $X^2-(m+1)X+m=0 \Leftrightarrow \hoac{&X=1\\&X=m}$. \\
Để $a\geq -1$ và $b \geq -1$ phải có $m \geq -1$. \\
Kết luận: $m\geq -1$.
\end{enumerate}
}
\end{bt}

\begin{bt} 
Cho hệ phương trình  $$ \heva{&x+y+xy=m\\&x^2y+xy^2=m-1}$$
Tìm giá trị của $m$ để hệ phương trình có nghiệm $(x;y)$ thỏa mãn $x<0$, $y<0$. 
\loigiai{
Đặt $x+y=a$, $xy=b$, ta có  \\
$\heva{&a+b=m\\&ab=m-1}$ nên $\heva{&a=1\\&b=m-1}$ hoặc $\heva{&a=m-1\\&b=1}$ \\
Loại trường hợp $a=1$, vì khi đó $x+y=1$, trái với $x<0$, $y<0$. \\
Với $\heva{&a=m-1\\&b=1}$ thì $x,y$ là nghiệm của phương trình $X^2-(m-1)X+1=0$. \\
Ta tìm được điều kiện để hai nghiệm cùng âm là $m\leq -1.$
}
\end{bt}


\begin{bt} 
Giải các hệ phương trình 
\begin{multicols}{2}
\item[a)] $\heva{&x+y+xy=5\\&y+z+yz=9\\&z+x+xz=14}$
\item[b)] $\heva{&x^2+4=4y\\&y^2+4=4z\\&z^2+4=4x}$
\end{multicols}
\loigiai{
\begin{enumerate}
\item[a)] Đưa về $\heva{&(x+1)(y+1)=6\\&(y+1)(z+1)=10\\&(z+1)(x+1)=15}$\\
Nhân từng vế các phương trình. \\
Đáp số: $(2;1;4)$, $(-4;-3;-6)$.
\item[b)] Đưa về $(x-2)^2+(y-2)^2+(z-2)^2=0$. \\
Đáp số: $(2;2;2)$. 
\end{enumerate}
}
\end{bt}

\begin{bt} Giải các hệ phương trình
\begin{multicols}{2}
\item[a)] $\heva{&2x^2+y^2+z^2=21\\&2x+xy+3y=17}$
\item[b)] $\heva{&x^2+y^2+z^2=1\\&x^3+y^3+z^3=1}$
\end{multicols}
\loigiai{
\begin{enumerate}
\item[a)] $\heva{&2x^2+y^2+z^2=21 &&(1)\\&4x+2xy+6y=34 &&(2)}$
Lấy $(1)$ trừ $(2)$ rồi đưa về $(x-2)^2+(x-y)^2+(z-3)^2=0.$ \\
Đáp số: $(2;2;3)$. 
\item[b)] Giải tương tự cách $2$ của ví dụ $28$. \\
Đáp số: $(1;0;0)$, $(0;1;0)$, $(0;0;1)$. 
\end{enumerate}
}
\end{bt}

\begin{bt} Tìm các số dương $x,y,z$ sao cho
$$\heva{&x+y+z=3\\&\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=3}$$
\loigiai{
Nhân từng vế hai phương trình được $(x+y+z)\left (\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\right )=9$. \\
Áp dụng bất đẳng thức $(x+y+z)\left (\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\right ) \geq 9$ với $x,y,z>0$. \\
Đáp số: $(1;1;1)$. 
}
\end{bt}


\begin{bt} Giải các hệ phương trình
\begin{multicols}{2}
\item[a)] $\heva{&x^2=4y-1\\&y^2=4z-1\\&z^2=4x-1}$
\item[b)] $\heva{&x^3=6y^2-12y+8\\&y^3=6z^2-12z+8\\&z^3=6t^2-12t+8\\&t^3=6x^2-12x+8}$
\end{multicols}
\loigiai{
\begin{enumerate}
\item[a)] $x^2=4y-1 \Rightarrow y \geq \dfrac{1}{4}$, tương tự $z \geq \dfrac{1}{4}, x \geq \dfrac{1}{4}$.\\
Ta sẽ chứng minh $x=y=z$. \\
Giả sử $x \geq y \Rightarrow 4y-1 \geq 4z-1 \Rightarrow y \geq z \Rightarrow y^2 \geq z^2 \Rightarrow 4z-1 \geq 4x-1 \Rightarrow z \geq x$. Từ $x \geq y \geq z \geq z $ suy ra $x=y=z$. \\
Đáp số: $(2+\sqrt{3};2+\sqrt{3};2+\sqrt{3})$,  $(2-\sqrt{3};2-\sqrt{3};2-\sqrt{3})$. 
\item[b)] Trước hết chứng minh $x,y,z,t$ đều dương. Cộng từng vế các phương trình được \\
$(x-2)^3+(y-2)^3+(z-2)^3+(t-2)^3=0$. Giải tương tự ví dụ $35$. \\
Đáp số: $(2;2;2;2)$.
\end{enumerate}
}
\end{bt}
