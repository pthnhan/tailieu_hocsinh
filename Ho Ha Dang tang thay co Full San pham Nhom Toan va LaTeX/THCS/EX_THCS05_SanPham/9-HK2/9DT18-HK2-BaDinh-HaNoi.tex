\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Ba Đình, Hà Nội]%[9D1B1]%[9D1B1]%[9D1K1]
Cho biểu thức $A=\left( \dfrac{2\sqrt{x}}{x-9}+\dfrac{1}{\sqrt{x}-3}\right):\dfrac{3}{\sqrt{x}-3}$ với $x\geq 0;x\ne 9$.
\begin{enumerate}
	\item Rút gọn biểu thức $A$.
	\item Tìm $x$ để $A=\dfrac{5}{6}$.
	\item Tìm giá trị nhỏ nhất của biểu thức $A$.
\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item
			 \begin{align*}
A=&\left(\dfrac{2\sqrt{x}}{x-9}+\dfrac{1}{\sqrt{x}-3}\right):\dfrac{3}{\sqrt{x}-3}\\
&\left(\dfrac{2\sqrt{x}}{(\sqrt{x}-3)(\sqrt{x}+3)}+\dfrac{\sqrt{x}+3}{(\sqrt{x}-3)(\sqrt{x}+3)}\right).\dfrac{\sqrt{x}-3}{3}\\
&=\dfrac{2\sqrt{x}+\sqrt{x}+3}{(\sqrt{x}-3)(\sqrt{x}+3)}.\dfrac{\sqrt{x}-3}{3}\\
&=\dfrac{\sqrt{x}+1}{\sqrt{x}+3}
\end{align*}
\item Để $A=\dfrac{5}{6}$ thì\\
$\dfrac{\sqrt{x}+1}{\sqrt{x}+3}=\dfrac{5}{6}\Leftrightarrow 6(\sqrt{x}+1)=5(\sqrt{x}+3)\Leftrightarrow\sqrt{x}=9\Leftrightarrow x=81$ (tmđk).\\
Vậy $x=81$.
\item $A=\dfrac{\sqrt{x}+1}{\sqrt{x}+3}=1-\dfrac{2}{\sqrt{x}+3}$.\\
Do $x\geq 0\Leftrightarrow A\geq \dfrac{1}{3}$ với mọi $x$ thỏa mãn đkxđ.\\
Dấu "=" xảy ra $\Leftrightarrow x=0$ (tmđk).\\
Vậy giá trị nhỏ nhất của $A$ là $\dfrac{1}{3}$ xảy ra khi $x=0$.
\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Ba Đình, Hà Nội]%[9D3B8]
	Hai đội công nhân cùng làm một công việc thì lam xong trong $8$ giờ. Nếu mỗi đội làm một mình xong công việc đó, đội thứ nhất cần ít thời gian hơn so với đội thứ hai là $12$ giờ. Hỏi mỗi đội làm một mình xong công việc đó trong bao lâu?
	\loigiai{
	Gọi thời gian đội thứ nhất làm một mình xong việc là $x$(giờ).\\
	Vậy thời gian đội thứ hai làm một mình xong công việc là $x+12$(giờ).\\
	Mỗi giờ đội thứ nhất làm được $\dfrac{1}{x}$(công việc).\\
	Mỗi giờ đội thứ hai làm được $\dfrac{1}{x+12}$(công việc).\\
	Theo bài ra, mỗi giờ cả hai đội làm được $\dfrac{1}{8}$ công việc ta có phương trình:
	\begin{center}
		$\dfrac{1}{x}+\dfrac{1}{x+12}=\dfrac{1}{8}$
	\end{center}
	Giải phương trình ta được $x=-8$(ktmdk);x=12(tmđk).\\
	Vậy thời gian đội thứ nhất làm một mình xong việc là $12$ giờ; thời gian đội thứ hai làm một mình xong việc là $24$ giờ.
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Ba Đình, Hà Nội]%[9D3B2]%[9D3B3]%[9D4K6]
\begin{enumerate}
	\item[1.] Giải hệ phương trình $\heva{&\vert x+5\vert -\dfrac{2}{\sqrt{y}-2}=4\\&\vert x+5\vert -\dfrac{1}{\sqrt{y}-2}=3}$.
	\item[2.] Cho phương trình $x^2 -2(m+1)x+m^2=0$.
	\begin{enumerate}
		\item Giải phương trình khi $m=4$.
		\item Tìm $m$ để phương trình có hai nghiệm phân biệt $x_1;x_2$ sao cho $x_1^2+x_2^2=4\sqrt{x_1.x_2}$
	\end{enumerate}
\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item[1.] Đk: $y\geq 0;y\ne 4$.\\
	Đặt $a=\vert x+5\vert;b=\dfrac{1}{\sqrt{y}-2}(a\geq 0)$.\\
	Ta có hệ $\heva{&a-2b=4\\&a+b=3}\Leftrightarrow \heva{&a=\dfrac{10}{3}\\&b=-\dfrac{1}{3}}$.\\
	Ta có $\sqrt{y}-2=-3\Leftrightarrow\sqrt{y}=-1$(vô nghiệm).\\
	Vậy hệ phương trình vô nghiệm.
	\item[2.]
	\begin{enumerate}
		\item Với $m=4$, giải phương trình $x^2-10x+16=0$ được $x\in\{2;8\}$.
		\item Phương trình đã cho có $2$ nghiệm phân biệt $\Leftrightarrow \Delta'>0\Leftrightarrow m>\dfrac{-1}{2}$.\\
		Theo Vi-et có $x_1+x_2=2(m+1);x_1.x_2=m^2$.\\
		Xét $x_1^2+x_2^2=4\sqrt{x_1.x_2}\Leftrightarrow(x_1+x_2)^2-2x_1.x_2=4\sqrt{x_1.x_2}\\\Leftrightarrow 4(m+1)^2-2m^2=4\vert m\vert =0$.\\
		Th1: $\dfrac{-1}{2}<m<0\Rightarrow m^2+6m+2=0\Leftrightarrow m_1=-2-\sqrt{7} (\text{loại});m_2=-3+\sqrt{7} (tmdk)$.\\
		Th2: $m>0\Rightarrow m^2+2m+2=0$. Không có $m$ thỏa mãn.\\
		Vậy $m=-3+\sqrt{7}$ thỏa mãn yêu cầu đề bài.
	\end{enumerate}	
	\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Ba Đình, Hà Nội]%[9H3Y7]%[9H3B3]%[9H3K3]%[9H3G2]
	Cho tam giác $ABC$ có ba góc nhọn nội tiếp đường tròn tâm $O$ bán kính $R$ và $AH$ là đường cao của tam giác $ABC$. Gọi $M,N$ thứ tự là hình chiếu của $H$ trên $AB$,  $AC$.
	\begin{enumerate}
		\item Chứng minh tứ giác $AMHN$ là tứ giác nội tiếp.
		\item Chứng minh $\widehat{ABC}=\widehat{ANM}$.
		\item Chứng minh $OA$ vuông góc với $MN$.
		\item Cho biết $AH=R\sqrt{2}$. Chứng minh $M,O,N$ thẳng hàng.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\tkzInit[xmin=-4,ymin=-1,xmax=4,ymax=6]
			\tkzDefPoints{-2/4/A,-4/0/B,3/0/C}
			\tkzDefCircle[circum](A,B,C)
			\tkzGetPoint{O}
			\tkzDefPointBy[projection = onto B--C](A)
			\tkzGetPoint{H}
			\tkzDefPointBy[projection = onto A--B](H)
			\tkzGetPoint{M}
			\tkzDefPointBy[projection = onto A--C](H)
			\tkzGetPoint{N}
			\tkzDefPointBy[symmetry = center O](A)
			\tkzGetPoint{D}
			\tkzDrawCircle[circum](A,B,C)
			\pgfresetboundingbox
			\tkzDrawSegments(A,B C,B A,C A,D A,H H,M H,N M,N)
			\tkzMarkRightAngles[<tùy chọn>](A,H,B H,M,A H,N,A)
			\tkzDrawPoints(A,B,C,M,N,H,D,O)
			\tkzLabelPoint[above](A){$A$}
			\tkzLabelPoints[below](D,H)
			\tkzLabelPoints[right](O,N,C)
			\tkzLabelPoints[left](M,B)
			\end{tikzpicture}
		\end{center}
	\begin{enumerate}
		\item Ta có $\widehat{AMH}=\widehat{ANH}=90^\circ$ (vì $M,N$ thứ tự là hình chiếu của $H$ trên $AB$,  $AC$)\\
		$\Rightarrow \widehat{AMH}+\widehat{ANH}=180^\circ\Rightarrow AMHN$ là tứ giác nội tiếp.
		\item Vì $AMHN$ nội tiếp nên ta có $\widehat{ANM}=\widehat{MHA}$ (cùng chắn cung $AM$)\\
		mà $\widehat{ABC}=\widehat{AHM}$(cùng phụ với $\widehat{MHB}$)\\
		$\Rightarrow \widehat{ABC}=\widehat{ANM}$.
		\item Ta có tam giác $AOC$ cân tại $O$ nên ta có $\widehat{OAC}=\dfrac{180^\circ-\widehat{AOC}}{2}=90^\circ-\dfrac{\widehat{AOC}}{2}=90^\circ-\widehat{ABC}$ (Vì góc ở tâm gấp đôi góc nội tiếp chắn cùng một cung).\\
		mà $\widehat{ABC}=\widehat{AHM}$\\
		$\Rightarrow \widehat{OAC}+\widehat{ANM}=90^\circ\Rightarrow AO\perp MN$.
		\item Kẻ đường kính $AD$.\\
		Vì tam giác $AHC$ vuông tại $H$ và có đường cao $HM$ nên ta có $AN.AC=AH^2=2R^2=AO.AD\Rightarrow \dfrac{AN}{AD}=\dfrac{AO}{AC}$\\
		$\Rightarrow \triangle AON\backsim\triangle ADC(c.g.c)$.\\
		$\Rightarrow \widehat{AOM}=\widehat{ADB}=90^\circ$.\\
		Vậy $\widehat{AOM}+\widehat{AON}=180^\circ\Rightarrow O,M,N$ thẳng hàng.
	\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Ba Đình, Hà Nội]%[9D5G1]
	Cho $a,b>0$ thỏa mãn $a+b\leq 2$. Tìm giá trị lớn nhất của biểu thức
	\begin{center}
		$P=\sqrt{a(b+1)}+\sqrt{b(a+1)}$
	\end{center}
	\loigiai{
		Có $\sqrt{2}P=\sqrt{2x(b+1)}+\sqrt{2b(a+1)}$.\\
		Áp dụng BĐT Cô si cho hai số không âm\\
		$\sqrt{2a(b+1)}\leq \dfrac{2a+b+1}{2};\sqrt{2b(a+1)}\leq \dfrac{2b+a+1}{2}.$\\
		$\Rightarrow \sqrt{2}P\leq \dfrac{3(a+b)+2}{2}\leq 4 $\\
		$\Rightarrow P\leq 2\sqrt{2}$.\\
		Dấu "=" xảy ra $\Leftrightarrow\heva{&2a=b+1\\&2b=a+1}\Leftrightarrow a=b=1$.\\
		Vậy $P$ có GTLN là $2\sqrt{2}$ khi $a=b=1$.
	}
\end{ex}
