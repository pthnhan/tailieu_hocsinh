\begin{enumerate}[1.]
\item Rút gọn các biểu thức sau:
	\begin{multicols}{2}
	\begin{enumerate}[a)]
	\item $A=\left(2\sqrt{3}-5\sqrt{27}+4\sqrt{12}\right):\sqrt{3}$;
	\item $B=\dfrac{2}{\sqrt{7}-\sqrt{6}}-\sqrt{28}+\sqrt{54}$.
	\end{enumerate}
	\end{multicols}
\item Giải hệ phương trình $\heva{&2x-y=3\\ &3x+2y=8}$.
\item Xác định hệ số $a$ và $b$ của đường thẳng $(d):y=ax+b$, biết đường thẳng $(d)$ song song với đường thẳng $(d'): y=x+2017$ và đi qua điểm $A(-1;2015)$.
\end{enumerate}
\loigiai{
\begin{enumerate}[1.]
\item	\begin{enumerate}[a)]
	\item $A=\left(2\sqrt{3}-5\sqrt{27}+4\sqrt{12}:\sqrt{3}\right)=\left(2\sqrt{3}-15\sqrt{3}+8\sqrt{3}\right):\sqrt{3}=-5$.
	\item $B=\dfrac{2}{\sqrt{7}-\sqrt{6}}-\sqrt{28}+\sqrt{54}=\dfrac{2\left(\sqrt{7}+\sqrt{6}\right)}{\left(\sqrt{7}-\sqrt{6}\right)\left(\sqrt{7}+\sqrt{6}\right)}-2\sqrt{7}+3\sqrt{6}=5\sqrt{6}$.
	\end{enumerate}
\item 
\textit{Cách 1: Phương pháp thế}\\
$\heva{&2x-y=3\\ &3x+2y=8}\Leftrightarrow \heva{&y=2x-3\\ &3x+2(2x-3)=8}\Leftrightarrow \heva{&x=2\\ &y=1}$. \\
\textit{Cách 2: Phương pháp cộng đại số}\\
$\heva{&2x-y=3\\ &3x+2y=8}\Leftrightarrow \heva{&4x-2y=6\\ &3x+2y=8} \Leftrightarrow \heva{&x=2\\ &y=1}$.
\end{enumerate}
}

\begin{ex}%[Đề thi vào 10, Sở Hải Phòng, 2016]%[9D4Y4]%[9D4K6]%[9D4B7]
\begin{enumerate}[1.]
\item Cho phương trình $x^2-mx-4=0\quad (1)$ (với $m$ là tham số).
	\begin{enumerate}[a)]
	\item Giải phương trình $(1)$ khi $m=3$.
	\item Tìm các giá trị của $m$ để phương trình $(1)$ có hai nghiệm $x_1$, $x_2$ thoả mãn
	$$x_1(x_2^2+1)+x_2(x_1^2+1)>6.$$
	\end{enumerate}
\item Cho tam giác vuông có cạnh huyền bằng $20\,\mathrm{cm}$. Hai cạnh góc vuông có độ dài hơn kém nhau $4\,\mathrm{cm}$. Tính độ dài mỗi cạnh góc vuông của tam giác vuông đó.
\end{enumerate}
\loigiai{\begin{enumerate}[1.]
\item	\begin{enumerate}[a)]
			\item Khi $m=3$ phương trình trở thành $x^2-3x-4=0$.\\
			Phương trình có $\Delta =(-3)^2-4\cdot 1\cdot(-4)=25$.\\
			Vậy phương trình có hai nghiệm là $x_1=-1$ và $x_2=4$.
			
			\item Phương trình đã cho có $\Delta =m^2+16>0$. Do đó phương trình luôn có hai nghiệm $x_1$ và $x_2$. Theo định lí Viete, ta có $\heva{& x_1+x_2=m\\ &x_1x_2=-4}$. Suy ra
			\begin{align*}
			&x_1(x_2^2+1)+x_2(x_1^2+1)>6\\
			\Leftrightarrow &(x_1x_2+1)(x_1+x_2)>6\\
			\Leftrightarrow &-3m>6\\
			\Leftrightarrow &m<-2.
			\end{align*}
			Vậy $m<-2$.
			\end{enumerate}
\item Giả sử độ dài hai cạnh góc vuông của tam giác vuông đó là $x$, $y$ ($x>y$). Theo đề bài ta có
\begin{align*}
\heva{&x^2+y^2=400\\ &x-y=4}\Leftrightarrow \heva{&(y+4)^2+y^2=400\\&x=y+4}\Leftrightarrow \heva{&x=16\\ &y=12}.
\end{align*}
Vậy độ dài hai cạnh góc vuông cần tìm là $16$ cm và $12$ cm.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở Hải Phòng, 2016]%[9H3B7]
Cho tam giác $ABC$ có ba góc nhọn nội tiếp đường tròn $(O)$, kẻ $AH$ vuông góc với $BC$ tại $H$. Gọi $I$, $K$ lần lượt là hình chiếu vuông góc của $A$ lên các tiếp tuyến tại $B$ và $C$ của đường tròn $(O)$.
\begin{enumerate}
\item Chứng minh tứ giác $AHCK$ nội tiếp đường tròn;
\item Chứng minh $\widehat{AHK}=\widehat{ABC}$ và $AH^2=AI\cdot AK$;
\item Gọi $M$, $N$ theo thứ tự là trung điểm của $AI$ và $AK$. Chứng minh rằng nếu $AH=AM+AN$ thì ba điểm $A$, $O$, $H$ thẳng hàng.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzDefPoints{0/4.5/A, -2/0/B, 3/0/C, 0/0/H}
\tkzDefCircle[circum](A,B,C)\tkzGetPoint{O}
\tkzTangent[at=B](O)\tkzGetPoint{x}
\tkzTangent[at=C](O)\tkzGetPoint{y}
\tkzDrawAltitude(B,x)(A)\tkzGetPoint{I}
\tkzDrawAltitude(C,y)(A)\tkzGetPoint{K}
\tkzDefMidPoint(A,I)\tkzGetPoint{M}
\tkzDefMidPoint(A,K)\tkzGetPoint{N}
\tkzDrawCircle(O,A)

\tkzMarkRightAngle(A,H,B)
\tkzMarkRightAngle(A,K,C)
\tkzMarkRightAngle(A,I,B)
\tkzMarkAngle[size=0.4](K,H,A)
\tkzMarkAngle[size=0.4](C,B,A)
\tkzMarkAngle[size=0.4](K,C,A)

\tkzLabelPoints[below](H)
\tkzLabelPoints[above](A,O)
\tkzLabelPoints[above left](I,M)
\tkzLabelPoints[above right](K,N)
\tkzLabelPoints[below left](B)
\tkzLabelPoints[below right](C)

\tkzDrawSegments(A,B B,C C,A A,I A,K H,K B,I C,K M,N A,H H,I)

\tkzDrawPoints[fill=white,size=8](A,B,C,H,I,K,M,N,O)
\end{tikzpicture}
\end{center}
\begin{enumerate}[a)]
\item Vì $\widehat{AHC}=\widehat{AKC}=90^\circ$ nên tứ giác $AHCK$ nội tiếp đường tròn đường kính $AC$.
\item Do tứ giác $AHCK$ nội tiếp nên $\widehat{AHK}=\widehat{ACK}$. Do tính chất góc tạo bởi tiếp tuyến và dây cung nên $\widehat{ACK}=\widehat{ABC}$. Vậy $\widehat{AHK}=\widehat{ABC}$.\\
Ta có $\widehat{AKH}=\widehat{ACB}$.
Tương tự ý a), ta có tứ giác $AHBI$ nội tiếp đường tròn đường kính $AB$ nên $\widehat{AIH}=\widehat{ABC}$. Tương tự chứng minh trên ta cũng có $\widehat{AHI}=\widehat{ACB}$. Từ đó suy ra $\triangle AIH \backsim \triangle AHK$ (g.g). Do đó $\dfrac{AI}{AH}=\dfrac{AH}{AK}$ hay $AH^2=AI\cdot AK$.
\item Ta có $AH=AM+AN=\dfrac{AI+AK}{2}$. Kết hợp với ý b) ta có $\dfrac{(AI+AK)^2}{4}=AI\cdot AK$ hay $AI=AK$. Suy ra $AH=AI=AK$. Mặt khác theo ý b) ta có $\widehat{IAH}=\widehat{KAH}$ nên $\triangle AIH=\triangle AKH$ (c.g.c). Suy ra $\widehat{ABC}=\widehat{AHK}=\widehat{AHI}=\widehat{ACB}$ hay tam giác $ABC$ cân tại $A$. Do đó $A$, $O$, $H$ thẳng hàng.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở Hải Phòng, 2016]%[9D5G1]
\begin{enumerate}
\item Cho $a>0$, $b>0$, $c>0$. Chứng minh rằng $(a+b+c)\left(\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}\right)\ge 9$;
\item Cho ba số thực dương $a$, $b$, $c$ thoả mãn $a+b+c=1$. Tìm giá trị nhỏ nhất của biểu thức
$$P=\dfrac{9}{2(ab+bc+ca)}+\dfrac{2}{a^2+b^2+c^2}.$$
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item Áp dụng bất đẳng thức Cauchy cho ba số dương ta có
\begin{align*}
a+b+c&\ge 3\sqrt[3]{abc},\\
\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}&\ge 3\sqrt[3]{\dfrac{1}{abc}}.
\end{align*}
Nhân hai bất đẳng thức trên vế với vế, ta được điều phải chứng minh.
\item Đặt $x=ab+bc+ca$ và $y=a^2+b^2+c^2$. Ta có $\heva{& 2x+y=1\\ &0< x\le y}$. Suy ra
\begin{align*}
P&= \dfrac{9}{2x}+\dfrac{2}{y}=\dfrac{13}{3x}+\dfrac{1}{6x}+\dfrac{2}{y}\\
&\ge \dfrac{13}{3x}+\dfrac{1}{6y}+\dfrac{2}{y}=\dfrac{13}{3x}+\dfrac{13}{6y}\\
&=\dfrac{13}{6}\left(\dfrac{2}{x}+\dfrac{1}{y}\right)=\dfrac{13}{6}(x+x+y)\left(\dfrac{1}{x}+\dfrac{1}{x}+\dfrac{1}{y}\right)\\
&\ge \dfrac{13}{6}\cdot 9=\dfrac{39}{2}.
\end{align*}
Đẳng thức xảy ra khi $a=b=c=\dfrac{1}{3}$.\\
Vậy giá trị nhỏ nhất của $P$ là $\dfrac{39}{2}$.
\end{enumerate}
}
\end{ex}
