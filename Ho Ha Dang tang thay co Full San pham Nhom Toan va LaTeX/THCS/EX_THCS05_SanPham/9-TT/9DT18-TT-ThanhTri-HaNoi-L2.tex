\begin{ex}%[TT Thanh Trì HN, 2018]%[9D1B6]%[9D1B8]
Cho các biểu thức
$$A=\dfrac{\sqrt{x}+1}{\sqrt{x}-2}\quad\text{và}\quad B=\dfrac{\sqrt{x}}{\sqrt{x}+1}+\dfrac{1-\sqrt{x}}{\sqrt{x}-2}-\dfrac{\sqrt{x}+4}{x-\sqrt{x}-2}\,\,\text{với}\,\, x\ge 0, x\neq 4.$$
\begin{enumerate}
\item Tính giá trị của $A$ khi $x=7+4\sqrt{3}$.
\item Chứng minh rằng $B=\dfrac{3}{2-\sqrt{x}}$.
\item Tìm $x$ để $\dfrac{B}{A}<-1$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $x=7+4\sqrt{3}=\left(\sqrt{3}+2\right)^2$ nên $\sqrt{x}=2+\sqrt{3}$. Do đó
$$A=\dfrac{2+\sqrt{3}+1}{2+\sqrt{3}-2}=\dfrac{\sqrt{3}\left(\sqrt{3}+1\right)}{\sqrt{3}}=\sqrt{3}+1.$$
\item Ta có 
\begin{eqnarray*}
A & = & \dfrac{\sqrt{x}\left(\sqrt{x}-2\right)+\left(1-\sqrt{x}\right)\left(\sqrt{x}+1\right)-\left(\sqrt{x}+4\right)}{\left(\sqrt{x}+1\right)\left(\sqrt{x}-2\right)}\\
& = & \dfrac{-3\sqrt{x}-3}{\left(\sqrt{x}+1\right)\left(\sqrt{x}-2\right)}=\dfrac{3}{2-\sqrt{x}}.
\end{eqnarray*}
Do đó ta có điều phải chứng minh.
\item Ta có
$$\dfrac{B}{A}<-1\Leftrightarrow\dfrac{3}{2-\sqrt{x}}:\dfrac{\sqrt{x}+1}{\sqrt{x}-2}<-1\Leftrightarrow \dfrac{3}{\sqrt{x}+1}>1\Leftrightarrow\sqrt{x}<2\Leftrightarrow x<4.$$
Vậy $0\le x<4$ thì thỏa mãn yêu cầu đề bài.
\end{enumerate}
}
\end{ex}

\begin{ex}%[TT Thanh Trì HN, 2018]%[9D3Y3]%[9D3B3]%[9D3K3]
Cho hệ phương trình $\begin{cases}
-2mx+y=5\\ mx+3y=1
\end{cases}$ (với $m$ là tham số).
\begin{enumerate}
\item Giải hệ phương trình với $m=1$.
\item Tìm $m$ để phương trình có nghiệm $(x,y)$ thỏa mãn $x-y=2$.
\item Chứng minh rằng nếu hệ phương trình có nghiệm $(x_0;y_0)$ thì điểm $M(x_0;y_0)$ luôn nằm trên một đường thẳng cố định khi $m$ thay đổi.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Với $m=1$ hệ trở thành
$$\begin{cases}
-2x+y=5\\ x+3y=1
\end{cases}
\Leftrightarrow \begin{cases}
y=2x+5\\ x+3(2x+5)=1
\end{cases}
\Leftrightarrow\begin{cases}
x=-2\\ y=1.
\end{cases}$$
Vậy nghiệm hệ phương trình là $(-2;1)$.
\item Ta có $x-y=2\Leftrightarrow y=x-2$. Thế vào hệ ta được
$$\begin{cases}
-2mx+x-2=5\\ mx+3(x-2)=1
\end{cases}
\Leftrightarrow\begin{cases}
-2mx+x=7\\ 2mx+6x=14
\end{cases}
\Leftrightarrow\begin{cases}
x=3\\ m=-\dfrac{2}{3}.
\end{cases}$$
Vậy $m=-\dfrac{2}{3}$ thì thỏa mãn yêu cầu đề bài.
\item Vì $(x_0,y_0)$ là nghiệm hệ phương trình nên 
$$\begin{cases}
-2mx_0+y_0=5\\ mx_0+3y_0=1
\end{cases}\Leftrightarrow\begin{cases}
-2mx_0+y_0=5\\ 2mx_0+6y_0=2
\end{cases}
$$
Cộng vế hai phương trình ta được $7y_0=7\Leftrightarrow y_0=1$. Do đó điểm $M(x_0;y_0)$ luôn nằm trên đường thẳng $y=1$ cố định.
\end{enumerate}
}
\end{ex}

\begin{ex}%[TT Thanh Trì HN, 2018]%[9D3K5]
(Giải bài toán bằng cách lập phương trình hoặc hệ phương trình)\\
Một đội công nhân được giao làm $1200$ sản phẩm trong thời gian nhất định. Sau khi làm $5$ ngày với năng suất dự kiến, đội đã tăng năng suất mỗi ngày thêm $10$ sản phẩm. Do đó, đội đã hoàn thành công việc được giao sớm $5$ ngày. Hỏi theo kế hoạch đội phải hoàn thành công việc trong bao nhiêu ngày?
\loigiai{
Gọi thời gian và năng suất dự kiến của đội công nhân là $x$ (ngày) và $y$ (sản phẩm).\\
Điều kiện $y\in\mathbb{N}$, $x,y>0$.\\
Vì sản phẩm được giao là $1200$ sản phẩm nên ta có phương trình $xy=1200$.\hfill (1)\\
Số sản phẩm $5$ ngày đầu thực tế là $5y$ (sản phẩm).\\
Do đội làm xong sớm $5$ ngày nên số ngày làm còn lại là $x-10$ (ngày).\\
Vì năng suất những này còn lại tăng $10$ sản phẩm nên số sản phẩm làm trong những ngày còn lại là $(y+10)(x-10)$ (sản phẩm).\\
Do đó ta có phương trình $5y+(y+10)(x-10)=1200$.\hfill (2)\\
Từ (1) và (2) ta có hệ phương trình
$$\begin{cases}
xy=1200\\ 5y+(y+10)(x-10)=1200
\end{cases}\Leftrightarrow\begin{cases}
xy=1200\\ 10x-5y=100
\end{cases}\Leftrightarrow\begin{cases}
y=2x-50\\ x(2x-50)=1200.\quad(\ast)
\end{cases}$$
Giải $(\ast)$ ta được $x=40$ (thỏa mãn) hoặc $x=-15$ (loại).\\
Vậy số ngày đội phải hoàn thành công việc theo kế hoạch là $40$ ngày.
}
\end{ex}

\begin{ex}%[TT Thanh Trì HN, 2018]%[9H3Y3]%[9H3B7]%[9H3K7]%[9H3G7]
Cho $\triangle MAB$ vuông tại $M$ ($MA<MB$) có đường cao $MH$ ($H\in AB$). Đường tròn $(O)$ đường kính $MH$ cắt $MA$, $MB$ lần lượt tại $E$ và $F$ ($E,F\neq M$).
\begin{enumerate}
\item Chứng minh tứ giác $MEHF$ là hình chữ nhật.
\item Chứng minh tứ giác $AEFB$ là tứ giác nội tiếp.
\item Đường thẳng $EF$ cắt đường tròn $(O')$ ngoại tiếp $\triangle MAB$ tại các điểm $P,Q$ ($P\in\wideparen{MA}$). Chứng minh $\triangle MPQ$ cân.
\item Gọi $I$ là giao điểm thứ hai của $(O)$ và $(O')$, $K$ là giao điểm của đường thẳng $EF$ với đường thẳng $AB$. Chứng minh $M,I,K$ thẳng hàng.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1.8]
\tkzDefPoints{-1.5/0/A, -0.2/0/H, 2.5/0/B}
\clip (-4,-2.1) rectangle (3.2,2.7);
\tkzDefLine[perpendicular=through H](A,B)\tkzGetPoint{h}
\coordinate (O') at ($(A)!.5!(B)$);
\tkzDrawCircle[radius](O',A)
\tkzInterLC(H,h)(O',A)\tkzGetPoints{m}{M}
\coordinate (O) at ($(M)!.5!(H)$);
\tkzDrawCircle[radius](O,M)
\tkzInterLC(M,A)(O,M)\tkzGetPoints{E}{e}
\tkzInterLC(M,B)(O,M)\tkzGetPoints{f}{F}
\tkzInterLC(E,F)(O',A)\tkzGetPoints{P}{Q}
\tkzInterCC(O,M)(O',A)\tkzGetPoints{i}{I}
\tkzInterLL(E,F)(A,B)\tkzGetPoint{K}
\tkzInterLL(M,O')(E,F)\tkzGetPoint{J}

\tkzDrawSegments(M,A M,B M,H K,B H,E H,F K,Q M,O')
\tkzDrawSegments[dashed](K,M)
\tkzDrawPoints[fill=white,size=8](A,H,B,O',M,O,E,F,P,Q,I,K,J)
\tkzMarkRightAngles[size=0.15](A,M,B M,H,A H,E,M H,F,M)

\tkzLabelPoints[above](M)
\tkzLabelPoints[above left](P,O,I)
\tkzLabelPoints[above right](Q,F,J)
\tkzLabelPoints[below](K,H,O')
\tkzLabelPoints[below=6pt](E)
\tkzLabelPoints[below left](A)
\tkzLabelPoints[right](B)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Ta có $\widehat{MEH}=\widehat{MFH}=90^\circ$ (góc nội tiếp chắn nửa đường tròn).\\
Mà $\widehat{EMF}=90^\circ$ nên $MEHF$ là hình chữ nhật.
\item $\triangle OFM$ cân tại $O$ nên $\widehat{OFM}=\widehat{OMF}$.\\
Lại có $\widehat{OMF}=\widehat{EABA}$ (cùng phụ với góc $\widehat{HMA}$).\\
Suy ra $\widehat{MFE}=\widehat{EAB}$. Mà $\widehat{MFE}+\widehat{EFB}=180^\circ$.\\
Suy ra $\widehat{EAB}+\widehat{EFB}=180^\circ$. Vậy $AEFB$ là tứ giác nội tiếp.
\item Ta có $\widehat{EAB}+\widehat{O'BM}=90^\circ$, mà $\widehat{O'BM}=\widehat{O'MB}$ (vì $\triangle O'BM$ cân tại $O'$).\\
Theo chứng mình trên lại có $\widehat{MFE}=\widehat{EAB}$.\\
Do đó $\widehat{MFE}+\widehat{O'MB}=90^\circ$. Suy ra $\triangle MJF$ vuông tại $J$ (với $J=MO'\cap PQ$).\\
Suy ra $O'M\perp PQ$ và $J$ là trung điểm $PQ$, hay $O'M$ là đường trung trực của $PQ$.\\
Do đó $MP=MQ$ hay $\triangle MPQ$ cân tại $M$.
\item Xét $\triangle KMO'$ có $KJ$ và $MH$ là đường cao cắt nhau tại $O$ nên $O$ là trực tâm của $\triangle KMO'$. Suy ra $OO'\perp KM$. \hfill (1)\\
Lại có $\begin{cases}
OM=OI\\ O'M=O'I
\end{cases}$ nên $OO'$ là đường trung trực của $IM$ hay $OO'\perp IM$.\hfill (2)\\
Từ (1) và (2) suy ra $M,I,K$ thẳng hàng.
\end{enumerate}
}
\end{ex}

\begin{ex}%[TT Thanh Trì HN, 2018]%[9D5G1]
Cho $x>1$, $y>1$. Tìm giá trị nhỏ nhất của biểu thức
$P=\dfrac{x^2}{y-1}+\dfrac{y^2}{x-1}.$
\loigiai{
Theo BĐT Cô-si (AM-GM, Cauchy) ta có
$$P=\dfrac{x^2}{y-1}+\dfrac{y^2}{x-1}\ge 2\sqrt{\dfrac{x^2}{y-1}\cdot\dfrac{y^2}{x-1}}=2\sqrt{\dfrac{x^2}{x-1}\cdot\dfrac{y^2}{y-1}}.$$
Lại có
$$\dfrac{x^2}{x-1}=x+1+\dfrac{1}{x-1}=x-1+\dfrac{1}{x-1}+2\ge 2+2=4.$$
Tương tự cũng có $\dfrac{y^2}{y-1}\ge 4$.
Do đó $P\ge 8$, dấu bằng xảy ra khi $x=y=2$.\\
Vậy GTNN của $P$ là $8$ khi $x=y=2$.
}
\end{ex}
