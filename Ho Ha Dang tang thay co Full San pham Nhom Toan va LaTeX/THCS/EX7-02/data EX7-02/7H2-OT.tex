\begin{vd}[Đề thi HKI, PGD quận 6, năm 2016 - 2017]
Cho $\triangle ABC$ có $AB=AC$, gọi $M$ là trung điểm của đoạn thẳng $BC$.
\begin{enumerate}
\item Chứng minh $\triangle AMB=\triangle AMC$;
\item Chứng minh $AM$ là tia phân giác của góc $\widehat{BAC}$ và $AM\perp BC$;
\item Gọi $I$ là trung điểm của đoạn thẳng $AC$. Trên tia đối của tia $IM$, lấy điểm $D$ sao cho $ID=IM$. Tính số đo góc $\widehat{ADC}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Xét $\triangle AMB$ và $\triangle AMC$ có\\
$\left. 
\begin{array}{l}
AB=AB \quad \text{(gt)}\\ 
BM=CM \quad \text{(gt)}\\ 
AM: \text{ cạnh chung}
\end{array} 
\right\}\Rightarrow \triangle AMB =\triangle AMC$ (c-c-c).
\item Do $\triangle AMB =\triangle AMC$ nên $\widehat{BAM}=\widehat{CAM}$ và $\widehat{AMB}=\widehat{AMC}=90^\circ$.\\
$\Rightarrow AM$ là tia phân giác của $\widehat{BAC}$ và $AM\perp BC$.
\item 
\immini{
Xét $\triangle AID$ và $\triangle CIM$ có\\
$\left. 
\begin{array}{l}
IM=ID \quad \text{(gt)}\\ 
\widehat{AID}=\widehat{CIM} \text{ (đối đỉnh)}\\
IA=IC \quad \text{(gt)}
\end{array} 
\right\}$\\
$\Rightarrow \triangle AID =\triangle CIM$ (c-g-c).\\
$\Rightarrow \widehat{ADI}=\widehat{CMI}$ (hai góc tương ứng).\tagEX{1}
Xét $\triangle AIM$ và $\triangle CID$ có\\
$\left. 
\begin{array}{l}
IM=ID \quad \text{(gt)}\\ 
\widehat{AIM}=\widehat{CID} \text{ (đối đỉnh)}\\
IA=IC \quad \text{(gt)}
\end{array} 
\right\}$\\
$\Rightarrow \triangle AIM =\triangle CID$ (c-g-c).\\
$\Rightarrow \widehat{AMI}=\widehat{CDI}$ (hai góc tương ứng).\tagEX{2}

\begin{vd}[Đề thi HKI, PGD quận 3, năm 2014 - 2015]

\begin{enumerate}
\item Tính số đo của $\widehat{ABC}$ khi $\widehat{ACB}=40^\circ$;
\item Chứng minh $\triangle AMB=\triangle EMC$ và $AB\parallel EC$;
\item Từ $C$ kẻ đường thẳng $d$ song song với $AE$. Kẻ $EK$ vuông góc với $d$ tại $K$. Chứng minh $\widehat{KEC}=\widehat{BCA}$.
\end{enumerate}
