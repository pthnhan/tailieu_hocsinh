	\hfill
	\begin{enumerate}[1)]
		\item Cho biểu thức $P=\dfrac{\sqrt{a^3}-\sqrt{b^3}}{a-b}-\dfrac{a}{\sqrt{a}+\sqrt{b}}-
		\dfrac{b}{\sqrt{b}-\sqrt{a}}$ với $a$, $b$ là các số dương khác nhau. Rút gọn $P$.
		\item Cho hai số dương $a$, $b$ và số $c$ khác $0$ thỏa mãn điều kiện $\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}=0$. Chứng minh rằng $\sqrt{a+b}=\sqrt{a+c}+\sqrt{b+c}$.
		
	\end{enumerate}
	
	\loigiai{
		\begin{enumerate}[1)]
			\item Ta biến đổi	\[ \begin{array}{rl}
			P&=\dfrac{a\sqrt{a}-b\sqrt{b}-a(\sqrt{a}-\sqrt{b})+b(\sqrt{a}+\sqrt{b})}{a-b}\\
			&=\dfrac{\sqrt{ab}(\sqrt{a}+\sqrt{b})}{a-b}\\
			&=\dfrac{\sqrt{ab}}{\sqrt{a}-\sqrt{b}}.
			\end{array}\]
			\item $\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}=0\Rightarrow \heva{&\dfrac{1}{c}=-\left(\dfrac{1}{a}+\dfrac{1}{b}\right)\\ &\dfrac{ab+bc+ca}{abc}=0}
			\Rightarrow \heva{&c<0 \\ & ab+bc+ca=0}.$\\
			Suy ra \[\begin{array}{rl}
			&\sqrt{a+b}=\sqrt{a+c}+\sqrt{b+c}\\
			\Leftrightarrow & a+b=a+b+2c+2\sqrt{(a+c)(b+c)}\\
			\Leftrightarrow & c+\sqrt{ab+bc+ca+c^2}=0\\
			\Leftrightarrow & c+|c|=0. \text{ Điều này đúng vì } c<0.
			\end{array}\]
			
		\end{enumerate}
	}
\end{ex}

	Cho $x=\sqrt[3]{3+2\sqrt{2}}-\sqrt[3]{3-2\sqrt{2}}$; $y=\sqrt[3]{17+12\sqrt{2}}-\sqrt[3]{17-12\sqrt{2}}$. Tính giá trị biểu thức $M=(x-y)^3+3(x-y)(xy+1)$.
	\loigiai{
		Áp dụng hằng đẳng thức $(a-b)^3=a^3-b^3-3ab(a-b)$.\\
		Ta có $x^3=4\sqrt{2}-3x,\; y^3=24\sqrt{2}-3y$.\\
		Suy ra \[ 
		\begin{array}{rl}
		M&=x^3-y^3-3xy(x-y)+3xy(x-y)+3(x-y)\\
		&=x^3-y^3+3(x-y)=-20\sqrt{2}.
		\end{array}
		\]	
	}
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Lào Cai, tỉnh Lào Cai, năm 2018]%[Nguyễn Quang Tân, Dự án EX-TS10C-18]%[9D4K8]
Một công ty vận tải dự định dùng loại xe lớn để chở $20$ tấn rau theo hợp đồng. Nhưng khi vào việc, công ty không còn xe lớn nên phải thay bằng những xe có trọng tải nhỏ hơn $1$ tấn so với xe lớn ban đầu. Để đảm bảo hợp đồng, công ty phải dùng một số lượng xe nhiều hơn số xe dự định là 1 xe. Hỏi trọng tải của mỗi xe nhỏ là bao nhiêu tấn?
\loigiai{
	Giả sử một xe tải nhỏ có thể chở được $x$ tấn rau ($x>0$).\\
	Suy ra một xe tải lớn trở được $x+1$ tấn rau.\\
	Theo đề bài ta có phương trình sau
	\[\dfrac{20}{x}=\dfrac{20}{x+1}+1
	\Leftrightarrow \dfrac{20}{x(x+1)}=1
	\Leftrightarrow  x^2+x-20=0
	\Leftrightarrow  \hoac{&x=4\\ &x=-5 \text{ (loại).}}\]
	Vậy trọng tải của xe tải nhỏ là $4$ tấn.
}	
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Lào Cai, tỉnh Lào Cai, năm 2018]%[Nguyễn Quang Tân, Dự án EX-TS10C-18]%[9D4K6]
 Tìm tất cả các giá trị nguyên của $m$ để phương trình $x^3-3x+m-4=0$ (trong đó $x$ là ẩn) có hai nghiệm phân biệt $x_1$, $x_2$ thỏa mãn $\dfrac{x_1+x_2}{(x_1x_2)^{2019}}$ là số nguyên.

	\loigiai{ 
		 Phương trình có 2 nghiệm phân biệt khi $\Delta'=\dfrac{9}{4}-m+4>0\Leftrightarrow m<\dfrac{25}{4}.$\\
			Áp dụng định lý Vi-et ta có $\dfrac{x_1+x_2}{(x_1x_2)^{2019}}=\dfrac{3}{(m-4)^{2019}}.$\\
			Vì $m$ là số nguyên nên biểu thức trên là số nguyên khi $(m-4)^{2019}$ là ước của $3$, suy ra $m-4$ là ước của $3$. Xét 2 trường hợp:
			\begin{itemize}
				\item Nếu $m-4=\pm 3 \Rightarrow (m-4)^{2019}=\pm 3^{2019}$ không phải là ước của $3.$\\
				\item Nếu $m-4=\pm 1\Rightarrow (m-4)^{2019}=\pm 1$ là ước của $3$. Khi đó $m=5$ hoặc $m=3$.
			\end{itemize}
			Vậy đáp án của bài toán là $m=3$ và $m=5$.
	}
\end{ex}

	Cho các số thực dương $a,b,c$ thỏa mãn điều kiện $a+b+c=\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}$. Chứng minh rằng
	\[3(a+b+c)\geq \sqrt{8a^2+1}+\sqrt{8b^2+1}+\sqrt{8c^2+1}.\]
	\loigiai{
		Áp dụng bất đẳng thức AM - GM ta có
		\[
		\begin{array}{rl}
		18(a+b+c)&=\left(8a+\dfrac{1}{a}+9a\right)+\left(8b+\dfrac{1}{b}+9b\right)+\left(8c+\dfrac{1}{c}+9c\right)\\
		&=\left(\dfrac{8a^2+1}{a}+9a\right)+\left(\dfrac{8b^2+1}{b}+9b\right)+\left(\dfrac{8c^2+1}{c}+9c\right)\\
		&\geq 6\sqrt{8a^2+1}+6\sqrt{8b^2+1}+6\sqrt{8c^2+1}.
		\end{array}
		\] 
		Đẳng thức xảy ra khi 
		$
		\heva{&8a+\dfrac{1}{a}=9a\\&8ab+\dfrac{1}{b}=9b\\&8c+\dfrac{1}{c}=9c\\&a+b+c=\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}}
		\Leftrightarrow a=b=c=1.
		$
	}
\end{ex} 
\begin{ex}%[Đề thi vào 10, chuyên Lào Cai, tỉnh Lào Cai, năm 2018]%[Nguyễn Quang Tân, Dự án EX-TS10C-18]%[9D0K5]
	Tìm tất cả các cặp số nguyên $(x;y)$ thỏa mãn $y^2+2xy-3x-2=0$.
\loigiai{
	Giả sử phương trình sau có nghiệm nguyên 
	\[y^2+2xy-3x-2=0\tag{1}\]
	Coi $x$ là ẩn và $y$ là tham số, thì biệt thức $\Delta =4x^2+12x+8$ phải là số chính phương. Tức là
	\[\begin{array}{rl}
	&4x^2+12x+8=k^2, \; k\in\mathbb{N}\\
	\Rightarrow & (2x+3)^2-k^2=1\\
	\Rightarrow & (2x+3-k)(2x+3+k)=1.
	\end{array}\]
	Xét 2 trường hợp 
	\begin{itemize}
		\item Nếu $2x+3-k=2x+3+k=1$ thì $x=-1$, thay vào phương trình (1) ta được $y^2-2y+1=0\Leftrightarrow y=1.$
		\item Nếu $2x+3-k=2x+3+k=-1$ thì $x=-2$, thay vào phương trình (1) ta được $y^2-4y+4=0\Leftrightarrow y=2.$
	\end{itemize}
	Vậy phương trình đã cho có 2 nghiệm $(x;y)=(-1;1)$ và $(x;y)=(-2;2)$.
}
\end{ex}

\begin{ex} %[Đề thi vào 10, chuyên Lào Cai, tỉnh Lào Cai, năm 2018]%[Nguyễn Quang Tân, Dự án EX-TS10C-18]%[9D0G4]
	Cho $m$, $n$ là hai số nguyên thỏa mãn $4(m+n)^2-mn$ chia hết cho $225$. Chứng minh $mn$ cũng chia hết cho $225.$
	\loigiai{
	Đặt $A=4(m+n)^2-mn$ thì $4A=15(m+n)^2+(m-n)^2$.\\
			Vì $4A\,\vdots\, 225$ nên $\heva{&4(m-n)^2\, \vdots \, 3\\ &4(m-n)^2 \,\vdots \, 5.}$\\
			Mà $3$, $5$ là các số nguyên tố nên $m-n$ chia hết cho $3$ và $5$.\\
			Suy ra $(m-n)^2$ chia hết cho $9$ và $25$ mà hai số này nguyên tố cùng nhau nên $(m-n)^2\, \vdots \, 225$.\\
			Do đó $15(m+n)^2\, \vdots \, 225$ nên $(m+n)^2\vdots 15$ nên $m+n$ chia hết cho $3$ và $5$.\\
			Từ đó suy ra: $\heva{& m-n\, \vdots\, 15\\ &m+n\vdots 15} \Rightarrow \heva{& m\, \vdots \, 15\\& n\, \vdots\, 15 }\Rightarrow mn\, \vdots\, 225.$
	}
\end{ex}
