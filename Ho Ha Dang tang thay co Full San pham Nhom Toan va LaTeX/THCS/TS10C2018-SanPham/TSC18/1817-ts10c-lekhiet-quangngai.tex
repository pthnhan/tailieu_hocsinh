\begin{ex}%[Đề thi vào 10, chuyên Lê Khiết, tỉnh Quảng Ngãi, năm 2018]%[Phan Anh Tiến, Dự án EX-TS10C-18]%[9D0K2]%[9D0K2]
\begin{listEX}
\item Cho $x\ne 1,$ hãy rút gọn biểu thức sau $$ A=\dfrac{5x+1}{x^3-1}-\dfrac{1-2x}{x^2+x+1}-\dfrac{2}{1-x}. $$
\item Tìm cặp số thực $(x;y)$ với $y$ lớn nhất thỏa mãn điều kiện $x^2+5y^2+2y-4xy-3=0.$
\item Cho $a,b,c$ là các số thực khác $0$ thỏa điều kiện $\heva{& a^2+a=b^2\\& b^2+b=c^2\\& c^2+c=a^2.}$ 
Chứng minh rằng $(a-b)(b-c)(c-a)=1.$
\end{listEX}
\loigiai{
\begin{listEX}
\item Ta có \begin{align*}
				A&=&\dfrac{5x+1}{(x-1)(x^2+x+1)}+\dfrac{2x-1}{x^2+x+1}+\dfrac{2}{x-1}\\ 
				&=&\dfrac{(5x+1)+(2x-1)(x-1)+2(x^2+x+1)}{(x-1)(x^2+x+1)}\\
				&=&\dfrac{4(x^2+x+1)}{(x-1)(x^2+x+1)}\\
				&=&\dfrac{4}{x-1}.
			\end{align*}
\item Phương trình viết lại thành $x^2-4xy+5y^2+2y-3=0.$ Phương trình có nghiệm khi $\Delta'=-y^2-2y+3\ge 0\Leftrightarrow       -3\le y\le 1.$\\
Vì $y$ lớn nhất nên chọn $y=1,$ thay vào phương trình ta được $x=2.$ Vậy $(2;1)$ là cặp nghiệm cần tìm.
\item Cộng theo vế ta được $a+b+c=0.$ Cộng đẳng thức thứ nhất và thứ hai ta được 
\begin{align*}
a+b&=c^2-a^2=(c-a)(c+a)\\ 
-c&=(-b)(c+a)\\
\end{align*}
Tương tự ta thu được
\begin{align*}
-b&=(-a)(b-c)\\
-a&=(-c)(a-b) 
\end{align*}
Nhân các đẳng thức trên và thu gọn ta được $(a-b)(b-c)(c-a)=1.$
\end{listEX}
	}
\end{ex}

\begin{ex} %[Đề thi vào 10, chuyên Lê Khiết, tỉnh Quảng Ngãi, năm 2018]%[Phan Anh Tiến, Dự án EX-TS10C-18]%[9D0K4]%[9D0G4]
\begin{listEX}
\item Chứng minh rằng với mọi số tự nhiên $n$ thì $n^3-9n+27$ không chia hết cho $3.$
\item Một số nguyên dương được gọi là \textit{số may mắn} nếu số đó bằng $99$ lần tổng tất cả các chữ số của nó. Tìm \textit{số may mắn} đó.
\end{listEX}	
\loigiai{
		\begin{enumerate}
			\item Giả sử tồn tại số tự nhiên $n$ để $n^3-9n+27$ chia hết cho $81.$ Suy ra $n^3-9n+27$ chia hết cho $3,$ hay $n=3k.$ 
\\Khi đó $n^3-9n+27=27(k^3-k+1),$ suy ra $k^3-k+1\vdots 3.$ Nhưng $k^3-k+1=(k-1)k(k+1)+1$ không chia hết cho $3$ với mọi $k.$
\\Vậy với mọi số tự nhiên $n$ thì $n^3-9n+27$ không chia hết cho $81.$ 
			\item Giả sử số cần tìm là $\overline {a_1a_2\cdots a_m}.$ Theo giả thiết $\overline {a_1a_2\cdots a_m}=99(a_1+a_2+\cdots+a_m).$ Ta xét các trường hợp sau:
\begin{itemize}
\item $m\le 3$ kiểm tra trực tiếp không có số may mắn nào.
\item $m\ge 5$ Ta luôn có $\heva{&\overline {a_1a_2\cdots a_m}\ge 10^{m-1}\\&99(a_1+a_2+\cdots+a_m)\le 99.9.m}$ suy ra $10^{m-1}\le 891m.$\\
Do đó khi $m\ge 5$ thì bất đẳng thức trên không còn đúng.
\item $m=4$ Khi đó, $1000a_1+100a_2+10a_3+a_4=99(a_1+a_2+a_3+a_4)$ hay $901a_1+a_2=89a_3+98a_4.$\\
Do $89a_3+98a_4\le(89+98)\times 9=1683$ nên $a_1=1.$ Khi đó $a_3=10-a_4+\dfrac{11+a_2-9a_4}{89}$ suy ra $11+a_2-9a_4=0$ hay $a_2=7, a_4=2, a_3=8 và a_1=1.$ Vậy số cần tìm là $1782.$
\end{itemize}
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Lê Khiết, tỉnh Quảng Ngãi, năm 2018]%[Phan Anh Tiến, Dự án EX-TS10C-18]%[9D0K2]%[9D0B2]
\begin{listEX}
\item Giải phương trình $\sqrt{x+1}+\sqrt{1-3x}=x+2.$
\item Giải hệ phương trình $\heva{& x-2y+xy=2\\ &x^2+4y^2=4.}$
\end{listEX}	
\loigiai{
\begin{listEX} 
\item Điều kiện $-1\le x\le \frac{1}{3}$\\
Phương trình lại \begin{align*}
&&(\sqrt{x+1}-1)+(\sqrt{1-3x}-1)=x	\\ 
&\Leftrightarrow & \dfrac{x}{\sqrt{x+1}+1}-\dfrac{3x}{\sqrt{1-3x}+1}=x\\
&\Leftrightarrow & x(1-\dfrac{1}{\sqrt{x+1}+1}+\dfrac{3}{\sqrt{1-3x}+1})=0\\
&\Leftrightarrow &\left[ \begin{gathered}
  x=0 \hfill \\
  1-\dfrac{1}{\sqrt{x+1}+1}+\dfrac{3}{\sqrt{1-3x}+1}=0 \hfill \\ 
\end{gathered}  \right.
\end{align*}
Mà phương trình $1-\dfrac{1}{\sqrt{x+1}+1}+\dfrac{3}{\sqrt{1-3x}+1}=0$ vô nghiệm, nên nghiệm của phương trình ban đầu là $x=0.$
\item Hệ viết lại thành 
$\begin{cases}
(x-2y)+xy=0&\\
(x-2y)^2+4xy=4.&
\end{cases}$\\
Đặt 
$\begin{cases}(x-2y)=a&\\
xy=b&
\end{cases}$ khi đó ta có hệ 
$\begin{cases}
a+b=2&\\
a^2+4b=4.& 
\end{cases}$
Giải hệ phương trình ta được $\begin{cases}a=2&\\
b=0&
\end{cases}$ $\Leftrightarrow$ $\begin{cases}x-2y=2&\\
xy=0&
\end{cases}$ suy ra $\begin{cases}x=0&\\
y=-1&
\end{cases}$ hoặc $\begin{cases}
x=2&\\
y=0.&
\end{cases}$
\end{listEX}
	}
\end{ex}

	Cho hình vuông $ABCD$ nội tiếp trong đường tròn $(O).$ Gọi $M$ là một điểm bất kỳ trên cạnh $BC$ ($M$ khác $B$ và $C$), $N$ là điểm trên cạnh $CD$ sao cho $BM=CN.$ Gọi $H, I$ là giao điểm của $AM$ với  $BN, DC.$

			\item Đường thẳng $DM$ cắt đường tròn $(O)$ tại $P$ ($P$ khác $D$). Gọi $S$ là giao điểm của $AP$ và $BD.$ Chứng minh $SM$ song song $AC.$
