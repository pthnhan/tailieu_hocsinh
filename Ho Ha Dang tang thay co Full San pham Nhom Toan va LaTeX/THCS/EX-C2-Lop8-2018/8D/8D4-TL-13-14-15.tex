\section{Phương trình chứa ẩn trong dấu giá trị tuyệt đối}
\subsection{Tóm tắt lí thuyết}
Để giải các phương trình chứa ẩn trong dấu giá trị tuyệt đối, cần khử dấu giá trị tuyệt đối. Ta nhớ lại: Giá trị tuyệt đối của một biểu thức bằng chính nó nếu biểu thức không âm, bằng số đối của nó nếu biểu thức âm:
\[ \left|A\right| = \heva{
	&A\: \text{nếu}\: A\ge 0\\
	-&A\: \text{nếu}\: A\le 0 } \]
Do đó, để khử dấu giá trị tuyêt đối, cần xét giá trị của biến làm cho biểu thức không âm hay âm. Nếu biểu thức nằm trong dấu giá trị tuyệt đối là nhị thức bậc nhất, ta cần nhớ định lí sau:\\

\begin{dl}
	Định lí về dấu của nhị thức bậc nhất $ ax+b (a \neq 0) $\\
	Nhị thức $ ax+b (a \neq 0) $
	\begin{itemize}
		\item Cùng dấu với $ a  $ với các giá trị của $ x $ lớn hơn nghiệm của nhị thức;
		\item Trái dấu với $ a $ với các giá trị của $ x $ nhỏ hơn nghiệm của nhị thức.
	\end{itemize}
\end{dl}
\textit{Chứng minh.} Gọi $ x_0 $ là nghiệm của nhị thức $ ax+b (a \neq 0) $ thì $ x_0=-\dfrac{b}{a} $.  Xét
\[
\dfrac{ax+b}{a}=x+\dfrac{b}{a}=x-x_0.
\]
Nếu $ x> x_0 $ thì $ x-x_0>0 \Rightarrow \dfrac{ax+b}{a}>0 \Rightarrow ax+b $ cùng dấu với $ a $.\\
Nếu $ x< x_0 $ thì $ x-x_0<0 \Rightarrow \dfrac{ax+b}{a}<0 \Rightarrow ax+b $ trái dấu với $ a $.\\

Xét ba khoảng giá trị của biến $ x $:
\begin{enumEX}{1}
	\item $ x<-3$: phương trình $ (1) $ có dạng:
	\[ (5-x)-(x+3)=3x-1,
	\]
	tìm được $ x=\dfrac{3}{5} $, loại vì giá trị này không thuộc khoảng đang xét.
	
	\item $ -3\le x<5$: phương trình $ (1) $ có dạng:
	\[ (5-x)+(x+3)=3x-1 \Leftrightarrow -3x=-9
	\]
	tìm được $ x=3$ thuộc khoảng đang xét.
	
	\item $ 5<x$: phương trình $ (1) $ có dạng:
	\[ (x-5)+(x+3)=3x-1 \Leftrightarrow -x=1
	\]
	tìm được $ x=-1$, không thuộc khoảng đang xét.
	
	\item[]\textit{Kết luận:} $ S=\left\lbrace 3\right\rbrace . $
\end{enumEX}
}
\end{vd}

\begin{vd}
	Giải phương trình:
	\[ \left| 2x-1\right| +\left| 2x-5\right| =4. \tag{1}
	\]
\loigiai{
\textit{Cách 1.}
\begin{enumEX}{1}
	\item Xét khoảng $ x<\dfrac{1}{2} $. (1) có dạng:
	\[ 1-2x+5-2x=4,
	\]
tìm được $x=\dfrac{1}{2}$, không thuộc khoảng đang xét.
	
	\item Xét khoảng $ \dfrac{1}{2}\le x \le \dfrac{5}{2}, $ (1) có dạng:
	\[ 1-2x+5-2x=4 \Leftrightarrow 0x=0. \]
	Phương trình nghiệm đúng với mọi $ x $ thuộc khoảng đang xét, tức là \[ \dfrac{1}{2}\le x\le \dfrac{5}{2} .\]

	\item  Xét khoảng $ x>\dfrac{5}{2}, $ (1) có dạng:
	\[ 2x-1+2x+5=4, \]
	tìm được $ x=\dfrac{5}{2} $, không thuộc khoảng đang xét.\\
	\textit{Kết luận:} $ S=\left\lbrace x\: \left| \: \dfrac{1}{2}\le x \le \dfrac{5}{2} \right\rbrace \right.  .$
	
\end{enumEX}
\textit{Cách 2:}\\
Áp dụng hai lần bất đẳng thức $ \left| A\right| \ge A $ (xảy ra đẳng thức khi và chỉ khi $ A \ge 0 $), ta có:
\[
\left| 2x-1\right| +\left\| 5-2x \right| \ge (2x-1)+(5-2x)=4 
\]
Theo đề bài, phải xảy ra đẳng thức, do đó
\[
\heva{
&2x-1 \ge 0\\
&5-2x\ge 0 }
\Leftrightarrow
\heva{
&x \ge \dfrac{1}{2}\\
&x \le \dfrac{5}{2}
}
\Leftrightarrow \dfrac{1}{2}\le x \le \dfrac{5}{2}.
\]
}
\end{vd}

\begin{vd}

	\[ \left| \left|x \right|-3  \right|  =x+1 \tag{1} \] 
\loigiai{
\begin{enumEX}{1}
	\item Xét khoảng $ X \ge 0 $, (1) có dạng
	\[
	\left| x-3 \right| =x+1 \tag{2}
	\]
	Lại xét hai trường hợp:\\
	Với $ x\ge 3$, (2) có dạng $ x-3=x-1 $, vô nghiệm.\\
	Với $ 0\le x <3 $, (2) có dạng $ 3-x=x-1 \Leftrightarrow x=1 $, thuộc khoảng đang xét.
	\item Xét khoảng $ x<0 $, (1) có dạng $ \left|-x-3 \right| =x+1 $ tức là
	\[
	\left| x-3\right| =x+1.\tag{3}
	\]
	Lại xét hai trường hợp:\\
	Với $ -3 \le x <0 ,$ (3) có dạng $ x+3=x+1 $, vô nghiệm.\\
	Với $ x<-3 $, (3) có dạng $ -x-3=x+1 \Leftrightarrow x=-2$, không thuộc khoảng đang xét.\\
	\shortsol{
	\textit{Kết luận}: $ S=\left\lbrace 1\right\rbrace  $
}
\end{enumEX}
}
\end{vd}

\section{Bất phương trình chứa ẩn trong dấu trị tuyệt đối}
\subsection{Tóm tắt lí thuyết}
Để giải bất phương trình loại này, ta cũng khử dấu giá trị tuyệt đối như với phương trình chứa ẩn trong dấu giá trị tuyệt đối.

\begin{vd}
Giải bất phương trình
\[ \left| x\right| -x+2 \le 2\left| x-4\right|. \tag{1} \]

\loigiai{ 
	Lập bảng xét dấu các biểu thức $ x $ và $ x-4 $.
\begin{center}
	\begin{tikzpicture}[scale=1, font=\footnotesize, line join=round, line cap=round, >=stealth]
	\tkzTabInit[nocadre=false,lgt=1.2,espcl=2.5,deltacl=0.6]{$x$/1,$x$ /1, $x-4$ /1}{, $0$ , $4$ ,}
	\tkzTabLine{ ,-, 0 ,+, t ,+ }
	\tkzTabLine{ ,-, t ,-, 0 ,+ }
	\end{tikzpicture}
\end{center}
\begin{enumEX}{1}
	\item Xét khoảng $ x<0 $, (1) có dạng:
	\[ -x-x+2\le 2\left( 4-x\right)   \Leftrightarrow 0x \le 6, \]
	nghiệm đúng với mọi $ x $ thuộc khoảng đang xét $ x<0. $
	\item Xét khoảng $ 0 \le x<4 $, (1) có dạng:
	\[ x-x+2\le 8-2x \Leftrightarrow x \le 3. \]
	Nghiệm của bất phương trình (1) thuộc khoảng đang xét $ 0 \le x \le 3. $
	\item Xét khoảng $ x \ge 4,$ (1) có dạng:
	\[ x-x+2\le 2x-8 \Leftrightarrow x \ge 5, \] thỏa mãn điều kiện $ x \ge 4. $
\end{enumEX}
\shortsol{
		Kết luận: Nghiệm của bất phương trình đã cho là: $ x \le 3; \: x \ge 5. $
	}

}
\end{vd}
\begin{nx}
	Trong cách giải trên, ta khử dấu giá trị tuyệt đối bằng cách xét từng khoảng giá trị của biến. Trong một số trường hợp, có thể giải nhanh hơn cách phương pháp chung nói trên bởi các biến đổi tương đương sau: 
\end{nx}
\textit{Dạng 1:}
\begin{enumerate}
	\item Với $ a $ dương, ta có: $ \left| f(x)\right| <a \Leftrightarrow -a <f(x)<a. $
	\item $ \left| f(x)\right| <g(x) \Leftrightarrow -g(x)<f(x)<g(x). $
\end{enumerate}

\textit{Dạng 2:}
\begin{enumEX}{1}
		\item Với $ a $ là số dương ta có:
		\[
		 \left| f(x)\right| >a \Leftrightarrow 
		 \hoac{
			f(x)&<-a\\
			f(x)&>a
			}
		 \]
		 
		 
		 \item Đối với hàm $ g(x) $
		 \[ \left| f(x)\right| >g(x) \Leftrightarrow  
		\hoac{
		 f(x)&<-g(x)\\
		 f(x)&>g(x).
			}
		  \]
\end{enumEX}

\textit{Dạng 3}
\[ \left| f(x)\right| >\left| g(x)\right|  \Leftrightarrow\left| f(x)\right| ^2>\left| g(x)\right| ^2. \]


\begin{vd}

\begin{array}{l l}

 &\Leftrightarrow -8x<-2 \\

}

\begin{bt}%[Phùng Lê Vũ, EX-C2-8-2018]%[8D4G5]
	Tìm điều kiện của $ x $ để biểu thức sau có giá trị âm:

\[ A=\left(\dfrac{1-x}{x+3} -\dfrac{x+3}{x-1}\right):\left(\dfrac{x+3}{x-1}-\dfrac{x-1}{x+3}\right).\]
	\loigiai{
	$ A=-\dfrac{x^2+2x+5}{4\left(x+1\right)};
	$\\
	$ A<0 \Leftrightarrow x>-1 $ đồng thời $ x \neq 1. $
	}
\end{bt}


\begin{bt}%[Phùng Lê Vũ, EX-C2-8-2018]%[8D4G5]
	Tìm điều kiện của $ x $ và $ y $ để biểu thức sau có giá trị dương:
	\[ A=\left(\dfrac{x^2-xy}{y^2+xy} +\dfrac{x^2-y^2}{x^2+xy}\right):\left(\dfrac{y^2}{x^3-xy^2}+\dfrac{1}{x-y}\right).\]
	\loigiai{
	$ A=\dfrac{\left(x-y\right)^2}{y};
	$\\
	$ A>0 \Leftrightarrow y>0; \: x \neq 0, \: x \neq y. $
}
\end{bt}

\begin{bt}%[Phùng Lê Vũ, EX-C2-8-2018]%[8D4G5]
	Tìm điều kiện của $ x $ và $ y $ để biểu thức sau lớn hơn $ 1 $:
	\[ A=\left(\dfrac{x}{y^2+xy} +\dfrac{x-y}{x^2+xy}\right):\left(\dfrac{y^2}{x^3-xy^2}+\dfrac{1}{x-y}\right):\dfrac{x}{y}.\]
	\loigiai{
		$ A=\dfrac{x-y}{x}=1- \dfrac{y}{x};
		$\\
		$ A>1 \Leftrightarrow xy<0; \: x +y \neq 0. $
	}
\end{bt}
