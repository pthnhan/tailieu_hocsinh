\begin{bt}%[Đề HK1, Thủ Đức, Hồ Chí Minh 17-18]%[Ngọc Diệp]%[0D3Y2-1]%[0D3B2-4]
	Giải các phương trình sau
	\begin{multicols}{2}
		\begin{enumerate}
			\item $ \left|  x^2-5x+4 \right|  =x+4$.
			\item $3x-\sqrt{18x+1}+1=0$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item \begin{align*}
			\left|  x^2-5x+4 \right|  =x+4 \Leftrightarrow \heva {&x+4 \ge 0 \\& { \left[\begin{aligned} & x^2 -5x+4=x+4 \\ & x^2 -5x+4=-x-4 \end{aligned}\right. }} 
			 \Leftrightarrow \heva {&x+4 \ge 0 \\& { \left[\begin{aligned} & x^2 -6x=0 \\ & x^2 -4x+8=0 \end{aligned}\right. }} \Leftrightarrow \heva {&x+4 \ge 0 \\& { \left[\begin{aligned} & x=6 \text{ \,(thỏa mãn)}\\ & x=0 \text{ \,(thỏa mãn)}. \end{aligned}\right. }} 
			\end{align*} 
			Vậy $x=0$, $x=6$ là nghiệm của phương trình.
			\item $$3x-\sqrt{18x+1}+1=0 \Leftrightarrow \left\{ \begin{array}{rcl}
			& 3x+1 &\ge 0\\
			& 18x+1&= \left( 3x+1\right)^2 
			\end{array}\right. \Leftrightarrow \left\{ \begin{array}{rcl}
			& x \ge \dfrac{-1}{3}\\
			& 9x^2 -12x= 0
			\end{array}\right. 
			\Leftrightarrow \left[\begin{aligned} x&=0\\
			x&=\dfrac{4}{3}.  \end{aligned}\right.$$
			Vậy $x=0$, $x=\dfrac{4}{3}$ là nghiệm của phương trình.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề HK1, Thủ Đức, Hồ Chí Minh 17-18]%[Ngọc Diệp]%[0D3B3-3]
	Giải hệ phương trình
	$\begin{cases}
	\dfrac{3}{x-1} +\dfrac{2}{y+5} &= 5  \\
	\dfrac{6}{x-1} - \dfrac{4}{y+5} &= 2. 
	\end{cases}$ \\
	\loigiai{
		Điều kiện xác định: $\begin{cases}
		x &\ne 1  \\
	y&\ne -5. 
		\end{cases}$ \\
Khi đó ta có \\
	$\begin{cases}
\dfrac{3}{x-1} +\dfrac{2}{y+5} &= 5  \\
\dfrac{6}{x-1} - \dfrac{4}{y+5} &= 2 
\end{cases} \Leftrightarrow \begin{cases}
\dfrac{1}{x-1}  &= 1  \\
\dfrac{1}{y+5} &= 1
\end{cases}  \Leftrightarrow \begin{cases}
x  &= 2  \\
y &= -4.
\end{cases}$ \\
Vậy hệ phương trình có nghiệm $(2;-4)$.
}
\end{bt}

\begin{bt}%[Đề HK1, Thủ Đức, Hồ Chí Minh 17-18]%[Ngọc Diệp]%[0D3K2-6]%[0D3K2-5]
	Cho phương trình $x^2 -2(m-1)x +2m-3=0 \, \ \ (1)$ với $m$ là tham số.
	\begin{enumerate}
		\item Chứng minh phương trình $(1)$ luôn có $2$ nghiệm với mọi $m \in \mathbb{R}$.
		\item Tìm $m$ để phương trình $(1)$ có $2$ nghiệm $x_1$, $x_2$ là độ dài các cạnh của $\triangle ABC$ có $\widehat{A}=120^\circ$ và độ dài cạnh $BC=\sqrt{7}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Phương trình $(1)$ có hệ số $a=1\ne0$, do đó $(1)$ là phương trình bậc hai với mọi $m \in \mathbb{R}$. Xét biệt thức
			$$\Delta'= (m-1)^2 -(2m-3) = m^2 -4m+4 = (m-2)^2 \ge 0.$$
			Do đó phương trình $(1)$ luôn có $2$ nghiệm với mọi $m \in \mathbb{R}$.
			\item Giả sử phương trình có $2$ nghiệm $x_1$, $x_2$ là độ dài các cạnh của $\triangle ABC$, $\left( x_1 >0, x_2 >0 \right)$ . Ta có $S=2(m-1)$, $P=2m-3$.\\
			Theo định lý Cosin và giả thiết ta có
			\begin{align*}
			x_1 ^2 + x_2 ^2 - 2 x_1 x_2 \cos 120^\circ =BC^2 &\Rightarrow S^2 -2P +P=7 \\
			&\Rightarrow 4(1-m)^2 -(2m-3)-7 =0 \\
			&\Leftrightarrow 4m^2 -10m =0 \Leftrightarrow \left[\begin{aligned} m&=0\\
			m&=\dfrac{5}{2}.  \end{aligned}\right.
			\end{align*}
			Thử lại với điều kiện $x_1 >0$ và $x_2 >0$, ta có $m= \dfrac{5}{2}$ thỏa mãn yêu cầu bài toán.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề HK1, Thủ Đức, Hồ Chí Minh 17-18]%[Ngọc Diệp]%[0D4K1-5]
	Tìm giá trị nhỏ nhất của biểu thức $f(x)= 2x-3 + \dfrac{8}{x-1}$ với $x>1$.
	\loigiai{
		Áp dụng định lí Cauchy cho $2$ số dương $2(x-1)$ và $\dfrac{8}{x-1}$ ta có
			\begin{align*}
	&f(x)= 2x-3 + \dfrac{8}{x-1} = 2(x-1) + \dfrac{8}{x-1} -1 \ge 2\sqrt{2 \cdot 8} -1 \\
	&\Rightarrow f(x) \ge 7.
		\end{align*}
		Dấu $"="$ xảy ra khi $2(x-1) = \dfrac{8}{x-1} \Leftrightarrow \left[\begin{aligned} x&=3 \text{ \,(thỏa mãn)}\\
		x&=-1 \text{\,(loại)}.  \end{aligned}\right.$\\
		Vậy giá trị nhỏ nhất của $f(x)=7$ khi $x=3$.
	}
\end{bt}

\begin{bt}%[Đề HK1, Thủ Đức, Hồ Chí Minh 17-18]%[Ngọc Diệp]%[0D3B3-5]
	Dung tích phổi của mỗi người phụ thuộc vào một số yếu tố, trong đó hai yếu tố quan trọng là chiều cao và độ tuổi. Công thức ước tính dung tích chuẩn phổi ở nam giới: $$P = 0,057h-0,022a-4,23.$$
	Trong đó: \begin{enumerate}
		\item [] $h$: chiều cao tính bằng cen-ti-mét (cm).
		\item [] $a$: tuổi tính bằng năm.
			\item [] $P$: dung tích chuẩn phổi tính bằng lít (l).
	\end{enumerate}
Bạn Huy năm nay $16$ tuổi, chiều cao của bạn Huy (tính bằng cen-ti-mét) là một số tự nhiên có $3$ chữ số, trong đó chữ số hàng trăm là $1$, chữ số hàng chục hơn chữ số hàng đơn vị là $5$ và năm lần chữ số hàng đơn vị lớn hơn chữ số hàng chục là $3$. Hãy tính dung tích chuẩn phổi của bạn Huy.
	\loigiai{
	Gọi $x$, $y$ lần lượt là chữ số hàng chục và chữ số hàng đơn vị của dung tích chuẩn phổi của bạn Huy. Theo bài ta ta có
	$$\begin{cases}
	b  &= c+5  \\
	5c &= b+3
	\end{cases} \Leftrightarrow \begin{cases}
	b  &= 7  \\
	c &= 2.
	\end{cases} $$
	Vậy chiều cao của bạn Huy là $172$ cm. Do đó  dung tích chuẩn phổi của bạn Huy là
	$$P=  0,057h-0,022a-4,23 \Rightarrow P= 0,057\cdot 172-0,022 \cdot 16-4,23 = 5,222  \text{ (l)}. $$
	}
\end{bt}

\begin{bt}%[Đề HK1, Thủ Đức, Hồ Chí Minh 17-18]%[Ngọc Diệp]%[0H1Y4-3]%[0H2K2-3]%[0H2B2-4]
	Trong mặt phẳng với hệ tọa độ $Oxy$, cho $\triangle ABC$ có $A\left(3;-5 \right) $, $B\left( -3;3\right) $, $C\left( -1;-8\right) $.	
	\begin{enumerate}
		\item Tam giác $\triangle ABC$ là tam giác gì? Tính diện tích $\triangle ABC$ và xác định tâm $I$ của đường tròn ngoại tiếp $\triangle ABC$.
		\item Tìm điểm $M$ trên tia $Oy$ sao cho $\vec{MA} \cdot \vec{MB} =-9$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $\vec{AB}= \left( -6;8\right) $, $\vec{AC}= \left( -4;-3\right) $, suy ra $$\vec{AB} \cdot \vec{AC} = (-6)\cdot (-4) + 8\cdot (-3) =0 \Rightarrow \vec{AB} \perp \vec{AC}.$$
			Do đó tam giác $ABC$ vuông tại $A$.\\
			Ta có $AB=10$, $AC=5$, diện tích tam giác $ABC$ là 
			$$ S_{ABC} = \dfrac{1}{2} AB \cdot AC = \dfrac{1}{2} 10 \cdot 5 =25. $$
			Vì tam giác $ABC$ vuông tại $A$ nên tâm $I$ của đường tròn ngoại tiếp tam giác $ABC$ là trung điểm của $BC$, do đó $I \left( -2 ; \dfrac{-5}{2} \right) $.
			\item Vì $M$ thuộc tia $Oy$ nên $M(0;y)$, $y>0$. Ta có $\vec{MA} = (3;-5-y)$, $\vec{MB} = (-3;3-y)$, suy ra 
			$$\vec{MA} \cdot \vec{MB} =-9 \Leftrightarrow -9 + (y+5) (y-3) =-9 \Leftrightarrow \left[\begin{aligned} y&=-5 \text{ \,(loại)}\\
			y&=3 \text{ \,(thỏa mãn)}.  \end{aligned}\right.$$
			Vậy $M(0;3)$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề HK1, Thủ Đức, Hồ Chí Minh 17-18]%[Ngọc Diệp]%[0H2B3-1]
	Cho hình bình hành $ABCD$ có $AB=1$, $AD=2$, $AC=2BD$. Tính độ dài cạnh $AC$ và diện tích tam giác $ABC$.
	\loigiai{
				\immini{
					Gọi $E$ là tâm hình bình hành $ABCD$. \\Ta có $AC=2AE=2BD \Rightarrow AE=BD=x$, $x>0$. \\
					Áp dụng công thức đường trung tuyến trong tam giác $ABD$ ta có}{	 
			\begin{tikzpicture}[scale=0.6]
			\tkzInit[xmin=-1, xmax=8, ymin=-1, ymax=4]
			\tkzClip
			\tkzDefPoints{0/0/A,1/3/B,7/3/C,6/0/D}
			\tkzDefMidPoint(B,D)\tkzGetPoint{E}
			\tkzDrawPoints(A,B,C,D,E)
			\tkzDrawSegments(A,B B,C C,A C,D A,D D,B)
			\tkzLabelPoints[above](E,B)
			\tkzLabelPoints[below](D)
			\tkzLabelPoints[below left](A)
			\tkzLabelPoints[above right](C)
			\end{tikzpicture} }
		\begin{align*}
		AE^2 = \dfrac{2(AB^2 +AD^2)-BD^2}{4} \Rightarrow x^2 = \dfrac{2(1+4)-x^2}{4} \Leftrightarrow x^2 =2 \Rightarrow x= \sqrt{2} \Rightarrow AC=2\sqrt{2}.
		\end{align*}
		Ta có $p= \dfrac{AB+BC+CA}{2} = \dfrac{3+2\sqrt{2}}{2} $. \\
		Diện tích tam giác $ABC$ là $$S= \sqrt{p(p-a)(p-b)(p-c)} = \dfrac{\sqrt{7}}{4}.$$
	}
\end{bt}
