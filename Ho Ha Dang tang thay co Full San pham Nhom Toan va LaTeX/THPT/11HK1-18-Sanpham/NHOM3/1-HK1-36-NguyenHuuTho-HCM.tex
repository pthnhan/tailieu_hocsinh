\begin{bt}%[Đề thi Học Kì 1, 2017 - 2018 trường THPT Nguyễn Hữu Thọ, Hồ Chí Minh]%[1D1Y2-1]
	Giải phương trình lượng giác $2\sin \left(3x - \dfrac{\pi}{12}\right) + \sqrt{2} = 0$
\loigiai{
	\begin{eqnarray*}
	&&2\sin \left(3x - \dfrac{\pi}{12}\right) + \sqrt{2} = 0\\
	&\Leftrightarrow &\sin \left(3x - \dfrac{\pi}{12}\right) = -\dfrac{\sqrt{2}}{2}\\
	&\Leftrightarrow &\sin \left(3x - \dfrac{\pi}{12}\right) = \sin \left(-\dfrac{\pi}{4}\right)\\
	&\Leftrightarrow &\hoac{&3x - \dfrac{\pi}{12} = -\dfrac{\pi}{4} + k2\pi\\&3x - \dfrac{\pi}{12} = \dfrac{5\pi}{4} + k2\pi}\\
	&\Leftrightarrow &\hoac{&x=-\dfrac{\pi}{18}+k\dfrac{2\pi}{3}\\&x = \dfrac{4\pi}{9} + k\dfrac{2\pi}{3}}(k\in \mathbb{Z}).
	\end{eqnarray*}
	Vậy phương trình có các nghiệm là $x=-\dfrac{\pi}{18}+k\dfrac{2\pi}{3},x = \dfrac{4\pi}{9} + k\dfrac{2\pi}{3},k\in \mathbb{Z}.$
}
\end{bt}
\begin{bt}%[Đề thi Học Kì 1, 2017 - 2018 trường THPT Nguyễn Hữu Thọ, Hồ Chí Minh]%[1D1B3-1]
	Giải phương trình lượng giác $4\sin^2 2x - 4\sin 2x - 3 = 0$ với $x \in (0;5)$.
\loigiai{
	Đặt $t = \sin 2x \Rightarrow t \in [-1;1]$.\\
	Phương trình trở thành $4t^2 - 4t - 3=0 \Rightarrow \hoac{&t = -\dfrac{1}{2}\\&t = \dfrac{3}{2} \text{ (loại)}}$.\\
	Với $t = -\dfrac{1}{2} \Leftrightarrow \sin 4x = -\dfrac{1}{2} \Leftrightarrow \hoac{&2x = -\dfrac{\pi}{6}+k2\pi\\&2x = \dfrac{7\pi}{6}+k2\pi} \Leftrightarrow \hoac{&x = -\dfrac{\pi}{12}+k\pi\\&x = \dfrac{7\pi}{12}+k\pi}(k\in \mathbb{Z}).$\\
	Vì $x \in (0;5)$ nên $x$ nhận các giá trị $\dfrac{11\pi}{12}$, $\dfrac{7\pi}{12}$, $\dfrac{19\pi}{12}$.
}
\end{bt}

\begin{bt}%[Đề thi Học Kì 1, 2017 - 2018 trường THPT Nguyễn Hữu Thọ, Hồ Chí Minh]%[1D2B3-2]
Tìm hệ số của $x^{26}$ trong khai triển $\left(\dfrac{1}{x^4}+x^7\right)^{10}$ với $x \neq 0$.
\loigiai{
	Ta có $\left(\dfrac{1}{x^4}+x^7\right)^{10} = \left(x^{-4}+x^7\right)^{10} = \displaystyle\sum\limits_{k=0}^{10}\mathrm{C}_{10}^k(x^{-4})^{10-k}x^{7k} = \sum\limits_{k=0}^{10}\mathrm{C}_{10}^k x^{11k-40}$.\\
	Để tìm hệ số của $x^{26}$ trong khai triển ta tìm $k$ thỏa mãn $11k-40 = 26 \Leftrightarrow k = 6$.\\
	Vậy hệ số cần tìm là $\mathrm{C}^{6}_{10} = 210$.
}
\end{bt}

\begin{bt}%[Đề thi Học Kì 1, 2017 - 2018 trường THPT Nguyễn Hữu Thọ, Hồ Chí Minh]%[1D2K3-2]
Tìm hệ số của $x^{4}$ trong khai triển đa thức $P(x) = \left(1-x-3x^3\right)^{10}$.
\loigiai{
	Ta có
	\begin{eqnarray*}
	&P(x) &= [1-(x+3x^3)]^{10} \\
	&&= \displaystyle\sum\limits_{k=0}^{10}\mathrm{C}_{10}^k(-1)^k(x+3x^3)^k\\
	&&=\sum\limits_{k=0}^{10}\mathrm{C}_{10}^k(-1)^k x^k (1+3x^2)^k \\
	&&= \sum\limits_{k=0}^{10}\sum\limits_{l=0}^k\mathrm{C}_{10}^k\mathrm{C}_{k}^l (-1)^k3^l x^{k+2l}.
	\end{eqnarray*}
	Số hạng chứa $x^4$ tương ứng với $\heva{&k+2l = 4\\&0 \leq k \leq 10\\&0 \leq l \leq k} \Leftrightarrow \hoac{&\heva{&k=4\\&l=0}\\&\heva{&k=2\\&l=1}}$.\\
	Vậy hệ số chứa $x^4$ là $\mathrm{C}_{10}^4\cdot\mathrm{C}_4^0 + 3\mathrm{C}_{10}^2\cdot\mathrm{C}_2^1 = 480$.
}
\end{bt}

\begin{bt}%[Đề thi Học Kì 1, 2017 - 2018 trường THPT Nguyễn Hữu Thọ, Hồ Chí Minh]%[1D3Y5-2]%[1D3Y5-2]
Trong một thùng có 24 lon nước ngọt Pepsi, trong đó có 2 lon bên trong nắp khoen ghi ``Chúc mừng bạn đã trúng thưởng một Iphone 8''. Anh Quân được chọn 2 lon bất kỳ trong thùng. Tính xác suất để:
\begin{enumerate}
\item Cả hai lon được chọn không trúng thưởng.
\item Cả hai lon được chọn đều trúng thưởng.
\end{enumerate}
\loigiai{
	Ta có $n(\Omega) = \mathrm{C}_{24}^2 = 276$.
	\begin{enumerate}
	\item Gọi A là biến cố ``2 lon lấy ra không trúng thưởng''. Ta có $n(A) = \mathrm{C}_{22}^2 = 231$.\\
	Do đó $P(A) = \dfrac{231}{276} = \dfrac{77}{92}$.
	\item Gọi B là biến cố ``2 lon lấy ra đều trúng thưởng''. Ta có $n(B) = \mathrm{C}_2^2 = 1$.\\
	Do đó $P(B)= \dfrac{1}{276}$.
	\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề thi Học Kì 1, 2017 - 2018 trường THPT Nguyễn Hữu Thọ, Hồ Chí Minh]%[1D3B3-2]
Tìm số hạng đầu và công sai của cấp số cộng $(u_n)$ biết $\heva{&u_3+u_5 = 14\\&S_{12} = 129}$.
\loigiai{
	Ta có $$\heva{&u_3+u_5 = 14\\&S_{12} = 129}\Leftrightarrow \heva{&(u_1+2d) + (u_1+4d ) =14\\&\dfrac{12(2u_1+11d)}{2} = 129} \Leftrightarrow \heva{&2u_1+6d = 14\\&12u_1 + 66d = 129} \Leftrightarrow \heva{&u_1 = \dfrac{5}{2}\\&d = \dfrac{3}{2}}.$$
Vậy cấp số cộng $(u_n)$ có số hạng đầu $u_1 = \dfrac{5}{2}$ và công sai $d = \dfrac{3}{2}$.
}
\end{bt}

Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình bình hành tâm $O$. Gọi $M$, $N$ lần lượt là trung điểm của $SA$ và $CD$.
