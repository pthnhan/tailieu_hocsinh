\section{Một bài toán phân tích đa thức}
\begin{center}
	\textbf{\color{violet}{Lưu Bá Thắng- Tạp chí Pi tháng 1 năm 2017}}
\end{center}

\begin{bt}[Kì thi học sinh giỏi quốc gia, Japan, 1999]
Cho số nguyên dương $n$, chứng minh rằng đa thức $$ \left( {x^2+1^2} \right)\left( {x^2+2^2} \right)\cdots\left( {x^2+n^2} \right)+1$$ không phân tích được thành tích của hai đa thức với hệ số nguyên có bậc lớn hơn hay bằng $1$. 
\loigiai{Đặt $f(x)=\left( {x^2+1^2} \right)\left( {x^2+2^2} \right)\cdots\left( {x^2+n^2} \right)+1$. Giả sử $f(x)=g(x)h(x)$ với $ g(x), h(x) \in \mathbb{Z}(x)$ và $\deg g(x), \deg h(x) \ge 1$. Do $f(x)$ là đa thức có hệ số bậc cao nhất bằng $1$ nên ta có thể giả sử hệ số bậc cao nhất của $g(x), h(x)$ cũng bằng $1$. Do $f(ki)=1$ với $k=-n, -n+1,\dots,-1, 1,\dots, n$ và $i$ là số phức đơn vị nên $g(ki)h(ki)=1$ với $k=-n, -n+1,\dots,-1, 1, 2,\dots, n$. Chú ý rằng $g(ki)$ và $h(ki)$ là các số có dạng $a+bi$ với $a,b \in \mathbb{Z}$ nên từ $g(ki)h(ki)=1$ ta suy ra 
	$$\left( {g(ki), h(ki)} \right) \in \left\{ {(-1,-1), (1,1), (i,-i), (-i,i)} \right\}.$$
Do đó $g(ki)=\overline {h(ki)}$, với $\overline {h(ki)}$ là liên hợp của $h(ki)$, với mọi $k=-n, -n+1,\dots,n$. Ta có $\overline {h(ki)}=h(\overline{ki})=h(-ki)$ nên $g(ki)=h(-ki)$. Từ đó, đa thức $k(x)=g(x)-h(-x)$ là đa thức có hệ số nguyên và có bậc nhỏ hơn $2n$, nhận $2n$ phần tử phân biệt 
$$-ni, (-n+1)i,\dots,-i,i,2i,\dots,ni$$
làm nghiệm nên $k(x) \equiv 0$ hay $g(x)=-h(x)$ với mọi $x$. Suy ra $g(0)=h(0)$. Và do đó $f(0)=g(0)^2$ là số chính phương. Mà $f(0)=(n!)^2+1$ nên $ (n!)+1$ là số chính phương (mâu thuẫn). Vậy ta có điều phải chứng minh.
}
\end{bt}
Từ bài toán trên ta đặt câu hỏi
\begin{bt}
	Cho số nguyên dương $n$. Đa thức 
	$$ \left( {x+1^2} \right)+\left( {x+2^2} \right)\cdots\left( {x+n^2} \right)+1$$
	có phân tích được thành tích của hai đa thức với hệ số nguyên có bậc lớn hơn hay bằng $1$ hay không?
	\loigiai{Câu trả lời là đa thức trên không phân tích được thành tích của hai đa thức với hệ số nguyên có bậc lớn hơn hay bằng $1$. Thật vậy, đặt $f(x)=\left( {x+1^2} \right)\left( {x+2^2} \right)\cdots+\left( {x+n^2} \right)+1$. Giả sử $f(x)=g(x)h(x)$ với $g(x),h(x) \in \mathbb{Z}(x)$ và $\deg g(x) \ge 1, \deg h(x) \ge 1$. Do $f(x)$ là đa thức có hệ số bậc cao nhất bằng $1$ nên ta có thể giả sử hệ số bậc cao nhất của $g(x), h(x)$ cũng bằng $1$. Do $f(-k^2)=1$ nên $g(-k^2)h(-k^2)=1$ với $k=1,2,\dots,n$. Do $g(-k^2), h(-k^2) \in \mathbb{Z}$ nên ta suy ra $g(-k^2)=h(-k^2)=1$ hoặc $-1$. Do đó đa thức $k(x)=g(x)-h(x)$ là đa thức có hệ số nguyên và có bậc nhỏ hơn $n$, nhận $n$ phần tử phân biệt $-1^2,-2^2,\dots,-n^2$ làm nghiệm nên $k(x)\equiv 0$ hay $g(x)=h(-x)$ với mọi $x$. Suy ra $g(0)=h(0)$. Và do đó $f(0)=g(0)^2$ là số chính phương. Mà $f(0) = (n!)^2+1 $ là số chính phương (mâu thuẫn).		}
\end{bt}
Tiếp tục ta lại đặt câu hỏi

\begin{bt}
Cho số nguyên dương $n$. Đa thức 	$$ \left( {x^4+1^2} \right)\left( {x^4+2^2} \right)\cdots\left( {x^4+n^2} \right)+1$$
có phân tích được thành tích của hai đa thức với hệ số nguyên có bậc lớn hơn hay bằng $1$ hay không?
	\loigiai{Rõ ràng với cách tiếp cận như cách giải hai bài toán $1$ và $2$ chúng ta dường như sẽ không thể trả lời cho bài toán trên. Liệu có một cách tiếp cận khác? Tình cờ chúng tôi phát hiện ra một Bổ đề, đó là bài toán thi chọn đội tuyển IMO của Rumania năm 2003. nội dung của bài toán như sau.
		
\begin{bode}[TST Romania 2003]
	Cho $f(x)$ là đa thức với hệ số nguyên với $\deg f(x) \ge 1$, có hệ số bậc cao nhất bằng $1$, không phân tích được thành tích của hai đa thức với hệ số nguyên có bậc lớn hơn hay bằng $1$. Giả sử rằng $|f(0)|$ là số không chính phương. Khi đó $f(x^2)$ cũng không phân tích được thành tích của hai đa thức với hệ số nguyên có bậc lớn hơn hay bằng $1$.
\end{bode}
\noindent{\bf Chứng minh.} Giả sử $f(x^2)=g(x)h(x)$ với $g(x), h(x) \in \mathbb{Z}$ và $\deg g(x) \ge 1, \deg h(x) \ge 1$. Do $f(x)$ là đa thức có hệ số bậc cao nhất bằng $1$ nên ta có thể giả sử hệ số bậc cao nhất của $g(x), h(x)$ cũng bằng 1. Ta có $f(x^2)=g(-x)h(-x)$ nên $g(x)g(-x)h(x)h(-x)=f(x^2)^2$. Do $g(x)g(-x)$ và $h(x)h(-x)$ là các hàm chẵn nên chúng là các đa thức không có hạng tử bậc lẻ, suy ra tồn tại $r(x), s(x) \in \mathbb{Z}[x]$ sao cho $r(x^2)=g(x)g(-x), s(x^2)=h(x)h(-x)$. Suy ra $f(x)^2=r(x)s(x)$ và do đó $f(x)=r(x)=s(x)$ với mọi $x$ hoặc $-f(x)=r(x)=s(x)$ với mọi $x$. Từ đó suy ra $s(x)=r(x)$, với mọi $x$ hay $g(x)g(-x)=h(x)h(-x)$ với mọi $x$. Do đó $g(0)^2=h(0)^2$, nên $|f(0)|=h(0)^2$ là số chính phương (mâu thuẫn). Vậy điều giả sử là sai và ta có điều phải chứng minh.\\
Chú ý rằng điều kiện $|f(0)|$ là số không chính phương là chặt chẽ vì nếu $|f(0)|$ là số chính phương thì ta có phản ví dụ cho thấy Bổ đề là không đúng. Ví dụ $f(x)=x^2+4$ không phân tích được thành tích của hai đa thức với hệ số nguyên có bậc lớn hơn hay bằng $1$ nhưng đa thức 
$$f(x^2)=x^4+4=(x^2+2)^2-4x^2=(x^2-2x+2)(x^2+2x+2).$$
Từ bổ đề trên, kết hợp với Bài toán $2$, ta suy ra Bài toán $1$ và Bài toán $3$ bằng cách xét đa thức $$f(x)=(x+1^2)(x+2^2)\dots(x+n^2)+1.$$ Từ cách tiếp cận này chúng ta có định lý sau:

\begin{dl}
	Với mỗi số nguyên dương $n$ và số tự nhiên $k$, đa thức 
	$$ \left( {x^{2^k}+1^2} \right)\left( {x^{2^k}+2^2} \right)\cdots\left( {x^{2^k}+n^2} \right)+1$$
	không phân tích được thành tích của hai đa thức với hệ số nguyên có bậc lớn hơn hay bằng $1$.
\end{dl}
Qua các phân tích trên, chúng ta thấy rằng một bài toán có thể có nhiều cách tiếp cận khác nhau. Mỗi cách cho chúng ta một trải nghiệm và khám phá thú vị. Hi vọng bài viết này sẽ giúp các thầy cô và các em trong việc giảng dạy và học Toán sơ cấp một cách khoa học và năng động hơn.
	}

\end{bt}
