\begin{ex}%[Đề thi thử vào lớp 10 năm học 2018-2019, Trường THCS - THPT Lương Thế Vinh, Hà Nội]%[Nguyễn Thị Hồng Nhung, EX-THCS05]%[9D1B8]%[9D1B8]%[9D1K8]
	Cho biểu thức $P=\dfrac{\sqrt{x}-2}{\sqrt{x}-3}+\dfrac{\sqrt{x}+1}{\sqrt{x}+3}+\dfrac{x-4\sqrt{x}-9}{9-x}; Q=\dfrac{\sqrt{x}+5}{3-\sqrt{x}}$ với $x\geq 0;x\ne 9.$
	\begin{enumerate}
		\item Rút gọn biểu thức $P.$
		\item Tìm  $x$ sao cho  $P=3$.
		\item Đặt   $M=P\div Q$. Tìm giá trị của $x$ để   $|M|<\dfrac{1}{2}$.
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Với $x\geq 0; x\ne 9$ ta có\\ $\begin{aligned} P&=\dfrac{{\left(\sqrt{x}-2\right)\cdot \left(\sqrt{x}+3 \right)}+{\left(\sqrt{x}+1\right)\cdot \left(\sqrt{x}-3 \right)}-x+4\sqrt{x}+9}{\left(\sqrt{x}-3\right)\cdot \left(\sqrt{x}+3 \right)}\\&=\dfrac{x+\sqrt{x}-6+x-2\sqrt{x}-3-x+4\sqrt{x}+9}{\left(\sqrt{x}-3\right)\cdot \left(\sqrt{x}+3 \right)} \\&=\dfrac{x+3\sqrt{x}}{\left(\sqrt{x}-3\right)\cdot \left(\sqrt{x}+3 \right)}\\&=\dfrac{{\sqrt{x}\cdot \left(\sqrt{x}+3 \right)}}{\left(\sqrt{x}-3\right)\cdot \left(\sqrt{x}+3 \right)}=\dfrac{\sqrt{x}}{\sqrt{x}-3}. \end{aligned}$
		\item Ta có   
		$P=3 \Leftrightarrow \dfrac{\sqrt{x}}{\sqrt{x}-3} = 3 \Rightarrow \sqrt{x}=3\sqrt{x}-9\Leftrightarrow \sqrt{x}=\dfrac{9}{2} \Leftrightarrow x= \dfrac{81}{4}$ (Thỏa mãn).
			\item Ta có\\ $\begin{aligned} &M=P\div Q=\dfrac{\sqrt{x}}{\sqrt{x}-3}\div\dfrac{\sqrt{x}+5}{3-\sqrt{x}} = \dfrac{\sqrt{x}}{\sqrt{x}-3}\cdot\dfrac{3-\sqrt{x}}{\sqrt{x}+5} =\dfrac{-\sqrt{x}}{\sqrt{x}+5}.\\ &|M|<\dfrac{1}{2}\Leftrightarrow \dfrac{\sqrt{x}}{\sqrt{x}+5}<\dfrac{1}{2}\Leftrightarrow \dfrac{\sqrt{x}}{\sqrt{x}+5}-\dfrac{1}{2}<0 \Leftrightarrow \dfrac{\sqrt{x}-5}{\sqrt{x}+5}<0\\
			\Leftrightarrow &\sqrt{x}-5<0\Leftrightarrow \sqrt{x}<5\Leftrightarrow x<25.\end{aligned}$
	\end{enumerate}
 Vậy $25>x\geq 0; x\ne 9$ thì $|M|<\dfrac{1}{2}$.
}
\end{ex}
\begin{ex}%[Đề thi thử vào lớp 10 năm học 2018-2019, Trường THCS - THPT Lương Thế Vinh, Hà Nội]%[Nguyễn Thị Hồng Nhung, EX-THCS05]%[9D3K6]%[9D4K8]
	\textbf{\textit{Giải bài toán bằng cách lập phương trình hoặc hệ phương trình}}\\
	Hai vòi nước cùng chảy vào một bể nước cạn (không có nước) trong $1$ giờ $12$ phút thì đầy bể. Nếu mở vòi thứ nhất chảy trong $30$ phút và vòi thứ hai chảy trong $1$ giờ thì được $\dfrac{7}{12}$ bể. Hỏi mỗi vòi chảy một mình thì sau bao lâu đầy bể?
	\loigiai{
	Gọi thời gian vòi thứ nhất chảy một mình đầy bể là $x$ (giờ), thời gian vòi thứ hai chảy một mình đầy bể là $y$ (giờ). Điều kiện: $x,y>1{,}2$.\\
	Trong $1$ giờ,
	\begin{itemize}
		\item vòi thứ nhất chảy được $\dfrac{1}{x}$ (bể);
		\item vòi thứ hai chảy được: $\dfrac{1}{y}$ (bể);
		\item cả hai vòi chảy được $\dfrac{1}{1{,}2}=\dfrac{5}{6}$ (bể). 
	\end{itemize}
	Suy ra $\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{5}{6}$.\\
	Vì mở vòi thứ nhất chảy trong $30$ phút ($0{,}5$ giờ) và vòi thứ hai chảy trong $1$ giờ thì được $\dfrac{7}{12}$ bể nên ta có phương trình $\dfrac{0{,}5}{x}+\dfrac{1}{y}=\dfrac{1}{2x}+\dfrac{1}{y}=\dfrac{7}{12}.$\\	
	Ta có hệ phương trình $\begin{cases}
	\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{5}{6}\\
	\dfrac{1}{2x}+\dfrac{1}{y}=\dfrac{7}{12}.
	\end{cases}$\\
	Giải hệ phương trình được nghiệm $x=2$, $y=3$ (thỏa mãn đều kiện).\\
	Vậy thời gian vòi thứ nhất chảy một mình đầy bể là $2$ (giờ), thời gian vòi thứ hai chảy một mình đầy bể là $3$ (giờ).
	}
\end{ex}
