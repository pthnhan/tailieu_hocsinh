\subsection{Tính các đại lượng hình học bằng cách lập phương trình}
Phương trình đại số là công cụ giúp chúng ta giải quyết nhiều bài toán hình học, nhất là bài toán về tính toán.
\begin{vd}%[Nguyễn Tất Thu, EX-C2-L8-2018]%[8H2K3]
	Tính diện tích của tam giác có độ dài ba cạnh bằng $10$ cm, $17$ cm, $21$ cm.
	\loigiai{
	\immini{	Xét $\triangle ABC$ có $AB=10$ cm, $AC=17$ cm, $BC=21$ cm. Gọi $AH$ là đường cao của tam giác. Vì $BC$ là cạnh lớn nhất của tam giác nên $\widehat{B},\ \widehat{ C} <90^\circ$, do đó $H$ nằm trong đoạn $BC$.
		Đặt $HC=x,\ HB=y$ ta có $$x+y=21. \eqno (1)$$
		Mặt khác $AH^2=10^2-y^2$, $AH^2=17^2-x^2$ nên
		$$x^2-y^2=17^2-10^2=189. \eqno (2)$$
		Từ (1) và (2) ta có $x+y=21$ và $x-y=9$. Do đó $x=15,\ y=6$. Ta có $AH=10^2-6^2=64$, nên $AH=8$ cm.

\begin{vd}%[Nguyễn Tất Thu, EX-C2-L8-2018]%[8H2G3] 
	Cho tam giác $ABC$ có $BC=a,\ CA=b,\ AB=c,$ diện tích $S$. Chứng minh rằng $S=\sqrt{p(p-a)(p-b)(p-c)}$ với $p$ là nửa chu vi của tam giác (công thức Hê-rông).
	\loigiai{ 
	\immini{	Giả sử $a\geq b\geq c$. Vẽ đường cao $AH$. Do $a$ là cạnh lớn nhất của tam giác nên $B$ và $C$ là các góc nhọn, do đó $H$ nằm giữa $B$ và $C$ ( chú ý nếu không sắp xếp $a\geq b\geq c$ thì phải xét hai trường hợp: $H$ thuộc hoặc không thuộc đoạn $BC$, phức tạp hơn).\\
		Trước hết ta tính $CH$ theo $a,\ b,\ c$. Đặt $CH=x$, ta có $AB^2-BH^2=AC^2-CH^2$ (cùng bằng $AH^2$), do đó:
		$$\begin{aligned} c^2-(a-x)^2=b^2-x^2& \Rightarrow 2ax =a^2+b^2+c^2\\ & \Rightarrow x=\dfrac{a^2+b^2-c^2}{2a}.  \end{aligned}$$}
		{\begin{tikzpicture}
		\tkzInit[ymin=-1.5,ymax=3.5,xmin=-1,xmax=5.5]
		\tkzClip
		\tkzSetUpPoint[color=black,fill=black,size=6]

		\tkzDrawSegments(A,B B,C C,A A,H)

		\node at (2.5,2.4) [below]{$b$};
		\node at (2.4,-0.5) [below]{$a$};
		\draw[->] (2.7,-0.75)--(5,-0.75);
		\draw[<-] (0,-0.75)--(2.1,-0.75);
		\end{tikzpicture}\\ \centering \textit{Hình 39}}
	
		Trong tam giác $AHC$ ta có:
		$$\begin{aligned}    AH^2 & =AC^2-CH^2=b^2-\left(\dfrac{a^2+b^2-c^2}{2a} \right)^2\\ & = \dfrac{4a^2b^2-(a^2+b^2-c^2)^2}{4a^2}.\end{aligned} $$
		Suy ra 
		$$S^2=\left(\dfrac{1}{2}AH\cdot BC \right)^2=\dfrac{4a^2b^2-(a^2+b^2-c^2)^2}{16}.$$
		Mà 
		$$\begin{aligned}4a^2b^2-(a^2+b^2-c^2)^2 & = (2ab+ a^2+b^2-c^2)(2ab-a^2-b^2+c^2) \\ & = \left[(a+b)^2-c^2 \right] \left[c^2-(a-b)^2 \right] \\ & = (a+b+c)(a+b-c)(a+c-b)(c+b-a)\\ &  =16p(p-a)(p-b)(p-c).\end{aligned}$$
		Vậy $S=\sqrt{p(p-a)(p-b)(p-c)}.$
		}
	\end{vd}
