\begin{bt}%[Thi GHK, Tây Thạnh - Hồ Chí Minh, 2018]%[Trương Quan Kía, EX-DT-10&11]%[1D1K2-1][1D1B3-1][1D1K3-3][1D1B3-2][1D1G3-5]
Giải các phương trình lượng giác sau:
\begin{enumerate}
	\item $ \cot \left( 2x-\dfrac{\pi}{3}\right) =\sqrt{3} $ với $ 0<x<2\pi $
	\item $ \cos 4x -4\sin^2 x+3=0 $
	\item $ 2\sin x\left( \sin x -\sqrt{3}\cos x\right) =1-2\sin x $
	\item $ 3\cos^2 x-\sin 2x +2\sin^2 x=\dfrac{3}{2} $
	\item $ \cos 2x+\sin 2x+3=\sqrt{2}\left( 3\cos x+\sin x\right)  $
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $ \cot \left( 2x-\dfrac{\pi}{3}\right) =\sqrt{3}\Leftrightarrow 2x-\dfrac{\pi}{3}=\dfrac{\pi}{6}+k\pi\Leftrightarrow x=\dfrac{\pi}{4}+\dfrac{k\pi}{2},k\in \mathbb{Z} .$\\
	Vì $ 0<x<2\pi\Leftrightarrow 0<\dfrac{\pi}{4}+\dfrac{k\pi}{2}<2\pi\Leftrightarrow -\dfrac{1}{2}<k<\dfrac{7}{2},k\in \mathbb{Z}\Leftrightarrow k\in \left\lbrace0;1;2;3\right\rbrace $.\\
	Vậy $ x\in \left\lbrace\dfrac{\pi}{4};\dfrac{3\pi}{4};\dfrac{5\pi}{4};\dfrac{7\pi}{4}\right\rbrace  $.
	\item $ \cos 4x -4\sin^2 x+3=0\Leftrightarrow 2\cos^2 2x -1-4\cdot\dfrac{1-\cos 2x}{2}+3=0 $\\
	$ \Leftrightarrow \cos 2x\left( \cos 2x+1\right) =0\Leftrightarrow \hoac{&\cos 2x=0\\&\cos 2x+1=0}\Leftrightarrow \hoac{&x=k\pi\\&x=\dfrac{\pi}{2}+k\pi.} (k\in \mathbb{Z})$
	\item $ 2\sin x\left( \sin x -\sqrt{3}\cos x\right) =1-2\sin x \Leftrightarrow \sqrt{3}\sin 2x +\cos 2x=2\sin x$\\
	$ \Leftrightarrow \sin\left( 2x+\dfrac{\pi}{6}\right) =\sin x\Leftrightarrow \hoac{&x=-\dfrac{\pi}{6}+k2\pi\\&x=\dfrac{5\pi}{18}+\dfrac{k2\pi}{3}} \qquad \left( k\in \mathbb{Z}\right) $
	\item $ 3\cos^2 x-\sin 2x +2\sin^2 x=\dfrac{3}{2} \Leftrightarrow 6\cos^2 x-2\sin 2x+4\sin^2 x=3\left( \sin^2 x+\cos^2 x\right) $\\
	$ \Leftrightarrow 3\cos^2 x-4\sin x\cos x+\sin^2 x=0 $\\
	Nhận xét: $ \cos x=0 $ không là nghiệm phương trình.\\
	Với $ \cos x\ne 0 $ chia hai vế phương trình cho $ \cos^2 x $ ta được\\
	$ \tan^2 x-4\tan x +3=0\Leftrightarrow \hoac{&\tan x=1\\&\tan x=3}\Leftrightarrow \hoac{&x=\dfrac{\pi}{4}+k\pi\\&x=\arctan 3+k\pi} \qquad\left(k\in \mathbb{Z} \right) $.
	\item  
	\begin{eqnarray*}
		&&\cos 2x+\sin 2x+3=\sqrt{2}\left( 3\cos x+\sin x\right)\\
		&\Leftrightarrow&\left( \cos 2x-3\sqrt{2}\cos x+3\right) +\left( 2\sin x\cos x-\sqrt{2}\sin x\right) =0\\
		&\Leftrightarrow& \left( \cos x-\sqrt{2}\right) \left( 2\cos x-\sqrt{2}\right) +\sin x\left( 2\cos x-\sqrt{2}\right)=0\\
		&\Leftrightarrow&\left( 2\cos x-\sqrt{2}\right) \left( \sin x+\cos x-\sqrt{2}\right) =0\\
		&\Leftrightarrow&\left( 2\cos x-\sqrt{2}\right)\left(\sqrt{2}\sin\left(x+\dfrac{\pi}{4} \right)-\sqrt{2}  \right)=0\\
		&\Leftrightarrow&\hoac{&\cos x=\dfrac{\sqrt{2}}{2}\\&\sin\left(x+\dfrac{\pi}{4} \right)=1} \Leftrightarrow \hoac{&x=\pm \dfrac{ \pi}{4}+k2\pi\\&x=\dfrac{\pi}{4}+k2\pi} \qquad\left(k\in \mathbb{Z} \right)		 .		
	\end{eqnarray*}	
\end{enumerate}
}
\end{bt}
\begin{bt}%[Thi GHK, Tây Thạnh - Hồ Chí Minh, 2018]%[Trương Quan Kía, EX-DT-10&11]%[1D1B1-3]
	Xét tính chẵn lẻ của hàm số $ y=f(x)=\sqrt{2+\sin x}+\sqrt{2-\sin x} $.	
	\loigiai{
	Tập xác định $ D=\mathbb{R} $.\\
	Với mọi $x\in D\Rightarrow -x\in D $.\\
	$ f(-x)=\sqrt{2+\sin (-x)}+\sqrt{2-\sin (-x)}=\sqrt{2-\sin x}+\sqrt{2+\sin x}=f(x) $.\\
	Vậy hàm số đã cho chẵn trên $ D $.		
	}
\end{bt}

 Tìm giá trị lớn nhất, giá trị nhỏ nhất của hàm số $ y=f(x)=\sin\left( x-\dfrac{\pi}{4}\right)+\sin\left(x+\dfrac{\pi}{4}\right)   $.	
	\loigiai{
	$ y=f(x)=\sin\left( x-\dfrac{\pi}{4}\right)+\cos\left(x-\dfrac{\pi}{4}\right)=\sqrt{2}\sin x    $\\
	Vậy GTLN của hàm số bằng $ \sqrt{2} $ khi $ \sin x=1\Leftrightarrow x=\dfrac{\pi}{2}+k2\pi $.\\
	GTNN của hàm số bằng $ -\sqrt{2} $ khi $ \sin x=-1\Leftrightarrow x=-\dfrac{\pi}{2}+k2\pi $.	
	}
