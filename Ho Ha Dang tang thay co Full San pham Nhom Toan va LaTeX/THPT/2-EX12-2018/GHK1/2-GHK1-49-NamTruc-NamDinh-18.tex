\choice
{$3$}
{$1$}
{$4$}
{\True $2$}
\loigiai{
Ta có: $y'=\dfrac{x^2-6x-m-8}{\left( {x-3} \right)^2}$, $y'=0 \Leftrightarrow x^2-6x-m-8=0$. Do hàm số đạt cực trị tại các điểm $x_1$, $x_2$ nên $x_1$, $x_2$ là nghiệm của phương trình $y'=0$. Theo hệ thức vi-et ta có $\heva{&x_1+x_2=6\\&x_1x_2=-m-8}$
\begin{align*}
\text{Ta có:} \left| {\dfrac{y\left( {x_1} \right)-y\left( {x_2} \right)}{x_1-x_2}} \right|&=\left| {\dfrac{\dfrac{x_1^2+3x_1+m-1}{x_1-3}-\dfrac{x_2^2+3x_2+m-1}{x_2-3}}{x_1-x_2}} \right|=\left| {\dfrac{x_1x_2-3\left( {x_1+x_2} \right)-m-8}{x_1x_2-3\left( {x_1+x_2} \right)+9}} \right|\\
&=\left| {\dfrac{-m-8-3 \cdot 6-m-8}{-m-8-3 \cdot 6+9}} \right|=\left| {\dfrac{-2m-34}{-m-17}} \right|=2
\end{align*}}

Bất phương trình $2\left( {\sqrt{x+1}-\sqrt{x^3+1}+\sqrt{x^2-x+1}} \right)<m+x^2-1$ có tập nghiệm là $\left( {-1;+\infty} \right)$ khi và chỉ khi

\loigiai{

\\$2\left( {\sqrt{x+1}-\sqrt{x+1}\sqrt{x^2-x+1}+\sqrt{x^2-x+1}} \right)<m+\left( {\sqrt{x+1}} \right)^2+\left( {\sqrt{x^2-x+1}} \right)^2-3$
\\$\Leftrightarrow 2\left( {\sqrt{x+1}+\sqrt{x^2-x+1}} \right)<\left( {\sqrt{x+1}+\sqrt{x^2-x+1}} \right)^2+m-3$ 
\\$\Leftrightarrow \left( {\sqrt{x+1}+\sqrt{x^2-x+1}} \right)^2-2\left( {\sqrt{x+1}+\sqrt{x^2-x+1}} \right)-3>-m$, $\left( {1} \right)$
\\Xét hàm số $f\left( {x} \right)=\sqrt{x+1}+\sqrt{x^2-x+1}$, với $x>-1$.
\\Ta có: $f'\left( {x} \right)=\dfrac{1}{2\sqrt{x+1}}+\dfrac{2x-1}{2\sqrt{x^2-x+1}}=\dfrac{\sqrt{x^2-x+1}+\left( {2x-1} \right)\sqrt{x+1}}{2\sqrt{x+1}\sqrt{x^2-x+1}}$.
\begin{align*}
\text{Trên khoảng } \left( {-1;+\infty} \right):\text{ }f'\left( {x} \right)=0 &\Leftrightarrow \heva{&\sqrt{x^2-x+1}=\left( {1-2x} \right)\sqrt{x+1}\\&x>-1}\\
&\Leftrightarrow \heva{&x>-1\\&1-2x>0\\&x^2-x+1=\left( {4x^2-4x+1} \right)\left( {x+1} \right)}\\
&\Leftrightarrow \heva{&x \le \dfrac{1}{2}\\&4x^3-x^2-2x=0}\\
&\Leftrightarrow \hoac{&x=0\\&x=\dfrac{1-\sqrt{33}}{8}}
\end{align*}
