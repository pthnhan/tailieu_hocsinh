	$A=\dfrac{3+2\sqrt{3}}{\sqrt{3}}-\dfrac{2}{\sqrt{3}-1}=\dfrac{\sqrt{3}(\sqrt{3}+2)}{\sqrt{3}}-\dfrac{2(\sqrt{3}+1)}{3-1}=(\sqrt{3}+2)-(\sqrt{3}+1)=1.$

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Đống Đa, Hà Nội]%[Lê Ngọc Hiếu, dự án 9DT19-HK1]%[9D1K4]
	Giải phương trình $\sqrt{4x-8}-\dfrac{1}{5}\sqrt{25x-50}=3\sqrt{x-2}-1.$
	\loigiai{Điều kiện $x \ge 2$.
		\begin{align*}
	&\sqrt{4x-8}-\dfrac{1}{5}\sqrt{25x-50}=3\sqrt{x-2}-1\\
\Leftrightarrow & 2\sqrt{x-2}-\dfrac{1}{5}\cdot 5 \sqrt{x-2}=3\sqrt{x-2}-1\\
\Leftrightarrow & -2\sqrt{x-2}=-1
\Leftrightarrow \sqrt{x-2}=\dfrac{1}{2}
\Leftrightarrow  x-2=\left (\dfrac{1}{2}\right )^2
\Leftrightarrow  x=\dfrac{9}{4}.		
		\end{align*}
	Vậy phương trình có nghiệm $x=\dfrac{9}{4}.$	
	}
\end{bt}

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Đống Đa, Hà Nội]%[Lê Ngọc Hiếu, dự án 9DT19-HK1]%[9D1K8]
	Cho hai biểu thức $A=\dfrac{2\sqrt{x}-1}{\sqrt{x}-1}$ và $B=\dfrac{\sqrt{x}}{\sqrt{x}-1}+\dfrac{3}{\sqrt{x}+1}-\dfrac{6\sqrt{x}-4}{x-1}$, với $x \ge 0,\; x \ne 1$.
	\begin{enumerate}
		\item Tính giá trị biểu thức $A$ khi $x=25$.
		\item Rút gọn biểu thức $B$.
		\item Đặt $P=A\cdot B$. Tìm giá trị nguyên của $x$ để $P<1$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item $A=\dfrac{2\sqrt{x}-1}{\sqrt{x}-1}=\dfrac{\left(2\sqrt{x}-1\right)\left(\sqrt{x}+1\right)}{\left(\sqrt{x}-1\right)\left(\sqrt{x}+1\right)}=\dfrac{2x+\sqrt{x}-1}{x-1}.$
		
		Thay $x=25$, ta có $A=\dfrac{2\cdot 25+\sqrt{25}-1}{25-1}=\dfrac{9}{4}.$
		\item
	$B=\dfrac{\sqrt{x}}{\sqrt{x}-1}+\dfrac{3}{\sqrt{x}+1}-\dfrac{6\sqrt{x}-4}{x-1}
	=\dfrac{\sqrt{x}(\sqrt{x}+1)+3(\sqrt{x}-1) -(6\sqrt{x}-4)}{x-1}
	=\dfrac{x-2\sqrt{x}+1}{x-1}.$
	\item $P=A \cdot B=\dfrac{2x+\sqrt{x}-1}{x-1} \cdot \dfrac{x-2\sqrt{x}+1}{x-1}=\dfrac{\left(2\sqrt{x}-1\right)\left(\sqrt{x}+1\right)(\sqrt{x}-1)^2}{(x-1)^2}.$
	
	Vì $\sqrt{x}+1>0;\; (\sqrt{x}-1)^2 \ge 0 ; \; (x-1)^2\ge 0$ nên 
	\begin{align*}
	&P<1 \Leftrightarrow P-1<0 \Leftrightarrow \dfrac{\left(2\sqrt{x}-1\right)\left(\sqrt{x}+1\right)(\sqrt{x}-1)^2}{(x-1)^2} -1 <0\\
	\Leftrightarrow & \dfrac{\left(2\sqrt{x}-1\right)\left(\sqrt{x}+1\right)(\sqrt{x}-1)^2 - (x-1)^2}{(x-1)^2}<0\\
	\Leftrightarrow &\dfrac{(2\sqrt{x}-1)\left(\sqrt{x}+1\right)(\sqrt{x}-1)^2-(\sqrt{x}-1)^2(\sqrt{x}+1)^2}{(x-1)^2}<0\\
	\Leftrightarrow & \dfrac{(\sqrt{x}-1)^2 (\sqrt{x}+1)(2\sqrt{x}-1 - (\sqrt{x}+1))}{(x-1)^2}<0\\
	\Leftrightarrow & 2\sqrt{x}-1-\sqrt{x}-1<0 \Leftrightarrow \sqrt{x}-2<0\Leftrightarrow\sqrt{x}<2 \Leftrightarrow x<4.
	\end{align*}
	Từ điều kiện của $x$ nên ta có $x=\{0;2;3\}.$
	\end{enumerate}	
	}
\end{bt}

	Cho đường tròn $(O;R)$ và một điểm $A$ sao cho $OA=2R$, vẽ các tiếp tuyến $AB$, $AC$ với $(O;R)$, $B$ và $C$ là các tiếp điểm. Vẽ đường kính $BOD$.
	\begin{enumerate}
		\item Chứng minh $4$ điểm $A$, $B$, $O$, $C$ cùng thuộc một đường tròn.
		\item Chứng mình rằng $DC \parallel OA$.
		\item Đường trung trực của $BD$ cắt $AC$ và $CD$ lần lượt tại $S$ và $E$. Chứng minh rằng $OCEA$ là hình thang cân.
		\item Gọi $I$ là giao điểm của đoạn $OA$ và $(O)$, $K$ là giao điểm của tia $SI$ và $AB$. Tính theo $R$ diện tích tứ giác $AKOS$. 
	\end{enumerate}
	\loigiai{

\begin{enumerate}
	\item Chứng minh $4$ điểm $A$, $B$, $O$, $C$ cùng thuộc một đường tròn.
	
	$AB$, $AC$ là tiếp của đường tròn nên $\widehat{ABO}=\widehat{ACO}=90^\circ$
	
	$\Rightarrow$ $B$, $C$ cùng thuộc đường tròn đường kính $AO.$
	
	Vậy $A$, $B$, $O$, $C$ cùng thuộc một đường tròn.
	\item Chứng mình rằng $DC \parallel OA$.
	
	$AB$, $AC$ là tiếp của đường tròn nên $AO \perp BC.$
	
	$\triangle BDC$ nội tiếp đường tròn đường kính $BD$ nên $\widehat{DCB}=90^\circ$ hay $DC \perp BC.$
	
	Vậy $OA \parallel DC$ (cùng vuông góc với $BC$).
	\item Đường trung trực của $BD$ cắt $AC$ và $CD$ lần lượt tại $S$ và $E$. Chứng minh rằng $OCEA$ là hình thang cân.
	
	Xét $\triangle ODE$ và $\triangle BOA$ có
	
	$\heva{&\widehat{EOD}=\widehat{ABO}\\
	& DO= OB(=R)\\ & \widehat{EDO}=\widehat{AOB}}\Rightarrow \triangle ODE =\triangle BOA \text{ (g.c.g)}.$

	$\Rightarrow OE = BA.$
	
	Xét tứ giác $OCEA$ có 
	
	$\heva{& CE \parallel OA \\ & AC = OE \; (=AB).}$
	
	$\Rightarrow$ $OCEA$ là hình thang cân.
	\item Gọi $I$ là giao điểm của đoạn $OA$ và $(O)$, $K$ là giao điểm của tia $SI$ và $AB$. Tính theo $R$ diện tích tứ giác $AKOS$. 
	
	Xét hình thang cân $OCEA$ có hai đường chéo $OE$ và $CA$ cắt nhau tại $S$ nên $SO=SA$
	
	$\Rightarrow \triangle SOA$ cân tại $S$
	
	$\Rightarrow SI \perp OA$ ($I$ là trung điểm $OA$)
	
	$\Rightarrow SK$ là tiếp tuyến $(O).$
	
	Suy ra $\widehat{SOI}=\widehat{SOC}$ hay $\widehat{SOI}=\dfrac{\widehat{COI}}{2}.$
	
	Tương tự $\widehat{KOI}=\dfrac{\widehat{BOI}}{2}.$
	
	Suy ra tam giác $OKS$ cân tại $O$ ($OI$ vừa là phân giác, vừa là đường cao).
	
	$\Rightarrow$ $I$ là trung điểm của đoạn thẳng $KS$
	
	$\Rightarrow$ $KOSA$ là hình thoi. 
		
	Tam giác $OBA$ vuông tại $B$ nên $\sin BAO = \dfrac{OB}{OA}=\dfrac{1}{2} \Rightarrow \widehat{BAO}=30^\circ.$
	
	Tam giác $KIA$ vuông tại $I$ nên $\tan KAI = \dfrac{KI}{IA} \Rightarrow KI = IA\cdot \tan KAI= \dfrac{R\sqrt{3}}{3}$
	
	$\Rightarrow KS=2KI=\dfrac{2R\sqrt{3}}{3}.$
	
	Diện tích hình thoi $AKOS$ bằng $\dfrac{OA \cdot KS}{2}=\dfrac{2R \cdot \dfrac{2R\sqrt{3}}{3}}{2}=\dfrac{2R^2\sqrt{3}}{3}\text{đvdt}.$ 
\end{enumerate}	
	}

	Giải phương trình $4\sqrt{x+1}=x^2-5x+14$
	\loigiai{Điều kiện $x \ge -1$.
		\begin{align*}
				&4\sqrt{x+1}=x^2-5x+14\\
\Leftrightarrow &x^2-5x+14-4\sqrt{x+1}=0\\
\Leftrightarrow & (4-2\cdot 2  \sqrt{x+1}+x+1)+(x^2-6x+9)=0\\
\Leftrightarrow&(2-\sqrt{x+1})^2+(x-3)^2=0\\
\Leftrightarrow & \heva{2-\sqrt{x+1}&=0\\ x-3&=0} \Leftrightarrow x=3 \; \text{(thỏa điều kiện).}
		\end{align*}
Vậy phương trình có nghiệm $x=3$.		
	}
