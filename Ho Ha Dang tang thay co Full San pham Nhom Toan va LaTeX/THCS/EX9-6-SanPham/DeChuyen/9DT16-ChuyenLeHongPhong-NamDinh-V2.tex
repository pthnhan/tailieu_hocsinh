\begin{enumerate}
\item[a)] Đơn giản biểu thức $\sqrt{x+2+2\sqrt{x+1}}-\sqrt{x+2-2\sqrt{x+1}}$ với $x>0$.
\item[b)]  Cho $a,b,c$ là các số thực thỏa mãn điều kiện $a+b+c=6$ và $\dfrac{1}{a+b}+\dfrac{1}{b+c}+\dfrac{1}{c+a}=\dfrac{47}{60}$. Tính giá trị biểu thức: $\dfrac{a}{b+c}+\dfrac{b}{c+a}+\dfrac{c}{a+b}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a)] 
$\sqrt{x+2+2\sqrt{x+1}}-\sqrt{x+2-2\sqrt{x+1}}=\sqrt{(\sqrt{x+1}+1)^2} - \sqrt{(\sqrt{x+1}-1)^2}=\sqrt{x+1}+1 - \sqrt{x+1}+1=2$
\item[b)]   
$\dfrac{a}{b+c}+\dfrac{b}{c+a}+\dfrac{c}{a+b} +3 = (a+b+c)(\dfrac{1}{a+b} +\dfrac{1}{b+c} +\dfrac{1}{c+a}) =\dfrac{47}{10}$ $\Longrightarrow \dfrac{a}{b+c}+\dfrac{b}{c+a}+\dfrac{c}{a+b}= \dfrac{17}{10}.$
\end{enumerate}
}
\end{ex}

\begin{ex}%[Chuyên Lê Hồng Phong Nam Định vòng 2, 2016]%[9D5K2]
\hfill
\begin{enumerate}
\item[a)] Chứng minh rằng tồn tại vô hạn bộ ba số nguyên $(x,y,z)$ thỏa mãn $xyz\neq 0$ và $x^5+8y^3+7z^2=0$.
\item[b)]  Tìm tất cả các số nguyên không âm $a,b,c$ thỏa mãn $(a-b)^2+(b-c)^2+(c-a)^2=6abc$ và $a^3+b^3+c^3+1$ chia hết $a+b+c+1$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a)] Xét bộ ba các số nguyên có dạng $(t^6,-t^{10},t^{15})$ với $t \in \mathbb{Z}$, ta có $(t^6)^5+8(-t^{10})^3+7(t^{15})^2=0$. Vậy phương trình  tồn tại vô hạn bộ ba số nguyên $(x,y,z)$.
\item[b)] Ta có $(a-b)^2+(b-c)^2+(c-a)^2=6abc \Longrightarrow (a+b+c)^2-3(ab+bc+ca)=3abc $. \\
$a^3+b^3+c^3+1 = (a+b+c)^3+1 - 3(a+b+c)(ab+bc+ca)+3abc 
\\= \underbrace{(a+b+c)^3+1}_{\textrm{chia hết cho $a+b+c+1$}}-3(a+b+c+1)(ab+bc+ca) + (a+b+c)^2$, cho nên $(a+b+c)^2$ chia hết cho $a+b+c+1$ $\Longrightarrow a+b+c+1 =1$ hoặc $a+b+c+1=-1$,  mặt khác $a,b,c$ không âm cho nên 
$a+b+c=0 \Longrightarrow a=b=c=0$.

\end{enumerate}
}
\end{ex}
\begin{ex}%[Chuyên Lê Hồng Phong Nam Định vòng 2, 2016]%[9D5G2]%[9D5G1]

\begin{enumerate}
\item[a)] Cho $x,y,z$ là các số thực thỏa mãn $(x-y)(x-z)=1$ và $y\neq z$. Chứng minh:
$\dfrac{1}{(x-y)^2}+\dfrac{1}{(y-z)^2}+\dfrac{1}{(z-x)^2}\geq 4$.
\item[b)] Trên bảng ban đầu ghi số $2$ và số $4$. Ta thực hiện cách viết thêm các số lên bảng như sau: nếu trên bảng đã có hai số, giả sử là $a, b$, $a\neq b$, ta viết thêm lên bảng số có giá trị là $a+b+ab$. Hỏi với cách thực hiện như vậy có thể xuất hiện số $2016$ được không? Giải thích.
\end{enumerate}
\end{ex}
\loigiai{
\begin{enumerate}
\item[a)] Đặt $a=x-y$, $b=x-z$, ta có $ab=1$, BĐT tương đương với $\dfrac{1}{a^2}+\dfrac{1}{a^2+b^2-2} +\dfrac{1}{b^2} \geq 4 \iff a^2+b^2 +\dfrac{1}{a^2+b^2-2} \geq 4 \iff a^2+b^2-2 +\dfrac{1}{a^2+b^2-2} \geq 2$. (ĐPCM) 
\item[b)] Dễ thấy tất cả các số ghi trên bảng là các số nguyên dương.
Giả sử xuất hiện $2016$, ta có $a+b+ab=2016 \iff (a+1)(b+1) =2017$, do $2017$ nguyên tố và $a,b >0$ suy ra $a+1=1$ hoặc $b+1=1$ (vô lý). 
\end{enumerate}
