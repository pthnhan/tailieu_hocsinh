\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Vĩnh Bảo, Hải Phòng]%[9D1K1]
    \begin{enumerate}[1.] 
    \item Tìm $x$ để biểu thức sau có nghĩa.
        \begin{enumerate}
            \item $\sqrt{2x-5}$ 
            \item $\dfrac{1}{x-1}+\sqrt{-2x+3}$
        \end{enumerate}
    \item  Rút gọn các biểu thức sau:
        \begin{enumerate}
            \item $A=\sqrt{75}+\sqrt{48}-\dfrac{1}{2}\sqrt{300}$
            \item $B=\left(\dfrac{\sqrt{x}}{\sqrt{x}+3}+\dfrac{\sqrt{x}}{\sqrt{x}-3}\right):\dfrac{2\sqrt{x}}{x-9}$ (với $x \geq 0$ và $x \neq 9$)
        \end{enumerate}
    \end{enumerate}    
\loigiai{
    \begin{enumerate}[1.]
        \item 
            \begin{enumerate}
            \item Biểu thức $A=\sqrt{2x-5}$ có nghĩa khi: $2x-5\geq 0 \Leftrightarrow x \geq \dfrac{5}{2}$\\
            Vậy $x \geq \dfrac{5}{2}$ thì biểu thức $A$ có nghĩa.
            \item Biểu thức $B=\dfrac{1}{x-1}+\sqrt{-2x+3}$ có nghĩa khi:\\
            $\begin{cases}-2x+3 \geq 0 \\ x-1 \neq 0\end{cases}
            \Leftrightarrow \begin{cases}
            x \leq \dfrac{3}{2} \\ x \neq 1
            \end{cases}$\\
            Vậy $\begin{cases}
            x \leq \dfrac{3}{2}\\ x \neq 1
            \end{cases}$ thì biểu thức $B$ có nghĩa.            \end{enumerate}
        \item 
            \begin{enumerate}
            \item $A=\sqrt{75}+\sqrt{48}-\dfrac{1}{2}\sqrt{300}=5\sqrt{3}+4\sqrt{3}-5\sqrt{3}=4\sqrt{3}$
            \item $B=\left(\dfrac{\sqrt{x}}{\sqrt{x}+3}+\dfrac{\sqrt{x}}{\sqrt{x}-3}\right):\dfrac{2\sqrt{x}}{x-9}=\dfrac{\sqrt{x}\cdot \left(\sqrt{x}-3\right)+\sqrt{x}\cdot \left(\sqrt{x}+3\right)}{\left(\sqrt{x}+3\right) \cdot \left(\sqrt{x}-3 \right)}:\dfrac{2\sqrt{x}}{x-9}$ \\
            $B=\dfrac{2x}{x-9}:\dfrac{2\sqrt{x}}{x-9}=\dfrac{2x}{x-9}\cdot \dfrac{x-9}{2\sqrt{x}}=\sqrt{x}$
            \end{enumerate}
        \end{enumerate}        
}
\end{ex}

    \end{enumerate}

        Vẽ đồ thị đường thẳng $(d)$

        Mà $(d')$ đi qua $B(-2;2)\Rightarrow x=-2$ và $y=2$ thỏa mãn phương trình $(d')\Rightarrow b=-6$ (t/m). Vậy phương trình $(d')$ cần tìm là: $y=-4x-6$.

\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Vĩnh Bảo, Hải Phòng]%[9D4K7]%[9D3B4]%[9H1B2]
    \begin{enumerate}
        \item Giải phương trình: $\left(\sqrt{x}-1\right)^2-x+2=0$
        \item Cho phương trình đường thẳng $2x-y=3$ $(d) $và phương trình đường thẳng $x+y=6$ $(d')$. Giải hệ phương trình gồm đường thẳng $(d)$ và $(d')$?
        \item Bóng của một cây trên mặt đất là $12$m, tia nắng mặt trời chiếu xiên một góc $30^\circ$ so với mặt đất. Tính chiều cao của cây?
    \end{enumerate}
\loigiai{
    \begin{enumerate}
        \item $\left( \sqrt{x}-1\right)^2-x+2=0$\qquad ĐK: $x \geq 0$\\
        $\Leftrightarrow x-2\sqrt{x}+1-x+2=0$\\
        $2\sqrt{x}=3$\\
        $\sqrt{x}=\dfrac{3}{2}$ vì $x \geq 0$\\
        $\Leftrightarrow x=\dfrac{9}{4}$\\
        Vậy  $\Leftrightarrow x=\dfrac{9}{4}$ là nghiệm của phương trình.
        \item Từ hệ ta có $(d):y=2x-3$ và $(d'):y=-x+6$\\
        Xét phương trình hoành độ giao điểm: $2x-3=-x+6$
        $\Leftrightarrow x=3$\\
        Thay  $\Leftrightarrow x=3$ vào $(d') \Rightarrow y=3 \Rightarrow D(3;3).$\\
        Vậy giao điểm của $(d)$ và $(d')$ là $D(3;3)$; hệ phương trình có nghiệm $(x;y)=(3;3)$.
        \item Gọi cây có chiều cao $AB$ ($AB$ không âm) và có bóng trên mặt đất là $AC$.\\
        Do cây trồng vuông góc với mặt đất nên tam giác $ABC$ vuông tại $A$.\\
        $\Rightarrow AB=AC \cdot \tan C=12 \cdot \tan 30^\circ=12 \cdot \dfrac{1}{\sqrt{3}}=4\sqrt{3}$ (t/m)\\
        Vậy cây đó cao $4\sqrt{3}$m.
    \end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Vĩnh Bảo, Hải Phòng]%[]
    \begin{enumerate}
        \item Cho $a,b>0$; chứng minh rằng: $3(b^2+2a^2) \geq (b+2a)^2$
        \item Cho $a,b,c>0$ thỏa mãn $ab+bc+ca=abc$. Chứng minh rằng:
        $$\dfrac{\sqrt{b^2+2a^2}}{ab}+\dfrac{\sqrt{c^2+2b^2}}{bc}+\sqrt{\dfrac{a^2+2c^2}{ca}}\geq \sqrt{3}$$
    \end{enumerate}
\loigiai{
    \begin{enumerate}
        \item $3(b^2+2a^2) \geq (b+2a)^2$\\
        $\Leftrightarrow 3b^2+6a^2 \geq b^2+4ab+4a^2$\\
        $2(a-b)^2 \geq 0 ~\forall a;~b$\\
        Dấu "=" xảy ra khi và chỉ khi $a=b$
        \item Theo câu a ta có:
        \begin{align*}
            3(b^2+2a^2) \geq (b+2a)^2 &\Rightarrow \sqrt{b^2+2a^2}\geq \dfrac{b+2a}{\sqrt{3}}\\
            &\Rightarrow \dfrac{\sqrt{b^2+2a^2}}{ab} \geq \dfrac{bc+2ac}{\sqrt{3}abc} \qquad (1)
        \end{align*}
             \text{Chứng minh tương tự:}
        \begin{align*}
            \dfrac{\sqrt{c^2+2b^2}}{bc} \geq \dfrac{ca+2ab}{\sqrt{3}abc}\qquad (2)\\
            \dfrac{\sqrt{a^2+2c^2}}{ca} \geq \dfrac{ab+2bc}{\sqrt{3}abc}\qquad (3)\\
            \text{Cộng (1), (2) và (3) vế với vế ta được:}\\
            \dfrac{\sqrt{b^2+2a^2}}{ab}+\dfrac{\sqrt{c^2+2b^2}}{bc}+\sqrt{\dfrac{a^2+2c^2}{ca}}\geq \sqrt{3}
        \end{align*}     
\end{enumerate}
}
\end{ex}
