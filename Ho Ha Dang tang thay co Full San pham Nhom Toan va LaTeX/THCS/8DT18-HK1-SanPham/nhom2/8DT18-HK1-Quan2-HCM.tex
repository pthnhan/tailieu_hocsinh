	\begin{enumEX}{2}

		\item $x^2 + 2x - y^2 + 2y.$
		\item $25x^2 - 4y^2 + 9 + 30x.$
		\item $x^2 - x - 56.$
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{2}
			\item  $\begin{aligned}[t]
			&&	x^3y + 2x^2y^2 + xy^3\\
			&= & xy(x^2 + 2xy + y^2)\\
			&= &xy(x+y)^2.
			\end{aligned}$
			\item $ \begin{aligned}[t]
			&&x^2 + 2x - y^2 + 2y \\
			&=& (x^2 - y^2) +(2x + 2y)\\
			&=&(x-y)(x+y) + 2(x+y) \\
			&=&(x + y)(x - y + 2).
			\end{aligned}$
			\item $\begin{aligned}[t]
			&&25x^2 - 4y^2 + 9 + 30x \\
			&=&25x^2 + 30x + 9 - 4y^2\\
			&=&(5x+3)^2 -(2y)^2\\
			&= &(5x +3 + 2y)( 5x +3 - 2y).
			\end{aligned}$
			\item  $\begin{aligned}[t]
			&&x^2 - x - 56 \\
			&=& (x^2 - 8x) + (7x - 56) \\
			&=&x(x-8) + 7(x-8) \\
			&=&(x-8)(x+7).
			\end{aligned}$
			
			\end{enumEX}
}

=&\dfrac{x^2-x^2+1+x-2}{x(x-1)}\\
