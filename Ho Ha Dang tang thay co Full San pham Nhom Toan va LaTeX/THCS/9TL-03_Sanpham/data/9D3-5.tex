\section{ĐỀ KIỂM TRA 1 TIẾT}
\subsection{Đề số 1 (Dành cho học sinh đại trà)}
\setcounter{bt}{0}
\begin{bt}%[9D3Y2]%[9D3G2]
Giải các hệ phương trình sau
\begin{listEX}[2]
\item $\heva{&2x-y=-5\\&5x+y=-9}$;
\item $\heva{&3\sqrt{5}x-4y=15-2\sqrt{7}\\&-2\sqrt{5}x+8\sqrt{7}y=18}$.
\end{listEX}
\loigiai{
\begin{listEX}[]
\item Ta có
\begin{eqnarray*}
\heva{&2x-y=-5\\&5x+y=-9}\Leftrightarrow \heva{2x-y&=-5\\7x&=-14}\Leftrightarrow \heva{&x=-2\\&y=1.}
\end{eqnarray*}
Vậy hệ phương trình đã cho có nghiệm $(x;y)=(-2;1).$
\item Ta có
\begin{eqnarray*}
&&\heva{&3\sqrt{5}x-4y=15-2\sqrt{7}\\&-2\sqrt{5}x+8\sqrt{7}y=18}\Leftrightarrow \heva{&6\sqrt{5}x-8y=30-4\sqrt{7}\\&-6\sqrt{5}x+24\sqrt{7}y=54}\\ &\Leftrightarrow &\heva{&6\sqrt{5}x-8y=30-4\sqrt{7}\\&\left(24\sqrt{7}-8\right)y=84-4\sqrt{7}}\Leftrightarrow \heva{&6\sqrt{5}x-8y=30-4\sqrt{7}\\&\left(3\sqrt{7}-1\right)8y=4\left(21-\sqrt{7}\right)}\\ &\Leftrightarrow &\heva{&6\sqrt{5}x-8y=30-4\sqrt{7}\\&y=\dfrac{21-\sqrt{7}}{6\sqrt{7}-2}=\dfrac{\left(21-\sqrt{7}\right)\left(6\sqrt{7}+2\right)}{248}=\dfrac{\sqrt{7}}{2}}\Leftrightarrow \heva{&x=\sqrt{5}\\&y=\dfrac{\sqrt{7}}{2}.}
\end{eqnarray*}
Vậy hệ phương trình đã cho có nghiệm $(x;y)=\left(\sqrt{5};\dfrac{\sqrt{7}}{2}\right).$
\end{listEX}
}
\end{bt}
\begin{bt}%[9D3B2]
Tìm $m$ để hệ phương trình sau vô nghiệm $$\heva{&mx+2y=-2\\&3x-y=5}.$$
\loigiai{
Hệ phương trình $\heva{&mx+2y=-2\\&3x-y=5}$ vô nghiệm $\Leftrightarrow \heva{&\dfrac{m}{3}=\dfrac{2}{-1}\\&\dfrac{m}{3}\ne \dfrac{-2}{5}}\Leftrightarrow m=-6.$\\
Vậy $m=-6$ là giá trị cần tìm.
}
\end{bt}
\begin{bt}%[9D3B3]
Một trường chuyên tuyển $70$ học sinh vào hai lớp 10 Toán và lớp 10 Tin. Biết rằng nếu chuyển $5$ học sinh của lớp 10 Toán sang lớp 10 Tin thì số học sinh của hai lớp bằng nhau. Tính số học sinh ban đầu của mỗi lớp.
\loigiai{
Gọi $x$ (học sinh) là số học sinh ban đầu của lớp 10 Toán $(x\in \mathbb{N}^*,x<70).$\\
Gọi $y$ (học sinh) là số học sinh ban đầu của lớp 10 Tin $(y\in \mathbb{N}^*,y<70).$\\
Tổng số học sinh của 2 lớp là 70 nên ta có $x+y=70\quad (1).$\\
Số học sinh lớp 10 Toán lúc sau là $x-5$ (học sinh).\\
Số học sinh lớp 10 Tin lúc sau là $y+5$ (học sinh).\\
Theo đề bài ta có $x-5=y+5$ hay $x-y=10\quad\quad\quad (2).$\\
Từ (1) và (2) ta có hệ
\begin{eqnarray*}
\heva{&x+y=70\\&x-y=10} \Leftrightarrow \heva{&x+y=70\\&2x=80}\Leftrightarrow \heva{&x=40\text{ (thỏa điều kiện)}\\&y=30\text{ (thỏa điều kiện)}.}
\end{eqnarray*}
Vậy số học sinh ban đầu của lớp 10 Toán là $40$ học sinh, số học sinh ban đầu của lớp 10 Tin là $30$ học sinh.
}
\end{bt}
\begin{bt}%[9D3K1]
Tìm $a,b\in \mathbb{R}$ để đường thẳng $y=ax+b$ $(d)$ đi qua hai điểm $A(2;2),B(-1;5).$
\loigiai{
Ta có
\begin{eqnarray*}
\heva{&A\in (d)\\&B\in (d)}\Leftrightarrow \heva{2a+b&=2\\-a+b&=5}\Leftrightarrow \heva{3a&=-3\\-a+b&=5}\Leftrightarrow \heva{&a=-1\\&b=4.}
\end{eqnarray*}
Vậy $a=-1,b=4$ là giá trị cần tìm.
} 
\end{bt}
\subsection{Đề số 2 (Dành cho học sinh giỏi)}
\setcounter{bt}{0}
\begin{bt}%[9D3K2]%[9D3G2]
Giải các hệ phương trình sau
\begin{listEX}[2]
\item $\heva{&\dfrac{3}{2x-y}-\dfrac{6}{x+y}=-1\\&\dfrac{1}{2x-y}-\dfrac{1}{x+y}=0};$
\item $\heva{\dfrac{xy}{4x+3y}&=\dfrac{4}{11}\\ \dfrac{xy}{2x+y}&=\dfrac{4}{5}}.$
\end{listEX}
\loigiai{
\begin{listEX}[]
\item Đặt $a=\dfrac{1}{2x-y},b=\dfrac{1}{x+y}$ với $(2x\ne y,x\ne -y)$, hệ phương trình đã cho trở thành
\begin{eqnarray*}
\heva{3a-6b&=-1\\a-b&=0}\Leftrightarrow \heva{3b-6b&=-1\\a&=b}\Leftrightarrow \heva{&a=\dfrac{1}{3}\\&b=\dfrac{1}{3}.}
\end{eqnarray*}
Khi đó ta có
\begin{eqnarray*}
\heva{\dfrac{1}{2x-y}&=\dfrac{1}{3}\\ \dfrac{1}{x+y}&=\dfrac{1}{3}}\Leftrightarrow \heva{2x-y&=3\\x+y&=3}\Leftrightarrow \heva{3x&=6\\x+y&=3}\Leftrightarrow \heva{&x=2\text{ (thỏa điều kiện) }\\&y=1\text{ (thỏa điều kiện)}.}
\end{eqnarray*}
Vậy hệ phương trình đã cho có nghiệm $(x;y)=(2;1).$
\item Nhận thấy $(x;y)=(0;0)$ không là nghiệm của hệ phương trình.\\
Do đó $\heva{\dfrac{xy}{4x+3y}&=\dfrac{4}{11}\\ \dfrac{xy}{2x+y}&=\dfrac{4}{5}}\Leftrightarrow \heva{\dfrac{4x+3y}{xy}&=\dfrac{11}{4}\\ \dfrac{2x+y}{xy}&=\dfrac{5}{4}}\Leftrightarrow \heva{&\dfrac{4}{y}+\dfrac{3}{x}=\dfrac{11}{4}\\&\dfrac{2}{y}+\dfrac{1}{x}=\dfrac{5}{4}.}$\\
Đặt $a=\dfrac{1}{x},b=\dfrac{1}{y},$ với $(x\ne 0,y\ne 0)$ hệ phương trình đã cho trở thành
\begin{eqnarray*}
\heva{3a+4b&=\dfrac{11}{4}\\a+2b&=\dfrac{5}{4}}\Leftrightarrow \heva{3a+4b&=\dfrac{11}{4}\\2a+4b&=\dfrac{10}{4}}\Leftrightarrow \heva{3a+4b&=\dfrac{11}{4}\\a&=\dfrac{1}{4}}\Leftrightarrow \heva{&a=\dfrac{1}{4}\\&b=\dfrac{1}{2}.}
\end{eqnarray*}
Khi đó ta có
\begin{eqnarray*}
\heva{&\dfrac{1}{x}=\dfrac{1}{4}\\&\dfrac{1}{y}=\dfrac{1}{2}}\Leftrightarrow \heva{&x=4\text{ (thỏa mãn điều kiện) }\\&y=2\text{ (thỏa mãn điều kiện)}.}
\end{eqnarray*}
Vậy hệ phương trình đã cho có nghiệm $(x;y)=(4;2).$
\end{listEX}
}
\end{bt}
\begin{bt}%[9D3B2]
Giải và biện luận hệ phương trình theo tham số $m$ $$\heva{-2mx+y&=5\\mx+3y&=1}.$$
\loigiai{
Ta có 
$\heva{-2mx+y&=5\\mx+3y&=1}\Leftrightarrow \heva{-6mx+3y&=15\\mx+3y&=1}\Leftrightarrow \heva{-7mx&=14\\mx+3y&=1}\Leftrightarrow \heva{mx&=-2\\y&=\dfrac{1-mx}{3}}(*)$\\
+ Với $m=0$, $(*)\Leftrightarrow \heva{0x&=-2\\y&=\dfrac{1}{3}}$ (vô nghiệm).\\
+ Với $m\ne 0,$ $(*)\Leftrightarrow \heva{&x=\dfrac{-2}{m}\\&y=1.}$\\
Vậy với $m=0$, hệ phương trình đã cho vô nghiệm.\\
Với $m\ne 0$, hệ phương trình đã cho có nghiệm duy nhất $(x;y)=\left(\dfrac{-2}{m};1\right).$
}
\end{bt}
\begin{bt}%9D3K3]
Hai vòi nước cùng chảy vào một bể cạn thì sau $12$ giờ sẽ đầy bể. Nếu cho vòi thứ nhất chảy trong $4$ giờ rồi khóa vòi thứ nhất lại, cho vòi thứ hai chảy trong $6$ giờ nữa thì được $\dfrac{2}{5}$ bể. Hỏi nếu chảy riêng thì mỗi vòi chảy trong bao lâu sẽ đầy bể?
\loigiai{
Gọi $x$ (giờ) là thời gian vòi thứ nhất chảy riêng thì đầy bể $(x\in \mathbb{R},x>12).$\\
Khi đó số phần bể vòi thứ nhất chảy vào trong một giờ là $\dfrac{1}{x}$ (bể).\\
Gọi $y$ (giờ) là thời gian vòi thứ hai chảy riêng thì đầy bể $(y\in \mathbb{R},y>12).$\\
Khi đó số phần bể vòi thứ hai chảy vào trong một giờ là $\dfrac{1}{y}$ (bể).\\
Số phần bể cả 2 vòi cùng chảy vào là $\dfrac{1}{12}$ (bể) nên ta có $\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{1}{12}.$\tagEX{1}
Theo đề bài ta có $\dfrac{4}{x}+\dfrac{6}{y}=\dfrac{2}{5}.$ \tagEX{2}
Từ (1), (2) ta có hệ
\begin{eqnarray*}
\heva{&\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{1}{12}\\&\dfrac{4}{x}+\dfrac{6}{y}=\dfrac{2}{5}}&\Leftrightarrow &\heva{&\dfrac{4}{x}+\dfrac{4}{y}=\dfrac{1}{3}\\&\dfrac{4}{x}+\dfrac{6}{y}=\dfrac{2}{5}}\Leftrightarrow \heva{&\dfrac{4}{x}+\dfrac{4}{y}=\dfrac{1}{3}\\&\dfrac{2}{y}=\dfrac{1}{15}}\\ &\Leftrightarrow &\heva{&\dfrac{1}{x}=\dfrac{1}{20}\\&\dfrac{1}{y}=\dfrac{1}{30}}\Leftrightarrow \heva{&x=20\text{ (thỏa điều kiện) }\\&y=30\text{ (thỏa điều kiện).}} 
\end{eqnarray*}
Vậy vòi thứ nhất chảy riêng một mình thì sau $20$ giờ thì đầy bể. Vòi thử hai chảy riêng một mình thì sau $30$ giờ thì đầy bể.
}
\end{bt}
\begin{bt}%[9D3G1]
Tìm nghiệm nguyên của phương trình $2x+3y=21$.
\loigiai{
Ta có $3y\ \vdots\ 3,21\ \vdots\ 3$ nên $2x\ \vdots\ 3\Rightarrow x\ \vdots\ 3.$\\
Đặt $x=3a\ (a\in \mathbb{Z})$, ta có\\
$6a+3y=21\Leftrightarrow 2a+y=7\Leftrightarrow y=-2a+7.$\\
Nghiệm nguyên của phương trình là $(x=a,y=-2a+7)$ với $a\in \mathbb{Z}.$
}
\end{bt}
