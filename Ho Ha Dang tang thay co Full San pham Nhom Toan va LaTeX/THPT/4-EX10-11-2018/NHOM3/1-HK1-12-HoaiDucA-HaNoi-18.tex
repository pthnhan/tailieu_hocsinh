\begin{bt}%[Đề thi HK1 THPT Hoài Đức A, Hà Nội, 2018]%[1D1B3]
Giải các phương trình sau
\begin{multicols}{2}
\begin{enumerate}
\item $3\sin x-\cos 2x+2=0$.
\item $2\cos 2x=\cos x+\sqrt{3}\sin x$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item Phương trình tương đương với 
\begin{eqnarray*}
&& 2\sin^2 x+3\sin x+1=0\\
& \Leftrightarrow & \hoac{& \sin x=-1\\ & \sin x=-\dfrac{1}{2}}\Leftrightarrow\hoac{& x=-\dfrac{\pi}{2}+k2\pi\\ & x=-\dfrac{\pi}{6}+k2\pi\\ & x=\dfrac{7\pi}{6}+k2\pi},\,\,\, (k\in\mathbb{Z}).
\end{eqnarray*}
Vậy nghiệm của phương trình là $x\in\left\{\left. -\dfrac{\pi}{2}+k2\pi;\,\, -\dfrac{\pi}{6}+k2\pi;\,\, \dfrac{7\pi}{6}+k2\pi\,\,\right|\,\, k\in\mathbb{Z}\right\}$.
\item Phương trình tương đương với
\begin{eqnarray*}
&& \cos 2x=\dfrac{1}{2}\cos x+\dfrac{\sqrt{3}}{2}\sin x\\
& \Leftrightarrow & \cos 2x=\cos\left(x-\dfrac{\pi}{3}\right)\\
& \Leftrightarrow & \hoac{ & 2x=x-\dfrac{\pi}{3}+k2\pi\\ & 2x=-x+\dfrac{\pi}{3}+k2\pi}\Leftrightarrow\hoac{& x=-\dfrac{\pi}{3}+k2\pi\\ & x=\dfrac{\pi}{9}+k\dfrac{2\pi}{3}},\,\, (k\in\mathbb{Z}).
\end{eqnarray*}
Vậy nghiệm của phương trình là $x\in\left\{\left. -\dfrac{\pi}{3}+k2\pi;\,\, \dfrac{\pi}{9}+k\dfrac{2\pi}{3}\right|\,\, k\in\mathbb{Z}\right\}$.
\end{enumerate}
}
\end{bt}

\item Ta có $\left(x^3+\dfrac{2}{x^2}\right)^{20}=\displaystyle\sum_{k=0}^{20}\mathrm{C}_{20}^{k}\cdot\left(x^3\right)^{20-k}\cdot\left(\dfrac{2}{x^2}\right)^k=\displaystyle\sum_{k=0}^{20}\mathrm{C}_{20}^{k}\cdot 2^k\cdot x^{60-5k}$.\\

\item Số phần tử của không gian mẫu là $n(\Omega)=\mathrm{C}_{12}^{2}$.\\
Số cách lấy ra $1$ bi vàng và $1$ bi đỏ khác số là $4\cdot 4=16$ (cách).\\
Số cách lấy ra $1$ bi xanh và $1$ bi đỏ khác số là $3\cdot 4=12$ (cách).\\
Số cách lấy ra $1$ bi xanh và $1$ bi vàng khác số là $3\cdot 3=9$ (cách).\\
Xác suất cần tính là $P=\dfrac{16+12+9}{\mathrm{C}_{12}^{2}}=\dfrac{37}{66}$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề thi HK1 THPT Hoài Đức A, Hà Nội, 2018]%[1D3G2]
Cho dãy số $(u_n)$ xác định bởi $\heva{& u_1=2\\ & u_{n+1}=2u_n+3n-1}$, $n\in\mathbb{N^{\ast}}$. Tìm công thức của số hạng tổng quát $u_n$.
\loigiai{
Ta có $u_n=2u_{n-1}+3(n-1)-1=2u_{n-1}+3n-4\Rightarrow u_n-2u_{n-1}=3n-4$.\\
Đặt $g(n)=an+b$ thỏa mãn $g(n)-2g(n-1)=u_n-2u_{n-1}=3n-4$. Tức là
\begin{eqnarray*}
&& an+b-2\left[a(n-1)+b\right]=3n-4,\,\,\forall n\in\mathbb{N^{\ast}}\\
& \Leftrightarrow & -an-b+2a=3n-4,\,\,\forall n\in\mathbb{N^{\ast}}\\
& \Leftrightarrow & \heva{& a=-3\\ & -b+2a=-4}\Leftrightarrow \heva{& a=-3\\ & b=-2}.
\end{eqnarray*}
Do đó $g(n)=-3n-2$. Bây giờ ta có
$$u_n-g(n)=2\left[u_{n-1}-g(n-1)\right]=2^2\left[u_{n-2}-g(n-2)\right]=\cdots=2^{n-1}\left[u_1-g(1)\right].$$
$$\Rightarrow u_n=g(n)+2^{n-1}\left[u_1-g(1)\right]=-3n-2+7\cdot 2^{n-1}.$$
Vậy công thức của số hạng tổng quát là $u_n=-3n-2+7\cdot 2^{n-1}$.
}
\end{bt}
