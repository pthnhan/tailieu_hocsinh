	{$\mathscr{D}= [-2; 2]$}
	{\True $\mathscr{D}= \mathbb{R}$}

	\loigiai{

\begin{ex}%[HK1, THPT Bến Tre, Vĩnh Phúc 17-18]%[Đào Anh Dương, 11HK1-18]%[1D2Y1-2]

	\choice
	{\True $64$}
	{$16$}
	{$32$}
	{$20$}
	\loigiai{
		Số cách chọn mua một cây bút mực là $8$ cách.\\
		Số cách chọn mua một cây bút chì là $8$ cách.\\

\begin{ex}%[HK1, THPT Bến Tre, Vĩnh Phúc 17-18]%[Đào Anh Dương, 11HK1-18]%[1D2B5-2]
	Một hộp đựng $4$ bi xanh và $6$ bi đỏ. Lần lượt rút $2$ viên bi, xác xuất để rút được một bi xanh và một bi đỏ là
	\choice
	{$\dfrac{2}{15}$}
	{$\dfrac{6}{25}$}
	{\True $\dfrac{8}{15}$}
	{$\dfrac{4}{15}$}
	\loigiai{
		Số cách lấy ra lần lượt $2$ bi từ hộp là $\mathrm{C}_{10}^1 \cdot \mathrm{C}_9^1 \Rightarrow n(\Omega) = \mathrm{C}_{10}^1 \cdot \mathrm{C}_9^1.$\\
		Số cách lấy ra lần lượt một bi xanh và một bi đỏ và ngược lại là $2 \cdot \mathrm{C}_{4}^1 \cdot \mathrm{C}_6^1 \Rightarrow n(A) = 2 \cdot \mathrm{C}_{4}^1 \cdot \mathrm{C}_6^1.$
		Vậy xác suất để lấy được một bi xanh và một bi đỏ là $\mathrm{P}(A)= \dfrac{n(A)}{n(\Omega)}=\dfrac{2 \cdot \mathrm{C}_{4}^1 \cdot \mathrm{C}_6^1}{\mathrm{C}_{10}^1 \cdot \mathrm{C}_9^1}= \dfrac{8}{15}.$
	}	
\end{ex}

\begin{ex}%[HK1, THPT Bến Tre, Vĩnh Phúc 17-18]%[Đào Anh Dương, 11HK1-18]%[1H1Y2-2]
	Trong mặt phẳng $Oxy$, cho điểm $M(1; -2)$. Phép tịnh tiến theo $\vec{v}=(-1; 1)$ biến điểm $M$ thành điểm $N$. Tìm tọa độ điểm $N$.
	\choice
	{\True $N(0; -1)$}
	{$N(2; -3)$}
	{$N(-2; 3)$}
	{$N(-1; 0)$}
	\loigiai{
		Gọi $N(x; y)$ là ảnh của $M(1;-2)$ qua phép tịnh tiến $\mathrm{T}_{\vec{v}}$.\\
		Khi đó ta có
			$$\mathrm{T}_{\vec{v}}(M)=N\Leftrightarrow \heva{&x= x_M -1\\ &y= y_M + 1} \Leftrightarrow \heva{&x= 1-1=0\\ &y=-2+1=-1}.$$
		Vậy $N(0; -1)$ là điểm cần tìm.
	}	
\end{ex}

\begin{ex}%[HK1, THPT Bến Tre, Vĩnh Phúc 17-18]%[Đào Anh Dương, 11HK1-18]%[1H1B2-2]
	Tìm ảnh của đường thẳng $(d) \colon 2x+ 3y- 1= 0$ qua phép tịnh tiến $\vec{v}= (2; 5)$.
	\choice
	{\True $2x+ 3y- 20= 0$}
	{$2x+ 3y- 18= 0$}
	{$2x+ 3y- 17= 0$}
	{$2x+ 3y- 16= 0$}
	\loigiai{
		Lấy điểm $M(x,y)$ tùy ý thuộc đường thẳng $(d)$.
		Gọi $M_1(x_1;y_1)$ là ảnh của $M$ qua $\mathrm{T}_{\vec{v}}$.\\
		 Khi đó ta có
		$$\mathrm{T}_{\vec{v}}(M)=M_1 \Leftrightarrow \heva{&x_1= x+ 2\\ &y_1= y+ 5} \Leftrightarrow \heva{&x= x_1- 2\\ &y= y_1- 5}.$$
		Do $M\in (d)$ nên ta có $2(x_1- 1)+ 3(y_1- 5)-1=0\Rightarrow 2x_1+ 3y_1- 20=0.$\\
		Vậy ảnh của $(d)$ qua phép tịnh tiến là đường thẳng $(d_1): 2x+ 3y- 20=0.$
	}	
\end{ex}

\begin{ex}%[HK1, THPT Bến Tre, Vĩnh Phúc 17-18]%[Đào Anh Dương, 11HK1-18]%[1H1B7-2]
	Phép vị tự tâm $O$ tỉ số $k=-2$ biến đường tròn $(x- 1)^2+ (y-2)^2= 4$ thành đường tròn
	\choice
	{$(x- 2)^2+ (y- 4)^2= 16$}
	{$(x- 4)^2+ (y- 2)^2= 16$}
	{$(x- 4)^2+ (y- 2)^2= 4$}
	{\True $(x+ 2)^2+ (y+ 4)^2= 16$}
	\loigiai{
		Đường tròn $(C)$ có tâm $I(1; 2)$ và bán kính $R= 2$. Gọi $I'(x; y)$ là ảnh của $I$ qua phép vị tự tâm $O$ tỉ số $k= -2$.\\
		Ta có $$\vec{OI'}= -2 \vec{OI} \Leftrightarrow \heva{&x= -2\\ &y= -4}.$$
		Bán kính đường tròn qua phép vị tự tâm $O$ tỉ số $k=-2$ là $R'=|k| \cdot R = 4$.\\
		Vậy đường tròn cần tìm là $(x+ 2)^2+ (y+ 4)^2= 16$.
	}	
\end{ex}

\begin{ex}%[HK1, THPT Bến Tre, Vĩnh Phúc 17-18]%[Đào Anh Dương, 11HK1-18]%[1H1B5-3]

\noindent\textbf{II. PHẦN TỰ LUẬN}

\begin{bt}%[HK1, THPT Bến Tre, Vĩnh Phúc 17-18]%[Đào Anh Dương, 11HK1-18]%[1D1Y3-1]%[1D1B3-3]
	Giải các phương trình sau.
	\begin{enumerate}
		\item $2\sin^2x - \sin x - 1 = 0$.
		\item $\cos x - \sqrt{3} \sin x = 2$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate} 
			\item 
			\begin{eqnarray*}
				2\sin^2x - \sin x - 1 = 0 &\Leftrightarrow &\hoac{&\sin x = 1 \\ &\sin x = -\dfrac{1}{2}}\\
				&\Leftrightarrow &\hoac{&x = \dfrac{\pi}{2} + k2\pi \\&x = -\dfrac{\pi}{6} + k2\pi \\&x = \dfrac{7\pi}{6} + k2\pi},k \in \mathbb{Z}.
			\end{eqnarray*}
			\item 
			\begin{eqnarray*}
				\cos x - \sqrt{3} \sin x = 2 &\Leftrightarrow &\dfrac{1}{2} \cos x - \dfrac{\sqrt{3}}{2} \sin x = 1\\
				&\Leftrightarrow &\cos \dfrac{\pi}{3} \cos x - \sin \dfrac{\pi}{3} \sin x = 1\\
				&\Leftrightarrow &\cos \left( x + \dfrac{\pi}{3} \right) = 1\\
				&\Leftrightarrow &x = -\dfrac{\pi}{3} + k2\pi, k \in \mathbb{Z}.
			\end{eqnarray*}
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[HK1, THPT Bến Tre, Vĩnh Phúc 17-18]%[Đào Anh Dương, 11HK1-18]%[1D2K3-2]%[1D2K3-2]
	\begin{enumerate}
		\item Cho khai triển $(x+1)^n = \mathrm{C}_n^0x^n + \mathrm{C}_n^1x^{n-1} + \mathrm{C}_n^2x^{n-2} + \ldots + \mathrm{C}_n^n$. Tìm tổng các hệ số trong khai triển, biết $\mathrm{C}_n^n + \mathrm{C}_n^{n-1} + \mathrm{C}_n^{n-2} = 79$.
		\item Tìm số hạng không chứa $x$ trong khai triển $\left( x + \dfrac{8}{x^2} \right)^9$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			\begin{align*}
			\mathrm{C}_n^n + \mathrm{C}_n^{n-1} + \mathrm{C}_n^{n-2} = 79 &\Leftrightarrow 1 + n + \dfrac{n(n-1)}{2} = 79\\
			&\Leftrightarrow n^2 + n - 156 = 0 \\
			&\Leftrightarrow \hoac{&n=12 \\ &n=-13 \,\text{(loại)}.}
			\end{align*}
			Do đó, tổng các hệ số trong khai triển là $(1+1)^{12} = 4096$.
			\item Ta có số hạng tổng quát thứ $k$ là $T_{k+1} = \mathrm{C}_9^k \cdot x^{9-k} \cdot 8^k \cdot x^{-2k}$.\\
			Yêu cầu bài toán tương đương với $9 - k - 2k = 0 \Leftrightarrow k = 3$.\\
			Vậy số hạng không chứa $x$ là $\mathrm{C}_9^3 \cdot 8^3 = 43008$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[HK1, THPT Bến Tre, Vĩnh Phúc 17-18]%[Đào Anh Dương, 11HK1-18]%[1D2B1-3]
	Cho tập hợp $A = \{0;1;2;3;4;5\}$. Từ tập hợp $A$ có thể lập được bao nhiêu số tự nhiên có $4$ chữ số khác nhau. Trong đó có bao nhiêu số chia hết cho $5$?
	\loigiai{
		\begin{itemize}
			\item Gọi số tự nhiên có $4$ chữ số là $\overline{abcd}; a,b,c,d \in A, a \ne 0$.\\
			Chọn $a$ có $5$ cách.\\
			Chọn ba số còn lại có $\mathrm{A}_5^3$ cách.\\
			Vậy có thể lập được $5 \cdot \mathrm{A}_5^3 = 300$ số tự nhiên có $4$ chữ số khác nhau.
			\item Để $\overline{abcd}$ chia hết cho $5$ ta có hai trường hợp\\
			TH1: $d=5$, khi đó\\
			Chọn $a$ có $4$ cách.\\
			Chọn hai số còn lại có $\mathrm{A}_4^2 = 12$ cách.\\
			Do đó, theo trường hợp $1$ có $4 \cdot 12 = 48$ số.\\
			TH2: $d=0$, khi đó\\
			Chọn ba số còn lại có $\mathrm{A}_5^3 = 60$ cách.\\
			Do đó, theo trường hợp $2$ có $60$ số.\\
			Vậy, trong các số tự nhiên có $4$ chữ số tìm được có $108$ số chia hết cho $5$.
		\end{itemize}
	}
\end{bt}

\begin{bt}%[HK1, THPT Bến Tre, Vĩnh Phúc 17-18]%[Đào Anh Dương, 11HK1-18]%[1D2B5-2]
	Để kiểm tra chất lượng từ một công ty sữa, người ta gửi đến bộ phận kiểm nghiệm $5$ hộp sữa cam, $4$ hộp sữa dâu và $3$ hộp sữa nho. Bộ phận kiểm nghiệm chọn ngẫu nhiên $3$ hộp sữa để phân tích mẫu. Tính xác suất để ba hộp sữa được chọn có cả ba loại.
	\loigiai{
	Chọn ngẫu nhiên 3 hộp sữa trong 12 hộp sữa để phân tích mẫu nên $n(\Omega) = \mathrm{C}_{12}^3 = 220$.\\
	Gọi $A$ là biến cố: \lq\lq Cả ba hộp sữa được chọn có cả ba loại\rq\rq, ta có $n(A) = \mathrm{C}_5^1 \cdot \mathrm{C}_4^1 \cdot \mathrm{C}_3^1 = 60$.\\
	Xác suất để ba hộp sữa được chọn có cả ba loại là $\mathrm{P}(A) = \dfrac{n(A)}{n(\Omega)} = \dfrac{60}{220} = \dfrac{3}{11}$.
	}
\end{bt}

\begin{bt}%[HK1, THPT Bến Tre, Vĩnh Phúc 17-18]%[Đào Anh Dương, 11HK1-18]%[1H2B1-2]%[1H2B2-3]
	Cho hình chóp $S.ABCD$ có các cặp cạnh đáy không song song với nhau. Trên $AB$ lấy điểm $M$, trên $SC$ lấy điểm $N$ ($M,N$ không trùng với các đầu mút).
	\begin{enumerate}
		\item Tìm giao tuyến của $(AMN)$ và $(SCD)$.
		\item Tìm giao điểm của $AN$ với $(SBD)$.
	\end{enumerate}
	\loigiai 
	{
		\begin{center}
			\begin{tikzpicture}[>=stealth,scale=0.9, line join = round, line cap = round]
			\tkzDefPoints{0/0/A,7/0/B,2/-4/D,7/-2.5/C,2/6/S}
			\coordinate (M) at ($(A)!0.73!(B)$);
			\coordinate (N) at ($(S)!0.3!(C)$);
			\tkzInterLL(A,C)(B,D)    \tkzGetPoint{P}
			\tkzInterLL(A,B)(C,D)    \tkzGetPoint{H}
			\tkzInterLL(N,H)(S,B)    \tkzGetPoint{g}
			\tkzInterLL(S,P)(A,N)    \tkzGetPoint{I}
			\tkzDrawPolygon(S,g,H,D,A)
			\tkzDrawSegments(S,C S,D N,g)
			\tkzDrawSegments[dashed](A,C B,D g,B A,H A,N N,M S,P B,C)
			\tkzDrawPoints(A,B,C,D,S,M,N,P,H,I)
			\tkzLabelPoints[above](S,N,B)
			\tkzLabelPoints[below](D,C,P,M)
			\tkzLabelPoints[left](A,I)
			\tkzLabelPoints[right](H)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Ta có $N$ là điểm chung thứ nhất.\\
			Gọi $H$ là giao điểm của $AB$ và $CD$. Ta có\\
			$\heva{&H \in AB \subset (AMN)\\&H \in CD \subset (SCD).}$\\
			Vậy $NH$ là giao tuyến cần tìm.
			\item Ta có $AN \subset (SAC)$. Trong mặt phẳng $(ABCD)$, gọi $P = AC \cap BD \Rightarrow (SAC) \cap (SBD) = SP$.\\
			Trong $(SAC)$, gọi $I = AN \cap SP$.\\
			Ta có $\heva{&I \in AN \\ &I \in SP \subset (SBD)} \Rightarrow I = AN \cap (SBD)$.
		\end{enumerate}
	}
\end{bt}
