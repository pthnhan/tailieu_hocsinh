\begin{ex}%[Đề thi vào 10, Sở giáo dục Long An, 2017]%[9D1Y2]%[9D1K8]%[9D1B2]
\hfill
	\begin{enumerate}[a)]
		\item Rút gọn: $3\sqrt{75}-12\sqrt{3}+\sqrt{12}$.
		\item Rút gọn: $N=\dfrac{x-2\sqrt{x}+1}{\sqrt{x}-1}-\dfrac{\sqrt{x}+x}{\sqrt{x}}$.
		\item Giải phương trình: $\sqrt{4x^2-12x+9}=9$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item $3\sqrt{75}-12\sqrt{3}+\sqrt{12}=15\sqrt{3}-12\sqrt{3}+2\sqrt{3}=5\sqrt{3}$.
			\item đk: $\heva{x>0 \\ x\neq 1 }$ \\
			$N=\dfrac{(\sqrt{x}-1)^2}{\sqrt{x}-1}-\dfrac{\sqrt{x}(\sqrt{x}+1)}{\sqrt{x}}=\sqrt{x}-1-\sqrt{x}-1=-2$.	
			\item $\sqrt{4x^2-12x+9}=9 \Leftrightarrow \sqrt{(2x-3)^2}=9 \Leftrightarrow |2x-3|=9 \Leftrightarrow \hoac{2x-3&=-9\\ 2x-3&=9} \Leftrightarrow \hoac{x&=-3\\x&=6}$.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Long An, 2017]%[9D4Y5]%[9D4K6]
\hfill
	\begin{enumerate}[a)]
		\item Giải phương trình: $3x^2+2x-8=0$.
		\item Cho phương trình: $x^2-2(m+1)x+m^2-3=0$. Tìm tất cả giá trị của tham số $m$ để phương trình trên có hai nghiệm phân biệt $x_1, x_2$ thỏa mãn: $\dfrac{x_1}{x_2}+\dfrac{x_2}{x_1}=2.$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item Ta có $\Delta ' = b'^2-ac=1+3.8=25>0$.\\ Vậy phương trình có hai nghiệm phân biệt:
			\begin{align*}
			x_1&=\dfrac{-b'-\sqrt{\Delta'}}{a}=\dfrac{-1-5}{3}=-2.\\
			x_2&=\dfrac{-b'+\sqrt{\Delta'}}{a}=\dfrac{-1+5}{3}=\dfrac{4}{3}.
			\end{align*}
			\item Ta có: $\Delta' = (m+1)^2-m^2+3=2m+4$.\\
			Để phương trình có hai nghiệm phân biệt thì: $2m+4>0 \Leftrightarrow m>-2. \qquad (1)$\\
			Theo hệ thức Vi-ét ta có: $x_1+x_2=2(m+1)$ và $x_1.x_2=m^2-3$. \\
			Ta có :
			\begin{align*}
			&\dfrac{x_1}{x_2}+\dfrac{x_2}{x_1}=2\\
			\Leftrightarrow &(x_1+x_2)^2-4x_1x_2=0	\Leftrightarrow 4(m+1)^2-4(m^2-3)=0\\
			\Leftrightarrow &4m^2+8m+4-4m^2+12=0 \Leftrightarrow m=-2. \qquad (2)
			\end{align*}				
			Từ (1) và (2) suy ra không có giá trị m thỏa mãn để: $\dfrac{x_1}{x_2}+\dfrac{x_2}{x_1}=2.$			
		\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Long An, 2017]%[9H1Y2]%[9H3Y7]%[9H2G5]%[9H2K2]	
	Cho đường tròn $(O)$ đường kính $  AC $. Trên bán kính $ OC $ lấy điểm $ B $ tùy ý ($ B $ không trùng $ O $ và $ C $). Gọi $ M $ là trung điểm của đoạn thẳng $ AB $. Qua $ M $ kẻ dây cung $ DE $ vuông góc với $ AB $. Kẻ $ BI $ vuông góc với $CD \quad (I \in CD).$
	\begin{enumerate}[a)]
		\item Cho $ AM = 4cm; MC = 9cm $. Tính độ dài đoạn thẳng $ MD  $ và $\tan A$ của tam giác $ MDA $.
		\item Chứng minh: $ BMDI $ là tứ giác nội tiếp
		\item Chứng minh $ ADBE $ là hình thoi và ba điểm $ I; B; E $ thẳng hàng.
		\item Gọi $ O' $ là tâm đường tròn đường kính $ BC $. Chứng minh: $ MI $ là tiếp tuyến của $(O')$
	\end{enumerate}	
	\loigiai{
		\begin{enumerate}[a)]
			\item Ta có $\widehat{ADC}=90^{\circ}$(góc nội tiếp chắn nữa đường tròn)\\
			\begin{minipage}[h]{0.45\textwidth}
				$\triangle ADC$ vuông tại $ D $, đường cao $ AH $ do đó ta có:\\
				$DM^2=AM.CM=4.9=36 \Rightarrow DM = 6$cm.\\
				$\triangle MDA$ vuông tại $ M $, ta có:\\
				$\tan A =\dfrac{DM}{MA}=\dfrac{6}{4}=\dfrac{3}{2}$.				
				\item Ta có: $\widehat{DMB}+\widehat{BID}=90^{\circ}+90^{\circ}=180^{\circ}$.\\
				$\Rightarrow$ Tứ giác $ BMDI $ nội tiếp.				
			\end{minipage}\hspace{1cm}			
			\begin{minipage}[h]{0.5\textwidth}			
				\begin{center}
					\begin{tikzpicture}[scale=1.4]
					\tkzInit[xmin=-2.5,xmax=2.5,ymin=-2.5,ymax=2.5]
					\tkzClip
					\tkzDefPoint(-2,0){A}
					\tkzDefPoint(2,0){C}					
					\tkzDefMidPoint(A,C) \tkzGetPoint{O} 
					\tkzDrawCircle[diameter](A,C)
					\tkzDefShiftPoint[C](0:-1.3){B} 
					\tkzDefMidPoint(A,B) \tkzGetPoint{M}
					\tkzDefMidPoint(B,C) \tkzGetPoint{O'}
					\tkzDefLine[perpendicular = through M](A,B) \tkzGetPoint{d} góc với AB
					\tkzInterLC(M,d)(O,A)	\tkzGetPoints{E}{D} 
					\tkzDefPointBy[projection=onto C--D](B) \tkzGetPoint{I} cho BI vuông góc với CD
					\tkzDrawCircle[diameter](B,C)					
					\tkzDrawPoints[color=blue](A,C,O,B,M,D,E,I,O')
					\tkzDrawSegments[thick, color=blue](A,C D,E C,D B,I M,D A,D I,E M,I O',I)
					\tkzMarkRightAngles[color=red](B,I,D D,M,B) 
					\tkzLabelPoints[below,font=\tiny](O,O')
					\tkzLabelPoints[below left,font=\tiny](A,E)
					\tkzLabelPoints[above,font=\tiny](D)
					\tkzLabelPoints[font=\tiny](C)
					\tkzLabelPoints[above left,font=\tiny](B)
					\tkzLabelPoints[above right,font=\tiny](I)
					\tkzLabelPoints[below left,font=\tiny](M)
					\end{tikzpicture}			
				\end{center}			
			\end{minipage}
			\item 	Ta có $DE \perp AB(gt) \Rightarrow MD=ME$\\
			Xét tứ giác $ ADBE $ có: \\ 
			$\heva{MD&=ME(cmt)\\MA&=MB(gt)}$
			$\Rightarrow$ Tứ giác ADBE là hình bình hành.\\
			Mặt khác ta lại có: $DE \perp AB(gt)$\\
			$\Rightarrow$ Tứ giác $ ADBE $ là hình thoi.\\
			$\Rightarrow DE \parallel  BE$ \\
			Mà $\widehat{ADC}=90^{\circ} \Rightarrow AD \perp DC \Rightarrow BE \perp DC$.\\
			Mặt khác $BI \perp DC$.
			Do đó 3 điểm $ E, B, I $ thẳng hàng. 
			\item Ta có: $\widehat{BIC}=90^{\circ} \Rightarrow I$ thuộc đường tròn tâm $O'$. \\
			Ta lại có: \\
			$\heva{&\triangle DIC \text{ vuông tại I}\\
				&\text{M là trung điểm của DE}}$ 
			$\Rightarrow IM=DM=\dfrac{DE}{2}$ \\
			$\Rightarrow \triangle MDI$ cân tại $M \Rightarrow \widehat{MDI}=\widehat{MID}.\qquad$ (1)\\
			Mặt khác ta có : $O'C=O'I$(Bán kính đường tròn tâm $O'$)\\
			$\Rightarrow \triangle O'CI$ cân tại $O'$ $\Rightarrow \widehat{O'CI}=\widehat{O'IC}. \qquad$ (2)\\
			Mà $\widehat{O'CI}+\widehat{MDI}=90^{\circ}. \qquad$ (3)\\
			Từ (1), (2) và (3) suy ra $\widehat{MIO'}=90^{\circ} \Rightarrow O'I \perp IM$ \\
			$\Rightarrow MI$ là tiếp tuyến của đường tròn tâm $O'$.			
		\end{enumerate}	}
\end{ex}
