	\begin{ex}%[Đề Giữa HK1, C Nghĩa Hưng, Nam Định, 2017]%[2H1G3]
		Cho hình hộp $ABCD.A'B'C'D'$ có $AB=a$, $B'C'=a\sqrt{5}$, các đường thẳng $A'B$ và $B'C$ cùng tạo với mặt phẳng $(ABCD)$ một góc $45^\circ$, tam giác $A'AB$ vuông tại $B$, tam giác $A'CD$ vuông tại $D$. Tính thể tích $V$ của khối hộp $ABCD.A'B'C'D'$ theo $a$.
		\choice
		{\True $V=2a^3$}
		{$V=\dfrac{2a^3}{3}$}
		{$V=\dfrac{a^3\sqrt{6}}{2}$}
		{$V=\dfrac{a^3\sqrt{6}}{6}$}
		\loigiai{
			\begin{center}
				\begin{tikzpicture}[scale=.6]
				\tkzDefPoints{0/0/A, -3/-2/B, 5/-2/C}

				\tkzDrawSegments[line width=1](B,B' B,C C,D D,D' C,C' B',C)

			Ta có $A'D\perp CD\Rightarrow A'D\perp AB$, vậy 
			$\heva{&A'B\perp AB\\&A'D\perp AB}\Rightarrow AB\perp(A'BD)\Rightarrow(A'BD)\perp(ABCD)$ theo giao tuyến $BD$.\\
			Ta có $B'C\parallel A'D$ nên $A'D$ tạo với $(ABCD)$ góc $45^\circ$.\\
			Gọi $H$ là hình chiếu của $A'$ xuống $(ABCD)$, $H\in BD$, ta có $\widehat{A'BH}=\widehat{A'DH}=45^\circ$ nên $\triangle A'BD$ vuông cân tại $A'$.\\
			Vậy $H$ là trung điểm của $BD$.\\
			Có $AB\perp(A'BD)\Rightarrow\triangle ABD$ vuông tại $B\Rightarrow BD=\sqrt{AD^2-AB^2}=2a$, $S_{ABD}=\dfrac{1}{2}AB\cdot BD=a^2$.\\
			Có $A'H=\dfrac{BD}{2}=a\Rightarrow V_{ABCD.A'B'C'D'}=A'H\cdot 2S_{ABD}=2a^3$.}
