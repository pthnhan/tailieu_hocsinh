\begin{bt}%[Đề HK1 lớp 11, Phú Nhuận, Hồ Chí Minh 2017 - 2018]%[1D1K3]%[1D1B3]%[1D1K3]
	Giải các phương trình sau
	\begin{enumerate}
		\item $\sin2x-\sin7x=\sqrt{3}(\cos2x+\cos7x)$.
		\item $\cos4x+6=7\cos2x$.
		\item $\dfrac{\cos x+\cos2x+\cos3x}{\sin\left(x-\dfrac{\pi}{4}\right)}=0$.
	\end{enumerate}
	\loigiai{\begin{enumerate}
			\item Phương trình tương đương với\\
			$\sin2x-\sqrt{3}\cos2x=\sin7x+\sqrt{3}\cos7x\Leftrightarrow\dfrac{1}{2}\sin2x-\dfrac{\sqrt{3}}{2}\cos2x=\dfrac{1}{2}\sin7x+\dfrac{\sqrt{3}}{2}\cos7x$\\
			$\Leftrightarrow\sin\left(2x-\dfrac{\pi}{3}\right)=\sin\left(7x+\dfrac{\pi}{3}\right)\Leftrightarrow\hoac{&2x-\dfrac{\pi}{3}=7x+\dfrac{\pi}{3}+k2\pi\\&2x-\dfrac{\pi}{3}=\pi-7x-\dfrac{\pi}{3}+k2\pi}$\\
			$\Leftrightarrow\hoac{&x=-\dfrac{2\pi}{15}-k\dfrac{2\pi}{5}\\&x=\dfrac{\pi}{9}+k2\pi}$, $k\in\mathbb{Z}$.
			\item Phương trình tương đương với\\
$2\cos^22x-1+6-7\cos2x=0\Leftrightarrow2\cos^22x-7\cos2x+5=0\Leftrightarrow\hoac{&\cos2x=1\\&\cos2x=\dfrac{5}{2}.}$\\
			Do $\cos2x\in[-1;1]$ nên $\cos2x=1\Leftrightarrow2x=k2\pi\Leftrightarrow x=k\pi$, $k\in\mathbb{Z}$.
			\item Điều kiện: $\sin\left(x-\dfrac{\pi}{4}\right)\ne0\Leftrightarrow x-\dfrac{\pi}{4}\ne k\pi\Leftrightarrow x\ne\dfrac{\pi}{4}+k\pi$.\\
			Phương trình tương đương với \\
			$\cos x+\cos2x+\cos3x=0\Leftrightarrow\cos2x+2\cos2x\cos x=0\Leftrightarrow\cos2x(2\cos x+1)=0$\\
			$\Leftrightarrow\hoac{&\cos2x=0\\&\cos x=-\dfrac{1}{2}}\Leftrightarrow\hoac{&x=\dfrac{\pi}{4}+k\dfrac{\pi}{2}\\&x=\pm\dfrac{2\pi}{3}+k2\pi.}$\\
			So điều kiện ta có nghiệm phương trình là $\hoac{&x=\dfrac{3\pi}{4}+k\pi\\&x=\pm\dfrac{2\pi}{3}+k2\pi}$, $k\in\mathbb{Z}$.
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Đề HK1 lớp 11, Phú Nhuận, Hồ Chí Minh 2017 - 2018]%[1D2K3]
Xét khai triển của biểu thức $(x+1)^{10}(x+2)=a_{11}\cdot x^{11}+a_{10}\cdot x^{10}+a_9\cdot x^9+\cdots+a_1\cdot x+a_0$. Hãy xác định giá trị của hệ số $a_6$.
	\loigiai{Ta có $(x+1)^{10}(x+2)=x(x+1)^{10}+2(x+1)^{10}=x\cdot\sum\limits_{k=0}^{10}{\mathrm{C}_{10}^k\cdot x^{10-k}}+2\cdot\sum\limits_{\ell=0}^{10}{\mathrm{C}_{10}^\ell\cdot x^{10-\ell}}$\\
		$=\sum\limits_{k=0}^{10}{\mathrm{C}_{10}^k\cdot x^{11-k}}+\sum\limits_{\ell=0}^{10}{2\mathrm{C}_{10}^\ell\cdot x^{10-\ell}}$\\
		Hệ số $a_6$ là của $x^6$ nên ta có $\heva{&11-k=6\\&10-\ell=6}\Leftrightarrow\heva{&k=5\\&\ell=4}$.\\
		Vậy $a_6=\mathrm{C}_{10}^5+2\mathrm{C}_{10}^4=672$.
	}
\end{bt}

\begin{bt}%[Đề HK1 lớp 11, Phú Nhuận, Hồ Chí Minh 2017 - 2018]%[1D2K5]
	Một lớp học có $45$ học sinh được đánh số thứ tự từ $1$ đến $45$, thầy giáo gọi ngẫu nhiên $3$ học sinh lên bảng để trả bài. Tính xác suất để $3$ học sinh được gọi có tổng các số thứ tự là một số lẻ.
	\loigiai{\begin{itemize}
			\item Số cách gọi ngẫu nhiên $3$ học sinh lên bảng trả bài là $\big|\Omega\big|=\mathrm{C}_{45}^3$.
			\item Gọi $A$ là biến cố các bạn được gọi có tổng các số thứ tự là một số lẻ. Trong các số từ $1$ đến $45$, có $23$ chữ số lẻ và $22$ chữ số chẵn nên ta có hai trường hợp:
			\begin{itemize}
				\item \underline{TH1:} Ba số được gọi đều là số lẻ, có $\mathrm{C}_{23}^3$ cách để gọi.
				\item \underline{TH2:} Có hai số chẵn và một số lẻ được gọi, có $\mathrm{C}_{22}^2\cdot\mathrm{C}_{23}^1$ cách để gọi.
			\end{itemize}
			\item Vậy ta có $\big|\Omega_A\big|=\mathrm{C}_{23}^3+\mathrm{C}_{23}^1\cdot\mathrm{C}_{22}^2$, suy ra $\mathrm{P}(A)=\dfrac{\big|\Omega_A\big|}{\big|\Omega\big|}=\dfrac{\mathrm{C}_{23}^3+\mathrm{C}_{23}^1\cdot\mathrm{C}_{22}^2}{\mathrm{C}_{45}^3}=\dfrac{322}{645}$.
	\end{itemize}}
\end{bt}
