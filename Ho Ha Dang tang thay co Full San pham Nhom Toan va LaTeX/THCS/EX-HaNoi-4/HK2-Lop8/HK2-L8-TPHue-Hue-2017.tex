\begin{ex}%[HK2, Toán 8, TP Huế - Thừa Thiên Huế, năm 2017]%[8D3B4]%[8D3B5]
Giải phương trình
\begin{enumerate}
\item $9x^2-1=(3x+1)(2x+5)$.
\item $\dfrac{2x}{x-1}+\dfrac{3(x+1)}{x}=5$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item \begin{eqnarray*}
&&9x^2-1=(3x+1)(2x+5)\\ 
& \Leftrightarrow& (3x-1)(3x+1)=(3x+1)(2x+5)\\
&\Leftrightarrow&(3x+1)(3x-1-2x-5)=0\\
&\Leftrightarrow &(3x+1)(x-6)=0\\
& \Leftrightarrow &\hoac{&3x+1=0\\&x-6=0}\\
 &\Leftrightarrow &\hoac{&x=-\dfrac{1}{3}\\&x=6.}
\end{eqnarray*}
\item $\dfrac{2x}{x-1}+\dfrac{3(x+1)}{x}=5$.\quad (1)\\
Điều kiện: $\heva{&x \ne 1\\&x\ne 0.}$
\begin{eqnarray*}
(1) &\Leftrightarrow &2x^2+3(x+1)(x-1)=5x(x-1) \\ 
&\Leftrightarrow&  5x^2-3=5x^2-5x\\
&\Leftrightarrow&x=\dfrac{3}{5}.
\end{eqnarray*}
Vậy phương trình có nghiệm $x=\dfrac{3}{5}$.
\end{enumerate}
}
\end{ex}

\hfill
\begin{enumerate}
\item Giải bất phương trình $x-3+\dfrac{x-4}{6} \le \dfrac{4x-5}{3}$ và biểu diễn tập nghiệm trên trục số.
\item Chứng minh $2(a^2+b^2) \ge (a+b)^2$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item  \begin{eqnarray*}
&&x-3+\dfrac{x-4}{6} \le \dfrac{4x-5}{3}\\ 
& \Leftrightarrow &6(x-3)+x-4 \le 2(4x-5)\\
& \Leftrightarrow &7x-22 \le 8x-10\\
& \Leftrightarrow & x \ge -12.
\end{eqnarray*}
Biểu diễn tập nghiệm trên trục số:

\item Ta có $2(a^2+b^2)-(a+b)^2=2a^2+2b^2-a^2-2ab-b^2=a^2+b^2-2ab=(a-b)^2 \ge 0$.\\

\begin{ex}%[HK2, Toán 8, TP Huế - Thừa Thiên Huế, năm 2017]%[8D3K6]
Một vườn hoa hình tam giác có chiều cao bằng $\dfrac{1}{2}$ cạnh đáy tương ứng. Nếu tăng chiều cao $3$ m và thu hẹp cạnh đáy tương ứng $2$ m thì được vườn hoa tam giác mới có diện tích tăng thêm $21$ m$^2$ so với vườn cũ. Tính chiều cao và cạnh đáy tương ứng của vườn hoa sau khi tu sửa.
\loigiai{
Gọi chiều cao ban đầu của vườn hoa hình tam giác là $x$ (m) ($x>0$).\\
Khi đó cạnh đáy của tam giác ban đầu là $2x$ (m).\\
Theo đề ta có phương trình
\begin{eqnarray*}
&&\dfrac{1}{2}(x+3) \cdot (2x-2)=21+\dfrac{1}{2}x \cdot 2x\\ 
&\Leftrightarrow& 2x^2+4x-6=42+2x^2\\
&\Leftrightarrow &x=12.
\end{eqnarray*}
Vậy chiều cao và cạnh đáy tương ứng của vườn hoa sau khi tu sửa là $15$ m và $28$ m.
}
\end{ex}

\begin{ex}%[HK2, Toán 8, TP Huế - Thừa Thiên Huế, năm 2017]%[8H3B8]%[8H3K8]%[8H3G8]
Cho tam giác $ABC$ vuông tại $A$ $(AB<AC)$, đường cao $AH$. Qua $B$ vẽ đường thẳng song song với $AC$ cắt $AH$ tại $D$.
\begin{enumerate}
\item Chứng minh tam giác $AHC$ đồng dạng với tam giác $DHB$.
\item Chứng minh $AB^2=AC.BD$.
\item Gọi $M, N$ lần lượt là trung điểm của $BD, AC$. Chứng minh $3$ điểm $M, H, N$ thẳng hàng.
\end{enumerate}
\loigiai{
\immini{
\begin{enumerate}
\item Xét $\triangle AHC$ và $\triangle DHB$ có\\
 $\heva{&\widehat{AHC}=\widehat{DHB}=90^\circ\\&\widehat{DBH}=\widehat{ACH} \;\text{(slt)}} 
\Rightarrow \triangle AHC \backsim \triangle DHB$.
\end{enumerate}
}
{

\begin{tikzpicture}[scale=1]
\clip (-2.5,-4.5) rectangle (4.5,0.5);
\coordinate (A) at (0,0);
\coordinate (C) at (-40:5 cm);

\tkzDrawAltitude(B,C)(A) \tkzGetPoint{H}
\tkzDefLine[parallel=through B](A,C) \tkzGetPoint{l}
%\tkzDefPointBy[projection=onto A--B](B)\tkzGetPoint{d}
\tkzInterLL(A,H)(B,l) \tkzGetPoint{D}
\tkzDefMidPoint(D,B)    \tkzGetPoint{M}
\tkzDefMidPoint(A,C)    \tkzGetPoint{N}
\draw (A)--(B)--(C)--(A) (A)--(D) (B)--(D) (M)--(N);
\tkzDrawPoints(D,A,B,C,M,N,H)
\tkzLabelPoints[right](N,C) 
\tkzLabelPoints[left](B,P) 
\tkzLabelPoints[above](A) 
\tkzLabelPoints[below](D,M)
\tkzLabelPoints[below right](H)
\tkzMarkRightAngles(B,A,C A,B,D A,H,B)
\tkzMarkSegments[mark=|](M,B M,D)
\tkzMarkSegments[mark=||](N,A N,C)
\end{tikzpicture}
}

\begin{enumerate}
\item[b)] Ta có $\heva{&AC \perp AB\\&BD \parallel AC} \Rightarrow BD \perp AB \Rightarrow \widehat{ABD}=90^\circ$.\\
Xét $\triangle ABC$ và $\triangle BDA$ có 
$$\heva{&\widehat{BAC}=\widehat{DBA}=90^\circ\\&\widehat{BAD}=\widehat{BCA} \;\text{(cùng phụ góc $\widehat{HAC}$)}} \Rightarrow \triangle ABC \backsim \triangle BDA \;\text{(g.g)}$$
$\triangle ABC \backsim \triangle BDA \Rightarrow \dfrac{AB}{BD}=\dfrac{AC}{AB} \Rightarrow AB^2=AC \cdot BD$.
\item[c)] Vì $\triangle AHC \backsim \triangle DHB \Rightarrow \dfrac{HB}{HC}=\dfrac{DB}{AC}$.\\
Mà $DB=2BM; AC=2CN \Rightarrow \dfrac{HB}{HC}=\dfrac{BM}{CN}$.\\
Xét $\triangle HBM$ và $\triangle HCN$ có $\heva{&\widehat{HBM}=\widehat{HCN}\\&\dfrac{HB}{HC}=\dfrac{BM}{CN}} \Rightarrow \triangle HBM \backsim \triangle HCN$ (c.g.c).\\
$\triangle HBM \backsim \triangle HCN \Rightarrow \widehat{BHM}=\widehat{CHN} \Rightarrow H, N, M$ thẳng hàng.
\end{enumerate}

}
\end{ex}
