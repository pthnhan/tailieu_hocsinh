\begin{ex}%[Đề thi vào 10, Sở giáo dục Lâm Đồng, 2017]%[9D3B4]

	\loigiai{$\heva{&x+2y=-4\\&3x-y=9}\Leftrightarrow \heva{x+2y=-4\\6x-2y=18} \Leftrightarrow \heva{&x+2y=-4\\&7x=14}\Leftrightarrow \heva{&2+2y=-4\\&x=2}\Leftrightarrow \heva{&x=2\\&y=-3.}$\\

\begin{ex}%[Đề thi vào 10, Sở giáo dục Lâm Đồng, 2017]%[9D2K5]
	Viết phương trình đường thẳng $(d)$ đi qua điểm $B(1;-2)$ và song song với đường thẳng $(d'): y=3x$.
	\loigiai{Vì $(d)$ song song với $(d')$ nên phương trình có dạng $y=3x+m$ với $m\neq 0$. Mà $(d)$ đi qua $B(1;-2)$ nên ta có $-2=3.1+m\Leftrightarrow m=-5$ (thoả mãn $m\neq 0$ ).

\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Lâm Đồng, 2017]%[9D4K7]
	Giải phương trình $9x^4+8x^2-1=0$.
	\loigiai{Đặt $t=x^2$ với điều kiện $t\ge 0$ ta có phương trình $9t^2+8t-1=0\Leftrightarrow \hoac{&t=-1\,\, (\textrm{loại })\\&t=\dfrac{1}{9}.}$ Với $t=\dfrac{1}{9}$ ta có $x^2=\dfrac{1}{9}\Leftrightarrow \hoac{&x=\dfrac{1}{3}\\&x=-\dfrac{1}{3}.}$\\
		Vậy phương trình đã cho có hai nghiệm $x=\dfrac{1}{3}$ và $x=-\dfrac{1}{3}$.}
\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Lâm Đồng, 2017]%[9H4K2]
	Một hình nón có diện tích toàn phần bằng $90\pi \mathrm{ cm}^2$ và độ dài đường sinh bằng $13$ cm. Tính thể tích của hình nón đó.
	\loigiai{Gọi $r$ là bán kính đáy của hình nón. Theo công thức tính diện tích toàn phần của hình nón ta có phương trình $\pi r^2+13\pi r=90\pi\Leftrightarrow r^2+13r-90=0\Leftrightarrow \hoac{&r=5\\&r=-18 \textrm{ (loại).}}$\\
		Gọi $h$ là đường cao của hình nón ta có $h=\sqrt{13^2-r^2}=12$.\\
		Áp dụng công thức tính thể tích của hình nón ta có $V=\dfrac{1}{3}\pi r^2h=100\pi$ $(\mathrm{ cm}^3)$.}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Lâm Đồng, 2017]%[9D4K2]
	Trong mặt phẳng toạ độ $Oxy$, cho Parabol $(P): y=x^2$ và đường thẳng $(d): y=2(m+3)x+1-4m$ ($m$ là tham số). Với giá trị nào của $m$ thì $(d)$ cắt $(P)$ tại hai điểm phân biệt cùng nằm bên phải trục tung.
	\loigiai{Yêu cầu bài toán tương đương phương trình $x^2=2(m+3)x+1-4m$ có hai nghiệm dương phân biệt. Hay phương trình
		$x^2-2(m+3)x-1+4m=0$ có hai nghiệm dương phân biệt, tức là:
		$$\heva{&\Delta'=(m+3)^2+1-4m>0\\&m+3>0\\&-1+4m>0}\Leftrightarrow \heva{&(m+1)^2+9>0\\&m>-3\\&m>-\dfrac{1}{4}}\Leftrightarrow m>\dfrac{1}{4}.$$
		Kết luận: với $m>\dfrac{1}{4}$ thì $(d)$ cắt $(P)$ tại hai điểm phân biệt cùng nằm bên phải trục tung. }
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Lâm Đồng, 2017]%[9D4K3]
	Tìm giá trị nhỏ nhất của biểu thức $P=x-\sqrt{x-2017}+\dfrac{1}{4}$.
	\loigiai{Đặt $t=\sqrt{x-2017}, t\ge 0$ ta có $t^2=x-2017$ hay $x=t^2+2017$.\\
		Từ đó ta có $P=t^2+2017-t+\dfrac{1}{4}=\left(t-\dfrac{1}{2}\right)^2+2017\geq 2017$, đẳng thức xảy ra khi $t=\dfrac{1}{2}$ tức là khi $x=2017+\dfrac{1}{4}=\dfrac{8069}{4}$.
		Vậy giá trị nhỏ nhất của $P$ bằng $2017$ đạt được tại $x=\dfrac{8069}{4}$.}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Lâm Đồng, 2017]%[9D4G4]
	Cho phương trình: $ax^2+bx+c=0$ ($a,b,c$ là các hệ số và $a>0$). Chứng minh rằng nếu $b>a+c$ thì phương trình luôn có hai nghiệm phân biệt.
	\loigiai{Ta xét hai trường hợp.\\
		\textbf{Trường hợp 1:} $c<0$, khi đó hiển nhiên $a$ và $c$ trái dấu nên phương trình có hai nghiệm phân biệt trái dấu.\\
		\textbf{Trường hợp 2:} $c\ge 0$, từ giả thiết ta có $$b>a+c>0\Rightarrow b^2>(a+c)^2\Leftrightarrow b^2-4ac>a^2-2ac+c^2\Leftrightarrow b^2-4ac>(a-c)^2.$$
		Mà $(a-c)^2\ge 0$ nên $\Delta=b^2-4ac>0$ suy ra phương trình đã cho có hai nghiệm phân biệt.\\
		Vậy phương trình đã cho luôn có hai nghiệm phân biệt.}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Lâm Đồng, 2017]%[9H3G8]
	Cho tam giác $ABC$ nhọn, nội tiếp đường tròn $(O)$. Gọi $E$ là điểm chính giữa của cung nhỏ $BC$, $AE$ cắt $BC$ tại $D$. Chứng minh rằng $AC^2.BD=AD.CD.AE$.
	\loigiai{\immini{Từ giả thiết dễ dàng thấy rằng $\Delta ADC\sim \Delta BDE$\\ nên $AC.BD=AD.BE$.\\
			Từ đó ta có $AC^2.BD=AC.BD.AC=AD.BE.AC$\\
			Vậy ta cần chứng minh $BE.AC=CD.AE$.\\
			Thật vậy dễ thấy:\\
			$\Delta CDA\simeq \Delta EBA\Rightarrow \dfrac{CD}{CA}=\dfrac{EB}{EA}\Rightarrow \dfrac{CD}{CA}=\dfrac{EC}{EA}$\\
			$\Rightarrow CD.EA=EC.AC\Leftrightarrow CD.AE=BE.AC$.\\ Từ đó ta có điều phải chứng minh.}

			\tkzDefPoints{4/4/O, 1.89/7.4/A, .55/1.98/B, 7.47/2/C, 4/0/E}
			\tkzDrawSegments(A,B B,C C,A A,E)
			\tkzLabelPoints[left](O, B)
			\tkzLabelPoints[left, above](A)
			\tkzLabelPoints[right](C)
			\tkzLabelPoints[below](E)
			\draw (O) circle (4); \draw (O) circle (0.01);
			\node[right, above] at (3.1,1) {\normalsize{$D$}};

\end{ex}
