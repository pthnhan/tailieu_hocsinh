\section{HỆ THỨC VI-ÉT VÀ ỨNG DỤNG}

\subsection{Tóm tắt lý thuyết}
\subsubsection{Hệ thức Vi -ét }
Phương trình bậc hai tổng quát $ax^2 + bx + c=0$ $\ (a\ne 0)$. \hfill(1)\\
Nếu phương trình $(1) $ có hai nghiệm $x_1,\ x_2$ thì $\begin{cases} S= x_1+ x_2 = -\dfrac{b}{a}\\ P= x_1\cdot x_2= \dfrac{c}{a}. \end{cases}$\\
Đảo lại nếu hai số $x_1,\ x_2$ thỏa mãn $\begin{cases} S= x_1+x_2\\ P=x_1x_2. \end{cases}$ thì $x_1,\ x_2$ là nghiệm của phương trình $x^2 - Sx + P =0$ ( điều kiện $S^2-4P\ge0 $).\\
Các hệ thức liên hệ giữa hai nghiệm thường được vận dụng để giải toán.
\begin{listEX}
	\item $x_1^2 + x_2^2 = (x_1+x_2)^2 -2x_1x_2.$
	\item $x_1^3 + x_2^3 =(x_1+x_2)^3 -3x_1x_2(x_1+x_2).$
	\item $x_1^4 +x_2^4 = (x_1^2+x_2^2)^2 -2x_1^2x_2^2 =\left[(x_1+x_2)^2 -2x_1x_2\right]^2 -2x_1^2x_2^2.$
	\item $|x_1-x_2| = \sqrt{(x_1-x_2)^2} = \sqrt{(x_1+x_2)^2 -4x_1x_2}.$
	\item $\dfrac{x_1}{x_2} + \dfrac{x_2}{x_1} = \dfrac{x_1^2+x_2^2}{x_1x_2} = \dfrac{(x_1+x_2)^2 -2x_1x_2}{x_1x_2}$ với $x_1,\ x_2 \ne 0$.
	\item $\dfrac{1}{x_1^2 } +\dfrac{1}{x_2^2} = \dfrac{x_1^2+x_2^2}{x_1^2x_2^2} = \dfrac{(x_1+x_2)^2-2x_1x_2}{(x_1x_2)^2}$ với $x_1,\ x_2 \ne 0$
	\item $(x_1-x_2)^2 = (x_1+x_2)^2 -4x_1x_2.$
\end{listEX}
\subsection{Áp dụng giải toán}
\begin{dang} {Tính giá trị biểu thức đối xứng giữa các nghiệm}
	Biểu thức đối xứng giữa các nghiệm $x_1$ và $x_2$ của phương trình $ax^2 + bx + c= 0$ là biểu thức có giá trị không thay đổi khi ta hoán vị $x_1$ và $x_2$.\\
	Ta có thể biểu thị được các biểu thức đối xứng giữa các nghiệm $x_1$ và $x_2$ theo $S$ và $P$, ví dụ như
	\begin{itemize}
	\item $x_1^2 + x_2^2 = (x_1 +x_2)^2 - 2x_1x_2 = S^2 -2P.$
	\item $\dfrac{1}{x_1} + \dfrac{1}{x_2} = \dfrac{x_1+x_2}{x_1x_2} = \dfrac{S}{P}$ với $x_1,\ x_2 \ne 0$
	\item $x_1^3 + x_2^3 =(x_1 +x_2)^3 - 3x_1x_2(x_1+x_2) = S^3 -3SP$ với $x_1,\ x_2 \ne 0$
	\item $\dfrac{1}{x_1^2} + \dfrac{1}{x_2^2} = \dfrac{x_1^2+x_2^2}{x_1^2x_2^2} = \dfrac{S^2 -P}{P^2}.$
	\end{itemize}
	\end{dang}
\begin{vd}
	Giả sử phương trình $ax^2 + bx+ c=0.$
	\begin{enumEX}{2}
		\item $-x_1$ và $-x_2$.
		\item $2x_1$ và $2x_2$.
		\item $x_1^2$ và $x_2^2.$
		\item $x_1 + x_2$ và $x_1x_2$.
		\item $\dfrac{1}{x_1}$ và $\dfrac{1}{x_2}$.
		\end{enumEX}
\end{vd}
\loigiai{Giả sử phương trình $ax^2 + bx+c=0$ có hai nghiệm $x_1, x_2$ ta có\\
$\begin{cases} S= x_1+x_2 = -\dfrac{b}{a}\\ P = x_1x_2 = \dfrac{c}{a}. \end{cases}$ 
\begin{enumEX}{1}	\item 	Ta có $\begin{cases} (-x_1) +(-x_2) =-S\\ (-x_1)(-x_2)=P . \end{cases}$\\	
	Do đó $-x_1 $ và $-x_2$ là nghiệm của phương trình $X^2 -SX + P=0.$
\item Ta có $\begin{cases}  2x_1+2x_2 = 2S\\ 2 x_12x_2 = 4P. \end{cases}$ \\
Do đó $2x_1 $ và $2x_2$ là nghiệm của phương trình $X^2 -2SX + 4P=0.$
\item Ta có $\begin{cases}  x_1^2 + x_2^2 =S^2 - 2P\\ x_1^2 \cdot x_2^2 = P^2. \end{cases}$ \\
Do đó $x_1^2 $ và $x_2^2$ là nghiệm của phương trình $X^2 -(S^2 -2P)X + P^2=0.$
\item Ta có $\begin{cases}  (x_1+x_2)+x_1x_2 =S+P\\ (x_1+x_2)\cdot x_1x_2 =S\cdot P. \end{cases}$ \\
Do đó $x_1+x_2 $ và $x_1x_2$ là nghiệm của phương trình $X^2 -(S+P)X + S\cdot P=0.$
\item Ta có $\begin{cases}  \dfrac{1}{x_1}+ \dfrac{1}{x_2} = \dfrac{S}{P}\\\dfrac{1}{x_1}\cdot \dfrac{1}{x_2} = \dfrac{1}{P}. \end{cases}$ \\
Do đó $\dfrac{1}{x_1} $ và $\dfrac{1}{x_2}$ là nghiệm của phương trình $X^2 -\dfrac{S}{P}X + \dfrac{1}{P}=0.$
\end{enumEX}}
\begin{dang}{Tìm giá trị của tham số khi biết hệ đối xứng giữa các nghiệm}
	\begin{itemize}
		\item Ta tìm điều kiện cho tham số để phương trình đã cho có hai nghiệm $x_1, x_2$ (thường là $a\ne 0$ và $\Delta \ge 0$).
		\item Từ biểu thức nghiệm đã cho, áp dụng hệ thức Vi-ét để giải phương trình (có ẩn là tham số).
		\item Đối chiếu với điều kiện xác định của tham số rồi sau đó xác định giá trị cần tìm.
	\end{itemize}
\end{dang}

\begin{vd}
Cho phương trình $x^2 - 5x + m =0$ ($m$ là tham số).
\begin{listEX}
	\item Giải phương trình trên khi $m = 6.$
	\item Tìm $m$ để phương trình có hai nghiệm $x_1,\ x_2$ thỏa mãn $|x_1 - x_2| = 3.$
	\end{listEX}
\end{vd}
\loigiai{\begin{listEX}
		\item Với $m = 6$, ta có phương trình $x^2 - 5x+ 6 = 0$.\\
		$\Delta = 25 - 4\cdot 6 = 1$. Suy ra phương trình có hai nghiệm $x_1 = 3;\ x_2= 2.$
		\item Ta có $\Delta = 25- 4\cdot m.$\\
		Để phương trình đã cho có nghiệm thì $\Delta \ge 0 \Leftrightarrow m \le \dfrac{25}{4}$ \hfill (*)\\
		Theo hệ thức Vi-ét, ta có $\begin{aligned}[t]
		x_1+ x_2 = 5 \quad (1)\\
		x_1 x_2 = m \quad (2)
		\end{aligned}$\\
		Mặt khác theo đề bài ta có $|x_1 - x_2 |=3$. \hfill (3)\\
		Từ $(1)$ và $(3)$ suy ra $|x_1 - 5+x_1| =3 \Leftrightarrow |2x_1 -5|=3 \Leftrightarrow \left[\begin{aligned} 2x_1 -5 & =3\\ 2x_1-5 &=-3 \end{aligned}\right.\Leftrightarrow \left[\begin{aligned}x_1& =4\\ x_1&=1. \end{aligned}\right.$\\
		Suy ra $x_1 = 4; \ x_2= 1$ hoặc $x_1= 1;\ x_2 =4.$ \hfill(4)\\
		Từ $(2)$ và $(4)$ suy ra $m = 4$. Thử lại thì thỏa mãn.\\
		Vậy với $m = 4$ thỏa yêu cầu bài toán.
	\end{listEX}}
\begin{vd}
	Giải phương trình $x^2 -2(m-1)x -m-3=0 \quad (1)$ (với $m$ là tham số).
	\begin{enumEX}{1}
		\item Giải phương trình với $m = -3$.
		\item Với giá trị nào của $m$ thì phương trình $(1)$ có các nghiệm $x_1,\  x_2$ thỏa mãn $x_1^2 + x_2^2= 10.$
		\end{enumEX}
\end{vd}
\loigiai{\begin{enumEX}{1}
		\item Với $m = -3$ ta có phương trình $x^2 + 8x =0 \Leftrightarrow x(x+8) = 0 \Leftrightarrow  \left[\begin{aligned} x & =0\\ x &=-8. \end{aligned}\right.$ 
		
		\item Phương trình $(1)$ có $2$ nghiệm phân biệt khi\\
		$\begin{aligned}[t]
		\Delta' \ge 0 \Leftrightarrow (m-1)^2 + (m+3) \ge 0 \Leftrightarrow& \ m^2 - 2m + 1+ m+ 3 \ge 0\\ \Leftrightarrow \ &\ m^2 - m+ 4\ge 0\\ \Leftrightarrow\ &\  \left(m- \dfrac{1}{2}\right)^2 + \dfrac{15}{4} > 0\ \text{đúng với mọi m}
		\end{aligned}$ .\\
		Vậy chứng tỏ phương trình có $2$ nghiệm phân biệt với mọi $m$.\\
		Theo hệ thức Vi-ét ta có $\begin{cases} x_1 + x_2 =2(m-1) \quad (1)\\x_1x_2 = -m-3 \quad (2). \end{cases}$\\
		Ta có  \begin{eqnarray*}
		x_1^2 + x_2^2 =10 \Leftrightarrow &(x_1+x_2)^2 -2x_1x_2 
		&=10\\ \Leftrightarrow &4(m-1)^2 +2(m+3) &=10\\ \Leftrightarrow &4m^2 -6m +10 &=10 \\\Leftrightarrow &2m(2m-3) &=0 \Leftrightarrow \left[\begin{aligned} m & =0\\ m &= \dfrac{3}{2}. \end{aligned}\right.
		\end{eqnarray*}
	\end{enumEX}}

Vậy với $m= 0$ hoặc $m = \dfrac{3}{2}$ thỏa yêu cầu bài toán.
\begin{dang}{Tìm hai số khi biết tổng và tích của chúng}
Cho hai số $x, y$ biết $x+y =S;\  x\cdot y =P$ thì $x,\ y$ là hai nghiệm của phương trình bậc hai $X^2 - S\cdot X + P=0$ ( điều kiện của hai số đó là $S^2 - 4P \ge 0$).

\subsection{Luyện tập}

\begin{bt}
	Cho phương trình $x^2-5x+2=0$. Không giải phương trình, gọi $x_1, \ x_2$ là hai nghiệm của phương trình. Hãy tính giá trị của biểu thức
	\begin{enumEX}{2}  
		\item $A=x_1^2+x_2^2$.
		\item $B=\left|x_1-x_2\right|$.
		\item $C=\dfrac{1}{x_1^3}+\dfrac{1}{x_2^3}$.
		\item $D=\dfrac{x_1}{\sqrt{x_2}}+\dfrac{x_2}{\sqrt{x_1}}$.
	\end{enumEX}
	\loigiai{
		Vì $\Delta=5^2-4\cdot 1\cdot 2=17>0$ nên phương trình đã cho có $2$ nghiệm phân biệt.\\
		Theo hệ thức Vi-ét, ta có
		\[x_1+x_2=5, x_1x_2=2.\]
		\begin{enumEX}{1}
			\item 
			Ta có
			\[A=(x_1+x_2)^2-2x_1x_2=5^2-2\cdot 2=21.\]
			\item
			Ta có
			\[B=\left|x_1-x_2\right|=\sqrt{(x_1-x_2)^2}=\sqrt{(x_1+x_2)^2-4x_1x_2}=\sqrt{5^2-4\cdot2}=\sqrt{17}.\]
			\item
			Vì $x_1+x_2>0$ và $x_1x_2>0$ nên $x_1>0, x_2>0$. Ta có
			\[C=\dfrac{x_1^3+x_2^3}{x_1^3\cdot x_2^3}=\dfrac{(x_1+x_2)^3-3x_1x_2(x_1+x_2)}{\left(x_1 x_2\right)^3}
			=\dfrac{5^3-3\cdot 2\cdot 5}{2^3}=\dfrac{95}{8}.\]
			\item
			Ta có
			\[\sqrt{x_1}+\sqrt{x_2}=\sqrt{\left(\sqrt{x_1}+\sqrt{x_2}\right)^2}=
			\sqrt{x_1+x_2+2\sqrt{x_1x_2}}=\sqrt{5+2\sqrt{2}}.\]
			Suy ra,
			\[D=\dfrac{x_1\sqrt{x_1}+x_2\sqrt{x_2}}{\sqrt{x_1}\cdot \sqrt{x_2}}=
			\dfrac{\left(\sqrt{x_1}+\sqrt{x_2}\right)\left(x_1+x_2-\sqrt{x_1x_2}\right)}{\sqrt{x_1x_2}}
			=\dfrac{(5-\sqrt{2})\sqrt{5+2\sqrt{2}}}{\sqrt{2}}.\]		
		\end{enumEX}		
	}
\end{bt}

\begin{bt}
	Tìm hai số $x$ và $y$ biết
	\begin{enumEX}{2} 
		\item $x+y=18$ và $xy=77$.
		\item $x+y=-3$ và $xy=5$.
		\item $x-y=2\sqrt{3}$ và $xy=1$.
		\item $x^2+y^2=34$ và $xy=-15$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item 
			Vì $x+y=18$ và $xy=77$ nên $x,y$ là hai nghiệm của phương trình
			\[t^2-18t+77=0.\]
			Giải phương trình trên ta được $t=7,t=11$.\\
			Vậy hai số $x,y$ cần tìm là $(x;y)=(7;11)$, $(x;y)=(11;7)$.
			\item
			Vì $x+y=-3$ và $xy=5$ nên $x,y$ là hai nghiệm của phương trình
			\[t^2+3t+5=0.\]
			Phương trình trên có $\Delta = -11<0$ nên vô nghiệm.\\
			Vậy không tồn tại hai số $x,y$ thỏa đề bài.
			\item
			Vì $x+(-y)=2\sqrt{3}$ và $x\cdot (-y)=-1$ nên $x,-y$ là hai nghiệm của phương trình
			\[t^2-2\sqrt{3}t-1=0.\]
			Giải phương trình trên ta được các nghiệm $t=2-\sqrt{3}, t=2+\sqrt{3}$.\\
			Do đó, 
			\begin{center}
				$\heva{&x=2+\sqrt{3}\\&-y=-2+\sqrt{3}.}$ hoặc $\heva{&x=-2+\sqrt{3}\\&-y=2+\sqrt{3}.}$
			\end{center}
			Vậy hai số $x,y$ cần tìm là $(x;y)=\left(2+\sqrt{3};2-\sqrt{3}\right)$, $(x;y)=\left(-2+\sqrt{3};-2-\sqrt{3}\right)$.
			\item
			Ta có
			\[x^2+y^2=34 \,\Leftrightarrow\, (x+y)^2-2xy=34 \,\Leftrightarrow\, (x+y)^2=4 \,\Leftrightarrow\, \hoac{&x+y=2\\&x+y=-2.}\]
			$\bullet\;$ Với $x+y=2$, ta được hệ $\heva{&x+y=2\\&xy=-15.}$\\
			Suy ra, $x,y$ là hai nghiệm của phương trình
			$t^2-2t-15=0 \,\Leftrightarrow\, \hoac{&t=-3\\&t=5.}$\\
			Do đó, 
			\begin{center}
				$\heva{&x=-3\\&y=5}$ hoặc $\heva{&x=5\\&y=-3.}$
			\end{center}
			$\bullet\;$ Với $x+y=-2$, ta được hệ $\heva{&x+y=-2\\&xy=-15}$.\\
			Suy ra, $x,y$ là hai nghiệm của phương trình
			$t^2+2t-15=0 \,\Leftrightarrow\, \hoac{&t=3\\&t=-5.}$\\
			Do đó, 
			\begin{center}
				$\heva{&x=3\\&y=-5}$ hoặc $\heva{&x=-5\\&y=3.}$
			\end{center}\
			Vậy hai số $x,y$ cần tìm là $(x;y)=\left\lbrace(-3;5),(5;-3),(3;-5);(-5;3)\right\rbrace$.			
		\end{enumEX}		
	}
\end{bt}

\begin{bt}
	Cho phương trình $x^2-2(m+1)x+4m-1=0 \quad (1)$, với $m$ là tham số.
	\begin{enumEX}{1} 
		\item Tìm $m$ để phương trình có hai nghiệm $x_1,\ x_2$ thỏa mãn $x_1^2+x_2^2=10$.
		\item Tìm hệ thức liên hệ giữa hai nghiệm $x_1,\ x_2$ không phụ thuộc vào tham số $m$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item 
			Vì
			\[\Delta'=(m+1)^2-(4m-1)=m^2-2m+2=(m-1)^2+1 \geqslant 1, \forall m\in \mathbb{R}.\]
			nên phương trình luôn có hai nghiệm phân biệt $x_1, \ x_2$.\\
			Hệ thức Vi-ét:
			\[\heva{&x_1+x_2=2m+2\\&x_1x_2=4m-1.}\]
			Theo đề
			\begin{eqnarray*}
				x_1^2+x_2^2=10 &\Leftrightarrow& (x_1+x_2)^2-2x_1x_2=10\\
				&\Leftrightarrow& 4(m+1)^2-2(4m-1)=10\\
				&\Leftrightarrow& m^2=1 \;\Leftrightarrow\; m=\pm 1.
			\end{eqnarray*}
			Vậy $m=-1, m=1$ là giá trị cần tìm.
			\item
			Hệ thức Vi-ét: $\heva{&x_1+x_2=2m+2&&(1)\\&x_1x_2=4m-1.&&(2)}$\\
			Từ $(1)$, ta được $2m=x_1+x_2-2$. Thay vào $(2)$ ta được
			\[x_1x_2=2(x_1+x_2-2)-1 \,\Leftrightarrow\, 2x_1+2x_2-x_1x_2=5.\]
			Biểu thức $2x_1+2x_2-x_1x_2=5$ luôn đúng với mọi $m$. Vậy đây là biểu thức cần tìm.
		\end{enumEX}		
	}
\end{bt}

\begin{bt}

	}

\begin{bt} 
	Cho phương trình
	\[x^2+7x-2=0. \tag{1}\]  
	Hãy lập một phương trình bậc hai có các nghiệm là lũy thừa bậc $6$ của các nghiệm của phương trình $(1)$.
	\loigiai{		
		\allowbreak
		Vì phương trình $(1)$ có $ac=-1<0$ nên $(1)$ luôn có hai nghiệm trái dấu.\\
		Gọi $x_1, \ x_2$ là các nghiệm của phương trình đã cho, theo Vi-ét ta có
		\[x_1+x_2=-7, \  x_1x_2=-2.\]
		Gọi $y_1,y_2$ là các nghiệm của phương trình phải lập, khi đó giả sử
		\[y_1=x_1^6,\  y_2=x_2^6.\]
		Ta có
		\[x_1^2+x_2^2=(x_1+x_2)^2-2x_1x_2=7^2-2\cdot(-2)=53.\]
		Do đó,
		\[y_1+y_2=x_1^6+x_2^6=\left(x_1^2+x_2^2\right)^3-3x_1^2x_2^2\left(x_1^2+x_2^2\right)=53^2-3\cdot(-2)^2\cdot 53=2173.\]
		và
		\[y_1y_2=x_1^6x_2^6=\left(x_1+x_2\right)^6=(-2)^6=64.\]
		Phương trình phải lập có tổng hai nghiệm bằng $2173$ và tích hai nghiệm bằng $64$ nên có dạng là $$y^2-2173y+64=0.$$		
	}
\end{bt}

\begin{bt}
	Cho phương trình $x^2-mx+m-1=0 \quad (1)$, với $m$ là tham số.
	\begin{enumEX}{1} 
		\item Chứng minh rằng phương trình $(1)$ luôn có hai nghiệm $x_1,x_2$ với mọi giá trị $m$.
		\item Tìm giá trị lớn nhất và giá trị nhỏ nhất của biểu thức 
		$P=\dfrac{2x_1x_2+3}{x_1^2+x_2^2+2(x_1x_2+1)}$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item 
			Vì \[\Delta=m^2-4(m-1)=(m-2)^2 \geqslant 0, \forall m\in\mathbb{R}\]
			nên phương trình $(1)$ luôn có hai nghiệm $x_1, \ x_2$.			
			\item
			Hệ thức Vi-ét: \[\heva{&x_1+x_2=m\\&x_1x_2=m-1}\]
			Do đó,
			\[P=\dfrac{2x_1x_2+3}{(x_1+x_2)^2+2}=\dfrac{2m+1}{m^2+2}\]
			Ta có $$1-P=\dfrac{(m-1)^2}{m^2+2} \geqslant 0 \;\Rightarrow P \leqslant 1$$
			và $$P+\dfrac{1}{2}=\dfrac{(m+2)^2}{2(m^2+2)} \geqslant 0 \;\Rightarrow P\geqslant -\dfrac{1}{2}.$$
			Vậy giá trị nhỏ nhất của $P$ bằng $-\dfrac{1}{2}$ khi $m=-2$; giá trị lớn nhất của $P$ bằng $2$ khi $m=1$.				
		\end{enumEX}		
	}
\end{bt}

\begin{bt}

			\item \textbf{Điều kiện 2}. Với điều kiện $(1)$, theo hệ thức Vi-ét, ta có $\heva{&x_1+x_2=\dfrac{-2m+38}{m-16}\\&x_1x_2=\dfrac{m-14}{m-16}.}$\\

				& \Leftrightarrow &

\begin{bt}
	Cho phương trình $x^2-2x+2-m=0$, $m$ là tham số. Tìm các giá trị của $m$ để phương trình có hai nghiệm $x_1,\ x_2$ thỏa mãn hệ thức $2x_1^3+(m+2)x_2^2=5$.
	\loigiai{
		\begin{itemize}
			\item \textbf{Điều kiện 1}. Phương trình có hai nghiệm $x_1, \ x_2$ khi và chỉ khi
			\[\Delta'\geqslant 0
			\;\Leftrightarrow\; 
			m-1 \geqslant 0 \;\Leftrightarrow\; m\geqslant 1 \tag{1}.\]
			\item \textbf{Điều kiện 2}. Với điều kiện $(1)$, theo hệ thức Vi-ét, ta có $\heva{&x_1+x_2=2\\&x_1x_2=2-m.}$\\

				&\Leftrightarrow& m^2+8m-9=0 \;\Leftrightarrow\; \hoac{&m=1\\&m=-9.}			
			\end{eqnarray*}
			Đối chiếu với điều kiện $(1)$, ta được $m=1$ là giá trị cần tìm.		
		\end{itemize}
	}
\end{bt}
