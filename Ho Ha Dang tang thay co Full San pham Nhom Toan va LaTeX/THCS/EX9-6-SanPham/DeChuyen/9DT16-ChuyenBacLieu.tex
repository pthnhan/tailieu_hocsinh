\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Bạc Liêu, 2016]%[9D5G5]%[9D5G3]
	\hfill
	\begin{enumerate}[a)]
		\item Chứng minh tích của 4 số nguyên dương liên tiếp không thể là số chính phương.
		\item Tìm nghiệm nguyên của phương trình $x^2y^2-xy=x^2+3y^2$.
	\end{enumerate}
\loigiai{
	\begin{enumerate}[a)]
		
		\item Giả sử có 4 số nguyên dương liên tiếp là $n$, $n+1$, $n+2$, $n+3$ ($n\in {{\mathbb{N}}^{*}}$).\\	
		Ta có: $A=n(n+1)( n+2)(n+3)$$=(n^2+3n)(n^2+3n+2)$	
		$=(n^2+3n)^2+2(n^2+3n)$\\ 	
		$\Rightarrow (n^2+3n)^2<A<(n^2+3n+1)^2$.\\	
		Vậy $A$ không thể là số chính phương.
		\item Ta có $x^2y^2-xy=x^2+3y^2$$\Leftrightarrow (x^2-3)y^2-xy-x^2=0$. $(1)$\\	
		Vì $x^2-3\ne 0$ nên $(1)$ là phương trình  bậc hai đối với $y$, có $\Delta =x^2+4x^2(x^2-3)=x^2(4x^2-11)$.\\	
		Phương trình $(1)$ có nghiệm nguyên khi $\Delta $ là số chính phương.\\
		Nếu $x=0$ thì $y=0$. Nếu $x\ne 0$ thì $4x^2-11$ phải là số chính phương.\\	
		Đặt $4x^2-11=z^2$ ($z\in \mathbb{N}$) ta được:\\
		$4x^2-z^2=11$$\Leftrightarrow (2x-z)( 2x+z)=11$$\Leftrightarrow \heva{
		& 2x+z=11 \\
		& 2x-z=1}$ hoặc $\heva{
		& 2x+z=-1 \\
		& 2x-z=-11}$.\\	
		Suy ra $x=3$ hoặc $x=-3$.\\	
		Với $x=3$, ta được $2y^2-y-3=0\Leftrightarrow y=-1$ hoặc $y=\dfrac{3}{2}$. (loại)\\
		Với $x=-3$, ta được $2y^2+y-3=0\Leftrightarrow y=1$ hoặc $y=-\dfrac{3}{2}$. (loại)\\
		Vậy phương trình đã cho có các nghiệm nguyên $( x;y)$ là $( 0;0)$, $( 3;-1)$ và $( -3;1)$.		
	\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Bạc Liêu, 2016]%[9D1K8]%[9D3G2]
	\hfill
	\begin{enumerate}[a)]
		\item Tính giá trị của biểu thức $A=\dfrac{\sqrt{3}}{\sqrt{2}+\sqrt{2+\sqrt{3}}}-\dfrac{\sqrt{3}}{\sqrt{2}-\sqrt{2-\sqrt{3}}}$.
		\item Giải hệ phương trình: $\heva{
		& x^2+3xy+2y+2=0 \\
		& y^2-xy-3x-5y=0}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item Ta có $A=\dfrac{\sqrt{6}}{2+\sqrt{{{\left( \sqrt{3}+1 \right)}^2}}}-\dfrac{\sqrt{6}}{2-\sqrt{{{\left( \sqrt{3}-1 \right)}^{2}}}}$\\
$=\dfrac{\sqrt{6}}{3+\sqrt{3}}-\dfrac{\sqrt{6}}{3-\sqrt{3}}$$=\dfrac{\sqrt{2}}{\sqrt{3}+1}-\dfrac{\sqrt{2}}{\sqrt{3}-1}$$=-\sqrt{2}$	
			\item Ta có  $\heva{
			& x^2+3xy+2y+2=0(1) \\
			& y^2-xy-3x-5y=0(2)}$.\\
			Lấy $(1)$ cộng $(2)$ ta được $( x+y)^2-3( x+y)+2=0$$\Leftrightarrow x+y=1$ hoặc $x+y=2$.\\	
			* $x+y=1$$\Leftrightarrow y=1-x$, thay vào $(1)$ được $2x^2-x-4=0$$\Leftrightarrow x=\dfrac{1\pm \sqrt{33}}{4}$.\\	
			* $x+y=2$$\Leftrightarrow y=2-x$, thay vào $(1)$ được $x^2-2x-3=0$$\Leftrightarrow x=-1$ hoặc $x=3$.\\	
			Vậy hệ phương trình có bốn nghiệm $( -1;3)$, $( 3;-1)$, $\left( \dfrac{1+\sqrt{33}}{4};\dfrac{3-\sqrt{33}}{4} \right)$, $\left( \dfrac{1-\sqrt{33}}{4};\dfrac{3+\sqrt{33}}{4} \right)$.			
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Bạc Liêu, 2016]%[9D4K6]%[9D5G1]
	\hfill
	\begin{enumerate}[a)]
		\item Cho phương trình $x^2-2mx+m^2-m-6=0$ ($m$ là tham số). Xác định giá trị của $m$ để phương trình có hai nghiệm $x_1$, $x_2$ thỏa mãn $\dfrac{x_1}{x_2}+\dfrac{x_2}{x_1}=\dfrac{18}{7}$.
		\item Cho ba số thực dương $x$, $y$, $z$ thỏa mãn $x+y+z=2$. Chứng minh:
		$\dfrac{x^2}{y}+\dfrac{y^2}{z}+\dfrac{z^2}{x}\ge \dfrac{3}{2}( x^2+y^2+z^2)$.
		\end{enumerate}
	\loigiai
	{
		\begin{enumerate}[a)]
			\item Phương trình đã cho có hai nghiệm khi ${\Delta }'\ge 0\Leftrightarrow m\ge -6$.  (*)\\
			Khi đó, theo định lí Vi-ét, ta có: $x_1+x_2=2m$; $x_1x_2=m^2-m-6$.\\	
			Do đó, $\dfrac{x_1}{x_2}+\dfrac{x_2}{x_1}=\dfrac{18}{7}$$\Leftrightarrow \heva{
			& 7( x_1+x_2)^2-32x_1x_2=0 \\
			& x_1x_2\ne 0}$	
			$\Leftrightarrow \heva{
			& m^2-8m-48=0 \\
			& m^2-m-6\ne 0}$	
			$\Leftrightarrow \hoac{
			& m=-4 \\
			& m=12}$ (thỏa (*))\\	
			Vậy $m=-4$ và $m=12$ là các giá trị cần tìm.		
			\item Ta có: $\dfrac{x^2}{y}+\dfrac{y^2}{z}+\dfrac{z^2}{x}\ge \dfrac{3}{2}\left( x^2+y^2+z^2 \right)$  (1)
			$\Leftrightarrow \left( x+y+x \right)\left( \dfrac{x^2}{y}+\dfrac{y^2}{z}+\dfrac{z^2}{x} \right)\ge 3\left( x^2+y^2+y^2 \right)$	
			$\Leftrightarrow \dfrac{x^2(x+z)}{y}+\dfrac{y^2(y+x)}{z}+\dfrac{z^2(z+y)}x\ge 2( x^2+y^2+y^2)$.\\	
			Theo bất đẳng thức $AM-GM$, ta có: $\dfrac{x^2(x+z)}{y}+(x+z)y\ge 2x(x+z)$;
			$\dfrac{y^2(y+x)}{z}+(y+x)z\ge 2y(y+x)$; $\dfrac{z^2(z+y)}{x}+(z+y)x\ge 2z(z+y)$.\\
			Cộng vế theo vế ta được: $\dfrac{x^2(x+z)}{y}+\dfrac{y^2(y+x)}{z}+\dfrac{z^2(z+y)}{x}\ge 2\left( x^2+y^2+y^2 \right)$.\\
			Do đó, bất đẳng thức $(1)$ được chứng minh. Đẳng thức xảy ra khi $x=y=z=\dfrac{2}{3}$.					
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Bạc Liêu, 2016]%[9H3K6]%[9H3G6]
	
	Từ một điểm $M$ ở ngoài đường tròn $(O)$ vẽ hai tiếp tuyến $MA$, $MB$ và cát tuyến $MCD$ của đường tròn đó ($C$ nằm giữa $M$ và $D$, dây $CD$ không đi qua $O$). Gọi $E$ là trung điểm của dây $CD$. Chứng minh:
	\begin{enumerate}
		\item $\widehat{BAC}=\widehat{DAE}$.
		\item $AC.BD=BC.DA=\dfrac{1}{2}.AB.CD$.
	\end{enumerate}
	\loigiai{Ta có hình vẽ:

ymax=2.5]

\tkzGetPoint{E}

		\begin{enumerate}[a)]
		\item Vì $\widehat{OAM}=\widehat{OBM}=\widehat{OEM}={{90}^{0}}$ 	
		nên năm điểm $O$, $M$, $A$, $B$, $E$ cùng thuộc đường tròn đường kính $OM$,
		suy ra $\widehat{AEM}=\widehat{ABM}$ (cùng chắn $\overset\frown{AM}$).\\	
		Mà $\widehat{ADE}+\widehat{DAE}=\widehat{AEM}$, $\widehat{ABC}+\widehat{BAC}=\widehat{ABM}$	
		và $\widehat{ADE}=\widehat{ABC}$ nên suy ra $\widehat{BAC}=\widehat{DAE}$.
		
		\item Xét $\Delta ABC$ và $\Delta ADE$, có $\widehat{BAC}=\widehat{DAE}$ (chứng minh trên) và $\widehat{ADE}=\widehat{ABC}$ (chắn $\overset\frown{AC}$).\\
		Suy ra $\Delta ABC\backsim \Delta ADE$	
		$\Rightarrow \dfrac{AB}{AD}=\dfrac{BC}{DE}\Rightarrow AD.BC=AB.DE=\dfrac{1}{2}.AB.CD$.  (1)\\	
		Tương tự, chứng minh được: $\Delta ABD\backsim \Delta ACE$				$\Rightarrow \dfrac{AB}{AC}=\dfrac{BD}{CE}\Rightarrow AC.BD=AB.CE=\dfrac{1}{2}.AB.CD$.  (2)\\	
		Từ $(1)$ và $(2)$ suy ra điều phải chứng minh.
		\end{enumerate}
	}

\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Bạc Liêu, 2016]%[9H5G4]
	Cho điểm $M$ nằm trong tam giác $ABC$ có $BC=a$, $AC=b$, $AB=c$ và diện tích là $S$. Gọi khoảng cách từ $M$ đến các cạnh $BC$, $AC$, $AB$ lần lượt là $x$, $y$, $z$. Xác định vị trí của điểm $M$ trong tam giác $ABC$ sao cho biểu thức $Q=\dfrac{a}{x}+\dfrac{b}{y}+\dfrac{c}{z}$ đạt giá trị nhỏ.
	\loigiai{
Ta có hình vẽ:
    \begin{center}
    \begin{tikzpicture}[scale=1]
\tkzInit[xmin=-3,ymin=-0.5,xmax=4,
ymax=3.5]
\tkzClip
%% khai bao A B C thay dổi nhe
\tkzDefPoint(-2,0){A}\tkzDefPoint(2,0){B}\tkzDefPoint(-1,3){C}
\tkzDefCircle[in](A,B,C) \tkzGetPoint{I}
\tkzDefPointBy[projection= onto A--B](I)
\tkzGetPoint{M1}
\tkzDefPointBy[projection= onto A--C](I)
\tkzGetPoint{M2}
\tkzDefPointBy[projection= onto C--B](I)
\tkzGetPoint{M3}
\tkzDrawCircle[radius](I,M1)
\tkzLabelPoints[above](C)
\tkzLabelPoints[below](A,B)
\tkzLabelPoints[right](I)
\tkzDrawSegments(A,B A,C B,C I,M1 I,M2 I,M3)
\end{tikzpicture}
\end{center}
		Ta có $S=S_{MBC}+S_{MAB}+S_{MAC}=\dfrac{1}{2}\left( ax+by+cz \right)$	
		$\Rightarrow 2SQ=\left( ax+by+cz \right)\left( \dfrac{a}{x}+\dfrac{b}{y}+\dfrac{c}{z} \right)$	
		$=a^2+b^2+c^2+ab\left( \dfrac{x}{y}+\dfrac{y}{x} \right)+bc\left( \dfrac{y}{z}+\dfrac{z}{y} \right)+ca\left( \dfrac{x}{z}+\dfrac{z}{x} \right)$.\\ 	
		Mà $\dfrac{x}{y}+\dfrac{y}{x}\ge 2$;\\ $\dfrac{y}{z}+\dfrac{z}{y}\ge 2$;\\
 $\dfrac{x}{z}+\dfrac{z}{x}\ge 2$.\\	
		$\Rightarrow 2SQ\ge a^2+b^2+c^2+2ab+2bc+2ca=\left( a+b+c \right)^2$\\	
		$\Rightarrow Q\ge \dfrac{{{\left( a+b+c \right)}^{2}}}{2S}$.\\	
		Dấu bằng xảy ra $\Leftrightarrow x=y=z\Leftrightarrow M$là tâm đường tròn nội tiếp tam giác $ABC$.\\
		Vậy $Q$ nhỏ nhất khi $M$ là tâm đường tròn nội tiếp tam giác $ABC$.	
		
	}
\end{ex}
