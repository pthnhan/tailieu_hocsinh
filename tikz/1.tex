\begin{ec}
	Cho nửa đường tròn tâm $O$, đường kính $AB=2R$ . Trên nửa đường tròn lấy điểm $M$ sao cho $MB=R$. Tiếp tuyến tại $M$ của đường tròn cắt các tiếp tuyến $Ax$ và $By$ lần lượt tại $C$ và $D$ ($Ax$ và $By$ cùng thuộc một nửa mặt phẳng có bờ $AB$ chứa điểm $M$).
	\lietke{
		\item Chứng minh: Tam giác $COD$ vuông và $AC+BD=CD$
		\item Tính $OC$ theo $R$?
		\item $BC$ cắt đường tròn tại $F$ ($F$ khác $B$), đường thẳng qua $O$ vuông góc với $BC$ cắt $By$ tại $E$. Chứng minh $EF$ là tiếp tuyến của đường tròn $(O)$.

\begin{ec}
	Từ một điểm $A$ nằm ngoài đường tròn $(O)$, kẻ các tiếp tuyến $AB, AC$ với $(O)$ ($B$ và $C$ là 2 tiếp điểm).
	\begin{enumerate}[\quad a)]
		\item Chứng minh 4 điểm $A, B, O, C$ cùng thuộc 1 đường tròn và $AO\perp BC$.
		\item Trên cung nhỏ $BC$ của $(O)$, lấy điểm $M$ bất kì ($M\ne B, M\ne C, M\notin AO$). Tiếp tuyến tại $M$ cắt $AB, AC$ lần lượt tại $D, E$. Chứng minh chu vi $\Delta ADE$ bằng $2AB$.
		\item Đường thẳng vuông góc với $AO$ tại $O$ cắt $AB$ và $AC$ lần lượt tại $P$ và $Q$. Chứng minh $4PD\cdot QE=PQ^2$.
		\item $DE$ cắt $AO$ tại $N$. $BC$ cắt $OM$ tại $K$. Chứng minh $AM\parallel NK$.
	\end{enumerate}

		\path let
		\p1=($(#2)-(#1)$),\p2=($(#2)-(#3)$),
		\n1={veclen(\x1,\y1)},\n2={veclen(\x2,\y2)}
		in ($(#1)!scalar(\n1/(\n1+\n2))!(#3)$) coordinate (#4);
