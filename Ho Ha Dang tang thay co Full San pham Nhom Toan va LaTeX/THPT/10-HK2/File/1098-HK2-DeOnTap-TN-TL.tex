\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[Ôn tập HK2, Nguyễn Khắc Hưởng]%[0D4B5]%%[0D4K5]
Giải các bất phương trình sau:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\dfrac{2x}{2x^2-3x+1}\geq \dfrac{1}{x-2}$.
			\item $\sqrt{3x^2+x-4}\geq x+1$.
		\end{enumerate}
	\end{multicols}
\loigiai{
		\begin{enumerate}
			\item $\dfrac{2x}{2x^2-3x+1}\geq \dfrac{1}{x-2}\Leftrightarrow \dfrac{2x(x-2)-(2x^2-3x+1)}{(x-2)(2x^2-3x+1)}\geq 0\Leftrightarrow \dfrac{-x-1}{(x-2)(x-1)(2x-1)}\geq 0$ $\Leftrightarrow \hoac{&-1\leq x<\dfrac{1}{2}\\&1<x<2}$.\\
			Vậy bất phương trình có tập nghiệm là $S=\left[-1;\dfrac{1}{2}\right)\cup \left(1;2\right)$.
			\item $\sqrt{3x^2+x-4}\geq x+1\Leftrightarrow \hoac{&\heva{&x+1<0\\&3x^2+x-4\geq 0}\\&\heva{&x+1\geq 0\\&3x^2+x-4\geq (x+1)^2}}\Leftrightarrow \hoac{&\heva{&x<-1\\&(x-1)(3x+4)\geq 0}\\&\heva{&x\geq -1\\&2x^2-x-5\geq 0}}$\\
			$\Leftrightarrow \hoac{&x\leq -\dfrac{4}{3}\\&x\geq \dfrac{1+\sqrt{41}}{4}}$.\\
			Vậy bất phương trình có tập nghiệm $T=\left(-\infty;-\dfrac{4}{3}\right]\cup\left[\dfrac{1+\sqrt{41}}{4};+\infty\right)$.
		\end{enumerate}
	}
\end{bt}

	Cho $\sin x=\dfrac{3}{5}$ với $\dfrac{\pi}{2}<x<\pi$. Tính giá trị của biểu thức $P=\cos 2x-\dfrac{1}{2}\sin 2x$.
	\loigiai{ Ta có $\sin^2x+\cos^2x=1\Leftrightarrow \dfrac{9}{25}+\cos^2x=1\Leftrightarrow\cos^2x=\dfrac{16}{26}\Leftrightarrow\cos x=-\dfrac{4}{5}$ (do $\dfrac{\pi}{2}<x<\pi$ nên $\cos x<0$).\\
		$P=\cos 2x-\dfrac{1}{2}\sin 2x=2\cos^2x-1-\sin x\cos x=2\cdot \dfrac{16}{25}-1-\dfrac{3}{5}\cdot \left(-\dfrac{4}{5}\right)=\dfrac{14}{25}$.
