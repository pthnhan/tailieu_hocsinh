\begin{ex}%[Nguyễn Bình Nguyên-LT1]%[2D1B1-1]
	Giả  sử  hàm số $f(x)$  có đạo hàm trên $(a;b)$. Trong  các  phát biểu sau, phát biểu nào đúng?
	\choice
	{Nếu $f'(x) \le 0,\ \forall x \in (a;b)$ thì  hàm số  $y=f(x)$  nghịch biến trên $(a;b)$}
	{ \True Hàm số $y=f(x)$ nghịch biến trên $(a;b)$  khi  $f'(x) <0,\ \forall x \in (a;b)$}
	{ Nếu $f'(x) \ge 0, \ \forall x \in (a;b)$ thì  hàm số  $y=f(x)$  đồng biến trên $(a;b)$}
	{Hàm số $y=f(x)$ nghịch biến trên $(a;b)$  khi và  chỉ  khi $f'(x) \le 0, \ \forall x\in (a;b)$}
	\loigiai{
		\begin{itemize}
			\item Xét mệnh đề  \lq \lq Nếu $f'(x) \le 0\ \forall x \in (a;b)$ thì  hàm số  $y=f(x)$  nghịch biến trên $(a;b)$\rq \rq. Đây là mệnh đề sai do $y=C\Rightarrow y'=0,\ \forall x \in (a;b) $ hàm số không nghịch biến trên $(a;b)$.
			\item Xét mệnh đề  \lq \lq Nếu $f'(x) \ge 0, \ \forall x \in (a;b)$ thì  hàm số  $y=f(x)$  đồng biến trên $(a;b)$\rq \rq. Đây là mệnh đề sai do $y=C\Rightarrow y'=0,\ \forall x \in (a;b) $ hàm số không đồng biến trên $(a;b)$.
			\item Xét mệnh đề  \lq \lq Hàm số $y=f(x)$ nghịch biến trên $(a;b)$  khi và  chỉ  khi $f'(x) \le 0, \ \forall x\in (a;b)$\rq \rq. Đây là mệnh đề sai do: Xét hàm số $y=C\Rightarrow y'=0,\ \forall x \in (a;b)$ hàm số không nghịch biến trên $(a;b)$. 
			\item Mệnh đề \lq \lq Hàm số $y=f(x)$ nghịch biến trên $(a;b)$  khi  $f'(x) <0,\ \forall x \in (a;b)$\rq\rq  là mệnh đề đúng.
	\end{itemize}}
	
\end{ex}

\subsection{Cực trị hàm số}
\begin{ex}%[Vũ Nguyễn Hoàng Anh, Dự án LT1]%[2D1Y2-7]
	Cho hàm số $f(x)$ liên tục trên khoảng $\mathscr{\mathscr{D}}\subset\mathbb{R}$ và $x_0\in \mathscr{\mathscr{D}}$. Mệnh đề nào sau đây là \textbf{đúng}?
	\choice
	{$x_0$ là điểm cực đại của hàm số $f$ nếu $f(x)>f(x_0)$ với mọi $x\in \mathscr{D}$}
	{$x_0$ là điểm cực đại của hàm số $f$ nếu với mọi $(a;b)\subset \mathscr{D}$ chứa $x_0$ ta đều có $f(x)>f(x_0)$ với mọi $x\in (a;b)\setminus\{x_0\}$}
	{\True $x_0$ là cực đại của hàm số $f$ nếu tồn tại $(a,b)\subset \mathscr{D}$ chứa $x_0$ sao cho $f(x)<f(x_0)$ với mọi $x\in(a;b)\setminus\{x_0\}$}
	{$x_0$ là điểm cực đại của hàm số $f$ nếu $f(x)<f(x_0)$ với mọi $x\in (a;b)\subset \mathscr{D}$}
	\loigiai{
		Theo định nghĩa của điểm cực đại của hàm số thì mệnh đề \lq\lq $x_0$ là cực đại của hàm số $f$ nếu tồn tại $(a,b)\subset \mathscr{D}$ chứa $x_0$ sao cho $f(x)<f(x_0)$ với mọi $x\in(a;b)\setminus\{x_0\}$\rq\rq\ là mệnh đề đúng.
	}
\end{ex}

\begin{ex}%[Vũ Nguyễn Hoàng Anh, Dự án LT1]%[2D1Y2-7]
	Cho hàm số $f(x)$ có đạo hàm cấp hai trên $(a;b)$ và $x_0\in (a;b)$. Mệnh đề nào sau đây là đúng?
	\choice
	{Nếu $x_0$ là điểm cực trị của $f$ thì $f''(x_0)=0$}
	{Nếu $f'(x_0)=0$ thì $f$ đạt cực trị tại điểm $x_0$}
	{Nếu $f'(x_0)=0$ và $f''(x_0)>0$ thì hàm số $f$ đạt cực đại tại điểm $x_0$}
	{\True Nếu $f'(x)$ đổi dấu khi $x$ qua $x_0$ thì $f$ đạt cực trị tại điểm $x_0$}
	\loigiai{
		\begin{itemize}
			\item Mệnh đề \lq\lq Nếu $x_0$ là điểm cực trị của $f$ thì $f''(x_0)=0$\rq\rq\ sai. Nếu $x_0$ là điểm cực trị của hàm số $f$ thì $f'(x_0)=0$ và $f''(x_0)\ne 0$.
			\item Mệnh đề \lq\lq Nếu $f'(x_0)=0$ thì $f$ đạt cực trị tại điểm $x_0$\rq\rq\ sai. Ví dụ xét hàm số $f(x)=x^3$. Dễ thấy $f'(0)=0$ nhưng $x_0=0$ không là điểm cực trị của $f$.
			\item Mệnh đề \lq\lq Nếu $f'(x_0)=0$ và $f''(x_0)>0$ thì hàm số $f$ đạt cực đại tại điểm $x_0$\rq\rq\ sai. Nếu $f'(x_0)=0$ và $f''(x_0)>0$ thì $x_0$ là điểm cực tiểu của hàm số $f$.
			\item Mệnh đề \lq\lq Nếu $f'(x)$ đổi dấu khi $x$ qua $x_0$ thì $f$ đạt cực trị tại điểm $x_0$\rq\rq\ đúng, theo định lí SGK.
		\end{itemize}
	}
\end{ex}

\begin{ex}%[Vũ Nguyễn Hoàng Anh, Dự án LT1]%[2D1G2-7]
	Cho hàm số $f(x)$ có đạo hàm trên khoảng $(a;b)$, $x_0\in (a;b)$ và các khẳng định
	\begin{enumerate}[(i)]
		\item Nếu tồn tại $h>0$ sao cho $f'(x)<0$ với $x\in (x_0-h;x_0)$ và $f'(x)>0$ với $x\in (x_0;x_0+h)$ thì $x_0$ là điểm cực tiểu của hàm số.
		\item Nếu tồn tại $h>0$ sao cho $f'(x)>0$ với $x\in (x_0-h;x_0)$ và $f'(x)<0$ với $x\in (x_0;x_0+h)$ thì $x_0$ thì $x_0$ là điểm cực đại của hàm số.
		\item Nếu $x_0$ là điểm cực tiểu của hàm số $f$ thì tồn tại $h>0$ sao cho $f'(x)<0$ với $x\in (x_0-h;x_0)$ và $f'(x)>0$ với $x\in (x_0;x_0+h)$
		\item Nếu $x_0$ là điểm cực đại của hàm số $f$ thì tồn tại $h>0$ sao cho $f'(x)>0$ với $x\in (x_0-h;x_0)$ và $f'(x)<0$ với $x\in (x_0;x_0+h)$
	\end{enumerate}
	Số khẳng định \textbf{sai} là
	\choice
	{$0$}
	{$1$}
	{\True $2$}
	{$3$}
	\loigiai{
		Mệnh đề \lq\lq Nếu $x_0$ là điểm cực tiểu của hàm số $f$ thì tồn tại $h>0$ sao cho $f'(x)<0$ với $x\in (x_0-h;x_0)$ và $f'(x)>0$ với $x\in (x_0;x_0+h)$\rq\rq\ sai. Thật vậy, xét hàm số
		$$f(x)=\begin{cases} x^2\left(1+\sin\dfrac{1}{x}\right) &\text{nếu}\ x\in \left( -\dfrac{2}{\pi};\dfrac{2}{\pi} \right)\setminus\{0\}\\ 0 &\text{nếu}\ x=0.\end{cases}$$
		Dễ thấy $f(x)\ge 0$ với mọi $x\in \left( -\dfrac{2}{\pi};\dfrac{2}{\pi} \right)$ và $f(x)=0$ khi và chỉ khi $x=0$. Do đó $x_0=0$ là điểm cực tiểu của hàm số $f$. Ta có
		\begin{align*}
		\dfrac{\left|x^2\left( 1+\sin \dfrac{1}{x} \right)\right|}{|x|}=\left| x\left( 1+\sin \dfrac{1}{x}\right) \right|\le 2|x|\longrightarrow 0\ \text{khi}\ x\to 0.
		\end{align*}
		Do đó
		$$f'(x)=\begin{cases} 2x\left( 1+\sin \dfrac{1}{x} \right)-\cos \dfrac{1}{x} &\text{nếu}\ x\ne 0\\ 0 &\text{nếu}\ x=0.\end{cases}$$
		Bây giờ giả sử tồn tại $h>0$ sao cho $f'(x)<0$ với $x\in (-h;0)$ và $f'(x)>0$ với $x\in (0;h)$. Với số nguyên $k$ đủ lớn thì $x_1=-\dfrac{2}{\pi+k4\pi}\in (-h;0)$ và $x_2=\dfrac{2}{\pi+k4\pi}\in (0;h)$, tuy nhiên $f'(x_1)=f'(x_2)=0$. Ta gặp mâu thuẫn. Vậy không tồn tại $h>0$ thỏa mãn yêu cầu.\\
		Hoàn toàn tương tự, bằng cách xét hàm $g(x)=-f(x)$, ta cũng chỉ ra được mệnh đề \lq\lq Nếu $x_0$ là điểm cực đại của hàm số $f$ thì tồn tại $h>0$ sao cho $f'(x)>0$ với $x\in (x_0-h;x_0)$ và $f'(x)<0$ với $x\in (x_0;x_0+h)$\rq\rq\ sai.
	}
\end{ex}

		\end{itemize}
	}
\end{ex}

\subsection{Giá trị lớn nhất, giá trị nhỏ nhất của hàm số}
\begin{ex}%[Võ Thị Quỳnh Trang, dự án LT1]%[2D1K3-2]
	Cho các mệnh đề sau:
	\begin{enumerate}
		\item Cho hàm số $y=f(x)$ xác định trên $\mathbb{R}$, giá trị lớn nhất của $f(x)$ trong $(-1;2)$ là $M$ khi và chỉ khi $ \heva{f(x) \le M \hspace{0.3cm} \forall x \in (-1;2) \\ \exists x_0 \in (-1;2); f(x_0)=M}.$
		\item Nếu hàm số $f(x)$ đạt giá trị lớn nhất $M$ trong khoảng $K$ thì có thể tổn tại nhiều giá trị của $x_0 \in K$ sao cho $f(x_0)=M.$
		\item $\mathop {\min }\limits_{K} \,\,f(x) = m \Leftrightarrow f(x) \ge m \hspace{0.3cm} \forall x \in K.$
		\item Có thể tồn tại nhiều giá trị nhỏ nhất của $f(x)$ trên khoảng $K$.  
	\end{enumerate}
	Trong các mệnh đề trên, có bao nhiêu mệnh đề đúng?

\subsection{Đồ thị hàm số và bài toán liên quan}

\begin{ex}%[Nguyễn Thế Út-LT1]%[2D1B5-6]
	Cho hàm số $y=\dfrac{ax+b}{cx+d}$, $(ad-bc\neq 0)$ có đồ thị $(C)$ và điểm $I$ là giao điểm hai đường tiệm cận. Hỏi qua $I$ có thể kẻ được bao nhiêu đường thẳng tiếp xúc với $(C)$?	
	\choice
	{\True $0$}
	{$1$}
	{$2$}
	{$1$ hoặc $2$}
	\loigiai{
		Tập xác định của hàm số $\mathscr{D}=\mathbb{R}\setminus \left\lbrace-\dfrac{c}{d} \right\rbrace $.\\
		Giao điểm của hai đường tiệm cận là $I\left(-\dfrac{d}{c};\dfrac{a}{c}\right)$. Gọi $M(x_0;y_0)$ là điểm bất kì trên $(C)$.\\ Đường thẳng tiếp xúc với $(C)$ tại $M$ có dạng \[d\colon y=\dfrac{ad-bc}{(cx_0+d)^2}(x-x_0)+\dfrac{ax_0+b}{cx_0+d}.\]
		Đường thẳng $d$ đi qua $I\left(-\dfrac{d}{c};\dfrac{a}{c}\right)$ khi và chỉ khi \[ \dfrac{a}{c}=\dfrac{ad-bc}{(cx_0+d)^2}(-\dfrac{d}{c}-x_0)+\dfrac{ax_0+b}{cx_0+d}.\tag{1}\] Phương trình $(1)$ vô nghiệm nên không tồn tại tiếp tuyến nào của $(C)$ đi qua $I$. 
	}
\end{ex}

\begin{ex}%[Nguyễn Thế Út-LT1]%[2D1B5-6]
	Cho hàm số $y=ax^3+bx^2+cx+d$, $(a\neq 0)$ có đồ thị $(C)$ và $I$ là điểm uốn. Hỏi qua $I$ có thể kẻ được bao nhiêu đường thẳng tiếp xúc với $(C)$?
	\choice
	{$0$}
	{\True $1$}
	{$2$}
	{$3$}
	\loigiai{
		Gọi $M\left(x_0;f(x_0)\right)$ là điểm trên $(C)$.\\ Đường thẳng qua $M$ có dạng $$d\colon y=k(x-x_0)+f(x_0).$$ $d$ tiếp xúc với $(C)$ thì hệ sau có nghiệm, số nghiệm $ x_0 $ của hệ phương trình chính là số tiếp điểm.
		\begin{eqnarray*}
			& & \heva{&k=3ax^2+2bx+c\\&k(x-x_0)+ax_0^3+bx_0^2+cx_0+d=ax^3+bx^2+cx+d}\\
			&\Leftrightarrow & \heva{&k=3ax^2+2bx+c\\&(x-x_0)^2\left(x-\dfrac{-b-ax_0}{2a}\right)=0}\\
			&\Leftrightarrow & \heva{&k=3ax_0^2+2bx_0+c\\&\hoac{&x=x_0\\&x=\dfrac{-b-ax_0}{2a}.}}
		\end{eqnarray*}
		Do đó, nếu $x_0=\dfrac{-b-ax_0}{2a}\Leftrightarrow x_0=-\dfrac{b}{3a}$ thì $I$ kẻ được $1$ tiếp tuyến với $(C)$. Ngược lại $ x_0 \neq-\dfrac{b}{3a}$ thì từ điểm $M\left(x_0;f(x_0)\right)$ kẻ được $2$ tiếp tuyến với $(C)$.
	}
\end{ex}

		\item Hàm số không có cực trị khi và chỉ khi $y'=0$ vô nghiệm.
		\item Đồ thị hàm số không có tiệm cận.

		\draw [] plot(\x,{(--3.-0.*\x)/6.});
		\fill (0cm,0cm) circle (1.5pt);
		\end{tikzpicture}
	}
	\loigiai{
		Ta có $y(0) < 0 \Rightarrow \dfrac{b}{d} < 0 \Rightarrow bd < 0$. Do đó $b$, $d$ trái dấu. \hfill $(1)$\\
		Lại có $y\left(-\dfrac{b}{a}\right) = 0$ nên từ đồ thị ta thấy hàm số có một nghiệm duy nhất $\Rightarrow -\dfrac{b}{a} > 0 \Rightarrow ab < 0$. Do đó $a$, $b$ trái dấu. \hfill $(2)$\\
		Từ $(1)$ và $(2)$ suy ra $a$ và $d$ cùng dấu nên $ad > 0$.
	}
\end{ex}
