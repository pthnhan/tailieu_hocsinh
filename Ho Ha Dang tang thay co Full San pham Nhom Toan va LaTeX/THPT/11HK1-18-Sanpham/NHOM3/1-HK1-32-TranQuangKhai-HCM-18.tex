\begin{bt}%[HK1-32-TRANQUANGKHAI-HOCHIMINH-2018]%[Lê Văn Thiện]%[1D1Y2-1]
	Giải phương trình lượng giác sau $2\cos x-\sqrt{3}=0$.
	\loigiai{
		Ta có $2\cos x-\sqrt{3}=0\Leftrightarrow \cos x = \dfrac{\sqrt{3}}{2}= \cos\dfrac{\pi}{6}\Leftrightarrow  \hoac{&x=\dfrac{\pi}{6}+k2\pi\\&x=-\dfrac{\pi}{6}+k2\pi}, k\in \mathbb{Z}$.\\
		Vậy tập nghiệm của phương trình là $S=\left\{\dfrac{\pi}{6}+k2\pi,-\dfrac{\pi}{6}+k2\pi\right\}$ với $k\in \mathbb{Z}$.
	}
\end{bt}

\begin{bt}%[HK1-32-TRANQUANGKHAI-HOCHIMINH-2018]%[Lê Văn Thiện]%[1D1B3-1]
	Giải phương trình lượng giác sau $\cos2x + 5\sin x -3=0$.
	\loigiai{
	\begin{eqnarray*}
		&&\cos2x + 5\sin x -3=0\\
		&\Leftrightarrow& 1 - 2\sin^2x +5\sin x - 3=0\\
		&\Leftrightarrow& -2\sin^2x +5\sin x - 2=0\\
		&\Leftrightarrow& \hoac{&\sin x=\dfrac{1}{2}\\&\sin x=2 \text{ (loại vì $-1\le \sin x\le 1$)}}
	\end{eqnarray*}
	Với $\sin x = \dfrac{1}{2}=\sin \dfrac{\pi}{6} \Leftrightarrow  \hoac{&x=\dfrac{\pi}{6}+k2\pi\\&x=\dfrac{5\pi}{6}+k2\pi}, k\in \mathbb{Z}$.\\
	Vậy tập nghiệm của phương trình là $S=\left\{\dfrac{\pi}{6}+k2\pi,\dfrac{5\pi}{6}+k2\pi\right\}$ với $k\in \mathbb{Z}$.
	}
\end{bt}

\begin{bt}%[HK1-32-TRANQUANGKHAI-HOCHIMINH-2018]%[Lê Văn Thiện]%[1D1B3-5]
	Giải phương trình lượng giác sau $\cos 2x + \sin 2x + \cos x-\sin x -1=0$.
	\loigiai{
	Ta có 
	\begin{eqnarray*}
		&&\cos 2x + \sin 2x + \cos x-\sin x -1=0\\
		&\Leftrightarrow& -2\sin^2 x+2\sin x\cos x+\cos x-\sin x=0\\
		&\Leftrightarrow& 2\sin x(\cos x-\sin x)+\cos x-\sin x=0\\
		&\Leftrightarrow& (\cos x -\sin x)(2\sin x+1)=0\\
		&\Leftrightarrow& \hoac{&\cos x -\sin x=0\\& 2\sin x+1=0}\\
		&\Leftrightarrow& \hoac{&\cos x =\sin x=\cos\left(\dfrac{\pi}{2}-x\right)\\& \sin x=\dfrac{1}{2}=\sin\dfrac{\pi}{6}}\\
		&\Leftrightarrow& \hoac{&x=\dfrac{\pi}{4}+k\pi\\&x=\dfrac{\pi}{6}+k2\pi\\&x=\dfrac{5\pi}{6}+k2\pi}, k\in \mathbb{Z}.
	\end{eqnarray*}
	Vậy tập nghiệm của phương trình là $S=\left\{\dfrac{\pi}{4}+k\pi, \dfrac{\pi}{6}+k2\pi, \dfrac{5\pi}{6}+k2\pi\right\}$, với $k\in \mathbb{Z}$.
	}
\end{bt}

\begin{bt}%[HK1-32-TRANQUANGKHAI-HOCHIMINH-2018]%[Lê Văn Thiện]%[1D2K3-2]
	Tìm hệ số của số hạng chứa biến $x^{48}$ trong khai triển nhị thức sau $\left(2x^3-\dfrac{x}{2}\right)^{24}$.
	\loigiai{
		Ta có
		\begin{eqnarray*}
			\left(2x^3-\dfrac{x}{2}\right)^{24}&=& \sum\limits_{k=0}^{24} \mathrm{C}^k_{24} \left(2x^3\right)^k \left(-\dfrac{x}{2}\right)^{24-k}\\
			&=& \sum\limits_{k=0}^{24} \mathrm{C}^k_{24}\cdot(-1)^{24-k}\cdot 2^{2k-24}\cdot x^{24+2k}.
		\end{eqnarray*}
	Điều kiện để  có số hạng chứa biến $x^{48}$ là $24+2k=48\Leftrightarrow k=12$.\\
	Vậy số hạng chứa biến $x^{48}$ là $(-1)^{24-12}\cdot 2^{2\cdot 12-24}\cdot x^{48}=x^{48}$.
	}
\end{bt}

\begin{bt}%[HK1-32-TRANQUANGKHAI-HOCHIMINH-2018]%[Lê Văn Thiện]%[1D2B5-2]
	Lớp $11A$ có $10$ học sinh giỏi, gồm $5$ học sinh giỏi Toán, $3$ học sinh giỏi Hóa và $2$ học sinh giỏi Lý.
	\begin{enumerate}
		\item Có bao nhiêu cách chọn ra $4$ học sinh, sao cho có đúng $2$ học sinh giỏi Toán.
		\item Tính xác suất để trong $4$ học sinh chọn ra có học sinh giỏi của cả $3$ môn.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Chọn ra $4$ học sinh, trong đó có đúng $2$ học sinh giỏi Toán, vậy còn lại $2$ bạn giỏi các môn khác. Khi đó số cách chọn là $\mathrm{C}^2_{5}\cdot \mathrm{C}^2_{5}=100$.
		\item Chọn ngẫu nhiên $4$ bạn bất kỳ, không gian mẫu $|\Omega| = \mathrm{C}^4_{10}=210$.\\
		Gọi $A$là biến cố có học sinh giỏi của cả 3 môn.
		Khi đó ta có 3 trường hợp cho biến cố $A$ là \\
		$\bullet$ Trường hợp 1 - Có $2$ học sinh giỏi Toán; $1$ học sinh giỏi Hóa và $1$ học sinh giỏi Lý, \\suy ra có $\mathrm{C}^2_{5}\cdot \mathrm{C}^1_{3}\cdot \mathrm{C}^1_{2}=60$ cách chọn.\\
		$\bullet$ Trường hợp 2 - Có $1$ học sinh giỏi Toán; $2$ học sinh giỏi Hóa và $1$ học sinh giỏi Lý, \\suy ra có $\mathrm{C}^1_{5}\cdot \mathrm{C}^2_{3}\cdot \mathrm{C}^1_{2}=30$ cách chọn.\\
		$\bullet$ Trường hợp 3 - Có $1$ học sinh giỏi Toán; $1$ học sinh giỏi Hóa và $2$ học sinh giỏi Lý, \\suy ra có $\mathrm{C}^1_{5}\cdot \mathrm{C}^1_{3}\cdot \mathrm{C}^2_{2}=15$ cách chọn.\\
		Do đó số cách chọn cho biến cố $A$ là $|A|=60+30+15=105$.\\
		Vậy xác suất để trong $4$ học sinh chon ra có học sinh giỏi của cả $3$ môn là $$\mathrm{P}(A)=\dfrac{105}{210}=0,875.$$
	\end{enumerate}
	}
\end{bt}

\begin{bt}%[HK1-32-TRANQUANGKHAI-HOCHIMINH-2018]%[Lê Văn Thiện]%[1D3B3-3]
	Tìm số hạng đầu $u_1$ và công sai $d$ của cấp số cộng $(u_n)$ biết $\heva{&u_3+u_7=22\\&u_9-u_2=28}$.
	\loigiai{
	Ta có \\
	$$\heva{&u_3+u_7=22\\&u_9-u_2=28}\Leftrightarrow \heva{&u_1+2d+u_1+6d=22\\&u_1+8d-(u_1+d)=28} \Leftrightarrow \heva{&2u_1+8d=22\\&7d=28}\Leftrightarrow\heva{&u_1=-5\\&d=4}.$$
	Vậy số hạng đầu $u_1=-5$ và công sai $d=4$.
	}
\end{bt}

\begin{bt}%[HK1-32-TRANQUANGKHAI-HOCHIMINH-2018]%[Lê Văn Thiện]%[1H2B3-2]
	Cho hình chóp $S.ABCD$, đáy $ABCD$ là hình bình hành tâm $O$. Gọi $M,N$ lần lượt là trung điểm của $SB,SD$.
	\begin{enumerate}
		\item Chứng minh rằng $ON$ song song với mặt phẳng $(SAB)$.
		\item Tìm giao tuyến hai mặt phẳng $(CMN)$ và $(ABCD)$.
		\item Tìm điểm $K$ là giao điểm của đường thẳng $SA$ và mặt phẳng $(CMN)$.
	\end{enumerate}

	\loigiai{
	\begin{center}
		\begin{tikzpicture}[scale=1.5]
		\tkzInit[ymin=-0.5,ymax=7,xmin=-0.5,xmax=7.9]
		\tkzClip
		\tkzDefPoints{0/0/D,5/0/C,2.2/2/A,7.2/2/B, 3.8/6/S}
		
		
		\tkzDefMidPoint(S,B)\tkzGetPoint{M}
		\tkzDefMidPoint(S,D)\tkzGetPoint{N}
		
		\tkzInterLL(A,C)(B,D) \tkzGetPoint{O}
		\tkzInterLL(M,N)(S,O) \tkzGetPoint{I}
		\tkzInterLL(C,I)(S,A) \tkzGetPoint{K}
		
		
		\tkzLabelPoints[above, right](S,M,K)
		\tkzLabelPoints[above left](N,A)
		\tkzLabelPoints[below right](B,C)
		\tkzLabelPoints[below](D,O)
		\tkzLabelPoints[below left](I)
		
		\tkzDefLine[parallel=through C](B,D)
		\tkzDrawLine[add = .5 and -.6](C,tkzPointResult)
		\draw (7.2,1.0) node[below]{$x$};
		\tkzDrawSegments[dashed](A,B D,A A,S A,C B,D O,N M,N S,O C,K)
		\tkzDrawSegments(C,B D,C S,D S,B S,C C,N C,M)
		
			
		
		\foreach \X in {A,B,C,D,S,M,N,O,I,K}
		{\draw[fill=black] (\X) circle (1.0pt);}
		\end{tikzpicture}
	\end{center}

	\begin{enumerate}
		\item Xét tam giác $SBD$ có $N$ và $O$ lần lượt là trung điểm của $SD$ và $BD$, \\ 
		do đó $ON\parallel AB$ (tính chất đường trung bình).\\
		Mà $SB\subset (SAB)$ nên $ON$ song song với mặt phẳng $(SAB)$.
		\item Xét tam giác $SBD$ có $N$ và $M$ lần lượt là trung điểm của $SD$ và $SB$,\\
		do đó $MN\parallel BD$ (tính chất đường trung bình).\\
		Khi đó $\heva{&C\in (ABCD)\cap (CMN)\\&MN\parallel BD\\&MN\subset (CMN)\\&BD\subset (ABCD)}$.\\
		Vậy $(ABCD)\cap (CMN)=Cx$, sao cho $Cx\parallel MN\parallel BD$.
		\item Trong mặt phẳng $(SBD)$, gọi $I$ là giao điểm của $SO$ và $MN$.\\
		Trong mặt phẳng $(SAC)$, gọi $K$ là giao điểm của $CI$ và $SA$.\\
		Khi đó $\heva{&K\in SA\\&K\in CI\subset (CMN)}$.\\
		Vậy $K$ là giao điểm của đường thẳng $SA$ và mặt phẳng $(CMN)$.
	\end{enumerate}
	}
\end{bt}
