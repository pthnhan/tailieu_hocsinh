\section{HỆ SỐ GÓC CỦA ĐƯỜNG THẲNG $Y=AX+B\ (A \ne 0)$}
\subsection{Nền tảng kiến thức}
\begin{enumEX}[1.]{1}
	\item Đường thẳng $y=ax+b$ có hệ số góc là $a$. 
	\item Gọi $\alpha$ là góc tạo bởi đường thẳng $y=ax+b$ với tia $Ox$\\
		\hspace{2cm} a) Nếu $\alpha < 90^\circ$ thì $a>0$;\\
		\hspace{2cm} b) Nếu $\alpha > 90^\circ$ thì $a<0$.\\
	Các đường thẳng có cùng hệ số $a$ ($a$ là hệ số của $x$) thì tạo với $Ox$ các góc bằng nhau.
\end{enumEX}
\subsection{Các dạng toán}
	\begin{dang}{XÁC ĐỊNH HỆ SỐ GÓC CỦA ĐƯỜNG THẲNG}
		\textbf{\textit{Phương pháp giải:}}
		\begin{enumEX}[1.]{1}
			\item Xác định hàm số $y=ax+b$.
			\item Hai đường thẳng song song với nhau thì có hệ số góc bằng nhau.
			\item $\tan \alpha = \dfrac{\text{Cạnh kề}}{\text{Cạnh đối}}$.
		\end{enumEX}
	\end{dang}
\begin{vd}
	Cho hàm số bậc nhất $y=ax+3$. Xác định hệ số góc $a$ biết rằng đồ thị của hàm số đi qua điểm $M(2;6)$.
	\loigiai{
	Đồ thị của hàm số đã cho đi qua điểm $M(2;6)$ nên ta có $6 = 2 \cdot a + 3 \Leftrightarrow 2 \cdot a = 3 \Leftrightarrow a = \dfrac{3}{2}$.	
}
\end{vd}
\begin{vd}
	Cho hàm số bậc nhất $y=ax+b \ (a \ne 0)$. Xác định hệ số góc $a$ biết rằng độ thị hàm số đi qua hai điểm $A(0;1)$, $B(2;5)$.
	\loigiai{
	Vì đồ thị hàm số đi qua hai điểm $A(0;1)$, $B(2;5)$ nên ta có hệ $\heva{&1=0 \cdot a + b \\&5 = 2 \cdot a + b} \Leftrightarrow \heva{&a=2 \\ &b=1.}$ \\
	Vậy hệ số góc $a=2$.
}
\end{vd}

\begin{vd}
	Tính hệ số góc của đường thẳng $d \colon (m-2)x + 3$, biết $d$ song song với đường thẳng $d'\colon 2x-y-1=0$.
	\loigiai{
	Đường thẳng $d'$ có phương trình $2x-y-1=0$ hay $y=2x-1$. Hệ số góc của $d'$ là $k'=2$. \\
	Ta có $d \parallel d' \Leftrightarrow \heva{&m-2=2 \\&3 \ne -1} \Leftrightarrow m=4$. \\
	Vậy đường thẳng $d$ có hệ số góc là $k=2$.	
}
\end{vd}

\end{dang}

\begin{dang}{XÁC ĐỊNH ĐƯỜNG THẲNG DỰA VÀO HỆ SỐ GÓC}
	\textbf{\textit{Phương pháp giải}}
	\begin{enumerate} 
		\item [$\bullet$] Gọi phương trình đường thẳng cần tìm là $y=ax+b$. Ta cần xác định $a$ và $b$.
		\item [$\bullet$] Chú ý rằng: Gọi $\alpha$ là góc nhọn tạo bởi đường thẳng $y=ax+b \ (a \ne 0)$ với trục $Ox$, ta có \\
		- Khi góc $\alpha$ nhọn thì $a = \tan \alpha$.\\
		- Khi góc $\alpha$ tù thì $a = - \tan \left(180^\circ - \alpha \right)$.
	\end{enumerate}
\end{dang}
\begin{vd}
	Xác định phương trình đường thẳng $d$ biết $d$ đi qua điểm $A(-2;3)$ và có hệ số góc bằng $-2$.
	\loigiai{
	Gọi phương tình đường thẳng $d$ là $y = ax + b$.\\
	Vì $d$ có hệ số góc là $-2$ nên $a=-2$, suy ra phương trình $d$ có dạng $y = -2x + b$.
	Lại có $d$ đi qua $A(-2;3)$ nên có phương trình $3 = (-2) \cdot (-2) + b \Leftrightarrow b = -1$.\\
	Vậy phương trình đường thẳng $d$ là $y=-2x-1$.	
}
\end{vd}
\subsection{Luyện tập}
\begin{bt}
	Cho hàm số bậc nhất $y=ax-3$ $(1)$. Hãy xác định hệ số $a$ biết đồ thị hàm số $(1)$ cắt đường thẳng $y=2x-1$ tại điểm có hoành độ bằng $2$.
	\loigiai{
		Gọi $N\left(x_N;y_N \right)$ là giao điểm. Theo đề ta có $x_N=2 \Rightarrow y_N=3$. Vậy $N(2;3)$.\\
		Đồ thị hàm số $(1)$ đi qua điểm $N(2;3)$ nên ta có $3 = a \cdot 2 - 3 \Leftrightarrow a = 3$.
	}
\end{bt}
\begin{bt}
	Xác định phương trình đường thẳng $d$ biết $d$ đi qua điểm $A(0;1)$ và tạo với đường thẳng $y=2$ một góc $60^\circ$.
	\loigiai{
	Phương trình đường thẳng $d$ có dạng $y = ax + b$.  \\
	Vì $d$ đi qua $A(0;1)$ nên ta có $ 1 = a \cdot 0 + b \Rightarrow b = 1$.\\
	Vì đường thẳng $y=2$ song song với trục hoành nên từ đề bài ta có $d$ tạo với trục $Ox$ một góc $60^\circ$. Ta có $a = \tan \alpha = \tan 60^\circ = \sqrt{3}$.\\
	Vậy phương trình đường thẳng $d \colon y = \sqrt{3}x + 1$.	
}
\end{bt}
\begin{bt}
	Cho đường thẳng $d \colon y = mx + 3$. Tính góc tạo bởi $d$ với trục $Ox$. Biết $d$ đi qua điểm $B\left(6;-3\right)$.
	\loigiai{
	Vì đường thẳng $d$ đi qua điểm $B(6;-3)$ nên ta có phương trình $-3 = 6 \cdot m + 3 \Leftrightarrow m =-1$.\\
	Gọi $\alpha$ là góc tạo bởi đường thẳng $d$ và trục $Ox$. \\
	Vì $m = -1$ nên $\tan \alpha = -1 \Leftrightarrow \alpha = 135^\circ$.	
}
\end{bt}
\subsection{Thử Thách}
\begin{bt}
	Tìm hệ số góc của đường thẳng đi qua gốc tọa độ $O$ và điểm $M(1;2)$.
	\loigiai{
		Gọi $y=ax+b$ là phương trình đường thẳng $d$ cần tìm. \\
		Vì đường thẳng $d$ đi qua gốc tọa độ $O$ và điểm $M(1;2)$ nên ta có hệ $\heva{&b=0 \\&2 = a \cdot 1} \Leftrightarrow \heva{&a=2 \\&b=1.}$\\
		Vậy hệ số góc của đường thẳng đã cho là $a=2$.
	}	
\end{bt}
\begin{bt}
	Tìm hệ số góc của đường thẳng $d$ đi qua hai điểm $A(1;2)$ và $B(3;4)$.
	\loigiai{
	Gọi $y=ax+b$ là phương trình của đường thẳng $d$ cần tìm.\\
	Vì đường thẳng $d$ đi qua hai điểm $A(1;2)$ và $B(3;4)$ nên ta có hệ $\heva{&a+b=2 \\&3a+b = 4} \Leftrightarrow \heva{&a=1 \\&b=1.}$ \\
	Đường thẳng $d \colon y = x + 1$ nên có hệ số góc $a = 1$.
	
}
\end{bt}
\begin{bt}
	Xác định đường thẳng $d$ đi qua điểm $A(0;3)$ và tạo với đường thẳng $y=3$ một góc $60^\circ$.
	\loigiai{
	Phương trình đường thẳng $d$ có dạng $y = ax + b$.  \\
	Vì $d$ đi qua $A(0;3)$ nên ta có $ 3 = a \cdot 0 + b \Rightarrow b = 3$.\\
	Vì đường thẳng $y=4$ song song với trục hoành nên từ đề bài ta có $d$ tạo với trục $Ox$ một góc $60^\circ$. Ta có $a = \tan \alpha = \tan 60^\circ = \sqrt{3}$.\\
	Vậy phương trình đường thẳng $d \colon y = \sqrt{3}x + 3$.	
	}
\end{bt}
