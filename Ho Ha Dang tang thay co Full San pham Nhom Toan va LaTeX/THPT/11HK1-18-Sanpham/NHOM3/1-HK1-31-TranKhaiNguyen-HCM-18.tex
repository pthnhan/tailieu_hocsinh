\begin{bt}%[HK1, Toán 11, Trần Khai Nguyên, HCM 2017]%[Nguyễn Thị Kiều Ngân - Dự án 11HK1-18]%[1D1B2-1]
	Giải phương trình $\sin \left( 2x+\dfrac{\pi}{3} \right) -\cos \left( x-\dfrac{\pi}{4} \right) =0$.
	\loigiai{
		 Ta có
		 \begin{eqnarray*}
		 & & \sin \left( 2x+\dfrac{\pi}{3} \right) -\cos \left( x-\dfrac{\pi}{4} \right) =0
		 \Leftrightarrow \sin \left( 2x+\dfrac{\pi}{3} \right) =\cos \left( x-\dfrac{\pi}{4} \right)\\
		 & \Leftrightarrow & \sin \left( 2x+\dfrac{\pi}{3} \right) =\sin \left(\dfrac{\pi}{2} -x+\dfrac{\pi}{4} \right)
		 \Leftrightarrow \sin \left( 2x+\dfrac{\pi}{3} \right) =\sin \left(\dfrac{3\pi}{4} -x \right)\\
		 & \Leftrightarrow & \hoac{&2x+\dfrac{\pi}{3}=\dfrac{3\pi}{4} -x +k2\pi\\&2x+\dfrac{\pi}{3}=\dfrac{\pi}{4} +x +k2\pi}
		 \Leftrightarrow \hoac{&3x=\dfrac{5\pi}{12} +k2\pi\\&x=-\dfrac{\pi}{12} +k2\pi}
		 \Leftrightarrow \hoac{&x=\dfrac{5\pi}{36} +\dfrac{k2\pi}{3}\\&x=-\dfrac{\pi}{12} +k2\pi} (k\in \mathbb{Z}).
		 \end{eqnarray*}
	 	Vậy tập nghiệm $S=\left\{ \left. \dfrac{5\pi}{36} +\dfrac{k2\pi}{3}; -\dfrac{\pi}{12} +k2\pi \right| k\in \mathbb{Z} \right\}$.
	}
\end{bt}

	Giải phương trình $\sin^2 x -4\sin x +3=0$.
	\loigiai{
		Ta có $\sin^2 x -4\sin x +3=0
			\Leftrightarrow \hoac{&\sin x =1 \quad \text{(nhận)}\\&\sin x=3 \quad \text{(loại)}}
			\Leftrightarrow x=\dfrac{\pi}{2}+k2\pi$ $(k\in \mathbb{Z}).$\\
		Vậy tập nghiệm $S=\left\{ \left. \dfrac{\pi}{2} +k2\pi \right| k\in \mathbb{Z} \right\}$.
	}
\end{bt}

\begin{bt}%[HK1, Toán 11, Trần Khai Nguyên, HCM 2017]%[Nguyễn Thị Kiều Ngân - Dự án 11HK1-18]%[1D2B2-1]
	Một nhóm học sinh có $7$ em nam và $3$ em nữ. Người ta cần chọn ra $5$ em trong nhóm tham gia đồng diễn thể dục. Trong $5$ em được chọn, có không quá một em nữ. Hỏi có bao nhiêu cách chọn?
	\loigiai{
		Chọn $5$ em trong đó không quá một em nữ nên có các trường hợp sau:
		\begin{itemize}
			\item TH1: Chọn $1$ nữ và $4$ nam có $\mathrm{C}_3^1 \cdot \mathrm{C}_7^4 =105$ cách.
			\item TH2: Chọn $5$ nam có $\mathrm{C}_7^5 =21$ cách. 
		\end{itemize}
		Vậy có $105+21=126$ cách chọn thỏa yêu cầu đề bài.
	}
\end{bt}

\begin{bt}%[HK1, Toán 11, Trần Khai Nguyên, HCM 2017]%[Nguyễn Thị Kiều Ngân - Dự án 11HK1-18]%[1D2B5-2]
	Một hộp chứa $5$ viên bi xanh, $6$ viên bi vàng và $3$ viên bi đỏ. Lấy ngẫu nhiên ra $4$ viên bi. Tính xác suất sao cho:
	\begin{enumerate}
		\item Có đúng $2$ viên bi xanh, $1$ viên bi vàng và $1$ viên bi đỏ.
		\item Có ít nhất $1$ viên bi đỏ.
	\end{enumerate}
	\loigiai{
		Lấy ngẫu nhiên ra $4$ viên bi trong số $14$ viên bi thì không gian mẫu có số phần tử là $$n(\Omega)=\mathrm{C}_{14}^4=1001.$$
		\begin{enumerate}
			\item Gọi $A$ là biến cố \lq\lq Có đúng $2$ viên bi xanh, $1$ viên bi vàng và $1$ viên bi đỏ\rq\rq.\\
			Khi đó $n(A)=\mathrm{C}_5^2 \cdot \mathrm{C}_6^1 \cdot \mathrm{C}_3^1 =180$.\\
			Vậy xác suất cần tìm là $\mathrm{P}(A)=\dfrac{n(A)}{n(\Omega)}=\dfrac{180}{1001}$.
			\item Gọi $B$ là biến cố \lq\lq Trong $4$ viên lấy ra có ít nhất $1$ viên bi đỏ\rq\rq.\\
			Suy ra biến cố đối của $B$ là $\overline{B}:$ \lq\lq Trong $4$ viên lấy ra không có viên bi đỏ nào\rq\rq.\\
			Khi đó $n(\overline{B})=\mathrm{C}_{11}^4=330$. Do đó $n(B)=n(\Omega)-n(\overline{B})=671$.\\
			Vậy xác suất cần tìm là $\mathrm{P}(B)=\dfrac{n(B)}{n(\Omega)}=\dfrac{671}{1001}=\dfrac{61}{91}$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[HK1, Toán 11, Trần Khai Nguyên, HCM 2017]%[Nguyễn Thị Kiều Ngân - Dự án 11HK1-18]%[1D2B3-1]
	Dùng công thức nhị thức Niu-tơn hãy khai triển $\left( x^3+\dfrac{3}{x^2} \right)^4$, $x \ne 0$.
	\loigiai{
		\begin{eqnarray*}
		\left( x^3+\dfrac{3}{x^2} \right)^4
		&=&\mathrm{C}_4^0 \cdot (x^3)^4 
		+\mathrm{C}_4^1 \cdot (x^3)^3 \cdot \left( \dfrac{3}{x^2} \right)^1 +\mathrm{C}_4^2 \cdot (x^3)^2 \cdot \left( \dfrac{3}{x^2} \right)^2 +\mathrm{C}_4^3 \cdot (x^3)^1 \cdot \left( \dfrac{3}{x^2} \right)^3 +\mathrm{C}_4^4 \cdot \left( \dfrac{3}{x^2} \right)^4\\
		&=& x^{12} +12x^7 +54x^2 +\dfrac{108}{x^3} +\dfrac{81}{x^8}.
		\end{eqnarray*}
	}
\end{bt}

\begin{bt}%[HK1, Toán 11, Trần Khai Nguyên, HCM 2017]%[Nguyễn Thị Kiều Ngân - Dự án 11HK1-18]%[1D3B2-1]
	Cho dãy số $(u_n)$ xác định bởi $\heva{&u_1=2\\&u_{n+1}=2u_n +3n+2, n\ge 1.}$.
	\begin{enumerate}
		\item Tìm $5$ số hạng đầu của dãy.
		\item Bằng phương pháp quy nạp, chứng minh rằng với mọi $n \in \mathbb{N^*}$ ta có $u_n=5\cdot 2^n -3n-5$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $5$ số hạng đầu của dãy là
				\begin{eqnarray*}
					u_1 &=& 2,\\
					u_2 &=& 2u_1+3\cdot 1 +2 =9,\\
					u_3 &=& 2u_2+3\cdot 2 +2 =26,\\
					u_4 &=& 2u_3+3\cdot 3 +2 =63,\\
					u_5 &=& 2u_4+3\cdot 4 +2 =140.
				\end{eqnarray*}
			\item Bằng phương pháp quy nạp, chứng minh rằng với mọi $n \in \mathbb{N^*}$ ta có $u_n=5\cdot 2^n -3n-5$.
				\begin{itemize}
					\item Với $n=1$, ta có $u_1 =5\cdot 2^1 -3\cdot 1-5 =2$. Do đó biểu thức đúng với $n=1$.
					\item Giả sử biểu thức đúng với $n=k \ge 1$ ($k \in \mathbb{N^*}$), tức là $u_k=5\cdot 2^k -3k-5$.\\
					Ta cần chứng minh biểu thức đúng với $n=k+1$, tức là chứng minh $$u_{k+1}=5\cdot 2^{k+1} -3(k+1)-5.$$
					Thật vậy, ta có
					\begin{eqnarray*}
						u_{k+1}&=&2u_k +3k+2 =2(5\cdot 2^k -3k-5)+3k+2\\
						&=& 5\cdot 2^{k+1} -6k-10+3k+2\\
						&=& 5\cdot 2^{k+1} -3k-8\\
						&=& 5\cdot 2^{k+1} -3(k+1)-5.
					\end{eqnarray*}
					Do đó biểu thức đúng với $n=k+1$.
				\end{itemize}
			Vậy $u_n=5\cdot 2^n -3n-5$, $\forall n \in \mathbb{N^*}$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[HK1, Toán 11, Trần Khai Nguyên, HCM 2017]%[Nguyễn Thị Kiều Ngân - Dự án 11HK1-18]%[1D3K3-3]
	Một tam giác vuông có chu vi bằng $3a$, và $3$ cạnh lập thành một cấp số cộng. Tính độ dài ba cạnh của tam giác theo $a$.
	\loigiai{
		Gọi ba cạnh của tam giác vuông lần lượt là $x$, $y$, $z$.\\
		Không mất tính tổng quát có thể giả sử $x \le y < z$ với $z$ là độ dài cạnh huyền.\\
		Theo định lý Pi-ta-go, ta có $x^2+y^2=z^2 \quad (1)$.\\
		Chu vi của tam giác là $3a$ nên $x+y+z=3a \quad (2)$.\\
		Ba cạnh $x$, $y$, $z$ theo thứ tự đó lập thành một cấp số cộng khi và chỉ khi $x+z=2y \quad (3)$.\\
		Từ $(2)$ và $(3)$ suy ra $y=a$.\\
		Ta có hệ $\heva{&x^2+a^2=z^2\\&x+z=2a\\&y=a}$
		$\Leftrightarrow \heva{&x^2+a^2=(2a-x)^2\\&z=2a-x\\&y=a}$
		$\Rightarrow 4ax=3a^2 \Rightarrow x=\dfrac{3a}{4} \Rightarrow z=\dfrac{5a}{4}$.\\
		Vậy độ dài ba cạnh của tam giác là $\dfrac{3a}{4}$, $a$, $\dfrac{5a}{4}$.
	}
\end{bt}

\begin{bt}%[HK1, Toán 11, Trần Khai Nguyên, HCM 2017]%[Nguyễn Thị Kiều Ngân - Dự án 11HK1-18]%[1H2Y1-2]%[1H2B2-3]%[1H2K1-5]
	Cho hình chóp $S.ABCD$ có $ABCD$ là hình thang, đáy lớn là $AB$, $O$ là giao điểm của $AC$ và $BD$. Gọi $M$ là trung điểm $SC$ và $G$ là trọng tâm tam giác $SBC$.
	\begin{enumerate}
		\item Tìm giao tuyến của các mặt phẳng $(SAC)$ và $(SBD)$, $(SDG)$ và $(SAB)$.
		\item Tìm giao điểm $N$ của $SD$ và $(MAB)$.
		\item Gọi $K$ là giao điểm của $AM$ và $NB$. Chứng minh $S$, $K$, $O$ thẳng hàng.
	\end{enumerate}
	\loigiai{
		\begin{center}
		\begin{tikzpicture}[>=stealth, scale=.7]
			\tkzDefPoints{0/0/A, 7/0/B, 5/-2.5/C, 1.5/-2.5/D, 1/6/S}
			\tkzDefMidPoint(S,C)\tkzGetPoint{M}
			\tkzDefMidPoint(B,C)\tkzGetPoint{E}
			\tkzDefMidPoint(S,D)\tkzGetPoint{N}
			\tkzCentroid(S,B,C)\tkzGetPoint{G}
			\tkzInterLL(A,C)(B,D)\tkzGetPoint{O}
			\tkzInterLL(D,E)(A,B)\tkzGetPoint{F}
			\tkzInterLL(A,M)(B,N)\tkzGetPoint{K}
			\tkzDrawSegments(S,A S,C S,D A,D C,D S,E S,F E,F C,E M,G M,N)
			\tkzDrawSegments[dashed](S,B A,B A,C B,D S,O D,G D,E B,F B,E M,A B,G B,N)
			\tkzDrawPoints[fill=black](A,B,C,D,S,O,M,G,E,F,N,K)
			\tkzLabelPoints[above](S,M)		
			\tkzLabelPoints[above right](B,O,G)
			\tkzLabelPoints[below](C,D,K)
			\tkzLabelPoints[below right](E)
			\tkzLabelPoints[right](F)
			\tkzLabelPoints[left](A,N)
			\tkzLabelSegment[pos=.7, above](M,N){$d$}
		\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Tìm giao tuyến của các mặt phẳng $(SAC)$ và $(SBD)$, $(SDG)$ và $(SAB)$.
				\begin{itemize}
					\item $S \in (SAC) \cap (SBD)$.
					\item Ta có $O = AC \cap BD \Rightarrow \heva{&O \in AC, AC\subset (SAC)\\&O \in BD, BD \subset (SBD)}
					\Rightarrow \heva{&O \in (SAC)\\&O \in (SBD)}\\
					\Rightarrow O \in (SAC) \cap (SBD)$.
				\end{itemize}
				Vậy $(SAC) \cap (SBD)=SO$.
				\begin{itemize}
					\item $S \in (SDG) \cap (SAB)$.
					\item $SG$ cắt $BC$ tại $E$ trong $(SBC)$ nên mặt phẳng $(SDG)$ cũng chính là mặt phẳng $(SDE)$.
					\item Trong $(ABCD)$ gọi $F= DE \cap AB$, suy ra
					$\heva{&F \in DE, DE \subset (SDG)\\&F \in AB, AB\subset (SAB)}
					\Rightarrow \heva{&F \in (SDG)\\&F \in (SAB)}\\
					\Rightarrow F \in (SDG) \cap (SAB)$.
				\end{itemize}
				Vậy $(SDG)\cap (SAB) =SF$.
			\item Tìm giao điểm $N$ của $SD$ và $(MAB)$
				\begin{itemize}
					\item $SD \subset (SCD)$.
					\item Ta có $\heva{&M \in (MAB) \cap (SCD)\\& AB \parallel CD}$ suy ra $(MAB) \cap (SCD) =d$ với $d$ qua $M$ và $d \parallel CD$.\\
					\item Trong $(SCD):$ $N=d \cap SD$.
				\end{itemize}
				Vậy $N=SD \cap (MAB)$.
			\item Chứng minh $S$, $K$, $O$ thẳng hàng.\\
				$K$ là giao điểm của $AM$ và $NB$ suy ra
				$\heva{&K \in AM, AM \in (SAC)\\& K \in BN, BN \in (SBD)}
				\Rightarrow \heva{&K \in (SAC)\\& K \in (SBD)}\\
				\Rightarrow K \in (SAC)\cap (SBD)$.\\
				Mà $(SAC)\cap (SBD)=SO$ nên $K \in SO$.\\
				Vậy $K$, $S$, $O$ thẳng hàng.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[HK1, Toán 11, Trần Khai Nguyên, HCM 2017]%[Nguyễn Thị Kiều Ngân - Dự án 11HK1-18]%[1H2G2-3]
	Cho hình chóp $S.ABC$, $M$ là một điểm nằm trong tam giác $ABC$. Các đường thẳng qua $M$ và song song $SA$, $SB$ và $SC$ cắt các mặt $(SBC)$, $(SCA)$, $(SAB)$ lần lượt tại các điểm $A'$, $B'$, $C'$. Nêu cách dựng các điểm $A'$, $B'$, $C'$ và chứng minh $\dfrac{MA'}{SA}+\dfrac{MB'}{SB}+\dfrac{MC'}{SC}=1$.
	\loigiai{
		\immini{
			Gọi $N$ là giao điểm của $AM$ và $BC$. Trong mặt phẳng $(SAN)$ kẻ đường thẳng đi qua $M$ và song song với $SA$, đường này cắt $SN$ tại $A'$.\\
			Gọi $F$ là giao điểm của $BM$ và $AC$. Trong mặt phẳng $(SBF)$ kẻ đường thẳng đi qua $M$ và song song với $SB$, đường này cắt $SF$ tại $B'$.\\

				\item $\dfrac{MB'}{BS}=\dfrac{MF}{BF}=\dfrac{S_{\triangle MAC}}{S_{\triangle BAC}}$.
