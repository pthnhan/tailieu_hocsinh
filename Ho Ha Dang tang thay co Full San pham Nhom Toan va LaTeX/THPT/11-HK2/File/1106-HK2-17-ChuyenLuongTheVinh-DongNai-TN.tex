\begin{ex}%[HK2 chuyên Lương Thế Vinh, Đồng Nai, 2017]%[1D4Y2-3]%[Nguyễn Tất Thu, 11-HK2]
Giá trị của $\lim\limits_{x \to 1} \dfrac{x^3 - 3x^2 + 2}{x^2 - 4x + 3}$ là
	\choice
	{\True $\dfrac{3}{2}$}
	{$\dfrac{5}{2}$}
	{$\dfrac{7}{5}$}
	{$\dfrac{8}{7}$}
	\loigiai{
	Ta có $$\lim\limits_{x \to 1} \dfrac{x^3 - 3x^2 + 2}{x^2 - 4x + 3}=\lim\limits_{x \to 1}\dfrac{(x-1)(x^2-2x-2)}{(x-1)(x-3)}=\lim\limits_{x \to 1}\dfrac{x^2-2x-2}{x-3}=\dfrac{3}{2}.$$
	}
	\end{ex}

\begin{ex}%[HK2 chuyên Lương Thế Vinh, Đồng Nai, 2017]%[1D4B2-7]%[Nguyễn Tất Thu, 11-HK2]
	Giá trị của số thực $m$ sao cho $\lim\limits_{x \to  - \infty } \dfrac{(2x^2 - 1)(mx + 3)}{x^3 + 4x + 7} = 6$ là
	\choice
	{$m = -3$}
	{$m = 2$}
	{\True $m = 3$}
	{$m = -2$}
	\loigiai{
	Ta có $$\lim\limits_{x \to -\infty}\dfrac{(2x^2-1)(mx+3)}{x^3+4x+7}=\lim\limits_{x \to -\infty}\dfrac{\left(2-\dfrac{1}{x^2} \right)\left(m+\dfrac{3}{x} \right)}{1+\dfrac{4}{x^2}+\dfrac{7}{x^3}}=2m.$$
	Suy ra $2m=6\Leftrightarrow m=3.$
	}
	\end{ex}

		\item[(IV)] Nếu phương trình $ f(x)=0 $ có nghiệm trên  $ (a,b) $ thì hàm số $ f(x) $ liên tục trên  $ (a,b) $

	{$1$}
	{$2$}
	{\True $4$}
	{$3$}

Do đó mệnh đề (IV) là mệnh đề sai.
