Thực hiện phép tính 
	\begin{enumerate}
	\item $\left(x + 3\right)\left(4 - 9x\right) + \left(3x + 2\right)^{2}$
	\item $\left(2x^3 - 3x^2 - 11x + 6\right)\colon\left(x - 3\right)$
	\item $\dfrac{x^2}{3x + 9} + \dfrac{6x + 9}{3x + 9}$
	\item $\dfrac{1 - 3x}{2x} + \dfrac{3x - 2}{2x - 1} + \dfrac{3x - 2}{2x - 4x^2}$
\end{enumerate}
	\loigiai{
		\begin{enumerate}
		\item Ta có
		\begin{eqnarray*}
			\left(x + 3\right)\left(4 - 9x\right) + \left(3x + 2\right)^{2}
			&=& 4x - 9x^2 + 12 - 27x + 9x^2 + 12x + 4\\
			&=&16 - 11x.
		\end{eqnarray*}
		\item  Điều kiện $x - 3\neq 0\Leftrightarrow x\neq 3\quad (*)$\\
		Với điều kiện $(*)$ ta có	
			\begin{eqnarray*}
			\left(2x^3 - 3x^2 - 11x + 6\right)\colon\left(x - 3\right) &=&\dfrac{2x^3 - 3x^2 -  11x + 6}{x - 3}\\
			&= &\dfrac{2x^3 -  6x^2 + 3x^2 -  9x  - 2x + 6}{x - 3}\\
			& = &\dfrac{2x^2\left(x - 3\right) + 3x\left(x - 3\right) - 2\left(x - 3\right) }{x - 3}\\
			& = &\dfrac{\left(x - 3\right)\left(2x^2 + 3x - 2\right)}{x - 3} = 2x^2 + 3x - 2.
			\end{eqnarray*}
		\item Điều kiện $3x +  9\neq 0\Leftrightarrow x\neq - 3\quad (*)$\\
		Với điều kiện $(*)$ ta có
		\begin{eqnarray*}
			\dfrac{x^2}{3x + 9} + \dfrac{6x + 9}{3x + 9} &=& \dfrac{x^2 + 6x + 9}{3x + 9}\\
			 &=& \dfrac{\left(x + 3\right)^2}{3\left(x + 3\right)} = \dfrac{x + 3}{3}.
		\end{eqnarray*}
	\item Điều kiện $\heva{&2x\neq 0 \\ & 2x - 1\neq 0\\&2x -4x^2\neq 0}\Leftrightarrow\heva{&2x\neq 0 \\ & 2x - 1\neq 0\\&2x\left(1 - 2x\right)\neq 0}\Leftrightarrow\heva{&x\neq 0 \\ &x\neq\dfrac{1}{2}}\quad (*)$\\
	Với điều kiện $(*)$ ta có
	\begin{eqnarray*}
	\dfrac{1 - 3x}{2x} + \dfrac{3x - 2}{2x - 1} + \dfrac{3x - 2}{2x - 4x^2} &=&\dfrac{1 - 3x}{2x} + \dfrac{3x - 2}{2x - 1} + \dfrac{3x - 2}{2x\left(1 - 2x\right)}\\
		&=&\dfrac{\left(1 - 3x\right)\left(1 - 2x\right)}{2x\left(1 - 2x\right)} - \dfrac{2x\left(3x - 2\right)}{2x\left(1 - 2x\right)} + \dfrac{3x - 2}{2x\left(1 - 2x\right)}\\
		&=&\dfrac{\left(1 - 3x\right)\left(1 - 2x\right) - 2x\left(3x - 2\right) + 3x - 2}{2x\left(1 - 2x\right)}\\
		&=&\dfrac{\left(1 - 3x\right)\left(1 - 2x\right) + \left(3x - 2\right)\left(1 - 2x\right)}{2x\left(1 - 2x\right)}\\
		&=&\dfrac{-\left(1 - 2x\right)}{2x\left(1 - 2x\right)} = - \dfrac{1}{2x}.
			\end{eqnarray*}
	\end{enumerate}
	}
\end{bt}

Phân tích đa thức sau thành nhân tử
		\begin{enumerate}
			\item $2x\left(x - 4\right) - xy + 4y$
			\item $x^2 - 4x + 4 - 25y^2$
			\item $x^2 - 2x - 4y^2 + 4y$
		\end{enumerate}
\loigiai{		
	\begin{enumerate}
		\item Ta có 
		\begin{eqnarray*}
			2x\left(x - 4\right) - xy + 4y
			&=& 2x\left(x - 4\right) - y\left(x - 4\right)\\
			&=&\left(x - 4\right)\left(2x -  y\right).
		\end{eqnarray*}
		\item Ta có
			\begin{eqnarray*}
			x^2 - 4x + 4 - 25y^2
			&=&\left(x - 2\right)^2 - \left(5y\right)^2\\
			&=&\left(x - 2 + 5y\right)\left(x -  2 - 5y\right)\\
			&=&\left(x + 5y - 2\right)\left(x -  5y - 2\right).
		\end{eqnarray*}
		\item 
		Ta có
		\begin{eqnarray*}
			x^2 - 2x - 4y^2 + 4y
			&=&x^2 - 2x + 1 - 4y^2 + 4y - 1\\
			&=&\left(x - 1\right)^2 - \left(2y - 1\right)^2\\
			&=&\left(x - 1 + 2y - 1\right)\left(x -  1 - 2y + 1\right)\\
			&=&\left(x + 2y - 2\right)\left(x -  2y\right).
			\end{eqnarray*}
	\end{enumerate}
}	
\end{bt}
\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Phú Nhuận, Hồ Chí Minh]%[Bùi Quốc Hoàn, dự án 8DT18-HK1]%[8D1B6]%[8D1B5] 
Tìm $x$ biết
	\begin{enumerate}
		\item $x^2 - 4x = 0$
		\item $\left(2x + 5\right)\left(2x - 5\right) + 10 = \left(2x - 3\right)^2$
		\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $x^2 - 4x=0\Leftrightarrow x\left(x - 4\right) = 0\Leftrightarrow \left[\begin{aligned}&x = 0\\ &x - 4 = 0\end{aligned}\right.  \Leftrightarrow \left[\begin{aligned}&x  = 0\\ &x = 4. \end{aligned}\right.$\\
 Vậy phương trình có hai nghiệm là $x = 0$ và $x = 4$.
			
			\item Ta có 
				\begin{eqnarray*}
				&{ }&\left(2x + 5\right)\left(2x - 5\right) + 10 = \left(2x - 3\right)^2\\
				&\Leftrightarrow& 4x^2 - 25 + 10 = 4x^2 - 12x + 9\\
				&\Leftrightarrow&12x = 24\Leftrightarrow x = 2.
			\end{eqnarray*}
		\end{enumerate}
	Vậy phương trình có hai nghiệm: $x =2$.
}
\end{bt}

Chào mừng ngày nhà giáo Việt Nam 20/11, trường của An tổ chức cho các khối $8$ thi đấu bóng đá theo thể thức vòng tròn một lượt (mỗi đội đều  gặp nhau một trận). Tổng số các trận các bạn đã thi đấu được tính bởi biểu thức $T = \dfrac{x\left(x - 1\right)}{2}$ ($T$ là tổng số trận đấu, $x$ là số đội tham gia). Em hãy tính xem có bao nhiêu đội tham gia thi đấu nếu tổng số trận đấu là $28$ trận. 
	\loigiai{Do giả thiết suy ra $x\geq 2$. Để thỏa mãn bài toán khi 
			\begin{eqnarray*}
			\dfrac{x\left(x - 1\right)}{2}  = 28 &\Leftrightarrow& x^2 - x - 56 = 0\\
			&\Leftrightarrow& x^2 - 64 - x + 8 = 0\\
			&\Leftrightarrow&\left(x - 8\right)\left(x + 8\right) - \left(x - 8\right) = 0\\
			&\Leftrightarrow&\left(x - 8\right)\left(x + 7\right) = 0\\
		 &\Leftrightarrow& \left[\begin{aligned} &x - 8 = 0\\ &x + 7 = 0\end{aligned}\right.\Leftrightarrow \left[\begin{aligned} &x = 8\\ &x = - 7.\end{aligned}\right.
				\end{eqnarray*}
	So sánh với điều kiện suy ra $x = 8$.
Vậy số đội tham gia thi đấu là $8$ đội.
}
\end{bt}
