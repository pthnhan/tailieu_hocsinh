		\begin{enumerate}
		\item Rút gọn biểu thức
		$$P=\left({\dfrac{3x+\sqrt{{16x}}-7}{x+2\sqrt{x}-3}-\dfrac{\sqrt{x}+1}{\sqrt{x}+3}-\dfrac{\sqrt{x}+7}{\sqrt{x}-1}}\right):\left({2-\dfrac{\sqrt{x}}{\sqrt{x}-1}}\right),$$
		với $x\geqslant 0,x\ne 1,x\ne 4$.
		\item Cho $a=\dfrac{13}{\sqrt{{19+8\sqrt{3}}}}$. Không sử dụng máy tính cầm tay, hãy tính giá trị của biểu thức: 
		$$A=\dfrac{a^4-6a^3-2a^2+18a+23}{a^2-8a+15}.$$
	\end{enumerate}	
	\loigiai{
		\begin{enumerate}
			\item Với $0<x\ne 1,x\ne 4$, ta có
			{\allowdisplaybreaks
				\begin{align*}
				P &= \left[ {\frac{{3x + 4\sqrt x  - 7}}{{\left( {\sqrt x  - 1} \right)\left( {\sqrt x  + 3} \right)}} - \frac{{\sqrt x  + 1}}{{\sqrt x  + 3}} - \frac{{\sqrt x  + 7}}{{\sqrt x  - 1}}} \right]:\left( {2 - \frac{{\sqrt x }}{{\sqrt x  - 1}}} \right)\\
				& = \left[ {\frac{{3x + 4\sqrt x  - 7 - \left( {x - 1} \right) - \left( {x + 10\sqrt x  + 21} \right)}}{{\left( {\sqrt x  - 1} \right)\left( {\sqrt x  + 3} \right)}}} \right]:\left( {\frac{{2\sqrt x  - 2 - \sqrt x }}{{\sqrt x  - 1}}} \right)\\
				&= \left[ {\frac{{x - 6\sqrt x  - 27}}{{\left( {\sqrt x  - 1} \right)\left( {\sqrt x  + 3} \right)}}} \right].\frac{{\sqrt x  - 1}}{{\sqrt x  - 2}}\\
				&= \frac{{\left( {\sqrt x  - 9} \right)\left( {\sqrt x  + 3} \right)}}{{\left( {\sqrt x  - 1} \right)\left( {\sqrt x  + 3} \right)}}.\frac{{\sqrt x  - 1}}{{\sqrt x  - 2}} = \frac{{\sqrt x  - 9}}{{\sqrt x  - 2}}
				\end{align*}}Kết luận: $P=\dfrac{\sqrt{x}-9}{\sqrt{x}-2},x\geqslant 0,x\ne 1,x\ne 4$.
			\item Ta có 
			$$a = \frac{{13}}{{\sqrt {19 + 8\sqrt 3 } }} = \frac{{13}}{{\sqrt {{{\left( {4 + \sqrt 3 } \right)}^2}} }} = \frac{{13}}{{4 + \sqrt 3 }} = 4 - \sqrt 3 \,\,\,\left( {0 < a < 4} \right).$$
			Khi đó $4-a=\sqrt{3}\Leftrightarrow 16-8a+a^2=3 \Rightarrow a^2-8a+15=2$.\\
			Mặt khác
			{\allowdisplaybreaks
				\begin{align*}
				A&=\dfrac{a^4-6a^3-2a^2+18a+23}{a^2-8a+15}=a^2+2a-1+\dfrac{38-20a}{a^2-8a+15}\\
				&=a^2+2a-1+\dfrac{38-20a}{2}=a^2+2a-1+19-10a=a^2-8a+18=5.
				\end{align*}}Kết luận: $A=5 $.
		\end{enumerate}

\begin{ex}%[Đề thi HSG 9, tỉnh Quảng Bình, năm 2018]%[Nguyễn Văn Vũ, Dự án EX-9G-18]%[9D0G2]%[9D0G2]
    	\begin{enumerate}
    	\item Cho phương trình $x^2-2\left({m+1}\right)x+2m+10=0$ ($m$ là tham số). Tìm $m$ để phương trình có hai nghiệm  $x_1, x_2$ thỏa mãn $P=10x_1x_2+x_1^2+x_2^2$ đạt giá trị nhỏ nhất.
    	\item Giải hệ phương trình $\left\{ \begin{array}{l}
    	\sqrt {x + 5}  + \sqrt {y - 2}  = 7\\
    	\sqrt {x - 2}  + \sqrt {y + 5}  = 7.
    	\end{array} \right.$
    \end{enumerate} 
    \loigiai{
    	\begin{enumerate}
    		\item Ta có  $\Delta '=\left({m+1}\right)^2-2m-10=m^2-9$.
    		Phương trình có hai nghiệm  $x_1, x_2$ khi 
    		$ \Delta '\geqslant 0$, hay
    		$$m^2-9\geqslant 0\Leftrightarrow \left[{\begin{aligned}&{m\leqslant -3} \\ &{m\geqslant 3.} \\ \end{aligned}}\right.$$
    		Theo định lí Viet, ta có
    		$x_1+x_2=2\left({m+1}\right);x_1.x_2=2m+10$.\\
    		Theo bài ra ta có
    		{\allowdisplaybreaks
    			\begin{align*}
    			P&=10\text{x}_1x_2+x_1^2+x_2^2=\left({x_1+x_2}\right)^2+8\text{x}_1x_2\\
    			&=4\left({m+1}\right)^2+8\left({2m+10}\right)=4\left({m^2+6m+21}\right)\\
    			&=4\left[{{\left({m+3}\right)}^2+12}\right]=4\left({m+3}\right)^2+48\geqslant 48.
    			\end{align*}}Dấu "=" xảy ra khi và chỉ khi $m=-3$.\\
    		Kết luận: $m=-3$.
    		\item Điều kiện $x,y\geqslant 2$.
    		Đặt: 
    		$\left\{ {\begin{array}{*{20}{c}}
    			{u = \sqrt {x - 2} }\\
    			{v = \sqrt {y - 2} }
    			\end{array}} \right.\,\,\,\left( {u,v \ge 0} \right) \Leftrightarrow \left\{ {\begin{array}{*{20}{c}}
    			{x = {u^2} + 2}\\
    			{y = {v^2} + 2.}
    			\end{array}} \right.$\\
    		Khi đó hệ phương trình đã cho trở thành:
    		$\left\{ \begin{array}{ll}
    		\sqrt {{u^2} + 7}  + v = 7&(1)\\
    		u + \sqrt {{v^2} + 7}  = 7.&(2)
    		\end{array} \right.$\\
    		Lấy (1) trừ (2), ta được: 
    		{\allowdisplaybreaks
    			\begin{align*}
    			&\sqrt {{u^2} + 7}  + v = u + \sqrt {{v^2} + 7} \\
    			\Leftrightarrow& {u^2} + 7 + 2v\sqrt {{u^2} + 7}  + {v^2} = {v^2} + 7 + 2u\sqrt {{v^2} + 7}  + {u^2}\\
    			\Leftrightarrow& v\sqrt {{u^2} + 7}  = u\sqrt {{v^2} + 7} \\
    			\Leftrightarrow& u = v
    			\end{align*}}Thế $u=v$ vào (1), ta có:
    		$\sqrt{{u^2+7}}+u=7\Leftrightarrow u^2+7=\left({7-u}\right)^2,\left({0\leqslant u\leqslant 7}\right)\Leftrightarrow u=3$.\\
    		Khi đó: $\left\{ {\begin{array}{*{20}{c}}
    			{\sqrt {x - 2}  = 3}\\
    			{\sqrt {y - 2}  = 3}
    			\end{array}} \right. \Leftrightarrow x = y = 11$.\\
    		Kết luận: $(x;y)=(11;11)$ là nghiệm của hệ phương trình.
    	\end{enumerate} 
    }
\end{ex}

Cho $x,y,z$ là các số thực dương thỏa mãn $x+y+z=4$. 
	Chứng minh rằng: $$\dfrac{1}{2xy+xz+yz}+\dfrac{1}{xy+2xz+yz}+\dfrac{1}{xy+xz+2yz}\leqslant \dfrac{1}{xyz}.$$
	\loigiai{Với $a, b>0$, ta có: $\left({a+b}\right)^2\geqslant 4ab\Leftrightarrow \dfrac{1}{a+b}\leqslant \dfrac{a+b}{4ab}\Leftrightarrow \dfrac{1}{a+b}\leqslant \dfrac{1}{4}\left({\dfrac{1}{a}+\dfrac{1}{b}}\right)$. 
		Dấu "=" xảy ra khi và chỉ khi $a=b$.
		Áp dụng kết quả trên ta có:
		{\allowdisplaybreaks
			\begin{align*}
			\dfrac{1}{2xy+xz+yz}&\leqslant \dfrac{1}{4}\left({\dfrac{1}{xy+xz}+\dfrac{1}{xy+yz}}\right)\\
			&\leqslant \dfrac{1}{4}\left[{\dfrac{1}{4}\left({\dfrac{1}{xy}+\dfrac{1}{xz}}\right)+\dfrac{1}{4}\left({\dfrac{1}{xy}+\dfrac{1}{yz}}\right)}\right]\\ &=\dfrac{1}{16}\left({\dfrac{2}{xy}+\dfrac{1}{xz}+\dfrac{1}{yz}}\right)=\dfrac{1}{16}\left({\dfrac{2z+y+x}{xyz}}\right).\tag{1}
			\end{align*}}Tương tự, ta cũng có:
		{\allowdisplaybreaks
			\begin{align*}
			&\dfrac{1}{xy+2xz+yz}\leqslant \dfrac{1}{16}\left({\dfrac{z+2y+x}{xyz}}\right).\tag{2}\\
			&\dfrac{1}{xy+xz+2yz}\leqslant \dfrac{1}{16}\left({\dfrac{z+y+2x}{xyz}}\right).\tag{3}
			\end{align*}}Vậy $\dfrac{1}{2xy+xz+yz}+\dfrac{1}{xy+2xz+yz}+\dfrac{1}{xy+xz+2yz}\leqslant \dfrac{1}{16}\left({\dfrac{4x+4y+4z}{xyz}}\right)=\dfrac{1}{xyz}$.
		Các bất đẳng thức (1), (2), (3) dấu "=" xảy ra khi và chỉ khi $x=y=z$.
		Do đó  đẳng thức xảy ra khi và chỉ khi $x=y=z=\dfrac{4}{3}$.
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Quảng Bình, năm 2018]%[Nguyễn Văn Vũ, Dự án EX-9G-18]%[8D0G5]
Cho $n$ là một số nguyên dương thỏa mãn $n+1$ và $2n+1$ đồng thời là hai số chính phương. Chứng minh rằng $n$ chia hết cho $24$.
	\loigiai{Đặt $n+1=k^2$, $2n+1=m^2$ $(k,m\in \mathbb{N})$.
		Vì $2n+1$ là một số lẻ nên $m$ là một số lẻ.
		Đặt $m=2t+1$ $(t\in \mathbb{N})$. Khi đó: $$2n+1=\left({2t+1}\right)^2\Rightarrow n=2t\left({t+1}\right).$$
		Suy ra $n$ chẵn nên $k$ lẻ.
		Ta lại có: 
		$$n+1=k^2\Leftrightarrow n=\left({k-1}\right)\left({k+1}\right).$$
		Vì $k$ lẻ nên $k-1$ và $k+1$ là hai số chẵn liên tiếp.
		Do đó $n$ chia hết cho $8$.\hfill$(1)$\\
		Mặt khác $k^2+m^2=\left({n+1}\right)+\left({2n+1}\right)=3n+2$ nên $k^2+m^2$ chia cho $3$ dư $2$.
		Do $k^2$ và $m^2$ chia cho $3$ luôn dư $0$ hoặc dư 1.
		Suy ra cả hai số $k^2$ và $m^2$ khi chia cho 3 đều dư 1, nên $m^2-k^2$ chia hết cho 3, mà $m^2-k^2=2n+1-\left({n+1}\right)=n$. 
		Do đó  $n$  chia hết cho 3.\hfill$(2)$\\ 
		Từ (1) và (2), kết hợp với $(3, 8)=1$  suy ra $n$  chia hết cho $24$.					 
	}
\end{ex}
