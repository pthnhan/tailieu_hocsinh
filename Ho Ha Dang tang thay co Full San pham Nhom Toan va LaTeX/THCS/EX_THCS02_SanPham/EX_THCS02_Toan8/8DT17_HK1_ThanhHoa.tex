			\item [b)] $x.(x^2-2x+5)$.

			\end{enumerate}

		Phân tích các đa thức sau thành nhân tử:
		\begin{enumerate}
			\item  [a)] $5x^2y-10xy^2$.
			\item [b)] $3(x+3)-x^2+9$.
			\item [c)] $x^2-y^2+xz-yz$.
		\end{enumerate}
		\loigiai{
			\begin{enumerate}
				\item  [a)] $5x^2y-10xy^2=5xy(x-y)$.
				\item [b)] $3(x+3)-x^2+9=3(x+3)+(9-x^2)=3(x+3)+(3+x)(3-x)=(3+x)(3+3-x)=(3+x)(6-x)$.
				\item [c)] $x^2-y^2+xz-yz=(x-y)(x+y)+z(x-y)=(x-y)(x+y+z)$.
			\end{enumerate}
		}
	
	\end{ex}
	\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2016-2017, Phòng GD Thành Phố - Thanh Hóa]%[8D2B1]%[8D2B5]%[8D2B9]
		Cho biểu thức: $A=\dfrac{x^2}{x^2-4}-\dfrac{x}{x-2}+\dfrac{2}{x+2}$.
		\begin{enumerate}
			\item  [a)] Với điều kiện nào của $x$ thì giá trị biểu thức $A$ được xác định?
			\item [b)] Rút gọn biểu thức $A$.
			\item [c)] Tính giá trị của biểu thức $A$ tại $x=1$.
		\end{enumerate}
		\loigiai{
			\begin{enumerate}
				\item  [a)] $x-2\ne0\Leftrightarrow x\ne2 ~\text{và}~x+2\ne0\Leftrightarrow x\ne-2$.\\
				Vậy với $x\ne2$ và $x\ne-2$ thì giá trị của biểu thức $A$ được xác định.
				\item [b)] Ta có $A=\dfrac{x^2}{x^2-4}-\dfrac{x}{x-2}+\dfrac{2}{x+2}=\dfrac{x^2-x(x+2)+2(x-2)}{(x+2)(x-2)}
				\\~~~~~~~~~~~=\dfrac{x^2-x^2-2x+2x-4}{x^2-4}=\dfrac{-4}{x^2-4}$.
				\item [c)] Khi $x=1$,  ta có $A=\dfrac{-4}{1^2-4}=\dfrac{4}{3}$.
			\end{enumerate}	
		}
	\end{ex}

		Cho tam giác $MNP$ vuông tại $M$, đường cao $MH$. Gọi $D,E$ lần lượt là chân các đường vuông góc hạ từ $H$ xuống $MN$ và $MP$.
		\begin{enumerate}
			\item  [a)] Chứng minh tứ giác $MDHE$ là hình chữ nhật.
			\item [b)] Gọi $A$ là trung điểm của $HP$. Chứng minh tam giác $DEA$ vuông.
			\item [c)] Tam giác $MNP$ cần có thêm điều kiện gì để $DE=2EA$.
		\end{enumerate}	
		\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\tkzDefPoints{0/0/M, 0/4/N, 3/0/P}

		\tkzInterLC(N,P)(K1,P)

		\tkzMarkRightAngle(M,D,H)

		\tkzLabelPoints[right](H,A)

			\begin{enumerate}
				\item  [a)] Chứng minh tứ giác $MDHE$ là hình chữ nhật.\\
				Ta có: $\widehat{DME}=90^\circ~$(do $\triangle{MNP}$ vuông tại $M$).\\
				$HD\bot MD$ nên $\widehat{HDM}=90^\circ$.\\
				$HE\bot ME$ nên $\widehat{HEM}=90^\circ$.\\
				Vậy tứ giác $MDHE$ là hình chữ nhật (do có $3$ góc vuông).
				\item [b)] Gọi $A$ là trung điểm của $HP$. Chứng minh tam giác $DEA$ vuông.\\
				Gọi $O$ là tâm hình chữ nhật $MDHE$, ta có $OH=OE$ nên $\widehat{OHE}=\widehat{OEH}$.\\
				Tam giác $EHP$ vuông tại $E$ có $A$ là trung điểm $PH$ nên $AE=AH$, suy ra $\widehat{EHA}=\widehat{HEA}$.\\
				$\Rightarrow \widehat{AEO}=\widehat{AHO}=90^\circ$.\\
				Do đó $\widehat{AEO}=90^\circ$ hay tam giác $DEA$ vuông tại $E$.
				\item [c)] Tam giác $MNP$ cần có thêm điều kiện gì để $DE=2EA$.\\

				$\Leftrightarrow{MDEH}$ là hình vuông.\\
				$\Leftrightarrow{MH}$ là phân giác của góc $M$ và $MH$ là đường cao nên tam giác $MNP$ vuông cân tại $M$.
			\end{enumerate}

			Cho $a+b=1$. Tính giá trị của các biểu thức sau: $M=a^3+b^3+3ab(a^2+b^2)+6a^2b^2(a+b)$.
			\loigiai{
				Ta có 	
				$M=a^3+b^3+3ab(a^2+b^2)+6a^2b^2(a+b).\\
				~~~~=(a+b)(a^2-ab+b^2)+3ab((a+b)^2-2ab)+6a^2b^2(a+b).\\
				~~~~=(a+b)[(a+b)^2-3ab)]+3ab[(a+b)^2-2ab)]+6a^2b^2(a+b)$.\\
				Vì~$a+b=1$~\text{nên}~$M=(1-3ab)+3ab(1-2ab)+6a^2b^2=1-3ab+3ab-6a^2b^2+6a^2b^2=1$.\\
			}
