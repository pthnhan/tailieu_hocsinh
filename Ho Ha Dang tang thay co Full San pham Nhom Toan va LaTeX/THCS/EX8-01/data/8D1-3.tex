\section{Những hằng đẳng thức đáng nhớ}
\subsection{Tóm tắt lí thuyết}
\begin{enumerate}[\tickEX]
	\item Bảy hằng đẳng thức đáng nhớ là những đẳng thức cơ bản nhất mà mỗi người học toán cần phải nắm vững. Các đẳng thức được chứng minh bằng phép nhân đa thức với đa thức. Những đẳng thức này được sử dụng thường xuyên trong các bài toán liên quan đến giải phương trình, nhân chia các đa thức, biến đổi biểu thức tại cấp học trung học cơ sở và trung học phổ thông. Nắm được bảy hằng đẳng thức đáng nhớ giúp giải nhanh những bài toán phân tích đa thức thành nhân tử.
	\item Dùng $A, B$ để chỉ các biểu thức đại số ta có $7$ hằng đẳng thức đáng nhớ sau
\end{enumerate}
\begin{center}
	
	\begin{tabular}{|l|l|}
	\hline 
	Bình phương của một tổng & $(A+B)^2 = A^2 + 2AB + B^2$\\ 
	\hline 
	Bình phương của một hiệu & $(A-B)^2 = A^2 - 2AB + B^2$\\ 
	\hline 
	Hiệu hai bình phương & $A^2 - B^2 = (A-B)(A+B)$\\ 
	\hline 
	Lập phương của một tổng &  $(A+B)^3 = A^3 + 3A^2B + 3AB^2 + B^3$\\ 
	\hline 
	Lập phương của một hiệu & $(A-B)^3 = A^3 - 3A^2B + 3AB^2 - B^3$\\ 
	\hline 
	Tổng hai lập phương & $A^3 + B^3 = (A+B)(A^2 - AB + B^2)$\\ 
	\hline 
	Hiệu hai lập phương & $A^3 - B^3 = (A-B)(A^2 + AB + B^2) $\\ 
	\hline 
\end{tabular}
\end{center} 
\begin{note}
Ta cũng có thể viết hằng đẳng thức bình phương của một hiệu bằng cách sử dụng hằng đẳng thức bình phương của một tổng 
	\[\left(A-B\right)^2 = \left[A+(-B)\right]^2 = A^2 + 2A \cdot (-B) + B^2 = A^2 - 2AB + B^2.\]
	Tương tự cho hằng đẳng thức lập phương của một hiệu
	\[(A-B)^3 = \left[A+(-B)\right]^3 = A^3 + 3A^2(-B) + 3A(-B)^2 + (-B)^3 = A^3 - 3A^2B + 3AB^2 - B^3.\]
\end{note}
\subsection{Các dạng toán}
\begin{dang}[Vận dụng hằng đẳng thức để tính]
	Xem biểu thức  đã cho thuộc dạng hằng đẳng thức nào thì vận dụng hằng đẳng thức ấy để khai triển và ngược lại.	
\end{dang}
\begin{vd}
	Áp dụng hằng đẳng thức bình phương một tổng hoặc bình phương một hiệu để tính
	\begin{listEX}[2]
		\item $(m+1)^2$;
		\item $(k-2)^2$;
		\item $(4x + 7)^2$;
		\item $\left(6x - \dfrac{1}{3}y\right)^2$.
	\end{listEX}
\loigiai{
	\begin{listEX}
		\item $(m+1)^2 = m^2 + 2 \cdot 1 \cdot m + 1^2 = m^2 + 2m + 1$;
		\item $(k-2)^2 = k^2 - 2 \cdot 2 \cdot k + 2^2 = k^2 -4k +4$;
		\item $(4x + 7)^2 = (4x)^2 + 2 \cdot 4x \cdot 7 + 7^2 = 16x^2 + 56x + 49$;
		\item $\displaystyle \left(6x - \frac{1}{3}y\right)^2 = (6x)^2 - 2 \cdot 6x \left( \frac{1}{3}y\right) + \left( \frac{1}{3} y\right)^2 = 36x^2 - 4xy + \dfrac{1}{9} y^2$.	
	\end{listEX}
}
\end{vd}

\begin{vd}
	Áp dụng hằng đẳng thức hiệu hai bình phương để tính
	\begin{listEX}[2]
		\item $5^2 - 4^2$;
		\item $a^2 - 4$;
		\item $t^2 - u^2$;
		\item $\left(3x^2 - 5xy^3\right)(3x^2 + 5xy^3)$.
	\end{listEX}
\loigiai{
	\begin{listEX}
		\item $5^2 - 4^2 = (5-4)(5+4) = 1 \cdot 9 = 9$;
		\item $a^2 - 4 = a^2 - 2^2 = (a-2)(a+2)$;
		\item $t^2 - u^2 = (t-u)(t+u)$;
		\item $\left(3x^2 - 5xy^3\right)(3x^2 + 5xy^3) = \left(3x^2\right)^2 - \left(5xy^3\right)^2 = 9x^4 - 25x^2y^6$.
	\end{listEX}
}
\end{vd}

\begin{vd}
	Áp dụng hằng đẳng thức lập phương của một tổng hoặc lập phương của một hiệu để tính
	\begin{listEX}[2]
		\item $(u+1)^3$;
		\item $(r-1)^3$;
		\item $\left(2x^2 + 5y\right)^3$;
		\item $\left(3x^3 - 4xy\right)^3$.
	\end{listEX}
\loigiai{
	\begin{listEX}
		\item $(u+1)^3 = u^3 + 3u^2 + 3u + 1$;
		\item $(r-1)^3 = r^3 - 3 \cdot r^2 \cdot 1 + 3r \cdot 1^2 - 1 =  r^3 - 3r^2 + 3r - 1$;
		\item Ta có {\allowdisplaybreaks\begin{eqnarray*}			
			\left(2x^2 + 5y\right)^3 &=& \left(2x^2\right)^3 + 3 \cdot \left(2x^2\right)^2 \cdot (5y) + 3 \cdot \left(2x^2\right)\left(5y\right) + 3 \cdot \left(2x^2\right)\left(5y\right)^2 + \left(5y\right)^2\\
			&=& 8x^6 + 60x^4y + 150x^2y^2 + 125y^3.
		\end{eqnarray*}} 
		\item Ta có {\allowdisplaybreaks\begin{eqnarray*}
			\left(3x^3 - 4xy\right)^3 &=& \left(3x^3\right)^3 - 3 \cdot \left(3x^3\right)^2 \cdot \left(4xy\right) + 3 \left(3x^3\right)\left(4xy\right)^2 - \left(4xy\right)^3\\
			&=& 27x^9 - 108x^7y + 144x^5y^2 - 64x^3y^3.
		\end{eqnarray*}}
	\end{listEX}
}
\end{vd}
\begin{vd}
	Áp dụng hằng đẳng thức tổng hai lập phương hoặc hiệu hai lập phương để tính
	\begin{listEX}[2]
		\item $y^3 + 2^3$;
		\item $u^3 - v^3$;
		\item $\left(6x + \dfrac{1}{2}\right)\left(36x^2 - 3x + \dfrac{1}{4}\right)$;
		\item $\left(x - 5y^2\right)\left(x^2 + 5xy^2 + 25y^4\right)$.
	\end{listEX}
\loigiai{
	\begin{listEX}
		\item $y^3 - 2^3 = (y+2)(y^2 - 2y + 4)$;
		\item $u^3 - v^3 = (u-v)(u^2 + uv + v^2)$;
		\item $\left(6x + \dfrac{1}{2}\right)\left(36x^2 - 3x + \dfrac{1}{4}\right) = (6x)^3 + \left(\frac{1}{2}\right)^3 = 216x^3 +  \frac{1}{8}$;
		\item $\left(x - 5y^2\right)\left(x^2 + 5xy^2 + 25y^4\right) = x^3 - \left(5y^2\right)^3 = x^3 - 125y^6$.
	\end{listEX}
}
\end{vd}
\begin{vd}
	Viết các đa thức sau dưới dạng bình phương hay lập phương của một tổng hay hiệu
	\begin{listEX}[2]
		\item $25x^2 - 5xy + \dfrac{1}{4} y^2$;
		\item $8x^3 - 12x^2y + 6xy^2 - y^3$.
	\end{listEX} 
\loigiai{
	\begin{listEX}
		\item $25x^2 - 5xy + \dfrac{1}{4} y^2 = \left(5x\right)^2 - 2\cdot\left(5x\right) \cdot \left( \dfrac{1}{2}y\right) + \left( \dfrac{1}{2}y\right)^2 = \left(5x - \dfrac{1}{2}\right)^2$;
		\item $8x^3 - 12x^2y + 6xy^2 - y^3  = \left(2x\right)^3 - 3 \cdot \left(2x\right)^2 \cdot y + 3 \cdot (2x) \cdot \left(y^2\right) - y^3 = (2x-y)^3$.
	\end{listEX}
}	
\end{vd}
\begin{vd}
	Điền các  đơn thức thích hợp vào ô trống
	\begin{listEX}
		\item $\left(x - \dfrac{1}{x}\right)^2 = x^2 - \boxEX[1cm]{$2$} + \dfrac{1}{x^2}$;
		\item $\left( \dfrac{1}{2} + \boxEX[1cm]{$\dfrac{1}{3} y$} \right)\left( \dfrac{1}{4}x^2 - \boxEX[1cm]{$\dfrac{1}{6} xy$} + \dfrac{1}{9} y^2\right) = \dfrac{1}{8} x^3 + \dfrac{1}{27} y^3$.
	\end{listEX}
\end{vd}
\begin{dang}[Rút gọn và tính giá trị của biểu thức]
\begin{enumerate}[\tickEX]
	\item Vận dụng các hằng đẳng thức đáng nhớ để khai triển các lũy thừa, khai triển các tích rồi rút gọn.
	\item Thay các giá trị của biến vào biểu thức đã rút gọn rồi thực hiện phép các phép tính.
\end{enumerate}
\end{dang}
\begin{vd}
	Rút gọn các biểu thức
	\begin{listEX}
		\item $(7x + 4)^2 - (7x + 4)(7x - 4)$;
		\item $(x + 2y)^3 - 6xy(x + 2y)$;
		\item $(3x + y)\left(9x^2 - 3xy + y^2\right) - (3x - y)^3 - 27x^2y$.
	\end{listEX}
\loigiai{
\begin{listEX}
	\item Ta có {\allowdisplaybreaks\begin{eqnarray*}
		(7x + 4)^2 - (7x + 4)(7x - 4) &=& 49x^2 + 56x + 16 - \left(49x^2 - 16\right)\\
		&=& 49x^2 + 56x + 16 - 49x^2 + 16\\
		&=& 56x + 32.
	\end{eqnarray*}}
	\item Ta có {\allowdisplaybreaks\begin{eqnarray*}
	(x + 2y)^3 - 6xy(x + 2y) &=& x^3 + 6x^2y + 12xy^2 + 8y^3 - 6x^2y - 12xy^2\\
	&=&x^3 + 8y^3.
	\end{eqnarray*}}
	\item Ta có {\allowdisplaybreaks\begin{eqnarray*}
		&& (3x + y)\left(9x^2 - 3xy + y^2\right) - (3x - y)^3 - 27x^2y \\
		&=& 27x^3 + y^3 - \left(27x^3 - 27x^2y + 9xy^2 - y^3\right) - 27x^2y\\
		&=& 27x^3 + y^3 - 27x^3 + 27x^2y - 9xy^2 + y^3 - 27x^2y\\
		&=& 2y^3 - 9xy^2.
	\end{eqnarray*}}
\end{listEX}
}
\end{vd}
\begin{vd}
Cho biểu thức $A = 5(x+3)(x-3) + (2x + 3)^2 + (x-6)^2$. Rút gọn rồi tính giá trị của biểu thức $A$ với $x = - \dfrac{1}{5}$.
\loigiai{
Ta có
	{\allowdisplaybreaks\begin{eqnarray*}
		A &=& 5(x+3)(x-3) + (2x + 3)^2 + (x-6)^2\\
		&=& 5\left(x^2 - 9\right) + \left(4x^2 + 12x + 9\right) + \left(x^2 - 12x + 36\right)\\
		&=& 5x^2 - 45 + 4x^2 + 12x + 9 + x^2 - 12x + 36\\
		&=& 10x^2.
	\end{eqnarray*}}
Với $x = - \dfrac{1}{5}$ thì $A = 10 \cdot \left(- \dfrac{1}{5}\right)^2 = 10 \cdot \dfrac{1}{25} = \dfrac{2}{5}$.
}
\end{vd}
\begin{vd}
Cho biết $x + y = 15$ và $xy = -100$. Tính giá trị của biểu thức $B = x^2 + y^2$.
\loigiai{
Ta có
	{\allowdisplaybreaks\begin{eqnarray*}
		B &=& x^2 + y^2 + 2xy - 2xy\\
		&=& (x+y)^2 -2xy\\
		&=& 15^2 - 2 \cdot (-100) = 425.
	\end{eqnarray*}}
}	
\end{vd}
\begin{vd}
Tính nhanh giá trị của biểu thức
\begin{listEX}[2]
	\item $C = 39^2 + 78 \cdot 61 + 61^2$;
	\item $D = 50^2 - 49 \cdot 51$.
\end{listEX}
\loigiai{
	\begin{listEX}
	\item Ta có {\allowdisplaybreaks\begin{eqnarray*}
			C &=& 39^2 + 78 \cdot 61 + 61^2\\
			&=& (39 + 61)^2\\
			&=& 100^2 = 10000.
		\end{eqnarray*}}
	\item Ta có {\allowdisplaybreaks\begin{eqnarray*}
		D &=& 50^2 - 49 \cdot 51\\
		&=& 50^2 - (50-1)(50+1)\\
		&=& 50^2 - (50^2 -1)\\
		&=& 1.
	\end{eqnarray*}}
	\end{listEX}	
}
\end{vd}
\begin{dang}[Chứng minh giá trị của biểu thức không phụ thuộc vào các biến]
	Vận dụng các hằng đẳng thức để biến đổi biểu thức đã cho thành một biểu thức không chứa biến.
\end{dang}
\begin{vd}
	Chứng minh giá trị của biểu thức sau không phụ thuộc vào giá trị của biến 
	\[A = (3x + 2)\left(9x^2 - 6x + 4\right)\left(9x^3 -2\right).\]
\loigiai{
Ta có
	{\allowdisplaybreaks\begin{eqnarray*}
		A &=& (3x + 2)\left(9x^2 - 6x + 4\right)-3\left(9x^3 -2\right)\\
		&=& 27x^3 + 8 - 27x^3 + 6\\
		&=& 14.
	\end{eqnarray*}}
Vậy giá trị của biểu thức $A$ không phụ thuộc vào giá trị của biến.
}
\end{vd}
\begin{vd}
	Giá trị của biểu thức sau có phụ thuộc vào giá trị của biến không?
	\[B = (x+1)^3 - (x-1)\left(x^2 + x +1 \right) - 3x(x+1).\]
\loigiai{
Ta có
	{\allowdisplaybreaks\begin{eqnarray*}
		B &=& (x+1)^3 - (x-1)\left(x^2 + x +1 \right) - 3x(x+1)\\
		&=& x^3 + 3x^2 + 3x +1 - \left(x^3 -1\right) - 3x^2 - 3x\\
		&=& 2.
	\end{eqnarray*}}
Vậy giá trị của biểu thức $B$ không phụ thuộc vào biến.
}
\end{vd}
\begin{dang}[Chứng minh đẳng thức]
	Vận dụng các hằng đẳng thức để biến đổi vế này thành vế kia hoặc biến đổi hai vế về cùng bằng một biểu thức.
\end{dang}
\begin{vd}
	Chứng minh đẳng thức
	\[(x+y)^2 - (x-y)^2 = 4xy.\]
\loigiai{
	Biến đổi vế trái (\text{VT}) ta được
	{\allowdisplaybreaks\begin{eqnarray*}
		VT &=& (x+y)^2 - (x-y)^2\\
		&=& x^2 + 2xy + y^2 - \left(x^2 - 2xy + y^2\right)\\
		&=& 4xy.
	\end{eqnarray*}}
Ta thấy vế trái đúng bằng vế phải nên đẳng thức đã cho đúng. Ta cũng có thể dùng hằng đẳng thức số $3$ để biến đổi vế trái.
}
\end{vd}
\begin{vd}
	Chứng minh đẳng thức
	\[3\left(x^2 + y^2 +z^2\right) - (x-y)^2 - (y-z)^2 - (z-x)^2 = (x+y+z)^2.\]
\loigiai{
Biến đổi vế trái (\text{VT}) ta được
	{\allowdisplaybreaks\begin{eqnarray*}
		\text{VT} &=& 3\left(x^2 + y^2 +z^2\right) - (x-y)^2 - (y-z)^2 - (z-x)^2\\
		&=& 3x^2 + 3y^2 + 3z^2 - (x-y)^2 - (y-z)^2 - (z-x)^2\\
		&=& 3x^2 + 3y^2 + 3z^2 - (x^2 - 2xy + y^2) - (y^2 - 2yz + z^2) - (z^2 -2zx + x^2)\\
		&=& 3x^2 + 3y^2 + 3z^2 - x^2 + 2xy - y^2 - y^2 + 2yz - z^2 - z^2 + 2zx - x^2\\
		&=& x^2 + y^2 + z^2 + 2xy + 2xz + 2yz.
	\end{eqnarray*}}
Biến đổi vế phải (\text{VP}) ta được
{\allowdisplaybreaks\begin{eqnarray*}
	\text{VP} &=& (x+y+z)^2 \\
	&=& [(x+y)+z]^2\\
	&=& (x+y)^2 + 2z(x+y) + z^2\\
	&=& x^2 + 2xy + y^2 + 2xz + 2yz + z^2\\
	&=& x^2 + y^2 + z^2 + 2xy + 2xz + 2yz.
\end{eqnarray*}}
Ta thấy vế trái $\text{VT}$ đúng bằng vế $\text{VP}$. Vậy đẳng thức đã cho là đúng.
\begin{note}
	Bạn nên nhớ kết quả $(x+y+z)^2 = x^2 + y^2 + z^2 + 2xy + 2xz + 2yz$ để vận dụng tính nhanh kết quả.
\end{note}
}
\end{vd}
\begin{dang}[Tìm $x$ thỏa mãn đẳng thức]
	\begin{enumerate}[\tickEX]
		\item Vận dụng các hằng đẳng thức đáng nhớ để khai triển ra rồi thu gọn về dạng $ax = b$.
		\item Suy ra $x = \dfrac{b}{a}$ nếu $a \neq 0, \forall x \in \mathbb{R}$ nếu $a = b = 0,$ không có $x$ nếu $a = 0, b \neq 0$.
	\end{enumerate}
\end{dang}
\begin{vd}
	Tìm $x$ biết rằng \[(2x+1)(1-2x)+(2x-1)^2 = 22.\]
	\loigiai{
	Ta có
		{\allowdisplaybreaks\begin{eqnarray*}
			(2x+1)(1-2x)+(2x-1)^2 &=& 22\\
			1 - 4x^2 + 4x^2 - 4x + 1 &=& 22\\
			-4x + 2 &=& 22\\
			-4x &=& 20\\
			x &=& -5.
		\end{eqnarray*}}
	}
\begin{note}
	Khi vận dụng hằng đẳng thức $(a+b)(a-b) = a^2 - b^2$ để tính $(2x+1)(1-2x)$ thì $b$ là $2x$ chứ không phải là $1$.
\end{note}
\end{vd}
\begin{vd}
	Tìm $x$ biết rằng \[(x-5)^2 + (x-3)(x+3) -2(x+1)^2 = 0.\]
	\loigiai{
	Ta có
		{\allowdisplaybreaks\begin{eqnarray*}
			(x-5)^2 + (x-3)(x+3) -2(x+1)^2 &=& 0\\
			\left(x^2 - 10x + 25\right) + \left(x^2 - 9\right) - 2\left(x^2 + 2x +1\right) &=& 0\\
			x^2 - 10x + 25 + x^2 - 9 - 2x^2 - 4x - 2 &=& 0\\
			-14x + 14 &=& 0\\
			-14x &=& 14\\
			x &=& -1.
		\end{eqnarray*}}
	}
\end{vd}
\begin{dang}[Chứng minh chia hết]
	Vận dụng các hằng đẳng thức đáng nhớ để biến đổi số đã cho về dạng $a = k \cdot b$ $(k\neq 0)$. Lúc đó $a \,\vdots\, k$. 
\end{dang}
\begin{vd}
	Chứng minh rằng hiệu các bình phương của hai số chẵn liên tiếp thì chia hết cho $4$.
	\loigiai{
		Gọi hai số chẵn liên tiếp là $2a$ và $2a+2$ $(a \in \mathbb{Z})$. Hiệu bình phương của chúng là 
		{\allowdisplaybreaks\begin{eqnarray*}
			(2a+2)^2 - (2a)^2 &=& 4a^2 + 8a + 4 - 4a^2\\
			&=& 8a + 4 = 4(2a + 1) \,\vdots\, 4.
		\end{eqnarray*}}
	}
\end{vd}
\begin{dang}[Chứng minh giá trị của một biểu thức luôn luôn dương (hay âm) với mọi giá trị của biến]
	\begin{enumerate}[\tickEX]
		\item Muốn chứng minh giá trị của một biểu thức luôn luôn dương với mọi giá trị của biến, ta vận dụng các hằng đẳng thức $A^2 \pm 2AB^2 + B^2 = (A \pm B)^2$, để biến đổi biểu thức về dạng $\left[f(x)\right]^2 + k$ với $k > 0$.
		\item Muốn chứng minh giá trị của một biểu thức luôn luôn âm với mọi giá trị của biến, ta biến đổi biểu thức về dạng $- \left[f(x)\right]^2 + k$ với $k < 0$.
	\end{enumerate}
\end{dang}
\begin{vd}

\begin{dang}[Tìm giá trị nhỏ nhất, giá trị lớn nhất của biểu thức \newline $P(x) = ax^2 + bx +c$]
	\begin{enumerate}[\tickEX]
		\item Muốn tìm giá trị nhỏ nhất của biểu thức $P(x)$, ta vận dụng các hằng đẳng thức $A^2 \pm 2AB + B^2 = (A+B)^2$ để biến đổi $P(x)$ về dạng $[f(x)]^2 + k$ ($k$ là hằng số). Vì $[f(x)]^2 \geq 0$ nên $P(x) \geq k$. Do đó giá trị nhỏ nhất của $P(x)$ là $k$ (ta phải tìm $x$ để $f(x) = 0$). Ta viết $\min P(x) =k$.
		\item Muốn tìm giá trị lớn nhất của biểu thức $P(x)$, ta biến đổi $P(x)$ về dạng $-[f(x)]^2 + k$ ($k$ là hằng số). Vì $[f(x)]^2 \leq 0$ nên $P(x) \leq k$. Do đó giá trị lớn nhất của $P(x)$ là $k$ (ta phải tìm $x$ để $f(x) = 0$). Ta viết $\max P(x) =k$.

\end{dang}

		\item Cho các số thực $a$, $b$, $c$ thỏa mãn $a+b+c = 0$. Chứng minh rằng $a^3+b^3+c^3 = 3abc$.
		\item Giả sử $x,y,z$ là các số thực thỏa mãn $x+y+z  =3$. Tìm giá trị nhỏ nhất của biểu thức \[M = x^4+y^4+z^4 +12(1-x)(1-y)(1-z).\]
	\end{listEX}
	\loigiai{
		\begin{listEX}
			\item Sử dụng hằng đẳng thức $(A+B)^3 = A^3+3A^2B+3AB^2+B^3 = A^3+B^3 + 3AB(A+B)$. Ta có
			{\allowdisplaybreaks\begin{eqnarray*}
				(a+b+c)^3 &=& (a+b)^3+c^3+3c(a+b)(a+b+c) \\
				&=& a^3+b^3+3ab(a+b)+c^3 + 0\\
				& =& a^3+b^3+3ab(-c)+c^3.
			\end{eqnarray*}}
			Suy ra $a^3+b^3+c^3 = 3abc$.
			\item Đặt $a = 1-x$, $b=1-y$, $c = 1-z$ ta có $a+b+c = 0$. Ngoài ra
			{\allowdisplaybreaks\begin{eqnarray*}
				M &=& (1-a)^4 + (1-b)^4 + (1-c)^4 + 12abc \\
				&=& a^4+b^4+c^4 - 4(a^3+b^3+c^3) + 6(a^2+b^2+c^2) - 4(a+b+c) + 3 + 4(a^3+b^3+c^3) \\
				&=& a^4+b^4+c^4 + 6(a^2+b^2+c^2) + 3 \\
				&\geq& 3.
			\end{eqnarray*}}
			Suy ra giá trị nhỏ nhất của $M$ là $3$, đạt được khi $a=b=c=0$ hay $x=y=z=1$.
		\end{listEX}
	}
\end{vn}

				&=& \left[ (a+b)+c\right]^3 -\left(a^3+b^3+c^3\right)\\
				&=& (a+b)^3+c^3 +3c(a+b)(a+b+c) -\left(a^3+b^3+c^3\right)\\

				&=& 3(a+b)\left[(ab+c(a+b+c)\right]\\
				&=& 3(a+b)(ab+ac+bc+c^2)\\
				&=& 3(a+b)(a+c)(b+c).		
			\end{eqnarray*}}
			Suy ra điều phải chứng minh.
			\item 
			Áp dụng kết quả câu $a)$ ta có phương trình đã cho tương đương với

			\end{eqnarray*}}
		\end{enumerate}
