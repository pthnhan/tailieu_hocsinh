\section{Sử dụng công thức diện tích để thiết lập quan hệ về độ dài của các đoạn thẳng}
	Các công thức diện tích cho ta các quan hệ về độ dài của các đoạn thẳng, chúng rất có ích để giải nhiều bài toán.
\subsection{Các ví dụ}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-L8]%[8H2K3]
	Cho tam giác đều $ABC$.
	\begin{enumerate}
		\item Chứng minh rằng nếu điểm $M$ thuộc miền trong của tam giác $ABC$ thì tổng các khoảng cách từ $M$ đến ba cạnh của tam giác bằng chiều cao của tam giác.
		\item Quan hệ trên thay đổi như thế nào nếu điểm $M$ thuộc miền ngoài của tam giác?
	\end{enumerate}
	\loigiai{
		Gọi $a$ và $h$ lần lượt là cạnh và chiều cao của tam giác $ABC$, $MA'$, $MB'$, $MC'$ là các khoảng cách từ $M$ đến $BC$, $AC$, $AB$.
		\immini{
			\begin{enumerate}
			\item Nếu $M$ thuộc miền trong tam giác thì
					\begin{eqnarray*}
						& & S_{MBC} + S_{MAC} + S_{MAB} = S_{ABC}\\
						& \Rightarrow & \dfrac{1}{2}BC \cdot MA' + \dfrac{1}{2}AC \cdot MB' + \dfrac{1}{2}AB \cdot MC' = \dfrac{1}{2}BC \cdot AH\\
						& \Rightarrow & \dfrac{a}{2}(MA' + MB' + MC') = \dfrac{a}{2} \cdot h\\
						& \Rightarrow & MA' + MB' + MC' = h.
					\end{eqnarray*}
			\end{enumerate}
		}{
			\begin{tikzpicture}[scale=.9]
				\clip (-.5,-.5) rectangle (4.5,4.5);	
				\tkzDefPoints{0/0/B, 4/0/C, 2.8/1/M}%Định nghĩa điểm
				\tkzDefTriangle[equilateral](B,C) \tkzGetPoint{A}
				\tkzDrawPolygon(A,B,C)
				\tkzDrawAltitude(B,C)(A) \tkzGetPoint{H}
				\tkzDrawAltitude(B,C)(M) \tkzGetPoint{A'}
				\tkzDrawAltitude(A,C)(M) \tkzGetPoint{B'}
				\tkzDrawAltitude(A,B)(M) \tkzGetPoint{C'}
				\tkzDrawSegments[dashed](M,A M,B M,C)%Vẽ đoạn thẳng nét đứt			
				\tkzDrawPoints(A,B,C,M,H,A',B',C')%Vẽ điểm
				\tkzLabelPoints[above](A)
				\tkzLabelPoints[below](H,A')
				\tkzLabelPoints[above left](C')
				\tkzLabelPoints[above right](B')
				\tkzLabelPoints[below left](B,M)
				\tkzLabelPoints[below right](C)			
				\tkzMarkRightAngles(A,H,C M,A',C M,B',C M,C',B)%Vẽ góc vuông			
			\end{tikzpicture}
		}
		\immini{
			\begin{enumerate}
				\item[b)] Nếu $M$ thuộc miền ngoài tam giác $ABC$ và thuộc miền trong góc $A$ (miền 2) thì
						\begin{eqnarray*}
							& & S_{MAC} + S_{MAB} - S_{MBC} = S_{ABC}\\
							& \Rightarrow & \dfrac{a}{2}\cdot MB' + \dfrac{a}{2}\cdot MC' - \dfrac{a}{2}\cdot MA' = \dfrac{a}{2}\cdot h\\
							& \Rightarrow & MB' + MC' - MA' = h.
						\end{eqnarray*}
						Tương tự
						\begin{itemize}
							\item Nếu $M$ thuộc miền ngoài tam giác $ABC$ và thuộc miền trong góc $B$ (miền 3) thì
									$$MA' + MC' - MB' = h.$$		
						\end{itemize}
			\end{enumerate}
		}{
			\begin{tikzpicture}[scale=.8]
				\clip (-1.5,-3) rectangle (6,5.5);	
				\tkzDefPoints{0/0/B, 4/0/C, 2.6/-2/M}%Định nghĩa điểm
				\tkzDefTriangle[equilateral](B,C) \tkzGetPoint{A}				
				\tkzDrawLine[add = 0.5 and 0.5](A,B)
				\tkzDrawLine[add = 0.5 and 0.5](B,C)
				\tkzDrawLine[add = 0.5 and 0.5](C,A)				
				\tkzDrawAltitude(B,C)(M) \tkzGetPoint{A'}
				\tkzDrawAltitude(A,C)(M) \tkzGetPoint{B'}
				\tkzDrawAltitude(A,B)(M) \tkzGetPoint{C'}
				\tkzDrawSegments[dashed](M,A M,B M,C)%Vẽ đoạn thẳng nét đứt			
				\tkzDrawPoints(A,B,C,M,A',B',C')%Vẽ điểm
				\tkzLabelPoints[above](A')
				\tkzLabelPoints[below](M)
				\tkzLabelPoints[left](A,C')
				\tkzLabelPoints[right](B')
				\tkzLabelPoints[above left](B)
				\tkzLabelPoints[above right](C)	
				\tkzMarkRightAngles(M,A',C M,B',C M,C',B)%Vẽ góc vuông
				\draw (1.8,1.8)node{$1$} (1,-2)node{$2$} (4.5,2)node{$3$} (-.5,2)node{$4$} (2,4.5)node{$5$} (-1,-1)node{$6$} (5.8,-1)node{$7$};	
			\end{tikzpicture}
		}
		\begin{enumerate}
			\item[] \begin{itemize}
							\item Nếu $M$ thuộc miền ngoài tam giác $ABC$ và thuộc miền trong góc $C$ (miền 4) thì
									$$MA' + MB' - MC' = h.$$	
							\item Nếu $M$ thuộc miền trong góc đối đỉnh với góc $A$ (miền 5) thì
									$$MA' - MB' - MC' = h.$$
							\item Nếu $M$ thuộc miền trong góc đối đỉnh với góc $B$ (miền 6) thì
									$$MB' - MA' - MC' = h.$$
							\item Nếu $M$ thuộc miền trong góc đối đỉnh với góc $C$ (miền 7) thì
									$$MC' - MA' - MB' = h.$$
						\end{itemize} 
		\end{enumerate}
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-L8]%[8H2K3]
	Các điểm $E$, $F$ nằm trên các cạnh $AB$, $BC$ của hình bình hành $ABCD$ sao cho $AF = CE$. Gọi $I$ là giao điểm của $AF$, $CE$. Chứng minh rằng $ID$ là tia phân giác của góc $AIC$.
	\loigiai{
		\immini{
			Kẻ $DH \perp IA$ tại $H$, $DK \perp IC$ tại $K$.\\
			Ta có $S_{AFD} = S_{CED} = \dfrac{1}{2}S_{ABCD}$\\
			$\Rightarrow \dfrac{1}{2}DH \cdot AF = \dfrac{1}{2}DK \cdot CE$\\
			$\Rightarrow DH = DK$ (Vì $AF = CE$).\\
			$\Rightarrow D$ thuộc tia phân giác của góc $AIC$.\\
			Vậy $ID$ là tia phân giác của góc $AIC$.
		}{
			\begin{tikzpicture}[scale=.9]	
				\clip (-1.5,-4) rectangle (4.5,1);	
				\tkzDefPoints{0/0/A, 4/0/B, -1/-3.5/D}%Định nghĩa điểm
				\tkzDefPointBy[translation = from A to B](D) \tkzGetPoint{C}%Phép tịnh tiến
				\tkzInterLC[R](A,B)(C,3.9 cm) \tkzGetPoints{E}{E'}
				\tkzInterLC[R](B,C)(A,3.9 cm) \tkzGetPoints{F'}{F}
				\tkzInterLL(A,F)(C,E) \tkzGetPoint{I}
				\tkzDrawPolygon(A,B,C,D)
				\tkzDrawSegments(A,F C,E D,I)%Vẽ đoạn thẳng nét liền
				\tkzDrawAltitude(A,F)(D) \tkzGetPoint{H}
				\tkzDrawAltitude(C,E)(D) \tkzGetPoint{K}	
				\tkzDrawPoints(A,B,C,D,E,F,I,H,K)%Vẽ điểm
				\tkzLabelPoints[above](E)
				\tkzLabelPoints[below](H)
				\tkzLabelPoints[right](K,F)
				\tkzLabelPoints[above left](A)
				\tkzLabelPoints[above right](B,I)
				\tkzLabelPoints[below left](D)
				\tkzLabelPoints[below right](C)			
				\tkzMarkRightAngles(D,H,A D,K,I)%Vẽ góc vuông			
			\end{tikzpicture}
		}
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-L8]%[8H2K3]
	Cho tam giác $ABC$ có $\widehat{A} \geq 90^\circ$, $D$ là điểm nằm giữa $A$ và $C$. Chứng minh rằng tổng các khoảng cách từ $A$ và từ $C$ đến $BD$ lớn hơn đường cao kẻ từ $A$ và nhỏ hơn đường cao kẻ từ $C$ của tam giác $ABC$.
	\loigiai{
		\immini{
			Gọi $AH$, $CK$ là các đường cao của tam giác $ABC$. Kẻ $AE$ và $CF$ vuông góc với $BD$. Đặt $S_{ABC} = S$. Ta có $AE = \dfrac{2S_{ABD}}{BD}$, $CF = \dfrac{2S_{CBD}}{BD}$ nên
			$$AE + CF = \dfrac{2S}{BD}.$$			
		}{
			\begin{tikzpicture}[scale=.8]	
				\clip (-.5,-1) rectangle (8,3.5);
				\tkzDefPoints{0/0/B, 7/0/C, 1.5/2.5/A}%Định nghĩa điểm
				\coordinate (D) at ($(A)!.3!(C)$);
				\tkzDrawAltitude(B,C)(A) \tkzGetPoint{H}
				\tkzDrawAltitude(B,D)(A) \tkzGetPoint{E}
				\tkzDrawAltitude(A,B)(C) \tkzGetPoint{K}
				\tkzDrawAltitude(B,D)(C) \tkzGetPoint{F}
				\tkzDrawSegments(B,K B,C C,A B,F)%Vẽ đoạn thẳng nét liền
				\tkzDrawPoints(A,B,C,D,E,F,H,K)%Vẽ điểm
				\tkzLabelPoints[above](F)
				\tkzLabelPoints[below](D,H)
				\tkzLabelPoints[left](A,K)
				\tkzLabelPoints[below left](B)
				\tkzLabelPoints[below right](E,C)			
				\tkzMarkRightAngles(A,H,C A,E,D B,K,C B,F,C)%Vẽ góc vuông			
			\end{tikzpicture}
		}
		\noindent Ta lại có $AH = \dfrac{2S}{BC}$, $CK = \dfrac{2S}{BA}$.\\
		Do $\widehat{A} \geq 90^\circ$ nên $BA < BD < BC$, do đó $AH < AE + CF < CK$.
	}
\end{vd}

\subsection{Bài tập}
\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-L8]%[8H2K3]
	Có tam giác nào mà độ dài ba đường cao bằng $3$ cm, $4$ cm, $7$ cm không?
	\loigiai{
		Giả sử tồn tại tam giác có độ dài ba cạnh lần lượt là $a$, $b$, $c$ thỏa mãn yêu cầu bài toán.\\
		Khi đó, ta có $3a = 4b = 7c$ (cùng bằng hai lần diện tích tam giác).\\
		Ta có $3a = 4b = 7c \Leftrightarrow \dfrac{a}{\frac{1}{3}} = \dfrac{b}{\frac{1}{4}} = \dfrac{c}{\frac{1}{7}} = \dfrac{a + b + c}{\frac{1}{3} + \frac{1}{4} + \frac{1}{7}} = \dfrac{84P}{61}$.\\
		Suy ra $a = \dfrac{28P}{61}$, $b = \dfrac{21P}{61}$, $c = \dfrac{12P}{61}$.\\
		Do đó, ta có $a + b > c$, $b + c > a$, $c + a > b$ hay $a$, $b$, $c$ là độ dài ba cạnh của một tam giác.\\
		Vậy tồn tại tam giác mà độ dài ba đường cao bằng $3$ cm, $4$ cm, $7$ cm.
	}
\end{bt}

\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-L8]%[8H2K3]
	Độ dài hai cạnh của một tam giác bằng $6$ cm và $4$ cm, nửa tổng các chiều cao ứng với hai cạnh ấy bằng chiều cao ứng với cạnh thứ ba. Tính độ dài cạnh thứ ba.
	\loigiai{
		Giả sử tam giác có độ dài ba cạnh lần lượt là $a$, $b = 6$ cm, $c = 4$ cm và $h_a$, $h_b$, $h_c$ lần lượt là độ dài ba đường cao tương ứng của tam giác. Khi đó, ta có
		$$ah_a = bh_b = ch_c \Rightarrow \dfrac{a(h_b + h_c)}{2} = 6h_b = 4h_c \Rightarrow \dfrac{h_b + h_c}{\frac{2}{a}} = \dfrac{h_b}{\frac{1}{6}} = \dfrac{h_c}{\frac{1}{4}}.$$
		Theo tích chất của dãy tỉ số bằng nhau, ta có $\dfrac{2}{a} = \dfrac{1}{6} + \dfrac{1}{4} \Rightarrow a = 4{,}8$ cm.\\
		Vậy độ dài cạnh thứ ba của tam giác bằng $4{,}8$ cm.
	}
\end{bt}

\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-L8]%[8H2K3]
	Chứng minh rằng một tam giác là tam giác vuông nếu các chiều cao $h_a$, $h_b$, $h_c$ của nó thỏa mãn điều kiện $\left(\dfrac{h_a}{h_b}\right)^2 + \left(\dfrac{h_a}{h_c}\right)^2 = 1$.
	\loigiai{
		Gọi $S$ là diện tích của tam giác, ta có
		$$\left(\dfrac{h_a}{h_b}\right)^2 + \left(\dfrac{h_a}{h_c}\right)^2 = 1 \Rightarrow \left(\dfrac{\dfrac{2S}{a}}{\dfrac{2S}{b}}\right)^2 + \left(\dfrac{\dfrac{2S}{a}}{\dfrac{2S}{c}}\right)^2 = 1 \Rightarrow \left(\dfrac{b}{a}\right)^2 + \left(\dfrac{c}{a}\right)^2 = 1 \Rightarrow b^2 + c^2 = a^2.$$
		Suy ra tam giác đã cho là tam giác vuông.
	}
\end{bt}

\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-L8]%[8H2K3]
	Tính các cạnh của một tam giác có ba đường cao bằng $12$ cm, $15$ cm, $20$ cm.
	\loigiai{
		Giả sử tam giác $ABC$ có độ dài ba đường cao là $h_a = 12$ cm, $h_b = 15$ cm, $h_c = 20$ cm tương ứng với ba cạnh $a$, $b$, $c$. Khi đó, ta có
		$$\left(\dfrac{b}{a}\right)^2 + \left(\dfrac{c}{a}\right)^2 = \left(\dfrac{h_a}{h_b}\right)^2 + \left(\dfrac{h_a}{h_c}\right)^2 = \left(\dfrac{12}{15}\right)^2 + \left(\dfrac{12}{20}\right)^2 = 1.$$
		$\Rightarrow b^2 + c^2 = 1 \Rightarrow$ tam giác $ABC$ vuông tại $A$.\\
		Do đó, ta có $AB = 15$ cm, $AC = 20$ cm và $BC = \sqrt{AB^2 + AC^2} = \sqrt{15^2 + 20^2} = 25$ cm.
	}
\end{bt}

\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-L8]%[8H2K3]
	Gọi $h_a$, $h_b$, $h_c$ là ba đường cao của một tam giác, chứng minh rằng $\dfrac{1}{h_a} < \dfrac{1}{h_b} + \dfrac{1}{h_c}$.
	\loigiai{
		Gọi $a$, $b$, $c$ là ba cạnh tương ứng với các đường cao $h_a$, $h_b$, $h_c$ và $S$ là diện tích tam giác. Ta có
		$$a < b + c \Rightarrow \dfrac{2S}{h_a} < \dfrac{2S}{h_b} + \dfrac{2S}{h_c} \Rightarrow \dfrac{1}{h_a} < \dfrac{1}{h_b} + \dfrac{1}{h_c}.$$
	}
\end{bt}

\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-L8]%[8H2K3]
	Tam giác $ABC$ có độ dài các cạnh là $a$, $b$, $c$, các chiều cao tương ứng là $h_a$, $h_b$, $h_c$. Biết rằng $a + h_a = b + h_b = c + h_c$. Chứng minh rằng tam giác $ABC$ là tam giác đều.
	\loigiai{
		Xét $a + h_a = b + h_b$, ta có
		\begin{eqnarray*}
			& & a - b = h_b - h_a = \dfrac{2S}{b} - \dfrac{2S}{a} = 2S\left(\dfrac{1}{b} - \dfrac{1}{a}\right) = 2S \cdot \dfrac{a - b}{ab}\\
			& \Rightarrow & (a - b)\left(1 - \dfrac{2S}{ab}\right) = 0\\
			& \Rightarrow & a = b \,\,\text{hoặc} \,\,S = \dfrac{1}{2}ab.
		\end{eqnarray*}
		Suy ra tam giác $ABC$ cân tại $C$ hoặc vuông tại $C$. \hfill $(1)$\\
		Tương tự, tam giác $ABC$ cân tại $B$ hoặc vuông tại $B$. \hfill $(2)$\\
		\phantom{Tương tự,} tam giác $ABC$ cân tại $A$ hoặc vuông tại $A$. \hfill $(3)$\\
		Xảy ra cả $(1)$, $(2)$ $\&$ $(3)$ khi và chỉ khi tam giác $ABC$ đều.
	}
\end{bt}

\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-L8]%[8H2G3]
	Cho điểm $O$ thuộc miền trong tam giác $ABC$. Các tia $AO$, $BO$, $CO$ cắt các cạnh của tam giác $ABC$ theo thứ tự ở $A'$, $B'$, $C'$. Chứng minh rằng:
	\begin{enumerate}
		\item $\dfrac{OA'}{AA'} + \dfrac{OB'}{BB'} + \dfrac{OC'}{CC'} = 1$;
		\item $\dfrac{OA}{AA'} + \dfrac{OB}{BB'} + \dfrac{OC}{CC'} = 1$ (bài toán của Giec-gôn, nhà toán học Pháp);
		\item $M = \dfrac{OA}{OA'} + \dfrac{OB}{OB'} + \dfrac{OC}{OC'} \geq 6$. Tìm vị trí của $O$ để tổng $M$ có giá trị nhỏ nhất;
		\item $N = \dfrac{OA}{OA'} \cdot \dfrac{OB}{OB'} \cdot \dfrac{OC}{OC'} \geq 8$. Tìm vị trí của $O$ để tích $N$ có giá trị nhỏ nhất
	\end{enumerate}
	\loigiai{
		\immini{
			Gọi $S$ là diện tích tam giác $ABC$, kí hiệu $S_1$, $S_2$, $S_3$ như hình vẽ. Ta có\\
			$\dfrac{OA}{OA'} = \dfrac{S_2}{S_{OA'C}} = \dfrac{S_3}{S_{OA'B}} = \dfrac{S_2 + S_3}{S_1}$, \hfill $(1)$\\
			$\dfrac{OA'}{AA'} = \dfrac{S_{OA'C}}{S_{AA'C}} = \dfrac{S_{OA'B}}{S_{AA'B}} = \dfrac{S_{OA'C} + S_{OA'B}}{S_{AA'C} + S_{AA'B}} = \dfrac{S_1}{S}$. \hfill $(2)$\\
			Từ $(1) \,\,\&\,\, (2)$ suy ra $\dfrac{OA'}{AA'} =  \dfrac{S_2 + S_3}{S}$.
		}{
			\begin{tikzpicture}[scale=.9]	
				\tkzDefPoints{0/0/B, 1/3/A, 4/0/C, 1.3/1/O}%Định nghĩa điểm
				\tkzInterLL(A,O)(B,C) \tkzGetPoint{A'}
				\tkzInterLL(B,O)(A,C) \tkzGetPoint{B'}
				\tkzInterLL(C,O)(A,B) \tkzGetPoint{C'}
				\tkzDrawPolygon(A,B,C)
				\tkzDrawSegments(O,A O,B O,C)%Vẽ đoạn thẳng nét liền		
				\tkzDrawPoints(A,B,C,A',B',C',O)%Vẽ điểm
				\tkzLabelPoints[above](A)
				\tkzLabelPoints[below](O,A')
				\tkzLabelPoints[left](C')
				\tkzLabelPoints[right](B')
				\tkzLabelPoints[below left](B)
				\tkzLabelPoints[below right](C)
				\draw (2,.4)node{$1$} (.8,1.4)node{$3$} (2,1.4)node{$2$};			
			\end{tikzpicture}
		}
		\begin{enumerate}
			\item Ta có $\dfrac{OA'}{AA'} + \dfrac{OB'}{BB'} + \dfrac{OC'}{CC'} = \dfrac{S_1}{S} + \dfrac{S_2}{S} + \dfrac{S_3}{S} = 1$.
			\item Ta có $\dfrac{OA}{AA'} + \dfrac{OB}{BB'} + \dfrac{OC}{CC'} = \dfrac{S_2 + S_3}{S} + \dfrac{S_3 + S_1}{S} + \dfrac{S_1 + S_2}{S} = 2$.
			\item Ta có $M = \dfrac{OA}{OA'} + \dfrac{OB}{OB'} + \dfrac{OC}{OC'}$\\
					\phantom{Ta có $M$} $= \dfrac{S_2 + S_3}{S_1} + \dfrac{S_3 + S_1}{S_2} + \dfrac{S_1 + S_2}{S_3}$\\
					\phantom{Ta có $M$} $= \left(\dfrac{S_1}{S_2} + \dfrac{S_2}{S_1}\right) + \left(\dfrac{S_2}{S_3} + \dfrac{S_3}{S_2}\right) + \left(\dfrac{S_3}{S_1} + \dfrac{S_1}{S_3}\right) \geq 2 + 2 + 2 = 6$.\\
					Đẳng thức xảy ra khi và chỉ khi $O$ là trọng tâm tam giác $ABC$.
			\item Ta có $N = \dfrac{OA}{OA'} \cdot \dfrac{OB}{OB'} \cdot \dfrac{OC}{OC'}$\\
					\phantom{Ta có $N$} $= \dfrac{(S_2 + S_3)(S_3 + S_1)(S_1 + S_2)}{S_1S_2S_3}$\\
					$\Rightarrow N^2 = \dfrac{(S_2 + S_3)^2(S_3 + S_1)^2(S_1 + S_2)^2}{(S_1S_2S_3)^2}$\\
					\phantom{$\Rightarrow$} $\geq \dfrac{4S_2S_3 \cdot 4S_1S_3 \cdot 4S_1S_2}{(S_1S_2S_3)^2} = 64$.\\
					Suy ra $N \geq 8$.\\
					Đẳng thức xảy ra khi và chỉ khi $O$ là trọng tâm tam giác $ABC$.
		\end{enumerate}
	}
\end{bt}
