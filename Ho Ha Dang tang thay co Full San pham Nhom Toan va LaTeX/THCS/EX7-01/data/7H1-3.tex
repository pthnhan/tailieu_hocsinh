\section{HAI ĐƯỜNG THẲNG SONG SONG}
\subsection{Tóm tắt lí thuyết}
\begin{enumerate}[\tickEX] 
	\item Hai đường thẳng được gọi là song song khi chúng không có điểm chung. Ký hiệu $a \parallel b.$
	\begin{center}
		\begin{tikzpicture}[line width=1pt]
		\tkzInit[xmin=-5.5,ymin=-0.5,xmax=5.5,ymax=1.5]
		\tkzClip
		\draw[line width = 1pt] (-5,0)--(5,0)node[midway, above]{$a$};
		\draw[line width = 1pt] (-5,1)--(5,1)node[midway, above]{$b$};
		\end{tikzpicture}
	\end{center}
	\item Cho đường thẳng $c$ cắt hai đường thẳng $a$ và $b$. Nếu trong các cặp góc tạo thành có một cặp góc so le trong bằng nhau thì
	\begin{enumerate}
		\item Cặp góc so le trong còn lại bằng nhau.
		\item Hai cặp góc đồng vị bằng nhau.
	\end{enumerate}
	\item  Dấu hiệu nhận biết hai đường thẳng song song. Nếu đường thẳng $c$ cắt hai đường thẳng $a, b$ và trong các góc tạo thành có
	\begin{enumerate}
		\item Hoặc một  cặp góc so le trong bằng nhau;
		\item Hoặc một cặp góc đồng vị bằng nhau;
		\item Hoặc một cặp góc trong cùng phía bù nhau.
	\end{enumerate}
\end{enumerate}
