\begin{vd}%[7H2K7]
	Một tam giác có độ dài ba đường cao là $ 4{,}8 $ cm, $ 6 $ cm và $ 8 $ cm có phải là tam giác vuông không?
	\loigiai{
		Gọi $ a $, $ b $, $ c $ là độ dài ba cạnh của tam giác ứng với các đường cao theo thứ tự đã cho, $ S $ là diện tích tam giác $ ABC $. Ta có
		\begin{align*}
			4{,}8a = 6b = 8c = 2S,
		\end{align*}
		do đó
		\begin{align*}
			a = \dfrac{2S}{4{,}8} = \dfrac{5}{12}S \quad ; \quad
			b = \dfrac{2S}{6} = \dfrac{1}{3}S \quad ; \quad
			c = \dfrac{2S}{8} = \dfrac{1}{4}S.
		\end{align*}
		Ta có
		\begin{align*}
			& b^2 + c^2 = \dfrac{1}{9}S^2 + \dfrac{1}{16}S^2 = \dfrac{25}{144} S^2\\
			& a^2 = \dfrac{25}{144}S^2.
		\end{align*}
		Rõ ràng $ a^2 = b^2 + c^2 $ nên tam giác đã cho là tam giác vuông, đỉnh góc vuông ứng với đường cao có độ dài $ 4{,}8 $ cm.
	}
\end{vd}

\begin{bt}%[7H2B7]
	Một tam giác vuông có độ dài hai cạnh góc vuông tỉ lệ với $ 3 $; $ 4 $. Biết cạnh huyền dài $ 55 $ cm. Tính độ dài hai cạnh góc vuông ấy.
	\loigiai{
		Gọi độ dài các cạnh góc vuông là $ b $ và $ c $. Ta có
		\begin{align*}
			\dfrac{b}{3} = \dfrac{c}{4} \text{ và } b^2 + c^2 = 55^2.
		\end{align*}
		Do đó
		\begin{align*}
			\left(\dfrac{b}{3}\right)^2 = \left(\dfrac{c}{4}\right)^2 = \dfrac{b^2 + c^2}{25} = \dfrac{55^2}{25} = 11^2.
		\end{align*}
		Suy ra $ b = 11 \cdot 3 = 33 $ (cm), $ c = 11 \cdot 4 = 44 $ (cm).
	}
\end{bt}

\begin{bt}%[7H2B7]
	Chứng minh rằng độ dài cạnh huyền của một tam giác vuông cân có cạnh góc vuông là số nguyên, luôn là một số vô tỷ.
	\loigiai{
		$ \triangle ABC $ vuông tại $ A $ có $ AB = AC = x $ ($ x $ nguyên). Theo định lý Py-ta-go ta có
		\begin{align*}
			BC^2 = AB^2 + AC^2  = x^2 + x^2  = 2x^2 \Rightarrow BC = x\sqrt{2}.
		\end{align*}
		$ \sqrt{2} $ là số vô tỉ. Vậy $ x \sqrt{2} $ là số vô tỉ.
	}
\end{bt}
