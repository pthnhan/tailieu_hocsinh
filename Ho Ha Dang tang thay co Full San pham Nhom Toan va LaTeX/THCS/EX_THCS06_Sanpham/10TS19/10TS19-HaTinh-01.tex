	Rút gọn các biểu thức sau
	\begin{enumerate}
		\item $P=\sqrt{45} -\sqrt{5}$.
		\item $Q=\left(1+\dfrac{2}{\sqrt{x} -2} \right):\dfrac{x}{\sqrt{x} -2} $ với $x>0$ và $x\ne 4$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $P=\sqrt{9\cdot 5}-\sqrt{5} =3\sqrt{5} -\sqrt{5} =2\sqrt{5}$.
			\item Ta có $Q=\dfrac{\sqrt{x} -2+2}{\sqrt{x} -2} \cdot\dfrac{\sqrt{x} -2}{x} =\dfrac{\sqrt{x}}{\sqrt{x} -2} \cdot\dfrac{\sqrt{x} -2}{x} =\dfrac{\sqrt{x}}{x} =\dfrac{1}{\sqrt{x}}$.
		\end{enumerate}
	}

\begin{ex}%[Đề Tuyển sinh vào lớp 10 năm học 2018-2019, Hà Tĩnh]%[Hồ Minh Hòa, dự án EXTHCS-06]%[9D4B2]%[9D4K6]
	\begin{enumerate}
		\item Xác định hệ số $a$ của hàm số $y=ax^{2} \left(a\ne 0\right)$, biết đồ thị của nó đi qua điểm $M\left(-\dfrac{1}{3} ;1\right)$.
		\item Cho phương trình $x^2 -2\left(m-1\right)x+m^2 -m=0$ ($m$ là tham số). Tìm giá trị của $m$ để phương trình đã cho có hai nghiệm $x_1$, $x_2$ thỏa mãn $\left(1+x_1 \right)^2 +\left(1+x_2 \right)^2 =6$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Đồ thị của hàm số đi qua điểm $M\left(-\dfrac{1}{3} ;1\right)$ nên $x=-\dfrac{1}{3} $; $y=1$ thay vào đẳng thức $y=ax^{2} $ được $a\left(-\dfrac{1}{3} \right)^{2} =1\Leftrightarrow a=9$.
			\item Để phương trình có hai nghiệm thì $\Delta '\ge 0\Rightarrow \left(m-1\right)^{2} -\left(m^{2} -m\right)\ge 0$
			
			\noindent  $\Leftrightarrow m^{2} -2m+1-m^{2} +m\ge 0\Leftrightarrow m\le 1.$ \\
			Theo định lý Vi-et thì $\begin{cases}x_1 +x_2 =2\left(m-1\right) \\ x_1x_2 =m^{2} -m.\end{cases}$
			
			Ta có $\left(1+x_{1} \right)^{2} +\left(1+x_{2} \right)^{2} =6$\\
			$\Leftrightarrow 1+2x_{1} +x_{1}^{2} +1+2x_{2} +x_{2}^{2} =6$\\
			$\Leftrightarrow \left(x_{1} +x_{2} \right)^{2} -2x_{1} x_{2} +2\left(x_{1} +x_{2} \right)=4$\\
			$\Rightarrow 4\left(m-1\right)^{2} -2\left(m^{2} -m\right)+4\left(m-1\right)=4$ \\
			$\Leftrightarrow 4m^{2} -8m+4-2m^{2} +2m+4m-4=4 $\\
			$\Leftrightarrow 2m^{2} -2m-4=0$
			$\Leftrightarrow m^{2} -m-2=0$ \\
			$\Leftrightarrow m^{2} -2m+m-2=0\Leftrightarrow m\left(m-2\right)+m-2=0$ \\
			$\Leftrightarrow \left(m-2\right)\left(m+1\right)=0\Leftrightarrow \left[\begin{aligned} & m=2 \\ &m=-1 \end{aligned}\right.$ \\
			Đối chiếu với điều kiện $m\le 1$ ta được $m=-1$ thỏa mãn bài toán.
		\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề Tuyển sinh vào lớp 10 năm học 2018-2019, Hà Tĩnh]%[Hồ Minh Hòa, dự án EXTHCS-06]%[9D3K6]
	Hai người công nhân cùng làm chung một công việc thì hoàn thành trong 16 giờ. Nếu người thứ nhất làm $3$ giờ và người thứ hai làm $2$ giờ thì họ làm được $\dfrac{1}{6} $ công việc. Hỏi nếu làm một mình thì mỗi người hoàn thành công việc đó trong bao lâu?
	\loigiai{
		Gọi thời gian người công nhân $A$ làm một mình xong công việc là $x$ (giờ) $\left(x>6\right)$.\\
		Gọi thời gian người công nhân $B$ làm một mình xong công việc là $y$ (giờ) $\left(y>16\right)$.\\
		Mỗi giờ $A$ làm được $\dfrac{1}{x} $ (công việc), $B$ làm được $\dfrac{1}{y} $ (công việc), cả hai người làm được $\dfrac{1}{16} $ (công việc). \\
		Ta có phương trình $\dfrac{1}{x} +\dfrac{1}{y} =\dfrac{1}{16} \Rightarrow \dfrac{1}{y} =\dfrac{1}{16} -\dfrac{1}{x} $ \quad $\left(1\right)$.\\
		Vì $A$ làm $3$ giờ và $B$ làm $2$ giờ thì họ làm được $\dfrac{1}{16} $ công việc nên ta có phương trình $$\dfrac{3}{x} +\dfrac{2}{y} =\dfrac{1}{6} \quad \left(2\right)$$
		Thế $\left(1\right)$ vào $\left(2\right)$ ta được $\dfrac{3}{x} +\dfrac{2}{16} -\dfrac{2}{x} =\dfrac{1}{6} \Leftrightarrow \dfrac{1}{x} =\dfrac{1}{24} \Rightarrow x=24$  thay vào $\left(1\right)$ được $$\dfrac{1}{y} =\dfrac{1}{16} -\dfrac{1}{24} \Leftrightarrow \dfrac{1}{y} =\dfrac{1}{48} \Rightarrow y=48.$$
		Đối chiếu với điều kiện ta có thời gian người thứ nhất làm một mình xong công việc là $24$ giờ, người thứ hai là $48$ giờ.
	}
\end{ex}
\begin{ex}%[Đề Tuyển sinh vào lớp 10 năm học 2018-2019, Hà Tĩnh]%[Hồ Minh Hòa, dự án EXTHCS-06]%[9H3B7]%[9H3K3]%[9H3G7]
	Cho tam giác $ABC$ có ba góc nhọn, $AB<AC$, nội tiếp đường tròn $\left(O;R\right)$. Vẽ đường kính $AD$ của đường tròn $\left(O\right)$, đường cao $AH$ của tam giác $ABC$ ($H$ thuộc $BC$) và $BE$ vuông góc với $AD$ (B thuộc $AD$).
	\begin{enumerate}
		\item Chứng minh rằng tứ giác $AEHB$ nội tiếp.
		\item Chứng minh $AH\cdot DC=AC\cdot BH$.
		\item Gọi $I$ là trung điểm $BC$. Chứng minh rằng $IH=IE$.
	\end{enumerate}
	
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\tkzDefPoints{0.8/4/A, 0/0/B, 4/0/C}
			\tkzDrawCircle[circum](A,B,C) 
			\tkzCircumCenter(A,B,C) \tkzGetPoint{O}
			\tkzDrawAltitude(B,C)(A)\tkzGetPoint{H}
			\tkzDrawAltitude(B,C)(O)\tkzGetPoint{I}
			\tkzInterLC(O,A)(O,A) \tkzGetPoints{D}{}
			\tkzDrawAltitude(A,D)(H)\tkzGetPoint{E}
			\tkzDrawPoints(A,B,C,H,O,I,D,E)
			\tkzLabelPoints[above](A) 
			\tkzLabelPoints[below](H,I,D)
			\tkzLabelPoints[left](B) 
			\tkzLabelPoints[right](C,O,E)
			\tkzDrawSegments[](A,B B,C C,A A,D C,D B,O I,E H,E B,E)
			\tkzMarkRightAngle[size=0.2](A,H,B)
			\tkzMarkRightAngle[size=0.2](H,E,A)
			\tkzMarkRightAngle[size=0.2](O,I,B)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Ta có $BH\bot AE$ (gt) và $AH\bot BC$ (gt)
			$\Rightarrow \widehat{AEB}=\widehat{AHB}=90^{\circ}$.\\
			Suy ra đỉnh $E$, $B$ cùng nhìn đoạn thẳng $AB$ dưới một góc vuông nên tứ giác $AEHB$ nội tiếp đường tròn.
			\item Ta có  $\widehat{ADC}=90^{\circ}$ (góc nội tiếp chắn nửa đường tròn).\\
			Xét $\Delta AHB$ và $\Delta ACD$ ta có\\
			$\widehat{AHB}=\widehat{ACD}=90^{\circ}$\\
			$\widehat{ABH}=\widehat{ADC}$ (góc nội tiếp cùng chắn cung $AC$).\\
			$\Rightarrow \Delta AHB \sim \Delta ACD$ (góc -- góc)\\
			$\Rightarrow \dfrac{AH}{AC} =\dfrac{BH}{DC}$\\ $\Rightarrow AH\cdot DC=AC\cdot BH$.
			
			\item Theo câu a) tứ giác $AEHB$ nội tiếp nên $\widehat{BAD}=\widehat{EHI}$ (cùng bù với $\widehat{BHE}$).\\
			Ta có $\widehat{BAD}=\dfrac{1}{2} \widehat{BOD}$ (góc nội tiếp, góc ở tâm cùng chắn cung $BD)$
			$\Rightarrow \widehat{EHI}=\dfrac{1}{2} \widehat{BOD}  \quad (1)$\\
			Mặt khác: $IB=IC$ (gt)
			$\Rightarrow OI\bot BC$.\\
			Do đó $\widehat{BIO}=\widehat{BEO}=90^{\circ}$.\\
			Suy ra đỉnh $E,I$ cùng nhìn đoạn $BO$ dưới một góc vuông nên tứ giác $BIEO$ nội tiếp, suy ra $ \widehat{EIC}=\widehat{EHI}+\widehat{IEH}$ (góc ngoài của $\Delta EIH$) \quad $(2)$.\\			
			Từ $\left(1\right)$ và $\left(2\right)$ suy ra $\widehat{EHI}=\dfrac{1}{2} \widehat{EIC}\Rightarrow \widehat{EIC}=2\widehat{EHI}$ mà $\widehat{EIC}=\widehat{EHI}+\widehat{IEH}$ (góc ngoài của $\Delta EIH$) $\Rightarrow \widehat{EHI}=\widehat{IEH}\Rightarrow \Delta EIH$ cân tại $I$ $\Rightarrow IH=IE$.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề Tuyển sinh vào lớp 10 năm học 2018-2019, Hà Tĩnh]%[Hồ Minh Hòa, dự án EXTHCS-06]%[9D1G1]
	Cho $a$, $b$ là các số thực thỏa mãn đẳng thức $\left(a+2\right)\left(b+2\right)=\dfrac{25}{4} $. Tìm giá trị nhỏ nhất của biểu thức $P=\sqrt{1+a^4} +\sqrt{1+b^4}$. 
	\loigiai{
		Áp dụng bất đẳng thức Minicopski ta có $$P=\sqrt{1+a^{4} } +\sqrt{1+b^{4} } \ge \sqrt{\left(1+1\right)^{2} +\left(a^{2} +b^{2} \right)^{2} } =\sqrt{\left(a^{2} +b^{2} \right)^{2} +4}. $$ 
		Từ giả thuyết ta có $2a+2b+ab=\dfrac{9}{4} $ .\\
		Ta có $a^{2} +b^{2} \ge 2ab\Leftrightarrow \dfrac{a^{2} +b^{2} }{2} \ge ab$  \quad $\left(1\right)$\\
		$\begin{cases} 4a^{2} +1\ge 4a \\4b^{2} +1\ge 4b\end{cases}\Rightarrow 4a^{2} +4b^{2} \ge 4\left(a+b\right)-2$  $\Leftrightarrow 2\left(a^{2} +b^{2} \right)\ge 2\left(a+b\right)-1$\quad $(2)$. \\
		Cộng các bất đẳng thức $\left(1\right)$ và $\left(2\right)$ vế theo vế ta được
		$\dfrac{5\left(a^{2} +b^{2} \right)}{2} \ge ab+2\left(a+b\right)-1=\dfrac{9}{4} -1=\dfrac{5}{4} $ 
		$\Rightarrow a^{2} +b^{2} \ge \dfrac{1}{2} \Rightarrow P\ge \sqrt{\dfrac{1}{4} +4} =\dfrac{\sqrt{17} }{2} $.\\ 
		Do đó giá trị nhỏ nhất của $P$ bằng $\dfrac{\sqrt{17} }{2} $ khi và chỉ khi $a=b=\dfrac{1}{2} $.
	}
\end{ex}
