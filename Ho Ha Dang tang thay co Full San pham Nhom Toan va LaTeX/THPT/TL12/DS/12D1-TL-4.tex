\section{ĐƯỜNG TIỆM CẬN}
\subsection{TÓM TẮT LÝ THUYẾT}
\subsubsection{Đường tiệm cận ngang}
\begin{dn}
Cho hàm số $y=f(x)$ xác định trên một khoảng vô hạn (là khoảng dạng $(a;+\infty)$, $(-\infty;b)$ hoặc $(-\infty;+\infty)$). Đường thẳng $y=y_0$ là đường tiệm cận ngang (hay tiệm cận ngang) của đồ thị hàm số $y=f(x)$ nếu ít nhất một trong các điều kiện sau được thỏa mãn
$$\lim\limits_{x\to +\infty}f(x)=y_0; \lim\limits_{x\to -\infty} =y_0.$$
\end{dn}

\begin{vd}%[Học Toán]%[2D1B4]
Cho hàm số $f(x)=\dfrac{1}{\sqrt{x}}+1$ xác định trên khoảng $(0;+\infty)$. 
\loigiai{
Đồ thị hàm số có tiệm cận ngang $y=1$ vì $\lim\limits_{x\to +\infty}f(x) =\lim\limits_{x\to +\infty} \left(\dfrac{1}{\sqrt{x}}+1\right)=1.$
}
\end{vd}

\begin{vd}%[Học Toán]%[2D1K4]
Tìm tiệm cận ngang của đồ thị hàm số $y=\dfrac{\sqrt{x^2-2x+6}}{x-1}\cdot$
\loigiai{
Đồ thị hàm số có hai đường tiệm cận ngang $y=1$ và $y=-1$ vì\\
$\lim\limits_{x\to +\infty} \dfrac{\sqrt{x^2-2x+6}}{x-1}=\lim\limits_{x\to +\infty} \dfrac{|x|\cdot\sqrt{1- \dfrac{2}{x}+\dfrac{6}{x^2}}}{x}=\lim\limits_{x\to -\infty} \dfrac{x\cdot\sqrt{1- \dfrac{2}{x}+\dfrac{6}{x^2}}}{x}=1$\\
$\lim\limits_{x\to -\infty} \dfrac{\sqrt{x^2-2x+6}}{x-1}=\lim\limits_{x\to -\infty} \dfrac{|x|\cdot\sqrt{1- \dfrac{2}{x}+\dfrac{6}{x^2}}}{x}=\lim\limits_{x\to -\infty} \dfrac{-x\cdot\sqrt{1- \dfrac{2}{x}+\dfrac{6}{x^2}}}{x}=-1$
}
\end{vd}

\subsubsection{Đường tiệm cận đứng}
\begin{dn}
Đường thẳng $x=x_0$ được gọi là đường tiệm cận đứng (hay tiệm cận đứng) của đồ thị hàm số $y=f(x)$ nếu ít nhất một trong các điều kiện sau được thỏa mãn
\begin{multicols}{2}
\begin{itemize}
	\item $\lim\limits_{x\to x_0^+} f(x)=+\infty$
	\item $\lim\limits_{x\to x_0^+} f(x)=-\infty$
	\item $\lim\limits_{x\to x_0^-} f(x)=-\infty$
	\item $\lim\limits_{x\to x_0^-} f(x)=+\infty$
\end{itemize}
\end{multicols}
\end{dn}

\begin{vd}%[Học Toán]%[2D1B4]
Tìm các tiệm cận đứng và tiệm cận ngang của đồ thị $(C)$ của hàm số $y=\dfrac{x-1}{x+2}\cdot$
\loigiai{
Vì $\lim\limits_{x\to -2^+} \dfrac{x-1}{x+2}=-\infty$ (hoặc $\lim\limits_{x\to -2^-} \dfrac{x-1}{x+2}=+\infty$) nên đường thẳng $x=-2$ là tiệm cận đứng của $(C)$.\\
\immini{
	Vì $\lim\limits_{x\to \pm\infty} \dfrac{x-1}{x+2}=1$ nên đường thẳng $y=1$ là tiệm cận ngang của $(C)$. \\
	Đồ thị của hàm số được cho bởi hình bên
}{
	\begin{tikzpicture}[thick,>=stealth,x=1cm,y=1cm,scale=.5] 
	\draw[->] (-8,0) -- (8,0) node[below] {\small $x$};
	\draw[->] (0,-4) -- (0,5) node[right] {\small $y$};
	\draw [fill=white,draw=black] (0,0) circle (1pt)node[above left] {\footnotesize $O$};
	\clip(-8,-4) rectangle (8,5);
	\draw[thick,smooth,samples=100,domain=-8:-2.1] plot(\x,{((\x)-1)/((\x)+2)});
	\draw[thick,smooth,samples=100,domain=-1.9:8] plot(\x,{((\x)-1)/((\x)+2)});
	\draw[thick,smooth,samples=100,domain=-8:8] plot(\x,{1});
	\draw (-2,-4)--(-2,5);
	\draw (-2,0)node[below left]{$-2$};
	\draw (0,-1/2)node[below right]{$-\frac{1}{2}$};
	\draw (1,0) node [above]{$1$};
	\draw (-2.5,-1)node[rotate=90, left]{$x=-2$};
	\draw (7,1)node[above]{$y=1$};
	\end{tikzpicture}
}
}
\end{vd}

\begin{vd}%[Học Toán]%[2D1B4]
Tìm tiệm cận đứng của đồ thị hàm số $y=\dfrac{2x^2+x+1}{2x-3}\cdot$
\loigiai{
Vì $\lim\limits_{x\to \left(\dfrac{3}{2}\right)^+} \dfrac{2x^2+x+1}{2x-3}=+\infty$ (hoặc $\lim\limits_{x\to \left(\dfrac{3}{2}\right)^-} \dfrac{2x^2+x+1}{2x-3}=-\infty$) nên đường thẳng $x=\dfrac{3}{2}$ là tiệm cận đứng của đồ thị hàm số đã cho.
}
\end{vd}


\begin{note}
Tiệm cận có thể cắt đồ thị hàm số.
\end{note}

\begin{vd}%[Học Toán]%[2D1K4]
Tìm tiệm cận ngang của đồ thị hàm số $y=\dfrac{x}{x^2+x+1}\cdot$
\loigiai{
\immini{

	Ta thấy rằng đồ thị hàm số đi qua điểm $A(0;0)\Rightarrow$ tiệm cận ngang $y=0$ cắt đồ thị tại điểm $A(0;0).$

\begin{bt}%[Lê Phước Thật]%[2D1K4]
Tìm các đường tiệm cận ngang (nếu có) của đồ thị các hàm số
\begin{multicols}{2}
\begin{enumerate}
	\item $y=\sqrt{x^2+1}-\sqrt{x^2-2x-4}$.
	\item $y=\sqrt{x^2+2x+3}-\sqrt{x^2-4x+7}$.
	\item $y=\sqrt{x^2+1}-x$.
	\item $y=\dfrac{\sqrt{4x^2-1}+3x^2+2}{x^2-x}$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
	\item Ta có $\lim\limits_{x \to +\infty} y = \lim\limits_{x \to +\infty} \dfrac{2x-3}{\sqrt{x^2+1}+\sqrt{x^2-2x-4}} = \lim\limits_{x \to +\infty} \dfrac{2-\dfrac{3}{x}}{\sqrt{1+\dfrac{1}{x^2}}+\sqrt{1-\dfrac{2}{x}+\dfrac{4}{x^2}}}=1$.\\
	Suy ra $y=1$ là tiệm cận ngang của đồ thị hàm số.\\
	$\lim\limits_{x \to -\infty} y = \lim\limits_{x \to -\infty} \dfrac{2x-3}{\sqrt{x^2+1}+\sqrt{x^2-2x-4}} = \lim\limits_{x \to -\infty} \dfrac{2-\dfrac{3}{x}}{-\sqrt{1+\dfrac{1}{x^2}}-\sqrt{1-\dfrac{2}{x}+\dfrac{4}{x^2}}}=-1$.\\
	Suy ra $y=-1$ là tiệm cận ngang của đồ thị hàm số.
	\item $\lim\limits_{x \to -\infty} y = \lim\limits_{x \to -\infty} \dfrac{6x-4}{\sqrt{x^2+2x+3}+\sqrt{x^2-4x+7}} = -3$ nên $y=-3$ là tiệm cận ngang.\\
	$\lim\limits_{x \to +\infty} y = \lim\limits_{x \to +\infty} \dfrac{6x-4}{\sqrt{x^2+2x+3}+\sqrt{x^2-4x+7}} = 3$ nên $y=3$ là tiệm cận ngang.
	\item $\lim\limits_{x \to -\infty} y = -\infty$ và $\lim\limits_{x \to +\infty} y = \lim\limits_{x \to +\infty} \dfrac{1}{\sqrt{x^2+1}+x} = 0$ nên $y=0$ là tiệm cận ngang.\\
	
	\item $\lim\limits_{x \to - \infty} y = \lim\limits_{x \to - \infty} \dfrac{-\sqrt{\dfrac{4}{x^2}-\dfrac{1}{x^4}}+3+\dfrac{2}{x^2}}{1-\dfrac{1}{x}}=3$.\\
	$\lim\limits_{x \to + \infty} y = \lim\limits_{x \to + \infty} \dfrac{\sqrt{\dfrac{4}{x^2}-\dfrac{1}{x^4}}+3+\dfrac{2}{x^2}}{1-\dfrac{1}{x}}=3$.\\
	Suy ra $y=3$ là tiệm cận ngang.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Lê Phước Thật]%[2D1G4]
Tìm tất cả giá trị của tham số $m$ để đồ thị các hàm số sau có hai đường tiệm cận.
\begin{multicols}{2}
\begin{enumerate}
	\item $y=\dfrac{x+m}{x-1}$.
	\item $y=\dfrac{x+m^2}{x+4}$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
	\item Đồ thị hàm số $y=\dfrac{ax+b}{cx+d}, (c \ne 0)$ có hai đường tiệm cận khi và chỉ khi $ad-bc \ne 0$.\\
	Do đó ta có $-11-m \ne 0 \Leftrightarrow m \ne -1$. Vậy $m \in \mathbb{R} \setminus \{-1\}$.
	\item Ta có $4-m^2 \ne 0 \Leftrightarrow m \ne \pm 2$. Vậy $m \in \mathbb{R} \setminus \{\pm 2\}.$
\end{enumerate}
}
\end{bt}

\begin{bt}%[Lê Phước Thật]%[2D1G4]
Tìm tất cả giá trị của tham số $m$ để đồ thị các hàm số sau có ba đường tiệm cận.
\begin{multicols}{2}
\begin{enumerate}
	\item $y=\dfrac{2}{x^2+mx+m}$.
	\item $y=\dfrac{x-2}{x^2+mx+m}$.
\end{enumerate}
\end{multicols}
\loigiai{
	\begin{enumerate}
	\item Ta có $\lim\limits_{x \to -\infty}y=0$ và $\lim\limits_{x \to +\infty}y=0$ nên $y=0$ là tiệm cận ngang của đồ thị hàm số.\\
	Để đồ thị hàm số có ba đường tiệm cận thì $x^2+mx+m=0$ có hai nghiệm phân biệt.\\
	$\Delta >0 \Leftrightarrow m^2-4m>0 \Leftrightarrow m \in (-\infty;0) \cup (4;+\infty)$.
	\item Ta có $\lim\limits_{x \to -\infty}y=0$ và $\lim\limits_{x \to +\infty} y=0$ nên $y=0$ là tiệm cận ngang của đồ thị hàm số.\\
	Để đồ thị hàm số có ba đường tiệm cận thì $x^2+mx+m=0$ có $2$ nghiệm phân biệt khác $2$.\\
	$\left\{\begin{aligned}&\Delta >0\\&4+2m+m \ne 0\end{aligned}\right. \Leftrightarrow \left\{\begin{aligned}&m^2-4m>0\\&m\ne -\dfrac{4}{3}\end{aligned}\right. \Leftrightarrow m \in (-\infty;0) \cup (4;+\infty) \setminus \left\{-\dfrac{4}{3}\right\}$.
	\end{enumerate}}
\end{bt}

\begin{bt}%[Lê Đình Mẫn]%[2D1B4]
Tìm tất cả các giá trị của tham số $m$ để đồ thị hàm số $y=\dfrac{x-1}{x^2+2(m-1)x+m^2-2}$ có đúng hai đường tiệm cận đứng.
\loigiai{Đồ thị hàm số đã cho có đúng hai tiệm cận đứng khi và chỉ khi phương trình\newline
$x^2+2(m-1)x+m^2-2=0$ có đúng hai nghiệm khác $1$. Ta có $\Delta' =(m-1)^2-(m^2-2)=3-2m$.\\
Do đó, yêu cầu bài toán tương đương với $\left\{\begin{aligned}&3-2m>0\\&1^2+2(m-1)+m^2-2\ne 0\end{aligned}\right.\Leftrightarrow \left\{\begin{aligned}&m<\dfrac{3}{2}\\&m\ne 1,m\ne -3\end{aligned}\right.$.}
\end{bt}

\begin{bt}%[Lê Đình Mẫn]%[2D1B4]
Tìm tất cả các giá trị của tham số $m$ để đồ thị hàm số $y=\dfrac{x-2}{\sqrt{(m-1)x^2+1}}$ có hai đường tiệm cận ngang.
\loigiai{\begin{enumerate}[$\bullet$]
\item Nếu $m=1$ thì $y=x-2$ nên lúc này đồ thị hàm số không có tiệm cận ngang.
\item Xét $m<1$: Hàm số có tập xác định $\mathscr{D}=\left(-\dfrac{1}{\sqrt{1-m}};\dfrac{1}{\sqrt{1-m}}\right)$ nên trong trường hợp này đồ thị hàm số cũng không có tiệm cận ngang.
\item Xét $m>1$: Hàm số có tập xác định là $\mathscr{D}=\left(-\infty;-\dfrac{1}{\sqrt{1-m}}\right)\cup \left(\dfrac{1}{\sqrt{1-m}};+\infty\right)$ và tính các giới hạn sau:\\
+ $\lim\limits_{x\to +\infty}y=\lim\limits_{x\to +\infty}\dfrac{x-2}{\sqrt{(m-1)x^2+1}}=\lim\limits_{x\to +\infty}\dfrac{1-\dfrac{2}{x}}{\sqrt{m-1+\dfrac{1}{x^2}}}=\dfrac{1}{\sqrt{m-1}}$;\\
+ $\lim\limits_{x\to -\infty}y=\lim\limits_{x\to -\infty}\dfrac{x-2}{\sqrt{(m-1)x^2+1}}=\lim\limits_{x\to -\infty}\dfrac{1-\dfrac{2}{x}}{-\sqrt{m-1+\dfrac{1}{x^2}}}=-\dfrac{1}{\sqrt{m-1}}$.\\
Do đó, trường hợp này đồ thị hàm số có hai tiệm cận ngang.
\end{enumerate}}
\end{bt}

\begin{bt}%[Lê Đình Mẫn]%[2D1B4]
Giả sử đường thẳng $d:x=a$, $a>0$ cắt đồ thị hàm số $y=\dfrac{2x+1}{x-1}$ tại một điểm duy nhất, biết khoảng cách từ điểm đó đến tiệm cận đứng của đồ thị hàm số bằng $1$. Tìm tọa độ điểm đó.
\loigiai{Đường thẳng $d$ cắt đồ thị hàm số $y=\dfrac{2x+1}{x-1}$ tại điểm $M\left(a;\dfrac{2a+1}{a-1}\right)$. Đồ thị có tiệm cận đứng $\Delta: x=1$.\\
Ta có $d(M,\Delta)=1\Leftrightarrow \dfrac{|a-1|}{1}=1\Leftrightarrow \left[\begin{aligned}&a=0\\&a=2\end{aligned}\right.\Rightarrow a=2>0$.\\
Với $a=2\Rightarrow x_0=2\Rightarrow y_0=5$.}
\end{bt}

\begin{bt}%[Lê Đình Mẫn]%[2D1K4]
Cho hàm số $y=\dfrac{(m-1)x+1}{2x+m}$ có đồ thị là $(C)$. Gọi $I$ là giao điểm của hai đường tiệm cận của $(C)$, $O$ là gốc tọa độ và $A(4;-6)$. Tìm tất cả các giá trị của tham số $m$ để ba điểm $O,I,A$ thẳng hàng.
\loigiai{Giao điểm của hai đường tiệm cận của đồ thị $(C)$ là $I\left(-\dfrac{m}{2};\dfrac{m-1}{2}\right)$.\\
Xét $\overrightarrow{OI}=k\overrightarrow{OA}\Leftrightarrow\left\{\begin{aligned}&-\dfrac{m}{2}=4k\\&\dfrac{m-1}{2}=-6k\end{aligned}\right.\Leftrightarrow\left\{\begin{aligned}&k=\dfrac{1}{4}\\&m=-2\end{aligned}\right.$.\\
Vậy, với $m=-2$ thì ba điểm $O,I,A$ thẳng hàng.}
\end{bt}

\begin{bt}%[Lê Đình Mẫn]%[2D1K4]
Cho hàm số $y=\dfrac{x+2}{x-1}$ có đồ thị là $(C)$. Chứng minh rằng với mọi điểm $M\in (C)$ ta luôn có tiếp tuyến của $(C)$ tại điểm $M$ tạo với hai đường tiệm cận của nó một tam giác có diện tích không đổi.
\loigiai{Gọi $M\left(x_0;\dfrac{x_0+2}{x_0-1}\right),x_0\ne 1$. Các đường tiệm cận ngang, tiệm cận đứng của đồ thị có phương trình lần lượt là $y=1,x=1$.\\
Phương trình tiếp tuyến với $(C)$ tại điểm $M$ có dạng: $y=-\dfrac{3}{(x_0-1)^2}(x-x_0)+\dfrac{x_0+2}{x_0-1}$.\\
Tiếp tuyến giao với đường tiệm cận ngang và tiệm cận đứng của đồ thị $(C)$ lần lượt là $A\left(2x_0-1;1\right)$ và $B\left(1;\dfrac{x_0+5}{x_0-1}\right)$. Giao điểm của hai đường tiệm cận là $I(1;1)$.\\
Diện tích tam giác $IAB$ là $S=\dfrac{1}{2}IA\cdot IB=\dfrac{1}{2}|2x_0-2|\cdot \dfrac{6}{|x_0-1|}=6$. Bài toán được chứng minh.}
\end{bt}

\begin{bt}%[Lê Đình Mẫn]%[2D1G4]
Cho hàm số $y=\dfrac{2x+1}{x-3}$ có đồ thị là $(C)$. Tìm tọa độ điểm $M$ trên đồ thị $(C)$ sao cho tổng khoảng cách từ $M$ đến hai tiệm cận của đồ thị đạt giá trị nhỏ nhất.
\loigiai{Gọi $M\left(x_M;\dfrac{2x_M+1}{x_M-3}\right),x_M\ne 3$. Các đường tiệm cận ngang, tiệm cận đứng của đồ thị có phương trình lần lượt là $y=2,x=3$.\\
Tổng khoảng cách từ điểm $M$ đến hai tiệm cận là\newline $d=|x_M-3|+\left|\dfrac{2x_M+1}{x_M-3}-2\right|=|x_M-3|+\dfrac{7}{\left|x_M-3\right|}\ge 2\sqrt{7}$.\\
Đẳng thức xảy ra khi và chỉ khi $|x_M-3|=\dfrac{7}{\left|x_M-3\right|}\Leftrightarrow \left[\begin{aligned}&x_M=3+\sqrt{7}\\&x_M=3-\sqrt{7}\end{aligned}\right.$\\
Vậy, tọa độ các điểm thỏa mãn yêu cầu bài toán là $M\left(3+\sqrt{7};2+\sqrt{7}\right)$ và $M\left(3-\sqrt{7};2-\sqrt{7}\right)$.}
\end{bt}
