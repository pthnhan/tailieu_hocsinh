	\begin{enumerate}[a)]

		\item Rút gọn biểu thức $B=\sqrt{7-4\sqrt{3}}+\dfrac{1}{2-\sqrt{3}}$.

		\end{enumerate}
	}

\begin{ex}%[TS 10,Năm 2018 – 2019,Vĩnh Long]%[Nguyễn Bình Nguyên-Dự án EX_THCS06]%[9D3B3]%[9D4B4]%[9D4B7]
Giải các phương trình và hệ phương trình sau:
	\begin{enumerate}[a)]
		\item $x^2-3x+2=0$
		\item $x^2-2\sqrt{3}x+3=0$
		\item $x^4-9x^2=0$
		\item $\begin{cases}x-y=3\\3x-2y=8 \end{cases}$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item $x^2-3x+2=0$ \\
			Ta có $\Delta =1>0$. \\ 
			Phương trình có $ 2 $ nghiệm $x_1=1$, $x_2=2$.			
			\item $x^2-2\sqrt{3}x+3=0$ \\
			Ta có $\Delta =0$. \\
			Phương trình có nghiệm kép $x_1=x_2=\sqrt{3}$.
			\item $x^4-9x^2=0$\\
			Đặt $t=x^2$, $t\ge 0$, phương trình trở thành $t^2-9t=0$.\\
			Giải ra được $t=0$ (nhận); $t=9$ (nhận)\\
			Khi $t=9$, ta có $x^2=9\Leftrightarrow x=\pm 3$.\\
			Khi $t=0$, ta có $x^2=0\Leftrightarrow x=0$.
			\item Ta có 
			$\begin{cases}x-y=3 \\3x-2y=8\end{cases}
			\Leftrightarrow \begin{cases}x=y+3 \\ 3(y+3)-2y=8\end{cases} 
			\Leftrightarrow \begin{cases} x=2 \\ y=-1 \end{cases}$\\
			Vậy hệ phương trình có nghiệm là $(x,y)=(2;-1)$.
		\end{enumerate}
	}
\end{ex}
\begin{ex}%[TS 10,Năm 2018 – 2019,Vĩnh Long]%[Nguyễn Bình Nguyên-Dự án EX_THCS06]%[9D4B2]%[9D4G6]
	\begin{enumerate}[a)]
		\item Trong mặt phẳng tọa độ $Oxy$, cho Parabol $(P)\colon y=x^2$. Vẽ đồ thị Parabol $(P)$. 
		\item Cho phương trình $x^2-(m-1)x-m=0$ (1) (với $x$ là ẩn số, $m$ là tham số). Xác định các giá trị của $ m $ để phương trình (1) có hai nghiệm phân biệt $x_1$, $x_2$ thỏa mãn điều kiện: $x_1(3-x_2)+20\ge 3(3-x_2).$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item 
			Bảng giá trị giữa $x$ và $y$
			\begin{center}
				\begin{tabular}{|c|c|c|c|c|c|}
					\hline 
					$x$ & -2 & -1 & 0 & 1 & 2 \\ 
					\hline 
					$y$ & 4 & 1 & 0 & 1 & 4 \\ 
					\hline 
				\end{tabular}
			\end{center}
			Đồ thị
			\begin{center}
				{\begin{tikzpicture}[x=1cm,y=1cm, scale=0.6] 
					\draw[->, color=black, thick] (-3,0) -- (3,0); 
					\foreach \x in {-2,-1,1,2}
					\draw[shift={(\x,0)},color=black,thick] (0pt,2pt) -- (0pt,-2pt) node[below] { $\x$};
					\draw[->,color=black,thick] (0,-2) -- (0,5); %vẽ trục oy
					
					\foreach \y in {1,4}

					\clip(-3,-2) rectangle (3,5); 
					\draw[smooth,very thick,samples=100,domain=-3:3] plot(\x,{(\x)^2}); 
					\draw[dashed] (0,4) -- (-2,4) -- (-2,0); 
					\draw[dashed] (0,4) -- (2,4) -- (2,0); 
					\draw[dashed] (-1,0) -- (-1,1) -- (0,1);
					\draw[dashed] (1,0) -- (1,1) -- (0,1);
					\node at (2.9,0) [above] {$x$};
					\node at (0,4.5) [right] {$y$};
					\node at (0,0) [below right] {$O$};

					\end{tikzpicture}}
			\end{center}
			\item 
			Ta có $\Delta =( m-1 )^2+4m=( m+1 )^2$.\\
			Phương trình (1) có hai nghiệm phân biệt 
			$x_1$; $x_2$ khi $\Delta >0\Leftrightarrow m\ne -1$.\\
			Ta có
			$\begin{cases} x_1+x_2=m-1 \\x_1\cdot x_2=-m \end{cases}$ \\
			Theo đề bài ta có
			$$x_1(3-x_2) + 20 \ge 3(3-x_2)
			\Leftrightarrow 3(x_1+x_2)-x_1x_2 \ge -11$$
			$$\Leftrightarrow 3(m-1)+m \ge -11
			\Leftrightarrow 4m\ge -8
			\Leftrightarrow m\ge -2.$$
			
			Vậy $m\ge -2$; $m\ne -1$ thì phương trình (1) có hai nghiệm phân biệt thỏa mãn yêu cầu bài toán.
		\end{enumerate}
	}
\end{ex}
\begin{ex}%[TS 10,Năm 2018 – 2019,Vĩnh Long]%[Nguyễn Bình Nguyên-Dự án EX_THCS06]%[9D4K8]
	Quãng đường  dài $160$ km. Hai xe khởi hành cùng một lúc từ $A$ để đi đến $B$. Vận tốc của xe thứ nhất lớn hơn vận tốc của xe thứ hai là $10$ km/h nên xe thứ nhất đến  sớm hơn xe thứ hai là $48$ phút. Tính vận tốc của xe thứ hai. 
	\loigiai{
		Gọi vận tốc của xe thứ hai là $x$ (km/h). Điều kiện: $x>0$.Vận tốc của xe thứ nhất là  $x+10$ (km/h).
		Thời gian đi quãng đường  $AB$ của xe thứ nhất là $\dfrac{160}{x+10}$ (h) 
		và thời gian của xe thứ hai là $\dfrac{160}{x}$  (h).
		Theo đề bài ta có phương trình  $$ \frac{160}{x}-\frac{160}{x+10}=\frac{48}{60}. $$
		Giải phương trình ta được: $x=40$ (nhận), $x=-50$ (loại).
		Vậy vận tốc của xe thứ hai là $40$ km/h.
	}
\end{ex}

\begin{ex}%[TS 10,Năm 2018 – 2019,Vĩnh Long]%[Nguyễn Bình Nguyên-Dự án EX_THCS06]%[9H3K7]

	\item Biết $\widehat{EBC}=30^\circ$. Tính số đo $\widehat{EMC}$.

\begin{ex}%[TS 10,Năm 2018 – 2019,Vĩnh Long]%[Nguyễn Bình Nguyên-Dự án EX_THCS06]%[9D4G6]
Cho $a=\dfrac{\sqrt2-1}{2}$; $b=\dfrac{\sqrt2+1}{2}$. Tính $a^7+b^7$.
\loigiai{
	Từ giả thiết ta có $ a+b=\sqrt{2}$ và $ ab=\dfrac{1}{4}$.
	\begin{eqnarray*}
		a^7+b^7&= & \left(a^4+b^4\right)\left(a^3+b^3\right)-a^3b^3(a+b)\\
		&= &\left\lbrace\left[ (a+b)^2-2ab\right]^2-2a^2b^2  \right\rbrace\left[(a+b)^3-3ab(a+b)\right]-a^3b^3(a+b)\\
		&= &\left[\left(2-\dfrac{1}{2} \right)^2-\dfrac{1}{8}\right]\cdot \left(2\sqrt{2}-\dfrac{3\sqrt{2}}{4}\right)-\dfrac{\sqrt{2}}{64}\\ 
		&= &\dfrac{169\sqrt{2}}{64}.
	\end{eqnarray*}
		}
\end{ex}
