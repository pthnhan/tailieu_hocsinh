\begin{center}
 \large \textbf{TỔNG QUAN VỀ CHUYÊN ĐỀ}
\end{center}
Nội dung chuyên đề này bao gồm:\\
- Phương trình chứa ẩn trong dấu căn bậc ba.\\
- Phương trình chứa ẩn trong dấu căn bậc bốn.\\
Chúng ta sẽ thấy những điểm giống nhau và những điểm khác nhau trong cách giải các phương trình trên so với phương trình chứa căn thức bậc hai ở Chuyên đề 5.\\
{\bf {Thử trí thông minh!}}\\
\begin{center}
	{\bf {VÌ SAO THỪA NGHIỆM?}}
\end{center}
Bạn Thu phải giải phương trình $\sqrt[3]{2x-3}+\sqrt[3]{x-2}=1,\quad (1).$ Bạn đã giải như sau\\ Lập phương hai vế PT(1) ta được
	$$ (1) \Leftrightarrow (2x-3)+(x-2) +3 \sqrt[3]{(x-2)(2x-3)} \left( \sqrt[3]{2x-3}+ \sqrt[3]{x-2}\right)=1$$
	Thay $\sqrt[3]{2x-3}+\sqrt[3]{x-2}=1$ vào phương trình trên ta được $3x-5+3\sqrt[3]{(x-2)(2x-3)}=1$ 
		\begin{align*}
		 &\Leftrightarrow \sqrt[3]{(x-2)(2x-3)}=2-x\\& \Leftrightarrow (x-2)(2x-3)=(2-x)^3\\& \Leftrightarrow (x-2)(x-1)^2=0\\& \Leftrightarrow \hoac{&x=1\\&x=2}.
		\end{align*}
Nhưng thay $x=1$ vào phương trình (1) lại được $-2=1 (!)$\\
Thu không hiểu tại sao như vậy, bạn hãy giải thích giúp.\\
{\bf{Giải}} \\
Do có phép biến đổi ở trên không tương đương (Phép thế $\sqrt[3]{2x-3}+\sqrt[3]{x-2}$ bởi 1) đã làm xuất hiện nghiệm ngoại lai $x=1$.\\
Do đó, sau khi tìm được $x=1$ và $x=2$, phải thử lại. Thử lại ta được phương trình có nghiệm duy nhất là $x=2.$

\subsection{Phương trình chứa căn bậc ba}
Một số phương pháp thường dùng để giải phương trình chứa căn thức bậc ba:\\
- Lập phương hai vế của phương trình.\\
- Đặt ẩn phụ.\\
- Dùng biểu thức liên hợp, tức là khử căn thức ở tử bằng cách nhân và chia với biểu thức liên hợp:\\
$$ \sqrt[3]{A}+\sqrt[3]{B}=\dfrac{A+B}{\sqrt[3]{A^2}- \sqrt[3]{AB}+\sqrt[3]{B^2}};
$$ $$ \sqrt[3]{A}-\sqrt[3]{B}=\dfrac{A-B}{\sqrt[3]{A^2}+ \sqrt[3]{AB}+\sqrt[3]{B^2}}.
$$ \\
- Dùng bất đẳng thức.


\begin{vd}
    Giải phương trình $\sqrt[3]{2x-3}+\sqrt[3]{x-2}=1$.\hfill$(1)$
    \loigiai{ Ngoài cách nói ở trên, còn có một số cách khác. \\
    {\bf{Cách 2}}: Đặt $a=\sqrt[3]{2x-3}; b= \sqrt[3]{x-2}.$ \\ Ta có $\heva{&a+b=1\\&a^3-2b^3=1} \Leftrightarrow \heva{&a=1\\&b=0} \Rightarrow x=2.$\\
    {\bf{Cách 3}}: Ta thấy $x=2$ là nghiệm của phương trình đã cho.
    \begin{itemize}
    	\item  Với $x>2$ ta có $\heva{& \sqrt[3]{2x-3}>1\\& \sqrt[3]{x-2}>0} \Rightarrow $ VT$(1) >1,$ do đó $x>2$ không thỏa mãn phương trình.
    	\item Với $x<2$ ta có $\heva{& \sqrt[3]{2x-3}<1\\& \sqrt[3]{x-2}<0} \Rightarrow $ VT$(1) <1,$ do đó $x<2$ không thỏa mãn phương trình.
    \end{itemize}
    Vậy phương trình có nghiệm duy nhất $x=2.$
    }
\end{vd}
\begin{vd}
	Giải phương trình $x^3 -1 =2 \sqrt[3]{2x+1}.$
	\loigiai{ Đặt $\sqrt[3]{2x+1}=y$, ta có hệ phương trình $\heva{&x^3-1=2y\\&y^3-1=2x.}$\\
Giải hệ phương trình trên (Xem VD 26) ta được $x=-1$ và $x= \dfrac{1\pm \sqrt{5}}{2}.$\\
Vậy phương trình có 3 nghiệm là $x=-1;  x= \dfrac{1\pm \sqrt{5}}{2}.$

}
\end{vd}
\begin{vd}
	Giải phương trình $\sqrt[3]{(x-2)^2}+ \sqrt[3]{x^2-4}=2\sqrt[3]{(x+2)^2}.$
	\loigiai{Đặt $\heva{& a= \sqrt[3]{x-2}\\& b=\sqrt[3]{x+2}}.$  Phương trình đã cho trở thành $a^2+ab-2b^2=0 $
		\begin{align*}
		&\Leftrightarrow (a-b)(a+2b)=0 \\& \Leftrightarrow \hoac{&a=b\\&a=-2b}.
		\end{align*}
	\begin{itemize}
		\item Với $a=b \Rightarrow x-2=x+2$ vô lý.
		\item Với $a=-2b \Rightarrow \sqrt[3]{x-2}=-2 \sqrt[3]{x+2} \Leftrightarrow x=-\dfrac{14}{9}.$
	\end{itemize}	
	Vậy phương trình có nghiệm duy nhất $x=-\dfrac{14}{9}.$
}
\end{vd}
\begin{vd}
	Giải phương trình $\sqrt[3]{2x+1}- \sqrt[3]{3x-2}=(2x-6) \sqrt{x-1}.$
	\loigiai{ ĐKXĐ: $x \geq 1.$\\
		Phương trình đã cho tương đương với $$\dfrac{(2x+1)-(3x-2)}{\sqrt[3]{(2x+1)^2}+\sqrt[3]{(2x+1)(3x-2)} +\sqrt[3]{(3x-2)^2}} = 2(x-3) \sqrt{x-1}$$
		\begin{align*}
		&\Leftrightarrow \dfrac{3-x}{\sqrt[3]{(2x+1)^2}+\sqrt[3]{(2x+1)(3x-2)} +\sqrt[3]{(3x-2)^2}} = 2(x-3) \sqrt{x-1}\\&
		\Leftrightarrow \dfrac{x-3}{\sqrt[3]{(2x+1)^2}+\sqrt[3]{(2x+1)(3x-2)} +\sqrt[3]{(3x-2)^2}} + 2(x-3) \sqrt{x-1}=0\\&
		\Leftrightarrow (x-3) \left(\dfrac{1}{\sqrt[3]{(2x+1)^2}+\sqrt[3]{(2x+1)(3x-2)} +\sqrt[3]{(3x-2)^2}} + 2 \sqrt{x-1} \right)=0\\& \Leftrightarrow  \hoac{&x=3\\&\dfrac{1}{\sqrt[3]{(2x+1)^2}+\sqrt[3]{(2x+1)(3x-2)} +\sqrt[3]{(3x-2)^2}} + 2 \sqrt{x-1} =0\, \text{: loại vì vế trái luôn dương}.}
		\end{align*}
	Vậy phương trình có nghiệm duy nhất $x=3.$
	
}
\end{vd}
\subsection{Phương trình chứa căn bậc bốn}
\begin{vd}
	Giải phương trình $\sqrt[4]{x+21}+\sqrt[4]{61-x}=4.$
	\loigiai{ ĐKXĐ: $-21 \leq x \leq 61.$\\
		Đặt $\heva{& a= \sqrt[4]{x+21}\\& b=\sqrt[4]{61-x}},$ ta có $\heva{&a^4+b^4=82 \quad (1)\\&a+b=4 \quad (2)}$\\
		Thay $b=4-a$ vào PT(1) ta được $a^4+(4-a)^4=82 \Leftrightarrow  a^4+(a-4)^4=82.$\\
		Đặt $a-2=y$, phương trình trên trở thành $(y+2)^4+(y-2)^4=82$
		\begin{align*}
		&\Leftrightarrow y^4+24y^2-25=0\\& \Leftrightarrow \hoac{&y^2=1\\&y^2=-25.}
		\end{align*}
			\begin{itemize}
			\item Với $y=1 \Rightarrow a=3 \Rightarrow x=60.$
			\item Với $y=-1 \Rightarrow a=1 \Rightarrow x=-20.$	
		\end{itemize}
	Vậy phương trình có hai nghiệm phân biệt $x=-20; x=60.$
}
\end{vd}
\begin{vd}
	Giải phương trình $\sqrt[4]{1-2x}+\sqrt[4]{1+2x} + \sqrt[4]{1-4x^2}=3.$
	\loigiai{ ĐKXĐ: $-\dfrac{1}{2} \leq x \leq \dfrac{1}{2}$.\\
			Đặt $\heva{& a= \sqrt{1-2x}\\& b=\sqrt{1+2x}},$ điều kiện $a \geq 0; b\geq 0$.\\
			Phương trình đã cho trở thành $\sqrt{a}+\sqrt{b} + \sqrt{ab}=3.$\\
			Ta sẽ chứng minh $\sqrt{a}+\sqrt{b} + \sqrt{ab} \leq 3.$\\
			Thật vậy, ta có $\heva{& \sqrt{a}\leq \dfrac{a+1}{2}\\&\sqrt{b} \leq \dfrac{b+1}{2}\\& \sqrt{ab} \leq \dfrac{a+b}{2}} \Rightarrow \sqrt{a}+\sqrt{b} + \sqrt{ab} \leq a+b+1 \, (1).$\\
			Mặt khác $\heva{&a=\sqrt{1-2x} \leq \dfrac{1+1-2x}{2}=1-x\\&b=\sqrt{1+2x} \leq \dfrac{1+1+2x}{2}=1+x } \Rightarrow a+b \leq 2 \Rightarrow a+b+1 \leq 3 \, (2).$\\
			Từ $(1)$ và $(2)$ suy ra $\sqrt{a} +\sqrt{b} +\sqrt{ab} \leq 3.$\\
			Đẳng thức xảy ra khi $\heva{& 1-2x=1\\&1+2x=1\\&a=1\\&b=1} \Leftrightarrow x=0,$ thử lại thỏa mãn.\\
			Vậy phương trình có nghiệm duy nhất $x=0.$
	
}
	
\end{vd}

\begin{bt} Giải các phương trình sau
    	\begin{itemize}
    	\item [a)] $x^3+2=3 \sqrt[3]{3x-2};$
    	\item [b)] $2x^3-1 = \sqrt[3]{\dfrac{x+1}{2}};$
    	\item [c)] $\sqrt[3]{x+8}+ \sqrt{1-x}=3;$
    	\item [d)] $\sqrt[3]{x+6}+\sqrt{x-1}=x^2-1.$
    \end{itemize}
    \loigiai{
        	\begin{itemize}
        	\item [a)] Xét phương trình $x^3+2=3 \sqrt[3]{3x-2}$. \\
        	ĐKXĐ: $x\in \mathbb{R}.$\\
        	Đặt $y=\sqrt[3]{3x-2},$ ta có $\heva{&x^3+2=3y\\&y^3+2=3x} $
        	\begin{align*}
        	&\Rightarrow x^3-y^3+3(x-y)=0 \\&\Leftrightarrow (x-y)(x^2+xy+y^2+3)=0.
        	\end{align*}
        	Do $x^2+xy+y^2+3>0$ nên $x=y$. Khi đó $x^3-3x+2=0 $
        	\begin{align*}
        		&\Leftrightarrow (x-1)^2(x+2)=0\\&\Leftrightarrow \hoac{&x=1\\&x=-2.}
        	\end{align*}
        	Vậy phương trình có 2 nghiệm là $x=1; x=-2.$
        	\item [b)] Xét phương trình $2x^3-1 = \sqrt[3]{\dfrac{x+1}{2}}.$ \\
        	ĐKXĐ  $x\in \mathbb{R}.$\\ Đặt $y= \sqrt[3]{\dfrac{x+1}{2}},$ ta có $\heva{&2x^3-1=y\\&2y^3-1=x}$
        	\begin{align*}
        	 &\Rightarrow 2(x^3-y^3)+(x-y)=0 \\&\Leftrightarrow (x-y)\left[ 2(x^2+xy+y^2)+1\right] =0 \\&\Leftrightarrow x=y.
        	\end{align*}
        	Khi đó $2x^3-x-1=0 \Leftrightarrow (x-1)(2x^2+2x+1)=0 \Leftrightarrow x=1.$\\
        	Vậy phương trình có nghiệm duy nhất $x=1.$
        	\item [c)] Xét phương trình $\sqrt[3]{x+8}+ \sqrt{1-x}=3.$\\
        	ĐKXĐ  $x \leq 1.$\\
        	Đặt $\heva{&a= \sqrt[3]{x+8}\\&b=\sqrt{1-x}\, b\geq 0},$ ta có $\heva{&a+b=3\\&a^3+b^2=9} \Rightarrow a(a+3)(a-2)=0$.\\
        	Từ đó suy ra phương trình có $3$ nghiệm là $x=-8; x=-35; x=0.$
        	\item [d)] Xét phương trình $\sqrt[3]{x+6}+\sqrt{x-1}=x^2-1.$\\
        	ĐKXĐ: $x\geq 1$, ta thấy $x=2$ là một nghiệm của phương trình.\\
        	Phương trình đã cho tương đương với $\left( \sqrt[3]{x+6} -2\right) +\left( \sqrt{x-1}-1\right)=x^2-4$
        	\begin{align*}
        	&\Leftrightarrow \dfrac{x-2}{\sqrt[3]{(x+6)^2}+2 \sqrt[3]{x+6}+4} +\dfrac{x-2}{\sqrt{x-1}+1}=(x-2)(x+2)\\&\Leftrightarrow (x-2)\left( \dfrac{1}{\sqrt[3]{(x+6)^2}+2 \sqrt[3]{x+6}+4} +\dfrac{1}{1+\sqrt{x-1}} -(x+2) \right)=0\\&\Leftrightarrow x=2.
        	\end{align*}
        	Ta có phương trình $\dfrac{1}{\sqrt[3]{(x+6)^2}+2 \sqrt[3]{x+6}+4} +\dfrac{1}{1+\sqrt{x-1}} =(x+2) \, (*)$ vô nghiệm vì $VT(*) < \dfrac{3}{4}$ và $VP(*)\geq 3.$
        \end{itemize}  
    }
\end{bt}
\begin{bt} Giải các phương trình sau
		\begin{itemize}
		\item [a)] $\sqrt[3]{7+\sqrt{x}}+ \sqrt[3]{9- \sqrt{x}}=4;$
		\item [b)] $\sqrt[3]{x-1}+ \sqrt[3]{x+4}= \sqrt[3]{2x+3};$
		\item [c)] $ \sqrt[3]{2x+1}+ \sqrt[3]{2-x} +\sqrt[3]{x+2} =\sqrt[3]{2x+5};$
		\item [d)] $\sqrt[3]{(5+x)^2}+2 \sqrt[3]{(5-x)^2}=3 \sqrt[3]{25-x^2}.$
	\end{itemize}
\loigiai{
\begin{itemize}
	\item [a)] Lập phương hai vế của phương trình $\sqrt[3]{7+\sqrt{x}}+ \sqrt[3]{9- \sqrt{x}}=4.$\\
	Đáp số: $x=1.$
	\item [b)] Lập phương hai vế của phương trình $\sqrt[3]{x-1}+ \sqrt[3]{x+4}= \sqrt[3]{2x+3}$ ta được $$(x-1)+(x+4)+3 \sqrt[3]{(x-1)(x+4)}\left( \sqrt[3]{x-1}+\sqrt[3]{x+4}\right) =2x+3 $$
	Thay $\sqrt[3]{x-1}+\sqrt[3]{x+4} $ bởi $\sqrt[3]{2x+3}$ vào PT trên ta được 
	\begin{align*}
	& 2x+3 \sqrt[3]{(x-1)(x+4)(2x+3)}=2x+3\\& \Leftrightarrow \sqrt[3]{(x-1)(x+4)(2x+3)}=0 \\& \Leftrightarrow \hoac{&x=1\\&x=-4\\&x=-\dfrac{3}{2}}.
	\end{align*}
	Thử lại thỏa mãn. Vậy phương trình có 3 nghiệm $x=1 ;x=-4; x=-\dfrac{3}{2}.$\\
	{\bf {Cách 2:}} Đặt $\heva{& a= \sqrt[3]{x-1}\\& b= \sqrt[3]{x+4}}$ ta có $a^3+b^3 =2x+3 $. \\
	Phương trình đã cho trở thành $a+b=\sqrt[3]{a^3+b^3}$
	\begin{align*}
	&\Leftrightarrow a^3+b^3+3ab(a+b)=a^3+b^3 \\&\Leftrightarrow 3ab(a+b)=0\\& \Leftrightarrow \hoac{&a=0\\&b=0\\&a+b=0.}
	\end{align*}
	 Từ đó suy ra phương trình có 3 nghiệm $x=1 ;x=-4; x=-\dfrac{3}{2}.$
	\item [c)] Xét phương trình $ \sqrt[3]{2x+1}+ \sqrt[3]{2-x} +\sqrt[3]{x+2} =\sqrt[3]{2x+5}\, (1).$
	ĐKXĐ: $x\in \mathbb{R}.$\\
	Đặt $\heva{& 2x+1=a^3 \\& 2-x=b^3\\&x+2=c^3} \Rightarrow 2x+5 =a^3+b^3+c^3.$\\
	Phương trình $(1)$ trở thành $a+b+c= \sqrt[3]{a^3+b^3+c^3}$
	\begin{align*}
	&\Leftrightarrow (a+b+c)^3=a^3+b^3+c^3\\& \Leftrightarrow 3(a+b)(b+c)(c+a)=0.
	\end{align*}
	Từ đó suy ra PT $(1)$ có hai nghiệm phân biệt $x=-3; x=-1.$
	\item [d)] Xét phương trình $\sqrt[3]{(5+x)^2}+2 \sqrt[3]{(5-x)^2}=3 \sqrt[3]{25-x^2}.$\\
	Ta thấy $x=\pm 5$ đều không là nghiệm của phương trình trên.\\
	Với $x \neq \pm 5$, chia hai vế phương trình  trên cho $\sqrt[3]{25-x^2}$, ta được $\sqrt[3]{\dfrac{5+x}{5-x}}+2\cdot \sqrt[3]{\dfrac{5-x}{5+x}}=3.$\\
	Đặt $\sqrt[3]{\dfrac{5+x}{5-x}} =y,$ phương trình trên trở thành $y+ \dfrac{2}{y}-3=0 \Leftrightarrow \hoac{&y=1\\&y=2.}$\\
	Từ đó suy ra PT có hai nghiệm phân biệt là $x=0; x= \dfrac{35}{9}.$ 
\end{itemize}

}
\end{bt}

\end{itemize}
