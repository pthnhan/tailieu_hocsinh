		Mà $ 2f(x)>1\Rightarrow 2mx-1>0 $.\\
		Ta lại có 
		\begin{align*}
		\dfrac{2f(x)-1}{2mx-1}=\dfrac{m^2x^2+\dfrac{1}{4}+m(1-x)}{f^2(x)-f(x)+m+\dfrac{1}{4}}&\Leftrightarrow\dfrac{2f(x)-1}{2mx-1}=\dfrac{(2mx-1)^2+4m}{[2f(x)-1]^2+4m} \\&\Leftrightarrow [2f(x)-1]^3+4m[2f(x)-1]=(2mx-1)^3+4m(2mx-1)\\&\Leftrightarrow 2f(x)-1=2mx-1\\&\Leftrightarrow f(x)=mx \ \mbox{ (vì hàm số } g(t)=t^3+4mt \mbox{ đồng biến trên } \mathbb{R}).
		\end{align*}
		Do đó hàm số đã cho trở thành $y=\dfrac{mx-9m+8}{x-m}$. \\
		Xét hàm số $ y=\dfrac{mx-9m+8}{x-m} $, ta có $ y'=\dfrac{-m^2+9m-8}{(x-m)^2} $.
		\begin{align*}
		\mbox{Hàm số đã cho đồng biến trên từng khoảng xác định của nó } \Leftrightarrow y'>0,\ \forall x\in D &\Leftrightarrow -m^2+9m-8>0 \\&\Leftrightarrow 1<m<8.
		\end{align*}
		Suy ra giá trị nguyên của $ m $ là $ 2,3,4,5,6,7 $.
	}
\end{ex}
