
\section{PHÉP TRỪ VÀ PHÉP CHIA}
\subsection{Tóm tắt lý thuyết}
\begin{enumerate}
\item Cho hai số tự nhiên $a$ và $b$, nếu có số tự nhiên $x$ sao cho $b + x = a$ thì ta có phép trừ $a - b = x$.\\
Điều kiện để phép trừ thực hiện được là số bị trừ lớn hơn hoặc bằng số trừ.
\item Cho hai số tự nhiên $a$ và $b$, trong đó $b \neq 0$, nếu có số tự nhiên $x$ sao cho $b\cdot x = a$ thì ta nói $a$ chia hết cho $b$ và ta có phép chia hết $a \colon b = x$.
\item Phép chia có dư:
\begin{eqnarray*}
	&\mbox{Số bị chia} & =  \mbox{Số chia} \times \mbox{Thương} + \mbox{Số dư}\\
		&a & =  b \cdot q + r, \left( 0 < r <b\right)
\end{eqnarray*}
Số dư $r$ bao giờ cũng nhỏ hơn số chia $b$.
\begin{note}
	Chú ý: Tính chất phân phối của phép nhân đối với phép cộng có thể mở rộng:
\begin{eqnarray*}
	& \left( a - b\right) \cdot c & = a \cdot c - b \cdot c\\
	& \left( a + b\right) : c & = a : c + b : c \mbox{ (nếu các phép chia đều là phép chia hết)}
\end{eqnarray*}
\end{note}
\end{enumerate}



\subsection{Các dạng toán}
\begin{dang}{Tính nhanh, tính hợp lí bằng cách áp dụng các tính chất của phép trừ, phép chia}
Tùy theo đặc điểm các thành phần trong phép tính mà ta có thể dùng các tính chất sau:
\begin{itemize}
	\item Tổng của hai số không đổi nếu ta thêm vào số hạng này và bớt đi ở số hạng kia cùng một số đơn vị.
	\item Hiệu của hai số không thay đổi nếu ta cùng thêm (hay bớt) vào số bị trừ và số trừ cùng một số đơn vị.
	\item Tích của hai số không thay đổi nếu ta nhân thừa số này và chia thừa số kia cho cùng một số tự nhiên khác $0$.
	\item Thương của hai số không đổi nếu ta nhân cả số bị chia và số chia với cùng một số tự nhiên khác $0$.
\end{itemize}
\end{dang}
\begin{vd}%[6D1B6]
Tính nhẩm:
\begin{enumEX}{2}
	\item $97 + 214$;
	\item $65 + 129$.
\end{enumEX}
\loigiai{
\begin{enumEX}{1}
	\item $97 + 214 = \left( 97 + 3\right) + \left( 214 - 3\right) = 100 + 211 = 311$.
	\item $65 + 129 = \left( 65 - 1\right) + \left( 129 + 1\right) = 64 + 130 = 194$.
\end{enumEX}
}
\end{vd}
\begin{vd}%[6D1B6]
Tính nhẩm:
\begin{enumEX}{2}
	\item $431 - 199$;
	\item $8250 : 250$.
\end{enumEX}
\loigiai{
	\begin{enumEX}{1}
		\item $431 - 199 = \left( 431 + 1\right) - \left( 199 + 1\right) = 432 - 200 = 232$.
		\item $8250 : 250 = 825 : 25 = \left( 825\cdot 4\right) : \left( 25 \cdot 4\right) = 3300 : 100 = 33$.
	\end{enumEX}
}
\end{vd}
\begin{vd}%[6D1B6]
	Tính nhẩm:
	\begin{enumEX}{2}

		\item $273 : 13$.
	\end{enumEX}
\loigiai{

		\item $42\cdot 167 - 42\cdot 67 = 42\cdot\left( 167 - 67\right) = 42\cdot 100 = 4200$;
		\item $
		273 : 13  =\left( 260 + 13\right) : 13 = 260 : 13 + 13 : 13 = 20 + 1 = 21$.
	\end{enumEX}
	}
\end{vd}

		\item $\overline{abcabc} : \overline{abc} = \left( \overline{abc000} + \overline{abc}\right)  ; \overline{abc} = 1000 + 1 = 1001$.

\begin{dang}{Toán về phép chia có dư}
	\begin{itemize}
		\item Trong phép chia có dư
		\begin{eqnarray*}
			&\mbox{Số bị chia} & =  \mbox{Số chia} \times \mbox{Thương} + \mbox{Số dư}\\
			&a & =  b \cdot q + r
		\end{eqnarray*}
	trong đó số dư $r$ nhỏ hơn số chia $b$ và lớn hơn $0$.
	\item Suy ra:
	$$b = \left( a - r\right) : q;\quad q = \left( a - r\right) : b;\quad r = a - b\cdot q$$
	\end{itemize}
\end{dang}
\begin{vd}%[6D1B6]
	Tìm số bị chia trong một phép chia có dư mà số chia, thương, số dư lần lượt là $43, 10, 26$.
	\loigiai{
	Vì số bị chia $=$ số chia $\times$ thương $+$ số dư nên số bị chia bằng $43\cdot 10 + 26 = 456$.
	}
\end{vd}
\begin{vd}%[6D1G6]
	Trong một phép chia có dư, số bị chia là $100$ và số dư là $9$. Tìm số chia và thương.
	\loigiai{
		Ta gọi số bị chia, số chia, thương và số dư lần lượt là $a, b, q, r$.\\
		Ta có $a = b \cdot q + r \quad \left( 0 < r < b\right)$.\\
		Suy ra $b\cdot q = a - r = 100 - 9 = 91$.\\
		Vì $91 = 1\cdot 91 = 7\cdot 13 = b \cdot q$.\\
		Mặt khác $b > 9$ nên ta chọn $ b = 91, q = 1$ hoặc $b = 13, q = 7$.\\
		Do đó ta có hai đáp số: $91$ và $1$ hoặc $13$ và $7$.
	}
\end{vd}
\begin{vd}%[6D1G6]
	Trong một phép chia có thương là $12$ và số dư là $19$. Hỏi số bị chia nhỏ nhất là bao nhiêu?
	\loigiai{
		\begin{center}
			Vì số dư là $19$ nên số chia nhỏ nhất là $20$. Do đó số bị chia nhỏ nhất là:\\
		$a = b\cdot q + r = 20\cdot 12 + 19 = 259$
		\end{center}
	
	}
\end{vd}

\section{LŨY THỪA VỚI SỐ MŨ TỰ NHIÊN. NHÂN HAI LŨY THỪA CÙNG CƠ SỐ}





\section{CHIA HAI LŨY THỪA CÙNG CƠ SỐ}
\subsection{Tóm tắt lý thuyết}
\begin{enumerate}
	\item Lũy thừa bậc $n$ của số $a$ là tích của $n$ thừa số bằng nhau, mỗi thừa số bằng $a$.
	$$a^n = \underbrace{a\cdot a\ldots a}_{n \mbox{ thừa số}} \quad \left( n \neq 0\right) $$
	$a$ gọi là cơ số, $n$ gọi là số mũ.\\
	Quy ước: $a^1 = a$.
	\begin{itemize}
		\item Ta gọi $a^2$ là $a$ bình phương; $a^3$ là $a$ lập phương.
		\item Số chính phương là số bằng bình phương của một số tự nhiên.
	\end{itemize}
	\item Khi nhân hai lũy thừa cùng cơ số, ta giữ nguyên cơ số và cộng các số mũ.
	$$a^m\cdot a^n = a^{m+n}$$
	\item Khi chia hai lũy thừa cùng cơ số (khác $0$), ta giử nguyên cơ số và trừ các số mũ.
	$$a^m : a^n = a^{m-n} \quad \left( a\neq 0; m \geq n\right) $$
	Quy ước: $a^0 = 1 \quad \left( a \neq 0\right) $.
	\begin{note}
		Chú ý: Mỗi số tự nhiên được viết dưới dạng tổng các lũy thừa của $10$:
		$$ \overline{abcd} = a\cdot 10^3 + b\cdot 10^2 + c\cdot 10^1 + d\cdot 10^0$$
	\end{note}
\end{enumerate}
\subsection{Các dạng toán}
\begin{dang}{Viết gọn các tích}
\end{dang}
\begin{vd}%[6D1B8]
Viết gọn các tích sau:
\begin{enumEX}{3}
\item $3\cdot 3\cdot 3 \cdot 3\cdot 3$;
\item $12\cdot 12\cdot 3\cdot 4$;	
\item $100\cdot 10\cdot 10$.
\end{enumEX}
\loigiai{
\begin{enumEX}{2}
	\item $3\cdot 3\cdot 3 \cdot 3\cdot 3 = 3^5$;
	\item $12\cdot 12\cdot 3\cdot 4 = 12\cdot 12\cdot 12 = 12^3$;	
	\item $100\cdot 10\cdot 10 = 10\cdot 10\cdot 10\cdot 10 = 10^4$ hoặc $100\cdot 10\cdot 10 = 100\cdot 100 = 100^2$.
\end{enumEX}
}
\end{vd}
\begin{vd}%[6D1B8]
Viết các số sau dưới dạng lũy thừa của $10$:
\begin{enumEX}{3}
	\item $100$;
	\item $1000$;	
	\item $1\underbrace{00\ldots 0}_{ n \mbox{ chữ số } 0}$.
\end{enumEX}
\loigiai{
\begin{enumEX}{2}
	\item $100 = 10\cdot 10 = 10^2$;
	\item $1000 = 10\cdot 10\cdot 10 = 10^3$;	
	\item $1\underbrace{00\ldots 0}_{ n \mbox{ chữ số } 0} = \underbrace{10\cdot 10\ldots 10}_{n \mbox{ thừa số}} = 10^n$.
\end{enumEX}
}
\end{vd}
\begin{vd}%[6D1B8]
	Tính giá trị của lũy thừa:
	\begin{enumEX}{2}
		\item $2^5$;
		\item $5^4$.	
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{2}
	\item $2^5 = 2\cdot 2\cdot 2\cdot 2\cdot 2 = 32$;
	\item $5^4 = 5\cdot 5 \cdot 5\cdot 5 = 625$.	
		\end{enumEX}
	}
\end{vd}

\begin{dang}{So sánh hai lũy thừa}
Tính giá trị của mỗi lũy thừa rồi sao sánh hai kết quả.
\end{dang}
\begin{vd}%[6D1K8]
	So sánh:
	\begin{enumEX}{2}
		\item $5^3$ và $3^5$;
		\item $2^4$ và $4^2$.	
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{2}
			\item 
			\begin{tabular}[t]{l}
			Ta có: $5^3 = 5\cdot 5\cdot 5 = 125$;\\
			$3^5 = 3\cdot 3\cdot 3\cdot 3\cdot 3 = 243$\\
			Vì $125 < 243$ nên $5^3 < 3^5$
			\end{tabular}
			\item 
			\begin{tabular}[t]{l}
			Ta có: $2^4 = 2\cdot 2\cdot 2\cdot 2 = 16$;\\
			$4^2 = 4\cdot 4 = 16$\\
			Vậy $2^4 = 4^2$	
			\end{tabular}
		\end{enumEX}
	}
\end{vd}
\begin{vd}%[6D1K8]
	So sánh các lũy thừa sau:\\
	\centerline{$ \left( 100 - 99\right) ^{2000} \mbox{ và} \left( 100 + 99\right) ^0$}
	\loigiai{
	$\begin{array}{ll}
	\mbox{Ta có} & \left( 100 - 99\right) ^{2000} = 1^{2000} = 1\\
	& \left( 100 + 99\right)^0 = 199^0 = 1\\
	\mbox{Vậy} & \left( 100 - 99\right) ^{2000} = \left( 100 + 99\right)^0
	\end{array}$
	
}
\end{vd}
\begin{dang}{Viết một số dưới dạng một lũy thừa với số mũ lớn hơn $1$}
\centerline{Vận dụng công thức 	$\underbrace{a\cdot a\ldots a}_{n \mbox{ thừa số}}  = a^n$.}
\end{dang}
\begin{vd}%[6D1B8]
	Viết mỗi số sau thành bình phương của một số tự nhiên: $64; 121; 225$.
		
	\loigiai{
		\begin{enumEX}{3}
			\item $ 64 = 8\cdot 8 = 8^2$;
			\item $121 = 11\cdot 11 = 11^2$;
			\item $225 = 15\cdot 15 = 15^2$.	
		\end{enumEX}
	}
\end{vd}
\begin{vd}%[6D1B8]
	Viết mỗi số sau thành lập phương của một số tự nhiên: $64; 125; 343$.
	
	\loigiai{
		\begin{enumEX}{3}
			\item $ 64 = 4\cdot 4\cdot 4 = 4^3$;
			\item $125 = 5\cdot 5\cdot 5 = 5^3$;
			\item $343 = 7\cdot 7\cdot 7 = 7^3$.	
		\end{enumEX}
	}
\end{vd}
\begin{vd}%[6D1B8]
Trong các số sau, số nào là số chính phương?
$$ 0; 1; 18; 25; 49; 81; 90; 200; 1000$$.

\loigiai{
	Số chính phương trong các số trên là:
	\begin{enumEX}{3}
		\item $0$ vì $0  =  0^2$;
		\item $1$ vì $1  =  1^2$;
		\item $25$ vì $25  =  5^2$;
		\item $49$ vì $49  =  7^2$;
		\item $81$ vì $81  =  9^2$.	
	\end{enumEX}
}
\end{vd}
\begin{dang}{Viết kết quả của phép tính dưới dạng một lũy thừa}
	Áp dụng các công thức:
	$$a^m\cdot a^n = a^{m+n};\quad  a^m : a^n = a^{m-n} \quad \left( a\neq 0; m \geq n\right)$$
\end{dang}
\begin{vd}%[6D1B8]
	Viết kết quả của các phép toán sau dưới dạng một lũy thừa:
	\begin{enumEX}{3}
		\item $7^5\cdot 7^2\cdot 7$;
		\item $2^0\cdot 2\cdot 2^4$;
		\item $5^8 : 5^2$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{3}
			\item $7^5\cdot 7^2\cdot 7 = 7^{5+2+1}=7^8$;
			\item $2^0\cdot 2\cdot 2^4 = 2^{0+1+4}=2^5$;
			\item $5^8 : 5^2 = 5^{8-2}=5^6$.
		\end{enumEX}
	}
\end{vd}
\begin{vd}%[6D1K8]
	Viết kết quả của các phép toán sau dưới dạng một lũy thừa:
	\begin{enumEX}{2}
		\item $4^2\cdot 2^5$;
		\item $1^3 + 2^3 + 3^3 + 4^3 + 5^3$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
		\item $4^2\cdot 2^5 = 2^4\cdot 2^5 = 2^9$;
		\item $1^3 + 2^3 + 3^3 + 4^3 + 5^3 = 1+ 8 + 27 + 64 + 125 = 225 = 15^2$.
		\end{enumEX}
	}
\end{vd}
\begin{dang}{Tìm số mũ của lũy thừa trong một đẳng thức}
	\begin{itemize}
		\item Viết hai vế của đẳng thức thành hai lũy thừa của cùng một cơ số.
		\item Sử dụng tính chất: Với $a \neq 0; a\neq 1$, nếu $a^m = a^m$ thì $m = n$.
	\end{itemize}
\end{dang}
\begin{vd}%[6D1K8]
Tìm số tự nhiên $n$ biết:
\begin{enumEX}{2}
	\item $3^n\cdot 3 = 243$;
	\item $7^n : 7^4 = 49$.
\end{enumEX}
\loigiai{
	\begin{enumEX}{2}
		\item    
		\begin{tabular}[t]{ll}
			Ta có & $3^n\cdot 3 = 243$\\
			Suy ra & $ 3^{n+1} = 3^5$\\
			Do đó & $ n + 1 = 5$\\
			 &$n = 4$
		\end{tabular}
		\item    
		\begin{tabular}[t]{ll}
		Ta có & $ 7^n : 7^4 = 49$\\
		Suy ra & $ 7^{n-4} = 7^2$\\
		Do đó & $ n - 4 = 2$\\
		& $n = 6$
	\end{tabular}
	\end{enumEX}
}
\end{vd}
\begin{vd}%[6D1K8]
	Tìm $n \in \mathbb{N}$ biết: $ 5\cdot 4^n + 13 = 1293$.
	\loigiai{
		
	\hspace{2cm}	\begin{tabular}{ll}
			Ta có & $ 5\cdot 4^n + 13 = 1293$\\
			Suy ra & $ 5 \cdot 4^n = 1293 - 13 = 1280$\\
			& $4^n = 1280 \colon 5 = 256$\\
			hay & $4^n = 4^4$\\
			Do đó & $n = 4$
		\end{tabular}
	}
\end{vd}
\begin{dang}{Tìm cơ số của lũy thừa trong một đẳng thức}
	\begin{itemize}
		\item Viết hai vế của đẳng thức thành hai lũy thừa của cùng một cơ số.
		\item Sử dụng tính chất: Với $n \neq 0, a, b$ là số tự nhiên, nếu $a^n = b^n$ thì $a = b$.
	\end{itemize}
\end{dang}
\begin{vd}%[6D1B8]
	Tìm $x \in \mathbb{N}$ biết: $ x^4 = 81$.
	\loigiai{
		
	\hspace{2cm}	\begin{tabular}{ll}
			Ta có & $ x^4 = 81$\\
			Suy ra & $ x^4 = 3^4$\\
			Do đó & $x = 3$
		\end{tabular}
	
	}
\end{vd}
\begin{vd}%[6D1K8]
	Tìm $x \in \mathbb{N}$ biết: $\dfrac{2\cdot x^3}{3^2} = 48$.
	\loigiai{
		
	\hspace{2cm}	\begin{tabular}{ll}
			Ta có & $\dfrac{2\cdot x^3}{3^2} = 48$\\
			Suy ra & $\dfrac{2\cdot x^3}{9} = 48$\\
			hay & $2\cdot 3^x = 48\cdot 9 = 432$\\
			& $x^3 = 432 : 2 = 216$\\
			& $x^3 = 6^3$\\

		\end{tabular}
		
	}
\end{vd}

\begin{bt}%[6D1B8]
Viết gọn các tích sau bằng cách dùng lũy thừa:
\begin{enumEX}{3}
	\item $4\cdot 8\cdot 8 \cdot 2$;
	\item $3\cdot 3\cdot 3\cdot 9$;
	\item $2\cdot 2\cdot 2\cdot 5\cdot 5$.
\end{enumEX}
\loigiai{
\begin{enumEX}{2}
\item $4\cdot 8\cdot 8 \cdot 2 = 8\cdot 8\cdot 8 = 8^3$;
\item $3\cdot 3\cdot 3\cdot 9 = 3\cdot 3\cdot 3\cdot 3^2 = 3^5$;
\item $2\cdot 2\cdot 2\cdot 5\cdot 5= 2^3\cdot 5^2$.
\end{enumEX}
}
\end{bt}
\begin{bt}%[6D1B8]
Tính giá trị của các lũy thừa:
\begin{enumEX}{4}
	\item $2^7$;
	\item $3^4$;
	\item $5^3$;
	\item $10^5$.
\end{enumEX}
\loigiai{
\begin{enumEX}{2}
	\item $2^7 = 2\cdot 2\cdot 2\cdot 2\cdot 2\cdot 2\cdot 2\cdot 2 = 128$;
	\item $3^4 = 3\cdot 3\cdot 3\cdot 3 =81$;
	\item $5^3 = 5\cdot 5\cdot 5 = 125$;
	\item $10^5 = 10\cdot 10\cdot 10\cdot 10\cdot 10 = 100000$.
\end{enumEX}
}
\end{bt}
\begin{bt}%[6D1B8]
Trong các số sau, số nào là lũy thừa của một số tự nhiên với số mũ lớn hơn $1$?
$$16; 32; 49; 99; 100$$
	\loigiai{
		Các số là lũy thừa của một số tự nhiên với số mũ lớn hơn $1$ là:
	\begin{enumEX}{3}
	\item $16 = 4\cdot 4 = 4^2 $ hoặc $16 = 2\cdot 2\cdot 2\cdot 2 = 2^4$;
	\item $49 = 7\cdot 7 = 7^2$;
	\item $100 = 10\cdot 10 = 10^2$.
\end{enumEX}	
	}
\end{bt}
\begin{bt}%[6D1B8]
	Viết kết quả các phép tính dưới đây dưới dạng một lũy thừa:
	\begin{enumEX}{4}
		\item $5\cdot 5^2\cdot 5^3$;
		\item $x\cdot x^4$;
		\item $7^5 : 7^2$;
		\item $10^{12} : 10^3$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{2}
		\item $5\cdot 5^2\cdot 5^3 = 5^{1+2+3} = 5^6$;
		\item $x\cdot x^4 = x^{1+4} = x^5$;
		\item $7^5 : 7^2 = 7^{5-2} = 7^3$;
		\item $10^{12} : 10^3 = 10^{12-3} = 10^9$.
	\end{enumEX}
	}
\end{bt}

\begin{bt}%[6D1K8]

	\begin{enumEX}{3}
		\item $6^2 + 8^2$;
		\item $13^2 - 12^2$;

\begin{bt}%[6D1K8]

			& $n = 2$
		\end{tabular}
	\end{enumEX}
	}
\end{bt}
