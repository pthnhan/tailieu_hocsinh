\section{ĐƠN THỨC}
\subsection{Tóm tắt lý thuyết}
\subsubsection{Đơn thức}
\begin{dn}
Đơn thức là biểu thức đại số chỉ gồm một số, hoặc một biến, hoặc một tích giữa các số và các biến.\\
Số $0$ được gọi là đơn thức không.
\end{dn}
\textbf{\textit{Nhận xét:}} Một biểu thức đại số nếu có chứa một trong các phép toán $+$, $-$, $:$ (số cho biến hoặc biến cho biến) thì sẽ không phải là đơn thức.

\subsubsection{Đơn thức thu gọn}
Đơn thức thu gọn là đơn thức chỉ gồm tích của một số với các biến mà mỗi biến đã được nâng lên luỹ thừa với số mũ nguyên dương.\\
Trong đó:\\
	$\bullet$ Số nói trên được gọi là hệ số của đơn thức thu gọn.\\
	$\bullet$ Phần còn lại được gọi là phần biến của đơn thức thu gọn.\\
\textbf{\textit{Chú ý:}}\\
$\bullet$ Ta cũng coi một số là đơn thức thu gọn.\\
$\bullet$ Trong đơn thức thu gọn, mỗi biến chỉ được viết một lần. Thông thường, khi viết đơn thức thu gọn ta viết hệ số trước, phần biến sau và các biến được viết theo bảng chữ cái.
\subsubsection{Bậc của một đơn thức}	
Bậc của đơn thức có hệ số khác $0$ là tổng của số mũ của tất cả các biến có trong đơn thức đó.\\
\textbf{\textit{Chú ý:}} Một số là đơn thức thu gọn có bậc bằng $0$.
\subsubsection{Nhân hai đơn thức}
Để nhân hai đơn thức, ta nhân các hệ số với nhau và nhân các phần biến với nhau.

\textbf{\textit{Nhận xét:}}\\
$\bullet$ Tích của hai đơn thức là một đơn thức.\\
$\bullet$ Mỗi đơn thức đều có thể viết thành một đơn thức thu gọn, thí dụ: $$3x^2y(-4)xy^3(-8)y=96x^3y^5.$$

\subsection{Phương pháp giải toán}
\begin{vd}%[Đào Trung Kiên, dự án EX-C2-L7-2018]%[7D4B3]
	Trong các biểu thức sau, biểu thức nào là đơn thức (Vì sao?) và nếu là đơn thức hãy chỉ ra bậc của nó:
	\begin{listEX}[4]
		\item $-\dfrac{1}{2}$.
		\item $a$.
		\item $2xy^2\cdot 3z$.
		\item $2x-y$
	\end{listEX}
\loigiai{
\begin{listEX}
	\item $-\dfrac{1}{2}$ là đơn thức, vì nó chỉ gồm một số. Đơn thức này có bậc bằng $0$.
	\item $a$ là đơn thức, vì nó chỉ gồm một biến. Đơn thức này có bậc bằng $1$.
	\item $2xy^2\cdot 3z$ là đơn thức, vì nó là một tích giữa các số và các biến. Bậc của đơn thức này bằng $4$.
	\item $2x-y$ không là đơn thức, vì nó chứa phép trừ
\end{listEX}
}
\end{vd}

\begin{vd}%[Đào Trung Kiên, dự án EX-C2-L7-2018]%[7D4B3]
	Cho hai chữ số $x, y$. hãy lập hai biểu thức đại số, trong đó một biểu thức là đơn thức còn biểu thức kia không là đơn thức.
	\loigiai{Với hai chữ số $x$, $y$ ta có:\\
	$\bullet$ Đơn thức là $2xy$ (hoặc $x$, $3y$, $4x^2y^3$, $\cdots$)\\
$\bullet$ $x-y$ (hoặc $x+2y$, $\dfrac{1}{x}+\dfrac{1}{y}, \dfrac{x}{y},\cdots$) không là đơn thức. }
\end{vd}

\begin{vd}%[Đào Trung Kiên, dự án EX-C2-L7-2018]%[7D4B3]
	Cho hai chữ số $x$, $y$. Hãy lập hai đơn thức thu gọn, trong đó một là đơn thức bậc $4$ còn một là đơn thức bậc $6$.
\loigiai{Với hai chữ số $x$, $y$ ta có:\\
	$\bullet$ Đơn thức thu gọn bậc $4$ là $6x^3y$ (hoặc $4x^4$, $3y^4$, $2xy^3$, $\cdots$)\\
	$\bullet$ Đơn thức thu gọn bậc $6$ là $5xy^5$ (hoặc $x^6$, $3y^6, 4x^3y^3,\cdots$).}
\end{vd}

\begin{vd}%[Đào Trung Kiên, dự án EX-C2-L7-2018]%[7D4K3]
	Cho đơn thức $3xy^2z^3(-2xy^4)$. Thu gọn đơn thức và chỉ ra hệ số cùng bậc của nó.
\loigiai{Viết lại đơn thức dưới dạng: $$3xy^2z^3(-2xy^4)=[3\cdot(-2)]\cdot(xy^2z^3\cdot xy^4)=-6\cdot(x\cdot x)(y^2\cdot y^4)\cdot z^3=-6x^2y^6z^3.$$
Như vậy đơn thức có:\\
$\bullet$ Hệ số bằng $-6$.\\
$\bullet$ Có bậc bằng $2+6+3=11$.}
\end{vd}

\begin{vd}%[Đào Trung Kiên, dự án EX-C2-L7-2018]%[7D4B3]
	Cho hai đơn thức $\dfrac{2}{3}a^2b$ và $\dfrac{3}{2}ab^2c$. Tính tích của hai đơn thức và xác định hệ số, bậc của đơn thức thu được.
\loigiai{Ta có: $\dfrac{2}{3}a^2b\cdot\dfrac{3}{2}ab^2c=\left(\dfrac{2}{3}\cdot\dfrac{3}{2}\right)\cdot(a^2b\cdot ab^2c)=(a^2\cdot a)\cdot(b\cdot b^2)\cdot c=a^3b^3c$.\\
Khi đó, đơn thức $a^3b^3c$ có hệ số bằng $1$ và có bậc bằng $7$.}
\end{vd}

\textbf{\textit{Nhận xét:}}
\begin{enumEX}{1}
	\item Tích của hai đơn thức là một đơn thức.
	\item Mỗi đơn thức đều có thể viết thành một đơn thức thu gọn, thí dụ: $$3x^2y(-4)xy^3(-8)y=96x^3y^5.$$
\end{enumEX}

\begin{vd}%[Đào Trung Kiên, dự án EX-C2-L7-2018]%[7D4K3]
	Cho hai đơn thức $-x^8y^8z^9$ và $6xy^3$. Tính tích của hai đơn thức và xác định hệ số, bậc của đơn thức thu được.
	\loigiai{Ta có: $-x^8y^8z^9\cdot 6xy^3=(-1\cdot 6)\cdot (x^8y^8z^9\cdot xy^3)=-6(x^8\cdot x)\cdot(y^8\cdot y^3)\cdot z^9=-6x^9y^{11}z^9$.\\
		Khi đó, đơn thức $-6x^9y^{11}z^9$ có hệ số bằng $-6$ và có bậc bằng $9+11+9=29$.}
\end{vd}

\begin{vd}%[Đào Trung Kiên, dự án EX-C2-L7-2018]%[7D4K3]
	Thu gọn các đơn thức rồi chỉ ra phần hệ số và bậc của chúng:
	\begin{listEX}[2]
		\item $6x\cdot(-8x^2y)\cdot(9x^3y^2z)$.
		\item $2x^6yz^4\cdot\left(-\dfrac{1}{4}y^2z^3\right)\cdot(2xz^6)$.
	\end{listEX}
	\loigiai{
		\begin{enumerate}
		\item Viết lại đơn thức dưới dạng: $$[6\cdot(-8)\cdot 9]\left(x\cdot x^2y\cdot x^3y^2z\right) =-432\cdot(x\cdot x^2\cdot x^3)(y\cdot y^2)\cdot z=-432x^6y^3z.$$
			Như vậy, đơn thức có hệ số bằng $-432$ và có bậc bằng $10$.
		\item Viết lại đơn thức dưới dạng: $$\left[2\cdot \left(-\dfrac{1}{4}\right)\cdot 2\right]\cdot(x^6x)\cdot (y\cdot y^2)\cdot(z^4\cdot z^3\cdot z^6)=-x^7y^3z^{13}.$$
				Như vậy, đơn thức có hệ số bằng $-1$ và có bậc bằng $23$.
	\end{enumerate}
}
\end{vd}

\begin{vd}%[Đào Trung Kiên, dự án EX-C2-L7-2018]%[7D4K3]
	Viết biểu thức tính diện tích hình chữ nhật, biết hình chữ nhật có chiều dài gấp hai lần chiều rộng.
		\loigiai{Giả sử hình chữ nhật có chiều rộng bằng $x$, suy ra chiều dài bằng $2x$ do đó biểu thức diện tích: $$2x\cdot x=2x^2.$$}
\end{vd}

\begin{bt}%[Đào Trung Kiên, dự án EX-C2-L7-2018]%[7D3K3]
	Cho ba chữ số $x$, $y$, $z$.
	\begin{enumerate}
		\item Hãy lập hai biểu thức đại số, trong đó một biểu thức là đơn thức còn một biểu thức không là đơn thức.
		\item Hãy lập hai đơn thức thu gọn, trong đó một là đơn thức bậc $8$ còn một là đơn thức bậc $9$.
	\end{enumerate}
	\loigiai{
\begin{enumerate}
	\item Với ba chữ số $x, y, z$ ta có:\\
	$\bullet$ Đơn thức là $xyz^2$ (hoặc $x$, $6y$, $8z$, $9xyz^2, \cdots$)\\
	$\bullet$ $xy+z$ (hoặc $x-6yz$, $\dfrac{1}{x}$, $\dfrac{9x}{yz},\cdots$) không là đơn thức.
	\item Với ba chữ số $x, y, z$ ta có:\\
	$\bullet$ Đơn thức thu gọn bậc $8$ là $x^6yz$ (hoặc $2x^8$, $8y^8$, $9z^8$, $x^2y^6, \cdots$)\\
	$\bullet$ Đơn thức thu gọn bậc $9$ là $6x^4y^2z^3$ (hoặc $x^9$, $7y^9$, $9z^9$, $x^3z^6, \cdots$)
\end{enumerate}	
}
\end{bt}

\begin{bt}%[Đào Trung Kiên, dự án EX-C2-L7-2018]%[7D3K3]

		\item $0{,}12^2abc^{11}$.

	\item Đơn thức $0{,}12^2abc^8$ có phần hệ số là $0{,}12^2$ và phần biến là $abc^{11}$.\\
	Thay $a=1$, $b=2$ và $c=-1$ vào $0{,}12^2abc^{11}$ ta được $$0{,}12^2\cdot 1\cdot 2\cdot(-1)^{11}=0{,}0288.$$

	Khi đó đơn thức $-5a^8b^3c^6d^9$ có hệ số bằng $-5$ và có bậc bằng $26$.
	\item Ta có: $2^4xy^4\cdot\left(\dfrac{3}{4}x^6y^8z^9\right)=\left(2^4\cdot\dfrac{3}{4}\right)\cdot\left(xy^4\cdot x^6y^8z^9\right)=12x^7y^{12}z^9$.\\
	Khi đó đơn thức $12x^7y^{12}z^9$ có hệ số bằng $12$ và có bậc bằng $28$.
\end{listEX}	
}
\end{bt}

\section{ĐƠN THỨC ĐỒNG DẠNG}
\subsection{Tóm tắt lý thuyết}
\subsubsection{Đơn thức đồng dạng}
Hai đơn thức đồng dạng là hai đơn thức có hệ số khác $0$ và có cùng phần biến.\\
\textbf{\textit{Chú ý:}} Tất cả các số khác $0$ được coi là những đơn thức đồng dạng.
\subsubsection{Cộng, trừ các đơn thức đồng dạng}
Để cộng (hoặc trừ) hai đơn thức đồng dạng, ta cộng (hoặc trừ) các hệ số với nhau và giữ nguyên phần biến.\\
\textbf{\textit{Chú ý:}} Phép cộng, trừ các đơn thức đồng dạng còn được sử dụng trong bài toán tính giá trị biểu thức.

\subsection{Phương pháp giải toán}
\begin{vd}%[Đào Trung Kiên, dự án EX-C2-L7-2018]%[7D4B4]
Trong các đơn thức sau, hãy chỉ ra đơn thức đồng dạng với đơn thức $6ab^6$.
\begin{listEX}[4]
	\item  $-ab^6$.
	\item $\dfrac{2}{ab^6}$.
	\item $\dfrac{1}{4}ab^6$.
	\item  $ab^6-a$.
\end{listEX}
\loigiai{
	$\bullet$ Các đơn thức $-ab^6$, $\dfrac{1}{4}ab^6$ đồng dạng với đơn thức $6ab^6$.\\
	$\bullet$ Các biểu thức $\dfrac{2}{ab^6}$, $ab^6-a$ không đồng dạng với đơn thức $6ab^6$.

}
\end{vd}

	\loigiai{
	$\bullet$ Các đơn thức $-2xy^2z$, $\dfrac{3}{2}xy^2z$, $8xzy^2$ đồng dạng với nhau.\\
	$\bullet$ Các đơn thức $6x^2yz$, $\dfrac{3}{4}x^2yz$ đồng dạng với nhau.

\begin{vd}%[Đào Trung Kiên, dự án EX-C2-L7-2018]%[7D4B4]
	Cho biểu thức $3x\cdot 2xy-\dfrac{2}{3}x^2y-4\cdot x^2\cdot \dfrac{1}{3}y$.
	\begin{listEX}
	\item Thực hiện đơn giản biểu thức.
	\item Tính giá trị của biểu thức với $x=-2$, $y=\dfrac{1}{8}$.
	\end{listEX}
\loigiai{
\begin{listEX}
	\item Ta có $3x\cdot 2xy-\dfrac{2}{3}x^2y-4\cdot x^2\cdot \dfrac{1}{3}y=6x^2y-\dfrac{2}{3}x^2y-\dfrac{4}{3}x^2y=\left(6-\dfrac{2}{3}-\dfrac{4}{3}\right)x^2y=4x^2y$.

}
\end{vd}

\textit{Cách 2:} Giả sử hình chữ nhật có chiều dài là $x$, suy ra chiều rộng bằng $x-6$. Khi đó, chu vi của hình chữ nhật được cho bởi: $$2(x+x-6)=2(2x-6)=4x-12.$$}

		\item Thực hiện đơn giản biểu thức.

		\end{listEX}
	}
