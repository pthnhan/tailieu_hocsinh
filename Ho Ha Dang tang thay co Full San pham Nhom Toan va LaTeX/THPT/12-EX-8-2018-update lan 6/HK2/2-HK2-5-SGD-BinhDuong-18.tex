Cho tích phân $I=\displaystyle\int\limits_{0}^{\frac{\pi}{2}}\sqrt{2+\cos x}\cdot \sin x\mathrm{\,d} x$. Nếu đặt $t=2+\cos x$ thì kết quả nào sau đây đúng?

{$I=\displaystyle\int\limits_{3}^{2}\sqrt{t}\mathrm{\,d} t$}
{\True $I=\displaystyle\int\limits_{2}^{3}\sqrt{t}\mathrm{\,d} t$}
{$I=2\displaystyle\int\limits_{3}^{2}\sqrt{t}\mathrm{\,d} t$}
{$I=\displaystyle\int\limits_{0}^{\frac{\pi}{2}}\sqrt{t}\mathrm{\,d} t$}

Đổi cận
\begin{align*}
& x=0\Rightarrow t=3\\
& x=\dfrac{\pi}{2}\Rightarrow t=2.
\end{align*}
Vậy tích phân đã cho trở thành
$$I=\displaystyle\int\limits_{3}^{2}\sqrt{t}(-\mathrm{\,d} t)=\displaystyle\int\limits_{2}^{3}\sqrt{t}\mathrm{\,d} t.$$

\begin{ex}%[Thi HK2, Sở GD Bình Dương, 2018]%[2D3K1-2]%[Trần Bá Huy, 12EX-8-2018]
Hàm số $f(x)=x\sqrt{x+1}$ có một nguyên hàm là $F(x)$. Nếu $F(0)=2$ thì $F(3)$ bằng
\choice
{$\dfrac{116}{15}$}
{\True $\dfrac{146}{15}$}
{$\dfrac{886}{105}$}
{$3$}
\loigiai{
{\bf Cách 1.} Ta có $F(x)=\displaystyle\int f(x)\mathrm{\,d} x=\displaystyle\int x\sqrt{x+1}\mathrm{\,d} x$.\\
Đặt $t=\sqrt{x+1}\Rightarrow t^2=x+1$. Suy ra $x=t^2-1$ và $\mathrm{\,d} x=2t\mathrm{\,d} t$.
$$F(x)=\displaystyle\int(t^2-1)\cdot t\cdot 2t\mathrm{\,d} t=\displaystyle\int(2t^4-2t^2)\mathrm{\,d} t=\dfrac{2t^5}{5}-\dfrac{2t^3}{3}+C=\dfrac{2(x+1)^2\sqrt{x+1}}{5}-\dfrac{2(x+1)\sqrt{x+1}}{3}+C.$$
Từ $$F(0)=2\Leftrightarrow \dfrac{2}{5}-\dfrac{2}{3}+C=2\Leftrightarrow C=\dfrac{34}{15}.$$
Vậy $F(x)=\dfrac{2(x+1)^2\sqrt{x+1}}{5}-\dfrac{2(x+1)\sqrt{x+1}}{3}+\dfrac{34}{15}$ nên $F(3)=\dfrac{2\cdot 4^2\cdot 2}{5}-\dfrac{2\cdot 4\cdot 2}{3}+\dfrac{34}{15}=\dfrac{146}{15}$.\\
{\bf Cách 2.} Ta có $F(3)-F(0)=\displaystyle\int\limits_{0}^{3}f(x)\mathrm{\,d} x$ nên $F(3)=F(0)+\displaystyle\int\limits_{0}^{3}f(x)\mathrm{\,d} x=2+\displaystyle\int\limits_{0}^{3}x\sqrt{x+1}\mathrm{\,d} x$.\\
Tính tích phân $I=\displaystyle\int\limits_{0}^{3}x\sqrt{x+1}\mathrm{\,d} x$.\\
Đặt $t=\sqrt{x+1}\Rightarrow t^2=x+1$. Suy ra $x=t^2-1$ và $\mathrm{\,d} x=2t\mathrm{\,d} t$.\\
Đổi cận
\begin{align*}
& x=0\Rightarrow t=1\\
& x=3\Rightarrow t=2.
\end{align*}
Từ đó $$I=\displaystyle\int\limits_{1}^{2}(t^2-1)\cdot t\cdot 2t\mathrm{\,d} t=\displaystyle\int\limits_{1}^{2}(2t^4-2t^2)\mathrm{\,d} t=\left(\dfrac{2t^5}{5}-\dfrac{2t^3}{3}\right)\Bigg|_1^2=\dfrac{116}{15}.$$
Vậy $F(3)=2+\dfrac{116}{15}=\dfrac{146}{15}$.
}
\end{ex}
