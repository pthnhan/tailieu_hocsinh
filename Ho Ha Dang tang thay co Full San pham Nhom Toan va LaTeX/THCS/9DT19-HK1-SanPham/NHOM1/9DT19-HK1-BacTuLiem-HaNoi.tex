	Cho biểu thức $ A=\dfrac{\sqrt{x}-2}{\sqrt{x}} $ với $ x>0 $.
	\begin{enumerate}
		\item Tính giá trị của biểu thức $ A $ khi $ x=16 $.
		\item Rút gọn biểu thức $ P=A\cdot\left(\dfrac{1}{\sqrt{x}+2}+\dfrac{1}{\sqrt{x}-2}\right) $ với $ x>0;x\neq 4 $.
		\item Tìm các giá trị của $ x $ để $ P>\dfrac{1}{3} $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Với $ x=16>0 $, ta có
			\[A=\dfrac{\sqrt{16}-2}{\sqrt{16}}=\dfrac{4-2}{4}=\dfrac{1}{2}.\]
			\item Với $ x>0 $; $ x\neq 4 $, ta có
			\begin{eqnarray*}
				P&=&\dfrac{\sqrt{x}-2}{\sqrt{x}}\cdot\left(\dfrac{1}{\sqrt{x}+2}+\dfrac{1}{\sqrt{x}-2}\right)\\
				&=&\dfrac{\sqrt{x}-2}{\sqrt{x}}\cdot \left[\dfrac{\sqrt{x}-2+\sqrt{x}+2}{\left(\sqrt{x}+2\right)\left(\sqrt{x}-2\right)}\right]\\
				&=&\dfrac{\left(\sqrt{x}-2\right)\cdot 2\sqrt{x}}{\sqrt{x}\left(\sqrt{x}+2\right)\left(\sqrt{x}-2\right)}=\dfrac{2}{\sqrt{x}+2}.
			\end{eqnarray*}
		\item Ta có $ P>\dfrac{1}{3}\Leftrightarrow\dfrac{2}{\sqrt{x}+2}>\dfrac{1}{3}\Leftrightarrow \sqrt{x}<4\Leftrightarrow 0<x<16  $.\\ Kết hợp với điều kiện xác định của $ P $, ta được $ 0<x<16;x\neq 4 $.
		\end{enumerate}	}
\end{bt}

		\begin{listEX}[2]
			\item $ \sqrt{x^2-4x+4}=1 $
			\item $ \sqrt{x^2-3x}-\sqrt{x-3}=0 $
		\end{listEX}
	\loigiai{
		 \begin{enumerate}
				\item $ \sqrt{x^2-4x+4}=1 \Leftrightarrow\sqrt{(x-2)^2}=1\Leftrightarrow|x-2|=1\Leftrightarrow\hoac{&x-2=1\\&x-2=-1}\Leftrightarrow\hoac{&x=3\\&x=1.} $
				\item Điều kiện $ x\ge 3 $.\\
				Phương trình tương đương với 
				\begin{eqnarray*}
				 \sqrt{x^2-3x}=\sqrt{x-3}&\Leftrightarrow & x^2-3x=x-3\\ &\Leftrightarrow & x(x-3)-(x-3)=0\\ &\Leftrightarrow & (x-3)(x-1)=0\\ &\Leftrightarrow&\hoac{&x=1\,(\text{loại})\\&x=3.}
				\end{eqnarray*}
			\end{enumerate}
		}

	Cho $ x,y $ là các số dương và $ \dfrac{1}{x}+\dfrac{4}{y}=1 $.\\Tìm giá trị nhỏ nhất của biểu thức $ P=x+y $.
	\loigiai{
Ta có $$ P=x+y=(x+y)\left(\dfrac{1}{x}+\dfrac{4}{y}\right)=5+\dfrac{y}{x}+\dfrac{4x}{y}. $$ 
Áp dụng BĐT Cauchy, ta có
\[\dfrac{y}{x}+\dfrac{4x}{y}\ge 2\sqrt{\dfrac{y}{x}\cdot\dfrac{4x}{y}}=4, \]
dấu \lq\lq $ = $ \rq\rq xảy ra khi và chỉ khi \[\dfrac{y}{x}=\dfrac{4x}{y}\text{ hay } y=2x.\]
Do đó, \[P\ge 5+4=9. \]
Vậy $ P_{\min}=9 $, tại $ x=3 $ và $ y=6 $.	
}
