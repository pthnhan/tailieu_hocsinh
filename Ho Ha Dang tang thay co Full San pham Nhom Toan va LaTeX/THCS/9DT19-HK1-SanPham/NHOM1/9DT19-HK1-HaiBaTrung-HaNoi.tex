	Thực hiện phép tính $A=2\sqrt{75}-8\sqrt{27}+4\sqrt{48}$.
	\loigiai{
		Ta có $2\sqrt{75}-8\sqrt{27}+4\sqrt{48}=2\cdot5\sqrt{3}-8\cdot 3\sqrt{3}+4\cdot 4\sqrt{3}=(10-24+16)\sqrt{3}=2\sqrt{3}$.
	}

\begin{bt}%[HK1-9, Hai Bà Trưng, Hà Nội,1819]%[Lê Minh An, dự án 9DT19]%[9D1B3]
	Thực hiện phép tính $B=\dfrac{\sqrt{15}-\sqrt{5}}{\sqrt{3}-1}+\sqrt{(2-\sqrt{5})^2}-2 \sqrt{5}$.
	\loigiai{
	Ta có $B=\dfrac{\sqrt{5}\left(\sqrt{3}-1\right)}{\sqrt{3}-1}+\sqrt{5}-2-2\sqrt{5}=\sqrt{5}+\sqrt{5}-2-2\sqrt{5}=-2$.	
	}
\end{bt}

\begin{bt}%[HK1-9, Hai Bà Trưng, Hà Nội,1819]%[Lê Minh An, dự án 9DT19]%[9D1B8]
	Cho hai biểu thức $A=\dfrac{\sqrt{x}+1}{\sqrt{x}-2}$ và $B=\dfrac{\sqrt{x}+2}{\sqrt{x}-3}+\dfrac{\sqrt{x}-8}{x-5 \sqrt{x}+6}$ với $x \geq 0$, $x \neq 4$, $x \neq 9$.
	\begin{enumerate}
		\item Tính giá trị biểu thức $A$ khi $x=\dfrac{1}{4}$.
		\item Rút gọn biểu thức $B$.
		\item Tìm tất cả các giá trị nguyên của $x$ để $B<A$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Với $x=\dfrac{1}{4}\Rightarrow\sqrt{x}=\dfrac{1}{2}$. Thay vào biểu thức $A$ ta được
			$A=\dfrac{\dfrac{1}{2}+1}{\dfrac{1}{2}-2}=\dfrac{3}{2}:\dfrac{-3}{2}=-1$.
			\item Với $x \geq 0$, $x \neq 4$, $x \neq 9$ ta có
			\begin{eqnarray*}
				B & = & \dfrac{\sqrt{x}+2}{\sqrt{x}-3}+\dfrac{\sqrt{x}-8}{\left(\sqrt{x}-2\right)\left(\sqrt{x}-3\right)}\\
				& = & \dfrac{\left(\sqrt{x}+2\right)\left(\sqrt{x}-2\right)+\sqrt{x}-8}{\left(\sqrt{x}-2\right)\left(\sqrt{x}-3\right)}\\
				& = & \dfrac{x+\sqrt{x}-12}{\left(\sqrt{x}-2\right)\left(\sqrt{x}-3\right)}=\dfrac{\left(\sqrt{x}+4\right)\left(\sqrt{x}-3\right)}{\left(\sqrt{x}-2\right)\left(\sqrt{x}-3\right)}=\dfrac{\sqrt{x}+4}{\sqrt{x}-2}.
			\end{eqnarray*}
		\item Với $x \geq 0$, $x \neq 4$, $x \neq 9$ ta có
		\begin{eqnarray*}
			B<A & \Leftrightarrow &  \dfrac{\sqrt{x}+4}{\sqrt{x}-2}<\dfrac{\sqrt{x}+1}{\sqrt{x}-2}\Leftrightarrow \dfrac{\sqrt{x}+4}{\sqrt{x}-2}-\dfrac{\sqrt{x}+1}{\sqrt{x}-2}<0\\
			& \Leftrightarrow & \dfrac{3}{\sqrt{x}-2}<0\Leftrightarrow \sqrt{x}-2<0\Leftrightarrow x<4.
		\end{eqnarray*}
		Vậy $0\le x<4$ thì $B<A$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[HK1-9, Hai Bà Trưng, Hà Nội,1819]%[Lê Minh An, dự án 9DT19]%[9D2B4]%[9D2K3]
	Cho hàm số $y=(m+1)x+3$ (với $m\neq -1$) có đồ thị là đường thẳng $(d)$.
	\begin{enumerate}
		\item Vẽ đồ thị hàm số khi $m=2$.
		\item Tìm $m$ để $(d)$ song song với đường thẳng $(d')\colon y=-2x+1$.
		\item Tìm $m$ để $(d)$ cắt hai trục tọa độ $Ox$, $Oy$ tạo thành một tam giác có diện tích bằng $9$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item \immini
			{
				Với $m=2$ thì $y=3x+3$\quad $(d)$.
				\begin{itemize}
					\item Cho $x=0\Rightarrow y=3\Rightarrow A(0;3)\in (d)$.
					\item Cho $y=0\Rightarrow x=-1\Rightarrow B(-1;0)\in (d)$.
				\end{itemize}
			Đồ thị hàm số là đường thẳng $AB$.
			}
			{
				\begin{tikzpicture}[scale=0.8, line join=round, line cap=round,>=stealth,thick]
				\tikzset{label style/.style={font=\footnotesize}}
				%%Nhập giới hạn đồ thị và hàm số cần vẽ
				\def \xmin{-2}
				\def \xmax{2}
				\def \ymin{-1}
				\def \ymax{4}
				\def \hamso{3*\x+3}
				
				\draw[->] (\xmin,0)--(\xmax,0) node[below left] {$x$};
				\draw[->] (0,\ymin)--(0,\ymax) node[below left] {$y$};
				\draw (0,0) node [below right] {$O$};
				\begin{scope}
				\clip (\xmin+0.01,\ymin+0.01) rectangle (\xmax-0.01,\ymax-0.01);
				\draw[samples=350,domain=\xmin+0.01:\xmax-0.01,smooth,variable=\x] plot (\x,{\hamso});
				\end{scope}
				\draw (0,3)node[right]{$3$};
				\draw (-1,0)node[above left]{$-1$};
				\end{tikzpicture}

			}
		\item $(d)\parallel (d')\colon y=-2x+1\Leftrightarrow \heva{& m+1=-2 \\ & 3\neq 1}\Leftrightarrow m=-3$.
		\item Gọi $M=(d)\cap  Ox$, cho $y=0\Rightarrow x=\dfrac{-3}{m+1}\Rightarrow M\left(\dfrac{-3}{m+1};0\right)\Rightarrow OM=\dfrac{3}{|m+1|}$.\\
		Gọi $N=(d)\cap Oy$, cho $x=0\Rightarrow y=3\Rightarrow N(0;3)\Rightarrow ON=3$.\\
		Vì $M\in Ox$, $N\in Oy$ nên $\triangle OMN$ vuông tại $O$.\\
		$\Rightarrow S_{OMN}=\dfrac{1}{2}\cdot OM\cdot ON\Leftrightarrow \dfrac{9}{2|m+1|}=9\Leftrightarrow 2|m+1|=1\Leftrightarrow\hoac{& m=-\dfrac{1}{2} \\ & m=-\dfrac{3}{2}.}$
		\end{enumerate}
		
	}
\end{bt}

\begin{bt}%[HK1-9, Hai Bà Trưng, Hà Nội,1819]%[Lê Minh An, dự án 9DT19]%[9H2B1]%[9H2B5]%[9D2G5]
	Cho đường tròn $(O;R)$ đường kính $AB$. Điểm $C$ thuộc đường tròn sao cho $AC>CB$; $C\neq A,B$. Kẻ $CH\perp AB$ ($H\in AB$), $OI\perp AC$ ($I\in AC$).
	\begin{enumerate}
		\item Chứng minh bốn điểm $C$, $H$, $O$, $I$ cùng thuộc một đường tròn.
		\item Kẻ tiếp tuyến $Ax$ của đường tròn $(O;R)$, tia $OI$ cắt $Ax$ tại $M$, chứng minh $OI\cdot OM=R^2$. Tính độ dài đoạn $IO$ biết $OM=2R$ và $R=6$ cm.
		\item Gọi giao điểm $BM$ với $CH$ là $K$. Chứng minh $\triangle AMO\backsim \triangle HCB$ và $KC=KH$.
		\item Giả sử $(O;R)$ cố định, điểm $C$ thay đổi trên đường tròn nhưng vẫn thỏa mãn điều kiện đề bài. Xác định vị trí của $C$ để chu vi $\triangle OHC$ đạt giá trị lớn nhất. Tìm giá trị lớn nhất đó theo $R$.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[line join = round, line cap = round,>=stealth,font=\footnotesize,scale=1]
			\tkzDefPoints{-2/0/A, 2/0/B}
			\clip (-2.8,-2) rectangle (2.8,3);
			\coordinate (O) at ($(A)!0.5!(B)$);
			\coordinate (H) at ($(O)!0.3!(B)$);
			\tkzDefLine[perpendicular=through H](A,B)\tkzGetPoint{h}
			\tkzInterLC(H,h)(O,A)\tkzGetPoints{C}{D}
			\tkzDefLine[perpendicular=through A](A,B) \tkzGetPoint{a}
			\tkzDefLine[perpendicular=through O](A,C) \tkzGetPoint{o}
			\tkzInterLL(A,a)(O,o)    \tkzGetPoint{M}
			\tkzInterLL(A,C)(O,M)    \tkzGetPoint{I}
			\tkzInterLL(M,B)(C,H)    \tkzGetPoint{K}
			\coordinate (J) at ($(O)!0.5!(C)$);
			\coordinate (E) at ($(A)!0.5!(M)$);
			\tkzDrawPoints[fill=black](A,B,C,H,O,M,K,J,E)
			\tkzDrawSegments(A,B B,M M,A A,C M,O C,H O,C J,I J,H C,B O,E)
			\tkzDrawCircle[radius](O,A)
			\tkzMarkRightAngles(O,I,A C,H,O O,A,M)
			\tkzLabelPoints[left](A,M,E)			
			\tkzLabelPoints[right](B)
			\tkzLabelPoints[below](O,H)
			\tkzLabelPoints[below left](J)
			\tkzLabelPoints[above](I)
			\tkzLabelPoints[above right](K,C)	
			\end{tikzpicture}
		\end{center}
	\begin{enumerate}
		\item Gọi $J$ là trung điểm $OC\Rightarrow JO=JC=\dfrac{1}{2}OC$.\qquad (1)\\
		Lại có $\triangle CIO$ vuông tại $I$ nên $IJ=\dfrac{1}{2}OC$.\qquad (2)\\
		Mà $\triangle OHC$ vuông tại $H$ nên $HJ=\dfrac{1}{2}OC$.\qquad (3)\\
		Từ (1), (2) và (3) suy ra $JO=JC=JI=JH$ nên bốn điểm $C$, $H$, $O$, $I$ cùng thuộc đường tròn tâm $J$, đường kính $OC$, bán kính $\dfrac{R}{2}$.
		\item Ta có $AM\perp OA$ (tính chất tiếp tuyến). Nên $\triangle AMO$ vuông tại $A$, mà $AI\perp OM$, suy ra $AO^2=OI\cdot OM$ hay $OI\cdot OM=R^2$.\\
		Với $OA=R=6$ cm, $OM=2R=12$ cm thì $OI=\dfrac{R^2}{OM}=\dfrac{36}{12}=3$ cm.
		\item Ta có $\triangle ABC$ nội tiếp đường tròn đường kính $AB$ nên $\triangle ABC$ vuông tại $C$, suy ra $BC\perp AC$.\\
		Mà $OM\perp AC$, suy ra $OM\parallel BC$ và $\widehat{AOM}=\widehat{HBC}$.\\
		Xét $\triangle AMO$ và $\triangle HCB$ có $\widehat{AOM}=\widehat{HBC}$ và $\widehat{OAM}=\widehat{BHC}=90^\circ$ nên $\triangle AMO\backsim \triangle HCB$.\\
		$\Rightarrow \dfrac{HB}{AO}=\dfrac{HC}{AM}$.\qquad (4)\\
		Mặt khác, gọi $E$ là trung điểm $AM$, suy ra $OE$ là đường trung bình $\triangle ABM$ nên $OE\parallel BM$.\\
		$\Rightarrow \widehat{AOE}=\widehat{HBK}$ và $\triangle AOE\backsim \triangle HBK\Rightarrow \dfrac{HB}{AO}=\dfrac{HK}{AE}$\qquad (5).\\
		Từ (4) và (5) suy ra $\dfrac{HC}{AM}=\dfrac{HK}{AE}\Rightarrow \dfrac{HC}{HK}=\dfrac{AM}{AE}=2$.\\
		Suy ra $HC=2HK$ hay $KC=KH$.
		\item Trước hết, ta có $(a-b)^2\ge 0\Leftrightarrow 2(a^2+b^2)\ge (a+b)^2\Leftrightarrow a+b\le \sqrt{2\left(a^2+b^2\right)}$.\\
		Chu vi của $\triangle OHC$ là
		\begin{eqnarray*}
			P & = & OC+OH+HC=R+(OH+HC)\\
			& \le & R+\sqrt{2\left(OH^2+HC^2\right)}=R+\sqrt{2\cdot OC^2}=\left(1+\sqrt{2}\right)R.
		\end{eqnarray*}
	Đẳng thức xảy ra khi $OH=HC=\dfrac{OC}{\sqrt{2}}=\dfrac{R}{\sqrt{2}}$ và $\widehat{BOC}=45^\circ$.\\
	Vậy điểm $C$ trên đường tròn sao cho $\widehat{BOC}=45^\circ$ thì chu vi $\triangle HOC$ lớn nhất, bằng $\left(1+\sqrt{2}\right)R$.
	\end{enumerate}
	}
\end{bt}

\begin{bt}%[HK1-9, Hai Bà Trưng, Hà Nội,1819]%[Lê Minh An, dự án 9DT19]%[9D5G1]
	Cho các số thực $x$, $y$, $z>0$ và $x+2y+3z\ge 20$. Tìm giá trị nhỏ nhất của biểu thức
	$$P=x+y+z+\dfrac{3}{x}+\dfrac{9}{2 y}+\dfrac{4}{z}.$$
	\loigiai{
		Theo bất đẳng thức Cô-si ta có 
		\begin{eqnarray*}
			P & = & \dfrac{3x}{4}+\dfrac{3}{x}+\dfrac{y}{2}+\dfrac{9}{2y}+\dfrac{z}{4}+\dfrac{4}{z}+\left(\dfrac{x}{4}+\dfrac{y}{2}+\dfrac{3z}{4}\right)\\
			& \ge & 2\sqrt{\dfrac{3x}{4}\cdot\dfrac{3}{x}}+2\sqrt{\dfrac{y}{2}\cdot\dfrac{9}{2y}}+2\sqrt{\dfrac{z}{4}\cdot\dfrac{4}{z}}+\dfrac{1}{4}(x+2y+3z)\\
			& \ge & 3+3+2+\dfrac{20}{4}=13.
		\end{eqnarray*}
	Đẳng thức xảy ra khi $x=2$, $y=3$, $z=4$. Vậy giá trị nhỏ nhất của $P$ là $13$ khi $(x;y;z)=(2;3;4)$.
	
	}
\end{bt}
