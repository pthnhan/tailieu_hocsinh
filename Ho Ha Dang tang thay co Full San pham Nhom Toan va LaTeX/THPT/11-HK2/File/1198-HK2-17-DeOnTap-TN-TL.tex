\begin{bt}%[HK2 (2016-2017), Đề ôn tập]%[1D4B2-3]%[1D4B2-7]%[1D4B2-4]%[Đức Nguyễn]
	Tính các giới hạn sau:
	\begin{enumerate}
		\item $\displaystyle \lim_{x\to 1}\dfrac{x^2+3x-4}{1-x^3}$
		\item $\displaystyle \lim_{x\to -\infty}\dfrac{2x-3}{4x-\sqrt{4x^2+x-3}}$
		\item $\displaystyle \lim_{x\to -\infty}\left(\sqrt{9x^2-3x+1}+3x\right)$
	\end{enumerate}
	\loigiai{\hfill
		\begin{enumerate}
			\item $\displaystyle \lim_{x\to 1}\dfrac{x^2+3x-4}{1-x^3}=\displaystyle \lim_{x\to 1}\dfrac{(x-1)(x+4)}{(1-x)(x^2+x+1)}=-\displaystyle \lim_{x\to 1}\dfrac{x+4}{x^2+x+1}=-\dfrac{5}{3}$.
			\item $\displaystyle \lim_{x\to -\infty}\dfrac{2x-3}{4x-\sqrt{4x^2+x-3}}=\displaystyle \lim_{x\to -\infty}\dfrac{2x-3}{4x+x\sqrt{4+\dfrac{1}{x}-\dfrac{3}{x^2}}}=\displaystyle \lim_{x\to -\infty}\dfrac{2-\dfrac{3}{x}}{4+\sqrt{4+\dfrac{1}{x}-\dfrac{3}{x^2}}}=\dfrac{1}{3}$.
			\item Ta có\\
			$\begin{aligned}
			&\displaystyle \lim_{x\to -\infty}\left(\sqrt{9x^2-3x+1}+3x\right)=\displaystyle \lim_{x\to -\infty}\dfrac{\left(\sqrt{9x^2-3x+1}+3x\right)\left(\sqrt{9x^2-3x+1}-3x\right)}{\left(\sqrt{9x^2-3x+1}-3x\right)}\\
			=&\displaystyle \lim_{x\to -\infty}\dfrac{-3x+1}{\sqrt{9x^2-3x+1}+3x}=\displaystyle \lim_{x\to -\infty}\dfrac{-3x+1}{x\left(-\sqrt{9-\dfrac{3}{x}+\dfrac{1}{x^2}}-3\right)}\\
			=&\displaystyle \lim_{x\to -\infty}\dfrac{-3+\dfrac{1}{x}}{-\sqrt{9-\dfrac{3}{x}+\dfrac{1}{x^2}}-3}=\dfrac{1}{2}.
			\end{aligned}$
		\end{enumerate}
	}
