\section{HAI TAM GIÁC BẰNG NHAU. TRƯỜNG HỢP BẰNG NHAU THỨ NHẤT CỦA TAM GIÁC: CẠNH - CẠNH - CẠNH (C.C.C)}
\subsection{Tóm tắt lí thuyết}
\begin{enumerate}[\tickEX]
	\item Định nghĩa hai tam giác bằng nhau
	\begin{itemize}
		\item Hai tam giác bằng nhau là hai tam giác có các cạnh tương ứng bằng nhau, các góc tương ứng bằng nhau.
		\item Nếu có $\triangle ABC$ bằng $\triangle A_1B_1C_1$ ta kí hiệu $\triangle ABC=\triangle A_1B_1C_1$.
		\item $\triangle ABC=\triangle A_1B_1C_1\Rightarrow\heva{& AB=A_1B_1,\, BC=B_1C_1,\, CA=C_1A_1\\& \widehat{A_1}=\widehat{A},\,\widehat{B_1}=\widehat{B},\,\widehat{C_1}=\widehat{C}.}$
		\begin{center}
			\begin{tikzpicture}[scale=1, line join=round, line cap=round]
			\tkzDefPoints{0/0/B,4/0/C,1/2/A,6/0/B_1,10/0/C_1,7/2/A_1}
			\tkzDrawPolygon(A,B,C)
			\tkzDrawPolygon(A_1,B_1,C_1)
			\tkzDrawPoints(A,B,C,A_1,B_1,C_1)
			\tkzLabelPoints[above](A,A_1)
			\tkzLabelPoints[below](C,B,B_1,C_1)
			\tkzMarkSegments[mark=|](A,B A_1,B_1)
			\tkzMarkSegments[mark=||](B,C B_1,C_1)
			\tkzMarkSegments[mark=x](A,C A_1,C_1)
			\tkzMarkAngles[size=0.3cm,arc=l](B,A,C B_1,A_1,C_1)
			\tkzMarkAngles[size=0.3cm,arc=ll](C,B,A C_1,B_1,A_1)
			\tkzMarkAngles[size=0.7cm,arc=ll,mark=|](A,C,B A_1,C_1,B_1)
			\end{tikzpicture}
		\end{center}
	\end{itemize}
	\item Trường hợp bằng nhau thứ nhất của tam giác: cạnh-cạnh-cạnh (c.c.c)
	\begin{itemize}
		\item Nếu ba cạnh của tam giác này bằng ba cạnh của tam giác kia thì hai tam giác đó bằng nhau.
		\item Nếu hai tam giác $\triangle ABC$ và $\triangle {A_1}{B_1}{C_1}$ thỏa mãn:\\
		$AB = {A_1}{B_1}$; $BC = {B_1}{C_1}$; $CA = {C_1}{A_1}$ thì $\triangle ABC = \triangle {A_1}{B_1}{C_1}$ và khi đó ta có ngay: $\widehat {A_1} = \widehat A$; $\widehat {B_1} = \widehat B$; $\widehat {C_1} = \widehat C$.
		\begin{center}
		\begin{tikzpicture}[scale=1, line join=round, line cap=round]
		\tkzDefPoints{0/0/B,4/0/C,1/2/A,6/0/B_1,10/0/C_1,7/2/A_1}
		\tkzDrawPolygon(A,B,C)
		\tkzDrawPolygon(A_1,B_1,C_1)
		\tkzDrawPoints(A,B,C,A_1,B_1,C_1)
		\tkzLabelPoints[above](A,A_1)
		\tkzLabelPoints[below](C,B,B_1,C_1)
		\tkzMarkSegments[mark=|](A,B A_1,B_1)
		\tkzMarkSegments[mark=||](B,C B_1,C_1)
		\tkzMarkSegments[mark=x](A,C A_1,C_1)
		\end{tikzpicture}
		\end{center}
	\end{itemize}
\end{enumerate}	

\begin{nx}
	\hfill
	\begin{itemize}
		\item Như vậy, để chứng tỏ hai tam giác bằng nhau chúng ta chỉ cần khẳng định ba cặp cạnh bằng nhau mà không cần phải nêu đủ $6$ yếu tố bằng nhau ($3$ yếu tố cạnh và $3$ yếu tố góc).
		\item Bằng việc khẳng định được sự bằng nhau của hai tam giác chúng ta sẽ bắt đầu làm quen với việc chứng minh các tính chất trong tam giác.
	\end{itemize}
\end{nx}

\subsection{Các dạng toán}
\begin{dang}[Xác định các cạnh bằng nhau, các góc bằng nhau của hai tam giác bằng nhau]
	\begin{enumerate}[\tickEX]
		\item Căn cứ vào cách viết các đỉnh tương ứng của hai tam giác bằng nhau theo đúng thứ tự, ta viết được các góc bằng nhau, các đoạn thẳng bằng nhau.
	\end{enumerate}
\end{dang}

\begin{vd}
Hai tam giác trong các hình sau có bằng nhau không? Nếu có, hãy viết kí hiệu sự bằng nhau của hai tam giác đó.
\begin{center}
\begin{tikzpicture}
\tkzInit[xmin=-1,ymin=-2,xmax=8,ymax=3]
\tkzDefPoints{0/0/B, 3/0/C, 1/2/A, 4/-1/O}
\tkzDefPointBy[rotation = center O angle -90](B)\tkzGetPoint{N}
\tkzDefPointBy[rotation = center O angle -90](A)\tkzGetPoint{M}
\tkzDefPointBy[rotation = center O angle -90](C)\tkzGetPoint{P}
\tkzDrawSegments(A,B B,C C,A M,N N,P P,M)
\tkzDrawPoints(A,B,C,M,N,P)
\tkzLabelPoints[above](A,M,N)
\tkzLabelPoints[below](B,C,P)
\tkzMarkSegments[mark=|](A,C M,P)
\tkzMarkSegments[mark=||](B,C N,P)
\tkzMarkSegments[mark=x](A,B M,N)
\tkzLabelAngles[pos=0.5](B,A,C){$60^\circ$}
\tkzLabelAngles[pos=-0.5](N,M,P){$60^\circ$}
\tkzLabelAngles[pos=0.6](C,B,A){$70^\circ$}
\tkzLabelAngles[pos=0.8](M,P,N){$50^\circ$}
\end{tikzpicture}
\end{center}
\loigiai{
Theo hình vẽ
\begin{itemize}
\item Trong $\triangle ABC$, ta suy ra $\widehat{C}=180^\circ-\widehat{A}-\widehat{B}=180^\circ-60^\circ-70^\circ=50^\circ$.
\item Trong $\triangle MNP$ ta suy ra $\widehat{N}=180^\circ-\widehat{M}-\widehat{P}=180^\circ-60^\circ-50^\circ=70^\circ$.
\end{itemize}
Khi đó, với $\triangle ABC$ và $\triangle MNP$ ta có
\begin{itemize}
\item $AB=MN$, $BC=NP$, $CA=PM$ các cạnh bằng nhau.
\item $\widehat{A}=\widehat{M}$, $\widehat{B}=\widehat{N}$, $\widehat{C}=\widehat{P}$ các góc bằng nhau
\end{itemize}
Suy ra $\triangle ABC=\triangle MNP$.
}
\end{vd}

\begin{nx}
Qua ví dụ trên, ta thấy ngay được việc có thể giảm bớt một yếu tố giả thiết về góc mà hai tam giác vẫn bằng nhau. Điều này chính là lí do để chúng ta cần thiết đi xem xét ``Có bao nhiêu trường hợp bằng nhau của hai tam giác''.
\end{nx}

\begin{vd}
Cho $\triangle ABC=\triangle HIK$.
\begin{enumerate}
\item Tìm cạnh tương ứng với cạnh $BC$. Tìm góc tương ứng với góc $H$.
\item Tìm các cạnh bằng nhau, tìm các góc bằng nhau.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có 
\begin{itemize}
\item Cạnh tương ứng với cạnh $BC$ là cạnh $IK$.
\item Góc tương ứng với góc $H$ là góc $A$.
\end{itemize}
\item Vì $\triangle ABC=\triangle HIK$ suy ra $AB=HI$, $BC=IK$, $AC=HK$, $\widehat{A}=\widehat{H}$, $\widehat{B}=\widehat{I}$, $\widehat{C}=\widehat{K}$.
\end{enumerate}
}
\end{vd}


\begin{vd}
Cho $\triangle ABC=\triangle MNP$.
\begin{enumerate}
\item Viết đẳng thức trên dưới một dạng khác.
\item Biết $AB=4$ cm, $BC=6$ cm và $MP=5$ cm. Tính chu vi của mỗi tam giác nói trên.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có thể viết $\triangle ACB=\triangle MPN$ hoặc $\triangle BCA=\triangle NPM$.
\item Từ giả thiết $\triangle ABC=\triangle MNP$\\
Suy ra $MN=AB=4$ cm, $NP=BC=6$ cm và $MP=AC=5$ cm.\\
Khi đó chu vi của tam giác $\triangle ABC$ và $\triangle MNP$ lần lượt là
\begin{itemize}
\item $C_{\triangle ABC}=AB+BC+CA=4+6+5=15$ cm.
\item $C_{\triangle MNP}=MN+NP+PM=4+6+5=15$ cm.
\end{itemize}
\end{enumerate}
}
\end{vd}

\begin{nx}
Qua ví dụ trên, ta thấy ngay ``Hai tam giác bằng nhau sẽ có chu vi bằng nhau''.
\end{nx}

\begin{vd}
\immini{Cho hình vẽ, chứng tỏ rằng hai tam giác $\triangle ABC$ và $\triangle ADC$ 
bằng nhau.
}{
\begin{tikzpicture}
\tkzDefPoints{0/0/D, 3/0/C, 1/2/A}
\coordinate (B) at ($(A)+(C)-(D)$);
\tkzDrawSegments(A,B B,C C,D D,A A,C)
\tkzDrawPoints(A,B,C,D)
\tkzLabelPoints[below](D,C)
\tkzLabelPoints[above](A,B)
\tkzMarkSegments[mark=||](D,A C,B)
\tkzMarkSegments[mark=|](D,C A,B)
\tkzLabelAngles[pos=0.7](C,A,B A,C,D){$30^\circ$}
\tkzLabelAngles[pos=0.6](B,C,A){$80^\circ$}
\tkzLabelAngles[pos=0.7](C,D,A){$70^\circ$}
\end{tikzpicture}
}
\loigiai{
Theo hình vẽ, ta có
\begin{itemize}
\item Trong $\triangle ABC$ suy ra $\widehat{B}=180^\circ-\widehat{BAC}-\widehat{ACB}=180^\circ-30^\circ-80^\circ=70^\circ$.
\item Trong $\triangle ADC$, suy ra $\widehat{DAC}=180^\circ-\widehat{ACD}-\widehat{D}=180^\circ-30^\circ-70^\circ=80^\circ$.
\end{itemize}
Khi đó với $\triangle ABC$ và $\triangle ADC$, ta có
\begin{itemize}
\item $AB=CD$, $BC=DA$, $AC$ chung (các cạnh bằng nhau).
\item $\widehat{CAB}=\widehat{ACD}$, $\widehat{B}=\widehat{D}$, $\widehat{ACB}=\widehat{CAD}$ (các góc bằng nhau).
\end{itemize}
Suy ra $\triangle ABC=\triangle CDA$.
}
\end{vd}

\begin{vd}
Cho $\triangle ABC$ có $\widehat{A}=80^\circ$. Tính số đo của các góc $\widehat{B}$, $\widehat{C}$ biết rằng tam giác $\triangle ABC=\triangle ACB$.
\loigiai{
Từ giả thiết $\triangle ABC=\triangle ACB\Rightarrow \widehat{B}=\widehat{C}$ (dựa trên sự tương ứng đỉnh).\\
Mặt khác, ta có $\widehat{A}+\widehat{B}+\widehat{C}=180^\circ\Rightarrow 80^\circ+\widehat{B}+\widehat{B}=180^\circ\Rightarrow\widehat{B}=50^\circ$.\\
Vậy $\triangle ABC$ có $\widehat{B}=\widehat{C}=50^\circ$.
}
\end{vd}

\begin{vd}
Cho $\triangle ABC=\triangle MNP$. Biết có $\widehat{A}=80^\circ$, $\widehat{N}=75^\circ$. Tính số đo các góc còn lại của tam giác.
\loigiai{
Từ giả thiết $\triangle ABC=\triangle MNP$ suy ra
\begin{itemize}
\item $\widehat{A}=\widehat{M}=80^\circ$.
\item $\widehat{B}=\widehat{N}=75^\circ$.
\item $\widehat{C}=\widehat{P}=180^\circ-\widehat{M}-\widehat{N}=180^\circ-80^\circ-75^\circ=25^\circ$.
\end{itemize}
}
\end{vd}


\begin{dang}[Chứng minh hai tam giác bằng nhau]
\begin{enumerate}[\tickEX]
	\item Xét hai tam giác;
	\item Kiểm tra ba điều kiện bằng nhau: cạnh-cạnh-cạnh;
	\item Kết luận hai tam giác đó bằng nhau.
\end{enumerate}	
\end{dang}
\begin{vd}%[7H2Y3]
	Cho đoạn thẳng $AB$. Vẽ hai cung tròn tâm $A$, tâm $B$ bán kính $AB$, chúng cắt nhau tại $C$ và $D$. Chứng minh rằng
	\begin{listEX}[2]
		\item $\triangle ABC = \triangle ABD$;
		\item $\triangle ACD = \triangle BCD$.
	\end{listEX}
	\loigiai{
		Theo cách dựng của giả thiết, ta có: $AB = AC = BC = AD = BD$.
		\immini{
			\begin{enumerate}
				\item  Xét $\triangle ABC$ và $\triangle ABD$ có
				\begin{itemize}
					\item $AB$ là cạnh chung;
					\item $AC=AB$ (cmt);
					\item $BC=BD$ (cmt).
				\end{itemize}
				$ \Rightarrow \triangle ABC = \triangle ABD$ (c–c–c).
				\item  Xét $\triangle ACD$ và $\triangle BCD$ có
				\begin{itemize}
					\item $CD$ là cạnh chung;
					\item $AC=BC$ (cmt);
					\item $AD=BD$ (cmt).
				\end{itemize}
				$ \Rightarrow \triangle ACD = \triangle BCD$ (c–c–c).
			\end{enumerate}	
		}{
			\begin{tikzpicture}[scale=1]
			\tkzDefPoints{0/0/A,4/0/B}
			\tkzCompass[color=red,delta=100,dashed](A,B)
			\tkzCompass[color=red,delta=100,dashed](B,A)
			\tkzInterCC(A,B)(B,A)\tkzGetPoints{C}{D}
			\tkzDrawPoints(A,B,C,D)
			\tkzLabelPoints[right](B)
			\tkzLabelPoints[left](A)
			\tkzLabelPoints[above](C)
			\tkzLabelPoints[below](D)
			\draw (A)--(D)--(B)--(C)--(A) (A)--(B) (C)--(D);
			\end{tikzpicture}
		}
		\begin{nx}
			\hfill
			\begin{itemize}
				\item Như vậy, để chứng tỏ hai tam giác bằng nhau chúng ta chỉ cần khẳng định ba cặp cạnh bằng nhau mà không cần phải nêu ra đủ $6$ yếu tổ bằng nhau nữa ($3$ yếu tố cạnh và $3$ yếu tố góc).
				\item Bằng việc khẳng định được sự bằng nhau của hai tam giác chúng ta sẽ bắt đầu làm quen với việc chứng minh các tính chất trong tam giác, các ví dụ sau sẽ minh họa điều này.
			\end{itemize}
		\end{nx}
	}
\end{vd}



\begin{dang}[Sử dụng hai tam giác bằng nhau để chứng minh hai góc bằng nhau]
\begin{enumerate}[\tickEX]
	\item Chọn hai tam giác có hai góc cần chứng minh bằng nhau.
	\item Chứng minh hai tam giác ấy bằng nhau theo trường hợp c.c.c.
	\item Suy ra hai góc tương ứng bằng nhau.
\end{enumerate}	
\end{dang}

% Ví dụ 2
\begin{vd}%[7H2B3]
	Cho $\triangle ABC$ có $AB = AC$. Gọi $M$ là trung điểm của $BC$. Chứng minh rằng $AM$ là đường trung trực của $BC$.
	\loigiai{
		Để chứng minh $AM$ là đường trung trực của $BC$, ta chỉ cần chứng minh $AH \perp BC$.\\
		\immini{
			Xét hai tam giác $\triangle ABM$ và $\triangle ACM$, ta có:
			\begin{itemize}
				\item $AB = AC$ (giả thiết);
				\item $BM = CM$ (vì $M$ là trung điểm của $BC$);
				\item $AM$ chung.
			\end{itemize}
		}{
			\begin{tikzpicture}
			\tkzDefPoints{0/0/B,6/0/C,3/0/M,3/3/A}
			\tkzDrawPoints(A,B,C,M)
			\tkzLabelPoints[right](C)
			\tkzLabelPoints[left](B)
			\tkzLabelPoints[above](A)
			\tkzLabelPoints[below](M)
			\draw (A)--(B)--(C)--(A) (A)--(M);
			\tkzLabelAngle[pos=0.6,rotate=0](B,A,M){\footnotesize $1$}
			\tkzLabelAngle[pos=0.6,rotate=0](M,A,C){\footnotesize $2$}
			\tkzLabelAngle[pos=0.3,rotate=0](A,M,C){\footnotesize $2$}
			\tkzLabelAngle[pos=0.3,rotate=0](B,M,A){\footnotesize $1$}
			\end{tikzpicture}}
		\noindent Suy ra: $\triangle ABM = \triangle ACM$ (c.c.c) $ \Rightarrow \widehat {M_1} = \widehat {M_2}$.\\
		Mặt khác, ta lại có:
		$$\widehat {M_1} + \widehat {M_2} = 180^\circ  \Rightarrow \widehat {M_1} + \widehat {M_1} = 180^\circ  \Rightarrow 2\widehat {M_1} = 180^\circ 
		\Rightarrow \widehat {M_1} = 90^\circ  \Rightarrow AH \perp BC.$$
		Vậy $AM$ là đường trung trực của $BC$.
		
		\begin{nx}
			\hfill
			\begin{itemize}
				\item Cũng từ kết quả $\triangle ABM = \triangle ACM$, ta suy ra được: 
				$\widehat {A_1} = \widehat {A_2} \Rightarrow AM$ là đường phân giác của góc $\widehat A$.
				\item Như vậy, trong $\triangle ABC$ có $AB = AC$ ($\triangle ABC$ cân tại $A$) thì $AM$ vừa là trung tuyến, vừa là đường cao, vừa là đường trung trực, vừa là đường phân giác – \textit{Đây chính là tính chất cơ bản của tam giác cân.}
			\end{itemize}
		\end{nx}
	}
\end{vd}

\begin{vd}%[7H2B3]
	Cho $\triangle ABC$. Vẽ cung tròn tâm $A$ bán kính $BC$, vẽ cung tròn tâm $C$ bán kính $BA$, chúng cắt nhau tại $D$ ($D$ và $B$ nằm khác phía đối với $AC$). Chứng minh rằng $AD\parallel BC$.
	\loigiai{
		\immini{
			Từ cách vẽ, ta nhận được: $AD = BC$, $CD = AB$.\\
			Xét hai tam giác $\triangle ABC$ và $\triangle CDA$, ta có:\\
			$AB = CD$, $BC = DA$, $AC$ chung $ \\
			\Rightarrow \triangle ABC = \triangle CDA$ (c.c.c)\\
			$ \Rightarrow \widehat {A_1} = \widehat {C_1} \Rightarrow AD\parallel BC$, vì có hai góc so le trong bằng nhau.
		}
		{\begin{tikzpicture}
			\tkzDefPoints{0/0/A,4/0/B}
			\tkzDefShiftPoint[A](-60:3){C}
			\tkzDefPointBy[translation=from B to A](C)\tkzGetPoint{D}
			\tkzCompass[color=red,length=1.5,dashed](A,D)
			\tkzCompass[color=black,length=2,dashed](C,D)
			\tkzDrawPoints(A,B,C,D)
			\tkzLabelPoints[right](C)
			\tkzLabelPoints[left](D)
			\tkzLabelPoints[above](A,B)
			\draw (A)--(B)--(C)--(D)--(A) (A)--(C);
			\tkzLabelAngle[pos=0.6,rotate=0](B,C,A){\footnotesize $1$}
			\tkzLabelAngle[pos=0.6,rotate=0](D,A,C){\footnotesize $1$}
			\end{tikzpicture}}
		\begin{note}
			Kết quả của ví dụ trên, cho phép chúng ta có thêm một phương pháp "Dựng đường thẳng $a$ đi qua điểm $A$ và song song với đường thẳng $d$ cho trước ($A \notin d$)". Thật vậy:
			\immini{
				\begin{itemize}
					\item Lấy hai điểm $B$, $C$ trên đường thẳng $d$.
					\item Vẽ cung tròn tâm $A$ bán kính $BC$, vẽ cung tròn tâm $C$ bán kính $BA$, chúng cắt nhau tại $D$ ($D$ và $B$ nằm khác phía đối với $AC$).
					\item Khi đó, $a$ chính là đường thẳng đi qua hai điểm $A$ và $D$.
				\end{itemize}
			}{
				\begin{tikzpicture}
				\tkzDefPoints{0/0/A,4/0/D}
				\tkzDefShiftPoint[A](-60:3){C}
				\tkzDefPointBy[translation=from D to A](C)\tkzGetPoint{B}
				\tkzCompass[color=red,length=1.5,dashed](A,D)
				\tkzCompass[color=black,length=2,dashed](C,D)
				\tkzDrawPoints(A,B,C,D)
				\tkzLabelPoints[above right](D)
				\tkzLabelPoints[above](A)
				\tkzLabelPoints[below](B,C)
				\tkzDrawLines(A,D C,B)
				\draw (A)--(B);
				\end{tikzpicture}}
		\end{note}
	}
\end{vd}
