Tìm tập xác định của hàm số sau:
	\begin{multicols}{2}
		\begin{enumerate}
\item $y=\dfrac{2x-1}{x-2}$.
			\item  $y=\dfrac{\left(2x-3\right)\sqrt{x-1}}{x-5}+\sqrt{6-x}$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item Điều kiện xác định: $x-2 \neq 0 \Leftrightarrow x \neq 2\\ \Rightarrow$ TXĐ: $ \mathscr{D}=\mathbb{R} \backslash \{2\} $.
			\item Điều kiện xác định: $\heva{x-1\geq 0\\ 6-x\geq 0\\ x-5\neq 0} \Leftrightarrow \heva{x\geq1\\x\leq 6\\ x\neq 5} \Leftrightarrow \heva{1 \leq x \leq 6 \\ x\neq 5}$\\ $\Rightarrow$ TXĐ: $\mathscr{D}=\left[1;6\right] \backslash \{5\}$ .
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề kiểm tra HKI Toán 10 - Năm học 2017 - 2018 - THPT Hậu Lộc 4 - Thanh Hóa]%[0D3Y2]%[0D3K2]
Giải phương trình sau
	\begin{multicols}{2}
		\begin{enumerate}
			\item $|3x - 1| = |2x - 3|$.
			\item $(x - 1)^4 - 3(x^2 - 2x) - 7 = 0$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item $|3x - 1| = |2x - 3|$ 
			$\Leftrightarrow \hoac{&3x - 1 = 2x - 3 \\ &3x - 1 = -(2x - 3)\\}$ 
			$\Leftrightarrow \hoac{&x = - 2 \\ &x = \dfrac{4}{5}\\}$. \\
			Vậy tập nghiệm của phương trình là $S = \left\{ 2 ; \dfrac{4}{5} \right\}$.
			\item $(x - 1)^4 - 3(x^2 - 2x) - 7 = 0$ 
			$\Leftrightarrow (x^2 - 2x + 1)^2 - 3(x^2 - 2x) - 7 = 0$. \\
			Đặt $t = x^2 - 2x$. Phương trình trở thành \\
			$(t + 1)^2 - 3t - 7 = 0$ 
			$\Leftrightarrow t^2 - t - 6 = 0$ 
			$\Leftrightarrow t = 3 \vee t = -2$. \\
			Với $t = 3$, ta có $x^2 - 2x - 3 = 0 \Leftrightarrow x = -1 \vee x = 3$. \\
			Với $t = -2$, ta có $x^2 - 2x + 2 = 0$ (vô nghiệm).\\
			Vậy tập nghiệm của phương trình là $S = \{ -1;3 \}$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề kiểm tra HKI Toán 10 - Năm học 2017 - 2018 - THPT Hậu Lộc 4 - Thanh Hóa]%[0D3G3]
Giải hệ phương trình $\heva{&2x^3 + x^2y + 4xy + 2y^2 = 9\\&x^2 + 2x + 3y = 6\\}$
	\loigiai{
Ta có hệ tương đương với $\heva{&(x^2 + 2y)(2x + y) = 9\\&(x^2 + 2y) + (2x + y) = 6\\}$. \\
Đặt $u = x^2 + 2y, v = 2x + y$, hệ trở thành $\heva{&uv = 9\\&u + v = 6\\}$.\\
Áp dụng định lý Vi-ét đảo, ta có $u, v$ là nghiệm của phương trình $X^2 - 6X + 9 = 0$. Từ đây suy ra $u = v = 3$.\\
Vậy ta có $\heva{&x^2 + 2y = 3\\&2x + y = 3\\}$ 
$\Leftrightarrow \heva{&x^2 + 2(3 - 2x) = 3\\&y = 3 - 2x\\}$ 
$\Leftrightarrow \heva{&x^2  - 4x +3 = 0\\&y = 3 - 2x\\}$\\
$\Leftrightarrow \heva{&x = 1 \vee x = 3\\&y = 3 - 2x\\}$
$\Leftrightarrow \heva{&x = 1\\&y= 1} \vee \heva{&x = 3\\&y = -3\\}$\\
Vậy hệ phương trình có hai nghiệm $(x,y) = (1,1)$, $(x,y) = (3,-3)$.
	}
\end{bt}
