\subsection{Nền tảng kiến thức}
\begin{itemize}
	\item [Bước 1.] Lập phương trình.
	\begin{itemize}
		\item Chọn ẩn số và đặt điều kiện thích hợp của ẩn.
		\item Biểu diễn các đại lượng chưa biết theo ẩn số và các đại lượng đã biết.
		\item Lập phương trình biểu thị mối quan hệ giữa các đại lượng.
	\end{itemize}
\item [Bước 2.] Giải phương trình vừa lập.
\item [Bước 3.] Trả lời: Kiểm tra xem trong các nghiệm của phương trình, nghiệm nào thích hợp với bài toán và trả lời.
\end{itemize}
\subsection{Các dạng bài tập và phương pháp giải}
   \begin{dang}{Toán số học, phần trăm}
   \begin{itemize}
   	\item 	Biểu diễn: 
   	\begin{eqnarray*}
 &&\overline{ab}=10\cdot a+b\; (a,b\in \mathbb{N}, 0<a\leq 9, 0\leq b\leq 9).\\
 &&\overline{abc}=100\cdot a+10\cdot b+c\; (a,b,c \in \mathbb{N}, 0<a\leq 9, 0\leq b,c\leq 9).
   	\end{eqnarray*}
   \item Tỉ số của hai số $a$ và $b$ ($b\neq 0$) là $\dfrac{a}{b}$.
   \item Tổng của hai số $x$, $y$ là $x+y$.
   \item Tổng bình phương hai số $x$; $y$ là $x^2+y^2$.
   \item Tổng nghịch đảo của hai số $x$; $y$ là $\dfrac{1}{x}+\dfrac{1}{y}$.
   \end{itemize}
   	\end{dang}
   \begin{vd}
   	Trong lúc học nhóm, bạn Hùng yêu cầu bạn Minh và bạn Lan mỗi người chọn một số sao cho hai số này hơn kém nhau là $5$ và tích của chúng phải bằng $150$. Vậy hai bạn Minh và Lan phải chọn những số nào?
   	\loigiai{
   	Gọi số mà một bạn đã chọn là $x$ và số bạn kia là $x+5$.\\
   	Tích của hai số là $x(x+5)$.\\
   	Theo đầu bài, ta có phương trình
   	\[x(x+5)=150 \Leftrightarrow x^2+5x-150=0.\]
   	Giải phương trình $\Delta =25-4\cdot (-150)=625=25^2$ nên $x_1=10$, $x_2=-15$.\\
   	Trả lời:\\
   	- Nếu bạn Minh chọn số $10$ thì bạn Lan chọn số $15$ hoặc ngược lại.\\
   	- Nếu bạn minh chọn số $-15$ thì bạn Lan chọn số $-10$ hoặc ngược lại.
   
}
   \end{vd}
\begin{vd}
	Tìm hai số biết rằng tổng của chúng bằng $17$ đơn vị. Nếu số thứ nhất tăng thêm $3$ đơn vị, số thứ $2$ tăng thêm $2$ đơn vị thì tích của chúng bằng $105$ đơn vị.
	\loigiai{
	Gọi số thứ nhất là $x$, số thứ hai là $y$.\\
	Theo đề bài tổng của hai số đó bằng $17$ đơn vị nên ta có phương trình $x+y=17$. \hfill (1)\\
	Số thứ nhất tăng thêm $3$ đơn vị, số thứ hai tăng thêm $2$ đơn vị thì tích của chúng bằng $105$ đơn vị nên ta có phương trình $(x+3)(y+2)=105.$ \hfill (2)\\
	Từ (1) và (2), ta có hệ phương trình
	$$\heva{&x+y=17\\&(x+3)(y+2)=105.}$$
	Rút $y$ từ (1) thế vào (2) và thu gọn, ta được
	\[x^2-16x+48=0.\]
	Giải phương trình ta được $x_1=12$ (thỏa mãn) và $x_2=4$ (thỏa mãn).\\
	Vậy nếu số thứ nhất là $12$ thì số thứ hai là $5$; nếu số thứ nhất là $4$ thì số thứ hai là $13$.

}
\end{vd}

		\item Năng suất $=$ Khối lượng công việc $\div$ Thời gian.
		\item Thời gian $=$ Khối lượng công việc $\div$ Năng suất.
	\end{itemize}

\begin{vd}
	Một công nhân dự định làm $70$ sản phẩm trong thời gian quy định. Nhưng do áp dụng kĩ thuật nên đã tăng năng suất thêm $5$ sản phẩm mỗi giờ. Do đó, không những hoàn thành kế hoạch trước thời hạn $40$ phút mà còn vượt mức $10\%$ sản phẩm. Tính năng suất dự định.
	\loigiai{
	Gọi năng suất dự định là $x$ (sản phẩm/giờ, $x\in \mathbb{N}^{*}$);\\
	Thời gian dự định làm $70$ sản phẩm là $\dfrac{70}{x}$ giờ;\\
	Thời gian thực tế làm $80$ sản phẩm với năng suất $x+5$ (sản phẩm/giờ) là $\dfrac{80}{x+5}$ giờ.\\
	Theo đề bài, công nhân hoàn thành trước kế hoạch $40$ phút ($=\dfrac{2}{3}$ giờ).\\
	Ta có phương trình \[\dfrac{70}{x}-\dfrac{80}{x+5}=\dfrac{2}{3} \Leftrightarrow x^2+20x-525=0.\]
	$\Delta =20^2-4\cdot (-525)=2500 >0$ nên phương trình có nghiệm $x_1=15$ (nhận); $x_2=-35$ (loại).\\
	Vậy năng suất dự định là $15$ sản phẩm/giờ.

}
\end{vd}
\begin{vd}
	Một công nhân dự định làm $72$ sản phẩm trong thời gian quy định. Nhưng thực tế xí nghiệp lại giao $80$ sản phẩm. Vì vậy mặc dù người đó đã làm mỗi giờ thêm 1 sản phẩm, song thời gian hoàn thành công việc vẫn chậm hơn so với dự định 12 phút. Tính năng suất dự kiến, biết rằng mỗi giờ người đó làm không quá $20$ sản phẩm.
	\loigiai{
	Gọi năng suất dự kiến là $x$ (sản phẩm/giờ, $x\in \mathbb{N}$, $0<x<20$);\\
	Vậy năng suất thực tế là $x+1$ (sản phẩm/giờ).\\
	Thời gian dự định làm $72$ sản phẩm là $\dfrac{72}{x}$ (giờ).\\
	Thời gian thực tế làm $80$ sản phẩm là $\dfrac{80}{x+1}$ (giờ).\\
	Theo đề bài, thời gian hoàn thành vẫn chậm hơn so với dự định là $12$ phút ($=\dfrac{1}{5}$ giờ) ta có phương trình 
	\[\dfrac{80}{x+1}-\dfrac{72}{x}=\dfrac{1}{5} \Leftrightarrow 80\cdot 5\cdot x-72\cdot 5(x+1)=x(x+1) \Leftrightarrow x^2-39x+360=0.\]
	Ta có
	$\Delta =39^2-4\cdot 360=81 >0$, nên phương trình có nghiệm $x_1=24$ (loại); $x_2=15$ (nhận).\\
	Vậy năng suất dự kiến là $15$ (sản phẩm/giờ).
}
\end{vd}

\begin{dang}{Toán chuyển động}
	\begin{itemize}
		\item Sử dụng công thức $S=V\cdot t$; trong đó, $S$ là quãng đường, $V$ là vận tốc, $t$ là thời gian.\\
		Suy ra $V=\dfrac{S}{t}$; $t=\dfrac{S}{V}$.
		\item Nếu chuyển động trong dòng chảy thì
		\begin{eqnarray*}
		&&V_{\text{xuôi dòng}}=V_{\text{riêng}}+V_{\text{dòng nước}}\\
			&&V_{\text{ngược dòng}}=V_{\text{riêng}}-V_{\text{dòng nước}}.
		\end{eqnarray*}
	\end{itemize}
	\end{dang}

\begin{vd}
	Một xuồng du lịch đi từ thành phố Cà Mau đến Đất Mũi theo một đường sông dài $120$ km. Trên đường đi, xuồng có nghỉ lại một giờ ở thị trấn Nam Căn. Khi về, xuồng đi theo đường khác dài hơn đường đi $5$ km và với vận tốc lúc về nhỏ hơn vận tốc lúc đi là $5$ km/h. Tính vận tốc của xuồng lúc đi, biết rằng thời gian về bằng thời gian đi.
	\loigiai{
	Gọi vận tốc của xuồng lúc đi là $x$ (km/h), $x>0$, thì vận tốc lúc về là $x-5$ (km/h).\\
	Thời gian đi $120$ km là $\dfrac{120}{x}$ (giờ).\\
	Vì khi đi có nghỉ một giờ nên thời gian đi hết tất cả là $\dfrac{120}{x}+1$ (giờ).\
	Độ dài quãng đường về $120+5=125$ (km).\\
	Thời gian về là $\dfrac{125}{x-5}$ (giờ).\\
	Theo đầu bài, ta có phương trình \[\dfrac{120}{x}+1=\dfrac{125}{x-5} \Leftrightarrow x^2-10x-600=0.\]
	Ta có
	$\Delta =10^2-4\cdot (-600)= 2500>0$, nên phương trình có nghiệm $x_1=30$ (nhận); $x_2=-20$ (loại).\\
	Vậy vận tốc của xuồng đi là $30$ km/h.
}
\end{vd}
\begin{vd}
	Bác Hiệp và cô Liên đi xe đạp từ làng lên tỉnh trên quãng đường dài $30$ km, khởi hành cùng một lúc. Vận tốc xe của bác hiệp lớn hơn vận tốc xe của cô Liên là $3$ km/h nên bác Hiệp đã đến tỉnh trước cô Liên nửa giờ. Tính vận tốc xe của mỗi người.
	\loigiai{
	Gọi vận tốc xe của bác Hiệp là $x$ (km/h, $x>0$); khi đó vận tốc xe của  cô Liên là $x-3$ (km/h).\\
	Thời gian bác Hiệp đi từ làng lên tỉnh là $\dfrac{30}{x}$ (giờ).\\
	Thời gian cô Liên đi từ làng lên tỉnh là $\dfrac{30}{x-3}$ (giờ).\\
	Vì bác Hiệp đến trước cô Liên nửa giờ, tức là thời gian đi của bác Hiệp ít hơn thời gian đi của cô Liên nửa giờ nên ta có phương trình
	\[\dfrac{30}{x-3}-\dfrac{30}{x}=\dfrac{1}{2} \Leftrightarrow x^2-3x-180=0.\]
	Ta có
	$\Delta =3^2-4\cdot (-180)= 729>0$, nên phương trình có nghiệm $x_1=-12$ (loại); $ x_2=15$ (nhận).\\
	Vậy vận tốc xe của bác Hiệp là $15$ km/h. Vận tốc xe của cô Liên là $12$ km/h.
}
\end{vd}
\begin{dang}{Dạng toán có nội dung hình học}
	Áp dụng các công thức sau:
	\begin{itemize}
		\item Định lý Pi-ta-go: $\Delta ABC$ vuông tại $A \Leftrightarrow AB^2+AC^2=BC^2$.
		\item Diện tích hình chữ nhật: $S=a\cdot b$; với $a$ là chiều dài, $b$ là chiều rộng.
		\item Diện tích hình thang: $S=\dfrac{a+b}{2}$ hoặc $S=m\cdot h$. Trong đó, $a$, $b$ là độ dài hai đáy; $h$ là đường cao; $m$ là độ dài đường trung bình.
	\end{itemize}
	
	\end{dang}

\begin{vd}
	Một hình chữ nhật có chiều rộng bằng $\dfrac{2}{3}$ chiều dài, diện tích hình chữ nhật là $5400$ cm$^2$. Tính chu vi hình chữ nhật.
	\loigiai{
	Gọi chiều dài hình chữ nhật là $x$ (cm, $x>0$).\\
	Chiều rộng hình chữ nhật là $\dfrac{2}{3}x$ (cm).\\
	Theo đầu bài, ta có phương trình \[x\cdot \dfrac{2}{3}x=5400 \Leftrightarrow x^2=8100 \Leftrightarrow \hoac{&x=90\\&x=-90\;(\text{loại}).}\]
	Vậy chiều dài hình chữ nhật là $90$ cm, chiều rộng hình chữ nhật là $60$ cm. Do đó, chu vi hình chữ nhật là $(90+60)\cdot 2=300$ cm.
}
\end{vd}
\begin{vd}
	Một thửa ruộng hình chữ nhật có diện tích là $100$  m$^2$. Tính độ dài các cạnh của thửa ruộng. Biết rằng nếu tăng chiều rộng của thửa ruộng lên $2$ m và giảm chiều dài thửa ruộng đi $5$ m thì diện tích thửa ruộng tăng thêm $5$ m$^2$.
	\loigiai{
	Gọi chiều dài hình chữ nhật là $x$ (m, $x>0$), thì chiều rộng hình chữ nhật là $\dfrac{100}{x}$ (m).\\
	Theo đầu bài, nếu tăng chiều rộng thửa ruộng lên $2$ m và giảm chiều dài thửa ruộng đi $5$ m thì diện tích thửa ruộng tăng thêm $5$ m$^2$, ta có phương trình
	\[(x-5)\cdot \left(\dfrac{100}{x}+2\right) =100+5 \Leftrightarrow 2x^2-15x-500=0\]
	Ta có
	$\Delta =15^2-4\cdot 2 \cdot(-500)= 4225>0$, nên phương trình có nghiệm $x_1=-\dfrac{25}{2}$ (loại); $ x_2=20$ (nhận).\\
	Vậy chiều dài mảnh đất hình chữ nhật là $20$ m, chiều rộng là $5$ m.
}
\end{vd}

		Gọi thời gian công nhân thứ nhất làm một mình xong công việc là $ x $ (giờ, $ x>6\dfrac{2}{3} $).\\

		Mỗi giờ công nhân thứ nhất làm được $ \dfrac{1}{x} $  (công việc).\\
		Mỗi giờ công nhân thứ hai làm được $ \dfrac{1}{x+3} $  (công việc).\\
		Theo đầu bài, hai công nhân cùng làm thì hoàn thành công việc trong $  6\dfrac{2}{3} $ giờ. Nên mỗi giờ họ cùng làm được $ 1: 6\dfrac{2}{3}=\dfrac{3}{20} $ (công việc).\\
		Ta có phương trình \\
		\[ \dfrac{1}{x}+\dfrac{1}{x+3}=\dfrac{3}{20} \Leftrightarrow3x^2-31x-60=0 .\]
		Ta có $ \Delta =31^2-4\cdot 3 \cdot (-60)=1681>0$ nên phương trình có nghiệm là $ x_1=-\dfrac{5}{3} $ (loại); $ x_2=12 $ (nhận).\\
		Vậy thời gian công nhân thứ nhất làm xong công việc là $ 12 $ giờ.\\
		Thời gian công nhân thứ hai làm một mình xong công việc là $ 15 $ giờ.
}
\end{vd}
\begin{vd}

	}

	Khối lượng muối có trong $300$ gam dung dịch muối II là $\dfrac{300y}{100}=3y$ (gam).\\
	Khối lượng muối có trong $200+300$ gam dung dịch hỗn hợp là $500\cdot \dfrac{33}{100}=165$ (gam).\\
	Vì lượng muối của dung dịch hỗn hợp bằng tổng  lượng muối của muối I và muối II nên ta có phương trình

	\[\heva{&x-y=20\\&2x+3y=165}\Leftrightarrow \heva{&x=y+20\\&2(y+20)+3y=165}\Leftrightarrow \heva{&x=45\\&y=25}\;(\text{Thỏa mãn}).\]

	Khi xuôi dòng $108$ km và ngược dòng $63$ km, ca nô lần lượt phải đi trong $\dfrac{108}{x+y}$ giờ và $\dfrac{63}{x-y}$ giờ. Ta có phương trình $\dfrac{108}{x+y}+\dfrac{63}{x-y}=7$.\\
	Khi xuôi dòng $81$ km và ngược dòng $84$ km, ca nô lần lượt phải đi trong $\dfrac{81}{x+y}$ giờ và $\dfrac{84}{x-y}$ giờ. Ta có phương trình $\dfrac{81}{x+y}+\dfrac{84}{x-y}=7$.\\

	\[\heva{&\dfrac{108}{x+y}+\dfrac{63}{x-y}=7\\&\dfrac{81}{x+y}+\dfrac{84}{x-y}=7.}\]

	\[\heva{&108u+63v=7\\&81u+84v=7}\Leftrightarrow \heva{&u=\dfrac{1}{27}\\&v=\dfrac{1}{21}.}\]

	\[\heva{&x+y=27\\&x-y=21} \Leftrightarrow \heva{&x=24\\&y=3}\; (\text{thỏa mãn}).\]
