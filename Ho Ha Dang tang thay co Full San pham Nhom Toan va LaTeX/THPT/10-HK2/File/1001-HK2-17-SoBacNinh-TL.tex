\begin{bt}%[Sở GD BẮC NINH TOÁN 10 năm học 2016 - 2017]%[0D4K5]%[Nguyễn Thành Sơn]
	Tìm tất cả các giá trị thực của tham số $m$ để hàm số $y = \sqrt{mx^2 -2(m + 1)x + 4}$ có tập xác định là $\mathscr{D} = \mathbb{R}$. 
	\loigiai{
	Hàm số đã cho có tập xác định là $\mathscr{D} = \mathbb{R} \Leftrightarrow mx^2 -2(m + 1)x + 4 \geq 0, \forall x \in \mathbb{R}$ \hfill(1)\\
	Đặt $f(x) = mx^2 -2(m + 1)x + 4$.\\
	$\bullet$ Trường hợp $1$: Với $m = 0$ \\
	$f(x) = -2x + 4 \geq 0 \Leftrightarrow x \leq 2$: Không thỏa điều kiện $(1)$. \\
	$\bullet$ Trường hợp $2$: Với $m \neq 0$ \\
	$f(x) \geq 0, \forall x \in \mathbb{R} \Leftrightarrow \heva{&m > 0 \\& \left(m + 1 \right)^2 - 4m \leq 0} \Leftrightarrow \heva{&m > 0 \\ &\left(m - 1 \right)^2 \leq 0} \Leftrightarrow m = 1$. \\
	Vậy $m = 1$ thỏa yêu cầu bài.
	}
\end{bt}
\begin{bt}%[Sở GD BẮC NINH TOÁN 10 năm học 2016 - 2017]%[0D6K2]%[Nguyễn Thành Sơn]
	Cho $\tan \alpha = - 2$. Tính giá trị của biểu thức $P = \dfrac{- \sin \alpha + 4 \cos \alpha}{\sin \alpha + 3 \cos \alpha}$.
	\loigiai{
	Ta có: $P = \dfrac{- \dfrac{\sin \alpha}{\cos \alpha} + 4}{\dfrac{\sin \alpha}{\cos \alpha} + 3} = \dfrac{- \tan \alpha + 4}{\tan \alpha + 3} = \dfrac{2 + 4}{-2 + 3} = 6$.
	}
\end{bt}
\begin{bt}%[Sở GD BẮC NINH TOÁN 10 năm học 2016 - 2017]%[0D6Y2]%[Nguyễn Thành Sơn]
	Tính giá trị của biểu thức $P = \cos \dfrac{\pi}{7} + \cos \dfrac{3 \pi}{7} + \cos \dfrac{5 \pi}{7}$.
	\loigiai{
	Ta có $\cos \alpha = - \cos( \pi - \alpha)$. \\
	Nên $P = \cos \dfrac{\pi}{7} + \cos \dfrac{3 \pi}{7} + \cos \dfrac{5 \pi}{7} = - \left(\cos \dfrac{2\pi}{7} + \cos \dfrac{4 \pi}{7}  + \cos \dfrac{6 \pi}{7} \right)$ \\
	$\Leftrightarrow -2 \sin \dfrac{\pi}{7} P = \sin \dfrac{\pi}{7} \left(\cos \dfrac{2\pi}{7} + \cos \dfrac{4 \pi}{7}  + \cos \dfrac{6 \pi}{7} \right)$ \\
	$\Leftrightarrow -2 \sin \dfrac{\pi}{7} P = \sin \dfrac{3 \pi}{7} - \sin \dfrac{\pi}{7} + \sin \dfrac{5\pi}{7} - \sin \dfrac{3\pi}{7} + \sin \dfrac{7\pi}{7} - \sin \dfrac{5\pi}{7} \Leftrightarrow - 2P \sin \dfrac{\pi}{7} = -\sin \dfrac{\pi}{7}$\\
	$P = \dfrac{1}{2}$.
	}
\end{bt}
\begin{bt}%[Sở GD BẮC NINH TOÁN 10 năm học 2016 - 2017]%[0H3K2]%[Nguyễn Thành Sơn]
	Trong mặt phẳng với hệ tọa độ $Oxy$, cho các điểm $A\left(1;1 \right), B\left(2;3 \right), C\left(5; - 1 \right)$. Chứng minh $A, B, C$ là ba đỉnh của một tam giác và lập phương trình đường tròn ngoại tiếp tam giác $ABC$.
	\loigiai{
	Ta có $\overrightarrow{AB} = \left(1;2 \right)$, $\overrightarrow{AC} = \left(4; -2 \right)$. \\
	$\bullet$ Suy ra $\overrightarrow{AB} \cdot \overrightarrow{AC} = 0 \Rightarrow A, B, C$ lập thành một tam giác vuông tại $A$. \\
	$\bullet$ $\overrightarrow{BC} = \left(3; -4 \right) \Rightarrow BC = 5$.\\
	Gọi $I$ là trung điểm cạnh $BC \Rightarrow I \left(\dfrac{7}{2}; 1 \right)$.\\
	Đường tròn $\left( \mathscr{T} \right)$ ngoại tiếp tam giác $ABC$ có tâm $I \left(\dfrac{7}{2}; 1 \right)$ và bán kính $R = \dfrac{BC}{2} = \dfrac{5}{2}$. \\
	Phương trình đường tròn $\left( \mathscr{T} \right): \left(x - \dfrac{7}{2} \right)^2 + (y -1)^2 = \dfrac{25}{4}$.
	}
\end{bt}
\begin{bt}%[Sở GD BẮC NINH TOÁN 10 năm học 2016 - 2017]%[0H1B4]%[Nguyễn Thành Sơn]
	Trong mặt phẳng với hệ tọa độ $Oxy$, cho tam giác $ABC$ với $A\left(1;1 \right), B\left(2;3 \right), C\left(5; - 1 \right)$. Tìm tọa độ điểm $D$ sao cho tứ giác $ABDC$ là hình bình hành.
	\loigiai{
	Tứ giác $ABDC$ là hình bình hành $\Leftrightarrow \overrightarrow{AB} = \overrightarrow{CD} \Leftrightarrow \heva{&x_D - 5 = 1 \\&y_D + 1 = 2} \Leftrightarrow \heva{&x_D = 6 \\ &y_D = 1}.$ \\
	Vậy $D(6;1)$ là điểm cần tìm.
	}
\end{bt}
\begin{bt}%[Sở GD BẮC NINH TOÁN 10 năm học 2016 - 2017]%[0H3K1]%[Nguyễn Thành Sơn]
	Trong mặt phẳng với hệ tọa độ $Oxy$, cho tam giác $ABC$ với $A\left(1;1 \right), B\left(2;3 \right), C\left(5; - 1 \right)$. Viết phương trình đường thẳng $AM$ với $M$ là điểm thuộc đường thẳng $BC$ sao cho $S_{\triangle ABM} = \dfrac{1}{2} \cdot S_{\triangle ABC}$.
	\loigiai{
	$\overrightarrow{BC} = \left(3; -4 \right)$; $\overrightarrow{BM} = \left(x_M - 2; y_M - 3 \right)$. \\
	Ta có $S_{\triangle ABM} = \dfrac{1}{2} \cdot S_{\triangle ABC} \Leftrightarrow \dfrac{1}{2} d(A,BM) \cdot BM = \dfrac{1}{4}d(A, BC) \cdot BC \Leftrightarrow BC = 2BM \Leftrightarrow \hoac{& \overrightarrow{BC} = 2 \overrightarrow{BM} \\& \overrightarrow{BC} = -2 \overrightarrow{BM}}$	\\
	$\bullet$ Trường hợp $1$: $\overrightarrow{BC} = 2 \overrightarrow{BM} \Leftrightarrow \heva{& 3 = 2 \left( x_M - 2 \right) \\ &-4 = 2 \left(y_M -3 \right)} \Leftrightarrow \heva{&x_M = \dfrac{7}{2} \\ &y_M = 1}$. \\
	Khi đó $\overrightarrow{AM} = \left(\dfrac{5}{2}; 0 \right)$. \\
	Đường thẳng $AM$ đi qua điểm $A(1;1)$ và có vec-tơ pháp tuyến $\overrightarrow{n} = (0;1)$. \\
	Vậy phương trình tổng quát đường thẳng $AM$ là $y = 1$. \\
	$\bullet$ Trường hợp $2$:$\overrightarrow{BC} = -2 \overrightarrow{BM} \Leftrightarrow \heva{& 3 = -2 \left( x_M - 2 \right) \\ &-4 = -2 \left(y_M -3 \right)} \Leftrightarrow \heva{&x_M = \dfrac{1}{2} \\ &y_M = 5}$. \\
	Khi đó $\overrightarrow{AM} = \left(-\dfrac{1}{2}; 4 \right)$. \\
	Đường thẳng $AM$ đi qua điểm $A(1;1)$ và có vec-tơ pháp tuyến $\overrightarrow{n_1} = (8;1)$. \\
	Phương trình đường thẳng $AM: 8(x - 1) + 1(y - 1) = 0 \Leftrightarrow 8x + y - 9 = 0$.
	}
\end{bt}
\begin{bt}%[Sở GD BẮC NINH TOÁN 10 năm học 2016 - 2017]%[0D4G4]%[Nguyễn Thành Sơn]
	Một nhóm bạn dự định tổ chức một chuyến du lịch sinh thái, chi phí chia đều cho mỗi người. Sau khi đã hợp đồng xong, vào giờ chót có hai người bận việc đột xuất không đi được. Vì vậy mỗi người phải trả thêm $300.000$ đồng so với dự kiến ban đầu. Tính số người lúc đầu dự định đi du lịch và giá của chuyến du lịch sinh thái biết rằng giá của chuyến du lịch này trong khoảng $7.000.000$ đồng đến $7.500.000$ đồng.
	\loigiai{
	Gọi $x$ (đồng) là số tiền mỗi người dự định đóng góp cho chuyến du lịch, $y$ (người) là số người dự định đi lúc đầu $\left(x,y > 0, \forall y \in \mathbb{N}, y > 2 \right)$. \\
	Theo giả thiết $xy = \left(x + 300000 \right) \cdot \left(y - 2 \right) \leq 150000 y - 300000$ \hfill (1) \\
	Mặt khác $7000000 \leq xy \leq 7500000$ \hfill (2) \\
	Từ $(1)$ và $(2)$ ta có $7000000 \leq y \left(150000y - 300000 \right) \leq 7500000$ \\ $\heva{&3y^2 - 6y - 140 \geq 0 \\& y^2 - 2y - 50 \leq 0} \Leftrightarrow \dfrac{3 + \sqrt{429}}{3} \leq y \leq 1 + \sqrt{51}$. \\
	Do $y \in \mathbb{N} \Rightarrow y = 8 \Rightarrow x = 900000$. \\
	Vậy theo dự kiến ban đầu có $8$ người đi du lịch và mỗi người phải đóng góp là $900000$ đồng.
	}
\end{bt}
