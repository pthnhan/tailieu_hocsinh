\begin{ex}%[Đề tuyển sinh 10 năm học 2018-2019 tỉnh Quảng Ninh]%[Nguyễn Thành Khang, dự án EXTHCS-06]%[9D1Y4]
	Thực hiện phép tính: $\dfrac{\sqrt{27}}{\sqrt{3}}$.
	\loigiai{
		Ta có $\dfrac{\sqrt{27}}{\sqrt{3}}=\sqrt{\dfrac{27}{3}}=\sqrt{9}=3$.	
	}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018-2019 tỉnh Quảng Ninh]%[Nguyễn Thành Khang, dự án EXTHCS-06]%[9D1B8]
	Rút gọn biểu thức: $P=\left(\dfrac{\sqrt{x}}{3+\sqrt{x}}+\dfrac{9+x}{9-x}\right)\cdot \left(3\sqrt{x}-x\right)$ với $x\ge 0$ và $x\ne 9$.
	\loigiai{
		Điều kiện: $x\ge 0;\ x\ne 9$. Ta có
		\begin{eqnarray*}
			P &=& \left(\dfrac{\sqrt{x}}{3+\sqrt{x}}+\dfrac{9+x}{9-x}\right)\cdot \left(3\sqrt{x}-x\right)\\
			&=& \left(\dfrac{\sqrt{x}\left(3-\sqrt{x}\right)}{\left(3+\sqrt{x}\right)\left(3-\sqrt{x}\right)}+\dfrac{9+x}{\left(3+\sqrt{x}\right)\left(3-\sqrt{x}\right)}\right)\cdot \left(3\sqrt{x}-x\right) \\
			&=& \dfrac{9+3\sqrt{x}}{\left(3+\sqrt{x}\right)\left(3-\sqrt{x}\right)}\cdot \sqrt{x}\cdot \left(3-\sqrt{x}\right)\\
			&=& 3\sqrt{x}.
		\end{eqnarray*}
	}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018-2019 tỉnh Quảng Ninh]%[Nguyễn Thành Khang, dự án EXTHCS-06]%[9D2B3]
	Xác định các hệ số $a,b$ để đồ thị hàm số $y=ax+b$ đi qua hai điểm $A(2;-2)$ và $B(-3;2)$.
	\loigiai{
		Đồ thị hàm số $y=ax+b$ đi qua hai điểm $A(2;-2)$ và $B(-3;2)$ nên ta có hệ phương trình
		$$\left\{\begin{aligned}&2a+b=-2 \\ &-3a+b=2 \end{aligned}\right. \Leftrightarrow \left\{\begin{aligned}&5a=-4 \\ &2a+b=-2\end{aligned}\right. \Leftrightarrow \left\{\begin{aligned}&a=-\dfrac{4}{5} \\ &b=-\dfrac{2}{5}.\end{aligned}\right.$$
		Vậy $a=-\dfrac{4}{5};b=-\dfrac{2}{5}$.
	}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018-2019 tỉnh Quảng Ninh]%[Nguyễn Thành Khang, dự án EXTHCS-06]%[9D4Y3]
	Giải phương trình: $x^2-4x+4=0$.
	\loigiai{
		Ta có $$x^2-4x+4=0 \Leftrightarrow (x-2)^2=0 \Leftrightarrow x=2.$$
		Vậy phương trình có nghiệm $x=2$.
	}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018-2019 tỉnh Quảng Ninh]%[Nguyễn Thành Khang, dự án EXTHCS-06]%[9D4B6]
	Tìm giá trị của $m$ để phương trình $x^2-2(m+1)x+m^2+3=0$ có hai nghiệm $x_1,x_2$ thoả mãn $\left|x_1\right|+\left|x_2\right|=10$.
	\loigiai{
		Phương trình có hai nghiệm $x_1,x_2$ khi và chỉ khi $$\Delta'\ge 0 \Leftrightarrow (m+1)^2-(m^2+3)\ge 0\Leftrightarrow 2m-2\ge 0 \Leftrightarrow m\ge 1.$$
		Khi đó, áp dụng định lý Vi-ét ta có $\left\{\begin{aligned}&x_1+x_2=2(m+1) \\ &x_1\cdot x_2=m^2+3\end{aligned}\right.$.\\
		Vì $x_1\cdot x_2=m^2+3>0,\ \forall m\ge 1$ nên $\left|x_1\cdot x_2\right|=x_1\cdot x_2=m^2+3$. Khi đó ta có:
		\begin{eqnarray*}
			\left|x_1\right|+\left|x_2\right|=10 &\Leftrightarrow& \left(\left|x_1\right|+\left|x_2\right|\right)^2=100 \\
			&\Leftrightarrow& x_1^2+2\left|x_1\cdot x_2\right|+x_2^2=100 \\
			&\Leftrightarrow& x_1^2+2x_1\cdot x_2+x_2^2=100 \\			
			&\Leftrightarrow& \left(x_1+x_2\right)^2=100.
		\end{eqnarray*}
		Mà $x_1+x_2=2(m+1)$ nên ta có $$4(m+1)^2=100 \Leftrightarrow (m+1)^2=25 \Leftrightarrow \left[\begin{aligned}&m=4 &(\text{thoả mãn}) \\ &m=-6 &(\text{loại}).\end{aligned}\right.$$
		Vậy $m=4$ thoả mãn đề bài.
	}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018-2019 tỉnh Quảng Ninh]%[Nguyễn Thành Khang, dự án EXTHCS-06]%[9D4K8]
	\textit{Giải bài toán sau bằng cách lập phương trình hoặc hệ phương trình:}\\
	Một xe ô tô đi từ A đến B theo đường quốc lộ cũ dài $156$ km với vận tốc không đổi. Khi từ B về A, xe đi đường cao tốc mới nên quãng đường giảm được $36$ km so với lúc đi và vận tốc tăng so với lúc đi là $32$ km/h. Tính vận tốc ô tô khi đi từ A đến B, biết thời gian đi nhiều hơn thời gian về là $1$ giờ $45$ phút.
	\loigiai{
		Đổi: $1$ giờ $45$ phút $=\dfrac{7}{4}$ giờ.\\
		Gọi vận tốc của ô tô khi đi từ A đến B là $x$ km/h ($x>0$).\\
		Thời gian ô tô đi từ A đến B là $\dfrac{156}{x}$ giờ.\\
		Quãng đường lúc về là $120$ km và vận tốc của ô tô lúc về là $x+32$ km/h nên thời gian ô tô đi từ B về A là $\dfrac{120}{x+32}$ giờ.\\
		Theo bài ra ta có phương trình
		\begin{eqnarray*}
			\dfrac{156}{x}-\dfrac{120}{x+32}=\dfrac{7}{4} &\Leftrightarrow& \dfrac{36x+4992}{x(x+32)}=\dfrac{7}{4} \\
			&\Leftrightarrow& 4(36x+4992)=7x(x+32) \\
			&\Leftrightarrow& 7x^2+80x-19968=0 \\
			&\Leftrightarrow& \left[\begin{aligned}&x=48 &(\text{thoả mãn}) \\ &x=-\dfrac{416}{7} &(\text{loại}).\end{aligned}\right.
		\end{eqnarray*}
		Vậy vận tốc của ô tô khi đi từ A đến B là $48$ km/h.
	}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018-2019 tỉnh Quảng Ninh]%[Nguyễn Thành Khang, dự án EXTHCS-06]%[9H3Y4]%[9H3B7]%[9H3K7]
	Cho đường tròn tâm $O$, đường kính $AB=2R$. Trên đường tròn $(O)$ lấy điểm $C$ bất kì ($C$ không trùng với $A$ và $B$). Tiếp tuyến của đường tròn $(O)$ tại $A$ cắt tia $BC$ ở điểm $D$. Gọi $H$ là hình chiếu của $A$ trên đường thẳng $DO$. Tia $AH$ cắt đường tròn $(O)$ tại điểm $F$ (không trùng với $A$). Chứng minh:
	\begin{enumerate}
		\item $DA^2=DC\cdot DB$.
		\item Tứ giác $AHCD$ nội tiếp.
		\item $CH\perp CF$.
		\item $\dfrac{BH\cdot BC}{BF}=2R$.
	\end{enumerate}
	\loigiai{
		\immini{
			\vspace*{-2em}
			\begin{enumerate}
				\item Ta có $\widehat{ACB}=90^\circ$ (góc nội tiếp chắn nửa đường tròn tâm $O$) nên $AC\perp CB$ hay $AC\perp BD$.\\
				Vì $AD$ là tiếp tuyến của đường tròn tâm $O$ tại $A$ nên $\widehat{DAB}=90^\circ$.\\
				Áp dụng hệ thức lượng trong tam giác vuông $ABD$ ta có $DA^2=DC\cdot DB$.
				\item Ta có $\widehat{AHD}=\widehat{ACD}=90^\circ$ nên $C,H$ cùng nhìn cạnh $AD$ dưới một góc vuông nên tứ giác $AHCD$ nội tiếp.
			\end{enumerate}		
		}{
			\begin{tikzpicture}
				\tkzInit[xmin=-1, xmax=5.5, ymin=-2.6, ymax=3.5]
				\tkzClip
				\tkzDefPoints{0/0/A,5/0/B,0/3/D}
				\tkzDefMidPoint(A,B)\tkzGetPoint{O}
				\tkzDrawCircle(O,A)
				\tkzInterLC(D,B)(O,A)\tkzGetFirstPoint{C}
				\tkzDefPointBy[projection=onto O--D](A)\tkzGetPoint{H}
				\tkzDefPointBy[symmetry=center H](A)\tkzGetPoint{F}
				\tkzDrawPoints[fill=black](A,B,O,C,D,H,F)
				\tkzDrawSegments(A,B A,C D,B A,F C,H D,O H,B C,F F,B)
				\tkzMarkRightAngles(A,H,D)
				\tkzDrawLines[add = 1.0 and 0.5](A,D)
				\tkzLabelPoints[above](C,F)
				\tkzLabelPoints[below](O,H)
				\tkzLabelPoints[left](A,D)
				\tkzLabelPoints[right](B)
			\end{tikzpicture}
		}
		\begin{enumerate}
			\item[c)] Tứ giác $ADCH$ nội tiếp nên suy ra $\widehat{HCB}=\widehat{DAH}$.\\
			Mà $\widehat{FCB}=\widehat{FAB}$ nên $\widehat{HCF}=\widehat{HCB}+\widehat{FCB}=\widehat{DAH}+\widehat{FAB}=90^\circ$, hay $CH\perp CF$.
			\item[d)] Áp dụng hệ thức lượng trong tam giác vuông $ADO$ ta có $OA^2=OH\cdot OD$, mà $OA=OB$ nên $OB^2=OH\cdot OD \Rightarrow \dfrac{OB}{OH}=\dfrac{OD}{OB}\Rightarrow \triangle OBD\backsim \triangle OHB$ (c.g.c) $\Rightarrow \widehat{ODB}=\widehat{OBH}$.\\
			Lại có $\widehat{ODB}=\widehat{CAH}=\widehat{CBF} \Rightarrow \widehat{OBH}=\widehat{CBF} \Rightarrow \widehat{ABC}=\widehat{HBF}$.\\
			Mà $\widehat{ACB}=\widehat{HFB}=90^\circ \Rightarrow \triangle ABC\backsim \triangle HBF\Rightarrow \dfrac{AB}{HB}=\dfrac{BC}{BF} \Rightarrow \dfrac{BH\cdot BC}{BF}=AB=2R$.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018-2019 tỉnh Quảng Ninh]%[Nguyễn Thành Khang, dự án EXTHCS-06]%[9D5G1]
	Cho $x,y$ là các số thực dương thoả mãn $xy+1\le x$. Tìm giá trị lớn nhất của biểu thức $Q=\dfrac{x+y}{\sqrt{3x^2-xy+y^2}}$.
	\loigiai{
		Ta có $xy+1\le x \Rightarrow y+\dfrac{1}{x}\le 1$.\\
		Áp dụng bất đẳng thức Cô-si ta có $$1\ge y+\dfrac{1}{x}\ge 2\sqrt{\dfrac{y}{x}} \Rightarrow \dfrac{y}{x}\le \dfrac{1}{4} \Rightarrow x-4y\ge 0.$$
		Khi đó $Q=\dfrac{x+y}{\sqrt{3x^2-xy+y^2}}=\dfrac{x+y}{\sqrt{\dfrac{9}{5}(x+y)^2+\dfrac{1}{5}(x-4y)(6x+y)}}\le \dfrac{x+y}{\sqrt{\dfrac{9}{5}(x+y)^2}}=\dfrac{\sqrt{5}}{3}$.\\
		Dấu bằng xảy ra khi và chỉ khi $\left\{\begin{aligned}&xy+1=x \\ &x=4y\end{aligned}\right. \Leftrightarrow \left\{\begin{aligned}&x=2 \\ &y=\dfrac{1}{2}\end{aligned}\right.$.\\
		Vậy giá trị lớn nhất của $Q$ là $\dfrac{\sqrt{5}}{3}$.
	}
\end{ex}
