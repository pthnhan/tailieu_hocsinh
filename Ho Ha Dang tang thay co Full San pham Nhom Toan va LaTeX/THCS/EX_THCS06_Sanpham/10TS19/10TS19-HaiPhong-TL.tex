\begin{bt}%[Tuyển sinh 10 Hải Phòng 2019]%[Nguyễn Kim Đông]%[9D1K6]
Cho biểu thức 
\begin{align*}
	&A=\sqrt{3}\left(\sqrt{3}-3\sqrt{12}+2\sqrt{27}\right),\\
	&B=\left(1+\dfrac{x+\sqrt{x}}{\sqrt{x}+1}\right) \cdot \left(1-\dfrac{x-\sqrt{x}}{\sqrt{x}-1}\right) (\text{ với } x>0,x\neq 1).
\end{align*}
\begin{enumerate}[1)]
	\item Rút gọn biểu thức $A$ và $B$.
	\item Tìm các giá trị của $x$ sao cho $A\cdot B\leq 0$.
\end{enumerate}
\loigiai{
\begin{enumerate}[1)]
	\item Ta có
	\begin{align*}
	A&=\sqrt{3}\left(\sqrt{3}-3\sqrt{12}+2\sqrt{27}\right)\\
	&=\sqrt{3}\left(\sqrt{3}-6\sqrt{3}+6\sqrt{3}\right)\\
	&=3.	
	\end{align*}
	Với $x>0$ và $x \neq 1$, ta có
	\begin{align*}
		B&=\left(1+\dfrac{x+\sqrt{x}}{\sqrt{x}+1}\right) \cdot \left(1-\dfrac{x-\sqrt{x}}{\sqrt{x}-1}\right)\\
		&=\dfrac{x+2\sqrt{x}+1}{\sqrt{x}+1}\cdot \dfrac{-x+2\sqrt{x}-1}{\sqrt{x}-1}\\
		&=\dfrac{\left(\sqrt{x}+1\right)^2}{\sqrt{x}+1}\cdot \dfrac{-\left(\sqrt{x}-1\right)^2}{\sqrt{x}-1}\\
		&=-\left(\sqrt{x}+1\right)\cdot \left(\sqrt{x}-1\right)\\
		&=1-x. 
	\end{align*}
	\item $A\cdot B \leq 0 \Leftrightarrow 3\left(1-x\right)\leq 0 \Leftrightarrow x \geq 1$.\\
	Vậy $x \geq 1$ thỏa yêu cầu bài toán.
\end{enumerate}}
\end{bt}

	\loigiai{ Vì đồ thị hàm số $y=ax+b$ song song với đường thẳng $y=2x-1$ nên ta có $a=2$ và $b \neq -1$. Khi đó, ta có đường thẳng $y=2x+b$.\\
			Vì đồ thị hàm số $y=2x+b$ đi qua điểm có tọa độ $\left(0;3\right)$ nên ta có $3=b$.\\
			Vậy $a=2$ và $b=3$.
		
	}
\end{bt}

\begin{bt}%[Tuyển sinh 10 Hải Phòng 2019]%[Nguyễn Kim Đông]%[9D3G4]
Giải hệ phương trình: $\begin{cases}
	3x+\sqrt{y+6}=11 \\ 
	5x-\sqrt{y+6}=13.\end{cases}$	
	\loigiai{
	 Điều kiện xác định: $y \geq -6$.\\
	Đặt $b=\sqrt{y+6}$, hệ phương trình đã cho trở thành 
	$\begin{cases} 3x+b=11 \\ 5x-b=13 \end{cases}\Leftrightarrow \begin{cases} x=3\\ b=2.	\end{cases}$\\
	Với $b=2$, ta có $\sqrt{y+6}=2 \Leftrightarrow y+6=4 \Leftrightarrow y=-2$ ( thỏa điều kiện).\\
	Vậy $(x;y)=(3;-2)$.		
	}
\end{bt}
\begin{bt}%[Tuyển sinh 10 Hải Phòng 2019]%[Nguyễn Kim Đông]%[9D4K4][9D4G6]
	Cho phương trình ẩn $ x$: $x^2-2\left(m+1\right)x+m^2+1=0$ $(*)$ ($m$  là tham số).
	\begin{enumerate}
		\item Giải phương trình $(*)$ khi $m=2$.
		\item Xác định các giá trị của tham số $m$ để phương trình $(*)$ có hai nghiệm phân biệt $x_1$, $x_2$ thỏa mãn điều kiện $x_1-2x_2=-1$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Khi $m=2$, phương trình $(*)$ trở thành $x^2-6x+5=0$ $(1)$.\\
			Vì $1-6+5=0$ nên phương trình $(1)$ có hai nghiệm phân biệt $x_1=1$ và $x_2=5$.
			\item $\Delta \rq = (m+1)^2-(m^2+1)=2m$.\\
			Phương trình $(*)$ có hai nghiệm phân biệt $x_1$, $x_2$ khi và chỉ khi $\Delta \rq =2m>0 \Leftrightarrow m>0$.\\
			Với $m>0$, theo định lí Vi-et, ta có $\begin{cases} x_1+x_2=2m+2\\ x_1 \cdot x_2 =m^2+1.	\end{cases}$\\
			Xét hệ phương trình $\begin{cases} x_1+x_2=2m+2\\x_1-2x_2=-1 \end{cases}$, ta có $\begin{cases} x_1=\dfrac{4m+3}{3}\\ x_2=\dfrac{2m+3}{3}.
			\end{cases}$\\
			Khi đó,
			\begin{equation*}
			x_1 \cdot x_2 =m^2+1 \Leftrightarrow \dfrac{4m+3}{3}\cdot \dfrac{2m+3}{3}=m^2+1 \Leftrightarrow m^2-18m=0 \Leftrightarrow \left[\begin{aligned} & m=0 \text{ (loại)}\\ & m=18 \text{ (thỏa mãn).}\end{aligned}\right. 
			\end{equation*}
			Vậy $m=18$ thỏa mãn yêu cầu bài toán.
		\end{enumerate}
	}
\end{bt}



\begin{bt}%[Tuyển sinh 10 Hải Phòng 2019]%[Nguyễn Kim Đông]%[9D3G5]
\textbf{ Bài toán thực tế:}
		Khoảng cách giữa hai thành phố $A$ và $B$ là $144$ km. Một ô tô khởi hành từ thành phố $A$ đến thành phố $B$ với vận tốc không đổi trên cả quãng đường. Sau khi ô tô thứ nhất đi được $20$ phút, ô tô thứ $2$ cũng đi từ thành phố $A$ đến thành phố $B$ với vận tốc lớn hơn vận tốc của ô tô thứ nhất là $6$ km/h (vận tốc không đổi). Biết rằng cả hai ô tô cùng đến thành phố $B$ cùng một lúc.
		\begin{enumerate}
		\item Tính vận tốc của mỗi xe ô tô.
	\item Nếu trên đường đó có biển báo cho phép xe chạy với vận tốc tối đa là 50km/h thì hai xe ô tô trên, xe nào vi phạm về tốc giới hạn tốc độ?
		\end{enumerate}
	\loigiai{ 
		\begin{enumerate}
			\item Gọi $x$ (km/h) $(x>0)$ là vận tốc xe ô tô thứ nhất.\\
			$y$ (km/h) $(x>0)$ là vận tốc xe ô tô thứ hai.\\
			Ta có hệ phương trình 
			\begin{center}
				$\begin{cases} y-x=6\\ \dfrac{144}{x}-\dfrac{144}{y}=\dfrac{1}{3}
				\end{cases} \Leftrightarrow \begin{cases} y=x+6\\ \dfrac{144}{x}-\dfrac{144}{x+6}=\dfrac{1}{3}. \quad (1)
				\end{cases}$
			\end{center}
		Giải $(1)$: \\
		\begin{equation*}
			\dfrac{144}{x}-\dfrac{144}{x+6}=\dfrac{1}{3} \Leftrightarrow  x^2+6x-2592=0 \Leftrightarrow \left[\begin{aligned} & x=-54 \text{ (loại)}\\ & x=48 \text{ (thỏa mãn).}\end{aligned}\right.
		\end{equation*}
		Với $x=48$, ta có $y=54$.\\
		Vậy vận tốc xe ô tô thứ nhất là $48$ km/h, vận tốc xe ô tô thứ hai là $54$ km/h.
		\end{enumerate}
	\item Xe thứ hai vi phạm về giới hạn tốc độ ( vì $54$ km/h $> 48$ km/h).	
	}
\end{bt}

\begin{bt}%[Tuyển sinh 10 Hải Phòng 2019]%[Nguyễn Kim Đông]%[9H3K7][9H3K7][9H3G7]
 Cho tam giác $ABC$ ($AB<AC$) có ba góc nhọn nội tiếp đường tròn $(O)$, $AH$ là đường cao của tam giác $ABC$. Kẻ đường kính $AD$ của đường tròn $(O)$. Từ hai điểm $B $ và $C$ kẻ $BE\perp AD$ tại $E$ và $CF \perp AD$ tại $F$.
		\begin{enumerate}
			\item Chứng minh tứ giác $ABHE$ nội tiếp đường tròn.
			\item Chứng minh: $HE \parallel CD$.
			\item Gọi $I$ là trung điểm của $BC$. Chứng minh: $IE=IF$.
		\end{enumerate}

	\loigiai{ \begin{enumerate}
			\item \begin{center}
				\begin{tikzpicture}
				\tkzDefPoints{3/5/A,1/0/B,7/0/C}
				\tkzDefCircle[circum](A,B,C)     \tkzGetPoint{O} %tâm đường tròn
				\tkzDrawCircle[circum](A,B,C)
				\tkzDefPointBy[projection=onto B--C](A)\tkzGetPoint{H}
				\tkzInterLC(A,O)(O,A)    \tkzGetPoints{A}{D}
				\tkzDefPointBy[projection=onto A--D](B)\tkzGetPoint{E}
				\tkzDefPointBy[projection=onto A--D](C)\tkzGetPoint{F}
				\tkzDefMidPoint(B,C)\tkzGetPoint{I}
				\tkzMarkRightAngles[size=0.2,fill=green!50](A,H,B)
				\tkzMarkRightAngles[size=0.2,fill=green!50](B,E,A)
				\tkzMarkRightAngles[size=0.2,fill=green!50](C,F,A)
				\tkzMarkSegments[mark=||](I,C I,B)
				\tkzDefMidPoint(B,A) \tkzGetPoint{N}
				\tkzDefMidPoint(A,C)	\tkzGetPoint{K}
				\tkzMarkSegments[mark=|](N,A N,B)
				\tkzMarkSegments[mark=|||](K,A K,C)
				\tkzDrawSegments(A,B B,C C,A A,H A,D B,E C,F H,E C,D H,F I,E I,F I,N I,K B,D) 
				\tkzDrawPoints(A,B,C,I,H,D,E,F,O,N,K)
				\tkzLabelPoints[left](B,O,N)
				\tkzLabelPoints[right](C,K)
				\tkzLabelPoints[above](A,I,E)
				\tkzLabelPoints[below left](H,F,D)
				\end{tikzpicture}
			\end{center}
			Xét tứ giác $ABHE$ có $\widehat{AHB}=90^\circ$ và $\widehat{AEB}=90^\circ$. Vì hai góc $AHB$ và $AEB$ ở vị trí đối nhau và $\widehat{AHB}+\widehat{AEB}=180^\circ$ nên tứ giác $ABHE$ là tứ giác nội tiếp.
			\item Vì tứ giác $ABHE$ là tứ giác nội tiếp nên $\widehat{BAD}=\widehat{CHE}$ ( vì cùng bù với góc $\widehat{BHE}$).\\
			Mặt khác, $\widehat{BAD}=\widehat{BCD}$ ( vì cùng chắn cung $BD$). Do đó, $\widehat{EHC}=\widehat{HCD}$.\\
			Vì hai góc $EHC$ và $HCD$ ở vị trí so le trong nên $HE \parallel CD$.
			\item Gọi $N$ và $K$ lần lượt là trung điểm của các cạnh $AB$ và $AC$. Khi đó, $IN$ là đường trung bình của tam giác $ABC$ mêm $IN \parallel AC$.\\
			Vì $\begin{cases} AC \perp CD\\ CD \parallel HE	\end{cases}$ nên $HE \perp AC$.\\
			Mặt khác, vì $IN \parallel AC$ nên $HE \perp IN$.
			Vì tứ giác $ABHE$ nội tiếp đường tròn tâm $N$ và $IN \perp HE$ nên $IN$ là đường trung trực của đoạn thẳng $HE$. Suy ra, $IE=IH$. \quad $(1)$.\\
			Tứ giác $AHFC$ có $\widehat{AHC}$ và $\widehat{AFC}$ cùng nhìn cạnh $AC$ dưới một góc $90^\circ$ nên tứ giác $AHFC$ nội tiếp đường tròn tâm $K$ ( vì $K$ là trung điểm của đoạn thẳng $AC$). \\
			Khi đó, vì  $\widehat{CHF}=\widehat{CAF}$ ( cùng chắn cung $CF$) và $\widehat{CAF}=\widehat{CBD}$ (cùng chắn cung $CD$) nên $\widehat{CHF}=\widehat{CBD}$. Mặt khác, hai góc này ở vị trí đồng vị nên $HF \parallel BD$.\\
			Vì $\begin{cases} AB \perp BD	\\ IK \parallel AB		\end{cases}$ nên $IK \perp BD$. Mà $HF \parallel BD$ nên $IK \perp IK$.\\
			Vì tứ giác $AHFC$ là tứ giác nội tiếp đường tròn tâm $K$ nên $IK$ là đường trung trực của đoạn thẳng $HF$. Do đó, $IH=IF$ \quad $(2)$.\\
			Từ $(1)$ và $(2)$, suy ra $IE=IF$.
		\end{enumerate}	
	}
\end{bt}

\begin{bt}%[Tuyển sinh 10 Hải Phòng 2019]%[Nguyễn Kim Đông]%[9D5G1][9D5G1]
	\begin{enumerate}
		\item Chứng minh với mọi số thực $a$, $b$, $c$ ta có: $ab+bc+ca \leq \dfrac{{\left(a+b+c\right)}^2}{3}$.
		\item Cho ba số dương $x$, $ y$, $z$ thỏa mãn điều kiện: $x+y+z=\dfrac{3}{4}$.Chứng minh: 
		\begin{equation*}
			6\left(x^2+y^2+z^2\right)+10\left(xy+yz+zx\right)+2\left(\dfrac{1}{2x+y+z}+\dfrac{1}{x+2y+z}+\dfrac{1}{x+y+2z}\right)\geq 9.
		\end{equation*}
		Đẳng thức xảy ra khi nào?
		
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Ta có
		\begin{align*}
			&ab+bc+ca \leq \dfrac{{\left(a+b+c\right)}^2}{3}\\
			\Leftrightarrow & 3\left(ab+bc+ca\right) \leq \left(a+b+c\right)^2\\
			\Leftrightarrow & ab+bc+ca \leq a^2+b^2 +c^2\\
			 \Leftrightarrow & 2(ab+bc+ca) \leq 2(a^2+b^2+c^2)\\
			\Leftrightarrow & (a-b)^2+(b-c)^2 +(c-a)^2 \geq 0 \quad (*).
		\end{align*}
		Vì bất đẳng thức $(*)$ luôn đúng với mọi số thực $a$, $b$, $c$ nên ta có điều phải chứng minh.
		\item Trước hết, ta có $x^2+y^2+z^2 \geq \dfrac{1}{3} (x+y+z)^2$, với mọi số dương $x$, $y$, $z$;\\
		và 
		\begin{equation*}
			\dfrac{1}{2x+y+z}+\dfrac{1}{x+2y+z}+\dfrac{1}{x+y+2z} \geq \dfrac{9}{4(x+y+z)}=3.
		\end{equation*}
		 Đẳng thức xảy ra khi và chỉ khi $x=y=z$.\\
		Khi đó, ta có
		\begin{align*}
			&6\left(x^2+y^2+z^2\right)+10\left(xy+yz+zx\right)+2\left(\dfrac{1}{2x+y+z}+\dfrac{1}{x+2y+z}+\dfrac{1}{x+y+2z}\right)\\
			= & 5(x+y+z)^2 +x^2+y^2+z^2 +2\cdot \left(\dfrac{1}{2x+y+z}+\dfrac{1}{x+2y+z}+\dfrac{1}{x+y+2z}\right) \\
			= & 5\cdot \dfrac{9}{16}+x^2+y^2+z^2 +2\cdot \left(\dfrac{1}{2x+y+z}+\dfrac{1}{x+2y+z}+\dfrac{1}{x+y+2z}\right) \\
			\geq & \dfrac{45}{16}+ \dfrac{1}{3} (x+y+z)^2 +2\cdot \dfrac{9}{4(x+y+z)}\\
			=& \dfrac{45}{16}+ \dfrac{1}{3}\cdot \dfrac{9}{16}+2\cdot 3=9.
		\end{align*}
	Đẳng thức xảy ra khi và chỉ khi $\begin{cases} x=y=z>0\\ x+y+z=\dfrac{3}{4}		\end{cases} \Leftrightarrow x=y=z=\dfrac{1}{4}.$
	\end{enumerate}	
	}
\end{bt}
