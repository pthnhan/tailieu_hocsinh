\begin{ex}%[Đề thi tuyển sinh vào 10 toán chuyên, tỉnh Bắc Giang, năm 2018]%[Huỳnh Thị Sâm, Dự án EX-TS10C-18]%[9D0G1]%[9D0K4]
Cho biểu thức $A = \left(\dfrac{x+ 4\sqrt x +4}{x+ \sqrt x -2} + \dfrac{x+ \sqrt x}{1-x}\right) : \left(\dfrac{1}{\sqrt x +1} - \dfrac{1}{1- \sqrt x}\right)$ (với $x> 0; x\ne 1$).
	\begin{enumerate}
		\item Rút gọn biểu thức $A $.
		\item Có bao nhiêu giá trị nguyên của $x$ để $A \ge \dfrac{1+ \sqrt{2018}}{\sqrt{2018}}$.
	\end{enumerate}
	\loigiai{	\begin{enumerate}
			\item Với $x> 0;\ x\ne 1.$\\
			Ta có $\begin{aligned}[t]
			A =& \left(\dfrac{x+ 4\sqrt x +4}{x+ \sqrt x -2} + \dfrac{x+ \sqrt x}{1-x}\right) : \left(\dfrac{1}{\sqrt x +1} - \dfrac{1}{1- \sqrt x}\right)\\
			=& \left[\dfrac{(\sqrt x+2)^2}{(\sqrt x-1)(\sqrt x+2)} + \dfrac{\sqrt x(\sqrt x+1)}{(1-\sqrt x)(\sqrt x +1)}\right]: \left(\dfrac{1}{\sqrt x +1} - \dfrac{1}{1-\sqrt x}\right)\\
			=& \left[\dfrac{\sqrt x +2}{\sqrt x-1}+ \dfrac{\sqrt x}{1-\sqrt x}\right]: \dfrac{1-\sqrt x - \sqrt x-1}{1-x}\\
			=& \dfrac{\sqrt x + 2-\sqrt x}{\sqrt x -1} \cdot \dfrac{x-1}{2\sqrt x}
			=\dfrac{2}{\sqrt x -1}\cdot \dfrac{(\sqrt x -1)(\sqrt x+1)}{2\sqrt x}= \dfrac{\sqrt x +1}{\sqrt x}.
			\end{aligned}$
			\item Với $A\ge \dfrac{1+ \sqrt{2018}}{\sqrt{2018}}$ thì\\
			$\dfrac{\sqrt x +1}{\sqrt x} \ge \dfrac{1+ \sqrt{2018}}{\sqrt{2018}} \Leftrightarrow \sqrt{2018}(\sqrt x +1) \ge \sqrt x (1+ \sqrt{2018}) \Leftrightarrow \sqrt {2018}\ge \sqrt x > 0 \\\Leftrightarrow 0 < x \le 2018$.\\
			Mà $x$ nguyên .\\
			Vậy có tất cả $2018$ giá trị $x$ để $A\ge \dfrac{1+ \sqrt{2018}}{\sqrt{2018}}.$
			\end{enumerate}} 
\end{ex}

	Cho phương trình $x^2 - (m+1)x - 3 = 0 \quad (1)$, với $x$ là số ẩn, $m$ là tham số. Gọi $x_1$, $x_2$ là hai nghiệm của phương trình (1). Đặt $B = \dfrac{3x_1^2 + 3x_2^2 + 4x_1 + 4x_2 -5}{x_1^2 + x_2^2 -4}$. Tìm $m$ để $B$ đạt giá trị lớn nhất.
	\loigiai{Để phương trình $(1)$ đã cho có hai nghiệm phân biệt thì $\begin{cases} a=1\ne 0\\\Delta >0. \end{cases}\\ \Leftrightarrow (m-1)^2 - 4\cdot (-3) = (m-1)^2 +12 > 0 $ với mọi $m$.\\
		Theo định lý Vi-et ta có $\begin{cases} x_1x_2= -3\\ x_1+x_2= m+1.\end{cases}$\\
		Khi đó $\begin{aligned}[t]
		B = \dfrac{3x_1^2 + 3x_2^2 + 4x_1 + 4x_2 -5}{x_1^2 + x_2^2 -4}=&  \dfrac{3[(x_1+x_2)^2 -2x_1x_2] +4(x_1+x_2)-5}{(x_1+x_2)^2 - 2x_1x_2-4}\\
		=& \dfrac{3[(m+1)^2 - 2\cdot(-3)] +4(m+1)-5}{(m+1)^2 -2\cdot (-3) -4}\\
		=&\dfrac{3(m^2 +2m+1+6)+ 4m +4 -5}{m^2 +2m +1+2}
		= \dfrac{3m^2 +10m +20}{m^2+2m+3}  .
		\end{aligned}$ \\
		$\Rightarrow \begin{aligned}[t]
		B(m^2 +2m +3) = 3m^2 +10m +20 \Leftrightarrow&  (B -3)m^2 +(2B -10)m +3B- 20 =0\\
		\Leftrightarrow & (B-3)m^2 +2(B -5)m + 3B -20=0. \refstepcounter{equation}\hfill (\theequation)\label{1837.1}
		\end{aligned}$	\\
	Từ  \eqref{1837.1} có  nghiệm  thì 
		$\begin{aligned}[t]
		\Delta' >0 \Leftrightarrow &(B-5)^2 - (B-3)(3B-20) \geq 0 \\
		\Leftrightarrow & B^2 - 10B + 25 - 3B^2 + 20B + 9B -60 \geq 0\\
		\Leftrightarrow& -2B^2 +19B - 35 \geq 0 \Leftrightarrow \dfrac{5}{2} \leq B \leq 7. \\
		\end{aligned}$\\
		Vậy $B$ đạt giá trị lớn nhất bằng $7$ khi và chỉ khi $4m^2 + 4m + 1= 0 \Leftrightarrow (2m+1)^2 = 0 \Leftrightarrow m= -\dfrac{1}{2}.$}
	\end{ex}

	\loigiai{	ĐKXĐ  $\begin{cases} x\ge \dfrac{6}{5}\\ y \le \dfrac{16}{3}. \end{cases} $\\
		Ta có	$\begin{cases} x^2 -xy -x + 3y -6 =0 \refstepcounter{equation}\hfill (\theequation)\label{1837.2}\\ \sqrt{5x -6}+\sqrt{16 -3y} = 2x^2 - 2x + y -4 \refstepcounter{equation}\hfill (\theequation)\label{1837.3} . \end{cases}  $\\
		Từ phương trình \eqref{1837.2} ta có $x^2 - xy - x + 3y - 6 =0 \Leftrightarrow (x-3)(x+2-y)=0\Leftrightarrow \left[\begin{aligned} x & =3\\ y &=x+2. \end{aligned}\right.$\\
		Với $x = 3$ thay vào  \eqref{1837.3} ta được\\
		$\begin{aligned}[t]
		\sqrt{5\cdot 3 - 6} + \sqrt{16- 3y} = 2\cdot 3^2 -2\cdot 2+ y-4 \Leftrightarrow& 3+ \sqrt{16-3y} = 10-y\\ \Leftrightarrow &  \sqrt{16-3y} = 7-y \\
		\Leftrightarrow & \begin{cases} y \le7\\ 16 -3y = 49 -14y + y^2  \end{cases} \\\Leftrightarrow &\begin{cases} y \le 7\\ y^2 -11y +33= 0 \quad (PTVN). \end{cases}  \end{aligned}$.\\

		&\sqrt{5x-6} + \sqrt{10-3x} = 2x^2 -2x + x+2 -4\\ \Leftrightarrow &\sqrt{5x-6} + \sqrt{10-3x} = 2x^2 -x -2\\
		\Leftrightarrow&\sqrt{5x-6}-2 + \sqrt{10-3x}-2 = 2x^2 -x -6\\
		\Leftrightarrow & \dfrac{5x -10}{\sqrt{5x-6}+2} + \dfrac{6-3x}{\sqrt{10-3x}+2} = (x-2)(2x+3)\\
		\Leftrightarrow & \dfrac{5(x-2)}{\sqrt{5x-6}+2} - \dfrac{3(x-2)}{\sqrt{10-3x}+2} = (x-2)(2x+3)\\
		\Leftrightarrow& (x-2)\left(\dfrac{5}{\sqrt{5x-6}+2 }- \dfrac{3}{\sqrt{10-3x}+2}-2x -3\right)  =0\\
		\Leftrightarrow & x=2.
		\end{aligned}$\\
		(Vì $\dfrac{5}{\sqrt{5x-6}+2 }- \dfrac{3}{\sqrt{10-3x}+2}- (2x+3) <0$ với $x \ge \dfrac{6}{5}$) .\\
		Với $x = 2 \Rightarrow y= 4$.\\

\begin{ex}%[Đề thi tuyển sinh vào 10 toán chuyên, tỉnh Bắc Giang, năm 2018]%[Huỳnh Thị Sâm, Dự án EX-TS10C-18]%[9D0G4]
Chứng minh rằng không tồn tại số tự nhiên $n$ để $2018 + n^2$ là số chính phương.
	
	\loigiai{Giả sử $n^2 + 2018$ là số chính phương, đặt $n^2 + 2018 = p^2$ ($p$ là số tự nhiên lớn hơn $0$).\\
			Ta được $n^2 - p^2 + 2018 =0 \Leftrightarrow n^2 - p^2 = -2018 \Leftrightarrow (n-p)(n+p)=-2018 = (-1)\cdot 2018 = (-2018)\cdot 1 = (-1009)\cdot 2 = (-2)\cdot 1009.$\\
			Gọi $a = n - p,\ b = n+p$ ($a,\ b$ cũng là các số nguyên).\\
			Vì tích của $a$ và $b$ bằng $-2018$ là một số chẵn, suy ra trong 2 số $a$ và $b$ phải có ít nhất một số chẵn. $\refstepcounter{equation}\hfill (\theequation)\label{1837.4}$\\
			Mặt khác $a+ b = (n-p + n+p)= 2n$ là một số chẵn.\\
			Suy ra $a$ và $b$ phải cùng chẵn hoặc cùng lẻ. $\refstepcounter{equation}\hfill (\theequation)\label{1837.5}$\\
			 Từ \eqref{1837.4} và \eqref{1837.5} suy ra $a$ và $b$ đều là số chẵn.\\
			Do đó $a = 2k, b = 2l$ (với $k,\l$ là số nguyên).\\
			Theo trên ta có $a\cdot b = 2018 $ hay $2k\cdot 2l = 2018 \Leftrightarrow 4\cdot k\cdot l =2018.$\\
			Vì $k,l$ là số nguyên nên suy ra $2018$ phải chia hết cho $4$ (điều này vô lý, vì $2018$ không chia hết cho $4$).\\
			Vậy không tồn tại số nguyên $n$ thỏa mãn yêu cầu bài toán (đpcm).
}
\end{ex}
