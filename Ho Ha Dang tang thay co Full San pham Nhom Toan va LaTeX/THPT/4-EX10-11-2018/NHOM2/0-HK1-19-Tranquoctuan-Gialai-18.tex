\begin{bt}%[Đề HK1, Trần Quốc Tuấn, Gia Lai 2017]%[0D3B3]  
Giải các phương trình sau:
\begin{multicols}{2}
	\begin{enumerate}[a)]
		\item %[Đề HK1, Trần Quốc Tuấn, Gia Lai 2017]%[0D3B3] 
		 $x^4-3x^2-4=0$.
		\item %[Đề HK1, Trần Quốc Tuấn, Gia Lai 2017]%[0D3K3] 
		 $\sqrt{3x+7}=x+3$.
		\item %[Đề HK1, Trần Quốc Tuấn, Gia Lai 2017]%[0D3G3] 
		 $\dfrac{3}{2{x^2} + 13x + 6} + \dfrac{5}{2{x^2} + 7x + 6}=\dfrac{10}{21x}$.	
	\end{enumerate}
\end{multicols}	
\loigiai{
	\begin{enumerate}[a)]
		\item 	Ta có
		$\begin{aligned} x^4-3x^2-4=0 \Leftrightarrow \left[ \begin{array}{l}
		{x^2} =  - 1\\
		{x^2} = 4
		\end{array}  
		\right.
		 \Leftrightarrow x^2=4 \Leftrightarrow x=\pm 2.
		\end{aligned}$
		\item 	
		$\begin{aligned} \sqrt{3x+7}=x+3 \Leftrightarrow \left\{ \begin{array}{l}
		x + 3 \ge 0\\
		3x + 7 = \left( {x + 3} \right)^2
		\end{array} \right.\Leftrightarrow \left\{ \begin{array}{l}
		x + 3 \ge 0\\
		x^2+3x+2=0
		\end{array} \right.		 \Leftrightarrow \left[ \begin{array}{l}
		x =  - 2\\
		x =  - 1
		\end{array} \right.	
		\end{aligned}  $.
		\item 
	Điều kiện: $\left\{ \begin{array}{l}
	2x^2 + 13x + 6 \ne 0\\
	2x^2 + 7x + 6 \ne 0\\
	x \ne 0
	\end{array} \right.$\\
	Ta có, $\dfrac{3}{2x^2 + 13x + 6} + \dfrac{5}{2x^2 + 7x + 6} = \dfrac{10}{21x} \Leftrightarrow \dfrac{3}{2x + 13 + \dfrac{6}{x}} + \dfrac{5}{2x + 7 + \dfrac{6}{x}} = \dfrac{10}{21}$.\\
	Đặt $t = 2x + \dfrac{6}{x} + 7$, ta được $\dfrac{3}{t + 6} + \dfrac{5}{t} = \dfrac{10}{21} \Leftrightarrow 5t^2 - 54t - 315 = 0 \Leftrightarrow \left[ \begin{array}{l}
		t = 15\\
		t =  - \dfrac{21}{5}
		\end{array} \right.$ 
	\begin{itemize}
		\item $t = 15$ thì  $2x + \dfrac{6}{x} + 7 = 15 \Leftrightarrow x^2 - 4x + 3 = 0 \Leftrightarrow \left[ \begin{array}{l}
			x = 1\\
			x = 3
			\end{array} \right.$
	\end{itemize}
	\begin{itemize}
		\item $t =  - \dfrac{21}{5}$  \text{thì}  $2x + \dfrac{6}{x} + 7 =  - \dfrac{21}{5}$ $\Leftrightarrow 5x^2 + 28x + 15 = 0 \Leftrightarrow \left[ \begin{array}{l}
			x =  - 5\\
			x =  - \dfrac{3}{5}
			\end{array} \right.$
	\end{itemize}
	Vậy tập nghiệm của phương trình là $S = \left\{  - 5, - \dfrac{3}{5};1,3 \right\}.$
	\end{enumerate}
}
\end{bt}
