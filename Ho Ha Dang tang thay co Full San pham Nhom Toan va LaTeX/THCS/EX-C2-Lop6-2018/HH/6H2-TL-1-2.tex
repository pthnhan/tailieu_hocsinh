\section{Nửa mặt phẳng}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Mặt phẳng}
\begin{dn}
Mặt hồ nước yên lặng, mặt bảng là hình ảnh của mặt phẳng.
\end{dn}

\subsubsection{Nửa mặt phẳng}
\begin{dn}
Hình gồm đường thẳng $a$ và một phần mặt phẳng bị chia ra bởi $a$ được gọi là \textit{một nửa mặt phẳng bờ $a$}.
\end{dn}
\begin{tc}
Bất kì đường thẳng nào nằm trên mặt phẳng cũng là bờ chung của hai nửa mặt phẳng đối nhau.\\		
\immini{Trong hình bên:
\begin{itemize}
\item  Nửa mặt phẳng (I) và nửa mặt phẳng (II) là hai nửa mặt phẳng đối nhau có bờ chung là $a$.
\item  Hai điểm $M$, $N$ cùng thuộc nửa mặt phẳng (I), với $M, N\notin a$ thì đoạn thẳng $MN$ không cắt $a$.
\item  Hai điểm $M$, $P$ thuộc hai nửa mặt phẳng đối nhau bờ $a$, với $M, P\notin a$ thì đoạn thẳng $MP$ cắt $a$.
\end{itemize}
}{\begin{tikzpicture}
\tkzDefPoints{0/1/M,2/-1/P,3/2/N,-2/0/a,5/0/b}
\tkzDrawSegments(a,b N,M M,P)
\tkzDrawPoints(M,N,P)
\tkzLabelPoints[above](M,N,a)
\tkzLabelPoints[below](P)
\draw (4.7,0) node[below]{(II)};
\draw (4.7,0) node[above]{(I)};
\end{tikzpicture}}
\end{tc}

\subsection{Các dạng toán và phương pháp giải}

\begin{dang}{Đoạn thẳng cắt hay không cắt đường thẳng}
\textbf{\textit{Phương pháp giải}}
\begin{itemize}
\item Nếu hai điểm $A, B$ thuộc hai nửa mặt phẳng đối nhau bờ $a$ ($A, B\notin a$) thì đoạn thẳng $AB$ cắt $a$.
\item Nếu hai điểm $A, B$ cùng thuộc nửa mặt phẳng bờ $a$ ($A, B\notin a$) thì đoạn thẳng $AB$ không cắt $a$.
\end{itemize}
\end{dang}
\begin{vd}%[Đào Trung Kiên, Dự án Ex-C2-L6-2018-L1]%[6H2B1]
Cho ba điểm $M, N, P$ không thẳng hàng nằm ngoài đường thẳng $a$. Biết rằng đường thẳng $a$ cắt đoạn thẳng $MN$ nhưng không cắt đoạn $MP$.
\immini{\begin{enumEX}{1}
\item Gọi tên hai nửa mặt phẳng đối nhau bờ $a$.
\item Đoạn thẳng $NP$ có cắt đường thẳng $a$ không? Vì sao?
\end{enumEX}}{\begin{tikzpicture}

\begin{enumerate}
\item Hai nửa mặt phẳng đối nhau bờ $a$ là
\begin{itemize}
\item  Nửa mặt phẳng bờ $a$ chứa điểm $M$ và điểm $P$.
\item  Nửa mặt phẳng bờ $a$ chứa điểm $N$.
\end{itemize}
\item Đường thẳng $a$ cắt đoạn $MN$ nên $M, N$ thuộc hai nửa mặt phẳng đối nhau bờ $a$ \quad (1).\\
Đường thẳng $a$ không cắt đoạn $MP$ nên $M, P$ thuộc cùng một nửa mặt phẳng đối nhau bờ $a$ \quad (2).\\

\end{enumerate}
