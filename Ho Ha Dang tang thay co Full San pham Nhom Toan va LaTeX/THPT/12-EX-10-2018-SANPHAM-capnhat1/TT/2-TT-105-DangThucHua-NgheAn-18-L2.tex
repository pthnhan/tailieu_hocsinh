\begin{ex}%[TT-Đặng Thúc Hứa- Nghệ An lần 2 - 2018]%[PhanMinhTâm, Ex10]%[2D1B4-1]
	Đồ thị hàm số nào sau đây \textbf{không} có tiệm cận ngang?
	\choice
	{\True $y=\sqrt{x^2-1}$}
	{$y=\dfrac{2x-1}{x+1}$}
	{$y=\dfrac{x^2-3x+2}{x^2-x-2}$}
	{$y=x-\sqrt{x^2+1}$}
	\loigiai{
	\begin{itemize}
		\item Xét hàm số $y=\sqrt{x^2-1}$ với $ x \in \left(-\infty;-1\right] \cup \left[1;+\infty\right) $. Ta có:
			\begin{itemize}
			\item[+] $\lim\limits_{x\to +\infty}\sqrt{x^2-1}=\lim\limits_{x\to +\infty}x\sqrt{1-\dfrac{1}{x^2}}=+\infty $.
			\item[+] $\lim\limits_{x\to -\infty}\sqrt{x^2-1}=\lim\limits_{x\to +\infty} -x\sqrt{1-\dfrac{1}{x^2}}=+\infty.$
		\end{itemize}
Nên đồ thị hàm số $y=\sqrt{x^2-1}$ không có tiệm cận ngang.
\item Xét hàm số $y=\dfrac{2x-1}{x+1}$ có $\lim\limits_{x\to +\infty} \dfrac{2x-1}{x+1}=2.$ \\Nên đồ thị hàm số có một tiệm ngang là $ y = 2. $
\item Xét hàm số $y=\dfrac{x^2-3x+2}{x^2-x-2}$ có $\lim\limits_{x\to +\infty} \dfrac{x^2-3x+2}{x^2-x-2}=1.$ \\Nên đồ thị hàm số có một tiệm ngang là $ y = 1. $
\item Xét hàm số $y=x-\sqrt{x^2+1}$ có
$\lim\limits_{x\to \pm \infty} \left(x-\sqrt{x^2+1}\right)=\lim\limits_{x\to \pm \infty} \dfrac{-1}{x+\sqrt{x^2+1}}=0.$\\
Nên đồ thị hàm số có một tiệm cận ngang là $ y = 0. $

\end{itemize}
	}
\end{ex}
