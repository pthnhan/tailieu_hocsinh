	Cho biểu thức $$P=\dfrac{1+2\sqrt{x}}{(1+x)(1-\sqrt{x})}:\left(\dfrac{2x+3\sqrt{x}+1}{1-x}+\dfrac{2x\sqrt{x}+3x+\sqrt{x}}{1+x\sqrt{x}}\right)\ \text{với}\  x\geq 0; x\neq 1.$$
	\begin{enumerate}
	\item Rút gọn biểu thức $P.$
	\item Tìm $x$ để $P$ có giá trị nguyên.

	\end{enumerate}
	\loigiai{ \begin{enumerate}
	\item 
	Đặt $A=\left(\dfrac{2x+3\sqrt{x}+1}{1-x}+\dfrac{2x\sqrt{x}+3x+\sqrt{x}}{1+x\sqrt{x}}\right).$\\
	Ta có \begin{itemize}
	\item $\dfrac{2x+3\sqrt{x}+1}{1-x}=\dfrac{2x+2\sqrt{x}+\sqrt{x}+1}{1-x}=\dfrac{(\sqrt{x}+1)(2\sqrt{x}+1)}{(1-\sqrt{x})(1+\sqrt{x})}=\dfrac{2\sqrt{x}+1}{1-\sqrt{x}}.$
	\item $\dfrac{2x\sqrt{x}+3x+\sqrt{x}}{1+x\sqrt{x}}=\dfrac{(\sqrt{x}+1)(2x+\sqrt{x})}{(\sqrt{x}+1)(x-\sqrt{x}+1)}=\dfrac{2\sqrt{x}+1}{x-\sqrt{x}+1}.$
	\end{itemize}
	Suy ra $A=\dfrac{2\sqrt{x}+1}{1-\sqrt{x}}+\dfrac{2\sqrt{x}+1}{x-\sqrt{x}+1}=\dfrac{1}{(1-\sqrt{x})(x-\sqrt{x}+1)}.$\\
	Vậy $P=\dfrac{1+2\sqrt{x}}{(1+x)(1-\sqrt{x})}\cdot \dfrac{(1-\sqrt{x})(x-\sqrt{x}+1)}{1}=\dfrac{x-\sqrt{x}+1}{1+x}=1-\dfrac{\sqrt{x}}{1+x}.$

	 Ta có $0\leq \dfrac{\sqrt{x}}{1+x}\leq \dfrac{1}{2},$ suy ra $\dfrac{\sqrt{x}}{1+x}\in \mathbb{Z}\Leftrightarrow x=0.$\\
	 Vậy $P$ đạt giá trị nguyên khi $x=0.$
	\end{enumerate}

    \begin{enumerate}
        \item Giải phương trình $4\sqrt{x+3}+2\sqrt{2x+7}=(x+1)(x^2+4x+2).$
        \item Tìm các bộ số nguyên dương $(x;y;z)$ thỏa mãn $x^2+15^y=2^z .$
    \end{enumerate}
\loigiai
    { \begin{enumerate}
        \item Điều kiện $\heva{&x\ge -3\\&x \ge -\dfrac{7}{2}} \Leftrightarrow x\ge -3.$\\
        Ta có  $x=-3$ là một nghiệm của phương trình.\\
        Xét $x>-3$ khi đó phương trình đã cho tương đương với
        \begin{align*}
       & 4\sqrt{x+3}+2\sqrt{2x+7}=x^3+5x^2+6x+2\\
       \Leftrightarrow \;& x^3+5x^2+3x-9+2x+6-4\sqrt{x+3}+x+5-2\sqrt{2x+7}=0\\
       \Leftrightarrow \;& (x-1)(x+3)^2+\dfrac{2(x^2+2x-3)}{x+3+2\sqrt{x+3}}+\dfrac{x^2+2x-3}{x+5+2\sqrt{2x+7}}=0\\
       \Leftrightarrow \;& (x^2+2x-3)\left( x+3 + \dfrac{2}{x+3+2\sqrt{x+3}}+\dfrac{1}{x+5+2\sqrt{2x+7}} \right)=0\\
       \Leftrightarrow \;& \hoac{&x^2+2x-3 =0 \\& x+3 + \dfrac{2}{x+3+2\sqrt{x+3}}+\dfrac{1}{x+5+2\sqrt{2x+7}}=0\;(\text{vô nghiệm do\;} x>-3)}\\
       \Leftrightarrow \;& x=1\;(\text{do}\;x>-3).
       \end{align*}
       Từ hai trường hợp ta có nghiệm của phương trình là $x=1, x=-3.$
        \item Tìm các bộ số nguyên dương $(x;y;z)$ thỏa mãn $x^2+15^y=2^z .$
        \begin{itemize}
        \item Xét $z$ lẻ.\\
        Do $2\equiv -1\pmod 3 \Rightarrow 2^z\equiv -1 \pmod 3$\\
        Mà $x^2 \equiv 0,1 \pmod 3, 15^y\  \vdots \ 3\Rightarrow VT\equiv 0,1 \pmod 3$ mâu thuẫn.\\
        Vậy $z$ chẵn, đặt $z = 2k, (k\in \mathbb{N}^*)\Rightarrow x^2 + 15^y =4^k$. Suy ra $x$ lẻ.
        \item Xét $y$ chẵn.\\
        Khi đó $15^y \equiv 1 \pmod 4 \Rightarrow 15^y + x^2 \equiv 2 \pmod 4$ mâu thuẫn.\\
        Vậy $y$ lẻ.\\ Đặt $\heva{&y=2m+1\\&x=2t+1} (m,t\in \mathbb{Z}^+).$\\
       Ta có  \begin{align*}
       x^2+15^y=2^z &\Leftrightarrow (2t+1)^2 + 15^{2m+1}=4^k\\
       &\Leftrightarrow 4t(t+1) + (15^{2m+1}+1) =4^k\\
       &\Leftrightarrow 4t(t+1) + 16(15^{2m} - 15^{2m-1} + \cdots +1)=4^k.
\end{align*}        
		Do $15^{2m} - 15^{2m-1} + \cdots +1$ có $2m+1$ số nên $15^{2m} - 15^{2m-1} + \cdots +1$ lẻ.\\
		 Vậy $16(15^{2m} - 15^{2m-1} + \cdots +1) \vdots 16$ là ước chẵn lớn nhất.
		 \item Với $4t(t+1)\vdots 8$ là ước chẵn lớn nhất $\Rightarrow 4t(t+1) + 16(15^{2m} - 15^{2m-1} + \cdots +1)\vdots 8$ là ước chẵn lớn nhất.\\
		 Mâu thuẫn do với $k=1\Rightarrow 4^k\not \vdots 8$, còn với $k\geq 2\Rightarrow 4^k \vdots 16$.
		 \item Với $4t(t+1)\vdots 16$ (hoặc hơn $16$).\\
		 Mà $16(15^{2m} - 15^{2m-1} + \cdots +1) \vdots 16$ là ước chẵn lớn nhất.\\
		  Suy ra $4^k \vdots 16$ là ước chẵn lớn nhất $\Rightarrow 4^k =16 \Rightarrow x=y =1, z =4$.
		 
        \end{itemize}

    \end{enumerate}
    }

\begin{ex}%[Đề thi HSG 9, tỉnh Hà Nam, năm 2018]%[Trần Như Ngọc, Dự án EX-9G-18]%[9D0G7]
Cho $a,b,c$ là các số thực dương. Chứng minh rằng
$$ \left(\dfrac{a}{b}+\dfrac{b}{c}+\dfrac{c}{a}\right)^2 \geq \dfrac{3}{2}\left(\dfrac{a+b}{c}+\dfrac{b+c}{a}+\dfrac{c+a}{b}\right).$$
Dấu bằng xảy ra khi nào?
\loigiai
{ Đặt $ x=\dfrac{a}{b} >0, y=\dfrac{b}{c}>0, z=\dfrac{c}{a}>0 \Rightarrow xyz=1.$ Khi đó bất đẳng thức cần chứng minh trở thành 
\begin{align*}
&(x+y+z)^2\geq \dfrac{3}{2}\left(x+y+z+\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\right)\\
\Leftrightarrow & 2x^2+2y^2+2z^2+4xy+4yz+4zx\geq 3x+3y+3z+3xy+3yz+3zx\\
\Leftrightarrow &2(x^2+y^2+z^2)+(xy+yz+zx) -3(x+y+z)\geq 0.
\end{align*}
Đặt $t=x+y+z$ ta có $t\geq 3.$ 
Áp dụng các bất đẳng thức $$(xy+yz+xz)^2\geq 3(x+y+z)=3t\ \text{và}\  x^2+y^2+z^2\geq \dfrac{(x+y+z)^2}{3}=\dfrac{t^2}{3}.$$
Suy ra $$2(x^2+y^2+z^2)\geq \dfrac{2}{3}t^2 ,\  (xy+yz+xz)\geq \sqrt{3t}.$$
Do đó ta cần chứng minh $\dfrac{2}{3}t^2+\sqrt{3t}-3t\geq 0.$ Vì $t\geq 3$ nên bất đẳng thức này luôn đúng.
}
\end{ex}
