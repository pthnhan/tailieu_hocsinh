\begin{ex}%[Đề thi vào 10, Chuyên Lê Hồng Phong Nam Định (Vòng 1), 2016]%[9D1B2]%[9D1B2]%[9H3B2]%[9D2Y2]
\hfill
\begin{enumerate}
\item Tìm điều kiện xác định của biểu thức $A=\sqrt{x-1}+\dfrac{2}{3-x}$.
\item Tính giá trị biểu thức $B=\sqrt{x^2-6x+9}$ với $x=3-\sqrt{3}$.
\item Tính bán kính đường tròn ngoại tiếp hình vuông có cạnh bằng $5$ cm.
\item Tìm tọa độ các giao điểm của đường thẳng $y=-x+2$ với parabol $y=x^2$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Điều kiện xác định của biểu thức là $\heva{& x-1\geq 0\\& 3-x\neq 0} \Leftrightarrow \heva{& x\geq 1\\ & x\neq 3.}$
\item Với $x=3-\sqrt{3}$ thì $B=\sqrt{x^2-6x+9}=\sqrt{(x-3)^2}=|x-3|=|3-\sqrt{3}-3|=\sqrt{3}$.
\item Gọi $ABCD$ là hình vuông $AB=5$ cm. Đường tròn ngoại tiếp $ABCD$ có tâm là điểm $O=AC\cap BD$.
Do đó bán kính đường tròn ngoại tiếp tiếp $ABCD$ là $R=OA=\dfrac{AC}{2}$.\\
Mà $AC^2=AB^2+AD^2=5\sqrt{2}\Rightarrow R=\dfrac{5\sqrt{2}}{2}$.
\item Xét phương trình hoành độ giao điểm
$$-x+2=x^2\Leftrightarrow x^2+x-2=0\Leftrightarrow \hoac{& x=1\\ & x=-2.}$$
Vậy tọa độ các giao điểm là $A(1;1)$, $B(-2;4)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Lê Hồng Phong Nam Định (Vòng 1), 2016]%[9D1B8]
Cho biểu thức $P=\dfrac{3(x+2\sqrt{x})}{x+\sqrt{x}-2}-\dfrac{\sqrt{x}+2}{\sqrt{x}-1}-\dfrac{\sqrt{x}+1}{\sqrt{x}+2}$\qquad (Với $x\ge 0$; $x\neq 1$).
\begin{enumerate}
\item Chứng minh rằng $P=\dfrac{\sqrt{x}+3}{\sqrt{x}+2}$.
\item Chứng minh rằng nếu $x\geq 0$; $x\neq 1$ thì $P\leq \dfrac{3}{2}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Điều kiện: $x\ge 0$, $x\neq 1$.
\begin{eqnarray*}
P & = & \dfrac{3(x+2\sqrt{x})-(\sqrt{x}+2)^2-(\sqrt{x}+1)(\sqrt{x}-1)}{(\sqrt{x}-1)(\sqrt{x}+2)}
=\dfrac{3x+6\sqrt{x}-x-4\sqrt{x}-4-x+1}{(\sqrt{x}-1)(\sqrt{x}+2)}\\
& = & \dfrac{x+2\sqrt{x}-3}{(\sqrt{x}-1)(\sqrt{x}+2)}
=\dfrac{(\sqrt{x}-1)(\sqrt{x}+3)}{(\sqrt{x}-1)(\sqrt{x}+2)}=\dfrac{\sqrt{x}+3}{\sqrt{x}+2}.
\end{eqnarray*}
\item Do $\sqrt{x}+2>0$, $\forall x\ge 0$, $x\neq 1$ nên
$$P\leq \dfrac{3}{2}\Leftrightarrow \dfrac{\sqrt{x}+3}{\sqrt{x}+2}\leq \dfrac{3}{2}\Leftrightarrow 2(\sqrt{x}+3)\leq 3(\sqrt{x}+2)\Leftrightarrow \sqrt{x}\ge 0\quad \text{(luôn đúng)}.$$
Vậy ta có điều phải chứng minh.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Lê Hồng Phong Nam Định (Vòng 1), 2016]%[9D4K6]%[9D4G7]
\hfill
\begin{enumerate}[1)]
\item Cho phương trình $x^2-(m+1)x+2m-2=0$ (1) (Với $m$ là tham số).
\begin{enumerate}[a)]
\item Tìm $m$ để phương trình có hai nghiệm phân biệt $x_1,x_2$ thỏa mãn $x_1^2+x_2^2=4+x_1x_2$.
\item Tìm $m$ để phương trình có nghiệm lớn hơn $2$.
\end{enumerate}
\item Giải hệ phương trình $\heva{& 2x^2-y^2-xy+x-y=0\\ &\sqrt{2x+y-2}+2-2x=0.}$
\end{enumerate}
\loigiai{
\begin{enumerate}[1)]
\item $\Delta = m^2+3>0$ $\forall m\in\mathbb{R}$. Nên phương trình luôn có hai nghiệm phân biệt $x_1,x_2$.\\
Theo định lí Vi-ét ta có $\heva{& x_1+x_2=m+1\\ & x_1x_2=2m-2.}$
\begin{enumerate}[a)]
\item Theo giả thiết ta có
\begin{eqnarray*}
x_1^2+x_2^2=4+x_1x_2 & \Leftrightarrow & (x_1+x_2)^2-3x_1x_2-4=0\\
& \Leftrightarrow & m^2-4m+3=0\Leftrightarrow \hoac{& m=1\\ & m=3.}
\end{eqnarray*}

\item Xét hai trường hợp
\begin{itemize}
\item Chỉ có đúng một nghiệm lớn hơn 2. Nghĩa là
$$(x_1-2)(x_2-2)<0 \Leftrightarrow x_1x_2-2(x_1+x_2)+4<0 \Leftrightarrow 1<0\quad \text{(Vô lý)}.$$
\item Cả hai nghiệm đều lớn hơn 2. Nghĩa là
$$\heva{& (x_1-2)(x_2-2)>0\\ & x_1+x_2>4} \Leftrightarrow \heva{& 1>0\\ & m+1>4} \Leftrightarrow m>3.$$
\end{itemize}
\end{enumerate}
\item Điều kiện: $2x+y-2\geq 0$.\\
Biến đổi phương trình thứ nhất
\begin{eqnarray*}
&& 2x^2-y^2-xy+x-y=0\\
& \Leftrightarrow & (2x^2-2xy)+(xy-y^2)+(x-y)=0\\
& \Leftrightarrow & (x-y)(2x+y+1)=0\Leftrightarrow \hoac{& y=x\\ & y=-2x-1.}
\end{eqnarray*}
\begin{itemize}
\item Với $y=x$ phương trình thứ hai trở thành
$$\sqrt{3x-2}=2x-2 \Leftrightarrow \heva{& x\ge 1\\ & 4x^2-11x+6=0}\Leftrightarrow x=2\Rightarrow y=2.$$
\item Với $y=-2x-1$ thì $2x+y-2=-3<0$ (Không thỏa mãn điều kiên).
\end{itemize}
Vậy nghiệm hệ là $(x;y)=(2;2)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Lê Hồng Phong Nam Định (Vòng 1), 2016]%[8H1K9]%[9H3K7]%[9H1K1]
Cho hình chữ nhật $ABCD$, kẻ $AH\perp BD$ tại $H$ và $HE\perp AB$ tại $E$ và $HF\perp AD$ tại $F$. Gọi $K,M$ lần lượt là trung điểm của $HD$, $BC$ và $I$ là giao điểm của $AH$ với $EF$.
\begin{enumerate}
\item Chứng minh rằng $I$ là trực tâm của $\triangle ABK$.
\item Chứng minh rằng $ABMK$ là tứ giác nội tiếp.
\item Chứng minh rằng $AH^3=BE.BD.DF$.
\end{enumerate}
\loigiai{

\begin{enumerate}
\item Ta có $\widehat{EAE}=\widehat{AFH}=\widehat{AEH}=90^{\circ}$ nên $AFHE$ là hình chữ nhật.\\
$\Rightarrow$ $I$ là trung điểm $AH$, do đó $IK$ là đường trung bình của $\triangle AHD$. Tức là $IK\parallel AD$.\\
$\Rightarrow KI\perp AB$. Mà $AI\perp BK$ nên $I$ là trực tâm $\triangle ABK$.
\item Do $IK$ là đường trung bình của $\triangle AHD$ nên $IK=\dfrac{1}{2}AD$ $\Rightarrow IK=BM$ và $IK\parallel BM$.\\
$\Rightarrow KIBM$ là hình bình hành, nên $BI\parallel MK$.\\
Mà $I$ là trực tâm $\triangle ABK$ nên $BI\perp AK$.\\
$\Rightarrow MK\perp AK\Rightarrow \widehat{AKM}=90^{\circ}$, mà $\widehat{ABM}=90^{\circ}$ nên $ABMK$ nội tiếp đường tròn đường kính $AM$. 
\item Áp dụng các hệ thức lượng trong tam giác vuông ta được
\begin{eqnarray*}
BE.BD.DF.AH & = & (BD.AH).BE.DF\\
& = & (AB.AD).BE.DF = (AB.BE)(AD.DF)\\
& = & HB^2.HD^2=(HB.HD)^2=AH^4.
\end{eqnarray*}
$\Rightarrow BE.BD.DF=AH^3$.
\end{enumerate}
}

\begin{ex}%[Đề thi vào 10, Chuyên Lê Hồng Phong Nam Định (Vòng 1), 2016]%[9D5G1]
Cho $x,y,z>0$ và $xy+yz+zx=1$. Tìm giá trị nhỏ nhất của biểu thức
$$P=\dfrac{1}{4x^2-yz+2}+\dfrac{1}{4y^2-zx+2}+\dfrac{1}{4z^2-xy+2}.$$
\loigiai{
Vì $xy+yz+zx=1$ nên
$$4x^2-yz+2=4x^2-yz+2(xy+yz+zx)=4x^2+2xy+yz+2xz=(2x+y)(2x+z).$$
Tương tự cũng có
$$4y^2-zx+2=(2y+x)(2y+z);\qquad 4z^2-xy+2=(2z+x)(2z+y).$$
Do đó, theo BĐT Cô si ta có các biến đổi sau
\begin{eqnarray*}
P & = & \dfrac{1}{(2x+y)(2x+z)}+\dfrac{1}{(2y+x)(2y+z)}+\dfrac{1}{(2z+x)(2z+y)}\\
& = & \dfrac{yz}{(2xz+yz)(2xy+yz)}+\dfrac{xz}{(2yz+xz)(2xy+xz)}+\dfrac{xy}{(2yz+xy)(2xz+xy)}\\
& \ge & \dfrac{4yz}{(2xz+2xy+2yz)^2}+\dfrac{4xz}{(2yz+2xy+2xz)^2}+\dfrac{4xy}{(2yz+2xz+2xy)^2}\\
& = & \dfrac{4(yz+xz+xy)}{4}=1.
\end{eqnarray*}
Dấu "$=$" xảy ra khi $x=y=z=\dfrac{1}{\sqrt{3}}$.\\
Vậy giá trị nhỏ nhất của $P$ là $1$ khi $x=y=z=\dfrac{1}{\sqrt{3}}$.
}
\end{ex}
