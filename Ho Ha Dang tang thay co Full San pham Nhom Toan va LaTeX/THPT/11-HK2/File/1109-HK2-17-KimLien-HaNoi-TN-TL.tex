		Ba số có tổng bằng $\dfrac{15}{2}$ theo thứ tự là ba số hạng liên tiếp của một cấp số cộng. Nếu cộng thêm $8$ đơn vị vào số hạng thứ ba, ta được ba số hạng liên tiếp là một cấp số nhân. Tìm ba số đó.
		\loigiai{
		Theo đề bài ta có:
		\begin{center}
			 $\heva{&a+b+c=\dfrac{15}{2}\,\,\,\,(1)\\&a+c=2b\,\,\,\,(2)\\&b^2=a(c+8)\,\,\,\,(3)}$
		\end{center}
		Từ $(1)$ và $(2)$ suy ra $b=\dfrac{5}{2}$ và $a=5-c$.
		Thay vào $(3)$ ta được $a^2-13a+\dfrac{25}{4}=0$.\\
		Giải phương trình ta được $a=\dfrac{1}{2}$ hoặc $a=\dfrac{25}{2}$.\\
		Với $a=\dfrac{1}{2}\Rightarrow c=\dfrac{9}{2}$. Vậy ba số cần tìm là $a=\dfrac{1}{2}$, $b=\dfrac{5}{2}$, $c=\dfrac{9}{2}$.\\
		Với $a=\dfrac{25}{2}\Rightarrow c=-\dfrac{15}{2}$. Vậy ba số cần tìm là $a=\dfrac{25}{2}$, $b=\dfrac{5}{2}$, $c=-\dfrac{15}{2}$.
		}
	\end{bt}
\begin{bt}%[HK2 (2016-2017), THPT Kim Liên, Hà Nội]%[1D5B2-1]%[1D5B2-3]%[Võ Thanh Phong]
	\begin{enumerate}
		\item Cho hàm số $f(x)=\sqrt{x^2+x+7}$. Giải bất phương trình $f'(x)\ge \dfrac{1}{2}$.
		\item Cho hàm số $f(x)=x^3+3x+1$ có đồ thị $(C)$. Viết phương trình tiếp tuyến của $(C)$ biết tiếp tuyến song song với đường thẳng $6x-y-1=0$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item 	Ta có: $f'(x)=\dfrac{2x+1}{2\sqrt{x^2+x+7}}$. Do $x^2+x+7>0$ với mọi $x\in \mathbb{R}$ nên theo đề bài ta suy ra được\\
			$f'(x)\ge \dfrac{1}{2}\Rightarrow \dfrac{2x+1}{2\sqrt{x^2+x+7}}\ge \dfrac{1}{2}\Leftrightarrow \sqrt{x^2+x+7} \le 2x+1\Leftrightarrow \heva{&x\ge -\dfrac{1}{2}\\&x^2+x-2\ge 0 } \Leftrightarrow x\ge 1$.
			\item Ta có: $y'=3x^2+3$.\\
			Do tiếp tuyến song song với đường thẳng $6x-y-1=0$ nên $y'(x_0)=6\Leftrightarrow x_0=\pm 1$.\\
			Với $x_0=1$  thì $y_0=5$. Suy ra $y=6x-1$ (loại).\\
			Với $x_0=-1$  thì $y_0=-3$. Suy ra $y=6x+3$ (thỏa).\\
			Vậy phương trình tiếp tuyến cần tìm là $y=6x+3$.
		\end{enumerate}
		}
\end{bt}
