		&\log_2\left (x-\sqrt{x^2-1}\right )\cdot \log_3 \left (x+\sqrt{x^2-1}\right )=\log_6 \left |x-\sqrt{x^2-1}\right | \\
		\Leftrightarrow & \log_2\left (x-\sqrt{x^2-1}\right )\cdot \log_3 \left (\dfrac{1}{x-\sqrt{x^2-1}}\right )=\log_6 \left |x-\sqrt{x^2-1}\right | \\
		\Leftrightarrow &-\log_2\left (x-\sqrt{x^2-1}\right )\cdot \log_3 \left (x-\sqrt{x^2-1}\right )=\log_6 \left (x-\sqrt{x^2-1}\right ) \\
		\Leftrightarrow &-\log_26\cdot \log_6\left (x-\sqrt{x^2-1}\right )\cdot \log_3 \left (x-\sqrt{x^2-1}\right )=\log_6 \left (x-\sqrt{x^2-1}\right ) \\
		\Leftrightarrow &\log_6 \left (x-\sqrt{x^2-1}\right )=0 \vee \log_26\cdot \log_3 \left (x-\sqrt{x^2-1}\right )=1.
		\end{align*}
		\begin{itemize}
			\item $\log_6 \left (x-\sqrt{x^2-1}\right )=0\Leftrightarrow x-\sqrt{x^2-1}=1 \Leftrightarrow x=1$.
			\item $\log_26\cdot \log_3 \left (x-\sqrt{x^2-1}\right )=1\Leftrightarrow x-\sqrt{x^2-1}=3^{\log_6\frac{1}{2}}\Leftrightarrow x=\dfrac{1}{2}\left (3^{\log_62}+3^{-\log_62}\right )$.
		\end{itemize}		
	}
\end{ex}
