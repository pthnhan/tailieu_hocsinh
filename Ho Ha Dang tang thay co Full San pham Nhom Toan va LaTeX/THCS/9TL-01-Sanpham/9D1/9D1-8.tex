\subsection{Rút gọn biểu thức không chứa căn.}
\begin{dang}{Rút gọn biểu thức không chứa căn.}
    Chúng ta thực hiện bài toán rút gọn biểu thức không chứa căn thông qua những bước như sau:
    \begin{enumerate}
    \item[B1:] Tìm điều kiện xác định.
    \item[B2:] Đưa các mẫu về dạng tích.
    \item[B3:] Quy đồng mẫu.
    \item[B4:] Rút gọn biểu thức.
	\end{enumerate}
    \begin{note}
        Nếu trong biểu thức tồn tại dạng $A:B$ thì chú ý điều kiện $B\neq 0$.
    \end{note}
\end{dang}
\begin{vd}
    Cho biểu thức $P=\dfrac{1}{x+5}+\dfrac{2}{x-5}-\dfrac{2x+10}{(x+5)(x-5)}$.\\
    \begin{enumerate}
    	\item Tìm điều kiện xác định của $P$.
    	\item Rút gọn biểu thức $P$.
    \end{enumerate}
       \loigiai{
    \begin{enumerate}
    	\item   Điều kiện xác định\\
    	$\heva{&x+5\ne 0\\&x-5\ne 0\\&(x-5)(x+5)\ne 0}\Leftrightarrow \heva{&x\ne 5\\&x\ne -5}\Leftrightarrow x\ne \pm 5$.
    	\item Ta có $\begin{aligned}[t]
    	P&=\dfrac{1}{x+5}+\dfrac{2}{x-5}-\dfrac{2x+10}{(x+5)(x-5)}\\
    	&=\dfrac{x-5}{(x+5)(x-5)}+\dfrac{2(x+5)}{(x-5)(x+5)}-\dfrac{2x+10}{(x+5)(x-5)}\\
    	&=\dfrac{x-5+2(x+5)-(2x+10)}{(x+5)(x-5)}=\dfrac{x-5}{(x+5)(x-5)}=\dfrac{1}{x+5}.
    	\end{aligned}$

    \end{enumerate}   
    }
\end{vd}
\begin{vd}

		\item Rút gọn biểu thức $P$.

		\begin{enumerate}
			\item Biểu thức $P$ xác định khi
			$\heva{&x+3\ne 0\\&x-3\ne 0\\&9-x^2\ne 0}\Leftrightarrow x\ne \pm 3$.
			\item Ta có
			$\begin{aligned}[t]
			P&=\dfrac{3}{x+3}+\dfrac{1}{x-3}-\dfrac{18}{9-x^2}\\
			&=\dfrac{3(x-3)}{(x-3)(x+3)}+\dfrac{x+3}{(x-3)(x+3)}+\dfrac{18}{(x-3)(x+3)}\\
			&=\dfrac{3x-9+x+3+18}{(x-3)(x+3)}=\dfrac{4x+12}{(x-3)(x+3)}\\
			&=\dfrac{4(x+3)}{(x-3)(x+3)}=\dfrac{4}{x-3}.
			\end{aligned}$
		\end{enumerate}   
	}
\end{vd}

\begin{vd}

	\begin{enumerate}
		\item Tìm điều kiện xác định của $P$.
		\item Rút gọn biểu thức $P$.
		\item Tìm giá trị $x$ để $P>0$.
		\item Tìm giá trị nguyên của $x$ để $P$ là số nguyên.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Biểu thức $P$ xác định khi 
			$\heva{&x+3\ne 0\\&x-3\ne 0\\&9-x^2\ne 0}\Leftrightarrow x\ne \pm 3$.
			\item Ta có
			\begin{align*}
			P&=\dfrac{3}{x+3}+\dfrac{1}{x-3}-\dfrac{18}{9-x^2}\\
			&=\dfrac{3(x-3)}{(x-3)(x+3)}+\dfrac{x+3}{(x-3)(x+3)}+\dfrac{18}{(x-3)(x+3)}\\
			&=\dfrac{3x-9+x+3+18}{(x-3)(x+3)}\\&=\dfrac{4x+12}{(x-3)(x+3)}=\dfrac{4(x+3)}{(x-3)(x+3)}=\dfrac{4}{x-3}
			\end{align*}
			\item 
			$P>0\Leftrightarrow \dfrac{4}{x-3}>0\Leftrightarrow x-3>0\Leftrightarrow x>3$.
			\item Với $x\in\mathbb{Z}$ thì\\
			$P\in\mathbb{Z}\Leftrightarrow x-3 \in \text{Ư}(4) \Leftrightarrow x-3 \in\{\pm 1;\pm 2;\pm 4\}\Leftrightarrow x\in\{2;4;1;5;-1;7\}$.\\
		Vậy $x\in\{2;4;1;5;-1;7\}$.
		\end{enumerate}   
	}
\end{vd}

	\item Tìm $x$ để $\dfrac{P}{Q} <2$.
\end{enumerate}

		$Q=2+\dfrac{7-2x}{x-3}= \dfrac{1}{x-3}$.
		\item

	\end{enumerate}
}
\end{vd}

\begin{bt}
	Cho biểu thức $ P=\left(\dfrac{x+3}{x+2}-\dfrac{2-x}{3-x}-\dfrac{x-8}{x^{2}-x-6} \right):\left(1-\dfrac{x+6}{2x+4} \right)$.
\begin{enumerate}
	\item Rút gọn $ P $ và tìm điều kiện xác định của $ P $.
	\item Tính giá trị của $ P $ biết $ |2x-3|+1=x $.
	\item Tìm $ x $ nguyên để $ P\cdot(x- 1) $ có giá trị là số tự nhiên.
\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Ta có
		\begin{align*}
		P&=\left(\dfrac{x+3}{x+2}-\dfrac{2-x}{3-x}-\dfrac{x-8}{x^{2}-x-6} \right):\left(1-\dfrac{x+6}{2x+4} \right)\\
		&=\left(\dfrac{x+3}{x+2}-\dfrac{2-x}{3-x}-\dfrac{x-8}{(x+2)(x-3)} \right):\left(\dfrac{2x+4-x-6}{2x+4} \right)\\
		&=\left(\dfrac{x^{2}-9+4-x^{2}-x+8}{(x+2)(x-3)} \right):\left(\dfrac{x-2}{2(x+2)} \right)\\
		&=\left(\dfrac{3-x}{(x+2)(x-3)} \right)\cdot\left(\dfrac{2(x+2)}{x-2} \right)\\
		&=-\dfrac{2}{x-2}.
		\end{align*}
		Điều kiện xác định $ \heva{&x\neq \pm 2\\ &x\neq 3} $.
		\item $ |2x-3|+1=x\Leftrightarrow |2x-3|=x-1 $ (điều kiện $ x\geq 1 $)\\
		$ \Leftrightarrow \hoac{&2x-3=x-1\\ &2x-3=1-x}\Leftrightarrow\hoac{&x=2\ (\mbox{loại})\\ &x=\dfrac{4}{3}\ (\mbox{thỏa mãn}).} $\\
		Với $ x=\dfrac{4}{3} $ ta có $ P=3 $.
		\item $ P\cdot (x-1)=-\dfrac{2(x-1)}{x-2}=-2-\dfrac{2}{x-2} $.\\
		Để $ P\cdot (x-1) $ là số tự nhiên thì $ x-2\in \textrm{Ư}(2)=\{\pm 1;\pm 2\} $. Ta có bảng giá trị
		\begin{center}
			\begin{tabular}{|p{2cm}<{\centering}|p{2cm}<{\centering}|p{2cm}<{\centering}|p{2cm}<{\centering}|p{2cm}<{\centering}|}\hline
				$x-2$&$-2$&$-1$&$1$&$2$ \\ \hline
				$x$&$0$&$1$&$3$&$4$ \\ \hline
				Kết luận&loại&thỏa mãn&loại&loại\\ \hline
			\end{tabular}
		\end{center}
		Vậy với $x=1$ thì $ P\cdot(x- 1) $ có giá trị là số tự nhiên.
	\end{enumerate}
}
\end{bt}
\begin{bt}

	\begin{enumerate}

		\item Rút gọn biểu thức $A$.
		\item Tìm giá trị biểu thức $A$ tại $x=1$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Giá trị của biểu thức $A$ xác định
			$\Leftrightarrow \heva{&x^2-4\ne 0\\&x-2\ne 0\\&x+2\ne 0}\Leftrightarrow x\ne\pm2$.
			\item Ta có 
			\begin{align*}
			A&=\dfrac{x^2}{x^2-4}-\dfrac{x}{x-2}+\dfrac{2}{x+2}\\
			&=\dfrac{x^2}{(x-2)(x+2)}-\dfrac{x(x+2)}{(x-2)(x+2)}+\dfrac{2(x-2)}{(x-2)(x+2)}\\
			&=\dfrac{x^2-x^2-2x+2x-4}{(x-2)(x+2)}=\dfrac{-4}{(x-2)(x+2)}.
			\end{align*}
			\item Tại $x=1$ thì ta có $A=\dfrac{-4}{(1-2)(1+2)}=\dfrac{4}{3}$.
		\end{enumerate}    
	}
\end{bt}

			P&=\dfrac{x+2}{x+3}-\dfrac{5}{x^2+x-6}+\dfrac{1}{2-x}\\

			\end{align*}

		P=\dfrac{-3}{4}
		&\Leftrightarrow \dfrac{x-4}{x-2}=\dfrac{-3}{4}
		\Leftrightarrow 4(x-4)=-3(x-2)\Leftrightarrow 4x-16=-3x+6\\
		&\Leftrightarrow 7x=22\Leftrightarrow x=\dfrac{22}{7}.
		\end{aligned}$

		Hay $x\in\{\pm 1;\pm 2\}\Leftrightarrow x\in\{0;1;3;4\}$.

\begin{dang}{Tính giá trị của biểu thức khi biết $x$}
	Đối với dạng toán này sau khi rút gọn biểu thức ta mới thay giá trị cụ thể của $x$ vào.\\
	Trong nhiều bài tập $x$ được cho dưới dạng $x=a^2\pm 2a\sqrt{b}+b$ ta cần biến đổi $x=(a\pm \sqrt{b})^2.$
	Chú ý thêm một số công thức:
	\begin{enumerate}[i)]
		\item $x\sqrt{x}=(\sqrt{x})^3$
		\item $a^3-b^3=(a-b)(a^2+ab+b^2)$
	\end{enumerate}
	\begin{note}
		Giá trị cụ thể của $x$ phải thỏa mãn điều kiện xác định của biểu thức ban đầu (chứ không phải biểu thức đã rút gọn) thì ta mới thực hiện tính.
	\end{note}
\end{dang}

\subsection{Giải phương trình chứa căn}

\begin{dang}{Giải phương trình chứa căn}
	Để giải phương trình có căn bậc hai, căn bậc ba, ta thực hiện theo các bước:
	\begin{itemize}
		\item Đặt điều kiện phương trình có nghĩa ($\sqrt{A}$ có nghĩa $\Leftrightarrow A\geq 0$).
		\item Đưa thừa số ra ngoài dấu căn: $\sqrt{A^2B}=|A|\sqrt{B}$; $\sqrt[3]{A^3B}=A\sqrt[3]{B}$.
		\item Rút gọn các căn thức đồng dạng.
		\item Biến đổi phương trình về dạng: $\sqrt{A}=B \Leftrightarrow A=B^2$ (với $B\geq 0$) hoặc\\ $\sqrt[3]{A}=B \Leftrightarrow A=B^3$.
	\end{itemize}
\end{dang}

\begin{vd}%[Nguyễn Cao Cường]%[9D1B8]
	Giải các phương trình sau:
	\begin{enumerate}
		\begin{multicols}{2}
			\item $\sqrt{x^2+x}=x-1$;
			\item $\sqrt{x^2+3}=\sqrt{4x}$;
			\item $2\sqrt{3x}-4\sqrt{3x}=27-3\sqrt{3x}$;
			\item $\sqrt{x^2-3}=0$;
			\item $2-\sqrt{x^2-2}=0$;
			\item $x+\sqrt{3x+10}=0$.
		\end{multicols}
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $$\sqrt{x^2+x}=x-1 \Leftrightarrow \heva{&x-1\geq 0\\&x^2+x=(x-1)^2} \Leftrightarrow \heva{&x-1\geq 0\\&x^2+x=x^2-2x+1} \Leftrightarrow \heva{&x\geq 1\\&3x=1} \Leftrightarrow \heva{&x\geq 1\\&x=\dfrac{1}{3} \text{ loại.} }$$
			Vậy phương trình vô nghiệm.
			\item Ta có
			$$\sqrt{x^2+3}=\sqrt{4x} \Leftrightarrow \heva{&4x\geq 0\\&x^2+3=4x} \Leftrightarrow \heva{&x\geq 0\\&x^2-4x+3=0} \Leftrightarrow \heva{&x\geq 0\\&\hoac{&x=1\\&x=3}}\Leftrightarrow \hoac{&x=1\\&x=3.}$$

			Vậy phương trình có nghiệm $x=243$.

			\end{align*}
			Vậy phương trình có 2 nghiệm là $x=1$ hoặc $x=\sqrt{6}-1$.
		\end{enumerate}

\end{bt}

		\begin{enumerate}

			\Leftrightarrow &\; \sqrt{x-2}=3 \Leftrightarrow x-2=9 \Leftrightarrow x=11.\end{align*}

		\item $\sqrt[3]{x+2}-\sqrt{x+1}=1$.

			\item Ta có
