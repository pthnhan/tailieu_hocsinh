\section{Ôn tập chương}
\subsection{Trọng tâm kiến thức}
\begin{enumerate}
	\item Đoạn thẳng tỉ lệ, định lí Ta-lét, định lí đảo và hệ quả của định lí Ta-lét.
	\item Tính chất đường phân giác trong tam giác.
	\item Các trường hợp đồng dạng của hai tam giác.
	\begin{itemize}
		\item Cạnh - cạnh - cạnh.
		\item Cạnh - góc - cạnh.
		\item Góc - góc.
	\end{itemize}
	\item Các trường hợp đồng dạng của hai tam giác vuông.
	\item Ứng dụng thực tế của hai tam giác đồng dạng.
\end{enumerate}

\begin{vd}%[8H3K2]
	Cho tam giác $ ABC $ có $ BM $, $ CN $ là các đường trung tuyến $ (BM < CN) $, $ G $ là trọng tâm. Từ điểm $ D $ bất kỳ thuộc cạnh $ BC $, kẻ $ DE \parallel CN $, $ DF \parallel BM $ ($ E \in AB $; $ F \in AC $). Gọi $ I $, $ K $ là giao điểm của $ EF $ với $ BM $, $ CN $. Chứng minh $ EI = IK = KF $.
	\loigiai{
		\immini{
			Gọi $ P $ là giao điểm của $ BM $ và $ DE $; $ Q $ là giao điểm $ DF $ và $ CN $.\\
			Áp dụng hệ quả định lý Ta-lét ta có
			\begin{align*}
				\dfrac{EP}{NG} = \dfrac{BE}{BN} = \dfrac{ED}{NC}
				\Rightarrow \dfrac{EP}{ED} = \dfrac{NG}{NC} = \dfrac{1}{3}.
			\end{align*}
			Do đó $ \dfrac{EP}{ED} = \dfrac{1}{3} $.\\
			Mặt khác
			\begin{align*}
				\dfrac{EI}{EF} = \dfrac{EP}{ED} = \dfrac{1}{3} \Rightarrow EI = \dfrac{1}{3} EF.
			\end{align*}
			Tương tự, ta có $ FK = \dfrac{1}{3} EF $.\\
