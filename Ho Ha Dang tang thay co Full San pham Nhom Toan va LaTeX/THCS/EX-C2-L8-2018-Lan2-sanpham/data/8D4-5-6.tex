\section{Phương trình chứa dấu giá trị tuyệt đối}
\subsection{Trọng tâm kiến thức}
\subsubsection{Nhắc lại về giá trị tuyệt đối}

\begin{eqnarray*}
|a|& = a \quad &\text{khi} \, a \ge 0 \\
|a|& =  -a \quad &\text{khi} \, a < 0.
\end{eqnarray*}

\subsubsection{Giải phương trình chứa dấu giá trị tuyệt đối}
\textbf{Thực hiện các bước:}
\begin{itemize}
	\item Xét dấu của biểu thức trong dấu giá trị tuyệt đối để bỏ dấu giá trị tuyệt đối theo định nghĩa.
	\item Giải phương trình thu được sau khi bỏ dấu giá trị tuyệt đối.
	\item Nhận xét nghiệm có thỏa mãn điều kiện đang xét hay không.
\end{itemize}

\subsection{Các dạng bài tập và phương pháp giải}

\begin{dang}{Giải phương trình $|A(x)|=k$ với $k$ là hằng số ($k>0$)}
\[|A(x)| = k \Leftrightarrow \hoac{& A(x) = k & \qquad (1) \\ & A(x) = -k & \qquad(2)}\]
Giải (1) và (2) rồi lấy tất cả các nghiệm vừa tìm được.
\end{dang}

%%VD 01
\begin{vd}%[8D4B5]
Giải phương trình $|5x-1|=9$.
\loigiai{
\begin{itemize}
	\item Xét trường hợp $5x-1=9 \Leftrightarrow 5x = 10 \Leftrightarrow x = 2$.
	\item Xét trường hợp $5x - 1 = -9 \Leftrightarrow 5x = -8 \Leftrightarrow x = -\dfrac{8}{5}$.
\end{itemize}
Vậy tập nghiệm của phương trình đã cho là $\mathsf{S} = \left\{ -\dfrac{8}{5};2 \right\}$.
}
\end{vd}

%%VD 02
\begin{vd}%[8D4B5]
Giải phương trình $13 - |x+10| = 4$.
\loigiai{
\[ 13 - |x+10| = 4 \Leftrightarrow |x+10| = 9. \]
\begin{itemize}
	\item Xét trường hợp $x + 10 = 9 \Leftrightarrow x = -1$.
	\item Xét trường hợp $x + 10 = -9 \Leftrightarrow x = -19$.
\end{itemize}
Vậy tập hợp nghiệm của phương trình đã cho là $\mathsf{S} = \{-19;-1\}$.
}
\end{vd}

%%02
\begin{dang}{Giải phương trình $|A(x)| = |B(x)|$ \quad(*)}
	\[|A(x)| = |B(x)| \Leftrightarrow \hoac{& A(x) = B(x) & \qquad(1) \\ & A(x) = - B(x) & \qquad(2)} \]
	Tập nghiệm của (*) là hợp tất cả các nghiệm của (1) và (2).
\end{dang}

%%01
\begin{vd}%[8D4B5]
	Giải phương trình $\left| x^2 - 8x \right| = \left| x^2 + 8 \right|$.
\loigiai{
\begin{itemize}
		\item Trường hợp 1. $x^2 - 8x = x^2 + 8 \Leftrightarrow	-8x = 8 \Leftrightarrow x = -1$.
		\item Trường hợp 2. $x^2 - 8x = -x^2 - 8 \Leftrightarrow 2x^2 - 8x + 8 = 0\Leftrightarrow 2(x-2)^2 = 0 \Leftrightarrow x = 2$.
\end{itemize}
	Vậy tập nghiệm của phương trình đã cho là $S = \{ -1;2 \}$.
}
\end{vd}

%%02
\begin{vd}%[8D4B5]
	Giải phương trình $|4-x| = |2x+3|$.
	\loigiai{
\begin{itemize}
	\item Trường hợp 1. $4 - x = 2x + 3 \Leftrightarrow -x-2x = 3 - 4 \Leftrightarrow -3x = -1 \Leftrightarrow x = \dfrac{2}{3}$.
	\item Trường hợp 2.	$4-x = -(2x+3) \Leftrightarrow	4-x = -2x-3 \Leftrightarrow -x+2x = -3-4 \Leftrightarrow x = -7$.
\end{itemize}
Vậy tập nghiệm của phương trình đã cho là $S = \left\{ \dfrac{2}{3}; -7 \right\}$.
}
\end{vd}

%%03
\begin{dang}{Giải phương trình $|A(x)| = B(x)$ \quad(*)}
	\begin{itemize}
		\item Xét trường hợp $A(x) \ge 0$, khi đó (*) trở thành $A(x) = B(x)$ \hfill(1). \\
		Giải (1) rồi đối chiếu với khoảng đang xét để chọn nghiệm thích hợp.
		\item Xét trường hợp $A(x) < 0$, khi đó (*) trở thành $-A(x) = B(x)$ \hfill(2). \\
		Giải (2) rồi đối chiếu với khoảng đang xét để chọn nghiệm thích hợp.
		\item Kết luận: Nghiệm của (*) là tất cả các nghiệm thích hợp tìm được trong hai trường hợp trên.
	\end{itemize}
\end{dang}

%%01
\begin{vd}%[8D4B5]
	Giải phương trình $|2x+5| = x+6$ \quad(*).
	\loigiai{
\begin{itemize}
	\item Nếu $2x+5 \ge 0 \Leftrightarrow x \ge -\dfrac{5}{2}$ thì (*) trở thành $2x+5 = x+6 \Leftrightarrow x = 1$ (nhận).
	\item Nếu $2x+5 < 0 \Leftrightarrow x < -\dfrac{5}{2}$ thì (*) trở thành $-(2x+5) = x+6 \Leftrightarrow -2x-5 = x+6 \Leftrightarrow x = - \dfrac{11}{3}$ (nhận).
\end{itemize}
Vậy tập nghiệm của (*) là $S = \left\{ -\dfrac{11}{3};1 \right\}$.
}
\end{vd}

%%02
\begin{vd}%[8D4B5]
	Giải phương trình $|x-4| = 3x-2$ \quad(*).
	\loigiai{
\begin{itemize}
	\item Nếu $x-4 \ge 0 \Leftrightarrow x \ge 4$ thì (*) trở thành $x-4 = 3x-2 \Leftrightarrow -2x = 2 \Leftrightarrow x = -1$ (loại).
	\item Nếu $x-4 < 0 \Leftrightarrow x < 4$ thì (*) trở thành  $-(x-4) = 3x-2 \Leftrightarrow -4x = -6 \Leftrightarrow x = \dfrac{3}{2}$ (nhận).
\end{itemize}
Vậy tập nghiệm của (*) là $S = \left\{ \dfrac{3}{2} \right\}$.
}
\end{vd}

%%03
\begin{vd}%[8D4B5]
	Giải phương trình $|5x-2| = 2x-3$ \quad (*).
	\loigiai{
\begin{itemize}
	\item Nếu $5x-2 \ge 0 \Leftrightarrow x \ge \dfrac{2}{5}$ thì (*)  trở thành $5x-2 = 2x-3 \Leftrightarrow 3x = -1 \Leftrightarrow x = -\dfrac{1}{3}$ (loại).
	\item Nếu $5x - 2 < 0 \Leftrightarrow x < \dfrac{2}{5}$ thì (*) trở thành $-(5x-2) = 2x-3 \Leftrightarrow -7x = -5 \Leftrightarrow x = \dfrac{5}{7}$ (loại).
\end{itemize}
Vậy tập nghiệm của (*) là $S = \varnothing$.
}
\end{vd}

%%04
\begin{vd}%[8D4B5]
	Giải phương trình $|4x-1| = 4x-1$.
	\loigiai{
Ta có $|4x-1| = 4x-1 \Leftrightarrow 4x-1 \ge 0 \Leftrightarrow 4x \ge 1 \Leftrightarrow x \ge \dfrac{1}{4}$.\\
Vậy tập nghiệm của phương trình đã cho là $S = \left\{ x\left| x \ge \dfrac{1}{4} \right. \right\}$.
}
\end{vd}

%%05
\begin{vd}%[8D4B5]
	Giải phương trình $|x^2 - 3x| = 3x - x^2$.
	\loigiai{
Ta có $|x^2 - 3x| = 3x - x^2 \Leftrightarrow x^2 - 3x \le 0 \Leftrightarrow x(x-3) \le 0 \Leftrightarrow \heva{& x \ge 0 \\ & x-3 \le 0} \Leftrightarrow \heva{& x \ge 0 \\ & x \le 3} \Leftrightarrow 0 \le x \le 3$ (vì $x > x - 3$). \\
Vậy tập nghiệm của phương trình đã cho là $S = \left\{ x \left| 0 \le x \le 3 \right. \right\}$.
}
\end{vd}

%%04
\begin{dang}{Giải phương trình $|A(x)| + |B(x)| = C(x)$ \quad(*)}
	\begin{itemize}
		\item Xét từng khoảng thích hợp của $x$ để bỏ cả hai dấu giá trị tuyệt đối.
		\item Giải phương trình ứng với mỗi  khoảng của $x$ để chọn nghiệm thích hợp.
		\item Kết luận: Tập nghiệm của phương trìn đã cho là tất cả các nghiệm thích hợp tìm được ở trên.
	\end{itemize}
\end{dang}

%%01
\begin{vd}%[8D4B5]
	Giải phương trình $5|x+1| - |x-3| = x+12$ biết $-1 \le x \le 3$.
	\loigiai{
Với $x \ge -1$ thì $x + 1 \ge 0$, do đó $|x+1| = x+1$. \\
Với $x \le 3$ thì $x-3 \le 0$, do đó $|x-3| = -(x-3)$. \\
Phương trình đã cho trở thành
\begin{eqnarray*}
	5(x+1) + (x-3) &=& x+12 \\
	5x + 5 + x - 3 - x &=& 12 \\
	5x &=& 10 \\
	x &=& 2 \, \text{(thõa mãn điều kiện)}.
\end{eqnarray*}
}
\end{vd}

%%02
\begin{vd}%[8D4B5]
	Giải phương trình $3|x-1| + 2|x-4| = x+17$.
	\loigiai{
\begin{itemize}
	\item Nếu $x < 1$ thì $x - 1 < 0$ và $x - 4 < 0$, khi đó phương trình đã cho trở thành
	\begin{eqnarray*}
		3(1-x) + 2(4-x) &=& x+17 \\
		3 - 3x + 8 - 2x - x &=& 17 \\
		-6x &=& 6 \\
		x &=& -1 \, \text{(thuộc khoảng đang xét)}.
	\end{eqnarray*}
	\item Nếu $1 \le x \le 4$ thì $x-1 \ge 0$ và $x-4 \le 0$, khi đó phương trình đã cho trở thành
	\begin{eqnarray*}
		3(x-1) +2(4-x) &=& x+17 \\
		3x - 3 + 8 - 2x - x &=& 17 \\
		0x &=& 12 \, \text{(vô nghiệm)}.
	\end{eqnarray*}
	\item Nếu $x>4$ thì $x-1 > 0$ và $x-4 > 0$, khi đó phương trình đã cho trở thành
	\begin{eqnarray*}
		3(x-1) + 2(x-4) &=& x+17 \\
		3x - 3 + 2x -8 &=& x+17 \\
		4x &=& 28 \\
		x &=& 7 \, \text{(thuộc khoảng đang xét)}.
	\end{eqnarray*}
\end{itemize}
Vậy tập nghiệm của phương trình đã cho là $S = \{ -1;7 \}$.
}
\end{vd}

%%01
\begin{bt}%[8D4B5]
	Giải các phương trình.
	\begin{enumEX}{2}
		\item $|5x-7| + 8 = 11$.
		\item $\dfrac{1}{2} |x - 9| - 5 = -6$.
	\end{enumEX}
	\loigiai{
\begin{enumerate}
	\item Ta có $|5x - 7| + 8 = 11 \Leftrightarrow |5x - 7| = 3$.
		\begin{itemize}
			\item Xét trường hợp $5x - 7 = 3 \Leftrightarrow 5x = 10 \Leftrightarrow x = 2$.
			\item Xét trường hợp $5x - 7 = - 3 \Leftrightarrow 5x = 4 \Leftrightarrow x = \dfrac{4}{5}$.
		\end{itemize}
	Vậy tập nghiệm của phương trình đã cho là $S = \left\{ 2; \dfrac{4}{5} \right\}$.
	\item Ta có $\dfrac{1}{2} |x - 9| - 5 = -6 \Leftrightarrow \dfrac{1}{2} |x - 9| = -1 \Leftrightarrow |x - 9| = -2$ (vô nghiệm).
\end{enumerate}
}
\end{bt}

%%02
\begin{bt}%[8D4B5]
	Giải các phương trình.
	\begin{enumEX}{2}
		\item $|2x+1| = |x+11|$.
		\item $|3x-1| = |x-3|$.
	\end{enumEX}
	\loigiai{
\begin{enumerate}
	\item  Ta có $|2x+1| = |x+11|$.
		\begin{itemize}
			\item Trường hợp 1. $2x + 1 = x + 11 \Leftrightarrow 2x - x = 11 - 1 \Leftrightarrow x = 10$.
			\item Trường hợp 2. $2x + 1 = - (x + 11) \Leftrightarrow 2x + 1 = - x - 11 \Leftrightarrow 3x = - 12 \Leftrightarrow x = -4$.
		\end{itemize}
	Vậy tập nghiệm của phương trình đã cho là $S = \{ 10; -4 \}$.
	\item Ta có $3x-1| = |x-3|$.
		\begin{itemize}
			\item Trường hợp 1. $3x-1 = x-3 \Leftrightarrow 3x - x = -3 + 1 \Leftrightarrow 2x = -2 \Leftrightarrow x = -1$.
			\item Trường hợp 2. $3x-1 = - (x-3) \Leftrightarrow 3x - 1 = - x + 3 \Leftrightarrow 4x = 4 \Leftrightarrow x = 1$.
		\end{itemize}
	Vậy tập nghiệm của phương trình đã cho là $S = \{ -1; 1 \}$.
\end{enumerate}
}
\end{bt}

%%03
\begin{bt}%[8D4B5]
	Giải các phương trình.
	\begin{enumEX}{2}
		\item $|7x+3| = -x + 5$.
		\item $|3x-1| + 2 = x$.
	\end{enumEX}
	\loigiai{
\begin{enumerate}
	\item Ta có $|7x+3| = -x + 5$ \quad(1).
		\begin{itemize}
			\item Nếu $7x + 3 \ge 0 \Leftrightarrow x \ge  - \dfrac{3}{7}$ thì (1) trở thành $7x + 3 = -x + 5 \Leftrightarrow 8x = 2 \Leftrightarrow x = \dfrac{1}{4}$ (nhận).
			\item Nếu $7x + 3 < 0 \Leftrightarrow x <  - \dfrac{3}{7}$ thì (1) trở thành $- (7x + 3) = -x + 5 \Leftrightarrow 6x = - 8 \Leftrightarrow x = - \dfrac{4}{3}$ (nhận).
		\end{itemize}
	Vậy tập nghiệm của (1) là $S = \left\{ \dfrac{1}{4}; - \dfrac{4}{3} \right\}$.
	\item Ta có $|3x-1| + 2 = x \Leftrightarrow |3x - 1| = x - 2$ \quad(2).
	\begin{itemize}
		\item Nếu $3x-1 \ge 0 \Leftrightarrow x \ge \dfrac{1}{3}$ thì (2) trở thành $3x - 1 = x - 2 \Leftrightarrow 2x = - 1 \Leftrightarrow x = - \dfrac{1}{2}$ (loại).
		\item Nếu $3x-1 < 0 \Leftrightarrow x < \dfrac{1}{3}$ thì (2) trở thành $- (3x - 1) = x - 2 \Leftrightarrow 4x = 3 \Leftrightarrow x = \dfrac{3}{4}$ (loại).
	\end{itemize}
		Vậy tập nghiệm của (2) là $S = \varnothing$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[8D4B5]
	Giải các phương trình.
	\begin{enumEX}{2}
		\item $|x-2| + |x-3| = 0$.
		\item $|x-2| + |x-3| = 15$.
	\end{enumEX}
	\loigiai{
\begin{enumerate}
	\item Ta có $|x-2| + |x-3| = 0$.
	\begin{itemize}
		\item Nếu $x < 2$ thì $x - 2 < 0$ và $x - 3 < 0$, khi đó phương trình đã cho trở thành
		\[ (2 - x) + (3 - x) = 0 \Leftrightarrow 2x = 5 \Leftrightarrow x = \dfrac{5}{2} \, \text{(loại)}. \]
		\item Nếu $2 \ge x \ge 3$ thì $x - 2 \ge 0$ và $x - 3 \ge 0$, khi đó phương trình đã cho trở thành
		\[ (x - 2) + (3 - x) = 0 \Leftrightarrow 0x = -1 \, \text{(vô nghiệm)}. \]
		\item Nếu $x > 3$ thì $x - 2 > 0$ và $x - 3 > 0$, khi đó phương trình đã cho trở thành
		\[ (x - 2) + (x - 3) = 0 \Leftrightarrow 2x = 5 \Leftrightarrow x = \dfrac{5}{2} \, \text{(loại)}. \]
	\end{itemize}
	Vậy tập nghiệm của phương trình đã cho là $S = \varnothing$.
	\item Ta có $|x-2| + |x-3| = 15$.
	\begin{itemize}
		\item Nếu $x < 2$ thì $x - 2 < 0$ và $x - 3 < 0$, khi đó phương trình đã cho trở thành
		\[ (2 - x) + (3 - x) = 15 \Leftrightarrow 2x = -10 \Leftrightarrow x = -5 \, \text{(nhận)}. \]
		\item Nếu $2 \ge x \ge 3$ thì $x - 2 \ge 0$ và $x - 3 \ge 0$, khi đó phương trình đã cho trở thành
		\[ (x - 2) + (3 - x) = 15 \Leftrightarrow 0x = 14 \, \text{(vô nghiệm)}. \]
		\item Nếu $x > 3$ thì $x - 2 > 0$ và $x - 3 > 0$, khi đó phương trình đã cho trở thành
		\[ (x - 2) + (x - 3) = 15 \Leftrightarrow 2x = 20 \Leftrightarrow x = 10 \, \text{(nhận)}. \]
	\end{itemize}
	Vậy tập nghiệm của phương trình đã cho là $S = \{ -5; 10 \}$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[8D4K5]
	Chứng minh rằng phương trình
	\[ \left| x^2 + mx + \dfrac{m^2}{4} \right| = x^2 + mx + \dfrac{m^2}{4} \]
	có nghiệm bất kì và không phụ thuộc vào giá trị của $m$.

Ta có $\left| x^2 + mx + \dfrac{m^2}{4} \right| = x^2 + mx + \dfrac{m^2}{4} \Leftrightarrow x^2 + mx + \dfrac{m^2}{4} \ge 0 \Leftrightarrow \left( x + \dfrac{m}{2} \right)^2 \ge 0$.\\

\section{Ôn tập chương IV}
\subsection{Trọng tâm kiến thức}

\begin{itemize}
	\item Bất đẳng thức và liên hệ giữa thứ tự và phép tính.
	\item Bất phương trình và hai quy tắc biến đổi tương đương bất phương trình.
	\item Giải bất phương trình.
	\item Giải phương trình chưa dấu giá trị tuyệt đối.
\end{itemize}

\subsection{Các dạng bài tập và phương pháp giải}

%%01
\begin{dang}{Chứng minh bất đẳng thức}
	Có thể xét hiệu hai vế, có thể biến đổi tương đương hoặc dùng các tính chất của bất đẳng thức.
\end{dang}

%%01
\begin{vd}%[9D5G1]
	Chứng minh bất đẳng thức $\dfrac{1}{x} + \dfrac{1}{y} \ge \dfrac{4}{x+y}$ với $x$, $y > 0$.
	\loigiai{
Xét hiệu
\begin{eqnarray*}
	\dfrac{1}{x} + \dfrac{1}{y} - \dfrac{4}{x+y} =& \dfrac{y(x+y) + x(x+y) - 4xy}{xy(x+y)} &= \dfrac{xy + y^2 + x^2 + xy - 4xy}{xy(x+y)} \\
	=& \dfrac{x^2 - 2xy + y^2}{xy(x+y)} &= \dfrac{(x-y)^2}{xy(x+y)}.
\end{eqnarray*}
Ta có $(x-y)^2 \ge 0$ (dấu \lq\lq =\rq\rq{} xảy ra khi và chỉ khi $x=y$) và $xy(x+y) > 0$ (vì $x$, $y > 0$).\\
Do đó $\dfrac{1}{x} + \dfrac{1}{y} - \dfrac{4}{x+y} \ge 0$.\\
Vậy $\dfrac{1}{x} + \dfrac{1}{y} \ge \dfrac{4}{x+y}$ (dấu \lq\lq =\rq\rq{} xảy ra khi và chỉ khi $x=y$).
}
\end{vd}

%%02
\begin{vd}%[9D5G1]
	Chứng minh bất đẳng thức $3(a^2 + b^2 + c^2) \ge (a + b + c)^2$.
	\loigiai{
Ta có
\begin{eqnarray*}
	& & 3(a^2 + b^2 + c^2) \ge (a + b + c)^2 \\
	&\Leftrightarrow & 3(a^2 + b^2 + c^2) - (a + b + c)^2 \ge 0 \\
	&\Leftrightarrow & 3a^2 + 3b^2 + 3c^2 - a^2 - b^2 - c^2 - 2ab - 2bc - 2ca \ge 0 \\
	&\Leftrightarrow & 2a^2 + 2b^2 + 2c^2 - 2ab - 2bc - 2ca \ge 0 \\
	&\Leftrightarrow & (a-b)^2 + (b-c)^2 + (c-a)^2 \ge 0.
\end{eqnarray*}
Bất đẳng thức cuối cùng đúng nên bất đẳng thức đã cho đúng (dấu \lq\lq =\rq\rq{} xảy ra khi và chỉ khi $a = b = c$).
}
\end{vd}

%%03
\begin{vd}
	Chứng minh bất đẳng thức $\dfrac{x^2 - x + 1}{x^2 + x + 1} \ge \dfrac{1}{3}$.
	\loigiai{
Ta có
\begin{eqnarray*}
	\dfrac{x^2 - x + 1}{x^2 + x + 1} \ge \dfrac{1}{3}
	&\Leftrightarrow & \dfrac{x^2 - x + 1}{x^2 + x + 1} - \dfrac{1}{3} \ge 0
	\Leftrightarrow \dfrac{3x^2 - 3x + 3 - x^2 - x - 1}{3(x^2 + x + 1)} \ge 0 \\
	&\Leftrightarrow & \dfrac{2(x^2 - 2x + 1)}{3(x^2 + x + 1)} \ge 0 \Leftrightarrow \dfrac{2(x-1)^2}{3(x^2 + x + 1)} \ge 0.
\end{eqnarray*}
Bất đẳng thức cuối cùng đúng nên bất đẳng thức đã cho là đúng (dấu \lq\lq =\rq\rq{} xảy ra khi và chỉ khi $x=1$).
}
\end{vd}

\begin{vd}%[9D5G1]
	Chứng minh rằng $(ax + by)^2 \le (a^2 + b^2)(x^2 + y^2)$. Áp dụng: Cho $3x + 4y = 5$, chứng minh rằng $x^2 + y^2 \ge 1$.
	\loigiai{
Ta có
\begin{eqnarray*}
	& &(ax + by)^2 \le (a^2 + b^2)(x^2 + y^2) \\
	&\Leftrightarrow & (ax + by)^2 - (a^2 + b^2)(x^2 + y^2) \le 0 \\
	&\Leftrightarrow & a^2x^2 + 2abxy + b^2y^2 - a^2x^2 - a^2y^2 - b^2x^2 - b^2y^2 \le 0 \\
	&\Leftrightarrow & -(a^2y^2 - 2abxy + b^2x^2) \le 0 \\
	&\Leftrightarrow & (ay - bx)^2 \ge 0.
\end{eqnarray*}
Bất đẳng thức cuối cùng đúng nên bất đẳng thức đã cho là đúng (dấu \lq\lq =\rq\rq{} xảy ra khi và chỉ khi $ay = bx$).\\
Áp dụng $(ax + 4y)^2 \le (3^2 + 4^2)(x^2 + y^2) \Leftrightarrow 5^2 \le 25(x^2 + y^2)$.\\
Do đó $x^2 + y^2 \ge 1$ (dấu \lq\lq =\rq\rq{} xảy ra khi và chỉ khi $\dfrac{x}{3} = \dfrac{y}{5}$).
}
\end{vd}

\subsection{Bài tập tự luyện}

%%01
\begin{bt}%[9D5G1]
	Cho biết trong các khẳng định sau, khẳng định nào đúng, khẳng định nào \textbf{sai}?
	\begin{enumerate}
		\item\TF{Đ} $x > 3$ và $x > 5$ viết gộp lại thành $x > 5$.
		\item\TF{Đ} $x < 2$ và $x < 7$ viết gộp lại thành $x < 2$.
		\item\TF{Đ} $x > 2$ và $x < 11$ viết gộp lại thành $2 < x < 11$.
		\item\TF{S} $x > 2$ hoặc $x < -1$ viết gộp lại thành $2 < x < -1$.
	\end{enumerate}
\end{bt}

%%02
\begin{bt}%[9D5G1]
	Hai bất phương trình sau có bao nhiêu nghiệm nguyên chung.
	\[ \dfrac{1}{3}x - 2 \le 1 \quad \text{và} \quad 3(5 - x) < - 4{,}5. \]
	\loigiai{
\begin{itemize}
	\item Ta có $\dfrac{1}{3}x - 2 \le 1 \Leftrightarrow \dfrac{1}{3}x \le 3 \Leftrightarrow x \le 9$.
	\item Ta có $3(5-x) < -4{,}5 \Leftrightarrow 5 - x \le - \dfrac{3}{2}x \Leftrightarrow - x \le - \dfrac{13}{2} \Leftrightarrow x \ge \dfrac{13}{2}$.
\end{itemize}
Vậy nghiệm nguyên chung của hai bất phương trình thỏa mãn $\dfrac{13}{2} \le x \le 9 \Leftrightarrow 6{,}5 \le x \le 9$. \\
Kết luận $x \in \{ 7;8;9 \}$.
}
\end{bt}

%%03
\begin{bt}%[9D5G1]
	Tìm giá trị của $m$ để phương trình sau có nghiệm $x > -1$
	\[ \dfrac{x + m}{3} + \dfrac{x + 1}{2} = 2. \]
	\loigiai{
Ta có $\dfrac{x + m}{3} + \dfrac{x + 1}{2} = 2 \Leftrightarrow 2x + 2m + 3x + 3 = 12 \Leftrightarrow 5x = -2m + 9 \Leftrightarrow x = \dfrac{9 - 2m}{5}$. \\
Theo yêu cầu bài toán $x > -1 \Leftrightarrow \dfrac{9 - 2m}{5} > -1 \Leftrightarrow 9 - 2m > -5 \Leftrightarrow -2m > -14 \Leftrightarrow m < 7$.
}
\end{bt}

%%04
\begin{bt}%[9D5G1]
	Giải bất phương trình $\dfrac{3x + 5}{x^2 + 1} \ge 1$.
	\loigiai{
Ta có $\dfrac{3x + 5}{x^2 + 1} \ge 1 \Leftrightarrow \dfrac{3x + 5}{x^2 + 1} - 1 \ge 0 \Leftrightarrow \dfrac{-x^2 + 3x + 4}{x^2 + 1} \ge 0$.\\
Vì $x^2 + 1 > 0$ với mọi $x$ nên
\[ \dfrac{-x^2 + 3x + 4}{x^2 + 1} \ge 0 \Leftrightarrow (-x^2 + 3x + 4) \ge 0 \Leftrightarrow (x+1)(x-4) \le 0. \]
\begin{itemize}
	\item Trường hợp 1. $\heva{& x + 1 \ge 0 \\ & x - 4 \le 0} \Leftrightarrow \heva{& x \ge -1 \\ & x \le 4} \Leftrightarrow -1 \le x \le 4$.
	\item Trường hợp 2. $\heva{& x + 1 \le 0 \\ & x - 4 \ge 0} \Leftrightarrow \heva{& x \le -1 \\ & x \ge 4} \, \text{(vô nghiệm)}$.
\end{itemize}
Vậy nghiệm của bất phương trình là $ -1 \le x \le 4$.
}
\end{bt}

%%05
\begin{bt}%[9D5G1]
	Giải phương trình $|x - 2| + 8 = 4x$.
	\loigiai{
Ta có $|x - 2| + 8 = 4x \Leftrightarrow |x - 2| = 4x - 8$.
\begin{itemize}
	\item Nếu $x - 2 \ge 0 \Leftrightarrow x \ge 2$ thì phương trình đã cho trở thành
	\[ x - 2 - 4x + 8 = 0 \Leftrightarrow 3x = 6 \Leftrightarrow x = 2 \, \text{(thuộc khoảng đang xét)}. \]
	\item Nếu $x - 2 < 0 \Leftrightarrow x < 2$ thì phương trình đã cho trở thành
	\[ - (x - 2) - 4x + 8 = 0 \Leftrightarrow - x + 2 - 4x + 8 = 0 \Leftrightarrow 5x = 10 \Leftrightarrow x = 2 \, \text{(không thuộc khoảng đang xét)}. \]
\end{itemize}
Vậy phương trình đã cho có nghiệm là $x = 2$.
}
\end{bt}

\begin{bt}%[9D5G1]

Ta có  $ F = \dfrac{4x - 1}{x^2 + 3} = \dfrac{x^2 + 3 - (x^2 - 4x + 4}{x^2 + 3} = 1 - \dfrac{(x - 2)^2}{x^2 + 3} \le 1$. \\
