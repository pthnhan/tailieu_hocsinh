\section{ÔN TẬP CHƯƠNG II}
\subsection{Tóm tắt lí thuyết}
\begin{enumerate}[\tickEX]
	\item  Nếu đại lượng $y$ liên hệ với đại lượng $x$ theo công thức $y=kx$, với $k$ là hằng số khác $0$, thì ta nói $y$ tỉ lệ thuận với $x$ theo hệ số tỉ lệ $k$.\\
 Cho $y$ tỉ lệ thuận với $x$ theo hệ số tỉ lệ $k$, tức là $y=kx$ hay $x=\dfrac{1}{k}y$. Khi đó
 \begin{itemize}
  \item $x$ cũng tỉ lệ thuận với $y$, do vậy ta thường nói hai đại lượng này tỉ lệ thuận với nhau.
  \item $x$ tỉ lệ thuận với $y$ theo hệ số tỉ lệ $\dfrac{1}{k}$.
 \end{itemize}
	\item  Nếu đại lượng $y$ liên hệ với đại lượng $x$ theo công thức $y=\dfrac{k}{x}$ (hoặc $xy=k$), với $k$ là hằng số khác $0$, thì ta nói $y$ tỉ lệ nghịch với $x$ theo hệ số tỉ lệ $k$.\\
 Cho $y$ tỉ lệ nghịch với $x$ theo hệ số tỉ lệ $k$, tức là $xy=k$ hay $x=\dfrac{k}{y}$. Khi đó
 \begin{itemize}
  \item $x$ cũng tỉ lệ nghịch với $y$, do vậy ta thường nói hai đại lượng này tỉ lệ nghịch với nhau.
  \item $x$ tỉ lệ nghịch với $y$ theo hệ số tỉ lệ $k$.
 \end{itemize}
\item Nếu đại lượng $y$ phụ thuộc vào đại lượng thay đổi $x$ và với mỗi giá trị của $x$ ta luôn xác định được chỉ một giá trị tương ứng của $y$ thì $y$ được gọi là \textit{hàm số} của $x$ và $x$ gọi là \textit{biến số}.
\item Đồ thị hàm số $ y = f(x) $ là tập hợp tất cả các điểm biểu diễn các cặp giá trị tương ứng $ \left( x; y \right) $ trên mặt phẳng tọa độ.
\item Đồ thị hàm số $ y = ax $ ($ a \ne 0 $) là một đường thẳng đi qua gốc tọa độ.\\
Để vẽ đồ thị hàm số $ y = ax $ ($ a \ne 0 $), ta thực hiện như sau
\begin{itemize}
	\item Xác định thêm một điểm $ A\left(x_A;ax_A \right) $ với $ x_A \ne 0. $
	\item Vẽ đường thẳng qua hai điểm $ O $ với $A $ ta được đồ thị hàm số $ y = ax. $
\end{itemize} 
\end{enumerate}
\subsection{Các dạng toán}
\begin{dang}[Một số bài toán về đại lượng tỉ lệ thuận, đại lượng tỉ lệ nghịch]
\begin{enumerate}[\tickEX]
	\item Nếu hai đại lượng $y$ và $x$ tỉ lệ thuận với nhau thì
		\begin{itemize}
		\item Tỉ số giữa hai giá trị tương ứng của chúng luôn không đổi.
		\item Tỉ số giữa hai giá trị bất kì của đại lượng này bằng tỉ số giữa hai giá trị tương ứng của đại lượng kia.
		\end{itemize}
	\item Nếu hai đại lượng $y$ và $x$ tỉ lệ nghịch với nhau thì
		\begin{itemize}
		\item Tích giữa hai giá trị tương ứng của chúng luôn không đổi.
		\item Tỉ số giữa hai giá trị bất kì của đại lượng này bằng nghịch đảo của tỉ số giữa hai giá trị tương ứng của đại lượng kia.
		\end{itemize}
\end{enumerate}
\end{dang}

\begin{vd}
Từ một sợi dây đồng, bác Khoa cắt lấy hai sợi dây $18$m và $14$m. Hỏi mỗi sợi dây đồng này nặng bao nhiêu gam, biết rằng dây đồng thứ hai nhẹ hơn dây đồng thứ nhất $30$ gam.
\loigiai{
Gọi khối lượng của hai sợi dây đồng thứ nhất và thứ hai lần lượt là $x$ (g) và $y$ (g).Ta có $x-y=30$. Vì khối lượng của sợi dây đồng tỉ lệ thuận với chiều dài của sợi dây đó nên ta có \[\dfrac{x}{18}=\dfrac{y}{14}.\]
Áp dụng tính chất của dãy tỉ số bằng nhau ta có \[\dfrac{x}{18}=\dfrac{y}{14}=\dfrac{x-y}{18-14}=\dfrac{30}{4}=7,5.\]
Từ đó suy ra $x=7,5\cdot 18=135$ (g) và $y=7,5\cdot 14=105$ (g).\\
Vậy khối lượng của sợi dây đồng thứ nhất, thứ hai lần lượt là $135$ gam, $105$ gam.
}
\end{vd}

\begin{vd}
Đi bộ từ nhà $A$ đến trường $B$ anh Tuấn cần $50$ phút, còn đi xe đạp thì chỉ cần $0,3$ giờ. Tính quãng đường từ nhà $A$ đến trường $B$, biết vận tốc xe đạp lớn hơn vận tốc đi bộ là $8$ km/h.
\loigiai{
Gọi vận tốc và thời gian của anh Tuấn đi từ nhà tới trường khi đi bộ là $v_1$ và $t_1$; khi đi xe đạp là $v_2$ và $t_2$. Theo đề bài ta có $v_2-v_1=8$ (km/h), $t_1=\dfrac{5}{6}$ (giờ) và $t_2=0,3$ (giờ).\\
Trên cùng quãng đường $AB$, vận tốc và thời gian tỉ lệ nghịch, do đó
\[\dfrac{v_1}{v_2}=\dfrac{t_2}{t_1}=0,3:\dfrac{5}{6}=\dfrac{9}{25}.\]
Suy ra $\dfrac{v_1}{9}=\dfrac{v_2}{25}=\dfrac{v_2-v_1}{25-9}=\dfrac{8}{16}=0,5$ hay $v_2=0,5\cdot 25=12,5$ (km/h).\\
Vậy quãng đường đi từ nhà đến trường là $AB=v_2\cdot t_2=12,5\cdot 0,3=3,75$ (km).
}
\end{vd}


\begin{dang}[Một số bài toán về chia tỉ lệ]
\begin{enumerate}[\tickEX]
	\item Nếu chia số $m$ thành ba phần $x,y,z$ tỉ lệ với $a,b,c$ thì ta có \[\dfrac{x}{a}=\dfrac{y}{b}=\dfrac{z}{c}=\dfrac{x+y+z}{a+b+c}=\dfrac{m}{a+b+c}\cdot\]
	\item Chia số $m$ thành ba phần $x,y,z$ tỉ lệ nghịch với $a,b,c$, ta có $ax=by=cz$ và $x+y+z=m.$ Do đó \[\dfrac{x}{\frac{1}{a}}=\dfrac{y}{\frac{1}{b}}=\dfrac{z}{\frac{1}{c}}=\dfrac{x+y+z}{\frac{1}{a}+\frac{1}{b}+\frac{1}{c}}=\dfrac{m}{\frac{1}{a}+\frac{1}{b}+\frac{1}{c}}\cdot\]
\end{enumerate}
\end{dang}
\begin{vd}
Hãy chia số $42$ thành ba phần tỉ lệ thuận với $\dfrac{2}{3}; \dfrac{1}{4}; \dfrac{5}{6}\cdot$
\loigiai{
Gọi ba phần được chia từ số $42$ là $a$, $b$, $c$. Ta có $a+b+c=42$.\\
Vì $a$, $b$, $c$ tỉ lệ thuận với $\dfrac{2}{3}$; $\dfrac{1}{4}$; $\dfrac{5}{6}$ nên ta có
\[\dfrac{a}{\frac{2}{3}}=\dfrac{b}{\frac{1}{4}}=\dfrac{c}{\frac{5}{6}}\Rightarrow \dfrac{a}{\frac{2}{3}}=\dfrac{b}{\frac{1}{4}}=\dfrac{c}{\frac{5}{6}}=\dfrac{a+b+c}{\frac{2}{3}+\frac{1}{4}+\frac{5}{6}}=\dfrac{42}{\frac{21}{12}}=24.\]
Từ đó suy ra $a=24\cdot \dfrac{2}{3}=16$; $b=24\cdot \dfrac{1}{4}=6$; $c=24\cdot \dfrac{5}{6}=20$.
}
\end{vd}

\begin{vd}
Chia số $m$ thành ba số tỉ lệ thuận với $4;7;9$. Tổng các bình phương của ba số đó là $1314$. Tìm số $m$.
\loigiai{
Giả sử chia số $M$ thành ba phần $a$, $b$, $c$ tỉ lệ thuận với  $4$; $7$; $9$. Ta có
\[M=a+b+c;\; \dfrac{a}{4}=\dfrac{b}{7}=\dfrac{c}{9};\;a^2+b^2+c^2=1314.\]
Đặt $a=4t$, $b=7t$, $c=9t$. Khi đó
\[\left(4t\right)^2+\left(7t\right)^2+\left(9t\right)^2=1314\Rightarrow 16t^2+49t^2+81t^2=1314\Rightarrow t^2=9\Rightarrow t=\pm 3.\]
\begin{itemize}
\item Nếu $t=3$ thì $a=4t=12$; $b=7t=21$; $c=9t=27$. Vậy $M=60$.
\item Nếu $t=-3$ thì $a=4t=-12$; $b=7t=-21$; $c=9t=-27$. Vậy $M=-60$.
\end{itemize}
}
\end{vd}

\begin{vd}
Hãy chia số $168$ thành ba phần tỉ lệ nghịch với $5$; $3$; $6$.
\loigiai{
Giả sử chia số $168$ thành ba phần $a$, $b$, $c$ tỉ lệ nghịch với $5$; $3$; $6$. Ta có $a+b+c=168$.\\
Vì $a$, $b$, $c$ tỉ lệ nghịch với $5$; $3$; $6$ nên 
\[\dfrac{a}{\frac{1}{5}}=\dfrac{b}{\frac{1}{3}}=\dfrac{c}{\frac{1}{6}}\Rightarrow \dfrac{a}{\frac{1}{5}}=\dfrac{b}{\frac{1}{3}}=\dfrac{c}{\frac{1}{6}}=\dfrac{a+b+c}{\frac{1}{5}+\frac{1}{3}+\frac{1}{6}}=\dfrac{168}{\frac{21}{30}}=240.\]
Từ đó suy ra $a=240\cdot \dfrac{1}{5}=48$; $b=240\cdot \dfrac{1}{3}=80$; $c=240\cdot \dfrac{1}{6}=40$.\\
Vậy ba phần cần tìm lần lượt là $48$, $80$, $40$.
}
\end{vd}

\begin{vd}
Hãy chia số $128$ thành bốn phần tỉ lệ nghịch với $2$; $5$; $7$; $14$.
\loigiai{
Giả sử chia số $128$ thành ba phần $a$, $b$, $c$, $d$ tỉ lệ nghịch với $2$; $5$; $7$; $14$. Ta có $a+b+c+d=128$.\\
Vì $a$, $b$, $c$, $d$ tỉ lệ nghịch với $2$; $5$; $7$; $14$ nên 
\[\dfrac{a}{\frac{1}{2}}=\dfrac{b}{\frac{1}{5}}=\dfrac{c}{\frac{1}{7}}=\dfrac{d}{\frac{1}{14}}\Rightarrow \dfrac{a}{\frac{1}{5}}=\dfrac{b}{\frac{1}{3}}=\dfrac{c}{\frac{1}{6}}=\dfrac{d}{\frac{1}{14}}=\dfrac{a+b+c+d}{\frac{1}{5}+\frac{1}{3}+\frac{1}{6}+\frac{1}{14}}=\dfrac{128}{\frac{32}{35}}=140.\]
Từ đó suy ra $a=140\cdot \dfrac{1}{2}=70$; $b=140\cdot \dfrac{1}{5}=28$; $c=140\cdot \dfrac{1}{7}=20$; $d=140\cdot \dfrac{1}{14}=10$ .\\
Vậy bốn phần cần tìm lần lượt là $70$, $28$, $20$, $10$.
}
\end{vd}

\begin{vd}
 Cho tam giác $ABC$ có số đo các góc $\widehat{A}$, $\widehat{B}$, $\widehat{C}$ lần lượt tỉ lệ với $1$; $4$; $7$. Tính số đo các góc của tam giác $ABC$.
 \loigiai
  {
  Trong tam giác $ABC$, ta có $\widehat{A}+\widehat{B}+\widehat{C}=180^\circ$.\\
  Từ giả thiết ta có
  \begin{eqnarray*}
  	\dfrac{\widehat{A}}{1} = \dfrac{\widehat{B}}{4} = \dfrac{\widehat{C}}{7} \Rightarrow \dfrac{\widehat{A}}{1} = \dfrac{\widehat{B}}{4} = \dfrac{\widehat{C}}{7} = \dfrac{\widehat{A}+\widehat{B}+\widehat{C}}{1+4+7} = \dfrac{180^\circ}{12} = 15^\circ.
  \end{eqnarray*}
  Từ đó suy ra $\widehat{A}=15^\circ$, $\widehat{B}=60^\circ$, $\widehat{C}=105^\circ$.\\
  Vậy ta được số đo các góc của tam giác $ABC$ là $\widehat{A}=15^\circ$, $\widehat{B}=60^\circ$, $\widehat{C}=105^\circ$.
  }
\end{vd}

\begin{vd}
 Cho tam giác $ABC$ có chu vi bằng $22$ cm và các cạnh $a$, $b$, $c$ của tam giác tỉ lệ với các số $2$; $4$; $5$. Tính độ dài các cạnh của tam giác.
 \loigiai
  {
  Từ giả thiết ta có $a+b+c=22$ cm.\\
  Lại có
  \begin{eqnarray*}
  	\dfrac{a}{2} = \dfrac{b}{4} = \dfrac{c}{5} \Rightarrow \dfrac{a}{2}=\dfrac{b}{4} = \dfrac{c}{5}=\dfrac{a+b+c}{2+4+5}= \dfrac{22}{11} = 2.
  \end{eqnarray*}
  Từ đó suy ra $a=4$ cm, $b=8$ cm và $c=10$ cm.
  }
\end{vd}

\begin{vd}
Tìm ba số tự nhiên biết rằng bội chung nhỏ nhất của chúng bằng $360$, số thứ nhất và số thứ hai tỉ lệ nghịch với $3$ và $2$, số thứ hai và số thứ ba tỉ lệ thuận với $2$ và $3$.
\loigiai{
Gọi ba số tự nhiên cần tìm lần lượt là $a$, $b$, $c$. Theo bài ra ta có BCNN$(a,b,c)=360$; $3a=2b$; $3b=2c$. Ta có
\[\left\{\begin{aligned}3a=2b\\ 3b=2c\end{aligned}\right.\Rightarrow \left\{\begin{aligned}\dfrac{a}{2}=\dfrac{b}{3}\\ \dfrac{b}{2}=\dfrac{c}{3}\end{aligned}\right.\Rightarrow \left\{\begin{aligned}\dfrac{a}{4}=\dfrac{b}{6}\\ \dfrac{b}{6}=\dfrac{c}{9}\end{aligned}\right.\Rightarrow \dfrac{a}{4}=\dfrac{b}{6}=\dfrac{c}{9}.\]
Đặt $\dfrac{a}{4}=\dfrac{b}{6}=\dfrac{c}{9}=k$, suy ra $a=4k=2^2\cdot k $; $b=6k=2\cdot 3\cdot k$; $c=9k=3^2\cdot k$. Do đó BCNN$(a,b,c)=2^2\cdot 3^2\cdot k=36k$. Theo bài ra ta có \[36k=360\Rightarrow k=10\Rightarrow a=4\cdot 10=40; b=6\cdot 10=60; c=9\cdot 10=90.\]
Vậy ba số cần tìm là $40$; $60$; $90$.
}
\end{vd}

\begin{vd}
Biết tổng hai số đầu hơn số thứ ba là $2000$. Tìm ba số, nếu số thứ nhất bằng $45\%$ số thứ hai và số thứ ba bằng $135\%$ số thứ hai.
\loigiai{
Gọi ba số phải tìm là $x$, $y$, $z$. Ta có $x+y-z=2000$.\\
Theo đề bài ta có $x:y=45\%\Rightarrow x:y=9:20$ và $z:y=135\%\Rightarrow z:y=27:20$.\\
Suy ra dãy tỉ số bằng nhay $\dfrac{x}{9}=\dfrac{y}{20}=\dfrac{z}{27}$.\\
Áp dụng tính chất của dãy tỉ số bằng nhau ta có 
\[k=\dfrac{x}{9}=\dfrac{y}{20}=\dfrac{z}{27}=\dfrac{x+y-z}{9+20-27}=\dfrac{2000}{2}=1000.\]
Suy ra $x=9000$; $y=20000$; $z=27000$.
}
\end{vd}

\begin{vd}
Một xe tải chạy từ thành phố $A$ đến hải cảng $B$ gồm ba chặng đường dài bằng nhau, nhưng chất lượng mặt đường xấu tốt khác nhau nên vận tốc trên mỗi chặng đường lần lượt bằng $40$, $24$ và $60$ (km/h). Biết tổng thời gian đi từ $A$ đến $B$ là $5$ giờ, tính độ dài quãng đường $AB$.
\loigiai{
Gọi vận tốc và thời gian xe tải đi trên ba chặng đường lần lượt là $v_1$, $v_2$, $v_3$; $t_1$, $t_2$, $t_3$. Ta có \[t_1+t_2+t_3=5.\]
Vì ba chặng đường cùng nhau nên cùng quãng đường, vận tốc và thời gian là hai đại lượng tỉ lệ nghịch. Ta có
\[t_1:t_2:t_3=\dfrac{1}{v_1}:\dfrac{1}{v_2}:\dfrac{1}{v_3}=\dfrac{1}{40}:\dfrac{1}{24}:\dfrac{1}{60}=3:5:2.\]
Áp dụng tính chất của dãy tỉ số bằng nhau ta có 
\[\dfrac{t_1}{3}=\dfrac{t_2}{5}=\dfrac{t_3}{2}=\dfrac{t_1+t_2+t_3}{3+5+2}=\dfrac{5}{10}=0,5.\]
Suy ra $t_1=0,5\cdot 3=1,5$ (h) và quãng đường $AB=3\cdot (40\cdot 1,5)=180$ (km).
}
\end{vd}



\begin{dang}[Một số bài toán về hàm số]

\begin{vn}
Cho hàm số $f(x)$ xác định với mọi $x\in\mathbb{Q}$ thỏa mãn điều kiện
\[f(x_1+x_2)=f(x_1)+f(x_2),\;\;\forall x_1,x_2\in\mathbb{Q}.\]
\begin{enumerate}
	\item Chứng minh rằng $f(-x)=-f(x), \forall x\in\mathbb{Q}$;
	\item Cho $f(1)=a\in\mathbb{Q}$. Chứng minh rằng $f(x)=ax, \forall x\in\mathbb{Q}$.
\end{enumerate}
\loigiai
{
\begin{enumerate}
\item Ta có \[f(x_1+x_2)=f(x_1)+f(x_2),\;\;\forall x_1,x_2\in\mathbb{Q}.\tag{1}\]
Thay $x_1=x_2=0$ vào $(1)$ ta được 
\[f(0+0)=f(0)+f(0)\Rightarrow f(0)=0.\]
Thay $x_1=x$ và $x_2=-x$ vào $(1)$ ta được 
\[f(x-x)=f(x)+f(-x)\Rightarrow f(-x)=-f(x).\]
Vậy $f(-x)=-f(x), \forall x\in\mathbb{Q}$.
\item Với $n\in\mathbb{N}$, $n\ge 2$, ta có
\[f\left(\dfrac{1}{n}+\dfrac{1}{n}\right)=f\left(\dfrac{1}{n}\right)+f\left(\dfrac{1}{n}\right)\Rightarrow f\left(\dfrac{2}{n}\right)=2f\left(\dfrac{1}{n}\right).\]
Tương tự ta có
\[f\left(\dfrac{3}{n}\right)=f\left(\dfrac{2}{n}+\dfrac{1}{n}\right)=f\left(\dfrac{2}{n}\right)+f\left(\dfrac{1}{n}\right)=2f\left(\dfrac{1}{n}\right)+f\left(\dfrac{1}{n}\right)=3f\left(\dfrac{1}{n}\right).\]
Cứ tiếp tục như vậy ta có
\begin{align*}
f\left(\dfrac{n}{n}\right)&=f\left(\underbrace{\dfrac{1}{n}+\dfrac{1}{n}+\cdots+\dfrac{1}{n}}_{n\;\text{số hạng $\frac{1}{n}$}}\right)=\underbrace{f\left(\dfrac{1}{n}\right)+f\left(\dfrac{1}{n}\right)+\cdots+f\left(\dfrac{1}{n}\right)}_{n\;\text{số hạng $f\left(\frac{1}{n}\right)$}}\\ 
&=nf\left(\dfrac{1}{n}\right)\\
\Rightarrow\;f(1)&=nf\left(\dfrac{1}{n}\right)\Rightarrow f\left(\dfrac{1}{n}\right)=\dfrac{f(1)}{n}=\dfrac{a}{n}.\tag{3}
\end{align*}
Với $x\in\mathbb{Q}$ bất kì thì $x=\dfrac{m}{n}$ với $m,n\in\mathbb{Z}$, $n\ne 0$. Từ $(3)$ và từ giả thiết ta có
\begin{align*}
f\left(\dfrac{m}{n}\right)&=f\left(\underbrace{\dfrac{1}{n}+\dfrac{1}{n}+\cdots+\dfrac{1}{n}}_{m\;\text{số hạng $\frac{1}{n}$}}\right)=\underbrace{f\left(\dfrac{1}{n}\right)+f\left(\dfrac{1}{n}\right)+\cdots+f\left(\dfrac{1}{n}\right)}_{m\;\text{số hạng $f\left(\frac{1}{n}\right)$}}\\ 
&=mf\left(\dfrac{1}{n}\right)=m\cdot\dfrac{a}{n}=a\cdot \dfrac{m}{n}=ax.
\end{align*}
Vậy $f(x)=ax, \forall x\in\mathbb{Q}$.
\end{enumerate}
}
\end{vn}
