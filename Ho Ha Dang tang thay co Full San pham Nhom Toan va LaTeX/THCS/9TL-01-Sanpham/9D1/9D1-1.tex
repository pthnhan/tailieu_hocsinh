\section{CĂN BẬC HAI}
\subsection{Kiến thức cần nhớ}
\begin{dn}\hfill
	\begin{enumerate}
		\item Căn bậc hai của số thực $a$ là số $x$ sao cho $x^2 =a$.
		\begin{itemize}
			\item Mỗi số dương $a$ đều có đúng hai căn bậc hai là hai số đối nhau, số dương kí hiệu là $\sqrt{a}$ còn số âm kí hiệu là $-\sqrt{a}$.
			\item Số $0$ có đúng một căn bậc hai chính là số $0$, ta viết $\sqrt{0}=0$.
			\item Số âm không có căn bậc hai.
		\end{itemize}
		\item Với mỗi số dương $a$, số $\sqrt{a}$ được gọi là \textbf{\textit{căn bậc hai số học}} của $a$.
	\end{enumerate}
	\begin{note} Chú ý
		\begin{itemize}
			\item Số $0$ cũng được gọi là căn bậc hai số học của $0$.
			\item $\sqrt{a}$ xác định khi và chỉ khi $a\ge 0$.
		\end{itemize}
	\end{note}
\end{dn}  

\begin{dl}
	Với hai số $a$, $b$ không âm ta có $a<b \Leftrightarrow \sqrt{a}<\sqrt{b}$.
\end{dl}

\subsection{Các dạng toán}

\begin{dang}{Tìm căn bậc hai hoặc căn bậc hai số học của một số}
	Phương pháp giải: Sử dụng định nghĩa hoặc máy tính cầm tay.
\end{dang}

\begin{vd}
	Tính căn bậc hai của các số sau
	\begin{multicols}{2}
		\begin{enumerate}
			\item $1$
			\item $9$
			\item $\dfrac{16}{9}$
			\item $0{,}36$
		\end{enumerate}
	\end{multicols}
	\loigiai{
		Ta có 
		\begin{enumerate}
			\item Căn bậc hai của số $1$ là $\pm 1$ vì $1^2=(-1)^2 =1$.
			\item Căn bậc hai của số $9$ là $\pm 3$ vì $3^2=(-3)^2 =9$.
			\item Căn bậc hai của số $\dfrac{16}{9}$ là $\pm \dfrac{4}{3}$ vì $\left(\dfrac{4}{3}\right)^2=\left(-\dfrac{4}{3}\right)^2 =\dfrac{16}{9}$.
			\item Căn bậc hai của số $0{,}36$ là $\pm 0{,}6$ vì $(0{,}6)^2=(-0{,}6)^2 =0{,}36$.
		\end{enumerate}
		\begin{note}
			Học sinh có thể sử dụng máy tính cầm tay để kiểm tra kết quả.
		\end{note}
	}
\end{vd}

\begin{vd}
	Tính căn bậc hai số học của các số sau
	\begin{multicols}{2}
		\begin{enumerate}
			\item $0{,}01$
			\item $0{,}04$
			\item $0{,}25$
			\item $\dfrac{4}{9}$
		\end{enumerate}
	\end{multicols}
	\loigiai{
		Ta có 
		\begin{multicols}{2}
		\begin{enumerate}
			\item $\sqrt{0{,}01} = 0{,1}$ vì $0{,}1^2 = 0{,}01$.
			\item $\sqrt{0{,}04} = 0{,2}$ vì $0{,}2^2 = 0{,}04$.
			\item $\sqrt{0{,}25} = 0{,5}$ vì $0{,}5^2 = 0{,}25$.
			\item $\sqrt{\dfrac{4}{9}} = \dfrac{2}{3}$ vì $\left(\dfrac{2}{3}\right)^2 = \dfrac{4}{9}$.
		\end{enumerate}
		\end{multicols}
		\begin{note}
			Học sinh có thể sử dụng máy tính cầm tay để kiểm tra kết quả.
		\end{note}
	}
\end{vd}

\begin{vd}
	Tính tổng $S=\sqrt{0{,}49} + \sqrt{\dfrac{1}{9}}-\sqrt{\dfrac{25}{4}}$.
	\loigiai{
		Ta có $S=0{,}7 + \dfrac{1}{3}-\dfrac{5}{2}=-\dfrac{22}{15}$}.
	\begin{note}
		Học sinh có thể sử dụng máy tính cầm tay để kiểm tra kết quả.
	\end{note}
\end{vd}

\begin{dang}{So sánh các căn bậc hai}
	Ta thường sử dụng tính chất cơ bản của bất đẳng thức, cụ thể:
	\begin{itemize}
		\item Nếu $\heva{&a>b\\&c >d}$ thì $a+c > b+d$.
		\item Nếu $\heva{&a>b\\ &c>0}$ thì $ac >bc$.
		\item Nếu $\heva{&a>b\\ &c<0}$ thì $ac <bc$.
		\item Nếu $\heva{&a>b>0\\&c >d>0}$ thì $ac > bd$.
		\item Với hai số $a,b$ không âm ta có $\sqrt{a}<\sqrt{b} \Leftrightarrow a<b \Leftrightarrow a^2 < b^2$.
	\end{itemize}
\end{dang}

\begin{vd}
	So sánh các số sau:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\sqrt{26}$ và $5$.
			\item $\sqrt{7}+\sqrt{15}$ và $7$.
			\item $\sqrt{2}+\sqrt{11}$ và $\sqrt{3}+5$.
			\item $-5\sqrt{35}$ và $-30$.
		\end{enumerate}
	\end{multicols}
	\loigiai{\begin{enumerate}
			\item Ta có $26 > 25 \Rightarrow \sqrt{26} > \sqrt{25}$ hay $\sqrt{26}>5$.
			\item Ta có $\heva{&7<9\\&15<16} \Rightarrow \heva{&\sqrt{7}<\sqrt{9}\\&\sqrt{15}<\sqrt{16}}\Rightarrow \heva{&\sqrt{7}<3\\&\sqrt{15}<4.}$\\
			Như vậy $\sqrt{7}+\sqrt{15}<3+4=7$.
			\item Ta có $\heva{&2<3\\&11<25} \Rightarrow \heva{&\sqrt{2}<\sqrt{3}\\&\sqrt{11}<\sqrt{25}}\Rightarrow \heva{&\sqrt{2}<\sqrt{3}\\&\sqrt{11}<5.}$\\
			Như vậy $\sqrt{2}+\sqrt{11}<\sqrt{3}+5$.
			\item Ta có $35<36\Rightarrow \sqrt{35}<\sqrt{36}=6\Rightarrow -5\sqrt{35}>(-5)\cdot 6 \Rightarrow -5\sqrt{35}>-30$.
		\end{enumerate}
	}
\end{vd}

\begin{vd}
	Cho $a>0$. Chứng minh rằng 
	\begin{multicols}{2}
		\begin{enumerate}
			\item Nếu $a>1$ thì $a>\sqrt{a}$.
			\item Nếu $a<1$ thì $a<\sqrt{a}$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item Ta có tính chất, nếu $a>b>0$ thì $\sqrt{a}> \sqrt{b}$, do đó từ giả thiết $a>1\Rightarrow \sqrt{a}> \sqrt{1}=1$.\\
			Nhân cả hai vế với $\sqrt{a}>0$ ta được $a>\sqrt{a}$.
			\item Tương tự như trên ta có $a<1\Rightarrow \sqrt{a}<\sqrt{1}=1$.\\
			Nhân cả hai vế với $\sqrt{a}>0$ ta được $a<\sqrt{a}$.
		\end{enumerate}
	}
\end{vd}

\begin{dang}{Tìm $x$}
	Phương pháp giải: Thường biến đổi  biểu thức về dạng $\sqrt{f(x)}=a. \hfill(*)$
	\begin{itemize}
		\item Nếu $a<0$ thì $(*)$ vô nghiệm.
		\item Nếu $a=0$ thì $(*)\Leftrightarrow f(x)=0$.
		\item Nếu $a>0$ thì $(*) \Leftrightarrow f(x)=a^2$.
	\end{itemize}
	\begin{note}
		Nếu không biến đổi tương đương được các phương trình thì có thể dùng phép biến đổi suy ra sau đó phải thử lại.
	\end{note}
\end{dang}
\begin{vd}
	Tìm $x$ thỏa mãn:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\sqrt{x}=-2018$.
			\item $\sqrt{x+1}-1=2$.
		\end{enumerate} 
	\end{multicols}    
	\loigiai{
		\begin{enumerate}
			\item Vì $\sqrt{x}\ge 0$ và $-2018<0$ nên không tồn tại $x$ thỏa mãn.
			\item Điều kiện $x+1 \ge 0 \Leftrightarrow x\ge -1$.\\
			Khi đó $\sqrt{x+1}-1=2 \Leftrightarrow \sqrt{x+1}=3 \Leftrightarrow x+1=9 \Leftrightarrow x=8$ (thỏa mãn điều kiện).\\
			Vậy $x=8.$
		\end{enumerate}          
	}
\end{vd}

\begin{vd}
	Tìm $x$ thỏa mãn 
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\sqrt{x^2+5x+20}=4$.
			\item $3-\sqrt{x^2+5}=4$.
		\end{enumerate} 
	\end{multicols}    
	\loigiai{
		\begin{enumerate}
			\item Ta có $x^2+5x+20 = x^2 + 5x+ \dfrac{25}{4}+\dfrac{55}{4}=\left(x+\dfrac{5}{2}\right)^2 + \dfrac{55}{4}>0, \forall x \in \Bbb R$. Khi đó
			\begin{eqnarray*}
				& &\sqrt{x^2+5x+20}=4 \Leftrightarrow x^2+5x+20=16 \\
				&\Leftrightarrow & x^2+5x+4=0 \Leftrightarrow (x+1)(x+4)=0\\
				& \Leftrightarrow &\hoac{&x=-1\\&x=-4.}
			\end{eqnarray*}		      
			Vậy $x=-1$ hoặc $x=-4$.
			\item Điều kiện $x^2+5 \ge 0$ (luôn đúng). Ta có
			$$3-\sqrt{x^2+5}=4 \Leftrightarrow \sqrt{x^2+5} =3-4=-1.$$
			Vì $\sqrt{x^2+5} >0$ còn $-1<0$ nên không tồn tại $x$ thỏa mãn.
		\end{enumerate}          
	}
\end{vd}

\subsection{Luyện tập}
\begin{bt}
	Tìm căn bậc hai số học và căc bậc hai của các số sau:
\begin{multicols}{2}
	\columnsep=0pt
	\columnseprule=0pt
	\begin{enumerate}
		\item $ 0{,}25 $.
		\item $ 81 $.
		\item $ 169 $.
		\item $ 2{,}25 $.
	\end{enumerate}
\end{multicols}	
\loigiai{
	\begin{enumerate}
\item Vì $ 0{,}25=0{,}5^{2} $ nên căn bậc hai số học của $ 0{,}25$ là $0{,}5 $ và căn bậc hai của $ 0{,}25$ là $\pm 0{,}5$.
\item Vì $ 81=9^{2} $ nên căn bậc hai số học của $ 81$ là $9$ và căn bậc hai của $ 81$ là $\pm 9$.
\item Vì $ 169=13^{2} $ nên căn bậc hai số học của $169$ là $13$ và căn bậc hai của $169$ là $\pm 13$.
\item Vì $ 2{,}25=1{,}5^{2} $ nên căn bậc hai số học của $2{,}25$ là $1{,}5$ và căn bậc hai của $2{,}25$ là $\pm 1{,}5$.
	\end{enumerate}
}
\end{bt}
\begin{bt}
Rút gọn biểu thức:
\begin{multicols}{2}
	\columnsep=0pt
\columnseprule=0pt
		\begin{enumerate}
		\item $A=2\sqrt{27}+5\sqrt{12}-3\sqrt{48}$.
		\item $B=\sqrt{147}+\sqrt{54}-4\sqrt{27} $.
		\item $C=3\sqrt{2}(4-\sqrt{2})+3(1-2\sqrt{2})^{2}$.
		\item $D=2\sqrt{5}-\sqrt{125}-\sqrt{80}+\sqrt{605}$.
		\end{enumerate}
	\end{multicols}	
\loigiai{
\begin{enumerate}
	\item $A=2\sqrt{27}+5\sqrt{12}-3\sqrt{48}=6\sqrt{3}+10\sqrt{3}-12\sqrt{3}=4\sqrt{3}$.
	\item $B=\sqrt{147}+\sqrt{75}-4\sqrt{27}=7\sqrt{3}+5\sqrt{3}-12\sqrt{3}=0$.
	\item $C=3\sqrt{2}(4-\sqrt{2})+3(1-2\sqrt{2})^{2}=12\sqrt{2}-6+3(1-4\sqrt{2}+8)=12\sqrt{2}-6+27-12\sqrt{2}=21$.
	\item $D=2\sqrt{5}-\sqrt{125}-\sqrt{80}+\sqrt{605}=2\sqrt{5}-5\sqrt{5}-4\sqrt{5}+11\sqrt{5}=4\sqrt{5}$.
\end{enumerate}
}
\end{bt}
\begin{bt}
So sánh các số sau:
\begin{multicols}{2}
	\columnsep=0pt
\columnseprule=0pt
	\begin{enumerate}
		\item $6$ và $\sqrt{41}$.
		\item $2\sqrt{27}$ và $\sqrt{147}$.
		\item $-3\sqrt{5}$ và $-5\sqrt{3}$.
		\item $2\sqrt{2}-1$ và $2$.
	\end{enumerate}
\end{multicols}	
	\loigiai{
\begin{enumerate}
\item Ta có $ 6=\sqrt{36} $. Mà $ 36<41 $ nên $ 6<\sqrt{41} $.
\item Ta có $ 2\sqrt{27}=\sqrt{108} $. Mà $ 108<147 $ nên $ 2\sqrt{27}<\sqrt{147} $.
\item Ta có $ 3\sqrt{5}=\sqrt{45} $ và $ 5\sqrt{3}=\sqrt{75} $. Mà $ 45<75 $ nên $ 3\sqrt{5}<5\sqrt{3}\Rightarrow -3\sqrt{5}>-5\sqrt{3} $.
\item Ta có $ 2\sqrt{2}-1=\sqrt{8}-1 $ và $ 2=3-1=\sqrt{9}-1 $. Mà $ 8<9 $ nên $ 2\sqrt{2}-1<2 $.
\end{enumerate}		
}
\end{bt}

\begin{bt}
	Tìm số thực $ x $ thỏa mãn:
	\begin{multicols}{2}
		\columnsep=0pt
		\columnseprule=0pt
		\begin{enumerate}
			\item $\sqrt{-2x^{2}-9}=2$.
			\item $\sqrt{x^{2}+1}+2=0$.
			\item $\sqrt{3x-1}=4$.
			\item $\sqrt{-3x+4}=12$.
			\item $\sqrt{(\sqrt{x}-7)(\sqrt{x}+7)}=2$.
			\item $\sqrt{9(x-1)}-19=2$.
		\end{enumerate}
	\end{multicols}	
	\loigiai{
		\begin{enumerate}
			\item Điều kiện xác định $ -2x^{2}-9\geq 0$ (vô lí).\\
			Vậy không tồn tại $ x $ thỏa mãn đề bài.
			\item Điều kiện xác định $ x^{2}+1\geq 0$ (luôn đúng).\\
			Ta có $ \sqrt{x^{2}+1}+2=0\Leftrightarrow \sqrt{x^{2}+1}=-2 $ (vô lí vì $ \sqrt{x^{2}+1}>0 $ với mọi $ x $).\\
			Vậy không tồn tại $ x $ thỏa mãn đề bài.
			\item Điều kiện xác định $ 3x-1\geq 0\Leftrightarrow x\geq \dfrac{1}{3} $.\\
			Ta có $ \sqrt{3x-1}=4\Leftrightarrow 3x-1=16\Leftrightarrow x=\dfrac{17}{3} $ (thỏa mãn điều kiện).\\
			Vậy $ x=\dfrac{17}{3} $.
			\item Điều kiện xác định $ -3x+4\geq 0\Leftrightarrow x\leq \dfrac{4}{3} $.\\
			Ta có $ \sqrt{-3x+4}=12\Leftrightarrow -3x+4=144\Leftrightarrow x=-\dfrac{140}{3} $ (thỏa mãn điều kiện).\\
			Vậy $ x=-\dfrac{140}{3} $.
			\item Điều kiện xác định $ \heva{&x\geq 0\\ &(\sqrt{x}-7)(\sqrt{x}+7)\geq 0}\Leftrightarrow\heva{&x\geq 0\\ &x-49\geq 0}\Leftrightarrow x\geq 49 $.\\
			Ta có $ \sqrt{(\sqrt{x}-7)(\sqrt{x}+7)}=2\Leftrightarrow x-49=4\Leftrightarrow x=53 $ (thỏa mãn điều kiện).\\
			Vậy $ x=53 $.
			\item Điều kiện xác định $ 9(x-1)\geq 0\Leftrightarrow x\geq 1 $.\\
			Ta có $\sqrt{9(x-1)}-19=2\Leftrightarrow \sqrt{9(x-1)}=21\Leftrightarrow 9(x-1)=441\Leftrightarrow x-1=49\Leftrightarrow x=50 $ (thỏa mãn điều kiện).\\
			Vậy $ x=50 $.
		\end{enumerate}		
	}
\end{bt}

\begin{bt}
(*) Chứng minh rằng $\sqrt{2}$ là một số vô tỉ.
\loigiai{
	Giả sử $\sqrt{2}$ là số hữu tỉ. Suy ra $\sqrt{2} = \dfrac{m}{n}$, trong đó $m$, $n \in \mathrm{N}^*$ và phân số $\dfrac{m}{n}$ là phân số tối giản.\\
	Khi đó $\sqrt{2} = \dfrac{m}{n} \Leftrightarrow 2 = \left(\dfrac{m}{n}\right)^2 \Leftrightarrow m^2 = 2n^2$. $\hfill (1)$\\
	Do $2n^2 ~\vdots ~ 2$ nên $m^2 ~\vdots ~ 2 \Rightarrow m ~\vdots ~ 2 \Rightarrow m=2m_1, m_1\in \mathrm{N}^* \Rightarrow m^2 =4m_1^2$.\\
	Thay vào $(1)$ suy ra $2n^2 =4m_1^2 \Leftrightarrow n^2 =2m_1^2 ~\vdots ~ 2 \Rightarrow n^2 ~\vdots ~ 2 \Rightarrow n~\vdots ~ 2$.\\
	Do đó $m,n$ cùng chia hết cho $ 2 $ nên phân số $\dfrac{m}{n}$ không tối giản, điều này mâu thuẫn với giả sử ở trên.\\
	Vậy $\sqrt{2}$ là số vô tỉ.
	}
\end{bt}

\begin{bt}
	(*) Chứng minh rằng $\sqrt{5}$ là một số vô tỉ.
	\loigiai{
		Giả sử $\sqrt{5}$ là số hữu tỉ. Suy ra $\sqrt{5} = \dfrac{m}{n}$, trong đó $m$, $n \in \Bbb N^*$ và phân số $\dfrac{m}{n}$ là phân số tối giản.\\
		Khi đó   $\sqrt{5} = \dfrac{m}{n} \Leftrightarrow 5 = \left(\dfrac{m}{n}\right)^2 \Leftrightarrow m^2 = 5n^2.$ $\hfill (1)$\\
		Do $5n^2 ~\vdots ~ 5$ nên $m^2 ~\vdots ~ 5 \Rightarrow m ~\vdots ~ 5 \Rightarrow m=5m_1, m_1\in \Bbb N^* \Rightarrow m^2 =25m_1^2$.\\
		Thay vào $(1)$ suy ra $5n^2 =25m_1^2 \Leftrightarrow n^2 =5m_1^2 ~\vdots ~ 5 \Rightarrow n^2 ~\vdots ~ 5 \Rightarrow n~\vdots ~ 5$.\\
		Do đó $m$, $n$ cùng chia hết cho 5 nên phân số $\dfrac{m}{n}$ không tối giản, điều này mâu thuẫn với giả sử ở trên.\\
		Vậy $\sqrt{5}$ là số vô tỉ.
		
	}
\end{bt}
