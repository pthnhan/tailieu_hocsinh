	Trong các mệnh đề sau, mệnh đề nào \textbf{sai}?
	\choice
	{Hình tứ diện luôn luôn có mặt cầu ngoại tiếp}
	{Hình lăng trụ đứng có đáy tam giác luôn luôn có mặt cầu ngoại tiếp}
	{\True Hình hộp luôn luôn có mặt cầu ngoại tiếp}
	{Hình lăng trụ tam giác đều luôn luôn có mặt cầu ngoại tiếp}
	\loigiai{
		\immini{\textbf``Hình tứ diện luôn luôn có mặt cầu ngoại tiếp''}. Đúng do tứ diện bất kỳ có 4 đỉnh luôn có 1 mặt cầu duy nhất chứa 4 đỉnh của tứ diện này.\\  
		\textbf{``Hình lăng trụ đứng có đáy tam giác luôn luôn có mặt cầu ngoại tiếp''}. Đúng do hình lăng trụ đứng, có đáy là 1 đa giác nội tiếp được trong 1 đường tròn thì lăng trụ đứng này sẽ có mặt cầu ngoại tiếp.\\
		\textbf{``Hình hộp luôn luôn có mặt cầu ngoại tiếp''}. Sai do hình hộp nếu đáy là một hình bình hành (không phải là hình chữ nhật) thì đáy không nội tiếp đường tròn được nên cả hình hộp không có mặt cầu ngoại tiếp.\\
		\textbf{``Hình lăng trụ tam giác đều luôn luôn có mặt cầu ngoại tiếp''}. Đúng do giống câu \textbf{``Hình lăng trụ đứng có đáy tam giác luôn luôn có mặt cầu ngoại tiếp''}. (thật ra đáy là tam giác là đã có mặt cầu ngoại tiếp, ở câu này còn đặc biệt đáy là tam giác đều).}
\end{ex}

\begin{ex}%[Đỗ Hữu Nhân - (EX-LT1)]%[2H2B2-1]

	\choice
	{$2$}
	{$0$}
	{\True $1$}
	{Vô số}
	\loigiai{
		Trên đường tròn $(C)$ lấy điểm điểm ${{M}_{0}}$ cố định. Gọi $(\alpha )$ là mặt phẳng trung trực của $A{{M}_{0}}$ và đường thẳng $\Delta $ là trục của $(C)$. Gọi $I$ giao điểm của $(\alpha )$ và $\Delta$ thì mặt cầu tâm $I$ thỏa mãn yêu cầu đề bài.\\
		Ta sẽ chứng minh tâm $I$ là duy nhất. Giả sử $M$ là điểm bất kì khác nằm trên đường tròn $(C)$, gọi $(\alpha ')$ là mặt phẳng trung trực của $AM$ và $I'=(\alpha ')\cap \Delta $ thì mặt cầu tâm tâm $I'$ thỏa mãn yêu cầu đề bài. Ta có:
		$I'A=I'M=I'{{M}_{0}}$ $\Rightarrow $ $I'$ thuộc mặt phẳng trung trực $(\alpha )$ của $A{{M}_{0}}$ nên $I'=(\alpha )\cap \Delta $.	Từ đó suy ra $I'\equiv I$. Vậy chỉ có duy nhất 1 mặt cầu thỏa mãn yêu cầu bài toán.
	}
\end{ex}

	Cách thay đổi kích thước nào dưới đây không làm thay đổi thể tích của một khối nón?
	\choice
	{\True Tăng bán kính đáy lên $2$ lần đồng thời giảm chiều cao lại $4$ lần}
	{Tăng bán kính đáy lên $2$ lần đồng thời giảm chiều cao lại $2$ lần}
	{Tăng bán kính đáy lên $2$ lần đồng thời giảm độ dài đường sinh lại $2$ lần}
	{Tăng bán kính đáy lên $2$ lần đồng thời giảm độ dài đường sinh lại $4$ lần}
	\loigiai{
		Với $r$, $h$ lần lượt là bán kính đáy và chiều cao của một khối nón thì thể tích của khối nón đó tính bởi công thức $V=\dfrac{1}{3}\pi r^2h$. Suy ra $h=\dfrac{3V}{\pi}\cdot \dfrac{1}{r^2}$.\\
		Như vậy với $V$ không đổi, $r$ tăng lên $k$ lần thì $h$ giảm lại $k^2$ lần. Áp dụng vào câu hỏi này, để thể tích của khối nón không đổi, tăng bán kính đáy lên $2$ lần thì phải giảm chiều cao lại $4$ lần.}
\end{ex}

		\begin{eqnarray*}
			MA^2+MB^2+MC^2+MD^2&=& \vec{MA}^2+\vec{MB}^2+\vec{MC}^2+\vec{MD}^2\\

			&=& 4MG^2+\dfrac{3a^2}{2}
