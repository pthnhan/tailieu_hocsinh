\begin{ex}%[HSG Toán 9, huyện Bình Giang, tỉnh Hải Dương, năm học 2012 - 2013]%[9D1K8]
	Cho biểu thức $A=\dfrac{x+2}{x\sqrt{x}-1} + \dfrac{\sqrt{x}+1}{x+\sqrt{x}+1} + \dfrac{1}{1-\sqrt{x}}$ với $x\ge 0$, $x\ne 1$.
    \begin{enumerate}
    \item Rút gọn $A$.
    \item Chứng tỏ rằng $A < \dfrac{1}{3}$.
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
    \item 
    	\begin{align*}
    	A & = \dfrac{x+2}{(\sqrt{x}-1)(x+\sqrt{x}+1)} + \dfrac{\sqrt{x}+1}{x+\sqrt{x}+1} + \dfrac{1}{1-\sqrt{x}}\\
    	A & = \dfrac{x+2+x-1-x-\sqrt{x}-1}{(\sqrt{x}-1)(x+\sqrt{x}+1)} = \dfrac{x-\sqrt{x}}{(\sqrt{x}-1)(x+\sqrt{x}+1)}\\
    	A & = \dfrac{\sqrt{x}(\sqrt{x}-1)}{(\sqrt{x}-1)(x+\sqrt{x}+1)} = \dfrac{\sqrt{x}}{x+\sqrt{x}+1},\ \text{với} \ x\ge 0, x\ne 1.
    	\end{align*}
    \item Xét $\dfrac{1}{3} - A = \dfrac{1}{3} - \dfrac{\sqrt{x}+1}{x+\sqrt{x}+1} = \dfrac{(\sqrt{x}-1)^2}{x+\sqrt{x}+1}$.\\
    	Với mọi $x\ge 0$, $x\ne 1$ ta có $(\sqrt{x}-1)^2 > 0$ và $x+\sqrt{x}+1 = \left(\sqrt{x}+\dfrac{1}{2} \right)^2 + \dfrac{3}{4} > 0$ nên suy ra $\dfrac{1}{3} - A > 0 \Leftrightarrow A < \dfrac{1}{3}$.
    \end{enumerate}
    }
\end{ex}

\begin{ex}%[HSG Toán 9, huyện Bình Giang, tỉnh Hải Dương, năm học 2012 - 2013][9D4K7] %[9D5G3]
	\begin{enumerate}
	\item Giải phương trình $x-\sqrt{x-15}=17$.
	\item Tìm $x,y$ sao cho $5x -2\sqrt{x}(2+y)+y^2+1=0$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item Điều kiện xác định $x\ge 15$.\\
		$x-\sqrt{x-15}=17 \Leftrightarrow x-15 - \sqrt{x-15} - 2 = 0.$\\
		Đặt $t=\sqrt{x-15}$ ($t\ge 0$) $\Rightarrow t^2 - t - 2 = 0 \Leftrightarrow \left[ \begin{array}{l} t = 2 \\ t = -1 \ (\text{loại}) \end{array} \right.$\\
		Với $t=2$, suy ra $\sqrt{x-15} = 2 \Leftrightarrow x - 15 = 4 \Leftrightarrow x = 19.$
	\item Điều kiện xác định $x\ge 0$.
		\begin{align*}
		5x -2\sqrt{x}(2+y)+y^2+1 0 & \Leftrightarrow 4x - 4\sqrt{x} + 1 + x -2y\sqrt{x} +y^2 = 0 \\
		& \Leftrightarrow \left(2\sqrt{x} - 1\right)^2 + \left(\sqrt{x} -y \right)^2 = 0. (1)
		\end{align*}
	Vì $\left(2\sqrt{x} - 1\right)^2 + \left(\sqrt{x} -y \right)^2 \ge 0$ với mọi $x\ge 0$, $y$ nên để (1) xảy ra thì 
	$$\begin{cases} 2\sqrt{x} - 1 = 0 \\ \sqrt{x} - y = 0 \end{cases} \Leftrightarrow 
		\begin{cases} x = \dfrac{1}{4} \\ y = \dfrac{1}{2}. \end{cases}$$
	\end{enumerate}
	}
\end{ex}

	\begin{enumerate}

	\item Tìm $m$ để hàm số bậc nhất $y=\dfrac{m^2 - 2013m+2012}{m^2 - 2\sqrt{2}m + 3} x - 2011$ là hàm số nghịch biến.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item Theo bài ta có $p = x^2 + x = x(x+1).$ Mà $x, x+1$ là hai số nguyên liên tiếp nên suy ra $p$ là số chẵn.\\
		Mặt khác $p$ là số nguyên tố nên $p=2$.\\
		Suy ra $x^2 + x - 2 = 0 \Leftrightarrow x = 1 \vee x = - 2$.
	\item Để hàm số $y=\dfrac{m^2 - 2013m+2012}{m^2 - 2\sqrt{2}m + 3} x - 2011$ nghịch biến thì $\dfrac{m^2 - 2013m+2012}{m^2 - 2\sqrt{2}m + 3} < 0$. (1)\\
		Ta có $m^2 - 2\sqrt{2}m + 3 = \left(\sqrt{2} - m \right)^2 + 1 \ge 0$ với mọi $m$ nên (1) tương đương với
			$$m^2 - 2013 m + 2012 <0 \Leftrightarrow (m-1)(m-2012) < 0 \Leftrightarrow 
			\begin{cases} m - 1 > 0 \\ m- 2012 < 0 \end{cases} 
			\Leftrightarrow 1 < m < 2012.$$
	\end{enumerate}
	}
\end{ex}`

\begin{ex}%[HSG Toán 9, huyện Bình Giang, tỉnh Hải Dương, năm học 2012 - 2013]%[9D5G1]
	Cho $x+3y \ge 1$. Tìm giá trị nhỏ nhất của biểu thức $C = x^2 +y^2$.
	\loigiai{
	Do $x+3y \ge 1$ nên đặt $x+3y = 1 +a$, với $a\ge 0$. Suy ra $x = 1 + a - 3y$, thay vào biểu thức $C$ ta có
		\begin{align*}
		C & =10y^2 -6ay - 6y^2 +a^2 +2a +1\\
		& = 10\left[ y- \dfrac{3}{10} (a+1) \right]^2 + \dfrac{1}{10} (a^2 +2a) +\dfrac{1}{10} \ge \dfrac{1}{10}.
		\end{align*}
	Dấu ``$=$'' xảy ra khi $\begin{cases} y - \dfrac{3}{10} (a+1) = 10 \\ a = 0 \end{cases} \Leftrightarrow \begin{cases}  x = \dfrac{1}{10} \\ y = \dfrac{3}{10}. \end{cases}$ \\
	Vậy giá trị nhỏ nhất của $C$ là $\dfrac{1}{10}$ khi $ x = \dfrac{1}{10}$ và $y = \dfrac{3}{10}$.
	}
\end{ex}
