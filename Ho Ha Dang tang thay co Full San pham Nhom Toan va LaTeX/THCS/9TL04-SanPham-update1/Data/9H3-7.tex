\section{Tứ giác nội tiếp}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Định nghĩa và tính chất}
\immini{
	\vspace*{-0.8cm}
\begin{dn}
	Một tứ giác có bốn đỉnh nằm trên một đường tròn được gọi là tứ giác nội tiếp đường tròn (gọi tắt là tứ giác nội tiếp).
\end{dn}
\listenumerate{dl,note}
\begin{dl}
	\begin{enumerate}
		\item
		Trong một tứ giác nội tiếp, tổng số đo hai góc đối nhau bằng $180^\circ$.
		\item
		Nếu một tứ giác có tổng số đo hai góc đối nhau bằng $180^\circ$ thì tứ giác đó nội tiếp được đường tròn.
	\end{enumerate}
\end{dl}

\vspace*{0.2cm}

\begin{note} Từ định lí trên ta có
	\begin{enumerate}
	\item
		Với $A$, $B$, $C$, $D$ là bốn đỉnh của một tứ giác thì:
		$$ABCD \text{ nội tiếp }\Leftrightarrow\hoac{&\widehat{A}+\widehat{C}=180^\circ\\&\widehat{B}+\widehat{D}=180^\circ .}$$
	\item
		Hình thang nội tiếp được trong một đường tròn khi và chỉ khi nó là hình thang cân.
	\end{enumerate}	
\end{note}
\subsubsection{Dấu hiệu nhận biết tứ giác nội tiếp}
\begin{itemize}
	\item Dựa vào định nghĩa của tứ giác nội tiếp.
	\item Chứng minh tứ giác đó có hai góc đối bù nhau (hoặc tứ giác đó có một góc bằng góc ngoài tại đỉnh đối diện).
	\item Dựa vào khái niệm cung chứa góc: Tứ giác có hai đỉnh liên tiếp nhìn đoạn thẳng nối hai đỉnh còn lại dưới hai góc bằng nhau thì tứ giác đó nội tiếp được một đường tròn.	
\end{itemize}

\begin{vd}%[Phạm An Bình]
	Cho tam giác $ABC$ vuông ở $A$ ($AB<AC)$, đường cao $AH$. Trên nửa mặt phẳng bờ $BC$ chứa điểm $A$, vẽ nửa đường tròn đường kính $BH$ cắt $AB$ tại $E$, nửa đường tròn đường kính $HC$ cắt $AC$ tại $F$.
	\begin{enumerate}
		\item Chứng minh tứ giác $AFHE$ là hình chữ nhật.
		\item Chứng minh tứ giác $BEFC$ là tứ giác nội tiếp đường tròn.
	\end{enumerate}

					\item $\widehat{CFH}=90^\circ$ (góc nội tiếp chắn nửa đường tròn đường kính $CH$)

\begin{vd}%[Phạm An Bình]
	Cho nửa đường tròn tâm $O$ đường kính $AB$. $C$ là một điểm nằm giữa $O$ và $A$. Đường thẳng vuông góc với $AB$ tại $C$ cắt nửa đường tròn trên tại $I$. $K$ là một điểm bất kỳ nằm trên đoạn thẳng $CI$ ($K$ khác $C$ và $I$), tia $AK$ cắt nửa đường tròn $(O)$ tại $M$, tia $BM$ cắt tia $CI$ tại $D$. Gọi $E$ đối xứng với $B$ qua $C$.
	\begin{enumerate}
		\item Chứng minh $ACMD$ là tứ giác nội tiếp đường tròn.
		\item Chứng minh $\triangle ABD \backsim \triangle MBC$.
		\item Chứng minh $AKDE$ là tứ giác nội tiếp.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Chứng minh $ACMD$ là tứ giác nội tiếp đường tròn.
			\immini{
				$(O)$ có $\widehat{AMB}$ nội tiếp chắn nửa đường tròn.\\
				$\Rightarrow \widehat{AMB}=90^\circ$.\\
				Vì $CI \perp AB$ nên $\widehat{ACD}=90^\circ$.\\				
				Tứ giác $ACMD$ có $C$, $M$ là hai đỉnh kề nhau và $\widehat{ACD}=\widehat{AMD}=90^\circ$ nên
				$ACMD$ nội tiếp.
				\item Chứng minh $\triangle ABD \backsim \triangle MBC$.\\
				Xét $\triangle ABD$ và $\triangle MBC$ có\\
				$\heva{&\widehat{ABD}\text{ chung}\\&\widehat{ADB}=\widehat{MCB}\,\,(\text{vì } ACMD \text{ nội tiếp.})}$\\
				$\Rightarrow \triangle ABD \backsim \triangle MBC$ (g-g).
			}{
				\begin{tikzpicture}[scale=.8, line join=round, line cap=round,>=stealth, font=\footnotesize, color=black]
				\tikzset{label style/.style={font=\footnotesize}}
				\tkzInit[xmin=-4.2, xmax=4.5, ymin=-.7, ymax=6]\tkzClip
				\tkzDefPoints{0/0/C, -1/0/A, 0/2/I, 3/3/t}
				\tkzDefLine[perpendicular=through I,K=-1](I,A)\tkzGetPoint{b}
				\tkzInterLL(A,H)(I,b)\tkzGetPoint{B}
				\tkzDefMidPoint(A,B)\tkzGetPoint{O}
