\begin{ex}%[Thi thử L1, Phan Châu Trinh, Đà Nẵng, 2018]%[Nguyễn Ngọc Dũng, dự án EX6]%[2D2G5-3]
	Tìm tất cả các giá trị của $m$ để phương trình $9^x+3^{1+x}-m=0$ có nghiệm.
	\choice
	{$m>-\dfrac{81}{4}$}
	{\True $m>0$}
	{$m\geq -\dfrac{81}{4}$}
	{$m\geq 0$}
	\loigiai{
		$9^x+3^{1+x}-m=0\Leftrightarrow \left(3^x\right)^2+3\cdot 3^x-m=0$. \hfill $(1)$\\
		Đặt $t=3^x; (t> 0)$. Khi đó: $(1)\Leftrightarrow t^2+3t-m=0$. \hfill $(2)$\\
		Ta có $\Delta =9+4m$.\\
		$(1)$ có nghiệm $\Leftrightarrow (2)$ có nghiệm $t> 0$. Ta xét các trường hợp sau:
		\begin{itemize}
			\item $\Delta =0\Leftrightarrow m=-\dfrac{9}{4}$. Phương trình $(2)$ có nghiệm kép $t_1=t_2=-\dfrac{1}{2}<0$.\\
			Suy ra $m=-\dfrac{9}{4}$ không thỏa bài toán.
			\item $\Delta >0\Leftrightarrow m>-\dfrac{9}{4}$. Phương trình $(2)$ có hai nghiệm phân biệt $t_1,t_2$.\\
			Do $t_1+t_2=-3<0$ nên $(2)$ không thể có hai nghiệm dương.
			Ta xét trường hợp $t_1,t_2$ trái dấu, nghĩa là $t_1\cdot t_2<0 \Leftrightarrow -m<0 \Leftrightarrow m>0$.\\
			Kết hợp điều kiện $m>-\dfrac{9}{4}$ ta được $m>0$ thỏa bài toán.
		\end{itemize}
		Kết luận $m>0$ là giá trị cần tìm của $m$.
	}
\end{ex}
