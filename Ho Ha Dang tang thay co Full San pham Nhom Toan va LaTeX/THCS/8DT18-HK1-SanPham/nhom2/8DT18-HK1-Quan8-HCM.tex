\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 8, TP.HCM]%[Nguyễn Cao Cường, 8DT18]%[8D1B3]%[8D2B5]%[8D2B6]
	Thực hiện phép tính
	\begin{enumerate}
		\item $2x(1-2x)+(x-1)^2$.
		\item $\dfrac{x-4}{2x+5}+\dfrac{x+9}{2x+5}$, $\left(x\neq -\dfrac{5}{2}
		\right)$.
		\item $\left(\dfrac{x+1}{x}-\dfrac{x}{x-1}\right)\cdot (x-1)$, $\left(x\neq 1; x\neq 0\right)$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $2x(1-2x)+(x-1)^2=2x-4x^2+x^2-2x+1=-3x^2+1$.
	\item $\dfrac{x-4}{2x+5}+\dfrac{x+9}{2x+5}= \dfrac{x-4+x+9}{2x+5}=\dfrac{2x+5}{2x+5}=1$.
	\item $\left(\dfrac{x+1}{x}-\dfrac{x}{x-1}\right)\cdot (x-1) =\left(\dfrac{x^2-1-x^2}{x(x-1)}\right)\cdot(x-1)=-\dfrac{1}{x(x-1)}(x-1)=-
	\dfrac{1}{x}.$
\end{enumerate}
}
\end{bt}
\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 8, TP.HCM]%[Nguyễn Cao Cường, 8DT18]%[8D1B9]
	Phân tích các đa thức sau thành nhân tử
	\begin{enumerate}
		\item $x^3-6x^2+9x$.
		\item $4x^2-9$.
		\item $3x+3y+x^2-y^2$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $x^3-6x^2+9x=x(x^2-6x+9)=x(x-3)^2$.
	\item $4x^2-9=(2x)^2-3^2=(2x-3)(2x+3)$.
	\item $3x+3y+x^2-y^2=3(x+y)+(x-y)(x+y)=(x+y)(3+x-y)$.
\end{enumerate}
}
\end{bt}
\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 8, TP.HCM]%[Nguyễn Cao Cường, 8DT18]%[8D5B5]
	Trong dịp khai trương một cửa hàng điện máy, mỗi chiếc máy lạnh 	đều được giảm $7\%$ so với giá niêm yết. Nhân dịp này, một khách hàng mua một chiếc máy
	lạnh có giá niêm yết là $8 100 000$ đồng. Hỏi khách hàng đó đã trả bao nhiêu tiền?
	\loigiai{
	Số tiền khách hàng được giảm là $8 100 000\cdot 7\%=567 000$ đồng.\\
	Số tiền khách hàng phải trả là $8 100 000 -567 000= 7 533 000$ đồng.\\
Đáp số: Số tiền khách hàng phải trả là $7 533 000$ đồng.
}
\end{bt}
\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 8, TP.HCM]%[Nguyễn Cao Cường, 8DT18]%[8H5B4]
	\immini{
	Một miếng đất hình thang vuông (xem hình bên)
	được dùng để trồng rau và trồng hoa. Phần diện tích có dạng hình chữ nhật để trồng hoa, phần còn lại để trồng rau. Tính 	diện tích trồng rau, biết diện tích đất trồng hoa là $40 \text{m}^2$.}{
	\begin{tikzpicture}[scale=0.6]
	\tkzDefPoints{2/4/A, 6/6/B, 6/4/C, 6/1/D, 2/1/E}

	\tkzDrawPoints(A,B,C,D,E)
	\draw (A)--(B)--(D)--(E) (E)--(A)--(C);
	\tkzLabelPoints[below](E,D)
	\tkzLabelPoints[right](C)
	\tkzLabelPoints[above](A,B)
	\tkzDrawPolygon[color=black,line width =1pt,pattern=dots,
	pattern color=gray, opacity=.4](A,C,D,E)
	\draw (1.3,2.5) node {$5\text{m}$};
	\draw (6.7,5.1) node {$3\text{m}$};
	\end{tikzpicture}}
	\loigiai{
	Xét hình chữ nhật trồng hoa $AEDC$ có diện tích là $40\;\text{m}^2$. Gọi $x$ là độ dài $AC$, theo công thức diện tích hình chữ nhật, ta có
	$$5\cdot x=40 \Rightarrow x=8\;\text{m}.$$
	Vậy diện tích phần trồng rau có hình tam giác là
	$$S=\dfrac{1}{2}\cdot AC\cdot CB=\dfrac{1}{2}\cdot 8\cdot 3=12\;\text{m}^2.$$
Đáp số: Diện tích đất trồng rau là $12\;\text{m}^2$.

}
\end{bt}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 8, TP.HCM]%[Nguyễn Cao Cường, 8DT18]%[8DHB3]
	\immini{
	Một con kiến bò trên cạnh của một miếng gỗ
	hình tam giác vuông (xem hình) bắt đầu từ vị trí điểm $A$ đến vị trí
	điểm $B$ với vận tốc $0,25$ cm/s trong thời gian $1$ phút $20$ giây, sau đó
	kiến bò tiếp đến vị trí ở điểm $C$ với vận tốc $0,2$ cm/s, nhưng sau khi bò
	được $1$ phút con kiến đó đã ở điểm $D$ (chính giữa cạnh $BC$) và kiến đổi
	hướng bò về $A$ với con đường ngắn nhất. Tính quãng đường mà kiến
	bò được từ lúc xuất phát đến lúc về lại điểm $A$.}{
	\begin{tikzpicture}[scale=0.6]
	\tkzDefPoints{1/1/A, 1/6/B, 5/1/C}
	\coordinate (D) at ($(C)!0.5!(B)$);
	\tkzDrawPoints(A,B,C,D)
	\draw (A)--(B)--(C) (C)--(A)--(D);
	\tkzLabelPoints[below](A,C)
	\tkzLabelPoints[above right](D)
	\tkzLabelPoints[above](B)
	\tkzMarkRightAngles[size=0.5](B,A,C)
	\tkzMarkSegments[size=3,mark=s|,color=violet](B,D D,C)
	\end{tikzpicture}}
	\loigiai{
		Ta có $1$ phút $20$ giây = $80$ giây; $1$ phút =$60$  giây.\\
	Độ dài quãng đường $AB$
	$$S_{AB}=0{,}25\cdot 80=20\;\text{cm}.$$
	Độ dài quãng đường $BD$
	$$S_{BD}=0{,}2\cdot 60=12\;\text{cm}.$$
	Vì $D$ là trung điểm $BC$ nên $BC=2BD=24$ cm.
	Xét tam giác vuông $ABC$ có $D$ là trung điểm $BC$ nên $AD$ là đường trung tuyến nên $AD=\dfrac{1}{2}BC=12$ cm.\\
	Con kiến bò từ $D$ về $A$ ngắn nhất khi nó bò trên đường thẳng $AD$. Khi đó quãng đường mà kiến đã bò là
	$$S=AB+BD+DA=20+12+12=44\;\text{cm}.$$
Đáp số: $44$ cm.
}
\end{bt}
\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 8, TP.HCM]%[Nguyễn Cao Cường, 8DT18]%[6D1K9]
	Một miếng đất hình chữ nhật có chiều dài $90$ m và chiều rộng là $50$m.	Người ta chia miếng đất ra thành những miếng đất nhỏ hình vuông để trồng từng loại rau trên từng miếng hình vuông đó. Hỏi có ít nhất bao nhiêu hình vuông được chia?
	\loigiai{
	Vì miếng đất được chia ra có hình vuông, tức là $4$ cạnh bằng nhau và số hình vuông là ít nhất nên khi đó độ dài cạnh hình vuông là ước chung lớn nhất của chiều dài và chiều rộng miếng đất hình chữ nhật lớn. Ta có
	$$\text{ƯCLN}(90,50)= 10.$$
	Vậy hình vuông được chia ra có cạnh $10$ cm. \\
	Diện tích miếng đất hình chữ nhật ban đầu là 
	$$S_{hcn}=90\cdot 50= 4500\text{m}^2.$$
	Diện tích mỗi hình vuông được chia ra
	$$S_{hv}=10\cdot 10=100\;\text{m}^2.$$
	Vậy số miếng đất hình vuông được chia ra là
	$$4500:100=45.$$
	Đáp số: $45$ hình vuông.

}
\end{bt}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 8, TP.HCM]%[Nguyễn Cao Cường, 8DT18]%[8H1B9]
	Cho $\triangle ABC$ vuông tại $B$ ($BC < AB$). Gọi $M$ là trung điểm của $AC$.
	Gọi $D$ là điểm đối xứng với $B$ qua $M$.
	\begin{enumerate}
		\item Chứng minh Tứ giác $ABCD$ là hình chữ nhật.
		\item Gọi $E$ là điểm đối xứng với $A$ qua $B$. Chứng minh tứ giác $DCEB$ là hình bình hành.
		\item Gọi $I$ là giao điểm của $BC$ và $EM$. Chứng minh $AD = 3BI$.
	\end{enumerate}
\loigiai{
\immini{
\begin{enumerate}
	\item Xét tứ giác $ABCD$ có $CM=MA$ và $BM=MD$ nên $ABCD$ là hình bình hành. \\
	Lại có $\widehat{CBA}=90^{\circ}$ nên $ABCD$ là hình chữ nhật.
	\item Ta có $EB=BA$ (do $E$ là điểm đối xứng với $A$ qua $B$).\hfill(1)\\ Do $ABCD$ là hình chữ nhật nên $BA=CD$.\hfill(2)
	\\Từ (1) và (2), suy ra $EB=BA=CD$.\\
	Vậy $EB=CD$ và $EB\parallel CD$ nên $EBDC$ là hình bình hành.
	\item Ta có $DA=CB$ vì $ABCD$ là hình chữ nhật.\hfill(3)\\
	Xét tam giác $ECA$ có $I$ là giao điểm hai đường trung tuyến nên $I$ là trọng tâm tam giác $ECA$. Do đó $CB=3BI$. \\
	Kết hợp (3), ta có $AD=CB=3BI$.
\end{enumerate}
}{
\begin{tikzpicture}[scale=0.6]
\tkzDefPoints{9/2/A, 5/2/B, 5/7/C, 9/7/D, 1/2/E}
\tkzInterLL(A,C)(B,D)    \tkzGetPoint{M}
\tkzInterLL(E,M)(B,C)    \tkzGetPoint{I}
\tkzDrawPoints(A,B,C,D,E,I,M)
\draw (B)--(A)--(D)--(C)--(E)--(B) (B)--(M)--(E) (A)--(C) (C)--(B)--(D);
\tkzLabelPoints[below](E,B,A)
\tkzLabelPoints[above left](I)
\tkzLabelPoints[above](C,D)
\tkzLabelPoints[right](M)
\tkzMarkRightAngles[size=0.5](A,B,C)
\tkzMarkSegments[size=3,mark=s||,color=violet](B,M M,D)
\tkzMarkSegments[size=3,mark=s|,color=violet](C,M M,A)
\tkzMarkSegments[size=3,mark=x,color=violet](E,B B,A)
\end{tikzpicture}}
}
