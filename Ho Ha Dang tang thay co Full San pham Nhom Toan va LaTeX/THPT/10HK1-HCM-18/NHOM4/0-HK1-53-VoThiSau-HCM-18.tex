\begin{bt}%[HK1, 2018, THPT Võ Thị Sáu, TP HCM]%[Đức Nguyễn]%[0D3K2-5]
Tìm  $m$ để phương trình $x^2-(3m-2)x+m+1=0$ có hai nghiệm $x_1$, $x_2$ thỏa mãn $x_1=3x_2$.
\loigiai{Phương trình đã cho là phương trình bậc hai với $\Delta=\left[-\left(3m-2\right)\right]^2-4\cdot\left(m+1\right)=9m^2-16m$.\\
Phương trình có hai nghiệm $x_1$, $x_2\Leftrightarrow \Delta \geq 0\Leftrightarrow 9m^2-16m\geq 0.\quad (*)$\\
Khi đó, áp dụng định lý Vi-ét ta có : $\heva{&x_1+x_2=3m-2\quad (1)\\&x_1\cdot x_2=m+1.\quad (2)}$\\
Theo giả thiết $x_1=3x_2$, thay vào $(1)$ và $(2)$ ta được \\
$\heva{&4x_2=3m-2\\&3x_2^2=m+1}\Leftrightarrow \heva{&x_2=\dfrac{3m-2}{4}\\&3x_2^2=m+1}\Rightarrow 3\left(\dfrac{3m-2}{4}\right)^2=m+1\Leftrightarrow 27m^2-52m-4=0\Leftrightarrow \hoac{&m=-\dfrac{2}{27}\\&m=2.}$\\
Cả hai giá trị của $m$ đều thỏa điều kiện $(*)$.\\
Vậy giá trị cần tìm của $m$ là $m=2$ hoặc $m=-\dfrac{2}{27}$.	

}
\end{bt}

\begin{bt}%[HK1, 2018, THPT Võ Thị Sáu, TP HCM]%[Đức Nguyễn]%[0D3K2-2]
	Giải phương trình $\dfrac{\left|2x-1\right|}{x+1}=\dfrac{3}{4}\left|x\right|$.
	\loigiai{Điều kiện của phương trình : $x\neq -1$.\\
	Phương trình đã cho tương đương với $\dfrac{\left|2x-1\right|}{x+1}-\dfrac{3}{4}\left|x\right|=0.\quad (1)$\\
	Bảng phá dấu giá trị tuyệt đối của phương trình $(1)$ như sau
	\begin{center}
		\begin{tikzpicture}
		\tkzTabInit[lgt=2,espcl=3.75]%
		{$x$/1,$\vert x \vert$/1, $\vert 2x-1 \vert $/1, $VT$/2}%
		{$-\infty$,$0$,$\dfrac{1}{2}$,$+\infty$}%
		\tkzTabLine{ , -x , z , x , t , x , }
		\tkzTabLine{ , 1-2x , t , 1-2x, z, 2x-1, }
		\tkzTabLine{ , \dfrac{1-2x}{x+1}+\dfrac{3}{4}x ,t, \dfrac{1-2x}{x+1}-\dfrac{3}{4}x, t, \dfrac{2x-1}{x+1}-\dfrac{3}{4}x, }
		\end{tikzpicture}
	\end{center}
Ta xét các trường hợp sau:
\begin{itemize}
		\item Nếu $x<0$ thì $(1)\Leftrightarrow \dfrac{1-2x}{x+1}+\dfrac{3}{4}x=0\Leftrightarrow \dfrac{3x^2-5x+4}{4(x+1)}=0$. Trường hợp này vô nghiệm vì phương trình $3x^2-5x+4=0$ vô nghiệm.
	\item Nếu $0\leq x<\dfrac{1}{2}$ thì $(1)\Leftrightarrow \dfrac{1-2x}{x+1}-\dfrac{3}{4}x=0\Leftrightarrow -\dfrac{3x^2+11x-4}{4(x-1)}=0\Rightarrow 3x^2+11x-4=0\Leftrightarrow \hoac{&x=-4\mbox{ (loại)}\\&x=\dfrac{1}{3}\mbox{ (nhận).}}$
	\item Nếu $x \geq\dfrac{1}{2}$ thì $(1)\Leftrightarrow \dfrac{2x-1}{x+1}-\dfrac{3}{4}x=0\Leftrightarrow \dfrac{-3x^2+5x-4}{4(x-1)}=0$. Trường hợp này vô nghiệm.
\end{itemize}
Kết hợp với điều kiện của bài toán suy ra nghiệm của phương trình đã cho là $x=\dfrac{1}{3}$.
	}
\end{bt}

\begin{bt}%[HK1, 2018, THPT Võ Thị Sáu, TP HCM]%[Đức Nguyễn]%[0D3K2-4]
	Giải phương trình $\left(x+1\right)\left(x+4\right)-3\sqrt{x^2+5x+2}=6$.
	\loigiai{Điều kiện của phương trình: $x^2+5x+2\geq 0$.\\
		Phương trình đã cho tương đương với phương trình $x^2+5x+4-3\sqrt{x^2+5x+2}=6.\quad (1)$\\
		Đặt $t=\sqrt{x^2+5x+2}$ ($t\geq 0$). Phương trình $(1)$ trở thành \\
		$t^2+2-3t=6\Leftrightarrow t^2-3t-4=0\Leftrightarrow \hoac{&t=-1\mbox{ (loại)}\\&t=4\mbox{(nhận)}.}$\\
		Với $t=4\Rightarrow \sqrt{x^2+5x+2}=4\Leftrightarrow x^2+5x+2=16\Leftrightarrow x^2+5x-14=0\Leftrightarrow \hoac{&x=-7\\&x=2.}$\\
		Cả hai nghiệm trên đều thỏa mãn điều kiện của phương trình.\\
		Vậy phương trình đã cho có nghiệm là $x=-7$, $x=2$.		
	}
\end{bt}

\begin{bt}%[HK1, 2018, THPT Võ Thị Sáu, TP HCM]%[Đức Nguyễn]%[0D3B2-3]
	Giải phương trình $\dfrac{\left(x+4\right)\sqrt{x+3}}{x^2}=\dfrac{\sqrt{x+3}}{6-x}.\quad (1)$
	\loigiai{Điều kiện của phương trình: $\heva{&x\neq 0\\&6-x\neq 0\\&x+3\geq 0}\Leftrightarrow \heva{&x\geq -3\\&x\neq 0\\&x\neq 6.}$\\
		Với điều kiện trên ta có
		\begin{eqnarray*}
		(1)&\Rightarrow& (x+4)(6-x)\sqrt{x+3}=x^2\sqrt{x+3}\Leftrightarrow \sqrt{x+3}\left(2x^2-2x-24\right)=0\\
		&\Leftrightarrow& \hoac{&\sqrt{x+3}=0\\&2x^2-2x-24=0}\Leftrightarrow \hoac{&x=-3\\&x=4.} 
		\end{eqnarray*}
		Đối chiếu với điều kiện suy ra nghiệm của phương trình đã cho là $x=-3$, $x=4$.
	}
\end{bt}
\begin{bt}%[HK1, 2018, THPT Võ Thị Sáu, TP HCM]%[Đức Nguyễn]%[0H2B2-1]
Cho tam giác $ABC$ có $AB=3$ cm, $AC=2\sqrt{2}$ cm, $\widehat{BAC}=135^\circ$. Tính tích vô hướng $\vec{BA}\cdot \vec{BC}$.
	\loigiai{$\vec{BA}\cdot \vec{BC}=\vec{BA}\left(\vec{BA}+\vec{AC}\right)=\vec{BA}^2-\vec{AB}.\vec{AC}=BA^2-\vec{AB}.\vec{AC}$.\\
	Ta có	$\vec{AB}\cdot \vec{AC}=\left|\vec{AB}\right|\cdot \left|\vec{AC}\right|\cdot \cos \left(\vec{AB},\vec{AC}\right)=AB\cdot AC\cdot \cos \widehat{BAC}=3\cdot 2\sqrt{2}\cdot\cos 135^\circ=-6$.\\
	Vậy $\vec{BA}\cdot \vec{BC}=BA^2-\vec{AB}.\vec{AC}=3^2+6=15$.	
	}
\end{bt}

\begin{bt}%[HK1, 2018, THPT Võ Thị Sáu, TP HCM]%[Đức Nguyễn]%[0H2B3-1]
	Cho tam giác $ABC$ có $AB=3a$, $BC=7a$, $AC=5a$. Tính góc $\widehat{BAC}$ và độ dài đường cao của tam giác hạ từ đỉnh $A$.
	\loigiai{
		\begin{itemize}
			\item $\cos \widehat{BAC}=\dfrac{AB^2+AC^2-BC^2}{2\cdot AB\cdot AC}=\dfrac{(3a)^2+(5a)^2-(7a)^2}{2\cdot 3a\cdot 5a}=-\dfrac{1}{2}\Rightarrow \widehat{BAC}=120^\circ$.
			\item Diện tích tam giác $ABC$ là $S_{ABC}=\dfrac{1}{2}AB\cdot AC\cdot \sin \widehat{BAC}=\dfrac{1}{2}\cdot3a\cdot 5a\cdot \sin 120^\circ=\dfrac{15a^2\sqrt{3}}{4}$.\\
			Gọi $AH$ là đường cao hạ từ đỉnh $A$ của tam giác $ABC$, ta có\\
			$S_{ABC}=\dfrac{1}{2}AH\cdot BC\Rightarrow AH=\dfrac{2S_{ABC}}{BC}=\dfrac{15a\sqrt{3}}{14}$.
		\end{itemize}
	}
\end{bt}

\begin{bt}%[HK1, 2018, THPT Võ Thị Sáu, TP HCM]%[Đức Nguyễn]%[0H2K2-4]
Cho $A\left(1;5\right)$, $B\left(-7;11\right)$. Tìm tọa độ điểm $M$ sao cho $M$, $A$, $B$ thẳng hàng và $MA=3MB$.
	\loigiai{Giả sử $M\left(x;y\right)\Rightarrow \vec{MA}=\left(1-x;5-y\right)$, $\vec{MB}=\left(-7-x;11-y\right)$.\\
		Yêu cầu bài toán $\Leftrightarrow \vec{MA}$, $\vec{MB}$ cùng phương và $MA=3MB\Leftrightarrow \hoac{&\vec{MA}=3\vec{MB}\\&\vec{MA}=-3\vec{MB}.}$
		\begin{itemize}
			\item Với $\vec{MA}=3\vec{MB}\Leftrightarrow \heva{&1-x=3\left(-7-x\right)\\&5-y=3\left(11-y\right)}\Leftrightarrow \heva{&x=-11\\&y=14.}$
			\item Với $\vec{MA}=-3\vec{MB}\Leftrightarrow \heva{&1-x=-3\left(-7-x\right)\\&5-y=-3\left(11-y\right)}\Leftrightarrow \heva{&x=-5\\&y=\dfrac{19}{2}.}$
		\end{itemize}
		Vậy có hai điểm thỏa mãn yêu cầu bài toán là $M_1(-11;14)$ và $M_2\left(-5;\dfrac{19}{2}\right)$.
	}
\end{bt}
