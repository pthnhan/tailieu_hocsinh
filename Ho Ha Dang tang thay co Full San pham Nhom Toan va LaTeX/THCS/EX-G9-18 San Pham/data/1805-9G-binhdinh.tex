\begin{ex}%[Đề thi HSG 9, tỉnh Bình Định, năm 2018]%[Trần Bá Huy, Dự án EX-9G-18]%[8D0K5]
Chứng minh $n^6-2n^4+n^2$ chia hết cho $36$ với mọi $n$ nguyên dương.
\loigiai{
Ta có
\begin{align*}
n^6-2n^4+n^2&=n^2(n^4-2n^2+1)=n^2(n^2-1)^2=n^2(n-1)^2(n+1)^2\\
& =\left[(n-1)n(n+1)\right]^2.
\end{align*}
Vì $(n-1)n(n+1)$ (với $n$ nguyên dương) là tích của $3$ số tự nhiên liên tiếp nên chia hết cho $2$ và $3$. Vì $2$ và $3$ nguyên tố cùng nhau nên $(n-1)n(n+1)$ chia hết cho $2\cdot 3=6$.\\
Từ đó suy ra $n^6-2n^4+n^2=\left[(n-1)n(n+1)\right]^2\ \vdots\ 36$.
}
\end{ex}


\begin{ex}%[Đề thi HSG 9, tỉnh Bình Định, năm 2018]%[Trần Bá Huy, Dự án EX-9G-18]%[8D0G8]
Cho ba số phân biệt $a,\ b,\ c$. Đặt
$$x=(a+b+c)^2-9ab,\ y=(a+b+c)^2-9bc,\ z=(a+b+c)^2-9ac.$$
Chứng minh rằng trong ba số $x,\ y,\ z$ có ít nhất một số dương.
\loigiai{
Với $a,\ b,\ c$ phân biệt, ta có
\begin{align}
& (a+b+c)^2=a^2+b^2+c^2+2(ab+bc+ca).\label{1805.1}\\
& 2a^2+2b^2+2c^2-2ab-2bc-2ca=(a-b)^2+(b-c)^2+(c-a)^2> 0,\notag\\
\Rightarrow\ & a^2+b^2+c^2> ab+bc+ca.\label{1805.2}
\end{align}
Từ \eqref{1805.1} và \eqref{1805.2} suy ra
$$(a+b+c)^2>3(ab+bc+ca).$$
Từ đó
\begin{align*}
x+y+z=\ &3(a+b+c)^2-9(ab+bc+ca)\\
>\ & 3\cdot 3(ab+bc+ca)-9(ab+bc+ca)=0.
\end{align*}
Vì $x+y+z>0$ nên trong ba số $x,\ y,\ z$ phải có ít nhất một số dương.
}
\end{ex}



\begin{ex}%[Đề thi HSG 9, tỉnh Bình Định, năm 2018]%[Trần Bá Huy, Dự án EX-9G-18]%[8D0G6]
Tìm nghiệm nguyên của phương trình $(x-y)(2x+y+1)+9(y-1)=13.$
\loigiai{
Ta có
\begin{align*}
(x-y)(2x+y+1)+9(y-1)-13=\ &2x^2-xy-y^2+x+8y-22\\
=\ & (2x+y-5)(x-y+3)-7.
\end{align*}
Do đó phương trình đã cho tương đương
$$(2x+y-5)(x-y+3)=7.$$
Vì $7=1\cdot 7=7\cdot 1=(-1)(-7)=(-7)(-1)$ nên có $4$ trường hợp xảy ra.
\begin{itemize}
\item {\bf Trường hợp 1.} $\heva{&2x+y-5=1\\ &x-y+3=7}\Leftrightarrow \heva{&2x+y=6\\ &x-y=4}\Leftrightarrow \heva{& x=\dfrac{10}{3}\\ &y=-\dfrac{2}{3}.}$\\
Loại vì không có nghiệm nguyên.
\item {\bf Trường hợp 2.} $\heva{&2x+y-5=7\\ &x-y+3=1}\Leftrightarrow \heva{&2x+y=12\\ &x-y=-2}\Leftrightarrow  \heva{& x=\dfrac{10}{3}\\ &y=\dfrac{16}{3}.}$\\
Loại vì không có nghiệm nguyên.
\item {\bf Trường hợp 3.} $\heva{&2x+y-5=-1\\ &x-y+3=-7}\Leftrightarrow \heva{&2x+y=4\\ &x-y=-10}\Leftrightarrow  \heva{& x=-2\\ &y=8.}$\\
$(x;y)=(-2;8)$ là một nghiệm.
\item {\bf Trường hợp 4.} $\heva{&2x+y-5=-7\\ &x-y+3=-1}\Leftrightarrow \heva{&2x+y=-2\\ &x-y=-4}\Leftrightarrow  \heva{& x=-2\\ &y=2.}$\\
$(x;y)=(-2;2)$ là một nghiệm.
\end{itemize}
Vậy phương trình đã cho có hai nghiệm $(x;y)$ là $(-2;8)$ và $(-2;2)$.
}
\end{ex}



\begin{ex}%[Đề thi HSG 9, tỉnh Bình Định, năm 2018]%[Trần Bá Huy, Dự án EX-9G-18]%[9D0K2]
Giải phương trình $x^2+\sqrt{x+2018}=2018.$
\loigiai{
Điều kiện $x\geq -2018$. Đặt $t=\sqrt{x+2018}\ (t\geq 0)$, thu được
$$\heva{&x^2+t=2018\\ &t^2-x=2018.}$$
Lấy phương trình trên trừ phương trình dưới, ta được
$$x^2-t^2+t+x=0\Leftrightarrow (x+t)(x-t+1)=0\Leftrightarrow \hoac{&x+t=0\\ &x-t+1=0.}$$
\begin{itemize}
\item Xét $t=-x$, thu được phương trình
$$\sqrt{x+2018}=-x\Leftrightarrow \heva{&-x\geq 0\\ &x+2018=x^2}\Leftrightarrow \heva{&x\leq 0\\ &x=\dfrac{1\pm 3\sqrt{897}}{2}}\Leftrightarrow x=\dfrac{1-3\sqrt{897}}{2}.$$
\item Xét $t=x+1$, thu được phương trình
$$\sqrt{x+2018}=x+1\Leftrightarrow \heva{&x+1\geq 0\\ &x+2018=(x+1)^2}\Leftrightarrow \heva{& x\geq -1\\ & x=\dfrac{-1\pm \sqrt{8069}}{1}}\Leftrightarrow x=\dfrac{-1+\sqrt{8069}}{2}.$$
\end{itemize}
Vậy phương trình đã cho có hai nghiệm là $\dfrac{1-3\sqrt{897}}{2}$ và $\dfrac{-1+\sqrt{8069}}{2}$.
}
\end{ex}



\begin{ex}%[Đề thi HSG 9, tỉnh Bình Định, năm 2018]%[Trần Bá Huy, Dự án EX-9G-18]%[9D0G7]
Cho $a,\ b,\ c$ là ba số không âm thỏa mãn điều kiện: $a^2+b^2+c^2\leq 2(ab+bc+ca)$ và $p$, $q$, $r$ là ba số thỏa mãn $p+q+r=0$. Chứng minh rằng $apq+bqr+crp\leq 0$.
\loigiai{
Ta có
{\allowdisplaybreaks
\begin{align*}
& apq+bqr+crp\leq 0\Leftrightarrow -r(bq+cp)-apq\geq 0\\
\Leftrightarrow\ & (p+q)(bq+cp)-apq\geq 0\Leftrightarrow bq^2+cp^2+(b+c-a)pq\geq 0\\
\Leftrightarrow\ & \left(\sqrt{b}|q|-\sqrt{c}|p|\right)^2+2\sqrt{bc}|pq|+(b+c-a)pq\geq 0.
\end{align*}}
Cần chứng minh $2\sqrt{bc}|pq|+(b+c-a)pq\geq 0.  \refstepcounter{equation}\hfill (\theequation)\label{1805.3}$\\
Từ giả thiết bài toán
{\allowdisplaybreaks
\begin{align*}
& a^2+b^2+c^2\leq 2(ab+bc+ca)\Leftrightarrow a^2+b^2+c^2+2(ab+bc+ca)\leq 4(ab+bc+ca)\\
\Leftrightarrow\ & 4bc\geq a^2+b^2+c^2-2ab-2ac+2bc=(b+c-a)^2\\
\Rightarrow\ & 2\sqrt{bc}\geq |b+c-a|\\
\Rightarrow\ & 2\sqrt{bc}|pq|\geq |b+c-a|\cdot |pq|\geq -(b+c-a)pq.
\end{align*}}
Vậy \eqref{1805.3} đúng. Từ đó ta có điều phải chứng minh.
}
\end{ex}



\begin{ex}%[Đề thi HSG 9, tỉnh Bình Định, năm 2018]%[Trần Bá Huy, Dự án EX-9G-18]%[9D0G7]
Cho các số dương $a$, $b$ thỏa mãn điều kiện $ab=1$. Tìm giá trị nhỏ nhất của biểu thức $$M=(a+b+1)(a^2+b^2)+\dfrac{4}{a+b}.$$
\loigiai{
Ta có $$M=(a+b)(a^2+b^2)+(a^2+b^2)+\dfrac{4}{a+b}=\dfrac{(a+b)(a^2+b^2)}{2}+\dfrac{(a+b)(a^2+b^2)}{2}+\dfrac{4}{a+b}+a^2+b^2.$$
Áp dụng bất đẳng thức Cô-si cho $3$ số dương $\dfrac{(a+b)(a^2+b^2)}{2}$, $\dfrac{(a+b)(a^2+b^2)}{2}$ và $\dfrac{4}{a+b}$, ta được
\begin{align*}
M\geq\ & 3\sqrt[3]{\dfrac{(a+b)(a^2+b^2)}{2}\cdot \dfrac{(a+b)(a^2+b^2)}{2}\cdot \dfrac{4}{a+b}}+a^2+b^2\\
=\ & 3\sqrt[3]{(a+b)(a^2+b^2)^2}+a^2+b^2\\
\geq\ & 3\sqrt[3]{2\sqrt{ab}(2ab)^2}+2ab=8.
\end{align*}
Đẳng thức xảy ra khi
$$\heva{& \dfrac{(a+b)(a^2+b^2)}{2}=\dfrac{4}{a+b}\\ & a=b\\ & ab=1}\Leftrightarrow a=b=1.$$
Vậy $\min M=8\Leftrightarrow a=b=1$.
}
\end{ex}

Cho tam giác nhọn $ABC$ có các đường cao $AD$, $BE$, $CF$ và trực tâm là $H$.
\begin{enumerate}

\item Gọi $I,\ J$ lần lượt là trung điểm của $AH$ và $BC$. Đường tròn đường kính $AH$ cắt đoạn thẳng $IJ$ tại $K$. \begin{ex}%[Đề thi HSG 9, tỉnh Bình Định, năm 2018]%[Trần Bá Huy, Dự án EX-9G-18]%[8H0K8]%[8H0K8]
Cho tam giác $ABC$ vuông cân tại $A$. Các điểm $D$, $E$ theo thứ tự di chuyển trên các cạnh $AB$, $AC$ sao cho $BD=AE$. Xác định vị trí điểm $D$, $E$ sao cho
\begin{enumerate}
\item $DE$ có độ dài nhỏ nhất.
\item Tứ giác $BDEC$ có diện tích nhỏ nhất.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Xác định vị trí của $D$, $E$ để đoạn $DE$ có độ dài nhỏ nhất.
\immini{
Đặt $AB=a$, $BD=AE=x$ với $a\geq x\geq 0,\ a>0$. Áp dụng định lí Pi-ta-go trong tam giác vuông $ADE$, ta có\\
$DE^2=AD^2+AE^2=(a-x)^2+x^2$.\\
Tiếp tục áp dụng bất đẳng thức $a^2+b^2\geq \dfrac{(a+b)^2}{2}$, thu được
}{
\begin{tikzpicture}
\clip (-0.5,-1) rectangle (5.5,3);
\tkzDefPoints{0/0/B, 5/0/C}
\coordinate (H) at ($(B)!0.5!(C)$);
\path (B)--(H)--([turn]90:1) coordinate (h);
\tkzInterLC(H,h)(H,B) \tkzGetSecondPoint{A}
\coordinate (D) at ($(B)!0.4!(A)$);
\coordinate (E) at ($(A)!0.4!(C)$);
\draw (A)--(B)--(C)--cycle (D)--(E);
\tkzDrawPoints(B,C,A,D,E)
\tkzLabelPoints[above](A)
\tkzLabelPoints[below](B,C)
\tkzLabelPoints[above left](D)
\tkzLabelPoints[above right](E)
\tkzMarkRightAngle[size=0.3](B,A,C)
\end{tikzpicture}
}
$DE^2=(a-x)^2+x^2\geq \dfrac{\left[(a-x)+x\right]^2}{2}=\dfrac{a^2}{2}$.\\
Từ đó suy ra $\min DE=\dfrac{a}{\sqrt{2}}\Leftrightarrow a-x=x\Leftrightarrow x=\dfrac{a}{2}$. \\
Vậy $DE$ nhỏ nhất khi $D$ và $E$ lần lượt là trung điểm của $AB$ và $AC$.
\item Xác định vị trí của $D$, $E$ để tứ giác $BDEC$ có diện tích nhỏ nhất.\\
Tứ giác $BDEC$ có diện tích nhỏ nhất khi và chỉ khi tam giác $ADE$ có diện tích lớn nhất.\\
Ta có $$S_{\text{ADE}}=\dfrac{1}{2}AD\cdot AE=\dfrac{1}{2}\cdot (a-x)x\leq \dfrac{1}{2}\cdot \left(\dfrac{a-x+x}{2}\right)^2=\dfrac{a^2}{8}.$$
Đẳng thức xảy ra khi và chỉ khi $a-x=x\Leftrightarrow x=\dfrac{a}{2}$.\\
Vậy $\max S_{\text{ADE}}=\dfrac{a^2}{8}\Leftrightarrow x=\dfrac{a}{2}$, hay tứ giác $BDEC$ có diện tích nhỏ nhất khi $D$ và $E$ lần lượt là trung điểm của $AB$ và $AC$.
\end{enumerate}
}
\end{ex}
