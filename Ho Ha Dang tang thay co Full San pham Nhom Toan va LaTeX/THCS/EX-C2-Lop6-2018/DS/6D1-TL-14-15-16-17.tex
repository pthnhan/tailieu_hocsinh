\section{PHÂN TÍCH MỘT SỐ RA THỪA SỐ NGUYÊN TỐ}
\subsection{Tóm tắt lý thuyết}
\begin{enumEX}[1)]{1}
	\item Số nguyên tố là số tự nhiên lớn hơn $1$, chỉ có hai ước là $1$ và chính nó. \\
	$\bullet$ Hợp số là số tự nhiên lớn hơn $1$, có nhiều hơn hai ước. \\
	$\bullet$ Số $2$ là số nguyên tố nhỏ nhất, là số nguyên tố chẵn duy nhất.
	\item Phân tích một số tự nhiên lớn hơn $1$ ra thừa số nguyên tố là viết số đó dưới dạng một tích các thừa số nguyên tố.\\
	$\bullet$ Có thể phân tích một số ra thừa số nguyên tố \lq \lq theo cột dọc\rq \rq \, hoặc \lq \lq theo hàng ngang\rq \rq \, và viết các ước nguyên tố theo thứ tự từ nhỏ tới lớn.\\
	$\bullet$ Dù phân tích ra thừa số nguyên tố theo cách nào thì ta cũng được cùng một kết quả.
\end{enumEX}
\subsection{Các dạng toán}
\begin{dang}{Nhận biết số nguyên tố, hợp số}
	\textit{\textbf{Phương pháp giải:}}\\
	$\bullet$ Dựa vào định nghĩa của số nguyên tố, hợp số.\\
	$\bullet$ Dựa vào các dấu hiệu chia hết.\\
	$\bullet$ Có thể dùng bảng số nguyên tố nhỏ hơn $1000$.
\end{dang}
\begin{vd}%[6D1K15]
	Trong các số sau, số nào là số nguyên tố, số nào là hợp số? 
	\begin{center}
		$0;1;87;73;1675;547.$
	\end{center}
	\loigiai{
	$\bullet$ số $0$ và $1$ không phải là số nguyên tố, không phải là hợp số.\\
	$\bullet$ Số $87$ là hợp số vì $87>1$ và $87 \,\vdots\, 3$ (ngoài $1$ và chính nó).\\
	$\bullet$ Số $1675$ là hợp số vì $1675>1$ và $1675 \,\vdots\, 5$ (ngoài $1$ và chính nó).\\
	$\bullet$ Số $73$ là số nguyên tố vì $73>1$ và $73$ chỉ chia hết cho $1$ và chính nó. \\
	$\bullet$ Số $547$ là số nguyên tố (vì có trong bảng các số nguyên tố nhỏ hơn $1000$).
	}
\end{vd}
\begin{vd}%[6D1K15]
	Chứng minh rằng tích của hai số nguyên tố là một hợp số.
	\loigiai{
	Gọi $p_1$ và $p_2$ là hai số nguyên tố. Xét tích $p_1 \times p_2$, tích này lớn hơn $1$, chia hết cho $1$ và chính nó. Ngoài ra tích này còn chia hết cho $p_1$ và $p_2$ nên tích $p_1 \times p_2$ là hợp số.	
}
\end{vd}
\begin{vd}
	Tổng $S = 5 \cdot 6 \cdot 7 + 10 \cdot 11 \cdot 13$ là số nguyên hay hợp số?
	\loigiai{
	Số $5 \cdot 6 \cdot 5 \,\vdots\, 5$ (vì tích này có một thừa số là $5$). \\
	Số $10 \cdot 11 \cdot 13 \,\vdots\, 5$ (vì $10 \,\vdots\, 5$).\\
	Do đó tổng $S = 5 \cdot 6 \cdot 7 + 10 \cdot 11 \cdot 13$ chia hết cho $5$.\\
	Tổng $S>5$ và chia hết cho $5$ nên nó là hợp số.
}
\end{vd}
\begin{dang}{Điền chữ số để được số nguyên tố hay hợp số}
	\textbf{\textit{Phương pháp giải}}\\
	$\bullet$ Dùng các dấu hiệu chia hết để xét. \\
	$\bullet$ Có thể dùng bảng số nguyên tố nhỏ hơn $1000$.
\end{dang}
\begin{vd}%[6D1K15]
	Cho số $\overline{10*}$. Điền chữ số thích hợp vào $*$ để được:
	\begin{enumEX}[a)]{2}
		\item	Hợp số.
		\item	Số nguyên tố.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}[a)]{1}
			\item Với $\overline{10*}$ ta có thể chọn $*\in \{0;2;4;6;8\}$ để $\overline{10*}$ chia hết cho $2$, có thể chọn $*$ là $5$ để $\overline{10*}$ chia hết cho $5$.\\
			Vậy để cho $\overline{10*}$ là hợp số ta có thể chọn $* \in \{0;2;4;6;8;5\}$.
			\item Các số $101$; $103$; $107$; $109$ đều là số nguyên tố (dùng bảng số nguyên tố nhỏ hơn $1000$).
			Vậy để $\overline{10*}$ là số nguyên tố, ta chọn $*\in \{1;3;5;7;9\}$.
		\end{enumEX}
}
\end{vd}
\begin{vd}%[6D1B15]
	Cho biết $\overline{x1}$ là một số nguyên tố. Hỏi chữ số $x$ là chữ số nào?
	\loigiai{
	Các số $11$; $31$; $41$; $71$ đều là số nguyên tố, các số $21$; $51$; $81$; $91$ đều là hợp số do đó $x\in \{1;3;4;6;7\}$.	
}
\end{vd}
\begin{vd}%[6D1G15]
	Tìm $k \in \mathbb{N}$ để tích $19 \cdot k$ là số nguyên tố.
	\loigiai{
	$\bullet$ Với $k=0$ thì $19 \cdot k = 0$, số $0$ không phải là số nguyên tố.\\
	$\bullet$ Với $k=1$ thì $19 \cdot k = 19$, số $19$ là số nguyên tố. \\
	$\bullet$ Với $k \geq 2$ thì $19 \cdot k$ là hợp số vì ngoài các ước là $1$ và chính nó còn có ước là $19$.\\
	Vậy với $k=1$ thì $19 \cdot k$ là số nguyên tố.
}
\end{vd}
\begin{dang}{Phân tích một số ra thừa số nguyên tố}
	\textbf{\textit{Phương pháp giải}}\\
	\textit{Cách 1.} Phân tích theo cột dọc.\\
	\textit{Cách 2.} Phân tích theo hàng ngang.\\
	Nên nhớ:	$10 = 2 \cdot 5$; $100 = 2^2 \cdot 5^2$; $1000 = 2^3 \cdot 5^3$; $10^n = 2^n \cdot 5^n$.
\end{dang}

\begin{vd}%[6D1B15]
	Phân tích các số sau ra thừa số nguyên tố:
	\begin{enumEX}[a)]{2}
		\item 1500;
		\item 40000.
	\end{enumEX}
	\loigiai{
	\begin{enumEX}[a)]{1}
		\item $1500 = 15 \cdot 100 = 3 \cdot 5 \cdot \left(2^2 \cdot 5^2\right)= 2^2 \cdot 3 \cdot 5^3$.
		\item $40000 = 4 \cdot 10000 = 2^2 \cdot \left(2^4 \cdot 5^4\right) = 2^6 \cdot 5^4$.
	\end{enumEX}
}
\end{vd}
\begin{dang}{Phân tích một số ra thừa số nguyên tố để tìm các ước của một số, để tính số lượng các ước số của số đó}
	\textbf{\textit{Phương pháp giải}} \\
	$\bullet$ Khi phân tích số $m$ ra thừa số nguyên tố, giả sử $m = a \cdot b$. Lúc đó ta được các ước của $m$ là $1$, $a$, $b$ và $a \cdot b$.\\
	$\bullet$ Khi phân tích số $m$ ra thừa số nguyên tố:
	\begin{itemize}
		\item  Nếu $m = a^x$ thì $m$ có $x+1$ ước.
		\item  Nếu $m = a^x \cdot b^y$ thì $m$ có $(x+1)\cdot(y+1)$ ước.
		\item  Nếu $m = a^x \cdot b^y \cdot c^z$ thì $m$ có $(x+1) \cdot (y+1) \cdot (z+1)$ ước.
	\end{itemize}
\end{dang}
\begin{vd}%[6D1K15]
	Tìm các ước của mỗi số sau:
	\begin{enumEX}[a)]{3}
		\item $33$;
		\item $81$;
		\item $45$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}[a)]{1}
			\item  $33 = 3 \cdot 11$ $\Rightarrow \text{Ư}(33) = \{1;3;11;33\}$.
			\item  $81=3^4$ $\Rightarrow \text{Ư}(81)= \{1;3;3^2;3^3;3^4\} = \{1;3;9;27;81\}$.
			\item  $45=3^2 \cdot 5$ $\Rightarrow \text{Ư}(45) = \{1;3;9;5;15;45\}$.
		\end{enumEX}
	}
\end{vd}
\begin{vd}%[6D1K15]
	Mỗi số sau có bao nhiêu ước?
	\begin{enumEX}[a)]{2}
		\item  $200$;
		\item  $720$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}[a)]{1}
		\item $200 = 2 \cdot 100 = 2^3 \cdot 5^2$. \\
		Số lượng các ước của $200$ là $(3+1)\cdot(2+1)= 12$ (ước).
		\item $720 = 9 \cdot 9 \cdot 10 = 2^3 \cdot 3^2 \cdot 5$. \\
		Số lượng các ước của $720$ l à $(4+1) \cdot (2+1) \cdot (1+1) = 30$ (ước).
		\end{enumEX}
	}
\end{vd}

\begin{vd}%[6D1G15]

	}

		\item $5 \cdot 8 - 3 \cdot 13 = 1$. Nên $5 \cdot 8 - 3 \cdot 13 = 1$ không phải là số nguyên tố, không phải là hợp số.

}

\section{ƯỚC CHUNG LỚN NHẤT}
\subsection{Tóm tắt lý thuyết}
	\begin{enumEX}[1)]{1}
		\item Ước chung lớn nhất (ƯCLN) của hai hay nhiều số là số lớn nhất trong tập hợp các ước chung của các số đó.
		\item Muốn tìm Ước chung lớn nhất của hai hay nhiều số lớn hơn 1, ta thực hiện ba bước sau:\\
		\textit{Bước 1:} Phân tích mỗi số ra thừa số nguyên tố. \\
		\textit{Bước 2:} Chọn ra các thừa số nguyên tố chung.\\
		\textit{Bước 3:} Lập tích các thừa số đã chọn, mỗi thừa số lấy với số mũ nhỏ nhất của nó. Tích đó là ƯCLN phải tìm.
		\begin{note} \textbf{\textit{Chú ý}}\\
		$\bullet$Nếu các số đã cho không có thừa số nguyên tố chung thì ƯCLN của chúng bằng $1$. Hai hay nhiều số có ƯCLN bằng $1$ gọi là các số nguyên tố cùng nhau.\\
		$\bullet$ Trong các số đã cho, nếu số nhỏ nhất là ước của các số còn lại thì ƯCLN của các số đã cho chính là số nhỏ nhất đó.
		\end{note}
		\item Cách tìm ước chung thông qua ƯCLN. \\
		Để tìm ước chung của các số đã cho, ta có thể tìm các ước của ƯCLN của các số đó.
	\end{enumEX}
\subsection{Các dạng toán}
\begin{dang}{Tìm ƯCLN của hai hay nhiều số}
	\textbf{\textit{Phương pháp giải}}\\ Thực hiện theo ba bước trong quy tắc tìm ƯCLN.
\end{dang}
\begin{vd}%[6D1K17]
	Tìm ƯCLN của các số sau:
	\begin{enumEX}[a)]{2}
		\item $108$ và $240$;
		\item $450$; $1260$ và $945$.
	\end{enumEX}
	\loigiai{
	\begin{enumEX}[a)]{1}
		\item $108 = 2^2 \cdot 3^2$ \\
		$240 = 2^4 \cdot 3 \cdot 5$. \\
		Vậy ƯCLN$(108;240) = 2^2 \cdot 3 = 12$.
		\item $450 =2 \cdot 3^2 \cdot 5^2$;\\
		$1260 = 2^2 \cdot 3^2 \cdot 5 \cdot 7$; \\
		$945 = 3^3 \cdot 5 \cdot 7$. \\

\begin{vd}%[6D1G17]
	Chứng tỏ rằng hai số lẻ liên tiếp là hai số nguyên tố cùng nhau.
	\loigiai{
	Gọi hai số lẻ liên tiếp là $2n+1$ và $2n+3$ $(n \in \mathbb{N})$. \\
	Giả sử $d$ là ƯCLN của chúng. \\
	Ta có $(2n+1) \, \vdots \, d$ và $(2n+3) \, \vdots \, d$.\\
	Do đó $[(2n+1)-(2n+3)] \, \vdots \, d$ hay $2 \, \vdots \, d$, suy ra $d \in \{1;2\}$.\\
	Vì $d$ là ước của số lẻ nên $d \ne 2$. \\
	Vậy $d=1$, do đó hai số $2n+1$ và $2n+3$ nguyên tố cùng nhau, tức là hai số lẻ liên tiếp nguyên tố cùng nhau.
}
\end{vd}
\begin{vd}%[6D1G17]
	Cho hai số tự nhiên $2n+1$ và $4n+3$ $(n \in \mathbb{N})$. Chứng tỏ rằng hai số này nguyên tố cùng nhau.
	\loigiai{
	Gọi $d$ là ƯCLN$(2n+1;2n+3)$. \\
	Suy ra $(4n+3) \, \vdots \, d$, $(2n+1)\, \vdots \, d \Rightarrow 2\cdot(2n+1) \, \vdots \, d$.	\\
	Do đó $\left[ (4n+3)- 2 \cdot (2n+1) \right] \, \vdots \, d$ hay $(4n+3-4n-2) \, \vdots \, d$, tức là $1\, \vdots \, d$, do đó $d = 1$. \\
	Vậy hai số $2n+1$ và $4n+3$ nguyên tố cùng nhau.
}
\end{vd}
