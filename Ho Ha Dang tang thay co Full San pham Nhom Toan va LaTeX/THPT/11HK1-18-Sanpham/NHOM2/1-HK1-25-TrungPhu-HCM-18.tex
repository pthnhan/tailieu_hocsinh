\begin{bt}[HK1 lớp 11 THPT Trung Phú-HCM , 2018][Võ Tấn Đạt, dự án 11HK1-18][1D1B3-3]
	Giải phương trình: $ \sqrt{3}\sin 2x - \cos 2x = 2\cos x $.
	\loigiai{
		Chia hai vế cho $ 2 $ ta có
	$$ \dfrac{\sqrt{3}}{2}\sin 2x - \dfrac{1}{2}\cos 2x = \cos x \Leftrightarrow \sin 2x\cdot \cos \dfrac{\pi}{6} - \cos 2x\cdot\sin \dfrac{\pi}{6} = \sin\left(\dfrac{\pi}{2} -x \right).  $$
		Áp dụng công thức cộng ta thu được
		$$ \sin\left(2x - \dfrac{\pi}{6} \right) =  \sin\left(\dfrac{\pi}{2} -x \right) \Leftrightarrow \hoac{&2x - \dfrac{\pi}{6} = \dfrac{\pi}{2} -x + k2\pi\\
			&2x - \dfrac{\pi}{6} = \pi - \dfrac{\pi}{2} + x + k2\pi.} $$
		Giải ra ta được họ nghiệm của phương trình là $ \hoac{&x = \dfrac{2\pi}{9} + \dfrac{k2\pi}{3}\\&x = \dfrac{2\pi}{3} + k2\pi},\ k\in \mathbb{Z} $.
	}
\end{bt} 
\begin{bt}[HK1 lớp 11 THPT Trung Phú-HCM , 2018][Võ Tấn Đạt, dự án 11HK1-18][1D1B3-1]
	Giải phương trình: $ 2\sin^2x + 5\cos x + 1= 0 $.
	\loigiai{
		Sử dụng công thức cơ bản $ \sin^2x = 1-\cos^2x $ ta có
		$$ 2(1-\cos^2x) + 5\cos x  +1= 0 \Leftrightarrow -2\cos^2x + 5\cos x + 3 = 0 \Leftrightarrow \hoac{&\cos x = 3\\&\cos x = -\dfrac{1}{2}.} $$
		Rõ ràng loại trường hợp $ \cos x = 3 $ vì ta có $ -1 \le \cos x \le 1, \forall x\in \mathbb{R} $.\\
		Giải $ \cos x = -\dfrac{1}{2} $ ta thu được họ nghiệm là $ x= \pm \dfrac{2\pi}{3} + k2\pi,\ k\in \mathbb{Z} $.
	}
\end{bt}
\begin{bt}[HK1 lớp 11 THPT Trung Phú-HCM , 2018][Võ Tấn Đạt, dự án 11HK1-18][1D2B3-2]
	Tìm hệ số của số hạng chứa $ x^{12} $ trong khai triển nhị thức Niu-tơn $ \left(x^2 - \dfrac{2}{x} \right)^{18}  $.
	\loigiai{
		Ta có công thức số hạng tổng quát là
		$$ T_{k+1} = \mathrm{C}_n^k a^{n-k} b^k = \mathrm{C}_{18}^k\left(x^2\right)^{18-k}\left(\dfrac{-2}{x} \right)^k = \mathrm{C}_{18}^k x^{36-3k}(-2)^k.  \quad  \quad  ({ \text{ Điều kiện } 0 \le k \le 18.) }    $$
		Số hạng chứa $ x^{12} $ sẽ thỏa $ 36 - 3k = 12 \Leftrightarrow k = 8 $.\\
		Vậy hệ số của số hạng chứa $ x^{12} $ là $ \mathrm{C}_{18}^8(-2)^8 = 11202048$.

	}
\end{bt}

\begin{bt}[HK1 lớp 11 THPT Trung Phú-HCM , 2018][Võ Tấn Đạt, dự án 11HK1-18][1D3K1-1]
	Chứng minh rằng với mọi $n \in \mathbb{N}^*$, ta có $1^2+2^2+3^2+\cdots+n^2=\dfrac{n(n+1)(2n+1)}{6}. \quad (*)$
	\loigiai{Với $n=1$ thì $\heva{&VT{(*)}=1\\&VP{(*)}=1}$.\\
		Vậy $(*)$ đúng với $n=1 \quad (1)$.\\
		Giả sử $(*)$ đúng với $n=k, k\in \mathbb{N}^*$, ta có	$1^2+2^2+3^2+\cdots+k^2=\dfrac{k(k+1)(2k+1)}{6} $.\\
		Ta chứng minh $(*)$ đúng với $n=k+1$, nghĩa là $1^2+2^2+3^2+\cdots+(k+1)^2=\dfrac{(k+1)(k+2)(2k+3)}{6}$.\\
		Thật vậy\\ $\begin{aligned}
		1^2+2^2+3^2+\cdots+(k+1)^2&=1^2+2^2+3^2+\cdots+k^2+(k+1)^2\\&=\dfrac{k(k+1)(2k+1)}{6}+(k+1)^2\\&=\dfrac{(k+1)[k(2k+1)+6(k+1)]}{6}\\&=\dfrac{(k+1)(2k^2+7k+6)}{6}\\&=\dfrac{(k+1)(k+2)(2k+3)}{6}.
		\end{aligned}$
		\\
		Do đó 	$(*)$ đúng với $n=k+1 \quad (2)$.\\
		Từ $(1)$ và $(2)$ suy ra $(*)$ đúng với mọi $n \in \mathbb{N}^*$.
	}
\end{bt}

		Với $q=-2 \Rightarrow u_1=1.$\\

	}

\begin{bt}[HK1 lớp 11 THPT Trung Phú-HCM , 2018][Võ Tấn Đạt, dự án 11HK1-18][1D2K2-1] 
	Một hộp có $6$ bi đỏ, $7$ bi xanh, $8$ bi vàng. Có bao nhiêu cách chọn ra bốn bi biết rằng bốn bi có đủ ba màu.
	\loigiai{
	Để chọn được bồn bi có đủ ba màu ta sẽ có các trường hợp:
		\begin{itemize}
			\item Trường hợp $1$\\
			Chọn $2$ bi đỏ, $1$ bi xanh, $1$ bi vàng, số cách chọn là $\mathrm{C}_{6}^2 \cdot \mathrm{C}_{7}^1 \cdot \mathrm{C}_{8}^1=840$.
			\item Trường hợp $2$\\
			Chọn $1$ bi đỏ, $2$ bi xanh, $1$ bi vàng, số cách chọn là $\mathrm{C}_{6}^1 \cdot \mathrm{C}_{7}^2 \cdot \mathrm{C}_{8}^1=1008$.
			\item Trường hợp $3$\\
			Chọn $1$ bi đỏ, $1$ bi xanh, $2$ bi vàng, số cách chọn là $\mathrm{C}_{6}^1 \cdot \mathrm{C}_{7}^1 \cdot \mathrm{C}_{8}^2=1176$.
		\end{itemize}
		Vậy số cách chọn được bốn bi có đủ ba màu là $840+1008+1176=3024$.
	}

\begin{bt}[HK1 lớp 11 THPT Trung Phú-HCM , 2018][Võ Tấn Đạt, dự án 11HK1-18][1D2K5-2]
	Có hai cái hộp: hộp thứ nhất chứa $3$ bi xanh và $2$ bi đỏ, hộp thứ hai chứa $5$ bi vàng. Biết rằng các bi hoàn toàn giống nhau chỉ khác màu. Chọn ngẫu nhiên từ mỗi hộp $3$ bi. Tính xác suất để chọn được $6$ bi có đủ ba màu.

		\end{enumerate}

	}

\begin{bt}[HK1 lớp 11 THPT Trung Phú-HCM , 2018][Võ Tấn Đạt, dự án 11HK1-18][1D2K2-6]
	Số đường chéo của đa giác lồi $n$ cạnh bằng $405$. Tìm $n$?
	\loigiai{Nhận xét đa giác lồi $n$ cạnh nên có $n$ đỉnh. Do bất kỳ $2$ đỉnh khác nhau nào cũng tạo thành một đoạn thẳng (bao gồm các đường chéo và các cạnh của đa giác) nên số đường chéo của đa giác $n$ cạnh này là $\mathrm{C}^2_n-n$. Theo đề bài ta có
		$$\mathrm{C}^2_n-n=405\Leftrightarrow \dfrac{n!}{2!(n-2)!}-n=405\Leftrightarrow \dfrac{n(n-1)}{2}-n=405\Leftrightarrow n^2-3n-810=0\Rightarrow n=30.$$
		Vậy $n=30$.
	}
\end{bt}

	Giải phương trình sau $3\sin^3 2x+4\cos^3 2x=3\sin 2x$.
	\loigiai{
	$	\begin{aligned}
		3\sin^3 2x+4\cos^3 2x=3\sin 2x&\Leftrightarrow 3\sin^32x-3\sin 2x+4\cos^32x=0\\
		&\Leftrightarrow 3\sin 2x(\sin^22x-1)+4\cos^32x=0\\
		&\Leftrightarrow 4\cos^32x-3\sin 2x\cos^2 2x=0\\
		&\Leftrightarrow \cos^2 2x\left(4\cos 2x-3\sin 2x\right)=0\\
		&\Leftrightarrow \hoac{&\cos 2x=0\\&4\cos 2x-3\sin 2x=0}\Leftrightarrow \hoac{&\cos 2x=0\\&\tan 2x=\dfrac{4}{3}}\\
		&\Leftrightarrow \hoac{&2x=\dfrac{\pi}{2}+k\pi\\&2x=\arctan \dfrac{4}{3}+k\pi}\Leftrightarrow \hoac{&x=\dfrac{\pi}{4}+\dfrac{k\pi}{2}\\&x=\dfrac{1}{2}\arctan \dfrac{4}{3}+\dfrac{k\pi}{2}}, \quad k\in \mathbb{Z}.
		\end{aligned}$
\\ Vậy phương trình đã cho có các nghiệm $\hoac{&x=\dfrac{\pi}{4}+\dfrac{k\pi}{2}\\&x=\dfrac{1}{2}\arctan \dfrac{4}{3}+\dfrac{k\pi}{2}}, \quad k\in \mathbb{Z}.$ 	}
