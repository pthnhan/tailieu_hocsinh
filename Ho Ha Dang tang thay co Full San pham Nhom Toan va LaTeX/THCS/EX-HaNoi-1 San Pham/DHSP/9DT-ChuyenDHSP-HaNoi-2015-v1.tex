\begin{ex}%[ChuyenSPHN-2015]%[9D5K2]
Cho biểu thức 
$P=\dfrac{\left ( \dfrac{a}{b} +\dfrac{b}{a} +1\right )\left (\dfrac{1}{a}-\dfrac{1}{b}\right )^2}{\dfrac{a^2}{b^2} + \dfrac{b^2}{a^2} - \dfrac{a}{b} - \dfrac{b}{a}}$ với $a>0$, $b>0$, $a \neq b$. 
\begin{enumerate}
\item[a)] Chứng minh rằng $P=\dfrac{1}{ab}$.
\item[b)] Giả sử $a,b$ thay đổi sao cho $4a+b+\sqrt{ab} =1$. Tìm giá trị nhỏ nhất của $P$. 
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a)] Ta có $\dfrac{a^2}{b^2} + \dfrac{b^2}{a^2} - \dfrac{a}{b} - \dfrac{b}{a} = \left( \dfrac{a}{b} + \dfrac{b}{a} \right )^2 - \left( \dfrac{a}{b} + \dfrac{b}{a} \right ) -2 = \left ( \dfrac{a}{b} + \dfrac{b}{a} +1 \right ) \left( \dfrac{a}{b} + \dfrac{b}{a} -2\right )$ \\
$= \left ( \dfrac{a}{b} + \dfrac{b}{a} +1 \right ) \dfrac{(a-b)^2}{ab}$; 
$\left (\dfrac{1}{a}-\dfrac{1}{b}\right )^2 = \dfrac{(a-b)^2}{(ab)^2}$. Thay vào biểu thức ta có $P=\dfrac{1}{ab}$. 
\item[b)] Đặt $z=4a$, $P= \dfrac{4}{zb}$,  $4a+b+\sqrt{ab} =1 \Leftrightarrow z+b+\dfrac{1}{2}\sqrt{zb} =1$. \\
Ta có $1=z+b+\dfrac{1}{2}\sqrt{zb} \geq 2\sqrt{zb} + \dfrac{1}{2}\sqrt{zb} = \dfrac{5}{2}\sqrt{zb} \Rightarrow zb \leq \dfrac{4}{25}$. Vậy $P_{\min} =25$ khi $a=\dfrac{1}{10}$ và $b=\dfrac{2}{5}$. 
\end{enumerate}
}
\end{ex}



\begin{ex}%[ChuyenSPHN-2015]%[9D3K3]
Cho hệ phương trình  $\heva{&x-my=2-4m\\&mx+y=3m+1}$ với $m$ là tham số.
\begin{enumerate}
\item[a)] Giải hệ phương trình khi $m=2$.
\item[b)] Chứng minh hệ luôn có nghiệm với mọi giá trị của m. Giả sử $(x_0, y_0)$ là một nghiệm của của hệ phương trình, chứng minh rằng $ x_0^2 + y_0^2 - 5 ( x_0 + y_0 ) + 10 = 0 $
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a)] $\heva{&x-2y=-6\\&2x+y=7} \Leftrightarrow \heva{&5x=8\\& y=7-2x} \Leftrightarrow \heva{&x= \dfrac{8}{5}\\&y=\dfrac{19}{5}}$ 
\item[b)] $\heva{&x-my=2-4m\\&mx+y=3m+1} \Leftrightarrow \heva{&(m^2+1)x = 3m^2-3m+2\\&y=3m+1-mx}$\\
Dễ thấy $m^2+1 >0 $ với mọi $m$ cho nên hệ luôn có nghiệm $\heva{&x= \dfrac{3m^2-3m+2}{m^2+1}\\&y=\dfrac{4m^2+m+1}{m^2+1}}$. \\
$x_0,y_0$ là nghiệm của hệ suy ra $\heva{&x_0-my_0=2-4m\\&mx_0+y_0=3m+1} \Rightarrow \heva{&x_0^2+m^2y_0^2 -2mx_0y_0=(2-4m)^2\\&m^2x_0^2+y_0^2+2mx_0y_0=(3m+1)^2}$ \\
$\Rightarrow (m^2+1)(x_0^2+y_0^2) = (2-4m)^2+(3m+1)^2 \Rightarrow x_0^2+y_0^2 = \dfrac{25 m^2 - 10 m + 5}{m^2+1}$ \\
$x_0^2 + y_0^2 - 5 ( x_0 + y_0 ) + 10 = \dfrac{25 m^2 - 10 m + 5 -5(3m^2-3m+2 +  4m^2+m+1) + 10(m^2+1) }{m^2+1} = 0$. 
\end{enumerate}
}
\end{ex}

\begin{ex}%[ChuyenSPHN-2015]%[9D4K3]
Cho $a, b$ là các số thực khác  $0$. Biết rằng phương trình   $a(x - a)^2 + b(x - b)^2 = 0$
có nghiệm duy nhất. Chứng minh rằng $|a| = |b|$. 
\loigiai{
$a(x - a)^2 + b(x - b)^2 = 0 \Leftrightarrow (a+b)x^2 -2(a^2+b^2)x + a^3+b^3=0$. \\
Nếu $a+b=0$ phương trình có nghiệm duy nhất $x=0$. \\
Nếu $a+b \neq 0$,  phương trình có nghiệm duy nhất khi  $\Delta' = (a^2+b^2)^2-(a+b)(a^3+b^3) = 0 \Leftrightarrow -ab(a-b)^2=0 \Leftrightarrow a=b$. 
}
\end{ex}

\begin{ex}%[ChuyenSPHN-2015]%[9H3K3]%[9H3K7]
Cho tam giác ABC có các góc $ABC$ và góc $ACB$ nhọn,  góc $BAC = 60^\circ$. Các đường phân giác trong $BB_1, CC_1$ của tam giác $ABC$ cắt nhau tại $I$.
\begin{enumerate}
\item[a)] Chứng minh tứ giác $AB_1IC_1$ nội tiếp.
\item[b)] Gọi $K$ là giao điểm thứ hai khác $B$ của đường thẳng $BC$ với đường tròn ngoại tiếp tam giác $BC_1I$. Chứng minh tứ giác $CKIB_1$ nội tiếp. 
\item[c)]  Chứng minh   $AK$ vuông góc $B_1C_1$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzDefPoints{1.5/3/A,-1/-3/B,6/-3/C}
\tkzInCenter(A,B,C) \tkzGetPoint{I}
\tkzInterLL(A,B)(C,I) \tkzGetPoint{C1}
\tkzInterLL(A,C)(B,I) \tkzGetPoint{B1}
\tkzCircumCenter(B,I,C1) \tkzGetPoint{O1}
\tkzInterLC(B,C)(O1,B) \tkzGetPoints{K}{K1}
\pgfresetboundingbox
\tkzDrawSegments(A,B B,C C,A B1,C1 A,K B,B1 C,C1 I,K C1,K A,I)
\tkzDrawCircle(O1,B)
\tkzLabelPoints[above](A)
\tkzLabelPoints[right=4pt](I)
\tkzLabelPoints[below left=-2pt](B)
\tkzLabelPoints[below right=-2pt](K,C)
\tkzLabelPoint[above left=-2pt](C1){$C_1$}
\tkzLabelPoint[above right=-2pt](B1){$B_1$}
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item[a)] $\widehat{B_1IC_1} = \widehat{BIC} = 180^\circ - \dfrac{\widehat{B} +\widehat{C}}{2} = 180^\circ - 60^\circ =120^\circ$ suy ra $\widehat{B_1IC_1} + \widehat{A} =180^\circ$. 
\item[b)] $\widehat{B_1IC_1} = \widehat{BIC}=120^\circ \Rightarrow \widehat{BIC_1} = \widehat{CIB_1}=\widehat{C_1KB}=\widehat{A}=60^\circ $ suy ra tứ giác $ACKC_1$ nội tiếp, mặt khác theo a)  tứ giác $AB_1IC_1$ nội tiếp cho nên $BK \cdot BC = BC_1 \cdot BA=BI \cdot BB_1$ suy ra tứ giác  
$CKIB_1$ nội tiếp. 
\item[c)] Tứ giác $ACKC_1$ và $AB_1IC_1$ nội tiếp suy ra $\widehat{KAC_1} = \dfrac{\widehat{C}}{2}$, $\widehat{AC_1B_1} =\widehat{AIB_1} = \dfrac{\widehat{A} + \widehat{B}}{2}$, $\widehat{KAC_1} + \widehat{AC_1B_1} = \dfrac{\widehat{A} + \widehat{B} + \widehat{C}}{2} =90^\circ$. 
\end{enumerate}
}
\end{ex}


\begin{ex}%[ChuyenSPHN-2015]%[9D5K2]
Tìm các số thực không âm $a$ và $b$ thỏa mãn \[\left( a^2 + b + \dfrac{3}{4} \right) \left( b^2 + a + \dfrac{3}{4} \right) = \left( 2a + \dfrac{1}{2}\right) \left( 2b + \dfrac{1}{2} \right)\]
\loigiai{
Ta có $a^2+\dfrac{1}{4} \geq 2\sqrt{a^2 \cdot \dfrac{1}{4}} =a$; $b^2+\dfrac{1}{4} \geq 2\sqrt{b^2 \cdot \dfrac{1}{4}} =b$. \\
$\left( a^2 + b + \dfrac{3}{4} \right) \left( b^2 + a + \dfrac{3}{4}\right) \geq \left(a+b+\dfrac{1}{2}\right)^2 =a^2+b^2+2ab+a+b+\dfrac{1}{4} \geq 4ab +a+b+\dfrac{1}{4} = \left( 2a + \dfrac{1}{2}\right) \left( 2b + \dfrac{1}{2} \right)$, dấu bằng xảy ra khi $a=b=\dfrac{1}{2}$. 
}
\end{ex}
