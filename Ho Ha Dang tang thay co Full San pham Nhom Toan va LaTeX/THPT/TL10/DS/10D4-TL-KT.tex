\begin{bt}%[Lê Thanh Nin]%[0D4B3]
	\textbf{(2 điểm)} Giải các bất phương trình sau: 
	\begin{enumerate}
		\item $ 8x-5>\dfrac{15x-8}{2} $.
		\item $ \dfrac{1-3x}{1+2x}\leq-2 $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $ 8x-5>\dfrac{15x-8}{2}\Leftrightarrow 16x-10>15x-8 \Leftrightarrow x>2 $.\\
			Vậy tập nghiệm của bất phương trình là: $ \text{T}=(2;+\infty) $
			\item $ \dfrac{1-3x}{1+2x}\leq-2 \Leftrightarrow \dfrac{1-3x}{1+2x}+2\leq 0 \Leftrightarrow \dfrac{x+3}{1+2x}\leq 0 \Leftrightarrow -3 \leq x < -\dfrac{1}{2}  $.\\
			Vậy tập nghiệm của bất phương trình là : $ \text{T}=\left[-3;-\dfrac{1}{2} \right)  $
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Lê Thanh Nin]%[0D4K5]
	\textbf{(2 điểm)} Giải bất phương trình $ x^2-x+\left| 3x-2\right| >0 $.
	\loigiai{
		
		Bất phương trình tương đương: \\
		$ \hoac {& \heva{&3x-2\geq 0\\& x^2+2x-2>0 }\\& \heva{&3x-2< 0\\& x^2-4x+2>0 }}  \Leftrightarrow 	 \hoac {& \heva{&x\geq \dfrac{2}{3}\\& x<-1-\sqrt{3}  \, \, \text{hoặc} \, \,  x>-1+\sqrt{3}}\\ & \heva{& x< \dfrac{2}{3}\\& x<2-\sqrt{2}  \, \, \text{hoặc} \, \,  x>2+ \sqrt{2} }} \Leftrightarrow \hoac{&x>-1+\sqrt{3}\\&x<2-\sqrt{2}}$.\\
		Vậy tập nghiệm của bất phương trình là: $ \text{T}=\left(-\infty;2-\sqrt{2}\right) \cup \left( -1+\sqrt{3};+\infty\right) $
	}
\end{bt}


\begin{bt}%[Lê Thanh Nin]%[0D4K5]
	\textbf{(4 điểm)} Cho biểu thức $f(x)=(m+1)x^2-2(2m+1)x+1$  ($ m $ là tham số)
	\begin{enumerate}
		\item Tìm các giá trị $ m $ để phương trình $f(x)=0$ có hai nghiệm dương phân biệt. 
		\item Tìm các giá trị $ m $ để bất phương trình $f(x)>0$ có nghiệm đúng $\forall x\in \mathbb{R}$. 
	\end{enumerate}
	\loigiai{
		
		\begin{enumerate}
			\item Xét phương trình: $(m+1)x^2-2(2m+1)x+1=0$ (*).\\
			Phương trình (*) có hai nghiệm dương phân biệt \\
			$\Leftrightarrow \heva{ &m+1\neq 0\\ & \Delta '>0 \\  & P>0 \\ 
				& S>0 } \Leftrightarrow \heva{ & 4m^2+3m>0 \\  & \dfrac{1}{m+1}>0 \\ 
				& \dfrac{2(2m+1)}{m+1}>0 } \Leftrightarrow \heva{ & m<\dfrac{-3}{4} \,\, \text{hoặc} \, \, m>0 \\  & m>-1 \\ 
				& m<-1\,\, \text{hoặc} \, \, m>\dfrac{-1}{2}} \Leftrightarrow m>0.$
			
			\item Xét bất phương trình: $(m+1)x^2-2(2m+1)x+1>0$ (**).\\
			TH1: Nếu $m=-1$ thì (**)$\Leftrightarrow 2x+1>0\Leftrightarrow x>\dfrac{-1}{2}$ không có nghiệm đúng $\forall x\in \mathbb{R}$\\
			TH2: Nếu $m\ne -1$ (**) có nghiệm đúng $\forall x\in \mathbb{R}$\\
			$\Leftrightarrow \heva{ & a>0 \\  & \Delta '<0 \\  } \Leftrightarrow \heva{ & m+1>0 \\  & 4m^2+3m<0 \\  } \Leftrightarrow \heva{ & m>-1 \\  & \dfrac{-3}{4}<m<0 \\  } \Leftrightarrow \dfrac{-3}{4}<m<0.$ 
			
		\end{enumerate}
	}
\end{bt}


\begin{bt}%[Lê Thanh Nin]%[0D4G1]
	\textbf{(2 điểm)} Chứng minh rằng $ 2a^2+b^2+c^2\geq 2a(b+c) $ với mọi $ a,b,c \in \mathbb {R} $.
	\loigiai{
		
		Ta có $ (a-b)^2 \geq 0 $ với mọi $ a,b \in \mathbb {R} $ nên $ a^2 + b^2\geq 2ab $ (1).\\
		Tương tự ta có: $ a^2 + c^2\geq 2ac $ (2) với mọi $ a,b \in \mathbb{R} $.\\
		Cộng từng vế (1) và (2) ta được $ 2a^2+b^2+c^2\geq 2a(b+c) $ với mọi $ a,b,c \in \mathbb {R} $.\\
		Đẳng thức xảy ra khi và chỉ khi với mọi $ a=b=c $.
	}
\end{bt}

\begin{bt}%[Lê Thanh Nin]%[0D4K5]

\begin{bt}%[Lê Thanh Nin]%[0D4K5]

	\end{enumerate}
	\loigiai{
		
		\begin{enumerate}

			$\Leftrightarrow \heva{&m-1\neq 0\\ & \Delta '>0 \\  & P>0 \\ 
				& S>0   } \Leftrightarrow \heva{ & 4m^2+5m>0 \\  & \dfrac{-1}{m-1}>0 \\

				& m<\dfrac{-1}{2}\, \, \text{hoặc} \, \,m>1  } \Leftrightarrow m<\dfrac{-5}{4}.$
			
			
			\item Xét bất phương trình: $(m-1)x^2-2(2m+1)x-1<0$ (**)\\

			$\Leftrightarrow \heva{ & a<0 \\  & \Delta '<0 \\  } \Leftrightarrow \heva{ & m-1<0 \\  & 4m^2+5m<0 \\  } \Leftrightarrow \heva{ & m<1 \\  & \dfrac{-5}{4}<m<0 \\  } \Leftrightarrow \dfrac{-5}{4}<m<0$. 
			
			
		\end{enumerate}
	}
\end{bt}


\begin{bt}%[Lê Thanh Nin]%[0D4G1]
	\textbf{(2 điểm)} Cho $a, b, c>0$. Chứng minh rằng 
	$ab+\dfrac{a}{b}+\dfrac{b}{a}\geq a+b+1$.
	\loigiai{
		
		Áp dụng bất đẳng thức Côsi  cho hai số không âm:
		\begin{align*}
		ab+\dfrac{a}{b}&\geq 2\sqrt{ab. \dfrac{a}{b}}=2a.\\ 
		\dfrac{a}{b}+\dfrac{b}{a}&\geq 2\sqrt{\dfrac{a}{b}. \dfrac{b}{a}}=2. \\
		\dfrac{b}{a}+ab&\geq 2\sqrt{\dfrac{b}{a}. ab}=2b. \\
		\Rightarrow 2(ab+\dfrac{a}{b}+\dfrac{b}{a})&\geq 2(a+b+1)\Leftrightarrow ab+\dfrac{a}{b}+\dfrac{b}{a}\geq a+b+1. 
		\end{align*}
		Đẳng thức sảy ra khi và chỉ khi $ a=b=1 $.
	}
\end{bt}

	}
