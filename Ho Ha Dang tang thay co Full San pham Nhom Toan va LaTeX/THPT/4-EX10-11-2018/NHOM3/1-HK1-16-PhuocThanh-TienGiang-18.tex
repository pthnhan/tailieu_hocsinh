			\item  $\left(\sin 2x+\cos 2x\right)\cos x+2\cos 2x-\sin x=0$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item $2\cos^27x-5\cos 7x-7=0\Leftrightarrow \hoac{
				&\cos 7x=-1\;\;(N)  \\
				&\cos 7x=\dfrac{7}{2}\quad(L)  \\
			}$\\
			$\cos 7x=-1\Leftrightarrow x=\dfrac{\pi}{7}+k\dfrac{2\pi}{7},k\in \mathbb{Z}.$
			
			\item $\left(\sin 2x+\cos 2x\right)\cos x+2\cos 2x-\sin x=0$ \\
			$\Leftrightarrow \sin x\left(2\cos^2x-1\right)+\cos 2x\left(\cos x+2\right)=0$\\
			$\Leftrightarrow \cos 2x\left(\sin x+\cos x+2\right)=0\Leftrightarrow \hoac{
				& \cos 2x=0  \\
				& \sin x+\cos x+2=0  \\
			}$\\
			$\Leftrightarrow x=\dfrac{\pi}{4}+k\dfrac{\pi}{2},k\in \mathbb{Z}.$ 
			
		\end{enumerate}
	}
\end{bt}
