\section{LIÊN HỆ GIỮA PHÉP CHIA VÀ PHÉP KHAI PHƯƠNG}
\subsection{Kiến thức cần nhớ}
    	\begin{dl}
    	Với $ A\ge 0, B>0 $ thì $ \sqrt{\dfrac{A}{B}}=\dfrac{\sqrt{A}}{\sqrt{B}}. $
    \end{dl}   
\subsection{Các dạng toán}
\begin{dang}{Khai phương một thương}
	\textbf{Quy tắc.} Muốn khai phương một thương $ \dfrac{A}{B} $ của hai biểu thức $ A\ge 0, B>0 $, ta có thể khai phương lần lượt biểu thức bị chia $ A $ và biểu thức chia $ B $. Sau đó lấy kết quả thứ nhất chia cho kết quả thứ hai.
\end{dang}
\begin{vd}
	Tính:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $ A=\sqrt{\dfrac{49}{81}} $;
			\item $ B=\dfrac{\sqrt{3+\sqrt{5}}}{\sqrt{2}} .$
		\end{enumerate}
	\end{multicols}
\loigiai{\begin{enumerate}
		\item Ta có $A=\sqrt{\dfrac{49}{81}}=\dfrac{\sqrt{49}}{\sqrt{81}}=\dfrac{7}{9}.$
		\item Ta có \[B=\sqrt{\dfrac{3+\sqrt{5}}{2}}=\sqrt{\dfrac{6+2\sqrt{5}}{4}}=\dfrac{\sqrt{5+2\sqrt{5}+1}}{\sqrt{4}}=\dfrac{\sqrt{\left(\sqrt{5}+1\right)^2}}{2}=\dfrac{\sqrt{5}+1}{2}.\]
\end{enumerate}}
\end{vd}

\begin{vd}
	Rút gọn các biểu thức:
	\begin{multicols}{2}
    \begin{enumerate}
    \item $ A=\sqrt{\dfrac{a^2}{b}}\cdot \sqrt{\dfrac{a^6}{b^3}} $, với $ b>0 $;
    \item $ B=b^5\sqrt{\dfrac{a^2+6a+9}{b^8}}. $
    \end{enumerate}
	\end{multicols}
\loigiai{
\begin{enumerate}
	\item Ta sử dụng quy tắc nhân hai căn bậc hai rồi biến đổi tiếp \[A=\sqrt{\dfrac{a^2}{b}}\cdot \sqrt{\dfrac{a^6}{b^3}}=\sqrt{\dfrac{a^2}{b}\cdot\dfrac{a^6}{b^3}}=\sqrt{\dfrac{a^8}{b^4}}=\dfrac{\sqrt{a^8}}{\sqrt{b^4}}=\dfrac{a^4}{b^2}.\]
	\item Ta biến đổi
	\begin{align*}
		B&=b^5\cdot\sqrt{\dfrac{\left(a+3\right)^2}{b^8}}=b^5\cdot\dfrac{\sqrt{\left(a+3\right)^2}}{\sqrt{b^8}}=b^5\cdot\dfrac{|a+3|}{b^4}=b\cdot|a+3|\\
		&=\heva{b\cdot(a+3)&\text{ nếu } a\ge -3\\-b\cdot(a+3)&\text{ nếu } a<-3.}
	\end{align*}
\end{enumerate}
}
\end{vd}

\begin{dang}{Chia các căn bậc hai}
    \textbf{Phương pháp giải:} \\
    Quy tắc chia hai căn thức bậc hai: Muốn chia căn bậc hai của biểu thức không âm $ A $ cho căn bậc hai của số dương $ B $, ta có thể chia biểu thức $ A $ cho biểu thức $ B $ rồi lấy căn bậc hai của thương đó.
\end{dang}

\begin{vd}
    Thực hiện phép tính:
    \begin{multicols}{2}
    	\begin{enumerate}
    		\item $ A=\sqrt{98}\colon\sqrt{2} $;
    		\item $ B=\left(\sqrt{48}-\sqrt{27}+\sqrt{3}\right)\colon \sqrt{3} $;
    		\item $ C=\left(5\sqrt{3}+3\sqrt{5}\right)\colon\sqrt{15}. $
    	\end{enumerate}
    \end{multicols}
    \loigiai{
        \begin{enumerate}
        	\item Ta có $ A=\sqrt{98}\colon\sqrt{2}=\sqrt{98\colon 2}=\sqrt{49}=7. $
        	\item Ta có $  B=\left(\sqrt{48}-\sqrt{27}+\sqrt{3}\right)\colon \sqrt{3}=\sqrt{48:3}-\sqrt{27:3}+\sqrt{3:3}=\sqrt{16}-\sqrt{9}+1=2. $
        	\item $C=\left(5\sqrt{3}+3\sqrt{5}\right)\colon\sqrt{15}=\left(5\sqrt{3}+3\sqrt{5}\right):\left( \sqrt{5}\cdot\sqrt{3}\right) =\dfrac{5\sqrt{3}}{\sqrt{5}\cdot\sqrt{3}}+\dfrac{3\sqrt{5}}{\sqrt{5}\cdot\sqrt{3}}=\sqrt{5}+\sqrt{3}.  $
        \end{enumerate}
    }
\end{vd}
\begin{dang}{Rút gọn, tính giá trị biểu thức}
    Ta sử dụng tính chất
    \begin{center}
    	Với $ a \ge 0 $ và $ b >0 $ thì ta có $ \sqrt{\dfrac{a}{b}} = \dfrac{\sqrt{a}}{\sqrt{b}} $.
    \end{center}
    \begin{note}
        Chú ý khi giải bài dạng này phải xét trong điều kiện có nghĩa của biểu thức chứa căn.
    \end{note}
\end{dang}

\begin{vd}
    Rút gọn các biểu thức sau:
    \begin{multicols}{2}
    	\begin{enumerate}
    		\item $ \sqrt{\dfrac{(x-1)^2}{16}} $ với $ x \ge 1 $;
    		\item $ \sqrt{\dfrac{x^4}{(a-1)^2}} $ với $ a < 1 $.
    	\end{enumerate}
    \end{multicols}
    \loigiai{
      \begin{enumerate}
      	\item Vì $ (x-1)^2 \ge 0 $ và $ 16 > 0 $ nên ta có $$ \sqrt{\dfrac{(x-1)^2}{16}} = \dfrac{\sqrt{(x-1)^2}}{\sqrt{16}} = \dfrac{|x-1|}{4} \overset{\text{ vì }x\ge 1}{=} \dfrac{x-1}{4}. $$ 
      	\item Vì $ x^4 \ge 0 $ và $ (a-1)^2 > 0 $ nên ta có
      	\[ \sqrt{\dfrac{x^4}{(a-1)^2}} = \dfrac{\sqrt{x^4}}{\sqrt{(a-1)^2}} = \dfrac{x^2}{|a-1|} \overset{\text{vì }a < 1}{=} \dfrac{x^2}{1-a}. \]
      \end{enumerate}
    }
\end{vd}
\begin{vd}
	Rút gọn các biểu thức sau:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $ \dfrac{\sqrt{27(x-5)^2}}{\sqrt{3}} $ với $ x\ge5 $;
			\item $ \dfrac{\sqrt{(x-4)^4}}{\sqrt{9(x-4)^2}} $ với $ x < 4 $.
		\end{enumerate}
	\end{multicols}
\loigiai{
\begin{enumerate}
	\item Với $ x \ge 5 $ ta có
	\[ \dfrac{\sqrt{27(x-5)^2}}{\sqrt{3}} = \sqrt{\dfrac{27(x-5)^2}{3}} = \sqrt{9(x-5)^2} = \sqrt{9}\cdot\sqrt{(x-5)^2} = 3\cdot|x-5| \overset{\text{ vì } x\ge5}{=} 3(x-5). \]
	\item Với $ x < 4 $ ta có
	\[ \dfrac{\sqrt{(x-4)^4}}{\sqrt{9(x-4)^2}} = \sqrt{\dfrac{(x-4)^4}{9(x-4)^2}} = \sqrt{\dfrac{(x-4)^2}{9}} = \dfrac{\sqrt{(x-4)^2}}{\sqrt{9}} = \dfrac{|x-4|}{3} \overset{\text{ vì } x < 4}{=} \dfrac{4-x}{3}. \]
\end{enumerate}	
}
\end{vd}
\begin{vd}
	Rút gọn biểu thức với điều kiện đã cho và tính giá trị của nó:
	\begin{enumerate}
		\item $ \sqrt{\dfrac{x-2\sqrt{x}+1}{x+2\sqrt{x}+1}} $ với $ x\ge0 $; tính giá trị tại $ x = 4 $.
		\item $ \sqrt{\dfrac{(x-2)^4}{(3-x)^2}} + \dfrac{x^2-1}{x-3} $ với $  x<3 $;  tính giá trị tại $ x = 0{,}5 $.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Với $ x \ge 0 $ ta có
	\[ \sqrt{\dfrac{x-2\sqrt{x}+1}{x+2\sqrt{x}+1}} = \sqrt{\dfrac{(\sqrt{x}-1)^2}{(\sqrt{x}+1)^2}} = \dfrac{\sqrt{(\sqrt{x}-1)^2}}{\sqrt{(\sqrt{x}+1)^2}} = \dfrac{|\sqrt{x}-1|}{|\sqrt{x}+1|} = \dfrac{|\sqrt{x}-1|}{\sqrt{x}+1}. \]
	Thay $ x = 4 $ vào ta có
	\[ \dfrac{|\sqrt{4}-1|}{\sqrt{4}+1} = \dfrac{1}{3}. \]
	\item Với $ x < 3 $ ta có
	 \begin{align*}
	 	\sqrt{\dfrac{(x-2)^4}{(3-x)^2}} + \dfrac{x^2-1}{x-3} = \dfrac{\sqrt{(x-2)^4}}{\sqrt{(3-x)^2}} + \dfrac{x^2-1}{x-3}& = \dfrac{(x-2)^2}{|3-x|} + \dfrac{x^2-1}{x-3}\\ &= \dfrac{x^2-4x+4}{3-x} - \dfrac{x^2-1}{3-x}\\ & = \dfrac{-4x+5}{3-x}.
	 \end{align*}
	Thay $ x = 0{,}5 $ vào ta có
	\[ \dfrac{-4\cdot0{,}5+5}{3-0{,}5} = \dfrac{6}{5}. \]
\end{enumerate}	
}
\end{vd}

\begin{dang}{Giải phương trình}
    Tìm điều kiện của biểu thức có mặt trong phương trình.\\
    Thu gọn trong điều kiện của biến sau đó giải phương trình.
    \begin{note}
    	Chú ý khi giải bài dạng này phải xét điều kiện có nghĩa của biểu thức chứa căn và cuối cùng phải đối chiếu nghiệm tìm được với điều kiện.
    \end{note}
\end{dang}

\begin{vd}
    Tìm $ x $ thỏa mãn điều kiện:
    \begin{multicols}{2}
    	\begin{enumerate}
    		\item $ \sqrt{\dfrac{(x-1)^4}{x^2-2x+1}} = 2 $;
    		\item $ \dfrac{\sqrt{x-1}}{\sqrt{x^2-x}} = 3 $.
    	\end{enumerate}
    \end{multicols}
    \loigiai{
    \begin{enumerate}
    	\item Điều kiện là $ \dfrac{(x-1)^4}{x^2-2x+1} \ge 0 $. Vì $ (x-1)^4 \ge 0 $ với mọi $ x $ nên điều kiện là $ x^2-2x+1 > 0 $ hay $ (x-1)^2 > 0 $ nghĩa là $ x \ne 1 $. Khi đó ta có
    	\[  \sqrt{\dfrac{(x-1)^4}{x^2-2x+1}} = 2 \Leftrightarrow  \sqrt{\dfrac{(x-1)^4}{(x-1)^2}} = 2 \Leftrightarrow \sqrt{(x-1)^2} = 2 \Leftrightarrow |x-1| = 2. \]
    	Ta xét hai trường hợp:
    	\begin{enumerate}[TH1:]
    		\item $ x-1 = 2 $ hay $  x = 3 $;
    		\item $ x-1 = -2 $ hay $ x = -1 $.
    	\end{enumerate}
    Đối chiếu điều kiện ta nhận $ x = 3 $ và $ x = -1 $.
    \item Điều kiện là $ x-1 \ge 0 $ và $ x^2-x > 0 $. Vì $ x^2 - x = x(x-1) $ nên điều kiện sẽ là $ x-1 > 0 $ và $ x > 0 $. Từ đó suy ra điều kiện là $ x > 1 $. Khi đó ta có
    \[ \dfrac{\sqrt{x-1}}{\sqrt{x^2-x}} = 3 \Leftrightarrow \sqrt{\dfrac{x-1}{x^2-x}} = 3 \Leftrightarrow \sqrt{\dfrac{1}{x}} = 3 \Leftrightarrow \sqrt{x} = \dfrac{1}{3}. \]
    Sử dụng định nghĩa của căn bậc hai số học ta có
    \[ x = \dfrac{1}{9}. \]
    Đối chiếu điều kiện ta loại $ x = \dfrac{1}{9} $. Vậy không có giá trị $ x $ thỏa mãn đề bài.
    \end{enumerate}     
    }
\end{vd}
\begin{vd}
	Tìm $ x $ thỏa mãn điều kiện:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $ \sqrt{\dfrac{2x-3}{x-1}} = 2 $;
			\item $ \dfrac{\sqrt{x-2}}{\sqrt{3x-4}} = 1 $.
		\end{enumerate}
	\end{multicols}
\loigiai{
\begin{enumerate}
	\item Điều kiện $ \dfrac{2x-3}{x-1} \ge 0 $, nghĩa là $ x $ thỏa mãn một trong hai trường hợp sau:
	\begin{enumerate}[TH1.]
		\item $ 2x-3 \ge 0 $ và $ x-1 > 0 $, ta tìm được $ x \ge \dfrac{3}{2} $;
		\item $ 2x-3 \le 0 $ và $ x-1 < 0 $, ta tìm được $ x <1 $.
	\end{enumerate}
Khi đó ta có
\[ \sqrt{\dfrac{2x-3}{x-1}} = 2 \Rightarrow \dfrac{2x-3}{x-1} = 4 \Rightarrow 2x-3 = 4x-4 \Rightarrow x = \dfrac{1}{2}. \]
Đối chiếu điều kiện ta thấy $ x = \dfrac{1}{2} $ thỏa mãn.
\item Điều kiện là $ x - 2 \ge 0 $ và $ 3x-4 > 0 $, suy ra $ x \ge 2 $ là điều kiện. Khi đó
\[ \dfrac{\sqrt{x-2}}{\sqrt{3x-4}} = 1 \Leftrightarrow \sqrt{\dfrac{x-2}{3x-4}} = 1 \Rightarrow \dfrac{x-2}{3x-4} = 1 \Rightarrow x = 1. \]
Đối chiếu điều kiện ta loại $ x = 1 $. Vậy không có giá trị $ x $ thỏa mãn đề bài.
\end{enumerate}	
}
\end{vd}

\begin{bt}
Áp dụng quy tắc khai phương một thương, hãy tính
\begin{multicols}{4}
\begin{enumerate}
\item $\sqrt{\dfrac{121}{9}}$;
\item $\sqrt{\dfrac{144}{169}}$;
\item $\sqrt{3\dfrac{6}{25}}$;
\item $\sqrt{4\dfrac{21}{25}}$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{multicols}{2}
\begin{enumerate}
\item $\sqrt{\dfrac{121}{9}}=\dfrac{\sqrt{121}}{\sqrt{9}}=\dfrac{11}{3}$.
\item $\sqrt{\dfrac{144}{169}}=\dfrac{\sqrt{144}}{\sqrt{169}}=\dfrac{12}{13}$.
\item $\sqrt{3\dfrac{6}{25}}=\sqrt{\dfrac{81}{25}}=\dfrac{\sqrt{81}}{\sqrt{25}}=\dfrac{9}{5}$.
\item $\sqrt{4\dfrac{21}{25}}=\sqrt{\dfrac{121}{25}}=\dfrac{\sqrt{121}}{\sqrt{25}}=\dfrac{11}{5}$.
\end{enumerate}
\end{multicols}
}
\end{bt}

\begin{bt}
Áp dụng quy tắc chia hai căn bậc hai, hãy tính
\begin{multicols}{4}
\begin{enumerate}
\item $\dfrac{\sqrt{999}}{\sqrt{444}}$;
\item $\dfrac{\sqrt{160}}{\sqrt{0{,}4}}$;
\item $\dfrac{\sqrt{9+6\sqrt{2}}}{\sqrt{3}}$;
\item $\sqrt{2+\sqrt{3}}:\sqrt{\dfrac{1}{2}}$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item $\dfrac{\sqrt{999}}{\sqrt{444}}=\sqrt{\dfrac{999}{444}}=\sqrt{\dfrac{9}{4}}=\dfrac{\sqrt{9}}{\sqrt{4}}=\dfrac{3}{2}$.
\item $\dfrac{\sqrt{160}}{\sqrt{0{,}4}}=\sqrt{\dfrac{160}{0{,}4}}=\sqrt{\dfrac{1600}{4}}=\dfrac{\sqrt{1600}}{\sqrt{4}}=\dfrac{40}{2}=20$.
\item $\dfrac{\sqrt{9+6\sqrt{2}}}{\sqrt{3}}=\sqrt{\dfrac{9+6\sqrt{2}}{3}}=\sqrt{3+2\sqrt{2}}=\sqrt{\left(\sqrt{2}+1\right)^2}=\sqrt{2}+1$;
\item $\sqrt{2+\sqrt{3}}:\sqrt{\dfrac{1}{2}}=\sqrt{\left(2+\sqrt{3}\right):\dfrac{1}{2}}=\sqrt{4+2\sqrt{3}}=\sqrt{\left(\sqrt{3}+1\right)^2}=\sqrt{3}+1$.
\end{enumerate}
}
\end{bt}

\begin{bt}
Rút gọn các biểu thức sau
\begin{multicols}{3}
\begin{enumerate}
\item $\sqrt{1\dfrac{9}{16}\cdot 5\dfrac{4}{9}\cdot 0{,}01}$;
\item $\sqrt{\dfrac{165^2-124^2}{164}}$;
\item $\sqrt{\dfrac{149^2-76^2}{457^2-384^2}}$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item $\sqrt{1\dfrac{9}{16}\cdot 5\dfrac{4}{9}\cdot 0{,}01}=\sqrt{\dfrac{25}{16}\cdot\dfrac{49}{9}\cdot\dfrac{1}{100}}=\sqrt{\dfrac{25}{16}}\cdot\sqrt{\dfrac{49}{9}}\cdot\sqrt{\dfrac{1}{100}}=\dfrac{5}{4}\cdot\dfrac{7}{3}\cdot\dfrac{1}{10}=\dfrac{7}{24}$.
\item $\sqrt{\dfrac{165^2-124^2}{164}}=\sqrt{\dfrac{(165-124)(165+124)}{164}}=\sqrt{\dfrac{41\cdot 289}{164}}=\sqrt{\dfrac{289}{4}}=\dfrac{\sqrt{289}}{\sqrt{4}}=\dfrac{17}{2}$.
\item $\sqrt{\dfrac{149^2-76^2}{457^2-384^2}}=\sqrt{\dfrac{(149-76)(149+76)}{(457-384)(457+384)}}=\sqrt{\dfrac{73\cdot 225}{73\cdot 841}}=\sqrt{\dfrac{225}{841}}=\dfrac{\sqrt{225}}{\sqrt{841}}=\dfrac{15}{29}$.
\end{enumerate}
}
\end{bt}

\begin{bt}
Rút gọn các biểu thức sau
\begin{multicols}{3}
\begin{enumerate}
\item $\dfrac{\sqrt{96x^5}}{\sqrt{24x}}$ ($x>0$);
\item $\dfrac{\sqrt{18(x+1)^3}}{\sqrt{2}x+\sqrt{2}}$ ($x>-1$);
\item $\dfrac{\sqrt{3x^4y^4}}{\sqrt{27x^2y^4}}$ ($x<0$, $y>0$).
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item $\dfrac{\sqrt{96x^5}}{\sqrt{24x}}=\sqrt{\dfrac{96x^5}{24x}}=\sqrt{4x^4}=2x^2$.
\item $\dfrac{\sqrt{18(x+1)^3}}{\sqrt{2}x+\sqrt{2}}=\dfrac{\sqrt{18(x+1)^3}}{\sqrt{2(x+1)^2}}=\sqrt{\dfrac{18(x+1)^3}{2(x+1)^2}}=\sqrt{9(x+1)}=3\sqrt{x+1}$ (với $x>-1$).
\item $\dfrac{\sqrt{3x^4y^4}}{\sqrt{27x^2y^4}}=\sqrt{\dfrac{3x^4y^4}{27x^2y^4}}=\sqrt{\dfrac{x^2}{9}}=\dfrac{\sqrt{x^2}}{\sqrt{9}}=\dfrac{-x}{3}$ (với $x<0$, $y>0$).
\end{enumerate}
}
\end{bt}

\begin{bt}
Rút gọn các biểu thức sau
\begin{multicols}{2}
\begin{enumerate}
\item $(2x-2)\sqrt{\dfrac{(x-2)^4}{(x-1)^2}}$ ($x<1$);
\item $\sqrt{\dfrac{4+12a+9a^2}{b^2}}$ ($a\le -\dfrac{2}{3}$).
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item $(2x-2)\sqrt{\dfrac{(x-2)^4}{(x-1)^2}}=2(x-1)\cdot\dfrac{\sqrt{(x-2)^4}}{\sqrt{(x-1)^2}}=2(x-1)\cdot\dfrac{(x-2)^2}{-(x-1)}=-2(x-2)^2$ (với $x<1$).
\item $\sqrt{\dfrac{4+12a+9a^2}{b^2}}=\sqrt{\dfrac{(2+3a)^2}{\sqrt{b^2}}}=\dfrac{\sqrt{(2+3a)^2}}{b^2}=\dfrac{-(2+3a)}{|b|}$ (với $a\le -\dfrac{2}{3}$).
\end{enumerate}
}
\end{bt}

\begin{bt}
Giải các phương trình sau
\begin{multicols}{2}
\begin{enumerate}
\item $\sqrt{\dfrac{(x+1)^4}{x}}=4\sqrt{x}$;
\item $(x-1)\sqrt{\dfrac{2x-x^2-1}{x}}-2x=-2$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item Điều kiện $\heva{&x>0 \\ &\dfrac{(x+1)^4}{x}\ge 0} \Leftrightarrow x>0.$
Khi đó phương trình tương đương với
\[ \dfrac{\sqrt{(x+1)^4}}{\sqrt{x}}=4\sqrt{x}\Leftrightarrow (x+1)^2=4x\Leftrightarrow (x-1)^2=0\Leftrightarrow x=1\,\,\text{(thỏa mãn).}\]
Vậy nghiệm phương trình là $x=1$.
\item Điều kiện $$\dfrac{2x-x^2-1}{x}\ge 0\Leftrightarrow \dfrac{-(x-1)^2}{x}\ge 0\Leftrightarrow \hoac{& x-1=0\\ & x<0}\Leftrightarrow \hoac{& x=1\\ & x<0.}$$
Khi đó phương trình tương đương với
\begin{eqnarray*}
&& (x-1)\sqrt{\dfrac{(x-1)^2}{-x}}-2(x-1)=0\\
& \Leftrightarrow & (x-1)\left(\sqrt{\dfrac{(x-1)^2}{-x}}-2\right)=0\\
& \Leftrightarrow & \hoac{ & x-1=0\\ & \sqrt{\dfrac{(x-1)^2}{-x}}-2=0}
\Leftrightarrow \hoac{ & x=1\quad\text{(thỏa mãn)}\\ & \sqrt{\dfrac{(x-1)^2}{-x}}=2.\quad (1)}
\end{eqnarray*}
Phương trình (1) tương đương với
\begin{eqnarray*}
&& \dfrac{\sqrt{(x-1)^2}}{\sqrt{-x}}=2\Leftrightarrow \sqrt{(x-1)^2}=2\sqrt{-x}\\
& \Leftrightarrow & \sqrt{(x-1)^2}=\sqrt{-4x}\Leftrightarrow (x-1)^2=-4x\\
& \Leftrightarrow & (x+1)^2=0\Leftrightarrow x=-1\quad \text{(thỏa mãn)}.
\end{eqnarray*}
Vậy nghiệm phương trình là $x=\pm 1$.
\end{enumerate}
}
\end{bt}

\begin{bt}
Chứng minh các bất đẳng thức sau
\begin{enumerate}
\item $\sqrt{\dfrac{a+b}{2}}\ge\dfrac{\sqrt{a}+\sqrt{b}}{2}$, với $a,b\ge 0$;
\item $\sqrt{a}+\sqrt{\dfrac{1}{a}}\ge 2$, với $a>0$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $$\left(\sqrt{a}-\sqrt{b}\right)^2\ge 0\Leftrightarrow a+b\ge 2\sqrt{ab}\Leftrightarrow 2(a+b)\ge a+b+2\sqrt{ab}\Leftrightarrow2(a+b)\ge \left(\sqrt{a}+\sqrt{b}\right)^2$$
$$\Leftrightarrow \dfrac{a+b}{2}\ge \dfrac{\left(\sqrt{a}+\sqrt{b}\right)^2}{4}\Leftrightarrow\sqrt{\dfrac{a+b}{2}}\ge\sqrt{\dfrac{\left(\sqrt{a}+\sqrt{b}\right)^2}{4}}.$$
Mà $\sqrt{\dfrac{\left(\sqrt{a}+\sqrt{b}\right)^2}{4}}=\dfrac{\sqrt{\left(\sqrt{a}+\sqrt{b}\right)^2}}{\sqrt{4}}=\dfrac{\sqrt{a}+\sqrt{b}}{2}$. Nên $\sqrt{\dfrac{a+b}{2}}\ge\dfrac{\sqrt{a}+\sqrt{b}}{2}$, với $a,b\ge 0$.
\item Ta có
$$\sqrt{a}+\sqrt{\dfrac{1}{a}}=\sqrt{a}+\dfrac{1}{\sqrt{a}}=\dfrac{\left(\sqrt{a}\right)^2+1}{\sqrt{a}}=\dfrac{a+1}{\sqrt{a}}.$$
Mà $\left(\sqrt{a}-1\right)^2\ge 0\Leftrightarrow a+1\ge 2\sqrt{a}$. Nên $\sqrt{a}+\sqrt{\dfrac{1}{a}}\ge \dfrac{2\sqrt{a}}{\sqrt{a}}=2$.
\end{enumerate}
}
\end{bt}

{$x=-\dfrac{\sqrt{a}}{\sqrt{b}}$}
{\True $x=\dfrac{\sqrt{-a}}{\sqrt{-b}}$}
{$x=-\dfrac{\sqrt{-a}}{\sqrt{-b}}$}
\loigiai{
Ta có $a,b<0$ nên $-a>0$ và $-b>0$ do đó
$$\sqrt{\dfrac{a}{b}}=\sqrt{\dfrac{-a}{-b}}=\dfrac{\sqrt{-a}}{\sqrt{-b}}.$$
}
\end{bt}

\begin{bt}
Điều kiện để $\sqrt{\dfrac{x}{x+1}}=\dfrac{\sqrt{x}}{\sqrt{x+1}}$ là
\choice{$x\in\mathbb{R}$}
{\True $x\ge 0$}
{$x\ge -1$}
{$x>-1$}
\loigiai{
Để xảy ra đẳng thức thì điều kiện là 
$$\heva{& x\ge 0\\ & x+1>0}\Leftrightarrow \heva{& x\ge 0\\ & x>-1}\Leftrightarrow x\ge 0.$$
}
\end{bt}

\begin{bt}
Cho $\dfrac{\sqrt{10-4\sqrt{6}}}{\sqrt{2}}=\sqrt{a}-\sqrt{b}$ với $a,b$ là các số nguyên dương. Khi đó $a-b$ bằng?
\choice{\True $1$}
{$-1$}
{$2$}
{$-2$}
\loigiai{
Ta có $$\dfrac{\sqrt{10-4\sqrt{6}}}{\sqrt{2}}=\sqrt{\dfrac{10-4\sqrt{6}}{2}}=\sqrt{5-2\sqrt{6}}=\sqrt{\left(\sqrt{3}-\sqrt{2}\right)^2}=\sqrt{3}-\sqrt{2}.$$
Do đó $a=3$, $b=2$ và $a-b=1$.
}
\end{bt}
