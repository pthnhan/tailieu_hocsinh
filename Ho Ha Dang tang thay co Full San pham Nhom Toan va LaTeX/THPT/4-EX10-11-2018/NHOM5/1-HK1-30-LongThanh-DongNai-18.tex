\begin{ex}%[Đề cuối HK1 - khối 11, 2018, Long Thành, Đồng Nai]%[1D1K3]

	\choice
	{$\dfrac{\pi}{2}$}
	{$-\dfrac{\pi}{2}$}
	{$ \pi $}
	{\True $0$}
	\loigiai{
	
	\underline{\textbf{Cách 1:}} Giải phương trình
	 
	$ \begin{aligned} 
	& \sin^2 x= \dfrac{1}{4} + \cos^4 x \Leftrightarrow 
	\sin^4 x -3 \sin^2 x + \dfrac{5}{4}=0 \\
	\Leftrightarrow & \left[ \begin{aligned}
	& \sin^2 x= \dfrac{5}{2} \text{ (loại) } \\
	& \sin^2 x= \dfrac{1}{2}
	\end{aligned}\right.
	\Leftrightarrow \cos 2x=0 \Leftrightarrow x= \dfrac{\pi}{4} + k \dfrac{\pi}{2} (k \in \mathbb{Z})
	\end{aligned}$.
	
	Với $k=0$ ta có $b= \dfrac{\pi}{4}$ là nghiệm dương nhỏ nhất của phương trình.
	
	Với $k=-1$ ta có $a= -\dfrac{\pi}{4}$ là nghiệm âm lớn nhất của phương trình.
	
	Vậy $a+b=0$.
	
	\underline{\textbf{Cách 2:}} Xét hàm số $f(x)= \sin^2 x - \cos^4 x - \dfrac{1}{4}$. Hàm $f(x)$ là hàm số chẵn trên tập xác định $\mathbb{R}$, do đó nghiệm âm lớn nhất $a$ và nghiệm dương nhỏ nhất $b$ của phương trình ban đầu là hai số đối nhau. Vậy ta có $a+b=0$.
	}
\end{ex}

	a) Tìm giao tuyến của hai mặt phẳng $(SAB)$ và $(SCD)$.
	
	b) Tìm giao điểm của mặt phẳng $(OMN)$ và đường thẳng $CD$. Xác định thiết diện của mặt phẳng $(OMN)$ với hình chóp. Thiết diện đó là hình gì?
	\loigiai{\immini{
		\begin{enumerate} 
			\item Có $S \in (SAB) \cap (SCD)$ 
			
			và $AB \parallel CD, AB \subset (SAB), CD \subset (SCD)$, 
			
			suy ra $(SBC) \cap (SAD)=Sx \parallel AB \parallel CD$.
			\item %$(OMN) \cap (ABCD)=Oy \parallel AD \parallel BC$.
			Qua $O$ kẻ đường thẳng song song với $AD$ cắt $AB, CD$ lần lượt tại $I, J$.
			%Trong $(ABCD)$, gọi $Oy \cap CD =I$. 
			Ta có $I \in CD$.
			
			Mà $IJ \subset (OMN) \Rightarrow CD \cap (OMN)=I$. 
			
			Thiết diện của $(OMN)$ và hình chóp là tứ giác $MNIJ$.
			
			Vì $MN \parallel IJ$ nên $MNIJ$ là hình thang có hai đáy là $MN$ và $IJ$.
		\end{enumerate}
	}{
	\begin{tikzpicture}[scale=0.7]
	\tkzDefPoints{0/0/A, 1/5/S, 6/0/D, -2/-2/B}
	\coordinate (C) at ($(B)+(D)-(A)$);
	\coordinate (K) at ($(B)+(S)-(A)$);
	\coordinate (L) at ($(S)+(D)-(C)$);
	\tkzInterLL(A,C)(B,D) \tkzGetPoint{O}
	\tkzDefMidPoint(A,S)\tkzGetPoint{M}
	\tkzDefMidPoint(D,S)\tkzGetPoint{N}
	\tkzDefMidPoint(C,D)\tkzGetPoint{I}
	\tkzDefMidPoint(A,B)\tkzGetPoint{J}
	\tkzDrawSegments[dashed](S,A A,B A,D A,C B,D O,N O,M M,N M,J I,J)
	\tkzDrawSegments(S,B S,C S,D B,C C,D N,I K,L)
	\tkzLabelPoints[above](S)
	\tkzLabelPoints[below](O,B,C)
	\tkzLabelPoints[right](D,I)
	\tkzLabelPoints[left](J)
	\tkzLabelPoints[above right](M,A,N)
	\tkzLabelSegment[above](S,L){$x$}
	\end{tikzpicture}
	}
