	\loigiai
	{Đặt ước chung lớn nhất của $a$ và $b$ là $d=(a,b)$ thì $d$ là ước của $a=n^3+2n$ nên $d$ là ước của $n(n^3+2n)=n^4+2n^2$. Từ đó và do $d$ là ước của $b=n^4+3n^+1$ nên $d$ là ước của
		\begin{align}
		n^4+3n^2+1-(n^4+2n^2)=n^2+1,\tag{*} 
		\end{align}
		suy ra $d$ là ước của $n(n^2+1)=n^3+n$ nên $d$ là ước của $n^3+2n^2-(n^3+n)=n$.

	}

	\loigiai
	{
		\begin{align}
		(1) &\Leftrightarrow xy(x^2+2xy+y^2)-4(x+y)+4=0 \nonumber \\
		&\Leftrightarrow xy(x+y)^2-4(x+y)+4=0 \nonumber \\
		&\Leftrightarrow (x+y)^2(1-xy)=(x+y-2)^2 \tag{2}
		\end{align}
		Nhận thấy $x+y=0$ không thỏa mãn $(2)$. Nên khi đó $x+y\neq 0$, chia hai vế của $(2)$ cho $(x+y)^2$ ta có: 
		\begin{align}
		1-xy=\left( 1-\dfrac{2}{x+y}\right)^2 \tag{3}
		\end{align}
		Do $x,y \in \mathbb{Z}$ nên $1-xy \in \mathbb{Z}$ suy ra $\dfrac{2}{x+y} \in \mathbb{Z}$. Do đó $x+y \in \{ \pm 1, \pm 2\} $. Kết hợp với (3) xảy ra các trường hợp sau:
		\begin{itemize}

			x+y &=1\\
			1-xy &=1
			\end{cases}
			\Leftrightarrow
			\begin{cases}
			x+y&=1\\
			xy&=0
			\end{cases}
			\Leftrightarrow
			\left[ 
			\begin{array}{ll} 
			x=1;y=0   \\ 
			x=0;y=1  \\
			\end{array} 
			\right.$			
			\item $
			\begin{cases}
			x+y &=-1\\
			1-xy &=9
			\end{cases}
			\Leftrightarrow
			\begin{cases}
			x+y&=-1\\
			xy&=-8
			\end{cases}
			$\\
			(không có nghiệm nguyên).\\
			\item $ \begin{cases}
			x+y &=2\\
			1-xy &=0
			\end{cases}
			\Leftrightarrow
			\begin{cases}
			x+y&=2\\
			xy&=1
			\end{cases}
			\Leftrightarrow
			\begin{cases}
			x&=1\\
			y&=1
			\end{cases}$
			\item $	\begin{cases}

	\begin{center}
		$\dfrac{a^3(a+b)}{a^2+b^2}+\dfrac{b^3(b+c)}{b^2+c^2}+\dfrac{c^3(c+a)}{c^2+a^2}\ge a^2+b^2+c^2. $
	\end{center}
	
	\loigiai{ BĐT đầu bài tương đương với bất đẳng thức sau:
	{\allowdisplaybreaks
	\begin{align*}
	&\dfrac{a^3(a+b)}{a^2+b^2}+\dfrac{b^3(b+c)}{b^2+c^2}+\dfrac{c^3(c+a)}{c^2+a^2}\ge a^2+b^2+c^2\\
		\Leftrightarrow& \dfrac{a^3b-a^2b^2}{a^2+b^2}+\dfrac{b^3c-b^2c^2}{b^2+c^2}+\dfrac{c^3a-c^2a^2}{c^2+a^2} \ge 0\\

	\end{align*}}Mà $a^2b(a-b)+b^2c(b-c)+c^2a(c-a) \ge 0 \hspace*{0.1cm} (*)$ theo bài 6 kỳ thi toán quốc tế 1983. \\

		$$(y+z)^2(z+x)(y-x)+(z+x)^2(x+y)(z-y)+(x+y)^2(y+z)(x-z) \ge 0. $$

		\begin{center}
			$(xy^3+yz^3+zx^3)(z+x+y) \ge \left[ \sqrt{xyz}(x+y+z)^2 \right]. $\\
		\end{center}

		{\allowdisplaybreaks
		\begin{align*}
		&2(ab^3+bc^3+ca^3-a^2b^2-b^2c^2-c^2a^2) \\
		=&\left[a^2(b-c)^2(a+c-b)+b^2(c-a)^2(b+a-c)+c^2(a-b)^2(c+b-a) \right] \ge 0\\
		&\text{(vì $a+c-b, b+c-a, c+a-b$ là các số dương do $a, b, c$ là ba cạnh của một tam giác). }
		\end{align*}}

\begin{bt}[T7/485, Toán học \& Tuổi trẻ, tháng 11 năm 2017]%[Nguyễn Thị Kiều Ngân, Dự án EX-TapChi03]%[2D0G2]
	Cho hàm số $f(x)$ liên tục trên $[a;b]$ và có đạo hàm trên $(a;b)$. Chứng minh rằng tồn tại $c \in (a;b)$ sao cho
	$$f'(c)=\dfrac{1}{a-c} +\dfrac{1}{b-c} +\dfrac{1}{a+b}.$$
\loigiai{
	Xét hàm số $g(x)=(x-a)(x-b)\mathrm{e}^{f(x)-\tfrac{x}{a+b}}$, $x \in [a;b].$\\
	Khi đó $g(x)$ cũng liên tục trên $[a;b]$, có đạo hàm trên $(a;b)$ và $g(a)=g(b)=0$.\\
	Ta có
	\begin{align*}
		g'(x) &= (x-b)\mathrm{e}^{f(x)-\tfrac{x}{a+b}} +(x-a)\mathrm{e}^{f(x)-\tfrac{x}{a+b}} +(x-a)(x-b)\mathrm{e}^{f(x)-\tfrac{x}{a+b}}\left( f'(x)-\dfrac{1}{a+b} \right)\\
		&= \mathrm{e}^{f(x)-\tfrac{x}{a+b}} \left[ (x-a)+(x-b)+(x-a)(x-b)\left( f'(x)-\dfrac{1}{a+b} \right) \right].
	\end{align*}
	Theo định lý Rolle tồn tại $c \in (a;b)$ để $g'(c)=0$.
	Suy ra
	{\allowdisplaybreaks
	\begin{align*}
		&(c-a)+(c-b)+(c-a)(c-b)\left( f'(c)-\dfrac{1}{a+b} \right)=0\\
		\Leftrightarrow & f'(c)=\dfrac{1}{a-c}+\dfrac{1}{b-c}+\dfrac{1}{a+b}.
	\end{align*}}
	\begin{nx}
		Cách giải trên đã sử dụng định lý Rolle, nội dung định lý Rolle như sau: \textit{Cho hàm $f(x)$ liên tục trên $[a;b]$, có đạo hàm trên $(a;b)$. Nếu $f(a)=f(b)$ thì tồn tại $c \in (a;b)$ sao cho $f'(c)=0$.} 
		Ý nghĩa hình học của định lý Rolle là: Nếu các điều kiện của định lý Rolle được thỏa mãn thì trên đồ thị hàm $y=f(x)$, $x \in (a;b)$, tồn tại điểm $M(c;f(c))$ mà tiếp tuyến với đồ thị tại $M$ song song hoặc trùng với trục $Ox$.
		Cần chú ý rằng nội dung của định lý Rolle chỉ là điều kiện đủ mà không là điều kiện cần, nghĩa là: Có những hàm $f$ liên tục trên $[a;b]$, có đạo hàm trên $(a;b)$ và tồn tại $c \in (a;b)$ sao cho $f'(c)=0$ nhưng điều kiện $f(a)=f(b)$ không thỏa mãn. Chẳng hạn: hàm $f(x)=x^3$ liên tục trên $[-1;1]$, có đạo hàm trên $(-1;1)$, có $f'(0)=0$ nhưng $f(-1) \ne f(1)$.
	\end{nx}
	}
	\end{bt}

	Cho $a$, $b$, $c$ là ba số thực không âm thỏa mãn: $ab+bc+ca=1$. Tìm giá trị nhỏ nhất của biểu thức: $P=\dfrac{(a^3+b^3+c^3+3abc)^2}{a^2b^2+b^2c^2+c^2a^2}$.
	\loigiai{
		Trước hết ta sẽ chứng minh bất đẳng thức (BĐT) sau:
		$$a^3+b^3+c^3+3abc \ge ab(a+b)+bc(b+c)+ca(c+a)$$
		với $a$, $b$, $c$ là các số thực không âm bất kỳ (được gọi là BĐT Schur).\\
		Thật vậy, do vai trò của $a$, $b$, $c$ trong BĐT là như nhau, không mất tính tổng quát ta giả sử $a\ge b \ge c$. Do đó
		\begin{align*}
			& \ a^3+b^3+c^3+3abc-ab(a+b)-bc(b+c)-ca(c+a)\\
			=& \ (a+b)\left( a^2+b^2-ab \right)-ab(a+b)+c\left( c^2+3ab-b^2-a^2-bc-ca \right)\\
			=& \ (a+b)(a-b)^2+c\left[ (c-a)(c-b)-(a-b)^2 \right]\\
			=& \ (a+b-c)(a-b)^2+c(c-a)(c-b) \ge 0.
		\end{align*}
		Đẳng thức xảy ra khi và chỉ khi $a=b$ và $c=a$ hoặc $c=0$. Áp dụng BĐT trên ta có:
		\begin{align*}
			\left( a^3+b^3+c^3+3abc \right)^2 & \ge \left[ab(a+b)+bc(b+c)+ca(c+a) \right]^2\\
			&= \left[ \left( a^2b+b^2c+c^2a \right)+\left( ab^2+bc^2+ca^2 \right) \right]^2\\
			&= \left[ \left( a^2b+b^2c+c^2a \right)-\left( ab^2+bc^2+ca^2 \right) \right]^2+\\
			& \qquad 4\left( a^2b+b^2c+c^2a \right) \left( ab^2+bc^2+ca^2 \right)\\
			& \ge  4\left( a^2b+b^2c+c^2a \right) \left(ab^2+bc^2+ca^2 \right). \tag{1}
		\end{align*} 
		Mặt khác:
		\begin{align*}
			& \left( a^2b+b^2c+c^2a \right) \left( ab^2+bc^2+ca^2 \right)-\left( a^2b^2+b^2c^2+c^2a^2 \right)\\
			=& \left( a^2b+b^2c+c^2a \right) \left( ab^2+bc^2+ca^2 \right)-\left( a^2b^2+b^2c^2+c^2a^2 \right)(ab+bc+ca)\\
			=& \left[ a^3b^3+b^3c^3+c^3a^3+abc \left( a^3+b^3+c^3+3abc \right) \right]-\\
			&\left\{ a^3b^3+b^3c^3+c^3a^3+abc \left[ ab(a+b)+bc(b+c)+ca(c+a) \right] \right\}\\
			=& \ abc\left[ a^3+b^3+c^3+3abc-ab(a+b)-bc(b+c)-ca(c+a) \right] \ge 0. \tag{2} 
		\end{align*}
		Từ $(1)$ và $(2)$ suy ra $P=\dfrac{\left( a^3+b^3+c^3+3abc \right)^2}{a^2b^2+b^2c^2+c^2a^2} \ge 4$.\\
		Đẳng thức xảy ra khi và chỉ khi hoặc $a=b=c=\dfrac{1}{\sqrt{3}}$ hoặc trong $a$, $b$, $c$ có hai số bằng $1$ và một số bằng $0$.
	}
