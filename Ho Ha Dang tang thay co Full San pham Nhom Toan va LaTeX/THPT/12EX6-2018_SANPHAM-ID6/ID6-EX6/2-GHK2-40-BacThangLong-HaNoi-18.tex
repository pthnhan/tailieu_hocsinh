\begin{ex}%[Đề TT lần 1, Bắc Thăng Long, Hà Nội, 2018]%[Nguyện Ngô,12EX6]%[2D3G2-1]
Cho hàm số $y=f(x)$ có đạo hàm liên tục trên $[1; 2]$ thỏa mãn $\displaystyle\int \limits_{1}^{2}f'(x)\mathrm{\,d}x=10$ và
$\displaystyle\int \limits_{1}^{2}\dfrac{f'(x)}{f(x)}\mathrm{\,d}x=\ln 2$. Biết rằng $f(x)>0, \forall x\in [1; 2]$. Tính $f(2)$.
\choice
{$f(2)=-10$}
{$f(2)=10$}
{$f(2)=-20$}
{\True $f(2)=20$}
\loigiai{Ta có $\displaystyle\int \limits_{1}^{2}f'(x)\mathrm{\,d}x=10\Leftrightarrow f(x)\Big|_1^2=10\Leftrightarrow f(2)-f(1)=10$ $(1)$.\\
Ta có\\
$\begin{aligned}\displaystyle\int \limits_{1}^{2}\dfrac{f'(x)}{f(x)}\mathrm{\,d}x=\ln2
&\Leftrightarrow\ln|f(x)|\Big|_1^2=\ln2\Leftrightarrow\ln|f(2)|-\ln|f(1)|=\ln2\\
&\Leftrightarrow\ln f(2)-\ln f(1)=\ln 2\quad\text{vì}\quad  f(2)>0, f(1)>0\\
&\Leftrightarrow\ln\dfrac{f(2)}{f(1)}=\ln2\Leftrightarrow\dfrac{f(2)}{f(1)}=2\quad (2).
\end{aligned}$\\
Từ $(1)$, $(2)$ ta có $f(2)-\dfrac{f(2)}{2}=10\Leftrightarrow f(2)=20$.
}
\end{ex}

\begin{ex}%[Đề TT lần 1, Bắc Thăng Long, Hà Nội, 2018]%[Nguyện Ngô,12EX6]%[2D2G3-3]

\choice
{$\log(a+b)=\dfrac{1}{2}(\log a + \log b)$}
{\True $\log(a+b)=\dfrac{1}{2}(1 + \log a + \log b)$}
{$\log(a+b)=\dfrac{1}{2} + \log a + \log b$}
{$\log(a+b)=1 + \log a + \log b$}

\begin{center}
$\begin{aligned}a^2+b^2=8ab&\Leftrightarrow (a+b)^2=10ab\Leftrightarrow \log(a+b)^2=\log(10ab)\\
&\Leftrightarrow 2\log(a+b)=\log(10) + \log a + \log b\\
&\Leftrightarrow \log(a+b)=\dfrac{1}{2}(1 + \log a + \log b).
\end{aligned}
$
\end{center}
Vậy với mọi số thực dương $a$ và $b$ thỏa mãn $a^2+b^2=8ab$, mệnh đề đúng là\\

Gọi $(a; b; c)$ là bộ số thỏa mãn $F(x)$ là một nguyên hàm của $f(x)$. Khi đó $a+b+c$ bằng
\choice
{$1$}	
{$5$}
{\True $3$}
{$7$}
\loigiai{Ta có
\begin{center}
$\begin{aligned} f(x)&=10x\sqrt{2x-3}+\dfrac{7}{\sqrt{2x-3}}=5(2x-3)\sqrt{2x-3}+15\sqrt{2x-3}+\dfrac{7}{\sqrt{2x-3}}\\
&=5(2x-3)^{\tfrac{3}{2}}+15(2x-3)^{\tfrac{1}{2}}+\dfrac{7}{\sqrt{2x-3}}.
\end{aligned}$
\end{center}
Suy ra $\displaystyle\int f(x)\mathrm{\,d}x=(2x-3)^{\tfrac{5}{2}}+5(2x-3)^{\tfrac{3}{2}}+7\sqrt{2x-3}+C=(4x^2-2x+1)\sqrt{2x-3}+C$.\\
Suy ra $F(x)=(4x^2-2x+1)\sqrt{2x-3}$ hay $a=4, b=-2, c=1\Rightarrow a+b+c=3$.
}
\end{ex}

\begin{ex}%[Đề TT lần 1, Bắc Thăng Long, Hà Nội, 2018]%[Nguyện Ngô,12EX6]%[2D3G1-3]
Cho $F(x)=-\dfrac{1}{3x^3}$ là một nguyên hàm của hàm số $\dfrac{f(x)}{x}$. Tìm nguyên hàm của hàm số $f'(x)\ln x$.
\choice
{$\displaystyle\int f'(x)\ln x\mathrm{\,d}x=-\dfrac{\ln x}{x^3}+\dfrac{1}{3x^3}+C$}
{\True $\displaystyle\int f'(x)\ln x\mathrm{\,d}x=\dfrac{\ln x}{x^3}+\dfrac{1}{3x^3}+C$}
{$\displaystyle\int f'(x)\ln x\mathrm{\,d}x=\dfrac{\ln x}{x^3}-\dfrac{1}{5x^5}+C$}
{$\displaystyle\int f'(x)\ln x\mathrm{\,d}x=\dfrac{\ln x}{x^3}+\dfrac{1}{5x^5}+C$}
\loigiai{
Ta có $F'(x)=\dfrac{f(x)}{x}\Leftrightarrow\dfrac{1}{x^4}=\dfrac{f(x)}{x}\Rightarrow f(x)=\dfrac{1}{x^3}$.\\
Đặt $\heva{&u=\ln x\\&\mathrm{\,d}v=f'(x)\mathrm{\,d}x}\Rightarrow\heva{&\mathrm{\,d}u=\dfrac{1}{x}\mathrm{\,d}x\\&v=f(x).}$\\
Suy ra
\begin{center}
$\displaystyle\int f'(x)\ln x\mathrm{\,d}x=f(x)\cdot\ln x-\displaystyle\int\dfrac{f(x)}{x} \mathrm{\,d}x
=\dfrac{\ln x}{x^3}-F(x)+C=\dfrac{\ln x}{x^3}+\dfrac{1}{3x^3}+C$.
\end{center}
}
\end{ex}

\begin{ex}%[Đề TT lần 1, Bắc Thăng Long, Hà Nội, 2018]%[Nguyện Ngô,12EX6]%[2H3K2-7]
Trong không gian với hệ tọa độ $Oxyz$, cho điểm $I(1;2;3)$ và mặt phẳng $(P):2x-2y-z-4=0.$ Mặt cầu tâm $I$ tiếp xúc mặt phẳng $(P)$  tại điểm $H$.
Tìm tọa độ điểm $H$.
\choice
{$H(-3;0;-2)$}
{\True $H(3;0;2)$}
{$H(-1;4;4)$}
{$H(-1;4;1)$}
\loigiai{Ta có $I\not\in (P)$. Mặt cầu tâm $I$ tiếp xúc mặt phẳng $(P)$  tại điểm $H\Leftrightarrow H$ là hình chiếu của $I$ lên $(P)$\\
$\Leftrightarrow\heva{&H\in (P)\\&\vec{IH}=(x_H-1; y_H-2; z_H-3)\quad \text{ cùng phương }\quad \vec{n}=(2;-2;-1)}$\\
$\Leftrightarrow \heva{&2x_H-2y_H-z_H-4=0\\&\dfrac{x_H-1}{2}=\dfrac{y_H-2}{-2}=\dfrac{z_H-3}{-1}=t}$\\
$\Leftrightarrow \heva{&x_H=2t+1\\&y_H=-2t+2\\&z_H=-t+3\\&2(2t+1)-2(-2t+2)-(-t+3)-4=0}
\Leftrightarrow \heva{&t=1\\&x_H=3\\&y_H=0\\&z_H=2.}$\\
Vậy $H(3;0;2)$.	
}
\end{ex}

\begin{ex}%[Đề TT lần 1, Bắc Thăng Long, Hà Nội, 2018]%[Nguyện Ngô,12EX6]%[2D2G4-4]
Cho $x, y$ là các số dương thỏa mãn $xy\leq 4y-1$. Giá trị nhỏ nhất của $P=\dfrac{6\left(2x+y \right)}{x}+\ln \dfrac{x+2y}{y}$ là $a+\ln b$. Tính $ab$.
\choice
{$ab=45$}
{\True $ab=81$}
{$ab=115$}
{$ab=108$}
\loigiai{
Từ  $xy\leq 4y-1$ chia 2 vế cho $y^2$ ta được $ \dfrac{x}{y}\leq \dfrac{4}{y}-\dfrac{1}{y^2}=4-\left(2-\dfrac{1}{y}\right)^2\leq 4$.\\
Đặt $\dfrac{x}{y}=t$ thì $0<t\leq 4$.\\
Khi đó $P=f(t)=12+\dfrac{6}{t}+\ln (t+2)$ có $f'(t)=\dfrac{-6}{t^2}+\dfrac{1}{t+2}=\dfrac{t^2-6t-12}{t^2(t+2)}$.\\
Ta có $t^2-6t-12<0\Leftrightarrow t\in (3-\sqrt{21}; 3+\sqrt{21})\supset (0; 4]$. Suy ra $f'(t)<0,\quad \forall t\in (0; 4].$\\
Vậy $\underset{(0; 4]}{\min}f(t)=f(4)= \dfrac{27}{2}+\ln 6$.\\
 Suy ra giá trị nhỏ nhất của $P$  bằng $ \dfrac{27}{2}+\ln 6$ khi $y=\dfrac{1}{2}, x=2$.\\
 Khi đó $a=\dfrac{27}{2}; b=6\Rightarrow ab=81$.
	}
\end{ex}

\begin{ex}%[Đề TT lần 1, Bắc Thăng Long, Hà Nội, 2018]%[Nguyện Ngô,12EX6]%[2D2K4-5]
Ông Bình gửi tổng cộng $320$ triệu đồng ở hai ngân hàng X và Y theo phương thức lãi kép. Số tiền thứ nhất gửi ở ngân hàng X với lãi suất $2{,}1$ \% một quý,
gửi trong $15$ tháng. Số tiền còn lại gửi ở ngân hàng Y với lãi suất $0{,}73$ \% một tháng trong vòng $9$ tháng.
Tổng lợi tức ông Bình có được sau khi rút tiền ở hai ngân hàng là $27507768{,}13$ đồng. Hỏi số tiền ông Bình lần lượt gửi ở hai ngân hàng X và Y là bao nhiêu?
\choice
{\True $140$ triệu đồng và $180$ triệu đồng}
{$200$ triệu đồng và $120$ triệu đồng}
{$180$ triệu đồng và $140$ triệu đồng}
{$120$ triệu đồng và $200$ triệu đồng}
\loigiai{
Đổi $15$ tháng $=$ $5$ quý. Gọi số tiền gửi trong ngân hàng X là $x$ triệu đồng và trong ngân hàng Y là $320-x$ triệu đồng.

\begin{center}
$x(1+2{,}1\%)^5+(320-x)(1+0{,}73\%)^9-320 =27507768{,}13$.
\end{center}
Giải ra $x=140$ triệu đồng. Vậy số tiền ông Bình lần lượt gửi ở hai ngân hàng X và Y là $140$ triệu đồng và $180$ triệu đồng.
