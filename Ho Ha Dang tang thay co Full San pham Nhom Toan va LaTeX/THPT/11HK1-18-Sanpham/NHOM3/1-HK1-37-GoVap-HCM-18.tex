\begin{bt}%[HK1, 2017-2018, THPT Gò Vấp, TP.HCM]%[Nguyễn Thành Khang - DA 11HK1-18]%[1D1Y2-1]%[1D1Y3-1]%[1D1B3-5]
	Giải các phương trình sau:
	\begin{enumerate}
		\item $2\cos\left(2x+\dfrac{\pi}{6}\right)+1=0$.
		\item $\cos^2 2x-2\sin 2x+2=0$.
		\item $(2\cos x-1)(2\sin x+\cos x)=\sin 2x-\sin x$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có 
			\allowdisplaybreaks{
				\begin{eqnarray*}
					2\cos\left(2x+\dfrac{\pi}{6}\right)+1=0 &\Leftrightarrow& \cos\left(2x+\dfrac{\pi}{6}\right)=-\dfrac{1}{2} \\
					&\Leftrightarrow& \hoac{&2x+\dfrac{\pi}{6}=\dfrac{2\pi}{3}+k2\pi \\ &2x+\dfrac{\pi}{6}=-\dfrac{2\pi}{3}+k2\pi},k\in\mathbb{Z}\\
					&\Leftrightarrow& \hoac{&x=\dfrac{\pi}{4}+k\pi \\ &x=-\dfrac{5\pi}{12}+k\pi},k\in\mathbb{Z}.
				\end{eqnarray*}
			}
			Vậy phương trình có nghiệm $x\in\left\{\dfrac{\pi}{4}+k\pi;-\dfrac{5\pi}{12}+k\pi\Big|k\in\mathbb{Z}\right\}$.
			\item Ta có
			\allowdisplaybreaks{
				\begin{eqnarray*}
					\cos^2 2x-2\sin 2x+2=0 &\Leftrightarrow& 1-\sin^2 2x -2\sin 2x+2=0 \\
					&\Leftrightarrow& (\sin 2x+3)(1-\sin 2x)=0\\
					&\Leftrightarrow& \sin 2x=1 \\
					&\Leftrightarrow& 2x=\dfrac{\pi}{2}+k2\pi,k\in\mathbb{Z}\\
					&\Leftrightarrow& x=\dfrac{\pi}{4}+k\pi,k\in\mathbb{Z}.
				\end{eqnarray*}
			}
			Vậy phương trình có nghiệm $x=\dfrac{\pi}{4}+k\pi,k\in\mathbb{Z}$.
			\item Ta có 
			\allowdisplaybreaks{
				\begin{eqnarray*}
					& & (2\cos x-1)(2\sin x+\cos x)=\sin 2x-\sin x\\ &\Leftrightarrow& (2\cos x-1)(2\sin x+\cos x)=2\sin x\cos x-\sin x \\
					&\Leftrightarrow& (2\cos x-1)(\sin x+\cos x)=0 \\
					&\Leftrightarrow& (2\cos x-1)\cdot\sqrt{2}\sin\left(x+\dfrac{\pi}{4}\right)=0 \\
					&\Leftrightarrow& \hoac{&\cos x=\dfrac{1}{2} \\ &\sin\left(x+\dfrac{\pi}{4}\right)=0} \Leftrightarrow \hoac{&x=\pm\dfrac{\pi}{3}+k2\pi \\ &x=-\dfrac{\pi}{4}+k\pi},k\in\mathbb{Z}.
				\end{eqnarray*}
			}
			Vậy phương trình có nghiệm $x\in\left\{\pm\dfrac{\pi}{3}+k2\pi;-\dfrac{\pi}{4}+k\pi\Big|k\in\mathbb{Z}\right\}$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[HK1, 2017-2018, THPT Gò Vấp, TP.HCM]%[Nguyễn Thành Khang - DA 11HK1-18]%[1D2B2-6]
	Giải phương trình sau: $3\mathrm{C}_x^3+2\mathrm{C}_x^2=3\mathrm{A}_x^2$.
	\loigiai{
		Điều kiện: $x\in\mathbb{N}, x\ge 3$. Ta có
		\begin{eqnarray*}
			& & 3\mathrm{C}_x^3+2\mathrm{C}_x^2=3\mathrm{A}_x^2 \\
			&\Leftrightarrow& \dfrac{3\cdot x!}{3!\cdot(x-3)!}+\dfrac{2\cdot x!}{2!\cdot(x-2)!}=\dfrac{3\cdot x!}{(x-2)!} \\
			&\Leftrightarrow& \dfrac{x(x-1)(x-2)}{2}+x(x-1)=3x(x-1)\quad (\text{vì }x\ge 3) \\
			&\Leftrightarrow& \dfrac{x-2}{2}+1=3 \\
			&\Leftrightarrow& x=6 \text{ (thoả mãn)}.
		\end{eqnarray*}
		Vậy $x=6$.
	}
\end{bt}

\begin{bt}%[HK1, 2017-2018, THPT Gò Vấp, TP.HCM]%[Nguyễn Thành Khang - DA 11HK1-18]%[1D2B3-2]
	Tìm hệ số của số hạng chứa $x^{11}$ trong khai triển $\left(x^3+2x\right)^5$.
	\loigiai{
		Ta có $\displaystyle \left(x^3+2x\right)^5=\sum\limits_{k=0}^5 \mathrm{C}_5^k \left(x^3\right)^{5-k}(2x)^k=\sum\limits_{k=0}^5 2^k\mathrm{C}_5^k x^{15-2k}$.\\
		Vì $15-2k=11\Leftrightarrow k=2$ nên hệ số của số hạng chứa $x^{11}$ trong khai triển trên là $2^2\mathrm{C}_5^2=40$.
	}
\end{bt}

\begin{bt}%[HK1, 2017-2018, THPT Gò Vấp, TP.HCM]%[Nguyễn Thành Khang - DA 11HK1-18]%[1D3B1-1]
	Chứng minh với $\forall n\in \mathbb{N}^*$ ta có đẳng thức: $$2+5+8+\cdots+(3n-1)=\dfrac{n(3n+1)}{2}.$$
	\vspace*{-2em}
	\loigiai{
		Với $n=1$ ta có $2=\dfrac{1\cdot4}{2}$ nên đẳng thức đúng với $n=1$.\\
		Giả sử đẳng thức đúng với $n=k,k\in\mathbb{N}^*$, hay $$2+5+8+\cdots+(3k-1)=\dfrac{k(3k+1)}{2}.$$
		Ta chứng minh đẳng thức đúng với $n=k+1$, hay $$2+5+8+\cdots+(3k-1)+(3k+2)=\dfrac{(k+1)(3k+4)}{2}.$$
		Thật vậy, ta có
		$$2+5+8+\cdots+(3k-1)+(3k+2)=\dfrac{k(3k+1)}{2}+(3k+2)=\dfrac{3k^2+7k+4}{2}=\dfrac{(k+1)(3k+4)}{2}.$$
		Vậy theo nguyên lý quy nạp, đẳng thức đúng với mọi $n\in\mathbb{N}^*$.
	}
\end{bt}

\begin{bt}%[HK1, 2017-2018, THPT Gò Vấp, TP.HCM]%[Nguyễn Thành Khang - DA 11HK1-18]%[1D2B1-3]
	Có bao nhiêu số tự nhiên chẵn gồm $4$ chữ số đôi một khác nhau?
	\loigiai{
		Giả sử số có $4$ chữ số cần lập là $\overline{abcd}$, với $a,b,c,d\in\mathbb{N}, 0\le a,b,c,d\le 9, a\ne 0$.\\
		Ta xét hai trường hợp sau:
		\begin{enumerate}[Trường hợp 1:]
			\item $a$ chẵn, khi đó có $4$ cách chọn $a$, $4$ cách chọn $d$, $8$ cách chọn $b$, $7$ cách chọn $c$. Vậy trong trường hợp này có $4\cdot 4\cdot 8\cdot 7=896$ số thoả mãn.
			\item $a$ lẻ, khi đó có $5$ cách chọn $a$, $5$ cách chọn $d$, $8$ cách chọn $b$, $7$ cách chọn $c$. Vậy trong trường hợp này có $5\cdot 5\cdot 8\cdot 7=1400$ số thoả mãn.
		\end{enumerate}
		Vậy có tổng cộng $896+1400=2296$ số thoả mãn.
	}
\end{bt}

\begin{bt}%[HK1, 2017-2018, THPT Gò Vấp, TP.HCM]%[Nguyễn Thành Khang - DA 11HK1-18]%[1D2B5-2]
	Chọn ngẫu nhiên $3$ số từ tập hợp $\{1,2,3,4,\ldots,11\}$. Tìm xác suất để tổng ba số được chọn là một số lẻ.
	\loigiai{
		Số phần tử của không gian mẫu là $n(\Omega)=\mathrm{C}_{11}^3=165$.\\
		Gọi biến cố $A$ là: ``tổng ba số được chọn là một số lẻ''. Biến cố $A$ xảy ra khi trong ba số được chọn có đúng $1$ số lẻ hoặc có đúng $3$ số lẻ, suy ra $n(A)=\mathrm{C}_6^1\cdot \mathrm{C}_5^2+\mathrm{C}_6^3=80$.\\
		Vậy $\mathrm{P}(A)=\dfrac{n(A)}{n(\Omega)}=\dfrac{80}{165}=\dfrac{16}{33}$.
	}
\end{bt}

\begin{bt}%[HK1, 2017-2018, THPT Gò Vấp, TP.HCM]%[Nguyễn Thành Khang - DA 11HK1-18]%[1H1B7-2]
	Trong mặt phẳng toạ độ $Oxy$, cho điểm $I(2;3)$ và đường thẳng $d\colon x+3y-5=0$. Hãy viết phương trình ảnh của đường thẳng $d$ qua phép vị tự tâm $I$ tỉ số $k=2$.
	\loigiai{
		Lấy $A(5;0), B(2;1)$ thuộc $d$ và gọi $M,N$ lần lượt là ảnh của $A,B$ qua phép vị tự tâm $I$ tỉ số $k=2$. Ta có $\vec{IM}=(x_M-2;y_M-3), \vec{IN}=(x_N-2;y_N-3), \vec{IA}=\cdot(3;-3), \vec{IB}=(0;-2)$ và
		$$\heva{&\vec{IM}=2\vec{IA} \\ &\vec{IN}=2\vec{IB}}\Leftrightarrow \heva{&x_M-2=2\cdot 3\\ &y_M-3=2\cdot(-3) \\ &x_N-2=2\cdot 0 \\ &y_N-3=2\cdot (-2)} \Leftrightarrow \heva{&x_M=8 \\ &y_M=-3 \\ &x_N=2 \\ &y_N=-1.}$$
		Vậy toạ độ $M,N$ là $M(8;-3),N(2;-1)$.\\
		Ảnh của đường thẳng $d$ qua phép vị tự tâm $I$ tỉ số $k=2$ chính là đường thẳng $MN$ và có phương trình là $x+3y+1=0$.
	}
\end{bt}

\begin{bt}%[HK1, 2017-2018, THPT Gò Vấp, TP.HCM]%[Nguyễn Thành Khang - DA 11HK1-18]%[1H2B1-3]
	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình bình hành. Gọi $H,K$ lần lượt là trung điểm của $AB,SC$.
	\begin{enumerate}
		\item Xác định giao điểm $E$ của $AK$ với $(SBD)$ và tính $\dfrac{EA}{EK}$.
		\item Xác định giao điểm $F$ của $HK$ với $(SBD)$ và tính $\dfrac{FH}{FK}$.
	\end{enumerate}
	\loigiai{
		\immini{
			\vspace*{-1em}
			\begin{enumerate}
				\item Gọi $O$ là giao điểm của $AC$ và $BD$, khi đó giao điểm $E$ của $AK$ với $(SBD)$ chính là giao điểm của $AK$ và $SO$. Ta có $E$ là trọng tâm của tam giác $SAC$ nên $\dfrac{AE}{EK}=2$.
				\item Gọi $I$ là giao điểm của $HC$ và $BD$, khi đó giao điểm $F$ của $HK$ với $(SBD)$ chính là giao điểm của $HK$ và $SI$. Ta có $I$ là trọng tâm tam giác $ABC$ nên $IC=2HI$, gọi $J$ là trung điểm $IC$ thì $HI=IJ$. Mà $KJ$ là đường trung bình của tam giác $SIC$ nên $KJ\parallel SI\Rightarrow \dfrac{HF}{FK}=\dfrac{HI}{IJ}=1$.
			\end{enumerate}

				\tkzClip

				\tkzDrawPoints[fill=black](A,B,C,D,S,H,K,O,E,I,F,J)
