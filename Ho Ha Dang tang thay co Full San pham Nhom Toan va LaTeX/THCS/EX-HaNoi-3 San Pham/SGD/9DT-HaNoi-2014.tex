\begin{ex}%[9D1Y1]
Tính giá trị của biểu thức $A=\dfrac{\sqrt{x}+1}{\sqrt{x}-1}$ khi $x=9$.
\loigiai{
Khi $x = 9$ thì $A = \dfrac{\sqrt{9}+1}{\sqrt{9}-1} = \dfrac{3+1}{3-1}=2$.
}

\begin{ex}%[9D1B8]%[9D1B8]
Cho biểu thức $P=\left( \dfrac{x-2}{x+2\sqrt{x}}+\dfrac{1}{\sqrt{x}+2} \right) : \dfrac{\sqrt{x}-1}{\sqrt{x}+1}$ với $x>0$ và $x \neq 1$.
	\begin{enumerate}
		\item Chứng minh rằng $P=\dfrac{\sqrt{x}+1}{\sqrt{x}}$.
		\item Tìm các giá trị của $x$ để $2P=2\sqrt{x}+5$ 
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Với $x > 0$ và $x \neq 1$ ta có 
\begin{align*}
P &= \left(\dfrac{x-2}{\sqrt{x}(\sqrt{x}+2)}+\dfrac{1}{\sqrt x + 2}\right)\cdot \dfrac{\sqrt{x}+1}{\sqrt{x}-1}\\
& = \dfrac{x - 2 + \sqrt{x}}{\sqrt{x}(\sqrt{x}+2)}\cdot \dfrac{\sqrt{x}+1}{\sqrt{x}-1} \\
&= \dfrac{(\sqrt x - 1)(\sqrt x + 2)}{\sqrt{x}(\sqrt x + 2)}\cdot\dfrac{\sqrt{x}+1}{\sqrt{x}-1}\\
& = \dfrac{\sqrt{x}+1}{\sqrt{x}}
\end{align*}
\item Với $x > 0$ và $x \neq 1$ ta có 
\begin{align*}
&2P=2\sqrt{x}+5\\
\Leftrightarrow & 2\cdot\dfrac{\sqrt{x}+1}{\sqrt{x}} = 2 \sqrt{x} + 5\\
\Leftrightarrow & 2\sqrt{x} + 2 = 2x + 5\sqrt{x}\\
\Leftrightarrow & 2x + 3\sqrt{x} - 2 = 0\\
\Leftrightarrow & \sqrt{x} = \dfrac{1}{2} \Leftrightarrow x = \dfrac{1}{4}
\end{align*}
\end{enumerate}
}
\end{ex}
 
\begin{ex}%[9D4B8]
\textit{Giải bài toán sau bằng cách lập phương trình hoặc hệ phương trình:}\\
	Một phân xưởng theo kế hoạch phải sản xuất 1100 sản phẩm trong một số ngày quy định. Do mỗi ngày phân xưởng đó sản xuất vượt mức 5 sản phẩm nên phân xưởng đã hoàn thành kế hoạch sớm hơn thời gian quy định 2 ngày. Hỏi theo kế hoạch, mỗi ngày phân xưởng phải sản xuất bao nhiêu sản phẩm?
	\loigiai{
			Gọi $x$ là số sản phẩm mỗi ngày xưởng làm được. ($x\in \mathbb N$, $0< x < 1100$).\\
			Số ngày mà xưởng làm xong theo kế hoạch là $\dfrac{1100}{x}$ (ngày).\\
			Mỗi ngày xưởng làm vượt mức $5$ sản phẩm nên số ngày mà xưởng làm xong là $\dfrac{1100}{x+5}$ (ngày).\\
			Vì xưởng xong sớm $2$ ngày nên ta có $$\dfrac{1100}{x+5}+2 = \dfrac{1100}{x}.$$
			Giải phương trình ta có $\hoac{&x=50\text{ (nhận)}\\&x=-55\text{ (loại)}}$.\\
			Vậy mỗi ngày xưởng làm $50$ sản phẩm.
	}
\end{ex}  
\begin{ex}%[9D3K4]
Giải hệ phương trình:  $ \begin{cases}
&\dfrac{4}{x+y}+\dfrac{1}{y-1}=5 \\ 
&\dfrac{1}{x+y}-\dfrac{2}{y-1}=-1 \\ 
\end{cases} $
\loigiai{
	Điều kiện $x+y\neq 0$ và $y\neq 1$\\
	Đặt $X = \dfrac{1}{x+y}$ và $Y = \dfrac{1}{y-1}$ ta có \\
	$\begin{cases}
		4X + Y =5\\
		X-2Y = -1
	\end{cases} \Leftrightarrow \begin{cases}
		X = 1\\
		Y = 1
	\end{cases}$\\
	Do đó $\begin{cases}
	x+y=1\\
	y-1=1
\end{cases}	\Leftrightarrow \begin{cases}
	x = -1\\
	y = 2
\end{cases} $
}
\end{ex}

\begin{ex}%[9H2Y1]%[9H4B7]%[9H4K5]%[9H3G1]
	Cho đường tròn $(O; R)$ có đường kính AB cố định. Vẽ đường kính $MN$ của đường tròn $(O; R)$ ($M$ khác $A$, $M$ khác $B$). Tiếp tuyến tại $B$ của đường tròn $(O; R)$ cắt các đường thẳng $AM$, $AN$ lần lượt tại các điểm $Q$, $P$.
\begin{enumerate}
\item Chứng minh tứ giác $AMBN$ là hình chữ nhật.
\item Chứng minh 4 điểm $M$, $N$, $P$, $Q$ cùng thuộc một đường tròn.
\item Gọi $E$ là trung điểm của $BQ$. Đường thẳng vuông góc với $OE$ tại $O$ cắt $PQ$ tại điểm $F$. Chứng minh $F$ là trung điểm của $BP$ và $ME\parallel NF$.
\item Khi đường kính $MN$ quay quanh tâm $O$ và thỏa mãn điều kiện đề bài, xác định vị 	trí của đường kính $MN$ để tứ giác $MNPQ$ có diện tích nhỏ nhất.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[>=stealth, line join=round, line cap = round,scale=0.4]
\tkzDefPoints{0/0/O, 0/4/A, 0/-4/B}
\tkzDrawCircle(O,B)
\tkzDefLine[perpendicular=through B](A,B)\tkzGetPoint{S}
\tkzDefPointBy[rotation = center O angle 60](A)\tkzGetPoint{M}
\tkzDefPointBy[rotation = center O angle -120](A)\tkzGetPoint{N}
\tkzInterLL(A,M)(S,B)\tkzGetPoint{Q}
\tkzInterLL(A,N)(S,B)\tkzGetPoint{P}
\tkzDefMidPoint(B,Q)\tkzGetPoint{E}
\tkzDefLine[perpendicular=through O](O,E)\tkzGetPoint{S'}
\tkzInterLL(P,Q)(S',O)\tkzGetPoint{F}
\tkzDefPointBy[projection=onto M--N](A)\tkzGetPoint{H}
\tkzDrawSegments(O,E A,B M,N A,P A,Q P,Q B,M B,N O,F M,E N,F A,H)
\tkzLabelPoints[left](M,Q)
\tkzLabelPoints[right](P,N,O)
\tkzLabelPoints[below](E,B,F,H)
\tkzLabelPoints[above](A)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Tứ giác $AMBN$ có hai đường chéo $AB$, $MN$ bằng nhau và cắt nhau tại trung điểm mỗi đường nên $AMBN$ là hình chữ nhật.
\item Do câu a) ta có $\widehat{ANM}  = \widehat{ABM}$\\
Hơn nữa $\widehat{ABM} = \widehat{MQP}$ (cùng phụ với $\widehat{QAB}$)\\
Do đó $\widehat{ANM} = \widehat{MQP} \Rightarrow M,N,P,Q$ cùng thuộc một đường tròn.
\item Trong $\triangle ABQ$ ta có $OE$ là đường trung bình nên $OE \parallel AQ$.\\
Ta lại có $OF \perp OE$ và $AP \perp AQ$ nên $OF \parallel AP$.\\
Trong tam giác $ABP$ có $O$ là trung điểm $AB$ và $OF \parallel AP$ nên $F$ là trung điểm $BP$.\\
Ta có $\widehat{MEF} = \dfrac{1}{2}(\text{sđ } \wideparen{MAB}-\text{sđ }\wideparen{MB})$ và $\widehat{NFE} = \dfrac{1}{2}(\text{sđ }\wideparen{BAN}-\text{sđ }\wideparen{BN})$.\\
Do đó $\widehat{MEF}+\widehat{NFE} = \text{sđ }\wideparen{MAN}=180^\circ \Rightarrow ME \parallel NF$ (góc trong cùng phía bù nhau).
\item Do $E,F$ là trung điểm của $BQ,BP$ nên $PQ = 2EF \geq 2MN$.\\
$S_{MNPQ} = S_{APQ} - S_{AMN}$. Ta cần tìm vị trí của $M$, $N$ sao cho $S_{APQ}$ nhỏ nhất và $S_{AMN}$ lớn nhất. Thật vậy,\\
$S_{APQ} = \dfrac{AB\cdot PQ}{2} = EF\cdot AB \geq MN\cdot AB = 4R^2$. Dấu bằng xảy ra khi $MN\perp AB$.\\
Hơn nữa, $S_{AMN} = \dfrac{MN\cdot AH}{2} \leq \dfrac{MN\cdot OA}{2} = R^2$. Dấu bằng xảy ra khi $MN \perp AB$.\\
Vậy $MN \perp AB$ để $S_{MNPQ}$ nhỏ nhất.
\end{enumerate}
}	
\end{ex}  
\begin{ex}%[9D5G1]
Với $a$, $b$, $c$ là các số dương thỏa mãn điều kiện $a+b+c=2$. Tìm giá trị lớn nhất của biểu thức 	$Q=\sqrt{2a+bc}+\sqrt{2b+ca}+\sqrt{2c+ab}$.
\loigiai{
	Vì $a+b+c=2$ nên $2a+bc = a(a+b+c)+bc = (a+b)(a+c)$\\
	$\Rightarrow \sqrt{2a+bc} = \sqrt{(a+b)(a+c)} \leq \dfrac{a+b+a+c}{2}$.\\
	Tương tự ta có $\sqrt{2b+ca}\leq \dfrac{b+c+b+a}{2}$ và $\sqrt{2c+ab}\leq \dfrac{c+a+c+b}{2}$.\\
	Cộng theo vế ta có $Q \leq 2(a+b+c)=4$.
}
\end{ex}
