\begin{bt}%[HK2 (2016-2017), THPT ĐẦM DƠI, CÀ MAU]%[Lê Thanh Nin + Thọ Bùi]%[0D4B5]
	Giải các bất phương trình sau:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\left(x^2-7x+12\right)(5-x)>0$.
			\item $\dfrac{2(x-1)^2+1}{x^2-x-6}+\dfrac{1}{2}\leq 0$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
