\begin{bt}%[Đề thi Học Kì 1, 2016 - 2017 trường THPT Nguyễn Thị Minh Khai, Hồ Chí Minh]%[Phạm Doãn Lê Bình, 11-HK1-18]%[1D1Y3-5]
	Giải phương trình $\cos^2 x - \sin^2 x = \sin 3x + \cos 4x$.
\loigiai{
	Phương trình ban đầu tương đương với 
		\begin{align*}
		& \cos 2x = \sin 3x + \cos 4x \Leftrightarrow \sin 3x + \cos 4x - \cos 2x = 0 \\
		\Leftrightarrow & \sin 3x - 2\sin 3x \sin x = 0 \Leftrightarrow \sin 3x (1-2\sin x)=0\\
		\Leftrightarrow & \hoac{& \sin 3x = 0 \\ & \sin x = \dfrac{1}{2}} \Leftrightarrow \hoac{& x = \dfrac{k\pi}{3} \\ & x = \dfrac{\pi}{6}+k2\pi \\ & x = \dfrac{5\pi}{6}+k2\pi}, (k\in\mathbb{Z}).
		\end{align*}
Vậy tập nghiệm của hệ phương trình là $S=\{ \dfrac{k\pi}{3}; \dfrac{\pi}{6}+k2\pi; \dfrac{5\pi}{6}+k2\pi|k\in\mathbb{Z}\}$.
}
\end{bt}

\begin{bt}%[Đề thi Học Kì 1, 2016 - 2017 trường THPT Nguyễn Thị Minh Khai, Hồ Chí Minh]%[Phạm Doãn Lê Bình, 11-HK1-18]%[1D1K3-6]
	Giải phương trình $\dfrac{\sqrt{3}\sin 2x+\cos 2x-2\cos \left(\dfrac{\pi}{3}-x \right)}{\sin x} = 0$.
	\loigiai{
		Điều kiện của phương trình $x\ne k\pi, k \in \mathbb{Z}$.
		Phương trình ban đầu tương đương với 
		\begin{align*}
		& \sqrt{3}\sin 2x+\cos 2x = 2 \cos \left(\dfrac{\pi}{3}-x \right) \Leftrightarrow \dfrac{\sqrt{3}}{2}\sin 2x + \dfrac{1}{2}\cos 2x = \cos \left(\dfrac{\pi}{3}-x \right)\\
		\Leftrightarrow & \sin \dfrac{\pi}{3}\sin 2x + \cos \dfrac{\pi}{3}\cos 2x = \cos \left(\dfrac{\pi}{3}-x \right) 
		\Leftrightarrow \cos \left( 2x - \dfrac{\pi}{3}\right) = \cos \left(\dfrac{\pi}{3}-x \right)\\
		\Leftrightarrow & \hoac{& 2x - \dfrac{\pi}{3} = \dfrac{\pi}{3} - x + m2\pi\\ & 2x - \dfrac{\pi}{3} = - \dfrac{\pi}{3} + x + m 2\pi} \Leftrightarrow \hoac{& x = \dfrac{2\pi}{9} + \dfrac{m2\pi}{3} \\ & x = m2\pi}, (m\in\mathbb{Z}). 
		\end{align*}
	So sánh với điều kiện ban đầu, ta có họ nghiệm của phương trình là $x=\dfrac{2\pi}{9} + \dfrac{m2\pi}{3}$, ($m\in\mathbb{Z}$).
	}
\end{bt}

\begin{bt}%[Đề thi Học Kì 1, 2016 - 2017 trường THPT Nguyễn Thị Minh Khai, Hồ Chí Minh]%[Phạm Doãn Lê Bình, 11-HK1-18]%[1D2B5-2]
	Lớp $11A$ có $15$ học sinh nam và $25$ học sinh nữ. Lớp $11B$ có $12$ học sinh nam và $18$ học sinh nữ. Trường chọn ngẫu nhiên từ mỗi lớp ra $2$ học sinh để tham gia vào đội nhảy cổ động. Gọi $A$ là biến cố ``Trong $4$ học sinh được chọn có $2$ nam và $2$ nữ''. Tính xác suất của biến cố $A$.
	\loigiai{
		Số phần tử của không gian mẫu là $n(\Omega)= \mathrm{C}_{40}^2 \cdot \mathrm{C}_{30}^2 = 339300$.
		\begin{itemize}
		\item {\bf TH1:} Chọn $2$ nam lớp $11A$, $2$ nữ lớp $11B$, có $\mathrm{C}_{15}^2 \cdot \mathrm{C}_{18}^2=16065$ cách.
		\item {\bf TH2:} Chọn $2$ nữ lớp $11A$, $2$ nam lớp $11B$, có $\mathrm{C}_{25}^2 \cdot \mathrm{C}_{12}^2=19800$ cách.
		\item {\bf TH3:} Mỗi lớp chọn $1$ nam, $1$ nữ, có $\mathrm{C}_{15}^1 \cdot \mathrm{C}_{25}^1 \cdot \mathrm{C}_{12}^1 \cdot \mathrm{C}_{18}^1=81000$ cách.
		\end{itemize}
		Số phần tử của biến cố $A$ là $n(A)=16065 + 19800 + 81000 = 116865$.\\
		Vậy xác suất của biến cố $A$ là $P(A)=\dfrac{n(A)}{n(\Omega)}=\dfrac{116865}{339300}=\dfrac{2597}{7540}$.
	}
\end{bt}

\begin{bt}%[Đề thi Học Kì 1, 2016 - 2017 trường THPT Nguyễn Thị Minh Khai, Hồ Chí Minh]%[Phạm Doãn Lê Bình, 11-HK1-18]%[1D2B2-6]
	Giải bất phương trình $\mathrm{A}_x^3 + 2 \mathrm{C}_x^{x-2}\le 9x$.
	\loigiai{
		Điều kiện của bất phương trình: $x\in\mathbb{N}$, $x \ge 3$.\\
		Bất phương trình ban đầu tương đương với $$\dfrac{x!}{(x-3)!}+ 2 \dfrac{x!}{(x-2)! 2!} \le 9x \Leftrightarrow (x-1)^2 \le 9 \Leftrightarrow -2 \le x \le 4.$$
		So với điều kiện ban đầu, ta có nghiệm của bất phương trình là $x\in\{3;4\}$.
	}
\end{bt}

\begin{bt}%[Đề thi Học Kì 1, 2016 - 2017 trường THPT Nguyễn Thị Minh Khai, Hồ Chí Minh]%[Phạm Doãn Lê Bình, 11-HK1-18]%[1D2K3-2]
	Cho khai triển nhị thức Newton của $P(x)=\left( 3x^5 - \dfrac{1}{x} \right)^{3n}$ (với $n\in\mathbb{N}^*$) và biết rằng tổng các hệ số của khai triển là $32768$. Tìm hệ số của số hạng chứa $x^9$ trong khai triển.
	\loigiai{
		Cho $x=1$ ta có $2^{3n}=32768 \Leftrightarrow n = 5$.\\
		$\displaystyle P(x) = \sum \limits_{k=0}^{15} \mathrm{C}_{15}^k 3^{15-k} (-1)^k x^{75-6k}$.\\
		Cho $75-6k=9\Leftrightarrow k=11$.\\
		Vậy hệ số cần tìm là $-110565$.
	}
\end{bt}

\begin{bt}%[Đề thi Học Kì 1, 2016 - 2017 trường THPT Nguyễn Thị Minh Khai, Hồ Chí Minh]%[Phạm Doãn Lê Bình, 11-HK1-18]%[1D3K3-3]
	Một cấp số cộng $(u_n)$ có $5$ số hạng và có $u_1<0$ và công sai là $d\ne 0$. Hãy tìm các số hạng của cấp số cộng đó, biết rằng $\heva{& \dfrac{1}{u_1 u_2} + \dfrac{1}{u_2 u_3} + \dfrac{1}{u_3 u_4} + \dfrac{1}{u_4 u_5} = \dfrac{1}{2} \\ & u_5 = 2u_1.}$
	\loigiai{
	Ta có $\dfrac{1}{u_1u_2} = \dfrac{1}{u_2-u_1}\cdot \dfrac{u_2-u_1}{u_1u_2} = \dfrac{1}{d} \left( \dfrac{1}{u_1} - \dfrac{1}{u_2} \right)$. \\
	Chứng minh tương tự ta có $\dfrac{1}{u_2u_3} = \dfrac{1}{d}\left(\dfrac{1}{u_2}-\dfrac{1}{u_3}\right), \ldots, \dfrac{1}{u_4u_5} = \dfrac{1}{d}\left(\dfrac{1}{u_4}-\dfrac{1}{u_5}\right).$\\
		Hệ phương trình ban đầu tương đương với 
		\begin{align*}
		& \heva{& \dfrac{1}{d} \left( \dfrac{1}{u_1} - \dfrac{1}{u_2} + \dfrac{1}{u_2}- \dfrac{1}{u_3} + \dfrac{1}{u_3} - \dfrac{1}{u_4} + \dfrac{1}{u_4} - \dfrac{1}{u_5} \right) = \dfrac{1}{2} \\ & u_5 = 2u_1}\\
		\Leftrightarrow & \heva{& \dfrac{1}{d} \cdot \dfrac{u_5-u_1}{u_1u_5} = \dfrac{1}{2} \\ & u_5 = 2u_1} \Leftrightarrow  \heva{& u_1 u_5 = 8 \\ & u_5 = 2u_1} \Leftrightarrow \heva{& u_1 = -2 \\ & u_5 = -4}
		\Leftrightarrow \heva{& u_1 = -2 \\ & d = - \dfrac{1}{2}.}
		\end{align*}
	Vậy cấp số cộng cần tìm là $-2;-\dfrac{5}{2}; -3;-\dfrac{7}{2};-4$.
	}
\end{bt}

\begin{bt}%[Đề thi Học Kì 1, 2016 - 2017 trường THPT Nguyễn Thị Minh Khai, Hồ Chí Minh]%[Phạm Doãn Lê Bình, 11-HK1-18]%[1H2Y3-5]%[1H2B3-3]%[1H2B3-4]%[1H2B2-5]
	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình bình hành tâm $O$. Gọi $M$ là trung điểm của $SC$ và $G$ là trọng tâm tam giác $ABC$.
		\begin{enumerate}[a)]
			\item Tìm giao điểm $I$ của $AM$ và mặt phẳng $(SBD)$. Chứng minh $I$ là trọng tâm tam giác $SBD$.
			\item Chứng minh $IG$ song song với mặt phẳng $(SAB)$.
			\item Mặt phẳng $(P)$ chứa $AM$ và song song với $BD$ cắt $SB,SD$ lần lượt tại hai điểm $E$ và $F$. Tìm thiết diện của mặt phẳng $(P)$ và hình chóp $S.ABCD$. 
			\item Gọi $K$ là giao điểm của $ME$ và $BC$, $J$ là giao điểm của $MF$ và $CD$. Chứng minh ba điểm $K,A,J$ nằm trên một đường thẳng song song với $EF$. Tính tỉ số $\dfrac{EF}{KJ}$. 
		\end{enumerate}
	\loigiai{
	\begin{center}
	\begin{tikzpicture}
	\tkzDefPoints{0/0/C, -2/-2/B, 3/-2/A}
	\coordinate (D) at ($(A)+(C)-(B)$);
	\coordinate (O) at ($(A)!.5!(C)$);
	\coordinate (S) at ($(O)+(0,5)$);
	\tkzDefMidPoint(C,S)	\tkzGetPoint{M}
	\tkzDefMidPoint(C,B)	\tkzGetPoint{a}
	\tkzDefPointBy[homothety=center A ratio 0.667](a)\tkzGetPoint{G}
	\tkzInterLL(A,M)(S,O)	\tkzGetPoint{I}
	\tkzDefPointBy[translation = from O to G](I)	\tkzGetPoint{i}
	\tkzInterLL(S,B)(I,i)	\tkzGetPoint{E}
	\tkzInterLL(S,D)(I,i)	\tkzGetPoint{F}
	\tkzInterLL(M,E)(B,C)	\tkzGetPoint{K}
	\tkzInterLL(M,F)(D,C)	\tkzGetPoint{J}
	\tkzDrawSegments[dashed](S,C A,B A,D A,C B,D S,O A,M I,G E,F M,E M,F C,J C,K E,B F,D)
	\tkzDrawSegments(S,E S,A S,F J,K E,K F,J A,E A,F)
	\tkzLabelPoints[left](E,K)
	\tkzLabelPoints[right](J)
	\tkzLabelPoints[above](S)
	\tkzLabelPoints[below](G,A,B,C,D)
	\tkzLabelPoints[above right](F,O,M,I)
	%\tkzLabelPoints[above left](I)
	\tkzDrawPoints(S,A,B,C,D,M,G,O,I,E,F,J,K)
	\end{tikzpicture}
	\end{center}
	\begin{enumerate}[a)]
		\item Gọi $O$ là giao điểm của $AC$ và $BD$.\\
		Trong mặt phẳng $(SAC)$, $AM$ cắt $SO$ tại $I$. Ta có $\heva{& I \in AM \\ & I \in SO \subset (SBD)} \Rightarrow I = AM \cap (SBD)$.\\
		Tam giác $SAC$ có hai đường trung tuyến $AM$, $SO$ cắt nhau ở $I$ nên $I$ là trọng tâm tam giác $SAC$, suy ra $\dfrac{SI}{SO} = \dfrac{2}{3}$.\\
		Xét tam giác $SBD$ có đường trung tuyến $SO$; $I$ là một điểm thuộc $SO$ và thỏa $\dfrac{SI}{SO} = \dfrac{2}{3}$ nên $I$ là trọng tâm tam giác $SBD$.
		\item Xét tam giác $SOB$, có $\dfrac{SI}{SO} = \dfrac{BG}{BO} = \dfrac{2}{3}$ nên suy ra $IG \parallel SB$, từ đó suy ra $IG \parallel (SBD)$.
		\item Ta có $\heva{& (P)\cap (SBD)=EF \\ & BD \parallel (P)} \Rightarrow BD \parallel EF$. \\
		Vậy, qua $I$ kẻ đường thẳng song song với $BD$,  cắt $SB, SD$ lần lượt tại $EF$. Thiết diện của $(P)$ và $S.ABCD$ là tứ giác $AEMF$.
		\item Ta có $K,A,J$ thẳng hàng vì cùng nằm trên giao tuyến của hai mặt phẳng $(P)$ và $(ABCD)$.\\
		Ta có $(P)\cap (SBD)=EF$; $(P)\cap (ABCD)=KJ$; $(SBD)\cap (ABCD)=BD)$. Mà $EF \parallel BD$ nên suy ra $EF \parallel KJ$.\\
		Ta có $\dfrac{EF}{KJ} = \dfrac{MI}{MA} = \dfrac{1}{3}$. 
	\end{enumerate}
}
\end{bt}
