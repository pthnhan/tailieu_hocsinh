\begin{ex}[Đề thi vào 10, Sở giáo dục Lào Cai, 2016][9D1B8]
\newline
Cho biểu thức $P=\left(\dfrac{x}{x-2\sqrt{x}}+\dfrac{x}{\sqrt{x}-2}\right):\dfrac{\sqrt{x}+1}{x-4\sqrt{x}+4}$ với $x>0,x\ne 4$.
    \begin{enumerate}    
        \item[a.] Rút gọn biểu thức $P$.
        \item[b.] Tìm tất cả các giá trị của $x$ để $P>0$.
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item[a.] Với điều kiện $x>0,x\ne 4$ ta có:\\
        $
        \begin{aligned}
        P=&\left(\dfrac{x}{x-2\sqrt{x}}+\dfrac{x}{\sqrt{x}-2}\right):\dfrac{\sqrt{x}+1}{x-4\sqrt{x}+4}
        =\left(\dfrac{\left(\sqrt{x}\right)^2}{\sqrt{x}\left(\sqrt{x}-2\right)}+\dfrac{x}{\sqrt{x}-2}\right):\dfrac{\sqrt{x}+1}{\left(\sqrt{x}-2\right)^2}\\
        &=\left(\dfrac{\sqrt{x}}{\sqrt{x}-2}+\dfrac{x}{\sqrt{x}-2}\right).\dfrac{\left(\sqrt{x}-2\right)^2}{\sqrt{x}+1}
        =\dfrac{\sqrt{x}+x}{\sqrt{x}-2}.\dfrac{\left(\sqrt{x}-2\right)^2}{\sqrt{x}+1}\\
        &=\dfrac{\sqrt{x}\left(\sqrt{x}+1\right)}{\sqrt{x}-2}.\dfrac{\left(\sqrt{x}-2\right)^2}{\sqrt{x}+1}=\sqrt{x}\left(\sqrt{x}-2\right).
        \end{aligned}
        $
        \item [b.] Với điều kiện $x>0,x\ne 4$ ta có:\\
         $
        \begin{aligned}
        P>0&\Leftrightarrow\sqrt{x}\left(\sqrt{x}-2\right)>0\Leftrightarrow \sqrt{x}-2>0\\
        &\Leftrightarrow \sqrt{x}>2\Leftrightarrow x>4.
        \end{aligned}
        $\\
        Kết hợp điều kiện ta có $x>4$ thỏa mãn yêu cầu bài toán.
    \end{enumerate}
    }
\end{ex}

\begin{ex}[Đề thi vào 10, Sở giáo dục Lào Cai, 2016][9D2B4][9D3B4]
    \begin{enumerate}
        \item[1.] Tìm tất cả các giá trị của tham số $a$ để đường thẳng $(d): y=3x+2$ song song với đường thẳng $(d'): y=\left(2a-1\right)x+8$.
        \item[2.] Cho hệ phương trình $\heva{&2x+y=3\\&x-2y=m}\ (I)$.
        \begin{itemize}
        \item[a.] Giải hệ phương trình $(I)$ khi $m=-1$.
        \item[b.] Tìm $m$ để hệ phương trình $(I)$ có nghiệm duy nhất $\left(x;y\right)$ sao cho $x>y$.
        \end{itemize}
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item[1.] Đường thẳng $(d)$ song song với đường thẳng $(d')$ khi và chỉ khi:\\ $\heva{&2a-1=3\\&8\ne 2}\Leftrightarrow 2a-1=3\Leftrightarrow a=2$.
        \item[2.]
        \begin{itemize}
        \item[a.] Khi $m=-1$, hệ $(I)$ có trở thành:\\
        $\heva{&2x+y=3\\&x-2y=-1}\Leftrightarrow \heva{&2x+y=3\\&2x-4y=-2}\Leftrightarrow\heva{&5y=5\\&2x+y=3}\Leftrightarrow\heva{&x=1\\&y=1}$.
        
      
        Vậy hệ phương trình có nghiệm: $(x;y)=(1;1)$.
        \item[b.] $\heva{&2x+y=3\\&x-2y=m}\Leftrightarrow \heva{&5y=3-2m\\&2x+y=3}\Leftrightarrow \heva{&y=\dfrac{3-2m}{5}\\&x=\dfrac{m+6}{5}}$.\\
        Hệ trên luôn có nghiệm duy nhất $(x;y)=\left(\dfrac{m+6}{5};\dfrac{3-2m}{5}\right)$ với mọi $m$.\\
        Khi đó: $x>y\Leftrightarrow \dfrac{m+6}{5}>\dfrac{3-2m}{5}\Leftrightarrow m+6>3-2m\Leftrightarrow m>-1$.\\
        Vậy với $m>-1$ hệ phương trình có nghiệm duy nhất thỏa mãn $x>y$.
        \end{itemize}
    \end{enumerate}
    }
\end{ex}
\begin{ex}[Đề thi vào 10, Sở giáo dục Lào Cai, 2016][9D4B4][9D4K6]
\newline
Cho phương trình: $x^2-2x+2m-5=0$ \  $(1)$ ($m$ là tham số)
\begin{enumerate}
\item[a.] Giải phương trình $(1)$ khi $m=-5$.
\item[b.] Tìm $m$ để phương trình $(1)$ có hai nghiệm phân biệt $x_1,x_2$ sao cho biểu thức $A=\left(x_1^2-2\right)\left(x_2^2-2\right)$ đạt giá trị nhỏ nhất.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a.] Khi $m=-5$, phương trình đã cho trở thành: $x^2-2x-15=0$.\\
Ta có: $\Delta = b^2-4ac=64>0\Rightarrow $ Phương trình có hai nghiệm phân biệt là:
\begin{center}
$x_1=\dfrac{2+\sqrt{64}}{2}=5$ và $x_2=\dfrac{2-\sqrt{64}}{2}=-3$.
\end{center}
\item[b.] Phương trình có hai nghiệm phân biệt khi và chỉ khi: $\Delta '=6-2m>0\Leftrightarrow m<3$.\\
Áp dụng định lý Vi-ét ta có: $\heva{&x_1+x_2=2\\&x_1x_2=2m-5}$.\\
Khi đó:\\$
\begin{aligned}
A&=\left(x_1^2-2\right)\left(x_2^2-2\right)=\left(x_1x_2\right)^2-2\left(x_1^2+x_2^2\right)+4\\
&=\left(x_1x_2\right)^2-2\left(x_1+x_2\right)^2+4x_1x_2+4=4m^2-12m+1\\&=\left(2m-3\right)^2-8\ge -8,\ \forall m\in\mathbb{R}.
\end{aligned}
$\\
Vậy $A$ đạt giá trị nhỏ nhất bằng $-8$ khi $m=\dfrac{3}{2}$.
\end{enumerate}
}
\end{ex}

\indent Cho nửa đường tròn $(O)$ đường kính $AB=2R$. Dựng đường thẳng $d$ là tiếp tuyến của $(O)$ tại điểm $A$. Trên cung $AB$ lấy điểm $C$ tùy ý ( $C$ khác $A$ và $B$). Tia $BC$ cắt đường thẳng $d$ tại điểm $D$. Gọi $I$ là trung điểm của $BC$. Tia $IO$ cắt đường thẳng $d$ tại điểm $K$.
    \begin{enumerate}
        \item[a.] Chứng minh rằng tứ giác $OADI$ là tứ giác nội tiếp.
        \item[b.] Chứng minh rằng $IB.ID=IO.IK$.
        \item[c.] Xác định vị trí của điểm $C$ trên cung $AB$ để $BD+4BI$ đạt giá trị nhỏ nhất.
    \end{enumerate}
\loigiai{
    \begin{center}
     \begin{tikzpicture}[scale=.8]
    \tkzInit[xmin=-4,xmax=4,ymin=-5,ymax=5]
\tkzClip
  \tkzDefPoint(-3,0){B} \tkzDefPoint(3,0){A}\tkzDefPoint(0,0){O}

\tkzDrawCircle[R](O,\rOA cm)
\tkzDefPointBy[rotation= center O angle 70](A)
\tkzGetPoint{C}
\tkzDefMidPoint(B,C) \tkzGetPoint{I}
\tkzDefLine[perpendicular=through A,
K=1](A,B)
\tkzGetPoint{c}\tkzDrawLine(A,c)
\tkzDefLine[perpendicular=through A,
K=-1](A,B)
\tkzGetPoint{d}\tkzDrawLine(A,d)
\tkzInterLL(A,c)(B,C) \tkzGetPoint{D}
\tkzInterLL(I,O)(A,d) \tkzGetPoint{K}
\tkzDrawPoints(A,B,O,C,D,I,K)
\tkzLabelPoints[left](B)\tkzLabelPoints[right](A)\tkzLabelPoints[below left](O)\tkzLabelPoints[above](C)\tkzLabelPoints[right](D)\tkzLabelPoints[above left](I)\tkzLabelPoints[right](K)

    \begin{enumerate}
        \item[a.] Vì $d$ là tiếp tuyến của đường tròn $(O)$ tại $A$ nên $OA\perp AD\Rightarrow \widehat{OAD}=90^\circ$  $(1)$.\\
        Vì $I$ là trung điểm của $BC$ nên $OI\perp BC\Rightarrow \widehat{OID}=90^\circ$  $(2)$.\\
        Từ $(1)$ và $(2)$ ta có: $\widehat{OAD}+\widehat{OID}=180^\circ$.\\
        Suy ra tứ giác $OADI$ nội tiếp một đường tròn.
        \item[b.] Xét hai tam giác $\Delta IBO$ và $\Delta IKD$ có:\\
        $\circ$ $\widehat{BIO}=\widehat{KID}=90^\circ$.\\
        $\circ$ $\widehat{IBO}=\widehat{IKD}$ (cùng phụ với $\widehat{IDK}$).\\
        Suy ra: $\Delta IBO\backsim\Delta IKD$ (g-g) $\Rightarrow \dfrac{IB}{IK}=\dfrac{IO}{ID}\Leftrightarrow IB.ID=IO.IK$.
        \item[c.] Áp dụng bất đẳng thức Cô-si ta có:\\
        $BD+4BI=BD+2BC\ge 2\sqrt{2BD.BC}=2\sqrt{2.AB^2}=4R\sqrt{2}$.\\
        Vậy $BD+4BI$ nhỏ nhất bằng $4R\sqrt{2}$ đạt được khi và chỉ khi:\\
        $BD=2BC\Leftrightarrow C$ là trung điểm của $BD\Leftrightarrow C$ là điểm chính giữa cung $AB$. 
    \end{enumerate}
    }
