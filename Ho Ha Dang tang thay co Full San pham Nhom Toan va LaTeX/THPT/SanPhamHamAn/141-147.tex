\begin{ex}%[WTT Bài toán hàm ẩn, Phan Quốc Trí]%[2D1G2-2]
\immini{
Cho hàm số $y=f(x)$ có đồ thị như hình bên. Hàm số $g(x)=f(f(x))$ có bao nhiêu điểm cực trị?
	\choice
	{$3$}
	{\True $4$}
	{$5$}
	{$6$}
	}{
	\begin{tikzpicture}[scale=0.6]
\def \xmin{-1.5}
\def \xmax{4.0}
\def \ymin{-5}
\def \ymax{2.3}
\tkzDefPoints{0/0/O, 2/-4/M}
\draw[->] (\xmin,0) -- (\xmax,0);
\draw[->,color=black] (0,\ymin) -- (0,\ymax);

\draw[color=black] (.1,\ymax) node[right] {\normalsize $y$};

\begin{scope}

%Vẽ đồ thị
\draw[color=black,smooth,samples=700,domain=\xmin+0.3:\xmax-0.3]

\end{scope}
%vẽ râu ria
\draw[dashed] (0,-4)--(2,-4)--(2,0);
\node at (2,0.4){ 2};
\node at (-0.4,-4){-4};
\tkzDrawPoints[fill=black](O,M)
\end{tikzpicture} 
	}
	\loigiai{
\begin{description}
\item[Cách 1]
Dựa vào đồ thị ta thấy $f(x)$ đạt cực trị tại $x=0$ và $x=2$. \\
Suy ra $f'(x)=\hoac{&x=0 \text{ (nghiệm đơn) }\\&x=2 \text{ (nghiệm đơn)} }$. Ta có $g'(x)=f'(x)\cdot f' \left[ f(x) \right]$\\
nên $g'(x)=0 \Leftrightarrow \hoac{&f'(x)=0\\& f'\left[ f(x) \right]=0. } \Leftrightarrow \hoac{&x=0 \text{ (nghiệm đơn) }\\&x=2 \text{ (nghiệm đơn)}\\& f(x)=0 \quad (1) \\& f(x) = 2. \quad (2) }$
\begin{center}
\begin{tikzpicture}[scale=0.6]
\def \xmin{-1.5}
\def \xmax{5.5}
\def \ymin{-5}
\def \ymax{3.3}
\tkzDefPoints{0/0/O, 2/-4/M}
\draw[->] (\xmin,0) -- (\xmax,0);
\draw[->,color=black] (0,\ymin) -- (0,\ymax);
\draw[color=black](\xmax-0.3,.2)node[right] {$x$};
\draw[color=black] (.1,\ymax) node[right] {\normalsize $y$};
\draw[color=black] (0pt,8pt) node[left] {\small $O$};
\begin{scope}
\clip(\xmin ,\ymin) rectangle (\xmax-.3,\ymax-.3);
%Vẽ đồ thị
\draw[color=black,smooth,samples=700,domain=\xmin+0.3:\xmax-0.3] 
plot(\x,{(\x)^3-3*(\x)^2+0*(\x)});
\end{scope}
%vẽ râu ria
\draw[dashed] (0,-4)--(2,-4)--(2,0);
\draw (\xmin+0.2,2)--(\xmax-0.2,2);
\node at (2,0.4){ 2};
\node at (-0.4,-4){-4};
\node at (4.5,2.4){$ y=2 $};
\tkzDrawPoints[fill=black](O,M)
\end{tikzpicture} 
\end{center}
Dựa vào đồ thị ta có:
\begin{itemize}
\item Phương trình $(1)$ có hai nghiệm $x=0$ (nghiệm kép) và $x=a \; (a>2)$.
\item  Phương trình $(2)$ có một nghiệm $x=b \; (b>a)$
\end{itemize}
Vậy phương trình $g'(x)=0$ có bốn nghiệm bội lẻ là $x=0$, $x=2$, $x=a$ và $x=b$. Suy ra hàm số $g(x)=f(f(x))$ có $4$ điểm cực trị.
\item[Cách 2] 
Với $u=f(x)$ thì $f'(f(x))_x = f^{'}_u \cdot u'_x = f^{'}_u \cdot f^{'}_x$ 
$$\Rightarrow f'(f(x)) =0 \Leftrightarrow \hoac{&f'_u=0\\&f'_x = 0} \Leftrightarrow \hoac{&u=f(x)=0\\&u=f(x)=2\\&x=0\\&x=2}$$
\begin{itemize}
\item Ta thấy $f(x)=0$ có hai nghiệm $x_{1,2}=0$ và $x_3>2$.
\item Ta thấy $f(x)=2$ có hai nghiệm $x_4>x_3$
\end{itemize}
$\Rightarrow f'(f(x))=0$ có nghiệm $x=0$ bậc 3, $x=2$, $x_3$, $x_4$ bậc 1, suy ra hàm số có 4 cực trị.
\end{description}
	}
\end{ex}

$$g'(x)=0 \Leftrightarrow \hoac{& x=0\\& f'(x^2+1)=0} \Leftrightarrow \hoac{& x=0\\& x^2-1=-2\\&x^2-1=1} \Leftrightarrow \hoac{& x=0 \text{ (nghiệm đơn)} \\ & x=0 \text{ (nghiệm kép)}} \Leftrightarrow x=0 \text{ (nghiệm bội 3)} $$

\begin{ex}%[WTT Bài toán hàm ẩn, Phan Quốc Trí]%[2D1G1-2]
Cho hàm số $y=f(x)$ có bảng biến thiên như hình vẽ.
\begin{center}
	\begin{tikzpicture}
	\tkzTabInit[nocadre=false, lgt=1, espcl=2.0]{$x$ /0.6,$y'$ /0.6,$y$ /2}{$-\infty$,$-2$,$0$,$2$,$+\infty$}
	\tkzTabLine{,+,$0$,-,$0$,+,$0$,-,}
	\tkzTabVar{-/ $+\infty$,+/$3$,-/$-1$,+/$3$,-/$-\infty$}
	\end{tikzpicture}
\end{center}
Hàm số $g(x)=f(x^2-2)$ nghịch biến trên khoảng nào dưới đây?
	\choice
	{$(-2;0)$}
	{\True $(2;+\infty )$}
	{$(0;2)$}
	{$(-\infty; -2)$}
	\loigiai{
Ta có $y'=2x \cdot f'(x^2-2)$.
\begin{eqnarray*}
y'>0 & \Leftrightarrow & \hoac{&\heva{&x>0\\ & f'(x^2-2) > 0 } \\&\heva{&x<0 \\ & f'(x^2-2)<0}} \Leftrightarrow 
 \hoac{&\heva{&x>0\\ & \hoac{&x^2-2<-2\\& 0 < x^2-2 <2} } \\&\heva{&x<0 \\ & \hoac{& x^2-2>2\\ & -2< x^2 -2 <0}}}\\
 & \Leftrightarrow & \hoac{& \sqrt{2} < x <2\\ & x<-4 \\ & -\sqrt{2} <x < 0.}
\end{eqnarray*}
Do đó hàm số $y=f(x^2-2)$ đồng biến trên các khoảng $(-\infty;-4)$, $(-\sqrt{2}; 0)$, $(\sqrt{2}; 2)$ và nghịch biến trên các khoảng $\left( -4; -\sqrt{2} \right)$, $\left(0; \sqrt{2} \right)$, $(2; +\infty )$.
	}
\end{ex}
