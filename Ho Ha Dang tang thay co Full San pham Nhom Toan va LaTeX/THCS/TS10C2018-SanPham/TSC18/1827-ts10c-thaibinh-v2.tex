\begin{ex}%[Đề thi vào 10, chuyên Thái Bình, năm 2018]%[Nguyễn Quang Hiệp, Dự án EX-TS10C-18]%[9D3G6]
Cho phương trình $x^2-2mx+m^2-2m+4=0$ (với $m$ là tham số). Tìm $m$ để phương trình có hai nghiệm không âm $x_1,x_2$. Tính theo $m$ giá trị biểu thức $P=\sqrt{x_1}+\sqrt{x_2}$ và tìm giá trị nhỏ nhất của $P$.
\loigiai{
Phương trình: $x^2-2mx+m^2-2m+4=0$ \tagEX{1}
\noindent
Phương trình $(1)$ có hai nghiệm không âm $x_1,x_2$\\
$\Leftrightarrow \heva{
& \Delta '\ge 0 \\ 
& x_1+x_2\ge 0 \\ 
& x_1x_2\ge 0 }\Leftrightarrow \heva{
& m^2-(m^2-2m+4)\ge 0 \\ 
& 2m\ge 0 \\ 
& m^2-2m+4\ge 0 }\Leftrightarrow \heva{
& 2m-4\ge 0 \\ 
& m\ge 0 \\ 
& (m-1)^2+3\ge 0 }\Leftrightarrow m\ge 2$.\\
Xét $P=\sqrt{x_1}+\sqrt{x_2}\ge 0$
\begin{align*}
&\Rightarrow P^2={\left(\sqrt{x_1}+\sqrt{x_2}\right)}^2=x_1+x_2+2\sqrt{x_1x_2}=2m+2\sqrt{m^2-2m+4} \\ 
& \Rightarrow P=\sqrt{2m+2\sqrt{m^2-2m+4}}. 
\end{align*}
Với $m\ge 2$, ta có:\\
$\begin{aligned}
& P^2=2m+2\sqrt{m(m-2)+4}\ge 2.2+2\sqrt{0+4}=8 \\ 
& \Rightarrow P\ge 2\sqrt{2}. 
\end{aligned}$
Dấu ``='' xảy ra $\Leftrightarrow m=2$.\\
Vậy $\min P=2\sqrt{2}$ khi $m=2$.
}
\end{ex}

Với $x\in \mathbb{Z}$, ta có:
$y\in \mathbb{Z}\Leftrightarrow \dfrac{6}{x+2}\in \mathbb{Z}\Leftrightarrow x+2 \in$ Ư($6$) \\
hay $x+2\in \left\{1;-1;2;-2;3;-3;6;-6\right\}\Leftrightarrow x\in \left\{-1;-3;0;-4;1;-5;4;-8\right\}$.\\
Vậy $x\in \left\{-1;-3;0;-4;1;-5;4;-8\right\}$ là các giá trị cần tìm.
 
}

\begin{ex}%[Đề thi vào 10, chuyên Thái Bình, năm 2018]%[Nguyễn Quang Hiệp, Dự án EX-TS10C-18]%[9D4G4]
	Cho các số $a,\ b,\ c$ thỏa mãn điều kiện $a+2b+5c=0$. Chứng minh phương trình $ax^2+bx+c=0$ có nghiệm.
	\loigiai{
	Phương trình $ax^2+bx+c=0$ \tagEX{1}
	\noindent
	Xét $2$ trường hợp:
	   \begin{enumerate}[\it Trường hợp 1.]
	   	\item Với $a=0 \Rightarrow $ phương trình $(1)$ trở thành $bx+c=0$ \tagEX{2}
		\begin{enumerate}[+]
			\item Nếu $b=0$ thì từ điều kiện $a+2b+5c=0$ suy ra $c=0$.\\
			$\Rightarrow $ Phương trình $(2)$ nghiệm đúng với mọi $x.$\\
			$\Rightarrow $ Phương trình $(1)$ có nghiệm.
			\item Nếu $b\ne 0$ thì phương trình $(2)$ có nghiệm duy nhất $x=-\dfrac{c}{b}$
			$\Rightarrow $ Phương trình $(1)$ có nghiệm.
		\end{enumerate}
			\item Với $a\ne 0 \Rightarrow $ phương trình $(1)$ là phương trình bậc hai ẩn $x.$\\
			Từ $a+2b+5c=0\Rightarrow b=-\dfrac{a+5c}{2}\Rightarrow b^2=\dfrac{(a+5c)^2}{4}$. Do đó:
			\begin{align*}
			\Delta &=b^2-4ac=\dfrac{(a+5c)^2}{4}-4ac=\dfrac{a^2+10ac+25c^2-16ac}{4} \\ 
			& =\dfrac{a^2-6ac+25c^2}{4}=\dfrac{a^2-6ac+9c^2+16c^2}{4}=\dfrac{(a-3c)^2}{4}+4c^2\ge 0.
			\end{align*}
			$\Rightarrow $ Phương trình $(1)$ có nghiệm.
	   \end{enumerate}
	   Kết luận: Phương trình $(1)$ luôn có nghiệm với các số $a,\ b,\ c$ thỏa mãn điều kiện $a+2b+5c=0$.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Thái Bình, năm 2018]%[Nguyễn Quang Hiệp, Dự án EX-TS10C-18]%[9D0G2]
	Giải phương trình $(4x^3-x+3)^3=x^3+\dfrac{3}{2}$.
	\loigiai{
 $(4x^3-x+3)^3=x^3+\dfrac{3}{2} \Leftrightarrow 2(4x^3-x+3)=3+2x^3$.\tagEX{1}
 \noindent
Đặt $y=2x$ khi đó phương trình $(1)$ tương đương với
\begin{align*}
    2 \left(\dfrac{y^3}{2}-\dfrac{y}{2}+3 \right)^3=3+\dfrac{y^3}{4} \Leftrightarrow (y^3-y+6)^3=12+y^3. \tag{2}
\end{align*}
Đặt $z=y^3-y+6$ khi đó phương trình $(2)$ tương đương với 
$\heva{
&z=y^3-y+6\\
&y^3+12=z^3. \qquad (*)
}$\\
Ta thấy $z^3-z=y^3+12-y^3+y-6=y+6 \Rightarrow y=z^3-z-6.$\\
Nên hệ đã cho tương đương với $\heva{&z=y^3-y+6\\&y=z^3-z-6.}$\\
Từ đó ta có $(y+z)(y^2-yz+z^2-2)=0 \Leftrightarrow \hoac{&z=-y\\&y^2-yz+z^2-2=0.}$\\
Với $z=-y$ thay vào $(*)$ ta được $y^3=-6 \Leftrightarrow y = \sqrt[3]{6} \Rightarrow x=-\sqrt[3]{\dfrac{3}{4}}\cdot$\\
Với $y^2-yz+z^2-2=0$ ta có hệ $\heva{&y^2-yz+z^2-2=0\\&y^3+12=z^3} \Leftrightarrow \heva{&y^2-yz+z^2-2=0\\&(y-z)(y^2+yz+z^2)=-12. \qquad (**)} $\\
Theo bất đẳng thức Cô-si áp dụng  cho ba số không âm $(y-z)^2,\ y^2+yz+z^2,\ y^2+yz+z^2 $ ta có 
\begin{align*}
    (y-z)^2(y^2+yz+z^2)^2 &\leq \left(\dfrac{(y-z)^2+(y^2+yz+z^2)+(y^2+yz+z^2)}{3} \right)^3\\
    &=(y^2+z^2)^3 \leq (2(y^2-yz+z^2))^3=4^3<12^2
\end{align*}
Điều này mâu thuẩn với $(**)$ do đó hệ phương trình vô nghiệm.\\
Do đó nghiệm của phương trình là $S=\left\lbrace-\sqrt[3]{\dfrac{3}{4}} \right\rbrace$. 
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Thái Bình, năm 2018]%[Nguyễn Quang Hiệp, Dự án EX-TS10C-18]%[8D3G7]
	Hai cây nến cùng chiều dài và làm bằng các chất liệu khác nhau, cây nến thứ nhất cháy hết với tốc độ đều trong $3$ giờ, cây nến thứ hai cháy hết với tốc độ đều trong $4$ giờ. Hỏi phải cùng bắt đầu đốt lúc mấy giờ chiều để đến $4$ giờ chiều, phần còn lại của cây nến thứ hai dài gấp đôi phần còn lại của cây nến thứ nhất?
	\loigiai{
	   Giả sử chiều dài ban đầu của hai cây nến là $h$ (cm).\\
	Gọi thời điểm cùng bắt đầu đốt hai cây nến là $x$ (giờ)($x > 0$).
	Sau $x$ (giờ) thì:
	\begin{enumerate}[-]
	\item Cây nến thứ nhất cháy được $x\cdot \dfrac{h}{3}=\dfrac{hx}{3}$ (cm).
	\item Cây nến thứ hai cháy được $x\cdot \dfrac{h}{4}=\dfrac{hx}{4}$ (cm).
	\item Phần còn lại của cây nến thứ nhất là $h-\dfrac{hx}{3}=h\left(1-\dfrac{x}{3}\right)$ (cm).
	\item Phần còn lại của cây nến thứ hai là $h-\dfrac{hx}{4}=h\left(1-\dfrac{x}{4}\right)$ (cm).
	\end{enumerate}
	Theo đề bài ta có phương trình:
	\begin{align*}
	 h\left(1-\dfrac{x}{4}\right)=2\cdot h\left(1-\dfrac{x}{3}\right) \Leftrightarrow 1-\dfrac{x}{4}=2-\dfrac{2x}{3}  \Leftrightarrow \left(\dfrac{2}{3}-\dfrac{1}{4}\right)x=1 \Leftrightarrow x=2,4 \ (\text{thỏa mãn điều kiện}).
	\end{align*}
	\noindent
	Vậy thời điểm cùng bắt đầu đốt hai cây nến là $4-2,4 = 1,6$ (giờ) hay $1$ giờ $36$ phút chiều.

	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Thái Bình, năm 2018]%[Nguyễn Quang Hiệp, Dự án EX-TS10C-18]%[9D0G7]
	Cho các số $x, y $ dương thỏa mãn điều kiện $(x+\sqrt{1+x^2})(y+\sqrt{1+y^2})=2018$. Tìm giá trị nhỏ nhất của biểu thức $P=x+y$.
	\loigiai{
	   Từ giả thiết ta có 
		 \begin{align*}
		 	x+\sqrt{1+x^2}=\dfrac{2018}{y+\sqrt{1+y^2}}=\dfrac{2018 \left(y- \sqrt{1+x^2}\right)}{-1}=2018 \left(\sqrt{1+y^2}-y \right).\tag{1}
		 \end{align*}
		 \noindent
		Tương tự ta cũng có 
		\begin{align*}
			y+\sqrt{1+y^2}=2018 \left(\sqrt{1+x^2} -x\right). \tag{2} 
		\end{align*}
		\noindent
		Cộng vế theo vế $(1)$ và $(2)$ ta được $$2019(x+y)=2017 \left(\sqrt{1+x^2} +\sqrt{1+y^2}\right).$$
		Ta thấy 
		\begin{align*}
		    \left(\sqrt{1+x^2} +\sqrt{1+y^2} \right)^2=2+x^2+y^2+2\sqrt{(1+x^2)(1+y^2)} \ge 2+x^2+y^2+2(1+xy)=4+(x+y)^2. 
		\end{align*}
		Đẳng thức xảy ra khi $x=y$.
		Khi đó
		\begin{align*}
		    &2019(x+y) \ge 2017 \sqrt{4+(x+y)^2} \Rightarrow 2019\cdot P \geq 2017 \sqrt{4+P^2}.\\
		    &\Rightarrow2019^2\cdot P^2 \ge 2017^2.(4+P^2)\Rightarrow P^2 \ge \dfrac{2017^2}{2018} \Rightarrow P \ge \dfrac{2017}{\sqrt{2018}}.
		\end{align*}
		Vậy $\min P =\dfrac{2017}{\sqrt{2018}} \Leftrightarrow  x=y=\dfrac{2017}{2\sqrt{2018}}$. 
	}	
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Thái Bình, năm 2018]%[Nguyễn Quang Hiệp, Dự án EX-TS10C-18]%[9D0G7]
	Cho $a,\ b,\ c$ là ba số thực dương thỏa mãn điều kiện $\dfrac{1}{a^2}+\dfrac{1}{b^2}+\dfrac{1}{c^2}=1$. 
Tìm giá trị lớn nhất của biểu thức 
\begin{align*}
	P=\dfrac{1}{\sqrt{5a^2+2ab+2b^2}}+\dfrac{1}{\sqrt{5b^2+2bc+2c^2}}+\dfrac{1}{\sqrt{5c^2+2ca+2a^2}}.
\end{align*}
\loigiai{
  Trước hết chúng ta chứng minh bất đẳng thức 
    \begin{align*}
        \dfrac{a^2}{x}+\dfrac{b^2}{y}\geq \dfrac{(a+b)^2}{x+y} \tag{1}
    \end{align*}
    trong đó $a,\ b,\ x,\ y$ là các số thực và $x,\ y \geq 0.$\\
    Ta có $(1) \Leftrightarrow a^2y(x+y)+b^2x(x+y)\geq (a+b)^2xy \Leftrightarrow (ay-bx)^2 \geq 0 $ (luôn đúng).\\
    Đẳng thức xảy ra khi và chỉ khi $\dfrac{a}{x}=\dfrac{b}{y}.$ Do đó bất đẳng thức $(1)$ được chứng minh.\\
Ta tiếp tục chứng minh bất đẳng thức sau
\begin{align*}
  3(a^2+b^2+c^2)\geq (a+b+c)^2   \tag{2}
\end{align*}
Ta có
\begin{align*}
    &3(a^2+b^2+c^2)\geq (a+b+c)^2\\ 
    \Leftrightarrow &3(a^2+b^2+c^2) \geq a^2+b^2+c^2+2ab+2bc+2ca \\
    \Leftrightarrow &2(a^2+b^2+c^2)-2ab -2bc-2ca \geq 0\\ 
    \Leftrightarrow &(a-b)^2+(b-c)^2+(c-a)^2 \geq 0\ (\text{luôn đúng}).   
\end{align*}
Đẳng thức xảy ra khi và chỉ khi $a=b=c$. Do đó bất đẳng thức $(2)$ được chứng minh.\\
Ta tiếp tục chứng minh bất đẳng thức sau
\begin{align*}
    (a+b+c)\left(\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}\right) \geq 9 \tag{3}
\end{align*}
Thật vậy
$(a+b+c)\left(\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}\right)
    =3+\dfrac{a}{b}+\dfrac{b}{a}+\dfrac{b}{c}+\dfrac{c}{b}+\dfrac{c}{a}+\dfrac{a}{c}
     \geq 3+2 \sqrt{\dfrac{a}{b}\cdot \dfrac{b}{a}}+2 \sqrt{\dfrac{b}{c}\cdot \dfrac{c}{b}}+2 \sqrt{\dfrac{c}{a}\cdot \dfrac{a}{c}}=9$.
Đẳng thức xảy ra khi và chỉ khi $a=b=c$. Do đó bất đẳng thức $(3)$ được chứng minh.\\
Áp dụng bất đẳng thức $(1)$ và $(3)$ ta có
\begin{align*}
	\dfrac{1}{\sqrt{5a^2+2ab+2b^2}}=\dfrac{1}{\sqrt{(2a+b)^2+(a-b)^2}}\leq \dfrac{1}{2a+b} \leq	\dfrac{1}{9}\left(\dfrac{2}{a}+\dfrac{1}{b} \right)
\end{align*}
Chứng minh tương tự như trên ta có
\begin{align*}
	&\dfrac{1}{\sqrt{5b^2+2bc+2c^2}}=\dfrac{1}{9}\left(\dfrac{2}{b}+\dfrac{1}{c} \right)\\
	&\dfrac{1}{\sqrt{5c^2+2ca+2a^2}}=\dfrac{1}{9}\left(\dfrac{2}{c}+\dfrac{1}{a} \right).
\end{align*}
Từ đó ta được 
\begin{align*}
	P \leq \dfrac{1}{3a}+\dfrac{1}{3b}+\dfrac{1}{3c}.
\end{align*}
Áp dụng bất đẳng thức $(2)$ ta có
\begin{align*}
	\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c} \leq \sqrt{3 \left(\dfrac{1}{a^2}+\dfrac{1}{b^2}+\dfrac{1}{c^2} \right)}=\sqrt{3}.
\end{align*}
Từ đó ta có $P\leq \dfrac{\sqrt{3}}{3}$.\\
Đẳng thức xảy ra $\Leftrightarrow a=b=c=\sqrt{3}$.\\
Vậy $\max P = \dfrac{\sqrt{3}}{3} \Leftrightarrow a= b= c = \sqrt{3}$.
