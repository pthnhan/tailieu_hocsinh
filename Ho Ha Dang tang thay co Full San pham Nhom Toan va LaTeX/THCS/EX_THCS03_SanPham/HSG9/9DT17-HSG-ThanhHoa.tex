\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2016-2017, Thanh Hóa]%[9D3B3]
	Cho biểu thức $P=\left({\dfrac{x+2}{x\sqrt{x}-1}+\dfrac{\sqrt{x}}{x+\sqrt{x}+1}+\dfrac{1}{1-\sqrt{x}}}\right):\dfrac{\sqrt{x}-1}{2}$, với $0\leqslant x\ne 1$.
	\begin{enumerate}
		\item Rút gọn biểu thức $P$.
		\item Tìm $x$ để $P=\dfrac{2}{7}$.
		\item So sánh: $P^2$ và $2P$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Điều kiện $0\leqslant x\ne 1$. Ta biến đổi biểu thức $P$\\
		$P=\left({\dfrac{x+2}{x\sqrt{x}-1}+\dfrac{\sqrt{x}}{x+\sqrt{x}+1}+\dfrac{1}{1-\sqrt{x}}}\right):\dfrac{\sqrt{x}-1}{2}$.\\
	    $=\left({\dfrac{x+2}{\left(\sqrt{x}\right)^3-1}+\dfrac{\sqrt{x}}{x+\sqrt{x}+1}-\dfrac{1}{\sqrt{x}-1}}\right):\dfrac{\sqrt{x}-1}{2}$\\
		$=\dfrac{x+2+\sqrt{x}\left({\sqrt{x}-1}\right)-\left({x+\sqrt{x}+1}\right)}{\left({\sqrt{x}-1}\right)\left({x+\sqrt{x}+1}\right)}:\dfrac{\sqrt{x}-1}{2}$ 
		$=\dfrac{x-2\sqrt{x}+1}{\left({\sqrt{x}-1}\right)\left({x+\sqrt{x}+1}\right)}.\dfrac{2}{\sqrt{x}-1}$ \\
		$=\dfrac{2}{x+\sqrt{x}+1}$.
		\item Điều kiện $0\leqslant x\ne 1$. Ta có:\\
		$P=\dfrac{2}{7}\Leftrightarrow \dfrac{2}{x+\sqrt{x}+1}=\dfrac{2}{7}$
		$\Leftrightarrow x+\sqrt{x}+1=7\Leftrightarrow x+\sqrt{x}-6=0$ 
		$\Leftrightarrow \left({\sqrt{x}-2}\right)\left({\sqrt{x}+3}\right)=0$\\ 
		Vì $\sqrt{x}+3>0$ nên $\sqrt{x}-2=0$$\Leftrightarrow x=4$, thỏa mãn.
		Vậy $P=\dfrac{2}{7}$ khi $x=4$.
		\item Vì $x\geqslant 0\Rightarrow x+\sqrt{x}+1\geqslant 1$\\
		$\Leftrightarrow 0<\dfrac{2}{x+\sqrt{x}+1}\leqslant 2$
		$\Leftrightarrow 0<P\leqslant 2$ 
		$\Leftrightarrow P\left({P-2}\right)\leqslant 0\Leftrightarrow P^2\leqslant 2P$.\\ 
		Dấu “=” xảy ra khi $P=2\Leftrightarrow x=0$.
		Vậy $P^2\leqslant 2P$.
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2016-2017, Thanh Hóa]%[9D3K3]
\begin{enumerate}
        \item  Tìm $x,y\in \mathbb{Z}$ thỏa mãn $2y^2x+x+y+1=x^2+2y^2+xy.$
        \item .Cho $a,b,c$ là các số nguyên khác 0 thỏa mãn điều kiện:
$\left({\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}}\right)^2=\dfrac{1}{a^2}+\dfrac{1}{b^2}+\dfrac{1}{c^2} $.Chứng minh rằng: $a^3+b^3+c^3$ chia hết cho $3.$
    \end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có
\begin{align*}
& 2y^2x+x+y+1=x^2+2y^2+xy\\ &\Leftrightarrow 2y^2x+x+y+1-x^2-2y^2-xy=0\\
&\Leftrightarrow\left({x-1}\right)(2y^2-y-x)=-1
\end{align*}
Vì $x,y\in \mathbb{Z} $ nên $x-1\in \left\{1;-1\right\}$
\begin{itemize}
\item Nếu $x-1=1\Rightarrow x=2.$ Khi đó $2y^2 -y-2=-1\Leftrightarrow y=1\vee y= -\dfrac{1}{2}\notin \mathbb{Z}.$
\item Nếu $x - 1 = -1 \Rightarrow x = 0.$ 
Khi đó $2y^2 - y = 1 \Leftrightarrow y = 1 \vee y =- \dfrac{1}{2}\notin \mathbb{Z}.$

Vậy
$\begin{cases}
x=2 \\ y=1
\end{cases};\begin{cases}
x=0 \\ y=1
\end{cases}.
$
\end{itemize}
\item Từ giả thiết  
$$(\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c})^2=\dfrac{1}{a^2}+\dfrac{1}{b^2}+\dfrac{1}{c^2}\Leftrightarrow2(\dfrac{1}{ab}+\dfrac{1}{bc}+\dfrac{1}{ca})=0$$
Vì $a,b,c \ne 0$ nên $a+b+c=0\Rightarrow a+b=-c.$ Từ đó
$$\left({a+b}\right)^3=\left({-c}\right)^3 \Rightarrow a^3+b^3+3ab(a+b)=-c^3\Rightarrow  a^3+b^3+c^3=3abc  $$

Vậy $a^3+b^3+c^3 \vdots 3$ với $a,b,c \in \mathbb{Z}. $

\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2016-2017, Thanh Hóa]%[9D3K3]
\begin{enumerate}
\item Giải phương trình sau: $\sqrt{{4x^2+20x+25}}+\sqrt{{x^2+6x+9}}=10x-20.$
\item Cho $x,y$ là 2 số thực thoả mãn: $x^2+2y^2+2xy+7x+7y+10=0.$
Tìm giá trị nhỏ nhất và giá trị lớn nhất của biểu thức: $A =x+y+1$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Đkxđ: $\forall x\in \mathbb{R}.$
$$\sqrt{{4x^2+20x+25}}+\sqrt{{x^2+6x+9}}=10x-20.$$
Vì $\sqrt{{4x^2+20x+25}}+\sqrt{{x^2+6x+9}}\geqslant 0,\forall x$ nên 
$10x-20 \geqslant 0\Rightarrow x\geqslant 2.$\\
Tacó
\begin{align*}
&\sqrt{4x^2+20x+25}+\sqrt{x^2+6x+9}=10x-20\\ 
& \Leftrightarrow \left|2x+5\right|+\left|x+3\right|=10x-20\\
&\Leftrightarrow 2x+5+x+3=10x-20\Leftrightarrow 7x=28 \Leftrightarrow x=4
\end{align*}
Vậy phương trình có nghiệm là $x=4.$
\item Ta có
\begin{align*}
&x^2+2y^2+2xy+7x+7y+10=0\\ 
& \Leftrightarrow \left({x+y}\right)^2+7(x+y)+10=-y^2\\
&\Rightarrow (x+y+2)(x+y+5)=-y^2 \leq 0\\
&\Rightarrow -4 \leq x+y+1\leq-1
\end{align*}
\begin{itemize}
\item $x+y+1=-4$ khi $x=-5;y=0.$
\item $x+y+1=-1$ khi $x=-2;y=0.$
\end{itemize}
Vậy $A_{min}=-4$ khi $x=-5;y=0.$ $A_{max}=-1$ khi $x=-2;y=0.$
\end{enumerate}

}
\end{ex}

Cho hình vuông $ABCD$ có cạnh bằng $a$. $N$ là điểm tùy ý thuộc cạnh $AB$. Gọi $E$ là giao điểm của $CN$ và $DA$. Vẽ tia $Cx$ vuông góc với $CE$ và cắt $AB$ tại $F$. Lấy $M$ là trung điểm của $EF$.
\begin{enumerate}
	\item Chứng minh $CM$ vuông góc với $EF$. 
	\item Chứng minh $NB.DE=a^2$ và $B$, $D$, $M$ thẳng hàng.
	\item Tìm vị trí của $N$ trên $AB$ sao cho diện tích của tứ giác $AEFC$ gấp $3$ lần diện tích của hình vuông $ABCD$.
\end{enumerate}

	\end{enumerate}}

\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2016-2017, Thanh Hóa]%[9D3G2]
Cho $a,b,c>0.$ Chứng minh rằng:
$$\dfrac{a}{a+b}+\dfrac{b}{b+c}+\dfrac{c}{c+a}<\sqrt{{\dfrac{a}{b+c}}}+\sqrt{{\dfrac{b}{c+a}}}+\sqrt{{\dfrac{c}{a+b}}}.$$
\loigiai{
Vì $a,b,c>0$ nên $\dfrac{a}{a+b}<1\Rightarrow \dfrac{a}{a+b}<\dfrac{a+c}{a+b+c}.$\\
Tương tự: $\dfrac{b}{b+c}<\dfrac{b+a}{a+b+c};\quad \dfrac{c}{c+a}<\dfrac{c+b}{a+b+c}$ suy ra
$$ \dfrac{a}{a+b}+\dfrac{b}{b+c}+\dfrac{c}{c+a}<2(1) $$
Ta có $\sqrt{{\dfrac{a}{b+c}}}=\dfrac{a}{\sqrt{a(b+c)}}.$
Vì $a,b,c>0$ nên theo bất đẳng thức Cô-si ta có:
\begin{align*}
&\dfrac{a+(b+c)}{2}\geqslant \sqrt{{a(b+c)}}>0\\ 
& \Leftrightarrow\dfrac{2}{a+b+c}\leqslant\dfrac{1}{\sqrt{{a(b+c)}}}\\
&\Leftrightarrow \dfrac{2a}{a+b+c}\leqslant \dfrac{a}{\sqrt{{a(b+c)}}}\\
&\Leftrightarrow \dfrac{2a}{a+b+c}\leqslant \sqrt{{\dfrac{a}{b+c}}}
\end{align*}
Tươngtự $\dfrac{2b}{a+b+c}\leqslant \sqrt{{\dfrac{b}{a+c}}};\quad \dfrac{2c}{a+b+c}\leqslant \sqrt{{\dfrac{c}{b+a}}}.$\\
Từ đó suy ra $\sqrt{{\dfrac{a}{b+c}}}+\sqrt{{\dfrac{b}{c+a}}}+\sqrt{{\dfrac{c}{a+b}}}\geqslant 2.$\\
Vậy $\dfrac{a}{a+b}+\dfrac{b}{b+c}+\dfrac{c}{c+a}<\sqrt{{\dfrac{a}{b+c}}}+\sqrt{{\dfrac{b}{c+a}}}+\sqrt{{\dfrac{c}{a+b}}}. $
}
\end{ex}
