\subsection{Hỏi đáp nhanh}
\begin{ex}%[7H3K1]
	Tại sao trong một tam giác, góc đối diện với cạnh nhỏ nhất là góc nhọn?
	\loigiai{
	Vì góc đối diện với cạnh nhỏ nhất là góc nhỏ nhất nên nếu góc đó không nhọn (tức lớn hơn hoặc bằng $90^\circ$) thì tổng số đo $3$ góc của tam giác giác lớn hơn hoặc bằng $270^\circ$, vô lí. Vậy góc đối diện với cạnh nhỏ nhất là góc nhọn.

\subsection{Học giải toán}
\begin{vd}%[7H3B1]
	Cho $\triangle ABC$ có $AB>AC$. Kẻ $BN$ là tia phân giác của $\widehat{ABC}$ ($N\in AC$). Kẻ $CM$ là tia phân giác của $\widehat{ACB}$ ($M\in AB$), $BN$ và $CM$ cắt nhau tại $I$.
	\begin{enumerate}
		\item So sánh $IC$ và $IB$.
		\item Một học sinh nhận xét $\widehat{C_1}=\widehat{C_2}$ (giả thiết). Vậy $AM=MB$ vì là hai cạnh đối diện với hai góc bằng nhau. Đúng hay sai? Tại sao?
	\end{enumerate}
	\loigiai{
	\immini
	{
			\begin{enumerate}
			\item Xét $\triangle ABC$ có $AB>AC$ (giả thiết). Suy ra $\widehat{C}>\widehat{B}$ (định lí). Mà $\widehat{C_1}=\dfrac{1}{2}\widehat{C}$ và $\widehat{B_1}=\dfrac{1}{2}\widehat{B}$ nên $\widehat{C_1}>\widehat{B_1}$.\\
			Trong $\triangle CIB$ có $\widehat{C_1}>\widehat{B_1}$ nên $IB>IC$ (định lí).
			\item Từ $\widehat{C_1}=\widehat{C_2}$ suy ra $AM=MB$ là sai vì $\widehat{C_1}$ là góc của $\triangle BCM$; $\widehat{C_2}$ là góc của $\triangle CMA$. Các cạnh $AM$ và $MB$ cũng là cạnh của hai tam giác khác nhau. Định lí ở trên chỉ xét trong một tam giác.
		\end{enumerate}
	}
	{
		\begin{tikzpicture}[scale=1, line join=round, line cap=round,>=stealth]
		\tkzInit[xmin=-1,xmax=7,ymin=-1,ymax=5]
		\tkzClip
		\tkzDefPoints{4/4/A,0/0/B,6/0/C}

		\tkzDefLine[bisector](A,C,B) \tkzGetPoint{c}
		\tkzInterLL(C,c)(A,B)\tkzGetPoint{M}
		\tkzInterLL(B,N)(C,M) \tkzGetPoint{I}
		\tkzDrawPoints(A,B,C,M,N,I)
		\tkzDrawSegments(A,B B,C C,A B,N C,M)
		\tkzLabelPoints[above](A,I) \tkzLabelPoints[left](B,M)
		\tkzLabelPoints[right](C,N)
		\tkzMarkAngles[size=0.5](C,B,N) \tkzMarkAngles[size=0.6](N,B,A)
		\tkzMarkAngles[size=0.5,mark=|](A,C,M) \tkzMarkAngles[size=0.6,mark=|](M,C,B)

	}
\end{vd}
\begin{vd}%[7H3K1]
	Cho tam giác $ABC$ có $AB<AC$, $AD$ là tia phân giác của góc $BAC$ ($D\in BC$). Chứng minh rằng:
	\begin{enumerate}
		\item $\widehat{ADB}<\widehat{ADC}$.
		\item $CD>DB$.
	\end{enumerate}
	\loigiai{
	\immini
	{
		\begin{enumerate}
			\item Xét $\triangle ABC$ có $AB<AC$ (giả thiết) $\Rightarrow \widehat{C}<\widehat{B}$ (định lí).\\
			Xét $\triangle ABD$ và $\triangle ACD$ có\\ $\widehat{A_1}+\widehat{B}+\widehat{ADB}=\widehat{A_2}+\widehat{C}+\widehat{ADC}=180^\circ$ (định lí tổng ba góc trong tam giác).\\
			Mà $\widehat{A_1}=\widehat{A_2}$ (giả thiết) và $\widehat{C}<\widehat{B}$ (chứng minh trên) nên $\widehat{ADB}<\widehat{ADC}$.
		\end{enumerate}	
	}
	{
		\begin{tikzpicture}[scale=.7, line join=round, line cap=round,>=stealth]
		\tkzInit[xmin=-2,xmax=8,ymin=-1,ymax=5]
		\tkzClip
		\tkzDefPoints{2/4/A,0/0/B,6/0/C}
		\tkzDefLine[bisector](B,A,C) \tkzGetPoint{d}
		\tkzInterLL(A,d)(B,C)\tkzGetPoint{D}
		\tkzDefPointBy[reflection=over A--D](B)\tkzGetPoint{E}
		\tkzDrawPoints(A,B,C,D,E)
		\tkzDrawSegments(A,B B,C C,A A,D D,E)
		\tkzLabelPoints[above](A) \tkzLabelPoints[below](B,C,D)
		\tkzLabelPoints[right](E)
		\tkzMarkAngles[size=0.5](D,A,C) \tkzMarkAngles[size=0.6](B,A,D)
		\tkzLabelAngle[pos=1](B,A,D){$1$} \tkzLabelAngle[pos=-.5](A,E,D){$1$}
		\tkzLabelAngle[pos=1](D,A,C){$2$}
		\tkzLabelAngle[pos=.4](D,E,C){$2$}
		\tkzMarkSegments[pos=0.5,mark=|](A,B A,E)
		\end{tikzpicture}
	}
	\begin{enumerate}
		\item[b)] Vì $AC>AB$ (giả thiết) nên trên cạnh $AC$ ta lấy được một điểm $E$ sao cho $AE=AB$.\\
		Xét $\triangle ABD$ và $\triangle AED$ có:
		\begin{itemize}
			\item $AB=AE$ (theo cách dựng).
			\item $\widehat{A_1}=\widehat{A_2}$ (giả thiết).
			\item $AD$ là cạnh chung.
		\end{itemize}
		Do vậy $\triangle ABD=\triangle AED$ (c.g.c).\\
		$\Rightarrow BD=ED$ (hai cạnh tương ứng) và $\widehat{B}=\widehat{E_1}$ (hai góc tương ứng).\\
		Ta lại có $\widehat{E_2}=180^\circ-\widehat{E_1}$ (hai góc kề bù).\\
		Suy ra $\widehat{E_2}=180^\circ -\widehat{B}=\widehat{BAC}+\widehat{C}\Rightarrow \widehat{E_2}>\widehat{C}$.\\
		Xét $\triangle ECD$ có $\widehat{E_2}>\widehat{C}$ (chứng minh trên) $\Rightarrow CD>ED$ (định lí).\\
		Mà $ED=BD$ (chứng minh trên) nên suy ra $CD>BD$.
	\end{enumerate}
	}
\end{vd}
\begin{vd}%[7H3K1]
	Cho tam giác $ABC$ có $AB<AC$, $M$ là trung điểm của cạnh $BC$.
	\begin{enumerate}
		\item Chứng minh rằng $\widehat{AMB}<\widehat{AMC}$.
		\item Trên đoạn $AM$ lấy một điểm $E$ tùy ý. Chứng minh $EB<EC$.
	\end{enumerate}
	\loigiai{
	\immini
	{
		\begin{enumerate}
			\item Trên tia đối của tia $MA$ lấy điểm $A'$ sao cho $M$ là trung điểm của $ABOVE'$.\\
			Khi đó, ta có $\triangle ABM=\triangle A'CM$ (c.g.c)\\
			$\Rightarrow A'C=AB$ (hai cạnh tương ứng) và $\widehat{A_1}=\widehat{A'}$ (hai góc tương ứng).\hfill$(1)$\\
			Vì $AB<AC$ (giả thiết) nến $A'C<AC$.\\
			Xét $\triangle ACA'$, có $A'C<AC$ (chứng minh trên) $\Rightarrow \widehat{A_2}<\widehat{A'}$ (quan hệ góc - cạnh đối diện).\hfill$(2)$\\
			Từ $(1)$ và $(2)$ suy ra $\widehat{A_1}>\widehat{A_2}$. Lại có $\widehat{B}>\widehat{C}$ (do $AC>AB$). Vậy $\widehat{AMB}<\widehat{AMC}$.
		\end{enumerate}
	}
	{
		\begin{tikzpicture}[scale=.7, line join=round, line cap=round,>=stealth]
		\tkzInit[xmin=-2,xmax=8,ymin=-4.2,ymax=4.2]
		\tkzClip
		\tkzDefPoints{2/3.5/A,0/0/B,6/0/C}
		\tkzDefMidPoint(B,C)\tkzGetPoint{M}
		\tkzDefPointBy[translation= from A to M](M) \tkzGetPoint{A'}
		\tkzDefPointBy[homothety=center A ratio .7](M)\tkzGetPoint{E}
		\tkzDrawPoints(A,B,C,M,A',E)
		\tkzDrawSegments(A,B B,C C,A A,A' A',C E,B E,C)
		\tkzLabelPoints[above](A) \tkzLabelPoints[below](A')
		\tkzLabelPoints[below right](C) \tkzLabelPoints[below left](B,M)
		\tkzLabelPoints[above right](E)
		\tkzLabelAngle[pos=1](B,A,M){$1$} \tkzLabelAngle[pos=.5](A,M,B){$1$}
		\tkzLabelAngle[pos=1](M,A,C){$2$}
		\tkzLabelAngle[pos=.4](C,M,A){$2$}
		\tkzMarkSegments[pos=0.5,mark=|](B,M C,M)
		\end{tikzpicture}
	}
	\begin{enumerate}
		\item[b)] Giả sử $EB=EC$ thì $\triangle EMB=\triangle EMC$ (c.c.c). Suy ra $\widehat{M_1}=\widehat{M_2}$, trái với a).\\
		Giả sử $EB>EC$. Lập luận tương tự câu a) suy ra $\widehat{M_1}>\widehat{M_2}$, vô lí. Vậy $EB<EC$.
	\end{enumerate}
	\begin{note}
		Các tam giác $AMB$ và $AMC$ có hai cặp cạnh tương ứng bằng nhau, cặp cạnh thứ ba không bằng nhau. Góc $M_2$ đối diện với cạnh $AC$ lớn hơn nên nó lớn hơn góc $M_1$. Ta có kết quả tổng quát sau.\\
		Nếu hai tam giác có hai cặp cạnh bằng nhau nhưng cặp cạnh thứ ba không bằng nhau thì góc đối diện với cạnh lớn hơn là góc lớn hơn. Ngược lại, nếu hai tam giác có hai cặp cạnh bằng nhau nhưng hai góc xem giữa không bằng nhau thì cạnh đối diện với góc lớn hơn là cạnh lớn hơn.\\
		Hai tam giác $ABC$ và $A'B'C'$ có $AB=A'B'$; $AC=A'C'$, thì $BC<B'C'\Leftrightarrow \widehat{A}<\widehat{A'}$.
	\end{note}
	}
\end{vd}
\begin{vd}%[7H3K1]
	Cho tam giác $ABC$ có $\widehat{B}=75^\circ$, $\widehat{C}=45^\circ$. Điểm $M$ nằm trong tam giác $ABC$ sao cho $\widehat{MBC}=\widehat{MCB}=30^\circ$. Chứng minh rằng $AM$ vuông góc với $BM$.
	\loigiai{
	\immini
	{
		Ta có\\
		$\widehat{BAC}=180^\circ-\left( \widehat{ABC}+\widehat{ACB}\right)=180^\circ-(75^\circ+45^\circ)=60^\circ$.\hfill$(1)$\\
		$\widehat{B_2}=\widehat{ABC}-\widehat{B_1}=75^\circ-30^\circ=45^\circ$.\\
		$\widehat{C_2}=\widehat{ACB}-\widehat{C_1}=45^\circ-30^\circ=15^\circ$.\\
		Giả sử $MA$ không vuông góc với $MB$, tức là $\widehat{AMB}\ne 90^\circ$, ta xét hai trường hợp sau:
	}
	{
		\begin{tikzpicture}[scale=.65, line join=round, line cap=round,>=stealth]
		\tkzInit[xmin=-1,xmax=7,ymin=-1,ymax=5]
		\tkzClip
		\tkzDefPoints{0/0/B,5/0/C}
		\tkzDefPointBy[rotation= center B angle 75](C) \tkzGetPoint{b}
		\tkzDefPointBy[rotation= center C angle -45](B) \tkzGetPoint{c}
		\tkzInterLL(B,b)(C,c) \tkzGetPoint{A}
		\tkzDefPointBy[rotation= center B angle 30](C) \tkzGetPoint{e}
		\tkzDefPointBy[rotation= center C angle -30](B) \tkzGetPoint{f}
		\tkzInterLL(B,e)(C,f) \tkzGetPoint{M}
		\tkzDrawSegments(A,B B,C C,A M,A M,B M,C)
		\tkzDrawPoints[fill=black](A,B,C,M)
		\tkzLabelPoints[above](A) \tkzLabelPoints[below](B,C,M)
		\tkzLabelAngle[pos=1.5](M,A,C){$1$} \tkzLabelAngle[pos=1](M,C,B){$1$}
		\tkzLabelAngle[pos=1](C,B,M){$1$}
		\tkzLabelAngle[pos=.8](B,A,M){$2$}
		\tkzLabelAngle[pos=.8](M,B,A){$2$}
		\tkzLabelAngle[pos=1.5](A,C,M){$2$}
		\end{tikzpicture}
	}
	\noindent Trường hợp 1: $\widehat{AMB}<90^\circ$.\\
	Xét $\triangle AMB$ có $\widehat{A_2}=180^\circ-\left( \widehat{AMB}+45^\circ\right)>45^\circ$.\\
	$\Rightarrow \widehat{A_2}>\widehat{B_2}\Rightarrow MB>MA$ (định lí).\\
	Vì $\triangle MBC$ cân tại $M$ nên $MB=MC\Rightarrow MC>MA\Rightarrow \widehat{A_1}>\widehat{C_2}$ (định lí).\\
	Từ đó suy ra $\widehat{A_1}+\widehat{A_2}>\widehat{B_2}+\widehat{C_2}$ hay $\widehat{BAC}>60^\circ$, trái với $(1)$.\\
	Trường hợp 2: $\widehat{AMB}>90^\circ$.\\
	Lập luận tương tự, ta có $\widehat{BAC}<60^\circ$, trái với $(1)$.\\
	Vậy $\widehat{AMB}=90^\circ$ hay $MA\perp MB$.	
	}
\end{vd}

\subsubsection{Cơ bản}
\begin{bt}%[7H3B1]
	Cho $\triangle ABC$, lấy $M$ là trung điểm của cạnh $BC$. Chứng minh rằng:
	\begin{enumerate}
		\item Nếu $AM>\dfrac{1}{2}BC$ thì $\widehat{BAC}$ là góc nhọn.
		\item Nếu $AM=\dfrac{1}{2}BC$ thì $\widehat{BAC}$ là góc vuông.
		\item Nếu $AM<\dfrac{1}{2}BC$ thì $\widehat{BAC}$ là góc tù.
	\end{enumerate}
	\loigiai{
	\immini
	{
		\begin{enumerate}
			\item Trong $\triangle ABM$ có $AM>BM\Rightarrow \widehat{B}>\widehat{A_1}$.\hfill$(1)$\\
			Trong $\triangle AMC$ có $AM>MC$ nên $\widehat{C}>\widehat{A_2}$.\hfill$(2)$\\
			Cộng từng vế $(1)$ và $(2)$ có $$\widehat{B}+\widehat{C}>\widehat{A_1}+\widehat{A_2}=\widehat{A}.$$
			Mà $\widehat{A}+\widehat{B}+\widehat{C}=180^\circ$ nên $\widehat{A}<90^\circ$. Vậy $\widehat{A}$ nhọn.
		\end{enumerate}
	}
	{
		\begin{tikzpicture}[scale=.7, line join=round, line cap=round,>=stealth]
		\tkzInit[xmin=-1,xmax=7,ymin=-1,ymax=5]
		\tkzClip
		\tkzDefPoints{2/4/A,0/0/B,6/0/C}
		\tkzDefMidPoint(B,C)\tkzGetPoint{M}
		\tkzDrawPoints[fill=black](A,B,C,M)
		\tkzDrawSegments(A,B B,C C,A A,M)
		\tkzLabelPoints[above](A) \tkzLabelPoints[below](B,C,M)
		\tkzLabelAngle[pos=1](M,A,C){$2$}
		\tkzLabelAngle[pos=1](B,A,M){$1$}
		\tkzMarkSegments[pos=0.5,mark=|](B,M C,M)
		\tkzMarkAngles[size=0.6](B,A,M) \tkzMarkAngles[size=0.5,mark=|](M,A,C)
		\end{tikzpicture}
	}
	\immini
	{
		\begin{enumerate}
			\item[b)] Trong $\triangle ABM$ có $AM=BM\Rightarrow \widehat{B}=\widehat{A_1}$.\hfill$(3)$\\
			Trong $\triangle AMC$ có $AM=MC$ nên $\widehat{C}=\widehat{A_2}$.\hfill$(4)$\\
			Cộng từng vế $(3)$ và $(4)$ có $$\widehat{B}+\widehat{C}=\widehat{A_1}+\widehat{A_2}=\widehat{A}.$$
			Mà $\widehat{A}+\widehat{B}+\widehat{C}=180^\circ$ nên $\widehat{A}=90^\circ$. Vậy $\widehat{A}$ vuông.
		\end{enumerate}
	}
	{
		\begin{tikzpicture}[scale=.7, line join=round, line cap=round,>=stealth]
		\tkzInit[xmin=-1,xmax=7,ymin=-1,ymax=5]
		\tkzClip
		\tkzDefPoints{0/0/B,6/0/C}
		\tkzDefPointBy[rotation= center B angle 50](C) \tkzGetPoint{b}
		\tkzDefPointBy[rotation= center C angle -40](B) \tkzGetPoint{c}
		\tkzInterLL(B,b)(C,c) \tkzGetPoint{A}
		\tkzDefMidPoint(B,C)\tkzGetPoint{M}
		\tkzDrawPoints[fill=black](A,B,C,M)
		\tkzDrawSegments(A,B B,C C,A A,M)
		\tkzLabelPoints[above](A) \tkzLabelPoints[below](B,C,M)
		\tkzLabelAngle[pos=1](M,A,C){$2$}
		\tkzLabelAngle[pos=1](B,A,M){$1$}
		\tkzMarkSegments[pos=0.5,mark=|](B,M C,M)
		\tkzMarkAngles[size=0.6](B,A,M) \tkzMarkAngles[size=0.5,mark=|](M,A,C)
		\end{tikzpicture}
	}
	\immini
	{
		\begin{enumerate}
			\item[c)] Trong $\triangle ABM$ có $AM<BM\Rightarrow \widehat{B}<\widehat{A_1}$.\hfill$(5)$\\
			Trong $\triangle AMC$ có $AM<MC$ nên $\widehat{C}<\widehat{A_2}$.\hfill$(6)$\\
			Cộng từng vế $(5)$ và $(6)$ có $$\widehat{B}+\widehat{C}<\widehat{A_1}+\widehat{A_2}=\widehat{A}.$$
			Mà $\widehat{A}+\widehat{B}+\widehat{C}=180^\circ$ nên $\widehat{A}>90^\circ$. Vậy $\widehat{A}$ tù.
		\end{enumerate}
	}
	{
		\begin{tikzpicture}[scale=.7, line join=round, line cap=round,>=stealth]
		\tkzInit[xmin=-1,xmax=7,ymin=-1,ymax=4]
		\tkzClip
		\tkzDefPoints{0/0/B,6/0/C}
