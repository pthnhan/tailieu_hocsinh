\begin{bt}%[Thi GHK, Nguyễn Chí Thanh - Hồ Chí Minh, 2018]%[Phan Quốc Trí, EX-DT-10&11]%[1D1B3-3]%[1D1B3-2]%[1D1K2-1]
	Giải các phương trình lượng giác sau
	\begin{listEX}
		\item $ 2\sin x +2 \cos x = \sqrt{2} $.
		\item  $ 2\sin^2 x -3 \sin x \cos x - \cos^2 x =2 $.
		\item $ \tan 3x + \tan x =0 $.
	\end{listEX}
	\loigiai{
		\begin{enumerate}
			\item Ta có 
			\begin{eqnarray*}
				&&2\sin x +2 \cos x = \sqrt{2} \\
				&\Leftrightarrow & \sin x + \cos x = \dfrac{\sqrt{2}}{2}\\
				&\Leftrightarrow  & \sin \left( x+\dfrac{\pi}{4} \right) = \dfrac{1}{2}\\
				& \Leftrightarrow & \hoac{&x+\dfrac{\pi}{4} = \dfrac{\pi}{6}+k 2 \pi\\& x+\dfrac{\pi}{4} = \pi -\dfrac{\pi}{6}+k 2 \pi  }\\
				& \Leftrightarrow & \hoac{&x=-\dfrac{\pi}{12}+ k 2 \pi\\& x = \dfrac{7\pi}{12}+k 2 \pi} \; (k \in \mathbb{Z}).
			\end{eqnarray*}
\item Ta thấy $x = -\dfrac{\pi}{4}+ k \pi$ thỏa mãn phương trình đã cho nên là nghiệm.\\
Với $x\ne \dfrac{\pi}{2} + k \pi \; (k\in \mathbb{Z})$, ta có
\begin{eqnarray*}
&&2\sin^2 x -3 \sin x \cos x -2 \cos^2 x =2\\
& \Leftrightarrow & 2 \tan^2 x -3 \tan x -1 = 2(1+ \tan^2 x )\\
& \Leftrightarrow & \tan x =-1\\
&\Leftrightarrow & x = -\dfrac{\pi}{4}+ k \pi \; ( k\in \mathbb{Z})
\end{eqnarray*}
Vậy phương trình đã cho có các nghiệm $x = -\dfrac{\pi}{4}+ k \pi; \; x = -\dfrac{\pi}{4}+ k \pi, \; (k\in \mathbb{Z}) $.
\item Điều kiện: $\heva{& \cos 3x \ne 0\\ & \cos x \ne 0} \Leftrightarrow \heva{&x \ne \dfrac{\pi}{6}+k \dfrac{\pi}{3}\\ & x \ne \dfrac{\pi}{2}+k \pi.}$
\begin{eqnarray*}
&& \tan 3x + \tan x =0\\
& \Leftrightarrow & \dfrac{\sin 3x}{\cos 3x}+ \dfrac{\sin x }{\cos x }=0 \\
& \Leftrightarrow & \sin 3x \cos x + \cos 3x \sin x = 0\\
& \Leftrightarrow & \sin 4x =0\\
& \Leftrightarrow & x = \dfrac{k\pi}{4},\; ( k\in \mathbb{Z})
\end{eqnarray*}
So sánh với điều kiện, suy ra phương trình có các nghiệm $x=\pi + k 2 \pi$ và $x = \dfrac{\pi}{4}+k \dfrac{\pi}{2}, \; (k \in \mathbb{Z})$.
		\end{enumerate}	
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Chí Thanh - Hồ Chí Minh, 2018]%[Phan Quốc Trí, EX-DT-10&11]%[1D2B1-2]
	Từ tập hợp $ X=\left\{ 0;1;2;3;4;5\right\} $, có bao nhiêu cách lập các số tự nhiên chẵn gồm $ 4 $ chữ số đôi một khác nhau?
	\loigiai{
Gọi $x=\overline{abcd}$ là số cần tìm.
\begin{description}
\item[TH1:] $x=\overline{abc0}$. Trường hợp này có $5\cdot 4 \cdot 3 = 60$ số.
\item[TH2:] $x=\overline{abcd}$ với $d\in \left\{2,4  \right\}$. Trường hợp này có $2 \cdot 4 \cdot 4 \cdot 3 = 96$ số
\end{description}	
Vậy có $60+96 = 156$ số thỏa yêu cầu bài toán.
	}
\end{bt}


\begin{bt}%[Thi GHK, Nguyễn Chí Thanh - Hồ Chí Minh, 2018]%[Phan Quốc Trí, EX-DT-10&11]%[1D2K2-1]
	Đội học sinh giỏi của trường gồm $18$	em; trong đó có $7$ học sinh khối $ 12 $; $ 6 $ học sinh khối $ 11 $ và $ 5 $ học sinh khối $ 10 $. Hỏi có bao nhiêu cách cử $ 8 $ học sinh trong đội đi dự trại Hè sao cho mỗi khối có ít nhất một em được chọn?
	\loigiai{
Ta có $\mathrm{C}_{18}^8 = 43758$ cách chọn ra $8$ học sinh bất kì để tham dự trại hè.\\
Gọi $x$ là số cách chọn ra $8$ học sinh tham dự trại hè sao cho trong $8$ học sinh này không có 	đủ học sinh của ba khối lớp 10, 11 và 12. Ta có 
$$x=\mathrm{C}_{13}^8+ \mathrm{C}_{12}^8+\mathrm{C}_{11}^8=1947.$$
Do đó có $43758-1947 = 41811$ cách chọn $8$ học sinh thỏa yêu cầu bài toán.
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Chí Thanh - Hồ Chí Minh, 2018]%[Phan Quốc Trí, EX-DT-10&11]%[1D2B2-3]
	Cho đa giác có $ n $ cạnh $ (n \ge 4) $. Tìm $ n $ để đa giác có số đường chéo bằng số cạnh.
	\loigiai{
Số đường chéo của một đa giác có $n$ cạnh là $\mathrm{C}_{n}^2 -n $. Do đó, đa giác có số đường chéo bằng số cạnh khi và chỉ khi $\mathrm{C}_{n}^2 -n = n \Leftrightarrow n^2 -5n = 0 \Leftrightarrow n =5$.
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Chí Thanh - Hồ Chí Minh, 2018]%[Phan Quốc Trí, EX-DT-10&11]%[1D2K2-2]
	Một người có $ 8 $ bì thư khác nhau và $ 6 $ tem thư khác nhau. Người đó cần gửi thư cho ba người bạn. Hỏi người đó có bao nhiêu cách chọn $ 3 $ bì thư và $ 3 $ tem thư sau đó dán mỗi tem thư lên bì thư để gửi?
	\loigiai{
\begin{itemize}
\item Có $\mathrm{C}_{8}^3=56$ cách chọn ra $3$ bì thư.
\item Có $\mathrm{C}_{6}^3=20$ cách chọn ra $3$ tem thư.
\item Có $3!=6$ cách dán ba tem thư lên ba bì thư.
\end{itemize}
Do đó có $56 \cdot 20 \cdot 6 = 6720$ cách để người đó thực hiện.
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Chí Thanh - Hồ Chí Minh, 2018]%[Phan Quốc Trí, EX-DT-10&11]%[1D2B3-2]
	Tìm hệ số của $ x^7 $	trong khai triển nhị thức $ \left( 3x^2 -\dfrac{2}{x}\right)^8\; (x\ne 0) $.
	\loigiai{
Số hạng tổng quát của khai triển là $T_{k+1}= \mathrm{C}_{8}^k (3x^2)^{8-k} \left( - \dfrac{2}{x} \right)^k = \mathrm{C}_{8}^k   3^{8-k} (-2)^k x^{16-3k}$.\\
Số hạng chứa $x^7$ ứng với $16-3k=7 \Leftrightarrow k = 3$.\\
Hệ số của $x^7$ là $ \mathrm{C}_{8}^3 \cdot  3^5 \cdot (-2)^3 = -108864$.
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Chí Thanh - Hồ Chí Minh, 2018]%[Phan Quốc Trí, EX-DT-10&11]%[1D2K3-2]
	Tìm hệ số của $ x^3 $ trong khai triển thành đa thức của biểu thức sau: $ A=(x+1)^5+(x-2)^7 $.	
	\loigiai{
Ta có $(x+1)^5+(x-2)^7 = \sum \limits_{i=0}^5 \mathrm{C}_5^i x^{5-i} + \sum \limits_{j=0}^7 \mathrm{C}_7^j x^{7-j} (-2)^{j} $.\\
Số hạng chứa $x^3$ ứng với $\heva{&5-i=3\\& 7-j=3} \Leftrightarrow \heva{& i= 2\\& j =4. }$\\
Hệ số của $x^3$ là $\mathrm{C}_5^2 + \mathrm{C}_7^4 (-2)^4= 570.$
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Chí Thanh - Hồ Chí Minh, 2018]%[Phan Quốc Trí, EX-DT-10&11]%[1D2K3-2]
	Biết hệ số của số hạng thứ ba theo số mũ giảm dần của $ x $	trong khai triển nhị thức $ \left(x^5+\dfrac{3}{x^4}\right)^{n+6} $ bằng $ 594 $. Tìm $ n $?
	\loigiai{
Điều kiện: $n\ge -4$.\\
Số hạng thứ $k+1$ theo thứ tự số mũ giảm dần của $x$ trong khai triển là $$T_{k+1} =\mathrm{C}_{n+6}^k \left( x^5 \right)^{n+6-k} \left(\dfrac{3}{x^4} \right)^{k}=\mathrm{C}_{n+6}^k 3^k x^{5n +30 -9k}  $$
Hệ số của số hạng thứ ba (ứng với $k=2$) bằng $594$, do đó $$\mathrm{C}_{n+6}^2 \cdot 3^2 = 594 \Leftrightarrow \mathrm{C}_{n+6}^2 = 66 \Leftrightarrow n^2 + 11n -102 = 0 \Leftrightarrow n = 6. $$
	}
\end{bt}
