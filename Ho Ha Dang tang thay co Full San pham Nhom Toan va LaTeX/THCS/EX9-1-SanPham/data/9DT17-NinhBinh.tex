\begin{ex}%[Đề thi vào 10, Sở giáo dục Ninh Bình, 2017]%[9D3B4]%[9D1B3]%[9D2B4]
\hfill
\begin{enumerate}
\item Rút gọn biểu thức $A=\sqrt{3}\left(\sqrt{12}-\sqrt{3}\right)$.
\item Tìm $m$ để đường thẳng $y = (m-1)x + 3$ song song với đường thẳng $y = 2x +1$.
\item Giải hệ phương trình $\heva{&x+2y=4\\&5x-2y=8}$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $A=\sqrt{36} - 3 = 6-3 =3$.
\item Đường thẳng  $y = (m-1)x + 3$ song song với đường thẳng $y = 2x +1$ khi và chỉ khi 
$\heva{&m-1=2\\&3 \neq 1} \iff m=3$. 
\item $\heva{&x+2y=4\\&5x-2y=8} \iff \heva{&6x=12\\&5x-2y=8} \iff \heva{&x=2\\&y=1}$
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Ninh Bình, 2017]%[9D4B5]%[9D4K6]
Cho phương trình $x^2+2(m+2)x +4m-1 =0$  ($x$ là ẩn số, $m$ là tham số)  (1). 
\begin{enumerate}
\item Giải phương trình (1) khi $m=2$.
\item Chứng minh rằng với mọi giá trị của tham số $m$ thì phương trình (1) luôn có hai
nghiệm phân biệt. Gọi $x_1, x_2$ là hai nghiệm của phương trình (1), tìm $m$ để
$x_1^2+x_2^2 =30$. 
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Khi $m=2$, ta có phương trình $x^2+8x+7=0$, $a-b+c=0$ cho nên $x_1=-1,x_2=-7$. 
\item $\Delta'= (m+2)^2 - 4m+1 = m^2 +5 > 0$ với mọi $m$ cho nên phương trình có  nghiệm
với mọi $m$. Theo định lí Vi-ét ta có $x_1^2+x_2^2 =(x_1 +x_2)^2 -2x_1x_2 = 4(m+2)^2-2(4m-1)
=4m^2+8m+18  =30 \iff m= 1,  m =-3$.   
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Ninh Bình, 2017]%[9D4K8]
Một ô tô dự định đi từ bến xe A đến bến xe B cách nhau 90 km với vận tốc
không đổi. Tuy nhiên, ô tô khởi hành muộn 12 phút so với dự định. Để đến bến xe B
đúng giờ ô tô đã tăng vận tốc lên 5 km/h so với vận tốc dự định. Tìm vận tốc dự định
của ô tô.
\loigiai{\\
Gọi vận tốc ô tô dự định đi từ A đến B là $x$ (km/h),  $x > 0$, khi đó  vận tốc ô tô thực tế đã đi từ A đến B là $x + 5$ (km/h)\\
Thời gian ô tô đi hết quãng đường AB với vận tốc dự định là : $\dfrac{90}{x}$ (h) \\
Thời gian ô tô đã đi hết quãng đường AB là : $\dfrac{90}{x+5}$ (h) \\
Theo giả thiết ta có phương trình $\dfrac{90}{x} - \dfrac{90}{x+5} = \dfrac{1}{5}~ (1) $ (12 phút đổi thành $\dfrac{1}{5}$ giờ) \\
$(1) \iff x^2+5x+2250=0 \iff x_1= 45, x_2 = -50 \textrm{ (loại) }$.\\
Vậy vận tốc dự định của ô tô là $45$ (km/h)
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Ninh Bình, 2017]%[9H2K6]%[9H3K4]
Cho đường tròn tâm $O$, bán kính $R$. Từ điểm $C$ nằm ngoài đường tròn kẻ hai tiếp
tuyến $CA, CB$ và cát tuyến $CMN$ với đường tròn $(O)$ ($A, B$ là hai tiếp điểm, $M$ nằm
giữa $C$ và $N$). Gọi $H$ là giao điểm của $CO$ và $AB$.
\begin{enumerate}
\item Chứng minh tứ giác $AOBC$ nội tiếp.
\item Chứng minh $CH.CO = CM.CN$.
\item Tiếp tuyến tại $M$ của đường tròn $(O)$ cắt $CA, CB$ theo thứ tự tại $E$ và $F$. Đường
vuông góc với $CO$ tại $O$ cắt $CA, CB$ theo thứ tự tại $P, Q$. Chứng minh
$\widehat {POE} =\widehat{OFQ}$.
\item Chứng minh  $PE +QF \geq PQ$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[thick]
\coordinate[label=above : $A$] (A) at (-2,3); \coordinate[label=below : $B$] (B) at (-2,-3);
\coordinate[label=left : $C$] (C) at (-6.5,0); \coordinate[label=above : $E$] (E) at (-3.06,2.29); 
\coordinate[label=below : $F$] (F) at (-4.25,-1.5); \coordinate[label=below left : $H$] (H) at (-2,0);
\coordinate[label=above left : $M$] (M) at (-3.44,1.08); \coordinate[label=above : $N$] (N) at (2,3);
\coordinate[label=right : $O$] (O) at (0,0); \coordinate[label=above : $P$] (P) at (0,4.33);
\coordinate[label=below : $Q$] (Q) at (0,-4.33);
\foreach \i in {A,B,C,E,F,M,N,H,O,P,Q} 
\draw[fill=black] (\i) circle (1.4pt) ;
\draw (O) circle (3.61) ; 
\draw (O)--(A)--(B)--(O)--(C)--(P)--(Q)--(C)--(N)--(A)--(M)--(O)--(F)--(E)--cycle; 
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Ta có $\widehat{CAO} = \widehat{CBO} = 90^{\circ}$ cho nên tứ giác $AOBC$ nội tiếp.
\item Chứng minh $CH.CO = CM.CN$. \\
+ Tam giác $ACO$ vuông tại $A$ với $AH$ là đường cao cho nên $CH.CO=CA^2$ (1). \\
+ Hai tam giác $CAM$ và $CNA$ có góc $\widehat{ACN}$ chung, $\widehat{CAM} = \widehat{CNA}$ (góc
giữa tiếp tuyến và dây cung) cho nên $\Delta CAM$  đồng dạng với $\Delta CNA$, kéo theo $\dfrac{CA}{CN} = \dfrac{CM}{CA} \iff CA^2=CM.CN$ (2).\\
+ Từ (1) và (2) ta có $CH.CO = CM.CN$.
\item  Chứng minh $\widehat {POE} =\widehat{OFQ}$. \\
+ Dễ thấy $\widehat{POA}=\widehat{BOQ}$ và $\widehat{AOE} = \widehat{EOM}$. \\
+ Ta có $\widehat{OFQ} = 90^{\circ} - \dfrac{180^{\circ} - \widehat{BOQ}-  \widehat{POM}}{2} = 
\widehat{POA} + \widehat{AOE} = \widehat{POE}$. 
\item Chứng minh  $PE +QF \geq PQ$. \\
+ Theo ý 3. $\widehat {POE} =\widehat{OFQ}$ suy ra hai tam giác $OPE$ và $FQO$ đồng dạng cho nên 
$\dfrac{OP}{FQ} = \dfrac{PE}{QO} \iff  OP.PQ = PE.QF \iff \dfrac{PQ^2}{4} = PE.QF \iff PQ = 2\sqrt{PE.QF}. $
  (1) \\
+ Theo bất đẳng thức Cô-si ta có $2\sqrt{PE.QF} \leq PE+QF$ (2) \\
+ Từ (1) và (2) ta có $PE +QF \geq PQ$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Ninh Bình, 2017]%[9D5G1]
Cho các số thực không âm $a, b, c$ thỏa mãn $\sqrt{a} + \sqrt{b} +\sqrt{c} =3$. Tìm giá trị nhỏ
nhất của biểu thức $P = \sqrt{3a^2+2ab+3b^2} + \sqrt{3b^2+2bc+3c^2} + \sqrt{3c^2+2ca+3a^2}$. 
\loigiai{ \\
 Với các số thực bất kì $x,y,z$ ta có $3x^2+2xy+3y^2 - 2(x+y)^2 = (x-y)^2 \geq 0 $, cho nên   $P = \sqrt{3a^2+2ab+3b^2} + \sqrt{3b^2+2bc+3c^2} + \sqrt{3c^2+2ca+3a^2} \geq \sqrt{2} (a+b +b+c +c+a ) = 2\sqrt{2} (a+b+c) $ (1) \\
Mặt khác  $3(a+b+c) - (\sqrt{a} +\sqrt{b} +\sqrt{c})^2= (\sqrt{a} - \sqrt{b})^2 + (\sqrt{b} - \sqrt{c})^2 +(\sqrt{c} - \sqrt{a})^2 \geq 0 \Longrightarrow  3(a+b+c)  \geq  (\sqrt{a} +\sqrt{b} +\sqrt{c})^2 =9 \Longrightarrow a+b+c \geq 3$ (2) \\
Từ (1) và (2) ta có $P \geq  6\sqrt{2}$. Dấu bằng xảy ra khi và chỉ khi $a=b=c=1$. Vậy $P_{\min} =6\sqrt{2}$. 
}
\end{ex}
