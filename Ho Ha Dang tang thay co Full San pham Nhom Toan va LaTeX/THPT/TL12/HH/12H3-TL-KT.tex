\begin{ex}[3 điểm]%[Trần Chiến]%[2H3Y1]%[2H3B2] %[2H3B1]
	Trong không gian với hệ trục tọa độ $Oxyz$, cho các điểm  $A(1; 0; 0)$, $B(0; 1; 0)$, $C(0; 0; 1)$, $D(-2; 1; -1)$.
	\begin{itemize}
	\item[a)] Tìm tọa độ trọng tâm $G$ của tam giác $BCD$.
	\item[b)] Viết phương trình mặt phẳng $(BCD)$.
	\item[c)] Viết phương trình mặt cầu $(S)$ có tâm $A$ và tiếp xúc với mặt phẳng $(BCD)$.
	\end{itemize}
	\loigiai{a) Giả sử $G\left(x_G;y_G;z_G\right)$. Ta có $\overrightarrow{OG}=\dfrac{1}{3}\left(\overrightarrow{OB}+\overrightarrow{OC}+\overrightarrow{OD}\right)$ nên 
	$$\heva{& x_G=\dfrac{x_B+x_C+x_D}{3}=-\dfrac{2}{3}\\
		& y_G=\dfrac{y_B+y_C+y_D}{3}=\dfrac{2}{3}\\
		& z_G=\dfrac{z_B+z_C+z_D}{3}=0.}$$
	Vậy $G\left(-\dfrac{2}{3};\dfrac{2}{3};0\right)$.\\
	b) Ta có $\overrightarrow{BC}=(0;-1;1)$ và $\overrightarrow{BD}=(-2;0;-1)$ suy ra $\overrightarrow{n}=\left[\overrightarrow{BC},\overrightarrow{BD}\right]=(1;-2;-2)$ là một VTPT của mặt phẳng $(BCD)$.\\
	Phương trình mặt phẳng $(BCD)$ qua $B(0;1;0)$ có VTPT $\overrightarrow{n}=(1;-2;-2)$ là $$(BCD):x-2y+2z+2=0.$$
	c) Ta có $d(A,(BCD))=\dfrac{\vert 1+2 \vert}{\sqrt{1^2+2^2+2^2}}=1$.\\
	Vì mặt cầu $(S)$ tiếp xúc với mặt phẳng $(BCD)$ nên bán kính của $(S)$ là $R=d(A, (BCD))=1$.\\				
	Vậy phương trình mặt cầu $(S)$ tâm $A(1;0;0)$ bán kính $R=1$ là $(S):(x-1)^2+y^2+z^2=1$.
	}
\end{ex}

\begin{ex}[2 điểm]%[Trần Chiến]%[2H3Y3]%[2H3K1]
	Trong không gian với hệ trục tọa độ $Oxyz$ cho hai điểm $A(1;3;-2)$ và $B(0;3;-1)$.
	\begin{itemize}
	\item[a)] Viết phương trình tham số của đường thẳng $AB$.
	\item[b)] Giả sử $AB$ cắt mặt cầu tâm $O$ bán kính $\sqrt{10}$ tại hai điểm phận biệt $M,N$. Tính độ dài $MN$.
	\end{itemize}
	\loigiai{a) Ta có $\overrightarrow{AB}=(-1;0;1)$ nên phương trình đường thẳng $AB$ qua $A(1;3;-2)$ và có VTCP $\overrightarrow{AB}=(-1;0;1)$ là $$AB:\heva{& x=1-t\\ & y=3 \\ & z=-2+t} \, (t\in\mathbb{R}).$$
	b) Gọi $(S)$ là mặt cầu tâm $O$ bán kính $\sqrt{10}$, khi đó $(S):x^2+y^2+z^2=10$.\\
	Để tìm giao điểm của $AB$ và $(S)$, ta giải phương trình $(1-t)^2+3^2+(-2+t)^2=10\Leftrightarrow \hoac{& t=1\\ & t=2.}$\\
	Với $t=1$ thì $M(0;3;-1)$, với $t=2$ thì $N(-1;3;0)$ (hoặc ngược lại). Do đó, $MN=\sqrt{2}$.
	}
\end{ex}

\begin{ex}[2 điểm]%[Trần Chiến]%[2H3K2]
	Trong không gian với hệ trục tọa độ $Oxyz$, viết phương trình mặt phẳng $(P)$ biết rằng $(P)$ chứa trục $Ox$ và khoảng cách từ $A(0;1;2)$ đến $(P)$ bằng $1$.
	\loigiai{Gọi $\overrightarrow{n}=(a;b;c) \neq \overrightarrow{0}$ là một VTPT của $(P)$. \\
	Khi đó, phương trình của $(P)$ là $ax+by+cz+d=0$ với $a,b,c\in\mathbb{R}$.\\
	Vì $(P)$ chứa $Ox$ nên cũng chứa $O$, do đó $d=0$.\\
	Hơn nữa, $\overrightarrow{n} \perp \overrightarrow{i}=(1;0;0)$ nên $\overrightarrow{n}\cdot\overrightarrow{i}=0$, suy ra $a=0$.\\
	Ta có $d\left(A,(P)\right)=1 \Leftrightarrow \dfrac{\vert b+2c \vert}{\sqrt{b^2+c^2}}=1 \Leftrightarrow 4bc=-3c^2 \Leftrightarrow \hoac{& c=0 \\ & b=-\dfrac{3}{4}c \neq 0.}$\\
	Với $c=0$, kết hợp với $a=d=0$ suy ra $b\neq 0$. Chọn $b=1$ ta được phương trình của $(P)$ là $y=0$.\\
	Với $b=-\dfrac{3}{4}c \neq 0$, chọn $b=3, c=-4$ ta được phương trình của $(P)$ là $3y-4z=0.$
	}
\end{ex}

\begin{ex}[3 điểm]%[Trần Chiến]%[2H3B3]%[2H3G3]
	Trong không gian với hệ trục tọa độ $Oxyz$, cho hai đường thẳng $d_1$ và $d_2$ có phương trình $d_1:\dfrac{x-3}{2}=\dfrac{y-3}{2}=z-3$ và $d_2:\dfrac{x+5}{6}=\dfrac{y+2}{3}=\dfrac{z}{2}$. 	
	\begin{itemize}
	\item[a)] Chứng minh hai đường thẳng $d_1$ và $d_2$ cắt nhau, gọi giao điểm là $I$.
	\item[b)] Tìm tọa độ các điểm $A,B$ lần lượt thuộc $d_1,d_2$ sao cho tam giác $AIB$ cân tại $I$ và có diện tích bằng $\dfrac{\sqrt{41}}{42}$.
	\end{itemize}
	\loigiai{a) Ta có $$\heva{& \dfrac{x-3}{2}=\dfrac{y-3}{2}=z-3\\ & \dfrac{x+5}{6}=\dfrac{y+2}{3}=\dfrac{z}{2}} \Leftrightarrow \heva{& x=1\\ & y=1 \\ & z=2.}$$
	Vậy $d_1$ cắt $d_2$ tại $I(1;1;2)$.\\
	b) $d_1$ đi qua điểm $M_1(3;3;3)$ và có VTCP $\overrightarrow{u_1}=(2;2;1)$;\\
	$d_2$ đi qua điểm $M_2(-5;-2;0)$ và có VTCP $\overrightarrow{u_2}=(6;3;2)$. \\
	Gọi $\alpha$ là góc giữa hai đường thẳng $d_1$ và $d_2$, ta có
	$$\cos\alpha=\dfrac{\vert\overrightarrow{u_1}\cdot\overrightarrow{u_2}\vert}{\vert\overrightarrow{u_1}\vert\cdot\vert\overrightarrow{u_1}\vert}=\dfrac{20}{21} \Rightarrow \sin\alpha=\sqrt{1-\cos^2\alpha}=\dfrac{\sqrt{41}}{21}.$$
	Đặt $IA=IB=a>0$ thì diện tích tam giác $ABC$ là $$S=\dfrac{1}{2}\cdot IA\cdot IB\cdot\sin\alpha=a^2\cdot\dfrac{\sqrt{41}}{42}=\dfrac{\sqrt{41}}{42}\Rightarrow a=1.$$
	Vì $A\in d_1$ nên $A(3+2t;3+2t;3+t)$, do đó $\overrightarrow{IA}=(2t+2;2t+2;t+1)$, suy ra $$IA^2=1\Leftrightarrow 9(t+1)^2=1\Leftrightarrow \hoac{& t=-\dfrac{2}{3}\\ & t=-\dfrac{4}{3}}\, \text{suy ra}\, A\left(\dfrac{5}{3};\dfrac{5}{3};\dfrac{7}{3}\right)\, \text{hoặc}\, A\left(\dfrac{1}{3};\dfrac{1}{3};\dfrac{5}{3}\right).$$
	Vì $B\in d_2$ nên $B(-5+6t;-2+3t;2t)$, do đó $\overrightarrow{IB}=(6t-6;3t-3;2t-2)$, suy ra $$IB^2=1\Leftrightarrow 49(t-1)^2=1\Leftrightarrow \hoac{& t=\dfrac{8}{7}\\ & t=\dfrac{6}{7}}\, \text{suy ra}\, B\left(\dfrac{13}{7};\dfrac{10}{7};\dfrac{16}{7}\right)\, \text{hoặc}\, B\left(\dfrac{1}{7};\dfrac{4}{7};\dfrac{12}{7}\right).$$
	Vậy có bốn cặp điểm $A,B$ thỏa yêu cầu là\\ $\left\{A\left(\dfrac{5}{3};\dfrac{5}{3};\dfrac{7}{3}\right), B\left(\dfrac{13}{7};\dfrac{10}{7};\dfrac{16}{7}\right)\right\}$ hoặc $\left\{A\left(\dfrac{5}{3};\dfrac{5}{3};\dfrac{7}{3}\right), B\left(\dfrac{1}{7};\dfrac{4}{7};\dfrac{12}{7}\right)\right\}$ hoặc\\ $\left\{A\left(\dfrac{1}{3};\dfrac{1}{3};\dfrac{5}{3}\right), B\left(\dfrac{13}{7};\dfrac{10}{7};\dfrac{16}{7}\right)\right\}$ hoặc $\left\{A\left(\dfrac{1}{3};\dfrac{1}{3};\dfrac{5}{3}\right), B\left(\dfrac{1}{7};\dfrac{4}{7};\dfrac{12}{7}\right)\right\}$.
	}
\end{ex}

	Trong không gian với hệ trục tọa độ $Oxyz$, cho các điểm  $A(2,5,1)$, $B(3,2,4)$, $C(-1,7,3)$, $D(2,-4,8)$.
	\begin{itemize}
	\item[a)] Tìm tọa độ trọng tâm $G$ của tam giác $ABC$.
	\item[b)] Viết phương trình mặt phẳng $(ABC)$.
	\item[c)] Viết phương trình mặt cầu $(S)$ có tâm $D$ và tiếp xúc với mặt phẳng $(ABC)$.
	\end{itemize}
	\loigiai{a) Giả sử $G\left(x_G;y_G;z_G\right)$. Ta có $\overrightarrow{OG}=\dfrac{1}{3}\left(\overrightarrow{OB}+\overrightarrow{OC}+\overrightarrow{OD}\right)$ nên 
	\begin{align*}
		\heva{& x_G=\dfrac{x_A+x_B+x_C}{3}=\dfrac{4}{3}\\
		& y_G=\dfrac{y_A+y_B+y_C}{3}=\dfrac{14}{3}\\
		& z_G=\dfrac{z_A+z_B+z_C}{3}=\dfrac{8}{3}.}
	\end{align*}
	Vậy $G\left(\dfrac{4}{3};\dfrac{14}{3};\dfrac{8}{3}\right)$.\\
	b) Ta có $\overrightarrow{AB}=(1;-3;3)$ và $\overrightarrow{AC}=(-3;2;2)$ suy ra $\overrightarrow{n}=\left[\overrightarrow{AB},\overrightarrow{AC}\right]=(-12;-11;-7)$ là một VTPT của mặt phẳng $(ABC)$.\\
	Phương trình mặt phẳng $(BCD)$ qua $A(2;5;1)$ có VTPT $\overrightarrow{n}=(-12;-11;-7)$ là $$(BCD):12x+11y+7z-86=0.$$
	c) Ta có $d(D,(ABC))=\dfrac{\vert 12\cdot 2+11\cdot(-4)+7\cdot 8 -86 \vert}{\sqrt{12^2+11^2+7^2}}=\dfrac{50}{\sqrt{314}}$.\\
	Vì mặt cầu $(S)$ tiếp xúc với mặt phẳng $(BCD)$ nên bán kính của $(S)$ là $R=d(D, (ABC))=\dfrac{50}{\sqrt{314}}$.\\				
	Vậy phương trình mặt cầu $(S)$ tâm $D(2,-4,8)$ bán kính $R=\dfrac{50}{\sqrt{314}}$ là $$(S):(x-2)^2+(y+4)^2+(z-8)^2=\dfrac{1250}{157}.$$
	}
\end{ex}

\begin{ex}[3 điểm]%[Trần Chiến]%[2H3Y2]%[2H3K2]

	Ta chọn $a=2$ suy ra $b=3$, với $c=6$ thì $d=-12$, với $c=-6$ thì $d=0$.\\
	Vậy $(Q):2x+3y+6z-12=0$ hoặc $(Q):2x+3y-6z=0$.
	}

	Với $s=t=0$ thì $M(0;0;0), N(-1;0;1)$, trường hợp này loại do $M,N\in (P)$, tức $MN\subset (P)$.\\
	Vậy hai điểm $M$ và $N$ cần tìm có tọa độ là $M\left(\dfrac{4}{7};\dfrac{4}{7};\dfrac{8}{7}\right)$ và $N\left(\dfrac{1}{7};-\dfrac{4}{7};\dfrac{3}{7}\right)$.

\begin{ex}[2 điểm]%[Trần Chiến]%[2H3G1]
