\begin{ex}%[Sở giáo dục Lai Châu, 2017]%[9D4B4]%[9D3B4]
 Giải các phương trình và hệ phương trình sau
\begin{enumerate}
        	\item $2x-1=0$.
        	\item $x^2-6x-7=0$.
	\item $\heva{&x+2y=1 \\ &2x-y=7}.$
\end{enumerate}
\loigiai
    {
 \begin{enumerate}
       \item Ta có $2x-1=0 \Leftrightarrow x=\dfrac{1}{2}$. Vậy $\mathcal{S}= \left\{\dfrac{1}{2} \right\}$.
       \item Ta có \\
$\begin{aligned}
& x^2-6x-7= 0\\ 
& \Leftrightarrow x^2+x-7x-7 =0 \\
& \Leftrightarrow x(x+1) -7(x+1)=0\\
& \Leftrightarrow (x+1)(x-7)=0\\
& \Leftrightarrow \hoac{& x=-1 \\ &x=7}.
\end{aligned}$\\
Vậy $\mathcal{S}= \left\{-1;7 \right\}$.
\item  Ta có \\
$ \heva{&x+2y=1 \\ &2x-y=7} \Leftrightarrow \heva{&x+2y=1 \\ &4x-2y=14}
 \Leftrightarrow \heva{&5x =15 \\ &x+2y=1}\Leftrightarrow \heva{&x=3 \\&y=-1}.
$\\
Vậy nghiệm của hệ $(x;y)=(3;-1)$.
\end{enumerate}
    }
\end{ex}

\begin{ex}%[Sở giáo dục Lai Châu, 2017]%[9D1B8]
Cho biểu thức $A= \dfrac{1}{\sqrt x +1}+\dfrac{x}{\sqrt x -x}$, $(x >0; x\ne 1)$.\\
\begin{enumerate}
\item Rút gọn biểu thức $A$.
\item Tìm $x$ để $A =2017$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có \\
$\begin{aligned}
A & = \dfrac{1}{\sqrt x +1}+\dfrac{x}{\sqrt x (1-\sqrt x)}\\ 
& =\dfrac{1}{\sqrt x +1}+\dfrac{\sqrt x}{1-\sqrt x}\\
&=\dfrac{1+x}{1-x}\,\, (x \ne 1; x >0).
\end{aligned}$
\item Để $A=2017$ thì $\dfrac{1+x}{1-x} =2017 \Leftrightarrow  2018x=2016 \Leftrightarrow x =\dfrac{1008}{1009}$ (thỏa mãn).\\
Vậy $\mathcal S =  \left\{\dfrac{1008}{1009} \right\}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Sở giáo dục Lai Châu, 2017]%[9D4K8]
Một người đi xe đạp từ thành phố Lai Châu đến thị trấn Tam Đường cách nhau $36$ km. Khi đi từ thị trấn Tam Đường về thành phố Lai Châu, người đó tăng vận tốc thêm $3$ km/h vì vậy thời gian về ít hơn thời gian đi là $36$ phút. Tính vận tốc của người đi xe đạp khi đi từ thành phố Lai Châu về thị trấn Tam Đường.

\loigiai{
Đổi $36$ phút = $\dfrac{3}{5}$ giờ.\\
Gọi vận tốc của người đi xe đạp từ Lai Châu đến Tam Đường là $x$ (km/h), ($x >0$).\\
Thời gian người đó đi từ Lai Châu đến Tam Đường là $t_1=\dfrac{36}{x}$ (h).\\
Lúc về, người đó tăng vận tốc thêm $3$ (km/h) nên vận tốc khi về là $x+3$ (km/h).\\
Do đó, người đó đi về hết thời gian là $t_2=\dfrac{36}{x+3}$ (h).\\
Thời gian về ít hơn thời gian đi là $36$ phút nên ta có phương trình
\[\begin{aligned}
&\dfrac{36}{x}-\dfrac{36}{x+3}=\dfrac{3}{5}\\ 
&\Leftrightarrow x^2+3x-180=0 \\
&\Leftrightarrow \hoac{&x=12 (\text{thỏa mãn}) \\ & x=-15 (\text{ko thỏa mãn})}.
\end{aligned}
\]
Vậy vận tốc của người đó khi đi từ Lai Châu tới Tam Đường là $12$ (km/h).
}
\end{ex}

\begin{ex}%[Sở giáo dục Lai Châu, 2017]%[9D4K6]
Cho phương trình $x^2-(2m+1)x+m^2+1=0$.
\begin{enumerate}
\item Tìm $m$ để phương trình có nghiệm.
\item Gọi $x_1, x_2$ là hai nghiệm của phương trình. Tìm $m$ để $x_1=2x_2$.
\end{enumerate}
\loigiai{
Ta có $\Delta=4m-3$.\\
\begin{enumerate}
\item Để phương trình có nghiệm thì $\Delta \ge 0 \Leftrightarrow m \ge \dfrac{4}{3}$.
\item Phương trình có nghiệm khi $m \ge \dfrac{4}{3}$.\\
Theo định lý Viet $\heva{& x_1+x_2=2m+1  (1) \\ & x_1 x_2=m^2+1 (2) }$.\\
Theo bài ra $x_1=2 x_2$, thay vào (1) ta được $\heva{& x_2=\dfrac{2m+1}{3}\\ &x_1 =\dfrac{2(2m+1)}{3}}$.\\
Thay $x_1, x_2$ vào (2) ta được phương trình $m^2-8m+7=0$. Suy ra $\hoac{& m =1 (\text{loại})\\ & m=7 (\text{thỏa mãn})}$.\\
Vậy $m=7$ thỏa mãn bài toán.
\end{enumerate}
}
\end{ex}


\begin{ex}%[Sở giáo dục Lai Châu, 2017]%[9D3K7]
Cho đường tròn $(O;R)$ có dây $MN$ cố định $(MN < 2R)$, $P$ là một điểm trên cung lớn $MN$ sao cho tam giác $MNP$ có ba góc nhọn. Các đường cao $ME$ và $NK$ của tam giác $MNP$ cắt nhau tại $H.$
\begin{enumerate}
\item Chứng minh rằng tứ giác $PKHE$ nội tiếp đường tròn.
\item Kéo dài $PO$ cắt đường tròn tại $Q$. Chứng minh rằng $\widehat{KNM} =\widehat{NPQ}$.
\item Chứng minh rằng khi $P$ thay đổi trên đường tròn $(O)$ thì độ dài đoạn $PH$ không đổi.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
\clip[] (0,2) circle (4.4cm);
\tkzDefPoints{-3/0/M, 3/0/N, -2/5/P, 0/2/O, -1.45/4.17/Q'}
\tkzDefLine[orthogonal= through M](P,N)\tkzGetPoint{X}
\tkzDefLine[orthogonal= through N](P,M)\tkzGetPoint{Y}
\tkzInterLL(M,X)(P,N)\tkzGetPoint{E}
\tkzInterLL(N,Y)(P,M)\tkzGetPoint{K}
\tkzInterLL(M,X)(N,Y)\tkzGetPoint{H}
\tkzDefPointBy[symmetry=center O](P)\tkzGetPoint{Q}
\tkzDrawCircle(O,M)
\tkzMarkRightAngle(M,E,P)
\tkzMarkRightAngle(N,K,P)
\tkzInterLL(H,Q)(M,N)\tkzGetPoint{F}

\tkzDrawPoint[size=4pt, color=black](O)
\tkzLabelPoints[above](P)
\tkzLabelPoints[below right](N, Q)
\tkzLabelPoints[below](F, H)
\tkzLabelPoints[below left](M)
\tkzLabelPoints[left](K, O)
\tkzLabelPoints[above right](E)
\tkzDrawSegments(M,N M,P P,N P,H P,Q M,E N,K H,Q O,F M,Q)
\tkzMarkAngle(K,N,M)
\tkzDrawArc[rotate, color=black](P,Q')(11.31)
\end{tikzpicture}
\end{center}

\begin{enumerate}
\item Xét tứ giác $KHEP$ có:
\[ \widehat{HKP}+\widehat{HEP}=90^\circ+90^\circ=180^\circ\]
Mà $K, E$ là hai đỉnh đối nhau của tứ giác. Suy ra tứ giác $KHEP$ nội tiếp đường tròn.
\item Ta có $\widehat{PNQ}=90^\circ$ (góc nội tiếp chắn nửa đường tròn).
\\
Suy ra  $\widehat{NPQ}+\widehat{NQP}=90^\circ$ (định lí tổng ba góc).\\
Mà $\heva{&\widehat{MNK}+\widehat{NMP}=90^\circ \,\,\, (\text{định lí tổng ba góc}) \\ & \widehat{NQP}=\widehat{NMP} \,\,\,(\text{hai góc nội tiếp cùng chắn cung $PQ$})}$.\\
Suy ra $\widehat{KNM} =\widehat{NPQ}$.

\item {\bf TH1: Tam giác $PMN$ không cân tại $P$.}
\\ Dễ chứng minh được tứ giác $MQNH$ là hình bình hành (hai cặp cạnh đối song song). Suy ra $HQ \cap MN$ tại $F$ là trung điểm của $MN$ và $HQ$.\\
Suy ra $OF \perp MN$ (tính chất). \\
Do $(O;R)$ cố định, $MN$ không đổi nên $OF$ không đổi.\\
Xét tam giác $QHP$ có $OF$ là đường trung bình, suy ra $PH = 2. OF$. Vậy $PH$ có độ dài không đổi.\\
{\bf TH2: Tam giác $PMN$ cân tại $P$. }\\
Khi đó, $PM=PN$ nên $P$ là điểm chính giữa cung lớn $MN$; dây $MN$ cố định nên $P$ cố định. Do đó, $\Delta PMN$ cố định suy ra $PH$ có độ dài không đổi.
\end{enumerate}

}
\end{ex}
