\begin{ex}%[Đề thi HSG 9, tỉnh Phú Yên, năm 2018]%[Thái Văn Đạt, Dự án EX-9G-18]%[9D0K1]
	Tính giá trị của $P=\dfrac{\dfrac{2+\sqrt{3}}{2}}{1+\dfrac{\sqrt{4+2\sqrt{3}}}{2}}+\dfrac{\dfrac{2-\sqrt{3}}{2}}{1-\dfrac{\sqrt{4-2\sqrt{3}}}{2}}$.
	\loigiai{
		Ta có: \begin{align*}
			P&=\dfrac{2+\sqrt{3}}{2+\sqrt{(1+\sqrt{3})^2}}+\dfrac{2-\sqrt{3}}{2-\sqrt{(1-\sqrt{3})^2}}\\
			&=\dfrac{2+\sqrt{3}}{2+(1+\sqrt{3})}+\dfrac{2-\sqrt{3}}{2-(\sqrt{3}-1)}\\
			&=\dfrac{2+\sqrt{3}}{3+\sqrt{3}}+\dfrac{2-\sqrt{3}}{3-\sqrt{3}}\\
			&=\dfrac{(2+\sqrt{3})(3-\sqrt{3})+(2-\sqrt{3})(3+\sqrt{3})}{(3+\sqrt{3})(3-\sqrt{3})}\\
			&=\dfrac{6}{6}=1.
		\end{align*}
		Vậy $P=1$.
		}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Phú Yên, năm 2018]%[Thái Văn Đạt, Dự án EX-9G-18]%[9D0K2]
	Giải phương trình $\dfrac{(2017-x)^2+(2017-x)(x-2018)+(2018-x)^2}{(2017-x)^2-(2017-x)(x-2018)+(2018-x)^2}=\dfrac{13}{37}$.
	\loigiai{
	Tập xác định $\mathscr{D}=\mathbb{R}$.\\
	Đặt $a=2017-x$ và $b=x-2018$ ($a,b\in \mathbb{R}$). Khi đó phương trình trở thành:
	\begin{align*}
		&\dfrac{a^2+ab+b^2}{a^2-ab+b^2}=\dfrac{13}{37}\\
		\Leftrightarrow &37(a^2+ab+b^2)=13(a^2-ab+b^2)\\
		\Leftrightarrow &12a^2+25ab+12b^2=0\\
		\Leftrightarrow &(3a+4b)(4a+3b)=0\\
		\Leftrightarrow &\hoac{ &3a+4b=0.\\ &4a+3b=0.}
	\end{align*}
	\begin{enumerate}[\bfseries(-)]
		\item Với $3a+4b=0$ thì $3(2017-x)+4(x-2018)=0$ hay $x=2021$.
		\item Với $4a+3b=0$ thì $4(2017-x)+3(x-2018)=0$ hay $x=2014$.
	\end{enumerate}
	Thử lại ta thấy $x=2014$ và $x=2021$ đều thỏa mãn.\\
	Vậy $\mathbb{S}=\{1;2\}.$
}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Phú Yên, năm 2018]%[Thái Văn Đạt, Dự án EX-9G-18]%[9D0G7]
	Cho các số thực dương $a$, $b$, $c$. Chứng minh rằng:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\sqrt{\dfrac{a}{a+2b}}>\dfrac{a}{a+b}$.
			\item $\sqrt{\dfrac{a}{a+2b}}+\sqrt{\dfrac{b}{b+2c}}+\sqrt{\dfrac{c}{c+2a}}>1$.
		\end{enumerate}
	\end{multicols}
\loigiai{
\begin{enumerate}
	\item Theo bất đẳng thức Cauchy cho hai số dương $a$ và $a+2b$ ta có: $$\sqrt{a(a+2b)}\le\dfrac{a+(a+2b)}{2}=a+b.$$
	Mà dấu bằng không xảy ra nên $\sqrt{a(a+2b)}<a+b$.\\
	Suy ra: $\sqrt{\dfrac{a}{a+2b}}=\dfrac{a}{\sqrt{a(a+2b)}}>\dfrac{a}{a+b}$, ta thu được điều phải chứng minh.
	\item Tương tự cách làm ở câu a ta thu được: 
	$$\sqrt{\dfrac{a}{a+2b}}+\sqrt{\dfrac{b}{b+2c}}+\sqrt{\dfrac{c}{c+2a}}>\dfrac{a}{a+b}+\dfrac{b}{b+c}+\dfrac{c}{c+a}.$$
	Mà $a$, $b$, $c$ dương nên $a+b<a+b+c$, $b+c<a+b+c$ và $c+a<a+b+c$ nên $$\dfrac{a}{a+b}+\dfrac{b}{b+c}+\dfrac{c}{c+a}>\dfrac{a}{a+b+c}+\dfrac{b}{a+b+c}+\dfrac{c}{a+b+c}=1.$$
	Từ đó ta có điều phải chứng minh!
\end{enumerate}}
\end{ex}

		\item Chứng minh rằng: $CA=CK$ và $BA=BL$.
		\item Đường thẳng qua $G$ song song với $BC$ cắt $AD$, $AE$ thứ tự tại $I$, $J$. Gọi $H$ là hình chiếu vuông góc của $G$ lên $BC$. Chứng minh rằng tam giác $IHJ$ vuông cân.

\begin{ex}%[Đề thi HSG 9, tỉnh Phú Yên, năm 2018]%[Thái Văn Đạt, Dự án EX-9G-18]%[9H0G6]
	Cho tam giác $ABC$ vuông cân tại $A$. Điểm $M$ chuyển động trên cạnh $BC$ ($M$ khác $B$, $C$). Gọi $H$, $K$ thứ tự là hình chiếu vuông góc của $M$ lên cạnh $AB$, $AC$. Vẽ đường tròn $(H)$ có tâm $H$ bán kính $HM$ và đường tròn $(K)$ tâm $K$ bán kính $KM$.
	\begin{enumerate}
		\item Chứng minh rằng: Hai đường tròn $(H)$ và $(K)$ luôn cắt nhau tại hai điểm phân biệt.
		\item Gọi $N$ là giao điểm thứ hai của hai đường tròn $(H)$ và $(K)$. Chứng minh rằng đường thẳng $MN$ luôn đi qua một điểm cố định.
	\end{enumerate}
\loigiai{
\immini{
	\begin{enumerate}
		\item Do $HK<HM+KM$ nên hai đường tròn $(H)$ và $(K)$ luôn cắt nhau tại hai điểm phân biệt.
		\item Dựng hình bình hành $ABDC$.\\ 
		Do tam giác $ABC$ vuông cân nên $ABDC$ là hình vuông.\\
		$\Rightarrow$ $\triangle BMD = \triangle BMA(c-g-c)$ $\Rightarrow$ $\widehat{BDM}=\widehat{BAM}. \refstepcounter{equation}\hfill (\theequation)\label{1829.1}$\\
		Lại có: $MHAK$ là hình chữ nhật nên
		\begin{equation} \label{1829.2} \widehat{BAM}=\widehat{HMK}=90^\circ-\widehat{MHK}.\end{equation}
	\end{enumerate}
	}
	{\begin{tikzpicture}[scale=0.5,x=1cm,y=1cm]

	\noindent Mặt khác: $M$, $N$ là giao điểm của hai đường tròn $(H)$, $(K)$ nên $MN\perp HK$\\
	$\Rightarrow$ $\widehat{HMN}=90^\circ-\widehat{MHK}. \refstepcounter{equation}\hfill (\theequation)\label{1829.3}$\\
	Từ (\ref{1829.1}), (\ref{1829.2}) và (\ref{1829.3}) ta thu được: $\widehat{BDM}=\widehat{HMN}$.\\
	Mà $MH\parallel BD(\perp AB)$ nên $M$, $N$, $D$ thẳng hàng.\\
	Vậy đường thẳng $MN$ đi qua điểm $D$ cố định.	
}

\begin{ex}%[Đề thi HSG 9, tỉnh Phú Yên, năm 2018]%[Thái Văn Đạt, Dự án EX-9G-18]%[9D0K4]
	Tìm tất cả các số nguyên tố $p$ sao cho $7p+1$ bằng lập phương của một số tự nhiên.
	\loigiai{
	Dễ kiểm tra được $p=2$ và $p=3$ không thỏa mãn bài toán nên ta chỉ cần xét với $p\ge 5$.\\
	Đặt $7p+1=a^3$ ($a\in\mathbb{N}$). Do $p\ge 5$ nên $a>3$.\\
	Ta có: $7p+1=a^3\Leftrightarrow 7p=(a-1)(a^2+a+1)$.\\
	Do $a>3$ nên $a-1$, $a^2+a+1$ đều là các số nguyên dương lớn hơn 1 và $a^2+a+1>7$.\\ Từ đó chỉ có thể xảy ra trường hợp $a-1=7$ và $a^2+a+1=p$.\\
	Khi đó: $p=73$ là số nguyên tố thỏa mãn.\\
	Vậy $p=73$.}
\end{ex}
