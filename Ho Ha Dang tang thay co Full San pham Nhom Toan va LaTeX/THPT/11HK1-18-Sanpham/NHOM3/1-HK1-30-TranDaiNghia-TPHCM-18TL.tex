\begin{bt}%[HK1 THPT Chuyên Trần Đại Nghĩa, TP. HCM]%[Huỳnh Xuân Tín - DA 11HK1-18]%[1D1K3-6]
Giải phương trình: $\dfrac{\sin{2x}-\sqrt{3}\cos2x-2\sin x}{2\cos x-1}=0.$
	\loigiai{
		Tập xác định $\mathscr{D}=\mathbb{R}\setminus \left\lbrace \pm \dfrac{\pi}{3}+k2\pi\right\rbrace   .$\\
Với điều kiện trên phương trình được viết lại thành
\begin{eqnarray*}
	& & \sin{2x}-\sqrt{3}\cos2x-2\sin x=0 \Leftrightarrow \sin{2x}-\sqrt{3}\cos2x=2\sin x\\
		&\Leftrightarrow& \dfrac{1}{2}\sin2x-\dfrac{\sqrt{3}}{2}\cos2x=\sin x\Leftrightarrow \sin\left(2x-\dfrac{\pi}{3} \right) =\sin x\\
				&\Leftrightarrow& \hoac{&2x-\dfrac{\pi}{3}=x+k2\pi\\
					&2x-\dfrac{\pi}{3}=\pi-x+k2\pi	} \Leftrightarrow  \hoac{&x=\dfrac{\pi}{3}+k2\pi\\
					&x=\dfrac{4\pi}{9}+\dfrac{k2\pi}{3}	.}
				\end{eqnarray*}
So với điều kiện ta chọn nghiệm	$x=\dfrac{4\pi}{9}+\dfrac{k2\pi}{3}$ với $k\in \mathbb{Z}.$ }
\end{bt}

Chứng minh với mọi số nguyên dương $n$, ta có
$$\dfrac{1}{2}+\dfrac{1}{4}+\dfrac{1}{8}+\cdots+\dfrac{1}{2^n}=\dfrac{2^n-1}{2^n}.$$
	\loigiai{
	Ta chứng minh đẳng thức $\dfrac{1}{2}+\dfrac{1}{4}+\dfrac{1}{8}+\cdots+\dfrac{1}{2^n}=\dfrac{2^n-1}{2^n}\quad\quad (1)$ bằng phương pháp quy nạp.\\
	Với $n=1$ thì (1) đúng vì $\dfrac{1}{2}=\dfrac{2^1-1}{2^1}$.\\
	Giả sử (1) đúng với $n=k$, tức là 
	$$\dfrac{1}{2}+\dfrac{1}{4}+\dfrac{1}{8}+\cdots+\dfrac{1}{2^k}=\dfrac{2^k-1}{2^k}\quad\quad (2).$$
	Ta chứng minh (1) đúng với $n=k+1$, tức là chứng minh 
	$$\dfrac{1}{2}+\dfrac{1}{4}+\dfrac{1}{8}+\cdots+\dfrac{1}{2^{k+1}}=\dfrac{2^{k+1}-1}{2^{k+1}}\quad\quad (3).$$
	Thật vậy
	\begin{eqnarray*}
		VT(3)= \dfrac{1}{2}+\dfrac{1}{4}+\dfrac{1}{8}+\cdots+\dfrac{1}{2^k}+\dfrac{1}{2^{k+1}} = \dfrac{2^k-1}{2^k}+\dfrac{1}{2^{k+1}} = \dfrac{2^{k+1}-1}{2^{k+1}} = VP(3).
		\end{eqnarray*}
	Khi đó (3) đúng với $n=k+1$.
\\
Vậy theo nguyên lý quy nạp (1) đúng với mọi số nguyên dương $n$.
	}
\end{bt}

\begin{bt}%[HK1 THPT Chuyên Trần Đại Nghĩa, TP. HCM]%[Huỳnh Xuân Tín - DA 11HK1-18]%[1D3K2-3]
	Xét tính tăng, giảm của dãy số $\left(u_n \right) $ với $u_n=\dfrac{3n^2+5}{2n^2+1}$.
	\loigiai{
		Ta có $u_n=\dfrac{3n^2+5}{2n^2+1}=\dfrac{3}{2}+\dfrac{7}{2\left(2n^2+1 \right) }$. Khi đó
		\begin{eqnarray*}
	u_{n+1}-u_n		& =&\left( \dfrac{3}{2}+\dfrac{7}{2\left(2(n+1)^2+1 \right) }\right) -\left( \dfrac{3}{2}+\dfrac{7}{2\left(2n^2+1 \right) }\right) \\
	&=& \dfrac{7}{2}\left(\dfrac{1}{2(n+1)^2+1} -\dfrac{1}{2n^2+1}\right) \\
		&=& \dfrac{7}{2}\left(\dfrac{2n^2+1-2(n+1)^2-1}{\left( 2(n+1)^2+1\right) \left(2n^2+1 \right) }\right)   \\
			&=& \dfrac{7}{2}\left(\dfrac{-4n-2}{\left( 2(n+1)^2+1\right) \left(2n^2+1 \right) }\right)  \le 0,\,\,\,\, \forall n\in \mathbb{N}.
	\end{eqnarray*}	
Vậy dãy số $\left(u_n \right) $ giảm.
	}
\end{bt}
\begin{bt}%[HK1 THPT Chuyên Trần Đại Nghĩa, TP. HCM]%[Huỳnh Xuân Tín - DA 11HK1-18]%[1D2G5-5]
Một đề thi trắc nghiệm gồm $50$ câu hỏi, mỗi câu có $4$ phương án trả lời, trong đó chỉ có một phương án đúng. Một thí sinh chọn ngẫu nhiên các phương án trả lời, tính xác suất để thí sinh được $52$ điểm, biết rằng mỗi câu trả lời đúng được 2 điểm, mỗi câu trả lời sai bị trừ $1$ điểm.
	\loigiai{
	Gọi $x$ là số câu trả lời đúng của học sinh trên. Khi đó $50-x$ là số câu trả lời sai của học sinh đó.\\
	Số điểm học sinh đó đạt được là
	$$2\cdot x-(50-x)\cdot 1=52 \Leftrightarrow x=34.$$
	Xác suất để học sinh đúng một câu là $\dfrac{1}{4}$, và sai một câu là $\dfrac{3}{4}$.\\
	Số cách chọn ra $34$ câu đúng là $\mathrm{C}_{50}^{34}$.\\
	Vậy xác suất để học sinh đó trả lời được 52 điểm, tức là đúng được 34 câu là
	$$\mathrm{P}=\mathrm{C}_{50}^{34}\left(\dfrac{1}{4} \right)^{34}\cdot\left(\dfrac{3}{4} \right)^{16}=\dfrac{\mathrm{C}_{50}^{34}\cdot 3^{16}}{4^{50}}.$$
	}
\end{bt}

		Xét tính tăng, giảm của dãy số $\left(u_n \right) $ với $u_n=\dfrac{3n+2}{2n+3}$.
	\loigiai{
	Ta có $u_n=\dfrac{3n+2}{2n+3}=\dfrac{3}{2}-\dfrac{5}{2\left(2n+3 \right) }$. Khi đó
	\begin{eqnarray*}
		u_{n+1}-u_n		& =&\left( \dfrac{3}{2}-\dfrac{5}{2\left(2(n+1)+3 \right) }\right) -\left( \dfrac{3}{2}-\dfrac{5}{2\left(2n+3 \right) } \right) \\
		&=& \dfrac{5}{2}\left(\dfrac{1}{2n+3} -\dfrac{1}{2(n+1)+3}\right) \\
		&=& \dfrac{7}{2}\left(\dfrac{2n+5-2n-3}{\left( 2n+5\right) \left(2n+3 \right) }\right)   \\
	&=& \dfrac{7}{2}\left(\dfrac{2}{\left( 2n+5\right) \left(2n+3 \right)}\right)  \ge 0,\,\,\,\, \forall n\in \mathbb{N}.
	\end{eqnarray*}	
	Vậy dãy số $\left(u_n \right) $ tăng.	
	}
\end{bt}
