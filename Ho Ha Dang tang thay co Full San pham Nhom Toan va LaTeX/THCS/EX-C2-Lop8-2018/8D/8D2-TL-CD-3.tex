\section{Tính chia hết đối với đa thức}
\subsection{Tìm dư của phép chia mà không thực hiện phép chia}
\subsubsection{Đa thức chia có dạng $x-a$ ($a$ là hằng)}

\begin{vd}%[Nguyễn Cao Cường]%[8D1G12]
	Chứng minh rằng số dư khi chia đa thức $f(x)$ cho nhị thức $x-a$ bằng giá trị của đa thức $f(x)$ tại $x=a$.\\
	Định lí Bê-du (\textit{Bézout}, 1730-1783, nhà toán học Pháp).
	\loigiai{
	Do đa thức chia $x-a$ có bậc nhất nên số dư khi chia $f(x)$ cho $x-a$ là hằng số $r$.\\
	Ta có
	$$f(x)=(x-a)\cdot Q(x)+r.$$
	Đẳng thức trên đúng với mọi $x$ nên với $x=a$, ta có
	$$f(a)=0\cdot Q(x)+r \text{ hay }f(a)=r.$$
\begin{note}
	Từ định lí Bê-du ta suy ra\\
	Đa thức $f(x)$ chia hết cho $x-a$ khi và chỉ khi $f(a)=0$ (tức là khi $a$ là nghiệm của đa thức).
	\end{note}
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[8D1G12]
	Chứng minh rằng nếu đa thức $f(x)$ có tổng các hệ số bằng $0$ thì đa thức ấy chia hết cho $x-1$.
	\loigiai{
Gọi $f(x)=a_0x^n+a_1x^{n-1}+\ldots+a_{n-1}x+a_n$.\\
Theo giả thuyết, $a_0+a_1+\ldots+a_n=0$.\hfill (1)\\
Theo định lý Bê-du, số dư khi chia $f(x)$ cho $x-1$ là
\[r=f(1)=a_0+a_1+\ldots+a_{n-1}+a_{n}.\tag{2}\]	
Từ (1) và (2), suy ra $r=0$.\\
Vậy $f(x)$ chia hết cho $x-1$.
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[8D1G12]
	Chứng minh rằng nếu đa thức $f(x)$ có tổng các hệ số của các hạng tử bậc chẵn bằng tổng các hệ số của các hạng tử bậc lẻ thì đa thức ấy chia hết cho $x+1$.
	\loigiai{
	Gọi $f(x)=a_0x^{2n}+a_1x^{2n-1}+a_2x^{2n-2}+\ldots+a_{2n-2}x^2+a_{2n-1}x+a_{2n}$, trong đó, $a_0$ có thể bằng $0$.
Theo giả thuyết, $a_0+a_2+..+a_{2n}=a_1+a_3+\ldots+a_{2n-1}$ nên
\[(a_0+a_2+\ldots+a_{2n})-(a_1+a_3+\ldots+a_{2n-1})=0.\tag{1}\]
Theo định lí Bê-du, số dư khi chia $f(x)$ cho $x+1$ bằng
{\allowdisplaybreaks
\begin{align*}
r&=f(-1)=a_0-a_1+a_2-a_3+\ldots+a_{2n-2}-a_{2n-a}+a_{2n}\\
&=(a_0+a_2+\ldots+a_{2n})-(a_1+a_3+\ldots+a_{2n-1})\tag{2}
\end{align*}}
Từ (1) và (2), suy ra $r=0$.\\
Vậy $f(x)$ chia hết cho $x+1$.
}
\end{vd}
\subsubsection{Đa thức có bậc tử từ bậc hai trở lên}
\begin{vd}%[Nguyễn Cao Cường]%[8D1G12]
	Tìm dư khi chia $$x^7+x^5+x^3+1\text{ cho  } x^2-1.$$
	\loigiai{
	Để tìm dư trong trường hợp này, ta thường dùng các cách sau:\\
	\textit{Cách 1. (Tách ra ở đa thức bị chia những đa thức chia hết cho đa thức chia).}\\
	Ta biết rằng $x^n-1$ chia hết cho $x-1$ với mọi số tự nhiên $n$ nên $x^{2n}-1$ chia hết cho $x^2-1$.\\
	Do đó $x^4-1$, $x^6-1$,\ldots chia hết cho $x^2-1$.\\
	Ta có
	\begin{eqnarray*}
	x^7+x^5+x^3+1&=&x^7-x+x^5-x+x^3-x+3x+1\\
	&=&x(x^6-1)+x(x^4-1)+x(x^2-1)+(3x+1).
	\end{eqnarray*}
Dư khi chia $x^7+x^5+x^3+1$ cho $x^2-1$ là $3x+1$.\\
\textit{Cách 2. (Xét giá trị riêng).}\\
Gọi thương là $Q(x)$, dư là $ax+b$. Ta có
$$x^7+x^5+x^3+1=(x+1)(x-1)\cdot  Q(x)+ax+b, \forall x.$$
Đẳng thức đúng với mọi $x$ nên với $x=1$, ta được $4=a+b$. \hfill (1)\\\ Với $x=-1$ ta được $-2=-a+b$. \hfill (2)\\
 Từ (1) và (2), suy ra $a=3$, $b=1$.\\Dư phải tìm là $3x+1$.
\begin{note}
	Để tách ra các đa thức chia hết cho $x^2-1$ hoặc $x^2+1$, cần nhớ lại các hằng đẳng thức $8$ và $9$:
	\begin{center}
		$a^n-b^n$ chia hết cho $a-b$ ($a\neq b$);\\
		$a^n+b^n$ ($n$ lẻ) chia hết cho $a+b$ ($a \neq -b$).
	\end{center}
\end{note}
}
\end{vd}

\subsubsection{Sơ đồ Hoóc-ne}
Tổng quát với đa thức bị chia là $a_0x^n+a_1x^{n-1}+a_2x^{n-2}+\ldots+a_{n-1}x+a_n$, đa thức chia là $x-a$, thương là $b_0x^{n-1}+b_1x^{n-2}+\ldots+b_{n-2}x+b_{n-1}$, dư $r$. Ta cần chứng minh rằng
\begin{eqnarray*}
&b_0&=a_0\\
&b_1&=ab_0+a_1\\
&b_2&=ab_1+a_2\\
&\ldots&\\
&b_{n-1}&=ab_{n-2}+a_{n-1}\\
&r&=ab_{n-1}+a_n.
\end{eqnarray*}
Thật vậy, thực hiện phép tính
\[(x-1)(b_0x^{n-1}+b_1x^{n-2}+\ldots+b_{n-2}x+b_{n-1})+r\]
rồi rút gọn, ta được
\[b_0x^{n}+(b_1-ab_0)x^{n-1}+\ldots+(b_{n-1}-ab_{n-2})x-ab_{n-1}+r.\]
Đồng nhất đa thức này với đa thức bị chia, ta được
\begin{eqnarray*}
	&b_0&=a_0\\
	&b_1-ab_0&=a_1\\
	&b_2-ab_1&=a_2\\
	&\ldots&\\
	&b_{n-1}-ab_{n-2}&=a_{n-1}\\
	&r-ab_{n-1}&=a_n.
\end{eqnarray*}
Từ đó, suy ra điều phải chứng minh.
\subsubsection{Áp dụng sơ đồ Hoóc-ne để tính giá trị của đa thức $f(x)$ tại $4x=a$}
Sơ đồ Ho óc-ne cho ta thương và dư khi chia đa thức $f(x)$ cho nhị thức $x-a$. Chú ý rằng theo định lí Bê-du, số dư khi chia $f(x)$ cho $x-a$ bằng $f(a)$. Do đó, dùng sơ đồ Hoóc-ne ta cũng tính được giá trị của đa thức $f(x)$ tại $x=a$.
\begin{vd}%[Nguyễn Cao Cường]%[8D1G12]
	Tính giá trị của đa thức $f(x)=x^3+3x^2-4$ tại $x=37$.
	\loigiai{
	Theo định lý Bê-du, $f(37)$ là số dư khi chia $f(x)$ cho $x-37$. Ta lập sơ đồ Hoóc-ne
\begin{center}
	\begin{tabular}{c|c|c|c|c}
		&$1$ & $3$  &$0$  & $-4$  \\ 
		\hline 
		$a=37$& $1$  & $37\cdot 1+3=40$ & $37\cdot 40+0=1480$  & $37\cdot 1480-4=54756$
	\end{tabular} 
\end{center}
Vậy $f(37)=54756$.	
}
\end{vd}

\subsection{Chứng minh một đa thức chia hết cho một đa thức khác}
Ta chỉ xét các đa thức một biến, thường có các cách sau:
\subsubsection{Cách 1.} Phân tích đa thức bị chia thành nhân tử, trong đó có một nhân tử là đa thức chia.
\begin{vd}%[Nguyễn Cao Cường]%[8D1G12]
	Chứng minh rằng $x^{8n}+x^{4n}+1$ chia hết cho $x^{2n}+x^{n}+1$, với mọi số tự nhiên $n$.
	\loigiai{
	Ta có
	\begin{eqnarray*}
	x^{8n}+x^{4n}+1&=&x^{8n}+2x^{4n}+1-x^{4n}=(x^{4n}+1)^2-\left(x^{2n}\right)^2\\
	&=&(x^{4n}+x^{2n}+1)(x^{4n}-x^{2n}+1).
	\end{eqnarray*}
Tiếp tục phân tích
\begin{eqnarray*}
x^{4n}+x^{2n}+1&=&x^{4n}+2x^{2n}+1-x^{2n}=(x^{2n}+1)^2-(x^n)^2\\&=&(x^{2n}+x^{n}+1)(x^{2n}-x^{n}+1).
\end{eqnarray*}
Vậy $x^{8n}+x^{4n}+1$ chia hết cho $x^{2n}+x^{n}+1$, với mọi số tự nhiên $n$.
}
\end{vd}
\subsubsection{Cách 2.} Biến đổi đa thức bị chia thành một tổng các đa thức chia.
\begin{vd}%[Nguyễn Cao Cường]%[8D1G12]
	Chứng minh rằng $x^{3m+1}+x^{3n+2}+1$ chia hết cho $x^2+x+1$ với mọi số tự nhiên $m$, $n$.
	\loigiai{
	Ta có 
	\begin{eqnarray*}
	x^{3m+1}+x^{3n+2}+1&=x^{3m+1}-x+x^{3n+2}-x^2+x^2+x+1\\&=x(x^{3m}-1)+x^2(x^{3n}-1)+(x^2+x+1).
	\end{eqnarray*}
	Ta thấy $x^{3m}-1$ và $x^{3n}-1$ chia hết cho $x^3-1$, do đó chia hết cho $x^2+x+1$.\\ Vậy $x^{3m+1}+x^{3n+2}+1$ chia hết cho $x^2+x+1$ với mọi số tự nhiên $m$, $n$.
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[8D1G12]
	Chứng minh rằng với mọi số tự nhiên $m$, $n$ thì 
	$$x^{6m+4}+x^{6n+2}+1 \; \text{chia hết cho }x^2-x+1.$$
	\loigiai{
	\begin{eqnarray*}
	x^{6m+4}+x^{6n+2}+1&=&x^{6m+4}-x^4+x^{6n+2}-x^2+x^4+x^2+1\\
	&=&x^4(x^{6m}-1)+x^2(x^{6n}-1)+(x^4+x^2+1).
	\end{eqnarray*}
Do $x^{6m}-1\;\vdots\; x^6-1$, $x^{6n}-1\;\vdots x^6-1$ và 
\begin{center}
	$x^6-1=(x^3+1)(x^3-1)\;\vdots \;x^2-x+1$;\\
	$x^4+x^2+1=(x^2+1)^2-x^2\;\vdots\; x^2-x+1$.
\end{center}
Nên suy ra điều cần chứng minh.
}
\end{vd}
\subsubsection{Cách 3.}
Sử dụng các biến đổi tương đương, chẳng hạn để chứng minh $f(x)\;\vdots\; g(x)$, có thể chứng minh $f(x)+g(x)\;\vdots \; g(x)$ hoặc $f(x)-g(x)\; \vdots \; g(x)$.\\
Xem bài tập $268$.
\subsubsection{Cách 4.}
Chứng tỏ rằng mọi nghiệm của đa thức chia đều là nghiệm của đa thức bị chia (ta công nhận rằng điều này dẫn đến đa thức bị chia chia hết cho đa thức chia).
\begin{vd}%[Nguyễn Cao Cường]%[8D1G12]
	Cho $f(x)=(x^2+x-1)^{10}+(x^2-x+1)^{10}-2$. Chứng minh rằng $f(x)$ chia hết cho $x^2-x$.
	\loigiai{
	Đa thức chia có hai nghiệm $x=0$ và $x=1$. Ta sẽ chứng tỏ rằng $x=0$ và $x=1$ cũng là nghiệm của  đa thức bị chia.\\
	Ta có $f(0)=1+1-2=0$ nên $f(x)$ chia hết cho $x$. Ta lại có $f(1)=1+1-2=0$ nên $f(x)$ chia hết cho $x-1$. Các nhân tử $x$ và $x-1$ không chứa nhân tử chung.\\
	Do đó $f(x)$ chia hết cho $x(x-1)$.
}
\end{vd}

\begin{bt}%[Nguyễn Cao Cường]%[8D1G12]
	Cho đa thức $f(x)$ có các hệ số nguyên. Biết rằng $f(0)$, $f(1)$ là các số lẻ. Chứng minh rằng đa thức $f(x)$ không có nghiệm nguyên.
	\loigiai{
	Giả sử $a$ là nghiệm nguyên của $f(x)$. Với mọi $x$, ta có $f(x)=(x-a)\cdot Q(x)$, trong đó $Q(x)$ là đa thức có hệ số nguyên, do đó
	$$f(0)=-a\cdot Q(0);\; f(1)=(1-a)\cdot Q(1).$$
	Do $f(0)$ là số lẻ nên $a$ là số lẻ, do $f(1)$ là số lẻ nên $1-a$ là số lẻ, mâu thuẫn với nhau.
}
