\begin{ex}%[Đề thi vào 10, Sở giáo dục Trà Vinh, 2017]%[9D1Y7]%[9D3B3]%[9D4Y4]
    \hfill
    \begin{enumerate}
        \item Rút gọn biểu thức: $A=\dfrac{1}{3+2\sqrt{2}}+\dfrac{1}{3-2\sqrt{2}}$.
        \item Giải hệ phương trình $\heva{3x-y&=7\\ 5x+y&=9.}$
        \item Giải phương trình: $x^2-3x-10=0$.
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item Ta có: $A=\dfrac{1}{3+2\sqrt{2}}+\dfrac{1}{3-2\sqrt{2}}=\dfrac{3-2\sqrt{2}+3+2\sqrt{2}}{\left( {3 + 2\sqrt 2 } \right)\left( {3 - 2\sqrt 2 } \right)}=6$.\\
        Vậy $A=6$.
        \item Ta có: $\heva{3x-y&=7\\ 5x+y&=9}$ $\Leftrightarrow$ $\heva{8x&=16\\ 5x+y&=9}$ $\Leftrightarrow$ $\heva{x&=2\\ y&=-1}$.\\
        Vậy hệ phương trình đã cho có nghiệm duy nhất $(x; y)=(2; -1)$.
        \item Ta có: $\Delta=(-3)^2-4.1.(-10)=49$.\\
        Vậy phương trình đã cho có hai nghiệm phân biệt: $x_1=\dfrac{3+\sqrt{49}}{2}=5$; $x_2=\dfrac{3-\sqrt{49}}{2}=-2$.
    \end{enumerate}
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Trà Vinh, 2017]%[9D2B3]%[9D4B2]%[9D4B4]
Cho hai hàm số $y=x+2$ và $y=x^2$ có đồ thị lần lượt là $(d)$ và $(P)$.
    \begin{enumerate}    
        \item Vẽ $(d)$ và $(P)$ trên cùng một hệ trục tọa độ.
        \item Bằng phép toán tìm tọa độ giao điểm của $(d)$ và $(P)$.
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item Bảng giá trị của hàm số $y=x+2$ là:
        		\begin{center}
        			\begin{tabular}{|c|c|c|}
        			\hline 
        			$x$ & $-2$ & $0$ \\ 
        			\hline 
        			$y=x+2$ & $0$ & $2$ \\
        			\hline
        			\end{tabular}
        		\end{center}
        		Bảng giá trị của hàm số $y=x^2$ là:
        		\begin{center}
        			\begin{tabular}{|c|c|c|c|c|c|}
        			\hline 
        			$x$ & $-2$ & $-1$	& $0$ & $1$ & $2$ \\ 
        			\hline 
        			$y=x^2$ & $4$ & $1$ & $0$ & $1$ & $4$\\ 
        			\hline
        			\end{tabular}
        		\end{center}
        		Đồ thị:
        		\begin{center}
        			\begin{tikzpicture}[line width=1pt,>=stealth,x=1cm,y=1cm,scale=.8]
        				%\draw[thin,color=gray!50] (-4,-1) grid (4,5);        				
        				\draw[->] (-4,0) -- (4,0) node[right] {$x$};
        				\foreach \x in {-3,-2,-1,1,2,3} 
        				\draw[shift={(\x,0)},color=black] (0pt,2pt) -- (0pt,-2pt) node[below] {\footnotesize $\x$};       	
        				\draw[->] (0,-1) -- (0,5) node[above] {$y$};
        				\foreach \y in {1,2,3,4}
        				\draw[shift={(0,\y)},color=black] (2pt,0pt) -- (-2pt,0pt) node[left] {\footnotesize $\y$};
        				\draw (0,0) node[below left]{\footnotesize $O$};
        				\clip (-4,-1) rectangle (4,5);
        				\draw[color=red]    plot (\x,\x+2)   node[right] {$(d)$};
        				\draw[samples=200,color=blue]   plot (\x,{(\x)^2}) 	node[right] {$(P)$};
        				\draw[dashed] (-1,0) -- (-1,1) -- (1,1) -- (1,0);
        				\draw[dashed] (-2,0) -- (-2,4) -- (2,4) -- (2,0);
        				\fill (0,0) circle (1.5pt);
        				\fill (-1,1) circle (1.5pt)  node[left] {$A$};
        				\fill (2,4) circle (1.5pt)   node[right] {$B$};
        				\fill (1,1) circle (1.5pt);
        				\fill (-2,4) circle (1.5pt);
        				\fill (-2,0) circle (1.5pt);
        				\fill (0,2) circle (1.5pt);   				
        			\end{tikzpicture}
        		\end{center}        		
        \item Phương trình hoành độ giao điểm của $(d)$ và $(P)$ là:\\
        		\hspace*{1cm} $x^2=x+2 \Leftrightarrow x^2-x-2=0 \Leftrightarrow \hoac{x&=-1 \\ x&=2}$\\
       			Vậy $(d)$ cắt $(P)$ tại hai điểm là: $A(-1; 1)$ và $B(2; 4)$.
    \end{enumerate}
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Trà Vinh, 2017]%[9D4K5]%[9D4G6]
    Cho phương trình $x^2-2(m-2)x-6m=0$ $(1)$ (với $m$ là tham số).
    \begin{enumerate}
        \item Chứng minh rằng phương trình (1) luôn có hai nghiệm phân biệt với mọi giá trị của $m$.
        \item Gọi $x_1$, $x_2$ là hai nghiệm của phương trình $(1)$. Tìm giá trị nhỏ nhất của biểu thức $P=x_1^2 + x_2^2$.
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item Ta có: $\Delta '=(m-2)^2-1.(-6m)=m^2+2m+4=(m+1)^2+3>0$, với mọi $m$.\\
        		Vậy phương trình $(1)$ luôn có hai nghiệm phân biệt với mọi giá trị của $m$.
        \item Theo định lí Vi-et, ta có: $x_1+x_2=2(m-2)$ và $x_1 x_2=-6m$.\\
        		Ta có:\\
        		 $
       			 \begin{aligned}
       				 P&=x_1^2 + x_2^2\\
        			&=\left(x_1 + x_2\right)^2-2x_1x_2\\ 
        			&=4(m-2)^2-2.(-6m)\\
        			&=4m^2-4m+16\\
        			&=(2m-1)^2+15\geq 15
        		\end{aligned}
        		$\\
        		Vậy giá trị nhỏ nhất của $P$ là $15$, đạt được khi $m=\dfrac{1}{2}$.
    \end{enumerate}
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Trà Vinh, 2017]%[9H3B3]%[9H3K3]%[9H1G4]
Cho đường tròn tâm $O$ bán kính $R$, đường kính $BC$. Gọi $A$ là một điểm thuộc đường tròn ($A$ khác $B$ và $C$). Đường phân giác $\widehat{BAC}$ cắt $BC$ tại $D$ và cắt đường tròn tại $M$.
    \begin{enumerate}
        \item Chứng minh $MB=MC$ và $OM$  vuông góc với $BC$.
        \item Gọi $E$, $F$ lần lượt là hình chiếu của $D$ lên $AB$, $AC$. Tứ giác $AEDF$ là hình gì?
        \item Cho $\widehat{ABC}=60^\circ$. Tính diện tích tam giác $MDC$ theo $R$.
    \end{enumerate}
\loigiai
    {
    \begin{center}
    	\begin{tikzpicture}[scale=.7]
    		\tkzDefPoints{0/0/O, -3/0/C, 3/0/B, 1.5/2.6/A, 0/-3/M}    		
    		\tkzDrawCircle[diameter](B,C)
    		\tkzDrawPolygon(A,B,M,C)
    		\tkzDrawSegments(A,M B,C M,O)
    		\tkzInterLL(A,M)(B,C) 		\tkzGetPoint{D}
    		\tkzDrawAltitude(A,B)(D)		\tkzGetPoint{E}
    		\tkzDrawAltitude(A,C)(D)	\tkzGetPoint{F}
    		\tkzDrawPoints(A,B,C,M,D,E,F,O)
    		\tkzLabelPoints[below right](D)
    		\tkzLabelPoints[above](O,A,F)
    		\tkzLabelPoints[right](B)
    		\tkzLabelPoints[below](M,E)
    		\tkzLabelPoints[left](C)
    		\tkzMarkRightAngle(C,A,B)
    		\tkzMarkRightAngle(D,E,A)
    		\tkzMarkRightAngle(D,F,A)
    		\tkzMarkRightAngle(M,O,C)
    	\end{tikzpicture}
    \end{center}
    \begin{enumerate}
        \item Vì $AM$ là tia phân giác của $\widehat{BAC}$ nên $\widehat{MAB}=\widehat{MAC}$.\\
        		$\Rightarrow$ $\stackrel\frown{MB} = \stackrel\frown{MC}$ $\Rightarrow$ $MB=MC$.\\
        		Tam giác $MBC$ có $MB=MC$ (\textit{cmt}) nên tam giác $MBC$ cân tại $M$.\\
        		Mà $MO$ là đường trung tuyến của tam giác $MBC$ nên $MO$ cũng là đường cao của tam giác $MBC$.\\
        		$\Rightarrow$ $MO \perp BC$.
        \item Ta có: $\widehat{BAC}=90^\circ$ (góc nội tiếp chắn nửa đường tròn).\\
        		$\Rightarrow$ Tứ giác $AEDF$ là hình chữ nhật (tứ giác có ba góc vuông).\\
        		Mà $AM$ là tia phân giác của $\widehat{BAC}$ nên $\widehat{MAC}=45^\circ$.\\
        		$\Rightarrow$ Tam giác $AED$ vuông cân tại $E$ $\Rightarrow$ $ED=ED$.\\
        		Vậy tứ giác $AEDF$ là hình vuông.
        \item Tam giác $ABC$ vuông tại $A$ có $\widehat{ABC}=60^\circ$ nên ta có:\\
       			 \hspace*{0.5cm} $AB=BC.\cos 60^\circ=R$ và $AC=BC.\sin 60^\circ=R\sqrt{3}$.\\
       			 Tam giác $ABC$ có $AD$ là đường phân giác nên ta có: $\dfrac{DB}{DC}=\dfrac{AB}{AC}=\dfrac{R}{R\sqrt{3}}=\dfrac{1}{\sqrt{3}}$.\\
       			 $\Rightarrow$ $\dfrac{DB}{1}=\dfrac{DC}{\sqrt{3}}=\dfrac{DB+DC}{1+\sqrt{3}}=\dfrac{BC}{1+\sqrt{3}}=\dfrac{2R}{1+\sqrt{3}}$.\\
       			 $\Rightarrow$ $DC=\dfrac{2R\sqrt{3}}{1+\sqrt{3}}$.\\
       			 Vậy diện tích tam giác $MDC$ là: $S_{\Delta MDC}=\dfrac{1}{2}MO.DC=\dfrac{1}{2}.R.\dfrac{2R\sqrt{3}}{1+\sqrt{3}}=\dfrac{R^2\sqrt{3}}{1+\sqrt{3}}$ (đvdt).
    \end{enumerate}
    }
\end{ex}
