\begin{ex} %[Đề thi vào 10, Chuyên Hạ Long - Quảng Ninh, 2017]%[9D1B8]
	
Cho biểu thức: 
\begin{align*}
A  =  \left(\dfrac{\sqrt{3}}{x^2  +  x\sqrt{3}  +  3}  +  \dfrac{3}{x^3  -  \sqrt{27}}\right)\left(\dfrac{x}{\sqrt{3}}  +  \dfrac{\sqrt{3}}{x}  +  1\right)\text{ với } x\ne 0;x\ne \sqrt{3}.
\end{align*}
\begin{enumerate}
\item Rút gọn biểu thức $A$.
\item Tính giá trị của biểu thức $A$ khi $x  =  \sqrt{3}  +  \sqrt{\sqrt{5}  -  \sqrt{3  -  \sqrt{29  -  12\sqrt{5}}}}$.
\end{enumerate}
\loigiai{\\
1. Rút gọn biểu thức A
\begin{align*}
A&=\left(\dfrac{\sqrt{3}}{x^2+x\sqrt{3}+3}+\dfrac{3}{x^3-\sqrt{27}}\right)\left(\dfrac{x}{\sqrt{3}}+\dfrac{\sqrt{3}}{x}+1\right)\\
&=\left(\dfrac{\sqrt{3}}{x^2+x\sqrt{3}+3}+\dfrac{3}{(x-\sqrt{3})(x^2+x\sqrt{3}+3)}\right)\left(\dfrac{x^2+x\sqrt{3}+3}{\sqrt{3}x}\right)\\
&=\left(\dfrac{(x-\sqrt{3})\sqrt{3}+3}{(x-\sqrt{3})(x^2+x\sqrt{3}+3)}\right)\left(\dfrac{x^2+x\sqrt{3}+3}{\sqrt{3}x}\right)=\dfrac{1}{x-\sqrt{3}}
\end{align*}
2. Ta có: 
\begin{align*}
x &= \sqrt{3} + \sqrt{\sqrt{5} - \sqrt{3 - \sqrt{29 - 12\sqrt{5}}}}=  \sqrt{3} + \sqrt{\sqrt{5} - \sqrt{3 - \sqrt{{{(2\sqrt{5} - 3)}^{2}}}}}\\
&=\sqrt{3} + \sqrt{\sqrt{5} - \sqrt{6 - 2\sqrt{5}}}= \sqrt{3} + \sqrt{\sqrt{5} - \sqrt{{{(\sqrt{5} - 1)}^{2}}}}=\sqrt{3} + 1
\end{align*}
Thay $x=\sqrt{3}+1$ vào A ta có $A=\dfrac{1}{x-\sqrt{3}}=\dfrac{1}{\sqrt{3}+1-\sqrt{3}}=1$
}
\end{ex}

\begin{ex} %[Đề thi vào 10, Chuyên Hạ Long - Quảng Ninh, 2017]%[9D4K7]%[9D3K3] 
	\hfill
\begin{enumerate}
\item Giải phương trình $x^3 - x^2 - x\sqrt{x - 1} -2 = 0$.
\item Giải hệ phương trình $\heva{& x^2 + xy - 2y^2 = 0 \\ & xy + 3y^2 + x = 3}$
\end{enumerate}
\loigiai{
1. ĐK: $x\ge 1$. Biến đổi về phương trình $x^2(x - 1) - x\sqrt{x - 1}  -2 = 0$.\\
Đặt $t = x\sqrt{x - 1}(t\ge 0)\Rightarrow t^2 = x^2(x - 1)$. Phương trình đã cho trở thành:
\begin{align*}
t^2 - t - 2 = 0\Leftrightarrow \left[\begin{matrix}
t = -1 \\ 
t =2 \\ 
\end{matrix} \right.
\end{align*}
Kết hợp với điều kiện, ta được $t = 2$.\\
Với $t = 2 \Rightarrow x\sqrt{x - 1} = 2\Leftrightarrow x^3 - x^2 = 4\Leftrightarrow (x - 2)(x^2 + x + 2) = 0 \Leftrightarrow x=2$\\
2. Giải hệ phương trình $\heva{&x^2 + xy - 2y^2 =0 & (1) \\ & xy + 3y^2 + x = 3 & (2)}$\\ 
Phương trình (1) $\Leftrightarrow \left(x^2  -  y^2\right) + y(x - y)  = 0 \Leftrightarrow (x - y)\left(x + 2y\right)  = 0$, 
ta được $x = y$ hoặc $x = -2y$
\begin{itemize}
\item Với $x = y$, từ (2) ta có: $4x^2 + x -3 = 0$, ta được $x_1 =- 1 ,x_2 = \dfrac{3}{4}$. Khi đó, $x_1 = y_1 = -  1,x_2 = y_2 = \dfrac{3}{4}$.
\item Với $x = -2y$, từ (2) ta có $y^2 -2y - 3 = 0$, ta được $y_1 = - 1,y_2 = 3$
Nếu $y = -1\Rightarrow x = 2$. Nếu $y = 3\Rightarrow x = -6$.
\end{itemize}
Vậy hệ phương trình có nghiệm $(x; y)$ là: $(-1;-1); \left(\dfrac{3}{4};\dfrac{3}{4}\right); (2;-1); (-6; )$.
}
\end{ex}
\begin{ex} %[Đề thi vào 10, Chuyên Hạ Long - Quảng Ninh, 2017]%[9D5G5] 
Tìm các số tự nhiên $n$ để $A = n^{2018} + n^{2008} + 1$ là số nguyên tố.
\loigiai{
Tìm số tự nhiên   để $A = n^{2018} + n^{2008} + 1$ là số nguyên tố.
\begin{itemize}
\item Xét $n = 0$ thì $A = 1$ không là số nguyên tố; 
\item Xét $n = 1$ thì $A = 3$ là số nguyên tố.
\item Xét $n > 1$, ta thấy $A > n^2 + n + 1$
\begin{align*}
A&= n^{2018} - n^2 + n^{2008} - n + n^2 + n + 1\\
&=  n^2((n^3)^{672} - 1) + n.((n^3)^{669} - 1) + (n^2 + n + 1)
\end{align*}
\end{itemize}
 mà $(n^3)^{672} - 1$ chia hết cho $n^3 - 1$, suy ra $(n^3)^{672} - 1$ chia hết cho $n^2 + n + 1$.\\
 Tương tự: $(n^3)^{669} – 1$ chia hết cho $n^2 + n + 1$\\
 Khi đó $A$ chia hết cho $n^2 + n + 1 >1$ và $A > n^2 + n + 1$ nên $A$ là hợp số.\\
 Tóm lại số tự nhiên cần tìm là $n = 1$.
}
\end{ex}

\begin{ex} %[Đề thi vào 10, Chuyên Hạ Long - Quảng Ninh, 2017]%[9H2K6]%[9H2G6]
Cho đường tròn $(O; R)$, đường kính $AB, M$ là một điểm tùy ý thuộc đường tròn ($M$ khác $A$ và $B$). Qua $A$ và $B$ lần lượt kẻ các đường thẳng $d$ và $d'$ là tiếp tuyến với đường tròn. Tiếp tuyến tại $M$ của đường tròn cắt $d$ và $d'$ lần lượt tại $C$ và $D$. Đường thẳng $BM$ cắt $d$ tại $E$.
\begin{enumerate}
\item Chứng minh $CM  =  CA  =  CE$.
\item Chứng minh $AD\perp OE$.
\item Tính độ dài đoạn $AM$ theo $R$, nếu $AE  =  BD$.
\end{enumerate}
\loigiai{ \\
1. Gọi $F$ là giao điểm của $OC$ và $AM$, ta có $OC\perp AM$.\\
Ta có, $CM  =  CA$ (tính chất hai tiếp tuyến cắt nhau).\\
Hai tam giác vuông $AME$ và $AFC$ đồng dạng, nên $\dfrac{AE}{AC}  =  \dfrac{AM}{AF}  =  2\Rightarrow AE  =  2AC\Rightarrow AC  =  CE$.\\
Vậy $CM  =  CA  =  CE$.\\
\immini
{
2. Gọi giao điểm của $EO$ với $d’$ là $I$, Chứng minh được $AEBI$ là hình bình hành $\Rightarrow BE \parallel AI$.\\
Ta có, $OD\bot BE$, $OD\perp AI$ mà $AB \bot DI\Rightarrow $ O là trực tâm của $\Delta ADI\Rightarrow  OI\bot  AD \Rightarrow OE\perp  AD$ (đpcm).\\
3. Tam giác $COD$ vuông tại $O$ (vì $OC, OD$ là hai phân giác của hai góc kề bù), có $OM$ là đường cao nên $OM^2  =  CM.MD$.\\
Theo phần 1, ta có $EC = CA = CM \Rightarrow 2CM = AE$,\\
mà $BD = MD và AE = BD (gt) \Rightarrow  2CM = MD$.\\
$\Rightarrow 2CM^2 = R^2$ (do $MO = R$ và $OM^2 = CM.MD$) $\Rightarrow CM = \dfrac{R\sqrt{2}}{2}\Rightarrow  AE = R\sqrt{2}$ (do $AE = 2CM$).\\

Do trong giác vuông $AEB$ tại $A$, ta có $\dfrac{1}{AM^2} = \dfrac{1}{AE^2} + \dfrac{1}{AB^2}\Rightarrow AM = \dfrac{AE.AB}{\sqrt{AE^2 + AB^2}} = \dfrac{2R\sqrt{3}}{3}$.
}

\begin{ex} %[Đề thi vào 10, Chuyên Hạ Long - Quảng Ninh, 2017]%[9D5G1]
Cho $a;\ b$ thoả mãn $\left| a\right|\ge 2;\ \left| b\right|\ge 2$. Chứng minh rằng:
\begin{align*}
(a^2 + 1)(b^2 + 1)\ge (a + b)(ab + 1) + 5
\end{align*}
\loigiai{\\
Xét hiệu 
\begin{align*}
M  &= (a^2 + 1)(b^2 + 1) - (a + b)(ab + 1) - 5\\
&= (a^2b^2 - a^2b - ab^2 + ab) + (a^2 + b^2 - a - b - ab) -4\\
&= ab(a - 1)(b - 1)  +  \dfrac{1}{2}\left[{{(a - b)}^{2}} + a(a - 2) +  b(b - 2)\right] - 4
\end{align*}
Chỉ ra với $\left| a\right|\ge 2$ thì $a(a  -  1)\ge 2$ và $a(a  -  2)\ge 0$ $\left| b\right|\ge 2$ thì $b(b  -  1)\ge 2$ và $b(b  -  2)\ge 0$
nên $ab(a - 1)(b - 1)\ge 4$; $\dfrac{1}{2}\left[{{(a - b)}^{2}} + a(a - 2) + b(b -2)\right]\ge 0 \Rightarrow M\ge 0$ hay $(a^2 + 1)(b^2 +1)\ge (a + b)(ab + 1) + 5$.
}
\end{ex}
