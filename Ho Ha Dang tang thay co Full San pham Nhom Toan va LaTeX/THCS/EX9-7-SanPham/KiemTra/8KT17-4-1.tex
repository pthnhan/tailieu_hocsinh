\begin{ex}%[Đề kiểm tra học kì 2, Toán 8, Quận 1, HCM, 2016]%[8D3Y3]%[8D3B3]%[8D3G3]%[8D4K5]
Giải các phương trình sau:
\begin{multicols}{2}
\begin{enumerate}
\item 3( $x - 2) = 2(x - 4)$.  
\item $9{x^2} - 1  = (3x - 1)(5x + 8)$.
\item $\dfrac{{x + 3}}{{x - 3}} + \dfrac{{36}}{{9 - {x^2}}} = \dfrac{{x - 3}}{{x + 3}}$.
\item $2\left| {3x + 1} \right| - 6 = 0$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item $3(x - 2) = 2(x - 4)   \Leftrightarrow  3x - 6 = 2x - 8  \Leftrightarrow x = - 2$. Vậy PT có nghiệm $x = - 2$.	
\item $9x^2 - 1  = (3x - 1)(5x + 8)  \Leftrightarrow   (3x - 1)(3x + 1) - (3x - 1)(5x + 8) = 0$\\
 $ \Leftrightarrow   (3x - 1)( - 2x - 7)  = 0  \Leftrightarrow x =  \dfrac{1}{3}$   hoặc $x = - \dfrac{7}{2}$ .\\
Vậy PT có nghiệm $x = \dfrac{1}{3}$  ; $x = - \dfrac{7}{2}$ .	
\item $\dfrac{x + 3}{x - 3} + \dfrac{36}{9 - x^2} = \dfrac{x - 3}{x + 3}$  $ \Leftrightarrow $  $\dfrac{(x + 3)^2 - 36 - (x - 3)^2}{(x - 3)(x + 3)} = 0$  (1). (ĐKXĐ :  $x \ne  \pm 3$ )\\
PT (1) trở thành :  $(x + 3)^2 - 36 - (x - 3)^2 = 0   \Leftrightarrow  12x  - 36  = 0  \Leftrightarrow  x = 3$.\\
 So với ĐKXĐ giá trị $x= 3$ không thỏa mãn. Vậy PT đã cho vô nghiệm.	
\item  $2\left| 3x + 1 \right| - 6 = 0  \Leftrightarrow   \left| {3x + 1} \right| =3  \Leftrightarrow \hoac{& 3x + 1 = 3\\&3x + 1 = -3 } \Leftrightarrow  \hoac{&x =  \dfrac{2}{3}  \\&x = - \dfrac{4}{3}.}$ \\
  Vậy PT có nghiệm $x = \dfrac{2}{3}$  ; $x = - \dfrac{4}{3}$ .
\end{enumerate}
}
\end{ex}


\begin{ex}%[Đề kiểm tra học kì 2, Toán 8, Quận 1, HCM, 2016]%[8D4B3]%[8D4G3]
    \hfill
\begin{enumerate}
\item Giải bất phương trình sau và biểu diễn tập nghiệm trên trục số:
					 $$\dfrac{{6x + 5}}{2} - \dfrac{{10x + 3}}{4} \ge 2x + \dfrac{{2x + 1}}{2}.$$ 
\item Cho $x$, $y$ dương thỏa mãn: $x + y = 3$. Chứng minh rằng: $x^2y \le  4$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $\dfrac{{6x + 5}}{2} - \dfrac{{10x + 3}}{4} \ge 2x + \dfrac{{2x + 1}}{2}$  $ \Leftrightarrow $  $\dfrac{{12x + 10}}{4} - \dfrac{{10x + 3}}{4} \ge \dfrac{{8x}}{4} + \dfrac{{4x + 2}}{4}$  $ \Leftrightarrow $ $10x  \le  5  \Leftrightarrow  x  \le   \dfrac{1}{2}$ .\\
Vậy tập nghiệm BPT là $S=\left\{ {x \in \mathbb{R}/x \le \dfrac{1}{2}} \right\}$ .	
\item Ta có : $x + y = 3  \Rightarrow  y = 3 - x$. \\
Vậy $x^2y  \le  4  \Leftrightarrow   x^2(3 - x )  \le  4  \Leftrightarrow  x^3 - 3x^2 + 4  \ge  0  \Leftrightarrow (x - 2)^2(x + 1)  \ge  0$ (BĐT đúng, vì $x>0$ )	
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề kiểm tra học kì 2, Toán 8, Quận 1, HCM, 2016]%[8D3K6]
Giải toán bằng cách lập phương trình:	  \\ 
Một ô tô đi từ $A$ đến $B$ với vận tốc $42$ km/h và đi từ $B$ về $A$ với vận tốc lớn hơn vận tốc lúc đi là 	$6$ km/h. Tính quãng đường $AB$ biết thời gian cả đi và về mất $5$ h.
\loigiai{
Gọi $x$ (km) là độ dài quãng đường $AB$ (ĐK : $x > 0$).\\
Thời gian đi từ $A$ đến $B$ là :  $\dfrac{x}{{42}}$ (h) .\\
Thời gian đi từ $B$ về $A$ là :  $\dfrac{x}{{42 + 6}} = \dfrac{x}{{48}}$ (h). \\
Do thời gian cả đi và về mất $5$ h, nên ta có phương trình :\\
  $\dfrac{x}{{42}} + \dfrac{x}{{48}}= 5  \Leftrightarrow   \dfrac{{5x}}{{112}} = 5  \Leftrightarrow  x = 112$.\\
So với ĐK $x = 112$ thỏa mãn. Vậy quãng đường $AB$ dài $112$ (km).
}
\end{ex}

\begin{ex}%[Đề kiểm tra học kì 2, Toán 8, Quận 1, HCM, 2016]%[8H3B8]%[8H3K9]%[8H3G9]
Cho tam giác $ABC$ vuông tại $A$, vẽ đường cao $AH$.
\begin{enumerate}
\item Chứng minh rằng:  $\triangle ABC \backsim  \triangle HBA$. Từ đó suy ra $AB^2 = BH. BC$.
\item Chứng minh rằng:  $\triangle HAB \backsim \triangle HCA$.  Từ đó suy ra $AH^2 = BH. CH$.
\item Vẽ $HD$ vuông góc $AC$ tại $D$. Đường trung tuyến $CM$ của tam giác $ABC$ cắt $HD$ tại $N$. Chứng minh rằng  $\dfrac{{HN}}{{BM}} = \dfrac{{CN}}{{CM}}$ và $HN = DN$.
\item  Qua $A$ vẽ đường thẳng $d$ song song với $BC$. Trên đường thẳng $d$ lấy điểm $E$ ($E$ và $C$ nằm trên  cùng nửa mặt phẳng bờ $AH$) sao cho  $\dfrac{{AE}}{{BC}} = \dfrac{{AD}}{{CD}}$ . Gọi $I$ là giao điểm của $AH$ và $CM$.\\
      Chứng minh rằng ba điểm $B$,$ E$,$ I$ thẳng hàng.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1]
\tkzInit[xmin=-3.5,xmax=4.5,ymin=-3.5,ymax=3.5]
\clip (-1,-2.2) rectangle (4.5,4.5); 
\tkzDefPoint[label=below left:$A$](0,0){A}
\tkzDefPoint[label=left:$B$](0,3){B}
\tkzDefPoint[label=right:$C$](4,0){C}
\tkzDefPointBy[projection=onto B--C](A)
\tkzGetPoint{H}
\tkzDefPointBy[projection=onto A--C](H)
\tkzGetPoint{D}
\tkzDefMidPoint(A,B)  \tkzGetPoint{M}
\tkzDefLine[parallel=through A](B,C) 
\tkzGetPoint{d}
\tkzInterLL(C,M)(H,D) \tkzGetPoint{N}
\tkzInterLL(C,M)(A,H) \tkzGetPoint{I}
\tkzInterLL(A,d)(B,I) \tkzGetPoint{E}
\tkzMarkRightAngles[fill=green!60!cyan,size=0.2,opacity=1](B,A,C A,H,B H,D,C)
\tkzDrawLine[color=blue,add = 0.5 and 0.2](A,d)
\tkzDrawSegments[color=blue](A,C A,B C,B D,H A,H C,M B,E)
\tkzDrawPoints[color=red](A,B,C,D,M,I,E,H,N)
\tkzLabelPoints[above](H)
\tkzLabelPoints[above right](N)
\tkzLabelPoints[below](I)
\tkzLabelPoints[below left](D,E,M)

\end{tikzpicture} 
\end{center}

\begin{enumerate}
\item Xét $\triangle ABC$ và $ \triangle HBA$ có:
\begin{itemize}
\item $\widehat{BAC}=\widehat{BHA}=90^\circ$.
\item $\widehat{B}$ chung.
\end{itemize}
$\Rightarrow \triangle ABC \backsim \triangle HBA$ (g.g)  $ \Rightarrow   \dfrac{{AB}}{{BH}} = \dfrac{{BC}}{{AB}}   \Rightarrow  AB^2 = BH. BC$.
\item Xét $\triangle  HAB$ và $ \triangle  HCA$ có:
\begin{itemize}
\item $\widehat{AHB}=\widehat{CHA}=90^\circ$.
\item $\widehat{HAB}=\widehat{HCA}$ (cùng phụ với góc $\widehat{HAC}$).
\end{itemize}
$\Rightarrow \triangle  HAB \backsim \triangle  HCA$ (g.g)
 $ \Rightarrow   \dfrac{{AH}}{{CH}} = \dfrac{{BH}}{{AH}}   \Rightarrow  AH^2 = BH. CH$.
\item Ta có : $AB \parallel   DH$ ( vì cùng vuông góc $AC$). \\
Xét  $\triangle BCM$ có $HN \parallel BM \Rightarrow \dfrac{{HN}}{{BM}} = \dfrac{{CN}}{{CM}}$  (hệ quả của định lý Talét).\\
Xét  $\triangle ACM$ có $DN \parallel AM \Rightarrow  \dfrac{{DN}}{{AM}} = \dfrac{{CN}}{{CM}}$ . \\
Do đó :  $\dfrac{{HN}}{{BM}} = \dfrac{{DN}}{{AM}}$ mà $BM = AM$ (gt) nên : $HN = DN$.
\item  Xét $\triangle DAE$  và $ \triangle DCB$ có:
\begin{itemize}
\item $\widehat{DAE}=\widehat{DCB}$ (so le trong, $AE \parallel BC$).
\item $\dfrac{AD}{DC}=\dfrac{AE}{BC}$ (GT).
\end{itemize}
$\Rightarrow \triangle DAE \backsim \triangle DCB$ (c.g.c)  $ \Rightarrow \widehat {ADE} = \widehat {BDC}$.\\
Ta có :  $\widehat {ADE} + \widehat {ADB} = \widehat {BDC} + \widehat {ADB} = {180^\circ}$
 $ \Rightarrow  B, D, E$ thẳng hàng  (1).\\
 $\triangle IAM$ có $HN \parallel AM \Rightarrow  \dfrac{{IH}}{{IA}} = \dfrac{{HN}}{{AM}}   \Rightarrow   \dfrac{{IH}}{{IA}} = \dfrac{{2HN}}{{2AM}} = \dfrac{{DH}}{{AB}}   \Rightarrow   \triangle  IHD \backsim \triangle  IAB$ (c.g.c)\\
 $ \Rightarrow $  $\widehat {HID} = \widehat {AIB}$  $ \Rightarrow $  $\widehat {BIH} + \widehat {HID} = \widehat {BIH} + \widehat {AIB} = {180^\circ}$  $ \Rightarrow  B, I, D$ thẳng hàng  (2).\\
Từ (1); (2) suy ra $B, D, E, I $ thẳng hàng. Vậy : $B, E, I$ thẳng hàng.	
\end{enumerate}
}
 \end{ex}
