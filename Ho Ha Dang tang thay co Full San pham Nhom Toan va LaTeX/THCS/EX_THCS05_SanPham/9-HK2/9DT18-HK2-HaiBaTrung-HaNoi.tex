\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Hai Bà Trưng, Hà Nội ]%[9D1B6]%[9D1B8]%[9D5G1]
Cho hai biểu thức $A=\dfrac{\sqrt{x}+1}{\sqrt{x}-3}$ và $B=\dfrac{2\sqrt{x}}{\sqrt{x}+3}+\dfrac{\sqrt{x}}{\sqrt{x}-3}-\dfrac{3x+3}{x-9}$ với $x\ge 0$, $x\ne 9$.
\begin{enumerate}
\item Tính giá trị của $A$ khi $x=25$.
\item Rút gọn biểu thức $P=B : A$.
\item Tìm giá trị lớn nhất của biểu thức $P$.
\end{enumerate}
	\loigiai{\text{}
	\begin{enumerate}
	\item Thay $x=25$ (thỏa mãn điều kiện $x\ge 0$, $x\ne 9$) vào $A$ ta được $A=\dfrac{\sqrt{25}+1}{\sqrt{25}-3}=3$.
	\item  \begin{eqnarray*}
B&=&\dfrac{2\sqrt{x}}{\sqrt{x}+3}+\dfrac{\sqrt{x}}{\sqrt{x}-3}-\dfrac{3x+3}{x-9}\\
&=&\dfrac{2\sqrt{x}(\sqrt{x}-3)}{(\sqrt{x}-3)(\sqrt{x}+3)}+\dfrac{\sqrt{x}(\sqrt{x}+3)}{(\sqrt{x}-3)(\sqrt{x}+3)}-\dfrac{3x+3}{(\sqrt{x}-3)(\sqrt{x}+3)}\\
&=&\dfrac{-3(\sqrt{x}+1)}{(\sqrt{x}-3)(\sqrt{x}+3)}.
\end{eqnarray*} 
Ta có $P=B:A=\dfrac{-3(\sqrt{x}+1)}{(\sqrt{x}-3)(\sqrt{x}+3)}:\dfrac{\sqrt{x}+1}{\sqrt{x}-3}=\dfrac{-3}{\sqrt{x}+3}$.
\item $P=\dfrac{-3}{\sqrt{x}+3}$. Vì $x \geq 0$, $x\ne 1$ nên ta có $\sqrt{x}+3 \geq 3 \Leftrightarrow \dfrac{-3}{\sqrt{x}+3} \geq -1 \Leftrightarrow P \geq -1$.\\
Dấu bằng xảy ra khi và chỉ khi $x=0$ (thỏa mãn điều kiện).
Vậy $P_{\min}=-1$ khi $x=0$.
	\end{enumerate} 
	}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Hai Bà Trưng, Hà Nội ]%[9D4T8]
\textit{Giải bài toán sau bằng cách lập phương trình hoặc hệ phương trình}:
Hai người cùng làm chung một công việc trong $4$ giờ $48$ phút thì xong. Thời gian người thứ nhất làm một mình xong công việc nhiều hơn thời gian để người thứ hai làm một mình xong công việc là $4$ giờ. Hỏi mỗi người làm một mình xong công việc trong bao lâu hoàn thành công việc?
\loigiai{ 
Gọi thời gian người thứ hai làm một mình xong công việc là $x$ giờ,  $\left( x>\dfrac{24}{5} \right)$.\\
Thời gian người thứ nhất làm một mình xong công việc là $x+4$ giờ.\\
Trong một giờ người thứ hai làm được $\dfrac{1}{x}$ công việc.\\
Trong một giờ người thứ nhất làm được $\dfrac{1}{x+4}$ công việc.\\
Theo bài ra ta có trong một giờ cả hai người làm được $\dfrac{5}{24}$ công việc.\\
Ta có phương trình $\dfrac{1}{x}+\dfrac{1}{x+4}=\dfrac{5}{24}$.\\
Giải phương trình trên ta được $x_1=\dfrac{-12}{5}$ (loại); $x_2=8$.\\
Vậy thời gian người thứ nhất làm một mình xong công việc là $12$ giờ, thời gian người thứ hai làm một mình xong công việc là $8$ giờ.
}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Hai Bà Trưng, Hà Nội ]%[9D4B4]%[9D4K4]%[9D4K6]
Trên mặt phẳng $Oxy$ cho Parabol $(P): y=x^2$ và đường thẳng $(d): y=x-m+3$.
\begin{enumerate}
\item Tìm tọa độ giao điểm của $(d)$ và $(P)$ khi $m=1$.
\item Tìm $m$ để $(d)$ cắt $(P)$ tại hai điểm phân biệt.
\item Với giá trị nào của $m$ thì $(d)$ cắt $(P)$ tại hai điểm phân biệt $M(x_1;y_1)$, $N(x_2;y_2)$ sao cho $y_1+y_2=3(x_1+x_2)$.
\end{enumerate}
	\loigiai
	{\begin{enumerate}
\item Tìm tọa độ giao điểm của $(d)$ và $(P)$ khi $m=1$.\\
Khi $m=1$ ta có $y=x+2$.\\
Phương trình hoành độ giao điểm $x^2=x+2 \Leftrightarrow x^2-x-2=0 \Leftrightarrow x_1=2,x_2=-1$.\\
Với $x_1=2 \Rightarrow y_1=4 \Rightarrow A(2;4)$.\\
Với $x_2=-1 \Rightarrow y_2=1 \Rightarrow B(-1;1)$.
\item Tìm $m$ để $(d)$ cắt $(P)$ tại hai điểm phân biệt.\\
Xét phương trình hoành độ giao điểm của $(d)$ và $(P)$ là $x^2-x+m-3=0 \hfill (*)$.\\
Ta có $\Delta=13-4m$.\\
Để $(d)$ cắt $(P)$ tại hai điểm phân biệt $M,N$ khi phương trình $(*)$ có hai nghiệm phân biệt $\Leftrightarrow \Delta >0 \Leftrightarrow m<\dfrac{13}{4}$.
\item Với giá trị nào của $m$ thì $(d)$ cắt $(P)$ tại hai điểm phân biệt $M(x_1;y_1)$, $N(x_2;y_2)$ sao cho $y_1+y_2=3(x_1+x_2)$.\\
Áp dụng hệ thức Vi-ét ta có 
$\heva{& x_1+x_2=\dfrac{-b}{a}=1\\&x_1x_2=\dfrac{c}{a}=m-3}.$\\
Ta có
\allowdisplaybreaks
 \begin{eqnarray*}
& & y_1+y_2=3(x_1+x_2)\\
&\Leftrightarrow &\left( x_1-m+3\right) + \left( x_2-m+3 \right)=3 \left( x_1+x_2 \right)\\
&\Leftrightarrow & -2m+6=2\\
&\Leftrightarrow & m=2.
\end{eqnarray*}
Vậy $m=2$ là giá trị cần tìm.
\end{enumerate}
}
\end{ex}

Cho $x \geq 0; y \geq 0$ và $x+y=1$. Tìm giá trị lớn nhất của biểu thức $A=\dfrac{x}{y+1}+\dfrac{y}{x+1}$.
\loigiai{ Ta có $A=\dfrac{x^2+y^2+x+y}{(x+1)(y+1)}=\dfrac{(x+y)^2-2xy+x+y}{xy+x+y+1}=\dfrac{2-2xy}{xy+2}$.\\
Vì $x+y \geq 2 \sqrt{xy} \Rightarrow 4xy \leq 1 \Rightarrow 0 \leq xy \leq \frac{1}{4}$. \\
Đặt $xy=t$ thì $0 \leq t \leq \dfrac{1}{4}$.\\
Ta có $A=\dfrac{2-2t}{2+t}=-2+\dfrac{6}{2+t}$\\
Ta có $0 \leq t \leq \dfrac{1}{4} \Rightarrow 2 \leq t+2 \leq \dfrac{9}{4}$.\\
Suy ra $A \leq 1 \Leftrightarrow t+2=2 \Leftrightarrow t=0 \Leftrightarrow \heva{&xy=0\\&x+y=1\\&x,y \geq 0} \Leftrightarrow \hoac{&x=0;y=1\\&x=1;y=0.}$\\
Vậy $A_{\max}=1$ tại $(x;y)=(0;1)$ hoặc $(x;y)=(1;0)$. 
}
\end{ex}
