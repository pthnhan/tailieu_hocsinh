\subsection{Đề số 1a}
\setcounter{ex}{0}
\begin{ex} (2 điểm) %[Phạm Tuấn]%[0D3B2]
Giải phương trình:  $\sqrt{x^2-3x+3} = 2-x.$
\loigiai{
\newline
$PT \iff \heva{& 2-x \geq 0\\&x^2-3x+3 = (2-x)^2}$ \dotfill     (1 điểm)\\
        $\iff \heva{& x \leq 2\\& x=1}$\\
        $\iff   x=1$\dotfill     (1 điểm)\\
 }
\end{ex}

\begin{ex} (2 điểm)  %[Phạm Tuấn]%[0D3B2]
Tìm $m$ sao cho phương trình $(x-1)\left ((m+1)x^2 -2x+1\right )=0$ có ba nghiệm phân biệt.
\loigiai{
\newline
Phương trình có ba nghiệm phân biệt khi và chỉ  khi $\heva{& (m+1) -2+1 \neq 0\\& m+1 \neq 0 \\& \Delta' >0}$ \dotfill  (1 điểm)\\
$ \iff \heva{& m \neq -1 \\&  m<0} $  \dotfill  (1 điểm)\\
}
\end{ex}

\begin{ex} (2 điểm)  %[Phạm Tuấn]%[0D3B2]
Giải phương trình $\sqrt{x^2-5x+4} = |x+1|+1$. 
\loigiai{
\newline
\textbf{TH1:} $x \geq -1$. \\
$PT \iff \heva{& x+2 \geq 0\\&x^2-5x+4 = (x+2)^2}$ \dotfill     (0,5 điểm)\\
$\iff \heva{& x \geq -2\\& x=0} \iff x=0$ \dotfill     (0,5 điểm)\\
\textbf{TH2:} $x <-1$. \\
$PT \iff x^2-5x+4 = x^2$ \dotfill     (0,5 điểm)\\
$\iff  x=\dfrac{4}{5}$ (loại) \dotfill     (0,5 điểm)\\
}
\end{ex}

\begin{ex} (2 điểm)  %[Phạm Tuấn]%[0D3K3]
Tìm $m$ sao cho hệ phương trình sau có nghiệm duy nhất 
\begin{center}
$\heva{& x+2y-z=1 \\&mx -y +2z = 2\\&x+(1-m)y+z=-2}$
\end{center}
\loigiai{
\newline
$\heva{& x+2y-z=1 \\&mx -y +2z = 2\\&x+(1-m)y+z=-2} \iff \heva{& z=x+2y-1 \\&(m+2)x +3y = 4\\&2x+(3-m)y=-1} $. Hệ phương trình có nghiệm duy nhất khi và chỉ khi hệ $\heva{&(m+2)x +3y = 4\\&2x+(3-m)y=-1}$ có nghiệm duy nhất. \dotfill (1 điểm)\\
$\iff \left|\begin{array}{cc}
m+2 & 3 \\ 
2 & 3-m
\end{array} \right| \neq 0 \iff -m^2+m \neq 0 \iff m \neq 0$ và $m\neq 1$. \dotfill(1 điểm)\\
}
\end{ex}

\begin{ex} (2 điểm)  %[Phạm Tuấn]%[0D3K2]
Giải phương trình $x^3-x^2+1=\sqrt{2x-1}$. 
\loigiai{ Điều kiện $x \geq \dfrac{1}{2}$\\
$x^3-x^2+1=\sqrt{2x-1} \iff x^3-x^2-x+1 + x - \sqrt{2x-1} =0 
 \iff  (x-1)^2(x+1) +\dfrac{(x-1)^2}{x+\sqrt{2x-1}}=0 $ $\iff (x-1)^2\left (x+1+\dfrac{1}{x+\sqrt{2x-1}}\right )=0$ \dotfill (1 điểm)\\
Vì $x \geq \dfrac{1}{2}$ nên $ x+1+\dfrac{1}{x+\sqrt{2x-1}} >0$.  Vậy phương trình có nghiệm duy nhất $x=1$ 
\dotfill (1 điểm)\\
}
\end{ex}

\begin{ex} (2 điểm)  %[Phạm Tuấn]%[0D3B3]

\loigiai{
\newline
$\heva{&xy-x-y=1 \\&x^2+y^2=13} \iff \heva{&xy-x-y=1 \\&(x+y)^2-2xy=13} \iff \heva{&xy=x+y+1 \\&(x+y)^2-2(x+y)-15=0} $ \dotfill     (0,5 điểm)\\
$\heva{&xy=6 \\&x+y=5}$ hoặc   $\heva{&xy=-2 \\&x+y=-3}$\dotfill     (0,5 điểm)\\
$\heva{&xy=6 \\&x+y=5} \iff \heva{&xy=6 \\&x+y=5} \iff \heva{&x=2 \\&y=3}$ hoặc $\heva{&x=3 \\&y=2}$ \dotfill     (0,5 điểm)\\
$\heva{&xy=-2 \\&x+y=-3} \iff \heva{&x=\dfrac{-3-\sqrt{17}}{2} \\&y=\dfrac{-3+\sqrt{17}}{2}}$ hoặc $\heva{&x=\dfrac{-3+\sqrt{17}}{2} \\&y=\dfrac{-3-\sqrt{17}}{2}}$\dotfill     (0,5 điểm)\\
}
\end{ex}

\loigiai{

m+1 & 2
\end{array} \right|=-m^2-m+2$, $D_x=\left|\begin{array}{cc}
2 & m \\ 
1 & 2
\end{array} \right|=4-m$, \\  $D_y=\left|\begin{array}{cc}
1 & 2 \\ 
m+1 & 1
\end{array} \right|=-1-2m$\dotfill (0,5 điểm)\\
Hệ có nghiệm khi $D \neq 0$ hoặc $D_x=D_y=D=0$. \dotfill(0,5 điểm)\\
$D \neq 0 \iff -m^2-m+2 \neq 0 \iff m \neq 1$ hoặc $m \neq -2$. \dotfill(0,5 điểm)\\
 $D_x=D_y=D=0$ (không tồn tại $m$). \dotfill(0,5 điểm)\\
}
\end{ex}

\begin{ex} (2 điểm)  %[Phạm Tuấn]%[0D3K2]

\dotfill (1 điểm)\\
}
\end{ex}

\begin{ex}(2 điểm)%[Ân Thọ]%[0D3B2]
Giải phương trình $\dfrac{x}{x-2}-\dfrac{2}{x+1} = 2$.
\loigiai{
ĐK: $x \ne 2, x \ne -1$ \dotfill     (0.5 điểm)\\
$PT \Leftrightarrow x(x+1) - 2(x-2) = 2(x+1)(x-2) \Leftrightarrow x^2 - x - 8 = 0$ \dotfill     (0.5 điểm)\\
$\Leftrightarrow \hoac{&x=\dfrac{1+\sqrt{33}}{2}\\&x=\dfrac{1-\sqrt{33}}{2}}$ \dotfill     (0.5 điểm)\\
Đối chiếu ĐK, PT đã cho có tập nghiệm là $S=\{\dfrac{1\pm \sqrt{33}}{2}\}$. \dotfill     (0.5 điểm)\\
}
\end{ex}

\begin{ex}(2 điểm)%[Ân Thọ]%[0D3B2]
Giải phương trình $|2x-1| = |x+1|$.
\loigiai{
$PT \Leftrightarrow \hoac{&2x-1=x+1\\&2x-1=-x-1}$ \dotfill     (0.5 x 2 điểm)\\
$\Leftrightarrow \hoac{x=2\\x=0}$ \dotfill     (0.5 x 2 điểm)\\
}
\end{ex}  

\begin{ex}(2 điểm)%[Ân Thọ]%[0D3B2]
Giải phương trình:  $\sqrt{3x^2-3x+3} = 1-2x.$
\loigiai{
\newline
$PT \Leftrightarrow \heva{& 1-2x \geq 0\\&3x^2-3x+3 = (1-2x)^2}$ \dotfill     (0.5 điểm)\\
        $\Leftrightarrow \heva{& x \leq \dfrac{1}{2}\\& x^2 -x -2=0}$ \dotfill     (0.5 điểm)\\
        $\Leftrightarrow \heva{& x \leq \dfrac{1}{2}\\& \hoac{&x=2\\&x=-1}}$ \dotfill     (0.5 điểm)\\
        $\Leftrightarrow   x=-1$ \dotfill     (0.5 điểm)\\
 }
\end{ex}

\begin{ex}(2 điểm)%[Ân Thọ]%[0D3K3]
Giải hệ phương trình $\heva{&2x - 3y -12xy = 4 \\&4x^2 + 9y^2 -18xy= 5}$.
\loigiai{
$HPT \Leftrightarrow \heva{S + 2P = 4\\S^2 + P = 5}$, (trong đó $\heva{&S=2x-3y\\&P=2x(-3x)=-6xy}$, và $S^2 \ge 4P$) \dotfill     (0.5 điểm)\\
$\Leftrightarrow \heva{&2S^2 - S - 6 = 0\\&S+2P=4} \Leftrightarrow \heva{S=2\\P=1}$ hoặc $\heva{S=-\dfrac{3}{2}\\P=\dfrac{11}{4}}$ (vô nghiệm)\dotfill     (0.5 điểm)\\
suy ra $2x$ và $-3y$ là các nghiệm của phương trình $t^2 - 2t + 1 = 0$, suy ra $2x = -3y = 1$\dotfill     (0.5 điểm)\\
Vậy hệ phương trình đã cho có nghiệm là $\heva{&x=\dfrac{1}{2}\\&y=-\dfrac{1}{3}}$.\dotfill     (0.5 điểm)\\
}
\end{ex}

\begin{ex}(2 điểm)%[Ân Thọ]%[0D3G2]
Giải phương trình $\sqrt{x^2 + x}+\sqrt{x-2} = \sqrt{3(x^2 - 2x - 2)}$.
\loigiai{
ĐK: $x \ge 1+ \sqrt{3}$.\\
$PT \Leftrightarrow x^2 - 4x -2 = \sqrt{x(x+1)(x-2)}$ \dotfill     (0.5 điểm)\\
$\Leftrightarrow (x^2 - 2x)- \sqrt{(x+1)(x^2 - 2x)} - 2(x+1) = 0$ \\
$\Leftrightarrow \left( \sqrt{x^2 - 2x}+\sqrt{x+1}\right) \left(\sqrt{x^2-2x}-2\sqrt{x+1} \right)=0$ \dotfill     (0.5 điểm)\\
$\Leftrightarrow \sqrt{x^2-2x}-2\sqrt{x+1}=0$ (vì $\sqrt{x^2-2x}+\sqrt{x+1}>0, \forall x \ge 1 + \sqrt{3})$\\
$\Leftrightarrow x^2 -6x - 4 = 0 \Leftrightarrow \hoac{&x=3+\sqrt{13}\\&x=3-\sqrt{13}}$ \dotfill     (0.5 điểm)\\
Đối chiếu ĐK, PT đã cho có tập nghiệm là $S = \{3+\sqrt{13}\}$. \dotfill     (0.5 điểm)
}
\end{ex}
