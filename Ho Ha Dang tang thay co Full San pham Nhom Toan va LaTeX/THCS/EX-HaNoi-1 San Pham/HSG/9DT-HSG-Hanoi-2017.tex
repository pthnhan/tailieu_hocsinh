\begin{ex}%[Đề thi HSG, TP Hà Nội năm 2017]%[9D5B2]%[9D5K3]
    \hfill
    \begin{enumerate}
  \item Chứng minh $n^5+5n^3-6n$ chia hết cho $30$, với mọi số nguyên dương $n$.
\item Tìm tất cả các cặp số nguyên dương $(x;\ y)$ sao cho $x^2+8y$ và $y^2+8x$ là các số chính phương.
    \end{enumerate}
\loigiai{\hfill
\begin{enumerate}
\item Ta có $$x=n^5+5n^3-6n=n(n^4+5n^2-6)=n(n-1)(n+1)(n^2+6).$$
Do $n,\ n-1,\ n+1$ là ba số nguyên liên tiếp nên $n(n-1)(n+1)\vdots 6$, suy ra $x \vdots 6.$\\
Nếu $n \equiv 0 \pmod 5$ hoặc $n\equiv \pm 1 \pmod 5$ thì $n(n-1)(n+1) \equiv 0 \pmod 5$ nên $x \vdots 5$.\\
Nếu $n \equiv \pm 2 \pmod 5$ thì $n^2+6 \equiv 10 \equiv 0 \pmod 5$ nên $x \vdots 5$. Do đó ta có $x \vdots 5$ với mọi $n$.\\
Từ đó ta suy ra được $x \vdots 30$ với mọi số nguyên dương $n$.
\item Không mất tính tổng quát, ta giả sử $x \geq y$. Khi đó $$x^2< x^2+8y \leq x^2+8x <x^2+8x+16=(x+4)^2.$$
Mà $x^2+8y$ là số chính phương nên ta có các trường hợp sau
\begin{itemize}
\item $x^2+8y=(x+1)^2 \Leftrightarrow 8y=2x+1$ (loại).
\item $x^2+8y=(x+2)^2 \Leftrightarrow 2y=x+1$, khi đó 
$$y^2+8x=y^2+16y-8=(y+8)^2-72=a^2 \Leftrightarrow (y+8-a)(y+8+a)=72=3^2.2^3.$$
Do $y+8+a>y+8-a$ và hai số đó cùng tính chẵn lẻ, nên ta có\\  
$\begin{cases} y+a+8=2.3^2 \\y+8-a=2^2  \end{cases} \Rightarrow y=3,x=5. $ hoặc $\begin{cases} y+a+8=2^2.3^2 \\y+8-a=2  \end{cases} \Rightarrow y=11,\ x=21. $ hoặc $\begin{cases} y+a+8=2^2.3 \\y+8-a=2.3  \end{cases} \Rightarrow y=1,\ x=1. $
\item $x^2+8y=(x+3)^2 \Leftrightarrow 8y=6x+9$ (loại).
\end{itemize} 
Vậy các cặp $(x;y)$ cần tìm là: $(1;1),\ (3;5),\ (5;3),\ (11;21),\ (21;11).$
\end{enumerate}
}    
\end{ex}

\begin{ex}%[Đề thi HSG, TP Hà Nội năm 2017]%[9D4K5]%[9D4K5]
\hfill
\begin{enumerate}
\item Giải phương trình $\sqrt{2x-\dfrac{3}{x}}+\sqrt{\dfrac{6}{x}-2x}=1+\dfrac{3}{2x}$.
\item Giải hệ phương trình $\left\{\begin{matrix} \sqrt{\dfrac{4x}{5y}}=\sqrt{x+y}-\sqrt{x-y} & & \\ \sqrt{\dfrac{5y}{x}}=\sqrt{x+y}+\sqrt{x-y} & & \end{matrix}\right.$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Đặt $a=\sqrt{2x-\dfrac{3}{x}},\ b=\sqrt{\dfrac{6}{x}-2x}$, ta có $a,\ b \geq 0$ và $a^2+b^2=\dfrac{3}{x}$. Do đó,ta có 
$$\begin{aligned} &a+b=1+\dfrac{1}{2}\left(a^2+b^2\right)\geq 1+\dfrac{1}{4}(a+b)^2\\ & \Rightarrow (a+b-2)^2 \leq 0 \Rightarrow \begin{cases} a+b-2=0\\a=b \end{cases} \Rightarrow a=b=1. \end{aligned}$$
Từ đây ta có $\dfrac{3}{x}=2\Leftrightarrow x=\dfrac{3}{2}$. Ta thấy $x=\dfrac{3}{2}$ là nghiệm của phương trình đã cho.\\
\textbf{Cách 2:} Áp dụng bất đẳng thức Cô si ta có
$$\begin{cases}\sqrt{2x-\dfrac{3}{x}} \leq \dfrac{1+2x-\dfrac{3}{x}}{2} \\ \sqrt{\dfrac{6}{x}-2x} \leq \dfrac{1+\dfrac{6}{x}-2x}{2}  \end{cases} \Rightarrow \sqrt{2x-\dfrac{3}{x}}+\sqrt{\dfrac{6}{x}-2x}\leq 1+\dfrac{3}{2x}. $$
Đẳng thức xảy ra khi $x=\dfrac{3}{2}$.
\item  Điều kiện $\begin{cases} xy>0 \\ x-y \geq 0 \\x+y \geq 0 \end{cases}$.\\ 
Cộng và trừ theo vế hai phương trình của hệ ta có
$$\begin{cases} \sqrt{\dfrac{4x}{5y}}+\sqrt{\dfrac{5y}{x}}=2\sqrt{x+y} \\ -\sqrt{\dfrac{4x}{5y}}+\sqrt{\dfrac{5y}{x}}=2\sqrt{x-y} \end{cases} \Rightarrow \begin{cases}\dfrac{4x}{5y}+4+\dfrac{5y}{x} =4(x+y) \\\dfrac{4x}{5y}-4+\dfrac{5y}{x} =4(x-y) \end{cases},$$
suy ra $8=8y \Rightarrow y=1.$ Thay vào hệ ta có 
$$\dfrac{4x}{5}+\dfrac{5}{x}=4x \Leftrightarrow x= \dfrac{5}{4}\ (\text{do}\ y>0 \ \text{nên}\ x>0).$$
Vậy nghiệm của hệ là $(x;y)=\left(\dfrac{5}{4};1 \right)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi HSG, TP Hà Nội năm 2017]%[9D5K1]%[9D5G1]
 Với các số thực không âm $x,\ y,\ z$ thỏa mãn $x^2+y^2+z^2=2$ 
 \begin{enumerate}
 \item Chứng minh $x+y+z\leq 2+xy$
\item Tìm GTLN và GTNN của biểu thức $P=\dfrac{x}{2+yz}+\dfrac{y}{2+zx}+\dfrac{z}{2+xy}$.
 \end{enumerate}

\loigiai{ \hfill
\begin{enumerate}
\item Áp dụng bất đẳng thức $a+b \leq \sqrt{2(a^2+b^2)}$ ta có
$$x+y+z \leq \sqrt{2\left[(x+y)^2+z^2 \right]}=\sqrt{2(x^2+y^2+z^2+2xy)}=2\sqrt{1+xy}\leq 1+(1+xy)=2+xy.$$
\item Theo chứng minh trên ta có $2+xy \geq x+y+z$ nên $\dfrac{z}{2+xy} \leq \dfrac{z}{x+y+z}$.\\
Tương tự $\dfrac{x}{2+yz} \leq \dfrac{x}{x+y+z}$, $\dfrac{y}{2+xz} \leq \dfrac{y}{x+y+z}$.\\
Suy ra $P \leq 1$. Đẳng thức xảy ra khi trong ba số $x,\ y,\ z$ có hai số bằng $1$ và một số bằng $0$. Do đó $\max P=1.$\\
Ta có $$2+yz \leq 2+\dfrac{y^2+z^2}{2}=\dfrac{6-x^2}{2},$$ do đó với $x\neq$, ta có $$\dfrac{x}{2+yz} \geq \dfrac{2x}{6-x^2} =\dfrac{2x^2}{x(6-x^2)}=\dfrac{2\sqrt{2}x^2}{\sqrt{2x^2(6-x^2)(6-x^2)}}. $$
Mà $$2x^2(6-x^2)(6-x^2) \leq \left(\dfrac{2x^2+6-x^2+6-x^2}{3} \right)^3=4^3,$$ nên $$\dfrac{x}{2+yz} \geq \dfrac{\sqrt{2}x^2}{4} . \eqno (1)$$
Và ta thấy (1) cũng đúng khi $x=0$, do đó (1) đúng với mọi $x \geq 0.$ Tương tự ta cũng có $$\dfrac{y}{2+zx} \geq \dfrac{\sqrt{2} y^2}{4},\ \dfrac{z}{2+xy} \geq \dfrac{\sqrt{2}z^2}{4}.$$
Do đó $P \geq \dfrac{\sqrt{2}}{4} \left(x^2+y^2+z^2 \right)=\dfrac{\sqrt{2}}{2}.$\\ Đẳng thức xảy ra chẳng hạn khi $x=y=0,\ z=\sqrt{2}$. Vậy $\min P=\dfrac{\sqrt{2}}{2}.$

\end{enumerate}
    }
\end{ex}

     \begin{enumerate}
     \item Chứng minh bốn điểm $A,\ B,\ C,\ P$ cùng thuộc một đường tròn
\item Đường thẳng $H$ song song với $AO$ cắt cạnh $BC$ tại $E$. Gọi $F$ là điểm trên cạnh $BC$ sao cho $CF=BE$. Chứng minh ba điểm $A,\ F,\ O$ thẳng hàng
\item Gọi $N$ là tâm đường tròn ngoại tiếp tam giác $ABM$. Chứng minh rằng $PN=PO$.
         \end{enumerate}
\loigiai{

\begin{enumerate}
\item Ta có $P,\ H$ là trực tâm các tam giác $BHC$ và $ABC$ nên ta có 
$$\widehat{BPC}=180^\circ-\widehat{BMC}=180^\circ-\widehat{BHC}=\widehat{BAC},$$
suy ra tứ giác $BAPC$ nội tiếp.
\item Kẻ đường kính $AK$, ta có $BKCH$ là hình bình hành.\\
Xét hai tam giác $BHE$ và $CKF$ có $$BH=CK,\ BE=CF\ \text{và}\ \widehat{FCK}=\widehat{EBH}$$ nên $\Delta BHE=\Delta CKF$, suy ra $\widehat{BEH}=\widehat{CFK}$. Dẫn tới $HE \parallel KF$\\
Mặt khác $HE \parallel AK $ nên ta có $A,\ K,\ F$ thẳng hàng. Bài toán được chứng minh.
\item Gọi $I$ là tâm đường tròn ngoại tiếp tam giác $BHC$. Vì  $\Delta BHC=\Delta BKC$ nên $IB=IC=OB=OC$.\\
Ta có $ON$ là đường trung trực cạnh $AB$, $OI$ là đường trung trực cạnh $BC$ và $IN$ là đường trung trực cạnh $BM$, nên $\widehat{ONI}=\widehat{ABM}$ và $\widehat{OIN}=\widehat{MBC}$. \\ Mặt khác $\widehat{ABM}=\widehat{MCB}=\dfrac{1}{2}\widehat{ABC}$ nên $\widehat{ONI}=\widehat{OIN}=\dfrac{1}{2}\widehat{ABC}.$ Suy ra $\Delta ONI$ cân tại $O$ và $\widehat{NOI}=180^\circ-\widehat{ABC} $. Lại có 
$$\widehat{POB}=2\widehat{PCB}=2(90^\circ-\widehat{MBC})=180^\circ-2\widehat{MBC}=180^\circ-\widehat{ABC}.$$
Nên ta có $\widehat{NOI}=\widehat{POB}$, suy ra $\widehat{NOP}=\widehat{IOB}$.\\
Từ đó dẫn đến $\Delta OBI=\Delta OPN$, mà $\Delta OBI$ cân tại $B$ nên $PN=PO$.

\end{enumerate}

\begin{ex}%[Đề thi HSG, TP Hà Nội năm 2017]%[9D5G5]
Trên bàn có $100$ tấm thẻ được đánh số từ $1$ đến $100$. Hai người $A$ và $B$ lần lượt mỗi người lấy một tấm thẻ trên bàn sao cho nếu người $A$ lấy tấm thẻ đánh số $n$ thì đảm bảo người $B$ chọn được tấm thẻ đánh số $2n+2$. Hỏi người $A$ có thể lấy được nhiều nhất bao nhiêu tấm thẻ trên bàn thỏa mãn yêu cầu trên?
\loigiai{
 Vì $B$ bốc thẻ $2n + 2$ nên $2n + 2 \leq 100$. Suy ra $n \leq 49$. Do đó, $A$ chỉ được bốc các thẻ đánh từ $1$ đến $49$.\\
Bây giờ, ta chia tập $\{1;\ 2;\ \cdots ;\ 49 \}$ thành $33$ tập con như sau:\\
$ \{ 1;\ 4 \},\ \{ 3;\ 8 \},\ \{ 5;\ 12 \};\cdots,\  \{ 23;\ 49 \}$ ($12$ nhóm);\\
 $\{ 2;\ 6 \},\ \{10;\ 22 \}, \{ 14;\ 30 \},\ \{ 18;\ 38 \}$ ($4$ nhóm);\\
 $\{25 \},\ \{ 27\}, \{ 29 \}, \cdots, \{ 49 \}$ ($13$ nhóm);\\
$ \{ 26\},\ \{ 32 \},\ \{ 42 \},\ \{ 46 \} $($4$ nhóm).\\
Ở mỗi nhóm, $A$ được chọn tối đa một số. Nếu $A$ chọn nhiều hơn $34$ số trong các số từ $1$ đến $49$ thì theo nguyên lý Dirichlet, tồn tại hai số thuộc cùng một nhóm (vô lý). Do đó, $A$ được chọn không quá $33$ số.\\
Mặt khác, $A$ có thể chọn $33 $ số sau :
$$\{ 1,\ 3,\ 5,\ \cdots, 23,\ 2,\ 10,\ 14,\ 18,\ 25,\ 27,\ 29,\ \cdots, 49,\ 26,\ 32,\ 42,\ 46 \}$$
thì thỏa mãn yêu cầu đề bài.\\
Vậy người $A$ có thể lấy nhiều nhất $33$ tấm thẻ trên bàn thỏa mãn yêu cầu trên.



    }
\end{ex}
