		Cho hai tập hợp $ A=\left\{0; 1; 2; 4; 7; 9; 11 \right\} $ và $ B=\left\{-2; -1; 0; 2; 4; 9 \right\} $. Tìm các tập hợp $ A \cup B $; $ A \cap B $; $ A \setminus B $; $ B \setminus A $?
		\loigiai{
			\begin{itemize}
				\item $ A\cup B=\left\{-2; -1; 0; 1; 2; 4; 7; 9; 11 \right\} $.
				\item $ A\cap B=\left\{0; 2; 4; 9 \right\} $.
				\item $ A \setminus B=\left\{ 1; 7; 11 \right\} $.
				\item $ B \setminus A=\left\{ -2; -1 \right\} $.
			\end{itemize}
		}
	\end{bt}
	
	\begin{bt}%[Đề HK1, Khai Minh, Hồ Chí Minh 17-18]%[Dương Xuân Lợi, 10EX-HK1]%[0D3B2-2]%[0D3B2-4]
		Giải các phương trình sau:
		\begin{multicols}{2}
			\begin{enumerate}
				\item $ \left| 2x+3 \right|=\left| 3x+2 \right| $.
				\item $ \sqrt{x+1}-\sqrt{x-1}=1 $.
			\end{enumerate}
		\end{multicols}
		\loigiai{
			\begin{enumerate}
				\item Ta có
				\begin{eqnarray*}
					& &\left| 2x+3 \right|=\left| 3x+2 \right|\\
					& \Leftrightarrow & \hoac{& 2x+3=3x+2 \\ & 2x+3=-3x-2}\\
					& \Leftrightarrow & \hoac{& x=1 \\ & x=-1.}
				\end{eqnarray*}
				Vậy phương trình có tập nghiệm $S=\left\{-1; 1 \right\}$.
				\item  $\sqrt {x + 1} - \sqrt {x - 1} = 1$.\\
				Điều kiện xác định: $\heva{&x + 1 \ge 0\\&x - 1 \ge 0}  \Leftrightarrow \heva{&x \ge - 1\\ &x \ge 1} \Leftrightarrow x \ge 1$.
				\begin{eqnarray*}
					& & \sqrt {x + 1} - \sqrt {x - 1} = 1 \\
					& \Rightarrow & -2 \sqrt {x^2 - 1} = 1 - 2x\\
					& \Rightarrow & 4\left({x^2 - 1} \right) = 1 - 4x + 4x^2\\ 
					& \Leftrightarrow & 4x - 5 = 0\\
					& \Leftrightarrow & x = \dfrac {5}{4}.
				\end{eqnarray*}\\
				Thử lại, ta thấy $ x= \dfrac {5}{4} $ thỏa mãn đầu bài.\\
				Vậy phương trình có tập nghiệm $ S = \left\{ \dfrac {5}{4} \right\} $.
			\end{enumerate}
		}
	\end{bt}

		Giải các hệ phương trình sau:
		\begin{multicols}{2}
			\begin{enumerate}
				\item $ \heva{ & 2x-3y=-4 \\  & 3x+4y=11} $.
				\item $ \heva{ & x+3y-z=8 \\  & 2x-y+4z=-4 \\  & 3x-y+2z=-1} $.
			\end{enumerate}
		\end{multicols}
		\loigiai{
			\begin{enumerate}
				\item Ta có
				$ \heva{& 2x - 3y = - 4\\ &3x + 4y = 11}$ 
				$ \Leftrightarrow \heva{& 8x - 12y = - 16\\ &9x + 12y = 33} $
				$ \Leftrightarrow \heva{&17x = 17\\ &2x - 3y = - 4 } $
				$ \Leftrightarrow \heva{&x = 1\\ &y = 2.} $\\
				Vậy hệ phương trình có nghiệm là: $\left(1; 2 \right)$.
				\item 
				\begin{eqnarray*}
					& & \heva{&x + 3y - z = 8\\&2x - y + 4z = - 4\\&3x - y + 2z = - 1} \\
					& \Leftrightarrow & \heva{&x + 3y - z = 8\\ & -7y + 6z = -20\\ &- 10y + 5z = - 25} \\
					& \Leftrightarrow & \heva{&x + 3y - z = 8\\ & -35y + 30z = - 100\\ &-60y + 30z = - 150} \\ 
					& \Leftrightarrow & \heva{&x + 3y - z = 8\\ &- 35y + 30z = - 100\\ &25y = 50} \\ 
					& \Leftrightarrow & \heva{&x = 1\\&y = 2\\ &z = - 1.} 
				\end{eqnarray*}
				Vậy hệ phương trình có nghiệm là: $\left( 1; 2; -1 \right)$.
			\end{enumerate}
		}
	\end{bt}

	\begin{bt}%[Đề HK1, Khai Minh, Hồ Chí Minh 17-18]%[Dương Xuân Lợi, 10EX-HK1]%[0D3B2-6]
		Giải và biện luận phương trình sau theo tham số $m$: $x^2+2\left( m+1 \right)x+m^2+5=0$.
		\loigiai{
			$x^2+2\left(m+1 \right)x+m^2+5=0$.\\
			Ta có: $ \Delta'=\left(m+1 \right)^2-\left(m^2+5 \right)=2m-4$.
			\begin{itemize}
				\item Nếu $\Delta ' > 0 \Leftrightarrow 2m - 4 > 0 \Leftrightarrow m > 2.$\\
				Phương trình có hai nghiệm phân biệt: $\hoac{&x = - \left( m + 1 \right) - \sqrt {2m - 4}\\&x = - \left( m + 1 \right) + \sqrt {2m - 4}.}$
				\item Nếu $\Delta ' = 0 \Leftrightarrow 2m - 4 = 0 \Leftrightarrow m = 2$. \\
				Phương trình có nghiệm kép: $x = - 3.$
				\item Nếu $\Delta ' < 0 \Leftrightarrow 2m - 4 < 0 \Leftrightarrow m < 2.$\\
				Phương trình vô nghiệm.
			\end{itemize}
			Kết luận:
			\begin{itemize}
				\item $m > 2$ phương trình có hai nghiệm phân biệt: $\hoac{&x = - \left( m + 1 \right) - \sqrt {2m - 4}\\&x = - \left( m + 1 \right) + \sqrt {2m - 4}.}$
				\item  $m = 2$ phương trình có nghiệm kép: $x = - 3$.
				\item  $m < 2$ phương trình vô nghiệm.
			\end{itemize}
		}
	\end{bt}
	
	\begin{bt}%[Đề HK1, Khai Minh, Hồ Chí Minh 17-18]%[Dương Xuân Lợi, 10EX-HK1]%[0D3K3-5]
		Một tam giác vuông có độ dài cạnh dài nhất lớn hơn độ dài cạnh thứ hai là $ 2 $m, độ dài cạnh thứ hai lớn hơn độ dài cạnh ngắn nhất là $ 23 $m. Tính diện tích của tam giác vuông đó?
		\loigiai{
			Gọi $ x $(m) là độ dài cạnh dài nhất của tam giác vuông, điều kiện $ \left({x > 25} \right). $\\
			Độ dài cạnh thứ hai của tam giác vuông là: $ x - 2 $ (m).\\
			Độ dài cạnh ngắn nhất của tam giác vuông là: $ x - 25 $ (m).\\
			Áp dụng định lý Pytago ta có:
			\begin{eqnarray*}
				& & {x^2} = \left( x - 2 \right)^2 + \left( x - 25 \right)^2 \\
				& \Leftrightarrow & x^2 - 54x + 629 = 0\\ 
				& \Leftrightarrow & \hoac{&x = 37 \text{ (thỏa mãn)}\\&x = 17\text{ (loại)}.}
			\end{eqnarray*}
			Diện tích tam giác vuông là:
			$S = \dfrac {1}{2} \left( x - 2 \right)\left( x - 25 \right) = \dfrac {1}{2} \left(37 - 2 \right)\left(37 - 25 \right) = 210 \left(\text{m}^2 \right)$.
		}
	\end{bt}
	
	\begin{bt}%[Đề HK1, Khai Minh, Hồ Chí Minh 17-18]%[Dương Xuân Lợi, 10EX-HK1]%[0H2Y2-1]%[0H2B2-1]%[0H1B4-4]
		Trong mặt phẳng tọa độ $ Oxy $, cho $ \overrightarrow{a}(-3;4) $, $ \overrightarrow{b}\left( 8;-6 \right) $, $ \overrightarrow{c}\left( 18;-10 \right) $.
		\begin{enumerate}
			\item Tính các tích vô hướng: $\overrightarrow{a} \cdot \overrightarrow{b} $ và $\overrightarrow{a} \cdot \overrightarrow{c} $?
			\item Tính giá trị biểu thức: $ S=\left( \overrightarrow{a} + \overrightarrow{b} \right)^2+\left( \overrightarrow{a} - \overrightarrow{b} \right)^2 $?
			\item Hãy phân tích vec-tơ $ \overrightarrow{c} $ theo hai vectơ  $ \overrightarrow{a} $ và  $ \overrightarrow{b} $.
		\end{enumerate}
		\loigiai{
			\begin{enumerate}
				\item $\overrightarrow{a} \cdot \overrightarrow{b}= \left(-3\right)\cdot 8 + 4 \cdot \left(-6\right)=-48.$\\
				$\overrightarrow{a} \cdot \overrightarrow{c}=\left(-3\right) \cdot 18 + 4 \cdot \left(-10\right)=94$.
				\item Ta có: $\left| {\overrightarrow a} \right| = 5$ và $\left| {\overrightarrow b} \right| = 10$.\\
				$S = \left(\overrightarrow a + \overrightarrow b \right)^2 + \left(\overrightarrow a - \overrightarrow b \right)^2 = 2\left| \overrightarrow a \right| + 2\left| \overrightarrow b \right| = 250$.\\
				Vậy $S = 250$.
				\item Gọi $ h $ và $ k $ là hai số thực sao cho: 
				\begin{eqnarray*} 
					& & \overrightarrow c = h \cdot \overrightarrow  a + k \cdot \overrightarrow  b\\
					& \Leftrightarrow & \heva{& -3h + 8k = 18\\ & 4h - 6k = - 10}\\
					& \Leftrightarrow & \heva{&h = 2\\&k = 3.}
				\end{eqnarray*}
				Vậy $\overrightarrow c = 2\overrightarrow a + 3\overrightarrow b$.			
			\end{enumerate}
		}
	\end{bt}

		Trong mặt phẳng tọa độ $ Oxy $, cho $ \triangle ABC $ với $A(2; 4) $, $ B-2; 1) $, $ C(4; -2)$.
		\begin{enumerate}
			\item Tính chu vi $ \triangle ABC $? (kết quả làm tròn đến số thập phân thứ nhất).
			\item Tìm tọa độ trọng tâm $ G $ và trực tâm $ H $ của $ \triangle ABC $?
		\end{enumerate}
		\loigiai{
			\begin{enumerate}
				\item $\overrightarrow {AB} = \left(-4; -3 \right) $, $AB = \left| \overrightarrow {AB} \right| = \sqrt {\left( -4 \right)^2 + \left( -3 \right)^2} = 5$.\\
				$\overrightarrow {AC} = \left( 2; -6 \right)$, $AC = \left| \overrightarrow {AC} \right| = \sqrt {2^2 + \left(- 6 \right)^2} = 2\sqrt {10}$.\\
				$\overrightarrow {BC} = \left( 6; -3 \right)$, $BC = \left| \overrightarrow {BC} \right| = \sqrt {6^2 + \left( -3 \right)^2} = 3\sqrt 5$.\\
				Chu vi tam giác $ ABC $ là: $ AB +AC + BC =18 $.
				\item 
				\begin{itemize}
					\item	Gọi $G\left(x_G; y_G \right)$ là trọng tâm của tam giác $ABC$.\\
					$\heva{&x_G = \dfrac {x_A + x_B + x_C} {3} = \dfrac {2 - 2 + 4}{3} = \dfrac {4}{3}\\
						& y_G = \dfrac {y_A + y_B + y_C} {3} = \dfrac {4 - 6 - 10}{3} = - 4.}$\\
					Vậy $G\left(\dfrac {4}{3}; - 4 \right)$.
					\item Gọi $H\left(x_H; y_H \right)$ là trực tâm của tam giác $ ABC $.\\
					$ \overrightarrow {AH} = (x_H - 2; y_H - 4)$, $ \overrightarrow {BH} = (x_H + 2; y_H - 1) $. \\
					Vì $H$ là trực tâm của tam giác $ ABC $ nên
					\begin{eqnarray*}
						& &\heva{&AH \perp BC\\&BH \perp AC}\\
						& \Leftrightarrow & \heva{&\overrightarrow {AH} \perp \overrightarrow {BC} \\ &\overrightarrow {BH} \perp \overrightarrow {AC}} \\
						& \Leftrightarrow & \heva{&\overrightarrow {AH} \cdot \overrightarrow {BC} = 0\\ &\overrightarrow {BH} \cdot \overrightarrow {AC} = 0}\\
						& \Leftrightarrow & \heva{&6\left(x_H - 2 \right) - 3\left(y_H - 4 \right) = 0\\&2\left(x_H + 2 \right) - 6\left(y_H - 1 \right) = 0} \\
						& \Leftrightarrow & \heva{&6x_H - 3y_H = 0\\ &2x_H - 6y_H = - 10} \\
						& \Leftrightarrow & \heva{&x_H = 1\\&y_H = 2.}
					\end{eqnarray*}
					Vậy $H\left( 1; 2 \right)$.
				\end{itemize}
			\end{enumerate}
		}
	\end{bt}
