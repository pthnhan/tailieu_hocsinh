\section{BẤT ĐẲNG THỨC VÀ CỰC TRỊ DẠNG CĂN THỨC}

\begin{center}
 \large \textbf{TỔNG QUAN VỀ CHUYÊN ĐỀ}
\end{center}
Chuyên đề này bao gồm hai nội dung:
\begin{enumerate}[--]
	\item Bất đẳng thức dạng căn thức.
	\item Cực trị dạng căn thức.
\end{enumerate}
Các bài toán về cực trị dạng căn thức thường khó, do đó ngoài các dạng đơn giản, trong chuyên đề đã phân loại các dạng sau:
\begin{enumerate}[--]
	\item Dạng $\sqrt{f^2(x)}+\sqrt{g^2(x)}$ trong đó $f(x)+g(x)=m. $
	\item Dạng $\sqrt{f(x)}-\sqrt{g(x)}$ trong đó $f(x)-g(x) =m >0.$
	\item Dạng $a\sqrt{f(x)}+b\sqrt{g(x)}$ trong đó $f(x)+g(x) =m.$
	\item Dạng $f(x)+\sqrt{g(x)}.$
	\item Dạng $f(x)+x\sqrt{g(x)}.$
	\item Dạng $\sqrt{f(x).g(x)}+\sqrt{h(x).k(x)}$ trong đó $f(x)+h(x)$ và $g(x)+k(x)$ đều là hằng số.
	\item Dạng $\sqrt{f(x).g(x)}-\sqrt{h(x).k(x)}$ trong đó $f(x)-h(x)$ và $g(x)-k(x)$ đều là hằng số.
	\item Dạng $\sqrt{f^2+g^2}+\sqrt{h^2+k^2}$ trong đó $f+h$ và $g+k$ đều là hằng số.	
	\item Dạng $\sqrt{f^2+g^2}-\sqrt{h^2+k^2}$ trong đó $f-h$ và $g-k$ đều là hằng số.
	\item Dạng chứa phân thức.
\end{enumerate}
\begin{center}
	\textbf{CHỨNG MINH BẤT ĐẲNG THỨC BẰNG HÌNH HỌC}
\end{center}
Xuân đố Mai chứng minh bất đẳng thức sau bằng hình học:
\begin{align*}
	\sqrt{a-1}+\sqrt{b-1} \leq \sqrt{ab}.
\end{align*}
Mai chưa tìm ra cách làm. Bạn hãy giúp cho Mai.\\

\subsection{Bất đẳng thức dạng căn thức}

\begin{vd}
    Chứng minh bất đẳng thức \[\sqrt{a-1}+\sqrt{b-1} \leq \sqrt{ab}.\tag{1}\]
    \loigiai{
        \begin{enumerate}[Cách 1.]
        	\item (đặt ẩn phụ và biến đổi tương)\\
        		Đặt $a-1=x,\b-1=y.$ Ta có\\
        		$\begin{aligned}
        			(1) &\Leftrightarrow \sqrt{x}+\sqrt{y} \leq \sqrt{(x+1)(y+1)}\\
        				&\Leftrightarrow x+y+2 \sqrt{xy} \leq xy + x + y +1\\
        				&\Leftrightarrow 2\sqrt{xy} \leq xy +1\\
        				&\Leftrightarrow (\sqrt{xy}-1)^2 \geq 0,\ \text{đúng. Vậy } (1)\ \text{đúng.}
        		\end{aligned} $ \\
        		Xảy ra đẳng thức khi và chỉ khi $xy=1 \Leftrightarrow a+b = ab.$
        	\item (dùng bất đẳng thức Cô-si)\\
        	$(1) \Leftrightarrow \sqrt{\dfrac{a-1}{ab}}+\sqrt{\dfrac{b-1}{ab}} \leq 1.$ Gọi vế trái là $A$ ta có\\
        	$2A=2 \sqrt{\dfrac{a-1}{a}.\dfrac{1}{b}}+2 \sqrt{\dfrac{b-1}{b}.\dfrac{1}{a}} \leq \left(\dfrac{a-1}{a}+\dfrac{1}{b} \right) + \left(\dfrac{b-1}{b}+\dfrac{1}{a} \right)=2$\\
        	nên $A\leq 1.$ Vậy $(1)$ đúng.
        	\item (dùng bất đẳng thức Bu-nhi-a-cốp-xki)\\
        	$(\sqrt{a-1}+\sqrt{b-1})^2=(\sqrt{a-1}.1+1.\sqrt{b-1})^2 \leq (a-1+1)(1+b-1)=ab $\\
        	$\Rightarrow \sqrt{a-1}+\sqrt{b-1}\leq \sqrt{ab.} $
        	\item (dùng hình học)\\
        	Xem cách giải ở phần \textit{Tổng quan về chuyên đề.}
        \end{enumerate}
    }
\end{vd}

\begin{vd}
	Cho các số dương $a,\ b,\ c$ thỏa mãn $ab+bc+ca=1.$ Chứng minh bất đẳng thức \[\sqrt{1+a^2} +\sqrt{1+b^2}+\sqrt{1+c^2} \leq 2(a+b+c).\]
	\loigiai{
		Ta có $1+a^2 =ab+bc+ca+a^2=(a+b)(a+c)$ nên\\
		$\sqrt{1+a^2}=\sqrt{(a+b)(a+c)}\leq \dfrac{a+b+a+c}{2}=a+\dfrac{b+c}{2}.$\\
		Từ đó\\
		$\sqrt{1+a^2} +\sqrt{1+b^2}+\sqrt{1+c^2}\leq \left(a+\dfrac{b+c}{2}\right)+ \left(b+\dfrac{c+a}{2}\right)+\left(c+\dfrac{a+b}{2}\right)=2(a+b+c).$\\
		Xãy ra đẳng thức khi và chỉ khi $a=b=c=\dfrac{1}{\sqrt{3}}.$
	}
\end{vd}
 
 \begin{vd}
 	Chứng minh bất đẳng thức \[\dfrac{(x+y)^2}{2}+\dfrac{x+y}{4} \geq x \sqrt{y}+y \sqrt{x}. \]
 	\loigiai{
 		Ta có $\dfrac{(x+y)^2}{2}+\dfrac{x+y}{4}=\dfrac{x+y}{2}\left(xy+\dfrac{1}{2}\right) \geq \sqrt{xy}\left(x+y+\dfrac{1}{2} \right). \hfill (1)$\\
 		Ta sẽ chứng minh $\sqrt{xy}\left(x+y+\dfrac{1}{2} \right) \geq x \sqrt{y} + y \sqrt{x}. \hfill (2)$\\
 		$(2)\Leftrightarrow \sqrt{xy}\left(x+y+\dfrac{1}{2}-\sqrt{x}-\sqrt{y} \right) \geq 0 \Leftrightarrow \sqrt{xy}\left[\left(\sqrt{x}-\dfrac{1}{2} \right)^2 +\left(\sqrt{y}-\dfrac{1}{2} \right)^2\right] \geq 0,$ đúng.\\
 		Từ $(1)$ và $(2)$ suy ra điều phải chứng minh.\\
 		Xảy ra đẳng thức khi và chỉ khi $x=y=0$ hoặc $x=y=\dfrac{1}{4}.$
 	}
 \end{vd}

 \begin{vd}
 	Chứng minh bất đẳng thức \[ \dfrac{a}{b+c} + \dfrac{b}{c+a}+\sqrt{\dfrac{2c}{a+b}} \geq 2 \ \text{với}\ a,\ b,\ c >0. \]
 	\loigiai{
 		Gọi vế trái là $A.$ Ta có
 		\[\sqrt{\dfrac{2c}{a+b}}=\dfrac{2c}{\sqrt{2c(a+b)}} \geq \dfrac{2c}{\dfrac{2c+a+b}{2}}=\dfrac{4c}{2c+a+b}. \tag{1} \]
 		Áp dụng bất đẳng thức $\dfrac{1}{x}+\dfrac{1}{y} \geq \dfrac{4}{x+y}$ với $x,\ y >0$ ta có\\
 		\[ \left(\dfrac{a}{b+c}+1 \right)+\left(\dfrac{b}{c+a}+1\right)=(a+b+c)\left(\dfrac{1}{b+c} +\dfrac{1}{c+a}\right) \geq \dfrac{4(a+b+c)}{2c+a+b}. \tag{2}\]
 		Từ $(1)$ và $(2)$ suy ra \[A+2 \geq \dfrac{4(a+b+c)+4c}{2c+a+b+c}=4 \Rightarrow A \geq 2. \]
 		Xảy ra đẳng thức khi và chỉ khi $a=b=c.$
 	}
 \end{vd}
\subsection{Cực trị dạng căn thức}
 	\begin{enumerate}[1.]
 		\item \textbf{Dạng $\mathbf{\sqrt{f^2(x)}+\sqrt{g^2(x)}}$ trong đó $\mathbf{f(x)+g(x) =m}$}\\
 		Để tìm giá trị nhỏ nhất của dạng trên, có hai cách:
 		\begin{enumerate}[--]
 			\item Đưa về bất đẳng thức $|f(x)|+|g(x)|\geq f(x) +g(x)=m.$
 			\item Xét bình phương của biểu thức.
 		\end{enumerate}
 		\begin{vd}
 			Tìm giá trị nhỏ nhất của \[A=\sqrt{x+\sqrt{2x-1}}+\sqrt{x- \sqrt{2x-1}}. \]
 			\loigiai{
 				ĐKXĐ : $x\geq \dfrac{1}{2}.$
 				\begin{enumerate}[Cách 1.]
 					\item Đặt $2x-1 =y.$ Ta có \\
 					$\begin{aligned}
 						A \sqrt{2} &= \sqrt{2x+2 \sqrt{2x-1}}+\sqrt{2x-2 \sqrt{2x-1}}\\
 								   &= \sqrt{y+1+2 \sqrt{y}}+\sqrt{y+1-2 \sqrt{y}}\\
 								   &=\sqrt{(\sqrt{y}+1)^2}+\sqrt{(\sqrt{y}-1)^2}\\
 								   &=\sqrt{y}+1+|1- \sqrt{y}|\geq \sqrt{y} +1 +1- \sqrt{y}=2
 					\end{aligned}$ \\
 					$\Rightarrow A \geq \sqrt{2} \Rightarrow \min A = \sqrt{2} \Leftrightarrow y=1 \Leftrightarrow x=1.$
 					\item Ta có\\
 					 $\begin{aligned}
 						A^2 &=x+\sqrt{2x-1}+x- \sqrt{2x-1}+2 \sqrt{x^2-(2x-1)}\\
 							&=2x+2|1-x| \geq 2x+2(1-x)=2.
 					\end{aligned} $\\
 					$\Rightarrow \min A = \sqrt{2} \Leftrightarrow x=1.$
 				\end{enumerate}
 			}
 		\end{vd}
 		\item \textbf{Dạng $\mathbf{\sqrt{f(x)}-\sqrt{g(x)}}$ trong đó $\mathbf{f(x)-g(x)=m>0}$ }\\
 		Để tìm giá trị lớn nhất của dạng trên, ta dùng bất đẳng thức \[ \sqrt{f(x)}-\sqrt{g(x)} \leq \sqrt{f(x)-g(x)}=\sqrt{m}.\]
 		Xảy ra đẳng thức khi và chỉ khi $g(x)=0$ hoặc $f(x)=g(x).$
 		\begin{vd}
 			Tìm giá trị lớn nhất của biểu thức \[A=\sqrt{x^2-x+3}-\sqrt{x^2-x-6} \]
 			\loigiai{
 				ĐKXĐ: $x \leq -2$ hoặc $x \geq 3.$\\
 				Áp dụng bất đẳng thức $\sqrt{a}-\sqrt{b} \leq \sqrt{a-b}$ ta có \[A \leq \sqrt{(x^2-x+3)-(x^2-x-6)}=\sqrt{9}=3.\]
 				Vậy $\max A = 3 \Leftrightarrow \hoac{&x^2-x-6=0 \\ &x^2-x+3 =x^2-x-6} \Leftrightarrow \hoac{&x=3\\&x=-2.}$
 			}
 		\end{vd}
 		\item \textbf{Dạng $\mathbf{a \sqrt{f}+b \sqrt{g}}$ trong đó $\mathbf{f(x)+g(x)=m}$}\\
 		Để tìm giá trị lớn nhất của dạng trên, ta dùng bất đẳng thức Bu-nhi-a-cốp-xki \[(a \sqrt{f}+b \sqrt{g})^2 \leq (a^2+b^2)(f+g) =(a^2 +b^2)m. \]
 		Trong một số trường hợp, có thể xét $(a \sqrt{f}+b \sqrt{g}){2}$ rồi dùng bất đẳng thức Cô-si.\\
 		Trong một số trường hợp, có thể tìm được giá trị nhỏ nhất của dạng  trên bằng cách xét bình phương của biểu thức.\\
 		Với dạng $\sqrt{f(x)}+\sqrt{g(x)}$ trong đó $f(x)+g(x)=m,$ ta các thể tìm được giá trị nhỏ nhất bằng cách xét bình phương của biểu thức hoặc dùng bất đẳng thức \[\sqrt{f(x)}+\sqrt{g(x)} \geq \sqrt{f(x)+g(x)}=\sqrt{m}.\]
 		\begin{vd}
 			Cho biểu thức $A=3 \sqrt{x}+\sqrt{10-x}.$
 			\begin{enumerate}[a)]
 				\item Tìm giá trị nhỏ nhất của $A.$
 				\item Tìm giá trị lớn nhất của $A.$
 			\end{enumerate}
 			\loigiai{
 				ĐKXĐ: $0 \leq x \leq 10.$
 				\begin{enumerate}[a)]
 					\item $A^2=9x+(10-x)+6 \sqrt{x(10-x)}=8x+10+6 \sqrt{x(10-x)} \geq 10 $ (do $x \geq 0$)\\
 					Do đó $\min A = \sqrt{10} \Leftrightarrow \heva{&x=0\\&x(10-x)=0} \Leftrightarrow x=0.$
 					\item Theo bất đẳng thức Bu-nhi-a-cốp-xki ta có
 					\[A^2=(3 \sqrt{x}+1.\sqrt{10-x})^2 \leq (3^2+1^2)(x+10-x)=100. \]
 					Vậy $\max A = 10 \Leftrightarrow \heva{&9=x \qquad (1) \\&1=10-x  \qquad (2)} \Leftrightarrow x=9.$\\
 					(Ta gặp may mãn vì $x=9$ ở $(1)$ thỏa mãn $(2)$).
 				\end{enumerate}
 			}
 		\end{vd}
 		\begin{vd}
 			Tìm giá trị lớn nhất của \[A = 3 \sqrt{x}+ 2 \sqrt{1-4x}.\]
 			\loigiai{
 				ĐKXĐ: $0 \leq x \leq \dfrac{1}{4}.$\\
 				$2A=6 \sqrt{x}+4 \sqrt{1-4x}=3 \sqrt{4x}+4 \sqrt{1-4x}.$\\
 				Theo bất đẳng thức Bu-nhi-a-cốp-xki ta có 
 				\[(2A)^2 =(3 \sqrt{4x}+4 \sqrt{1-4x})^2 \leq (3^2+4^2)(4x+1-4x)=25\]
 				$\Rightarrow 2A \leq 5 \Rightarrow A \leq \dfrac{5}{2}.$\\
 				Vậy $\max A = \dfrac{5}{2} \Leftrightarrow \dfrac{\sqrt{4x}}{3}=\dfrac{\sqrt{1-4x}}{4}\Leftrightarrow x=\dfrac{9}{100}.$
 				\textit{Lưu ý.} Cũng có thể viết $A$ dưới dạng $A=3 \sqrt{x}+ 4 \sqrt{\dfrac{1}{4}-x}$ rồi dùng bất đẳng thức Bu-nhi-a-cốp-xki được 
 				\[ A^2 \leq (3^2+4^2)\left(x+\dfrac{1}{4}-x \right)=\dfrac{25}{4} \Rightarrow A \leq \dfrac{5}{2}.\]
 			}
 		\end{vd}

 		\begin{vd}
 			Cho các số dương $a,\ b,\ c$ thỏa mãn $a+b+c=3.$ Tìm giá trị lớn nhất của:
 			\begin{enumerate}[a)]
 				\item $A=\sqrt{2-a}+\sqrt{2-b}+\sqrt{2-c};$
 				\item $B=\sqrt{5-a}+\sqrt{5-b}+\sqrt{5-c}.$
 			\end{enumerate}
 			\loigiai{
 				\begin{enumerate}[a)]
 					\item Áp dụng bất đẳng thức Cô-si 
 						\[\sqrt{2-a} =\sqrt{1.(2-a)} \leq \dfrac{1+(2-a)}{2}=\dfrac{3-a}{2}.\]
 						Từ đó 
 						\[2A \leq (3-a)+(3-b)+(3-c) = 9 -(a+b+c)=9-3=6 \Rightarrow A \leq 3. \]
 						Vậy $\max A = 3 \Leftrightarrow \heva{&1=2-a\\&1=2-b\\&1=2-c} \Leftrightarrow a=b=c=1.$
 						\item Áp dụng bất đẳng thức Cô-si
 						\[2 \sqrt{5-a}=\sqrt{4(5-a)} \leq \dfrac{4+5-a}{2} =\dfrac{9-a}{2}\]
 						Từ đó
 						\[4B = (9-a)+(9-b)+(9-c)=27-(a+b+c)=27-3=24 \Rightarrow B \leq 6.\]
 						Vậy $\max B = 6 \Leftrightarrow \heva{&4=5-a\\&4=5-b\\&4=5-c} \Leftrightarrow a=b=c=1.$\\
 						\textit{Lưu ý.} Ở câu a) ta viết $2\sqrt{2-a}=\sqrt{1.(2-a)}$, biểu thức trong dấu căn nhân với $k_1.$\\
 						Ở câu b) ta viết $2 \sqrt{5-a}=\sqrt{4(5-a)},$ biểu thức trong dấu căn nhân với $k_2=4.$\\
 						Có sự khác nhau nói trên vì ở bất đẳng thức $\sqrt{xy} \leq \dfrac{x+y}{2},$ đẳng thức xảy ra tại $x=y,$ nên ta phải chọn $k_1 $ và $k_2 $ sao cho $k_1=2-a $ và $k_2=5-a. $\\
 						Ở cả hai câu a và b do vai trò bình đẳng của $a,\ b,\ c$ ta dự đoán cực trị xảy ra khi $a = b = c =1,$ do đó $k_1=2-a = 2-1 = 1 $ và $k_2=5-a = 5 -1 =4. $
 				\end{enumerate}
 			}
 		\end{vd}
	\item \textbf{Dạng $\mathbf{f(x)-\sqrt{g(x)}}$}\\
	Để tìm giá trị lớn nhất của dạng trên, tùy từng trường hợp mà ta sử dụng bất đẳng thức Cô-si hay Bu-nhi-a-cốp-xki. Cần chú ý những dạng cơ bản sau:
	\begin{enumerate}[a)]
	\item Dạng $A=mx+\sqrt{ax+b}$ trong đó $m+\dfrac{a}{2}=0$. Dùng bất đẳng thức Cô-si ta có\\
	$\sqrt{1.(ax+b)}\leq \dfrac{1+b+ax}{2} \Rightarrow A \leq mx+\dfrac{1+b}{2}+\dfrac{ax}{2}=\dfrac{1+b}{2}$.
	\item Dạng $B=mx+n\sqrt{a-x^2}$, Dùng bất đẳng thức Bu-nhi-a-cốp-ski ta có\\
	$B^2=\left(mx+n\sqrt{a-x^2}\right)^2\leq(m^2+n^2)(x^2+a-x^2)=a(m^2+n^2).$
	\end{enumerate}
\begin{vd}
	Tìm giá trị lớn nhất của $A=x+\sqrt{1-2x}.$
	\loigiai{
	ĐKXĐ: $x\leq \dfrac{1}{2}$.\\
	\textit{Cách 1.} Áp dụng bất dẳng thức Cô-si: $A=x+\sqrt{1.(1-2x)}\leq x+\dfrac{1+(1-2x)}{2}=1$\\
	$\max A=1 \Leftrightarrow 1=1-2x \Leftrightarrow x=0.$\\
	\textit{Cách 2.} Đặt $\sqrt{1-2x}=y$ thì $x=\dfrac{1-y^2}{2}$. Ta có\\
	$A=\dfrac{1-y^2}{2}+y \Rightarrow 2A=1-y^2+2y=2-(y-1)^2\leq2.$\\
	$\max A=1 \Leftrightarrow y=1 \Leftrightarrow x=0.$
	}
\end{vd}
\begin{vd}
	Tìm giá trị lớn nhất của $A=x+\sqrt{-x^2-2x+3}.$
	\loigiai{
	ĐKXĐ; $-2\leq x \leq 1.$\\
	\textit{Cách 1.} $A=x+\sqrt{4-(x+1)^2}$. Đặt $x+1=y$ ta có $A=-1+y+\sqrt{4-y^2}$.\\
	Áp dụng bất đẳng thức Bu-nhi-a-cốp-xki ta có\\
	$\left(1.y+1.\sqrt{4-y^2}\right)^2\leq(1^2+1^2)(y^2+4-y^2)=8
	\Rightarrow y+\sqrt{4-y^2} \leq \sqrt{8}$\\ 
	$\Rightarrow A \leq -1+\sqrt{8} =2\sqrt{2}-1.$\\
	$\max A=2\sqrt{2}-1 \Leftrightarrow y=\sqrt{4-y^2} \Leftrightarrow y=\sqrt{2}\Leftrightarrow x=\sqrt{2}-1.$\\
	\textit{Cách 2.} Gọi $a$ là giá trị biểu thức của $A$. Ta có $a=x+\sqrt{-x^2-2x+3}$\\
	$\Leftrightarrow a-x=\sqrt{-x^2-2x+3}.$ \hfill$(1)$ \\
	Với $a\geq x$ thì $(1) \Leftrightarrow a^2+x^2-2ax=-x^2-2x+3 \Leftrightarrow 2x^2-2(a-1)x+(a^2-3)=0.$\\
	Để tồn tại $x$, phải có $\Delta'\geq0 \Rightarrow a^2+2a-7 \leq 0$
	$\Rightarrow (a+1)^2\leq 8 \Rightarrow a \leq 2\sqrt{2}-1$.\\
	$\max A=2\sqrt{2}-1 \Leftrightarrow x=\sqrt{2}-1$,khi đó $a=2\sqrt{2}-1$, thỏa mãn $x\leq a$.  
}
\end{vd}
\begin{vd}
	Tìm giá trị lớn nhất của $A=x+\sqrt{1-2x-2x^2}.$
	\loigiai{
	ĐKXĐ: $1-2x-2x^2\geq 0$.\\
	Có thể giải hai cách như ở Ví dụ $139$. Cách giải dùng bất đẳng thức Bu-nhi-a-cốp-xki như sau:\\

\subsection*{Bất đẳng thức dạng căn thức}

\begin{bt}
    Chứng minh các bất đẳng thức:
    \begin{enumerate}
    \item[a)] $\sqrt{a^2+b^2}+\sqrt{c^2+d^2} \geq \sqrt{(a-c)^2+(b-d)^2}$;
    \item[b)] $x\sqrt{y-1}+y\sqrt{x-1} \geq xy$;
    \item[c)] $\dfrac{x^2+y^2}{xy}+\dfrac{\sqrt{xy}}{x+y} \geq 2,5$ với $x,\ y>0$;
    \item[d)] $\dfrac{1+a+b+c}{2} \geq \sqrt{1+\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}}$ với $a,\ b,\ c >0$ và $abc=1$;
    \item[e)] $\dfrac{a}{\sqrt{1-a^2}}+\dfrac{b}{\sqrt{1-b^2}}+\dfrac{c}{\sqrt{1-c^2}}>2$ với $a,\ b,\ c>0$ và $a^2+b^2+c^2=1$.
    \end{enumerate}
    \loigiai{
     \begin{enumerate}
     \item[a)] Bình phương hai vế của bất đẳng thức, ta được bất đẳng thức tương đương
     $$a^2+b^2+c^2+d^2+2\sqrt{(a^2+b^2)(c^2+d^2)} \geq (a-c)^2+(b-d)^2.$$
     Tiếp tục khai triển vế phải và rút gọn, thu được
     \begin{align*}
     \sqrt{(a^2+b^2)(c^2+d^2)} \geq -ac-bd.\tag{*}
     \end{align*}
     Sử dụng bất đẳng thức Bu-nhi-a-cốp-xki cho hai dãy số $-a,\ -b$ và $c,\ d$, ta có
     $$\left(a^2+b^2\right)\left(c^2+d^2\right) \geq \left(-ac-bd\right)^2\Leftrightarrow \sqrt{\left(a^2+b^2\right)\left(c^2+d^2\right)} \geq \left|-ac-bd\right| \geq -ac-bd.$$
     Vậy bất đẳng thức $(*)$ đúng. Từ đó bất đẳng thức đã cho được chứng minh xong.
     \item[b)] Điều kiện $x\geq 1,\ y\geq 1$.\\
     Với điều kiện trên, ta có
     $$\sqrt{y-1}=\sqrt{1\cdot (y-1)} \leq \dfrac{1+y-1}{2}=\dfrac{y}{2}\Rightarrow x\sqrt{y-1} \leq \dfrac{xy}{2}.$$
     Tương tự $$y\sqrt{x-1} \leq \dfrac{xy}{2}$$
     Cộng theo vế hai bất đẳng thức trên lại, ta được
     $$x\sqrt{y-1}+y\sqrt{x-1} \leq xy$$
     Xảy ra đẳng thức khi và chỉ khi $x=y=2$.
     \item[c)] Biến đổi tương đương
     \begin{align*}
     & \dfrac{x^2+y^2}{xy}+\dfrac{\sqrt{xy}}{x+y} \geq 2,5\\
     \Leftrightarrow &\ \left(\dfrac{x^2+y^2}{xy}-2\right)+\left(\dfrac{\sqrt{xy}}{x+y}-\dfrac{1}{2}\right) \geq 0\\
     \Leftrightarrow &\ \dfrac{(x-y)^2}{xy}-\dfrac{\left(\sqrt{x}-\sqrt{y}\right)^2}{2(x+y)} \geq 0\\
     \Leftrightarrow &\ \left(\sqrt{x}-\sqrt{y}\right)^2\left[\dfrac{\left(\sqrt{x}+\sqrt{y}\right)^2}{xy}-\dfrac{1}{2(x+y)}\right] \geq 0. \tag{1}
     \end{align*}
     Gọi $A$ là biểu thức trong dấu ngoặc vuông. Ta có
     $$\dfrac{\left(\sqrt{x}+\sqrt{y}\right)^2}{xy} \geq \dfrac{4\sqrt{xy}}{xy}=\dfrac{4}{\sqrt{xy}}=\dfrac{8}{2\sqrt{xy}} \geq \dfrac{8}{x+y}$$
     nên
     $$A \geq \dfrac{8}{x+y}-\dfrac{1}{2(x+y)}=\dfrac{15}{2(x+y)}>0.$$
     Vậy $(1)$ đúng nên bất đẳng thức đã cho được chứng minh. Xảy ra dấu đẳng thức khi và chỉ khi $x=y$.
     \item[d)] Do $abc=1$ nên
     \begin{align*}
     &\ \dfrac{1+a+b+c}{2} \geq \sqrt{1+\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}}\\
     \Leftrightarrow &\ \dfrac{1+a+b+c}{2} \geq \sqrt{1+\dfrac{bc+ca+ab}{abc}}\\ 
     \Leftrightarrow &\ \dfrac{1+a+b+c}{2} \geq \sqrt{1+ab+bc+ca}.
     \end{align*}
     Ta có 
     $$\dfrac{(1+a)+(b+c)}{2} \geq \sqrt{(1+a)(b+c)}=\sqrt{b+c+ab+ac}$$
     Do đó chỉ cần chứng minh $b+c \geq 1+bc$.\\
     Do $abc=1$ nên trong ba số $a,\ b,\ c$ tồn tại một số lớn hơn hoặc bằng $1$ và một số nhỏ hơn hoặc bằng $1$, chẳng hạn hai số đó là $b$ và $c$. Khi đó $(b-1)(c-1) \leq 0\Leftrightarrow b+c \geq 1+bc$.     
     \item[e)] Từ giả thiết suy ra $0<a,\ b,\ c<1$.\\
     Ta có
     $$2a\sqrt{1-a^2}\leq a^2+(1-a^2)=1\Rightarrow \dfrac{1}{\sqrt{1-a^2}}\geq 2a\Rightarrow \dfrac{a}{\sqrt{1-a^2}}\geq 2a^2.$$
     Do đó
     $$\dfrac{a}{\sqrt{1-a^2}}+\dfrac{b}{\sqrt{1-b^2}}+\dfrac{c}{\sqrt{1-c^2}}\geq 2(a^2+b^2+c^2)=2.$$
     Xảy ra đẳng thức khi và chỉ khi $a=b=c=\dfrac{1}{\sqrt{2}}$, trái với $a^2+b^2+c^2=1$.\\
     Vậy $\dfrac{a}{\sqrt{1-a^2}}+\dfrac{b}{\sqrt{1-b^2}}+\dfrac{c}{\sqrt{1-c^2}} > 2$.
\end{enumerate}            
    }
\end{bt}
