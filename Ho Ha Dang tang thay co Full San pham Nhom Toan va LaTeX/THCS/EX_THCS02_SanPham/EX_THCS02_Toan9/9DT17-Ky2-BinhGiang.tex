\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2015-2016, Bình Giang]%[9D4Y5]%[9D3Y4]
	Giải phương trình và hệ phương trình sau:
	\begin{itemize}
	\item[1)] $x^2-10x+16=0$
	\item[2)] $\begin{cases} x+2y=-3\\ 2x-y=4 \end{cases}$
	\end{itemize}
	\loigiai{
		\begin{itemize}
	\item[1)] Ta có $x^2-10x+16=0$\\
	$\Rightarrow \bigtriangleup'=3$\\
	Vậy phương trình có hai nghiệm $x_1=8,x_2=2$.
	\item[2)] 
	\begin{align*}
	&\begin{cases} x+2y=-3\\ 2x-y=4 \end{cases}\\
	\Leftrightarrow &\begin{cases} x=-3-2y\\ 2x-y=4 \end{cases}\\
	\Leftrightarrow &\begin{cases} x=-3-2y\\ 2(-3-2y)-y=4 \end{cases}\\
	\Leftrightarrow &\begin{cases} x=1\\ y=-2 \end{cases} 
	\end{align*}
	\end{itemize}
	}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2015-2016, Bình Giang]%[9D4K6]
	Cho phương trình bậc hai: $x^2-8x+m+2=0 $ (*)
	\begin{itemize}
	\item[1)] Tìm $m$ để phương trình (*) có nghiệm kép, tính nghiệm kép đó.
	\item[2)] Tìm $m$ để phương trình có hai nghiệm $x_1,x_2$ thỏa mãn $x_1 - 2x_2=2$.
	\end{itemize}
	\loigiai{
	Ta có $\delta'=16-(m+2)=14-m$.
	\begin{itemize}
	\item[1)] Để phương trình (*) có nghiệm kép thì 
	$\delta'=0$\\
	$\delta'=0 \Leftrightarrow m =14$.\\
	Vậy với $m=14$ thì phương trình (*) có nghiệm kép $x_1=x_2=4$
	\item[2)] Để phương trình (*) có hai nghiệm $x_1,x_2$ thì $\delta' \geq 0 \Rightarrow m \leq 14$.\\
	Theo hệ thức Vi-ét và kết hợp với giả thiết, ta có:
	\begin{align*}
	\begin{cases} x_1-2x_2 &=2\\ x_1+x_2& =8\\ x_1x_2&=m+2 \end{cases}
	\end{align*}
	$\Rightarrow m+2 =12\Rightarrow m= 10 \text{ (kết hợp với điều kiện)}$\\
	Vậy $m=10$ thì phương trình (*) có hai nghiệm thỏa mãn $x_1-2x_2=2$.
	\end{itemize}
	}
\end{ex}

\begin{ex} %[Đề thi Toán 9 Học kỳ 2 năm học 2015-2016, Bình Giang]%[9D2B5]%[9D3B6]
	\begin{itemize}
	\item[1)] Cho đường thẳng $y=2x+m$ cắt đồ thị hàm số $y=x^2$ tại điểm có hoành độ bằng $-3$. Tìm $m$?.
	\item[2)] Hai ô tô khởi hành cùng một lúc đi từ A đến B. Vận tốc ô tô thứ nhất lớn hơn vận tốc ô tô thứ hai là 10 km/h nên đến B sớm hơn ô tô thứ hai một giờ. Tính vận tốc của mỗi ô tô, biết A và B cách nhau 300 km.
	\end{itemize}	 
	\loigiai{
	\begin{itemize}
	\item[1)] Phương trình hoành độ giao điểm của hai đồ thị hàm số là: $x^2=2x+m$\\
	Vì 2 đồ thị hàm số cắt nhau tại điểm có hoành độ bằng -3 nên $(-3)^2=2\cdot (-3)+m$\\
	$\Rightarrow m=15$
	\item[2)] Gọi vận tốc của ô tô thứ nhất là $x$ (km/h) $x\geq0$\\
	$\Rightarrow$ vận tốc của ô tô thứ hai là $x-10$ (km/h).\\
	$\Rightarrow$ thời gian của ô tô thứ nhất đến B là $\dfrac{300}{x}$ (h).\\
	$\Rightarrow$ thời gian ô tô thứ hai đến B là $\dfrac{300}{x-10}$ (h)\\
	Theo đề bài, ta có:
	$\dfrac{300}{x-10}-\dfrac{300}{x}=1$\\
	$\Rightarrow x = 60$.\\
	Vậy vận tốc của ô tô thứ nhất là 60 (km/h).\\
	vận tốc của ô tô thứ hai là 50 (km/h)	
	
	\end{itemize}
	}
\end{ex}

\begin{ex} %[Đề thi Toán 9 Học kỳ 2 năm học 2015-2016, Bình Giang]%[2H3K8]
Từ điểm A ở ngoài đường tròn (O) kẻ hai tiếp tuyến AB, AC với đường tròn (O) với B, C là hai tiếp điểm. Kẻ cát tuyến ADE với đường tròn (O) $(AD<AE)$. \\
Chứng minh rằng
	\begin{itemize}
	\item[1)] Tứ giác ABOC nội tiếp.
	\item[2)] $AB^2=AD \cdot AE$.
	\item[3)] $BD \cdot CE=CD \cdot BE$.
	\end{itemize}	 
		\loigiai
	{
		\immini
		{
	1) Ta có: $\widehat{OBA}=\widehat{OCA}=90^\circ$ (tính chất tiếp tuyến)\\
	$\Rightarrow \widehat{OBA}+\widehat{OCA}= 180^\circ$\\
	$\Rightarrow$ Tứ giác ABOC nội tiếp đường tròn.\\
	2) Xét $\triangle ABD$ và $\triangle AEB$ có:
	\begin{align*}
	& \widehat{BAE} \text{ góc chung}\\
	& \widehat{ABD}=\widehat{AEB}=\dfrac{1}{2} \cdot \text{sđ} \stackrel\frown{AB}
	\end{align*}
		$\Rightarrow \triangle ABD \backsim  \triangle AEB$\\
		$\Leftrightarrow \dfrac{AB}{AE}=\dfrac{AD}{AB}$ hay $AB^2=AD \cdot AE$ (đpcm).\\
		}{
	\begin{tikzpicture}[]
\tkzDefPoint(0,0){O}
\tkzDefPoint(70:3.8){A}
\tkzDrawCircle[R](O,2 cm)
\tkzTangent[from with R=A](O,2 cm)
\tkzGetPoints{B}{C}
\tkzDrawPoints(O,A,B,C)
\tkzLabelPoints[right](A,C)
\tkzLabelPoints[below](O)
\tkzLabelPoints[left](B)
\tkzDrawSegments[](O,B O,C C,A A,B)
\tkzMarkRightAngles(O,B,A A,C,O)
\tkzDefPoint(80:2){D}
\tkzInterLC[R](A,D)(O,2 cm)
\tkzGetPoints{E}{D}
\tkzDrawPoints(D,E)
\tkzLabelPoints[below](D)
\tkzLabelPoints[left](E)
\tkzDrawSegments[](A,D D,E B,D C,E B,E D,C)
\tkzFillAngle[draw=black,size=.6](D,B,A)
\tkzFillAngle[draw=black,size=.6](A,E,B)
\tkzMarkAngle[mark=|,size=.8](A,C,D)
\tkzMarkAngle[mark=|,size=.8](C,E,A)
\end{tikzpicture}

	}
	3) Ta có: $\triangle ABD \backsim  \triangle AEB$ (cmt)\\
	$\Rightarrow \dfrac{AB}{AE}=\dfrac{BD}{BE}$ (1)\\
	Xét $\triangle ACD$ và $\triangle AEC$ có
	\begin{align*}
	& \widehat{EAC} \text{ góc chung}\\
	& \widehat{ACD}=\widehat{AEC}=\dfrac{1}{2} \cdot \text{sđ}\stackrel\frown{DC}
	\end{align*}
	$\Rightarrow \triangle ACD \backsim  \triangle AEC$\\
	$\Rightarrow \dfrac{AC}{AE}=\dfrac{CD}{CE}$ \\
	Mà $AB = AC$ (tính chất hai tiếp tuyến cắt nhau)\\
	Nên $\dfrac{AB}{AE}=\dfrac{CD}{EC}$ (2)\\
	Từ (1) và (2) suy ra $\dfrac{BD}{BE}=\dfrac{CD}{CE}$\\
	Hay $BD\cdot CE=CD \cdot BE$ (đpcm)	
	}
\end{ex}

\begin{ex} %[Đề thi Toán 9 Học kỳ 2 năm học 2015-2016, Bình Giang]%[9D5G1]
Cho $x, y$ thỏa mãn $x+3y=5$. Tìm giá trị nhỏ nhất của biểu thức:
$$A=x^2+y^2+2x+16y$$	 
	\loigiai{
	Từ $x+3y=5$ ta có $x=5-3y$.\\
	Do đó 
	\begin{align*}
	A & = (5-3y)^2+y^2+2(5-3y)+16 \\
	&=25-30y+9y^2+y^2+10-6y+16y\\
	&=35+10y^2-20y\\
	&=10(y-1)^2+25 \geq 25
	\end{align*}

	Vậy $\min A = 2$ đạt được khi $x=2,y=1$.
	}
\end{ex}
