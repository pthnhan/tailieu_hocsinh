\begin{ex}%[Đề thi HSG 9, tỉnh Tuyên Quang, năm 2018]%[Trần Chiến, Dự án EX-9G-18]%[9D0B1]
	Cho $x=\dfrac{\sqrt{3+\sqrt{5}}+\sqrt{3-\sqrt{5}}}{\sqrt{2}}$. Tính $P=\left(1+5x^{2015}-x^{2017}\right)^{2018}$.
	\loigiai{
		Ta có $x=\dfrac{\sqrt{6+2\sqrt{5}}+\sqrt{6-2\sqrt{5}}}{2}=\dfrac{\sqrt{(\sqrt{5}+1)^2}+\sqrt{(\sqrt{5}-1)^2}}{2}=\dfrac{\sqrt{5}+1+\sqrt{5}-1}{2}=\sqrt{5}.$\\
		Khi đó, $P=\left(1+5(\sqrt{5})^{2015}-(\sqrt{5})^{2017}\right)^{2018}=\left(1+(\sqrt{5})^{2017}-(\sqrt{5})^{2017}\right)^{2018}=1.$
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Tuyên Quang, năm 2018]%[Trần Chiến, Dự án EX-9G-18]%[9D0G2]
	Giải phương trình và hệ phương trình sau:
	\begin{enumerate}
	\item $\sqrt{4x^2+5x+1}-2\sqrt{x^2-x+1}=9x-3$.
	\item $\heva{& \sqrt{x+y+1}+1=(x+y)^2+\sqrt{2x+2y}\\
	& x^2-xy=3.}$
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item Điều kiện: $\heva{& 4x^2+5x+1\geq 0 \\ & x^2-x+1\geq 0}\Leftrightarrow \heva{& x\leq -1\ \text{hoặc}\ x\geq-\dfrac{1}{4}\\ & x \in\mathbb{R}} \Leftrightarrow x\leq -1\ \text{hoặc}\ x\geq-\dfrac{1}{4}.$\\
	Ta có $\sqrt{4x^2+5x+1}-2\sqrt{x^2-x+1}=9x-3\Leftrightarrow \sqrt{4x^2+5x+1}-\sqrt{4x^2-4x+4}=9x-3$.\\
	Đặt $a=\sqrt{4x^2+5x+1}$, $b=\sqrt{4x^2-4x+4}$, ta có
	$$\heva{& a-b=9x-3 \\ & a^2-b^2=9x-3} \Rightarrow a-b=(a-b)(a+b) \Leftrightarrow \hoac{& a-b=0 \\ & a+b=1}\Leftrightarrow \hoac{& 9x-3=0\\ & a+b=1}\Leftrightarrow \hoac{& x=\dfrac{1}{3}\\ & a+b=1.}$$
	Ta chú ý rằng $b=\sqrt{4x^2-4x+4}=\sqrt{(2x-1)^2+3}\geq \sqrt{3}$ do đó $a+b\geq \sqrt{3}>1$.\\
	Vậy phương trình chỉ có nghiệm $x=\dfrac{1}{3}$.
	
	\item Điều kiện $\heva{& x+y+1\geq 0\\ & 2x+2y\geq 0}\Rightarrow x+y\geq 0$.\\
	Ta xét phương trình thứ nhất của hệ:
	\allowdisplaybreaks{
	\begin{eqnarray*}
		&& \sqrt{x+y+1}+1=(x+y)^2+\sqrt{2x+2y}\\
		&\Leftrightarrow& \sqrt{x+y+1}-\sqrt{2x+2y}=(x+y)^2-1\\
		&\Leftrightarrow& \dfrac{x+y+1-(2x+2y)}{\sqrt{x+y+1}+\sqrt{2x+2y}}=(x+y-1)(x+y+1)\\
		&\Leftrightarrow& \dfrac{1-x-y}{\sqrt{x+y+1}+\sqrt{2x+2y}}=(x+y-1)(x+y+1)\\
		&\Leftrightarrow& (x+y-1)\underbrace{\left(x+y+1+\dfrac{1}{\sqrt{x+y+1}+\sqrt{2x+2y}}\right)}_{>0}=0 \\
		&\Leftrightarrow& x+y-1=0\\
		&\Leftrightarrow& y=1-x.
	\end{eqnarray*}}
	Thay vào phương trình thứ hai của hệ ta có $$ x^2-xy=3 \Leftrightarrow x^2-x(1-x)=3 \Leftrightarrow 2x^2-x-3=0\Leftrightarrow x=\dfrac{3}{2}\ \text{hoặc}\ x=-1.$$
	Với $x=\dfrac{3}{2}$ thì $y=-\dfrac{1}{2}$, với $x=-1$ thì $y=2$.\\
	Vậy hệ phương trình có hai nghiệm là $\left(\dfrac{3}{2};-\dfrac{1}{2}\right)$ và $(-1;2)$.
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Tuyên Quang, năm 2018]%[Trần Chiến, Dự án EX-9G-18]%[9D0G5]%[9D0K4]
	\hfill
	\begin{enumerate}
	\item Chứng minh rằng $A=n^8+4n^7+6n^6+4n^5+n^4$ chia hết cho $16$ với mọi $n$ là số nguyên.
	\item  Tìm nghiệm nguyên dương của phương trình $x^2+y^2+z^2+xyz=20$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item Ta có $A=n^4(n^4+4n^3+6n^2+4n+1)=n^4(n+1)^4=[n(n+1)]^4$.\\
	Vì $n$ và $n+1$ là hai số nguyên liên liên nên có một số là số chẵn, do đó $n(n+1)$ chia hết cho $2$. Suy ra $[n(n+1)]^4$ chia hết cho $16$. \\
	Vậy $A$ chia hết cho $16$ với mọi $n$ là số nguyên.
	\item Do vai trò của $x,y,z$ như nhau nên không mất tính tổng quát, ta giả sử $x\geq y\geq z>0$.\\
	Ta có $3z^2<x^2+y^2+z^2+xyz=20\Rightarrow z^2<\dfrac{20}{3}$. Mà $z$ nguyên dương nên $z\in\{1;2\}$.\\
	Với $z=1$ phương trình ban đầu trở thành
	\begin{eqnarray*}
		&& x^2+y^2+xy=19\\
		&\Rightarrow & 3y^2<x^2+y^2+xy=19\\
		&\Rightarrow & y\in\{1;2\}.
	\end{eqnarray*} 
	Với $y=1$ thay vào ta có $x^2+1+x=19\Leftrightarrow x^2+x-18=0\Leftrightarrow x=\dfrac{-1\pm\sqrt{73}}{2}$ (loại).\\
	Với $y=2$ thay vào ta có $x^2+4+2x=19\Leftrightarrow x^2+2x-15=0\Leftrightarrow \hoac{& x=-5\ \text{(loại)}\\ &x=3}$.\\
	Với $z=2$ phương trình ban đầu trở thành
	\allowdisplaybreaks{
	\begin{eqnarray*}
		&& x^2+y^2+2xy=16\\
		& \Leftrightarrow & (x+y)^2=16\\
		& \Rightarrow & x+y=4\\
		& \Rightarrow & 2y\leq x+y=4\\
		& \Rightarrow & y\in\{1;2\}.
	\end{eqnarray*}}
	Mà $y\geq z=2$ nên ta nhận $y=2$. Từ đó suy ra $x=2$.\\
	Vậy các nghiệm nguyên dương của phương trình là $(2;2;2)$, $(3;2;1)$ và các hoán vị của nó.
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Tuyên Quang, năm 2018]%[Trần Chiến, Dự án EX-9G-18]%[9H0K1]
	Từ một điểm $E$ ở ngoài đường tròn $(O)$ vẽ hai tiếp tuyến $EB, ED$ với đường tròn ($B, D$ là tiếp điểm) và một cát tuyến qua $E$ cắt đường tròn tại hai điểm phân biệt $A,C$.
	\begin{enumerate}
	\item Chứng minh hai tam giác $EAD$ và $EDC$ đồng dạng.
	\item Chứng minh $AD\cdot BC= AB\cdot DC$.
	\item Gọi $d$ là đường thẳng qua $B$ và song song với $ED$, $d$ cắt $DA,DC$ lần lượt tại $M$ và $N$. Chứng minh $BM=BN$.
	\end{enumerate}
	\loigiai{
	\begin{center}
		\begin{tikzpicture}[scale=0.9]
		\tkzDefPoints{0/0/A,1.5/2/B,1.5/-3.5/D}
		\tkzCircumCenter(A,B,D)\tkzGetPoint{O}
		\tkzDrawCircle(O,A)
		\tkzTangent[from=B](O,A) \tkzGetSecondPoint{b}
		\tkzTangent[from=D](O,A) \tkzGetSecondPoint{d}
		\tkzInterLL(B,b)(D,d)\tkzGetPoint{E}
		\tkzInterLC(E,A)(O,A) \tkzGetSecondPoint{C}
		\tkzDefLine[parallel=through B](E,D)\tkzGetPoint{X}
		\tkzInterLL(B,X)(A,D)\tkzGetPoint{M}
		\tkzInterLL(B,X)(C,D)\tkzGetPoint{N}
		\tkzInterLC(B,X)(O,A)\tkzGetSecondPoint{P}
		\tkzInterLC(D,O)(O,A)\tkzGetSecondPoint{G}
		\tkzInterLL(D,G)(M,N)\tkzGetPoint{F}
		\tkzDrawPolygon(E,B,C,D)
		\tkzDrawSegments(M,D M,P D,G E,C O,B O,D B,D B,A)
		\tkzMarkRightAngles(E,D,O E,B,O)
		\tkzDrawPoints(O,A,B,C,D,E,P,G,M,N,F)
		\tkzLabelPoints[left](E)
		\tkzLabelPoints[above](B,M)
		\tkzLabelPoints[below](D,N,F)
		\tkzLabelPoints[right](P,O)
		\tkzLabelPoints[below left](A)
		\tkzLabelPoints[above right](C,G)
		\end{tikzpicture}
	\end{center}
	\begin{enumerate}
	\item Xét $\Delta EAD$ và $\Delta EDC$ có\\
	$\widehat{CED}$ chung\\
	$\widehat{EDA}=\widehat{ECD}$ (tính chất của góc tạo bởi tiếp tuyến và dây cung)\\
	Suy ra $\Delta EAD \backsim \Delta EDC$.

	\item Từ $\Delta EAD \backsim \Delta EDC$ suy ra $\dfrac{EA}{ED}=\dfrac{AD}{DC}$. Chứng minh tương tự ta có $\dfrac{EA}{EB}=\dfrac{AB}{BC}$.\\
	Mà $EB=ED$ nên ta có $\dfrac{AD}{DC}=\dfrac{AB}{BC}$, suy ra $AD\cdot BC= AB\cdot DC$.
	
	\item Ta có $\widehat{DMB}=\widehat{ADE}$ (so le trong).\\
	Mà $\widehat{ADE}=\widehat{DCA}$ (câu a), $\widehat{DCA}=\widehat{DBA}$ (cùng chắn $\wideparen{AD}$). Do đó, $\widehat{DMB}=\widehat{DBA}$. \\
	Từ đó, $\Delta DMB \backsim \Delta DBA$. Suy ra $\dfrac{BA}{MB}=\dfrac{DA}{DB}\Rightarrow \dfrac{DB}{MB}=\dfrac{AD}{AB}$. \hfill (1)\\
	Kéo dài $MN$ cắt $(O)$ tại $P$. Vẽ đường kính $DG$, $DG$ cắt $MN$ tại $F$.\\
	Vì $OD\perp ED$, $MN\parallel ED$ nên $OD\perp MN$.\\
	Suy ra $OD$ đi qua trung điểm $F$ của $BP$. Suy ra $G$ là điểm chính giữa cung $\wideparen{BP}$.\\
	Mặt khác, $DG$ là đường kính của $(O)$ nên $\wideparen{DB}=\wideparen{DP}$. \\
	Suy ra $\widehat{BCD}=\dfrac{1}{2}\text{sđ}\wideparen{DB}=\dfrac{1}{2}\text{sđ}\wideparen{DP}=\widehat{DBN}$.\\
	Từ đó, $\Delta DBC \backsim \Delta DNB$. Suy ra $\dfrac{DC}{BC}=\dfrac{DB}{NB}$.\hfill (2)\\
	Mặt khác, từ câu b) ta có $AD\cdot BC= AB\cdot DC\Leftrightarrow \dfrac{AD}{AB}=\dfrac{DC}{BC}$. \hfill (3)\\
	Từ (1), (2) và (3) suy ra $MB=NB$.
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Tuyên Quang, năm 2018]%[Trần Chiến, Dự án EX-9G-18]%[9D0K7]
	Cho số thực $x$ thỏa mãn $2\leq x\leq 3$. Tìm giá trị lớn nhất và giá trị nhỏ nhất của biểu thức $$M=\dfrac{5+x}{x}+\dfrac{10-x}{5-x}.$$
	\loigiai{
	Vì $2\leq x\leq 3$ nên $(x-2)(3-x)\geq 0\Leftrightarrow 5x-x^2\geq 6$.\\
	Ta có $M=\dfrac{5}{x}+\dfrac{5}{5-x}+2=\dfrac{25}{x(5-x)}+2\leq \dfrac{25}{6}+2=\dfrac{37}{6}$.\\
	Dấu bằng xảy ra khi và chỉ khi $x=2$ hoặc $x=3$.\\
	Lại có $M=\dfrac{5}{x}+\dfrac{5}{5-x}+2\geq 5\cdot \dfrac{4}{x+(5-x)}+2=5\cdot\dfrac{4}{5}+2=6$.\\
	Dấu bằng xảy ra khi và chỉ khi $\heva{& 2\leq x\leq 3\\ & \dfrac{1}{x}=\dfrac{1}{5-x}}\Leftrightarrow x=\dfrac{5}{2}$.\\
	Vậy GTLN của $M$ là $\dfrac{37}{6}$ khi $x=2$ hoặc $x=3$; GTNN của $M$ là $6$ khi $x=\dfrac{5}{2}$.
	}
\end{ex}
