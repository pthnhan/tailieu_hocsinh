\noindent\textbf{II. PHẦN TỰ LUẬN}
\setcounter{ex}{0}
\begin{ex}%[8D1B6]%[8D1G6]
Phân tích các đa thức sau thành nhân tử:
\begin{enumerate}
\item $ x^3-9x $
\item $ 2x^2-5x-7 $
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $ x^3-9x = x (x-3)(x+3)$
\item $ 2x^2-5x-7 =(2x-7)(x+1) $
\end{enumerate}
}
\end{ex}
\begin{ex}%[8D1K6]%[8D1G7]
Tìm $ x $ biết:
\begin{enumerate}
\item $ 3x(2x-5)-4(5-2x)=0 $
\item $ (2x+3)^2-(5x-2)^2=0 $
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $ \displaystyle 3x(2x-5)-4(5-2x)=0  \Leftrightarrow (2x-5)(3x+4)=0 \Leftrightarrow x = \dfrac{5}{2} \text{ hoặc } x= \dfrac{-4}{3} $
\item $\displaystyle (2x+3)^2-(5x-2)^2=0 \Leftrightarrow (2x+3+5x-2)(2x+3-5x+2)=0$\\
$\Leftrightarrow (7x+1)(-3x+5)=0 \Leftrightarrow x=\dfrac{-1}{7} \text{ hoặc } x=\dfrac{5}{3} $
\end{enumerate}
}
\end{ex}
\begin{ex}%[8H1K7]%[8H1K9]
Cho hình bình hành $ ABCD $ có cạnh $ AD = a $ và $ AB=2a $. Gọi $ M, N $ lần lượt là trung điểm của $ AB $ và $ CD $.
\begin{enumerate}
\item Chứng minh $ \Delta ADN $ cân và $ AN $ là phân giác của $ \widehat{BAD} $
\item Chứng minh rằng $ MD \parallel NB $
\item Gọi $ P $ là giao điểm của $ AN $ với $ DM $, $ Q $ là giao điểm của $ CM $ với $ BN $. Chứng minh tứ giác $ PMQN $ là hình chữ nhật
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1]
\clip (-3,-5) rectangle (8,1);
\tkzDefPoints{0/0/A, 7/0/B, -2/-3/D}
\tkzDefPointBy[translation = from A to B](D)
\tkzGetPoint{C}
\tkzDefMidPoint(A,B)
\tkzGetPoint{M}
\tkzDefMidPoint(C,D)
\tkzGetPoint{N}
\tkzInterLL(A,N)(M,D)
\tkzGetPoint{P}
\tkzInterLL(N,B)(C,M)
\tkzGetPoint{Q}
\tkzDrawPoints(A,B,C,D,M,N,P,Q)
\tkzLabelPoints[below](C,D,N)
\tkzLabelPoints[above](P,Q)
\tkzLabelPoints[above](A,B,M)
\tkzDrawSegments(A,B B,C C,D D,A D,M M,C A,N N,B)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item $ DN = \dfrac{1}{2} DC  $. Mà $ DC = AB $, nên $ DN = a $. Trong $ \Delta ADN $, $ DN = AD $.\\
Vậy $ \Delta ADN $ cân tại $ D $.\\
Ta có $ \Delta ADN = \Delta AMN (c-c-c) $, nên $ \widehat{DAN}=\widehat{NAM} $, do đó AN là phân giác của góc $ \widehat{BAD} $
\item Xét tứ giác $ DMBN $ là hình bình hành vì có các cặp cạnh đối song song và bằng nhau ($ MB=DN =a, MB \parallel DN $ ).\\
Do đó  $ MD \parallel NB  $.
\item Xét tứ giác $PMQN$ có:\\
$MD \parallel BN$ nên $MP \parallel NQ$.\\
Tương tự ta có $NP \parallel MQ$.\\
Suy ra tứ giác $MNPQ$ là hình bình hành.\\
Ta lại có $AN$ là phân giác góc $BAD$ nên $AP$ là phân giác của góc $DAM$.\\
Mà $\triangle ADM$ cân tại $A$ nên $AP \perp DN$.\\
Do đó tứ giác $ PMQN $ là hình chữ nhật.
\end{enumerate}
}
\end{ex}
\begin{ex}%[8D1G12]
Tìm các số thực $ a,b $ để đa thức $ f(x)=x^4-3x^3+ax+b $ chia hết cho đa thức $ g(x)=x^2-3x-4 $.
\loigiai{Lấy hai đa thức chia với nhau được $x^4-3x^3+ax+b = (x^2-3x-4)(x^2+4)+(a+12)x+b-16 $.\\
Để đa thức $f(x)$ chia hết cho $g(x)$ thì $(a+12)x+b-16 =0  \Leftrightarrow \heva{&a = -16 \\\ &b=16}$  }
\end{ex}
