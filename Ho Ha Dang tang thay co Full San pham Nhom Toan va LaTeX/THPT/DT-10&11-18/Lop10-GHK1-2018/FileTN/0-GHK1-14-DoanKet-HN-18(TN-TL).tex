\begin{bt}
	Cho $\mathrm{C}_{\mathbb{R}}A=(-\infty;5)\cup[ơ7;+\infty)$ và $\mathrm{C}_{\mathbb{R}}B=[6;9]$. Xác định tập $X=A\cap B$.
	\loigiai{
	$\mathrm{C}_{\mathbb{R}}A=(-\infty;5)\cup[ơ7;+\infty)\Rightarrow A=[5;7)$.\\
	$\mathrm{C}_{\mathbb{R}}B=[6;9]\Rightarrow B=(-\infty;6)\cup(9;+\infty)$.\\
	Vậy $X=A\cap B=[5;6)$.}
\end{bt}

\begin{bt}%[Thi GHK 1, THPT Đoàn Kết - Hà Nội, 2018]%[Huỳnh Quy, EX-DT-10&11]%[0H1B2-5]
	Cho tam giác $ABC$ đều có độ dài cạnh bằng $2a$. Dựng và tính độ dài véc-tơ $\vec{AB}-\vec{CA}$.
	\loigiai{\immini{
	Dựng hình bình hành $ABDC$, ta có:\\ $\left|\vec{AB}-\vec{CA}\right|=\left|\vec{AB}+\vec{AC}\right|=\left|\vec{AD}\right|=AD$.\\
	Vì $ABDC$ là hình bình hành và $\triangle ABC$ là tam giác đều nên $ABDC$ là hình thoi.}{\begin{tikzpicture}[line cap=round,line join=round,>=stealth,x=1cm,y=1cm]
	\tkzDefPoints{0/0/A, 3/0/B}
	\tkzDefPointBy[rotation=center A angle 60](B)\tkzGetPoint{C}
	\coordinate (D) at ($(B)+(C)-(A)$);
	\coordinate (H) at ($(B)!0.5!(C)$);
	\tkzDrawPolygon(A,B,D,C)
	%\tkzFillPolygon[color=yellow, fill opacity=0.5](A,B,D,C)
	\tkzDrawSegments(A,D B,C)
	\tkzDrawPoints[fill=black](A,B,C,D,H)
	%\tkzLabelPoints[above](A)
	\tkzLabelPoints[below](H)
	\tkzLabelPoints[left](A,C)
	\tkzLabelPoints[right](B,D)
	\end{tikzpicture}}
	Gọi $H$ là giao điểm của $AD$ và $BC$ thì ta có $AD=2AH$.\\
	Xét tam giác $ABH$ vuông tại $H$ có $\widehat{ABH}=60^{\circ}$, $AB=2a$ nên ta có\\ 
	$AH=AB\cdot\sin 60^{\circ}=2a\cdot \dfrac{\sqrt{3}}{2}=a\sqrt{3}\Rightarrow AD=2a\sqrt{3}$.\\
	Vậy $\left|\vec{AB}-\vec{CA}\right|=\left|\vec{AD}\right|=AD=2a\sqrt{3}$.}
\end{bt}

\begin{bt}%[Thi GHK 1, THPT Đoàn Kết - Hà Nội, 2018]%[Huỳnh Quy, EX-DT-10&11]%[0D1K4-2]
	Cho hai tập khác rỗng: $A=(m-2;6]$, $B=(-2;2m+2)$, với $m\in\mathbb{R}$. Xác định $m$ để $A\subset B$.
	\loigiai{
	Điều kiện của $m$ để các tập $A$, $B$ khác rỗng: $\heva{&m-2<6\\&-2<2m+2}\Leftrightarrow\heva{&m<8\\&-2<m}\Leftrightarrow -2<m<8$.\hfill$(*)$\\
	Khi đó điều kiện để $A\subset B$ là: $\heva{&m-2\geq -2\\&6<2m+2}\Leftrightarrow \heva{&m\geq 0\\&m>2}\Leftrightarrow m>2$.\\
	Kết hợp với điều kiện $(*)$ ta được các giá trị cần tìm của $m$ là $2<m<8$.}
\end{bt}

\begin{bt}%[Thi GHK 1, THPT Đoàn Kết - Hà Nội, 2018]%[Huỳnh Quy, EX-DT-10&11]%[0D1G1-2]
	Cho các số $a$, $b$, $c$ thỏa mãn các điều kiện $\heva{&a+b+c>0&(1)\\&ab+bc+ca>0&(2)\\&abc>0&(3)}$\\
	Chứng minh rằng cả ba số $a$, $b$, $c$ đều dương.
	\loigiai{
	\textbf{Cách 1}: Để chứng minh cả ba số $a$, $b$, $c$ đều dương ta có thể chứng minh bằng phản chứng như sau:\\
	Giả sử tồn tại số không dương, không mất tính tổng quát, giả sử $a\leq 0$.\\
	Từ $(3)$ ta có: $\heva{&abc>0\\&a\leq 0}\Rightarrow \heva{&bc<0\\&a<0.}$\\
	Từ $(2)$ ta có $ab+bc+ca>0\Leftrightarrow a(b+c)+bc>0$. Mà $bc<0\Rightarrow a(b+c)>0\Rightarrow b+c<0$ (do $a<0$).\\
	Suy ra $a+b+c<0$ (mâu thuẫn với $(1)$).\\
	Vậy cả ba số $a$, $b$, $c$ đều dương.\\
	\textbf{Cách 2}: Để chứng minh cả ba số $a$, $b$, $c$ đều dương ta có chứng minh bằng phản chứng như sau:\\
	\textbf{Trường hợp 1}: Cả ba số $a$, $b$, $c$ đều âm.\\
	Ta có: $\heva{&a<0\\&b<0\\&c<0}\Rightarrow a+b+c<0$ trái với giả thiết $(1)$ nên trường hợp 1 không đúng.\\	
	\textbf{Trường hợp 2}: Có một số âm và hai số dương.\\
	Không mất tính tổng quát, ta giả sử $\heva{&a<0\\&b>0\\&c>0}\Rightarrow abc<0$ trái với giả thiết $(3)$ nên trường hợp $2$ không đúng.\\
	\textbf{Trường hợp 3}: Có một số dương và hai số âm.\\
	Không mất tính tổng quát, ta giả sử $\heva{&a>0\\&b<0\\&c<0}\Rightarrow b+c<0\Rightarrow -(b+c)>0$.\\
	Khi đó từ $(1)$ ta có $a>-(b+c)>0\Rightarrow a(b+c)<-(b+c)\cdot(b+c)=-(b+c)^2$.\\
	Từ $(2)$ ta có $ab+ac+ca=a(b+c)+bc<-(b+c)^2+bc<-b^2-2bc+c^2+bc=-b^2-bc-c^2$.\\
	$=\left[\left(b+\dfrac{c}{2}\right)^2+\dfrac{3c^2}{4}\right]<0$ trái với giả thiết $(2)$ nên trường hợp $3$ không đúng.\\
	Vậy cả ba số $a$, $b$, $c$ đều dương.}
\end{bt}
