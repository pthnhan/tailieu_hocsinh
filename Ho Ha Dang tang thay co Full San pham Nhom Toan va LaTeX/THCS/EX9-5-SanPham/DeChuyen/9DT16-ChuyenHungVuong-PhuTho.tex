\begin{ex}%[Đề tuyển sinh lớp 10, chuyên Hùng Vương, Phú Thọ, 2016]%[9D4Y4]%[9D4K5]
\textbf{(2,0 điểm)}
\begin{enumerate}[a)]
\item[a] Cho các số $a,b$ thỏa mãn $2a^2+11ab-3b^2=0,b\neq 2a,b\neq-2a$. Tính giá trị biểu thức $T=\dfrac{a-2b}{2a-b}+\dfrac{2a-3b}{2a+b}$.
\item[b] Cho các số nguyên dương $x,y,z$ và biểu thức $$P=\dfrac{(x^2-y^2)^3+(y^2-z^2)^3+(z^2-x^2)^3}{x^2(y+z)+y^2(z+x)+z^2(x+y)+2xyz}$$
Chứng minh rằng $P$ là số nguyên chia hết cho $6$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item Ta có\\
$T=\dfrac{(a-2b)(2a+b)+(2a-3b)(2a-b)}{(2a-b)(2a+b)}=\dfrac{6a^2-11ab+b^2}{4a^2-b^2}$\\
Từ giả thiết suy ra $11ab=3b^2-2a^2$, thay vào $T$ ta được:\\
$T=\dfrac{6a^2+2a^2-3b^2+b^2}{4a^2-b^2}=\dfrac{2(4a^2-b^2)}{4a^2-b^2}=2$.
\item Ta có: $a^3+b^3+c^3=(a+b+c)(a^2+b^2+c^2-ab-bc-ca)$.\\
Suy ra nếu $a+b+c=0$ thì $a^3+b^3+c^3=3abc$.\\
Vì $(x^2-y^2)+(y^2-z^2)+(z^2-x^2)=0$ nên theo trên ta có:\\
$(x^2-y^2)^3+(y^2-z^2)^3+(z^2-x^2)^3=3(x^2-y^2)(y^2-z^2)(z^2-x^2)$\hfill(1)\\
và $x^2(y+z)+y^2(z+x)+z^2(x+y)+2xyz=x^2y+xy^2+z^2(x+y)+(2xyz+y^2z+x^2z)\\
=xy(x+y)+z^2(x+y)+z(x+y)^2=(x+y)(xy+z^2+zx+zy)\\
=(x+y)(y+z)(z+x)$.\hfill(2)\\
Thay (1) và (2) vào $P$ ta được:\\
$P=\dfrac{3(x^2-y^2)(y^2-z^2)(z^2-x^2)}{(x+y)(y+z)(z+x)}=3(x-y)(y-z)(z-x)\in\mathbb{Z}$.\\
Trong ba số nguyên $x,y,z$ luôn có hai số cùng tính chẵn lẻ, giả sử đó là $x,y\Rightarrow (x-y)\vdots 2$. Do đó $P\vdots 6$.
\end{enumerate}
}
\end{ex}

\begin{ex}
(\textbf{2,0 điểm})
\begin{enumerate}[a)]
\item Tìm các số nguyên $x,y$ thỏa mãn $2x^3+2x^2y+x^2+2xy=x+10$.
\item Cho $19$ điểm phân biệt nằm trong một tam giác đều có cạnh bằng $3$, trong đó không có $3$ điểm nào thẳng hàng. Chứng minh rằng luôn tìm được một tam giác có $3$ đỉnh là $3$ trong $19$ điểm đã cho mà có diện tích không lớn hơn $\dfrac{\sqrt{3}}{4}$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item Ta có\\
$2x^3+2x^2y+x^2+2xy=x+10\Leftrightarrow 2x^2(x+y)+2x(x+y)-(x^2+x)=10 \\
\Leftrightarrow 2(x+y)(x^2+x)-(x^2+x)=10\Leftrightarrow (x^2+x)(2(x+y)-1)=10$.\hfill(1)\\
Mà $x^2+x=x(x+1)$ là số chẵn; $2(x+y)-1$ là số lẻ và $x^2+x=\left(x+\dfrac{1}{2}\right)^2-\dfrac{1}{4}>-1\Rightarrow x^2+x\geq 0$ (vì $x\in\mathbb{Z}$)\\
Do đó từ (1) ta có $\heva{&x^2+x=10\\&2(x+)-1=1}$ hoặc $\heva{&x^2+x=2\\&2(x+y)-1=5}$.\\
Trường hợp $\heva{&x^2+x=10\\&2(x+)-1=1}$. Phương trình $x^2+x=10$ không có nghiệm nguyên.\\
Trường hợp $\heva{&x^2+x=2\\&2(x+)-1=5}\Leftrightarrow\heva{&\hoac{&x=1\\&y=2}\\&\hoac{&x=-2\\&y=5}}$.\\
Vậy có hai bộ số $(x;y)$ thỏa mãn là: $(1;2),(-2;5)$.
\item Chia tam giác đều $ABC$ thành $9$ tam giác đều, có cạnh bằng $1$.\\
Suy ra mỗi tam giác nhỏ có diện tích là $S=\dfrac{\sqrt{3}}{4}$.\\
Vì có $19$ điểm nằm trong $9$ tam giác nhỏ nên có ít nhất $3$ điểm cùng thuộc một hình tam giác nhỏ. Giả sử ba điểm đó là $M,N,P$.\\
Khi đó tam giác $MNP$ nằm trong một tam giác nhỏ nên $S_{\triangle MNP}\leq \dfrac{\sqrt{3}}{4}$.
\end{enumerate}
}
\end{ex}

\begin{ex}
(\textbf{2,0 điểm})
\begin{enumerate}[a)]
\item Giải phương trình $\sqrt{2x+1}-\sqrt{x-3}=2$.
\item Giải hệ phương trình $\heva{&2x^3+x^2y+2x^2+xy+6=0\\&x^2+3x+y=1}$.	
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item Điều kiện $x\geq 3$. Phương trình đã cho tương đương với\\
$\sqrt{2x+1}=2+\sqrt{x-3}\Leftrightarrow 2x+1=4+x-3+4\sqrt{x-3}\Leftrightarrow 4\sqrt{x-3}=x\\
\Leftrightarrow 16(x-3)=x^2\Leftrightarrow x^2-16x+48=0\Leftrightarrow\hoac{&x=4\\&x=12}$.\\
Cả hai nghiệm đều thỏa mãn điều kiện.
\item $\heva{&2x^3+x^2y+2x^2+xy+6=0\\&x^2+3x+y=1}\Leftrightarrow\heva{&(x^2+x)(2x+y)=-6\\&(x^2+x)+(2x+y)=1}$.\\
Đặt $u=x^2+x; v=2x+y$. \\
Hệ đã cho trở thành: $\heva{&uv=-6\\&u+v=1}\Leftrightarrow\hoac{&\heva{&u=-2\\&v=3}\\&\heva{&=3\\&v=-2}}$.\\
Với $\heva{&u=-2\\&v=3}\Rightarrow\heva{&x^2+x=-2\\&2x+y=3}$. Hệ này vô nghiệm.\\
Với $\heva{&u=3\\&v=-2}\Rightarrow\heva{&x^2+x=3\\&2x+y=-2}$. \\
Giải hệ này được hai nghiệm $\heva{&x=\dfrac{-1-\sqrt{13}}{2}\\&y=\sqrt{13}-1}$ và $\heva{&x=\dfrac{-1+\sqrt{13}}{2}\\&y=-\sqrt{13}-1}$.
\end{enumerate}
}
\end{ex}

\begin{ex}
(\textbf{3,0 điểm})
Cho đường tròn $(O;R)$ và dây cung $BC$ cố định. Gọi $A$ là điểm di động trên cung lớn $BC$ sao cho tam giác $ABC$ nhọn. Bên ngoài tam giác $ABC$ dựng các hình vuông $ABDE, ACFG$ và hình bình hành $AEKG$.
\begin{enumerate}[a)]
\item Chứng minh rằng $AK=BC$ và $AK\perp BC$.
\item $DC$ cắt $BF$ tại $M$. Chứng minh rằng $A,K,M$ thẳng hàng.
\item Chứng minh rằng khi $A$ thay đổi trên cung lớn $BC$ của $(O;R)$ thì $K$ luôn thuộc một đường tròn cố định.
\end{enumerate}
\loigiai{
\begin{center}
	\begin{tikzpicture}[>=stealth,scale=.7]
	\tkzDefPoints{0/0/O}
	\tkzDrawCircle[dashed,R](O,3cm)
	\tkzDefShiftPoint[O](-30:3){B}
	\tkzDefShiftPoint[O](220:3){C}
	\tkzDefShiftPoint[O](100:3){A}
	\tkzDefPointBy[rotation=center A angle 90](B)\tkzGetPoint{E}
	\tkzDefPointBy[rotation=center B angle -90](A)\tkzGetPoint{D}
	\tkzDefPointBy[rotation=center A angle -90](C)\tkzGetPoint{G}
	\tkzDefPointBy[rotation=center C angle 90](A)\tkzGetPoint{F}
	\tkzDefPointBy[translation=from A to G](E)\tkzGetPoint{K}
	\tkzInterLL(A,K)(B,C)\tkzGetPoint{H}
	\tkzInterLL(B,F)(D,C)\tkzGetPoint{M}
	\tkzDrawPoints(A,B,C,O,D,E,F,G,K,H,M)
	\tkzDrawSegments(A,B A,C B,C H,K G,K K,E B,F K,C C,D K,B)
	\tkzDrawPolygon(A,C,F,G)
	\tkzDrawPolygon(A,E,D,B)
	\tkzLabelPoints[left](C,F,G)
	\tkzLabelPoints[right](B,D,E)
	\tkzLabelPoints[below](O,H)
	\tkzLabelPoints[above left](M,K,A)
	\tkzMarkRightAngles(A,C,F A,B,D A,H,C)
	\end{tikzpicture}
\end{center}
a. Ta có $\widehat{KEA}+\widehat{EAG}=180^\circ,\widehat{BAC}+\widehat{EAG}=180^\circ\Rightarrow \widehat{KEA}=\widehat{BAC}$. \\
Lại có $EK=AG=AC; EA=AB\Rightarrow\triangle AEK=\triangle BAC\Rightarrow AK=BC$.\\
Ta có $\triangle AEK=\triangle BAC\Rightarrow \widehat{EAK}=\widehat{ABC}$. Gọi $H$ là giao điểm của $KA$ và $BC$, ta có:\\
$\widehat{BAH}+\widehat{ABC}=\widehat{BAH}+\widehat{EAK}=90^\circ\Rightarrow AH\perp BC$. Vậy $AK\perp BC$.\\
b. Vì $\widehat{KAC}=\widehat{KAG}+90^\circ; \widehat{BCF}=\widehat{ACB}+90^\circ$ mà $\widehat{KAG}=\widehat{ACB}\Rightarrow \widehat{KAC}=\widehat{BCF}$.\\
Vì $KA=BC;AC=CF;\widehat{KAC}=\widehat{BCF}\Rightarrow \triangle KAC=\triangle BCF\Rightarrow \widehat{CKH}=\widehat{FBC}$. \\
Ta lại có $\widehat{CKH}+\widehat{KCH}=90^\circ\Rightarrow BF\perp KC$.\hfill(1)\\
Tương tự, ta có $KB\perp CD$.\hfill(2)\\
Từ (1) và (2) suy ra $M$ là trực tâm của $\triangle KBC$, suy ra $M\in KH$. Vậy $A,K,M$ thẳng hàng.\\
c. Dựng hình vuông $BCC'B'$ trên nửa mặt phẳng bờ $BC$ chứa cung lớn $BC$, suy ra $B'C'$ cố định.\\
Ta có $AKB'B$ là hình bình hành (vì $BB', KA$ cùng vuông góc $BC$ suy ra $BB'\parallel KA; BB'=KA=BC$). Do đó $B'K\parallel BA\Rightarrow \widehat{B'KA}=\widehat{BAH}$.\\
Tương tự, ta có $AKC'C$ là hình bình hành, suy ra $KC'\parallel AC\Rightarrow \widehat{AKC'}=\widehat{HAC}$.\\
Suy ra $\widehat{B'KC'}=\widehat{B'KA}+\widehat{AKC'}=\widehat{BAH}+\widehat{HAC}=\widehat{BAC}$. Vì khi $A$ thay đổi trên cung lớn $BC$ của đường tròn $(O;R)$ thì $K$ luôn nhìn đoạn $B'C'$ cố định một góc không đổi $\alpha=\widehat{BAC}$. Do đó $K$ thuộc quỹ tích cung chứa góc $\alpha$ dựng trên đoạn $B'C'$ cố định.
}
\end{ex}

\begin{ex}
(\textbf{1,0 điểm})
Cho các số dương $x,y$. Tìm giá trị nhỏ nhất của biểu thức $$P=\dfrac{2}{\sqrt{(2x+y)^3+1}-1}+\dfrac{2}{\sqrt{(x+2y)^3+1}-1}+\dfrac{(2x+y)(x+2y)}{4}-\dfrac{8}{3(x+y)}$$
\loigiai{
Đặt $2x+y=a; 2y+x=b; a,b>0$ thì $P=\dfrac{2}{\sqrt{a^3+1}-1}+\dfrac{2}{\sqrt{b^3+1}-1}+\dfrac{ab}{4}-\dfrac{8}{a+b}$\\
Ta có $\sqrt{a^3+1}=\sqrt{(a+1)(a^2-a+1)}\leq \dfrac{a+1+a^2-a+1}{2}=\dfrac{a^2+2}{2}\Rightarrow \sqrt{a^3+1}-1\leq \dfrac{a^2}{2}$.\\
Tương tự $\sqrt{b^3+1}-1\leq \dfrac{b^2}{2}$.\\
Mặt khác $\dfrac{4}{a+b}\leq \dfrac{1}{a}+\dfrac{1}{b}\Rightarrow -\dfrac{8}{a+b}\geq -\dfrac{2}{a}-\dfrac{2}{b}$.\\
Vậy $P\geq \dfrac{4}{a^2}+\dfrac{4}{b^2}+\dfrac{ab}{4}-\dfrac{2}{a}-\dfrac{2}{b}=\left(\dfrac{4}{a^2}+1\right)+\left(\dfrac{4}{b^2}+1\right)+\dfrac{ab}{4}-\dfrac{2}{a}-\dfrac{2}{b}-2\geq \dfrac{4}{a}+\dfrac{4}{b}+\dfrac{ab}{4}-\dfrac{2}{a}-\dfrac{2}{b}$\\
Hay $P\geq \dfrac{2}{a}+\dfrac{2}{b}+\dfrac{ab}{4}-2\geq 3\sqrt[3]{\dfrac{2}{a}.\dfrac{2}{b}.\dfrac{ab}{4}}-2=1$.\\
Do đó $\min P=1\Leftrightarrow \heva{&a+1=a^2-a+1\\&b+1=b^2-b+1\\&\dfrac{4}{a^2}=\dfrac{4}{b^2}=1\\&\dfrac{2}{a}=\dfrac{2}{b}=\dfrac{ab}{4}\\&a=b}\Rightarrow a=b=2\Rightarrow x=y=\dfrac{2}{3}$.
}
\end{ex}
