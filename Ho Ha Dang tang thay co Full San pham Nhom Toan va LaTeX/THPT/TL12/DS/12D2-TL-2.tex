\section{HÀM SỐ LŨY THỪA}
\subsection{Lý thuyết cơ bản}
\begin{dn}[Lũy thừa với số mũ nguyên]
Cho $n$ là một số nguyên dương. Với $a$ là số thực tùy ý, lũy thừa bậc $n$ của $a$ là tích của $n$ thừa số $a$.
$$a^n=\underset{n \text{ thừa số}}{a\cdot a\cdot a\ldots\cdot a}$$
với $a$ là cơ số, $n$ là số mũ.
\end{dn}

\begin{dn}[Căn bậc $n$]\hfill
\begin{itemize}
\item Cho số thực $b$ và số nguyên dương $n\geq 2$. Số $a$ được gọi là căn bậc $n$ của số $b$ nếu $a^n=b$.
\item Với $n$ lẻ, $b\in\mathbb{R}$ thì phương trình có duy nhất một căn bậc $n$ của $b$, ký hiệu: $\sqrt[n]{b}$.
\item Với $n$ chẵn:
\begin{enumerate}[+]
\item $b<0$: Không tồn tại căn bậc $n$ của $b$.
\item $b=0$: Có một căn bậc $n$ của $b$ là số $0$.
\item $b>0$: Có hai căn bậc $n$ của $b$ là $-\sqrt[n]{b}$ và $\sqrt[n]{b}$.
\end{enumerate}
\end{itemize}
\end{dn}

\begin{tc}[Tính chất căn bậc $n$]\hfill
\begin{multicols}{3}
\begin{enumerate}
\item $\sqrt[n]{a}\cdot\sqrt[n]{b}=\sqrt[n]{ab}$.
\item $\left(\sqrt[n]{a}\right)^m=\sqrt[n]{a^m}$.
\item $\sqrt[n]{\sqrt[m]{a}}=\sqrt[n\cdot m]{a}$.
\item $\dfrac{\sqrt[n]{a}}{\sqrt[n]{b}}=\sqrt[n]{\dfrac{a}{b}}$.
\item $\sqrt[n]{a^n}=\heva{&-a\text{ khi } n \text{ lẻ}\\&a\text{ khi }n\text{ chẵn}}$.
\end{enumerate}
\end{multicols}
\end{tc}

\begin{dn}[Lũy thừa với số mũ hữu tỉ]\hfill
Cho số thực $a$ dương và  số hữu tỉ $r=\dfrac{m}{n}$, trong đó $m\in\mathbb{Z},n\in\mathbb{N}^*$.\\
Lũy thừa của $a$ với số mũ $r$ là $a^r$ được xác định bởi $$a^r=a^{\frac{m}{n}}=\sqrt[n]{a^m}$$
\end{dn}

\begin{dn}[Lũy thừa với số mũ vô tỉ]
Cho $a$ là một số dương, $\alpha$ là một số vô tỉ. Ta thừa nhận rằng luôn có một dãy số hữu tỉ $(r_n)$ có giới hạn bằng $\alpha$ và dãy số ứng $\left(\alpha^{r_n}\right)$ có giới hạn không phụ thuộc vào việc chọn dãy số $(r_n)$.\\
Ta gọi giới hạn của dãy số $\left(\alpha^{r_n}\right)$ là lũy thừa của $a$ với số mũ $\alpha$. Ký hiệu là $a^{\alpha}$.
$$a^{\alpha}=\lim\limits_{x\to +\infty}a^{r_n}$$ với $\alpha=\lim\limits_{x\to +\infty}r_n$.
\end{dn}

\begin{tc}[Lũy thừa với số mũ thực]\hfill
\begin{multicols}{2}
\begin{enumerate}
\item $a^{\alpha}\cdot a^{\beta}=a^{\alpha+\beta}$.
\item $\dfrac{a^{\alpha}}{a^{\beta}}=a^{\alpha-\beta}$.
\item $\left(a^{\alpha}\right)^{\beta}=a^{\alpha\cdot\beta}$.
\item $(a\cdot b)^{\alpha}=a^{\alpha}\cdot b^{\alpha}$.
\item $\dfrac{a^{\alpha}}{b^{\alpha}}=\left(\dfrac{a}{b}\right)^{\alpha}=\left(\dfrac{b}{a}\right)^{-\alpha}$.
\item Nếu $a>1$ thì $a^{\alpha}>a^{\beta}\Leftrightarrow \alpha>\beta$.
\item Nếu $0<a<1$ thì $a^{\alpha}>a^{\beta}\Leftrightarrow \alpha<\beta$.
\end{enumerate}
\end{multicols}
\end{tc}

\begin{dn}[Công thức lãi kép]
Lãi kép là phần lãi của kì sau được tính trên số tiền gốc kì trước cộng với phần lãi của kì trước.\\
\textbf{Công thức:} Giả sử số tiền gốc là $A$ ; lãi suất $r\%$/kì hạn gửi (có thể là tháng, quý hay năm).
\begin{itemize}
\item Số tiền nhận được cả gốc và lãi sau $n$ kì hạn gửi là $A(1+r)^n$.
\item Số tiền lãi nhận được sau $n$ kì hạn gửi là $A(1+r)^n-A$.
\end{itemize}
\end{dn}
\subsection{Các dạng bài tập}
\begin{dang}{Tính toán - Rút gọn biểu thức lũy thừa}
	Áp dụng các tính chất của lũy thừa để tính giá trị của biểu thức, rút gọn một biểu thức, chứng minh một biểu thức không phụ thuộc tham số.
\end{dang}

\begin{vd}%[Tran Phong]%[2D2Y2]
	Tính giá trị của các biểu thức sau
	\begin{enumerate}
		\item $A=4^{3+\sqrt2}\cdot 2^{1-\sqrt2}\cdot 2^{-4-\sqrt2}$
		\item $B=\left(\dfrac{1}{16}\right)^{-0{,}75}+(0{,}25)^{-\frac{5}{2}}+(0{,}04)^{-1{,}5}-(0{,}125)^{-\frac{2}{3}}$
		\item $C=\left( \left(\sqrt5\right)^{\sqrt2} \right)^{\sqrt2}+4^{1-2\sqrt3}\cdot 16^{1+\sqrt3}$
		\item $D=\dfrac{2^3\cdot 2^{-1}+5^{-3}\cdot 5^4}{10^{-3}:10^{-2}-(0{,}25)^0}$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $A=4^{3+\sqrt2}\cdot 2^{1-\sqrt2}\cdot 2^{-4-\sqrt2}=2^{6+2\sqrt2}\cdot 2^{1-\sqrt2}\cdot 2^{-4-\sqrt2}=2^{6+2\sqrt2+1-\sqrt2-4-\sqrt2}=2^{3}=8.$
			\item $B=\left(\dfrac{1}{16}\right)^{-0{,}75}+(0{,}25)^{-\frac{5}{2}}+(0{,}04)^{-1{,}5}-(0{,}125)^{-\frac{2}{3}}=\left(2^{-4}\right)^{-\frac{3}{4}}+(2^{-2})^{-\frac{5}{2}}+(5^{-2})^{-\frac{3}{2}}-(2^{-3})^{-\frac{2}{3}}$
			\item[] $=2^{3}+2^5+5^3-2^2=8+32+125-4=161.$
			\item $C=\left( \left(\sqrt5\right)^{\sqrt2} \right)^{\sqrt2}+4^{1-2\sqrt3}\cdot 16^{1+\sqrt3}=\left(\sqrt5\right)^2+4^{1-2\sqrt3}\cdot 4^{2+2\sqrt3}=5+4^3=69.$
			\item $D=\dfrac{2^3\cdot 2^{-1}+5^{-3}\cdot 5^4}{10^{-3}:10^{-2}-(0{,}25)^0}=\dfrac{2^2+5}{10^{-1}-1}=-10.$
		\end{enumerate}
	}
\end{vd}

\begin{vd}%[Tran Phong]%[2D2Y2]
	Đơn giản các biểu thức sau
	\begin{enumerate}
		\item $A=\dfrac{a^{\frac{4}{3}}b+ab^{\frac{4}{3}}}{\sqrt[3]a+\sqrt[3]b}$
		\item $B=\sqrt[3]{\sqrt{a^6b^{12}}}-\left(\sqrt[5]{ab^2}\right)^5$
		\item $C=\dfrac{a-1}{a^{\frac{3}{4}}+a^{\frac{1}{2}}}.\dfrac{\sqrt{a}+\sqrt[4]a}{\sqrt{a}+1}:\left(a^{\frac{1}{4}}+1\right)$
		\item $D=\dfrac{a-a^{-2}}{a^{\frac{1}{2}}-a^{-\frac{1}{2}}}-\dfrac{2}{a^{\frac{3}{2}}}-\dfrac{1-a^{-2}}{a^{\frac{1}{2}}+a^{-\frac{1}{2}}}$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $A=\dfrac{a^{\frac{4}{3}}b+ab^{\frac{4}{3}}}{\sqrt[3]a+\sqrt[3]b}=\dfrac{ab\left(a^{\frac{1}{3}}+b^{\frac{1}{3}}\right)}{a^{\frac{1}{3}}+b^{\frac{1}{3}}}=ab.$
			\item $B=\sqrt[3]{\sqrt{a^6b^{12}}}-\left(\sqrt[5]{ab^2}\right)^5=\sqrt[3]{a^3b^6}-ab^2=ab^2-ab^2=0.$
			\item $C=\dfrac{a-1}{a^{\frac{3}{4}}+a^{\frac{1}{2}}}\cdot \dfrac{\sqrt{a}+\sqrt[4]a}{\sqrt{a}+1}:\left(a^{\frac{1}{4}}+1\right)=\dfrac{\left(a^{\frac{1}{2}}-1\right)\left(a^{\frac{1}{2}}+1\right) }{a^{\frac{1}{2}}\left(a^{\frac{1}{4}}+1\right)}\cdot \dfrac{ a^{\frac{1}{4}}\left(a^{\frac{1}{4}}+1\right)}{a^{\frac{1}{2}}+1}\cdot \dfrac{1}{a^{\frac{1}{4}}+1}$
			\item[] $=\dfrac{\left(a^{\frac{1}{4}}-1\right)\left(a^{\frac{1}{4}}+1\right) }{a^{\frac{1}{4}}\left(a^{\frac{1}{4}}+1\right)}=\dfrac{a^{\frac{1}{4}}-1}{a^{\frac{1}{4}}}.$
			\item $D=\dfrac{a-a^{-2}}{a^{\frac{1}{2}}-a^{-\frac{1}{2}}}-\dfrac{2}{a^{\frac{3}{2}}}-\dfrac{1-a^{-2}}{a^{\frac{1}{2}}+a^{-\frac{1}{2}}}=\dfrac{a^3-1}{a^{\frac{5}{2}}-a^{\frac{3}{2}}}-\dfrac{2}{a^{\frac{3}{2}}}-\dfrac{a^2-1}{a^{\frac{5}{2}}+a^{\frac{3}{2}}}=\dfrac{a^3-1}{a^{\frac{3}{2}}(a-1)}-\dfrac{2}{a^{\frac{3}{2}}}-\dfrac{a^2-1}{a^{\frac{3}{2}}(a+1)}$
			\item[] $=\dfrac{a^2+a+1}{a^{\frac{3}{2}}}-\dfrac{2}{a^{\frac{3}{2}}}-\dfrac{a-1}{a^{\frac{3}{2}}}=\dfrac{a^2}{a^{\frac{3}{2}}}=a^{\frac{1}{2}}=\sqrt{a}.$
		\end{enumerate}
	}
\end{vd}

		\item $B=\dfrac{a^{\frac{4}{3}}-8a^{\frac{1}{3}}b}{a^{\frac{2}{3}}+2\sqrt[3]{ab}+4b^{\frac{2}{3}}}:\left(1-2\sqrt[3]{\dfrac{b}{a}}\ \right)$
		\item $C=\dfrac{\left(a^{\frac{1}{p}}-a^{\frac{1}{q}}\right)^2+4a^{\frac{(p+q)}{pq}}}{\left(a^{\frac{2}{p}}-a^{\frac{2}{q}}\right)\left(\sqrt[p]{a^{p+1}}+\sqrt[q]{a^{q+1}}\right)}$

	\end{enumerate}
	\loigiai{

			\item[] $=\dfrac{\left(a^{\frac{1}{2}}+2\right)\left(a^{\frac{1}{2}}-1\right)-\left(a^{\frac{1}{2}}-2\right)\left(a^{\frac{1}{2}}+1\right)}{\left(a^{\frac{1}{2}}-1\right)\left(a^{\frac{1}{2}}+1\right)^2}\cdot\dfrac{a^{\frac{1}{2}}+1}{a^{\frac{1}{2}}}=\dfrac{2a^{\frac{1}{2}}}{a^{\frac{1}{2}}(a-1)}=\dfrac{2}{a-1}.$
			
			\item $B=\dfrac{a^{\frac{4}{3}}-8a^{\frac{1}{3}}b}{a^{\frac{2}{3}}+2\sqrt[3]{ab}+4b^{\frac{2}{3}}}:\left(1-2\sqrt[3]{\dfrac{b}{a}}\ \right)=\dfrac{a^{\frac{1}{3}}\left(a-8b\right)}{a^{\frac{2}{3}}+2a^{\frac{1}{3}}b^{\frac{1}{3}}+4b^{\frac{2}{3}}}:\left(1-2\dfrac{b^{\frac{1}{3}}}{a^{\frac{1}{3}}}\ \right)$
			\item[] $=\dfrac{a^{\frac{1}{3}}\left(a^{\frac{1}{3}}-2b^{\frac{1}{3}}\right)\left( a^{\frac{2}{3}}+2a^{\frac{1}{3}}b^{\frac{1}{3}}+4b^{\frac{2}{3}}\right)}{a^{\frac{2}{3}}+2a^{\frac{1}{3}}b^{\frac{1}{3}}+4b^{\frac{2}{3}}}:\left(\dfrac{a^{\frac{1}{3}}-2b^{\frac{1}{3}}}{a^{\frac{1}{3}}}\right)=a^{\frac{1}{3}}\cdot \left(a^{\frac{1}{3}}-2b^{\frac{1}{3}}\right)\cdot \dfrac{a^{\frac{1}{3}}}{a^{\frac{1}{3}}-2b^{\frac{1}{3}}}=a^{\frac{2}{3}}.$

			\item[] $=\dfrac{\left(a^{\frac{1}{p}}+a^{\frac{1}{q}}\right)^2}{\left(a^{\frac{1}{p}}-a^{\frac{1}{q}}\right)\left(a^{\frac{1}{p}}+a^{\frac{1}{q}}\right)\left(a^{1+\frac{1}{p}}+a^{1+\frac{1}{q}}\right)}=\dfrac{1}{a\left(\sqrt[p]{a}-\sqrt[q]{a}\right)}.$

\begin{dang}{So sánh lũy thừa hay căn số}
\begin{enumerate}
\item So sánh lũy thừa cùng cơ số $a$ ta áp dụng kết quả sau:
\begin{itemize}
\item Với $a>1$ thì $a^{x_1}>a^{x_2}\Leftrightarrow x_1>x_2$.
\item Với $0<a<1$ thì $a^{x_1}>a^{x_2}\Leftrightarrow x_1<x_2$.
\end{itemize}
\item So sánh hai lũy thừa cùng số mũ ta dùng kết quả sau:\\
Với $a,b\neq 1$ và $0<b^x<a^x\Leftrightarrow\heva{&x>0\Leftrightarrow b^x<a^x\\&x<0\Leftrightarrow b^x>a^x}$.
\item Với hai biểu thức chứa căn, ta cần đưa về các căn đồng bậc.
\end{enumerate}
\end{dang}

\begin{vd}%[Dũng Lê]%[2D2Y2]
Không dùng máy tính, so sánh các số sau: $3^{600}$ và $5^{400}$.
\loigiai{
Ta có: $3^{600}=3^{3\cdot 200}=27^{200}$ và $5^{400}=5^{2\cdot 200}=25^{200}$.\\
Mà $27>25>1$, suy ra $3^{600}>5^{400}$.
}
\end{vd}

\begin{vd}%[Dũng Lê]%[2D2Y2]
So sánh $32^{10}$ và $16^{15}$.
\loigiai{
Các cơ số $32$ và $16$ tuy khác nhau nhưng đều là luỹ thừa của $2$ nên ta tìm cách đưa $32^{10}$ và $16^{15}$ về luỹ thừa cùng cơ số $2$.\\
Ta có $32^{10} = (2^5)^{10} = 2^{50}$ và $16^{15} = (2^4)^{15} = 2^{60}$.\\
Vì $2^{50} < 2^{60}$ suy ra $32^{10} < 16^{15}$. 
}
\end{vd}

\begin{vd}
So sánh  $\sqrt{10}$  và $\sqrt[3]{30}$.
\loigiai{
Đưa về các căn thức cùng bậc $6$, ta có:\\
$\sqrt{10}=\sqrt[6]{10^3}=\sqrt[6]{1000}$.\\
$\sqrt[3]{30}=\sqrt[6]{30^2}=\sqrt[6]{900}$.\\
Mà $900<1000$ nên $\sqrt[3]{30}<\sqrt{10}$.
}
\end{vd}

\begin{vd}%[Dũng Lê]%[2D2K2]
Tìm xem $2^{100}$ có bao nhiêu chữ số trong cách viết ở hệ thập phân.
\loigiai{
Muốn biết $2^{100}$ có bao nhiêu chữ số trong cách viết thập phân ta so sánh $2^{100}$ với $10^{30}$ và $10^{31}$.\\
Ta có: $2^{100}=(2^{10})^{10}=1024^{10}$.\\
$10^{30}=(10^3)^{10}=1000^{10}$.  Mà $1024>1000$, suy ra $2^{100}>10^{30}$.\\
$10^{31}=(2\cdot 5)^{31}=2^{31}5^{31}=2^{31}\cdot 625^7\cdot 5^3$.\\
$2^{100}=2^{31}\cdot 2^{69}=2^{31}\cdot 625^7\cdot 4^3$.\\
Mà $2^{31}\cdot 625^7\cdot 5^3>2^{31}\cdot 625^7\cdot 4^3$, suy ra $2^{100}<10^{31}$.\\
Do đó $10^{30}<2^{100}<10^{31}$, suy ra $2^{100}$ có $31$ chữ số ở cách viết ở hệ thập phân.
}
\end{vd}

\begin{bt}%[Dũng Lê]%[2D2B2]
So sánh hai số $A=\dfrac{19^{30}+5}{19^{31}+5}$ và $B=\dfrac{19^{31}+5}{19^{32}+5}$.
\loigiai{
Ta có $A=\dfrac{19^{30}+5}{19^{31}+5}\Leftrightarrow 19A=\dfrac{19^{31}+95}{19^{31}+5}=1+\dfrac{90}{19^{31}+5}$.\\
$B=\dfrac{19^{31}+5}{19^{32}+5}\Leftrightarrow 19B=\dfrac{19^{32}+95}{19^{32}+5}=1+\dfrac{90}{19^{32}+2}$.\\
Vì $\dfrac{90}{19^{31}+5}>\dfrac{90}{19^{32}+2}
\Rightarrow 19A>19B\Rightarrow A>B$.
}
\end{bt}

\begin{bt}%[Dũng Lê]%[2D2B2]
So sánh $A=\dfrac{1+5+5^2+\ldots+5^9}{1+5+5^2+\ldots+5^8}$ và $B=\dfrac{1+3+3^2+\ldots+3^9}{1+3+3^2+\ldots+3^8}$.
\loigiai{
Ta có: $A=\dfrac{1+5+5^2+\ldots+5^9}{1+5+5^2+\ldots+5^8}=\dfrac{1+5(1+5+5^2+\ldots+5^8)}{1+5+5^2+\ldots+5^8}=5+\dfrac{1}{1+5+5^2+\ldots+5^8}$.\\
$B=\dfrac{1+3+3^2+\ldots+3^9}{1+3+3^2+\ldots+3^8}=\dfrac{1+3(1+3+3^2+\ldots+3^8)}{1+3+3^2+\ldots+3^8}=3+\dfrac{1}{1+3+3^2+\ldots+3^8}$.\\
Do đó: $A>5>4>B$.
}
\end{bt}

\begin{bt}%[Dũng Lê]%[2D2B2]

\loigiai{
Ta có $3^{500}=(3^5)^{100}=243^{100}$.\\
$7^{300}=(7^3)^{100}=343^{100}$.\\
Vì $243<343$, suy ra $3^{500}<7^{300}$.

\begin{bt}%[Dũng Lê]%[2D2B2]
So sánh hai số sau $107^{50}$ và $73^{75}$.
\loigiai{
Ta có $107^{50}=(107^2)^{25}=11449^{25}$.\\
$73^{75}=(73^3)^{25}=389017^{25}$.\\

\begin{dang}{Bài toán lãi kép}
	\textbf{- Bài toán 1.} Ông Dũng gửi số tiền $P$ đồng vào ngân hàng T với lãi suất $r\% / \mbox{năm}$. Biết rằng nếu không rút tiền ra khỏi ngân hàng thì cứ sau mỗi năm, số tiền lãi sẽ được nhập vào vốn ban đầu. Hỏi ông Dũng được lãnh bao nhiêu tiền (cả vốn ban đầu và lãi) sau $n$ năm, nếu trong khoảng thời gian này không rút tiền ra và lãi suất không thay đổi?\\
	Số tiền ông Dũng được lãnh sau $n$ năm là 
	\begin{center}
		$P_n=P(1+r)^n$. 
	\end{center}
	\textbf{- Bài toán 2.} Đầu mỗi tháng, bà Vân gửi số tiền $P$ đồng vào ngân hàng N với lãi suất $r\% / \mbox{tháng}$. Hỏi bà Vân được lãnh bao nhiêu tiền (cả vốn ban đầu và lãi) sau $n$ tháng, nếu trong khoảng thời gian này không rút tiền ra và lãi suất không thay đổi?\\ 
	Số tiền bà Vân được lãnh sau $n$ tháng là 
	\begin{center}
		$S_n=\dfrac{P}{r}\left[ (1+r)^n-1\right].(1+r)$. 
	\end{center}
	\textbf{- Bài toán 3.} Ông Hùng gửi số tiền $P$ đồng vào ngân hàng N với lãi suất $r\% / \mbox{tháng}$. Mỗi tháng ông vào ngân hàng này tính lãi, rút ra số tiền $Q$ đồng. Hỏi số tiền còn lại sau $n$ tháng là bao nhiêu?\\
	Số tiền còn lại của ông Hùng sau $n$ tháng là
	\begin{center}
		$S_n=P(1+r)^n-Q\dfrac{(1+r)^n-1}{r}$. 
	\end{center}
	\textbf{- Bài toán 4.} Tính dân số.\\
	Dân số thế giới được ước tính theo công thức\\
	\begin{center}
		$S=A.e^{ni}$.
	\end{center}
	Trong đó $A$ là dân số của năm lấy làm mốc tính, $S$ là dân số sau $n$ năm, $i$ là tỉ lệ tăng dân số hàng năm.
\end{dang}
