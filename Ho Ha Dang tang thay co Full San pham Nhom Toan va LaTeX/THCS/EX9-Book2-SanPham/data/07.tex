\section{BẤT ĐẲNG THỨC VỀ PHÂN SỐ VÀ CĂN BẬC HAI}
Các bài toán trong chuyên đề này thuộc các dạng chính sau:
\begin{itemize}
\item[-] \textit{Dạng 1:} Chứng minh một biểu thức lớn hơn hoặc nhỏ hơn một số cho trước
\item[-] \textit{Dạng 2:} Tìm giá trị của biến để một hoặc nhiều bất đẳng thức được thỏa mãn
\item[-] \textit{Dạng 3:} Tìm giá trị nhỏ nhất, giá trị lớn nhất của một biểu thức.
\end{itemize}
Để chứng minh biểu thức $A$ nhỏ hơn hằng số $m$ ta thường làm trội biểu thức $A$ thành biểu thức $B$ thỏa mãn hai điều kiện: Biểu thức $B$ dễ tính toán và $B \leq m.$
Để chứng minh biểu thức $A$ lớn hơn hằng số $m$ ta thường \textit{làm giảm} biểu thức $A$ thành biểu thức $C$ thỏa mãn hai điều kiện: Biểu thức $C$ dễ tính toán và $C \geq m.$
 
\subsection{Bất đẳng thức về phân số}

Khi chứng minh các bất đẳng thức về phân số, nhiều khi ta phải tính tổng các phân số được viết theo quy luật. Khi đó, ta thường tìm cách viết mỗi phân số dưới dạng hiệu của hai phân số để xuất hiện trong biểu thức những phân số đối nhau.
\begin{vd}
Cho $A=\dfrac{1}{1\cdot 3}+\dfrac{1}{2\cdot 4}+\dfrac{1}{3\cdot 5}\cdots +\dfrac{1}{20\cdot 22}.$ Chứng minh rằng $A<\dfrac{3}{4}.$
\loigiai{
Ta có $$\begin{aligned} 2A&=\left(\dfrac{2}{1\cdot 3}+\dfrac{2}{3\cdot 5}+\cdots+\dfrac{1}{19\cdot 21} \right)+\left(\dfrac{2}{2\cdot 4}+\dfrac{2}{4\cdot 6}+\cdots+\dfrac{1}{20\cdot 22} \right) \\& =\left(\dfrac{1}{1}-\dfrac{1}{3}+\dfrac{1}{3}-\dfrac{1}{5}+\cdots +\dfrac{1}{19}-\dfrac{1}{21} \right)+\left(\dfrac{1}{2}-\dfrac{1}{4}+\dfrac{1}{4}-\dfrac{1}{6}+\cdots +\dfrac{1}{20}-\dfrac{1}{22} \right)\\& =1-\dfrac{1}{21}+\dfrac{1}{2}-\dfrac{1}{22}<\dfrac{3}{2}. \end{aligned}$$
Vậy $A<\dfrac{3}{4}.$
}
\end{vd}
\begin{vd}
Cho $A=\dfrac{1}{3^3}+\dfrac{1}{4^3}+\cdots+\dfrac{1}{99^3}$. Chứng minh rằng 
\begin{multicols}{2}
\begin{enumerate}[a)]
\item $A<\dfrac{1}{12}$
\item $A>\dfrac{1}{27}$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}[a)]
\item Với $n>1$ ta có $$n^3>n^3-n=(n-1)n(n+1) \Rightarrow \dfrac{1}{n^3} <\dfrac{1}{(n-1)n(n+1)}.$$
Mặt khác $\dfrac{2}{(n-1)n(n+1)}=\dfrac{1}{(n-1)n}-\dfrac{1}{n(n+1)}$, nên ta có 
$$2A<\dfrac{1}{2\cdots 3}-\dfrac{1}{3\cdot 4}+\dfrac{1}{3\cdot 4}-\dfrac{1}{4\cdot 5}+\cdots+\dfrac{1}{98\cdot 99}+\dfrac{1}{99\cdot 100}=\dfrac{1}{6}-\dfrac{1}{99\cdot 100}<\dfrac{1}{6}.$$
Vậy $A<\dfrac{1}{12}.$
\item Ta có $A>\dfrac{1}{3^3}=\dfrac{1}{27}.$
\end{enumerate}
}
\end{vd}
\begin{vd}
   Cho $A=\dfrac{1}{3^1}+\dfrac{3}{3^2}+\dfrac{5}{3^3}+\cdots+ \dfrac{2n-1}{3^n}$. Chứng minh rằng $A<1.$
    \loigiai{
       Ta có $$\dfrac{2n-1}{3^n}=\dfrac{3n-(n+1)}{3^n}=\dfrac{n}{3^{n-1}}-\dfrac{n+1}{3^n}.$$
       Do đó 
       $$A=\dfrac{1}{3^0}-\dfrac{2}{3^1}+\dfrac{2}{3^1}-\dfrac{3}{3^2}+\cdots +\dfrac{n}{3^{n-1}}-\dfrac{n+1}{3^n}=1-\dfrac{n+1}{3^n}<1.$$
    }
\end{vd}

\subsection{Bất đẳng thức về căn bậc hai}
Khi chứng minh bất đẳng thức về căn bậc hai, ta cũng gặp những trường hợp cần tính tổng những số viết theo quy luật. Ta cũng tìm cách viết mỗi số hạng thành một hiệu để xuất hiện trong biểu thức những số hạng đối nhau.
\begin{vd}
Cho $A=\dfrac{1}{\sqrt{1}+\sqrt{2}}+\dfrac{1}{\sqrt{3}+\sqrt{4}}+\cdots+\dfrac{1}{\sqrt{98}+\sqrt{99}}$. Chứng minh rằng $A>4{,}5$.
    \loigiai{Ta có:
    \begin{align*}
    A&=\dfrac{1}{\sqrt{1}+\sqrt{2}}+\dfrac{1}{\sqrt{3}+\sqrt{4}}+\cdots+\dfrac{1}{\sqrt{98}+\sqrt{99}}\\
    \Rightarrow A &> \dfrac{1}{\sqrt{2}+\sqrt{3}}+\dfrac{1}{\sqrt{4}+\sqrt{5}}+\cdots+\dfrac{1}{\sqrt{99}+\sqrt{100}}
    \end{align*}
Cộng vế với vế ta được $2A>\dfrac{1}{\sqrt{1}+\sqrt{2}}+\dfrac{1}{\sqrt{2}+\sqrt{3}}+\cdots+\dfrac{1}{\sqrt{99}+\sqrt{100}}$\\
Suy ra $2A>\dfrac{\sqrt{1}-\sqrt{2}}{1-2}+\dfrac{\sqrt{2}-\sqrt{3}}{2-3}+\cdots+\dfrac{\sqrt{99}-\sqrt{100}}{99-100}=\dfrac{\sqrt{1}-\sqrt{100}}{-1}=9$.\\
Vậy $A>4{,}5$.}
\end{vd}
\begin{vd}
Cho $A=\dfrac{1}{\sqrt{1}+\sqrt{3}}+\dfrac{1}{2(\sqrt{3}+\sqrt{5})}+\dfrac{1}{3(\sqrt{5}+\sqrt{7})}+\cdots+\dfrac{1}{40(\sqrt{79}+\sqrt{81})}$.\\
Chứng minh rằng $A<\dfrac{8}{9}$.
\loigiai{Ta có
	{\allowdisplaybreaks
	\begin{align*} \dfrac{1}{n(\sqrt{2n-1}+\sqrt{2n+1})}&=\dfrac{\sqrt{2n+1}-\sqrt{2n-1}}{2n}=\dfrac{\sqrt{2n+1}-\sqrt{2n-1}}{\sqrt{4n^2}}\\
	&<\dfrac{\sqrt{2n+1}-\sqrt{2n-1}}{\sqrt{4n^2-1}}\\
&=\dfrac{\sqrt{2n+1}-\sqrt{2n-1}}{\sqrt{(2n+1)(2n-1)}}=\dfrac{1}{\sqrt{2n-1}}-\dfrac{1}{\sqrt{2n+1}}.
\end{align*}}Do đó $A<\dfrac{1}{\sqrt{1}}-\dfrac{1}{\sqrt{3}}+\dfrac{1}{\sqrt{3}}-\dfrac{1}{\sqrt{5}}+\dfrac{1}{\sqrt{5}}-\dfrac{1}{\sqrt{7}}+\cdots+\dfrac{1}{\sqrt{79}}-\dfrac{1}{\sqrt{81}}=1-\dfrac{1}{9}=\dfrac{8}{9}$.}
\end{vd}
\begin{vd}
	\hfill\\
	\begin{enumerate}
		\item Chứng minh rằng với mọi số nguyên dương $n$ thì $$\dfrac{1}{(\sqrt{n}+\sqrt{n+2})^3}<\dfrac{1}{8}\cdot\left(\dfrac{1}{\sqrt{n}}-\dfrac{1}{\sqrt{n+2}}\right).$$
		\item Áp dụng kết quả trên chứng minh rằng $$\dfrac{1}{(1+\sqrt{3})^3}+\dfrac{1}{(\sqrt{3}+\sqrt{5})^3}+\cdots+\dfrac{1}{(\sqrt{623}+\sqrt{625})^3}<\dfrac{3}{25}.$$
	\end{enumerate}
\loigiai{\begin{enumerate}
		\item \begin{align*}
		&\dfrac{1}{(\sqrt{n}+\sqrt{n+2})^3}<\dfrac{1}{8}\cdot\left(\dfrac{1}{\sqrt{n}}-\dfrac{1}{\sqrt{n+2}}\right)\\
		\Leftrightarrow\quad&(\sqrt{n}+\sqrt{n+2})^3(\sqrt{n+2}-\sqrt{n})>8\sqrt{n}\sqrt{n+2}\\
		\Leftrightarrow\quad&(\sqrt{n}+\sqrt{n+2})^2(n+2-n)>8\sqrt{n}\sqrt{n+2}\\
		\Leftrightarrow\quad&(\sqrt{n}+\sqrt{n+2})^2>4\sqrt{n}\sqrt{n+2}\\
		\Leftrightarrow\quad&(\sqrt{n+2}-\sqrt{n})^2>0\quad (\textrm{luôn đúng}).
		\end{align*}
		Vậy ta có điều phải chứng minh.
		\item Theo câu a ta có
		\begin{align*}
		&\dfrac{1}{(1+\sqrt{3})^3}+\dfrac{1}{(\sqrt{3}+\sqrt{5})^3}+\ldots+\dfrac{1}{(\sqrt{623}+\sqrt{625})^3}\\
		<\quad&\dfrac{1}{8}\cdot\left(\dfrac{1}{\sqrt{1}}-\dfrac{1}{\sqrt{3}}+\dfrac{1}{\sqrt{3}}-\dfrac{1}{\sqrt{5}}+\ldots+\dfrac{1}{\sqrt{623}}-\dfrac{1}{\sqrt{625}}\right)=\dfrac{1}{8}\cdot\left(1-\dfrac{1}{25}\right)=\dfrac{3}{25}.		
		\end{align*}
		Vậy ta có điều phải chứng minh.
\end{enumerate}}
\end{vd}

    \textbf{BÀI TẬP}
\end{center}
   \begin{bt}
	Cho $A = \dfrac{1}{2}\cdot \dfrac{3}{4} \cdot \dfrac{5}{6} \cdots \dfrac{95}{96}$. Chứng minh rằng
    \begin{enumerate}
    \item $A < \dfrac{1}{9}$
    \item $A < \dfrac{1}{12}$
    \end{enumerate}
    \loigiai{
    	\begin{enumerate}
    	\item Ta có $A = \dfrac{1}{2}\cdot \dfrac{3}{4} \cdot \dfrac{5}{6} \cdots \dfrac{95}{96} \Rightarrow A < \dfrac{2}{3}\cdot \dfrac{4}{5} \cdot \dfrac{6}{7} \cdots \dfrac{96}{97}$.\\
        Do đó $A^2 < \dfrac{1 \cdot 3 \cdots 95}{2 \cdot 4 \cdots  96} \cdot \dfrac{2 \cdot 4 \cdots 96}{3 \cdot 5 \cdots  97} < \dfrac{1}{81}\Rightarrow A < \dfrac{1}{9}$.
        \item Trước hết ta chứng minh với $n>1$ thì $\left(\dfrac{2n-1}{2n}\right)^2 < \dfrac{3n-2}{3n+1}$ (bạn đọc tự chứng minh). Do đó
        $$A^2 = \left(\dfrac{1}{2}\right)^2\cdot \left(\dfrac{3}{4}\right)^2 \cdot \left(\dfrac{5}{6}\right)^2 \cdot\ldots\cdot \left(\dfrac{95}{96}\right)^2 < \dfrac{1}{4}\cdot \dfrac{4}{7}\cdot\ldots\cdot\dfrac{142}{145} = \dfrac{1}{145}< \dfrac{1}{144}.$$
        Vậy $A < \dfrac{1}{12}$.
    	\end{enumerate}
    }
\end{bt}

\begin{bt}
	Chứng minh với mọi số nguyên dương $n$ thì
    \begin{enumerate}
    \item $\dfrac{1}{3^2}+\dfrac{1}{5^2}+\ldots+\dfrac{1}{(2n+1)^2} < \dfrac{1}{4}$;
    \item $\dfrac{1}{1^2+2^2}+\dfrac{1}{2^2+3^2}+\ldots+\dfrac{1}{n^2+(n+1)^2} < \dfrac{1}{2}$;
    \item $\dfrac{1}{1^4+1^2+1}+\dfrac{2}{2^4+2^2+1}+\ldots+\dfrac{n}{n^4+n^2+1} < \dfrac{1}{2}$.
    \end{enumerate}
    \loigiai{
    	\begin{enumerate}
    	\item Sử dụng nhận xét $\dfrac{1}{(2n+1)^2} = \dfrac{1}{4n^2+4n+1}< \dfrac{1}{4n(n+1)} = \dfrac{1}{4}\cdot\left(\dfrac{1}{n}-\dfrac{1}{n+1}\right)$.
        \item Sử dụng nhận xét $\dfrac{1}{n^2+(n+1)^2}=\dfrac{1}{2n^2+2n+1}<\dfrac{1}{2n(n+1)}=\dfrac{1}{2}\cdot\left(\dfrac{1}{n}-\dfrac{1}{n+1}\right)$
        \item Sử dụng nhận xét \begin{align*}
        &\dfrac{2n}{n^4+n^2+1}=\dfrac{2n}{(n^2+1)^2-n^2} = \dfrac{1}{(n^2+n+1)(n^2-n+1)} \\
        = &\dfrac{(n^2+n+1)-(n^2-n+1)}{(n^2+n+1)(n^2-n+1)} = \dfrac{1}{n(n-1)+1}-\dfrac{1}{n(n+1)+1}.\end{align*}
    	\end{enumerate}
    }
\end{bt}

\begin{bt}
	Chứng minh rằng với mọi số nguyên dương $n$ thì $$\dfrac{1}{2}+\dfrac{2}{2^2}+\ldots+\dfrac{n}{2^n} < 2.$$
    \loigiai{
    	$A = \dfrac{1}{2} + \dfrac{2}{2^2} + \ldots + \dfrac{n}{2^n}$. Ta có\\
        $\dfrac{n}{2^n} = \dfrac{2(n+1)-(n+2)}{2^n} = \dfrac{n+1}{2^{n-1}} - \dfrac{n+2}{2^n}$.\\
        Do đó $A = \dfrac{2}{2^0} - \dfrac{3}{2^1} + \dfrac{3}{2^1} - \dfrac{4}{2^2}+\ldots+\dfrac{n+1}{2^{n-1}} - \dfrac{n+2}{2^n} = 2 - \dfrac{n+2}{2^n} < 2$.
    }
\end{bt}

\begin{bt}
	Chứng minh bất đẳng thức $$\dfrac{2}{5+1}+\dfrac{4}{5^2+1}+\dfrac{8}{5^4+1}+\dfrac{16}{5^8+1}+\dfrac{32}{5^{16}+1}+\dfrac{64}{5^{32}+1}<\dfrac{1}{2}.$$
    \loigiai{
    	Đặt $A = \dfrac{2}{5+1}+\dfrac{4}{5^2+1}+\dfrac{8}{5^4+1}+\dfrac{16}{5^8+1}+\dfrac{32}{5^{16}+1}+\dfrac{64}{5^{32}+1}$.\\
        Ta có $\dfrac{1}{k+1} = \dfrac{k-1}{k^2-1} = \dfrac{k+1-2}{k^2-1} = \dfrac{1}{k-1}-\dfrac{2}{k^2-1}$.\\
        Suy ra $\dfrac{n}{k+1} = \dfrac{n}{k-1} - \dfrac{2n}{k^2-1}$.\\
        Do đó $A = \left(\dfrac{2}{5-1}-\dfrac{4}{5^2-1}\right)+\left(\dfrac{4}{5^2-1}-\dfrac{8}{5^4-1}\right)+\ldots+\left(\dfrac{64}{5^{32}-1}-\dfrac{128}{5^{64}-1}\right) < \dfrac{1}{2}$
    }
\end{bt}

\begin{bt}
	Cho các số nguyên dương $a,b,c,d$ thỏa mãn $1\le a \le b \le c \le d \le 100$. Tìm giá trị nhỏ nhất của $B = \dfrac{a}{b}+\dfrac{c}{d}$.
    \loigiai{
    	Do $1 \leq a \leq b \leq c$ nên $\dfrac{a}{b} \geq \dfrac{1}{b} \geq \dfrac{1}{c}$.\hfill$(1)$\\
        Dấu bằng xảy ra khi $a=1;b=c$.\\
        Do $d \leq 100$ nên $\dfrac{c}{d} \geq \dfrac{c}{100}$.\hfill$(2)$\\
        Từ (1) và (2) suy ra $$\dfrac{a}{b}+\dfrac{c}{d} \geq \dfrac{1}{c} + \dfrac{c}{100} = \dfrac{c^2 + 100}{100c} = \dfrac{c^2 - 20c + 100 + 20c}{100c} = \dfrac{(c-10)^2}{100c}+\dfrac{1}{5} \geq \dfrac{1}{5}.$$
        Vậy giá trị nhỏ nhất của $B$ là $\dfrac{1}{5}$ tại $a=1;b=c=10;d=100$.
    }
\end{bt}

\begin{bt}
	Chứng minh rằng số $2^{300}$ có $91$ chữ số khi viết ở dạng thập phân.
    \loigiai{
    	Để chứng minh $2^{300}$ có $91$ chữ số thập phân ta sẽ chứng minh $10^{90} < 2^{300} < 10^{91}$.\\
        Ta có $2^{10}>10^3$ nên $2^{300} = (2^{10})^{30} > (10^3)^{30} = 10^{90}$.\hfill$(1)$\\
        Ta sẽ chứng minh $2^{300} < 10^{91}\Leftrightarrow \left(\dfrac{2^{10}}{10^3}\right)^{30} < 10$.\\
        Thật vậy, xét $\dfrac{2^{10}}{10^3} = \dfrac{1024}{1000}<\dfrac{1025}{1000}=\dfrac{41}{40}$.\\
    Suy ra     $\left(\dfrac{2^{10}}{10^3}\right)^{30} < \left(\dfrac{41}{40}\right)^{30} < \underbrace{\dfrac{41}{40}\cdot \dfrac{40}{39}\cdot\ldots\cdot\dfrac{12}{11}}_{30 \text{ phân số}}=\dfrac{41}{11}<10 \Rightarrow 2^{300} < 10^{91}$.\hfill$(2)$\\
        Từ (1) và (2) ta có $10^{90} < 2^{300} < 10^{91}$. Vậy $2^{300}$ có $91$ chữ số khi viết ở dạng thập phân.
    }
\end{bt}

\begin{bt}
	Chứng minh bất đẳng thức $$\dfrac{1}{\sqrt{1}+\sqrt{3}}+\dfrac{1}{\sqrt{5}+\sqrt{7}}+\ldots+\dfrac{1}{\sqrt{77}+\sqrt{79}}>2.$$
    \loigiai{
    	Đặt $A = \dfrac{1}{\sqrt{1}+\sqrt{3}}+\dfrac{1}{\sqrt{5}+\sqrt{7}}+\ldots+\dfrac{1}{\sqrt{77}+\sqrt{79}}$.\\
        Ta có $A > \dfrac{1}{\sqrt{3}+\sqrt{5}}+\dfrac{1}{\sqrt{7}+\sqrt{9}}+\ldots+\dfrac{1}{\sqrt{79}+\sqrt{81}}$\\
        nên $2A > \dfrac{1}{\sqrt{1}+\sqrt{3}}+\dfrac{1}{\sqrt{3}+\sqrt{5}}+\ldots+\dfrac{1}{\sqrt{79}+\sqrt{81}}$\\
        $\Rightarrow 2A > \dfrac{\sqrt{1}-\sqrt{3}}{-2}+\dfrac{\sqrt{3}-\sqrt{5}}{-2}+\ldots+\dfrac{\sqrt{79}-\sqrt{81}}{-2} = \dfrac{\sqrt{81}-\sqrt{1}}{2} = 4$.\\
        Vậy ta có $A > 2$.
    }
\end{bt}

\begin{bt}
	Chứng minh bất đẳng thức $$\dfrac{1}{(\sqrt{2}+\sqrt{5})^3}+\dfrac{1}{(\sqrt{5}+\sqrt{8})^3}+\ldots+\dfrac{1}{(\sqrt{95}+\sqrt{98})^3}<\dfrac{1}{19}.$$
    \loigiai{
    	Áp dụng $(x+y)^2 > 4xy$ với $x\neq y$, ta có\\
        $(\sqrt{n} + \sqrt{n+3})^3 = (\sqrt{n} + \sqrt{n+3})(\sqrt{n} + \sqrt{n+3})^2 > (\sqrt{n} + \sqrt{n+3}) \cdot 4 \cdot \sqrt{n} \cdot \sqrt{n+3}$\\
        $\Rightarrow \dfrac{1}{(\sqrt{n}+\sqrt{n+3})^3} < \dfrac{1}{(\sqrt{n} + \sqrt{n+3}) \cdot 4 \cdot \sqrt{n} \cdot \sqrt{n+3}} = \dfrac{\sqrt{n+3} - \sqrt{n}}{(n+3 - n) \cdot 4 \cdot \sqrt{n} \cdot \sqrt{n+3}}$\\
        $\Rightarrow \dfrac{1}{(\sqrt{n}+\sqrt{n+3})^3} < \dfrac{1}{12}\cdot \left( \dfrac{1}{\sqrt n} - \dfrac{1}{\sqrt{n+3}} \right)$\\
        Do đó $A < \dfrac{1}{12}\left( \dfrac{1}{\sqrt{2}} - \dfrac{1}{\sqrt{5}} + \dfrac{1}{\sqrt{5}} - \dfrac{1}{\sqrt{8}} + \ldots + \dfrac{1}{\sqrt{95}} - \dfrac{1}{\sqrt{98}} \right)$\\
        $\Rightarrow A < \dfrac{1}{12}\left( \dfrac{1}{\sqrt{2}} - \dfrac{1}{7\sqrt{2}}\right)=\dfrac{1}{14\sqrt{2}} < \dfrac{1}{19}$.
    }
\end{bt}
