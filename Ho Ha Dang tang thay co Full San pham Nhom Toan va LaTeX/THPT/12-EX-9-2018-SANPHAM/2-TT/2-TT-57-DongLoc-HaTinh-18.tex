\begin{ex}%[TT lần 2, THPT Đồng Lộc, Hà Tĩnh 2018]%[Hữu Bình,12EX9]%[2D1G3-3]
	Tìm giá trị nhỏ nhất của $ A $ sao cho với mỗi tam thức bậc hai $ f(x) $ thỏa mãn điều kiện $ |f(x)| \le 1, \forall x \in [0;1] $	nghiệm đúng bất đẳng thức $ f'(0) \le A $.
	\choice
	{$ 1 $}
	{$ 2 $}
	{\True $ 8 $}
	{$ 4 $}
	\loigiai{
		Giả sử tồn tại tam thức bậc hai $ f(x)=ax^2+bx+c $ với $ a \ne 0 $, $ a,b,c \in \mathbb{R} $ thỏa mãn yêu cầu bài toán. Đặt $ M=\max \limits_{[0;1]}|f(x)| $. Theo giả thiết, suy ra $ M \le 1 $.\\
		Khi đó, ta có $ f(1)=a+b+c $, $ f(0)=c $, $ f\left(\dfrac{1}{2}\right)=\dfrac{a}{4}+\dfrac{b}{2}+c $ và
		$$8\ge 8M \ge |f(1)|+4\bigg|f\left(\dfrac{1}{2}\right)\bigg|+3|f(0)| \ge \bigg|f(1)-4f\left(\dfrac{1}{2}\right)+3f(0)\bigg|=|b|.$$
		Dấu đẳng thức xảy ra khi và chỉ khi $$\begin{cases}
		|f(1)|=\bigg|f\left(\dfrac{1}{2}\right)\bigg|=|f(0)|=1\\ f(1);-f\left(\dfrac{1}{2}\right);f(0) \text{ từng đôi một có tích không âm}
		\end{cases} \Leftrightarrow \hoac{&f(1)=-f\left(\dfrac{1}{2}\right)=f(0)=1 \hfill(1)\\&f(1)=-f\left(\dfrac{1}{2}\right)=f(0)=-1\hfill(2)}$$
		Xét (1), ta có $ \begin{cases}
		a+b+c=1\\ -\dfrac{a}{4}-\dfrac{b}{2}-c=1\\c=1
		\end{cases} \Leftrightarrow \begin{cases}
		a=8\\b=-8\\c=1.
		\end{cases}$\\
		Vậy, $ f(x)=8x^2-8x+1 $ thỏa mãn $ |f(x)| \le 1,\forall x\in[0;1] $. Do đó, $ A\ge f'(0)=-8$.\\
		Xét (2), ta có $ \begin{cases}
		a+b+c=-1\\ -\dfrac{a}{4}-\dfrac{b}{2}-c=-1\\c=-1
		\end{cases} \Leftrightarrow \begin{cases}
		a=-8\\b=8\\c=-1.
		\end{cases}$\\
		Vậy, $ f(x)=-8x^2+8x-1 $ thỏa mãn $ |f(x)| \le 1,\forall x\in[0;1] $. Do đó, $ A\ge f'(0)=8$.}	
\end{ex}
\begin{ex}%[TT lần 2, THPT Đồng Lộc, Hà Tĩnh 2018]%[Hữu Bình,12EX9]%[1D2G5-5]
	Có hai hộp đựng bi, mỗi viên bi chỉ mang một màu đen hoặc trắng. Lấy ngẫu nhiên từ mỗi hộp đúng $ 1 $ viên bi. Biết tổng số bi trong hai hộp là $ 20 $ và xác suất lấy được $ 2 $ viên bi đen là $ \dfrac{55}{84} $. Tính xác suất để lấy được $ 2 $ viên bi trắng?
	\choice
	{\True $ \dfrac{1}{28} $}
	{$ \dfrac{23}{84} $}
	{$ \dfrac{3}{28} $}
	{$ \dfrac{13}{84} $}
	\loigiai{
		Gọi hộp $ 1 $ có $ a $ viên bi trắng, $ b $ viên bi đen; hộp $ 2 $ có $ c $ viên bi trắng, $ d $ viên bi đen với $ a,b,c,d $ là các số nguyên dương.\\
		Theo giả thiết, ta có $ (a+b)+(c+d)=20 $ và $ \dfrac{bd}{(a+b)\cdot(c+d)}=\dfrac{55}{84} $.\\
		Ta có $ 100=\left[\dfrac{(a+b)+(c+d)}{2}\right]^2 \ge (a+b)(c+d) \Rightarrow (a+b)(c+d)=84$.\\
		Khi đó, ta có $$\begin{cases}
		(a+b)+(c+d)=20\\(a+b)(c+d)=84
		\end{cases} \Rightarrow \begin{cases}
		a+b=14\\ c+d=6
		\end{cases} \text{ hoặc } \begin{cases}
		a+b=6\\c+d=14.
		\end{cases}$$
		Không giảm tổng quát, xét $ a+b=14 $, $ c+d=6 $.\\
		Lại có $ bd=55=5\cdot 11 $, chọn $ b=11; d= 5 \Rightarrow a= 3; c=1$.\\
		Xác suất để lấy được hai viên bi trắng là $ \mathrm{P}=\dfrac{ac}{(a+b)(c+d)}= \dfrac{3}{84}=\dfrac{1}{28}$. }	
\end{ex}
