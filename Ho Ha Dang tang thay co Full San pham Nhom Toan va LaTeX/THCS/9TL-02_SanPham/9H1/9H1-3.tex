
\section{HỆ THỨC VỀ CẠNH VÀ GÓC TRONG TAM GIÁC VUÔNG}
\subsection{Kiến thức cần nhớ}
\begin{dl}
	Trong tam giác vuông, mỗi cạnh góc vuông bằng:
	\begin{listEX}
		\item Cạnh huyền nhân với $\pmb\sin$ góc đối hoặc nhân với $\pmb\cos$ góc kề;
		\item Cạnh góc vuông kia nhân với $\pmb\tan$ góc đối hoặc nhân với $\pmb\cot$ góc kề.
	\end{listEX}
\end{dl}   
\immini{
	Vậy, trong tam giác $ABC$ vuông tại $A$, ta có các hệ thức
	\begin{itemize}
		\item $b = a \cdot \sin B = a \cdot \cos C$.
		\item $b = c \cdot \tan B = c \cdot \cot C$.
		\item $c = a \cdot \sin C = a \cdot \cos B$.
		\item $c = b \cdot \tan C = b \cdot \cot B$.
\end{itemize}}
{
	\begin{tikzpicture}[scale=0.9]
	\clip (-0.5,-0.5) rectangle (3.6,4.2);
	\tkzDefPoints{0/0/A, 3/0/B, 0/4/C}
	\tkzDrawPoints[size=5,fill=black](A,B,C)
	\tkzLabelPoints[left](A)
	\tkzLabelPoints[right](B,C)
	\tkzDrawSegments(A,B B,C A,C)
	\tkzMarkRightAngle[size=0.2](B,A,C)
	\tkzLabelSegment[below](A,B){$c$}
	\tkzLabelSegment[left](A,C){$b$}
	\tkzLabelSegment[right](B,C){$a$}
	\end{tikzpicture}
}
\subsection{Các dạng toán}
\begin{dang}{Giải tam giác vuông}
	Sử dụng mối quan hệ giữa cạnh và góc trong tam giác vuông để giải.
\end{dang}
\begin{vd}
	Cho tam giác $ABC$ với các cạnh góc vuông $AB = 5$, $AC = 8$. Hãy giải tam giác vuông $ABC$.
	\loigiai{
		\immini{
			Theo định lí Py-ta-go, ta có 
			\begin{eqnarray*}
				&BC&=\sqrt{AB^2+AC^2}\\
				&  &=\sqrt{5^2+8^2}\\
				&  &\approx 9{,}43.
			\end{eqnarray*}   
			Mặt khác $\tan C = \dfrac{AB}{AC} = \dfrac{5}{8} = 0{,}625$. \\
			Tra bảng hay dùng máy tính bỏ túi, ta tìm được $\widehat{C} \approx 32^\circ$. \\
			Do đó $\widehat{B} \approx 90^\circ - 32^\circ = 58^\circ$.}
		{
			\begin{tikzpicture}[scale=0.9]
			\clip (-0.5,-0.5) rectangle (3.6,4.2);
			\tkzDefPoints{0/0/A, 3/0/B, 0/4/C}

			\tkzLabelPoints[left](A)
			\tkzLabelPoints[right](B,C)
			\tkzDrawSegments(A,B B,C A,C)
			\tkzMarkRightAngle[size=0.2](B,A,C)
			\tkzLabelSegment[above](A,B){$5$}
			\tkzLabelSegment[right](A,C){$8$}
			\end{tikzpicture}	
		}
	}
\end{vd}

\begin{vd}

	\begin{listEX}[3]
		\item $AC$.
		\item $BC$.
		\item Phân giác $BD$.

\begin{vd}
	Cho tam giác $ABC$, trong đó $BC = 11$ cm, $\widehat{ABC} = 38^\circ$, $\widehat{ACB} = 30^\circ$. Gọi điểm $N$ là chân của đường vuông góc kẻ từ $A$ đến cạnh $BC$. Hãy tính độ dài đoạn thẳng $AN$.
	\loigiai{
		\immini{
			Ta có $\tan 38^\circ = \dfrac{AN}{BN} \Rightarrow BN = \dfrac{AN}{\tan 38^\circ}$.\\ Tương tự $NC = \dfrac{AN}{\tan 30^\circ}$. Khi đó, ta có
			\begin{eqnarray*}
				&&BC = BN + NC\\
				&\Leftrightarrow & 11 = \dfrac{AN}{\tan 38^\circ} + \dfrac{AN}{\tan 30^\circ}\\
				&\Leftrightarrow & 11 = AN \cdot \left(\dfrac{1}{\tan 38^\circ} + \dfrac{1}{\tan 30^\circ}\right)\\
				&\Rightarrow& AN = \dfrac{11}{\dfrac{1}{\tan 38^\circ} + \dfrac{1}{\tan 30^\circ}} \approx 3{,}65.
		\end{eqnarray*}}

			\tkzInterLL(B,E)(C,F)\tkzGetPoint{A}
			\tkzDefPointBy[projection=onto B--C](A) \tkzGetPoint{N}
			\tkzDrawPoints(A,B,C,N)
			\draw (A)--(B)--(C)--(A) (A)--(N);
			\tkzLabelPoints[below](B,N,C)
			\tkzLabelPoints[above](A)
			\tkzLabelAngles[pos=0.8,rotate=0](C,B,A){$38^\circ$}

\begin{vd}
	Cho tam giác $ABC$ có $BC = 6$ cm, $\widehat{B}= 60^\circ$, $\widehat{C} = 40^\circ$. Hãy tính
	\begin{listEX}[2]
		\item Chiều cao $CH$ và cạnh $AC$.
		\item Diện tích tam giác $ABC$.
	\end{listEX}
\loigiai{
	\immini{
		\begin{enumerate}
			\item Tam giác $BHC$ vuông tại $H$:\\
			$\sin \widehat{HBC} = \dfrac{CH}{BC} \Rightarrow CH = BC \cdot \sin\widehat{HBC} = 6 \sin 60^\circ = 3 \sqrt{3}$. \\
			Mà $\widehat{CAB} = 180^\circ - 40^\circ - 60^\circ = 80^\circ$.\\
			Tam giác $AHC$ vuông tại $H$:\\ $\sin \widehat{CAH} = \dfrac{CH}{AC} \Rightarrow AC = \dfrac{3 \sqrt{3}}{\sin 80^\circ} \approx 5,28$ cm.
		\end{enumerate}
			}
	{
	\begin{tikzpicture}[scale=0.8]
	\tkzDefPoints{0/0/C, 5/0/B, 3.5/3/A}
	\draw (A)--(B)--(C)--(A);
	\tkzDrawAltitude(A,B)(C) \tkzGetPoint{H}
	\tkzDrawPoints(A,B,C,H)
	\tkzLabelPoints[below](B,C)
	\tkzLabelPoints[above right](A,H)
	\tkzMarkAngles[size=1cm,opacity=.4,draw=black,mark=||,mksize=2](A,B,C)
	\tkzMarkAngles[size=1cm,opacity=.4,draw=black,mksize=2](B,C,A)
	\tkzLabelAngles[pos=1.1,rotate=0](B,C,A){$40^\circ$}
	\tkzLabelAngles[pos=0.7,rotate=0](C,B,A){$60^\circ$}
	\tkzMarkRightAngles[size=.2,opacity=.4,draw=black](C,H,A)
	\tkzLabelSegment[below right](B,C){$6$ cm}
	\end{tikzpicture}	
}	
		\begin{enumerate}
			\item[b)]
			Ta có $\tan \widehat{CAH} = \dfrac{CH}{AH} \Rightarrow AH = \dfrac{CH}{\tan 80^\circ} = \dfrac{3\sqrt{3}}{\tan 80^\circ}$. \\
			Do vậy $\tan \widehat{HBC} = \dfrac{CH}{HB} \Rightarrow HB = \dfrac{CH}{\tan 60^\circ} = 3$ cm.				
	\end{enumerate}	
		Ta có $S_{ABC} = \dfrac{1}{2} \cdot CH \cdot AB = \dfrac{1}{2} \cdot CH \cdot \left(AH + HB\right) = \dfrac{1}{2} \cdot 3\sqrt{3} \cdot \left(\dfrac{3\sqrt{3}}{\tan 80^\circ} + 3\right) \approx 10{,}17$ cm$^2$.		
	}
	
\end{vd}
