\begin{ex}%[TT, Chuyên Lê Quý Đôn, Lai Châu, 2018]%[2D3K2-1]%[Nguyễn Tiến Thùy, 12EX-8]

\begin{ex}%[TT, Chuyên Lê Quý Đôn, Lai Châu, 2018]%[2D1G1-4]%[Nguyễn Tiến Thùy, 12EX-8]
Có bao nhiêu giá trị nguyên $m$ để phương trình
$$\sin x\cdot\!\!\sqrt[2018]{2019-\cos^2x}-(\cos x+m)\cdot\!\!\sqrt[2018]{2019-\sin^2x+m^2+2m\cos x}=\cos x-\sin x+m$$
có nghiệm thực.
\choice
{$1$}
{\True $3$}
{$2$}
{$0$}
\loigiai{
Ta có phương trình
$$\sin x\sqrt[2018]{2018+\sin^2x}+\sin x=(\cos x+m)\sqrt[2018]{2018+(\cos x+m)^2}+(\cos x+m).\, (*)$$
Xét hàm số $f(t)=t\sqrt[2018]{2018+t^2}+t$ có
$$f'(t)=\sqrt[2018]{2018+t^2}+\dfrac{2t^2}{2018\sqrt[2018]{(2018+t^2)^{2017}}}+1>0,\, \forall t\in\mathrm{R}.$$
Từ đó, kết hợp với $(*)$ ta có 
$$f(\sin x)=f(\cos x+m)\Leftrightarrow \sin x-\cos x=m.\, (**)$$
Để $(**)$ có nghiệm khi và chỉ khi 
$$1^2+(-1)^2\ge m^2\Leftarrow m\le 2\Leftrightarrow -\sqrt{2}\le m\le \sqrt{2}.$$
Vậy có $3$ giá trị nguyên của $m$ để phương trình có nghiệm.
}
\end{ex}

\begin{ex}%[TT, Chuyên Lê Quý Đôn, Lai Châu, 2018]%[2D3G2-4]%[Nguyễn Tiến Thùy, 12EX-8]
Cho hai hàm số $f(x)$ và $g(x)$ có đạo hàm trên $[1;4]$ và thỏa mãn hệ thức sau với mọi $x\in[1;4]$
$$\heva{&f(1)=2g(1)=2\\ &f'(x)=\dfrac{1}{x\sqrt{x}}\cdot\dfrac{1}{g(x)}\\ &g'(x)=-\dfrac{2}{x\sqrt{x}}\cdot\dfrac{1}{f(x)}.
}$$
Tính $I=\displaystyle\int\limits_1^4[f(x)g(x)]\mathrm{\,d}x$.
\choice
{$I=4\ln2$}
{\True $I=4$}
{$I=2\ln2$}
{$I=2$}
\loigiai{
Theo bài ra ta có 
$$[f(x)g(x)]'=f'(x)g(x)+g'(x)f(x)=\dfrac{1}{x\sqrt{x}}-\dfrac{2}{x\sqrt{x}}=-\dfrac{1}{x\sqrt{x}}$$
$$\Rightarrow f(x)g(x)=-\displaystyle\int\dfrac{1}{x\sqrt{x}}\mathrm{\,d}x=\dfrac{2}{\sqrt{x}}+C.$$
Kết hợp với giả thiết ta có
$$f(1)g(1)=2=\dfrac{2}{\sqrt{1}}+C\Rightarrow C=0.$$
Từ đó suy ra
$$I=\displaystyle\int\limits_1^4[f(x)g(x)]\mathrm{\,d}x=\displaystyle\int\limits_1^4\dfrac{2}{\sqrt{x}}\mathrm{\,d}x=4.$$
}
\end{ex}
