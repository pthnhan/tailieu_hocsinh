\section{Cực trị căn thức}

%Biên soạn: Thịnh Trần

\begin{bt}%[Trần Đức Thịnh]
	Tìm giá trị lớn nhất của biểu thức $A=\sqrt{x-1}+\sqrt{y-2}$ biết $x+y=4$.
\loigiai{
	Điều kiện xác định của biểu thức là $\heva{&x\geq 1\\&y\leq 2.}$\\
{\bf Cách 1.}
	\begin{note}
	Bất đẳng thức Cô-si cho phép làm giảm một tổng $\dfrac{a+b}{2}\geq \sqrt{ab}$ với $a$, $b$ không âm.\\
	Ở đây ta muốn làm tăng một tổng. Ta dùng bất đẳng thức Bunhiacopxki
	$a+b\leq \sqrt{2\left(a^2+b^2\right)} .$
	\end{note}
	\noindent Ta có $A=\sqrt{x-1}+\sqrt{y-2}\leq \sqrt{2(x-1+y-2)}=\sqrt{2}$.\\
	Vậy $\max A=\sqrt{2}$ khi và chỉ khi $\heva{&x-1=y-2\\&x+y=4}\Leftrightarrow\heva{&x=1{,}5\\&y=2{,}5.}$\\
{\bf Cách 2.}\\
	Ta có $A\geq 0$ và
	\begin{eqnarray*}
		A^2&=&x+y-3+2\sqrt{(x-1)(x-2)}\\
		&=&1+2\sqrt{(x-1)(y-2)}\\
		&\leq& 1+(x-1)+(y-2)=2\\
		\Rightarrow A&\leq& \sqrt{2}
	\end{eqnarray*}
	Vậy $\max A=\sqrt{2}$ khi và chỉ khi $\heva{&x-1=y-2\\&x+y=4}\Leftrightarrow\heva{&x=1{,}5\\&y=2{,}5.}$\\
{\bf Cách 3.}\\	
	Áp dụng bất đẳng thức $\dfrac{\sqrt{a}+\sqrt{b}}{2}\leq \sqrt{\dfrac{a+b}{2}}$, ta có:
	$$\dfrac{A}{2}\leq \sqrt{\dfrac{x-1+y-2}{2}}=\sqrt{\dfrac{1}{2}}.$$
	Do đó $A\leq \sqrt{2}.$\\
	Vậy $\max A=\sqrt{2}$ khi và chỉ khi $\heva{&x-1=y-2\\&x+y=4}\Leftrightarrow\heva{&x=1{,}5\\&y=2{,}5.}$
	}
\end{bt}

\begin{bt}
	Tìm giá trị lớn nhất của biểu thức $A=3-2x+\sqrt{5-x^2+4x}$.
\loigiai{
	Biểu thức $A$ có nghĩa khi $5-x^2+4x\geq 0\Leftrightarrow (x+1)(x-5)\leq 0\Leftrightarrow -1\leq x\leq 5$.\\
	Ta có $A=3-2x+\sqrt{5-x^2+4x}\Leftrightarrow A+2x-3=\sqrt{5-x^2+4x}$. \hfill $(*)$\\
	Điều kiện $A+2x-3\geq 0$.\hfill $(1)$\\
	Khi đó $(*)\Leftrightarrow (A+2x-3)^2=5-x^2+4x\Leftrightarrow (5x+2A-8)^2+(A^2+2A-44)=0$.\\
	Vì $(5x+2A-8)^2\geq 0$ suy ra 
	\begin{eqnarray*}
	A^2+2A-44\leq 0&\Leftrightarrow& (A+1+3\sqrt{5})(A-1+3\sqrt{5})\leq 0\\
	&\Leftrightarrow& -1-3\sqrt{5}\leq A\leq -1+3\sqrt{5}
	\end{eqnarray*}
	Vậy $\max A=3\sqrt{5}-1\Leftrightarrow x=\dfrac{8-24}{5}=2-\dfrac{6\sqrt{5}}{2}$ (thỏa mãn điều kiện có nghĩa và $(1)$).
	\begin{note}
		Khi cho học sinh giải bài này một số em đã cho lời giải sai như sau:\\
		Biểu thức $A$ có nghĩa khi $5-x^2+4x\geq 0\Leftrightarrow -1\leq x\leq 5$.\\
		Theo bất đẳng thức Cô-si ta có
		$$ A=3-2x+\sqrt{1(5-x^2-4x)}\leq 3-2x+\dfrac{1+5-x^2+4x}{2}=6-\dfrac{x^2}{2}\leq 6.$$\\
		Vậy $\max A=6\Leftrightarrow x=0$ (thỏa mãn điều kiện xác định).\\
		Sai lầm ở đây là dấu bằng xảy ra $\Leftrightarrow\heva{&5-x^2+4x=1\\&x=0}$. Nhưng hệ này vô nghiệm.	
	\end{note}	
}
\end{bt}

\begin{bt}
	Giả thiết $-1\leq x\leq 1$. Tìm giá trị lớn nhất và giá trị nhỏ nhất của biểu thức
	$$A=\dfrac{\sqrt{1+\sqrt{1-x^2}}\cdot \left(\sqrt{(1+x)^3}-\sqrt{(1-x)^3}\right)}{2+\sqrt{1-x^2}} .$$
\loigiai{
	Đặt $u=\sqrt{1+x}$, $v=\sqrt{1-x}$. Khi đó $u^2+v^2=2$ và biến đổi
	$$A=\dfrac{\sqrt{1+uv}(u^3-v^3)}{u^2+v^2+uv}=\sqrt{1+uv}\cdot (u-v)=\sqrt{\dfrac{2+2uv}{2}}\cdot (u-v) .$$
	Thay $2=u^2+v^2$ được $A=\dfrac{(u+v)(u-v)}{\sqrt{2}}=\dfrac{u^2-v^2}{\sqrt{2}}=x\sqrt{2}.$\\
	Vậy $\max A=\sqrt{2}$ khi $x=1$,\, $\max A=-\sqrt{2}$ khi $x=-1$.
}
\end{bt}
\begin{bt}
	Giả sử $x\geq -\dfrac{3}{2}$. Tìm giá trị nhỏ nhất của biểu thức sau $$A=\sqrt{x^2+2x+\dfrac{3}{4}+\sqrt{x^2+2x+\dfrac{3}{4}+\sqrt{x^2+3x+\dfrac{9}{4}}}}.$$
\loigiai{
	Vì $\sqrt{x^2+3x+\dfrac{9}{4}}=\sqrt{\left(x+\dfrac{3}{2}\right)^2}=\left|x+\dfrac{3}{2}\right|$ và $x\geq -\dfrac{3}{2}$ nên ta có
	$$x^2+2x+\dfrac{3}{4}+\sqrt{x^2+3x+\dfrac{9}{4}}=x^2+2x+\dfrac{3}{4}+\left|x+\dfrac{3}{2} \right|=x^2+3x+\dfrac{9}{4} .$$
	Do đó $A=\sqrt{x^2+3x+\dfrac{9}{4}}=x+\dfrac{3}{2}\geq 0$ khi $x\geq -\dfrac{3}{2}$.\\
	Vậy $\min A=0$ khi $x=-\dfrac{3}{2}$.
}
\end{bt}

\begin{bt}
	Tìm giá trị nhỏ nhất của biểu thức $A=\sqrt{x^2+1}+\sqrt{x^2-2x+5}$.
\loigiai{
	Áp dụng bất đẳng thức $\sqrt{a^2+b^2}+\sqrt{c^2+d^2}\geq \sqrt{(a+c)^2+(b+d)^2}$, ta có
	$$A=\sqrt{x^2+1^2}+\sqrt{(1-x)^2+2^2}\geq \sqrt{(x+1-x)^2+(1+2)^2}=\sqrt{10} .$$
	Vậy $\min A=\sqrt{10}$ khi và chỉ khi $\dfrac{1-x}{x}=2\Leftrightarrow x=\dfrac{1}{3}$.
}
\end{bt}

\begin{bt}
	Tìm giá trị lớn nhất, giá trị nhỏ nhất của biểu thức $A=2x+\sqrt{5-x^2}$.
\loigiai{
	Điều kiện $x^2\leq 5$.
	\begin{itemize}
	\item 
	Áp dụng bất đẳng thức Bu-nhi-a-cốp-xki được
	$$A^2=\left(2x+1\cdot\sqrt{5-x^2}\right)^2\leq (2^2+1^2)(x^2+5-x^2)=25\Rightarrow A^2\leq 25.$$
	$A^2=25\Leftrightarrow \heva{&\dfrac{x}{2}=\sqrt{5-x^2}\\&x^2\leq 5}\Leftrightarrow\heva{&x\geq 0\\&x^2=4(5-x^2)\\&x^2\leq 5}\Leftrightarrow x=2$.\\
	Với $x=2$ thì $A=5$. Vậy $\max A=5$ khi $x=2$.
	\item 
	Chú ý rằng tuy từ $A^2\leq 25\Rightarrow -\sqrt{5}\leq x\leq \sqrt{5}$ nhưng không xảy ra $A=-5$.\\
	Do điều kiện xác định của $A$, ta có $x^2\leq 5\Rightarrow -\sqrt{5}\leq x\leq\sqrt{5}$.\\
	Do đó $2x\geq -2\sqrt{5}$. Ta lại có $\sqrt{5-x^2}\geq 0$ nên $A=2x+\sqrt{5-x^2}\geq -2\sqrt{5}$.\\
	Suy ra $\min A=-2\sqrt{5}$ khi $x=-\sqrt{5}$.
	\end{itemize}
}
\end{bt}

\begin{bt}%[Trần Đức Thịnh]
	Tìm giá trị nhỏ nhất của biểu thức $A=\sqrt{x^2-x+1}+\sqrt{x^2+x+1}$.	
	\loigiai{
		Biểu thức xác định với mọi $x\in\mathbb{R}$.\\
		Ta có $x^2-x+1=\left(x-\dfrac{1}{2}\right)^2+\dfrac{3}{4}\geq \dfrac{3}{4}$ với mọi $x$\\
		Ta có $x^2+x+1=\left(x+\dfrac{1}{2}\right)^2+\dfrac{3}{4}\geq \dfrac{3}{4}$ với mọi $x$.\\	
		Áp dụng bất đẳng thức Cô-si cho hai số dương $\sqrt{x^2-x+1}$, $\sqrt{x^2+x+1}$ ta có\\
		$$\sqrt{x^2-x+1}+\sqrt{x^2+x+1}\geq 2\sqrt{\sqrt{x^2-x+1}\cdot \sqrt{x^2+x+1}}=2\sqrt[4]{x^4+x^2+1}\geq 2.$$\\
		Vậy $\min A=2$ khi và chỉ khi $\heva{&x^4+x^2+1=1\\&\sqrt{x^2-x+1}=\sqrt{x^2+x+1}}\Leftrightarrow x=0$ (thỏa mãn).
	}
\end{bt}

\begin{bt}%[Trần Đức Thịnh]
	Cho $x\geq-\dfrac{1}{2}$. Tìm giá trị lớn nhất của biểu thức $A=\sqrt{2x^2+5x+2}+2\sqrt{x+3}-2x$.
	\loigiai{
		Ta có 
		$A=\sqrt{2x^2+5x+2}+2\sqrt{x+3}-2x=\sqrt{(2x+1)(x+2)}+2\sqrt{x+3}-2x$.\\
		Với $x\geq -\dfrac{1}{2}$ ta có $\heva{&2x+1\geq 0\\&x+2>0}$, theo bất đẳng thức Cô-si ta có $$\dfrac{2x+1+x+2}{2}\geq \sqrt{(2x+1)(x+2)}\Leftrightarrow \dfrac{3x+3}{2}\geq \sqrt{(2x+1)(x+2)}.$$
		Dấu bằng xảy ra khi $2x+1=x+2\Leftrightarrow x=1$.\\
		Áp dụng bất đẳng thức Cô-si cho hai số dương $x+3$ và $4$ ta có
		$$\dfrac{x+3+4}{2}\geq \sqrt{4(x+3)}\Leftrightarrow \dfrac{x+7}{2}\geq 2\sqrt{x+3} .$$
		Dấu bằng xáy ra khi $x+3=4\Leftrightarrow x=1$.\\
		Do đó $A\leq \dfrac{x+7}{2}+\dfrac{3x+3}{2}-2x=5$.\\
		Vậy $\max A=5$ khi và chỉ khi $x=1$.
	}
\end{bt}

\begin{bt}%[Trần Đức Thịnh]
\begin{enumerate}
	\item Tìm giá trị nhỏ nhất của biểu thức $A=\sqrt{x^2-2x+1}+\sqrt{x^2-6x+9}$.
	\item Tìm giá trị lớn nhất của biểu thức $A=\sqrt{a+3-4\sqrt{a-1}}-\sqrt{a+15-8\sqrt{a-1}}$.
\end{enumerate}	
\loigiai{
	\begin{enumerate}
	\item
	Biểu thức xác định với mọi $x\in\mathbb{R}$. Ta có
	\begin{eqnarray*}
		A&=&\sqrt{x^2-2x+1}+\sqrt{x^2-6x+9}\\
		&=&\sqrt{(x-1)^2}+\sqrt{(x-3)^2}\\
		&=&|x-1|+|x-3|\\
		&=&|x-1|+|3-x|\\
		&\geq&|x-1+3-x|=2.
	\end{eqnarray*}	
	Dấu bằng xảy ra khi $x-1$ và $3-x$ cùng dấu $\Leftrightarrow (x-1)(3-x)\geq 0\Leftrightarrow 1\leq x\leq 3$.\\
	Vậy $\min A=2$ khi và chỉ khi $1\leq x\leq 3.$
	\item
	Điều kiện xác định của biểu thức là $a\geq 1$. Với $a\geq 1$ ta có\\
		\begin{eqnarray*}
			A&=&\sqrt{a+3-4\sqrt{a-1}}-\sqrt{a+15-8\sqrt{a-1}}\\
			&=&\sqrt{(\sqrt{a-1}-2)^2}-\sqrt{(\sqrt{a-1}-4)^2}\\
			&=&|\sqrt{a-1}-2|-|\sqrt{a-1}-4|\\
			&\leq&|\sqrt{a-1}-2-\sqrt{a-1}+4|=2.
		\end{eqnarray*}	
		Dấu bằng xảy ra khi $\sqrt{a-1}-2$ và $\sqrt{a-1}-4$ cùng dấu\\
		$\Leftrightarrow (\sqrt{a-1}-2)(\sqrt{a-1}-4)\geq 0\Leftrightarrow \hoac{&\sqrt{a-1}\leq 2\\ &\sqrt{a-1}\geq 4}\Leftrightarrow\hoac{&a\leq 5\\&a\geq 17} $.\\
		Vậy $\max A=2$ khi và chỉ khi $1\leq a\leq 5$ hoặc $a\geq 17$.
	\end{enumerate}
	}
\end{bt}

\begin{bt}%[Trần Đức Thịnh]
	Tìm giá trị lớn nhất của biểu thức $A=\sqrt{3x-5}+\sqrt{7-3x}$.
\loigiai{
	Điều kiện xác định của biểu thức $\heva{&3x-5\geq 0\\&7-3x\geq 0}\Leftrightarrow \dfrac{5}{3}\leq x\leq \dfrac{7}{3}$.\\
	Bình phương hai vế ta có $A^2=2+2\sqrt{(3x-5)(7-3x)}$.\\
	Với $\dfrac{5}{3}\leq x\leq \dfrac{7}{3}$, áp dụng bất đẳng thức Cô-si cho $(3x-5)$ và $(7-3x)$ ta có\\
	$$(3x-5)+(7-3x)\geq 2\sqrt{(3x-5)(7-3x)}\Leftrightarrow 2\geq 2\sqrt{(3x-5)(7-3x)} .$$	
	Do đó $A^2\leq 4\Rightarrow A\leq 2$.\\
	Dâu bằng xảy ra khi và chỉ khi $3x-5=7-3x\Leftrightarrow x=2$.\\
	Vậy $\max A=2\Leftrightarrow x=2$.
	}
\end{bt}

\begin{bt}%[Trần Đức Thịnh]
	Tìm giá trị nhỏ nhất của biểu thức $A=\sqrt{-x^2+2x+8}-\sqrt{-x^2+x+2}$.
\loigiai{
	Điều kiện xác định của biểu thức $$\heva{&-x^2+2x+8\geq 0\\&-x^2+x+2\geq 0}\Leftrightarrow\heva{&(x+2)(x-4)\leq 0\\&(x+1)(x-2)\leq 0}\Leftrightarrow \heva{&-2\leq x\leq 4\\&-1\leq x\leq 2}\Leftrightarrow -1\leq x\leq 2.$$
	Khi đó 	$(-x^2+2x+8)-(-x^2+x+2)=x+6>0\Rightarrow A>0$.\\
	Bình phương hai vế ta có
	\begin{eqnarray*}
		A^2&=&(-x^2+2x+8)+(-x^2+x+2)-2\sqrt{-x^2+2x+8}\sqrt{-x^2+x+2}\\
		&=&-2x^2+3x+10-2\sqrt{(x+2)(4-x)(x+1)(2-x)}\\
		&=&(2-x)(x+2)+(x+1)(4-x)+2-2\sqrt{(2-x)(x+2)\cdot (x+1)(4-x)}\\
		&=&\left(\sqrt{4-x^2}\right)^2-2\sqrt{(2-x)(x+2)\cdot (x+1)(4-x)}+\left(\sqrt{(x+1)(4-x)}\right)^2+2\\
		&=&\left(\sqrt{4-x^2}-\sqrt{(x+1)(4-x)}\right)^2+2\geq 2\\
		\Rightarrow A&\geq&\sqrt{2}
	\end{eqnarray*}
	Vậy $\min A=\sqrt{2}\Leftrightarrow 4-x^2=(x+1)(4-x)\Leftrightarrow x=0$.
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Tìm giá trị nhỏ nhất, giá trị lớn nhất của biểu thức $A=\dfrac{1}{3-\sqrt{8-x^2}}.$
	\loigiai{
		Điều kiện: $|x|\leq 2\sqrt{2}$.\\
		Ta có 
		\allowdisplaybreaks{
			\begin{eqnarray*}
				0\leq \sqrt{8-x^2}\leq \sqrt{8}=2\sqrt{2} & \Rightarrow & 0 \geq -\sqrt{8-x^2} \geq -2\sqrt{2}\\
				& \Rightarrow & 3 \geq 3-\sqrt{8-x^2}\geq 3-2\sqrt{2} \\
				& \Rightarrow & \dfrac{1}{3} \leq \dfrac{1}{3-\sqrt{8-x^2}} \leq \dfrac{1}{3-2\sqrt{2}} \\
				& \Rightarrow & \dfrac{1}{3} \leq A \leq 3+2\sqrt{2}.
			\end{eqnarray*}
		}
		\begin{itemize}
			\item $A=\dfrac{1}{3}\Leftrightarrow \sqrt{8-x^2}=0\Leftrightarrow x=\pm 2\sqrt{2}$.
			\item $A=3+2\sqrt{2}\Leftrightarrow \sqrt{8-x^2}=\sqrt{8}\Leftrightarrow x=0$.
		\end{itemize}
		Vậy giá trị nhỏ nhất của $A$ là $\dfrac{1}{3}$ khi $x=\pm 2\sqrt{2}$; giá trị lớn nhất của $A$ là $3+2\sqrt{2}$ khi $x=0$.
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Tìm giá trị lớn nhất của biểu thức $A=\sqrt{x-2}+\sqrt{4-x}$.
	\loigiai{
		Điều kiện xác định: $2 \leq x \leq 4$.\\
		Ta có $A^2=(1\cdot \sqrt{x-2}+1\cdot \sqrt{4-x})^2 \leq (1^2+1^2)(x-2+4-x)=4$.\\
		Suy ra $0<A\leq 2$.\\
		Dấu bằng xảy ra khi và chỉ khi $\sqrt{x-2}=\sqrt{4-x}\Leftrightarrow x=3$.\\
		Vậy giá trị lớn nhất của $A$ là $2$ khi $x=3$.
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Tìm giá trị nhỏ nhất của biểu thức $P=x+\sqrt{x}+1$.
	\loigiai{
		Điều kiện: $x\geq 0$.\\
		Ta có $x\geq 0\Rightarrow x+\sqrt{x}\geq 0\Rightarrow x+\sqrt{x}+1\geq 1$.\\
		Dấu bằng xảy ra khi và chỉ khi $x=0$.\\
		Vậy giá trị nhỏ nhất của $P$ là $1$ khi $x=0$.
		\begin{note}[Lời giải sai]
			$P=\left(\sqrt{x}+\dfrac{1}{2}\right)^2+\dfrac{3}{4}\geq \dfrac{3}{4}$.\\
			Đến đây kết luận giá trị nhỏ nhất của $P$ là $\dfrac{3}{4}$ là sai vì khi đó không có giá trị nào của $x$ để dấu bằng xảy ra vì phương trình $\sqrt{x}+\dfrac{1}{2}=0$ vô nghiệm.
		\end{note}
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Tìm giá trị nhỏ nhất của biểu thức $M=\dfrac{5-3x}{\sqrt{1-x^2}}$.
	\loigiai{
		Điều kiện xác định: $-1 < x < 1$. Khi đó $5-3x>0$, do đó $M>0$. Ta có $$M^2=\dfrac{(5-3x)^2}{1-x^2}=\dfrac{25-30x+9x^2}{1-x^2}=\dfrac{9-30x+25x^2+16-16x^2}{1-x^2}=\dfrac{(3-5x)^2}{1-x^2}+16\geq 16.$$
		Dấu bằng xảy ra khi và chỉ khi $3-5x=0\Leftrightarrow x=\dfrac{3}{5}$.\\
		Vậy giá trị nhỏ nhất của $M$ là $16$ khi $x=\dfrac{3}{5}$.
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Tìm giá trị nhỏ nhất của biểu thức $M=\dfrac{\sqrt{x+1}}{\sqrt{x}+2}$.
	\loigiai{
		Điều kiện $x\geq 0$. Ta có
		\allowdisplaybreaks{
			\begin{eqnarray*}
				M^2 & = & \dfrac{x+1}{(\sqrt{x}+2)^2}=\dfrac{5x+5}{5(\sqrt{x}+2)^2}\\
				&=& \dfrac{(4x-4\sqrt{x}+1)+(x+4\sqrt{x}+4)} {5(\sqrt{x}+2)^2}\\
				&=& \dfrac{(2\sqrt{x}-1)^2+(\sqrt{x}+2)^2}
				{5(\sqrt{x}+2)^2}\\
				& = & \dfrac{(2\sqrt{x}-1)^2}{5(\sqrt{x}+2)^2} +\dfrac{1}{5}\\
				& \geq & \dfrac{1}{5}.
			\end{eqnarray*}
		}
		Mà $M>0$ nên $M\geq \dfrac{\sqrt{5}}{5}$.\\
		Dấu bằng xảy ra khi và chỉ khi $2\sqrt{x}-1=0\Leftrightarrow x=\dfrac{1}{4}$.\\
		Vậy giá trị nhỏ nhất của $M$ là $\dfrac{\sqrt{5}}{5}$ khi $x=\dfrac{1}{4}$.
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Tìm giá trị lớn nhất của biểu thức $M=\dfrac{\sqrt{x-1}}{x}+\dfrac{\sqrt{y-4}}{y}$.
	\loigiai{
		Điều kiện: $x\geq 1$, $y\geq 4$.\\
		Ta có $$\heva{
			& \dfrac{\sqrt{x-1}}{x}=\dfrac{\sqrt{1 \cdot (x-1)}}{x} \leq \dfrac{1+(x-1)}{2x}=\dfrac{1}{2} \\
			& \dfrac{\sqrt{y-4}}{y}=\dfrac{\sqrt{4 \cdot (y-4)}}{2y} \leq \dfrac{4+(y-4)}{4y}=\dfrac{1}{4}.}$$
		Suy ra $M\leq \dfrac{1}{2}+\dfrac{1}{4}=\dfrac{3}{4}$.\\
		Dấu bằng xảy ra khi và chỉ khi $\heva{& x-1=1\\ & y-4=4}\Leftrightarrow \heva{& x=2\\ & y=8}$ (thỏa mãn điều kiện).\\
		Vậy giá trị lớn nhất của $M$ là $\dfrac{3}{4}$ khi $x=2$, $y=8$.
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Cho $a\geq 4$, $b\geq 9$. Tìm giá trị nhỏ nhất và giá trị lớn nhất của biểu thức $$N=\dfrac{b\sqrt{a-4}+a\sqrt{b-9}}{ab}.$$
	\loigiai{
		Ta có $N\geq 0$. Dấu bằng xảy ra khi và chỉ khi $a=4, b=9$.\\
		Vậy giá trị nhỏ nhất của $N$ là $0$ khi $a=4$, $b=9$.\\
		Ta có $N=\dfrac{\sqrt{a-4}}{a}+\dfrac{\sqrt{b-9}}{b}$.\\
		Mà $a=(a-4)+4\geq 2\sqrt{4(a-4)}=4\sqrt{a-4}$ suy ra $\dfrac{\sqrt{a-4}}{a}\leq \dfrac{1}{4}$.\\
		Dấu bằng xảy ra khi và chỉ khi $a-4=4\Leftrightarrow a=8$.\\
		Tương tự, ta có $\dfrac{\sqrt{b-9}}{b}\leq \dfrac{1}{6}$. Dấu bằng xảy ra khi và chỉ khi $b=18$.\\
		Vậy $N\leq \dfrac{1}{4}+\dfrac{1}{6}=\dfrac{5}{12}$.\\
		Vậy giá trị lớn nhất của $N$ là $\dfrac{5}{12}$ khi $a=8$, $b=18$.
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Tìm giá trị lớn nhất của biểu thức $A=x\sqrt{1-x^2}$.
	\loigiai{
		Điều kiện: $-1\leq x \leq 1$.\\
		Với $-1\leq x\leq 0$ thì $A\leq 0$. Dấu bằng xảy ra khi và chỉ khi $x=0$, $x=-1$.\\
		Với $0<x\leq 1$ thì $A=\sqrt{x^2(1-x^2)}\leq \dfrac{x^2+(1-x^2)}{2}=\dfrac{1}{2}$.\\
		Dấu bằng xảy ra khi và chỉ khi $\heva{& x>0\\ & x^2=1-x^2}\Leftrightarrow x=\dfrac{\sqrt{2}}{2}$.\\
		Vậy giá trị lớn nhất của $A$ là $\dfrac{1}{2}$ khi $x=\dfrac{\sqrt{2}}{2}$.
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Cho $a$ và $b$ là hai số thực bất kì thỏa mãn $a+b=a^2+b^2$. Tìm giá trị nhỏ nhất và giá trị lớn nhất của biểu thức $$P=\sqrt{a-b+1}.$$
	\loigiai{
		Ta có $P\geq 0$. Dễ thấy $P=0$ khi $a=0$, $b=1$ (thỏa mãn điều kiện).\\
		Vậy giá trị nhỏ nhất của $P$ là $0$ khi $a=0$, $b=1$.\\
		Lại có $P=\sqrt{a-b+1}=\sqrt{\dfrac{(a-b)(a+b)}{a^2+b^2}+1}=\sqrt{\dfrac{a^2-b^2}{a^2+b^2}+1}=\sqrt{2-\dfrac{2b^2}{a^2+b^2}}\leq \sqrt{2}$.\\
		Dấu bằng xảy ra khi và chỉ khi $b=0$, khi đó $a=1$.\\
		Vậy giá trị lớn nhất của $P$ là $\sqrt{2}$ khi $a=1$, $b=0$.
	}
\end{bt}

\begin{bt}%[Trần Chiến]
	Tìm giá trị nhỏ nhất của biểu thức $A=x+\sqrt{x^2+\dfrac{1}{x}}$ với $x>0$.
	\loigiai{
		Biểu thức $A$ nhận giá trị $m$ khi và chỉ khi phương trình ẩn $x$ sau đây có nghiệm:
		$$m=x+\sqrt{x^2+\dfrac{1}{x}} \Leftrightarrow m-x=\sqrt{x^2+\dfrac{1}{x}}.$$
		Với $m\geq x>0$, phương trình tương đương với 
		$$m^2-2mx+x^2=x^2+\dfrac{1}{x} \Leftrightarrow 2mx^2-m^2x+1=0. \qquad (*)$$
		Do $m\neq 0$ nên $(*)$ là phương trình bậc hai. Ta có 
		$$\Delta=m^4-8m=m(m^3-8)\geq 0 \Leftrightarrow \hoac{& m\geq 2\\ & m\leq 0.}$$
		Kết hợp với điều kiện $m>0$ ta có $m\geq 2$.\\
		Ta thấy $m=2\Leftrightarrow x=\dfrac{1}{2}$ (thỏa mãn điều kiện $0<x\leq m$).\\
		Vậy giá trị nhỏ nhất của $A$ là $2$ khi $x=\dfrac{1}{2}$.
	}
\end{bt}
