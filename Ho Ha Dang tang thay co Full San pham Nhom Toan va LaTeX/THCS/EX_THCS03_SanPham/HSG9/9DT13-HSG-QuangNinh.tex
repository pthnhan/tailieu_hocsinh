\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2012-2013 tỉnh Quảng Ninh]%[9D1K9]
Minh]
	Chứng minh đẳng thức $\sqrt[3]{\sqrt[3]{2}-1}=\sqrt[3]{\dfrac{1}{9}}-\sqrt[3]{\dfrac{2}{9}}+\sqrt[3]{\dfrac{4}{9}}$.
	\loigiai{
		Đặt $\sqrt[3]{2}=a\Leftrightarrow 2=a^3$.\\
		Đẳng thức cần chứng minh tương đương với $\sqrt[3]{a-1}=\dfrac{1-a+a^2}{\sqrt[3]{9}}$\\
		$\Leftrightarrow \sqrt[3]{9(a-1)}=a^2-a+1\Leftrightarrow (a^2-a+1)^3=9(a-1)$.\\
		Ta có
		\begin{eqnarray*}
			(a^2-a+1)^3 &=& (a^2-a+1)^2(a^2-a+1) = (a^4+a^2+1-2a^3-2a+2a^2)(a^2-a+1) \\
			&=& (2a+3a^2+1-4-2a)(a^2-a+1)=3(a^2-1)(a^2-a+1)\\
			&=& 3(a-1)(a+1)(a^2-a+1)=3(a-1)(a^3+1)\\
			&=& 9(a-1) \text{ (đpcm)}.
		\end{eqnarray*}
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2012-2013 tỉnh Quảng Ninh]%[9D5G%]
	Giải hệ phương trình $\begin{cases} x^2(2013y-2012)=1 \\ x(y^2+2012)=2013. \end{cases}$
	\loigiai{
		Dễ thấy $x\neq 0$ nên hệ phương trình đã cho tương đương với $\begin{cases} 2013y-2012=\dfrac{1}{x^2} \\ y^2+2012=\dfrac{2013}{x}. \end{cases}$\\
		Đặt $\dfrac{1}{x}=t$, hệ trở thành $\begin{cases} t^2-2013y+2012=0 \\ y^2-2013t+2012=0. \end{cases}$\\
		Trừ vế với vế của phương trình thứ nhất cho phương trình thứ hai ta được\\
		$t^2-y^2-2013y+2013t=0\Leftrightarrow (t-y)(t+y+2013)=0\Leftrightarrow \left[\begin{aligned} & y=t \\ & y=-t-2013.\end{aligned}\right.$
		\begin{itemize}
			\item Trường hợp 1: $y=t.$\\
			Ta có phương trình $t^2-2013t+2012=0\Leftrightarrow \left[\begin{aligned} & t=1 \\ & t=2012.\end{aligned}\right.$\\
			Vậy hệ phương trình đã cho có nghiệm $(x;y)=(1;1)$ và $(x;y)=\left( \dfrac{1}{2012};2012\right)$.
			\item Trường hợp 2: $y=-t-2013.$\\
			Ta có phương trình $t^2-2013(-t-2013)+2012=0$\\
			$\Leftrightarrow t^2+2013t+4054181=0$ (vô nghiệm do $\Delta<0$).
		\end{itemize}
		Vậy hệ phương trình đã cho có đúng 2 nghiệm $(x;y)$ là $(1;1)$ và $\left( \dfrac{1}{2012};2012\right)$.
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2012-2013 tỉnh Quảng Ninh]%[9D4K4]%[9D2K3]
	Cho hàm số bậc nhất $y=mx+m-1$ $(*)$ (với $m$ là tham số).
	\begin{enumerate}[a)]
		\item Tìm các giá trị của $m$ để đồ thị của hàm số $(*)$ tạo với trục tọa độ $Oxy$ một tam giác có diện tích bằng $2$.
		\item Chứng minh rằng đồ thị của hàm số $(*)$ luôn đi qua một điểm cố định với mọi giá trị của $m$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item Vì $(*)$ là hàm số bậc nhất nên $m\neq 0$.\\
			Đồ thị của $(*)$ cắt trục tung tại điểm $A(0;m-1)$ và cắt trục hoành tại điểm $B\left( \dfrac{1-m}{m};0\right)$.\\
			$OAB$ tạo thành tam giác khi và chỉ khi $m\neq 1$.\\
			Khi đó ta có\\
			$S_{ABC}=2\Leftrightarrow \dfrac{1}{2}OA\cdot OB=2\Leftrightarrow OA\cdot OB=4$\\
			$\Leftrightarrow |m-1|\cdot \left| \dfrac{1-m}{m}\right| =4\Leftrightarrow (m-1)^2=4|m|\quad (1)$.
			\begin{itemize}
				\item Trường hợp 1: $m>0$.\\
				$(1)$ trở thành $m^2-2m+1=4m\Leftrightarrow m^2-6m+1=0\Leftrightarrow \left[\begin{aligned} & m=3-2\sqrt{2}\text{ (thỏa mãn)} \\ & m=3+2\sqrt{2}\text{ (thỏa mãn).}\end{aligned}\right.$
				\item Trường hợp 2: $m<0$.\\
				$(1)$ trở thành $m^2-2m+1=-4m\Leftrightarrow m^2+2m+1=0\Leftrightarrow m=-1\text{ (thỏa mãn)}$.
			\end{itemize}
			Vậy với $m\in\{-1;3+2\sqrt{2};3-2\sqrt{2}\}$ thì đồ thị của hàm số $(*)$ tạo với các trục tọa độ $Oxy$ một tam giác có diện tích bằng $2$.
			\item $M(x_0;y_0)$ là điểm cố định thuộc đồ thị của hàm số $(*)$ khi và chỉ khi\\
			$y_0=mx_0+m-1\quad \forall m\in\mathbb{R}$\\
			$\Leftrightarrow (x_0+1)m-(y_0+1)=0\quad \forall m\in\mathbb{R}$\\
			$\Leftrightarrow \begin{cases} x_0+1=0 \\ y_0+1=0 \end{cases}\Leftrightarrow \begin{cases} x_0=-1 \\ y_0=-1. \end{cases}$\\
			Vậy đồ thị của hàm số $(*)$ luôn đi qua điểm cố định $M(-1;-1)$ với mọi giá trị của $m$.
		\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2012-2013 tỉnh Quảng Ninh]%[9D5G1]
	Cho $x,y,z$ là ba số thực dương thỏa mãn $xyz=1$.\\
	Tìm giá trị lớn nhất của biểu thức $A=\dfrac{1}{x^3+y^3+1}+\dfrac{1}{y^3+z^3+1}+\dfrac{1}{z^3+x^3+1}$.
	\loigiai{
		Với $\forall x,y\in\mathbb{R}$, ta có $(x-y)^2\geq 0\Leftrightarrow x^2-xy+y^2\geq xy$.\\
		Mà $x>0,y>0$ nên $x+y>0$.\\
		Suy ra $x^3+y^3=(x+y)(x^2-xy+y^2)\geq (x+y)xy$.\\
		$\Rightarrow x^3+y^3+1=x^3+y^3+xyz\geq (x+y)xy+xyz=xy(x+y+z)>0$.\\
		Chứng minh tương tự ta có
		$$y^3+z^3+1\geq yz(x+y+z)>0$$
		$$z^3+x^3+1\geq zx(x+y+z)>0$$
		Vậy ta có $A\leq \dfrac{1}{xy(x+y+z)}+\dfrac{1}{yz(x+y+z)}+\dfrac{1}{zx(x+y+z)}=\dfrac{x+y+z}{xyz(x+y+z)}=\dfrac{1}{xyz}=1$.\\
		Đẳng thức xảy ra khi và chỉ khi $x=y=z=1$.\\
		Vậy giá trị lớn nhất của $A$ là $1$ khi $x=y=z=1$.
		 
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2012-2013 tỉnh Quảng Ninh]%[9H3K7]%[9H3G7]
	Cho tam giác $ABC$ có ba góc nhọn nội tiếp đường tròn tâm $O$. Gọi $I$ là một điểm trên cung nhỏ $AB$ ($I$ không trùng với $A$ và $B$). Gọi $M,N,P$ theo thứ tự là hình chiếu của điểm $I$ trên các đường thẳng $BC,AC,AB$.
	\begin{enumerate}[a)]
		\item Chứng minh rằng ba điểm $M,N,P$ thẳng hàng.
		\item Xác định vị trí của điểm $I$ để đoạn thẳng $MN$ có độ dài lớn nhất.
	\end{enumerate}
	\loigiai{
	\begin{center}
	\begin{tikzpicture}[scale=0.47,thick]
	\tkzInit[ymin=-5,ymax=10,xmin=-9,xmax=5]
	\tkzClip
	\tkzDefPoints{0/7/A, -8/0/B, 4/0/C}
	\tkzCircumCenter(A,B,C) \tkzGetPoint{O}
	\tkzDefPointBy[rotation = center O angle 70](A)
	\tkzGetPoint{I}
	\tkzDefPointBy[projection= onto B--C](I) \tkzGetPoint{M}
	\tkzDefPointBy[projection= onto A--B](I) \tkzGetPoint{P}
	\tkzDefPointBy[projection= onto A--C](I) \tkzGetPoint{N}
	
	\tkzDrawSegments(A,B B,C C,A M,I N,I I,P M,N A,N I,B I,A)
	\tkzDrawCircle[radius](O,A)
	\tkzDrawPoints[size=2](O)
	
	\tkzLabelPoints(C,P) \tkzLabelPoints[below left](B) \tkzLabelPoints[above](A,N) \tkzLabelPoints[below](M,O)
	\tkzLabelPoints[above left](I)
	
	\tkzMarkRightAngles[size=0.5](C,M,I A,P,I I,N,A)
	\end{tikzpicture}
	\end{center}
	\begin{enumerate}
		\item Xét tứ giác $IPAN$ có $\widehat{IPA}+\widehat{INA}=90^\circ+90^\circ=180^\circ$.\\
		$\Rightarrow$ Tứ giác $IPAN$ là tứ giác nội tiếp.\\
		$\Rightarrow \widehat{IPN}=\widehat{IAN}$ (cùng chắn cung $IN$).\\
		Xét tứ giác $IPMB$ có $\widehat{IPB}=\widehat{IMB}=90^\circ$.\\
		$\Rightarrow$ Tứ giác $IPMB$ là tứ giác nội tiếp.\\
		$\Rightarrow \widehat{IBM}+\widehat{IPM}=180^\circ$.\\
		Mà $\widehat{IBM}+\widehat{IAC}=180^\circ$ (do tứ giác $IACB$ là tứ giác nội tiếp).\\
		$\Rightarrow \widehat{IPM}=\widehat{IAC}$.\\
		$\Rightarrow \widehat{IPM}+\widehat{IPN}=\widehat{IAC}+\widehat{IAN}=180^\circ$.\\
		Vậy $M,N,P$ là ba điểm thẳng hàng.
		\item Vì tứ giác $IPMB$ là tứ giác nội tiếp nên $\widehat{IBP}=\widehat{IMP}$ (cùng chắn cung $IP$).\\
		Lại có tứ giác $IPAN$ là tứ giác nội tiếp nên $\widehat{IAP}=\widehat{INP}$ (cùng chắn cung $IP$).\\
		Xét $\triangle IBA$ và $\triangle IMN$ có\\
		$\begin{cases} \widehat{IBA}=\widehat{IMN} \\ \widehat{IAB}=\widehat{INM} \end{cases}\Rightarrow \triangle IBA\backsim \triangle IMN$ (g.g).\\
		$\Rightarrow \dfrac{MN}{AB}=\dfrac{IM}{IB}$.\\
		Mà tam giác $IBM$ vuông tại $M$ nên $IM\leq IB$ hay $\dfrac{IM}{IB}\leq 1$.\\
		Vậy $MN\leq AB$.\\
		Dấu ``$=$'' xảy ra khi và chỉ khi $\begin{cases} M\equiv B \\ N\equiv A \end{cases}\Leftrightarrow \widehat{IAC}=\widehat{IBC}=90^\circ$\\
		$\Leftrightarrow CI$ là đường kính của $(O)$.\\
		Vậy $MN$ lớn nhất bằng $AB$ khi và chỉ khi $I$ là điểm đối xứng với $C$ qua $O$.
	\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2012-2013 tỉnh Quảng Ninh]%[9D5G5]
	Giải phương trình sau $(x+3)\sqrt{(4-x)(12+x)}+x=28$.
	\loigiai{
	Điều kiện xác định $-12\leq x\leq 4$.\\
	Đặt $\begin{cases} x+3=u \\ \sqrt{(4-x)(12+x)}=v \end{cases}$   ($v\geq 0$).\\
	Phương trình đã cho trở thành $uv=28-x\quad (1)$.\\
	Ta có $u^2+v^2-1=x^2+6x+9+48-8x-x^2-1=56-2x=2(28-x)\quad (2)$.\\
	Từ $(1)$ và $(2)$ ta có $u^2+v^2-1=2uv\Leftrightarrow (u-v)^2=1$\\
	$\Leftrightarrow \left[\begin{aligned} & u-v=1 \\ & u-v=-1\end{aligned}\right.\Leftrightarrow \left[\begin{aligned} & v=u-1 \\ & v=u+1.\end{aligned}\right.$
	\begin{itemize}
		\item Trường hợp 1: $v=u-1$.\\
		$\Leftrightarrow \sqrt{(4-x)(12+x)}=x+2$\\
		$\Leftrightarrow \begin{cases} 48-8x-x^2=x^2+4x+4 \\ x\geq -2 \end{cases}\Leftrightarrow \begin{cases} 2x^2+12x-44=0 \\ x\geq -2 \end{cases}$\\
		$\Leftrightarrow x=-3+\sqrt{31}$ (thỏa mãn).
		\item Trường hợp 2: $v=u+1$.\\
		$\Leftrightarrow \sqrt{(4-x)(12+x)}=x+4$\\
		$\Leftrightarrow \begin{cases} 48-8x-x^2=x^2+8x+16 \\ x\geq -4 \end{cases}\Leftrightarrow \begin{cases} 2x^2+16x-32=0 \\ x\geq -4 \end{cases}$\\
		$\Leftrightarrow x=-4+4\sqrt{2}$ (thỏa mãn).
	\end{itemize}
	Vậy phương trình đã cho có tập nghiệm là $S=\{-3+\sqrt{31};-4+4\sqrt{2}\}$.
	}
\end{ex}
