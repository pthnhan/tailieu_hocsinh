	Giải các phương trình sau
		\begin{enumerate}
			\item $2\sin \left(x+15^\circ\right) =\sqrt{2}$.
			\item $4\sin^2x+23\cos x-19=0$.
			\item $\sqrt{3}\cos 2x-\sin 2x=\sqrt{3}$.
		\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Phương trình tương đương với\\
		$\sin \left(x + 15^\circ\right)=\dfrac{\sqrt{2}}{2} \Leftrightarrow \hoac{&	x+15^\circ =45^\circ +k\cdot360^\circ \\&x + 15^\circ =135^\circ+k\cdot360^\circ} \Leftrightarrow \hoac{&x=30^\circ + k\cdot 360^\circ\\&x=120^\circ+k\cdot360^\circ} \left(k\in \mathbb{Z}\right)$.
		\item Phương trình tương đương với 
		\begin{eqnarray*} 
		&& 4\left(1-\cos^2x\right)+23\cos x-19=0\\
		&\Leftrightarrow& - 4\cos^2x + 23\cos x - 15=0 \Leftrightarrow \hoac{&			\cos x=5 \text{ (loại)} \\&\cos x=\dfrac{3}{4} \text{ (nhận)}.}
		\end{eqnarray*}
		Với $\cos x=\dfrac{3}{4}\Leftrightarrow \hoac{&x=\arccos \dfrac{3}{4} + k2\pi \\&x= - \arccos \dfrac{3}{4} + k2\pi} \left(k\in \mathbb{Z}\right)$.
		\item Phương trình tương đương với $\dfrac{\sqrt{3}}{2}\cos 2x - \dfrac{1}{2}\sin 2x=\dfrac{\sqrt{3}}{2}$
		\begin{eqnarray*}
			& \Leftrightarrow& \sin \dfrac{\pi}{3}\cos 2x - \cos \dfrac{\pi}{3}\sin 2x=\dfrac{\sqrt{3}}{2}\\		&\Leftrightarrow& \sin \left(\dfrac{\pi}{3} - 2x\right)=\dfrac{\sqrt{3}}{2} \\ 
			& \Leftrightarrow& \hoac{&\dfrac{\pi}{3} - 2x=\dfrac{\pi}{3} + k2\pi \\&\dfrac{\pi}{3} - 2x=\pi-\dfrac{\pi}{3} + k2\pi}
			\Leftrightarrow \hoac{&x= - k\pi\\&x= - \dfrac{\pi}{6} - k\pi} \left(k\in \mathbb{Z}\right).
		\end{eqnarray*}
		\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Củ Chi, Hồ Chí Minh, 2017]%[Phùng Hoàng Em, dự án 11-EX-HCM]%[1D2B2-1]%[1D2B5-2]
	\item
	\begin{enumerate}
		\item Từ các chữ số $0$, $1$, $2$, $3$, $4$, $5$, $6$ có thể lập được bao nhiêu số tự nhiên gồm năm chữ số khác nhau và số tạo thành phải chia hết cho $5$.
		\item Một hộp đựng $20$ thẻ được đánh số từ $1$ đến $20$. Lấy ngẫu nhiên hai thẻ cùng lúc. Tính xác suất để hai thẻ lấy ra có ít nhất một thẻ mang số chia hết cho $4$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Gọi số cần tìm có dạng $\overline{abcde}$.\\
		Số cần tìm chia hết cho $5$ nên chữ số tận cùng phải là $0$ hoặc $5$.
		\begin{itemize}
		\item Trường hợp $e=0$:\\
		Chọn $\overline{abcd}$ có $\mathrm{A}_6^4$ cách\\
		Trường hợp này có $1\cdot \mathrm{A}_6^4 = 360$ số được lập.
		\item Trường hợp $e=5$:\\
		Chọn $a$ có $5$ cách $(a\ne 0 \text{ và } a\ne 5)$; Chọn $\overline{bcd}$ có $\mathrm{A}_5^3$ cách\\
		Trường hợp này có $1\cdot 5 \cdot \mathrm{A}_5^3 = 300$ số được lập.
		\end{itemize}
		Vậy ta có thể lập được $360 + 300=660$ số.
		\item Ta có $n\left(\Omega\right)=\mathrm{C}_{20}^2=190$.\\
		Từ $1$ đến $20$ có năm số chia hết cho $4$.\\
		Gọi $A$ là biến cố \lq\lq Hai thẻ lấy ra có ít nhất một thẻ mang số chia hết cho $4$ \rq\rq.
		\begin{itemize}
		\item Trường hợp cả hai thẻ lấy ra đều được số chia hết cho $4$, ta được $\mathrm{C}_{5}^2$ cách.
		\item Trường hợp hai thẻ lấy ra có đúng một thẻ mang số chia hết cho $4$, ta được $\mathrm{C}_{5}^1 \cdot \mathrm{C}_{15}^1$ cách.
		\end{itemize}
		Suy ra $n(A)=\mathrm{C}_{5}^2+\mathrm{C}_{5}^1 \cdot \mathrm{C}_{15}^1=85$.\\ 
		Xác suất của biến cố $A$ là $\mathrm{P}(A)=\dfrac{n(A)}{n\left(\Omega\right)}=\dfrac{85}{190}=\dfrac{17}{38}$.
		
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Củ Chi, Hồ Chí Minh, 2017]%[Phùng Hoàng Em, dự án 11-EX-HCM]%[1D2B3-1]%[1D2K3-2]
	\item
	\begin{enumerate}
	\item Khai triển nhị thức $\left(2x-\dfrac{1}{3}\right)^6$.
	\item Tìm hệ số của số hạng chứa $x^9$ trong khai triển $\left(x^2-\dfrac{3}{x}\right)^{12}$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item 
	\begin{eqnarray*}
		 \left(2x-\dfrac{1}{3}\right)^6&=&\mathrm{C}_6^0(2x)^6+\mathrm{C}_{6}^1(2x)^5\left(- \dfrac{1}{3}\right)+\mathrm{C}_{6}^2(2x)^4\left(- \dfrac{1}{3}\right)^2 +\mathrm{C}_{6}^3(2x)^3\left(- \dfrac{1}{3}\right)^3+ \\
		 &&\mathrm{C}_{6}^4(2x)^2\left(- \dfrac{1}{3}\right)^4 +\mathrm{C}_{6}^5(2x)\left(- \dfrac{1}{3}\right)^5 +\mathrm{C}_{6}^6\left(- \dfrac{1}{3}\right)^6 \\
		 &=&64x^6 - 64x^5 + \dfrac{80}{3}x^4 - \dfrac{160}{27}x^3 + \dfrac{20}{27}x^2 - \dfrac{4}{81}x + \dfrac{1}{729}. 
	\end{eqnarray*}
	\item Số hạng tổng quát trong khai triển $\left(x^2-\dfrac{3}{x}\right)^{12}$ là $$\mathrm{C}_{12}^k(x^2)^{12-k}\left(- \dfrac{3}{x}\right)^k=\mathrm{C}_{12}^k(-3)^k\cdot x^{24-3k}, \text{ với } k=0,1,2,\ldots,12.$$
	Số hạng chứa $x^9$ tương ứng với $24-3k=9 \Leftrightarrow k=5$.\\
	Với $k=5$, ta được hệ số của số hạng chứa $x^9$ là $\mathrm{C}_{12}^5(-3)^5=-192456$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Củ Chi, Hồ Chí Minh, 2017]%[Phùng Hoàng Em, dự án 11-EX-HCM]%[1H1Y7-2]%[1H1B2-2]
	Trong mặt phẳng $Oxy$, cho điểm $A(-1;4)$ và đường thẳng $d\colon 5x-2y+3=0$.
	\begin{enumerate}
		\item Tìm tọa độ $A'$ là ảnh của điểm $A$ qua phép vị tự tâm $O$, tỉ số $k=\dfrac{1}{3}$.
		\item Tìm phương trình đường thẳng $d'$ là ảnh của đường thẳng $d$ qua phép tịnh tiến theo véc-tơ $\vec{u}=\left(\dfrac{3}{2};-\dfrac{1}{2} \right)$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item $\mathrm{V}_{\left(O,\tfrac{1}{3}\right)}(A)=A' \Leftrightarrow \vec{OA'}=\dfrac{1}{3}\cdot\vec{OA}\Leftrightarrow \heva{&x_{A'}=\dfrac{1}{3}\cdot (-1)=-\dfrac{1}{3}\\&y_{A'}=\dfrac{1}{3}\cdot (4)=\dfrac{4}{3}.}$\\
		Tọa độ điểm $A'\left(-\dfrac{1}{3};\dfrac{4}{3}\right)$.
		\item Do $d'$ song song hoặc trùng $d$ nên phương trình $d'$ có dạng $5x-2y+m=0$.\\
		Lấy $M(1;4) \in d$.\\
		$\mathrm{T}_{\vec{u}}(M)=M'(x';y')\Leftrightarrow \heva{&x'=1+\dfrac{3}{2}=\dfrac{5}{2}\\&y'=4-\dfrac{1}{2}=\dfrac{7}{2}}\Rightarrow M'\left(\dfrac{5}{2};\dfrac{7}{2} \right)$.\\
		$M' \in d'$, suy ra $5\cdot \dfrac{5}{2}-2\cdot \dfrac{7}{2}+m=0 \Leftrightarrow m=-\dfrac{11}{2}$.\\
		Vậy, phương trình $d'\colon 5x-2y-\dfrac{11}{2}=0$.
	\end{enumerate}
	}
\end{bt}

	Cho hình chóp $SABCD$ có $ABCD$ là hình thang đáy lớn $CD = 2 AB$. Lấy $M,N$ lần lượt là trung điểm $BC$ và $SC$.
	\begin{enumerate}
		\item Chứng minh $MN\parallel (SBD)$.
		\item Tìm giao tuyến của $(SAB)$ và $(AMN)$.
		\item Tìm giao điểm của $SD$ và $(AMN)$.
		\item Gọi $G$ là trọng tâm của tam giác $SAD$. Chứng minh $AG\parallel (SBC)$.
	\end{enumerate}

		\end{center}
		\begin{enumerate}
			\item Ta có $MN\parallel SB$ (do $MN$ là đường trung bình của tam giác $CSB$).\\
			Mặt khác $MN$ không thuộc $(SBD)$, do đó $MN\parallel (SBD)$.
			\item Ta có $A$ là điểm chung của $(SAB)$ và $(AMN)$.\\
			Mặt khác ta có $\heva{&SB\subset (SAB)\\&MN\subset (AMN)\\&MN\parallel SB.}$\\
			Do đó giao tuyến của $(SAB)$ với $(AMN)$ là đường thẳng $d$ đi qua $A$ và song song với $SB$ và $MN$.
			\item Gọi $I=AM\cap DC$ và $F=IN\cap SD$.\\
			Khi đó $\heva{&F \in SD\\&F \in IN \subset (AMN)}$. Suy ra $F$ chính là giao điểm của $SD$ và $(AMN)$.
			\item Gọi $P$ là trung điểm của $SD$, khi đó ta có $\heva{&PN\parallel CD\\&PN=\dfrac{1}{2}CD.}$\\
			Theo giả thiết ta có $ABCD$ là hình thang đáy lớn $CD = 2 AB$ nên suy ra $\heva{&PN\parallel AB\\&PN=AB.}$\\
			Suy ra tứ giác $ABNP$ là hình bình hành nên $AP\parallel BN$.\\
			Từ đó suy ra $AG\parallel (SBC)$.
		\end{enumerate}

\begin{bt}%[Đề HK1, Củ Chi, Hồ Chí Minh, 2017]%[Phùng Hoàng Em, dự án 11-EX-HCM]%[1D3K1-1]%[1D1G1-5]
	\item
	\begin{enumerate}
	\item Chứng minh $5 \cdot 2^{3n-2}+3^{3n-1}$ chia hết cho $19$, với mọi $n \in \mathbb{N^*}$.
	\item Tìm giá trị lớn nhất và giá trị nhỏ nhất của hàm số $$y=\cos\left( \dfrac{6055\pi}{3}-2x\right)-\sqrt{3}\cos\left( \dfrac{\pi}{2}+x\right)+\cos\left( \pi-x\right)-2.$$
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item Đặt $A_n=5 \cdot 2^{3n-2}+3^{3n-1}$.\\
	Với $n=1$, ta được $A_1=19$ chia hết cho $19$.\\
	Giả sử với $n=k \ge 1$, ta có $A_k=5 \cdot 2^{3k-2}+3^{3k-1}$ chia hết cho $19$.\\
	Ta chứng minh $A_{k+1}=5 \cdot 2^{3(k+1)-2}+3^{3(k+1)-1}$ chia hết cho $19$.\\
	Ta có
	\begin{eqnarray*}
		A_{k+1}&=&5 \cdot 2^{3k+1}+3^{3k+2}=40\cdot2^{3k-2}+27\cdot3^{3k-1}\\
			   &=&8\left(5 \cdot 2^{3k-2}+3^{3k-1}\right)+19\cdot 3^{3k-1}=8\cdot A_k+19\cdot 3^{3k-1}.
	\end{eqnarray*}
	Theo giả thiết quy nạp thì $A_k$ chia hết cho $19$ và $19\cdot3^{3k-1}$ cũng chia hết cho $19$ nên $A_{k+1}$ chia hết cho $19$.\\
	Vậy theo phương pháp quy nạp thì $A_n$ chia hết cho $19$, với mọi $n \in \mathbb{N^*}$.
	\item Ta biến đổi
	\begin{eqnarray*}
	y&=&\cos\left(2018\pi+ \dfrac{\pi}{3}-2x\right)-\sqrt{3}\cos\left( \dfrac{\pi}{2}+x\right)+\cos\left( \pi-x\right)-2\\
	&=&\cos\left( \dfrac{\pi}{3}-2x\right)+ \sqrt{3}\sin x - \cos x-2\\
	&=& \cos \left[2\left(\dfrac{\pi}{6}-x\right)\right]+2\sin \left(\dfrac{\pi}{6}-x \right)-2\\
	&=& 1-2\sin^2\left(\dfrac{\pi}{6}-x\right)+2\sin \left(\dfrac{\pi}{6}-x \right)-2\\
	&=& -2\sin^2\left(\dfrac{\pi}{6}-x\right)+2\sin \left(\dfrac{\pi}{6}-x \right)-1.
	\end{eqnarray*}
	Đặt $t=\sin \left(\dfrac{\pi}{6}-x \right)$, với $t \in \left[-1;1\right]$. Ta được $f(t)=-2t^2+2t-1$.\\
	Đây là phương trình của parabol có tọa độ đỉnh là $\left(\dfrac{1}{2};-\dfrac{1}{2}\right)$.\\
	Bảng biến thiên trên miền $\left[-1;1\right]$:

	Suy ra giá trị lớn nhất của hàm số bằng $-\dfrac{1}{2}$; giá trị nhỏ nhất của hàm số bằng $-5$.
	\end{enumerate}
}
