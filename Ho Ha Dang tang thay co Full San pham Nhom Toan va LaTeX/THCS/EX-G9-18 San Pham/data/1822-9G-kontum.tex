\begin{ex}%[Đề thi HSG 9, tỉnh Kon Tum, năm 2018]%[Đoàn Nhật Thiện, Dự án EX-9G-18]%[9D0G1]
Rút gọn biểu thức $$P=\dfrac{4+\sqrt{5}}{\sqrt{2}+\sqrt{3+\sqrt{5}}}+\dfrac{4-\sqrt{5}}{\sqrt{2}-\sqrt{3-\sqrt{5}}}.$$
\loigiai{
	Ta có
\begin{eqnarray*}
P &=& \dfrac{4+\sqrt{5}}{\sqrt{2}+\sqrt{3+\sqrt{5}}} + \dfrac{4-\sqrt{5}}{\sqrt{2} - \sqrt{3-\sqrt{5}}}\\
&=& \dfrac{4\sqrt{2}+\sqrt{10}}{2+\sqrt{6+2\sqrt{5}}} + \dfrac{4\sqrt{2}-\sqrt{10}}{2 - \sqrt{6-2\sqrt{5}}}\\
&=& \dfrac{4\sqrt{2}+\sqrt{10}}{2+\sqrt{(\sqrt{5}+1)^2}} + \dfrac{4\sqrt{2}-\sqrt{10}}{2 - \sqrt{(\sqrt{5}-1)^2}}\\
&=& \dfrac{4\sqrt{2}+\sqrt{10}}{3+\sqrt{5}} + \dfrac{4\sqrt{2}-\sqrt{10}}{3-\sqrt{5}}\\
&=& \dfrac{24\sqrt{2} - 10\sqrt{2}}{4}\\
&=& \dfrac{7\sqrt{2}}{2}.
\end{eqnarray*}
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Kon Tum, năm 2018]%[Đoàn Nhật Thiện, Dự án EX-9G-18]%[9D0G4]
	Tìm các số nguyên $x$ để $\sqrt{x^2+x+13}$ có giá trị là số hữu tỉ.
	\loigiai{
	Ta có $x\in \mathbb{Z}$ nên $x^2+x+13\in \mathbb{Z}$. Căn bậc hai của một số nguyên chỉ có thể là một số nguyên hoặc một số vô tỉ. Vì vậy $\sqrt{x^2+x+13}=|a|$ với $a\in \mathbb{Z}$.\\
	Suy ra $x^2+x+13=a^2\Leftrightarrow 4x^2+4x+52=4a^2\Leftrightarrow (2x+1-2a)(2x+1+2a)=-51$. 
	\begin{itemize}
		\item TH1: $\heva{&2x+1-2a=1\\ &2x+1+2a=-51}\Leftrightarrow \heva{&x=-13\\ &a=-13}$ (nhận).
		\item TH2: $\heva{&2x+1-2a=-1\\ &2x+1+2a=51}\Leftrightarrow \heva{&x=12\\ &a=13}$ (nhận).
		\item TH3: $\heva{&2x+1-2a=3\\ &2x+1+2a=-17}\Leftrightarrow \heva{&x=-4\\ &a=-5}$ (nhận).
		\item TH4: $\heva{&2x+1-2a=-3\\ &2x+1+2a=17}\Leftrightarrow \heva{&x=3\\ &a=5}$ (nhận).
	\end{itemize}
Vậy $x\in \{-13;12;-4;3\}$.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Kon Tum, năm 2018]%[Đoàn Nhật Thiện, Dự án EX-9G-18]%[9H0G1]
	Cho hai đường tròn $(O;R)$, $(O';R')$ cắt nhau tại $A$ và $B$. Từ điểm $C$ trên tia đối của tia $AB$ kẻ hai tiếp tuyến $CD$ và $CE$ với đường tròn $(O)$ ($C$ không trùng với $A$; $D$, $E$ là hai tiếp điểm không trùng nhau, $E$ nằm trong đường tròn $(O')$). Các đường thẳng $AD$, $AE$ cắt đường tròn $(O')$ tại điểm thứ hai lần lượt là $M$, $N$.
	\begin{enumerate}
		\item Chứng minh $AD\cdot EB=AE\cdot DB$.
		\item Chứng minh đường thẳng $DE$ đi qua trung điểm của $MN$.
	\end{enumerate}
\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=0.8]
			\tkzInit[ymin=-5,ymax=9,xmin=-9,xmax=7]
			\tkzClip
			\tkzDefPoints{-4/0/O,0/2.5/A,0/-2.5/B,2.5/0/O',-6/4.3/D,0.7/0.6/E}
			\tkzDrawCircle[radius](O,A)
			\tkzDrawCircle[radius](O',B)
			\tkzTangent[at=D](O) \tkzGetPoint{c}
			\tkzTangent[at=E](O) \tkzGetPoint{d}
			\tkzInterLL(D,c)(E,d)	\tkzGetPoint{C}
			\tkzInterLC(D,A)(O',B)\tkzGetSecondPoint{M}
			\tkzInterLC(A,E)(O',B)\tkzGetFirstPoint{N}
			\tkzDefMidPoint(M,N)
			\tkzGetPoint{I}
			\tkzDrawSegments(A,B D,E D,C E,C A,C A,D A,M A,E E,N M,N E,I D,B O,D O,E B,M B,I B,N)			
			\tkzDrawPoints[fill=black](A,B,O,O',D,E,C,M,N,I)
			\tkzMarkRightAngle[size=0.3,opacity=.5](O,D,C)
			\tkzMarkRightAngle[size=0.3,opacity=.5](O,E,C)
			\tkzLabelPoints[above](C)
			\tkzLabelPoints[above right](O',M,E)
			\tkzLabelPoints[above left](D)
			\tkzLabelPoints[below](B,N,O,I)
			\draw ($(A)+(-0.5,1)$) node {$A$};
			\end{tikzpicture}
		\end{center}
	\begin{enumerate}
				\item Xét $\triangle AEC$ và $\triangle EBC$, có $\widehat{C}$ chung, $\widehat{AEC}= \widehat{CBE}$ (cùng bằng $\dfrac{1}{2}$ số đo cung $\wideparen{AE}$).
				Suy ra $\triangle AEC \backsim \triangle EBC$ (g.g).
				\begin{equation}\label{1822.1}
					\Rightarrow \dfrac{AE}{BE}= \dfrac{EC}{BC}= \dfrac{CD}{BC},\text{ do }EC= CD.
				\end{equation}
				Xét $\triangle ACD$ và $\triangle DCB$, có\\
				$\widehat{C}$ chung,\\
				$\widehat{CDB}= \widehat{CDA}$(cùng bằng $\dfrac{1}{2}$ số đo cung $\wideparen{AD}$),\\
				\begin{equation}\label{1822.2}
					\Rightarrow \triangle ACD \backsim \triangle DCB \Rightarrow \dfrac{CD}{CB}= \dfrac{AD}{BD}.
				\end{equation}
				Từ \eqref{1822.1} và \eqref{1822.2}, suy ra $\dfrac{AE}{BE}= \dfrac{AD}{BD} \Leftrightarrow AE \cdot BD= AD \cdot BE$.
				\item Gọi $I$ là giao điểm của $DE$ và $MN$.\\
				Ta có $\widehat{BDE}= \widehat{BAE}$ (hai góc nội tiếp cùng chắn cung $\wideparen{BE}$ của đường tròn tâm $O$),\\
				$\widehat{BAE}= \widehat{BMN}$ (hai góc nội tiếp cùng chắn cung $\wideparen{BN}$ của đường tròn tâm $O^{'}$),\\
				Suy ra, $\widehat{BDI}= \widehat{BMI}$, mà hai đỉnh $D$ và $M$ kề nhau và cùng nhìn cạnh $BI$ dưới một góc bằng nhau. Nên tứ giác $DBIM$ nội tiếp.\\
				$\Rightarrow \widehat{BDA}= \widehat{NIB}$ (do cùng phụ với $\widehat{BIM}$),\\
				Mà trong đường tròn tâm $O^{'}$ có tứ giác $ABNM$ nội tiếp, suy ra $\widehat{DAB}= \widehat{BNI}$ (cùng phụ với $\widehat{BAM}$).\\
				\begin{equation}\label{1822.3}
					\Rightarrow \triangle ADB \backsim \triangle NIB \Rightarrow \dfrac{IN}{AD}= \dfrac{IB}{DB} \Rightarrow \dfrac{IN}{IB}= \dfrac{AD}{BD}.
				\end{equation}
				Xét $\triangle ABE$ và $\triangle MBI$, có\\
				$\widehat{BAE}= \widehat{BMI}$ (cùng bằng $\dfrac{1}{2}$ số đo cung $BN$),\\
				$\widehat{AEB}= \widehat{BIM}$ (cùng phụ với $\widehat{BDA}$),\\
				\begin{equation}\label{1822.4}
					\Rightarrow \triangle ABE \backsim \triangle MBI \Rightarrow \dfrac{MI}{AE}= \dfrac{BI}{BE} \Rightarrow \dfrac{MI}{BI}= \dfrac{AE}{BE}.
				\end{equation}
				Từ \eqref{1822.3} và \eqref{1822.4}, suy ra $\dfrac{MI}{BI}= \dfrac{AE}{BE}= \dfrac{AD}{BD}= \dfrac{IN}{IB} \Rightarrow \dfrac{MI}{BI}= \dfrac{IN}{BI} \Rightarrow MI= IN.$
				
			\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Kon Tum, năm 2018]%[Đoàn Nhật Thiện, Dự án EX-9G-18]%[9D0G2]
	Giải phương trình:
	$$x^2+2x\sqrt{x-\dfrac{1}{x}}=3x+1.$$
	\loigiai{
	Điều kiện $-1 \le x < 0 $ và $x \ge 1$.
	Từ điều kiện suy ra $x \ne 0$, khi đó chia cả hai vế phương trình cho $x$ ta được
	$$x + 2\sqrt{x - \dfrac{1}{x}} = 3 + \dfrac{1}{x}.$$  
	Đặt $t = \sqrt{x - \dfrac{1}{x}} \ge 0$ ta được phương trình
	$$t^2 + 2t - 3 = 0.$$
	Giải phương trình ta được hai nghiệm $t=1$ hoặc $t=-3\text{ (loại)}$.
	Với $t=1$, ta có 
	$$\sqrt{x - \dfrac{1}{x}} = 1.$$
	Phương trình tương đương với
	$$x^2 - x - 1 = 0.$$
	So sánh với điều kiện, phương trình có hai nghiệm đều thỏa mãn
	$$x= \dfrac{1 - \sqrt{5}}{2}\text{ hoặc }x= \dfrac{1 + \sqrt{5}}{2}.$$
}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Kon Tum, năm 2018]%[Đoàn Nhật Thiện, Dự án EX-9G-18]%[9D0G7]
	Cho $x$ là số thực dương. Tìm giá trị nhỏ nhất của biểu thức: $$P=\dfrac{x}{(x+2018)^2}.$$
	\loigiai{
	Đặt $y=x+2018\Rightarrow x=y-2018$ thì $$P=\dfrac{y-2018}{y^2}=\dfrac{1}{y}-\dfrac{2018}{y^2}.$$
	Đặt $z=\dfrac{1}{y}$ thì $P=z-2018z^2$. Ta có 
	\begin{eqnarray*}
		P&=& z-2018z^2\\
		&=& -2018\left(z^2-2\cdot \dfrac{1}{4036}z+\dfrac{1}{4036^2}\right)+\dfrac{1}{8072}\\
		&=&-2018\left(z-\dfrac{1}{4036}\right)^2+\dfrac{1}{8072}\\
		&\le& \dfrac{1}{8072}.
	\end{eqnarray*}
Dấu \lq\lq$=$\rq\rq\, xảy ra khi $x=2018.$ Vậy giá trị lớn nhất của $P$ là $\dfrac{1}{8072}$. 
}
\end{ex}
