\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Hoàn Kiếm, Hà Nội]%[9D3B3]
	Hãy tính giá trị của:
\begin{enumerate}[a)]
	\item $M=(2\sqrt{300}+3\sqrt{48}-4\sqrt{75}):\sqrt{3}$,
	\item $N=\sqrt{(\sqrt{3}-2)^2}+\sqrt{4-2\sqrt{3}}$,
	\item $P=\dfrac{2	}{\sqrt{3}+1}-\dfrac{1}{\sqrt{3}-2}+\dfrac{12}{\sqrt{3}+3}$.
\end{enumerate}
	\loigiai{
\begin{enumerate}[a)]
	\item $M=(20\sqrt{3}+12\sqrt{3}-20\sqrt{3}):\sqrt{3}=12\sqrt{3}:\sqrt{3}=12$.
	\item $N=\left| \sqrt{3}-2 \right|+\left| 1-\sqrt{3} \right|=2-\sqrt{3}+\sqrt{3}-1=1$.
	\item $P=-1+\sqrt{3}+2+\sqrt{3}+6-2\sqrt{3}=7$.
\end{enumerate}	
	}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Hoàn Kiếm, Hà Nội]%[9D4Y4]
	Cho các biểu thức: 
	\begin{center}
		$A=1-\dfrac{\sqrt{x}}{1+\sqrt{x}}$ và $B=\dfrac{\sqrt{x}+3}{\sqrt{x}-2}+\dfrac{\sqrt{x}+2}{3-\sqrt{x}}+\dfrac{\sqrt{x}+2}{x-5\sqrt{x}+6}$
	\end{center}
		với $x\ge 0, x\neq 4$ và $x\neq 9.$
\begin{enumerate}[a)]
	\item Hãy tính giá trị của $A$ khi $x=16.$
	\item Rút gọn $B.$
	\item Xét biểu thức $T=\dfrac{A}{B}.$ Hãy tính giá trị nhỏ nhất của $T.$ 
\end{enumerate}
	\loigiai{
\begin{enumerate}[a)]
	\item $A=\dfrac{1}{1+\sqrt{x}};x=16\Rightarrow A=\dfrac{1}{1+\sqrt{16}}=\dfrac{1}{5}$.
	\item $B=\dfrac{(\sqrt{x}+3)(\sqrt{x}-3)-(\sqrt{x}+2)( \sqrt{x}-2)+\sqrt{x}+2}{(\sqrt{x}-2)(\sqrt{x}-3)}=\dfrac{\sqrt{x}-2}{(\sqrt{x}-2)(\sqrt{x}-3)}=\sqrt{x}-3$.
	\item $T=\dfrac{A}{B}=\dfrac{1}{(\sqrt{x}+1)(\sqrt{x}-3)} =\dfrac{1}{(\sqrt{x}-1)^2+(-4)}$.\\
	$T_{\min }\Leftrightarrow \left[(\sqrt{x}-1)^2+(-4) \right]_{\max }$. Ta có $(\sqrt{x}-1)^2+(-4)\ge -4\,\,\forall x\ge 0;x\ne 4;x\ne 9$.\\
	Do đó $\left[(\sqrt{x}-1)^2+(-4)\right]_{\max}\Leftrightarrow (\sqrt{x}-1)^2+(-4)=-4\Leftrightarrow \sqrt{x}-1=0\Leftrightarrow x=1$.\\
	Vậy $T_{\min }=-\dfrac{1}{4}$ khi $x=1$.
\end{enumerate}
		}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Hoàn Kiếm, Hà Nội]%[8H2G3]
	Cho hàm số $y=(2-m)x+m+1$ (với $m$ là tham số và $m\neq 2$) có đồ thị là đường thẳng $d$.
	\begin{enumerate}[a)]
		\item Khi $m=0$, hãy vẽ $d$ trên hệ trục tọa độ $Oxy.$
		\item Tìm $m$ để $d$ cắt đường thẳng $y=2x-5$ tại điểm có hoành độ bằng $2.$
		\item Tìm $m$ để $d$ cùng với trục tọa độ $Ox$, $Oy$ tạo thành một tam giác có diện tích bằng $2.$
	\end{enumerate}
	\loigiai{
\begin{enumerate}[a)]
	\item $m=0\Rightarrow y=2x+1$. Đồ thị:
\begin{center}
\begin{tikzpicture}[thick,>=stealth,x=1cm,y=1cm,scale=1.5] 
\draw[->] (-1.5,0) -- (2,0) node[above] {$x$};
\draw[->] (0,-1) -- (0,3.4) node[right] {$y$};
\draw [fill=white,draw=black] (0,0) circle (1pt)node[below left] {\footnotesize $O$};
\draw[thick,smooth,samples=100,domain=-1:1.2] plot(\x,{2*(\x)+1});
\draw[fill=black] (-0.5,0)node [above]{\footnotesize $\dfrac{1}{2}$} circle(0.03);
\draw[fill=black] (0,1)node [left]{\footnotesize $1$} circle(0.03);
\end{tikzpicture}
\end{center}
	\item Phương trình hoành độ giao điểm của $d$ và đường thẳng $y=2x-5$ là:
\begin{center}
	$\left( 2-m \right)x+m+1=2x-5\Leftrightarrow mx=m+6$ (*)
\end{center}
Theo đề ta có (*) có nghiệm là $x=2\Rightarrow 2m=m+6\Leftrightarrow m=6$.\\
Vậy $m=6.$ 
\item Gọi giao điểm của $d$ với $Ox,Oy$ lần lượt là $A;B\Rightarrow A\left( \dfrac{m+1}{m-2};0 \right)$ và $B\left( 0;m+1 \right)$ với $m\ne 2.$\\
Khi đó ta có: \\
$S_{OAB}=\dfrac{1}{2}\left| \dfrac{m+1}{m-2} \right|.\left| m+1 \right|\Leftrightarrow \left| \dfrac{{{\left( m+1 \right)}^{2}}}{m-2} \right|=4
\Leftrightarrow 
\left[\begin{array}{ll} 
m^2+2m+9=0 \\ 
m^2+6m-7=0   
\end{array}\right.
\Leftrightarrow 
\left[\begin{array}{ll}
m=1 \ \ \ \ \text{(nhận)}\\
m=-7\ \  \text{(nhận)}
\end{array}\right.$
\end{enumerate}	
}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Hoàn Kiếm, Hà Nội]%[8H2G3]
	Cho đường tròn $(O;R)$ và điểm $A$ nằm ngoài $(O).$ Từ $A$ kẻ hai tiếp tuyến $AB, AC$ với $(O)$ $(B,C$ là các tiếp điểm$).$ Gọi $H$ là giao điểm của $OA$ và $BC.$
	\begin{enumerate}[a)]
		\item Chứng minh bốn điểm $A, B, O, C$ cùng thuộc một đường tròn.
		\item Chứng minh $OA$ là đường trung trực của $BC.$
		\item Lấy $D$ đối xứng với $B$ qua $O.$ Gọi $E$ là giao điểm của đoạn thẳng $AD$ với $(O)$ ($E$ không trùng với $D$). Chứng minh $\dfrac{DE}{BE}=\dfrac{BD}{BA}.$
	\end{enumerate}
	\loigiai{
	\begin{center}
	\begin{tikzpicture}
			\tkzDefPoints{0/0/O, 8/2/A}
			\newcommand*{\R}{3cm}
			\tkzDrawCircle[R](O,\R) 
			\tkzTangent[from with R= A](O,\R)  \tkzGetPoints{B}{C} 
			\coordinate (D) at ($(B)!2!(O)$);
			\tkzInterLC[R](A,D)(O,\R) \tkzGetFirstPoint{E} 
			\tkzInterLL(O,A)(B,C) \tkzGetPoint{H}
			\tkzDrawSegments(A,B A,C B,D A,D B,C O,A O,C B,E)
			\pgfresetboundingbox
			\tkzDrawPoints[size=10, fill=white](A,B,C,D,O,H,E)
			\tkzLabelPoints[above](B)
			\tkzLabelPoints[below](D)
			\tkzLabelPoints[right](A)
			\tkzLabelPoints[left](O,E)
			\tkzLabelPoints(C,H)
		\end{tikzpicture}
		\end{center}
	\begin{enumerate}[a)]
	\item Ta có $AB,AC$ là các tiếp tuyến của $(O)$ nên $\widehat{ABO}=90^\circ$ và $\widehat{ACO}=90^\circ$. \\
	Suy ra tứ giác$ABOC$ nội tiếp đường tròn (vì tứ giác có tổng hai góc đối nhau bằng $180^\circ$). \\
	Vậy $4$ điểm $A,B,O,C$ cùng thuộc một đường tròn.
	\item Ta có: $OA\bot BC$ tại $H$  (tính chất tiếp tuyến của đường tròn). 	(1)\\
	Mặt khác ta lại có $\Delta OBC$ cân tại $O$ (vì $OA=OB=R$) nên suy ra $OH$ là đường trung tuyến của $\Delta OBC$ hay $H$ là trung điểm của $BC$. 				(2) \\
	Từ (1) và (2) suy ra $OA$ là trung trực của đoạn thẳng $BC$.
	\item Ta có $\widehat{ABD}=90^\circ$ (tính chất tiếp tuyến) và $\widehat{BED}=90^\circ$ (góc nội tiếp chắn nửa đường tròn).\\
	Xét $\Delta ABD$ và $\Delta BED$ có: $$\left\{\begin{array}{ll}
	\widehat{ABD}=\widehat{BED}=90^\circ \\ 
	\widehat{BDE}=\widehat{BDA}
	\end{array}\right.$$
	Suy ra $\Delta ABD\sim \Delta BED\,\,\left( g-g \right).$\\
	Suy ra $\dfrac{AB}{BE}=\dfrac{BD}{ED}$ hay $\dfrac{DE}{BE}=\dfrac{BD}{BA}$. 
	\end{enumerate}
}	
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Hoàn Kiếm, Hà Nội]%[8H2G3]
	Cho $x>0, y>0$ thỏa mãn $xy=6.$ Tìm giá trị nhỏ nhất của biểu thức: $$Q=\dfrac{2}{x}+\dfrac{3}{y}+\dfrac{6}{3x+2y}.$$
	\loigiai{
	\begin{alignat*}{3}
		&Q& \quad = \quad & \dfrac{2y+3x}{6} +\dfrac{6}{3x+2y}\\
		&& \quad= \quad&\dfrac{(3x+2y)^2+36}{6(3x+2y)}\\
		&& \quad= \quad&\dfrac{(3x+2y-6)^2+12(3x+2y)}{6(3x+2y)}\\
		&& \quad\ge \quad &\dfrac{12(3x+2y)}{6(3x+2y)}
	\end{alignat*}
	Suy ra $Q_{\min}=2$.	
	}
\end{ex}
