\begin{ex}%[HSG Kazakhstan năm 2018, vòng 1]%[Đặng Đức Quý, Dự án EX-Nat.MO-18]%[9H0K2]
	Trong một hình thang cân $ABCD$, gọi $O$ là trung điểm của cạnh $AD$. Một đường tròn tâm $O$ có bán kính $OB$ sao cho $AB$ là tiếp tuyến với đường tròn tâm $O$. Đoạn thẳng $AC$ cắt đường tròn $(O; OB)$ tại điểm $K$ $(K \neq C)$ và $M$ là một điểm sao cho tứ giác $ABCM$ là một hình bình hành. Đường tròn ngoại tiếp tam giác $CMD$ cắt đoạn thẳng $AC$ tại điểm $L$ $(L\neq C)$. Chứng minh rằng $AK = CL$. 
	\loigiai{ 
		\begin{center}
				\begin{tikzpicture}[>=stealth,line join=round,line cap=round,font=\footnotesize,scale=.8]
			\tkzDefPoints{0/0/O,-4/3/B,4/3/C,-6.25/0/A,6.25/0/D,1.75/0/M,2.73/2.63/L,-4.99/0.37/K}
			\tkzCalcLength[cm](O,B) \tkzGetLength{Tempt}
			\draw[name path=c] (O) circle (\Tempt cm);
			\tkzDefCircle[circum](C,M,D) \tkzGetPoint{O'}
			\tkzCalcLength[cm](O',C) \tkzGetLength{Tempt}
			\draw[name path=m] (O') circle (\Tempt cm);
			\tkzDrawSegments(A,B B,C C,A A,D C,D M,C O,B B,K M,L)
			\tkzMarkRightAngles[size=0.3,thin](O,B,A)
			\tkzDrawPoints(A,B,C,D,O,M,L,K,O')
			\tkzLabelPoints[left](A,B)
			\tkzMarkSegments[mark=|](A,B M,C C,D)
			\tkzLabelPoints[above left](K)
			\tkzLabelPoints[above](C,O')
			\tkzLabelPoints[below right](L,D)
			\tkzLabelPoints[below](O)
			\tkzLabelPoints[below left](M)
			\tkzDrawPoints[fill=black](A,B,C,D,O,M,L,K)
			\end{tikzpicture}
		\end{center}
		Từ giả thiết $ABCM$ là một hình bình hành và $ABCD$ là hình thang cân, ta có
		$$\widehat{CDM} = \widehat{MAB} = \widehat{BCM}$$
		từ đó suy ra $BC$ là tiếp tuyến của đường tròn ngoại tiếp tứ giác $MLCD$.\\
		Mặt khác $AB$ là tiếp tuyến của đường tròn tâm $O$ ngoại tiếp $\triangle BCK$ suy ra $$\widehat{LMC} = \widehat{LCB} = \widehat{KBA}.$$
		Ta lại có $\widehat{BAK}=\widehat{LCM}$ nên suy ra $\triangle ABK = \triangle CML\ \text{(cạnh - góc - cạnh)} \Rightarrow AK = CL$.
		
	}
\end{ex}
\begin{ex}%[HSG Kazakhstan năm 2018, vòng 1]%[Đặng Đức Quý, Dự án EX-Nat.MO-18]%[1D0G1]
	Cho số tự nhiên $ m \geq 2$. Dãy các số tự nhiên $(b_0, b_1, \ldots,b_m)$ được gọi là dãy số lõm nếu $b_k + b_{k-2} \leq 2b_{k-1}$ với $ 2\leq k \leq m$. Chứng minh rằng có không quá $2^m$ dãy số lõm bắt đầu bằng $b_0 = 1$ và \ $b_1 = 2$. 
	\loigiai{
		Từ điều kiện bài toán
		$$ b_m-b_ {m-1} \le b_ {m-1} -b_ {m-2} \le \ldots \le b_1-b_0 = 1. $$
		Một dãy số là dãy lõm khi dãy số có dạng
		$$ 1,2, \ldots, n, n, \ldots, n, b_ {n + k}, \ldots, b_m$$ 
		trong đó $2 \leq n \leq m+1$, số $n$ được viết $(k+1)$ lần với
		$ 0 \leq k \leq m + 1-n$ và $ n > b_ {n + k}> \ldots> b_m. $ 
		Dãy số $ b_ {n + k}, \ldots, b_m $ không lớn hơn $ C_ {n-1} ^ {m + 1-nk} $  (do $ b_ {n + k}, \ldots, b_m $  là các số nguyên dương khác nhau và nhỏ hơn $n$).\\
		Với $n = m + 1$, dãy số bằng $1$.\\
		Với $2 \leq n \leq m$, dãy số 
		$$ C_ {n-1} ^ {m + 1-n} + C_ {n-1 }^{m - n} + \ldots + C_ {n-1} ^ {0}  \leq C_ {n-1} ^ {0} + C_ {n-1} ^ {1} + \ldots + C_ {n- 1} ^ {n-1} = 2 ^ {n-1} $$ ( điều này luôn đúng với mọi số $k = 0,1, \ldots, m + 1-n).$ \\
		Vì vậy tổng của dãy số là 
				$$ 2 ^ 1 + 2 ^ 2 + \ldots + 2 ^ {m-1} + 1 = 2 ^ m-1 <2 ^ m. $$
	}
\end{ex}
\begin{ex}%[HSG Kazakhstan năm 2018, vòng 1]%[Đặng Đức Quý, Dự án EX-Nat.MO-18]%[0D0G3]
	Liệu có tồn tại một hàm số $f:\mathbb{N}^*\to \mathbb{N}^*$ thỏa mãn
	\[f\left( {mf(n)} \right) = f(m)f(m + n) + n,\,\,\forall m,n \in \mathbb{N}^*?\tag{1}\]
	\loigiai{Giải sử tòn tại hàm số $f$ thỏa mãn các yêu cầu của đề bài. Đặt $f(1)=c$. Từ $(1)$ cho $m=n=1$ ta được
		$f(c) = cf(2) + 1$. Từ $(1)$ cho $m=1$, ta được
		\[f\left( {f(n)} \right) = cf(n + 1) + n,\,\,\forall n \in \mathbb N^*.\tag{2}\]
		Từ $(1)$ cho $n=1$, ta được
		{\allowdisplaybreaks
			\begin{align*}
			&f\left( {cm} \right) = f(m)f(m + 1) + 1,\,\,\forall m \in \mathbb{N^*}\\
			\Leftrightarrow& f\left( {cn} \right) = f(n)f(n + 1) + 1,\,\,\forall n \in \mathbb{N^*}.\tag{3}
			\end{align*}}
		Từ $(2)$ suy ra $f\left( {f(n)} \right) \equiv n\left( {\bmod c} \right),\,\,\forall n \in\mathbb {N^*}$.$\quad(4)$\\
		Nếu $n\,\, \vdots \,\,c$ thì từ $(3)$ thay $n$ bởi $f(n)$, ta được
		\[f\left( {cf(n)} \right) = f\left( {f(n)} \right) \cdot f\left( {f(n) + 1} \right) + 1 \equiv 1 \left( {\bmod c} \right).\tag{5}\]
		Từ $(1)$ cho $m=c$, ta được
		\[f\left( {cf(n)} \right) = f(c)f(c + n) + n.\tag{6}\]
		Như thế, nếu $n\,\, \vdots \,\,c$ thì từ $(5)$ và $(6)$ suy ra
		\[f(c)f(c + n) + n \equiv 1\left( {\bmod c} \right),\]
		từ đậy kết hợp với $f(c) = cf(2) + 1$, ta được 
		\[f(c + n)  \equiv 1\left( {\bmod c} \right).\]
		Suy ra $ f \equiv 1 \pmod c $ với mọi số $n$ chia hết cho $c$.
		Suy ra $f\left( {cf(n)} \right) \equiv 1\left( {\bmod c} \right)$, kết hợp với $(6)$, ta được
		{\allowdisplaybreaks
			\begin{align*}
			&f(c)f(c + n) + n \equiv 1\left( {\bmod c} \right),\,\,\forall n \in\mathbb {N^*}\\
			\Rightarrow& \left( {cf(2) + 1} \right)f(c + n) + n \equiv 1\left( {\bmod c} \right),\,\,\forall n \in\mathbb {N^*}\\
			\Rightarrow& f(c + n) + n \equiv 1\left( {\bmod c} \right),\,\,\forall n \in\mathbb {N^*}\\
			\Rightarrow& f(c + n) \equiv 1 - n\left( {\bmod c} \right),\,\,\forall n \in\mathbb {N^*}.\tag{7}
			\end{align*}}
		Từ $(7)$ lấy $n=c+2$ rồi sử dụng $(3)$, ta được
		$$ 1 \equiv f (c + 2) f (c + 3) +1 \equiv (-1) \cdot (-2) +1 \equiv 3 \pmod c, $$
		điều đó không đúng khi $c\ge 3$. Vậy không tồn tại hàm số thỏa mãn bài toán.  
	}
\end{ex}
