{$3360$}
{\True $3360x^6$}
{$13440$}
{$13440x^6$}
\loigiai{

$\begin{aligned}
&2\mathrm{C}_n^2+3\mathrm{A}_n^2-360=0\\
&\Leftrightarrow \dfrac{n!}{(n-2)!\cdot 2!}+3\dfrac{n!}{(n-2)!}-360=0\\
&\Leftrightarrow 4\dfrac{n!}{(n-2)!}=360\\
&\Leftrightarrow n(n-1)-90=0\\
&\Leftrightarrow n^2-n-90=0\\
&\Leftrightarrow \hoac{&n=10\\&n=-9.}
\end{aligned}$\\

\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[HKI, Sở Giáo Dục và Đào Tạo Bà Rịa-Vũng Tàu, 2018]%[1D1B4] 
Giải các phương trình sau:
\begin{enumerate}
\item $2\sin{x}-1=0$.
\item $\sin^{2}{x}-\cos{x}+1=0$.
\item $\sin{x}-\sqrt{3}\cos{x}=1$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item 
$2\sin{x}-1=0$\\
$\begin{aligned}
&\Leftrightarrow \sin{x}=\dfrac{1}{2}\\
&\Leftrightarrow \sin{x}=\sin{\dfrac{\pi}{6}}\\
&\Leftrightarrow \hoac{&x=\dfrac{\pi}{6}+k2\pi\\&x=\dfrac{5\pi}{6}+k2\pi},\, k\in \mathbb{Z}.
\end{aligned}$\\
Vậy tập nghiệm của phương trình $S=\left\lbrace \dfrac{\pi}{6}+k2\pi; \dfrac{5\pi}{6}+k2\pi,\, k\in \mathbb{Z} \right\rbrace$.
\item $\sin^{2}{x}-\cos{x}+1=0$\\
$\begin{aligned}
&\Leftrightarrow \sin^{2}{x}-2\sin^{2}{\dfrac{x}{2}}=0\\
&\Leftrightarrow 4\sin^{2}{\dfrac{x}{2}}\cos^{2}{\dfrac{x}{2}}+2\sin^{2}{\dfrac{x}{2}}=0\\
&\Leftrightarrow \sin^{2}{\dfrac{x}{2}}\cdot \left(2\cos^{2}{\dfrac{x}{2}}+1\right)=0\\
&\Leftrightarrow \hoac{&2\cos^{2}{\dfrac{x}{2}}+1=0\quad (\text{vô nghiệm.})\\&\sin^{2}{\dfrac{x}{2}}=0}\\
&\Leftrightarrow \sin^{2}{\dfrac{x}{2}}=0\\
&\Leftrightarrow \sin{\dfrac{x}{2}}=0\\
&\Leftrightarrow \dfrac{x}{2}=k\pi,\, k\in \mathbb{Z}\\
&\Leftrightarrow x=\dfrac{k\pi}{2},\, k\in \mathbb{Z}.
\end{aligned}$\\

$\begin{aligned}
&\sin{2x}+m\cos{x}-4\sin{x}-2m=0\\
&\Leftrightarrow 2\sin{x}\cos{x}-3\cos{x}-4\sin{x}-2m=0\\
&\Leftrightarrow 2\sin{x}(\cos{x}-2)+m(\cos{x}-2)=0\\

&\Leftrightarrow \hoac{&\cos{x}-2=0\quad (\text{vô nghiệm})\\&2\sin{x}+m=0}\\
&\Leftrightarrow 2\sin{x}+m=0\\
&\Leftrightarrow \sin{x}=-\dfrac{m}{2}.
\end{aligned}$\\
Để phương trình đã cho có nghiệm thì $\left|-\dfrac{m}{2}\right|\leq 1\Leftrightarrow -2\leq m\leq 2$.
