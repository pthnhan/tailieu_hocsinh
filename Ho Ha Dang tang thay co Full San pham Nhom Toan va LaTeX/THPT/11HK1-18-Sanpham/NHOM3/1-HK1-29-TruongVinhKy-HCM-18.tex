\begin{bt}%[HK1 THPT Trương Vĩnh Ký, 2017-2018]$[Nhật Thiện]%[1D1B2-1]
	Giải phương trình lượng giác $\sin \left( 2x-\dfrac{\pi}{3}\right) =\sin \left(x+\dfrac{\pi}{4}\right)$.
	\loigiai{
	\begin{eqnarray*}
		& &\sin \left( 2x-\dfrac{\pi}{3}\right) =\sin \left(x+\dfrac{\pi}{4}\right)\\
		&\Leftrightarrow& \hoac{&2x-\dfrac{\pi}{3}=x+\dfrac{\pi}{4}+k2\pi\\ &2x-\dfrac{\pi}{3}=\pi-x-\dfrac{\pi}{4}+k2\pi}\\
		&\Leftrightarrow& \hoac{&x=\dfrac{7\pi}{12}+k2\pi\\ &x=\dfrac{13\pi}{36}+\dfrac{k2\pi}{3}}\qquad (k\in \mathbb{Z}).
	\end{eqnarray*}	
}
\end{bt}
\begin{bt}%[HK1 THPT Trương Vĩnh Ký, 2017-2018]$[Nhật Thiện]%[1D1B3-3]
	Giải phương trình lượng giác $3\sin x+\sqrt{3}\cos x=-\sqrt{3}$.
	\loigiai{Ta có
	\begin{eqnarray*}
		& &3\sin x+\sqrt{3}\cos x=-\sqrt{3}\\
		&\Leftrightarrow& \dfrac{\sqrt{3}}{2}\sin x+\dfrac{1}{2}\cos x=-\dfrac{1}{2}\\
		&\Leftrightarrow& \sin \left(x+\dfrac{\pi}{6}\right)=\sin \left(-\dfrac{\pi}{6}\right)\\
		&\Leftrightarrow& \hoac{&x+\dfrac{\pi}{6}=-\dfrac{\pi}{6}+k2\pi\\ &x+\dfrac{\pi}{6}=\dfrac{7\pi}{6}+k2\pi}\\
		&\Leftrightarrow& \hoac{&x=-\dfrac{\pi}{3}+k2\pi\\ &x=\pi+k2\pi.}\qquad (k\in\mathbb{Z}).
	\end{eqnarray*}
}
\end{bt}
\begin{bt}%[HK1 THPT Trương Vĩnh Ký, 2017-2018]$[Nhật Thiện]%[1D1K3-6]
	Giải phương trình lượng giác $5\left(\sin x+\dfrac{\cos 3x+\sin 3x}{1+2\sin 2x}\right)=3+\cos 2x$.
	\loigiai{
	Điều kiện: $\sin 2x\ne -\dfrac{1}{2}$.\\
	Ta có 
	\begin{eqnarray*}
		& &5\left(\sin x+\dfrac{\cos 3x+\sin 3x}{1+2\sin 2x}\right)=3+\cos 2x\\
		&\Leftrightarrow& 5\left(\dfrac{\sin x+2\sin 2x\cdot \sin x+\cos 3x+\sin 3x}{1+2\sin 2x}\right)=3+\cos 2x\\
		&\Leftrightarrow& 5\left(\dfrac{2\sin 2x\cos x+\cos x-\cos 3x+\cos 3x}{1+2\sin 2x}\right)=3+\cos 2x\\
		&\Leftrightarrow& 5\left(\dfrac{2\sin 2x\cos x+\cos x}{1+2\sin 2x}\right)=3+\cos 2x\\
		&\Leftrightarrow& 5\left(\dfrac{\cos x(2\sin 2x+1)}{1+2\sin 2x}\right)=3+\cos 2x\\
		&\Leftrightarrow& 5\cos x=3+\cos 2x\Leftrightarrow 2\cos^2 x-5\cos x+2=0\\
		&\Leftrightarrow& \hoac{&\cos x=2\\ &\cos x=\dfrac{1}{2}}
	\end{eqnarray*}
Với $\cos x=2$, phương trình vô nghiệm.\\
Với $\cos x=\dfrac{1}{2}\Leftrightarrow \cos x=\cos \dfrac{\pi}{3}\Leftrightarrow \hoac{&x=\dfrac{\pi}{3}+k2\pi\\ &x=-\dfrac{\pi}{3}+k2\pi}$, $k\in \mathbb{Z}$.
}
\end{bt}
\begin{bt}%[HK1 THPT Trương Vĩnh Ký, 2017-2018]$[Nhật Thiện]%[1D2B2-1]
	Cho tập hợp $X=\{1;2;3;4;5;6;7;8;9\}$. Từ tập hợp $X$ có thể lập được bao nhiêu số tự nhiên có $5$ chữ số khác nhau và luôn có mặt chữ số $7$.
	\loigiai{
	Gọi số tự nhiên có $5$ chữ số khác nhau là $\overline{abcbe}$, $a\ne 0$.\\
	Số có $5$ chữ số khác nhau tùy ý có $\mathrm{A}^5_9$ số.\\
	Số có $5$ chữ số khác nhau không có mặt số $7$ có $\mathrm{A}^5_8$ số.\\
	Vậy có $\mathrm{A}^5_9-\mathrm{A}^5_8=8400$ số thỏa đề bài.
}
\end{bt}
\begin{bt}%[HK1 THPT Trương Vĩnh Ký, 2017-2018]$[Nhật Thiện]%[1D2B3-2]
	Tìm số hạng không chứa $x$ trong khai triển $\left( 2x-\dfrac{1}{x^3}\right)^{16}$ (với $x\ne 0$).
	\loigiai{
	Ta có $\left( 2x-\dfrac{1}{x^3}\right)^{16}=\sum\limits_{k=0}^{16}\mathrm{C}^k_{16}(2x)^{16-k}\left(-\dfrac{1}{x^3}\right)^k.$\\
	Khi đó, số hạng tổng quát trong khai triển là 
	\begin{eqnarray*}
		& &T_{k+1}=\mathrm{C}^k_{16}\cdot (2x)^{16-k}\cdot \left(-\dfrac{1}{x^3}\right)^k\\
		&\Leftrightarrow& T_{k+1}=\mathrm{C}^k_{16}\cdot 2^{16-k}\cdot (-1)^k\cdot x^{16-4k}.
	\end{eqnarray*}
Số hạng không chứa $x$ ứng với $16-4k=0\Leftrightarrow k=4$.\\
Vậy số hạng cần tìm là $C\mathrm{C}^4_{16}2^{12}(-1)^4=\mathrm{C}^4_{16}2^{12}.$
}
\end{bt}
\begin{bt}%[HK1 THPT Trương Vĩnh Ký, 2017-2018]$[Nhật Thiện]%[1D2B5-2]
	Một bình chứa $12$ viên bi, trong đó có $4$ viên bi trắng, $6$ viên bi đen, $2$ viên bi đỏ. Lấy ngẫu nhiên $7$ viên bi. Tính xác suất lấy được $3$ viên bi trắng.
	\loigiai{
	Lấy ngẫu nhiên $7$ viên bi từ $12$ viên bi suy ra  $n(\Omega)=\mathrm{C}^7_{12}$.\\
	Gọi $A$ là biến cố lấy được $3$ viên bi trắng suy ra $n(A)=\mathrm{C}^3_4\cdot \mathrm{C}^4_8$.\\
	Vậy $\mathrm{P}(A)=\dfrac{n(A)}{n(\Omega)}=\dfrac{35}{99}$.
}
\end{bt}
\begin{bt}%[HK1 THPT Trương Vĩnh Ký, 2017-2018]$[Nhật Thiện]%[1D3B3-3]
	Tìm số hạng đầu và công sai của cấp số cộng $(u_n)$, biết rằng $\heva{&u_2+u_5-u_3=10\\ &u_4+u_6=26}$.
	\loigiai{
	Gọi $u_1$ là số hạng đầu tiên, $d$ là công sai. Ta có
	\begin{eqnarray*}
		& &\heva{&u_2+u_5-u_3=10\\ &u_4+u_6=26}\\
		&\Leftrightarrow& \heva{&(u_1+d)+(u_1+4d)-(u_1+2d)=10\\ &(u_1+3d)+(u_1+5d)=26}\\
		&\Leftrightarrow& \heva{&u_1+3d=10\\ &2u_1+8d=26}\\
		&\Leftrightarrow& \heva{&u_1=1\\ &d=3.}
	\end{eqnarray*}
}
\end{bt}

	Cho tứ diện $ABCD$ có các cạnh bằng nhau và bằng $6a$. Gọi $I$, $J$ lần lượt là trung điểm của $AC$ và $BC$. Gọi $K$ là một điểm trên cạnh $BD$ với $BK=2KD$.

		\item Xác định thiết diện của tứ diện $ABCD$ với mặt phẳng $(IJK)$. Thiết diện là hình gì? Tính diện tích thiết diện theo $a$.

	Ta có: $\heva{&D\in (DIJ)\cap (DAB)\\ &IJ\parallel AB\\ &IJ\subset (DIJ), AB\subset(DAB)}\Rightarrow (DIJ)\cap (DAB)=Dx\parallel AB\parallel IJ.$
