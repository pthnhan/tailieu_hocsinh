KHi đó: $z=\dfrac{3+4i}{i^{2019}}=\dfrac{3+4i}{-i}=-4+3i$.	
} 
\end{ex}

	}

\begin{bt}%[HKII, Tam Quan - Bình Định, 2018]%[Nguyễn Hồng Điệp, 12EX-9-2018]%[2D4B3-3]
	Tìm số phức $z$ thỏa mãn $\left(3+i\right)\overline{z}+\left(1+2i\right)z=3-4i$.
	\loigiai{
Gọi $z=a+bi, (a,b\in \mathbb{R})\Rightarrow \overline{z}=a-bi$.\\
Ta có: 
\begin{eqnarray*}
&\left(3+i\right)\overline{z}+\left(1+2i\right)z&=3-4i\\
\Leftrightarrow&\left(3+i\right)(a-bi)+\left(1+2i\right)(a+bi)&=3-4i\\
\Leftrightarrow& 4a-b+(3a-2b)i&=3-4i\\
\Leftrightarrow& \heva{&4a-b=3\\&3a-2b=-4}\quad \Leftrightarrow \heva{&a=2\\&b=5.}
\end{eqnarray*}
Vậy $z=2+5i$.	}
\end{bt}
\begin{bt}%[HKII, Tam Quan - Bình Định, 2018]%[Nguyễn Hồng Điệp, 12EX-9-2018]%[2H3B3-2]%[2H3B3-6]%[2H3B3-7]
Trong không gian với hệ tọa độ $Oxyz$, cho  mặt phẳng $(P)\colon2x-y+2z+4=0$ và điểm $M\left(2;1;1\right)$.
\begin{enumerate}
	\item Viết phương trình đường thẳng $d$ đi qua $M$ và vuông góc với mặt phẳng $(P)$.
\item Tìm hình chiếu vuông góc của điểm $M$ trên mặt phẳng $(P)$.
\item Viết phương trình mặt cầu $(S)$ tâm $M$ và tiếp xúc với mặt phẳng $(P)$.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Véc-tơ chỉ phương của $d$ là $\overrightarrow{u}_d=\overrightarrow{n}_P=\left(2;-1;2\right)$.\\
	$d$ có phương trình tham số $\heva{&x=2+2t\\&y=1-t\\&z=1+2t}$.
	\item Tọa độ hình chiếu $H$ của $M$ lên $(P)$ là nghiệm hệ phương trình
	$$\heva{&2x-y+2z+4=0\\&x=2+2t\\&y=1-t\\&z=1+2t}\Leftrightarrow \heva{&t=-1\\&x=0\\&y=2\\&z=-1.}$$
	Vậy $H\left(0;2;-1\right)$.
	\item Ta có: $R=\mathrm{d}\left(M,(P)\right)=\dfrac{|4-4+2+4|}{\sqrt{4+1+4}}=3$.\\
	Mặt cầu $(S)$ có phương trình $(x-2)^2+(y-1)^2+(z-1)^2=9$.
\end{enumerate}
}
\end{bt}
