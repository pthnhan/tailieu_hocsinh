\begin{ex}%[Đề thi chuyên Toán Tin, Sở Giáo dục Hà Nội năm 2004]%[9D5G2]
    Chứng minh rằng số tự nhiên $$P = 1\cdot 2\cdot 3\cdots 2003\cdot 2004\cdot\left(1 + \dfrac{1}{2} + \dfrac{1}{3} + \cdots + \dfrac{1}{2003} + \dfrac{1}{2004}\right)$$ chia hết cho $2005$.
    \loigiai{
        Đặt $A=1\cdot 2\cdot 3\cdots 2003\cdot 2004, B=1 + \dfrac{1}{2} + \dfrac{1}{3} + \cdots + \dfrac{1}{2003} + \dfrac{1}{2004}$
        
        Ta có:
        
        $\begin{array}{rl}
            B &=\left(1+\dfrac{1}{2004}\right)+\left(\dfrac{1}{2}+\dfrac{1}{2003}\right)+\ldots+\left(\dfrac{1}{1002}+\dfrac{1}{1003}\right) \\
             &=\dfrac{2005}{2004}+\dfrac{2005}{2\cdot 2003}+\ldots+\dfrac{2005}{1002\cdot 1003} \\
             &=2005\underbrace{\left(\dfrac{1}{2004}+\dfrac{1}{2\cdot 2003}+\ldots+\dfrac{1}{1002\cdot 1003}\right)}_{C}
        \end{array}$
        
        Suy ra: $P=2005\cdot A\cdot C$.
        
        Mà $A\cdot C$ là số nguyên nên ta có $P$ chia hết cho $2005$.
    }
\end{ex}
\begin{ex}%[Đề thi chuyên Toán Tin, Sở Giáo dục Hà Nội năm 2004]%[9D4G7]
	Cho phương trình: $x+3\left(m-3x^2\right)^2=m$   
	\begin{enumerate}
	    \item Giải phương trình khi $m=2$.
	    \item Tìm  $m$ để phương trình có nghiệm.
    \end{enumerate}
	\loigiai{
	    Đặt $m-3x^2=y\Rightarrow y+3x^2=m$. Khi đó ta có hệ: $\heva{&x+3y^2=m&(1) \\&y+3x^2=m&(2)}$
	    
	    Lấy (1) trừ (2) vế với vế ta được:
	    $$\left(x-y\right)+3\left(y-x\right)\left(y+x\right)=0\Leftrightarrow \left(y-x\right)\left(3x+3y-1\right)=0\Leftrightarrow \hoac{&x=y \\&3x+3y=1}$$
	    
	    Thế vào phương trình (1) ta được: $\hoac{&3x^2+x-m=0 \\&9x^2-3x+1-3m=0}\qquad (*)$
	    \begin{enumerate}
	        \item Với $m=2$, ta có: $\hoac{&3x^2+x-2=0 \\&9x^2-3x-5=0}\Leftrightarrow \hoac{&x=-1 \\&x=\dfrac{2}{3} \\&x=\dfrac{1\pm\sqrt{21}}{6}}$
	        
	        Vậy với $m=2$, phương trình có nghiệm $x\in\left\{-1;\dfrac{2}{3};\dfrac{1\pm\sqrt{21}}{6}\right\}$.
	        \item Phương trình đã cho có nghiệm khi hệ $(*)$ có nghiệm
	        $$\Leftrightarrow \hoac{&\Delta_1\ge 0 \\&\Delta_2\ge 0}\Leftrightarrow \hoac{&1+12m\ge 0 \\&9-36(1-3m)\ge 0}\Leftrightarrow m\ge-\dfrac{1}{12}$$
	        Vậy phương trình có nghiệm khi $m\ge -\dfrac{1}{12}$.
	    \end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi chuyên Toán Tin, Sở Giáo dục Hà Nội năm 2004]%[9D5G1]
	Giải bất phương trình sau: $\sqrt[3]{25x\left(2x^2+9\right)}\ge 4x+\dfrac{3}{x}$.
	\loigiai{
	    Điều kiện: $x\ne 0$.
	    \begin{itemize}
	        \item Với $x>0$, bất phương trình tương đương với: $\sqrt[3]{25x^4\left(2x^2+9\right)}\ge 4x^2+3 \qquad (1)$
	        
	        Áp dụng bất đẳng thức AM-GM cho 3 số dương ta có:
	        
	        $5x^2+5x^2+\left(2x^2+9\right)\ge \sqrt[3]{25x^4\left(2x^2+9\right)}$ hay $4x^2+3 \ge \sqrt[3]{25x^4\left(2x^2+9\right)} \qquad (2)$
	        
	        Từ (1), (2) suy ra để bất phương trình có nghiệm thì dấu đẳng thức phải xảy ra ở bất đẳng thức (2), lúc đó $5x^2=2x^2+9\Leftrightarrow x^2=3\Leftrightarrow x=\sqrt3$ (vì $x>0$).
	        \item Với $x<0$, bất phương trình tương đương với: $\sqrt[3]{25x^4\left(2x^2+9\right)}\le 4x^2+3$
	        
	        Suy ra bất phương trình đúng với $\forall x<0$.
	        
	        Vậy tập nghiệm của bất phương trình là: $x\in\left(-\infty;0\right)\cup\left\{\sqrt{3}\right\}$.
	    \end{itemize}
    }
\end{ex}
\begin{ex}%[Đề thi chuyên Toán Tin, Sở Giáo dục Hà Nội năm 2004]%[9H3G3]%[9D5G1]
    Cho tam giác $ABC$ có ba góc nhọn, kẻ hai đường cao $BE$, $CF$.
    \begin{enumerate}	
        \item Biết góc $\widehat{BAC}=60^\circ$, tính độ dài $EF$ theo $BC=a$.   
        \item Trên nửa đường tròn đường kính $BC$ không chứa $E$, $F$ lấy một điểm $M$ bất kỳ. Gọi $H$, $I$, $K$ lần lượt là hình chiếu vuông góc của $M$ trên $BC$, $CE$, $EB$. Tìm giá trị nhỏ nhất của tổng $S=\dfrac{BC}{MH}+\dfrac{CE}{MI}+\dfrac{EB}{MK}$
    \end{enumerate}
	\loigiai{
	    \begin{tikzpicture}[thick]
	        \tkzInit[ymin=-4.1,ymax=7.5,xmin=-4,xmax=10]
	        \tkzClip
	        \tkzDefPoints{0/0/B,1/0/X,8/0/C,5.75/0/H}
	        \tkzDefMidPoint(B,C)\tkzGetPoint{O}
	        \tkzDrawCircle(O,B)
	        \tkzDefLine[orthogonal=through X](B,C) \tkzGetPoint{d}
	        \tkzInterLC(X,d)(O,B)\tkzGetSecondPoint{F}
	        \tkzInterCC(F,O)(O,B)\tkzGetFirstPoint{E}
	        \tkzInterLL(B,F)(C,E)\tkzGetPoint{A}
	        \tkzDefLine[orthogonal=through H](B,C) \tkzGetPoint{e}
	        \tkzInterLC(H,e)(O,B)\tkzGetSecondPoint{M}
	        \tkzDefLine[orthogonal=through M](A,C) \tkzGetPoint{f}
	        \tkzInterLL(M,f)(A,C)\tkzGetPoint{I}
	        \tkzInterLL(I,H)(B,E)\tkzGetPoint{K}
	        \tkzDefLine[parallel=through E](B,C) \tkzGetPoint{h}
	        \tkzInterLC(E,h)(O,B)\tkzGetFirstPoint{D}
	        \tkzInterLL(M,D)(B,C)\tkzGetPoint{P}
	        \tkzDrawPoints[fill=blue](B,C,O,F,E,A,H,M,I,K,P)
	        \tkzDrawSegments(A,B B,C A,I B,E C,F M,K M,H M,I M,P M,B M,C M,E E,F E,O F,O)
	        \tkzLabelPoints[above](A,P,H)
	        \tkzLabelPoints[left](B)
	        \tkzLabelPoints[right](I)
	        \tkzLabelPoints[below](O,M)
	        \tkzLabelPoints[above right](E,C)
	        \tkzLabelPoints[above left](F,K)
	        \tkzMarkAngle[arc=l,size=1 cm,fill=red,opacity=0.5](P,M,B)
	        \tkzMarkAngle[arc=l,size=1 cm,fill=red,opacity=0.5](C,M,E)
	        \tkzMarkRightAngles(M,H,C M,I,C M,K,B)
	    \end{tikzpicture}
	    \begin{enumerate}
	        \item Gọi $O$ là trung điểm $BC$.
	        
	        Do $\widehat{BAC}=60^\circ \Rightarrow \widehat{ACF}=30^\circ \Rightarrow \widehat{EOF}=2\widehat{ECF}=60^\circ\Rightarrow \triangle EOF$ đều.
	        
	        Từ đó suy ra: $EF=EO=\dfrac{BC}{2}=\dfrac{a}{2}$.
	        \item Lấy điểm $P$ trên $BC$ sao cho $\widehat{BMP}=\widehat{EMC}$.
	        
	        Ta có: $\triangle CME\sim\triangle PMB$ (g-g) $\Rightarrow \dfrac{BP}{MH}=\dfrac{EC}{MI}$.
	        
	        Lại có: $\triangle MEB\sim\triangle MCP$ (g-g) $\Rightarrow \dfrac{EB}{MK}=\dfrac{CP}{MH}$.
	        
	        Suy ra: $\dfrac{BP+PC}{MH}=\dfrac{EC}{MI}+\dfrac{EB}{MK}\Rightarrow\dfrac{BC}{MH}=\dfrac{EC}{MI}+\dfrac{EB}{MK}\Rightarrow S=\dfrac{2BC}{MH}$.
	        
	        Để $S$ đạt giá trị nhỏ nhất thì $MH$ lớn nhất, hay $M$ là điểm chính giữa cung $\wideparen{BC}$.
	        
	        Khi đó: $H\equiv O\Rightarrow MH=MO\Rightarrow S=\dfrac{2BC}{MO}=4$.
	        
	        Vậy giá trị nhỏ nhất của $S$ là 4 khi $M$ là điểm chính giữa cung $\wideparen{BC}$.
	    \end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi chuyên Toán Tin, Sở Giáo dục Hà Nội năm 2004]%[9D5G4]
	Cho một đa giác có chu vi bằng $1$, chứng minh rằng có một hình tròn bán kính $r=\dfrac{1}{4}$ chứa toàn bộ đa giác đó.
	\loigiai{
	    \immini{
	        Lấy điểm $A$ trên biên của đa giác. Lấy điểm $B$ trên biên đa giác sao cho $AB$ chia chu vi đa giác thành 2 phần có độ dài mỗi phần bằng $\dfrac{1}{2}$.
		
    		Gọi $O$ là trung điểm của $AB$.
    		
    		Giả sử $M$ là một điểm tùy ý thuộc biên hoặc miền trong đa giác, lấy $N$ đối xứng với $M$ qua $O$.
    		
    		Khi đó ta có tứ giác $AMBN$ là hình bình hành và $AM+MB<\dfrac{1}{2}$.
    		
    		Mà $MN<AM+AN=AM+MB\Rightarrow MN<\dfrac{1}{2}\Rightarrow OM<\dfrac{1}{4}$, suy ra $M$ nằm trong đường tròn tâm $O$ bán kính $r=\dfrac{1}{4}$.
	    }
	        {
	        \begin{tikzpicture}[thick]
		        \tkzInit[ymin=-1,ymax=7,xmin=-0.5,xmax=7]
                \tkzClip
                \tkzDefPoints{0/3/C,6/1/D,7/4/E,6/6/F,4/6/G,1/5/H,5/6/A,3/2/B,2/3.5/M}
                \tkzDefMidPoint(A,B)\tkzGetPoint{O}
                \tkzDefPointBy[symmetry=center O](M)\tkzGetPoint{N}
                \tkzDrawPoints[fill=blue](A,B,O,M,N)
                \tkzDrawSegments(C,D D,E E,F F,G G,H H,C A,B A,M M,B A,N N,B M,N)
                \tkzLabelPoints[above](A)
                \tkzLabelPoints[below left](B)
                \tkzLabelPoints[left](M)
                \tkzLabelPoints[below right](O,N)
            \end{tikzpicture}
        }	
	}
\end{ex}
