
\section{BẤT ĐẲNG THỨC VÀ CỰC TRỊ DẠNG ĐA THỨC}

\begin{center}
 \large \textbf{TỔNG QUAN VỀ CHUYÊN ĐỀ}
\end{center}
Bất đẳng thức và toán cực trị là một nội dung quan trọng và hấp dẫn. Các bài toán trong chuyên đề này bao gồm:
\begin{itemize}
	\item Bất đẳng thức dạng đa thức, các cách chứng minh một bất đẳng thức, các bất đẳng thức thông dụng, trong đo có bất đẳng thức Cô-si và bất đẳng thức Bu-nhi-a-cốp-xki.
	\item Tìm giá trị nhỏ nhất, giá trị lớn nhất của các biểu thức có dạng đa thức với một biến hoặc nhiều biến.
\end{itemize}
\textbf{Vài nét lịch sử}
\begin{center}
	\textbf{CÔ-SI VÀ BU-NHI-A-CỐP-XKI}
\end{center}
Cô-si (\textit{Augustin Louis Cauchy}, 1789 - 1857) là nhà toán học Pháp, Viện sĩ Viện Hàn Lâm  Khoa học Pa-ri. Ông có trên 800 công trình về nhiều lĩnh vực như Số học, Giải tích, Cơ học, Quang học, Thiên văn học. Ông đã xây dụng nhiều vấn đề lí thuyết một cách chặt chẽ, khoa học, giúp cho Toán học có những bước tiến đáng kể.\medskip

Bất đẳng thức về trung bình cộng và trung bình nhân của $n$ số không âm là 

$$\dfrac{a_1+a_2+a_3+\ldots+a_n}{n}\geq\sqrt[n]{a_1a_2\ldots a_n}.$$

Bất đẳng thức trên cũng được gọi là \textit{bất đẳng thức Cô-si} vì Cô-si đã đưa ra một cách chứng minh độc đáo, mặc dù ông không phải là người đầu tiên đề xuất bất đẳng thức này.\medskip

Bu-nhi-a-cốp-xki (\textit{Viktor Bunyakovsky}, 1804 - 1889) là nhà toán học Nga, Phó Chủ tịch Viện Hàn lâm khoa học Pê-tec-bua. Ông học Toán tại Pa-ri và là học trò của Cô-si.\medskip

Ông có 128 công trình về Lí thuyết số, Lí thuyết xác suất, Giải tích, Hình học và Đại số. Ông có nhiều đóng góp nâng cao trình độ khoa học trong giảng dạy toán học ở các trường đại học và trung học. Để ghi nhớ công lao của ông với nền giáo dục Nga, năm 1875, một giải thường mang tên ông đã được lập ra để trao cho những sáng tạo về Toán.\medskip

Bất đẳng thức Bu-nhi-a-cốp-xki với hai bộ $n$ số $(a_1,a_2,a_3,\ldots,a_n)$ và $(b_1,b_2,b_3,\ldots,b_n)$ là $$(a_1^2+a_2^2+\ldots+a_n^2)(b_1^2+b_2^2+\ldots+b_n^2)\geq(a_1b_1+a_2b_2+\ldots+a_nb_n)^2.$$

Bất đẳng thức trên còn được gọi là \textit{bất đẳng thức Cô-si - Bu-nhi-a-cốp-xki - Svac}, vì Cô-si đã đề xuất bất thức đó, Bu-nhi-a-cốp-xki đã mở rộng kết quả cho tích phân, còn Svac (\textit{Schwarz}, nhà toán học Đức, 1843 - 1921) mở rộng kết quả trên cho không gian vec-tơ.\medskip

\noindent \textbf{Bài toán thực tế}
\begin{center}
	\textbf{KHU ĐẤT NHỐT GIA SÚC}
\end{center}
Bác Tâm có một cuộn lưới sắt dài $60$ (m), bác muốn dùng lưới căng thành ba đoạn thẳng $AB$, $BC$, $CD$ cùng với bức tường có sẵn làm thành một hình chữ nhất $ABCD$ để nhốt gia súc.\\
Hãy tính độ dài $AB$ để khu đất hình chữ nhật $ABCD$ có diện tích lớn nhất.\\
\textbf{Giải}\\
Đặt $AB=CD=x$ (m) thi $BC=60-2x$ (m).\\
Diện tích $S$ của hình chữ nhật $ABCD$ bằng 
\begin{align*}
S&=x(60-2x)=-2x^2+60x=-2(x^2-30x)\\
&=-2(x^2-30x+225)+450=-2(x-15)^2+450\leq450.
\end{align*}
Vậy giá trị lớn nhất của khu đất là $450$ (m$^2$) khi $AB=x=15$ (m).
\subsection{Bất đẳng thức dạng đa thức}
\begin{enumerate}[1.]
	\item Ta gọi hệ thức dạng $a>b$ (hoặc $a<b$, $a\geq b$, $a\leq b$) là một bất đẳng thức.
	\item Để chứng minh bất đẳng thức, ta thường dùng các cách sau:
	\begin{itemize}
		\item Dùng định nghĩa : Để chứng minh $a>b$, ta chứng minh $a-b>0$.
		\item Dùng biến đổi tương đường : Chứng tỏ bất đẳng thức phải chứng minh tương đương với một bất đẳng thức đúng.
		\item Dùng tính chất của bất đẳng thức : Từ các bất đẳng thức đã biết, suy ra bất đẳng thức phải chứng minh.
		\item Dùng phương pháp phản chứng : Để chứng minh $a\geq b$, ta chứng tỏ $a<b$ là \textbf{sai}.
	\end{itemize}
\item Cần nhớ một số bất đẳng thức thông dụng sau:
\begin{enumerate}
	\item Liên hệ giữa trung bình cộng và trung bình nhân (bất đẳng thức Cô-si):
	\begin{align}
	\dfrac{a+b}{2}\geq\sqrt{ab} \textrm{ với } a,b\geq0
	\end{align}
	Xảy ra đẳng thức khi và chỉ khi $a=b$.
	\item Liên hệ giữa tổng các bình phương và bình phương của tổng:
	\begin{align}
	&2(a^2+b^2)\geq(a+b)^2\label{2}\\
	&3(a^2+b^2+c^2)\geq(a+b+c)^2\geq3(ab+bc+ca)\label{3}
	\end{align}
	Tổng quát của bất đẳng thức \eqref{2} và \eqref{3}, ta có bất đẳng thức Bu-nhi-a-cốp-xki:
	\begin{align}
	&(a^2+b^2)(x^2+y^2)\geq(ax+by)^2\label{4}\\
	&(a^2+b^2+c^2)(x^2+y^2+z^2)\geq(ax+by+cz)^2\label{5}
	\end{align}
	Xảy ra đẳng thức khi và chỉ khi $\dfrac{a}{x}=\dfrac{b}{y}=\dfrac{c}{z}$, với $x,y,z\ne0$.\\
	Bất đẳng thức \eqref{2} là trường hợp đặc biệt của bất đẳng thức \eqref{4} với $x=y=1$.\\
	Bất đẳng thức \eqref{3} là trường hợp đặc biệt của bất đẳng thức \eqref{5} với $x=y=z=1$.\\
	Khi $a,b,c,x,y,z$ không âm thì các bất đẳng thức \eqref{4} và \eqref{5} còn được viết dưới dạng
	\begin{align*}
	&(a+b)(x+y)\geq\left(\sqrt{ax}+\sqrt{by}\right)^2\\
	&(a+b+c)(x+y+z)\geq\left(\sqrt{ax}+\sqrt{by}+\sqrt{cz}\right)^2
	\end{align*}
\end{enumerate}
\end{enumerate}
\begin{vd}
Chứng minh bất đẳng thức $(a^2-b^2)(c^2-d^2)\leq(ac-bd)^2$\hfill $(1)$
\loigiai{\begin{align*}
	(1)&\Leftrightarrow a^2c^2-a^2d^2-b^2c^2+b^2d^2\leq a^2c^2-2abcd+b^2d^2\\
	&\Leftrightarrow a^2d^2-2abcd+b^2c^2\geq0\Leftrightarrow(ad-bc)^2\geq0
	\end{align*}
Bất đẳng thức cuối đúng, vậy bất đẳng thức $(1)$ đúng.\\
Đẳng thức xảy ra khi và chỉ khi $ad=bc$.}
\end{vd}
\begin{vd}
Chứng minh với $x,y$ không âm, ta có bất đẳng thức $(x+y)(x^3+y^3)\geq(x^2+y^2)^2$.
    \loigiai{\begin{itemize}
    		\item \textbf{Cách 1:} Xét hiệu hai vế:
    		\begin{align*}
    		&(x+y)(x^3+y^3)-(x^2+y^2)^2\\
    		=&x^4+xy^3+x^3y+y^4-x^4-2x^2y^2-y^4\\
    		=&xy(x^2+y^2-2xy)=xy(x-y)^2\geq0
    		\end{align*}
    		Vậy $(x+y)(x^3+y^3)\geq(x^2+y^2)^2$.\\
    		Đẳng thức xảy ra khi và chỉ khi $x=0$, hoặc $y=0$, hoặc $x=y$.
    		\item \textbf{Cách 2:} Với $a,b,x,y\geq0$, theo bất đẳng thức Bu-nhi-a-cốp-xki $$(a+b)(x+y)\geq\left(\sqrt{ax}+\sqrt{by}\right)^2$$
    		Với $a=x^3$, $b=y^3$, ta có $$(x^3+y^3)(x+y)\geq\left(\sqrt{x^3\cdot x}+\sqrt{y^3\cdot y}\right)^2=(x^2+y^2)^2.$$
    		\end{itemize}}
\end{vd}
\begin{vd}
Chứng minh bất đẳng thức $x+y\leq xy+1$ với $x\geq1$, $y\geq1$.
\loigiai{Do $x\geq1$, $y\geq1$ nên $(x-1)(y-1)\geq0\Rightarrow xy-x-y+1\geq0\Rightarrow x+y\geq xy+1$.\\
Đẳng thức xảy ra khi và chỉ khi $x=1$ hoặc $y=1$.}
\end{vd}

\begin{vd}
Cho $a+b\geq2$. Chứng minh bất đẳng thức 
\begin{align}
a^3+b^3\geq a^2+b^2.\label{1}
\end{align}
\loigiai{\begin{itemize}
		\item \textbf{Cách 1:}
		\begin{align}
		\eqref{1}\Leftrightarrow a^3-a^2+b^3-b^2\geq0\Leftrightarrow a^2(a-1)+b^2(b-1)\geq0\label{2}
		\end{align}
		Đặt $a=1+x$ và $b=1+y$. Do $a+b\geq2$ nên $x+y\geq0$. Ta có
		\begin{align}
		\eqref{2}\Leftrightarrow(1+x)^2x+(1+y)^2y\geq0\Leftrightarrow x+y+2(x^2+y^2)+(x^3+y^3)\geq0\label{3}
		\end{align}
		Ta có $x^2+y^2\geq0$ và do $x+y\geq0$ nên $x^3+y^3=(x+y)(x^2-xy+y^2)\geq0$.\\
		Suy ra \eqref{3} đúng. Vậy \eqref{1} đúng. Đẳng thức xảy ra khi và chỉ khi $a=1$, $b=1$.
		\item \textbf{Cách 2:} Do $a+b\geq2$ nên $a+b-2\geq0$.\\
		Để chứng minh $a^3+b^3-a^2-b^2\geq0$, ta chứng minh $a^3+b^3-a^2-b^2\geq a+b-2$\hfill$(*)$
		\begin{align}
		(*)&\Leftrightarrow a^3+b^3-a^2-b^2-a-b+2\geq0\nonumber\\
		&\Leftrightarrow a^3-a^2-a++b^3-b^2-b+1\geq0\nonumber\\
		&\Leftrightarrow (a-1)^2(a^2+a+1)+(b-1)^2(b^2+b+1)\geq0\label{4}
		\end{align}
		Do \eqref{4} đúng nên $(*)$ đúng. Vậy \eqref{1} đúng.
\end{itemize}}
\end{vd}
\begin{vd}
Cho các số $a$ và $b$ không âm thỏa mãn $a^2+b^2=1$. Chứng minh các bất đẳng thức sau:
\begin{multicols}{2}
	\begin{enumerate}
		\item $a^3+b^3\leq1$.
		\item $a^3+b^3\geq\dfrac{1}{\sqrt{2}}$.
	\end{enumerate}
\end{multicols}
\loigiai{\begin{enumerate}
		\item Do $a^2+b^2=1$ nên $a\leq1$ và $b\leq1$. Suy ra $a^3\leq a^2$ và $b^3\leq b^2$.\\
		Do đó nên $a^3+b^3\leq a^2+b^2=1$.\\
		Đẳng thức xảy ra khi và chỉ khi $a=b=1$.
		\item Ta chứng minh được $(a+b)(a^3+b^3)\geq(a^2+b^2)^2$.\\
		$\Rightarrow a^3+b^3\geq\dfrac{(a^2+b^2)^2}{a+b}=\dfrac{1}{a+b}$\hfill$(1)$\\
		Ta lại có $(a+b)^2\leq2(a^2+b^2)=2\Rightarrow a+b\leq\sqrt{2}$\hfill$(2)$\\
		Từ $(1)$ và $(2)$ suy ra $a^3+b^3\geq\dfrac{1}{\sqrt{2}}$.\\
		Đẳng thức xảy ra khi và chỉ khi $a=b=\dfrac{\sqrt{2}}{2}$.
	\end{enumerate}}
\end{vd}
\subsection{Cực trị dạng đa thức}
\begin{vd}
Tìm giá trị nhỏ nhất của biểu thức $A=x^2-3x+3$ với 
\begin{multicols}{2}
	\begin{enumerate}
		\item $x$ bất kì.
		\item $x\geq2$.
	\end{enumerate}
\end{multicols}
\loigiai{\begin{enumerate}
		\item $A=x^2-3x+\dfrac{9}{4}+\dfrac{3}{4}=\left(x-\dfrac{3}{2}\right)^2+\dfrac{3}{4}\geq\dfrac{3}{4}$.\\
		Vậy $\min A=\dfrac{3}{4}$ tại $x=\dfrac{3}{2}$.
		\item \begin{itemize}
			\item \textbf{Cách 1:} Do $x\geq2$, ta đặt $x=2+y$ với $y\geq0$.\\
			Ta có $A=(2+y)^2-3(2+y)+3=y^2+y+1\geq1$ (do $y\geq0$).\\
			Vậy $\min A=1\Leftrightarrow y=0\Leftrightarrow x=2$.
			\item \textbf{Cách 2:} Ta có $A(2)=2^2-3\cdot2+3=1$.\\
			Ta sẽ chứng minh với $x\geq2$ thì $A\geq1$.\\
			Ta có $A-x^2-3x+2=(x-1)(x-2)$.\\
			Với $x\geq2$ thì $x-1>0$, $x-2\geq0$ nên $A-1\geq0$, do đó $A\geq1$.\\
			Vậy $\min A=1$ tại $x=2$.
		\end{itemize}
\end{enumerate}}
\end{vd}
\begin{note}
\begin{itemize}
	\item Khi biến $x$ nhận giá trị tùy ý thì tam thức bậc hai $ax^2+bx+c$, ($a\ne0$) đạt cực trị tại $x=-\dfrac{b}{2a}$.
	\item Khi biến $x$ nhận giá trị hạn chế không chứ $-\dfrac{b}{2a}$ thì tam thức bậc hai đạt cực trị tại giá trị biên của biến.
\end{itemize}
\end{note}
\begin{vd}
Tìm giá trị nhỏ nhất của các biểu thức sau:
\begin{multicols}{2}
	\begin{enumerate}
		\item $A=x^3-48x$ với $x\geq0$.
		\item $B=x^4-32x$.
	\end{enumerate}
\end{multicols}
\loigiai{\begin{enumerate}
		\item \begin{align*}
		A&=x^3-48x=x^3+8x^2-8x^2-64x+16x+128-128\\
		&=(x^2-8x+16)(x+8)-128=(x-4)^2(x+8)-128\geq-128.
		\end{align*}
		Vậy $\min A=-128$ tại $x=4$.
		\item \begin{align*}
		B&=x^4-32x=x^4+16-8x62+8x^2+32-32x-48\\
		&=(x^2-4)^2+8(x-2)^2-48\geq-48.
		\end{align*}
		Vậy $\min B=-48$ tại $x=2$.
\end{enumerate}}
\end{vd}
\begin{vd}
Tìm giá trị nhỏ nhất của $A=x^2(x+4)$ với $x\geq2$.
\loigiai{Do $x\geq2$, ta đặt $x=2+y$ với $y\geq0$. Ta có\\
\begin{align*}
A&=(2+y)^2(6+y)=y^3+10y^2+28y+24\\
&=y(y^2+10y+25)+3y+24=y(y+5)^2+3y+24\geq24.
\end{align*}
Vậy $\min A=24$ tại $y=0$, tức là tại $x=2$.}
\end{vd}

\begin{vd}
Cho các số $x$, $y$ không âm thỏa mãn $x^3+y^3=2$. Tìm giá trị lớn nhất của 
\begin{multicols}{2}
	\begin{enumerate}
		\item $A=x+y$.
		\item $B=x^2+y^2$.
	\end{enumerate}
\end{multicols}
\loigiai{\begin{enumerate}
		\item Ta có $2=x^3+y^3=(x+y)(x^2-xy+y^2)=(x^2-xy+y^2)\cdot A$\hfill$(1)$\\
		$x^2-xy+y^2=(x+y)^2-3xy=A^2-3xy$\hfill$(2)$\\
		$4xy\leq(x+y)^2=A^2\Rightarrow-3xy\geq-\dfrac{3}{4}A^2$\hfill$(3)$\\
		Từ $(2)$ và $(3)$ suy ra $x^2-xy+y^2\geq A^2-\dfrac{3}{4}A^2=\dfrac{A^2}{4}$.\hfill$(4)$\\
		Từ $(1)$ và $(4)$ suy ra $2\geq\dfrac{A^2}{4}\cdot A\Rightarrow A^3\leq8\Rightarrow A\leq2$.\\
		Vậy $\max A=2\Leftrightarrow\heva{&x=y\\&x^3+y^3=2}\Leftrightarrow x=y=1$.
		\item Với $x$, $y$ không âm ta có $(x+y)(x^3+y^3)\geq(x^2+y^2)^2$.\\
		Nên ta suy ra $2\cdot A\geq B^2\Rightarrow B^2\leq2A\leq2\cdot2=4$.\\
		Vậy $\max B=2\Leftrightarrow x=y=1$.
\end{enumerate}}
\end{vd}
\begin{vd}
	Cho $x\geq4$ và $x+y\geq6$. Tìm giá trị nhỏ nhất của $A=x^2+y^2$.
	\loigiai{Chọn $x$ nhỏ nhất là $4$, khi $x=4$ thì $y\geq2$.\\
		Theo bất đẳng thức Bu-nhi-a-cốp-xki ta có $(4^2+2^2)(x^2+y^2)\geq(4x+2y)^2$.\hfill$(1)$\\
		Do $2x\geq8$ và $2x+2y\geq12$ nên $4x+2y\geq20$.\hfill$(2)$\\
		Từ $(1)$ và $(2)$ suy ra $20A\geq20A^2\Rightarrow A\geq20$.\\
		Vậy $\min A=20\Leftrightarrow\heva{&\dfrac{x}{4}=\dfrac{y}{2}\\&x=4\\&x+y=6}\Leftrightarrow\heva{&x=4\\&y=2.}$
	}
\end{vd}
\begin{vd}
Cho các số dương $a$, $b$ thỏa mãn $a+b=1$. Tìm giá trị lớn nhất của biểu thức \begin{align*}
A=ab(a^2+b^2).
\end{align*}
\loigiai{Áp dụng bất đẳng thức Cô-si, ta có $\sqrt{2A}=\sqrt{2ab(a^2+b^2)}\leq\dfrac{2ab+a^2+b^2}{2}=\dfrac{(a+b)^2}{2}=\dfrac{1}{2}$\\
$\Rightarrow2A\leq\dfrac{1}{4}\Rightarrow A\leq\dfrac{1}{8}$. Vậy $\max A=\dfrac{1}{8}\Leftrightarrow a=b=\dfrac{1}{2}$.}
\end{vd}
\begin{vd}
Cho các số dương $a$, $b$, $c$ thỏa mãn $(a+b)(a+c)=8$. Tìm giá trị lớn nhất của 
\begin{align*}
A=abc(a+b+c).
\end{align*}
\loigiai{$(a+b)(a+c)=8\Rightarrow a(a+c)+ab+bc=8\Rightarrow a(a+b+c)+bc=8$.\hfill$(1)$\\
Áp dụng bất đẳng thức Cô-si  vào do $(1)$ ta có\\
$\sqrt{A}=\sqrt{a(a+b+c)\cdot bc}\leq\dfrac{a(a+b+c)+bc}{2}=\dfrac{8}{2}=4\Rightarrow A\leq16$.\\
Dấu bằng xảy ra khi và chỉ khi $a(a+b+c)=bc=4$.\\
Vậy $\max A=16$ khi chẳng hạn $\heva{&b=c=2\\&a+2=\sqrt{8}}\Leftrightarrow\heva{&b=c=2\\&a=2\sqrt{2}-2.}$}
\end{vd}

\begin{enumerate}
\item $a+b+c=5$.

		\item $3abc\left(a + b + c\right) \leq 1$ với $ab + bc + ca = 1$.

			\item Xét hiệu

			\left(a^4 + b^4 + c^4\right) - \left( a^3b + b^3c + c^3 a \right) & = a^3\left(a - b\right) + b^3 \left( b - c\right) + c^3 \left( c - a\right) \\

			& \geq 0

			\item Đặt $ab = x, bc = y, ca = z$ thì $x + y + z = 1$. Ta có\medskip

	}

\begin{bt}
	Tìm giá trị lớn nhất của các biểu thức sau:
	\begin{enumerate}[a)]
		
		\item $A = a^3 + b^3 + c^3,$ biết $0 \leq c \leq b \leq a \leq 2$ và $a + b + c = 3$;
		
		\item $B = a\left(b^2 + c^2\right) + b \left(c^2 + a^2\right) + c \left(a^2 + b^2\right)$ biết $a \geq b \geq c \geq 0$ và $a + b + c = 1$. 
	\end{enumerate}
	
	\loigiai{\begin{enumerate}[a)]
			\item $A = a^3 + b^3 + c^3 \leq a^3 + \left[ b^3 + c^3 + 3bc\left(b+ c\right) \right]  = a^3 + \left(b+ c\right)^3$ \medskip
			
			$= a^3 + \left( 3 - a\right)^3 = 9 \left( a^2 - 3a - 3\right) = 9 \left[ 1 + \left(a-1\right)\left(a-2\right)\right]$. \hspace{\fill} $(1)$\medskip
			
			Do $3 = a + b + c \leq 3a$ nên $a \geq 1$. Ta lại có $a \leq 2$ nên $\left(a - 1\right)\left(a-2\right) \leq 0$. \hspace{\fill} $(2)$\medskip
			
			Từ $(1)$ và $(2)$ suy ra $A \leq 9$; $\max A = 9 \Leftrightarrow a = 2, b = 1, c = 0$.
			
			\item $B = ab\left(a+b\right) + bc\left(b+c\right)+ca\left(c + a\right) = ab\left(1 - c\right) + bc\left(1-a\right)+ ca\left(1-b\right)$\medskip
			
			$ = \left(ab + bc + ca\right) - 3abc = ab - 3abc + ac + bc = ab\left(1 - 3c\right) + c \left(1 - c\right).$ \hspace{\fill} $(1)$\medskip
			
			Ta có $ab \leq \dfrac{\left(a+b\right)^2}{4} = \dfrac{\left(1 - c\right)^2}{4}$ \hspace{\fill} $(2)$\medskip
			
			Từ $(1)$ và $(2)$ suy ra \begin{align*}
			4B\leq \left(1 - 2c + c^2\right)\left(1 - 3c\right) + c\left(1-c\right) = -3c^3 + 3c^2 - c + 1 = -3c\left(c^2 - c + \dfrac{1}{3}\right) + 1. 
			\end{align*}
			
			Lại có $c\geq 0$ và $c^2 - c + \dfrac{1}{3} > 0$ nên $4B \leq 1$. \medskip
			
			$\max B = \dfrac{1}{4} \Leftrightarrow a = b = \dfrac{1}{2}$ và $c = 0$. 
		\end{enumerate}
	}
\end{bt}

\begin{bt}
	Cho các số $a, b, c$ thỏa mãn $0 \leq a, b, c \leq 1$. Tìm giá trị nhỏ nhất, giá trị lớn nhất của biểu thức $A = \left(a + b + c\right) - \left(ab + bc + ca\right)$. \medskip
	
	\loigiai{ \begin{enumerate}[$\bullet$]
			\item Ta có $A = \left(a + b + c\right) - \left(ab + bc + ca\right) = a\left(1 - b\right) + b \left(1-c\right) + c \left(1-a\right) \geq 0$ (do $0 \leq a, b, c \leq 1$).\medskip
			
			$\min A = 0 \Leftrightarrow a = b = c = 0$ hoặc $a = b = c = 1$. \medskip
			
			\item Do $a, b, c \leq 1$ nên $\left( 1-a\right)\left(1-b\right)\left(1-c\right) \geq 0$ \medskip
			
			$\Rightarrow \left(a + b + c\right) - \left(ab + bc + ca\right) \leq 1 - abc \leq 1$. \medskip
			
			$\max A = 1 \Leftrightarrow$ trong $a, b, c$ có một số bằng 1, một số bằng 0.
		\end{enumerate}
	}
\end{bt}

\begin{bt}
	Tìm giá trị nhỏ nhất, giá trị lớn nhất của $a$, biết: 
	
	a) $\heva{& a + b + c = 5 \\ & a^2 + b^2 + c^2 = 11};$ \hspace{4cm} b) $\heva{& a + b + c = 4 \\ & ab + bc + ca = 5};$ \medskip
	
	c) $\heva{& a + b + c + d = 1 \\ & a^2 + b^2 + c^2 + d^2 = 1}.$ \medskip
	
	\loigiai{ \begin{enumerate}
			\item Từ giả thiết suy ra $b + c = 5 - a$ và $b^2 + c^2 = 11 - a^2$. \medskip
			
			Do bất đẳng thức $2\left(b^2 + c^2\right) \geq \left(b+c\right)^2$ nên $2\left(11-a^2\right) \geq \left(5 - a\right)^2$ \medskip
			
			$\Rightarrow 3a^2 -10a + 3 \leq 0 \Leftrightarrow \dfrac{1}{3} \leq a \leq
			3$. \medskip
			
		    $\min a = \dfrac{1}{3} \Leftrightarrow b = c = \dfrac{7}{3}$; $\max a = 3 \Leftrightarrow b = c = 1$. \medskip
			
			\item Giải tương tự câu a) được $3a^2 - 8a + 4 \leq 0$ \medskip
			
			$\Leftrightarrow \left(a - 2\right) \left(3a - 2\right) \leq 0 \Leftrightarrow \dfrac{2}{3} \leq a \leq 2$. \medskip
			
			$\min a = \dfrac{2}{3} \Leftrightarrow b = c = \dfrac{5}{3}$; $\max a = 2 \Leftrightarrow b = c = 1$. \medskip
			
			\item Giải tương tự Ví dụ 97. \medskip
			
			$\min a = -\dfrac{1}{2} \Leftrightarrow b = c = d = \dfrac{1}{2}$. \medskip
			
			$\max a = 1 \Leftrightarrow b = c = d = 0$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}
	Tìm giá trị nhỏ nhất của $A = \left(a+b\right)\left(a + b + c\right),$ biết \medskip
	
	$\heva{& a + b + c + d = 4 \\ & abcd = \dfrac{1}{2} \\ & a, b, c, d > 0}.$ \medskip
	
	\loigiai{Áp dụng bất đẳng thức $\left(x+y\right)^2 \geq 4xy,$ ta có \medskip
		
		$\left(a + b\right)^2 \geq 4ab$ \medskip
		
		$\left(a + b + c\right)^2 \geq 4c\left(a + b\right)$ \medskip
		
		$\left(a + b + c + d\right)^2 \geq 4d \left(a + b + c\right).$ \medskip
		
		Suy ra $A^2\left(a + b + c + d\right)^2 \geq 64abcd \left(a+b\right)\left(a+b+c\right)$ \medskip
		
		$\Rightarrow 16A^2 \geq 64 \times \dfrac{1}{2}A \Rightarrow A \geq 2$ \medskip
		
		$\min A = 2 \Leftrightarrow a = b =\dfrac{1}{2}, c = a, d= 2$.}
	
\end{bt}
