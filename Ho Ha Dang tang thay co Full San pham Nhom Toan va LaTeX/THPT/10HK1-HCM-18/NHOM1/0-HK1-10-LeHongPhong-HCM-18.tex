\begin{bt}%[Đề HK1, Lê Hồng Phong, Hồ Chí Minh 17-18]%[Hoàng Hải, 10EX-HK1]%[0D3B2-4]
Giải phương trình $ \sqrt{2x^2+3x-1}=x+1 $.
\loigiai{
	Phương trình tương đương
	$$ \heva{&x+1\ge 0 \\ &2x^2+3x-1=(x+1)^2} \Leftrightarrow \heva{&x\ge -1 \\ &x^2+x-2=0} \Leftrightarrow \heva{&x\ge -1 \\ &x=1\vee x=-2} \Leftrightarrow x=1. $$
	 Vậy tập nghiệm của phương trình là $ S=\{1\} $.
}	
\end{bt}

\begin{bt}%[Đề HK1, Lê Hồng Phong, Hồ Chí Minh 17-18]%[Hoàng Hải, 10EX-HK1]%[0D3K3-4]
Giải hệ phương trình $ \heva{&x^2+y^2+3xy=-1 \\ &x+y-xy=1.} $
\loigiai{
	Hệ phương trình tương đương 
	\begin{eqnarray*}
		& & \heva{&(x+y)^2+xy=-1 \\ &x+y-xy=1} \\
		& \Leftrightarrow & \heva{&(x+y)^2+(x+y)=0 \\ &x+y-xy=1}\\
		& \Leftrightarrow & \heva{&(x+y)(x+y+1)=0 \\ &x+y-xy=1}\\
		& \Leftrightarrow & \heva{&x+y=0 \\ &x+y-xy=1} \vee \heva{&x+y+1=0 \\ &x+y-xy=1}\\
		& \Leftrightarrow & \heva{&x=-y \\ &y^2=1} \vee \heva{&x=-1-y \\ &y^2+y-2=0}\\
		& \Leftrightarrow & \heva{&x=1 \\ &y=-1} \vee \heva{&x=-1\\&y=1} \vee \heva{&x=-2 \\ &y=1} \vee \heva{&x=1 \\ &y=-2.}
	\end{eqnarray*}
	Vậy tập nghiệm của hệ phương trình là $ S=\{(1;-1),(-1;1),(-2;1),(1,-2) \} $.
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Hồng Phong, Hồ Chí Minh 17-18]%[Hoàng Hải, 10EX-HK1]%[0D2B3-2]
Tìm Parabol $ (P)\colon y=ax^2+bx+c$ $(a\ne 0) $ biết $ (P) $ có đỉnh $I(2;1)$ và đi qua điểm $A(3;2)$.
\loigiai{
	Ta có 
	\begin{eqnarray*}
		& &\heva{&(P)\text{ có đỉnh }I(2;1) \\ &(P)\text{ đi qua }A(3;2)}\\
		& \Leftrightarrow & \heva{&\dfrac{-b}{2a}=2\\&4a+2b+c=1\\&9a+3b+c=2}\\
		& \Leftrightarrow & \heva{&4a+b=0\\&4a+2b+c=1\\&9a+3b+c=2}\\
		& \Leftrightarrow & \heva{&a=1\\&b=-4\\&c=5}
	\end{eqnarray*}
	Vậy Parabol $ (P)\colon y=x^2-4x+5 $.
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Hồng Phong, Hồ Chí Minh 17-18]%[Hoàng Hải, 10EX-HK1]%[0D3G2-6]
Cho phương trình $ (x-3)(x^2-2mx+m-4)=0 $ $ (1) $ với $ m $ là tham số. Định $m$ để phương trình $ (1) $ có ba nghiệm phân biệt, trong đó có hai nghiệm dương và một nghiệm âm.	
\loigiai{
	Phương trình tương đương $ \hoac{&x-3=0 \\ &x^2-2mx+m-4=0} \Leftrightarrow \hoac{&x=3 \\ &x^2-2mx+m-4=0 \quad (2)} $\\
	Yêu cầu bài toán trở thành tìm $ m $ để phương trình $ (2) $ có hai nghiệm $ x_1,x_2 $ trái dấu khác $ 3 $, điều này tương đương với
	\begin{equation*}
		\heva{&\Delta'>0\\&x_1x_2<0 \\ &9-6m+m-4\ne 0}
		\Leftrightarrow \heva{&a\cdot c < 0 \\&9-6m+m-4\ne 0 }
		\Leftrightarrow \heva{&m-4<0 \\ &m\ne 1.}
		\Leftrightarrow \heva{&m<4 \\ &m\ne 1}
	\end{equation*}
	Vậy $m<4$ và $ m\ne 1 $ thì thỏa yêu cầu bài toán.
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Hồng Phong, Hồ Chí Minh 17-18]%[Hoàng Hải, 10EX-HK1]%[0D3G3-2]
Tìm các giá trị nguyên âm của $ m $ để hệ phương trình $ \heva{&mx+y=2m+1 \\ &x+my=3m} $ có nghiệm duy nhất $ (x;y) $ sao cho $ x,y $ là các số nguyên.
\loigiai{
	Xét $ m<0 $, hệ phương trình tương đương 
		$ \heva{&mx+y=2m+1 \\ &mx+m^2y=3m^2} \Leftrightarrow \heva{&mx+y=2m+1&(1) \\ &(m^2-1)y=3m^2-2m-1 &\quad(2)}$
	Xét phương trình $ (2) $, với $ m< 0 $, ta có
		\begin{eqnarray*}
		 (2)& \Leftrightarrow & (m-1)(m+1)y=(m-1)(3m+1) \\
			& \Leftrightarrow & (m+1)y=3m+1 \\
			& \Leftrightarrow & y=\dfrac{3m+1}{m+1}=\dfrac{3(m+1)-2}{m+1}=3-\dfrac{2}{m+1}.
		\end{eqnarray*}
	Để $ y $ là số nguyên thì $ (m+1)\in\{-1;1;-2;2\} \Rightarrow m \in \{-2;0;-3;1 \} \Rightarrow m \in \{-2;-3 \} $ (vì $ m<0 $).
	\begin{itemize}
		\item Với $ m=-2 $ ta có $\heva{&x=4 \\ &y=5}$ (nhận).
		\item Với $ m=-3 $ ta có $\heva{&x=3 \\ &y=4}$ (nhận).
	\end{itemize}
	Vậy $ m=-3 $ hoặc $ m=-2 $ thì thỏa yêu cầu bài toán.
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Hồng Phong, Hồ Chí Minh 17-18]%[Hoàng Hải, 10EX-HK1]%[0D4G2-5]
Định $ m $ để bất phương trình $ (m^2+2m-3)x+2m-1\le 0 $ vô nghiệm ($ m $ là tham số).
\loigiai{
	Bất phương trình tương đương $ (m^2+2m-3)x\le-2m+1 $, bất phương trình này vô nghiệm khi và chỉ khi
	$ \heva{&m^2+2m-3=0 \\ &-2m+1<0}\Leftrightarrow \heva{&m=1\vee m=-3 \\ &m>\dfrac{1}{2}} \Leftrightarrow m=1 $.\\
	Vậy $ m=1 $ thì thỏa yêu cầu bài toán.
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Hồng Phong, Hồ Chí Minh 17-18]%[Hoàng Hải, 10EX-HK1]%[0H2B2-3]
Trong mặt phẳng $ Oxy $ cho tam giác $ ABC $ với $ A(1;2),B(3;5),C(4;7) $. Tìm tọa độ trực tâm $ H $ của tam giác $ ABC $.
\loigiai{
	Gọi $ H(x;y) $, ta có 
	$ \heva{&\vec{AH} \cdot \vec{BC}=0 \\ &\vec{BH} \cdot \vec{AC}=0} \Leftrightarrow \heva{&(x-1)+2(y-2)=0 \\ &3(x-3)+5(y-5)=0} \Leftrightarrow \heva{&x+2y=5\\ &3x+5y=34} \Leftrightarrow \heva{&x=43 \\ &y=-19.} $\\
	Vậy trực tâm của $ \triangle ABC $ có tọa độ $ H(43;-19) $.
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Hồng Phong, Hồ Chí Minh 17-18]%[Hoàng Hải, 10EX-HK1]%[0H2B2-1]
Cho tam giác $ ABC $ có $ AB=7,AC=8,BC=13 $. Tính $ \vec{AB}\cdot\vec{AC} $.
\loigiai{
	\textbf{Cách 1:}
	 $$ \vec{AB} \cdot \vec{AC} = AB \cdot AC \cdot \cos{\widehat{BAC}} = AB \cdot AC \cdot \dfrac{AB^2+AC^2-BC^2}{2AB \cdot AC}
	=\dfrac{1}{2}(AB^2+AC^2-BC^2) = -28. $$
	 \textbf{Cách 2:}
	 $$ \vec{AB} - \vec{AC} = \vec{CB} \Rightarrow (\vec{AB} - \vec{AC})^2 = \vec{CB}^2 \Leftrightarrow \vec{AB} \cdot \vec{AC} = \dfrac{1}{2} (AB^2+AC^2-CB^2) = -28.  $$
	
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Hồng Phong, Hồ Chí Minh 17-18]%[Hoàng Hải, 10EX-HK1]%[0H2B3-1]
Cho tam giác $ ABC $ có $ AB=2,AC=3$ và $ \widehat{BAC}=120^{\circ} $. Tính độ dài cạnh $ BC $, diện tích $ \triangle ABC $, bán kính đường tròn ngoại tiếp và độ dài đường phân giác trong $ AD $ của $ \triangle ABC $.
\loigiai{
	\begin{center}
	\begin{tikzpicture}[scale=2,thick]
	\tkzDefPoints{0/0/A,-1.5/-1/B,2.5/-1/C}
	\tkzLabelPoints[above](A)\tkzLabelPoints[left](B)\tkzLabelPoints[right](C)
	\tkzDefLine[bisector](B,A,C) \tkzGetPoint{d} \tkzInterLL(A,d)(B,C)\tkzGetPoint{D}\tkzLabelPoints[below](D)
	\tkzDrawSegments(A,B A,C B,C A,D)
	\tkzMarkAngles[size=0.2cm,mark=|](B,A,D)
	\tkzMarkAngles[size=0.15cm,pos=0.1,mark=|](D,A,C)
	\draw($(A)!0.5!(B)$)node[left]{\rotatebox{37}{$2$}};
	\draw($(A)!0.5!(C)$)node[right=0.2]{\rotatebox{-23}{$3$}};
	\draw(A)node[right=1.5]{$ 120^\circ $};
	\draw [->,dashed] ($(A)+(0.03,-0.2)$) to [out=300,in=270] ($(A)+(1,-0.15)$);
	\end{tikzpicture}\vspace*{-5.5cm}
	\end{center}
	Ta có 
	\begin{itemize}
		\item $ BC^2 = AB^2+AC^2-2AB \cdot AC \cdot \cos\widehat{BAC} = 19 \Rightarrow BC=\sqrt{19} $.
		\item $ S_{\triangle ABC} = \dfrac{1}{2}AB\cdot AC \cdot \sin\widehat{BAC} = \dfrac{3\sqrt{3}}{2} $.
		\item $ S_{\triangle ABC} = \dfrac{AB \cdot AC \cdot BC}{4R} \Rightarrow R = \dfrac{AB \cdot AC\cdot BC}{4S_{\triangle ABC}} = \dfrac{\sqrt{57}}{3} $.\\
			Với $ R $ là bán kính đường tròn ngoại tiếp $ \triangle ABC $.
		\item 
		\textbf{Cách 1:} Ta có
		\begin{eqnarray*}
			& & S_{\triangle ABC} = S_{\triangle ABD} + S_{\triangle ACD} \\
			& \Leftrightarrow & \dfrac{1}{2}AB\cdot AC \sin 120^\circ = \dfrac{1}{2}AB\cdot AD \sin 60^\circ + \dfrac{1}{2}AC\cdot AD \sin 60^\circ \\
			& \Leftrightarrow & AD = \dfrac{6}{5}.
		\end{eqnarray*} 
		\textbf{Cách 2: (chú ý khi chọn cách này)} \\
		$ AD $ là tia phân giác trong của $ \widehat{BAC} $ nên
			$$ \heva{&\dfrac{DB}{DC} = \dfrac{AB}{AC} \\ &DB+DC=BC} \Leftrightarrow \heva{&3DB-2DC=0 \\ &DB+DC=\sqrt{19}} \Leftrightarrow \heva{&DB = \dfrac{2\sqrt{19}}{5} \\ &DC=\dfrac{3\sqrt{19}}{5}.} $$
		Trong $\triangle ABD$, ta có
		$$ BD^2 = AB^2+AD^2-2AB \cdot AD \cdot \cos\widehat{BAD} \Leftrightarrow AD^2-2AD + \dfrac{24}{25} = 0 \Leftrightarrow \hoac{&AD=\dfrac{4}{5} \\ &AD = \dfrac{6}{5}.} $$
		\textit{Chú ý: Vì $ \triangle ABC $ xác định nên $ AD $ phải là duy nhất, lý do tìm được 2 giá trị như trên là do $ \triangle ABD $ ta biết độ dài hai cạnh $ AB,BD $ và $ \widehat{BAD} $ \textbf{không xen giữa} hai cạnh $ AB,BD $. Trường hợp này tam giác xác định không duy nhất, do đó ta phải kiểm tra lại.}\\
		Với $ AD=\dfrac{4}{5} $, trong $ \triangle ADC $, ta có
			$$ DC^2 = AD^2+AC^2-2AD \cdot AC \cdot \cos\widehat{DAC} = \dfrac{181}{25} \ne \left(\dfrac{3\sqrt{19}}{5} \right)^2. $$
		Vậy loại $ AD=\dfrac{4}{5} $ và nhận $ AD=\dfrac{6}{5} $.\\
		\textit{Ta chỉ cần kiểm tra một giá trị, nếu sai thì loại và nhận giá trị còn lại, nếu đúng thì nhận và loại giá trị còn lại}.
	\end{itemize}
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Hồng Phong, Hồ Chí Minh 17-18]%[Hoàng Hải, 10EX-HK1]%[0D4K1-5]
Tìm giá trị nhỏ nhất của biểu thức $ f(x)=\dfrac{1}{x}+\dfrac{2}{1-x} $ với $ 0<x<1 $.
\loigiai{
	Ta có $$ f(x)=\dfrac{(1-x)+x}{x}+\dfrac{2x+2(1-x)}{1-x}=3+\dfrac{1-x}{x}+\dfrac{2x}{1-x}. $$
	Vì $ 0<x<1 $ nên $ \dfrac{1-x}{x}>0,\dfrac{2x}{1-x}>0$. Áp dụng bất đẳng thức Cauchy ta có
	$$ \dfrac{1-x}{x}+\dfrac{2x}{1-x}\ge 2\sqrt{2}.$$
	Đẳng thức xảy ra khi và chỉ khi
	 \begin{eqnarray*}
	 	& & \dfrac{1-x}{x}=\dfrac{2x}{1-x}\\
	 	& \Leftrightarrow & (1-x)^2=2x^2\\
	 	& \Leftrightarrow & (1-x-\sqrt{2}x)(1-x+\sqrt{2}x)=0\\
	 	& \Leftrightarrow & \hoac{&x=\dfrac{1}{1+\sqrt{2}}\quad\text{(thỏa)} \\ &x=\dfrac{1}{1-\sqrt{2}}\quad\text{(loại)}.}
	 \end{eqnarray*}
	 Vậy giá trị nhỏ nhất của $ f(x) $ trên $ (0;1) $ là $ 3+2\sqrt{2} $.
}
\end{bt}
