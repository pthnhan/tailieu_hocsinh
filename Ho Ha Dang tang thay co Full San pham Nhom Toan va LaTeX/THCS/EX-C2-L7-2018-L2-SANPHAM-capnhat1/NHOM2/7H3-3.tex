\section{Quan hệ giữa ba cạnh của tam giác}
\subsection{Hỏi đáp nhanh}
\begin{ex}%[7H3K3]
Bạn Minh lập luận: Ba đoạn thẳng $a=10$cm, $b=4$cm, $c=5$cm thỏa mãn $a+b>c$ nên chúng có thể lập thành một tam giác. Lập luận đó đúng hay sai?
\loigiai
{
Lập luận của bạn Minh là sai. Ta thấy $b+c<a$.
}
\end{ex}
\begin{ex}%[7H3K3]
Chu vi một tam giác cân là $20$cm, một cạnh của tam giác dài $5$cm. Hỏi cạnh đó là cạnh bên hay cạnh đáy?
\loigiai
{
Nếu cạnh đó là cạnh bên thì tổng hai cạnh bên là $5+5=10$cm. Trong đó cạnh đáy còn lại là $20-10=10$cm, vi phạm bất đẳng thức tam giác. Vậy cạnh $5$cm phải là cạnh đáy.
}
\end{ex}
\begin{ex}%[7H3K3]
Ba điểm $A,B, C$ phân biệt thỏa mãn $AB-AC=BC$ thì vị trí ba điểm $A,B,C$ như thế nào?
\loigiai
{
	Từ giả thiết ta có  $AB=AC+BC$. Vậy $A,B,C$ là ba điểm thẳng hàng và $C$ nằm giữa $A,B$.
}
\end{ex}

\begin{vd}%[7H3G3]
Cho $\triangle ABC$ có $AB>AC$. Điểm $M$ là trung điểm của $BC$. Chứng minh rằng
$$\dfrac{AB-AC}{2}<AM<\dfrac{AB+AC}{2}$$
\loigiai
{
	
\begin{minipage}{.6\textwidth}
Trên tia đối của tia $MA$ ta lấy điểm $A'$ sao cho $MA'=MA$. 
\begin{itemize}
	\item Xét hai tam giác $ABM$ và $A'MC$ có
	\begin{align*}
	AM&=MA'\ \ \text{(Theo cách vẽ)}\\
	BM&=MC \ \ \text{(Theo giả thiết)}\\
	\widehat {AMB}&=\widehat {CMA'} \ \ \text{(Góc đối đỉnh)}.
	\end{align*}
	Vậy $\triangle ABM$ = $\triangle A'MC$ (c.g.c).
	
	Suy ra $AB=A'C$.
	\item Xét $\triangle ACA'$ có $A'C-AC<AA'<AC+A'C$, do $AB=A'C$ và $AA'=2AM$ nên ta có
	$$AB-AC<2AM<AC+AB$$
\end{itemize}
Vậy $\dfrac{AB-AC}{2}<AM<\dfrac{AB+AC}{2}$.

\begin{enumEX}{2}
	\item $AB=7$cm, $AC=4$cm.
	\item $AB=9$cm, $AC=4$cm.
\end{enumEX}

{
\begin{enumEX}{1}
	\item Nếu $AB$ là cạnh bên thì ba cạnh là $7$cm, $7$cm và $4$cm thỏa mãn bất đẳng thức tam giác. Vậy chu vi tam giác là $18$cm.

\end{enumEX} 
}

\begin{bt}%[7H3K3]
	Chứng minh rằng trong một tam giác, độ dài cạnh lớn nhất sẽ lớn hơn hoặc bằng $\dfrac{1}{3}$ chu vi của tam giác nhưng nhỏ hơn nửa chu vi của tam giác ấy.
	\loigiai
	{
	Giả sử độ dài ba cạnh là $a, b, c$. Không mất tính tổng quát, giả sử $a \ge b \ge c >0$. 
	
	Ta có $a<b+c \Rightarrow a+a <a+b+c \Rightarrow a<\dfrac{a+b+c}{2}$.
	
	Từ $a\ge b, a\ge c$ ta có $3a \ge a+b+c \Rightarrow a\ge \dfrac{a+b+c}{3}$.
	
	Vậy $\dfrac{a+b+c}{3}\le a <\dfrac{a+b+c}{2}$.
	}
\end{bt}
