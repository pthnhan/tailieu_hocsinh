\begin{bt}%[Đề kiểm tra hk1, Toán 11, Ân Thi, Hưng Yên 2017]%[1D1K3]
	Giải phương trình lượng giác sau $\sqrt 3 \sin x + \cos x = 2$.
	\loigiai{Ta có
	\begin{equation*}
		\sqrt 3 \cdot \sin x + \cos x = 2 \Leftrightarrow \dfrac{\sqrt 3}{2}\sin x + \dfrac{1}{2}\cos x = 1 \Leftrightarrow \sin \left(x + \dfrac{\pi}{6} \right) = 1  \Leftrightarrow x = \dfrac{\pi}{3} + k2\pi, \quad k \in \mathbb{Z}.
\end{equation*}
Vậy phương trình có 1 họ nghiệm $x = \dfrac{\pi}{3} + k2\pi, \quad k \in \mathbb{Z}$.}

	Cho hai số thực $x$, $y$ thay đổi thỏa mãn hệ thức ${x^2} + {y^2} = 1$, tìm giá trị lớn nhất và nhỏ nhất của biểu thức $P = \dfrac{2(x^2 + 6xy)}{1 + 2xy + 2y^2}$. 
	\loigiai{Ta có $x^2 + y^2 = 1$ nên $\heva{&\left|x\right| \leq 1\\&\left|y\right| \leq 1}$. Đặt $x=\sin t$, $y=\cos t$ với $t \in \left[0;2\pi\right]$.\\
	Khi đó,  $P = \dfrac{1-\cos2t + 6\sin2t}{2 + \sin2t + \cos2t}= \Leftrightarrow (P-6)\sin2t + (P+1)\cos2t = 1 - 2P$. \quad $(*)$.\\
	Phương trình $(*)$ có nghiệm khi và chỉ khi
\begin{equation*}
	(1 - 2P)^2\leq(P-6)^2+(P+1)^2 \Leftrightarrow P^2+3P-18\leq 0 \Leftrightarrow - 6\leq P \leq 3.
\end{equation*}}
