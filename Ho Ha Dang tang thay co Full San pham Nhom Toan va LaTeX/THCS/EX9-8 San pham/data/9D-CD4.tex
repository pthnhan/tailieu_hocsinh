\section{Tam thức bậc hai}
\begin{dang}{Không dùng công thức nghiệm giải phương trình bậc hai một ẩn cho trước}
   Ta có thể sử dụng một trong các cách sau:
    \begin{itemize}
\item\textbf{Cách 1:} Đưa PT đã cho về dạng tích
\item \textbf{Cách 2:} Đưa PT đã cho về PT mà vế trái là một bình phương còn vế phải là một hằng số.
  \end{itemize}
\end{dang}
\setcounter{ex}{0}
\begin{ex}%[Nguyễn Thế Anh]%[9D4Y3]%[9D4Y3]%[9D4B3]%[9D4B3]
   Giải các phương trình:
\begin{enumerate}
	\begin{multicols}{2}
\item $5x^2-7x=0$.
\item $-3x^2+9=0$.
\item $x^2-6x+5=0$.
\item $3x^2+12x+1=0$.
	\end{multicols}
\end{enumerate}
    \loigiai{
    \begin{enumerate}
\item  $5x^2-7x=0\Leftrightarrow x(5x-7)=0\Leftrightarrow \hoac{
& x=0 \\
& x=\dfrac{7}{5} \\
}$
\item  $-3x^2+9=0\Leftrightarrow (3-\sqrt{3}x)(3+\sqrt{3}x)\Leftrightarrow \hoac{
& x=-\sqrt{3} \\
& x=\sqrt{3} \\
}$
\item $x^2-6x+5=0\Leftrightarrow \left(x-1\right)(x-5)=0\Leftrightarrow \hoac{
& x=1 \\
& x=5 \\
}$
\item Ta có:
\begin{align*}
& 3x^2+12x+1=0\Leftrightarrow 3(x^2+2.2x+4)-11=0\Leftrightarrow 3(x+2)^2-11=0 \\
& \Leftrightarrow (x+2)^2=\dfrac{11}{3}\Leftrightarrow \hoac{
& x=\dfrac{-6-\sqrt{33}}{3} \\
& x=\dfrac{-6+\sqrt{33}}{3} \\
} \\
\end{align*}
\end{enumerate}
}
\end{ex}

\begin{enumerate}

\item $x^2-x-9=0$.
\item $3x^2+6x+5=0$.
	\end{multicols}
\end{enumerate}
    \loigiai{

& x=0 \\
& x=2\sqrt{3} \\
}$.
\item $-\dfrac{3}{5}x^2-\dfrac{7}{2}=0\Leftrightarrow-\dfrac{3}{5}x^2=\dfrac{7}{2}\Leftrightarrow x^2=-\dfrac{35}{6}$(vô lý)\\
Vậy phương trình vô nghiệm.
\item Ta có:$x^2-x-9=0\Leftrightarrow x^2-2.\dfrac{1}{2}x+\dfrac{1}{4}-9-\dfrac{1}{4}=0\Leftrightarrow {\left(x-\dfrac{1}{2}\right)}^2=\dfrac{37}{4}\Leftrightarrow \hoac{
& x=\dfrac{1-\sqrt{37}}{2} \\
& x=\dfrac{1+\sqrt{37}}{2} \\
}$.
\item $3x^2+6x+5=0\Leftrightarrow 3(x^2+2x+1)+2=0\Leftrightarrow 3(x+1)^2+2=0$(vô lý).\\
Vậy phương trình vô nghiệm.
\end{enumerate}
}
\end{ex}

\begin{dang}{Giải phương trình bậc hai bằng cách sử dụng công thức nghiệm}
   Xét phương trình bậc hai: $ax^2+bx+c=0~~~(a\ne 0)$
   \begin{itemize}   
\item \textbf{Cách 1:}Xác định các hệ số a, b, c và tính biệt thức $\Delta =b^2-4ac$
\item \textbf{Cách 2:} Kết luận\\
-	Nếu $\Delta <0$ thì phương trình vô nghiệm.\\
-	Nếu  $\Delta =0$ thì phương trình có nghiệm kép: \\ $$x_1=x_2=-\dfrac{b}{a}$$
-	Nếu $\Delta >0$ thì phương trình có hai nghiệm phân biệt: \\ $${x}_{1,2}=\dfrac{-b\pm \sqrt{\Delta}}{2a}$$
\end{itemize}
\end{dang}
\begin{ex}%[Nguyễn Thế Anh]%[9D4Y3]%[9D4Y3]%[9D4B3]%[9D4B3]
   Xác định các hệ số $a$, $b$, $c$; tính biệt thức $\Delta$ rồi tìm nghiệm của các phương trình:
 \begin{enumerate}
 	\begin{multicols}{2}
\item $2x^2-3x-5=0$.
\item $x^2-6x+8=0$.
\item $9x^2-12x+4=0$.
\item$-3x^2+4x-4=0$.
 	\end{multicols}
\end{enumerate}
    \loigiai{
\begin{enumerate}
\item  Ta có $a=2,b=-3,c=-5\Rightarrow \Delta =49>0\Rightarrow $ Phương trình có hai nghiệm phân biệt : \\ $x_{1,2}=\dfrac{-b\pm \sqrt{\Delta}}{2a}\Rightarrow \hoac{
& x=-1 \\
& x=\dfrac{5}{2} \\
}$
\item Ta có $a=1,b=-6,c=8\Rightarrow \Delta =4>0\Rightarrow $ Phương trình có hai nghiệm phân biệt : ${x}_{1,2}=\dfrac{-b\pm \sqrt{\Delta}}{2a}\Rightarrow \hoac{
& x=4 \\
& x=2 \\
}$
\item Ta có: $a=9,b=-12,c=4\Rightarrow \Delta =0\Rightarrow $ Phương trình có nghiệm kép : $x_1=x_2=\dfrac{2}{3}.$
\item Ta có: $a=-3,b=4,c=-4\Rightarrow \Delta =-32<0\Rightarrow $ Phương trình vô nghiệm.
    \end{enumerate}
    }
\end{ex}

\begin{dang}{Xác định số nghiệm của phương trình bậc hai}
    Xét phương trình bậc hai:\\
$$ ax^2+bx+c=0$$ $$(a\ne 0)$$
 \begin{enumerate}
\item[1.] PT có nghiệm kép $\Leftrightarrow \heva{
& a\ne 0 \\
& \Delta =0 \\
}$
\item[2.] PT có hai nghiệm phân biệt $\Leftrightarrow \heva{
& a\ne 0 \\
& \Delta >0 \\
}$
\item[3.] PT có đúng một nghiệm $\Leftrightarrow \hoac{
& a=0;b\ne 0 \\
& a\ne 0;b=0 \\
}$
\item[4.] PT vô nghiệm $\Leftrightarrow \hoac{
& a=0;b=0;c\ne 0 \\
& a\ne 0;\Delta <0 \\
}$
 \end{enumerate}
\end{dang}

\begin{ex}%[Nguyễn Thế Anh]%[9D4B3]%[9D4B3]%[9D4B3]%[9D4B3]

\item Vô nghiệm.
\item Có đúng một nghiệm.

\end{enumerate}
    \loigiai{
 \begin{enumerate}
 \item Phương trình có hai nghiệm phân biệt
\begin{align*}
& \Leftrightarrow \heva{
& a\ne 0 \\
& \Delta >0 \\
}\Leftrightarrow \heva{
& m\ne 0 \\
& (-2(m-1))^2-4m(m-3)>0 \\
}\Leftrightarrow \heva{
& m\ne 0 \\

& \Leftrightarrow \heva{
& m\ne 0 \\

}\Leftrightarrow \heva{
& m\ne 0 \\
& m>-1 \\
} \\
\end{align*}
\item Phương trình có nghiệm kép $\Leftrightarrow \heva{
& a\ne 0 \\
& \Delta =0 \\
}\Leftrightarrow \heva{
& m\ne 0 \\
& 4m+4=0 \\
}\Leftrightarrow m=-1$.
\item Phương trình vô nghiệm vô nghiệm $\Leftrightarrow \hoac{
& a=0;b=0;c\ne 0 \\
& a\ne 0;\Delta <0 \\
}$.\\

Nếu $a\ne 0\Leftrightarrow m\ne 0$ thì $\Delta =4m+4<0\Leftrightarrow m<-1$.\\
Vậy $m<-1$ .
\item Phương trình có đúng một nghiệm $\Leftrightarrow \hoac{
& a=0;b\ne 0 \\
& a\ne 0;b=0 \\
}$.\\
Ta có: Nếu $a=0\Leftrightarrow m=0$ thì phương trình (1) thành $2x-3=0\Rightarrow x=\dfrac{3}{2}$.\\
Do đó $m=0$ (nhận).\\
Nếu $a\ne 0\Leftrightarrow m\ne 0$ thì $\Delta =4m+4=0\Leftrightarrow m=-1$.\\
Vậy $m=-1,m=0$ .
\end{enumerate}
}
\end{ex}

\begin{ex}%[Thanh Hai Le]%[Canada 82]%[9D4G6]

\begin{dang}{Tìm GTNN hoặc GTLN của biểu thức chứa nghiệm}
	Để tìm GTLN hoặc giá trị nhỏ nhất của biểu thức chứa nghiệm của phương trình bậc hai $ax^2+bx+c=0\ (1)$ ở đó $a\neq 0$ và $a, b, c$ chứa tham số $m$  ta cần thực hiện các bước sau:
	\begin{itemize}
		\item Tìm điều kiện của tham số $m$ để phương trình đã cho có nghiệm.
		\item Biến đổi biểu thức cần tìm GTLN hoặc GTNN thông qua các biểu thức $x_1+x_2$ và $x_1x_2$, ở đó $x_1, x_2$ là các nghiệm của phương trình $(1)$.
		\item Sử dụng định lý Vi-ét xác định $\heva{&x_1+x_2=-\dfrac{b}{a}\\&x_1x_2=\dfrac{c}{a}}$, từ đó đưa biểu thức đang xét về biểu thức có chứa tham số $m$.
		\item Tìm GTLN hoặc GTNN của biểu thức đang xét với điều kiện của tham số $m$ để $(1)$ có nghiệm.
	\end{itemize}
\end{dang}

		Ta có $|x_0|^3 = |ax_0^2 + bx_0+c| \leq M(|x_0|^2 + |x_0| + 1)$ suy ra $\dfrac{|x_0|^3}{|x_0|^2 + |x_0| + 1} \leq M \Leftrightarrow |x_0|-1 + \dfrac{1}{|x_0|^2 + |x_0| + 1} \leq M$,  $\dfrac{1}{|x_0|^2 + |x_0| + 1} >0$ suy ra $|x_0| -1 < M$ (ĐPCM). 
	}
