\begin{ex}%[Đề thi hk 1 Toán 12, Thạch Thành I, Thanh Hoá 2018]%[Đào Trung Kiên, Dự án 12EX-5]%[2D1G3-5]
	Cho các số thực $x, y$ thoả mãn $x^2+2xy+3y^2=4$. Giá trị lớn nhất của biểu thức $P=(x-y)^2$ là
	\choice
	{$\mathrm{Max}P=8$}
	{\True $\mathrm{Max}P=12$}
	{$\mathrm{Max}P=16$}
	{$\mathrm{Max}P=4$}
	\loigiai{Từ giả thiết $x^2+2xy+3y^2=4$ ta xét $\dfrac{P}{4}=\dfrac{x^2-2xy+y^2}{x^2+2xy+3y^2}$.\\
		Với $x=0$ thì ta có $P=\dfrac{4}{3}$.\\
		Với $x\neq 0$ ta có $\dfrac{P}{4}=\dfrac{1-2\dfrac{y}{x}+\left(\dfrac{y}{x}\right)^2}{1+2\dfrac{y}{x}+3\left(\dfrac{y}{x}\right)^2}$. Bằng cách đặt $t=\dfrac{y}{x}$ ta có $\dfrac{P}{4}=\dfrac{1-2t+t^2}{1+2t+3t^2}$.\\
		Ta dễ thấy $P$ thuộc tập giá trị khi phương trình $\dfrac{P}{4}=\dfrac{1-2t+t^2}{1+2t+3t^2}$ có nghiệm $t$ suy ra $0\leq \dfrac{P}{4}\leq 3$. Hay $0\leq P\leq 12$.\\
		Ta thấy $P=12$ tại $x=\dfrac{-4}{\sqrt{3}}, y=\dfrac{2}{\sqrt{3}}$ nên  $\textrm{Max} P=12$.}
\end{ex}
