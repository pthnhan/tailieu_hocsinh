\begin{bt}%[Đề HK1, Chu Văn An, HCM 2017-2018]%[Đỗ Đường Hiếu, dự án TL-HK1-11]%[1D1B2-1]
	Giải phương trình $\sin \left(2x-\dfrac{\pi}{5}\right)=\sin \left(\dfrac{\pi}{5}+x\right)$.
	\loigiai{
	Ta có 
	\begin{eqnarray*}
	\sin \left(2x-\dfrac{\pi}{5}\right)=\sin \left(\dfrac{\pi}{5}+x\right) &\Leftrightarrow & \hoac{&2x-\dfrac{\pi}{5}=\dfrac{\pi}{5}+x+k2\pi\\&2x-\dfrac{\pi}{5}=\pi-\left( \dfrac{\pi}{5}+x\right) +k2\pi},k\in\mathbb{Z}\\
	&\Leftrightarrow & \hoac{&x=\dfrac{2\pi}{5}+k2\pi\\&3x=\pi+k2\pi},k\in\mathbb{Z}\\
	&\Leftrightarrow & \hoac{&x=\dfrac{2\pi}{5}+k2\pi\\&x=\dfrac{\pi}{3}+k\dfrac{2\pi}{3}},k\in\mathbb{Z}.
	\end{eqnarray*}
	Vậy, phương trình đã cho có nghiệm $\hoac{&x=\dfrac{2\pi}{5}+k2\pi\\&x=\dfrac{\pi}{3}+k\dfrac{2\pi}{3}},k\in\mathbb{Z}$.
	}
\end{bt}

\begin{bt}%[Đề HK1, Chu Văn An, HCM 2017-2018]%[Đỗ Đường Hiếu, dự án TL-HK1-11]%[1D1B3-3]
	Giải phương trình $\sqrt{3}\cos 3x -\sin 3x=2$.
	\loigiai{
	Ta có
	\begin{eqnarray*}
	\sqrt{3}\cos 3x -\sin 3x=2&\Leftrightarrow &\sqrt{3}\cos 3x -\sin 3x=2\\
	&\Leftrightarrow &\dfrac{\sqrt{3}}{2}\cos 3x -\dfrac{1}{2}\sin 3x=1\\
	&\Leftrightarrow &\cos 3x \cos \dfrac{\pi}{6}-\sin 3x\sin \dfrac{\pi}{6}=1\\
	&\Leftrightarrow &\cos \left(3x+\dfrac{\pi}{6}\right)=1\\
	&\Leftrightarrow &3x+\dfrac{\pi}{6}=k2\pi, \; k\in \mathbb{Z}\\
	&\Leftrightarrow &x=-\dfrac{\pi}{18}+k\dfrac{2\pi}{3}, \; k\in \mathbb{Z}.
	\end{eqnarray*}
	Vậy, phương trình đã cho có nghiệm là $x=-\dfrac{\pi}{18}+k\dfrac{2\pi}{3}, \; k\in \mathbb{Z}$.
	}
\end{bt}

\begin{bt}%[Đề HK1, Chu Văn An, HCM 2017-2018]%[Đỗ Đường Hiếu, dự án TL-HK1-11]%[1D2B2-1]
	Có bao nhiêu cách xếp $4$ học sinh vào một bàn học có $4$ chỗ ngồi?
	\loigiai{
	Mỗi cách xếp $4$ học sinh vào một bàn học có $4$ chỗ ngồi là một hoán vị của $4$ học sinh.\\
	Vậy, có tất cả $\mathrm{P}_4=4!=24$ cách xếp $4$ học sinh vào một bàn học có $4$ chỗ ngồi.
	}
\end{bt}

\begin{bt}%[Đề HK1, Chu Văn An, HCM 2017-2018]%[Đỗ Đường Hiếu, dự án TL-HK1-11]%[1D2K2-1]
	Có bao nhiêu cách chọn ra $3$ bông hoa từ $7$ bông hoa có màu khác nhau để cắm vào $3$ lọ hoa khác nhau (mỗi lọ cắm một bông)?
	\loigiai{
	Mỗi cách chọn ra $3$ bông hoa từ $7$ bông hoa có màu khác nhau để cắm vào $3$ lọ hoa khác nhau (mỗi lọ cắm một bông) là một chỉnh hợp chập $3$ của $7$ bông hoa.\\
	Vậy, tổng số cách chọn là $\mathrm{A}^3_7=210$ cách.
	}
\end{bt}

\begin{bt}%[Đề HK1, Chu Văn An, HCM 2017-2018]%[Đỗ Đường Hiếu, dự án TL-HK1-11]%[1D2K5-2]%[1D2K5-3]
	Từ một hộp chứa $16$ thẻ được đánh số từ $1$ đến $16$, chọn ngẫu nhiên $4$ thẻ.
	\begin{enumerate}
		\item Tính xác suất để $4$ thẻ được chọn đều được đánh số chẵn.
		\item Tính xác suất để chọn được ít nhất một thẻ mang số chẵn trong $4$ thẻ được chọn ra.
	\end{enumerate}
	\loigiai{
	Số phần tử của không gian mẫu là $n\left(\Omega\right)=\mathrm{C}^4_{16}$.
	\begin{enumerate}
		\item Trong $16$ thẻ được đánh số từ $1$ đến $16$, có $8$ thẻ được đánh số lẻ và có $8$ thẻ được đánh số chẵn.\\
		Gọi $A$ là biến cố \lq\lq $4$ thẻ được chọn đều được đánh số chẵn\rq\rq.\\
		Ta có $n(A)=\mathrm{C}^4_8$.\\
		Vậy $\mathrm{P}(A)=\dfrac{n(A)}{n\left(\Omega\right)}=\dfrac{\mathrm{C}^4_8}{\mathrm{C}^4_{16}}=\dfrac{1}{26}$.
		
		\item Gọi $B$ là biến cố \lq\lq $4$ thẻ được chọn có ít nhất một thẻ mang số chẵn\rq\rq.\\
		Suy ra $\overline{B}$ là biến cố \lq\lq $4$ thẻ được chọn không có thẻ nào mang số chẵn\rq\rq.\\
		Ta có $n(\overline{B})=\mathrm{C}^4_8$.\\ 
		Vậy nên $\mathrm{P}(B)=1-\mathrm{P}(\overline{B})=1-\dfrac{n(\overline{B})}{n\left(\Omega\right)}=1-\dfrac{\mathrm{C}^4_8}{\mathrm{C}^4_{16}}=1-\dfrac{1}{26}=\dfrac{25}{26}$.
		
	\end{enumerate}
	}
\end{bt}


\begin{bt}%[Đề HK1, Chu Văn An, HCM 2017-2018]%[Đỗ Đường Hiếu, dự án TL-HK1-11]%[1D3B4-3]
	Tìm số hạng đầu và công sai của cấp số cộng, biết $\heva{&u_1+u_5-u_3=10\\&u_1+u_6=17}$.
	\loigiai{
	Goi $d$ là công sai của cấp số cộng đã cho, ta có
	\begin{eqnarray*}
	&\heva{&u_1+u_5-u_3=10\\&u_1+u_6=17}&\Leftrightarrow \heva{&u_1+(u_1+4d)-(u_1+2d)=10\\&u_1+(u_1+5d)=17}\\
	&&\Leftrightarrow \heva{&u_1+2d=10\\&2u_1+5d=17}\\
	&&\Leftrightarrow \heva{&u_1=16\\&d=-3.}
	\end{eqnarray*}
	Vậy, cấp số cộng đã cho có số hạng đầu $u_1=16$ và công sai $d=-3$.
	}
\end{bt}


\begin{bt}%[Đề HK1, Chu Văn An, HCM 2017-2018]%[Đỗ Đường Hiếu, dự án TL-HK1-11]%[1D3B4-1]
	Cho dãy số $\left(u_n\right)$ có số hạng tổng quát $u_n=2\cdot 3^n$, $n\in\mathbb{N}^*$.
	\begin{enumerate}
		\item Chứng minh $\left(u_n\right)$ là cấp số nhân.
		\item Tìm số hạng đầu tiên và công bội.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Với $\forall n\in\mathbb{N}^*$, ta có
		\begin{eqnarray*}
		\dfrac{u_{n+1}}{u_n}=\dfrac{2\cdot 3^{n+1}}{2\cdot 3^n}=3.
		\end{eqnarray*}
		Do đó, dãy số $\left(u_n\right)$ là một cấp số nhân.
		\item Cấp số nhân $\left(u_n\right)$ có số hạng đầu $u_1=2\cdot 3^1=6$ và công bội $q=3$.
	\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề HK1, Chu Văn An, HCM 2017-2018]%[Đỗ Đường Hiếu, dự án TL-HK1-11]%[1H2B1-2]%[1H2B3-2]%[1H2K3-5]
	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình bình hành tâm $O$. Gọi $I$ là trung điểm của $SA$.
	\begin{enumerate}
		\item Tìm giao tuyến của hai mặt phẳng $(SAC)$ và $(IBD)$.
		\item Chứng minh đường thẳng $SC$ song song với mặt phẳng $(IBD)$.
		\item Tìm giao điểm $K$ của đường thẳng $SB$ và mặt phẳng $(ICD)$.
	\end{enumerate}
	\loigiai{
	\begin{center}
		\begin{tikzpicture}[scale=1]
		\tkzDefPoints{0/0/A, -2/-2/B, 6/0/D}
		\coordinate (C) at ($(B)+(D)-(A)$);  
		\coordinate (O) at ($(A)!0.5!(C)$);
		\coordinate (S) at ($(O)+(-1,5)$);
		\coordinate (I) at ($(S)!0.5!(A)$);
		\coordinate (K) at ($(S)!0.5!(B)$);
		\tkzDrawPoints(S,A,B,C,D,O,I);
		\draw (S)--(B)--(C)--(D)--(S)--(C) (K)--(C);
		\draw[dashed] (S)--(A)--(B) (A)--(D) (A)--(C) (B)--(D) (S)--(O) (B)--(I)--(D) (I)--(O) (I)--(K);
		\tkzDrawPoints(S,A,B,C,D,O,K);
		\tkzLabelPoints[below](A,B,C,D,O);
		\tkzLabelPoints[above](S);
		\tkzLabelPoints[left](I,K);
		\end{tikzpicture}
	\end{center}
	\begin{enumerate}
		\item Ta có $I\in (IBD)$ và $I\in SA$, $SA\subset (SAC)$ nên $I$ là một điểm chung của hai mặt phẳng $(SAC)$ và $(IBD)$.\\
		Vì $O=AC\cap BD$ và $AC\subset (SAC)$, $BD\subset (IBD)$ nên $O$ là điểm chung của hai mặt phẳng $(SAC)$ và $(IBD)$.\\
		Vậy, giao tuyến của hai mặt phẳng $(SAC)$ và $(IBD)$ là đường thẳng $IO$.
		\item Do $I$ và $O$ lần lượt là trung điểm $AS$ và $AC$ nên $IO\parallel SC$.\\
		Từ $\heva{& SC\parallel IO\\& IO \subset (IBD)\\& SC\not\subset (IBD) }$ suy ra $SC\parallel (IBD)$.
		\item Hai mặt phẳng $(ICD)$ và $(SAB)$ có điểm chung là $I$. Mặt khác $CD\subset (ICD)$, $AB\subset (SAB)$ và $CD\parallel AB$, nên giao tuyến của hai mặt phẳng $(ICD)$ và $(SAB)$ là đường thẳng $d$ đi qua $I$ và song song với $AB$, cắt $SB$ tại $K$. Điểm $K$ đó chính là giao điểm của đường thẳng $SB$ và mặt phẳng $(ICD)$.
	\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề HK1, Chu Văn An, HCM 2017-2018]%[Đỗ Đường Hiếu, dự án TL-HK1-11]%[1D2K3-2]
	Tìm hệ số của $x^4y^9$ trong khai triển $(2x-y)^{13}$.
	\loigiai{
	Số hạng thứ $k+1$ của khai triển $(2x-y)^{13}$ là
	$$T_{k+1}=\mathrm{C}^k_{13}\cdot (2x)^{13-k}\cdot (-y)^k=2^{13-k}\cdot (-1)^k\cdot\mathrm{C}^k_{13}\cdot x^{13-k}\cdot y^k.$$
	Số hạng chứa $x^4y^9$ ứng với $\heva{&13-k=4\\&k=9}\Leftrightarrow k=9$.\\
	Hệ số của $x^4y^9$ trong khai triển $(2x-y)^{13}$ là: $2^4\cdot(-1)^9\cdot\mathrm{C}^9_{13}=-11440$.
	}
\end{bt}
