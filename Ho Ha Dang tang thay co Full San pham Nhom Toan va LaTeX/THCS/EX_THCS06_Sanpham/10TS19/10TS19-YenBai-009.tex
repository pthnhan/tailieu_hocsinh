\begin{ex}%[10TS19 - SGD Yên Bái, Nguyễn Tuấn Kiệt]%[9D2B3]
	Xác định giá trị của $m$ để các đường thẳng $y=2x+4$, $y=3x+5$, $y=-mx$ cùng đi qua một điểm.
	\choice
		{$m=\dfrac{1}{2}$}
		{$m=-\dfrac{1}{2}$}
		{\True $ m=2$}
		{$m=-2$}
\loigiai{
Gọi $M(x_M;y_M)$ là giao điểm của 2 đường thẳng $y=2x+4$, $y=3x+5$. Tọa độ $M$ thỏa hệ phương trình
$\begin{cases}
2x_M-y_M=-4\\
3x_M-y_M=-5
\end{cases}
$. Vậy $M(-1;2)$.\\
Theo yêu cầu bài toán thì $M(-1;2)$ phải thuộc đường thẳng $y=-mx$.\\
Do đó $2=-m(-1)\Rightarrow m=2$.
}
\end{ex}

\begin{ex}%[10TS19 - SGD Yên Bái, Nguyễn Tuấn Kiệt]%[9H2B7]%[9H2B8]
	Cho hai đường tròn ($O$;$4$cm) và ($O'$;$3$cm) có $OO'=5$cm. Hai đường tròn trên cắt nhau tại $A$ và $B$. Tính độ dài $AB$.
	\choice
		{$AB = 3{,}2$cm}
		{\True $AB = 4{,}8$cm}
		{$AB = 2{,}4$cm}
		{$AB = 3{,}6$cm}
\loigiai{
\begin{center}
\begin{tikzpicture}[>=stealth,scale=.7, line join = round, line cap = round]
\tkzInit[xmin=-4,xmax=8,ymin=-4,ymax=4]
\tkzClip
\tikzset{label style/.style={font=\footnotesize}}
\tkzDefPoints{0/0/O, 5/0/O', 4/0/E, 2/0/F}
\tkzDrawCircle(O,E)\tkzDrawCircle(O',F)
\tkzInterCC(O,E)(O',F)\tkzGetFirstPoint{A}\tkzGetSecondPoint{B}
\tkzInterLL(O,O')(A,B)\tkzGetPoint{H}
\tkzLabelPoints[above](A,O)
\tkzLabelPoints[above right](H,O')
\tkzLabelPoints[below](B)
\tkzDrawPoints(O,O',A,H,B)
\tkzDrawSegments(O,O' A,B O,A O',A)
\end{tikzpicture}
\end{center}
Áp dụng định lý Pytago đảo cho $\triangle{OAO'}$ ta có $OO'^2=OA^2+O'A^2 \Leftrightarrow 5^2=4^2+3^2$.\\
Suy ra $\triangle{OAO'}$ vuông tại $A$.\\
Gọi $H$ là giao của $AB$ và $OO'$. Dựa vào hai tam giác đồng dạng $\triangle{OAO'}$ và $\triangle{OBO'}$ dễ dàng chứng minh $AH$ là đường cao của $\triangle{OAO'}$.\\
Ta có $\dfrac{1}{AH^2}=\dfrac{1}{4^2}+\dfrac{1}{3^2} \Leftrightarrow AH=\dfrac{12}{5}=2,4$cm.\\
Do đó $AB=2AH=2.2{,}4=4{,}8$cm.
}
\end{ex}

\begin{ex}%[10TS19 - SGD Yên Bái, Nguyễn Tuấn Kiệt]%[9H3B8]

	\choice
		{$OB = \sqrt{10}$ cm}
		{$OB = \sqrt{3}$ cm}
		{\True $OB = \sqrt{5}$ cm}
		{$OB = \sqrt{2}$ cm}
\loigiai{

\tkzInCenter(A,B,C)\tkzGetPoint{O}
\tkzDefPointBy[projection=onto A--B](O)\tkzGetPoint{H}
\tkzInterLL(B,O)(A,C)\tkzGetPoint{K}

\tkzDrawCircle(O,H)
\tkzDrawSegments(A,B B,C C,A O,B O,H O,K)
\tkzLabelPoints[above right](B,O,H,K)

\end{center}
Kẻ $BO$ cắt $AC$ tại $K$.\\
Ta có $\dfrac{AB}{AC}=\dfrac{AK}{CK}=\dfrac{3}{5} \Rightarrow \dfrac{AK}{3}=\dfrac{CK}{5}=\dfrac{AK+KC}{3+5}=\dfrac{AC}{8}=\dfrac{4}{8}=\dfrac{1}{2}$.\\
Suy ra $AK=\dfrac{3}{2}=1,5 \Rightarrow BK=\sqrt{AB^2+AK^2}=\sqrt{3^2+(1,5)^2}=\dfrac{3\sqrt{5}}{2}$.\\
Kẻ $OH \perp AB \Rightarrow OH \parallel AC$ ta có\\
$\dfrac{BH}{BA}=\dfrac{BO}{BK}=\dfrac{2}{3} \Rightarrow BO=\dfrac{2}{3}BK=\dfrac{2}{3}.\dfrac{3\sqrt{5}}{2}=\sqrt{5}$cm.
}
\end{ex}

\begin{ex}%[10TS19 - SGD Yên Bái, Nguyễn Tuấn Kiệt]%[8D2B3]
	Kết quả rút gọn  biểu thức $A=\dfrac{x}{x-4}+\dfrac{1}{\sqrt{x}-2}+\dfrac{1}{\sqrt{x}+2}$ (với $x \geq 0$, $x \neq 4$) có dạng $\dfrac{\sqrt{x}-m}{\sqrt{x}+n}$. Tính giá trị của $m^2+n^2$.
	\choice
		{\True $m^2+n^2 = 4$}
		{$m^2+n^2 = 9$}
		{$m^2+n^2 = 5$}
		{$m^2+n^2 = 1$}

$=\dfrac{x+\sqrt{x}+2+\sqrt{x}-2}{x-4}
= \dfrac{x+2\sqrt{x}}{x-4}
= \dfrac{\sqrt{x}(\sqrt{x}+2)}{(\sqrt{x}+2)(\sqrt{x}-2)}
= \dfrac{\sqrt{x}}{\sqrt{x}-2}$.\\
Giá trị $m^2+n^2 = 0^2+(-2)^2=4$.
}

\begin{ex}%[10TS19 - SGD Yên Bái, Nguyễn Tuấn Kiệt]%[7D1B8]
	Cho các số thực $x,y,z$ thỏa mãn $\dfrac{12x-15y}{7}=\dfrac{20z-12x}{9}=\dfrac{15y-20z}{11}$ và $x+y+z=-48$. Tìm $y$.
	\choice
		{\True $y = -16$}
		{$y = -12$}
		{$y = -10$}
		{$y = -20$}
\loigiai{
Ta có: $\dfrac{12x-15y}{7}=\dfrac{20z-12x}{9}=\dfrac{15y-20z}{11}=\dfrac{12x-15y+20z-12x+15y-20z}{7+9+11}=0$\\
Do đó $\dfrac{12x}{7}-\dfrac{15y}{7}=0; \dfrac{20z}{9}-\dfrac{12x}{9}=0; \dfrac{15y}{11}-\dfrac{20z}{11}=0$\\
$\Leftrightarrow \dfrac{12x}{7}=\dfrac{15y}{7}; \dfrac{20z}{9}=\dfrac{12x}{9}; \dfrac{15y}{11}=\dfrac{20z}{11}$\\
$\Leftrightarrow 12x=15y; 20z=12x; 15y=20z$\\
$\Leftrightarrow \dfrac{x}{15}=\dfrac{y}{12}; \dfrac{z}{12}=\dfrac{x}{20}; \dfrac{y}{20}=\dfrac{z}{15}$\\
$\Leftrightarrow \dfrac{x}{5.15}=\dfrac{y}{60}=\dfrac{z}{3.15}=\dfrac{x+y+z}{75+60+45}=\dfrac{-48}{180}=\dfrac{-4}{15} \Rightarrow y=-16$.
}
\end{ex}

\begin{ex}%[10TS19 - SGD Yên Bái, Nguyễn Tuấn Kiệt]%[9D4K6]
	Cho phương trình $x^2-2(m-1)x+m^2-3m-1=0$ ($m$ là tham số). Tìm tất cả các giá trị dương của $m$ để phương trình đã cho có hai nghiệm phân biệt $x_1, x_2$ thỏa mãn $x_1^2+x_2^2=10$.
	\choice
		{$m = 3$}
		{$m =  4$}
		{\True $m = 2$}
		{$m = 1$}
\loigiai{
Phương trình có 2 nghiệm phân biệt\\
$\Leftrightarrow \Delta' >0 \Leftrightarrow [-(m-1)]^2-1(m^2-3m-1) > 0 \Leftrightarrow m+2 > 0 \Leftrightarrow m > -2$.\\
Áp dụng định lí Vi-ét ta có $x_1^2 + x_2^2 = (x_1 + x_2)^2 - 2x_1x_2 = (2m-2)^2 - 2(m^2 - 3m - 1)$\\
$= 2m^2 - 2m + 6 = 10 \Leftrightarrow m^2 - m - 2 = 0 \Leftrightarrow \left[\begin{array}{ll}
m = -1 \\ m = 2
\end{array}\right.$.\\
Vì $m$ dương nên chọn $m = 2$.
}
\end{ex}
