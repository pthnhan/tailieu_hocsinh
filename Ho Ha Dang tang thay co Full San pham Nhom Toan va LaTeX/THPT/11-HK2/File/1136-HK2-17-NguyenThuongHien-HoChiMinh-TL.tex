\begin{bt}%[HK2 (2016-2017), THPT Nguyễn Thượng Hiền, Hồ Chí Minh]%[1D3B3-5]%[Trần Phong]
	Cho cấp số cộng $(u_n)$ thỏa mãn $\heva{& u_2-3u_5+2u_7=3\\& u_1+2u_4-2u_6=-7}$. Tính tổng của $20$ số hạng đầu tiên của cấp số cộng này.
	\loigiai{
		Gọi $d$ là công sai của cấp số cộng $(u_n)$. Ta có
		$$\heva{& u_2-3u_5+2u_7=3\\& u_1+2u_4-2u_6=-7}\Leftrightarrow \heva{& (u_1+d)-3(u_1+4d)+2(u_1+6d)=3\\& u_1+2(u_1+3d)-2(u_1+5d)=-7}\Leftrightarrow \heva{& d=3\\& u_1=5}.$$
		Suy ra $S_{20}=\dfrac{20}{2}(2u_1+19d)=670$.
	}
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Nguyễn Thượng Hiền, Hồ Chí Minh]%[1D4K2-7][1D4K2-3]%[Trần Phong]
	Tính các giới hạn sau
	\begin{enumerate}
		\item $\lim\limits_{x\rightarrow -\infty}\dfrac{\sqrt[3]{8x^3+3x^2+1}-x}{\sqrt{4x^2-x+2}+3x}$.
		\item $\lim\limits_{x\rightarrow 1}\dfrac{x-1}{\sqrt{2x+2}-x^2-1}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $\lim\limits_{x\rightarrow -\infty}\dfrac{\sqrt[3]{8x^3+3x^2+1}-x}{\sqrt{4x^2-x+2}+3x}=\lim\limits_{x\rightarrow -\infty}\dfrac{x\sqrt[3]{8+\dfrac{3}{x}+\dfrac{1}{x^3}}-x}{-x\sqrt{4-\dfrac{1}{x}+\dfrac{2}{x^2}}+3x}=\lim\limits_{x\rightarrow -\infty}\dfrac{\sqrt[3]{8+\dfrac{3}{x}+\dfrac{1}{x^3}}-1}{-\sqrt{4-\dfrac{1}{x}+\dfrac{2}{x^2}}+3}=1$.
			
			\item $\lim\limits_{x\rightarrow 1}\dfrac{x-1}{\sqrt{2x+2}-x^2-1}=\lim\limits_{x\rightarrow 1}\dfrac{x-1}{\sqrt{2x+2}-2+1-x^2}=\lim\limits_{x\rightarrow 1}\dfrac{x-1}{ \dfrac{2(x-1)}{\sqrt{2x+2}+2}+(1-x)(1+x)}$
			\item[] $=\lim\limits_{x\rightarrow 1}\dfrac{1}{ \dfrac{2}{\sqrt{2x+2}+2}-(x+1)}=-\dfrac{2}{3}$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Nguyễn Thượng Hiền, Hồ Chí Minh]%[1D4K3-5]%[Trần Phong]
	Định $a$ để hàm số sau liên tục tại $x_0=-3$: $f(x)$ $=\heva{& \dfrac{x^4-13x^2+36}{x^3+3x^2+x+3}& \text{khi}\ x\ne-3 \\& ax^2-12 & \text{khi}\ x=-3}$.
	\loigiai{
		\begin{itemize}
			\item Ta có $f(-3)=9a-12$.
			\item $\lim\limits_{x\rightarrow -3}f(x)=\lim\limits_{x\rightarrow -3}\dfrac{x^4-13x^2+36}{x^3+3x^2+x+3}=\lim\limits_{x\rightarrow -3}\dfrac{(x^2-9)(x^2-4)}{(x+3)(x^2+1)}=\lim\limits_{x\rightarrow -3}\dfrac{(x-3)(x^2-4)}{x^2+1}=-3$.
			\item Hàm số liên tục tại $x_0=-3\Leftrightarrow \lim\limits_{x\rightarrow -3}f(x)=f(-3) \Leftrightarrow 9a-12=-3\Leftrightarrow a=1$.
		\end{itemize}
	}
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Nguyễn Thượng Hiền, Hồ Chí Minh]%[1D5K3-1]%[Trần Phong]
	Tính đạo hàm các hàm số sau
	\begin{enumerate}
		\item $y=\sin2x\cdot (x^2-3\cos x)^7$.
		\item $y=\sin^3(2x^2+5x-1)$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $y=\sin2x\cdot (x^2-3\cos x)^7$.
			
			$\begin{aligned} y' & =(\sin2x)'\cdot (x^2-3\cos x)^7+\sin2x\cdot [(x^2-3\cos x)^7]' \\& =2\cos2x \cdot (x^2-3\cos x)^7+7\sin2x\cdot (x^2-3\cos x)^6 \cdot (x^2-3\cos x)' \\& =2\cos2x \cdot (x^2-3\cos x)^7+7\sin2x\cdot (x^2-3\cos x)^6 \cdot (2x+3\sin x) \\& =(x^2-3\cos x)^6\cdot [2\cos2x\cdot (x^2-3\cos x)+7\sin2x\cdot(2x+3\sin x)]. \end{aligned}$
			
			\item $y=\sin^3(2x^2+5x-1)$.
			
			$\begin{aligned} y' & =3\sin^2(2x^2+5x-1)\cdot [\sin(2x^2+5x-1)]'\\& = 3(4x+5)\cdot\sin^2(2x^2+5x-1)\cdot\cos(2x^2+5x-1). \end{aligned}$
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Nguyễn Thượng Hiền, Hồ Chí Minh]%[1D5B2-3]%[Trần Phong]
	Cho hàm số $y=f(x)=x^3-2x^2+1$ có đồ thị $(C)$. Viết phương trình tiếp tuyến của đồ thị $(C)$ biết hệ số góc tiếp tuyến là $k=-1$.
	\loigiai{
		\begin{itemize}
			\item Tập xác định $\mathscr{D}=\mathbb R$. Ta có $y'=3x^2-4x$.
			\item Gọi $M(x_0;y_0)$ là tọa độ tiếp điểm.
			\item Do tiếp tuyến có hệ số góc $k=-1$ nên ta có $y'(x_0)=-1\Leftrightarrow 3x_0^2-4x_0+1=0\Leftrightarrow \hoac{& x_0=1\\& x_0=\dfrac{1}{3}}$.
			\item Với $x_0=1\Rightarrow y_0=0$, ta có tiếp tuyến $d:y=(-1)\cdot(x-1)+0=-x+1$.
			\item Với $x_0=\dfrac{1}{3}\Rightarrow y_0=\dfrac{22}{27}$, ta có tiếp tuyến $d:y=(-1)\cdot\left(x-\dfrac{1}{3}\right)+\dfrac{22}{27}=-x+\dfrac{31}{27}$.
		\end{itemize}
	}
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Nguyễn Thượng Hiền, Hồ Chí Minh]%[1H3B3-2][1H3B4-3][1H3G4-2][1H3G5-3]%[Trần Phong]
	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình chữ nhật với $AB=a,AD=2a$. Tam giác $SAB$ là tam giác đều và nằm trong mặt phẳng vuông góc với đáy. Gọi $I$ là trung điểm $AB$.
	\begin{enumerate}
		\item Chứng minh $SI\perp(ABCD)$ và tam giác $SBC$ vuông.
		\item Xác định và tính góc giữa hai mặt phẳng $(SBD)$ và mặt phẳng $(ABCD)$.
		\item Gọi $H$ là hình chiếu của $C$ lên $BD$ và $K$ là trung điểm của $DH$, chứng minh $(SCK)$ vuông góc với $(SIK)$.
		\item Gọi $M$ là điểm trên cạnh $BC$ sao cho $MC=2BM$ và $J$ là giao điểm của $AM$ và $BD$, tính khoảng cách từ $J$ đến mặt phẳng $(SCD)$.
	\end{enumerate}
	\loigiai{
		
		\begin{center}
			\begin{tikzpicture}[scale=1.0]

			\tkzDrawSegments(S,B B,C  S,C S,D C,D S,N)
			\tkzLabelPoints[left](B,I,H,J)

		\begin{enumerate}
			\item Ta có $\heva{& (SAB)\perp(ABCD)\\& (SAB)\cap (ABCD)=AB\\& SI\subset (SAB),SI\perp AB} \Rightarrow SI\perp(ABCD)$.
			\item[] Ta có $\heva{& BC\perp SI,BC\perp AB\\& SI,AB\subset (SAB)\\& SI\cap AB=I}\Rightarrow BC\perp(SAB)\Rightarrow BC\perp SB\Rightarrow \triangle SBC$ vuông tại $B$.
			
			\item Dựng $IE\perp BD\ (E\in BD)$. Do $BD\perp SI$ nên $BD\perp(SIE)\Rightarrow BD\perp SE$.
			\item[] Ta có $\heva{& (SBD)\cap (ABCD)=BD\\& SE\perp BD, SE\subset (SBD)\\& IE\perp BD, IE\subset (ABCD)}\Rightarrow [(SBD),(ABCD)]=(SE,IE)=\widehat{SEI}$.
			\item[] Ta có $\triangle BIE\backsim\triangle BDA$ nên $\dfrac{IE}{AD}=\dfrac{BI}{BD}\Rightarrow IE=\dfrac{a\sqrt5}{5}$, $SI=AB\cdot\dfrac{\sqrt3}{2}=\dfrac{a\sqrt3}{2}$.
			\item[] Suy ra $\tan\widehat{SEI}=\dfrac{SI}{IE}=\dfrac{\sqrt{15}}{2}\Rightarrow \widehat{SEI}=\arctan\dfrac{\sqrt{15}}{2}$.
			\item[] Vậy $[(SBD),(ABCD)]=\arctan\dfrac{\sqrt{15}}{2}$.
			
			\item Gọi $F$ là trung điểm của $CH$, khi đó $KF$ là đường trung bình của $\triangle CDH$ nên $KF\parallel CD$ và $KF=\dfrac{1}{2}CD$, suy ra tứ giác $BIKF$ là hình bình hành. 
			\item[] Do $CD\perp BC$ nên $KF\perp BC$, xét $\triangle BKC$ có 2 đường cao $CH,KF$ cắt nhau tại $F$ nên $F$ là trực tâm tam giác, suy ra $BF\perp CK$ mà $BF\parallel IK$ nên ta có $IK\perp CK\ (1)$.
			\item[] Lại có $CK\perp SI\ (2)$. Từ $(1)$ và $(2)$ suy ra $CK\perp(SIK)$, do đó ta có $(SCK)\perp(SIK)$.
			
			\item Gọi $N$ là trung điểm $CD$, khi đó ta có $\heva{& CD\perp IN\\& CD\perp SI}\Rightarrow CD\perp(SIN)$.
			\item[] Dựng $IP\perp SN$ thì suy ra $IP\perp(SCD)$. Vậy $d(I,(SCD))=IP$.
			\item[] Ta có $\dfrac{1}{IP^2}=\dfrac{1}{SI^2}+\dfrac{1}{IN^2}=\dfrac{19}{12a^2}\Rightarrow IP=\dfrac{2\sqrt{57}a}{19}$.
			\item[] Ta có $\triangle BJM\backsim\triangle DJA\Rightarrow \dfrac{BJ}{DJ}=\dfrac{BM}{DA}=\dfrac{1}{3}\Rightarrow \dfrac{BD}{JD}=\dfrac{4}{3}$.
			\item[] Suy ra $\dfrac{d(J,(SCD))}{d(B,(SCD))}=\dfrac{JD}{BD}=\dfrac{3}{4}$
			\item[] $\Rightarrow d(J,(SCD))=\dfrac{3}{4}d(B,(SCD))=\dfrac{3}{4}d(I,(SCD))=\dfrac{3\sqrt{57}a}{38}$.
		\end{enumerate}
