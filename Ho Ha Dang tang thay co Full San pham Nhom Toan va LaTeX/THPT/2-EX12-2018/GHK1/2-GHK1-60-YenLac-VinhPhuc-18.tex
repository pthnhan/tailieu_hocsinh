\begin{ex}%[Đề TT Lần 1, Yên Lạc, Vĩnh Phúc 2018]%[2D1B5]

Đồ thị sau đây là của hàm số nào?
\choice
{\True $y=\dfrac{2x+1}{x+1}$}
{$y=\dfrac{x+3}{1-x}$}
{$y=\dfrac{x+2}{x+1}$}
{$y=\dfrac{x-1}{x+1}$}
}
{
\begin{tikzpicture}[scale=.8][>=stealth,x=1cm,y=1cm]
\draw[->,line width = 1pt] (-4,0)--(0,0)%
node[below right]{$O$}--(3,0) node[below]{$x$};
\draw[->,line width = 1pt] (0,-1.5) --(0,5) node[right]{$y$};
\foreach \x in {,,-1,,}
{\draw (\x,-.1)--(\x,.1) node[above left]{$\x$};}
\foreach \y in {,,2,,}
{\draw[-] (-.1,\y)--(.1,\y) node[above right]{$\y$};}
\clip (-10,-1.5) rectangle (6,5);
\draw [line width = 1.2pt]
plot[domain=-4:-1.2, samples=100] (\x, {(2*(\x)+1)/(\x+1)});
\draw [line width = 1.2pt]
plot[domain=-0.8:2.5, samples=100] (\x, {(2*(\x)+1)/(\x+1)});
\draw [-](-4,2)--(2.5,2);
\draw [-](-1,-1.5)--(-1,5);
\end{tikzpicture}
}

\loigiai{
\textbf{Lời giải tự luận}: Gọi hàm số cần tìm là $y=\dfrac{ax+b}{cx+d}$.\\
Tiệm cận đứng $x=-\dfrac{d}{c}=-1\Leftrightarrow d=c$.
Tiệm cận ngang $y=\dfrac{a}{c}=2\Leftrightarrow a=2c$.\\
Đồ thị hàm số qua điểm $A(0;1)$ nên $\dfrac{b}{d}=1\Leftrightarrow b=d$.\\
Đồ thị hàm số nhận $I(-1;2)$ làm tâm đối xứng nên cũng đi qua điểm $B(-2;3)\Rightarrow \dfrac{-2a+b}{-2c+d}=3$.\\
Từ đó ta tìm được $a=2$, $b=c=d=1$.\\
\textbf{Lời giải nhanh}: Tiệm cận đứng của đồ thị hàm số là $x=-1$ nên ta loại hàm $y=\dfrac{x+3}{1-x}$. Tiệm cận ngang của đồ thị hàm số là $y=2$ nên ta loại cách hàm $y=\dfrac{x+2}{x+1}$ và $y=\dfrac{x-1}{x+1}$.
}
\end{ex}

\begin{ex}%[Đề TT Lần 1, Yên Lạc, Vĩnh Phúc 2018]%[2D1B4]

{$1$}
{$2$}
{\True $3$}
{$4$}
