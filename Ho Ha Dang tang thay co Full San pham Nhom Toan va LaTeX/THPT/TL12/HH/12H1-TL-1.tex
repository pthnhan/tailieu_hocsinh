\section{KHÁI NIỆM VỀ KHỐI ĐA DIỆN}
\subsection{Khái niệm về hình đa diện và khối đa diện}
	\begin{dn}%Đinh Mạnh Hùng
		Hình $\mathscr H$ gồm các đa giác phẳng thoả mãn hai điều kiện sau đây được gọi là một hình đa diện\begin{itemize}
			\item Hai đa giác bất kỳ hoặc không có điểm chung, hoặc có một đỉnh chung, hoặc có một cạnh chung.
			\item Mỗi cạnh của  một đa giác là cạnh chung của đúng hai đa giác.
		\end{itemize}
	\end{dn}
	\begin{note}%Đinh Mạnh Hùng
		Mỗi hình đa diện $\mathscr H$ chia các điểm trong không gian thành hai miền không giao nhau là miền trong và miền ngoài của khối đa diện, trong đó chỉ có miền ngoài là chứa hoàn toàn một đường thẳng nào đó.
	\end{note}
\begin{dn}%Đinh Mạnh Hùng
	Hình $\mathscr H$ và các điểm nằm trong hình $\mathscr H$ được gọi là khối đa diện giới hạn bởi hình $\mathscr H.$    
\end{dn}
	
\setcounter{vd}{0}	
\begin{vd}%Đinh Mạnh Hùng%[2H1K1]
	Chứng minh rằng nếu một khối đa diện có các mặt đều là tam giác thì số mặt của nó là một số chẵn.
	\loigiai{Gọi $n$ là số mặt của đa diện đó, $p$ là số cạnh của đa diện đó\\Do các mặt của khối đa diện là tam giác nên số cạnh của $n$ tam giác đó là $3n$\\Nhưng mỗi cạnh của đa diện là cạnh chung của đúng hai mặt nên $2p=3n$, vậy $n$ phải là số chẵn.      }
\end{vd}
\begin{vd}%Đinh Mạnh Hùng%[2H1K1]
	Chứng minh rằng nếu một khối đa diện có mỗi đỉnh là đỉnh chung của một số lẻ các mặt, thì tổng số đỉnh của nó là một số chẵn.
	\loigiai{
	Gọi $k$ là số đỉnh và $p$ là số cạnh của khối đa diện\\Gọi $A_1;A_2;..A_k$ là các đỉnh và $c_1;c_2;...;c_k$ là số cạnh qua các đỉnh tương ứng.\\Vậy số cạnh qua $k$ đỉnh là: $c_1+c_2+..+c_k$ với $c_1;...:c_k$ là các số lẻ\\Nhưng mỗi cạnh lại đi qua hai đỉnh nên $2p=c_1+c_2+..+c_k$  vì vậy nên $k$ là số chẵn        }
\end{vd}

	\loigiai{Gọi $M_1$ là một mặt của khối đa diện, do $M_1$ là đa giác nên $M_1$ có ít nhất $3$ cạnh là $c_1;c_2;c_3$\\
	Gọi $M_2$ có cạnh chung $c_1$ với $M_1$, $M_2\neq M_1$\\Gọi $M_3$ có cạnh chung $c_2$ với $M_1$, $M_3\neq M_1$  , vì $c_1$ không thuộc $M_3$ nên $M_2\neq M_3$ \\  Gọi $M_4$ có cạnh chung $c_3$ với $M_1$, $M_4\neq M_1$, vì $c_1, c_2$ không thuộc $M_4$ nên $M_4\neq M_2\neq  M_3$      \\Vậy đa diện có ít nhất 4 mặt. } 
\end{bt}

	\loigiai{Gọi $d$ và $c$ lần lượt là số đỉnh và số cạnh.\\Do mỗi đỉnh là đỉnh chung của ít nhất $3$  cạnh  và qua $2$ đỉnh có đúng một cạnh nên $2c\geq 3d\Rightarrow c>d$    }
\end{bt}


\subsection{Hai đa diện bằng nhau}
\subsubsection{Phép dời hình trong không gian}
Trong không gian, quy tắc đặt tương ứng mỗi điểm $M$ với điểm $M'$ xác định duy nhất được gọi là một phép biến hình trong không gian.\\
Phép biến hình trong không gian được gọi là phép dời hình nếu nó bảo toàn khoảng cách giữa hai điểm tùy ý.
\subsubsection{Một số phép dời hình thường gặp}
\immini{
a) \textbf{Phép tịnh tiến theo vectơ} $\overrightarrow{v}$\\
 là phép biến hình biến mỗi điểm $M$ thành điểm $M'$ sao cho $\overrightarrow{MM'} = \overrightarrow{v}$
}
{
\begin{tikzpicture}[>=stealth,line cap=round,line join=round]

\end{tikzpicture}
}

\immini{
b) \textbf{Phép đối xứng qua mặt phẳng $(P)$.} \\là phép biến hình biến mỗi điểm thuộc $(P)$ thành chính nó, biến mỗi điểm không thuộc $(P)$ thành điểm $M'$ sao cho $(P)$ là mặt phẳng trung trực của $MM'$. \\
Nếu phép đối xứng qua mặt phẳng $(P)$ biến hình $(H)$ thành chính nó thì $(P)$ được gọi là mặt phẳng đối xứng của $(H)$.

Nếu phép đối xứng tâm $O$ biến hình $(H)$ thành chính nó thì $O$ được gọi là tâm đối xứng của $(H)$.

Nếu phép đối xứng qua đường thẳng $\Delta$ biến hình $(H)$ thành chính nó thì $\Delta$ gọi là trục đối xứng của $(H)$.

\textbf{Nhận xét}
\begin{enumerate}
\item[•] Thực hiện liên tiếp các phép dời hình sẽ được một phép dời hình.
\item[•] Phép dời hình biến đa diện $(H)$ thành đa diện $(H')$, biến đỉnh, cạnh, mặt của $(H)$ thành  đỉnh, cạnh, mặt tương ứng của $(H')$.
\end{enumerate}

\subsection{Phân chia và lắp ghép khối đa diện}
Ta nói khối đa diện $(H)$ có thể được chia thành $n$ khối đa diện $(H_1), (H_2), \dots , (H_n)$ hay có thể lắp ghép $n$ khối đa diện $(H_1), (H_2), \dots , (H_n)$ thành khối đa diện $(H)$ nếu  là $(H)$ hợp của $n$ khối đa diện $(H_1), (H_2), \dots , (H_n)$ và hai khối đa diện bất kì $(H_i)$ và $(H_j)$ $(i \neq j)$ không có điểm chung trong. \\
Một khối đa diện bất kì luôn có thể phân chia được thành những khối tứ diện.

\begin{vd}%[Phạm Tuấn]%[2H1B1]
Chia một khối tứ diện đều thành bốn khối tứ diện bằng nhau.
\loigiai{
Cho tứ diện đều $ABCD$, gọi $A',B',C',D'$ là trọng tâm các tam giác $BCD, ACD, ABD, ABC$. Các đường thẳng $AA',BB',CC',DD'$ đồng quy tại điểm $G$,  $G$ được trọng tâm của tứ diện. Tứ diện $ABCD$ đều nên $C$ đối xứng với $D$ qua mặt phẳng $(ABA')$ cho nên hai  tứ diện $GABD$ và $GABC$ đối xứng với nhau qua  mặt phẳng $(ABA')$, suy ra hai khối tứ diện bằng nhau, chứng minh tương tự ta suy ra bốn khối tứ diện $GABD$, $GABC$, $GBCD$ và $GACD$   bằng nhau. Vậy  khối tứ diện đều $ABCD$ được chia thành bốn khối tứ diện bằng nhau.

\draw (A')--(C')--(C)--(D)--(D')--(C);
\end{tikzpicture}

Khối lập phương $ABCD.A'B'C'D'$ được phân chia thành sáu khối tứ diện bằng nhau $A'B'BC$, $A'B'C'C$,
$A'ABC$, $A'ACD$, $A'CC'D'$ và  $A'CDD'$.  
}
\end{bt}
