\begin{bt}%[HK1, Lê Quý Đôn, Tp.Hồ Chí Minh, 2018]%[1D1B3]%[1D1B3]
	Giải các phương trình lượng giác sau
	\begin{enumerate}
	\item $\cos2x+5\sin x+2=0$.
	\item $\sin5x-\sqrt{3}\cos5x=-\sqrt{2}$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Phương trình tương đương với 
	$$1-2\sin^2x+5\sin x +2=0$$
	$$\Leftrightarrow -2\sin^2x+5\sin x+3=0\Leftrightarrow \hoac{\sin x&=-\dfrac{1}{2}\\ \sin x&=3\, (\mbox{loại}).}$$
	Từ đó ta có nghiệm của phương trình là $x=-\dfrac{\pi}{6}+k2\pi$ và $x=\dfrac{7\pi}{6}+k2\pi$.
	\item Phương trình tương đương với
	$$\dfrac{1}{2}\sin5x-\dfrac{\sqrt{3}}{2}\cos5x=-\dfrac{\sqrt{2}}{2}$$
	$$\Leftrightarrow \sin\left(5x-\dfrac{\pi}{3}\right)=\sin\left(-\dfrac{\pi}{4}\right)$$
	$$\Leftrightarrow \hoac{x&=\dfrac{\pi}{60}+\dfrac{k2\pi}{5}\\ x&=\dfrac{19\pi}{60}+\dfrac{k2\pi}{5}},\, k\in\mathbb{Z}.$$
\end{enumerate}
}
\end{bt}
\begin{bt}%[HK1, Lê Quý Đôn, Tp.Hồ Chí Minh, 2018]%[1D2B2]
	Giải phương trình $A_x^1+C_x^2+C_x^3=14$.
	\loigiai{
	+ Điều kiện: $x\ge 3; x\in\mathbb{N}$.\\
	+ Ta có phương trình
	$$\dfrac{x!}{(x-1)!}+\dfrac{x!}{(x-2)!2!}+\dfrac{x!}{(x-3)!3!}=14$$
	$$\Leftrightarrow x^3+5x-84=0\Leftrightarrow x=4.$$
	Đối chiếu điều kiên, ta có nghiệm của phương trình là $x=4$.
	}
\end{bt}
\begin{bt}%[HK1, Lê Quý Đôn, Tp.Hồ Chí Minh, 2018]%[1D2K3]
	Tìm số hạng không chứa $x$ trong khai triển $\left(2x^3-\dfrac{1}{x^2}\right)^{10}, (x\ne 0)$.
	\loigiai{
	Số hạng tổng quát $T_{k+1}=(-1)^k\cdot C^k_{10}\cdot\left(2x^3\right)^{10-k}\cdot\left(\dfrac{1}{x^2}\right)^k=(-1)^k\cdot C^k_{10}\cdot 2^{10-k}\cdot x^{30-5k}$.\\
	Để $T_{k+1}$ không chứa $x$ khi và chỉ khi $30-5k=0\Leftrightarrow k=6$.\\
	Vậy số hạng không chứa $x$ là $T_7=(-1)^6\cdot
	C^6_{10}\cdot2^4=3360$.
 	}
\end{bt}
\begin{bt}%[HK1, Lê Quý Đôn, Tp.Hồ Chí Minh, 2018]%[1D2B1]%[1D2K1]
	Từ các chữ số $0;2;4;5;7;8;9$ có thể lập được bao nhiêu số tự nhiên:
	\begin{enumerate}
	\item Gồm $5$ chữ số đôi một khác nhau và là số chẵn.
	\item Gồm $5$ chữ số đôi một khác nhau sao cho phải có mặt hai chữ số $2$ và $9$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item Gọi số có $5$ chữ số phải tìm là $\overline{abcde}$, ($a\ne 0$).\\
	+ Trường hợp $e=0$ thì có $A^4_6$ số dạng $\overline{abcd}$.\\
	+ Trường hợp $e\ne 0$ thì $e$ có $3$ cách chọn, $a$ có $5$ cách chọn, $\overline{bcd}$ có $A^3_5$ cách chọn. Suy ra trường hợp này có $3\cdot 5\cdot A^3_5$ số.\\
	Tóm lại, số số gồm $5$ chữ số đôi một khác nhau và là số chẵn là $A^4_6+15\cdot A^3_5=1260$.
	\item Thay số $0$ bằng số $1$.\\
	Ta có số số gồm $5$ chữ số đôi một khác nhau sao cho luôn có mặt chữ số $2$ và chữ số $9$ là $C^3_5\cdot P_5$.\\
	Số bộ số có dạng $[0abcd]$ sao cho luôn có mặt chữ số $2$ và chữ số $9$ là $C^2_4\cdot P_4$.\\
	Vậy số chữ số gồm $5$ chữ số đôi một khác nhau sao cho luôn có mặt chữ số $2$ và chữ số $9$ là
	$$C^3_5\cdot P_5-C^2_4\cdot P_4=1056.$$
	\end{enumerate}
	}
\end{bt}
\begin{bt}%[HK1, Lê Quý Đôn, Tp.Hồ Chí Minh, 2018]%[1D2K2]%[1D2G2]
	Người ta xếp ngẫu nhiên $4$ quyển sách Toán, $5$ quyển sách Lí và $6$ quyển sách Hóa vào kệ dài. Tính xác suất sao cho
	\begin{enumerate}
	\item Xếp được các quyển sách Toán đứng cạnh nhau, các quyển sách Lí đứng cạnh nhau và các quyển sách Hóa đứng cạnh nhau.
	\item Không có quyển sách Lí nào đứng cạnh nhau.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item Hoán vị $4$ quyển sách Toán có $4!$ cách, đóng thành gói $1$.\\
	Hoán vị $5$ quyển sách Lí có $5!$ cách, đóng thành gói $2$.\\
	Hoán vị $6$ quyển sách Hóa có $6!$ cách, đóng thành gói $3$.\\
	Hoán vị $3$ gói ta có $3!$ cách xếp.\\
	Vậy tổng số cách xếp là $3!4!5!6!=12441600$.
	\item Xếp $10$ quyển gồm Toán và Hóa theo thứ tự ta có $10!$ cách xếp.\\
	Xếp $5$ quyển Lí vào khoảng giữa các quyển sách Toán và Hóa ta có $A^5_{11}$ cách.\\
	Vậy tổng số cách xếp là $10!\cdot A^5_{11}$.
	\end{enumerate}
	}
\end{bt}
