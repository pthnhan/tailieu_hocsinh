\begin{ex}%[WTT Bài toán hàm ẩn, Nguyễn Như Thường, Phản biện Ngô Quang Anh]%[2D1G2-6]
	Cho hàm số $f(x)=mx^3-3mx^2+(3m-2)x+2-m$ với $m$ là tham số thực. Có bao nhiêu giá trị nguyên của tham số $m\in [-10;10]$ để hàm số $g(x)=\left|f(x)\right|$ có \textbf{đúng} $5$ điểm cực trị?
\choice
	{$7$}	
	{$9$}
	{\True $10$}
	{$11$}
\loigiai{
		\textbf{Cách 1: }Để $g(x)=\left|f(x)\right|$ có $5$ điểm cực trị $\Leftrightarrow f(x)=0$ có $3$ nghiệm phân biệt.$\quad (*)$\\ 
		Xét $f(x)=0\Leftrightarrow (x-1)(mx^2-2mx+m-2)=0\Leftrightarrow \left[\begin{aligned}&x=1\\&mx^2-2mx+m-2=0\quad (1)\end{aligned}\right.$.
		\begin{eqnarray*}
		\text{Do đó (*)}&\Leftrightarrow&\text{phương trình (1) có hai nghiệm phân biệt khác 1}\\&\Leftrightarrow&\left\{\begin{aligned}&m\ne 0\\&\Delta'=m^2-m(m-2)>0\\&f(1)=-2\ne0\end{aligned}\right.\\
		&\Leftrightarrow& m>0 \xrightarrow[{m\in[-10;10]}]{m\in\mathbb{Z}}m\in\left\{1;2;3;\ldots;10\right\}.
		\end{eqnarray*}
		\textbf{Cách 2: }Hàm số $y=\left|mx^3-3mx^2+(3m-2)x+2-m\right|$ có $5$ điểm cực trị\\
		$\Leftrightarrow$ đồ thị hàm số $y=mx^3-3mx^2+(3m-2)x+2-m$ cắt trục $Ox$ tại $3$ điểm phân biệt\\
		$\Leftrightarrow$ Phương trình $mx^3-3mx^2+(3m-2)x+2-m=0\ (1)$ có $3$	nghiệm phân biệt.\\
		Ta có $(1)\Leftrightarrow (x-1)(mx^2-2mx+m-2)=0\Leftrightarrow \left[\begin{aligned}&x=1\\&f(x)=mx^2-2mx+m-2=0\ (2)\end{aligned}\right.$.
		\begin{eqnarray*}
		\text{Yêu cầu bài toán}&\Leftrightarrow&\text{phương trình (2) có hai nghiệm phân biệt khác 1}\\
		&\Leftrightarrow&\left\{\begin{aligned}&m\ne0\\&\Delta'=m^2-m(m-2)>0\\&f(1)=-2\ne0\end{aligned}\right.\Leftrightarrow m>0.
		\end{eqnarray*}
		Vì $m$ nguyên và $m\in[-10;10]$, nên $m\in\left\{1,2,3,\ldots\right\}$.\\
		Vậy có $10$ giá trị của  thỏa mãn yêu cầu bài toán.
	}
\end{ex}
