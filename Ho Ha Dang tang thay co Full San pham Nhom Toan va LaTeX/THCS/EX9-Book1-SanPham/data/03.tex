
\section{PHƯƠNG PHÁP DÙNG TÍNH CHẤT CỦA SỐ CHÍNH PHƯƠNG}

\subsection{Sử dụng tính chất về chia hết của số chính phương}
Số chính phương là bình phương của một số tự nhiên.\\
Các tính chất thường dùng của số chính phương về tính chia hết là
\begin{itemize}
\item Số chính phương không tận cùng bằng $2$, $3$, $7$, $8$.
\item Số chính phương chia hết cho số nguyên tố $p$ thì chia hết cho $p^2$.
\item Số chính phương chia cho $3$ có số dư là $0$ hoặc $1$.
\item Số chính phương chia cho $4$ có số dư là $0$ hoặc $1$.
\item Số chính phương chia cho $8$ có số dư là $0$, hoặc $1$, hoặc $4$.
\end{itemize}

\begin{vd}
Tìm các số nguyên $x$ để $9x+5$ là tích của hai số nguyên liên tiếp.
\loigiai{
\textbf{Cách 1.} Giả sử $9x+5=n(n+1)$ với $n$ nguyên thì 
\begin{align*}
36x+20=4n^2+4 & \Leftrightarrow 36x+21=4n^2+4n+1\\
& \Leftrightarrow 3(12x+7)=(2n+1)^2.
\end{align*}
Số chính phương $(2n+1)^2$ chia hết cho $3$ nên cũng chia hết cho $3^2=9$. Ta lại có $12x+7$ không chia hết cho $3$ nên $3(12x+7)$ không chia hết cho $9$.\\
Mâu thuẫn trên chứng tỏ không tồn tại số nguyên $x$ nào để $9x+5=n(n+1)$.\\
\textbf{Cách 2.} Giả sử $9x+5=n(n+1)$ với $n$ nguyên.\\
Biến đổi $n^2+n-(9x+5)=0$.\\
Để phương trình bậc hai đối với $n$ có nghiệm nguyên, ta phải có $\Delta$ là số chính phương.\\
Ta thấy $\Delta=1+4(9x+5)=36x+21$, chia hết cho $3$ nhưng không chia hết cho $9$, nên không là số chính phương.\\
Vậy không tồn tại số nguyên $n$ nào để $9x+5=n(n+1)$, tức là không tồn tại số nguyên $x$ để $9x+5$ là tích của hai số nguyên liên tiếp.
}
\end{vd}

\subsection{Tạo ra bình phương đúng}

\begin{vd}
Tìm nghiệm nguyên của phương trình $2x^2+4x=19-3y^2$.
\loigiai{
Ta có $2x^2+4x+2=21-3y^2\Leftrightarrow 2(x+1)^2=3(7-y^2).$\hfill$ (1)$\\
Ta thấy $3(7-y^2)\, \vdots \, 2\Rightarrow (7-y^2)\, \vdots \, 2\Rightarrow y$ lẻ.\\
Ta lại có $7-y^2\geq 0\Rightarrow y^2=1$.\\
Khi đó $(1)$ có dạng $2(x+1)^2=18$.\\
Ta được $x+1=3$ hoặc $x+1=-3$, do đó $x_1=2$; $x_2=-4$.\\
Các cặp số $(x;y)$ bằng $(2;1)$, $(2;-1)$, $(-4;1)$, $(-4;-1)$ thỏa mãn $(1)$ nên là nghiệm của phương trình.
}
\end{vd}

\begin{note}
Có thể viết phương trình đã cho dưới dạng phương trình bậc hai đối với ẩn $x$ rồi sử dụng điều kiện $\Delta\geq 0$ để có $7-y^2\geq 0$.
\end{note}

\subsection{Tạo ra tổng các số chính phương}

\begin{vd}
Tìm nghiệm nguyên của phương trình $4x^2+4x+y^2-6y=24$.
\loigiai{
Ta có $4x^2+4x+y^2-6y=24\Leftrightarrow (2x+1)^2+(y-3)^2=34$.\\
Viết $34$ dưới dạng $a^2+b$, trong đó $a$ lẻ, ta có $34=1^2+33=3^2+25=5^2+9$.\\
Chỉ có hai trường hợp cho $b$ là số chính phương $34=3^2+5^2=5^2+3^2$.\\
Chú ý rằng $2x+1\geq 3$ và $y-3\geq -2$ nên $\heva{& 2x+1=3\\& y-3=5}$ hoặc $\heva{& 2x+1=5\\& y-3=3}$\\
Nghiệm nguyên dương $(x;y)$ của phương trình là $(1;8)$, $(2;6)$.
}
\end{vd}

\begin{note}
Có thể viết phương trình đã cho dưới dạng phương trình bậc hai đối với $y$ rồi sử dụng điều kiện $\Delta '\geq 0$ để có $(2x+1)^2\geq 34$. Cách này dài hơn.
\end{note}

\subsection{Xét các số chính phương liên tiếp}
Hiển nhiên giữa hai số chính phương liên tiếp, không có số chính phương nào. Do đó với mọi số nguyên $a$, thì:
\begin{itemize}
\item Không tồn tại số nguyên $x$ nào để $a^2<x^2<(a+1)^2.$
\item Nếu có số nguyên $x$ sao cho $a^2<x^2<(a+2)^2$ thì $x^2=(a+1)^2$.
\end{itemize}

\begin{vd}
Tìm nghiệm nguyên của phương trình $x^4-y^4=3y^2+1$.
\loigiai{
Ta có 
\begin{align*}
& x^4=y^4+3y^2+1\geq y^4+2y^2+1=(y^2+1)^2\tag{1}
\\& x^4=y^4+3y^2+1<y^4+4y^2+4=(y^2+2)^2\tag{2}
\end{align*}
Từ $(1)$ và $(2)$ suy ra $(y^2+1)^2\leq x^4<(y^2+2)^2$.\\
Do đó $(y^2+1)^2=x^4$.\hfill$(3)$\\
Thay $x^4$ bởi $y^4+3y^2+1$ vào $(3)$, ta được $y^4+2y^2+1=y^4+3y^2+1\Leftrightarrow y^2=0\Leftrightarrow y=0$.\\
Suy ra $x=1$ hoặc $x=-1$.\\
Đáp số: Nghiệm $(x;y)$ của phương trình là $(1;0)$, $(-1;0)$.
}
\end{vd}

\begin{note}
Có thể giải phương trình đã cho bằng cách đưa về $(2x^2)^2=(2y^2+3)^2-5.$
\end{note}

\begin{vd}
Chứng minh rằng với mọi số nguyên $k$ cho trước, không tồn tại số nguyên dương $x$ sao cho $x(x+1)=k(k+2)$.
\loigiai{
Giả sử $x(x+1)=k(k+2)$ với $k$ nguyên, $x$ nguyên dương. Ta có
\begin{align*}
& x^2+x=k^2+2k\\
\Leftrightarrow & x^2+x+1=(k+1)^2.
\end{align*}
Do 
\begin{align*}
x>0\Rightarrow x^2<x^2+x+1=(k+1)^2\tag{1}
\end{align*}
Cũng do $x>0$ nên 
\begin{align*}
(k+1)^2=x^2+x+1<x^2+2x+1=(x+1)^2\tag{2}
\end{align*}
Từ $(1)$ và $(2)$ suy ra $x^2<(k+1)^2<(x+1)^2$, điều này không xảy ra.\\
Vậy với $k$ đã cho không tồn tại số nguyên dương $x$ để $x(x+1)=k(k+2).$
}
\end{vd}


\subsection{Sử dụng điều kiện biệt số $\Delta$ là số chính phương}
Ở những phương trình dạng $f(x,y)=0$ với hệ số nguyên có thể viết được dưới dạng phương trình bậc hai đối với một ẩn, chẳng hạn đối với ẩn $x$, ngoài điều kiện $\Delta \geq 0$ để phương trình ẩn $x$ có nghiệm, muốn phương trình có nghiệm nguyên còn cần $\Delta$ là số chính phương, vì nếu $\Delta$ không là số chính phương thì $x$ là số vô tỉ.
Chú ý rằng $\Delta$ là số chính phương là điều kiện cần nhưng chưa đủ để phương trình có nghiệm nguyên. Do đó phải thử giá trị tìm được vào phương trình đã cho hoặc tìm ra cụ thể nghiệm nguyên của phương trình đó.


\begin{vd}
  Tìm nghiệm nguyên của phương trình
  \[x^2+2y^2+3xy+2x+3y+4=0.\tag{1}\]
    \loigiai{
     Viết (1) dưới dạng phương trình bậc hai đối với ẩn $x$ được
     \[x^2+(3y+2)x+ (2y^2+3y+4)=0.\tag{2}\]
 Ta có     $\Delta=(3y+2)^2-4(2y^2+3y+4)=y^2-12.$
  Để phương trình (2) có nghiệm nguyên, cần có $y^2-12$ là số chính phương.
     Đặt $y^2-12=m^2$ với $m\in \mathbb{N},$ ta có
     $$y^2-m^2=12\Leftrightarrow (y+m)(y-m)=12.$$
     Ta có $y+m$ và $y-m$ là các ước của $12$ và $y+m-(y-m)=2m$ nên $y+m$ và $y-m$ cùng tính chẵn lẻ và $y+m\geq y-m.$ Do đó
     $\heva {y+m=6\\
     y-m=2}$ hoặc $\heva{y+m=-2\\ y-m=-6.}$\\
     Từ đó $y=4$ hoặc $y=-4.$
     \begin{itemize}
     \item Với $y=4,$ thay vào (2) được $x^2+14x+48=0.$ Từ đó ta có $x_1=-6;x_2=-8.$
     \item Với $y=-4,$ thay vào (2) được $x^2-10x+24=0.$ Ta có $x_3=4;x_4=6.$
     \end{itemize}
     \textit{Đáp số:} Nghiệm $(x;y)$ của phương trình là $(-6;4), (-8;4), (4;-4), (6;-4).$\\ 
     \textbf{Lưu ý:} Nếu chỉ sử dụng điều kiện $\Delta\geq 0,$ ta được $y^2\geq 12.$ Chưa chặn được giá trị của $y.$
    }
\end{vd}
\subsection{Sử dụng tính chất:} \textit{Nếu hai số nguyên dương nguyên tố cùng nhau có tích là một số chính phương trình mỗi số đều là số chính phương.}\\
Giả sử $ab=c^2$ với $a,b,c\in\mathbb{N^*}, (a,b)=1.$\\
Nếu trong $a$ và $b$ có một số, chẳng hạn $a$ chứa thừa số nguyên tố $p$ với số mũ lẻ thì $b$ không chứa thừa số $p$ nên $c^2$ chứa thừa số nguyên tố $p$ với số mũ lẻ, trái với giả thiết $c^2$ là số chính phương.\\
Vậy $a$ và $b$ đều chỉ chứa thừa số nguyên tố với số mũ chẵn, tức là $a$ và $b$ đều là số chính phương.
\begin{vd}
Giải phương trình ba ẩn $x,y,z$ với nghiệm nguyên dương
\[xy=z^2.\tag{1}\]
\loigiai{
Trước hết ta xét ƯCLN $(x,y,z)=1.$ Thật vậy nếu bộ ba số $(x_0;y_0;z_0)$ thỏa mãn $(1)$ và có ƯCLN bằng $d,$ giả sử $x_0=dx_1, y_0=dy_1, z_0=dz_1$ thì bộ $(x_1,y_1,z_1)$ cũng là nghiệm của phương trình (1) với ƯCLN $(x_1,y_1,z_1)=1.$ \\
Với ƯCLN $(x,y,z)=1$ thì $x,y,z$ đôi một nguyên tố cùng nhau, vì nếu hai trong  ba số $x,y,z$ có ước chung là $d$ thì số còn lại cũng chia hết cho $d.$\\
Ta có $z^2=xy$ mà $(x,y)=1$ nên $x=a^2, y=b^2$ với $a,b\in\mathbb{N^*}.$\\
Suy ra $z^2=xy=(ab)^2,$ do đó $z=ab$.\\
Như vậy : $\heva{x=ta^2\\ y=tb^2\\ z=tab}$ (với $t,a,b$ là các số nguyên dương tùy ý).\\
Thử lại các số $x,y,z$ có dạng trên thỏa mãn (1).\\
Công thức trên cho ta các nghiệm nguyên dương của phương trình (1).
}
\end{vd}
\textbf{Lưu ý:} Mệnh đề sau không đúng: Nếu hai số nguyên dương nguyên tố cùng nhau có tích chia hết cho một số chính phương thì tồn tại không hai số đó là số chính phương (!)\\
Chẳng hạn: $7.8\, \vdots \, 4$ nhưng cả $7$ và $8$ đều không phải là số chính phương.
\subsection{Sử dụng tính chất:} Nếu hai số nguyên dương liên tiếp có tích là một số chính phương thì một trong hai số đó bằng $0.$\\
Giả sử $a(a+1)=k^2\quad (1)$ với $a\in \mathbb{Z}, k\in\mathbb{N}.$\\
Giả sử $a\neq 0, a+1\neq 0$ thì $k^2\neq 0.$ Do $k\in \mathbb{N}$ nên $k>0.$\\
Từ (1) suy ra $a^a+a=k^2\Leftrightarrow 4a^2+4a=4k^2\Leftrightarrow (2a+1)^2=4k^2+1.\hfill (2)$\\
Do $k>0$ nên $4k^2<4k^2+1<4k^2+4k+1.\hfill (3)$\\
Từ (2)và (3) suy ra $(2k)^2<(2a+1)^2<(2k+1)^2,$ điều này không xảy ra.\\
Vậy nếu $a(a+1)=k^2$ thì tồn tại một trong hai số $a$ hoặc $a+1$ bằng $0.$\\
\textbf{Lưu ý:} Nếu trong hai số  nguyên liên tiếp có tích là một số chính phương thì chưa thể kết luận mỗi số đều là một số chính phương. Chẳng hạn: hai số $-1$ và $0.$
\begin{vd}\label{chu1ng}
Tìm nghiệm nguyên của phương trình 
\[x^2+xy+y^2=x^2y^2.\tag{1}\]
\loigiai{
Thêm $xy$ vào hai vế ta được $x^2+2xy+y^2=x^2y^2+xy\Leftrightarrow (x+y)^2=xy(xy+1).$\\
Ta thấy $xy$ và $xy+1$ là hai số chính phương liên tiếp, có tích là một số chính phương nên tồn tại một số bằng $0.$

\begin{itemize}
\item Xét $xy=0.$ Từ (1) có $x^2+y^2=0$ nên $x=y=0.$
\item Xét $xy+1=0.$ Ta có $xy=-1$ nên $(x,y)=(1;-1)$ hoặc $(-1;1).$
\end{itemize}
Thử lại, $(x;y)$ lấy các giá trị $(0;0), (1;-1), (-1;1)$ đều là nghiệm nguyên của phương trình đã cho.\\
\textbf{Cách giải khác.}\\
\textit{Cách 2.} Đưa về phương trình ước số:
{\allowdisplaybreaks
\begin{align*}
&4x^2+4xy+4y^2=4x^2y^2\\
\Leftrightarrow& 4x^2+8xy+4y^2=4x^2y^2+4xy\\
\Leftrightarrow&(2x+2y)^2=(2xy+1)^2-1\\
\Leftrightarrow&(2xy+1)^2-(2x+2y)^2=1\\
\Leftrightarrow&(2xy+1+2x+2y)(2xy+1-2x-2y)=1.
\end{align*}}
\textit{Cách 3.} Dùng tính chất của số chính phương và đưa về phương trình ước số:
{\allowdisplaybreaks
	\begin{align*}
&4x^2+4xy+4y^2=4x^2y^2\\
\Leftrightarrow&(2x+y)^2+3y^2=4x^2y^2\\
\Leftrightarrow&(2x+y)^2=y^2(4x^2-3).
\end{align*}}
\begin{itemize}
\item Nếu $y=0$ thì $x=0,$ ta có: $(x;y)=(0;0)$ là một nghiệm.
\item Nếu $y\neq 0$ thì $4x^2-3$ phải là số chính phương.

Ta có $4x^2-3=k^2 (k\in\mathbb{N}),$ đưa về phương trình ước số
$$(2x+k)(2x-k)=3.$$
Ta tìm được $x_1=1;x_2=-1.$ Từ đó ta tìm được $y.$
\end{itemize}
\textit{Cách 4. } Dùng bất đẳng thức sắp thứ tự các ẩn.\\
Không mất tính tổng quát giả sử $|x|\leq |y|,$ thế thì $x^2\leq y^2, xy\leq |xy|\leq y^2.$\\
Do đó $x^2y^2=x^2+xy+y^2\leq y^2+y^2+y^2=3y^2.$\\
\begin{itemize}
\item Nếu $y=0$ thì $x=0.$
\item Nếu $y\neq 0$ thì chia cả hai vế cho $y^2$ ta được $x^2\leq 3$. Do đó $x^2=1$. Ta có thêm hai nghiệm $(x;y)$ là $(1;-1)$ và $(-1;1)$.
\end{itemize}
\textit{Cách 5.} Dùng bất đẳng thức xét từng khoảng giá trị của ẩn.\\
Với $|x|\geq 2$ và $|y|\geq 2$ thì $x^2\geq 4$ và $y^2\geq 4$ nên $x^2y^2\geq 4y^2$ và $x^2x^2\geq 4 x^2.$ 
Do đó
$$\begin{array}{rl}
x^2y^2\geq 2(x^2+y^2)&=x^2+y^2+x^2+y^2\geq x^2+y^2+2|xy|\\
&=x^2+y^2+|xy|+|xy|\\
&\geq x^2+y^2+xy+4>x^2+y^2=xy,
\end{array}$$
Trái với đề bài.\\
Vậy $|x|<2$ hoặc $|y|<2$.
Do vai trò $x,y$ trong phương trình là như nhau nên ta chỉ cần xét $|x|<2$.\\
Thử với $x=0,x=1, x=-1$ ta được các nghiệm $(x;y)$ là:
$$(0;0), (1;-1), (-1;1).$$
\textit{Cách 6.} Đưa về phương trình bậc hai đối với $x$.
\[(y^2-1)x^2-yx-y^2=0.\tag{2}\]
\begin{itemize}
\item Xét $y=1$ thì (2) có dạng $-x-1=0$ được $x=-1$.
\item Xét $y=-1$ thì (2) trở thành $x-1=0$ được $x=1.$
\item Xét $y\neq \pm 1$ thì (2) là một phương trình bậc hai đối với $x.$
$$\Delta =y^2+4y^2(y^2-1)=y^2(4y^2-3).$$
Ta phải có $\Delta $ là số chính phương.\\
Nếu $y=0$ thì từ (2) suy ra $x=0.$\\
Nếu $y\neq 0$ thì $4y^2-3$ là số chính phương.\\
Ta có $4y^2-3=k^2 (k\in\mathbb{N})$ nên $(2y+k)(2y-k)=3.$\\
Ta tìm được $y=\pm 1,$ loại vì đang xét $y\neq \pm 1.$
\end{itemize}
}
\end{vd}

\begin{bt}
Tìm nghiệm nguyên của phương trình $3x^2+4y^2=6x+13$.
\loigiai{
Biến đổi: $3x^2-6x+3=16-4y^2\Leftrightarrow 3(x-1)^2=4(4-y^2).$\\
Ta có $4-y^2\geq 0$ và $(4-y^2)\, \vdots \, 3$ nên $y^2=1$ hoặc $y^2=4$.\\
Đáp số: Nghiệm $(x;y)$ là $(3;1)$, $(3;-1)$; $(-1;1)$, $(-1;-1)$, $(1;2)$, $(1;-2)$.
}
\end{bt}
  
\begin{bt}
Tìm các số nguyên $x$ thỏa mãn một trong các điều kiện sau:
\begin{multicols}{2}
\begin{enumerate}
\item $2x+1$ là số chính phương.
\item $4x+1$ là số chính phương.
\item $8x+1$ là số chính phương.
\item $16x+1$ là số chính phương.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item Đặt $2x+1=(2k+1)^2$ với $k\in\mathbb{N}$.\\
Từ đó $2x+1=4k^2+4k+1\Leftrightarrow x=2k(k+1)$.
\item Đặt $4x+1=(2k+1)^2$ với $k\in\mathbb{N}$.\\
$\Rightarrow x=k(k+1)$.
\item Đặt $8x+1=(2k+1)^2$ với $k\in\mathbb{N}$.\\
$\Rightarrow x=\dfrac{k(k+1)}{2}\cdot$
\item Đặt $16x+1=(2k+1)^2$ với $k\in\mathbb{N}$.\\
Ta có $16x=4k^2+4k\Rightarrow x=\dfrac{k(k+1)}{4}$.\\
Để $x$ nguyên ta phải có $k\, \vdots \, 4$ hoặc $(k+1)\, \vdots \, 4$.
\begin{itemize}
\item Với $k=4t$, $(t\in\mathbb{N})\Rightarrow x=t(4t+1)$.
\item Với $k+1=4t$, $(t\in\mathbb{N})\Rightarrow x=t(4t-1)$.
\end{itemize}
\end{enumerate}
}
\end{bt}

\begin{bt}
Chứng minh rằng $2x^2+3$ không là số chính phương với mọi số tự nhiên $x$.
\loigiai{
Giả sử $2x^2+3=y^2$ với $y\in\mathbb{N}$.\\
Ta có $2x^2+2=y^2-1=(y-1)(y+1)\Rightarrow y-1$ và $y+1$ phải cùng chẵn.\\
Đặt $y-1=2k$, $y+1=2k+2$ với $k\in\mathbb{N}$.\\
Ta có $2x^2+2=2k(2k+2)$ nên $x^2+1=2k(k+1)$.\hfill$(1)$\\
Vế phải của $(1)$ chia hết cho $4$, còn vế trái chia $4$ dư $1$ (nếu $x$ chẵn), chia cho $4$ dư $2$ (nếu $x$ lẻ).\\
Vậy $2x^2+3$ không là số chính phương với mọi số tự nhiên $x$.
}
\end{bt}

\begin{bt}
Tìm nghiệm nguyên của mỗi phương trình sau:
\begin{multicols}{2}
\begin{enumerate}
\item $x^2-6x+y^2+10y=24$.
\item $x^2-3x+y^2-6y+10=0$.
\item $(x+y+1)^2=3(x^2+y^2+1)$.
\item $x^2+5y^2-4xy-4y+4=0$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item Viết phương trình dưới dạng $(x-3)^2+(y+5)^2=58$.\\
Viết $58$ thành tổng $a^2+b$ trong đó $a^2<b$ ta có $$58=1^2+57=2^2+54=3^2+49=4^2+42=5^2+33,$$
chỉ có một trường hợp cho $b$ là số chính phương $58=3^2+7^2.$\\
Ta có bảng giá trị sau

\hline
$x-3$ & $3$ & $3$ & $-3$ & $-3$ & $7$ & $7$ & $-7$ & $-7$\\
\hline
$y+5$ & $7$ & $-7$ & $7$ & $-7$ & $3$ & $-3$ & $3$ & $-3$\\
\hline
\end{tabular}
\end{center}
Nên có các nghiệm $(x;y)$ là
\begin{center}
\begin{tabular}{|m{1cm}|m{1cm}|m{1cm}|m{1cm}|m{1cm}|m{1cm}|m{1cm}|m{1cm}|m{1cm}|}
\hline
$x$ & $6$ & $6$ & $0$ & $0$ & $10$ & $10$ & $-4$ & $-4$\\\hline
$y$ & $2$ & $-12$ & $2$ & $-12$ & $-2$ & $-8$ & $-2$ & $-8$\\\hline
\end{tabular}
\end{center}
\item Viết phương trình dưới dạng $(2x-3)^2+(2y-6)^2=5$.\\
Số $5$ chỉ có một cách viết thành tổng hai số chính phương là $1^2+2^2$. Do $2x-3$ là số lẻ nên có bảng giá trị
\begin{center}
\begin{tabular}{|m{2cm}|m{1cm}|m{1cm}|m{1cm}|m{1cm}|}

Do đó nghiệm $(x;y)$ là 
\begin{center}
\begin{tabular}{|m{2cm}|m{1cm}|m{1cm}|m{1cm}|m{1cm}|}
\hline
$x$ & $2$ & $2$ & $1$ & $1$\\\hline
$y$ & $4$ & $2$ & $4$ & $2$\\\hline
\end{tabular}
\end{center}
\item Viết phương trình dưới dạng $(x-y)^2+(x-1)^2+(y-1)^2=0.$\\
Đáp số: Nghiệm $(x;y)$ là $(1;1)$.
\item Viết phương trình dưới dạng $(x-2y)^2+(y-2)^2=0$.\\
Đáp số: Nghiệm $(x;y)$ là $(4;2)$.
\end{enumerate}
}
\end{bt}

\begin{bt}
Tìm nghiệm tự nhiên của phương trình $x^2-y^2=y+1$.
\loigiai{
Ta có $x^2=y^2+y+1\leq y^2+2y+1$ nên $y^2<x^2\leq (y+1)^2$.\\
Do đó $x^2=(y+1)^2\Leftrightarrow y^2+y+1=y^2+2y+1\Leftrightarrow y=0.$\\
Đáp số: Nghiệm $(x;y)$ là $(1;0)$.
}
\end{bt}

\begin{bt}
   Có tồn tại hay không hai số nguyên dương $x$ và $y$ sao cho $x^2+y$ và $y^2+x$ đều là số chính phương?
    \loigiai{
        Giả sử $y\leq x.$ Ta có: $x^2<x^2+y\leq x^2+x<(x+1)^2.$        
        Số $x^2+y$ nằm giữa hai số chính phương liên tiếp nên không thể là số chính phương. 
    }
\end{bt}
   \begin{bt}
  Tìm nghiệm nguyên của phương trình
   $$x^2y^2-xy=x^2+2y^2.$$
    \loigiai{
       Viết phương trình đã cho dưới dạng $(x^2-2)y^2-xy-x^2=0.\quad\quad (1)$
       
       Vì $x^2-2\neq 0$ nên (1) là phương trình bậc hai đối với $y.$
       $$\Delta =x^2+4x^2(x^2-2)=4x^4-7x^2=x^2(4x^2-7).$$
       Ta có $\Delta$ là số chính phương.
       
       Nếu $x=0$ thì $y=0.$ Nếu $x\neq 0$ thì $4x^2-7$ phải là số chính phương.
       
       Đặt $4x^2-7=m^2 (m\in\mathbb{N})$ ta được
       
       $4x^2-m^2=7\Leftrightarrow (2x+m)(2x-m)=7.$ Ta xét bảng giá trị sau
      \begin{center}
       
       \begin{tabular}{|c|c|c|}
       \hline
       $2x+m$&$7$&$1$\\
       \hline
       $2x-m$&$-1$&$-7$\\
       \hline
       \end{tabular}
      \end{center}
      Từ đó suy ra $x$ bằng $2$ hoặc $-2$.
      
      Đáp số: Nghiệm $(x;y)$ là $(0;0), (2;-1), (2;2), (-2;1), (-2;-2).$
      
      Lưu ý. Nếu viết phương trình đã cho thành phương trình bậc hai với $x$
      
      $(y^2-1)x^2-yx-2y^2=0,$ ta phải xét $y=1, y=-1, y\neq \pm 1.$
      
      Với $y\neq \pm 1$ thì $\Delta =y^2(8y^2-7).$ Ta gặp khó khăn khi giải điều kiện $8y^2-7$ là số chính phương.
    }
\end{bt}
\begin{bt}
   Tìm nghiệm nguyên dương của mỗi phương trình sau:
   \begin{multicols}{2}
  \begin{enumerate}[a)]
  \item $x(x^2+x+1)=4y(y+1)$;
  \item $x^4-2y^2=1.$
  \end{enumerate} 
  \end{multicols} 
    \loigiai{
        \begin{enumerate}[a)]
        \item $x(x^2+x+1)=4y(y+1).$ Cộng $1$ vào hai vế ta được 
        $$x^3+x^2+x+1=4y^2+4y+1\Leftrightarrow (x^2+1)(x+1)=(2y+1)^2.$$
        Vế phải là số lẻ nên $x^2+1$ và $x+1$ cùng lẻ, đồng thời $x^2+1$ và $x+1$ cùng là số dương. Ta lại thấy $(x^2+1, x+1)=1.$ Nên $x^2+1$ và $x+1$ là số chính phương. Nên $x^2$ và $x^2+1$ là hai số liên tiếp và cùng là 2 số chính phương nên $x=0$, từ đó suy ra $y=0$ hoặc $y=-1.$
        
        Đáp số: Nghiệm $(x;y)$ là $(0;0), (0;-1).$
        \item Xét phương trình $x^4-2y^2=1$.\hfill$(1)$\\        
        $x$ và $y$ chỉ có số mũ chẵn, do đó ta giả sử $x\geq 0, y\geq 0.$
        
        Hiển nhiên $x$ lẻ, do đó $x^4$ chia $4$ dư $1$, suy ra $y^2$ chẵn, do đó $y$ chẵn.
        
        Đặt $x=2a+1, y=2b$ với $a,b\in\mathbb{N}.$
        
        Thay các biểu thức này vào (1) ta được:
      $$  \begin{array}{rl}
        (4a^2+4a+1)^2-1=8b^2&\Leftrightarrow (4a^2+4a+2)(4a^2+4a)=8b^2\\
        & (2a^2+2a+1)(a^2+a)=b^2.
      \end{array}$$
        Đặt $a^2+a=n,$ ta có $(2n+1).n=b^2$.
        \begin{itemize}
        \item Với $a=0$ thì $x=1.$ Thay vào (1) được $y=0.$
        \item Với $a\geq 1$ thì $n$ và $2n+1$ là hai số nguyên dương. Vì chúng nguyên tố cùng nhau và tích của chúng là $b^2$ nên mỗi số đều là số chính phương.
        
        Đặt $n=k^2 (k\in\mathbb{N}),$ ta có $a^2+a=k^2.$ Điều này không xảy ra vì 
        $$a^2<a^2+a<a^2+2a+1=(a+1)^2.$$
        
        \end{itemize}
        Đáp số: Nghiệm $(x;y)$ là $(1;0)$ và $(-1;0)$.
\end{enumerate}          
    }
\end{bt}

\begin{bt}
  Chứng minh rằng có vô hạn số nguyên $x$ để biểu thức sau là số chính phương
  $$(1+2+3+\cdots+x)(1^2+2^2+3^2+\cdots+x^2).$$
    \loigiai{
       Ta đặt $(1+2+\cdots+x)(1^2+2^2+\cdots+x^2)=y^2.$ Áp dụng các công thức 
       {\allowdisplaybreaks
       \begin{align*}
       &1+x+\cdots+x=\frac{x(x+1)}{2};\\ &1^2+2^2+\cdots+x^2=\frac{1}{6}x(x+1)(2x+1)
       \end{align*}}ta được $$\frac{x(x+1)}{2}.\frac{x(x+1)(2x+1)}{6}=y^2\Leftrightarrow \frac{x^2(x+1)^2}{4}.\frac{2x+1}{3}=y^2.$$
   Ta phải có $\dfrac{2x+1}{3}=(2n+1)^2$ với $n$ là số nguyên.      
       Phương trình này có vô số nghiệm dạng $x=6n^2+6n+1$ nên có vô hạn số $x$ thỏa mãn điều kiện đề bài.
    }
\end{bt}
