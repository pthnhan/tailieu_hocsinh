	\loigiai{

			 $\lim\limits_{x \rightarrow - \infty} \dfrac{\sqrt{x^2}+3}{2x+7}= \lim\limits_{x \rightarrow - \infty} \dfrac{x\left(-1+\dfrac{3}{x}\right)}{x\left(2+\dfrac{7}{x}\right)}=\lim\limits_{x \rightarrow - \infty} \dfrac{-\sqrt{1}+\dfrac{3}{x}}{2+\dfrac{7}{x}}= \dfrac{-1}{2}.$

\begin{bt}%[HK2 (2016-2017),Chuyên Hà Nội - Amsterdam, Hà Nội]%[1D4K3-5]%[Võ Tấn Đạt]
	Cho hàm số $y=f(x) = \begin{cases}
	\dfrac{	x^2-\sqrt{6-5x}}{x+2}, \; &x \ne -2 \\
	2ax+1, \; &x =-2
	\end{cases}$ ($a$ là tham số).\\
	Tính giá trị của tham số $a$ để hàm số liên tục tại $x=-2$.
	\loigiai{
		Ta có $f(-2)=-4a+1$.\\
		$\begin{aligned}
		\lim\limits_{x \rightarrow -2} f(x)
		&= \lim\limits_{x \rightarrow -2} \dfrac{	x^2-\sqrt{6-5x}}{x+2}\\
		&= \lim\limits_{x \rightarrow -2} \dfrac{	x^4+5x-6}{(x+2)\left(x^2+\sqrt{6-5x}\right)}\\
		&=\lim\limits_{x \rightarrow -2} \dfrac{	(x+2)(x^3-2x^2+4x-3)}{(x+2)\left(x^2+\sqrt{6-5x}\right)}\\
		&=\lim\limits_{x \rightarrow -2} \dfrac{	(x^3-2x^2+4x-3)}{\left(x^2+\sqrt{6-5x}\right)}\\
		&=\dfrac{-27}{8}.
		\end{aligned}$\\
		Để hàm số liên tục tại $x=-2$ thì $\lim\limits_{x \rightarrow -2} f(x)=f(-2) \Leftrightarrow -4a+1= \dfrac{-27}{8} \Leftrightarrow a= \dfrac{35}{32}.$}
\end{bt}

\begin{bt}%[HK2 (2016-2017),Chuyên Hà Nội - Amsterdam, Hà Nội]%[1D4G3-6]%[Võ Tấn Đạt]
	Cho $a,b,c$ là các số thực. Biết $a\neq 0$ và $2a+3b+8c=0$. Chứng minh rằng phương trình $ax^2+bx+c=0$ luôn có nghiệm thuộc khoảng $(0;1)$.
	\loigiai{Xét hàm số $f(x)=ax^2+bx+c$ với $a\neq 0$ và $2a+3b+8c=0$.
		\begin{itemize}
			\item Nếu $c=0$ thì $2a+3b=0\Leftrightarrow b=-\dfrac{2a}{3}$. Khi đó phương trình đã cho trở thành
			$$ax^2-\dfrac{2a}{3} x=0\Leftrightarrow \hoac{&x=0\\ &x=\dfrac{2}{3}}.$$
			Suy ra phương trình có nghiệm $x=\dfrac{2}{3}\in (0;1)$.
			\item Nếu $c\neq 0$. Ta có $f(0)=c$ và $f\left(\dfrac{2}{3}\right)= \dfrac{4}{9} a+\dfrac{2}{3}b+c$. Khi đó
			$$f(0)\cdot f\left(\dfrac{2}{3}\right)=c\left(\dfrac{4}{9} a+\dfrac{2}{3}b+c\right)=\dfrac{\left[2(2a+3b+8c)-7c\right]c}{9}=-\dfrac{7c^2}{9}<0 \text{ (do 2a+3b+8c=0)}.$$
			Hơn nữa do hàm số $f(x)=ax^2+bx+c$ liên tục trên $\mathbb{R}$ nên liên tục trên $\left[0;\dfrac{2}{3}\right]$. Do đó $f(x)=0$ có nghiệm thuộc $\left(0;\dfrac{2}{3}\right)\subset (0;1)$.
		\end{itemize}
		Vậy phương trình đã cho luôn có nghiệm thuộc $(0;1)$.
	}
\end{bt}

\begin{ex}%[HK2 (2016-2017),Chuyên Hà Nội - Amsterdam, Hà Nội]%[1D4K1-3]%[Võ Tấn Đạt]

		\begin{align*}
		\lim \left(\dfrac{n-n^2}{3+2n^2}-\dfrac{1}{\sqrt{n}}\right)=\lim \left[\dfrac{n^2\left(\dfrac{1}{n}-1\right)}{n^2\left(\dfrac{3}{n^2}+2\right)}-\dfrac{1}{\sqrt{n}}\right]=\lim \left(\dfrac{\dfrac{1}{n}-1}{\dfrac{3}{n^2}+2}-\dfrac{1}{\sqrt{n}}\right)=\dfrac{0-1}{0+2}-0=-\dfrac{1}{2}.
