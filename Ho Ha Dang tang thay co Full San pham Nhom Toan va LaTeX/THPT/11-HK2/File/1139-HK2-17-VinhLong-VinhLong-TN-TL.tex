\begin{ex}%[HK2 (2016-2017), THPT Vĩnh Long, Vĩnh Long]%[1D3G2-1]%[Huỳnh Văn Quy]
	Với $n\geq0$\\
	Dãy số \textbf{Fibonacci} được \textbf{Leonardo Fibonacci}, một nhà toán học người Ý, công bố vào năm 1202 trong cuốn sách \lq\lq Liber Abacci\rq\rq - Sách về toán đồ qua hai bài toán: \lq\lq \textit{Bài toán con thỏ}\rq\rq{} và bài toán \lq\lq \textit{số các cụ tổ của một ong đực}\rq\rq. Dãy số này hầu như biến hóa vô tận. Chính điều đó làm cho nhiều nhà Toán học (chuyên nghiệp lẫn nghiệp dư) và cả những người bình thường nghiên cứu, khám phá về nó. Người ta chứng minh được rằng công thức tổng quát cho dãy Fibonacci là:
	\[F_{(n)}=\dfrac{1}{\sqrt{5}}\left[\left(\dfrac{1+\sqrt{5}}{2}\right)^n -\left(\dfrac{1-\sqrt{5}}{2}\right)^n \right]\]
	Số \textbf{Lucas} là một dãy số được đặt tên nhằm vinh danh nhà Toán học \textbf{Francois Edouard Anatole Lucas}, người đã nghiên cứu dãy số Fibonacci, dãy số Lucas và các dãy tương tự. Dãy số gồm thương giữa hai số Lucas liền nhau sẽ hội tụ đến giới hạn bằng \textbf{tỉ lệ vàng} $\left(\varphi=\dfrac{1+\sqrt{5}}{2} \right)$. Công thức tổng quát của số \textbf{Lucas}:
	\[L_n=\left(\dfrac{1+\sqrt{5}}{2}\right)^n+\left(\dfrac{1-\sqrt{5}}{2}\right)^n\]
	Số \textbf{Lucas} liên hệ với số Fibonacci bởi hàng đẳng thức sau:
	\[L_n=F_{n-2}+F_n\]
	Hãy tìm tổng $S_{(n)}=F_{n-2}+F_{n-1}+F_{n}+F_{n+1}$, biết $L_{n}=18$.
	\choice
	{\True $47$}
	{$29$}
	{$13$}
	{$8$}
	\loigiai{
		Ta có: $S_{(n)}=F_{n-2}+F_{n-1}+F_{n}+F_{n+1}=(F_{n-2}+F_n)+(F_{n-1}+F_{n+1})=L_n+L_{n+1}$.\\
	Mà $L_n=18\Leftrightarrow 18=\left(\dfrac{1+\sqrt{5}}{2}\right)^n+\left(\dfrac{1-\sqrt{5}}{2}\right)^n \Leftrightarrow n=6$.\\
	Suy ra $L_{n+1}=\left(\dfrac{1+\sqrt{5}}{2}\right)^{n+1}+\left(\dfrac{1-\sqrt{5}}{2}\right)^{n+1}=\left(\dfrac{1+\sqrt{5}}{2}\right)^{7}+\left(\dfrac{1-\sqrt{5}}{2}\right)^{7}=29$.\\
	Vậy $S_{n}=L_n+L_{n+1}=18+29=47$.}
	
\end{ex}

\begin{bt}%[HK2 (2016-2017), THPT Vĩnh Long, Vĩnh Long]%[1D4B2-3]%[Huỳnh Văn Quy]
	Tính giới hạn sau: $\lim\limits_{x\to -4}\dfrac{x^2+3x-4}{x^2+4x}$.
	\loigiai{Ta có: $\lim\limits_{x\to -4}\dfrac{x^2+3x-4}{x^2+4x}=\lim\limits_{x\to -4}\dfrac{(x-1)(x+4)}{x(x+4)}=\lim\limits_{x\to -4}\dfrac{x-1}{x}=\dfrac{5}{4}$.}
\end{bt}
\begin{bt}

	Tính giới hạn sau: $\lim\limits_{x\to 1}\dfrac{x-\sqrt{2x-1}}{x^2-12x+11}$.
	
	\loigiai{$\lim\limits_{x\to 1}\dfrac{x-\sqrt{2x-1}}{x^2-12x+11}=\lim\limits_{x\to 1}\dfrac{x^2-(2x-1)}{(x-1)(x-11)(x+\sqrt{2x-11})}\lim\limits_{x\to 1}\dfrac{x-1}{(x-11)(x+\sqrt{2x-11})}=0$.}
\end{bt}
\begin{bt}%[HK2 (2016-2017), THPT Vĩnh Long, Vĩnh Long]%[1D5B3-1]%[Huỳnh Văn Quy]
	Tính đạo hàm của hàm số $f(x)=2\sin2x+\cos2x$. Tìm giá trị lớn nhất và giá trị nhỏ nhất của hàm số $g(x)=f'(x)+2\sin2x-3\cos2x$.
\loigiai{Ta có: $f'(x)=(2\sin2x+\cos2x)'=4\cos2x-2\sin2x$.\\
	Do đó: $g(x)=4\sin2x+2\cos2x+2\sin2x-3\cos2x=6\sin2x-\cos2x$.\\
	Áp dụng bất đẳng thức BCS, ta có:\\
	$|g(x)|\leq\sqrt{(6^2+(-1)^2)(\sin^22x+\cos^22x)}\Leftrightarrow|g(x)|\leq\sqrt{37}$.\\
	Vậy $\max g(x)=\sqrt{37}\Leftrightarrow\dfrac{\sin 2x}{\cos2x}=-6\Leftrightarrow\tan2x=-6\Leftrightarrow x=\dfrac{1}{2}\arctan(-6)+\dfrac{k\pi}{2}$.\\
	$\min{g(x)}=-\sqrt{37}$.}
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Vĩnh Long, Vĩnh Long]%[1D5B2-2]%[Huỳnh Văn Quy]
	Viết phương trình tiếp tuyến với đường cong $(C)\colon y=f(x)=x^3\sqrt{2}-2x^2\sqrt{2}+2$ tại điểm có hoành độ là $2$.
	\loigiai{$f'(x)=3\sqrt{2}x^2-4\sqrt{2}x\Rightarrow f'(2)=12\sqrt{2}-8\sqrt{2}=4\sqrt{2}$.\\
		$f(2)=2$.\\
		Vậy phương trình tiếp tuyến cần tìm là: $y=4\sqrt{2}(x-2)+2\Leftrightarrow y=4\sqrt{2}x+2-8\sqrt{2}$.}
