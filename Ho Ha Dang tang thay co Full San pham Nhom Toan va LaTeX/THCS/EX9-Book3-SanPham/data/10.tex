\subsection{Biến đổi đồng nhất biểu thức đại số}

\begin{bt}[KT học kì I lớp 8 Diễn Lâm, Diễn Châu, Nghệ An 2016]
	Cho các số $x$, $y$ thỏa mãn đẳng thức $5x^2+5y^2+8xy-2x+2y+2=0$.\\
	Tính giá trị của biểu thức $M=(x+y)^{2015}+(x-2)^{2016}+(y+1)^{2017}.$
	\loigiai{
	Ta có\\
	$\begin{aligned}
		& 5x^2+5y^2+8xy-2x+2y+2=0\\
		\Leftrightarrow\; & (4x^2+8xy+4y^2)+(x^2-2x+1)+(y^2+2y+1)=0\\
		\Leftrightarrow\; & (2x+2y)^2+(x-1)^2+(y+1)^2=0\\
		\Leftrightarrow\; & x=1,\ y=-1.
	\end{aligned}$\\
	Thay $x=1,y=-1$ vào biểu thức $M$ ta được $M=1$.
	}
\end{bt}
\begin{bt}[KT học kì I lớp 8 Ba Đình, Hà Nội 2017]
	Cho $a$ thỏa mãn $a^2-5a+2=0$. Tính giá trị của biểu thức
	$P=a^5-a^4-18a^3+9a^2-5a+2017+(a^4-40a^2+4):a^2.$
	\loigiai{
	Ta có\\
	$\begin{aligned}
	& a^5-a^4-18a^3+9a^2-5a+2017\\
	=\; & (a^3+4a^2+1)(a^2-5a+2)+2015=2015.
	\end{aligned}$\\
	Vì $a^2-5a+2=0$ nên $a^4=(5a-2)^2$.\\
	$\Rightarrow a^4-40a^2+4=-15a^2-20a+8=-19a^2+4(a^2-5a+2)=-19a^2.$\\
	Vậy $P=2015+\dfrac{-19a^2}{a^2}=2015-19=1996.$
	}
\end{bt}
\begin{bt}[Đề thi học sinh giỏi lớp 8 Quốc Oai, Hà Nội 2017]
	Cho $x=\dfrac{b^2+c^2-a^2}{2bc}$ và $y=\dfrac{a^2-(b-c)^2}{(b+c)^2-a^2}$. Tính giá trị của biểu thức $B=x+y+xy$.
	\loigiai{
	$x+1=\dfrac{(b+c)^2-a^2}{2bc}\Rightarrow y+xy=y(x+1)=\dfrac{a^2-(b-c)^2}{2bc}\Rightarrow B=\dfrac{2bc}{2bc}=1$.
	}
\end{bt}
\begin{bt}[Đề thi học sinh giỏi lớp 8 Việt Yên, Bắc Giang 2017]
	\hfill
	\begin{enumerate}
		\item Phân tích đa thức sau thành nhân tử: $A=21x^4+3x^3+2039x^2+3x+2018$
		\item Tìm giá trị nhỏ nhất của biểu thức $B=(x^2-2x)(y^2+6y+12)+3y^2+18y+2048$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có\\
			$\begin{aligned}
			A&=(21x^4+21x^2+2018x^2+2018)+(3x^3+3x)\\
			&=(x^2+1)(21x^2+2018)+3x(x^2+1)\\
			&=(x^2+1)(21x^2+3x+2018).
			\end{aligned}$
			\item Ta có\\
			$\begin{aligned}
			B&=(x^2-2x)(y^2+6y+12)+3(y^2+6y+12)+2012\\
			&=(x^2-2x+3)(y^2+6y+12)+2012\\
			&\ge 2\cdot 3+2012=2018.

	}
\end{bt}
\begin{bt}[Đề thi học sinh giỏi lớp 8 Việt Yên, Bắc Giang 2017]
	Cho ba số $a,b,c$ khác nhau và khác $0$ thỏa mãn $\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}=0$.\\
	Rút gọn biểu thức $N=\dfrac{1}{a^2+2bc}+\dfrac{1}{b^2+2ca}+\dfrac{1}{c^2+2ab}$.
	\loigiai{
	Từ giả thiết suy ra $ab+bc+ca=0$.\\
	Vậy ta có\\
	$\begin{aligned}
	N & =\dfrac{1}{a^2+2bc}+\dfrac{1}{b^2+2ca}+\dfrac{1}{c^2+2ab}\\
	& =\dfrac{1}{a^2+bc-ab-ca}+\dfrac{1}{b^2+ca-ab-bc}+\dfrac{1}{c^2+ab-bc-ca}\\
	& =\dfrac{1}{(a-b)(a-c)}+\dfrac{1}{(b-c)(b-a)}+\dfrac{1}{(c-a)(c-b)}\\
	& =\dfrac{c-b+a-c+b-a}{(a-b)(b-c)(c-a)}\\
	& =0.
	\end{aligned}$
	}
\end{bt}
\begin{bt}[Đề thi học sinh giỏi lớp 9 Bắc Giang 2017]
	Cho biểu thức $$B=\dfrac{1}{16}+\dfrac{2}{16^2}+\dfrac{3}{16^3}+\cdots+\dfrac{2018}{16^{2018}}.$$
	Hãy so sánh hai số $B^{2017}$ và $B^{2018}$.
	\loigiai{
	Ta có $16\cdot B=1+\dfrac{2}{16}+\dfrac{3}{16^2}+\cdots+\dfrac{2018}{16^{2017}}.$\\
	Suy ra $15\cdot B=1+\dfrac{1}{16}+\dfrac{1}{16^2}+\cdots+\dfrac{1}{16^{2017}}-\dfrac{2018}{18^{2018}}.$\\
	Đặt $C=	\dfrac{1}{16}+\dfrac{1}{16^2}+\cdots+\dfrac{1}{16^{2017}}.$\\
	$\Rightarrow 16\cdot C=1+\dfrac{1}{16}+\cdots+\dfrac{1}{16^{2016}}$\\
	$\Rightarrow 15\cdot C=1-\dfrac{1}{16^{2017}}<1$.\\
	$\Rightarrow C<1$\\
	$\Rightarrow 15\cdot B<2 \Rightarrow B<1$.\\
	Vì $0<B<1$ nên $B^{2017}-B^{2018}=B^{2017}(1-B)>0$.\\
	Do đó $B^{2017}>B^{2018}$.
	}
\end{bt}
\begin{bt}[Đề thi học sinh giỏi lớp 9 Bắc Ninh 2017]
	Cho $x$ là số thực dương thỏa mãn điều kiện $x^2+\dfrac{1}{x^2}=7$. Tính giá trị các biểu thức $A=x^3+\dfrac{1}{x^3}$; $B=x^7+\dfrac{1}{x^7}$.
	\loigiai{
	Ta có $x^2+\dfrac{1}{x^2}=7\Leftrightarrow \left( x+\dfrac{1}{x}\right) ^2-2=7\Leftrightarrow \left( x+\dfrac{1}{x}\right) ^2=9.$\\
	Do $x$ là số thực dương nên $x+\dfrac{1}{x}=3$.\\
	Vậy $A=	x^3+\dfrac{1}{x^3}=\left( x+\dfrac{1}{x}\right) ^3-3\left( x+\dfrac{1}{x}\right) =3^3-3\cdot 3=18.$\\
	Ta có $x^4+\dfrac{1}{x^4}=\left( x^2+\dfrac{1}{x^2}\right) ^2-2=7^2-2=47$.\\
	Do đó $B=x^7+\dfrac{1}{x^7}=\left( x^3+\dfrac{1}{x^3}\right) \left( x^4+\dfrac{1}{x^4}\right) -\left( x+\dfrac{1}{x}\right) =18\cdot 47-3=843.$
	}
\end{bt}
\begin{bt}[Đề thi học sinh giỏi lớp 9 Hưng Yên 2017]
	Cho hai số thực dương $a,b$ thỏa mãn $\dfrac{1}{a}+\dfrac{1}{b}=\dfrac{1}{2018}$. Chứng minh rằng $\sqrt{a+b}=\sqrt{a-2018}+\sqrt{b-2018}$.
	\loigiai{
	Vì $a,b$ là các số thực dương và $\dfrac{1}{a}+\dfrac{1}{b}=\dfrac{1}{2018}$ nên $2018=\dfrac{ab}{a+b}.$\\
	Vậy ta có\\
	$\begin{aligned}
	& \sqrt{a-2018}+\sqrt{b-2018}=\sqrt{a-\dfrac{ab}{a+b}}+\sqrt{b-\dfrac{ab}{a+b}}\\
	=\; & \sqrt{\dfrac{a^2}{a+b}}+\sqrt{\dfrac{b^2}{a+b}}=\dfrac{a}{\sqrt{a+b}}+\dfrac{b}{\sqrt{a+b}}\\
	=\; & \dfrac{a+b}{\sqrt{a+b}}=\sqrt{a+b}.
	\end{aligned}$
	}
\end{bt}
\begin{bt}[Đề thi học sinh giỏi lớp 9 Hưng Yên 2017]
	Cho $a$ là nghiệm dương của phương trình $6x^2+\sqrt{3}x-\sqrt{3}=0$. Tính giá trị biểu thức $A=\dfrac{a+2}{\sqrt{a^4+a+2}-a^2}$.
	\loigiai{
	Ta có $a$ là nghiệm dương của phương trình $6x^2+\sqrt{3}x-\sqrt{3}=0$ nên $6a^2+\sqrt{3}a-\sqrt{3}=0$.\\
	$\Rightarrow a=\dfrac{\sqrt{3}-6a^2}{\sqrt{3}}=1-2\sqrt{3}a^2>0\Rightarrow a<\dfrac{1}{2\sqrt{3}}\Leftrightarrow a^2<\dfrac{1}{12}<\sqrt{3}\Leftrightarrow a^2-\sqrt{3}<0$.\\
	Do đó\\
	$\begin{aligned}
	A & =\dfrac{a+2}{\sqrt{a^4+a+2}-a^2}=\sqrt{a^4+a+2}+a^2=\sqrt{a^4+1-2\sqrt{3}a^2+2}+a^2\\
	& =\sqrt{\left( a^2-\sqrt{3}\right) ^2} +a^2=\left| a^2-\sqrt{3}\right| +a^2=\sqrt{3}-a^2+a^2=\sqrt{3}.
	\end{aligned}$
	}
\end{bt}
\begin{bt}[Đề thi vào lớp 10 Quảng Ninh 2013]
	Cho các số thực dương $x,y,z$ thỏa mãn $\left( \sqrt{x}-\sqrt{y}\right) ^3+\left( \sqrt{y}-\sqrt{z}\right) ^3+\left( \sqrt{z}-\sqrt{x}\right) ^3=0.$\\
	Tính giá trị của biểu thức $T=\left( \sqrt{x}-\sqrt{y}\right) ^{2013}+\left( \sqrt{y}-\sqrt{z}\right) ^{2013}+\left( \sqrt{z}-\sqrt{x}\right) ^{2013}$.
	\loigiai{
	Ta có\\
	$\begin{aligned}
	& \left( \sqrt{x}-\sqrt{y}\right) ^3+\left( \sqrt{y}-\sqrt{z}\right) ^3+\left( \sqrt{z}-\sqrt{x}\right) ^3=0\\
	\Leftrightarrow & \sqrt{x}\sqrt{y}(\sqrt{x}-\sqrt{y})+\sqrt{y}\sqrt{z}(\sqrt{y}-\sqrt{z})+\sqrt{z}\sqrt{x}(\sqrt{z}-\sqrt{x})=0\\
	\Leftrightarrow & (\sqrt{x}-\sqrt{y})(\sqrt{y}-\sqrt{z})(\sqrt{x}-\sqrt{z})=0\\
	\Leftrightarrow & \left[\begin{aligned}
		\sqrt{x}=\sqrt{y}\\ \sqrt{y}=\sqrt{z}\\ \sqrt{z}=\sqrt{x}.
	\end{aligned}\right.
	\end{aligned}$\\
	Nếu $\sqrt{x}=\sqrt{y}$ thì $T=\left( \sqrt{x}-\sqrt{z}\right) ^{2013}+\left( \sqrt{z}-\sqrt{x}\right) ^{2013}=0.$\\
	Tương tự với các trường hợp $\sqrt{y}=\sqrt{z}$ và $\sqrt{z}=\sqrt{x}$, ta có $T=0$.
	}
\end{bt}
\begin{bt}[Đề thi vào lớp 10 Quảng Ninh 2017]
	Tính giá trị của biểu thức $M=a^2+b^2$ biết $a$ và $b$ thỏa mãn $\begin{cases}
	\dfrac{3a^2}{b^2}+\dfrac{1}{b^3}=1\\[0.4 cm] \dfrac{3b^2}{a^2}+\dfrac{2}{a^3}=1.
	\end{cases}$
	\loigiai{
	Ta có
	$\begin{cases}
	\dfrac{3a^2}{b^2}+\dfrac{1}{b^3}=1\\[0.4 cm] \dfrac{3b^2}{a^2}+\dfrac{2}{a^3}=1.
	\end{cases}\Rightarrow \begin{cases}
	3a^2+1=b^3\\ 3ab^2+2=a^3
	\end{cases}\Rightarrow \begin{cases}
	(b^3-3a^2b)^2=1\\ (a^3-3ab^2)^2=4.
	\end{cases}$\\
	$\begin{aligned}
		&\Rightarrow b^6-6a^2b^4+9a^4b^2+a^6-6a^4b^2+9a^2b^4=5\\
		&\Leftrightarrow a^6+3a^4b^2+3a^2b^4+b^6=5\\
		&\Leftrightarrow (a^2+b^2)^3=5\\
		&\Leftrightarrow a^2+b^2=\sqrt[3]{5}.
	\end{aligned}$
	}
\end{bt}
\begin{bt}[Thi vào lớp 10 chuyên Hạ Long 2015]
	Cho $(x_0;y_0;z_0)$ là nghiệm của hệ phương trình $\begin{cases}
	x^2=y+z\\ y^2=x+z\\ z^2=2xy
	\end{cases}$ với $xyz\ne 0$.\\
	Tính giá trị biểu thức $T=x_0+y_0+1$.
	\loigiai{
	Trừ vế với vế của hai phương trình đầu ta được $$(x-y)(x+y+1)=0.$$
	\begin{itemize}
		\item Trường hợp 1: $x+y+1=0$ hay $y=-x-1$.\\
		Từ phương trình đầu suy ra $z=x^2+x+1$.\\
		Thế vào phương trình cuối ta được $(x^2+x+1)^2=-2x^2-2x\quad(1)$.\\
		Mà $(x^2+x+1)^2\ge \dfrac{9}{16}>\dfrac{1}{2}$ và $-2x^2-2x\le \dfrac{1}{2}$ nên phương trình $(1)$ vô nghiệm.
		\item Trường hợp 2: $x=y$.\\
		Hệ trở thành $\begin{cases}
		x^2=x+z\\ z^2=2x^2.
		\end{cases}$\\
		Nếu $z=\sqrt{2}x$ thì $x^2=(1+\sqrt{2}x)\Leftrightarrow x=1+\sqrt{2}$ (do $x\ne 0$).\\
		Nếu $z=-\sqrt{2}x$ thì $x^2=(1-\sqrt{2}x)\Leftrightarrow x=1-\sqrt{2}$ (do $x\ne 0$).\\
		Vậy nếu $(x_0;y_0;z_0)$ là nghiệm của hệ phương trình đã cho thì $T=x_0+y_0+1=3\pm 2\sqrt{2}.$
	\end{itemize}	
	}
\end{bt}
\begin{bt}[Đề thi Chuyên ĐHSPHN năm 2011, Vòng 1]
	Chứng minh bất đẳng thức
	\[
	\dfrac{1}{\sqrt{1}+\sqrt{2}}+\dfrac{1}{\sqrt{3}+\sqrt{4}}+\dfrac{1}{\sqrt{5}+\sqrt{6}}+...+\dfrac{1}{\sqrt{79}+\sqrt{80}}>4
	\]
	\loigiai{
		Đặt $A=\dfrac{1}{\sqrt{1}+\sqrt{2}}+\dfrac{1}{\sqrt{3}+\sqrt{4}}+\dfrac{1}{\sqrt{5}+\sqrt{6}}+...+\dfrac{1}{\sqrt{79}+\sqrt{80}}$.\\
		$\begin{aligned}
		\Rightarrow 2A&=\dfrac{2}{\sqrt{1}+\sqrt{2}}+\dfrac{2}{\sqrt{3}+\sqrt{4}}+\dfrac{2}{\sqrt{5}+\sqrt{6}}+...+\dfrac{2}{\sqrt{79}+\sqrt{80}}\\
		\Rightarrow 2A&>\dfrac{1}{\sqrt{1}+\sqrt{2}}+\dfrac{1}{\sqrt{2}+\sqrt{3}}+\dfrac{1}{\sqrt{3}+\sqrt{4}}+...+\dfrac{1}{\sqrt{79}+\sqrt{80}}+\dfrac{1}{\sqrt{80}+\sqrt{81}}\\ 
		&>\dfrac{\sqrt{2}-\sqrt{1}}{(\sqrt{2}+\sqrt{1})(\sqrt{2}-\sqrt{1})}+\dfrac{\sqrt{3}-\sqrt{2}}{(\sqrt{3}+\sqrt{2})(\sqrt{3}-\sqrt{2})}+...+\dfrac{\sqrt{81}-\sqrt{80}}{(\sqrt{81}+\sqrt{80})(\sqrt{81}-\sqrt{80})}\\ 
		&=\sqrt{2}-\sqrt{1}+\sqrt{3}-\sqrt{2}+...+\sqrt{81}-\sqrt{80}=9-1=8.\\
		\Rightarrow A&>4.
		\end{aligned}$
	}
\end{bt}

\begin{bt}[Đề thi vào Chuyên KHTN Hà Nội năm 2015, vòng 2]
	Với $a, b, c$ là các số thực thỏa mãn:
	$(3a+3b+3c)^3=24+(3a+b-c)^3+(3b+c-a)^3+(3c+a-b)^3$.
	Chứng minh rằng: $(a+2b)(b+2c)(c+2a)=1$.

	3a+b-c=x\\ 3b+c-a=y\\ 3c+a-b=z.
	\end{cases}$\\

	$\begin{aligned}
	&(3a+3b+3c)^3=24+(3a+b-c)^3+(3b+c-a)^3+(3c+a-b)^3\\
	\Rightarrow &(x+y+z)^3=24+x^3+y^3+z^3\\

	\Rightarrow &24-3(2a+4b)(2b+4c)(2c+4a)=0\\

\subsection{Phương trình bậc nhất. Phương trình bậc hai}

\begin{bt}[Đề thi vào 10, Chuyên Kiên Giang, 2016,V2]%[9D4G6]%[9D5G1]
   	Giải phương trình
   	$$\dfrac{1}{x^2+x-2} + \dfrac{1}{x^2+7x+10} + \dfrac{1}{x^2+x-20}+ \dfrac{1}{x^2-5x+4} = \dfrac{-2}{7}.$$
   	\loigiai{Điều kiện của phương trình $\heva{&x^2+x-2\neq 0\\&x^2+x-20\neq 0\\&x^2-5x+4\neq 0\\&x^2+7x+10\neq 0}\Leftrightarrow \heva{&x\neq -5\\&x\neq -2\\&x\neq 1\\&x\neq 4}$.\\
   		Khi đó phương trình đã cho tương đương với phương trình\\
   		\begin{align*}
   		&\dfrac{1}{(x-1)(x+2)}+\dfrac{1}{(x+5)(x+2)}+\dfrac{1}{(x+5)(x-4)}+\dfrac{1}{(x-4)(x-1)}=-\dfrac{2}{7}\\
   		\Leftrightarrow &\dfrac{(x-4)(x+5)+(x-4)(x-1)+(x-1)(x+2)+(x+2)(x+5)}{(x-1)(x+2)(x-4)(x+5)}=-\dfrac{2}{7}\\
   		\Leftrightarrow &\dfrac{4}{(x-4)(x+5)}=-\dfrac{2}{7}\Leftrightarrow \dfrac{x^2+x-6}{(x-4)(x+5)}=0\\
   		\Rightarrow &x^2+x-6=0\Leftrightarrow \hoac{&x=-3\\&x=2} \mbox{(thỏa điều kiện).}
   		\end{align*}
   		Vậy phương trình đã cho có nghiệm $x=2$ và $x=-3$.
   	}
   \end{bt}
   \begin{bt}[Đề thi vào 10, Trường Đại học Sư phạm Hà nội, 2014]%[9D4K4]%[9D4K6]
   	Trong mặt phẳng tọa độ $Oxy$ cho parabol $(P):y=x^2$ và đường thẳng $(d):y=-\dfrac{2}{3}(m+1)x+\dfrac{1}{3}$ (với $m$ là tham số)
   	\begin{enumerate}
   		\item Chứng minh rằng với mỗi giá trị của $m$ đường thẳng $d$ cắt $(P)$ tại $2$ điểm phân biệt.
   		\item Gọi $x_1,x_2$ là hoành độ giao điểm của $d$ và $(P)$, đặt $f(x)=x^3+(m+1)x^2-x$. Chứng minh đẳng thức $f(x_1)-f(x_2)=-\dfrac{1}{2}(x_1-x_2)^3$.
   	\end{enumerate}
   	\loigiai
   	{
   		\begin{enumerate}
   			\item Xét phương trình hoành độ giao điểm của $(P)$ và $(d)$:\\
   			$x^2=-\dfrac{2}{3}(m+1)x+\dfrac{1}{3}\Leftrightarrow x^2+\dfrac{2}{3}(m+1)x-\dfrac{1}{3}=0\Leftrightarrow 3x^2 + 2(m+1)x-1=0$ $(1)$\\
   			Ta có $\Delta ' =(m+1)^2-3(-1)=(m+1)^2+3>0$ với mọi $m$\\
   			$\Rightarrow$ phương trình  $(1)$ luôn có $2$ nghiệm phân biệt với mọi $m$\\
   			Do đó đường thẳng $(d)$ và $(P)$ luôn cắt nhau tại $2$ điểm phân biệt với mọi $m$.
   			\item Ta có 		
   			\begin{align*}
   			f(x_1)-f(x_2) = &x_1^3+(m+1)x_1^2-x_1-x_2^3-(m+1)x_2^2+x_2 \\
   			=&(x_1^3-x_2^3)-(x_1-x_2)+(m+1)(x_1^2-x_2^2)\\
   			=& (x_1-x_2)(x_1^2+x_1x_2+x_2^2-1+(m+1)(x_1+x_2))\\
   			\end{align*}
   			Vì $x_1;x_2$ là nghiệm của phương trình $(1)$ nên theo định lý Vi-ét, ta có\\
   			$\heva {&x_1+x_2 = \dfrac{-2(m+1)}{3}\\
   				&x_1x_2 = \dfrac{-1}{3}}\Leftrightarrow \heva{&m+1=-\dfrac{3}{2}(x_1+x_2)\\&x_1x_2=-\dfrac{1}{3}}$\\
   			Từ đó suy ra \begin{align*}
   			f(x_1)-f(x_2)&=(x_1-x_2)\left[x_1^2-\dfrac{1}{3}+x_2^2-1-\dfrac{3}{2}(x_1+x_2)(x_1+x_2)\right]\\

   	\loigiai{
   		$\heva{&(m-1)x-my=3m-1\\&2x-y=m+5}\Leftrightarrow \heva{&(m-1)x-my=3m-1\\&y=2x-m-5}\Leftrightarrow \heva{&(m+1)x=(m+1)^3\ (*)\\&y=2x-m-5}$.\\
   		Hệ phương trình có nghiệm duy nhất $\Leftrightarrow $ phương trình $(*)$ có nghiệm duy nhất $\Leftrightarrow m\ne -1$.\\
   		Với $m\ne -1$, hệ phương trình đã cho có nghiệm duy nhất $(x;y)=(m+1;m-3)$.\\
   		Ta có: $x^3-y^3=64\Leftrightarrow (m+1)^3-(m-3)^3=64\Leftrightarrow 12m^2-24m-36=0\Leftrightarrow \hoac{&m=-1\\&m=3}$.\\
   		Kết hợp với điều kiện ta được $m=3$ thỏa yêu cầu bài toán.
   	}
   \end{bt}

\subsection{Phương trình chứa căn bậc hai}

\begin{bt}%[Đề tuyển sinh vào lớp 10 năm 2016-2017, Chuyên Lê Hồng Phong, TP.HCM]%[Cao Thành Thái]
	Giải phương trình $2(x + 2)\sqrt{3x - 1} = 3x^2 - 7x - 3$
	\loigiai
	{
		\noindent{\textbf{Cách 1}}\\
		Điều kiện xác định $x \geq \dfrac{1}{3}$.\\
		Từ phương trình đã cho ta suy ra
		\begin{align*}
		& 4(x + 2)^2 (3x - 1) = (3x^2 - 7x - 3)^2 \\
		\Leftrightarrow & 9x^4 - 54x^3 - 13x^2 + 10x + 25 = 0 \\
		\Leftrightarrow & (x^2 - 7x + 5)(9x^2 + 9x + 5) = 0 \\
		\Leftrightarrow & \left[\begin{aligned} & x^2 - 7x + 5 = 0 \\ & 9x^2 + 9x + 5 = 0 \, \text{(vô nghiệm)}\end{aligned}\right.\\
		\Leftrightarrow & x = \dfrac{7 \pm \sqrt{29}}{2}.
		\end{align*}
		Đối chiếu với điều kiện xác định và thử lại ta nhận $x = \dfrac{7 + \sqrt{29}}{2}$.\\
		Vậy tập nghiệm của phương trình đã cho là $S = \left\{ \dfrac{7 + \sqrt{29}}{2} \right\}$.\\
		\noindent{\textbf{Cách 2}}\\
		Điều kiện xác định $x \geq \dfrac{1}{3}$.
		\begin{align*}
		& 2(x + 2)\sqrt{3x - 1} = 3x^2 - 7x - 3 \\
		\Leftrightarrow & 2(x + 2)\sqrt{3x - 1} = 3x^2 - 4x - (3x - 1) - 4.
		\end{align*}
		Đặt $t = \sqrt{3x - 1}$ ($t \geq 0$). Phương trình đã cho trở thành
		\begin{align*}
		& 3x^2 - 4x - t^2 - 4 - 2(x + 2)t = 0 \\
		\Leftrightarrow & 3x^2 - 2(t + 2)x - t^2 - 4t - 4 = 0. \tag{*}
		\end{align*}
		Ta xem $(*)$ là phương trình bậc hai theo biến $x$, tham số $t$. Ta có
		$$\Delta' = (t + 2)^2 - 3 (-t^2 - 4t - 4) = 4t^2 + 16t + 16 = (2t + 4)^2.$$
		Khi đó 
		\begin{align*}
		\left[\begin{aligned}
		& x = \dfrac{t + 2 - (2t + 4)}{3} = \dfrac{-t - 2}{3} \leq -\dfrac{2}{3} \, (\text{mâu thuẫn với ĐKXĐ do } t \geq 0) \\
		& x = \dfrac{t + 2 + (2t + 4)}{3} = t + 2.
		\end{aligned}
		\right.
		\end{align*}
		Ta giải phương trình
		\begin{align*}
		& x = t + 2 \\
		\Leftrightarrow & \sqrt{3x - 1} = x - 2 \\
		\Leftrightarrow & \begin{cases}  x \geq 2 \\ 3x - 1 = x^2 - 4x + 4 \end{cases} \\
		\Leftrightarrow & x = \dfrac{7 + \sqrt{29}}{2}.
		\end{align*}
		Thử lại ta được tập nghiệm của phương trình là $S = \left\{ \dfrac{7 + \sqrt{29}}{2} \right\}$.
	}
\end{bt}

	Giải phương trình $2(x + 1) \sqrt{x + 1} = \left( \sqrt{x + 1} + \sqrt{1 - x} \right)\left( 2 - \sqrt{1 - x^2} \right)$.
	\loigiai
	{

		Đặt $a = \sqrt{x + 1}$ và $b = \sqrt{1 - x}$ với điều kiện $a \geq 0$ và $b \geq 0$. Khi đó $a^2 + b^2 = 2$.\\

\begin{bt}[Đề thi tuyển sinh Chuyên Hùng Vương Phú Thọ 2017]%[9D5G1]

		& \ \ \ =1+\dfrac{1}{2}\left[ 9xyz-4\left( xy+yz+zx \right) \right]

		\noindent Ta có:
		$a>0\Rightarrow g(\alpha )\le g(x)\le g(\beta );a=0\Rightarrow g(x)=g(\alpha );a<0\Rightarrow g(\beta )\le g(x)\le g(\alpha ).$
		
		\noindent Vậy $\min \left\{ g(\alpha );g(\beta ) \right\}\le g(x)\le \max \left\{ g(\alpha );g(\beta ) \right\}$.
		
		\noindent Không mất tính tổng quát có thể giả sử $x\ge y\ge z\Rightarrow \dfrac{1}{3}\le x\le 1$.
		
		\noindent Ta có $9xyz-4\left( xy+yz+zx \right)=\left( 9yz-4y-4z \right)x-4yz$.
		
		\noindent Đặt $f(x)=\left( 9yz-4y-4z \right)x-4yz\ \ \left( \dfrac{1}{3}\le x\le 1 \right)$.

		\noindent Ta có $P=1$ khi $x=1;y=z=0$ hoặc $y=1;x=z=0$ hoặc $z=1;y=x=0$. 
		Vậy $\max P=1$.
		
	}
\end{bt}

\begin{bt}[Đề thi HSG Phú Thọ 2017]%[9D5G1]

\subsection{Bất đẳng thức và cực trị dạng phân thức}

\begin{bt}%[Đề thi vào 10, Sở GD-ĐT Bắc Ninh, 2017]%[9D5G1]
	Cho bốn số thực dương $x,y,z,t$ thỏa mãn $x+y+z+t=2$. Tìm giá trị nhỏ nhất của biểu thức $A=\dfrac{(x+y+z)(x+y)}{xyzt}.$
	\loigiai{
		Áp dụng bất đẳng thức AM-GM cho các số thực dương ta có
		\begin{align*}
		& x+y \geq 2\sqrt{xy}\\
		& (x+y)+z \geq 2\sqrt{(x+y)z}\\
		& (x+y+z)+t \geq 2\sqrt{(x+y+z)t}.
		\end{align*}
		Nhân vế với vế của bất đẳng thức trên và kết hợp điều kiện $x+y+z+t=2$ ta được
		\begin{align*}
		& 2(x+y)(x+y+z)\geq 8\sqrt{xyzt(x+y)(x+y+z)}\\
		\Rightarrow & \sqrt{(x+y)(x+y+z} \geq 4\sqrt{xyzt}\\
		\Rightarrow & \dfrac{(x+y)(x+y+z)}{xyzt}  \geq 16.
		\end{align*}
		Đẳng thức xảy ra khi và chỉ khi $\heva{&x=y\\&x+y=z\\&x+y+z=t\\&x+y+z+t=2} \Leftrightarrow \heva{&x=y=\dfrac{1}{4}\\&z=\dfrac{1}{2}\\&t=1.}$
	}
\end{bt}

\begin{bt}%[Đề thi vào 10, Sở giáo dục Đắklak, 2017]]%[9D1G8]
	Cho hai số thực dương $x,y$ thỏa mãn $xy = 1$. Tìm giá trị nhỏ nhất của biểu thức:
	\begin{center}
		$M=x^2+y^2+\dfrac{3}{x+y+1}$.
	\end{center}
	\loigiai
	{Ta có $(x+y)^2\geq 4xy=4\Rightarrow x+y\geq 2 $ (vì $x+y>0$).\\
		Đặt $t=x+y\geq 2$. Ta có\\
		$\begin{array}{ll}
		M& = x^2+y^2+\dfrac{3}{x+y+1}=(x+y)^2+\dfrac{3}{x+y+1}-2xy \\ 
		& =t^2+\dfrac{3}{t+1}-2=\dfrac{t^3+t^2-2t+1}{t+1}.
		\end{array}$\\
		$\begin{array}{ll}
		\Rightarrow M-3 &=\dfrac{t^3+t^2-2t+1}{t+1}-3=\dfrac{t^3+t^2-5t-2}{t+1} \\
		&=\dfrac{(t-2)(t^2+3t+1)}{t+1}\geq 0  \text{ (vì }t\geq 2\text{)}.\\
		\Rightarrow m \ge 3.
		\end{array}$\\
		Đẳng thức xảy ra khi $\heva{&x+y=2\\&xy=1}\Leftrightarrow x=y=1 $. Vậy $\min M=3\Leftrightarrow x=y=1.$    
	}
\end{bt}

\begin{bt}%[Đề thi vào 10, Sở GD-ĐT Hải Dương, 2017]%[9D5G1]
	Cho $x,y,z$ là ba số thực dương thỏa mãn $x+y+z=3$. Tìm giá trị nhỏ nhất của biểu thức 
	$Q=\dfrac{x+1}{1+y^2}+\dfrac{y+1}{1+z^2}+\dfrac{z+1}{1+x^2}.$
	\loigiai
	{
		Ta có \\ 
		$ \begin{array}{ll}
		Q & =\dfrac{x+1}{1+y^2}+\dfrac{y+1}{1+z^2}+\dfrac{z+1}{1+x^2}\\ 
		& =\left(\dfrac{x}{1+y^2}+\dfrac{y}{1+z^2}+\dfrac{z}{1+x^2}\right)+\left(\dfrac{1}{1+y^2}+\dfrac{1}{1+z^2}+\dfrac{1}{1+x^2}\right)\\
		&=M+N.\\
		\end{array}$\\
		Xét biểu thức $M=\dfrac{x}{1+y^2}+\dfrac{y}{1+z^2}+\dfrac{z}{1+x^2}$. \\
		Áp dụng bất đẳng thức Cô-si ta có \\
		\begin{center}
			$\dfrac{x}{1+y^2}=\dfrac{x(1+y^2)-xy^2}{1+y^2}=x-\dfrac{xy^2}{1+y^2}\geq x-\dfrac{xy^2}{2y}=x-\dfrac{xy}{2}.$
		\end{center}
		Tương tự  $\dfrac{y}{1+z^2}\geq y-\dfrac{yz}{2}$;  $\dfrac{z}{1+x^2}\geq z-\dfrac{zx}{2}$.\\ 
		Nên suy ra \\
		$M=\dfrac{x}{1+y^2}+\dfrac{y}{1+z^2}+\dfrac{z}{1+x^2}\geq x+y+z-\dfrac{xy+yz+zx}{2}=3-\dfrac{xy+yz+zx}{2}$.\\
		Lại có\\
		$x^2+y^2+z^2\geq xy+yz+zx\Rightarrow (x+y+z)^2\geq 3(xy+yz+zx)\Rightarrow xy+yz+zx\leq 3$.\\
		Suy ra $M\geq 3-\dfrac{xy+yz+zx}{2}\geq 3-\dfrac{3}{2}=\dfrac{3}{2}$.\\
		Dấu "=" xãy ra $\Leftrightarrow x=y=z=1$.\\
		Xét $N=\dfrac{1}{1+y^2}+\dfrac{1}{1+z^2}+\dfrac{1}{1+x^2}$, ta có
		\begin{align*}
		3-N&=\left(1-\dfrac{1}{1+y^2}\right)+\left(1-\dfrac{1}{1+z^2}\right)+\left(1-\dfrac{1}{1+x^2}\right)\\
		&=\dfrac{y^2}{1+y^2}+\dfrac{z^2}{1+z^2}+\dfrac{x^2}{1+x^2}\leq \dfrac{y^2}{2y}+\dfrac{z^2}{2z}+\dfrac{x^2}{2x}=\dfrac{x+y+z}{2}=\dfrac{3}{2}.
		\end{align*}
		Suy ra $N\geq 3-\dfrac{3}{2}=\dfrac{3}{2}$.\\
		Dấu "=" xãy ra $\Leftrightarrow x=y=z=1$.\\
		Từ đó suy ra  $Q\geq 3$. Dấu "=" xãy ra khi $x=y=z=1$.\\
		Vậy giá trị nhỏ nhất của $Q$ là  $Q_{\min}=3$ đạt được khi $x=y=z=1$.
	}
\end{bt}

\begin{bt}%[Đề thi vào 10, Sở giáo dục Hải Phòng, 2017]%[9D5G1]
	\hfill
	\begin{enumerate}
		\item[a)] Cho hai số $x>0$, $y>0$. Chứng minh rằng
		$$\dfrac{1}{x+y} \le \dfrac{1}{4} \left( \dfrac{1}{x}+ \dfrac{1}{y} \right).$$
		\item[b)] Cho số dương $a$, $b$, $c$ thỏa mãn $\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}=16$. Chứng minh rằng
		$$\dfrac{1}{3a+2b+c}+\dfrac{1}{a+3b+2c}+\dfrac{1}{2a+b+3c} \le \dfrac{8}{3}.$$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item[a)] Vì $x>0$, $y>0$ nên áp dụng Bất đẳng thức Cô-si ta có:
			$$xy \le \left(\dfrac{x+y}{2} \right)^2 = \dfrac{(x+y)^2}{4}.$$
			Suy ra
			$$\dfrac{1}{x+y} \le \dfrac{1}{4}\cdot \dfrac{x+y}{xy} = \dfrac{1}{4}\cdot \left(\dfrac{1}{x}+\dfrac{1}{y} \right).$$
			Dấu $``="$ xảy ra khi $x=y$.
			\item[b)] Với ba số $a$, $b$, $c>0$, ta chứng minh bất đẳng thức
			$$\dfrac{1}{a+b+c} \le \dfrac{1}{9} \left(\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c} \right). \quad (\ast)$$
			Thật vậy, 
			\begin{align*}
			(\ast) 
			&\Leftrightarrow \dfrac{1}{a+b+c} \le \dfrac{1}{9} \dfrac{ab+bc+ca}{abc}\\
			&\Leftrightarrow 9abc \le (a+b+c)(ab+bc+ca)\\
			&\Leftrightarrow 18abc \le (a+b+c)(2ab+2bc+2ca). \quad (\ast \ast)
			\end{align*}
			Ta có: $(a+b+c)^2=a^2+b^2+c^2+2ab+2bc+2ca$.\\
			Mà $a^2+b^2+c^2 \ge 0$ nên $2ab+2bc+2ca \le (a+b+c)^2$.
			\par Khi đó $(\ast \ast)$ trở thành $18abc \le (a+b+c)^3$.
			\par Theo Bất đẳng thức Cô-si ta có: $(a+b+c)^3 \ge 27abc \ge 18abc$.
			\par Vậy $\dfrac{1}{a+b+c} \le \dfrac{1}{9} \left( \dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c} \right)$. Dấu $``="$ xảy ra khi $a=b=c$.
			Áp dụng bất đẳng thức $(\ast)$ và bất đẳng thức $\dfrac{1}{x+y} \le \dfrac{1}{4} \left( \dfrac{1}{x}+\dfrac{1}{y} \right)$ ta có:
			\begin{align*}
			\rm VT &=\dfrac{1}{3a+2b+c}+\dfrac{1}{a+3b+2c}+\dfrac{1}{2a+b+3c}\\
			&=\dfrac{1}{a+b+a+b+a+c} + \dfrac{1}{a+b+b+c+b+c} + \dfrac{1}{a+c+b+c+a+c}\\
			&\le \dfrac{1}{9} \left( \dfrac{1}{a+b} +\dfrac{1}{a+b}+\dfrac{1}{a+c}+\dfrac{1}{a+b}+\dfrac{1}{b+c}+\dfrac{1}{b+c}+\dfrac{1}{a+c}+\dfrac{1}{b+c}+\dfrac{1}{a+c} \right)\\
			&\le \dfrac{1}{9}.\dfrac{1}{4} \left( \dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{a}+\dfrac{1}{c}+\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{b}+\dfrac{1}{c}+\dfrac{1}{b}+\dfrac{1}{c}+\dfrac{1}{a}+\dfrac{1}{c}+\dfrac{1}{b}+\dfrac{1}{c}+\dfrac{1}{a}+\dfrac{1}{c} \right)\\
			&=\dfrac{1}{36}.6 \left(\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c} \right) = \dfrac{1}{6}\cdot16=\dfrac{8}{3}.
			\end{align*}
			Dấu $``="$ xảy ra khi $\heva{&a+b=a+c=b+c\\&a=b=c\\&\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}=16} \Leftrightarrow a=b=c=\dfrac{3}{16}$.
			\par Vậy bất đẳng thức được chứng minh.
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Đề thi vào 10, Sở giáo dục Hà Nam, 2017]%[9D5G1]
	Cho $a$, $b$, $c$ là các số thực không âm thỏa mãn $ab+bc+ca=3$ và $a\geq c$. Tìm giá trị nhỏ nhất của biểu thức $P=\dfrac{1}{(a+1)^{2}}+\dfrac{2}{(b+1)^{2}}+\dfrac{3}{(c+1)^{2}}$.
	\loigiai{
		
		{\bf Cách 1}: Sử dụng $x^{2}+y^{2}+z^{2}\geq xy+yz+zx$ và $(x+y+z)^{2}\geq 3(xy+yz+zx)$.\\
		Đặt $x=\dfrac{1}{a+1}$, $y=\dfrac{1}{b+1}$, $z=\dfrac{1}{c+1}$ ($x,y,z>0$). Vì $a\geq c$ nên $z\geq x$.\\
		Ta có $P\geq 2(x^{2}+y^{2}+z^{2})\geq 2(xy+yz+zx)$.\\
		Mà $(a+b+c)^{2}\geq 3(ab+bc+ca)=9\Rightarrow a+b+c\geq 3$.\\
		Lại có $3=ab+bc+ca\geq 3\sqrt[3]{a^{2}b^{2}c^{2}}\Rightarrow abc\leq 1\Rightarrow 3\geq 3abc\Rightarrow a+b+c\geq 3abc$.\\
		Xét $xy+yz+zx=\dfrac{3(a+b+c+3)}{3abc+3(a+b+c)+12}\geq \dfrac{3(a+b+c+3)}{4(a+b+c+3)}=\dfrac{3}{4}\Rightarrow P\geq \dfrac{3}{2}$.\\
		Dấu bằng xảy ra khi $a=b=c=1$.
		
		{\bf Cách 2}: Sử dụng $(x+y+z)^{2}\geq 3(xy+yz+zx)$.\\
		Đặt $x=a+1$, $y=b+1$, $z=c+1$ ($x,y,z>0$). Vì $a\geq c$ nên $x\geq z\Rightarrow \dfrac{1}{z^{2}}\geq \dfrac{1}{x^{2}}$.\\
		Ta có $P\geq 2\left(\dfrac{1}{x^{2}}+\dfrac{1}{y^{2}}+\dfrac{1}{z^{2}}\right)\geq \dfrac{2}{3}\left(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\right)^{2}$.\\
		Từ giả thiết ta có \\ $(x-1)(y-1)+(y-1)(z-1)+(z-1)(x-1)=3\Leftrightarrow \dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=2\left(\dfrac{1}{xy}+\dfrac{1}{yz}+\dfrac{1}{zx}\right)$.\\
		Lại có \\
		$\left(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\right)^{2}\geq 3\left(\dfrac{1}{xy}+\dfrac{1}{yz}+\dfrac{1}{zx}\right)=\dfrac{3}{2}\left(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\right)\Rightarrow \dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\geq \dfrac{3}{2}\Rightarrow P\geq \dfrac{3}{2}$.
		\\
		Dấu bằng xảy ra khi $a=b=c=1$.
		
	}
\end{bt}

\begin{bt}%[Đề thi vào 10, Sở giáo dục Hưng Yên, 2017]%[9D5G1]
	Cho hai số thực dương $x,y$ thỏa mãn điều kiện $x+y\leq 4$. Tìm giá trị nhỏ nhất của biểu thức $P=\dfrac{2}{x^2+y^2}+\dfrac{35}{xy}+2xy.$
	\loigiai{
		Ta có
		\begin{align*}
		P &=\dfrac{2}{x^2+y^2}+\dfrac{35}{xy}+2xy  \\
		&=\dfrac{2}{x^2+y^2}+\dfrac{x^2+y^2}{32}+\dfrac{35}{xy}+\dfrac{35xy}{16}+2xy-\dfrac{x^2+y^2}{32}-\dfrac{35xy}{16}\\
		&=\dfrac{2}{x^2+y^2}+\dfrac{x^2+y^2}{32}+\dfrac{35}{xy}+\dfrac{35xy}{16}-\dfrac{x^2+y^2}{32}-\dfrac{xy}{16}-\dfrac{xy}{8}\\
		&=\dfrac{2}{x^2+y^2}+\dfrac{x^2+y^2}{32}+\dfrac{35}{xy}+\dfrac{35xy}{16}-\dfrac{(x+y)^2}{32}-\dfrac{xy}{8}.
		\end{align*}
		Sử dụng Cauchy cho các cặp sau $\dfrac{2}{x^2+y^2}$ và $\dfrac{x^2+y^2}{32}$; $\dfrac{35}{xy}$ và $\dfrac{35xy}{16}$ ta được
		\begin{align*}
		&\dfrac{2}{x^2+y^2}+\dfrac{x^2+y^2}{32}\geq 2\sqrt{\dfrac{2}{32}}=\dfrac{1}{2}.\\
		&\dfrac{35}{xy}+\dfrac{35xy}{16}\geq 2\sqrt{\dfrac{35^2}{16}}=\dfrac{35}{2}.
		\end{align*}
		Mặt khác $x+y\leq 4\Rightarrow xy\leq 4$ nên $ \dfrac{xy}{8}\leq\dfrac{1}{2}$ , $\dfrac{(x+y)^2}{32}\leq\dfrac{1}{2}$.\\
		Vậy ta có $\min P=\dfrac{1}{2}+\dfrac{35}{2}-\dfrac{1}{2}-\dfrac{1}{2}=17$. 
		Dấu "$=$" xảy ra $\Leftrightarrow x=y=2$ .\\
		\\
		Cách khác
		\begin{align*}
		P &=\dfrac{2}{x^2+y^2}+\dfrac{35}{xy}+2xy \\
		&=\dfrac{2}{x^2+y^2}+\dfrac{2}{2xy}+\dfrac{34}{xy}+\dfrac{17xy}{8}-\dfrac{xy}{8}\\
		&=2\left(\dfrac{1}{x^2+y^2}+\dfrac{1}{2xy}\right) +\left(\dfrac{34}{xy}+\dfrac{17xy}{8}\right)-\dfrac{xy}{8}\\
		&\geq \dfrac{2.4}{x^2+y^2+2xy}+2\sqrt{\dfrac{34.17}{8}}-\dfrac{(x+y)^2}{32}\\
		&\geq \dfrac{8}{16}+17-\dfrac{1}{2}= 17.
		\end{align*}
		Dấu "$=$" xảy ra $\Leftrightarrow x=y=2$ .
	}
\end{bt}

\begin{bt}%[Đề thi vào 10, Sở giáo dục Thanh Hóa, 2016]%[9D5G1]
	Cho các số dương $a, b, c$ thỏa mãn $\dfrac{1}{a+b}+\dfrac{1}{b+c}+\dfrac{1}{a+c}=2017$. Tìm giá trị lớn nhất của biểu thức $P=\dfrac{1}{2a+3b+3c}+\dfrac{1}{3a+2b+3c}+\dfrac{1}{3a+3b+2c}$.
	\loigiai
	{
		Ta có\\
		$\begin{array}{ll}
		P & =\dfrac{1}{2a+3b+3c}+\dfrac{1}{3a+2b+3c}+\dfrac{1}{3a+3b+2c} \\
		&=\dfrac{1}{(b+c)+(b+c)+(b+a)+(c+a)}+\dfrac{1}{(a+c)+(a+c)+(a+b)+(b+c)} \\

		&\leq\dfrac{1}{16}\left(\dfrac{1}{b+c}+\dfrac{1}{b+c}+\dfrac{1}{b+a}+\dfrac{1}{c+a}\right)+\dfrac{1}{16}\left(\dfrac{1}{a+c}+\dfrac{1}{a+c}+\dfrac{1}{a+b}+\dfrac{1}{b+c}\right) \\
		&+\dfrac{1}{16}\left(\dfrac{1}{a+b}+\dfrac{1}{a+b}+\dfrac{1}{a+c}+\dfrac{1}{b+c}\right) \\

\begin{bt}[{Đề thi vào lớp 10 tỉnh Bắc Ninh 2016-2017}] %Nguyễn Chính Tâm

	\loigiai
	{
		Trước hết ta đặt $$P=2\sqrt{a^2-ab+b^2}+\sqrt{a^2-2ac+4c^2}+\sqrt{b^2-2bc+4c^2}.$$
		Ta có $$\sqrt{a^2-ab+b^2}=\sqrt{\dfrac{3}{4}(a-b)^2+\dfrac{1}{4}(a+b)^2}\ge \dfrac{|a+b|}{2}=2|c|.$$

		&\sqrt{a^2-2ac+4c^2}=\sqrt{\dfrac{3}{4}(a-2c)^2+\dfrac{1}{4}(a+2c)^2}\ge \dfrac{|a+2c|}{2},\\ 
		&\sqrt{b^2-2bc+4c^2}=\sqrt{\dfrac{3}{4}(b-2c)^2+\dfrac{1}{4}(b+2c)^2}\ge \dfrac{|b+2c|}{2}.
		\end{aligned}$$
		Do đó ta có $$
		P\ge 4|c|+\dfrac{|a+2c|+|b+2c|}{2}\ge 4|c|+\dfrac{|a+2c+b+2c|}{2}=8|c|\ge 8c.$$
		Dấu bằng xảy ra khi và chỉ khi $a=b=2c\ge 0.$
		$\blacksquare$
	}
\end{bt}

\begin{bt}[{Toán tuổi thơ B3-85, 03-2010}]%Nguyễn Chính Tâm
	Cho $a,b,c$ là các số thực thỏa mãn điều kiện $a^2+b^2+c^2=1.$ Chứng minh rằng $$\sqrt{a^2+b^2c^2}+\sqrt{b^2+c^2a^2}+\sqrt{c^2+a^2b^2}\ge ab+bc+ca+1.$$
	\loigiai{
		Vì $a^2+b^2+c^2=1$ nên ta có $$\begin{aligned}
		a^2+b^2c^2&=a^2(a^2+b^2+c^2)+b^2c^2\\ 
		&=(a^2+b^2)(a^2+c^2)\ge \left(a^2+|bc|\right)^2.
		\end{aligned}$$
		Do đó $\sqrt{a^2+b^2c^2}\ge a^2+|bc|.$ Tương tự ta có $$\begin{aligned}
		&\sqrt{b^2+c^2a^2}\ge b^2+|ca|,\\ 
		&\sqrt{c^2+a^2b^2}\ge c^2+|ab|.
		\end{aligned}$$
		Do đó ta có $$\begin{aligned}
		\sqrt{a^2+b^2c^2}+\sqrt{b^2+c^2a^2}+\sqrt{c^2+a^2b^2}&\ge a^2+b^2+c^2+|ab|+|bc|+|ca|\\ 
		&\ge 1+ab+bc+ca.
		\end{aligned}$$
		Dấu bằng xảy ra khi $a^2(b^2-c^2)=b^2(c^2-a^2)=c^2(a^2-b^2)$ và $ab,bc,ca\ge 0$, suy ra $a=b=c=\pm \dfrac{1}{\sqrt{3}}$ hoặc $(a,b,c)$ là các hoán vị của một trong hai bộ số $(0,0,1)$, $(0,0,-1).$
	}
\end{bt}

\begin{bt}[{Toán tuổi thơ 87+88, 05+06-2010}]%Nguyễn Chính Tâm
	Cho $a,b,c$ là các số thực dương thỏa mãn $a+b+c=3.$ Chứng minh rằng $$\sqrt{a}+\sqrt{b}+\sqrt{c}\ge \left(a^2+b^2+c^2\right)\left(4-\dfrac{1}{a^2}-\dfrac{1}{b^2}-\dfrac{1}{c^2}\right).$$
	\loigiai{Áp dụng bất đẳng thức AM-GM cho ba số dương ta có 
		$$\sqrt{a}+\sqrt{a}+a^2\ge 3\sqrt[3]{\sqrt{a}\cdot\sqrt{a}\cdot a^2}=3a\Leftrightarrow 2\sqrt{a}+a^2\ge 3a.$$
		Tương tự ta có $$2\sqrt{b}+b^2\ge 3b, 2\sqrt{c}+c^2\ge 3c.$$
		Do đó kết hợp với giả thiết $a+b+c=3$ ta có
		$$\begin{aligned}
		&2\left(\sqrt{a}+\sqrt{b}+\sqrt{c}\right)+a^2+b^2+c^2\ge 3(a+b+c)=(a+b+c)^2\\ 
		\Leftrightarrow\;&\sqrt{a}+\sqrt{b}+\sqrt{c}\ge ab+bc+ca.
		\end{aligned}$$
		Mặt khác ta có 
		{\allowdisplaybreaks
			\begin{align*}
		&ab+bc+ca-\left(a^2+b^2+c^2\right)\left(4-\dfrac{1}{a^2}-\dfrac{1}{b^2}-\dfrac{1}{c^2}\right)\\ 
		=&\left(\dfrac{a^2}{b^2}-2+\dfrac{b^2}{a^2}\right)+\left(\dfrac{b^2}{c^2}-2+\dfrac{c^2}{b^2}\right)+\left(\dfrac{c^2}{a^2}-2+\dfrac{a^2}{c^2}\right)\\
		+&9-4(a^2+b^2+c^2)+ab+bc+ca\\
		=&\dfrac{(a^2-b^2)^2}{a^2b^2}+\dfrac{(b^2-c^2)^2}{b^2c^2}+\dfrac{(c^2-a^2)^2}{c^2a^2}\\
		+&(a+b+c)^3-4(a^2+b^2+c^2)+ab+bc+ca\\
		=&\dfrac{(a-b)^2(a+b)^2}{a^2b^2}+\dfrac{(b-c)^2(b+c)^2}{b^2c^2}+\dfrac{(c-a)^2(c+a)^2}{c^2a^2}\\
		-&3(a^2+b^2+c^2-ab-bc-ca)\\
		=&\dfrac{(a-b)^2(a+b)^2}{a^2b^2}+\dfrac{(b-c)^2(b+c)^2}{b^2c^2}+\dfrac{(c-a)^2(c+a)^2}{c^2a^2}\\
		-&\dfrac{3}{2}\left[(a-b)^2+(b-c)^2+(c-a)^2\right]\\
		=&(a-b)^2\left[\dfrac{(a+b)^2}{a^2b^2}-\dfrac{3}{2}\right]+(b-c)^2\left[\dfrac{(b+c)^2}{b^2c^2}-\dfrac{3}{2}\right]+(c-a)^2\left[\dfrac{(c+a)^2}{c^2a^2}-\dfrac{3}{2}\right].
		\end{align*}}Nếu $x,y$ là hai số thực dương thỏa mãn $x+y<3$ thì áp dụng bất đẳng thức AM-GM cho hai số dương ta có
		$$\dfrac{(x+y)^2}{x^2y^2}=\dfrac{16(x+y)^2}{\left(2\sqrt{xy}\right)^4}=\dfrac{16(x+y)^2}{\left(x+y\right)^4}=\dfrac{16}{(x+y)^2}>\dfrac{16}{9}>\dfrac{3}{2}.$$
		Do đó ta có $$\sqrt{a}+\sqrt{b}+\sqrt{c}\ge \left(a^2+b^2+c^2\right)\left(4-\dfrac{1}{a^2}-\dfrac{1}{b^2}-\dfrac{1}{c^2}\right).$$
		Dấu bằng xảy ra khi và chỉ khi $a=b=c=1.$}
\end{bt}

\begin{bt}[{Toán tuổi thơ 90+91, 08+09-2010}]%Nguyễn Chính Tâm
	Cho $a,b,c$ là các số thực dương. Tìm giá trị nhỏ nhất của biểu thức $$P=\sqrt{\dfrac{a}{b+c}}+\sqrt{\dfrac{b}{c+a}}+\sqrt{\dfrac{c}{a+b}}+\dfrac{\sqrt{2}(a+b+c)}{\sqrt{ab}+\sqrt{bc}+\sqrt{ca}}\cdot$$
	\loigiai{Áp dụng bất đẳng thức Cauchy-Schwarz ta có 
		$$\begin{aligned}
		&\left(\sqrt{\dfrac{a}{b+c}}+\sqrt{\dfrac{b}{c+a}}+\sqrt{\dfrac{c}{a+b}}\right)\left(\sqrt{a(b+c)}+\sqrt{b(c+a)}+\sqrt{c(a+b)}\right)\\ 
		&\ge \left(\sqrt{a}+\sqrt{b}+\sqrt{c}\right)^2.
		\end{aligned}$$
		Mặt khác, áp dụng bất đẳng thức AM-GM ta có 
		$$\begin{aligned}
		&\sqrt{a(b+c)}+\sqrt{b(c+a)}+\sqrt{c(a+b)}\\ 
		&=\dfrac{\sqrt{2}}{4}\left(2\sqrt{2a(b+c)}+2\sqrt{2b(c+a)}+2\sqrt{2c(a+b)}\right)\\
		&\le \dfrac{\sqrt{2}}{4}(2a+b+c+2b+c+a+2c+a+b)=\sqrt{2}(a+b+c).
		\end{aligned}$$
		Do đó ta có $$\begin{aligned}
		P&\ge \dfrac{\left(\sqrt{a}+\sqrt{b}+\sqrt{c}\right)^2}{\sqrt{2}(a+b+c)}+\dfrac{\sqrt{2}(a+b+c)}{\sqrt{ab}+\sqrt{bc}+\sqrt{ca}}\\ 
		&=\dfrac{a+b+c+2\left(\sqrt{ab}+\sqrt{bc}+\sqrt{ca}\right)}{\sqrt{2}(a+b+c)}+\dfrac{\sqrt{2}(a+b+c)}{\sqrt{ab}+\sqrt{bc}+\sqrt{ca}}\\ 
		&=\dfrac{1}{\sqrt{2}}+\sqrt{2}\cdot\left(\dfrac{\sqrt{ab}+\sqrt{bc}+\sqrt{ca}}{a+b+c}+\dfrac{a+b+c}{\sqrt{ab}+\sqrt{bc}+\sqrt{ca}}\right)\\ 
		&\ge \dfrac{1}{\sqrt{2}}+\sqrt{2}\cdot 2=\dfrac{5}{\sqrt{2}}\cdot
		\end{aligned}$$
		Vậy giá trị nhỏ nhất của $P$ là $\dfrac{5}{\sqrt{2}}$, đạt được khi $a=b=c.$
	}
\end{bt}
\begin{bt}[{Toán tuổi thơ B4-94, 12-2010}] %Nguyễn Chính Tâm
	Cho $a,b,c$ là các số dương thỏa mãn $$\sqrt{a^2+b^2}+\sqrt{b^2+c^2}+\sqrt{c^2+a^2}=3\sqrt{2}.$$
	Tìm giá trị nhỏ nhất của biểu thức $$P=\dfrac{a^2}{b+c}+\dfrac{b^2}{c+a}+\dfrac{c^2}{a+b}\cdot$$
	\loigiai
	{
		Trước hết, ta đặt $$x=\sqrt{b^2+c^2}, y=\sqrt{c^2+a^2}, z=\sqrt{a^2+b^2}.$$
		Khi đó $a^2=\dfrac{y^2+z^2-x^2}{2}$. Do đó 
		$$\begin{aligned}
		\dfrac{a^2}{b+c}&\ge \dfrac{a^2}{\sqrt{2(b^2+c^2)}}=\dfrac{y^2+z^2-x^2}{2\sqrt{2}x}\\ 
		&=\dfrac{1}{2\sqrt{2}}\left(\dfrac{y^2}{x}+\dfrac{z^2}{x}-x\right)=\dfrac{1}{2\sqrt{2}}\left[\left(\dfrac{y^2}{x}+x\right)+\left(\dfrac{z^2}{x}+x\right)-3x\right]\\
		&\ge \dfrac{1}{2\sqrt{2}}\left(2y+2z-3x\right).
		\end{aligned}$$
		Tương tự ta có $$\begin{aligned}
		&\dfrac{b^2}{c+a}\ge \dfrac{1}{2\sqrt{2}}\left(2z+2x-3y\right),\\ 
		&\dfrac{c^2}{a+b}\ge  \dfrac{1}{2\sqrt{2}}\left(2x+2y-3z\right).
		\end{aligned}$$
		Từ đó ta có $$P\ge \dfrac{1}{2\sqrt{2}}(x+y+z)=\dfrac{3}{2}.$$
		Vậy giá trị nhỏ nhất của $P$ là $\dfrac{3}{2}$, đạt được khi $a=b=c=1.$
	}
\end{bt}
