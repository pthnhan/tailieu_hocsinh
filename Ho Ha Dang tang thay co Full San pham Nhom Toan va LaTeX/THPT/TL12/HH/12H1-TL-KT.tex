\section{ĐỀ KIỂM TRA CHƯƠNG I}
\subsection{Đề số 1a}
	\setcounter{ex}{0}
\begin{bt}%[Phan Quốc Trí]%[2H1B3][2H1K3][2H1G3]
Cho khối tứ diện đều $ABCD$ có cạnh bằng $a$. Gọi $M,N$ lần lượt là trung điểm của các cạnh $AB,BC$ và $E$ là điểm đối xứng với $B$ qua $D$.
\begin{enumerate}[a.]
\item Tính thể tích của khối tứ diện $ABCD$.
\item Tính tỉ số thể tích giữa hai khối chóp  $EBMN$ và $ABCD$.
\item Mặt phẳng $(MNE)$ chia khối tứ diện $ABCD$ thành hai khối đa diện. Tính thể tích khối đa diện chứa đỉnh $A$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a.]
\immini{
\item Gọi $O$ là hình chiếu của $A$ trên $(ABCD)$. Ta có $OD=\dfrac{2}{3}DN=\dfrac{a\sqrt{3}}{3}$ và \\$AO=\sqrt{AD^2-OD^2}= a\sqrt{\dfrac{2}{3}}$.\\
$V_{ABCD}=\dfrac{1}{3} \cdot S_{ABC} \cdot AO=\dfrac{a^3\sqrt{2}}{12}$.
\item Ta có $S_{BMN}=\dfrac{1}{2} \cdot BM \cdot BN \sin B $\\
$= \dfrac{1}{2}\cdot \dfrac{1}{2} \cdot \dfrac{1}{2} \cdot BA \cdot BC \sin B=\dfrac{1}{4} \cdot S_{\Delta ABC}$\\
và $d(E,(ABC))=2d(D,(ABC))$ nên
}{
\begin{tikzpicture}[scale=0.7, line width=1.2pt]%1
\tkzDefPoints{0/0/B, 2.5/-2/C, 6/0/D, 0/6/E}
\tkzDefMidPoint(B,C) \tkzGetPoint{N}
\tkzDefPointBy[homothety=center D ratio 0.6](N) \tkzGetPoint{O}
\tkzDefPointsBy[translation=from B to E](O){A}
\tkzDefPointBy[symmetry=center D](B) \tkzGetPoint{E}
\tkzDefMidPoint(B,A) \tkzGetPoint{M}
\tkzInterLL(E,M)(A,D) \tkzGetPoint{Q}
\tkzInterLL(D,C)(E,N) \tkzGetPoint{P}
\tkzDrawSegments(B,C C,P A,B A,C A,Q E,Q E,P M,N P,Q)
\tkzDrawSegments[dashed](E,B D,N A,O P,D Q,D M,Q N,P)
\tkzLabelPoints(D,C,O,E,P,Q)
\tkzLabelPoints[left](B,M,N)
\tkzLabelPoints[above](A)
\end{tikzpicture}
}
\begin{eqnarray*}
V_{EBMN}&=&\dfrac{1}{3} \cdot S_{BMN}d(E,(ABC))
= \dfrac{1}{3} \cdot \dfrac{1}{4} \cdot S_{\Delta ABC} \cdot 2 d(D,(ABC))\\
&=&\dfrac{1}{2} \cdot \dfrac{1}{3}S_{\Delta ABC} \cdot d(D,(ABC))
=\dfrac{1}{2}V_{ABCD}
\end{eqnarray*}
Vậy $\dfrac{V_{EBMN}}{V_{ABCD}}=\dfrac{1}{2}$.
\item   Đặt $V=V_{ABCD}$.\\
Ta có $Q, P$ lần lượt là trọng tâm các tam giác $ABE, BCE$.
Áp dụng tỉ số thể tích ta có $\dfrac{V_{E.PQD}}{V_{E.BMN}}=\dfrac{2}{9}$. Suy ra $V_{E.PQD}=\dfrac{2}{9}\cdot V_{E.BMN}\Rightarrow V_{BMNDQP}=\dfrac{7}{9}\cdot V_{E.BMN}=\dfrac{7}{18}.V$\\
	Tức là phần khối đa diện không chứa điểm $A$ có thể tích bằng $\dfrac{7}{18}V$, nên phần chứa điểm $A$ có thể tích là $\dfrac{11}{18}V=\dfrac{11\sqrt{2}a^3}{216}.$ 
\end{enumerate}
}
\end{bt}
\begin{bt}%[Phan Quốc Trí]%[2H1B3][2H1K3]
Cho lăng trụ tam giác $ABC.A'B'C'$ có tam giác $ABC$ vuông cân tại $B$ và $BC=AB=a$. Cạnh $A'A=2a$ và hợp với mặt phẳng $(ABC)$ một góc $60^{\circ}$ . Gọi $M,N$ lần lượt là trung điểm của $BB',CC'$.
\begin{enumerate}[a.]
\item Tính thể tích khối lăng trụ $ABC.A'B'C'$.
\item Mặt phẳng $A'MN$ chia khối lăng trụ thành hai khối đa diện. Đặt $V_1$ là thể tích của khối đa diện chứa điểm $B$, $V_2$ là phần còn lại. Tính tỉ số $\dfrac{V_1}{V_2}$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a.]
\immini{
\item 
Gọi $H$ là hình chiếu của $A'$ trên $(ABC)$\\
$\Rightarrow (A'A,(ABC))= \widehat{A'AH}=60^{\circ}$\\
$A'H=AA' \cdot \sin 60^{\circ} = 2a \dfrac{\sqrt{3}}{2} = a\sqrt{3}$.\\
$V_{ABC.A'B'C'} = S_{ABC} \cdot A'H=\dfrac{a^3\sqrt{3}}{2}$.
\item Ta có $S_{MNC'B'}=\dfrac{1}{2}S_{BCC'B'}$, suy ra \\ $V_{A'.MNC'B'}=\dfrac{1}{2}V_{A'.BCC'B'}=\dfrac{1}{2} \left( V_{ABC.A'B'C'}-V_{A'.ABC} \right)$.\\
Mà $V_{A'.ABC}=\dfrac{1}{3}V_{ABC.A'B'C'}$ nên \\
}{
\begin{tikzpicture}[scale=0.6, line width = 1.2 pt]
\tkzDefPoints{0/0/A, 2.5/-2/B, 7/0/C, 2.5/6/A', 2.5/-0.75/H}

}

$V_{A'MNC'B'}=\dfrac{1}{2}\left( V_{ABC.A'B'C'}-\dfrac{1}{3}V_{ABC.A'B'C'} \right)=\dfrac{1}{3}V_{ABC.A'B'C'}$.\\
Vậy $\dfrac{V_1}{V_2}=\dfrac{V_{A'MNABC}}{V_{A'MNC'B'}}=\dfrac{V_{ABC.A'B'C'}-\dfrac{1}{3}V_{ABC.A'B'C'}}{\dfrac{1}{3}V_{ABC.A'B'C'}} = 2$.
\end{enumerate}
}
\end{bt}

	\begin{enumerate}
		\item Thể tích khối chóp $S.ABCD$.
		\item Thể tích khối tứ diện $BCMN$.
		\item Thể tích khối chóp $S.ABMN$.
	\end{enumerate}

\end{tikzpicture}

\begin{enumerate}
	\item 
	\begin{itemize}
	\item $V_{S.ABCD}=\dfrac{1}{3}SO\cdot S_{ABCD}$. \dotfill $1{,}0$ điểm.
	\item Có $BO^2=AB^2-AO^2=5a^2-4a^2=a^2\Rightarrow BO=a$.\\
	$S_{ABCD}=\dfrac{1}{2}AC\cdot BD=\dfrac{1}{2}\cdot 4a\cdot 2a=4a^2$.\\
	Vậy  $V_{S.ABCD}=\dfrac{1}{3}SO\cdot S_{ABCD}=\dfrac{1}{3}\cdot 2\sqrt{2}a\cdot 4a^2=\dfrac{8\sqrt{2}a^3}{3}$.\dotfill $1{,}0$ điểm.
	\end{itemize}
	\item 
	\begin{itemize}
	\item $V_{BCMN}=\dfrac{1}{3}d(N,(BCM))\cdot S_{BCM}$.\\
	Có $d(N,(BCM))=\dfrac{1}{2}d(D,(SBC))$ và $S_{BCM}=\dfrac{1}{2}S_{SBC}$.\\
	Suy ra $V_{BCMN}=\dfrac{1}{4}V_{D.SBC}$ \dotfill $1{,}0$ điểm.
	\item Có $V_{D.SBC}=V_{S.BCD}=\dfrac{1}{2}V_{S.ABCD}$.\\
	Vậy  $V_{BCMN}=\dfrac{1}{8}V_{S.ABCD}=\dfrac{\sqrt{2}a^3}{3}$ .\dotfill $1{,}0$ điểm.
	\end{itemize}
    \item 
	\begin{itemize}
	\item $V_{S.ABMN}=V_{S.ABN}+V_{S.BMN}$.\\
	Có $\dfrac{V_{S.ABN}}{V_{S.ABD}}=\dfrac{SA}{SA}\cdot \dfrac{SB}{SB}\cdot \dfrac{SN}{SD}=\dfrac{1}{2}$\\
	Suy ra $V_{S.ABN}=\dfrac{1}{2}V_{S.ABD}=\dfrac{1}{4}V_{S.ABCD}.$ \dotfill $1{,}0$ điểm.
	\item Có $\dfrac{V_{S.BMN}}{V_{S.BCD}}=\dfrac{SB}{SB}\cdot \dfrac{SM}{SC}\cdot \dfrac{SN}{SD}=\dfrac{1}{4}$\\
	Suy ra $V_{S.BMN}=\dfrac{1}{4}V_{S.ABD}=\dfrac{1}{8}V_{S.ABCD}.$\\
	Vậy $V_{S.ABMN}=V_{S.ABN}+V_{S.BMN}=\dfrac{3}{8}V_{S.ABCD}=\sqrt{2}a^3$. \dotfill $1{,}0$ điểm.
	\end{itemize}
\end{enumerate}
	}
\end{bt}

	Vậy $V_{A'B'C'ABCD}=V_{A'B'C'D'.ABCD}-V_{D.A'C'D'}=\dfrac{5}{6}a^3$. \dotfill $1{,}0$ điểm.

	\end{align*}

	$\dfrac{V_{A'MBD}}{V_{ABCD.A'B'C'D'}}=\dfrac{1}{4}\Leftrightarrow \dfrac{a^2(a+x)}{6a^3}=\dfrac{1}{4}\Leftrightarrow 2(a+x)=3a\Leftrightarrow x=\dfrac{a}{2}$.\\
Vậy khi $M$ là trung điểm cạnh $CC'$ thì  $\dfrac{V_{A'MBD}}{V_{ABCD.A'B'C'D'}}=\dfrac{1}{4}$. \dotfill $1{,}0$ điểm.
	\end{itemize}
\end{enumerate}
	}
