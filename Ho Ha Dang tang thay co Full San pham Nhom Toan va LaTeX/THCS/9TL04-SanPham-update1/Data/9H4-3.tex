 \subsubsection{Diện tích mặt cầu và thể tích hình cầu}
 - Diện tích mặt cầu: $S=4\pi R^2$ hay $S=\pi d^2$, với $R$ là bán kính; $d$ là đường kính.\\
 - Thể tích hình cầu $V=\dfrac{4}{3}\pi R^3$.
 
 
\subsection{Các ví dụ}
\begin{vd}%[Nguyễn Cao Cường]%[9H4B3]
	Một phao cơ hình cầu tự động đóng nước chảy vào bể khi bể đầy. Biết diện tích bề mặt của phao là $804$ cm$^2$, tính bán kính của phao.
	\loigiai{
	Từ công thức $S=4\pi R^2 \Rightarrow R=\sqrt{\dfrac{S}{4\pi}}$.\\
	Bán kính của phao là $R=\sqrt{\dfrac{804}{4\pi}}\approx 8$ cm.
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[9H4B3]
	Phần trên của một chiếc cốc chân cao có dạng nửa hình cầu. Biết cốc này có thể chứa được $56,5$ ml nước. Tính đường kính của miệng cốc.
	\loigiai{
	Vì dung dích của cốc là $56{,}5$ ml nên thể tích của cốc là $56{,}5$ cm$^3$.\\
	Ta có $V=\dfrac{4}{3}\pi R^3$ do đó có thể tích của nửa hình cầu là $\dfrac{2}{3}\pi R^3$.\\
	Theo đề bài, ta có $\dfrac{2}{3}\pi R^3=56{,}5 \Rightarrow R^3=\dfrac{3\cdot 56\cdot 5}{2\pi}\approx 27$ cm$^3$, suy ra $R=3$ cm.\\
	Vậy đường kính của miệng cốc là $3\cdot 2=6$ cm.
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[9H4K3]
	Một trái dưa có dạng hình cầu. Bổ đôi trái dưa này ra thì mặt cắt có diện tích là $314$ cm$^2$. Tính thể tích của trái dưa đó.
	\loigiai{
Khi bổ đôi trái dưa thì mặt cắt là một hình tròn.\\
Ta có: $S=\pi R^2\Rightarrow R=\sqrt{\dfrac{S}{\pi}}\approx \sqrt{\dfrac{314}{3{,}14}}=10$ cm.\\
Vậy bán kính của trái dưa là $10$ cm.\\
Thể tích của trái dưa là:
$$V=\dfrac{4}{3}\pi R^3=\dfrac{4}{3}\pi\cdot 10^3\approx 4187 \; \text{cm}^3.$$

}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[9H4B3]
	Trái đất có bán kính $6400$ km. Diện tích biển và đại dương chiếm $\dfrac{3}{4}$ bề mặt trái đất. Hãy tính diện tích biển và đại dương của trái đất (làm tròn đến triệu km$^2$).
	\loigiai{
	Diện tích bề mặt trái đất là $S=4\pi R^2=4\cdot \pi \cdot 6400^2 \approx 514 457 600$ km$^2$.\\
Diện tích các biển và đại dương là 
$ 514 457 600 \cdot \dfrac{3}{4}\approx 386000000$ km$^2$.
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường]%[9H4G3]
	\immini{
	Hình bên minh họa bộ phận lọc của một bình nước. Bộ phận này gồm một hình trụ và một nửa hình cầu  với kích thước ghi trên hình. Hãy tính
	\begin{enumerate}
		\item Thể tích của bộ phận đó;
		\item Diện tích mặt ngoài của bộ phận này.
	\end{enumerate}}{
\begin{tikzpicture}
\tkzInit[xmin=0.5, xmax=6, ymin=0, ymax=6]
\tkzClip
\tkzDefPoints{1/4/A, 5/4/B, 1/1/C, 5/1/D, 3/4/O, 3/1/F}
\draw[dashed] (B) arc(0:180:2 cm and 1 cm);

\loigiai{
	\begin{enumerate}
		\item Thể tích phần hình trụ là $V_1=\pi R^2h=\pi\cdot 5^2\cdot 6=150\pi$ cm$^3$.\\
Thể tích nửa hình cầu:
$$V_2=\dfrac{1}{2}\cdot \dfrac{4}{3}\pi R^3=\dfrac{2}{3}\pi \cdot 5^3=\dfrac{250}{3}\pi \;\text{cm}^3.$$
Thể tích bộ phận lọc là:
$$V=V_1+V_2=150\pi +\dfrac{250}{3}\pi=\dfrac{700}{3}\pi\;\text{cm}^3\approx 733\;\text{cm}^3.$$
\item Diện tích xung quanh của hình trụ là:
$$S_1=2\pi Rh=2\pi\cdot 5\cdot 6=60\pi\;\text{cm}^2.$$
Diện tích đáy hình trụ là:
$$S_2=\pi\cdot R^2=\pi\cdot 5^2=25\pi\;\text{cm}^3.$$
Diện tích nửa mặt cầu là:
$$S_3=\dfrac{1}{2}\cdot 4\pi R^2=2\pi\cdot 5^2=50\pi\;\text{cm}^3.$$
Diện tích mặt ngoài của bộ phận lọc:
$$S=S_1+S_2+S_3=60\pi+25\pi+50\pi=135\pi \;\text{cm}^2\approx 424 \;\text{cm}^2.$$
\end{enumerate}
}
\end{vd}

\begin{enumerate}
