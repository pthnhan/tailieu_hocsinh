\begin{ex}%[Đề thi vào 10, chuyên Khoa Học Tự Nhiên, tỉnh Hà Nội, năm 2018, vòng 2, Dự án EX-TS10C-18]%[9D0G2]
\begin{enumerate}
	\item Giải hệ phương trình $\heva{&xy(x+y)=2\\&x^3+y^3+x^3y^3+7(x+1)(y+1)=31.}$
	\item Giải phương trình $9+3\sqrt{x(3-2x)}=7\sqrt{x}+5\sqrt{3-2x}$.
	\end{enumerate}	
	\loigiai{
\begin{enumerate}
	\item Đặt $a=x+y$ và $b=xy(a^2\geq 4b)$. Hệ phương trình có thể được viết thành
	\begin{center}
		$\heva{&ab=2\\&a^3-3ab+b^3+7(a+b+1)=31.}$
	\end{center}
Từ đó ta có 
\begin{eqnarray*}
&31 &=a^3-3ab+b^3+7(a+b+1)\\
& &=(a+b)^3-3ab(a+b)+7(a+b)+1\\
& &=(a+b)^3+(a+b)+1.
\end{eqnarray*}
Suy ra $(a+b)^3+(a+b)-30=0$ hay $a+b=3$ (vì nếu $a+b>3$ thì $(a+b)^3+(a+b)-30>0$, còn nếu $a+b<3$ thì $(a+b)^3+(a+b)-30<0$).\\
Giải hệ phương trình $a+b=3$ và $ab=2$ với chú ý $a^2\geq 4b$ ta được $a=2;b=1$. Từ đó dễ dàng tìm được $x=y=1$. Vậy hệ phương trình đã cho có nghiệm duy nhất $(x;y)=(1;1)$.
\item Điều kiện: $0\leq x\leq \dfrac{1}{2}$.
\\
Đặt $a=\sqrt{x}$ và $b=\sqrt{3-2x}$ thì ta có $a,b\geq 0$ và $2a^2+b^2=3$. Ngoài ra, từ giả thiết, ta cũng có
\begin{align*}
9+3ab=7a+5b.\tag{1}
\end{align*}
Từ đó, ta có\\
$2a^2+b^2+3ab+6=7a+5b\Leftrightarrow (a+b-2)(2a+b-3)=0$
suy ra  $b=2-a$ hoặc $b=3-2a$.
\begin{itemize}
	\item Với $b=2-a$ thay vào (1) ta được $9+3a(2-a)=2a+10$. Giải phương trình này, ta được $a=\dfrac{1}{3}$ (tương ứng, $b=\dfrac{2}{3}$ và $x=\dfrac{1}{9}$), hoặc $a=1$ (tương ứng, $b=1$ và $x=1$).
	\item Với $b=3-2a$, thay vào (1) ta được $9+3a(3-2a)=2a+10$. Giải phương trình này, ta được $a=1$ (tương ứng, $b=1$ và $x=1$).
\end{itemize} 
Vậy phương trình đã cho có nghiệm là $x=\dfrac{1}{9}$ và $x=1$.
	\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Khoa Học Tự Nhiên, tỉnh Hà Nội, năm 2018, vòng 2, Dự án EX-TS10C-18]%[9D0G4]%[9D0G8]
	\begin{enumerate}
		\item Cho $x,y$ là các số nguyên sao cho $x^2-2xy-y$ và $xy-2y^2-x$ đều chia hết cho $5$.
		\item Cho $a_1,a_2,\cdots,a_{50}$ là các số nguyên thỏa mãn
			$1\leq a_1\leq a_2\leq \cdots\leq x_{50}\leq 50$\\
			 và $a_1+a_2+\cdots+a_{50}=100$.\\
		Chứng minh rằng từ các số đã cho ta có thể chọn được một vài số có tổng bằng $50$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Từ giả thiết, ta suy ra $(x+y)(x-2y-1)=(x^2-2xy-y)+(xy-2y^2-x)$ chia hết cho $5$. Do đó, trong hai số $x+y$ và $x-2y-1$ có ít nhất một số chia hết cho $5$.
			\begin{itemize}
				\item Nếu $x+y$ chia hết cho $5$ thì $y\equiv x\pmod 5$, suy ra
				$$0\equiv x^2-2xy-y\equiv x^2+2x^2+x\equiv x(3x+1) \pmod 5.$$
				Do đó $x$ chia hết cho $5$ hoặc $x$ chia cho $5$ dư $3$.
				\begin{itemize}
					\item Nếu $x$ chia hết cho $5$ thì ta cũng có $y$ chia hết cho $5$ (do $x+y$ chia hết cho $5$) nên hiển nhiên $2x^2+y^2+2x+y$ chia hết cho $5$.
					\item Nếu $x$ chia $5$ dư $3$ thì $y$ chia $5$ dư $2$ (do $x+y$ chia hết cho $5$), suy ra
					$$2x^2+y^2+2x+y\equiv 2\cdot 3^2+2^2+2\cdot 3+2\equiv 0 \pmod 5.$$
				\end{itemize}
				\item Nếu $x-2y-1$ chia hết cho $5$ thì ta có  $x\equiv 2y+1\pmod 5$, suy ra
				$$0\equiv x^2-2xy-y\equiv (2x+1)^2-2y(2y+1)-y\equiv y+1\pmod 5.$$
				Tóm lại, trong mọi trường hợp, ta luôn có $2x^2+y^2+2x+y$ chia hết cho $5$.
			\end{itemize}
		\item Nếu tồn tại chỉ số $n$ ($1\leq n\leq 50$) sao cho $a_1+a_2+\cdots+a_n=50$ thì kết luận của bài toán là hiển nhiên.\\
		Xét trường hợp ngược lại, khi đó tồn tại chỉ số $n$ ($1\leq n\leq 49$) sao cho
		\begin{align*}
	a_1+a_2+\cdots+a_n\leq 49,a_1+a_2+\cdots+a_{n+1}\geq 51.\tag{1}
			\end{align*}
		Từ đây suy ra $a_{n+1}\geq 2$. Ta xét hai trường hợp sau\\
		\begin{enumerate}
			\item[\textbf{Trường hợp 1}]: $a_{n+1}=2$. Từ (1), dễ thấy $a_1+a_2+\cdots+a_n=49$, suy ra
			$$a_{n+2}+a_{n+3}+\cdots+a_{50}=49.$$
			Nếu $n\leq 24$ thì do $a_1\leq a_{n+2},a_2\leq a_{n+3},\cdots.a_n\leq a{2n+1}$ nên\\ $49=a_1+a_2+\cdots +a_n\leq a_{n+2}+a_{n+3}+\cdots+a_{2n+1}< a_{n+2}+a_{n+3}+\cdots+a_{49}+a_{50}$, mâu thuẫn. Vậy $n\geq 25$, suy ra
			$$49=a_1+a_2+\cdots+a_n\geq na_1\geq 25a_1.$$
			Từ đây, ta có $a_1<2$ nên $a_1=1$. Do đó $a_2+\cdots+a_n=48$ và $a_2+\cdots+a_{n+1}=50$. 
			\item[\textbf{Trường hợp 2}]: $a_{n+1}\geq 3$. Lúc này, do
			\begin{align*}
		a_{n+2}+a_{n+3}+\cdots+a_{50}-100-(a_1+a_2+\cdots+a_{n+1})\leq 49.\tag{2}
						\end{align*}
			nên $49\geq (49-n)a_{n+2}\geq (49-n)\cdot 3$, suy ra $n\geq 33$. Do đó, ta có
			$$49\geq (a_1+\cdots+a_{16})+(a_{17}+\cdots+a_n)\geq 16+(n-16)a_{17}\geq 16+17a_{17},$$
				suy ra $a_{17}<2$. Mà $1\leq a_1\leq a_2\leq\cdots\leq a_{17}$ và $a_i$ là các số nguyên nên
				$$a_1=a_2=\cdots=a_{17}.$$
				\begin{itemize}
					\item Nếu $a_{n+1}\leq 18$, đặt $a_1+a_2+\cdots+a_{n+1}=50+k$ với $k\in\mathbb{Z},k\geq 1$, ta có
					$$18\geq a_{n+1}\geq (50+k)-49=k+1,$$
					suy ra $k\leq 17$. Từ đây, ta có $a_{k+1}+\cdots+a_{n+1}$ vì $a_1=a_2=\cdots=a_k=1$.
					\item Nếu $a_{n+1}\geq 19$, từ (2) và từ giả thiết, ta có
					$$49\geq (49-n)a_{n+2}\geq (49-n)\cdot 19.$$
					suy ra $n\geq 47$. Từ đây, ta có $a_1=a_2=\cdots=a_{45}=1$, vì nếu $a_{45}\geq 2$ thì $(a_1+a_2+\cdots+a_{44})+(a_{45}+\cdots+a_n)\geq 44+(n-44)a_{45}\geq 44+(47-44)\cdot2>49$.\\
					Đặt $a_{n+1}=50-k(k\in\mathbb{Z},0\leq k\leq 31)$. Khi đó, ta có $a_1+\cdots+a_k+a_{n+1}=50$ vì $a_1=a_2=\cdots=a_k=1$.
				\end{itemize}
		\end{enumerate}
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Khoa Học Tự Nhiên, tỉnh Hà Nội, năm 2018, vòng 2, Dự án EX-TS10C-18]%[9D0G7]
	Cho $a,b,c$ là các số thực dương. Chứng minh rằng
	$$\left(\sqrt{\dfrac{ab}{a+b}}+\sqrt{\dfrac{bc}{b+c}} \right)\left(\dfrac{1}{\sqrt{a+b}}+\dfrac{1}{\sqrt{b+c}} \right)\leq 2$$.  
	\loigiai{
		Sử dụng các bất đẳng thức Cauchy và AM-GM, ta có
		\begin{eqnarray*}
		&\text{VT}&\leq \sqrt{2\left(\dfrac{ab}{a+b}+\dfrac{bc}{b+c}\right) }\cdot\sqrt{2\left(\dfrac{1}{a+b}+\dfrac{1}{b+c} \right) }\\
		& &=2\sqrt{\left(\dfrac{a}{a+b}+\dfrac{c}{b+c} \right)\left(\dfrac{b}{a+b}+\dfrac{b}{b+c} \right) }\\
		& &\leq \left(\dfrac{a}{a+b}+\dfrac{c}{b+c} \right) +\left(\dfrac{b}{a+b}+\dfrac{b}{b+c} \right) \\
		& &=2.
		\end{eqnarray*}
	Ta có điều phải chứng minh. Đẳng thức xảy ra khi và chỉ khi $a=b=c$.
	}
