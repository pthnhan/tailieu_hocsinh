\begin{ex}%[Đề thi vào lớp 10 chuyên, Sở giáo dục Lâm Đồng, 2017]%[9D5G2]
    Cho $a$, $b$, $c$ là các số nguyên. Chứng minh rằng nếu $a^{2014}+b^{2015}+c^{2016}$ chia hết cho $6$ thì $a^{2016}+b^{2017}+c^{2018}$ chia hết cho $6$.
    \loigiai{Đặt $A=a^{2014}+b^{2015}+c^{2016}$, $B=a^{2016}+b^{2017}+c^{2018}$.\\
    Ta có: $B-A=a^{2014}(a^2-1)+b^{2015}(b^2-1)+c^{2016}(c^2-1)$.\\
Nhận thấy rằng $a(a^2-1)=(a-1)a(a+1)\vdots6\Rightarrow a^{2014}(a^2-1)\vdots6$.\\
Tương tự ta có $b^{2014}(b^2-1)\vdots6$, $c^{2014}(c^2-1)\vdots6$ nên ta suy ra $B-A\vdots6$.\\
Mà $A\vdots6$ nên $B\vdots6$.}
\end{ex}
\begin{ex}%[Đề thi vào lớp 10 chuyên, Sở giáo dục Lâm Đồng, 2017]%[9D3G3]
Giải hệ phương trình: $\heva{&x-\dfrac{1}{x}=y-\dfrac{1}{y}\\&2x^2-xy=1}$.
\loigiai{$x-\dfrac{1}{x}=y-\dfrac{1}{y}\Leftrightarrow(x-y)\left(1+\dfrac{1}{xy}\right)=0\Leftrightarrow\hoac{&x=y\\&y=-\dfrac{1}{x}}$.\\
Trường hợp 1: $x=y$, $(2)\Leftrightarrow2x^2-x^2=1\Leftrightarrow\hoac{&x=y=1\\&x=y=-1}$.\\
Trường hợp 2: $y=-\dfrac{1}{x}$, $(2)\Leftrightarrow2x^2+1=1\Leftrightarrow x=0$ (loại)\\
Vậy hệ phương trình có hai nghiệm là $(1,1)$, $(-1,-1)$.}
\end{ex}

\begin{ex}%[Đề thi vào lớp 10 chuyên, Sở giáo dục Lâm Đồng, 2017]%[9D5G5]
Giải phương trình $\dfrac{3x}{\sqrt{3x+10}}=\sqrt{3x+1}-1$.
\loigiai{Điều kiện: $x\ge-\dfrac{1}{3}$.\\
pt$\Leftrightarrow\dfrac{3x}{\sqrt{3x+10}}=\dfrac{3x}{\sqrt{3x+1}+1}\Leftrightarrow\hoac{&x=0\\&\sqrt{3x+10}=\sqrt{3x+1}+1(*)}$\\
$(*)\Leftrightarrow3x+10=3x+2+2\sqrt{3x+1}\Leftrightarrow\sqrt{3x+1}=4\Leftrightarrow x=5$.\\
Vậy phương trình có hai nghiệm là $x=0$ và $x=5$.}
\end{ex}
\begin{ex}%[Đề thi vào lớp 10 chuyên, Sở giáo dục Lâm Đồng, 2017]%[9D5G1]
Cho hai số dương $x$, $y$. Xác định giá trị nhỏ nhất của biểu thức $A=\sqrt{x^2+\dfrac{1}{y^2}}+\sqrt{y^2+\dfrac{1}{x^2}}$.
\loigiai{Ta có: $A\ge2\sqrt[4]{\left(x^2+\dfrac{1}{y^2}\right)\left(y^2+\dfrac{1}{x^2}\right)}=2\sqrt[4]{x^2y^2+\dfrac{1}{x^2y^2}+\dfrac{x^2}{y^2}+\dfrac{y^2}{x^2}}\ge2\sqrt[4]{2+2}=2\sqrt{2}$.\\
Dấu bằng xảy ra khi và chỉ khi $x=y=1$. Vậy $\min A=2\sqrt{2}$.}
\end{ex}

\begin{ex}%[Đề thi vào lớp 10 chuyên, Sở giáo dục Lâm Đồng, 2017]%[9D5G5]
Chứng minh rằng nếu $n$ là số tự nhiên lớn hơn $1$ thì $2^n-1$ không thể là số chính phương.
\loigiai{Giả sử $2^n-1$ là một số chính phương, do $2^n-1$ là số nguyên lẻ nên $2^n-1=(2k+1)^2\Leftrightarrow2^n=2(2k^2+2k+1)\Leftrightarrow2k^2+2k+1=2^{n-1}$, đây là điều vô lý vì $2^{n-1}$ là một số nguyên chẵn. Vậy ta có điều phải chứng minh, nghĩa là $2^n-1$ không thể là một số chính phương.}
\end{ex}

\begin{ex}%[Đề thi vào lớp 10 chuyên, Sở giáo dục Lâm Đồng, 2017]%[9D4K6]
Cho phương trình $x^2+mx+n=0$, trong đó $m$, $n$ là các tham số thỏa mãn $m+n=6$. Tìm giá trị của $m$, $n$ để phương trình có hai nghiệm phân biệt $x_1$, $x_2$ sao cho $x_1=x^2_2+x_2+2$.
\loigiai{Để phương trình có hai nghiệm phân biệt thì $\Delta=m^2-4n\ge0$.\\
Có $\heva{&x_1+x_2=-m\\&x_1x_2=n}\Leftrightarrow\heva{&x^2_2+2x_2+2=-m\\&x^3_2+x^2_2+2x_2=n}$\\
$\Rightarrow x^3_2-2=n+m=6\Leftrightarrow x^3_2=8\Leftrightarrow x_2=2\Rightarrow\heva{&m=-10\\&n=16}$(nhận).
}
\end{ex}

\begin{ex}%[Đề thi vào lớp 10 chuyên, Sở giáo dục Lâm Đồng, 2017]%[9D5G5]
Cho $a$, $b$, $c$ là các số dương thỏa mãn điều kiện $a+b+c+\sqrt{2abc}=2$. Chứng minh rằng $\sqrt{a(2-b)(2-c)}+\sqrt{b(2-c)(2-a)}+\sqrt{c(2-a)(2-b)}=\sqrt{8}+\sqrt{abc}$.
\loigiai{Ta có: $a(2-b)(2-c)=a(4-2(b+c)+bc)=a(2a+2\sqrt{2abc}+bc)=a\left(\sqrt{2a}+\sqrt{bc}\right)^2$.\\
Vậy $VT=\sqrt{a}\left(\sqrt{2a}+\sqrt{bc}\right)+\sqrt{b}\left(\sqrt{2b}+\sqrt{ca}\right)+\sqrt{c}\left(\sqrt{2c}+\sqrt{ab}\right)=\sqrt{2}(a+b+c)+3\sqrt{abc}$\\
$=\sqrt{2}\left(2-\sqrt{2abc}\right)+3\sqrt{abc}=2\sqrt{2}+\sqrt{abc}=VP$.}
\end{ex}

\begin{ex}%[Đề thi vào lớp 10 chuyên, Sở giáo dục Lâm Đồng, 2017]%[9H3K3]
Cho tam giác $ABC$ nội tiếp đường tròn $(O,R)$ với đường cao $AH=R\sqrt{2}$. Gọi $D$, $K$ lần lượt là hình chiếu của $H$ lên $AB$ và $AC$. Chứng mình rằng $S_{ADK}=\dfrac{1}{2}S_{ABC}$, với $S_{ADK}$ và $S_{ABC}$ lần  lượt là diện tích của hai tam giác $ADK$ và $ ABC $.

\immini{Ta có $AD.AB=AH^2=AK.AC\Rightarrow\dfrac{AD}{AC}=\dfrac{AK}{AB}$\\
Vậy $\triangle ADK\backsim\triangle ACB\Rightarrow\dfrac{S_{ADK}}{S_{ACB}}=\left(\dfrac{AD}{AC}\right)^2$.\\
Kẻ đường kính $AE$, ta có $\widehat{AHD}=\widehat{ABH}=\widehat{AEC}$\\
Nên $\triangle ADH\backsim\triangle ACE$\\
$\Rightarrow\dfrac{AD}{AC}=\dfrac{AH}{AE}=\dfrac{R\sqrt{2}}{2R}=\dfrac{\sqrt{2}}{2}$\\
$\Rightarrow\dfrac{S_{ADK}}{S_{ACB}}=\left(\dfrac{\sqrt{2}}{2}\right)^2=\dfrac{1}{2}\Rightarrow S_{ABC}=2S_{ADK}$.}
