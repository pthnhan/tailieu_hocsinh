\begin{bt}%[Đề thi học kì 1 lớp 11, THPT Nguyễn Thị Minh Khai,Thành phố Hồ Chí Minh 2018]%[Bùi Mạnh Tiến]%[1D1B3-3]%[1D1B2-1]%[1D1B3-6]
	Giải các phương trình sau:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\sqrt{3}\sin x-\cos x=2\sin 2x$.
			\item $\sin ^22x-\cos ^2 2x+\cos 6x=\sin x$.
			\item $1-\cot 2x=\dfrac{1+\cos 2x}{\sin ^2 2x}.$
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			\begin{eqnarray*}
				\sqrt{3}\sin x-\cos x=2\sin 2x
				&\Leftrightarrow & \dfrac{\sqrt{3}}{2}\sin x-\dfrac{1}{2}\cos x=\sin 2x\\
				&\Leftrightarrow & \sin 2x=\sin \left(x-\dfrac{\pi }{6}\right)\\
				&\Leftrightarrow & \hoac{&x=-\dfrac{\pi}{6}+k2\pi \\&x=\dfrac{7\pi}{18}+\dfrac{k2\pi}{3}} \left(k\in \mathbb{Z}\right).
			\end{eqnarray*}
			Vậy phương trình có các nghiệm là $x=-\dfrac{\pi}{6}+k2\pi$ và $x=\dfrac{7\pi}{18}+\dfrac{k2\pi}{3}$ với $k\in \mathbb{Z}$.
			\item Ta có
			\begin{eqnarray*}
				\sin ^2 2x-\cos ^2 2x+\cos 6x=\sin x
				&\Leftrightarrow & \cos 6x-\cos 4x=\sin x\\
				&\Leftrightarrow & 2\sin 5x \cdot \sin x+\sin x=0\\
				&\Leftrightarrow & \sin x \cdot \left(2\sin 5x + 1\right)=0\\
				&\Leftrightarrow & \hoac{&\sin x =0\\&\sin 5x=-\dfrac{1}{2}}\\
				&\Leftrightarrow & \hoac{&x=k\pi\\&x=-\dfrac{\pi}{30}+\dfrac{k2\pi}{5}\\&x=\dfrac{7\pi}{30}+\dfrac{k2\pi}{5}} \left(k\in \mathbb{Z}\right).
			\end{eqnarray*}
			Vậy phương trình có các nghiệm là $x=k\pi$, $x=-\dfrac{\pi}{30}+\dfrac{k2\pi}{5}$ và $x=\dfrac{7\pi}{30}+\dfrac{k2\pi}{5}$ với $k\in \mathbb{Z}$.
			\item Điều kiện: $\sin 2x\ne 0\Leftrightarrow x\ne k\pi$.\\
			Ta có
			\begin{eqnarray*}
				\dfrac{\sin 2x-\cos 2x}{\sin 2x}=\dfrac{1+\cos 2x}{1-\cos ^2 2x}
				&\Leftrightarrow & \dfrac{\sin 2x-\cos 2x}{\sin 2x}=\dfrac{1}{1-\cos 2x}\\
				&\Leftrightarrow &\left(\sin 2x-\cos 2x\right)\left(1-\cos 2x\right)=\sin 2x\\
				&\Leftrightarrow & \left(\cos 2x-\sin 2x - 1\right)\cdot \cos 2x=0\\
				&\Leftrightarrow & \hoac{&\cos 2x=0\\&\cos\left(2x+\dfrac{\pi}{4}\right)=\cos \dfrac{\pi}{4}}\\
				&\Leftrightarrow & \hoac{&x=\dfrac{\pi}{4}+\dfrac{k\pi}{2} \text{ (nhận)}\\&x=k\pi \text{ (loại)}\\& x=-\dfrac{\pi}{4}+k\pi \text{ (nhận)}} \left(k\in \mathbb{Z}\right).
			\end{eqnarray*}
			Vậy các nghiệm của phương trình là $x=\dfrac{\pi}{4}+\dfrac{k\pi}{2}$ với $k\in \mathbb{Z}$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề thi học kì 1 lớp 11, THPT Nguyễn Thị Minh Khai,Thành phố Hồ Chí Minh 2018]%[Bùi Mạnh Tiến]%[1D2B5-5]
	Có $2$ lô hàng
	\begin{itemize}
		\item Lô $1$: Có $9$ sản phẩm đạt chuẩn và $2$ phế phẩm.
		\item Lô $2$: Có $8$ sản phẩm đạt chuẩn và $3$ phế phẩm.
	\end{itemize}
	Lấy ngẫu nhiên mỗi lô hàng $1$ sản phẩm. Tính xác suất để lấy được $2$ sản phẩm mà trong đó có đúng $1$ sản phẩm đạt chuẩn.
	\loigiai
	{
		Gọi không gian mẫu $\Omega $ là tập hợp các cách lấy ra ở mỗi lô $1$ sản phẩm nên $|\Omega|= 11\cdot 11=121.$\\
		Gọi $A$ là biến cố \lq\lq$2$ sản phẩm lấy ra có đúng $1$ sản phẩm đạt chuẩn\rq\rq.
		Các trường hợp xảy ra là
		\begin{itemize}
			\item Lấy $1$ sản phẩm đạt chuẩn ở lô $1$ và $1$ phế phẩm ở lô $2$: Có $\mathrm{C}_9^1\cdot \mathrm{C}_3^1$ (cách).
			\item Lấy $1$ sản phẩm đạt chuẩn ở lô $2$ và $1$ phế phẩm ở lô $1$: Có $\mathrm{C}_8^1\cdot \mathrm{C}_2^1$ (cách).
		\end{itemize} 
		Vậy $|A|=\mathrm{C}_9^1\cdot \mathrm{C}_3^1+\mathrm{C}_8^1\cdot \mathrm{C}_2^1=43.$\\
		Xác suất cần tìm là $P(A)=\dfrac{|A|}{|\Omega|}=\dfrac{43}{121}.$
	}
\end{bt}

\begin{bt}%[Đề thi học kì 1 lớp 11, THPT Nguyễn Thị Minh Khai,Thành phố Hồ Chí Minh 2018]%[Bùi Mạnh Tiến]%[1D2B3-2]
	Tìm số hạng chứa $x^5$ trong khai triển $\left(2x^5-\dfrac{3}{x^3}\right)^n$, biết rằng $\mathrm{C}_{n+1}^{n-1}\cdot \left(\mathrm{A}_{2n}^1\right)^2=4n^3.$	\loigiai{
		Điều kiện: $\heva{&n\in \mathbb{N^*}\\&n\ge 2.}$\\
		Ta có 
		\begin{eqnarray*}
			\mathrm{C}_{n+1}^{n-1}\cdot \mathrm{A}_{n}^{2}-\left(\mathrm{A}_{2n}^{1}\right)^2=4n^3
			&\Leftrightarrow &	\dfrac{(n+1)n}{2}\cdot n(n-1)-(2n)^2=4n^3\\
			&\Leftrightarrow & (n+1)(n-1)-8n-8=0\\
			&\Leftrightarrow & \hoac{&n=9\text{ (nhận)}\\&n=-1 \text{ (loại).}}
		\end{eqnarray*}
		Ta có $\left(2x^5-\dfrac{3}{x^3}\right)^9=\displaystyle\sum\limits_{k=0}^{9} \mathrm{C}_9^k\cdot \left(2x^5\right)^{9-k}\cdot \left(-\dfrac{3}{x^3}\right)^k=\displaystyle\sum\limits_{k=0}^{9} \mathrm{C}_9^k\cdot (2)^{9-k}\cdot (-3)^k\cdot x^{45-8k}.$\\
		Số hạng chứa $x^5$ ứng với $k$ thỏa mãn $45-8k=5\Leftrightarrow k=5$.\\
		Số hạng cần tìm là $\mathrm{C}_9^5\cdot 2^4\cdot (-3)^5\cdot x^5=-489888\cdot x^5.$
	}
\end{bt}

\begin{bt}%[Đề thi học kì 1 lớp 11, THPT Nguyễn Thị Minh Khai,Thành phố Hồ Chí Minh 2018]%[Bùi Mạnh Tiến]%[1D3B1-1]
	Dùng qui nạp, chứng minh rằng với mọi số $n$ nguyên dương ta có
	\begin{center}
		$2\cdot 7+3\cdot 10+4\cdot 13+\cdots +(n+1)\cdot (3n+4)=(n+1)\cdot (n+2)^2-4.$
	\end{center}
	\loigiai{
		Ta cần chứng minh qui nạp biểu thức
		$$2\cdot 7+3\cdot 10+4\cdot 13+\cdots +(n+1)\cdot (3n+4)=(n+1)\cdot (n+2)^2-4\quad(1)$$
		\begin{itemize}
			\item Với $n=1$ thì vế trái của $(1)$ là $2\cdot 7=14$ và vế phải của $(1)$ là $(1+1)\cdot (1+2)^2-4=14.$\\
			Vậy $(1)$ đúng với $n=1.$
			\item Giả sử $(1)$ đúng với $n=k$ tức là $2\cdot 7+3\cdot 10+4\cdot 13+\cdots +(k+1)\cdot (3k+4)=(k+1)\cdot (k+2)^2-4.$
			\item Ta cần chứng minh $(1)$ cũng đúng với $n=k+1$ tức là cần chứng minh 
			$$2\cdot 7+3\cdot 10+4\cdot 13+\cdots +(k+1)(3k+4)+\left(k+2\right)\cdot \left(3k+7\right)=(k+2)\cdot \left(k+3\right)^2-4 \quad (2)$$
			Vế trái của $(2)$ bằng
			\begin{eqnarray*}
				(k+1)\cdot (k+2)^2-4+(k+2)\cdot (3k+7)
				&=&(k+2)\left[(k+1)\cdot (k+2)+(3k+7)\right]-4\\
				&=&(k+2)\cdot (k^2+6k+9)-4\\
				&=&(k+2)\cdot (k+3)^2-4.
			\end{eqnarray*}
		\end{itemize}
		Vậy biểu thức $(2)$ đúng.\\
		Theo nguyên lí qui nạp ta có $2\cdot 7+3\cdot 10+\cdots +(n+1)\cdot (3n+4)=(n+1)\cdot (n+2)^2-4,$ $\forall n\in \mathbb{N^*}.$
	}
\end{bt}

	Tìm số hạng đầu tiên $u_1$ và công sai $d$ của $1$ cấp số cộng $(u_n)$ biết
	\begin{center}
			$\heva{&u_3+u_5+u_7=45\\&\left(u_4\right)^2-\left(u_2\right)^2=108\\&u_1<0.}$
	\end{center}
	\loigiai{
		Ta có 
		\begingroup
		\allowdisplaybreaks
		\begin{eqnarray*}
			\heva{&u_3+u_5+u_7=45\\&\left(u_4\right)^2-\left(u_2\right)^2=108}
			&\Leftrightarrow &\heva{&(u_1+2d)+(u_1+4d)+(u_1+6d)=45\\&(u_1+3d)^2-(u_1+d)^2=108}\\
			&\Leftrightarrow & \heva{&u_1=15-4d\\&2d^2-15d+27=0}\\
			&\Leftrightarrow & \hoac{&\heva{&u_1=3\\&d=3}\text{ (loại)}\\&\heva{&u_1=-3\\&d=\dfrac{9}{2}}\text{ (nhận)}.} 
		\end{eqnarray*}
		\endgroup
	}
\end{bt}

\begin{bt}%[Đề thi học kì 1 lớp 11, THPT Nguyễn Thị Minh Khai,Thành phố Hồ Chí Minh 2018]%[Bùi Mạnh Tiến]%[1H2B1-2]%[1H2B1-3]%[1H2B3-2]%[1H2K3-4]
	Cho hình chóp $S.ABCD$ có mặt đáy $ABCD$ là hình thang, cạnh đáy lớn $AD=2BC$. Gọi $I,K$ lần lượt là trung điểm của $AD$ và $SI$. Trên cạnh $AB$ lấy điểm $M$ sao cho $AM=2MB$.
	\begin{enumerate}
		\item Tìm giao tuyến của $2$ mặt phẳng $(SAD)$ và $(KBC)$.
		\item Tìm giao điểm $J$ của đường thẳng $BC$ và mặt phẳng $(SKM)$.
		\item Gọi $G$ là trọng tâm $\triangle SAD$. Chứng minh rằng $JK\parallel (GMC).$
		\item Chứng minh thiết diện tạo bởi mặt phẳng $(KBC)$ với hình chóp $S.ABCD$ là $1$ hình bình hành.
	\end{enumerate}
	\loigiai
	{
		\begin{center}
			\begin{tikzpicture}[scale=1, line join = round, line cap = round]
			
			\tkzDefPoints{0/0/A,1/-3/B,5/-3/C, 8/0/D}
			\coordinate (S) at ($(A)+(0,5)$);
			\coordinate (I) at ($(A)!0.5!(D)$);
			\coordinate (K) at ($(S)!0.5!(I)$);
			\coordinate (E) at ($(S)!0.5!(A)$);
			\coordinate (F) at ($(S)!0.5!(D)$);
			\tkzCentroid(S,A,D)\tkzGetPoint{G}
			\tkzDefPointWith[linear,K=0.6666](A,B)\tkzGetPoint{M}
			\tkzInterLL(I,M)(B,C)\tkzGetPoint{J}
			\tkzInterLL(J,K)(M,S)\tkzGetPoint{T}
			\tkzInterLL(A,B)(J,T)\tkzGetPoint{R}
			\tkzInterLL(B,E)(K,J)\tkzGetPoint{U}
			\tkzLabelSegment[above](E,K){$d$}
						
			\tkzDrawPolygon(S,B,C,D)
			\tkzDrawSegments(S,C S,B U,E C,F J,B T,J S,A A,R)
			\tkzDrawSegments[dashed](A,D E,F S,I I,J M,C C,I T,K M,G R,B B,U)
			\tkzDrawPoints[fill = black](D,C,A,B,S,I,K,E,F,G,J,M)
			\tkzLabelPoints[above](S,K,I)
			\tkzLabelPoints[left](A,E,M)
			\tkzLabelPoints[right](D,F,G)
			\tkzLabelPoints[below](B,C,J)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item $\heva{&AD\parallel BC\\&AD\subset (SAD),\text{ }BC\subset (KBC)\\&K\in (SAD)\cap (KBC)}\Rightarrow (SAD)\cap (KBC)=d$ qua $K$ và song song với $AD$, $BC$.
			\item Trong mặt phẳng $(ABCD)$, gọi $J=BC\cap IM$.\\
			Từ $\heva{&J\in IM, \text{ }IM\subset (SKM)\\&J\in BC}\Rightarrow J=BC\cap (SKM).$
			\item Do $AI\parallel BJ$ nên theo định lí Thales ta có $\dfrac{MI}{MJ}=\dfrac{MA}{MB}=2\Rightarrow MI=2MJ.$\\
			Do $G$ là trọng tâm $\triangle SAD$ nên $GI=2GK$.\\
			Trong $\triangle KIJ$ có $\dfrac{MI}{MJ}=\dfrac{GI}{GK}=2$ nên theo định lí đảo của định lí Thales ta có $JK\parallel MG.$\\
			Mà $MG\subset (GMC)$ nên $JK\parallel (GMC).$ 
			\item Trong mặt phẳng $(SAD)$. Gọi $E$, $F$ lần lượt là giao điểm của $d$ với $SA$ và $SD$.\\
			Ta có $(KBC)\cap (SAB)=BE$, $(KBC)\cap (SAD)=EF$, $(KBC)\cap (SCD)=FC$, $(KBC)\cap (SBC)=CB$. Vậy thiết diện cần tìm là tứ giác $BEFC$.\\
			Ta có $EF$ đi qua trung điểm $K$ của $SI$ và $EF\parallel AD$ nên $EF$ là đường trung bình của tam giác $SAD$ suy ra $EF\parallel AD\parallel BC$ và $EF=\dfrac{1}{2}AD=BC.$\\
			Từ đây suy ra $EF\parallel BC$, $EF=BC$ hay $BEFC$ là hình bình hành. 
		\end{enumerate}
	}
