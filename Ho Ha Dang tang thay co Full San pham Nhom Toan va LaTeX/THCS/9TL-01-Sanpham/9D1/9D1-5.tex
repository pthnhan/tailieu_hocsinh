\section{BIẾN ĐỔI ĐƠN GIẢN BIỂU THỨC CHỨA CĂN THỨC BẬC HAI}
\subsection{Kiến thức cần nhớ}
\begin{enumerate}
\item Đưa thừa số ra ngoài dấu căn \\
Với $B\ge 0$, ta có $\sqrt{A^2B}=|A|\sqrt{B}=\begin{cases}
A\sqrt{B} &\text{nếu } A\ge 0\\
-A\sqrt{B} &\text{nếu } A<0.
\end{cases}$
\item Đưa thừa số vào trong dấu căn\\
	Với $B\ge 0$, ta có $A\sqrt{B}=\begin{cases}
\sqrt{A^2B} &\text{nếu } A\ge 0\\
-\sqrt{A^2B} &\text{nếu } A<0.
\end{cases}$
\item Khử mẫu của biểu thức lấy căn\\
Với $A,B$ mà $AB\ge 0$, $B\ne 0$ ta có $\sqrt{\dfrac{A}{B}}=\sqrt{\dfrac{AB}{B^2}}=\dfrac{\sqrt{AB}}{|B|}\cdot$
\item Trục căn thức ở mẫu
\begin{itemize}
\item Với $B>0$ ta có $\dfrac{A}{\sqrt{B}}=\dfrac{A\sqrt{B}}{B}\cdot$
\item Với $A\ge 0, A\ne B^2$ ta có $$\dfrac{C}{\sqrt{A}+B}=\dfrac{C\left(\sqrt{A}-B\right)}{A-B^2};\;\;\;\dfrac{C}{\sqrt{A}-B}=\dfrac{C\left(\sqrt{A}+B\right)}{A-B^2}\cdot$$
\item Với $A\ge 0, B\ge 0, A\ne B$ ta có $$\dfrac{C}{\sqrt{A}+\sqrt{B}}=\dfrac{C\left(\sqrt{A}-\sqrt{B}\right)}{A-B};\;\;\;\dfrac{C}{\sqrt{A}-\sqrt{B}}=\dfrac{C\left(\sqrt{A}+\sqrt{B}\right)}{A-B}\cdot$$
\end{itemize}
\end{enumerate}
\subsection{Các dạng toán}

\begin{dang}{Đưa thừa số ra ngoài dấu căn}
    \textbf{Phương pháp giải}: Cách đưa thừa số $A^2$ ra ngoài dấu căn:
$$\sqrt{A^2B}=|A|\sqrt{B}=\begin{cases}
A\sqrt{B} &\text{nếu } A\ge 0\\
-A\sqrt{B} &\text{nếu } A<0.
\end{cases} \text{ với $B\ge 0$}.$$
\end{dang}

\begin{vd}
Đưa thừa số ra ngoài dấu căn:
\begin{multicols}{2}
	\begin{enumerate}
		\item $\sqrt{8x^2}$ với $x\ge 0$;
		\item $\sqrt{27xy^2}$ với $x\ge 0$; $y\le 0$.
	\end{enumerate}
\end{multicols}

    \loigiai
    {
    	\begin{enumerate}[a)]
    		\item Vì $x\ge 0$ nên $\sqrt{8x^2}=\sqrt{2^2x^2\cdot 2}=2|x|\sqrt{2}=2x\sqrt{2}.$
    		\item Vì $x\ge 0$ và $y\le 0$ nên $\sqrt{27xy^2}=\sqrt{3^2y^2\cdot 3x}=3|y|\sqrt{3x}=-3y\sqrt{3x}.$
    	\end{enumerate}
    }
\end{vd}

\begin{vd}
Đưa thừa số ra ngoài dấu căn:

a) $\sqrt{25x^3}$ với $x> 0$;\hspace*{3cm} b) $\sqrt{48(x-1)^2y^4}$ với $x<1$; $y\in\mathbb{R}$.
    \loigiai{
a) $\sqrt{25x^3}=\sqrt{5^2x^2\cdot x}=5|x|\sqrt{x}=5x\sqrt{x}$ (do $x>0$)\\
b) $\sqrt{48(x-1)^2y^4}=4|x-1|y^2\sqrt{3}=4(1-x)y^2\sqrt{3}$ (do $x<1$).
    }
\end{vd}

\begin{vd}
Rút gọn các biểu thức sau:

a) $A=5\sqrt{4x}-3\sqrt{\dfrac{100x}{9}}-\dfrac{4}{x}\sqrt{\dfrac{x^3}{4}}$ với $x> 0$;

b) $B=\dfrac{1}{3}\sqrt{9+6y+y^2}+\dfrac{4y}{3}+5$ với $y\le -3$.
    \loigiai{
a) Vì $x>0$ nên ta có $$\begin{aligned}
A&=5\sqrt{4x}-3\sqrt{\dfrac{100x}{9}}-\dfrac{4}{x}\sqrt{\dfrac{x^3}{4}}\\ 
&=5\cdot 2\sqrt{x}-3\cdot \dfrac{10}{3}\sqrt{x}-\dfrac{4}{x}\cdot\dfrac{x}{2}\sqrt{x}\\
&=10\sqrt{x}-10\sqrt{x}-2\sqrt{x}=-2\sqrt{x}.
\end{aligned}$$
b) Vì $y\le -3$ nên $\sqrt{9+6y+y^2}=\sqrt{(3+y)^2}=|3+y|=-3-y$. Do đó $$B=\dfrac{1}{3}(-3-y)+\dfrac{4y}{3}+5=y+4.$$

    }
\end{vd}
\begin{dang}{Đưa thừa số vào trong dấu căn}
    \textbf{Phương pháp giải}: Cách đưa thừa số vào trong dấu căn:
$$A\sqrt{B}=\begin{cases}
\sqrt{A^2B} &\text{nếu } A\ge 0\\
-\sqrt{A^2B} &\text{nếu } A<0.
\end{cases}$$
\end{dang}

\begin{vd}
Đưa thừa số vào trong dấu căn:

a) $a\sqrt{11}$ với $a\ge 0$;\hspace*{3cm} b) $a\sqrt{\dfrac{-23}{a}}$ với $a< 0$.
    \loigiai{
a) Vì $a\ge 0$ nên $a\sqrt{11}=\sqrt{11a^2}.$\\
b) Vì $a<0$ nên $a\sqrt{\dfrac{-23}{a}}=-\sqrt{a^2\cdot\dfrac{-23}{a}}=-\sqrt{-23a}.$
    }
\end{vd}

\begin{vd}
Đưa thừa số vào trong dấu căn:

a) $\dfrac{a}{2}\sqrt{\dfrac{12}{a}}$ với $a> 0$;\hspace*{3cm} b) $(a-2)\sqrt{3}$ với $a<2$.
    \loigiai{
a) Vì $a>0$ nên $\dfrac{a}{2}\sqrt{\dfrac{12}{a}}=\sqrt{\dfrac{a^2}{2^2}\cdot\dfrac{12}{a}}=\sqrt{3a}$.\\
b) Vì $a<2$ nên $(a-2)\sqrt{3}=-\sqrt{3(2-a)^2}$.
    }
\end{vd}

\begin{dang}{Khử mẫu}
    \textbf{Phương pháp giải:}\\
    Vận dụng công thức $\sqrt{\dfrac{A}{B}}=\dfrac{\sqrt{AB}}{|B|}\ (A \cdot B \geq 0; B \neq 0).$ Cụ thể gồm các bước sau:
    \begin{itemize}
        \item Biến đổi mẫu thành bình phương của một số hoặc một biểu thức (nếu cần);
        \item Khai phương mẫu và đưa ra ngoài dấu căn.
    \end{itemize}
\end{dang}

\begin{vd}
    Khử mẫu của biểu thức lẫy căn 
    \begin{multicols}{2}
        \begin{enumerate}
            \item $\sqrt{\dfrac{5}{2}};$
            \item $\sqrt{\dfrac{3m}{5n}}$ với $m \cdot n >0;$
            \item $\sqrt{\dfrac{5}{14}};$
            \item $\sqrt{\dfrac{7x}{18y}}$ với $x \cdot y >0.$ 
        \end{enumerate}
    \end{multicols}
    \loigiai{
        \begin{enumerate}
            \item $\sqrt{\dfrac{5}{2}}=\sqrt{\dfrac{5.2}{3.2}}=\dfrac{\sqrt{10}}{\sqrt{2^2}}=\dfrac{\sqrt{10}}{2}.$
            \item $\sqrt{\dfrac{3m}{5n}}=\sqrt{\dfrac{3m\cdot 5n}{5n\cdot 5n}}=\dfrac{\sqrt{15mn}}{(5n)^2}=\dfrac{\sqrt{15mn}}{5|n|}.$
            \item $\sqrt{\dfrac{5}{14}}=\sqrt{\dfrac{5\cdot 14}{14 \cdot 14}}=\dfrac{\sqrt{70}}{\sqrt{14^2}}=\dfrac{\sqrt{70}}{14}.$
            \item $\sqrt{\dfrac{7x}{18y}}=\sqrt{\dfrac{7x\cdot 18y}{18y \cdot 18y}}=\dfrac{\sqrt{126xy}}{(18y)^2}=\dfrac{3\sqrt{14xy}}{18|y|}=\dfrac{\sqrt{14xy}}{6|y|}.$ 
        \end{enumerate}
        \begin{note} 
              Trong câu d) ta có cách khác xử lý bài toán đơn giản hơn\\
              $$\sqrt{\dfrac{7x}{18y}}=\sqrt{\dfrac{7x\cdot 2y}{18y \cdot 2y}}=\dfrac{\sqrt{14xy}}{(6y)^2}=\dfrac{\sqrt{14xy}}{6|y|}.$$ 
        \end{note}
    }
\end{vd}

\begin{vd} 
      Khử mẫu của biểu thức lấy căn
      \begin{multicols}{2}
            \begin{enumerate}
                \item $\sqrt{\dfrac{1}{x^3+3x^2+3x+1}},$ với $x > -1;$
                \item $\sqrt{\dfrac{1}{a^2}-\dfrac{1}{a^3}},$ với $a < 0$ hoặc $a \geq 1.$
            \end{enumerate}             
      \end{multicols}
      \loigiai{
         \begin{enumerate}
            \item $\sqrt{\dfrac{1}{x^3+3x^2+3x+1}}=\sqrt{\dfrac{1}{(x+1)^3}}=\sqrt{\dfrac{x+1}{(x+1)^4}}=\dfrac{\sqrt{x+1}}{(x+1)^2}.$
            \item $\sqrt{\dfrac{1}{a^2}-\dfrac{1}{a^3}}=\sqrt{\dfrac{a-1}{a^3}}=\sqrt{\dfrac{a(a-1)}{a^4}}=\dfrac{\sqrt{a(a-1)}}{a^2}$.
          \end{enumerate} 
      }
 \end{vd}

\begin{vd} 
    Khử mẫu của các biểu thức dưới dấu căn
    \begin{multicols}{2}
        \begin{enumerate}
            \item $-x^2y \sqrt{\dfrac{y}{x}}$ với $x >0; y \geq 0;$
            \item $\sqrt{\dfrac{-2x}{15}}$ với $x<0;$
            \item $\sqrt{\dfrac{5x}{13y}}$ với $x \geq 0; y>0;$
            \item $\sqrt{\dfrac{2x-1}{2x+1}}$ với $x\geq \dfrac{1}{2}.$
        \end{enumerate}         
    \end{multicols}     
    \loigiai{
         \begin{enumerate}
            \item $-x^2y \sqrt{\dfrac{y}{x}}=-x^2y \sqrt{\dfrac{xy}{x^2}}=\dfrac{-x^2y\sqrt{xy}}{x}=-xy \sqrt{xy}.$
            \item $\sqrt{\dfrac{-2x}{15}}=\sqrt{\dfrac{-30x}{15^2}}=\dfrac{\sqrt{-2x}}{15}.$
            \item $\sqrt{\dfrac{5x}{13y}}=\sqrt{\dfrac{65xy}{(13y)^2}}=\dfrac{\sqrt{65xy}}{13y}.$
            \item $\sqrt{\dfrac{2x-1}{2x+1}}=\sqrt{\dfrac{(2x-1)(2x+1)}{(2x+1)^2}}=\dfrac{\sqrt{4x^2-1}}{2x+1}.$
          \end{enumerate} 
      }  
\end{vd}


\begin{dang}{Trục căn thức ở mẫu}
    \textbf{Phương pháp giải:}\\
    Để trục căn thức ở mẫu, ta lựa chọn một trong hai cách sau:
    \begin{enumerate}[Cách 1:] 
        \item Phân tích tử và mẫu ra thừa số chung chứa căn rồi rút gọn thừa số đó.
        \item Nhân tử và mẫu với thừa số thích hợp để làm mất căn thức ở mẫu. Có các dạng cơ bản sau:
        \begin{enumerate}[1.]
            \item $\dfrac{A}{\sqrt{B}}=\dfrac{A \sqrt{B}}{B}\ (B>0). $
            \item $\dfrac{1}{\sqrt{A}\pm \sqrt{B}}=\dfrac{\sqrt{A}\mp \sqrt{B}}{A-B}$ với $A>0;B>0;A\neq B.$
        \end{enumerate}
    \end{enumerate}
    
\end{dang}

\begin{vd}
    Trục căn thức ở mẫu của các phân thức sau
    \begin{multicols}{2}
        \begin{enumerate}
        \item $\dfrac{3+\sqrt{3}}{3\sqrt{3}};$
        \item $\dfrac{5+\sqrt{5}}{\sqrt{5}+1};$
        \item $\dfrac{3 \sqrt{5}+\sqrt{15}}{\sqrt{5}}$
        \item $\dfrac{1}{2-3 \sqrt{3}}.$
    \end{enumerate}
    \end{multicols}
    \loigiai{
        \begin{enumerate}
            \item $\dfrac{3+\sqrt{3}}{3\sqrt{3}}=\dfrac{\sqrt{3}(\sqrt{3}+1)}{3\sqrt{3}}=\dfrac{\sqrt{3}+1}{3};$
            \item $\dfrac{5+\sqrt{5}}{\sqrt{5}+1}=\dfrac{\sqrt{5}(\sqrt{5}+1)}{\sqrt{5}+1}=\sqrt{5};$
            \item $\dfrac{3 \sqrt{5}+\sqrt{15}}{\sqrt{5}}=\dfrac{ (3 \sqrt{5}+\sqrt{15})\cdot \sqrt{5}}{\sqrt{5}\cdot \sqrt{5}}=\dfrac{15+5 \sqrt{3}}{5}=3+\sqrt{3};$
            \item $\dfrac{1}{2-3 \sqrt{3}}=\dfrac{2+3 \sqrt{3}}{(2-3 \sqrt{3})(2+3 \sqrt{3})}=\dfrac{2+3 \sqrt{3}}{4-27}=-\dfrac{2+3 \sqrt{3}}{23}.$
        \end{enumerate}
    }
\end{vd}

\begin{vd} 
        Trục căn thức ở mẫu của các phân thức sau, với các biểu thức đều có nghĩa.  
        \begin{multicols}{2}
            \begin{enumerate}
                \item $\dfrac{1}{\sqrt{m}+n};$
                \item $\dfrac{2}{\sqrt{m}-\sqrt{n}};$
                \item $\dfrac{3}{2\sqrt{m}+1};$
                \item $\dfrac{2ab}{2\sqrt{a}+3\sqrt{b}}.$
            \end{enumerate}
        \end{multicols}
\loigiai{
    \begin{enumerate}
        \item $\dfrac{1}{\sqrt{m}+n}=\dfrac{\sqrt{m}-n}{(\sqrt{m}-n)(\sqrt{m}+n)}=\dfrac{\sqrt{m}-n}{m-n^2};$
        \item $\dfrac{2}{\sqrt{m}-\sqrt{n}}=\dfrac{2(\sqrt{m}+\sqrt{n})}{(\sqrt{m}+\sqrt{n})(\sqrt{m}-\sqrt{n})}=\dfrac{2(\sqrt{m}-\sqrt{n})}{m-n};$
        \item $\dfrac{3}{2\sqrt{m}+1}=\dfrac{3(2\sqrt{m}-1)}{(2\sqrt{m}-1)(2\sqrt{m}+1)}=\dfrac{6\sqrt{m}-3}{4m-1};$
        \item $\dfrac{2ab}{2\sqrt{a}+3\sqrt{b}}=\dfrac{2ab(2\sqrt{a}-3\sqrt{b})}{(2\sqrt{a}+3\sqrt{b})(2\sqrt{a}-3\sqrt{b})}=\dfrac{2ab(2\sqrt{a}-3\sqrt{b})}{4a-9b}.$
    \end{enumerate}
           
}
\end{vd}

\begin{vd} 
   Trục căn thức ở mẫu của các phân thức sau  
    \begin{multicols}{2}
        \begin{enumerate}
            \item $\dfrac{1}{\sqrt{3}+\sqrt{2}-1};$
            \item $\dfrac{1}{\sqrt{14}-\sqrt{6+\sqrt{35}}}$.
        \end{enumerate}
    \end{multicols}
 \loigiai{

   \begin{enumerate}
        \item  
        { \allowbreak
        \begin{align*}
            \dfrac{1}{\sqrt{3}+\sqrt{2}-1}=\dfrac{\sqrt{3}+\sqrt{2}+1}{\left(\sqrt{3} + \sqrt{2}\right)^2-1}=\dfrac{\sqrt{3}+\sqrt{2}+1}{2 \sqrt{6}+4}&=\dfrac{(\sqrt{3}+\sqrt{2}+1)(2 \sqrt{6}-4)}{24-16}\\ &=\dfrac{(\sqrt{3}+\sqrt{2}+1)(2 \sqrt{6}-4)}{8};
        \end{align*}
        }
        \item  
        {\allowbreak
        \begin{align*}
                \dfrac{1}{\sqrt{14}-\sqrt{6+\sqrt{35}}}=\dfrac{\sqrt{2}}{2 \sqrt{7}-\sqrt{12+2 \sqrt{35}}} &=\dfrac{\sqrt{2}}{2 \sqrt{7}-\sqrt{(\sqrt{5}+\sqrt{7})^2}}\\ &=\dfrac{\sqrt{2}}{\sqrt{7}-\sqrt{5}}=\dfrac{\sqrt{2}(\sqrt{7}+\sqrt{5})}{7-5}=\dfrac{\sqrt{14}-\sqrt{10}}{2}.
            \end{align*}
        }
    \end{enumerate}
     
 }
\end{vd}

\begin{bt}

$$\begin{aligned}
(1)&\Leftrightarrow 3\sqrt{2x+1}-2\sqrt{2x+1}+\dfrac{1}{3}\sqrt{2x+1}=4\\ 
&\Leftrightarrow \sqrt{2x+1}=3 \Leftrightarrow 2x+1=9\Leftrightarrow x=4.

    }

\begin{bt}
Giải phương trình \[25\sqrt{\dfrac{x-3}{25}}-7\sqrt{\dfrac{4x-12}{9}}-7\sqrt{x^2-9}+18\sqrt{\dfrac{9x^2-81}{81}}=0.\tag{2}\]
    \loigiai{
\textbf{Cách 1}. Điều kiện $x\ge 3$.
$$\begin{aligned}
(2)&\Leftrightarrow \dfrac{1}{3}\sqrt{x-3}=\sqrt{x^2-9}\\ 
&\Leftrightarrow \sqrt{x-3}=3\sqrt{x^2-9}\Leftrightarrow \begin{cases}  x\ge 3\\(x-3)=9(x^2-9)  \end{cases}\Rightarrow x=3.
\end{aligned}$$
Vậy phương trình $(2)$ có nghiệm là $x=3$.\\
\textbf{Cách 2}. Điều kiện $x\ge 3$.
$$\begin{aligned}
(2)&\Leftrightarrow \dfrac{1}{3}\sqrt{x-3}=\sqrt{x^2-9}\\

&\Leftrightarrow \begin{cases}  x=3 &\text{(thỏa mãn)}\\ x=-\dfrac{26}{9} &\text{(loại)}  \end{cases}\Rightarrow x=3.
\end{aligned}$$
Vậy phương trình $(2)$ có nghiệm là $x=3$.
    }
\end{bt}
\begin{bt}

M&=\left(4x^5+4x^4-5x^3+2x-2\right)^{2018}+2019\\

    }

\begin{bt}

	\loigiai{

	}

\begin{bt}
	Đưa thừa số vào trong dấu căn của biểu thức $(1-x)\sqrt{\dfrac{x}{x-1}}$ với $x>1$, ta được
	\choice
	{$\sqrt{x(x-1)}$}
	{$\sqrt{x(1-x)}$}
	{\True $-\sqrt{x(x-1)}$}
	{$-\sqrt{x(1-x)}$}
	\loigiai{
		Vì $x>1$ nên  $(1-x)\sqrt{\dfrac{x}{x-1}}=-(x-1)\sqrt{\dfrac{x}{x-1}}=-\sqrt{(x-1)^2\cdot\dfrac{x}{x-1}}=-\sqrt{x(x-1)}.$
	}
\end{bt}

\begin{bt}

\begin{bt}
	Trục căn thức của biểu thức $\dfrac{m}{\sqrt{5m^3}},\ m>0$ ta được

	}
