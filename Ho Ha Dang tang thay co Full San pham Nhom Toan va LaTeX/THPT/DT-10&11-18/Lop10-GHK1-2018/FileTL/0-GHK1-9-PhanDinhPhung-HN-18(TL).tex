\begin{bt}%[Thi GHK1, Phan Đình Phùng - Hà Nội, 2018]%[Lê Mạnh Thắng, EX-DT-10&11]%[0D2B1-2]
	Tìm tập xác định của hàm số:
	\begin{enumerate}
		\item $y=\sqrt{6-x}+\dfrac{2x+1}{1+\sqrt{x-1}}$.
		\item $y=\sqrt{|x^2-25|}$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Hàm số xác định khi $\heva{&6-x\geq 0\\&x-1\geq 0} \Leftrightarrow 1\leq x \leq 6$.\\
		Do đó, tập xác định $\mathscr{D}=\left[1;6\right]$.
		\item Hàm số xác định khi $|x^2-25| \geq 0 \Leftrightarrow x\in \mathbb{R}$.\\
		Vậy, tập xác định $\mathscr{D}=\mathbb{R}$.
	\end{enumerate}
	}
\end{bt}

\begin{bt}%[Thi GHK1, Phan Đình Phùng - Hà Nội, 2018]%[Lê Mạnh Thắng, EX-DT-10&11]%[0D2B1-4]
	Đồ thị hàm số $y=\dfrac{x^4+3x^2-2}{x^2-2}$ có trục đối xứng hay tâm đối xứng? Vì sao?
	\loigiai{
		Hàm số xác định khi $x^2-2\neq 0 \Leftrightarrow \heva{&x\neq \sqrt{2}\\&x\neq -\sqrt{2}}$.	Do đó, tập xác định $\mathscr{D}=\mathbb{R}\setminus \left\lbrace-\sqrt{2};\sqrt{2}\right\rbrace$.\\
		Với mọi $x\in\mathscr{D}$, ta có:
		\begin{itemize}
			\item $x \in \mathscr{D} \Rightarrow -x\in \mathscr{D}$.
			\item $f(-x)=\dfrac{(-x)^4+3(-x)^2-2}{(-x)^2-2}=\dfrac{x^4+3x^2-2}{x^2-2}=f(x)$
		\end{itemize}
	Suy ra, hàm số đã cho là hàm số chẵn. Đồ thị của hàm số chẵn nhận trục tung làm trục đối xứng.
	}
\end{bt}

\begin{bt}%[Thi GHK1, Phan Đình Phùng - Hà Nội, 2018]%[Lê Mạnh Thắng, EX-DT-10&11]%[0D2B2-2][0D2B2-4]
	Cho đường thẳng $d\colon y=f(x)=ax+b$.
	\begin{enumerate}
		\item Viết phương trình $d$ biết đồ thị của hàm số đi qua $A(1;4)$ và song song với đường thẳng $y=2x+1$.
		\item Lập bảng biến thiên và vẽ đồ thị của hàm số $y=|f(x)|$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Đường thẳng $d\colon y=ax+b$ song song với đường thẳng $y=2x+1$ nên $d\colon y=2x+b$ $(b\neq 1)$.\\
			mà $A(1;4) \in d$, suy ra $4=2\cdot 1+b \,\Leftrightarrow\, b=2$.\\
			Vậy $d\colon y=2x+2$.
			\item Hàm số $y=|f(x)|=|2x+2|=\heva{&2x+2, x\geq-1\\&-2x-2, x<-1}$.\\
			Bảng biến thiên:

				{$-\infty$,$-1$,$+\infty$}
				\tkzTabVar{+/$+\infty$,-/$0$,+/$+\infty$}
				\end{tikzpicture}
			\end{center}
		Đồ thị của hàm số $y=|f(x)|=|2x+2|$
			\begin{center}

	}

\begin{bt}%[Thi GHK1, Phan Đình Phùng - Hà Nội, 2018]%[Lê Mạnh Thắng, EX-DT-10&11]%[0D2B3-2][0D2B3-3][0D2K3-1][0D2K3-4]
	Cho hàm số $y=ax^2+bx+c$ có đồ thị là parabol.
	\begin{enumerate}
		\item Xác định hàm số biết rằng parabol có đỉnh $I(2;-1)$ và cắt trục tung tại điểm có tung độ bằng $3$.
		\item Lập bảng biến thiên và vẽ đồ thị của hàm số $(P)\colon y=f(x)=x^2-4x+3$.
		\item Tìm giá trị lớn nhất và giá trị nhỏ nhất của hàm số $y=f(x)=x^2-4x+3$ trên đoạn $[-1;3]$.
		\item Tìm $m$ để $d\colon y=2x+3m$ cắt $(P)$ tại hai điểm phân biệt cùng nằm về bên phải của trục tung.
		\item Tùy theo $m$ hãy biện luận số giao điểm của hai đồ thị $y=m$ và $y=|f(x)|$.
	\end{enumerate}
	\loigiai{
			\begin{enumerate}
			\item Ta có $\heva{&I(2;-1)\in(P)\\&A(0;3)\in(P)\\&x=-\dfrac{b}{2a}=2} \Leftrightarrow \heva{&4a+2b+c=-1\\&c=3\\&4a+b=0} \Leftrightarrow \heva{&4a+2b=-4\\&4a+b=0\\&c=3} 
			\Leftrightarrow \heva{&a=1\\&b=-4\\&c=3.}$\\
			Vậy parabol của hàm số $y=x^2-4x+3$.
			\item Đỉnh $I(2;-1)$, trục đối xứng $x=2$.\\
			Bảng biến thiên:
			\begin{center}
				\begin{tikzpicture}[>=stealth, scale=1]
				\tkzTabInit[lgt=1.3,espcl=2.5,deltacl=0.6]
				{$x$/.7,$f(x)$/2}
				{$-\infty$,$2$,$+\infty$}
				\tkzTabVar{+/$+\infty$,-/$-1$,+/$+\infty$}
				\end{tikzpicture}
			\end{center}
			Đồ thị của $(P)$:
			\begin{center}
				\begin{tikzpicture}[>=stealth,x=.8cm,y=.8cm,scale=1]
				\draw[->] (-2,0)--(0,0)
				node[shift={(-0.3,-0.3)}]{$O$}--(5,0) node[below]{$x$};
				\draw[->] (0,-1.5)--(0,4.5) node[right]{$y$};
				\draw plot[domain=-0.3:4.3, samples=200] (\x, {(\x)^2-4*(\x)+3});
				\draw[dashed] (0,-1)--(2,-1);
				\draw[dashed] (2,-1.5)--(2,4.2);
				\draw (1,0)node[below]{$1$} (3,0)node[below]{$3$};
				\draw (0,-1) node[below left]{$-1$};
				\draw (0,3) node[right]{$3$};
				\fill (0,0) circle (1pt);
				\end{tikzpicture}
			\end{center}
			\item Ta có $y(-1)=8$, $y(3)=0$. Dựa vào đồ thị của $(P)$, ta có
			\begin{itemize}
				\item Hàm số đạt giá trị nhỏ nhất bằng $-1$ tại $x=2$.
				\item Hàm số đạt giá trị lớn nhất bằng $8$ tại $x=-1$.
			\end{itemize}
			\item Phương trình hoành độ giao điểm của $(P)$ và $d$ là
				\[x^2-4x+3=2x+3m \,\Leftrightarrow\, x^2-6x+3-3m=0.\tag{1}\]
				$(P)$ và $d$ cắt nhau tại hai điểm phân biệt nằm bên phải trục tung khi và chỉ khi phương trình $(1)$ có hai nghiệm phân biệt dương, nghĩa là
				\[\heva{&\Delta'>0\\&S>0\\&P>0} \Leftrightarrow \heva{&9-(3-3m)>0\\&6>0\\&3-3m>0}\Leftrightarrow -2<m<1.\]
			\item Ta có $y=|f(x)|=\heva{&f(x), \ \text{nếu} \ f(x)\geq 0\\&-f(x), \ \text{nếu} \ f(x)<0}$\\
			Suy ra, đồ thị hàm số $y=|f(x)|$ có được bằng cách giữ lại phần đồ thị của $y=f(x)$ nằm trên trục $Ox$ và lấy đối xứng qua trục $Ox$ phần đồ thị của $y=f(x)$ nằm dưới trục $Ox$.\\
			Đồ thị hàm số $y=|f(x)|$ như hình bên dưới
			\begin{center}
				\begin{tikzpicture}[>=stealth,x=.8cm,y=.8cm,scale=1]
				\draw[->] (-2,0)--(0,0)
				node[shift={(-0.3,-0.3)}]{$O$}--(5,0) node[below]{$x$};
				\draw[->] (0,-1)--(0,4.5) node[right]{$y$};
				\draw plot[domain=-0.3:4.3, samples=200] (\x, {abs((\x)^2-4*(\x)+3)});
				\draw[dashed] (0,1)--(2,1);
				\draw[dashed] (2,-0.5)--(2,4.2);
				\draw (1,0)node[below]{$1$} (3,0)node[below]{$3$};
				\draw (0,1) node[below left]{$1$};
				\draw (0,3) node[right]{$3$};
				\fill (0,0) circle (1pt);
				\end{tikzpicture}
			\end{center}
			Dựa vào đồ thị hàm số $y=|f(x)|$, ta có
			\begin{itemize}
				\item $m<0$: hai đồ thị không có giao điểm.
				\item $m=0$: hai đồ thị có hai giao điểm.
				\item $0<m<1$: hai đồ thị có bốn giao điểm.
				\item $m=1$: hai đồ thị có ba giao điểm.
				\item $m>1$: hai đồ thị có hai giao điểm.
			\end{itemize}
			\end{enumerate}
	}
\end{bt}

\begin{bt}%[Thi GHK1, Phan Đình Phùng - Hà Nội, 2018]%[Lê Mạnh Thắng, EX-DT-10&11]%[0D2G3-4]
	Chứng minh parabol $y=2x^2-4(2m-1)x+8m^2-3$ luôn tiếp xúc với một đường thẳng cố định.
	\loigiai{
		Gọi $\Delta\colon y=ax+b$ là đường thẳng cố định luôn tiếp xúc $(P)$.\\
		Khi đó, phương trình
		\begin{eqnarray*}
			&&2x^2-4(2m-1)x+8m^2-3=ax+b, \ \text{có nghiệm kép} \ \forall m\\
			&\Leftrightarrow& 2x^2-(8m-4+a)x+8m^2-3-b=0, \ \text{có nghiệm kép} \ \forall m\\
			&\Leftrightarrow& \Delta = 0, \forall m\\
			&\Leftrightarrow& (8m-4+a)^2-8(8m^2-3-b) = 0, \forall m\\
			&\Leftrightarrow& 16(a-4)m+(a-4)^2+8(b+3) = 0, \forall m\\
			&\Leftrightarrow& \heva{&a-4=0\\&(a-4)^2+8(b+3)=0} \;\Leftrightarrow\; \heva{&a=4\\&b=-3.}
		\end{eqnarray*}
		Vậy parabol đã cho luôn tiếp xúc với đường thẳng $y=4x-3$.
	}
\end{bt}
