\begin{ex}%[Đề thi Chuyên ĐHSPHN năm 2011, Vòng 1]%[8D1G5]
Cho biểu thức\\
$A=\left(\dfrac{x-y}{2y-x}+\dfrac{x^{2}+y^{2}+y-2}{2y^{2}+xy-x^{2}}\right):\dfrac{4x^{4}+4x^{2}y+y^{2}-4}{x^{2}+y+xy+x}$ (với $x>0$, $y>0$, $x\neq 2y$, $y\neq 2-2x^{2}$).
\begin{enumerate}
\item Rút gọn biểu thức $A$.
\item Cho $y=1$. Hãy tìm $x$ để $A=\dfrac{2}{5}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $A=\left(\dfrac{x-y}{2y-x}+\dfrac{x^{2}+y^{2}+y-2}{(x+y)(2y-x)}\right).\dfrac{(x+y)(x+1)}{(2x^{2}+y-2)(2x^{2}+y+2)}$\\
$=\dfrac{2x^{2}+y-2}{2y-x}.\dfrac{x+1}{(2x^{2}+y-2)(2x^{2}+y+2)}=\dfrac{x+1}{(2y-x)(2x^{2}+y+2)}$.
\item Với $y=1$, ta có $A=\dfrac{x+1}{(2-x)(2x^{2}+3)}=\dfrac{2}{5}\Leftrightarrow 4x^{3}-8x^{2}+11x-7=0\Leftrightarrow x=1$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi Chuyên ĐHSPHN năm 2011, Vòng 1]%[9D4K8]
Một nhóm công nhân đặt kế hoạch sản xuất $200$ sản phẩm. Trong $4$ ngày đầu họ thực hiện đúng mức đề ra, những ngày còn lại họ làm vượt mức mỗi ngày $10$ sản phẩm, nên đã hoàn thành sớm $2$ ngày. Hỏi theo kế hoạch mỗi ngày nhóm công nhân cần sản xuất bao nhiêu sản phẩm?
\loigiai{
Gọi số sản phẩm phải làm theo kế hoạch là $x$ (sản phẩm) ($x >0$, $x\in\mathbb N$).\\
Số ngày làm theo kế hoạch là $\dfrac{200}{x}$ (ngày).\\
Số ngày làm $x+10$ sản phẩm là $\dfrac{200-4x}{x+10}$ (ngày).\\
Vì nhóm công nhân làm vượt $2$ ngày nên ta có phương trình\\
$\dfrac{200}{x}-\dfrac{200-4x}{x+10}=2\Rightarrow x=20$ (thỏa mãn).\\
Vậy theo kế hoạch mỗi ngày nhóm công nhân làm $20$ sản phẩm.
}
\end{ex}
\begin{ex}%[Đề thi Chuyên ĐHSPHN năm 2011, Vòng 1]%[9D4K6]
Cho Parabol $(P): y= x^{2}$ và đường thẳng $(d): y=mx-m2+3$ ($m$ là tham số). Tìm tất cả các giá trị $m$ để đường thẳng $(d)$ cắt parabol $(P)$ tại hai điểm phân biệt có hoành độ $x_{1}$, $x_{2}$. Với giá trị nào của $m$ thì $x_{1}$, $x_{2}$ là độ dài cạnh góc vuông của tam giác vuông có độ dài cạnh huyền bằng $\sqrt{\dfrac{5}{2}}$.
\loigiai{
Xét phương trình hoành độ giao điểm $x^{2}-mx+m^{2}-3=0$ (1). Để $(P)$ cắt $(d)$ tai 2 điểm phân biệt thì $(1)$ phải có $2$ nghiệm phân biệt $\Leftrightarrow \Delta >0\Leftrightarrow 12-3m^{2}>0\Leftrightarrow -2<m<2$.\\
Theo Viet ta có $\heva{&x_{1}+x_{2}=m\\ &x_{1}x_{2}=m^{2}-3}$. Vì $x_{1}$, $x_{2}$ là độ dài các cạnh của tam giác nên\\
$\heva{&x_{1}+x_{2}=m>0\\ &x_{1}x_{2}=m^{2}-3>0}\Rightarrow m>\sqrt{3}$.\\
Vì $x_{1}^{2}+x_{2}^{2}=\dfrac{5}{2}\Leftrightarrow (x_{1}+x_{2})^{2}-2x_{1}x_{2}=\dfrac{5}{2}\Leftrightarrow 2m^{2}=7\Leftrightarrow m=\pm \dfrac{\sqrt{14}}{2}$.\\
Vì $m>\sqrt{3}$, $-2<m<2$ nên $m=\dfrac{\sqrt{14}}{2}$.
}
\end{ex}
\begin{ex}%[Đề thi Chuyên ĐHSPHN năm 2011, Vòng 1]%[9H3B1]
Cho đường tròn $(O)$ đường kính $AB=10$. Dây cung $CD$ vuông góc với $AB$ tại điểm $E$ sao cho $AE =1$. Các tiếp tuyến tại $B$ và $C$ của đường tròn $(O)$ cắt nhau tại $K$, $AK$ và $CE$ cắt nhau tại $M$.
\begin{enumerate}
\item Chứng minh tam giác $AEC$ đồng dạng với tam giác $OBK$. Tính $BK$.
\item Tính diện tích tam giác $CKM$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[line cap=round,line join=round,>=stealth,x=1.0cm,y=1.0cm, scale=0.7, thick]
\tkzInit[xmax=17,xmin=-1,ymax=11,ymin=-1]
\tkzDefPoint(0,0){K}
\tkzDefShiftPoint[K](0:15){B}
\tkzDefShiftPoint[B](90:10){A}
\tkzDrawCircle[diameter](A,B)\tkzGetPoint{O}
\tkzTangent[from=K](O,A)\tkzGetSecondPoint{C}
\tkzDefLine[orthogonal=through C](B,A)\tkzGetPoint{c}
\tkzInterLC(C,c)(O,A)\tkzGetSecondPoint{D}
\tkzInterLL(B,A)(C,D)\tkzGetPoint{E}
\tkzInterLL(A,K)(C,D)\tkzGetPoint{M}
\tkzDrawSegments(A,B A,K C,D C,O K,B K,C K,O C,A)
\pgfresetboundingbox
\tkzLabelPoints[above](A)
\tkzLabelPoints[above left](C)
\tkzLabelPoints[right](O,D)
\tkzLabelPoints[below right](E)
\tkzLabelPoints[below](B,K,M)
\tkzMarkRightAngle(O,C,K)
\tkzMarkRightAngle(A,E,D)
\tkzMarkRightAngle(O,B,K)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Theo tính chất góc ngoài tam giác cân $AOC$ thì $\widehat{COB}=2\widehat{OAC}\Rightarrow \widehat{BOK}=\widehat{OAC}$\\
$\Rightarrow AC\parallel OK$. Mà $CE\parallel KB$ nên $\Delta AEC\backsim\Delta OBK$ (g.g) $\Rightarrow \dfrac{BK}{CE}=\dfrac{OB}{AE}$.\\
Xét tam giác vuông $OEC$, áp dụng Pitago có $CE=\sqrt{5^{2}-4^{2}}=3\Rightarrow BK=15$.
\item Ta có $\Delta AEM\backsim\Delta ABK\Rightarrow \dfrac{AE}{AB}=\dfrac{ME}{BK}\Rightarrow ME=\dfrac{3}{2}\Rightarrow CM=CE-ME=\dfrac{3}{2}$\\
$\Rightarrow S_{\Delta CKM}=\dfrac{1}{2}.CM.BE=\dfrac{27}{4}$ (đvdt).
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi Chuyên ĐHSPHN năm 2011, Vòng 1]%[9H3G8]
Cho hình thoi $ABCD$ có góc $\widehat{BAD} =120^{\circ}$. Các điểm $M$, $N$ chạy trên $BC$ và $CD$ tương ứng sao cho góc $MAN=30^{\circ}$. Chứng minh rằng tâm đường tròn ngoại tiếp tam giác $MAN$ chạy trên đường thẳng cố định.
\loigiai{
\begin{center}
\begin{tikzpicture}[line cap=round,line join=round,>=stealth,x=1.0cm,y=1.0cm, scale=0.7, thick]
\tkzInit[xmax=17,xmin=-1,ymax=11,ymin=-1]
\def\r{4}
\tkzDefPoint(0,0){O}
\tkzDefShiftPoint[O](110:\r){A}
\tkzDefPointBy[homothety=center O ratio -1.163](A)\tkzGetPoint{C}
\tkzDefPointBy[rotation=center A angle -60](C)\tkzGetPoint{B}
\tkzDefPointBy[rotation=center A angle 60](C)\tkzGetPoint{D}
\tkzDrawCircle[radius](O,A)
\tkzTangent[from=C](O,A)\tkzGetPoints{M}{N}
\tkzDrawSegments(A,B B,C C,D D,A A,C A,M A,N O,M O,N M,N)
\pgfresetboundingbox
\tkzLabelPoints[above](A,D)
\tkzLabelPoints[left](B,O)
\tkzLabelPoints[right](M)
\tkzLabelPoints[below](C,N)
\tkzMarkRightAngle(O,N,C)
\tkzMarkRightAngle(C,M,O)
\end{tikzpicture}
\end{center}
Gọi $O$ là tâm đường tròn ngoại tiếp tam giác $AMN\Rightarrow \Delta MON$ cân tại $O$.\\
Mà $\widehat{MON}=\widehat{MAN}= 60^{\circ}$ nên tam giác $MAN$ đều.\\
Mặt khác $\widehat{MON}+\widehat{MCN}=60^{\circ}+120^{\circ}=180^{\circ}\Rightarrow$ tứ giác $MCNO$ nội tiếp $\Rightarrow\widehat{MCN}=\widehat{MNA}=60^{\circ}$. Vì $ABCD$ là hình thoi nên $\widehat{MCA}=60^{\circ}$. Vậy $\widehat{MCA}=\widehat{MCO}$ nên $O$ thuộc $AC$.
}
\end{ex}
\begin{ex}%[Đề thi Chuyên ĐHSPHN năm 2011, Vòng 1]%[9D5G1]
Chứng minh bất đẳng thức
\[
\dfrac{1}{\sqrt{1}+\sqrt{2}}+\dfrac{1}{\sqrt{3}+\sqrt{4}}+\dfrac{1}{\sqrt{5}+\sqrt{6}}+...+\dfrac{1}{\sqrt{79}+\sqrt{80}}>4
\]
\loigiai{
Đặt $A=\dfrac{1}{\sqrt{1}+\sqrt{2}}+\dfrac{1}{\sqrt{3}+\sqrt{4}}+\dfrac{1}{\sqrt{5}+\sqrt{6}}+...+\dfrac{1}{\sqrt{79}+\sqrt{80}}$.
\begin{align*}
\Rightarrow 2A&=\dfrac{2}{\sqrt{1}+\sqrt{2}}+\dfrac{2}{\sqrt{3}+\sqrt{4}}+\dfrac{2}{\sqrt{5}+\sqrt{6}}+...+\dfrac{2}{\sqrt{79}+\sqrt{80}}\\
\Rightarrow 2A&>\dfrac{1}{\sqrt{1}+\sqrt{2}}+\dfrac{1}{\sqrt{2}+\sqrt{3}}+\dfrac{1}{\sqrt{3}+\sqrt{4}}+...+\dfrac{1}{\sqrt{79}+\sqrt{80}}+\dfrac{1}{\sqrt{80}+\sqrt{81}}\\ 
&>\dfrac{\sqrt{2}-\sqrt{1}}{(\sqrt{2}+\sqrt{1})(\sqrt{2}-\sqrt{1})}+\dfrac{\sqrt{3}-\sqrt{2}}{(\sqrt{3}+\sqrt{2})(\sqrt{3}-\sqrt{2})}+...+\dfrac{\sqrt{81}-\sqrt{80}}{(\sqrt{81}+\sqrt{80})(\sqrt{81}-\sqrt{80})}\\ 
&=\sqrt{2}-\sqrt{1}+\sqrt{3}-\sqrt{2}+...+\sqrt{81}-\sqrt{80}=9-1=8.\\
\Rightarrow A&>4.
\end{align*}
}
% +\dfrac{\sqrt{4}-\sqrt{3}}{(\sqrt{4}+\sqrt{3})(\sqrt{4}-\sqrt{3})}
\end{ex}
