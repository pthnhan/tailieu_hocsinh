\subsection{Các dạng toán và công thức giải nhanh}

\begin{dang}{Xác định tọa độ tâm đường tròn nội tiếp tam giác khi biết tọa độ $3$ đỉnh tam giác}
Cho tam giác $ABC$ ngoại tiếp đường tròn tâm $I$. Khi đó, tọa độ tâm $I$ được xác định theo công thức sau\\
\impicinpar{
$\heva{&x_I=\dfrac{x_A \cdot BC+x_B \cdot AC+x_C \cdot AB}{AB+AC+BC}\\ &y_I=\dfrac{y_A \cdot BC+y_B \cdot AC+y_C \cdot AB}{AB+AC+BC}}$}{
\begin{tikzpicture}[scale=.22]
\tkzInit[xmin=0,xmax=8,ymin=-10,ymax=6]
\tkzDefPoints{3/5/A,1/0/B,7/0/C}
\tkzDefCircle[circum](A,B,C)     \tkzGetPoint{I} %tâm đường tròn
\tkzDrawCircle(I,B)
\tkzDrawSegments(A,B B,C C,A)
\tkzDrawPoints(A,B,C,I)
\tkzLabelPoints[left](B,I)
\tkzLabelPoints[right](C)
\tkzLabelPoints[above left](A)
%\tkzLabelPoints[below]()
\end{tikzpicture}
}
\end{dang}
\begin{vd}
Cho tam giác $ABC$ sao cho $A(-2;3), B\left(\dfrac{1}{4};0 \right), C(2;0)$. Tìm tọa độ $I$ là tâm đường tròn nội tiếp tam giác $ABC$.\\
\choice
{$\left( \dfrac{1}{3};\dfrac{1}{3} \right) $}
{$\left( \dfrac{1}{4};\dfrac{1}{4} \right) $} 
{\True $\left( \dfrac{1}{2};\dfrac{1}{2} \right)$} 
{$(1;1)$} 
\loigiai{
\textbf{Cách 1}: Sử dụng công thức tính nhanh.\\
$\vec{AB}=\left( \dfrac{9}{4};-3 \right) \Rightarrow AB=\dfrac{15}{4}$\\ 
$\vec{AC}=(4;-3) \Rightarrow AC=5$\\ 
$\vec{BC}=\left( \dfrac{7}{4};0 \right) \Rightarrow BC=\dfrac{7}{4}$\\ 
Áp dụng công thức trên, ta được\\
$\heva{&x_I=\dfrac{(-2)\cdot \frac{7}{4}+\frac{1}{4}\cdot 5+2\cdot \frac{15}{4}}{\frac{7}{4}+5+\frac{15}{4}} = \dfrac{1}{2} \\ &y_I=\frac{3\cdot \frac{7}{4}+0\cdot 5+0\cdot \frac{15}{4}}{\frac{7}{4}+5+\frac{15}{4}} = \dfrac{1}{2}}$\\
\textbf{Cách 2}: Giải tự luận thông thường.\\
Gọi $AD$ là tia phân giác $\widehat{BAC}$\\
$\vec{AB}=(\dfrac{9}{4};-3) \Rightarrow AB=\dfrac{15}{4}$\\ 
$\vec{AC}=(4;-3) \Rightarrow AC=5$\\ 
Khi đó $-\dfrac{AB}{AC}=-\dfrac{3}{4}$\\
Suy ra $\vec{DB}=-\dfrac{3}{4}\cdot \vec{DC} \Rightarrow D(1;0)$\\
$\vec{BD}=(\dfrac{3}{4};0) \Rightarrow BD=\dfrac{3}{4}$\\ 
Khi đó $-\dfrac{AB}{BD}=-5$\\
Suy ra $\vec{IA}=-5\cdot \vec{ID} \Rightarrow I(\dfrac{1}{2};\dfrac{1}{2})$\\
}
\end{vd}
\begin{vd}
Cho tam giác $ABC$ sao cho $A(2;1), B(1;0), C(2;0)$. Tìm tọa độ $I$ là tâm đường tròn nội tiếp tam giác $ABC$.\\
\choice
{\True $\left( \dfrac{2+\sqrt{2}}{2};\dfrac{2-\sqrt{2}}{2}\right)$}
{$\left( \dfrac{2-\sqrt{2}}{2};\dfrac{2-\sqrt{2}}{2}\right)$}
{$\left( \dfrac{2-\sqrt{2}}{2};\dfrac{2+\sqrt{2}}{2}\right)$}
{$\left( \dfrac{2+\sqrt{2}}{2};\dfrac{2+\sqrt{2}}{2}\right)$}
\loigiai{
\textbf{Cách 1}: Sử dụng công thức tính nhanh.\\
$\vec{AB}=(-1;1) \Rightarrow AB=\sqrt{2}$\\ 
$\vec{AC}=(0;-1) \Rightarrow AC=1$\\ 
$\vec{BC}=(1;0) \Rightarrow BC=1$\\ 
Áp dụng công thức trên, ta được\\
$\heva{&x_I=\dfrac{2\cdot 1+1\cdot 1+2\cdot \sqrt{2}}{\sqrt{2}+1+1} = \dfrac{2+\sqrt{2}}{2} \\ &y_I=\dfrac{1\cdot 1+0\cdot 1+0\cdot \sqrt{2}}{\sqrt{2}+1+1}=\dfrac{2-\sqrt{2}}{2}}$\\
\textbf{Cách 2}: Giải tự luận thông thường.\\
Kẻ $AD$ là tia phân giác góc $\widehat{BAC}$\\
$\vec{AB}=(-1;1) \Rightarrow AB=\sqrt{2}$\\ 
$\vec{AC}=(0;-1) \Rightarrow AC=1$\\ 
Khi đó $-\dfrac{AB}{AC}=-\sqrt{2}$\\
Suy ra $\vec{DB}=-\sqrt{2}\cdot \vec{DC} \Rightarrow D(3-\sqrt{2};0)$\\
$\vec{BD}=(2-\sqrt{2};0) \Rightarrow BD=2-\sqrt{2}$\\ 
Khi đó $-\dfrac{AB}{BD}=-1-\sqrt{2}$\\
Suy ra $\vec{IA}=(-1-\sqrt{2})\cdot \vec{ID} \Rightarrow I\left( \dfrac{2+\sqrt{2}}{2};\dfrac{2-\sqrt{2}}{2}\right) $\\
}
\end{vd}

\begin{dang}{Tìm bán kính đường tròn ngoại tiếp tam giác khi biết tọa độ $3$ đỉnh tam giác}
	Cho tam giác $ABC$ bất kỳ. Khi đó bán kính đường tròn ngoại tiếp tam giác $ABC$ được xác định bởi công thức sau (đưa vào không gian ba chiều với cao độ bằng $0$)\\
	$$R=\dfrac{AB\cdot BC\cdot CA}{2\mid [ \vec{AB},\vec{AC} ] \mid }$$
\end{dang}
\begin{vd}
	Cho tam giác $ABC$ sao cho $A(2;1), B(1;0), C(2;0)$ nội tiếp đường tròn tâm $I$. Tìm bán kính đường tròn ngoại tiếp tam giác $ABC$.\\
\choice
{\True $\dfrac{\sqrt{2}}{2}$}
{$1$}
{$\sqrt{2}$}
{$2$}
\loigiai{
\textbf{Cách 1}: Sử dụng công thức tính nhanh.\\
Ta có:\\
$\vec{AB}=(-1;1;0) \Rightarrow AB=\sqrt{2}.$\\
$\vec{AC}=(0;-1;0) \Rightarrow AC=1.$\\ 
$\vec{BC}=(1;0;0) \Rightarrow AC=1.$\\ 
$\mid [\vec{AB},\vec{AC}] \mid = 1.$\\
Do đó, theo công thức tính nhanh\\
$R=\dfrac{\sqrt{2}\cdot 1\cdot 1}{2\cdot 1}=\dfrac{\sqrt{2}}{2}.$\\
\textbf{Cách 2}: Giải tự luận thông thường.\\
Gọi $I(a;b)$\\
$\vec{IA}=(2-a;1-b)\Rightarrow IA^{2}=a^{2}+b^{2}-4a-2b+5.$\\
$\vec{IB}=(1-a;-b)\Rightarrow IB^{2}=a^{2}+b^{2}-2a+1.$\\
$\vec{IC}=(2-a;-b)\Rightarrow IC^{2}=a^{2}+b^{2}-4a+4.$\\
Ta có hệ phương trình sau:
\begin{eqnarray*}
&&\heva{&a^{2}+b^{2}-4a-2b+5 = a^{2}+b^{2}-2a+1 \\ &a^{2}+b^{2}-4a-2b+5 = a^{2}+b^{2}-4a+4}.\\
&\Leftrightarrow &\heva {& a=\dfrac{3}{2} \\ &b=\dfrac{1}{2}}
\end{eqnarray*}
Khi đó $IA^{2}=\dfrac{1}{2} \Leftrightarrow R = IA = \dfrac{\sqrt{2}}{2}.$
}
\end{vd}
\begin{vd}
	Cho tam giác $ABC$ sao cho $A(2;1), B(3;0), C(1;-2)$ nội tiếp đường tròn tâm $I$. Tìm bán kính đường tròn ngoại tiếp tam giác $ABC$.\\
\choice
{\True $\dfrac{\sqrt{10}}{2}$}
{$\dfrac{1}{\sqrt{2}}$}
{$\sqrt{2}$}
{$1$}
\loigiai{
\textbf{Cách 1}: Sử dụng công thức tính nhanh.\\
Ta có:\\
$\vec{AB}=(1;-1;0) \Rightarrow AB=\sqrt{2}.$\\
$\vec{AC}=(-1;-3;0) \Rightarrow AC=\sqrt{10}.$\\ 
$\vec{BC}=(-2;-2;0) \Rightarrow AC=2\sqrt{2}.$\\ 
$\mid [\vec{AB},\vec{AC}] \mid = 4.$\\
Do đó, theo công thức tính nhanh\\
$R=\dfrac{\sqrt{2}\cdot \sqrt{10}\cdot 2\sqrt{2}}{2\cdot 4}=\dfrac{\sqrt{10}}{2}.$\\
\textbf{Cách 2}: Giải tự luận thông thường.\\
Gọi $I(a;b)$\\
$\vec{IA}=(2-a;1-b)\Rightarrow IA^{2}=a^{2}+b^{2}-4a-2b+5$.\\
$\vec{IB}=(3-a;-b)\Rightarrow IB^{2}=a^{2}+b^{2}-6a+9$.\\
$\vec{IC}=(1-a;-2-b)\Rightarrow IC^{2}=a^{2}+b^{2}-2a+4b+5$.\\
Ta có hệ phương trình sau: 
\begin{eqnarray*}
&&\heva{&a^{2}+b^{2}-4a-2b+5 = a^{2}+b^{2}-6a+9 \\ &a^{2}+b^{2}-4a-2b+5 = a^{2}+b^{2}-2a+4b+5}.\\
&\Leftrightarrow &\heva {& a=\dfrac{3}{2} \\ &b=-\dfrac{1}{2}}.
\end{eqnarray*}
Khi đó $IA^{2}=\dfrac{5}{2} \Leftrightarrow R = IA = \dfrac{\sqrt{10}}{2}$.
	} 
\end{vd}
\begin{dang}{Tìm phương trình tiếp tuyến của đường tròn tại điểm $M_0(x_0;y_0)$ thuộc đường tròn}
	Nếu đường tròn có phương trình dạng: $(x-a)^2+(y-b)^2=R^2$ thì phương trình tiếp tuyến tại $M_0$ là:
	$$(x-a)(x_0-a)+(y-b)(y_0-b)=R^2$$
	Nếu đường tròn có phương trình dạng: $x^2+y^2-2ax-2by+c=0$ thì phương trình tiếp tuyến tại $M_0$ là:
	$$xx_0+yy_0-a(x+x_0)-b(y+y_0)+c=0$$
\end{dang}
\begin{vd}
	Cho đường tròn $(C)$ có phương trình $(x-2)^2+(y-3)^2=25$ và điểm $M(-2;6)$ thuộc đường tròn. \\
	Viết phương trình tiếp tuyến của đường tròn $(C)$ tại $M$.
\choice
{\True $4x-3y+26=0$}
{$-4x+3y+26=0$}
{$-4x-3y+26=0$}
{$4x+3y+26=0$}
\loigiai{
\textbf{Cách 1}: Sử dụng công thức tính nhanh.\\
Theo công thức tính nhanh, ta có phương trình\\
$$(x-2)(-2-2)+(y-3)(6-3)=25 \Leftrightarrow 4x-3y+26=0$$\\
\textbf{Cách 2}: Giải tự luận thông thường.\\
Ta có tâm của đường tròn  $I(2;3)$\\
$\vec{IM}=(-4;3)$\\
Phương trình tiếp tuyến $\Delta$ cần tìm có dạng: $-4(x+2)+3(y-6)=0 \Leftrightarrow 4x-3y+26=0$\\
}
\end{vd}
