\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[Đề thi học kì 1, Bình Lục, Hà Nam 2017-2118]%[Dương Xuân Lợi-11HK1-18]%[1D1Y2-1]%[1D1B3-1]%[1D1K3-5]
	Giải các phương trình sau:
		\begin{enumerate}
			\item $\sin x=\sin \dfrac{\pi}{5}$.
			\item $2\cos^2x+3\cos x+1=0$.
			\item $\dfrac{\sin2x\cos2x+4\sin x\cos^2x-3\sin 2x+\cos 2x-2\cos x+1}{\cos2x-1}=2$.
		\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item  $\sin x=\sin \dfrac{\pi}{5} \Leftrightarrow \hoac{&x=\dfrac{\pi}{5}+k2\pi\\&x=\dfrac{4\pi}{5}+k2\pi},k \in \mathbb{Z}.$
			\item $2\cos^2x+3\cos x+1=0 \Leftrightarrow \hoac{&\cos x=-1\\&\cos x=\dfrac{-1}{2}}\Leftrightarrow \hoac{&x=\pi+k2\pi\\&x=\pm\dfrac{2\pi}{3}+k2\pi},k\in \mathbb{Z}.$
			\item Điều kiện $x\ne l\pi$, $l\in \mathbb{Z}$.
				\begin{eqnarray*}
					& &\dfrac{\sin2x\cos2x+4\sin x\cos^2x-3\sin 2x+\cos 2x-2\cos x+1}{\cos2x-1}=2\\
					& \Leftrightarrow &\sin2x\cos2x+4\sin x\cos^2x-3\sin 2x-\cos 2x-2\cos x+3=0\\
					& \Leftrightarrow &\left(\sin2x\cos2x-\cos2x\right)+(-3\sin 2x+3)+\left(4\sin x\cos^2x-2\cos x\right)=0\\
					& \Leftrightarrow &\cos2x\left(\sin2x-1\right)-3(\sin 2x-1)+2\cos x\left(\sin2x-1\right)=0\\
					& \Leftrightarrow &\left(\sin2x-1\right)\left(\cos2x+2\cos x-3\right)=0\\
					& \Leftrightarrow &\left(\sin2x-1\right)\left(2\cos^2x+2\cos x-4\right)=0\\
					& \Leftrightarrow &\hoac{&\sin2x=1\\&\cos x=1\\&\cos x=-2 \text{ (vô nghiệm)}}\\
					& \Leftrightarrow &\hoac{&x=\dfrac{\pi}{4}+k\pi\\&x=k2\pi},k\in\mathbb{Z}.
				\end{eqnarray*}
				Đối chiếu với điều kiện, phương trình có nghiệm là $x=\dfrac{\pi}{4}+k\pi$, $k \in \mathbb{Z}$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề thi học kì 1, Bình Lục, Hà Nam 2017-2118]%[Dương Xuân Lợi-11HK1-18]%[1D2Y4-1]%[1D2B5-2]
	Đội tuyển học sinh giỏi khối $11$ của trường THPT B Bình Lục có $10$ học sinh nữ và $7$ học sinh nam. Xét phép thử ban giám hiệu cần chọn ngẫu nhiên $4$ học sinh tham dự trại hè.
		\begin{enumerate}
			\item Tính số phần tử của không gian mẫu.
			\item Tính xác suất của biến cố chọn được ít nhất một học sinh nữ.
		\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Chọn $4$ học sinh từ $17$ học sinh, ta có $\mathrm{C}_{17}^4$ cách nên $n\left(\Omega\right)=\mathrm{C}_{17}^4$.
			\item Gọi biến cố $A$: \lq\lq chọn được ít nhất một học sinh nữ\rq\rq.\\
				  Nên biến cố $\overline{A}$: \lq\lq chọn được $4$ học sinh nam\rq\rq.\\
				  Chọn $4$ bạn nam từ $7$ học sinh nam có $\mathrm{C}_7^4$ cách suy ra $n\left(\overline{A}\right)=\mathrm{C}_7^4$.\\
				  Vì vậy $\mathrm{P}\left(\overline{A}\right)=\dfrac{n\left(\overline{A}\right)}{n\left(\Omega\right)}=\dfrac{1}{68}$ nên $\mathrm{P}(A)=\dfrac{67}{68}$.				  
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề thi học kì 1, Bình Lục, Hà Nam 2017-2118]%[Dương Xuân Lợi-11HK1-18]%[1H2B3-3]%[1H2B3-2]%[1H2K3-4]
	Cho hình chóp $S.ABCD$ có đáy là hình thang với đáy lớn $AB=3CD$.
		\begin{enumerate}
			\item Tìm giao tuyến của hai mặt phẳng (SAB) và (SCD).
			\item Gọi $E$, $F$ lần lượt là trung điểm $AD$, $BC$. Chứng minh rằng đường thẳng $EF$ song song với mặt phẳng $(SAB)$.
			\item Gọi $G$ là trọng tâm tam giác $SAB$. Xác định thiết diện của hình chóp cắt bởi mặt phẳng $(EFG)$. Thiết diện là hình gì?
		\end{enumerate}
	\loigiai{
		\begin{center}\begin{tikzpicture}[scale=1, line join=round, line cap=round]
		\tkzDefPoints{0/0/A,6/0/B,3.2/-1.3/C,1.2/-1.3/D}
		\coordinate (S) at ($(A)+(0.4,3.5)$);
		\coordinate (P) at ($(A)+(-0.4,3.5)$);
		\coordinate (Q) at ($(A)+(5,3.5)$);
		\coordinate (E) at ($(A)!0.5!(D)$);
		\coordinate (F) at ($(B)!0.5!(C)$);
		\coordinate (M) at ($(A)!0.5!(B)$);
		\coordinate (G) at ($(S)!2/3!(M)$);
		\coordinate (I) at ($(S)!2/3!(B)$);
		\coordinate (J) at ($(S)!2/3!(A)$);
		\tkzDrawPolygon(S,B,C,D)
		\tkzDrawSegments(S,C F,I P,Q A,S A,D J,E)
		\tkzDrawSegments[dashed](A,B E,F S,M G,E G,F F,I I,J)
		\tkzDrawPoints[fill=black,size=4](D,C,A,B,S,E,F,M,G,I,J)
		\tkzLabelPoints[above](S)
		\tkzLabelPoints[above right](G,I)
		\tkzLabelPoints[below](B,C,D,F,M)
		\tkzLabelPoints[below left](E)
		\tkzLabelPoints[left](A,J)
		\tkzLabelSegment(P,Q){$\Delta$}		
		\end{tikzpicture}\end{center}
		\begin{enumerate}
			\item Ta có $\heva{&S \in (SAB) \cap (SCD)\\&AB \subset (SAB),CD \subset (SCD)\\&AB \parallel CD}$ nên giao tuyến của hai mặt phẳng $(SAB)$ và $(SCD)$ là đường thẳng $\Delta$ đi qua điểm $S$ và song song với $AB$, $CD$.
			\item Ta có $\heva{&EF\parallel AB\\&AB\subset (SAB)\\&EF \not\subset (SAB)} \Rightarrow EF \parallel (SAB)$.
			\item Ta có $\heva{&G \in (SAB) \cap (GEF)\\&AB \subset (SAB),EF \subset (GEF)\\&AB \parallel EF}$ nên giao tuyến của hai mặt phẳng $(SAB)$ và $(GEF)$ là đường thẳng đi qua điểm $G$, song song với $AB$, $EF$ và cắt $SB$, $SA$ tại $I$, $J$.\\
			Thiết diện của hình chóp cắt bởi mặt phẳng $(EFG)$ là tứ giác $EFIJ$.\\
			Theo cách xác định thiết diện $EF \parallel IJ$ và $IJ=\dfrac{2}{3}AB$.\\
			Mặt khác $EF$ là đường trung bình của hình thang $ABCD$ nên $$EF=\dfrac{AB+CD}{2}=\dfrac{AB+\dfrac{1}{3}AB}{2}=\dfrac{2}{3}AB.$$
			Vậy thiết diện là hình bình hành.						
		\end{enumerate}
	}
