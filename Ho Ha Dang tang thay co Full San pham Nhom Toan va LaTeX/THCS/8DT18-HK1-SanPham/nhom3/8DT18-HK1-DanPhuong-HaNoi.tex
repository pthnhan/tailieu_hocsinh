	\begin{enumerate}
		\item Phân tích đa thức thành nhân tử $x\left( x-y \right) + 2\left( x-y \right)$.
		\item Tính nhanh giá trị của biểu thức $x^2-6xy + 9y^2$ tại $x = 16$ và $y = 2$.
		\item Tìm $x$, biết $2x\left( x-5 \right)-x\left( 2x + 3 \right) = 26$. 
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có 
			$$x\left( x-y \right) + 2\left( x-y \right) = \left( x+2 \right) \cdot \left( x-y \right).$$
			\item Ta có $x^2-6xy + 9y^2 = \left( x-3y\right)^2$.\\
			Thay $x = 16$ và $y = 2$ ta được $\left( x-3y\right)^2 = \left( 16-3 \cdot 2\right)^2=10^2=100$.\\
			Vậy khi $x = 16$ và $y = 2$ thì biểu thức $x^2-6xy + 9y^2$ có giá trị là $100$.
			\item Ta có
			\begin{eqnarray*}
				2x\left( x-5 \right)-x\left( 2x + 3 \right) = 26 & \Leftrightarrow & 2x^2-10x-\left( 2x^2+3x\right) = 26\\
				& \Leftrightarrow & 2x^2-10x-2x^2-3x=26\\
				& \Leftrightarrow & -13x=26\\
				& \Leftrightarrow & x=-2.
			\end{eqnarray*}
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Đan Phượng, Hà Nội]%[Lê Minh Đức, dự án 8DT18-HK1]%[8D2Y3]%[8D2Y8]%[8D2B5]
	\begin{enumerate}
		\item Rút gọn biểu thức $\dfrac{x^2 + xy}{x^2-y^2}$.
		\item Thực hiện phép tính $\dfrac{4x + 12}{(x+1)^2}:\dfrac{3(x+3)}{x + 1}$.
		\item Thực hiện phép tính $\dfrac{4}{x + 2} + \dfrac{3}{x-2} + \dfrac{-5x-2}{x^2-4}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Điều kiện xác định $x \ne \pm y$.\\
			Với $x$, $y$ thỏa mãn điều kiện xác định, ta có
			$$\dfrac{x^2 + xy}{x^2-y^2} = \dfrac{x \cdot (x+y)}{(x+y) \cdot (x-y)} = \dfrac{x}{(x-y)}.$$
			\item Điều kiện xác định $x \ne -1$; $x \ne -3$.\\
			Với $x$ thỏa mãn điều kiện xác định, ta có
			$$\dfrac{4x + 12}{(x+1)^2}:\dfrac{3(x+3)}{x + 1} = \dfrac{4(x+3)}{(x+1)^2} \cdot \dfrac{x+1}{3(x+3)}=\dfrac{4}{3(x+1)}.$$
			Vậy $\dfrac{4x + 12}{(x+1)^2}:\dfrac{3(x+3)}{x + 1} = \dfrac{4}{3(x+1)}$.
			\item Điều kiện xác định $x \ne \pm 2$.\\
			Với $x$ thỏa mãn điều kiện xác định, ta có
			\begin{eqnarray*}
				\dfrac{4}{x + 2} + \dfrac{3}{x-2} + \dfrac{-5x-2}{x^2-4} & = & \dfrac{4(x-2)}{(x+2)(x-2)}+\dfrac{3(x+2)}{(x-2)(x+2)}+ \dfrac{-5x-2}{(x-2)(x+2)}\\
				& = & \dfrac{4x-8}{(x+2)(x-2)}+\dfrac{3x+6}{(x-2)(x+2)}+ \dfrac{-5x-2}{(x-2)(x+2)}\\
				& = & \dfrac{4x-8+3x+6-5x-2}{(x+2)(x-2)} = \dfrac{2x-4}{(x+2)(x-2)}\\
				& = & \dfrac{2(x-2)}{(x-2)(x+2)} = \dfrac{2}{x+2}.
			\end{eqnarray*}
			Vậy $\dfrac{4}{x + 2} + \dfrac{3}{x-2} + \dfrac{-5x-2}{x^2-4}= \dfrac{2}{x+2}.$
		\end{enumerate}
	}
\end{bt}

	Cho hai đa thức $A = 2x^2 + 3x + 3$ và $B = 2x-1$.
	\begin{enumerate}
		\item Thực hiện phép chia $A$ cho $B$. 
		\item Tìm các giá trị nguyên của $x$ để giá trị của đa thức $A$ chia hết cho giá trị của đa thức $B$. 
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Thực hiện phép chia $A$ cho $B$ bằng cách đặt phép chia
			\begin{center}
				\polylongdiv[style=D]{2x^2 + 3x + 3}{2x-1}
			\end{center}
			\item Ta có $\dfrac{A(x)}{B(x)}=x+2+\dfrac{5}{2x-1}$.\\
			Điều kiện $x \ne \dfrac{1}{2}$.\\
			Để giá trị của đa thức $A(x)$ chia hết cho giá trị của đa thức $B$ thì $\dfrac{5}{2x-1}$ phải là số nguyên.\\
			Mà $x$ là số nguyên nên $2x-1$ là ước của $5$.\\
			$\Rightarrow 2x-1 \in \left\{ -5;-1;1;5 \right\} $.\\
			Ta có bảng sau
			\begin{center}
				\begin{tabular}{|c|c|c|c|c|}
					\hline 
					$2x-1$ & $-5$ & $-1$ & $1$ & $5$ \\ 
					\hline 
					$x$ & $-2$ & $0$ & $1$ & $3$ \\
					\hline 
					So với điều kiện & thỏa mãn & thỏa mãn & thỏa mãn & thỏa mãn \\
					\hline 
				\end{tabular}
			\end{center}
			Vậy với $x \in \left\{ -2; 0; 1; 3 \right\}$ thì giá trị của đa thức $A$ chia hết cho giá trị của đa thức $B$.
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Đan Phượng, Hà Nội]%[Lê Minh Đức, dự án 8DT18-HK1]%[8H1B4]%[8H1B9]%[8H1K1]%[8H1G4]
	Cho tam giác $ABC$ cân tại $A$. Gọi $H$, $K$ lần lượt là trung điểm của $BC$ và $AC$.
	\begin{enumerate}
		\item Chứng minh tứ giác $ABHK$ là hình thang;
		\item Trên tia đối của tia $HA$ lấy điểm $E$ sao cho $H$ là trung điểm của cạnh $AE$. Chứng minh tứ giác $ABEC$ là hình thoi;
		\item Qua $A$ vẽ đường thẳng vuông góc với $AH$ cắt tia $HK$ tại $D$. Chứng minh $AD = BH$;
		\item Vẽ $HN \perp AB$ $\left( N \in AB \right)$, gọi $I$ là trung điểm của $AN$. Trên tia đối của tia $BH$ lấy điểm $M$ sao cho $B$ là trung điểm của $HM$. Chứng minh $MN \perp HI$. 
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=1, line join=round, line cap=round,>=stealth]
			\tikzset{label style/.style={font=\footnotesize}}
			\tkzDefPoints{0/0/A,3/-5/B,-3/-5/C,-3/0/D}
			\tkzDefMidPoint(B,C)\tkzGetPoint{H}
			\tkzDefMidPoint(A,C)\tkzGetPoint{K}
			\tkzDefPointBy[ rotation = center H angle 180](A) \tkzGetPoint{E}
			\tkzDefPointBy[ rotation = center B angle 180](H) \tkzGetPoint{M}
			\tkzDefMidPoint(A,H)\tkzGetPoint{1}
			\tkzCalcLength[cm](1,A) \tkzGetLength{Tempt}
			\path[name path=dt] (1) circle (\Tempt cm);
			\draw[name path=AB] (A)--(B);
			\path [name intersections={of=AB and dt,by={3,N}}];
			\draw[name path=HN] (H)--(N);
			\tkzDefMidPoint(A,N)\tkzGetPoint{I}
			\draw[name path=HI] (H)--(I);
			\draw[name path=MN] (M)--(N);
			\tkzInterLL(H,I)(M,N) \tkzGetPoint{J}
			\tkzDefMidPoint(H,N)\tkzGetPoint{G}
			\tkzInterLL(G,I)(H,B) \tkzGetPoint{4}
			\tkzDrawSegments(B,C C,A H,K A,E B,E C,E A,D D,K J,N B,M I,4 B,G)
			\tkzMarkRightAngles[size=0.2](D,A,H)
			\tkzMarkRightAngles[size=0.2](H,N,B)
			\tkzMarkRightAngles[size=0.2](I,4,B)
			\tkzDrawPoints[fill=black](A,B,C,H,K,E,D,M,N,I,J,G)
			\tkzLabelPoints[below right=-2pt](B,H,M)
			\tkzLabelPoints[above right](A,N,I)
			\tkzLabelPoints[left](C,K,E,D,G,J)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Xét tam giác $ABC$ có $H$, $K$ lần lượt là trung điểm của $BC$ và $AC$ nên $HK$ là đường trung bình trong $\triangle ABC$ (dấu hiệu nhận biết).\\
			$\Rightarrow HK \parallel AB$.\\
			Xét tứ giác $ABHK$ có $HK \parallel AB$ (chứng minh trên) nên $ABHK$ là hình thang (dấu hiệu nhận biết hình thang).
			\item Vì $H$ là trung điểm của cạnh $AE$ (giả thiết) và $H$ là trung điểm của $BC$ (giả thiết) suy ra tứ giác $ABEC$ có $2$ đường chéo cắt nhau tại trung điểm của mỗi đường nên $ABEC$ là hình bình hành (dấu hiệu nhận biết hình bình hành).\\
			Tam giác $ABC$ cân tại $A$ có $AH$ là trung tuyến nên $AH$ đồng thời là đường cao 
			$$\Rightarrow AH \perp BC \Rightarrow AE \perp BC.$$
			$\Rightarrow ABEC$ là hình bình hành có $2$ đường chéo vuông góc với nhau.\\
			$\Rightarrow ABEC$ là hình thoi (dấu hiệu nhận biết hình thoi).
			\item Dễ dàng chứng minh được $\triangle CKH = \triangle AKD$ (góc cạnh góc).\\
			$\Rightarrow AD=CH$ ($2$ cạnh tương ứng của $2$ tam giác bằng nhau).  \hfill$(1)$\\
			Mà $H$ là trung điểm của $BC$ (giả thiết) $\Rightarrow CH=BH$.  \hfill$(2)$\\
			Từ $(1)$ và $(2)$ suy ra $AD=BH$ (điều phải chứng minh).
			\item Gọi trung điểm của $HN$ là $G$, vì $I$ là trung điểm của $AN$ (giả thiết) $\Rightarrow IG$ là đường trung bình trong tam giác $AHN$ suy ra $IG \parallel AH$.\\
			Mà $AH \perp BC$ nên $IG \perp BC$ hay $IG \perp HB$.  \hfill$(3)$\\
			Lại có $HN \perp AB$ (giả thiết) hay $HN \perp IB$.  \hfill$(4)$\\
			Từ $(3)$ và $(4)$ suy ra $G$ là trực tâm tam giác $HIB$ $\Rightarrow BG \perp HI$.  \hfill$(5)$\\
			Mà $B$ là trung điểm của $HM$ (giả thiết), $G$ là trung điểm $HN$ (do ta lấy) $\Leftrightarrow BG$ là đường trung bình trong tam giác $HMN$ suy ra $BG \parallel MN$.  \hfill$(6)$\\
			Từ $(5)$ và $(6)$ suy ra $MN \perp HI$, suy ra điều phải chứng minh.
		\end{enumerate}
	}
\end{bt}

	Cho $x$, $y$, $z$ là ba số thỏa mãn điều kiện
	$$4x^2 + 2y^2 + 2z^2-4xy-4xz + 2yz-6y-10z + 34 = 0.$$
	Tính $S = \left( x-4 \right)^{2017} + \left( y-4 \right)^{2017} + \left( z-4 \right)^{2017}$ .
	\loigiai{
		Ta có
		\allowdisplaybreaks
		\begin{eqnarray*}
			& & 4x^2 + 2y^2 + 2z^2-4xy-4xz + 2yz-6y-10z + 34 = 0 \\
			& \Leftrightarrow & 4x^2 + y^2+z^2+2yz-4x(y+z)+y^2-6y+9+z^2-10z+25=0 \\
			& \Leftrightarrow & 4x^2-4x(y+z)+(y+z)^2+(y-3)^2+(z-5)^2=0 \\
			& \Leftrightarrow & (2x-y-z)^2+(y-3)^2+(z-5)^2=0 \\
			& \Leftrightarrow & \heva{& 2x-y-z=0\\& y-3=0\\&z-5=0} \Leftrightarrow \heva{&x=4\\&y=3\\&z=5.}
		\end{eqnarray*}
		Thay vào biểu thức ta có
		$$ S= (4-4)^{2017}+(3-4)^{2017}+(5-4)^{2017}=0^{2017}-1^{2017}+1^{2017}=0-1+1=0.$$
		Vậy với $x$, $y$, $z$ là ba số thỏa mãn điều kiện
		$$4x^2 + 2y^2 + 2z^2-4xy-4xz + 2yz-6y-10z + 34 = 0$$
		thì $S = \left( x-4 \right)^{2017} + \left( y-4 \right)^{2017} + \left( z-4 \right)^{2017} = 0$.
	}
