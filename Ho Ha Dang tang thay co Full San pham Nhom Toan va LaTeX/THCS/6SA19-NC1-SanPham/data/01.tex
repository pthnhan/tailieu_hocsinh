
\section{Các phép tính về số tự nhiên}
\subsection{Kiến thức cơ bản}
\begin{enumerate}
	\item Tập hợp các số tự nhiên là $\mathbb{N}=\{0;1;2;3 \ldots \}$.\\
	Tập hợp các số tự nhiên khác $0$ là $\mathbb{N}^*=\{1;2;3;4 \ldots \}$.
	\item Với phép cộng và phép nhân, cần nhớ các tính chất giao hoán và kết hợp. Giữa phép nhân và phép cộng có tính chất phân phối:
	$$ a(b+c)=ab+ac. $$
	\item Điều kiện để có hiệu $a-b$ là $a\geq b$.
	\item Số tự nhiên $a$ chia hết cho số tự nhiên $b$ khác $0$ nếu có số tự nhiên $k$ sao cho $a=bk$.
	\begin{itemize}
		\item Trong phép chia hết ta có
		$$ \text{Số bị chia}=\text{Số chia} \times \text{Thương} .$$
		\item Trong phép chia có dư ta có
		$$ \text{Số bị chia}=\text{Số chia} \times \text{Thương}+\text{Số dư} $$
		$$ a=bk+r \ \text{với} \ 0<r<b. $$
	\end{itemize}
	\item Lũy thừa bậc $n$ của $a$ là tích của $n$ thừa số bằng nhau, mỗi thừa số bằng $a$.\\
	Công thức nhân, chia hai lũy thừa cùng cơ số:
	$$ a^m\cdot a^n=a^{m+n} $$
	$$ a^m \colon a^n=a^{m-n} \ (a\ne 0, m\geq n).$$
	\item Quy ước về thứ tự thực hiện phép tính:\\
	Lũy thừa $\to$ Nhân, chia $\to$ Cộng, trừ (đối với biểu thức không có dấu ngoặc);\\
	$ ( \ )  \to [ \ ] \to \{ \ \}$ (đối với biểu thức có dấu ngoặc).
\end{enumerate}

\begin{vd}%[8D3Y2]
	Tính hiệu $a-b$ biết:\\
	$a= 1\cdot 2+2\cdot 3+3\cdot 4+\cdots+98\cdot99$ và
	$b=1^2+2^2+3^2+\cdots+98^2.$
	\loigiai{
		Ta có:
		\begin{eqnarray*}
			a&=&1\cdot 2+2\cdot 3+3\cdot 4+\cdots+98\cdot99\\
			&=& 1(1+1)+2(2+1)+3(3+1)+\cdots+98(98+1)\\
			&=& 1^2+1+2^2+2+3^2+3+\cdots+98^2+98\\
			&=& b+(1+2+3+\cdots+98)\\
			&=& b+(1+98)\cdot 98 :2\\
			&=&b+4851.
		\end{eqnarray*}	
		Vậy $a-b=4851$.	
	}
\end{vd}

\begin{dang}{TÌM SỐ CHƯA BIẾT}
\end{dang}

\begin{vd}%[8D3B2]
	Tìm các số tự nhiên $x$ và $y$ ($x<y$) sao cho $2^x+2^y=20.$
	\loigiai{
		Xét các lũy thừa của $2$ nhỏ hơn $20$, ta có
		$2^0=1$, $2^1=2$, $2^2=4$, $2^3=8$, $2^4=16$.\\
		Chỉ có hai số có tổng bằng $20$ là $4+16$, tức là $2^2+2^4=20$. 
		Vậy $x=2$ và $y=4$.
	}
\end{vd}

\begin{vd}%[8D3Y2]
	Tìm một phép chia có dư, số bị chia bằng $24$, thương bằng $3$. Tìm số chia và số dư.
	\loigiai{
		Gọi số chia là $b$, số dư là $r$, ta có: $24=3b+r$ với $0<r<b$.\\
		Từ $r=24-3b$ và $r>0$ suy ra $3b<24$ nên $b<8$.\hfill$(1)$\\	
		Từ $r=24-3b$ và $r<b$ suy ra $24-3b<b$ nên $24<4b$, do đó $b>6$. \hfill$(2)$\\
		Do $b$ là số tự nhiên nên từ $(1)$ và $(2)$ suy ra $b=7$. Do đó $r=24-3\cdot 7=3$.\\
		Vậy số chia bằng $7$, số dư bằng $3$.
	}
\end{vd}

\begin{vd}%[8D3Y2]
	Tìm số tự nhiên có hai chữ số, biết rằng nếu viết thêm chữ số $4$ vào trước chữ số hàng chục thì được số $A$, nếu viết thêm chữ số $8$ vào sau chữ số hàng đơn vị thì được số $B$, trong đó $B$ gấp đôi $A$.
	\loigiai{
		\textbf{Cách 1:} \\
		Gọi số phải tìm là $\overline{ab}$, ta có phép tính
		\begin{center}
			\begin{tikzpicture}[scale=0.65]
			\draw(4.0,-0.5)node[left]{\normalsize $4\,a\,b$ };
			\draw(4.0,-1.5)node[left]{\normalsize $2$};
			\draw(1.2,-1)node[right]{\normalsize $\times$};
			\draw(2.0,-1.9)--(3.8,-1.9);
			\draw(4.0,-2.5)node[left]{$a\,b\,8$};
			\end{tikzpicture}
		\end{center}
		Xét phép nhân ở hàng trăm được $2\cdot 4\le a$ nên $a\ge 8$ \hfill$(1)$.\\
		Xét phép nhân ở cột hàng đơn vị được $2b=8$ hoặc $2b=18$, tức là $b=4$ hoặc $b=9$.
		\begin{itemize}
			\item Trường hợp $b=4$, ta có phép tính 	
			\begin{center}
				\begin{tikzpicture}[scale=0.65]
				\draw(4.0,-0.5)node[left]{\normalsize $4\,a\,4$ };
				\draw(4.0,-1.5)node[left]{\normalsize $2$};
				\draw(1.2,-1)node[right]{\normalsize $\times$};
				\draw(2.0,-1.9)--(3.8,-1.9);
				\draw(4.0,-2.5)node[left]{$a\,4\,8$};
				\end{tikzpicture}
			\end{center}
			Xét phép nhân ở cột hàng chục được $2a=4$ hoặc $2a=14$, tức là $a=2$ hoặc $a=7$.\\
			Loại cả $a=2$ và $a=7$ vì trái với $(1)$.
			\item Trường hợp $b=9$, ta có phép tính 	
			\begin{center}
				\begin{tikzpicture}[scale=0.65]
				\draw(4.0,-0.5)node[left]{\normalsize $4\,a\,9$ };
				\draw(4.0,-1.5)node[left]{\normalsize $2$};
				\draw(1.2,-1)node[right]{\normalsize $\times$};
				\draw(2.0,-1.9)--(3.8,-1.9);
				\draw(4.0,-2.5)node[left]{$a\,9\,8$};
				\end{tikzpicture}
			\end{center}
			Xét phép nhân ở cột hàng chục được $2a=8$ hoặc $2a=18$, tức là $a=4$ hoặc $a=9$.\\
			Loại $a=4$ vì trái với $(1)$.\\
			Với $a=9$, ta được 
			\begin{center}
				\begin{tikzpicture}[scale=0.65]
				\draw(4.0,-0.5)node[left]{\normalsize $4\,9\,9$ };
				\draw(4.0,-1.5)node[left]{\normalsize $2$};
				\draw(1.2,-1)node[right]{\normalsize $\times$};
				\draw(2.0,-1.9)--(3.8,-1.9);
				\draw(4.0,-2.5)node[left]{$9\,9\,8$};
				\end{tikzpicture}
			\end{center}
			Thỏa mãn yêu cầu bài toán. Vậy số phải tìm là $99$.
		\end{itemize}
		\textbf{Cách 2:} \\
		Gọi số phải tìm là $\overline{ab}$, ta có $\overline{ab8}=\overline{4ab}\cdot 2$.\\
		Đặt $\overline{ab}=x$, ta được
		\begin{eqnarray*}
			10x+8 &=&(400+x)\cdot 2\\
			10x+8&=&800+2x \\
			8x & = & 792 \\
			x & =& 99. 	
		\end{eqnarray*}
		Vậy ta tìm được $x=99$.
	}
\end{vd}

\begin{dang}{Tìm chữ số tận cùng của một lũy thừa}
	Trong nhiều trường hợp, ta cần tìm một hoặc nhiều chữ số tận cùng của một số. Ở chuyên đề 3, ta sẽ thấy: Để xét xem một số có chia hết cho $2$, cho $5$ hay không, ta chỉ cần biết chữ số tận cùng của số ấy; để xét xem một số có chia hết cho $4$, cho $25$ hay không, ta cần biết hai chữ số tận cùng của số ấy.
	\begin{listEX}
		\item 	Tìm một số tận cùng.\\
		Khi tìm chữ số tận cùng của một lũy thừa, cần chú ý đến những số vẫn giữ nguyên chữ số tận cùng khi nâng lên lũy thừa bất kì (khác $0$):
		\begin{itemize}
			\item Các số có tận cùng là $0, 1, 5, 6$ nâng lên lũy thừa $n$ (khác $0$) vẫn giữ nguyên chữ số tận cùng:\\
			$(\cdots 0)^n=\cdots 0; (\cdots 1)^n=\cdots 1; (\cdots 5)^n=\cdots 5; (\cdots 6)^n=\cdots 6$.
			\item Với các số có tận cùng khác $0, 1, 5, 6$, ta đưa về số có tận cùng là $1$ hoặc $6$:
			\begin{center}
				$(\cdots 2)^4=\cdots 6; (\cdots 3)^4=\cdots 1; (\cdots 4)^2=\cdots 6; (\cdots 7)^4=\cdots 1;(\cdots 8)^4=\cdots 6; (\cdots 9)^2=\cdots 1$.    
			\end{center}
		\end{itemize} 
		\item 	Tìm hai chữ số tận cùng.\\
		Khi tìm hai chữ số tận cùng của một lũy thừa, cần chú ý đến những số vẫn giữ nguyên hai chữ số tận cùng khi nâng lên lũy thừa bất kì (khác $0$).
		\begin{itemize}
			\item Các số có tận cùng là $00, 01, 25, 76$ nâng lên lũy thừa $n$ (khác $0$) vẫn giữ nguyên hai chữ số tận cùng:
			\begin{center}
				$(\cdots 00)^n=\cdots 00; (\cdots 01)^n=\cdots 01; (\cdots 25)^n=\cdots 25; (\cdots 76)^n=\cdots 76.$ 
			\end{center}
			\item Với những trường hợp khác, ta thường nâng lên lũy thừa thích hợp để được số có tận cùng $01$ hoặc $76$.
		\end{itemize}
	\end{listEX}
\end{dang}

\begin{vd}%[8D3B2]
	Tìm chữ số tận cùng của $3^{2015}$. $\underbrace{ 11\ldots1}_{100\ \text{chữ số}}$
	\loigiai{
		Ta thấy $3^4=81$, số tận cùng bàng $1$ nâng lên lũy thừa nào cũng tận cùng bằng $1$ nên
		\begin{center}
			$3^{2015}=\underbrace{3^4\cdot 3^4\cdots\cdots 3^4}_{503 \ \text{chữ số} \ 3^4} \cdot 3^3=81^{503}\cdot 27=(\cdots 1)\cdot 27=\cdots 7$.
		\end{center}
		Vậy chữ số tận cùng của $3^{2015}$ là $7$.
	}
\end{vd}
\begin{vd}%[8D3B2]
	Tìm hai chữ số tận cùng của $6^{2011}$.
	\loigiai{
		Ta thấy $6^5$ tận cùng bằng $76$, số tận cùng là $76$ nâng lên lũy thừa $n$ (khác $0$) vẫn tận cùng là $76$ nên
		\begin{center}
			$6^{2011}=\underbrace{ 6^5\cdot 6^5\cdots\cdots 6^5\cdot }_{402 \ \text{chữ số } 6^5} 6=(\cdots 76)^{402}\cdot 6=(\cdots 76)\cdot 6=\cdots 56.$
		\end{center}
		Vậy hai chữ số tận cùng của $6^{2011}$ là $56$.
	}
\end{vd}

			Đáp số$\colon 1400$.
			
		\end{enumerate}
	}

	Người ta đánh số trang của một cuốn sách bằng dãy số tự nhiên $1$,$2$, $3$, $ \ldots $

	Từ (1) và (2) suy ra $ A=a(9a+1)-a=9a^{2}=(3a)^{2} $. Vậy $ A $ là số chính phương.
	}
\end{bt}

	}

\begin{bt}
	Tìm hai chữ số tận cùng của 
	\begin{enumEX}{2}
		\item $ 351^{2011} $;
		\item $ 218^{218} $.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item Số tận cùng là $ 51 $ khi bình phương lên sẽ được số tận cùng là $ 01 $ và số tận cùng là $ 01 $ khi lũy thừa bậc $ n $ lên sẽ được tận cùng là $ 01 $. Do đó
			\[
			351^{2011}=\underbrace{351^{2}\cdots 351^{2}}_{1005\text{ lần }351^{2}}\cdot 351=(\ldots 01)^{1005}\cdot 351=\ldots 51.
			\]
			Vậy $ 351^{2011} $ có hai chữ số tận cùng là $ 51 $.
			\item Số tận cùng là $ 18 $ khi lũy thừa bậc $ 4 $ thì tận cùng là $ 76 $ và số tận cùng là $ 76 $ khi nâng lũy thừa bậc $ n $ (khác $ 0 $) cũng tận cùng là $ 76 $. Do đó
			\[
			218^{218}=\underbrace{218^{4}\cdots 218^{4}}_{54\text{ lần }218^{4}}\cdot 218^{2}=(\ldots 76)^{54}\cdot (\ldots 24)=\ldots 24.
			\]
			Vậy $ 218^{218} $ có hai chữ số tận cùng là $ 24 $.
		\end{enumerate}
	}
\end{bt}
