	\begin{dang}{Giải và biện luận phương trình bậc nhất}
		{\textbf{Phương pháp giải:}}
		\begin{enumerate}
			\item $a\neq 0$: Phương trình có một nghiệm duy nhất $x=-\dfrac{b}{a}$.
			\item $a=0$ và $b \neq 0$: Phương trình vô nghiệm.
			\item $a=0$ và $b=0$: Phương trình nghiệm đúng với mọi $x \in \mathbb{R}$.
		\end{enumerate}
	\end{dang}

	\begin{vd}%%[Nguyễn Kim Đông]%[0D2K2]
		Giải và biện luận phương trình sau theo tham số $m$
		\begin{equation*}
		m^2 x+2=x+2m \quad \left(1\right)
		\end{equation*}
		\loigiai{
			Ta có biến đổi tương đương 
			\begin{equation*}
			\left(1\right) \Leftrightarrow m^2 x-x=2m-2 \Leftrightarrow \left(m^2-1\right)x=2\left(m-1\right) \quad (2)
			\end{equation*}
			Ta xét các trường hợp sau đây:\\
			\textbf{Trường hợp 1:} Khi $m \neq \pm 1$, ta có $m^2-1 \neq 0$ nên $(2)$ có nghiệm
			\begin{equation*}
			x=\dfrac{2\left(m-1\right)}{m^2-1}=\dfrac{2}{m+1}.
			\end{equation*}
			Đây là nghiệm duy nhất của phương trình.\\
			\textbf{Trường hợp 2:} Khi $m=1$, phương trình $(2)$ trở thành $0.x=0$. Phương trình này có nghiệm đúng với mọi số thực $x$ nên phương trình $(1)$ cũng có nghiệm đúng với mọi số thực $x$.
			\textbf{Trường hợp 3: } Khi $m=-1$, phương trình $(2)$ trở thành $0.x=-4$. Phương trình này vô nghiệm nên phương trình $(1)$ cũng vô nghiệm.\\
			
			\textbf{Kết luận:}
			\begin{itemize}
				\item	Với $m\neq \pm 1$: $(1)$ có nghiệm duy nhất $x= \dfrac{2}{m+1}$.
				\item Với $m=-1$: $(1)$ vô nghiệm.
				\item Với $m=1$: $(1)$ có vô số nghiệm.
			\end{itemize}
		}
	\end{vd}

	\begin{vd}%%[Nguyễn Kim Đông]%[0D3G2]
		Giải và biện luận phương trình $\dfrac{2x+a}{a-2}-\dfrac{a-2x}{a+2}=\dfrac{6a}{a^2-4}$. \quad $(1)$
		\loigiai{
			Ta có $\heva{&a-2 \neq 0\\& a+2 \neq 0\\& a^2-4 \neq 0}\Leftrightarrow a \neq \pm 2.$\\
			Phương trình trên được viết lại dưới dạng
			\begin{equation*}
			\left(2x+a\right)\left(a+2\right)-\left(a-2x\right)\left(a-2\right)=6a \Leftrightarrow 2ax=a.\quad (2)
			\end{equation*}
			\textbf{Trường hợp 1:} Nếu $a \neq 0$ thì $ (2)\Leftrightarrow x=\dfrac{2a}{4a}=\dfrac{1}{2}$.\\
			\textbf{Trường hợp 2:} Nếu $ a=0$ thì $(2)\Leftrightarrow 0.x =0$, phương trình có nghiệm đúng với mọi số thực $x$.\\
			\textbf{Kết luận:}
			\begin{itemize}
				\item Với $a \neq 0$ và $a \neq \pm 2$ thì phương trình có một nghiệm duy nhất $x=\dfrac{1}{2}$.
				\item Với $a=0$ thì phương trình có nghiệm đúng với mọi số thực $x$.
				\item Với $a =\pm 2$ thì phương trình đã cho vô nghiệm. 
			\end{itemize}
		}
	\end{vd}

	\begin{vd}%%[Nguyễn Kim Đông]%[0D3K2]
	Tìm giá trị của tham số $m$ để phương trình sau có tập hợp nghiệm là $\mathbb{R}$
	\begin{equation*}
	m\left(m^2 x-1\right)=1-x \quad (1)
	\end{equation*}
	\loigiai{Phương trình đã cho viết dưới dạng $\left(m^3+1\right)x=m+1$. \quad $(2)$\\
		Do đó, phương trình $(1)$ có tập nghiệm là $\mathbb{R}$ khi và chỉ khi phương trình $(2)$ có tập nghiệm $\mathbb{R}$ $\Leftrightarrow \heva{&m^3+1=0\\&m+1=0}\Leftrightarrow m=-1.$\\
		Vậy với $m=-1$ thì phương trình $(1)$ có tập nghiệm là $\mathbb{R}$.}
\end{vd}

\begin{vd}%%[Nguyễn Kim Đông]%[0D3K2]
	Tìm giá trị tham số $m$ để phương trình sau có nghiệm $x>2$
	\begin{equation*}
	2x-3m=1 \quad (1)
	\end{equation*}
	\loigiai{Phương trình đã cho được viết lại dưới dạng $x=\dfrac{3m+1}{2}$.\\ Phương trình $(1)$ có nghiệm $x >2$ khi và chỉ khi $\dfrac{3m+1}{2}>2 \Leftrightarrow m>1$.\\
		Vậy $m>1$ thỏa yêu cầu bài toán.}
\end{vd}

		\loigiai{

		Giải và biện luận phương trình $m\left(x-2m\right)=x+m+2$. \quad $(1)$
		\loigiai{
			Phương trình $(1)$ được viết lại dưới dạng $\left(m-1\right)x=2m^2+m+2$. $(2)$
			\begin{itemize}
				\item Với $m=1$, phương trình $(2)$ trở thành $0.x=5$. Điều này vô lí, phương trình đã cho vô nghiệm.
				\item Với $m\neq 1$, phương trình có nghiệm duy nhất là $x=\dfrac{m^2+2+m}{m-1}$.
			\end{itemize}
		}
	\end{bt}

	\begin{bt}%%[Nguyễn Kim Đông]%[0D3K2]
		Giải và biện luận phương trình $m^2x+2=x+2m$. \quad $(1)$
		\loigiai{
			Phương trình $(1)$ được viết lại dưới dạng $\left(m^2-1\right)x=2m-2$. \quad $(2)$
			\begin{itemize}
				\item Với $m \neq \pm 1$, phương trình $(2)$ có nghiệm duy nhất $x=\dfrac{2m-2}{m^2-1}=\dfrac{2}{m+1}$.
				\item Với $m=1$, phương trình $(2)$ trở thành $0.x=0$. Phương trình đúng với mọi số thực $x$.
				\item Với $m=-1$, phương trình $(2)$ trở thành $0.x=-4$. Điều này vô lí nên phương trình đã cho vô nghiệm.
			\end{itemize}
		}
	\end{bt}

	\begin{bt}%%[Nguyễn Kim Đông]%[0D3K2]
		Giải và biện luận phương trình $m^2x+1=\left(m-1\right)x+m$. \quad $(1)$.
		\loigiai{
			Phương trình $(1)$ được viết lại dưới dạng $\left(m^2-m+1\right)x=m-1$. \quad $(2)$.\\
			Vì $m^2-m+1 \neq 0, \forall x \in \mathbb{R}$ nên phương trình $(2)$ luôn có nghiệm duy nhất $x=\dfrac{m-1}{m^2-m+1}$.
		}
	\end{bt}

	\begin{bt}%%[Nguyễn Kim Đông]%[0D3K2]

			\end{itemize}
		}
	\end{bt}

	\begin{bt}%%[Nguyễn Kim Đông]%[0D3K2]
		Tìm giá trị tham số $m$ để phương trình $m^2\left(mx-1\right)=2m\left(2x+1\right)$ \quad $(1)$ có tập nghiệm là $\mathbb{R}$. 
		\loigiai{
			Phương trình $(1)$ được viết lại dưới dạng $\left(m^3-4m\right)x=2m+m^2$. \quad $(2)$.\\
			Phương trình $(1)$ có tập nghiệm là $\mathbb{R}$ khi và chỉ khi phương trình $(2)$ có tập nghiệm là $\mathbb{R}$. Điều này xảy ra khi và chỉ khi $\heva{&m^3-4m=0\\&2m+m^2=0}\Leftrightarrow \hoac{&m=0\\&m=-2}$.
		}
	\end{bt}

	\begin{bt}%%[Nguyễn Kim Đông]%[0D3K2]

		\loigiai{
			Phương trình $(1)$ được viết lại dưới dạng $\left(m-2\right)x=m^2-3m+2$. \quad $(2)$.\\
			Phương trình $(1)$ có tập nghiệm là $\mathbb{R}$ khi và chỉ khi phương trình $(2)$ có tập nghiệm là $\mathbb{R}$. Điều này xảy ra khi và chỉ khi $\heva{&m-2=0\\&m^2-3m+2=0}\Leftrightarrow m=2$.
		}

	\begin{bt}%%[Nguyễn Kim Đông]%[0D3G2]
		Giải và biện luận phương trình $\left(x-1\right)\left(x-mx+2\right)=0$.
		\loigiai{
			Phương trình $(1)$ tương đương với
			\begin{equation*}
			\hoac{&x=1\\&\left(1-m\right)x=-2 \quad (*)}
			\end{equation*}
			\begin{itemize}
				\item  Với $m=1$, phương trình $(*)$ trở thành $0.x=-2$. Điều này vô lí nên phương trình $(*)$ vô nghiệm. Phương trình $(1)$ có nghiệm duy nhất $x=1$.
				\item Với $m=3$, phương trình $(*)$ trở thành $-2x=-2$. Phương trình có nghiệm duy nhất $x=1$. Do đó, phương trình $(1)$ có nghiệm duy nhất $x=1$.
				\item Với $m \neq 1$ và $m\neq 3$, phương trình $(*)$ có nghiệm duy nhất $x=-\dfrac{2}{1-m} \neq 1$. Do đó, phương trình $(1)$ có hai nghiệm $x=1$ và $x=-\dfrac{2}{1-m}$.
			\end{itemize}.
		}
	\end{bt}

	\begin{bt}%%[Nguyễn Kim Đông]%[0D3G2]
		Giải và biện luận phương trình $\left(x^2-4\right) \left(mx-3\right)=0$.
		\loigiai{
			Phương trình $(1)$ tương đương với
			\begin{equation*}
			\hoac{&x=\pm 2\\&mx=3 \quad (*)}
			\end{equation*}
			\begin{itemize}
				\item  Với $m=0$, phương trình $(*)$ trở thành $0.x=3$. Điều này vô lí nên phương trình $(*)$ vô nghiệm. Phương trình $(1)$ có hai nghiệm $x=\pm 2$.
				\item Với $m=\dfrac{3}{2}$, phương trình $(*)$ trở thành $\dfrac{3}{2}x=3$. Phương trình $(*)$ có nghiệm duy nhất $x=2$. Do đó, phương trình $(1)$ có hai nghiệm $x= \pm 2$.
				\item Với $m=-\dfrac{3}{2}$, phương trình $(*)$ trở thành $-\dfrac{3}{2}x=3$. Phương trình $(*)$ có nghiệm duy nhất $x=-2$. Do đó, phương trình $(1)$ có hai nghiệm $x= \pm 2$.
				\item Với $m \neq \pm 2$ và $m\neq 0$, phương trình $(*)$ có nghiệm duy nhất $x=-dfrac{3}{m} \neq \pm 2$. Do đó, phương trình $(1)$ có ba nghiệm $x=\pm 2$ và $x=\dfrac{3}{m}$.
			\end{itemize}.
		}
	\end{bt}
	
	

\begin{dang}{Phương trình chứa ẩn dưới dấu căn}
	Nguyên tắc cơ bản trong giải phương trình chứa ẩn dưới dấu căn là phải tìm cách làm mất dấu căn. Có các phương pháp thường dùng như: bình phương hai vế, đặt ẩn phụ, đưa phương trình về dạng tích, \ldots\\
	
	\noindent\textbf{Phương pháp 1. Bình phương hai vế.}\\
	Thiết lập điều kiện rồi sau đó bình phương hai vế.
	\begin{itemize}
		\item $\sqrt{A}=\sqrt{B}\Leftrightarrow\heva{&B\geq 0\\&A=B.}$
		\item $\sqrt{A}=B\Leftrightarrow\heva{&B\geq 0\\&A=B^2.}$
	\end{itemize}
\noindent\textbf{Phương pháp 2. Đặt ẩn phụ.} \\
Nhiều phương trình, việc bình phương không thể làm mất hết căn hoặc lại đưa về những phương trình bậc cao hơn hai. Những câu như vậy ta không nên bình phương hai vế mà nên sử dụng phương pháp khác.\\
Sau đây là một số dạng hay gặp trong đặt ẩn phụ:
\begin{itemize}
	\item $af\left( x\right) +b\sqrt{ f\left( x\right) }=c  $. Đặt $ \sqrt{ f\left( x\right) }=t$.
	\item $ a(\sqrt{A}\pm\sqrt{B})+b\sqrt{A.B}=c $ (A,B là biểu thức của $x$). Đặt $ \sqrt{A}\pm\sqrt{B}=t \Rightarrow \sqrt{A.B}=\cdots $(Bình phương $t$ để đưa ra $\sqrt{A.B}$).
\end{itemize}

\noindent\textbf{Phương pháp 3. Đưa về dạng tích.}\\
Nếu phương trình đưa được về tích ta có thể chuyển về các phương trình dễ giải hơn. Chúng ta có thể thực hiện theo một trong những hướng sau:
\begin{itemize}
	\item Ghép nhóm tạo ra nhân tử chung.
	\item Biến đổi liên hợp $ \sqrt{A}-\sqrt{B}=\dfrac{A-B}{\sqrt{A}+\sqrt{B}} $.
	\item Khi nhẩm được nghiệm thì thêm bớt hệ số để liện hợp tạo ra nhân tử chung.
\end{itemize}

\end{dang}

\begin{dang}{Phương trình chứa ẩn trong dấu giá trị tuyệt đối}
	Nguyên tắc cơ bản trong giải phương trình chứa ẩn trong dấu giá trị tuyệt đối là phải tìm cách làm mất dấu giá trị tuyệt đối. Các phương pháp thường dùng là: biến đổi tương đương, chia khoảng trên trục số, \ldots\\
	
	\noindent\textbf{Phương pháp 1. Biến đổi tương đương.}\\ 
	Với $f(x)$, $g(x)$ là các hàm số. Khi đó
	\begin{align*}
	& |f(x)|=g(x)\Leftrightarrow\heva{& g(x)\geq 0\\&\hoac{& f(x)=g(x)\\& f(x)=-g(x)}}\\&
	|f(x)|=|g(x)|\Leftrightarrow\hoac{& f(x)=g(x)\\& f(x)=-g(x)}\\&
	|f(x)|+|g(x)|=|f(x)+g(x)|\Leftrightarrow f(x).g(x)\geq 0
	\end{align*}  
	
	\noindent\textbf{Phương pháp 2. Chia khoảng trên trục số}\\
	Ta lập bảng xét dấu của các biểu thức trong dấu giá trị tuyệt đối rồi xét các trường hợp để khử dấu giá trị tuyệt đối.\\
	
	\noindent\textbf{Một số cách khác}
	\begin{enumerate}
		\item Đặt ẩn phụ.
		\item Sử dụng bất đẳng thức ta so sánh $f(x)$ và $g(x)$ từ đó tìm nghiệm của phương trình $f(x)=g(x)$.
		\item Sử dụng đồ thị cần chú ý số nghiệm của phương trình $f(x)=g(x)$ là số giao điểm của hai đồ thị hàm số $y=f(x)$ và $y=g(x)$. Phương pháp này thường áp dụng cho các bài toán biện luận nghiệm.
	\end{enumerate}
	
\end{dang}

	\textbf{Loại 1. Phương trình chứa ẩn ở mẫu thức} 
	\begin{itemize}
		\item Đặt điều kiện xác định của phương trình.
		\item Biến đổi phương trình đã cho về phương trình bậc nhất, bậc hai đã biết cách giải.
		\item Chọn nghiệm thỏa điều kiện xác định của phương trình.
	\end{itemize}
\begin{note}
	Khi giải phương trình chứa ẩn ở mẫu thức, ta phải chú ý điều kiện xác định của phương trình.
\end{note}
	\textbf{Loại 2. Phương trình trùng phương}\\
	Để giải phương trình trùng phương dạng $ax^4+bx^2+c=0\;(\star)$ ta đặt $t=x^2\ge 0$ để đưa về phương trình bậc hai $at^2+bt+c=0\;(\star')$.
	\begin{itemize}
		\item Nếu phương trình $(\star')$ vô nghiệm hoặc chỉ có nghiệm âm thì phương trình $(\star)$ vô nghiệm.
		\item Nếu phương trình $(\star')$ có nghiệm $t=0$ thì phương trình $(\star)$ có nghiệm $x=0$.
		\item Nếu phương trình $(\star')$ có một nghiệm $t=t_0>0$ thì phương trình $(\star)$ có hai nghiệm $x=\pm\sqrt{t_0}$.
	\end{itemize}
\end{dang}

	\textbf{Định lí Viète}\\
	Nếu phương trình bậc hai $ax^2+bx+c=0\, (a\neq0)$ có hai nghiệm $x_1$, $x_2$ thì
	$$x_1+x_2=-\dfrac{b}{a}\,;\quad x_1x_2=\dfrac{c}{a}.$$

	$$x^2-Sx+P=0.$$
\end{dang}

\begin{vd}%[Đào-V- Thủy]%[0D3B1]
