\begin{ex}%[Đề thi Thử tuyển sinh vào 10 đợt 1, 2018, THPT Thăng Long, Hà Nội]%[9D1B8]
	Cho hai biểu thức: $ A = \dfrac{2x-3\sqrt{x} - 2}{\sqrt{x}-2} $ và $ B = \dfrac{\sqrt{x^3}- \sqrt{x}+2x-2}{\sqrt{x}+2} $ với $ x \ge 0 $ và $ x\ne 4 $.
	\begin{enumerate}
		\item Tính giá trị của $ A $ khi $ x = 4-2\sqrt{3} $.
		\item Tìm giá trị của $ x $ để $ B = A + 1 $.
		\item Tìm giá trị nhỏ nhất của biểu thức $ C = B-A $.
	\end{enumerate}
\loigiai{
Với $ x \ge 0,\ x\ne 4 $, ta có
\begin{align*}
A = \dfrac{2x-3\sqrt{x}-2}{\sqrt{x}-2} = \dfrac{(2x-4\sqrt{x}) + (\sqrt{x}-2)}{\sqrt{x}-2} = \dfrac{(\sqrt{x}-2)(2\sqrt{x}+1)}{\sqrt{x}-2} = 2\sqrt{x}+1.
\end{align*}
\begin{align*}
B = \dfrac{\sqrt{x^3}-\sqrt{x}+2x-2}{\sqrt{x}+2} = \dfrac{(\sqrt{x^3}-\sqrt{x})+(2x-2)}{\sqrt{x}+2} = \dfrac{(\sqrt{x}+2)(x-1)}{\sqrt{x}+2} = x-1.
\end{align*}
\begin{enumerate}
	\item Khi $ x = 4-2\sqrt{3} = 3 - 2\sqrt{3} + 1 = \left(\sqrt{3}-1 \right)^2 $, thay vào $ A $, ta được
	\[ A = 2\sqrt{x} + 1 = 2\sqrt{\left(\sqrt{3}-1 \right)^2}  +1 = 2(\sqrt{3}-1) +1 = 2\sqrt{3} -1. \]
	\item 
	\begin{eqnarray*}
	&B = A + 1 \\
	\Leftrightarrow &x - 1 = 2\sqrt{x}+1+1 \\
	\Leftrightarrow &(\sqrt{x}+1)(\sqrt{x}-3) = 0\\
	 \Leftrightarrow &\sqrt{x}-3 = 0\\
	 \Leftrightarrow &x = 9.
	\end{eqnarray*}
Đối chiếu điều kiện ta thấy $ x = 9 $ thỏa mãn.
	\item $ C = B-A = (x-1) - (2\sqrt{x}+1) = x-2\sqrt{x} = (\sqrt{x}-1)^2 -3 $.\\
	Với $ x \ge 0, x\ne 4 $ thì $ (\sqrt{x}-1)^2 \ge 0 $, nên $ \left(\sqrt{x}-1 \right)^2 - 3\ge -3  $.\\
	Dấu bằng xảy ra khi $ \left(\sqrt{x}-1\right)^2 = 0 \Leftrightarrow x = 1$.\\
	Vậy giá trị nhỏ nhất của biểu thức $ C $ là $ -3 $ khi $ x = 1 $.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi Thử tuyển sinh vào 10 đợt 1, 2018, THPT Thăng Long, Hà Nội]%[9D3B6]
	\textit{Giải bài toán bằng cách lập phương trình hoặc hệ phương trình}\\
	Một ô tô dự định đi từ $ A $ đến $ B $ trong một khoảng thời gian đã định. Nếu chạy xe với vận tốc $ 35\mathrm{\ km/h} $ thì đến $ B $ chậm mất $ 2 $ giờ. Nếu xe chạy với vận tốc $ 50\mathrm{\ km/h} $ thì đến sớm hơn $ 1 $ giờ. Tính quãng đường $ AB $ và thời gian dự định đi lúc đầu.
	\loigiai{
	Gọi $ x $ (giờ) là thời gian dự định đi lúc ban đầu, $ x>0 $.\\
	Thời gian khi chạy với vận tốc $ 35\mathrm{\ km/h} $ và $ 50\mathrm{\ km/h} $ lần lượt là $ x+2 $ và $ x-1 $.
	Theo đề bài ta có phương trình sau
	\[ 35(x+2) = 50(x-1)  \Leftrightarrow 35x + 70 = 50x - 50 \Leftrightarrow x = 8\text{ (nhận)}.  \]
	Vậy thời gian dự định đi lúc ban đầu là $ 8 $ giờ. Quãng đường $ AB $ là $ 35(8+2) = 350\mathrm{\ km} $.	
	}
\end{ex}

\begin{ex}%[Đề thi Thử tuyển sinh vào 10 đợt 1, 2018, THPT Thăng Long, Hà Nội]%[9D3B6]%[9D2B5]%[9D2K4]
	\begin{enumerate}[1)]
		\item Giải hệ phương trình $ \begin{cases}
		\sqrt{\dfrac{x+3}{x}}+\dfrac{2y}{y-2} = 8\\2\sqrt{\dfrac{x+3}{x}} + \dfrac{3y}{y-2} = 13
		\end{cases} $.
		\item Trong mặt phẳng tọa độ $ Oxy $, cho hai đường thẳng $ (d_1)\colon y = -mx + m +1 $ và \\ $ (d_2)\colon y = \dfrac{1}{m}x-1-\dfrac{5}{m} $ với $ m $ là tham số khác $ 0 $.
		\begin{enumerate}[a)]
			\item Chứng minh rằng $ (d_1) $ và $ (d_2) $ luôn vuông góc với nhau với mọi giá trị của tham số $ m\ne 0 $.
			\item Tìm điểm cố định mà đường thẳng $ (d_1) $ luôn đi qua. Chứng minh rằng giao điểm của hai đường thằng luôn thuộc một đường cố định.
		\end{enumerate}
	\end{enumerate}
\loigiai{
	\begin{enumerate}[1)]
		\item ĐKXĐ $ \dfrac{x+3}{x} \ge 0, y\ne 2 $. Đặt $ \begin{cases}
		\sqrt{\dfrac{x+3}{x}} = a\ (a\ge 0) \\
		\dfrac{y}{y-2} = b\ (b > 0)
		\end{cases} $. Khi đó, hệ phương trình trở thành
		\[ \begin{cases}
		a+2b = 8\\
		2a + 3b = 13
		\end{cases} \Leftrightarrow \begin{cases}
		a  = 2\\
		b = 3.
		\end{cases} \]
		Thay lại theo biến $ x $ ta thu được
		\[ \begin{cases}
		\sqrt{\dfrac{x+3}{x}} = 2\\
		\dfrac{y}{y-2} = 3
		\end{cases} \Leftrightarrow \begin{cases}
		x = 1\\
		y=3.
		\end{cases} \]
		Đối chiếu điều kiện thấy $ x = 1, y = 3 $ thỏa mãn.
		\item a) Hệ số góc của đường thẳng $ (d_1) $ là $ -m $ và hệ số góc của đường thẳng $ (d_2) $ là $ \dfrac{1}{m} $.\\
		Vì $ -m\cdot\dfrac{1}{m} = -1, \forall m \ne 0 $, nên hai đường thẳng $ (d_1) $ và $ (d_2) $ vuông góc với nhau với mọi giá trị $ n\ne 0 $.\\
		b) Giả sử $ N(x_N,y_N) $ là điểm cố định của $ (d_1) $ với mọi $ m $, khi đó $ y_N = -mx_N + m + 1 $ với mọi $ m $ nghĩa là phương trình ẩn $ m $: $ m(1-x_N) + 1-y_N = 0 $ có nghiệm với mọi $ m $, điều này xảy ra khi và chỉ khi $ \begin{cases}
		1-x_N = 0\\ 1-y_N = 0
		\end{cases} \Leftrightarrow \begin{cases}
		x_N = 1\\
		y_N = 1
		\end{cases} $. Vậy điểm cố định mà đường thẳng $ (d_1) $ luôn đi qua là $ N(1;1) $.\\
		Giả sử $ M(x_0,y_0) $ là giao điểm của $ (d_1) $ và $ (d_2) $ ta có hệ sau\\
		\begin{eqnarray*}
		&&\begin{cases}
		y_0 - 1 = m(1-x_0)\\
		y_0 + 1 = \dfrac{1}{m}(x_0-5)
		\end{cases}\\
		\Rightarrow&& (y_0+1)(y_0-1) = (1-x_0)(x_0-5) \\
		\Leftrightarrow &&y_0^2 - 1 = -x_0^2 + 6x_0 - 4\\
		\Leftrightarrow && (x_0-3)^2 + y_0^2 = 5.
		\end{eqnarray*}
		Đặt $ I(3;0) $. Khi đó ta có $ IM = \sqrt{(x_0-3)^2 + y_0^2} = \sqrt{5} $ không đổi. Vậy $ M $ thuộc đường tròn tâm $ I $ bán kính $ \sqrt{5} $.
	\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi thử vào 10, 2018, THPT Thăng Long, Hà Nội]%[9H3B3]%[9H2K5]%[9H3G7]
	Cho đường tròn tâm $O$, bán kính $R$. Điểm $A$ thuộc đường tròn, $BC$ là một
	đường kính $(A\neq B, A\neq C)$. Vẽ $AH$ vuông góc với $BC$ tại $H$. Gọi $E$, $M$ lần lượt là trung điểm của
	$AB$, $AH$ và $P$ là giao điểm của $OE$ với tiếp tuyến tại $A$ của đường tròn $(O;R)$.
	\begin{enumerate}
		\item Chứng minh rằng $AB^2=BH\cdot BC$.
		\item Chứng minh $PB$ là tiếp tuyến của đường tròn $(O)$.
		\item Chứng minh ba điểm $P,M,C$ thẳng hàng.
		\item Gọi $Q$ là giao điểm của đường thẳng $PA$ với tiếp tuyến tại $C$ của đường tròn $(O)$. Khi $A$ thay đổi trên đường tròn $(O)$, tìm giá trị nhỏ nhất của tổng $OP+OQ$.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[line join=round, line cap=round]
			\def\r{3}
			%\draw[cyan, dashed] (-5,-5) grid (5,5);
			\tkzInit[xmax=3.7, xmin=-3.7, ymax=6.2, ymin=-3.2]
			\tkzClip
			\coordinate[label=below:$O$] (O) at (0,0);
			\draw (O) circle (\r cm);
			\coordinate[label={below left}:$B$] (B) at ($(O)+(-\r,0)$);
			\coordinate[label={below right}:$C$] (C) at ($(O)+(\r,0)$);
			\tkzDefPointBy[rotation= center O angle 120](C) \tkzGetPoint{A} \tkzLabelPoints[above left](A)
			\tkzDefPointBy[projection= onto B--C](A) \tkzGetPoint{H} \tkzLabelPoints[below](H)
			\coordinate[label={below}:$E$] (E) at ($(A)!0.5!(B)$);
			\coordinate[label={above right}:$M$] (M) at ($(A)!0.5!(H)$);
			\tkzDefPointWith[orthogonal](A,O) \tkzGetPoint{x}
			\tkzInterLL(O,E)(A,x) \tkzGetPoint{P} \tkzLabelPoints[above left](P)
			\tkzDefPointWith[orthogonal](C,O) \tkzGetPoint{y}
			\tkzInterLL(C,y)(A,x) \tkzGetPoint{Q} \tkzLabelPoints[above right](Q)
			\draw (P)--(B)--(C)--(Q)--(P)--(C)--(A)--(B) (P)--(O)--(A)--(H);
			\tkzMarkRightAngles(A,H,C)
			\tkzDrawPoints[fill=black](A,B,C,P,Q,M,O,H,E)
			\end{tikzpicture}
		\end{center}
		
		\begin{enumerate}
			\item \textbf{Chứng minh rằng $AB^2=BH\cdot BC$.}\\
			Ta có $\widehat{BAC}$ là góc nội tiếp chắn nửa đường tròn nên $\widehat{BAC}=90^\circ$.\\
			Xét tam giác $\triangle ABC$ vuông tại $A$, đường cao $AH$, ta có $AB^2=BH\cdot BC$.
			\item \textbf{Chứng minh $PB$ là tiếp tuyến của đường tròn $(O)$.}\\
			Có $E$ là trung điểm của $AB\Rightarrow AB\perp OE \Rightarrow OE$ là đường trung trực của $AB$.\\
			$\Rightarrow PA=PB\Rightarrow \triangle OPA=\triangle OPB$ (c-g-c) $\Rightarrow \widehat{PAO}=\widehat{PBO}=90^\circ \Rightarrow PB\perp BO$.\\
			$\Rightarrow PB$ là tiếp tuyến của đường tròn $(O)$.
			\item \textbf{Chứng minh ba điểm $P,M,C$ thẳng hàng.}\\
			Giả sử $PC$ cắt $AH$ tại $N$. Do \\
			 $\begin{cases}	\triangle APO\backsim \triangle ABC \\ 
			AE\perp PO, AH\perp BC\end{cases} \Rightarrow \dfrac{AP^2}{AB^2}=\dfrac{PE\cdot PO}{BH \cdot BC}=\dfrac{PO^2}{BC^2} \Rightarrow \dfrac{PE}{BH}=\dfrac{PO}{BC}\Rightarrow \dfrac{PE}{PO}=\dfrac{BH}{BC}$. \hfill $(1)$\\
			Mà $\dfrac{BH}{BC}=\dfrac{PN}{PC}$ (do $HN\parallel PB$) \hfill $(2)$\\
			Từ $(1)$ và $(2)$ suy ra $\dfrac{PE}{PO}=\dfrac{PN}{PC}\Rightarrow EN\parallel BH$ (Ta-lét).\\
			Lại có $E$ là trung điểm của $AB\Rightarrow N$ là trung điểm $AH\Rightarrow N\equiv M$.\\
			Vậy $P,M,C$ thẳng hàng.
			\item \textbf{Tìm giá trị nhỏ nhất của tổng $OP+OQ$.}\\
			Theo bất đẳng thức Cô-si ta có
			$$OP+OQ\geqslant 2\sqrt{OP\cdot OQ}.$$
			Mà $OP\cdot OQ=OA\cdot PQ=PQ\cdot R$.\\
			$\Rightarrow OP\cdot OQ$ đạt giá trị nhỏ nhất khi $PQ$ nhỏ nhất $\Leftrightarrow PQ$ là khoảng cách giữa hai đường $BP$ và $CQ$.\\
			$\Rightarrow PQ\parallel BC\Rightarrow A$ là điểm chính giữa đường tròn.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi thử vào 10, 2018, THPT Thăng Long, Hà Nội]%[9D5G1]
	Cho các số thực không âm $x,y,z$ thỏa mãn $x\leqslant 1, y\leqslant 1,z\leqslant 1$ và $x+y+z=\dfrac{3}{2}$.\\
	Tìm giá trị lớn nhất và giá trị nhỏ nhất của biểu thức $P=x^2+y^2+z^2$.
	\loigiai{
		\begin{enumerate}
			\item \textbf{Tìm giá trị lớn nhất}\\
			Ta có $0\leqslant x,y,z\leqslant 1$. Do vai trò $x,y,z$ như nhau nên giả sử $x\geqslant y\geqslant z$. Khi đó $1\geqslant x\geqslant \dfrac{1}{2}$.\\
			Ta có
			\begin{center}
				$y+z=\dfrac{3}{2}-x\Rightarrow y^2+z^2+2yz=\dfrac{9}{4}-3x+x^2$\\ 
				$\Leftrightarrow x^2+y^2+z^2=\dfrac{9}{4}-3x+2x^2-2yz\leqslant\dfrac{9}{4}-3x+2x^2=\dfrac{5}{4}+(x-1)(2x-1)\leqslant\dfrac{5}{4}$.
			\end{center}
			Suy ra $P\leqslant \dfrac{5}{4}$.\\
			Vậy $\max P=\dfrac{5}{4}$ khi $(x,y,z)=\left( 1;\dfrac{1}{2};0\right) $ và các hoán vị $x,y,z$.
			\item \textbf{Tìm giá trị nhỏ nhất}\\
			Áp dụng bất đẳng thức Cauchy cho 2 số dương, ta có $x^2+\dfrac{1}{4}\geqslant 2\sqrt{x^2\cdot \dfrac{1}{4}}=x$.\\
			Tương tự $y^2+\dfrac{1}{4}\geqslant y;z^2+\dfrac{1}{4}\geqslant z$.\\
			Cộng theo vế các bất đẳng thức ta có $x^2+y^2+z^2+\dfrac{3}{4}\geqslant x+y+z=\dfrac{3}{2}$.\\
			Hay $x^2+y^2+z^2\geqslant \dfrac{3}{2}$.\\
			Đẳng thức xảy ra khi và chỉ khi $x=y=z=\dfrac{1}{2}$.\\
			Vậy $\min P=\dfrac{3}{2}$ khi $x=y=z=\dfrac{1}{2}$.
	\end{enumerate}}
\end{ex}
