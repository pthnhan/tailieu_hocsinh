\begin{bt}%[Thi GHK, Nguyễn Khuyến - Hồ Chí Minh, 2018]%[Trịnh Văn Luân, EX-DT-10&11]%[1D1B2-1]
	Giải các phương trình sau
	\begin{listEX}
		\item $ \cos 2x+ \sin^2 x+ 2\cos x+ 1= 0 $.
		\item  $ 4\sin^2 x -\sqrt{3} \sin 2x - 2\cos^2 x =4 $.
		\item $ \sqrt{3} \cos x+ \sin 2x= 0, \text{với}\ \dfrac{\pi}{2}< x< \dfrac{3 \pi}{2} $.
	\end{listEX}
	\loigiai{
		\begin{enumerate}
			\item Ta có 
			{\allowdisplaybreaks
			\begin{eqnarray*}
				&&\cos 2x+ \sin^2 x+ 2\cos x+ 1= 0 \\
				&\Leftrightarrow & 2\cos^2- 1+ 1- \cos^2+ 2\cos x+ 1= 0\\
				&\Leftrightarrow  & \cos^2 x+ 2\cos x+ 1= 0\\
				& \Leftrightarrow & \left(\cos x+ 1\right)^2= 0\\
				& \Leftrightarrow & x= \pi + k2 \pi \; (k \in \mathbb{Z}).
			\end{eqnarray*}}
Vậy nghiệm của phương trình đã cho là $x= \pi + k2 \pi$.
\item Dễ thấy $x= \dfrac{\pi}{2}+ k \pi; (k \in \mathbb{Z})$ là nghiệm của phương trình.\\
Với $x \ne \dfrac{\pi}{2}+ k \pi; (k \in \mathbb{Z})$. Chia cả hai vế của phương trình cho $\cos^2 x$, ta được
\begin{eqnarray*}
	&&4 \tan^2 x- 2\sqrt{3} \tan x- 2= 4(1+ \tan^2 x) \\
	&\Leftrightarrow & -2\sqrt{3} \tan x= 6\\
	&\Leftrightarrow  & \tan x= \tan \left(\dfrac{-\pi}{3}\right)\\
	& \Leftrightarrow & x= \dfrac{-\pi}{3}+ k \pi\; (k \in \mathbb{Z}).
\end{eqnarray*}
Vậy phương trình có nghiệm $x= \dfrac{\pi}{2}+ k \pi$, hoặc $x= \dfrac{-\pi}{3}+ k \pi\; (k \in \mathbb{Z}).$
\item Ta có
{\allowdisplaybreaks
\begin{eqnarray*}
	&&\sqrt{3} \cos x+ \sin 2x= 0 \\
	&\Leftrightarrow & \cos x(\sqrt{3}+ 2 \sin x)= 0\\
	&\Leftrightarrow  & \hoac{&\cos x= 0\\ &\sin x= \dfrac{-\sqrt{3}}{2}}\\
	& \Leftrightarrow & \hoac{&x= \dfrac{\pi}{2}+ k \pi\\&x= \dfrac{-\pi}{3}+ k2 \pi\\&x= \dfrac{4 \pi}{3}+ k2 \pi}; (k \in \mathbb{Z}).
\end{eqnarray*}}
Kết hợp với điều kiện, vậy phương trình có nghiệm $x= \dfrac{\pi}{2};\text{hoặc}\ x= \dfrac{4 \pi}{3}.$
		\end{enumerate}	
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Khuyến - Hồ Chí Minh, 2018]%[Trịnh Văn Luân, EX-DT-10&11]%[[1D1K3-3]]
	Tìm giá trị lớn nhất và giá trị nhỏ nhất của hàm số $y= \dfrac{\cos x+ 2\sin x+ 3}{\sin x+ 2}$.
	\loigiai{ Ta có
\begin{eqnarray*}
	&&y= \dfrac{\cos x+ 2\sin x+ 3}{\sin x+ 2} \\
	&\Leftrightarrow & y \sin x+ 2y= \cos x+ 2\sin x+ 3\\
	&\Leftrightarrow  & (y- 2) \sin x- \cos x= 3- 2y.
\end{eqnarray*}
Phương trình có nghiệm khi và chỉ khi
\begin{eqnarray*}
	&&(y- 2)^2+ 1 \geq (3- 2y)^2 \\
	&\Leftrightarrow & y^2- 4y+ 5 \geq 9- 12y+ 4y^2\\
	&\Leftrightarrow  & 3y^2- 8y+ 4 \leq 0\\
	&\Leftrightarrow  & \dfrac{2}{3} \leq y \leq 2.
\end{eqnarray*}
Vậy GTNN của $y$ là $\dfrac{2}{3}$; GTLN của $y$ là $2$.
	}
\end{bt}

	\item  Trong mặt phẳng với hệ trục tọa độ Oxy vuông góc, cho điểm $B(0; 4)$ và đường tròn $(C): x^2+ y^2- 10x+ 2y- 1=0$. Hãy viết phương trình ảnh của đường tròn $(C)$ qua phép vị tự tâm $B$ tỉ số $-2$.

\end{listEX}
