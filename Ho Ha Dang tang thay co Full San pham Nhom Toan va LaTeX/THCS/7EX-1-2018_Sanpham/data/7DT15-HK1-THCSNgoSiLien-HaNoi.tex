	\noindent\textbf{II. PHẦN TỰ LUẬN}
	\setcounter{ex}{0}
	\begin{ex}%[Đề kiểm tra học kỳ I, THCS Ngô Sĩ Liên, 2018]%[Đỗ Đức Thảo, dự án 7EX-1-2018]%[7D1B11]
		Thực hiện phép tính (tính nhanh nếu có thể)
		\begin{enumerate}
			\item $\sqrt{\dfrac{25}{81}}:2\dfrac{2}{5}-4\dfrac{5}{9}:2\dfrac{2}{5}$.
			\item $2^3\cdot \left( \dfrac{1}{4}\right)^3-6\dfrac{3}{5}-\dfrac{5}{8}+4\dfrac{3}{5} $.
		\end{enumerate}
			\loigiai{
				\begin{enumerate}
					\item {\allowdisplaybreaks
					\begin{eqnarray*}
						\sqrt{\dfrac{25}{81}}:2\dfrac{2}{5}-4\dfrac{5}{9}:2\dfrac{2}{5}&=&
						\dfrac{5}{9}\cdot \dfrac{5}{12}-\dfrac{41}{9}\cdot \dfrac{5}{12}\\
						&=&\dfrac{5}{12}\left( \dfrac{5}{9}-\dfrac{41}{9}\right) \\
						&=&\dfrac{5}{12}\cdot (-4)=-\dfrac{5}{3}.
				\end{eqnarray*}}
					
					\item {\allowdisplaybreaks
					\begin{eqnarray*}
						2^3\cdot \left( \dfrac{1}{4}\right)^3-6\dfrac{3}{5}-\dfrac{5}{8}+4\dfrac{3}{5} 
						&=&\dfrac{1}{8}-\dfrac{33}{5}-\dfrac{5}{8}+\dfrac{23}{5}\\
						&=&\dfrac{-1}{2}-2\\
						&=&\dfrac{-5}{2}.
				\end{eqnarray*}}
				\end{enumerate}	
			}
	\end{ex}

	\begin{ex}%[Đề kiểm tra học kỳ I, THCS Ngô Sĩ Liên, 2018]%[Đỗ Đức Thảo, dự án 7EX-1-2018]%[7D1B6]
		Tìm $x$ thỏa mãn:
		\begin{enumerate}
			\item $\dfrac{2}{3}:x=2\dfrac{1}{2}:(-0,3)$.
			\item $\left(x-\dfrac{1}{5} \right)^3\cdot 2-\dfrac{4}{5}=-16,8 $.
			
		\end{enumerate}
		\loigiai{
			\begin{enumerate}
				\item {\allowdisplaybreaks
				\begin{eqnarray*}
					&&\dfrac{2}{3}:x=2\dfrac{1}{2}:(-0,3)\\
					&\Leftrightarrow& \dfrac{2}{3}:x=-\dfrac{25}{3}\\
					&\Leftrightarrow& x=\dfrac{2}{3}:\dfrac{-25}{3}\\ &\Leftrightarrow& x=-\dfrac{2}{25}.
			\end{eqnarray*}}
				\item {\allowdisplaybreaks
				\begin{eqnarray*}
					&&\left(x-\dfrac{1}{5} \right)^3\cdot 2-\dfrac{4}{5}=-16,8 \\
					&\Leftrightarrow& \left(x-\dfrac{1}{5} \right)^3=-8\\
					&\Leftrightarrow& x-\dfrac{1}{5}=-2\\
					&\Leftrightarrow& x=-\dfrac{9}{5}.
			\end{eqnarray*}}
				
			\end{enumerate}
		}
	\end{ex}
	
	\begin{ex}%[Đề kiểm tra học kỳ I, THCS Ngô Sĩ Liên, 2018]%[Đỗ Đức Thảo, dự án 7EX-1-2018]%[7D1B2]
		Một ô tô đi từ $ A $ đến $ B $ mất $ 6 $ giờ, đi từ $ B $ về $ A $ mất $ 4,5 $ giờ. Tính quãng đường $ AB $ biết vận tốc ô tô lúc về hơn lúc đi là $ 2 $ km/h.
		\loigiai
		{
			Gọi $ x $ (km) là độ dài quãng đường $ AB $. Khi đó\\
			Vận tốc ô tô khi đi từ $ A $ đến $ B $ là $ \dfrac{x}{6} $.\\ 
			Vận tốc ô tô khi đi từ $ B $ đến $ A $ là $ \dfrac{x}{4,5} $.\\ 
			Theo đề bài ta có vận tốc lúc về hơn lúc đi là $ 2$  km/h, do đó 
			\[ \dfrac{x}{4,5} - \dfrac{x}{6} = 2 \Leftrightarrow x=36\]
			Vậy quãng đường $ AB $ dài $ 36 $ km.
		}
	\end{ex}
	
	\begin{ex}%[Đề kiểm tra học kỳ I, THCS Ngô Sĩ Liên, 2018]%[Đỗ Đức Thảo, dự án 7EX-1-2018]%[7H2K8]%[7H2B3]%[7H2K4]
		Cho tam giác $ABC$ có $AB=AC$. Lấy $M$ là trung điểm $B,C$.
		\begin{enumEX}{1}
			\item Chứng minh $\triangle ABM=\triangle ACM$ và tia $AM$ là tia phân giác của góc $BAC$.
			\item Lấy điểm $D$ thuộc tia đối của tia $BC$ và điểm $E$ thuộc tia đối của tia $CB$ sao cho $BD=CE$. Kẻ $BH$ vuông góc với $AD$ tại $H$, kẻ $CI$ vuông góc với $AE$ tại $I$. Chứng minh $\triangle ABD=\triangle ACE$ và $ DH=EI$. 
			\item Trong trường hợp $BA = BD$ và góc $\widehat{BAC}={90}^\circ$, tính góc $BDA$. 
		\end{enumEX}
		\loigiai{
			\begin{center}
				\begin{tikzpicture}[scale=0.7, font=\footnotesize, line join=round, line join=round, line cap = round,>=stealth]
				%tkzInit[xmin=-4,ymin=-4,xmax=4,ymax=4] \tkzAxeXY \tkzGrid 
				\tkzInit[ymin=-1,ymax=7,xmin=-4,xmax=8]
				\tkzClip
				\tkzDefPoints{0/0/B,2/6/A,4/0/C,-3/0/D,7/0/E}
				\coordinate (M) at ($(C)!0.5!(B)$); 
				\tkzDrawAltitude[draw=black](A,D)(B) \tkzGetPoint{H}
				\tkzDrawAltitude[draw=black](A,E)(C) \tkzGetPoint{I}
				\tkzMarkRightAngles[size=0.25](B,H,A C,I,A)
				\tkzMarkSegments[mark=|](M,C M,B)
				\tkzDrawSegments(A,B A,C D,E A,D A,E A,M)
				\tkzLabelPoints[below](B,C,D,E,M)
				\tkzLabelPoints[above](A)
				\tkzLabelPoints[left](H)
				\tkzLabelPoints[right](I)
				\tkzDrawPoints(A,B,C,D,E,I,M) 
				
				\end{tikzpicture}
			\end{center}
			
			\begin{enumEX}{1}
				
				\item Xét $\triangle ABM$ và $\triangle ACM$ có 
				\\ $AM$ chung, $AB=AC$ (giả thuyết), $BM=CM$ ($M$ là trung điểm).
				\\ Vậy $\triangle ABM = \triangle ACM$ (c.c.c), suy ra $\widehat{BAM}=\widehat{CAM}$ nên $AM$ là phân giác góc $BAC$.
				\item Ta có $\widehat{ABC}=\widehat{ACB}$ ( do $\triangle ABM = \triangle ACM$).
				\\ $\Rightarrow \widehat{ABC}+\widehat{BAC}=\widehat{ACB}+\widehat{BAC}\Rightarrow \widehat{ACE}=\widehat{ABD}$.
				\\ Xét $\triangle ABD$ và $\triangle ACE$ có
				\\ $BD=CE$ (giả thuyết), $\widehat{ABD}=\widehat{ACE}$ (chứng minh trên), $AB=AC$ ($\triangle ABM = \triangle ACM$).
				\\ Vậy  $\triangle ABD = \triangle ACE$ (c.g.c).
				\\ Xét $\triangle ABH$ và $\triangle ACI$ có
				\\ $\widehat{AHB}=\widehat{AIC}=90^\circ$ (giả thuyết), $AB=AC$, $\widehat{BAH}=\widehat{CAI}$ ($\triangle ABD = \triangle ACE$).
				\\ Vậy $\triangle ABH = \triangle ACI$ (cạnh huyền-góc nhọn).
				\\ Suy ra $AH=AI$ mà $AD=AE$ ($\triangle ABD = \triangle ACE$).
				\\ Nên $AD-AH=AE-AI\Rightarrow DH=EI$.
				\item Ta có $\widehat{BAC}=90^\circ \Rightarrow \widehat{BAM}=\widehat{CAM}=45^\circ \Rightarrow \widehat{ABC}=\widehat{ACB}=\widehat{45^\circ}$.
				\\ Lại có $\widehat{ABD}=\widehat{ACB}+\widehat{BAC}=45^\circ+90^\circ=135^\circ$.
				\\ Xét $\triangle AHB$ và $\triangle DHB$ có
				\\ $\widehat{AHB}=\widehat{DHB}=90^\circ$, $AB=DB$ (giả thuyết), $HB$ chung.
				\\ Vậy $\triangle ABH= \triangle DBH$ (cạnh huyền-cạnh góc vuông).
				\\ Suy ra $\widehat{ABH}=\widehat{DBH}\Rightarrow \widehat{ABH}=\widehat{DBH}=\dfrac{1}{2}\widehat{ABD}=67,5^\circ$.
				\\ Do đó $\widehat{BDA}=90^\circ-67,5^\circ=22,5^\circ.$
			\end{enumEX}
			
		}
	\end{ex}
	
	\begin{ex}%[Đề kiểm tra học kỳ I, THCS Ngô Sĩ Liên, 2018]%[Đỗ Đức Thảo, dự án 7EX-1-2018]%[7D1G1]
		Tìm $ x \in \mathbb{Z} $ để $ \dfrac{3x^2-6x+1}{x-2} \in \mathbb{Z} $.
		\loigiai
		{
			Do $ \dfrac{3x^2-6x+1}{x-2}=3x+\dfrac{1}{x-2} $ nên để $\dfrac{3x^2-6x+1}{x-2} \in \mathbb{Z} $ thì $ \dfrac{1}{x-2} \in \mathbb{Z} $ (do đã có $ 3x \in \mathbb{Z} $).\\
			Để $ \dfrac{1}{x-2} \in \mathbb{Z} $ thì $ x-2 $ phải là ước của $ 1 $, nghĩa là $ x-2=1 $ hoặc $ x-2=-1 $.\\ Giải ra ta được $ x=3 $ hoặc $ x=1 $.
		}
	\end{ex}
