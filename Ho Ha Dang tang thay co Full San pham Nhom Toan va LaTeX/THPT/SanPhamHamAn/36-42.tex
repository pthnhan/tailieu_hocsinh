\begin{ex}%[WTT Bài toán hàm ẩn, Đinh Thanh Hoàng]%[2D1K1-2]
	\impicinpar{
		Cho hàm số $y=f(x)$. Đồ thị hàm số $y=f'(x)$ như hình bên và $f(-2)=f(2)=0$. Hàm số $g(x)=\left[f(3-x)\right]^2$ nghịch biến trên khoảng nào trong các khoảng sau?
		\motcot
		{$(-2;-1)$}
		{$(1;2)$}
		{\True $(2;5)$}
		{$(5;+\infty)$}
	}{
		\begin{tikzpicture}[scale=.65,>=stealth]
			%\tikzset{label style/.style={font=\footnotesize}}
			\draw[->] (-3,0)--(3,0) node[above left]{\footnotesize $x$};
			\draw[->] (0,-4)--(0,2) node[below left] {\footnotesize $y$};
			\filldraw (0,0) circle(2pt) node [below left] {\footnotesize $O$};	
			\filldraw (-2,0) circle(2pt) node [below left] {\footnotesize $-2$};	
			\filldraw (1,0) circle(2pt) node [below] {\footnotesize $1$};	
			\filldraw (2,0) circle(2pt) node [below] {\footnotesize $2$};
			\begin{scope}
				\clip (-3,-4) rectangle (3,2);
				\draw[samples=100,smooth,variable=\x] plot (\x,{-0.5*(\x+2)*(\x-1)*(\x-2)});
			\end{scope}
		\end{tikzpicture}
	}
	\loigiai{
		Dựa vào đồ thị hàm số $y=f'(x)$, suy ra bảng biến thiên của hàm số $f(x)$ như sau
		\begin{center}
			\begin{tikzpicture}
				\tkzTabInit[nocadre=false]
				{$x$ /1,$y'$ /1,$y$ /2}
				{$-\infty$, $-2$,$1$,$2$,$+\infty$}
				\tkzTabLine{,+,$0$,-,$0$,+,$0$,-,}
				\tkzTabVar{-/ $-\infty$,+ /$0$,-/$y(1)$,+/$0$,-/$-\infty$}
			\end{tikzpicture}
		\end{center}
		Từ bảng biến thiên suy ra $f(x)\leqslant 0, \forall x\in \mathbb{R}$. \\
		Ta có $g'(x)=-2f'(3-x)\cdot f(3-x)$. \\
		Xét $g'(x)<0 \Leftrightarrow f'(3-x)\cdot f(3-x)>0 \Leftrightarrow \left\{\begin{aligned}
		&f'(3-x)<0\\
		&f(3-x)<0\\
		\end{aligned}\right. \Leftrightarrow \left[\begin{aligned}
		&-2<3-x<1\\
		&3-x>2\\
		\end{aligned}\right. \Leftrightarrow \left\{\begin{aligned}
		&2<x<5\\
		&x<1.\\
		\end{aligned}\right. $ \\
		Suy ra hàm số $g(x)$ nghịch biến trên các khoảng $(-\infty;1)$, $(2;5)$.}
\end{ex}

\begin{ex}%[WTT Bài toán hàm ẩn, Đinh Thanh Hoàng]%[2D1G1-2]
	\impicinpar{
		Cho hàm số $y=f(x)$. Đồ thị hàm số $y=f'(x)$ như hình bên. Hàm số $g(x)=f\left(|3-x|\right)$ đồng biến trên khoảng nào trong các khoảng sau?
		\motcot
		{$(-\infty;-1)$}
		{\True $(-1;2)$}
		{$(2;3)$}
		{$(4;7)$}
	}{
		\begin{tikzpicture}[scale=.625,>=stealth]
			%\tikzset{label style/.style={font=\footnotesize}}
			\draw[->] (-2,0)--(5,0) node[above left]{\footnotesize $x$};
			\draw[->] (0,-3)--(0,3) node[below left] {\footnotesize $y$};
			\filldraw (0,0) circle(2pt) node [below left] {\footnotesize $O$};	
			\filldraw (-1,0) circle(2pt) node [above left] {\footnotesize $-1$};	
			\filldraw (1,0) circle(2pt) node [above right] {\footnotesize $1$};	
			\filldraw (4,0) circle(2pt) node [above left] {\footnotesize $4$};
			\begin{scope}
				\clip (-2,-3) rectangle (5,3);
				\draw[samples=100,smooth,variable=\x] plot (\x,{0.25*(\x+1)*(\x-1)*(\x-4)});
			\end{scope}
		\end{tikzpicture}
	}
	\loigiai{
		Dựa vào đồ thị, suy ra $f'(x)>0 \Leftrightarrow \left[\begin{aligned}
		&-1<x<1\\
		&x>4\\
		\end{aligned}\right. $ và $f'(x)<0 \Leftrightarrow \left[\begin{aligned}
		&x<-1\\
		&1<x<4\\
		\end{aligned}\right. $. \\
		Với $x>3$ khi đó $g(x)=f(x-3)\Rightarrow g'(x)=f'(x-3)>0 \Leftrightarrow \left[\begin{aligned}
		&-1<x-3<1\\
		&x-3>4\\
		\end{aligned}\right. \Leftrightarrow \left[\begin{aligned}
		&2<x<4\\
		&x>7.\\
		\end{aligned}\right. $ \\
		$\Rightarrow$ hàm số $g(x)$ đồng biến trên các khoảng $(3;4)$, $(7;+\infty)$. \\
		Với $x<3$ khi đó $g(x)=f(3-x)\Rightarrow g'(x)=-f'(3-x)>0 \Leftrightarrow f'(3-x)<0$ \\
		$ \Leftrightarrow \left[\begin{aligned}
		&3-x<-1\\
		&1<3-x<4\\
		\end{aligned}\right. \Leftrightarrow \left[\begin{aligned}
		&x>4 \;\text{(loại)}\\
		&-1<x<2.\\
		\end{aligned}\right. $\\
		$\Rightarrow$ hàm số $g(x)$ đồng biến trên khoảng $(-1;2)$.}
\end{ex}

\begin{ex}%[WTT Bài toán hàm ẩn, Đinh Thanh Hoàng]%[2D1G1-2]
	\immini{
		Cho hàm số $y=f(x)$. Đồ thị hàm số $y=f'(x)$ như hình bên. Hàm số $g(x)=f\left(\sqrt{x^2+2x+2}\right)$ nghịch biến trên khoảng nào trong các khoảng sau?
		\motcot
		{\True $\left(-\infty;-1-2\sqrt{2}\right)$}
		{$(-\infty;1)$}
		{$\left(1;2\sqrt{2}-1\right)$}
		{$\left(2\sqrt{2}-1;+\infty \right)$}
	}{
		\begin{tikzpicture}[scale=.475,>=stealth]
			%\tikzset{label style/.style={font=\footnotesize}}
			\draw[->] (-2,0)--(4,0) node[above left]{\footnotesize $x$};
			\draw[->] (0,-4)--(0,4) node[below left] {\footnotesize $y$};
			\filldraw (0,0) circle(2pt) node [below left] {\footnotesize $O$};	
			\filldraw (-1,0) circle(2pt) node [above left] {\footnotesize $-1$};	
			\filldraw (1,0) circle(2pt) node [above right] {\footnotesize $1$};	
			\filldraw (3,0) circle(2pt) node [above left] {\footnotesize $3$};
			\begin{scope}
				\clip (-2,-4) rectangle (4,4);
				\draw[samples=100,smooth,variable=\x] plot (\x,{(\x+1)*(\x-1)*(\x-3)});
			\end{scope}
		\end{tikzpicture}
	}
	\loigiai{
		Dựa vào đồ thị, suy ra $f'(x)=0 \Leftrightarrow \left[\begin{aligned}
		&x=-1\\
		&x=1\\
		&x=3\\
		\end{aligned}\right. $. Ta có $g'(x)=\dfrac{x+1}{\sqrt{x^2+2x+2}}f'\left(\sqrt{x^2+2x+2}\right)$; \\
		$g'(x)=0 \Leftrightarrow \left[\begin{aligned}
		&x+1=0\\
		&f'\left(\sqrt{x^2+2x+2}\right)=0\\
		\end{aligned}\right. \overset{\text{theo đồ thị}\; f'(x)}{\longleftrightarrow}\left[\begin{aligned}
		&x+1=0\\
		&\sqrt{x^2+2x+2}=1\\
		&\sqrt{x^2+2x+2}=3\\
		\end{aligned}\right. \Leftrightarrow \left[\begin{aligned}
		&x=-1 \;\text{(nghiệm bội ba)}\\
		&x=-1-2\sqrt{2}\\
		&x=-1+2\sqrt{2}.\\
		\end{aligned}\right. $ \\
		Bảng xét dấu $g'(x)$:
		\begin{center}
			\begin{tikzpicture}
				\tkzTabInit[nocadre=false]
				{$x$ /1,$g'(x)$ /1}%,$g(x)$ /2}
				{$-\infty$, $-1-2\sqrt{2}$,$-1$,$-1+2\sqrt{2}$,$+\infty$}
				\tkzTabLine{,-,$0$,+,$0$,-,$0$,+,}
				%\tkzTabVar{-/ $-\infty$,+ /$0$,-/$y(1)$,+/$0$,-/$-\infty$}
			\end{tikzpicture}
		\end{center}
		\textbf{Chú ý:} Cách xét dấu $g'(x)$ như sau: Ví dụ xét trên khoảng $\left(-1;-1+2\sqrt{2}\right)$ ta chọn $x=0$. Khi đó $g'(0)=\dfrac{1}{\sqrt{2}}f'\left(\sqrt{2}\right)<0$ vì dựa vào đồ thị $f'(x)$ ta thấy tại $x=\sqrt{2}\in (1;3)$ thì $f'\left(\sqrt{2}\right)<0$. Các nghiệm của phương trình $g'(x)=0$ là nghiệm bội lẻ nên qua nghiệm đổi dấu.}
\end{ex}

\begin{ex}%[WTT Bài toán hàm ẩn, Đinh Thanh Hoàng]%[2D1G1-2]
	\impicinpar{
		Cho hàm số $y=f(x)$. Đồ thị hàm số $y=f'(x)$ như hình bên. 	Hàm số $g(x)=f\left(\sqrt{x^2+2x+3}-\sqrt{x^2+2x+2}\right)$ đồng biến trên khoảng nào sau đây?

			\item $0<u=\sqrt{x^2+2x+3}-\sqrt{x^2+2x+2}=\dfrac{1}{\sqrt{(x+1)^2+2}+\sqrt{(x+1)^2+1}}\leqslant \dfrac{1}{\sqrt{2}+1}<1$ $\xrightarrow{\text{theo đồ thị}\; f'(x)}f'(u)>0, \forall x\in \mathbb{R}$. \hfill $(2)$
		\end{itemize}
		Từ $(1)$ và $(2)$, suy ra dấu của $g'(x)$ phụ thuộc vào dấu của nhị thức $x+1$ (ngược dấu)\\
