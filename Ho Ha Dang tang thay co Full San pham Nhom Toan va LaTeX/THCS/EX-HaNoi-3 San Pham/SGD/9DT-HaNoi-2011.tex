\begin{ex}%[2,5 điểm]%[Đề thi vào 10 TP Hà Nội năm 2011]%[9D1K8]
Cho  $A=\dfrac{\sqrt{x}}{\sqrt{x}-5}-\dfrac{10\sqrt{x}}{x-25}-\dfrac{5}{\sqrt{x}+5}$ , với $x \neq 0$ và $x \geq 25$.
\begin{enumerate}
	\item 	Rút gọn biểu thức $A$.
	\item	Tìm giá trị của $A$ khi $x = 9$.
	\item 	Tìm $x$ để $A < \dfrac{1}{3}$.
\end{enumerate}
\loigiai{
	\begin{enumerate}
	\item 
	Với $x \neq 0$ và $x \neq 25$, ta có 
	\begin{align*}
	 A&=\dfrac{\sqrt{x}}{\sqrt{x}-5}-\dfrac{10\sqrt{x}}{x-25}-\dfrac{5}{\sqrt{x}+5}\\& =\dfrac{\sqrt{x}(\sqrt{x}+5)}{x-25} - \dfrac{10 \sqrt{x}}{x-25}- \dfrac{5(\sqrt{x}-5)}{x-25}\\&= \dfrac{x+5\sqrt{x} -10\sqrt{x} -5\sqrt{x}+25}{x-25} \\&= \dfrac{x-10\sqrt{x}+25}{x-25}\\&= \dfrac{(\sqrt{x}-5)^2}{(\sqrt{x}-5)(\sqrt{x}+5)}\\&=\dfrac{\sqrt{x}-5}{\sqrt{x}+5}.
	\end{align*}
	\item Với $x=9\Rightarrow A= \dfrac{\sqrt{9}-5}{\sqrt{9}+5}=-\dfrac{1}{4}$
	\item $A< \dfrac{1}{3} \Leftrightarrow \dfrac{\sqrt{x}-5}{\sqrt{x}+5} < \dfrac{1}{3} \Leftrightarrow 2\sqrt{x} <20 \Leftrightarrow 0 \leq x <100.$ \\
	Kết hợp với kiều kiện xác định ta có $\heva{& 0<x<100\\& x\neq 25}.$
	\end{enumerate}
}
\end{ex} 
 
\begin{ex}%[2,5 điểm] %[Đề thi vào 10 TP Hà Nội năm 2011]%[9D3B6]
\textit{Giải bài toán sau bằng cách lập phương trình hoặc hệ phương trình:}\\
	Một đội xe theo kế hoạch chở hết 140 tấn hàng trong một số ngày quy định. Do mỗi ngày đội đó chở vượt mức 5 tấn nên đội đã hoàn thành kế hoạch sớm hơn thời gian quy định 1 ngày và chở thêm được 10 tấn. Hỏi theo kế hoạch đội xe chở hàng hết bao nhiêu ngày?
\end{ex}  
\loigiai{
Gọi $a$ (tấn), $a\geq 0$: số tấn hàng mỗi ngày.\\
Gọi $b$ (ngày), $b \in \mathbb{N}^*$: số ngày.\\
Theo đề bài ta có $$\heva{& a\times b=140\\& (a+5)(b-1)=140+10} \Leftrightarrow \heva{& ab=140\\& 5b-a=15} \Rightarrow 5b^2-15b -140=0 \Leftrightarrow \hoac{&b=7\\&b=-4 \, \text{(loại)}}.$$
Vậy đội xe chở hết hàng theo kế hoạch trong $7$ ngày.

}
\begin{ex}%[1,0 điểm]%[Đề thi vào 10 TP Hà Nội năm 2011]%[9D4B6]%[9D4B3]
Cho parabol $(P)$: $y = x^2$ và đường thẳng $(d)$: $y =2x-m^2+9$.
\begin{enumerate}
\item Tìm tọa độ các giao điểm của parabol $(P)$ và đường thẳng $(d)$ khi $m=1$. 
\item	Tìm $m$ để đường thẳng $(d)$ cắt parabol $(P)$ tại hai điểm nằm về hai phía của trục tung.	
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Phương trình hoành độ giao điểm của $(P)$ và $(d)$ khi $m=1$ là $x^2=2x+8 \Leftrightarrow \hoac{& x=-2\\&x=4.}$
	\begin{itemize}
		\item Với $x=-2 \Rightarrow y=4 \Rightarrow A(-2;4)$
		\item  Với $x=4 \Rightarrow y=16 \Rightarrow B(4;16)$
	\end{itemize}
Vậy tọa độ giao điểm của $(P)$ và $(d)$ là $A(-2;4); B(4;16).$
\item Phương trình hoành độ giao điểm của $(P)$ và $(d)$ là $x^2=2x-m^2+9 \Leftrightarrow x^2-2x+m^2-9=0\, (1).$\\
Yêu cầu bài toán tương đương với phương trình $(1)$ có hai nghiệm phân biệt trái dấu $\Leftrightarrow a\times c <0 \Leftrightarrow m^2-9<0 \Leftrightarrow -3<m<3.$\\
Vậy $-3<m<3.$
\end{enumerate}
}
\end{ex}

\begin{ex}[0,5 điểm]
Với $x > 0$, tìm giá trị nhỏ nhất của biểu thức: $M =4x^2-3x+\dfrac{1}{4x}+2011$.
\loigiai{
Ta có $M=4\left( x-\dfrac{1}{2}\right)^2 +x +\dfrac{1}{4x} +2010 \geq 2 \sqrt{x\cdot \dfrac{1}{4x}}+2010 =2011.$ Đẳng thức xảy ra khi $ x =\dfrac{1}{2}.$\\
Vậy giá trị nhỏ nhất của $M$ là $2011$.
}
\end{ex}
