\begin{bt}%[Đề HK1, Nguyễn Khuyến, Hồ Chí Minh 17-18]%[Quỷ Lệ]%[0D3K2-3] 
Giải phương trình $\dfrac{3x+4}{x-2}-\dfrac{1}{x+2}=\dfrac{4}{x^2-4}+3$.
\loigiai{Điều kiện: $\mathscr{D}=\mathbb{R}\setminus \{\pm 2\}$.\\
Khi đó, phương trình đã cho tương đương với \begin{eqnarray*}
& & \dfrac{(3x+4)(x+2)}{(x-2)(x+2)}-\dfrac{x-2}{(x-2)(x+2)}=\dfrac{4}{(x-2)(x+2)}+\dfrac{3(x-2)(x+2)}{(x-2)(x+2)}\\ &\Leftrightarrow &(3x+4)(x+2)-(x-2)=4+3(x-2)(x+2)\\ &\Leftrightarrow &9x+18=0 \Leftrightarrow x=-2\,(\text{ không thỏa mãn}).
\end{eqnarray*}
Vậy phương trình vô nghiệm.
}
\end{bt}

 \begin{bt}%[Đề HK1, Nguyễn Khuyến, Hồ Chí Minh 17-18]%[Quỷ Lệ]%[0D3K2-4]
Giải phương trình  $\sqrt{2x^2-8x+12}+4x=x^2-6$.
\loigiai{Ta có  $2x^2-8x+12\ge 0,\,\forall x\in\mathbb{R} \Rightarrow$ Tập xác định: $\mathscr{D}=\mathbb{R}$.\\ Phương trình đã cho có thể viết lại \[2x^2-8x+12-2\sqrt{2x^2-8x+12}-24=0\tag{1}.\]
Đặt $t=\sqrt{2x^2-8x+12}\ge 0,$ phương trình $(1)$ trở thành $$t^2-2t-24=0\Leftrightarrow\hoac{&t=6\\&t=-4\,(\text{loại}).}$$ Với $t=6$, ta có $\sqrt{2x^2-8x+12}=6\Leftrightarrow x^2-4x-12=0\Leftrightarrow\hoac{&x_1=6\\&x_2=-2.}$\\ Vậy phương trình có hai ngiệm $x=-2;x=6.$
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Khuyến, Hồ Chí Minh 17-18]%[Quỷ Lệ]%[0D3K2-6]
Giải và biện luận phương trình $m^2x-3=4x-(m-1),$\,$m$ là tham số.
\loigiai{Tập xác định: $\mathscr{D}=\mathbb{R}$.\\ Phương trình đã cho được viết lại thành \[(m-2)(m+2)x=4-m\tag{1}\]\begin{itemize}
\item Xét $m=2,$ phương trình $(1)$ trở thành $0x=2$\,(vô nghiệm).\\ $\Rightarrow $ Với $ m=2$ thì phương trình vô nghiệm.
\item Xét $m=-2$, phương trình $(1)$ trở thành $0x=6$\,(vô nghiệm).\\ $\Rightarrow$ Với $m=-2$ thì phương trình vô nghiệm.
\item Xét $m\neq \pm 2$, phương trình $(1)$ trở thành $x=\dfrac{4-m}{(m-2)(m+2)}.$\\ $\Rightarrow$ Với $m\neq \pm 2$ thì phương trình có nghiệm duy nhất $x=\dfrac{4-m}{(m-2)(m+2)}.$
\end{itemize}
\textbf{Kết luận.}
\begin{itemize}
\item Với $m=\pm 2$, phương trình vô nghiệm.
\item Với $m\neq \pm 2$, phương trình có nghiệm duy nhất $x=\dfrac{4-m}{(m-2)(m+2)}.$
\end{itemize}

}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Khuyến, Hồ Chí Minh 17-18]%[Quỷ Lệ]%[0D3B2-5]
Tìm $m$ để phương trình $3x^2-2x+m-1=0$ có hai nghiệm dương phân biệt.
\loigiai{
Phương trình có hai nghiệm dương phân biệt khi và chỉ khi
$$\heva{&\Delta'=4-3m>0\\&S=\dfrac{2}{3}>0\\&P=\dfrac{m-1}{3}>0}\Leftrightarrow\heva{&m<\dfrac{4}{3}\\&m>1}\Leftrightarrow 1<m<\dfrac{4}{3}.$$ Vậy, $1<m<\dfrac{4}{3}$ là giá trị cần tìm.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Khuyến, Hồ Chí Minh 17-18]%[Quỷ Lệ]%[0D3K2-5]
Cho phương trình $x^2-(2m+2)x+m^2+4=0$. Tìm $m$ để phương trình có hai nghiệm phân biệt $x_1,x_2$ thỏa điều kiện $x_1=2x_2.$
\loigiai{Phương trình có hai nghiệm phân biệt $x_1,x_2$ khi và chỉ khi \[\Delta'=(m+1)^2-\left(m^2+4\right)>0\Leftrightarrow 2m-3>0\Leftrightarrow m>\dfrac{3}{2}\tag{1}.\]
Khi đó, theo Vi-ét ta có $\heva{&x_1+x_2=2m+2&\quad (2)\\&x_1x_2=m^2+4&\quad (3).}$\\ Thay $x_1=2x_2$ vào $(2)$, ta được $x_2=\dfrac{2(m+1)}{3}\Rightarrow x_1=\dfrac{4(m+1)}{3}.$ \\Thay vào $(3)$, ta được \begin{eqnarray*}
& & \dfrac{4(m+1)}{3}\cdot\dfrac{2(m+1)}{3}=m^2+4\\ &\Leftrightarrow
 & 8(m+1)^2=9(m^2+4)\\ &\Leftrightarrow & m^2-16m+28=0\Leftrightarrow \hoac{&m=14\\&m=2}(\text{thỏa mãn điều kiện } (1)). 
\end{eqnarray*}
Vậy $m=2;m=14$ là giá trị cần tìm.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Khuyến, Hồ Chí Minh 17-18]%[Quỷ Lệ]%[0D3K3-4]
Giải hệ phương trình $\heva{&x^2y+xy^2=6\\&\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{3}{2}.}$
\loigiai{Điều kiện: $xy\neq 0.$\\ Hệ phương trình được viết lại thành \[\heva{&xy(x+y)=6\\&2(x+y)=3xy}\tag{I}\] Đặt $u=x+y;\,v=xy$, hệ phương trình $(I)$ trở thành $$\heva{&uv=6\\&2u=3v}\Leftrightarrow \heva{&\dfrac{3}{2}v^2=6\\&u=\dfrac{3}{2}v}\Leftrightarrow\hoac{&v^2=4\\&u=\dfrac{3}{2}v}\Leftrightarrow\heva{&\hoac{&v=2\\&v=-2}\\&u=\dfrac{3}{2}v}\Leftrightarrow\hoac{&\heva{&u=3\\&v=2}\\&\heva{&u=-3\\&v=-2.}}$$
\begin{itemize}
\item Với $u=3;\,v=2$ thì $x+y=3;\,xy=2$. Khi đó, $x,y$ là hai nghiệm của phương trình $$X^2-3X+2=0\Leftrightarrow \hoac{&X=1\\&X=2.}$$ Đối chiếu điều kiện, $(x;y)=(1;2)$ và $(x;y)=(2;1)$ là hai nghiệm của hệ.
\item Với $u=-3;\,v=-2$ thì $x+y=-3;\,xy=-2$. Khi đó, $x,y$ là hai nghiệm của phương trình $$X^2+3X-2=0\Leftrightarrow X=\dfrac{-3\pm\sqrt{17}}{2}.$$ Đối chiếu điều kiện, $(x;y)=\left(\dfrac{-3-\sqrt{17}}{2};\dfrac{-3+\sqrt{17}}{2}\right)$ và $(x;y)=\left(\dfrac{-3+\sqrt{17}}{2};\dfrac{-3-\sqrt{17}}{2}\right)$ là hai nghiệm của hệ
 \end{itemize}
 Vậy hệ phương trình có $4$ nghiệm như trên.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Khuyến, Hồ Chí Minh 17-18]%[Quỷ Lệ]%[0H2B3-1]
Cho $\triangle ABC$, có $AB=6,BC=8,\widehat{B}=120^\circ$. Tính diện tích $S$, bán kính đường tròn nội tiếp $r$ của $\triangle ABC$.
\loigiai{
Diện tích tam giác $ABC$ là $$S=\dfrac{1}{2}\cdot AB\cdot BC\sin 120^\circ=\dfrac{1}{2}\cdot 6\cdot 8\cdot\dfrac{\sqrt{3}}{2}=12\sqrt{3}.$$
Áp dụng định lí Cô-sin cho tam giác $ABC$, ta có
\[CA^2=AB^2+BC^2-2AB\cdot BC\cos\widehat{ABC}=6^2+8^2-2\cdot 6\cdot 8\cdot \left(-\dfrac{1}{2}\right)=148,\]từ đó suy ra $CA=2\sqrt{37}$ và $p=\dfrac{AB+BC+CA}{2}=7+\sqrt{37}.$\\ 
Bán kính đường tròn nội tiếp tam giác $ABC$ là $$r=\dfrac{S}{p}=\dfrac{12\sqrt{3}}{7+\sqrt{37}}=-\sqrt{111}+7\sqrt{3}.$$
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Khuyến, Hồ Chí Minh 17-18]%[Quỷ Lệ]%[0H2K3-1]
Trong mặt phẳng với hệ tọa độ $Oxy$, lấy các điểm $A(-1;3)$,\,$B(0;-4)$,\,$C(5;1)$. Tính chu vi và diện tích tam giác $ABC$.
\loigiai{Ta có $\vec{AB}=(1;-7),\,\vec{AC}=(6;-2),\,\vec{BC}=(5;5)$. Do đó, $AB=5\sqrt{2};\, AC=2\sqrt{10};\, BC=5\sqrt{2}$.\\ Chu vi tam giác $ABC$ là $AB+AC+BC=10\sqrt{2}+2\sqrt{10}$.\\ Từ đó, $p=5\sqrt{2}+\sqrt{10}.$ Diện tích tam giác $ABC$ là $$S=\sqrt{p(p-AB)(p-BC)(p-AC)}=\sqrt{(5\sqrt{2}+\sqrt{10})\cdot\sqrt{10}\cdot\sqrt{10}\cdot\left(5\sqrt{2}-\sqrt{10}\right)}=20.$$

}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Khuyến, Hồ Chí Minh 17-18]%[Quỷ Lệ]%[0H2B2-1] 
Cho tam giác $ABC$ có $AB=5,\,AC=6,\,BC=\sqrt{31}$. Tính tích vô hướng $\vec{AB}\cdot\vec{AC}$ và số đo góc $A$.
\loigiai{
Áp dụng định lý Cô-sin, ta có $$\cos \widehat{BAC}=\dfrac{AB^2+AC^2-BC^2}{2AB\cdot AC}=\dfrac{5^2+6^2-31}{2\cdot 5\cdot 6}=\dfrac{1}{2}.$$ Từ đó, $\widehat{BAC}=60^\circ$ và tích vô hướng $\vec{AB}\cdot\vec{AC}$ là $$\vec{AB}\cdot\vec{AC}=AB\cdot AC\cdot \cos \widehat{BAC}=5\cdot 6\cdot \dfrac{1}{2}=15.$$
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Khuyến, Hồ Chí Minh 17-18]%[Quỷ Lệ]%[0H2G2-4] 
Cho đoạn thẳng $AB=2a$. Tìm tập hợp các điểm di động $M$ biết $\vec{MA}\cdot\vec{MB}=a^2$.
\loigiai{Gọi $I$ là trung điểm của $AB$. Ta có
$$a^2=\vec{MA}\cdot\vec{MB}=\left(\vec{MI}+\vec{IA}\right)\left(\vec{MI}+\vec{IB}\right)=\vec{MI}^2+\vec{MI}\cdot\left(\vec{IA}+\vec{IB}\right)+\vec{IA}\cdot\vec{IB}=\vec{MI}^2-\vec{IA}^2,$$ do $\vec{IA}^2=\left(\dfrac{AB}{2}\right)^2=a^2$ và $\vec{IA}+\vec{IB}=\vec{0}$ nên suy ra $\vec{MI}^2=2a^2$ hay $MI^2=2a^2$. \\ Do đó, tập hợp các điểm $M$ là đường tròn tâm $I$ cố định bán kính $R=a\sqrt{2}.$

}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Khuyến, Hồ Chí Minh 17-18]%[Quỷ Lệ]%[0D3G2-3]
Định $m$ để phương trình $\dfrac{(m-1)x+m-2}{x-2}=0$ có một nghiệm nguyên duy nhất.
\loigiai{
Điều kiện: $x\neq 2.$\\ Phương trình đã cho trở thành \[(m-1)x+m-2=0,\text{ với }x\neq 2.\tag{1}\]
Phương trình đã cho có nghiệm duy nhất khi và chỉ khi $(1)$ có nghiệm duy nhất khác $2$, hay
\[\heva{&m-1\neq 0\\&\dfrac{2-m}{m-1}\neq 2}\Leftrightarrow \heva{&m\neq 1\\&m\neq \dfrac{4}{3}.}\tag{2}\]
Khi đó, $x_0=\dfrac{2-m}{m-1}=\dfrac{1}{m-1}-1\in\mathbb{Z}$ thì $m-1\in \text{Ư}(1)=\{\pm 1\}\Rightarrow \hoac{&m=0\\&m=2.}\,(\text{thỏa mãn }(2)).$\\
Vậy $m=0;m=2$ là giá trị cần tìm.

}
\end{bt}
