\begin{ex}%[GHK1, THPT Nhân Chính, Hà Nội, 2018-2019]%[Trần Bá Huy, EX-DT-10_11-18]%[0D1B1-2]
Trong các mệnh đề sau, mệnh đề nào đúng?
\choice
{$\forall x\in \mathbb{N}:\ x^2\ \vdots\ x$}
{$\forall x\in \mathbb{R}:\ x^2\geq x$}
{$\exists x\in \mathbb{R}:\ x^2+1<2x$}
{\True $\exists x\in \mathbb{R}:\ x^2=x+1$}
\loigiai{
\begin{itemize}
\item Mệnh đề ``$\forall x\in \mathbb{N}:\ x^2\ \vdots\ x$'' sai khi $x=0$.
\item Mệnh đề ``$\forall x\in \mathbb{R}:\ x^2\geq x$'' sai khi $0<x<1$.
\item Mệnh đề ``$\exists x\in \mathbb{R}:\ x^2+1<2x$'' sai vì $(x-1)^2\geq 0\Leftrightarrow x^2+1\geq 2x,\ \forall x\in\mathbb{R}$.
\item Mệnh đề đúng là ``$\exists x\in \mathbb{R}:\ x^2=x+1$'' vì phương trình 
$$x^2=x+1\Leftrightarrow x^2-x-1=0$$
có $ac<0$ nên luôn có nghiệm trong tập $\mathbb{R}$.
\end{itemize}
}
\end{ex}

\end{ex}

\loigiai{

}
