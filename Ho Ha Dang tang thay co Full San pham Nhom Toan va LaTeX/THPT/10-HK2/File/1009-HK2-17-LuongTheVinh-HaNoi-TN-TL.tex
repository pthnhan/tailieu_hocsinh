\begin{ex}%[HK2 (2016-2017), THPT Lương Thế Vinh, Hà Nội]%[0D6G3]%[Hoàng Mạnh Linh]

Cho Elip ($E$) đi qua điểm $A(-3;0)$ và có tâm sai $e = \dfrac{5}{6}$. Tiêu cự của $(E)$ là
\choice
{$10$}
{$\dfrac{5}{3}$}
{\True $5$}
{$\dfrac{10}{3}$}	
\loigiai{

\begin{ex}%[HK2 (2016-2017), THPT Lương Thế Vinh, Hà Nội]%[0D4Y5]%[Hoàng Mạnh Linh]
Giá trị $x = 3$ thuộc tập nghiệm của bất phương trình nào sau đây ?
\choice
{$\dfrac{x^2 - x + 1}{x - 1} \geq x + 1$}
{$|2x - 1| > x^2$}
{\True $x^2 - \sqrt{x^2 + 1 } < 6$}
{$2x^2 - 5x + 2$}	
\loigiai{
Thử $x = 3$ vào từng phương án ta được:
\begin{itemize}
	\item $\dfrac{3^2 - 3 + 1}{3 - 1} \geq 3 + 1 \Leftrightarrow \dfrac{7}{2} \geq 4 \Leftrightarrow 3,5 \geq 4$ (vô lý). Vậy $x = 3$ không thuộc tập nghiệm bất phương trình $\dfrac{x^2 - x + 1}{x - 1} \geq x + 1$
	\item $|2.3 - 1| > 3^2 \Leftrightarrow |5| > 9 \Leftrightarrow 5 > 9$ (vô lý). Vậy $x = 3$ không thuộc tập nghiệm bất phương trình $|2x - 1| > x^2$ .
	\item $3^2 - \sqrt{3^2 + 1} < 6 \Leftrightarrow 9 - \sqrt{9 + 1} < 6 \Leftrightarrow 9 - \sqrt{10} < 6$ (Thỏa mãn). Vậy $x = 3$ thuộc tập nghiệm bất phương trình $x^2 - \sqrt{x^2 + 1 } < 6$ .
	\item Với $2x^2 - 5x + 2$ không phải là bất phương trình đang xét
\end{itemize}
}

\begin{bt}%[HK2 (2016-2017), THPT ABC, Hà Nội]%[0D4K2]%[Đỗ Vũ Minh Thắng]
	Giải bất phương trình $\sqrt{x^2+2x-3}\ge 2x-2$.
	\loigiai{
		Tập xác định: $\mathscr{D}=(\infty ;3] \cup [1; +\infty)$.\\
		Trường hợp $2x-2<0\Leftrightarrow x<1$ thì bất phương trình luôn nghiệm đúng.\\
		Trường hợp $2x-2>0\Leftrightarrow x>1$ thì bất phương trình tương đương với
		\[
		x^2+2x-3\ge (2x-2)^2\Leftrightarrow 3x^2-3x+7\le 0\Leftrightarrow 1\le x\le \dfrac{7}{3}
		\]	
		Vậy $S=(\infty ;3] \cup \left [1; \dfrac{7}{3}\right ]$.
	}
\end{bt}

	}
