\begin{bt}%[HK2 (2015-2016), THPT Chu Văn An, Hà Nội]%[0D3B2]%[Dương BùiĐức]
Giải phương trình $\sqrt{x+8}-\sqrt{1+x}=1$.
\loigiai{
Điều kiện xác định $\heva{&x+8\geq 0\\ &1+x\geq 0}\Leftrightarrow x\geq -1\Rightarrow \mathscr{D}=[-1;+\infty)$.\\
Phương trình tương đương $\sqrt{x+8}=\sqrt{1+x}+1\Leftrightarrow x+8=x+2+2\sqrt{x+1}$\\
$\Leftrightarrow \sqrt{x+1}=3\Leftrightarrow x+1=9\Leftrightarrow x=8$ (thỏa mãn).
}
\end{bt}
\begin{bt}%[HK2 (2015-2016), THPT Chu Văn An, Hà Nội]%[0D4B4]%[0D4K4]%[Dương BùiĐức]
Cho bất phương trình $x^{2}-4x+\sqrt{x^{2}+4x+1}+2-m>0$ ($1$) (với $m$ là tham số).
\begin{enumerate}
\item Giải bất phương trình $(1)$ khi $m=3$.
\item Tìm các giá trị của $m$ để bất phương trình $(1)$ có nghiệm $x\in [4;5]$.
\end{enumerate}
\loigiai{
Tập xác định $x^{2}+4x+1\geq 0\Rightarrow \mathscr{D}=(-\infty;-2-2\sqrt{2}]\cup [-2+2\sqrt{2};+\infty)$.
\begin{enumerate}
\item Với $m=3$, ta có $x^{2}-4x+\sqrt{x^{2}+4x+1}-1>0$.\\
Đặt $t=\sqrt{x^{2}+4x+1}$ ($t\geq 0$) $\Rightarrow x^{2}+4x=t^{2}-1$. Khi đó, ta có\\
$t^{2}+t-2>0\Leftrightarrow -2<t<1\Rightarrow 0\leq t<1\Rightarrow \heva{& x^{2}+4x+1\geq 0\\ &x^{2}+4x+1<1}$\\
$\Rightarrow \heva{& x\in(-\infty;-2-2\sqrt{2}]\cup [-2+2\sqrt{2};+\infty)\\ &-4<x<0}\Rightarrow x\in\varnothing$.
\item Xét $y=x^{2}+4x+1$ trên đoạn $[4;5]$. Ta có tọa độ đỉnh $(-2;-3)\Rightarrow$ bảng biến thiên
\begin{center}
\begin{tikzpicture}
\tkzTabInit[nocadre=false,lgt=1,espcl=3]
{$x$ /1,$y$ /1.5}{$4$,$5$}
\tkzTabVar{-/$31$,+/$46$}
\end{tikzpicture}
\end{center}
Đặt $t=\sqrt{x^{2}+4x+1}\Rightarrow \sqrt{31}\leq t\leq \sqrt{46}$. Bất phương trình $(1)$ có dạng
\[
t^{2}+t+1-m>0\Leftrightarrow t^{2}+t+1>m
\]
Xét $f(t)=t^{2}+t+1$ trên đoạn $[\sqrt{31};\sqrt{46}]$, ta có bảng biến thiên
\begin{center}
\begin{tikzpicture}
\tkzTabInit[nocadre=false,lgt=1,espcl=3]
{$x$ /1,$y$ /1.5}{$\sqrt{31}$,$\sqrt{46}$}
\tkzTabVar{-/$32+\sqrt{31}$,+/$47+\sqrt{46}$}
\end{tikzpicture}
\end{center}
Để bất phương trình $(1)$ có nghiệm $x\in[4;5]$ thì $m<47+\sqrt{46}$.
\end{enumerate}
}
\end{bt}
\begin{bt}%[HK2 (2015-2016), THPT Chu Văn An, Hà Nội]%[0D6K3]%[Dương BùiĐức]
Cho $A=\sin 7x-2\sin x(\cos 4x+\cos 6x)-\cos\left(3x-\dfrac{\pi}{2}\right)$. Chứng minh rằng giá trị của biểu thức $A$ không phụ thuộc vào $x$.
\loigiai{
$A=\sin 7x-2\sin x\cos 4x-2\sin x\cos 6x-\sin 3x$\\
$=\sin 7x-(\sin 5x-\sin 3x)-(\sin 7x-\sin 5x)-\sin 3x=0$.
}
\end{bt}
\begin{bt}%[HK2 (2015-2016), THPT Chu Văn An, Hà Nội]%[0D6K3]%[Dương BùiĐức]
Cho $A$, $B$, $C$ là ba góc của tam giác $ABC$. Chứng minh rằng:
\[
\sin A+\sin B-\sin C=4\sin\dfrac{A}{2}\sin\dfrac{B}{2}\cos\dfrac{C}{2}.
\]
\loigiai{
Xét $VT=\sin A+\sin B-\sin C=2\sin\dfrac{A+B}{2}\cos\dfrac{A-B}{2}-2\sin\dfrac{C}{2}\cos\dfrac{C}{2}$\\
$=2\cos\dfrac{C}{2}\cos\dfrac{A-B}{2}-2\sin\dfrac{C}{2}\cos\dfrac{C}{2}=2\cos\dfrac{C}{2}\left(\cos\dfrac{A-B}{2}-\cos\dfrac{A+B}{2}\right)$\\
$=4\sin\dfrac{A}{2}\sin\dfrac{B}{2}\cos\dfrac{C}{2}=VP$.
}
\end{bt}
\begin{bt}%[HK2 (2015-2016), THPT Chu Văn An, Hà Nội]%[0H3B2]%[Dương BùiĐức]
Trong mặt phẳng với hệ tọa độ $Oxy$, cho đường tròn $(C)\colon x^{2}+y^{2}-4x+4y+7=0$. Viết phương trình các tiếp tuyến của $(C)$ biết tiếp tuyến vuông góc với đường thẳng $\Delta\colon 3x-4y+1=0$.
\loigiai{
Xét $(C)\colon x^{2}+y^{2}-4x+4y+7=0\Rightarrow$ tâm $I(2;-2)$, bán kính $R=1$. Vì tiếp tuyến của $(C)$ vuông góc với $\Delta\colon 3x-4y+1=0$ nên tiếp tuyến có dạng $d\colon 4x+3y+m=0$. Áp dụng điều kiện tiếp xúc của $d$ với $(C)$, ta có
\[
\dfrac{|m+2|}{5}=1\Leftrightarrow |m+2|=5\Leftrightarrow\hoac{&m=3\\ &m=-7}
\]
\begin{itemize}
\item Với $m=2\Rightarrow$ phương trình tiếp tuyến $d\colon 4x+3y+2=0$.
\item Với $m=-7\Rightarrow$ phương trình tiếp tuyến $d\colon 4x+3y-7=0$.
\end{itemize}
}
\end{bt}

\begin{bt}%[HK2 (2015-2016), THPT Chu Văn An, Hà Nội]%[0H3B3]%[0H3K3]%[Dương BùiĐức]
Trong mặt phẳng với hệ tọa độ $Oxy$, cho elip $(E)\colon\dfrac{x^{2}}{25}+\dfrac{y^{2}}{16}=1$.
\begin{enumerate}
\item Tìm tọa độ của tiêu điểm trái $F_{1}$ và tiêu điểm phải $F_{2}$ của $(E)$.
\item Tìm tọa độ điểm $M$ thuộc $(E)$ sao cho $MF_{1}=4MF_{2}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $a^{2}=25\Rightarrow a=5$; $b^{2}=16\Rightarrow b=4\Rightarrow c=3$\\
$\Rightarrow$ tọa độ các tiêu điểm $F_{1}(-3;0)$, $F_{2}(3;0)$.
\item Gọi $M(x;y)$ là điểm nằm trên $(E)$. Áp dụng công thức bán kính qua tiêu ta có $\heva{&MF_{1}=5+\dfrac{3}{5}x\\ &MF_{2}=5-\dfrac{3}{5}x}$.\\
Vì $MF_{1}=4MF_{2}$ nên $5+\dfrac{3}{5}x=4\left(5-\dfrac{3}{5}x\right)\Rightarrow x=5\Rightarrow y=\pm 4$.\\
Vậy điểm $M$ cần tìm là $(5;4)$ hoặc $(5;-4)$.
\end{enumerate}
}
\end{bt}
\begin{bt}%[HK2 (2015-2016), THPT Chu Văn An, Hà Nội]%[0D4G4]%[Dương BùiĐức]
Giải phương trình $2\sqrt{x^{2}+x+1}+\sqrt{x+1}=4x+3$.
\loigiai{
Tập xác định $\mathscr{D}=[-1;+\infty)$.\\
$2(\sqrt{x^{2}+x+1}-1)+(\sqrt{x+1}-1)=4x\Leftrightarrow \dfrac{2(x^{2}+x)}{\sqrt{x^{2}+x+1}+1}+\dfrac{x}{\sqrt{x+1}+1}=4x$\\
$\Leftrightarrow\hoac{&x=0\ (\mbox{thỏa mãn})\\ &\dfrac{2(x+1)}{\sqrt{x^{2}+x+1}+1}+\dfrac{1}{\sqrt{x+1}+1}-4=0\ (*).}$\\
Vì $x\geq -1\Rightarrow \sqrt{x^{2}+x+1}+1\geq \sqrt{x^{2}}=|x|\Rightarrow \sqrt{x^{2}+x+1}+1\geq |x|+1\geq x+1\geq 0$.
\begin{itemize}
\item Xét $x=-1$ ta thấy không thỏa mãn $(*)$.
\item Xét $x>-1$, ta có $\dfrac{2(x+1)}{\sqrt{x^{2}+x+1}+1}<\dfrac{2(x+1)}{x+1}=2$ và $\dfrac{1}{\sqrt{x+1}+1}<1\Rightarrow (*)$ vô nghiệm.
\end{itemize}
Vậy nghiệm của phương trình là $x=0$.
}
\end{bt}
