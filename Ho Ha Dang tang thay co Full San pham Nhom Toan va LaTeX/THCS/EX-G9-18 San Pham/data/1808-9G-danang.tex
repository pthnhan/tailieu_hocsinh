\begin{ex}%[Đề thi HSG 9, Đà Nẵng, năm 2018]%[Đỗ Viết Lân, Dự án EX-9G-18]%[9D0Y1]
Tính $A = \dfrac{1+\sqrt{11}}{2+\sqrt{11}}+\sqrt{\dfrac{2}{18-5\sqrt{11}}}$
\loigiai{
	Ta có
\begin{eqnarray*}
&A &= \dfrac{(1+\sqrt{11})(2-\sqrt{11})}{4 - 11} + \sqrt{\dfrac{4}{36-10\sqrt{11}}} \\
&&= \dfrac{9-\sqrt{11}}{7} + \dfrac{2}{\sqrt{(5-\sqrt{11})^2}} \\
&&= \dfrac{9-\sqrt{11}}{7} + \dfrac{2}{5-\sqrt{11}} \\
&&= \dfrac{9-\sqrt{11}}{7} + \dfrac{5+\sqrt{11}}{7} = 2.
\end{eqnarray*}	
}

\begin{ex}%[Đề thi HSG 9, Đà Nẵng, năm 2018]%[Đỗ Viết Lân, Dự án EX-9G-18]%[9D0B1]
Cho biểu thức $A = \left( \dfrac{x+2}{x\sqrt{x}-1} + \dfrac{\sqrt{x}}{x+\sqrt{x}+1} + \dfrac{1}{1-\sqrt{x}} \right):\dfrac{\sqrt{x}-1}{2\sqrt{x}}$, với $x > 0, x\neq 1$. Rút gọn $A$ và chứng minh $A < \dfrac{2}{3}$.
\loigiai{
Ta có 
\begin{eqnarray*}
& A &= \left( \dfrac{x+2}{(\sqrt{x}-1)(x+\sqrt{x}+1)} + \dfrac{\sqrt{x}}{x+\sqrt{x}+1} + \dfrac{1}{1-\sqrt{x}} \right)\cdot\dfrac{2\sqrt{x}}{\sqrt{x}-1} \\
&&= \dfrac{x+2 - \sqrt{x}(1-\sqrt{x})-(x+\sqrt{x}+1)}{(\sqrt{x}-1)(x+\sqrt{x}+1)}\cdot\dfrac{2\sqrt{x}}{\sqrt{x}-1}\\
&&=\dfrac{x+2 + x-\sqrt{x}-x-\sqrt{x}-1}{(\sqrt{x}-1)(x+\sqrt{x}+1)}\cdot\dfrac{2\sqrt{x}}{\sqrt{x}-1}\\
&&=\dfrac{x-2\sqrt{x}+1}{(\sqrt{x}-1)(x+\sqrt{x}+1)}\cdot\dfrac{2\sqrt{x}}{\sqrt{x}-1}\\
&& = \dfrac{2\sqrt{x}(\sqrt{x}-1)^2}{(1-\sqrt{x})^2(x+\sqrt{x}+1)}\\
&& = \dfrac{2\sqrt{x}}{x+\sqrt{x}+1}
\end{eqnarray*}
Ta lại có
\begin{eqnarray*}
&A - \dfrac{2}{3} = &\dfrac{2\sqrt{x}}{x+\sqrt{x}+1} - \dfrac{2}{3} \\
&= &\dfrac{2(3\sqrt{x}-x-\sqrt{x}-1)}{x+\sqrt{x}+1}\\
&= &\dfrac{-2(1-\sqrt{x})^2}{x+\sqrt{x}+1} <0.
\end{eqnarray*}
Vậy $A < \dfrac{2}{3}$.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, Đà Nẵng, năm 2018]%[Đỗ Viết Lân, Dự án EX-9G-18]%[9D2G3]
Cho dường thẳng $d_m$ có phương trình $y = mx + 2m - 1$, với $m$ là tham số.
\begin{enumerate}
\item Khi tham số $m$ thay đổi thì đường thẳng $d_m$ luôn đi qua một điểm $H$ cố định. Tìm tọa độ điểm $H$.
\item Tìm giá trị của $m$ sao cho khoảng cách từ $A(1;2)$ đến $d_m$ lớn nhất.
\end{enumerate}
\loigiai{
	\begin{enumerate}
	\item Gọi tọa độ điểm $H$ cần tìm là $H(a;b)$.\\
	Vì $d_m$ luôn đi qua $H$ nên ta có $b=ma+2m-1\Leftrightarrow (a+2)m -(b+1)=0$ luôn đúng với mọi $m$.\\
	Do đó $\heva{&a+2 = 0\\&b+1 = 0} \Leftrightarrow \heva{&a = -2\\&b = -1}$.\\
	Vậy $H(-2;-1)$.
	\item \immini{Gọi $M$ là hình chiếu của $A$ lên $d_m$. Khi đó khoảng cách từ $A$ đến $d_m$ là $AM$.\\
	Ta có $AM \leq AH = const$ với mọi $m$.\\
	Do đó khoảng cách từ $A$ đến $d_m$ lớn nhất là $AH$ khi $d_m \perp AH$ tại $H$.\\
	Đường thẳng $AH$ có dạng $y = ax +b$. Do $A,H \in AH$ nên $\heva{&2=a+b\\&-1 = -2a+b} \Leftrightarrow \heva{&a=1 \\&b=1}$.\\
	Đường thẳng $d_m$ vuông góc với $AH$ nên $$a_{d_m} \cdot a_{AH} = -1 \Rightarrow a_{d_m} = -1 \Rightarrow m = -1.$$}
	{\begin{tikzpicture}[scale=0.7]
	\clip (0,0) rectangle (5.5,5.5);
	\tkzDefPoints{0/0/C,5/3/D,1/5/A}
	\tkzDefPointBy[homothety = center C ratio 0.3](D)    \tkzGetPoint{H}
	\tkzDefPointBy[projection = onto C--D](A)    \tkzGetPoint{M}
	\tkzDrawSegments(C,D A,M A,H)
	\tkzDrawPoints(M,A,H)
	\tkzLabelPoints[below right](M,H)
	\tkzLabelPoints[right](A)
	\end{tikzpicture}}
	\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, Đà Nẵng, năm 2018]%[Đỗ Viết Lân, Dự án EX-9G-18]%[9D0B2]
Tìm tất cả các giá trị của $x$ thỏa mãn $\sqrt{x - 4\sqrt{x-2}+2} + \sqrt{x + 6\sqrt{x-2}+7} = 7$.
\loigiai{
	Điều kiện $x \geq 2$. Ta có
	\begin{eqnarray*}
	&&\sqrt{x - 4\sqrt{x-2}+2} + \sqrt{x + 6\sqrt{x-2}+7} = 7\\
	&\Leftrightarrow & \sqrt{(\sqrt{x-2}-2)^2} + \sqrt{(\sqrt{x-2}+3)^2} = 7\\
	&\Leftrightarrow & |\sqrt{x-2}-2| + \sqrt{x-2}+3 = 7\\
	&\Leftrightarrow & \hoac{&\sqrt{x-2}-2 + \sqrt{x-2}+3 = 7,\text{ nếu } x \geq 6\\&2-\sqrt{x-2} + \sqrt{x-2}+3 = 7,\text{ nếu } x < 6}\\
	&\Leftrightarrow & \hoac{&2\sqrt{x-2} = 6\,\,(x \geq 6)\\&5 = 7\text{ (vô lý)}}\\
	&\Leftrightarrow & x-2 = 9\,\,(x \geq 6)\\
	&\Leftrightarrow & x = 11 \text{ (thỏa)}.
	\end{eqnarray*} 
	Vậy $x = 11$ là nghiệm của phương trình.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, Đà Nẵng, năm 2018]%[Đỗ Viết Lân, Dự án EX-9G-18]%[9D0K2]
Tìm tất cả các bộ số $x,y,z$ thỏa $\heva{&x^2-2x = y\\&y^2 +2y = z\\&x+y+z+1+\sqrt{x-1} = 0}$.
\loigiai{
	Ta có $x^2 - 2x = y \Leftrightarrow (x-1)^2 = y+1$.\\
	và $y^2 + 2y = z \Leftrightarrow z + 1 = (y+1)^2 = (x-1)^4.$\\
	Do đó ta có
	\begin{eqnarray*}
	&&x+y+z+1+\sqrt{x-1} = 0\\
	&\Leftrightarrow &(x-1)+(y+1)+(z+1) + \sqrt{x-1} = 0\\
	&\Leftrightarrow &(x-1)+(x-1)^2 + (x-1)^4 + \sqrt{x-1} = 0\\
	&\Leftrightarrow & \sqrt{x-1}\left(\sqrt{x-1}^7+\sqrt{x-1}^3+\sqrt{x-1}+1\right) = 0\\
	&\Leftrightarrow & \sqrt{x-1} = 0\\
	&\Leftrightarrow & x = 1.
	\end{eqnarray*}
	Suy ra $y = z = -1$.\\
	Vậy nghiệm hệ là $(x;y;z) = (1;-1;-1)$.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, Đà Nẵng, năm 2018]%[Đỗ Viết Lân, Dự án EX-9G-18]%[9D3B5]
Một thửa ruông hình chữ nhật, nếu giảm chiều rộng đi $1$ m và tăng chiều dài thêm $2$ m thì diện tích không đổi. Ngoài ra nếu giảm chiều dài đi $4$ m đồng thời tăng chiều rộng thêm $3$m ta được hình vuông. Tính diện tích thửa ruộng ban đầu.
\loigiai{
	Gọi $x,y$ (m) lần lượt là chiều dài, chiều rộng của thửa ruộng (điều kiện $x > 4$, $y>1$).\\
	Nếu giảm chiều rộng đi $1$ m và tăng chiều dài thêm $2$ m thì diện tích không đổi nên $$(x+2)(y-1) = xy \quad (1)$$
	Nếu giảm chiều dài đi $4$ m đồng thời tăng chiều rộng thêm $3$m ta được hình vuông nên $$x-4 = y+3 \quad (2)$$
	Từ $(1)$ và $(2)$ ta có hệ $\heva{&xy-x+2y-2 = xy\\&x=y+7} \Leftrightarrow \heva{&-(y+7)+2y-2=0\\&x=y+7} \Leftrightarrow \heva{&y=9\\&x=16}$.\\
	Diện tích thửa ruộng ban đầu là $S = 9\times 16 = 144$ m$^2$.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, Đà Nẵng, năm 2018]%[Đỗ Viết Lân, Dự án EX-9G-18]%[9H0G4]

	$\Rightarrow \alpha + \beta < 90^\circ$\\
	Ta có $S_{ABC} = \dfrac{1}{2}BC\cdot AH = \dfrac{1}{2}AB\cdot AC \cdot \sin (\alpha + \beta)$\\
	$\Rightarrow \dfrac{AH\cdot BC}{AB\cdot AC} = \sin (\alpha + \beta) $}{\begin{tikzpicture}
