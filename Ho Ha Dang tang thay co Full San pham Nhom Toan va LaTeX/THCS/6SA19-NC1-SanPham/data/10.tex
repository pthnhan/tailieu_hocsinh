\section{Các phép tính về phân số}
\subsection{Kiến thức cơ bản}
\begin{enumerate}
	\item Muốn cộng hai phân số cùng mẫu, ta cộng các tử và giữ nguyên mẫu. Muốn cộng hai phân số không cùng mẫu, ta viết chúng dưới dạng hai phân số có cùng mẫu, rồi cộng các tử và giữ nguyên mẫu chung.
	\item Hai phân số gọi là đối nhau nếu tổng của chúng bằng $0$.
	\item Muốn trừ một phân số cho một phân số, ta cộng số bị trừ với số đối của số trừ.
	\item Muốn nhân hai phân số, ta nhân các tử với nhau và nhân các mẫu với nhau.
	\item Hai số gọi là nghịch đảo nếu tích của chúng bằng $1$.
	\item Muốn chia một phân số cho một phân số, ta nhân số bị chia với số nghịch đảo của số chia.
\end{enumerate}
\subsection{Ví dụ minh họa}
	\begin{vd}
		Tìm các số tự nhiên $a$ và $b$ sao cho $\dfrac{a}{5}+\dfrac{b}{3}=\dfrac{13}{15}$.
		\loigiai{
		$\dfrac{a}{5}+\dfrac{b}{3}=\dfrac{13}{15}$ $\Leftrightarrow \dfrac{3a+5b}{15}=\dfrac{13}{15}$ $\Leftrightarrow 3a+5b=15$.\\
		Nếu $b=0$ thì $3a=13$, loại.\\
		Nếu $b=1$ thì $3a=8$. loại.\\
		Nếu $b=2$ thì $3a=3$ nên $a=1$.\\
		Nếu $b \geq 3$ thì $3a+5b \geq 15$, loại.\\
		Vậy $a=1$ và $b=2$. 
		}
	\end{vd}
	\begin{vd}
		\begin{enumerate}
			\item Cho trước phân số $\dfrac{a}{b}$ khác $-1$. Tìm phân số $\dfrac{c}{d}$ sao cho $\dfrac{a}{b}-\dfrac{c}{d}=\dfrac{a}{b}.\dfrac{c}{d}$.
			\item Tìm phân số $\dfrac{c}{d}$ có tính chất trên, nếu phân số $\dfrac{a}{b}$ bằng $\dfrac{1}{3}$, bằng $\dfrac{3}{5}$.
		\end{enumerate}
		\loigiai{
		\begin{enumerate}
			\item $\dfrac{a}{b}-\dfrac{c}{d}=\dfrac{a}{b}.\dfrac{c}{d}$ $\Leftrightarrow$ $ad-bc=ac$ $\Leftrightarrow$ $ad=c\left(a+b\right)$ $\Leftrightarrow$ $\dfrac{c}{d}=\dfrac{a}{a+b}$.
			\item  \hfill
			\begin{center} \renewcommand{\arraystretch}{2.5}
				\begin{tabular}{|c|c|c|}
					\hline
					$\dfrac{a}{b}$&$\dfrac{c}{d}$& Kiểm tra $\dfrac{a}{b}-\dfrac{c}{d}=\dfrac{a}{b}.\dfrac{c}{d}$\\ \hline
					$\dfrac{1}{3}$&$\dfrac{1}{4}$& $\dfrac{1}{3}-\dfrac{1}{4}=\dfrac{1}{3}.\dfrac{1}{4}$\\ \hline
					$\dfrac{3}{5}$&$\dfrac{3}{8}$& $\dfrac{3}{5}-\dfrac{3}{8}=\dfrac{3}{5}.\dfrac{3}{8}$\\ \hline
				\end{tabular}
			\end{center}
			
		\end{enumerate}
		}
	\end{vd}

					&=&9-\left(1-\dfrac{1}{10}\right)=9-\dfrac{9}{10}=\dfrac{81}{10}.

			}
		}

	\begin{vd}
		Cho $A=\dfrac{1}{4}.\dfrac{3}{6}.\dfrac{5}{8}.\cdots .\dfrac{43}{46}.\dfrac{45}{48}$ và $B=\dfrac{2}{5}.\dfrac{4}{7}.\dfrac{6}{9}.\cdots .\dfrac{44}{47}.\dfrac{46}{49}$.
		\begin{multicols}{2}
			\begin{enumerate}
				\item So sánh $A$ và $B$.
				\item Chứng minh rằng $A<\dfrac{1}{133}$.
			\end{enumerate}
		\end{multicols}
		\loigiai{
		\begin{enumerate}
			\item Khi cộng tử và mẫu của một phân số nhỏ hơn $1$ (tử và mẫu đều dương) với $1$ thì giá trị của phân số tăng thêm. Do đó: $$\dfrac{1}{4}<\dfrac{2}{5},\dfrac{3}{6}<\dfrac{4}{7},\dfrac{5}{8}<\dfrac{6}{9},\cdots$$
			Suy ra $A<B$.	
			\item \hfil
				{\allowdisplaybreaks
					\begin{eqnarray*}
						A.B&=&\dfrac{1.3.5.\cdots.43.45}{4.6.8.\cdots.46.48}.\dfrac{2.4.6.\cdots.44.46}{5.7.9.\cdots.47.49}\\
						&=&\dfrac{1.3.\left(5.\cdots.43.45\right)}{\left(4.6.8.\cdots.46\right).48}.\dfrac{2.\left(4.6.\cdots.44.46\right)}{\left(5.7.9.\cdots.45\right).47.49}\\
						&=&\dfrac{1.3}{48}.\dfrac{2}{47.49}=\dfrac{1}{8.47.49}<\dfrac{1}{361.49}=\left(\dfrac{1}{133}\right)^2.
					\end{eqnarray*}
				}
				Ta lại có: $A<B$ nên $A.A<A.B<\left(\dfrac{1}{133}\right)^2$. Vậy $A<\dfrac{1}{133}$.
		\end{enumerate}
		}
	\end{vd}
	\begin{vd}
		Tìm phân số $\dfrac{a}{b}$ lớn nhất sao cho khi chia mỗi phân số $\dfrac{12}{35}$, $\dfrac{8}{21}$, $\dfrac{52}{91}$ cho $\dfrac{a}{b}$ ta đều được các số tự nhiên.
		\loigiai{
		Phân số $\dfrac{a}{b}$ phải là phân số dương. giả sử $a>0$, $b>0$ và $\left(a,b\right)=1$.\\
		Rút gọn $\dfrac{52}{91}=\dfrac{4}{7}$. Theo đề bài:\\
			\begin{equation} \label{VD10.5.1}
				\dfrac{12}{35}:\dfrac{a}{b}=\dfrac{12b}{35a} \in \mathbb{N}.
			\end{equation}
			\begin{equation} \label{VD10.5.2}
				\dfrac{8}{21}:\dfrac{a}{b}=\dfrac{8b}{21a} \in \mathbb{N}.
			\end{equation}
			\begin{equation} \label{VD10.5.3}
				\dfrac{4}{7}:\dfrac{a}{b}=\dfrac{4b}{7a} \in \mathbb{N}.
			\end{equation}
	Từ \eqref{VD10.5.1} suy ra $12b\ \vdots \ 35$, mà $\left(12,35\right)=1$ nên $b\ \vdots  \ 35$.\\
	Từ \eqref{VD10.5.2} suy ra $8b \vdots \ 21$, mà $\left(8,21\right)=1$ nên $b\ \vdots \ 21$.\\
	Từ \eqref{VD10.5.3} suy ra $4b\ \vdots \ 7$, mà $\left(4,7\right)=1$ nên $b\ \vdots \ 7$.\\
	Do đó $b \in \text{BC} \left(35,21,7\right)$.\\	
	Từ \eqref{VD10.5.1} suy ra $12b\ \vdots \ a$, mà $\left(a,b\right)=1$ nên $12\ \vdots \ a$.\\
	Từ \eqref{VD10.5.2} suy ra $8b\ \vdots \ a$, mà $\left(a,b\right)=1$ nên $8\ \vdots \ a$.\\
	Từ \eqref{VD10.5.3} suy ra $4b\ \vdots \ a$, mà $\left(a,b\right)=1$ nên $4\ \vdots \ a$.\\
	Do đó $a \in \text{ƯC} \left(12,8,4\right)$.\\
	Để $\dfrac{a}{b}$ lớn nhất, ta chọn $a=\text{ƯCLN}\left(12,8,4\right)=4$ và $b=\text{BCNN}\left(35,21,7\right)=105$.\\
	Vậy $\dfrac{a}{b}=\dfrac{4}{105}$.
		}
	\end{vd}
\begin{vd}
	Tìm số nguyên $n$ để phân số $\dfrac{20n+3}{4n+3}$có giá trị nhỏ nhất.	
	
	\loigiai{
		{\allowdisplaybreaks
			\begin{eqnarray*}
				A=\dfrac{20n+3}{4n+3}=\dfrac{20n+5-2}{4n+3}=\dfrac{5\left(4n+3\right)}{4n+3}-\dfrac{2}{4n+3}=5-\dfrac{2}{4n+3}.
		\end{eqnarray*}}
		$A$ nhỏ nhất $\Leftrightarrow \dfrac{2}{4n+3}$ lớn nhất$\Leftrightarrow 4n+3$ là số nguyên dương nhỏ nhất.\\
		
		Nếu $4n+3=1$ thì $n$ không là số nguyên.\\
		
		Nếu $4n+3=2$ thì $n$ không là số nguyên.\\
		
		Nếu $4n+3=3$ thì $n=0$.\\
		
		Vậy $\dfrac{20n+3}{4n+3}$ nhỏ nhất tại $n=0$, khi đó phân số bằng $\dfrac{13}{3}$.
	}
\end{vd}

\begin{vd}
	Tìm các số nguyên dương $x$ và $y (x<y)$ sao cho $\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{1}{8}$.  
	\loigiai{
		Do $x<y$ nên $\dfrac{1}{x}>\dfrac{1}{y}$. Ta có $\dfrac{1}{8}=\dfrac{1}{x}+\dfrac{1}{y} <\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{2}{x}$ nên $x<16$.\\
		Ta lại có $\dfrac{1}{x}<\dfrac{1}{8}$ nên $x>8$.\\
		Vậy $x\in \{9;10;11;12;13;14;15\}$.\\
		
		- Với $x=9$ thì $\dfrac{1}{y}=\dfrac{1}{8}-\dfrac{1}{9}=\dfrac{1}{72}$ nên $y=72$.\\
		
		- Với $x=10$ thì $\dfrac{1}{y}=\dfrac{1}{8}-\dfrac{1}{10}=\dfrac{1}{40}$ nên $y=40$.\\
		
		- Với $x=11$ thì $\dfrac{1}{y}=\dfrac{1}{8}-\dfrac{1}{11}=\dfrac{3}{88}$, loại.\\
		
		- Với $x=12$ thì $\dfrac{1}{y}=\dfrac{1}{8}-\dfrac{1}{12}=\dfrac{1}{24}$ nên $y=24$.\\
		
		- Với $x=13$ thì $\dfrac{1}{y}=\dfrac{1}{8}-\dfrac{1}{13}=\dfrac{5}{104}$, loại.\\
		
		- Với $x=14$ thì $\dfrac{1}{y}=\dfrac{1}{8}-\dfrac{1}{14}=\dfrac{3}{56}$, loại.\\
		
		- Với $x=15$ thì $\dfrac{1}{y}=\dfrac{1}{8}-\dfrac{1}{15}=\dfrac{7}{120}$, loại.\\
		Vậy 
		\begin{center}
			\begin{tabular}{|l|l|l|l|}
				\hline 
				$x$ & $9$&$10$&$12$\\ 
				\hline 
				$y$ & $72$&$40$&$24$ \\ 
				\hline
			\end{tabular}
		\end{center}
	}			
\end{vd}

\begin{vd}
	Thay $a$ và $b$ bởi các chữ số thích hợp $\overline{0,ab}\cdot(a+b)=0,36$.	
	\loigiai{{\allowdisplaybreaks
			\begin{eqnarray*}
				\overline{0,ab}\cdot(a+b)=0,36 \quad\text{nên}\quad \overline{ab}\cdot(a+b)=36.
		\end{eqnarray*}}
		Suy ra $a+b$ là ước của 36. Ta lại có $0\leq a\leq 9$ và $0\leq b\leq 9$ nên $a+b\leq 18$ (chú ý rằng $a$ có thể bằng $0$).\\
		Ta có bảng sau:
		\begin{center}
			\begin{tabular}{|l|l|l|l|l|l|l|l|l|}
				\hline 
				$a+b$ & $1$&$2$&$3$&$ 4 $&$ 6 $&$ 9 $&$ 12 $&$ 18 $\\ 
				\hline 
				$\overline{ab}$ & $36$&$18$&$12$&$ 9 $&$ 6 $&$ 4 $&$ 3 $&$ 2 $\\ 
				\hline
			\end{tabular}
		\end{center}
		
		xét các số $\overline{ab}$ tìm được, tính tổng các chữ số (là $a+b$) xem có phù hợp với dòng trên hay không, ta thấy có hai đáp số sau:
		{\allowdisplaybreaks\begin{eqnarray*}
				\overline{ab}=12 \quad\text{thì}\quad a+b=3;\\
				\overline{ab}=06 \quad\text{thì}\quad a+b=6.
		\end{eqnarray*}}
		Vậy {\allowdisplaybreaks
			\begin{eqnarray*}
				0,12(1+2)=0,36;\\
				0,06(0+6)=0,36.
		\end{eqnarray*}}
	}
\end{vd}
