\begin{ex}%[Đề thi HSG Lớp 9 - TP Hà Nội năm 2010]%	[9D5G2]
Tìm tất cả các số tự nhiên $n$ để $7^{2013} + 3^n$ có chứa số hàng đơn vị $8$.
    \loigiai{
    Ta có $7^4$ tận cùng bởi $1$, suy ra $7^{2013}=\left({7^4}\right)^{503}\cdot7$ tận cùng bởi $7$. Để số $7^{2013}+3^n$ có chữ số hàng đơn vị là $8$ thì $3^n$ phải có chữ số hàng đơn vị là $1$.\\
    		Xét $3^n=3^{4k+r}=81^k\cdot3^r$ với $r\in \left\{0;1;2;3\right\}$ nên $3^n$ lần lượt có chữ số hàng đơn vị là $1, 3, 9, 7$.\\
    		Vậy số $7^{2013}+3^n$ có chữ số hàng đơn vị là $8$ khi và chỉ khi $r=0 \Leftrightarrow n=4k$ với $k \in \mathbb{N}$.
    }
\end{ex}
\begin{ex}%[Đề thi HSG Lớp 9 - TP Hà Nội năm 2010]%[9D5G2]
Cho $a$, $b$ là các số tự nhiên lớn hơn $2$ và $p$ là số tự nhiên thỏa mãn $\dfrac{1}{p} = \dfrac{1}{a^2} + \dfrac{1}{b^2}$. Chứng minh $p$ là  hợp số.
\loigiai{
Giả sử $p$ là số nguyên tố. Từ $a^2b^2=p\left(a^2+b^2\right) \Rightarrow a^2b^2 \, \vdots \, p\Rightarrow \hoac{& a \, \vdots \,p \\
	& b \, \vdots \, p}$, (1).\\
Suy ra $a^2b^2 \, \vdots \, p^2\Rightarrow p\left(a^2+b^2\right)  \, \vdots \, p^2\Rightarrow \left(a^2+b^2\right)   \, \vdots \, p$, (2) \\
Từ (1) và (2) suy ra $\heva{& a  \, \vdots \, p \\
	& b  \, \vdots \, p}$. \\
Từ đó $a\geqslant p$ và $b\geqslant p$, suy ra $\dfrac{1}{a^2}+\dfrac{1}{b^2}\leqslant \dfrac{2}{p^2}\Rightarrow \dfrac{1}{p}\leqslant \dfrac{2}{p^2}\Rightarrow p\leqslant 2$, (3).\\
Từ $a, b$ là các số tự nhiên lớn hơn $2$, suy ra $\dfrac{1}{a^2}+\dfrac{1}{b^2}<\dfrac{1}{2}\Rightarrow p>2$, (4).\\
Từ (3) và (4) dẫn đến mâu thuẫn. Vậy $p$ là hợp số. }
\end{ex}
\begin{ex}%[Đề thi HSG Lớp 9 - TP Hà Nội năm 2010]%[9D5G3] 
Tìm các số nguyên $x$, $y$ thỏa mãn $x^2 - 3y^2 + 2xy - 2x +  6y  - 8 = 0$.
\loigiai
    { $x^2 - 3y^2 + 2xy - 2x + 6y - 8 = 0$\\
    		$\Leftrightarrow \left(x-y+1\right)\left(x+3y-3\right)=5=1\cdot5=5\cdot1=\left(-1\right)\cdot\left(-5\right)=\left(-5\right)\cdot\left(-1\right).$\\
    		Ta giải bốn hệ phương trình\\
    		$\heva{& x-y+1=1 \\
    		& x+3y-3=5} $; $\heva{& x-y+1=5 \\
    		& x+3y-3=1}$; $\heva{& x-y+1=-1 \\
    		& x+3y-3=-5}$; $\heva{& x-y+1=-5 \\
    		& x+3y-3=-1}$.\\
    	Từ đó phương trình đã cho có bốn nghiệm nguyên là $\left(2;2\right), \left(-2;0\right), \left(4;0\right), \left(-4;2\right)$.
    	
    }
\end{ex}
\begin{ex}%[Đề thi HSG Lớp 9 - TP Hà Nội năm 2010]%[9D5G5]
Giải hệ phương trình:   $\heva{&
	2x^2  + xy + 3y^2 - 2y - 4 = 0\\
	&3x^2  + 5y^2 + 4x - 12 = 0.}$
\loigiai{
 $\heva{&
	2x^2 + xy + 3y^2 - 2y - 4 = 0\\&
	3x^2 + 5y^2 + 4x - 12 = 0}$\\
Lấy phương trình thứ nhất nhân với $2$ rồi trừ đi phương trình hai ta được
\begin{align*}
x^2+2xy+y^2-4y-4x+4=0\\
 \Leftrightarrow \left({x+y}\right)^2-4\left({x+y}\right)+4=0 \\
 \Leftrightarrow x+y=2\\
 \Leftrightarrow y=2-x.
\end{align*}
Thay vào phương trình thứ hai ta được $ 8x^2-16x+8=0 \Leftrightarrow x=1$.\\
Từ đó hệ có nghiệm duy nhất là $\left(1;1\right)$. 
}
\end{ex}
\begin{ex}%[Đề thi HSG Lớp 9 - TP Hà Nội năm 2010]%[9D5G1]
       Với  $a$, $b$ là các số thực thỏa mãn $a + b  + 4ab  = 4a^2 +  4b^2$. Tìm giá trị lớn nhất của   
       $$A =  20\big(a^3 + b^3\big) - 6\big(a^2 + b^2\big) + 2013.$$ 
\loigiai
    {Ta có $\dfrac{1}{4}\left(a+b\right)=a^2+b^2-ab\geqslant \dfrac{{\left(a+b\right)}^2}{4}\Rightarrow 0\leqslant a+b\leqslant 1$.\\
    	Mặt khác $A\leqslant 20\left(a+b\right)\left(a^2+b^2-ab\right)-6\cdot\dfrac{{\left(a+b\right)}^2}{2}+2013$.\\
    	Suy ra $A\leqslant 20\left({a+b}\right)\cdot\dfrac{a+b}{4}-3\left({a+b}\right)^2+2013=2\left({a+b}\right)^2+2013\leqslant 2015$.\\
    	Dấu bằng xảy ra khi và chỉ khi $a=b=\dfrac{1}{2}$.
    	Vậy $\max A=2015$ khi $a=b=\dfrac{1}{2}$.
    }
\end{ex}

\begin{ex}%[Đề thi HSG Lớp 9 - TP Hà Nội năm 2010]%[9D5G4]
    Trong mặt phẳng cho $6$ điểm $A_1$, $A_2$,  $A_3$, $A_4$, $A_5$, $A_6$ trong đó không có $3$ điểm nào thẳng hàng và trong $3$ điểm luôn có $2$ điểm có khoảng cách nhỏ hơn $671$. Chứng minh rằng trong $6$ điểm đã cho luôn tồn tại $3$ điểm là $3$ đỉnh của một tam giác có chu vi nhỏ hơn $2013$. 
\loigiai
      {
    	Với $A_iA_k$ bất kì $(1\leq i<k \leq 6),$ ta quy định:
    	\begin{itemize}
    		\item Nếu $A_iA_k<671$ thì $A_i, A_k$ được nối với nhau bởi đoạn thẳng màu xanh.
    		\item Nếu $A_iA_k\geq 671$ thì $A_i, A_k$ được nối với nhau bởi đoạn thẳng màu đỏ.
    	\end{itemize}
    	Xét $5$ đoạn thẳng $A_1A_2,A_1A_3,\dotsc, A_1A_6.$ Theo nguyên lí Dirichlet, tồn tại ba đoạn thẳng cùng màu. Không mất tính tổng quát, giả sử ba đoạn $A_1A_1, A_1A_3, A_1A_4$ cùng màu.\\
    	\textit{Trường hợp 1}: Ba đoạn $A_1A_1, A_1A_3, A_1A_4$ cùng màu xanh. Do đó tồn tại một cạnh trong tam giác $A_2A_3A_4$ màu xanh, chẳng hạn cạnh $A_2A_3.$ Khi đó tam giác $A_1A_2A_3$ thỏa đề bài.\\
    	\textit{Trường hợp 2}: Ba đoạn $A_1A_1, A_1A_3, A_1A_4$ cùng màu đỏ. Lập luận tương tự , tam giác $A_2A_3A_4$ thỏa đề bài.
    }
\end{ex}
