\section{LÔGARIT}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Định nghĩa và tính chất}
\begin{dn}
Cho hai số dương $a, b$ với $a\neq 1$. Số $\alpha$ thỏa mãn đẳng thức $a^{\alpha}=b$ được gọi là lôgarit cơ số $a$ của $b$ và kí hiệu là $\log_ab$.
$$\alpha=\log_ab \Leftrightarrow a^{\alpha}=b, (a,b>0; a\neq 1).$$
\end{dn}
\begin{tc}
Cho hai số dương $a, b$ với $a\neq 1$, ta có tính chất sau:
$$\log_a1=0; \log_aa=1; a^{\log_ab}=b; \log_aa^{\alpha}=\alpha.$$
\end{tc}
\begin{vd}%[2D2B3]
Hãy tìm lôgarit của mỗi số $3$; $1$; $\dfrac{1}{3\sqrt{3}}$ theo sơ số $3$.
\loigiai{
Ta lần lượt có:
$$\log_33=1; \log_31=0; \log_3\dfrac{1}{3\sqrt{3}}=\log_33^{\frac{-3}{2}}=-\dfrac{3}{2}.$$
}
\end{vd}

\begin{vd}%[2D2B3]
Tính $\log_{0.5}\dfrac{1}{2}$; $\log_{\frac{1}{4}}\dfrac{1}{64}$.
\loigiai{
Ta lần lượt có:
$$\log_{0.5}\dfrac{1}{2}=\log_{0.5}0.5=1.$$
$$\log_{\frac{1}{4}}\dfrac{1}{64}=\log_{\frac{1}{4}}\left(\dfrac{1}{4}\right)^3=3.$$
}
\end{vd}

\subsubsection{Các quy tắc tính lôgarit}
Cho ba số dương $a$, $b_1$, $b_2$ với $a\neq 1$, ta có các quy tắc sau:
\begin{itemize}
\item $\log_ab_1b_2=\log_ab_1+\log_ab_2$;
\item $\log_a\dfrac{b_1}{b_2}=\log_ab_1-\log_ab_2$. Đặc biệt, với $a, b>0$, $a\neq 1$ thì $\log_a\dfrac{1}{b}=-\log_ab$.
\item $\log_ab_1^{\alpha}=\alpha \log_ab_1$. Đặc biệt: $\log_a\sqrt[n]{b}=\dfrac{1}{n}\log_ab$.
\end{itemize}

\begin{vd} %[2D2B3]
Tính $\dfrac{\log_716}{\log_7 15 - \log_7 30} .$
\loigiai{

Ta có: 	$\dfrac{\log_716}{\log_7 15 - \log_7 30} = \dfrac{\log_7 16}{\log_7 \dfrac{15}{30}}= \dfrac{\log_7 2^ 4}{\log_7 2^{-1}} = \dfrac{4 \log_7 2}{- \log_7 2}=-4.$}
\end{vd}

\begin{vd}%[2D2B3]
Hãy so sánh $\log 2+\log 3$ và $\log 5$.
\loigiai{
Ta có $$\log 2+\log 3=\log 6>\log 5.$$
}
\end{vd}
\subsubsection{Công thức đổi cơ số}
Cho ba số dương $a, b, c$ và $a\neq 1, c\neq 1$, ta có $\log_ab=\dfrac{\log_cb}{\log_ca}$. 

Đặc biệt: $\log_ab=\dfrac{1}{\log_ba}$, với $b\neq 1$; $\log_a^{\alpha}b=\dfrac{1}{\alpha}\log_ab$, với $\alpha\neq 0$.

\begin{vd}%[2D2B3]
	Tính $\log_\frac{1}{4} \left(\log_3 4. \log_ 2 3 \right)$.
	\loigiai{
Ta có $$\log_\frac{1}{4} \left( \log_3 4. \log_2 3 \right) = \log_\frac{1}{4} \left( 2 \log_3 2. \log_2 3 \right)  = \log_\frac{1}{4} \left( 2\log_3 3 \right)  = \log_\frac{1}{4} 2 = \log_2^{-2} 2 =-\dfrac{1}{2} \log_2 2 = -\dfrac{1}{2}.$$ }
\end{vd}

\subsubsection{Lôgarit thập phân và lôgarit tự nhiên}
- Lôgarit cơ số $10$ gọi là lôgarit thập phân, $\log_{10}N$, ($N>0$) thường được gọi là $\lg N$ hay $\log N$.\\
- Lôgarit cơ số $e$ gọi là lôgarit tự nhiên, $\log_eN$, ($N>0$), được viết là $\ln N$.

\begin{vd}%[2D2B3]
Biểu diễn $\log 64$ theo $a = \log_2 5$.	

\loigiai{
Ta có $\log 64 = \dfrac{\log_2 64}{ \log_2 10} = \dfrac{\log_2 2^6}{\log_2(2.5)} = \dfrac{6 \log_2 2}{\log_2 2 + \log_2 5} =\dfrac{6}{1 + a}$.}
\end{vd}

\begin{dang}{Biểu diễn logarit theo các tham số.}%Nguyễn Bình Nguyên.
    \begin{enumerate}[\textbf{Bước} 1:]
        \item Biến đổi các biểu thức logarit phụ thuộc vào tham số $a$ và $b$ về dạng logarit với cơ số và đối số là tích các số nguyên tố.
        \item Đặt các biểu thức logarit của các số nguyên tố là các ẩn $x, y, z$,… Từ đó ta thu được phương trình hoặc hệ phương trình với các ẩn là $x, y, z$,… ta tìm các ẩn này theo $a, b$.
        \item Giải hệ tìm được tìm $x, y, z$,… theo $a, b$. Từ đó tính được biểu thức theo các tham số $a, b$.
    \end{enumerate}
    Các công thức nền tảng là $\log_a b = \dfrac{\log_c b}{\log_c a}$ và $\dfrac{1}{\log_a b}= \log_b a$. 
\end{dang}

        Khi đó $VT=a^{\log_cb}=\left(b^{\log_ba}\right)^{\log_cb}=b^{\log_cb \cdot \log_ba}=b^{\log_ca}=VP$.
        \hfill \rule{2mm}{2mm}
    }
\end{vd}

\begin{vd}%[Hồ Sỹ Trường]%[2D2B3]
    Cho $a,b,c$ là các số dương và khác $1$. Chứng minh rằng $\dfrac{\log_ac}{\log_{ab}c}=1+\log_ab$.
    \loigiai{
        Ta có $\log_{ab}c=\dfrac{1}{\log_cab}$.\\
        Do đó $VT=\log_ac\cdot \log_cab=\log_ac\left(\log_ca+\log_cb\right)=\log_ac\cdot\log_ca+\log_ac\cdot\log_cb$\\
        $=1+\log_ab=VP$.
        \hfill \rule{2mm}{2mm}
    }
\end{vd}

\begin{vd}%[Hồ Sỹ Trường]%[2D2B3]
    Chứng minh rằng $\dfrac{7}{16}\ln\left(3+2\sqrt{2}\right)-4\ln\left(\sqrt{2}+1\right)-\dfrac{25}{8}\ln \left(\sqrt{2}-1\right)=0$.
    \loigiai{
        Ta có $VT=\dfrac{7}{16}\ln\left(\sqrt{2}+1\right)^2-4\ln\left(\sqrt{2}+1\right)-\dfrac{25}{8}\ln\left(\sqrt{2}+1\right)^{-1}$\\
        $=\dfrac{7}{8}\ln\left(\sqrt{2}+1\right)-4\ln\left(\sqrt{2}+1\right)+\dfrac{25}{8}\ln\left(\sqrt{2}+1\right)$\\
        $=\left(\dfrac{7}{8}-4+\dfrac{25}{8}\right)\ln\left(\sqrt{2}+1\right)=0=VP$.
        \hfill \rule{2mm}{2mm}
    }
\end{vd}

\begin{bt}%[Hồ Sỹ Trường]%[2D2K3]
    Cho $a,b$ là các số thực dương và khác $1$. Chứng minh rằng $a^{\sqrt{\log_ab}}-b^{\sqrt{\log_ba}}=0$.
    \loigiai{
        Ta có $VT=a^{\sqrt{\log_ab}}-b^{\sqrt{\log_ba}}=b^{\log_ba\sqrt{\log_ab}}-b^{\sqrt{\log_ba}}=b^{\sqrt{\left(\log_ba\right)^2\log_ab}}-b^{\sqrt{\log_ba}}$\\
        $=b^{\sqrt{\log_ba}}-b^{\sqrt{\log_ba}}=0=VP$.
        \hfill \rule{2mm}{2mm}
    }
\end{bt}

\begin{bt}%[Hồ Sỹ Trường]%[2D2B3]
    Cho $a,x$ là các số thực dương, khác $1$ và $n\in \mathbb{N}^*$. Chứng minh rằng
    $$\dfrac{1}{\log_ax}+\dfrac{1}{\log_{a^2}x}+\ldots+\dfrac{1}{\log_{a^n}x}=\dfrac{n\left(n+1\right)}{2\log_ax}.$$
    \loigiai{
        Ta có $VT=\dfrac{1}{\log_ax}+\dfrac{1}{\tfrac{1}{2}\log_ax}+\ldots+\dfrac{1}{\tfrac{1}{n}\log_ax}=\dfrac{1}{\log_ax}+\dfrac{2}{\log_ax}+\ldots+\dfrac{n}{\log_ax}$\\
        $=\left(1+2+\ldots+n\right)\cdot \dfrac{1}{\log_ax}=\dfrac{n\left(n+1\right)}{2\log_ax}=VP$.
        \hfill \rule{2mm}{2mm}
    }
\end{bt}

\begin{bt}%[Hồ Sỹ Trường]%[2D2K3]
    Cho $a,b$ là các số thực dương và thỏa mãn $a^2+b^2=7ab$. Chứng minh rằng $\log_7\dfrac{a+b}{3}=\dfrac{1}{2}\left(\log_7a+\log_7b\right)$.
    \loigiai{
        Ta có $a^2+b^2=7ab \Rightarrow \left(a+b\right)^2=9ab\Rightarrow\left(\dfrac{a+b}{3}\right)^2=ab$.\\
        Suy ra $\log_7\left(\dfrac{a+b}{3}\right)^2=\log_7\left(ab\right)\Rightarrow\log_7\dfrac{a+b}{3}=\dfrac{1}{2}\left(\log_7a+\log_7b\right)$.
        \hfill \rule{2mm}{2mm}
    }    
\end{bt}

\begin{bt}%[Hồ Sỹ Trường]%[2D2K3]
    Cho $x,y$ là các số thực dương và thỏa mãn $x^2+4y^2=12xy$. Chứng minh rằng $2\log\left(x+2y\right)=\log x+\log y+4\log2$.
    \loigiai{
        Ta có $x^2+4y^2=12xy\Rightarrow \left(x+2y\right)^2=16xy$.\\
        Suy ra $\log\left(x+2y\right)^2=\log\left(16xy\right)\Rightarrow 2\log\left(x+2y\right)=\log x+\log y+4\log 2$.
        \hfill \rule{2mm}{2mm}
    }
\end{bt}

\begin{bt}%[Hồ Sỹ Trường]%[2D2K3]
    Cho $a=\log_{12}18$ và $b=\log_{24}54$. Chứng minh rằng $5\left(a-b\right)+ab=1$.
    \loigiai{
        Ta có $\heva{&a=\log_{12}18=\dfrac{\log_318}{\log_312}=\dfrac{\log_32+2}{2\log_32+1}\\&b=\log_{24}54=\dfrac{\log_354}{\log_324}=\dfrac{\log_32+3}{3\log_32+1}}$.\\
        Đặt $x=\log_32$, khi đó ta có $\heva{&a=\dfrac{x+2}{2x+1}\\&b=\dfrac{x+3}{3x+1}}$.\\
        Ta có 
        \begin{eqnarray*}
            VT=5\left(a-b\right)+ab&=&5\left(\dfrac{x-2}{2x+1}-\dfrac{x+3}{3x+1}\right)+\dfrac{x+2}{2x+1}\cdot\dfrac{x+3}{3x+1}\\
            &=&\dfrac{5\left[\left(x+2\right)\left(3x+1\right)-\left(x+3\right)\left(2x+1\right)\right]+\left(x+2\right)\left(x+3\right)}{\left(2x+1\right)\left(3x+1\right)}\\
            &=&\dfrac{6x^2+3x+1}{\left(2x+1\right)\left(3x+1\right)}=1=VP.
        \end{eqnarray*}
        \hfill \rule{2mm}{2mm}
    }    
\end{bt}

\begin{bt}%[Hồ Sỹ Trường]%[2D2K3]
 Chứng minh rằng $\log_23>\log_34$.
 \loigiai{
Ta có $\log_23\cdot\log_34=2$.\\
Mặt khác $3>\sqrt{8}\Rightarrow 3>2^{\tfrac{3}{2}}>2^{\sqrt{2}}\Rightarrow\log_23>\sqrt{2}$.\\
Do đó ta suy ra được $\log_34<\sqrt{2}$.\\
Vậy $\log_23>\log_34$.
  \hfill \rule{2mm}{2mm}
}   
\end{bt}

\begin{bt}%[Hồ Sỹ Trường]%[2D2K3]
Cho $a,b,c$ là ba cạnh của một tam giác vuông, trong đó $c$ là cạnh huyền. Chứng minh rằng $\log_{c+b}a+\log_{c-b}a=2\log_{c+b}a\cdot\log_{c-b}a$.
\loigiai{
Theo định lý Py-ta-go, ta có: $c^2=a^2+b^2\Rightarrow \left(c+b\right)\left(c-b\right)=a^2$.\\
Ta có 
   \begin{eqnarray*}
   VP=2\log_{c+b}a\cdot\log_{c-b}a&=&\log_{c+b}a^2\cdot\log_{c-b}a    =\left[\log_{c+b}\left(c+b\right)\left(c-b\right)\right]\log_{c-b}a\\
  & =&\left(1+\log_{c+b}\left(c-b\right)\right)\log_{c-b}a
    =\log_{c-b}a+\log_{c+b}a=VT.
\end{eqnarray*}
\hfill \rule{2mm}{2mm}
}     
\end{bt}

\begin{bt}%[Hồ Sỹ Trường]%[2D2K3]
Cho ba số dương $a,b,c$ khác nhau và khác $1$. Cho $N>0, N\ne1$. Chứng minh rằng $\dfrac{\log_aN}{\log_cN}=\dfrac{\log_aN-\log_bN}{\log_bN-\log_cN}$ là điều kiện cần và đủ để $a,b,c$ theo thứ tự đó lập thành một cấp số nhân.
\loigiai{
Ta có 
 \begin{eqnarray*}
    \dfrac{\log_aN}{\log_cN}=\dfrac{\log_aN-\log_bN}{\log_bN-\log_cN}&\Leftrightarrow&\dfrac{\log_Nc}{\log_Na}=\dfrac{\tfrac{1}{\log_Na}-\tfrac{1}{\log_Nb}}{\tfrac{1}{\log_Nb}-\tfrac{1}{\log_Nc}}\\
    & \Leftrightarrow&\dfrac{\log_Nc}{\log_Na}=\dfrac{\log_N\tfrac{b}{a}\cdot\log_Nc}{\log_N\tfrac{c}{b}\cdot\log_Na}\Rightarrow\log_N\dfrac{b}{a}=\log_N\dfrac{c}{b}\\
    &\Leftrightarrow& \dfrac{b}{a}=\dfrac{c}{b}.
\end{eqnarray*}
Do đó $a,b,c$ lập thành cấp số nhân.
\hfill \rule{2mm}{2mm}
}
\end{bt}
