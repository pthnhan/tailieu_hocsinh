			\tkzInit[ymin=-1.5,ymax=3,xmin=-2,xmax=3]

			\draw[->] (-3,0)--(0,0)node[below right]{$O$} circle(1pt)--(2,0)node[below]{$x$}; 
			\draw[->] (0,-3)--(0,0)--(0,2.5)node[right]{$y$};
			\draw [  domain=-1.2:1.2, samples=100] plot (\x, {2*(\x)^2-1});
			\draw [  domain=-1.4142:1.4142, samples=100] plot (\x, {sqrt(2-(\x)^2)});
			\draw (-1.41,0) node[below]{$-\sqrt{2}$} circle(1pt);
			\fill[pattern=north east lines,draw=none](-1,1) plot[domain=-1:1] (\x,{sqrt(2-(\x)^2)}) (1,1) plot[domain=1:-1](\x, {2*(\x)^2-1});

			\draw (0,1.41) node[above left]{$\sqrt{2}$} circle(1pt);

	\begin{align*}
		&2x^2-1=\sqrt{2-x^2}\\
		\Leftrightarrow &\heva{&(2x^2-1)^2=2-x^2\\ &2x^2-1\ge 0\\ &2-x^2\ge 0}
		\Leftrightarrow \heva{&4x^4-3x^2-1=0\\ &\dfrac{1}{2}\le x^2\le x^2}\\
		\Leftrightarrow &x^2=1
		\Leftrightarrow x=\pm 1.
	\end{align*}

	\begin{align*}
	S_H &=\int\limits_{-1}^{1} |2x^2-1-\sqrt{2-x^2}| \mathrm{\,d}x

	=\dfrac{2}{3} + \left(\dfrac{1}{2}\sin(2t)+t\right)\bigg|_{\frac{-\pi}{4}}^{\frac{\pi}{4}}
	=\dfrac{3\pi + 10}{6}.  
	\end{align*}
	}

\begin{ex}%[Đề THPT L1, Hoàng Mai, Nghệ An 2018]%[Thái Văn Đạt, EX10]%[2D1G5-7]

Vậy $0<x_1<1<x_2<3<x_3<4$.
}
\end{ex}

\begin{ex}%[Đề THPT L1, Hoàng Mai, Nghệ An 2018]%[Thái Văn Đạt, EX10]%[2D1K3-2]
Gọi $S$ là tập hợp các giá trị nguyên dương của tham số $m$ để giá trị nhỏ nhất của hàm số $y=|x^2-4x+3|+4mx$ lớn hơn $2$. Số phần tử của $S$ là
\choice
