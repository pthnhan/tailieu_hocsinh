\chapter{MỘT SỐ DẠNG PHƯƠNG TRÌNH NGHIỆM NGUYÊN}

\section{PHƯƠNG TRÌNH MỘT ẨN}

\begin{vd}
Tìm nghiệm nguyên của phương trình $(x-2)(3x-2)(5x-2)(7x-2)=945$.
\loigiai{
Đặt $A=(x-2)(3x-2)(5x-2)(7x-2)$.
\begin{itemize}
\item Nếu $x\geq 3$ thì $\heva{&x-2\geq 1\\ &3x-2\geq 7\\ & 5x-2\geq 13\\ & 7x-2\geq 19}\Rightarrow A\geq 1.7.13.19=1729$ (loại).
\item Nếu $x\leq -2$ thì $\heva{&x-2\leq -4\\ &3x-2\leq -8\\ & 5x-2\leq -12\\ & 7x-2\leq -16}\Rightarrow A\geq 4.8.12.16=6164$ (loại).
\end{itemize}
Vì $x\in \mathbb Z$ nên $x\in \{-1;0;1;2\}$.
\begin{itemize}
\item Với $x=-1$ thì $A=(-3).(-5).(-7).(-9)=945$ (thỏa mãn).
\item Với $x=0$ thì $A=(-2)^{4}=16$ (loại).
\item Với $x=1$ thì $A=(-1).1.3.5=-15$ (loại).
\item Với $x=2$ thì $A=0$ (loại).
\end{itemize}
Kết luận: nghiệm của phương trình là $x=-1$.
}
\end{vd}
\begin{note}
Kinh nghiệm giải toán:
\begin{itemize}
\item Nếu khai triển ta phải giải phương trình bậc cao (bậc bốn), thật không đơn giản chút nào!
\item Ta thường sử dụng điều kiện $x$ là số nguyên và dùng phương pháp xét từng khoảng giá trị của ẩn để giải.
\end{itemize}
\end{note}
\begin{vd}
Tìm nghiệm nguyên của phương trình
\[
(x^{2}-1)(x^{2}-11)(x^{2}-21)(x^{2}-31)=-4224.
\]
\loigiai{
Đặt $A=(x^{2}-1)(x^{2}-11)(x^{2}-21)(x^{2}-31)$. Vì $A<0$ và là tích của bốn thừa số $x^{2}-1$, $x^{2}-11$, $x^{2}-21$, $x^{2}-31$ nên trong bốn thừa số trên phải có một hoặc ba thừa số âm.\\
Nhận thấy $x^{2}-1>x^{2}-11>x^{2}-21>x^{2}-31$ nên có hai trường hợp:
\begin{itemize}
\item Trường hợp có ba thừa số âm $\Rightarrow x^{2}-1>0>x^{2}-11\Rightarrow 1<x^{2}<11\Rightarrow x^{2}\in \{4;9\}$.
\begin{itemize}
\item Nếu $x^{2}=4$ thì $A=3.(-7).(-17).(-27)=-9639$ (loại).
\item Nếu $x^{2}=9$ thì $A=8.(-2).(-12).(-22)=-4224$ (thỏa mãn).
\end{itemize}
\item Trường hợp có một thừa số âm $\Rightarrow x^{2}-21>0>x^{2}-31\Rightarrow 21<x^{2}<31\Rightarrow x^{2}=25$\\
$\Rightarrow A=24.14.4.(-6)=-8064$ (loại).
\end{itemize}
Vậy $x^{2}=9\Rightarrow x=\pm 3$.
}
\end{vd}

\begin{bt}
Tìm nghiệm nguyên của mỗi phương trình sau:
\begin{enumerate}
\item $x^{4}=24x+9$.
\item $x^{3}-3x^{2}+490=0$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Xét $x^{4}-24x-9=x^{4}-81-24(x-3)=(x^{2}-9)(x^{2}+9)-24(x-3)=(x-3)(x^{3}+3x^{2}+9x+3)$. Do vậy\\
$x^{4}=24x+9\Leftrightarrow (x-3)(x^{3}+3x^{2}+9x+3)=0\Leftrightarrow \hoac{&x-3=0\\ & x^{3}+3x^{2}+9x+3=0.}$
\begin{itemize}
\item Với $x\geq 0\Rightarrow x^{3}+3x^{2}+9x+3>0$ (loại).
\item Với $x< 0\Rightarrow x\leq -1\Rightarrow x^{3}+3x^{2}+9x+3<-1+3-9+3=-4<0$ (loại).
\end{itemize}
Vậy $x=3$ là nghiệm nguyên của phương trình.
\item Xét $x^{3}-3x^{2}+490=0\Leftrightarrow x^{3}-3x^{2}=-490\Leftrightarrow x^{2}(x-3)=-49.10$.\\
Vì $x\in \mathbb Z$ nên $\heva{& x^{2}=49\\ & x-3=-10}\Rightarrow x=-7$.
\end{enumerate}
}
\end{bt}
\begin{bt}
Tìm nghiệm nguyên của mỗi phương trình sau:
\begin{enumerate}
\item $(2x-1)(3x-1)(4x-1)(5x-1)=24$.
\item $(x^{2}-17)(x^{2}-27)(x^{2}-37)=-4032$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Đặt $A=(2x-1)(3x-1)(4x-1)(5x-1)$.
\begin{itemize}
\item Nếu $x\geq 2\Rightarrow A\geq 3.5.7.9=945>24$ (loại).
\item Nếu $x\leq 0\Rightarrow A\leq (-1)^{4}=1<24$ (loại).
\end{itemize}
Vậy $0<x<2$. Vì $x\in\mathbb Z$ nên $x=1$.
\item Đặt $A=(x^{2}-17)(x^{2}-27)(x^{2}-37)$. Vì $A<0$ và $A$ là tích của ba số hạng $x^{2}-17$, $x^{2}-27$, $x^{2}-37$ nên có một số hạng hoặc cả ba số hạng trên đều âm.\\
Ta thấy $x^{2}-17>x^{2}-27>x^{2}-37$ nên ta có hai trường hợp sau:
\begin{itemize}
\item Trường hợp có một số hạng âm $\Rightarrow x^{2}-27>0>x^{2}-37\Rightarrow 27<x^{2}<37$\\
$\Rightarrow x^{2}=36\Rightarrow A=19.9.(-1)=-171$ (loại).
\item Trường hợp có ba số hạng âm $\Rightarrow x^{2}-17<0\Rightarrow x^{2}\in \{0;1;4;9;16\}$.
\begin{itemize}
\item Với $x^{2}=0\Rightarrow A=(-17).(-27).(-37)=-16983$ (loại).
\item Với $x^{2}=1\Rightarrow A=(-16).(-26).(-36)=-14976$ (loại).
\item Với $x^{2}=4\Rightarrow A=(-13).(-23).(-33)=-9867$ (loại).
\item Với $x^{2}=9\Rightarrow A=(-8).(-18).(-28)=-4032$ (thỏa mãn) $\Rightarrow x=\pm 3$.
\item Với $x^{2}=16\Rightarrow A=(-1).(-11).(-21)=-231$ (loại).
\end{itemize}
\end{itemize}
Vậy nghiệm nguyên của phương trình là $x=\pm 3$.
\end{enumerate}
}
\end{bt}
