\begin{ex}%[9D1B7]%[9D1K7]
	Với $x > 0$, cho hai biểu thức  $A=\dfrac{2+\sqrt{x}}{\sqrt{x}}$ và  $B =\dfrac{\sqrt{x}-1}{\sqrt{x}}+\dfrac{2\sqrt{x}+1}{x+\sqrt{x}}$. 
	\begin{enumerate}
		\item 	Tính giá trị biểu thức $A$ khi $x = 64$.
		\item	Rút gọn biểu thức $B$.
		\item 	Tính $x$ để  $\dfrac{A}{B}>\dfrac{3}{2}$. 
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Thay $x = 64$ (thỏa mãn) vào biểu thức $A$ ta được
		\begin{align*}
		A = \dfrac{2 + \sqrt{64}}{\sqrt{64}} = \dfrac{2 + 8}{8} = \dfrac{10}{8} = \dfrac{5}{4}.
		\end{align*}
		\item Mẫu thức chung của biểu thức $B$ là: $\sqrt{x}(\sqrt{x} + 1)$. 
		\begin{align*}
		B &=\dfrac{\sqrt{x}-1}{\sqrt{x}}+\dfrac{2\sqrt{x}+1}{x+\sqrt{x}}\\
		&= \dfrac{(\sqrt{x} - 1)(\sqrt{x} + 1)}{\sqrt{x}(\sqrt{x} + 1)} + \dfrac{2\sqrt{x} + 1}{\sqrt{x}(\sqrt{x} + 1)}\\
		&= \dfrac{x - 1 + 2\sqrt{x} + 1}{\sqrt{x}(\sqrt{x} + 1)}\\
		&= \dfrac{\sqrt{x}(\sqrt{x} + 2)}{\sqrt{x}(\sqrt{x} + 1)}\\
		&= \dfrac{\sqrt{x} + 2}{\sqrt{x} + 1}.
		\end{align*} 
		\item Ta có $\dfrac{A}{B} > \dfrac{3}{2}$ thì 
		\begin{align*}
		\dfrac{2 + \sqrt{x}}{\sqrt{x}}.\dfrac{\sqrt{x} + 1}{\sqrt{x} + 2} > \dfrac{3}{2} \Leftrightarrow \dfrac{\sqrt{x} + 1}{\sqrt{x}} > \dfrac{3}{2} \Leftrightarrow \dfrac{2(\sqrt{x} + 1)}{2\sqrt{x}} - \dfrac{3\sqrt{x}}{2\sqrt{x}} > 0 \Leftrightarrow \dfrac{2 - \sqrt{x}}{2\sqrt{x}} > 0
		\end{align*}
		Ta thấy $2\sqrt{x} > 0$ suy ra $\dfrac{2 - \sqrt{x}}{2\sqrt{x}} > 0 \Leftrightarrow 2 - \sqrt{x} > 0 \Leftrightarrow 2 > \sqrt{x} \Leftrightarrow x < 4$.\\
		Vậy $0 < x < 4$ thì $\dfrac{A}{B} > \dfrac{3}{2}$.
	\end{enumerate}
}
\end{ex} 

\begin{ex} %[9D3B6]
	\textit{Giải bài toán sau bằng cách lập phương trình:}\\
	Quãng đường từ A đến B dài $90$  km. Một người đi xe máy từ A đến B. Khi đến B, người đó nghỉ 30 phút rồi quay trở về A với vận tốc lớn hơn vận tốc lúc đi là $9$  km/h. Thời gian kể từ lúc bắt đầu đi từ A đến lúc trở về đến A là 5 giờ. Tính vận tốc xe máy lúc đi từ A đến B.
	\loigiai{
	Gọi vận tốc xe máy lúc đi là $x$ (km/h) (Điều kiện: $x > 0$).\\
	Vì vận tốc xe máy lúc về lớn hơn vận tốc lúc đi là $9$ (km/h) nên vận tốc lúc về là $x + 9$ (km/h).\\
	Suy ra thời gian lúc đi là $\dfrac{90}{x}$ (giờ). Thời gian lúc về là $\dfrac{90}{x + 9}$ (giờ).\\
	Khi đến $B$ xe nghỉ lại $30$ phút $ = \dfrac{1}{2}$ giờ. Mà tổng thời gian cả đi và về là $5$ giờ nên ta có phương trình
	\begin{align*}
	\dfrac{90}{x} + \dfrac{90}{x + 9} + \dfrac{1}{2} = 5 \Leftrightarrow 9x^2 -279x - 1620 = 0 \Leftrightarrow x^2 - 31x - 180 = 0 \Leftrightarrow \hoac{x &= -5\hspace{0.1cm}\text{(Loại)}\\ x&= 36\hspace{0.1cm}\text{(TM)}}
	\end{align*} 
	Vậy vận tốc xe máy lúc đi là $36$ km/h.
}
\end{ex}  
\begin{ex}%[2D4B7]%[9D4B6]
	\begin{enumerate}[1)]
		\item Giải hệ phương trình:  $\left\{ \begin{aligned}
		3(x+1)+2(x+2y)=4 \\ 
		4(x+1)-(x+2y)=9 \\ 
		\end{aligned} \right.$. 
		\item 	Cho parabol $(P)$:  $y=\dfrac{1}{2}x^2$  và đường thẳng (d):  $y=mx-\dfrac{1}{2}m^2+m+1$.
		\begin{enumerate}
			\item 	Với $m = 1$, xác định tọa độ giao điểm $A$, $B$ của $(d)$ và $(P)$.
			\item	Tìm các giá trị của $m$ để $(d)$ cắt $(P)$ tại hai điểm phân biệt có hoành độ $x_1$, $x_2$ sao cho:  $\big| x_1-x_2 \big|=2$.
		\end{enumerate}
	\end{enumerate}
	\loigiai{
\begin{enumerate}[1)]
	\item $\heva{&3(x + 1) + 2(x + 2y) = 4\\&4(x + 1) - (x + 2y) = 9} \Leftrightarrow \heva{&5x + 4y = 1\\&3x - 2y = 5} \Leftrightarrow \heva{x &= 1\\ y&= -1}$.\\
	Vậy nghiệm của hệ phương trình là $(x; y) = (1 ; -1)$.
	\item \begin{enumerate}
		\item Thay $m = 1$ vào $(d)$ ta có $y = x - \dfrac{1}{2} + 2 = x + \dfrac{3}{2}$.\\
		Xét phương trình hoành độ giao điểm của $(d)$ và $(P)$
		\begin{align*}
		\dfrac{1}{2}x^2 = x + \dfrac{3}{2} \Leftrightarrow x^2 = 2x + 3 \Leftrightarrow x^2 - 2x - 3 = 0 \Leftrightarrow \hoac{x &= -1\\ x &= 3}.
		\end{align*}
		Vậy phương trình có hai nghiệm $x_1 = -1, x_2 = 3$.
		\begin{itemize}
			\item Với $x = -1$ thay vào $(P): y = \dfrac{1}{2} \Rightarrow A\left( -1 ; \dfrac{1}{2}\right) $.
			\item  Với $x = 3$ thay vào $(P): y = \dfrac{9}{2} \Rightarrow B\left( 3 ; \dfrac{9}{2}\right) $.
		\end{itemize}
	\item Xét phương trình hoành độ của $(d)$ và $(P)$ ta có:
	\begin{align*}
	\dfrac{1}{2}x^2 = mx - \dfrac{1}{2}m^2 + m + 1 \Leftrightarrow x^2 - 2mx + m^2 - 2m - 2.
	\end{align*}
	Xét $\Delta ' = m^2 - (m^2 - 2m - 2) = 2m + 2$.\\
	 Để $(d)$ và $(P)$ giao nhau tại hai điểm phân biệt thì $\Delta ' > 0 \Leftrightarrow 2m + 2 > 0 \Leftrightarrow m > -1$.\\
	Theo Vi-et ta có: $\heva{&x_1 + x_2 = 2m\\ &x_1\cdot x_2 = m^2 - 2m - 2}$.\\
	Theo bài ra ta có: 
	\begin{align*}
	|x_1 - x_2| = 2 &\Leftrightarrow (x_1 - x_2)^2 = 4\\
	&\Leftrightarrow (x_1 + x_2)^2 - 4x_1x_2 = 4\\
	&\Leftrightarrow (2m)^2 - 4(m^2 - 2m - 2) = 4\\
	&\Leftrightarrow 2m + 2 = 1\\
	&\Leftrightarrow m = -\dfrac{1}{2}.
	\end{align*}
	Vậy $m = -\dfrac{1}{2}$ thỏa mãn điều kiện bài toán.
	\end{enumerate}
\end{enumerate}	
}
\end{ex}  

\begin{ex}%[9H3B7]
	Cho đường tròn $(O)$ và điểm $A$ nằm bên ngoài $(O)$. Kẻ hai tiếp tuyến $AM, AN$ với đường tròn $(O)$. Một đường thẳng $d$ đi qua $A$ cắt đường tròn $(O)$ tại hai điểm $B$ và $C$ ( $AB < AC$, $d$ không đi qua tâm $O$).
	\begin{enumerate}
		\item 	Chứng minh tứ giác $AMON$ nội tiếp.
		\item 	Chứng minh $AN^2 = AB \cdot AC$. Tính độ dài đoạn thẳng $BC$ khi $AB=4$ cm, $AN=6$ cm.
		\item	Gọi $I$ là trung điểm $BC$. Đường thẳng $NI$ cắt đường tròn $(O)$ tại điểm thứ hai $T$. Chứng minh: $MT \parallel AC$.
		\item 	Hai tiếp tuyến của đường tròn $(O)$ tại $B$ và $C$ cắt nhau tại $K$. Chứng minh $K$ thuộc một đường thẳng cố định khi $d$ thay đổi và thỏa mãn điều kiện đầu bài.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\tkzDefPoint(0,0){O}
			\tkzDefPoint(-5,0){A}
			\tkzDefPoint(-0.8,5){K}
			\tkzDrawCircle[R](O,2 cm)
			\tkzTangent[from with R=A](O, 2 cm)
			\tkzGetPoints{N}{M}
			\tkzTangent[from with R=K](O, 2 cm)
			\tkzGetPoints{B}{C}
			\tkzDefMidPoint(B,C)\tkzGetPoint{I}
			\tkzInterLC(N,I)(O,B)\tkzGetSecondPoint{T}
			\tkzLabelPoints[below right](O)
			\tkzLabelPoints[left](A)
			\tkzLabelPoints[above left](M)
			\tkzLabelPoints[below](N)
			\tkzLabelPoints[above](K)
			\tkzLabelPoints[above left](B)
			\tkzLabelPoints[right](C)
			\tkzLabelPoints[above right](T)
			\tkzLabelPoints(I)
			\tkzDrawLines[add=0 and 0.85](C,B)
			\tkzDrawLines[add=0 and 0.2](A,M)
			\tkzDrawLines[add=0 and 0.2](A,N)
			\tkzDrawLines[add=0 and 0.3](K,B)
			\tkzDrawLines[add=0 and 0.3](K,C)
			\tkzDrawPoints(O,A,M,N,K,B,C,I,T)
			\tkzDrawSegments(K,N O,M O,N O,B O,C O,K I,M N,T M,T A,O)
			\tkzMarkRightAngles(A,N,O A,M,O K,B,O K,C,O)
			\end{tikzpicture}
		\end{center}
	\begin{enumerate}
		\item Ta có $\widehat{AMO} = \widehat{AMO} = 90^\circ $ (Tính chất tiếp tuyến).\\
		Do đó: $\widehat{AMO} + \widehat{ANO} = 180^\circ$, mà hai góc này ở vị trí đối nhau nên tứ giác $AMON$ nội tiếp.
		\item Xét $\triangle AMB$ và $\triangle ACM$ có
		\begin{itemize}
			\item $\widehat{MAC}$ chung
			\item $\widehat{MCA} = \widehat{AMB}$ (cùng chắn cung $\wideparen{MB}$)
		\end{itemize}
	Suy ra $\triangle AMB \backsim \triangle ACM$.\\
	$\Rightarrow \dfrac{AM}{AB} = \dfrac{AC}{AM}$ (Tính chất tam giác đồng dạng)\\
	$\Rightarrow AM^2 = AB \cdot AC$.\\
	Mà $AM = AN$ (Tính chất hai tiếp tuyến cắt nhau).\\
	Vậy $AN^2 = AB \cdot AC$ (Đpcm).\\
	Ta có $AN^2 = AB \cdot AC \Rightarrow 36 = 4 \cdot AC \Rightarrow AC = 9$ (cm).\\
	Mà $AC = AB + BC \Rightarrow BC = AC - AB = 9 - 4 = 5$ (cm).
\item Vì $I$ là trung điểm của dây cung $BC$ không đi qua tâm $O$ nên $OI \perp BC$ hay $\widehat{AIO} = 90^\circ$, như vậy $I$ và $N$ cùng nhìn đoạn $AO$ dưới góc vuông nên $A, N, O, I$ cùng nằm trên một đường tròn đường kính $AO$, do đó: $\widehat{AIN} = \widehat{AON}$ (cùng chắn cung $\wideparen{NA}$).\\
Mặt khác $\widehat{MTN} = \dfrac{1}{2}\widehat{MON} = \widehat{AON}$. Suy ra $\widehat{MTN} = \widehat{AIN}$ và chúng ở vị trí đồng vị nên $MT \parallel AC$ (đpcm).
\item Xét $\triangle BOK$ vuông tại $B$ (Tính chất tiếp tuyến), có đường cao $BI$ nên $OB^2 = OI.OK$ mà $OB = OM \Rightarrow OM^2 = OI.OK \Rightarrow \dfrac{OM}{OI} = \dfrac{OK}{OM}$.\\ 
Góc $MOI$ chung nên $\triangle OIM \backsim \triangle OMK$ (c - g - c). Từ đó suy ra 
\begin{align}\label{1}
\widehat{MIO} = \widehat{OMK}
\end{align}
Ta có: $OM = ON$ nên
\begin{align}\label{2}
\widehat{OMN} = \widehat{ONM}
\end{align}
Vì bốn điểm $M, N, I, O$ cùng nằm trên một đường tròn đường kính $AO$ và từ \eqref{2} nên
\begin{align}\label{3}
\widehat{NMO} + \widehat{MIO} = \widehat{MNO} + \widehat{MIN} = 180^\circ
\end{align}
Từ \eqref{1} và \eqref{3}, suy ra: $\widehat{NMO} + \widehat{KMO} = 180^\circ$, do đó ba điểm $M, N, K$ thẳng hàng hay suy ra $K$ luôn nằm trên đường thẳng $MN$ cố định khi $d$ thay đổi.
\end{enumerate}
}
\end{ex}  

\begin{ex}%[9D5G1]
	Với $a$, $b$, $c$ là các số dương thỏa mãn điều kiện $a + b + c + ab + bc + ca = 6abc$.
	Chứng minh:  $\dfrac{1}{a^2}+\dfrac{1}{b^2}+\dfrac{1}{c^2} \geq 3$ .
\loigiai{
\begin{align*}
\left(\dfrac{1}{a} - \dfrac{1}{b} \right)^2 \geq 0 &\Leftrightarrow \dfrac{1}{a^2} + \dfrac{1}{b^2} \geq \dfrac{2}{ab}\Leftrightarrow \dfrac{1}{2}\left(\dfrac{1}{a^2} + \dfrac{1}{b^2} \right) \geq \dfrac{1}{ab}  
\end{align*}
Theo bất đẳng Cauchy ta có
\begin{itemize}
	\item $\dfrac{1}{2}\left(\dfrac{1}{a^2} + \dfrac{1}{b^2} \right) \geq \dfrac{1}{ab}; \dfrac{1}{2}\left(\dfrac{1}{b^2} + \dfrac{1}{c^2} \right) \geq \dfrac{1}{bc}; \dfrac{1}{2}\left(\dfrac{1}{a^2} + \dfrac{1}{c^2} \right) \geq \dfrac{1}{ac} $
	\item $\dfrac{1}{2}\left(\dfrac{1}{a^2} + 1 \right) \geq \dfrac{1}{a}; \dfrac{1}{2}\left(\dfrac{1}{b^2} + 1 \right) \geq \dfrac{1}{b}; \dfrac{1}{2}\left(\dfrac{1}{c^2} + 1 \right) \geq \dfrac{1}{c} $
\end{itemize}
Cộng lần lượt các vế 
\begin{align*}
&\dfrac{3}{2} + \dfrac{3}{2}\left(\dfrac{1}{a^2} + \dfrac{1}{b^2} + \dfrac{1}{c^2} \right) \geq \dfrac{1}{ab} + \dfrac{1}{ac} + \dfrac{1}{bc} + \dfrac{1}{a} + \dfrac{1}{b} + \dfrac{1}{c}\\
&\Leftrightarrow\dfrac{3}{2} + \dfrac{3}{2}\left(\dfrac{1}{a^2} + \dfrac{1}{b^2} + \dfrac{1}{c^2} \right)\geq \dfrac{c + b + a + bc + ac + ab}{abc}\\
&\Leftrightarrow\dfrac{3}{2} + \dfrac{3}{2}\left(\dfrac{1}{a^2} + \dfrac{1}{b^2} + \dfrac{1}{c^2} \right)\geq 6\\
 &\Leftrightarrow\dfrac{3}{2}\left(\dfrac{1}{a^2} + \dfrac{1}{b^2} + \dfrac{1}{c^2} \right) \geq 6 - \dfrac{3}{2} = 9\\
 &\Leftrightarrow\left(\dfrac{1}{a^2} + \dfrac{1}{b^2} + \dfrac{1}{c^2} \right) \geq 3.  
\end{align*}
Dấu "=" xảy ra khi $a = b = c = 1$ (đpcm).
}
\end{ex}
