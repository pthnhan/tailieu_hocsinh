		\end{enumerate}

	\begin{enumerate}

		\item Gọi $M$ là trung điểm của $BC$. Chứng minh $MA$ là tiếp tuyến chung của hai đường tròn $(O)$ và $(O')$.

		\begin{enumerate}
			\item 
			Ta có $OB \parallel O'C$ (cùng vuông góc với $BC$).\\
			$\Rightarrow$ Tứ giác $OBCO'$ là hình thang vuông.\\
			$\Rightarrow \widehat{BOO'}+ \widehat{CO'O}= 180^\circ$.\\
			$\triangle CO'A$ cân tại $O'$ có $\widehat{CAO’}=\dfrac{180^\circ-\widehat{CO’O}}{2}=90^\circ-\dfrac{\widehat{CO’O}}{2}. \hfill(1)$\\
			$\triangle BOA$ cân tại $O$ có 	$\widehat{BAO}=\dfrac{180^\circ-\widehat{BOO’}}{2}=90^\circ-\dfrac{\widehat{BOO’}}{2}.\hfill(2)$\\ 
			Từ $(1$) và $(2)$ 
			$\begin{aligned}[t]
			\Rightarrow\widehat{CAO’}+\widehat{BAO}&=90^\circ-\dfrac{\widehat{CO’O}}{2}+90^\circ-\dfrac{\widehat{BOO’}}{2}\\
			&=180^\circ-\dfrac{\widehat{BOO’}+\widehat{CO’O}}{2}\\
			&=180^\circ-90^\circ=90^\circ.\end{aligned}$.\\
			Lại có $\widehat{CAO'}+\widehat{BAO}+\widehat{BAC}= 180^\circ \Rightarrow \widehat{BAC}= 180^\circ-90^\circ = 90^\circ$.\\
			$\Rightarrow \triangle ABC$ vuông tại $A$.
			\item Ta có $M$ là trung điểm của cạnh huyền $BC$

	\end{enumerate}	
	}
\end{bt}
