\begin{ex}%[Đề thi HSG 9, tỉnh Quảng Trị, năm 2018]%[Hoàng Hải, Dự án EX-9G-18]%[9D0K1]%[9D0G7]
\begin{enumerate}
	\item Rút gọn biểu thức $ P = \sqrt{109-36\sqrt{7}} + \sqrt{109+36\sqrt{7}} $.
	\item Xét 3 số thực $ a,b,c $ thay đổi và thỏa mãn điều kiện $ \heva{&a^3+b^3+c^3=3abc\\&a+b+c\ne 0.} $ Chứng minh $ Q = \dfrac{a^2+3b^2+5c^2}{(a+b+c)^2} $ có giá trị không đổi.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có 
	\begin{align*}
		P = & \sqrt{109-36\sqrt{7}} + \sqrt{109+36\sqrt{7}}>0\\
		& \Leftrightarrow  P^2 & = 218+2\sqrt{(109-36\sqrt{7})(109+36\sqrt{7})}
	\end{align*}
		\begin{eqnarray*}
		  	& & P\hspace*{1ex} = \sqrt{109-36\sqrt{7}} + \sqrt{109+36\sqrt{7}}>0\\
			& \Leftrightarrow & P^2 = 218+2\sqrt{(109-36\sqrt{7})(109+36\sqrt{7})}\\
			& \Leftrightarrow & P^2 = 324\\
			& \Leftrightarrow & P\hspace*{1ex} = 18.
		\end{eqnarray*}
		Vậy $ P=18 $.
	\item Ta có $ (a+b+c)^3=a^3+b^3+c^3+3ab(a+b)+3ac(a+c)+3bc(b+c)+6abc $. \\
	Do đó
		\begin{eqnarray*}
			& & a^3+b^3+c^3=3abc\\
			& \Leftrightarrow & (a+b+c)^3-3ab(a+b)-3ac(a+c)-3bc(b+c)-6abc=3abc\\
			& \Leftrightarrow & (a+b+c)^3-3ab(a+b+c)-3c(a^2+ac+b^2+bc+2ab)=0\\
			& \Leftrightarrow & (a+b+c)^3-3ab(a+b+c)-3c(a+b)(a+b+c)=0\\
			& \Leftrightarrow & (a+b+c)\left[(a+b+c)^2-3ab-3ac-3bc \right]=0\\
			& \Leftrightarrow & \left[(a+b+c)^2-3ab-3ac-3bc \right]=0 \text{ (vì }a+b+c\ne 0)\\
			& \Leftrightarrow & a^2+b^2+c^2-ab-ac-bc=0\\
			& \Leftrightarrow & (a-b)^2+(a-c)^2+(b-c)^2=0 \text{ (nhân cả hai vế cho 2)}\\
			& \Leftrightarrow & a=b=c.
		\end{eqnarray*}
	Vậy $ Q=\cfrac{a^2+3a^2+5a^2}{(a+a+a)^2}=1 $ có giá trị không đổi.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Quảng Trị, năm 2018]%[Hoàng Hải, Dự án EX-9G-18]%[9D0K2]%[9D0G2]
\begin{enumerate}
	\item  Giải phương trình $ x^2-3x+3-\sqrt{x-2}-\sqrt{7-x}=0 $.
	\item Giải hệ phương trình $\heva{&x+3=2\sqrt{(3y-x)(y+1)}\\&\sqrt{2y-3}-\sqrt{x-y}=x-3.}$
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Điều kiện $ 2\le x\le 7 $. Phương trình tương đương
	\begin{eqnarray*}
		& & (x^2-3x)+(1-\sqrt{x-2})+(2-\sqrt{7-x})=0\\
		& \Leftrightarrow & x(x-3)+\dfrac{3-x}{1+\sqrt{x-2}}+\dfrac{x-3}{2+\sqrt{7-x}}=0\\
		& \Leftrightarrow & (x-3)\left(x-\dfrac{1}{1+\sqrt{x-2}} + \dfrac{1}{2+\sqrt{7-x}} \right) = 0\\
		&\Leftrightarrow&x=3.
	\end{eqnarray*}	
	Vì $ \heva{&x\ge 2\\&\dfrac{1}{1+\sqrt{x-2}}\le 1} \Rightarrow x-\dfrac{1}{1+\sqrt{x-2}} > 0 \Rightarrow x-\dfrac{1}{1+\sqrt{x-2}}+\dfrac{1}{2+\sqrt{7-x}} > 0 $.\\
	Vậy phương trình có nghiệm duy nhất $ x=3 $.
	\item Nhận xét $x\ge -3$ và $y\ge\dfrac{3}{2}$. Xét phương trình
	\begin{eqnarray*}
		& & x+3=2\sqrt{(3y-x)(y+1)} \hspace*{2ex}(x+3\ge 0)\\
		& \Leftrightarrow & (x+3)^2=4\sqrt{(3y-x)(y+1)}^2\\
		&\Leftrightarrow& x^2+2(5+2y)x+9-12y^2-12y=0 \qquad(1)\\
		& & \Delta_x^{'}=(5+2y)^2-(9-12y^2-12y)=16(y+1)^2 \ge 0 \\
		& \Rightarrow & \sqrt{\Delta_x^{'}}=4(y+1) \text{ (vì $ y\ge\dfrac{3}{2} $)}\\
		(1)& \Leftrightarrow &\hoac{&x=-6y-9\le -6\cdot\dfrac{3}{2}-9<-3 \quad\text{(loại)} \\&x=2y-1.}
	\end{eqnarray*}
	Thế $ x=2y-1 $ vào phương trình còn lại ta có
	\begin{eqnarray*}
		& & \sqrt{2y-3}-\sqrt{y-1}=2y-4\\
		& \Leftrightarrow&\sqrt{2y-3}-1+1-\sqrt{y-1}=2y-4\\
		& \Leftrightarrow&\dfrac{2(y-2)}{\sqrt{2y-3}+1}+\dfrac{2-y}{1+\sqrt{y-1}}=2(y-2)\\
		& \Leftrightarrow & (y-2)\left(\dfrac{2}{\sqrt{2y-3}+1}-\dfrac{1}{1+\sqrt{y-1}}-2\right)=0\\
		& \Leftrightarrow & \hoac{&y=2\\&\dfrac{2}{\sqrt{2y-3}+1}-\dfrac{1}{1+\sqrt{y-1}}-2=0.}
	\end{eqnarray*}
	Ta có $\dfrac{2}{\sqrt{2y-3}+1}\le 2 \Rightarrow\left(\dfrac{2}{\sqrt{2y-3}+1}-2 \right) -\dfrac{1}{1+\sqrt{y-1}}<0, \forall y\ge \dfrac{3}{2}$.\\
	Thế $ (x;y)=(3;2) $ vào hệ phương trình ban đầu ta thấy thỏa. \\
	Vậy hệ phương trình có nghiệm duy nhất $ (x;y)=(3;2) $.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Quảng Trị, năm 2018]%[Hoàng Hải, Dự án EX-9G-18]%[9D0K3]%[9D0G4]
\begin{enumerate}
	\item Cho đa thức $ P(x)=x^4+ax^3+bx^2+cx+d$ ($ a,b,c,d\in\mathbb{R} $) thỏa
		$ P(1)=3,P(2)=6,P(3)=11 $. Tính $ S=10P(4)+P(-2) $.
	\item Phân tích số $ 1603^{2018} $ thành tổng của một số số hạng nguyên dương. Gọi $ S $ là tổng các lập phương của tất cả các số hạng đó. Hỏi $ S $ chia $ 6 $ dư bao nhiêu?
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Đa thức $ P(x) $ được biểu diễn dưới dạng $ P(x)=(x-k)(x-1)(x-2)(x-3)+b'x^2+c'x+d' $.\\
	Ta có $\heva{&P(1)=3\\&P(2)=6\\&P(3)=11}  
		\Leftrightarrow \heva{&b'+c'+d'=3\\&4b'+2c'+d'=6\\&9b'+3c'+d'=11} 
		\Leftrightarrow \heva{&b'=1\\&c'=0\\&d'=2.}$\\
	Vậy $ P(x)=(x-k)(x-1)(x-2)(x-3)+x^2+2 $. Khi đó
	\begin{eqnarray*}
	  S & = & 10P(4)+P(-2)\\
		& = & 10\left[6(4-k)+18\right]-60(-2-k)+6\\
		& = & 546.
	\end{eqnarray*}
	Vậy $ S=546 $.
	\item Giả sử $ 1603^{2018}=n_1+n_2+\cdots n_k$ với $ n_1,n_2,\ldots,n_k $ là $ k $ số nguyên dương. Khi đó
	\begin{eqnarray*}
	  S & = & n_1^3+n_2^3+\cdots+n_k^3\\
		& = & (n_1^3-n_1)+(n_2^3-n_2)+\cdots+(n_k^3-n_k)+(n_1+n_2+\cdots+n_k)\\
		& = & (n_1^3-n_1)+(n_2^3-n_2)+\cdots+(n_k^3-n_k)+1603^{2018}.
	\end{eqnarray*}
	\begin{itemize}
		\item Với $ n\in\mathbb{N} $ ta có $ n^3-n=n(n-1)(n+1)\vdots 6$.
		\item Ta có $ (1603-1)\vdots 6 \Rightarrow (1603^{2018}-1^{2018})\vdots 6 \Rightarrow (1603^{2018}-1)\vdots 6 \Rightarrow 1603^{2018}$ chia $ 6 $ dư $ 1 $.
	\end{itemize}
	Vậy $ S $ chia $ 6 $ dư $ 1 $.
\end{enumerate}
}
\end{ex}

\begin{enumerate}
	\item Xét $ 3 $ số thực không âm $ a,b,c $ thỏa mãn $ a+b+c=1 $ và không có hai số nào đồng thời bằng $ 0 $. Tìm giá trị nhỏ nhất của biểu thức
	$$ P=\dfrac{2}{(a+b)(b+c)}+\dfrac{2}{(c+a)(a+b)}+(c+2)(3+a+b). $$
	\item\immini{Một hình tròn được chia thành $ 10 $ ô hình quạt bằng nhau. Trên mỗi ô người ta đặt một viên bi. Nếu ta thực hiện liên tục thao tác lấy ở $ 2 $ ô bất kỳ mỗi ô $ 1 $ viên bi rồi chuyển sang ô liền kề thì có thể chuyển được tất cả số viên bi về cùng một chỗ không?}{\begin{tikzpicture}
