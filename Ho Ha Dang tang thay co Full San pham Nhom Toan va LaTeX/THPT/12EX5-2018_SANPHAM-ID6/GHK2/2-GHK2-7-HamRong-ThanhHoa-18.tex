	Tìm tích tất cả các nghiệm của phương trình $4\cdot3^{\log\left(100x^2\right)}+9\cdot4^{\log(10x)}=13\cdot6^{1+\log x}$.
	\choice
	{\True $1$}
	{$0{,}1$}
	{$100$}
	{$10$}
	\loigiai
	{
		Đặt $t=\log x$, phương trình đã cho trở thành
		\begin{eqnarray*}
			& & 4\cdot3^{2+2t}+9\cdot2^{2+2t}=78\cdot6^t\\
			& \Leftrightarrow & 6\cdot\left(\dfrac{3}{2}\right)^{2t}-13\cdot\left(\dfrac{3}{2}\right)^t+6=0\\
			& \Leftrightarrow & \hoac{&\left(\dfrac{3}{2}\right)^t=\dfrac{3}{2}\\&\left(\dfrac{3}{2}\right)^t=\dfrac{2}{3}} \Leftrightarrow t=1 \text{ hoặc }t=-1.
		\end{eqnarray*} 
		Khi đó
		\begin{itemize}
			\item Với $t=1 \Leftrightarrow \log x=1 \Leftrightarrow x=10$.
			\item Với $t=-1 \Leftrightarrow \log x=-1 \Leftrightarrow x=10^{-1}$.
		\end{itemize}
		Vậy tích các nghiệm của phương trình đã cho là $10\times10^{-1}=1$.
	}
\end{ex}

				& \le & \dfrac{1}{6}\cdot SA\cdot SB \cdot SC=\dfrac{a^3\sqrt{6}}{6}.
			\end{eqnarray*}
			Suy ra thể tích $SABC$ lớn nhất bằng $\dfrac{a^3\sqrt{6}}{6}$ khi \[\sin \alpha =\sin \beta=1 \Rightarrow \alpha=\beta=90^\circ.\]
		}
		{\begin{tikzpicture}[line cap=round,line join=round,x=1.0cm,y=1.0cm,>=stealth,scale=1]
			\tkzDefPoints{3/-2/B, 5/0/C, 0/0/A}
			\coordinate (H) at ($(A)!0.5!(B)$);

			\coordinate (y) at ($(S)!(A)!(x)$);
			\tkzDrawSegments[dashed](A,C A,y)
			\tkzDrawSegments(S,C S,x)

			\tkzMarkAngle[size=1.2](B,S,C) \tkzLabelAngle[pos = 0.8](B,S,C){$\beta$}
			\tkzMarkAngle[size=1.,fill=yellow!30!blue,opacity=0.4](A,S,x) \tkzLabelAngle[pos = 0.8](A,S,x){$\alpha$}
			\tkzMarkRightAngles[size=0.2](A,y,x)
			\tkzDrawPoints[fill=black](A,B,C,S)

	}

		{

			\begin{eqnarray*}

			\end{eqnarray*}

		}
