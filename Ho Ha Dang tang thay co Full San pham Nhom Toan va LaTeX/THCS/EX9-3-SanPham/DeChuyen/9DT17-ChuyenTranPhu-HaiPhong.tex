\begin{ex}%[Đề thi vào 10, Chuyên Trần Phú, Hải Phòng năm 2017]%[9D1B8]%[9D4B5]
\begin{enumerate}[1)]
\item Cho biểu thức $Q=\left(\dfrac{1}{\sqrt{x}-1}-\dfrac{2}{x-1}\right)\left(\dfrac{x+\sqrt{x}}{\sqrt{x}+1}-\dfrac{1-\sqrt{x}}{\sqrt{x}-x}\right)$ với $x>0$, $x\ne 1$.
	\begin{enumerate}[a)]
	\item Rút gọn biểu thức $Q$.
	\item Tìm các giá trị của $x$ để $Q=-1$.
	\end{enumerate}
\item Cho phương trình $x^2-2(m-1)x-2017m^2-1=0$. Tìm $m$ để phương trình có hai nghiệm phân biệt $x_1<x_2$ thoả mãn $|x_1|-|x_2|=2018$.
\end{enumerate}
\loigiai{
\begin{enumerate}[1)]
\item \begin{enumerate}[a)]
\item Với $x>0$ và $x\ne 1$, ta có
\begin{align*}
Q&=\dfrac{1}{\sqrt{x}+1}\left(\sqrt{x}-\dfrac{1}	{\sqrt{x}}\right)=\dfrac{1}{\sqrt{x+1}}\cdot 		\dfrac{x-1}{\sqrt{x}}=\dfrac{\sqrt{x}-1}{\sqrt{x}}.
\end{align*}
\item Với $x>0$ và $x\ne 1$, ta có
$$Q=-1 \Leftrightarrow \sqrt{x}-1=-\sqrt{x}\Leftrightarrow \sqrt{x}=\dfrac{1}{2}\Leftrightarrow x=\dfrac{1}{4}.$$
\end{enumerate}

\item Dễ thấy phương trình đã cho có $ac=1\cdot (-2017m^2-1)<0$ nên phương trình đã cho luôn có hai nghiệm $x_1$, $x_2$ thoả mãn $x_1<0<x_2$. Từ đó ta có
$x_1=m-1-\sqrt{\Delta'}$ và $x_2=m-1+\sqrt{\Delta'}$. Suy ra
\begin{align*}
|x_1|-|x_2|=2018\Leftrightarrow -m+1+\sqrt{\Delta'}-m+1-\sqrt{\Delta'}=2018 \Leftrightarrow m=-1008.
\end{align*}

\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Trần Phú, Hải Phòng năm 2017]%[9D4K7]%[9D3K4]
\begin{enumerate}[a)]
\item Giải phương trình $\sqrt{x+1}-\sqrt{x-7}=\sqrt{12-x}$.
\item Giải hệ phương trình $\heva{&x^3+xy^2-10y=0\\ &x^2+6y^2=10}$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item Điều kiện của phương trình là $7\le x\le 12$. Với điều kiện đó, phương trình tương đương
\begin{align*}
&\sqrt{x+1}=\sqrt{12-x}+\sqrt{x-7}\\
\Leftrightarrow &x+1=12-x + 2\sqrt{12-x}\sqrt{x-7} + x-7\\
\Leftrightarrow &x-4=2\sqrt{12-x}\sqrt{x-7}\\
\Leftrightarrow &x^2-8x+16=-4x^2+76x-336\\
\Leftrightarrow &5x^2-84x+352=0\\
\Leftrightarrow &\hoac{&x=\dfrac{44}{5}\\ &x=8}
\end{align*}
Thử lại, ta được tập nghiệm của phương trình là $S=\left\{\dfrac{44}{5};8\right\}$.

\item Dễ thấy $x\ne 0$ và $y\ne 0$. Khi đó, hệ phương trình đã cho tương đương
\begin{align*}
\heva{&\left(\dfrac{x}{y}\right)^3+\dfrac{x}{y}=\dfrac{10}{y^2}\\ &\left(\dfrac{x}{y}\right)^2+6=\dfrac{10}{y^2}}
\end{align*}
Trừ hai phương trình theo vế ta được $\left(\dfrac{x}{y}\right)^3-\left(\dfrac{x}{y}\right)^2+\dfrac{x}{y}-6=0$ hay $\dfrac{x}{y}=2$. Thay vào phương trình thứ hai của hệ ta được $y^2=1$ hay $y=\pm 1$. Từ đó $x=\pm 2$.\\
Vậy các nghiệm của hệ phương trình đã cho là $(2;1)$ và $(-2;-1)$. 
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Trần Phú, Hải Phòng năm 2017]%[9H3B1]%[9H3B7]%[9H3K7]
Cho tam giác $ABC$ nhọn nội tiếp đường tròn $(O)$, $Y$ trên cạnh $CA$, $Z$ trên cạnh $AB$ sao cho $\widehat{AZY}>90^\circ$. Gọi $I$ là tâm đường tròn ngoại tiếp tam giác $AYZ$, $S$ là giao điểm khác $A$ của $AI$ và đường tròn $(O)$.
\begin{enumerate}[a)]
\item Chứng minh rằng $\widehat{SAC}=\widehat{AZY}-90^\circ$.
\item Gọi $X$ là giao điểm của $YZ$ và $BC$, $M$ là giao điểm khác $Y$ của các đường tròn ngoại tiếp tam giác $AYZ$ và $CXY$. Chứng minh rằng $M$ nằm trên đường tròn $(O)$.
\item Gọi $J$, $K$ là tâm các đường tròn ngoại tiếp tam giác $BZX$ và $CXY$, $T$ là giao điểm của $AI$ và $BJ$. Chứng minh rằng sáu điểm $T$, $O$, $M$, $I$, $J$, $K$ cùng nằm trên một đường tròn.
\end{enumerate}
\loigiai{

\begin{enumerate}[a)]
\item Ta có $\widehat{AZY}=\dfrac{360^\circ-\widehat{AIY}}{2}=180^\circ-\dfrac{\widehat{AIY}}{2} \Rightarrow \widehat{AZY}-90^\circ=90^\circ-\dfrac{\widehat{AIY}}{2}=\widehat{IAY}=\widehat{SAC}$.

\item Do $\widehat{AMC}=\widehat{AMY}+\widehat{YMC}=\widehat{BZY}+\widehat{YXC}=180^\circ-\widehat{ABC}$ nên $M$ thuộc đường tròn ngoại tiếp tam giác $ABC$ hay đường tròn $(O)$.

\item Ta có $\widehat{ZMX}=\widehat{ZMY}+\widehat{XMY}=\widehat{BAC}+\widehat{ACB}=180^\circ -\widehat{XBZ}$ hay $M$ thuộc đường tròn $(J)$ ngoại tiếp tam giác $BZX$.\\
Ta chứng minh $T\equiv S$. Thật vậy, ta có $\widehat{SBX}=\widehat{SAC}=\widehat{AZX}-90^\circ=90^\circ - \widehat{BZX}=90^\circ-\dfrac{\widehat{BJX}}{2}=\widehat{JBX}$. Do tam giác $BZX$ nhọn nên $J$ nằm trong tam giác $BZX$, suy ra $J$ nằm trên nửa mặt phẳng bờ $BC$ chứa $A$. Do tam giác $\widehat{AZY}>90^\circ$ nên $I$ nằm trong góc $\widehat{AZY}$ và ngoài tam giác $AZY$, suy ra tia $AC$ nằm giữa tia $AB$ và $AS$, do đó $S$ nằm trên nửa mặt phẳng bờ $BC$ chứa $A$. Vậy $J$ và $S$ nằm cùng nửa mặt phẳng bờ $BX$, do đó $B$, $J$, $S$ thẳng hàng. Suy ra $T\equiv S$.\\
Dễ thấy $OI$, $OJ$ là đường trung trực của $MA$, $MB$ tương ứng nên $\widehat{IOJ}=180^\circ-\widehat{AMB}=\widehat{ISJ}$ hay tứ giác $IOJS$ nội tiếp một đường tròn $(\mathfrak{C})$.\\
Tương tự $IK$ và $JK$ lần lượt là đường trung trực của $MY$ và $MX$ nên $\widehat{IKJ}=\widehat{XMY}=\widehat{ACB}=\widehat{ASB}=\widehat{ISJ}$ hay $K$ thuộc $(\mathfrak{C})$.\\
Dễ thấy $\widehat{JMB}=\widehat{JBM}=\widehat{IAM}=\widehat{IMA}$, từ đó $\widehat{IMJ}=\widehat{IMB}+\widehat{JMB}=\widehat{IMB}+\widehat{IMA}=\widehat{AMB}=\widehat{ISJ}$ hay $M$ thuộc đường tròn $(\mathfrak{C})$.\\
Vậy cả sáu điểm $T$, $O$, $M$, $I$, $J$, $K$ cùng thuộc đường tròn $(\mathfrak{C})$.
\end{enumerate}
}

\begin{ex}%[Đề thi vào 10, Chuyên Trần Phú, Hải Phòng năm 2017]%[9D5G1]
Cho các số dương $a$, $b$, $c$, chứng minh rằng
$$\dfrac{a^4}{b^3(c+2a)}+\dfrac{b^4}{c^3(a+2b)}+\dfrac{c^4}{a^3(b+2c)}\ge 1.$$
\loigiai{
Ta có các đánh giá sau
\begin{align*}
&\dfrac{a^4}{b^3(c+2a}+\dfrac{c+2a}{9a}+\dfrac{1}{3}\ge \dfrac{a}{b}\\
&\dfrac{b^4}{c^3(a+2b)}+\dfrac{a+2b}{9b}+\dfrac{1}{3}\ge \dfrac{b}{c}\\
&\dfrac{c^4}{a^3(b+2c)}+\dfrac{b+2c}{9c}+\dfrac{1}{3}\ge \dfrac{c}{a}. 
\end{align*}
Lấy tổng các bất đẳng thức trên theo vế ta được
\begin{align*}
&\dfrac{a^4}{b^3(c+2a)}+\dfrac{b^4}{c^3(a+2b)}+\dfrac{c^4}{a^3(b+2c)}\ge \dfrac{8}{9} \left( \dfrac{a}{b}+\dfrac{b}{c}+\dfrac{c}{a}\right)-\dfrac{5}{3} \ge \dfrac{8}{9}\cdot 3\sqrt[3]{\dfrac{a}{b}\cdot\dfrac{b}{c}\cdot\dfrac{c}{a}}-\dfrac{5}{3}=1.
\end{align*}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Trần Phú, Hải Phòng năm 2017]%[9D5G3]%[9D5G5]
\begin{enumerate}[a)]
\item Chứng minh rằng không tồn tại các số tự nhiên $x$, $y$, $z$ sao cho $x^{16}+y^{16}+2017=z^{16}$.
\item $A$ và $B$ chơi một trò chơi, $A$ chơi trước. Ban đầu có $n$ viên sỏi. Trong mỗi lượt chơi của mình, người chơi sẽ lấy ra $4$, $5$, hoặc $7$ viên sỏi. Quá trình đó tiếp tục như vậy. Ai đến lượt chơi của mình mà không thể lấy thêm sỏi là thua cuộc. Biết cả hai đều là người chơi thông minh, chứng minh rằng nếu $n$ có dạng $11k+l$ với $k,l\in \mathbb{N}, 0\le l\le 3$ thì $B$ thắng cuộc.
\end{enumerate}
\loigiai{\begin{enumerate}[a)]
\item Giả sử tồn tại các số tự nhiên $x$, $y$, $z$ thoả mãn bài toán.
	\begin{enumerate}
	\item[•] Trường hợp 1: $z$ là số chẵn.\\
	Khi đó $x$, $y$ khác tính chẵn lẻ. Ta có thể coi $x$ lẻ, $y$ chẵn. Suy ra
	$$x^{16}+2017\equiv z^{16}-y^{16}\equiv 0\ (\mathrm{mod}\ 2^{16}) \Rightarrow x^{16}+2017\equiv 0\ (\mathrm{mod}\ 64).$$
	Vì $x$ lẻ nên $x^2\equiv 1\ (\mathrm{mod}\ 8)$, do đó $x^{16}\equiv 1\ (\mathrm{mod}\ 64)$. Suy ra $x^{16}+2017\equiv 34\ (\mathrm{mod}\ 64)$. Ta gặp mâu thuẫn.
	
	\item[•] Trường hợp 2: $z$ là số lẻ.\\
	Khi đó $x$, $y$ cùng tính chẵn lẻ.
	Nếu $x$, $y$ cùng lẻ thì 
	$$1\equiv z^{16} \equiv x^{16}+y^{16}+2017\equiv 2019\ (\mathrm{mod}\ 64).$$ 
	hay $64\;|\; 2018$. Vô lí.\\
	Do đó $x$, $y$ cùng chẵn. Suy ra
	$$1\equiv z^{16} \equiv x^{16}+y^{16}+2017\equiv 2017\ (\mathrm{mod}\ 64).$$
	hay $64\;|\;2016$. Vô lí. 
	\end{enumerate}
	Vậy không tồn tại các số tự nhiên $x$, $y$, $z$ thoả mãn bài toán.

\item Ta chứng minh bằng phương pháp quy nạp theo $k$.\\
Nếu $k=0$ thì hiển nhiên $A$ thua.\\
Nếu $k=1$ thì ta có chiến thuật chơi của $B$ như sau. Trường hợp $0\le l\le 1$, nếu $A$ lấy $4$, $5$ hoặc $7$ viên sỏi thì $B$ tương ứng lấy $7$, $5$ và $4$ viên sỏi. Trường hợp $2\le l\le 3$, nếu $A$ lấy $4$, $5$ hoặc $7$ viên sỏi thì $B$ tương ứng lấy ra $7$, $7$ hoặc $4$ viên sỏi.
Giả sử với mọi $m\le k$, $m\in \mathbb{N}$ thì $B$ luôn có chiến thuật chơi để thắng $A$. Ta chứng minh với $m=k+1$ thì $B$ cũng thắng.\\
Thật vậy, khi đó số viên sỏi là $11(k+1)+l=11k+l+11$. Do giả thiết quy nạp nên $B$ có chiến thuật chơi sao cho với $11k+l$ thì $B$ thắng. Do đó chiến thuật chơi đó sẽ làm cho số sỏi còn dư lại của $11k+l$ viên là $p$ với $0\le p\le 3$. Ta quay lại trường hợp $k=1$. Do đó với $n=11(k+1)+l$ thì $B$ thắng.\\
Vậy theo nguyên lí quy nạp $B$ luôn thắng.
\end{enumerate}

}
\end{ex}
