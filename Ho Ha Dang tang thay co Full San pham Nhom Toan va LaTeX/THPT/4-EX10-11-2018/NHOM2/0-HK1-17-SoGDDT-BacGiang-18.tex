		\item $B\setminus A=B\Leftrightarrow A\cap B=\varnothing\Leftrightarrow \heva{&2\notin B\\ &-2\notin B.}$.
		\item $2\notin B\Leftrightarrow |2m-3|\neq 2m-3\Leftrightarrow 2m-3<0\Leftrightarrow m<\dfrac{3}{2}$.
		\item $-2\notin B\Leftrightarrow |-2m-3|\neq -2m-3\Leftrightarrow |2m+3|\neq -(2m+3)\Leftrightarrow 2m+3>0\Leftrightarrow m>-\dfrac{3}{2}$.
	\end{itemize}	
}
\end{ex}
