\section{ĐẠO HÀM VÀ VI PHÂN} %Viết in hoa
\subsection{Kiến thức trọng tâm}
\begin{description}
	\item[Định nghĩa đạo hàm, vi phân]\
    \begin{itemize}
        \item Cho hàm số $y=f(x)$ xác định trên khoảng $(a;b)$ và điểm $x_0$ thuộc khoảng đó, đặt $\Delta x=x-x_0$ là số gia của biến số và $\Delta y=f(x_0+\Delta x)-f(x_0)$ là số gia của hàm số. Đạo hàm của $f$ tại $x_0$ là $\displaystyle f'(x_0)=y'(x_0)=\lim_{x\rightarrow x_0}\dfrac{f(x)-f(x_0)}{x-x_0}=\lim_{x\rightarrow}\dfrac{\Delta y}{\Delta x}$ (hữu hạn).
        \item Vi phân của hàm số $y=f(x)$ tại điểm $x_0$ ứng với số gia $\Delta x$ được ký hiệu $\mathrm{d}f(x_0)$ là $\mathrm{d}f(x_0)=f'(x_0)\Delta x$ hay $\mathrm{d}y=f'(x)\mathrm{d}x=y'\mathrm{d}x$.
        \item Nếu hàm số $y=f(x)$ có đạo hàm tại $x_0$ thì liên tục tại $x_0$.
    \end{itemize}
	\item[Công thức và quy tắc]\
	\begin{multicols}{2}
	\begin{itemize}
	    \item $y=c\Rightarrow y'=0$, $y=x\Rightarrow y'=1$.
	    \item $y=x^n\Rightarrow y'=n\cdot x^{n-1}$.
	    \item $y=\sqrt{x}\Rightarrow y'=\dfrac{1}{2\sqrt{x}}$ ($x>0$).
	    \item $y=\sqrt[n]{x}\Rightarrow y'=\dfrac{1}{n\sqrt{x^n}}$.
	    \item $y=\sin x\Rightarrow y'=\cos x$.
	    \item $y=\cos x\Rightarrow y'=-\sin x$.
	    \item $y=\tan x\Rightarrow y'=\dfrac{1}{\cos^2 x}$.
	    \item $y=\cot x\Rightarrow y'=\dfrac{-1}{\sin^2 x}$.
	    \item $y=\arcsin x\Rightarrow y'=\dfrac{1}{\sqrt{1-x^2}}$.
	    \item $y=\arccos x\Rightarrow y'=\dfrac{-1}{\sqrt{1-x^2}}$.
	    \item $y=\arctan x\Rightarrow y'=\dfrac{1}{1+x^2}$.
	    \item $y=\mathrm{arccot}\,x\Rightarrow y'=\dfrac{-1}{1+x^2}$.
	    \item $(u+v)'=u'+v'$.
	    \item $(u-v)'=u'-v'$.
	    \item $(u\cdot v)'=u'\cdot v+u\cdot v'$.
	    \item $\left(\dfrac{u}{v}\right)'=\dfrac{u'\cdot v-u\cdot v'}{v^2}$.
	    \item Hàm hợp $\left[f(u(x))\right]'=f'(u)\cdot u'(x)$.
	    \item Hàm ngược $x'_y=\dfrac{1}{y'_x}$.
	\end{itemize}
	\end{multicols}
	\item[Đạo hàm cấp $n$]\
	\begin{itemize}
	    \item Định nghĩa: $f^{(n)}(x)=[f^{(n-1)}(x)]'$.
	    \item Ta có $(\sin x)^{(n)}=\sin\left(x+\dfrac{n\pi}{2}\right)$, $\left(\dfrac{1}{x}\right)^{(n)}=\dfrac{(-1)^n\cdot n!}{x^{n+1}}$.
	    \item Ý nghĩa cơ học: Chuyển động $s=s(t)$ có vận tốc tại điểm $t_0$ là $v(t_0)=s'(t_0)$, gia tốc tại điểm $t_0$ là $a(t_0)=v'(t_0)=s''(t_0)$.
	\end{itemize}
	\newpage
	\item[Tiếp tuyến và tiếp xúc]\
	    \begin{itemize}
	        \item 
            \immini
	        {
	        Đạo hàm của hàm số $y=f(x)$ tại điểm $x_0$ là hệ số góc $k$ của tiếp tuyến của đồ thị tại điểm $x_0$ hay $k=f'(x_0)$.\\
	        Phương trình tiếp tuyến tại điểm $M_0(x_0;f(x_0))$ là \[y=f'(x_0)(x-x_0)+f(x_0).\]
	        }
	        {
	        \begin{tikzpicture}[scale=0.6, font=\footnotesize,line join=round, line cap=round, >=stealth]
    \def\hs{0.25*(\x-1)^2+1}
    \def\xt{-1} \def\xp{7} \def\yt{6.5} \def\yd{-1}
    \draw[->] (\xt,0)--(\xp,0) node [below]{$x$};
    \draw[->] (0,\yd)--(0,\yt) node [left]{$y$};
    \node at (0,0) [below left]{$O$};
    \begin{scope}
    \clip (\xt,\yd) rectangle (\xp,\yt-0.1);
    \draw[smooth,samples=200,domain=0.5:5.7,thick,violet] plot(\x,{\hs})node[below left]{$(C)$};
    \draw[smooth,samples=200,domain=-1:6] plot(\x,{0.5*(\x)+0.25})node[below right]{$\Delta$};
    \draw[smooth,samples=200,domain=-1:6] plot(\x,{1.25*(\x)-1.25});
    \end{scope}
    \draw[dashed] (2,0)|-(0,1.25) (5,0)|-(0,5);
    \fill (2,1.25)node[below right]{$A$} circle(2pt);
    \fill (5,5)node[below right]{$B$} circle(2pt);
    \end{tikzpicture}
	        }
	        \item Tiếp tuyến đi qua điểm $K(a;b)$: Lập phương trình tiếp tuyến tại $x_0$ rồi cho tiếp tuyến đi qua điểm $K(a;b)$ thì tìm ra $x_0$.
	        \item Điều kiện tiếp xúc của hai đồ thị $f(x)$ và $g(x)$ là hệ $\heva{&f(x)=g(x)\\&f'(x)=g'(x)}$ có nghiệm. Nghiệm chung $x_0$ là hoành độ tiếp điểm.
	    \end{itemize}
	\item[Tính gần đúng]\ \\
	Ta có $f(x_0+\Delta x)\approx f(x_0)+f'(x_0)\Delta x$.
	\item[Quy tắc L'Hospital]\
    \begin{itemize}
        \item Giả sử hai hàm số $f$ và $g$ liên tục trên khoảng $(a;b)$ chứa $x_0$, có đạo hàm trên $(a;b)\setminus \{x_0\}$ và có $f(x_0)=g(x_0)=0$. Khi đó $\displaystyle\lim_{x\rightarrow x_0}\dfrac{f'(x)}{g'(x)}=L$ thì $\displaystyle\lim_{x\rightarrow x_0}\dfrac{f(x)}{g(x)}=L$.\\
	    \item Đặc biệt rằng $\displaystyle f'(x_0)=\lim_{x\rightarrow x_0}\dfrac{f(x)-f(x_0)}{x-x_0}\Rightarrow\lim_{x\rightarrow x_0}\dfrac{f(x)-f(x_0)}{x-x_0}=f'(x_0)$.
    \end{itemize}
\end{description}
\subsection{Các bài toán}
\begin{bt}%[Dự án Tex-hóa sách Bồi dưỡng HSG K10, Tác giả: Lê Hoành Phò]%[Danh Trần]%[1D5B1-1]
	Dùng định nghĩa tính đạo hàm của mỗi hàm số tại $x_0$ sau.
	\begin{listEX}[2]
	    \item $y=\dfrac{3x+1}{x-2}$, $x_0=3$.
	    \item $y=\sqrt[3]{x}$, $x_0=\dfrac{1}{8}$.
	\end{listEX}
	\loigiai{
	\begin{listEX}[1]
	    \item Cho $x_0=3$ số gia $\Delta x$ thì $\Delta y=f(3+\Delta x)-f(3)=\dfrac{3(3+\Delta x)+1}{(3+\Delta x)-2}-\dfrac{10}{1}=\dfrac{-7\Delta x}{1+\Delta x}$.\\
	    Ta có $\displaystyle \lim_{\Delta x\rightarrow 0}\dfrac{\Delta y}{\Delta x}=\lim_{\Delta x\rightarrow 0}\dfrac{-7}{1+\Delta x}=-7$. Vậy $f'(3)=-7$.
	    \item Cho $x_0=\dfrac{1}{8}$ số gia $\Delta x$ thì ta có \[\Delta y=f\left(\dfrac{1}{8}+\Delta x\right)-f\left(\dfrac{1}{8}\right)=\sqrt[3]{\dfrac{1}{8}+\Delta x}-\dfrac{1}{2}=\dfrac{\Delta x}{\sqrt[3]{\left(\dfrac{1}{8}+\Delta x\right)^2}+\dfrac{1}{2}\sqrt[3]{\dfrac{1}{8}+\Delta x}+\dfrac{1}{4}}.\]
	    Ta có $\displaystyle \lim_{\Delta x\rightarrow 0}\dfrac{\Delta y}{\Delta x}=\lim_{\Delta x\rightarrow 0}\dfrac{1}{\sqrt[3]{\left(\dfrac{1}{8}+\Delta x\right)^2}+\dfrac{1}{2}\sqrt[3]{\dfrac{1}{8}+\Delta x}+\dfrac{1}{4}}=\dfrac{1}{3\cdot\dfrac{1}{4}}=\dfrac{4}{3}$.
	    Vậy $f'\left(\dfrac{1}{8}\right)=\dfrac{4}{3}$.
	\end{listEX}
}
\end{bt}
\begin{bt}%[Dự án Tex-hóa sách Bồi dưỡng HSG K10, Tác giả: Lê Hoành Phò]%[Danh Trần]%[1D5G1-1]
	Dùng định nghĩa tính đạo hàm của mỗi hàm số sau.
	\begin{listEX}[2]
	    \item $y=\dfrac{1+x}{\sqrt{1-x}}$.
	    \item $y=x^n$ với $n$ nguyên dương.
	\end{listEX}
	\loigiai{
	\begin{listEX}[1]
	    \item Với mọi $x$ thuộc khoảng $(-\infty;1)$, ta có
	    \allowdisplaybreaks
	    \begin{eqnarray*}
	    & \Delta y & = f(x+\Delta x)-f(x)=\dfrac{1+x+\Delta x}{\sqrt{1-x-\Delta x}}-\dfrac{1+x}{\sqrt{1-x}}\\
	    & & = \dfrac{(1+x+\Delta x)\sqrt{1-x}-(1+x)\sqrt{1-x-\Delta x}}{\sqrt{1-x-\Delta x}\cdot\sqrt{1-x}}\\
	    & & =\dfrac{(1+x)(\sqrt{1-x}-\sqrt{1-x-\Delta x})+\Delta x\sqrt{1-x}}{\sqrt{1-x-\Delta x}\cdot\sqrt{1-x}}\\
	    & & =\dfrac{(1+x)\dfrac{\Delta x}{\sqrt{1-x}+\sqrt{1-x-\Delta x}}+\Delta x\sqrt{1-x}}{\sqrt{1-x-\Delta x}\cdot\sqrt{1-x}}.
	    \end{eqnarray*}
	    Ta có $\displaystyle \lim_{\Delta x\rightarrow 0}\dfrac{\Delta y}{\Delta x}=\lim_{\Delta x\rightarrow 0}\dfrac{1+x+\sqrt{1-x}(\sqrt{1-x}+\sqrt{1-x-\Delta x})}{\sqrt{1-x-\Delta x}\cdot\sqrt{1-x}(\sqrt{1-x}+\sqrt{1-x-\Delta x})}=\dfrac{3-x}{2\sqrt{(1-x)^3}}$.\\
	    Vậy $y'=\dfrac{3-x}{2\sqrt{(1-x)^3}}$, với $x<1$.
	    \item Với mọi $x\in\mathbb{R}$, cho số gia $\Delta x$, ta có
	    \allowdisplaybreaks
	    \[\Delta y=(x+\Delta x)^n-x^n=\mathrm{C}_n^1x^{n-1}\Delta x+\mathrm{C}_n^2 x^{n-2}\Delta x^2+\cdots+\mathrm{C}_n^{n-1}x\Delta x^{n-1}+\Delta x^n.\]
	    Ta có $\displaystyle\lim_{\Delta x\rightarrow 0}\dfrac{\Delta y}{\Delta x}=\lim_{\Delta x\rightarrow 0}\left(\mathrm{C}_n^1x^{n-1}+\mathrm{C}_n^2 x^{n-2}\Delta x+\cdots+\mathrm{C}_n^{n-1}x\Delta x^{n-2}+\Delta x^{n-1}\right)=\mathrm{C}_n^1x^{n-1}=nx^{n-1}$.\\
	    Vậy $y'=nx^{n-1}$.
	\end{listEX}
	}
\end{bt}
\begin{bt}%[Dự án Tex-hóa sách Bồi dưỡng HSG K11, Tác giả: Lê Hoành Phò]%[Danh Trần]%[1D5K2-1]
	Chứng minh các hàm số sau liên tục tại $x=0$ nhưng không có đạo hàm tại đó.
	\begin{listEX}[2]
	    \item $y=f(x)=\sqrt{|x|}$.
	    \item $y=f(x)=\dfrac{|x|}{x+1}$.
	\end{listEX}
	\loigiai{
	\begin{listEX}[1]
	    \item Ta có $f(0)=0$, $\displaystyle\lim{x\rightarrow 0}\sqrt{|x|}=0=f(0)$ nên $f$ liên tục tại $x=0$.\\
	    Cho $x=0$ số gia $\Delta x$, ta có $f(0+\Delta x)-f(0)=\sqrt{|\Delta x|}$.\\
	    Ta có $\displaystyle \lim_{x\rightarrow 0^+}\dfrac{\Delta y}{\Delta x}=\lim_{x\rightarrow 0^+}\dfrac{\sqrt{|\Delta x|}}{\Delta x}=\lim_{x\rightarrow 0^+}\dfrac{1}{\sqrt{|\Delta x|}}=+\infty$.\\
	    Vậy không tồn tại đạo hàm tại $x=0$.
	    \item Ta có $f(0)=0$, $\displaystyle\lim_{x\rightarrow 0}\dfrac{|x|}{x+1}=0=f(0)$ nên $f$ liên tục tại $x=0$.\\
	    Cho $x=0$ số gia $\Delta x$ thì $\Delta y=\dfrac{|\Delta x|}{\Delta x+1}-0=\dfrac{|\Delta x|}{\Delta x+1}$.\\
	    Ta có $\displaystyle\lim_{\Delta x\rightarrow 0^-}\dfrac{\Delta y}{\Delta x}=\lim_{\Delta x\rightarrow 0^-}\dfrac{-1}{\Delta x+1}=-1$ và $\displaystyle\lim_{\Delta x\rightarrow 0^+}\dfrac{\Delta y}{\Delta x}=\lim_{\Delta x\rightarrow 0^+}\dfrac{1}{\Delta x+1}=1\ne\lim_{\Delta x\rightarrow 0^-}\dfrac{\Delta y}{\Delta x}.$\\
	    Vậy $f$ không có đạo hàm tại $x=0$.
	\end{listEX}
	}
\end{bt}
\begin{bt}%[Dự án Tex-hóa sách Bồi dưỡng HSG K10, Tác giả: Lê Hoành Phò]%[Danh Trần]%[1D5K2-1]
	Cho hàm số $f(x)=\heva{&\dfrac{1-\sqrt{1-x}}{x}&\text{ khi } x\ne0 \\&\dfrac{1}{2}&\text{ khi } x=0}$. Chứng minh $f$ liên tục và có đạo hàm tại $x=0$.
	\loigiai{
	$f(x)-f(0)=\dfrac{1-\sqrt{1-x}}{x}-\dfrac{1}{2}=\dfrac{2-x-2\sqrt{1-x}}{2x}=\dfrac{(2-x)^2-4(1-x)}{2x(2-x+2\sqrt{1-x})}=\dfrac{x}{2x(2-x+2\sqrt{1-x})}$.
	Ta có $\displaystyle\lim_{x\rightarrow 0}\dfrac{f(x)-f(0)}{x-0}=\lim_{x\rightarrow 0}\dfrac{1}{2x(2-x+2\sqrt{1-x})}=\dfrac{1}{8}$.\\
	Vậy tồn tại $f'(0)=\dfrac{1}{8}$ nên $f$ liên tục và có đạo hàm tại $x=0$
	}
\end{bt}
\begin{bt}%[Dự án Tex-hóa sách Bồi dưỡng HSG K10, Tác giả: Lê Hoành Phò]%[Danh Trần]%[1D5G2-1]
	Cho hàm số $f(x)=\heva{&x^2\sin\dfrac{1}{x}&\text{ khi } x\ne0 \\&0&\text{ khi } x=0}$. Chứng minh $f$ liên tục và có đạo hàm trên $\mathbb{R}$.
	\loigiai{
		Ta có $f(x)=x^2\sin\dfrac{1}{x}$ xác định và liên tục trên $\mathbb{R}\setminus\{0\}$ và $\displaystyle\lim_{x\rightarrow0}f(x)=\lim_{x\rightarrow0}x^2\sin\dfrac{1}{x}=0$ vì $\displaystyle\heva{&\left|x^2\sin\dfrac{1}{x}\right|\le x^2\\&\lim_{x\rightarrow 0}x^2=0.}$\\
		Nên $\displaystyle\lim_{x\rightarrow0}f(x)=f(0)=0$, do đó $f$ liên tục tại $x=0$ nên liên tục trên $\mathbb{R}$.\\
		Khi $x\ne 0$ thì $f'(x)=2x\sin\dfrac{1}{x}-\cos x$. Ta tính $f'$ tại $x=0$. Ta có
		\[\dfrac{\Delta y}{\Delta x}=\dfrac{f(\Delta x)-f(0)}{\Delta x}=\dfrac{(\Delta x)^2\sin\dfrac{1}{\Delta x}-0}{\Delta x}=\Delta x\sin\dfrac{1}{\Delta x}.\]
		Từ đó $\displaystyle\lim_{\Delta x\rightarrow 0}\dfrac{\Delta y}{\Delta x}=0$ vì $\left|\Delta x\sin\dfrac{1}{\Delta x}\right|\le\Delta x\rightarrow 0$ nên $f'(0)=0$.\\
		Vậy $f(x)$ có đạo hàm trên $\mathbb{R}$.
	}
\end{bt}
\begin{bt}%[Dự án Tex-hóa sách Bồi dưỡng HSG K10, Tác giả: Lê Hoành Phò]%[Danh Trần]%[1D5K2-1]
	Tìm $a$, $b$ để hàm số $f(x)=\heva{&x^3-2&\text{ khi } x\le0 \\&x^2+ax+b&\text{ khi } x>0}$ có đạo hàm tại $x=0$. Khi đó tính $f'(0)$.
	\loigiai{
		Hàm số có đạo hàm tại $x=0$ thì liên tục tại $x=0$ nên \[\displaystyle\lim_{x\rightarrow 0^+}f(x)=f(0)\Rightarrow \displaystyle\lim_{x\rightarrow 0^+}(x^2+ax+b)=-2\Rightarrow b=-2.\]
		Ta có $\displaystyle\heva{&\lim_{x\rightarrow 0^+}\dfrac{f(x)-f(0)}{x-0}=\lim_{x\rightarrow0^+}\dfrac{x^2+ax}{x}=\lim_{x\rightarrow0^+}(x+a)=a\\&\lim_{x\rightarrow 0^-}\dfrac{f(x)-f(0)}{x-0}=\lim_{x\rightarrow0^-}\dfrac{x^3}{x}=\lim_{x\rightarrow0^-}x^2=0.}$\\
		Suy ra điều kiện tồn tại đạo hàm tại $x=0$ là $a=0$ và $b'=2$, khi đó $f'(0)=0$.
	}
\end{bt}
