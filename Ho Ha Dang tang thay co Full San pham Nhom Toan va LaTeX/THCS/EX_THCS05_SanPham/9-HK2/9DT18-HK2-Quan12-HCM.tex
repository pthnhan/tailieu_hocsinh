\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Quận 12, Thành phố Hồ Chí Minh]%[9D4B6]%[9D4B7]
	Giải các phương trình sau
	\begin{enumerate}
		\item $3x(x-2)=11-2x^2$.
		\item $(x+1)^2 -2x+1=0$.
	\end{enumerate} 
	\loigiai{
		\begin{enumerate}
			\item Ta có $3x(x-2)=11-2x^2 \Leftrightarrow 5x^2-6x-11=0$.\\
					Phương trình có dạng $a-b+c=5-(-6)+(-11)=0$ nên phương trình có hai nghiệm $x_1=-1$, $x_2=\dfrac{c}{a}=\dfrac{-11}{5}$.		
			\item Ta có $(x+1)^2 -2x+1=x^4 \Leftrightarrow x^4-x^2-2=0$.\\
					Đặt $t=x^2 (t \geq 0)$ phương trình trở thành $t^2-t-2=0\,(*)$. \\
					Ta có $a-b+c=1-(-1)+(-2)=0$ nên $(*)$ có hai nghiệm $t_1=-1$ (loại), $t_2=-\dfrac{-2}{1}=2$ (nhận).\\
					Với $t=2$, ta có $x^2=2 \Leftrightarrow x= \pm \sqrt{2}$.\\
					Vậy phương trình có nghiệm $x=\pm \sqrt{2}$. 
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Quận 12, Thành phố Hồ Chí Minh]%[9D3B5]
	Một mảnh đát hình chữ nhật có chu vi $80$m, biết ba lần chiều rộng kém hai lần chiều dài là $5$m. Tính diện tích mảnh đất đó.
	\loigiai{
		Gọi chiều rộng và chiều dài mảnh đất lần lượt là $x$(m) và $y$(m) $(x,y>0)$.\\
		Theo đề bài ta có hệ phương trình sau
		$$\begin{cases} 2(x+y)&=80\\ 3x-2y&=-5 \end{cases} \Leftrightarrow \begin{cases} x&=15\\y&=25   \end{cases}$$
		Diện tích mảnh đất bằng $15 \cdot 25 = 375$m$^2$.	
	}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Quận 12, Thành phố Hồ Chí Minh]%[9D4B5]%[9D4B6]%[9D4B6]
	Cho phương trình $x^2-2mx-4m-5=0$.
	\begin{enumerate}
		\item Chứng minh phương trình luôn có nghiệm với mọi giá trị $m$.
		\item Tính tổng và tích hai nghiệm theo $m$.
		\item Gọi $x_1,x_2$ là hai nghiệm của phương trình. Tìm $m$ để phương trình có hai nghiệm thỏa mãn $x_1 ^2 + x_2 ^2 -x_1 x-2 = 2x_1 +2x_2 +27$.		
	\end{enumerate}
\loigiai{	
	\begin{enumerate}
		\item $\Delta' = (-m)^2-(-4m-5)=m^2+4m+5=(m+2)^2+1>0$ với mọi $m$ nên phương trình luôn có nghiệm với mọi giá trị $m$.
		\item Gọi $S$ và $P$ lần lượt là tổng và tích hai nghiệm của phương trình. Khi đó 
				$S=-\dfrac{b}{a}=2m$ và $P=\dfrac{c}{a}=-4m-5$.
		\item \begin{align*}
					&x_1 ^2 + x_2 ^2 -x_1 x-2 = 2x_1 +2x_2 +27\\
					\Leftrightarrow & (x_1 + x_2)^2-3x_1 x_2=2(x_1 + x_2)+27\\
					\Leftrightarrow & (2m)^2-3(-4m-5)=2(2m)+27\\
					\Leftrightarrow & m^2+2m-3=0 \Leftrightarrow \left[\begin{aligned} & m=1 \\ & m=-3. \end{aligned}\right.
				\end{align*}
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Quận 12, Thành phố Hồ Chí Minh]%[8D3B6]
	Một trường tổ chức cho $250$ người bao gồm giáo viên và học sinh đi tham quan Suối Tiên. Biết giá vé vào cổng của một giáo viên là $80000$ đồng, vé vào cổng của một học sinh là $60000$ đồng. Biết rằng nhà trường tổ chức vào đúng dịp lễ Giỗ tổ Hùng Vương nên được giảm $5 \%$ cho mỗi vé, vì vậy nhà trường chỉ phải trả số tiền là $14535000$ đồng. Hỏi có bao nhiêu giáo viên và học sinh đi tham quan?
	\loigiai{
		Gọi $x$ là số giáo viên đi tham quan. $(x \in \mathbb{N}, x<250)$.\\
		Số học sinh đi tham quan là $250-x$.\\
		Tổng số tiền được giảm là $\left[80000x+60000(250-x)\right] \cdot 0{,}05$.\\
		Ta có phương trình 	$\left[80000x+60000(250-x)\right] \cdot \left(1-0{,}05\right)=14535000 \Leftrightarrow x=15$.\\ 
		Vậy có $15$ giáo viên và $235$ học sinh đi tham quan.
	}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Quận 12, Thành phố Hồ Chí Minh]%[9D4K8]
	\immini{Đường tròn đi qua hai đỉnh và tiếp xúc với một cạnh của hình vuông (xem hình vẽ). Tính bán kính $R$ của đường tròn đó, biết rằng cạnh hình vuông dài $12$cm.}{
		\begin{tikzpicture}[scale=0.4]
			\tkzDefPoints{0/0/O, -4/5/X}
			\tkzDrawCircle[radius](O,X)
			\tkzDefPointBy[rotation = center O angle -1](X)
			\tkzGetPoint{A}
			\tkzDefPointBy[rotation = center O angle 105](X)
			\tkzGetPoint{B}
			\tkzDefSquare(A,B)
			\tkzGetPoints{C}{D}
			\tkzDrawSegments(A,B B,C C,D D,A)
			\draw (8,0) node {$12$cm};
			\tkzDrawPoints(O)
		\end{tikzpicture}
	}
	\loigiai{\immini{
		Gọi $O$ là tâm của đường tròn; $H$, $K$, $T$ lần lượt là hình chiếu của $O$ lên các cạnh $AD$, $BC$, $AB$ của hình vuông $ABCD$ như hình vẽ.\\
		Đặt $R=x,x>0$.\\
		$OH^2=OA^2-AH^2=x^2-6^2=x^2-36$.\\
		$OB^2=OK^2+BK^2=x^2+6^2=x^2+36$ $(1)$.\\
		$OB^2=OT^2+BT^2=AH^2+(AB-AT)^2=6^2+(12-OH)^2=36+(12-\sqrt{x^2-36})^2$ $(2)$.\\
		Từ $(1),(2) \Rightarrow x^2+36=36+(12-\sqrt{x^2-36})^2$.\\
		$\Rightarrow x^2=(12-\sqrt{x^2-36})^2$.\\
		$\Rightarrow \left[\begin{aligned} & x=12-\sqrt{x^2-36} \ \ (3) \\ & x=\sqrt{x^2-36}-12 \ \ (4)\end{aligned}\right.$
	}{
		\begin{tikzpicture}[scale=0.4]
			\clip (-7,-7) rectangle (8,7);
			\tkzDefPoints{0/0/O, -4/5/X}
			\tkzDrawCircle[radius](O,X)
			\tkzDefPointBy[rotation = center O angle -1](X)	\tkzGetPoint{A}
			\tkzDefPointBy[rotation = center O angle 105](X)\tkzGetPoint{D}
			\tkzDefPointBy[projection = onto A--D](O)		\tkzGetPoint{H}
			\tkzDefPointBy[projection = onto B--A](O)
			\tkzGetPoint{T}
			\tkzDefPointBy[projection = onto B--C](O)		\tkzGetPoint{K}
			\tkzDefSquare(A,D)
			\tkzGetPoints{C}{B}
			\tkzDrawSegments(A,B B,C C,D D,A O,H O,T O,K O,A O,B)
			\tkzMarkRightAngles[size=0.5](A,H,O O,T,B O,K,C)
			\tkzLabelPoints[above left](A,T)
			\tkzLabelPoints[left](H)
			\tkzLabelPoints[below left](D)
			\tkzLabelPoints(O,K,C)
			\tkzLabelPoints[above right](B)
			\tkzDrawPoints(O,A,B,C,D,T,H,K)
		\end{tikzpicture}	}
		$(3) \Leftrightarrow \sqrt{x^2-36}=12-x \Leftrightarrow \begin{cases} &x^2-36=x^2-24x+144 \\ &x \leq 12 \end{cases} \Leftrightarrow x = \dfrac{15}{2}$.\\
		$(4) \Leftrightarrow \sqrt{x^2-36}=x-12 \Leftrightarrow \begin{cases} &x^2-36=x^2+24x+144 \\ &x \geq -12 \end{cases} \Leftrightarrow x = -\dfrac{15}{2}$ (loại vì $x>0$).\\
		Vậy $R=\dfrac{15}{2}$.
	}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Quận 12, Thành phố Hồ Chí Minh]%[9H3B7]%[9H3K3]%[9H3K7]
	Từ điểm $A$ nằm ở ngoài $(O)$, vẽ hai tiếp tuyến $AB,AC$ ($B,C$ là tiếp điểm), gọi $H$ là giao điểm của $OA$ và $BC$.
	\begin{enumerate}
		\item Chứng minh tứ giác $OBAC$ nội tiếp và $OA \perp BC$.
		\item Qua điểm $C$ vẽ đường thẳng $d$ song song với $OA$, qua điểm $O$ kẻ đường thẳng vuông góc với $OB$ cắt $(O)$ tại $F$ và cắt đường thẳng $d$ tại $K$ (điểm $O$ nằm giữa hai điểm $F$ và $K$), đoạn thẳng $AF$ cắt $(O)$ tại điểm $E$. Chứng minh $AB^2 =AE \cdot AF$. Từ đó suy ra $BE \cdot FC=BF \cdot EC$.
		\item Chứng minh tứ giác $OCKA$ là hình thang cân.
	\end{enumerate}
	\loigiai{\begin{enumerate}
		\immini{
		\item Xét tứ giác $OBAC$ có\\ $\widehat{OBA}+\widehat{OCA}=90^\circ + 90^\circ =180 ^ \circ $\\ nên tứ giác $OBAC$ nội tiếp.\\
				$OB$ và $OC$ là hai tiếp tuyến của $(O)$ nên $ OA \perp BC$.
		}{
		\begin{tikzpicture}[scale=0.8]
			\clip (-4,-4) rectangle (8,4);
			\tkzDefPoints{0/0/O, -2/2/X, 7/0/A}
			\tkzDrawCircle[radius](O,X)
			\tkzTangent[from=A](O,X) \tkzGetPoints{c}{d}
			\tkzDefPointBy[projection = onto A--c](O)\tkzGetPoint{B}
			\tkzDefPointBy[projection = onto A--d](O)\tkzGetPoint{C}
			\tkzInterLL(A,O)(C,B)\tkzGetPoint{H}
			\tkzDefLine[parallel = through C](O,A) \tkzGetPoint{d}
			\tkzDefLine[perpendicular=through O](O,B) \tkzGetPoint{a}
			\tkzInterLL(O,a)(C,d)\tkzGetPoint{K}
			\tkzInterLC(O,a)(O,X)\tkzGetPoints{Z}{F}
			\tkzInterLC(A,F)(O,X)\tkzGetPoints{F}{E}
			\tkzDrawSegments(A,B A,C A,O B,C O,B O,C F,K A,F B,E F,C C,K K,A B,F E,C)
			\tkzMarkRightAngles[size=0.3](O,B,A O,C,A F,O,B)
			\tkzLabelPoints[above](B)
			\tkzLabelPoints[left](F)
			\tkzLabelPoints(C,K,E,H,A)
			\tkzLabelPoints[below left](O)
			\tkzDrawPoints(O,A,B,C,F,H,E,K)
		\end{tikzpicture}}
		\item Hai tam giác $ABE$ và $AFB$ có $\widehat{BAF}$ chung và $\widehat{ABE}=\widehat{AFB}$ (cùng chắn cung $BE$).\\
				$\Rightarrow \triangle ABE \backsim \triangle AFB$ (g-g) $\Rightarrow \dfrac{AB}{AF}=\dfrac{AE}{AB} \Leftrightarrow AB^2=AE\cdot AF$.\\
				Ta có $\triangle ABE \backsim \triangle AFB$ (cmt) $\Rightarrow \dfrac{BE}{BF}=\dfrac{AB}{AF}$. \hfill $(1)$\\
				Tương tự, ta có $\triangle ACE \backsim \triangle AFC \Rightarrow \dfrac{EC}{FC}=\dfrac{AC}{AF}$. \hfill $(2)$\\
				Ta lại có $AB=AC$ (tính chất hai tiếp tuyến cắt nhau). \hfill $(3)$\\
				Từ $(1)$, $(2)$ $\&$ $(3)$ ta có $\dfrac{BE}{BF}=\dfrac{EC}{FC}$ hay $BE \cdot FC = BF \cdot EC$.
		\item Tứ giác $OCKA$ có $OA \parallel CK$ nên là hình thang.\\
				Ta có $AB \perp OB$ và $OK \perp OB$ nên $AB \parallel OK$ $\Rightarrow \widehat{BAO}=\widehat{AOK}$ (hai góc sole trong).\\
				Theo giả thiết, ta có $OA \parallel CK$ nên $\widehat{ACK}=\widehat{CAO}$ (hai góc sole trong).\\
				Mà $\widehat{BAO}=\widehat{CAO}$ (tính chất hai tiếp tuyến cắt nhau).\\
				Do đó, ta có $\widehat{ACK}=\widehat{AOK}$ nên tứ giác $OCKA$ nội tiếp.\\
				Suy ra $OCKA$ là hình thang cân.
\end{enumerate}
}
