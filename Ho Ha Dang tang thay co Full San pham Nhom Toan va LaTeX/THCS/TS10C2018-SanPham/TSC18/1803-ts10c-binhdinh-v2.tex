	\vspace{1pt}
	\begin{enumerate}[1.]
		\item Cho biểu thức $$ T=\dfrac{\left( \sqrt{a}-\sqrt{b}\right)^2+\sqrt{ab}}{\sqrt{a}+\sqrt{b}} : \left(\dfrac{a-b}{\sqrt{a}-\sqrt{b}}-\dfrac{\sqrt{a^3}-\sqrt{b^3}}{a-b}\right), $$ với $ a\ne b, a>0,b>0 $.
		\begin{enumerate}[a)]
			\item Rút gọn biểu thức $ T $.
			\item Chứng tỏ $ T >1 $.
		\end{enumerate}
		\item Cho $ n $ là số tự nhiên chẵn, chứng minh rằng số $ 20^n-3^n+16^n-1 $ chia hết cho số $ 323 $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[1.]
		\item \begin{enumerate}[a)]
				\item Rút gọn biểu thức $ T $.\\
				$ \begin{array}{ll}
				T &=\dfrac{\left( \sqrt{a}-\sqrt{b}\right)^2+\sqrt{ab}}{\sqrt{a}+\sqrt{b}} : \left(\dfrac{a-b}{\sqrt{a}-\sqrt{b}}-\dfrac{\sqrt{a^3}-\sqrt{b^3}}{a-b}\right) \\
				& = \dfrac{a+b-\sqrt{ab}}{\sqrt{a}+\sqrt{b}} : \left(\sqrt{a}+\sqrt{b} -\dfrac{a+b+\sqrt{ab}}{\sqrt{a}+\sqrt{b}} \right) \\
				&= \dfrac{a+b-\sqrt{ab}}{\sqrt{a}+\sqrt{b}} : \dfrac{\sqrt{ab}}{\sqrt{a}+\sqrt{b}}\\
				&= \dfrac{a+b-\sqrt{ab}}{\sqrt{ab}}.\\
				\end{array} $
				\item  Áp dụng bất đẳng thức Cauchy cho hai số dương $ a,b $, cùng với điều kiện $ a \ne b $, ta được \[ a+b >2\sqrt{ab} .\] 
				Do đó
				\[T = \dfrac{a+b-\sqrt{ab}}{\sqrt{ab}} > \dfrac{2\sqrt{ab}-\sqrt{ab}}{\sqrt{ab}} =1 .\]
				Vậy $ T >1 $.
			\end{enumerate}
			\item Bài này ta áp dụng kiến thức, với mọi số $ a,b $ và số tự nhiên $ n $, thì
			\begin{enumerate}[$\bullet$]
				\item $ a^n - b^n  $ chia hết cho $  a-b $, $ a \ne b $.
				\item $ a^{2n+1} + b^{2n+1}  $ chia hết cho $  a+b $.
				\item $ a^{2n} - b^{2n}  $ chia hết cho $ a +b $.
			\end{enumerate}
			Đặt $ A=  20^n-3^n+16^n-1  $.\\
			Ta phân tích $ 323 $ thành tích các số nguyên tố, khi đó $ 323 = 17\cdot 19 $.\\
			Ta sẽ chứng minh $ A $ chia hết cho $ 19 $ và $ A $ chia hết cho $17 $.
			\begin{enumerate}[$\bullet$]
				\item Do $ 20^n-3^n $ chia hết cho $ 20-3=17 $ và $ 16^n-1 $ chia hết cho $ 16+1=17 $ ($ n $ chẵn) nên $ A $ chia hết cho $ 17 $.
				\item Do $ 20^n-1 $ chia hết cho $ 20-1=19 $ và $ 16^n- 3^n $ chia hết cho $ 16+3=19$ ($ n $ chẵn) nên $ A $ chia hết cho $ 19 $.
			\end{enumerate}
			Vậy $  20^n-3^n+16^n-1 $ chia hết cho $ 17\cdot 19=323 $.
			
		\end{enumerate}
	}

	Cho phương trình: $(m-1)x^2-2(2m-3)x-5m+25=0$ ($m$ là tham số). Tìm các giá trị $m$ là số nguyên sao cho phương trình có nghiệm là số hữu tỉ.
	\loigiai{
		$\bullet$ TH1: $m=1$. Phương trình đã cho trở thành
		\begin{center}
			$2x+20=0\Leftrightarrow x=-10$ (thỏa ycbt).
		\end{center}
		$\bullet$ TH2: $m\neq 1$. Ta có $\Delta '=(2m-3)^2-(m-1)(-5m+25)=9m^2-42m+34$.\\
		Phương trình có nghiệm $\Leftrightarrow \Delta '\ge 0 \Leftrightarrow 9m^2-42m+34 \ge 0 \hfill (1)$\\
		Khi đó nghiệm của phương trình là $x_{1,2}=\dfrac{2m-3\pm \sqrt{\Delta}}{m-1}$.\\
		Vậy để phương trình có nghiệm hữu tỉ thì $\sqrt{\Delta} \in \mathbb{Z}\Leftrightarrow 9m^2-42m+34=n^2$ (với $n\in \mathbb{Z})$\\
		$\Leftrightarrow (3m-7)^2+15=n^2\Leftrightarrow (3m-7-n)(3m-7+n)=15$.\\
		Ta có $15=1\cdot 15=-1\cdot (-15)=3\cdot 5=-3\cdot (-5)$ và $3m-7-n<3m-7+n$ nên các trường hợp sau có thể xảy ra:
		\begin{enumerate}[$\bullet$]
			\item $\heva{&3m-7-n=1\\&3m-7+n=15}\Leftrightarrow \heva{&m=5\\&n=7}$ (thỏa (1)).
			\item $\heva{&3m-7-n=3\\&3m-7+n=5}\Leftrightarrow \heva{&m=\dfrac{11}{3}\\&n=1}$ (loại).
			\item $\heva{&3m-7-n=-5\\&3m-7+n=-3}\Leftrightarrow \heva{&m=1\\&n=1}$ (loại).
			\item $\heva{&3m-7-n=-15\\&3m-7+n=-1}\Leftrightarrow \heva{&m=-\dfrac{1}{3}\\&n=7}$ (loại).
		\end{enumerate}
		Vậy các giá trị của $m$ thỏa yêu cầu bài toán là $\{5;1\}$.
	}

	Cho hai số dương $a,b$ thỏa $a+\dfrac{1}{b}=1$. Chứng minh rằng $$\left( a+\dfrac{1}{a} \right)^2+\left( b+\dfrac{1}{b}\right)^2\geqslant \dfrac{25}{2} .$$
	\loigiai{
		Ta có $1=a+\dfrac{1}{b}\geqslant 2\sqrt{\dfrac{a}{b}}\Rightarrow \dfrac{a}{b}\leqslant \dfrac{1}{4} $.
		\begin{align*}
		\left( a+\dfrac{1}{a} \right)^2+\left( b+\dfrac{1}{b}\right)^2 &= a^2+\dfrac{1}{b^2}+\dfrac{1}{a^2}+b^2+4
		\\
		&=a^2+\dfrac{1}{b^2} +\dfrac{1}{a^2}+\dfrac{1}{\dfrac{1}{b^2}}+4\\
		&=a^2+\dfrac{1}{b^2}+\dfrac{a^2+\dfrac{1}{b^2}}{\dfrac{a^2}{b^2}}+4\\
		&\geqslant a^2+\dfrac{1}{b^2}+\dfrac{a^2+\dfrac{1}{b^2}}{\dfrac{1}{16}}+4\\
		&=17\left(a^2+\dfrac{1}{b^2} \right)+4\\
		&\geqslant \dfrac{17\left(a+\dfrac{1}{b} \right)^2}{2}+4\\
		&=\dfrac{25}{2}.
			\end{align*}
		Dấu bằng xảy ra khi và chỉ khi $a=\dfrac{1}{2},b=2$.
	}
\end{ex}
