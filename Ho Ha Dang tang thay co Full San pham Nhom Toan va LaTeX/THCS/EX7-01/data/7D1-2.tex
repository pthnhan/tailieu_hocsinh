\section{PHÉP CỘNG, TRỪ SỐ HỮU TỈ. GIÁ TRỊ TUYỆT ĐỐI}
\subsection{Tóm tắt lí thuyết}
\begin{enumerate}[1)]
\item Ta có thể cộng, trừ hai số hữu tỉ bằng cách viết chúng dưới dạng phân số có cùng một mẫu chung rồi áp dụng quy tắc cộng, trừ phân số.

\begin{enumerate}[\tickEX]
	\item Ta cộng (hay trừ) hai tử số, được tử số mới;
	\item Giữ nguyên phân số của hai mẫu số làm mẫu số mới.\\
	Tức là với $x=\dfrac{a}{m}$, $y=\dfrac{b}{m}$ $(a,b,m\in\mathbb{Z}, m>0$), ta có
	\begin{itemize}
		\item $x+y=\dfrac{a}{m}+\dfrac{b}{m}=\dfrac{a+b}{m}$;
		\item $x-y=\dfrac{a}{m}-\dfrac{b}{m}=\dfrac{a-b}{m}$
	\end{itemize}

	\item Trường hợp hai số hữu tỉ không cùng mẫu số, trước hết phải quy đồng mẫu số rồi tiến hành như trên.


\end{enumerate}
\item Phép cộng số hữu tỉ cũng có tính chất của phép cộng phân số: giao hoán, kết hợp, cộng với số $0$. Mỗi số hữu tỉ đều có một số đối.
\item Quy tắc ``\textit{chuyển vế}''\\
Khi chuyển một số hạng từ vế này sang vế kia của một đẳng thức, ta phải đổi dấu số hạng đó.\\
Tức là với mọi $x,y,z\in\mathbb{Q}$: $x+y=z\Rightarrow x=z-y$.
\item Trong $\mathbb{Q}$ cũng có những tổng đại số, trong đó có thể đổi chỗ các số hạng, đặt dấu ngoặc để nhóm các số hạng một cách tùy ý như các tổng đại số trong $\mathbb{Z}$.
\item Giá trị tuyệt đối (GTTĐ) của một số hữu tỉ, kí hiệu là $|x|$, là khoảng cách từ điểm $x$ đến điểm $0$ trên trục số. Ta có
\[|x|=\begin{cases}
x&\text{nếu}\: x\geq0\\
-x&\text{nếu}\: x<0.
\end{cases}\]
\item Từ định nghĩa giá trị tuyệt đối của số hữu tỉ $x$, ta suy ra một số tính chất như sau.
\begin{enumerate}[(i)]
	\item GTTĐ của hai số đối nhau thì bằng nhau, tức là $|-x|=|x|$;
	\item GTTĐ của một số thì không âm, tức là $|x| \geq 0$ (dấu ``$=$'' xảy ra khi $x=0$);
	\item GTTĐ của một số thì lớn hơn hoặc bằng số đó, tức là $|x| \geq x$ (dấu ``$=$'' xảy ra khi $x \geq 0$);
	\item GTTĐ của một số thì lớn hơn hoặc bằng số đối của số đó, tức là $|x| \geq -x$ (dấu ``$=$'' xảy ra khi $x \leq 0$.)
\end{enumerate}
\end{enumerate}
\begin{note}
Tránh mắc sai lầm trong tình huống $|x|=-x \Leftrightarrow x<0$, mà phải đúng là $|x|=-x \Leftrightarrow x \leq 0$.
\end{note}

\subsection{Các dạng toán}
\begin{dang}[Thực hiện phép tính cộng, trừ các số hữu tỉ]
	Phương pháp giải
\begin{enumerate}[\tickEX]
	\item Viết hai số hữu tỉ dưới dạng hai phân số có cùng mẫu dương (quy đồng mẫu số).
	\item Cộng, trừ hai tử số và giữ nguyên mẫu chung.
	\item Rút gọn kết quả đến phân số tối giản.
\end{enumerate}
\end{dang}
\begin{note}
	Đối với dãy tính có nhiều số hạng, ta có thể áp dụng tính chất giao hoán và kết hợp, quy tắc dấu ngoặc để thực hiện phép tính một cách hợp lí nhất.
\end{note}
\begin{vd}
	Hãy thực hiện các phép tính.
	\begin{listEX}[2]
		\item $\dfrac{9}{17}+\dfrac{5}{17}$;
		\item $\dfrac{9}{17}-\dfrac{5}{17}$.
	\end{listEX}
\loigiai{
\begin{listEX}[2]
	\item $\dfrac{9}{17}+\dfrac{5}{17}=\dfrac{9+5}{17}=\dfrac{14}{17}$.
	\item $\dfrac{9}{17}-\dfrac{5}{17}=\dfrac{9-5}{17}=\dfrac{4}{17}$.
\end{listEX}
}
\end{vd}
\begin{vd}
	Hãy thực hiện các phép tính.
\begin{listEX}[3]
	\item $3+\dfrac{2}{9}$;
	\item $\dfrac{-3}{8}+\dfrac{-5}{12}$;
	\item $\dfrac{-3}{14}+\dfrac{4}{35}$.
	
\end{listEX}
\loigiai{
\begin{enumerate}[a)]
	\item $3+\dfrac{2}{9}=\dfrac{3}{1}+\dfrac{2}{9}=\dfrac{3\cdot 9}{1\cdot 9}+\dfrac{2}{9}=\dfrac{27}{9}+\dfrac{2}{9}=\dfrac{29}{9}$. Vậy $3+\dfrac{2}{9}=\dfrac{29}{9}=3\dfrac{2}{9}$.
	\begin{note}
		Phép cộng này giải thích cách viết một hỗn số dưới dạng một phân số.
	\end{note}
	\item $\dfrac{-3}{8}+\dfrac{-5}{12}=\dfrac{-9}{24}+\dfrac{-10}{24}=\dfrac{-19}{24}$;
	\item $\dfrac{-3}{14}+\dfrac{4}{35}=\dfrac{-15}{70}+\dfrac{8}{70}=\dfrac{-7}{70}=\dfrac{-1}{10}$.
\end{enumerate}

}
\end{vd}

\begin{vd}
	Tính giá trị các biểu thức sau theo cách hợp lí nhất.
	\begin{listEX}
		\item $A=\left(\dfrac{1}{3}-\dfrac{8}{15}-\dfrac{1}{7}\right)+\left(\dfrac{2}{3}+\dfrac{-7}{15}+1\dfrac{1}{7}\right)$;
		\item $B=0,25+\dfrac{3}{5}-\left(\dfrac{1}{8}-\dfrac{2}{5}+1\dfrac{1}{4}\right)$.
	\end{listEX}
\loigiai{
Áp dụng các tính chất giao hoán và kết hợp, ta có:
\begin{enumerate}[a)]
	\item $A=\left(\dfrac{1}{3}+\dfrac{2}{3}\right)-\left(\dfrac{8}{15}+\dfrac{7}{15}\right)+\left(\dfrac{-1}{7}+1\dfrac{1}{7}\right)=1-1+1=1$;
	\item $B=\left(0,25-1\dfrac{1}{4}\right)+\left(\dfrac{3}{5}+\dfrac{2}{5}\right)-\dfrac{1}{8}=\left(\dfrac{1}{4}-1-\dfrac{1}{4}\right)+1-\dfrac{1}{8}=\dfrac{-1}{8}$.
\end{enumerate}}
\end{vd}

\begin{dang}[Tìm số hữu tỉ $x$ trong một đẳng thức chứa phép tính cộng trừ]
	Để tìm được số hữu tỉ $x$ trong một đẳng thức chứa phép tính cộng trừ, ta thực hiện như sau:
	\begin{enumerate}[\tickEX]
		\item Chuyển $x$ về một vế (thường là vế trái);
		\item Chuyển các số hạng đã biết sang vế còn lại;
		\item Rút gọn để tìm $x$.
	\end{enumerate}
\end{dang}
\begin{vd}
	Tìm số hữu tỉ $x$, biết:
	\begin{listEX}[2]
		\item $x+\dfrac{3}{5}=\dfrac{4}{7}$;
		\item $x-\dfrac{5}{6}=\dfrac{1}{6}$;
		\item $\dfrac{-5}{7}-x=\dfrac{-9}{10}$;
		\item $\dfrac{5}{7}-x=10$.
	\end{listEX}
\loigiai{
\begin{enumerate}[a)]
	\item $x+\dfrac{3}{5}=\dfrac{4}{7}\Leftrightarrow x=\dfrac{4}{7}-\dfrac{3}{5}\Leftrightarrow x=\dfrac{20}{35}-\dfrac{21}{35}\Leftrightarrow x=\dfrac{-1}{35}$.
	\item $x-\dfrac{5}{6}=\dfrac{1}{6}\Leftrightarrow x=\dfrac{5}{6}+\dfrac{1}{6}\Leftrightarrow x=1$.
	\item $\dfrac{-5}{7}-x=\dfrac{-9}{10}\Leftrightarrow x=\dfrac{-5}{7}+\dfrac{9}{10}\Leftrightarrow x=\dfrac{-50}{70}+\dfrac{63}{70}=\dfrac{13}{70}$.
	\item $\dfrac{5}{7}-x=10\Leftrightarrow x=\dfrac{5}{7}-10\Leftrightarrow x=\dfrac{5}{7}-\dfrac{70}{7}=\dfrac{-65}{7}$.
\end{enumerate}

}
\end{vd}

\begin{vd}
	Tìm số hữu tỉ $x$, biết:
	\begin{listEX}[2]
		\item $x-\dfrac{1}{5}=\dfrac{3}{2}+\dfrac{1}{6}$;
		\item $\dfrac{4}{5}-\left(\dfrac{-1}{8}\right)=\dfrac{7}{8}-x$;
		\item $-1,25+\dfrac{2}{15}-x=2$;
		\item $2\dfrac{1}{2}-x+\dfrac{4}{5}=\dfrac{2}{3}-\left(\dfrac{-4}{7}\right)$.
	\end{listEX}
	\loigiai{
		\begin{enumerate}[a)]
			\item $x=\dfrac{1}{5}+\dfrac{3}{2}+\dfrac{1}{6}=\dfrac{6}{30}+\dfrac{45}{30}+\dfrac{5}{30}=\dfrac{56}{30}=\dfrac{28}{15}=1\dfrac{13}{15}$.
			\item $x=\dfrac{7}{8}-\dfrac{4}{5}-\dfrac{1}{8}=\left(\dfrac{7}{8}-\dfrac{1}{8}\right)-\dfrac{4}{5}=\dfrac{3}{4}-\dfrac{4}{5}=\dfrac{-1}{20}$;
			\item $x=-1,25+\dfrac{2}{15}-2=-3-\dfrac{1}{4}+\dfrac{2}{15}=-3-\dfrac{7}{60}=-3\dfrac{7}{60}$;
			\item $x=2\dfrac{1}{2}+\dfrac{4}{5}-\dfrac{2}{3}-\dfrac{4}{7}=2+\left(\dfrac{1}{2}+\dfrac{4}{5}\right)-\left(\dfrac{2}{3}+\dfrac{4}{7}\right)
			=2+\dfrac{13}{10}-\dfrac{26}{21}=2+1+\dfrac{3}{10}-1-\dfrac{5}{21}=2\dfrac{13}{210}$.
			
			
		\end{enumerate}
	}
\end{vd}


\begin{dang}[Tìm số hữu tỉ $x$ trong một đẳng thức chứa dấu giá trị tuyệt đối]
	Ta thực hiện các bước sau:
	\begin{enumerate}[\tickEX]
		\item Biến đổi đẳng thức về dạng $|f(x)|=a$, trong đó $f(x)=x-b$ hoặc $f(x)=b-x$, $a$ và $b$ là số hữu tỉ.
		\item Ta có công thức $|f(x)|=a\Leftrightarrow f(x)=a$ hoặc $f(x)=-a$.
		\item Chuyển vế đổi dấu, hoặc rút gọn để tìm $x$.
	\end{enumerate}
\end{dang}
\begin{vd}
	Tìm $x\in\mathbb{Q}$, biết:
	\begin{listEX}[3]
		\item $|2,5-x|=1,3$;
		\item $1,6-|x-0,2|=0$;
		\item $\left|-x-\dfrac{4}{5}\right|=2\dfrac{1}{5}$.
	\end{listEX}
\loigiai{
	\begin{enumerate}[a)]
		\item $|2,5-x|=1,3\Rightarrow 2,5-x=1,3$ hoặc $2,5-x=-1,3$.
		\begin{itemize}
			\item Trường hợp 1: $2,5-x=1,3\Rightarrow  -x=1,3-2,5=-1,2\Rightarrow x=1,2$.
			\item Trường hợp 2: $2,5-x=-1,3\Rightarrow -x=-1,3-2,5=-3,8  \Rightarrow x=3,8$.
		\end{itemize}
		Vậy $x=1,2$ hoặc $x=3,8$.
		\item $1,6-|x-0,2|=0 \Rightarrow |x-0,2|=1,6\Rightarrow x-0,2=1,6$ hoặc $x-0,2=-1,6$.
		\begin{itemize}
			\item TH1: $x-0,2=1,6\Rightarrow x=1,6+0,2=1,8$. 
			\item TH2: $x-0,2=1,6\Rightarrow x=-1,6+0,2=-1,4$.
		\end{itemize}
		Vậy $x=1,8$ hoặc $x=-1,4$.
		\item $\left|-x-\dfrac{4}{5}\right|=2\dfrac{1}{5}\Rightarrow \left|-\left(x+\dfrac{4}{5}\right)\right|=2\dfrac{1}{5}\Rightarrow \left|x+\dfrac{4}{5}\right|=2\dfrac{1}{5}$.\\
		$\Rightarrow x+\dfrac{4}{5}=2\dfrac{1}{5}$ hoặc $x+\dfrac{4}{5}=-2\dfrac{1}{5}$.
		\begin{itemize}
			\item TH1:  $x+\dfrac{4}{5}=2\dfrac{1}{5}\Rightarrow x=2\dfrac{1}{5}-\dfrac{4}{5}=\dfrac{10}{5}+\dfrac{1}{5}-\dfrac{4}{5}=\dfrac{7}{5}$.
			\item TH2: $x+\dfrac{4}{5}=-2\dfrac{1}{5}\Rightarrow x=-2\dfrac{1}{5}-\dfrac{4}{5}=\dfrac{-10}{5}-\dfrac{1}{5}-\dfrac{4}{5}=\dfrac{-15}{5}=-3$.
		\end{itemize}
		Vậy $x=\dfrac{7}{5}$ hoặc $x=-3$.
	\end{enumerate}}
\end{vd}



\begin{dang}[So sánh hai số hữu tỉ âm]
Để so sánh hai số hữu tỉ âm ta thực hiện các bước sau:
\begin{enumerate}[\tickEX]
	\item Tính giá trị tuyệt đối của hai số hữu tỉ âm;
	\item Số nào có giá trị tuyệt đối lớn hơn thì nhỏ hơn.
\end{enumerate}
\end{dang}
\begin{vd}
So sánh các số hữu tỉ sau:
\begin{listEX}[2]
	\item $\dfrac{-13}{15}$ và $\dfrac{11}{-13}$;
	\item $\dfrac{84}{-83}$ và  $\dfrac{-337}{7331}$.
\end{listEX}
\loigiai
{\begin{enumerate}[a)]
		\item Đặt $x=\left|\dfrac{-13}{15}\right|=\dfrac{13}{15}$ và $y=\left|\dfrac{11}{-13}\right|=\dfrac{11}{13}$.\\
		Có $x=\dfrac{13}{15}=\dfrac{169}{195}$ và $y=\dfrac{11}{13}=\dfrac{165}{195}$.\\
		 Mà $\dfrac{169}{195}>\dfrac{165}{195}$ suy ra $x>y$.\\
		 Vậy $\dfrac{-13}{15}<\dfrac{11}{-13}$.
		 \item Đặt $x=\left|\dfrac{84}{-83}\right|=\dfrac{84}{83}$ và $y=\left|\dfrac{-337}{331}\right|=\dfrac{337}{331}$.\\
		 Có $x=\dfrac{84}{83}=1\dfrac{1}{83}=1\dfrac{6}{498}$ và $y=\dfrac{337}{331}=1\dfrac{6}{331}$.\\
		 Mà $1\dfrac{6}{498}<1\dfrac{6}{331}$ suy ra $x<y$. \\
		 Vậy $\dfrac{84}{-83}>\dfrac{-337}{331}$.
	\end{enumerate}
}
\end{vd}


\begin{dang}[Tìm GTLN, GTNN của biểu thức chứa dấu giá trị tuyệt đối]
	Ta sử dụng các tính chất sau:
	\begin{enumerate}[\tickEX]
		\item $|P|=|-P|$;
		\item $|M|\geq M$ (dấu ``='' xảy ra khi và chỉ khi $M\geq0$).
	\end{enumerate}
\end{dang}
\begin{note}
	Ta có một số tính chất quan trọng sau hay sử dụng để làm những bài tập nâng cao.
	
	Với mọi $x,y\in\mathbb{Q}$ thì
	\begin{listEX}[2]
	\item $|x+y|\leq |x|+|y|$;
	\item $|x-y|\geq |x|-|y|$.
	\end{listEX}
Thật vậy:
\begin{enumerate}[a)]
	\item Nếu $x+y\geq0$ thì $|x+y|=x+y$.\\
	Vì $x\leq |x|$, $y\leq|y|$ với mọi $x,y\in\mathbb{Q}$ nên  $|x+y|=x+y\leq|x|+|y|$.\\
	Dấu ``$=$'' xảy ra $\Leftrightarrow x\geq 0$ và $y\geq0$.\\
	Nếu $x+y<0$ thì $|x+y|=-(x+y)=(-x)+(-y)$.\\
	Vì $-x\leq |x|$, $-y\leq |y|$ với mọi $x,y\in\mathbb{Q}$ nên
	\[
	|x+y|=(-x)+(-y)\leq |x|+|y|.
	\]
	Dấu ``$=$'' xảy ra $\Leftrightarrow x\leq 0$ và $y\leq 0$.\\
	Vậy với mọi $x,y\in\mathbb{Q}$ ta luôn có $|x+y|\leq |x|+|y|$.\\
	Dấu ``$=$'' xảy ra $\Leftrightarrow xy\geq 0$.
	\item Theo câu a) ta có: \[|x|=|(x-y)+y|\leq |x-y|+|y|\]
	hay $|x-y|\geq|x|-|y|$.\\
	Dấu ``$=$'' xảy ra $\Leftrightarrow y(x-y)\geq 0$.
\end{enumerate}



\end{note}
\begin{vd}
	Tìm giá trị nhỏ nhất của biểu thức $A=|x+3|+|x-5|$.
	\loigiai{
	Ta có: $A=|x+3|+|x-5|=|x+3|+|5-x|$.\\
	Vì $|x+3|\geq x+3$; $|5-x|\geq 5-x$ nên $A\geq x+3+5-x\Rightarrow A\geq 8$.\\
	Đẳng thức $A=8$ xảy ra khi và chỉ khi\\
	$x+3\geq 0$ và $5-x\geq 0\Leftrightarrow x\geq -3$ và $x\leq 5$ $\Leftrightarrow -3\leq x\leq 5$.\\
	Vậy giá trị nhỏ nhất của biểu thức $A$ là $8$, đạt được khi $-3\leq x\leq 5$. 
	\begin{note}
		Có bạn cho rằng $|M|\geq0$ nên $|x+3|\geq0$ và $|x-5|\geq0$ mà $A=|x+3|+|x-5|$ nên $A\geq0$. Vậy giá trị nhỏ nhất của $A$ là $0$.\\
		Điều này không đúng vì $A=0$ khi và chỉ khi $|x+3|=0$ và $|x-5|=0$, tức là khi và chỉ khi $x=-3$ và $x=5$ đồng thời (vô lí).
	\end{note}}
\end{vd}

\begin{vd}
	Tìm giá trị nhỏ nhất của biểu thức $M=|x+5|+|x+2|+|x-7|+|x-8|$.
	\loigiai{Áp dụng bất đẳng thức $|M|\geq M$, ta có:
	\begin{align*}
	A&=|x+5|+|x+2|+|x-7|+|x-8|\\
	&=|x+5|+|x+2|+|7-x|+|8-x|\\
	&\geq x+5+x+2+7-x+8-x=22
	\end{align*}
	
	$A=22$ khi và chỉ khi $\begin{cases}
	x+5\geq0\\
	x+2\geq0\\
	7-x\geq0\\
	8-x\geq0
	\end{cases}
	\Leftrightarrow
	\begin{cases}
	x\geq -5\\
	x\geq-2\\
	x\leq7\\
	x\leq8
	\end{cases}
	\Leftrightarrow -2\leq x\leq 7$.\\
	Giá trị nhỏ nhất của $A$ là $22$, đạt được khi $-2\leq x\leq 7$.
}
\end{vd}
\begin{vd}
	Tìm giá trị lớn nhất của biểu thức $B=0,5-|x-4,5|$.
	\loigiai{Vì $|x-4,5|\geq 0$ nên $0,5-|x-4,5| \leq 0,5$. Do đó, $B=0,5-|x-4,5| \leq 0,5$. Dấu ``$=$'' xảy ra khi và chỉ khi $|x-4,5|=0\Leftrightarrow x=4,5$. Vậy giá trị lớn nhất của $B$ bằng $0,5$ khi $x=4,5$.}
\end{vd}

\begin{vd}
	Tìm giá trị lớn nhất của biểu thức $C=|x+5|-|x-2|$.
	\loigiai{
