\begin{ex}%[HK2, THPT Nguyễn Trãi - Hà Nội, 2018]%[Dương BùiĐức, 12EX9]%[2D1K5-4]

	\choice
	{$ m\in (0;1) $}
	{\True $ m\in (-1;0) $}

	{$ m\in \left(\dfrac{3}{2};2 \right)  $}
	\loigiai{
Xét phương trình hoành độ giao điểm
\[ (3m-1)x+6m+3=x^{3}-3x^{2}+1\Leftrightarrow x^{3}-3x^{2}-(3m-1)x-6m-2=0.\tag{1} \]
Gọi $ x_{1} $, $ x_{2} $, $ x_{3} $ là nghiệm của phương trình ($ 1 $), trong đó $ x_{1}<x_{2}<x_{3} $. Để ba giao điểm trong đó có một điểm cách đều hai điểm kia thì $ 2x_{2}=x_{1}+x_{3} $.\\
Vì $ x_{1} $, $ x_{2} $, $ x_{3} $ là các nghiệm của phương trình ($ 1 $) nên 
\[ 
(x-x_{1})(x-x_{2})(x-x_{3})=0\Leftrightarrow x^{3}-(x_{1}+x_{2}+x_{3})x^{2}+(x_{1}x_{2}+x_{2}x_{3}+x_{3}x_{1})x-x_{1}x_{2}x_{3}=0.\tag{2}
\]
Từ ($ 1 $) và ($ 2 $), ta có $ \heva{& x_{1} + x_{2} + x_{3}=3\\ & x_{1}x_{2} + x_{2}x_{3} + x_{3}x_{1}=-3m+1\\ & x_{1}x_{2}x_{3}=6m+2. } $\\ 
Vì $ 2x_{2}=x_{1}+x_{3} $ nên $ 3x_{2}=3\Rightarrow x_{2}=1 $.\\
Thay $ x=1 $ vào phương trình ($ 1 $) ta được $ -9m-3=0\Leftrightarrow m=-\dfrac{1}{3} $.\\
Với $ m=-\dfrac{1}{3} $, ta có
\[ 
(1)\Leftrightarrow x^{3}-3x^{2}+2x=0\Leftrightarrow\hoac{&x=0\\ &x=1\\ &x=2}\ (\mbox{thỏa mãn}).
\]
Vậy $ m=-\dfrac{1}{3}\in(-1;0) $.
}
\end{ex}
