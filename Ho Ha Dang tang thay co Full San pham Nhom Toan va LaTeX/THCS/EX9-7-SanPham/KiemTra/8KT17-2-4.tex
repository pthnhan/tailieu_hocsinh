\noindent\textbf{I. PHẦN TRẮC NGHIỆM}
\begin{ex}%[8D1B7]
Kết quả rút gọn của $\dfrac{-4x^{2}-4x}{1-x^{2}}$ là
\choice
{$\dfrac{-4x}{1+x}$}
{$\dfrac{-4x}{x-1}$}
{\True $\dfrac{4x}{x-1}$}
{$\dfrac{4x}{x+1}$}
\loigiai{
$\dfrac{-4x^{2}-4x}{1-x^{2}}=\dfrac{-4x(1+x)}{(1-x)(1+x)}=\dfrac{4x}{x+1}$.
}
\end{ex}
\begin{ex}%[8D1B12]
Thực hiện phép chia $x^{3}+27$ cho $3x-x^{2}-9$ ta được thương là
\choice
{\True $-x-3$}
{$x+3$}
{$x-3$}
{$-x+3$}
\loigiai{
$x^{3}+27=(x+3)(x^{2}-3x+9)\Rightarrow (x^{3}+27):(3x-x^{2}-9)=-x-3$.
}
\end{ex}

\begin{ex}%[8D1B7]%[8D1B7]%[8D1B8]
Phân tích các đa thức sau thành nhân tử:
\begin{multicols}{2}
\columnsep=2pt
\columnseprule=0pt
\begin{enumerate}
\item $A=x^{2}-9x-9y-y^{2}$.
\item $B=2xy-x^{2}-y^{2}+16$.
\item $C=x^{3}-x^{2}-6x$.
\end{enumerate}
\end{multicols}
\loigiai{
\hfill
\begin{enumerate}
\item $A=(x-y)(x+y)-9(x+y)=(x+y)(x-y-9)$.
\item $B=16-(x-y)^{2}=(4+x-y)(4-x+y)$.
\item $C=x(x^{2}-x-6)=x(x-3)(x+2)$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[8D2B9]%[8D2B9]%[8D2K9]
Cho biểu thức $A=\left(\dfrac{3x^{2}}{x^{2}-4}-\dfrac{3}{x+2}+\dfrac{3}{2-x}\right):\dfrac{x+3}{x+2}$ (với điều kiện $x\neq \pm 2$; $x\neq -3$).
\begin{enumerate}
\item Rút gọn biểu thức $A$.
\item Tính giá trị biểu thức $A$ khi $|x-2|=4$.
\item Tìm giá trị nguyên của $x$ để biểu thức $A$ có giá trị là một số nguyên.
\end{enumerate}
\loigiai{
\hfill
\begin{enumerate}
\item $A=\left(\dfrac{3x^{2}}{(x-2)(x+2)}-\dfrac{3(x-2)+3(x+2)}{(x-2)(x+2)}\right):\dfrac{x+3}{x+2}=\dfrac{3x^{2}-6x}{(x-2)(x+2)}:\dfrac{x+3}{x+2}$\\
\hspace*{0.4cm}$=\dfrac{3x}{x+2}:\dfrac{x+3}{x+2}=\dfrac{3x}{x+3}$.
\item $|x-2|=4\Leftrightarrow \hoac{& x-2=4\\ & x-2=-4}\Leftrightarrow \hoac{& x=6\ (\mbox{TM})\\ & x=2\ (\mbox{Loại})}$.\\
Với $x=6\Rightarrow A=2$.
\item Ta có $A=3-\dfrac{9}{x+3}$. Để $A\in \mathbb Z\Rightarrow x+3\in\mbox{Ư($9$)}=\{\pm 1;\pm 3;\pm 9\}$.\\
Lập bảng giá trị:
\begin{center}
\begin{tikzpicture}
\tkzTabInit[lgt=2,espcl=1.5]
{$x+3$ /.7,$x$ /.7}{$-9$,$-3$,$-1$,$1$,$3$,$9$}
\tkzTabLine{-$12$,,-$6$,,-$4$,,-$2$,,$0$,,$6$}
\end{tikzpicture}
\end{center}
Vậy $x\in \{-12;-6;-4;-2;0;6\}$
\end{enumerate}
}
\end{ex}
\begin{ex}%[8H1B7]%[8H1B9]%[8H1K7]%[8H1K7]
Cho hình chữ nhật $ABCD$. Kẻ $AN$ và $CM$ cùng vuông góc với $BD$ ($M$, $N\in BD$).
\begin{enumerate}
\item Chứng minh tứ giác $ANCM$ là hình bình hành.
\item Gọi $K$ là điểm đối xứng với điểm $A$ qua $N$. Chứng minh $MNKC$ là hình chữ nhật.
\item Tứ giác $DKCB$ là hình gì? Tại sao?
\item Tia $AM$ cắt tia $KC$ tại $P$. Chứng minh các đường thẳng $PN$, $AC$, $KM$ đồng quy.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1]
\tkzInit[xmax=12,ymax=8]
\tkzDefPoint(0,0){A}
\tkzDefShiftPoint[A](0:7){B}
\tkzDefShiftPoint[A](90:3.5){D}
\tkzDefShiftPoint[B](90:3.5){C}
\tkzDefPointBy[projection = onto D--B](A)\tkzGetPoint{N}
\tkzDefPointBy[projection = onto D--B](C)\tkzGetPoint{M}
\tkzDefPointBy[symmetry=center N](A)\tkzGetPoint{K}
\tkzInterLL(A,M)(C,K)\tkzGetPoint{P}
\tkzDrawSegments(A,B B,C C,D D,A A,K A,P K,P N,C N,P M,C M,K A,C B,D D,K)
\pgfresetboundingbox
\tkzLabelPoints[right](B,C,K,P)
\tkzLabelPoints[left](A,D,N)
\tkzLabelPoints[below](M)
\tkzMarkRightAngle(A,N,B)
\tkzMarkRightAngle(C,M,B)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Vì $ABCD$ là hình chữ nhật nên $\widehat{ADN}=\widehat{CBM}$\\
$\Rightarrow \widehat{DAN}=90^{\circ}-\widehat{ADN}=90^{\circ}-\widehat{CBM}=\widehat{BCM}$.\\
Xét hai tam giác $\Delta ADN$ và $\Delta CBM$ có $\widehat{ADN}=\widehat{CBM}$, $\widehat{DAN}=\widehat{BCM}$, $AD=BC$\\
$\Rightarrow \Delta ADN=\Delta CBM$ (g.c.g) $\Rightarrow AN=CM$.\\
Vì $AN\perp BD$, $CM\perp BD\Rightarrow AN\parallel CM$. Mà $AN=CM$ (cmt) nên $ANCM$ là hình bình hành.
\item Vì $A$, $K$ đối xứng nhau qua $N$ nên $AN=NK=CM$ và $KN\parallel CM$. Do đó $MNKC$ là hình bình hành. Mà $AN\perp BD\Rightarrow KN\perp BD\Rightarrow MNKC$ là hình chữ nhật.
\item Vì $A$, $K$ đối xứng nhau qua $N$ nên $AD=KD=BC$. Do $AD\parallel BC$ và $KD$ cắt $AD$ nên $KD$ cắt $BC$.\\
Lại có $MNKC$ là hình chữ nhật nên $KC\parallel BD\Rightarrow DKCB$ là hình thang. Mà $DK=BC$ (cmt) nên $DKCB$ là hình thang cân.
\item Vì $CM=KN=NA\Rightarrow CM=\dfrac{1}{2}AK$. Mà $CM\parallel AK\Rightarrow CM$ là đường trung bình trong tam giác $AKP\Rightarrow C$, $M$ lần lượt là các trung điểm của $KP$, $AP$.\\
Xét tam giác $AKP$ có $AC$, $KM$, $PN$ là các đường trung tuyến nên $AC$, $KM$, $PN$ đồng quy tại trọng tâm của tam giác $AKP$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[8D1G5]
Cho $x$ và $y$ là hai số thực khác $0$ thỏa mãn $2x^{2}+\dfrac{y^{2}}{4}+\dfrac{1}{x^{2}}=4$. Tìm giá trị lớn nhất và giá trị nhỏ nhất của $A=xy+2016$.
\loigiai{
\hfill
\begin{enumerate}
\item $4=\left(x-\dfrac{1}{x}\right)^{2}+\left(x+\dfrac{y}{2}\right)^{2}-xy+2\Rightarrow xy=\left(x-\dfrac{1}{x}\right)^{2}+\left(x+\dfrac{y}{2}\right)^{2}-2\geq -2$\\
$\Rightarrow A=xy+2016\geq 2014$ khi $x=1$, $y=-2$ hoặc $x=-1$, $y=2$.
\item $4=\left(x-\dfrac{1}{x}\right)^{2}+\left(x-\dfrac{y}{2}\right)^{2}+xy+2\Rightarrow xy=-\left(x-\dfrac{1}{x}\right)^{2}-\left(x+\dfrac{y}{2}\right)^{2}+2\leq 2$\\
$\Rightarrow A=xy+2016\leq 2018$ khi $x=1$, $y=2$ hoặc $x=-1$, $y=-2$.
\end{enumerate}
Vậy $A_{\min}=2014$ khi $\hoac{& x=1;y=-2\\ & x=-1; y=2}$ và $A_{\max}=2018$ khi $\hoac{& x=1; y=2\\ & x=-1; y=-2}$.
}
\end{ex}
