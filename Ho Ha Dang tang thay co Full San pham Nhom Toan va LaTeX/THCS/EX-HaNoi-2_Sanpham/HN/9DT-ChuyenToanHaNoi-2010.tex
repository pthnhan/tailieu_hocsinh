\begin{ex}{\it{(2,0 điểm)}}%[9D5G2]%[9D5G2]
\begin{enumerate}
		\item Cho $n$ là số nguyên, chứng minh $A = n^3 + 11n$ chia hết cho $6$.
		\item Tìm tất cả các số tự nhiên $n$ để $B = n^4 - 3n^2 + 1$ là số nguyên tố.
		\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $A=(n-1)n(n+1)+12n\ \vdots\ 6$.
\item Ta có $B=(n^{2}-n-1)(n^{2}+n-1)$. Vì $n^{2}-n-1<n^{2}+n-1$ nên để $B$ là số nguyên tố thì $n^{2}-n-1=1\Rightarrow \hoac{&n=-1\ (\mbox{Loại})\\ &n=2\ (\mbox{Thỏa mãn})}\Rightarrow n=2$.
\end{enumerate}
}

\begin{ex}{\it{(2,0 điểm)}}%[9D4K6]%[9D4G6]
Cho phương trình $\big(m^2 + 2m + 2\big)x^2 - \big(m^2 - 2m + 2\big)x - 1 = 0$. Gọi $x_1$, $x_2$ là hai nghiệm của phương trình đã cho. 
	\begin{enumerate}
		\item Tìm tất cả các giá trị của $m$ để $x^2_1 + x^2_2 = 2x_1x_2\big(2x_1x_2 - 1\big)$.
		\item Tìm giá trị lớn nhất và giá trị nhỏ nhất của biểu thức $S = x_1 + x_2$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
\item Theo Viet ta có $\heva{&x_{1}+x_{2}=\dfrac{m^{2}-2m+2}{m^{2}+2m+2}\\ &x_{1}x_{2}=\dfrac{-1}{m^{2}+2m+2}}$. Theo giả thiết\\
$(x_{1}+x_{2})^{2}=4x_{1}^{2}x_{2}^{2}\Rightarrow (m^{2}-2m+2)^{2}=4\Rightarrow \hoac{&m^{2}-2m+2=2\\ &m^{2}-2m+2=-2}$\\
$\Rightarrow \hoac{&m^{2}-2m=0\\ &m^{2}-2m+4=0\ (\mbox{Vô nghiệm})}\Rightarrow\hoac{&m=0\\ &m=2}$.
\begin{itemize}
\item Với $m=0$, thay vào phương trình ta thấy thỏa mãn.
\item Với $m=2$, thay vào phương trình ta thấy thỏa mãn.
\end{itemize}
Vậy các giá trị của $m$ cần tìm là $m=0$, $m=2$.
\item Ta có $S=x_{1}+x_{2}=\dfrac{m^{2}-2m+2}{m^{2}+2m+2}$.
\begin{itemize}
\item $S-(3-2\sqrt{2})=\dfrac{(2\sqrt{2}-2)m^{2}-4(2-\sqrt{2})m-4(1-\sqrt{2})}{m^{2}+2m+2}$\\
$=\dfrac{(\sqrt{2}-1)(\sqrt{2}m-2)^{2}}{m^{2}+2m+2}\geq 0\Rightarrow S\geq 3-\sqrt{2}$. Dấu bằng xảy ra khi $m=\sqrt{2}$.
\item $S-(3+2\sqrt{2})=\dfrac{(-2\sqrt{2}-2)m^{2}-4(2+\sqrt{2})m-4(2+\sqrt{2})}{m^{2}+2m+2}$\\
$=\dfrac{-(\sqrt{2}+1)(\sqrt{2}m+2)^{2}}{m^{2}+2m+2}\leq 0\Rightarrow S\leq 3+\sqrt{2}$. Dấu bằng xảy ra khi $m=-\sqrt{2}$.
\end{itemize}
Vậy $S_{\max}=3+\sqrt{2}$ khi $m=-\sqrt{2}$ và $S_{\min}=3-\sqrt{2}$ khi $m=\sqrt{2}$.
\end{enumerate}
}
\end{ex}
\begin{ex}{\it{(2,0 điểm)}}%[9D5G1]%[9D5G3]
	\begin{enumerate}
		\item Cho $a$ bất kỳ, chứng minh rằng $\dfrac{a^{2010} + 2010}{\sqrt{a^{2010} + 2009}} > 2$.
		\item Tìm các số nguyên $x$, $y$ thỏa mãn phương trình $y^2 - x\big(x - 2\big)\big(x^2 - 2x + 2\big) = 0$. 
	\end{enumerate}
\loigiai{
\begin{enumerate}
\item Áp dụng Cauchy cho $a^{2010}+2009$ và $1$ ta có\\
$a^{2010}+2009+1\geq 2\sqrt{a^{2010}+2009}$. Dấu bằng xảy ra khi $a^{2010}+2009=1$ (vô lí) $\Rightarrow$ đpcm.
\item Đặt $t=(x-1)^{2}
 (t\geq 0)\Rightarrow$ phương trình có dạng $y^{2}-(t-1)(t-1)=0\Rightarrow (y-t)(y+t)=-1$.\\
Vì $t\geq 0$ nên $y+t\geq y-t\Rightarrow$ nên ta có $\heva{&y-t=-1\\ &y+t=1}\Rightarrow\heva{&y=0\\ &t=1}\Rightarrow\heva{&y=0\\ &x=2}$ hoặc $\heva{&y=0\\ &x=0}$.
\end{enumerate}
}
\end{ex}
\begin{ex}{\it{(3,0 điểm)}}%[9H3K4]%[9H3K4]%[9H3G7]
Cho đường tròn $\big(O; R\big)$ và một điểm $M$ nằm ngoài đường tròn. Đường tròn đường kính $OM$ cắt đường tròn $\big(O; R\big)$ tại hai điểm $E$, $F$.
\begin{enumerate}	
\item Chứng minh giao điểm $I$ của đoạn thẳng $OM$ với đường tròn $\big(O; R\big)$ là tâm của đường tròn nội tiếp tam giác $MEF$. 
\item  Cho $A$ là một điểm bất kỳ thuộc cung $EF$ chứa điểm $M$ của đường tròn đường kính $OM$ ($A$ khác $E$ và $F$). Đoạn thẳng $OA$ cắt đoạn thẳng $EF$ tại điểm $B$. Chứng minh $OA.OB = R^2$. 
\item Cho biết $OM = 2R$ và $N$ là điểm bất kỳ thuộc cung $EF$ chứa điểm $I$ của đường tròn $\big(O; R\big)$ ($N$ khác $E$ và $F$).  Gọi $d$ là đường thẳng qua $F$ và vuông góc với đường thẳng $EN$ tại điểm $P$, $d$ cắt đường tròn đường kính $OM$ tại điểm $K$ ($K$ khác $F$). Hai đường thẳng $FN$ và $KE$ cắt nhau tại điểm $Q$. 
Chứng minh rằng $PK.PK + QN.QK \leq \dfrac{\sqrt{3}}{2}R^2$.
\end{enumerate}
