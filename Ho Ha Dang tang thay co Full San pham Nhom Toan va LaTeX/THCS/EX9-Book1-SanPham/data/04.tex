
\section{PHƯƠNG PHÁP LÙI VÔ HẠN, NGUYÊN TẮC CỰC HẠN}

\begin{vd}
    Tìm nghiệm nguyên của phương trình
    	\begin{align}
    		x^3 + 2y^3 = 4z^3 \tag{1}
    	\end{align}
    \loigiai{
        Hiển nhiên $x \vdots 2$. Đặt $x = 2x_1$ với $x_1$ là số nguyên. Thay vào $(1)$ rồi chia hai vế cho $2$ ta được
        	\begin{align}
    			4x_1^3 + y^3 = 2z^3 \tag{2}
    		\end{align}      
        Do đó $y \vdots 2$. Đặt $y = 2y_1$ với $y_1$ là số nguyên. Thay vào $(2)$ rồi chia hai vế cho $2$ ta được
        	\begin{align}
    			2x_1^3 + 4y_1^3 = z^3 \tag{3}
    		\end{align}       
        Do đó $z \vdots 2$. Đặt $z = 2z_1$ với $z_1$ là số nguyên. Thay vào $(3)$ rồi chia hai vế cho $2$ ta được
        	\begin{align}
    			x_1^3 + 2y_1^3 = 4z_1^3 \tag{4}
    		\end{align}        
        Như vậy nếu $(x; y; z)$ là nghiệm của $(1)$ thì $(x_1; y_1; z_1)$ cũng là nghiệm của $(1)$ trong đó $x = 2x_1$, $y = 2y_1$, $z = 2z_1$.\\
        Lập luận tương tự như trên, $(x_2; y_2; z_2)$ cũng là nghiệm của $(1)$ trong đó $x_1 = 2x_2$, $y_1 = 2y_2$, $z_1 = 2z_2$.\\
        Cứ tiếp tục như vậy ta đi đến $x$, $y$, $z$ đều chia hết cho $2^k$ với $k$ là số tự nhiên tùy ý. Điều này chỉ xảy ra khi $x = y = z = 0$.\\
        Đó là nghiệm duy nhất của $(1)$.\\
       \textbf{\textit{Lưu ý.}} Ta gọi phương pháp giải trên là \textit{phương pháp lùi vô hạn}.\\ 
       Nếu ví dụ 20 được cho dưới dạng: Tìm các nghiệm nguyên dương của phương trình $x^3 + 2y^3 = 4z^3 \quad (1)$, ta có thể trình bày chứng minh bằng \textit{nguyên tắc cực hạn}.\\
       Giả sử $(x_0; y_0; z_0)$ là nghiệm nguyên dương của $(1)$, trong đó $x_0$ là giá trị nguyên dương nhỏ nhất trong các giá trị mà $x$ có thể nhận.\\
       Lập luận như trong cách giải trên ta được $(x_1; y_1; z_1)$ cũng là nghiệm nguyên dương của $(1)$ mà $x_0 = 2x_1$, tức là $x_1 < x_0$. Điều này trái với giả thiết $x_0$ là số nguyên dương nhỏ nhất trong các giá trị nhận được của $x$.\\
       Vậy phương trình $(1)$ không có nghiệm nguyên dương.
    }
\end{vd}

\begin{bt}
    $x^3 - 3y^3 = 9z^3$.
    \loigiai{
    	Ta có $x^3 - 3y^3 = 9z^3 \Leftrightarrow x^3 = 3y^3 + 9z^3$ \hfill $(1)$.\\
    	Hiển nhiên $x \vdots 3$. Đặt $x = 3x_1$ với $x_1$ là số nguyên. Thay vào $(1)$ rồi chia hai vế cho $3$ ta được
        	\begin{align}
    			9x_1^3 = y^3 + 3z^3 \tag{2}
    		\end{align}      
        Do đó $y \vdots 3$. Đặt $y = 3y_1$ với $y_1$ là số nguyên. Thay vào $(2)$ rồi chia hai vế cho $3$ ta được
        	\begin{align}
    			3x_1^3 = 9y_1^3 + z^3 \tag{3}
    		\end{align}       
        Do đó $z \vdots 3$. Đặt $z = 3z_1$ với $z_1$ là số nguyên. Thay vào $(3)$ rồi chia hai vế cho $3$ ta được
        	\begin{align}
    			x_1^3 = 3y_1^3 + 9z_1^3 \tag{4}
    		\end{align}        
        Như vậy nếu $(x; y; z)$ là nghiệm của $(1)$ thì $(x_1; y_1; z_1)$ cũng là nghiệm của $(1)$ trong đó $x = 3x_1$, $y = 3y_1$, $z = 3z_1$.\\
        Lập luận tương tự như trên, $(x_2; y_2; z_2)$ cũng là nghiệm của $(1)$ trong đó $x_1 = 3x_2$, $y_1 = 3y_2$, $z_1 = 3z_2$.\\
        Cứ tiếp tục như vậy ta đi đến $x$, $y$, $z$ đều chia hết cho $3^k$ với $k$ là số tự nhiên tùy ý. Điều này chỉ xảy ra khi $x = y = z = 0$.\\
        Đó là nghiệm duy nhất của $(1)$.
    }
\end{bt}

\begin{bt}
    \begin{enumerate}
    	\item $x^2 + y^2 = 3z^2$;
    	\item $x^2 + y^2 = 6(z^2 + t^2)$.
    \end{enumerate}
    \loigiai{
    	Ta thấy $x^2$, $y^2$ chia cho $3$ dư $0$ hoặc $1$ mà $(x^2 + y^2) \vdots 3$ nên $x^2$ và $y^2$ đều chia hết cho $3$. Do đó $x$ và $y$ đều chia hết cho $3$.
    	\begin{enumerate}
    		\item Đặt $x = 3x_1$, $y = 3y_1$ với $x_1$, $y_1$ là các số nguyên. Thay vào phương trình đã cho rồi chia hai vế phương trình cho $3$, ta được
    					\begin{align}
    						3x_1^2 + 3y_1^2 = z^2 \tag{1.2}
    					\end{align}
    				 Do đó $z \vdots 3$. Đặt $z = 3z_1$ với $z_1$ là số nguyên. Thay vào $(1.2)$ rồi chia hai vế cho $3$ ta được
        				\begin{align}
    						x_1^2 + y_1^2 = 3z_1^2 \tag{1.3}
    					\end{align}        
        			Như vậy nếu $(x; y; z)$ là nghiệm của phương trình đã cho thì $(x_1; y_1; z_1)$ cũng là nghiệm của phương trình đã cho trong đó $x = 3x_1$, $y = 3y_1$, $z = 3z_1$.\\
        			Lập luận tương tự như trên, $(x_2; y_2; z_2)$ cũng là nghiệm của phương trình đã cho trong đó $x_1 = 3x_2$, $y_1 = 3y_2$, $z_1 = 3z_2$.\\
        			Cứ tiếp tục như vậy ta đi đến $x$, $y$, $z$ đều chia hết cho $3^k$ với $k$ là số tự nhiên tùy ý. Điều này chỉ xảy ra khi $x = y = z = 0$.\\
        			Đó là nghiệm duy nhất của phương trình đã cho.
        	\item Đặt $x = 3x_1$, $y = 3y_1$ với $x_1$, $y_1$ là các số nguyên. Thay vào phương trình đã cho rồi chia hai vế phương trình cho $3$, ta được
    					\begin{align}
    						3x_1^2 + 3y_1^2 = 2(z^2 + t^2) \tag{2.2}
    					\end{align}
    				 Do đó $(z^2 + t^2) \vdots 3$. Suy ra $z$ và $t$ đều chia hết cho $3$. Đặt $z = 3z_1$, $t = 3t_1$ với $z_1$, $t_1$ là các số nguyên. Thay vào $(2.2)$ rồi chia hai vế phương trình cho $3$, ta được
    				 	\begin{align}
    						x_1^2 + y_1^2 = 6(z_1^2 + t_1^2) \tag{2.3}
    					\end{align}
    				Như vậy nếu $(x; y; z; t)$ là nghiệm của phương trình đã cho thì $(x_1; y_1; z_1; t_1)$ cũng là nghiệm của phương trình đã cho trong đó $x = 3x_1$, $y = 3y_1$, $z = 3z_1$, $t = 3t_1$.\\
        			Lập luận tương tự như trên, $(x_2; y_2; z_2; t_2)$ cũng là nghiệm của phương trình đã cho trong đó $x_1 = 3x_2$, $y_1 = 3y_2$, $z_1 = 3z_2$, $t_1 = 3t_2$.\\
        			Cứ tiếp tục như vậy ta đi đến $x$, $y$, $z$, $t$ đều chia hết cho $3^k$ với $k$ là số tự nhiên tùy ý. Điều này chỉ xảy ra khi $x = y = z = = t = 0$.\\
        			Đó là nghiệm duy nhất của phương trình đã cho.
    	\end{enumerate}
    }
\end{bt}

\begin{bt}
    $x^2 + y^2 + z^2 = 2xyz$.
    \loigiai{
    	Ta thấy $(x^2 + y^2 + z^2) \vdots 2$. Xảy ra hai trường hợp:
    	\begin{itemize}
    		\item Trong ba số $x$, $y$, $z$ có một số chẵn, hai số lẻ, chẳng hạn $x$ chẵn, $y$ và $z$ lẻ. Khi đó vế trái $x^2 + y^2 + z^2$  chia cho $4$ dư $2$, còn vế phải $2xyz$ chia hết cho $4$, loại.
    		\item Ba số $x$, $y$, $z$ đều chẵn. Đặt $x = 2x_1$, $y = 2y_1$, $z = 2z_1$ với $x_1$, $y_1$, $z_1$ là các số nguyên. Thay vào phương trình đã cho rồi chia hai vế phương trình cho $4$, ta được
    					\begin{align}
    						x_1^2 + y_1^2 + z_1^2 = 4x_1y_1z_1 \tag{1}
    					\end{align}   				
    	\end{itemize}
    Lập luận như trên, ta có ba số $x_1$, $x_2$, $x_3$ cũng là các số chẵn. 
    Cứ tiếp tục như vậy ta đi đến $x$, $y$, $z$ đều chia hết cho $2^k$ với $k$ là số tự nhiên tùy ý. Điều này chỉ xảy ra khi $x = y = z = 0$.\\
    Đó là nghiệm duy nhất của phương trình đã cho.
    }
\end{bt}

\begin{bt}
    \begin{enumerate}
    	\item Tìm nghiệm nguyên của phương trình $x^2 + y^2 = 7z^2$.
    	\item Chứng minh rằng số $7$ không viết được dưới dạng tổng các bình phương của hai số hữu tỉ.
    	\item Tìm nghiệm nguyên của phương trình $x^2 + y^2 = az^2$, trong đó $a$ là số tự nhiên có dạng $4k - 1$ $(k \in \mathbb{N})$.
    \end{enumerate}
    \loigiai{
    \begin{enumerate}
    	\item $x^2 + y^2 = 7z^2$ \hfill $(1)$.\\
    			Một số chính phương chia cho $7$ thì dư $0$, $1$, $2$, $4$. Theo $(1)$ thì $x^2 + y^2$ chi hết cho $7$ nên $x$ và $y$ đều chia hết cho $7$. Từ đó $z$ chia hết cho $7$.\\
    			Đặt $x = 7x_1$, $y = 7y_1$, $z = 7z_1$. Thay vào $(1)$ và rút gọn, ta được $x_1^2 + y_1^2 = 7z_1^2$.\\
    			Như vậy $(x_1; y_1; z_1)$ cũng là nghiệm của $(1)$.\\
    			Đáp số: Nghiệm $(x; y; z)$ là $(0; 0; 0)$.
    	\item Giả sử $7 = \left(\dfrac{x}{z}\right)^2 + \left(\dfrac{y}{z}\right)^2$ với $x$, $y$, $z$ là các số nguyên  và $z \neq 0$.\\
    			Suy ra $7z^2 = x^2 + y^2$.\\
    			Phương trình trên chỉ có nghiệm là $(0; 0; 0)$ (câu a), trái với điều kiện $z \neq 0$.\\
    			Vậy số $7$ không viết được dưới dạng tổng các bình phương của hai số hữu tỉ.
    	\item $x^2 + y^2 = (4k - 1)z^2 \Leftrightarrow x^2 + y^2 + z^2 = 4kz^2$ \hfill $(2)$\\
    			Do $x^2 + y^2 + z^2$ là số chẵn nên xảy ra hai trường hợp:
    			\begin{itemize}
    				\item Trong ba số $x$, $y$, $z$ có một số chẵn, hai số lẻ thì vế trái của $(2)$ chia cho $4$ dư $2$, còn vế phải chia hết cho $4$, loại.
    				\item Cả ba số $x$, $y$, $z$ đều chẵn. Đặt $x = 2x_1$, $y = 2y_1$, $z = 2z_1$ với $x_1$, $y_1$, $z_1$ là các số nguyên. Thay vào $(2)$ được
    						$$4x_1^2 + 4y_1^2 + 4z_1^2 = 16kz_1^2 \Leftrightarrow x_1^2 + y_1^2 + z_1^2 = 4kz_1^2.$$
    						Như vậy $(x_1; y_1; z_1)$ cũng là nghiệm của phương trình $(2)$.    						
    			\end{itemize}
    			Đáp số: Nghiệm $(x; y; z)$ là $(0; 0; 0)$.
    \end{enumerate}
    }
\end{bt}
