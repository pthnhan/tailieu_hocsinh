	\choice
	{\True $1$}
	{$\dfrac{1}{2}$}
	{$-1$}
	{$-\dfrac{1}{2}$}
	\loigiai{
	$ \lim\limits_{x \to +\infty}\dfrac{x^2-1}{x^2+3x+2}  =\lim\limits_{x \to +\infty} \dfrac{1-\dfrac{1}{x^2}}{1+\dfrac{3}{x}+\dfrac{2}{x^2}}=1.$

\begin{ex}%[HK2 (2016-2017), THPT THỚI LAI, CẦN THƠ]%[Lê Thanh Nin]%[1D4B2-3]
	Tính $ \lim\limits_{x \to 3}\dfrac{\sqrt{x+1}-2}{9-x^2} $ bằng
	\choice
	{\True $-\dfrac{1}{24}$}
	{$\dfrac{1}{24}$}
	{$\dfrac{1}{6}$}
	{$-\dfrac{1}{6}$}
	\loigiai{
	$ \lim\limits_{x \to 3}\dfrac{\sqrt{x+1}-2}{9-x^2}
	=  \lim\limits_{x \to 3} \dfrac{\left(\sqrt{x+1}-2\right)\left(\sqrt{x+1}+2\right)}{(3-x)(3+x)\left(\sqrt{x+1}+2\right)} \\
	=\lim\limits_{x \to 3} \dfrac{x-3}{(3-x)(3+x)\left(\sqrt{x+1}+2\right)}
	=\lim\limits_{x \to 3} \dfrac{-1}{(3+x)\left(\sqrt{x+1}+2\right)}
	=-\dfrac{1}{24}$.	
	}
\end{ex}

	Một vật được thả rơi tự do ở độ cao $ 147 $ m có phương trình chuyển động $ S(t)=\dfrac{1}{2}gt^2 $, trong đó $ g=9,8 $ m/s$ ^2 $ và $ t $ tính bằng giây(s). Tính vận tốc của vật tại thời điểm vật tiếp đất.

	Vậy vận tốc của vật tại thời điểm chạm đất là : $ v(\sqrt{30})= 9,8\cdot \sqrt{30} =\dfrac{49\sqrt{30}}{5}$ (m/s).

\begin{bt}%[HK2 (2016-2017), THPT THỚI LAI, CẦN THƠ]%[Lê Thanh Nin]%[1D5K2-3]
	Viết phương trình tiếp tuyến của đồ thị hàm số $ (C): y=\dfrac{2x-5}{x+2} $, biết tiếp tuyến song song với đường thẳng $ d: y=x-2017 $.
	\loigiai{
	Gọi $ (x_0;y_0) $ là tọa độ tiếp điểm.\\
	Vì $ d:y=x-2017 $ có hệ số góc là $ k=1 $\\
	Suy ra hệ số góc của tiếp tuyến là :
	 $ y'(x_0)=1\Leftrightarrow \dfrac{9}{(x_0+2)^2}=1
	 \Leftrightarrow x_0^2+4x_0-5=0\Leftrightarrow \hoac{&x_0=-5\\&x_0=1}$. \\
	 Với $ x_0=1\Rightarrow y_0=-1 $ khi đó phương trình tiếp tuyến là: $ y=x-2 $.\\
	 Với $ x_0=-5\Rightarrow y_0=5 $ khi đó phương trình tiếp tuyến là: $ y=x+10 $.
	}

	\loigiai{

	}

\begin{bt}%[HK2 (2016-2017), THPT THỚI LAI, CẦN THƠ]%[Lê Thanh Nin]%[1D5B3-1]
	Tính đạo hàm của hàm của các hàm số sau : 
	\begin{enumerate}
		\item $ y=\dfrac{\sin x}{\sin x-\cos x} $
		\item $ y=\cos^2\left(2x-\dfrac{\pi}{3}\right) $
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
		\item $ y'=\dfrac{(\sin x)'(\sin x-\cos x)-\sin x(\sin x-\cos x)'}{(\sin x-\cos x)^2}
		=\dfrac{\cos x(\sin x-\cos x)-\sin x(\cos x+\sin x)}{(\sin x-\cos x)^2}\\
		=-\dfrac{1}{(\sin x-\cos x)^2} $
		\item $ y'=2\cos\left(2x-\dfrac{\pi}{3}\right)\left(\cos\left(2x-\dfrac{\pi}{3}\right)\right)'
		=-4\cos \left(2x-\dfrac{\pi}{3}\right)\sin\left(2x-\dfrac{\pi}{3}\right)
		=-2\sin\left(4x-\dfrac{2\pi}{3}\right)$
	\end{enumerate}	
	}
\end{bt}


\begin{bt}%[HK2 (2016-2017), THPT THỚI LAI, CẦN THƠ]%[Lê Thanh Nin]%[1H3B3-2]%[1H3K3-3]%[1H3K5-2]
	Cho hình chóp $ S.ABCD $ có đáy $ ABCD $ là hình vuông cạnh $ a, SA\perp (ABCD) $ và $ SA=a\sqrt{10} $. Gọi $ M,N $ lần lượt là trung điểm của $ BC $ và $ CD $.
	\begin{enumerate}
		\item Chứng minh : $ BD\perp (SAC) $.
		\item Tính góc giữa $ SM $ và $ (ABCD) $.
		\item Tính khoảng cách từ điểm $ C $ đến mặt phẳng $ (SMN) $.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[line cap=round,line join=round, >=stealth,scale=1]
			\def \xa{-1.3} 
			\def \xb{-1.5}
			\def \y{4}

			\coordinate (D) at ($(A)+(\y,0)$);

			\coordinate (S) at ($ (A)+(0,\z) $);
			\tkzDefMidPoint(B,C)    \tkzGetPoint{M}
			\tkzDefMidPoint(D,C)    \tkzGetPoint{N}
			\tkzInterLL(A,C)(B,D) \tkzGetPoint{O}
			\tkzInterLL(A,C)(M,N) \tkzGetPoint{I}

			\tkzLabelPoints[above](S,I,O)

	\begin{enumerate}
		\item Ta có $ \heva{&BD\perp AC\\&BD\perp SA}\Rightarrow BD\perp (SAC) $.
		\item Hình chiếu vuông góc của $ SM $ lên mặt phẳng $ (ABCD) $ là $ AM $.\\
		Nên $ \widehat{(SM,(ABCD))}=\widehat{(SM,AM)}=\widehat{SMA} $.\\
		Xét $ \triangle  SAM $ vuông tại $ A $, ta có:\\
		$ \tan \widehat{SMA}=\dfrac{SA}{AM}=\dfrac{a\sqrt{10}}{\dfrac{a\sqrt{5}}{2}}=2\sqrt{2}\Rightarrow \widehat{SMA}\approx 70^\circ31'. $
		\item Gọi $ O=AC\cap BD; I=AC\cap MN\Rightarrow CI=OI=\dfrac{1}{3}AI. $\\

		Vậy $ d(C,(SMN))=d(O,(SMN))=\dfrac{1}{3}d(A,(SMN))=\dfrac{AH}{3}=\dfrac{a\sqrt{10}}{\sqrt{89}} $.
