\begin{bt}%[Đề KT học kì 1,2017-2018 THPT Nguyễn Hữu Cầu, Hồ Chí Minh]%[1D2K1-3]
	Từ các số chữ số $ 0,1,2,3,4,5,6,7 $ có thể lập được bao nhiêu số tự nhiên chẵn gồm ba chữ số khác
	nhau?
	\loigiai{
	Gọi số cần tìm có dạng $ \overline{abc} .$
	\begin{itemize}
		\item TH 1$\colon c=0 \Rightarrow c $ có 1 cách chọn. Chọn $ a,b $ có $ \mathrm{A}_7^2 $ cách.\\
		Vậy có $ \mathrm{A}_7^2 = 42 $ số.
		\item TH 2$\colon c \in \left\lbrace 2;4;6 \right\rbrace  \Rightarrow c $ có 3 cách chọn.\\
		Chọn $ a $ có 6 cách, chọn $ b $ có 6 cách.\\
		Vậy có $ 3 \cdot 6 \cdot 6 = 108 $ số.
		\item Kết luận$\colon $ có $ 42+108=150 $ số cần tìm.
	\end{itemize}
	}
\end{bt}
\begin{bt}%[Đề KT học kì 1,2017-2018 THPT Nguyễn Hữu Cầu, Hồ Chí Minh]%[1D2B3-2]
	Tìm số hạng không chứa $x$ trong khai triển của  $ \left( 2x^3-\dfrac{1}{2x^2}\right)^{10}.$
	\loigiai{
	\begin{itemize}
		\item Số hạng tổng quát $ \mathrm{C}_{10}^k \cdot (2x^3)^{10-k} \cdot \left( -\dfrac{1}{2x^2}\right)^k=\mathrm{C}_{10}^k \cdot 2^{10-k} \cdot \left( -\dfrac{1}{2}\right)^k \cdot x^{30-5k}   $.
		\item Theo đề bài ta có $ 30-5k=0 \Leftrightarrow k = 6 $.
		\item Số hạng không chứa $ x $ là $ \mathrm{C}_{10}^6 \cdot 2^4 \cdot \left( -\dfrac{1}{2}\right) ^6=\dfrac{105}{2} $.
	\end{itemize}
	}
\end{bt}
\begin{bt}%[Đề KT học kì 1,2017-2018 THPT Nguyễn Hữu Cầu, Hồ Chí Minh]%[1D2B5-2]
	Một hộp đựng $ 4 $ viên bi đỏ, $ 5 $ viên bi trắng và $ 6 $ viên bi vàng. Chọn ngẫu nhiên $ 4 $ viên bi từ hộp.	Tính xác suất để $ 4 $ viên bi được chọn không đủ cả ba màu.
	\loigiai{
		\begin{itemize}
			\item Gọi $ A $ là biến cố cần tìm.
			\item Số phần tử không gian mẫu là $ n\left( \Omega\right) = \mathrm{C}_{15}^4=1365 $.
			\item Chọn 4 viên bi có đủ cả ba màu có $ \mathrm{C}_4^2\cdot\mathrm{C}_5^1\cdot\mathrm{C}_6^1+\mathrm{C}_4^1\cdot\mathrm{C}_5^2\cdot\mathrm{C}_6^1+\mathrm{C}_4^1\cdot\mathrm{C}_5^1\cdot\mathrm{C}_6^2=720 $ cách.
			\item Số phần tử của biến cố$\colon n\left( A\right) = 1365 - 720=645 .$
			\item Xác suất của biến cố của biến cố $A\colon P(A)=\dfrac{645}{1365}=\dfrac{43}{91}  $.

		\end{itemize}
	}
\end{bt}
\begin{bt}%[Đề KT học kì 1,2017-2018 THPT Nguyễn Hữu Cầu, Hồ Chí Minh]%[1D3B4-3]
	Cho cấp số nhân $ \left( u_n\right) $ có $ \heva{& u_2+u_3+u_4= 39 \\ & u_1-u_4=-26} $. Tìm công bội $ q $ và $ u_1 $.
	\loigiai{
	\begin{eqnarray*}
	\heva{& u_2+u_3+u_4= 39 \\ & u_1-u_4=-26}\Leftrightarrow \heva{  & u_1q\left(1 + q + q^2\right)=39 \\   & u_1\left(1 - q^3\right)= - 26 } \Leftrightarrow \heva{  & \dfrac{1 - q}{q}= - \dfrac{26}{39} \\   & u_1\left(1 - q^3\right)= - 26} \Leftrightarrow \heva{  & q=3 \\   & u_1=1.} 		
	\end{eqnarray*}
    Vậy $ u_1=1 $ và $ q=3. $
	}
\end{bt}
\begin{bt}%[Đề KT học kì 1,2017-2018 THPT Nguyễn Hữu Cầu, Hồ Chí Minh]%[1D3K3-3]
	Cho cấp số cộng $ \left( u_n \right)  $ có công sai $ d=4 $, số hạng $ u_n = 37  $ và $ S_n=u_1+u_2+u_3+ \cdots +u_n=190 $. Tìm $ n $.
	\loigiai{
	\begin{eqnarray*}
	&&\heva{  & u_n=37 \\   & u_1 + u_2 +... + u_n=190} \Leftrightarrow \heva{  & u_1 + (n - 1)d=37 \\   & \dfrac{\left(u_1 + u_n\right)n}{2}=190} \Leftrightarrow \heva{  & u_1=37 - 4(n - 1) \\   & \left(37 - 4(n - 1) + 37\right)n=380.}\\
	&\Rightarrow& 2n^2 - 39n + 190=0 \Leftrightarrow \hoac{& n=10 \,\,(\text{nhận})\\ 
		& n=\dfrac{19}{2} \,\,(\text{loại})}. 	 	
	\end{eqnarray*}
	Vây $ n=10. $
	}
\end{bt}
\begin{bt}%[Đề KT học kì 1,2017-2018 THPT Nguyễn Hữu Cầu, Hồ Chí Minh]%[1D1K3-5]
	Giải phương trình $ 2\sin ^2x+\sqrt{3}\sin 2x+2\sqrt{3}\sin x+2\cos x - 2 = 0 $.
	\loigiai{
	\begin{eqnarray*}
	&& 2{\sin}^2x+\sqrt{3}\sin 2x+2\sqrt{3}\sin x+2\cos x - 2 = 0\\ 
	&\Leftrightarrow& 3{\sin}^2x + 2\sqrt{3}\sin x \cdot \cos x + {\cos}^2x + 2\sqrt{3}\sin x + 2\cos x - 3=0 \\
	&\Leftrightarrow& \left(\sqrt{3}\sin x + \cos x\right)^2 + 2\left(\sqrt{3}\sin x + \cos x\right) - 3=0 \\
	&\Leftrightarrow& \hoac{& \sqrt{3}\sin x + \cos x=1 \\ 
		& \sqrt{3}\sin x + \cos x= - 3 
		 \,(\text{ptvn})}
	\Leftrightarrow \sin \left(x + \dfrac{\pi}{6}\right)=\dfrac{1}{2}
	\Leftrightarrow \hoac{
		& x=k2\pi \\ 
		& x=\dfrac{2\pi}{3} + k2\pi.} \left( k \in \mathbb{Z} \right)
	\end{eqnarray*}
	Tập nghiệm $ S=\left\lbrace k2\pi;\dfrac{2\pi}{3}+k2\pi\right\rbrace.$
	}
\end{bt}
