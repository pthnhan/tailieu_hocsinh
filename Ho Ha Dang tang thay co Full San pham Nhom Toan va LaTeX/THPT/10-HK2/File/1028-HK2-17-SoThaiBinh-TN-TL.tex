\begin{ex}%[HK2, Sở Thái Bình, 2016-2017]%[Nguyễn Tiến Thùy]%[0D4G1]
Cho các số thực $x, y$ thỏa mãn $x+y=2\left(\sqrt{x-3}+\sqrt{y+3}\right)$. Tìm giá trị nhỏ nhất của biểu thức $P=4(x^2+y^2)+15xy$.
\choice
{\True $P_{\min}=-83$}
{$P_{\min}=-80$}
{$P_{\min}=-91$}
{$P_{\min}=-63$}
\loigiai{
Điều kiện $x\ge 3, y\ge -3$, từ đó ta có
$$(x+3)(y+3)\ge 0\Leftrightarrow xy\ge -3(x+y)-9.$$
Xét trường hợp $x=-y=3$ suy ra $P=-63$.\\
Trường hợp $x+y>0$, từ giả thiết suy ra $$(x+y)^2=4(x+y)+8\sqrt{(x-3)(y+3)}\ge 4(x+y)$$
$$\Leftrightarrow x+y\ge 4\, (\mbox{do } x+y>0).$$
Đánh giá $P$ ta được
$$P=4(x^2+y^2)+15xy=4(x+y)^2+7xy\ge 4(x+y)^2-21(x+y)-63.$$
Do hàm số $f(t)=4t^2-21t-63$ đồng biến trên khoảng $\left(\dfrac{21}{8};+\infty\right)$ suy ra 
$$P=f(x+y)\ge f(4)=-83.$$
Vậy $P_{\min}=-83$ khi $x=7, y=-3$.
}	
\end{ex}

\begin{bt}%[HK2, Sở Thái Bình, 2016-2017]%[Nguyễn Tiến Thùy]%[0D3B2]
Giải phương trình sau $\sqrt{3x^2-9x+1}+2=x$.

$$\Leftrightarrow \heva{&x-2\ge 0\\ &3x^2-9x+1=x^2-4x+4}\Leftrightarrow \heva{&x\ge 2\\ &2x^2-5x-3=0}\Leftrightarrow x=3.$$

}
\end{bt}

\begin{bt}%[HK2, Sở Thái Bình, 2016-2017]%[Nguyễn Tiến Thùy]%[0D4K5]
Giải bất phương trình sau $(x-3)\sqrt{x^2+4}\le x^2-9$.
\loigiai{
- Trường hợp $x=3$ thỏa mãn bất phương trình.\\
- Trường hợp $x>3$ \\
Ta có bất phương trình tương đương với 
$$\sqrt{x^2+4}\le x+3\Leftrightarrow x^2+4\le x^2+6x+9\Leftrightarrow x\ge -\dfrac{5}{6}.$$
Đối chiếu điều kiện ta được $x\ge 3$.\hfill $(1)$\\
- Trường hợp $x<3$ \\
Bất phương trình tương đương với
$\sqrt{x^2+4}\ge x+3$. \hfill $(2)$\\
Nhận thấy nếu $x\le -3$ thì bất phương trình $(2)$ luôn đúng. \hfill $(3)$\\
Xét trường hợp $-3<x<3$ ta có $(2)$ tương đương với $x^2+4\ge x^2+6x+9\Leftrightarrow x\le -\dfrac{5}{6}$.\hfill $(4)$\\
Từ $(1)(3)(4)$ ta thu được $x\in\left(-\infty;-\dfrac{5}{6}\right]\cup [3;+\infty)$.
}
\end{bt}
