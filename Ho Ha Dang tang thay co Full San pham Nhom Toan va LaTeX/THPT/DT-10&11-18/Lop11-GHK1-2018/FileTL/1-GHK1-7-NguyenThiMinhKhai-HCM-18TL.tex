\begin{bt}%[Thi giữa học kì I,Nguyễn Thị Minh Khai HCM, 2018]%[Nguyễn Anh Quốc, dự án EX-DT-10&11-2018]%[1D1B3-1]
	Giải các phương trình lượng giác sau
	\begin{enumerate}
		\item $4\sin x\cos x+1=0$.
		\item $\cos4x-2\cos^2x+1=0$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}	
		\item $4\sin x\cos x+1=0$.
		\begin{eqnarray*}
			&&4\sin x\cos x+1=0\\
			&\Leftrightarrow&2\sin2x+1=0\\
			&\Leftrightarrow&\sin2x=-\dfrac{1}{2}\\
			&\Leftrightarrow&\sin2x=\sin \left(-\dfrac{\pi}{6} \right)\\
			&\Leftrightarrow&\hoac{&2x=-\dfrac{\pi}{6}+k2\pi\\&2x=\pi+\dfrac{\pi}{6}+k2\pi}\\
			&\Leftrightarrow&\hoac{&x=-\dfrac{\pi}{12}+k\pi\\&x=\dfrac{7\pi}{12}+k\pi}\ \left(k\in \mathbb{Z} \right).
		\end{eqnarray*}
		Vậy nghiệm phương trình là $x=-\dfrac{\pi}{12}+k\pi$ hoặc $x=\dfrac{7\pi}{12}+k\pi$ $\left(k\in \mathbb{Z} \right)$.
		\item $\cos4x-2\cos^2x+1=0$.
		\begin{eqnarray*}
			&&\cos4x-2\cos^2x+1=0\\
			&\Leftrightarrow&\cos4x-\cos2x=0\\
			&\Leftrightarrow&\cos4x=\cos2x\\
			&\Leftrightarrow&\hoac{&4x=2x+k2\pi\\&4x=-2x+k2\pi}\\
			&\Leftrightarrow&\hoac{&2x=k2\pi\\&6x=k2\pi}\\
			&\Leftrightarrow&\hoac{&x=k\pi\\&x=\dfrac{k\pi}{3}}\ \left(k\in \mathbb{Z} \right).
		\end{eqnarray*}	
		Vậy nghiệm phương trình là $x=k\pi$ hoặc $x=\dfrac{k\pi}{3}\ \left(k\in \mathbb{Z} \right)$.
	\end{enumerate}
		
	}
\end{bt}	
\begin{bt}%[Thi giữa học kì I,Nguyễn Thị Minh Khai HCM, 2018]%[Nguyễn Anh Quốc, dự án EX-DT-10&11-2018]%[1D1B3-3]
	Giải phương trình $\sqrt{3}\cos \left( x+\dfrac{\pi}{6}\right)=2-\sin \left(x+\dfrac{\pi}{6} \right)$.	
	\loigiai{
		\begin{eqnarray*}
			&&\sqrt{3}\cos \left( x+\dfrac{\pi}{6}\right)=2-\sin \left(x+\dfrac{\pi}{6} \right)\\
			&\Leftrightarrow& \sqrt{3}\cos \left( x+\dfrac{\pi}{6}\right)+\sin \left(x+\dfrac{\pi}{6} \right)	=2\\
			&\Leftrightarrow& \dfrac{\sqrt{3}}{2}\cos \left( x+\dfrac{\pi}{6}\right)+\dfrac{1}{2}\sin \left(x+\dfrac{\pi}{6} \right)	=1\\
			&\Leftrightarrow&\sin \dfrac{\pi}{3}\cos \left( x+\dfrac{\pi}{6}\right)+\cos \dfrac{\pi}{3}\sin \left(x+\dfrac{\pi}{6} \right)	=1\\
			&\Leftrightarrow&\sin \left( \dfrac{\pi}{3}+x+\dfrac{\pi}{6} \right)=1\\
			&\Leftrightarrow& \sin \left(x+\dfrac{\pi}{2} \right)=1\\
			&\Leftrightarrow& x+\dfrac{\pi}{2}=\dfrac{\pi}{2}+k2\pi\\
			&\Leftrightarrow& x=k2\pi\ \left(k\in \mathbb{Z}\right).
		\end{eqnarray*}
		Vậy nghiệm phương trình là $x=k2\pi\ \left(k\in \mathbb{Z}\right)$.
		
	}
\end{bt}

\begin{bt}%[Thi giữa học kì I,Nguyễn Thị Minh Khai HCM, 2018]%[Nguyễn Anh Quốc, dự án EX-DT-10&11-2018]%[1D1B3-1]
	Giải phương trình $\cos4x-5\sin2x+2=0$.	
	\loigiai{
		\begin{eqnarray*}
			&&\cos4x-5\sin2x+2=0\\
			&\Leftrightarrow& 1-2\sin^22x-5\sin2x+2=0\\
			&\Leftrightarrow&-2\sin^22x-5\sin2x+3=0\\
			&\Leftrightarrow&\hoac{&\sin2x=\dfrac{1}{2}\\&\sin2x=-3\text{ (loại)}}\\
			&\Leftrightarrow& \sin2x=\dfrac{1}{2}\\
			&\Leftrightarrow&\sin2x =\sin \dfrac{\pi}{6}\\
			&\Leftrightarrow&\hoac{&2x=\dfrac{\pi}{6}+k2\pi\\&2x=\dfrac{5\pi}{6}+k2\pi}\\
			&\Leftrightarrow&\hoac{&x=\dfrac{\pi}{12}+k\pi\\&x=\dfrac{5\pi}{12}+k\pi}\ \left(k\in\mathbb{Z} \right).
		\end{eqnarray*}	
	}
\end{bt}
Vậy nghiệm phương trình là $x=\dfrac{\pi}{12}+k\pi $ hoặc $x=\dfrac{5\pi}{12}+k\pi\ \left(k\in\mathbb{Z} \right)$.


\begin{bt}%[Thi giữa học kì I,Nguyễn Thị Minh Khai HCM, 2018]%[Nguyễn Anh Quốc, dự án EX-DT-10&11-2018]%[1D1K3-2]
	Giải phương trình $	2\cos2x+4\left(\sin x+\cos x \right)^2=7+2\cos ^2x$.
	\loigiai{
		\begin{eqnarray*}
			&&2\left(2\cos^2x-1 \right)+4\left(\sin x+\cos x \right)^2=7+2\cos ^2x\\
			&\Leftrightarrow& 4\cos^2x-2+4\sin^2x+4\cos^2x+8\sin x \cos x=7+2\cos^2x\\
			&\Leftrightarrow& 4\sin^2x+8\sin x\cos x+6\cos^2x=9.\quad (1)
		\end{eqnarray*}
		\begin{itemize}
			\item Xét $\cos x=0 \Leftrightarrow x=\dfrac{\pi}{2}+k\pi$ thế vào (1) ta được $4=9$ vô lí. Nên $x=\dfrac{\pi}{2}+k\pi$, $k\in \mathbb{Z}$ không phải là nghiệm phương trình.
			\item Xét $\cos x\ne 0$, chia hai vế phương trình $(1)$ cho $\cos^2x$ ta được\\
			$4\tan^2x+8\tan x+6=\dfrac{9}{\cos^2x}=9(1+\tan^2x)$\\
			$\Leftrightarrow-5\tan^2+8\tan x-3=0\Leftrightarrow \hoac{&\tan x=1\\&\tan x=\dfrac{3}{5}}\Leftrightarrow \hoac{&x=\dfrac{\pi}{4}+k\pi\\&x=\arctan \dfrac{3}{5}+k\pi}$\  $(k\in \mathbb{Z})$.
		\end{itemize}
		Vậy nghiệm phương trình là $x=\dfrac{\pi}{4}+k\pi$ hoặc $x=\arctan \dfrac{3}{5}+k\pi $, $(k\in \mathbb{Z})$.
	}
\end{bt}
\begin{bt}%[Thi giữa học kì I,Nguyễn Thị Minh Khai HCM, 2018]%[Nguyễn Anh Quốc, dự án EX-DT-10&11-2018]%[1D1K3-6]
	$\dfrac{\cos x-\sin 2x}{2\cos ^2x-\sin x-1}=\sqrt 3$.
	\loigiai{
	Điều kiện: $2\cos^2x-\sin x-1\ne 0\Leftrightarrow -2\sin^2x-\sin x+1\ne 0\\\Leftrightarrow \heva{&\sin x\ne -1\\&\sin x\ne \dfrac{1}{2}}\Leftrightarrow \heva{&x\ne-\dfrac{\pi}{2}+k2\pi\\&x\ne \dfrac{\pi}{6}+m2\pi\\&x\ne \dfrac{5\pi}{6}+n2\pi}$, $(k,m,n\in \mathbb{Z})$.\\
		Với điều kiện trên, ta có:
		\begin{eqnarray*}
			\dfrac{\cos x-\sin 2x}{2\cos ^2x-\sin x-1}=\sqrt 3&\Leftrightarrow& \cos x-2\sin x\cos x=\sqrt 3\left(\sin x +1\right)\left(1-2\sin x\right)\\
			&\Leftrightarrow& \left(1-2\sin x\right)\left(\cos x-\sqrt 3\sin x -\sqrt 3\right)=0\\
			&\Leftrightarrow& \hoac{&\sin x=\dfrac{1}{2} \text { (không thỏa điều kiện)}\\&\cos x-\sqrt 3\sin x=\sqrt 3}\\
			&\Leftrightarrow& \dfrac{1}{2}\cos x -\dfrac{\sqrt{3}}{2}\sin x=\dfrac{\sqrt{3}}{2}\\
			&\Leftrightarrow& \cos\left(x+\dfrac{\pi}{3}\right)=\dfrac{\sqrt 3}{2}\\
			&\Leftrightarrow& \hoac{&x=-\dfrac{\pi}{6}+k2\pi\\&x=-\dfrac{\pi}{2}+k2\pi\text { (không thỏa điều kiện $\sin x\ne -1$)}} \ (k \in \mathbb{Z}).
		\end{eqnarray*}
		Vậy phương trình có nghiệm là $x=-\dfrac{\pi}{6}+k2\pi\ (k\in \mathbb{Z})$.
	}
\end{bt}

\begin{bt}%[Thi giữa học kì I,Nguyễn Thị Minh Khai HCM, 2018]%[Nguyễn Anh Quốc, dự án EX-DT-10&11-2018]%[1D1G3-5]
	$\left(2\sin x+1\right)\left(4\cos 5x+3\sin x-6\right)+2\cos 2x=1$.
	\loigiai{
		\begin{eqnarray*}
			& &\left(2\sin x+1\right)\left(4\cos 5x+3\sin x-6\right)+2\cos 2x=1\\ &\Leftrightarrow& \left(2\sin x+1\right)\left(4\cos 5x+3\sin x-6\right)+\left(1-4\sin^2x\right)=0\\
			&\Leftrightarrow& \left(2\sin x+1 \right)\left(4\cos 5x+\sin x-5 \right)=0\\
			&\Leftrightarrow& \hoac{&2\sin x+1=0\quad (1)\\&4\cos 5x+\sin x-5=0\quad (2).}
		\end{eqnarray*}
		\begin{itemize}
			\item Giải phương trình $(1)$.\\
			$(1)\Leftrightarrow \sin x=-\dfrac{1}{2}\Leftrightarrow \hoac{&x=-\dfrac{\pi}{6}+k2\pi\\&x=\dfrac{7\pi}{6}+k2\pi},\ (k\in \mathbb{Z})$.
			\item Giải phương trình $(2)$.\\
			Ta có $$4\cos 5x\le 4 \text{ và } \sin x\le 1.$$
			nên $$4\cos 5x+\sin x\le 5.$$
			Dấu "=" xảy ra khi và chỉ khi
			$$\cos 5x = \sin x =1.$$
			Điều này vô lý nên phương trình $(2)$ vô nghiệm.
		\end{itemize}
		Vậy phương trình đã cho có nghiệm là $x=-\dfrac{\pi}{6}+k2\pi$, $x=\dfrac{7\pi}{6}+k2\pi \ (k\in \mathbb{Z})$.
	}
\end{bt}
