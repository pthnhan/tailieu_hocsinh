\noindent\textbf{I. PHẦN TRẮC NGHIỆM}
\begin{ex}%[HK2 (2016-2017), THPT Kim Sơn A, Ninh Bình]%[0D4B2]
Tìm tập xác định $\mathscr{D}$ của hàm số $f(x)=\dfrac{x^2-x}{\sqrt{2x+6}}-\sqrt{10-x}$.
\choice
{\True $\mathscr{D}=(-3;10]$}
{$\mathscr{D}=[-3;10]$}
{$\mathscr{D}=(-3;10)$}
{$\mathscr{D}=[-3;10)$}
\loigiai{
Hàm số xác định khi $\heva{&2x+6>0\\&10-x\geq 0}\Leftrightarrow \heva{&x>-3\\&x\leq 10}\Leftrightarrow -3<x\leq 10$.\\
Vậy tập xác định của hàm số là $\mathscr{D}=(-3;10]$
}
\end{ex}
\begin{ex}%[HK2 (2016-2017), THPT Kim Sơn A, Ninh Bình]%[0D4B5]
Tìm tất cả các giá trị thực của tham số $m$ để phương trình $x^2-2mx+m^2-1=0$ có hai nghiệm dương phân biệt?
\choice
{$m\in [1;+\infty)$}
{\True $m\in (1;+\infty)$}
{$m\in (-\infty;-1)\cup (1;+\infty)$}
{$m\in (-\infty;+\infty)$}
\loigiai{
Ta có $\Delta'=m^2-m+1=1>0$ với mọi $m\in \mathbb{R}$, do đó phương trình đã cho luôn có hai nghiệm phân biệt $x_1, x_2$ với mọi $m\in \mathbb{R}$.\\
Suy ra phương trình đã cho có hai nghiệm dương phân biệt khi và chỉ khi $$\heva{&x_1+x_2=2m>0\\&x_1\cdot x_2=m^2-1>0}\Leftrightarrow m>1.$$
Vậy $m\in (1;+\infty)$.
}
\end{ex}
\begin{ex}%[HK2 (2016-2017), THPT Kim Sơn A, Ninh Bình]%[0D5B4]
Thống kê điểm kiểm tra môn toán (thang điểm 10) của một nhóm gồm 6 học sinh ta có bảng số liệu sau:
\begin{center}
\begin{tabular}[c]{|p{2.5cm}|p{1cm}|p{1cm}|p{1cm}|p{1cm}|p{1cm}|p{1cm}|}
\hline
{Tên học sinh}&{Kim} & {Sơn} &{Ninh}&{Bình} &{Việt}&{Nam}\\
\hline
{Điểm}&{$9$} & {$8$} &{$7$}&{$10$} &{$8$}&{$9$}\\
\hline
\end{tabular}
 \end{center}
 Tìm độ lệch chuẩn $s$ của bảng số liệu trên (làm tròn đến hàng phần trăm).
\choice
{$s=0,92$}
{$s=0,95$}
{\True $s=0,96$}
{$s=0,91$}
\loigiai{
Ta có $\overline{x}=\dfrac{9+8+7+10+8+9}{6}=\dfrac{51}{6}=8,5$.\\
Suy ra $s^2=\dfrac{1}{6}\left[2\left(9-8,5\right)^2+2\left(8-8,5\right)^2+\left(7-8,5\right)^2+\left(10-8,5\right)^2\right]=\dfrac{11}{12}$.\\
Do đó $s=\sqrt{\dfrac{11}{12}}=0,96$.
}
\end{ex}

\begin{ex}%[HK2 (2016-2017), THPT Kim Sơn A, Ninh Bình]%[0D6Y3]
 Cho cung $x$ thỏa mãn điều kiện tồn tại của các biểu thức. Mệnh đề nào sau đây \textbf{sai}?
\choice
{$\sin2x=2\tan x\cdot \cos^2x$}
{$\cos2x=\cos^4x-\sin^4x$}
{\True $\tan2x=2\tan^2x-1$}
{$\sin^22x+\cos^22x=1$}
\loigiai{
Do $\tan2x=\dfrac{2\tan x}{1-\tan^2x}$, suy ra mệnh đề $\tan2x=2\tan^2x-1$ là mệnh đề sai.
}
\end{ex}

\begin{ex}%[HK2 (2016-2017), THPT Kim Sơn A, Ninh Bình]%[0D6K2]
Biết biểu thức sau không phụ thuộc vào giá trị của cung $x$. Tính giá trị biểu thức $T=2\left(\sin^6x+\cos^6x\right)-3\left(\sin^4x+\cos^4x\right)+5.$
\choice
{$T=-1$}
{\True $T=4$}
{$T=6$}
{$T=5$}
\loigiai{
\emph{Cách 1}: Do biểu thức có giá trị không phụ thuộc vào $x$ nên thay $x=0$ ta được $T=4$.\\
\emph{Cách 2}: Ta có\\
$\sin^6x+\cos^6x=\left(\sin^2x+\cos^2x\right)^3-3\sin^2x\cos^2x\left(\sin^2x+\cos^2x\right)=1-3\sin^2x\cos^2x$.\\
$\sin^4x+\cos^4x=\left(\sin^2x+\cos^2x\right)^2-2\sin^2x\cos^2x=1-2\sin^2x\cos^2x$.\\
Từ đó suy ra
$T=2\left[1-3\sin^2x\cos^2x\right]-3\left[1-2\sin^2x\cos^2x\right]+5=4.$
}
\end{ex}

\begin{ex}%[HK2 (2016-2017), THPT Kim Sơn A, Ninh Bình]%[0H3K3]
Trong mặt phẳng toạ độ $Oxy$, phương trình nào dưới đây là phương trình chính tắc của elip $(E)$ có một tiêu điểm là $F_1(3;0)$ và có trục lớn dài hơn trục bé $2$ đơn vị.
\choice
{$\dfrac{x^2}{25}+\dfrac{y^2}{9}=1$}
{$\dfrac{x^2}{25}-\dfrac{y^2}{9}=1$}
{$\dfrac{x^2}{25}-\dfrac{y^2}{16}=1$}
{\True $\dfrac{x^2}{25}+\dfrac{y^2}{16}=1$}
\loigiai{
Trước hết nhận thấy các đường có phương trình $\dfrac{x^2}{25}-\dfrac{y^2}{9}=1$ và $\dfrac{x^2}{25}-\dfrac{y^2}{16}=1$ không phải là các đường elip.\\
Xét elip: $\dfrac{x^2}{25}+\dfrac{y^2}{9}=1$ có $\heva{&a^2=25\\&b^2=9} \Rightarrow \heva{&2a=10\\&2b=6}\Rightarrow 2a-2b=4.$ Elip này không thỏa mãn yêu cầu bài toán.\\
Xét elip: $\dfrac{x^2}{25}+\dfrac{y^2}{16}=1$ có $\heva{&a^2=25\\&b^2=16} \Rightarrow \heva{&a=5\\&b=4\\&c=\sqrt{a^2-b^2}=3.}$ Khi đó $2a-2b=2$ và $F(3;0)$ là một tiêu điểm của elip. Đây là elip thỏa mãn yêu cầu bài toán.
}
\end{ex}

\begin{ex}%[HK2 (2016-2017), THPT Kim Sơn A, Ninh Bình]%[0H3G1]
Trong mặt phẳng toạ độ $Oxy$, cho điểm $M(1;3)$. Tìm phương trình đường thẳng $(d)$ đi qua $M$ cắt các tia $Ox, Oy$ lần lượt tại $A$ và $B$ sao cho diện tích tam giác $OAB$ nhỏ nhất.
\choice
{$\dfrac{x}{3}+\dfrac{2y}{9}=1$}
{\True $\dfrac{x}{2}+\dfrac{y}{6}=1$}
{$\dfrac{2x}{3}+\dfrac{y}{9}=1$}
{$\dfrac{x}{4}+\dfrac{y}{4}=1$}
\loigiai{
Giả sử $A(a;0), B(0;b)$ với $a, b>0$. Khi đó $(d)$ có phương trình $\dfrac{x}{a}+\dfrac{y}{b}=1$.\\
Do $(d)$ đi qua $M$ nên ta có $\dfrac{1}{a}+\dfrac{3}{b}=1$.\\
Theo bất đẳng thức Cô-si ta có: $1=\dfrac{1}{a}+\dfrac{3}{b}\geq 2\sqrt{\dfrac{1}{a}\cdot\dfrac{3}{b}}=\dfrac{2\sqrt{3}}{\sqrt{ab}}$.\\
Từ đó suy ra $\sqrt{ab}\geq 2\sqrt{3}\Rightarrow ab\geq 12$, do đó $S_{\Delta OAB}=\dfrac{1}{2}OA\cdot OB = \dfrac{ab}{2}\geq 6$.\\
Dấu bằng xảy ra khi và chỉ khi $\heva{&\dfrac{1}{a}+\dfrac{3}{b}=1\\&\dfrac{1}{a}=\dfrac{3}{b}}\Rightarrow \heva{&a=2\\&b=6.}$\\
Vậy $(d)$ có phương trình {$\dfrac{x}{2}+\dfrac{y}{6}=1$}.
}
\end{ex}

Giải bất phương trình $\dfrac{x^2-3x}{2-x}\leq 0$.
\loigiai{
Điều kiện $x\neq 2$.\\
Đặt $f(x)=\dfrac{x^2-3x}{2-x}$, ta có $x^2-3x=0 \Leftrightarrow \hoac{&x=0\\&x=3.}$\\
Xét dấu $f(x)$

\begin{bt}%[HK2 (2016-2017), THPT Kim Sơn A, Ninh Bình]%[0D3B2]
Giải phương trình $\sqrt{x^2+2x-3}=2-x$.
\loigiai{
Ta có $\sqrt{x^2+2x-3}=2-x\Leftrightarrow \heva{&2-x\geq 0\\&x^2+2x-3=(2-x)^2}\Leftrightarrow\heva{&6x=7\\&x\leq 2}\Leftrightarrow x=\dfrac{7}{6}$.\\
Vậy phương trình có nghiệm $x=\dfrac{7}{6}$.
}
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Kim Sơn A, Ninh Bình]%[0D4K5]
Tìm tất cả các giá trị thực của tham số $m$ để $mx^2-mx+1>0$ với mọi $x\in\mathbb{R}$.
\loigiai{
\emph{Trường hợp 1}: $m=0$, bất phương trình trở thành $1>0$, suy ra bất phương trình nghiệm đúng với mọi $x\in\mathbb{R}$.\\
\emph{Trường hợp 2}: $m\neq 0$, khi đó bất phương trình nghiệm đúng với mọi $x\in\mathbb{R}$ khi và chỉ khi
$$\heva{&m>0\\&\Delta =m^2-4m<0}\Leftrightarrow m\in(0;4).$$
\emph{Kết luận}: Tập hợp các giá trị $m$ phải tìm là $[0;4)$.
}
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Kim Sơn A, Ninh Bình]%[0D6B3]
Cho $\pi<\alpha<\dfrac{3\pi}{2}$ và $\sin\alpha=-\dfrac{1}{3}$. Tính $\cos\alpha$ và $\cos2\alpha$.
\loigiai{
Do $\pi<\alpha<\dfrac{3\pi}{2}$ nên $\cos\alpha<0$.\\
Ta có $\cos^2\alpha=1-\sin^2\alpha=\dfrac{8}{9}$, từ đó suy ra $\cos\alpha=-\dfrac{2\sqrt{2}}{3}$.\\
Vì $\cos2\alpha=1-2\sin^2\alpha =1-2\dfrac{2}{9}=\dfrac{7}{9}$.
}
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Kim Sơn A, Ninh Bình]%[0H3B1]
Trong mặt phẳng tọa độ $Oxy$, cho điểm $A(-1;2)$ và đường thẳng $(\Delta): 3x-4y-2=0$. Tính khoảng cách từ $A$ tới $(\Delta)$, viết phương trình đường thẳng $(d)$ đi qua $A$ và song song song với $(\Delta)$.
\loigiai{
Ta có $\mathrm{d}\left(A;\Delta\right)=\dfrac{|3\cdot (-1)-4\cdot 2-2}{\sqrt{3^3+4^2}}=\dfrac{13}{5}.$\\
Đường thẳng $(d)$ song song song với $(\Delta)$ có phương trình dạng $3x-4y+C=0$ với $C\neq -2$.\\
Do $(d)$ qua $A$ nên $3\cdot (-1)-4\cdot 2+C=0 \Rightarrow C=11$.\\
Vậy $(d)$ có phương trình $3x-4y+11=0$.
}
\end{bt}
