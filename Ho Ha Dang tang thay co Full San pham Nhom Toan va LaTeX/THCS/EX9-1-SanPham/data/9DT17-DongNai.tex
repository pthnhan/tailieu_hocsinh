    \hfill
    \begin{enumerate}
        \item Giải phương trình $x^2-9x+20=0.$
        \item Giải hệ phương trình $\heva{&7x-3y=4\\&4x+y=5}$
        \item  Giải phương trình $x^4-2x^2-3=0.$
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item $x^2-9x+20=0\Leftrightarrow(x-4)(x-5)=0\Leftrightarrow
      \hoac{x=5\\x=4} $
        \item $\heva{&7x-3y=4\\&4x+y=5}\Leftrightarrow 
      \heva{
      	& 7x-3y=4\\&12x+3y=15
      } $ $\Leftrightarrow\heva{
      	&19x=19\\&4x+y=5}
      $ $
      \Leftrightarrow\heva{
      	&  x=1\\&y=1
      }$\\\\
        Vậy hệ phương trình có nghiệm duy nhất $(x;y)=(1;1)$.
        \item Đặt $t=x^2\ge0$, phương trình trở thành $t^2-2t-3=0.$\\\\
        Ta có $\Delta'=(-1)^2-1.(-3)=4\Rightarrow \hoac{&t=\dfrac{1-\sqrt{4}}{1}\\&t=\dfrac{1+\sqrt{4}}{1}}\Rightarrow \hoac{&t=-1\;\text{(loại)}\\&t=3\;(\text{thỏa mãn})}$ 
       \\\\Với $t=3$ suy ra $x^2=3\Leftrightarrow x=\pm\sqrt{3}.$
    \end{enumerate}
    }

	Cho hai hàm số $y=-\dfrac{1}{2}x^2$ và $y=x-4$ có đồ thị lần lượt là $(P)$ và $(d).$

		\item Vẽ hai đồ thị $(P),(d)$ trên cùng một mặt phẳng tọa độ.
		\item Tìm tọa độ giao điểm của hai đồ thị $(P)$ và $(d).$
	\end{enumerate}

\begin{enumerate}
		
		\item Vẽ hai đồ thị $(P),(d)$ trên cùng một mặt phẳng tọa độ.

\begin{ex}%[Đề thi vào 10, Sở giáo dục Đồng Nai, 2017]%[9D7B1]%[9D3B6]
	\hfill
	\begin{enumerate}
		\item Cho $a>0$ và $a\ne 4$. Rút gọn biểu thức $T=\left(\dfrac{\sqrt{a}-2}{\sqrt{a}+2}-\dfrac{\sqrt{a}+2}{\sqrt{a}-2}\right).\left(\sqrt{a}-\dfrac{4}{\sqrt{a}}\right).$
		\item  Một đội xe dự định chở $120$ tấn hàng. Để tăng sự  an toàn nên đến khi thực hiện đội xe được bổ sung thêm $4$ chiếc xe, lúc này số tấn hàng của mỗi xe chở ít hơn số tấn hàng của mỗi xe dự định chở là $1$ tấn. Tính số tấn hàng của mỗi xe dự định chở, biêt số tấn hàng của  mỗi xe chở khi dự định là bằng nhau, khi thực hiện là bằng nhau.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Với $a>0$ và $a\ne 4,$ ta có:
		\\\\ $\begin{array}{ll}T&=\left(\dfrac{\sqrt{a}-2}{\sqrt{a}+2}-\dfrac{\sqrt{a}+2}{\sqrt{a}-2}\right).\left(\sqrt{a}-\dfrac{4}{\sqrt{a}}\right)
		=\left(\dfrac{(\sqrt{a}-2)^2-(\sqrt{a}+2)^2}{(\sqrt{a}-2)(\sqrt{a}+2)}\right).\left(\dfrac{a-4}{\sqrt{a}}\right)\\\\
		&=\left(\dfrac{a-4\sqrt{a}+4-a-4\sqrt{a}-4}{a-4}\right).\left(\dfrac{a-4}{\sqrt{a}}\right)=\dfrac{-8\sqrt{a}}{\sqrt{a}}=-8.
		\end{array}$
		\item Gọi $x$ (xe) là số xe của đội lúc đầu, $x$ nguyên dương.\\
		Số tấn hàng mỗi xe dự định chở là $\dfrac{120}{x}$ (tấn).\\
		Số xe của đội lúc sau là $x+4$ (xe).\\
		Số tấn hàng mỗi xe khi thực hiện chở $\dfrac{120}{x+4}$ (tấn).\\
		Theo đề bài ta được phương trình:
		\begin{center} 
			$\begin{array}{ll}\dfrac{120}{x}-\dfrac{120}{x+4}=1
			&\Leftrightarrow 120(x+4)-120x=x(x+4)\\
			&\Leftrightarrow x^2+4x-480=0\Leftrightarrow\hoac{&x=20\\&x=-24\;(\text{loại})}
			\end{array}$
		
\end{center}
	Vậy số xe của đội lúc đầu là $20$ xe.
\end{enumerate}}
	\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Đồng Nai, 2017]%[9D4K6]
	Tìm các giá trị của tham số thực $m$ để phương trình $x^2+(2m-1)x+m^2-1=0$ có hai nghiệm phân biệt $x_1,x_2$ sao cho biểu thức $P=x_1^2+x_2^2$ đạt giá trị nhỏ nhất.
	\loigiai{\vskip0.1cm \noindent Phương trình $x^2+(2m-1)x+m^2-1=0$ có hai nghiệm phân biệt $x_1,x_2\Leftrightarrow\Delta>0\\\Leftrightarrow (2m-1)^2-4(m^2-1)>0\Leftrightarrow-4m+5>0\Leftrightarrow m<\dfrac{5}{4}. $\\
		Với $m<\dfrac{5}{4}$ , áp dụng định lý Vi-ét, ta được $\heva{
	&	x_1+x_2=1-2m\\&x_1x_2=m^2-1
		}$\\
		Ta có: \begin{center}
		$\begin{array}{ll}
		P&=x_1^2+x_2^2=(x_1+x_2)^2-2x_1x_2=(1-2m)^2-2(m^2-1)\\
		&=2m^2-4m+2+1	=2(m-1)^2+1\ge 1.	\end{array}$

		\end{center}
			Dấu $"="$ xảy ra $\Leftrightarrow m=1$ (thỏa mãn điều kiện $m<\dfrac{5}{4}$).\\
			Vậy với $m=1$ thì $P$ đạt giá trị nhỏ nhất.
	}
\end{ex}
