\chapter{Tích vô hướng và ứng dụng}
\subsection{Tóm tắt lí thuyết}
\begin{dn}
	Cho hai véc-tơ $\vec{a}$ và $\vec{b}$ khác véc-tơ $\vec{0}$. Tích vô hướng của $\vec{a}$ và $\vec{b}$ là một số, kí hiệu là $\vec{a} \cdot\vec{b}$, được xác định bởi công thức
	$$\vec{a} \cdot\vec{b} = |\vec{a}  | \cdot |\vec{b}  | \cos \left( \vec{a}, \vec{b} \right).  $$
\end{dn}

\begin{tc}
	\begin{enumerate}
	\item Với $\vec{a}$ và $\vec{b}$ khác véc-tơ $\vec{0}$ ta có $\vec{a}\cdot \vec{b} \Leftrightarrow \vec{ a} \perp \vec{b}$.
	\item Bình phương vô hướng của véc-tơ $\vec{ a}$ bằng bình phương độ dài của $\vec{ a}$:
	$\vec{a} ^2 = | \vec{ a} | ^2 . $
	\end{enumerate}
\end{tc}

\begin{tc}
Với ba véc-tơ $\vec{a}$, $\vec{b}$, $\vec{c}$ bất kì và mọi số $k$ ta có
\begin{enumerate}
	\item Tính chất giao hoán: $\vec{a} \cdot\vec{b} = \vec{b} \cdot\vec{a}$.
		\item Tính chất phân phối: $\vec{a} \cdot \left( \vec{b} + \vec{c} \right)  = \vec{a} \cdot \vec{b} + \vec{b} \cdot \vec{c}$.
		\item $\left( k \vec{ a} \right) \cdot \vec{b} = k\left( \vec{a} \cdot\vec{b} \right) = \vec{ a} \cdot \left( k \vec{b} \right)  $.
		\item $\vec{a} ^2 \ge 0$, $\vec{a} ^2 = 0 \Leftrightarrow \vec{ a} = \vec{0}$. 
		\end{enumerate}
\end{tc}

\begin{tc}
	\begin{enumerate}
	\item $\left( \vec{ a} + \vec{b} \right)^2 = \vec{a} ^2 + 2 \cdot \vec{a} \cdot\vec{b} +   \vec{b} ^2$.
	\item $\left( \vec{ a} - \vec{b} \right)^2 = \vec{a} ^2 - 2 \cdot \vec{a} \cdot\vec{b} +   \vec{b} ^2$.
		\item $\left( \vec{ a} + \vec{b} \right) \cdot  \left( \vec{ a} - \vec{b} \right) = \vec{a} ^2  -  \vec{b} ^2$.
\end{enumerate}
\end{tc}

\begin{dl}
Biểu thức tọa độ của tích vô hướng: Cho hai véc-tơ $\vec{ a} = (a_1 ;a_2)$, $\vec{b} = (b_1 ;b_2)$. Khi đó tích vô hướng $\vec{a} \cdot\vec{b}$ là $\vec{a} \cdot\vec{b} = a_1 b_1 +a_2 b_2.$
\end{dl}

\begin{dl}
Độ dài của véc-tơ $\vec{ a} = (a_1;a_2)$ là $| \vec{ a} | = \sqrt{a_1^2 + a_2^2}$.
\end{dl}

\begin{dl}
Góc giữa hai véc-tơ: Cho hai véc-tơ $\vec{ a} = (a_1 ;a_2)$ và $\vec{b} = (b_1 ;b_2)$ đều khác véc-tơ $\vec{0}$, ta có
$  \cos \left( \vec{a}, \vec{b} \right) = \dfrac{ \vec{a} \cdot\vec{b}}{| \vec{ a} | \cdot | \vec{ b} |} = \dfrac{a_1 b_1 +a_2 b_2}{\sqrt{a_1^2 + a_2^2} \cdot \sqrt{b_1^2 + b_2^2}}. $
\end{dl}

\begin{dl}
Khoảng cách giữa hai điểm $A(x_A;y_A)$ và $B(x_B;y_B)$:
$AB= \sqrt{(x_B-x_A)^2+(y_B-y_A)^2}.$
\end{dl}

\subsection{Các dạng toán và công thức giải nhanh}

\begin{dang}{Tính tích vô hướng}
	\begin{enumerate}
	\item Cho $\vec{a}$, $\vec{b}$
\begin{align*}
\vec{a} \cdot\vec{b} &= \dfrac{1}{4} \left( | \vec{ a} + \vec{ b} |^2 - | \vec{ a} - \vec{ b} |^2  \right) \\
&=\dfrac{1}{2} \left( | \vec{ a}  |^2 + | \vec{ b}  |^2- | \vec{ a} - \vec{ b} |^2  \right) \\
&=\dfrac{1}{2} \left( | \vec{ a} + \vec{ b} |^2 - | \vec{ a}  |^2 -  |  \vec{ b} |^2  \right).
\end{align*}
\loigiai{
Ta có $| \vec{ a} + \vec{ b} |^2 = | \vec{ a}  |^2 + | \vec{ b}  |^2 + 2\vec{a} \cdot\vec{b} $, $| \vec{ a} -\vec{ b} |^2 = | \vec{ a}  |^2 + | \vec{ b}  |^2 - 2\vec{a} \cdot\vec{b}$, thay vào vế trái, ta có đpcm.

}
\item Với mọi điểm $A$, $B$, $C$, ta có $ \vec{ AB} \cdot \vec{ AC} = \dfrac{1}{2} \left( AB^2 + AC^2 -BC^2 \right) $.
\loigiai{ Ta có \begin{align*}
\vec{ AB} \cdot \vec{ AC}&=- \vec{ BA} \cdot \vec{ AC}=- \dfrac{| \vec{ BA} + \vec{ AC}|^2 - |\vec{ BA}|^2 - |\vec{ AC}|^2 }{2}\\ &= -\dfrac{|BC|^2 - |\vec{ BA}|^2 - |\vec{ AC}|^2 }{2} = \dfrac{1}{2} \left( AB^2 + AC^2 -BC^2 \right).
	\end{align*} 
}
\item Với $4$ điểm $M$, $N$, $P$, $Q$ ta có $\vec{ MN} \cdot \vec{ PQ} = \dfrac{1}{2} \left( MQ^2 + NP^2 -MP^2 -NQ^2 \right)  $.
\loigiai{
Gọi $I$ là trung điểm $PQ$ ta có $\dfrac{1}{2} \left( \vec{ MQ}+ \vec{ MP} \right)  = \vec{ MI} $ và $\dfrac{1}{2} \left( \vec{ NQ}+ \vec{ NP} \right)  = \vec{ NI}$.\\
 Do đó
\begin{align*}
&\dfrac{1}{2} \left( MQ^2 + NP^2 -MP^2 -NQ^2 \right) = \dfrac{1}{2} \left(   \vec{MQ}^2 -\vec{ MP}^2 \right) -  \dfrac{1}{2} \left(  \vec{ NQ}^2 -\vec{ NP}^2 \right)\\
&= \dfrac{1}{2} \left(   \vec{ MQ} -\vec{ MP}\right) \left(   \vec{ MQ} +\vec{ MP}\right) - \dfrac{1}{2} \left(   \vec{ NQ} -\vec{ NP}\right) \left(   \vec{ NQ} +\vec{ NP}\right)\\
&=\vec{PQ}\cdot \vec{ MI} - \vec{PQ}\cdot \vec{ NI} = \vec{PQ}\cdot \left(  \vec{ MI} - \vec{ NI}\right) = \vec{ MN} \cdot \vec{ PQ}.
\end{align*}
}
	\end{enumerate}
\end{dang}

\begin{vd}%[Ngọc Diệp]%[0H2G2-1]

	\choice

{\True $-50$}
{$170$}

\loigiai{Ta có $\vec{ MN} \cdot \vec{ PQ} = \dfrac{1}{2} \left( MQ^2 + NP^2 -MP^2 -NQ^2 \right) = \dfrac{12^2 + 1^2 - \sqrt{26}^2 - 13^2}{2}  =-50$.
}
\end{vd}

\begin{vd}%[Ngọc Diệp]%[0H2K2-1]
Cho tam giác $ABC$ có $\vec{ AB} \cdot \vec{ CB} = 4$ và $ \vec{ AC} \cdot \vec{BC} = 9 $. Tính $AB^2 - AC^2$.
	\choice
{$5$}
{$16$}
{\True $-5$}
{$-16$}
\loigiai{ Ta có $\vec{ AB} \cdot \vec{ CB} +  \vec{ AC} \cdot \vec{BC} = \vec{ AB} \cdot \vec{ CB} +  \vec{ CA} \cdot \vec{CB} = \vec{CB} \cdot \left(\vec{ AB} +\vec{ CA}  \right) =  \vec{CB} \cdot \vec{CB}= CB^2  $.\\
	Suy ra $CB^2 = 13$.\\
	Ta lại có $\vec{ AB} \cdot \vec{ CB} = 4 \Leftrightarrow \vec{ BA} \cdot \vec{ BC} = 4 \Leftrightarrow AB^2 +BC^2- AC^2 =8 \Rightarrow AB^2 - AC^2 = -5$.
}
\end{vd}

\begin{dang}{Tính diện tích tam giác}
		\begin{enumerate}
		\item 	Cho $\vec{ a} = (a_1 ;a_2)$ và $\vec{b} = (b_1 ;b_2)$
		$$| \vec{ a}  |^2 \cdot |  \vec{ b} |^2 - \left( \vec{a} \cdot \vec{b} \right)^2 = (a_1b_2 - a_2 b_1)^2.  $$
		\loigiai{
			\begin{align*}
			VT &=| \vec{ a}  |^2 \cdot |  \vec{ b} |^2 - \left( \vec{a} \cdot \vec{b} \right)^2 \\
			&= (a_1^2 + a_2^2)(b_1^2 + b_2^2)-(a_1b_1 +a_2b_2)\\
			&= (a_1b_2)^2 + (a_2 b_1)^2 - 2a_1a_2b_1b_2\\
			&=(a_1b_2 - a_2 b_1)^2 = VP.
			\end{align*} 
		}
		\item Cho tam giác $ABC$ có $\vec{ AB} = (x_1; y_1)$, $\vec{AC} = (x_2 ;y_2)$.
		$$ S_{ \triangle ABC}= \dfrac{1}{2} \sqrt{AB^2 AC^2 - \left( \vec{ AB} \cdot \vec{AC} \right)^2 }= \dfrac{1}{2} | x_1 y_2 - x_2 y_1 | $$
		\loigiai{ \begin{align*}
			S_{ \triangle ABC}&= \dfrac{1}{2} AB \cdot AC \sin A \\
			&=  \dfrac{1}{2} AB \cdot AC \sqrt{ 1-cos ^2 A} \\
			&= \dfrac{1}{2} AB \cdot AC \sqrt{1- \left( \dfrac{ \vec{ AB} \cdot \vec{ AC}}{AB \cdot AC}\right)^2 } \\
			&=  \dfrac{1}{2} \sqrt{AB^2 AC^2 - \left( \vec{ AB} \cdot \vec{AC} \right)^2 } \\
			&=  \dfrac{1}{2} \sqrt{\left(x_1 y_2 - x_2 y_1 \right)^2 }\\
			&= \dfrac{1}{2}  | x_1 y_2 - x_2 y_1 |.
			\end{align*}
		}
	\end{enumerate}
\end{dang}

\begin{vd}%[Ngọc Diệp]%[0H2K2-1]
	 Tam giác $ABC$ có diện tích $S=\dfrac{3}{2}$, $A(0;2)$, $B(1;3)$. Tìm tọa độ điểm $C$, biết $C$ nằm trên đường thẳng $d: x+y-7=0$.
	\choice

{$C(4;3)$, $C(-1;6)$}

	Vì tam giác $ABC$ có diện tích $S=\dfrac{3}{2}$ nên $$\dfrac{1}{2}|5-x-x|= \dfrac{3}{2} \Leftrightarrow |5-2x |=3 \Leftrightarrow \hoac{&x=1\\ &x=4} \Leftrightarrow \hoac{&x=1,y=6\\ &x=4,y=3.}$$
	Vậy $C(4;3)$ hoặc $C(1;6)$.
	
}

\end{vd}

\begin{dang}{Tìm tọa độ một số điểm đặc biệt}
	\begin{enumerate}
	\item $M$ là trung điểm $AB$: $\vec{MA} + \vec{MB} = \vec{0} \Leftrightarrow \heva{&x_M = \dfrac{x_A+x_B}{2}\\ &y_M= \dfrac{y_A+y_B}{2}.}$
	\item $G$ là trọng tâm tam giác $ABC$: $\vec{GA} + \vec{GB} + \vec{GC}  = \vec{0} \Leftrightarrow \heva{&x_G = \dfrac{x_A+x_B+x_C}{3}\\ &y_G= \dfrac{y_A+y_B+y_C}{3}.}$
		\item $M$ chia đoạn $AB$ theo tỉ số $k$, $(k \ne 0)$: $\vec{MA} =k \vec{MB}  \Leftrightarrow \heva{&x_M = \dfrac{x_A-kx_B}{1-k}\\ &y_M= \dfrac{y_A-ky_B}{1-k}.}$
	\item $D$ là chân đường phân giác trong $\triangle ABC$, $\dfrac{AB}{AC} = k$: $\dfrac{ \vec{ DB}}{\vec{CD}} = \dfrac{AB}{AC} \Leftrightarrow \heva{&x_D = \dfrac{x_C+k x_B}{1+k}\\ &y_D= \dfrac{y_C+ky_B}{1+k}.}$
	\item $E$ là chân đường phân giác ngoài $\triangle ABC$, $\dfrac{AB}{AC} = k$: $\dfrac{ \vec{ EB}}{\vec{EC}} = \dfrac{AB}{AC} \Leftrightarrow \heva{&x_E = \dfrac{x_B-k x_C}{1-k}\\ &y_E= \dfrac{y_B-ky_C}{1-k}.}$
\item $H$ là hình chiếu vuông góc của $M(x_0;y_0)$ lên đường thẳng $\Delta: ax+by+c=0$, $H\left( x_0 -at;y_0 -bt\right) $, $t= \dfrac{a x_0 + b y_0 +c}{a^2 + b^2}$.
 \loigiai{ Đường thẳng $d$ đi qua $M(x_0;y_0)$ và có véc-tơ pháp tuyến $\vec{ n_{d} }= (-b;a)$ có phương trình
 	$$-b(x-x_0)+a(y-y_0) = 0 \Leftrightarrow -bx + ay + bx_0 - ay_0 = 0.$$
 	$H$ là giao điểm của $\Delta$ và $d$ nên
 	\begin{align*}
\heva{ &ax+by = -c\\ &-bx + ay= ay_0 - bx_0} & \Leftrightarrow \heva{&abx + b^2 y = -cb\\ & -abx + a^2 y = a^2 y_0 - ab x_0} \\
&\Leftrightarrow \heva{&x=x_0 - a\dfrac{a x_0 + b y_0 +c}{a^2 + b^2} \\ & y = y_0 - b \dfrac{a x_0 + b y_0 +c}{a^2 + b^2}.}
 	\end{align*}
}
\end{enumerate}
\end{dang}
