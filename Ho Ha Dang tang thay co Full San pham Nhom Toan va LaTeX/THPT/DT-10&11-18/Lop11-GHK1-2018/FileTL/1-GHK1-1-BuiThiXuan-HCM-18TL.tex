\begin{bt} %[Thi GHK1, Bùi Thị Xuân - Hồ Chí Minh, 2018]%[Trần Xuân Thiện, EX-DT-10&11]%[1D1K3-4]
Giải phương trình sau: 
$\sin x+\sqrt{3} \cos x = \sqrt{2}$.
\loigiai{
\begin{eqnarray*}
& & \sin x + \sqrt{3} \cos x = \sqrt{2}\\
&\Leftrightarrow & \dfrac{1}{2}\sin x + \dfrac{\sqrt{3}}{2} \cos x = \dfrac{\sqrt{2}}{2} \\
&\Leftrightarrow & \sin \left(x+\dfrac{\pi}{3}\right) = \sin \dfrac{\pi}{4}\\
&\Leftrightarrow & \hoac{& x + \dfrac{\pi}{3} = \dfrac{\pi}{4} + k2\pi  \\ & x + \dfrac{\pi}{3} = \pi-\dfrac{\pi}{4}+k2\pi  \\ }\\
&\Leftrightarrow & \hoac{& x = -\dfrac{\pi}{12}+k2\pi  \\ & x = \dfrac{5\pi}{12} + k2\pi  \\ } \left(k \in \mathbb{Z} \right).
\end{eqnarray*}
Kết luận: Phương trình đã cho có họ nghiệm là $\hoac{& x = -\dfrac{\pi}{12}+k2\pi  \\ & x = \dfrac{5\pi}{12}+k2\pi  \\ } \left(k \in \mathbb{Z} \right)$.
}
\end{bt}

\begin{bt} %[Thi GHK, Bùi Thị Xuân - Hồ Chí Minh, 2018]%[Trần Xuân Thiện, EX-DT-10&11]%[1D1K3-4]
Giải phương trình sau: 
$\sin \dfrac{x}{2}+\cos \dfrac{x}{2}-\sin x+1=0$.
\loigiai{
Đặt $t=\sin \dfrac{x}{2}+\cos \dfrac{x}{2}\left(-\sqrt{2}\le t\le \sqrt{2}\right)$ $\Rightarrow \sin x=2\sin \dfrac{x}{2}\cos \dfrac{x}{2}=t^2-1$.\\
Phương trình trở thành $-t^2+t+2=0$ $\Leftrightarrow \hoac{& t=-1 ~~( tm) \\ & t=2~~( l) \\ }$.\\
Với $t = -1 $, ta có: 
{\allowdisplaybreaks 
\begin{eqnarray*}
& & \sin \dfrac{x}{2}+\cos \dfrac{x}{2} = -1\\
&\Leftrightarrow & \sqrt{2}\sin \left(\dfrac{x}{2}+\dfrac{\pi}{4}\right)=-1\\
&\Leftrightarrow & \sin \left(\dfrac{x}{2}+\dfrac{\pi}{4}\right)=-\dfrac{\sqrt{2}}{2}\\
&\Leftrightarrow & \sin \left(\dfrac{x}{2}+\dfrac{\pi}{4}\right)=\sin \left(-\dfrac{\pi}{4}\right)\\
&\Leftrightarrow & \hoac{& \dfrac{x}{2}+\dfrac{\pi}{4}=-\dfrac{\pi}{4}+k2\pi  \\ & \dfrac{x}{2}+\dfrac{\pi}{4}=\dfrac{5\pi}{4}+k2\pi  \\ }\\
&\Leftrightarrow & \hoac{& x=-\pi+k4\pi  \\ & x=2\pi+k4\pi  \\ } \left(k\in \mathbb{Z}\right).
\end{eqnarray*}
}
Kết luận: Phương trình đã cho có họ nghiệm là $\hoac{& x=-\pi+k4\pi  \\ & x=2\pi+k4\pi  \\ }$ $\left(k\in \mathbb{Z}\right)$.
}
\end{bt}

\begin{bt} %[Thi GHK, Bùi Thị Xuân - Hồ Chí Minh, 2018]%[Trần Xuân Thiện, EX-DT-10&11]%[1D1K3-1]
Giải các phương trình sau: 
$\cos 2x+\cos 3x-\sin x-\cos 4x=\sin 6x$.
\loigiai{
\begin{eqnarray*}
& & \cos 2x + \cos 3x - \sin x - \cos 4x = \sin 6x \\
&\Leftrightarrow & 2\sin 3x \cos 3x + \left(\cos 4x - \cos 2x\right)+\left(\sin x - \cos 3x\right)=0 \\
&\Leftrightarrow & 2\sin 3x \cos 3x - 2\sin 3x \sin x + \left(\sin x - \cos 3x\right)=0 \\
&\Leftrightarrow & 2\sin 3x\left(\cos 3x - \sin x\right)+\left(\sin x - \cos 3x\right)=0 \\
&\Leftrightarrow & \left(\cos 3x - \sin x\right)\left(2\sin 3x - 1\right)=0 \\
&\Leftrightarrow & \hoac{& \cos 3x = \sin x \\ & \sin 3x = \dfrac{1}{2}  } \\
&\Leftrightarrow & \hoac{& \cos 3x = \cos \left(\dfrac{\pi}{2} - x\right) \\ & \sin 3x = \sin \dfrac{\pi}{6} \\ } \\
&\Leftrightarrow & \hoac{& 3x=\dfrac{\pi}{2}-x+k2\pi  \\ & 3x=x-\dfrac{\pi}{2}+k2\pi  \\ & 3x=\dfrac{\pi}{6}+k2\pi  \\ & 3x=\dfrac{5\pi}{6}+k2\pi  \\ } \\
&\Leftrightarrow & \hoac{& x=\dfrac{\pi}{8}+\dfrac{k\pi}{2} \\ & x=-\dfrac{\pi}{4}+k\pi  \\ & x=\dfrac{\pi}{18}+\dfrac{k2\pi}{3} \\ & x=\dfrac{5\pi}{18}+\dfrac{k2\pi}{3} \\ } \left(k\in \mathbb{Z}\right).
\end{eqnarray*} 
Kết luận: Phương trình đã cho có nghiệm là $\hoac{& x=\dfrac{\pi}{8}+\dfrac{k\pi}{2} \\ & x=-\dfrac{\pi}{4}+k\pi  \\ & x=\dfrac{\pi}{18}+\dfrac{k2\pi}{3} \\ & x=\dfrac{5\pi}{18}+\dfrac{k2\pi}{3} \\ }$ $\left(k\in \mathbb{Z}\right)$.
}
\end{bt}

\begin{bt} %[Thi GHK, Bùi Thị Xuân- Hồ Chí Minh, 2018]%[Trần Xuân Thiện, EX-DT-10&11]%[1H2B1-2][1H2K1-3][1H2K1-5]
Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình bình hành tâm $O$. Gọi $E$ là trung điểm của $SA$, $F$ là điểm trên cạnh $SB$ sao cho $SF=2FB$ và $G$ là trọng tâm của tam giác $SCD$.
\begin{enumerate}
\item Tìm giao điểm $I$ của đường thẳng $SG$ và mặt phẳng $\left(ABCD\right)$. Tìm giao tuyến của hai mặt phẳng $\left(SBI\right)$ và $\left(SAC\right)$.
\item Tìm giao tuyến của hai mặt phẳng $\left(SAC\right)$ và $\left(EFG\right)$. Tìm giao điểm $K$ của đường thẳng $SC$ và mặt phẳng $\left(EFG\right)$.
\item  Gọi $L$ là giao điểm của đường thẳng $SD$ và mặt phẳng $\left(EFG\right)$. Chứng minh rằng ba đường thẳng $EK$, $FL$ và $SO$ đồng quy.
\end{enumerate}

\begin{enumerate}
\item  Trong mặt phẳng $\left(SCD\right)$, gọi $I=SG\cap CD$ $\Rightarrow \heva{
& I\in SG \\ 
& I\in CD,CD\subset \left(ABCD\right) \\ 
}$.\\
 Vậy $I=SG\cap \left(ABCD\right)$.\\
Trong mặt phẳng $\left(ABCD\right)$, gọi $J=BI\cap AC$ $\Rightarrow \heva{
& J\in BI,BI\subset \left(SBI\right) \\ 
& J\in AC,AC\subset \left(SAC\right) \\ 
}$. \\
Vậy $J\in \left(SBI\right)\cap \left(SAC\right)$\\
Mà $S\in \left(SBI\right)\cap \left(SAC\right)$ $\Rightarrow $ $SJ = \left(SBI\right)\cap \left(SAC\right)$.
\item Trong mặt phẳng $\left(SBI\right)$, gọi $M=FG\cap SJ$ $\Rightarrow \heva{
& M\in FG,FG\subset \left(EFG\right) \\ 
& M\in SJ,SJ\subset \left(SAC\right) \\ 
}$.\\
 Vậy $M\in \left(EFG\right)\cap \left(SAC\right)$.\\
Mà $E\in SA\Rightarrow E\in \left(SAC\right)$ $\Rightarrow E\in \left(EFG\right)\cap \left(SAC\right)$.\\
Vậy $ME=\left(EFG\right)\cap \left(SAC\right)$.\\
Trong mặt phẳng $\left(SAC\right)$, gọi $K=ME\cap SC$ $\Rightarrow \heva{
& K\in ME,ME\subset \left(EFG\right) \\ 
& K\in SC \\ 
}$.\\
 Vậy $K=SC\cap \left(EFG\right)$.
\item Do $KG \in \left(EFG)\right)$, gọi $L = KG \cap SD$.

& N\in EK,EK\subset \left(SAC\right) \\ 
& N\in FL,FL\subset \left(SBD\right) \\ 
}$.\\
 Vậy $N\in \left(SAC\right)\cap \left(SBD\right)$.\\
Mà $AC\cap BD=O$ $\Rightarrow \heva{
& O\in AC,AC\subset \left(SAC\right) \\ 
& O\in BD,BD\subset \left(SBD\right) \\ 
}$. \\
Vậy $O\in \left(SAC\right)\cap \left(SBD\right)$.\\
Lại có $S\in \left(SAC\right)\cap \left(SBD\right)$.\\
Vậy $S$, $N$, $O$ thẳng hàng hay $EK$, $FL$ và $SO$ đồng quy tại $N$.
\end{enumerate}
