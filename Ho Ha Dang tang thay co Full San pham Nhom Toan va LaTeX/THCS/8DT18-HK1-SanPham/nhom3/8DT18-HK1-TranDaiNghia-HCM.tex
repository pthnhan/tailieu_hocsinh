	Phân tích đa thức sau thành nhân tử
	\begin{enumerate}
		\item $8a^3+2a-18ab^4-8a^2$,
		\item $(h^2-4h)^2+13(h^2-4h)+40$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có
	\begin{eqnarray*}
	8a^3+2a-18ab^4-8a^2&=&2a\cdot (4a^2+1-9b^4-4a)\\
	&=& 2a\cdot \left[ (2a-1)^2-(3b^2)^2\right]\\
	&=& 2a \cdot (2a-1-3b^2)\cdot (2a-1+3b^2).
	\end{eqnarray*}
	\item Ta có 
	\begin{eqnarray*}
	(h^2-4h)^2+13(h^2-4h)+40 &=& (h^2-4h)^2+5 (h^2-4h) +8 \left[(h^2-4h)+5\right]\\
	&=& \left[(h^2-4h)+5\right]\cdot \left[ (h^2-4h)+8\right].
	\end{eqnarray*}
\end{enumerate}
}

	Thực hiện các phép tính sau
	\begin{enumerate}
		\item $(12x^3-28x^2+21x-5):(6x-5)-(2x^2-4x)$,
		\item $\left(\dfrac{x+1}{x-3}+\dfrac{5x-39}{x^2-9}-\dfrac{11}{x+3}\right):\dfrac{x^2+2x+1}{2x+6}.$
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có 
	\begin{eqnarray*}
		\dfrac{12x^3-28x^2+21x-5}{6x-5}-(2x^2-4x)& = & \dfrac{(2x^2-3x +1)(6x-5)}{6x-5} -2x^2+4x \\
		&= & 2x^2-3x+1 -2x^2+4x\\
		&= & x+1.
	\end{eqnarray*}
	\item Ta có
	\begin{eqnarray*}
	\left(\dfrac{x+1}{x-3}+\dfrac{5x-39}{x^2-9}-\dfrac{11}{x+3}\right):\dfrac{x^2+2x+1}{2x+6} &=& \dfrac{(x+1)(x+3)+5x-39 -11(x-3)}{x^2-9} \cdot \dfrac{2x+6}{(x+1)^2}\\
	&=& \dfrac{x^2-2x-3}{x^2-9}\cdot \dfrac{2(x+3)}{(x+1)^2}\\
	&=& \dfrac{(x+1)(x-3)\cdot 2 (x+3)}{(x-3)(x+3)(x+1)^2}\\
	& =& \dfrac{2}{x+1}.
	\end{eqnarray*}
\end{enumerate}
}
\end{bt}

	Cho tam giác nhọn $ABC$ có $BC=2AB$. Gọi $D$ là trung điểm của $BC$. Từ $D$ kẻ tia $Dx \parallel AB$ và từ $A$ kẻ tia $Ay \parallel BC$ sao cho $Dx$ cắt $Ay$ tại $E$.
	\begin{enumerate}
		\item Chứng minh rằng tứ giác $ABDE$ là hình thoi.
		\item Chứng minh tứ giác $AECD$ là hình bình hành và $BE \perp CE$.
		\item Gọi $O$ là giao điểm của $BE$ và $AD$, $K$ là giao điểm của $DE$ và $OC$, $J$ là giao điểm của $BK$ và $EC$. Chứng minh rằng $JE=JC$.
		\item Gọi $I$ là giao điểm của $BK$ và $OD$. Chứng minh rằng $4\cdot IO =AD$.
	\end{enumerate}
\loigiai{
\begin{center}
	\begin{tikzpicture}[scale=1, line join=round, line cap=round]
	%\clip (-5,0.5) rectangle (5,6);
	\tkzDefPoints{-4/0/B,4/0/C,0/0/D}
	\tkzDefPointBy[rotation=center B angle 65](D)\tkzGetPoint{A}
	\tkzDefPointBy[translation=from B to D](A)\tkzGetPoint{E}
	\tkzInterLL(A,D)(B,E)\tkzGetPoint{O}
	\coordinate (J) at ($(E)!0.5!(C)$);
	\tkzInterLL(B,J)(E,D)\tkzGetPoint{K}
	\tkzInterLL(A,D)(B,J)\tkzGetPoint{I}
	
	\tkzDrawPolygon(A,B,C,E)
	\tkzDrawSegments(A,D D,E B,E C,O D,E B,J A,C)
	
	\tkzDrawPoints[size=5,fill=black](A,B,C,D,E,O,J,K,I)
	\tkzLabelPoints[above](A,E)
	\tkzLabelPoints[below](D,I)
	\tkzLabelPoints[left](B,O)
	\tkzLabelPoints[right](C,J)
	\tkzLabelPoints[above left](K)
	\tkzMarkSegments[mark=||](A,B B,D D,C)
	
	\end{tikzpicture}

\begin{enumerate}
	\item Do $\heva{&AB\parallel DE\\&AE\parallel BD} \Rightarrow $ tứ giác $ABDE$ là hình bình hành.\\
	Mặt khác $AB$, $BD$ là hai cạnh kề của hình bình hành và $AB=BD$. Suy ra tứ giác $ABDE$ là hình thoi.
	\item Do tứ giác $ABDE$ là hình thoi, suy ra $BE \perp AD$.\\ 
	Do $AE \parallel BD \Rightarrow AE \parallel DC $ và $AE=DC$ (vì cùng bằng $BD$), suy ra tứ giác $AECD$ là hình bình hành, suy ra $AD \parallel EC$.\\
	Do $\heva{& BE \perp AD\\& AD \parallel EC} \Rightarrow BE \perp EC$.
	\item  Xét $\triangle BEC$, do $O$ và $D$ lần lượt là trung điểm của $BE$ và $BC$, suy ra $CO$ và $ED$ là đường trung tuyến của $\triangle BEC$. Suy ra $K$ là trọng tâm của $\triangle BEC$. Do đó $BJ$ là đường trung tuyến của $\triangle BEC$, hay $J$ là trung điểm của $EC \Rightarrow JE=JC$.
	\item Do $O$ là trung điểm của $BE$ và $OI \parallel EJ$, suy ra $OI$ là đường trung bình của $\triangle BEJ$, suy ra $OI=\dfrac{1}{2}EJ$, mà $EJ=\dfrac{1}{2}EC \Rightarrow OI=\dfrac{1}{4}EC$.\\
	Mà tứ giác $ADCE$ là hình bình hành nên $EC=AD$. Do đó $OI=\dfrac{1}{4}AD$ hay $4\cdot IO =AD$.
\end{enumerate}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Trần Đại Nghĩa, HCM]%[Nguyễn Sỹ, dự án 8DT18-HK1]%[8D3B7]
	Cô An đi siêu thị mua một món hàng đang khuyến mãi giảm giá $15\% $, cô có thẻ khách hàng thân thiết của siêu thị nên được giảm thêm $2\%$ trên giá đã giảm, do đó cô chỉ phải trả $249900$ đồng cho món hàng đó. Hỏi giá ban đầu của món hàng nếu không khuyến mãi là bao nhiêu?
	\loigiai{
Gọi $x$ đồng là	giá ban đầu của món hàng nếu không khuyến mãi (điều kiện $x>249900$).\\
Do món hàng đang khuyến mãi giảm giá $15\%$, nên số tiến được giảm là $x\cdot \dfrac{15}{100}$.\\
Cô An có thẻ khách hàng thân thiện nên được giảm thêm $2\%$ trên giá đã giảm. Do đó số tiền cô An trả cho sản phẩm là $$\left(x-\dfrac{15}{100}x\right) - \left(x-\dfrac{15}{100}x\right) \dfrac{2}{100}.$$
Theo giả thiết ta có 
\begin{eqnarray*}
& & \left(x-\dfrac{15}{100}x\right) - \left(x-\dfrac{15}{100}x\right) \dfrac{2}{100}=249900\\ 
&\Leftrightarrow & \dfrac{85x}{100} \left( 1-\dfrac{1}{50}\right) =249900\\
& \Leftrightarrow & \dfrac{17x}{20}\cdot \dfrac{49}{50} =249900\\
&\Leftrightarrow& x= 300000\quad \text{(thỏa mãn)}.
\end{eqnarray*}

Vậy giá ban đầu của món hàng nếu không khuyến mãi là $300000$ đồng.
}
\end{bt}
