	Giải phương trình và hệ phương trình sau:
	\begin{enumerate}
		\item $x^2+9=6x$. 
		\item $\left\{\begin{aligned}& x+2y=-3\\& 3x+y=1.\end{aligned}\right.$
	\end{enumerate}
	\loigiai{\begin{enumerate}
			\item  Ta có \begin{eqnarray*}
				x^2+9=6x &\Leftrightarrow& x^2-6x+9=0\Leftrightarrow {\left(x-3\right)}^2=0\\
				&\Leftrightarrow &x-3=0\Leftrightarrow x=3.
				\end{eqnarray*}
			Vậy nghiệm của phương trình là $x=3$.
		\item Ta có \begin{eqnarray*}
		&&\left\{\begin{aligned}& x+2y=-3 \\& 3x+y=1\end{aligned}\right.\Leftrightarrow \left\{\begin{aligned}& x=-2y-3\\& 3\left(-2y-3\right)+y=1\end{aligned}\right.\Leftrightarrow \left\{\begin{aligned}& x=-2y-3 \\	&-6y-9+y=1\end{aligned}\right.\\
		&\Leftrightarrow &\left\{\begin{aligned}& x=-2y-3 \\&-5y=10\end{aligned}\right. \Leftrightarrow \left\{\begin{aligned}& x=1 \\& y=-2\end{aligned}\right.
		\end{eqnarray*}
		Vậy nghiệm của hệ phương trình là $\left\{\begin{aligned}& x=1\\&y=-2\end{aligned}\right.$.
		\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi thử vào lớp 10 năm học 2017-2018, Bình Giang, Hải Dương]%[9D1B8]%[9D2K4]
	\hfill
	\begin{enumerate}
		\item Cho $a>0,b>0,a\ne b$, rút gọn biểu thức:
		$A=\left(\dfrac{\sqrt{a}}{\sqrt{ab}-b}+\dfrac{\sqrt{b}}{\sqrt{ab}-a}\right):\dfrac{\sqrt{a}+\sqrt{b}}{a\sqrt{b}-b\sqrt{a}}$.
		\item Tìm giá trị của m để ba đường thẳng: $3 x-y=10$, $2x + 3y=-8$ và $y=mx+6$ cùng đi qua một điểm.
	\end{enumerate}
\loigiai{\hfill
	\begin{enumerate}
		\item Với $a>0,b>0,a\ne b$, ta có:
		\begin{eqnarray*}
		A&=&\left(\dfrac{\sqrt{a}}{\sqrt{ab}-b}+\dfrac{\sqrt{b}}{\sqrt{ab}-a}\right):\dfrac{\sqrt{a}+\sqrt{b}}{a\sqrt{b}-b\sqrt{a}}\\
		&=&\left(\dfrac{\sqrt{a}}{\sqrt{b}\left(\sqrt{a}-\sqrt{b}\right)}-\dfrac{\sqrt{b}}{\sqrt{a}\left(\sqrt{a}-\sqrt{b}\right)}\right)\cdot \dfrac{a\sqrt{b}-b\sqrt{a}}{\sqrt{a}+\sqrt{b}}\\
		&=&\left(\dfrac{\sqrt{a}}{\sqrt{b}\left(\sqrt{a}-\sqrt{b}\right)}-\dfrac{\sqrt{b}}{\sqrt{a}\left(\sqrt{a}-\sqrt{b}\right)}\right)\cdot \dfrac{a\sqrt{b}-b\sqrt{a}}{\sqrt{a}+\sqrt{b}}\\
		&=&\dfrac{a-b}{\sqrt{ab}\left(\sqrt{a}-\sqrt{b}\right)}\cdot \dfrac{\sqrt{ab}\left(\sqrt{a}-\sqrt{b}\right)}{\sqrt{a}+\sqrt{b}}\\
		&=&\dfrac{a-b}{\sqrt{a}+\sqrt{b}}=\sqrt{a}-\sqrt{b}.
		\end{eqnarray*}
		\item Tọa độ giao điểm của hai đường thẳng: $3x-y=10$, $2x+3y=-8$ là nghiệm của hệ phương trình:
		\begin{eqnarray*}
		&&\left\{\begin{aligned}
			& 3 x-y=10 \\& 2 x + 3y=-8\end{aligned}\right.\Leftrightarrow \left\{\begin{aligned}
			& y=3 x-10 \\& 2 x+3\left(3 x-10\right)=-8\end{aligned}\right.\\
		&\Leftrightarrow &\left\{\begin{aligned}& y=3x-10\\	&2x+9x-30=-8\end{aligned}\right. \Leftrightarrow \left\{\begin{aligned}& y=3x-10\\& 11x=22\end{aligned}\right.\\
		&\Leftrightarrow &\left\{\begin{aligned}&y=-4\\ & x=2.\end{aligned}\right.
		\end{eqnarray*}
		Ba đường thẳng: $3 x-y=10$, $2x + 3y=-8$ và $y=mx+6$ cùng đi qua một điểm khi điểm $\left(2;-4\right)$ thuộc đường thẳng $y=mx+6\Leftrightarrow-4=2m+6\Leftrightarrow m=-5$.\\
		Vậy $m=-5$.
		
	\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi thử vào lớp 10 năm học 2017-2018, Bình Giang, Hải Dương]%[9D3K6]%[9D2K4]
	\hfill
	\begin{enumerate}
		\item Hai người cùng làm chung một công việc trong vòng $8$ giờ thì xong. Nếu người thứ nhất làm $1$ giờ $30$ phút và người thứ hai làm tiếp $3$ giờ thì được $25\%$ công việc. Hỏi nếu làm riêng một mình thì mỗi người cần bao nhiêu thời gian để hoàn thành công việc?
		\item Tìm $m$ để đồ thị của hàm số $y=2x+m+5$($m$ là tham số) cắt trục tung tại điểm $A$, cắt trục hoành tại điểm $B$ sao cho diện tích tam giác $AOB$ bằng $6$ (với $O$ là gốc tọa độ).
	\end{enumerate}
\loigiai{\hfill
	\begin{enumerate}
		\item Gọi thời gian người thứ nhất làm riêng một mình xong công việc là $x$ (giờ), thời gian người thứ hai làm riêng một mình xong công việc là $y$ (giờ), điều kiện $x>8,\,y>8$.\\
		Trong một giờ: người thứ nhất làm được $\dfrac{1}{x}$ (công việc), người thứ hai làm được $\dfrac{1}{y}$ (công việc), cả hai người cùng làm chung một công việc trong vòng $8$ giờ thì xong nên ta có phương trình: $\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{1}{8}\quad (1)$.\\
		Đổi $1$ giờ $30$ phút = $\dfrac{3}{2}$ giờ. Do người thứ nhất làm $1$ giờ $30$ phút và người thứ hai làm tiếp $3$ giờ thì được $25\%$ công việc nên ta có phương trình: $\dfrac{3}{2x}+\dfrac{3}{y}=\dfrac{1}{4}\quad (2)$.\\
		Từ $(1)$ và $(2)$ ta có hệ phương trình: $\left\{\begin{aligned}&\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{1}{8} \\	& \dfrac{3}{2x}+\dfrac{3}{y}=\dfrac{1}{4}.\end{aligned}\right.$\\
		Đặt $a=\dfrac{1}{x},\, b=\dfrac{1}{y}$ ta có hệ phương trình:
		\begin{eqnarray*}
		&&\left\{\begin{aligned}& a+b=\dfrac{1}{8} \\&\dfrac{3}{2} a+3b=\dfrac{1}{4}\end{aligned}\right.\Leftrightarrow \left\{\begin{aligned}&8a+8b=1\\&6a+12b=1\end{aligned}\right.\\
		&\Leftrightarrow &\left\{\begin{aligned}&24a+24b=3\\& 12a+24b=2\end{aligned}\right.\Leftrightarrow \left\{\begin{aligned}&a=\dfrac{1}{12}\\			&b=\dfrac{1}{24}.\end{aligned}\right.
		\end{eqnarray*}
	Từ đó suy ra $\left\{\begin{aligned}& \dfrac{1}{x}=\dfrac{1}{12}\\& \dfrac{1}{y}=\dfrac{1}{24}\end{aligned}\right.\Leftrightarrow \left\{\begin{aligned}&x=12\\&y=24\end{aligned}\right.$ (thoả mãn).\\
	Vậy thời gian người thứ nhất làm riêng một mình xong công việc là $12$ (giờ), thời gian người thứ hai làm riêng một mình xong công việc là $24$ (giờ).	
		\item Đồ thị của hàm số $y=2x+m+5$ cắt trục tung tại điểm $A(0;m+5)$; đồ thị của hàm số $y=2x+m+5$ cắt trục hoành tại điểm $B\left(\dfrac{-\left(m+5\right)}{2};0\right)$\\
		Khi đó: $OA=\left|m+5\right|$, $OB=\left| \dfrac{-\left(m+5\right)}{2}\right|=\dfrac{\left| m+5\right|}{2}$.\\
		Diện tích tam giác $AOB$ bằng $6$ nên:
		\begin{eqnarray*}
		&\dfrac{1}{2}\cdot OA\cdot OB=&6\Leftrightarrow \dfrac{1}{2}\left| m+5\right|\cdot \dfrac{\left| m+5\right|}{2}=6\\
		&\Leftrightarrow {\left(m+5\right)}^2=&24\Leftrightarrow \left| m+5\right|=2\sqrt{6}\Leftrightarrow \left[\begin{aligned}&m+5=2\sqrt{6}\\&m+5=-2\sqrt{6}.\end{aligned}\right.
		\end{eqnarray*}
		Vậy $m=2\sqrt{6}-5$ hoặc $m=-2\sqrt{6}-5$.
		\end{enumerate}
}
\end{ex}

Cho đường tròn $O$ đường kính $AB=2R$. Điểm $C$ nằm giữa hai điểm $A$ và $B$, vẽ đường tròn $I$ đường kính $CA$ và đường tròn $K$ đường kính $CB$. Qua $C$ kẻ đường thẳng vuông góc với $AB$ cắt đường tròn $O$ tại $D$ và $E$. Đoạn thẳng $DA$ cắt đường tròn $I$ tại $M$, $DB$ cắt đường tròn $K$ tại $N$.
\begin{enumerate}
	\item Chứng minh rằng: Bốn điểm $C,M,D,N$ cùng thuộc một đường tròn.
	\item Chứng minh rằng: $MN$ là tiếp tuyến chung của đường tròn $I$ và đường tròn $K$.
	\item Xác định vị trí điểm C trên đường kính AB sao cho tứ giác $CMDN$ có diện tích lớn nhất. 
\end{enumerate}
\loigiai{
	\begin{center}
		\begin{tikzpicture}[scale=1.2]
		\tkzInit[xmin=-3.5,ymin=-3.5,xmax=3.5,ymax=3.5]
		\tkzClip
		\tkzDefPoints{-3/0/A,0/0/O}

		\tkzInterLC(A,D)(I,A)\tkzGetPoints{}{M}
		\tkzInterLC(B,D)(K,B)\tkzGetPoints{N}{}
		\tkzInterLL(M,N)(D,E)\tkzGetPoint{G}

		\tkzMarkRightAngles(D,C,B)

		\end{center}
	\begin{enumerate}
		\item Ta có $\widehat{ADB}=90^\circ$, $\widehat{AMC}=90^\circ$ (các góc nội tiếp chắn nữa đường tròn) $\Rightarrow \widehat{DMC}=90^\circ$ (bù với $\widehat{AMC}$).\\
		Chứng minh tương tự $\Rightarrow \widehat{BNC}=90^\circ\Rightarrow \widehat{DNC}=90^\circ$.\\
		Tứ giác $CMDN$ có $\widehat{CM D}=\widehat{ADB}=\widehat{CN D}=90^0$ nên $CMDN$ là hình chữ nhật.\\
		Gọi $G$ là giao điểm của $MN$ và $CD$ suy ra $GC=GD=GM=GN\Rightarrow$ bốn điểm $C$, $M$, $D$, $N$ cùng thuộc một đường tròn tâm $G$.
		\item Ta có $\triangle IMC$ cân tại $I$ ($IM=IC$) $\Rightarrow \widehat{IMC}=\widehat{ICM}$ 		và $\triangle GMC$ cân tại $G$ ($GM=GC$).\\ 
		$\Rightarrow \widehat{GMC}=\widehat{GCM}$ $\Rightarrow \widehat{GMC}+\widehat{IMC}=\widehat{ICM}+\widehat{GCM}\Rightarrow \widehat{IMG}=\widehat{ICG}=90^0$.\\
		$\Rightarrow MN$ là tiếp tuyến của đường tròn $(I)$.\\
		Ta có $\triangle KNC$ cân tại $K$ ($KN=KC$) $\Rightarrow \widehat{KNC}=\widehat{KCN}$ và $\triangle GNC$ cân tại $G$ ($GN=GC$).\\
		$\Rightarrow \widehat{GNC}=\widehat{GCN}$.\\
		$\Rightarrow \widehat{GNC}+\widehat{KNC}=\widehat{KCN}+\widehat{GCN}\Rightarrow \widehat{GNK}=\widehat{GCK}=90^\circ$.\\
		$\Rightarrow MN$ là tiếp tuyến của đường tròn $(K)$\\
		$\Rightarrow MN$ là tiếp tuyến chung của hai đường tròn $(I)$ và $(K)$.
		\item $\triangle AC D$ vuông tại $C$ nên $AD\cdot M D=CD^2\Rightarrow MD=\dfrac{CD^2}{AD}$.\\
		$\triangle BCD$ vuông tại $C$ nên $BD\cdot ND=CD^2\Rightarrow ND=\dfrac{CD^2}{BD}$.\\
		$\triangle ABD$ vuông tại $D$ nên $AD\cdot BD=AB\cdot CD$.\\

		Tứ giác $CMDN$ có diện tích lớn nhất bằng $\dfrac{R^2}{2}$ khi $CD=R$ $\Leftrightarrow C$ trùng với $O$.
	\end{enumerate}

\begin{ex}%[Đề thi thử vào lớp 10 năm học 2017-2018, Bình Giang, Hải Dương]%[9D5G1]
Cho hai số thực $x$, $y$ thỏa mãn $x>y$ và $xy=2$. Tìm giá trị nhỏ nhất của biểu thức $$M=\dfrac{2x^2-3xy+2y^2}{x-y}.$$
\loigiai{\hfill.\\
	Nhận xét: Cho hai số dương a, b ta có $a+b-2\sqrt{ab}=\left(\sqrt{a}-\sqrt{b}\right)^2\geq 0\Rightarrow a+b\geq 2\sqrt{ab}$, đẳng thức xảy ra khi $a=b$.\\
$M=\dfrac{2x^2-3xy+2y^2}{x-y}=\dfrac{2(x-y)^2+xy}{x-y}$.\\
Do $x>y$ và $xy=2$ nên $M=2\left((x-y)+\dfrac{1}{x-y}\right)\geq 4\sqrt{(x-y)\dfrac{1}{x-y}}=4$.\\
Đẳng thức xảy ra khi và chỉ khi 
\begin{eqnarray*}
&&\left\{\begin{aligned}&xy=2\\& x-y=\dfrac{1}{x-y}\\&x>y\end{aligned}\right. \Leftrightarrow \left\{\begin{aligned}& xy=2\\& x-y=1\end{aligned}\right.\\
&\Leftrightarrow &\left\{\begin{aligned}& x=y+1\\&y^2+y-2=0\end{aligned}\right. \Leftrightarrow \left[\begin{aligned}
&y=1,x=2\\& y=-2,x=-1.\end{aligned}\right.
\end{eqnarray*}
	Kết luận: $\min M = 4$ khi $\left[\begin{aligned}& x=2,y=1 \\& x=-1,y=-2.\end{aligned}\right.$
	}
\end{ex}
