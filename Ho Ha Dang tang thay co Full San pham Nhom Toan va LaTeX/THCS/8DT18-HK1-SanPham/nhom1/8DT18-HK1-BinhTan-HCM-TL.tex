\end{aligned}$
	\item
$\begin{aligned}[t]
 & x^2 + 2x + 1 - 25y^2 \\

\end{aligned}$
	\item
$\begin{aligned}[t]
& xy - 2x(x+3) + 3y \\
=& xy - 2x^2 - 6x + 3y \\
=& (xy + 3y) - (2x^2 + 6x) \\
=& y(x+3) - 2x(x+3) \\
=& (x+3)(y-2x).
\end{aligned}$
	\end{multicols}
\end{enumerate}
}

Tính và rút gọn.
\begin{enumerate}
	\item $(5x + 3)(2 - x) + (x-1)^2 + 4x^2 - 5x$.
	\item $\left( x^3y^2 - 2x^2y - \dfrac{-2}{3}x^2y^2 \right) \div \left( -\dfrac{1}{2}x^2y \right)$.
	\item $\dfrac{-3}{x} - \dfrac{5}{x+3} + \dfrac{5x}{x^2+3x}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\begin{multicols}{2}
	
	
	\item
$\begin{aligned}[t]
& (5x + 3)(2 - x) + (x-1)^2 + 4x^2 - 5x \\
=& 10x - 5x^2 + 6 - 3x + x^2 - 2x + 1 + 4x^2 - 5x \\
=& 0 + 0 + 7 \\
=& 7.
\end{aligned}$
	\item
$\begin{aligned}[t]
& \left( x^3y^2 - 2x^2y - \dfrac{-2}{3}x^2y^2 \right) \div \left( -\dfrac{1}{2}x^2y \right) \\
=& \left( x^3y^2 - 2x^2y + \dfrac{2}{3}x^2y^2 \right) \div \left( -\dfrac{x^2y}{2} \right) \\
=& \left( \dfrac{x^3y^2 - 2x^2y + 2x^2y^2}{3} \right) \cdot \left( -\dfrac{2}{x^2y} \right) \\
=& -\dfrac{x^2y \cdot (3xy - 6 + 2y)}{3} \cdot \dfrac{2}{x^2y} \\
=& -\dfrac{3xy - 6 + 2y}{3} \cdot 2 \\
=& -\dfrac{2(3xy - 6 + 2y)}{3} \\
=& -\dfrac{6xy - 12 + 4y}{3}.
\end{aligned}$
	\item
$\begin{aligned}[t]
 & \dfrac{-3}{x} - \dfrac{5}{x+3} + \dfrac{5x}{x^2+3x} \\
=& \dfrac{-3}{x} - \dfrac{5}{x+3} + \dfrac{5x}{x(x+3)} \\
=& \dfrac{-3}{x} - \dfrac{5}{x+3} + \dfrac{5}{x+3} \\
=& \dfrac{-3}{x}.
\end{aligned}$
\end{multicols}
\end{enumerate}
}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Bình Tân, TP.Hồ Chí Minh]%[Nguyễn Tuấn Kiệt, dự án 8DT18-HK1]%[8D2B9]
Nhà bạn Nghi chuẩn bị lát gạch tầng trệt ngôi nhà (gồm phòng khách và phòng ăn). Phòng khách là hình chữ nhật có kích thước là $5$ m và $6$ m, phòng ăn cũng là hình chữ nhật có kích thước là $4,5$ m và $4$ m. Tiền gạch lát phòng khách là $300\,000$ đồng/m$^2$; tiền gạch lát phòng ăn là $200\,000$ đồng/m$^2$ và tiền công lát (tính cả vật liệu) là $60\,000$ đồng/m$^2$. Hỏi nhà bạn Nghi phải tốn tổng cộng bao nhiêu tiền để lát gạch hết tầng trệt ngôi nhà?
\loigiai{
Diện tích phòng khách nhà bạn Nghi là
\[S_{PK} = 5 \cdot 6 = 30 \quad \text{(m$^2$)}\]
Diện tích phòng ăn nhà bạn Nghi là
\[S_{PA} = 4,5 \cdot 4 = 18 \quad \text{(m$^2$)}\]
Số tiền cần dùng lát gạch phòng khách là
\[30 \cdot 300\,000 = 9\,000\,000 \quad \text{(đồng)}\]
Số tiền cần dùng lát gạch phòng ăn là
\[18 \cdot 200\,000 = 3\,600\,000 \quad \text{(đồng)}\]
Số tiền công lát gạch của cả ngôi nhà là
\[(30 + 18) \cdot 60\,000 = 2\,880\,000 \quad \text{(đồng)}\]
Tổng số tiền lát gạch ngôi nhà là
\[9\,000\,000 + 3\,600\,000 + 2\,880\,000 = 15\,480\,000 \quad \text{(đồng)}\]
Vậy tổng số tiền nhà Nghi cần chi ra là $15\,480\,000 \quad \text{(đồng)}$.
}

\end{bt}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Bình Tân, TP.Hồ Chí Minh]%[Nguyễn Tuấn Kiệt, dự án 8DT18-HK1]%[8H3B9]
Cho tam giác $ABC$ vuông cân tại $B$, gọi $O$, $E$ lần lượt là trung điểm của $AC$ và $AB$; $D$ là điểm đối xứng của $B$ qua $O$. Trên cạnh $BC$ lấy điểm $M$ và trên tia đối của tia $DC$ lấy điểm $N$ sao cho $DN = BM$.
\begin{enumerate}
	\item Chứng minh tứ giác $BCOE$ là hình thang vuông.
	\item Tứ giác $ABCD$ là hình gì? Vì sao?
	\item Chứng minh tam giác $MAN$ là tam giác vuông cân.
	\item Chứng minh $BD$ đi qua trung điểm của $MN$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[>=stealth,scale=1, line join = round, line cap = round]
\tkzInit[ymin=-0.5,ymax=4.6,xmin=-0.7,xmax=5.5]

\tkzLabelPoints[left](B)
\tkzLabelPoints[right](A)
\pgfmathsetmacro{\so}{4} % lấy số

\tkzInterLC[R](B,b)(B,\so cm)\tkzGetSecondPoint{C}\tkzLabelPoint[above](C){$C$}
\tkzDefMidPoint(A,C)\tkzGetPoint{O}\tkzLabelPoint[right](O){$O$}
\tkzDefMidPoint(A,B)\tkzGetPoint{E}\tkzLabelPoint[below](E){$E$}
\tkzDefPointBy[symmetry=center O](B)\tkzGetPoint{D}
\tkzLabelPoint[above](D){$D$}
\tkzDefBarycentricPoint(B=2,C=1)
\tkzGetPoint{M}
\tkzLabelPoint[left](M){$M$}
\tkzCalcLength[cm](B,M)\tkzGetLength{rBM}

\begin{enumerate}
	\item Xét tam giác $ABC$ vuông tại $B$ có $EO$ là đường trung bình, suy ra
\[EO \parallel BC\]
Lại có $BC \perp BE$ nên tứ giác $BCOE$ là hình thang vuông.
	\item Xét tam giác vuông cân $ABC$ tại $B$ có $BO$ là đường trung tuyến cũng là đường cao, suy ra
\[BO \perp AC\]
Lại có $IB = ID$ do tính chất đối xứng tâm.\\
Xét tứ giác $ABCD$ có $AC$, $BD$ là hai đường chéo vuông góc với nhau tại trung điểm mỗi đường, suy ra $ABCD$ là hình thoi.\\
Mặt khác, $\widehat{ABC}=90^\circ$. Vậy $ABCD$ là hình vuông.
	\item Dễ dàng chứng minh được
\[\triangle{ABM} = \triangle{ADN}\]
theo trường hợp (c.g.c), suy ra
\[MA = NA\]
suy ra tam giác $MAN$ là tam giác cân tại $A$.\\
Ta có $\widehat{BAM} + \widehat{MAD} = 90^\circ$.\\
Lại có $\widehat{BAM} = \widehat{DAN}$, suy ra
\[\widehat{MAN} = \widehat{DAN} + \widehat{MAD} = 90^\circ\]
Vậy tam giác $MAN$ vuông cân tại $A$.
	\item Gọi $I$ là trung điểm của đoạn $MN$.\\
	Xét tam giác $MAN$ vuông cân tại $A$ có $AI$ là đường trung tuyến, suy ra
	\[AI = MI\]
	Xét tam giác $MCN$ vuông tại $C$ có $CI$ là đường trung tuyến, suy ra
	\[CI = MI\]
	Từ đó suy ra $AI = CI$. Vậy tam giác $CIA$ cân tại $I$ với mọi sự thay đổi từ điểm $M$.\\
	Suy ra điểm $I$ thuộc đường trung trực của đoạn $AC$.\\
	Hay điểm $I$ thuộc đoạn $BD$.
\end{enumerate}
}
