\section{Xác suất của biến cố}

\subsection{Tóm tắt lí thuyết}

\subsubsection{Định nghĩa cổ điển của xác suất}
\begin{dn}
	Xác suất của biến cố $A$ được tính bởi công thức:
	\begin{center}
		$\mathrm{P}(A)=\dfrac{n(A)}{n(\Omega)}$.
	\end{center}
	Trong đó $n(A)$ là số kết quả thuận lợi của biến cố $A$; $n(\Omega)$ là số kết quả có thể xảy ra của phép thử.
\end{dn}

\subsubsection{Tính chất của xác suất}
	\begin{dl}
	Giả sử $A$ và $B$ là các biến cố liên quan đến một phép thử có một số hữu hạn kết quả đồng khả năng xuất hiện. Khi đó, ta có
	\begin{enumerate}
		\item $\mathrm{P}(\varnothing)=0$, $\mathrm{P}(\Omega)=1$.
		\item $0\le \mathrm{P}(A)\le 1$, với mọi biến cố $A$.
		\item Nếu $A$ và $B$ xung khắc, thì
		$$\mathrm{P}(A\cup B)=\mathrm{P}(A)+\mathrm{P}(B)$$
		\hfill \textit{(công thức cộng xác suất).}
	\end{enumerate}
\end{dl}
\begin{note}
	Các biến cố $A$ và $B$ là xung khắc nếu và chỉ nếu chúng không khi nào cùng xảy ra.
\end{note}

\begin{hq}
	Với mọi biến cố $A$, ta có 
	$$\mathrm{P}\left(\overline{A}\right)=1-\mathrm{P}(A).$$
\end{hq}


\subsubsection{Các biến cố độc lập, công thức nhân xác suất}
\begin{kn}
	Trong một phép thử, nếu sự xảy ra của biến cố này không ảnh hưởng đến xác suất xảy ra của một biến cố khác thì ta nói hai biến cố đó \emph{độc lập}.
\end{kn}
\begin{vd}%\label{vd1_NXS}
	Có hai bạn, một người có đồng tiền và một người có súc sắc. Xét phép thử ``Bạn thứ nhất gieo đồng tiền, sau đó bạn thứ hai gieo súc sắc''. Gọi $A$ là biến cố ``Đồng tiền xuất hiện mặt sấp'' và $B$ là biến cố ``Con súc sắc xuất hiện mặt 6 chấm''. Khi đó sự xảy ra của biến cố $A$ rõ ràng không ảnh hưởng tới xác suất xảy ra của biến cố $B$. Do đó biến cố $A$ và $B$ là độc lập.
\end{vd}
\begin{tc}
	Với hai biến cố bất kỳ, ta có mối quan hệ sau (công thức nhân xác suất):
	$$A \text{ và } B \text{ là hai biến cố độc lập}\Leftrightarrow \mathrm{P}(A.B)=\mathrm{P}(A).\mathrm{P}(B).$$
\end{tc}

\subsubsection{Xác suất điều kiện}

\begin{dn}
	Xác suất có điều kiện của biến cố $A$ với điều kiện $B$ là một số được xác định bởi công thức
	$$\mathrm{P}(A|B)=\dfrac{\mathrm{P}(AB)}{\mathrm{P}(B)}\ \text{nếu}\ \mathrm{P}(B)>0.$$
\end{dn}


\begin{tc}\hfill
	\begin{enumerate}
		\item $\mathrm{P}(A|B)\ge 0$.
		\item $\mathrm{P}(\Omega|B)=\mathrm{P}(B|B)=1$,
		\item Nếu $A_i$, $i=1,\ldots,n$ là các biến cố đôi một xung khắc thì
		$\mathrm{P}\left(\bigcup\limits_{i=1}^n A_i | B\right)=\displaystyle\sum\limits_{i=1}^n\mathrm{P}(A_i|B)$.
		\item (Công thức nhân xác suất) $\mathrm{P}(AB)=\mathrm{P}(B)\mathrm{P}(A|B)=\mathrm{P}(A)\mathrm{P}(B|A)$.
	\end{enumerate}
\end{tc}

\begin{note}
	Xác suất điều kiện cho phép tính xác suất xảy ra của một biến cố khi biến cố khác đã xảy ra. Trong trường hợp hai biến cố $A$ và $B$ độc lập thì việc biến cố $B$ xảy ra không ảnh hưởng gì tới việc xảy ra biến cố $A$ nên $\mathrm{P}(A|B)=\mathrm{P}(A)$. Ta được công thức nhân xác suất thông thường.
\end{note}

\begin{dl}[Công thức xác suất toàn phần]
	Nếu $B_i$, $i=1,\ldots,n$, là hệ các biến cố đôi một xung khắc sao cho $\bigcup\limits_{i=1}^n B_i=\Omega$ thì với biến cố $A$ bất kì ta luôn có
	$$\mathrm{P}(A)=\displaystyle\sum\limits_{i=1}^n \mathrm{P}(B_i)\mathrm{P}(A|B_i).$$
	Hệ các biến cố $B_i$ ($i=1,\ldots,n$) như vậy được gọi là hệ đầy đủ.
\end{dl}

\begin{dl}[Công thức Bayes]
	Cho biến cố $A$ và hệ đầy đủ $B_i$ ($i=1,\ldots,n$) đều có xác suất dương. Khi đó
	$$\mathrm{P}(B_i|A)=\dfrac{\mathrm{P}(B_i)\mathrm{P}(A|B_i)}{\mathrm{P}(A)}=\dfrac{\mathrm{P}(B_i)\mathrm{P}(A|B_i)}{\displaystyle\sum\limits_{i=1}^n \mathrm{P}(B_i)\mathrm{P}(A|B_i)}.$$
\end{dl}

		\item $C$: "Tổng $3$ lần gieo là $5$".

	\end{enumerate}
	\loigiai{
		Gieo súc sắc $3$ lần sẽ có $n(\Omega)=6.6.6=216$ cách có thể xảy ra. Ta gọi kết quả $3$ lần gieo lần lượt theo dãy số $(x;y;z)$.

			\item $C=\{(1;1;3),(1;2;2),(1;3;1),;(2;1;2),(2;2;1),(3;1;1)\}$.\\ Do đó $n(C)=6\Rightarrow \mathrm{P}(C)=\dfrac{n(C)}{n(\Omega)}=\dfrac{1}{36}$.
			\item Kết quả thuận lợi của $D$ là việc chọn ra $3$ giá trị khác nhau trong $3$ lần gieo và sắp xếp từ bé đến lớn tức là số kết quả bằng số việc chon $3$ phần tử trong $6$ phần tử, vậy\\ $n(D)=C^3_6=20\Rightarrow \mathrm{P}(C)=\dfrac{n(C)}{n(\Omega)}=\dfrac{5}{54}$.

	}
\end{vd}

	\begin{dang}{Tính xác suất theo quy tắc cộng}
	{\textbf{Phương pháp giải:}}
	\begin{itemize}
		\item Sử dụng định nghĩa biến cố xung khắc, biến cố đối.
		\item Áp dụng các công thức
		\begin{enumerate}
			\item Nếu $A$ và $B$ xung khắc thì $\mathrm{P}(A\cup B)=\mathrm{P}(A)+\mathrm{P}(B)$.

		\end{enumerate}
	\end{itemize}

\begin{vd}%%[Đào-V- Thủy]%[1D2B5]

	\begin{enumerate}
		\item Tính xác suất để thu được hai viên bi cùng màu.
		\item Tính xác suất để thu được hai viên bi khác màu.
	\end{enumerate}
	\loigiai{
		Ta có $n(\Omega)=C_{10}^2$.
		\begin{enumerate}
			\item Gọi $A$ là biến cố: \lq\lq Lấy được hai viên bi cùng màu\rq\rq.\\
			$A_1$ là biến cố: \lq\lq Lấy được hai viên bi cùng màu xanh\rq\rq.\, $\Rightarrow \mathrm{P}(A_1)=\dfrac{C_6^2}{C_{10}^2}$.\\
			$A_2$ là biến cố: \lq\lq Lấy được hai viên bi cùng màu đỏ\rq\rq.
			\, $\Rightarrow \mathrm{P}(A_2)=\dfrac{C_4^2}{C_{10}^2}$.\\
			Vậy $\mathrm{P}(A)=\mathrm{P}(A_1\cup A_2)=\mathrm{P}(A_1)+\mathrm{P}(A_2)=\dfrac{C_6^2+C_4^2}{C_{10}^2}=\dfrac{7}{15}$.
			\item Gọi $B$ là biến cố: \lq\lq Lấy được hai viên bi khác màu\rq\rq.\\
			Vì chỉ có hai màu xanh hoặc đỏ nên ta có $B=\overline{A}$.\\
			Vậy $\mathrm{P}(B)=\mathrm{P}\left(\overline{A}\right)=1-\mathrm{P}(A)=\dfrac{8}{15}$.
		\end{enumerate}
	}
\end{vd}

			$$\mathrm{P}(AB)=\dfrac{1}{6}\left(\dfrac{4}{6}\right)^{k-1}\Rightarrow \mathrm{P}(B|A)=\dfrac{\mathrm{P}(AB)}{\mathrm{A}}=\left(\dfrac{4}{5}\right)^{k-1}.$$
			\item Gọi $C$ là biến cố mặt "bốn" xuất hiện trước mặt "ba", $C_1$, $C_2$, $C_3$ lần lượt là các biến cố "lần đầu ra mặt bốn", "lần đầu ra mặt ba", "lần đầu không ra cả mặt ba và bốn". Khi đó $C_1$, $C_2$, $C_3$ là hệ biến cố đầy đủ. Suy ra
			\begin{align*}
			\mathrm{P}(C)&=\mathrm{P}(C_1)\mathrm{P}(C|C_1)+ \mathrm{P}(C_2)\mathrm{P}(C|C_2) + \mathrm{P}(C_3)\mathrm{P}(C|C_3)\\
			&=\dfrac{1}{6}\cdot 1 + \dfrac{1}{6}\cdot 0 + \dfrac{4}{6} \cdot \mathrm{P}(C)\\
			\Rightarrow &\mathrm{P}(C)=\dfrac{1}{2}.
			\end{align*}
		\end{enumerate}
	}
\end{bt}

		Gọi $A$ là biến cố "Chọn được một số thỏa mãn $5$ chữ số phân biệt nhỏ hơn $45000$".\\
		Gọi số có $5$ chữ số được lập từ các số $1$, $2$, $3$, $4$, $5$ là $\overline{a_{1}a_{2}a_{3}a_{4}a_{5}}$ ($a_{1},a_{2},a_{3},a_{4},a_{5}\in \{1;2;3;4;5\}$)
		\begin{itemize}
			\item $a_{1}<4$:
			\begin{itemize}
				\item Vị trí $a_{1}$ có $3$ cách chọn.
				\item Số cách xếp $4$ số vào $4$ vị trí còn lại là $4!=24$ cách.
			\end{itemize}
			\item $a_{1}=4$, $a_{2}<5$:
			\begin{itemize}
				\item Vị trí $a_{1}$ có $1$ cách chọn.
				\item Vị trí $a_{2}$ có $3$ cách chọn.
				\item Số cách xếp $3$ số vào $3$ vị trí còn lại là $3!=6$ cách.
			\end{itemize}
		\end{itemize}
		\[
		\mathrm{P}(A)=\dfrac{90}{120}=\dfrac{3}{4}
		\]
	}
\end{bt}
\begin{bt}%[Dương BùiĐức]%[1D2G5]
	Gọi $X$ là tập hợp các số tự nhiên có $3$ chữ số phân biệt được lập từ các số $0$, $1$, $2$, $3$, $4$, $5$. Chọn ngẫu nhiên một số từ $X$. Tính xác suất để có thể chọn được một số không chia hết cho $3$.
	\loigiai{
		Số lượng phần tử của $X$ là $5.5.4=100$.\\
		Gọi $A$ là biến cố "Chọn được một số không chia hết cho $3$".\\
		Dấu hiệu chia hết cho $3$ là tổng các chữ số là một số chia hết cho $3$ nên ta xét các trường hợp sau.
		\begin{itemize}
			\item Tổng các chữ số bằng $12\Rightarrow$ xét tập các chữ số $\{3;4;5\}$: có $3!=6$ số.
			\item Tổng các chữ số bằng $9$:
			\begin{itemize}
				\item Xét tập các chữ số $\{0;4;5\}$: có $4$ số.
				\item Xét tập các chữ số $\{1;3;5\}$: có $6$ số.
				\item Xét tập các chữ số $\{2;3;4\}$: có $6$ số.
			\end{itemize}
			\item Tổng các chữ số bằng $6$:
			\begin{itemize}
				\item Xét tập các chữ số $\{0;1;5\}$: có $4$ số.
				\item Xét tập các chữ số $\{0;2;4\}$: có $4$ số.
				\item Xét tập các chữ số $\{1;2;3\}$: có $6$ số.
			\end{itemize}
			\item Tổng các chữ số bằng $3\Rightarrow$ xét tập các chữ số $\{0;1;2\}$: có $4$ số.
		\end{itemize}
		\[
		\mathrm{P}(A)=1-\dfrac{40}{100}=0,6
		\]
	}
\end{bt}
\begin{bt}%[Dương BùiĐức]%[1D2G5]
	Trong một kì thi một học sinh làm bài thi trắc nghiệm gồm $50$ câu (trong đó có $1$ câu đúng và $3$ câu sai ) và chấm theo thang điểm sau: mỗi câu đúng được $0,2$ điểm và mỗi câu sai bị trừ $0,05$ điểm. Một học sinh đó vì không học bài nên chọn đáp án ngẫu nhiên. Tính xác xuất để học sinh đó được $4,5$ điểm, biết cậu học sinh đó không bỏ khoanh câu nào.
	\loigiai{
		Gọi $a$, $b$ lần lượt là số câu đúng và số câu sai được chọn $\Rightarrow \heva{&a+b=50\\ &0,2a-0,05b=4,5}\Rightarrow \heva{&a=28\\ &b=22}$.\\
		Vì xác suất để chọn được một câu đúng là $0,25$ và xác suất chọn được một câu sai là $0,75$ nên xác suất để học sinh đó được $4,5$ điểm là $2,197.10^{-6}$.
	}
\end{bt}
