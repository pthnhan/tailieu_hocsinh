\begin{vd}
	\begin{multicols}{2}
		\textit{Một trăm con trâu,}\\
		\textit{Một trăm bó cỏ.} \\
		\textit{Trâu đứng ăn năm,}\\
		\textit{Trâu nằm ăn ba,} \\
		\textit{Ba con trâu già}\\
		\textit{Chỉ ăn một bó.}\\
		\textit{Tính xem trong đó}\\
		\textit{Mỗi loại mấy trâu?}
	\end{multicols}
	\loigiai{
	Gọi số trâu đứng là $x$, số trâu nằm là $y$, số trâu già là $z$. Ta có hệ hai phương trình ba ẩn với nghiệm nguyên dương 
	$$\heva{&x + y + z = 100 \\ &5x + 3y + \dfrac{z}{3} = 100 } \Leftrightarrow \heva{&x + y + z = 100 \\ & 15x + 9y + z = 300.}$$
	Suy ra $14x + 8y = 200 \Leftrightarrow 7x + 4y = 100$.\\
	Ta thấy $7x ~\vdots~ 4$ nên $x ~\vdots~ 4$. Đặt $x = 4k$ ($k$ là số nguyên dương), ta có \\
	$7k + y = 25 \Leftrightarrow y = 25 - 7k$ \\
	$z = 100 - 4k - ( 25 - 7k) = 75 + 3k$.\\
	Ta phải có: $\heva{&x = 4k > 0 \\&y = 25 - 7k > 0 \\ & z = 75 + 3k } \Leftrightarrow 0 < k < 3\dfrac{4}{7}$.\\
	Có ba đáp số: $4$ trâu đứng, $18$ trâu nằm, $78$ trâu già; \\
	$8$ trâu đứng, $11$ trâu nằm, $81$ trâu già; \\	
	$12$ trâu đứng, $4$ trâu nằm, $84$ trâu già;\\
	\textbf{\textit{Lưu ý.}} Sách Đại thành toán pháp của Lương Thế Vinh từ thế kỉ XV đã có bài toán \textit{Trăm trâu trăm cỏ} nói trên.
	}
\end{vd}
\begin{vd} Anh Tâm và bác Đức là hai công nhân của một xí nghiệp. Một ngày đầu năm $1999$, bác Đức hỏi anh Tâm:\\
- Năm nay cháu bao nhiêu tuổi rồi?\\
Anh Tâm hóm hỉnh trả lời: \\
- Tuổi cháu năm nay đúng bằng tổng các chữ số của năm sinh. Bác Đức tính ra ngay tuổi của anh Tâm. Bác gật gù nói: \\
- Lúc tuổi bác bằng tuổi cháu hiện nay, bác đang là chiến sĩ quân giải phóng miền Nam và cũng có tuổi bằng tổng các chư số của năm sinh. \\
Anh Tâm cũng tính đúng tuổi của bác Đức. \\
Hãy tính xem anh Tâm và bác Đức sinh năm nào?
\loigiai{
	Gọi năm sinh của anh Tâm là $\overline{19xy}$, ta có \\
	$$1999 - \overline{19xy} = 1 + 9 + x + y \Leftrightarrow 99 - (10x + y ) = 10 + x + y $$
	$$\Leftrightarrow 89 = 11x + 2y$$
	Ta thấy $0 \leq 2y \leq 18$ nên $71 \leq 11x \leq 89$, do đó $7 \leq x \leq 8$.\\
	$\bullet$ $x = 7$ thì $y = 6$.\\
	$\bullet$ $x = 8$ thì $2y = 1$, loại.\\
	Anh Tâm sinh năm $1976$, đến năm $1999$ thì $23$ tuổi. \\
	Bác Đức $23$ tuổi năm $\overline{19ab}$ và $ 1 + 9 + a + b = 23$. \\
	Do đó $a + b = 13$. Vì $1960 \leq \overline{19ab} \leq 1975 $ nên $\overline{ab} = 67$.\\
	Bác Đức sinh năm $ 1967 - 23 = 1944$.\\
	\textbf{\textit{Kinh nghiệm giải toán}} \\
	Có thể giải phương trình $11x + 2y = 89$ theo cách giải phương trình nghiệm nguyên được $(x;y)$ là $(2t + 1 ; 39 - 11t)$ với $t$ nguyên, kết hợp với $2t + 1 \geq 0$ và $39 - 22t \geq 0$ được $(x;y)$ là $(1;39)$, $(3; 28)$, $(5; 17)$, $(7;6)$. Do đó $x$ và $y$ là các chữ số nên $x = 7$ và $y = 6$. \\
	Cách giải này dài do không tận dụng được $x$ và $y$ là các số tự nhiên nhỏ hơn $10$ ngay từ ban đầu.
}
\end{vd}

\begin{bt}
	Ngày đầu năm mới, Thủy tính tuổi của mẹ và của mình, chợt phát hiện ra:\\
	- Mẹ ơi, tổng các chữ số của tuổi mẹ đúng bằng tuổi con. \\
	Mẹ Thủy hỏi lại:\\
	- Thế tổng các chữ số của tuổi con bằng tuổi ai?\\
	- A, đúng bằng tuổi của em Tuấn! \\
	Tuổi của mỗi người là bao nhiêu, nếu tuổi của ba mẹ con cộng lại là $54$.
	\loigiai{
	Gọi tuổi của mẹ Thủy là $\overline{ab}$ thì tuổi của Thủy là $a + b$. Tuổi của Thủy là một số có hai chữ số, vì nếu là một số có một chữ số thì tuổi của Thủy bằng tuổi của bé Tuấn. \\
	Ta lại có $a + b \leq 18$ nên Thủy $\overline{1m}$ tuổi, bé Tuấn $1 + m$ tuổi, ít hơn Thủy $9$ tuổi. Tuổi của bé Tuấn là $a + b - 9$. Ta có $\overline{ab} + (a + b) + (a + b - 9) = 54$. Rút gọn được $4a + b = 21$. Xét các trường hợp: \\
	$a = 3$ và $b = 9$, thỏa mãn $a + b \geq 10$.\\
	$a = 4$ và $b = 5$, loại vì $a + b < 10$. \\
	$a = 5$ và $b = 1$, loại vì $a + b < 10$. \\
	Đáp số: Mẹ của Thủy $39$ tuổi, Thủy $12$ tuổi, bé Tuấn $3$ tuổi.
	}
\end{bt}
\begin{bt}
	Nhân dịp Tết, các cụ phụ lão, các anh chị thanh niên và các em thiếu nhi tất cả gồm $15$ người, mang $50$ chiếc bánh chưng đến tặng đơn vị bộ đội. Mỗi cụ phụ lão mang $4$ chiếc bánh, mỗi anh chị thanh niên mang $6$ chiếc bánh, mỗi em thiếu nhi mang $1$ chiếc bánh. Có bao nhiêu phụ lão, thanh niên, thiếu nhi?
	\loigiai{
	Gọi số phụ lão, số thanh niên, số thiếu nhi theo thứ tự là $x$, $y$, $z$. Theo đề bài thì $x$, $y$, $z$ là các số nguyên lớn hơn $1$. Giải hệ phương trình 
	$ \heva{&x + y + z = 15 \\ &4x + 6y + z = 50}$ với nghiệm nguyên lớn hơn $1$.\\
	Đáp số: Có $5$ phụ lão, $4$ thanh niên, $6$ thiếu nhi.
	}
\end{bt}
\begin{bt}
	Các bạn Tuấn, Hùng, Cường cùng với các anh chị của mình là Mai, Vân, Nga (không nhất thiết viết đúng thứ tự) dạo chơi hội hoa xuân. Cô bán hàng nói với họ rằng ai mua bông hoa nào giá bao nhiêu nghìn (giá mỗi bông hoa là một số nguyên nghìn) thì phải mua từng ấy bông hoa đó. Tính ra mỗi bạn đều mua ít hơn chị của mình là $48$ nghìn đồng. Ngoài ra Tuấn mua ít hơn chị Vân $9$ bông, Hùng mua ít hơn chị Mai $7$ bông. \\
	Hãy xác định các cặp chị em và tính xem mỗi người mua bao nhiêu bông hoa, biết rằng mỗi người chỉ mua một loại hoa?
	\loigiai{
	Nếu một bạn nào đó mua $x$ bông hòa thì phải trả $x^2$ (nghìn đồng). Chị của bạn đó m$y$ bông hoa thì phải trả $y^2$ (nghìn đồng). Ta có $y^2 - x^2 = 48$. Giải phương trình trên với nghiệm nguyên dương ta được $\heva{&x_1 = 11 \\&y_1 = 13}$  $\heva{&x_2 = 4 \\&y_2 = 8}$  $\heva{&x_3 = 1 \\&y_3 = 7}$. \\
	Tuấn mua ít hơn chị Vân $9$ bông nên Tuấn mua $4$ bông, chị Vân mua $1$ bông, Hùng mua ít hơn chị Mai $7$ bông nên Hùng mua $1$ bông, chị Mai mua $8$ bông. Còn lại Cường mua $11$ bông, chị Nga mua $7$ bông. \\
	Các cặp chị em là: Vân - Cường, Mai - Tuấn, Nga - Hùng.
	}
\end{bt}
\begin{bt}
	Tân và Hùng gặp nhau trong hội nghị học sinh giỏi Toán. Tân hỏi số nhà của Hùng. Hùng trả lời: \\
	- Nhà mình ở chính giữa đoạn phố, đoạn phố ấy có tổng các số nhà bằng $161$, và không có số nhà nào đánh $a$, $b$,... \\
	Nghĩ một chút, Tân nói: \\
	- Bạn ở số nhà 23 chứ gì! \\
	Hỏi Tân đã tìm ra như thế nào?
	\loigiai{
	\textbf{Cách 1:} Gọi $n$ là số nhà của dãy $x + 2, x + 4, x + 6, ..., x + 2n$.\\
	Ta có $(x + 2) + (x + 4) + ... + (x + 2n) = 161 \Leftrightarrow \dfrac{(x + 2 + x + 2n) \cdot n}{2} = 161$ \\
	$\Leftrightarrow(x + n + 1) \cdot  n = 161$. \\
	Ta có $x + n + 1 > n > 1$ nên $\heva{&x + n + 1 = 23 \\ &n = 7} \Leftrightarrow \heva{&x = 15 \\ & n = 7}$.\\
	Các số nhà của đoạn phố là $17,19,21,23,25,27,29$. \\
	Số nhà ở chính giữa là $23$. \\
	\textbf{Cách 2:} Gọi số nhà của Hùng (ở chính giữa đoạn phố) là $x$. Trung bình cộng của hai số nhà cách đều nhà Hùng cũng bằng $x$. Gọi số các số nhà là $n$ thì $x \cdot n = 161$. \\
	$161$ có bốn ước tự nhiên là $1, 7, 23, 161$. \\
	Loại các trường hợp: \\
	Có $1$ nhà, số nhà là $161$;\\
	Có $23$ nhà, số nhà ở chính giữa là $7$; \\
	Có $161$ nhà, số nhà ở chính giữa là $1$; \\
	Còn một trường hợp: Có $7$ số nhà, số nhà ở chính giữa là $23$.
	
	}
\end{bt}
\begin{bt}
	Một hình chữ nhật có chiều dài và chiều rộng là các số nguyên dương khác nhau và số đo chu vi (tính bằng mét) bằng số đo diện tích (tính bằng mét vuông). Tìm chiều dài và chiều rộng của hình chữ nhật.
	\loigiai{
	Gọi chiều dài và chiều rộng của hình chữ nhật theo thứ tự là $x$ và $y$ (mét). \\
	Ta có $2(x + y) = xy \Leftrightarrow xy - 2x - 2y = 0 \Leftrightarrow (x - 2)(y - 2) = 4$.\\
	Ta tìm được $x = 6$, $y = 3$.\\
	Chiều dài là $6$m, chiều rộng là $3$m. 
	}
\end{bt}
\begin{bt}
	Tìm hai số nguyên dương, biết rằng tổng của chúng, hiệu của chúng (số lớn trừ số nhỏ), tích của chúng, thương của chúng ( số lớn chia số nhỏ) cộng lại bằng $50$.
	\loigiai{
	Gọi số lớn và số nhỏ theo thứ tự là $x$ và $y$. Ta có \\
	$$(x + y) + (x - y) + xy + \dfrac{x}{y} = 50 \Leftrightarrow x(y + 1)^2 = 50y.$$
	Ta có $50 \vdots (y + 1)^2$, mà $y$ và $y + 1$ nguyên tố cùng nhau nên $50 \vdots (y + 1)^2$. Ta có bảng giá trị sau
	\begin{center}
		{\renewcommand\arraystretch{2}
		\begin{longtable}{|c|c|c|c|}
		\hline $(y+1)^2$ & $1$ & $25$ \\ 
		\hline $y+1$ & $1$ & $5$ \\
		\hline $y$ & $0$, loại & $4$\\
		\hline
		\end{longtable}}
	\end{center}
	Với $y = 4$ thì $x = 8$.
	}
\end{bt}
\begin{bt}
	Ba người bạn đi câu được một số cá. Buổi tối họ ngủ lại bên bờ sông. Nữa đêm, người thứ nhất thức dậy, muốn về trước, thấy số cá chia cho $3$ dư $1$ nên quẳng một con xuống nước, lấy một phần ba mang về. Người thứ hai thức dậy tưởng hai người bạn còn ngủ, đếm thấy số cá chia $3$ dư $1$ nên cũng vứt một con xuống nước, lấy một phần ba mang về. Người thứ ba thức dậy, cũng vứt một con cá xuống nước, lấy một phần ba mang về. \\
	Hỏi cả ba người câu được bao nhiêu con cá, biết rằng họ là những người câu cá tồi?
	\loigiai{
	\textbf{Cách 1:} Gọi $x$ là số cá cả ba người câu được. \\
	Số cá còn lại sau khi người I lấy là $\dfrac{2}{3}(x - 1)$ tức là $\dfrac{2}{3}x - \dfrac{2}{3}$.\\
	Số cá còn lại sau khi người II lấy là $\dfrac{2}{3} \left(\dfrac{2}{3} - \dfrac{2}{3} - 1 \right)$ tức là $\dfrac{4}{9}x - \dfrac{10}{9}$. \\
	Số cá còn lại sau khi người III lấy là $\dfrac{2}{3} \left( \dfrac{4}{9}x - \dfrac{10}{9} - 1 \right)$ tức là $\dfrac{8}{27}x - \dfrac{38}{27}$. \\
	Gọi số cá còn lại sau khi người III lấy là $y$. \\
	Ta phải giải phương trình sau với nghiệm nguyên dương $\dfrac{8}{27}x - \dfrac{38}{27} = y$ tức là $8x - 27y = 38$ \hfill(1). \\
	Do $y \vdots 2$, đặt $y = 2k \left(k \in \mathbb{N}^{*} \right)$. Thay vào $(1)$ được $4x - 27k = 19 \Leftrightarrow x = \dfrac{27k + 19}{4} = 7k + 5 - \dfrac{k + 1}{4}$.\\
	Đặt $\dfrac{k + 1}{4} = t \left(t \in \mathbb{N}^{*} \right)$ thì $k = 4t - 1 \Rightarrow x = 7 \left(4t - 1 \right) + 5 - t = 27t - 2$. \\
	Vì ba chàng câu cá tồi, chọn $t = 1$ để $x$ nhận giá trị nguyên dương và nhỏ nhất, ta có $x = 25$.\\
	Ba chàng câu được $25$ con cá.\\
	Số cá họ mang về theo thứ tự là $8$ con, $5$ con, $3$ con.\\
	\textbf{Cách 2:} Gọi $x$ là số cá câu được. Gọi $a, b, c$ theo thứ tự là số cá người I, người II, người III mang về. Ta có hệ phương trình \\
	$$\heva{&x - 1 = 3a &(1) \\ &2a - 1 = 3b &(2) \\ &2b - 1 = 3c &(3)}$$
	Từ $(3)$ suy ra $c$ lẻ. Đặt  $c = 2k + 1 ( k \in \mathbb{N})$. \\
	Thay vào $(3)$ được $b = 3k + 2$.\\
	Thay vào $(2)$ được $2a = 9k + 7$.\\
	Đặt $k = 2t + 1 (t \in \mathbb{N})$ thì $2a \approx 18t + 16$ nên $a = 9t + 8$.\\
	Thay vào $(1)$ được $x = 27t + 25$.\\
	Vì ba chàng câu cá tồi, chọn $t = 0$, số cá họ câu là $25$ con. 
	}
\end{bt}
\begin{bt}
	Trong một đợt thi đua, An làm vượt mức $10$ sản phẩm, Bách vượt mức $13$ sản phẩm, Dũng vượt mức $26$ sản phẩm. Số sản phẩm vượt mức của mỗi người gồm loại I và loại II. Số sản phẩm vượt mức của ba người khác nhau nhưng do sản phẩm loại II được thưởng ít tiền hơn loại I nên ai cũng được thưởng $140$ nghìn đồng. Tính xem mỗi người làm vượt mức bao nhiêu sản phẩm từng loại và tiền thưởng cho một sản phẩm mỗi loại bao nhiêu?
	\loigiai{
	Gọi số sản phẩm làm vượt mức loại I của An, Bách, Dũng theo thứ tự là $x$, $y$, $z$ chiếc $(1 \leq x < 10; 1 \leq y < 16; 1 \leq z < 26)$ và $x, y, z \in \mathbb{N}^{*}$, tiền thưởng cho một sản phẩm loại I là $t$ nghìn đồng, loại II là $u$ nghìn đồng. \\
	Ta có hệ phương trình 
	$$\heva{&tx + u(10 - x) = 140 &(1) \\&ty + u(16 - y) = 140 &(2) \\ &tz + u(26 - z) = 140 &(3)} \Leftrightarrow \heva{&(t - u)x + 10u = 140 \\&(t - u)y + 16u = 140 \\& (t - u)z + 26u = 140}$$
	Lấy hai phương trình trên lần lượt trừ đi phương trình cuối được 
	$$\heva{&(t - u)(x - z) = 16u &(4) \\ &(t - u)(y - z) = 10u &(5)}$$
	Do đó $\dfrac{x - z}{y - z} = \dfrac{8}{5}$    (6)  \\
	Suy ra $(x - z) \vdots 8$. Theo đề bài, $t > u$ nên từ $(4)$ suy ra $x > z$. \\
	Ta lại có $1 \leq x \leq 9$ nên $1 \leq x - z \leq 8$. Do đó $x - z = 8$. \\
	Vậy $x = 9$; $z = 1$. Thay vào $(6)$ được $y = 6$. \\
	Thay các giá trị của $x$ và $z$ vào $(1)$ và $(3)$ được $\heva{&9t + u = 140 \\ &t + 25u = 140}$.\\
	Hệ phương trình này cho $t = 15$; $u = 5$.
	}
\end{bt}
\begin{bt}
	Hiện tại là $0$ giờ, các kim giờ, kim phút, kim giây đều chập nhau. Ngoài thời điểm trên, trong khoảng từ $0$ giờ đến $12$ giờ, còn các thời điểm nào để:
	\begin{enumerate}
		\item	Kim giờ và kim phút chập nhau?
		\item	Cả ba kim đều chập nhau?
	\end{enumerate}
	\loigiai{
	\begin{itemize}
	\item	[a)] Trong một giờ, kim phút quay được $1$ vòng, kim giờ quay được $\dfrac{1}{12}$ vòng. \\
	Khoảng thời gian để kim phút gặp lại kim giờ lần tiếp theo là $1: \left(1 - \dfrac{1}{12} \right) = \dfrac{12}{11}$ (giờ). \\
	Các thời điểm để kim giờ và kim phút chập nhau là $\dfrac{12}{11}m$ (giờ) với $m \in \mathbb{N}$. Do $0 \leq \dfrac{12}{11}m  \leq 12$ nên $0 \leq m \leq 11$. Ngoài thời điểm $0$ giờ, còn $11$ thời điểm khác nhau mà kim giờ và kim phút chập nhau là $1\dfrac{1}{11}$ giờ, $2\dfrac{2}{11}$ giờ, $3\dfrac{3}{11}$ giờ,...,$10\dfrac{10}{11}$ giờ, $12$ giờ.
	\item	[b)] Trong một giờ, kim giây quay được $60$ vòng, kim giờ quay được $\dfrac{1}{12}$ vòng. Khoảng thời gian để kim giây gặp lại kim giờ lần tiếp theo là $1 : \left(60 - \dfrac{1}{12} \right) = \dfrac{12}{719}$ (giờ). \\
	Các thời điểm để kim giờ và kim giây chập nhau là $\dfrac{12}{719}n$ (giờ) với $n \in \mathbb{N}$. \\
	Để tìm thời điểm cả ba kim đồng hồ chập nhau, ta phải giải phương trình sau với nghiệm tự nhiên $\dfrac{12m}{11} = \dfrac{12n}{719} \Leftrightarrow 719m = 11n$. \\
	Ta có $719m \vdots 11$, mà $(719,11) = 1$ nên $m \vdots 11$.\\
	Như vậy, ngoài thời điểm $0$ giờ, chỉ có thời điểm $12$ giờ là cả ba kim đồng hồ đều chập nhau.
	\end{itemize}
	\textbf{Lưu ý.} Có các thời điểm mà kim giờ và kim phút chập nhau, còn kim giờ lệch đi một chút. Ta xét tất cả các thời điểm mà kim giờ và kim phút chập nhau, trừ lúc $0$ giờ và lúc $12$ giờ, đó là: \\
	$1\dfrac{1}{11}$ giờ $=$ $1$ giờ $5$ phút $27 \dfrac{3}{11}$ giây; $2\dfrac{2}{11}$ giờ $=$ $2$ giờ $10$ phút $54\dfrac{6}{11}$ giây $3\dfrac{3}{11}$ giờ $=$ $3$ giờ $16$ p hút $21\dfrac{9}{11}$ giây; $4 \dfrac{4}{11}$ giờ $=$ $4$ giờ $21$ phút $49 \dfrac{1}{11}$ giây $5 \dfrac{5}{11}$ giờ $=$ $5$ giờ $27$ phút $16 \dfrac{4}{11}$ giây; $6\dfrac{6}{11}$ giờ $=$ $6$ giờ $32$ phút $43 \dfrac{7}{11}$ giây; $7 \dfrac{7}{11}$ giờ $=$ $7$ giờ $38$ phút $10 \dfrac{10}{11}$ giây; $8\dfrac{8}{11}$ giờ $=$ $8$ giờ $43$ phút $38 \dfrac{2}{11}$ giây; $9 \dfrac{9}{11}$ giờ $=$ $9$ giờ $49$ phút $5 \dfrac{5}{11}$ giây;$10 \dfrac{10}{11}$ giờ $=$ $10$ giờ $54$ phút $32 \dfrac{8}{11}$ giây. \\
	Xét các thời điểm trên, ta thấy vào lúc $3$ giờ $16$ phút $21 \dfrac{9}{11}$ giây và $8$ giờ $43$ phút $38\dfrac{2}{11}$ giây thì kim giờ và kim phút chập nhau, còn kim giây lệch đi một chút.\\
	Tại các thời điểm $3$ giờ $16$ phút $16$ giây và $8$ giờ $43$ phút $43$ giây, trong ba kim đồng hồ không có hai kim nào hoàn toàn chập nhau, nhưng cả ba kim rất sát nhau. \\
	Bạn đọc từ tìm các thời điểm kim giờ và kim giây chập nhau, còn kim phút lệch đi một chút; kim phút và kim giây chập nhau, còn kim giờ lệch đi một chút.
