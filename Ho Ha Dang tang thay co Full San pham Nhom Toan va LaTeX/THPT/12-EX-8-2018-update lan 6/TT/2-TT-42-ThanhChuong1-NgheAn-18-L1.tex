\begin{ex}%[Thi thử lần 1, Thanh Chương 1, Nghệ An 2018][2D2K4-6]%[Lê Mạnh Thắng, 12EX-8]
	Có bao nhiêu số tự nhiên $m$ để hàm số $y=\dfrac{x^4}{4}-\dfrac{2x^3}{3}-\dfrac{m-1}{2}x^2+mx-\ln{x}+2$ đồng biến trên $(2;+\infty)$?
	\choice
	{$3$}
	{$1$}
	{\True $2$}
	{$4$}
	\loigiai{
	Hàm số đồng biến trên $(2;+\infty)$ khi và chỉ khi
	\begin{eqnarray*}
		& & y'\geq0,\forall x\in (2;+\infty)\\
		&\Leftrightarrow & x^3-2x^2-(m-1)x+m-\dfrac{1}{x}\geq0, \forall x\in (2;+\infty)\\
		&\Leftrightarrow & m\leq\dfrac{x^4-2x^3+x^2-1}{x^2-x},\forall x\in (2;+\infty)\\
		&\Leftrightarrow & m\leq x^2-x-\dfrac{1}{x^2-x},\forall x\in (2;+\infty) \qquad (1).
	\end{eqnarray*}
	Đặt $t=x^2-x$, ta có $t'=2x-1>0, \forall x\in (2;+\infty)$. Do đó $x\in (2;+\infty) \Rightarrow t\in (2;+\infty)$.\\
	Xét hàm $f(t)=t-\dfrac{1}{t}$ trên $(2;+\infty)$. Ta có $f'(t)=1+\dfrac{1}{t^2}>0, \forall t\in (2;+\infty)$.\\
	Khi đó $(1)$ tương đương với
	\begin{eqnarray*}
		& & m\leq f(t),\forall t\in (2;+\infty)\\
		&\Leftrightarrow & m\leq \lim\limits_{t \rightarrow 2^+}f(t)=\dfrac{3}{2}.
	\end{eqnarray*}
	Vì $m\in \mathbb{N}$ nên $m=0$, $m=1$.	
	}
\end{ex}
