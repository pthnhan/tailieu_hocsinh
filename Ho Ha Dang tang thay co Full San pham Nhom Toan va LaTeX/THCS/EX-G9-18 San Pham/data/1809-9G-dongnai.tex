\begin{ex}%[Đề thi HSG 9, tỉnh Đồng Nai, năm 2018]%[Lục Hán Tường, Dự án EX-9G-18]%[0D0G1]
Cho $ a, b, c $ là ba số thực dương thỏa mãn $ ab + bc + ac = 1 $. Tính giá trị biểu thức
\begin{align*} P = a\sqrt{\dfrac{\left(1 + b^2\right)\left(1 + a^2\right)}{\left(1 + a^2\right)}} +b\sqrt{\dfrac{\left(1 + a^2\right)\left(1 + c^2\right)}{\left(1 + b^2\right)}} +c\sqrt{\dfrac{\left(1 + a^2\right)\left(1 + b^2\right)}{\left(1 + c^2\right)}}
\end{align*}
\loigiai{
Ta có
\begin{align*}
&1 + a^2 = ab + bc + ca + a^2 = b\left(a + c\right) + a\left(c + a \right) = \left(a + c\right)\left(a + b\right)\\
&1 + b^2 = ab + bc + ca + b^2 = b\left(a + b\right) + c\left(b + a \right) = \left(b + c\right)\left(a + b\right)\\
&1 + c^2 = ab + bc + ca + c^2 = b\left(a + c\right) + c\left(a + c \right) = \left(a + c\right)\left(b + c\right)
\end{align*}
Do đó
\begin{align*}
&\dfrac{\left(1 + b^2\right)\left(1 + c^2\right)}{\left(1 + a^2\right)} = \dfrac{\left(b + c\right)\left(a + b\right)\left(a + c\right)\left(b + c\right)}{\left(a + c\right)\left(a + b\right)} = \left(b + c\right)^2\\
&\Rightarrow a\sqrt{\dfrac{\left(1 + b^2\right)\left(1 + c^2\right)}{\left(1 + a^2\right)}} = a\left(b + c\right)
\end{align*}
Tính tương tự ta được các biểu thức
\begin{align*}
& b\sqrt{\dfrac{\left(1 + a^2\right)\left(1 + c^2\right)}{\left(1 + b^2\right)}} = b\left(a + c\right)\\
& c\sqrt{\dfrac{\left(1 + b^2\right)\left(1 + a^2\right)}{\left(1 + c^2\right)}} = c\left(b + a\right)
\end{align*}
Suy ra biểu thức cần tính
\begin{align*}
P = a\left(b + c\right) + b\left(a + c\right) + c\left(b + a\right) = 2\left(ab + bc + ac\right) = 2
\end{align*}
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Đồng Nai, năm 2018]%[Lục Hán Tường, Dự án EX-9G-18]%[0D0G2]
Giải các phương trình
\begin{enumerate}
	\item $ x^4 + 2x^3 = 4x + 4 $
	\item $ \dfrac{1}{x^2} + \sqrt{x + 2} = \dfrac{1}{x} + \sqrt{2x + 1} $
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Cộng $ x^2 $ vào hai vế phương trình ta được
	\begin{align*}
		 & \Leftrightarrow x^4 + 2x^3 + x^2 = 4x + 4 + x^2
		 \Leftrightarrow x^2\left(x + 1\right)^2 = \left(x + 2\right)^2\\
		 & \Leftrightarrow \hoac{&x\left(x + 1\right) = x + 2\\&x\left(x + 1\right) = \left(x + 2\right)}
		 \Leftrightarrow \hoac{&x^2 + 2x + 2 = 0\\& x^2 -2 = 0} \qquad (*)
	\end{align*}
	Do $ x^2 + 2x + 2 = \left(x + 1\right)^2 + 1 > 0 $ nên $ (*)\Leftrightarrow x^2 = 2 \Leftrightarrow x = \pm \sqrt{2} $.
	
	\item Điều kiện có nghiệm
	\begin{align*}
	\heva{&x\neq0\\&x \geqslant -2\\& x \geqslant -\dfrac{1}{2}} \Leftrightarrow \heva{& x \neq 0\\& x \geqslant -\dfrac{1}{2}}
	\end{align*}
	Chuyển vế phương trình đã cho
	\begin{align*}
		& \Leftrightarrow \dfrac{1}{x^2} - \dfrac{1}{x} + \sqrt{x + 2} - \sqrt{2x + 1} = 0\\
		& \Leftrightarrow \dfrac{1 - x}{x^2} + \dfrac{1 - x}{\sqrt{x + 2} + \sqrt{2x + 1}} = 0\\
		& \Leftrightarrow \left(1 - x\right)\left(\dfrac{1}{x^2} + \dfrac{1}{\sqrt{x + 2} + \sqrt{2x + 1}} \right) = 0\\
		& \Leftrightarrow \hoac{& x = 1\\& \dfrac{1}{x^2} + \dfrac{1}{\sqrt{x + 2} + \sqrt{2x + 1}} = 0} \qquad (*)
	\end{align*}
	Do $ \heva{& x^2 > 0 \\& \sqrt{x + 2} + \sqrt{2x + 1} > 0} \Rightarrow \heva{& \dfrac{1}{x^2} > 0 \\& \dfrac{1}{\sqrt{x + 2} + \sqrt{2x + 1}} > 0} \Rightarrow \dfrac{1}{x^2} + \dfrac{1}{\sqrt{x + 2} + \sqrt{2x + 1}} > 0 $.\\
	Do đó phương trình $ (*) \Leftrightarrow x = 1 $.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Đồng Nai, năm 2018]%[Lục Hán Tường, Dự án EX-9G-18]%[0D0G7]
Cho $ a, b, c $ là ba số không âm có tổng bằng $ 1 $. Chứng minh
\begin{enumerate}
	\item $ 3\left(ab + bc + ca \right) \leqslant 1 $
	\item $ a^2 + b^2 + c^2 \geqslant 4\left(ab + bc + ca \right) - 1 $
\end{enumerate}
\loigiai{
Chứng minh bất đẳng thức sau: $ \forall a, b, c \geqslant 0 $
\begin{align}
a^2 + b^2 + c^2 \geqslant ab + bc + ca \label{eq:1809.1}
\end{align}
Thật vậy, ta có
\begin{align*}
	& \heva{& \left(a - b\right)^2 \geqslant 0 \Leftrightarrow a^2 + b^2 \geqslant 2ab\\
	& \left(a - c\right)^2 \geqslant 0 \Leftrightarrow a^2 + c^2 \geqslant 2ac\\
	& \left(c - b\right)^2 \geqslant 0 \Leftrightarrow c^2 + b^2 \geqslant 2bc}
	\Rightarrow 2\left(a^2 + b^2 + c^2\right) \geqslant 2\left(ab + bc + ca\right)\\
	& \Leftrightarrow a^2 + b^2 + c^2 \geqslant ab + bc + ca
\end{align*}
\begin{enumerate}
\item Cộng hai vế bất đẳng thức \ref{eq:1809.1} với biểu thức $ 2ab + 2bc + 2ca $
\begin{align}
\eqref{eq:1809.1} 
& \Leftrightarrow 3\left(ab + bc + ca \right) \leqslant a^2 + b^2 + c^2 + 2ab + 2bc + 2ca \notag \\
& \Leftrightarrow 3\left(ab + bc + ca \right) \leqslant \left( a + b + c \right)^2 \notag \\
& \Leftrightarrow 3\left(ab + bc + ca \right) \leqslant 1 \label{eq:1809.2}
\end{align}
\item Cộng hai vế của hai bất đẳng thức \ref{eq:1809.1} và \ref{eq:1809.2}, ta được
\begin{align*}
	a^2 + b^2 + c^2 + 1 \geqslant 4\left(ab + bc + ca \right)\\
 \Leftrightarrow a^2 + b^2 + c^2 \geqslant 4\left(ab + bc + ca \right) - 1
\end{align*}
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Đồng Nai, năm 2018]%[Lục Hán Tường, Dự án EX-9G-18]%[0H0G5]%[0H0G1]
Cho tam giác $ ABC $ có ba góc nhọn và nội tiếp đường tròn $ (O) $, $ (AB > AC) $. Hai tiếp tuyến của $ (O) $ tại $ B $ và $ C $ cắt nhau tại $ K $. Đường tròn tâm $ K $ bán kính $ KB $ cắt tia $ AB, AC $ lần lượt tại $ D, E \ (D \neq B, E \neq C) $. Gọi $ M $ là trung điểm $ BC $.
\begin{enumerate}
	\item Chứng minh $ D, K, E $ thẳng hàng.
	\item Chứng minh $ \widehat{BAM} = \widehat{CAK} $.
	\item Gọi N là giao điểm của $ AK $ và $ BC $. Chứng minh $ \dfrac{NB}{NC} = \dfrac{AB^2}{AC^2} $

\begin{enumerate}

	\item Xét hai tam giác $ \triangle ABC $ và $ \triangle AED $ có

\begin{ex}%[Đề thi HSG 9, tỉnh Đồng Nai, năm 2018]%[Lục Hán Tường, Dự án EX-9G-18]%[0H0G1]%[0H0G7]
Cho tam giác $ ABC $ có $ A = 60^\circ $ và độ dài ba cạnh $ BC = a, CA = b, AB = c $ là ba số nguyên khác nhau.
\begin{enumerate}
	\item Chứng minh $ a^2 = b^2 + c^2 - bc $.
	\item Giả sử $ b < c $, thì $ b \geqslant 3 $.		
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item 
	\immini{
	Gọi $ H $ là chân đường cao kẻ từ $ C $.\\
	Ta xét $ \triangle ACH $ có
	\begin{align*}
	\heva{& A = 60^\circ\\& AC = b} \Rightarrow \heva{& AH = AC \ cos A = \dfrac{1}{2}b\\&CH = AC\sin A = \dfrac{\sqrt{3}}{2}b}
	\end{align*}
	Lại xét $ \triangle CHB $ có
	\begin{align*}
	& BC^2 = BH^2 + CH^2\\
	\Leftrightarrow & BC^2 = \left(AB - AH\right)^2 + CH^2\\
	\Leftrightarrow & a^2 = \left(c - \dfrac{b}{2}\right)^2 + \left(\dfrac{\sqrt{3}}{2}b\right)^2\\
	\Leftrightarrow & a^2 = b^2 + c^2 - bc
	\end{align*}
	}{
	\begin{tikzpicture}
	\tkzDefPoints{0/0/A}
	\def\x{5}
	\def\y{3}
	\coordinate (B) at ($(A)+(\x,0)$);
	\coordinate (C) at ($(A)+({\y*cos(60)},{\y*sin(60)})$);
	\tkzDefPointBy[projection=onto A--B](C)\tkzGetPoint{H}
	\tkzDrawSegments(A,B B,C A,C C,H)
	\tkzDrawPoints(A,B,C,H)
	\tkzLabelPoints[below left](A)
	\tkzLabelPoints[below](H)
	\tkzLabelPoints(B)
	\tkzLabelPoints[above](C)
	\tkzMarkAngles[label=$ 60^\circ $,size=.7cm](B,A,C)
	\tkzMarkRightAngle(A,H,C)
\end{tikzpicture}
	}
	\item Ta xét bài toán với hai trường hơp sau:\\
	Với $ b = 1 $ ta có $ a^2 = c^2 - c + 1 $. Vì $ c^2 < c^2 - c + 1 < \left(c - 1\right)^2 $ nên phương trình $ a^2 = c^2 - c + 1 $ không có nghiệm nguyên.\\
	Với $ b = 2 $ ta có $ a^2 = c^2 - 2c + 4 $. Vì $ \left(c - 1\right)^2 < c^2 - 2c + 4 < c^2 $ do $ (c > b = 2) $ nên phương trình $ a^2 = c^2 - 2c + 4 $ không có nghiệm nguyên.\\
	Do đó, vì luôn tìm được một tam giác $ ABC $ có ba cạnh có độ dài nguyên nên tìm được $ b \geqslant 3 $.
\end{enumerate}
}
