\begin{bt}%[Học Kì 1, THPT Bình Phú, HCM, 18]%[Đinh Thanh Hoàng - 11HK1-HCM]%[1D1B2-1]%[1D1B3-1]%[1D3B3-3]
	Giải các phương trình sau:
	\begin{enumerate}
		\item $2\sin x - 1 = 0$.
		\item $\cos 2x - 4\sin^2x + \cos x + 3 = 0$.
		\item $2(\sin x + \cos x) + \cos 2x = 0$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $2\sin x - 1 = 0 \Leftrightarrow \sin x = \dfrac{1}{2} \Leftrightarrow \hoac{&x = \dfrac{\pi}{6} + k2\pi \\ &x = \dfrac{5\pi}{6} + k2\pi} (k \in \mathbb{Z})$.\\
					Vậy phương trình đã cho có tập nghiệm $S = \left\lbrace \dfrac{\pi}{6} + k2\pi, \dfrac{5\pi}{6} + k2\pi \,\left|\, k\in \mathbb{Z}\right.\right\rbrace$.
			\item Ta có $\cos 2x - 4\sin^2x + \cos x + 3 = 0 \Leftrightarrow 2\cos^2x - 1 - 4(1 - \cos^2x) + \cos x + 3 = 0$\\
					\phantom{Ta có $\cos 2x - 4\sin^2x + \cos x + 3 = 0$} $\Leftrightarrow 6\cos^2x + \cos x - 2 = 0$\\
					\phantom{Ta có $\cos 2x - 4\sin^2x + \cos x + 3 = 0$} $\Leftrightarrow \hoac{&\cos x = \dfrac{1}{2} \\ &\cos x = -\dfrac{2}{3}}$\\
					\phantom{Ta có $\cos 2x - 4\sin^2x + \cos x + 3 = 0$} $\Leftrightarrow \hoac{&x = \pm\dfrac{\pi}{3} + k2\pi \\ &x = \pm\arccos\left(-\dfrac{2}{3}\right) + k2\pi} (k \in \mathbb{Z})$.\\
					Vậy phương trình đã cho có tập nghiệm $S = \left\lbrace \pm\dfrac{\pi}{3} + k2\pi, \pm\arccos\left(-\dfrac{2}{3}\right) + k2\pi \,\left|\, k\in \mathbb{Z}\right.\right\rbrace$.
			\item Ta có $2(\sin x + \cos x) + \cos 2x = 0 \Leftrightarrow 2(\sin x + \cos x) + \cos^2x - \sin^2x = 0$\\
					\phantom{Ta có $2(\sin x + \cos x) + \cos 2x = 0$} $\Leftrightarrow 2(\sin x + \cos x) + (\cos x - \sin x)(\cos x + \sin x) = 0$\\
					\phantom{Ta có $2(\sin x + \cos x) + \cos 2x = 0$} $\Leftrightarrow (\sin x + \cos x)(\cos x - \sin x + 2) = 0$\\
					\phantom{Ta có $2(\sin x + \cos x) + \cos 2x = 0$} $\Leftrightarrow \hoac{&\sin x + \cos x = 0 \\ &\cos x - \sin x + 2 = 0 \,\mbox{(vô nghiệm)}}$\\
					\phantom{Ta có $2(\sin x + \cos x) + \cos 2x = 0$} $\Leftrightarrow \tan x = -1 \Leftrightarrow x = -\dfrac{\pi}{4} + k\pi$ $(k \in \mathbb{Z})$.\\
					Vậy phương trình đã cho có tập nghiệm $S = \left\lbrace x = -\dfrac{\pi}{4} + k\pi \,\left|\, k\in \mathbb{Z}\right.\right\rbrace$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Học Kì 1, THPT Bình Phú, HCM, 18]%[Đinh Thanh Hoàng - 11HK1-HCM]%[1D2B3-2]
	Trong khai triển nhị thức Niu-tơn $\left(x^2 - \dfrac{2y}{x}\right)^{10}$. Tìm số hạng thỏa điều kiện  lũy thừa của  biến $x$ bằng lũy thừa của biến $y$.
	\loigiai{
		Số hạng thứ $k + 1$ trong khai triển nhị thức Niu-tơn $\left(x^2 - \dfrac{2y}{x}\right)^{10}$ là
		$$T_{k + 1} = \mathrm{C}_{10}^k\left(x^2\right)^{10 - k}\left(-\dfrac{2y}{x}\right)^k = \mathrm{C}_{10}^k (-2)^k x^{20 - 3k} y^k.$$
		Theo đề bài, ta có $20 - 3k = k \Leftrightarrow k = 5$.\\
		Vậy số hạng cần tìm là $\mathrm{C}_{10}^5 (-2)^5 x^5y^5 = -8064x^5y^5$.
	} 
\end{bt}

\begin{bt}%[Học Kì 1, THPT Bình Phú, HCM, 18]%[Đinh Thanh Hoàng - 11HK1-HCM]%[1D3B3-3]
	Tìm số hạng đầu $u_1$ và công sai $d$ của cấp số cộng $(u_n)$, biết $\heva{&u_2 - u_3 + u_5 = 10 \\ &u_1 + u_6 = 17.}$
	\loigiai{
		Ta có $\heva{&u_2 - u_3 + u_5 = 10 \\ &u_1 + u_6 = 17} \Leftrightarrow \heva{&u_1 + d - u_1 - 2d + u_1 + 4d = 10 \\ &u_1 + u_1 + 5d = 17} \Leftrightarrow \heva{&u_1 + 3d = 10 \\ &2u_1 + 5d = 17} \Leftrightarrow \heva{&u_1 = 1 \\ &d = 3.}$\\
		Vậy $u_1 = 1$ và $d = 3$.
	}
\end{bt}

\begin{bt}%[Học Kì 1, THPT Bình Phú, HCM, 18]%[Đinh Thanh Hoàng - 11HK1-HCM]%[1D2B5-2]
	Có hai cái hộp chứa bi. Hộp thứ nhất chứa $4$ viên bi đỏ và $3$ viên bi trắng, hộp thứ hai chứa $2$ viên bi đỏ và $4$ viên bi trắng. Lấy ngẫu nhiên từ mỗi hộp ra $1$ viên bi. Tính xác suất để $2$ viên bi được lấy ra có cùng màu.
	\loigiai{
		Số kết quả lấy ngẫu nhiên từ mỗi hộp ra $1$ viên bi là $n(\Omega) = 7 \cdot 6 = 42$.\\
		Gọi $A$ là biến cố ``$2$ viên bi được lấy ra có cùng màu''.
		\begin{itemize}
			\item $2$ viên bi được lấy ra có cùng màu đỏ, có $4 \cdot 2 = 8$ kết quả.
			\item $2$ viên bi được lấy ra có cùng màu trắng, có $3 \cdot 4 = 12$ kết quả.
		\end{itemize}
		Suy ra $n(A) = 8 + 12 = 20$.\\
		Vậy $\mathrm{P}(A) = \dfrac{n(A)}{n(\Omega)} = \dfrac{20}{42} = \dfrac{10}{21}$.
	}
\end{bt}

\begin{bt}%[Học Kì 1, THPT Bình Phú, HCM, 18]%[Đinh Thanh Hoàng - 11HK1-HCM]%[1D2K1-3]
	Từ các chữ số $0$, $1$, $2$, $3$, $4$, $5$, $6$. Ta lập được bao nhiêu số tự nhiên gồm $4$ chữ số khác nhau và số tự nhiên này lớn hơn $3452$?
	\loigiai{
		Gọi $\overline{abcd}$ là số tự nhiên gồm $4$ chữ số khác nhau và lớn hơn $3452$, trong đó $a$, $b$, $c$, $d \in \lbrace 0, 1, 2, 3, 4, 5, 6\rbrace$.
		\begin{itemize}
			\item $a \in \lbrace 4, 5, 6\rbrace$, có $3 \cdot 6 \cdot 5 \cdot 4 = 360$ (số).
			\item $a \in \lbrace 3\rbrace$, $b \in \lbrace 5, 6\rbrace$, có $2 \cdot 5 \cdot 4 = 40$ (số).
			\item $a \in \lbrace 3\rbrace$, $b \in \lbrace 4\rbrace$, $c \in \lbrace 6\rbrace$, có $4$ (số).
			\item $a \in \lbrace 3\rbrace$, $b \in \lbrace 4\rbrace$, $c \in \lbrace 5\rbrace$, có $1$ (số).			
		\end{itemize}
		Vậy theo quy tắc cộng, có $360 + 40 + 4 + 1 = 405$ (số).
	} 
\end{bt}

	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình bình hành. Gọi $M$ là trung điểm của $SA$, điểm $N$ trên cạnh $AD$ sao cho $DN = 2NA$.
	\begin{enumerate}
		\item Tìm giao tuyến của hai mặt phẳng $(SAC)$ và $(SBD)$.
		\item Tìm giao điểm $K$ của đường thẳng $BN$ và mặt phẳng $(SCD)$.
		\item  Gọi $F$ là điểm trên cạnh $SB$ sao cho $SF = 2FB$. Xác định thiết diện của hình chóp $S.ABCD$ khi cắt bởi mặt phẳng $(MFC)$.
	\end{enumerate}

			\coordinate (F) at ($(S)!.6666!(B)$);

			\tkzInterLL(B,N)(C,D) \tkzGetPoint{K}
			\tkzInterLL(M,F)(A,B) \tkzGetPoint{H}
			\tkzInterLL(H,C)(A,D) \tkzGetPoint{I}
			\tkzInterLL(M,I)(S,D) \tkzGetPoint{E}
			\tkzInterLL(B,K)(M,I) \tkzGetPoint{G}
			\tkzInterLL(C,D)(M,I) \tkzGetPoint{P}

			\tkzDrawSegments[dashed](S,A A,H A,I A,C B,D M,F M,E B,G S,O C,P E,D F,B B,C)
			\tkzDrawPoints(S,A,B,C,D,O,M,N,E,F,H,I,K)
			\tkzLabelPoints[above](S,K,E)	
			\tkzLabelPoints[below](O,N)

		\begin{enumerate}
			\item Trong mặt phẳng $(ABCD)$, gọi $O = AC \cap BD$, ta có $O \in (SAC) \cap (SBD)$.\\
					Vậy $SO$ là giao tuyến của hai mặt phẳng $(SAC)$ và $(SBD)$.
			\item Trong mặt phẳng $(ABCD)$, gọi $K = BN \cap CD$, ta có $\heva{&K \in BN \\ &K \in CD \subset (SCD).}$\\
					Suy ra $K = BN \cap (SCD)$.
			\item Trong mặt phẳng $(SAB)$, gọi $H = MF \cap AB$; trong mặt phẳng $(ABCD)$, gọi $I = HC \cap AD$; trong mặt phẳng $(SAD)$, gọi $E = MI \cap SD$.\\
					Khi đó thiết diện của hình chóp $S.ABCD$ khi cắt bởi mặt phẳng $(MFC)$ là tứ giác $MECF$.
		\end{enumerate}
	}
