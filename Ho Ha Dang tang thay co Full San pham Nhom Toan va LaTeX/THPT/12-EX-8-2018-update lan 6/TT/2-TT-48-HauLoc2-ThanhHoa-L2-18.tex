\begin{ex}%[Thi thử, Hậu Lộc 2, Thanh Hoá, L2, 2018]%[2D1G5-4]%[Dương Phước Sang, 12EX8]
	Biết rằng đồ thị hàm số $y=f(x)=ax^4+bx^3+cx^2+\mathrm{\,d}x+e$, $(a,b,c,d,e\in \mathbb{R};a\ne 0,b\ne 0)$ cắt trục hoành $Ox$ tại 4 điểm phân biệt. Khi đó đồ thị hàm số $y=g(x)$ cắt trục hoành $Ox$ tại bao nhiêu điểm, trong đó $g(x)=(4ax^3+3bx^2+2cx+d)^2-2(6ax^2+3bx+c)(ax^4+bx^3+cx^2+dx+e)$?
	\choice
	{$6$}
	{\True $0$}
	{$4$}
	{$2$}
	\loigiai{
		Giả sử đồ thị hàm số $y=f(x)=ax^4+bx^3+cx^2+\mathrm{\,d}x+e,(a,b,c,d,e\in \mathbb{R};a\ne 0,b\ne 0)$ cắt trục hoành $Ox$ tại 4 điểm phân biệt $x_1$, $x_2$, $x_3$, $x_4$. \\
		Đặt $A=x-x_1$; $B=x-x_2$; $C=x-x_3$; $D=x-x_4$ ta có \\
		$f(x)=a(x-x_1)(x-x_2)(x-x_3)(x-x_4)=aABCD$. \\
		Xét thấy $g(x)=[f'(x)]^2-f''(x)\cdot f(x)$ \\
		TH1: Nếu $x=x_i,i=1,2,3,4$ thì $g(x_i)={\left[f'(x_i)\right]}^2>0$.\\
		Do đó, $x=x_i,i=1,2,3,4$ không phải nghiệm của phương trình $g(x)=0$. \\
		TH2: Nếu $x\ne x_i,i=1,2,3,4$ thì ta viết lại \\
		$f'(x)=a[BCD+ACD+ABD+ABC]=f(x)\left(\dfrac{1}{A}+\dfrac{1}{B}+\dfrac{1}{C}+\dfrac{1}{D}\right)$. \\
		$f''(x)=f'(x)\left(\dfrac{1}{A}+\dfrac{1}{B}+\dfrac{1}{C}+\dfrac{1}{D}\right)-f(x)\left(\dfrac{1}{A^2}+\dfrac{1}{B^2}+\dfrac{1}{C^2}+\dfrac{1}{D^2}\right)$ \\
		$=f(x){\left(\dfrac{1}{A}+\dfrac{1}{B}+\dfrac{1}{C}+\dfrac{1}{D}\right)\!}^2-f(x)\left(\dfrac{1}{A^2}+\dfrac{1}{B^2}+\dfrac{1}{C^2}+\dfrac{1}{D^2}\right)$ \\
		Suy ra, $f''(x)\cdot f(x)=f^2(x){\left(\dfrac{1}{A}+\dfrac{1}{B}+\dfrac{1}{C}+\dfrac{1}{D}\right)\!}^2-f^2(x)\left(\dfrac{1}{A^2}+\dfrac{1}{B^2}+\dfrac{1}{C^2}+\dfrac{1}{D^2}\right)$ \\

		Từ đó suy ra phương trình $g(x)=0$ vô nghiệm. \\
		Vậy đồ thị hàm số $y=g(x)$ không cắt trục hoành. }
\end{ex}
