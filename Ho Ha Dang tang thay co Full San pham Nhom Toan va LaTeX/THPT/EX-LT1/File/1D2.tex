\subsection{Hai qui tắc đếm}
\begin{ex} %[1D2Y1-1] %[Trần Hoàng Trọng Ân]

	\choice
	{\True $ m+n $}
	{$ m-n $}
	{$ \dfrac{m}{n} $}
	{$ m\cdot n $}
	\loigiai{Áp dụng qui tắc cộng.
	}
\end{ex}

\subsection{Hoán vị, chỉnh hợp, tổ hợp}
\begin{ex}%[Nguyễn Tài Tuệ]%[1D2Y2-1]
	Kí hiệu $\mathrm{A}_n^k$ là số các chỉnh hợp chập $k$ của $n$ phần tử $(1\le k\le n)$. Mệnh đề nào dưới đây đúng?
	\choice
	{$\mathrm{A}_n^k=\dfrac{n!}{(n+k)!}$}
	{$\mathrm{A}_n^k=\dfrac{n!}{k!(n+k)!}$}
	{$\mathrm{A}_n^k=\dfrac{n!}{k!(n-k)!}$}
	{\True $\mathrm{A}_n^k=\dfrac{n!}{(n-k)!}$}
	\loigiai{Theo công thức ta có $\mathrm{A}_n^k=\dfrac{n!}{(n-k)!}$.
	}
\end{ex}
\begin{ex}%[Nguyễn Tài Tuệ]%[1D2Y2-1]
	Cho tập hợp $ A $ gồm $ n $ phần tử ($ n\ge 1 $). Khẳng định nào sau đây \textbf{sai}?
	\choice
	{Mỗi kết quả của sự sắp xếp thứ tự $ n $ phần tử của tập hợp $ A$ được gọi là một hoán vị của $ n $ phần tử đó}
	{Hai hoán vị của n phần tử chỉ khác nhau ở thứ tự sắp xếp}
	{Số các hoán vị của $ n $ phần tử kí hiệu là $\mathrm{ P}_n$ và $ \mathrm{ P}_n=n\cdot (n-1)\cdots 2\cdot1 =n!$}
	{\True Số các hoán vị của $ n $ phần tử kí hiệu là $ \mathrm{ P}_n$ và $ \mathrm{ P}_n=(n+1)\cdot n\cdot (n-1)\cdots 2\cdot1 =(n+1)!$}
	\loigiai{
		Ta có "Số các hoán vị của $ n $ phần tử kí hiệu là $ \mathrm{ P}_n$ và $\mathrm{ P}_n=n\cdot (n-1)\cdots 2\cdot1 =n!$" nên khẳng định "Số các hoán vị của $ n $ phần tử kí hiệu là $ \mathrm{ P}_n$ và $ \mathrm{ P}_n=(n+1)\cdot n\cdot (n-1)\cdots 2\cdot1 =(n+1)!$" là khẳng định \textbf{sai}.
	}
\end{ex}

\begin{ex}%[Nguyễn Tài Tuệ]%[1D2Y2-2]
	Cho số nguyên dương $n$ và số nguyên không âm $k$, $k \leq n$. Có bao nhiêu mệnh đề đúng trong các mệnh đề sau đây?
	\begin{enumerate}
		\item[(1)]   $\mathrm{A}_n^k = \dfrac{n!}{(n-k)!}$
		\item[(2)]   $\mathrm{C}_n^k = \dfrac{n!}{k!}$
		\item[(3)]  $n\cdot\mathrm{P}_n = \mathrm{P}_{n+1}$
		\item[(4)]  $\mathrm{C}_n^k\cdot\mathrm{P}_n = \mathrm{A}_n^k$
		\item[(5)]  $\mathrm{C}_n^k+\mathrm{C}_n^{k+1}=\mathrm{C}_{n+1}^{k+1}$
	\end{enumerate}
	\choice 
	{\True 2}
	{3}
	{4}
	{5}
	\loigiai{
		\begin{enumerate}
			\item Khẳng định "$\mathrm{A}_n^k = \dfrac{n!}{(n-k)!}$" là một khẳng định đúng theo cách tính số chỉnh hợp.
			\item  Khẳng định $\mathrm{C}_n^k = \dfrac{n!}{k!}$ là khẳng định sai vì $\mathrm{C}_n^k = \dfrac{n!}{k!\cdot (n-k)!}.$
			\item Khẳng định $n\cdot\mathrm{P}_n = \mathrm{P}_{n+1}$ là một khẳng định sai vì $ \mathrm{P}_{n+1} =(n+1)\mathrm{P}_n. $
			\item Khẳng định $\mathrm{C}_n^k\cdot\mathrm{P}_n = \mathrm{A}_n^k$ là khẳng định sai, vì $\mathrm{C}_n^k\cdot\mathrm{P}_k = \mathrm{A}_n^k$.
			\item Khẳng định $\mathrm{C}_n^k+\mathrm{C}_n^{k+1}=\mathrm{C}_{n+1}^{k+1}$ là khẳng định đúng theo công thức Pa-xcan.
		\end{enumerate}	
		Vậy có hai khẳng định đúng.
	}
\end{ex}
\begin{ex}%[Nguyễn Tài Tuệ]%[1D2Y2-2]

	\loigiai{
		Khẳng định $\mathrm{C}_n^k=\mathrm{C}_n^{n-k+1}$ là khẳng định sai vì $\mathrm{C}_n^k=\mathrm{C}_n^{n-k}$.
	}
\end{ex}


\begin{ex}%[Nguyễn Tài Tuệ]%[1D2Y2-2]

	{Số các hoán vị của tập có $n$ phần tử được ký hiệu là $P_n$}
	{\True Số các tổ hợp chập $k$ của $n$ phần tử được ký hiệu là $A^k_n$ với ($1\leq k\leq n$)}
	{Mỗi hoán vị của $n$ phần tử cũng chính là một chỉnh hợp chập $n$ của $n$ phần tử đó}

\begin{ex}
	Với $ k,n\in \mathbb{N}, n\ge k $. Mệnh đề nào sau đây \textbf{đúng}?
	\choice
	{$0!=0$}
	{$\mathrm{C}_n^k=\mathrm{C}_n^{k-n}$}
	{\True $1!=1$}
	{$\mathrm{C}_n^0=n$}
	\loigiai{
		Ta có
		\begin{enumerate}[+]
			\item $ 0!=1 $ nên khẳng định $ 0!=0 $ là khẳng định sai.
			\item  $\mathrm{C}_n^k=\mathrm{C}_n^{n-k}$ nên khẳng định $\mathrm{C}_n^k=\mathrm{C}_n^{k-n}$ là khẳng định sai.
			\item $\mathrm{C}_n^0=\dfrac{n!}{0!\cdot n!} =1$ nên khẳng định $\mathrm{C}_n^0=n$ là khẳng định sai.
	\end{enumerate} }
\end{ex}

\subsection{Phép thử và biến cố}
\begin{ex}%[Ha]%[1D2B4-1]
	Gọi $\Omega$ là không gian mẫu của phép thử. Phát biểu nào sau đây {\bf sai}?
	\choice
	{\True Tập $\varnothing$ không là một biến cố}{Biến cố là tập con của không gian mẫu $\Omega$}{Tập $\varnothing\cap \Omega$ là một biến cố}{$\Omega$ là một biến cố}
	\loigiai{ 
		Biến cố là tập con của không gian mẫu. Do đó tập $\varnothing$ cũng là một biến cố.
	}
\end{ex}
\begin{ex}%[Ha]%[1D2B4-1]
	Nếu phép thử là gieo một con xúc sắc cân đối đồng chất hai lần, thì số phần tử của không gian mẫu là
	
	\choice
	{\True $36$}{$6$}{$2$}{$1$}
	\loigiai{ 
		Số phần tử của không gian mẫu là $6^2$.
	}
\end{ex}

\begin{ex}%[Ha]%[1D2B4-1]
	Giả sử không gian mẫu $\Omega$ có $n$ phần tử. Xét các phát biểu sau:\\
	a) Số biến cố là hữu hạn.\\
	b) Có tất cả $n$ biến cố.\\
	c) Có tất cả $2^n-1$ biến cố.\\
	Số phát biểu đúng là
	\choice
	{\True 1}{2}{3}{0}
	\loigiai{ 
		Số tập  con của tập có $n$ phần tử là $2^n$. Do đó phát biểu $(b)$ và $(c)$ sai.
	}
\end{ex}
\begin{ex}%[Ha]%[1D2B4-1]
	Giả sử $A, B$ là hai biến cố xung khắc với không gian mẫu $\Omega$. Chọn phát biểu đúng?

	}

\begin{ex}%[Ha]%[1D2B4-1]

	}

\subsection{Xác suất của biến cố}
\begin{ex}%[BanNt]%[1D2B5-1]
	Giả sử $A$ là biến cố liên quan đến phép thử có một số hữu hạn kết quả đồng khả năng xuất hiện. Khẳng định nào dưới đây là đúng?
	\choice
	{$\mathrm{P}(\overline{A})=\mathrm{P}(A)-1$}
	{$\mathrm{P}(A)=1$}
	{\True $\mathrm{P}(\overline{A})=1-\mathrm{P}(A)$}
	{$\mathrm{P}(\overline{A})-\mathrm{P}(A)=1$}
	\loigiai{ 
		Với mọi biến cố $A$ ta có $\mathrm{P}(\overline{A})=1-\mathrm{P}(A)$.
	}
\end{ex}
\begin{ex}%[BanNt]%[1D2B5-1]
	Giả sử $A$, $B$ là  các biến cố liên quan đến phép thử có một số hữu hạn kết quả đồng khả năng xuất hiện. Trong các khẳng định dưới đây, có bao nhiêu khẳng định đúng?
	\begin{enumerate}
		\item $\mathrm{P}(\varnothing)=0.$
		\item $\mathrm{P}(\Omega)=1. $
		\item $0<\mathrm{P}(A)<1, \, \forall $ biến cố $A$.
		\item Nếu $A$ và $B$ xung khắc, thì $\mathrm{P}(A\cup B)= \mathrm{P}(A)+\mathrm{P}(B).$
	\end{enumerate}
	\choice
	{$1$}
	{$2$}
	{\True $3$}
	{$4$}
	\loigiai{ 
		Ta có $\mathrm{P}(\varnothing)=0$, $\mathrm{P}(\Omega)=1 $, $0\leq \mathrm{P}(A)\leq 1$ với mọi biến cố $A$ và nếu $A$ và $B$ xung khắc, thì $\mathrm{P}(A\cup B)= \mathrm{P}(A)+\mathrm{P}(B).$
	}
\end{ex}

	$B:$ ``Phương lấy được thẻ số $2$''\\

	$D:$ ``Nam lấy được viên bi màu vàng''\\

	}
