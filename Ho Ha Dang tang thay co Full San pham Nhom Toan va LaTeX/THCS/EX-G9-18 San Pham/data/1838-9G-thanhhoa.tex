\begin{ex}%[Đề thi HSG, Thanh Hóa, năm 2018]%[Nguyễn Tất Thu, Dự án EX-9G-18]%[9D0B1]
Cho biểu thức $P=\dfrac{x-2\sqrt{x}}{x\sqrt{x}-1}+\dfrac{\sqrt{x}+1}{x\sqrt{x}+x+\sqrt{x}}+\dfrac{1+2x-2\sqrt{x}}{x^2-\sqrt{x}}$ với $x>0,\ x\neq 1$ . Rút gọn $P$ và tìm tất cả các giá trị của $x$ sao cho giá trị của $P$ là một số nguyên.
\loigiai{
Ta có $$\begin{aligned}P&=\dfrac{x-2\sqrt{x}}{\left(\sqrt{x}-1\right)\cdot \left(x+\sqrt{x}+1 \right)} +\dfrac{\sqrt{x}+1}{\sqrt{x}\left(x+\sqrt{x}+1 \right)}+\dfrac{1+2x-2\sqrt{x}}{\sqrt{x}\left(\sqrt{x}-1\right)\cdot \left(x+\sqrt{x}+1 \right)}\\&=\dfrac{x\sqrt{x}-2x+x-1+1+2x-2\sqrt{x}}{\sqrt{x}\left(\sqrt{x}-1\right)\cdot \left(x+\sqrt{x}+1 \right)} \\&=\dfrac{x\sqrt{x}+x-2\sqrt{x}}{\sqrt{x}\left(\sqrt{x}-1\right)\cdot \left(x+\sqrt{x}+1 \right)}\\&=\dfrac{\sqrt{x}\left(\sqrt{x}-1 \right)\left(\sqrt{x}+2 \right)}{\sqrt{x}\left(\sqrt{x}-1\right)\cdot \left(x+\sqrt{x}+1 \right)} \\&= \dfrac{\sqrt{x}+2}{x+\sqrt{x}+1}.\end{aligned} $$
Ta có $P>0$ và $P-2=\dfrac{-2x-\sqrt{x}}{x+\sqrt{x}+1}<0\ \forall\ x>0,\ x\neq 1.$ Suy ra $0<P<2$, nên $P=1$, hay $$\dfrac{\sqrt{x}+2}{x+\sqrt{x}+1}=1 \Leftrightarrow x=1\ \text{(loại)}.$$
Vậy không có $x$ để $P$ nguyên.
} 
\end{ex}
\begin{ex}%[Đề thi HSG, Thanh Hóa, năm 2018]%[Nguyễn Tất Thu, Dự án EX-9G-18]%[9D0K1]
\item Tính giá trị của biểu thức $P=\dfrac{4(x+1)x^{2018}-2x^{2017}+2x+1}{2x^2+3x}$ tại $x=\sqrt{\dfrac{1}{2\sqrt{3}-2}-\dfrac{3}{2\sqrt{3}+2}}$.
\loigiai{
Ta có $$x=\sqrt{\dfrac{\sqrt{3}+1-3(\sqrt{3}-1)}{2(\sqrt{3}-1)(\sqrt{3}+1)}}=\sqrt{\dfrac{4-2\sqrt{3}}{2\cdot 2}}=\dfrac{1}{2}\sqrt{\left(\sqrt{3}-1\right)^2}=\dfrac{\sqrt{3}-1}{2}.$$
Suy ra $$(2x+1)^2=3 \Leftrightarrow 2x^2+2x=1.$$
Khi đó 
$$\begin{aligned}4(x+1)x^{2018}-2x^{2017}+2x+1&=2x^{2017}(2x^2+2x-1)+2x+1=2x+1 \\ 2x^2+3x&=2x^2+2x+x=x+1. \end{aligned}$$
Do đó $P=\dfrac{2x+1}{x+1}=\dfrac{\sqrt{3}}{\dfrac{\sqrt{3}-1}{2}}=3+\sqrt{3}.$
} 
\end{ex}
\begin{ex}%[Đề thi HSG, Thanh Hóa, năm 2018]%[Nguyễn Tất Thu, Dự án EX-9G-18]%[9D0B2]
 Biết phương trình $(m-2)x^2-2(m-1)x+m=0$ có hai nghiệm tương ứng là độ dài hai cạnh góc vuông của một tam giác vuông. Tìm $m$ để độ dài đường cao tương ứng với cạnh huyền của tam giác vuông đó bằng $\dfrac{2}{\sqrt{5}}$.
\loigiai{
Với $m\neq 2$ ta có $$(m-2)x^2-2(m-1)x+m=0 \Leftrightarrow x_1=1,\ x_2=\dfrac{m}{m-2}.$$
Vì $x_1,\ x_2$ là độ dài hai cạnh góc vuông của một tam giác vuông, nên $x_2>0$, hay $$\dfrac{m}{m-2}>0 \Leftrightarrow \hoac{&m<0 \\ &m>2}. \eqno (1)$$
Khi đó, đường cao của tam giác vuông ứng với cạnh huyền bằng $\dfrac{2}{\sqrt{5}}$, nên 
$$\dfrac{1}{x_1^2}+\dfrac{1}{x_2^2}=\dfrac{5}{4} \Leftrightarrow x^2_2=4 \Leftrightarrow x_2=\pm 2 \Leftrightarrow m=4\ \text{(thỏa (1))},\ m=\dfrac{4}{3}\ \text{(không thỏa (1))}.$$
Vậy $m=4$ là giá trị cần tìm.
} 
\end{ex}
\begin{ex}%[Đề thi HSG, Thanh Hóa, năm 2018]%[Nguyễn Tất Thu, Dự án EX-9G-18]%[9D0K2]
Giải hệ phương trình $\heva{& (x+y)^2(8x^2+8y^2+4xy-13)+5=0  \\ & 2x+\dfrac{1}{x+y}=1.}$
\loigiai{
Điều kiện: $x+y \neq 0.$\\
Ta có 
$$8x^2+8y^2+4xy=5(x+y)^2+3(x-y)^2,$$ nên hệ đã cho có thể viết thành
$$\heva{& 5\left((x+y)^2+\dfrac{1}{(x+y)^2} \right)+3(x-y)^2-13=0 \\& (x-y)+\left(x+y+\dfrac{1}{x+y} \right)=1 } \Leftrightarrow \heva{& 5\left[\left(x+y+\dfrac{1}{x+y}\right)^2-2 \right]+3(x-y)^2-13=0 \\& (x-y)+\left(x+y+\dfrac{1}{x+y} \right)=1 }. $$
Đặt $a=x+y+\dfrac{1}{x+y},\ |a| \geq 2$, $b=x-y$, ta có hệ
$$\heva{&5a^2+3b^2-23=0 \\& a+b=1} \Leftrightarrow \heva{&b=1-a \\& 5a^2+3(1-a)^2-23=0} \Leftrightarrow \heva{&b=1-a\\& 4a^2-3a-10=0.} $$
Ta có $4a^2-3a-10=0 \Leftrightarrow a=2\ \text{(nhận)},\ a=-\dfrac{5}{4}\ \text{(loại)}.$\\
Với $a=2\Rightarrow b=-1$ ta có $$\heva{&x+y+\dfrac{1}{x+1}=2 \\& x-y=-1} \Leftrightarrow \heva{&x+y=1\\&x-y=-1} \Leftrightarrow \heva{&x=0 \\ & y=1.}$$
Vậy nghiệm của hệ là $(x;y)=(0;1).$
} 
\end{ex}
\begin{ex}%[Đề thi HSG, Thanh Hóa, năm 2018]%[Nguyễn Tất Thu, Dự án EX-9G-18]%[9D0K5]
Tìm nghiệm nguyên của phương trình $y^2-5y+62=(y-2)x^2+(y^2-6y+8)x$.
\loigiai{
Phương trình đã cho tương đương với $$\begin{aligned}  &\left({y-2}\right)\left({y-3}\right)+56=(y-2)x^2+\left({y-2}\right)\left({y-4}\right)x \\ &\Leftrightarrow \left({y-2}\right)\left[{x^2+\left({y-4}\right)x-\left({y-3}\right)}\right]=56 \\ &\Leftrightarrow \left({x-1}\right)\left({y-2}\right)\left({x+y-3}\right)=56. \end{aligned}$$
Nhận thấy $\left({y-2}\right)+\left({x-1}\right)=x+y-3,$ nên ta phải phân tích số 56 thành tích của ba số nguyên mà tổng hai số đầu bằng số còn lại.
Như vậy ta có: 
\begin{itemize}
\item[+)] $56=1\cdot 7\cdot 8\Rightarrow \left({x;y}\right)=\left({2;9}\right).$
\item[+)] $56=7\cdot 1\cdot 8\Rightarrow\left({x;y}\right)=\left({8;3}\right).$
\item[+)] $56=\left({-8}\right)\cdot 1\cdot \left({-7}\right)\Rightarrow \left({x;y}\right)=\left({-7;3}\right).$
\item[+)] $56=1\cdot \left({-8}\right)\cdot \left({-7}\right)\Rightarrow\left({x;y}\right)=\left({2;-6}\right).$
\item[+)] $56=\left({-8}\right)\cdot 7\cdot \left({-1}\right)\Rightarrow \left({x;y}\right)=\left({-7;9}\right).$
\item[+)] $56=7\cdot \left({-8}\right)\cdot \left({-1}\right)\Rightarrow\left({x;y}\right)=\left({8;-6}\right).$

\end{itemize}
Vậy phương trình có 6 nghiệm nguyên như trên.

} 
\end{ex}
\begin{ex}%[Đề thi HSG, Thanh Hóa, năm 2018]%[Nguyễn Tất Thu, Dự án EX-9G-18]%[9D0G4]
Cho $a,\ b$ là các số nguyên dương thỏa mãn $p=a^2+b^2$ là các số nguyên tố và $p-5$ chia hết cho $8$. Giả sử $x,\ y$ là các số nguyên thỏa mãn $ax^2-by^2$ chia hết cho $p$. Chứng minh rằng cả hai số $x,\ y$ chia hết cho $p$.
\loigiai{
Do $p-5\vdots 8$ nên $p=8k+5(k\in \mathbb{N})$.\\
Vì $$\left({ax^2}\right)^{4k+2}-\left({by^2}\right)^{4k+2}\vdots \left({ax^2-by^2}\right)\ \vdots\ p,$$ nên $$a^{4k+2}\cdot x^{8k+4}-b^{4k+2}\cdot y^{8k+4}\ \vdots\ p.$$
Nhận thấy $$a^{4k+2}\cdot x^{8k+4}-b^{4k+2}\cdot y^{8k+4}=\left({a^{4k+2}+b^{4k+2}}\right)x^{8k+4}-b^{4k+2}\left({x^{8k+4}+y^{8k+4}}\right). $$
Do $$a^{4k+2}+b^{4k+2}=\left({a^2}\right)^{2k+1}+\left({b^2}\right)^{2k+1}\ \vdots\ \left({a^2+b^2}\right)=p, $$ và $b<p$ nên $x^{8k+4}+y^{8k+4}\ \vdots\ p$. \hfill (*)\\
Nếu trong hai số $x,\ y$ có một số chia hết cho $p$ thì từ (*) suy ra số thứ hai cũng chia hết cho $p$.\\
Nếu cả hai số $x,\ y$ đều không chia hết cho $p$ thì theo định lí Fecma ta có :
$$x^{8k+4}=x^{p-1}\equiv 1 \pmod p,\ y^{8k+4}=y^{p-1}\equiv 1 \pmod p  \Rightarrow x^{8k+4}+y^{8k+4}\equiv 2 \pmod p.$$ Mâu thuẫn với (*).\\ 
Vậy cả hai số $x$ và $y$ chia hết cho $p$.

} 
\end{ex}
\begin{ex}%[Đề thi HSG, Thanh Hóa, năm  2018]%[9H0B1]%[9H0K1]%[9H0K1]

Cho tam giác $ABC$ có $(O),\ (I),\ (I_a)$ theo thứ tự là các đường tròn ngoại tiếp, đường tròn nội tiếp và đường tròn bàng tiếp đối diện đỉnh $A$ của tam giác với các tâm tương ứng là $O,\ I,\ I_a.$ Gọi $D$ là tiếp điểm của $(I)$ với $BC$, $P$ là điểm chính giữa cung  $\wideparen{BAC}$ của $(O)$, $PI_a$ cắt $(O)$ tại điểm $K$. Gọi $M$ là giao điểm $PO$ và $BC$, $N$ là điểm đối xứng với $P$ qua $O$.
\begin{enumerate}
\item Chứng minh $IBI_aC$ là tứ giác nội tiếp.
\item Chứng minh $NI_a$ là tiếp tuyến của đường tròn ngoại tiếp tam giác $I_aMP$.
\item Chứng minh $\widehat{DAI}=\widehat{KAI_a}$.
\end{enumerate}

\begin{enumerate}
\item Ta có $I_a$ là tâm đường tròn bàng tiếp đối diện đỉnh $A$ và $I$ là tâm đường tròn nội tiếp tam giác $ABC$, từ đó suy ra $BI_a\perp BI,CI_a\perp CI$ ( Phân giác trong và phân giác ngoài cùng  một  góc thì vuông góc với nhau). \\
Xét tứ giác $IBI_aC$ có $\widehat{{IBI_a}}+\widehat{{ICI_a}}=180^0$.\\
Từ đó suy ra tứ giác $IBI_aC$ là tứ giác nội tiếp đường tròn đường kính $II_a$.
\item Nhận thấy  bốn điểm $A$, $I$, $N$, $I_a$ thẳng hàng (vì cùng thuộc tia phân giác của $\widehat{{BAC}}$).\\ 
Do $NP$ là đường kính của $(O)$ nên $\widehat{{NBP}}=90^0$, $M$ là trung điểm của $BC$ nên $PN\perp BC$ tại $M$.\\
Áp dụng hệ thức lượng trong tam giác vuông $PBN$ ta có $NB^2=NM\cdot NP$.\\
Vì $\widehat{BIN}$ là góc ngoài tại đỉnh $I$ của tam giác $ABI$  nên $$\widehat{BIN}=\dfrac12\left(\widehat{ABC}+\widehat{BAC}\right). \eqno (1)$$

$$\widehat{NBI}=\widehat{NBC}+\widehat{CBI}=\dfrac{1}{2}\left(\widehat{BAC}+\widehat{ABC}\right). \eqno (2)$$

$$\widehat{KAI_a}=\widehat{KAN}=\widehat{KPN}=\widehat{I_aPN}=\widehat{NI_aM}. \eqno (2) $$
Từ (1) và (2) ta có $\widehat{DAI}=\widehat{KAI_a}$.
\end{enumerate}
}

\begin{ex}%[Đề thi HSG, Thanh Hóa, năm 2018]%[Nguyễn Tất Thu, Dự án EX-9G-18]%[9D0G7]
Cho $x$, $y$, $z$ là các số thực dương thỏa mãn $x\geq z.$  Chứng minh rằng:
$$\dfrac{xz}{y^2+yz}+\dfrac{y^2}{xz+yz}+\dfrac{x+2z}{x+z}\geq \dfrac{5}{2}.$$
\loigiai{
Ta có  $$\begin{aligned} P & =\dfrac{xz}{y^2+yz}+\dfrac{y^2}{xz+yz}+\dfrac{x+2z}{x+z}=\dfrac{\dfrac{xz}{yz}}{\dfrac{y^2}{yz}+1}+\dfrac{\dfrac{y^2}{yz}}{\dfrac{xz}{yz}+1}+\dfrac{1+\dfrac{2z}x}{1+\dfrac{z} x} \\& =\dfrac{\dfrac{x} y}{\dfrac{y} z+1}+\dfrac{\dfrac{y} z}{\dfrac{x} y+1}+\dfrac{1+\dfrac{2z}x}{1+\dfrac{z} x} =\dfrac{a^2}{b^2+1}+\dfrac{b^2}{a^2+1}+\dfrac{1+2c^2}{1+c^2}, \end{aligned}$$ 
trong đó $a^2=\dfrac{x} y$, $b^2=\dfrac{y} z$, $c^2=\dfrac{z} x\ \left(a,\ b,\ c>0\right)$.\\
Nhận xét rằng $a^2\cdot b^2=\dfrac{x} z=\dfrac1{c^2}\geqslant 1\left({\text{do}\ x\geqslant z}\right).$\\
Xét $$\begin{aligned} \dfrac{a^2}{b^2+1}+\dfrac{b^2}{a^2+1}-\dfrac{2ab}{ab+1}&=\dfrac{a^2\left({a^2+1}\right)\left({ab+1}\right)+b^2\left({b^2+1}\right)\left({ab+1}\right)-2aba^2\left({a^2+1}\right)\left({b^2+1}\right)}{\left({a^2+1}\right)\left({b^2+1}\right)\left({ab+1}\right)} \\&=\dfrac{ab{\left({a^2-b^2}\right)}^2+\left({a-b}\right)\left({a^3-b^3}\right)+{\left({a-b}\right)}^2}{\left({a^2+1}\right)\left({b^2+1}\right)\left({ab+1}\right)}\geqslant 0. \end{aligned} $$
Do đó $$\dfrac{a^2}{b^2+1}+\dfrac{b^2}{a^2+1}\geqslant \dfrac{2ab}{ab+1}=\dfrac{\dfrac2c}{\dfrac1c+1}=\dfrac2{1+c}. \eqno \left(1\right).$$ Đẳng thức xảy ra khi $a=b$.\\ 
Khi đó  $$ \begin{aligned} \dfrac2{1+c}+\dfrac{1+2c^2}{c^2+1}-\dfrac52&=\dfrac{2\left({2\left({1+c^2}\right)+\left({1+c}\right)\left({1+2c^2}\right)}\right)-5\left({1+c}\right)\left({1+c^2}\right)}{2\left({1+c}\right)\left({1+c^2}\right)}\\& =\dfrac{1-3c+3c^2-c^3}{2\left({1+c}\right)\left({1+c^2}\right)}=\dfrac{{\left({1-c}\right)}^3}{2\left({1+c}\right)\left({1+c^2}\right)}\geqslant 0\ \left({\text{do}\ c\leqslant 1}\right).\ \ \left(2\right)\end{aligned}$$
Từ $\left(1\right)$ và $\left(2\right)$ suy ra điều phải chứng minh. Đẳng thức xảy ra khi $a=b,\ c=1  \Leftrightarrow x=y=z.$} 
\end{ex}
