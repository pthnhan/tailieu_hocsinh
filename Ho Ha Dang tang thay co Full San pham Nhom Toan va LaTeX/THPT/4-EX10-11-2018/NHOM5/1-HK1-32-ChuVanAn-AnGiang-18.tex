\begin{ex}%[HKI, THPT Chu Văn An, An Giang, 2017 - 2018]%[1H2K2]
	Cho đường thẳng $a$ nằm trên mặt phẳng $(P)$, đường thẳng $b$ cắt $(P)$ duy nhất tại điểm $A$ và $A$ không thuộc $a$. Kết luận nào sau đây về vị trí tương đối của $a$ và $b$ là đúng?
	\choice
	{$a$ và $b$ song song nhau}
	{\True $a$ và $b$ chéo nhau}
	{$a$ và $b$ trùng nhau}
	{$a$ và $b$ cắt nhau}
	\loigiai{
		Do $b \cap (P) = A$ nên $\left\{\begin{aligned}&A \in b \\&A \in (P) \end{aligned} \right.$. Theo giả thiết ta có $a \subset (P),~b \not\subset (P)$ và $A \notin a$.\\
		Nếu $b\equiv a$ thì từ $A \in b$ ta suy ra được $A \in a$ (mâu thuẫn với giả thiết $A \notin a$). \\
		Nếu $b\parallel a$ thì $(a,b)\equiv (a,A)\equiv (P) \Rightarrow b\subset (P)$ (mâu thuẫn giả thiết $b \not\subset (P)$). \\
		Nếu $b\cap a=B$ thì $\left\{\begin{aligned}&B \in a \subset (P)\\&B \in b  \end{aligned}\right.$. Kết hợp $\left\{\begin{aligned}&A\in (P)\\& A \in b,A \not\equiv B \end{aligned}\right.$ $\Rightarrow b\equiv AB\subset (P)$ (mâu thuẫn với  $b \not\subset (P)$). \\
		Tóm lại, chỉ có thể xảy ra trường hợp $a$ và $b$ chéo nhau.}
\end{ex}

\begin{ex}%[HKI, THPT Chu Văn An, An Giang, 2017 - 2018]%[1D1G1]
	Tìm chu kỳ tuần hoàn của hàm số $y=\cos 2x$.

	{\True $T=\pi $}

	{$T=\dfrac{\pi}{2}$}
	{$T=4\pi $}
	\loigiai{
		Với $y=f(x)=\cos 2x$ ta có $f\left(x+\pi \right)=\cos \left[2\left(x+\pi \right)\right]=\cos \left(2x+2\pi \right)=\cos 2x=f(x),\forall x \in \mathbb{R}$. \\

		Tuy nhiên, chỉ với $x=0$ thì $\cos \left[2\left(x+\alpha \right)\right]=\cos 2x \Leftrightarrow \cos(2\alpha)=1$ (*)\\

		Vậy $T=\pi $ là chu kỳ tuần hoàn của hàm số $y=\cos 2x$.}
\end{ex}

	Tìm tập xác định của hàm số $y=\dfrac{2\sin 3x}{\sin x+\cos x}$.
	\loigiai{
		Hàm số $y=\dfrac{2\sin 3x}{\sin x+\cos x}$ xác định khi và chỉ khi \\
		$\sin x+\cos x \neq 0 \Leftrightarrow \sqrt{2}\sin \left(x+\dfrac{\pi}{4}\right) \neq 0 \Leftrightarrow x+\dfrac{\pi}{4} \neq k\pi \Leftrightarrow x \neq -\dfrac{\pi}{4}+k\pi$ $(k \in \mathbb{Z})$. \\
		Vậy tập xác định của hàm số $y=\dfrac{2\sin 3x}{\sin x+\cos x}$ là $\mathscr{D}=\mathbb{R} \setminus \left\{-\dfrac{\pi}{4}+k\pi\left|\right.k \in \mathbb{Z}\right\}$.}
\end{bt}
