	Tính

		\item $\sqrt{12} + 4\sqrt{27} - \sqrt{108}-\dfrac{1}{4}\sqrt{192}$.
		\item $\dfrac{\sqrt{27} - 3\sqrt{2}}{\sqrt{3} - \sqrt{2}} + \dfrac{12}{3 + \sqrt{3}} + \dfrac{6}{\sqrt{3}}$.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			\begin{eqnarray*}
				\sqrt{12} + 4\sqrt{27} - \sqrt{108}-\dfrac{1}{4}\sqrt{192}&=&\sqrt{2^2\cdot 3}+4 \sqrt{3^2\cdot 3}-\sqrt{6^2\cdot 3}-\dfrac{1}{4}\sqrt{8^2\cdot 3}\\
				&=&2\sqrt{3}+12\sqrt{3}-6\sqrt{3}-2\sqrt{3}=6\sqrt{3}.
			\end{eqnarray*}
			\item Ta có
			\begin{eqnarray*}
				&&\dfrac{\sqrt{27} - 3\sqrt{2}}{\sqrt{3} - \sqrt{2}} + \dfrac{12}{3 + \sqrt{3}} + \dfrac{6}{\sqrt{3}}=\dfrac{\sqrt{3^2\cdot 3}-3\sqrt{2}}{\sqrt{3}-\sqrt{2}}+\dfrac{12\left(3-\sqrt{3}\right)}{\left(3+\sqrt{3}\right)\left(3-\sqrt{3}\right)}+\dfrac{6\sqrt{3}}{3}\\
				&=&\dfrac{3\left(\sqrt{3}-\sqrt{2}\right)}{\sqrt{3}-\sqrt{2}}+\dfrac{12\left(3-\sqrt{3}\right)}{6}+2\sqrt{3}=3+2\left(3-\sqrt{3}\right)+2\sqrt{3}=9.
			\end{eqnarray*}
		\end{enumerate}
	}

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Quận 1, Thành phố Hồ Chí Minh]%[Bùi Mạnh Tiến, dự án 9DT19-HK1]%[9D3B5]
	Cận thị trong học sinh có chiều hướng gia tăng. Lớp $9A$ có $40$ học sinh, trong đó $\dfrac{2}{7}$ số học sinh nam và $\dfrac{1}{4}$ số học sinh nữ không bị cận thị. Biết tổng số học sinh nam và học sinh nữ không bị cận thị là $11$. Tính số học sinh nam không bị cận thị.
	\loigiai{
		Gọi số học sinh nam không bị cận thị là $x$ ($x\in \mathbb{N}$).\\
		Gọi số học sinh nữ không bị cận thị là $y$ ($y\in \mathbb{N}$).\\
		Vì $\dfrac{2}{7}$ số học sinh nam không bị cận thị nên số học sinh nam là $\dfrac{7}{2}x$.\\
		Vì $\dfrac{1}{4}$ số học sinh nữ không bị cận thị nên số học sinh nữ là $4y$.
		Theo đầu bài ta có:
		\begin{eqnarray*}
			\heva{&\dfrac{7}{2}x+4y=40\\&x+y=11}\Rightarrow \heva{& \dfrac{7}{2}x+4(11-x)=40 \\ & y=11-4x}\Rightarrow \heva{&x=8\\&y=3.}
		\end{eqnarray*}
		Vậy số học sinh nam không bị cận thị là $8$ học sinh.
	}
\end{bt}

			&\Rightarrow & CC' = \sqrt{A'C^2-\left(A'D'^2+D'C'^2\right)}=\sqrt{13^2-3^2-4^2} = 12\text{ cm}

		Thể tích của hình hộp chữ nhật là $V=CC'\cdot S_{A'B'C'D'}=CC'\cdot D'A'\cdot D'C' = 12\cdot 3\cdot 4=144$ cm$^3$.

				Đường thẳng $y=x+3$ cắt trục hoành tại điểm $B(-3;0)$ và cắt trục tung tại điểm $M(0;3)$.
				\item Hoành độ giao điểm của $(d_1)$ và $(d_2)$ là nghiệm của phương trình $-5x+3=x+3\Rightarrow x = 0\Rightarrow y = 3$.\\

				\item Vì $(d_1)$, $(d_2)$ cắt nhau tại $M(0;3)$ thuộc trục tung nên để $(d_1)$, $(d_2)$, $(d)$ đồng quy tại một điểm thuộc trục tung thì $(d)$ phải đi qua $M$ hay

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Quận 1, Thành phố Hồ Chí Minh]%[Bùi Mạnh Tiến, dự án 9DT19-HK1]%[9H2BK]
	\immini
	{
		Từ nóc một cao ốc cao $75$ m, người ta nhìn thấy chân và đỉnh một cột ăng-ten với các góc hạ và góc nâng lần lượt là $64^\circ$ và $32^\circ$ (xem hình bên). Tính chiều cao của cột ăng-ten (làm tròn đến mét).
	}
	{
		\begin{tikzpicture}
			\tkzDefPoints{0/0/A}
			\coordinate (E) at ($(A)+(2,0)$);
			\coordinate (x) at ($(E)+(0,1)$);
			\tkzDefPointBy[rotation = center A angle 32](E)    \tkzGetPoint{M}
			\tkzDefPointBy[rotation = center A angle -64](E)    \tkzGetPoint{N}
			\tkzInterLL(E,x)(A,M)    \tkzGetPoint{D}
			\tkzInterLL(E,x)(A,N)    \tkzGetPoint{C}
			\tkzDefPointBy[translation = from E to C](A)    \tkzGetPoint{B}
			\tkzDrawSegments(A,B B,C C,D D,A A,E A,C)
			\tkzDrawPoints[fill=black](A,B,C,D,E)
			\tkzLabelPoints[left](A,B)
			\tkzLabelPoints[right](C,D,E)
			\tkzMarkAngles[size=0.5cm,arc=l](E,A,D)
			\tkzMarkAngles[size=0.4cm,arc=ll](C,A,E)
			\draw ($(A)+(0.9,0.2)$) node {$32^\circ$};
			\draw ($(A)+(0.7,-0.35)$) node {$64^\circ$};
		\end{tikzpicture}
	}
	\loigiai{
		Chiều cao tòa cao ốc là $75$ m nên $AB = EC = 75$ m.\\
		Tam giác $AEC$ vuông tại $E$ nên: $\tan \widehat{CAE} = \dfrac{EC}{AE}\Rightarrow AE = \dfrac{EC}{\tan \widehat{CAE}}$.\\
		Tam giác $AED$ vuông tại $E$ nên: $\tan \widehat{DAE}=\dfrac{DE}{AE}\Rightarrow DE = AE\cdot \tan \widehat{DAE} = EC\cdot \dfrac{\tan \widehat{DAE}}{\tan \widehat{CAE}}.$\\
		Độ dài cột ăng-ten là: $CD=CE+ED=75+75\cdot \dfrac{\tan 32^\circ}{\tan 64^\circ}\approx 98$ m.
	}
\end{bt}

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Quận 1, Thành phố Hồ Chí Minh]%[Bùi Mạnh Tiến, dự án 9DT19-HK1]%[9D3B3]
	Cho đường tròn $(O;R)$, đường kính $AB$. Qua điểm $M$ thuộc đường tròn ($M$ khác $A$ và $B$), vẽ tiếp tuyến với đường tròn cắt các tiếp tuyến tại $A$ và $B$ lần lượt tại $C$ và $D$.
	\begin{enumerate}
		\item Chứng minh rằng $AC+BD=CD$ và $\widehat{COD}=90^\circ$.
		\item Gọi $E$ là giao điểm của $AM$ và $OC$, $F$ là giao điểm của $BM$ và $OD$. Chứng minh tứ giác $MEOF$ là hình chữ nhật.
		\item Cho biết $AC=\dfrac{2}{3}R$. Tính độ dài $AD$ theo $R$.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[line join = round, line cap = round,>=stealth,font=\footnotesize,scale=1] 
			\pgfresetboundingbox 
			\tkzDefPoints{0/0/A}

			\def\b{6}
			\coordinate (B) at ($(A)+(\b,0)$);
			\tkzDefMidPoint(A,B)\tkzGetPoint{O}
			\tkzDefPointBy[rotation = center O angle \gocquay](B)    \tkzGetPoint{M}
			\tkzDefLine[perpendicular=through M](O,M) \tkzGetPoint{d}.
			\tkzDefLine[perpendicular=through A](O,A) \tkzGetPoint{x}.
			\tkzDefLine[perpendicular=through B](O,B) \tkzGetPoint{y}.
			
			\tkzInterLL(A,x)(M,d)    \tkzGetPoint{C}
			\tkzInterLL(B,y)(M,d)    \tkzGetPoint{D}
			\tkzInterLL(A,M)(O,C)    \tkzGetPoint{E}
			\tkzInterLL(B,M)(O,D)    \tkzGetPoint{F}
			\tkzDrawCircle[radius](O,A)
			\tkzDrawSegments(A,B O,M C,D A,C B,D A,M M,B O,C O,D)
			\tkzDrawPoints[fill=black](A,B,C,D,M,O,E,F)
			\tkzLabelPoints[left](A,C)
			\tkzLabelPoints[right](B,D)
			\tkzLabelPoints[below](O,E,F)
			\tkzLabelPoints[above](M)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Vì $CA$, $CM$ là tiếp tuyến của đường tròn $(O,R)$ nên $CM=CA$.\\
			Vì $DB$, $DM$ là tiếp tuyến của đường tròn $(O;R)$ nên $DB=DM$.\\
			Do đó $AC+BD=CM+DM=CD$.
			\item Vì $CM=CA$ và $OM=OA$ nên $OC$ là trung trực của $AM$,suy ra $E$ là trung điểm của $AM$ và $OE\perp EM\Rightarrow \widehat{OEA}=90^\circ$.\\
			Vì $OM = OB$ và $DM=DB$ nên $OD$ là đường trung trực của $MB$,suy ra $OD\perp MB\Rightarrow \widehat{OFM}=90^\circ$.\\
			Tam giác $AMB$ có $O$ và $E$ là trung điểm của $AB$ và $AM$ nên $OE$ là đường trung bình của $\triangle AMB\Rightarrow OE\parallel MB \Rightarrow AM\perp MB\Rightarrow \widehat{EMF}=90^\circ$.\\
			Tứ giác $MEOF$ có $\widehat{OME}=\widehat{EMF}=\widehat{MFO}=90^\circ\Rightarrow MEOF$ là hình chữ nhật.
			\item Vì $MEOF$ là hình chữ nhật nên $\widehat{EOF}=90^\circ\Rightarrow \widehat{COA}+\widehat{DOB}=90^\circ$.\\
			Tam giác $CAO$ vuông tại $A$ nên $\widehat{COA}+\widehat{OCA}=90^\circ\Rightarrow \widehat{OCA}=\widehat{DOB}$.\\
			Xét $\triangle OCA$ và $\triangle DOB$ có $\heva{&\widehat{OAC}=\widehat{OBD}=90^\circ\\&\widehat{OCA}=\widehat{DOB}}\Rightarrow \triangle OCA\backsim
			 \triangle DOB$ (g.g).\\
			Suy ra $\dfrac{OA}{AC}=\dfrac{DB}{OB}\Rightarrow DB=\dfrac{OA\cdot OB}{AC}=\dfrac{R\cdot R}{\dfrac{2}{3}R}=\dfrac{3R}{2}$.\\
			Vì tam giác $ADB$ vuông tại $B$ (do $BD$ là tiếp tuyến của $(O;R)$) nên theo định lí Pitago ta có:
			\begin{align*}
				AD^2=AB^2+BD^2=(2R)^2+\left(\dfrac{3R}{2}\right)^2=\dfrac{25R^2}{4}\Rightarrow AD = \dfrac{5R}{2}.
			\end{align*}
		\end{enumerate}
