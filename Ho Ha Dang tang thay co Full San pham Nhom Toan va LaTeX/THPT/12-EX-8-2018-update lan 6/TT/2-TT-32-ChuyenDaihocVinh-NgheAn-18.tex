		& y=\dfrac{34}{3}+2t \\
		& z=-\dfrac{34}{3}+t \\
		& 2x+2y+z-12=0}\Leftrightarrow \heva{& x=2 \\
		& y=10 \\
		& z=-\dfrac{38}{3} \\
		& t=-\dfrac{2}{3}}$.\\
		Vậy $J=\left(2;10;-\dfrac{38}{3}\right)$.
	}
\end{ex}

		& z=a-t \\
