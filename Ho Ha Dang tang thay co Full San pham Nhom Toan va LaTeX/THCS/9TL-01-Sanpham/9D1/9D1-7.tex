\section{CĂN BẬC BA}
\subsection{Kiến thức cần nhớ}
\begin{dn}
Cho $ a, b \in \mathbb{R}, \sqrt[3]{a} = b \Leftrightarrow a = b^3.$
\end{dn}

\begin{tc}
Cho $ a, b \in \mathbb{R}, b \neq 0 $. Khi đó ta có
\begin{enumerate}
	\item $ \sqrt[3]{ab} = \sqrt[3]{a} \cdot \sqrt[3]{b} $.
	\item $ \sqrt[3]{\dfrac{a}{b}} = \dfrac{\sqrt[3]{a}}{\sqrt[3]{b}} $.
	\item $ a \leqslant b \Leftrightarrow \sqrt[3]{a} \leqslant \sqrt[3]{b} $.
\end{enumerate}
\end{tc}
     
\subsection{Các dạng toán}

\begin{dang}{Tìm căn bậc ba của một số}
    Phương pháp giải\\
    Dùng các phép biến đổi đưa biểu thức dưới dấu căn về dạng $ a^3 $ rồi tính.
\end{dang}

\begin{vd}
    Tính $ \sqrt[3]{8} $; $ \sqrt[3]{-64} $.
    \loigiai{
        \begin{align*}
        	& \sqrt[3]{8} = \sqrt[3]{2^3} = 2\\
        	& \sqrt[3]{-64} = \sqrt[3]{\left(-4\right)^3} = -4
        \end{align*}
    }
\end{vd}

\begin{vd}
    Tính $ \sqrt[3]{\dfrac{1}{0{,}008}} $; $ \sqrt[3]{\left(-27\right) \cdot 8} $.
    \loigiai{
        \begin{align*}
        	& \sqrt[3]{\dfrac{1}{0{,}008}} = \dfrac{\sqrt[3]{1}}{\sqrt[3]{0{,}008}} = \dfrac{\sqrt[3]{1^3}}{\sqrt[3]{0{,}2^3}} = \dfrac{1}{0{,}2} = 5.\\
        	& \sqrt[3]{\left(-27\right) \cdot 8 } = \sqrt[3]{\left(-27\right)} \cdot \sqrt[3]{8} = \sqrt[3]{\left(-3\right)^3} \cdot \sqrt[3]{2^3} = -3 \cdot 2 = -6.
        \end{align*}
    }
\end{vd}

\begin{vd}
	Tính $ \dfrac{\sqrt[3]{36} \cdot \sqrt[3]{12}}{\sqrt[3]{16}} $.
	\loigiai{
	\begin{align*}
		& \dfrac{\sqrt[3]{36} \cdot \sqrt[3]{12}}{\sqrt[3]{16}} = \sqrt[3]{\dfrac{36 \cdot 12}{16}} = \sqrt[3]{\dfrac{3^2 \cdot 4 \cdot 4 \cdot 3 }{4^2}} = \sqrt[3]{3^3} = 3.
	\end{align*}
	}
\end{vd}

\begin{dang}{So sánh các căn bậc ba}
    Phương pháp giải
    \begin{itemize}
    	\item Đưa hai biểu tức cần so sánh về dạng cơ bản $ \sqrt[3]{a} \leqslant \sqrt[3]{b} $.
    	\item Áp dụng tính chất cơ bản suy ra kết quả.
    \end{itemize}
    \begin{note}
        Những bài không thể đưa ngay về dạng cơ bản thì có thể lập phương hai biểu thức đã cho rồi so sánh: Nếu $ a^3 \geqslant b^3 \Leftrightarrow a \geqslant b $.
    \end{note}
\end{dang}

\begin{vd}
    So sánh $ \sqrt[3]{5} $ và $ \sqrt[3]{4} $.
    \loigiai{
		Ta có $ 4 < 5 \Rightarrow \sqrt[3]{4} < \sqrt[3]{5} $.
		}
\end{vd}

\begin{vd}
    So sánh $ \sqrt[3]{5} $ và $ 2 $.
    \loigiai{
		Ta có $ 8 > 5 \Rightarrow \sqrt[3]{8} > \sqrt[3]{5} \Rightarrow 2 > \sqrt[3]{5} $.
		}
\end{vd}

\begin{vd}
    So sánh $ 4\sqrt[3]{5} $ và $ 5\sqrt[3]{4} $.
    \loigiai{
		Ta có 
		\begin{align*}
		5\sqrt[3]{4} = \sqrt[3]{5^3 \cdot 4} = \sqrt[3]{500}\\
		4\sqrt[3]{5} = \sqrt[3]{4^3 \cdot 5} = \sqrt[3]{320}
\end{align*} 
Suy ra $ 	 \sqrt[3]{320} < \sqrt[3]{500} \Leftrightarrow 4\sqrt[3]{5} < 5\sqrt[3]{4} $.
		}
\end{vd}

\begin{vd}
    So sánh $ \sqrt[3]{5} + \sqrt[3]{7} $ và $ \sqrt[3]{12} $.
    \loigiai{
		Ta có
		\begin{align*}
		& \left(\sqrt[3]{5} + \sqrt[3]{7}\right)^3 = 5 + 3\sqrt[3]{5^2 \cdot 7} + 3\sqrt[3]{5\cdot 7^2} + 7 = 12 + 3\sqrt[3]{175} + 3\sqrt[3]{245} > 12 = \left(\sqrt[3]{12}\right)^3\\
		& \Rightarrow \sqrt[3]{5} + \sqrt[3]{7} > \sqrt[3]{12}.
\end{align*}}
\end{vd}

\begin{dang}{Rút gọn biểu thức chứa căn bậc ba}
	Phương pháp giải\\
	Dùng các phép biến đổi đưa biểu thức dưới dấu căn về dạng $ a^3 $ hoặc mũ $3$ cả hai vế của biểu thức đưa về giải phương trình bậc ba.
\end{dang}

\begin{vd}
	Rút gọn biểu thức sau
	\begin{multicols}{2}
		\begin{enumerate}
			\item $A=\sqrt[3]{7+5\sqrt{2}}+\sqrt[3]{7-5\sqrt{2}}$.
			\item $B=\sqrt[3]{72-32\sqrt{5}}\cdot \sqrt{7+3\sqrt{5}}$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item 
			$\begin{aligned}[t]
				A&=\sqrt[3]{\left(\sqrt{2}\right)^3+3\left(\sqrt{2}\right)^2\cdot 1+3\sqrt{2}\cdot 1^2+1^3}+\sqrt[3]{\left(\sqrt{2}\right)^3-3\left(\sqrt{2}\right)^2\cdot 1+3\sqrt{2}\cdot 1^2-1^3}\\
				&=\sqrt[3]{\left(\sqrt{2}+1\right)^3}+\sqrt[3]{\left(\sqrt{2}-1\right)^3}=\left(\sqrt{2}+1\right)+\left(\sqrt{2}-1\right)=2\sqrt{2}.
			\end{aligned}$
			\item 
			$\begin{aligned}[t]
				B&=\sqrt[3]{(3)^3-3\cdot (3)^2 \cdot \sqrt{5}+3 \cdot 3\cdot \left(\sqrt{5}\right)^2-\left(\sqrt{5}\right)^3}\cdot \dfrac{\sqrt{14+6\sqrt{5}}}{\sqrt{2}}\\
				&=\sqrt[3]{\left(3-\sqrt{5}\right)^3}\cdot \dfrac{\sqrt{\left(3+\sqrt{5}\right)^2}}{\sqrt{2}} = \dfrac{\left(3-\sqrt{5}\right)\left(3+\sqrt{5}\right)}{\sqrt{2}}=2\sqrt{2}.
			\end{aligned}$
		\end{enumerate}
	}
\end{vd}

\begin{vd}
	Rút gọn biểu thức sau $A=\sqrt[3]{2+\sqrt{5}}+\sqrt[3]{2-\sqrt{5}}$.
	\loigiai{
		$A^3=2+\sqrt{5}+2-\sqrt{5}+3\sqrt[3]{2+\sqrt{5}}\cdot \sqrt[3]{2-\sqrt{5}}\left(\sqrt[3]{2+\sqrt{5}}+\sqrt[3]{2-\sqrt{5}}\right)=4-3A$.\\
		$\Rightarrow A^3+3A-4=0\Leftrightarrow (A-1)(A^2+A+4)=0\Leftrightarrow A=1$ (do $A^2+A+4=\left(A+\dfrac{1}{2}\right)^2+\dfrac{15}{4}>0,\forall A$)
	}
\end{vd}

\begin{dang}{Giải phương trình chứa căn bậc ba}
	Phương pháp giải\\
	Lũy thừa bậc ba hai vế của phương trình rồi đưa về dạng phương trình tích hoặc ta có thể đặt ẩn phụ.
\end{dang}

\begin{vd}
	Giải phương trình $\sqrt[3]{x+7}-3=1$
	\loigiai{
		$\sqrt[3]{x+7}-3=1\Leftrightarrow \sqrt[3]{x+7}=4\Leftrightarrow x+7=64\Leftrightarrow x=57$.
	}
\end{vd}

\begin{vd}
	Giải phương trình $\sqrt[3]{x}+\sqrt[3]{x+1}+\sqrt[3]{x+2}=0$.
	\loigiai{
		Phương trình tương đương với $\sqrt[3]{x}+\sqrt[3]{x+1}=-\sqrt[3]{x+2}$ \hfill(1)\\
		Lũy thừa bậc ba cả hai vế ta được 
		\begin{align*}
			&x+x+1+3\sqrt[3]{x(x+1)}\left(\sqrt[3]{x}+\sqrt[3]{x+1}\right)=-x-2\\
			\Leftrightarrow & 3x+3+3\sqrt[3]{x(x+1)}\left(\sqrt[3]{x}+\sqrt[3]{x+1}\right)=0\\
			\Rightarrow & x+1-\sqrt[3]{x(x+1)(x+2)}=0\\
			\Leftrightarrow & x(x+1)(x+2)=(x+1)^3\\
			\Leftrightarrow & x=-1.
		\end{align*}
		Thử lại ta thấy $x=-1$ là nghiệm của phương trình.
	}
\end{vd}
	
\begin{vd}
	Giải phương trình $\sqrt[3]{2+x}+\sqrt[3]{5-x}=1$.
	\loigiai{
		Đặt $a=\sqrt[3]{2+x}$, $b=\sqrt[3]{5-x}$.\\
		Ta có 
		\begin{eqnarray*}
			\heva{&a+b=1\\&a^3+b^3=2+x+5-x=7}&\Leftrightarrow &  \heva{&a+b=1\\&(a+b)(a^2-ab+b^2)=7}\\
			&\Leftrightarrow& \heva{&a+b=1\\&(a+b)^2-3ab=7}\\
			&\Leftrightarrow& \heva{&a+b=1\\&ab=-2}
		\end{eqnarray*}
		Khi đó $a,b$ là nghiệm của phương trình $X^2-X-2=0\Leftrightarrow \hoac{&X=-1\\&X=2}\Rightarrow \hoac{&\heva{&a=-1\\&b=2}\quad(1)\\&\heva{&a=2\\&b=-1}\quad(2)}$
		\begin{itemize}
			\item $(1)\Leftrightarrow \heva{&\sqrt[3]{2+x}=-1\\&\sqrt[3]{5-x}=2}\Leftrightarrow x=-3$.
			\item $(2)\Leftrightarrow \heva{&\sqrt[3]{2+x}=2\\&\sqrt[3]{5-x}=-1}\Leftrightarrow x=6$.
		\end{itemize}
		Vậy phương trình có hai nghiệm là $x=-3$ và $x=6$.
	}
\end{vd}

\subsection{Luyện tập}

\begin{bt}
Tính các căn bậc ba sau
\begin{multicols}{2}
\begin{enumerate}
	\item $ \sqrt[3]{64} $.
	\item $ \sqrt[3]{-512} $.
	\item $ \sqrt[3]{0{,}064} $.
	\item $ \sqrt[3]{-0{,}216} $.
	\item $ \dfrac{\sqrt[3]{500}}{\sqrt[3]{4}} + \sqrt[3]{12}\cdot \sqrt[3]{18} $.
	\item $ \dfrac{\sqrt[3]{12}\cdot \sqrt[3]{6}}{\sqrt[3]{576}} - \dfrac{\sqrt[3]{32}}{\sqrt[3]{4}} $.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
	\item $ \sqrt[3]{64} = 4 $.
	\item $ \sqrt[3]{-512} = -8 $.
	\item $ \sqrt[3]{0{,}064} = \sqrt[3]{\dfrac{64}{1000}} = \dfrac{\sqrt[3]{64}}{\sqrt[3]{1000}} = \dfrac{4}{10} $.
	\item $ \sqrt[3]{-0{,}216}  = \sqrt[3]{(-6)^3} = -6 $.
	\item $ \dfrac{\sqrt[3]{500}}{\sqrt[3]{4}} + \sqrt[3]{12}\cdot \sqrt[3]{18} = \sqrt[3]{\dfrac{500}{4}} + \sqrt[3]{12 \cdot 18} = \sqrt[3]{125} + \sqrt[3]{216} = 5 + 6 = 11 $.
	\item $ \dfrac{\sqrt[3]{12}\cdot \sqrt[3]{6}}{\sqrt[3]{576}} - \dfrac{\sqrt[3]{32}}{\sqrt[3]{4}} = \sqrt[3]{\dfrac{12\cdot 6}{576}} - \sqrt[3]{\dfrac{32}{4}} = \sqrt[3]{\dfrac{1}{8}} - \sqrt[3]{8} = \dfrac{1}{2} - 2 = -\dfrac{3}{2} $.
\end{enumerate}
}
\end{bt}

\begin{bt}
So sánh các biểu thức sau
\begin{multicols}{2}
\begin{enumerate}
	\item $ \sqrt[3]{9} $ và $ 2 $.
	\item $ \sqrt[3]{\dfrac{1}{8}} $ và $ \dfrac{3}{4} $.
	\item $ 2\sqrt[3]{3} $ và $ 3\sqrt[3]{2} $.
	\item $ -6\sqrt[3]{7} $ và $ 7\sqrt[3]{\left(-6\right)} $.
	\item $ \sqrt[3]{4} + \sqrt[3]{7} $ và $ \sqrt[3]{11} $.
	\item $ \sqrt[3]{10} - 2 $ và $ \sqrt[3]{2} $.
	\item $ \sqrt{2} + 1 $ và $ \sqrt[3]{7 + 5\sqrt{2}} $.
	\item $ \sqrt{3} - 2 $ và $ \sqrt[3]{15\sqrt{3} - 25} $.
\end{enumerate}
\end{multicols}

\loigiai{
\begin{enumerate}
	\item $ \sqrt[3]{9} > 2 $.
	\item $ \sqrt[3]{\dfrac{1}{8}} < \dfrac{3}{4} $.
	\item $\heva{& 2\sqrt[3]{3} = \sqrt[3]{2^3 \cdot 3} = \sqrt[3]{24} \\ & 3\sqrt[3]{2} = \sqrt[3]{3^3 \cdot 2} = \sqrt[3]{54}} \Rightarrow 2\sqrt[3]{3} < 3\sqrt[3]{2}.$
	\item 
	 $\heva{& -6\sqrt[3]{7} = \sqrt[3]{(-6)^3 \cdot 7} = \sqrt[3]{-1512} \\
	& 7\sqrt[3]{\left(-6\right)} = \sqrt[3]{7^3 \cdot (-6)} = \sqrt[3]{-2058}}   \Rightarrow -6\sqrt[3]{7} > 7\sqrt[3]{\left(-6\right)}.$
	\item 
	$\begin{aligned}[t]
		& \left(\sqrt[3]{4} + \sqrt[3]{7}\right)^3 = 4 + 3\sqrt[3]{112} + 3\sqrt[3]{196} + 7  = 11 + 3\sqrt[3]{112} + 3\sqrt[3]{196} > 11 = \left(\sqrt[3]{11}\right)^3\\
		\Rightarrow &\sqrt[3]{4} + \sqrt[3]{7} > \sqrt[3]{11}.
	\end{aligned}$
	\item $\begin{aligned}[t]
	&\heva{&\left(\sqrt[3]{10}\right)^3 = 10\\&\left(\sqrt[3]{2} + 2 \right)^3 = 2 + 6\sqrt[3]{4} + 12\sqrt[3]{2} + 8 = 10 + 6\sqrt[3]{4} + 12\sqrt[3]{2}}\\
	\Rightarrow &\left(\sqrt[3]{10}\right)^3 < \left(\sqrt[3]{2} + 2 \right)^3 \Rightarrow \sqrt[3]{10} < \sqrt[3]{2} + 2 \Leftrightarrow \sqrt[3]{10} - 2 < \sqrt[3]{2}.
	\end{aligned}$
	\item
	$\left( \sqrt{2} + 1 \right)^3 = 2\sqrt{2} + 6 + 3\sqrt{2} + 1 = 7 + 5\sqrt{2} \Rightarrow \sqrt{2} + 1 = \sqrt[3]{7 + 5\sqrt{2}}.$
	\item $\begin{aligned}[t]
	& \left(\sqrt{3} - 2\right)^3 = 3\sqrt{3} - 18 + 12\sqrt{3} - 8 = 15\sqrt{3} - 26 < 15\sqrt{3} - 25\\
	\Rightarrow &\sqrt{3} - 2 < \sqrt[3]{15\sqrt{3} - 25}.
	\end{aligned}$
\end{enumerate}
}
\end{bt}

\begin{bt}
	Rút gọn các biểu thức sau:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $A=\sqrt[3]{20+14\sqrt{2}}+\sqrt[3]{20-14\sqrt{2}}$.
			\item $B=\sqrt[3]{182+\sqrt{33125}}+\sqrt[3]{182-\sqrt{33125}}$.
		\end{enumerate}	
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item 
			$\begin{aligned}[t]
				A&=\sqrt[3]{2^3+3\cdot 2^2\left(\sqrt{2}\right)+3\cdot 2\cdot \left(\sqrt{2}\right)^2+\left(\sqrt{2}\right)^3}+\sqrt[3]{2^3-3\cdot 2^2\left(\sqrt{2}\right)+3\cdot 2\cdot \left(\sqrt{2}\right)^2-\left(\sqrt{2}\right)^3}\\
				&=\sqrt[3]{\left(2+\sqrt{2}\right)^3}+\sqrt[3]{\left(2-\sqrt{2}\right)^3}=2+\sqrt{2}+2-\sqrt{2}=4.
			\end{aligned}$
			\item $B^3=182+\sqrt{33125}+182-\sqrt{33125}+3\sqrt[3]{182^2-33125}B=364-3B$.\\
			Nên $B^3+3B-364=0\Leftrightarrow (B-7)(B^2+7B+52)=0\Leftrightarrow B=7$ \\\Bigg(do $B^2+7B+52=\left(B+\dfrac{7}{2}\right)^2+ \dfrac{159}{4}>0$\Bigg).
		\end{enumerate}
	}
\end{bt}

\begin{bt}
	Rút gọn các biểu thức sau:
	\begin{enumerate}
		\item $A=\left[\left(\dfrac{1}{a}-\sqrt[6]{\dfrac{1}{a}}+\sqrt[3]{a^2}\right)+\left(\dfrac{a}{a^2}\sqrt[6]{a^5}-\dfrac{3}{a}\sqrt[3]{a^2}\right)\right] \cdot a\sqrt[3]{a}$.
		\item $B=\sqrt{a^2+\sqrt[3]{a^4b^2}}+\sqrt{b^2+\sqrt[3]{a^2b^4}}.$
	\end{enumerate}	
	\loigiai{
		\begin{enumerate}
			\item Ta có
			$\begin{aligned}[t]
				A&=\left[\left(\dfrac{1}{a}-\sqrt[6]{\dfrac{1}{a}}+\sqrt[3]{a^2}\right)+\left(\dfrac{a}{a^2}\sqrt[6]{a^5}-\dfrac{3}{a}\sqrt[3]{a^2}\right)\right]a\sqrt[3]{a}\\
				&= \left[\left(\dfrac{1}{a}-\sqrt[6]{\dfrac{1}{a}}+\sqrt[3]{a^2}\right)+\left(\sqrt[6]{\dfrac{1}{a}}-3\sqrt[3]{\dfrac{1}{a}}\right)\right]a\sqrt[3]{a}\\
				&= \left(\dfrac{1}{a}+\sqrt[3]{a^2}-3\sqrt[3]{\dfrac{1}{a}}\right)a\sqrt[3]{a}\\
				&= a^2-3a+\sqrt[3]{a}.
			\end{aligned}$
			\item Ta có
			$\begin{aligned}[t]
				B&=\sqrt[3]{a^2}\sqrt{\sqrt[3]{a^2}+\sqrt[3]{b^2}}+\sqrt[3]{b^2}\sqrt{\sqrt[3]{a^2}+\sqrt[3]{b^2}}\\
				&=\left(\sqrt[3]{a^2}+\sqrt[3]{b^2}\right)\sqrt{\sqrt[3]{a^2}+\sqrt[3]{b^2}}\\
				&=\sqrt{\left(\sqrt[3]{a^2}+\sqrt[3]{b^2}\right)^3}.
			\end{aligned}$
		\end{enumerate}
	}
\end{bt}

\begin{bt}
	Giải các phương trình sau
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\sqrt[3]{1000x}-\sqrt[3]{64x}-\sqrt[3]{27x}=15$
			\item $\sqrt[3]{x-3}+3=x$
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item $\sqrt[3]{1000x}-\sqrt[3]{64x}-\sqrt[3]{27x}=15\Leftrightarrow10\sqrt[3]{x}-4\sqrt[3]{x}-3\sqrt[3]{x}=15\Leftrightarrow \sqrt[3]{x}=5\Leftrightarrow x=125.$\\
			Vậy phương trình có một nghiệm $x=125$.
			\item Ta có
			$\begin{aligned}[t]
				\sqrt[3]{x-3}+3=x&\Leftrightarrow \sqrt[3]{x-3}=x-3\Leftrightarrow x-3=(x-3)^3\\
				&\Leftrightarrow  (x-3)\left[(x-3)^2-1\right]=0\\
				&\Leftrightarrow  (x-3)(x-2)(x-4)=0\\
				&\Leftrightarrow  \hoac{&x=3\\&x=2\\&x=4.}
			\end{aligned}$\\
			Vậy tập nghiệm của phương trình là $S=\left\{2;3;4\right\}.$
		\end{enumerate}
	}
\end{bt}

\begin{bt}
	Giải các phương trình sau
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\sqrt[3]{2x-1}+\sqrt[3]{x-1}=\sqrt[3]{3x-2}.$
			\item $\sqrt[3]{x+5}+\sqrt[3]{x+6}=\sqrt[3]{2x+11}.$
		\end{enumerate}	
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			$\begin{aligned}[t]
				&\sqrt[3]{2x-1}+\sqrt[3]{x-1}=\sqrt[3]{3x-2}\\
				&\Leftrightarrow  2x-1+x-1+3\sqrt[3]{(2x-1)(x-1)}\left(\sqrt[3]{2x-1}+\sqrt[3]{x-1}\right)=3x-2\\
				&\Rightarrow  3\sqrt[3]{(2x-1)(x-1)(3x-2)}=0
				\Leftrightarrow \hoac{&x=\dfrac{1}{2}\\&x=1\\&x=\dfrac{2}{3}.}
			\end{aligned}$\\
			Thử lại ta thấy các nghiệm đều thỏa mãn phương trình.\\
			Vậy tập nghiệm của phương trình là $S=\left\{\dfrac{1}{2};1;\dfrac{2}{3}\right\}.$
			\item Ta có
			$\begin{aligned}[t]
				&\sqrt[3]{x+5}+\sqrt[3]{x+6}=\sqrt[3]{2x+11}\\
				\Leftrightarrow &x+5+x+6+3\sqrt[3]{(x+5)(x+6)}\left(\sqrt[3]{x+5}+\sqrt[3]{x+6}\right)=2x+11\\
				\Rightarrow &3\sqrt[3]{(x+5)(x+6)(2x+11)}=0
				\Leftrightarrow \heva{&x=-5\\&x=-6\\&x=-\dfrac{11}{2}.}
			\end{aligned}$\\
			Thử lại ta thấy các nghiệm đều thỏa mãn phương trình.\\
			Vậy tập nghiệm của phương trình là $S=\left\{-5;-\dfrac{11}{2};-6\right\}.$
		\end{enumerate}
	}
\end{bt}
