\begin{ex}%[Thi vào 10, Sở GD&ĐT Hà Nội, 2015 - 2016]%[9D1B7]%[9D1K7]%%[Nguyễn Bình Nguyên]
	Cho hai biểu thức $P=\dfrac{x+3}{\sqrt{x}-2}$ và $Q=\dfrac{\sqrt{x}-1}{\sqrt{x}+2}+\dfrac{5\sqrt{x}-2}{x-4}$ với $x>0$, $x\neq 4$.
	\begin{enumerate}
		\item Tính giá trị của biểu thức $P$ khi $x=9$.
		\item Rút gọn biểu thức $Q$.
		\item Tìm giá trị của $x$ để $\dfrac{P}{Q}$ đạt giá trị nhỏ nhất.
	\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
	\item \textbf{Tính giá trị của biểu thức $P$ khi $x=9$}\\
	Thay $x=9$ vào $P=\dfrac{9+3}{\sqrt{9}-2}= \dfrac{12}{\sqrt{9}-2}=\dfrac{12}{3-2}=12$.
	\item \textbf{Rút gọn biểu thức $Q$}\\
	\begin{align*}
Q=&\dfrac{\sqrt{x}-1}{\sqrt{x}+2}+\dfrac{5\sqrt{x}-2}{x-4}\\
=&\dfrac{\left(\sqrt{x}-1\right)\left(\sqrt{x}-2\right)+5\sqrt{x}-2}{x-4}\\
=&\dfrac{x+2\sqrt{x}}{\left(\sqrt{x}-2\right)\left(\sqrt{x}+2\right)}\\
=&\dfrac{\sqrt{x}}{\sqrt{x}-2}.
	\end{align*}
	\item \textbf{Tìm giá trị của $x$ để $\dfrac{P}{Q}$ đạt giá trị nhỏ nhất}\\
	Ta có $\dfrac{P}{Q}=\dfrac{x+3}{\sqrt{x}}=\sqrt{x}+\dfrac{3}{\sqrt{x}}$\\
	Theo bất đẳng thức Cô-si, ta có $\sqrt{x}+\dfrac{3}{\sqrt{x}} \geq 2\sqrt{3}$.\\
	Dấu bằng xảy ra khi và chỉ khi $\sqrt{x}=\dfrac{3}{\sqrt{x}}\Leftrightarrow x=3$, thỏa mãn điều kiện.\\
	Vậy giá trị nhỏ nhất của $\dfrac{P}{Q}$ là $2\sqrt{3}$, đạt được khi $x=3$.
\end{enumerate}
}
\end{ex} 
\begin{ex}%[Thi vào 10, Sở GD&ĐT Hà Nội, 2015 - 2016]%[9D4B8]%[Nguyễn Bình Nguyên]
	\textit{Giải bài toán sau bằng cách lập phương trình hoặc hệ phương trình:}\\
	Một tàu tuần tra chạy ngược dòng $ 60 $ km, sau đó chạy xuôi dòng $ 48 $ km trên cùng một dòng sông có vận tốc của dòng nước là $ 2 $ km/giờ. Tính vận tốc của tàu tuần tra khi nước yên lặng, biết thời gian xuôi dòng ít hơn thời gian ngược dòng $ 1 $ giờ.
	\loigiai{
	Gọi vận tốc của tàu tuần tra khi nước yên lặng là $x$ (km/giờ), $x>2$.\\
	Thời gian tàu tuần tra ngược dòng là $\dfrac{60}{x-2}$ (giờ).\\
	Thời gian tàu tuần tra xuôi dòng là $\dfrac{48}{x+2}$ (giờ).\\
	Ta có phương trình $\dfrac{60}{x-2}-\dfrac{48}{x+2}=1\Rightarrow x^2-12x-220=0 \Leftrightarrow \hoac{&x=22\ \text{(thỏa mãn)}\\& x=-10\ \text{(loại)}.}$\\
	Vậy vận tốc của tàu tuần tra khi nước yên lặng là $22$ km/giờ.
}
\end{ex}  

\begin{ex}%[Thi vào 10, Sở GD&ĐT Hà Nội, 2015 - 2016]%[2D4K7]%[Nguyễn Bình Nguyên]   
 Giải hệ phương trình  $\heva{& 2\left( x+y \right)+\sqrt{x+1}=4 \\ 
		& \left( x+y \right)-3\sqrt{x+1}=-5.}$  
\loigiai{
\begin{itemize}

	\item Điều kiện xác định $x\geq -1$.
	\item 	Đặt $\heva{&a=x+y\\ &b=\sqrt{x+1}\geq\ 0} \Rightarrow \heva{&2a+b=4\\&a-3b=-5} \Leftrightarrow \heva{&a=1\\&b=2.}$
	\item Từ đó ta có $\heva{&x+y=1\\ &\sqrt{x+1}=2}\Leftrightarrow \heva{&x=3\\&y=-2}$ thỏa mãn điều kiện xác định.
	\item 	Vậy hệ phương trình có nghiệm $\left(3;-2\right)$.
\end{itemize}
}
\end{ex}  
\begin{ex}%[Thi vào 10, Sở GD&ĐT Hà Nội, 2015 - 2016]%[2D4K6]%[Nguyễn Bình Nguyên]  
	Cho phương trình $x^2-\left(  m+5 \right)x+3m+6=0$ ($x$ là ẩn số).
	\begin{enumerate}
		\item Chứng minh phương trình luôn có nghiệm với mọi số thực $m$.
		\item  Tìm $m$ để phương trình có hai nghiệm $x_1$, $x_2$ là độ dài hai cạnh góc vuông của một tam giác vuông có độ dài cạnh huyền bằng $ 5 $.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item \textbf{Chứng minh phương trình luôn có nghiệm với mọi số thực $m$}\\
	Ta có $\Delta =(m+5)^2-4(3m+6)=(m-1)^2\geq 0 \quad \forall m \in \mathbb{R}$ nên phương trình luôn có nghiệm với mọi giá trị của $m$.
	\item \textbf{Tìm $m$ để phương trình có hai nghiệm $x_1$, $x_2$ là độ dài hai cạnh góc vuông của một tam giác vuông có độ dài cạnh huyền bằng $ 5 $}\\
	Ta tính được hai nghiệm là $x_1=3$, $x_2=m+2$.\\
	Theo yêu cầu bài toán ta cần có $\heva{&x_1=3>0\\&x_2=m+2>0\\&x_1^2+x_2^2=25}$\\
	Giải điều kiện trên ta được $m=2$ , (chọn) hoặc $m=-6$ (loại).\\
	Vậy $m=2$ là giá trị cần tìm.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Thi vào 10, Sở GD&ĐT Hà Nội, 2015 - 2016]%[9D5G1]%[Nguyễn Bình Nguyên]   
	Với hai số thực không âm $a$, $b$ thỏa mãn $a^2+b^2=4$, tìm giá trị lớn nhất của biểu thức $M=\dfrac{ab}{a+b+2}$.
	\loigiai{
	Ta có $a^2+b^2=4\Rightarrow 2ab=(a+b)^2-4$\\
	$ \Rightarrow 2M=\dfrac{(a+b)^2-4}{a+b+2}=a+b-2 $.\\
	Ta có $a+b\leq \sqrt{2(a^2+b^2)}=2\sqrt{2}\Rightarrow M\leq \sqrt{2}-1$.\\
	Dấu bằng xảy ra khi và chỉ khi $a=b=\sqrt{2}$.\\
	Vậy giá trị lớn nhất của $M$ bằng $\sqrt{2}$ khi $a=b=\sqrt{2}$.
}
\end{ex}
