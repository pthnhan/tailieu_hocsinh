\begin{ex}%[Đề thi HSG Toán 9, 2013-2014, Ninh Thuận]%[9D5G3]
Tìm $x,y$ thỏa mãn phương trình: $4y\sqrt{x-2}+2x\sqrt{y-1}=y(3x-2)$
	\loigiai{Điều kiện của phương trình : $x\geq 2,y\geq 1$.\\
		Ta có: \begin{align*}
		&4y\sqrt{x-2}+2x\sqrt{y-1}=y(3x-2)\Leftrightarrow 4y\sqrt{x-2}+2x\sqrt{y-1}-3xy+2y=0\\
		&\Leftrightarrow -2xy+4y\sqrt{x-2}+2y-xy+2x\sqrt{y-1}=0\\
		&\Leftrightarrow -2y(x-2\sqrt{x-2}-1)-x(y-2\sqrt{y-1})=0\\
		&\Leftrightarrow -2y(x-2-2\sqrt{x-2}+1)-x(y-1-2\sqrt{y-1}+1)=0\\
		&\Leftrightarrow -2y(\sqrt{x-2}-1)^2-x(\sqrt{y-1}-1)^2=0\\
		&\Leftrightarrow 2y(\sqrt{x-2}-1)^2+x(\sqrt{y-1}-1)^2=0\Leftrightarrow \left\{\begin{aligned}&\sqrt{x-2}-1=0\\&\sqrt{y-1}-1=0\end{aligned}\right.\Leftrightarrow\left\{ \begin{aligned}& x=3 \\	& y=2\end{aligned}\right.
		\end{align*}
		Vậy $(x;y)=(3;2)$
}
\end{ex}
\begin{ex}%[Đề thi HSG Toán 9, 2013-2014, Ninh Thuận]%[9D5G1]
	Cho các số thực $a,b,c$ không âm. Chứng minh rằng:\\
	$\left(a+b+c\right)^3\geq a^3+b^3+c^3+24abc$
	\loigiai{
	Các số $a, b, c$ không âm, nên ta có:
	\begin{align*}
	&a(b-c)^2+b(c-a)^2+c(a-b)^2\geq 0\\
	&\Leftrightarrow ab^2-2abc+ac^2+bc^2-2abc+ba^2+ca^2-2abc+cb^2\geq 0\\
	&\Leftrightarrow ab^2+ac^2+bc^2+ba^2+ca^2+cb^2\geq 6abc \hspace{1cm}\text{(1)}
	\end{align*}
	Mặt khác ta có: 
	\begin{align*}
	(a+b+c)^3 &= [(a+b)+c]^3= (a+b)^3+3(a+b)^2c+3(a+b)c^2+c^3\\
	&= a^3+b^3+c^3+3(ab^2+ac^2+bc^2+ba^2+ca^2+cb^2)+6abc\hspace{1cm}\text{(2)}
	\end{align*}
	Từ $(1)$ và $(2)$ suy ra: $(a+b+c)^3\geq a^3+b^3+c^3+3.6abc+6abc
	\Leftrightarrow (a+b+c)^3\geq a^3+b^3+c^3+24abc$.
}
\end{ex}
\begin{ex}%[Đề thi HSG Toán 9, 2013-2014, Ninh Thuận]%[9H5G4]
	Cho tam giác $ABC$ có trung tuyến $AM=1$ và $\widehat{ABC}=\widehat{CAM}$. Tính độ dài cạnh $AB$.
	\loigiai
	{
		\immini
		{
		Ta có $\widehat{ABC}=\widehat{MAC}$ (Theo giả thiết) và $\widehat{ACM}$ chung, suy ra: $\Delta ABC\backsim\Delta MAC$\\ 
		$\Rightarrow \dfrac{AB}{MA}=\dfrac{BC}{AC}=\dfrac{AC}{MC}$ $(*)$\\
		Theo giả thiết $AM = 1$, nên từ $(*)$ suy ra: $AB=\dfrac{AC}{MC}$ $(1)$\\
		Mặt khác, theo giả thiết $AM$ là trung tuyến nên $M$ là trung điểm của $BC \Rightarrow BC = 2MC$.\\
		Do đó, từ $(*)$ suy ra: $\dfrac{AC}{MC}=\dfrac{BC}{AC}
		\Rightarrow \dfrac{AC}{MC}=\dfrac{2MC}{AC}\\
		\Leftrightarrow AC^2 = 2MC^2$
		$\Leftrightarrow \dfrac{AC^2}{MC^2}=2\\
		\Leftrightarrow \dfrac{AC}{MC}=\sqrt{2}$ $(2)$\\
		Do dó, từ $(1)$ và $(2)$ suy ra: $AB = \sqrt{2}$.
			}{
	\begin{tikzpicture}[scale=.7]
	\tkzDefPoints{0/0/M,-0.4/2.3/A,-3/0/B}
	\tkzDefPointsBy [symmetry=center M](B){C}
	\tkzMarkAngles(C,B,A M,A,C)
	\tkzDrawPolygon(C,B,A)
	\tkzDrawSegments(M,A)
	\tkzLabelSegment[left](A,M){$1$}
	\tkzLabelPoints[left](B)
	\tkzLabelPoints[below](M)
	\tkzLabelPoints[above](A)
	\tkzLabelPoints[right](C)
	\end{tikzpicture}
	}
}
\end{ex}
\begin{ex}%[Đề thi HSG Toán 9, 2013-2014, Ninh Thuận]%[9D5G2]
	Chứng minh rằng $ab(a^2-b^2)$ Chia hết cho 3 với mọi số nguyên $a$ và $b$.
	\loigiai{
		Xét hai trường hợp:
		\begin{itemize}
			\item Trường hợp $1$: $a$ hoặc $b$ chia hết cho $3$. Khi đó tích $ab \vdots 3$ nên $ab(a^2-b^2)\vdots 3$.
			\item Trường hợp $2$: Cả $a$ và $b$ không chia hết cho $3$. Khi đó: $a = 3p+r_1$ và $b = 3q+r_2$ với $p, q, r_1,r_2$ nguyên và $1 \le r_1, r_2 \le 2$.
			\begin{itemize}
				\item Nếu: $r_1=r_2$ thì $(a-b)\vdots 3$ suy ra $ab(a^2-b^2) = ab(a-b)(a+b)\vdots 3$
				\item Nếu: $r_1 \neq r_2 thì (a+b)\vdots 3$ suy ra $ab(a^2-b^2)=ab(a-b)(a+b)\vdots 3$
			\end{itemize}
		\end{itemize}
		Vậy: $ab(a^2-b^2)\vdots 3$ với mọi số nguyên $a, b$.
		}
\end{ex}
\begin{ex}%[Đề thi HSG Toán 9, 2013-2014, Ninh Thuận]%[9H5G4]
	Cho hình bình hành $ABCD$. Trên các cạnh $AB$ và $AD$ lần lượt lấy các điểm $E$ và $F$ ($E$, $F$ không trùng với các đỉnh của hình bình hành). Gọi $K$ là giao điểm của $ED$ và $FB$. Chứng minh rằng hai tứ giác $ABKD$ và $CEKF$ có diện tích bằng nhau.
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=1.5]
			\tkzDefPoints{0/0/A,1/3/D,5/0/B}
			\tkzDefPointBy[translation=from A to B](D)
			\tkzGetPoint{C}
			\coordinate(E) at ($(A)!0.8!(B)$);
			\coordinate(F) at ($(A)!0.6!(D)$);
			\tkzInterLL(E,D)(F,B)
			\tkzGetPoint{K}
			\tkzDefPointBy[projection=onto A--B](K) 
			\tkzGetPoint{M}
			\tkzDefPointBy[projection=onto C--D](K) 
			\tkzGetPoint{N}
			\tkzDrawSegments(E,D F,B F,C E,C A,B C,D A,D B,C M,N K,C K,A)
			\tkzMarkRightAngles(C,N,M)
			\tkzLabelPoints[left](A,D,F)
			\tkzLabelPoints[below](M,E)
			\tkzLabelPoints[above](N,K)
			\tkzLabelPoints[right](B,C)
			\end{tikzpicture}
		\end{center}
		Ta có: $S_{ABKD}=S_{\Delta AKB}+S_{\Delta AKD}$ và $S_{CEKF}=S_{\Delta CKE}+S_{\Delta CKF}$\\ 
		Qua $K$ kẻ đường thẳng vuông góc với $AB$ và $CD$ (vì $AB\parallel CD$), cắt $AB$ tại $M$ và cắt $CD$ tại $N$.\\
		Ta có: $S_{\Delta AKB}+S_{\Delta CKD}=\dfrac{1}{2}KM.AB+\dfrac{1}{2}KN.CD=\dfrac{1}{2}AB.MN=\dfrac{1}{2}S_{ABCD}$ $(1)$\\
		Ta lại có: $S_{\Delta CKE}+S_{\Delta CKD}=S_{\Delta CDE}=\dfrac{1}{2}CD.MN=\dfrac{1}{2}S_{ABCD}$ $(2)$\\
		Từ $(1)$ và $(2)$ suy ra: $S_{\Delta AKB}=S_{\Delta CKE}$ $(3)$\\
		Tương tự, ta cũng có: $S_{\Delta AKD}+S_{\Delta CKB}=\dfrac{1}{3}S_{ABCD}$ 
		và $S_{\Delta CKF}+S_{\Delta CKB}=S_{\Delta BFC}=\dfrac{1}{2}S_{ABCD}$ \\
		Suy ra:$S_{\Delta AKD}=S_{\Delta CKF}$ $(4)$\\
		Từ $(3)$ và $(4)$ suy ra: $S_{\Delta AKB}+S_{\Delta AKD}=S_{\Delta CKE}+S_{\Delta CKF}$\\
		Vậy: $S_{ABKD}=S_{CEKF}$
}
\end{ex}
