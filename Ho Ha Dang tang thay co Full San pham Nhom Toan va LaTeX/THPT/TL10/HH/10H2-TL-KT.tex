	\begin{enumerate}
		\item Tìm tọa độ trực tâm $H$ của tam giác $ABC$.
		\item Tìm tọa độ tâm đường tròn ngoại tiếp tam giác $ABC$.
		\item Tìm tọa độ hình chiếu vuông góc của điểm $A$ lên đường thẳng $BC$.
	\end{enumerate}
	\loigiai{
		
		\begin{enumerate}
			\item Tìm tọa độ trực tâm $H$ của tam giác $ABC$.
			Ta có $\overrightarrow{AB}=(3,-6)$; $\overrightarrow{BC}=(-8,-4)$
			, suy ra $\overrightarrow{AB}.\overrightarrow{BC}=0$, nên tam giác $ABC$ vuông tại $B$. Vậy trực tâm      
			$H(x,y)$ là điểm $B(4,-1)$.\\
			.\dotfill{1,0 điểm}      
			\item Tìm tọa độ tâm đường tròn ngoại tiếp tam giác $ABC$.
			
			Gọi $I(x,y)$ là tâm đường tròn ngoại tiếp tam giác
			, do đó $I$ là trung điểm của $AC$, có tọa độ $I\left(\dfrac{-3}{2},0 \right)$.       
			.\dotfill{1,0 điểm}        
			\item Tìm tọa độ hình chiếu vuông góc của điểm $A$ lên đường thẳng $BC$.
			
			Vì tam giác $ABC$ vuông tại $B$ nên hình chiếu của $A$ lên $BC$ là $B(4,-1)$. \\
			.\dotfill{1,0 điểm} 
		\end{enumerate}
		
	}
\end{bt}

\begin{bt}%[Đinh Bích Hảo]%[0H2K2]%4

	\loigiai{ Từ điều kiện bài toán ta có $$b^2+c^2-\dfrac{a^2}{2}=c^2+a^2-\dfrac{b^2}{2}+a^2+b^2-\dfrac{c^2}{2}$$

			.\dotfill{1,0 điểm}      
			\item Tìm tọa độ tâm đường tròn ngoại tiếp tam giác $ABC$
			
			Gọi $I(x,y)$. Khi đó $IA=IB$ và $IA=IC$. Do đó,
			ta có \\
			$(x+5)^2+(y-6)^2=(x+4)^2+(y+1)^2$\\
			$(x+5)^2+(y-6)^2=(x-4)^2+(y-3)^2$ \\
			suy ra $3x-y=6$ và $x-7y=22$,
			giải hệ phương trình ta được $I(\dfrac{-16}{5},\dfrac{-18}{5})$. 
			
			.\dotfill{1,0 điểm}        \\
			\item Tìm tọa độ hình chiếu vuông góc của điểm $A$ lên đường thẳng $BC$
			
			Gọi $M(x,y)$ là hình chiếu vuông góc của điểm $A$ lên cạnh $BC$.
			Ta có $\overrightarrow{AM}=(x+5,y-6)$
			và $\overrightarrow{BC}=(8,4)$; $\overrightarrow{BM}=(x+4,y+1)$
			
			Khi đó ta có    $\overrightarrow{AM}.\overrightarrow{BC}=0$ và $\overrightarrow{BM}=t.\overrightarrow{BC}$.
			
			Suy ra $t=\dfrac{2}{7}$ và $M(-2,0)$ \\
			.\dotfill{1,0 điểm} 
		\end{enumerate}
		
	}
\end{bt}

\begin{bt} (2 điểm)
	Cho nửa đường tròn đường kính $AB$. Có $AC$, $BD$ là hai dây cung thuộc nửa đường tròn cắt nhau tại $E$.

	\loigiai{ \\
		Ta có $\overrightarrow{AE}.\overrightarrow{AC}=\overrightarrow{AE}.(\overrightarrow{AB}+\overrightarrow{BC})=\overrightarrow{AE}.\overrightarrow{AB}+\overrightarrow{AE}.\overrightarrow{BC}=\overrightarrow{AE}.\overrightarrow{AB}$  (1).\\

		Cộng $(1)$ và $(2)$ theo vế ta được:\\
		$\overrightarrow{AE}.\overrightarrow{AC}+\overrightarrow{BE}.\overrightarrow{BD}=(\overrightarrow{AE}-\overrightarrow{BE})\overrightarrow{AB}=(\overrightarrow{AE}+\overrightarrow{EB})\overrightarrow{AB}=\overrightarrow{AB}^2=AB^2.$
	} 
\end{bt}

\begin{bt} (3 điểm)
	Cho tam giác $ABC$ có các cạnh $AC=b$, $BC=a$ và $AB=c$.   \\
	a) Tính $\overrightarrow{AB}.\overrightarrow{AC}$ theo $a, b, c$.\\
	b) Gọi $M$ là trung điểm $BC$ và $G$ là trọng tâm $\Delta ABC$. Tính độ dài $AM$ và $AG$.\\
	c) Tính $\cos$ góc tạo bởi $AG$ và $BC$ theo $a, b, c.$
	\loigiai{ \\
		a) Ta có : $\overrightarrow{BC}=\overrightarrow{AC}-\overrightarrow{AB}$ (1).\\
		Bình phương hai vế $(1)$ ta được:\\
		$\overrightarrow{BC}^2=(\overrightarrow{AC}-\overrightarrow{AB})^2 \Leftrightarrow BC^2=AC^2+AB^2-2.\overrightarrow{AC}.\overrightarrow{AB}$.\\
		Vậy ta có: $\overrightarrow{AC}.\overrightarrow{AB}=\dfrac{1}{2}(AC^2+AB^2-BC^2)=\dfrac{1}{2}(a^2+b^2-a^2)$.\\
		b) Theo đẳng thức véctơ về trung điểm ta có: $\overrightarrow{AM}=\dfrac{1}{2}(\overrightarrow{AC}+\overrightarrow{AB})$. (2)\\
		Bình phương hai vế $(2)$ ta được:\\
		$\overrightarrow{AM}^2 = \dfrac{1}{4} (\overrightarrow{AC}+\overrightarrow{AB})^2 = \dfrac{1}{4} (AC^2+AB^2+2.\overrightarrow{AC}.\overrightarrow{AB})\\
		=\dfrac{1}{4} \left[c^2+b^2 + 2.\dfrac{1}{2}\left( b^2+c^2-a^2 \right)\right] = \dfrac{1}{4}\left( 2b^2+2c^2-a^2 \right)$\\
