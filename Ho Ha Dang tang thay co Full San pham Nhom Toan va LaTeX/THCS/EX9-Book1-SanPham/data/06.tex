
\section{PHƯƠNG TRÌNH BẬC NHẤT VỚI HAI ẨN}

\subsection{Mở đầu}

\begin{vd}
Tìm nghiệm nguyên của phương trình 
\[11x+18y=120.\tag{1}\]
\loigiai{
Chú ý đến tính chia hết, ta thấy $11x \ \vdots \ 6$ nên $x\ \vdots  \ 6$. Đặt $x=6k  \ ( k \  \text{nguyên})$. Thay vào $(1)$ và rút gọn ta được $$ 11k+18y=20.$$  Biểu thị ẩn mà hệ số của nó có giá trị tuyệt đối nhỏ (là $y$)  theo $k$ ta được $$y=\dfrac{20-11k}{3}$$ Tách riêng  giá trị nguyên ở biểu thức này được $$y=7-4k+\dfrac{k-1}{3}$$ Lại đặt $\dfrac{k-1}{3}=t \ ( t\ \text{nguyên})$ suy ra $k=3t+1.$ Do đó 
{\allowdisplaybreaks
\begin{align*}
y&=7-4(3t+1)+t=3-11t.\\
x&=6k-6(3t+1)=18t+6.
\end{align*}}Thay các biểu thức của $x$ và $y$ vào $(1)$, phương trình được nghiệm đúng. \\ Vậy nghiệm nguyên của phương $(1)$ được biểu thị bởi công thức $$\heva{x=18t+6\\y=3-11t.} \ (\text{ với } t \ \text{là số nguyên tùy ý}).$$
\textbf{\textit{Lưu ý:}} Nếu đề bài yêu cầu tìm nghiệm nguyên dương của phương trình $(1)$ thì sau khi được nghiệm nguyên tổng quát, ta giải các điều kiện: $$\heva{18t+6>0\\3-11t>0} \Leftrightarrow -\dfrac{1}{3}<t<\dfrac{3}{11}.$$ Do đó $t=0$ (vì $t$ nguyên). Nghiệm nguyên dương $(x;y)$ của phương trình $(1)$ là $(6;3)$.\\ Trong trường hợp tìm nghiệm nguyên dương của phương trình $(1)$, ta còn có thể giải như sau: $$11x+18y=120$$ Do $y\geq 1$ nên  $11x\leq 120-18.1=102.$  \\ Do $x$ nguyên nên $x\leq 9.$ \\ Mặt khác $x \ \vdots \ 6$ và $x$ nguyên dương nên $x=6.$ Suy ra $y=3$.\\ $\blacktriangle$  \textbf{Kinh nghiệm giải toán}\\ Có nhiều cách tách  giá trị nguyên  của biểu thức $y=\dfrac{20-11k}{3}.$ chẳng hạn: \begin{align*}
y&=7-4k+\dfrac{k-1}{3} \ (\text{cách 1}) \\ y&=7-3k-\dfrac{1+2k}{3}\ ( \text{cách 2})\\ y&=6-3k+\dfrac{2(1-k)}{3} \ (\text{cách 3}.)
\end{align*}
Bạn đọc  tự giải theo cách trên để thấy:
\begin{itemize}
	\item Cách 1 gọn hơn cách 2 vì trong cách 1 hệ số của $k$ ở phân số bằng $1$, do đó sau khi đặt $\dfrac{k-1}{3}=t$ ta không cần dùng thêm một ẩn phụ nào nữa.
	\item Trong cách 3, nhờ đặt được thừa số chung mà hệ số của $k$ ở phân số bằng $-1$, do đó sau khi đặt $\dfrac{1-k}{3}=t$, ta cũng không cần dùng thêm một ẩn phụ nào nữa.
\end{itemize} 
}	
\end{vd}
\subsection{Cách giải phương trình bậc nhất hai ẩn  $ax+by=c$ với nghiệm nguyên $(a, b, c \in \mathbb{Z})$}

\begin{enumerate}[-]
	\item Rút gọn phương trình, chú ý đến tính chia hết của các ẩn.
	\item Biểu thị ẩn mà hệ số của nó có giá trị tuyệt đối nhỏ, chẳng hạn $x$, theo ẩn kia.
	\item Tách giá trị nguyên ở biểu thức của $x$
	\item Đặt điều kiện đề phân số trong biểu thức của $x$ bằng một	 số nguyên $t_1$, ta được một phương trình  bậc nhất hai ẩn $y$ và $t_1$.
	\item  Cứ tiếp tục làm như trên cho đến khi các ẩn đều được biểu thị dưới dạng một  đa thức với các hệ số nguyên.
\end{enumerate} 	
 Thực chất của cách giải này là thay việc giải phương trình $ax+by=c$  bởi việc giải lần lượt các phương trình 
 \begin{align*}
 a_1y+b_1t_1=c_1\\a_2t_1+b_2t_2=c_2\\ \ldots\ldots\ldots\ldots\ldots
 \end{align*}
 trong đó cách hệ số $a_1, a_2,\ldots ; b_1, b_2, \ldots $ có giá trị tuyệt đối nhỏ dần cho đến khi được một hệ số có giá trị tuyệt đối bằng $1$.\\
 \textbf{\textit{Lưu ý:}} Ngoài cách giải bằng phương pháp biểu thị một ẩn theo ẩn kia như trên, còn có thể giải phương trình $ax+by=c$ bằng phương pháp tìm một nghiệm riêng, xem trang 95.

\begin{bt}
Tìm nghiệm nguyên của mỗi phương trình sau:
\begin{enumerate}[a)]
	\item $12x-7y=45$;
	\item $9x+20y=547$;
	\item $11x+8y=73$;
\end{enumerate}
    \loigiai{
\begin{enumerate}[a)]
	\item Nhận thấy $y \ \vdots \ 3, $ đặt $y=3k\ (k\in \mathbb{Z})$. Rút gọn được $4x-7k=15.$\\ $x=\dfrac{7k+15}{4}=2k+4-\dfrac{k+1}{4}$. Đặt $\dfrac{k+1}{4}=t \ (t\in \mathbb{Z})$.
\\ Đáp số: $\heva{x&=7t+2\\y&=12t-3} \ (t \,\, \text{là số nguyên tùy ý}).$
\item $x=\dfrac{547-20y}{9}=61-2y-\dfrac{2(1+y)}{9}.$ Đặt $\dfrac{1+y}{9}=t\ (t \in \mathbb{Z}).$\\ Đáp số $\heva{x&=63-20t\\y&=9t-1} \ (t\ \text{là số nguyên tùy ý}).$
\item $y=\dfrac{73-11x}{8}=8-x+\dfrac{3(3-x)}{8}.$ Đặt $\dfrac{3-x}{8}=t \ (t\in \mathbb{Z})$.\\ Đáp số $\heva{x&=3-8t\\y&=5+11t} \  ( t\ \text{ là số nguyên tùy ý}).$

\end{enumerate} 
    }
\end{bt}
   
   \begin{bt}
Chứng minh rằng phương trình sau không có nghiệm nguyên dương: 
$$11x+1999y=11.1999.$$
   	\loigiai{
Ta thấy  $1999y \  \vdots \ 11 \Rightarrow y \ \vdots \ 11.$
\\ Do $y$ là số nguyên dương nên $y\geq 11.$ Do đó $1999y \geq \ 1999.11$. \\ Suy ra vế trái của phương trình  (là $11x+1999y$) lớn hơn vế phải ( là $ 11.1999$.) Vậy phương trình không có nghiệm nguyên dương. \\ \textbf{\textit{Lưu ý:}} Chứng minh tương tự như trên đối với  bài toán tổng quát: Nếu số nguyên dương $a, b$ nguyên tố cùng nhau thì phương trình $ax+by=ab$ không có nghiệm nguyên dương.
 
 } 
   \end{bt}
\begin{bt}
Cho phương trình $19x+83y=1983.$ \hfill $(1)$\\ Từ đẳng thức $19.100+83.1=1983,$ hãy viết $100$ dưới dạng $83+a,$ viết $1$ dưới dạng $b-19$ rồi tìm $a, b$ và tìm  một nghiệm nguyên khác $(x;y)=(100;1)$ của phương trình $(1)$.

\loigiai{
$(x;y)=(100;1)$ là một nghiệm của phương trình $19x+83y=1983$ \hfill (1) 
 $$a=100-83=17, b=19+1=20.$$
 Ta có \begin{align*}
 1983=19.1000+83.1&=19(83+17)+83(20-19)\\&=19.83+19.17+83.20-83.19\\&=19.17+83.20 
 \end{align*}
  Vậy $(x;y)=(17;20)$ là một nghiệm khác của phương trình $(1).$
  \\
  \textbf{\textit{Lưu ý:}} Phương trình $(1) $ chỉ cho hai nghiệm nguyên dương $(x;y)$ là $(100;1)$ và $(17;20).$ Thật vậy,  
  $19x=1983-83y$ nên $x=\dfrac{1983-83y}{19}=100-\dfrac{83(y-1)}{19}$.
  \\
  Đặt $\dfrac{y-1}{19}=k \ (k\ \text{là số nguyên })$ thì $y=19k+1; x=100-83k$. 
  $$\heva{100-83k>0\\19k+1>0} \text{ nên} \ -\dfrac{1}{19}<k< \dfrac{1}{19}.$$ Do $k\in \mathbb{Z}$ nên $k\in \{0;1\}.$} 
\begin{itemize}
	\item Với $k=0$ thì $x=100, y=1$.
	\item Với $k=1$ thì $x=17, y=20$.
\end{itemize}
\end{bt}
\begin{bt}
Tìm nghiệm nguyên của phương trình $$6x+8y=m+1,$$
trong đó $m$ là số nguyên cho trước.
\loigiai{
Dễ thấy $m$ lẻ. Đặt $m=2k-1$ ($k$ là số nguyên). Ta có $3x+4y=k$.
$$x=\dfrac{k-4y}{3}=-y+\dfrac{k-y}{3}. \text{Đặt} \dfrac{k-y}{3}=t \  (t\in \mathbb{Z}).$$
Đáp số: $\heva{&x=-k+4t\\&y=k-3t}$ \\ với $m$ là số lẻ, $k=\dfrac{m+1}{2},$ $t$ là số nguyên tùy ý.\\ Trường hợp $m$ chẵn, phương trình không có nghiệm nguyên.
}
\end{bt}
