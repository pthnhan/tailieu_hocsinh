\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Đống Đa, Hà Nội]%[9D3B3]
\begin{enumerate}
\item Phân tích đa thức thành nhân tử:
\begin{enumerate}
\item $x^2+2x+1$.
\item $x^2-xy+5x-5y$.
\end{enumerate}
\item Tìm $x$ biết $x^2 -2017x+2016=0$
\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item
	\begin{enumerate}
	\item Ta có : $x^2+2x+1$ = $(x+1)^2$\\
	\item Ta có: $x^2-xy+5x-5y$\\
		$=(x^2-xy)+(5x-5y)$\\
		$=(x(x-y))+5(x-y)$\\
		$=(x-y)(x+5)$\\
	\end{enumerate}
\item Ta có: 	\\
$x^2 -2017x+2016=0$\\
$x^2 -x-2016x+2016=0$\\
$(x-1)(x-2016)=0$\\
Suy ra: $x=1$ hoặc $x=2016.$
Vậy $x=1$ hoặc $x=2016.$
\end{enumerate}
 }
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Đống Đa, Hà Nội]%[9D4Y4]

	\begin{enumerate}
	\item $(x-3)(x^2+3x+9)-(x^3+3)$
	\item $(5x^2-10x):5x+(5x+2)^2:(5x+2)$

	\end{enumerate}
	\loigiai{

	\item $(x-3)(x^2+3x+9)-(x^3+3)$\\
		$=x^3+3x^2+9x-3x^2-9x-27-x^3-3$\\
		$=-30.$\\
	\item $(5x^2-10x):5x+(5x+2)^2:(5x+2)$\\
		$=x-2+5x+2$\\
		$=6x$\\
	\item $\dfrac{5x}{3+5x}+\dfrac{3}{3+5x}$\\
		$=\dfrac{5x+3}{3+5x}$\\

	\end{enumerate}

\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Đống Đa, Hà Nội]%[9D4Y4]
	Cho biểu thức $P=\dfrac{2a^2}{a^2-1}+\dfrac{a}{a+1}-\dfrac{a}{a-1}$.
	\begin{enumerate}
	\item Tìm $a$ để biểu thức $P$ có nghĩa.
	\item Rút gọn $P$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item
	Để biểu thức $P$ có nghĩa thì:
		$\begin{aligned}
		\begin{cases}
		a^2-1 & \neq0 \\
		a-1 &\neq0 \\
		a+1 &\neq0
		\end{cases}
		\end{aligned}$
		Suy ra $a\neq\pm 1$\\
	\item $P=\dfrac{2a^2}{a^2-1}+\dfrac{a}{a+1}-\dfrac{a}{a-1}$\\
		$P=\dfrac{2a^2}{(a-1)(a+1)}+\dfrac{a}{a+1}-\dfrac{a}{a-1}$\\
		$P=\dfrac{2a^2}{(a-1)(a+1)}+\dfrac{a(a-1)}{a+1}-\dfrac{a(a+1)}{a-1}$\\
		$P=\dfrac{2a^2}{(a-1)(a+1)}+\dfrac{a^2-a}{(a+1)(a-1)}-\dfrac{a^2+a}{(a+1)(a-1)}$\\
		$P=\dfrac{2a^2+a^2-a-a^2-a}{(a-1)(a+1)}$\\
		$P=\dfrac{2a^2-2a}{(a-1)(a+1)}$\\
		$P=\dfrac{2a(a-1)}{(a-1)(a+1)}$\\
		$P=\dfrac{2a}{a+1}$\\
	\end{enumerate}		
		}
	\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Đống Đa, Hà Nội]%[8H2G3]
	Cho hình chữ nhật $ABCD.$ Gọi $H$ là chân đường vuông góc kẻ từ $A$ đến $BD$. Gọi $M$ và $N$ theo thứ tự là trung điểm của các đoạn thẳng $AH$ và $DH.$
	\begin{enumerate}
	\item Chứng minh $MN||AD$
	\item Gọi $I$ là trung điểm của cạnh $BC$. Chứng minh tứ giác $BMNI$ là hình bình hành.
	\item Chứng minh tam giác $ANI$ vuông tại N.
	\end{enumerate}	
	\loigiai
	{
		\begin{enumerate}
		\item  Trong tam giác $AHD$, ta có:\\
		$M$ là trung điểm của $AH$ (gt)\\
		$N$ là trung điểm của $DH$ (gt)\\
		Do đó: $MN$ là đường trung bình của $\Delta ADH$.\\
		Suy ra: $MN\parallel AD$.	
	
		\item Vì $MN$ là đường trung bình của $\Delta ADH$ nên:\\
		$MN\parallel AD$ và $MN =\dfrac{AD}{2}$           (1)\\
		Vì $I$ là trung điểm của $BC$ nên: $IB = \dfrac{BC}{2}$       (2)\\
		Mặt khác $ABCD$ là hình chữ nhật nên $AD \parallel BC$ và $AD = BC$  (3)\\
		Từ (1), (2) và (3) suy ra $MN\parallel BI$ và $MN = BI$\\
		Do đó: $MNIB$ là hình bình hành.
		\item Vì $MN\parallel AD$ nên $MN\perp AB.$\\
		Trong $\Delta ABN$, ta có:\\
		$MN$: là đường cao (cmt).\\
		$AH$: là đường cao (gt)\\
		Do đó: $M$ là trực tâm của $\Delta ANB$.\\
		Suy ra: $BM \perp AN.$\\
		Mà $BM\parallel NI$ ( Vì $BMNI$ là hình bình hành)\\
		Suy ra: $NI\perp AN$ tại $N$.\\
		Vậy $\Delta AIN$ vuông tại $N$.\\
		Đặt $a=S_{\Delta MBC}; b=S_{\Delta MAC}; c=S_{\Delta MAB}$.\\
			Ta có \\
			$\dfrac{MA}{MP}=\dfrac{b+c}{a}\ge \dfrac{2\sqrt{bc}}{a}$\\
			$\dfrac{MB}{MQ}=\dfrac{a+b}{c}\ge \dfrac{2\sqrt{ab}}{c}$\\
			$\dfrac{MC}{MK}=\dfrac{a+c}{b}\ge \dfrac{2\sqrt{ac}}{b}$.				
	\begin{center}
	\begin{tikzpicture}[scale=0.6]
\tkzInit[xmin=-5,ymin=-3,xmax=5,
ymax=3]
\tkzClip
		\tkzDefPoints{-5/3/A,5/3/B,5/-3/C,-5/-3/D}
		\tkzDefPointBy[projection = onto B--D](A)		\tkzGetPoint{H}
		\tkzDefMidPoint(A,H)	\tkzGetPoint{M}
		\tkzDefMidPoint(D,H)	\tkzGetPoint{N}
		\tkzDefMidPoint(B,C)	\tkzGetPoint{I}
		\tkzMarkRightAngles[size=.4,fill=gray!50](A,H,B A,N,I)
		\tkzMarkSegments[mark=|](A,M M,H)
		\tkzMarkSegments[mark=||](D,N N,H)
		\tkzMarkSegments[mark=x](B,I I,C)
		\tkzDrawPolygon(A,B,C,D)
		\tkzDrawSegments(B,D A,N A,H M,N N,I M,B A,I)
		\tkzDrawPoints(A,B,C,D,H,M,N,I)
		\tkzLabelPoints[left](A,D,M)
		\tkzLabelPoints[right](B,I,C)
		\tkzLabelPoints[below](N,H)
	\end{tikzpicture}
\end{center}
	Suy ra	$\dfrac{MA}{MP}.\dfrac{MB}{MQ}.\dfrac{MC}{MK}\ge 8\Rightarrow MA.MB.MC\ge 8MP.MQ.MK.$\\
	Dấu "$=$" xảy ra khi $M$ là trọng tâm của tam giác $ABC$.
	\end{enumerate}
}
\end{ex}

	$P=(x+2)(x+4)(x+6)(x+8)+16$ là bình phương của một số hữu tỉ.

		$P=(x^2+10x+16)(x^2+10x+24)+16$\\
		Đặt $t=P=(x^2+10x+16)$ thì biểu thức $P$ trở thành:\\
		$P=t(t+8)+16$\\
		$P=t^2+8t+16$\\
		$P=(t+4)^2=(x^2+10x+20)^2$\\
		Vậy với mọi giá trị của $x\in\mathbb{Q}$ thì giá trị của đa thức $P$ là bình phương của một số hữu tỉ.
			
	}
