\begin{ex}%[TS 10, SGD Thái Bình, 2018-2019]%[Vũ Nguyễn Hoàng Anh, dự án EXTHCS-06]%[9D1B8]%[9D1Y1]%[9D1K8]
Cho biểu thức $A=\dfrac{1}{\sqrt{x}}+\dfrac{\sqrt{x}}{\sqrt{x}+1}$.
\begin{enumerate}
\item Tính giá trị biểu thức $A$ với $x=\dfrac{4}{9}$.
\item Tìm điều kiện để biểu thức $A$ có nghĩa.
\item Tìm $x$ để $A=\dfrac{3}{2}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Với $x=\dfrac{4}{9}$ ta có $A=\dfrac{1}{\sqrt{\dfrac{4}{9}}}+\dfrac{\sqrt{\dfrac{4}{9}}}{\sqrt{\dfrac{4}{9}}+1}=\dfrac{19}{10}$.
\item Biểu thức $A$ có nghĩa khi $x>0$.
\item Với $x>0$, ta có $A=\dfrac{x+\sqrt{x}+1}{x+\sqrt{x}}$. Do đó
\begin{align*}
A=\dfrac{3}{2}\Leftrightarrow\ & 2x+2\sqrt{x}+2=3x+3\sqrt{x} \Leftrightarrow x+\sqrt{x}-2=0\\
\Leftrightarrow\ & \left( \sqrt{x}-1 \right)\left( \sqrt{x}+2 \right)=0 \Leftrightarrow \sqrt{x}=1 \Leftrightarrow x=1.
\end{align*}
Vậy với $x=1$ thì $A=\dfrac{3}{2}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[TS 10, SGD Thái Bình, 2018-2019]%[Vũ Nguyễn Hoàng Anh, dự án EXTHCS-06]%[9D2B5]
Cho hàm số $y=kx+b$ có đồ thị là đường thẳng $\Delta$. Tìm $k$ và $b$ biết đường thẳng $\Delta$ có hệ số góc bằng $\sqrt{2}$ và đi qua điểm $M\left( -\sqrt{2};3 \right)$.
\loigiai{
Do đường thẳng $\Delta$ có hệ số góc bằng $\sqrt{2}$ nên $k=\sqrt{2}$. Lại có $M\in \Delta$ nên
\[3=\sqrt{2}\cdot \left(-\sqrt{2}\right)+b \Rightarrow b=5.\]
Vậy $k=\sqrt{2}$, $b=5$.
}
\end{ex}

\begin{ex}%[TS 10, SGD Thái Bình, 2018-2019]%[Vũ Nguyễn Hoàng Anh, dự án EXTHCS-06]%[9H3Y9]%[9H3B3]%[9H3K3]%[9H3K3]
Cho tam giác $ABC$ có $AB<AC$ và ba góc đều nhọn. Đường tròn tâm $O$, đường kính $BC$ cắt các cạnh $AB$, $AC$ lần lượt tại $E$ và $D$.
\begin{enumerate}
\item Giả sử $BC=6a$. Tính diện tích hình tròn $(O)$ theo $a$.
\item Gọi $H$ là giao điểm của $BD$ và $CE$, gọi $K$ là giao điểm của $AH$ và $BC$. Chứng minh rằng $AH$ vuông góc với $BC$.
\item Từ $A$ kẻ các tiếp tuyến $AM$, $AN$ đến đường tròn $(O)$ với $M$, $N$ là các tiếp điểm. Chứng minh rằng $\widehat{ANM}=\widehat{AKN}$.
\item Giả sử $F$ là điểm di động trên đường tròn $(O)$. Xác định vị trí của điểm $F$ để tam giác $FBC$ có diện tích lớn nhất.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[>=stealth]
\clip (-2,-2.5) rectangle (3.6,4.6);

%\tkzDrawArc[color=black](O,C)(B)
\tkzDrawCircle(O,B)
\tkzInterLL(C,E)(B,D)\tkzGetPoint{H}
\tkzInterLL(A,H)(B,C)\tkzGetPoint{K}
\tkzTangent[from=A](O,B)\tkzGetPoints{M}{N}
\tkzDefPointBy[rotation=center O angle -40](C)\tkzGetPoint{F}
\tkzDefPointBy[projection=onto B--C](F)\tkzGetPoint{P}

\end{tikzpicture}

\begin{enumerate}
\item Diện tích hình tròn $(O)$ là $\pi \cdot \dfrac{BC^2}{4}=9\pi a^2$.
\item Do $D$, $E$ thuộc đường tròn đường kính $BC$ nên $\widehat{BDC}=\widehat{BEC}=90^\circ$ hay $BD$, $CE$ là các đường cao của tam giác $ABC$. Suy ra $H$ là trực tâm của tam giác $ABC$. Vì vậy $AH\perp BC$.
\item Không mất tính tổng quát, giả sử $M$ và $B$ cùng phía đối với $OA$. Do $AM$, $AN$ là các tiếp tuyến với $(O)$ $\widehat{AMO}=\widehat{ANO}=90^\circ$, mà $\widehat{AKO}=90^\circ$ (chứng minh trên), nên năm điểm $A$, $M$, $K$, $O$, $N$ cùng thuộc đường tròn đường kính $AO$. \begin{ex}%[TS 10, SGD Thái Bình, 2018-2019]%[Vũ Nguyễn Hoàng Anh, dự án EXTHCS-06]%[9D5G1]
Cho các số không âm $a$, $b$, $c$ thỏa mãn $a^2+b^2+c^2=2$. Tìm giá trị lớn nhất của biểu thức
\[S=\dfrac{a}{2+bc}+\dfrac{b}{2+ca}+\dfrac{c}{2+ab}.\]
\loigiai{
Không mất tính tổng quát, giả sử $a\ge b\ge c$. Khi đó
\[(a-b)\left[ a(a-c)-b(b-c) \right]+c(c-a)(c-b)\ge 0\]
hay
\[a(a-b)(a-c)+b(b-c)(b-a)+c(c-a)(c-b)\ge 0.\tag{1}\]
Đặt $p=a+b+c$, $q=ab+bc+ca$, $r=abc$, ta có
\begin{align*}
&ab(a+b)+bc(b+c)+ca(c+a)=pq-3r,\tag{2}\\
&a^3+b^3+c^3=p^3-3pq+3r.\tag{3}
\end{align*}
Từ $(1)$ ta có
\begin{align*}
a^3+b^3+c^3+3abc-ab(a+b)-bc(b+c)-ca(c+a)\ge 0.\tag{4}
\end{align*}
Thay $(2)$ và $(3)$ vào $(4)$, ta được
\begin{align*}
9r&\ge p\left( 4q-p^2 \right).\tag{5}
\end{align*}
Mặt khác, ta có
\begin{align*}
p^2-2=a^2+b^2+c^2+2(ab+bc+ca)-2=2q.\tag{6}
\end{align*}
Thay $(6)$ vào $(5)$, ta được
\begin{align*}
9r\ge p\left( 2p^2-4-p^2 \right)=p^3-4p.\tag{7}
\end{align*}
Ta có
\begin{align*}
2S&=a\cdot\dfrac{2+bc-bc}{2+bc}+b\cdot\dfrac{2+ca-ca}{2+ca}+c\cdot\dfrac{2+ab-ab}{2+ab}\\
&=(a+b+c)-abc\left( \dfrac{1}{2+ab}+\dfrac{1}{2+bc}+\dfrac{1}{2+ca} \right).
\end{align*}
Áp dụng bất đẳng thức Cauchy-Schwarz, ta có
\begin{align*}
 \dfrac{1}{2+ab}+\dfrac{1}{2+bc}+\dfrac{1}{2+ca} \ge \dfrac{9}{6+ab+bc+ca},
\end{align*}
nên
\begin{align*}
2S\le p-\dfrac{9r}{6+q}.
\end{align*}
Nếu $p\le 2$ thì $2S\le 2$ hay $S\le 1$. Xét $p>2$, kết hợp $(7)$, ta được
\begin{align*}
p-\dfrac{9r}{6+q}\le p-\dfrac{p^3-4p}{6+\dfrac{p^2-2}{2}}=-\dfrac{(p-2)\left(p^2+4p-10\right)}{p^2+10}+2<2\ (\text{do}\ p>2).
\end{align*}
Do đó ta luôn có $S\le 1$. Đẳng thức xảy ra khi $a=1$, $b=1$, $c=0$ hoặc các hoán vị. Vậy giá trị lớn nhất của $S$ là $1$.
}

\end{ex}
