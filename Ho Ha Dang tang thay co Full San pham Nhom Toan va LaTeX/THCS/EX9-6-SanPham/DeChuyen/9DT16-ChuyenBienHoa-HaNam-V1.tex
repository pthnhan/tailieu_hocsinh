\begin{ex}%[Đề thi vào 10, Chuyên Biên Hòa Hà Nam (Vòng 1), 2016]%[9D1Y3]%[9D1B8]
Rút gọn các biểu thức sau
\begin{enumerate}
\item $A=2\sqrt{12}+3\sqrt{\dfrac{1}{3}}-\dfrac{2}{\sqrt{3}+1}-1$.
\item $B=\left(\dfrac{1}{\sqrt{x}+2}+\dfrac{1}{\sqrt{x}-2}\right)\cdot\dfrac{\sqrt{x}-2}{\sqrt{x}}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $A=4\sqrt{3}+\sqrt{3}-(\sqrt{3}-1)-1=4\sqrt{3}$.
\item $B=\dfrac{\sqrt{x}-2+\sqrt{x}+2}{(\sqrt{x}+2)(\sqrt{x}-2)}\cdot\dfrac{\sqrt{x}-2}{\sqrt{x}}=\dfrac{2\sqrt{x}}{\sqrt{x}\left(\sqrt{x}+2\right)}=\dfrac{2}{\sqrt{x}+2}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Biên Hòa Hà Nam (Vòng 1), 2016]%[9D4B2]%[9D4B7]
\hfill
\begin{enumerate}
\item Trong mặt phẳng tọa độ $Oxy$, cho Parabol $(P)$ có phương trình $y=\dfrac{1}{4}x^2$ và đường thẳng $d$ có phương trình $y=ax+b$. Xác định $a,b$ để $(d)$ cắt trục tung tại điểm có tung độ bằng $2$ và cắt $(P)$ tại điểm có hoành độ bằng $-2$.
\item Giải phương trình $4x^4-5x^2+1=0$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Gọi $A=d\cap Oy\Rightarrow A(0;2)\in d\Rightarrow 2=a.0+b\Leftrightarrow b=2$.\\
Gọi $B=d\cap (P)= B(-2;y_B)\Rightarrow y_B=\dfrac{1}{4}\cdot(-2)^2=1$.\\
$\Rightarrow B(-2;1)\in d\Rightarrow 1=a.(-2)+2\Leftrightarrow a=\dfrac{1}{2}$.\\
Vậy $(a,b)=\left(\dfrac{1}{2};2\right)$.
\item Ta có phương trình tương đương
$$4x^4-4x^2-(x^2-1)=0\Leftrightarrow (x^2-1)(4x^2-1)=0\Leftrightarrow \hoac{& x=\pm 1\\ & x=\pm \dfrac{1}{2}.}$$
Vậy tập nghiệm của phương trình là $S=\left\{\pm 1;\pm \dfrac{1}{2}\right\}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Biên Hòa Hà Nam (Vòng 1), 2016]%[9D3Y4]%[9D3K3]
Cho hệ phương trình $\heva{& mx-y=1\\ & x+my=2}$ ($m$ là tham số $m\neq 0$).
\begin{enumerate}
\item Giải hệ phương trình với $m=-2$.
\item Tìm $m$ để hệ phương trình có nghiệm duy nhất $(x,y)$ thỏa mãn $x+y=-1$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Với $m=-2$ hệ phương trình trở thành 
$$\heva{& -2x-y=1\\ & x-2y=2}\Leftrightarrow \heva{& -2x-y=1\\ & 2x-4y=4}\Leftrightarrow \heva{& -5y=5\\ & x=2y+2}\Leftrightarrow \heva{& x= 0\\ & y=-1.}$$
Vậy nghiệm hệ phương trình là $(x;y)=(0;-1)$.
\item Ta có $x+y=-1\Leftrightarrow y=-1-x$. Hệ trở thành
$$\heva{& mx-(-1-x)=1\\ & x+m(-1-x)=2}\Leftrightarrow \heva{& mx+x=0\\ & (1-m)x=m+2}\Leftrightarrow \heva{& \hoac{& x=0\\ & m=-1}\\ & (1-m)x=m+2.}$$
\begin{itemize}
\item Với $x=0\Rightarrow m=-2$, hệ có nghiệm duy nhất $(x;y)=(0;-1)$ (thỏa mãn).
\item Với $m=-1$, hệ có nghiệm duy nhất $(x;y)=\left(\dfrac{1}{2};-\dfrac{3}{2}\right)$ (thỏa mãn).
\end{itemize}
Vậy $m\in\{-2;-1\}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Biên Hòa Hà Nam (Vòng 1), 2016]%[9D5G1]
Cho các số thực $a,b,c$ thỏa mãn $\dfrac{27a^2}{2}+4b^2+c^2=1-2bc$. Tìm giá trị lớn nhất và giá trị nhỏ nhất của biểu thức $P=3a+2b+c$.
\loigiai{
Đặt $3a=x$, $2b=y$ và $c=z$ ta được
\begin{eqnarray*}
&& \dfrac{27a^2}{2}+4b^2+c^2=1-2bc
\Leftrightarrow \dfrac{3(3a)^2}{2}+(2b+c)^2=1+2bc\\
& \Leftrightarrow & 3x^2+2(y+z)^2=2+2yz\le 2+\dfrac{(y+z)^2}{2}
\Rightarrow 6x^2+3(y+z)^2\le 4.
\end{eqnarray*}
Theo bất đẳng thức Bunhiacopsky lại có
$$\left(\dfrac{1}{6}+\dfrac{1}{3}\right)\left[6x^2+3(y+z)^2\right]\ge (x+y+z)^2=P^2\Leftrightarrow |P|\le 2\Leftrightarrow -\sqrt{2}\le P\le \sqrt{2}.$$
Vậy $\min P=-\sqrt{2}$ khi $x=y=z=-\dfrac{\sqrt{2}}{3}$ hay $(a;b;c)=\left(-\dfrac{\sqrt{2}}{9};-\dfrac{\sqrt{2}}{6};-\dfrac{\sqrt{2}}{3}\right)$.\\
$\max P=\sqrt{2}$ khi $x=x=z=\dfrac{\sqrt{2}}{3}$ hay $(a;b;c)=\left(\dfrac{\sqrt{2}}{9};\dfrac{\sqrt{2}}{6};\dfrac{\sqrt{2}}{3}\right)$.
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Biên Hòa Hà Nam (Vòng 1), 2016]%[9H3Y7]%[9H2B4]%[9H2B5]%[9H3K2]
Cho nửa đường tròn $(O)$ đường kính $AB=2R$. Vẽ tiếp tuyến $Ax$ với nửa đường tròn $(O)$. Trên $Ax$ lấy điểm $M$ sao cho $AM>AB$. Gọi $N$ là giao điểm của đoạn thẳng $AB$ và nửa đường tròn $(O)$. Qua trung điểm $P$ của đoạn thẳng $AM$ dựng đường thẳng vuông góc với $AM$ cắt đoạn thẳng $BM$ tại $Q$.
\begin{enumerate}
\item Chứng minh tứ giác $APQN$ là tứ giác nội tiếp đường tròn.
\item Chứng minh $\widehat{BAN}=\widehat{OQB}$.
\item Chứng minh $PN$ là tiếp tuyến của nửa đường tròn $(O)$.
\item Giả sử đường tròn nội tiếp tam giác $ANP$ có độ dài đường kính bằng $R$. Tính diện tích $\triangle ABM$ theo $R$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1.5]
\tkzInit[xmin=-0.5,xmax=5.5,ymin=-0.5, ymax=6.5] \tkzClip[space=.1]
\draw[thick] (5,0) arc (0:180:2.5);   
\tkzDefPoints{0/0/A, 5/0/B, 0/6/M}

\tkzDefMidPoint(A,B) \tkzGetPoint{O}
\tkzDefMidPoint(A,M) \tkzGetPoint{P}
\tkzDefPointBy[translation = from A to O](P) \tkzGetPoint{Q}
\tkzInterLC(B,M)(O,A) \tkzGetPoints{N}{B}
\tkzInterLC(P,O)(O,A) \tkzGetPoints{I}{y}
\tkzInterLL(P,O)(N,A) \tkzGetPoint{J}

\tkzMarkRightAngle(Q,P,M)
\tkzMarkRightAngle(B,A,M)
\tkzMarkRightAngle(N,J,P)

\tkzDrawSegments(A,B A,M M,B P,Q P,N N,O O,P N,I A,N Q,O)
\tkzDrawPoints[fill=white](A,B,M,O,P,N,Q,I,J)

\tkzLabelPoints[above](I)
\tkzLabelPoints[left](A,P,M)
\tkzLabelPoints[right](B)
\tkzLabelPoints[above right](Q,N,O)
\tkzLabelPoints[below](J)

\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Ta có $\widehat{APQ}=\widehat{ANQ}=90^{\circ}$ nên $APQN$ nội tiếp đường tròn đường kính $AQ$.
\item Ta có $PQ\parallel AB$ (cùng vuông góc với $AM$) và $P$ là trung điểm $AM$ nên $Q$ là trung điểm $BM$. Do đó $OQ\parallel AM$ và $OQ\perp AB$, suy ra $\widehat{BAN}=\widehat{OQN}$ (cùng phụ với $\widehat{ABM}$).
\item Ta có $\triangle OAP=\triangle ONP$ (c.c.c) nên $\widehat{ONP}=\widehat{OAP}=90^{\circ}$. Do đó $PN$ là tiếp tuyến của $(O)$.
\item Gọi $I=PO\cap\wideparen{AN}$, $J=PO\cap AN$.\\
Ta có $PO$ là phân giác $\widehat{AON}$ nên $\wideparen{AI}=\wideparen{IN}$. Suy ra $IN$ là phân giác $\widehat{PNA}$.\\
Mà $PI$ là phân giác $\widehat{APN}$. Do đó $I$ là tâm đường tròn nội tiếp $\triangle ANP$.\\
Lại có $\triangle ANP$ cân tại $P$ nên $AJ$ là phân giác đồng thời là đường cao.\\
$\Rightarrow IJ$ là bán kính đường tròn nội tiếp $\triangle ANP$. Suy ra $IJ=\dfrac{R}{2}\Rightarrow JO=\dfrac{R}{2}$.\\
$\Rightarrow \sin\widehat{JAO}=\dfrac{JO}{AO}=\dfrac{1}{2}\Rightarrow \widehat{JAO}=30^{\circ}\Rightarrow \widehat{ABM}=60^{\circ}\Rightarrow AM=AB.\tan 60^{\circ}=2\sqrt{3}R$.
$\Rightarrow S_{ABM}=\dfrac{1}{2}AB.AM=2\sqrt{3}R^2$.
\end{enumerate}
}
\end{ex}
