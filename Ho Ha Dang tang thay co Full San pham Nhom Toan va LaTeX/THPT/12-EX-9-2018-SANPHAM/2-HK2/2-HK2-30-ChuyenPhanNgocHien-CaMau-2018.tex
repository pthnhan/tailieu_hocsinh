	\choice

	{$\dfrac{2}{5}$}

		\left|iz-3\right|=\left|z-2-i\right|&\Leftrightarrow & \left|xi-y-3\right|=\left|(x-2)+(y-1)i\right|\\
		&\Leftrightarrow& x^2+\left(y+3\right)^2=\left(x-2\right)^2+\left(y-1\right)^2\\& \Leftrightarrow& x+2y=-1\Leftrightarrow x=-2y-1.
