	\def\khvuong[size=#1](#2,#3,#4){
		\draw ($(#3)!#1!(#2)$) -- ($($(#3)!#1!(#2)$)+($(#3)!#1!(#4)$)-(#3)$)-- ($(#3)!#1!(#4)$);
	}
