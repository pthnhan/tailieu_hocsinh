\begin{ex}%[Đề thi Khảo sát chất lượng, năm học 2017-2018, Long Biên, Hà Nội]%[9D1B8]
	Cho hai biểu thức $P=\dfrac{a-9}{\sqrt{a}-3}$ và $Q=\dfrac{3}{\sqrt{a}-3}+\dfrac{2}{\sqrt{a}+3}+\dfrac{a-5\sqrt{a}-3}{a-9}$ với $a\geq0$, $a\ne9$.
	\begin{enumerate}[1)]
		\item Khi $a=81$, tính giá trị của biểu thức $P$.
		\item Rút gọn biểu thức $Q$.
		\item Với $a\geq9$, tìm giá trị nhỏ nhất của biểu thức $A=P\cdot Q$.
	\end{enumerate}
\loigiai{
\begin{enumerate}[1)]
\item $a=81$ (Thỏa mãn điều kiện xác định) suy ra $\sqrt{a}=9$.\\
Thay vào biểu thức $P$ ta tính được $P=12$.
\item Biến đổi $Q=\dfrac{3(\sqrt{a}+3)+2(\sqrt{a}-3)+a-5\sqrt{a}-3}{a-9}$.\\
Rút gọn về được $Q=\dfrac{a}{a-9}$.
\item Biến đổi $P\cdot Q=\dfrac{a}{\sqrt{a}-3}=\sqrt{a}+3+\dfrac{9}{\sqrt{a}-3}=(\sqrt{a}-3)+\dfrac{9}{\sqrt{a}-3}+6$.\\
Đánh giá được $(\sqrt{a}-3)+\dfrac{9}{\sqrt{a}-3}\geq2\sqrt{(\sqrt{a}-3)\cdot\dfrac{9}{\sqrt{a}-3}}=6$ (vì $a>9$).\\
Từ đó $\min A=12\Leftrightarrow a=36$ (thỏa mãn điều kiện xác định).	
\end{enumerate}}
\end{ex}

\begin{ex}%[Đề thi Khảo sát chất lượng, năm học 2017-2018, Long Biên, Hà Nội]%[9D3K6]
	\textit{Giải bài toán sau bằng cách lập phương trình hoặc hệ phương trình}:\\
	Hai đội công nhân làm chung một công việc và dự định $12$ ngày thì hoàn thành xong. Nhưng khi làm chung được $8$ ngày, thì đội $\mathrm{I}$ được điều động đi làm việc khác. Đội $\mathrm{II}$ tiếp tục làm nốt phần việc còn lại. Khi làm một mình, do cải tiến cách làm, năng suất của đội $\mathrm{II}$ tăng gấp đôi, nên đội $\mathrm{II}$ đã hoàn thành xong phần việc còn lại trong $3{,}5$ ngày. Hỏi với năng suất ban đầu, nếu mỗi đội làm một mình thì sau thời gian bao lâu sẽ hoàn thành công việc trên?
	\loigiai{
	Gọi thời gian đội $\mathrm{I}$ làm một mình (với năng suất ban đầu) để hoàn thành công việc là $x$ (đơn vị ngày, $x>12$)\\
	Gọi thời gian đội $\mathrm{II}$ làm một mình (với năng suất ban đầu) để hoàn thành công việc là $y$ (đơn vị ngày, $y>12$)\\
	Mỗi ngày đội $\mathrm{I}$ làm được $\dfrac{1}{x}$ công việc.\\
	Mỗi ngày đội $\mathrm{II}$ làm được $\dfrac{1}{y}$ công việc.\\
	$8$ ngày làm được $\dfrac{8}{12}=\dfrac{2}{3}$ (công việc),\\
	Năng suất mới của đội $\mathrm{II}$ là $\dfrac{2}{y}$ (công việc/ngày)\\
	Ta có hệ phương trình: $\begin{cases}
	\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{1}{12}\\
	\dfrac{2}{3}+\dfrac{2}{y}\cdot\dfrac{7}{2}=1
	\end{cases}$\\
	Giải hệ phương trình được nghiệm $x=28$, $y=21$ (thỏa mãn đều kiện).\\
	Vậy với năng suất ban đầu, để hoàn thành công việc, đội $\mathrm{I}$ làm trong $28$ ngày, đội $\mathrm{II}$ làm trong $21$ ngày.}
\end{ex}

\begin{ex}%[Đề thi Khảo sát chất lượng, năm học 2017-2018, Long Biên, Hà Nội]%[9D3B4]
	Giải hệ phương trình: $\begin{cases}
	\dfrac{1}{x-2}+\dfrac{1}{y-1}=2\\
	\dfrac{2}{x-2}-\dfrac{3}{y-1}=1
	\end{cases}$
	\loigiai{
	Điều kiện xác định: $\begin{cases}
	x\ne2\\
	y\ne1
	\end{cases}$.\\
	Đặt ẩn phụ $\dfrac{1}{x-2}=x$, $\dfrac{1}{y-1}=v$. Ta có hệ: $\begin{cases}
	u+v=2\\
	2u-3v=1
	\end{cases}$.\\
	Giải hệ ta được: $u=\dfrac{7}{5}$ và $v=\dfrac{3}{5}$.\\
	Trở lại ẩn ban đầu và giải được $x=\dfrac{19}{7}$, $y=\dfrac{8}{3}$ (thỏa mãn điều kiện).\\
	Vậy hệ phương trình có nghiệm duy nhất $\left(\dfrac{19}{7};\dfrac{18}{3}\right)$.}
\end{ex}

\begin{ex}%[Đề thi Khảo sát chất lượng, năm học 2017-2018, Long Biên, Hà Nội]%[9D4K2]
	Cho parabol $(P)\colon y=x^2$ và đường thẳng $d\colon y=(2m+1)x-2m$. ($x$ là ẩn số, $m$ là tham số)
	\begin{enumerate}[a)]
		\item Khi $m=1$. Xác định tọa độ giao điểm của $d$ và $(P)$.
		\item Tìm $m$ để $(d)$ và $(P)$ cắt nhau tại hai điểm phân biệt $A(x_1;y_1)$, $B(x_2;y_2)$ sao cho biểu thức $T=x_!^2+x_2^2-x_1x_2$ đạt giá trị nhỏ nhất.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}[a)]
		\item Khi $m=1$, hoành độ giao điểm của $d$ và $(P)$ là nghiệm của phương trình $x^2-3x+2=0$.\hfill$(1)$\\
		Giải phương trình $(1)$ ta được nghiệm $x_1=1$, $x_2=2$.\\
		Ta có các tung độ giao điểm tương ứng là $y_1=1$, $y_2=4$.\\
		Vậy tọa độ giao điểm của $d$ và $(P)$ là: $M(1;1)$, $N(2;4)$.
		\item Hoành độ giao điểm của $d$ và $(P)$ là nghiệm của phương trình $x^2-(2m+1)x+2m=0$.\hfill$(*)$\\
		Tính được $\Delta=(2m-1)^2$.\\
		$(P)$ và $d$ cắt nhau tại hai điểm phân biệt khi và chỉ khi phương trình $(*)$ có hai nghiệm phân biệt $\Leftrightarrow \Delta>0\Leftrightarrow m\ne\dfrac{1}{2}$.\\
		Khi đó ta có:
		\begin{align*}
		T&=x_1^2+x_2^2=x_1^2+x_2^2+2x_1x_2-3x_1x_2\\
		&=(x_1+x_2)^2-3x_1x_2
		\end{align*}
		Áp dụng hệ thức Vi-ét cho phương trình $(*)$ ta có: 
		$\begin{cases}
		x_1+x_2=2m+1\\
		x_1x_2=2m
		\end{cases}$\\
		Thay vào biểu thức $T$, ta được \\		$T=(2m+1)^2-3\cdot2m=4m^2-2m+1=\left(2m-\dfrac{1}{2}\right)^2+\dfrac{3}{4}\geq\dfrac{3}{4}$.\\
		Vậy $T_{\min}=\dfrac{3}{4}$ khi $m=\dfrac{1}{4}$ (thỏa mãn điều kiện).
	\end{enumerate}}
\end{ex}

\begin{ex}%[Đề thi Khảo sát chất lượng, năm học 2017-2018, Long Biên, Hà Nội]%[9H3B7]%[9H3B4]
	Từ một điểm $A$ nằm ngoài đường tròn $(O;R)$ vẽ tiếp tuyến $AB$, $AC$ với đường tròn ($B$, $C$ là tiếp điểm). Trên cung nhỏ $BC$ lấy một điểm $M$ bất kỳ, vẽ $MI$ vuông góc với $AB$, $MK$ vuông góc với $AC$ ($I\in AB$, $K\in AC$)
	\begin{enumerate}[a)]
		\item Chứng minh tứ giác $AIMK$ nội tiếp đường tròn.
		\item Vẽ $MP$ vuông góc với $BC$ ($P\in BC$). Chứng minh: $\widehat{MPK}=\widehat{MBC}$.
		\item Chứng minh rằng: $MI\cdot MK=MP^2$.
		\item Xác định vị trí điểm $M$ trên cung nhỏ $BC$ để tích $MI\cdot MK\cdot MP$ đạt giá trị lớn nhất.
	\end{enumerate}
\loigiai{
	\begin{center}
		\begin{tikzpicture}
		\tkzDefPoints{0/0/O, 3/0/J, -8/0/A}	
		\tkzDrawCircle[radius](O,J)
		\tkzTangent[from=A](O,J)%đn tiếp tuyến qua K đến đường tròn
		\tkzGetPoints{C}{B}
		\tkzDrawLines[add=0 and 0](A,B)
		\tkzDrawLines[add=0 and 0](A,C)
		\tkzDefPointBy[rotation = center O angle 200](J)
		\tkzGetPoint{M}	
		\tkzDrawAltitude(A,B)(M)%vẽ đường cao hạ từ K của tam giác KAB
		\tkzGetPoint{I}
		\tkzDrawAltitude(A,C)(M)%vẽ đường cao hạ từ K của tam giác KAB
		\tkzGetPoint{K}
		\tkzDrawAltitude(B,C)(M)%vẽ đường cao hạ từ K của tam giác KAB
		\tkzGetPoint{P}
		\tkzDrawAltitude(B,C)(O)%vẽ đường cao hạ từ K của tam giác KAB
		\tkzGetPoint{H}
		\tkzInterLL(M,O)(B,C)
		\tkzGetPoint{D}
		\pgfresetboundingbox
		\tkzDrawSegments(A,B A,C M,I M,K M,A M,B M,C M,P B,C O,M P,K O,B O,C)
		\tkzDrawPoints(O,A,B,C,M,I,K,P,H,D)
		\tkzLabelPoints[above](A,I,B)
		\tkzLabelPoints[below](C,K)
		\tkzLabelPoints[below right](P,D)
		\tkzLabelPoints[below left](M)
		\tkzLabelPoints[right](O)
		\tkzLabelPoints[left](H)
		\tkzMarkRightAngles[size=0.3](M,I,A O,H,C M,P,C M,K,A)
		\tkzMarkAngles[size=0.6](M,P,K M,B,C)
		\end{tikzpicture}
	\end{center}
\begin{enumerate}[a)]
	\item Ta có: $\widehat{AIM}=\widehat{AKM}=90^{\circ}$\\
	$\Rightarrow \widehat{AIM}+\widehat{AKM}=180^{\circ}$.\\
	Suy ra tứ giác $AIMK$ nội tiếp đường tròn đường kính $AM$.
	\item Tứ giác $CPMK$ có $\widehat{MPC}=\widehat{MKC}=90^{\circ}$.\\
	$\Rightarrow \widehat{MPC}+\widehat{MKC}=180^{\circ}$.\\
	Suy ra tứ giác $CPMK$ là tứ giác nội tiếp.\\
	$\Rightarrow \widehat{MPK}=\widehat{MCK}$ (góc nội tiếp cùng chắn cung $MK$)\hfill$(1)$\\
	Vì $KC$ là tiếp tuyến của $(O)$ nên ta có: $\widehat{MCK}=\widehat{MBC}$ (góc nội tiếp và góc tạo bởi tiếp tuyến và dây cung cùng chắn cung $MC$)\hfill$(2)$\\
	Từ $(1)$ và $(2)$ suy ra $\widehat{MPK}=\widehat{MBC}$\hfill$(3)$
	\item Chứng minh tương tự câu b) ta có tứ giác $BPMI$ nội tiếp.\\
	Suy ra $\widehat{MIP}=\widehat{MBP}=\widehat{MBC}$\hfill$(4)$\\
	Từ $(3)$ và $(4)$ suy ra $\widehat{MPK}=\widehat{MIP}$.\\
	Tương tự: $\widehat{MKP}=\widehat{MPI}\Rightarrow \triangle MPK=\triangle MIP$\\
	$\Rightarrow \dfrac{MP}{MK}\backsim\dfrac{MI}{MP}\Rightarrow MI\cdot MK=MP^2$.
	\item Từ câu c) suy ra $MI\cdot MK\cdot MP=MP^3$.\\ 
	Do đó $MI\cdot MK\cdot MP$ lớn nhất khi và chỉ khi $MP$ lớn nhất.\hfill$(5)$\\
	Gọi $H$ là hình chiếu của $O$ trên $BC$\\
	$\Rightarrow OH$ là hằng số (do $BC$ cố định).\\
	Gọi $D$ là giao điểm của $MO$ với $BC$.\\
	Ta có: $MP\leq MD$; $OH\leq OD$\\
	$MP+OH\leq MD+OD=MO$\\
	$\Rightarrow MP+OH\leq R\Rightarrow MP\leq R-OH$.\\
	Do đó $MP$ lớn nhất bằng $R=OH$ khi và chỉ khi $O,H,M$ thẳng hàng ($M$ nằm chính giữa cung nhỏ $BC$)\hfill$(6)$\\
	Từ $(5)$ và $(6)$ suy ra $\max(MI\cdot MK\cdot MP)=(R-OH)^3$ khi và chỉ khi $M$ nằm chính giữa cung nhỏ $BC$.
	\end{enumerate}}
\end{ex}

\begin{ex}%[Đề thi Khảo sát chất lượng, năm học 2017-2018, Long Biên, Hà Nội]%[9D5G5]
	Cho ba số $x$, $y$, $z$ không âm và $x^2+y^2+z^2\leq 3y$. Tìm giá trị nhỏ nhất của biểu thức $P=\dfrac{1}{(x+1)^2}+\dfrac{4}{(y+2)^2}+\dfrac{8}{(z+3)^2}$.
	\loigiai{Theo bất đẳng thức Cô-si ta có:\\
		$(x^1+1)+(y^2+4)+(z^2+1)\geq2x+4y+2z$\\
		$\Rightarrow 3y+6\geq2x+4y+2z$ (vì $x^2+y^2+z^3\leq3y$)\\
		Với hai số $a,b>0$ chứng minh được: $\dfrac{1}{a^2}+\dfrac{1}{b^2}\geq\dfrac{8}{(a+b)^2}$.\\
		Do đó:
		\begin{align*}
		P&=\dfrac{1}{(x+1)^2}+\dfrac{1}{\left(\dfrac{y}{2}+1\right)^2}+\dfrac{8}{(z+3)^2}\geq\dfrac{8}{\left(x+1+\dfrac{y}{2}+1\right)^2}+\dfrac{8}{(z+3)^2}\\
		&\geq\dfrac{64\cdot4}{(2x+y+2z+10)^2}\geq\dfrac{256}{(6+10)^2}=1
		\end{align*}
		Dấu \lq\lq{}$=$\rq\rq{} xảy ra khi và chỉ khi $x=1$, $y=2$, $z=1$.\\
		Vậy $P_{\min}=1$ khi và chỉ khi $x=1$, $y=2$, $z=1$.}
\end{ex}
