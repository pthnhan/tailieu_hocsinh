\begin{ex}%[Đề thi vào Chuyên KHTN Hà Nội năm 2013, vòng 1]%[9D4B7]
Giải phương trình $\sqrt{3x+1}+\sqrt{2-x}=3$.
\loigiai{
Điều kiện $-\dfrac{1}{3}\le x\le 2$.
\begin{eqnarray*}
\sqrt{3x+1}+\sqrt{2-x}=3 & \Leftrightarrow & 2x+3+2\sqrt{-3x^2+5x+2}=9\\
& \Leftrightarrow & \sqrt{-3x^2+5x+2}=-x+3\\
& \Leftrightarrow & 4x^2-11x+7=0\Leftrightarrow \hoac{& x=1\\ & x=\dfrac{7}{4}}\,\,\text{(thỏa mãn).} 
\end{eqnarray*}
Vậy nghiệm phương trình là $x\in\left\{1;\dfrac{7}{4}\right\}$.
}
\end{ex}

\begin{ex}%[Đề thi vào Chuyên KHTN Hà Nội năm 2013, vòng 1]%[9D5G5]
Giải hệ phương trình $\heva{& x+y+\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{9}{2}\\ & \dfrac{1}{4}+\dfrac{3}{2}\left(x+\dfrac{1}{y}\right)=xy+\dfrac{1}{xy}.}$
\loigiai{
Điều kiện: $x\neq 0$, $y\neq 0$.\\
Ta có $\left(x+\dfrac{1}{y}\right)\left(y+\dfrac{1}{x}\right)=xy+\dfrac{1}{xy}+2$. Nên ta có hệ phương trình tương đương
$$\heva{& \left(x+\dfrac{1}{y}\right)+\left(y+\dfrac{1}{x}\right)=\dfrac{9}{2}\\ & \dfrac{1}{4}+\dfrac{3}{2}\left(x+\dfrac{1}{y}\right)=\left(x+\dfrac{1}{y}\right)\left(y+\dfrac{1}{x}\right)-2.}$$
Đặt $u=x+\dfrac{1}{y}$, $v=y+\dfrac{1}{x}$ ta được hệ phương trình
$$\heva{& u+v=\dfrac{9}{2}\\ & \dfrac{1}{4}+\dfrac{3}{2}u=uv-2}\Leftrightarrow \heva{ & v=\dfrac{9}{2}-u\\ & \dfrac{9}{4}+\dfrac{3u}{2}=u\left(\dfrac{9}{2}-u\right)}\Leftrightarrow \heva{& u=\dfrac{3}{2}\\ & v=3.}$$
Khi đó
$$\heva{ & x+\dfrac{1}{y}=\dfrac{3}{2}\\ & y+\dfrac{1}{x}=3}\Leftrightarrow \heva{& 2xy+2=3y\\ & xy+1=3x}\Leftrightarrow \heva{ & y=2x\\ & 2x^2-3x+1=0}\Leftrightarrow \hoac{& x=1, y=2\\ & x=\dfrac{1}{2},y=1}\,\,\text{(thỏa mãn).}$$
Vậy nghiệm hệ phương trình là $(1;2)$; $\left(\dfrac{1}{2};1\right)$.
}
\end{ex}

\begin{ex}%[Đề thi vào Chuyên KHTN Hà Nội năm 2013, vòng 1]%[8D2G9]
Với $a,b,c$ là các số thực khác $0$ thỏa mãn đẳng thức $(a+b)(b+c)(c+a)=8abc$. Chứng minh rằng
$$\dfrac{a}{a+b}+\dfrac{b}{b+c}+\dfrac{c}{c+a}=\dfrac{3}{4}+\dfrac{ab}{(a+b)(b+c)}+\dfrac{bc}{(b+c)(c+a)}+\dfrac{ca}{(c+a)(a+b)}.$$
\loigiai{
Đẳng thức cần chứng minh tương đương với
\begin{eqnarray*}
&& \dfrac{a}{a+b}\left(1-\dfrac{b}{b+c}\right)+\dfrac{b}{b+c}\left(1-\dfrac{c}{c+a}\right)+\dfrac{c}{c+a}\left(1-\dfrac{a}{a+b}\right)=\dfrac{3}{4}\\
& \Leftrightarrow & \dfrac{ac}{(a+b)(b+c)}+\dfrac{ba}{(b+c)(c+a)}+\dfrac{cb}{(c+a)(a+b)}=\dfrac{3}{4}\\
& \Leftrightarrow & ac(a+c)+ba(b+a)+cb(c+b)=\dfrac{3}{4}(a+b)(b+c)(c+a)\\
& \Leftrightarrow & ac(a+c)+ba(b+a)+cb(c+b)=6abc\\
& \Leftrightarrow & ac(a+c)+b^2(a+c)+ab(a+c)+bc(a+c)=8abc\\
& \Leftrightarrow & (a+c)(ac+b^2+ab+bc)=8abc\\
& \Leftrightarrow & (a+b)(b+c)(c+a)=8abc\quad\text{(giả thiết).}
\end{eqnarray*}
Vậy ta có điều cần chứng minh.
}
\end{ex}

\begin{ex}%[Đề thi vào Chuyên KHTN Hà Nội năm 2013, vòng 1]%[9D5G2]
Có bao nhiêu số nguyên dương có năm chữ số $\overline{abcde}$ sao cho $\overline{abc}-(10d+e)$ chia hết cho $101$?
\loigiai{
Ta có
\begin{eqnarray*}
\overline{abcde} & = & \overline{abc}\cdot 100+\overline{de}\\
& = & \overline{abc}\cdot 101 - (\overline{abc}-\overline{de})\\
& = & \overline{abc}\cdot 101 - (\overline{abc}-(10+e)).
\end{eqnarray*}
Do đó, $\overline{abc}-(10d+e)$ chia hết cho $101$ khi và chỉ khi $\overline{abcde}$ chia hết cho $101$.\\
Ta có $101\cdot m\le 99999\Leftrightarrow m\le \dfrac{99999}{101}=990+\dfrac{9}{101}.$\\
Suy ra số có năm chữ số lớn nhất chia hết cho $101$ là $990\cdot 101$.\\
Lại có $101\cdot n> 9999\Leftrightarrow n>\dfrac{999}{101}=99$.\\
Suy ra số có năm chữ số nhỏ nhất chia hết cho $101$ là $100\cdot 101$.\\
Vậy số các số có năm chữ số thỏa mãn yêu cầu đề bài là $990-100+1=891$ số.
}
\end{ex}

\begin{ex}%[Đề thi vào Chuyên KHTN Hà Nội năm 2013, vòng 1]%[9H3G3]
Cho $\triangle ABC$ nội tiếp đường tròn $(O)$ với $AB<AC$. Đường phân giác của $\widehat{BAC}$ cắt $(O)$ tại điểm $D$ khác $A$. Gọi $M$ là trung điểm $AD$ và $E$ là điểm đối xứng với $D$ qua tâm $O$. Giả sử đường tròn ngoại tiếp tam giác $ABM$ cắt đoạn thẳng $AC$ tại điểm $F$ khác $A$. Chứng minh rằng
\begin{multicols}{2}
\begin{enumerate}
\item $\triangle BDM\backsim \triangle BCF$.
\item $EF\perp AC$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1.5]
\tkzInit[xmin=-1.5,xmax=5.5,ymin=-1.3, ymax=6] \tkzClip[space=.1]
\tkzDefPoints{0/0/B, 4/0/C, -1/3/A}
\tkzDefCircle[circum](A,B,C)\tkzGetPoint{O}
\tkzDrawCircle(O,A)
\tkzDefMidPoint(C,B) \tkzGetPoint{N}
\tkzInterLC(O,N)(O,A) \tkzGetPoints{E}{D}
\tkzDefMidPoint(A,D) \tkzGetPoint{M}
\tkzDefCircle[circum](A,B,M)\tkzGetPoint{I}
\tkzDrawCircle[dashed](I,A)
\tkzInterLC(A,C)(I,A) \tkzGetPoints{A}{F}
\tkzMarkRightAngles(E,A,D E,C,D B,N,D)


\tkzDrawSegments(A,B B,C C,A D,E A,D E,A E,F E,C F,B F,M M,B B,D D,C)
\tkzDrawPoints[fill=white,size=8](A,B,C,O,D,E,F,M,N)

\tkzLabelPoints[above](E)
\tkzLabelPoints[above left](A)
\tkzLabelPoints[right](O,F,M)
\tkzLabelPoints[below](D,B)
\tkzLabelPoints[below right](C,N)

\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Ta có $\widehat{BDM}=\widehat{BCA}$ (tính chất góc nội tiếp trong đường tròn).\\
Lại có $\widehat{FBM}=\widehat{FAM}=\widehat{CBD}$ (tính chất góc nội tiếp trong đường tròn).\\
Do đó $\widehat{FBM}+\widehat{MBC}=\widehat{CBD}+\widehat{MBC}$, hay $\widehat{FBC}=\widehat{MBD}$.\\
Nên $\triangle BDM\backsim \triangle BCF$ (g.g).
\item Ta sẽ chứng minh cho $\triangle CEF\backsim\triangle DEA$. Thật vậy, do $\triangle BDM\backsim \triangle BCF$ nên 
$$\dfrac{DM}{CF}=\dfrac{BD}{BC}\Leftrightarrow \dfrac{2\cdot DM}{CF}=\dfrac{2\cdot BD}{BC}\Leftrightarrow\dfrac{DA}{CF}=\dfrac{BD}{BN}.$$
Với $N=OD\cap BC$ là trung điểm $BC$. Mà $\triangle BDN\backsim \triangle EDC$ nên $\dfrac{BD}{BN}=\dfrac{DE}{CE}$.\\
Từ đó ta có $\dfrac{DM}{CF}=\dfrac{DE}{CE}$. Và $\triangle CEF\backsim\triangle DEA$ (c.g.c).\\
Suy ra $\widehat{EFC}= \widehat{EAD}=90^{\circ}$. Vậy $EF\perp AC$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào Chuyên KHTN Hà Nội năm 2013, vòng 1]%[9D5G1]
Với $a,b,c,d$ là các số thực dương thỏa mãn điều kiện $abc+bcd+cda+dab=1$. Tìm giá trị nhỏ nhất của biểu thức $P=4(a^3+b^3+c^3)+9d^3$.
\loigiai{
Với $\alpha$ là số thực dương, áp dụng BĐT Cô-si (Cauchy, AM-GM) cho ba số ta có 
\begin{eqnarray*}
\dfrac{d^3}{3}+\dfrac{a^3}{3\alpha ^3}+\dfrac{b^3}{3\alpha ^3} & \ge & \dfrac{dab}{\alpha ^2},\\
\dfrac{d^3}{3}+\dfrac{b^3}{3\alpha ^3}+\dfrac{c^3}{3\alpha ^3} & \ge & \dfrac{dbc}{\alpha ^2},\\
\dfrac{d^3}{3}+\dfrac{c^3}{3\alpha ^3}+\dfrac{a^3}{3\alpha ^3} & \ge & \dfrac{dca}{\alpha ^2},\\
\dfrac{a^3+b^3+c^3}{3\alpha ^2} & \ge & \dfrac{abc}{\alpha ^2}.
\end{eqnarray*}
Từ đó suy ra
$$d^3+\left(\dfrac{2}{3\alpha^3}+\dfrac{1}{3\alpha^2}\right)\left(a^3+b^3+c^3\right)\ge \dfrac{1}{\alpha ^2}(dab+dbc+dca+abc)=\dfrac{1}{\alpha^2}.$$
Bây giờ, ta chọn $\alpha >0$ sao cho $\dfrac{2}{3\alpha^3}+\dfrac{1}{3\alpha^2}=\dfrac{4}{9}\Leftrightarrow 4\alpha^3-3\alpha-6=0$.\\
Đặt $\alpha=\dfrac{1}{2}\left(x+\dfrac{1}{x}\right)$ ta được
$$\dfrac{1}{2}\left(x+\dfrac{1}{x}\right)^3-\dfrac{3}{2}\left(x+\dfrac{1}{x}\right)=6\Leftrightarrow x^6-12x^3+1=0\Leftrightarrow \hoac{& x=\sqrt[3]{6+\sqrt{35}}\\ & x=\sqrt[3]{6-\sqrt{35}}.}$$
Ta được $\alpha=\dfrac{1}{2}\left(\sqrt[3]{6+\sqrt{35}}+\sqrt[3]{6-\sqrt{35}}\right)$. Khi đó
$$d^3+\dfrac{4}{9}\left(a^3+b^3+c^3\right)\ge \dfrac{1}{\alpha ^2}\Leftrightarrow 9d^3+4\left(a^3+b^3+c^3\right)\ge \dfrac{9}{\alpha ^2}=\dfrac{36}{\left(\sqrt[3]{6+\sqrt{35}}+\sqrt[3]{6-\sqrt{35}}\right)^2}.$$
Đẳng thức xảy ra khi $a=b=c=\sqrt[3]{\dfrac{\alpha}{\alpha+3}}$, $d=\sqrt[3]{\dfrac{1}{\alpha^3+3\alpha^2}}$.\\
Vậy giá trị nhỏ nhất của $P$ là $\dfrac{36}{\left(\sqrt[3]{6+\sqrt{35}}+\sqrt[3]{6-\sqrt{35}}\right)^2}$.
}
\end{ex}
