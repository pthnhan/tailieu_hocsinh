\begin{ex}%[Tuyển sinh 10 Hà Nam, 2018 - 2019]%[Nguyễn Gia Minh, dự án EXTHCS-06]%[9D3B4]
Giải hệ phương trình: $\left\{\begin{aligned}&x+y=25\\&2x-1=y+4.\end{aligned}\right.$
\loigiai{
	\begin{eqnarray*}
	\left\{\begin{aligned}&x+y=25\\&2x-1=y+4\end{aligned}\right.&\Leftrightarrow& \left\{\begin{aligned}&x+y=25\\&2x-y=5\end{aligned}\right.\\
	&\Leftrightarrow& \left\{\begin{aligned}&x+y=25\\&3x=30\end{aligned}\right.\\
	&\Leftrightarrow&\left\{\begin{aligned}&x+y=25\\&x=10\end{aligned}\right.\\
	&\Leftrightarrow& \left\{\begin{aligned}&x=10\\&y=15.\end{aligned}\right.
	\end{eqnarray*}
Vậy hệ phương trình đã cho có nghiệm $\left\{\begin{aligned}&x=10\\&y=15.\end{aligned}\right.$
}
\end{ex}

\begin{ex}%[Tuyển sinh 10 Hà Nam, 2018 - 2019]%[Nguyễn Gia Minh, dự án EXTHCS-06]%[9D1K8]%[9D5G2]
Cho biểu thức $B=\dfrac{\sqrt{a}}{\sqrt{a}-3}-\dfrac{3}{\sqrt{a}+3}-\dfrac{a-2}{a-9}$ với $a\ge 0,a\ne 9$.\\ 
Rút gọn $B$. Tìm các số nguyên $a$ để $B$ nhận giá trị nguyên.
\loigiai{Với $a\ge 0,a\ne 9$, ta có 
	\begin{eqnarray*}
	B&=&\dfrac{\sqrt{a}}{\sqrt{a}-3}-\dfrac{3}{\sqrt{a}+3}-\dfrac{a-2}{a-9}\\
	&=&\dfrac{\sqrt{a}\left(\sqrt{a}+3\right)}{a-9}-\dfrac{3\left(\sqrt{a}-3\right)}{a-9}-\dfrac{a-2}{a-9}\\
	&=&\dfrac{11}{a-9}.
	\end{eqnarray*}
\begin{eqnarray*}
B&=&\dfrac{11}{a-9}\in\mathbb{Z}\Leftrightarrow 11 \text{ chia hết cho } a-9\\
&\Leftrightarrow&\left[\begin{aligned}&a-9=1\\&a-9=-1\\&a-9=11\\&a-9=-11\end{aligned}\right. \Leftrightarrow \left[\begin{aligned}&a=10\\&a=8\\&a=20\\&a=-2 \text{ (loại).}\end{aligned}\right.
\end{eqnarray*}
Vậy $a\in\{8,10,20\}$ thì $B$ nhận giá trị nguyên.
}
\end{ex}
\begin{ex}%[Tuyển sinh 10 Hà Nam, 2018 - 2019]%[Nguyễn Gia Minh, dự án EXTHCS-06]%[9D4B2]%[9D4K2]
Trong mặt phẳng tọa độ $Oxy$, cho parabol $(P)$ có phương trình $y=x^2$ và đường thẳng $d$ có phương trình: $y=5x-m+2$ ($m$ là tham số).
\begin{enumerate}
	\item Điểm $A\left(2;4\right)$ có thuộc đồ thị hàm số $P$ không? Tại sao?
	\item Tìm $m$ để đường thẳng $d$ cắt parabol $(P)$ tại hai điểm phân biệt có tung độ $y_1,y_2$ thỏa mãn $y_1+y_2+y_1y_2=25$.
\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Thay tọa độ của $A$ vào phương trình của $(P)$ ta có $4=2^2$ (thỏa mãn) $\Rightarrow A(2;4)\in (P)$.
		\item Xét phương trình hoành độ giao điểm của $d$ và $(P)$ : \\$x^2-5x+m-2=0$, có $\Delta=25-4(m-2)=33-4m$.\\
		Để $d$ cắt $(P)$ tại hai điểm phân biệt thì $33-4m>0$ hay $m<\dfrac{33}{4}$.
	\end{enumerate}
}
\end{ex}
\begin{ex}%[Tuyển sinh 10 Hà Nam, 2018 - 2019]%[Nguyễn Gia Minh, dự án EXTHCS-06]%[9H3K7]%[9H3K3]%[8H3K8]%[8H3G8]
Cho đường tròn $\left(O,R\right)$ và một điểm $A$ sao cho $OA=3R$. Qua $A$ kẻ hai tiếp tuyến $AB$, $AC$ với đường tròn $(O)$ ($B$, $C$ là hai tiếp điểm). Lấy điểm $M$ thuộc đường tròn $(O)$ sao cho $BM\parallel AC$. Gọi $N$ là giao điểm thứ hai của của đường thẳng $AM$ với đường tròn $(O)$, $K$ là giao điểm của hai đường thẳng $BN$ và $AC$.
\begin{enumerate}
	\item Chứng minh $ABOC$ là tứ giác nội tiếp.
	\item Chứng minh $KA^2=KB\cdot KN$.
	\item Tính độ dài đoạn thẳng $AK$ theo $R$.
	\item Tiếp tuyến tại $M$ và $N$ của $(O)$ cắt nhau tại $E$. Chứng minh $E$, $B$, $C$ thẳng hàng.
\end{enumerate}
\loigiai{
	\begin{center}
\begin{tikzpicture}[>=stealth,scale=.7, line join = round, line cap = round]

\tkzDefPoints{0/0/O,-2.5/0/A'}
\tkzDefPointBy[homothety = center O ratio 3](A')\tkzGetPoint{A}
\tkzDrawCircle[radius](O,A')
\tkzTangent[from=A](O,A')    \tkzGetPoints{C}{B}
\tkzDefLine[parallel=through B](A,C)\tkzGetPoint{b}
\tkzInterLC(B,b)(O,A')\tkzGetPoints{}{M}
\tkzInterLC(A,M)(O,A')\tkzGetPoints{N}{}
\tkzInterLL(A,C)(B,N)\tkzGetPoint{K}

\tkzInterLL(M,m)(N,n)\tkzGetPoint{E}
\tkzInterLL(A,O)(B,C)\tkzGetPoint{H}
\tkzInterLL(E,O)(M,N)\tkzGetPoint{I}

\end{tikzpicture}
	\end{center}
		\begin{enumerate}
		\item $AB$, $AC$ là tiếp tuyến của $(O)\Rightarrow \widehat{ABO}=\widehat{ACO}=90^\circ\Rightarrow \widehat{ABO}+\widehat{ACO}=180^\circ$\\
		$\Rightarrow$ Tứ giác $ABOC$ là tứ giác nội tiếp.
		\item 
		Vì $BM\parallel AC\Rightarrow \widehat{BMN}=\widehat{NAK}$.\\
		Mà $\widehat{BMN}=\widehat{NAK}$ (cùng chắn cung $NB$) $\Rightarrow \widehat{KAN}=\widehat{ABK}$.\\
		Xét $\triangle KAB$ và $\triangle KNA$ có: 
		\begin{eqnarray*}
			&&\widehat{KAN}=\widehat{ABK}\\
			&&\widehat{AKB} \text{ chung}\\
			&\Rightarrow& \triangle KAB\backsim \triangle KNA\\
			&\Rightarrow& \dfrac{KA}{KB}=\dfrac{KN}{KA}\Rightarrow KA^2=KB\cdot KN.
		\end{eqnarray*}	
	\item Chứng minh tương tự như câu b) suy ra $\triangle \backsim \triangle KCN\Rightarrow KC^2=KB\cdot KN$.\\
	Suy ra $KA=KC=\dfrac{1}{2}AC$.\\
	Tam giác $ACO$ vuông tại $C$, suy ra $AC^2=AO^2-OC^2=8R^2\Rightarrow AK=R\sqrt{2}$.
	\item Gọi $H$ là giao điểm của $AO$ và $BC$, $I$ là giao điểm của $EO$ và $MN$ $\Rightarrow AO\perp BC$ tại $H$, $EO\perp NM$ tại $I$.\\
	$\triangle AOC$ vuông tại $C$, có $CH$ là đường cao $\Rightarrow OH\cdot OA=OC^2=R^2.\quad (1)$\\
	$\triangle EMO$ vuông tại $M$, có $MI$ là đường cao $\Rightarrow OI\cdot OE=OM^2=R^2.\quad (2)$\\
	Từ $(1)$ và $(2)$ suy ra $OH\cdot OA=OI\cdot OE\Rightarrow \dfrac{OH}{OI}=\dfrac{OE}{OA}$.\\
	Xét $\triangle OHE$ và $\triangle OIA$ có: 
	\begin{eqnarray*}
	&&\dfrac{OH}{OI}=\dfrac{OE}{OA}\\
	&&\widehat{AOE} \text{ chung}\\
	&\Rightarrow& \triangle \backsim \triangle OIA\\
	&\Rightarrow& \widehat{OHE}=\widehat{OIA}=90^\circ \text{ hay $EH$ vuông góc với $AO$ tại $H$.}
	\end{eqnarray*}
Mà $BC\perp AO$ tại $H$ nên $E$, $B$, $C$ thẳng hàng.
	\end{enumerate}
}

\begin{ex}%[Tuyển sinh 10 Hà Nam, 2018 - 2019]%[Nguyễn Gia Minh, dự án EXTHCS-06]%[9D5G1]
Cho $a,b,c$ là độ dài ba cạnh của một tam giác có chu vi bằng $4$. Chứng minh $$\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}+8>9\left(\dfrac{1}{a+b}+\dfrac{1}{b+c}+\dfrac{1}{c+a}\right).$$
\loigiai{
	Ta có $a+b+c=4$.\\
	Bất đẳng thức cần chứng minh tương đương với \\
	$\left(\dfrac{9}{4-a}-\dfrac{1}{a}\right)+\left(\dfrac{9}{4-b}-\dfrac{1}{b}\right)+\left(\dfrac{9}{4-c}-\dfrac{1}{c}\right)<8$.\\
	Vì $a$, $b$, $c$ là ba cạnh của một tam giác
	\begin{eqnarray*}
	&\Rightarrow& a<b+c\Rightarrow 2a<a+b+c=4\\
	&\Rightarrow& a<2\Rightarrow a-2<0\\
	&\Rightarrow& (a-1)^2(a-2)\leq 0\quad (*) \text{ (luôn đúng)} \\
	(*)&\Leftrightarrow& a^3-4a^2+5a-2\leq 0\\
	&\Leftrightarrow& \dfrac{a^3-4a^2+5a-2}{a(4-a)}\leq 0\Leftrightarrow \dfrac{5a-2}{a(4-a)}\leq a\\
	&\Leftrightarrow& \dfrac{9}{2(4-a)}-\dfrac{1}{2a}\leq a\Leftrightarrow \dfrac{9}{4-a}-\dfrac{1}{a}\leq 2a\quad (1)
	\end{eqnarray*} 
Tương tự ta có 
\begin{eqnarray*}
&&\dfrac{9}{4-b}-\dfrac{1}{b}\leq 2b\quad (2)\\ 
&&\dfrac{9}{4-c}-\dfrac{1}{c}\leq 2c.\quad (3)
\end{eqnarray*}
Cộng hai vế tương ứng của $(1)$, $(2)$ và $(3)$ ta được
\begin{eqnarray*}
&&\left(\dfrac{9}{4-a}-\dfrac{1}{a}\right)+\left(\dfrac{9}{4-b}-\dfrac{1}{b}\right)+\left(\dfrac{9}{4-c}-\dfrac{1}{c}\right)\leq 2(a+b+c)=8\\
&\Leftrightarrow& \dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}+8\geq \left(\dfrac{1}{a+b}+\dfrac{1}{b+c}+\dfrac{1}{c+a}\right)
\end{eqnarray*}
Dấu ``='' xảy ra $\Leftrightarrow \left\{\begin{aligned}&a=b=c=1\\&a+b+c=4\end{aligned}\right.$ (vô lý).\\
Vậy $\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}+8> \left(\dfrac{1}{a+b}+\dfrac{1}{b+c}+\dfrac{1}{c+a}\right)$.
}
\end{ex}
