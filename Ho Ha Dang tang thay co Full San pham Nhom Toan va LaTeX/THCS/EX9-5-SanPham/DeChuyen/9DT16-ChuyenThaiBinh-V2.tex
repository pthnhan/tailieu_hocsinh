\begin{ex}%[Đề thi vào 10, Chuyên Thái Bình, 2016]%[9D4K7]%[9D4K3]
    \hfill
    \begin{enumerate}
        \item Giải phương trình $5x-(x+3)\sqrt{2x-1}-1=0$.
        \item Cho hai số thực $a$, $b$ bất kì. Chứng minh rằng ít nhất một trong hai phương trình sau có nghiệm
        \begin{align}
        x^2+2ax+3ab=0\\
        x^2+2bx-8ab=0.
		\end{align}   
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item Điều kiện: $x\ge \dfrac{1}{2}$.\\
        Phương trình tương đương với
        $$4(x-1)-(x+3)(\sqrt{2x-1}-1)=0$$
        $$\Leftrightarrow 4(x-1)-\dfrac{2(x+3)(x-1)}{\sqrt{2x-1}+1}=0$$
        $$\Leftrightarrow (x-1)\left(4-\dfrac{2(x+3)}{\sqrt{2x-1}+1}\right)=0$$
        $$\Leftrightarrow (x-1)\dfrac{(1-\sqrt{2x-1})(\sqrt{2x-1}-3)}{\sqrt{2x-1}+1}=0$$
        $$\Leftrightarrow \hoac{x&=1\, (\mbox{thỏa mãn})\\x&=5\, (\mbox{thỏa mãn}).}$$
        Vậy nghiệm của phương trình là $x=1$ và $x=5$.
        \item Phương trình $(1)$ có $\Delta_1=4a^2-12ab$\hfill $(3)$\\
        Phương trình $(2)$ có $\Delta_2'=b^2+8ab$\hfill $(4)$\\
        Từ $(3),(4)$ ta có $\Delta_1+\Delta_2'=4a^2-4ab+b^2=(2a-b)^2\ge 0$.\\
        Từ đó suy ra ít nhất một phương trình phải có nghiệm.
    \end{enumerate}
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Thái Bình, 2016]%[9D5G3]%[9D5G5]
\hfill
    \begin{enumerate}    
        \item Tìm các số nguyên $x$, $y$ thỏa mãn $9x^2+3y^2+6xy-6x+2y-35=0$.
        \item Cho $P(x)$ là đa thức bậc ba có hệ số bậc cao nhất bằng $1$ và thỏa mãn $P(2016)=2016$, $P(2016)=2018$. Tính giá trị của biểu thức $-3P(2018)+P(2019)$.
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item Ta có 
        $$9x^2+3y^2+6xy-6x+2y-35=0$$
        $$\Leftrightarrow 9x^2+y^2+1+6xy-6x-y+2y^2+4y+2=38$$
        $$(3x+y-1)^2+2(y+1)^2=38$$
        Từ đó suy ra $(3x+y-1)^2$ là số chính phương chẵn và nhỏ hơn $38$, nên ta có
        \begin{itemize}
        \item $(3x+y-1)^2=0\Rightarrow (y+1)^2=19$ (loại),
        \item $(3x+y-1)^2=4\Rightarrow (y+1)^2=17$ (loại),
        \item $(3x+y-1)^2=16\Rightarrow (y+1)^2=11$ (loại),
        \item $(3x+y-1)^2=36\Rightarrow (y+1)^2=1$ (thỏa mãn).
        \end{itemize} 
        Giải hệ $\heva{(3x+y-1)^2=36\\ (y+1)^2=1}$ ta được các cặp nghiệm nguyên: $(-1;-2); (3;-2)$.
        \item Đặt $Q(x)=P(x)-x-1$.\\ 
        Ta có $Q(2016)=Q(2017)=0$ nên $Q(x)=(x-2016)(x-2017)(x-a)$.\\
        Suy ra
        $$P(x)=(x-2016)(x-2017)(x-a)+x+1.$$
        Từ đó ta có
        $$-3P(2018)+P(2019)=-4031.$$
       
    \end{enumerate}
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Thái Bình, 2016]%[9D4G7]
    Giải hệ phương trình $$\heva{y^3+\sqrt{8x^4-2y}=2(2x^4+3)\\ \sqrt{2x^2+x+y}+2\sqrt{x+2y}=\sqrt{9x-2x^2+19y}.}$$
    
\loigiai
    {
Đặt $a=2x^2$, $b=x+y$, $c=x+2y$. Phương trình thứ hai trở thành
$$\sqrt{a+b}+2\sqrt{c}=\sqrt{10c-a-b}\Leftrightarrow a+b+2\sqrt{c(a+b)}=3c$$
$$\Leftrightarrow \left(\sqrt{a+b}+\sqrt{c}\right)^2=\left(2\sqrt{c}\right)^2\Leftrightarrow a+b=c.$$
	Từ đó suy ra $y=2x^2\ge 0\, (*)$, thay vào phương trình thứ nhất ta có
	$$y^3+\sqrt{2y^2-2y}=y^2+6$$
	$$\Leftrightarrow y^3-y^2-4+\sqrt{2y^2-2y}-2=0$$
	$$\Leftrightarrow (y-2)\left(y^2+y+2+\dfrac{2(y+1)}{\sqrt{2y^2-2y}+2}\right)=0\, (**)$$
	Do $(*)$ nên $(**)\Leftrightarrow y=2$.
	Từ đó suy ra $x=\pm 1$.
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Thái Bình, 2016]%[9H3K7]%[9H3K7]%[9H3G7]
    Từ một điểm $I$ nằm bên ngoài đường tròn $(O)$, vẽ các tiếp tuyến $IA$, $IB$ ($A$, $B$ là các tiếp điểm) và vẽ cát tuyến $ICD$ (không qua tâm $O$) với đường tròn ($C$ nằm giữa $I$ và $D$).
    \begin{enumerate}
        \item Chứng minh rằng $AC.BD=AD.BC$.
        \item Gọi $K$ là giao điểm của $CD$ và $AB$, $E$ là trung điểm của $OI$. Chứng minh rằng $KA.KB=OE^2-EK^2$.
        \item Gọi $H$ là trung điểm của $AB$. Chứng minh rằng $\widehat{ADH}=\widehat{IDB}$.
    \end{enumerate}
\loigiai
    {
    \begin{tikzpicture}
    \tkzDefPoints{0/0/I, 6/0/B, 6/3/O}
    \tkzDrawCircle[radius](O,B)
    \tkzDefCircle[radius](I,B)
    \tkzInterCC(O,B)(I,B)\tkzGetPoints{B}{A}
    \tkzDefPointBy[rotation = center O angle -50](B)\tkzGetPoint{C}
    \tkzInterLC(I,C)(O,B)\tkzGetFirstPoint{D}
    \tkzInterLL(A,B)(C,D)\tkzGetPoint{K}
    \tkzDefMidPoint(O,I)\tkzGetPoint{E}
    \tkzDefMidPoint(A,B)\tkzGetPoint{H}
    \tkzDrawPoints(I,B,O,A,C,D,K,E,H)
    \tkzDrawSegments(A,B I,A I,B I,O O,A O,B O,C O,D C,A C,B D,A D,B E,K D,H H,C)
    \tkzDrawLine[add = 0 and 0.1](I,D)
    \tkzLabelPoints[above left](A,E)
    \tkzLabelPoints[above left,yshift=-5pt](H)
    \tkzLabelPoints[below left](I)
    \tkzLabelPoints[below](B)
    \tkzLabelPoints[below,xshift=-4pt,yshift=-2pt](C,K)
    \tkzLabelPoints[xshift=-2.5pt](D,O)
    \tkzMarkAngles[](I,H,C D,H,O D,C,O O,D,C)
    \tkzMarkRightAngles(B,H,I O,B,I)
    \end{tikzpicture}
    \begin{enumerate}
        \item Ta có\\ $\left.
        			\begin{aligned}
        \widehat{BIC}&=\widehat{DIB}\\ 
        \widehat{IBC}&=\widehat{IDB}\, (\mbox{cùng chắn cung } \wideparen{BC})
                	\end{aligned}
             		\right\}\Rightarrow \triangle IBC\backsim \triangle IDB\,(\mbox{g-g})\Rightarrow \dfrac{IB}{ID}=\dfrac{IC}{IB}=\dfrac{BC}{DB}.$\hfill $(1)$\\
        	Ta lại có\\ $\left.
        			\begin{aligned}
        \widehat{AIC}&=\widehat{DIA}\\ 
        \widehat{IAC}&=\widehat{IDA}\, (\mbox{cùng chắn cung } \wideparen{AC})
                	\end{aligned}
             		\right\}\Rightarrow \triangle IAC\backsim \triangle IDA\,(\mbox{g-g})\Rightarrow \dfrac{IA}{ID}=\dfrac{AC}{DA}.$\hfill $(2)$\\
            Từ $(1)(2)$ suy ra $\dfrac{AC}{DA}=\dfrac{BC}{DB}\Rightarrow AC.BD=AD.BC$ (do $IA=IB$).
        \item Ta có
\begin{align*}
		KA.KB=&(AH+HK)(BH-HK)=AH^2-HK^2\\
		=&(AE^2-EH^2)-(EK^2-EH^2)=AE^2-EK^2=OE^2-EK^2.
\end{align*}             
        \item Dễ thấy tam giác $OBI$ vuông tại $B$ và có $BH$ là đường cao, suy ra $IB^2=IH.IO$.\hfill $(3)$\\
        Từ $(1)(3)$ ta được $IH.IO=IC.ID$, suy ra tứ giác $CDOH$ nội tiếp. Từ đó suy ra
        $$\widehat{OHD}=\widehat{OCD}=\widehat{ODC}=\widehat{CHI}\Rightarrow \widehat{CHK}=\widehat{DHK}.$$
        Vậy $HB$ là phân giác góc $\widehat{CHD}$ suy ra $\widehat{DHB}=\dfrac{1}{2}\widehat{DHC}$.\hfill $(4)$\\
        Cũng do tứ giác $CDOH$ nội tiếp, suy ra $\widehat{COD}=\widehat{DHC}$.\hfill $(5)$\\
        Từ $(4)(5)$ suy ra $\widehat{CAD}=\dfrac{1}{2}\widehat{COD}=\dfrac{1}{2}\widehat{CHD}=\widehat{DHB}$. Mặt khác $\widehat{DHB}+\widehat{DHA}=180^\circ$ và $\widehat{CBD}+\widehat{CAD}=180^\circ$ nên $\widehat{CBD}=\widehat{AHD}$.\hfill $(6)$\\
       	Mà theo bài ra, tứ giác $ACBD$ nội tiếp nên $\widehat{HAD}=\widehat{BCD}.$\hfill $(7)$\\
       	Từ $(6)(7)$ ta có $\triangle BCD\backsim \triangle HAD$ (g-g), suy ra $\widehat{ADH}=IDB$.
    \end{enumerate}
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Thái Bình, 2016]%[9D5G1]
    Cho các số thực $x\ge 1$, $y\ge 1$, $z\ge 1$ và thỏa mãn $3x^2+4y^2+5z^2=52$. Tìm giá trị nhỏ nhất của biểu thức $F=x+y+z$.
    \loigiai{
    Ta có $5(x^2+y^2+z^2)=52+2x^2+y^2\ge 52+2+1=55$ suy ra $x^2+y^2+z^2\ge 11.$\hfill $(1)$\\
    Ta lại có $(x-1)(y-1)\ge 0\Leftrightarrow xy+1\ge x+y$.\\
    Tương tự ta có $yz+1\ge y+z$; $xz+1\ge x+z$.\\
    Từ đó suy ra 
    $xy+yz+zx+3\ge 2x+2y+2z$
    $\Leftrightarrow 2(xy+yz+zx)+6\ge 4(x+y+z)$\hfill $(2)$\\
    Từ $(1),(2)$ suy ra 
    $$(x+y+z)^2+6\ge 4(x+y+z)+11$$
    $$\Leftrightarrow (x+y+z)^2-4(x+y+z)-5\ge 0$$
    Giải ra ta được $F=x+y+z\ge 5$, dấu bằng xảy ra khi và chỉ khi $x=y=1, z=3$.
    }
\end{ex}
