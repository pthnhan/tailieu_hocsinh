	Rút gọn biểu thức $\sqrt{6 + \sqrt{8} + \sqrt{12} + \sqrt{24}}$.
	\loigiai{
		Ta có $6 + \sqrt{8} + \sqrt{12} + \sqrt{24} = 3 + 2 + 1 + 2\sqrt{6} + 2\sqrt{3} + 2\sqrt{2} = \left(\sqrt{3} + \sqrt{2} + 1\right)^2$.\\
		Vậy $\sqrt{6 + \sqrt{8} + \sqrt{12} + \sqrt{24}} = \sqrt{\left(\sqrt{3} + \sqrt{2} + 1\right)^2} = \sqrt{3} + \sqrt{2} + 1$.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Lâm Đồng, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-9G-18]%[9H0K3]
	Cho hình bình hành $ABCD$ có $\widehat{B} = 120^\circ$, $AB = 4\mathrm{cm}$, $BC = 5\mathrm{cm}$. Tính độ dài đường chéo $BD$.
	\loigiai{
		\immini{
			$ABCD$ là hình bình hành có $\widehat{B} = 120^\circ$ $\Rightarrow \widehat{A} = 60^\circ$.\\
			Kẻ $DH \perp AB$ tại $H$. Khi đó tam giác $AHD$ vuông tại $H$ có $\widehat{A} = 60^\circ$ nên ta có\\
			$DH = AD\sin 60^\circ = \dfrac{5\sqrt{3}}{2}$ và $AH = AD\cos 60^\circ = \dfrac{5}{2} \Rightarrow BH = AB - AH = \dfrac{3}{2}$.\\
			Tam giác $BHD$ vuông tại $H$ nên\\ $BD^2 = HB^2 + HD^2 = \dfrac{9}{4} + \dfrac{75}{4} = 21$.\\
			Vậy $BD = \sqrt{21}\mathrm{cm}$.
		}{
			\begin{tikzpicture}
				\clip (-1,-1) rectangle (7,5);
				\tkzDefPoints{0/0/A, 4/0/B, 5/0/E}
				\tkzDefPointBy[rotation = center A angle 60](E) \tkzGetPoint{D}
				\tkzDefPointBy[translation = from A to B](D) \tkzGetPoint{C}
				\tkzDrawAltitude(A,B)(D) \tkzGetPoint{H}
				\tkzDrawPolygon(A,B,C,D)
				\tkzDrawSegments(B,D)  
				\tkzMarkRightAngle(B,H,D) 
				\tkzMarkAngles[size=0.7,fill=red,opacity=0.4](B,A,D)  \tkzLabelAngle[pos=0.45,rotate=40](B,A,D){\footnotesize $60^\circ$}
				\tkzDrawPoints(A,B,C,D,H)
				\tkzLabelPoints[above left](D)
				\tkzLabelPoints[above right](C)
				\tkzLabelPoints[below left](A)
				\tkzLabelPoints[below right](B)
				\tkzLabelPoints[below](H)
			\end{tikzpicture}
		}
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Lâm Đồng, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-9G-18]%[9D0B2]
	Giải phương trình $\sqrt{2x + 3} - \sqrt{x + 1} = 1$.
	\loigiai{
		Điều kiện: $x \geq -1$.\\
		Ta có $\sqrt{2x + 3} - \sqrt{x + 1} = 1 \Leftrightarrow \sqrt{2x + 3} = 1 + 	\sqrt{x + 1}$\\
		\phantom{Ta có $\sqrt{2x + 3} - \sqrt{x + 1} = 1$} $\Leftrightarrow 2x + 3 = 1 + 2\sqrt{x + 1} + x + 1$\\
		\phantom{Ta có $\sqrt{2x + 3} - \sqrt{x + 1} = 1$} $\Leftrightarrow x + 1 = 2\sqrt{x + 1}$\\
		\phantom{Ta có $\sqrt{2x + 3} - \sqrt{x + 1} = 1$} $\Leftrightarrow \sqrt{x + 1}\left(\sqrt{x + 1} - 2\right) = 0$\\
		\phantom{Ta có $\sqrt{2x + 3} - \sqrt{x + 1} = 1$} $\Leftrightarrow \hoac{&\sqrt{x + 1} = 0 \\ &\sqrt{x + 1} = 2}$\\
		\phantom{Ta có $\sqrt{2x + 3} - \sqrt{x + 1} = 1$} $\Leftrightarrow \hoac{&x = -1&&\text{(thỏa mãn)} \\ &x = 3&&\text{(thỏa mãn)}.}$\\
		Vậy phương trình đã cho có tập nghiệm $S = \left\lbrace -1; 3\right\rbrace$.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Lâm Đồng, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-9G-18]%[9D0B4]
	Giả sử $p$ là số nguyên tố không nhỏ hơn $5$. Chứng minh $(p^2 - 1)\,\vdots\,24$.
	\loigiai{
		Vì $p$ là số nguyên tố không nhỏ hơn $5$ nên $p$ là một số lẻ không nhỏ hơn $5$.\\
		Suy ra $p = 2n + 1$ với $n \geq 2$, $n \in \mathbb{N}$.\\
		Ta có $p^2 - 1 = (2n + 1)^2 - 1 = 4n^2 + 4n = 4n(n + 1)\,\vdots\,24$.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Lâm Đồng, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-9G-18]%[9D0K7]
	Cho $a > 0$, $b > 0$. Chứng minh $\sqrt{a}\left(\sqrt{\dfrac{a}{b}} - 1\right) \geq \sqrt{b}\left(1 - \sqrt{\dfrac{b}{a}}\right)$.
	\loigiai{
		Ta có $\sqrt{a}\left(\sqrt{\dfrac{a}{b}} - 1\right) - \sqrt{b}\left(1 - \sqrt{\dfrac{b}{a}}\right) = \dfrac{\sqrt{a}}{\sqrt{b}}\left(\sqrt{a} - \sqrt{b}\right) - \dfrac{\sqrt{b}}{\sqrt{a}}\left(\sqrt{a} - \sqrt{b}\right)$\\
		\phantom{Ta có $\sqrt{a}\left(\sqrt{\dfrac{a}{b}} - 1\right) - \sqrt{b}\left(1 - \sqrt{\dfrac{b}{a}}\right)$} $= \left(\sqrt{a} - \sqrt{b}\right)\left(\dfrac{\sqrt{a}}{\sqrt{b}} - \dfrac{\sqrt{b}}{\sqrt{a}}\right)$\\
		\phantom{Ta có $\sqrt{a}\left(\sqrt{\dfrac{a}{b}} - 1\right) - \sqrt{b}\left(1 - \sqrt{\dfrac{b}{a}}\right)$} $= \left(\sqrt{a} - \sqrt{b}\right)\cdot\dfrac{a - b}{\sqrt{ab}}$\\
		\phantom{Ta có $\sqrt{a}\left(\sqrt{\dfrac{a}{b}} - 1\right) - \sqrt{b}\left(1 - \sqrt{\dfrac{b}{a}}\right)$} $= \dfrac{\left(\sqrt{a} - \sqrt{b}\right)^2\left(\sqrt{a} + \sqrt{b}\right)}{\sqrt{ab}} \geq 0$.\\
		Đẳng thức xảy ra khi và chỉ khi $a = b > 0$.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Lâm Đồng, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-9G-18]%[8H0B3]
	Cho hình thang $ABCD$ $(AD \parallel BC)$. Gọi $M$, $N$ lần lượt là trung điểm của $BC$, $AD$. Một đường thẳng $d$ song song với hai đáy cắt $AB$, $MN$, $CD$ theo thứ tự tại $E$, $O$, $F$. Chứng minh $O$ là trung điểm của $EF$.

			Gọi $S$ là điểm đồng quy của ba đường thẳng $AB$, $MN$, $CD$. Ta có
			$$\dfrac{OE}{NA} = \dfrac{SO}{SN} = \dfrac{OF}{ND}.$$

				\tkzDefMidPoint(A,D)\tkzGetPoint{N}
				\tkzDefMidPoint(B,C)\tkzGetPoint{M}
				\tkzInterLL(M,N)(E,F)\tkzGetPoint{O}			
				\tkzDrawSegments(S,A S,D S,N B,C E,F A,D)			
				\tkzDrawPoints(S,A,B,C,D,M,N,E,F,O)

\begin{ex}%[Đề thi HSG 9, tỉnh Lâm Đồng, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-9G-18]%[9D0K7]
	Cho $x$, $y$, $z$ là các số dương thỏa $(4x^2 + 1)(9y^2 + 1)(16z^2 + 1) = 192xyz$. Tính giá trị $A = \left(\dfrac{x - y + z}{x + y - z}\right)^2$.
	\loigiai{
		Ta có $4x^2 + 1 \geq 4x$, $9y^2 + 1 \geq 6y$, $16z^2 + 1 \geq 8z$.\\
		Suy ra $(4x^2 + 1)(9y^2 + 1)(16z^2 + 1) \geq 4x \cdot 6y \cdot 8z = 192xyz$.\\
		Đẳng thức xảy ra khi và chỉ khi $x = \dfrac{1}{2}$, $y = \dfrac{1}{3}$, $z = \dfrac{1}{4}$.\\
		Vậy $A = \left(\dfrac{x - y + z}{x + y - z}\right)^2 = \left(\dfrac{\dfrac{1}{2} - \dfrac{1}{3} + \dfrac{1}{4}}{\dfrac{1}{2} + \dfrac{1}{3} - \dfrac{1}{4}}\right)^2 = \dfrac{25}{49}$.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Lâm Đồng, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-9G-18]%[9D0G2]
	Giải hệ phương trình $\heva{&2x^2 - y^2 = 1 \\ &xy + x^2 = 2.}$
	\loigiai{
		Dễ thấy $y = 0$ không thỏa mãn hệ phương trình đã cho.\\
		Xét $y \neq 0$, đặt $y = kx$, thay vào hệ phương trình đã cho, ta được $\heva{&2x^2 - k^2x^2 = 1 \\ &kx^2 + x^2 = 2.}$\\
		$\Rightarrow \dfrac{2 - k^2}{k + 1} = \dfrac{1}{2} \Leftrightarrow 2k^2 + k - 3 = 0 \Leftrightarrow \hoac{&k = 1 \\ &k = -\dfrac{3}{2}.}$\\
		Với $k = 1$, ta có $y = x$ thay vào phương trình thứ nhất của hệ phương trình đã cho, ta được:
		$$2x^2 - x^2 = 1 \Leftrightarrow \hoac{&x = 1 \Rightarrow y = 1 \\ &x = -1 \Rightarrow y = -1.}$$
		Với $k = -\dfrac{3}{2}$, ta có $y = -\dfrac{3}{2}x$ thay vào phương trình thứ nhất của hệ phương trình đã cho, ta được:
		$$2x^2 - \dfrac{9}{4}x^2 = 1 \Leftrightarrow x^2 = -4 \quad\text{(vô nghiệm)}.$$
		Vậy hệ phương trình đã cho có tập nghiệm $S = \left\lbrace (1; 1), (-1; -1)\right\rbrace$.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Lâm Đồng, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-9G-18]%[9H0K1]
	Cho tam giác $ABC$ nội tiếp đường tròn $(O)$. Phân giác góc $A$ cắt đường tròn $(O)$ tại $D$. Chứng minh $AD > \dfrac{AB + AC}{2}$.
	\loigiai{
		\immini{
			Gọi $H$, $K$ lần lượt là hình chiếu vuông góc của $D$ trên $AB$, $AC$. Khi đó ta có $DH = DK$.\\
			Hai tam giác $DHB$ vuông tại $H$ và $DKC$ vuông tại $K$ có $DB = DC$.\\
			$\Rightarrow \triangle DHB = \triangle DKC \Rightarrow BH = CK$.\\
			$\Rightarrow AB + AC = AH + AK$.\\
			Mà $AH < AD$ và $AK < AD$ nên $AD > \dfrac{AB + AC}{2}$.
		}{
			\begin{tikzpicture}[scale=.7]
				\clip (-4,-4) rectangle (5,4);
				\tkzDefPoints{0/0/O, 3/0/C}
				\tkzDefPointBy[rotation = center O angle 95](C) \tkzGetPoint{A}
				\tkzDefPointBy[rotation = center O angle -115](C) \tkzGetPoint{B}
				\tkzDrawBisector(B,A,C) \tkzGetPoint{E}
				\tkzInterLC(A,E)(O,C) \tkzGetPoints{A}{D}
				\tkzDrawAltitude(A,B)(D) \tkzGetPoint{H}
				\tkzDrawAltitude(A,C)(D) \tkzGetPoint{K}
				\tkzDrawCircle[radius](O,C) 						
				\tkzDrawSegments(A,B A,K A,D B,C D,H D,K D,B D,C)			
				\tkzDrawPoints(O,A,B,C,D,H,K)
				\tkzLabelPoints[above left](H)
				\tkzLabelPoints[above right](C)
				\tkzLabelPoints[below left](B)
				\tkzLabelPoints[below right](K)
				\tkzLabelPoints[below](O,D)
				\tkzLabelPoints[above](A)
				\tkzMarkRightAngles(D,H,B D,K,C) 
			\end{tikzpicture}
		}
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Lâm Đồng, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-9G-18]%[9D0K4]
	Tìm số nguyên tố $p$ biết phương trình $x^2 + px -12p = 0$ có hai nghiệm đều là số nguyên.
	\loigiai{
		Ta có $\Delta = p^2 - 4\cdot(-12p) = p^2 + 48p$.\\
		Để phương trình đã cho có nghiệm nguyên thì trước hết $\Delta$ phải là số chính phương.\\
		Vậy $\Delta = p^2 + 48p = p(p + 48)$ là số chính phương $\Rightarrow p + 48$ chia hết cho $p$.\\
		Vì $p$ là số nguyên tố nên ta có $p = 2$ hoặc $p = 3$.\\
		Thử lại ta thấy $p = 2$ thỏa mãn.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Lâm Đồng, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-9G-18]%[9D0K7]
	Cho $a$, $b$ là các số thỏa mãn điều kiện $a^2 + b^2 = 1$. Tìm giá trị lớn nhất của $M = ab\sqrt{3} + a^2$.
	\loigiai{
		Ta có $ab\sqrt{3} \leq \dfrac{a^2 + \left(b\sqrt{3}\right)^2}{2} = \dfrac{a^2 + 3b^2}{2}$.\\
		$\Rightarrow M \leq \dfrac{a^2 + 3b^2}{2} + a^2 = \dfrac{3(a^2 + b^2)}{2} = \dfrac{3}{2}$.\\
		Đẳng thức xảy ra $\Leftrightarrow \heva{&a^2 + b^2 = 1 \\ &a = b\sqrt{3}} \Leftrightarrow \heva{&a = \dfrac{\sqrt{3}}{2} \\ &b = \dfrac{1}{2}}$ hoặc $\heva{&a = -\dfrac{\sqrt{3}}{2} \\ &b = -\dfrac{1}{2}.}$\\
		Vậy giá trị lớn nhất của $M$ là $\dfrac{3}{2}$, đạt được khi $a = \dfrac{\sqrt{3}}{2}$, $b = \dfrac{1}{2}$ hoặc $a = -\dfrac{\sqrt{3}}{2}$, $b = -\dfrac{1}{2}$.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Lâm Đồng, năm 2018]%[Đinh Thanh Hoàng, Dự án EX-9G-18]%[9H0K6]
	Cho góc vuông $xAy$, điểm $B$ cố định trên $Ax$, điểm $C$ di động trên $Ay$. Đường tròn tâm $O$ nội tiếp tam giác $ABC$ tiếp xúc với $AC$, $BC$ theo thứ tự tại $E$ và $F$. Chứng minh đường thẳng $EF$ luôn đi qua một điểm cố định.
	\loigiai{
		\immini{
			Gọi $H$ là giao điểm của $EF$ và $OA$.\\
			Tam giác $OAB$ có $\widehat{AOB} = 180^\circ - \dfrac{\widehat{ABC} + \widehat{BAC}}{2} = 90^\circ + \dfrac{\widehat{ACB}}{2}$.\\
			Tứ giác $OECF$ có $\widehat{OEC} + \widehat{OFC} = 90^\circ + 90^\circ = 180^\circ$.\\
			$\Rightarrow$ tứ giác $OECF$ nội tiếp $\Rightarrow \widehat{OFE} = \widehat{OCE}  =\dfrac{\widehat{BCA}}{2}$.\\
			$\Rightarrow \widehat{BEH} = \widehat{BFO} + \widehat{OFE} = 90^\circ + \dfrac{\widehat{ACB}}{2}$.			
		}{
			\begin{tikzpicture}[scale=1]
				\clip (-1,-1) rectangle (7.5,5.5);
				\tkzDefPoints{0/0/A, 0/4/B, 6/0/C, 7/0/D, 0/5/G}
				\tkzDefCircle[in](A,B,C) \tkzGetPoint{O}				
				\tkzDrawAltitude(A,C)(O) \tkzGetPoint{E}
				\tkzDrawAltitude(B,C)(O) \tkzGetPoint{F}
				\tkzInterLL(A,O)(E,F) \tkzGetPoint{H}				
				\tkzDrawCircle[in](A,B,C)					
				\tkzDrawSegments(A,G A,D B,C E,F A,H O,E O,F B,H O,B O,C)			
				\tkzDrawPoints(A,B,C,O,H,E,F)				
				\tkzLabelPoints[below left](A)
				\tkzLabelPoints[below right](H)
				\tkzLabelPoints[below](E,C)
				\tkzLabelPoints[above](F)
				\tkzLabelPoints[left](B,O)
				\draw (0,5) node[right]{$y$};
				\draw (7,0) node[above]{$x$};			
				\tkzMarkRightAngles(C,F,O B,A,C C,E,O B,H,O) 
			\end{tikzpicture}
		}
		Do đó, ta có $\widehat{AOB} = \widehat{BEH}$ hay tứ giác $BOHE$ nội tiếp.\\
		$\Rightarrow \widehat{BHA} = \widehat{BFO} = 90^\circ$.\\
		Mà $B$ và $OA$ cố định nên $H$ cố định.\\
		Vậy đường thẳng $EF$ luôn đi qua điểm $H$ cố định.
	}
\end{ex}
