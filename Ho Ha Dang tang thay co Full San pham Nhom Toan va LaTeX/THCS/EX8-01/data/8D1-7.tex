\section{PHÂN TÍCH ĐA THỨC THÀNH NHÂN TỬ BẰNG CÁCH PHỐI HỢP NHIỀU PHƯƠNG PHÁP}

\subsection{Tóm tắt lí thuyết}
Khi phân tích đa thức thành nhân tử, nếu cần ta phải phối hợp nhiều phương pháp để phân tích được triệt để. Các phương pháp thông thường hay sử dụng:
\begin{enumerate}[\tickEX]
\item Phương pháp ưu tiên số một là đặt nhân tử chung;
\item Phương pháp ưu tiên số hai là dùng hằng đẳng thức;
\item Cuối cùng là nhóm các hạng tử. Mục đích của việc nhóm các hạng tử nhằm làm cho quá trình phân tích đa thức thành nhân tử được tiếp tục bằng cách đặt nhân tử chung hoặc dùng hằng đẳng thức.
\end{enumerate}
Ngoài ra, ta còn có thể sử dụng các phương pháp khác (nâng cao) như sau.
\begin{enumerate}[\tickEX]
\item Phương pháp tách một hạng tử thành nhiều hạng tử;
\item Phương pháp thêm và bớt cùng một hạng tử;
\item Phương pháp đổi biến.
\end{enumerate}

\subsection{Các dạng toán}
\begin{dang}[Phối hợp các phương pháp thông thường]
\begin{enumerate}[\tickEX]
\item Một số bài toán, nếu chỉ áp dụng một phương pháp thì ta không thể phân tích thành nhân tử được vì vậy ta phải kết hợp hai hoặc cả ba phương pháp đã nêu.
\item Khi phối phợp nhiều phương pháp, thông thường phương pháp đặt nhân tử chung được ưu tiên đầu tiên rồi đến nhóm hạng tử và hằng đẳng thức, một phương pháp có thể dùng nhiều lần.
\end{enumerate}
\end{dang}

\begin{vd}
Phân tích các đa thức sau thành nhân tử.
\begin{enumEX}{3}
\item $16x^3-54y^3$;
\item $5x^2-5y^2$;
\item $2x^4-32$.
\end{enumEX}
\loigiai{
\begin{enumerate}
\item $16x^3 - 54y^3 = 2 \left(8x^3 - 27y^3 \right) = 2(2x-3y) \left( 4x^2 + 6xy + 9y^2 \right)$.
\item $5x^2 - 5y^2 = 5 (x^2 - y^2 ) = 5 (x - y) (x + y)$.
\item $2x^4 - 32 = 2\left( x^4 - 16 \right) = 2 \left( x^2 - 4 \right) \left( x^2 + 4\right) = 2 ( x -2 )(x + 2) \left( x^2 + 4 \right)$.
\end{enumerate}
}
\end{vd}

\begin{vd}
Phân tích đa thức sau thành nhân tử.
\begin{enumEX}{2}
\item $3x^3-75x$;
\item $5x^2y-30xy^2+45y^3$.
\end{enumEX}
\loigiai{
\begin{enumerate}
\item $3x^3-75x=3x\left( x^2-25\right)= 3x(x-5)(x+5)$.
\item $5x^2y-30xy^2+45y^3=5y\left( x^2-6xy+9y^2\right) =5y(x-3y)^2$.
\end{enumerate}
}
\end{vd}

\begin{vd}
Phân tích đa thức $A= a^6 - a^4 + 2a^3 +2a^2$ thành nhân tử.
\loigiai
{
Ta có 
{\allowdisplaybreaks
\begin{eqnarray*}
A&=&a^6 - a^4 + 2a^3 +2a^2 = a^2 \left( a^4 - a^2 + 2a +2 \right)= a^2 \left[ a^2 \left(a^2 -1 \right) + 2(a + 1) \right]\\
&=& a^2 \left[ a^2 (a - 1) (a + 1) + 2(a + 1) \right]= a^2 (a - 1)(a + 1) \left(a^2 + 2 \right).
\end{eqnarray*}
}
}
\end{vd}

\begin{vd}
Phân tích đa thức $B=2x^3y - 2xy3 - 4xy^2 - 4x^2y$ thành nhân tử.
\loigiai{
Ta có 
{\allowdisplaybreaks
\begin{eqnarray*}
B&=&2x^3y - 2xy3 - 4xy^2 - 2xy = 2xy \left( x^2 - y^2 -2y -2x \right)\\
&=& 2xy \left[ (x - y)(x + y) - 2(x + y) \right]= 2xy (x + y)(x - y -2).
\end{eqnarray*}
}
}
\end{vd}

\begin{vd}
Phân tích đa thức sau thành nhân tử
\[M=(5x-10)\left( x^2-1\right) - (3x-6) \left( x^2-2x+1\right) .\]
\loigiai{
{\allowdisplaybreaks
\begin{eqnarray*}
M &=& (5x-10)\left( x^2-1\right) - (3x-6) \left( x^2-2x+1\right)\\
&=& 5(x-2)(x-1)(x+1) - 3(x-2) (x-1)^2\\
&=& (x-2)(x-1)\left[ 5(x+1)-3(x-1)\right]\\
&=& (x-2)(x-1)(2x+8)\\
&=& 2(x-2)(x-1)(x+4).
\end{eqnarray*}
}
}
\end{vd}

\begin{dang}[Phương pháp tách một hạng tử thành nhiều hạng tử]
\begin{enumerate}[\tickEX]
\item Tách các hạng tử của đa thức thành tổng hoặc hiệu của nhiều hạng tử, từ đó ta ghép cặp để được các nhóm hạng tử giống nhau và làm xuất hiện nhân tử chung.
\item Cách tổng quát để phân tích đa thức bậc hai $ax^2+bx+c$ thành nhân tử là 
\begin{itemize}
\item Tách $bx$ thành $b_1x+b_2x$ sao cho $b_1\cdot b_2=ac$.
\item Đặt nhân tử chung theo từng nhóm.
\end{itemize}
\item Đối với đa thức bậc ba trở lên thì tùy theo đặc điểm của các hệ số mà có cách tách riêng cho phù hợp. Một thủ thuật của loại này là dùng máy tính cầm tay nhẩm một nghiệm (thường là nghiệm nguyên, giả sử là $x_0$), khi đó ta tìm cách ghép cặp làm sao cho xuất hiện nhân tử $(x-x_0)$ là được.
\end{enumerate}
\end{dang}

\begin{vd}
Phân tích đa thức sau thành nhân tử
\[A=4x^2-8x+3.\]
\loigiai{
\begin{phantich}
Ta sẽ tách hạng tử $-8x$ thành $b_1x+b_2x$ sao cho $\heva{&b_1+b_2=-8\\ & b_1\cdot b_2=4\cdot 3=12.}$ \\
Ta chọn $b_1=-2$ và $b_2=-6$, nghĩa là $-8x=-6x-2x$.
\end{phantich}
Ta có 
\[A=4x^2-8x+3= 4x^2-2x-6x+3= 2x(2x-1)-3(2x-1)=(2x-1)(2x-3).\]
Vậy $A=(2x-1)(2x-3)$.
}
\end{vd}

\begin{vd}
Phân tích các đa thức sau thành nhân tử.
\begin{enumEX}{2}
\item $x^2+x-6$;
\item $x^2 - 2xy -8y^2$;
\item $2x + 12x^2 - 2 $;
\item $15y^2 - 18x^2 + 39xy $.
\end{enumEX}
\loigiai
{
\begin{enumerate}
\item
$x^2+x-6 = x^2+3x-2x-6 = x(x+3) - 2(x+3) = (x+3)(x-2)$.
\item $x^2 - 2xy -8y^2 = x^2 - 4xy +2xy -8y^2 = x(x-4y) + 2y(x-4y) = (x-4y)(x+2y)$.
\item  $2x + 12x^2 - 2 = 12x^2 + 6x - 4x - 2 = 6x(2x +1) - 2(2x+1) = 2(2x+1)(3x-1)$.
\item Ta có 
{\allowdisplaybreaks
\begin{eqnarray*}
15y^2 - 18x^2 + 39xy &=& -18x^2 + 45xy - 6xy + 15y^2 = 9x(-2x+5y) + 3y(-2x + 5y) \\
&=& 3(5y-2x)(3x+y).
\end{eqnarray*}}
\end{enumerate}
}
\end{vd}

\begin{vd}
Phân tích các đa thức sau thành nhân tử.
\begin{enumEX}{2}
\item $x^3-4x^2+x+6$;
\item $8x^3 - 14x^2 - 5x +2 $.
\end{enumEX}
\loigiai{
{\allowdisplaybreaks
\begin{enumerate}
\item Ta có 
\begin{eqnarray*}
x^3-4x^2+x+6 &=& x^3 + x^2 -5x^2 -5x + 6x +6 = x^2(x+1) -5x(x+1) + 6(x+1) \\
&=& (x+1)(x^2-5x+6) = (x+1)(x^2 - 3x - 2x +6) \\
&=& (x+1)\left [x(x-3) -2(x-3) \right ] = (x+1)(x-3)(x-2).
\end{eqnarray*}
\item Ta có
\begin{eqnarray*}
8x^3 - 14x^2 - 5x +2  &=& 8x^3 - 16x^2 + 2x^2 - 4x - x + 2 \\
&=& 8x^2(x-2) + 2x(x-2) - (x-2) \\
&=& (x-2)(8x^2 + 2x - 1) = (x-2)(8x^2 + 4x - 2x -1) \\
&=& (x-2)[4x(2x+1) - (2x+1)] = (x-2)(2x+1)(4x-1).
\end{eqnarray*}
\end{enumerate}
}}
\end{vd}

\begin{vd}[Đề thi HSG lớp 8, Lục Nam - Bắc Giang năm 2017]
Phân tích đa thức $x^2+6xy+5y^2-5y-x$ thành nhân tử.

\loigiai{
\begin{phantich}
Nhẩm các hệ số của biểu thức $x^2+6xy+5y^2$, ta thấy tổng các hệ số bằng nhau. Từ đó ta nghĩ đến việc tách hạng tử $6xy$ để được nhân tử chung.
\end{phantich}
Tách hạng tử $6xy$ thành $xy+5xy$, ta được
{\allowdisplaybreaks
\begin{eqnarray*}
&& x^2+6xy+5y^2-5y-x = x^2+xy+5xy+5y^2-5y-x\\
&=& x(x+y) +5y(x+y)-(x+5y)= (x+y)(x+5y)- (x+5y)=(x+5y)(x+y-1).
\end{eqnarray*}
}
}
\end{vd}

\begin{vd}[KSCL mũi nhọn, phòng GD\&ĐT Thanh Chương - Nghệ An năm 2013]
Phân tích đa thức sau thành nhân tử
\[ x^2-2xy+y^2+4x-4y-5.\]
\loigiai{
\begin{phantich}
Ta nhận thấy ngay biểu thức $x^2-2xy+y^2$ là hằng đẳng thức, kết hợp với phần sau và tách $5$ ra để được hiệu của hai bình phương.
\end{phantich}
{\allowdisplaybreaks
\begin{eqnarray*}
&& x^2-2xy+y^2+4x-4y-5 = (x-y)^2+4(x-y)-5 \\
&=& (x-y)^2+4(x-y)+4-9 = (x-y+2)^2-3^2=(x-y+5)(x-y-1).
\end{eqnarray*}
}
}
\end{vd}

\begin{dang}[Phương pháp thêm bớt cùng một hạng tử]
Khi phân tích đa thức thành nhân tử, đôi khi ta cần tăng thêm các hạng tử của đa thức bằng cách thêm và bớt cùng một hạng tử. Có hai cách thêm bớt thương gặp như sau:
\begin{enumerate}[\tickEX]
\item Thêm và bớt cùng một hạng tử làm xuất hiện hiệu của hai bình phương.
\item Thêm và bớt cùng một hạng tử làm xuất hiện nhân tử chung.
\end{enumerate}
\end{dang}

\begin{vd}[Đề HKI, trường THPT Hà Nội - Amsterdam năm 2014 - 2015]
Phân tích đa thức sau thành nhân tử
\[ A=x^4+4.\]
\loigiai{
\begin{phantich}
Ta nhận thấy các phương pháp thông thường không dùng được. Ta tăng thêm các hạng tử của $A$ bằng cách thêm bớt cùng một hạng tử $4x^2$. Lúc này ta sẽ xuất hiện hiệu của hai bình phương.
\end{phantich}
Thêm và bớt $4x^2$, ta được
{\allowdisplaybreaks
\begin{eqnarray*}
A&=& x^4+4 = x^4+4x^2+4-4x^2 =\left( x^2+2 \right)^2-(2x)^2\\
&=& \left( x^2+2x+2 \right) \left( x^2-2x+2 \right).
\end{eqnarray*}
}
}
\end{vd}

\begin{vd}[Đề HKI, trường THPT Hà Nội - Amsterdam năm 2005 - 2006]
Phân tích đa thức sau thành nhân tử
\[ A=(x-a)^4+4a^4.\]
\loigiai{
Thêm và bớt $4(x-a)^2a^2$, ta được
{\allowdisplaybreaks
\begin{eqnarray*}
A&=& (x-a)^4+4a^4 = (x-a)^4+4(x-a)^2a^2+4a^4-4(x-a)^2a^2 \\
&=& \left[ \left(x-a\right)^2 -a^2\right]^2-[2a(x-a)]^2=\left( x^2-2ax \right)^2-\left( 2ax-2a^2 \right)^2\\
&=& \left( x^2-2ax+2ax-2a^2 \right) \left( x^2-2ax-2ax+2a^2 \right) = \left( x^2-2a^2 \right) \left( x^2-4ax+2a^2 \right).
\end{eqnarray*}
}
}
\end{vd}

\begin{vd}
Phân tích đa thức sau thành nhân tử
\[ N=x^4+2x^3+6x-9.\]
\loigiai{
\begin{phantich}
Ta nhận thấy, nếu thêm bớt $x^2$ và ghép cặp thích hợp thì ta sẽ có nhân tử chung. Các bài tập loại này thường nâng cao, cần phải nhẩm nhanh để biết được thêm bớt hạng tử nào sẽ thích hợp.
\end{phantich}
Thêm và bớt $x^2$, ta được
{\allowdisplaybreaks
\begin{eqnarray*}
N&=& x^4+2x^3+x^2-x^2+6x-9 = \left( x^2+x \right)^2-\left( x^2-6x+9 \right)\\
&=& \left( x^2+x \right)^2-\left( x-3 \right)^2=\left( x^2+x+x-3 \right) \left( x^2+x-x+3 \right) =\left( x^2+2x-3 \right) \left( x^2+3 \right)\\
&=& \left( x^2-x+3x-3 \right) \left( x^2+3 \right)=\left[ x(x-1)+3(x-1) \right] \left( x^2+3 \right)=(x-1)(x+3)\left( x^2+3 \right).
\end{eqnarray*}
}
}
\end{vd}

\begin{vd}
Phân tích đa thức $D=x^5+x-1$ thành nhân tử.

\loigiai{
Thêm và bớt $x^2$, ta được
{\allowdisplaybreaks
\begin{eqnarray*}
D&=& x^5+x-1 = x^5+x^2-x^2+x-1=x^2\left( x^3+1 \right) -\left( x^2-x+1 \right)\\
&=& x^2(x+1)\left( x^2-x+1 \right) -\left( x^2-x+1 \right)=\left( x^2-x+1 \right)\left[ x^2(x+1)-1 \right]\\
&=& \left( x^2-x+1 \right)\left(x^3+x^2-1 \right).
\end{eqnarray*}
}
}
\end{vd}

\begin{dang}[Phương pháp đổi biến]
\begin{enumerate}[\tickEX]
\item Khi gặp một đa thức phức tạp, ta nên dùng cách đặt ẩn phụ (thay một đa thức của biến cũ bằng một biến mới để được một đa thức đơn giản hơn, dễ phân tích hơn).
\item Sau khi phân tích với biến mới, ta thay trở lại biến cũ để phân tích tiếp (nếu được).
\end{enumerate}
\end{dang}

\begin{vd}[Đề HKI, trường THPT Hà Nội - Amsterdam năm 2007 - 2008]
Phân tích đa thức sau thành nhân tử
\[C=\left( x^2+x-5\right)\left(x^2+x-7\right) +1.\]
\loigiai{
\begin{phantich}
Ta thấy hai biểu thức $\left( x^2+x-5\right)$ và $\left(x^2+x-7\right)$ gần giống nhau. Ta sẽ đặt $y$ là trung bình cộng của cả hai biểu thức đó, khi đó $C$ sẽ trở nên đơn giản hơn.
\end{phantich}
Đặt $y=\dfrac{\left( x^2+x-5\right)+ \left(x^2+x-7\right)}{2}=x^2+x-6$. Khi đó
\[C=\left( x^2+x-5\right) \left(x^2+x-7\right) +1=(y+1)(y-1)+1=y^2-1+1=y^2.\]
Suy ra $C=\left(x^2+x-6 \right)^2=\left(x^2-2x+3x-6 \right)^2=\left[ x(x-2)+3(x-2) \right]^2=(x-2)^2(x+3)^2$.
}
\end{vd}

\begin{vd}
Phân tích đa thức sau thành nhân tử
\[M=x^2-2xy+y^2+3x-3y-4.\]
\loigiai{
Ta có
\[M=x^2-2xy+y^2+3x-3y-4=(x-y)^2+3(x-y)-4.\]
Đặt $a=x-y$. Khi đó
\[M=a^2+3a-4=a^2-a+4a-4=a(a-1)+4(a-1)=(a-1)(a+4).\]
Vậy $M=(x-y-1)(x-y+4)$.
}
\end{vd}

\begin{vd}[Đề HKI, trường THPT Hà Nội - Amsterdam năm 2007 - 2008]
Phân tích đa thức sau thành nhân tử
\[D=(a+b+1)^2+(a+b-1)^2-4(a+b)^2.\]
\loigiai{
\begin{phantich}
Ta thấy biểu thức $(a+b)$ là biểu thức chính của $D$. Ta sẽ đặt $x=a+b$, khi đó $D$ sẽ trở nên đơn giản hơn.
\end{phantich}
Đặt $x=a+b$. Khi đó
\[D=(x+1)^2+(x-1)^2-4x^2=-2x^2+2=-2\left( x^2-1\right) =-2(x-1)(x+1).\]
Vậy $D=-2(a+b-1)(a+b+1)$.
}
\end{vd}

\begin{vd}[Đề HKI, trường THPT Hà Nội - Amsterdam năm 2005 - 2006]
Phân tích đa thức sau thành nhân tử
\[A=(x+1)(x+2)(x+3)(x+4)-120.\]
\loigiai{
\begin{phantich}
Đây là dạng bài khá quen thuộc. Ta sẽ ghép nhóm các thừa số sao cho tổng hệ số tự do của hai nhóm là bằng nhau, từ đó sẽ xuất hiện ẩn phụ.
\end{phantich}
Ta có 
{\allowdisplaybreaks
\begin{eqnarray*}
A&=&(x+1)(x+2)(x+3)(x+4)-120=[(x+1)(x+4)]\cdot [(x+2)(x+3)]-120\\
& =&\left( x^2+5x+4\right) \left( x^2+5x+6\right) -120.
\end{eqnarray*}
}
Đặt $y=\dfrac{\left( x^2+5x+4 \right) +\left(x^2+5x+6\right) }{2}=x^2+5x+5$, ta được
\[A=(y-1)(y+1)-120=y^2-1-120=y^2-121=(y-11)(y+11).\]
Suy ra 
{\allowdisplaybreaks
\begin{eqnarray*}
A&=&\left( x^2+5x+5-11 \right) \left( x^2+5x+5+11 \right) =\left( x^2+5x-6 \right) \left( x^2+5x+16 \right)\\
&=&\left( x^2+6x-x-6\right) \left( x^2+5x+16 \right)=\left[ x(x+6) - (x+6)\right] \left( x^2+5x+16 \right)\\
&=& (x+6)(x-1)\left( x^2+5x+16 \right).
\end{eqnarray*}
}
Vậy $A=(x+6)(x-1)\left( x^2+5x+16 \right)$.
}
\end{vd}

\begin{dang}[Tìm $x$ thỏa một đẳng thức cho trước]
Một tích bằng $0$ khi một trong các nhân tử của nó bằng $0$. Ta thực hiện theo các bước sau:
\begin{enumerate}[\tickEX]
\item Chuyển tất cả sang vế trái để vế phải bằng $0$.
\item Phân tích đa thức thành nhân tử để đưa về dạng tích.
\item Cho một trong các nhân tử bằng $0$ và tìm $x$.
\end{enumerate}
\begin{note}
Một số bài toán ta cần đặt ẩn phụ trước rồi mới giải.
\end{note}
\end{dang}
\begin{vd}
Tìm $x$, biết 
$x^2  -  5x +4  = 0$.
\loigiai{
Ta có
{\allowdisplaybreaks
\begin{eqnarray*}
x^2  -  5x +4  = 0 &\Rightarrow & \left( x^2  -  x \right)  -  (4x + 4)  =  0\\
