\section{SỐ PHỨC}
	\subsection{Tóm tắt lí thuyết}
	\subsubsection{Định nghĩa số phức}
	\begin{dn}
		Một \textbf{số phức} là một biểu thức dạng $ a+bi $, trong đó $ a $ và $ b $ là những số thực và số $ i $ thỏa mãn $\boxed{ i^2=-1} $. Kí hiệu số phức đó là $ z $ và viết $ z=a+bi $.\\
		$ i $ được gọi là \textit{đơn vị ảo}, $ a $ được gọi là \textit{phần thực}, $ b $ được gọi là \textit{phần ảo} của số phức $ z=a+bi $.\\
		Tập hợp các số phức được kí hiệu là $ \mathbb C $.
	\end{dn}
	\begin{vd}
		Các số sau là những số phức:\\
		$3-5i$ ; $ -\sqrt 3+5i$ ; $ 2+\left(-4\right)i$.
	\end{vd}
	\subsubsection{Số phức bằng nhau}
	\begin{dn}
		Hai số phức được gọi là \textbf{bằng nhau} nếu phần thực và phần ảo của chúng tương ứng bằng nhau. 
		$$\boxed{a+bi=c+di \Leftrightarrow a=c \text{ và }b=d.}$$
	\end{dn}
	\begin{vd}
		Tìm các số thực $ x,y $, biết $$\left(3x-y\right)+\left(2y-1\right)i=\left(x+1\right)+\left(y+2\right)i$$
	\end{vd}
	\loigiai{Từ định nghĩa hai số phức bằng nhau, ta có $$ \begin{cases}
		3x-y=x+1\\ 2y-1=y+2
		\end{cases} \Leftrightarrow \begin{cases}
		x=2\\y=3
		\end{cases}$$
		Vậy $ x=2 $ và $ y=3 $.}
	\begin{note}
		\begin{enumerate}
			\item Mỗi số thực $ a $ được gọi là một số phức với phần ảo bằng $ 0 $, tức là $ a=a+0i $.\\
			Như vậy, \textit{mỗi số thực cũng là một số phức}. Ta có $ \mathbb R \subset \mathbb C $.
			\item Số phức $ 0+bi $ được gọi là \textbf{số thuần ảo} và viết đơn giản là $ bi $, tức là $ bi=0+bi $.
		\end{enumerate}
	\end{note}

	\subsubsection{Môđun của số phức}

			Từ định nghĩa, suy ra $ |z|=\left|\overrightarrow{OM}\right| $ hay $ \left|a+bi\right|=\left|\overrightarrow{OM}\right| $. Khi đó $$ \boxed{\left|a+bi\right|=\sqrt{a^2+b^2}.}$$}

		$\overline{\overline{z}}=z$.
	\end{tc}

\subsection{Các dạng toán}
\begin{dang}{Xác định phần thực - phần ảo của số phức}
\begin{center}
Số phức $z=a+bi,\, a,b\in \mathbb{R}$ có
$a$ là phần thực, $b$ là phần ảo.
\end{center}
\end{dang}
\begin{vd}%[2D4Y1] 
Xác định phần thực, phần ảo của các số phức:
\begin{enumerate}
\begin{multicols}{2}
\item $z=2+3i$.
\item $z=2i-4$.
\item $z=3$.
\item $z=15i$.
\end{multicols}
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Số phức $z=2+3i$ có phần thực $a=2$ và phần ảo $b=3$.
\item Số phức $z=2i-4$ có phần thực $a=-4$ và phần ảo $b=2$.
\item Số phức $z=3$ có phần thực $a=3$ và phần ảo $b=0$.
\item Số phức $z=15i$ có phần thực $a=0$ và phần ảo $b=15$.
\end{enumerate}
}
\begin{center}
\textbf{BÀI TẬP TỰ LUYỆN}
\end{center}
\begin{bt}%[2D4Y1] 
Xác định phần thực, phần ảo của các số phức:
\begin{enumerate}
\begin{multicols}{2}
\item $z=4i$.
\item $z=-3i+4$.
\item $z=16$.
\item $z=-43+15i$.
\end{multicols}
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Số phức $z=4i$ có phần thực $a=0$ và phần ảo $b=4$.
\item Số phức $z=-3i+4$ có phần thực $a=4$ và phần ảo $b=-3$.
\item Số phức $z=16$ có phần thực $a=16$ và phần ảo $b=0$.
\item Số phức $z=-43+15i$ có phần thực $a=-43$ và phần ảo $b=15$.
\end{enumerate}
}
\end{bt}
\end{vd}
\begin{dang}{Xác định mô-đun của số phức}
Mô-đun của số phức $z=a+bi$ là $|z|=\sqrt{a^2+b^2}$.
\end{dang}
\setcounter{vd}{0}
\begin{vd}%[2D4Y1] 
Tìm mô-đun của các số phức sau:
\begin{enumerate}
\begin{multicols}{2}
\item $z=1+2i$.
\item $z=3-5i$.
\item $z=-5+4i$.
\item $z=-4i$.
\item $z=2$.
\end{multicols}
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $|z|=|1+2i|=\sqrt{1^2+2^2}=\sqrt{5}$.
\item Ta có $|z|=|3-5i|=\sqrt{3^2+(-5)^2}=\sqrt{34}$.
\item Ta có $|z|=|-5+4i|=\sqrt{(-5)^2+4^2}=\sqrt{41}$.
\item Ta có $|z|=|-4i|=\sqrt{(-4)^2}=4$.
\item Ta có $|z|=|2|=\sqrt{2^2}=2$.
\end{enumerate}
}
\end{vd}
\begin{center}
\textbf{BÀI TẬP TỰ LUYỆN}
\end{center}
\setcounter{bt}{0}
\begin{bt}%[2D4Y1] 
Tìm mô-đun của các số phức sau:
\begin{enumerate}
\begin{multicols}{2}
\item $z=\dfrac{1}{2}-\dfrac{\sqrt{3}}{2}i$.
\item $z=4i-3$.
\item $z=-3-4i$.
\item $z=-6$.
\item $z=-4i$.
\end{multicols}
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $|z|=\left|\dfrac{1}{2}-\dfrac{\sqrt{3}}{2}i\right|=\sqrt{\left(\dfrac{1}{2}\right)^2+\left(-\dfrac{\sqrt{3}}{2}\right)^2}=1$.
\item Ta có $|z|=|4i-3|=\sqrt{(-3)^2+4^2}=5$.
\item Ta có $|z|=|-3-4i|=\sqrt{(-3)^2+(-4)^2}=5$.
\item Ta có $|z|=|-6|=\sqrt{(-6)^2}=6$.
\item Ta có $|z|=|-4i|=\sqrt{(-4)^2}=4$.
\end{enumerate}
}
\end{bt}
\begin{dang}{Hai số phức bằng nhau}
Hai số phức $z=a+bi, z'=a'+b'i$ được gọi là bằng nhau nếu $\heva{&a=a'\\&b=b'.}$
\end{dang}
\setcounter{vd}{0}
\setcounter{bt}{0}
\begin{vd}%[2D4Y1] 
Tìm các số thực $x$, $y$ biết:
\begin{enumerate}
\item $x+2y+3i=4x-5y+(6-y)i$.
\item $-3x+6y-(8+4y)i=3x-4+(4x-y)i$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $x+2y+3i=4x-5y+(6-y)i \Leftrightarrow \heva{&x+2y=4x-5y\\&3=6-y}\Leftrightarrow \heva{&-3x+7y=0\\&y=3}\Leftrightarrow \heva{&x=7\\&y=3.}$
\item $-3x+6y-(8+4y)i=3x-4+(4x-y)i\\ \Leftrightarrow \heva{&-3x+6y=3x-4\\&-(8+4y)=4x-y}\Leftrightarrow \heva{&-6x+6y=-4\\&-4x-3y=8}\Leftrightarrow \heva{&x=-\dfrac{6}{7}\\&y=-\dfrac{32}{21}.}$
\end{enumerate}
}
\end{vd}
\begin{vd}%[2D4Y1] 
Cho $z=(3a+2)+(b-4)i$. Tìm các số $a,b$ để
\begin{enumerate}
\begin{multicols}{2}
\item $z$ là số thực.
\item $z$ là số thuần ảo.
\end{multicols}
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $z$ là số thực khi $b-4=0$ hay $b=4.$
\item $z$ là số thuần ảo khi $3a+2=0$ hay $a=-\dfrac{2}{3}$.
\end{enumerate}
}
\end{vd}
\begin{center}
\textbf{BÀI TẬP TỰ LUYỆN}
\end{center}
\begin{bt}%[2D4Y1] 
Tìm các số thực $x$, $y$, biết:
\begin{enumerate}
\item $(2x+1)+(3y-2)i=(x+2)+(y+4)i$.
\item $(1-3x)+(y+1)i=(x+y)-(2x+1)i$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $\heva{&2x+1=x+2\\&3y-2=y+4}\Leftrightarrow \heva{&x=1\\&y=3.}$
\item Ta có $\heva{&1-3x=x+y\\&y+1=-(2x+1)}\Leftrightarrow \heva{&4x+y=1\\&2x+y=-2}\Leftrightarrow \heva{&x=\dfrac{3}{2}\\&y=-5.}$
\end{enumerate}
}
\end{bt}
\begin{bt}%[2D4Y1] 
Tìm các số thực $x,y$, biết:
\begin{enumerate}
\item $2x+1+5i=-4+(3y-2)i$.
\item $(x-\sqrt{2})-4i=3-(y+1)i$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $\heva{&2x+1=4\\&5=3y-2}\Leftrightarrow \heva{&x=\dfrac{3}{2}\\&y=\dfrac{7}{3}.}$
\item Ta có $\heva{&x-\sqrt{2}=3\\&-4=-(y+1)}\Leftrightarrow \heva{&x=3+\sqrt{2}\\&y=3.}$
\end{enumerate}
}
\end{bt}
\begin{bt}%[2D4Y1] 
Cho số phức $z=(a-5)+(b+4)i$. Tìm các số $a,b$ để:
\begin{enumerate}
\begin{multicols}{2}
\item $z$ là số thuần ảo.
\item $z$ là số thực. 
\end{multicols}
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Để $z$ là số thuần ảo thì $a-5=0$ hay $a=5$.
\item Để $z$ là số thực thì $b+4=0$ hay $b=-4$.
\end{enumerate}
}
\end{bt}
\begin{bt}%[2D4Y1]
Cho số phức $z=(a^2-4b^2)+(a+2b)i$. Tìm các số $a,b$ để $z$ là số ảo.
\loigiai{
Để $z$ là số ảo thì $a^2-4b^2=0\Leftrightarrow \hoac{&a=2b\\&a=-2b.}$
}
\end{bt}

		\noindent Vậy ta có
		$$\cos \widehat{ACB}=\dfrac{\overrightarrow{CA}\cdot \overrightarrow{CB}}{\left| \overrightarrow{CA}\right|\cdot \left| \overrightarrow{CB}\right|}=\dfrac{6}{2\sqrt{3}\cdot 2}=\dfrac{\sqrt{3}}{2}.$$
		$$\cos \widehat{ADB}=\dfrac{\overrightarrow{DA}\cdot \overrightarrow{DB}}{\left| \overrightarrow{DA}\right|\cdot \left| \overrightarrow{DB}\right|}=\dfrac{6+4\sqrt{3}}{8+4\sqrt{3}}=\dfrac{\sqrt{3}}{2}.$$
		Do đó tứ giác $ABCD$ có $\widehat{ACB}=\widehat{ADB}=30^\circ$ nên $ABCD$ là tứ giác nội tiếp.
