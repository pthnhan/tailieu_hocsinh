\begin{enumerate} [\bf 1)]

						Nếu $3x-3\ne 0\Leftrightarrow $ $x\ne 1$\\

							\item Nếu $\sqrt{3x}-2\in \mathbb{I}$ thì $A\in \mathbb{I}$ (Vô lý).
							\item Nếu $\sqrt{3x}-2\in \mathbb{Z}$ thì $B=\sqrt{3x}+2+\dfrac{1}{\sqrt{3x}-2}$
						\end{itemize}
						Do đó $B\in \mathbb{Z}$ khi và chỉ khi $\sqrt{3x}-2\in \left\{-1;1\right\}$, từ đó tìm được $x=3$ thỏa mãn yêu  cầu.\\

\end{enumerate}

\begin{ex}%[HSG Lớp 9 - Quận Cầu Giấy - Hà Nội năm 2017-2018, Vòng 1]%[9D3G7]%[9D5G3]
\hfill
\begin{enumerate}    
	\item Giải phương trình: $x^2+6=4\sqrt{x^3-2x^2+3}$.
	\item Tìm các cặp số nguyên $(x; y)$ thỏa mãn $x^2\left(x-y\right)=5\left(y-1\right)$.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Điều kiện: $x\ge -1$. Khi đó, ta có:\\
			Phương trình $\Leftrightarrow (x^2-3x+3)+3(x+1) = 4\sqrt{(x+1)(x^2-3x+3)}$\\
			\phantom{Phương trình }$\Leftrightarrow 1+\dfrac{3(x+1)}{x^2-3x+3}=4\sqrt{\dfrac{x+1}{x^2-3x+3}}$ (vì $x^2-3x+3>0, \forall x$).\\
			Đặt $t=\sqrt{\dfrac{x+1}{x^2-3x+3}}$ $(t\ge 0)$, ta được phương trình:
			$$1+3t^2 = 4t \Leftrightarrow 3t^2-4t+1 = 0 \Leftrightarrow \hoac{& t=1 \,\text{(thỏa mãn điều kiện)} \\& t=\dfrac{1}{3} \,\text{(thỏa mãn điều kiện)}}$$
			\begin{itemize}
				\item Với $t = 1$, ta có: $\sqrt{\dfrac{x+1}{x^2-3x+3}}=1\Leftrightarrow x^2-4x+2=0\Leftrightarrow x=2\pm \sqrt{2}$ (thỏa mãn điều kiện).
				\item Với $t = \dfrac{1}{3}$, ta có: $\sqrt{\dfrac{x+1}{x^2-3x+3}}=\dfrac{1}{3}\Leftrightarrow x^2-12x-6=0\Leftrightarrow x=6\pm \sqrt{42}$ (thỏa mãn điều kiện).				
			\end{itemize}
			Vậy phương trình đã cho có tập nghiệm $S = \left\lbrace 2\pm \sqrt{2}; 6\pm \sqrt{42}\right\rbrace $.
	\item Ta có $x^2\left(x-y\right)=5\left(y-1\right) \Leftrightarrow x^2\left(x-y\right)=5\left(y-x\right)+5\left(x-1\right)$\\
			 \phantom{Ta có $x^2\left(x-y\right)=5\left(y-1\right)$ }$\Leftrightarrow \left(x-y\right)\left(x^2+5\right)=5\left(x-1\right) $\\
			Suy ra: $5\left(x-1\right)\vdots \left(x^2+5\right)$ \hfill $(1)$\\
			Do đó: $\left[5\left(x^2+5\right)-5x\left(x-1\right)\right]\vdots \left(x^2+5\right)\Rightarrow 25+5x\vdots \left(x^2+5\right)$ \hfill $(2)$\\
			Từ $(1)$ và $(2)$, suy ra: $\left(25+5x\right)-5\left(x-1\right)\vdots \left(x^2+5\right)\Rightarrow 30\vdots \left(x^2+5\right)$.\\
			Suy ra:$x^2+5\in \left\lbrace 5;6;10;15;30\right\rbrace \Leftrightarrow x^2\in \left\lbrace 0;1;5;10;25\right\rbrace $.\\
			Từ đó tìm được $x\in \left\lbrace 0;1;-1;5;-5\right\rbrace $.\\
			Thử lại vào phương trình đã cho, ta được các nghiệm nguyên của phương trình là $$(x;y) \in  \left\lbrace (0; 1); (1;1); (-5;-4)\right\rbrace .$$
\end{enumerate}
}

\begin{enumerate}
	\item Cho ba số dương $a$, $b$, $c$ thỏa mãn $abc=1$. Tìm giá trị lớn nhất của biểu thức
			$$A=\dfrac{1}{a^2+2b^2+3}+\dfrac{1}{b^2+2c^2+3}+\dfrac{1}{c^2+2a^2+3}$$
	\item Tìm tất cả các bộ số nguyên dương $\left(x;y;z\right)$ thỏa mãn $\dfrac{x+y\sqrt{2017}}{y+z\sqrt{2017}}$ là số hữu tỷ, đồng thời $x^2+y^2+z^2$ là số nguyên tố.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có $a^2+2b^2+3=a^2+b^2+b^2+1+2\ge 2ab+2b+2=2(ab+b+1)$\\
			Tương tự ta có $b^2+2c^2+3\ge 2(bc+c+1)$; $c^2+2a^2+3\ge 2(ca+a+1)$.\\
			Do đó $A\le \dfrac{1}{2}\left(\dfrac{1}{ab+b+1}+\dfrac{1}{bc+c+1}+\dfrac{1}{ca+a+1}\right)$.\\
			Ta chứng minh được  $\dfrac{1}{ab+b+1}+\dfrac{1}{bc+c+1}+\dfrac{1}{ca+a+1}=1$, suy ra $A\le \dfrac{1}{2}$.\\
			Dấu “=” xảy ra khi và chỉ khi $a=b=c=1$.\\
			Vậy giá trị lớn nhất của $A$ là $\dfrac{1}{2}$  khi  $a=b=c=1$.
	\item Ta có $\dfrac{x+y\sqrt{2017}}{y+z\sqrt{2017}}=\dfrac{m}{n}\left(m,n\in {\mathbb{N}}^{*},\left(m,n\right)=1\right)$.\\
			 \phantom{Ta có }$\Leftrightarrow nx-my=\left(mz-ny\right)\sqrt{2017}$ $\Rightarrow \heva{& nx-my=0 \\ & mz-ny=0}\Rightarrow \dfrac{x}{y}=\dfrac{y}{z}=\dfrac{m}{n}\Rightarrow xz=y^2$.\\
			Ta có $x^2+y^2+z^2={\left(x+z\right)}^2-2xz+y^2={\left(x+z\right)}^2-y^2=\left(x+y+z\right)\left(x+z-y\right)$\\
			Vì $x+y+z>1$ và $x^2+y^2+z^2$ là số nguyên tố nên $\heva{&x^2+y^2+z^2=x+y+z \\& x-y+z=1}$\\
			Từ đó suy ra $x=y=z=1$ (thỏa mãn).
\end{enumerate}
}

\begin{ex}%[HSG Lớp 9 - Quận Cầu Giấy - Hà Nội năm 2017-2018, Vòng 1]%[9H2K6]%[9H2G6]%[9D5G1]
Cho đường tròn tâm $I$ bán kính $r$ nội tiếp tam giác $ABC$ tiếp xúc với cạnh $BC$ tại $D$. Vẽ đường kính $DN$ của đường tròn $(I;r)$. Tiếp tuyến của đường tròn $(I)$ tại $N$ cắt cạnh $AB$, $AC$ theo thứ tự tại $P$ và $K$.
\begin{enumerate}
	\item Chứng minh rằng $NK.CD=r^2$.
	\item Gọi $E$ là giao điểm của $AN$ và $BC$. Chứng minh rằng $BD = CE$.
    \item Tìm giá trị nhỏ nhất của biểu thức $\dfrac{IA+IB+IC}{r}$.
\end{enumerate}
\loigiai{
\begin{center}

    	\tkzDrawAltitude[draw =blue](A,I)(C) \tkzGetPoint{C'}

\begin{enumerate}
	\item Ta có $\widehat{ICK} + \widehat{IKC} = \dfrac{1}{2}\left(\widehat{DCK} + \widehat{NKC}\right) = 90^\circ$.\\
			$\Rightarrow \widehat{KIC} = 90^\circ$ $\Rightarrow \widehat{NIK} = \widehat{ICD}$ (cùng phụ $\widehat{DIC}$).\\
			$\Rightarrow \triangle INK \backsim \triangle CDI \Rightarrow \dfrac{IN}{CD} = \dfrac{NK}{ID}$.\\
			Do đó, ta có $NK.CD = IN.ID = r^2$.
	\item Chứng minh tương tự câu a ta có $NP.DB=r^2$.\\
			Suy ra $NK.CD=NP.DB\Rightarrow \dfrac{NP}{NK}=\dfrac{DC}{DB}\Leftrightarrow \dfrac{NP}{DC}=\dfrac{NK}{DB}=\dfrac{NP+NK}{DC+DB}=\dfrac{PK}{BC}$\\
			Vì $PK\parallel BC$ nên theo định lí Talet ta có $\dfrac{NK}{EC}=\dfrac{AK}{AC}=\dfrac{PK}{BC}$.\\
			Suy ra $\dfrac{NK}{EC}=\dfrac{NK}{DB}\Rightarrow EC=DB$.
	\item Đặt $BC = a$; $AC = b$; $AB = c$. Gọi $B’$, $C’$ lần lượt là hình chiếu của $B$, $C$ trên $AI$.\\
			Ta có $cr=IA.BB';br=IA.CC'$ nên $(b+c)r=IA(BB'+CC')\le IA.a\Rightarrow \dfrac{IA}{r}\ge \dfrac{b+c}{a}$\\
			Tương tự ta có $\dfrac{IB}{r}\ge \dfrac{a+c}{b};\dfrac{IC}{r}\ge \dfrac{a+b}{c}$.\\
			Suy ra $\dfrac{IA+IB+IC}{r}\ge \left(\dfrac{a}{b}+\dfrac{b}{a}\right)+\left(\dfrac{b}{c}+\dfrac{c}{b}\right)+\left(\dfrac{a}{c}+\dfrac{c}{a}\right)\ge 6$.\\
			Dấu ``='' xảy ra khi $a=b=c$.\\
			Vậy $\dfrac{IA+IB+IC}{r}$ đạt giá trị nhỏ nhất bằng 6 khi $\triangle ABC$ đều.
\end{enumerate}

\begin{ex}%[HSG Lớp 9 - Quận Cầu Giấy - Hà Nội năm 2017-2018, Vòng 1]%[9D5G4]
Cho một đa giác đều có $2017$ đỉnh. Người ta ghi lên mỗi đỉnh của đa giác số $1$ hoặc số $2$. Biết rằng có $1007$ số $1$ và $1010$ số $2$ và các số trên $3$ đỉnh liên tiếp bất kỳ không đồng thời bằng nhau. Hãy tính tổng của tất cả các tích ba số trên $3$ đỉnh liên tiếp của đa giác trên.
\loigiai{
Có $2017$ đỉnh nên có $2017$ tích ba số trên ba đỉnh liên tiếp. Vì $3$ đỉnh liên tiếp bất kì các số không bằng nhau nên chỉ có $2$ loại tích:
\begin{itemize}
	\item Loại I: Ba số ở ba đỉnh liên tiếp chỉ có một số $2$, tích ba số này bằng $2$.
	\item Loại II : Ba số ở ba đỉnh liên tiếp có hai chữ số $2$, tích ba số này bằng $4$.
\end{itemize}
Gọi số loại I là $x$ $(x\in \mathbb{N})$ thì số tích loại II là $2017-x$. Mà số $2$ ở $2017$ tích này là $1010\times 3 = 3030$.\\
Ta có phương trình $x.1+(2017-x).2=3030\Leftrightarrow x=1004$.\\
Vậy tổng tất cả các tích cần tìm là: $1004\times 2+1013\times 4 = 6060$.
}
\end{ex}
