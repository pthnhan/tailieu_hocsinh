	Cho $\displaystyle\int \limits_0^1{x\cdot \ln \sqrt{3x^2+1}}\mathrm{\,d}x=a\cdot \ln {2}-\dfrac{b}{c}$ (với $a$ là số hữu tỉ, $b$ và $c$ là các số nguyên dương, $\dfrac{b}{c}$ là phân số tối giản). Hãy tính giá trị của $a\cdot b\cdot c$.
	\choice
	{$6$}
	{$3$}
	{\True $\dfrac{8}{3}$}
	{$\dfrac{4}{3}$}
	\loigiai{
	Đặt $I=\displaystyle\int \limits_0^1{x\cdot \ln \sqrt{3x^2+1}}\mathrm{\,d}x$.\\
	Đặt $t=\sqrt{3x^2+1} \Rightarrow t^2=3x^2+1 \Rightarrow \dfrac{t}{3}\mathrm{\,d}t=x\mathrm{\,d}x$.\\
	Đổi cận $x=0 \Rightarrow t=1;x=1 \Rightarrow t=2$.\\
	$$I=\displaystyle\int\limits_1^2{\dfrac{t}{3} \ln t}\mathrm{\,d}t.$$\\
	Đặt $\heva{&u=\ln t\\&\mathrm{d}v=\dfrac{t}{3}\mathrm{d}t} \Rightarrow \heva{&\mathrm{d}u=\dfrac{\mathrm{d}t}{t}\\&v=\dfrac{t^2}{6}.}$\\
	$$I=\left.\dfrac{t^2}{6} \ln t \right|_1^2- \displaystyle\int\limits_1^2{\dfrac{t}{6}}\mathrm{\,d}t= \dfrac{2}{3} \ln 2 -\left. \dfrac{t^2}{12}\right|_1^2=\dfrac{2}{3} \ln 2-\dfrac{1}{4}.$$\\
	Vậy $a\cdot b\cdot c=\dfrac{2}{3}\cdot 1 \cdot 4 =\dfrac{8}{3}$.
	}

\begin{ex}%[Thi thử L2, THPT Huỳnh Thúc Kháng Khánh Hòa, 2018]%[Dương Xuân Lợi, dự án 12(Ex-9)]%[2D1G1-4]
	Cho $a$ và $b$ là hai số dương thỏa mãn $2(a^2 +b^2) + ab = (a +b)(ab + 2)$. Biết giá trị nhỏ nhất của biểu thức $P=4 \left(\dfrac{a^3}{b^3} + \dfrac{b^3}{a^3}\right) -9\left(\dfrac{a^2}{b^2} + \dfrac{b^2}{a^2}\right)$ là $S=\dfrac{m}{n}$ (với $m, n$ là các số nguyên, $n > 0$ và $\dfrac{m}{n}$ là phân số tối giản). Hãy tính $m +n$.
	\choice
	{$-18$}
	{\True $-19$}
	{$-20$}
	{$-17$}
	\loigiai{
	Với $a, b$ là số dương ta có 
	\begin{eqnarray*}
	& &	2(a^2 +b^2) + ab = (a +b)(ab + 2)\\
	&\Leftrightarrow & 2(a^2 +b^2) + ab = a^2b +ab^2 +2(a +b) \\
	&\Leftrightarrow & 2\left(\dfrac{a}{b} +\dfrac{b}{a} \right)+1 = (a+b)+2\left(\dfrac{1}{a} +\dfrac{1}{b}\right)
	\end{eqnarray*}
	Mặt khác $(a+b) +2\left(\dfrac{1}{a} +\dfrac{1}{b}\right) \ge 2\sqrt{2(a+b)\left(\dfrac{1}{a} +\dfrac{1}{b}\right)} =2\sqrt{2\left(\dfrac{a}{b} +\dfrac{b}{a}+2\right)}$.\\
	Suy ra $2\left(\dfrac{a}{b} +\dfrac{b}{a} \right) + 1 \ge 2\sqrt{2\left(\dfrac{a}{b} +\dfrac{b}{a}+2\right)} \Rightarrow \dfrac{a}{b} +\dfrac{b}{a} \ge \dfrac{5}{2}$.\\
	Đặt $t=\dfrac{a}{b} +\dfrac{b}{a}$, $t \ge \dfrac{5}{2}$ suy ra $P=4(t^3-3t) -9(t^2-2)=4t^3-9t^2-12t+18$.\\
	Xét hàm số $f(t)=4t^3-9t^2-12t+18$, với $t \ge \dfrac{5}{2}$.\\
	Ta có $f'(t)=6(2t^2-3t-2)> 0$, với $\forall t \ge \dfrac{5}{2}$ suy ra $\min \limits_{\left[\tfrac{5}{2};+\infty\right)}f(t)=f \left(\dfrac{5}{2}\right)=\dfrac{-23}{4}$.\\
	Vậy $\min P=\dfrac{-23}{4} \Rightarrow m+n=-19$.  
	}
\end{ex}

	& \Leftrightarrow & \hoac{&x-\dfrac{\pi}{6}=\dfrac{\pi}{4}+k2\pi\\&x-\dfrac{\pi}{6}=\pi -\dfrac{\pi}{4}+k2\pi}, (k \in \mathbb{Z})\\
	& \Leftrightarrow & \hoac{&x= \dfrac{5\pi}{12}+k2\pi \\&x=\dfrac{11\pi}{12}+k2\pi}, (k \in \mathbb{Z}).
	\end{eqnarray*}
	Vì $x \in [-2\pi;2\pi]$ nên $x \in \left\{\dfrac{5\pi}{12};\dfrac{-19\pi}{12};\dfrac{11\pi}{12};\dfrac{-13\pi}{12} \right\}$.\\
	Tổng các nghiệm là $\dfrac{5\pi}{12}+\dfrac{-19\pi}{12}+\dfrac{11\pi}{12}+\dfrac{-13\pi}{12}=\dfrac{-4\pi}{3}$.
	}
\end{ex}
