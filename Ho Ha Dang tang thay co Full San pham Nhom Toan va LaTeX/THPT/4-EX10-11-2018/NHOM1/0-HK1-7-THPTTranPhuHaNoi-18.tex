\begin{ex}%[Đề HK1, THPT Trần Phú, Hà Nội 2017 - 2018]%[0D2B1]

\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[Đề HK1, THPT Trần Phú, Hà Nội 2017 - 2018]%[0D3K2]
	(\textbf{1 điểm}) Giải phương trình: $\sqrt{2x^2-5x+2}=x-2$
	\loigiai{
		Ta có: $\sqrt{2x^2-5x+2}=x-2 \Leftrightarrow  \heva{&2x^2-5x+2=(x-2)^2\\&x-2\ge 0}\Leftrightarrow  \heva{&x^2-x-2=0\\&x\ge 0}$\\
$\Leftrightarrow  \heva{&x=-1 \,\, \text{hoặc}\,\, x=2\\&x\ge 0}\Leftrightarrow  x=2$.\\
Vậy, phương trình có nghiệm duy nhất $x=2$.
	}
\end{bt}

\begin{bt}%[Đề HK1, THPT Trần Phú, Hà Nội 2017 - 2018]%[0D3B3]%[0D3K3]
	Cho hệ phương trình: $\heva{&2x-y+1=0\\&x^2-3xy+y^2=2x-5+m^2}$.
	\begin{enumerate}
		\item (\textbf{1 điểm}) Giải hệ phương trình với $m=0$.
		\item (\textbf{1 điểm})  Tìm $m$ để hệ phương trình có nghiệm.
\end{enumerate}
	\loigiai{Xét hệ phương trình: $\heva{&2x-y+1=0\;\; (1)\\&x^2-3xy+y^2=2x-5+m^2\;\; (2)}$.\\
		Từ phương trình $(1)$ ta có $y=2x+1\;\; (3)$. Thay $(3)$ vào phương trình $(2)$ ta được:\\
			$$x^2-3x(2x+1)+(2x+1)^2=2x-5+m^2\Leftrightarrow x^2+x+m^2-6=0\;\; (4)$$
	\begin{enumerate}
		\item Khi $m=0$ phương trình $(4)$ trở thành: $x^2+x-6=0\Leftrightarrow \hoac{&x=-3\\&x=2}$.\\
				Khi $x=-3$ thì $y=-5$.\\
				Khi $x=2$ thì $y=5$.\\
				Như vậy, $m=0$, hệ phương trình có hai nghiệm $(x;y)$ là $(-3;-5)$ và $(2;5)$ .
		\item Hệ phương trình đã cho có nghiệm khi và chỉ khi phương trình $(4)$ có nghiệm, điều này tương đương với:
				$$ \Delta =1-4(m^2-6)\ge 0 \Leftrightarrow m^2\le \dfrac{25}{4}\Leftrightarrow -\dfrac{5}{2}\le m\le\dfrac{5}{2} $$
				Vậy: $-\dfrac{5}{2}\le m\le\dfrac{5}{2}$.
	\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề HK1, THPT Trần Phú, Hà Nội 2017 - 2018]%[0H2B2]%[0H2K2]
	Trong mặt phẳng tọa độ $Oxy$ cho tam giác $ABC$ với $A(-1;1)$, $B(3;1)$, $C(2;4)$.
	\begin{enumerate}
		\item (\textbf{0,5 điểm}) Tính $\overrightarrow{AB}\cdot \overrightarrow{AC}$.
		\item (\textbf{0,5 điểm})  Chứng minh góc $\widehat{BAC}=45^\circ$.
\end{enumerate}
	\loigiai{Ta có $\overrightarrow{AB}=(4;0)$ và $\overrightarrow{AC}=(3;3)$.
	\begin{enumerate}
		\item $\overrightarrow{AB}\cdot \overrightarrow{AC}=4\cdot 3+0\cdot 3=12$.
		\item $\cos{\widehat{BAC}}=\cos{\left(\overrightarrow{AB}; \overrightarrow{AC}\right)}=\dfrac{\overrightarrow{AB}\cdot \overrightarrow{AC}}{\left|\overrightarrow{AB}\right|\cdot \left|\overrightarrow{AC}\right|}$\\
				$=\dfrac{12}{\sqrt{4^2+0^2}\cdot \sqrt{3^2+3^2}}=\dfrac{12}{4\cdot 3\sqrt{2}}=\dfrac{\sqrt{2}}{2}$.\\
		Do đó: $\widehat{BAC}=45^\circ$.
	\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề HK1, THPT Trần Phú, Hà Nội 2017 - 2018]%[0H1B3]%[0H2G2]
	Cho tam giác $ABC$. Gọi $I$ là trung điểm cạnh $BC$, $K$ là điểm thuộc cạnh $AC$ sao cho $KC=2AK$.
	\begin{enumerate}
		\item (\textbf{1 điểm}) Biểu diễn các vectơ $\overrightarrow{AI}$, $\overrightarrow{AK}$ theo hai  vectơ $\overrightarrow{AB}$, $\overrightarrow{AC}$. Chứng minh: $\overrightarrow{KI}=\dfrac{1}{2}\overrightarrow{AB}+\dfrac{1}{6}\overrightarrow{AC}$.
		\item (\textbf{0,5 điểm})  Xác định vị trí điểm $M$ sao cho $2MA^2+MB^2+MC^2$ đạt giá trị nhỏ nhất.
\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item 
\immini{ $I$ là trung điểm cạnh $BC$ nên $\overrightarrow{AI}=\dfrac{1}{2}\left(\overrightarrow{AB}+\overrightarrow{AC}\right)$.\\
				Vì $K$ là điểm thuộc cạnh $AC$ và $KC=2AK$ nên\\
				 $$\overrightarrow{AK}=\dfrac{1}{3}\overrightarrow{AC}$$.
				 Ta có:		
				 $$\overrightarrow{KI}=\overrightarrow{AI}-\overrightarrow{AK}=\dfrac{1}{2}\left(\overrightarrow{AB}+\overrightarrow{AC}\right)-\dfrac{1}{3}\overrightarrow{AC}$$
		$$=\dfrac{1}{2}\overrightarrow{AB}+\dfrac{1}{6}\overrightarrow{AC}$$}

		\item Gọi $D$ là điểm sao cho $2\overrightarrow{DA}+\overrightarrow{DB}+\overrightarrow{DC}=\overrightarrow{0}$.\\

			Như vậy $D$ là trung điểm của $AI$.\\
