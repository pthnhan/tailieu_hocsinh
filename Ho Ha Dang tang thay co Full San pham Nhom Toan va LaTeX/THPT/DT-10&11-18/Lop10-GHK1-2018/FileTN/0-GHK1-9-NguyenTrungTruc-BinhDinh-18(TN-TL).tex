\noindent\textbf{I. PHẦN TRẮC NGHIỆM}
\begin{ex}%[GHKI, Nguyễn Trung Trực, Bình Định, 2018]%[Lê Xuân Hòa, EX-DT-10\&11-18]%[0D2Y2-1]
Hàm số $y=(2+m)x +3m$ nghịch biến khi
	\choice
	{$m>2$}
	{$m=2$}
	{$m>-2$}
	{\True $m<-2$}
\loigiai{
Hàm số nghịch biến $\Leftrightarrow m+2 <0 \Leftrightarrow m <-2$.\\
Vậy $m<-2$ là giá trị cần tìm.
}
\end{ex}

\begin{ex}%[GHKI, Nguyễn Trung Trực, Bình Định, 2018]%[Lê Xuân Hòa, EX-DT-10\&11-18]%[0D2Y3-2]
Cho hàm số bậc hai $y =ax^2 +bx +c $ $(a\ne 0)$ có đồ thị $(P)$, đỉnh được xác định bởi công thức nào?
	\choice
	{$I\left(-\dfrac{b}{2a}; -\dfrac{\Delta}{2a}\right)$}
	{\True $I\left( -\dfrac{b}{2a}; -\dfrac{\Delta}{4a}\right)$}
	{$I\left( -\dfrac{b}{a}; \dfrac{\Delta}{4a}\right)$}
	{$I\left( -\dfrac{b}{a}; -\dfrac{\Delta}{2a}\right)$}
\loigiai{
Tọa độ đỉnh của $(P)\colon y =ax^2 +bx +c$ là $I\left(-\dfrac{b}{2a}; -\dfrac{\Delta}{4a}\right)$.
}
\end{ex}

\begin{ex}%[GHKI, Nguyễn Trung Trực, Bình Định, 2018]%[Lê Xuân Hòa, EX-DT-10\&11-18]%[0D2Y1-2]
Tập xác định của hàm số $y=\dfrac{\sqrt{x^2+1}}{x+1}$ là
	\choice
	{\True $\mathbb{R}\setminus \{-1\}$}
	{$(-1;1)$}
	{$\mathbb{R}\setminus \{1; -1\}$}
	{$\mathbb{R}$}
\loigiai{
Điều kiện để hàm số có nghĩa là $\heva{&x^2+1\ge 0\\&x-1\ne 0}\Leftrightarrow x\ne 1$.\\
Vậy $\mathscr{D} = \mathbb{R}\setminus \{-1\}$.
}
\end{ex}

\begin{ex}%[GHKI, Nguyễn Trung Trực, Bình Định, 2018]%[Lê Xuân Hòa, EX-DT-10\&11-18]%[0D2Y3-3]
Đồ thị hàm số $y=\heva{&2x+1\,\, \text{khi } x\le 2 \\& x^2 -3\,\, \text{khi } x>2}$ đi qua điểm có tọa độ
	\choice
	{$(0;3)$}
	{\True $(0;1)$}
	{$(0;-3)$}
	{$(-3;0)$}
\loigiai{
Tập xác định của hàm số là $\mathscr{D}=\mathbb{R}$. \\
Đồ thị gồm hai phần, phần đường thẳng  $y=2x+1$ với $x\le 2$ và phần Parabol $y=x^2-3$ với $x> 2$.\\
Thay tọa độ điểm $(0;1)$ vào phương trình đường thẳng $y=2x+1$ thỏa mãn.\\
Vậy đồ thị hàm số đi qua điểm $(0;1)$.
}
\end{ex}

\begin{ex}%[GHKI, Nguyễn Trung Trực, Bình Định, 2018]%[Lê Xuân Hòa, EX-DT-10\&11-18]%[0D2B3-2]
Parabol $y=ax^2 +bx +2$ đi qua $M(1;5)$ và $N(-2;8)$ có phương trình
	\choice
	{$y=-2x^2+x+2$}
	{\True $y=2x^2+x+2$}
	{$y=-2x^2-x-2$}
	{$y=2x^2-x+2$}
\loigiai{
Vì $M,N \in (P)$ nên ta có hệ phương trình $\heva{&a+b+2=5\\&4a-2b+2=8}\Leftrightarrow \heva{&a= 2\\& b= 1}$.\\
Vậy $(P)\colon y=2x^2+x+2$.
}
\end{ex}

\begin{ex}%[GHKI, Nguyễn Trung Trực, Bình Định, 2018]%[Lê Xuân Hòa, EX-DT-10\&11-18]%[0D2Y3-1]
Cho hàm số $y=x^2-4x+1$. Trong các khẳng định sau, khẳng định nào đúng?
	\choice
	{Hàm số đồng biến trên khoảng $(-\infty ; 2)$ và nghịch biến trên khoảng $(2;+\infty )$}
	{\True Hàm số nghịch biến trên khoảng $(-\infty ; 2)$ và đồng biến trên khoảng $(2;+\infty )$}
	{Hàm số đồng biến trên khoảng $(-\infty ; 0)$ và nghịch biến trên khoảng $(0;+\infty )$}
	{Hàm số nghịch biến trên khoảng $(-\infty ; 0)$ và đồng biến trên khoảng $(0;+\infty )$}
\loigiai{
Vì $a=1>0$ nên hàm số nghịch biến trên khoảng $(-\infty ; 2)$ và đồng biến trên khoảng $(2;+\infty )$.
}
\end{ex}

\begin{ex}%[GHKI, Nguyễn Trung Trực, Bình Định, 2018]%[Lê Xuân Hòa, EX-DT-10\&11-18]%[0D2Y2-4]
Cho hai đường thẳng $(d_1)\colon \, y=\dfrac{1}{2}x+100$ và $(d_2)\colon \, y=-\dfrac{1}{2}x +100$. Mệnh đề nào sau đây đúng?
	\choice
	{$(d_1)$ và $(d_2)$ trùng nhau}
	{\True $(d_1)$ và $(d_2)$ cắt nhau}
	{$(d_1)$ và $(d_2)$ vuông góc nhau}
	{$(d_1)$ và $(d_2)$ song song với nhau}
\loigiai{
Hệ số góc của đường thẳng $(d_1)$ là $k_1 = \dfrac{1}{2}$ và hệ số góc của đường thẳng $(d_2)$ là $k_2 =-\dfrac{1}{2}$. Vì $\dfrac{1}{2}\ne -\dfrac{1}{2}$ nên hai đường thẳng cắt nhau.\\
Vậy $(d_1)$ và $(d_2)$ cắt nhau.
}
\end{ex}

\begin{ex}%[GHKI, Nguyễn Trung Trực, Bình Định, 2018]%[Lê Xuân Hòa, EX-DT-10\&11-18]%[0D2B2-4]
Với giá trị nào của $k$ thì đồ thị hàm số $y=(k-1)x-2$ song song với trục hoành	
	\choice
	{$k=-1$}
	{\True $k=1$}
	{$k>1$}
	{$k<1$}
\loigiai{
Đường thẳng $y=ax +b$ song song với trục hoành $\Leftrightarrow \heva{&a=0\\&b\ne 0}\Leftrightarrow\heva{&k-1=0\\&-2\ne 0} \Leftrightarrow k=1$.\\
Vậy giá trị $k$ cần tìm là $k=1$.
}
\end{ex}

\begin{ex}%[GHKI, Nguyễn Trung Trực, Bình Định, 2018]%[Lê Xuân Hòa, EX-DT-10\&11-18]%[0D2B1-4]
Hàm số $f(x)=x\left( x^4 -3x^2 -5\right)$ là 
	\choice
	{Hàm số vừa chẵn, vừa lẻ}
	{\True Hàm số lẻ}
	{Hàm số chẵn}
	{Hàm số không chẵn, không lẻ}
\loigiai{
\begin{itemize}
\item Tập xác định $\mathscr{D} =\mathbb{R}$.
\item $\forall x\in \mathbb{R}\Rightarrow -x\in\mathbb{R}$.
\item $\forall x\in\mathbb{R}$, ta có 
$$f(-x) = (-x)\left[ (-x)^4 -3(-x)^2 -5\right) = -x\left( x^4 -3x^2 -5\right] =-f(x).$$
Vậy hàm số đã cho là hàm số lẻ.
\end{itemize}
}
\end{ex}

	\choice

\loigiai{
Dựa vào bảng biến thiên, suy ra hàm số bậc hai có đồ thị là một parabol với $a<0$ và tọa độ đỉnh $I(1;2)$.
\begin{itemize}
\item Hàm số $y=x^2-2x+3$ có $a=1>0$ loại.
\item Hàm số $y=-x^2-2x+5$ có $a=-1<0$ và tọa độ đỉnh $I(-1; 6)$ loại.

\end{itemize}
 Vậy hàm số cần tìm là $y=-x^2+2x +1$.
}
\end{ex}

\begin{ex}%[GHKI, Nguyễn Trung Trực, Bình Định, 2018]%[Lê Xuân Hòa, EX-DT-10\&11-18]%[0D2Y3-4]
	Tọa độ giao điểm của hai đồ thị hàm số $y=x^2-2x+3$ và $y=x^2+2x-1$.
	\choice
	{\True $(1;2)$}
	{$(0;4)$}
	{$(-1;6)$}
	{$(-1;-2)$}
\loigiai{
Phương trình hoành độ giao điểm của hai đường
\begin{align*}
x^2-2x+3&=x^2+2x-1\\
\Leftrightarrow x=1 \Rightarrow y= 2.
\end{align*}
Vậy tọa độ giao điểm của hai đồ thị hàm số là $(1;2)$.
}
\end{ex}

\noindent\textbf{II. PHẦN TỰ LUẬN}

\begin{bt}%[GHKI, Nguyễn Trung Trực, Bình Định, 2018]%[Lê Xuân Hòa, EX-DT-10\&11-18]%[0D2B1-2]
Tìm tập xác định của  hàm số sau $y=\sqrt{2+3x}-\dfrac{x-2}{x^2-5x-6}$.

\loigiai{
Để hàm số có nghĩa thì $\heva{&2+3x\ge 0\\& x^2-5x-6 \ne 0}\Leftrightarrow\heva{&x\ge -\dfrac{2}{3}\\& x\ne -1; x\ne 6}\Leftrightarrow\heva{&x\ge -\dfrac{2}{3}\\&x\ne 6}$.\\
Vậy $\mathscr{D}=\left[ -\dfrac{2}{3}; +\infty\right) \setminus \{ 6\} = \left[ -\dfrac{2}{3}; 6\right) \cup \left( 6; +\infty \right)$.
}
\end{bt}

\begin{bt}%[GHKI, Nguyễn Trung Trực, Bình Định, 2018]%[Lê Xuân Hòa, EX-DT-10\&11-18]%[0D2Y1-4]
Xét tính chẵn lẻ của hàm số $f(x)=\dfrac{-x^4-2x^2+4|x|}{9-x^2}$.
\loigiai{
\begin{itemize}
\item Tập xác định $\mathscr{D}=\mathbb{R}\setminus \{\pm 3\}$.
\item $\forall  \in \mathscr{D} \Rightarrow -x\in \mathscr{D}$.
\item $\forall x\in \mathscr{D}$, ta có $f(-x) =\dfrac{-(-x)^4-2(-x)^2 +4|-x|}{9-(-x)^2} = \dfrac{-x^4-2x^2+4|x|}{9-x^2} = f(x)$
\end{itemize}
Vậy hàm số đã cho là hàm số chẵn.
}
\end{bt}

\begin{bt}%[GHKI, Nguyễn Trung Trực, Bình Định, 2018]%[Lê Xuân Hòa, EX-DT-10\&11-18]%[0D2B3-2]%[0D2Y3-3]
Cho hàm số $y=ax^2+bx+3$.
\begin{enumerate}
\item Tìm $a$, $b$ biết Parabol đi qua điểm $A(-1;8)$ và có trục đối xứng $x=2$.
\item Lập bảng biến thiên và vẽ đồ thị hàm số khi $a=1$, $b=-4$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Vì Parabol đi qua $A(-1;8)$ và trục đối xứng $x=2=-\dfrac{b}{2a}$ nên ta có hệ phương trình $$\heva{&a-b+3=8\\&2=-\dfrac{b}{2a}}\Leftrightarrow\heva{&a=1\\&b=-4.}$$
Vậy $(P)\colon y = x^2 -4x +3$.
\item Với $a=1$; $b=-4$ thì Parabol có dạng $y=x^2-4x+3$.
\begin{itemize}
\item Tập xác định $\mathscr{D}=\mathbb{R}$.
\item Sự biến thiên\\
Vì $a=1>0$ nên hàm số đồng biến trong khoảng $(2: +\infty )$ và nghịch biến trên khoảng $(-\infty ; 2)$.
\item Bảng biến thiên
\begin{center}

\item Đồ thị hàm số là $1$ parabol có trục đối xứng $x=2$.\\
Với $x=0 \Rightarrow y=3$; $x=-1 \Rightarrow y = 8$.
\end{itemize}
\end{enumerate}
