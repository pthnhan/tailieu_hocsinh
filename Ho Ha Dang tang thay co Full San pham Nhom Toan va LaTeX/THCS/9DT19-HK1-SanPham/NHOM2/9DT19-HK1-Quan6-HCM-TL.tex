	Thực hiện các phép tính
	\begin{enumerate}
\item $5\sqrt{12}-\sqrt{27}+3\sqrt{48}-9\sqrt{3}$.
\item $\left(\dfrac{\sqrt{14}-\sqrt{7}}{\sqrt{2}-1}- \dfrac{\sqrt{15}-\sqrt{5}}{1-\sqrt{3}} \right): \dfrac{2}{\sqrt{7}-\sqrt{5}}$.
\end{enumerate}
	\loigiai{
		\begin{enumerate}
\item Ta có $5\sqrt{12}-\sqrt{27}+3\sqrt{48}-9\sqrt{3}=10\sqrt{3}-3\sqrt{3}+12\sqrt{3}-9\sqrt{3}=10\sqrt{3}$.
\item Ta có 
	\begin{eqnarray*}
	& & \left(\dfrac{\sqrt{14}-\sqrt{7}}{\sqrt{2}-1}- \dfrac{\sqrt{15}-\sqrt{5}}{1-\sqrt{3}} \right): \dfrac{2}{\sqrt{7}-\sqrt{5}}\\
		&=& \left( \frac{\sqrt{7}(\sqrt{2}-1)}{\sqrt{2}-1}+\frac{\sqrt{5}(1-\sqrt{3})}{1-\sqrt{3}} \right) \cdot \frac{\sqrt{7}-\sqrt{5}}{2}   \\
		&= &(\sqrt{7}+\sqrt{5})\frac{(\sqrt{7}-\sqrt{5})}{2}=1. 
	\end{eqnarray*}
\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Quận 6, Hồ Chí Minh]%[Phan Ngọc Toàn, dự án 9DT19-HK1]%[9H2K6]
	Cho đường tròn $(O)$, điểm $M$ nằm ngoài đường tròn. Kẻ các tiếp tuyến $MB$, $MC$ với đường tròn ($B$, $C$ là các tiếp điểm).
\begin{enumerate}
\item Chứng minh rằng $OM$ vuông góc với $BC$.
\item Vẽ đường kính $CE$. Chứng minh rằng $BE$ song song với $MO$.
\item Tính độ dài các cạnh của tam giác $BCM$; biết $OB=2$ cm, $OM=3$ cm.
\end{enumerate}
\loigiai{
\begin{center}
	\begin{tikzpicture}[line join = round, line cap = round,>=stealth,scale=1]
	\tkzDefPoints{0/0/O,-5/0/M}
	\tkzDefShiftPoint[O](0:2){A}
	\tkzDrawCircle[radius](O,A)
	\tkzTangent[from=M](O,A)    \tkzGetPoints{C}{B}
	\tkzInterLL(M,O)(C,B)    \tkzGetPoint{H}
	\tkzDefPointBy[symmetry = center O](C)    \tkzGetPoint{E}
	\tkzDrawSegments(M,B M,C O,B O,C B,C M,O B,E E,O)
	\tkzDrawPoints[fill=black](O,M,B,C,H,E)
	\tkzLabelPoints[left](M)
	\tkzLabelPoints[right](O)
	\tkzLabelPoints[above](B,E)
	\tkzLabelPoints[below](C)
	\tkzLabelPoints[above left](H)
	\tkzMarkRightAngles(M,B,O M,C,O C,B,E)
	\path (O)--(B) node[above,midway,sloped]{$2$cm};
	\path (O)--(M) node[above,midway,sloped]{$3$cm};
	\end{tikzpicture}	
\end{center}
\begin{enumerate}[a)]
	\item Gọi $H$ là giao điểm của $OM$ với $BC$.\\ 
	Ta xét $\triangle OMB$ và $\triangle OMC$ là hai tam giác vuông có\\
	$ \heva{& OM \,\, \text{cạnh chung}\\ &OB = OC = R}\Rightarrow \triangle OMB = \triangle OMC$ (cạnh huyền - cạnh góc vuông).\\
	Suy ra $\heva{&MB = MC\\&\widehat {BMH} = \widehat {CMH}\\&MH \,\, \text{cạnh chung}} \Rightarrow \triangle BMH = \triangle CMH$ (cạnh - góc - cạnh).\\ 
	Suy ra $BH = CH$, mà tam giác $BMC$ cân tại $M$, suy ra $OM \bot BC$.
	\item Do $CE$ đường kính, suy ra $\widehat {CBE} = 90^\circ  \Rightarrow BE \bot BC$.\\
	Mà $MO\perp BC$. Suy ra $BE\parallel MO$.
	\item Do tam giác $MOB$ vuông tại $B$, suy ra $MB^2 = OM^2 - OB^2 = 5 \Rightarrow MB = MC = \sqrt 5 $.\\
	Mặt khác $BH$ là đường cao ứng với cạnh huyền $OM$ của tam giác vuông $MOB$.
	Suy ra $BH\cdot OM = BM\cdot BO \Rightarrow BH = \dfrac {2\sqrt 5 } 3 \Rightarrow BC = 2BH = \dfrac {4\sqrt 5 } 3$.\\
	Vậy tam giác $MBC$  có $MB = MC = \sqrt 5$ cm; $BC = \dfrac{4\sqrt5}3$ cm.   
\end{enumerate}}
\end{bt}

Tính cạnh của một hình vuông, biết diện tích của nó bằng diện tích của hình chữ nhật có chiều rộng $4,5$ m và chiều dài $18$ m.
\loigiai{Gọi cạnh của hình vuông là $x$ (m), điều kiện $x > 0$.\\
Theo giả thiết ta có diện tích hình vuông $S = x^2 = 2\cdot \left(4{,}5 + 18 \right) = 45$.\\
Từ đó suy ra $x =  \pm \sqrt {45}  =  \pm 3\sqrt 5 $, do $x > 0$, suy ra $x = 3\sqrt 5 $.\\
Vậy $x = 3\sqrt 5\approx 6{,}7$ m.}
\end{bt}
