\begin{bt}%[Phạm An Bình]%[EX-TS10-ToanTT-2019]%[9H2B5]
	\immini{
		Một người ở ngọn hải đăng $A$ cao $40$ m so với mặt nước biển quan sát thấy đỉnh $B$ của chiếc tàu ở phía đường chân trời. Tính khoảng cách $AB$ từ ngọn hải đăng đến chiếc tàu, biết bán kính trái đất là $6370$ km và đỉnh tàu B cao $20$ m so với mặt nước biển (làm tròn đến km).
	}{
		\begin{tikzpicture}[scale=1, font=\footnotesize, line join=round, line cap=round, >=stealth]
		\tkzDefPoints{0/0/O, 0/1/M, -1/1/A, .5/1/B}
		
		\tkzDrawCircle(O,M)
		\tkzDrawPolygon(O,A,B)
		\tkzDrawSegments(O,M)
		\tkzDrawPoints[fill=black](O,M,A,B)
		\tkzLabelPoints[above](M,A,B)
		\tkzLabelPoints[below](O)
		\tkzMarkRightAngles(A,M,O)
		\end{tikzpicture}
	}
	\dapso{$39$ km}
	\loigiai{
		Khoảng cách từ $A$ đến tâm trái đất là $OA=6370+0{,}04=6370{,}04$ km.\\
		Khoảng cách từ $B$ đến tâm trái đất là $OB=6370+0{,}02=6370{,}02$ km.\\
		$\triangle AOM$ vuông tại $M$ có $AM^2+OM^2=OA^2\Rightarrow AM\approx 22{,}57$ km.\\
		$\triangle BOM$ vuông tại $M$ có $BM^2+OM^2=OB^2\Rightarrow BM\approx 15{,}96$ km.\\
		Vậy $AB=AM+BM\approx 39$ km.
	}
\end{bt}
