\begin{bt}%[Đề HK1, Hai Bà Trưng, Hồ Chí Minh 17-18]%[Huỳnh Thị Sâm, 10EX-HK1]%[0D3B2-4][0D3B2-6]
Giải các phương trình sau.
\begin{enumerate}
\item $ \sqrt{2x^2 + 3x - 5} = x+ 1. $
\item $|x^2 - 4x + 5| = 4x - 1.$
\item $ x^2 + \sqrt{x^2 -x} -9 = x+ 3. $
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $ \sqrt{2x^2 + 3x - 5} = x+ 1 \Leftrightarrow \heva {&x+1\ge 0\\&2x^2+3x -5 =(x+1)^2} \Leftrightarrow \heva{&x\ge -1\\& x^2 +x -6 = 0} \Leftrightarrow \heva{&x\ge -1\\& \hoac{&x=2 \quad \text{(nhận)}\\&x=-3 \quad \text{(loại)}.}} $\\
	Vậy $ 	\mathbf{S} =\left\{ 2 \right\}. $
	\item $|x^2 - 4x + 5| = 4x - 1 \Leftrightarrow \heva{&4x-1\ge 0\\&\hoac{&x^2 -4x +5 = 4x -1\\& x^2 -4x +5 = -4x + 1}} \Leftrightarrow \heva{&x\ge \dfrac{1}{4}\\& \hoac{&x^2 -8x +6 = 0\\& x^2 +4 = 0 \quad (VN)}} \Leftrightarrow \heva{&x\ge \dfrac{1}{4}\\& \hoac{&x = 4+ \sqrt{10}\quad \text{(nhận)}\\&x = 4-\sqrt{10}\quad \text{(nhận).}}} $\\
	Vậy $ 	\mathbf{S} =\left\{4+ \sqrt{10}; \ 4 - \sqrt{10} \right\}. $
	\item $ x^2 + \sqrt{x^2 -x} -9 = x+ 3 \quad (1)$\\
	Điều kiện: $ x^2 -x \ge 0 \Leftrightarrow \hoac{&x \le 0\\&x\ge 1}. $\\
Ta có	$ (1)\Leftrightarrow x^2 - x + \sqrt{x^2 -x} - 12 = 0. \quad (2)$\\
	Đặt $ t = \sqrt{x^2 -x} $, $ t\ge 0 $ khi đó $(2)  $ trở thành\\
	$ t^2 + t - 12 = 0 \Leftrightarrow \hoac{&t=3 \quad \text{(nhận)}\\&t=-4\quad \text{(loại)}.}   $\\
	Với $ t= 3  $ thì $ \sqrt{x^2 - x} = 3 \Leftrightarrow x^2 - x -9 = 0 \Leftrightarrow \hoac{&x= \dfrac{1+\sqrt{37}}{2}\\&x=\dfrac{1-\sqrt{37}}{2}.} $\\
		Vậy $ 	\mathbf{S} =\left\{\dfrac{1+\sqrt{37}}{2}; \dfrac{1-\sqrt{37}}{2} \right\}. $
\end{enumerate}

}
\end{bt}

\begin{bt}%[Đề HK1, Hai Bà Trưng, Hồ Chí Minh 17-18]%[Huỳnh Thị Sâm, 10EX-HK1]%[0D3K2-5]
Cho phương trình: $ mx^2 - 2(m+1)x + m + 3 = 0 $. Tìm $ m $ để phương trình có hai nghiệm $ x_1; x_2 $ thỏa $ \dfrac{1}{x_1} + \dfrac{1}{x_2} = 4. $
	
	\loigiai{ Để phương trình đã cho có 2 nghiệm phân biệt $ x_1;x_2 $ thì $ \heva{&m \ne 0\\& \Delta' > 0} \Leftrightarrow \heva{& m \ne 0\\ &[-(m+1)]^2 - m(m+3) > 0}\\ \Leftrightarrow \heva{&m \ne 0 \\ &-m+ 1> 0} \Leftrightarrow \heva{&m\ne 0\\& m< 1.} \quad (*)   $\\
		Theo định lí Vi-ét ta có $ \heva{&x_1+x_2 = \dfrac{2(m+1)}{m}\\& x_1 x_2 = \dfrac{m+3}{m}.} $\\
		Theo đề ta có $ \dfrac{1}{x_1} + \dfrac{1}{x_2} = 4 \Leftrightarrow \dfrac{x_1+x_2}{x_1\cdot x_2} = 4 \Rightarrow \dfrac{\dfrac{2(m+1)}{m}}{\dfrac{m+3}{m}}= 4 \Leftrightarrow \dfrac{2(m+1)}{m+3}= 4 \Leftrightarrow m = -5  \quad \text{(nhận).} $\\
		Vậy  với $ m = -5 $ thì phương trình đã cho có hai nghiệm $ x_1; \ x_2$ thỏa $ \dfrac{1}{x_1} + \dfrac{1}{x_2} = 4. $
	
	}
\end{bt}



\begin{bt}%[Đề HK1, Hai Bà Trưng, Hồ Chí Minh 17-18]%[Huỳnh Thị Sâm, 10EX-HK1]%[0D3K3-3]
Giải hệ phương trình sau: $ \heva{&2(x+y)^2 - xy - 1 = 0\\& x^2y + xy^2 = 0.} $
	
	\loigiai{ Ta có: $ \heva{&2(x+y)^2 -xy -1 = 0\\&x^2y + xy^2 =0}\Leftrightarrow \heva{& 2(x+y)^2 - xy -1 = 0\\&xy(x+y) = 0.}$\\
		Đặt $ \heva{&S = x+ y \\& P =xy.} $\\ Điều kiện: $ S^2 -4P\ge 0. $\\
		Khi đó hệ phương trình trên trở thành: $ \heva{&2S^2 -P -1 = 0\\& SP = 0} \Leftrightarrow \heva{&2S^2 -P-1 = 0\\&\hoac{&S=0\\&P=0.}}$ %\Leftrightarrow \hoac{&\heva{&P=0\\&S=\dfrac{1}{\sqrt2}}\\&\heva{&P=0\\&S=-\dfrac{1}{\sqrt2}}\\&\heva{&P=-1\\&S=0.}} $
	\begin{itemize}
		\item Trường hợp 1: $ \heva{&2S^2 - P -1 = 0\\& S=0} \Leftrightarrow \heva {&-P-1 =0\\&S=0} \Leftrightarrow \heva{&P = -1\\&S=0.}$	
		\begin{itemize}
		\item  Với $ \heva{&P=-1\\&S=0} $, khi đó $ (x;y) $ là nghiệm của phương trình \\
		$ X^2 -SX + P = 0  \Leftrightarrow X^2 -0X -1 = 0 \Leftrightarrow \hoac{&X=1\\&X=-1.} $\\
		Suy ra $ \heva{&x=1 \\&y=-1} \quad \text {hoặc}\quad \heva{&x=-1\\& y = 1.}$
		\end{itemize}
		\item Trường hợp 2: $ \heva{&2S^2 -P-1=0\\&P=0} \Leftrightarrow \heva{&2S^2 -1=0\\&P=0} \Leftrightarrow \heva{&S=\dfrac{1}{\sqrt2}\\&P=0} \quad \text{hoặc} \quad \heva{&S= - \dfrac{1}{\sqrt 2}\\&P=0.} $
		\begin{itemize}
		\item 	Với $ \heva{&P=0\\&S=\dfrac{1}{\sqrt2}} $, khi đó $ (x;y) $ là nghiệm của phương trình \\
		$ X^2 -SX + P = 0  \Leftrightarrow X^2 - \dfrac{1}{\sqrt 2} =0 \Leftrightarrow \hoac{&X = \dfrac{\sqrt2}{2}\\&X= 0.}$\\
		Suy ra $ \heva{&x=\dfrac{\sqrt2}{2} \\&y=0} \quad \text {hoặc}\quad \heva{&x=0\\& y = \dfrac{\sqrt2}{2}.}$
		\item 	Với $ \heva{&P=0\\&S=-\dfrac{1}{\sqrt2}} $, khi đó $ (x;y) $ là nghiệm của phương trình \\
		$ X^2 -SX + P = 0  \Leftrightarrow X^2 + \dfrac{1}{\sqrt 2} = 0 \Leftrightarrow \hoac{&X= -\dfrac{\sqrt2}{2} \\& X = 0.}$\\
		Suy ra $ \heva{&x=-\dfrac{\sqrt2}{2} \\&y=0} \quad \text{hoặc} \quad \heva{&x=0\\& y = -\dfrac{\sqrt2}{2}.}$	
		\end{itemize}
		
	\end{itemize}	
	Vậy hệ phương trình đã cho có các nghiệm là:\\ $ \mathbf{S} = \left\{ {\left( {\dfrac{{\sqrt 2 }}{2};0} \right),\left( { - \dfrac{{\sqrt 2 }}{2};0} \right),\left( {0;\dfrac{{\sqrt 2 }}{2}} \right),\left( {0; - \dfrac{{\sqrt 2 }}{2}} \right),\left( {1; - 1} \right),\left( { - 1;1} \right)} \right\}.$
		
	}
\end{bt}
