\begin{bt}%[Đề HK1, Ngô Gia Tự, Hồ Chí Minh 17-18]%[Phạm An Bình]%[0D2B1-2]
	Tìm tập xác định của hàm số $f(x)=\sqrt{3-x} - \dfrac{2}{\sqrt{x-1}}$.
	\loigiai{
		Điều kiện $\heva{&3-x \ge 0\\&\sqrt{x-1} \ne 0\\&x-1 \ge 0}
		\Leftrightarrow \heva{&x \le 3\\&x-1 \ne 0\\&x-1 \ge 0}
		\Leftrightarrow \heva{&x \le 3\\&x-1 > 0}
		\Leftrightarrow \heva{&x \le 3\\&x > 1}
		\Leftrightarrow 1 < x \le 3$.\\
		Vậy tập xác định của hàm số là $\mathscr{D} = \left( 1;3 \right]$.
	}
\end{bt}

\begin{bt}%[Đề HK1, Ngô Gia Tự, Hồ Chí Minh 17-18]%[Phạm An Bình]%[0D3B2-3]
Giải phương trình $\dfrac{2x+5}{2x} - \dfrac{x}{x+5} = 1$.
	\loigiai{
Ta có phương trình $\dfrac{2x+5}{2x} - \dfrac{x}{x+5} = 1$.\\
			Điều kiện $\heva{&2x\ne 0\\&x+5\ne 0}\Leftrightarrow \heva{&x\ne 0\\&x\ne -5.}$\\
			Với điều kiện trên, phương trình trở thành
			\begin{eqnarray*}
				& &\dfrac{(2x+5)(x+5)-2x^2}{2x(x+5)} = \dfrac{2x(x+5)}{2x(x+5)}\\
				&\Leftrightarrow &(2x+5)(x+5)-2x^2 = 2x(x+5)\\
				&\Leftrightarrow &2x^2+10x+5x+25-2x^2 = 2x^2+10x\\
				&\Leftrightarrow &-2x^2+5x+25 = 0\\
				&\Leftrightarrow &\hoac{&x=5&\text{(nhận)}\\ &x=-\dfrac{5}{2}&\text{(nhận)}.}
			\end{eqnarray*}
			Vậy tập nghiệm của phương trình là $S=\left\lbrace -\dfrac{5}{2};5 \right\rbrace$.
	}
\end{bt}

\begin{bt}%[Đề HK1, Ngô Gia Tự, Hồ Chí Minh 17-18]%[Phạm An Bình]%[0D3B2-4]
Giải phương trình $\sqrt{2x^2-9x-10} = x-2$.
	\loigiai{
			\begin{eqnarray*}
				\sqrt{2x^2-9x-10} = x-2
				&\Leftrightarrow &\heva{&x-2\ge 0\\&2x^2-9x-10=(x-2)^2}\\
				&\Leftrightarrow &\heva{&x\ge 2\\&2x^2-9x-10=x^2-4x+4}\\
				&\Leftrightarrow &\heva{&x\ge 2\\&x^2-5x-14=0}\\
				&\Leftrightarrow &\heva{&x\ge 2\\&\hoac{&x=-2\\&x=7}}\\
				&\Leftrightarrow &x=7.
			\end{eqnarray*}
			Vậy tập nghiệm của phương trình là $S=\left\lbrace 7 \right\rbrace$.
	}
\end{bt}

\begin{bt}%[Đề HK1, Ngô Gia Tự, Hồ Chí Minh 17-18]%[Phạm An Bình]%[0D3K2-4]
Giải phương trình $2\sqrt{x+1} - \sqrt{x-2} = 3$.
\loigiai{
$2\sqrt{x+1} - \sqrt{x-2} = 3$.\\
Điều kiện $\heva{&x+1\ge 0\\&x-2\ge 0}\Leftrightarrow \heva{&x\ge -1\\&x\ge 2} \Leftrightarrow x\ge 2$.\\
			Với điều kiện trên, phương trình trở thành
			\begin{eqnarray*}
				& &2\sqrt{x+1} = \sqrt{x-2} +3\\
				&\Leftrightarrow &4(x+1)=x-2+6\sqrt{x-2} +9\\
				&\Leftrightarrow &4x+4=x-2+6\sqrt{x-2} +9\\
				&\Leftrightarrow &6\sqrt{x-2} =3x-3\\
				&\Leftrightarrow &2\sqrt{x-2} =x-1\\
				&\Leftrightarrow &\heva{&x-1\ge 0\\&4(x-2)=(x-1)^2}\\
				&\Leftrightarrow &\heva{&x\ge 1\\&4x-8=x^2-2x+1}\\
				&\Leftrightarrow &\heva{&x\ge 1\\&x^2-6x+9=0}\\
				&\Leftrightarrow &\heva{&x\ge 1\\&x=3}\\
				&\Leftrightarrow &x=3 \text{ (nhận vì $x\ge 2$)}.
			\end{eqnarray*}
			Vậy tập nghiệm của phương trình là $S=\left\lbrace 3 \right\rbrace$.
	}
\end{bt}

\begin{bt}%[Đề HK1, Ngô Gia Tự, Hồ Chí Minh 17-18]%[Phạm An Bình]%[0D3B2-6]
	Giải và biện luận theo tham số $m$ số nghiệm phương trình $(2m+1)x - 2m = 3x-2$.
	\loigiai{
		Ta có
		\begin{eqnarray*}
			& &(2m+1)x-2m=3x-2\\
			&\Leftrightarrow &(2m+1)x-3x-2m+2=0\\
			&\Leftrightarrow &(2m-2)x=2m-2 \quad (1)
		\end{eqnarray*}
		\begin{enumerate}[TH 1.]
			\item $2m-2=0\Leftrightarrow m=1$.\\
			Phương trình $(1)$ trở thành
			\begin{eqnarray*}
				& &(2\cdot 1-2)x=2\cdot 1-2\\
				&\Leftrightarrow &0=0 \text{ (luôn đúng)}.
			\end{eqnarray*}
			Vậy phương trình $(1)$ vô số nghiệm.
			
			\item $2m-2\ne 0\Leftrightarrow m\ne 1$.\\
			Phương trình $(1)$ trở thành
			\begin{eqnarray*}
				& &x=\dfrac{2m-2}{2m-2}\\
				&\Leftrightarrow &x=1.
			\end{eqnarray*}
		\end{enumerate}
		Vậy
		\begin{itemize}
			\item với $m=1$ thì phương trình $(1)$ có vô số nghiệm
			\item với $m\ne 1$ thì phương trình $(1)$ có nghiệm duy nhất $x=1$.
		\end{itemize}
	}
\end{bt}

\begin{bt}%[Đề HK1, Ngô Gia Tự, Hồ Chí Minh 17-18]%[Phạm An Bình]%[0D3K2-5]
	Cho phương trình $x^2 -2(m+1)x + m^2 + 3 = 0. \quad (1)$\\
	Tìm tất cả giá trị của tham số $m$ để phương trình $(1)$ có hai nghiệm phân biệt $x_1$ và $x_2$ thỏa điều kiện $x_1^2 + x_2^2 = 2x_1x_2+8$.
	\loigiai{
		$\Delta '=b'^2-ac=\left[-\left(m+1\right)\right]^2 -(1)(m^2+3)=m^2+2m+1-m^2-3=2m-2$.\\
		Để phương trình $(1)$ có hai nghiệm phân biệt thì $\heva{&a\ne 0\\&\Delta>0} \Leftrightarrow \heva{&1\ne 0\\&2m-2>0} \Leftrightarrow m>1$.\\
		Với điều kiện $m>1$, áp dụng định lí Vi-ét ta có
		$$\heva{&x_1+x_2=-\dfrac{b}{a}=-\dfrac{-2\left(m+1\right)}{1}=2m+2\\&x_1x_2=\dfrac{c}{a}=\dfrac{m^2+3}{1}=m^2+3.}$$
		Ta có
		\begin{eqnarray*}
			& &x_1^2 + x_2^2 = 2x_1x_2+8\\
			&\Leftrightarrow &x_1^2 +2x_1x_2 + x_2^2 = 4x_1x_2+8\\
			&\Leftrightarrow &\left(x_1+x_2\right)^2 = 4x_1x_2+8.
		\end{eqnarray*}
		Thay $\heva{&x_1+x_2=2m+2\\&x_1x_2=m^2+3}$ vào phương trình trên ta được
		\begin{eqnarray*}
			& &\left(2m+2\right)^2 = 4\left(m^2+3\right)+8\\
			&\Leftrightarrow &4m^2+8m+4 = 4m^2+12+8\\
			&\Leftrightarrow &8m = 16\\
			&\Leftrightarrow &m=2. \text{ (nhận)}
		\end{eqnarray*}
		Vậy $m=2$ thỏa đề bài.
	}
\end{bt}

			Ta có $\vec{AH} \cdot \vec{AB}=\left| \vec{AH} \right| \left| \vec{AB} \right| \cos \left( \vec{AH};\vec{AB} \right)=\dfrac{3a^2}{4}$.

\begin{bt}%[Đề HK1, Ngô Gia Tự, Hồ Chí Minh 17-18]%[Phạm An Bình]%[0H2B3-3]%[0H2B3-1]%[0H2B2-1]%[0H2K2-4]
	Trong mặt phẳng $Oxy$, cho ba điểm $A(-2;4)$; $B(4;1)$; $C(1;-2)$.
	\begin{enumerate}
		\item Chứng minh $ABC$ là tam giác cân. Tính diện tích tam giác $ABC$.
		\item Tính $\cos$ góc $C$ của tam giác $ABC$.
		\item Tìm tọa độ điểm $Q$ khác điểm $A$, $B$ sao cho $A$, $B$, $Q$ thẳng hàng và $QA=3\sqrt{5}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Chứng minh $ABC$ là tam giác cân.
			\immini{
				$\vec{AB}=(x_B-x_A;y_B-y_A)=(6;-3)$\\
				$\Rightarrow AB=\sqrt{(6)^2+(-3)^2}=3\sqrt{5}$.\\
				$\vec{AC}=(x_C-x_A;y_C-y_A)=(3;-6)$\\
				$\Rightarrow AC=\sqrt{(3)^2+(-6)^2}=3\sqrt{5}$.\\
				$\vec{BC}=(x_C-x_B;y_C-y_B)=(-3;-3)$\\
				$\Rightarrow BC=\sqrt{(-3)^2+(-3)^2}=3\sqrt{2}$.\\
				Do $\dfrac{6}{3} \ne \dfrac{-3}{-6}$ nên $\vec{AB}$ và $\vec{AC}$ không cùng phương. Mặt khác $AB=AC$ ($=3\sqrt{5}$)	nên $\triangle ABC$ cân tại $A$.\\
				Tính diện tích tam giác $ABC$.\\
				Gọi $H$ là trung điểm của $BC$\\
				$\Rightarrow \heva{&x_H=\dfrac{x_B+x_C}{2}=\dfrac{5}{2}\\&y_H=\dfrac{y_B+y_C}{2}=\dfrac{-1}{2}.}$\\
				$AH=\sqrt{(x_H-x_A)^2+(y_H-y_A)^2}=\dfrac{9\sqrt{2}}{2}$.\\
				Do $\triangle ABC$ cân tại $A$ nên đường trung tuyến $AH$ cũng là đường cao 
				$\Rightarrow S_{\triangle ABC}=\dfrac{AH \cdot BC}{2}=\dfrac{27}{2}$.
			}
			{
				\begin{tikzpicture}[scale=0.5]
				\draw (-2,4) -- (4,1) -- (1,-2) -- (-2,4) -- (5/2,-1/2);
				\draw[thin] (2,-5/4) -- (1.5,-5/4);
				\draw[thin] (3,1/4) -- (3.5,1/4);
				\draw[fill=black] (-2,4) circle (1pt) node[above] {$A(-2;4)$} (4,1) circle (1pt) node[right] {$B(4;1)$} (1,-2) circle (1pt) node[left] {$C(1;-2)$} (5/2,-1/2) circle (1pt) node[right] {$H$};
				\end{tikzpicture}
			}
			
			\item Tính $\cos$ góc $C$ của tam giác $ABC$.\\
			$\vec{CA}=-\vec{AC}=(-3;6)$, $\vec{CB}=-\vec{BC}=(3;3)$.\\
			$\cos C=\cos (\vec{CA};\vec{CB})=\dfrac{\vec{CA}\cdot \vec{CB}}{\left|\vec{CA}\right|\cdot \left|\vec{CB}\right|}=\dfrac{-3\cdot 3+6\cdot 3}{3\sqrt{5}\cdot 3\sqrt{2}}=\dfrac{\sqrt{10}}{10}$.
			
			\item Tìm tọa độ điểm $Q$ khác điểm $A$, $B$ sao cho $A$, $B$, $Q$ thẳng hàng và $QA=3\sqrt{5}$.\\
			$\vec{AQ}=(x_Q-x_A;y_Q-y_A)=(x_Q+2;y_Q-4)\Rightarrow AQ=\sqrt{(x_Q+2)^2+(y_Q-4)^2}$.\\
			Ta có $A$, $B$, $Q$ thẳng hàng và $QA=3\sqrt{5}$
			\begin{eqnarray*}
				&\Leftrightarrow &\heva{&\dfrac{x_Q+2}{6}=\dfrac{y_Q-4}{-3}\\&QA=3\sqrt{5}}\\
				&\Leftrightarrow &\heva{&-3x_Q-6=6y_Q-24\\&(x_Q+2)^2+(y_Q-4)^2=45}\\
				&\Leftrightarrow &\heva{&x_Q=6-2y_Q\\&(x_Q+2)^2+(y_Q-4)^2=45}\\ &\Leftrightarrow &\heva{&x_Q=6-2y_Q\\&(8-2y_Q)^2+(y_Q-4)^2=45}\\ &\Leftrightarrow &\heva{&x_Q=6-2y_Q\\&5y_Q^2-40y_Q+35=0}\\
				&\Leftrightarrow &\heva{&x_Q=4\\&y_Q=1} \text{ (loại vì $Q$ phải khác $B$) hay }\heva{&x_Q=-8\\&y_Q=7} \text{ (nhận)}.
			\end{eqnarray*}
			Vậy $Q(-8;7)$.
		\end{enumerate}			
	}
\end{bt}
