\section{SỐ PHẦN TỬ CỦA MỘT TẬP HỢP - TẬP HỢP CON}
\subsection{Tóm tắt lý thuyết}
	\begin{enumerate}
		\item Một tập hợp có thể có một phần tử, có nhiều phần tử, có vô số phần tử, cũng có thể không có phần tử nào.\\
				Tập hợp không có phần tử nào gọi là tập hợp rỗng, kí hiệu là $\varnothing$.
		\item Nếu mọi phần tử của tập hợp $A$ đều thuộc tập hợp $B$ thì tập hợp $A$ gọi là tập hợp con của tập hợp $B$, kí hiệu $A \subset B$ hay $B \supset A$.
				\begin{note}
					Nếu $A \subset B$ và $B \subset A$ thì $A$ và $B$ là hai tập hợp bằng nhau, kí hiệu $A = B$.
				\end{note}
	\end{enumerate}
\subsection{Các dạng toán}
	\begin{dang}{Tìm số phần tử của một tập hợp}
		\begin{itemize}
			\item Tập hợp các số tự nhiên từ $a$ đến $b$ có $(b - a) + 1$ phần tử.
			\item Tập hợp các số chẵn từ số chẵn $a$ đến số chẵn $b$ có $(b - a):2 + 1$ phần tử.
			\item Tập hợp các số lẻ từ số lẻ $a$ đến số lẻ $b$ có $(b - a):2 + 1$ phần tử.
			\item Tập hợp các số tự nhiên từ $a$ đến $b$ mà bất cứ hai số liền nhau nào cũng cách nhau $d$ đơn vị thì có số phần tử là $(b - a):d + 1$.
		\end{itemize}
	\end{dang}
	
\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1K4]
	Tính số phần tử của các tập hợp sau:
	\begin{enumEX}{2}
		\item $A = \{15, 17, 19, \ldots, 49, 51\}$;
		\item $B = \{10, 12, 14, \ldots, 76, 78\}$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Tập hợp $A$ là tập hợp các số lẻ từ $15$ đến $51$ nên số phần tử của tập hợp $A$ là
					$$(51 - 15):2 + 1 = 19 \,\,\mbox{(phần tử)}.$$
			\item Tập hợp $B$ là tập hợp các số chẵn từ $10$ đến $78$ nên số phần tử của tập hợp $B$ là
					$$(78 - 10):2 + 1 = 35 \,\,\mbox{(phần tử)}.$$
		\end{enumEX}
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1K4]

	\loigiai{
		Tập hợp $C$ là tập hợp các số tự nhiên từ $17$ đến $113$, bất cứ hai số liền nhau nào cũng cách nhau $3$ đơn vị nên số phần tử của $C$ là

	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1K4]
	Tính số phần tử của các tập hợp sau:
	\begin{enumEX}{1}
		\item $A$ là tập hợp các số lẻ không vượt quá $46$;
		\item $B$ là tập hợp các số chẵn không vượt quá $46$;
		\item $C$ là tập hợp các số tự nhiên lớn hơn $46$;
		\item $D$ là tập hợp các số tự nhiên lớn hơn $46$ nhưng nhỏ hơn $47$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Tập hợp các số lẻ không vượt quá $46$ là tập hợp $A = \{1, 3, 5, \ldots, 45\}$.\\
					Số phần tử của tập hợp này là $(45 - 1):2 + 1 = 23$ (phần tử).
			\item Tập hợp các số chẵn không vượt quá $46$ là tập hợp $B = \{0, 2, 4, \ldots, 46\}$.\\
					Số phần tử của tập hợp này là $(46 - 0):2 + 1 = 24$ (phần tử).
			\item Tập hợp các số tự nhiên lớn hơn $46$ là tập hợp $C = \{47, 48, 49, \ldots\}$.\\
					Tập hợp này có vô số phần tử.
			\item Không có số tự nhiên nào lớn hơn $46$ nhưng nhỏ hơn $47$, do đó tập hợp $D$ không có phần tử nào.
		\end{enumEX}
	}
\end{vd}

	\begin{dang}{Xác định xem tập hợp $A$ có là tập hợp con của tập hợp $B$ không}
		Xem mọi phần tử của tập hợp $A$ có phải là phần tử của tập hợp $B$ không?

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1Y4]

	\begin{enumEX}{1}
		\item Các tập hợp $A$ và $B$ có phải là tập hợp con của tập hợp $M$ không?
		\item Tập hợp $A$ có phải là tập hợp con của tập hợp $B$ không?
	\end{enumEX}	
	\loigiai{
		\begin{enumEX}{1}
			\item Các phần tử của tập hợp $A$ đều là phần tử của tập hợp $M$ nên $A \subset M$.\\
					Các phần tử của tập hợp $B$ đều là phần tử của tập hợp $M$ nên $B \subset M$.
			\item Ta có $1 \in A$ nhưng $1 \not\in B$ nên tập hợp $A$ không phải là tập hợp con của tập hợp $B$.
		\end{enumEX}
	}
\end{vd}

\section{PHÉP CỘNG VÀ PHÉP NHÂN}
\subsection{Tóm tắt lý thuyết}
\begin{enumerate}
	\item Tổng và tích của hai số tự nhiên
			\begin{center}
				$a + b = c$ (tổng)\\
				$a \cdot b = d$ (tích)
			\end{center}
	\item Tính chất của phép cộng và phép nhân
			\begin{center}
				\begin{tabular}{|m{5cm}|m{5cm}|m{5cm}|}
					\hline
					\begin{center} \textbf{Tính chất} \end{center} & \begin{center} \textbf{Phép tính cộng}  \end{center} & \begin{center} \textbf{Phép tính nhân}	\end{center}\\
					\hline
					Giao hoán & $a + b = b + a$ & $a \cdot b = b \cdot a$\\
					\hline
					Kết hợp & $(a + b) + c = a + (b + c)$ & $(a \cdot b) \cdot c = a \cdot (b \cdot c)$\\
					\hline
					Cộng với số $0$ & $a + 0 = 0 + a = a$ & \\
					\hline
					Nhân với số $1$ & & $a \cdot 1 = 1 \cdot a = a$\\
					\hline
					Phân phối của phép nhân đối với phép cộng & \multicolumn{2}{|c|}{$a \cdot (b + c) = a \cdot b + a \cdot c$}\\
					\hline
				\end{tabular}
			\end{center}
\end{enumerate}
\subsection{Các dạng toán}
	\begin{dang}{Tính nhanh, tính hợp lí bằng cách áp dụng các tính chất của phép cộng và phép nhân}
		\begin{itemize}
			\item Nhận xét, phát hiện đặc điểm của các số hạng, các thừa số.
			\item Áp dụng tính chất của phép cộng và phép nhân để tính toán được nhanh chóng (tổng và tích là một số tròn chục, tròn trăm).
		\end{itemize}
	\end{dang}
	
\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1K5]
	Tính bằng cách hợp lí nhất:
	\begin{enumEX}{2}
		\item $217 + 31 + 46 + 183 + 154$;
		\item $125 \cdot 28 \cdot 8 \cdot 25$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Ta có $217 + 31 + 46 + 183 + 154 = (217 + 183) + (46 + 154) + 31$\\
					\phantom{Ta có $217 + 31 + 46 + 183 + 154$} $= 400 + 200 + 31 = 631$.
			\item Ta có $125 \cdot 28 \cdot 8 \cdot 25 = 125 \cdot 4 \cdot 7 \cdot 8 \cdot 25$\\
					\phantom{Ta có $125 \cdot 28 \cdot 8 \cdot 25$} $= (125 \cdot 8) \cdot (4 \cdot 25) \cdot 7$\\
					\phantom{Ta có $125 \cdot 28 \cdot 8 \cdot 25$} $= 1000 \cdot 100 \cdot 7 = 700000$.
	\end{enumEX}
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1B5]
	Tính nhẩm:
	\begin{enumEX}{3}
		\item $114 + 87$;
		\item $45 \cdot 11$;
		\item $23 \cdot 38 + 23 \cdot 43 + 23 \cdot 19$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Ta có $114 + 87 = 114 + (86 + 1) = (114 + 86) + 1 = 200 + 1 = 201$.
			\item Ta có $45 \cdot 11 = 45 \cdot (10 + 1) = 450 + 45 = 495$.
			\item Ta có $23 \cdot 38 + 23 \cdot 43 + 23 \cdot 19 = 23 \cdot (38 + 43 + 19) = 23 \cdot 100 = 2300$.
		\end{enumEX}
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1K5]
	Cho biết $37 \cdot 3 = 111$, $3003 \cdot 111 = 333333$. Hãy tính:
	\begin{enumEX}{2}
		\item $37 \cdot 21$;
		\item $3003 \cdot 37 \cdot 6$.		
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Ta có $37 \cdot 21 = (37 \cdot 3) \cdot 7 = 111 \cdot 7 = 777$.
			\item Ta có $3003 \cdot 37 \cdot 6 = 3003 \cdot (37 \cdot 3) \cdot 2 = (3003 \cdot 111) \cdot 2 = 333333 \cdot 2 = 666666$.		
		\end{enumEX}
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1G5]
	Tính:
	\begin{enumEX}{2}
		\item $a + 2 \cdot a + 3 \cdot a$;
		\item $\overline{ab} \cdot 101$.		
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Ta có $a + 2 \cdot a + 3 \cdot a = a \cdot (1 + 2 + 3) = a \cdot 6 = 6 \cdot a$.
			\item Ta có $\overline{ab} \cdot 101 = \overline{ab} \cdot (100 + 1) = \overline{ab00} + \overline{ab} = \overline{abab}$.		
		\end{enumEX}
	}
\end{vd}

	\begin{dang}{Tìm số chưa biết trong một đẳng thức}
		\begin{itemize}
			\item Mỗi số hạng của một tổng bằng tổng trừ đi số hạng đã biết.
			\item Mỗi thừa số của một tích bằng tích chia cho thừa số đã biết.
			\item Nếu một tích của hai thừa số mà bằng $0$ thì ít nhất có một thừa số bằng $0$:\\
					\begin{center}
						Nếu $a \cdot b = 0$ thì $a = 0$ hoặc $b = 0$.
					\end{center}
		\end{itemize}
	\end{dang}
	
\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1K5]
	Tìm số tự nhiên $x$, biết:
	\begin{enumEX}{2}
		\item $33 \cdot x + 135 = 26 \cdot 9$;
		\item $108 \cdot (x - 43) = 0$.		
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Ta có $33 \cdot x + 135 = 26 \cdot 9$\\
					\phantom{Ta có} $33 \cdot x = 234 - 135 = 99$\\
					\phantom{Ta có} $x = 99 : 33 = 3$.
			\item $108 \cdot (x - 43) = 0$\\
					Vì $108 \neq 0$ nên $x - 43 = 0$\\
					Do đó $x = 0 + 43 = 43$.	
		\end{enumEX}
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1K5]
	Trung bình cộng của hai số là $75$. Biết một số là số lẻ nhỏ nhất có ba chữ số. Tìm số kia.
	\loigiai{
		Tổng của hai số là $75 \cdot 2 = 150$.\\
		Số lẻ nhỏ nhất có ba chữ số là $101$.\\
		Số kia là $150 - 101 = 49$.
	}
\end{vd}

	\begin{dang}{So sánh hai tổng hoặc hai tích mà không tính giá trị cụ thể của chúng}
		\begin{itemize}
			\item Phát hiện đặc điểm của các số hạng, các thừa số trong hai tổng hoặc hai tích.
			\item Dựa vào tính chất của phép cộng và phép nhân để rút ra kết luận.
		\end{itemize}
	\end{dang}
	
\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1K5]
	So sánh hai tổng $576 + 429$ và $729 + 276$ mà không tính giá trị cụ thể của chúng.
	\loigiai{
		Ta có $576 + 429 = (500 + 76) + (229 + 200)$\\
		\phantom{Ta có $576 + 429$} $= (500 + 229) + (76 + 200)$\\
		\phantom{Ta có $576 + 429$} $= 729 + 276$.\\
		Vậy hai tổng trên bằng nhau.
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1G5]
	So sánh hai tích $A = 200 \cdot 200$ và $B = 199 \cdot 201$ mà không tính giá trị cụ thể của chúng.
	\loigiai{
		Ta có $A = 200 \cdot 200 = 200 \cdot (199 + 1) = 200 \cdot 199 + 200$ \hfill $(1)$\\
		\phantom{Ta có} $B = 199 \cdot 201 = 199 \cdot (200 + 1) = 199 \cdot 200 + 199$ \hfill $(2)$\\
		Vì $200 > 199$ nên từ $(1)$ \& $(2)$ suy ra $A > B$.
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1K5]
	Cho $a$, $b$, $c$ là ba số tự nhiên. Biết $a + 5 = b + 7 = c + 10$, hãy so sánh $a$, $b$, $c$.
	\loigiai{
		Ta có $a + 5 = b + 7$ mà $5 < 7$ nên $a > b$ \hfill $(1)$\\
		\phantom{Ta có} $b + 7 = c + 10$ mà $7 < 10$ nên $b > c$ \hfill $(2)$\\
		Từ $(1)$ \& $(2)$ suy ra $a > b > c$.
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1K5]
	Tích $2 \cdot 7 \cdot 125$ không bằng tích nào trong các tích dưới đây?
	\choice
	{$10 \cdot 5 \cdot 35$}
	{$50 \cdot 5 \cdot 7$}
	{\True $2 \cdot 120 \cdot 7$}
	{$2 \cdot 25 \cdot 35$}
	\loigiai{
		Vì $125 > 120$ nên $2 \cdot 7 \cdot 125 > 2 \cdot 120 \cdot 7$.
	}
\end{vd}

	\begin{dang}{Tính tổng các số hạng của một dãy các số tự nhiên mà bất cứ hai số liền nhau nào cũng cách nhau $d$ đơn vị}
		\begin{itemize}
			\item Xét xem tổng đó có bao nhiêu số hạng.
			\item Lấy số hạng đầu cộng số hạng cuối nhân với số các số hạng rồi chia cho $2$.
		\end{itemize}
	\end{dang}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1G5]
	Tính tổng $S = 1 + 3 + 5 + \cdots + 97 + 99$.
	\loigiai{
		Các số hạng của tổng là dãy số tự nhiên viết theo quy luật: hai số liền nhau cách nhau $2$ đơn vị.\\
		Số các số hạng của tổng $S$ là $(99 - 1):2 + 1 = 50$ (số hạng).\\
		Do đó $S = \dfrac{(1 + 99) \cdot 50}{2} = 2500$.
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1G5]
	Tính tổng tất cả các số chẵn có hai chữ số.
	\loigiai{
		Các số chẵn có hai chữ số là $10$, $12$, $14$, $\ldots$, $96$, $98$.\\
		Số các số hạng là $(98 - 10):2 + 1 = 45$ (số hạng).\\
		Tổng của chúng là $S = 10 + 12 + 14 + \cdots + 98 = \dfrac{(10 + 98) \cdot 45}{2} = 2430$.
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1K5]
	Cho biết $4 \cdot 44 < x < 4 + 44 + 144$ trong đó $x \in \mathbb{N}$. Tính tổng tất cả các giá trị của $x$.
	\loigiai{
		Ta có $4 \cdot 44 = 176$; $4 + 44 + 144 = 192$.\\
		Do đó $176 < x < 192$.\\
		Vì $x \in \mathbb{N}$ nên $x \in \{177, 178, 179, \ldots, 191\}$.\\
		Số các giá trị của $x$ là $191 - 177 + 1 = 15$.\\
		Tổng cần tìm là $S = \dfrac{(177 + 191) \cdot 15}{2} = 2760$.
	}
\end{vd}

\subsection{Bài tập tự luyện}
\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1K5]
	Tính bằng cách hợp lí nhất:
	\begin{enumEX}{1}
		\item $42 + 37 + 135 + 58 + 63$;
		\item $25 \cdot 17 \cdot 8 \cdot 4 \cdot 125$;
		\item $36 \cdot 23 + 62 \cdot 23 + 46$.		
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item $42 + 37 + 135 + 58 + 63 = (42 + 58) + (37 + 63) + 135 = 100 + 100 + 135 = 335$;
			\item $25 \cdot 17 \cdot 8 \cdot 4 \cdot 125 = 17 \cdot (25 \cdot 4) \cdot (8 \cdot 125) = 17 \cdot 100 \cdot 1000 = 1700000$;
			\item $36 \cdot 23 + 62 \cdot 23 + 46 = (36 + 62 + 2) \cdot 23 = 100 \cdot 23 = 2300$.		
		\end{enumEX}
	}
\end{bt}

\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1G5]
	So sánh các tích sau mà không tính giá trị cụ thể của chúng: $A = 98 \cdot 102$, $B = 100 \cdot 100$.
	\loigiai{
		Ta có $A = 98 \cdot 102 = 98 \cdot (100 + 2) = 98 \cdot 100 + 98 \cdot 2$ \hfill $(1)$\\
		\phantom{Ta có} $B = 100 \cdot 100 = (98 + 2) \cdot 100 = 98 \cdot 100 + 100 \cdot 2$ \hfill $(2)$\\
		Từ $(1)$ và $(2)$ suy ra $A < B$.
	}
\end{bt}

\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1K5]
	Tính giá trị của biểu thức sau với $a + b = 41$: $M = 73 \cdot a + 27 \cdot a + 100 \cdot b$.
	\loigiai{
		Biến đổi $M = (73 + 27) \cdot a + 100 \cdot b = 100 \cdot a + 100 \cdot b = 100 \cdot (a + b) = 100 \cdot 41 = 4100$.
	}
\end{bt}

\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1K5]
	Tìm $x$, biết:
	\begin{enumEX}{2}
		\item $47 \cdot x + 213 = 3 \cdot 7 \cdot 37$;
		\item $(83 + 519) \cdot x = 66 \cdot 100 + 22$.		
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Ta có $47 \cdot x + 213 = 3 \cdot 7 \cdot 37 = 777$\\
					\phantom{Ta có} $47 \cdot x = 777 - 213 = 564$\\
					\phantom{Ta có} $x = 564 : 47 = 12$.
			\item Ta có $(83 + 519) \cdot x = 66 \cdot 100 + 22$\\
					\phantom{Ta có} $602 \cdot x = 6600 + 22 = 6622$\\
					\phantom{Ta có} $x = 6622 : 602 = 11$.
		\end{enumEX}
	}
\end{bt}

\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1G5]
	Tính tổng của tất cả các số có ba chữ số mà cả ba chữ số này giống nhau.
	\loigiai{
		$S = 111 + 222 + 333 + \cdots + 999 = \dfrac{(999 + 111) \cdot 9}{2} = 4995$.
	}
\end{bt}

\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-Lớp 6-2018]%[6D1G5]
	Cho $M$ là một số có hai chữ số. Chữ số hàng chục gấp $6$ lần chữ số hàng đơn vị. $N$ là một số có hai chữ số, giá trị của số này gấp $3$ lần chữ số hàng đơn vị của nó. Tính tổng $M + N$.
	\loigiai{
		Ta có $M$ là một số có hai chữ số mà chữ số hàng chục gấp $6$ lần chữ số hàng đơn vị nên ta có $M = 61$.\\
		Lại có $N$ là một số có hai chữ số mà giá trị của số này gấp $3$ lần chữ số hàng đơn vị của nó nên $N = 15$.\\
		Do đó, ta có $M + N = 61 + 15 = 76$.
	}
\end{bt}
