\begin{bt}%[Nguyễn Cao Cường]%[EX-TS10-ToanTT-2019]%[9D1T2]
Bạn Bảo gửi $8$ triệu theo định kì $1$ năm và sau $2$ năm bạn nhận được $9159200$VNĐ.
\begin{enumerate}
	\item Tính lãi suất ngân hàng, biết rằng tiền lãi nhập vào tiền vốn.
	\dapso{$7\%$}
	\item Nếu bạn Bảo gửi theo định kì $4$ tháng với lãi suất $2{,}4\%$/kì và rút tiền sau mỗi kì thì sau $1$ năm, bạn nhận được cả vốn lẫn lãi là bao nhiêu?
	\dapso{$8576000$ đồng}
\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Áp dụng công thức: Khách hàng gửi vào ngân hàng số tiền là $A$ với lãi kép $r\%$ mỗi kì hạn thì số tiền khách hàng nhận được cả vốn lẫn lãi sau $n$ kì hạn là $S=A(1+r\%)^n$.\\
		(\lq \lq Lãi kép: Tiền lãi của kì hạn trước nếu người gửi không rút ra thì được cộng vào vốn để tính lãi cho kì hạn sau\rq \rq.)\\
		Gọi $x$ là lãi suất cần tìm, $2$ năm là $2$ kì hạn nên ta có phương trình
		\begin{eqnarray*}
		&&8000000(1+x)^2=9159200 \\
		&\Leftrightarrow& (1+x)^2=\dfrac{9159200}{8000000} \\
		&\Leftrightarrow& ( 1+x)^2=1{,}1449\\
			&\Leftrightarrow& 1+x=\sqrt{1{,}1449}\\
			&\Leftrightarrow& 1+x=1{,}07\\
				&\Leftrightarrow&  x=0{,}07=7\%.
		\end{eqnarray*}		
		\item Áp dụng công thức: Khách hàng gửi vào ngân hàng số tiền là $A$ với lãi suất đơn $r\%$ mỗi kì hạn thì số tiền khách hàng nhận được cả vốn lẫn lãi sau $n$ kì hạn là $S=A(1+n\cdot r\%)$.\\
		(\lq \lq Lãi suất đơn: Là số tiền lãi chỉ tính trên số tiền gốc mà không tính trên số tiền lãi do số tiền gốc sinh ra\rq \rq.)\\
		Theo định kì $4$ tháng thì một năm có $3$ kì, do đó số tiền bạn nhận được sau $1$ năm là
		$8000000(1+3\cdot2{,}4\%) =8576000$ đồng.
	\end{enumerate}
	
}
\end{bt}

\begin{bt}%[Tô Ngọc Thy]%[EX-TS10-ToanTT-2019]%[9D1G4]
	\immini{Hai người cùng xuất phát một lúc từ hai địa điểm $A$ và $B$ để đến chỗ làm ở $D$, người thứ $I$ ở $A$ đi trên đường thẳng và đi bộ với vận tốc $5$ km/giờ, người thứ $II$ đi bằng xe đạp lúc đầu đi trên đoạn đường thẳng $BC$ song song với $AD$, sau đó đi đoạn đường $CD$ với cùng vận tốc $12$ km/giờ và đến $D$ trước $4$ phút $24$ giây. Tính đoạn đường $CD$ biết rằng $AB$ vuông góc với $AD$, $AB=300$ m và $AD=700$ m.
	\dapso{$CD=500$ m}}
	{\begin{tikzpicture}[scale=0.9, font=\footnotesize, line join=round, line join=round, line cap = round,>=stealth]
		\tkzInit[ymin=-4,ymax=1,xmin=-1,xmax=8]
		\tkzClip
		\tkzDefPoints{0/0/A, 3/0/H, 7/0/D, 0/-3/B, 3/-3/C}	
		\tkzDrawSegments(A,B B,C C,D D,A C,H)
		\tkzLabelPoints[above](A,H,D)
		\tkzLabelPoints[below](B,C)
		\tkzDrawPoints(A,B,C,D,H)
		\tkzMarkRightAngle[size=.5](B,A,H)
		\tkzMarkRightAngle[size=.5](A,H,C)
		\end{tikzpicture}}
	\loigiai{
		Đổi  $4$ phút $24$ giây $=\dfrac{11}{150}$ h.\\
		Thời gian người thứ $I$ đi từ $A$ đến $D$ là $$0{,}7:5=\dfrac{7}{50} \text{ (h).}$$
		Gọi quãng đường từ $C$ đến $D$ là $x$, ($x>0{,}3$).\\
		Thời gian người thứ $II$ đi từ $A$ đến $D$ là	
		\begin{eqnarray*}
		&\dfrac{BC+CD}{12}=\dfrac{7}{50}-\dfrac{11}{150}\\
		\Leftrightarrow & \dfrac{0{,}7-\sqrt{x^2-0{,}09}+x}{12}=\dfrac{1}{15}\\
		\Leftrightarrow & 0{,}7-\sqrt{x^2-0{,}09}+x=0{,}8\\
		\Leftrightarrow & \sqrt{x^2-0{,}09}=x-0{,}1\\
		\Leftrightarrow & x^2-0{,}09=x^2-0{,}2x+0{,}01\\
		\Leftrightarrow & x=0{,}5 \quad\text{ (thỏa mãn)}.
		\end{eqnarray*}	
		Vậy quãng đường từ $C$ đến $D$ là $0{,}5$ km $=500$ m.		
	}
\end{bt}

\begin{bt}%[Nguyễn Cao Cường]%[EX-TS10-ToanTT-2019]%[9D1T8]
Bạn Khang gửi $8$ triệu theo kì hạn $1$ năm và tiền lãi được dồn vào tiền vốn. Sau $3$  năm Khang nhận được tổng cộng $9261000$VNĐ. Tính lãi suất ngân hàng.
\dapso{$5\%$}	
\loigiai{
	Áp dụng công thức: Khách hàng gửi vào ngân hàng số tiền là $A$ với lãi kép $r\%$ mỗi kì hạn thì số tiền khách hàng nhận được cả vốn lẫn lãi sau $n$ kì hạn là $S=A(1+r\%)^n$.\\
	(\lq \lq Lãi kép: Tiền lãi của kì hạn trước nếu người gửi không rút ra thì được cộng vào vốn để tính lãi cho kì hạn sau\rq\rq.)\\
	Gọi $x$ là lãi suất cần tìm. Ba năm là $3$ kì hạn, nên ta có phương trình
	\begin{eqnarray*}
	&&8000000(1+x)^3=9261000\\
	&\Leftrightarrow & (1+x)^3=\dfrac{9261000}{8000000}\\
	&\Leftrightarrow & (1+x)^3=1{,}157625 \\
	 	&\Leftrightarrow &1+x=\sqrt[3]{1{,}157625}=1{,}05\\ 	&\Leftrightarrow &x=0{,}05=5\%.
	\end{eqnarray*}
	
	Vậy lãi suất ngân hàng là $5\%$.
}
\end{bt}

\begin{bt}%[]%[EX-TS10-ToanTT-2019]%[9D1K9]
	Hàng ngày, bạn Tuấn đi xe đạp từ nhà đến trường cách nhà $0{,}6$ km. Nhưng hôm nay do xe trục trặc bạn phải đi bộ, nên phải đi sớm hơn ngày thường $9$ phút và bạn đến trường sớm hơn $3$ phút so với mọi hôm. Tính vận tốc khi đi xe đạp, vận tốc khi đi bộ, biết rằng tốc độ khi đi xe đạp nhiều hơn khi đi bộ là $8$ km/giờ.
	\dapso{Đáp số}
	\loigiai{
		Gọi $x$ là vận tốc của xe đạp ($x>0$, đơn vị km/h). Ta có vận tốc đi bộ là $x-8$.\\
		Thời gian đi xe đạp nhanh hơn đi bộ là
		$$9-3=6\text{ (phút)}=\dfrac{1}{10} \text{ (giờ)}$$
		Vậy theo đề bai ta có phương trình
		$$\dfrac{0{,}6}{x}-\dfrac{1}{10}=\dfrac{0{,}6}{x-8}$$
		Tiến hành giải phương trình tìm nghiệm ta có
		\begin{eqnarray*}
			\Leftrightarrow&\dfrac{0{,}6}{x}-\dfrac{1}{10}=&\dfrac{0{,}6}{x-8}\\
			\Leftrightarrow&\dfrac{6(x-8)}{10x(x-8)}-\dfrac{x(x-8)}{10x(x-8)}=&\dfrac{6x}{x(x-8)}\\
			\Rightarrow&6(x-8)-x(x-8)=&6x\\
			\Leftrightarrow&-x^2+8x-48=&0\\
			\Leftrightarrow&\hoac{&x=12\text{ (nhận)}\\&x=-4\text{ (loại)}}
		\end{eqnarray*}
		Vậy vận tốc đi xe đạp là $12$ km/h.
	}
\end{bt}
