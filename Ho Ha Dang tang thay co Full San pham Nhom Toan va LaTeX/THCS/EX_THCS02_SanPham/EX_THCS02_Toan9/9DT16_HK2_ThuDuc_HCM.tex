\begin{ex}%[Đề thi Toán 9 Học kỳ II năm học 2015-2016, Thủ Đức, Hồ Chí Minh]%[9D4B4]%[9D2B6]
Cho phương trình $x^2-2x+4m-1=0$ ($x$ là ẩn số)
\begin{enumerate}
\item Tìm điều kiện của $m$ để phương trình có nghiệm
\item Tìm $m$ để phương trình $x_1,x_2$ thoả $x_1^2+x_2^2+2x_1+2x_2=12$
\end{enumerate}
	\loigiai{
\begin{enumerate}		
	\item Xét phương trình:
	\begin{align*}
&	x^2-2x+4m-1=0\\
	\Leftrightarrow &x^2-2x+1=2-4m\\
	\Leftrightarrow &(x-1)^2=2-4m
	\end{align*}	
	Phương trình có nghiệm khi và chỉ khi $2-4m \geqslant 0\Leftrightarrow m \leqslant \dfrac{1}{2}$
	\item Áp đụng định lý Vi-et: $\begin{cases} x_1+x_2=2\\x_1.x_2=4m-1 \end{cases}$\\
	Mặt khác: 
	\begin{align*}
&	x_1^{2}+x_2^{2}+2x_1+2x_2=12\\
	\Leftrightarrow &(x_1 + x_2)^{2} + (x_1 + x_2) - 2x_1.x_2 = 12\\
	\Rightarrow &2^2+2.2-2(4m-1)=12\\
	\Leftrightarrow &m = - \dfrac{1}{4}
	\end{align*}
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ II năm học 2015-2016, Thủ Đức , Hồ Chí Minh]%[9H3Y7]%[9H3K7]%[9H3K3]%[9H3G7]
Cho tam giác $ABC$ có ba góc nhọn $(AB<AC)$ nội tiếp đường tròn $(O)$. Hai đường cao $BE,CF$ cắt nhau tại $H$.
\begin{enumerate}
\item Chứng minh tứ giác $AFHE$ và $BFEC$ nội tiếp
\item Đường thẳng $EF$ cắt đường thẳng $BC$ tại $M$. Chứng minh $\triangle MFB$ đồng dạng $\triangle MCE$
\item Vẽ đường kính $AK$ của đường tròn $(O)$. Chứng minh tứ giác $BHCK$ là hình bình hành
\item Đường thẳng $HK$ cắt đường tròn $(O)$ tại $I$ $(I \ne K)$. Chứng minh ba điểm $A,I,M$ thẳng hàng.
\end{enumerate}
	\loigiai{
	\begin{tikzpicture}[scale=0.5]
\tkzDefPoints{1/2/B}
\tkzDefPoints{9/2/C}
\tkzDefPoints{3.5/7/A}
\tkzDrawPoint(A),(B),(C)
\tkzLabelPoints[above](A)
\tkzLabelPoints[below](B,C)
\tkzDrawSegments(A,B A,C B,C)
\tkzCircumCenter(A,B,C)\tkzGetPoint{O}
\tkzOrthoCenter(A,B,C) \tkzGetPoint{H}
\tkzDrawPoints(O,H)
\tkzLabelPoints[below](H)
\tkzLabelPoints[below](O)
\tkzDrawCircle(O,A)
\tkzDrawSegments(B,H C,H)
\tkzInterLL(B,H)(A,C)\tkzGetPoint{E}
\tkzInterLL(C,H)(A,B)\tkzGetPoint{F}
\tkzDrawPoints(F,E)
\tkzLabelPoints[below](E,F)
\tkzDrawSegments(H,E H,F)
\tkzInterLL(E,F)(B,C)\tkzGetPoint{M}
\tkzDrawSegments(E,M C,M)
\tkzLabelPoints[below](M)
\tkzDefPointBy[symmetry=center O](A)\tkzGetPoint{K}
\tkzDrawSegments(A,K)
\tkzLabelPoints[below](K)
\tkzDrawSegments(B,K C,K H,K)
\tkzInterLC(K,H)(O,A)\tkzGetSecondPoint{I}
\tkzLabelPoints[above](I)
\tkzDrawSegments(M,A I,K)
\end{tikzpicture}
\begin{enumerate}		
	\item Xét tứ giác $AFHE$ ta có:\\
	$\widehat{AFH} + \widehat{AEH}=180 ^\circ$\\
	$\Rightarrow$ tứ giác $AFHE$ nội tiếp\\
	Xét tứ giác $BFEC$ ta có:\\
	$\widehat{BFC}=\widehat{BEC}=90^\circ$\\
	$\Rightarrow$ tứ giác $BFEC$ nội tiếp
	\item Do BFEC nội tiếp\\
	$\Rightarrow \widehat{BFE}+\widehat{BCE}=180^\circ$\\
	$\widehat{MFB}= \widehat{MCE}$\\
	Xét $\triangle MFB$ và $\triangle MCE$:\\
	$\begin{cases} \widehat{MFB}=\widehat{MCE}\\ \widehat{EMC} \end{cases}$\\
	$\Rightarrow \triangle MFB \backsim \triangle MCE$
	\item Do $AK$ là đường kính nên $\widehat{ACK} =90 ^\circ$ và $\widehat{ABK}= 90 ^\circ$\\
	$\widehat{ABK} = \widehat{AFC}$ và $\widehat{ACK} = \widehat{AEB}$\\
	$\begin{cases} CF \perp BK \\ BE \perp CK \end{cases}$\\
	$\Rightarrow$ $BHCK$ là hình bình hành
	\item Giả sử $AM$ cắt $(O)$ tại $J$ $\Rightarrow$ $AJBC$ nội tiếp $\Rightarrow MB \cdot MC=MJ \cdot MA$\\
	Mặt khác $BFEC$ nội tiếp $\Rightarrow MB \cdot MC=MF \cdot ME$\\
	Từ đó, suy ra: $MJ \cdot MA=MF \cdot ME$\\
	$\Rightarrow$ $AJFE$ nội tiếp $\Rightarrow$ $AJFH$ nội tiếp\\
	$\Rightarrow\widehat{AJH} = \widehat{AFH}=90 ^\circ$\\
	Mặt khác $\widehat{AIH}=90 ^\circ$ \\
	$\Rightarrow J \equiv I$\\
	$\Rightarrow$ $A,I,M$ thẳng hàng. 
\end{enumerate}			  
}	
\end{ex}

\item Sau hai năm Bác Minh nhận được số tiền cả vốn lẫn lãi là bao nhiêu? Biết rằng tiền lãi năm trước cộng vào vốn tính lãi năm sau và lãi suất không thay đổi.
\end{enumerate}

\begin{enumerate}		
	\item Số tiền lãi sau một năm là:$120000000 \cdot 6,8 \%=1160000$
	\item Sau hai năm bác minh nhận được cả vốn lẫn lãi là: $120000000 \cdot (1+6,8 \% )^{2} =136874880$
\end{enumerate}
