\begin{ex}%[Đề thi vào lớp 10 - TP Hà Nội năm 2010]%[9D1B8]%[9D1B8]%[9D1K8]
Cho $P=\dfrac{\sqrt{x}}{\sqrt{x}+3}+\dfrac{2\sqrt{x}}{\sqrt{x}-3}-\dfrac{3x+9}{x-9}$, ($x\geq 0$ và $x\neq 9$).
\begin{enumerate}
	\item 	Rút gọn $P$.
	\item	Tìm giá trị của $x$ để $P =\dfrac{1}{3}$.
	\item 	Tìm GTLN của $P$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $P=\dfrac{\sqrt{x}}{\sqrt{x}+3}+\dfrac{2\sqrt{x}}{\sqrt{x}-3}-\dfrac{3x+9}{x-9}$.\\
$P=\dfrac{\sqrt{x}}{\sqrt{x}+3}+\dfrac{2\sqrt{x}}{\sqrt{x}-3}-\dfrac{3x+9}{(\sqrt{x}-3)(\sqrt{x}+3)}$.\\
$P=\dfrac{\sqrt{x}(\sqrt{x}-3)+2\sqrt{x}(\sqrt{x}+3)-(3x+9)}{(\sqrt{x}-3)(\sqrt{x}+3)}$.\\
$P=\dfrac{x-3\sqrt{x}+2x+6\sqrt{x}-3x-9}{(\sqrt{x}-3)(\sqrt{x}+3)}$.\\
$P=\dfrac{3\sqrt{x}-9}{(\sqrt{x}-3)(\sqrt{x}+3)}$.\\
$P=\dfrac{3(\sqrt{x}-3)}{(\sqrt{x}-3)(\sqrt{x}+3)}$.\\
$P=\dfrac{3}{\sqrt{x}+3}$.
\item $P=\dfrac{1}{3} \Leftrightarrow \dfrac{3}{\sqrt{x}+3}=\dfrac{1}{3} \Leftrightarrow \sqrt{x}+3=9 \Leftrightarrow \sqrt{x}=6 \Leftrightarrow x=36$.
\item Ta có $x \ge 0 \Leftrightarrow \sqrt{x} \ge 0 \Leftrightarrow \sqrt{x}+3 \ge 3 \Leftrightarrow \dfrac{1}{\sqrt{x}+3} \le \dfrac{1}{3} \Leftrightarrow \dfrac{3}{\sqrt{x}+3} \le 1 \Leftrightarrow P \le 1$.\\
Vậy $P_{\max}=1$, dấu bằng xảy ra khi $x=0$.
\end{enumerate}
}
\end{ex} 
 
\begin{ex}%[Đề thi vào lớp 10 - TP Hà Nội năm 2010]%[9D4B8]
\textit{Giải bài toán sau bằng cách lập phương trình:}\\
	Một mảnh đất hình chữ nhật có độ dài đường chéo là $13$ m và chiều dài lớn hơn chiều rộng là $7$ m. Tính chiều dài và chiều rộng của mảnh đất đó.
\loigiai{
Gọi chiều rộng của hình chữ nhật là $x$ (m) ($3<x<13$).\\
Vì chiều dài lớn hơn chiều rộng là $7$ m nên chiều dài hình chữ nhật là $x+7$ (m).\\
Theo đề, ta có phương trình: $x^2+(x+7)^2=13^2 $\\
$\Leftrightarrow 2x^2+14x-120=0 \Leftrightarrow x^2+7x-60=0$.\\
$\Delta=289>0$ nên phương trình có hai nghiệm 
\begin{align*}
& x_1=5 \; \text{(thỏa mãn điều kiện)}\\ 
&x_2=-12 \;   \text{(loại)}
\end{align*}
Vậy chiều rộng của hình chữ nhật là $5$ m; chiều dài là $12$ m.
}
\end{ex}  

\begin{ex}%[Đề thi vào lớp 10 - TP Hà Nội năm 2010]%[9D4B6]%[9D4K6]
Cho Parabol $(P): y =-x^2$ và đường thẳng $(d): y =mx-1$.
\begin{enumerate}
\item Chứng minh rằng với mọi $m$ thì $(d)$ luôn cắt $(P)$ tại $2$ điểm phân biệt. 
\item 	Gọi $x_1, x_2$ là các hoành độ giao điểm của $(d)$ và $(P)$. Tìm giá trị của $m$ để $x_1^2x_2+x_2^2x_1- x_1x_2 =3$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Phương trình hoành độ giao điểm $-x^2=mx-1 \Leftrightarrow x^2+mx-1=0$.  \hfill (*)\\
Ta có $\Delta=m^2+4 >0 \; \forall m$, suy ra phương trình (*) luôn có hai nghiệm phân biệt với mọi $m$.\\
Vậy $d$ luôn cắt $(P)$ tại hai nghiệm phân biệt với mọi $m$.
\item Phương trình (*) luôn có hai nghiệm phân biệt với mọi $m$.\\
Áp dụng hệ thức Vi-et, ta có $\heva{&x_1+x_2=-m\\&x_1x_2=-1}$ \hfill (**)\\
Ta có $x_1^2x_2+x_2^2x_1- x_1x_2 =3 \Leftrightarrow x_1x_2(x_1+x_2)-x_1x_2=3$. \hfill (***)\\
Thay (**) vào (***) ta có $-1.(-m)-(-1)=3 \Leftrightarrow m=2$.
\end{enumerate}
}
\end{ex}  

\begin{ex}%[Đề thi vào lớp 10 - TP Hà Nội năm 2010]%[9H3B7]%[9H3K7]
	Cho đường tròn $(O;R)$ đường kính $AB =2R$ và điểm $C$ thuộc đường tròn đó ($C$ khác $A$, $B$), $D$ thuộc dây $BC$ ($D$ khác $B$, $C$). Tia $AD$ cắt cung nhỏ $BC$ tại $E$, tia $AC$ cắt $BE$ tại $F$.
\begin{enumerate}
\item 	Chứng minh tứ giác $FCDE$ nội tiếp.
\item 	Chứng minh $DA.DE = DB.DC$.
\item	Chứng minh $ \widehat{CFD}  = \widehat{OCB}$. Gọi $I$ là tâm đường tròn ngoại tiếp tứ giác $FCDE$, chứng minh $IC$ là 	tiếp tuyến của $(O)$.
\item 	Cho biết $DF =R$, chứng minh $\tan \widehat{AFB} = 2$.
\end{enumerate}
\loigiai{
\immini{
\begin{enumerate}
\item Ta có $\widehat{ACB}=90^\circ$ (góc nội tiếp chắn nửa đường tròn) suy ra $\widehat{DCF}=90^\circ$ (kề bù với góc $\widehat{ACB}$).\\
Tương tự $\widehat{DEF}=90^\circ$.\\
Tứ giác $FCDE$ có $\widehat{DCF}+\widehat{DEF}=180^\circ$\\
Vậy tứ giác $FCDE$ là tứ giác nội tiếp.
\item Xét $\triangle DCA$ và $\triangle DEB$ có
\begin{align*}
&\widehat{ACD}=\widehat{DEB}=90^\circ\\ 
& \widehat{ADC}=\widehat{BDE} \;\text{(hai góc đối đỉnh)}
\end{align*}
Suy ra $\triangle DCA \backsim \triangle DEB$ (g.g)\\
$\Rightarrow \dfrac{DA}{DB}=\dfrac{DC}{DE} \Leftrightarrow DA \cdot DE =DB \cdot DC$.
\end{enumerate}
}
{

\begin{tikzpicture}[scale=1]
\tkzDefPoints{0/0/O, -3/0/A, 3/0/B}
\tkzDrawCircle(O,A) 
\tkzDefPointBy[rotation = center O angle 130](B)\tkzGetPoint{C}
\tkzDefPointBy[homothety = center B ratio 0.7](C)\tkzGetPoint{D}
\tkzInterLC(A,D)(O,B)\tkzGetPoints{A}{E}
\tkzInterLL(A,C)(B,E) \tkzGetPoint{F}
\tkzDefMidPoint(D,F) \tkzGetPoint{I}
\tkzDrawPoints(O,A,B,C,D,E,F,I)
\tkzLabelPoints[above](F)
\tkzLabelPoints[left](A,C)
\tkzLabelPoints[right](E,B,I)
\tkzLabelPoints[below](O,D)
\tkzDrawSegments(A,B A,F B,F B,C O,C A,E D,F I,C)
\tkzMarkSegments[mark=|](I,F I,D)
\end{tikzpicture}
}

\begin{enumerate}
\item[c)] Xét đường tròn ngoại tiếp tứ giác $DCFE$, $\widehat{CFD}=\widehat{CED}$ (cùng chắn cung $CD$) \hfill (1)\\
Xét đường tròn $(O)$, $\widehat{CED}=\widehat{CBA}$  (cùng chắn cung $AC$) \hfill (2)\\
Mặt khác $OB=OC=R$ suy ra $\triangle OBC$ cân tại $O$ suy ra $\widehat{OBC}=\widehat{OCB}$ \hfill (3)\\
Từ (1), (2), (3) suy ra $\widehat{CFD}=\widehat{OCB}$.\\
$I$ là tâm đường tròn ngoại tiếp tứ giác $CDFE$ nên $I$ là trung điểm $DF$.\\
Xét đường tròn $(I)$, $IC=ID$ suy ra $\triangle ICD$ cân tại $I$ $\Rightarrow \widehat{ICD}=\widehat{IDC}$.\\
Ta có $\widehat{ICD}=90^\circ$ (góc nội tiếp chắn nửa đường tròn)\\
suy ra $\widehat{IDC}+\widehat{CFD}=90^\circ$, mà $\widehat{IDC}=\widehat{ICD}$ (cmt), $\widehat{CFD}=\widehat{OCB}$ (cmt)\\
suy ra $\widehat{ICD}+\widehat{OCB}=90^\circ \Leftrightarrow \widehat{ICO}=90^\circ \Leftrightarrow IC \perp OC$, mà $OC$ là bán kính của $(O)$ nên $IC$ là tiếp tuyến của $(O)$.
\item[d)]  Ta có $\triangle CBA \backsim \triangle CFD$ (g.g) $\Rightarrow \dfrac{CB}{CF}=\dfrac{BA}{FD}$.\\
Mà $FD=R; BA=2R$ nên $\dfrac{CB}{CF}=\dfrac{BA}{FD}=2$.\\
Ta có $\tan \widehat{AFB}=\tan \widehat{CFB}=\dfrac{CB}{CF}=2$.
\end{enumerate}
}
\end{ex}  



\begin{ex}%[Đề thi vào lớp 10 - TP Hà Nội năm 2010]%[9H3B7]%[9D4G7]
Giải phương trình $x^2+4x+7=(x+4)\sqrt{x^2+7}$.
\loigiai{
Đặt $t=\sqrt{x^2+7}$, phương trình đã cho trở thành $t^2+4x=(x+4)t$
\begin{align*}
&\Leftrightarrow t^2-(x+4)t+4x=0 \Leftrightarrow (t-x)(t-4)=0 \Leftrightarrow \hoac{&t=x\\&t=4.}\\ 
& \Leftrightarrow \hoac{&\sqrt{x^2+7}=4\\&\sqrt{x^2+7}=x} \Leftrightarrow \hoac{&x^2+7=16\\&\heva{&x \ge 0\\&x^2+7=x^2}} \Leftrightarrow x^2=9 \Leftrightarrow x= \pm 3.
\end{align*}
}
\end{ex}
