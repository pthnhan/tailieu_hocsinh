\begin{bt}%[Đề KSCL T10, TruongVinhKy, HCM 2017]%[0D2B1-2]
	Cho hàm số $$f(x)=\dfrac{4}{5-x}-\sqrt{x-3}.$$
	Hãy tìm tập xác định của hàm số và tính giá trị của hàm số tại $x=4$.
	\loigiai{
		\bigskip Điều kiện xác định $\heva{&x-3\ge0\\ &5-x\ne0} \Leftrightarrow \heva{&x\ge3\\ &x\ne5}$. Vậy $\mathscr{D}=\left[3;+\infty\right) \setminus\left\lbrace5\right\rbrace$.\\
		$f(4)=\dfrac{4}{5-4}-\sqrt{4-3}=3$.
	}
\end{bt}

\begin{bt}%[Đề KSCL T10, TruongVinhKy, HCM 2017]%[0D3K2--4][0D3K3-4]
	Giải phương trình $\sqrt{2x^2-3x+7}=3x-5$.
	\loigiai{
	\begin{eqnarray*}
		\sqrt{2x^2-3x+7}=3x-5&\Leftrightarrow &\heva{&3x-5\ge0\\ &2x^2-3x+7=(3x-5)^2}\\
		&\Leftrightarrow &\heva{&x\ge\dfrac{5}{3}\\ &7x^2-27x+18=0}\\
		&\Leftrightarrow &\heva{&x\ge\dfrac{5}{3}\\ &\hoac{&x=\dfrac{6}{7}\\ &x=3}}\\
		&\Leftrightarrow &x=3.
	\end{eqnarray*}}
\end{bt}
\begin{bt}%[0D3K3-4]
	Giải hệ phương trình $\heva{&x-2y=1\\ &x^2-10y^2=-1}$.
	\loigiai{
			$$\heva{&x-2y=1 &(1)\\ &x^2-10y^2=-1. &(2)}$$
		Từ $(1) \Rightarrow x=2y+1$. Thay vào $(2)$, ta được
		\begin{eqnarray*}
			& &(2y+1)^2-10y^2=-1\notag\\
			&\Leftrightarrow &6y^2-4y-2=0\notag\\
			&\Leftrightarrow &\hoac{&y=1\\ &y=-\dfrac{1}{3}}\notag\\
			&\Leftrightarrow &\hoac{&x=3\\ &x=\dfrac{1}{3}.}\notag
		\end{eqnarray*}
	Vậy hệ có nghiệm $(3;1)$ hoặc $\left(\dfrac{1}{3};-\dfrac{1}{3}\right) $.}
\end{bt}


\begin{bt}%[Đề KSCL T10, TruongVinhKy, HCM 2017]%[0D3K2-5]
		Cho phương trình $x^2-(m+3)x+2(m+1)=0$ ($m$ là tham số).
	Tìm $m$ để phương trình có 2 nghiệm $x_1$, $x_2$ thỏa $x_1-2x_2=0$.
	\loigiai{
		\begin{eqnarray*}
		&\Delta =b^2-4ac & =[-(m+3)]^2-4\cdot1\cdot2(m+1)\\
		& & =m^2+6m+9-8m-8\\
		& & =m^2-2m+1\\
		& & =(m-1)^2\ge0,\forall m.
		\end{eqnarray*}
		Vì $\heva{&a=1\ne0\\ &\Delta\ge0,\forall m}$ nên phương trình luôn có 2 nghiệm phân biệt với mọi $m\ne1$.\\
		Theo hệ thức Vi - ét, ta có
		$$\heva{&x_1+x_2=\dfrac{-b}{a}=m+3 &(1)\\ &x_1x_2=\dfrac{c}{a}=2(m+1). &(2)}$$
		Mà $x_1-2x_2=0 \quad (3)$\\
		Từ $(1)$ và $(3)$ suy ra $\heva{&x_1=\dfrac{2(m+3)}{3}\\ &x_2=\dfrac{m+3}{3}.}$\\
		Thay vào $(2)$
		\begin{eqnarray*}
		& &\dfrac{2(m+3)}{3}\cdot\dfrac{m+3}{3}=2(m+1)\\
		&\Leftrightarrow &2m^2+12m+18=18m+18\\
		&\Leftrightarrow &2m^2-6m=0\\
		&\Leftrightarrow &\hoac{&m=0 &\text{(thỏa)}\\ &m=3 &\text{(thỏa).}}
		\end{eqnarray*}
		Vậy với $m=0$ hoặc $m=3$ thì phương trình có 2 ngiệm phân biệt thỏa đề bài.
	}
\end{bt}

\begin{bt}%[Đề KSCL T10, TruongVinhKy, HCM 2017]%[0H1K3-2]
		Cho $\triangle ABC$. Trên các cạnh $AB$, $BC$ và $CA$ lấy các điểm $D$, $E$, $F$ sao cho:
	$DA=2DB$; $EB=2EC$; $FC=2FA$.\\
	Chứng minh: $\overrightarrow{AD}+\overrightarrow{AE}+\overrightarrow{AF}=\overrightarrow{AB}+\overrightarrow{AC}$.
	\loigiai{
		\immini {
			Vì $DA=2DB$ nên $AD=\dfrac{2}{3}AB \Rightarrow \overrightarrow{AD}=\dfrac{2}{3}\overrightarrow{AB}$\\
			Tương tự $\overrightarrow{BE}=\dfrac{2}{3}\overrightarrow{BC}$; $\overrightarrow{AF}=\dfrac{1}{3}\overrightarrow{AC}$.\\
			Khi đó
	\begin{eqnarray*}
	&VT & =\overrightarrow{AD}+\overrightarrow{AE}+\overrightarrow{AF}\notag\\
	& &=\dfrac{2}{3}\overrightarrow{AB}+(\overrightarrow{AB}+\overrightarrow{BE})+\dfrac{1}{3}\overrightarrow{AC}\notag\\
	& & =\dfrac{5}{3}\overrightarrow{AB}+\dfrac{1}{3}\overrightarrow{AC}+\dfrac{2}{3}\overrightarrow{BC}\notag\\
	& & =\dfrac{5}{3}\overrightarrow{AB}+\dfrac{1}{3}\overrightarrow{AC}+\dfrac{2}{3}(\overrightarrow{AC}-\overrightarrow{AB})\notag\\
	& & =\overrightarrow{AB}+\overrightarrow{AC}=VP.\notag
	\end{eqnarray*}
	Vậy $\overrightarrow{AD}+\overrightarrow{AE}+\overrightarrow{AF}=\overrightarrow{AB}+\overrightarrow{AC}$.
}
{

\tkzDefPoints{0/0/B, 1/3/A, 6/0/C} % lấy điểm theo tikz euclide

\coordinate (F) at ($(C)!0.66!(A)$);

}
}
\end{bt}

\begin{bt}%[Đề KSCL T10, TruongVinhKy, HCM 2017]%[0H2B1-4][0H1G4-3]
Trong mặt phẳng $Oxy$, cho $\triangle ABC$, biết $A(4;1)$,$B(1;5)$, $C(-4;-5)$.
\begin{enumerate}
	\item Chứng minh $\triangle ABC$ vuông tại $A$ và tính số đo góc $B$ (làm tròn đến phút).
	\item Phân giác trong góc $A$ cắt cạnh $BC$ tại $D$. Gọi $H$ là hình chiếu vuông góc của $D$ lên $AC$. Tìm tọa độ của điểm $H$.
\end{enumerate}
\loigiai{ 
	\begin{enumerate}
			\item
			\immini{
		  Ta có
		 \begin{eqnarray*}
			&\overrightarrow{AB}=(-3;4) &\Rightarrow AB=\sqrt{(-3)^2+4^2}=5,\\
			&\overrightarrow{AC}=(-8;-6) &\Rightarrow AC=\sqrt{(-8)^2+(-6)^2}=10,\\
			&\overrightarrow{BC}=(-5;-10) &\Rightarrow BC=\sqrt{(-5)^2+(-10)^2}=5\sqrt{5}.
		\end{eqnarray*}
			Vì $AB^2+AC^2=5^2+10^2=125$	và $BC^2=(5\sqrt{5})^2=125$ nên $AB^2+AC^2=BC^2$.\\
			Suy ra $\triangle ABC$ vuông tại $A$.\\
			Áp dụng hệ thức lượng vào tam giác vuông $ABC$\\
			$\Rightarrow \tan \widehat{B}=\dfrac{AC}{AB}=2 \Rightarrow \widehat{B}\approx63^\circ26'$.}

Lại có $DH\parallel AB$ (gt)
