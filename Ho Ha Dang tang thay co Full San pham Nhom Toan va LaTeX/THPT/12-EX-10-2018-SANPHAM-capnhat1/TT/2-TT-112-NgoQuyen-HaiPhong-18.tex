$$
u_n=\dfrac{1}{n}\left[f(1)+\sqrt{\dfrac{n}{3+n}}\cdot f\left(\sqrt{\dfrac{n+3}{n}}\right)+\sqrt{\dfrac{n}{6+n}}\cdot f\left(\sqrt{\dfrac{n+6}{n}}\right)+\cdots +\sqrt{\dfrac{n}{4n-3}}\cdot f\left(\sqrt{\dfrac{4n-3}{n}}\right)\right].
$$
\choice
{$\displaystyle\lim u_n=2$}
{\True $\displaystyle\lim u_n=\dfrac{2}{3}$}
{$\displaystyle\lim u_n=1$}
{$\displaystyle\lim u_n=\dfrac{4}{3}$}

Ta có:\\ 
$u_n=\dfrac{f(1)}{n}+\dfrac{1}{n}\left[\dfrac{1}{\sqrt{1+\dfrac{3}{n}}}f\left(\sqrt{1+\dfrac{3}{n}}\right)+\dfrac{1}{\sqrt{1+\dfrac{2\cdot 3}{n}}}f\left(\sqrt{1+\dfrac{2\cdot 3}{n}}\right)+\cdots +\dfrac{1}{\sqrt{1+\dfrac{3(n-1)}{n}}}f\left(\sqrt{1+\dfrac{3(n-1)}{n}}\right)\right]$.\\

Đặt $t=\sqrt{1+3x}\Rightarrow \mathrm{\,d}t=\dfrac{3}{2\sqrt{1+3x}}\mathrm{\,d}x\Rightarrow \lim u_n=\dfrac{2}{3}\displaystyle\int\limits_1^2f(t)\mathrm{\,d}t=\dfrac{2}{3}$.
}
\end{ex}

\begin{ex}%[Đề TT lần 2, Ngô Quyền, Hải Phòng 2018]%[Hung Tran,12EX-10]%[1D2K3-2]

{$a=a_{749}$}
{$a=a_{501}$}
{\True $a=a_{750}$}
{$a=a_{500}$}
