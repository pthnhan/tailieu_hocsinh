\begin{ex}%[Thi thử L2, THPT Cầu Xe- Hải Dương, 2018]%[2D4K3-4]%[Võ Tấn Đạt, 12EX8]
	Xét các số phức $ z = a+bi\ (a,b\in \mathbb{R}) $ thỏa mãn $ \left| z-3+3i\right| = \sqrt{2} $. Tính $ P = a+b $ khi $ |z-1+3i| + |z-3+5i| $ đạt giá trị lớn nhất.
	\choice
	{$ P = -2 $}
	{$ P = -8 $}
	{$ P = 8 $}
	{\True $ P = 2 $}
	\loigiai{
		Ta có $\left| z-3+3i\right| = \sqrt{2} \Leftrightarrow (a-3)^2 + (b+3)^2 = 2 $.\\ Tìm giá trị lớn nhất của
		$ |z-1+3i| + |z-3+5i| = \sqrt{(a-1)^2 + (b+3)^2} + \sqrt{(a-3)^2 + (b+5)^2} $.\\
		Áp dụng bất đẳng thức Cauchy-Schwars ta có
		\begin{align*}
		\left[\sqrt{(a-1)^2 + (b+3)^2} + \sqrt{(a-3)^2 + (b+5)^2}\right]^2 &\le 2\left[(a-1)^2 + (b+3)^2 + (a-3)^2 + (b+5)^2 \right]\\
		&\le 4(a^2-4a+b^2+8b+22).  
		\end{align*}
		Ta đổi biến $ a-3 = u, b+3 = v $. Từ đó điều kiện là $ u^2+v^2 = 2 $ và ta cần tìm giá trị lớn nhất của 
		\[ u^2 + v^2 + 2(u+v) + 4 \le 2 + 2 \sqrt{2(u^2+v^2)} + 4 = 10. \]
		Dấu bằng xảy ra khi và chỉ khi $ u=v=1 \Leftrightarrow \heva{&a=4\\&b=-2} $. Vậy $ P = a+b= 2 $.
	}
\end{ex}

\begin{ex}%[Thi thử L2, THPT Cầu Xe- Hải Dương, 2018]%[2D1K1-3]%[Võ Tấn Đạt, 12EX8]
