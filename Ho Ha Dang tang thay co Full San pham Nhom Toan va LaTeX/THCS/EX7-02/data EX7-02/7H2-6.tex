\section{ĐỊNH LÍ PY-TA-GO}
\subsection{Tóm tắt lí thuyết}
\begin{dl}[Py-ta-go]
Trong một tam giác vuông, bình phương của cạnh huyền bằng tổng các bình phương của hai cạnh góc vuông.
\end{dl}

\begin{dl}[Py-to-go đảo]
Nếu một tam giác có bình phương của một cạnh bằng tổng các bình phương của hai cạnh kia thì tam giác đó là tam giác vuông.
\end{dl}

\begin{vd}

\loigiai{
\immini{
Đặt $BH=HC=x$, $AB=AC=y$. Trong tam giác vuông $AHB$ ta có
$$ AB^2=AH^2+BH^2 \Rightarrow y^2=25+x^2 $$

$ (5k)^2=25+(3k)^2\Rightarrow k=\dfrac{5}{4}$. Do đó $x=\dfrac{15}{4}$. Từ đó, 
$S_{ABC}=\dfrac{1}{2}BC\cdot AH=\dfrac{1}{2}\cdot 2\cdot \dfrac{15}{4}\cdot 5=18{,}75$.

\begin{vd}
Các đường cao của tam giác $ABC$ có độ dài bằng $60$, $65$ và $156$. Chứng minh tam giác $ABC$ vuông.
\loigiai{
Giả sử các cạnh $a$, $b$, $c$ của tam giác $ABC$ có các đường cao tương ứng là $60$, $65$ và $156$. Ta có
$$ 60a=65b=156c \Rightarrow \frac{60a}{780}=\frac{65b}{480}=\frac{156c}{780} .$$
Hay $\dfrac{a}{13}=\dfrac{b}{12}=\dfrac{c}{5}=k$. Từ đó suy ra
$a=13k$, $b=12k$, $c=5k$.\\
Ta thấy $(13k)^2=(12k)^2+(5k)^2$ hay $a^2=b^2+c^2$ nghĩa là tam giác $ABC$ vuông.
}
\end{vd}

\begin{vd}
Cho hai đoạn thẳng $AC$ và $BD$ vuông góc với nhau và cắt nhau tại $O$. Chứng minh rằng
$$ AB^2+CD^2= AD^2+BC^2$$
\loigiai{
Áp dụng định lí Py-ta-go cho các tam giác vuông $OAB$, $OBC$, $OCD$, $ODA$ ta có
{\allowdisplaybreaks
\begin{align}
&AB^2=OA^2+OB^2 \tag{1}  \\
&BC^2=OB^2+OC^2 \tag{2} \\
&CD^2=OC^2+OD^2 \tag{3} \\
&DA^2=OD^2+OA^2 \tag{4}
\end{align}
}
\immini{
Cộng từng vế của (1) với (3) ta có $AB^2+CD^2=OA^2+OB^2+OC^2+OD^2$.\\
Cộng từng vế của (2) với (4) ta có $AD^2+BC^2=OA^2+OB^2+OC^2+OD^2$.\\
Suy ra $ AB^2+CD^2= AD^2+BC^2$.
}{

\end{tikzpicture}

}
\end{vd}


\begin{vd}
Cho tam giác $ABC$ có $\widehat{A}=90^\circ$. Gọi $M$ là trung điểm $AC$. Chứng minh rằng $$ BM^2=BC^2-\dfrac{3}{4} AC^2.$$
\loigiai{
Áp dụng định lí Py-ta-go cho các tam giác vuông $ABM$, $ABC$ ta có
	$$ BM^2=AB^2+AM^2. $$
Suy ra $BM^2=BC^2-AC^2+AM^2$.\\ Vậy $BM^2=BC^2-Ac^2+\dfrac{1}{4}AC^2$ hay $ BM^2=BC^2-\dfrac{3}{4} AC^2$.
}
\end{vd}




\begin{vd}
Chứng minh rằng trong một tam giác vuông, tổng các bình phương của các đường trung tuyến bằng $\dfrac{3}{2}$ bình phương của cạnh huyền.
\loigiai{
giả sử tam giác $ABC$ vuông tại $A$, các đường trung tuyến $AM$, $BN$, $CP$. Trong tam giác vuông $ABN$ ta có
\begin{align}
BN^2&=AB^2+AN^2=BC^2-AC^2+\frac{1}{4}AC^2\notag\\ 
&=BC^2-\frac{3}{4}AC^2  \tag{1}
\end{align}
Tương tự, ta cũng có $CP^2=BC^2-\dfrac{3}{4}AB^2. \quad (2)$ \\
Vì $AM=\dfrac{1}{2}BC$ nên $AM^2=\dfrac{1}{4}BC^2. \quad (3)$\\
Cộng vế theo vế (1), (2), (3) ta được 
\begin{align*}
AM^2+BN^2+CP^2&=\frac{1}{4}BC^2+BC^2-\frac{3}{4}AC^2+BC^2-\frac{3}{4}AB^2\\ 
&=\frac{9}{4}BC^2-\frac{3}{4}(AC^2+AB^2)=\frac{9}{4}BC^2-\frac{3}{4}BC^2=\frac{3}{2}BC^2.
\end{align*}
}
\end{vd}

\begin{vd}%[ ]%[, dự án ]%[]
Trong một hồ nước có một bông sen. Cả bông sen và nhánh sen nhô lên mặt nước $ 0{,}5 $ m. Chợt có một ngọn gió thổi tới làm bông sen bị dạt xuống, chìm hẳn vào nước. Hỏi hồ nước sâu bao nhiêu mét?
	\loigiai
	{
\immini
{

Gọi đỉnh của bông sen là điểm $ B $, gốc của bông sen là điểm $ A $, đường thẳng $ HC$ chính là mặt nước, điểm $ C $ là vị trí mà bông sen bị gió thổi chạm mặt nước. Theo đề bài ta có $ AB=AC$. Chiều sâu của hồ nước là độ dài đoạn thẳng $ AH $.\\
Xét tam giác $ AHC $ vuông tại đỉnh $ H $ có $ AH=x>0$; $ HC=1{,5}$; $ AC=x+0{,5} $. 
Áp dụng định lí Pi-ta-go ta có 
\begin{eqnarray*}
	&&AH^2+HC^2=AC^2\\

	&\Leftrightarrow& x^2+1{,}5^2=x^2+2\cdot x \cdot 0{,5}+0{,}5^2\\ 
	&\Leftrightarrow& x= 2.
\end{eqnarray*}
Vậy chiều sâu của hồ nước là $ AH=2 $m.	
}

\begin{vn}
Cho tam giác $ABC$ ($AB=AC$) $\widehat{A}<90^\circ$. kẻ $BH$ vuông góc với $AC$. Chứng minh rằng $$ AB^2+AC^2+BC^2=3BH^2+2AH^2+CH^2. $$
\loigiai{
Áp dụng định lí Py-ta-go cho các tam giác vuông $ABH$, $BCH$ ta có
{\allowdisplaybreaks
\begin{align}
&AB^2=AH^2+BH^2 \tag{1}  \\
&BC^2=BH^2+CH^2 \tag{2} \\
&AC^2=BH^2+AH^2 \tag{3} 
\end{align}
}
Cộng vế theo vế (1), (2), (3) ta được
	$$ AB^2+AC^2+BC^2=3BH^2+2AH^2+CH^2. $$
}
\end{vn}
