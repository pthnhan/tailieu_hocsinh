\subsection{Đề số 1a}
\setcounter{bt}{0}
\begin{bt}%[Lê Thanh Nin]%[2D2Y2][2D2B4]
	($2{,}0$ điểm) 
	\begin{enumerate}
\item 	Tìm tập xác định của hàm số sau : $y=\left(x^2-2x \right)^{-2} $.
\item   Tính đạo hàm của hàm số sau : $y=\mathrm{e}^x\sin x$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}	
		\item Điều kiện : $x^2-2x\neq 0 \Leftrightarrow \heva{&x\neq 0\\&x\neq 2.}$\\
		Vậy tập xác định của hàm số là : $ \mathscr{D}=\mathbb{R}\backslash \left\lbrace 1;2 \right\rbrace  .$
		\item Tập xác định :  $ \mathscr{D}=\mathbb{R}$\\
		Đạo hàm : $ y'=\left(\mathrm{e}^x\sin x\right)^{'}=\left(\mathrm{e}^x\right)^{'}\sin x+\mathrm{e}^x\left(\sin x \right)^{'}= \mathrm{e}^x\sin x+\mathrm{e}^x\cos x.$
		\end{enumerate}
	}
	
\end{bt}

\begin{bt}%[Lê Thanh Nin]%[2D2B5]
	($2{,}0$ điểm) Giải phương trình sau : $ 16^x-17\cdot 4^x +16=0 .$
	\loigiai{
	Đặt $ t=4^x>0 $ ta có phương trình $ t^2-17\cdot t +16=0 \Leftrightarrow \hoac{&t=1\\&t=16.} $
	\begin{itemize}
		\item $ t=1\Rightarrow 4^x=1 \Leftrightarrow x=0 $.
		\item $ t=16\Rightarrow  4^x=16 \Leftrightarrow x=2 $.
\end{itemize}
Vậy tập nghiệm của phương tình là : $ \mathbb{T}=\left\lbrace 0;2 \right\rbrace .$
}
\end{bt}
	
\begin{bt}%[Lê Thanh Nin]%[2D2B5]
	($2{,}0$ điểm) Giải phương trình sau : $ \log_2\sqrt{x-3}+\log_2\sqrt{3x-7}=2 .$
	\loigiai{
		Điều kiện : $ x>3 $.\\
	 $ \log_2\sqrt{x-3}+\log_2\sqrt{3x-7}=2\\
	 \Leftrightarrow \log_2\sqrt{(x-3)(3x-7)}=2\\
	 \Leftrightarrow \sqrt{3x^2-16x+21}=4\\
	 \Leftrightarrow 3x^2-16x+5=0
	 \Leftrightarrow \heva{&x=\dfrac{1}{3}\,\text {(loại)}\\&x=5.} $\\
	Vậy phương trình có nghiệm duy nhất : $ x=5 .$
	}	
\end{bt}

\begin{bt}%[Lê Thanh Nin]%[2D2K6]
	($2{,}0$ điểm) Giải bất phương trình sau : $ \log_{\frac{1}{2}}\left( x^2-5x-6\right) \ge -3 .$
	\loigiai{
		Điều kiện : $ x^2-5x-6>0\Leftrightarrow \hoac{&x<-1\\&x>6.} $\\
		$ \log_{\frac{1}{2}}\left( x^2-5x-6\right) \ge -3\\
		\Leftrightarrow  \log_{\frac{1}{2}}\left( x^2-5x-6\right) \ge \log_{\frac{1}{2}}8\\
		\Leftrightarrow x^2-5x-6\le 8\\
		\Leftrightarrow x^2-5x-14\le 0 \\
		\Leftrightarrow -2\le x\le 7 $\\ 
		Kết hợp điều kiện ta được : $ \hoac{&-2\le x<-1\\&6<x\le 7.} $\\
		Vậy tập nghiệm của bất phương trình là : $ \mathbb{T}=\left[-2;-1\right)\cup \left(6;7\right].$
	}	
\end{bt}
\begin{bt}%[Lê Thanh Nin]%[2D2G6]
($2{,}0$ điểm) Tìm tập hợp tất cả các giá trị của tham số $m$ để bất phương trình \break $4^{\sin^2x}+5^{\cos^2x}\le m\cdot 7^{\cos^2x}$ có nghiệm.
\loigiai{
	Ta có $4^{\sin^2x}+5^{\cos^2x}\le m.7^{\cos^2x}\Leftrightarrow 4\cdot \left({\dfrac{1}{28}}\right)^{\cos^2x}+\left(\dfrac{5}{7}\right)^{\cos^2x}\le m$.\\
	Đặt $t=\cos^2x,t\in \left[0;1\right]$ thì bất phương trình trở thành: $4\cdot \left(\dfrac{1}{28}\right)^t+\left(\dfrac{5}{7}\right)^t\le m$.\\
	Xét $f(t)=4\cdot \left(\dfrac{1}{28}\right)^t+\left(\dfrac{5}{7}\right)^t$ là hàm số nghịch biến trên $\left[0;1\right]$.\\
	Suy ra: $f(1)\le f(t)\le f(0)\Leftrightarrow \dfrac{6}{7}\le f(t)\le 5$.\\
	Từ đó bất phương trình có nghiệm $\Leftrightarrow m\ge \dfrac{6}{7}$.}
\end{bt}

	($2{,}0$ điểm) 
	\begin{enumerate}

		\item   Tính đạo hàm của hàm số sau : $y=\log_2\left( x^2+1\right) $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}	
			\item Điều kiện : $-x^2-2x+3>0 \Leftrightarrow -3<x<1$.\\
			Vậy tập xác định của hàm số là : $ \mathscr{D}=(-3;1).$
			\item Tập xác định :  $ \mathscr{D}=\mathbb{R}$.\\
			Đạo hàm : $ y'=\dfrac{\left( x^2+1\right)^{'}}{\left( x^2+1\right)\ln 2}=\dfrac{2x}{\left( x^2+1\right)\ln 2}.$
		\end{enumerate}
	}
	
\end{bt}
\begin{bt}%[Lê Thanh Nin]%[2D2B5]
	($2{,}0$ điểm) Giải phương trình sau : $ 9^x-3^x -6=0 $.
	\loigiai{
		Đặt $ t=3^x>0 $ ta có phương trình $ t^2-t-6=0 \Leftrightarrow \hoac{&t=3\\&t=-2 \,\text {(loại)}.} $\\
		Với $ t=3\Rightarrow  3^x=3 \Leftrightarrow x=1 $.\\
		Vậy phương trình có nghiệm duy nhất $ x=1 .$
	}

\end{bt}
\begin{bt}%[Lê Thanh Nin]%[2D2B5]
	($2{,}0$ điểm) Giải phương trình sau : $ \log_2^2x-\log_8x=\dfrac{4}{3} $.
	\loigiai{
		Điều kiện : $ x>0 $.\\
		Đặt $ t=\log_2x $. Ta có phương trình $ t^2-\dfrac{t}{3}-\dfrac{4}{3}=0 \Leftrightarrow \hoac{&t=-1\\&t=\dfrac{4}{3}.} $
		\begin{itemize}
		\item Với $ t=-1\Rightarrow  x=\dfrac{1}{2}$.
		\item Với $ t=\dfrac{4}{3}\Rightarrow  x=\sqrt[3]{16}$.
		\end{itemize}
	Vậy tập nghiệm của phương tình là : $ \mathbb{T}=\left\lbrace\dfrac{1}{2};\sqrt[3]{16} \right\rbrace .$	
	}
	
\end{bt}

\begin{bt}%[Lê Thanh Nin]%[2D2K6]
	($2{,}0$ điểm) Giải bất phương trình sau : $ \log_{\frac{1}{15}}( x-2)+\log_{\frac{1}{15}}( 10-x) \ge -1 $.
	\loigiai{
		Điều kiện : $ 2<x<10 $.\\
		Ta có : $ \log_{\frac{1}{15}}( x-2)+\log_{\frac{1}{15}}( 10-x) \ge -1\\ \Leftrightarrow \log_{\frac{1}{15}}\left[( x-2)( 10-x)\right]\ge \log_{\frac{1}{15}}15\\
		\Leftrightarrow ( x-2)( 10-x)\le 15\\ \Leftrightarrow x^2-12x+35\le 0 \\
		\Leftrightarrow x\le 5 $ hoặc $x\ge 7.$\\
		Kết hợp với điều kiện ta được tập nghiệm của bất phương trình là : $ \mathbb{T}=\left(2;5\right]\cup \left[7;10\right)$.
	}	
\end{bt}


\begin{bt}%[Lê Thanh Nin]%[2D2G6]
	($2{,}0$ điểm) Với giá trị nào của $m$ để bất phương trình $9^x-2\left(m+1\right).3^x-3-2m>0$ có nghiệm đúng với mọi số thực $x\in \mathbb{R}?$
	\loigiai{
		$9^x-2\left(m+1\right).3^x-3-2m>0$.\\
		Đặt $t=3^x>0 $. Bất phương trình trở thành: $t^2-2\left(m+1\right)t-3-2m>0,\forall t>0$.\\
		$\Leftrightarrow t^2-2mt-2t-3-2m>0,\forall t>0$.\\
		$\Leftrightarrow t^2-2t-3>2m\left(t+1\right),\forall t>0$.\\
		$\Leftrightarrow m<\dfrac{t^2-2t-3}{2\left(t+1\right)}$ vì $t+1>0,\forall t>0$.\\
		$\Leftrightarrow m<\dfrac{t-3}{2},\forall t>0$. (*)\\
		Xét hàm số $g(t)=\dfrac{t-3}{2}$ trên $\left(0;+\infty\right)$.\\
		$g'(t)=\dfrac{1}{2}>0$. Suy ra hàm số $g(t)$ luôn đồng biến trên $\left(0;+\infty\right)$.\\
		Mặt khác $g(0)=-\dfrac{3}{2} $. Do đó: $\left(*\right)\Leftrightarrow m\le -\dfrac{3}{2} $.}
\end{bt}

\begin{bt}\textbf{($2,0$ điểm)}%[2D2Y2][2D2Y1]%[LÊ QUỐC SANG]
	\begin{enumerate}
		\item Tìm tập xác định của hàm số $y=(2x-4)^{-8}$.
		\item Rút gọn biểu thức $P=x^{\frac{1}{3}} \cdot \sqrt[6]{x}$  với $x>0.$
	\end{enumerate}
	\loigiai
	{ 
		\begin{enumerate}
			\item Vì $-8$ là số nguyên âm nên điều kiện xác định của hàm số $y=(2x-4)^{-8}$ là:\\ $2x-4\neq 0\Leftrightarrow x\neq 2\Rightarrow \mathscr{D}=\mathbb{R}\setminus \left\{2\right\}$.
			\item Ta có $P=x^{\frac{1}{3}} \cdot \sqrt[6]{x}=x^{\frac{1}{3}}\cdot x^{\frac{1}{6}}=x^{\frac{1}{2}}=\sqrt{x}.$ 
		\end{enumerate}
	}
\end{bt}

\begin{bt}\textbf{($2,0$ điểm)}%[2D2B3]%[LÊ QUỐC SANG]
	Cho $a,\ b$ là hai số thực dương thỏa mãn $a\ne 1,\ a\ne\dfrac{1}{b}$ và $\log_a b=\sqrt{5}$. Tính $P=\log_{\sqrt{ab}}\dfrac{b}{\sqrt{a}}$.	
	\loigiai
	{
		Ta có $P=\dfrac{\log_a\dfrac{b}{\sqrt{a}}}{\log_a\sqrt{ab}}=\dfrac{\log_ab-\log_a\sqrt{a}}{\log_a\sqrt{a}+\log_a\sqrt{b}}=\dfrac{\log_ab-\dfrac{1}{2}}{\dfrac{1}{2}+\dfrac{1}{2}\log_ab}=\dfrac{11-3\sqrt{5}}{4}$.	
	}
\end{bt}

\begin{bt}\textbf{($2,0$ điểm)}%[2D2K3]%[LÊ QUỐC SANG]
	Cho $a>0$, $b>0$ thoả mãn $a^2+9b^2=10ab$. \\Chứng minh rằng: $\log \dfrac{a+3b}{4}=\dfrac{\log a+\log b}{2}$.
	\loigiai
	{
		Ta có $a^2+9b^2=10ab \Leftrightarrow a^2+9b^2+6ab=16ab \Leftrightarrow \left( {a+3b} \right)^2=16ab \Leftrightarrow \left( {\dfrac{a+3b}{4}} \right)^2=ab 
		\\\Leftrightarrow \log \left( {\dfrac{a+3b}{4}} \right)^2=\log ab \Leftrightarrow 2\log \dfrac{a+3b}{4}=\log a+\log b \Leftrightarrow \log \dfrac{a+3b}{4}=\dfrac{\log a+\log b}{2}$.}
\end{bt}

\begin{bt}\textbf{($2,0$ điểm)}%[2D2B6]%[LÊ QUỐC SANG]
	Giải bất phương trình $\log_2^2x-4\log_2x+3>0$.
	\loigiai
	{
		Điều kiện: $x>0$\\
		BPT $\Leftrightarrow \log_2^2x-4\log_2x+3>0\Leftrightarrow\log_2x\in(-\infty;1)\cup(3;+\infty)\Leftrightarrow x\in (0;2)\cup(8;+\infty)$.
	}
\end{bt}

\begin{bt}\textbf{($2,0$ điểm)}%[2D2G5]%[LÊ QUỐC SANG]
	Tìm các giá trị thực của tham số $m$ để phương trình $9^x - 2 \cdot 3^{x+1} + m = 0$ có hai nghiệm thực phân biệt $x_1$, $x_2$ thỏa mãn $x_1 + x_2 =1$.
	\loigiai
	{
		Đặt $ t = 3^x$ $(t > 0)$. Ta được phương trình $t^2 - 6t + m = 0$. \hfill (1)\\
		Phương trình ban đầu có hai nghiệm thỏa yêu cầu bài toán khi phương trình $(1)$ có hai nghiệm dương phân biệt $\Leftrightarrow \heva{& \Delta > 0 \\ & S > 0 \\ & P > 0}
		\Leftrightarrow \heva{& 3^2 - m > 0 \\ & 6 > 0 \\ & m > 0}  
		\Leftrightarrow 0<m<9$.\\
		Khi $0<m<9$, phương trình $(1)$ có 2 nghiệm $t_1,\ t_2$ và $t_1=3^{x_1},\ t_2=3^{x_2}$.\\
		Theo giả thiết $t_1\cdot t_2=3^{x_1+x_2}=3^1=3 \Leftrightarrow m=3$ (thỏa điều kiện).	
	}	
\end{bt}

\begin{bt}\textbf{($2,0$ điểm)}%[2D2Y4][2D2Y3]%[LÊ QUỐC SANG]
	\begin{enumerate}
		\item Tìm tập xác định của hàm số $y=\log_{\frac12}\left(x^2-3x+2\right)$.
		\item Cho $a$ là số thực dương khác $1$. Tính $\log_{\sqrt{a}}a.$
	\end{enumerate}	
	\loigiai
	{
		\begin{enumerate}	
			\item Điều kiện 
			$x^2-3x+2>0\Leftrightarrow \left[\begin{aligned}
			&x<1\\&x>2
			\end{aligned}\right.$ nên tập xác định của hàm số là: $\mathscr{D} =(-\infty;1)\cup(2;+\infty)$.
			\item Ta có $\log_{\sqrt{a}}a = \log _{a^{\tfrac{1}{2}}} a=2\log _a a=2.$
		\end{enumerate}	
	}
\end{bt}

\begin{bt}\textbf{($2,0$ điểm)}%[2D2K1]%[LÊ QUỐC SANG]
	Rút gọn biểu thức $A=(a-4)\left(\dfrac{a}{4-a}\right)^{\frac{1}{2}}+\left[a(4-a)\right]^{\frac{1}{2}},$ với $0<a<4.$
	\loigiai
	{
		Ta có
		\begin{align*} 
			A=(a-4)\left(\dfrac{a}{4-a}\right)^{\frac{1}{2}}+\left[a(4-a)^{\frac{1}{2}}\right]
			&=-\sqrt{(4-a)^2}\dfrac{\sqrt{a}}{\sqrt{4-a}}+\sqrt{a} \sqrt{4-a} \\
			&=-\sqrt{a}   \sqrt{4-a}+\sqrt{a}  \sqrt{4-a}=0.
		\end{align*}
	}
\end{bt}

\begin{bt}\textbf{($2,0$ điểm)}%[2D2K3]%[LÊ QUỐC SANG]
	Cho $x,y$ là các số dương lớn hơn 1 thỏa mãn $x^2+9y^2=6xy$. Tính \break $M=\dfrac{1+\log_{12} x+ \log_{12} y}{2 \log_{12} (x+3y)}$.
	\loigiai
	{
		Ta có $x^2+9xy=6xy \Leftrightarrow (x-3y)^2=0 \Leftrightarrow x=3y$.\\
		Khi đó $M=\dfrac{1+2\log_{12} 3y + \log_{12} y}{2\log_{12} 6y} =\dfrac{\log_{12} 12y+ \log_{12} 3y}{\log_{12} 36y^2} =\dfrac{\log_{12} 36y^2}{\log_{12} 36y^2}=1$.
	}
\end{bt}

\begin{bt}\textbf{($2,0$ điểm)}%[2D2B6]%[LÊ QUỐC SANG]
	Giải bất phương trình sau: $\log_{3}(x^{2}+4x)+\log_{\frac{1}{3}}(2x-3)>0$.
	\loigiai
	{
		Điều kiện xác định $x>\dfrac{3}{2}$.\\
		$\log_{3}(x^{2}+4x)+\log_{\dfrac{1}{3}}(2x-3)>0\Leftrightarrow \log_{3}(x^{2}+4x)>\log_{3}(2x-3)\Leftrightarrow x^{2}+4x>2x-3\Leftrightarrow x^{2}+2x+3>0$ (thỏa $\forall x \in \mathbb{R})$.\\
		Vậy bất phương trình có nghiệm $x \in \left(\dfrac{3}{2};+\infty \right)$.
	}
\end{bt}

\begin{bt}\textbf{($2,0$ điểm)}%[2D2G5]%[LÊ QUỐC SANG]
	Tìm các giá trị thực  của tham số $m$ để phương trình $4^x-m2^{x+1}+2m=0$ có hai nghiệm phân biệt $x_1,\ x_2$ thỏa mãn $x_1+x_2=3$.
	\loigiai
	{
		Đặt $t=2^x,\ t>0$. Phương trình trở thành $t^2-2mt+2m=0$. \hfill (1)\\
		Phương trình ban đầu có 2 nghiệm phân biệt khi phương trình $(1)$ có hai nghiệm phân biệt dương $\Leftrightarrow \heva{&\Delta'=m^2-2m>0\\ &S=2m>0\\&P=2m>0} \Leftrightarrow m>2$.\\
		Khi $m>2$, phương trình $(1)$ có 2 nghiệm $t_1,\ t_2$ và $t_1=2^{x_1},\ t_2=2^{x_2}$.\\
		Theo giả thiết $t_1\cdot t_2=2^{x_1+x_2}=2^3=8 \Leftrightarrow 2m=8 \Leftrightarrow m=4$ (thỏa điều kiện).	
	}
\end{bt}
