\begin{bt}%[Đề HKI, Bình Hưng Hòa, TP Hồ Chí Minh 2017]%[Hung Tran, 11-HK1-18]%[1D1B3-3]%[1D1B3-1]
	Giải các phương trình lượng giác sau:
	\begin{enumerate}
	\item $\cos 2x+9\cos x+5=0$.
	\item $\sqrt{3}\sin 4x+\cos 4x-\sqrt{3}=0$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item Ta có
	\begin{eqnarray*}
	\cos 2x+9\cos x+5=0 
	&\Leftrightarrow & 2\cos^2x+9\cos x+4=0\\
	&\Leftrightarrow& \hoac{&\cos x=-4\ (\text{VN})\\&\cos x=-\dfrac{1}{2}}\\
	&\Leftrightarrow & x=\pm\dfrac{2\pi}{3}+k2\pi,\ k\in\mathbb{Z}.
	\end{eqnarray*}
	Vậy phương trình có nghiệm $x=\pm\dfrac{2\pi}{3}+k2\pi,\ k\in\mathbb{Z}$.
	\item Ta có
	\begin{eqnarray*}
	\sqrt{3}\sin 4x+\cos 4x-\sqrt{3}=0 
	&\Leftrightarrow &\dfrac{\sqrt{3}}{2}\sin 4x+\dfrac{1}{2}\cos 4x=\dfrac{\sqrt{3}}{2}\\
	&\Leftrightarrow &\sin \left(4x+\dfrac{\pi}{6}\right)=\sin \dfrac{\pi}{3}\\
	&\Leftrightarrow &\hoac{&4x+\dfrac{\pi}{6}=\dfrac{\pi}{3}+k2\pi\\& 4x+\dfrac{\pi}{6}=\dfrac{2\pi}{3}+k2\pi}\\
	&\Leftrightarrow& \hoac{&x=\dfrac{\pi}{24}+k\dfrac{\pi}{2}\\&x=\dfrac{\pi}{8}+k\dfrac{\pi}{2}}, k\in\mathbb{Z}.
	\end{eqnarray*}
	Vậy phương trình có nghiệm $\hoac{&x=\dfrac{\pi}{24}+k\dfrac{\pi}{2}\\&x=\dfrac{\pi}{8}+k\dfrac{\pi}{2}}, k\in\mathbb{Z}.$.
	\end{enumerate}
	}
\end{bt}
\begin{bt}%[Đề HKI, Bình Hưng Hòa, TP Hồ Chí Minh 2017]%[Hung Tran, 11-HK1-18]%[1D2B2-1]
	Từ các chữ số $0,1,2,3,5,7,8$ có thể lập được bao nhiêu số tự nhiên gồm ba chữ số khác nhau và chia hết cho $5$?
	\loigiai{
	Gọi số cần tìm có dạng: $\overline{abc}$ với $a,b,c$ đôi một khác nhau thuộc tập $A=\left\{0,1,2,3,5,7,8\right\}$.\\
	Với $c=0\Rightarrow \overline{ab}$ có $\mathrm{A}_6^2=30$ số.\\
	Với $c=5\Rightarrow \overline{ab}$ có $\mathrm{A}_6^2-\mathrm{A}_5^1=25$ số.\\
	Theo quy tắc cộng có $30+25=55$ số cần tìm.
	}
\end{bt}
\begin{bt}%[Đề HKI, Bình Hưng Hòa, TP Hồ Chí Minh 2017]%[Hung Tran, 11-HK1-18]%[1D2B3-2]
	Tìm số hạng không chứa $x$ trong khai triển $\left(x^4-\dfrac{3}{2x^3}\right)^{14}$ với $x\ne 0$.
	\loigiai{
	Số hạng tổng quát trong khai triển: $\mathrm{C}_{14}^k\left(x^4\right)^{14-k}\cdot \left(-\dfrac{3}{2x^3}\right)^k=\mathrm{C}_{14}^k \left(-\dfrac{3}{2}\right)^k x^{56-7k}$.\\
	Hệ số không chứa $x$ trong khai triển ứng với $k=8$.\\
	Do đó số hạng cần tìm là $\mathrm{C}_{14}^k\left(-\dfrac{3}{2}\right)^8=\dfrac{19702683}{256}$.
	}
\end{bt}
\begin{bt}%[Đề HKI, Bình Hưng Hòa, TP Hồ Chí Minh 2017]%[Hung Tran, 11-HK1-18]%[1D2B2-6]
	Tìm số nguyên dương $n$ thỏa mãn $2\mathrm{C}_{n+1}^1-\mathrm{A}_n^2+128=0$.
	\loigiai{
	Điều kiện: $n\in\mathbb{N}, n\ge 2$.
	\begin{eqnarray*}
	2\mathrm{C}_{n+1}^1-\mathrm{A}_n^2+128=0
	&\Leftrightarrow & 2\cdot \dfrac{(n+1)!}{n!}-\dfrac{n!}{(n-2)!}+128=0\\
	&\Leftrightarrow & 2(n+1)-n(n-1)+128=0\\
	&\Leftrightarrow & n^2-3n-130=0\\
	&\Leftrightarrow & \hoac{&n=13\\&n=-10 (\text{L})}.
	\end{eqnarray*}
	Vậy $n=13$ thỏa mãn yêu cầu bài toán.
	}
\end{bt}
\begin{bt}%[Đề HKI, Bình Hưng Hòa, TP Hồ Chí Minh 2017]%[Hung Tran, 11-HK1-18]%[1D2B5-2]
	Một hộp chứa $7$ viên bi đỏ, $8$ viên bi trắng, $6$ viên bi vàng và tất cả các viên bi đều phân biệt. Lấy ngẫu nhiên trong hộp ra $4$ viên bi, tính xác suất để chọn được $4$ viên bi trong đó có nhiều nhất $2$ viên bi vàng.
	\loigiai{
	Số phần tử của không gian mẫu: $n\left(\Omega\right)=\mathrm{C}_{21}^4=5985$.\\
	Chọn được $0$ bi vàng và $4$ viên bi khác có: $\mathrm{C}_6^0\cdot \mathrm{C}_{15}^4$ cách.\\
	Chọn được $1$ bi vàng và $3$ viên bi khác có: $\mathrm{C}_6^1\cdot\mathrm{C}_{15}^3$ cách.\\
	Chọn được $2$ bi vàng và $2$ bi khác có: $\mathrm{C}_6^2\cdot \mathrm{C}_{15}^2$ cách.\\
	Gọi $A$ là biến cố: ``Chọn được $4$ viên bi trong đó có nhiều nhất $2$ viên bi vàng''.\\
	$\Rightarrow n(A)=\mathrm{C}_6^0\cdot \mathrm{C}_{15}^4+\mathrm{C}_6^1\cdot\mathrm{C}_{15}^3+\mathrm{C}_6^2\cdot \mathrm{C}_{15}^2=5670$.\\
	$P(A)=\dfrac{n(A)}{n(\Omega)}=\dfrac{5670}{5985}=\dfrac{18}{19}$.
	}
\end{bt}

\begin{bt}%[Đề HKI, Bình Hưng Hòa, TP Hồ Chí Minh 2017]%[Hung Tran, 11-HK1-18]%[1D2K3-2]
	Cho khai triển $\left(1+2x\right)^n=a_0+a_1x+a_2x^2+\cdots +a_nx^n$, trong đó $n\in\mathbb{N^*}$ và các hệ số thỏa mãn hệ thức $\dfrac{a_1}{2}+\dfrac{a_2}{4}+\cdots +\dfrac{a_n}{2^n}=4095$. Tìm số lớn nhất trong các số $a_0,a_1,a_2,\cdots ,a_n$.
	\loigiai{
	Ta có: $\left(1+2x\right)^n=\displaystyle\sum\limits_{k=0}^n\mathrm{C}_n^k 2^kx^k=a_0+a_1x+a_2x^2+\cdots +a_nx^n$.\\
	Hệ số ứng với $x^k$ là $a_k=\mathrm{C}_n^k 2^k$.\\
	Theo giả thiết 
	\begin{eqnarray*}
	\dfrac{a_1}{2}+\dfrac{a_2}{4}+\cdots +\dfrac{a_n}{2^n}=4095
	&\Leftrightarrow & \dfrac{\mathrm{C}_n^1\cdot 2}{2}+\dfrac{\mathrm{C}_n^2\cdot 2^2}{2^2}+\cdots +\dfrac{\mathrm{C}_n^n\cdot 2^n}{2^n}=4095\\
	&\Leftrightarrow &2^n=4096\Leftrightarrow n=12.
	\end{eqnarray*}
	$\circ$ Với $a_k>a_{k+1}\Leftrightarrow \dfrac{\mathrm{C}_{12}^k\cdot 2^k}{\mathrm{C}_{12}^{k+1}\cdot 2^{k+1}}>1\Leftrightarrow \dfrac{k+1}{2(12-k)}>1\Leftrightarrow k>\dfrac{23}{3}$.\\
	$\circ$ Với $a_k<a_{k+1}\Leftrightarrow \dfrac{\mathrm{C}_{12}^k\cdot 2^k}{\mathrm{C}_{12}^{k+1}\cdot 2^{k+1}}<1\Leftrightarrow \dfrac{k+1}{2(12-k)}<1\Leftrightarrow k<\dfrac{23}{3}$.\\
	Nghĩa là hệ số trong khai triển tăng dần với $k<\dfrac{23}{3}$ và giảm dần với $k>\dfrac{23}{3}$.\\
	Vậy hệ số lớn nhất trong khai triển là $a_8=\mathrm{C}_{12}^82^8=126720$.
	}
\end{bt}
