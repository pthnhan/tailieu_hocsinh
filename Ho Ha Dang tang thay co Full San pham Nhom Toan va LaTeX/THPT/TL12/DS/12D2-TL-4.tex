\subsection{Các dạng toán}
\begin{dang}{Tính giới hạn liên quan đến hàm số mũ và hàm số logarit}
Sử dụng các giới hạn sau
\begin{itemize}
\item $\displaystyle \lim_{x \to 0}\dfrac{\ln(1+x)}{x}=1$.
\item $\displaystyle \lim_{x \to 0}\dfrac{e^x-1}{x}=1$.
\end{itemize}
\end{dang}
\begin{vd}%[Đinh Bích Hảo]%[2D2B5]
Tính các giới hạn sau\\
\begin{enumerate}
        \item $\displaystyle \lim_{x \to 0}\dfrac{e^{3x}-1}{x}$. 
        \item $\displaystyle \lim_{x \to 0}\dfrac{e^{2x}-e^{3x}}{5x}$.
    \end{enumerate}
\loigiai{
\begin{enumerate}
         \item $\displaystyle \lim_{x \to 0}\dfrac{e^{3x}-1}{x}=3.\displaystyle \lim_{x \to 0} \dfrac{e^{3x}-1}{3x}=3$.
        \item $\displaystyle \lim_{x \to 0}\dfrac{e^{2x}-e^{3x}}{5x}=\displaystyle \lim_{x \to 0} \dfrac{e^{2x}-1}{2x} \cdot \dfrac{2}{5} -\displaystyle \lim_{x \to 0} \dfrac{e^{3x}-1}{3x} \cdot \dfrac{3}{5}=\dfrac{2}{5}-\dfrac{3}{5}=-\dfrac{1}{5}$.
\end{enumerate}

}
\end{vd}

\begin{vd}%[Đinh Bích Hảo]%[2D2B4]
Tính giới hạn 
\begin{enumerate}
        \item $\displaystyle \lim_{x \to 0} \dfrac{\ln(4x+1)}{x}$ 
        \item $\displaystyle \lim_{x \to 0}\dfrac{\ln(1+3x)}{\sin 2x}$.
    \end{enumerate}
\loigiai{
\begin{enumerate}
        \item $\displaystyle \lim_{x \to 0} \dfrac{\ln(4x+1)}{x}=4 \cdot \displaystyle \lim_{x \to 0} \dfrac{\ln(4x+1)}{4x}=4$. 
        \item $\displaystyle \lim_{x \to 0}\dfrac{\ln(1+3x)}{\sin 2x}=\displaystyle \lim_{x \to 0}\dfrac{\dfrac{\ln(1+3x)}{3x} }{\dfrac{\sin 2x}{2x}} \cdot \dfrac{3}{2}=\dfrac{3}{2}$.
    \end{enumerate}
    }
\end{vd}

\begin{dang}{Các bài toán liên quan đến đạo hàm hàm số mũ và hàm số logarit}
Sử dụng các công thức sau
\begin{itemize}
\item $(a^x)'=a^x \ln a$ và $\left(a^{u(x)}\right)'=u'(x) a^{u(x)} \ln a$.
\item $\left(\log_a x\right)'=\dfrac{1}{x \ln a}$ và $\left(\log_a u(x) \right)'=\dfrac{u'(x)}{u(x) \ln a}$.
\end{itemize}
\end{dang}

\begin{vd}%[BanNt]%[2D2K4]

\loigiai{
Gọi $P_n$ là số tiền còn lại sau tháng thứ $n$.
Sau tháng thứ nhất, số tiền gốc và lãi là: $a+ar=a(1+r)=ad$ với $d=1+r$.
Trả $x$ đồng thì số tiền còn lại sau tháng thứ nhất là: $P_1=ad-x=ad-x\dfrac{d-1}{d-1}$.
Sau tháng thứ hai số tiền gốc và lãi là: $ad-x+(ad-x)(1+r)=(ad-x)d$.
Trả $x$ đồng thì số tiền còn lại sau tháng thứ hai là: $$P_2=(ad-x)d-x=ad^2-xd-x=ad^2-x(d+1)=ad^2-x\dfrac{d^2-1}{d-1}.$$
Sau tháng thứ ba, số tiền gốc và lãi là: $$ad^2-x(d+1)+[ad^2-x(d+1)]r=[ad^2-x(d+1)](1+r)=[ad^2-xx(d+1)]d$$
Trả $x$ đồng thì số tiền còn lại sau tháng thứ ba là: $$ P_3=[ad^2-x(d+1)]d-x=ad^3-xd^2-xd-x=ad^3-x(d^2+d+1)=[ad^2-x(d+1)]d.$$
Bằng quy nạp toán học ta suy ra số tiền còn lại sau tháng thứ $n$ là: $$P_n=ad^n-x\dfrac{d^n-1}{d-1}\Leftrightarrow P_n=a(1+r)^n-x\dfrac{(1+r)^n-1}{r}\hspace{0.5cm} \text{với}\, d=1+r.$$
Do sau tháng thứ $n$ người vay tiền đã trả hết số tiền đã vay ta có: $$P_n=0\Leftrightarrow ad^n-x\dfrac{d^n-1}{d-1}=0\Leftrightarrow x=\dfrac{ad^n(d-1)}{d^n-1}.$$
Khi đó $x=\dfrac{a(1+r)^nr}{(1+r)^n-1}$.
}
\end{vd}

\begin{bt}%[BanNt]%[2D2K4]
Số lượng của một loài vi khuẩn sau $t$ giờ được xấp xỉ bởi đẳng thức $Q(t)=Q_0\cdot e ^{0{,}195 t}$, trong đó $Q_0$ là số lượng vi khuẩn ban đầu. Nếu số lượng vi khuẩn ban đầu là $5000$ con thì sau bao nhiêu giờ, số lượng vi khuẩn có $100000$ con?
\loigiai{
Từ giả thiết suy ra $Q(t)=5000 e^{0{,}195t}$.\\ Để số lượng vi khuẩn là $100000$ con thì $5000e^{0{,}195t}=100000\Leftrightarrow e^{0{,}195t}=2\Leftrightarrow t=\dfrac{1}{0{,}195}\ln 20\approx 15{,}36$ (h).
}
\end{bt}

\begin{bt}%[BanNt]%[2D2K4]
Theo số liệu của Tổng cục thống kê, năm $2016$ dân số Việt Nam ước tính $94444200$ người. Tỉ lệ tăng dân số hàng năm ở Việt Nam được duy trì ở mức $1{,}07\%$. Cho biết sự tăng dân số được tính theo công thức $S=A\cdot e^{Nr}$ (trong đó $A$ là dân số của năm lấy làm mốc tính, $S$ là dân số sau $N$ năm, $r$ là tỉ lệ tăng dân số hàng năm). Cứ tăng dân số với tỉ lệ như vậy thì đến năm nào, dân số nước ta ở mức $120$ triệu người?
\loigiai{
Gọi $n$ là số năm để dân số đạt mức $120$ triệu người tính mốc từ năm $2016$.\\
Ta có $120000000=94444200e^{n\cdot0{,}0107}\Rightarrow n\approx \dfrac{\ln 1{,}27}{0{,}0107}\approx 22{,}34$.\\
Vậy trong năm thứ $23$ (tức là năm $2016+23=2039$) thì dân số đạt mức $120 $ triệu người.
}
\end{bt}

\begin{bt}%[BanNt]%[2D2K4]
Sự tăng trưởng của loại vi khuẩn tuân theo công thức $S=A e^{rt}$, trong đó $A$ là số lượng vi khuẩn ban đầu, $r$ là tỉ lệ tăng trưởng ($r>0$), $t$ là thời gian tăng trưởng (tính theo đơn vị là giờ). Biết số vi khuẩn ban đầu là $100$ con và sau $5$ giờ có $300$ con. Thời gian để vi khuân tăng gấp đôi ban đầu là bao lâu?
\loigiai{
Ta có $300=100e^{5r}\Leftrightarrow e^{5r}=3\Leftrightarrow 5r=\ln 3\Leftrightarrow r=\dfrac{\ln 3}{5}$.

Theo yêu cầu bài toán ta lại có $$200=100e^{rt}\Leftrightarrow e^{rt}=2\Leftrightarrow rt=\ln 2 \Leftrightarrow t=\dfrac{5\ln 2}{\ln 3}\approx 3{,}15 \, (\text{h})$$
}
\end{bt}

\begin{bt}%[BanNt]%[2D2K4]
Một người gửi ngân hàng $100$ triệu theo thể thức lãi kép, lãi suất $0.5\%$ một tháng. Sau ít nhất bao nhiêu tháng, người đó có $125$ triệu?
\loigiai{
Sau một tháng, người đó nhận được $100+100\cdot 0{,}5 \%=100\cdot 1{,}005$ (triệu đồng).\\
Sau hai tháng, người đó nhận được: \\
$100\cdot 1{,}005+100\cdot 1{,}005\cdot 0{,}005=100\cdot1{,}005(1+0{,}005)=100\cdot 1{,}005^2$ triệu đồng.\\
Sau $n$ tháng, người đó nhận được $100\cdot 1{,}005^n$ triệu đồng.\\
Theo đề $100\cdot 1{,}005^n=125 \Rightarrow n=44{,}7$.\\
Do $n$ nguyên nên suy ra $n=45$ tháng.
}
\end{bt}

\begin{bt}%[BanNt]%[2D2K4]
Năm $2014$, một người đã tiết kiệm được $x$ triệu đồng và dùng số tiền đó để mua nhà nhưng trên thực tế người đó phải cần $1{,}55 x$ triệu đồng. Người đó quyết định gửi tiết kiệm vào ngân hàng với lãi suất $6{,}9\%$ một năm, theo hình thức lãi kép và không rút trước kì hạn. Hỏi năm nào người đó mua được căn nhà đó (giả sử rằng, giá bán nhà đó không thay đổi).
\loigiai{
Số tiền tiết kiệm sau n năm là $x(1+6{,}9 \%)^n$.\\
Ta cần tìm $n$ để $x(1+6{,}9\%)^n=1{,}55 x\Leftrightarrow (1+6{,}9\%)^n=1{,}55 \Leftrightarrow n\approx 6{,}56$.\\
Do đó, người gửi tiết kiệm cần gửi trọn $7$ kì hạn, tức là $7$ năm.\\
Vậy đến năm $2021$ người đó sẽ có đủ tiền cần thiết.
}
\end{bt}
\begin{bt}%[BanNt]%[2D2K4]
Một người gửi tiết kiệm với lãi suất $6{,}5\%$ năm và lãi hàng năm được nhập vào vốn. Hỏi khoảng bao nhiêu năm người đó thu được gấp đôi số tiền ban đầu?
\loigiai{
Gọi $x$ là số tiền gửi ban đầu.
Giả sử sau $n$ năm số tiền vốn và lãi là $2x$.\\
Ta có $2x=x\cdot 1{,}065^n\Leftrightarrow 1{,}065 ^n =2 \Leftrightarrow n= \log _2 1{,}065 \Leftrightarrow n\approx 11.$
}
\end{bt}
