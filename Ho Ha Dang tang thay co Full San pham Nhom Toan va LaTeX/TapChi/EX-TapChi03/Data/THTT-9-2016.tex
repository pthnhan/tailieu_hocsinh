	Cho các số nguyên $a$, $b$, $c$ thỏa mãn $a\cdot b\cdot c=2015^{2016}$. Tìm số dư của phép chia tổng $19a^2+5b^2+1890c^2$ cho $24$.
	\loigiai{
	Số $2015^{2016}$ là số lẻ mà các số $a$, $b$, $c$ đều là ước số của số $2015^{2016}$ nên $a$, $b$, $c$ đều là số lẻ.\\ 
	Đặt $a=2k+1$ thì $a^2-a=(a-1)(a+1)=2k(2k+2)=4k(k+1)$ chia hết cho $8$ (vì $k(k+1)$ là tích của hai số nguyên liên tiếp nên chia hết cho $2$).\\ 
	Tương tự $b^2-1$ và $c^2-1$ đều chia hết cho $8$.\hfill$(1)$\\
	Số $2015$ không chia hết cho $3$ vì tổng các chữ số bằng $8$, do đó số $2015^{2016}$ không chia hết cho $3$. Nếu $a=3k+1$ thì $a-1=3k$, nếu $a=3k+3$ thì $a+1=3(k+1)$, do đó $a^2-1=(a-1)(a+1)$ chia hết cho $3$.\\ 
	Tương tự $b^2-1$ và $c^2-1$ đều chia hết cho $3$.\hfill$(2)$\\
	Từ $(1)$ và $(2)$ với chú ý rằng $(3,8)=1$ suy ra $a^2-1$, $b^2-1$, $c^2-1$ đều chia hết cho $8\cdot 3=24$\hfill$(3)$\\
	Từ $(3)$ và $19a^2+5b^2+1890c^2=19(a^2-1)+5(b^2-1)+1890(c^2-1)+1914$ suy ra số dư của $19a^2+5b^2+1890c^2$ khi chia cho $24$ cũng là số dư của $1914$ chia cho $24$, và bằng $18$.
	\begin{nx}
		Mấu chốt để giải bài này là kết quả $(3)$, một số bạn không để ý điều đó nên giải khá dài hoặc đi lạc hướng khi khai thác điều kiện $a$, $b$, $c$ đều là ước số của số $2015^{2016}$.
	\end{nx}
}

	$$(a^2+ab+b^2)\,\,\vdots\,\, 10$$
	thì $(a^3-b^3)$ chia hết cho $1000$ ($a, b$ là các số nguyên).	
	
	\loigiai{Vì $(a^2+ab+b^2)$ chia hết cho $10$ nên $(a-b)(a^2+ab+b^2)$ chia hết cho $10$ hay $(a^3-b^3)$ chia hết cho $10$, suy ra $a^3$ và $b^3$ có chữ số tận cùng giống nhau\; $(1)$. Bằng cách lập bảng xét chữ số tận cùng (từ $0$ đến $9$) của các số tự nhiên và chữ số tận cùng của số lập phương tương ứng ta rút ra nhận xét:\textit{ Hai số nguyên có chữ số tận cùng khác nhau thì lập phương của chúng cũng có chữ số tận cùng khác nhau}. Do đó từ $(1)$ suy ra $a$ và $b$ có chữ số tận cùng giống nhau, khi đó ba số $a^2$, $ab$, và $b^2$ cũng có ba chữ số tận cùng giống nhau. Mà $(a^2+ab+b^2)$ có chữ số tận cùng là $0$ (giả thiết) nên $a^2$, $ab$, và $b^2$ đều có tận cùng là $0$, suy ra $a$ và $b$ có chữ số tận cùng là $0$. Từ đó $a^3$ và $b^3$ đều chia hết cho $1000$ nên $(a^3-b^3)$ chia hết cho $1000$.
	\begin{nx}
		Ngoài cách giải trên đây, đa số các bạn có lập luận như sau: Vì $(a^2+ab+b^2)$ chia hết cho $10$ nên $(a^2+ab+b^2)$ là số chẵn, suy ra $a$ và $b$ đồng thời là số chẵn.\\ Sau đó dựa vào nhận xét \lq\lq \textit{Một số chính phương chia hết cho $5$ chỉ cho số dư là $0; 1$ hoặc $4$}\rq\rq, viết lại $4(a^2+ab+b^2)=(2a+b)^2+3b^2$, từ giả thiết $(a^2+ab+b^2)$ chia hết cho $5$ suy ra đồng thời $(2a+b)^2$ và $3b^2$ cùng chia hết cho $5$ (vì $3b^2$ chia cho $5$ chỉ cho số dư là $0; 3$ hoặc $2$), suy ra $b^2$ (và $b$) chia hết cho $5$, từ đó $a^2$ (và $a$) chia hết cho $5$. Như vậy $a$ và $b$ cùng chia hết cho $10$ và có kết luận bài toán.  
	\end{nx}

	Giải phương trình $$\sqrt{2x-2}+\sqrt[3]{x-2}=\dfrac{9-x}{\sqrt[3]{8x-16}}\cdot$$	
	\loigiai{Điều kiện $ \heva{&2x-2\ge 0 \\&8x-16\ne 0} \Leftrightarrow \heva{&x\ge 0\\&x\ne 0.}$\\
		\textit{\textbf{Cách 1.}} Đặt $a=\sqrt{2x-2};b=\sqrt[3]{x-2}$  (với $a\ge 0,b\ne 0$) ta có  $a^2=2x-2=2(x-2)+2=2b^3+2$ nên $a^2=2b^3+2$.\hfill$(1)$\\
		Biến đổi phương trình đã cho thành $$\sqrt{2x-2}+\sqrt[3]{x-2}=\dfrac{16-(2x-2)}{4\sqrt[3]{x-2}}$$ 
		Nên $a+b=\dfrac{16-a^2}{4b}\Leftrightarrow (a+2b)^2=16\Leftrightarrow \left[\begin{matrix}
		a=4-2b  \\
		a=-4-2b.  \\
		\end{matrix}\right.$ \\
		Nếu $a=-4-2b$  thì do $a\ge 0$ nên $b\le-2$ ; từ (1) ta lại có  $2b^3=a^2-2\ge 2\Rightarrow b\ge-1$ (mâu thuẫn). Vậy $a=4-2b$. Thay vào (1) ta được 
		\[(4-2b)^2=2b^3+2\Leftrightarrow (b-1)(b^2-b+7)=0.\tag{2}\]
		Vì $b^2-b+7=(b-\dfrac{1}{2})^2+\dfrac{27}{4}>0$ nên từ (2) suy ra $b=1$, do đó $a=2$ .\\
		Từ đó tìm được $x=3$ (thỏa mãn điều kiện).\\
		\textit{\textbf{Cách 2.}} 
		Với $x=3$ thì\\ $VT=\sqrt{3.2-2}+\sqrt[3]{3-2}=2+1=3;$\\
		$VP=\dfrac{9-3}{\sqrt[3]{8.3-16}}=\dfrac{6}{2}=3$.\\ 
		Nhận thấy  VT $=$ VP, 
		nên $x=3$ là nghiệm của phương trình.
		\begin{itemize}
			\item Xét $x>3$ thì $\sqrt{2x-2}>\sqrt{2.3-2}=2;\sqrt[3]{x-2}>\sqrt[3]{3-2}=1$;  
			$9-x<9-3=6;\\ 0<\dfrac{1}{\sqrt[3]{8x-16}}<\dfrac{1}{\sqrt[3]{8.3-16}}=\dfrac{1}{2}$ 
			nên VT $>2+1=3$ và VP $<\dfrac{6}{2}=3.$ \\
			Do đó VT $>$ VP. 
			\item Xét $2<x<3$ thì $\sqrt{2x-2}<2;\sqrt[3]{x-2}<1$ ; $9-x>6;\dfrac{1}{\sqrt[3]{8x-16}}>\dfrac{1}{2}$, 
			nên VT $<3$ và VP $>3$. 
			Do đó VT $<$ VP.
			\item Xét $1\le x<2$ thì $\sqrt{2x-2}\ge 0;\sqrt[3]{x-2}\ge-1$; 
			nên VT $>-1; 9-x>7; -2\le \sqrt[3]{8x-16}<0$ 
			$\Rightarrow 0<-\sqrt[3]{8x-16}<2\Rightarrow-\dfrac{1}{\sqrt[3]{8x-16}}>\dfrac{1}{2}$ 
			$\Rightarrow-\dfrac{9-x}{\sqrt[3]{8x-16}}>\dfrac{7}{2}\\ \Rightarrow VP=\dfrac{9-x}{\sqrt[3]{8x-16}}<-\dfrac{7}{2}<-1$.  
			Do đó VT $>$ VP.
		\end{itemize} 
		Vậy phương trình đã cho có nghiệm duy nhất $x=3.$ 
	}	
\end{bt}

	Cho tam thức 
	$$f(x)=20x^2-11x+2016.$$
	Chứng minh rằng tồn tại số nguyên $\alpha$ sao cho $f(\alpha)\vdots 2^{20^{11^{1960}}}$.	
	\loigiai{
		\textit{\textbf{Cách 1:}} Ta sẽ chứng minh bằng quy nạp bài toán tổng quát: Với mọi số nguyên dương $n$ thì tồn tại số nguyên $\alpha_n$ sao cho $f\left(\alpha_n\right)\vdots 2^n$.\hfill$(\star)$\\
		Với $n=1$ ta lấy $\alpha_1=2$ thì $f(2)\vdots 2^1$.\\
		Giả sử bài toán đúng với $n=k$, nghĩa là tồn tại số nguyên $\alpha_k$ sao cho $\beta=f\left(\alpha_k\right)\vdots 2^k$.\\
		Ta sẽ chứng minh bài toán đúng với $n=k+1$. Chọn $\alpha_{k+1}=\alpha_k+\beta$, ta thu được 
		$$ f\left(\alpha_{k+1}\right)=10\beta\left(2\beta+4\alpha_k-1\right)\vdots 2^{k+1} $$
		Bài toán tổng quát $(\star)$ được chứng minh. Do đó tồn tại số nguyên $\alpha_n$ sao cho $f\left(\alpha_n\right)\vdots 2^n$.\\
		\textbf{\textit{Cách 2:}} Ta chứng minh kết quả tổng quát hơn sau: Với mọi số nguyên dương chẵn $n$ thì tồn tại số nguyên $\alpha$ sao cho 
			$f(\alpha)\vdots n$.\hfill$(\star\star)$.\\
			Chia $f(1), f(2).\ldots, f(n)$ cho $n$ sẽ có $n$ số dư. Giả sử trong đó có hai số có cùng số dư khi chia cho $n$ là $f(a)$ và $f(b) \left(a>b; a, b\in\left\{1,\ldots,n\right\}\right)$. Khi đó $f(a)-f(b)\vdots n$. Ta có		
		\begin{align*}
			f(a)-f(b)&=20\left(a^2-b^2\right)-11\left(a-b\right)\\ 
			&=\left(a-b\right)\left[20\left(a+b\right)-11\right]\vdots n
		\end{align*}
		Do $n$ là số chẵn, $20(a+b)-11$ là số lẻ nên $a-b\vdots n$, suy ra $a-b\geq n$.\\
		Mặt khác, do $a, b\in\left\{1,\ldots,n\right\}$ nên $a-b\leq n-1$, điều này mâu thuẫn với $a-b\geq n$. Vậy giả sử trên là sai. 
		Do đó các số $f(1), f(2).\ldots, f(n)$ khi chia cho $n$ sẽ có $n$ số dư đôi một khác nhau. Suy ra tồn tại số nguyên $\alpha$ sao cho $f(\alpha)\vdots n$.\\
		Vậy $f(\alpha)\vdots n$ với $n=2^{20^{11^{1960}}}$.
	}

	Giải phương trình $$\dfrac{\log_2x}{x^4-10x^2+26}=\dfrac{5-x^2}{\log_2^2x+1}.$$
	\loigiai{Điều kiện: $x>0$. Ta có
		{\allowdisplaybreaks
		\begin{align*}
			\dfrac{\log_2x}{x^4-10x^2+26}=\dfrac{5-x^2}{\log_2^2x+1}&\Leftrightarrow \log_2x\left(\log_2^2x+1\right)=\left(5-x^2\right)\left[\left(5-x^2\right)^2+1\right]\\ 
			&\Leftrightarrow \log_2^3x+\log_2x=\left(5-x^2\right)^3+5-x^2.\tag{1}
		\end{align*}}Xét hàm số $f(t)=t^3+t, t\in\mathbb{R}$.\\
		Ta có $f'(t)=3t^2+1>0$ nên hàm số đồng biến trên $\mathbb{R}$.\\
		Từ $(1)$ có $f\left(\log_2x\right)=f\left(5-x^2\right)\Rightarrow \log_2x=5-x^2\Leftrightarrow x^2+\log_2x-5=0$.\hfill$(2)$\\
		Xét hàm số $g(x)=x^2+\log_2x-5$ có $g'(x)=2x+\dfrac{1}{x\ln 2}>0, \forall x>0$.\\
		Hàm số $g(x)$ đồng biến trên khoảng $(0;+\infty)$. Mà $g(2)=0$ nên phương trình $(2)$ có nghiệm duy nhất $x=2$ (thỏa ĐK).
		Vậy phương trình ban đầu có nghiệm duy nhất.
\begin{nx}
		Rất nhiều bài toán về giải phương trình sử dụng mệnh đề sau: Nếu hàm số $f(t)$ luôn đồng biến (hoặc nghịch biến)  trong khoảng $(a;b)$ và $u(x), v(x)\in (a;b)$ thì $$f\left(u(x)\right)=f\left(v(x)\right)\Leftrightarrow u(x)=v(x).$$

	Cho các số thực dương $x,y$ thỏa mãn $\left(\sqrt{x}+1\right)\left(2\sqrt{y}+4\right)+y \geqslant 13$.
	Tìm giá trị nhỏ nhất của biểu thức $P=\dfrac{x^4}{y}+\dfrac{y^3}{x}+y$.
	
	\loigiai{
		Áp dụng bất đẳng thức Cauchy cho hai, ba số thực dương ta có:
		{\allowdisplaybreaks
			\begin{align*}
			13	&\leqslant \left(\sqrt{x}+1\right)\left(2\sqrt{y}+4\right)+y\\
			&=2\sqrt{xy}+2\sqrt{y}+4\sqrt{x}+4+y\\
			&\leqslant (x+y)+(y+1)+2(x+1)+4+y
			\end{align*}}
		Suy ra, $6 \leqslant 3(x+y) \,\Leftrightarrow\, 2 \leqslant x+y$.\\
		Ta có
		{\allowdisplaybreaks
			\begin{align*}
			P&=\dfrac{x^4}{y}+\dfrac{y^3}{x}+y=\left(\dfrac{x^4}{y}+y\right)+\left(\dfrac{y^3}{x}+x+1\right)-x-1\\
			&\geqslant 2\sqrt{\dfrac{x^4}{y}\cdot y} + 3\sqrt[3]{\dfrac{y^3}{x}\cdot x \cdot 1}-x-1\\
			&=2x^2+3y-x-1=2(x^2+1)+3y-x-3\\
			&\geqslant 2\cdot 2x+3y-x-3=3(x+y)-3 \geqslant 3\cdot 2-3 = 3.
			\end{align*}}
		\begin{nx}
			Bài toán bất đẳng thức trên thuộc dạng cơ bản, đơn giản của lớp 9, lớp 10.
		\end{nx}
	}

	Cho tam giác nhọn $ABC$. Chứng minh rằng
	\[\sqrt[3]{\cot{A}+\cot{B}}\geqslant \dfrac{16}{3}\cot{A}\cdot\cot{B}\cdot\cot{C}.\]\vspace*{-12pt}	
	\loigiai{
		Chú ý đến đẳng thức
		\[\cot{A}\cdot\cot{B}+\cot{B}\cdot\cot{C}+\cot{C}\cdot\cot{A}=1.\]
		Đặt $x=\cot{A}$, $y=\cot{B}$ $(x,y>0)$ thì $\cot{C}=\dfrac{1-xy}{x+y} \,\Rightarrow\, 1-xy>0$.\\
		Ta cần chứng minh bất đẳng thức:
		\[\sqrt[3]{x+y} \geqslant \dfrac{16}{3}xy \cdot \dfrac{1-xy}{x+y}\]
		hay \[3^3\cdot (x+y)^4 \geqslant 16^3 x^3y^3(1-xy)^3 \tag{*}\]
		Sử dụng bất đẳng thức Cauchy cho $2$ số dương, rồi bất đẳng thức Cauchy cho $4$ dương, ta có:
		{\allowdisplaybreaks
			\begin{align*}
			3^3\cdot (x+y)^4 \geqslant 3^3\cdot 16x^2y^2&=\dfrac{1}{3}\cdot 16 x^2y^2\left[3xy+(1-xy)+(1-xy)+(1-xy)\right]^4\\
			&\geqslant \dfrac{1}{3}\cdot 16 x^2y^2\cdot 4^4\cdot 3xy(1-xy)^3=16^3x^3y^3(1-xy)^3.
			\end{align*}}
		Vậy bất đẳng thức $(*)$ được chứng minh, do đó bất đẳng thức ở đề bài được chứng minh. Đẳng thức xảy ra khi và chỉ khi $\heva{&x=y\\&3xy=1-xy} \Leftrightarrow x=y=\dfrac{1}{2}$ hay $\cot{A}=\cot{B}=\dfrac{1}{2}$ và $\cot{C}=\dfrac{3}{4}$.
	}
\end{bt}

\begin{bt}[T9/471-Toán học \& Tuổi trẻ, tháng 9 năm 2016]%[Lê Mạnh Thắng,Dự án EX_TapChi03]%[0D0G2]
	Cho ba số thực dương $a,b,c$ sao cho $a+b+c=abc$. Chứng minh bất đẳng thức
	\[\sqrt{(1+a^2)(1+b^2)}+\sqrt{(1+b^2)(1+c^2)}+\sqrt{(1+c^2)(1+a^2)}-\sqrt{(1+a^2)(1+b^2)((1+c^2)}\geqslant 4.\]\vspace*{-12pt}	
	\loigiai{
		Ta có\vspace*{-12pt}
		{\allowdisplaybreaks
			\begin{align*}
			(1+a^2)(1+b^2)(1+c^2)&=(1+a^2b^2+b^2c^2+c^2a^2)+(a^2+b^2+c^2+a^2b^2c^2)\\
			&=(ab+bc+ca-1)^2+(a+b+c-abc)^2.
			\end{align*}}
		Do $a+b+c=abc$ nên
		\[(1+a^2)(1+b^2)(1+c^2)=(ab+bc+ca-1)^2 \tag{1}\]
		Vì $\heva{&(a+b+c)\left(\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}\right)\geqslant9\\&a+b+c=abc}$ nên
		\[ab+bc+ca\geqslant9 \,\Rightarrow\, ab+bc+ca-1>0.\]
		Do đó từ $(1)$ suy ra
		\[\sqrt{(1+a^2)(1+b^2)(1+c^2)}=ab+bc+ca-1.\]
		Vậy đẳng thức cần chứng minh trở thành
		\[\sqrt{(1+a^2)(1+b^2)}+\sqrt{(1+b^2)(1+c^2)}+\sqrt{(1+c^2)(1+a^2)} \geqslant ab+bc+ca+3 \tag{2}\]
		Ta có 
		\[(1+a^2)(1+b^2)=(ab+1)^2+(a-b)^2 \geqslant (ab+1)^2 \,\Rightarrow\, \sqrt{(1+a^2)(1+b^2)}\geqslant ab+1.\]
		Tương tự
		{\allowdisplaybreaks
			\begin{align*}
			&\sqrt{(1+b^2)(1+c^2)}\geqslant bc+1\\
			&\sqrt{(1+c^2)(1+a^2)}\geqslant ca+1.
			\end{align*}}
		Từ đó suy ra $(2)$. Đẳng thức xảy ra khi và chỉ khi $\heva{&a=b=c\\&a+b+c=abc} \Leftrightarrow\, a=b=c=\sqrt{3}.$
	}

\begin{bt}[T10/471-Toán học \& Tuổi trẻ, tháng 9 năm 2016]%[Lê Mạnh Thắng,Dự án EX_TapChi03]%[1D0G6]
	Một trung tâm mở ra $7$ lớp học hè. Biết rằng mỗi học sinh trong trung tâm tham gia ít nhất một trong các lớp học này. Số liệu thống kê sau $7$ lớp học cho thấy mỗi lớp học có số học sinh tham dự bằng nhau và bằng $40$. Ngoài ra cứ hai lớp học bất kỳ thì có không quá $9$ học sinh cả hai lớp đó. Chứng minh rằng trung tâm dạy học có ít nhất $120$ học sinh tham dự các lớp học đó.	
	\loigiai{
		Ký hiệu $n$ là số học sinh tham dự các lớp học đó và với học sinh thứ $j$ ký hiệu $a_j$ là số lớp học học sinh đó tham gia. Ta lập bảng có $7$ hàng và $n$ cột. Ô $(i,j)$ của dòng $i$ cột $j$ được đánh dấu $\times$ nếu học sinh $j$ tham gia lớp $i$.\\
		Tổng số các dấu $\times$ nếu tính theo cột là $a_1+a_2+\cdots+a_n$, và tổng này là
		$7\cdot 40 = 280$ nếu tính theo dòng (vì mỗi lớp có đúng $40$ học sinh tham gia). Bây giờ ta tính $S$ là số cặp $(\times,\times)$ theo cùng cột. Trên cột $j$ có $\mathrm{C}_{a_j}^{2}$ cặp, cho nên số $S$ các cặp $(\times,\times)$ theo cột là
		\[\mathrm{C}_{a_1}^{2}+\mathrm{C}_{a_2}^{2}+\cdots+\mathrm{C}_{a_n}^{2}=\dfrac{1}{2}(a_1^2+a_2^2+\cdots+a_n^2)-\dfrac{1}{2}(a_1+a_2+\cdots+a_n).\]
		Thay $a_1+a_2+\cdots+a_n=280$ và áp dụng bất đẳng thức quen thuộc
		\[a_1^2+a_2^2+\cdots+a_n^2 \geqslant \dfrac{(a_1+a_2+\cdots+a_n)^2}{n},\]
		ta có \[S \geqslant \dfrac{280^2}{2n}-140 \tag{*}\]
		Mặt khác với mỗi cặp $2$ dòng, chỉ có nhiều lắm có $9$ cột có cặp $(\times,\times)$ trên $2$ dòng này, cho nên tổng số các cặp $(\times,\times)$ cùng cột trên tất cả $\mathrm{C}_{7}^{2}$ cặp $2$ dòng không vượt quá $9\mathrm{C}_{7}^{2}$ cặp $(\times,\times)$.\\
		Ta có bất đẳng thức
		\[9\mathrm{C}_{7}^{2}\geqslant S.\]
		Thay vào $(*)$, ta có $9\mathrm{\,C}_{7}^{2} \geqslant \dfrac{280^2}{2n}-140$ và có $n\geqslant \dfrac{280^2}{2(9\mathrm{C}_{7}^{2}+140)}>119$. Vậy $n \geqslant 120$.
	}
\end{bt}

	Tìm tất cả các hàm số liên tục $f: \mathbb R \to \mathbb R$ thoả mãn
	\begin{equation}
	f(xy)=f\left(\dfrac{x^2+y^2}{2}\right) + (x-y)^2, \, \forall x, y \in \mathbb R. \tag{1}
	\end{equation}
	\loigiai
	{
		Đặt $f(0)=c.$ Thay $y=0$ vào (1), ta thu được
		\begin{equation*}
		f(0)= f\left(\dfrac{x^2}{2}\right) + x^2, \, \forall x \in \mathbb R
		\end{equation*}
		hay \begin{equation*}
		f(t)= -2t +c, \, \forall t\geq 0. \tag{2}
		\end{equation*}
		Thay $y=1$ vào (1), ta thu được
		\begin{equation*}
		f(x)= f\left(\dfrac{x^2+1}{2}\right) + (x-1)^2, \, \forall x \in \mathbb R.\tag{3}
		\end{equation*}
		Từ (2) và (3) suy ra
		\begin{equation*}
		f(x)=f\left(\dfrac{x^2+1}{2}\right) + (x-1)^2= -2\cdot\dfrac{x^2+1}{2} + c + (x-1)^2= c- 2x, \, \forall x \in \mathbb R.
		\end{equation*}
		Thử lại ta thấy hàm $f(x)= c -2x$ thoả mãn các điều kiện bài ra.
		
		\begin{nx}
			Đây là dạng toán cơ bản về phương trình hàm với cặp biến tự do trên $\mathbb R.$ Rất nhiều bạn gửi bài giải về toà soạn và giải theo cách ở trên, tức là không sử dụng đến giả thiết của bài ra về tính liên tục của hàm $f(x).$
		\end{nx}
	}
