\begin{ex}%[Đề HK1, THPT Lý Thánh Tông, Hà Nội,2017 - 2018]%[1D1K2]

	}

\begin{ex}%[Đề HK1, THPT Lý Thánh Tông, Hà Nội,2017 - 2018]%[1D1K3]
	Họ nghiệm nào dưới đây là nghiệm của phương trình $\cos^2 {2x}+\cos {2x}-\dfrac{3}{4}=0$?
	\choice
	{$x=\pm \dfrac{2\pi}{3}+k\pi$}
	{$x=\pm \dfrac{\pi}{3}+k\pi$}
	{\True $x=\pm \dfrac{\pi}{6}+k\pi$}
	{$x=\pm \dfrac{\pi}{6}+k2\pi$}
	\loigiai{
		$\cos^2 {2x}+\cos {2x}-\dfrac{3}{4}=0 \Leftrightarrow \hoac{\cos {2x}&=\dfrac{1}{2}\\
			\cos {2x}&=-\dfrac{3}{4}.}$ \\
		Với $\cos {2x}=\dfrac{1}{2}\Leftrightarrow 2x=\pm \dfrac{\pi}{3}+k2\pi \Leftrightarrow x=\pm \dfrac{\pi}{6}+k\pi$, $k\in \mathbb{Z}$.\\
		Với $\cos2x=-\dfrac{3}{4}\Leftrightarrow x=\pm\dfrac{1}{2}\arccos\left(-\dfrac{3}{4}\right)+k\pi,\, k\in\mathbb{Z}.$	
	}	
\end{ex}

\begin{ex}%[Đề HK1, THPT Lý Thánh Tông, Hà Nội,2017 - 2018]%[1D2K3]
	Tìm hệ số không chứa $x$ trong khai triển biểu thức $P(x)=\left(2x-\dfrac{1}{x^2} \right)^6$.	
	\choice
	{\True $240$}
	{$250$}
	{$260$}
	{$270$}
	\loigiai{
		Số hạng tổng quát trong khai triển là $\mathrm{C}_6^k\cdot (2x)^{6-k}\cdot \left( \dfrac{-1}{x^2}\right)^k =(-1)^k\cdot \mathrm{C}_n^k\cdot 2^{6-k}\cdot x^{6-3k}$ với $0\le k\le 6$. \\
		Hệ số không chứa $x$ ứng với $6-3k=0\Leftrightarrow k=2$.\\
		Hệ số không chứa $x$ là $\mathrm{C}_6^2\cdot 2^4=240$.
	}	
\end{ex}

\begin{ex}%[Đề HK1, THPT Lý Thánh Tông, Hà Nội,2017 - 2018]%[1D2B4]
Gieo ngẫu nhiên một con xúc sắc cân đối và đồng chất $3$ lần. Khi đó $n(\Omega)$ bằng bao nhiêu?

\begin{ex}%[Đề HK1, THPT Lý Thánh Tông, Hà Nội,2017 - 2018]%[1D2K5]
Cho $P(A)=\dfrac{1}{3}$, $P(A\cup B)=\dfrac{1}{2}$. Biết $A$, $B$ là $2$ biến cố độc lập thì $P(B)$ bằng bao nhiêu?	
	\choice
	{$\dfrac{1}{3}$}
	{$\dfrac{1}{8}$}
	{\True $\dfrac{1}{4}$}
	{$\dfrac{3}{4}$}
	\loigiai{
	Ta có $P(A\cup B)=P(A)+P(B)-P(A\cap B)=P(A)+P(B)-P(A)\cdot P(B)=\dfrac{1}{3}+P(B)-\dfrac{1}{3}P(B)=\dfrac{1}{2}$\\
	Suy ra $P(B)=\dfrac{1}{4}$
	}	
\end{ex}

\begin{ex}%[Đề HK1, THPT Lý Thánh Tông, Hà Nội,2017 - 2018]%[1D2K5]
Trên một giá sách có $9$ quyển sách Văn, $6$ quyển sách Anh. Lấy lần lượt $3$ quyển và không để lại trên giá. Xác suất để lấy được $2$ quyển đầu là sách Văn và quyển thứ $3$ là sách Anh là bao nhiêu?	
	\choice
	{\True $\dfrac{72}{455}$}
	{$\dfrac{73}{455}$}
	{$\dfrac{74}{455}$}
	{$\dfrac{71}{455}$}
	\loigiai{
		Gọi biến cố $A$ là "Lấy được $2$ quyển đầu là sách Văn và quyển thứ $3$ là sách Anh".\\
		Số cách chọn lần lượt $3$ quyển sách trong $15$ quyển sách là $n(\Omega)=15\cdot 14\cdot 13$.\\
		Do lấy lần lượt các quyển sách nên có $9\cdot 8\cdot 6$ cách chọn $2$ quyển sách Văn và $1$ quyển sách Anh.\\
		Suy ra $P(A)=\dfrac{n(A)}{n(\Omega)}=\dfrac{9\cdot 8\cdot 6}{15\cdot 14\cdot 13}=\dfrac{72}{455}$.
		
	}	
\end{ex}

\begin{bt}%[Đề HK1, THPT Lý Thánh Tông, Hà Nội,2017 - 2018]%[1D2K3]
Biết rằng số $n$ nguyên dương thỏa mãn $\mathrm{C}_{n+1}^2+2\mathrm{C}_{n+2}^2+2\mathrm{C}_{n+3}^2+\mathrm{C}_{n+4}^2=149$. Tìm hệ số chứa $x^5$  trong khai triển biểu thức $\left(\dfrac{x^3}{2}+\dfrac{1}{x^2} \right)^n$.
	\loigiai{
	Áp dụng công thức tổ hợp và biến đổi ta được\\ $\mathrm{C}_{n+1}^2+2\mathrm{C}_{n+2}^2+2\mathrm{C}_{n+3}^2+\mathrm{C}_{n+4}^2=149 \Leftrightarrow n^2+4n-45=0 \Leftrightarrow \hoac{n&=5\\ n&=9 \text{(loại).}}$\\
	Khi đó số hạng tổng quát của khai triển $\left(\dfrac{x^3}{2}+\dfrac{1}{x^2} \right)^5$ là $\mathrm{C}_{5}^k\dfrac{x^{15-5k}}{2^{5-k}}$.\\
	Hệ số chứa $x^5$ ứng với $15-5k=5\Leftrightarrow k=2$. Từ đó, hệ số chứ $x^5$ là $\mathrm{C}_{5}^2\cdot \dfrac{1}{2^3}=\dfrac{5}{4}$.
	}
\end{bt}
