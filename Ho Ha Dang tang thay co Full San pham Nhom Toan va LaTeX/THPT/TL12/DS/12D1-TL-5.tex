\begin{dang}{Khảo sát sự biến thiên và vẽ đồ thị của hàm bậc ba}
	$$y=ax^3+bx^2+cx+d\quad (a\neq 0).$$
\end{dang}
\begin{vd}%[Khuất Văn Thanh]%[2D1B5]
	Khảo sát dự biến thiên và vẽ đồ thị của hàm số $y=x^3-3x^2+1$.
	\loigiai{
		\begin{itemize}
			\item Tập xác định: $\mathbb{R}$.
			\item Sự biến thiên\\
			$y'=3x^2-6x$, xác định với $\forall x\in\mathbb{R}$;\\
			$y'=0\Leftrightarrow \hoac{&x=0\\&x=2.}$\\
			$y'>0\Leftrightarrow x\in (-\infty;0)\cup (2;+\infty)$;\\
			$y'<0\Leftrightarrow x\in (0;2)$.\\
			Suy ra hàm số đồng biến trên các khoảng $(-\infty;0)$ và $(2;+\infty)$; nghịch biến trên $(0;2)$.\\
			Hàm số đạt cực đại tại $x=0; y_{\text{CĐ}}=y(0)=1$.\\
			Hàm số đạt cực tiểu tại $x=2; y_{\text{CT}}=y(2)=-3$.\\
			$\lim\limits_{x\to -\infty}y=\lim\limits_{x\to -\infty}x^3\left(1-\dfrac{3}{x}+\dfrac{1}{x^3}\right)=-\infty$.\\
			$\lim\limits_{x\to +\infty}y=\lim\limits_{x\to +\infty}x^3\left(1-\dfrac{3}{x}+\dfrac{1}{x^3}\right)=+\infty$.\\
			Bảng biến thiên\\

\begin{bt}%[Khuất Văn Thanh]%[2D1K5]
	Cho hàm số $y=x^3-2x^2+(1-m)x+m$ có đồ thị $(\mathscr{C}_m)$.
	\begin{enumerate}
		\item[a)] Khảo sát sự biến thiên và vẽ đồ thị của hàm số đã cho khi $m=1$.
		\item[b)] Tìm $m$ để đồ thị $(\mathscr{C}_m)$ cắt trục $Ox$ tại $3$ điểm phân biệt có hoành độ $x_1, x_2, x_3$ thỏa mãn $x_1^2+x_2^2+x_3^2<2018$. 
	\end{enumerate}
	\loigiai{
		a) Khi $m=1$ hàm số có dạng $y=x^3-2x^2+1$.
		\begin{itemize}
			\item 
			Tập xác định: $\mathbb{R}$.\\
			$y'=3x^2-4x=0\Leftrightarrow \hoac{&x=0\\&x=\dfrac{4}{3}}$.\\
			$y'>0\Leftrightarrow x\in (-\infty;0)\cup \left(\dfrac{4}{3};+\infty\right)$; $y'<0\Leftrightarrow x\in \left(0;\dfrac{4}{3}\right)$. Suy ra hàm số đồng biến trên các khoảng $(-\infty;0)$ và $\left(\dfrac{4}{3};+\infty\right)$, nghịch biến trên $\left(0;\dfrac{4}{3}\right)$.\\
			Hàm số đạt cực đại tại $x=0, y_{\text{CĐ}}=y(0)=1$.\\
			Hàm số đạt cực tiểu tại $x=\dfrac{4}{3}, y_{\text{CT}}=y\left(\dfrac{4}{3}\right)=-\dfrac{5}{27}$.\\
			$\lim\limits_{x\to -\infty}y=-\infty$, $\lim\limits_{x\to +\infty}y=+\infty$.\\
			Bảng biến thiên

				\tkzTabVar{-/$-\infty$ , +/$1$,-/$-\dfrac{5}{27}$, +/$+\infty$}%
				\end{tikzpicture}
			\end{center}
			\item Đồ thị
			\immini{
				Đồ thị cắt $Oy$ tại $(0;1)$.\\
				$x^3-2x^2+1=0\Leftrightarrow (x-1)(x^2-x-1)=0$\\
				$\Leftrightarrow \hoac{&x=1\\&x=\dfrac{1-\sqrt{5}}{2}\\&x=\dfrac{1+\sqrt{5}}{2}}$\\
				Suy ra đồ thị cắt trục $Ox$ tại $(1;0)$, $\left(\dfrac{1-\sqrt{5}}{2};0\right)$, $\left(\dfrac{1+\sqrt{5}}{2};0\right)$.\\
				Đồ thị qua $\left(\dfrac{4}{3};-\dfrac{5}{27}\right)$.
			}
			{
				\begin{tikzpicture}[line cap=round,line join=round,x=1cm,y=1cm]

		b) $x_1, x_2, x_3$ là nghiệm của phương trình $x^3-2x^2+(1-m)x+m=0\Leftrightarrow (x-1)(x^2-x-m)=0$\\
		$\Leftrightarrow \hoac{&x=1\\&x^2-x-m=0\quad (*)}$\\
		Coi $x_1=1$. Khi đó $(\mathscr{C}_m)$ cắt $Ox$ tại $3$ điểm phân biệt thỏa mãn $x_1^2+x_2^2+x_3^2<2018$ khi và chỉ khi $(*)$ có hai nghiệm $x_2, x_3$ khác $1$ và thỏa mãn $x_2^2+x_3^2<2017$.\\
		$(*)$ có $2$ nghiệm phân biệt khác $1\Leftrightarrow \heva{&\Delta =1+4m>0\\&-m\neq 0}\Leftrightarrow -\dfrac{1}{4}<m\neq 0\quad (**)$.\\
		Khi đó, theo Định lý Viet ta có:\\
		$x_2^2+x_3^2<2017\Leftrightarrow (x_2+x_3)^2-2x_2x_3<2017\Leftrightarrow 1+2m<2017\Leftrightarrow m<1008\quad (***)$.\\
		Kết hợp $(**)$ và $(***)$, ta được $\heva{&-\dfrac{1}{4}<m<1008\\&m\neq 0}$. 
		
	}

\begin{bt}%[2D1K5]
	Cho hàm số $y=x^4-2mx^2+m-1\hspace{5pt}(2)$, với $m$ là tham số thực. Tìm điều kiện của tham số $m$ để đồ thị hàm số có ba điểm cực trị tạo thành một tam giác có bán kính đường tròn ngoại tiếp bằng $1$.
	\loigiai{Hàm số có ba cực trị khi và chỉ khi $y'=4x^3-4mx$ đổi dấu ba lần $\Leftrightarrow -2m<0\Leftrightarrow m>0$. Khi đó, tọa độ ba điểm cực trị là: $$A\left(0;m-1\right),B\left(-\sqrt{m};-m^2+m-1\right),C\left(\sqrt{m};-m^2+m-1\right).$$
		Đến đây ta có thể giải quyết bài toán theo hai cách sau:
		
		\noindent \textbf{Cách 1.} $S_{ABC}=\dfrac{1}{2}\cdot \left|y_B-y_A\right|\cdot \left|x_B-x_C\right|=m^2\sqrt{m},AB=AC=\sqrt{m^4+m},BC=2\sqrt{m}$. Lúc đó:
		$$\begin{array}{cll}
		R=1\Leftrightarrow \dfrac{AB.AC.BC}{4S_{ABC}}=1\Leftrightarrow 2\sqrt{m}\left(m^4+m\right)=4m^2\sqrt{m}&\Leftrightarrow& m^3-2m+1=0\\&\Leftrightarrow& m=1\vee m=\dfrac{\sqrt{5}-1}{2},\text{ (do $m<0$)}.
		\end{array}.$$
		\textbf{Cách 2.} Do đồ thị hàm số $(2)$ nhận $Oy$ làm trục đối xứng và $A\in Oy$ nên tâm $I$ của đường tròn ngoại tiếp $\Delta ABC$ nằm trên $Oy$. Suy ra $I(0;a),a\in \mathbb{R}$. Khi đó
		$$IA=IB=1\Leftrightarrow \left\{\begin{array}{cllr}
		\left|a-m+1\right|&=&1&(*)\\
		m+\left(a+m^2-m+1\right)^2&=&1&(**)
		\end{array}\right..$$
		Giải $(*)$ ta được $a=m\vee a=m-2$.
		
		Với $a=m$, $(**)\Leftrightarrow m+\left(m^2+1\right)^2\Leftrightarrow m^3+2m+1=0$, (vô lý do $m>0$).
		
		Với $a=m-2$, $(**)\Leftrightarrow m+\left(m^2-1\right)^2=0\Leftrightarrow m^3-2m+1=0\Leftrightarrow m=1\vee m=\dfrac{\sqrt{5}-1}{2}$, (do $m>0$).}

\begin{bt}%[Đỗ Đường Hiếu-1]%%[2D1K5]
	Cho hàm số $y=\dfrac{x-1}{x+1}$ có đồ thị $(C)$.
	Tìm $a$ và $b$ để đường thẳng $d:y=ax+b$ cắt $(C)$ tại hai điểm phân biệt đối xứng nhau qua đường thẳng $\Delta: x-2y+3=0$.
	\loigiai{
		Đường thẳng $\Delta$ viết lại là $y=\dfrac{1}{2}x+\dfrac{3}{2}$.\\
		Từ yêu cầu bài toán suy ra $d\perp \Delta$, do đó $a\cdot \dfrac{1}{2}=-1\Rightarrow a=-2$, như vậy $d:y=-2x+b$.\\
		Phương trình hoành độ giao điểm của đồ thị $(C)$ và đường thẳng $d$ là
		$$ \dfrac{x-1}{x+1}=-2x+b \Leftrightarrow \left\{\begin{aligned}
		&2x^2-(b-3)x-1-b=0 \;\; (1)\\ 
		&x\ne -1
		\end{aligned}\right.$$
		Để $d$ cắt $(C)$ tại hai điểm phân biệt điều kiện là phương trình $(1)$ có hai nghiệm phân biệt khác $-1$\\
		$\Leftrightarrow \left\{\begin{aligned}
		&\Delta =(b-3)^2+8(1+b) >0\\ 
		&2(-1)^2-(b-3)(-1)-1-b \ne 0
		\end{aligned}\right.
		\Leftrightarrow b^2+2b+17>0 \Leftrightarrow b\in \mathbb{R}$.\\
		Hoành độ giao điểm $A, B$ của đồ thị $(C)$ và đường thẳng $d$ là các nghiệm $x_1$, $x_2$ của phương trình $(1)$, nên hoành độ trung điểm $I$ của $AB$ là $x_I=\dfrac{x_1+x_2}{2}=\dfrac{b-3}{4}$.\\
		Từ đó ta có $I=\left(\dfrac{b-3}{4};\dfrac{b+3}{2}\right)$.\\
		Để $A, B$ đối xứng nhau qua $\Delta$, điều kiện là $I\Delta$, do đó:\\
		$$\left(\dfrac{b-3}{4}\right)-2\left(\dfrac{b+3}{2}\right)+3=0\Leftrightarrow b=-1.$$
		Vậy $a=-2$, $b=-1$.
	}
\end{bt}
