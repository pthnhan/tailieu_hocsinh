\begin{ex}%[Đề thi vào 10, Sở giáo dục Khánh Hòa, 2017]%[9D1B6]%[9D1B3] 
	\textit{(Không sử dụng máy tính cầm tay)}
	\begin{enumerate}[$a)$]
		\item Tính giá trị biểu thức $T = \sqrt{\dfrac{1}{2}} + \dfrac{\sqrt{5} -1}{\sqrt{10}-\sqrt{2}} - \sqrt{3 - 2\sqrt{2}}$.
		\item Giải phương trình $x - 3 \sqrt{x} - 10 = 0$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[$a)$]
			\item Ta có: 
			\begin{align*} 
			T &= \sqrt{\dfrac{1}{2}} + \dfrac{\sqrt{5} -1}{\sqrt{10}-\sqrt{2}} - \sqrt{3 - 2\sqrt{2}} \\
			&= \sqrt{\dfrac{1}{2}} + \dfrac{\sqrt{5} -1}{\sqrt{2}\left(\sqrt{5}-1 \right)} - \sqrt{2 - 2.1.\sqrt{2} + 1} \\
			&= \dfrac{1}{\sqrt{2}} + \dfrac{1}{\sqrt{2}} - \sqrt{\left( \sqrt{2} - 1 \right) ^2 } \\
			&= \dfrac{2}{\sqrt{2}} - \left( \sqrt{2} - 1 \right) = 1.
			\end{align*}
			\item Đặt $t = \sqrt{x}$ $(t \ge 0)$. Khi đó, $x = t^2$ và phương trình trở thành:
			$$ t^2 - 3t - 10 = 0.$$
			Giải phương trình này ta thu được hai nghiệm $t_1 = 5$ (nhận) và $t_2 = -2$ (loại). \\
			Với $t = 5$, ta có $\sqrt{x}=5\Leftrightarrow x=25$.\\
			Vậy phương trình có nghiệm $x = 25$.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Khánh Hòa, 2017]%[9D2K3]
	Trên mặt phẳng tọa độ $Oxy$, cho parabol $(P): y = -3x^2$ và hai điểm $A(1;-3)$ và $B(2;3)$.
	\begin{enumerate}[a)]
		\item Chứng tỏ rằng điểm $A$ thuộc parabol $(P)$.
		\item Tìm tọa độ điểm $C$ ($C$ khác $A$) thuộc parabol $(P)$ sao cho ba điểm $A, B, C$ thẳng hàng.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item Thay $x = 1$, $y = -3$ vào $(P): y = -3x^2$ ta có $-3 = -3.1^2$ (thỏa mãn). \\
			Vậy $A \in (P)$.
			\item $A, B, C$ thẳng hàng và $C \in (P)$, vậy $C$ là giao điểm của đường thẳng $AB$ và parabol $(P)$. Phương trình đường thẳng $AB$ có dạng $y = ax + b$, trong đó $a, b$ là nghiệm của hệ phương trình:
			$\begin{cases}
			-3 = a + b \\
			3 = 2a + b.
			\end{cases}$ \\
			Giải hệ phương trình ta có $a = 6; b = -9$. Vậy phương trình $AB$ là $y = 6x - 9$. \\
			Phương trình hoành độ giao điểm của đường thẳng $AB$ và parabol $(P)$ là: $$ 6x - 9 = -3x^2 \Leftrightarrow 3x^2 + 6x - 9 = 0.$$
			Giải phương trình này, ta có hai nghiệm $x_1 = 1$ và $x_2 = -3$. Vì $C$ không trùng với $A$ và $A$ có hoành độ bằng $1$, nên $C$ có hoành độ $-3.$ Từ đó suy ra tọa độ điểm $C$ là $C \left(-3; -27 \right)$.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Khánh Hòa, 2017]%[9D4Y6]%[9D4G8] 
	\begin{enumerate}[a)]
		\item Tìm hai số, biết tổng của chúng bằng $7$ và tích của chúng bằng $12$.
		\item Một hội trường có $300$ ghế ngồi (loại ghế một người ngồi) được xếp thành nhiều dãy với số lượng ghế mỗi dãy như nhau để tổ chức một sự kiện. Vì số người dự lên đến $351$ người nên người ta phải xếp thêm $1$ dãy ghế có số lượng ghế như dãy ghế ban đầu và sau đó xếp mỗi dãy $2$ ghế (kể cả dãy ghế xếp thêm) để vừa đủ mỗi người ngồi một ghế. \\
		Hỏi ban đầu hội trường đó có bao nhiêu dãy ghế?
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item Hai số cần tìm có tổng bằng $7$ và tích bằng $12$. Do đó hai số này là nghiệm (nếu có) của phương trình $$ X^2 - 7X + 12 = 0.$$
			Giải phương trình này, ta thu được hai nghiệm $X_1 = 3$ và $X_2 = 4$.\\
			Vậy hai số cần tìm là $3$ và $4$.
			\item Gọi $x$ là số dãy ghế ban đầu của hội trường $(x \in \mathbb{N}^*)$. \\ 
			Từ đó suy ra số ghế của mỗi dãy ban đầu là $\dfrac{300}{x}$. \\ 
			Thực tế, số dãy ghế là $x + 1$ và số ghế của một dãy là $\dfrac{351}{x + 1}$.\\ 
			Vì phải xếp thêm $2$ ghế mỗi dãy nên ta có phương trình:
			$$\frac{351}{x + 1} - 2 = \frac{300}{x}. $$
			Giải phương trình này, ta thu được hai nghiệm $x_1 = 12$ (nhận) và $x_2 = \dfrac{25}{2}$ (loại). \\
			Vậy số dãy ghế ban đầu là $12$ dãy.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Khánh Hòa, 2017]%[9H3G3] %[9H3G7]
	Cho đường tròn $(O; OA)$. Trên bán kính $OA$ lấy điểm $I$ sao cho $OI = \dfrac{1}{3} OA$. Vẽ dây $BC$ vuông góc với $OA$ tại điểm $I$ và vẽ đường kính $BD$. Gọi $E$ là giao điểm của $AD$ và $BC$. 
	\begin{enumerate}[a)]
		\item Chứng minh $DA$ là tia phân giác của 
		$\widehat {BDC}$.
		\item Chứng minh $OE$ vuông góc với $AD$.
		\item Lấy điểm $M$ trên đoạn $IB$ ($M$ khác $I$ và $B$). Tia $AM$ cắt đường tròn $(O)$ tại điểm $N$. Tứ giác $MNDE$ có phải là một tứ giác nội tiếp hay không? Vì sao?
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item Ta có:\\
			$\left. \begin{array}{rr}
			\widehat{BDA}=\widehat{ACB} & (\text{cùng chắn }\overarc{\em AB})\\
			\widehat{ADC}=\widehat{CBA}& (\text{cùng chắn }\overarc{\em AC})\\
			\widehat{ACB}=\widehat{CBA}& (\text{vì }\Delta ABC \text{ cân tại }A)\\
			\end{array} \right\rbrace\Rightarrow \widehat{BDA}=\widehat{ADC}$.\\
			\immini{ Suy ra $DA$ là tia phân giác của $\widehat{BDC}$.
				\item Ta có $I$ là trung điểm $BC$. Vậy $OI$ là đường trung bình của $\Delta BDC$. Từ đây ta có $DC = 2OI = \dfrac{2}{3} OA.$ \\
				Mặt khác $IA = OA - OI = \dfrac{2}{3} OA$. Vậy $IA = DC$.\\
				Xét $\Delta IAE$ và $\Delta CDE$ có: \\
				$\widehat{IAE} = \widehat{CDE}$ (hai góc ở vị trí so le trong) \\
				$\widehat{AIE} = \widehat{DCE} (= 90^\circ)$ \\
				$IA = CD$ (chứng minh trên).\\
				Suy ra $\Delta IAE = \Delta CDE$ (g.c.g). Từ đó ta có $AE = DE$ (cạnh tương ứng), hay $E$ là trung điểm của $AD$. Suy ra $OE \perp DA$.}
			{\begin{tikzpicture}
				\tkzDefPoints{0/0/O, 1/2.83/B, 1/0/I, 1/1.7/M}
				\tkzDrawCircle(O,B)
				\tkzInterLC(I,B)(O,B)\tkzGetPoints{B}{C}
				\tkzInterLC(I,O)(O,B)\tkzGetPoints{A}{A'}
				\tkzInterLC(B,O)(O,B)\tkzGetPoints{B}{D}
				\tkzInterLL(A,D)(C,B)\tkzGetPoint{E}
				\tkzInterLC(A,M)(O,B)\tkzGetPoints{N}{A}
				\tkzDrawSegments(B,D D,C C,A A,O A,D B,C O,E A,N N,D O,C A,B)
				\tkzMarkRightAngles(D,C,B O,I,B O,E,D D,A,B)
				\tkzMarkAngles[arc=l,size=0.7cm](A,D,B A,C,B C,B,A I,A,D)
				\tkzMarkAngles[arc=l,size=1cm](C,D,A)
				\tkzLabelPoints[left](O)
				\tkzLabelPoints[above right](B)
				\tkzLabelPoints[below right](I,C)
				\tkzLabelPoints[below left](D)
				\tkzLabelPoints[right](A,E,M)
				\tkzLabelPoints[above left](N)
				\tkzDrawPoints(O,B,I,C,A,D,E,M,N)
				\end{tikzpicture}}
			\item Ta có: \\
			$\widehat{MED} = \widehat{BED} = \dfrac{1}{2} \left( \text{sđ} \overarc{\em BND} + \text{sđ} \overarc{\em AC} \right)$ (góc có đỉnh nằm trong đường tròn)\\
			$\begin{array}{rl}
			\widehat{MND} =	& \widehat{AND} = \dfrac{1}{2} \text{sđ} \overarc{\em ACD} \text{ (góc nội tiếp)} \\ 
			=	&  \dfrac{1}{2} \left( \text{sđ} \overarc{\em AC} + \text{sđ} \overarc{\em CD} \right) = \dfrac{1}{2} \left( \text{sđ} \overarc{\em AB} + \text{sđ} \overarc{\em CD} \right)
			\end{array} $\\
			Vậy $\widehat{MED} + \widehat{MND} = \dfrac{1}{2} \left( \text{sđ} \overarc{\em BND} + \text{sđ} \overarc{\em DC} +  \text{sđ} \overarc{\em CA} + \text{sđ} \overarc{\em AB} \right) = \dfrac{1}{2} \cdot 360^\circ = 180^\circ$\\
			Suy ra tứ giác $MNDC$ nội tiếp.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Khánh Hòa, 2017]%[9H4K1]
	Tính diện tích xung quanh, diện tích toàn phần và thể tích của một hình trụ có chu vi hình tròn đáy là 16cm và chiều cao là 5cm.
	\loigiai{
		\immini{
			Ta có bán kính đáy của hình trụ là: $r = \dfrac{16}{2\pi} = \dfrac{8}{\pi}$. \\
			Vậy diện tích xung quanh là $S_{xq} = 2 \pi r h = 2 \pi \cdot \dfrac{8}{\pi} \cdot 5 = 80 $ (cm$^2$). \\
			Diện tích toàn phần là
			$$S_{tp} = S_{xq} + 2 S_{\text{đáy}} 
			= 80 + 2 \pi r^2 
			= 80 + 2 \pi . \left( \dfrac{8}{\pi} \right)^2 
			= 80 + \dfrac{128}{\pi}\; (\text{cm}^2).$$
			Thể tích là $V = \pi . r^2 . h =\pi .  \left( \dfrac{8}{\pi} \right)^2 . 5 = \dfrac{320}{\pi}$ (cm$^3$).}{\begin{tikzpicture}[scale=0.7]
			\coordinate[label={below left}:$O$] (o) at (0,-2);
			\coordinate[label={below left}:$O'$] (o') at (0,2);
			\coordinate (a) at (2,-2);
			\coordinate (a') at ($(a)+(-4,0)$);
			\coordinate (b) at (2,2);
			\coordinate (b') at ($(b)+(-4,0)$);
			\draw[dashed] (a) arc (0:180:2cm and 1cm);
			\draw (a) arc (0:-180:2cm and 1cm);
			\draw (b) arc (0:360:2cm and 1cm);
			\draw (a)--(b) (a')--(b');
			\draw[dashed]  (o)--(o');
			\tkzDrawPoints(o,o')
			\path (a)--(b) node[right,midway]{$5$\,cm};
			\end{tikzpicture}}
	}
\end{ex}
