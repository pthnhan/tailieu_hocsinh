\begin{ex}%[Đề thi HSG 9, tỉnh Bắc Giang, năm 2018]%[Nguyễn Thành Khang, Dự án EX-9G-18]%[9D1B6]
	Cho biểu thức: $$A=\left(\dfrac{\sqrt{x}+1}{\sqrt{x}-1}+\dfrac{\sqrt{x}-1}{\sqrt{x}+1}+\dfrac{3\sqrt{x}+1}{1-x}\right):\left(\dfrac{\sqrt{x}}{\sqrt{x}-1}-\dfrac{2}{x-1}\right)\text{ với } x\ge 0, x\ne 1.$$
	\begin{enumerate}
		\item Rút gọn biểu thức $A$.
		\item Tìm $x$ để biểu thức $A$ đạt giá trị nhỏ nhất.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			\vspace*{-1em}
			\begin{eqnarray*}
				A &=& \left(\dfrac{\sqrt{x}+1}{\sqrt{x}-1}+\dfrac{\sqrt{x}-1}{\sqrt{x}+1}+\dfrac{3\sqrt{x}+1}{1-x}\right):\left(\dfrac{\sqrt{x}}{\sqrt{x}-1}-\dfrac{2}{x-1}\right) \\ 
				&=& \dfrac{\left(\sqrt{x}+1\right)\left(\sqrt{x}+1\right)+\left(\sqrt{x}-1\right)\left(\sqrt{x}-1\right)-\left(3\sqrt{x}+1\right)}{x-1}:\dfrac{\sqrt{x}\left(\sqrt{x}+1\right)-2}{x-1} \\
				&=& \dfrac{2x-3\sqrt{x}+1}{x-1}\cdot \dfrac{x-1}{x+\sqrt{x}-2} \\
				&=& \dfrac{2x-3\sqrt{x}+1}{x+\sqrt{x}-2}=\dfrac{\left(\sqrt{x}-1\right)\left(2\sqrt{x}-1\right)}{\left(\sqrt{x}-1\right)\left(\sqrt{x}+2\right)} \\
				&=& \dfrac{2\sqrt{x}-1}{\sqrt{x}+2}.
			\end{eqnarray*}
			\vspace*{-2em}
			\item Ta có $A=\dfrac{2\sqrt{x}-1}{\sqrt{x}+2}=2-\dfrac{5}{\sqrt{x}+2} \ge 2-\dfrac{5}{2}=-\dfrac{1}{2}$.\\
			Dấu bằng xảy ra khi $\sqrt{x}=0 \Leftrightarrow x=0$.\\
			Vậy giá trị nhỏ nhất của $A$ bằng $-\dfrac{1}{2}$ khi $x=0$.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Bắc Giang, năm 2018]%[Nguyễn Thành Khang, Dự án EX-9G-18]%[9D4B6]
	Tìm giá trị của tham số $m$ để phương trình $x^2-2(m+1)x+m=0$ có hai nghiệm phân biệt $x_1,x_2$ thoả mãn $\dfrac{2x_1-1}{x_2}+\dfrac{2x_2-1}{x_1}=x_1x_2+\dfrac{3}{x_1x_2}$.
	\loigiai{
		Ta có $\Delta'=(m+1)^2-m=m^2+m+1=\left(m+\dfrac{1}{2}\right)^2+\dfrac{3}{4}>0$ với mọi $m$ nên phương trình đã cho luôn có $2$ nghiệm phân biệt $x_1,x_2$.\\
		Áp dụng định lý Vi-ét, ta có $\heva{&x_1+x_2=2(m+1), \\ &x_1x_2=m.}$\\
		Khi đó
		\allowdisplaybreaks{
		\begin{eqnarray*}
			\dfrac{2x_1-1}{x_2}+\dfrac{2x_2-1}{x_1}=x_1x_2+\dfrac{3}{x_1x_2} &\Leftrightarrow& \dfrac{2\left(x_1^2+x_2^2\right)-\left(x_1+x_2\right)}{x_1x_2}=x_1x_2+\dfrac{3}{x_1x_2} \\ 
			&\Leftrightarrow& 2\left(x_1+x_2\right)^2-4x_1x_2-\left(x_1+x_2\right)=x_1^2x_2^2+3 \\
			&\Leftrightarrow& 8(m+1)^2-4m-2(m+1)=m^2+3 \\
			&\Leftrightarrow& 7m^2+10m+3=0 \Leftrightarrow \hoac{&m=-1 \\ &m=-\dfrac{3}{7}.}
		\end{eqnarray*}
		}
		Vậy $m\in\left\{-1;-\dfrac{7}{3}\right\}$.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Bắc Giang, năm 2018]%[Nguyễn Thành Khang, Dự án EX-9G-18]%[9D0K2]
	Giải phương trình $\sqrt{2x^2+x+6}+\sqrt{x^2+x+2}=x+\dfrac{4}{x}$.
	\loigiai{
		Điều kiện: $x\ne 0$.\\
		Nhận thấy $\sqrt{2x^2+x+6}>\sqrt{x^2+x+2},\forall x\ne0$ nên phương trình đã cho tương đương với:
		\begin{eqnarray*}
		& &\dfrac{x^2+4}{\sqrt{2x^2+x+6}-\sqrt{x^2+x+2}}=\dfrac{x^2+4}{x} \\
		&\Leftrightarrow & \sqrt{2x^2+x+6}-\sqrt{x^2+x+2}=x \\
		&\Rightarrow & 2x^2+x+6=\left(x+\sqrt{x^2+x+2}\right)^2 \\
		&\Leftrightarrow & x\sqrt{x^2+x+2}=2 \\
		&\Leftrightarrow & \heva{&x\ge 0 \\ &x^2(x^2+x+2)=4} \\
		&\Leftrightarrow & \heva{&x\ge 0 \\ &(x-1)\left(x^3+2x^2+4x+4\right)=0} \Leftrightarrow x=1.
		\end{eqnarray*}
		Thử lại, $x=1$ thoả mãn phương trình đã cho.\\
		Vậy phương trình đã cho có nghiệm duy nhất $x=1$.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Bắc Giang, năm 2018]%[Nguyễn Thành Khang, Dự án EX-9G-18]%[9D0G2]
	Giải hệ phương trình $\heva{&x^3+x+2=2y \\ &3\left(x^2+x\right)=y^3-y}\quad \left(x,y\in\mathbb{R}\right)$.
	\loigiai{
		Cộng vế với vế của hai phương trình ta được
		\begin{eqnarray*}
		& & x^3+3x^2+4x+2=y^3+y \\ 
		&\Leftrightarrow & (x+1)^3+x+1-y^3-y=0 \\
		&\Leftrightarrow & (x-y+1)\left[(x+1)^2+y(x+1)+y^2\right]+(x-y+1)=0 \\
		&\Leftrightarrow & (x-y+1)\underbrace{\left[\left(x+1+\dfrac{y}{2}\right)^2+\dfrac{3}{4}y^2+1\right]}_{>0}=0 \\
		&\Leftrightarrow & x-y+1=0.
		\end{eqnarray*}
		Thay $y=x+1$ vào phương trình thứ nhất ta được
		$$ x^3+x+2=2(x+1) \Leftrightarrow x^3-x=0 \Leftrightarrow \hoac{&x=0 \\ &x=1 \\ &x=-1.} $$
		Từ đó suy ra hệ có các nghiệm $(x;y)\in \left\{(0;1),(1;2),(-1;0)\right\}$.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Bắc Giang, năm 2018]%[Nguyễn Thành Khang, Dự án EX-9G-18]%[9D0G5]
	Tìm tất cả các cặp số nguyên $(x;y)$ sao cho $3\left(x^4-y^2\right)=2\left(x^2-y\right)+7$.
	\loigiai{
		Phương trình đã cho tương đương với
		$$ 3\left(x^4-y^2\right)=2\left(x^2-y\right)+7 \Leftrightarrow \left(3x^2-1\right)^2-\left(3y-1\right)^2=21 \Leftrightarrow \left(x^2-y\right)\left(3x^2+3y-2\right)=7. $$
		Từ đó suy ra $x^2-y$ và $3x^2+3y-2$ đều là ước của $7$ và $3x^2+3y-2\equiv 1\pmod{3}$ nên
		$$ \hoac{&\heva{&x^2-y=1 \\ &3x^2+3y-2=7} \\ &\heva{&x^2-y=7 \\ &3x^2+3y-2=1}} \Leftrightarrow \hoac{&\heva{&x^2-y=1 \\ &x^2+y=3} \\ &\heva{&x^2-y=7 \\ &x^2+y=1}} \Leftrightarrow \hoac{&\heva{&x^2=2 \\ &y=1}\quad\text{(không có nghiệm nguyên)} \\ &\heva{&x^2=4 \\ &y=3}\Leftrightarrow \heva{&x=\pm 2 \\ &y=3.}}$$
		Vậy phương trình có các nghiệm nguyên $(x;y)\in \left\{(2;3),(-2;3)\right\}$.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Bắc Giang, năm 2018]%[Nguyễn Thành Khang, Dự án EX-9G-18]%[9D0K7]
	Cho biểu thức $B=\dfrac{1}{16}+\dfrac{2}{16^2}+\dfrac{3}{16^3}+\cdots+\dfrac{2018}{16^{2018}}$. Hãy so sánh hai số $B^{2017}$ và $B^{2018}$.
	\loigiai{
		Ta có $16B=1+\dfrac{2}{16}+\dfrac{3}{16^2}+\cdots+\dfrac{2018}{16^{2017}}$.\\
		Suy ra $15B=16B-B=1+\dfrac{1}{16}+\dfrac{1}{16^2}+\cdots+\dfrac{1}{16^{2017}}-\dfrac{2018}{16^{2018}}<1+\dfrac{1}{16}+\dfrac{1}{16^2}+\cdots+\dfrac{1}{16^{2017}}=A$.\\
		Lại có $16A=16+1+\dfrac{1}{16}+\cdots+\dfrac{1}{16^{2016}}$ nên $15A=16-\dfrac{1}{16^{2017}}<16$.\\
		Từ đó suy ra $B<1$, dẫn tới $B^{2017}>B^{2018}$.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Bắc Giang, năm 2018]%[Nguyễn Thành Khang, Dự án EX-9G-18]%[9H0K1]
	Cho hai đường tròn $(O;4),(I;2)$ cắt nhau tại hai điểm phân biệt $A,B$ sao cho $\widehat{OAI}\ne 90^\circ$. Tiếp tuyến của đường tròn $(O)$ tại $A$ cắt đường tròn $(I)$ tại $C$ khác $A$. Tiếp tuyến của đường tròn $(I)$ tại $A$ cắt đường tròn $(O)$ tại $D$ khác $A$. Gọi $E$ là giao điểm của $AB$ và $CD$. Gọi $P,Q$ lần lượt là trung điểm $AD,CD$. Chứng minh:
	\begin{enumerate}
		\item Hai tam giác $APQ,ABC$ đồng dạng.
		\item $ED=4EC$.
	\end{enumerate}
	\loigiai{			
		\begin{center}
			\begin{tikzpicture}
			\tkzInit[xmin=-4.5, xmax=7.5, ymin=-4, ymax=4.1]
			\tkzClip
			\tkzDefPoints{0/0/O,4/0/X,3/0/Y,5/0/I}
			\tkzDrawCircle(O,X)
			\tkzDrawCircle(I,Y)
			\tkzInterCC(O,X)(I,Y)\tkzGetPoints{A}{B}
			\tkzDefLine[orthogonal=through A](O,A) \tkzGetPoint{c}
			\tkzInterLC(A,c)(I,Y)\tkzGetFirstPoint{C}
			\tkzDefLine[orthogonal=through A](I,A) \tkzGetPoint{d}
			\tkzInterLC(A,d)(O,X)\tkzGetSecondPoint{D}
			\tkzInterLL(A,B)(C,D)\tkzGetPoint{E}
			\tkzDefMidPoint(A,D)\tkzGetPoint{P}
			\tkzDefMidPoint(C,D)\tkzGetPoint{Q}
			\tkzDrawSegments(A,C A,D C,D A,E A,Q B,C P,Q B,D O,A O,I A,I)
			\tkzDrawPoints[fill=blue](O,I,A,C,D,B,E,P,Q)
			\tkzLabelPoints[below](O,I,C,D,E,Q)
			\tkzLabelPoints[above right](B)
			\tkzLabelPoints[above left](P)
			\tkzLabelPoints[above](A)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item $PQ$ là đường trung bình của $\triangle ACD$ nên $PQ\parallel AC$ và $PQ=\dfrac{1}{2}AC$.\\
			Ta có $\widehat{DAB}=\widehat{ACB}, \widehat{BDA}=\widehat{BAC}\Rightarrow \triangle ABD\backsim \triangle CBA$ (g.g)\\
			$\Rightarrow \dfrac{AB}{AD}=\dfrac{CB}{AC}\Rightarrow \dfrac{AB}{2AP}=\dfrac{CB}{2PQ} \Rightarrow \dfrac{AB}{BC}=\dfrac{AP}{PQ}$.\\
			Lại có $\widehat{DPQ}=\widehat{DAC}=\dfrac{1}{2}\widehat{AIC}=180^\circ-\widehat{ABC}\Rightarrow \widehat{APQ}=\widehat{ABC}$.\\
			Suy ra $\triangle APQ\backsim \triangle ABC$.
			\item Ta có $\widehat{AIO}=\dfrac{1}{2}\widehat{AIB}=\widehat{ACB}$ và $\widehat{AOI}=\dfrac{1}{2}\widehat{AOB}=\widehat{ADB}=\widehat{BAC}$.\\
			Suy ra $\triangle OAI\backsim \triangle ABC$ (g-g) $\Rightarrow \dfrac{AB}{BC}=\dfrac{OA}{AI}=2$.\\
			Tương tự ta cũng có $\triangle ABD\backsim \triangle IAO\Rightarrow \dfrac{BD}{AB}=\dfrac{OA}{AI}=2$.\\
			Lại có $\triangle ABD\backsim \triangle CBA \Rightarrow \widehat{ABD}=\widehat{ABC}$, suy ra $BE$ là phân giác góc $\widehat{CBD}$, nên dẫn tới $\dfrac{DE}{EC}=\dfrac{BD}{BC}=\dfrac{AB}{BC}\cdot \dfrac{BD}{AB}=4$ hay $DE=4EC$.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Bắc Giang, năm 2018]%[Nguyễn Thành Khang, Dự án EX-9G-18]%[9H0G1]

Điểm $E$ thuộc cung nhỏ $CD$ của đường tròn $(O)$, $E$ khác $C$ và $D$. $EA$ cắt $DB,DC$ lần lượt tại $M$ và $N$; $EB$ cắt $CA,CD$ lần lượt tại $P$ và $Q$. Gọi $G$ là giao điểm của $CM$ và $DP$. \begin{ex}%[Đề thi HSG 9, tỉnh Bắc Giang, năm 2018]%[Nguyễn Thành Khang, Dự án EX-9G-18]%[9D0G7]
	Tìm giá trị nhỏ nhất của biểu thức $P=\dfrac{\left(x^3+y^3\right)-\left(x^2+y^2\right)}{(x-1)(y-1)}$, trong đó $x,y$ là các số thực lớn hơn $1$.
	\loigiai{
		Ta có 
		$$ \begin{array}{rl}
		P &= \dfrac{x^2}{y-1}+\dfrac{y^2}{x-1} \\ 
		&= \dfrac{x^2}{y-1}+4(y-1)+\dfrac{y^2}{x-1}+4(x-1)-4x-4y+8 \\
		&\ge 2\sqrt{\dfrac{x^2}{y-1}\cdot 4(y-1)}+ 2\sqrt{\dfrac{y^2}{x-1}\cdot 4(x-1)}-4x-4y+8 = 8. \\
		\end{array} $$
		Dấu bằng xảy ra khi và chỉ khi $x=y=2$.\\
		Vậy giá trị nhỏ nhất của $P$ bằng $8$.
	}
\end{ex}
