	\begin{bt}
		Giải phương trình $\sqrt{2x^2+x-2}=-x$. 
	\loigiai{
		Phương trình đã cho tương đương  $\heva{& -x\ge 0 \\ & 2x^2+x-2=x^2} \Leftrightarrow \heva{&x\le 0\\& \hoac{&x=1\text{ (loại)}\\&x=-2\text{ (nhận)}}}\Leftrightarrow x=-2$.\\
		Vậy tập nghiệm của phương trình là $S=\{-2\}$.
	}
	\end{bt}

	\begin{bt}
		Giải phương trình  $(x+1)^4-3(x^2+2x)-3=0$.
		\loigiai{
		\begin{eqnarray*}
			&& (x+1)^4-3(x^2+2x)-3=0\\
			&\Leftrightarrow& (x^2+2x+1)^2-3(x^2+2x)-3=0\\
			&\Leftrightarrow& (x^2+2x+1)^2-3(x^2+2x+1)=0\\
			&\Leftrightarrow& \hoac{&x^2+2x+1=0\\&x^2+2x+1=3}\\
			&\Leftrightarrow& \hoac{&x=-1\\&x=-1-\sqrt{3}\\&x=-1+\sqrt{3}}.
		\end{eqnarray*}
	Vậy tập nghiệm của phương trình là $S=\{-1,-1-\sqrt{3},-1+\sqrt{3}\}$.
		}
	\end{bt}

	\begin{bt}
		Tìm tất cả các giá trị của tham số $m$ để phương trình $x^2+(m^2-1)x-1=0$ có hai nghiệm phân biệt sao cho $x_1^2-x_2^2=0$.
		\loigiai{
		Phương trình có hai nghiệm phân biệt khi và chỉ khi
		$$\Delta >0\Leftrightarrow ( m^2-1)^2+4>0,\forall m\in \mathbb{R}.$$
		Với $x_1\ne x_2$, ta có  $ x_1^2-x_2^2=0  \Leftrightarrow (x_1-x_2)(x_1+x_2)=0 \Leftrightarrow x_1+x_2=0$.\\
		Theo định lý Viete, ta có $x_1+x_2=-\dfrac{m^2-1}{1}\Leftrightarrow m^2-1 =0\Leftrightarrow m=\pm 1$.\\
		Vậy $m=1$ hoặc $m=-1$ thì $x_1^2-x_2^2=0$.
		}
	\end{bt}

	\begin{bt}
		Tìm tất cả các giá trị của tham số $m$ để phương trình $(m-1)x^2+(2m-1)x+m=0$ có nghiệm.
	\loigiai{
		\begin{itemize}
		\item Trường hợp 1, $m-1=0\Leftrightarrow m=1$.\\
		Khi đó phương trình đã cho tương đương $x+1=0\Leftrightarrow x=-1$ (nhận).
		\item Trường hợp 2, $m-1\ne 0\Leftrightarrow m\ne 1$.\\
		Khi đó hương trình đã cho có nghiệm khi và chỉ khi
		$$\Delta \ge 0\Leftrightarrow (2m-1)^2-4m(m-1)\ge 0 \Leftrightarrow 1\ge 0, \forall m\ne 1.$$
		Vậy với mọi $m$ thì phương trình có nghiệm.
		
		\end{itemize}
		}	
\end{bt}

			\item[b)] Tìm tọa độ điểm $D$ trên trục $Ox$ sao cho tứ giác $ABCD$ là hình thang có hai đáy $AB,CD$. 
		\end{itemize}
	\loigiai{
		\begin{itemize}
			\item[a)] Điểm  $G$ thỏa $\overrightarrow{GA}+\overrightarrow{GB}+\overrightarrow{GC}=\overrightarrow{0}$ khi và chỉ khi $G$ là trọng tâm tam giác $ABC$.\\
			Vậy tọa độ điểm  $G\left( \dfrac{2}{3};-\dfrac{2}{3} \right)$.
			\item[b)] Vì $D\in Ox$ nên gọi tọa độ điểm $ D=(x;0)$, suy ra $\overrightarrow{AB}=( 0;-5)$ và $\overrightarrow{CD}=(x-2;1)$.\\
			Mà $ABCD$ là hình thang đáy $AB,CD$ nên 
			$\overrightarrow{AB},\overrightarrow{CD}$ cùng phương,\\
			do đó $x-2=0\Leftrightarrow x=2$.\\
			Vậy tọa độ điểm $D(2;0)$.
			
		\end{itemize}
		
		}
