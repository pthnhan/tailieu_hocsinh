\begin{ex}%[Đề thi vào 10, Chuyên Hà Nội, 2016]%[9D5G5]%[9D5G5]
\hfill
\begin{enumerate}
\item Giải phương trình $x^{4}-2x^{3}+x-\sqrt{2(x^{2}-x)}=0$.
\item Giải hệ phương trình $\heva{&x^{2}+2y-4x=0\\ &4x^{2}-4xy^{2}+y^{4}-2y+4=0}$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Điều kiện $x\leq 0$; $x\geq 1$.\\
Phương trình tương đương $(x^{2}-x)^{2}-(x^{2}-x)-\sqrt{2(x^{2}-x)}=0$. Đặt $t=\sqrt{x^{2}-x}$ ($t\geq 0$). Ta có
\[
t^{4}-t^{2}-\sqrt{2}t=0\Leftrightarrow t(t^{3}-t-\sqrt{2})=0\Leftrightarrow \hoac{&t=0\\ &t=\sqrt{2}}
\]
\begin{itemize}
\item $t=0\Rightarrow x^{2}-x=0\Rightarrow \hoac{&x=0\\ &x=1}$ (TM).
\item $t=\sqrt{2}\Rightarrow x^{2}-x=2\Rightarrow \hoac{&x=2\\ &x=-1}$ (TM).
\end{itemize}
Vậy tập nghiệm của phương trình là $\{-1;0;1;2\}$.
\item Xét $4x^{2}-4xy^{2}+y^{4}-2y+4=0\Leftrightarrow (2x-y^{2})^{2}+(x-2)^{2}=0\Rightarrow \heva{&2x-y=0\\ &x-2=0}\Rightarrow \heva{&x=2\\ &y=\pm 2}$.
\begin{itemize}
\item Thay $x=2$, $y=2$ vào phương trình còn lại ta thấy thỏa mãn.
\item Thay $x=2$, $y=-2$ vào phương trình còn lại ta thấy không thỏa mãn.
\end{itemize}
Vậy nghiệm của hệ là $(x;y)=(2;2)$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên Hà Nội, 2016]%[9D5G5]%[9D5G3]
\hfill
\begin{enumerate}
\item Cho các số thực $a$, $b$, $c$ đôi một khác nhau thỏa mãn $a^{3}+b^{3}+c^{3}=3abc$ và $abc\neq 0$. Tính giá trị của biểu thức $P=\dfrac{ab^{2}}{a^{2}+b^{2}-c^{2}}+\dfrac{bc^{2}}{b^{2}+c^{2}-a^{2}}+\dfrac{ca^{2}}{c^{2}+a^{2}-b^{2}}$.
\item Tìm tất cả các cặp số tự nhiên $(x;y)$ thỏa mãn $2^{x}.x^{2}=9y^{2}+6y+16$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $a^{3}+b^{3}+c^{3}=3abc\Rightarrow (a+b+c)(a^{2}+b^{2}+c^{2}-ab-bc-ca)=0\Rightarrow a+b+c=0$ (do $a$, $b$, $c$ khác nhau).\\
$\Rightarrow \dfrac{ab^{2}}{a^{2}+b^{2}-c^{2}}=\dfrac{ab^{2}}{b^{2}+(a-c)(a+c)}=\dfrac{ab^{2}}{b^{2}-b(a-c)}=\dfrac{ab}{b-a+c}=-\dfrac{b}{2}$.\\
Tương tự $\dfrac{bc^{2}}{b^{2}+c^{2}-a^{2}}=-\dfrac{c}{2}$ và $\dfrac{ca^{2}}{c^{2}+a^{2}-b^{2}}=-\dfrac{a}{2}\Rightarrow P=-\dfrac{a+b+c}{2}=0$.
\item Ta có $2^{x}.x^{2}=(3y+1)^{2}+15$.\\
Do $x$, $y\in\mathbb N$ và $\heva{&3y+1\equiv 1(\mbox{mod } 3)\\ &15\equiv 0(\mbox{mod } 3)}\Rightarrow (3y+1)^{2}+15\equiv 1(\mbox{mod }3)$.\\
Vì $x\in\mathbb N$ nên $x^{2}$ là số chính phương $\Rightarrow x^{2}\equiv 1(\mbox{mod 3})$ hoặc $x^{2}\equiv 0(\mbox{mod 3})$.\\
Do $2^{x}.x^{2}\equiv 1(\mbox{mod }3)$ nên $x^{2}\equiv 1(\mbox{mod 3})\Rightarrow 2^{x}\equiv 1(\mbox{mod }3)\Rightarrow x=2k$ ($k\in \mathbb N$).\\
Vậy ta có $2^{2k}.(2k)^{2}-(3y+1)^{2}=15\Leftrightarrow (2^{k}.2k-3y-1)(2^{k}.2k+3y+1)=15$. Vì $k$, $y\in\mathbb N$ nên $2^{k}.2k+3y+1>0\Rightarrow 2^{k}.2k-3y-1>0$ nên ta có các trường hợp sau:
\begin{itemize}
\item $\heva{&2^{k}.2k+3y+1=15\\ &2^{k}.2k-3y-1=1}\Leftrightarrow \heva{&3y+1=7\\ &2^{k}.k=4}$ (vô lí).
\item $\heva{&2^{k}.2k+3y+1=5\\ &2^{k}.2k-3y-1=3}\Leftrightarrow \heva{&3y+1=1\\ &2^{k}.k=2}\Leftrightarrow \heva{& y=0\\& k=1}$.
\end{itemize}
Vậy $(x;y)=(2;0)$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên Hà Nội, 2016]%[9D5G1]%[9D5G2]
\hfill
\begin{enumerate}
\item Cho các số thực dương $a$, $b$, $c$ thỏa mãn $a^{2}+b^{2}+c^{2}=3$. Chứng minh rằng
\[
\dfrac{2a^{2}}{a+b^{2}}+\dfrac{2b^{2}}{b+c^{2}}+\dfrac{2c^{2}}{c+a^{2}}\geq a+b+c.
\]
\item Cho số nguyên dương $n$ thỏa mãn $2+2\sqrt{12n^{2}+1}$ là số nguyên. Chứng minh $2+2\sqrt{12n^{2}+1}$ là số chính phương.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $3=a^{2}+b^{2}+c^{2}\geq \dfrac{(a+b+c)^{2}}{3}\Rightarrow a+b+c\leq 3$. Vì $a^{2}-2a+1\geq 0\Rightarrow a^{4}+a^{2}\geq 2a^{3}$. Do đó
\begin{align*}
&\dfrac{2a^{2}}{a+b^{2}}+\dfrac{2b^{2}}{b+c^{2}}+\dfrac{2c^{2}}{c+a^{2}}=\dfrac{4a^{4}}{2a^{3}+2a^{2}b^{2}}+\dfrac{4b^{4}}{2b^{3}+2b^{2}c^{2}}+\dfrac{4c^{4}}{2c^{3}+2c^{2}a^{2}}\\ 
&\geq \dfrac{(2a^{2}+2b^{2}+2c^{2})^{2}}{2a^{3}+2a^{2}b^{2}+2b^{3}+2b^{2}c^{2}+2c^{3}+2c^{2}a^{2}}\\ 
&\geq \dfrac{36}{a^{4}+a^{2}+b^{4}+b^{2}+c^{4}+c^{2}++2a^{2}b^{2}+2b^{2}c^{2}+2c^{2}a^{2}}\\ 
&=\dfrac{36}{(a^{2}+b^{2}+c^{2})^{2}+(a^{2}+b^{2}+c^{2})}=3\geq a+b+c.
\end{align*}
\item Vì $2+2\sqrt{12n^{2}+1}\in \mathbb N^*\Rightarrow 2\sqrt{12n^{2}+1}\in\mathbb N^{*}\Rightarrow \sqrt{12n^{2}+1}\in\mathbb Q\Rightarrow \sqrt{12n^{2}+1}=m\in\mathbb Q$\\
$\Rightarrow 12n^{2}=m^{2}-1\ \vdots\ 4\Rightarrow m=2k+1$ ($k\in \mathbb N$).\\
Ta có $12n^{2}=(2k+1)^{2}-1=4k(k+1)\Rightarrow k(k+1)=3n^{2}\ \vdots\ 3\Rightarrow k\ \vdots\ 3$ hoặc $k+1\ \vdots\ 3$.
\begin{itemize}
\item Xét $k=3p$ ($p\in\mathbb N^{*}$) $\Rightarrow 3n^{2}=3p(3p+1)\Rightarrow n^{2}=p(3p+1)$.\\
Vì $\left(p,3p+1\right)=1\Rightarrow \heva{&p=a^{2}\\&3p+1=b^{2}}$ (với $a$, $b\in\mathbb N^{*}$).\\
Mà $2+2\sqrt{12n^{2}+1}=2+2m=2+2(2k+1)=4+4k=4(k+1)=4(3p+1)=4b^{2}$ (đpcm).
\item Xét $k=3q-1$ ($q\in\mathbb N^{*}$) $\Rightarrow 3n^{2}=3q(3q-1)\Rightarrow n^{2}=q(3q-1)$.\\
Vì $\left(q,3q-1\right)=1\Rightarrow \heva{&q=a^{2}\\&3q-1=b^{2}}$ (với $a$, $b\in\mathbb N^{*}$) $\Rightarrow b^{2}=3a^{2}-1$ (vô lí vì số chính phương chia cho $3$ có dư là $0$ hoặc $1$).
\end{itemize}
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên Hà Nội, 2016]%[9H3G7]%[9H3G7]%[9H3G7]
Cho tam giác nhọn $ABC$ có $AB<AC$ và nội tiếp đường tròn $(O)$. Các đường cao $BB'$, $CC'$ cắt nhau tại $H$. Gọi $M$ là trung điểm của $BC$. Tia $MH$ cắt đường tròn $(O)$ tại $P$.
\begin{enumerate}
\item Chứng minh hai tam giác $BPC'$ và $CPB'$ đồng dạng.
\item Các đường phân giác của các góc $\widehat{BPC'}$, $\widehat{CPB'}$ lần lượt cắt $AB$, $AC$ tại các điểm $E$, $F$. Gọi $O'$ là tâm đường tròn ngoại tiếp tam giác $AEF$ và $K$ là giao điểm của $HM$ với $AO'$.
\begin{enumerate}
\item Chứng minh tứ giác $PEKF$ là tứ giác nội tiếp.
\item Chứng minh các tiếp tuyến tại $E$ và $F$ của đường tròn $(O')$ cắt nhau tại một điểm nằm trên đường tròn $(O)$.
\end{enumerate}
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Giả sử tia $AO$ cắt $(O)$ tại $D\Rightarrow AD$ là đường kính của đường tròn $(O)$\\
$\Rightarrow \widehat{ABD}=\widehat{ACD}=90^{\circ}\Rightarrow BD\parallel CH$ và $CD\parallel BH\Rightarrow$ tứ giác $HBDC$ là hình bình hành\\
$\Rightarrow DH$ cắt $BC$ tại trung điểm của mỗi đoạn $\Rightarrow D$, $M$, $H$, $P$ thẳng hàng\\
$\Rightarrow \widehat{APD}=90^{\circ}$ hay $\widehat{APH}=90^{\circ}\Rightarrow P$ nằm trên đường tròn đường kính $AH$.\\
Lại có $\widehat{AB'H}=\widehat{AC'H}=90^{\circ}\Rightarrow B'$, $C'$ nằm trên đường tròn đường kính $AH$\\
$\Rightarrow A$, $P$, $B'$, $C'$, $H$ nằm trên đường tròn đường kính $AH\Rightarrow \widehat{AB'P}=\widehat{AC'P}$.\\
Mà $\widehat{PC'B}+\widehat{PC'A}=90^{\circ}$ và $\widehat{PB'C}+\widehat{PB'A}=180^{\circ}$ nên $\widehat{PC'B}=\widehat{PB'C}$.\\
Xét đường tròn $(O)$ có $\widehat{PCB'}=\widehat{PBC'}$ (cùng chắn cung $\wideparen{AP}$) và $\widehat{PC'B}=\widehat{PB'C}$\\
$\Rightarrow \Delta PB'C\backsim\Delta PC'B$.

\item Vì $PE$, $PF$ lần lượt là phân giác $\widehat{BPC'}$, $\widehat{CPB'}$ nên $\heva{&\widehat{EPC'}=\dfrac{1}{2}\widehat{BPC'}\\ & \widehat{FPB'}=\dfrac{1}{2}\widehat{CPB'}}$.

\begin{ex}%[Đề thi vào 10, Chuyên Hà Nội, 2016]%[9D5G2]
Cho $2017$ số hữu tỷ dương được viết trên một đường tròn. Chứng minh tồn tại hai số được viết cạnh nhau trên đường tròn sao cho khi bỏ hai số đó thì $2015$ số còn lại không thể chia thành hai nhóm mà tổng các số mỗi nhóm bằng nhau.
\loigiai{
Ta có thể đưa về bài toán mà $2017$ số đã cho là các số nguyên dương, bằng cách quy đồng các số hữu tỷ ta được tử số là các số nguyên dương.\\
Ta gọi ước chung lớn nhất của $2017$ số nguyên dương này là $d$, ta chia tất cả các số này cho $d$ ta đưa bài toán về việc xét $2017$ số nguyên dương mà ước chung lớn nhất của chúng là $1$.
\begin{itemize}
\item Trường hợp 1: tổng của $2017$ số này là lẻ.\\
Nếu tất cả các cặp số đứng cạnh nhau là các số chẵn - lẻ (hoặc lẻ - chẵn) thì số cặp số như thế sẽ là một số chẵn, mà $2017$ là số lẻ nên sẽ tồn tại hai số chẵn (hoặc hai số lẻ) đứng cạnh nhau. Khi ta bỏ hai số này (có tổng là một số chẵn) thì $2015$ số còn lại sẽ có tổng là một số lẻ nên không thể chia $2015$ số còn lại thành hai phần mà có tổng bằng nhau.
\item Trường hợp 2: tổng của $2017$ số này là chẵn $\Rightarrow$ các số này không thể đều chẵn (1).\\
Vì ước chung lớn nhất của $2017$ số này là $1$ nên các số này không thể đều chẵn, do đó tồn tại ít nhất một số là lẻ (2).\\
Từ (1) và (2) ta thấy tồn tại một số chẵn và một số lẻ đứng cạnh nhau. Khi bỏ hai số này thì sẽ còn $2015$ số có tổng là một số lẻ. Do vậy không thể chia $2015$ số còn lại thành hai phần có tổng các số bằng nhau.
\end{itemize}
}
\end{ex}
