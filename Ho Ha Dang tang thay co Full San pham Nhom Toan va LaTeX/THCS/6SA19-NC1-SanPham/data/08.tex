
\section{Tính chất cơ bản của phân số. Rút gọn phân số}
\subsection{Trọng tâm kiến thức}
\begin{enumerate}
	\item Số có dạng $\dfrac{a}{b}$ với $a$ và $b$ là những số nguyên, $b\ne 0$ gọi là phân số.
	\item Hai phân số $\dfrac{a}{b}$ và $\dfrac{c}{d}$ gọi là bằng nhau nếu $ad=bc$.
	\item Nếu ta nhân cả tử và mẫu của một phân số với cùng một số nguyên khác 0 thì ta được một phân số bằng phân số đã cho \\
	\centerline{$\dfrac{a}{b}=\dfrac{am}{bm}$ với $m\in\mathbb{Z}$ và $m\ne 0$.}
	\item Nếu ta chia cả tử và mẫu của một phân số cho cùng một ước chung của chúng thì ta được một phân số bằng phân số đã cho.
	\item Mỗi một phân số có vô số phân số bằng nó. Các phân số bằng nhau là các cách viết khác nhau của cùng một số mà người ta gọi là các số hữu tỉ.
	\item Muốn rút gọn một phân số, ta chia cả tử và mẫu của phân số đó cho cùng một ước chung (khác $1$ và $-1$) của chúng.
	\item Phân số tối giản là phân số mà tử và mẫu chỉ có ước chung là $1$ và $-1$.\\
	 \centerline{ $\dfrac{a}{b}$ là phân số tối giản $\Leftrightarrow \text{ƯCLN}(|a|,|b|)=1$. } 
\end{enumerate}
\subsection{Các dạng bài tập và phương pháp giải}

\begin{vd}
	Quan sát dãy số sau để viết tiếp một phân số nữa theo quy luật của dãy: $$\dfrac{1}{2},\ \dfrac{5}{12},\ \dfrac{1}{3},\ \dfrac{1}{4},\ \dfrac{1}{6}.$$
	\loigiai{
	Viết dãy phân số dưới dạng như sau: 
	$$\dfrac{6}{12},\ \dfrac{5}{12},\ \dfrac{4}{12},\ \dfrac{3}{12},\ \dfrac{2}{12}.$$
	Do đó phân số tiếp theo của dãy là $\dfrac{1}{12}$.
}
\end{vd}

\begin{vd}
	Tìm số tự nhiên $n$ để cả ba phân số sau đều là số nguyên: $$\dfrac{15}{n},\ \dfrac{12}{n+2},\ \dfrac{6}{2n-5}.$$
	\loigiai{
	Để $\dfrac{15}{n}$ là số nguyên thì $n$ phải là ước của 15 hay $n\in\{1;3;5;15\}$.\\
	Để $\dfrac{12}{n+2}$ là số nguyên thì ta loại $n=3$, $n=5$ và $n=15$.\\
	Với $n=1$ thì ta thấy thỏa mãn vì khi đó ba phân số là $15$, $4$ và $-2$.\\
	Vậy $n=1$.
}
\end{vd}

\begin{vd}
	Cho phân số $\dfrac{1+2+3+\ldots+20}{6+7+8+\ldots+36}$.\\
	Hãy xóa một số hạng ở tử và một số hạng ở mẫu của phân số trên để giá trị của phân số không thay đổi.
	\loigiai{
	Ta có: $\dfrac{1+2+3+\ldots+20}{6+7+8+\ldots+36}=\dfrac{21\times 20:2}{42\times 31:2}=\dfrac{10}{31}$.\\
	Nếu xóa số 10 ở tử và xóa số 31 ở mẫu thì giá trị của phân số mới bằng $$\dfrac{10k-10}{31k-31}=\dfrac{10(k-1)}{31(k-1)}=\dfrac{10}{31}\ (\text{do}\ k-1\ne 0).$$
	Phân số mới bằng phân số đã cho nên yêu cầu bài toán được đảm bảo.\\
	Vậy ta sẽ xóa số 10 ở tử và xóa số 31 ở mẫu.
}
\end{vd}

\begin{vd}
	Tìm các số tự nhiên $a$ và $b$, biết rằng $\dfrac{a}{b}=\dfrac{132}{143}$ và $\text{BCNN}(a,b)=1092$.
	\loigiai{
	Ta có $\dfrac{a}{b}=\dfrac{132}{143}=\dfrac{12}{13}$ nên $a=12k$ và $b=13k$ với $k\in\mathbb{N}^*$.\\
	Do đó $\text{BCNN}(a,b)=\text{BCNN}(12k,13k)=12\times 13k$.\\
	Mà theo đề bài, $\text{BCNN}(a,b)=1092$ nên $12\times 13k=1092$ hay $k=7$.\\
	Vậy $a=12\times 7=84$ và $b=13\times 7=91$.
}
\end{vd}

\subsection{Bài tập tự luyện}
\begin{bt}
	Tìm số tự nhiên $x$, biết rằng:
	\begin{enumEX}{3}
		\item $\dfrac{7}{x}=\dfrac{x}{28}$;
		\item $\dfrac{10+x}{17+x}=\dfrac{3}{4}$;
		\item $\dfrac{40+x}{77-x}=\dfrac{6}{7}$.
	\end{enumEX}
	\loigiai{
	\begin{enumEX}{1}
		\item Ta có $\dfrac{7}{x}=\dfrac{x}{28}$ $\Leftrightarrow$ $7\times 28=x^2$ $\Leftrightarrow$ $x^2=196=14^2$ $\Leftrightarrow$ $x=14$ do $x\in\mathbb{N}$.\\
		Vậy $x=14$. 
		\item Ta có $\dfrac{10+x}{17+x}=\dfrac{3}{4}\Leftrightarrow 4(10+x)=3(17+x) \Leftrightarrow 40+4x=51+3x\Leftrightarrow x=11$.\\
		Vậy $x=11$.
		\item Điều kiện $x\ne 77$.\\
		Khi đó $\dfrac{40+x}{77-x}=\dfrac{6}{7}\Leftrightarrow 7(40+x)=6(77-x) \Leftrightarrow 280+7x=462-6x\\ \Leftrightarrow 13x=182 \Leftrightarrow x=14$.\\
		Vậy $x=14$.
	\end{enumEX}	
}
\end{bt}

\begin{bt}
	Tìm các số tự nhiên $a$ và $b$, biết $a^3+b^3=1216$ và phân số $\dfrac{a}{b}$ rút gọn được thành $\dfrac{3}{5}$.
	\loigiai{
	Ta có $\dfrac{a}{b}=\dfrac{3}{5}$ nên $a=3k$ và $b=5k$ với $k\in\mathbb{N}^*$.\\
	Theo đề bài, $a^3+b^3=1216$ nên $(3k)^3+(5k)^3=1216\Leftrightarrow 152k^3=1216\Leftrightarrow k^3=8\Leftrightarrow k=2$.\\
	Vậy $a=3\times 2=6$ và $b=5\times 2=10$.
}
\end{bt}

\begin{bt}
	Viết các phân số tối giản $\dfrac{a}{b}$ với $a$, $b$ là các số nguyên dương và $ab=100$.
	\loigiai{
	Do $100=2^2.5^2$ nên $\dfrac{a}{b}\in\{\dfrac{1}{100};\dfrac{100}{1};\dfrac{4}{25}; \dfrac{25}{4} \}$.
}
\end{bt}

\begin{bt}
	Rút gọn các phân số:
	\begin{enumEX}{2}
		\item $\dfrac{10.11+50.55+70.77}{11.12+55.60+77.84}$;
		\item $\dfrac{1.3.5.\ldots.49}{26.27.28.29. \ldots . 50}$.
	\end{enumEX}
	\loigiai{
	\begin{enumEX}{1}
		\item $\dfrac{10.11+50.55+70.77}{11.12+55.60+77.84}= \dfrac{10.11.(1+5.5+7.7)}{11.12(1+5.5+7.7)}=\dfrac{5}{6}$.
		\item Ta có $(26.27.\ldots .50).(2.4.\ldots . 50)=2^{25}.1.2.3.\ldots 50$ và
		$(1.3.5.\ldots.49).(2.4.\ldots . 50)=1.2.3.\ldots 50$ nên 
		$\dfrac{1.3.5.\ldots.49}{26.27.28.29. \ldots . 50}=\dfrac{1}{2^{25}}.$
	\end{enumEX}
}
\end{bt}

\begin{bt}
	Tìm số nguyên dương $n$ sao cho:
	\begin{enumEX}{1}
		\item $\dfrac{n+3}{n-2}$ là số nguyên âm;
		\item $\dfrac{n+7}{3n-1}$ là số nguyên;
		\item $\dfrac{3n+2}{4n-5}$ là số tự nhiên.
	\end{enumEX}
	\loigiai{
	\begin{enumEX}{1}
		\item $\dfrac{n+3}{n-2}$ là số nguyên $\Rightarrow$ $n+3\ \vdots \ n-2$ $\Leftrightarrow$ $n-2+5\ \vdots \ n-2$ $\Leftrightarrow$ $5\ \vdots \ n-2$.\\
		Do đó ta có bảng:
		\begin{center}
			\begin{tabular}{|c|c|c|c|c|}
				\hline
				$n-2$&1&-1&5&-5\\
				\hline
				$n$&3&1&7&-3\\
				\hline
				$\dfrac{n+3}{n-2}$ &6&-4&2&0\\
				\hline
			\end{tabular}
		\end{center}
		Vậy $n=1$.
		\item $\dfrac{n+7}{3n-1}$ là số nguyên $\Rightarrow$ $n+7 \ \vdots \ 3n-1$ $\Rightarrow$ $3n+21\ \vdots \ 3n-1$ $\Rightarrow$ $22\ \vdots \ 3n-1$.\\
		Do đó $3n-1\in\{-1;-22;2;11\}$. Ta có bảng:	
		\begin{center}
			\begin{tabular}{|c|c|c|c|c|}
				\hline
				$3n-1$&-1&-22&2&11\\
				\hline
				$n$&0&-7&1&4\\
				\hline
				$\dfrac{n+7}{3n-1}$ &-7&0&4&1\\
				\hline
			\end{tabular}
		\end{center}
		Vậy $n\in\{1;4\}$.
		\item $\dfrac{3n+2}{4n-5}$ là số tự nhiên $\Rightarrow$ $3n+2\ \vdots \ 4n-5$ $\Rightarrow$ $12n+8\ \vdots \ 4n-5$ $\Rightarrow$ $3(4n-5)+23\ \vdots \ 4n-5$ $\Rightarrow$ $23\ \vdots \ 4n-5$.\\
		Do đó $4n-5\in\{-1;23\}$ hay $n\in\{1;7\}$.\\
		Thử lại ta thấy $n=7$ thỏa mãn.\\
		Vậy $n=7$.
	\end{enumEX}
}
\end{bt}

\begin{bt}
	Chứng minh rằng $\dfrac{n-5}{3n-14}$ là phân số tối giản với mọi số nguyên $n$.
	\loigiai{
	Gọi $d$ là ước chung lớn nhất của $n-5$ và $3n-14$ ($d\in\mathbb{N}^*$).\\
	Do đó $n-5\ \vdots \ d$ và $3n-14\ \vdots \ d$.\\
	Mà $3n-14=3(n-5)+1$ nên $1\ \vdots \ d$, suy ra $d=1$.\\
	Vậy $\dfrac{n-5}{3n-14}$ là phân số tối giản với mọi số nguyên $n$. 
}
\end{bt}

\begin{bt}
	Tìm số nguyên $n$ để phân số $\dfrac{2n-1}{3n+2}$ rút gọn được.
	\loigiai{
	Gọi $d$ là ước chung lớn nhất của $2n-1$ và $3n+2$ ($d\in\mathbb{N}^*$).\\
	Do $3(2n-1)-2(3n+2)\ \vdots \ d$ nên $-7\ \vdots \ d$, suy ra $d=1$ hoặc $d=7$.\\
	Để phân số $\dfrac{2n-1}{3n+2}$ rút gọn được thì ước chung lớn nhất của $2n-1$ và $3n+2$ phải khác 1, do đó $2n-1\ \vdots \ 7$ $\Leftrightarrow$ $2n+6\ \vdots \ 7$ $\Leftrightarrow$ $n+3\ \vdots \ 7$ (vì $\text{ƯCLN}(2;7)=1$).\\
	Khi đó $n=7k-3$ với $k\in\mathbb{Z}$.\\
	Thử lại $\dfrac{2n-1}{3n+2}=\dfrac{2(7k-3)-1}{3(7k-3)+2}=\dfrac{14k-7}{21k-7} =\dfrac{2k-1}{3k-1}$ rút gọn được.\\
	Vậy $n=7k-3$ với $k\in\mathbb{Z}$. 
}
\end{bt}

\begin{bt}
	Tìm số tự nhiên $n$ nhỏ nhất để các phân số sau đều là các phân số tối giản:
	$$\dfrac{n+7}{3},\dfrac{n+8}{4},\dfrac{n+9}{5},\dfrac{n+10}{6},\dfrac{n+11}{7}.$$
	\loigiai{
	Viết các phân số đã cho dưới dạng
	$$\dfrac{n+4}{3}+1,\dfrac{n+4}{4}+1,\dfrac{n+4}{5}+1,\dfrac{n+4}{6}+1,\dfrac{n+4}{7}+1.$$
	Do đó để các phân số đã cho đều tối giản thì $n+4$ phải nguyên tố cùng nhau với 3, 4, 5, 6, 7.\\
	Theo bài ra thì $n$ nhỏ nhất nên $n+4$ phải là số nguyên tố nhỏ nhất lớn hơn 7, suy ra $n+4=11$ hay $n=7$.\\
	Vậy $n=7$.
}
\end{bt}

\begin{bt}
	Tìm các số tự nhiên $a$ và $b$, biết rằng $\dfrac{a}{b}=\dfrac{49}{56}$ và $\text{ƯCLN}(a,b)=12$.
	\loigiai{
	Ta có $\dfrac{a}{b}=\dfrac{49}{56}=\dfrac{7}{8}$ nên $a=7k$ và $b=8k$ với $k\in\mathbb{N}^*$.\\
	Suy ra $\text{ƯCLN}(a,b)=\text{ƯCLN}(7k,8k)=k.\text{ƯCLN}(7,8)=k$.\\
	Theo đề bài, $\text{ƯCLN}(a,b)=12$ nên $k=12$.\\
	Vậy $a=7.12=84$ và $b=8.12=96$.
}
\end{bt}

\begin{bt}
	Tìm các số tự nhiên $a$, $b$, $c$ và $d$ nhỏ nhất sao cho $$\dfrac{a}{b}=\dfrac{5}{14},\ \dfrac{b}{c}=\dfrac{21}{28},\ \dfrac{c}{d}=\dfrac{6}{11}.$$
	\loigiai{
	Do $\dfrac{a}{b}=\dfrac{5}{14}$ nên $a=5m$ và $b=14m$ với $m\in\mathbb{N}^*$.\\
	Do $\dfrac{b}{c}=\dfrac{21}{28}=\dfrac{3}{4}$ nên $b=3n$ và $c=4n$ với $n\in\mathbb{N}^*$.\\
	Do $\dfrac{c}{d}=\dfrac{6}{11}$ nên $c=6k$ và $d=11k$ với $k\in\mathbb{N}^*$.\\
	Từ $14m=3n$ suy ra $n\ \vdots \ 14$ do $\text{ƯCLN}(3,14)=1$.\\
	Từ $4n=6k$ $\Leftrightarrow$ $2n=3k$ suy ra $n\ \vdots \ 3$ do $\text{ƯCLN}(2,3)=1$.\\
	Để $n$ nhỏ nhất thì $n=\text{BCNN}(3,14)=42$, khi đó $m=9$ và $k=28$.\\
	Vậy $a=5.9=45$, $b=14.9=126$, $c=4.42=168$ và $d=11.28=308$.
	
}
\end{bt}
