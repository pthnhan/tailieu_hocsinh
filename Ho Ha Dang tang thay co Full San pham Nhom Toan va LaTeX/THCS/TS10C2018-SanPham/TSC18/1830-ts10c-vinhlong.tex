	\begin{listEX}
		\item Cho biểu thức $ A=\left(\dfrac{x+3\sqrt{x}+2}{x\sqrt{x}-8}-\dfrac{1}{\sqrt{x}-2}\right):\dfrac{1}{\sqrt{x}} $ với $x>0$ và $ x\neq 4$. Tìm giá trị của $ A $ tại $ x=14+6\sqrt{5} $.
		\item Tính giá trị biểu thức $ B=\sqrt{12-\sqrt{80-32\sqrt{3}}}-\sqrt{12+\sqrt{80-32\sqrt{3}}} $.
	\end{listEX}
	\loigiai{
		\begin{listEX}
			\item Với $x>0$, $ x\neq 4 $ ta có \begin{align*}
			A&= \left(\dfrac{x+3\sqrt{x}+2}{x\sqrt{x}-8}-\dfrac{1}{\sqrt{x}-2}\right):\dfrac{1}{\sqrt{x}}\\
			&=\left[\dfrac{x+3\sqrt{x}+2}{\left(\sqrt{x}-2\right)\left(x+2\sqrt{x}+4\right)}-\dfrac{x+2\sqrt{x}+4}{\left(\sqrt{x}-2\right)\left(x+2\sqrt{x}+4\right)}\right]\sqrt{x}\\
			&=\dfrac{\sqrt{x}-2}{\left(\sqrt{x}-2\right)\left(x+2\sqrt{x}+4\right)}\cdot \sqrt{x}\\
			&=\dfrac{\sqrt{x}}{x+2\sqrt{x}+4}.
			\end{align*}
			Ta lại có $ x=14+6\sqrt{5}=9+2\cdot 3\cdot\sqrt{5}+5=\left(3+\sqrt{5}\right)^2\Rightarrow \sqrt{x}=\left|3+\sqrt{5}\right|=3+\sqrt{5} $.\\
			Khi đó, giá trị của $ A $ tại $ x=14+6\sqrt{5} $ là $$ \dfrac{3+\sqrt{5}}{14+6\sqrt{5}+2\left(3+\sqrt{5}\right)+4}=\dfrac{3+\sqrt{5}}{24+8\sqrt{5}}=\dfrac{3+\sqrt{5}}{8\left(3+\sqrt{5}\right)}=\dfrac{1}{8}. $$
			\item Ta có $ B^2=24-8\sqrt{4+2\sqrt{3}}=\left(2\sqrt{3}-2\right)^2 $.\\
			Suy ra $ B=\pm\left(2\sqrt{3}-2\right) $.\\
			Do $ B<0 $ nên $ B=2-2\sqrt{3} $. 
		\end{listEX}
	}
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Vĩnh Long, tỉnh Vĩnh Long, năm 2018]%[Nguyễn Bình Nguyên, Dự án EX-TS10C-18]%[9D0K2]
	Cho phương trình $ x^2+\left(2m-3\right)x-m^2-1=0\refstepcounter{equation}\hfill (\theequation)\label{1830.1}$ ($x$ ẩn số, $m$ là tham số).
	\begin{listEX}
		\item Chứng tỏ rằng phương trình \eqref{1830.1} có hai nghiệm phân biệt với mọi giá trị của $ m $.
		\item Giả sử $ x_1$, $x_2 $ là hai nghiệm của phương trình \eqref{1830.1}. Tìm $m$ để phương trình  có hai nghiệm phân biệt $x_1<x_2$ thỏa mãn $ \left|x_1\right|-\left|x_2\right|=3 $.
	\end{listEX}
	\loigiai{
		\begin{listEX}
			\item Ta có $ ac=1\left(-m^2-1\right)=-m^2-1<0\ \forall m $ nên phương trình \eqref{1830.1} có hai nghiệm phân biệt trái dấu với mọi $m$.
			\item Do phương trình \eqref{1830.1} có hai nghiệm trái dấu và $ x_1<x_2 $ nên $ x_1<0 $, $ x_2>0 $.\\
			Suy ra $ \left|x_1\right|=-x_1, \left|x_2\right| =x_2$ và $$ \left|x_1\right|-\left|x_2\right|=3\Leftrightarrow -x_1-x_2=3\Leftrightarrow-\left(x_1+x_2\right)=3\Leftrightarrow 2m-3=3\Leftrightarrow m=3.$$
		\end{listEX}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Vĩnh Long, tỉnh Vĩnh Long, năm 2018]%[Nguyễn Bình Nguyên, Dự án EX-TS10C-18]%[9D0G2]
	\begin{listEX}
		\item Giải phương trình $ \left(x^2-9\right)^2=12x+1 $.
		\item Giải hệ phương trình $ \heva{& \sqrt{2x-y-9}-36+x^2=0\\ & y^2-xy+9=0.} $
	\end{listEX}
	\loigiai{
		\begin{listEX}
			\item Giải phương trình $ \left(x^2-9\right)^2=12x+1 $.\\
			Ta có \begin{eqnarray*}
				&&\left(x^2-9\right)^2=12x+1\\
				&\Leftrightarrow & x^4-18x^2+81=12x+1\\
				&\Leftrightarrow& x^4+18x^2+81=36x^2+12x+1\\
				&\Leftrightarrow & \left(x^2+9\right)^2=\left(6x+1\right)^2\\
				&\Leftrightarrow &\hoac{& x^2+9=6x+1\\ & x^2+9=-6x-1}\\
				&\Leftrightarrow &\hoac{ &x^2-6x+8=0\\ &x^2+6x+10=0\ (\text{vô nghiệm})}\\
				&\Leftrightarrow & \hoac{&x=2\\ &x=4.}
			\end{eqnarray*} 
			Vậy tập nghiệm của phương trình là $ S=\{2; 4\} $.
			\item Giải hệ phương trình $ \heva{& \sqrt{2x-y-9}-36+x^2=0 & \refstepcounter{equation}\hfill (\theequation)\label{1830.2}\\ & y^2-xy+9=0.& \refstepcounter{equation}\hfill (\theequation)\label{1830.3}} $\\
			Điều kiện $ 2x-y-9\geq 0 $.\\
			Ta có (\ref{1830.3}) $\Leftrightarrow 4y^2-4xy+x^2=x^2-36\Leftrightarrow (2y-x)^2=x^2-36.\refstepcounter{equation}\hfill (\theequation)\label{1830.4} $\\
			Thay (\ref{1830.4}) vào (\ref{1830.2}) ta được $$\sqrt{2x-y-9}+(2y-x)^2=0\Leftrightarrow\heva{& 2x-y-9=0\\ & 2y-x=0}\Leftrightarrow\heva{& x=6\\ &y=3}\ \text{thỏa mãn điều kiện.} $$ 
			Vậy hệ phương trình đã cho có nghiệm $\heva{& x=6\\ &y=3.}$
		\end{listEX}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Vĩnh Long, tỉnh Vĩnh Long, năm 2018]%[Nguyễn Bình Nguyên, Dự án EX-TS10C-18]%[9D0G5]
	\begin{listEX}
		\item Tìm tất cả các số tự nhiên $x$ để giá trị biểu thức $P=-x^4+x^2+14x+49$ là số nguyên tố.
		\item Tìm nghiệm nguyên của phương trình $x^2-xy+y^2=2x-3y-2$.
	\end{listEX}
	
	\loigiai{
		\begin{listEX}
			\item Ta có $ P=\left(x^2+14x+49\right)-\left(x^2\right)^2=(x+7)^2-\left(x^2\right)^2=\left(x+7-x^2\right)\left(x+7+x^2\right) $.\\
			Dễ thấy $x+7+x^2\geq x+7-x^2$ và $ x+7+x^2>1$ với mọi số tự nhiên $x$.\\
			Từ đó, kết hợp với $P$ nguyên tố suy ra $$x+7-x^2=1\Leftrightarrow x^2-x-6=0\Leftrightarrow\hoac{& x=3\\ &x=-2\ (\text{loại}).}$$
			Với $ x=3$ thì $P=19 $ thỏa mãn bài toán.
			Vậy số tự nhiên $ x $ cần tìm là $ x=3 $.
			\item Xét phương trình $ x^2-xy+y^2=2x-3y-2.\refstepcounter{equation}\hfill (\theequation)\label{1830.5}$\\
			Ta có (\ref{1830.5}) $\Leftrightarrow x^2-(y+2)x+y^2+3y+2=0. \refstepcounter{equation}\hfill (\theequation)\label{1830.6} $\\
			Phương trình (\ref{1830.6}) có $ \Delta=(y+2)^2-4\left(y^2+3y+2\right)=-3y^2-8y-4 $.\\
			Phương trình (\ref{1830.6}) có nghiệm khi và chỉ khi $$\Delta\geq 0\Leftrightarrow -3y^2-8y-4\geq 0\Leftrightarrow -2\leq y\leq -\dfrac{2}{3}. $$
			Vì $ y $ nguyên nên $ y=-2 $ hoặc $ y=-1 $.\\
			Với $ y=-2 $ thì từ (\ref{1830.6}) suy ra $ x^2=0\Leftrightarrow x=0 $.\\
			Với $ y=-1 $ thì từ (\ref{1830.6}) suy ra $ x^2-x=0\Leftrightarrow \hoac{ &x=1\\ &x=0.} $\\
			Vậy nghiệm nguyên $ (x; y) $ của phương trình đã cho là $ (0; -2), (0; -1), (1; -1) $.
		\end{listEX}
	}
\end{ex}
\begin{ex}%[Đề thi vào 10, Nguyễn Bỉnh Khiêm, tỉnh Vĩnh Long, năm 2018]%[Nguyễn Bình Nguyên, Dự án EX-TS10C-18]%[8H0K2]
Cho tam giác $ ABC$ vuông tại $A$ có $ AB=6$ cm, $AC=8$ cm. Các đường phân giác trong và phân giác ngoài của góc $B$ lần lượt cắt đường thẳng $AC$ tại $M$ và $N$. Tính diện tích của tam giác $ BMN$.

\begin{center}
\begin{tikzpicture}

\tkzMarkAngles[size=1cm,arc=ll,mark=|](A,B,M M,B,C)
\tkzMarkAngles[size=1cm,arc=ll,mark=|](A,B,M M,B,C)
\tkzMarkAngles[size=0.8cm,mark=||,mksize=2](C',B,N N,B,A)
\end{tikzpicture}	
\end{center}
\begin{itemize}
	\item $ \triangle ABC$ vuông tại $A$ suy ra $ BC=10$ cm.
	\item $ BM$ là đường phân giác trong của $ \triangle ABC \Rightarrow \dfrac{MA}{BA}=\dfrac{MC}{BC}$. Áp dụng tính chất dãy tỉ số bằng nhau ta có $\dfrac{MA}{BA}=\dfrac{MC}{BC}=\dfrac{MA+MC}{BA+BC}=\dfrac{AC}{BA+BC}=\dfrac{1}{2}\Rightarrow MA=3$ (cm).
	\item $ BN$ là đường phân giác ngoài của $ \triangle ABC \Rightarrow \dfrac{NA}{BA}=\dfrac{NC}{BC}$. Áp dụng tính chất dãy tỉ số bằng nhau ta có $\dfrac{NA}{BA}=\dfrac{NC}{BC}=\dfrac{NC-NA}{BC-BA}=\dfrac{AC}{BC-BA}=2\Rightarrow NA = 12$ (cm).
	\item Từ đó ta tính được $ NM=NA+MA=15$ (cm). Suy ra $ S_{\triangle BMN }= \dfrac{1}{2}BA \cdot NM =45$ (cm$^2$). 
\end{itemize}

\begin{ex}%[Đề thi vào 10, Nguyễn Bỉnh Khiêm, tỉnh Vĩnh Long, năm 2018]%[Nguyễn Bình Nguyên, Dự án EX-TS10C-18]%[9H0G2]
   Cho tam giác $ABC$ vuông tại $A$ với  $\left(AB<AC\right)$ và đường cao $AH$. Vẽ đường tròn $(O)$ đường kính $BC$. Trên cung nhỏ $AC$ lấy điểm $E$ ($E\ne A$, $E\ne C$) sao cho hai tia $AE$ và $BC$ cắt nhau tại $I$; $AC$ cắt $BE$ tại $N$. Kéo dài $AH$ cắt đường tròn $(O)$ tại điểm thứ hai là $D$, $DE$ cắt $BC$ tại $M$.
\begin{enumerate}[a)]
	\item Chứng minh $MN$ song song $AD$.
	\item Chứng minh hai tam giác $OME$ và $OEI$ đồng dạng.
\end{enumerate}
\loigiai
    {
 \begin{tikzpicture}[>=stealth,line join=round,line cap=round,font=\footnotesize,scale=1]
 \tkzDefPoint(-3,0){B}
 \tkzDefPoint(3,0){C}
 \tkzDefPoint(0,0){O}
 \tkzDrawCircle[diameter](B,C)
 \tkzDefPointBy[rotation = center O angle 120](C)\tkzGetPoint{A}
 \tkzDrawAltitude(C,B)(A) \tkzGetPoint{H}
 \tkzDefPointBy[symmetry = center H](A)    \tkzGetPoint{D}
 \tkzDefPointBy[rotation = center O angle 35](C)\tkzGetPoint{E}
 \tkzInterLL(A,E)(C,B)    \tkzGetPoint{I}
\tkzInterLL(D,E)(C,B)    \tkzGetPoint{M}
\tkzInterLL(B,E)(C,A)    \tkzGetPoint{N}
\tkzInterLC(E,O)(O,A)    \tkzGetPoints{}{F}

\tkzDrawSegments(B,C A,B A,C A,D B,E E,C D,E A,I I,C A,O E,F M,N)
\tkzDrawPoints(C,B,O,A,H,E,I,M,N,D,F)
\tkzLabelPoints[below right](C)
\tkzLabelPoints[below right](H)
\tkzLabelPoints[above](A,E,N)
\tkzLabelPoints[left](B)
\tkzLabelPoints[below](D,I,M,O,F)
\tkzMarkRightAngles(B,A,C D,H,B B,E,C)
\end{tikzpicture} 
\begin{enumerate}[a)]
	\item Chứng minh $MN\parallel AD$.\\
	Dễ thấy tứ giác $MNEC $ nội tiếp, do $\widehat{NEM}=\widehat{BED}=\widehat{ACB}=\widehat{NCM}$. Mà $\widehat{NEC}=90^\circ$, suy ra $\widehat{NMC}=90^\circ$.\\
	Ta có $ MN\perp BC$, $AD \perp BC$ suy ra $ MN \parallel AD$.
	\item Chứng minh $\triangle OME \backsim \triangle OEI$.\\
	Gọi $F$ là giao điểm của $OE$ với đường tròn $ (O) $, ($F$ khác với $E$).
	Ta có $\widehat{BOF}=\widehat{EOC}\Rightarrow \wideparen{EC}=\wideparen{BF}$.
	Suy ra $ \widehat{DEF}=\dfrac{1}{2}\text{sđ}\wideparen{FD}=\dfrac{1}{2}\left(\text{sđ}\wideparen{BD}-\text{sđ}\wideparen{BF}\right)$.\\
	Mặt khác $\widehat{AIB}=\dfrac{1}{2}\left(\text{sđ}\wideparen{AB}-\text{EC}\wideparen{BF}\right)$, (tính chất góc có đỉnh bên ngoài đường tròn). Suy ra $\widehat{AIB}=\widehat{DEF}$.\\
	Xét hai tam giác $OME$ và $OEI$ có
	\begin{itemize}
		\item 	$\widehat{EOI}$ chung.
		\item $\widehat{EIO}=\widehat{MEO}$.
	\end{itemize}
	$\Rightarrow \triangle OME \backsim \triangle OEI$.
\end{enumerate}

    }

\begin{ex}%[Đề thi vào 10, Nguyễn Bỉnh Khiêm, tỉnh Vĩnh Long, năm 2018]%[Nguyễn Bình Nguyên, Dự án EX-TS10C-18]%[9D0G7]
Cho $a$, $b$, $c$ là các số dương. Chứng minh rằng
\begin{enumerate}[a)]
	\item $\dfrac{a^3}{a^2+b^2}\geq a-\dfrac{b}{2}$.
	\item  $\dfrac{a^3}{a^2+ab+b^2}+\dfrac{b^3}{b^2+bc+c^2}+\dfrac{c^3}{c^2+ca+a^2}\geq \dfrac{a+b+c}{3}$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
	\item Ta có $\dfrac{a^3}{a^2+b^2}=\dfrac{a(a^2+b^2)-ab^2}{a^2+b^2}=a-\dfrac{ab^2}{a^2+b^2}\geq a -\dfrac{ab^2}{2ab}=a-\dfrac{b}{2}.$
	\item Tương tự theo câu a, ta có $\dfrac{b^3}{b^2+c^2}\geq b-\dfrac{c}{2}$ và $\dfrac{c^3}{a^2+c^2}\geq c-\dfrac{a}{2}$.\\
	Cộng vế theo vế ba bất đẳng thức trên ta có
	$$\dfrac{a^3}{a^2+b^2}+\dfrac{b^3}{c^2+b^2}+\dfrac{c^3}{a^2+c^2}\geq\dfrac{a+b+c}{2}.$$
	Ta có $\dfrac{a^3}{a^2+ab+b^2}\geq \dfrac{a^3}{a^2+\dfrac{a^2+b^2}{2}+b^2}=\dfrac{2}{3}\cdot \dfrac{a^3}{a^2+b^2}$.\\
	Tương tự ta cũng có $\dfrac{b^3}{b^2+bc+c^2} \geq \dfrac{2}{3}\cdot \dfrac{b^3}{b^2+c^2}$ và $\dfrac{c^3}{c^2+ac+a^2}\geq \dfrac{2}{3}\cdot \dfrac{c^3}{c^2+a^2}$.\\
	Cộng vế theo vế ba bất đẳng thức trên ta có
	$$\dfrac{a^3}{a^2+ab+b^2}+\dfrac{b^3}{b^2+bc+c^2}+\dfrac{c^3}{c^2+ac+a^2}\geq \dfrac{2}{3}\left(\dfrac{a^3}{a^2+b^2}+\dfrac{b^3}{c^2+b^2}+\dfrac{c^3}{a^2+c^2}\right)\geq \dfrac{a+b+c}{3}.$$
\end{enumerate}
}
\end{ex}
