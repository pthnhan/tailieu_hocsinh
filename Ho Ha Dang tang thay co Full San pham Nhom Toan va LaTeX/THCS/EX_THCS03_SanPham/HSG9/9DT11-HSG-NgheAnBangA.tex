\begin{ex}%[Đề thi HSG, Nghệ an, Bảng A, năm 2011]%[9D5B2]%[9D5K3]
\hfill
\begin{enumerate}
\item Cho các số nguyên $a_1,\ a_2,\ a_3,\cdots,\ a_n$. Đặt $S=a_1^3+a_2^3+\cdots+a_n^3$ và $P=a_1+a_2+\cdots+a_n$. Chứng minh rằng: $S$ chia hết cho $6$ khi và chỉ khi $P$ chia hết cho $6$.
\item Cho $A=n^6-n^4+2n^3+2n^2$ (với $n \in \mathbb{N},\ n>1$). Chứng minh rằng $A$ không là số chính phương
\end{enumerate}    
\loigiai{\hfill
\begin{enumerate}
\item Với số nguyên $x$, ta có $$x^3-x=x(x-1)(x+1) \vdots 6.$$
Do đó 
$$S-P=\left( a_1^3-a_1 \right)+\left( a_2^3-a_2 \right)+\cdots+\left( a_n^3-a_n \right) \vdots 6.$$
Suy ra $S \vdots 6 \Leftrightarrow P \vdots 6.$
\item Ta có $A=n^2\left(n^4-n^2+2n+2 \right)$.
\begin{itemize}
\item Với $n=2$, ta có $n^4-n^2+2n+2=18$ không là số chính phương nên $A$ không là số chính phương.
\item Với $n \geq 3$, ta có $n^2 -2n-2>0$ nên 
$$\left(n^2-1\right)^2=n^4-2n^2+1< n^4-n^2+2n+2<\left( n^2\right)^2,$$ do đó $n^4-n^2+2n+2$ không là số chính phương.
\end{itemize}
Từ đó ta có $A$ không là số chính phương với mọi $n \geq 2.$
\end{enumerate}

}    
\end{ex}

\begin{ex}%[Đề thi HSG, Nghệ an, Bảng A, năm 2011]%[9D4K5]%[9D4K5]
\hfill
\begin{enumerate}
\item Giải phương trình: $10\sqrt{x^3+1}=3x^2+6$.
\item Giải hệ phương trình: $\begin{cases} x+\dfrac{1}{y} =3\\ y+\dfrac{1}{z}=3 \\ z+\dfrac{1}{x}=3 \end{cases}$.
\end{enumerate}
\loigiai{ \hfill 
\begin{enumerate}
\item Điều kiện: $x \geq -1.$\\
Ta có $x^3+1=(x+1)(x^2-x+1),\ x^2+2=(x^2-x+1)+(x+1)$. \\ Đặt $a=\sqrt{x+1},\ b=\sqrt{x^2-x+1}$ ta có phương trình
$$10ab=3a^2+3b^2 \Leftrightarrow (3a-b)(a-3b)=0 \Leftrightarrow a=3b,\ b=3a. $$
\begin{itemize}
\item $a=3b \Leftrightarrow \sqrt{x+1}=3\sqrt{x^2-x+1} \Leftrightarrow 9x^2-10x+8=0$. Phương trình vô nghiệm.
\item $b=3a \Leftrightarrow \sqrt{x^2-x+1}=3\sqrt{x+1}\Leftrightarrow x^2-10x-8=0 \Leftrightarrow x= 5 \pm \sqrt{33}.$
\end{itemize}
Vậy nghiệm của phương trình là $x= 5 \pm \sqrt{33}$.
\item Từ phương trình $x+\dfrac{1}{y}=3$ ta có $x=\dfrac{3y-1}{y}$, thay vào phương trình thứ ba ta được
$$z=3-\dfrac{1}{x}=3-\dfrac{y}{3y-1}=\dfrac{8y-3}{3y-1}.$$
Thay $z$ và phương trình thứ hai ta có
$$y+\dfrac{3y-1}{8y-3}=3 \Leftrightarrow y^2-3y+1=0 \Leftrightarrow y=\dfrac{3\pm \sqrt{5}}{2}.$$
Từ đó ta có nghiệm của hệ là $x=y=z=\dfrac{3\pm \sqrt{5}}{2}.$
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi HSG, Nghệ an, Bảng A, năm 2011]%[9D5K1]%[9D5G1]
\hfill 
 \begin{enumerate}
 \item Cho các số thực dương $x,\ y,\ z$ thỏa mãn $\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=4.$ Chứng minh rằng $$\dfrac{1}{2x+y+z}+\dfrac{1}{x+2y+z}+\dfrac{1}{x+y+2z} \leq 1.$$
\item Cho các số thực dương thỏa mãn $x^{2011}+y^{2011}+z^{2011}=3.$ Tìm giá trị lớn nhất của biểu thức $M=x^2+y^2+z^2.$
 \end{enumerate}
 
\loigiai{ \hfill
\begin{enumerate}
\item Với $a,\ b,\ c,\ d>0$ ta có $$\dfrac{1}{a+b+c+d} \leq \dfrac{1}{16} \left(\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}+\dfrac{1}{d} \right).$$
Áp dụng bất đẳng thức trên ta có
$$\begin{aligned} &\dfrac{1}{2x+y+z} \leq \dfrac{1}{16}\left(\dfrac{2}{x}+\dfrac{1}{y}+\dfrac{1}{z} \right) \\&\dfrac{1}{x+2y+z} \leq \dfrac{1}{16}\left(\dfrac{1}{x}+\dfrac{2}{y}+\dfrac{1}{z} \right) \\& \dfrac{1}{x+y+2z} \leq \dfrac{1}{16}\left(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{2}{z} \right). \end{aligned} $$
Công các bất đẳng thức trên theo vế ta có đpcm.
\item Áp dụng bất đẳng thức Cauchy cho $2011$ số gồm $2$ số $x^{2011}$ và $2009$ số $1$ ta có 
$$2x^{2011}+2009=x^{2011}+x^{2011}+1+\cdots+1 \geq 2011 \sqrt[2011]{x^{2.2011}}=2011.x^2.$$
Tương tự:
$$2y^{2011}+2009 \geq 2011.y^2,\ 2z^{2011}+2009 \geq 2011 z^2.$$
Cộng các bất đẳng thức trên theo vế ta có
$$2011(x^2+y^2+z^2) \leq 2\left( x^{2011}+y^{2011}+z^{2011} \right) +3.2009=3.2011.$$
Suy ra $M \leq 3$. Đẳng thức xảy ra khi $x=y=z=1.$\\
Vậy $\max M=3.$
\end{enumerate}
    }
\end{ex}

\begin{ex}%[Đề thi HSG, Nghệ an, Bảng A, năm 2011]%[9H3B7]%[9H3K7][9H3G7]
Cho tam giác $ABC$ có ba góc nhọn nội tiếp đường tròn $(O)$, $H$ là trực tâm của tam giác. Gọi $M$ là một điểm nằm trên cung $BC$ không chứa $A$ ($M$ không trùng với $B$ và $C$). Gọi $N$ và $P$ lần lượt là điểm đối xứng của $M$ qua các đường thẳng $AB$ và $AC$.
\begin{enumerate}
\item Chứng minh rằng ba điểm $N,\ H,\ P$ thẳng hàng.
\item Khi $\widehat{BOC}=120^\circ$, xác định vị trí của điểm $M$ để $\dfrac{1}{MB}+\dfrac{1}{MC} $ đạt giá trị nhỏ nhất.
\end{enumerate}

\loigiai{
\begin{center}
\begin{tikzpicture} 
\tkzInit[xmax=6,xmin=-5,ymax=5,ymin=-2] \tkzClip
\tkzDefPoints{-1/0/B, 0/4/A, 4/0/C, 1/0.7/X}
\tkzDefCircle[circum](A,B,C) \tkzGetPoint{O}
\tkzDrawCircle(O,A)
\tkzDefLine[perpendicular=through A](B,C) \tkzGetPoint{a}
\tkzDefLine[perpendicular=through B](A,C) \tkzGetPoint{b}
\tkzInterLL(A,a)(B,b) \tkzGetPoint{H}
\tkzInterLC(A,X)(O,A) \tkzGetPoints{M}{A}
\tkzDefPointBy[reflection = over A--B](M)
\tkzGetPoint{N}
\tkzDefPointBy[reflection = over A--C](M)
\tkzGetPoint{P}
\tkzInterLL(A,H)(B,C) \tkzGetPoint{D}
\tkzInterLL(B,H)(A,C) \tkzGetPoint{E}
\tkzInterLL(C,H)(B,A) \tkzGetPoint{F}
\tkzInterLL(A,B)(M,N) \tkzGetPoint{T}
\tkzInterLL(M,P)(A,C) \tkzGetPoint{K'}
\tkzDefLine[perpendicular=through O](B,C) \tkzGetPoint{o}
\tkzInterLC(O,o)(O,A) \tkzGetPoints{L}{J}
\tkzInterLC(J,M)(M,B) \tkzGetPoints{K}{L'}

\tkzDrawSegments(A,B B,C C,A A,M M,N M,P N,P A,P A,N A,D B,E C,F B,T C,P J,M B,M B,K)
\tkzDrawPoints(A,B,C,O,H,M,N,P,D,E,F,T,K',J,K)
\tkzLabelPoints(C,M,P,D)
\tkzLabelPoints[above](E,J)
\tkzLabelPoints[above left](O)
\tkzLabelPoints[left](B,N,F)
\tkzLabelPoints[above left](A)
\tkzLabelPoints[below right](H,K)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Gọi $D,\ E,\ F$ lần lượt là chân đường cao hạ từ $A,\ B,\ C$. Ta có tứ giác $CDHE$ nội tiếp và $M,\ N$ đối xứng nhau qua $AB$, nên $$\widehat{AHE}=\widehat{ACB}=\widehat{AMB}=\widehat{ANB}.$$
Suy ra tứ giác $ANBH$ nội tiếp, nên $$\widehat{NHB}=\widehat{NAB} =\widehat{BAM}.$$
Tương tự $\widehat{PHC}=\widehat{CAM}$. Từ đó, suy ra 
$$\widehat{NHB}+\widehat{BHC}+\widehat{CHP}=\widehat{BAM}+\widehat{BHC}+\widehat{MAC}=\widehat{BAC}+\widehat{BHC}=180^\circ.$$
Suy ra $N,\ H,\ P$ thẳng hàng.
\item Gọi $J$ là điểm chính giữa cung lớn $BC$, trên đoạn $MJ$ lấy $K$ sao cho $MB=MK$. Ta có $\widehat{BIC}=120^\circ$ và $\Delta JKB=\Delta CM$, nên $JK=CM$. Do đó $BM+CM=MK+KJ=JM$. Khi đó
$$\dfrac{1}{MB}+\dfrac{1}{MC} \geq \dfrac{4}{MB+MC}=\dfrac{4}{JM} \geq \dfrac{4}{2R}.$$
Đẳng thức xảy ra khi $M$ là điểm chính giữa cung $BC$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi HSG, Nghệ an, Bảng A, năm 2011]%[9D5G5]
Cho tam giác $ABC$ nội tiếp đường tròn tâm $O$, một điểm $I$ chuyển động trên cung $BC$ không chứa điểm $A$ ($I$ không trùng với $B$ và $C$). Đường thẳng vuông góc với $IB$ tại $I$ cắt đường thẳng $AC$ tại $E$, đường thẳng vuông góc với $IC$ tại $I$ cắt đường thẳng $AB$ tại $F$. Chứng minh rằng đường thẳng $EF$ luôn đi qua một điểm cố định.
\loigiai{
\begin{center}
\begin{tikzpicture} 
\tkzInit[xmax=6,xmin=-5,ymax=5,ymin=-2] \tkzClip
\tkzDefPoints{-1/0/B, 0/4/A, 4/0/C, 1/0.7/X}
\tkzDefCircle[circum](A,B,C) \tkzGetPoint{O}
\tkzDrawCircle(O,A)
\tkzInterLC(A,X)(O,A) \tkzGetPoints{I}{A}
\tkzDefLine[perpendicular=through I](B,I) \tkzGetPoint{b}
\tkzDefLine[perpendicular=through I](I,C) \tkzGetPoint{c}
\tkzInterLL(I,b)(A,C) \tkzGetPoint{E}
\tkzInterLL(I,c)(A,B) \tkzGetPoint{F}
\tkzDefPointBy[reflection = over E--F](I)
\tkzGetPoint{K}


\tkzDrawSegments(A,B B,C C,A B,I C,I E,I F,I E,F K,E K,F K,B K,A K,I)
\tkzDrawPoints(A,B,C,O,I,E,F,K)
\tkzLabelPoints(C)
\tkzLabelPoints[above](K)
\tkzLabelPoints[above left](O)
\tkzLabelPoints[left](B,F)
\tkzLabelPoints[above left](A)
\tkzLabelPoints[below right](I,E)
\end{tikzpicture}
\end{center}
\begin{itemize}
\item Nếu $\widehat{BAC}=90^\circ$ thì $\widehat{BIC}=90^\circ$, khi đó $EF$ là đường kính nên $EF$ đi qua $O.$
\item Xét $\widehat{BAC}<90^\circ$, khi đó $\widehat{BIC}>90^\circ$. Gọi $K$ là điểm đối xứng với $I$ qua $EF$, ta có $\widehat{EIF}=\widehat{EAF}$ (cùng bù với $\widehat{BIC}$), $\widehat{EKF}=\widehat{EIF}$, suy ra $\widehat{EKF}=\widehat{EAF}$ nên tứ giác $AEKF$ nội tiếp, hay $K \in (O)$. Mà $EF$ là đường trung trực đoạn $IK$ nên $EF$ đi qua $O$.
\item $\widehat{BAC}>90^\circ$ thì $\widehat{BIC}<90^\circ$. Chứng minh tương tự ta cũng có $EF$ đi qua $O.$
\end{itemize}
Vậy $EF$ luôn đi qua điểm cố định $O.$
    }
\end{ex}
