\begin{ex}%[HK1, Sở Nam Định, 2018]%[2D1K2]

	{$m = -1$}
	{$m \ne 0$}
	{$m = 2$}
	{\True $m = 1$}
	\loigiai{
		$y = x^4 - 2mx^2 + 2m - 3$. TXĐ:  $\mathscr{D}= \mathbb{R}$. \\
		$y' = 4x^3 - 4mx$. \\
		$y' = 0 \Leftrightarrow \hoac{&x = 0\\&x^2}$. \\

		Dễ thấy: tam giác $ABC$ cân tại $A$. \\
		Yêu cầu bài toán  $\Leftrightarrow AB \perp AC$  $ \Leftrightarrow \overrightarrow{AB} \cdot \overrightarrow{AC} =0 \Leftrightarrow -m + m^4 = 0$ \\ $ \Leftrightarrow \hoac{&m = 0\\&m = 1}$. \\
		So với ĐK  $\left( * \right)$  suy ra:  $m = 1$  thoả mãn yêu cầu bài toán.}
\end{ex}
