\begin{bt}%[Tuyển sinh 10 Đà Nẵng 2019]%[Mai Sương]%[9D1B7]
Trục căn thức ở mẫu của biểu thức $A=\dfrac{1}{2-\sqrt{3}}.$
\loigiai{
$$A=\dfrac{1}{2-\sqrt{3}}=\dfrac{2+\sqrt{3}}{4-3}=2+\sqrt{3}.$$
}
\end{bt}

\begin{bt}%[Tuyển sinh 10 Đà Nẵng 2019]%[Mai Sương]%[9D1B8]
Cho $a \ge 0, a\ne 4.$ Chứng minh $\dfrac{\sqrt{a}}{\sqrt{a}+2}+\dfrac{2(\sqrt{a}-2)}{a-4}=1.$
\loigiai{
$$\dfrac{\sqrt{a}}{\sqrt{a}+2}+\dfrac{2(\sqrt{a}-2)}{a-4}
=\dfrac{\sqrt{a}(\sqrt{a}-2)+2(\sqrt{a}-2)}{a-4}
=\dfrac{a-2\sqrt{a}+2\sqrt{a}-4}{a-4}
=\dfrac{a-4}{a-4}=1.$$
}
\end{bt}

\begin{bt}%[Tuyển sinh 10 Đà Nẵng 2019]%[Mai Sương]%[9D3B4]
Giải hệ phương trình $\begin{cases} x+2y=14 \\ 2x+3y=24. \end{cases}$
\loigiai{
$\begin{cases} x+2y=14 \\ 2x+3y=24. \end{cases}
\Leftrightarrow \begin{cases} 2x+4y=28 \\ 2x+3y=24. \end{cases}
\Leftrightarrow \begin{cases} y=4 \\ 2x+3y=24. \end{cases}
\Leftrightarrow \begin{cases} y=4 \\ 2x+3\cdot 4=24. \end{cases}
\Leftrightarrow \begin{cases} y=4 \\ x=6. \end{cases}$\\
Vậy hệ phương trình có nghiệm $(x;y)=(6;4).$
}
\end{bt}

\begin{bt}%[Tuyển sinh 10 Đà Nẵng 2019]%[Mai Sương]%[9D4B7]
Giải phương trình $4x+\dfrac{3}{x-1}=11.$
\loigiai{
Điều kiện: $x \ne 1.$\\
$4x+\dfrac{3}{x-1}=11
\Leftrightarrow \dfrac{4x(x-1)+3}{x-1}=\dfrac{11(x-1)}{x-1}
\Leftrightarrow 4x^2-4x+3=11x-11
\Leftrightarrow 4x^2-15x+14=0.$\\
$\Delta = (-15)^2-4\cdot 4 \cdot 14 =1>0$.\\ Suy ra
\begin{center}
$x_1=\dfrac{15+1}{2 \cdot 4}= 2 \quad $ (chọn); 
$\qquad x_2=\dfrac{15-1}{2 \cdot 4}=\dfrac{7}{4} \quad $ (chọn).
\end{center}
Vậy phương trình đã cho có tập nghiệm $S=\left\{2;\dfrac{7}{4}\right\}.$
}
\end{bt}

\begin{bt}%[Tuyển sinh 10 Đà Nẵng 2019]%[Mai Sương]%[9D4G2]
Vẽ đồ thị của các hàm số $y=-\dfrac{1}{2}x^2$ và $y=x-4$ trên cùng một mặt phẳng tọa độ. Gọi $A$ và $B$ là các giao điểm của đồ thị hai hàm số trên. Tính bán kính của đường tròn ngoại tiếp tam giác $OAB$, với $O$ là gốc tọa độ (đơn vị đo trên các trục tọa độ là cen-ti-mét).
\loigiai{
\begin{itemize}
\item $y=-\dfrac{1}{2}x^2$
\begin{center}
\begin{tabular}{|c|c|c|c|c|c|}
\hline 
$x$ & $-4$ & $-2$ & $0$ & $2$ & $4$ \\ 
\hline
$y=-\dfrac{1}{2}x^2$ & $-8$ & $-2$ & $0$ & $-2$ & $-8$ \\ 
\hline 
\end{tabular}
\end{center}
\item $y=x-4$
\begin{center}
\begin{tabular}{|c|c|c|}
\hline 
$x$ & $0$ & $4$ \\ 
\hline 
$y=x-4$ & $-4$ & $0$ \\ 
\hline 
\end{tabular} 
\end{center}
\end{itemize}
Đồ thị
\begin{center}
\begin{tikzpicture}[scale=0.5,every node/.style={scale=0.7},>=stealth,x=1cm,y=1cm]
\draw[->] (-6,0)--(0,0) node[above left]{$O$}--(6,0) node [below] {$x$};
\draw[->] (0,-10)--(0,4) node [left] {$y$};
\foreach \x in {-4,-2,2,4}
\draw (\x,0.05)--(\x,-0.05) node [above] {\x};
\foreach \y in {-8,-6,-4,-2,2}
\draw (0.05,\y)--(-0.05,\y) node [left] {\y};
\draw[black,domain=-4.47:4.47, samples=100]plot(\x,{-0.5*(\x)^2})node[right]{$y=-\dfrac{x^2}{2}$};
\draw[black,domain=-6:6, samples=100]plot(\x,{\x-4})node[left]{$y=x-4$};
\tkzDefPoints{2/-2/A, -4/-8/B, -4/0/C, 2/0/D, 4/0/E, 0/0/O}
\tkzDefMidPoint(O,B) \tkzGetPoint{i}
\tkzDrawPoints(A,B,C,D,E,i)
\tkzLabelPoints(A,B,C,D,E)
\draw (B)--(O)--(A) (A)--(D) (B)--(C);
\end{tikzpicture}
\end{center}
Phương trình hoành độ giao điểm của hai đồ thị đã cho 
$$-\dfrac{x^2}{2}=x-4 
\Leftrightarrow -\dfrac{x^2}{2}-x+4=0
\Leftrightarrow x^2+2x-8=0
\Leftrightarrow (x-2)(x+4)=0
\Leftrightarrow \left[\begin{aligned} & x=2 \\ & x=-4.\end{aligned}\right.$$ 	
\begin{itemize}
\item $x=2 \Rightarrow y=-2 \Rightarrow A(2;-2).$
\item $x=-4 \Rightarrow y=-8 \Rightarrow B(-4;-8).$
\end{itemize}
Gọi các điểm $C(-4;0),D(2;0),E(4;0).$\\
Xét tam giác $OAE$ có $AD=OD=DE=2$ cm nên tam giác $OAD$ vuông tại $A$. Suy ra $OA \perp BE.$ \\
Tam giác $OAB$ vuông tại $A$ nên đường tròn ngoại tiếp tam giác $OAB$ có tâm là trung điểm cạnh $OB$ và bán kính $R=\dfrac{OB}{2}.$\\
Xét tam giác $OCB$ vuông tại $C$ có 
$$OB=\sqrt{OC^2+BC^2}=\sqrt{4^2+8^2}=\sqrt{80}=4\sqrt{5}.$$
Vậy bán kính đường tròn ngoại tiếp tam giác $OAB$ là $R=\dfrac{OB}{2}=2\sqrt{5}.$
}
\end{bt}

\begin{bt}%[Tuyển sinh 10 Đà Nẵng 2019]%[Mai Sương]%[9D4G6]
Cho phương trình $x^2+2(m-1)x+4m-11=0,$ với $m$ là tham số. Tìm tất cả các giá trị của $m$ để phương trình có hai nghiệm phân biệt $x_1,x_2$ thỏa mãn hệ thức $2(x_1-1)^2+(6-x_2)(x_1x_2+11)=72.$
\loigiai{
Phương trình đã cho có hai nghiệm phân biệt $x_1,x_2$
\begin{eqnarray*}
&\Leftrightarrow &\Delta' > 0\\
&\Leftrightarrow & (m-1)^2-(4m-11) > 0\\
&\Leftrightarrow & m^2-6m+12 >0\\
&\Leftrightarrow & (m-3)^2+3 >0 \quad \text{(đúng với mọi } m \in \mathbb{R}).
\end{eqnarray*}
Do đó phương trình đã cho luôn có hai nghiệm phân biệt $x_1,x_2$ với mọi $m$.\\
Áp dụng hệ thức Vi-ét, ta có: 
$\begin{cases}
x_1+x_2=-2(m-1)\\
x_1 x_2=4m-11.
\end{cases}$\\
Vì $x_1,x_2$ là hai nghiệm của phương trình $x^2+2(m-1)x+4m-11=0$ nên ta có
$$\begin{cases}
2x_1^2+4(m-1)x_1+8m-22=0\\
x_2^2+2(m-1)x_2+4m-11=0
\end{cases}
\Leftrightarrow
\begin{cases}
2x_1^2=-4(m-1)x_1-8m+22\\
x_2^2=-2(m-1)x_2-4m+11.
\end{cases}$$
Ta có 

\begin{eqnarray*}
&&2(x_1-1)^2+(6-x_2)(x_1x_2+11)=72\\
&\Leftrightarrow & 2x_1^2-4x_1+2+6x_1x_2+66-x_1x_2^2-11x_2-72=0\\
&\Leftrightarrow & -4(m-1)x_1-8m+22-4x_1+2+6x_1x_2+66-x_1(-2(m-1)x_2-4m+11)-11x_2-72=0\\
&\Leftrightarrow & -4mx_1+4x_1-8m+22-4x_1+2+6x_1x_2+66+2(m-1)x_1x_2+4mx_1-11x_1-11x_2-72=0\\
&\Leftrightarrow &-11(x_1+x_2)+(2m+4)x_1x_2-8m+18=0\\
&\Leftrightarrow &22(m-1)+(2m+4)(4m-11)-8m+18=0\\
&\Leftrightarrow &22m-22 + 8m^2-6m -44-8m +18=0\\
&\Leftrightarrow & 8m^2+8m-48=0\\
&\Leftrightarrow & m^2+m-6=0\\
&\Leftrightarrow & m^2-2m+3m-6=0\\
&\Leftrightarrow & m(m-2)+3(m-2)=0\\
&\Leftrightarrow & (m+3)(m-2)=0\\
&\Leftrightarrow & \left[\begin{aligned} & m=-3 \\ & m=2.\end{aligned}\right.
\end{eqnarray*}
Vậy $m=-3$ hoặc $m=2.$
}
\end{bt}

\begin{bt}%[Tuyển sinh 10 Đà Nẵng 2019]%[Mai Sương]%[9D4K8]
Cạnh huyền của một tam giác vuông bằng $17$ cm. Hai cạnh góc vuông có độ dài hơn kém nhau $7$ cm. Tính diện tích của tam giác vuông đó.
\loigiai{
Gọi độ dài một cạnh góc vuông lớn hơn của tam giác vuông là $x$ (cm). $(7<x<17)$\\
Khi đó cạnh góc vuông còn lại của tam giác vuông đó là $x-7$ (cm).\\
Áp dụng định lý Py-ta-go cho tam giác vuông này, ta có
\begin{eqnarray*}
&& x^2+(x-7)^2=17^2\\
& \Leftrightarrow & 2x^2-14x+49 = 289\\
& \Leftrightarrow & 2x^2-14x-240=0\\
& \Leftrightarrow & 2(x-15)(x+8)=0\\
& \Leftrightarrow & \left[\begin{aligned}&x-15=0\\&x+8=0\end{aligned}\right.\\
& \Leftrightarrow & \left[\begin{aligned}&x=15 &\quad \text{(chọn)}\\&x=-8 &\quad \text{(loại)}.\end{aligned}\right.
\end{eqnarray*}
Vậy diện tích của tam giác vuông đó là
$$\dfrac{1}{2}x(x-7)=\dfrac{1}{2}\cdot 15\cdot (15-7)=\dfrac{1}{2} \cdot 15\cdot 8 = 60 \text{ (cm}^2).$$
}
\end{bt}

\begin{bt}%[Tuyển sinh 10 Đà Nẵng 2019]%[Mai Sương]%[9H3K7]
Cho tam giác nhọn $ABC$ nội tiếp đường tròn tâm $O$ có $AB<AC.$ Trên cung nhỏ $\wideparen{AC}$ lấy điểm $M$ khác $A$ thỏa $MA<MC$. Vẽ đường kính $MN$ của đường tròn $(O)$ và gọi $H,K$ lần lượt là hình chiếu vuông góc của $A$ trên $MB,MN.$ Chứng minh rằng
\begin{enumerate}[a)]
\item Bốn điểm $A,H,K,M$ cùng nằm trên một đường tròn.
\item $AH\cdot AK=HB \cdot MK.$
\item Khi điểm $M$ di động trên cung nhỏ $\wideparen{AC}$ thì đường thẳng $HK$ luôn đi qua một điểm cố định.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=0.6,every node/.style={scale=0.8}]
\tkzDefPoints{0/0/B}
\tkzDefPoints{3.5/6/A}
\tkzDefPoints{10/0/C}
\tkzCircumCenter(A,B,C)\tkzGetPoint{O}
\tkzDrawCircle(O,A)
\tkzDrawPolygon(A,B,C)
\tkzDefPointBy[rotation= center O angle - 50](A)
\tkzGetPoint{M}
\tkzDefPointBy[rotation= center O angle 180](M)
\tkzGetPoint{N}
\tkzDefPointBy[projection=onto M--B](A) \tkzGetPoint{H}
\tkzDefPointBy[projection=onto M--N](A) \tkzGetPoint{K}
\tkzDefMidPoint(A,B) \tkzGetPoint{E}
\tkzDrawSegments(A,M M,N M,B M,C A,H A,K K,E A,N)
\tkzMarkRightAngles(A,H,M A,K,M)
%\tkzLabelPoints(A,B,C,O,M,N,H,K,E)
\tkzDrawPoints(A,B,C,O,M,N,H,K,E)
\tkzLabelPoints[above](A)
\tkzLabelPoints[right](C)
\tkzLabelPoints[left](E,B)
\tkzLabelPoints[below right](K,O,H)
\tkzLabelPoints[above right](M)
\tkzLabelPoints[below left](N)
\end{tikzpicture}
\end{center}

\begin{enumerate}[a)]
\item  Xét tứ giác $AHKM$ có $\widehat{AHM} =\widehat{AKM} =90^{\circ}$ (gt).\\
Suy ra $H,K$ cùng nhìn cạnh $AM$ dưới một góc vuông.\\
Suy ra tứ giác $AHKM$ nội tiếp. Hay bốn điểm $A,H,K,M$ cùng nằm trên một đường tròn.
\item Xét đường tròn $(O)$ có $\widehat{ABM}= \widehat{ANM}\quad$ (hai góc nội tiếp cùng chắn cung $\wideparen{AM}$).$\quad (1)$\\
Ta có $\widehat{MAN}=90^{\circ}\quad$ (góc nội tiếp chắn nửa đường tròn).\\
Suy ra $\widehat{ANM}=\widehat{MAK} \quad$  (cùng phụ với $\widehat{AMK}$).$\quad (2)$\\
$(1),(2) \Rightarrow \widehat{ABM}=\widehat{MAK}$ hay $\widehat{ABH}=\widehat{MAK}$.\\
Xét $\triangle AHB$ và $\triangle MAK$ có
\begin{eqnarray*}
&\widehat{AHB}&=\widehat{MKA} \quad (=90^{\circ})\\
&\widehat{ABH}&=\widehat{MAK} \quad \text{(cmt)}
\end{eqnarray*}
Suy ra $\triangle AHB \backsim \triangle MAK \quad $ \text{(g-g)}\\
Suy ra $\dfrac{AH}{MK}=\dfrac{HB}{KA} \Leftrightarrow AH\cdot AK=HB \cdot MK.$
\item Gọi $E$ là giao điểm của $HK$ và $AB$.\\
Ta có 
\begin{eqnarray*}
&\widehat{MAK} &= \widehat{MHK} \quad \text{(hai góc nội tiếp cùng chắn cung } \wideparen{MK}).\\
&\widehat{MHK} &= \widehat{EHB}\quad \text{(hai góc đối đỉnh).}\\
&\widehat{ABH}&=\widehat{MAK} \quad \text{(cmt)}
\end{eqnarray*} 
Suy ra $\widehat{ABH} =\widehat{EBH}$ hay $\widehat{EHB} =\widehat{EBH}.$\\
Suy ra $\triangle EBH$ cân tại $E$.\\
Suy ra $EH=EB.$ $\quad (3)$\\
Mặt khác, ta có 
\begin{eqnarray*}
&\widehat{EAH} + \widehat{EBH} &= 90^{\circ}\\
&\widehat{EHA} + \widehat{EHB} &= 90^{\circ}\\
&\widehat{EBH}=\widehat{EHB} &
\end{eqnarray*} 
Suy ra $\widehat{EAH} =\widehat{EHA}$ \\
Suy ra $\triangle EAH$ cân tại $E$.\\
Suy ra $EH=EA.$ $\quad (4)$\\
$(3),(4) \Rightarrow EA=EB \Rightarrow E$ là trung điểm của $AB$. Mà $A,B$ cố định nên $E$ cố định.\\
Vậy khi điểm $M$ di động trên cung nhỏ $\wideparen{AC}$ thì đường thẳng $HK$ luôn đi qua trung điểm $E$ của đoạn thẳng $AB$.
\end{enumerate}
}
\end{bt}
