	Thực hiện phép tính
	\begin{enumerate}
		\item $(5x+3)\cdot (4x^2-6x+2)$.
		\item $(x^3-5x^2+8x-4)\div (x-2)$.
		\item $\dfrac{x}{x-2}-\dfrac{x}{x+2}+\dfrac{8}{x^2-4}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item {\allowdisplaybreaks
				\begin{eqnarray*}
					(5x+3)\cdot (4x^2-6x+2)&=&20x^3-30x^2+10x+12x^2-18x+6\\
					&=&20x^3-18x^2-8x+6.
			\end{eqnarray*}}
			\item {\allowdisplaybreaks
				\begin{eqnarray*}
					(x^3-5x^2+8x-4)\div (x-2)&=&(x^3-5x^2+8x-4)\cdot \dfrac{1}{x-2}\\
					&=&\dfrac{x^3-2x^2-3x^2+6x+2x-4}{x-2}\\
					&=&\dfrac{(x-2)(x^2-3x+2)}{x-2}\\
					&=&x^2-3x+2.
			\end{eqnarray*}}
			\item {\allowdisplaybreaks
				\begin{eqnarray*}
					\dfrac{x}{x-2}-\dfrac{x}{x+2}+\dfrac{8}{x^2-4}&=&\dfrac{x(x+2)-x(x-2)+8}{(x-2)(x+2)}\\
					&=&\dfrac{x^2+2x-x^2+2x+8}{(x-2)(x+2)}\\
					&=&\dfrac{4(x+2)}{(x-2)(x+2)}\\
					&=&\dfrac{4}{x-2}.
			\end{eqnarray*}}
		\end{enumerate}
	}
\end{bt}

	Phân tích các đa thức sau thành nhân tử
	\begin{enumEX}{3}
		\item $2x^3y-18xy$.
		\item $x^2+xy+5x+5y$.
		\item $4x^2-12x-y^2+9$.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item $2x^3y-18xy = 2xy(x^2-9) = 2xy(x-4)(x+3)$.
			\item $x^2+xy+5x+5y = x(x+y)+5(x+y) = (x+5)(x+y)$.
			\item $4x^2-12x-y^2+9 = (4x^2-12xy+9) - y^2 = (2x-3)^2-y^2 = (2x+y-3)(2x-3-y)$.
		\end{enumerate}
	}
\end{bt}

	Tìm $x$ biết $x^{2}+5x+6=0$ ?
	\loigiai{
		Ta có $x^{2}+5x+6= x^{2}+3x+2x+6= x(x+3)+2(x+3)=(x+2)(x+3)$. \\
		Do đó $x^{2}+5x+6=0 \Leftrightarrow (x+2)(x+3)=0 \Leftrightarrow \hoac{& x+2=0 \\ & x+3=0} \Leftrightarrow \hoac{& x=-2 \\ & x=-3.}$
	}
\end{bt}

	Cho ba số $a, b, c$ là ba số thỏa mãn $a+b+c=0$. Chứng minh $a^{3}+ b^{3}+ c^{3}= 3abc$ ?
	\loigiai{
		Từ giả thiết $a+b+c=0$, ta có: $c=-a-b$.\\
		Khi đó:
		\begin{eqnarray*}
			a^{3}+ b^{3}+ c^{3}& = & a^{3}+ b^{3} - (a+b)^{3}\\
			&=& a^{3}+ b^{3} - \left( a^{3} + b^{3} +3a^{2}b+3ab^{2} \right)\\
			& = & 3ab(-a-b)=3abc \text{ (đpcm).}
		\end{eqnarray*}
		
	}
\end{bt}

		\item Chứng minh $ ADFE $ là hình thoi.
	\end{enumerate}

				\item Do $ DE $ là đường trung bình của tam giác $ ABC $ nên $ DC \parallel BC $. Do đó $ BDEC $ là hình thang. Hơn nữa, do tam giác $ ABC $ cân tại $ A $ nên $ \widehat{ABC} = \widehat{ACB} $. Vậy hình thang $ BDEC $ có hai góc kề một đáy bằng nhau nên $ BDEC $ là hình thang cân.
				\item Do $ K $ là điểm đối xứng của $ F $ qua $ E $ nên $ E $ là trung điểm của $ FK $. Mà tam giác $ ABC $ cân nên $ AB=AC $ hay $ AD=AE $. Khi đó, tứ giác $ ADFE $ có $ AD=DF=FE=EA $ nên tứ giác $ ADEF $ là hình thoi.
