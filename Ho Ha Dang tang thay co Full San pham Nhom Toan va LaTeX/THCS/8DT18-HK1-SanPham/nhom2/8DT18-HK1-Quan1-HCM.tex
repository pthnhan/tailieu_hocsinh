			\item \begin{align*}
			& x(x-11)-x+11=0\\
			  &x(x-1)-(x-11)=0\\
			  &(x-11)(x-1) =0\\
			  \Rightarrow & \left[\begin{aligned} x-11 & =0\\ x-1 &=0. \end{aligned}\right.\\
			  \Rightarrow & \left[\begin{aligned} x & =11\\ x &=1. \end{aligned}\right.
			\end{align*}
			\item \begin{align*}
				 & 5x^3=10x\\
				 &5x^3-10x=0\\
				 &5x(x^2-2) =0\\
				\Rightarrow & \left[\begin{aligned} 5x & =0\\ x^2-2 &=0. \end{aligned}\right.\\
				\Rightarrow & \left[\begin{aligned} x & =0\\ x &=\sqrt 2\\&x=-\sqrt 2. \end{aligned}\right.
			\end{align*}
			\end{enumEX}}
