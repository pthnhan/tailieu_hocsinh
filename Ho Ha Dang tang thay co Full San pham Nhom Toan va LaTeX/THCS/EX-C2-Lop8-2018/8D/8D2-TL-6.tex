\section{Các phép tính về phân thức}
\subsection{Tóm tắt lí thuyết}
Muốn cộng các phân thức, ta quy đồng mẫu thức, cộng các tử thức với nhau, giữ nguyên mẫu thức chung, rồi rút gọn phân thức vừa tìm được.\\
Muốn trừ đi một phân thức, ta lấy phân thức bị trừ cộng với phân thức đối của phân thức trừ.\\
Muốn nhân các phân thức, ta nhân các tử thức với nhau, các mẫu thức với nhau, rồi rút gọn phân thức vừa tìm được.\\
Muốn chia cho một phân thức khác $0$, ta lấy phân thức bị chia nhân với phân thức nghịch đảo của phân thức chia.


\subsection{Các dạng toán}

\begin{vd}%[Tô Ngọc Thy, dự án (EX-C2-L8-2018)]%[8D2K5]
 Cho $a+b+c=0$ và $a$, $b$, $c$ đều khác $0$. Rút gọn biểu thức
 $$ A = \dfrac{ab}{a^2+b^2-c^2}+\dfrac{bc}{b^2+c^2-a^2}+\dfrac{ca}{c^2+a^2-b^2}.$$
 \loigiai
  {
  Từ $a+b+c=0$ suy ra $a+b=-c$.\\
  Bình phương hai vế, ta được $a^2+b^2+2ab=c^2$ nên $a^2+b^2-c^2=-2ab$.\\
  Tương tự, $b^2+c^2-a^2=-2bc$ và $c^2+a^2-b^2=-2ca$.\\
  Do đó, $A = \dfrac{ab}{-2ab} +\dfrac{bc}{-2bc}+ \dfrac{ca}{-2ca}= -\dfrac{1}{2}-\dfrac{1}{2} -\dfrac{1}{2}= -\dfrac{3}{2}$. 	
  }
\end{vd}

\begin{vd}%[Tô Ngọc Thy, dự án (EX-C2-L8-2018)]%[8D2K5]
 Rút gọn biểu thức
 $$A = \dfrac{1}{1-x}+\dfrac{1}{1+x}+\dfrac{2}{1+x^2}+\dfrac{4}{1+x^4}+\dfrac{8}{1+x^8}.$$
 \loigiai
  {
  Do đặc điểm của bài toán, ta không quy đồng mẫu tất cả các phân thức mà cộng lần lượt từng phân thức.\\
  $$\begin{array}{ll}
  A & = \dfrac{2}{1-x^2}+\dfrac{2}{1+x^2}+\dfrac{4}{1+x^4}+\dfrac{8}{1+x^8}\\
  & = \dfrac{4}{1-x^4}+\dfrac{4}{1+x^4}+\dfrac{8}{1+x^8} = \dfrac{8}{1-x^8}+\dfrac{8}{1+x^8} = \dfrac{16}{1-x^{16}}.
  \end{array}$$	
  }
\end{vd}

\begin{vd}%[Tô Ngọc Thy, dự án (EX-C2-L8-2018)]%[8D2K5]
 Rút gọn biểu thức
 $$B = \dfrac{3}{(1\cdot 2)^2}+\dfrac{5}{(2\cdot 3)^2}+\dots + \dfrac{2n+1}{\left[ n(n+1)\right]^2}.$$
 \loigiai
  {
	Đương nhiên không thể quy đồng mẫu tất cả các phân thức. Ta tìm cách tách mỗi phân thức thành hiệu của hai phân thức rồi dùng phương pháp khử liên tiếp. Ta có
	$$\dfrac{2k+1}{k^2(k+1)^2} = \dfrac{(k+1)^2-k^2}{k^2(k+1)^2} = \dfrac{1}{k^2} - \dfrac{1}{(k+1)^2}.$$
	Do đó
	$$\begin{array}{ll}
	B& = \dfrac{1}{1^2}-\dfrac{1}{2^2}+\dfrac{1}{2^2}-\dfrac{1}{3^2}+\dots+\dfrac{1}{n^2}-\dfrac{1}{(n+1)^2}\\
	& = 1-\dfrac{1}{(n+1)^2} = \dfrac{n(n+2)}{(n+1)^2}.
	\end{array}$$	
  }
\end{vd}

\begin{vd}%[Tô Ngọc Thy, dự án (EX-C2-L8-2018)]%[8D2G5]
 Xác định các số $a$, $b$, $c$ sao cho
 \[\dfrac{1}{(x^2+1)(x-1)}=\dfrac{ax+b}{x^2+1}+\dfrac{c}{x-1}.\tag{1}\]
 \loigiai
  {
	Thực hiện phép cộng ở vế phải của $(1)$ ta được
	$$\begin{array}{ll}
	\dfrac{(ax+b)(x-1)+c(x^2+1)}{(x^2+1)(x-1)}& = \dfrac{ax^2-ax+bx-b+cx^2+c}{(x^2+1)(x-1)}\\
	& = \dfrac{(a+c)x^2+(b-a)x+(c-b)}{(x^2+1)(x-1)}.
	\end{array}$$
	Đồng nhất phân thức trên với phân thức $\dfrac{1}{(x^2+1)(x-1)}$, ta được
	$$\heva{& a+c=0\\&b-a=0\\&c-b=1}\Rightarrow \heva{&c+b=0\\&c-b=1}\Rightarrow c=\dfrac{1}{2}, b=-\dfrac{1}{2}.$$
	Do đó $a=-\dfrac{1}{2}$. Như vậy $\dfrac{1}{(x^2+1)(x-1)}=\dfrac{-\dfrac{1}{2}x-\dfrac{1}{2}}{x^2+1}+\dfrac{\dfrac{1}{2}
	}{x-1}$.
  }
\end{vd}

\begin{vd}%[Tô Ngọc Thy, dự án (EX-C2-L8-2018)]%[8D2G7]
 Cho $ \begin{aligned}[t]
 A &= \dfrac{1}{(x+y)^3}\left( \dfrac{1}{x^4}-\dfrac{1}{y^4}\right),\\
 B &= \dfrac{2}{(x+y)^4}\left( \dfrac{1}{x^3}-\dfrac{1}{y^3}\right),\\
 C &= \dfrac{2}{(x+y)^5}\left( \dfrac{1}{x^2}-\dfrac{1}{y^2}\right).
\end{aligned}$\\
 Thực hiện phép tính $A+B+C$.
 \loigiai
  {
	Ta có
	$$\begin{array}{ll}
	A & = \dfrac{y^4-x^4}{x^4y^4(x+y)^3}=\dfrac{(y^2+x^2)(y^2-x^2)}{x^4y^4(x+y)^3} = \dfrac{(y^2+x^2)(y-x)}{x^4y^4(x+y)^2}.\\
	B+C& =  \dfrac{2}{(x+y)^4}\left( \dfrac{1}{x^3}-\dfrac{1}{y^3}+\dfrac{1}{x+y}\cdot \dfrac{y^2-x^2}{x^2y^2}\right)\\
	& = \dfrac{2}{(x+y)^4}\left( \dfrac{1}{x^3}-\dfrac{1}{y^3}+ \dfrac{y-x}{x^2y^2}\right) = \dfrac{2}{(x+y)^4}\cdot \dfrac{y^3-x^3+xy(y-x)}{x^3y^3}\\
	& = \dfrac{2}{(x+y)^4}\cdot \dfrac{(y-x)(y^2+2yx+x^2)}{x^3y^3} = \dfrac{2(y-x)}{(x+y)^2x^3y^3}.
	\end{array}$$
	Do đó, 
	$$\begin{array}{ll}
	A+B+C & = \dfrac{(y^2+x^2)(y-x)}{x^4y^4(x+y)^2} + \dfrac{2(y-x)}{(x+y)^2x^3y^3}\\
	& =  \dfrac{(y^2+x^2)(y-x)+2xy(y-x)}{x^4y^4(x+y)^2}\\
	& = \dfrac{(y-x)(y^2+x^2+2xy)}{x^4y^4(x+y)^2}\\
	& = \dfrac{y-x}{x^4y^4}.
	\end{array}$$
	}
\end{vd}


\subsection{Bài tập tự luện}
\begin{bt}%[Tô Ngọc Thy, dự án (EX-C2-L8-2018)]%[8D2K6]
 Thực hiện các phép tính
	\begin{enumerate}
		\item $\dfrac{x+3}{x+1}-\dfrac{2x-1}{x-1}-\dfrac{x-3}{x^2-1}$;
		\item $\dfrac{1}{x(x+y)}+\dfrac{1}{y(x+y)}+\dfrac{1}{x(x-y)}+\dfrac{1}{y(y-x)}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item
			\begin{tabular}[t]{ll}
				$\dfrac{x+3}{x+1}-\dfrac{2x-1}{x-1}-\dfrac{x-3}{x^2-1}$ & $= \dfrac{(x+3)(x-1)-(2x-1)(x+1)-(x-3)}{x^2-1}$\\
				& $= \dfrac{-x^2+1}{x^2-1}=-1$.	
			\end{tabular} 
			\item
			\begin{tabular}[t]{ll}
				$\dfrac{1}{x(x+y)}+\dfrac{1}{y(x+y)}+\dfrac{1}{x(x-y)}+\dfrac{1}{y(y-x)}$& $= \dfrac{x+y}{xy(x+y)}+\dfrac{y-x}{xy(x-y)}$\\
				& = $\dfrac{1}{xy}-\dfrac{1}{xy}=0$.
			\end{tabular}
		\end{enumerate}
		
	}
\end{bt}

\begin{bt}%[Tô Ngọc Thy, dự án (EX-C2-L8-2018)]%[8D2G5]
	Thực hiện phép tính
	\begin{enumerate}
		\item $A = \dfrac{1}{(a-b)(a-c)}+\dfrac{1}{(b-a)(b-c)}+\dfrac{1}{(c-a)(c-b)}$;
		\item $B = \dfrac{1}{a(a-b)(a-c)}+\dfrac{1}{b(b-a)(b-c)}+\dfrac{1}{c(c-a)(c-b)}$;
		\item $C = \dfrac{bc}{(a-b)(a-c)}+\dfrac{ac}{(b-a)(b-c)}+\dfrac{ab}{(c-a)(c-b)}$;
		\item $D = \dfrac{a^2}{(a-b)(a-c)}+\dfrac{b^2}{(b-a)(b-c)}+\dfrac{c^2}{(c-a)(c-b)}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $A = \dfrac{1}{(a-b)(a-c)}+\dfrac{1}{(b-a)(b-c)}+\dfrac{1}{(c-a)(c-b)} = \dfrac{c-b+a-c+b-a}{(a-b)(b-c)(c-a)}=0$.
			\item Ta có
			\allowdisplaybreaks
			\begin{eqnarray*}
				B &=& \dfrac{1}{a(a-b)(a-c)}+\dfrac{1}{b(b-a)(b-c)}+\dfrac{1}{c(c-a)(c-b)} \\
				&=& \dfrac{bc(b-c)-ac(a-c)+ab(a-b)}{abc(a-b)(b-c)(a-c)}\\
				&=& \dfrac{c(b^2-bc-a^2+ac)+ab(a-b)}{abc(a-b)(b-c)(a-c)}\\
				&=& \dfrac{c\left[ (b-a)(b+a)-c(b-a)\right] +ab(a-b)}{abc(a-b)(b-c)(a-c)}\\
				&=& \dfrac{(a-b)(-cb-ca+c^2)+ab(a-b)}{abc(a-b)(b-c)(a-c)}\\
				&=& \dfrac{(a-b)(-cb-ac+c^2+ab)}{abc(a-b)(b-c)(a-c)}\\
				&=& \dfrac{(a-b)(b-c)(a-c)}{abc(a-b)(b-c)(a-c)} = \dfrac{1}{abc}.
			\end{eqnarray*}
			\item Ta có
			\allowdisplaybreaks
			\begin{eqnarray*}
				C &=&  \dfrac{bc}{(a-b)(a-c)}+\dfrac{ac}{(b-a)(b-c)}+\dfrac{ab}{(c-a)(c-b)}\\
				&=& \dfrac{bc(b-c)-ac(a-c)+ab(a-b)}{abc(a-b)(b-c)(a-c)}\\
				&=& \dfrac{c(b^2-bc-a^2+ac)+ab(a-b)}{(a-b)(b-c)(a-c)}\\
				&=& \dfrac{c\left[ (b-a)(b+a)-c(b-a)\right] +ab(a-b)}{(a-b)(b-c)(a-c)}\\
				&=& \dfrac{(a-b)(-cb-ca+c^2)+ab(a-b)}{(a-b)(b-c)(a-c)}\\
				&=& \dfrac{(a-b)(-cb-ac+c^2+ab)}{(a-b)(b-c)(a-c)}\\
				&=& \dfrac{(a-b)(b-c)(a-c)}{(a-b)(b-c)(a-c)} = 1.
			\end{eqnarray*}
			\item Ta có
			\allowdisplaybreaks
			\begin{eqnarray*}
				D &=& \dfrac{a^2}{(a-b)(a-c)}+\dfrac{b^2}{(b-a)(b-c)}+\dfrac{c^2}{(c-a)(c-b)}\\
				&=& \dfrac{a^2(b-c)-b^2(a-c)+c^2(a-b)}{(a-b)(b-c)(a-c)}\\
				&=& \dfrac{a^2(b-c)-b^2a+b^2c+c^2a-c^2b}{(a-b)(b-c)(a-c)}\\
				&=& \dfrac{a^2(b-c)-a(b^2-c^2)+bc(b-c)}{(a-b)(b-c)(a-c)}\\
				&=& \dfrac{(b-c)(a^2-ab-ac+cb)}{(a-b)(b-c)(a-c)}\\
				&=& \dfrac{(b-c)\left[ a(a-b)-c(a-b)\right]}{(a-b)(b-c)(a-c)}\\
				&=& \dfrac{(b-c)(a-b)(a-c)}{(a-b)(b-c)(a-c)} = 1.
			\end{eqnarray*}
		\end{enumerate}
		
	}
\end{bt}

\begin{bt}%[Tô Ngọc Thy, dự án (EX-C2-L8-2018)]%[8D2G5]
	Cho $a$, $b$, $c$ là các số nguyên khác nhau đôi một. Chứng minh rằng biểu thức sau có giá trị là một số nguyên:
	$$P = \dfrac{a^3}{(a-b)(a-c)}+\dfrac{b^3}{(b-a)(b-c)}+\dfrac{c^3}{(c-a)(c-b)}.$$
	\loigiai{
	Ta có
		\begin{eqnarray*}
			P &=& \dfrac{a^3}{(a-b)(a-c)}+\dfrac{b^3}{(b-a)(b-c)}+\dfrac{c^3}{(c-a)(c-b)}\\
			&=& \dfrac{a^3(b-c)+b^3(c-a)+c^3(a-b)}{(a-b)(b-c)(a-c)}.
		\end{eqnarray*}
		Phân tích tử thành nhân tử
		\allowdisplaybreaks
		\begin{eqnarray*}
		a^3(b-c)+b^3(c-a)+c^3(a-b) &=& a^3b-a^3c+b^3c-b^3a+c^3(a-b) \\
			&=& (a^3b-b^3a)-(a^3c-b^3c)+c^3(a-b)\\
			&=& ab(a-b)(a+b)-c(a-b)(a^2+ab+b^2)+c^3(a-b)\\
			&=& (a-b)(a^2b+ab^2-ca^2-cb^2-abc+c^3)\\
			&=& (a-b)\left[ (c^3-cb^2)-(abc-ab^2)+(a^2b-ca^2)\right]\\
			&=& (a-b)(b-c)(-cb-c^2+ab+a^2)\\
			&=& (a-b)(b-c)\left[ (ab-cb)+(a^2-c^2)\right]\\
			&=& (a-b)(b-c)(a-c)(a+b+c).
		\end{eqnarray*}
		Vậy $P = a+b+c$.
	}
\end{bt}

\begin{bt}%[Tô Ngọc Thy, dự án (EX-C2-L8-2018)]%[8D2K5]

\begin{bt}%[Tô Ngọc Thy, dự án (EX-C2-L8-2018)]%[8D2G5]
	Tìm $x$, $y$, $z$ biết rằng $\dfrac{x^2}{2}+\dfrac{y^2}{3}+\dfrac{z^2}{4}=\dfrac{x^2+y^2+z^2}{5}$.
	\loigiai{
		Từ $\dfrac{x^2}{2}+\dfrac{y^2}{3}+\dfrac{z^2}{4}=\dfrac{x^2+y^2+z^2}{5}$ suy ra

		Cho nên $\dfrac{3x^2}{10}+\dfrac{2y^2}{15}+\dfrac{z^2}{20}=0$. Do đó, $x=y=z=0$.
	}
\end{bt}

\begin{bt}%[Tô Ngọc Thy, dự án (EX-C2-L8-2018)]%[8D2G6]
	Tìm $x$, $y$ biết rằng $x^2+y^2+\dfrac{1}{x^2}+\dfrac{1}{y^2}=4$.
	\loigiai{
		Ta có	
		\begin{eqnarray*}
			x^2+y^2+\dfrac{1}{x^2}+\dfrac{1}{y^2}=4 & \Rightarrow & \left( x^2-2+\dfrac{1}{x^2}\right) + \left( y^2-2+\dfrac{1}{y^2}\right) = 0\\
			&\Rightarrow & \left( x - \dfrac{1}{x}\right)^2 + \left( y - \dfrac{1}{y}\right)^2 = 0\\
			& \Rightarrow & \heva{& x - \dfrac{1}{x} = 0\\ &  y - \dfrac{1}{y} = 0} \Rightarrow \heva{& x^2 = 1\\ & y^2 = 1.}
		\end{eqnarray*}\\
		Có bốn đáp án như bảng sau
		\begin{center}
			\begin{tabular}{|c|c|c|c|c|c|c|}
				\hline
				$x$ & $1$ & $1$ & $-1$ & $-1$  \\ \hline
				$y$ & $1$ & $-1$ & $1$ & $-1$ \\ \hline
				
			\end{tabular}
		\end{center}
		
	}
\end{bt}

\begin{bt}%[Tô Ngọc Thy, dự án (EX-C2-L8-2018)]%[8D2G9]
	Cho $\left( a+b+c\right)^2=a^2+b^2+c^2$ và $a$, $b$, $c$ khác $0$. Chứng minh rằng
	$$\dfrac{1}{a^3}+\dfrac{1}{b^3}+\dfrac{1}{c^3}=\dfrac{3}{abc}.$$
	\loigiai{
		Từ giả thiết suy ra $ab+bc+ca=0 \Rightarrow b+c=-\dfrac{bc}{a}$.\\
		Do đó $\dfrac{ab+bc+ca}{abc}=0$, tức là $\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}=0$. Suy ra
		\begin{eqnarray*}
			 \dfrac{1}{a}=-\left( \dfrac{1}{b}+\dfrac{1}{c}\right) & \Rightarrow & \dfrac{1}{a^3}=-\left( \dfrac{1}{b}+\dfrac{1}{c}\right)^3=-\dfrac{1}{b^3}-\dfrac{1}{c^3}-3\left( \dfrac{1}{b^2c}+\dfrac{1}{bc^2}\right)\\
			&\Rightarrow & \dfrac{1}{a^3}+\dfrac{1}{b^3}+\dfrac{1}{c^3}=-\dfrac{3(b+c)}{b^2c^2}=\dfrac{3bc}{ab^2c^2}=\dfrac{3}{abc}.	\end{eqnarray*}
	}
\end{bt}

\begin{bt}%[Tô Ngọc Thy, dự án (EX-C2-L8-2018)]%[8D2G5]
	Cho $\dfrac{a}{b}+\dfrac{b}{c}+\dfrac{c}{a}= \dfrac{b}{a}+\dfrac{a}{c}+\dfrac{c}{b}$. Chứng minh rằng trong ba số $a$,$b$,$c$ tồn tại hai số bằng nhau.
	\loigiai{
		Từ giả thiết suy ra
		\begin{eqnarray*}
			a^2c+b^2a+c^2b=b^2c+a^2b+c^2a &\Rightarrow & a^2(c-b)-a(c^2-b^2)+bc(c-b)=0\\
			&\Rightarrow & (c-b)(a^2-ac-ab+bc)=0\\
			&\Rightarrow & (c-b)(a-c)(a-b)=0.
		\end{eqnarray*}
		Tồn tại một trong các thừa số $c-b, a-c, a-b$ bằng $0$. Do đó, trong ba số $a, b, c$ tồn tại hai số bằng nhau.
	}
\end{bt}

\begin{bt}%[Tô Ngọc Thy, dự án (EX-C2-L8-2018)]%[8D1K12]
	Tìm các giá trị nguyên của $x$ để phân thức sau có giá trị là số nguyên:
	\begin{enumerate}
		\item $A = \dfrac{2x^3-6x^2+x-8}{x-3}$;
		\item $B = \dfrac{x^4-2x^3-3x^2+8x-1}{x^2-2x+1}$;
		\item $C = \dfrac{x^4+3x^3+2x^2+6x-2}{x^2+2}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $A = \dfrac{2x^3-6x^2+x-8}{x-3} = 2x^2+1-\dfrac{5}{x-3}$.\\
			$A$ nguyên khi $x$ nguyên, $x-3$ nguyên và nó là ước của $5$.\\
			Suy ra $x-3=1$ hoặc $x-3=-1$ hoặc $x-3=5$ hoặc $x-3=-5$.\\
			Hay $x=4$ hoặc $x=2$ hoặc $x=8$ hoặc $x=-2$.
			\item $B = \dfrac{x^4-2x^3-3x^2+8x-1}{x^2-2x+1} = x^2-4+\dfrac{3}{(x-1)^2}$.\\
			$B$ nguyên khi $x$ nguyên, $(x-1)^2$ nguyên và nó là ước của $3$.\\
			Suy ra $(x-1)^2=1$ hoặc $(x-1)^2=3$.\\
			Hay $x-1=-1$ hoặc $x-1=1$

  Giả sử mức sản xuất của xí nghiệp năm $2000$ là $1$ thì mức sản xuất năm $2001$ là $1+\dfrac{a}{100}$, mức sản xuất năm $2002$ là
  \[\left(1+\dfrac{a}{100}\right)\left(1+\dfrac{b}{100}\right) = 1+\dfrac{b}{100}+\dfrac{a}{100}+\dfrac{ab}{10000}, \]
  tăng so với năm $2000$ là $\dfrac{a+b}{100}+\dfrac{ab}{10000}$ hay $\left(a+b+\dfrac{ab}{100}\right)\%$.
  }
\end{bt}

\begin{bt}%[Cao Thành Thái, dự án EX-C2-L8-2018]%[8D2K9]
 Một số $a$ tăng $m \%$, sau đó lại giảm đi $n \%$ ($a$, $m$, $n$ là các số dương) thì được số $b$. Tìm liên hệ giữa $m$ và $n$ để $b>a$.
 \loigiai
  {
  Ta có $b=a\left(1+\dfrac{m}{100}\right)\left(1-\dfrac{n}{100}\right)$ nên $b-a = a\left[\left(1+\dfrac{m}{100}\right)\left(1-\dfrac{n}{100}\right)-1\right]$.\\
  Điều kiện để $b>a$ là $\left(1+\dfrac{m}{100}\right)\left(1-\dfrac{n}{100}\right) >1$.\\
  Rút gọn điều kiện trên ta được $100(m-n)>mn$.
  }
\end{bt}

\begin{bt}%[Cao Thành Thái, dự án EX-C2-L8-2018]%[8D2G5]
 Chứng minh tổng sau không là số nguyên với mọi số tự nhiên $n \geq 2$
 \begin{eqnarray*}
  A= \dfrac{1}{2} + \dfrac{1}{3} + \dfrac{1}{4} + \cdots + \dfrac{1}{n}.
 \end{eqnarray*}
 \loigiai
  {
  Gọi $k$ là số nguyên lớn nhất sao cho $2^k \leq n$. Chọn mẫu chung là $2^kP$ trong đó $P$ là tích các số lẻ không vượt quá $n$. Chỉ có duy nhất một thừa số phụ $\Big($của phân số $\dfrac{1}{2^k} \Big)$ là số lẻ, còn mọi thừa số phụ khác đều chẵn.\\
  Như vậy, sau khi quy đồng mẫu, mẫu là số chẵn, tử là số lẻ. Do đó $A$ không là số nguyên.
  }
\end{bt}

\begin{bt}%[Cao Thành Thái, dự án EX-C2-L8-2018]%[8D2G5]
 Chứng minh tổng sau không là số nguyên với mọi số tự nhiên $n \geq 1$
 \begin{eqnarray*}
  B = \dfrac{1}{3} + \dfrac{1}{5} + \dfrac{1}{7} + \cdots + \dfrac{1}{2n+1}.
 \end{eqnarray*}
 \loigiai
  {
  Gọi $k$ là số nguyên lớn nhất sao cho $3^k \leq 2n+1$. Chọn mẫu chung là $3^kP$ trong đó $P$ là tích các thừa số nguyên tố lẻ không vượt quá $2n+1$. Chỉ có duy nhất một thừa số phụ $\Big($của phân số $\dfrac{1}{3^k} \Big)$ không chia hết cho $3$, còn mọi thừa số phụ khác đều chia hết cho $3$.\\
  Như vậy, sau khi quy đồng mẫu, mẫu là số chia hết cho $3$, tử là số không chia hết cho $3$. Do đó $B$ không là số nguyên.
  }
\end{bt}
