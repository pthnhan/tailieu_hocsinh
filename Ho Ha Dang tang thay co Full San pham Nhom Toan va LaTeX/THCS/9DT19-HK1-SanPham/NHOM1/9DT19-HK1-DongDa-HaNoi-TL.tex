		\sqrt{9x-9}-1 = \sqrt{x-1}&\Leftrightarrow& 3\sqrt{x-1}-1 = \sqrt{x-1}\\
		&\Leftrightarrow& 2\sqrt{x-1} =1\\&\Leftrightarrow& 4(x-1) =1  \\		&\Leftrightarrow& x=\dfrac{5}{4}. 
		\end{eqnarray*}

	\end{enumerate}

	\begin{eqnarray*}
	A=\dfrac{2\sqrt{x}-1}{\sqrt{x}-3} \quad \text{và}\quad B=\dfrac{2x+3\sqrt{x}+9}{x-9}-\dfrac{\sqrt{x}}{\sqrt{x}+3}\quad \text{với}\quad x \geq 0, x\neq 9.
	\end{eqnarray*}
	\begin{enumerate}
		\item Tính giá trị của $A$ khi $x=25$.
		\item Rút gọn biểu thức $B$.
		\item Cho $P=\dfrac{A}{B}$, tìm giá trị nhỏ nhất của $P$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Thay $x=25$ vào biểu thức $A$ ta có
			\begin{eqnarray*}
		A=\dfrac{2\sqrt{25}-1}{\sqrt{25}-3}=\dfrac{10-1}{5-3}=\dfrac{9}{2}.
			\end{eqnarray*}  
			\item Rút gọn biểu thức $B$. Ta có
			\begin{eqnarray*}
		B&=&\dfrac{2x+3\sqrt{x}+9}{x-9}-\dfrac{\sqrt{x}}{\sqrt{x}+3}=\dfrac{2x+3\sqrt{x}+9-\sqrt{x}\left(\sqrt{x}-3\right)}{\left(\sqrt{x}-3\right) \left(\sqrt{x}+3\right)}\\ &=&\dfrac{x+6\sqrt{x}+9}{\left(\sqrt{x}-3\right) \left(\sqrt{x}+3\right)}=\dfrac{\left(\sqrt{x}+3\right)^{2}}{\left(\sqrt{x}-3\right) \left(\sqrt{x}+3\right)}=\dfrac{\sqrt{x}+3}{\sqrt{x}-3}.
			\end{eqnarray*}
			Vậy $B=\dfrac{\sqrt{x}+3}{\sqrt{x}-3}$.
			\item $\bullet$ Ta có $A=\dfrac{2\sqrt{x}-1}{\sqrt{x}-3}$ và $B=\dfrac{\sqrt{x}+3}{\sqrt{x}-3}$. Khi đó 
			\begin{eqnarray*}
			P&=&\dfrac{A}{B}=\dfrac{2\sqrt{x}-1}{\sqrt{x}-3}\cdot  \dfrac{\sqrt{x}-3}{\sqrt{x}+3},\quad \text{với}\quad x \geq 0, x\neq 9\\&=&\dfrac{2\sqrt{x}-1}{\sqrt{x}+3}=2-\dfrac{7}{\sqrt{x}+3} .
			\end{eqnarray*}
			$\bullet$ Ta có 
			\begin{eqnarray*}
			x \geq 0 \Leftrightarrow \sqrt{x}+3\geq3 \Leftrightarrow - \dfrac{7}{\sqrt{x}+3} \geq - \dfrac{7}{3}\Leftrightarrow P=2-\dfrac{7}{\sqrt{x}+3}\geq -\dfrac{1}{3}.
			\end{eqnarray*}
		$\bullet$ Với $x=0$ thì giá trị nhỏ nhất của $P$ là $-\dfrac{1}{3}$.
		\end{enumerate}	
		
	}

	\loigiai{$\bullet$ Áp dụng bất đẳng Côsi cho hai số dương $x^{2}$, $y^{2}$ nên ta có
		\begin{eqnarray*}
	x^{2} +y^{2}\geq 2xy.\qquad (1)
		\end{eqnarray*}  
	 $\bullet$ Áp dụng bất đẳng Côsi cho hai số dương $2x^{2}$, $\dfrac{z^{2}}{2}$ nên ta có
	 \begin{eqnarray*}
	 2x^{2}+\dfrac{z^{2}}{2}\geq 2xz.\qquad(2)
	 \end{eqnarray*}
	  $\bullet$ Áp dụng bất đẳng Côsi cho hai số dương $2y^{2}$, $\dfrac{z^{2}}{2}$ nên ta có
	  \begin{eqnarray*}
	  2x^{2}+\dfrac{z^{2}}{2}\geq 2yz.\qquad(3)
	  \end{eqnarray*} 
  $\bullet$ Từ $(1), (2), (3)$ suy ra
  \begin{eqnarray*}
  T=3x^{2}+3y^{2}+z^{2}\geq 2\left(xy+yz+xz\right) \Rightarrow T\geq 10.
  \end{eqnarray*}
$\bullet$Dấu bằng xảy ra và các số $x, y, z$ dương khi và chỉ khi
\begin{eqnarray*}
\heva{&x=y&\\&z=2x&} \text{ và thế vào phương trình:} \quad xy+yz+zx=5 \Rightarrow \heva{&x=y=1&\\&z=2.&}
\end{eqnarray*}
$\bullet$ Vậy giá trị nhỏ nhất của $T$ bằng $10$ khi $\heva{&x=y=1&\\&z=2.&}$
