\begin{ex}%[KSCL, Sở GD \& ĐT Hà Nam, 2018]%[Lê Quốc Hiệp, dự án 12EX10-18]%[1D2B5-2]

\begin{ex}%[KSCL, Sở GD \& ĐT Hà Nam, 2018]%[Lê Quốc Hiệp, dự án 12EX10-18]%[2D1G3-1]

			\[|m+14|\le30\Leftrightarrow-44\le m\le16.\qquad(4)\]

		\end{itemize}
		Vậy $m\in[0;16]$ và $m$ nguyên nên $m\in\{0;1;2;3;\ldots;15;16\}$.\\
		Khi đó $0+1+2+\cdots+15+16=136$.
