\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2016-2017, Ninh Bình]%[8D1Y1]%[8D1Y2]%[8D2Y5]
	Thực hiện phép tính:
	\begin{enumerate}
		\item $x(x+5)+(x+1)^2$
		\item $(2x^2+4x-16):(x-2)$
		\item $\dfrac{6x}{x+3}+\dfrac{18}{x+3}$
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
			\item $x(x+5)+(x+1)^2=x^2+5x+x^2+2x+1=2x^2+7x+1$.
			\item $(2x^2+4x-16):(x-2)=2(x^2+2x-8):(x-2)=2[(x+1)^2-3^2]:(x-2)=2(x+1+3)(x+1-3):(x-2)=2(x+4)(x-2):(x-2)=2(x+4)=2x+8$.
			\item $\dfrac{6x}{x+3}+\dfrac{18}{x+3}=\dfrac{6x+18}{x+3}=\dfrac{6(x+3)}{x+3}=6$.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2016-2017, Ninh Bình]%[8D2B1]%[8D2B5]%[8D2B9]
	Cho biểu thức $M=\dfrac{x}{2x-2}+\dfrac{x^2+1}{2-2x^2}$.
	\begin{enumerate}
		\item Tìm $x$ để biểu thức $M$ có nghĩa.
		\item Rút gọn biểu thức $M$.
		\item Tìm giá trị của $x$ để giá trị của biểu thức $M=\dfrac{1}{2}$.
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
			\item $M$ có nghĩa $\Leftrightarrow \begin{cases} 2x-2 \neq 0 \\ 2-2x^2 \neq 0 \end{cases} \Leftrightarrow \begin{cases} x \neq 1 \\ x^2 \neq 1 \end{cases} \Leftrightarrow \begin{cases} x \neq 1 \\ x \neq -1 \end{cases}$.
			\item Ta có $M=\dfrac{x}{2x-2}+\dfrac{x^2+1}{2-2x^2}=\dfrac{x}{2(x-1)}-\dfrac{x^2+1}{2(x-1)(x+1)}=\dfrac{x(x+1)-(x^2+1)}{2(x-1)(x+1)} \\ =\dfrac{x-1}{2(x-1)(x+1)}=\dfrac{1}{2(x+1)}$
			\item $M=\dfrac{1}{2} \Leftrightarrow \dfrac{1}{2(x+1)}=\dfrac{1}{2} \Leftrightarrow x+1=1 \Leftrightarrow x=0$ (thỏa mãn).
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2016-2017, Ninh Bình]%[8D1B6]
	Tìm $x$, biết:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $2x(x-2016)-x+2016=0$.
			\item $(x+2)^2-(x-2)(x+2)=0$.
		\end{enumerate}
	\end{multicols}
	\loigiai
	{
		Ta có:
		\begin{enumerate}
			\item $2x(x-2016)-x+2016=0 \Leftrightarrow 2x(x-2016)-(x-2016)=0 \Leftrightarrow (x-2016)(2x-1)=0 \\
			\Leftrightarrow \left[\begin{aligned}
				& x-2016=0 \\ & 2x-1=0
			\end{aligned}\right. \Leftrightarrow \left[\begin{aligned}
				& x=2016 \\ & x=\frac{1}{2}
			\end{aligned}\right.$.
			\item $(x+2)^2-(x-2)(x+2)=0 \Leftrightarrow (x+2)(x+2-x+2)=0 \Leftrightarrow 4(x+2)=0 \Leftrightarrow x+2=0 \Leftrightarrow x=-2$.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2016-2017, Ninh Bình]%[8H1B7]%[8H1K7]%[8H1G12]
	Cho hình bình hành $ABCD$. Trên các cạnh $AB$ và $CD$ lần lượt lấy các điểm $E$ và $F$ sao cho $AE=CF$. Trên các cạnh $AD$ và $BC$ lần lượt lấy các điểm $M$ và $N$ sao cho $AM=CN$.
	\begin{enumerate}
		\item Các tứ giác $AECF$, $MENF$ là những hình gì?
		\item Chứng minh rằng các đường thẳng $AC,BD,EF$ và $MN$ đồng quy.
		\item Nếu $ABCD$ là hình vuông, $AE=CF=\dfrac{AB}{2}$ và $AM=CN=\dfrac{AD}{2}$ thì tứ giác $MENF$ là hình gì?
	\end{enumerate}
	\loigiai
	{
		\begin{center}
			\begin{tikzpicture}[scale=1]
				\tkzDefPoints{-3/3/A,3/3/B,-4/0/D}
				\tkzDefPointBy[translation = from A to B](D)	\tkzGetPoint{C}
				\tkzDefMidPoint(A,C)	\tkzGetPoint{O}
				\tkzDefPointWith[linear,K=0.6](A,B)		\tkzGetPoint{E}
				\tkzDefPointBy[translation = from E to A](C)	\tkzGetPoint{F}
				\tkzDefPointWith[linear,K=0.3](A,D)		\tkzGetPoint{M}
				\tkzDefPointBy[translation = from M to A](C)	\tkzGetPoint{N}
				\tkzDrawSegments(A,B B,C C,D D,A A,C B,D M,E E,N N,F F,M M,N E,F A,F E,C)
				\tkzMarkSegments[mark=|](A,E C,F)
				\tkzMarkSegments[mark=||](A,M C,N)
				\tkzDrawPoints(A,B,C,D,O,E,F,M,N)
				\tkzLabelPoints[left](A,M,D)
				\tkzLabelPoints[right](B,N,C)
				\tkzLabelPoints[below](O,F)
				\tkzLabelPoints[above](E)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Ta có $AE \parallel FC$ (do $ABCD$ là hình bình hành) và $AE=FC$ (giả thiết) nên suy ra tứ giác $AECF$ là hình bình hành.\\
			Chứng minh $\triangle FCN = \triangle EAM~(c-g-c) \Rightarrow FN=ME~(1)$ \\
			Chứng minh $\triangle MDF = \triangle NBE~(c-g-c) \Rightarrow MF=NE~(2)$ \\
			Từ $(1)$ và $(2)$ suy ra tứ giác $MENF$ là hình bình hành.
			\item Gọi $O$ là giao điểm của 2 đường chéo $AC$ và $BD$ của hình bình hành $ABCD$. \\
			Vì tứ giác $AECF$ là hình bình hành có $O$ là trung điểm của đường chéo $AC$ nên $O$ là trung điểm của đường chéo $EF$. \\
			Tương tự vì tứ giác $MENF$ là hình bình hành có $O$ là trung điểm của đường chéo $EF$ nên $O$ là trung điểm của đường chéo $MN$. \\
			Vậy $AC,BD,EF,MN$ đồng quy tại $O$.
			\item Khi $ABCD$ là hình vuông thì $AC \perp BD$. \\
			Mà $ME \parallel BD$, $MF \parallel AC$ (tính chất đường trung bình của tam giác) nên $ME \perp MF~(3)$. \\
			Lại có $MN \perp EF~(4)$ (do $AB \perp BC$). \\
			Vì $MENF$ là hình bình hành và kết hợp với $(3)$ và $(4)$ nên $MENF$ là hình vuông.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2016-2017, Ninh Bình]%[9D5G1]
	Cho $x$ và $y$ thỏa mãn: $x^2+2xy+4x+4y+3y^2+3=0$. Tìm giá trị lớn nhất và nhỏ nhất của biểu thức $B=x+y+2017$.
	\loigiai
	{
		Ta có:
		\begin{align*}
			& x^2+2xy+4x+4y+3y^2+3=0 \\
			\Leftrightarrow & x^2+2xy+y^2+4x+4y+3=-2y^2 \le 0 \\
			\Leftrightarrow & (x+y)^2+2.(x+y).2 +2^2-1=-2y^2 \le 0 \\
			\Leftrightarrow & (x+y+2)^2-1=-2y^2 \le 0 \\
			\Leftrightarrow & (x+y+1)(x+y+3)=-2y^2 \le 0 \\
			\Leftrightarrow & (x+y+2017-2016)(x+y+2017-2014) = -2y^2 \le 0 \\
			\Leftrightarrow & (B-2016)(B-2014)=-2y^2 \le 0 \\
			\Rightarrow & \left[\begin{aligned}
				\begin{cases}
					B-2016 \le 0 \\
					B-2014 \ge 0
				\end{cases} \\
				\begin{cases}
					B-2016 \ge 0 \\
					B-2014 \le 0
				\end{cases}
			\end{aligned}\right.
			\Leftrightarrow \left[\begin{aligned}
			\begin{cases}
				B \le 2016 \\
				B \ge 2014
			\end{cases} \\
			\begin{cases}
				B \ge 2016 \\
				B \le 2014
			\end{cases}
			\end{aligned}\right.
			\Leftrightarrow 2014 \le B \le 2016
		\end{align*}
		GTLN của $B$ bằng 2016 khi $(x;y)=(-1;0)$. GTNN của $B$ bằng 2014 khi $(x;y)=(-3;0)$.
	}
\end{ex}
