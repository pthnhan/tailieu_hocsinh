\subsection*{Khó}
\begin{ex}%[Huỳnh Văn Quy] %[1001-1020 Lê Quốc Bảo]%[0H1G3-2]
	Cho tam giác $ABC$. Gọi $A'$, $B'$, $C'$ lần lượt là trung điểm của $BC$, $CA$, $AB$. Khẳng định nào sau đây \textbf{sai}?
	\choice{$\vec{AA'}+\vec{BB'}+\vec{CC'}=\vec{0}$}
	{$\vec{BC}=\dfrac{2}{3}\left(\vec{BB'}-\vec{CC'}\right)$}
	{\True$\vec{AB}=\dfrac{2}{3}\left(2\vec{BB'}+\vec{CC'}\right)$}
	{$\vec{CA}=\dfrac{2}{3}\left(\vec{BB'}+2\vec{CC'}\right)$}
	\loigiai{Xét tính đúng sai của từng khẳng định:\\
		Xét phương án \textbf{A}: $\vec{AA'}+\vec{BB'}+\vec{CC'}=\dfrac{1}{2}\left(\vec{AB}+\vec{AC}\right)+\dfrac{1}{2}\left(\vec{BA}+\vec{BC}\right)+\left(\vec{CA}+\vec{CB}\right) $\\
		$=\dfrac{1}{2}\left(\vec{AB}+\vec{AC}+\vec{BA}+\vec{BC}+\vec{CA}+\vec{CB}\right)=\dfrac{1}{2}\left(\vec{AA}+\vec{AA}+\vec{AA}\right)=\vec{0}$. Phương án \textbf{A} đúng.\\
		Xét phương án \textbf{B}: $\dfrac{2}{3}\left(\vec{BB'}-\vec{CC'}\right)=\dfrac{2}{3}\left[\dfrac{1}{2}\left(\vec{BA}+\vec{BC}\right)-\dfrac{1}{2}\left(\vec{CA}+\vec{CB}\right)\right]$\\
		$=\left(\vec{BA}-\vec{CA}+\vec{BC}-\vec{CB}\right)=\dfrac{1}{3}\left(\vec{BC}+2\vec{BC}\right)=\vec{BC}$. Phương án \textbf{B} đúng.\\
		Phương án \textbf{C}: có $\dfrac{2}{3}\left(2\vec{BB'}+\vec{CC'}\right)=\dfrac{2}{3}\left[2\cdot\dfrac{1}{2}\left(\vec{BA}+\vec{BC}\right)+\dfrac{1}{2}\left(\vec{CA}+\vec{CB}\right)\right]$\\
		$=\dfrac{1}{3}\left(2\vec{BA}+2\vec{BC}+\vec{CA}+\vec{CB}\right)=\dfrac{1}{3}\left(-2\vec{AB}+2\vec{BC}-\vec{BC}+\vec{CA}\right)$\\
		$=\dfrac{1}{3}\left(-2\vec{AB}+\vec{BC}+\vec{CA}\right)=\dfrac{1}{3}\left(-3\vec{AB}\right)=-\vec{AB}\ne\vec{AB}$. Phương án \textbf{C} sai.\\
		Xét phương án \textbf{D}: có $\dfrac{2}{3}\left(\vec{BB'}+2\vec{CC'}\right)=\dfrac{2}{3}\left[\dfrac{1}{2}\left(\vec{BA}+\vec{BC}\right)+2\cdot\dfrac{1}{2}\left(\vec{CA}+\vec{CB}\right)\right]$\\
		$=\dfrac{2}{3}\cdot\dfrac{1}{2}\left[\vec{BA}+\vec{BC}+2\left(\vec{CA}+\vec{CB}\right)\right]=\dfrac{1}{3}\left(\vec{BA}+\vec{BC}+2\vec{CA}+2\vec{CB}\right)=\dfrac{1}{3}\left(2\vec{CA}+\vec{CB}+\vec{BA}\right)$\\
		$=\dfrac{1}{3}\left(2\vec{CA}+\vec{CA}\right)=\dfrac{1}{3}\cdot\vec{CA}=\vec{CA}$. Phương án \textbf{D} đúng.}
\end{ex}

	Trên trục $(O;\vec{e})$ cho hai điểm $A$, $B$ có tọa độ lần lượt là $3$; $-2$. Tính độ dài đại số của véc-tơ $\vec{AB}$.
	\choice
	{\True $\overline{AB}=-5$}
	{$\overline{AB}=5$}
	{$\overline{AB}=1$}
	{$\overline{AB}=2$}
	\loigiai{
		Ta có $\overline{AB}=x_B-x_A=-2-3=-5$.
	}
\end{ex}
