\begin{dang}{Diện tích hình giới hạn bởi: đồ thị hàm số - trục hoành và hai cận}
	Tính diện tích hình giới hạn bởi: $(C): y=f(x)$, $Ox$, $x=a$, $x=b$ với $a<b$.\\
	{\bf Bước 1.} Tìm tất cả các nghiệm của phương trình $f(x)=0$ trong khoảng $(a;b)$.\\
	{\bf Bước 2.} Áp dụng công thức tính diện tích và tách ra thành nhiều đoạn thành phần để xét dấu phá trị tuyệt đối như sau
	$$S=\displaystyle\int\limits_a^b\left| f(x)\right|\mathrm{\,d}x=\displaystyle\int\limits_a^{x_1}\left| f(x)\right|\mathrm{\,d}x+\displaystyle\int\limits_{x_1}^{x_2}\left| f(x)\right|\mathrm{\,d}x+\cdots+\displaystyle\int\limits_{x_{n-1}}^{x_n}\left| f(x)\right|\mathrm{\,d}x+\displaystyle\int\limits_{x_n}^b\left| f(x)\right|\mathrm{\,d}x=$$
	$$=\left|\displaystyle\int\limits_a^{x_1} f(x)\mathrm{\,d}x\right|+\left|\displaystyle\int\limits_{x_1}^{x_2} f(x)\mathrm{\,d}x\right|+\cdots+\left|\displaystyle\int\limits_{x_{n-1}}^{x_n} f(x)\mathrm{\,d}x\right|+\left|\displaystyle\int\limits_{x_n}^b f(x)\mathrm{\,d}x\right|.$$
	Ở đó $x_1<x_2<\cdots<x_{n-1}<x_n$ là các nghiệm của phương trình hoành độ giao điểm nằm trong khoảng $(a;b)$.\\
	{\bf Bước 3.} Kết luận.
\end{dang}

\begin{bt}%[2D3K3]
	Tính diện tích hình phẳng giới hạn bởi các đường: $x=-1$, $x=2$, $y=0$ và $y=x^2-2x$.
	\loigiai{
		Ta có $x^2-2x=0\Leftrightarrow\hoac{x&=0\\x&=2}$.\\
		Diện tích hình giới hạn là
		$$S=\displaystyle\int\limits_{-1}^2\left| x^2-2x\right|\mathrm{\,d}x=\displaystyle\int\limits_{-1}^0\left| x^2-2x\right|\mathrm{\,d}x + \displaystyle\int\limits_{0}^2\left| x^2-2x\right|\mathrm{\,d}x = \left|\displaystyle\int\limits_{-1}^0(x^2-2x)\mathrm{\,d}x\right| + \left|\displaystyle\int\limits_{0}^2(x^2-2x)\mathrm{\,d}x\right|=$$
		$$=\left|\left.\left(\dfrac{x^3}{3}-x^2\right)\right|_{-1}^{0}\right|+\left|\left.\left(\dfrac{x^3}{3}-x^2\right)\right|_{0}^{2}\right|=\dfrac{8}{3}.$$
	}
\end{bt}

\begin{bt}%[2D3K3]
	Tính diện tích hình phẳng giới hạn bởi các đường: $y=x^2-x-2$, $y=0$, $x=-2$, $x=0$.
	\loigiai{
		Ta có phương trình
		$$x^2-x-2=0\Leftrightarrow \hoac{x&=-1\\ x&=2}.$$
		Diện tích hình giới hạn là
		$$S=\displaystyle\int\limits_{-2}^0\left|x^2-x-2\right|\mathrm{\,d}x=\displaystyle\int\limits_{-2}^{-1}\left|x^2-x-2\right|\mathrm{\,d}x+\displaystyle\int\limits_{-1}^0\left|x^2-x-2\right|\mathrm{\,d}x=$$
		$$=\left|\displaystyle\int\limits_{-2}^{-1}(x^2-x-2)\mathrm{\,d}x\right|+\left|\displaystyle\int\limits_{-1}^0(x^2-x-2)\mathrm{\,d}x\right|=\left|\left.\left(\dfrac{x^3}{3}-\dfrac{x^2}{2}-2x\right)\right|_{-2}^{-1}\right|+\left|\left.\left(\dfrac{x^3}{3}-\dfrac{x^2}{2}-2x\right)\right|_{-1}^{0}\right|=3.$$
	}
\end{bt}

\begin{bt}%[2D3K3]
	Tính diện tích hình phẳng giới hạn bởi các đường: $x=1$, $x=\mathrm{e}$, $y=0$, $y=\dfrac{\ln x}{2\sqrt{x}}$.
	\loigiai{
		Phương trình $\dfrac{\ln x}{2\sqrt{x}}=0$ vô nghiệm trong khoảng $(1;\mathrm{e})$.\\
		Diện tích hình phẳng là
		$$\displaystyle\int\limits_{1}^{\mathrm{e}}\left|\dfrac{\ln x}{2\sqrt{x}}\right|\mathrm{\,d}x=\left|\displaystyle\int\limits_{1}^{\mathrm{e}}\dfrac{\ln x}{2\sqrt{x}}\mathrm{\,d}x\right|=\left|\left.\sqrt{x}\ln x\right|_1^{\mathrm{e}}-\displaystyle\int\limits_1^{\mathrm{e}}\dfrac{1}{\sqrt{x}}\mathrm{\,d}x\right|=\left|\left.(\sqrt{x}\ln x-2\sqrt{x})\right|_1^{\mathrm{e}}\right|=2-\sqrt{\mathrm{e}}.$$
	}
\end{bt}

\begin{bt}%[2D3K3]
	Tính diện tích hình phẳng giới hạn bởi các đường: $y=\tan^3x$, $y=0$, $x=-\dfrac{\pi}{4}$, $x=\dfrac{\pi}{4}$.
	\loigiai{
		Nghiệm của phương trình $\tan^3x=0$ trong khoảng $\left(-\dfrac{\pi}{4};\dfrac{\pi}{4}\right)$ là $x=0$.\\
		Diện tích hình phẳng là 
		$$\displaystyle\int\limits_{-\frac{\pi}{4}}^{\frac{\pi}{4}}\left|\tan^3x\right|\mathrm{\,d}x=\displaystyle\int\limits_{-\frac{\pi}{4}}^{0}\left|\tan^3x\right|\mathrm{\,d}x+\displaystyle\int\limits_{0}^{\frac{\pi}{4}}\left|\tan^3x\right|\mathrm{\,d}x=\left|\displaystyle\int\limits_{-\frac{\pi}{4}}^{0}\tan^3x\mathrm{\,d}x\right|+\left|\displaystyle\int\limits_{0}^{\frac{\pi}{4}}\tan^3x\mathrm{\,d}x\right|=$$
		$$=\left|\displaystyle\int\limits_{-\frac{\pi}{4}}^{0}\left(\tan x(1+\tan^2x)-\dfrac{\sin x}{\cos x}\right)\mathrm{\,d}x\right|+\left|\displaystyle\int\limits_{0}^{\frac{\pi}{4}}\left(\tan x(1+\tan^2x)-\dfrac{\sin x}{\cos x}\right)\mathrm{\,d}x\right|=$$
		$$\left|\displaystyle\int\limits_{-\frac{\pi}{4}}^{0}\tan x\mathrm{\,d}(\tan x)+\displaystyle\int\limits_{-\frac{\pi}{4}}^{0}\dfrac{\mathrm{d}(\cos x)}{\cos x}\right|+\left|\displaystyle\int\limits_{0}^{\frac{\pi}{4}}\tan x\mathrm{\,d}(\tan x)+\displaystyle\int\limits_{0}^{\frac{\pi}{4}}\dfrac{\mathrm{d}(\cos x)}{\cos x}\right|=$$
		$$=\left|\left.\left(\dfrac{\tan^2x}{2}+\ln|\cos x|\right)\right|_{-\frac{\pi}{4}}^{0}\right|+\left|\left.\left(\dfrac{\tan^2x}{2}+\ln|\cos x|\right)\right|_{0}^{\frac{\pi}{4}}\right|=1+2\ln\left(\dfrac{\sqrt{2}}{2}\right).$$
	}
\end{bt}

\begin{bt}%[Phạm Tiến Đạt]%[2D3K3]

\begin{bt}%[Phạm Tiến Đạt]%[2D3G3]
	Tính thể tích khối tròn xoay sinh ra do hình phẳng giới hạn bởi các đường $y=\sqrt{x}.\mathrm{e}^\frac{x}{2} $, $y=0$, $x=2$, khi nó quay quanh trục $Ox$.	
	\loigiai{
		Phương trình hoành độ giao điểm của $y=\sqrt{x}.\mathrm{e}^\frac{x}{2} $ và $y=0$ là
		$$\sqrt{x}.\mathrm{e}^\frac{x}{2}=0  \Leftrightarrow \sqrt{x}=0 \; \text{(vì } \mathrm{e}^\frac{x}{2} >0) \Leftrightarrow x=0.$$
		Gọi V là thể tích cần tìm.	
		$$V=\pi \displaystyle\int\limits_{0}^{2}(\sqrt{x}.\mathrm{e}^\frac{x}{2})^2 \mathrm{\,d}x=\pi \displaystyle\int\limits_{0}^{2}(x.\mathrm{e}^x)\mathrm{\,d}x=\pi.I.$$
		Bây giờ, ta tính $I=\displaystyle\int\limits_{0}^{2}(x.\mathrm{e}^x)\mathrm{\,d}x$.\\
		Đặt $\heva{u&=x\\
			dv&=\mathrm{e}^x\mathrm{\,d}x} \Rightarrow \heva{du&=\mathrm{\,d}x\\v&=\mathrm{e}^x.}$\\
		Khi đó $$I=(x.\mathrm{e}^x) \Big|_0^2-\displaystyle\int\limits_{0}^{2}(\mathrm{e}^x)\mathrm{\,d}x=2\mathrm{e}^2-\mathrm{e}^x \Big|_0^2=2\mathrm{e}^2-\mathrm{e}^2+1=\mathrm{e}^2+1.$$
		Vậy $V=\pi (\mathrm{e}^2+1) \: \textit{(đvtt)}$.
	}
\end{bt}

	\loigiai{
		Vận tốc của vật $v(t)=\displaystyle\int\left(3t+t^2\right)\mathrm{\, d}x=\dfrac{t^3}{3}+\dfrac{3t^2}{2}+C$.\\
		Với $v(0)=10$ nên $C=10$. Do đó, $v(t)=\dfrac{t^3}{3}+\dfrac{3t^2}{2}+10$.\\
