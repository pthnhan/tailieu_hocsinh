\noindent\textbf{II. PHẦN TỰ LUẬN}

\begin{bt}%[Thi GHK, Thái Phiên - Hải Phòng, 2018]%[Lê Nguyễn Viết Tường, EX-DT-10&11]%[0H1B3-2]%[0H1K3-2]%[0H1K3-4]
Cho tam giác $ABC$.
\begin{enumerate}
\item Với điểm $Q$ bất kì. Chứng minh rằng $\vv{AB}+\vv{CQ}=\vv{AC}+\vv{BQ}$.
\item Gọi $A'$ là điểm đối xứng với $B$ qua $A$, $B'$ là điểm đối xứng với $C$ qua $B$, $C'$ là điểm đối xứng với $A$ qua $C$. Chứng minh rằng với mọi điểm $O$ bất kì ta luôn có $$\vv{OA}+\vv{OB}+\vv{OC}=\vv{OA'}+\vv{OB'}+\vv{OC'}.$$
\item Gọi $M$, $N$ là các điểm lần lượt trên $AB$, $AC$ sao cho $\vv{MA}+3\vv{MB}=\vv{0}$, $2\vv{NB}-3\vv{NC}=\vv{0}$. Hãy biểu diễn $\vv{MN}$ theo $\vv{AB}$ và $\vv{AC}$.
\end{enumerate}
\loigiai{
\immini{
a) Ta có $\vv{AB}-\vv{AC}=\vv{CB}$ và $\vv{BQ}-\vv{CQ}=\vv{CB}$.\\
Do đó ta có $\vv{AB}-\vv{AC}=\vv{BQ}-\vv{CQ}\Rightarrow \vv{AB}+\vv{CQ}=\vv{AC}+\vv{BQ}$.\medskip

b) Do $A'$ là điểm đối xứng của $B$ qua $A$ nên $A$ là trung điểm của $BA'$.\\
Do đó với mọi điểm $O$ ta có $\vv{OA'}+\vv{OB}=2\vv{OA}$.\\
Chứng minh tương tự ta có $\vv{OB'}+\vv{OC}=2\vv{OB}$ \\và $\vv{OC'}+\vv{OA}=2\vv{OC}$.
}{
\begin{tikzpicture}[>=stealth,line join=round,line cap=round,scale=1]
\tkzDefPoints{0/0/B,1.5/2/A,3.5/0/C}
\tkzDefBarycentricPoint(A=1,B=-2)\tkzGetPoint{A'}
\tkzDrawSegments[](B,C C,A A,A')
\tkzDrawPoints[fill=black](A,B,C,A')
\tkzLabelPoints[above](A)
\tkzLabelPoints[below left](A')
\tkzLabelPoints[left](B)
\tkzLabelPoints[below right](C)
\end{tikzpicture}
}
\medskip

\noindent Khi đó ta có \\$\vv{OA'}+\vv{OB}+\vv{OB'}+\vv{OC}+\vv{OC'}+\vv{OA}=2\vv{OA}+2\vv{OB}+2\vv{OC}$\\
$\Rightarrow \vv{OA}+\vv{OB}+\vv{OC}=\vv{OA'}+\vv{OB'}+\vv{OC'}$.\medskip

\noindent c) Ta có \\
$\vv{MA}+3\vv{MB}=\vv{0}\Rightarrow -\vv{AM}+3\left(\vv{AB}-\vv{AM}\right) =\vv{0}\Rightarrow\vv{AM}=\dfrac{3}{4}\vv{AB}$.\\
$2\vv{NB}-3\vv{NC}=\vv{0}\Rightarrow 2\left(\vv{AB}-\vv{AN}\right) -3\left(\vv{AC}-\vv{AN}\right) =\vv{0}\Rightarrow\vv{AN}=-2\vv{AB}+3\vv{AC}$.\\
Khi đó ta có\\
$\vv{MN}=\vv{AN}-\vv{AM}=-2\vv{AB}+3\vv{AC}-\dfrac{3}{4}\vv{AB}=-\dfrac{11}{4}\vv{AB}+3\vv{AC}$.
}
\end{bt}
\begin{bt}%[Thi GHK, Thái Phiên - Hải Phòng, 2018]%[Lê Nguyễn Viết Tường, EX-DT-10&11]%[0H1B4-3]%[0H1K4-3]
Trong mặt phẳng tọa độ $Oxy$, cho điểm $A(-3;5)$, $B(-4;-3)$, $C(1;1)$.
\begin{enumerate}
\item Tìm tọa độ điểm $D$ sao cho tứ giác $ABCD$ là hình bình hành.
\item Tìm tọa độ điểm $K$ thuộc trục hoành sao cho $KA+KB$ nhỏ nhất.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[>=stealth,line join=round,line cap=round,scale=.5]
\tkzDefPoints{-3/5/A,-4/-3/B,1/1/C}
\tkzDefPointBy[translation=from B to A](C)\tkzGetPoint{D}
\tkzDrawSegments[](A,B B,C C,D D,A)
\tkzLabelPoints[above left	](A)
\tkzLabelPoints[above right](D)
\tkzLabelPoints[below left](B)
\tkzLabelPoints[below right](C)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Tứ giác $ABCD$ là hình bình hành \\
$\Leftrightarrow\vv{AB}=\vv{DC}\Leftrightarrow (-1;-8)=(1-x_D;1-y_D)\Leftrightarrow\heva{&1-x_D=-1\\&1-y_D=-8}\Leftrightarrow\heva{&x_D=2\\&y_D=9.}$\\
Vậy $D(2;9)$ là điểm cần tìm.
\item Gọi $K(a;0)$ là điểm cần tìm.\\
Ta có $KA+KB\ge AB$.\\
Dấu \lq\lq$=$\rq\rq\, xảy ra khi $A,K,B$ thẳng hàng.\\
$\Leftrightarrow \vv{AK}=k\vv{AB}\Leftrightarrow (a+3;-5)=k(-1;-8)\Leftrightarrow\heva{&a+3=-k\\&-5=-8k}\Leftrightarrow\heva{&a=-\dfrac{29}{8}\\&k=\dfrac{5}{8}.}$\\
Vậy $K\left(-\dfrac{29}{8};0\right) $ thỏa yêu cầu bài toán.
\end{enumerate}
}
\end{bt}
\begin{bt}%[Thi GHK, Thái Phiên - Hải Phòng, 2018]%[Lê Nguyễn Viết Tường, EX-DT-10&11]%[0H1G3-5]
Cho tứ giác $ABCD$ không phải là hình bình hành. Gọi $M$ và $N$ là hai điểm lần lượt di động trên các đoạn thẳng $AB$, $CD$ sao cho $\dfrac{ND}{NC}=\dfrac{MB}{MA}$. Gọi $E$, $F$, $I$ lần lượt là trung điểm của đoạn thẳng $AC$, $BD$ và $MN$. Chứng minh rằng ba điểm $E$, $I$, $F$ thẳng hàng.
\loigiai{
\immini{
Gọi $O$ là trung điểm của $AD$.\\
Đặt $\dfrac{ND}{NC}=\dfrac{MB}{MA}=k$. Ta có $\vv{OF}=\dfrac{1}{2} \vv{AB},\vv{OE}=\dfrac{1}{2}\vv{DC}$,\medskip

$\vv{OI}=\dfrac{1}{2}\left(\vv{OM}+\vv{ON}\right) $.\\
Lại có $\vv{OM}=\vv{OA}+\vv{AM}=\vv{OA}+\dfrac{1}{k+1}\vv{AB}$\\
và $\vv{ON}=\vv{OD}+\vv{DN}=\vv{OD}+\dfrac{k}{k+1}\vv{DC}$.
}{
\begin{tikzpicture}[>=stealth,line join=round,line cap=round,scale=1]
\tkzDefPoints{0/0/D,1/3/A,5/0/C,3.5/3.6/B}
\tkzDefBarycentricPoint(A=2,B=3)\tkzGetPoint{M}
\tkzDefBarycentricPoint(D=3,C=2)\tkzGetPoint{N}
\tkzDefMidPoint(A,C)\tkzGetPoint{E}
\tkzDefMidPoint(B,D)\tkzGetPoint{F}
\tkzDefMidPoint(M,N)\tkzGetPoint{I}
\tkzDefMidPoint(A,D)\tkzGetPoint{O}
\tkzDrawPoints[fill=black](E,F,I,A,B,C,D,M,N,O)
\tkzDrawSegments[](A,B B,C C,D D,A A,C B,D M,N)
\tkzLabelPoints[above left](A,F)
\tkzLabelPoints[above right](B,E)
\tkzLabelPoints[below left](I,D)
\tkzLabelPoints[below right](C)
\tkzLabelPoints[above](M)
\tkzLabelPoints[below](N)
\tkzLabelPoints[left](O)
\end{tikzpicture}
}
\noindent $\Rightarrow\vv{OI}=\dfrac{1}{2}\left(\dfrac{1}{k+1}\vv{AB}+\dfrac{k}{k+1}\vv{DC}\right) =\dfrac{1}{k+1}\vv{OF}+\dfrac{k}{k+1}\vv{OE}=\vv{OF}+\dfrac{k}{k+1}\vv{FE}$\medskip

\noindent$\Rightarrow\vv{FI}=\dfrac{k}{k+1}\vv{FE}$.\medskip

\noindent Vậy $E,I,F$ thẳng hàng.
}
\end{bt}
