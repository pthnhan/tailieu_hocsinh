	Cho phương trình $ax^2+bx+c=0,\ (a\ne 0)$ thỏa mãn $3a+4b+6c=0$. Chứng minh rằng phương trình luôn có nghiệm trong khoảng $\left(0;\dfrac{7}{8}\right)$.
	\loigiai{
	Đặt $f(x)=ax^2+bx+c$.\\
	Ta có $f(0)=c$ và $f\left(\dfrac{3}{4}\right)=\dfrac{9}{16}a+\dfrac{3}{4}b+c$.\\
	$f(0)\cdot f\left(\dfrac{3}{4}\right)=\dfrac{9a+12b+16c}{16}\cdot c=\dfrac{3(3a+4b+6c)-2c}{16}\cdot c=-\dfrac{c^2}{8}$.\\
	Nếu $c=0$. Khi đó $f(x)=0$ có nghiệm $x=\dfrac{3}{4}\in\left(0;\dfrac{7}{8}\right)$.\\
	Nếu $c\ne 0\Rightarrow f(0)\cdot f\left(\dfrac{3}{4}\right)<0$.\\
	Hàm số $y=f(x)$ liên tục trên $\left[0;\dfrac{3}{4}\right]\in\mathbb{R}$ nên có ít nhất một nghiệm trong $\left(0;\dfrac{3}{4}\right)$ và do đó luôn có nghiệm trong khoảng $\left(0;\dfrac{7}{8}\right)$.
	}
\end{bt}
