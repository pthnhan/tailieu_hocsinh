	{$y'=\dfrac{-5}{(x+2)^2}$}
	{\True $y'=\dfrac{5}{(x+2)^2}$}
	{$y'=\dfrac{3}{(x+2)^2}$}
	{$y'=\dfrac{2}{(x+2)^2}$}
	\loigiai
	{
		Ta có $y'=\dfrac{(2x-1)'(x+2)-(x+2)'(2x-1)}{(x+2)^2}=\dfrac{2x+4-2x+1}{(x+2)^2}=\dfrac{5}{(x+2)^2}.$
	}
