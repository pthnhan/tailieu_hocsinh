\section{NGUYÊN HÀM}
\subsection{Tóm tắt lý thuyết}
\subsubsection{Nguyên hàm và tính chất}
\paragraph{NGUYÊN HÀM}
	\begin{dn}
	Cho hàm số $f(x)$ xác định trên $\mathscr{K}$. Hàm số $F(x)$ được gọi là \textbf{nguyên hàm} của hàm số $f(x)$ trên $\mathscr{K}$ nếu $F'(x) = f(x)$ với mọi $x \in \mathscr{K}$.
	\end{dn}
	\begin{dl}
		Nếu $F(x)$ là một nguyên hàm của hàm số $f(x)$ trên $\mathscr{K}$ thì với mỗi hằng số $C$, hàm số $G(x) = F(x) + C$ cũng là một nguyên hàm của hàm số $f(x)$ trên $\mathcal{K}$.
	\end{dl}
	\begin{dl}
		Nếu $F(x)$ là một nguyên hàm của hàm số $f(x)$ trên $\mathscr{K}$ thì mọi nguyên hàm của hàm số $f(x)$ trên $\mathscr{K}$ đều có dạng $F(x) + C$, với $C$ là một hằng số.
	\end{dl}
	\begin{dl}
		Mọi hàm số $f(x)$ liên tục trên $\mathscr{K}$ đều có nguyên hàm trên $\mathcal{K}$.
	\end{dl}
\paragraph{TÍNH CHẤT CỦA NGUYÊN HÀM}
	\begin{tc} $\displaystyle \int{f'(x)} \mathrm{\,d}x = f(x) + C$ \end{tc}
	\begin{tc} $\displaystyle \int{kf(x)} \mathrm{\,d}x = k \displaystyle \int{f(x)} \displaystyle \mathrm{\,d}x $ ($k$ là một hằng số khác $0$). \end{tc}
	\begin{tc} $\displaystyle \int{ \left[ f(x) \pm g(x) \right]} \mathrm{\,d}x =\displaystyle \int{f(x)} \mathrm{\,d}x \pm \displaystyle \int{g(x)} \mathrm{\,d}x$ \end{tc}
\subsubsection{Phương pháp tìm nguyên hàm}
\paragraph{PHƯƠNG PHÁP ĐỔI BIẾN SỐ}

	\begin{dl}
		Nếu $\displaystyle\int {f(u) \mathrm{\,d}u} = F(u) + C$ và $u =u(x)$ là hàm số có đạo hàm liên tục thì $$\displaystyle \int {f(u(x)) u'(x) \mathrm{\,d}x} = F(u(x)) +C.$$
	\end{dl}
\paragraph{PHƯƠNG PHÁP TÍNH NGUYÊN HÀM TỪNG PHẦN}
	\begin{dl}
		Nếu hai hàm số $u = u(x)$ và $v = v(x)$ có đạo hàm liên tục trên $\mathscr{K}$ thì $$\displaystyle {u(x).v'x()} \mathrm{\,d}x = u(x)v(x) - \displaystyle\int{u'(x)v(x)} \mathrm{\,d}x.$$
	\end{dl}
	\begin{note}
		Vì $u'(x) \mathrm{\,d}x = \mathrm{\,d}v, u'(x) \mathrm{\,d}x = \mathrm{\,d}u$ nên đẳng thức trên còn được viết ở dạng $\displaystyle \int {u} \mathrm{\,d}y = uv - \displaystyle \int{v} \mathrm{\,d}v$.
	\end{note}
	Để tính nguyên hàm $\displaystyle\int{f\left(x\right)}  \mathrm{\,d}x$ bằng từng phần ta làm như sau:\\
	\textbf{Bước 1.} Chọn $u,  v$ sao cho $f\left(x\right) \mathrm{\,d}x=u \mathrm{\,d}v$ (chú ý $ \mathrm{d}v=v'\left(x\right) \mathrm{\,d}x$). 
	Sau đó tính $v=\displaystyle\int{\mathrm{\,d}v}$ và $ \mathrm{\,d}u=u' \cdot \mathrm{\,d}x$.\\
	\textbf{Bước 2.} Thay vào công thức $\left(*\right)$ và tính $\displaystyle\int{v \mathrm{d}u}$.
	Chú ý. Cần phải lựa chọn $u$ và $dv$ hợp lí sao cho ta dễ dàng tìm được $v$ và tích phân $\displaystyle\int{v \mathrm{\,d}u}$ dễ tính hơn $\displaystyle\int{u \mathrm{\,d}v}$. Ta thường gặp các dạng sau\\
\textbf{Dạng 1.} $I = \displaystyle\int {P\left( x \right)\left[ \begin{array}{l} 	\sin x\\ 	\cos x  	\end{array} \right]{\mathrm{\,d}}x} $. Với dạng này, ta đặt $\left\{ \begin{array}{l}  u = P\left( x \right)\\  {\mathrm{\,d}}v = \left[ \begin{array}{l} \sin x\\ \cos x \end{array} \right]\mathrm{\,d}x \end{array} \right.$ \\
\textbf{Dạng 2.} $I = \displaystyle\int {P\left( x \right){e^{ax + b}}{\mathrm{\,d}}x} $, trong đó $P\left(x\right)$ là đa thức.  
Với dạng này, ta đặt $\heva{ 	& u=P\left(x\right) \\  	&  \mathrm{\,d}v={e}^{ax+b} \mathrm{d}x \\  }$.
\textbf{Dạng 3.} $I=\displaystyle\int{P\left(x\right)\ln \left(mx+n\right) \mathrm{\,d}x}$, trong đó $P\left(x\right)$ là đa thức.
Với dạng này, ta đặt  $\heva{ 	& u=\ln \left(mx+n\right) \\  	&  \mathrm{\,d}v=P\left(x\right) \mathrm{d}x \\  }$.\\
\textbf{Dạng 4.} $I = \displaystyle\int {\left[ \begin{array}{l} 	\sin x\\  	\cos x 	\end{array} \right]{\mathrm{e}^x}\mathrm{\,d}x} $. Với dạng này, ta đặt $\left\{ \begin{array}{l}
u = \left[ \begin{array}{l}
\sin x\\
\cos x
\end{array} \right]\\
\mathrm{\,d}v = {\mathrm{e}^x}\mathrm{\,d}x
\end{array} \right.$

\begin{dang}{Nguyên hàm đổi biến số loại I}
	\begin{itemize}
		\item Đôi khi gặp bài toán nguyên hàm $\displaystyle\int{f(x)\mathrm{\,d}x}$ ta thực hiện các bước giải như sau:
		\begin{itemize}
			\item[+] Bước 1: Đặt $x=\varphi(t)$, với $\varphi(t)$ có đạo hàm liên tục trên $K$, được chọn hợp lý.
			\item[+] Bước 2: Lấy vi phân của $x$ theo biến số $t$, cụ thể là $\mathrm{\,d}x=\varphi'(t)\mathrm{\,d}t$.
			\item[+] Bước 3: thay cả $x=\varphi(t)$ lẫn $\mathrm{\,d}x=\varphi'(t)\mathrm{\,d}t$ vào $\displaystyle\int{f(x)\mathrm{\,d}x}$ được bài mới theo $t$.
			\item[+] Bước 4: Giải nguyên hàm mới $\displaystyle\int{f\left(\varphi(t) \right)\cdot\varphi'(t)\mathrm{\,d}t}$, được kết quả $F(t)$ theo $t$, sau đó thay biểu thức $x=\varphi(t)$ vào $F(t)$ để tìm được nguyên hàm theo biến $x$.
		\end{itemize}
		\item Một số dấu hiện nhận dạng nguyên hàm đổi biến số loại I

\begin{vd}%[Dương Phước Sang]%[2D3B1]\\
	Tìm họ nguyên hàm của hàm số $f(x)=\sqrt{3-x^2}$.
	\loigiai{
	Xét $I=\displaystyle\int{f(x)\mathrm{\,d}x}=\displaystyle\int{\sqrt{3-x^2}\mathrm{\,d}x}$. (*) \\
	Đặt $x=\sqrt{3}\sin t~\left(-\dfrac{\pi}{2}\leqslant t \leqslant \dfrac{\pi}{2} \right)$ thì $\mathrm{\,d}x=\sqrt{3}\cos t\mathrm{\,d}t$ và $\cos t \geqslant 0$.\\
	Thay vào (*) ta được: $$I=\displaystyle\int{\sqrt{3-3\sin^2t}\cdot \sqrt{3}\cos t\mathrm{\,d}t}=\displaystyle\int{3\cos^2t\mathrm{\,d}t} \Leftrightarrow I=\displaystyle\int{\dfrac{3(1+\cos2t)}{2}\mathrm{\,d}t}=\dfrac{3}{2}t+\dfrac{3}{4}\sin2t+C.$$
	Ta có $x=\sqrt{3}\sin t~\left(-\dfrac{\pi}{2}\leqslant t \leqslant \dfrac{\pi}{2} \right) \Rightarrow \cos t=\sqrt{1-\sin^2t}=\sqrt{\dfrac{3-x^2}{3}}$
	$$\Rightarrow \sin2t=2\sin t\cos t=\dfrac{2x\sqrt{3-x^2}}{3}.$$
	Vậy $I=\displaystyle\int{f(x)\mathrm{\,d}x}=\displaystyle\int{\sqrt{3-x^2}\mathrm{\,d}x}=\dfrac{3}{2}\arcsin\dfrac{x}{\sqrt{3}}+\dfrac{x\sqrt{1-x^2}}{2}+C~(C \text{ là hằng số thực})$.
	}
\end{vd}
\begin{vd}%[Dương Phước Sang]%[2D3B1]\\
	Tìm họ nguyên hàm của hàm số $f(x)=\dfrac{1}{4+x^2}$.
	\loigiai{
	Xét nguyên hàm $I=\displaystyle\int{f(x)\mathrm{\,d}x}=\displaystyle\int{\dfrac{1}{4+x^2}\mathrm{\,d}x}$ (*).\\
	Đặt $x=2\tan t~\left(-\dfrac{\pi}{2} <t< \dfrac{\pi}{2} \right)$ thì $\mathrm{\,d}x=2\left(1+\tan^2t \right)\mathrm{\,d}t$.\\
	Thay vào (*) ta được $I=\displaystyle\int{\dfrac{1}{4+4\tan^2t}\cdot 2(1+\tan^2t)\mathrm{\,d}t}=\displaystyle\int{\dfrac{1}{2}\mathrm{\,d}t}=\dfrac{1}{2}t+C$.\\
	Vậy $I=\displaystyle\int{f(x)\mathrm{\,d}x}=\displaystyle\int{\dfrac{1}{4+x^2}\mathrm{\,d}x}=\dfrac{1}{2}\arctan \dfrac{x}{2}+C~(C \text{ là hằng số thực})$.
	}
\end{vd}

\begin{bt}%[Dương Phước Sang]%[2D3B1]\\
	Tìm họ nguyên hàm của hàm số $f(x)=\dfrac{1}{\sqrt{1-x^2}}$.
	\loigiai{
		Xét họ nguyên hàm $I=\displaystyle\int{f(x)\mathrm{\,d}x}=\displaystyle\int{\dfrac{1}{\sqrt{1-x^2}}\mathrm{\,d}x}$.\\
		Đặt $x=\sin t~\left(-\dfrac{\pi}{2} \leqslant t \leqslant \dfrac{\pi}{2} \right)$ thì $\mathrm{\,d}x=\cos t\mathrm{\,d}t$ và $\cos t \geqslant 0$.\\
		Từ đó ta có $I=\displaystyle\int{f(x)\mathrm{\,d}x}=\displaystyle\int{\dfrac{1}{\sqrt{1-\sin^2t}}\cdot \cos t\mathrm{\,d}t}=\displaystyle\int{\mathrm{\,d}t}=t+C=\arcsin x + C$.\\
		Vậy họ các nguyên hàm của hàm số $f(x)=\dfrac{1}{\sqrt{1-x^2}}$ là $\displaystyle\int{f(x)\mathrm{\,d}x}=\arcsin x+C$.
	}
\end{bt}
\begin{bt}%[Dương Phước Sang]%[2D3K1]\\
	Tìm họ các nguyên hàm của hàm số $f(x)=\dfrac{x^2}{\sqrt{4-x^2}}$.
	\loigiai{
	Xét họ nguyên hàm $I=\displaystyle\int{f(x)\mathrm{\,d}x}=\displaystyle\int{\dfrac{x^2}{\sqrt{4-x^2}}\mathrm{\,d}x}$.\\
	Đặt $x=2\sin t~\left(-\dfrac{\pi}{2} \leqslant t \leqslant \dfrac{\pi}{2} \right)$ thì $\mathrm{\,d}x=2\cos t\mathrm{\,d}t$ và $\cos t \geqslant 0$.\\
	Từ đó ta có $I=\displaystyle\int{\dfrac{4\sin^2t}{\sqrt{4-4\sin^2t}}\cdot 2\cos t\mathrm{\,d}t}=\displaystyle\int{2(1-\cos2t)\mathrm{\,d}}=2t-\sin2t + C$.\\
	Do $x=2\sin t~\left(-\dfrac{\pi}{2} \leqslant t \leqslant \dfrac{\pi}{2} \right)$ nên $\cos t=\sqrt{1-\sin^2t}=\dfrac{1}{2}\sqrt{4-x^2}$ $$\Rightarrow \sin2t=2\sin t\cos t=\dfrac{1}{2}x\sqrt{4-x^2}.$$
	Vậy họ nguyên hàm của hàm số $f(x)$ là $F(x)=2\arcsin\dfrac{x}{2}-\dfrac{x\sqrt{4-x^2}}{2}+C$.
	}
\end{bt}
\begin{bt}%[Dương Phước Sang]%[2D3K1]\\
	Tìm họ nguyên hàm của hàm số $f(x)=\dfrac{1}{\left(1+x^2\right)^2}$.
	\loigiai{
	Xét họ nguyên hàm $I=\displaystyle\int{f(x)\mathrm{\,d}x}=\displaystyle\int{\dfrac{1}{\left(1+x^2\right)^2}\mathrm{\,d}x}$.\\
	Đặt $x=\tan t~\left(-\dfrac{\pi}{2} < t < \dfrac{\pi}{2} \right)$ thì $\mathrm{\,d}x=(1+\tan^2t)\mathrm{\,d}t$ và $\cos t>0$.\\
	Thay vào họ nguyên hàm $I$ ta được:\\
	$$\begin{aligned} I&= \displaystyle\int{\dfrac{1}{\left(1+\tan^2t\right)^2}\cdot (1+\tan^2t)\mathrm{\,d}t}=\displaystyle\int{\dfrac{1}{1+\tan^2t}\mathrm{\,d}t}=\displaystyle\int{\cos^2t\mathrm{\,d}t} \\ &=\displaystyle\int{\dfrac{1+\cos2t}{2}\mathrm{\,d}t}=\dfrac{1}{2}t+\dfrac{1}{4}\sin2t+C=\dfrac{1}{2}t+\dfrac{\tan t}{2(1+\tan^2t)}+C. \end{aligned}$$
	Vậy họ nguyên hàm của hàm số $f(x)=\dfrac{1}{\left(1+x^2\right)^2}$ là $\displaystyle\int{f(x)\mathrm{\,d}x}=\dfrac{1}{2}\arctan x+\dfrac{x}{2(1+x^2)}+C$.
	}
\end{bt}
\begin{bt}%[Dương Phước Sang]%[2D3K1]\\
	Tìm họ nguyên hàm của hàm số $f(x)=\sqrt{\dfrac{2-x}{2+x}}$.
	\loigiai{
	Xét họ nguyên hàm $I=\displaystyle\int{f(x)\mathrm{\,d}x}=\displaystyle\int{\sqrt{\dfrac{2-x}{2+x}}\mathrm{\,d}x}$.\\
	Đặt $x=2\cos2t~\left(0 \leqslant t \leqslant \dfrac{\pi}{2} \right)$ thì $\mathrm{\,d}x=-4\sin2t\mathrm{\,d}t$ và $\sin t \geqslant 0,\,\cos t \geqslant 0$.\\
	Ngoài ra, $\sin2t=\sqrt{1-\cos^22t}=\sqrt{1-\dfrac{x^2}{4}}=\dfrac{1}{2}\sqrt{4-x^2}$.\\
	Như vậy $I=\displaystyle\int{\sqrt{\dfrac{2-2\cos2t}{2+2\cos2t}}.(-4)\sin2t\mathrm{\,d}t}=-4\displaystyle\int{\sqrt{\dfrac{\sin^2t}{\cos^2t}} \cdot \sin2t\mathrm{\,d}t}=-8\displaystyle\int{\sin^2t\mathrm{\,d}t}$
	$$\Rightarrow I=-4\displaystyle\int{(1-\cos2t)\mathrm{\,d}t}=-4t+2\sin2t+C=-2\arccos\dfrac{x}{2}+\sqrt{4-x^2}+C.$$
	}
\end{bt}

\begin{vd}%[Dương Phước Sang]%[2D3B1]\\
	Tìm họ nguyên hàm của hàm số $f(x)=\dfrac{\sin x}{1+3\cos x}$.
	\loigiai{
		Xét họ nguyên hàm $I=\displaystyle\int f(x)\mathrm{\,d}x=\displaystyle\int \dfrac{\sin 	x}{1+3\cos x}\mathrm{\,d}x$.\\
		Đặt $t=1+3\cos x$ thì $\mathrm{\,d}t=-3\sin x\mathrm{\,d}t \Rightarrow -\dfrac{1}{3}\mathrm{\,d}t=\sin x\mathrm{\,d}x$.\\
		Như vậy $I=-\dfrac{1}{3}\displaystyle\int \dfrac{\mathrm{\,d}t}{t}=-\dfrac{1}{3}\ln|t|+C=-\dfrac{1}{3}\ln|1+3\cos x|+C$.\\
		\underline{Dùng vi phân}\\
		$\begin{aligned}\text{Ta có } \displaystyle\int f(x)\mathrm{\,d}x=\displaystyle\int \dfrac{\sin x}{1+3\cos x}\mathrm{\,d}x&=-\dfrac{1}{3}\displaystyle\int \dfrac{-3\sin x}{1+3\cos x}\mathrm{\,d}x\\
			&=-\dfrac{1}{3}\displaystyle\int \dfrac{\mathrm{\,d}(1+3\cos x)}{1+3\cos x}=-\dfrac{1}{3}\ln|1+3\cos x|+C. \end{aligned}$
	}
\end{vd}
\begin{vd}%[Dương Phước Sang]%[2D3B1]\\

\begin{dang}{Tìm nguyên hàm bằng phương pháp từng phần}
	Nếu hai hàm số $u = u(x)$ và $v = v(x)$ có đạo hàm liên tục trên $\mathscr{K}$ thì $$\displaystyle {u(x).v'x()} \mathrm{\,d}x = u(x)v(x) - \displaystyle\int{u'(x)v(x)} \mathrm{\,d}x.$$ 
	Để tính nguyên hàm $\displaystyle\int{f\left(x\right)}  \mathrm{\,d}x$ bằng từng phần ta làm như sau:\\
	\textbf{Bước 1.} Chọn $u,  v$ sao cho $f\left(x\right) \mathrm{\,d}x=u \mathrm{\,d}v$ (chú ý $ \mathrm{d}v=v'\left(x\right) \mathrm{\,d}x$). 
	Sau đó tính $v=\displaystyle\int{\mathrm{\,d}v}$ và $ \mathrm{\,d}u=u' \cdot \mathrm{\,d}x$.\\
	\textbf{Bước 2.} Thay vào công thức $\left(*\right)$ và tính $\displaystyle\int{v \mathrm{d}u}$.
	Chú ý. Cần phải lựa chọn $u$ và $dv$ hợp lí sao cho ta dễ dàng tìm được $v$ và tích phân $\displaystyle\int{v \mathrm{\,d}u}$ dễ tính hơn $\displaystyle\int{u \mathrm{\,d}v}$. Ta thường gặp các dạng sau\\
\textbf{Dạng 1.} $I = \displaystyle\int {P\left( x \right)\left[ \begin{array}{l} 	\sin x\\ 	\cos x  	\end{array} \right]{\mathrm{\,d}}x} $. Với dạng này, ta đặt $\left\{ \begin{array}{l}  u = P\left( x \right)\\  {\mathrm{\,d}}v = \left[ \begin{array}{l} \sin x\\ \cos x \end{array} \right]\mathrm{\,d}x \end{array} \right.$ \\
\textbf{Dạng 2.} $I = \displaystyle\int {P\left( x \right){e^{ax + b}}{\mathrm{\,d}}x} $, trong đó $P\left(x\right)$ là đa thức.  
Với dạng này, ta đặt $\heva{ 	& u=P\left(x\right) \\  	&  \mathrm{\,d}v={e}^{ax+b} \mathrm{d}x \\  }$.\\
\textbf{Dạng 3.} $I=\displaystyle\int{P\left(x\right)\ln \left(mx+n\right) \mathrm{\,d}x}$, trong đó $P\left(x\right)$ là đa thức.
Với dạng này, ta đặt  $\heva{ 	& u=\ln \left(mx+n\right) \\  	&  \mathrm{\,d}v=P\left(x\right) \mathrm{d}x \\  }$.\\
\textbf{Dạng 4.} $I = \displaystyle\int {\left[ \begin{array}{l} 	\sin x\\  	\cos x 	\end{array} \right]{\mathrm{e}^x}\mathrm{\,d}x} $. Với dạng này, ta đặt $\left\{ \begin{array}{l}
u = \left[ \begin{array}{l}
\sin x\\
\cos x
\end{array} \right]\\
\mathrm{\,d}v = {\mathrm{e}^x}\mathrm{\,d}x
\end{array} \right.$
\end{dang}

\begin{dang}{Nguyên hàm hàm phân thức}
Xét nguyên hàm dạng $\displaystyle\int \dfrac{f(x)}{g(x)}\mathrm{\,d}x$ với $f(x)$, $g(x)$ là hai đa thức.

Nếu bậc $f(x)\,\ge $ bậc $g(x)$ thì ta chia $f(x)$ cho $g(x)$ được thương $p(x)$ và dư $r(x)$. Khi đó
$$\displaystyle\int \dfrac{f(x)}{g(x)}\mathrm{\,d}x=\displaystyle\int p(x) \mathrm{\,d}x+\displaystyle\int \dfrac{r(x)}{g(x)}\mathrm{\,d}x.$$	
Lúc này bậc $r(x)\,<$ bậc $g(x)$. Do đó ta chỉ cần xét trường hợp bậc $f(x)\,<$ bậc $g(x)$ qua các dạng thường gặp sau:
\begin{enumerate}[Dạng 1:]
	\item  $\displaystyle \int \dfrac{g'(x)}{g(x)}\mathrm{\,d}x=\ln |g(x)|+C.$
	\item  $\displaystyle \int \dfrac{1}{ax+b}\mathrm{\,d}x=\dfrac{\ln|ax+b|}{a}+C.$
	\item  $\displaystyle \int \dfrac{Ax+B}{ax^2+bx+c}\mathrm{\,d}x$.
		\begin{itemize}
			\item Nếu $ax^2+bx+c=0$ có hai nghiệm phân biệt $x_1\,x_2$ thì phân tích $ax^2+bx+c=a(x-x_1)(x-x_2)$ và $Ax+B=M(x-x_1)+N(x-x_2) \Leftrightarrow \heva{&M+N=A\\ &-x_1M-x_2N=B}$. Do đó $\dfrac{Ax+B}{ax^2+bx+c}=\dfrac{1}{a}\left(\dfrac{M}{x-x_2}+\dfrac{N}{x-x_1}\right)$. Khi đó 
			$$\displaystyle\int \dfrac{Ax+B}{ax^2+bx+c}\mathrm{\,d}x=\dfrac{1}{a}\left(M\ln |x-x_2|+N\ln|x-x_1|\right)+C.$$
			\item Nếu $ax^2+bx+c=0$ có nghiệm kép $x=x_0$ thì ta đổi biến $t=x-x_0$.
		\end{itemize}
	\item $\displaystyle \int \dfrac{Ax^2+Bx+C}{ax^3+bx^2+cx+d}\mathrm{\,d}x$.
		\begin{itemize}
			\item Nếu $ax^3+bx^2+cx+d=0$ có nghiệm bội $3$ là $x_0$, tức là $ax^3+bx^2+cx+d=a(x-x_0)^3$ thì ta đổi biến $t=x-x_0$.
			\item Nếu $ax^3+bx^2+cx+d=0$ có hai nghiệm $x_1, x_2$, tức là $ax^3+bx^2+cx+d=a(x-x_1)(x-x_2)^2$ và thì phân tích
			$$\dfrac{Ax^2+Bx+C}{ax^3+bx^2+cx+d}=\dfrac{1}{a}\left(\dfrac{M}{x-x_1}+\dfrac{N}{x-x_2}+\dfrac{P}{(x-x_2)^2}\right).$$
		\end{itemize}
	\item  $\displaystyle\int \dfrac{f(x)}{g(x)}\mathrm{\,d}x$. Trong một số trường hợp ta có thể đổi biến $t=g(x)$.
\end{enumerate}
\end{dang}

			\textbf{Cách 1:} Đặt $x=\sin t$ với $-\dfrac{\pi}{2}\leq t\leq \dfrac{\pi}{2}$, suy ra $\mathrm{\,d}x=\cos t \mathrm{\,d}t$.
			
			Khi đó:
			
			\begin{equation*}
			\begin{split}
			\displaystyle\int\limits \sqrt{1-x^2}\mathrm{\,d}x&=\displaystyle\int\limits \sqrt{1-\sin^2 t}\cdot \cos t\mathrm{\,d}t\\
			&=\displaystyle\int\limits \cos^2 t\mathrm{\,d}t=\dfrac{1}{2}\displaystyle\int\limits (1+\cos 2t)\mathrm{\,d}t\\
			&=\dfrac{1}{2}\left( t+\dfrac{1}{2}\sin 2t\right)+C.
			\end{split}
			\end{equation*}
			
			
			\textbf{Cách 2:} Đặt $x=\cos t$ với $0\leq t\leq \pi$, suy ra $\mathrm{\,d}x=-\sin t \mathrm{\,d}t$.
			
			Khi đó:
			
			\begin{equation*}
			\begin{split}
			\displaystyle\int\limits \sqrt{1-x^2}\mathrm{\,d}x&=\displaystyle\int\limits \sqrt{1-\cos^2 t}\cdot \sin t\mathrm{\,d}t=-\displaystyle\int\limits \sin^2 t\mathrm{\,d}t\\
			&=-\dfrac{1}{2}\displaystyle\int\limits (1-\cos 2t)\mathrm{\,d}t=-\dfrac{1}{2}\left( t-\dfrac{1}{2}\sin 2t\right)+C.		
			\end{split}
			\end{equation*}
		}
	\end{vd}

\begin{dang}{Nguyên hàm có yếu tố mũ và lôgarit}
Ta tiến hành theo các bước sau:
\begin{enumerate}[\bf Bước 1.]
\item Biến đổi để đưa về các hàm cơ bản.
\item Sử dụng bảng nguyên hàm để tìm nguyên hàm của các hàm cơ bản đó, nếu là tích của nhiều hàm thì ta kết hợp phương pháp đổi biến hoặc từng phần để tìm nguyên hàm.
\end{enumerate}
\end{dang}

Đặt $u=\mathrm{e}^x\Rightarrow \mathrm{\,d}u=u\mathrm{\,d}x \Rightarrow \mathrm{\,d}x=\dfrac{\mathrm{\,d}u}{u}.$

		$=\dfrac{1}{2}\mathrm{e}^{2x}-\mathrm{e}^x+\ln \mathrm{e}^x+C=\dfrac{1}{2}\mathrm{e}^{2x}-\mathrm{e}^x+x+C$.
}

		Đặt $ u=\cos x \Rightarrow \mathrm{\,d}u=-\sin x \mathrm{\,d}x$ .\\
		Vậy $ \displaystyle\int\limits \dfrac{\tan x}{\cos x-1} \mathrm{\,d}x=\displaystyle\int\limits \dfrac{\sin x}{\cos x(\cos x-1)} \mathrm{\,d}x =\displaystyle\int\limits \dfrac{-1}{u(u-1)} \mathrm{\,d}u
		=\displaystyle\int\limits \left(\dfrac{1}{u}-\dfrac{1}{u-1}\right) \mathrm{\,d}u\\=\ln\left|u \right|-\ln\left|u-1 \right|+C=\ln\left|\dfrac{\cos x}{\cos x-1}\right|+C.$
	}
\end{vd}

\begin{vd}%[Lê Thanh Nin]%[2D3G1]
	Tính nguyên hàm $ \displaystyle\int\limits \dfrac{1}{\sin x\cos^3x} \mathrm{\,d}x .$
	\loigiai{
		Đặt $ u=\tan x \Rightarrow \mathrm{\,d}u=\dfrac{1}{\cos^2x} \mathrm{\,d}x$ .\\
		Vậy $ \displaystyle\int\limits \dfrac{1}{\sin x\cos^3x} \mathrm{\,d}x=\displaystyle\int\limits \dfrac{1+u^2}{u} \mathrm{\,d}u =\displaystyle\int\limits \left(\dfrac{1}{u}+u\right) \mathrm{\,d}u
		=\dfrac{u^2}{2}+\ln\left|u \right|+C=\ln\left|\tan x\right|+\dfrac{\tan^2x}{2}+C.$
	}
\end{vd}

\begin{vd}%[Lê Thanh Nin]%[2D3K1]
	Tính nguyên hàm $ \displaystyle\int\limits \dfrac{1}{\sin x} \mathrm{\,d}x. $
	\loigiai{
		Đặt $ u=\cos x \Rightarrow \mathrm{\,d}u=-\sin x \mathrm{\,d}x$ .\\
		Vậy $ \displaystyle\int\limits \dfrac{1}{\sin x} \mathrm{\,d}x=\displaystyle\int\limits \dfrac{\sin x}{\sin^2x} \mathrm{\,d}x=\displaystyle\int\limits \dfrac{\sin x}{1-\cos^2x} \mathrm{\,d}x=\displaystyle\int\limits \dfrac{1}{u^2-1} \mathrm{\,d}u\\ =\displaystyle\int\limits\dfrac{1}{2} \left(\dfrac{1}{u-1}-\dfrac{1}{u+1}\right) \mathrm{\,d}u
		=\dfrac{1}{2}\left(\ln\left|u-1 \right|-\ln\left|u+1 \right|\right)+C=\dfrac{1}{2}\ln\left|\dfrac{\cos x-1}{\cos x+1} \right|+C.$
	}
\end{vd}

	}

\begin{bt}%[Lê Thanh Nin]%[2D3K1]

	\loigiai{
		Ta có $ \displaystyle\int\limits \tan^4x \mathrm{\,d}x=\displaystyle\int\limits \left(\tan^4x-1+1\right) \mathrm{\,d}x=\displaystyle\int\limits \left(\tan^2x-1\right)\left(\tan^2x+1\right) \mathrm{\,d}x+x\\
		=\displaystyle\int\limits \left(\tan^2x-1\right) \mathrm{\,d}\tan x+x=\dfrac{\tan^3x}{3}-\tan x+x+C.$
	}
\end{bt}

\begin{bt}%[Lê Thanh Nin]%[2D3K1]

\begin{bt}%[Lê Thanh Nin]%[2D3K1]

\begin{bt}%[Lê Thanh Nin]%[2D3G1]

	\loigiai{

\begin{bt}%[Lê Thanh Nin]%[2D3K1]
	Tính nguyên hàm $ \displaystyle\int\limits \dfrac{\cos^3x}{\sin^2x+\sin x} \mathrm{\,d}x .$
	\loigiai{
		Đặt $ u=\sin x \Rightarrow \mathrm{\,d}u=\cos x \mathrm{\,d}x$. \\
		Vậy $ \displaystyle\int\limits \dfrac{\cos^3x}{\sin^2x+\sin x} \mathrm{\,d}x=\displaystyle\int\limits \dfrac{\left(1-\sin^2x\right)\cos x}{\sin^2x+\sin x} \mathrm{\,d}x=\displaystyle\int\limits \dfrac{1-u^2}{u^2+u} \mathrm{\,d}u=\displaystyle\int\limits \dfrac{1+u-u-u^2}{u^2+u} \mathrm{\,d}u\\
		=\displaystyle\int\limits \left(\dfrac{1}{u}-1\right) \mathrm{\,d}u
		=\ln\left|u \right|-u+C=\ln\left|\sin x \right|-\sin x+C.$
	}
\end{bt}


\begin{bt}%[Lê Thanh Nin]%[2D3K1]
	Cho $ F(x) $ là một nguyên hàm của hàm số $ f(x)=\dfrac{\sin2x}{\sin^2x+3} $ thỏa mãn $ F(0)=0$. Tính $ F(x) $.
	\loigiai{
		Đặt $ u=\sin^2x+3 \Rightarrow \mathrm{\,d}u=\sin2x \mathrm{\,d}x$ .\\
		Vậy $ F(x)=\displaystyle\int\limits \dfrac{\sin2x}{\sin^2x+3} \mathrm{\,d}x=\displaystyle\int\limits \dfrac{1}{u} \mathrm{\,d}u
		=\ln\left|u\right|+C=\ln\left|\sin^2x+3\right|+C.$\\
		Theo giả thiết ta có : $ F(0) =0\Leftrightarrow \ln 3+C=0\Leftrightarrow C=-\ln3 $\\
		Vậy $ F(x)=\ln\left|\sin^2x+3\right|-\ln3 . $	
	}
\end{bt}


\begin{bt}%[Lê Thanh Nin]%[2D3K1]
	Cho $ F(x) $ là một nguyên hàm của hàm số $ f(x)=\dfrac{\sin2x}{1+\cos x} $ thỏa mãn $ F\left(\dfrac{\pi}{2}\right) =0$. Tính $ F(0) $.
	\loigiai{
		Đặt $ u=1+\cos x \Rightarrow \mathrm{\,d}u=-\sin x \mathrm{\,d}x.$ \\
		Vậy $ F(x)=\displaystyle\int\limits \dfrac{\sin2x}{1+\cos x} \mathrm{\,d}x=\displaystyle\int\limits \dfrac{2\sin x\cos x}{1+\cos x} \mathrm{\,d}x=\displaystyle\int\limits \dfrac{2(1-u)}{u} \mathrm{\,d}u=\displaystyle\int\limits \left(\dfrac{2}{u}-2\right) \mathrm{\,d}u\break 
		=-2u+2\ln\left|u\right|+C=-2\left(1+\cos x\right)+2\ln\left|1+\cos x\right|+C.$\\
		Theo giả thiết ta có : $ F\left(\dfrac{\pi}{2}\right) =0\Leftrightarrow -2+C=0\Leftrightarrow C=2. $\\
		Vậy $ F(x)=-2\left(1+\cos x\right)+2\ln\left|1+\cos x\right|+2 .$\\
		Từ đó suy ra $ F(0)=2\ln2-2 .$
	}
\end{bt}
