\begin{ex}%[8D3Y3]%[8D3Y4]%[8D3B3]%[8D3K5]%[8D3K3]%[Đề thi HK2 lớp 8, Q11, TP.HCM, 2018]%[Đào Anh Dương]
    Giải phương trình.
    \begin{enumerate}
        \item $3x+2=5x+12$.
        \item $(x-2)(2x+3)=0$.
        \item $\dfrac{3x-4}{5} + \dfrac{x}{6} = \dfrac{x+1}{2}$.
        \item $\dfrac{x+7}{x-4} - \dfrac{7}{x+4} = \dfrac{56}{x^2-16}$.
        \item $\left| 2x-3 \right|=3x$.
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
    	\item $$3x+2=5x+12 \Leftrightarrow 2x=-10 \Leftrightarrow x=-5.$$
    	Vậy $x=-5$ là nghiệm của phương trình.
    	\item $$(x-2)(2x+3)=0 \Leftrightarrow \hoac{&x-2=0 \\&2x+3=0} \Leftrightarrow \hoac{&x=2 \\&x=-\dfrac{3}{2}.}$$
    	Vậy $x=2$ và $-\dfrac{3}{2}$ là nghiệm của phương trình.
    	\item 
    	\begin{align*}
       	&\quad \quad \dfrac{3x-4}{5} + \dfrac{x}{6} = \dfrac{x+1}{2}\\
       	&\Leftrightarrow 6(3x-4) + 5x = 15(x+1)\\
       	&\Leftrightarrow 18x-24 + 5x = 15x+15\\
       	&\Leftrightarrow 8x = 39 \Leftrightarrow x=\dfrac{39}{8}.
       	\end{align*}
       	Vậy $x=\dfrac{39}{8}$ là nghiệm của phương trình.
    	\item ĐK: $x \ne \pm 4$.
    	\begin{align*}
    	& \quad \quad \dfrac{x+7}{x-4} - \dfrac{7}{x+4} = \dfrac{56}{x^2-16}\\
    	&\Leftrightarrow (x+7)(x+4) - 7(x-4)=56\\
    	&\Leftrightarrow x^2+11x+28 - 7x+28 = 56\\
    	&\Leftrightarrow x^2+4x=0\\
    	&\Leftrightarrow x(x+4)=0 \Leftrightarrow \hoac{&x=0\\&x=-4 \, \text{(loại).}}
    	\end{align*}
    	  	Vậy $x=0$ là nghiệm của phương trình.
    	\item $$\left| 2x-3 \right|=3x
    	\Leftrightarrow \hoac{&2x-3=3x &\mbox{ khi } x \ge \dfrac{3}{2} \\&3-2x=3x &\mbox{ khi } x<\dfrac{3}{2}} \Leftrightarrow \hoac{&x=-3 \, \text{(loại)}\\&x=\dfrac{3}{5}.}$$
    	Vậy $x=\dfrac{3}{5}$ là nghiệm của phương trình.
    \end{enumerate}
    }
\end{ex}

\begin{ex}%[8D3K7]%[Đề thi HK2 lớp 8, Q11, TP.HCM, 2018]%[Đào Anh Dương]
    Hai người đi xe máy khởi hành cùng một lúc đi từ $A$ đến $B$. Người thứ nhất đi với vận tốc $40$ km/h, người thứ hai đi với vận tốc $60$ km/h nên đã đến $B$ trước người thứ nhất $1$ giờ. Tính quãng đường từ $A$ đến $B$.
    \loigiai
    {
   Gọi quãng đường $AB$ là $x$ (km), $x>0$.\\
   Thời gian người thứ nhất đi $\dfrac{x}{40}$ (km/h).\\ 
   Thời gian người thứ hai đi $\dfrac{x}{60}$ (km/h).\\ 
   Theo đề bài ta có phương trình $\dfrac{x}{40} = \dfrac{x}{60} + 1$.\\
   Giải phương trình trên ta được $x=120$.\\
   Vậy quãng đường $AB$ dài $120$ km/h.
    }
\end{ex}

\begin{ex}%[8H3B8]%[8H3G8][Đề thi HK2 lớp 8, Q11, TP.HCM, 2018]%[Đào Anh Dương]
	Cho $\triangle ABC$ vuông tại $A$, đường cao $AH$.
	\begin{enumerate}[a)]
		\item Chứng minh $\triangle ABC \backsim HBA$, từ đó suy ra $AB \cdot AC = AH \cdot BC$.
		\item Chứng minh $AH^2 = HB \cdot HC$.
		\item Phân giác của $\widehat{ABC}$ cắt $AH$ và $AC$ lần lượt tại $I$ và $K$. Chứng minh rằng $AI^2 = IH \cdot KC$.
	\end{enumerate}
	\loigiai
	{
		\begin{center}
			\begin{tikzpicture}[x=1cm,y=1cm,scale=1.3]
			\tkzInit[xmin=-3.3,xmax=2.7,ymin=-0.7,ymax=3]
			%\tkzAxeXY
			\tkzClip
			\tkzDefPoints{-3/0/B,-0.8/3/a,2.5/0/C}
			\tkzDefPointBy[projection = onto a--B](C)
			\tkzGetPoint{A}
			\tkzDefPointBy[projection = onto B--C](A)
			\tkzGetPoint{H}
			\tkzDrawBisector(C,B,A)		\tkzGetPoint{K}
			\tkzInterLL(B,K)(A,H)		\tkzGetPoint{I}
			\tkzMarkRightAngles[fill=gray!50](B,A,C A,H,C)
			\tkzMarkAngles[draw=black,size=0.7, fill=gray!50, opacity=0.5](C,B,K)
			\tkzMarkAngles[draw=black,size=0.6, fill=gray!50, opacity=0.5](K,B,A)
			\tkzDrawSegments(A,B B,C C,A A,H)
			\tkzDrawPoints(A,B,C,H,K,I)
			\tkzLabelPoints[above](A,K)
			\tkzLabelPoints[below](B,H,C)
			\tkzLabelPoints(I)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}[a)]
			\item $\triangle ABC$ và $\triangle HBA$ là hai tam giác vuông có chung góc nhọn $B$ nên chúng đồng dạng, ta có
			$$\dfrac{AC}{BC} = \dfrac{AH}{AB} \Leftrightarrow AC \cdot AB = AH \cdot BC.$$
			\item $\triangle AHB$ và $\triangle CHA$ là hai tam giác vuông có $\widehat{ABH} = \widehat{CAH}$ (vì cùng phụ với $\widehat{ACH}$) nên chúng đồng dạng, ta có
			 $$\dfrac{AH}{HB} = \dfrac{CH}{HA} \Leftrightarrow AH^2 = HB \cdot HC.$$
			\item Ta có $\triangle ABC \backsim \triangle HBA$ nên 
			$$\dfrac{AB}{HB} = \dfrac{BC}{BA} \Leftrightarrow AB^2 = BH \cdot BC.$$
			Vì $BI$ là phân giác của $\triangle ABH$ nên $\dfrac{BH}{BA} = \dfrac{IH}{IA}.$\\
			Ta lại có $\triangle BAI \backsim \triangle BCK$ nên
			\begin{align*}
			\dfrac{BA}{BC} = \dfrac{AI}{CK} &\Leftrightarrow \dfrac{BA^2}{BC \cdot BA} = \dfrac{AI}{CK}\\
			&\Leftrightarrow \dfrac{BH \cdot BC}{BC \cdot BA} = \dfrac{AI}{CK} \\	
			&\Leftrightarrow \dfrac{BH}{BA} = \dfrac{AI}{CK} \\
			&\Leftrightarrow \dfrac{IH}{IA} = \dfrac{AI}{CK} \\
			&\Leftrightarrow AI^2 = IH \cdot KC.
			\end{align*}
		\end{enumerate}
	}
\end{ex}
