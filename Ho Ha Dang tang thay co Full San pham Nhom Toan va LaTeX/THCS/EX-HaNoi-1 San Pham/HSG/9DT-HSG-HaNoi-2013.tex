\begin{ex}[Đề thi học sinh giỏi Toán 9, TP Hà Nội, năm 2013][9D2B2][9D5G2]
    \begin{enumerate}
    \item Tìm các số thực $a,b$ sao cho đa thức $4x^4 - 11x^3 - 2ax^2 + 5bx - 6$ chia hết cho đa thức $x^2 - 2x - 3$.
    \item Cho biểu thức $P = \left(a^{2013} -8a^{2012} + 11a^{2011}\right) + \left(b^{2013} - 8b^{2012} + 11b^{2011} \right)$. Tính giá trị của $P$ với $a=4+\sqrt{5}$ và $b=4-\sqrt{5}$.
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
    \item Đặt $P(x) = 4x^4 - 11x^3 - 2ax^2 + 5bx - 6$.\\
    Ta có $x^2 - 2x - 3 = (x+1)(x-3)$. Do đó, $P(x)$ chia hết cho đa thức $x^2 - 2x - 3$ khi và chỉ khi $P(x)$ chia hết cho cả hai đa thức $x+1$ và $x-3$. Theo định lý Bézout về số dư của phép chia đa thức, ta phải có 
    	$$\begin{cases} P(-1) = 0 \\ P(3) = 0 \end{cases} 
    	\Leftrightarrow \begin{cases} 2a + 5b = 9 \\ 18a - 15b = 21 \end{cases}
    	\Leftrightarrow \begin{cases} a = 2 \\ b = 1. \end{cases}$$
    Vậy đa thức $4x^4 - 11x^3 - 2ax^2 + 5bx - 6$ chia hết cho đa thức $x^2 - 2x - 3$ khi và chỉ khi $a = 2, b = 1$.
    \item Ta nhận thấy $a,b$ là hai nghiệm của phương trình $x^2 - 8x + 11 = 0$. Do đó ta có $a^2 - 8a + 11 = 0$ và $b^2 - 8b + 11 = 0$. Suy ra
    	\begin{align*}
    	P & = \left(a^{2013} -8a^{2012} + 11a^{2011}\right) + \left(b^{2013} - 8b^{2012} + 11b^{2011} \right)\\
    	& = a^{2011} \left( a^2 - 8a + 11 \right) + b^{2011} \left( b^2 - 8b + 11 \right) = 0.
    	\end{align*}
    \end{enumerate}
    }
\end{ex}

\begin{ex}[Đề thi học sinh giỏi Toán 9, TP Hà Nội, năm 2013][9D4G7][9D5G3]
	\begin{enumerate}
	\item Giải hệ phương trình: $\begin{cases} 6x^2 - y^2 -xy + 5x + 5y - 6 = 0 \\ 20x^2 -y^2 -28x + 9 = 0. \end{cases}$
	\item Tìm các số nguyên $x,y$ thỏa mãn $ 6x^2 + 10y^2 +2xy -x - 28y +18 = 0$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item Ta có $6x^2 - y^2 -xy + 5x + 5y - 6 = 0 \Leftrightarrow y^2 + (x-5)y -6x^2 -5x +6 = 0$. \hfill $(1)$\\
	Coi phương trình $(1)$ như là phương trình bậc $2$ ẩn $y$ và tham số là $x$. Khi đó
	$$\Delta = (x-5)^2 - 4 \cdot 1 \cdot (-6x^2-5x+6) =25x^2 +10x + 1 = (5x+1)^2.$$
	Suy ra $y= \dfrac{-x+5 + 5x+1}{2} = 2x+3$ hoặc $y = \dfrac{-x+5-5x-1}{2} = -3x+2$.\\
	{\bf TH1:} $y=2x+3$, thay vào phương trình $20x^2 -y^2 -28x + 9 = 0$ ta được
		$$20x^2 - (2x+3)^2 - 28x + 9 = 0 \Leftrightarrow 16x^2-40x = 0 
		\Leftrightarrow \left[ \begin{array}{ll}x = 0 &\Rightarrow y = 3 \\ x = \dfrac{5}{2} & \Rightarrow y = 8. \end{array} \right.$$
	{\bf TH2:} $y=-3x+2$, thay vào phương trình $20x^2 -y^2 -28x + 9 = 0$ ta được
		$$20x^2 - (-3x+2)^2 - 28x + 9 = 0 \Leftrightarrow 11x^2 -16x + 5 = 0 
		\Leftrightarrow \left[ \begin{array}{ll}x = 1 &\Rightarrow y = -1 \\ x = \dfrac{5}{11} & \Rightarrow y = \dfrac{7}{11}. \end{array} \right.$$
	Vậy hệ phương trình có $4$ cặp nghiệm $(x;y)$ là  $(0;3)$, $\left(\dfrac{5}{2}; 8\right)$, $(1;-1)$, $\left(\dfrac{5}{11}; \dfrac{7}{11}\right)$.
	\item Ta có
		\begin{align*}
		& 6x^2 + 10y^2 +2xy -x - 28y +18 = 0 \\
		\Leftrightarrow & 12x^2 + 20 y^2 + 4xy - 2x - 56y + 36 = 0 \\
		\Leftrightarrow & (x+4y)^2 + (4y-7)^2 + (x-1)^2 +10x^2 - 14 = 0	\\
		\Leftrightarrow & 10x^2 - 14 = -(x+4y)^2 - (4y-7)^2 - (x-1)^2 \le 0\\
		\Rightarrow & x^2 \le \dfrac{7}{5} \Rightarrow x \in \{ -1; 0; 1 \} \ (\text{vì $x$ là số nguyên}).
		\end{align*}		 
	\begin{itemize}
	\item $x = -1$, suy ra $10y^2 - 30y + 25 = 0$ (vô nghiệm).
	\item $ x = 0$, suy ra $10y^2 - 28y + 18 = 0 \Leftrightarrow \left[ \begin{array}{l} y = 1 \\ y = \dfrac{9}{5}\ (\text{loại}). \end{array} \right.$ 
	\item $x = 1$, suy ra $10y^2 - 26y + 23 = 0$ (vô nghiệm).
	\end{itemize}
	Vậy phương trình có 1 cặp nghiệm nguyên $(x;y)$ là $(0;1)$.
	\end{enumerate}
	}
\end{ex}

\begin{ex}[Đề thi học sinh giỏi Toán 9, TP Hà Nội, năm 2013][9D5G1]
	Cho ba số thực dương $a,b,c$ thỏa mãn $\dfrac{1}{a} + \dfrac{2}{b} + \dfrac{3}{c} = 3$. Chứng minh:
		$$\dfrac{27a^2}{c (c^2 + 9a^2)} + \dfrac{b^2}{a(4a^2 + b^2)} + \dfrac{8c^2}{b(9b^2 +4c^2)} \ge \dfrac{3}{2}.$$
	\loigiai{
	Đặt $a = \dfrac{1}{x}$; $b = \dfrac{2}{y}$; $c=\dfrac{3}{z}$. Ta có $\dfrac{1}{a} + \dfrac{2}{b} + \dfrac{3}{c} = 3 \Leftrightarrow x + y + z = 3$. Khi đó, bất đẳng thức cần chứng minh tương đương với 
		$$\dfrac{z^3}{z^2+x^2} + \dfrac{x^3}{x^2 + y^2} + \dfrac{y^3}{y^2 + x^2} \ge \dfrac{3}{2}.$$
	Ta có
	\begin{align*}
	& \dfrac{z^3}{z^2+x^2} + \dfrac{x^3}{x^2 + y^2} + \dfrac{y^3}{y^2 + x^2}\\
	= & (x+y+z) - \left( \dfrac{x^2z}{x^2+z^2} + \dfrac{z^2 y}{z^2 + y^2} + \dfrac{y^2 z}{y^2 + x^2} \right) \\
	\ge & (x+y+z) - \left( \dfrac{x^2 z}{2xz} + \dfrac{z^2 y}{2zy} + \dfrac{y^2 x}{2yx} \right)\\
	\ge & (x+y+z) - \left(\dfrac{x}{2} + \dfrac{y}{2} + \dfrac{z}{2} \right) = 3 - \dfrac{3}{2} = \dfrac{3}{2}.
	\end{align*}
	Dấu ``$=$'' xảy ra khi và chỉ khi $x=y=z=1$ hay $a = 1, b = 2, c = 3$.
	}
\end{ex}`

\begin{ex}[Đề thi học sinh giỏi Toán 9, TP Hà Nội, năm 2013][9H3G7]
	Cho tam giác $ABC$ có ba góc nhọn, nội tiếp đường tròn $(O)$ và $AB < AC$. Các đường cao $AD$, $BE$, $CF$ của tam giác $ABC$ gặp nhau tại $H$. Gọi $I$ là giao điểm hai đường thẳng $EF$ và $CB$. Đường thẳng $AI$ cắt $(O)$ tại $MM$ ($M$ khác $A$).
	\begin{enumerate}
	\item Chứng minh năm điểm $A,M,F,H,E$ cùng nằm trên một đường tròn.
	\item Gọi $N$ là trung điểm của $BC$. Chứng minh ba điểm $M,H,N$ thẳng hàng.
	\item Chứng minh $BM\cdot AC + AM \cdot BC = AB \cdot MC$.
	\end{enumerate}
	\loigiai{
		\begin{center}
	    \begin{tikzpicture}
	    \tkzInit[xmin=-8,xmax=9.0,ymin=-5,ymax=5]
	    \tkzClip
	    \tkzDefPoints{2/4/A, 0/0/B, 6/0/C}
	    \tkzDrawCircle[circum](A,B,C) 
    	\tkzCircumCenter(A,B,C) \tkzGetPoint{O}
    	\tkzDefPointBy[projection=onto B--C](A)\tkzGetPoint{D} %Dựng đường cao AD.;
		\tkzDefPointBy[projection=onto A--C](B)\tkzGetPoint{E} %Dựng đường cao BE.;
		\tkzDefPointBy[projection=onto A--B](C)\tkzGetPoint{F}
		\tkzInterLL(A,D)(B,E)\tkzGetPoint{H} %Dựng trực tâm H.;
		\tkzInterLL(E,F)(B,C)\tkzGetPoint{I}
		\tkzInterLC(A,I)(O,A) \tkzGetFirstPoint{M}
		\tkzDrawCircle[circum, color = red](A,H,E) 
    	\tkzCircumCenter(A,H,E) \tkzGetPoint{O_1}
    	\tkzDefPointBy[homothety = center B ratio 0.5](C) \tkzGetPoint{N}
    	\tkzDrawCircle[circum, color = blue](B,E,C) 
    	\tkzDefPointBy[homothety = center A ratio 2](O) \tkzGetPoint{G}
    	\tkzDrawPoints(A,B,C,D,E,F,H,I,M,N,O,G)
		\tkzLabelPoints(B,C,D,H,I,O,G)
		\tkzLabelPoints[below](N,F)
		\tkzLabelPoints[above](A,M,E)
		\tkzDrawSegments[thick](A,B I,C C,A A,D B,E C,F I,A M,G I,E G,B G,C G,A B,M C,M)
		\tkzMarkRightAngles(A,F,C A,D,C B,E,A A,B,G A,C,G)
    	\end{tikzpicture}
    	\end{center}
    \begin{enumerate}
    \item Xét đường tròn tâm $(O)$ có hai cát tuyến $IMA$ và $IBC$. Suy ra $IM.IA = IB.IC$. \hfill $(1)$\\
    	Dễ thấy tứ giác $BFEC$ nội tiếp và $EF$ cắt $BC$ tại $I$ nên suy ra $IF.IE = IB.IC$. \hfill $(2)$\\
    	Từ $(1)$ và $(2)$ suy ra $IM.IA = IF.IE$, suy ra tứ giác $AMFE$ nội tiếp. \\
    	Ta lại có tứ giác $AFHE$ nội tiếp đường tròn đường kính $AH$. Suy ra năm điểm $A,M,F,H,E$ cùng nằm trên đường tròn đường kính $AH$.
    \item Kẻ đường kính $AG$ của đường tròn $(O)$. Ta chứng minh $G,H,N$ thẳng hàng.\\
    	Ta có $GC \parallel BE$ (cùng vuông góc với $AC$) và $GB \parallel CF$ (cùng vuông góc với $AB$) nên tứ giác $HCGB$ là hình bình hành. Suy ra $GN$ đi qua trung điểm $N$ của $BC$ hay $G,H,N$ thẳng hàng.\\
    	$\widehat{GMA} = 90^\circ$ (góc nội tiếp chắn nửa đường tròn) $\Rightarrow GM \bot AM$.\\
    	Do $M$ thuộc đường tròn đường kính $AH$ (cmt) nên ta có $\widehat{AMH} = 90^\circ$ (góc nội tiếp chắn nửa đường tròn) $\Rightarrow HM \bot AM$.\\
    	Suy ra $G, H, M$ thẳng hàng. Từ đó suy ra $4$ điểm $G,H,M,N$ thẳng hàng. Vậy ta chứng minh được $H,M,N$ thẳng hàng.
    \item Ta chứng minh định lý Ptoleme: ``Nếu một tứ giác nội tiếp đường tròn thì tích hai đường chéo bằng tổng các tích của các cặp cạnh đối diện.''\\
    Giả sử tứ giác $ABCD$ nội tiếp, ta chứng minh $AB \cdot CD + AD\cdot BC = AC\cdot BD$.\\
   	\immini{
   	Trên $AC$ lấy điểm $K$ sao cho $\widehat{ABK} = \widehat{CBD}$.
   	Xét tam giác $ABK$ và tam giác $DBC$ ta có:\\
   		$\widehat{ABK} = \widehat{CBD}$ (cách lấy)\\
 		$\widehat{BAK} = \widehat{BDC}$ (hai góc nội tiếp chắn cung $BC$)\\
 	$\Rightarrow \Delta ABK \backsim \Delta DBC$ (g-g)\\
 	$\Rightarrow \dfrac{AB}{BD} = \dfrac{AK}{CD}\Rightarrow AB\cdot CD = BD\cdot AK$.\\
 	Xét tam giác $CBK$ và tam giác $DBA$, ta có:\\
 	$\widehat{CBK} = \widehat{CBA}- \widehat{ABK} = \widehat{CBA} - \widehat{CBD} = \widehat{ABD}$\\
 	$\widehat{BCK} = \widehat{BDA}$ (hai góc nội tiếp cùng chắn cung $AB$)\\
 	$\Rightarrow \Delta CBK \backsim \Delta DBA$ (g-g)\\
 	$\Rightarrow \dfrac{BC}{BD} = \dfrac{CK}{AD} \Rightarrow BC\cdot AD = BD\cdot CK$.
   	}
   	{
   	    \begin{tikzpicture}
   	    \tkzInit[xmin=-1,xmax=7,ymin=-3,ymax=5]
	    \tkzClip
	    \tkzDefPoints{2/4/A, 0/0/B, 6/0/C}
	    \tkzDrawCircle[circum](A,B,D) 
    	\tkzCircumCenter(A,B,C) \tkzGetPoint{O}
    	\tkzDefPointBy[rotation = center O angle -50](A)
		\tkzGetPoint{D}
		\tkzFindAngle(D,B,C)
		\tkzGetAngle{ang}
		\tkzDefPointBy[rotation = center B angle \ang](A)
		\tkzGetPoint{b}
		\tkzInterLL(B,b)(A,C)\tkzGetPoint{K}
    	\tkzDrawPoints(A,B,C,D,K)
    	\tkzLabelPoints(C)
    	\tkzLabelPoints[above](A,D,K)
    	\tkzLabelPoints[left](B)
    	\tkzDrawSegments[thick](A,B B,C C,A C,D D,A D,B B,K)
    	\tkzMarkAngle[size=0.7](K,B,A)
    	\tkzMarkAngle[size = 0.5](C,B,D)
    	\end{tikzpicture}
   	}
 	Từ đó ta có
 	$$AB\cdot CD + BC\cdot AD = BD\cdot AK + BD\cdot CK = BD (AK + CK) = BD \cdot AC.$$
 	Áp dụng định lý Ptoleme vào tứ giác nội tiếp $BMAC$ ta có $BM\cdot AC + AM \cdot BC = AB \cdot MC$.
    \end{enumerate}
	}

\begin{ex}[Đề thi học sinh giỏi Toán 9, TP Hà Nội, năm 2013][9D5G4]
	Cho $2013$ điểm $A_1, A_2, \dots, A_{2013}$ và đường tròn $(O;1)$ tùy ý cùng nằm trong mặt phẳng. Chứng minh trên đường tròn $(O;1)$ đó, ta luôn có thể tìm được một điểm $M$ sao cho $MA_1 + MA_2 + \dots + MA_{2013} \ge 2013.$
	\loigiai{
	Vẽ đường kính $BC$ bất kì của đường tròn $(O;1)$. Ta có
	\begin{align*}
	&(BA_1+BA_2 + \dots + BA_{2013}) + (CA_1 + CA_2 + \dots + CA_{2013}) \\
	 = & (BA_1 + CA_1) + (BA_2 + CA_2) + \dots + (BA_{2013} + CA_{2013}) \\
	 \ge & 2013 BC = 4026.
	\end{align*}
	Nếu $\begin{cases} BA_1+BA_2 + \dots + BA_{2013} < 2013 \\
		CA_1 + CA_2 + \dots + CA_{2013} < 2013 \end{cases}$ \\
		thì $(BA_1+BA_2 + \dots + BA_{2013}) + (CA_1 + CA_2 + \dots + CA_{2013}) < 4026$ (vô lí).\\
	Suy ra $\left[ \begin{array}{l} 
			BA_1+BA_2 + \dots + BA_{2013} \ge 2013 \\ 
			CA_1 + CA_2 + \dots + CA_{2013} \ge 2013. 
			\end{array} \right.$
	\begin{itemize}
	\item Nếu $BA_1+BA_2 + \dots + BA_{2013} \ge 2013$ thì $B$ là điểm $M$ cần tìm.
	\item Nếu $CA_1 + CA_2 + \dots + CA_{2013} \ge 2013$ thì $C$ là điểm $M$ cần tìm.
	\end{itemize}
	}
\end{ex}
