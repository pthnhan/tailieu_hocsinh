\begin{bt}%[HK2 (2017-2018), THPT TRẦN PHÚ, TPHCM]%[QK Trương, dự án EX9]%[2D3B3-1]
	Tính diện tích hình phẳng giới hạn bởi đồ thị hàm số $y=x \cdot \mathrm{e}^x$ và đường thẳng $y=x \cdot \mathrm{e}$. 
	\loigiai{
		Ta có $x \cdot \mathrm{e}^x=x \cdot \mathrm{e}\Leftrightarrow \hoac{& x=0 \\ & x=1}$. \\
		Diện tích hình phẳng cần tìm $S=\displaystyle\int\limits_0^1{\left|x\cdot \mathrm{e}^x - x \cdot \mathrm{e}\right|\mathrm{\,d}x}=\displaystyle\int\limits_0^1{\left(x \cdot \mathrm{e} - x\cdot \mathrm{e}^x\right)\mathrm{\,d}x}=\underbrace{\displaystyle\int\limits_0^1{x \cdot \mathrm{e}\mathrm{\,d}x}}_{S_1} - \underbrace{\displaystyle\int\limits_0^1{x \cdot \mathrm{e}^x\mathrm{\,d}x}}_{S_2}$. \\
