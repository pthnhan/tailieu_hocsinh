	\choice
	{$2a^2$}
	{\True $a^2$}
	{$a$}
	{$2a$}
	\loigiai{Đặt $x=-t$, suy ra $\mathrm{d}x=-\mathrm{d}t$. Khi đó
		$$I= \displaystyle\int_{-a}^{a}f(x)\mathrm{d}x = -\displaystyle\int_{a}^{-a}f(-t)\mathrm{d}t=\displaystyle\int_{-a}^{a}f(-t)\mathrm{d}t=\displaystyle\int_{-a}^{a}\left[a-f(t)\right]\mathrm{d}x=\displaystyle\int_{-a}^{a}a\mathrm{d}x -I$$
	$$\Rightarrow I=\dfrac{1}{2}\displaystyle\int_{-a}^{a}a\mathrm{d}x=\dfrac{1}{2}a\cdot (a+a)=a^2.$$}
\end{ex}
