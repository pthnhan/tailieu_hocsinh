\begin{bt}%[Tuyển sinh 10 Thái Nguyên 2019]%[Bùi Mạnh Tiến]%[8D1B8]
	Không dùng máy tính cầm tay hãy giải phương trình
	\begin{center}
		$(x-2018)(x-2020)=2018-x.$
	\end{center}
	\loigiai
	{
		\begin{eqnarray*}
			(x-2018)(x-2020)=2018-x&\Leftrightarrow & (x-2018)(x-2020)+x-2018=0\\
			&\Leftrightarrow & (x-2018)(x-2019)=0\\
			&\Leftrightarrow & \left[\begin{aligned} x & =2018\text{ (nhận)}\\ x &=2019 \text{ (nhận)}. \end{aligned}\right.
		\end{eqnarray*}
		Vậy phương trình có hai nghiệm $x=2018$ và $x=2019.$
	}
\end{bt}
\begin{bt}%[Tuyển sinh 10 Thái Nguyên 2019]%[Bùi Mạnh Tiến]%[9D1B3]
	Không dùng máy tính cầm tay, tính giá trị của biểu thức
	\begin{center}
		$A=\dfrac{\sqrt{15}-\sqrt{12}}{\sqrt{5}-2}-\dfrac{1}{2-\sqrt{3}}.$
	\end{center}
	\loigiai{
		Ta có $A=\dfrac{\sqrt{3}\left(\sqrt{5}-2\right)}{\sqrt{5}-2}-\dfrac{2+\sqrt{3}}{\left(2-\sqrt{3}\right)\left(2+\sqrt{3}\right)}=\sqrt{3}-\left(2+\sqrt{3}\right)=-2.$
	}
\end{bt}
\begin{bt}%[Tuyển sinh 10 Thái Nguyên 2019]%[Bùi Mạnh Tiến]%[9D3B3]
	Rút gọn biểu thức $P=\left(\dfrac{3\sqrt{x}}{\sqrt{x}+2}+\dfrac{\sqrt{x}}{\sqrt{x}-2}-\dfrac{x-\sqrt{x}}{x-4}\right):\dfrac{3\sqrt{x}}{\sqrt{x}+2}$ với $x>0$, $x\ne 4.$
	\loigiai{
		Ta có
		\begin{eqnarray*}
			P&= & \left[\dfrac{3\sqrt{x}\left(\sqrt{x}-2\right)+\sqrt{x}\left(\sqrt{x}+2\right)-x+\sqrt{x}}{\left(\sqrt{x}-2\right)\left(\sqrt{x}+2\right)}\right]\cdot \dfrac{\sqrt{x}+2}{3\sqrt{x}}\\
			&= &\dfrac{3x+3\sqrt{x}}{\left(\sqrt{x}-2\right)\left(\sqrt{x}+2\right)}\cdot \dfrac{\sqrt{x}+2}{3\sqrt{x}}\\
			&= &\dfrac{3\sqrt{x}\left(\sqrt{x}+1\right)}{\sqrt{x-2}}\cdot \dfrac{1}{3\sqrt{x}}=\dfrac{\sqrt{x}+1}{\sqrt{x}-2}.
		\end{eqnarray*}
	}
\end{bt}
\begin{bt}%[Tuyển sinh 10 Thái Nguyên 2019]%[Bùi Mạnh Tiến]%[9D2Y1]
	Cho hàm số bậc nhất $y=mx+1$ với $m$ là tham số. Tìm $m$ để đồ thị hàm số đi qua điểm $A(1;4)$. Với giá trị $m$ vừa tìm được, hàm số đồng biến hay nghịch biến trên $\mathbb{R}.$
	\loigiai
	{
		Hàm số đi qua điểm $A(1;4)$ nên $4=m\cdot 1+1\Rightarrow m = 3.$\\
		Vì $m=3>0$ nên hàm số là đồng biến.	
	}
\end{bt}

\begin{bt}%[Tuyển sinh 10 Thái Nguyên 2019]%[Bùi Mạnh Tiến]%[9D3B4]
	Giải hệ phương trình $\begin{cases} 3(x+1)+2(x+2y)=4\\ 4(x+1)-(x+2y)=9. \end{cases}$
	\loigiai{
		Ta có 
		\begin{eqnarray*}
			\begin{cases} 3(x+1)+2(x+2y)=4\\ 4(x+1)-(x+2y)=9 \end{cases} &\Leftrightarrow & \begin{cases} 3(x+1)+2(x+2y)=4\\ 8(x+1)-2(x+2y)=18 \end{cases}\\
			&\Leftrightarrow & \begin{cases} 11(x+1)=22\\ 4(x+1)-(x+2y)=9 \end{cases}\\
			&\Leftrightarrow & \begin{cases} x=1\\ y=-1. \end{cases}
		\end{eqnarray*}
		Vậy cặp nghiệm $(x;y)$ của phương trình là $(1;-1).$
	}
\end{bt}

\begin{bt}%[Tuyển sinh 10 Thái Nguyên 2019]%[Bùi Mạnh Tiến]%[9D4B6]
	Cho phương trình $x^2-4x+4m-3=0$ với $m$ là tham số. Tìm giá trị của $m$ để phương trình có hai nghiệm $x_1$; $x_2$ thỏa mãn $x_1^2+x_2^2=14.$
	\loigiai{
		Ta có $\Delta ' = 2^2-(4m-3)=7-4m.$ Để phương trình có hai nghiệm thì $\Delta '\ge 0\Leftrightarrow m\le \dfrac{7}{4}.$\\
		Với $m\le \dfrac{7}{4}$. Theo định lí Vi-ét ta có $\begin{cases} x_1+x_2=4\\ x_1\cdot x_2=4m-3. \end{cases}$\\
		Khi đó $x_1^2+x_2^2=14 \Leftrightarrow (x_1+x_2)^2-2x_1x_2=14
		\Leftrightarrow 4^2-2(4m-3)=14
		\Leftrightarrow m=1$.
	}
\end{bt}

\begin{bt}%[Tuyển sinh 10 Thái Nguyên 2019]%[Bùi Mạnh Tiến]%[9H1B2]
	Cho tam giác $ABC$ vuông tại $A$, đường cao $AH$, biết $AC=16$ $\mathrm{cm}$ và $\sin \widehat{CAH}=\dfrac{4}{5}$. Tính độ dài các cạnh $BC$, $AB$.
	\loigiai
	{
		\immini{
			Trong $\triangle AHC$ vuông tại $H$ ta có 
			\begin{center}
				$\sin \widehat{CAH}=\dfrac{HC}{AC}\Rightarrow HC=AC\cdot \sin \widehat{CAH}=16\cdot \dfrac{4}{5}=12{,}8$ $\mathrm{cm}.$
			\end{center}
			Xét $\triangle ABC$ vuông tại $A$ ta có 
			\begin{center}
				$AC^2=BC\cdot HC\Rightarrow HC=\dfrac{AC^2}{BC}=20$ $\mathrm{cm}.$
			\end{center}
			Trong $\triangle ABC$ ta có $AB=\sqrt{BC^2-AC^2}=12$ $\mathrm{cm}.$\\
			Vậy $AB=12$ $\mathrm{cm}$, $BC=20$ $\mathrm{cm}$.
		}
		{
			\begin{tikzpicture}[scale = 0.8,>=stealth, line join=round, line cap = round]
			\clip (-1.4,-3.6) rectangle (5.4,0.4);

			\tkzDefPointBy[projection = onto M--B](C) \tkzGetPoint{A}
			\tkzDefPointBy[projection = onto B--C](A) \tkzGetPoint{H}
			\tkzDrawPolygon(A,B,C)
			\tkzDrawSegments(A,H)
			\tkzDrawPoints[fill = black](A,B,C,H)
			\tkzLabelPoints[above](A)
			\tkzLabelPoints[below](B,C,H)
			\tkzMarkRightAngle(A,H,C)
			\tkzMarkRightAngle(C,A,B)

		}
	}
\end{bt}

\begin{bt}%[Tuyển sinh 10 Thái Nguyên 2019]%[Bùi Mạnh Tiến]%[9H2B7]
	Cho hai đường tròn ($O;4$ $\mathrm{cm}$) và ($O';11$ $\mathrm{cm}$). Biết khoảng cách $OO'=2a+3$ $\mathrm{cm}$ với $a$ là số thực dương. Tìm $a$ để hai đường tròn tiếp xúc nhau.
	\loigiai
	{
		\begin{center}
			\begin{tikzpicture}[scale = 0.8,>=stealth, line join=round, line cap = round]
			\tkzDefPoints{0/0/O, 1/0/A, 3.75/0/O', 8/0/B, 9/0/I, 10.75/0/J}
			\tkzDrawCircle(O,A)
			\tkzDrawCircle(O',A)
			\tkzDrawCircle(I,B)
			\tkzDrawCircle(J,B)
			\tkzDrawPoints[fill = black](O,O',I,J,A,B)
			\tkzDrawSegments(O,O' B,J)
			\draw(O) node[below] {$O$} (O') node[below] {$O'$} (I) node[below] {$O$} (J) node[below] {$O'$};
			\draw (2.75,-3) node[below] {\text{Hình 1}};
			\draw (10.75,-3) node[below] {\text{Hình 2}};
			\end{tikzpicture}
		\end{center}
		Các trường hợp có thể xảy ra là 
		\begin{itemize}
			\item Hai đường tròn tiếp xúc ngoài (xem hình $1$), ta có 
			\begin{center}
				$OO'=R+R'\Leftrightarrow 2a+3=15\Leftrightarrow a=6$ $\mathrm{cm}$.
			\end{center}
			\item Hai đường tròn tiếp xúc trong (xem hình $2$), ta có 
			\begin{center}
				$OO'=|R-R'|\Leftrightarrow 2a+3=|4-11|\Leftrightarrow a=2$ $\mathrm{cm}.$
			\end{center}
		\end{itemize}
		Vậy $a=6$ $\mathrm{cm}$ và $a=2$ $\mathrm{cm}$
	}
\end{bt}

\begin{bt}%[Tuyển sinh 10 Thái Nguyên 2019]%[Bùi Mạnh Tiến]%[9H2B2]
	Cho đường tròn $(O)$, dây cung $AB$ không đi qua tâm $O$. Gọi $M$ là điểm chính giữa của cung nhỏ $\wideparen{AB}$. Vẽ dây cung $MC$ không đi qua $O$ cắt đoạn thẳng $AB$ tại $D$ ($D\ne A$, $D\ne B$). Đường thẳng vuông góc với $AB$ tại $D$ cắt $OC$ tại $K$. Chứng minh rằng tam giác $KCD$ là tam giác cân.
	\loigiai
	{
		\immini{
			Vì $M$ là điểm chính giữa cung $AB$ nên $OM\perp AB$ \hfill (1)\\
			Theo giả thiết thì $DK\perp AB$ \hfill(2)\\
			Từ $(1)$ và $(2)$ suy ra $DK\parallel OM$\\
			Suy ra $\widehat{CDK}=\widehat{CMO}$ (đồng vị)\hfill(3)\\
			Mà $\triangle OCM$ cân tại $O$ nên $\widehat{CMO}=\widehat{OCM}$ \hfill(4)\\
			Từ $(3)$ và $(4)$ suy ra $\widehat{KDC}=\widehat{KCD}\Rightarrow \triangle KCD$ cân tại $K$.
			
		}
		{
			\begin{tikzpicture}[>=stealth, line join=round, line cap = round]
				\clip (-0.5,-3.2) rectangle (6.2, 3.5);
				\tkzDefPoints{0/0/S, 3/0/O, 0/2/U, 6/2/V}

				\tkzDefPointBy[projection = onto A--B](O)    \tkzGetPoint{H}
				\tkzInterLC(O,H)(O,S)    \tkzGetPoints{N}{M}

				\tkzInterLL(A,B)(C,M)    \tkzGetPoint{D}
				\tkzDefLine[orthogonal=through D](B,A) \tkzGetPoint{d}
				\tkzInterLL(D,d)(C,O)    \tkzGetPoint{K}
								
				\tkzDrawSegments(A,B M,C O,C D,K O,M)

\begin{bt}%[Tuyển sinh 10 Thái Nguyên 2019]%[Bùi Mạnh Tiến]%[9H3B7]%[9H3K7]
	Cho tam giác $ABC$ có ba góc nhọn và $AB<AC$ nội tiếp đường tròn tâm $O$. Các đường cao $BE$, $CF$ của tam giác $ABC$ cắt nhau tại $H$.
	\begin{listEX}
		\item Chứng minh tứ giác $AFHE$ nội tiếp được trong một đường tròn. Xác định tâm và bán kính của đường tròn đó.
		\item Gọi $M$ là giao điểm của $EF$ và $BC$, đường thẳng $MA$ cắt đường tròn $(O)$ tại điểm thứ hai $I$ khác $A$. Chứng minh tứ giác $AEFI$ nội tiếp được trong một đường tròn.
	\end{listEX}
	\loigiai
	{
		\begin{center}
			\begin{tikzpicture}[scale = 1.2,>=stealth, line join=round, line cap = round]
			\clip (-3,-3.2) rectangle (6.2,3.5);
			\tkzDefPoints{0/0/S, 3/0/O, 0/-2/U, 6/-2/V, 2/0/T}
			\tkzDrawCircle(O,S)
			\tkzInterLC(U,V)(O,S)    \tkzGetPoints{C}{B}
			\tkzDefLine[orthogonal=through T](O,S) \tkzGetPoint{t}
			\tkzInterLC(T,t)(O,S)    \tkzGetPoints{A'}{A}
			\tkzDefPointBy[projection = onto A--C](B)    \tkzGetPoint{E}
			\tkzDefPointBy[projection = onto A--B](C)    \tkzGetPoint{F}
			\tkzInterLL(B,E)(C,F)    \tkzGetPoint{H}
			\tkzInterLL(E,F)(B,C)    \tkzGetPoint{M}
			\tkzInterLC(M,A)(O,S)   \tkzGetFirstPoint{I}				
			\coordinate (J) at ($(A)!0.5!(H)$);
			
			\tkzDrawPoints[fill = black](A,B,C,O,E,F,H,M,I,J)
			\tkzDrawSegments(A,B B,C C,A B,E C,F M,E M,B M,A B,I E,I F,I A,H E,J F,J)
			\draw (A) node[above] {$A$} (B) node[below left] {$B$} (C) node[below right] {$C$} (I) node[left] {$I$} (M) node[below] {$M$} (E) node[right] {$E$} (F) node[above right] {$F$} (O) node[left] {$O$} (H) node[below] {$H$} (J) node[right] {$J$};
			\tkzMarkRightAngle(B,E,C)
			\tkzMarkRightAngle(C,F,B)
			\end{tikzpicture}
		\end{center}
		\begin{listEX}
			\item Gọi $J$ là trung điểm của $AH$. \\
			Vì $\triangle AEH$ và $\triangle AFH$ là các tam giác vuông tại $E$ và $F$ nên $EJ=\dfrac{1}{2}AH=JA=JH=FJ$. Suy ra $AFHE$ nội tiếp đường tròn $J$ bán kính $\dfrac{AH}{2}.$
			\item Tứ giác $BFEC$ có $\widehat{BFC}=\widehat{BEC}=90^\circ\Rightarrow BFEC$ là tứ giác nội tiếp. Suy ra $\widehat{MFB}=\widehat{MCE}$.\\
			Xét tam giác $\triangle MFB$ và $\triangle MCE$ có
			$\begin{cases} \widehat{M} \text{ chung}\\ \widehat{MFB}=\widehat{MCE}\end{cases}\Rightarrow \triangle MFB\backsim \triangle MCE$ (g.g).\\
			Suy ra $\dfrac{MF}{MC}=\dfrac{MB}{ME}\Rightarrow MF\cdot ME=MB\cdot MC.$\hfill(1) \\
			Tứ giác $BCAI$ nội tiếp đường tròn $(O)$ nên $\widehat{MCA}=\widehat{MIB}.$\\
			Xét $\triangle MIB$ và $\triangle MCA$ có $\begin{cases} \widehat{M} \text{ chung}\\ \widehat{MIB}=\widehat{MCA}\end{cases}\Rightarrow \triangle MIB\backsim \triangle MCA$ (g.g).\\
			Suy ra $\dfrac{MB}{MA}=\dfrac{MI}{MC}\Rightarrow MA\cdot MI=MB \cdot MC$\hfill(2)\\
			Từ $(1)$ và $(2)$ suy ra $ME\cdot MF=MI\cdot MA\Rightarrow \dfrac{MF}{MA}=\dfrac{MI}{ME}.$\\
			Xét $\triangle MIF$ và $\triangle MEA$ có $\begin{cases}
			\widehat{M} \text{ chung}\\ \dfrac{MF}{MA}=\dfrac{MI}{ME} 
			\end{cases}\Rightarrow \triangle MIF\backsim \triangle MEA\Rightarrow \widehat{MFA}=\widehat{MAE}.$\\
			Từ đây suy ra tứ giác $AEFI$ nội tiếp được một đường tròn.
		\end{listEX}
	}
\end{bt}
