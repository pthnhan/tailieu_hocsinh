\begin{bt}%[Thi GHK, THPT Lương Thế Vinh - Hồ Chí Minh, 2018]%[Nguyễn Trung Kiên, EX-DT-10&11]%[1D1Y2-1]
	Giải phương trình $\tan \left(3x- \dfrac{\pi}{3} \right)= \dfrac{\sqrt{3}}{3}$.
	\loigiai{\allowdisplaybreaks
		\begin{align*}
		&\tan \left(3x- \dfrac{\pi}{3} \right)= \dfrac{\sqrt{3}}{3} \tag{1}\\
		\Leftrightarrow\ &\tan \left(3x- \dfrac{\pi}{3} \right)=\tan \dfrac{\pi}{6}\\
		\Leftrightarrow\ &3x- \dfrac{\pi}{3}= \dfrac{\pi}{6}+k\pi\\
		\Leftrightarrow\ &x= \dfrac{\pi}{6}+k \dfrac{\pi}{3},(k \in \mathbb{Z}).
		\end{align*}}
\end{bt}

\begin{bt}%[Thi GHK, THPT Lương Thế Vinh - Hồ Chí Minh, 2018]%[Nguyễn Trung Kiên, EX-DT-10&11]%[1D1B3-1]
	Giải phương trình $\cos{2x}-3 \sin{x}-2=0$.
	\loigiai{\allowdisplaybreaks
		\begin{align*}
		&\cos{2x}-3 \sin{x}-2=0 \tag{2}\\
		\Leftrightarrow\ & -2 \sin^2{x}-3 \sin{x} -1=0\\
		\Leftrightarrow\ & \hoac{& \sin{x}=-1\\ & \sin{x}=-\dfrac{1}{2}}\\
		\Leftrightarrow\ & \hoac{& x=-\dfrac{\pi}{2}+k2 \pi \\&x=-\dfrac{\pi}{6}+k2\pi \\&x= \dfrac{7\pi}{6}+k2 \pi},( k\in \mathbb{Z}).
		\end{align*}}
\end{bt}

\begin{bt}%[Thi GHK, THPT Lương Thế Vinh - Hồ Chí Minh, 2018]%[Nguyễn Trung Kiên, EX-DT-10&11]%[1D1B3-2]
	Giải phương trình $\sin^2{x}+3 \sin{x} \cos{x}-6 \cos^2{x}=-1$.
	\loigiai{\allowdisplaybreaks	
		\begin{align*}
		&\sin^2{x}+3 \sin{x} \cos{x}-6 \cos^2{x}=-1 \tag{3}\\
		\Leftrightarrow\ &2\sin^2{x}+3 \sin{x} \cos{x}-5 \cos^2{x}=0\\
		\Leftrightarrow\ &2\sin{x}\left( \sin{x}- \cos{x}\right)+5 \cos{x} \left(\sin{x}- \cos{x} \right)=0\\
		\Leftrightarrow\ &\left(\sin{x}- \cos{x}\right)\left(2\sin{x}+ 5\cos{x} \right)=0\\
		\Leftrightarrow\ &\hoac{&\sin{x}- \cos{x}=0\\ &2\sin{x}+5 \cos{x}=0}\\
		\Leftrightarrow\ &\hoac{&\tan{x}=1 \\&\tan{x}=-\dfrac{5}{2}}\\
		\Leftrightarrow\ &\hoac{& x=\dfrac{\pi}{4}+k\pi\\ &x=\mathrm{arctan}{\left(-\dfrac{5}{2}\right)}+k\pi},(k\in \mathbb{Z}).
		\end{align*}}
\end{bt}

\begin{bt}%[Thi GHK, THPT Lương Thế Vinh - Hồ Chí Minh, 2018]%[Nguyễn Trung Kiên, EX-DT-10&11]%[1D1K3-3]
	Giải phương trình $\sin{x}- \sin{2x}= \sqrt{3}(\cos{2x}- \cos{x}+1)$.
	\loigiai{\allowdisplaybreaks
		\begin{align*}
		&\sin{x}- \sin{2x}= \sqrt{3}(\cos{2x}- \cos{x}+1) \tag{4}\\
		\Leftrightarrow\ &\sqrt{3} \cos{2x}+ \sin{2x} -\left(\sin{x}+ \sqrt{3} \cos{x}\right) +\sqrt{3}=0\\
		\Leftrightarrow\ &2\cos\left(2x- \dfrac{\pi}{6}\right)-2 \cos\left(x- \dfrac{\pi}{6} \right)+ \sqrt{3}=0\\
		\Leftrightarrow\ &\cos\left(2x-\dfrac{\pi}{6}\right)+ \cos\dfrac{\pi}{6}- \cos\left( x- \dfrac{\pi}{6}\right)=0\\
		\Leftrightarrow\ &2\cos{x} \cos\left(x- \dfrac{\pi}{6}\right)- \cos\left(x- \dfrac{\pi}{6}\right)=0\\
		\Leftrightarrow\ &\hoac{&\cos\left(x- \dfrac{\pi}{6}\right)=0\\&\cos{x}= \dfrac{1}{2}}\\
		\Leftrightarrow\ &\hoac{& x=\dfrac{2\pi}{3}+k \pi\\& x= \pm \dfrac{\pi}{3}+k2 \pi}, (k\in \mathbb{Z}).
		\end{align*}
		$\text{Kết hợp lại phương trình có các nghiệm là } x=\pm \dfrac{\pi}{3}+k2 \pi; x=\dfrac{2\pi}{3}+k2 \pi, (k\in \mathbb{Z})$. 
	}
\end{bt}

\begin{bt}%[Thi GHK, THPT Lương Thế Vinh - Hồ Chí Minh, 2018]%[Nguyễn Trung Kiên, EX-DT-10&11]%[1H2K1-5]
	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là tứ giác có các cạnh đối không song song.
	\begin{enumerate}
		\item Tìm giao tuyến của $(SAC)$ và $(SBD)$.
		\item Gọi $M$ là điểm trên cạnh $SC$ ($M$ không trùng với $S$ và $C$). Tìm giao điểm của $AM$ và $(SBD)$.
		\item Gọi $H$ là giao điểm của $SD$ và $(ABM)$, $I$ là giao điểm của $BM$ và $(SAD)$. Chứng minh rằng $A$, $I$, $H$ thẳng hàng.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=1,line join=round,line cap=round]
			\tkzDefPoints{0/0/A,1.2/-2/B,4/-1.3/C,4.5/0/D}
			\coordinate (S) at ($(A)+(1,4)$);
			\coordinate (M) at ($(S)!0.6!(C)$);
			\tkzInterLL(A,C)(B,D)\tkzGetPoint{O}
			\tkzInterLL(A,M)(S,O)\tkzGetPoint{E}
			\tkzInterLL(B,E)(S,D)\tkzGetPoint{H}
			\tkzInterLL(B,C)(A,D)\tkzGetPoint{F}
			\tkzInterLL(B,M)(S,F)\tkzGetPoint{I}
			\tkzDrawPolygon(S,A,B,C)
			\tkzDrawSegments(S,B S,F C,F B,I)
			\tkzDrawSegments[dashed](A,C A,F B,D A,M S,O S,D C,D B,H A,I)
			\tkzDrawPoints[fill=black,size=4](A,B,C,S,D,M,E,F,H,I)
			\tkzLabelPoints[above](S,I,H)
			\tkzLabelPoints[below](B,C,F,O)
			\tkzLabelPoints[left](A)
			\tkzLabelPoints[right](M)
			\tkzLabelPoints[above right](D)
			\tkzLabelPoints[xshift=-18pt,yshift=12pt](E)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Tìm giao tuyến của $(SAC)$ và $(SBD)$.\\
			Trong mặt phẳng $(ABCD)$ gọi $O$ là giao điểm của $AC$ và $BD$. Xét hai mặt phẳng $(SAC)$ và $(SBD)$. Ta có:\\
			$S$ là một điểm chung của hai mặt phẳng.\\
			$O \in AC \Leftrightarrow O \in (SAC)$, $O \in BD \Leftrightarrow O \in (SBD)$ nên $O$ là điểm chung thứ hai.\\
			Mà $S \notin (ABCD)$ nên $S \neq O$.\\
			Vậy $(SAC) \cap (SBD)=SO$.
			\item Gọi $M$ là điểm trên cạnh $SC$ ($M$ không trùng với $S$ và $C$). Tìm giao điểm của $AM$ và $(SBD)$.\\
			Do $M \in SC$ nên $AM \subset (SAC)$. Trong mặt phẳng $(SAC)$, gọi $E$ là giao điểm của hai đường thẳng $SO$ và $AM$.\\
			Ta có $E \in SO,\ SO \subset (SBD) \Rightarrow E \in (SBD)$ và $E \in AM$.\\
			Vậy $E$ là giao điểm của $AM$ và mặt phẳng $(SBD)$.
			\item Gọi $H$ là giao điểm của $SD$ và $(ABM)$, $I$ là giao điểm của $BM$ và $(SAD)$. Chứng minh rằng $A$, $I$, $H$ thẳng hàng.\\
			Trong mặt phẳng $(ABCD)$, do $AD$ và $BC$ không song song nên chúng cắt nhau giả sử tại $F$. Ta có $SF \subset(SAD)$ và $SF \subset (SBC)$.\\
			Trong mặt phẳng $(SBC)$, gọi $J$ là giao điểm của $BM$ và $SF$, khi đó do $SF \subset (SAD)$ nên $J$ là điểm chung của hai mặt phẳng $(ABM)$ và $(SAD)$.\\
			Mặt khác mặt phẳng $(ABM)$ và $(SAD)$ phân biệt, có $A$ là điểm chung, $A \not\equiv J$.\\
			Giao tuyến của $(ABM)$ và $(SAD)$ là đường thẳng $\Delta$ qua hai điểm $A, J$.\\
			Theo bài ra $H$ là giao điểm của $SD$ và $(ABM)$ nên $H \in (ABM)$, $H \in SD \Rightarrow H \in (SAD)$. Do đó $H$ là một điểm chung của $(ABM)$ và $(SAD)$. Suy ra $H \in \Delta$.\\
			Lại có $\{I\}=BM \cap (SAD)$ nên $I \equiv J$. Suy ra $I \in \Delta$.\\
			Vậy $A$, $H$, $I$ thẳng hàng.
		\end{enumerate}			
	}
\end{bt}
