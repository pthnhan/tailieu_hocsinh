	\begin{enumerate}
		\item Cho $A=\dfrac{x^{2}-\sqrt{x}}{x+\sqrt{x}+1}+\dfrac{x^{2}+\sqrt{x}}{x-\sqrt{x}+1}$.\\
		 Rút gọn biểu thức $B=1-\sqrt{2\cdot A-4\sqrt{x}+1}$  với $0\leq x\leq \dfrac{1}{4}$.
		\item Cho $x,y,z$ là các số thực khác $0$ và đôi một khác nhau thỏa mãn $\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=0$. Chứng minh rằng $\left( \dfrac{1}{x^{2}+2yz}+\dfrac{1}{y^{2}+2xz}+\dfrac{1}{z^{2}+2xy}\right) (x^{2016}+y^{2017}+z^{2018})=xy+yz+xz$.
	
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			\begin{eqnarray*}
			A&=&\dfrac{x^{2}-\sqrt{x}}{x+\sqrt{x}+1}+\dfrac{\sqrt{x}(\sqrt{x^3}+1)}{x-\sqrt{x}+1}\\
			&=&\dfrac{\sqrt{x}(\sqrt{x^3}-1)}{x+\sqrt{x}+1}+\dfrac{\sqrt{x}(\sqrt{x^3}+1)}{x-\sqrt{x}+1}\\
			&=&\dfrac{\sqrt{x}(\sqrt{x}-1)(x+\sqrt{x}+1)}{x+\sqrt{x}+1}+\dfrac{\sqrt{x}(\sqrt{x}+1)(x-\sqrt{x}+1)}{x-\sqrt{x}+1}\\
			&=& x-\sqrt{x}+x+\sqrt{x}=2x.
			\end{eqnarray*}
			
			Khi đó, ta có $$B=1-\sqrt{2A-4\sqrt{x}+1}=1-\sqrt{4x-4\sqrt{x}+1}=1-\vert 2\sqrt{x}-1\vert .$$
			Vì $0\leq x\leq \dfrac{1}{4}$ nên $2\sqrt{x}-1\leq 0$. Suy ra 
			$B=1+(2\sqrt{x}-1) =2\sqrt{x}$.	
			\item Từ $\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=0\Rightarrow xy+yz+xz=0 \quad (1)$.\\
			Mặt khác ta có $x^2+2yz=x^2-xy-xz+yz=(x-y)(x-z).$\\ 
			Tương tự ta có $y^2+2xz=(y-x)(y-z);z^2+2xy=(z-x)(z-y)$.\\
			Từ đó, suy ra
			\begin{eqnarray*}
			\dfrac{1}{x^{2}+2yz}+\dfrac{1}{y^{2}+2xz}+\dfrac{1}{z^{2}+2xy}&=& \dfrac{1}{(x-y)(x-z)}+\dfrac{1}{(y-x)(y-z)}+\dfrac{1}{(z-x)(z-y)}\\
			&=& \dfrac{y-z+z-x+x-y}{(x-y)(x-z)(y-x)}\\
			&= & 0 \quad (2).
			\end{eqnarray*}
			Từ $(1)$ và $(2)$, ta có $VT=VP=0$. Điều phải chứng minh.
		\end{enumerate}
	}
\end{ex}

	\begin{enumerate}

		\item Giải hệ phương trình $\heva{&x^{2}+y^{2}-xy=2 \\ &x^{3}=x+y.}$

	\loigiai{
		\begin{enumerate}
			\item Điều kiện xác định $x\geq 2$.\\
			 Đặt $\heva{&a=\sqrt{x+5} \\ &b=\sqrt{x-2} } \quad (a,b\geq 0) \Rightarrow \heva{&a^{2}-b^{2}=7 \\ &(a-b)(1+ab)=7 }\\ \Rightarrow a^{2}-b^{2}=(a-b)(1+ab)\Leftrightarrow (a-b)(a+b-1-ab)=0\\\Leftrightarrow (a-b)(1-b)(a-1)=0\Leftrightarrow \hoac{&a=b\\&a=1\\&b=1.}$
			\begin{itemize}
				\item $a=b\Rightarrow  x+5=x-2$ (loại.)
				\item $a=1\Rightarrow  x+5=1\Leftrightarrow x=-4$ (loại).
				\item  $b=1\Rightarrow x-2=1\Leftrightarrow x=3$.
			\end{itemize}
			Vậy phương trình có nghiệm duy nhất $x=3$.
			\item Xét hệ phương trình  $\heva{&x^{2}+y^{2}-xy=2 \\ &x^{3}=x+y.}$ \\
			Nhân hai vế với vế các phương trình trong hệ ta được $$2x^3=(x^2+y^2-xy)(x+y)=x^3+y^3\Rightarrow x^3=y^3\Rightarrow x=y.$$
			Từ đó $x^3=2x\Leftrightarrow \hoac{&x=0\\&x=\sqrt{2}\\&x=-\sqrt{2}.}$\\
			Thử lại, ta thấy $x=0$ không thỏa mãn hệ phương trình đã cho.\\
			Vậy ta có hai cặp nghiệm $(x;y)$ là $(\sqrt{2};\sqrt{2});(-\sqrt{2};-\sqrt{2})$.
		\end{enumerate}	
	}
\end{ex}

\begin{itemize}

				\item Nếu $a> b$.\\
				Do $(a-b)(a+b)$ chia hết cho $3$ và $11$; $1\leq a; \leq b \leq 9$, suy ra $a+b=11.$\\
			+) Nếu $a=8,b=3$: không thỏa mãn.\\
			+) Nếu $a=7,b=4$: thỏa mãn.\\
			+) Nếu $a=6;b=5$: không thỏa mãn.\\

\begin{ex}%[Đề thi HSG 9, tỉnh Hải Dương, năm 2018]%[Nguyễn Sỹ, Dự án EX-9G-18]%[9D0G7]
	Cho các số thực dương $x,y,z$ thỏa mãn $x^{2}+y^{2}+z^{2}\leq 3xyz$. Tìm giá trị lớn nhất của biểu thức $P=\dfrac{x^{2}}{x^{4}+yz}+\dfrac{y^{2}}{y^{4}+xz}+\dfrac{z^{2}}{z^{4}+xy}$
	\loigiai{
		Từ giả thiết, ta có $x^{2}+y^{2}+z^{2}\leq 3xyz\Leftrightarrow \dfrac{x}{yz}+\dfrac{y}{xz}+\dfrac{z}{xy}\leq 3$.\\
		Áp dụng bất đẳng thức AM-GM, ta có
		\begin{eqnarray*}
			\dfrac{x}{yz}+\dfrac{y}{xz}&\geq& \dfrac{2}{z}\\
			\dfrac{y}{xz}+\dfrac{z}{xy}&\geq& \dfrac{2}{x}\\
			\dfrac{x}{yz}+\dfrac{z}{xy}&\geq& \dfrac{2}{y}.
		\end{eqnarray*}
		Cộng các bất đẳng thức trên lại với nhau\\
		$$\Rightarrow \dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\leq \dfrac{x}{yz}+\dfrac{y}{xz}+\dfrac{z}{xy}\leq 3.$$
		Từ đó, ta có
		\begin{align*}
		P&\leq \dfrac{x^{2}}{2x^{2}\sqrt{yz}}+\dfrac{y^{2}}{2y^{2}\sqrt{xz}}+\dfrac{z^{2}}{2z^{2}\sqrt{xy}}=\dfrac{1}{2}(\dfrac{1}{\sqrt{xy}}+\dfrac{1}{\sqrt{xz}}+\dfrac{1}{\sqrt{yz}})\\&\leq \dfrac{1}{2}(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z})\leq \dfrac{3}{2}.
		\end{align*}
		Vậy giá trị lớn nhất của $P$ là $\dfrac{3}{2}$ xảy ra khi $x=y=z=1.$
		
	}
\end{ex}
