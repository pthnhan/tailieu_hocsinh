\begin{bt}%[Đề HK1, Đinh Thiện Lý, Hồ Chí Minh 17-18]%[BanNt]%[0D3B2-4]
Giải phương trình 
\begin{multicols}{2}
\begin{enumerate}
\item $\sqrt{3x-1}=5$.
\item $\sqrt{2x^2-8x-1}+2=x$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item Điều kiện xác định $x\geq \dfrac{1}{3}$.\\
$$\sqrt{3x-1}=5\Leftrightarrow 3x-1=25 \Leftrightarrow 3x=26 \Leftrightarrow x=\dfrac{26}{3}\,\text{(thỏa mãn điều kiện)}.$$\\
Vậy nghiệm của phương trình là $x=\dfrac{26}{3}$.
\item Ta có
\begin{align*}
\sqrt{2x^2-8x-1}+2=x  &\Leftrightarrow\sqrt{2x^2-8x-1}=x-2\\
&\Leftrightarrow \heva{&x-2\geq 0\\&2x^2-8x-1=(x-2)^2}\\
&\Leftrightarrow \heva{&x\geq 2\\&x^2-4x-5=0}\\
&\Leftrightarrow\heva{&x\geq 2\\&\hoac{&x=-1\\&x=5}}
\Leftrightarrow x=5.
\end{align*}
Vậy phương trình có nghiệm $x=5.$
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Đinh Thiện Lý, Hồ Chí Minh 17-18]%[BanNt]%[0D3B1-3]
Giải phương trình $(x-3)^4-3(x-3)^2-4=0$.
\loigiai{
Ta có $(x-3)^4-3(x-3)^2-4=0\Leftrightarrow \hoac{&(x-3)^2=-1\,(\text{loại})\\&(x-3)^2=4}\Leftrightarrow \hoac{&x-3=2\\&x-3=-2}\Leftrightarrow \hoac{&x=5\\&x=1.}$\\
Vậy phương trình có nghiệm $x=5$ hoặc $x=1$.
}
\end{bt}

\begin{bt}%[Đề HK1, Đinh Thiện Lý, Hồ Chí Minh 17-18]%[BanNt]%[0D3B2-6]
Tìm $m$ để phương trình $m^2x-6=4x+3m$ nhận mọi $x\in \mathbb{R}$ là nghiệm.
\loigiai{
Ta có $m^2x-6=4x+3m\Leftrightarrow(m^2-4)x=3m+6$.\\
Điều kiện để phương trình có nghiệm với mọi $x$ thuộc $\mathbb{R}$ là \\
$$\heva{&m^2-4=0\\&3m+6=0}\Leftrightarrow \heva{&\hoac{&m=2\\&m=-2}\\&m=-2}\Leftrightarrow m=-2.$$\\
Vậy $m=-2$ thỏa mãn yêu cầu bài toán.
}
\end{bt}

\begin{bt}%[Đề HK1, Đinh Thiện Lý, Hồ Chí Minh 17-18]%[BanNt]%[0D3K2-5]
Xác định $m$ để phương trình $(m+2)x^2-2(m-1)x+m+5=0$ có hai nghiệm phân biệt thỏa mãn $10x_1x_2-3(x_1+x_2)=0.$

\loigiai{
Phương trình đã cho có hai nghiệm phân biệt khi và chỉ khi\\
$$\heva{&m+2\ne 0\\&\Delta'=(m-1)^2-(m+2)(m+5)>0}\Leftrightarrow \heva{&m\ne-2\\&-9m-9>0} \Leftrightarrow \heva{&m\ne -2\\& m<-1.}$$\\
Lại có  $10x_1x_2-3(x_1+x_2)=0\Leftrightarrow 10\cdot \dfrac{m+5}{m+2}-3\cdot\dfrac{2(m-1)}{m+2}\Leftrightarrow m=-14$ (thỏa mãn).\\
Vậy $m=-14$ thỏa mãn yêu cầu bài toán.
}
\end{bt}

\begin{bt}%[Đề HK1, Đinh Thiện Lý, Hồ Chí Minh 17-18]%[BanNt]%[0D6B2-2]
Cho $\cot x =-5$ với $90^{\circ}<x<180^{\circ}$. Tính $\sin x,\, \cos x,\, \tan x.$

\loigiai{
Theo bài $90^{\circ}<x<180^{\circ} \Rightarrow \sin x>0,\, \cos x< 0.$\\
Ta có $\cot x= -5 \Rightarrow \tan x=-\dfrac{1}{5}$.\\
$1+\cot ^2 x= \dfrac{1}{\sin ^2 x} \Rightarrow \sin x =\sqrt{\dfrac{1}{1+\cot ^2 x}}=\sqrt{\dfrac{1}{1+5^2}}=\dfrac{1}{\sqrt{26}}$.\\
$\sin ^2 x+\cos ^2 x=1 \Rightarrow \cos x=-\sqrt{1-\sin ^2 x}=-\dfrac{5}{\sqrt{26}}$.

}
\end{bt}

\begin{bt}%[Đề HK1, Đinh Thiện Lý, Hồ Chí Minh 17-18]%[BanNt]%[0H2B2-1]
Cho hình thoi $ABCD$ có độ dài cạnh bằng $a$ và $\widehat{ADC}=60^{\circ}$. Gọi $M$ là trung điểm của $CD$. Tính
\begin{enumerate}
\item $\overrightarrow{DA}\cdot \overrightarrow{DC}.$
\item $\overrightarrow{MA}\cdot\overrightarrow{CB}.$
\end{enumerate}

\loigiai{
\begin{center}
\begin{tikzpicture}[scale=0.7]
\tkzDefPoints{0/0/D, 6/0/C}
\tkzDefPointBy[rotation=center D angle 60](C)
\tkzGetPoint{A}
\tkzDefPointBy[translation= from D to C](A) \tkzGetPoint{B}
\tkzInterLL(A,C)(B,D) \tkzGetPoint{O}
\tkzDefMidPoint(C,D) \tkzGetPoint{M}
\tkzDefMidPoint(A,B) \tkzGetPoint{N}
\tkzDrawPoints(A, B, C, D, O, M, N)
\tkzLabelPoints[above](A,B, N)
\tkzLabelPoints[below](D,C,M)
\tkzLabelPoints[left ](O)
\tkzDrawSegments(A,B B,C C,D D,A A,C B,D A,M M,N)

\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Ta có $\overrightarrow{DA} \cdot \overrightarrow{DC}= DA\cdot DC \cdot \cos 60^{\circ}= a\cdot a \cdot \cos 60^{\circ}=\dfrac{a^2}{2}.$
\item Có $\triangle ADC$ là tam giác đều cạnh $a$ (do $DA=DC,\, \widehat{ADC}=60^{\circ}$), suy ra $AM=\dfrac{a\sqrt{3}}{2}$, $(\overrightarrow{MA},\overrightarrow{CB})=\widehat{AMN}=\widehat{DAM}=30^{\circ}$. (với $N$ là giao điểm của $MO$ và $AB$)\\
Khi đó $\overrightarrow{MA} \cdot \overrightarrow{CB}=MA\cdot CB \cdot \cos 30^{\circ}=\dfrac{a\sqrt{3}}{2}\cdot a \cdot \cos 30^{\circ}=\dfrac{3a^2}{4}$. 
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Đinh Thiện Lý, Hồ Chí Minh 17-18]%[BanNt]%[0H1K4-3][0H2B1-4]
Trong mặt phẳng $Oxy$, cho $A(-2;4),\, B(3;5),\, C(2;4)$.
\begin{enumerate}
\item Tính số đo $\widehat{ABC}$.
\item Xác định tọa độ chân đường cao $H$ kẻ từ $B$ của $\triangle ABC$.
\item Tìm tọa độ điểm $M$ sao cho $C$ là trực tâm của $\triangle ABM$.
\end{enumerate}
\loigiai{
\begin{center}
    		\begin{tikzpicture}[scale=0.8]
    		\tkzInit[xmin=-4.5,xmax=4 ,ymin=-1.4 ,ymax=4.2]
			\tkzClip[space=.4]
    		\tkzDefPoints{0/4/A,-2/-1/B,4/-1/M}
    		\tkzDrawAltitude(A,M)(B)\tkzGetPoint{N}
    		\tkzDrawAltitude(B,M)(A)\tkzGetPoint{H}
    		\tkzInterLL(A,H)(B,N) \tkzGetPoint{C}
    		%\tkzDefMidPoint(B,C)
    		%\tkzGetPoint{M}
    		%\tkzInCenter(A,B,C)\tkzGetPoint{I}
    		\tkzDrawPoints[fill=black](A,B,M,H,C)
    		\tkzDrawSegments(A,B B,M A,M)
    		\tkzMarkRightAngle(A,H,M)
    		\tkzMarkRightAngle(B,N,M)
    		%\tkzMarkSegments[mark=||](M,B M,C)
    		%\tkzLabelPoints[right](I)
    		\tkzLabelPoints[left](C)
    		\tkzLabelPoints[above](A)
    		\tkzLabelPoints[below](B,M,H)
    	 \end{tikzpicture}
    	\end{center}
\begin{enumerate}
\item Ta có $\overrightarrow{BA}=(-5;-1)\Rightarrow BA=\sqrt{26}$; $\overrightarrow{BC}=(-1;-1)\Rightarrow BC=\sqrt{2}$. Khi đó\\
$$\cos \widehat{ABC}=\dfrac{\overrightarrow{BA}\cdot\overrightarrow{BC}}{BA\cdot BC}=\dfrac{5+1}{\sqrt{26}\cdot\sqrt{2}}=\dfrac{6}{\sqrt{52}}\Rightarrow \widehat{ABC}=\arccos \left(\dfrac{6}{\sqrt{52}}\right)\approx 34^{\circ}.$$
\item 
 Giả sử $H(x;y)$. Khi đó $\overrightarrow{BH}=(x-3; y-5),\, \overrightarrow{AH}=(x+2; y-4),\, \overrightarrow{AC}=(4;0)$.\\
Ta có $H$ là chân đường cao kẻ từ $B$ của $\triangle ABC$ khi và chỉ khi
$$\heva{&\overrightarrow{BH}\cdot \overrightarrow{AC}=0\\& \overrightarrow{AH}, \overrightarrow{AC}\, \text{cùng phương}}\Leftrightarrow \heva{&4(x-3)=0\\&y-4=0}\Leftrightarrow \heva{&x=3\\&y=4.}$$
Vậy $H(3;4)$ thỏa mãn yêu cầu bài toán.
\item Giả sử $M(a;b)$. Ta có $\overrightarrow{BM}=(a-3; b-5),\, \overrightarrow{AM}=(a+2;b-4),\, \overrightarrow{AC}=(4;0)$.\\
$C$ là trực tâm của $\triangle ABM \Leftrightarrow \heva{&\overrightarrow{AC} \cdot \overrightarrow{BM}=0\\&\overrightarrow{BC} \cdot \overrightarrow{AM}=0} \Leftrightarrow \heva{&4(a-3)=0\\&-a-2-b+4=0} \Leftrightarrow \heva{&a=3\\&b=-1.}$\\
Vậy $M(3;-1)$ thỏa mãn yêu cầu bài toán.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Đinh Thiện Lý, Hồ Chí Minh 17-18]%[BanNt]%[0D4K1-5]
Điền chủ cho phép con trai rào một phần đất tùy ý trên cánh đồng để trồng rau hữu cơ, với điều kiện dùng đúng $600 \mathrm{m}^2$. Để canh tác, người con dự tính cải tạo thửa đất hình chữ nhật nhưng chưa chọn được kích thước ưng ý. Để giúp anh ta dự toán, em hãy cho biết số tiền tối thiểu phải trả để dựng hàng rào quanh thửa đất, biết chi phí cho cạnh dọc xa lộ là $ 2\mathdollar $ mỗi mét, và cho ba cạnh còn lại là $1\$ $ mỗi mét.
\loigiai{
\begin{center}
    		\begin{tikzpicture}[scale=0.8]
    		\tkzInit[xmin=-3,xmax=4 ,ymin=-0.5 ,ymax=4]
			\tkzClip[space=.4]
    		\tkzDefPoints{-1/0/A,-1/3/B,3/3/C, 3/0/D}
    		\tkzDrawPoints(A, B, C, D)
    		\tkzDrawSegments(A,B B,C D,C D,A)
    		\tkzLabelSegment[above](C,B){$x$ (m)}
    		\tkzLabelSegment[left](A,B){$\dfrac{600}{x}$ (m)}
    		\tkzMarkRightAngle(A,B,C)
    		\tkzMarkRightAngle(B,A,D)
    		%\tkzMarkSegments[mark=||](M,B M,C)
    		%\tkzLabelPoints[right](I)
    		\tkzLabelPoints[left](A, B)
    		\tkzLabelPoints[right](C, D)
    		%\tkzLabelPoints[below](B)
    	 \end{tikzpicture}
    	\end{center}
Gọi $x$ (m) là chiều dài hàng rào cạnh dọc xa lộ ($x>0$).\\
 Theo bài, chi phí phải trả để dựng hàng rào cho cạnh dọc xa lộ là $2\$$ mỗi mét, và cho ba cạnh còn lại là $1\$$ mỗi mét, nên số tiền phải trả để dựng hàng rào quanh thửa đất được biểu diễn bởi hàm số $$f(x)=2x+x+\dfrac{600}{x}+\dfrac{600}{x}=3x+\dfrac{1200}{x}.$$
Ta có $f(x)=3x+\dfrac{1200}{x}\geq 2\sqrt{3x\cdot\dfrac{1200}{x}}=2\sqrt{3600}=120.$\\
Như vậy chi phí tối thiểu là $120\$$, khi độ dài cạnh dọc xa lộ là $x=20$ m.
}
\end{bt}
