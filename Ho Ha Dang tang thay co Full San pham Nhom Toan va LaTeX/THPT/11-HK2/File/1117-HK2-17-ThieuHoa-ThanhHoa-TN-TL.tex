\begin{bt}%[HK2, Triệu Hóa, Thanh Hóa, 17]%[1D4G2-4]%[Trần Quang Thạnh]
	Tính $\lim\limits_{x\to 0}\dfrac{\sqrt{2x+1}-\sqrt[3]{3x+1}}{x^2}$.
	\loigiai{
		Ta có
		\begin{align*}
		&\lim\limits_{x\to 0}\dfrac{\sqrt{2x +1} -\sqrt[3]{3x+1}}{x^2}\\
		=&\lim\limits_{x\to 0}\left[\dfrac{\sqrt{2x +1}-(1+x)}{x^2} -\dfrac{\sqrt[3]{3x+1}-(1+x)}{x^2}\right]\\
		=&\lim\limits_{x\to 0}\left[\dfrac{2x +1-x^2-2x-1}{x^2\left(\sqrt{2x +1}+(1+x)\right)} -\dfrac{{3x+1}-x^3-3x^2-3x-1}{x^2\left(\sqrt[3]{(3x+1)^2}+(1+x)\sqrt[3]{3x+1}+(x+1)^2\right)}\right]\\
		=&\lim\limits_{x\to 0}\left[\dfrac{-x^2}{x^2\left(\sqrt{2x +1}+(1+x)\right)} +\dfrac{x^3+3x^2}{x^2\left(\sqrt[3]{(3x+1)^2}+(1+x)\sqrt[3]{3x+1}+(x+1)^2\right)}\right]\\
		=&\lim\limits_{x\to 0}\left[\dfrac{-1}{\sqrt{2x +1}+(1+x)} +\dfrac{x+3}{\sqrt[3]{(3x+1)^2}+(1+x)\sqrt[3]{3x+1}+(x+1)^2}\right]\\
		=&1-\dfrac{1}{2}=\dfrac{1}{2}.
		\end{align*}
	}
\end{bt}

\begin{bt}%[HK2, Triệu Hóa, Thanh Hóa, 17]%[1D4B3-3]%[Trần Quang Thạnh]

	}

\begin{bt}%[HK2, Triệu Hóa, Thanh Hóa, 17]%[1D4K3-6]%[Trần Quang Thạnh]
	Chứng minh rằng phương trình $-x^3+4x^2+x-2=0$ có ba nghiệm phân biệt trên khoảng $(-2;5)$.
	\loigiai{
		Xét hàm số $f(x)=-x^3+4x^2+x-2$ xác định và liên tục trên $\mathbb{R}$ nên liên tục trên $[-2;5]$.\\
		Ta có $f(-2)=20, f(0)=-2, f(2)=8, f(5)=-22$ nên 
		\begin{itemize}
			\item $f(-2) \cdot f(0)<0$, suy ra phương trình có ít nhất 1 nghiệm $x_1\in (-2;0)$;
			\item $f(0) \cdot f(2)<0$, suy ra phương trình có ít nhất 1 nghiệm $x_2\in (0;2)$;
			\item $f(2) \cdot f(5)<0$, suy ra phương trình có ít nhất 1 nghiệm $x_3\in (2;5)$.
		\end{itemize}
		Vì $-2<x_1<0<x_2<2<x_3<5$ nên ba nghiệm $x_1, x_2, x_3$ phân biệt.\\
		Vậy phương trình đã cho có ba nghiệm phân biệt trên khoảng $(-2;5)$.
		
	}
\end{bt}

\begin{bt}%[HK2, Triệu Hóa, Thanh Hóa, 17]%[1H3K3-4]%[Trần Quang Thạnh]
	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình vuông tâm $O$ cạnh $a$, $SA=a\sqrt{6}$ và $SA$ vuông góc mặt phẳng $(ABCD)$. Gọi $H, K$ lần lượt là chân đường cao hạ từ đỉnh $A$ của các tam giác $SAB$ và $SAD$.
	\begin{enumerate}
		\item Tính góc hợp bởi đường thẳng $SC$ với mặt phẳng $(ABCD)$.
		\item Chứng minh rằng đường thẳng $SC$ vuông góc với mặt phẳng $(AHK)$.
		\item Tính theo $a$ thiết diện của hình chóp khi cắt bởi mặt phẳng $(AHK)$.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=0.7]
			%Định nghĩa các điểm
			\tkzDefPoints{0/0/B, 8/0/C}
			\path(B)++(40:4)coordinate(A)++(0:8)coordinate(D);
			\path(A)++(90:6)coordinate(S);
			\tkzInterLL(A,C)(B,D) \tkzGetPoint{O}
			\tkzDefPointBy[homothety=center S ratio 6/7 ](B) \tkzGetPoint{H}
			\tkzDefPointBy[homothety=center S ratio 6/7 ](D) \tkzGetPoint{K}
			\tkzInterLL(S,O)(H,K) \tkzGetPoint{I}
			\tkzInterLL(A,I)(S,C) \tkzGetPoint{J}
			%Gán tên các điểm
			\tkzLabelPoint[below right](D){$D$}
			\tkzLabelPoint[below right](C){$C$}
			\tkzLabelPoints[below](B, A,O)
			\tkzLabelPoints[above](S,K)
			\tkzLabelPoints[right](J)
			\tkzLabelPoints[left](H,I)
			
			%Nối các điểm
			\tkzDrawSegments[style=dashed](S,A A,B A,D A,C B,D A,H A,K H,K S,O S,J)
			\tkzMarkRightAngle(D,A,S)
			\tkzDrawSegments(S,B S,D B,C C,D S,C H,J K,J)
			\tkzDrawPoints(S,A,B,C,D,O,H,K,I,J)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Ta có $\left(SC,(ABCD)\right)=(SC,AC)=\widehat{SCA}$ và 
			$\tan\widehat{SCA}=\dfrac{SA}{AC}=\sqrt{3}$ nên $\left(SC,(ABCD)\right)=60^{\circ}$.
			\item Vì $AH \perp SB$ và $AH\perp BC$ nên $AH \perp SC$. Tương tự $AK\perp SC$. Do đó $SC\perp (AHK)$.
			\item Gọi $I$ là giao điểm của $HK$ và $SO$, $J$ là giao điểm của $AI$ và $SC$.\\
			Thiết diện cần tìm là tứ giác $AHJK$.\\
			Ta có $AH=\dfrac{SA\cdot AB}{SB}=\dfrac{a\sqrt{42}}{7}$ và $AH\perp (SBC)$ nên $AH \perp HJ$.\\
			Tương tự $AK \perp KJ$ và $AK=\dfrac{a\sqrt{42}}{7}$.\\
			Từ đó suy ra hai tam giác vuông $AHJ$ và $SKJ$ bằng nhau, do đó $$S_{AHJK}=S_{AHJ}+S_{AKJ}=2S_{AHJ}.$$
			Ta có $SH^2=SA^2-AH^2 \Rightarrow SH=\dfrac{6a\sqrt{7}}{7}$.\\
			Vì $SC\perp (AHK)$ nên $SC\perp HJ$, do đó tam giác $SHJ$ vuông tại $J$. Suy ra hai tam giác vuông $SBC$ và $SJH$ đồng dạng nên $$\dfrac{JH}{BC}=\dfrac{SH}{SC} \Rightarrow JH=\dfrac{SH\cdot BC}{SC}=\dfrac{3a\sqrt{14}}{14}.$$
			Vậy $S_{AHJK}=2S_{AHJ}=AH\cdot JH=\dfrac{3a^2\sqrt{3}}{7}$.
			
		\end{enumerate}
	}
\end{bt}
