\begin{bt}%[HK2 (2016-2017), Sở GD \& ĐT Bắc Giang]%[1D4B2-2]%[1D4B2-6]%[Đào-V- Thủy]
	Tính các giới hạn
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\lim\limits_{x\to 2}\dfrac{x+1}{x^2+3}$.
			\item $\lim\limits_{x\to +\infty}\dfrac{3x-1}{\sqrt{9x^2+2x-1}}$.
		\end{enumerate}
	\end{multicols}
\loigiai{
\begin{enumerate}
	\item $\lim\limits_{x\to 2}\dfrac{x+1}{x^2+3}=\dfrac{3}{7}$.
	\item $\lim\limits_{x\to +\infty}\dfrac{3x-1}{\sqrt{9x^2+2x-1}}=\lim\limits_{x\to +\infty}\dfrac{x\left(3-\dfrac{1}{x}\right)}{x\sqrt{9+\dfrac{2}{x}-\dfrac{1}{x^2}}}=\lim\limits_{x\to +\infty}\dfrac{3-\dfrac{1}{x}}{\sqrt{9+\dfrac{2}{x}-\dfrac{1}{x^2}}}=1$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[HK2 (2016-2017), Sở GD \& ĐT Bắc Giang]%[1D4B3-3]%[Đào-V- Thủy]
	Xét tính liên tục của hàm số $f(x)=\heva{&\dfrac{3x^2-10x-25}{2x-10}&\text{ khi }x>5\\ &x^2-4x+5&\text{ khi }x\le 5}$ tại $x=5$.
	\loigiai{
\begin{itemize}
	\item $\lim\limits_{x\to 5^+}f(x)=\lim\limits_{x\to 5^+}\dfrac{3x^2-10x-25}{2x-10}=\lim\limits_{x\to 5^+}\dfrac{(x-5)(3x+5)}{2(x-5)}=\lim\limits_{x\to 5^+}\dfrac{3x+5}{2}=10$.
	\item $\lim\limits_{x\to 5^-}f(x)=\lim\limits_{x\to 5^-}\left(x^2-4x+5\right)=10$.
	\item $f(5)=10$.
\end{itemize}	
Ta thấy $\lim\limits_{x\to 5^+}f(x)=\lim\limits_{x\to 5^-}f(x)=f(5)$. Vậy hàm số đã cho liên tục tại $x=5$.
}
\end{bt}

\begin{bt}%[HK2 (2016-2017), Sở GD \& ĐT Bắc Giang]%[1D5K2-1]%[Đào-V- Thủy]
	Cho hàm số $y=\dfrac{1}{3}x^3-(2m+1)x^2+m^2x-4$, với $m$ là tham số. Tìm $m$ để $y'\ge 0,\ \forall x\in\mathbb{R}$.
	\loigiai{
Tập xác định $\mathscr{D}=\mathbb{R}$. Ta có $y'=x^2-2(2m+1)x+m^2$.\\
$y'$ là một tam thức bậc hai có $\Delta'=(2m+1)^2-m^2=3m^2+4m+1$.\\
Do đó $y'\ge 0,\ \forall x\in\mathbb{R}\Leftrightarrow \Delta'\le 0\Leftrightarrow 3m^2+4m+1\le 0\Leftrightarrow -1\le m\le -\dfrac{1}{3}$.	
}
\end{bt}

\begin{bt}%[HK2 (2016-2017), Sở GD \& ĐT Bắc Giang]%[1D5K2-3]%[Đào-V- Thủy]
	Cho hàm số $y=\dfrac{3x+1}{1-x}$ có đồ thị $(C)$. Viết phương trình tiếp tuyến của $(C)$ biết tiếp tuyến song song với đường thẳng $x-4y-5=0$.
	\loigiai{
Gọi $(x_0;y_0)$ là tọa độ tiếp điểm.\\		
Ta có $y'=\dfrac{4}{(1-x)^2}$.\\
Vì tiếp tuyến song song với đường thẳng $d\colon x-4y-5=0\Leftrightarrow d\colon y=\dfrac{1}{4}x-\dfrac{5}{4}$.\\
Suy ra $\dfrac{4}{(1-x_0)^2}=\dfrac{1}{4}\Rightarrow (1-x_0)^2=16\Rightarrow \hoac{&x_0=-3&\Rightarrow y_0=-2\\ &x_0=5&\Rightarrow y_0=-4.}$\\
Phương trình tiếp tuyến của đồ thị $(C)$ tại điểm $(-3;-2)$ là
$$y=\dfrac{1}{4}(x+3)-2\Leftrightarrow x-4y-5=0\quad (\text{loại vì trùng với }d).$$
Phương trình tiếp tuyến của đồ thị $(C)$ tại điểm $(5;-4)$ là
$$y=\dfrac{1}{4}(x-5)-4\Leftrightarrow x-4y-21=0\quad (\text{thỏa mãn}).$$
}
\end{bt}

\begin{bt}%[HK2 (2016-2017), Sở GD \& ĐT Bắc Giang]%[1D5B3-1]%[Đào-V- Thủy]
	Xét chuyển động có phương trình $s(t)=20\sin\left(\pi t+\dfrac{\pi}{6}\right)$ ($t$ tính bằng giây). Tính vận tốc của chuyển động tại thời điểm $t=4$ giây.
	\loigiai{
Vận tốc của chuyển động tại thời điểm $t$ là
	$$v(t)=s'(t)=20\cos\left(\pi t+\dfrac{\pi}{6}\right)\cdot \left(\pi t+\dfrac{\pi}{6}\right)'=20\pi\cos\left(\pi t+\dfrac{\pi}{6}\right).$$
Vậy vận tốc của chuyển động tại thời điểm $4$ giây bằng
$$v(4)=20\pi \cos\left(4\pi+\dfrac{\pi}{6}\right)=10\pi\sqrt 3.$$	
}
\end{bt}

\begin{bt}%[HK2 (2016-2017), Sở GD \& ĐT Bắc Giang]%[1H3B3-2]%[1H3K3-4]%[1H3K4-3]%[Đào-V- Thủy]
	Cho hình chóp $S.ABC$ có đáy là tam giác vuông tại $B$, $AB=a$, $AC=2a$, $SA\perp (ABC)$, $SA=2a$.
	\begin{enumerate}
		\item Chứng minh các mặt bên của hình chóp là các tam giác vuông.
		\item Gọi $M$ là điểm thuộc cạnh $AC$ sao cho $AM=\dfrac{2a}{3}$. Xác định thiết diện của hình chóp cắt bởi mặt phẳng $(P)$ đi qua $M$ và vuông góc với $AB$. Tính diện tích thiết diện theo $a$.
		\item Tính cosin của góc giữa hai mặt phẳng $(SAC)$ và $(SBC)$.
	\end{enumerate}
\loigiai{
\begin{center}
	\begin{tikzpicture}[line width=0.8, scale=1.0]
	\coordinate (A) at (0,0);
	\coordinate (B) at (4,-2);
	\coordinate (C) at (6,0);
	\coordinate (S) at (0,5);
	\coordinate (M) at ($(A)!1/3!(C)$);
	\coordinate (N) at ($(A)!1/3!(B)$);
	\coordinate (P) at ($(S)!1/3!(B)$);
	\coordinate (Q) at ($(S)!1/3!(C)$);
	\coordinate (H) at ($(S)!0.6!(B)$);
	\coordinate (K) at ($(S)!.55!(C)$);
	\draw (S)--(A)--(B)--(C)--(S)--(B) (N)--(P)--(Q) (A)--(H)--(K);
	\draw[dashed] (Q)--(M)--(N) (C)--(A)--(K);
	\tkzMarkRightAngle(S,H,A) 
	\tkzMarkRightAngle(S,K,A)
	\draw (S) node[above] {\footnotesize $S$} (A) node[left] {\footnotesize $A$} (B) node[below] {\footnotesize $B$} (C) node[right] {\footnotesize $C$} ($(M)+(.1,0)$) node[below] {\footnotesize $M$} (N) node[below] {\footnotesize $N$} (P) node[left] {\footnotesize $P$} ($(Q)+(0,.15)$) node[right] {\footnotesize $Q$} (H) node[right] {\footnotesize $H$} ($(K)+(0,.15)$) node[right] {\footnotesize $K$};
	\end{tikzpicture}
\end{center}
\begin{enumerate}
	\item Từ giả thiết có $SA\perp (ABC)$. Suy ra $\heva{&SA\perp AB\\ &SA\perp AC.}$\\
	Vậy $\triangle SAB$, $\triangle SAC$ là các tam giác vuông tại $A$.
	\item Ta có $\heva{&BC\perp AB\\ &BC\perp SA}\Rightarrow BC\perp (SAB)\Rightarrow BC\perp SB$. Vậy tam giác $SBC$ vuông tại $B$.\\
	Vì $(P)$ đi qua $M$, $(P)$ vuông góc với $AB$ nên $(P)$ song song với $BC$.\\
	Suy ra $(P)\cap (ABC)=MN$ với $N\in AB$, $MN\parallel BC$.\\
	Tương tự $(P)$ đi qua $N$, $(P)$ vuông góc với $AB$ nên $(P)$ song song với $SA$.\\
	Suy ra $(P)\cap (SAB)=NP$ với $P\in SB$, $NP\parallel SA$.\\
	Vì $(P)\parallel BC$ nên $(P)\cap (SBC)=PQ$ với $PQ\parallel BC$, $Q\in SC$.\\
	$(P)\cap (SAC)=QM$. Do $SA\perp BC$ nên $MN\perp NP$.\\
	Vậy thiết diện của hình chóp cắt bởi mặt phẳng $(P)$ là hình chữ nhật $MNPQ$.\\
	Có $MN=\dfrac{1}{3}BC=\dfrac{1}{3}a\sqrt 3$, $NP=\dfrac{2}{3}SA=\dfrac{4a}{3}$.\\
	Vậy $S_{MNPQ}=\dfrac{a\sqrt 3}{3}\cdot \dfrac{4a}{3}=\dfrac{4a^2\sqrt 3}{9}$.
	\item Gọi $H$, $K$ lần lượt là hình chiếu vuông góc của $A$ trên $SB$, $SC$.\\
	Ta có $\heva{&AH\perp SB\\ &AH\perp BC}\Rightarrow AH\perp (SBC)\Rightarrow AH\perp SC$.\\
	và $\heva{&SC\perp AH\\ &SC\perp AK}\Rightarrow SC\perp (AHK)\Rightarrow SC\perp HK$.\\
	Vì $\heva{&(SAC)\cap (SBC)=SC\\ &AK\subset (SAC), AK\perp SC\\ &HK\subset (SBC), HK\perp SC}$ Suy ra góc giữa $(SAC)$ và $(SBC)$ là $\widehat{AKH}$.\\
	Có $\dfrac{1}{AH^2}=\dfrac{1}{AS^2}+\dfrac{1}{AB^2}=\dfrac{1}{4a^2}+\dfrac{1}{a^2}=\dfrac{5}{4a^2}\Rightarrow AH=\dfrac{2a}{\sqrt 5}$.\\
	Tam giác $SAC$ vuông cân tại $A$, cạnh $AS=AC=2a$. Suy ra $AK=a\sqrt 2\Rightarrow HK=\dfrac{a\sqrt 6}{\sqrt 5}$.\\
	Vậy $\cos\widehat{AKH}=\dfrac{HK}{AK}=\sqrt{\dfrac{3}{5}}$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[HK2 (2016-2017), Sở GD \& ĐT Bắc Giang]%[1D3G4-7][Đào-V- Thủy]
	Hàng tháng ông An gửi vào ngân hàng một số tiền như nhau là $5000000$ đồng (vào ngày đầu mỗi tháng) với lãi suất $0{,}5\%$ một tháng, biết tiền lãi của tháng trước được nhập vào tiền gốc của tháng sau. Hỏi sau $36$ tháng ông An nhận được số tiền vốn và lãi là bao nhiêu? (làm tròn đến hàng đơn vị).
	\loigiai{
Gọi $a$ là số tiền ông An gửi vào hàng tháng, $r$ là lãi suất trên một tháng và $P_n$ là số tiền vốn và lãi ông An nhận được sau $n$ tháng.
\begin{itemize}
	\item Sau một tháng, ông An có số tiền là $P_1=a+ar=a(1+r)$.
	\item Đầu tháng thứ hai, ông An có số tiền là $P_1+a=a(1+r)+a$.
	\item Sau hai tháng, ông An có số tiền là $P_2=a(1+r)+a+\left[a(1+r)+a\right]r=a\left[(1+r)^2+(1+r)\right]$.
	\item Cuối tháng thứ $36$, ông An có số tiền là
	\begin{align*}
	P_{36}&=a\left[(1+r)^{36}+(1+r)^{35}+\ldots+(1+r)\right]\\
	&=a(1+r)\dfrac{(1+r)^{36}-1}{r}\\
	&=5000000\cdot (1+0{,}005)\cdot\dfrac{(1+0{,}005)^{36}-1}{0{,}005}\\
	&\approx 197 663 927 \quad \text{(đồng)}.
	\end{align*}
\end{itemize}
}
\end{bt}
