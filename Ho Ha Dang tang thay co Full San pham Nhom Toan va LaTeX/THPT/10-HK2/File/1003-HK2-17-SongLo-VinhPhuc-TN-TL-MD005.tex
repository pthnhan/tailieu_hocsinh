\noindent\textbf{I. PHẦN TRẮC NGHIỆM}
\begin{ex}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc]%[0H3B2]%[Ngô Quang Anh]
	Trong mặt phẳng với hệ tọa độ $Oxy,$ cho đường tròn $(C):x^2+y^2+4x-2y-5=0.$  Phát biểu nào sau đây \textbf{sai}?
	\choice
	{Đường tròn có tâm $I\left(-2;1\right)$}
	{\True Đường tròn có bán kính $R=10$}
	{Đường tròn đi qua điểm $A\left(1;2\right)$}
	{Đường tròn không đi qua điểm $B\left(0;5\right)$}
	\loigiai{
		Ta có $R=\sqrt{a^2+b^2-c}=\sqrt{(-2)^2+(1)^2+5}=\sqrt{10}.$}
\end{ex}
\begin{ex}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc]%[0D6B2]%[Ngô Quang Anh]
	Cho $\dfrac{\pi}{2}<\alpha <\pi.$ Chọn kết quả đúng
	\choice
	{$\cos \left(\alpha+\pi \right)<0$}
	{$\tan \left(\alpha-\pi \right)>0$}
	{\True$\sin \left(\alpha+\pi \right)<0$}
	{$\cot \left(\alpha+\pi \right)>0$}
	\loigiai{
		Ta có $\sin \left(\alpha+\pi \right)=-\sin \alpha.$ 
		Vì $\dfrac{\pi}{2}<\alpha <\pi $ nên $\sin \alpha >0\Rightarrow \sin \left(\alpha+\pi \right)<0.$}
\end{ex}

\begin{ex}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc]%[0D4B2]%[Ngô Quang Anh]
	Tập xác định của hàm số $y=\sqrt{x+6}+\sqrt{3-x}$ là
	\choice
	{$\mathscr{D}=\left(-\infty;3\right]$}
	{$\mathscr{D}=\left(-\infty;-6\right]\cup \left[3;+\infty \right)$}
	{\True $\mathscr{D}=\left[-6;3\right]$}
	{$\mathscr{D}=\left[3;6\right]$}
	\loigiai{
		Hàm số xác định khi $\left\{\begin{matrix}
		x+6\ge 0 \\
		3-x\le 0 \\
		\end{matrix}\right.\Leftrightarrow-6\le x\le 3.$}
\end{ex}
\begin{ex}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc]%[0D4B2]%[Ngô Quang Anh]
	Với giá trị nào của $m$ thì bất phương trình $2x-m<mx$ nghiệm đúng với mọi $x.$ 
	\choice
	{\True $m=0$}
	{$m=2$}
	{$m=-2$}
	{$m=-1$}
	\loigiai{
		$$2x-m<mx\Leftrightarrow \left(2-m\right)x<m.$$
		Với $ m=2\Rightarrow 0x<2 $ đúng với mọi $x.$\\
		Vậy bất phương trình đã cho nghiệm đúng với mọi $x$ khi $m=2.$}
\end{ex}

\begin{ex}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc]%[0D4B5]%[Ngô Quang Anh]

\noindent\textbf{II. PHẦN TỰ LUẬN} 
\begin{bt}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc]%[0D4B5]%[Ngô Quang Anh]
	 Giải bất phương trình $2x^2-5x-3>(x+1)(x-3).$		
	\loigiai{ $2x^2-5x-3>(x+1)(x-3) \Leftrightarrow x^2-3x>0 \Leftrightarrow \hoac{&x<0\\ &x>3.} $\\Vậy $ S=(-\infty;0)\cup(3;+\infty). $}		
\end{bt}
	\begin{bt}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc]%[0D4K5]%[Ngô Quang Anh]
		Giải bất phương trình $\dfrac{x^2-12x+32}{10-2x}\le 0.$
			\loigiai{\begin{enumerate}[+]
					\item $10-2x=0 \Leftrightarrow x=5$.
					\item $x^2-12x+32=0 \Leftrightarrow \hoac{&x=4\\&x=8.}$
				\end{enumerate}
				Bảng xét dấu 
				\begin{center}
					\begin{tikzpicture}
					\tkzTabInit[nocadre=false, lgt=3, espcl=2]{$x$ /1,$10-2x$ /1, $x^2-12x+32$/1, $ VT $ /1}{$-\infty$,$4$,$5$,$8$,$+\infty$}
					\tkzTabLine{,+,$|$,+,$0$,-,$|$,-,}
					\tkzTabLine{,+,$0$,-,$|$,-,$0$,+,}
					\tkzTabLine{,+,$0$,-,$||$,+,$0$,-,}
					\end{tikzpicture}
				\end{center}
				Dựa vào bảng xét dấu ta có tập nghiệm của bất phương trình là $S=\left[4;5\right)\cup \left[8;+\infty \right).$}
	\end{bt}

	\begin{bt}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc]%[0D4K5]%[Ngô Quang Anh]
		Giải bất phương trình $\left| x+2\right| \le 4x+3.$
			\loigiai{$\left| x+2\right| \le 4x+3 \Leftrightarrow \heva{&4x+3\ge 0\\&16x^2+24x+9\ge x^2+4x+4}$
				$$\Leftrightarrow \heva{&x\ge -\dfrac{3}{4}\\&15x^2+20x+5\ge 0}\Leftrightarrow \heva{&x\ge -\dfrac{3}{4}\\ &\hoac{&x\le -1\\& x\ge -\dfrac{1}{3}}} \Leftrightarrow x\ge -\dfrac{1}{3}\cdot $$
				Vậy $S=\left[-\dfrac{1}{3};+\infty \right).$}		
	\end{bt}
	
\begin{bt}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc]%[0D6B2]%[Ngô Quang Anh]
	Cho  $\tan \alpha=2$ với $0<\alpha<\dfrac{\pi}{2}$. Tính các giá trị lượng giác còn lại.
	\loigiai{Vì $0<\alpha <\dfrac{\pi}{2}\Rightarrow \cos \alpha >0$.
		Ta có $\dfrac{1}{\cos^2x}=1+\tan^2x=5\Rightarrow \cos x=\dfrac{1}{\sqrt{5}}\cdot $   \\
		$\sin \alpha =\tan \alpha \cdot \cos \alpha =\dfrac{2}{\sqrt{5}};\cot \alpha =\dfrac{1}{\tan \alpha}=\dfrac{1}{2}\cdot $ }
\end{bt}
\begin{bt}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc]%[0H3B1]%[Ngô Quang Anh]
	Trong mặt phẳng $ Oxy $, cho ba điểm $ A(-1;0), B(1;6), C(3;2). $ Viết phương trình tổng quát của đường thẳng $ AB $. Tính khoảng cách từ $ C $ đến $ AB $, khoảng cách này là đại lượng nào trong tam giác.			
	\loigiai{ 
	Ta có $AB:\dfrac{x+1}{2}=\dfrac{y}{6}\Leftrightarrow 3x-y+3=0.$\\
	Khoảng cách từ  $C$ đến đường thẳng $AB$: $d(C,AB)=\dfrac{\left| 3.3-2+3\right|}{\sqrt{3^2+(-1)^2}}=\sqrt{10}.$ Khoảng cách này chính là độ dài đường cao hạ từ đỉnh $C.$ }
\end{bt}

	\begin{bt}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc]%[0H3B2]%[Ngô Quang Anh]
		Trong mặt phẳng $ Oxy $, cho ba điểm $ A(-1;0), B(1;6), C(3;2).$ Viết phương trình đường tròn $ (C) $ có tâm là điểm $ C $ và đi qua đi qua $ A $.
	\loigiai{Ta có $R=AC=\sqrt{(3+1)^2+(2-0)^2}=\sqrt{20}.$ \\
		Suy ra phương trình đường tròn $ (C) $ có tâm là điểm $ C $ và đi qua đi qua $ A $ là $ (x-3)^2 +(y-2)^2=20.$}

\begin{bt}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc]%[0D4K5]%[Ngô Quang Anh]
Tìm $ m $ để bất phương trình $ m(m-4)x^2+2mx+2\ge 0 $ đúng với $\forall x\in \mathbb{R}.$ 
	\loigiai{
		\begin{itemize}
			\item Nếu $m=0$ thì bpt $\Leftrightarrow 2\ge 0$ đúng với mọi $x\in \mathbb{R}\Rightarrow m=0$ thỏa mãn.  
			\item Nếu $m=4$ thì bpt $\Leftrightarrow 8x+2\ge 0 \Leftrightarrow x \ge -\dfrac{1}{4}$ $\Rightarrow m=4$ không thỏa mãn.
			\item Nếu $m\ne 0$ và $m\ne 4$ thì bpt đúng $\forall x\in \mathbb{R}\Leftrightarrow \heva{
			& m(m-4)>0  \\
			&{\triangle}'=m^2-2m(m-4)\le 0}$ 
			$$\Leftrightarrow \left\{\begin{matrix}
			\left[\begin{matrix}
			m>4 \\
			m<0 \\
			\end{matrix}\right. \\
			\left[\begin{matrix}
			m\le 0 \\
			m\ge 8 \\
			\end{matrix}\right. \\
			\end{matrix}\right.\Leftrightarrow \left[\begin{matrix}
			m<0 \\
			m\ge 8. \\
			\end{matrix}\right.$$
			\end{itemize}
	Vậy $\hoac{&m\le 0\\&m\ge 8}$ thỏa yêu cầu bài toán.
		
	}
\end{bt}
