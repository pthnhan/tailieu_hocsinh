\dapso{$2500\left(\dfrac{5\pi}{6}-\sqrt{3} \right) km^2$}
\loigiai{

Tương tự $HA=\sqrt{AC^2-CH^2}=\sqrt{2500-CH^2}$.\\
Mà $AB=HB+HA=\sqrt{7500-CH^2}+\sqrt{2500-CH^2}$\\
$\Leftrightarrow 100^2 =7500 -CH^2 +2500 -CH^2 +2\sqrt{\left(7500-CH^2\right) \left(2500-CH^2\right)}$\\
$\Leftrightarrow CH=25\sqrt{3}$ (km).\\
Suy ra $S_{ABC}=2.S_{ACB} = 2 \cdot \dfrac{1}{2} \cdot CH \cdot AB = 2500\sqrt{3}$ (km$^2$).\\
Xét tam giác $CHB$ có $\sin \widehat{HBC}=\dfrac{CH}{BC}=\dfrac{25\sqrt{3}}{50\sqrt{3}} = \dfrac{1}{2}$, suy ra $\widehat{HBC} = 30^{\circ} \Rightarrow \widehat{DBC} = 60^{\circ}.$\\

	Để sửa một ngôi nhà cần một số thợ làm việc trong một thời gian quy định. Nếu giảm ba người thì thời gian kéo dài sáu ngày. Nếu tăng thêm hai người thì xong sớm hai ngày. Hỏi theo quy định cần bao nhiêu thợ và làm xong trong bao nhiêu ngày? Biết rằng khả năng lao động của mỗi thợ đều như nhau.
	\dapso{$8$ thợ làm trong $6$ ngày}
	\loigiai{
		Gọi $x$ (người) là số thợ theo quy định, $x \in \mathbb{N}^*$.\\
		Gọi $y$ (ngày) là số ngày làm xong việc, $y \in \mathbb{N}^*$.\\
		Sau khi giảm ba người ($x-3$) thì số ngày hoàn thành là $y+6$.\\
		Sau khi tăng thêm hai người ($x+2$) thì số ngày hoàn thành là $y-2$.\\
		Ta có hệ $\heva{&xy=(x-3)(y+6)\\&xy=(x+2)y-2)}$ $\Leftrightarrow \heva{&2x-y=6\\&x-y=-2} \Leftrightarrow \heva{&x=8\\&y=6 } \text{(thỏa mãn).}$
	}
\end{bt}

\begin{bt}%[Lê Mạnh Thắng]%[EX-TS10-ToanTT-2019]%[9D3K6]
	Hai dung dịch có khối lượng tổng cộng bằng $220\,$kg. Lượng muối trong dung dịch I là $5\,$kg, lượng muối trong dung dịch II là $4{,}8\,$kg. Biết nồng độ muối trong dung dịch I nhiều hơn nồng độ muối trong dung dịch II là $1\%$. Tính khối lượng mỗi dung dịch nói trên.
	\dapso{Dung dịch I là $100$ kg; dung dịch II là $120$ kg}
	\loigiai{
		Gọi $x$ (kg), $y$ (kg) lần lượt là khối lượng của dung dịch I và II (điều kiện $0<x<220$, $0<y<220$).\\
		Tổng khối lượng hai dung dịch là \[x+y=220 \Leftrightarrow x=220 -y \tag{1}.\]\\
		Nồng độ muối trong dung dịch I	nhiều hơn nồng độ muối trong dung dịch II là $1\%$, nên ta có
		\[\dfrac{5}{x}-\dfrac{4{,}8}{y}=\dfrac{1}{100} \Leftrightarrow 500y-480x=xy \tag{2}.\]
		Thay $(1)$ vào $(2)$ ta được:
		\[500y-480(220-y)=(220-y)y \Leftrightarrow y^2+760y-105600=0 \Leftrightarrow \hoac{&y=120\\&y=-880\ (\text{loại}).}\]
		Với $y=120 \Rightarrow x=100$.\\
		Vậy khối lượng dung dịch I là $100$ kg, khối lượng dung dịch II là $120$ kg.
	}
\end{bt}

\begin{bt}%[Lê Mạnh Thắng]%[EX-TS10-ToanTT-2019]%[9D3K6]
	Có hai loại quặng sắt: quặng loại A chứa $60\%$ sắt, quặng loại B chứa $50\%$ sắt. Người ta trộn một lượng quặng loại A với một lượng quặng loại B thì được hỗn hợp chứa $\dfrac{8}{15}$ sắt. Nếu lấy tăng hơn lúc đầu là $10$ tấn quặng loại A và lấy giảm hơn lúc đầu là $10$ tấn quặng loại B thì được hỗn hợp quặng chứa $\dfrac{17}{30}$ sắt. Tính khối lượng quặng mỗi loại đem trộn lúc đầu.
	\dapso{loại A: $10$ tấn, loại B: $20$ tấn}
	\loigiai{
		Gọi $x$ (tấn), $y$ (tấn) là khối lượng quặng loại A và loại B ban đầu, điều kiện $x,y>0$.\\
		Khối lượng sắt trong $x$ tấn quặng loại A là $60\%x=\dfrac{3}{5}x$.\\
		Khối lượng sắt trong $y$ tấn quặng loại B là $50\%y=\dfrac{1}{2}y$.\\
		Ban đầu, trộn $x$ tấn quặng loại A với $y$ tấn quặng loại B thì được hỗn hợp chứa $\dfrac{8}{15}$ sắt, suy ra
		\[\dfrac{3}{5}x+\dfrac{1}{2}y=\dfrac{8}{15}(x+y) \Leftrightarrow \dfrac{1}{15}x=\dfrac{1}{30}y \Leftrightarrow 2x=y. \tag{1}\]
		Nếu trộn $x+10$ tấn quặng loại A với $y-10$ tấn quặng loại B thì được hỗn hợp chứa $\dfrac{17}{30}$ sắt, suy ra
		\[\dfrac{3}{5}(x+10)+\dfrac{1}{2}(y-10)=\dfrac{17}{30}(x+10+y-10) \Leftrightarrow \dfrac{1}{30}x-\dfrac{1}{15}y=-1\Leftrightarrow x-2y=-30. \tag{2}\]
		Giải hệ $(1),(2)$: $\heva{&2x=y\\&x-2y=-30} \Leftrightarrow \heva{&2x=y\\&x-2\cdot 2x=-30} \Leftrightarrow \heva{&y=20\\&x=10.}$\\
		Vậy lúc đầu đem trộn $10$ tấn quặng loại A và $20$ tấn quặng loại B.
	}
\end{bt}

\begin{bt}%[Lê Mạnh Thắng]%[EX-TS10-ToanTT-2019]%[9D3K6]
	Một phòng họp có $80$ ghế ngồi, được xếp thành từng hàng, mỗi hàng có số lượng ghế bằng nhau. Nếu bớt đi $2$ hàng mà không làm thay đổi số lượng ghế trong phòng thì mỗi hàng còn lại phải xếp thêm $2$ ghế. Hỏi lúc đầu trong phòng có bao nhiêu hàng ghế?
	\dapso{$10$ hàng ghế}
	\loigiai{
		Gọi $x$, $y$ là số hàng ghế và số lượng ghế mỗi hàng lúc ban đầu (điều kiện $0<x<80$, $0<y<80$).\\
		Khi đó ta có
		$$\heva{&xy=80\\&(x-2)(y+2)=80} \Leftrightarrow \heva{&xy=80\\&xy+2x-2y-4=80} \Leftrightarrow \heva{&x\cdot (-y)=-80\\&x+(-y)=2.}$$
		$x$ và $-y$ là hai nghiệm của phương trình $t^2-2t-80=0 \Leftrightarrow \hoac{&t=10\\&t=-8}$.\\
		Vì $x>0, y>0$ nên suy ra $\heva{&x=10\\&-y=-8} \Leftrightarrow \heva{&x=10\\&y=8.}$
		Vậy lúc đầu trong phòng có $10$ hàng ghế.
	}
\end{bt}

\begin{bt}%[Đề thi tuyển sinh lớp $10$ THPT, Chuyên toán PTNK ĐHQG TP. HCM $2013 - 2014$]%[Lê Văn Thiện]%[EX-C2-TS10-2019]%[9D3T6]
	Có hai vòi nước $A$, $B$ cùng cung cấp cho một hồ cạn nước và vòi $C$ (đặt sát đáy hồ) lấy nước từ hồ cung cấp cho hệ thống tưới cây. Đúng $6$ giờ vòi $A$ và $B$ được mở; đến $7$ giờ vòi $C$ được mở; đến $9$ giờ thì đóng vòi $B$ và vòi $C$ ; đến $10$ giờ $45$ phút thì hồ đầy nước. Người ta thấy rằng nếu đóng vòi $B$ ngay từ đầu thì phải đến đúng $13$ giờ hồ mới đầy. Biết lưu lượng vòi $B$ là trung bình cộng của vòi $A$ và vòi $C$, hỏi một mình vòi $C$ tháo cạn hồ nước đầy trong bao lâu?
	\dapso{$12$ giờ.}
	\loigiai{
		Đổi $10$ giờ $45$ phút $=\dfrac{43}{4}$ giờ. \\
		Gọi thời gian để vòi $A$, $B$ chảy riêng vào đầy bể lần lượt là $x$ (giờ), $y$ (giờ), thời gian để vòi $C$ tháo cạn hồ nước đầy là $z$ (giờ). (Điều kiện $x, y, z> 0 $).\\
		Trong $1$ giờ vòi $A$ chảy vào được $\dfrac{1}{x}$ (hồ), vòi $B$ chảy vào được $\dfrac{1}{x}$ (hồ), vòi $C$ tháo ra $\dfrac{1}{x}$ (hồ).
		Đúng $6$ giờ vòi $A$ và $B$ được mở; đến $7$ giờ vòi $C$ được mở; đến $9$ giờ thì đóng vòi $B$ và vòi $C$; đến $10$ giờ $45$ phút thì hồ đầy. Ta có
		\[\left(\dfrac{43}{4} - 6\right)\cdot\dfrac{1}{x} + (9 - 6)\cdot\dfrac{1}{y} - (9 - 7)\cdot\dfrac{1}{z}= 1\Leftrightarrow\dfrac{19}{4x} + \dfrac{3}{y} - \dfrac{2}{z}= 1 \tag{1}\]
		Nếu đóng $B$ ngay từ đầu đến $13$ (h) hồ đầy nên
		\[(13 - 6)\dfrac{1}{x} - (9 - 7)\dfrac{1}{z}= 1\Leftrightarrow\dfrac{7}{x} - \dfrac{2}{z}= 1 \tag{2}\]
		Lưu lượng vòi $B$ là trung bình cộng của lưu lượng vòi $A$ và vòi $C$ nên
		\[\dfrac{1}{y}=\dfrac{1}{2}\left(\dfrac{1}{x} + \dfrac{1}{z}\right) \tag{3}\]
		Từ $(1), (2)$ và $(3)$ ta có
		\[\heva{&\dfrac{19}{4x} + \dfrac{3}{y} - \dfrac{2}{z}= 1\\ &\dfrac{7}{x} - \dfrac{2}{z}= 1\\ &\dfrac{1}{y}=\dfrac{1}{2}\left(\dfrac{1}{x} + \dfrac{1}{z}\right)} \Leftrightarrow \heva{&\dfrac{19}{4x} + \dfrac{3}{2}\left(\dfrac{1}{x} + \dfrac{1}{z}\right) - \dfrac{2}{z}= 1\\&\dfrac{7}{x} - \dfrac{2}{z}= 1\\&\dfrac{1}{y}=\dfrac{1}{2}\left(\dfrac{1}{x} + \dfrac{1}{z}\right)} \Leftrightarrow \heva{&\dfrac{25}{4x} - \dfrac{1}{2z}= 1\\ &\dfrac{7}{4x} - \dfrac{1}{2z}=\dfrac{1}{4} \\&\dfrac{1}{y}=\dfrac{1}{2}\left(\dfrac{1}{x} + \dfrac{1}{z}\right)} \Leftrightarrow \heva{&x=6\\&y=8\\&z=12.}\]
		Vậy một mình vòi $C$ tháo cạn hồn nước đầy trong $12$ giờ.
	}
\end{bt}

\begin{bt}%[Hoàng Ngọc Lâm]%[EX-TS10-ToanTT-2019]%[9D3T6]
	Trên một vùng biển được xem như bằng phẳng và không có chướng ngại vật. Vào lúc $6$h có một tàu cá đi thẳng qua tọa độ X theo hướng từ Nam đến Bắc với vận tốc không đổi. Đến $7$h một tàu du lịch cũng đi thẳng qua tọa độ X theo hướng từ Đông sang Tây với vận tốc lớn hơn vận tốc tàu cá $12$km/h. Đến $8$h khoảng cách giữa hai tàu là $60$km. Tính vận tốc mỗi tàu. 
	\dapso{tàu cá: $24$ km/h, tàu du lịch: $36$ km/h}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\draw [<-,very thick] (-4,0)--(0,0);
			\draw [->,very thick] (0,0)--(0,4);
			\tkzDefPoints{0/0/X,0/2/A,-3/0/B};
			\tkzLabelPoints(X);
			\draw [dashed] (0,2) --(-3,0);
			\tkzLabelSegment[right](A,A){\text{tàu cá}};
			\tkzLabelSegment[below](B,B){\text{tàu du lịch}};
			\tkzMarkRightAngle[size=.3](A,X,B);
			\end{tikzpicture}
		\end{center}
		Gọi $v_1$ (km/h) là vận tốc của tàu cá. (Điều kiện $v_1>0$)\\
		Gọi $v_2$ (km/h) là vận tốc của tàu du lịch. (Điều kiện $v_2>12$)\\
		Do vận tốc của tàu du lịch lớn hơn vận tốc tàu cá $12$ km/h nên ta có phương trình 
		\begin{equation}\label{ct1}
			v_2-v_1=12\Leftrightarrow v_2=v_1+12
		\end{equation}
		Lúc 8h, quãng đường tàu cá đã di chuyển được là $$v_1\cdot (8-6)=v_1\cdot 2\,\text{(km)}.$$ 
		Lúc 8h, quang đường tàu cá đã di chuyển được là $$v_2\cdot (8-7)=v_2\,\text{(km)}.$$
		Do phương chuyển động của hai tàu vuông góc với nhau nên ta có khoảng cách giữa hai tàu lúc 8h là $$\sqrt{(2v_1)^2+v_2^2}\,\text{(km)}.$$
		Theo giả thiết ta có 
		\begin{equation}\label{ct2}
			\sqrt{4v_1^2+v_2^2}=60\Leftrightarrow 4v_1^2+v_2^2=3600
		\end{equation}
		Thay \eqref{ct1} vào \eqref{ct2} ta được phương trình
		\begin{eqnarray*}
			&&4v_1^2+(v_1+12)^2=3600\\
			&\Leftrightarrow& 4v_1^2+v_1^2+24v_1+144=3600\\
			&\Leftrightarrow& 5v_1^2+24v_1-3456=0\\
			&\Leftrightarrow& \hoac{&v_1=24\,\text{(nhận)}\\&v_1=-\dfrac{144}{5}\,\text{(loại)}}
		\end{eqnarray*} 
	Suy ra vận tốc tàu cá là $24$ km/h. Vận tốc tàu du lịch là $24+12=36$ km/h.
	}
\end{bt}

\begin{bt}%[Hưng Nguyễn]%[EX-TS10-ToanTT-2019]%[9D3T6]
	Hai người cùng làm chung một công việc thì sau $8$ ngày sẽ xong. Sau khi cùng làm được $5$ ngày thì người thứ nhất bận không làm tiếp được nữa, một mình người thứ hai phải làm trong $9$ ngày mới xong phần công việc còn lại. Hỏi nếu mỗi người làm một mình thì cần bao nhiêu ngày để hoàn thành công việc?
	\dapso{Người thứ nhất: $12$ ngày; người thứ hai: $24$ ngày}
	\loigiai{
	Gọi thời gian một mình hoàn thành công việc của người thứ nhất và người thứ hai lần lượt là $x$ và $y$ (ngày) ($x$, $y>8$).\\
	Suy ra trong một ngày, người thứ nhất làm được $\dfrac{1}{x}$ công việc, người thứ hai làm được $\dfrac{1}{y}$ công việc.\\
	Vì hai người làm chung thì sau $8$ ngày hoàn thành công việc nên ta có phương trình $$\dfrac{8}{x}+\dfrac{8}{y}=1.$$
	Vì hai người cùng làm trong $5$ ngày và người thứ hai làm một mình thêm $9$ ngày thì hoàn thành công việc nên ta có phương trình $$\dfrac{5}{x}+\dfrac{5}{y}+\dfrac{9}{y}=1\Leftrightarrow \dfrac{5}{x}+\dfrac{14}{y}=1.$$
	Vậy ta có hệ phương trình
	\begin{eqnarray*}
		\heva{& \dfrac{8}{x}+\dfrac{8}{y}=1 \\ & \dfrac{5}{x}+\dfrac{14}{y}=1}&\Leftrightarrow& \heva{& \dfrac{40}{x}+\dfrac{40}{y}=5 \\ & \dfrac{40}{x}+\dfrac{112}{y}=8}\Leftrightarrow \heva{& \dfrac{72}{y}=3 \\ & \dfrac{5}{x}+\dfrac{14}{y}=1}\\
		&\Leftrightarrow & \heva{& y=24 \text{ (thỏa mãn)} \\ & \dfrac{5}{x}=1-\dfrac{14}{24}}\Leftrightarrow \heva{& y=24\\ & x=5:\dfrac{5}{12}}\Leftrightarrow \heva{& y=24 \\ & x=12 \text{ (thỏa mãn).}}
	\end{eqnarray*}
	Vậy người thứ nhất làm một mình trong $12$ ngày thì xong công việc, người thứ hai làm một mình trong $24$ ngày thì xong công việc.
	}
\end{bt}
