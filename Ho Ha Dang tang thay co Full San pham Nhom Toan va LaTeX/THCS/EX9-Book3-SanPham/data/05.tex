\section{PHƯƠNG TRÌNH CHỨA CĂN THỨC BẬC HAI}

\begin{center}
 \large \textbf{TỔNG QUAN VỀ CHUYÊN ĐỀ}
\end{center}
Ta gọi phương trình chứa căn thức bậc hai là phương trình chứa  ẩn trong dấu căn bậc hai. Đây là dạng toán thường gặp trong các kì thi học sinh giỏi vì nó đòi hỏi sự thành thạo và sáng tạo của học sinh. \\
Chuyên đề này giới thiệu những phương pháp thường  dùng để giải phương trình chứa căn thức bậc hai như:
\begin{itemize}
	\item Bình phương hai vế của phương trình.
	\item Đưa phương trình về dạng $A^2=B^2$.
	\item Đưa phương trình về dạng $A^2+B^2=0$.
	\item Đặt nhân tử chung.
	\item Dùng biểu thức liên hợp.
	\item Dùng bất đẳng thức.
\end{itemize}
\textbf{Bài toán cổ}
\begin{center}
 \large \textbf{BÀI TOÁN CỦA BÁT-XCA-RA}
\end{center}
\textit{Tìm các cạnh góc vuông của một tam giác vuông, biết rằng số đo cạnh huyền và số đo diện tích biểu thị bởi cùng một số.}
\loigiai
{
	Gọi $x$ và $y$ là độ dài các cạnh góc vuông thì độ dài cạnh huyền bằng $\sqrt{x^2+y^2}$ và diện tích bằng $\dfrac{xy}{2}$. Ta có phương trình
	\begin{align*}
		&\dfrac{xy}{2}=\sqrt{x^2+y^2}\Leftrightarrow xy =2\sqrt{x^2+y^2} \\
		\Leftrightarrow &x^2y^2=4(x^2+y^2)\Leftrightarrow  x^2\left(y^2-4\right)=4y^2\Leftrightarrow x^2=\dfrac{4y^2}{y^2-4}
	\end{align*}
Bài toán có vô số đáp số là $\left(\dfrac{2y}{\sqrt{y^2-4}};y\right)$ với $y$ tùy ý lớn hơn $2$.\\
Chẳng hạn $y=\sqrt{6}$, ta có $x=2\sqrt{3}$.
}
\subsection{Bình phương hai vế của phương trình}
Bình phương hai vế của phương trình giúp ta  khử dấu căn bậc hai. Phép bình phương hai vế  của  phương trình là tương đương nếu có thêm điều kiện hai vế cùng không âm (hoặc cùng không dương).

\begin{vd} Giải phương trình $x^2-x-2=2\sqrt{16x+1} \hfill  (1)$ 
     \loigiai{
       ĐKXĐ: $x\geq-\dfrac{1}{16}$. Thêm điều kiện $x^2-x-2\ge 0\ \ (2)$ thì 
       $$(1) \Leftrightarrow {\left(x^2-x-2\right)}^2 =4(16x+1) 
       \Leftrightarrow  x^4-2x^3-3x^2-50x=0
     \Leftrightarrow  x(x-5)(x^2+3x+12)=0$$
     \begin{itemize}
     	\item  $x=0$ loại vì trái với (2).
     	\item $x=5$ thỏa mãn (1) và (2).
     	\item $x^2+3x+12=0$ vô nghiệm.
     \end{itemize}
     \textit{Đáp số:} Phương trình có  nghiệm $x=5$.
    }
\end{vd}

\begin{vd} Giải phương trình $x^2+5x+6=2\sqrt{3x+4} \hfill  (1)$ 
     \loigiai{
       ĐKXĐ: $x\geq-\dfrac{4}{3}$. Thêm điều kiện $x^2+5x+6\ge 0\ \ (2)$ thì 
       $$(1) \Leftrightarrow {\left(x^2+5x+6\right)}^2 =4(3x+4) 
            \Leftrightarrow  {(x+1)}^2(x^2+8x+20)=0   $$
       \begin{itemize}
       	\item $x=-1$ thỏa mãn (1) và (2).
       	\item $x^2+8x+20=0$ vô nghiệm.
       \end{itemize}
     \textit{Đáp số:} Phương trình có nghiệm $x=-1$.
    }
\end{vd}

\subsection{Đưa phương trình về dạng $A^2=B^2$ (tức là $A^2-B^2=0$)}
\begin{vd} Giải phương trình $4x^2+8x=\sqrt{2x+6} \hfill  (1)$ 
     \loigiai{
       ĐKXĐ: $x\geq -3$. Cộng $2x+\dfrac{25}{4}$ vào hai vế thì 
       $$(1) \Leftrightarrow 4x^2+10x+\dfrac{25}{4}=2x+6+\sqrt{2x+6}+\dfrac{1}{4}
            \Leftrightarrow  {\left(2x+\dfrac{5}{2}\right)}^2 ={\left(\dfrac{1}{2}+\sqrt{2x+6} \right)}^2.$$
      \begin{itemize}
      	\item Xét $2x+\dfrac{5}{2}=\dfrac{1}{2}+\sqrt{2x+6}\Leftrightarrow 2x+2=\sqrt{2x+6}$ \\
      	$\Leftrightarrow \heva{& x\ge -1\\ & 2x^2+3x-1=0} \Leftrightarrow x=\dfrac{-3+\sqrt{17}}{4}$ thỏa mãn ĐKXĐ.
      	\item Xét $2x+\dfrac{5}{2}=-\dfrac{1}{2}-\sqrt{2x+6}\Leftrightarrow 2x+3=-\sqrt{2x+6}$ \\
      	$\Leftrightarrow \heva{& x\le -1,5 \\ & 4x^2+10x+3=0} \Leftrightarrow x=\dfrac{-5-\sqrt{13}}{4}$ thỏa mãn ĐKXĐ.
      \end{itemize}     
     \textit{Đáp số:} Phương trình có  nghiệm  $\dfrac{-3+\sqrt{17}}{4}$ ; $\dfrac{-5-\sqrt{13}}{4}$. 
    }
   \end{vd}

\begin{vd} Giải các phương trình 
\begin{multicols}{2}
	\begin{enumerate}[a)]
		\item $x^2-10x-12=4\sqrt{2x+3}$
		\item $2x^2+x-3=2\sqrt{2x+3}$
	\end{enumerate}
\end{multicols}
     \loigiai{
     	\begin{enumerate}[a)]
     		\item ĐKXĐ: $x\geq -\dfrac{3}{2}$. Cộng $8x+13$ vào hai vế ta được 
     		$$x^2-2x+1=4(2x+3)+4\sqrt{2x+3}+1
     		\Leftrightarrow  {\left(x-1\right)}^2 ={\left(1+2\sqrt{2x+3} \right)}^2.$$
     		\begin{itemize}
     			\item Xét $x-1=1+2\sqrt{2x+3}\Leftrightarrow x-2=2\sqrt{2x+3}$ \\
     			$\Leftrightarrow \heva{& x\ge 2\\ & x^2-12x-8=0} \Leftrightarrow x=6+2\sqrt{11}$ (thỏa mãn ĐKXĐ).
     			\item Xét $x-1=-1-2\sqrt{2x+3}\Leftrightarrow x=-2\sqrt{2x+3}$ \\
     			$\Leftrightarrow \heva{& x\le 0\\ & x^2-8x-12=0} \Leftrightarrow x=4-2\sqrt{7}$ (thỏa mãn ĐKXĐ).
     		\end{itemize}
     		\textit{Đáp số:} Phương trình có  nghiệm  $6+2\sqrt{11}; 4-2\sqrt{7}$.
     		\item ĐKXĐ: $x\geq-\dfrac{3}{2}$. Nhân hai vế  của phương trình với 2 ta được
     		$$
     		4x^2+2x-6=4\sqrt{2x+3}$$
     		Cộng $2x+7$ vào hai vế, ta có 
     		$$
     		4x^2+4x+1=2x+3+4\sqrt{2x+3}+4    \Leftrightarrow  {\left(2x+1\right)}^2 ={\left(2+\sqrt{2x+3} \right)}^2$$
     		\begin{itemize}
     			\item Xét $2x+1=2+\sqrt{2x+3}\Leftrightarrow 2x-1=\sqrt{2x+3}$ \\
     			$\Leftrightarrow \heva{& x\ge \dfrac{1}{2}\\ & 2x^2-3x-1=0} \Leftrightarrow x=\dfrac{3+\sqrt{17}}{4}$ (thỏa mãn ĐKXĐ).
     			\item Xét $2x+1=-2-\sqrt{2x+3}\Leftrightarrow 2x+3=-\sqrt{2x+3}$ \\
     			$\Leftrightarrow \heva{& x\le -\dfrac{3}{2}\\ & 2x^2+5x+3=0} \Leftrightarrow x=-\dfrac{3}{2}$ (thỏa mãn ĐKXĐ).
     		\end{itemize}
     		\textit{Đáp số:} Phương trình có  nghiệm: $\dfrac{3+\sqrt{17}}{4};-\dfrac{3}{2}$.
     	\end{enumerate}  
    }
   \end{vd}

\begin{vd} Giải phương trình $4x+1-x^2=2\sqrt{2x+1} \hfill  (1)$ 
     \loigiai{
       ĐKXĐ: $x\geq -\dfrac{1}{2}$.
$$(1) \Leftrightarrow x^2-4x-1=-2\sqrt{2x+1}$$
Cộng $2x+2$ vào hai vế ta được
$$
x^2-2x+2=2x+1-2\sqrt{2x+1}+1  \Leftrightarrow   {\left(x-1\right)}^2 ={\left(1-\sqrt{2x+1} \right)}^2$$
\begin{itemize}
	\item Xét $x-1=1-\sqrt{2x+1}\Leftrightarrow x-2=-\sqrt{2x+1}$ \\
	$\Leftrightarrow \heva{& x\le 2 \\ & x^2-6x+3=0} \Leftrightarrow x=3-\sqrt{6}$ thỏa mãn ĐKXĐ.
	\item Xét $x-1=\sqrt{2x+1}-1\Leftrightarrow x=\sqrt{2x+1}$ \\
	$\Leftrightarrow \heva{& x\ge 0\\ & x^2-2x-1=0} \Leftrightarrow x=1+\sqrt{2}$ thỏa mãn ĐKXĐ.
\end{itemize}
\textit{Đáp số:} Phương trình có  nghiệm: $3-\sqrt{6};1+\sqrt{2}$.
}
   \end{vd}

\begin{vd} Giải phương trình sau bằng cách đưa  về  dạng $A^2=B^2$ \[x^2-x-2=2\sqrt{16x+1}\]
     \loigiai{
       ĐKXĐ: $x\geq-\dfrac{1}{16}$. Nhân hai vế của phương trình với $4 $ ta được
       \[4x^2-4x-8=8\sqrt{16x+1}\]
       Cộng $16x+17$ vào hai vế được
       \[4x^2+12x+9=16x+1+8\sqrt{16x+1}+16 
 \Leftrightarrow   {\left(2x+3\right)}^2 ={\left(4+\sqrt{16x+1} \right)}^2\]
 \begin{itemize}
 \item Xét $2x+3=4+\sqrt{16x+1}\Leftrightarrow 2x-1=\sqrt{16x+1}$ \\
 $\Leftrightarrow \heva{& x\ge \dfrac{1}{2}\\ & 4x^2-20x=0} \Leftrightarrow x=5,$ thỏa mãn ĐKXĐ.
 \item  Xét $2x+3=-4-\sqrt{16x+1}\Leftrightarrow 2x+7=-\sqrt{16x+1}$ 
 $\Leftrightarrow \heva{& x\le- \dfrac{7}{2}\\ & x^2+3x+12=0}$ vô nghiệm.
 \end{itemize}
\textit{Đáp số:} Phương trình có  nghiệm $x=5.$
}
   \end{vd}

\begin{vd} Giải phương trình $2x^2-3x+2=x\sqrt{3x-2}$ 
     \loigiai{
       ĐKXĐ: $x\geq \dfrac{2}{3}$.
       $$2x^2-3x+2=x\sqrt{3x-2} \Leftrightarrow 2x^2=3x-2+x\sqrt{3x-2}$$
       Cộng $\dfrac{x^2}{4}$ vào hai vế ta được
       \[
       2x^2+\dfrac{x^2}{4}=3x-2+x\sqrt{3x-2} +\dfrac{x^2}{4}
        \Leftrightarrow   {\left(\dfrac{3x}{2}\right)}^2 ={\left(\dfrac{x}{2}+\sqrt{3x-2} \right)}^2\]
Do  $x\ge \dfrac{2}{3}$ nên $\dfrac{3x}{2}=\dfrac{x}{2}+\sqrt{3x-2}\Leftrightarrow x=\sqrt{3x-2}\Leftrightarrow  x^2-3x+2=0\Leftrightarrow \hoac{& x=1\\ & x=2}$ (thỏa mãn).\\
\textit{Đáp số:} Phương trình có $2$ nghiệm: $x=1, x=2$.
}
\end{vd}

\subsection{Đưa phương trình về dạng $A^2+B^2=0$}

\begin{vd} Giải phương trình sau bằng cách đưa  về dạng $A^2+B^2=0$
\[x^2+5x+6=2\sqrt{3x+4}\]
 \loigiai{
       ĐKXĐ: $x\geq - \dfrac{4}{3}$.\\
       Ta viết lại phương trình
       \begin{align*}
       		&x^2+2x+1 +3x+4-2\sqrt{3x+4}+1=0 \\
       		\Leftrightarrow &{(x+1)}^2 +{\left(1-\sqrt{3x+4}\right)}^2 =0 \Leftrightarrow \heva{& x=1=0 \\ & \sqrt{3x+4}=1} \Leftrightarrow x=-1
       \end{align*}       
       \textit{Đáp số:} Phương trình có  nghiệm $x=-1$.
}
\end{vd}


\subsection{Đặt nhân tử chung}

\begin{vd} Giải phương trình $\sqrt{x^2-2x}+\sqrt{2x^2+4x}=2x \hfill (1)$ 
     \loigiai{
       ĐKXĐ: $x\geq 0$.\\
       Phương trình tương đương 
       $$\sqrt{x(x-2)}+\sqrt{2x(x+2)}=2x$$
       \begin{itemize}
       	\item Ta thấy $x=0$ thỏa mãn phương trình.
       	\item Xét $x\ne 0$. Chia hai vế của phương trình cho $\sqrt{x}$ được
       	$$ \sqrt{x-2}+\sqrt{2(x+2)}=2\sqrt{x}$$
       	Bình phương hai vế, ta được
       	\[3x+2+2\sqrt{2(x+2)(x-2)} =4x
       	\Leftrightarrow 2\sqrt{2(x+2)(x-2)}=x-2 \tag{2}\]
       	\item Ta thấy $x=2$ thỏa  mãn (2).
       	\item Xét $x\ne 2$. Chia cả hai vế của (2)  cho $\sqrt{x-2}$ được
       	$$2\sqrt{2x+4}=\sqrt{x-2}\Leftrightarrow x=-\dfrac{18}{7}, \  \text{không thỏa mãn} \ x\ge 0$$
       \end{itemize}
       \textit{Đáp số:} Hai nghiệm  $0$ và $2$.
}
\end{vd}
       
\subsection{Đặt ẩn phụ}  
       
       \begin{vd} Giải phương trình $x+\sqrt{x+\dfrac{1}{2}+\sqrt{x+\dfrac{1}{4}}} =2$.
       \loigiai{
       ĐKXĐ: $x\geq - \dfrac{1}{4}$.\\
       Đặt $\sqrt{x+\dfrac{1}{4}}=y\ge 0$ thì $x=y^2-\dfrac{1}{4}$. Ta có
       \begin{align*}
       		&y^2-\dfrac{1}{4}+\sqrt{y^2+\dfrac{1}{4}+y}=2
       		\Leftrightarrow y^2-\dfrac{1}{4}+y+\dfrac{1}{2}=2 \\
       		\Leftrightarrow 
       		&4y^2+4y-7=0\Leftrightarrow y=\dfrac{-1\pm 2\sqrt{2}}{2}.
       \end{align*}
       Do $y\ge 0$ nên ta chọn $y=\dfrac{-1+2\sqrt{2}}{2}$. Từ đó $x=2-\sqrt{2}$. \\
       \textit{Đáp số:} Một nghiệm $x=2-\sqrt{2}$.
}
\end{vd}
       
       
\begin{vd}
Giải phương trình
\begin{align*}
5x^2-2x+1=\left(4x-1\right)\sqrt{x^2+1} \tag{1}
\end{align*}
\loigiai{ Đặt $\sqrt{x^2+1}=a>0$ thì $a^2=x^2+1$. Ta có
\begin{align*}
(1) &\Leftrightarrow \left(x^2+1\right)-\left(4x-1\right)\sqrt{x^2+1}+4x^2-2x=0\\
&\Leftrightarrow a^2-\left(4x-1\right)a+2x\left(2x-1\right)=0 \tag{2}
\end{align*}
 Xét phương trình (2) với ẩn $a$, tích hai nghiệm bằng $2x\left(2x-1\right)$, tổng hai nghiệm bằng $4x-1$ nên $a=2x$ hoặc $a=2x-1.$
\begin{itemize}
 \item Với $a=2x$ thì $\sqrt{x^2+1}=2x \Leftrightarrow \heva{&x\ge 0 \\&3x^2=1}\Leftrightarrow x=\dfrac{\sqrt{3}}{3}.$
\item Với $a=2x-1$ thì $\sqrt{x^2+1}=2x-1\Leftrightarrow \heva{&x\ge \dfrac{1}{2}\\&x\left(3x-4\right)=0}\Leftrightarrow x=\dfrac{4}{3}$.
\end{itemize}
\textit{Đáp số}: Hai nghiệm: $\dfrac{\sqrt{3}}{3}$ và $\dfrac{4}{3}$
}
\end{vd}
\begin{vd}
Giải phương trình sau bằng cách đặt ẩn phụ
 \begin{align*}
2x^2-3x+2=x\sqrt{3x-2}
\end{align*}
\loigiai
{
	ĐKXĐ: $x\ge \dfrac{2}{3}$. \\ 
	Đặt $\sqrt{3x-2}=a\ge 0$ thì $3x-2=a^2$. Ta có
\begin{align*}
&2x^2-\left(3x-2\right)=x\sqrt{3x-2}\\
\Leftrightarrow &a^2+xa-2x^2=0 \tag{1}
\end{align*}
Xét phương trình (1) với ẩn $a$, tích hai nghiệm bằng $-2x \cdot x$, tổng hai nghiệm bằng $-x$ nên $a=x$ hoặc $a=-2x$.
\begin{itemize}
	\item Với $a=x$ thì $\sqrt{3x-2}=x\Leftrightarrow \heva{&x\ge 0\\&x^2-3x+2=0}\Leftrightarrow \hoac{&x=1\\&x=2.}$ (thỏa mãn ĐKXĐ).
	\item Với $a=-2x$, do $x\ge \dfrac{2}{3}$ nên $a<0$, loại.
\end{itemize}
\textit{Đáp số}: Hai nghiệm: $1$ và $2$.
}
\end{vd}
\begin{vd}
Giải phương trình 
\begin{align*}
10\sqrt{x^3+1}=3\left(x^2+2\right) \tag{1}
\end{align*}
\loigiai{
ĐKXĐ: $x\ge -1$.\\
Đặt $\sqrt{x+1}=a\ge 0,\:\sqrt{x^2-x+1}=b>0$, ta có $a^2+b^2=x+1+x^2-x+1=x^2+2$. Khi đó
\[(1)\Leftrightarrow 10ab=3\left(a^2+b^2\right)\Leftrightarrow \left(a-3b\right)\left(3a-b\right)=0\]
\begin{itemize}
	\item Với $a=3b$ thì $\sqrt{x+1}=3\sqrt{x^2-x+1}\Leftrightarrow 9x^2-10x+8=0$, vô nghiệm.
	\item Với $3a=b$ thì $3\sqrt{x+1}=\sqrt{x^2-x+1}\Leftrightarrow x^2-10x-8=0\Leftrightarrow x=5\pm\sqrt{33}$, thỏa ĐKXĐ.
\end{itemize}
\textit{Đáp số}: Hai nghiệm $5\pm\sqrt{33}$.\\
}
\end{vd}
\begin{vd}
Giải phương trình 
\[\left(2x-1\right)\sqrt{10-4x^2}=5-2x \tag{1}\]

\loigiai{
ĐKXĐ: $x^2\le \dfrac{5}{2}.$\\
Đặt $2x-1=a,\:\sqrt{10-4x^2}=b\ge 0$ thì
\[a^2+b^2=4x^2-4x+1+10-4x^2=11-4x \tag{2}\]
Ta có $ab=5-2x$ \hfill (3)\\
Từ (2) và (3) suy ra $a^2+b^2-2ab=11-4x-2\left(5-2x\right)=1 \Leftrightarrow \left(a-b\right)^2=1 \Leftrightarrow a-b=\pm 1$.
\begin{itemize}
	\item Với $a-b=1$ thì $2x-1-\sqrt{10-4x^2}=1$\\
	$\Leftrightarrow 2x-2=\sqrt{10-4x^2}\Leftrightarrow \heva{&x\ge 1\\&4x^2-4x-3=0}\Leftrightarrow x=\dfrac{3}{2}$, thỏa ĐKXĐ.
	\item Với $a-b=-1$ thì $2x-1-\sqrt{10-4x^2}=-1$\\
	$\Leftrightarrow 2x=\sqrt{10-4x^2}\Leftrightarrow \heva{&x\ge 0\\&4x^2=5}\Leftrightarrow x=\dfrac{\sqrt{5}}{2}$, thỏa mãn ĐKXĐ.
\end{itemize}
Đáp số: Hai nghiệm: $\dfrac{3}{2}$ và $\dfrac{\sqrt{5}}{2}$.
}
\end{vd}

\begin{vd}
Giải phương trình 
\begin{align*}
\sqrt{2x-1}-\sqrt{x+1}=2x-4
\end{align*}
\loigiai{
ĐKXĐ: $x\ge \dfrac{1}{2}$.
Đặt $\sqrt{2x-1}=a\ge 0,\:\sqrt{x+1}=b\ge 0$, ta có $\heva{&a-b=2x-4\\&a^2-b^2=x-2.}$\\
Suy ra $a-b=2\left(a^2-b^2\right)\Leftrightarrow \left(a-b\right)\left(2a+2b-1\right)=0$.
\begin{itemize}
	\item Với $a=b$ thì $2x-1=x+1\Leftrightarrow x=2$, thỏa mãn ĐKXĐ.
	\item Với $2a+2b=1$ thì $2\sqrt{2x-1}+2\sqrt{x+1}=1$.\\
	Phương trình vô nghiệm vì với $x\ge \dfrac{1}{2}$ thì $2\sqrt{2x-1}\ge 0$ và $2\sqrt{x+1}>1$, vế trái $>1$.
\end{itemize}
\textit{Đáp số}: Một nghiệm $x=2$.
}
\end{vd}

\begin{vd}
Giải phương trình
\[\sqrt{2x^2-3x+10}+\sqrt{2x^2-5x+4}=x+3\]
\loigiai{
ĐKXĐ: $2x^2-3x+10\ge 0,\: 2x^2-5x+4\ge 0$.\\
Đặt $\sqrt{2x^2-3x+10}=a\ge 0,\: \sqrt{2x^2-5x+4}=b\ge 0$.\\
Ta có $\heva{&a+b=x+3\\&a^2-b^2=2x+6}$.\\
Suy ra $a^2-b^2=2\left(a+b\right)\Leftrightarrow \left(a-b-2\right)\left(a+b\right)=0$.
\begin{itemize}
	\item Với $a+b=0$ thì $a=b=0$ và $x=-3$, không thỏa ĐKXĐ.
	\item Với $a-b=2$ thì $\sqrt{2x^2-3x+10}=2+\sqrt{2x^2-5x+4}$.
	Bình phương hai vế rồi rút gọn.\\

\subsection{Dùng biểu thức liên hợp}
 Phương pháp dùng biểu thức liên hợp còn được gọi là phương pháp khử căn thức ở tử, thường dùng hơn cả là 
\begin{align*}
&\sqrt{A}-\sqrt{B}=\dfrac{A-B}{\sqrt{A}+\sqrt{B}} \tag{1}\\
&\sqrt{A}+\sqrt{B}=\dfrac{A-B}{\sqrt{A}-\sqrt{B}} \tag{2}
\end{align*}
Trong các công thức (1) và (2), khi nhân và chia vế trái với biểu thức liên hợp của nó, ta được vế phải. Mục đích của việc khử căn thức ở tử nhằm làm xuất hiện nhân tử chung.
\begin{vd}

\begin{align*}
x^2+5x+6=2\sqrt{3x+4} \tag{1}
\end{align*}
\loigiai{
ĐKXĐ: $x\ge -\dfrac{4}{3}$. Với nhận xét $-1$ là một nghiệm, ta biến đổi.
\begin{align*}
(1)& \Leftrightarrow \left(x+1\right)\left(x+4\right)+2=2\sqrt{3x+4}\\
&\Leftrightarrow \left(x+1\right)\left(x+4\right)=2\left(\sqrt{3x+4}-1\right)\\
&\Leftrightarrow \left(x+1\right)\left(x+4\right)=\dfrac{2(3x+4-1)}{\sqrt{3x+4}+1}\\
&\Leftrightarrow \left(x+1\right)\left(x+4\right)=\dfrac{6(x+1)}{\sqrt{3x+4}+1}
\end{align*}
Rõ ràng $x=-1$ thỏa mãn phương trình.\\
Với $x\ne -1$ ta có $x+4=\dfrac{6}{1+\sqrt{3x+4}}$ (2)\\
Với $x>-1$ thì (2) có vế trái lớn hơn $3$, vế phải nhỏ hơn $3$, vô nghiệm.\\
Với $-\dfrac{4}{3}\le x<-1$ thì (2) có vế trái nhỏ hơn $3$, vế phải lớn hơn $3$, vô nghiệm.\\
\textit{Đáp số}: Một nghiệm $x=-1$\\
\textit{Lưu ý}: Cách giải khác, xem Ví dụ 74.

\item $\sqrt{x+1}+\sqrt{3-x}-\sqrt{\left(x+1\right)\left(3-x\right)}=1$ \hspace{1cm} (1)
\item $\sqrt{x-\dfrac{1}{x}}-\sqrt{1-\dfrac{1}{x}}=\dfrac{x-1}{x}$ \hspace{3.3cm} (2)
\end{enumerate}
\loigiai{
	\begin{enumerate}[a)]
		\item ĐKXĐ: $-1\le x\le 3$\\
		Đặt $\sqrt{x+1}=a\ge 0;\:\sqrt{3-x}=b\ge 0$\\
		Từ phương trình (1) ta được
		\begin{align*}
		(1)&\Leftrightarrow a+b-ab=1\\
		&\Leftrightarrow \left(a-1\right)\left(b-1\right)=0\\
		&\Leftrightarrow \hoac{&a=1\\&b=1}\Leftrightarrow \hoac{&\sqrt{x+1}=1\\&\sqrt{3-x}=1}\Leftrightarrow \hoac{&x=0\\&x=2}
		\end{align*}
		\textit{Đáp số}: Hai nghiệm $x=0; x=2$.
		\item ĐKXĐ: $x\ge 1$ hoặc $-1\le x<1$\\
		Ta thấy $x=1$ thỏa phương trình (2)\\
		Đặt $\sqrt{x-\dfrac{1}{x}}=a\ge 0;\:\sqrt{1-\dfrac{1}{x}}=b>0$ ta được\\
		\begin{align*}
		&\Rightarrow \heva{&a^2-b^2=x-1\\&a-b=\dfrac{x-1}{x}}\Rightarrow a+b=x\\
		&\Rightarrow \heva{&a+b=x\\&a-b=1-\dfrac{1}{x}}\Rightarrow 2a=x-\dfrac{1}{x}+1=a^2+1
		&\Rightarrow a^2-2a+1=0\Rightarrow a=1\\
		&\Rightarrow x=\dfrac{1+\sqrt{5}}{2}
		\end{align*}
		\textit{Đáp số}: Hai nghiệm $x=1; x=\dfrac{1+\sqrt{5}}{2}$.
	\end{enumerate}
}
\end{bt}

\begin{bt} 
Giải các phương trình sau
\begin{enumerate}[a)]
\item $\sqrt{x^2+16}-\sqrt{x^2+7}=3x-8$ \hspace{1.89cm} (1)
\item $\sqrt{x^3+10}-\sqrt{x^3+5}=2x+3$ \hspace{1.89cm} (2)
\item $\sqrt{x^2-1}+\sqrt{2x^2+4x+3}=2x+1$ \hspace{1cm} (3)
\end{enumerate}
\loigiai{
	\begin{enumerate}[a)]
		\item $\sqrt{x^2+16}-\sqrt{x^2+7}=3x-8$ \hfill (1)
		\begin{align*}
		(1)&\Leftrightarrow \left(\sqrt{x^2+16}-5\right)-\left(\sqrt{x^2+7}-4\right)=3x-9\\
		&\Leftrightarrow \dfrac{(x^2+16)-25}{\sqrt{x^2+16}+5}-\dfrac{(x^2+7)-16}{\sqrt{x^2+7}+4}=3x-9\\
		&\Leftrightarrow \dfrac{x^2-9}{5+\sqrt{x^2+16}}-\dfrac{x^2-9}{4+\sqrt{x^2+7}}=3(x-3)
		\end{align*}
		Dễ thấy $x=3$ thỏa mãn phương trình\\
		Với $x\ne 3$ ta có $\dfrac{x+3}{5+\sqrt{x^2+16}}-\dfrac{x+3}{4+\sqrt{x^2+7}}=3$ \hfill (*)\\
		Do $x^2+16>x^2+7$ nên $3x>8$, do đó $x+3>0$\\
		Xét (*) có $x+3>0$ nên vế trái của (*) âm, suy ra (*) vô nghiệm.\\
		\textit{Đáp số}: Một nghiệm $x=3$.
		\item $\sqrt{x^3+10}-\sqrt{x^3+5}=2x+3$\\
		ĐKXĐ: $x^3\ge -5$\\
		Ta thấy $x=-1$ thỏa mãn phương trình (2)\\
		Xét $x\ne -1$, ta có
		$\dfrac{(x^3+10)-(x^3+5)}{\sqrt{x^3+10}+\sqrt{x^3+5}}=2x+3\Leftrightarrow \dfrac{5}{\sqrt{x^3+10}+\sqrt{x^3+5}}=2x+3$ \hfill (**)\\
		Nếu $x>-1$ thì (**) có vế trái nhỏ hơn $1$, vế phải lớn hơn $1$, suy ra (**) vô nghiêm.\\
		Nếu $x<-1$ thì (**) có vế trái nhỏ hơn $1$, vế phải lớn hơn $1$, suy ra (**) vô nghiêm.\\
		\textit{Đáp số}: Một nghiệm $x=-1$.
		\item $\sqrt{x^2-1}+\sqrt{2x^2+4x+3}=2x+1$\\
		ĐKXĐ: $x\ge 1$
		\begin{align*}
		(3)&\Leftrightarrow (2x+1)-\sqrt{2x^2+4x+3}=\sqrt{x^2-1}\\
		&\Leftrightarrow \dfrac{(2x-1)^2-(2x^2+4x+3)}{2x+1+\sqrt{2x^2+4x+3}}=\sqrt{x^2-1}\\
		&\Leftrightarrow \dfrac{2(x^2-1)}{2x+1+\sqrt{2x^2+4x+3}}=\sqrt{x^2-1}\\
		&\Leftrightarrow \sqrt{x^2-1}\left(\dfrac{2\sqrt{x^2-1}}{2x+1+\sqrt{2x^2+4x+3}}-1\right)=0
		\end{align*}
		Giá trị $x=1$ thỏa mãn (3). Với $x>1$ ta có\\
		$2\sqrt{x^2-1}=2x+1+\sqrt{2x^2+4x+3}$ \hfill (***)\\
		Phương trình (***) vô nghiệm.\\
		\textit{Đáp số}: Một nghiệm $x=1$.
	\end{enumerate}
}
\end{bt}

    Vậy $4x-2=0\Leftrightarrow x=\dfrac{1}{2}$.\\
    $\textit{Đáp số: }$ Một nghiệm $x=\dfrac{1}{2}$.
    \end{enumerate}    
    }

\begin{bt}Giải các phương trình

    $\textit{Đáp số: }$ Một nghiệm $x=-1$.
    \item $\sqrt{2x^2+4x+6}+\sqrt{x^4-2x^2+2}=2-2x-x^2$.\hspace{2cm} (1)\\

    Gọi vế trái của (2) là $A$, vế phải của (2) là $B$.\\
    Ta có $A\ge 3$ và $B\le 3$.\\
    $\textit{Đáp số: }$ Một nghiệm $x=-1$.
    \item $\sqrt{x^2+x-1}+\sqrt{x-x^2+1}=x^2-x+2$.\hspace{2cm} (1)\\

    $\Rightarrow A\le\dfrac{x^2+x+x-x^2+2}{2}=x+1$.\\Suy ra $A=x+1=B$.\\
    $\textit{Đáp số: }$ Một nghiệm $x=1$.
    \item $\sqrt{3x^2+3x}+\sqrt{x-x^2}=2x+1$.\hspace{2cm} (1)\\
    \textit{Cách 1. }$\sqrt{3(x^2+x)}+\sqrt{x-x^2}$.\\
    $=2\cdot\sqrt{\dfrac{3}{4}(x^2+x)}+2\cdot\sqrt{\dfrac{1}{4}(x-x^2)}\le\dfrac{3}{4}+x^2+x+1\dfrac{1}{4}+x-x^2=2x+1$.\\
    Xảy ra dấu bằng tại $\heva{&x^2+x=\dfrac{3}{4}\\&x-x^2=\dfrac{1}{4}}\Leftrightarrow x=\dfrac{1}{2}$.\\
    \textit{Cách 2. }Dùng bất đẳng thức Bu-nhi-a-cốp-xki.\\

$\textit{Đáp số: }$ Một nghiệm $x=\dfrac{1}{2}$.
    \end{enumerate}
