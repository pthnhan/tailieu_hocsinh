\begin{ex}%[Đề thi vào 10, Chuyên Tiền Giang, 2017]%[9D1K8]%[9D4G7]%[9D3G3]
\begin{enumerate}    
\item Rút gọn biểu thức $A=\dfrac{\sqrt{4+2\sqrt{3}}-\sqrt{3}}{(\sqrt{5}+2)\sqrt[3]{17\sqrt{5}-38}-2}$.
\item Giải phương trình $x^{2}-\sqrt{x^{3}+x}=6x-1$.
\item Giải hệ phương trình $\heva{&x^{2}+9y^{2}+8xy+24=0\\ & x-3y+xy=0}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $A=\dfrac{\sqrt{(\sqrt{3}+1)^{2}}-\sqrt{3}}{(\sqrt{5}+2)\sqrt[3]{(\sqrt{5}-2)^{3}}-2}=\dfrac{\sqrt{3}+1-\sqrt{3}}{(\sqrt{5}+2)(\sqrt{5}-2)-2}=-1$.
\item Điều kiện xác định: $x\geq 3+2\sqrt{2}$ hoặc $0\leq x\leq 3-2\sqrt{2}$.
\[
x^{2}-\sqrt{x^{3}+x}=6x-1\Leftrightarrow x^{2}+1-\sqrt{x(x^{2}+1)}-6x=0
\]
Đặt $a=\sqrt{x}$, $b=\sqrt{x^{2}+1}$ ($a\geq 0$, $b\geq 1$). Khi đó phương trình trở thành:
\[
b^{2}-ab-6a^{2}=0\Leftrightarrow (b-3a)(b+2a)=0\Leftrightarrow \hoac{&b=3a\ (\mbox{TM})\\ &b=-2a\ (\mbox{Loại})}.
\]
Với $b=3a\Leftrightarrow \sqrt{x^{2}+1}=3\sqrt{x}\Leftrightarrow x^{2}-9x+1=0\Leftrightarrow x=\dfrac{9\pm\sqrt{77}}{2}\ (\mbox{TM})$.\\
Kết luận: tập nghiệm của phương trình là $\left\{\dfrac{9\pm\sqrt{77}}{2}\right\}$.
\item $\heva{&x^{2}+9y^{2}+8xy+24=0\\ & x-3y+xy=0}\Leftrightarrow \heva{&(x-3y)^{2}+14xy+24=0\\ & (x-3y)+xy=0}$.\\
Đặt $a=x-3y$, $b=xy$. Hệ phương trình trở thành $\heva{&a^{2}+14b+24=0&(1)\\ & a+b=0&(2)}$\\
$(2)\Rightarrow b=-a$. Thay vào phương trình $(1)$ ta được $a^{2}-14a+24=0\Leftrightarrow a=2$ hoặc $a=12$.
\begin{itemize}
\item Với $a=2\Rightarrow b=-2$. Ta có $\heva{&x-3y=2\\ &xy=-2}\Leftrightarrow \heva{&x=3y+2\\ &3y^{2}+2y+2=0}$ (Vô nghiệm).
\item Với $a=12\Rightarrow b=-12$. Ta có $\heva{&x-3y=12\\ &xy=-12}\Leftrightarrow \heva{&x=3y+12\\ &y^{2}+4y+4=0}\Leftrightarrow \heva{&x=6\\ &y=-2}$.
\end{itemize}
Vậy nghiệm của hệ là $(6;-2)$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên Tiền Giang, 2017]%[9D4G6]%[9D4K6]
\begin{enumerate}
\item Trong mặt phẳng tọa độ $Oxy$, cho parabol $(P):y=x^{2}$ và đường thẳng $d:y=2(m+1)x-m^{2}$ (với $m$ là tham số). Tìm tất cả các giá trị của $m$ để đường thẳng $d$ cắt $(P)$ tại hai điểm $A(x_{1};y_{1})$, $B(x_{2};y_{2})$ thỏa mãn $(x_{1}-m)^{2}+x_{2}=3m$.
\item Cho phương trình $x^{2}+mx-2=0$ (với $m$ là tham số). Chứng minh rằng phương trình luôn có hai nghiệm phân biệt $x_{1}$, $x_{2}$ với mọi $m$. Tìm tất cả các giá trị của $m$ sao cho biểu thức $A=(x_{1}^{2}-1)(x_{2}^{2}-4)$ đạt giá trị lớn nhất.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Phương trình hoành độ giao điểm $x^{2}-2(m+1)x+m^{2}=0\ (*)$.\\
$(P)$ cắt $d$ tại hai điểm phân biệt $\Leftrightarrow \Delta' >0\Leftrightarrow 2m+1>0\Leftrightarrow m>-\dfrac{1}{2}$.\\
Vì $x_{1}$ là nghiệm của phương trình $(*)$ nên $x_{1}^{2}-2(m+1)x_{1}+m^{2}=0\Rightarrow (x_{1}-m)^{2}=2x_{1}$.\\
Theo đề bài $(x_{1}-m)^{2}+x_{2}=3m\Rightarrow 2x_{1}+x_{2}=3m$. Theo Viet ta có $\heva{&x_{1}+x_{2}=2m+2\\ &x_{1}x_{2}=m^{2}}$.\\
Ta có $\heva{&2x_{1}+x_{2}=3m\\ & x_{1}+x_{2}=2m+2}\Leftrightarrow \heva{& x_{1}=m-2\\ &x_{2}=m+4}$.\\
Do vậy $(m-2)(m+4)=m^{2}\Rightarrow 2m=8\Rightarrow m=4$ (TM).
\item Vì phương trình có $ac=-2<0$ nên phương trình đã cho luôn có hai nghiệm phân biệt với mọi $m$.\\
Theo Viet ta có $\heva{&x_{1}+x_{2}=-m\\ &x_{1}x_{2}=-2}\Rightarrow x_{1}=\dfrac{-2}{x_{2}}$.\\
Ta có $A=\left(\dfrac{4}{x_{2}^{2}}-1\right)(x_{2}^{2}-4)=\dfrac{(4-x_{2}^{2})^{2}}{-x_{2}^{2}}=-x_{2}^{2}-\dfrac{16}{x_{2}^{2}}+8\leq -2\sqrt{x_{2}^{2}.\dfrac{16}{x_{2}^{2}}}+8=0$.\\
Dấu $=$ xảy ra khi và chỉ khi $x_{2}=\pm 2$.
\begin{itemize}
\item Với $x_{2}=2\Rightarrow x_{1}=-1\Rightarrow m=-1$.
\item Với $x_{2}=-2\Rightarrow x_{1}=1\Rightarrow m=1$.
\end{itemize}
Vậy $\max A=0$ khi $m=\pm 1$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên Tiền Giang, 2017]%[9D5G3]
Tìm tất cả các số nguyên tố $p$ sao cho $p^{2}+p+6$ là một số chính phương.
\loigiai{
Vì $p^{2}+p+6$ là số chính phương nên $p^{2}+p+6=k^{2}$ ($k$ nguyên dương).\\
Ta có $p^{2}+p+6=k^{2}\Rightarrow 4p^{2}+4p+24=4k^{2}\Rightarrow (2k)^{2}-(2p+1)^{2}=23\Rightarrow (2k+2p+1)(2k-2p-1)=23$.\\
Vì $2k+2p+1>2k-2p-1$ nên $\heva{& 2k+2p+1=23\\ &2k-2p-1=1}\Rightarrow \heva{&k=6\\ &p=5}$.\\
Vậy $p=5$ thì $p^{2}+p+6$ là số chính phương.
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên Tiền Giang, 2017]%[9H3K4]%[9H3K4]%[9H3G4]
Cho hai đường tròn $(O;R)$ và $(O';R')$ cắt nhau tại $A$, $B$ ($R<R'$). Kẻ tiếp tuyến chung $CD$ của $(O)$ và $(O')$ với $C$, $D$ là các tiếp điểm ($C$ nằm trên $(O)$ và $D$ nằm trên $(O')$). Qua $B$ kẻ cát tuyến song song với $CD$ cắt $(O)$ tại $E$, cắt $(O')$ tại $F$. Gọi $G$, $H$ theo thứ tự là giao điểm của $DA$, $CA$ với $EF$ và $I$ là giao điểm của $EC$ với $FD$.
\begin{enumerate}
\item Chứng minh $\Delta BCD=\Delta ICD$.
\item Gọi $K$ là giao điểm của $AB$ và $CD$. Chứng minh $K$ là trung điểm của $CD$.
\item Chứng minh $BI$ là đường trung trực của đoạn thẳng $GH$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzDefPoints{-1.5/0/O, 2.5/0/O', 0/1/A, 0/-1/B}
\tkzDefPointBy[homothety = center O' ratio 3.026](O)\tkzGetPoint{V}
\tkzTangent[from=V](O,A)\tkzGetPoints{C'}{C}
\tkzTangent[from=V](O',A)\tkzGetSecondPoint{D}
\tkzDefPointBy[homothety = center V ratio 2](C)\tkzGetPoint{y}
\tkzDefMidPoint(V,C)\tkzGetPoint{x}
\tkzDefLine[parallel=through B](C,D)\tkzGetPoint{b}
\tkzInterLC(B,b)(O,A)\tkzGetSecondPoint{E}
\tkzInterLC(B,b)(O',A)\tkzGetFirstPoint{F}
\tkzInterLL(F,D)(E,C)\tkzGetPoint{I}
\tkzInterLL(A,D)(E,F)\tkzGetPoint{G}
\tkzInterLL(A,C)(E,F)\tkzGetPoint{H}
\tkzInterLL(A,B)(C,D)\tkzGetPoint{K}

\tkzDrawCircle[radius](O,A)
\tkzDrawCircle[radius](O',A)
\tkzDrawPoints[fill=black](A,B,O,O',C,D,E,F,I,G,K,H)
\tkzDrawLine[add=1 and 1](B,b)
\tkzDrawSegments[thick](B,K x,y I,E I,F I,H I,B C,B C,H D,G O,C O',D)
\tkzLabelPoints[above](I,D,K,x,y)
\tkzLabelPoints[above left](C)
\tkzLabelPoints[above right](F)

\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Vì $CD\parallel EF$ nên $\wideparen{CE}=\wideparen{CB}$ và $\wideparen{DF}=\wideparen{DB}\Rightarrow \widehat{xCE}=\widehat{DCB}$, $\widehat{yDF}=\widehat{CDB}$.\\
Mà $\widehat{ICD}=\widehat{xCE}$, $\widehat{IDC}=\widehat{yDF}$ (đối đỉnh) nên $\widehat{DCB}=\widehat{ICD}$, $\widehat{IDC}=\widehat{CDB}$.\\
Xét $\Delta BCD$ và $\Delta ICD$ có $CD$ chung, $\widehat{DCB}=\widehat{ICD}$, $\widehat{IDC}=\widehat{CDB}$\\
$\Rightarrow \Delta BCD=\Delta ICD$ (g.c.g).
\item Xét $\Delta KCA$ và $\Delta KBC$ có $\widehat{KCA}=\widehat{KBC}$ và $\widehat{K}$ chung nên\\
$\Delta KCA\backsim\Delta KBC\Rightarrow KC^{2}=KA.KB$ (1)\\
Xét $\Delta KDA$ và $\Delta KBD$ có $\widehat{KDA}=\widehat{KBD}$ và $\widehat{K}$ chung nên\\
$\Delta KDA\backsim\Delta KBD\Rightarrow KD^{2}=KA.KB$ (2)\\
Từ $(1)$, $(2)$ ta có $KC=KD\Rightarrow K$ là trung điểm của $CD$.
\item Vì $CD\parallel EF$ nên $\dfrac{KD}{BG}=\dfrac{KA}{AB}=\dfrac{KC}{BH}$. Mà $KC=KD$ nên $BG=BH$ (3).\\
Do $\Delta BCD=\Delta ICD$ nên $CI=CB$, $DI=DB\Rightarrow CD$ là đường trung trực của đoạn $IB\Rightarrow CD\perp IB$.\\
Mà $CD\parallel EH$ nên $IB\perp GH$ (4).\\
Từ $(3)$, $(4)$ ta suy ra $IB$ là đường trung trực của đoạn $GH$.
\end{enumerate}
}
