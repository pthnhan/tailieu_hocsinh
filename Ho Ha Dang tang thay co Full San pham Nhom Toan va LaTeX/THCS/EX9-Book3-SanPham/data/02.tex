\end{itemize}

\subsection{Phương trình bậc nhất một ẩn}
Cần chú ý đến các dạng phương trình đưa được về phương trình bậc nhất một ẩn:
\subsubsection{Phương trình chứa ẩn ở mẫu thức}
Ở dạng này, giá trị tìm được của ẩn phải thỏa mãn điều kiện làm cho mẫu thức khác $0$ (điều kiện của phương trình).

\setcounter{vd}{12}
\begin{vd}
Giải phương trình ($a,b$ là tham số):
\begin{align*}
\dfrac{ax-1}{x-1}+\dfrac{b}{x+1}=\dfrac{a(x^2+1)}{x^2}-1 \tag{1}
\end{align*}
 \loigiai{
Điều kiện xác định (ĐKXĐ) của phương trình là $x \ne \pm 1$.\\
Với điều kiện đó thì $(1) \Leftrightarrow (ax-1)(x+1)+b(x-1)=a(x^2+1) \Leftrightarrow (a+b-1)x=a+b+1. \hfill (2)$\\
Nếu $a+b\ne 1$ thì $x=\dfrac{a+b+1}{a+b-1}$. Giá trị này là nghiệm nếu $\heva{&\dfrac{a+b+1}{a+b-1}\ne 1\\&\dfrac{a+b+1}{a+b-1}\ne -1} \Leftrightarrow a+b \ne 0$.\\
Nếu  $a + b=1$  thì trở thành  $0x=2$ , vô nghiệm.\\
Kết luận:\\
Với  $a + b \ne 1$  và  $a + b \ne 0$ , phương trình có nghiệm  $x=\dfrac{{a + b + 1}}{{a + b - 1}}$.\\
Còn lại vô nghiệm.
    }
\end{vd}

\subsubsection{Phương trình chứa ẩn trong dấu giá trị tuyệt đối}
Ở dạng này, ta thường khử dấu giá trị tuyệt đối theo định nghĩa
 $$|A| =\heva{&A&\text{nếu}\;A \ge 0\\&-A&\text{nếu}\;A < 0}$$ 
\begin{vd}
Một học sinh giải phương trình  $\left| x + 1 \right| + 3x= - 5$  (1) như sau:\\
 $(1) \Leftrightarrow \left| x + 1\right|= - 3x - 5 \Leftrightarrow \hoac{&x + 1= - 3x - 5\\&x + 1=3x + 5} \Leftrightarrow \hoac{&4x= - 6\\ &- 2x=4}\Leftrightarrow \hoac{&x= - \dfrac{3}{2}\\&x= - 2.}$\\
Cách giải trên có đúng không?
\loigiai{
Giá trị  $x= - \dfrac{3}{2}$  không thỏa mãn (1) nên loại.\\
Cách giải đúng như sau:\\
Cách 1. Với điều kiện  $ - 3x - 5 \ge 0$  (2) thì \\
 $\left| x + 1 \right|= - 3x - 5 \Leftrightarrow \hoac{&x + 1= - 3x - 5\\&x + 1=3x + 5.}$\\
Giải như trên, loại  $x= - \dfrac{3}{2}$  vì trái với (2), chọn  $x= - 2$  vì thỏa mãn (2).\\
Cách 2.\\
Xét  $x \ge  - 1$  thì  $(1) \Leftrightarrow x + 1 + 3x= - 5 \Leftrightarrow x= - \dfrac{3}{2}$, không thỏa mãn  $x \ge  - 1$.\\
Xét  $x <  - 1$  thì  $(1) \Leftrightarrow  - x - 1 + 3x= - 5 \Leftrightarrow x= - 2$, thỏa mãn  $x <  - 1$.\\
Kết luận:  $x= - 2$.\\
Lưu ý:  $|f(x)|=g(x) \Leftrightarrow \heva{&f(x)= \pm g(x)\\&g(x) \ge 0}$.
}
\end{vd}

\begin{vd}
Tìm giá trị của tham số $a$ để phương trình  $|2x - a|=x + 1$  (1) có nghiệm duy nhất.
\loigiai{
$(1) \Leftrightarrow \hoac{&\heva{&x \ge \dfrac{a}{2}\\&2x - a=x + 1}&\text{(I)}\\&\heva{&x \le \dfrac{a}{2}\\&a - 2x=x + 1}&\text{(II)}}$ \\
$\text{(I)} \Leftrightarrow \heva{&x=a + 1\\&x \ge \dfrac{a}{2}}, a + 1 \ge \dfrac{a}{2} \Leftrightarrow a \ge  - 2$.\\
$\text{(II)}\Leftrightarrow \heva{&x=\dfrac{{a - 1}}{3}\\&x \le \dfrac{a}{2}}, \dfrac{{a - 1}}{3} \le \dfrac{a}{2} \Leftrightarrow a \ge  - 2$.\\
Để (1) có nghiệm duy nhất thì  $\heva{&a + 1=\dfrac{{a - 1}}{3}\\&a \ge  - 2} \Leftrightarrow a= - 2$.
}
\end{vd}

\subsection{Phương trình bậc hai một ẩn}
Cần chú ý đến các kiến thức sau:
\begin{enumerate}[1)]
\item Điều kiện để phương trình bậc hai có nghiệm là  $\Delta  \ge 0$.
\item Hệ thức Vi -ét: Nếu phương trình  $ax^2 + bx + c=0 \,(a \ne 0)$  có các nghiệm  $x_1$ và  $x_2$  thì  $x_1+x_2=-\dfrac{b}{a}$  và  $x_1.x_2=\dfrac{c}{a}$.
\item Cho phương trình  $ax^2 + bx + c=0$  có  $a \ne 0$.
\begin{description}
\item[-] Nếu  $a + b + c=0$  thì phương trình có hai nghiệm  $x_1=1$  và  $x_2=\dfrac{c}{a}$.
\item[-] Nếu  $a - b + c=0$  thì phương trình có hai nghiệm  $x_1= - 1$  và  $x_2= - \dfrac{c}{a}$.
\end{description}
\end{enumerate}

\begin{vd}
Cho phương trình  $x^2-2(m + 1)x + (m^2 + 2m)=0$.\\
Tìm giá trị của $m$ để các nghiệm  $x_1, x_2$  của phương trình thỏa mãn  $\left|x_1^3 - x_2^3\right|=8$.
\loigiai{
Phương trình đã cho có nghiệm với mọi $m$ vì $\Delta '=(m + 1)^2- (m^2 + 2m)=1 > 0$.\\
Theo hệ thức Vi-ét,  $x_1 + x_2= 2m + 2,x_1.x_2= m^2 + 2m$.\\
Ta có  $(x_1 - x_2)^2= (x_1 + x_2)^2- 4x_1.x_2= (2m + 2)^2 - 4(m^2 + 2m)=4$  nên  $\left| x_1- x_2\right|=2$.\\
Ta lại có  $x_1^2 + x_1.x_2 + x_2^2=(x_1 + x_2)^2- x_1.x_2=(2m + 2)^2 - (m^2 + 2m)=3m^2 + 6m + 4 > 0$.\\
Do đó  $\left| x_1^3 - x_2^3 \right|=2(3m^2+ 6m + 4)$.\\
Giải phương trình  $2(3m^2 + 6m + 4) = 8$  được  $m^2 + 2m=0 \Leftrightarrow m(m + 2)=0 \Leftrightarrow m=0$ hoặc  $m= - 2$.\\
Đáp số:  $m \in \left\{0; - 2 \right\}$.
}
\end{vd}

\begin{vd}
Cho phương trình  $x^2 + mx + 1 = 0$. Tìm giá trị của $m$ để các nghiệm  $x_1, x_2$  của phương trình thỏa mãn  $x_1^4 + x_2^4 = 2$.
\loigiai{
Điều kiện để phương trình  $x^2 + mx + 1 = 0$  có nghiệm là  $\Delta  \ge 0 \Leftrightarrow m^2 - 4 \ge 0 \Leftrightarrow m^2 \ge 4$.\\
Theo hệ thức Vi - ét:  $x_1 + x_2 =  - m$  và  $x_1.x_2 = 1$.\\
Ta có  $x_1^2 + x_2^2 = (x_1 + x_2)^2 - 2x_1.x_2= ( - m)^2 - 2$  nên  $$x_1^4 + x_2^4 = (x_1^2 + x_2^2)^2 - 2x_1^2x_2^2 = (m^2 - 2)^2 - 2 = m^4 - 4m^2 + 2.$$
Giải phương trình  $x_1^4 + x_2^4 = 2 \Leftrightarrow m^4 - 4m^2 + 2 = 2 \Leftrightarrow m^2(m^2 - 4) = 0$.\\
Loại  $m^2 = 0$  vì trái với (1), ta được  $m^2 = 4 \Leftrightarrow m =  \pm 2$.\\ 
Đáp số:  $m =  \pm 2$. 
}
\end{vd}

\begin{vd}
Cho các phương trình
\begin{align*}
&x^2 + mx - 5 = 0 \tag{1}\\ 
\text{và}\quad & 5x^2 - mx - 1 = 0 \tag{2}
\end{align*}
\begin{enumerate}
\item Chứng minh rằng các phương trình trên có nghiệm.
\item Gọi  $x_1$  là nghiệm dương của (1),  $x_2$  là nghiệm dương của (2). Chứng minh rằng  $x_1 + x_2 \ge 2$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Các phương trình (1) và (2) đều có  $ac < 0$  nên đều có hai nghiệm trái dấu.
\item Do  $x_1$  là nghiệm của (1) nên  $x_1^2 + mx_1 - 5 = 0 \Rightarrow 1 + m.\dfrac{1}{x_1} - \dfrac{5}{x_1^2} = 0$ 
 $ \Rightarrow 5\left(\dfrac{1}{x_1} \right)^2 - m\dfrac{1}{x_1} - 1 = 0 \Rightarrow \dfrac{1}{x_1}$ là nghiệm dương của (2)
 $ \Rightarrow x_2 = \dfrac{1}{x_1} \Rightarrow x_1.x_2 = 1.$ \\
Do  $x_1, x_2$  dương nên  $x_1 + x_2 \ge 2\sqrt {x_1x_2}  = 2$.
\end{enumerate}
}
\end{vd}

\begin{vd}Cho các phương trình 
\begin{align*}
&x^2 + x + a = 0 \tag{1}\\ 
\text{và}\quad & x^2 + 4x + b = 0 \tag{2}
\end{align*}
Tìm giá trị của $a$ và $b$ sao cho các nghiệm  $x_1, x_2$ của phương trình (1) và các nghiệm  $x_3, x_4$ của phương trình (2) thỏa mãn:
 $\dfrac{x_4}{x_3} = \dfrac{x_3}{x_2} = \dfrac{x_2}{x_1}$.
\loigiai{
Điều kiện để (1) và (2) có nghiệm là
 $\heva{&1 - 4a \ge 0\\&4 - b \ge 0} \Leftrightarrow \heva{&a \le \dfrac{1}{4}\\&b \le 4.}$\\
Đặt   $\dfrac{x_4}{x_3} = \dfrac{x_3}{x_2} = \dfrac{x_2}{x_1}=k$  thì  $x_2 = kx_1,x_3 = kx_2 = k^2x_1,x_4 = kx_3 = k^3x_1$.\\
Theo hệ thức Vi - ét:
 $x_1 + x_2 =  - 1 \Rightarrow x_1 + kx_1 =  - 1 \Rightarrow x_1(1 + k) =  - 1$ \hfill		(3)\\
 $x_3 + x_4 =  - 4 \Rightarrow k^2x_1 + k^3x_1 =  - 4 \Rightarrow k^2x_1(1 + k) =  - 4$ 	\hfill (4)\\
Từ (3) và (4) suy ra  $k^2 = 4$  nên  $k =  \pm 2$.\\
- Xét  $k = 2$, thay vào (3) được  $x_1 =  - \dfrac{1}{3},x_2 =  - \dfrac{2}{3},x_3 =  - \dfrac{4}{3},x_4=  - \dfrac{8}{3}.$ \\
Suy ra  $a = x_1x_2 = \dfrac{2}{9},b = x_3x_4 = \dfrac{32}{9}$, thỏa mãn  $a \le \dfrac{1}{4}$  và  $b \le 4$.\\
- Xét  $k = -2$, thay vào (3) được  $x_1 = 1, x_2 = -2, x_3 =4, x_4=-8.$ \\
Suy ra  $a = x_1x_2 =-2, b = x_3x_4 =-32$, thỏa mãn  $a \le \dfrac{1}{4}$  và  $b \le 4$.\\
Đáp số:  $a = \dfrac{2}{9},b = \dfrac{32}{9}$  hoặc  $a =  - 2, b =  - 32$.
}
\end{vd}

\subsection{Quan hệ giữa Parabol và đường thẳng}
Cho parabol  $y = ax^2 \; (a \ne 0)$ và đường thẳng  $y = mx + n$. Hoành độ giao điểm của parabol và đường thẳng là nghiệm của phương trình  $ax^2 = mx + n$  hay  $ax^2-mx-n=0$. \hfill	(1)\\
Nếu phương trình (1) có hai nghiệm phân biệt thì đường thẳng cắt parabol.\\
Nếu phương trình (1) có nghiệm kép thì đường thẳng tiếp xúc với parabol.\\
Nếu phương trình (1) vô nghiệm thì đường thẳng không giao với parabol.

\begin{vd}
Cho parabol  $y =x^2$. Gọi $A$ và $B$ là hai điểm thuộc parabol có hoành độ theo thứ tự là $a$ và $b$. Gọi $C$ là điểm thuộc parabol có hoành độ bằng  $a + b$. Chứng minh rằng $OC$ song song với $AB$.
\loigiai{
\immini{
Kẻ $AE, BF, CK$ vuông góc với $Ox$, kẻ $AH$ vuông góc với $BF$.\\
Ta có  $A(a;a^2), B(b;b^2), C(a + b,(a + b)^2)$.\\
Đường thẳng $AB$ có hệ số góc là  $$m = \dfrac{BH}{AH} = \dfrac{b^2 - a^2}{b - a} = b + a$$
Đường thẳng $OC$ có hệ số góc là  $$n = \dfrac{CK}{OK} = \dfrac{(a + b)^2}{a + b} = a + b$$
Do  $m = n$  nên  $AB \parallel OC$.
}

 \large \textbf{BÀI TẬP}

\setcounter{bt}{18}
\begin{bt}
Giải các phương trình sau:
\begin{enumerate}
\item $\dfrac{x - b - c}{a} + \dfrac{x - c - a}{b} + \dfrac{x - a - b}{c} = 3$. 
\item $\dfrac{x - a}{bc} + \dfrac{x - b}{ac} + \dfrac{x - c}{ab} = 2\left(\dfrac{1}{a} + \dfrac{1}{b} + \dfrac{1}{c}\right)$.
\item $\dfrac{x - a}{x - b} + \dfrac{x - b}{x - a} = 2$.
\end{enumerate}
 \loigiai{
\begin{enumerate}
\item Trừ $1$ vào mỗi phân thức ở vế trái, trừ $3$ vào vế phải.\\
Rút gọn được $(x-a-b-c)\left(\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}\right)=0$.\\
Nếu $\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c} \ne 0$ thì phương trình có nghiệm duy nhất $x=a+b+c$.\\
Nếu $\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c} = 0$ thì phương trình có nghiệm là số thực bất kì.
\item $\dfrac{x}{bc}+\dfrac{x}{ac}+\dfrac{x}{ab}=\dfrac{a}{bc}+\dfrac{b}{ac}+\dfrac{c}{ab}+\dfrac{2}{a}+\dfrac{2}{b}+\dfrac{2}{c} \Leftrightarrow x.\dfrac{a+b+c}{abc}=\dfrac{a^2+b^2+c^2+2bc+2ac+2ab}{abc} \Leftrightarrow (a+b+c)x=(a+b+c)^2$.\\
Nếu $a+b+c \ne 0$ (và $a, b, c \ne 0$) thì phương trình có nghiệm duy nhất $x=a+b+c$.\\
Nếu $a+b+c = 0$ (và $a, b, c \ne 0$) thì phương trình có nghiệm là số thực bất kì.
\item ĐKXĐ: $x \ne a, x \ne b$.\\
Đặt $x-a=y \ne 0$, đặt $x-b \ne 0$. Ta có\\
$\dfrac{y}{z}+\dfrac{z}{y}=2 \Leftrightarrow y^2+z^2=2yz \Leftrightarrow (y-z)^2=0 \Leftrightarrow y=z$.\\
Ta có $x-a=x-b \Leftrightarrow 0x=a-b$.\\
Nếu $a=b$ thì phương trình có nghiệm là số thực bất kì khác $a$.\\
Nếu $a \ne b$ thì phương trình vô nghiệm.
\end{enumerate}
}
\end{bt}
