\subsection{Lũy thừa, hàm số lũy thừa}
\begin{ex}%[Nguyễn Cao Cường]%[2D2B1-2]
	Cho $a$, $b$ là các số thực dương, $a\neq 0$ và $m$, $n$ là các số tùy ý. Công thức nào dưới đây \textbf{sai}?
	\choice
	{$a^m\cdot a^n=a^{m+n}$}
	{$\dfrac{a^m}{a^n}=a^{m-n}$}
	{\True $\left(a^m\right)^n=a^{m+n}$}
	{$(ab)^m=a^m\cdot b^m$}
	\loigiai{
		Các công thức cần nhớ: \\
		Với $a$, $b\in \mathbb{R}$; $a\neq 0, b\neq 0$; $m$, $n\in \mathbb{R}$, ta có:
		$$a^{m}\cdot a^n=a^{m+n}; \dfrac{a^m}{a^n}=a^{m-n}; (a^m)^n=a^{m\cdot n}; (ab)^m=a^m\cdot b^m;\left(\dfrac{a}{b}\right)^m=\dfrac{a^m}{b^m}.$$
	}
\end{ex}
\begin{ex}%[Nguyễn Cao Cường]%[2D2K1-2]
	Khẳng định nào sau đây là \textbf{sai}?
	\choice
	{Có một căn bậc $n$ của số $0$ là $0$}
	{$-\dfrac{1}{3}$ là căn bậc $5$ của $-\dfrac{1}{243}$}
	{\True $4$ có một căn bậc hai}
	{Các căn bậc $8$ của $2$ được viết là $\pm \sqrt[8]{2}$}
	\loigiai{
		Nhắc lại rằng:
		\begin{itemize}
			\item Nếu $n$ là số lẻ và $b\in \mathbb{R}$ thì có duy nhất một căn bậc $n$ của $b$, kí hiệu là $\sqrt[n]{b}$.
			\item Nếu $n$ là số chẵn:
			\begin{itemize}
				\item [-] Nếu $b<0$ thì không tồn tại căn bậc $n$ của $b$.
				\item [-] Nếu $b=0$ thì có một căn bậc $n$ của $b$ là $0$.
				\item [-] Nếu $b>0$ thì có hai cặn bậc $n$ của $b$ là $\sqrt[n]{b}$ và $-\sqrt[n]{b}$.
			\end{itemize}
		\end{itemize}
	}
\end{ex}


\begin{ex}%[Nguyễn Cao Cường]%[2D2K2-3]
	Cho $\alpha$ là một số thực và hàm số $y=\dfrac{1}{x^{\frac{2\alpha-1}{\alpha}}}$ đồng biến trên $(0; +\infty)$. Khẳng định nào sau đây là đúng?
	\choice
	{$\alpha <0$}
	{\True $0<\alpha<\dfrac{1}{2}$}
	{$\dfrac{1}{2}<\alpha<1$}
	{$\alpha>1$}
	\loigiai{
		Viết lại hàm số dưới dạng hàm lũy thừa: $y=x^{\frac{1-2\alpha}{\alpha}}$. Hàm số đồng biến khi và chỉ khi $$\dfrac{1-2\alpha}{\alpha}>0 \Leftrightarrow 0<\alpha<\dfrac{1}{2}.$$
	}
\end{ex}

\begin{ex}%[Nguyễn Cao Cường]%[2D2B2-3]
	Cho hàm số $y=x^{-\sqrt{2}}$. Hỏi khẳng định nào dưới đây là khẳng định đúng ?
	\choice
	{Đồ thị hàm số đã cho không có tiệm cận}
	{Đồ thị hàm số đã cho có một tiệm cận ngang và không có tiệm cận đứng}
	{Đồ thị hàm số đã cho không có tiệm cận ngang và có một tiệm cận đứng}
	{\True Đồ thị hàm số đã cho có một tiệm cận ngang và một tiệm cận đứng}
	\loigiai{ Tập khảo sát: $(0; +\infty)$.\\
		Do $-\sqrt{2}<0$ nên ta có hai giới hạn: $\displaystyle\lim_{x\to 0^+}x^{-\sqrt{2}}=+\infty$ và $\displaystyle\lim_{x\to +\infty}x^{-\sqrt{2}}=0$.\\
		Vậy trục $Ox$ là tiệm cận ngang và trục $Oy$ là tiệm cận đứng.
	}
\end{ex}

\begin{ex}%[Nguyễn Cao Cường]%[2D2B1-2]
	Cho $n$ nguyên dương thỏa mãn $n\geq 2$, khẳng định nào sau đây là khẳng định đúng?
	\choice
	{\True $a^{\frac{1}{n}}=\sqrt[n]{a}, \forall a>0$}
	{$a^{\frac{1}{n}}=\sqrt[n]{a}, \forall a\neq 0$}
	{$a^{\frac{1}{n}}=\sqrt[n]{a}, \forall a\geq 0$}
	{$a^{\frac{1}{n}}=\sqrt[n]{a}, \forall a\in \mathbb{R}$}
	\loigiai{ Áp dụng định nghĩa lũy thừa với số mũ hữu tỷ: $a^{\frac{m}{n}}=\sqrt[n]{a^m}$, với $a>0$.
	}
\end{ex}

\subsection{Hàm số mũ}
\begin{ex}%[Lê Minh An, Ex-LT1]%[2D2Y4-3]
	Cho hàm số $y=a^x$ ($a>0$, $a\neq 1$) đồng biến trên $\mathbb{R}$. Khẳng định nào sau đây đúng?
	\choice{$\ln a>1$}
	{$\lim\limits_{x\rightarrow -\infty} a^x=-\infty$}
	{$\lim\limits_{x\rightarrow +\infty} a^x=0$}
	{\True $\ln a>0$}
	\loigiai{
		Hàm số $y=a^x$ ($a>0$, $a\neq 1$) đồng biến trên $\mathbb{R}$ nên $a>1\Leftrightarrow \ln a>0$.\\
		Ngoài ra ta cũng có $\lim\limits_{x\rightarrow -\infty} a^x=0$, $\lim\limits_{x\rightarrow +\infty} a^x=+\infty$.
	}
\end{ex}

\begin{ex}%[Lê Minh An, Ex-LT1]%[2D2Y4-3]
	Đồ thị hàm số $y=a^x+1$ có một đường tiệm cận là
	\choice{\True $y=1$}
	{$y=-1$}
	{$y=0$}
	{$x=0$}
	\loigiai{
		Đồ thị $(C)$ của hàm số $y=a^x$ chỉ có một đường tiệm cận ngang $y=0$.\\
		Mà đồ thị $(C')$ của hàm số $y=a^x+1$ có được bằng cách tịnh tiến $(C)$ lên một đơn vị.\\
		Nên $(C')$ chỉ có một đường tiệm cận $y=1$.
	}
\end{ex}

\begin{ex}%[Lê Minh An, Ex-LT1]%[2D2B4-3]
	Cho hàm số $y=a^x$ ($a>0$, $a\neq 1$) có đồ thị là đường cong $(C)$. Biết rằng tiếp tuyến của $(C)$ tại một điểm bất kì trên đồ thị luôn tạo với trục $Ox$ một góc nhọn. Khẳng định nào sau đây là \textbf{sai}? 
	\choice{Hàm số $y=a^x$ đồng biến trên $\mathbb{R}$}
	{$\log a>0$}
	{Hàm số $y=a^{-x}$ nghịch biến trên $\mathbb{R}$}
	{\True $\ln a>1$}
	\loigiai{
		Tiếp tuyến tại một điểm bất kì của đồ thị hàm số $y=a^x$ có hệ số góc là $k=y'=a^x\cdot \ln a$.\\
		Do tiếp tuyến tạo với $Ox$ một góc nhọn nên $k>0\Leftrightarrow \ln a>0\Leftrightarrow a>1$.\\
		Điều này có nghĩa là $\log a>0$, hàm số $y=a^x$ đồng biến trên $\mathbb{R}$ và hàm số $y=a^{-x}$ nghịch biến trên $\mathbb{R}$.
	}
\end{ex}

\subsection{Lô-ga-rít}
\begin{ex}%[GT12-2]%[Lê Mạnh Thắng, EX-LT1]%[2D2B3-2]
	Cho $a,b$ là hai số dương với $a\neq 1$. Khẳng định nào sau đây là \textbf{sai}?
	\choice
	{$a^{\log_a{b}}=b$}
	{$\log_a{a^b}=b$}
	{\True $\log_b{b}=1$}
	{$b^{\log_a{1}}=1$}
	\loigiai{
		Theo định nghĩa, số $\alpha$ thỏa mãn đẳng thức $a^{\alpha}=b$ được gọi là lôgarit cơ số $a$ của $b$, kí hiệu là $\log_a{b}$. Do đó, $\alpha = \log_a{b} \,\Leftrightarrow\, a^{\alpha}=b$. Từ đây, ta có
		\begin{itemize}
			\item $a^{\log_a{b}}=a^{\alpha}=b$.
			\item $\log_a{a^b}=b \,\Leftrightarrow\, a^b=a^b$.
			\item $\log_a{1}=\log_a{a^0}=0 \,\Rightarrow\, b^{\log_a{1}}=b^0=1$.
			\item $\log_b{b}$ không xác định duy nhất khi $b=1$, do đó $\log_b{b}=1$ là mệnh đề sai.
		\end{itemize}		
	}
\end{ex}
\begin{ex}%[GT12-2]%[Lê Mạnh Thắng, EX-LT1]%[2D2Y3-2]
	Cho $a,b,c$ là ba số dương với $a\neq 1$. Khẳng định nào sau đây là đúng?
	\choice
	{$\log_a{(bc)}=\log_a{b}\cdot \log_a{c}$}
	{$\log_a{\dfrac{b}{c}}=\dfrac{\log_a{b}}{\log_a{c}}$}
	{\True $\log_a{b^c}=c\log_a{b}$}
	{$\log_a{(b+c)}=\log_a{b}+\log_a{c}$}
	\loigiai{
		Theo quy tắc tính lôgarit, ta có
		\begin{itemize}
			\item $\log_a{(bc)}=\log_a{b}+\log_a{c}$.
			\item $\log_a{\dfrac{b}{c}}= \log_a{b}-\log_a{c}$.
			\item $\log_a{b^{\alpha}}=\alpha\log_a{b}$, với $\alpha \in \mathbb{R}$.
		\end{itemize}		
	}
\end{ex}
\begin{ex}%[GT12-2]%[Lê Mạnh Thắng, EX-LT1]%[2D2B3-2]
	Phát biểu nào sau đây là \textbf{sai}?
	\choice
	{Lôgarit cơ số $2$ của $3$ là một số mà $2$ lũy thừa nó bằng $3$}
	{Không tồn tại Lôgarit cơ số $1$ của một số thực dương}
	{\True Với mọi số thực dương, lôgarit thập phân của nó luôn khác lôgarit tự nhiên của nó}
	{Tồn tại duy nhất một số thực mà $3$ lũy thừa nó bằng $2$}
	\loigiai{
		Với $a$ là số thực dương, $a\neq1$ ta có $\log_a{1}=0$. Do đó $\log{1}=\ln{1}=0$.
	}
\end{ex}
\begin{ex}%[GT12-2]%[Lê Mạnh Thắng, EX-LT1]%[2D2Y3-2]
	Cho $a,b,c$ là ba số dương với $a\neq 1$. Khẳng định nào sau đây là \textbf{sai}?
	\choice
	{\True $\log_a{b}=\dfrac{\log_b{c}}{\log_a{c}}$, $(b\neq1, c\neq1)$}
	{$\log_a{b}=\dfrac{1}{\log_b{a}}$, $(b\neq1)$}
	{$\log_a{b}\cdot \log_b{c} \cdot \log_c{a}=1$, $(b\neq1, c\neq1)$}
	{$\log_{a^{\alpha}}{b}=\dfrac{1}{\alpha}\log_a{b}$, $(\alpha \neq 0)$}
	\loigiai{
		Theo công thức đổi cơ số, ta có
		\begin{itemize}
			\item $\log_a{b}=\dfrac{\log_c{a}}{\log_c{b}}$, $\log_a{b}=\dfrac{\log_a{c}}{\log_b{c}}, \;(b\neq1, c\neq1)$.
			\item $\log_a{b}\cdot \log_b{c} \cdot \log_c{a}=\log_a{a}=1$.
		\end{itemize}
	}
\end{ex}
\begin{ex}%[GT12-2]%[Lê Mạnh Thắng, EX-LT1]%[2D2B3-2]
	Cho $a$ là số thực khác $0$. Khẳng định nào sau đây là đúng?
	\choice
	{$\log_{\mathrm{e}^6}{a^2}=\dfrac{1}{3}\ln{a}$}
	{$\log_{\mathrm{e}^6}{a^2}=\dfrac{1}{3}\log{\left|a\right|}$}
	{$\log_{\mathrm{e}^6}{a^2}=12\ln{a}$}
	{\True $\log_{\mathrm{e}^6}{a^2}=\dfrac{1}{3}\ln{\left|a\right|}$}
	\loigiai{
		Vì $a\neq0$ nên $\log_{\mathrm{e}^6}{a^2}=\dfrac{2}{6}\log_{\mathrm{e}}{\left|a\right|}=\dfrac{1}{3}\ln{\left|a\right|}$.
	}
\end{ex}


\subsection{Hàm số lô-ga-rít}
\begin{ex}%[Nguyễn Đắc Giáp]%[2D2Y4-1]
	Cho $0<a<1$. Tìm mệnh đề đúng trong các mệnh đề sau.
	\choice
	{Tập xác định của hàm số $y=\log_ax$ là $\mathscr{D}=\mathbb{R}$}
	{Tập xác định của hàm số $y=a^x$ là $\mathscr{D}=\left(0;+\infty\right)$}
	{Tập giá trị của hàm số $y=a^x$ là $\mathscr{T}=\mathbb{R}$}
	{\True Tập giá trị của hàm số $y=\log_ax$ là $\mathscr{T}=\mathbb{R}$}
	\loigiai{
		\begin{itemize}
			\item Mệnh đề ‘‘Tập xác định của hàm số $y=\log_ax$ là $\mathscr{D}=\mathbb{R}$'' sai vì hàm số $y=\log_ax$ có tập xác định là $\mathscr{D}=\left(0;+\infty\right)$.
			\item Mệnh đề ‘‘Tập xác định của hàm số $y=a^x$ là $\mathscr{D}=\left(0;+\infty\right)$'' sai vì hàm số $y=a^x$ có tập xác định là $\mathscr{D}=\mathbb{R}$.
			\item Mệnh đề ‘‘Tập giá trị của hàm số $y=a^x$ là $\mathscr{T}=\mathbb{R}$'' sai vì hàm số $y=a^x$ có tập giá trị là $\mathscr{T}=\left(0;+\infty\right)$.
			\item Mệnh đề ‘‘Tập giá trị của hàm số $y=\log_ax$ là $\mathscr{T}=\mathbb{R}$'' đúng vì theo tính chất của hàm số lôgarit.
			
		\end{itemize}
	}
\end{ex}

\begin{ex}%[Nguyễn Đắc Giáp]%[2D2K4-3]
	Cho hai hàm số $f(x)=2^x$ và $g(x)=\log_2x$. Xét các mệnh đề sau:
	\begin{itemize}
		\item $(I)$ Đồ thị hai hàm số đối xứng nhau qua đường thẳng $y=x$.
		\item $\left(II\right)$ Tập xác định của hai hàm số trên là $\mathscr{D}=\mathbb{R}$.
		\item $\left(III\right)$ Đồ thị hai hàm số đều cắt trục hoành tại đúng một điểm.
		\item $\left(IV\right)$ Hai hàm số đều đồng biến trên tập xác định của nó.
	\end{itemize}
	Có bao nhiêu mệnh đề đúng trong các mệnh đề trên?
	\choice
	{\True $2$}
	{$3$}
	{$1$}
	{$4$}
	\loigiai{
		\begin{itemize}
			\item Mệnh đề $(I)$ đúng vì $(x_0;2^{x_0})$ thuộc đồ thị hàm số $y=2^x$ khi và chỉ khi $(2^{x_0};x_0)$ thuộc đồ thị hàm số $y=\log_2x$.
			\item Mệnh đề $(II)$ sai vì hàm số $y=\log_2x$ có tập xác định là $\mathscr{D}=\left(0;+\infty\right)$.
			\item Mệnh đề $(III)$ sai vì $y=2^x>0$ với mọi số thực $x$ nên đồ thị hàm số $y=2^x$ không cắt trục hoành.
			\item Mệnh đề $(IV)$ đúng vì cơ số $a=2>1$ nên hàm số $y=\log_2x$ và $y=2^x$ đồng biến trên tập xác định của nó.
		\end{itemize}
	}
\end{ex}
\begin{ex}%[Nguyễn Đắc Giáp]%[2D2Y4-3]

\begin{ex}%[Nguyễn Đắc Giáp]%[2D2B4-3]
	Cho hàm số $y=\log_{\left(\sqrt{2}-1\right)}x$. Xét các mệnh đề sau:
	\begin{itemize}
		\item $(I)$ Hàm số nghịch biến trên khoảng $\left(0;+\infty\right)$.
		\item $\left(II\right)$ Hàm số có tập giá trị là $\left(0;+\infty\right)$.
		\item $\left(III\right)$ Đồ thị hàm số nhận trục $Oy$ là tiệm cận đứng.
		\item $\left(IV\right)$ Hàm số có tập giá trị là $\left(-\infty;+\infty\right)$.
	\end{itemize}
	Số mệnh đề đúng trong các mệnh đề trên là
	\choice
	{$0$}
	{$1$}
	{$2$}
	{\True $3$}
	\loigiai{
		\begin{itemize}
			\item Mệnh đề $(I)$ đúng vì hàm số $y=\log_{\left({\sqrt{2}-1}\right)}x$ có cơ số $a=\sqrt{2}-1\in \left({0;1}\right)$ nên nghịch biến trên khoảng $\left({0;+\infty}\right)$.
			\item Mệnh đề $\left(II\right)$ sai vì hàm số $y=\log_{\left(\sqrt{2}-1\right)}x$ có tập giá trị là $\mathbb{R}$.

		\end{itemize}
	}

\subsection{Phương trình, bất phương trình mũ, lô-ga-rít}
\begin{ex}%[Thanh Hai Le]%[2D2B6-1]
	Cho bất phương trình $\log_{a^2} x>\log_{a^2} 2$, $a$ là tham số. Điều kiện của $a$ để bất phương trình có nghiệm $0<x< 2$ là
	\choice
	{\True $\heva{&-1<a<1 \\& a\neq 0}$}
	{$0<a<1$}
	{$a>0$}
	{Với mọi $a\in \mathbb{R}$}
	\loigiai{
		Điều kiện $x>0$.
		\begin{itemize}
			\item Xét $a^2>1\Leftrightarrow \hoac{& x<-1\\& x>1}$, $\log_{a^2} x>\log_{a^2} 2\Leftrightarrow x>2$.
			\item Xét $0<a^2<1\Leftrightarrow \heva{&-1<a<1\\& a\neq 0}$, $\log_{a^2} x>\log_{a^2} 2\Leftrightarrow x<2$.
		\end{itemize}
		Vậy $\heva{&-1<a<1 \\& a\neq 0}$ thì phương trình có nghiệm $0<x<2$.
	}
\end{ex}
\begin{ex}%[Thanh Hai Le]%[Đề HK1, Nguyễn Thượng Hiền, Hồ Chí Minh 17-18]%[Trần Văn A]%[2D2B6-1]
	Cho các mệnh đề sau có bao nhiêu mệnh đề đúng?
	\begin{multicols}{3}
		\begin{enumerate}[(I)]
			\item $2^{\sqrt{x}}>2 \Leftrightarrow \sqrt{x} > 1$
			\item $a^n \ge 0, \; \forall a,n \in \mathbb{Z}$
			\item $a^n \le a^{n+1}, \forall a>0, n \in \mathbb{R}$
		\end{enumerate}
	\end{multicols}
	
	\choice
	{$0$}
	{\True $1$}
	{$2$}
	{$3$}
	\loigiai{
		$(I)$ đúng vì $2>1$.\\
		$(II)$ sai vì $0^0$ không có nghĩa.\\
		$(III)$ sai vì $0,5^2>0,5^3$.
	}
\end{ex}
\begin{ex}%[Thanh Hai Le]%[2D2Y6-1]
	Nếu $a^{\sqrt{3}} > a^{\sqrt{7}}$ thì kết luận nào sau đây là đúng ?
	\choice
	{$a<1$}
	{\True $0<a<1$}
	{$a>1$}
	{$ \sqrt{3}<a<\sqrt{7}$}
	\loigiai{
		Điều kiện xác định là $ a>0, a \ne 1 $.\\
		Do $\sqrt{3} <\sqrt{7}$ nên suy ra $ a<1 $. Vậy $ 0<a<1 $.\\
		Đáp án $a<1$ sai vì thiếu điều kiện $ a>0 $.\\
		Đáp án $ a>1 $ sai vì $\sqrt{3} <\sqrt{7}$.\\
		Đáp án $ \sqrt{3}<a<\sqrt{7}$ sai vì không đúng quy tắc giải bất phương trình mũ.
	}
\end{ex}
\begin{ex}%[Thanh Hai Le]%[2D2Y6-1]
	Cho $a>1$. Trong các khẳng định sau đây, khẳng định nào sai?
	\choice
	{$a^x>1$ khi $x>0$}
	{$0<a^x<1$ khi $x<0$}
	{Nếu $x_1<x_2$ thì $a^{x_1}<a^{x_2}$}
	{$\left(\dfrac{1}{a}\right)^{-3}<\left(\dfrac{1}{a}\right)^3$}
	\loigiai{
		Khẳng định "$a^x>1$ khi $x>0$" đúng vì $a>1$.\\
		Khẳng định "$0<a^x<1$ khi $x<0$" đúng vì $a>1$.\\
		Khẳng định "Nếu $x_1<x_2$ thì $a^{x_1}<a^{x_2}$" đúng vì $a>1$.\\
		Khẳng định "$\left(\dfrac{1}{a}\right)^{-3}<\left(\dfrac{1}{a}\right)^3$" sai vì $\heva{&\dfrac{1}{a}<1\\&-3<3}$.
	}
\end{ex}
\begin{ex}%[Thanh Hai Le]%[2D2K6-1]
	Trong các khẳng định sau, khẳng định nào là \textbf{đúng}?
	\choice
	{$\log _a f(x) \geq \log _a g(x)\Leftrightarrow \heva{&0<a\neq 1\\ & f(x)\geq g(x)>0}$}
	{$a^{\log _a f(x)}\geq a^{\log _a g(x)}\Leftrightarrow \heva{&0<a\neq 1\\ & \log_a f(x)\geq \log_a g(x)}$}
	{\True $\log _a f(x) \geq \log _a g(x)\Leftrightarrow \heva{&0<a\neq 1\\ & f(x)>0, g(x) >0 \\ & (a-1)\left[ f(x)-g(x) \right]\geq 0}$}
	{$\log _a f(x) < g(x)\Leftrightarrow \heva{&0<a\neq 1\\ & 0<f(x)< a^{g(x)}}$}
	\loigiai{
		\begin{itemize}
			\item Xét bất phương trình $\log _a f(x) \geq \log _a g(x)$. \quad $(1)$
			$$(1)\Leftrightarrow \hoac{& \heva{& 0<a<1\\ & 0< f(x)\leq g(x)}\\ & \heva{& a>1\\ & f(x)\geq g(x)>0}}\Leftrightarrow \heva{&0<a\neq 1\\ & f(x)>0, g(x) >0 \\ & (a-1)\left[ f(x)-g(x) \right]\geq 0}.$$
			Do đó khẳng định {\lq\lq} $\log _a f(x) \geq \log _a g(x)\Leftrightarrow \heva{&0<a\neq 1\\ & f(x)\geq g(x)>0}$ {\rq\rq} sai, khẳng định {\lq\lq} $\log _a f(x) \geq \log _a g(x)\Leftrightarrow \heva{&0<a\neq 1\\ & f(x)>0, g(x) >0 \\ & (a-1)\left[ f(x)-g(x) \right]\geq 0}$ {\rq\rq} đúng.
			\item Xét bất phương trình $a^{\log _a f(x)}\geq a^{\log _a g(x)}$. \quad $(2)$
			$$(2)\Leftrightarrow \hoac{& \heva{& 0<a<1\\ & \log _a f(x)\leq \log _a g(x)}\\ & \heva{& a>1\\ & \log _a f(x)\geq \log _a g(x)}}.$$
			Do đó khẳng định {\lq\lq} $a^{\log _a f(x)}\geq a^{\log _a g(x)}\Leftrightarrow \heva{&0<a\neq 1\\ & \log_a f(x)\geq \log_a g(x)}$ {\rq\rq} sai.
			\item Xét bất phương trình $\log _a f(x) < g(x)$. \quad $(3)$
			$$(3)\Leftrightarrow \log _a f(x) < \log _a a^{g(x)}\Leftrightarrow \hoac{& \heva{& 0<a<1\\ & a^{g(x)}< f(x)}\\ & \heva{& a>1\\ & 0<f(x)< a^{g(x)}}}.$$
			Do đó khẳng định {\lq\lq} $\log _a f(x) < g(x)\Leftrightarrow \heva{&0<a\neq 1\\ & 0<f(x)< a^{g(x)}}$ {\rq\rq} sai.
		\end{itemize}
	}
\end{ex}
