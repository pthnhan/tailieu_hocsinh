	Cho biểu thức $P=\left( \dfrac{\sqrt{x}-1}{\sqrt{x}+1}+ \dfrac{1}{\sqrt{x}}- \dfrac{2\sqrt{x}}{x+ \sqrt{x}}
	\right) : \dfrac{1}{\sqrt{x}+1}$; $(x > 0)$.
	\begin{enumerate}
		\item  Rút gọn biểu thức $P$.
		\item Tìm giá trị nhỏ nhất của biểu thức $P$.
        \item Tìm nghiệm nguyên sao cho $A$ cũng nhận giá trị nguyên.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Biểu thức A có nghĩa khi $\heva{
& x^2-1\ne 0 \\
& x-1\ne 0 \\
}\Rightarrow \heva{
& x+1\ne 0 \\
& x-1\ne 0 \\
}\Rightarrow x\ne \pm 1.$\\
Rút gọn
$A=\left(\dfrac{x^3+1}{x^2-1}-\dfrac{x^2-1}{x-1}\right):\left(x+\dfrac{x}{x-1}\right)$ \\
$ \Leftrightarrow \left[\dfrac{\left(x+1\right)\left(x^2-x+1\right)}{\left(x+1\right)\left(x-1\right)}-\dfrac{x^2-1}{x-1}\right]:\left(\dfrac{x^2}{x-1}\right)$ \\
$ \Leftrightarrow \left(\dfrac{-x+2}{x-1}\right):\left(\dfrac{x^2}{x-1}\right)$ \\
$ \Rightarrow A=\dfrac{-x+2}{x^2}$.
			\item Vì $A=3\Leftrightarrow \dfrac{2-x}{x^2}=3\Leftrightarrow 3x^2+x-2=0\Leftrightarrow \hoac{
& x=-1\left(\text{loại}\right) \\
& x=\dfrac{2}{3}\left(\text{chọn}\right) \\
}$.
            \item Vì $x\in \mathbb{Z}$ nên
$ A=\dfrac{2-x}{x^2}\in \mathbb{Z}\Rightarrow-x.A\in \mathbb{Z}\Leftrightarrow \dfrac{x^2-2x}{x^2}\in \mathbb{Z}$ \\
$\Leftrightarrow 1-\dfrac{2}{x}\in \mathbb{Z}\Leftrightarrow x\in \left\{\pm 1;\pm 2\right\}$ \\
Kết hợp với điều kiện ta suy ra $x\in \left\{-2;2\right\}$\\
Thử lại ta thấy thỏa, vậy $x\in \left\{-2;2\right\}$ thỏa yêu cầu bài toán.
	\end{enumerate}
}
\end{ex}

Cho các số thực $x;y$ thỏa mãn $\left(x+y\right)=1$; $x^3+y^3=2$. Tính các giá trị biểu thức?
	\begin{enumerate}
		\item $M=x.y$.
		\item $x^5+y^5$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có :
$ 2=x^3+y^3=\left(x+y\right)\left(x^2-xy+y^2\right)=x^2-xy+y^2$ \\
$ ={\left(x+y\right)}^2-3xy=1-3xy$ \\
$ \Rightarrow xy=-\dfrac{1}{3}$
			\item Ta có :
$ x^5+y^5=\left(x^3+y^3\right)\left(x^2+y^2\right)-x^3y^2-x^2y^3$ \\
$ =2\left[{\left(x+y\right)}^2-2xy\right]-x^2y^2\left(x+y\right)$ \\
$ =2\left(1-2xy\right)-{\left(xy\right)}^2.\left(x+y\right)$ \\
$ =2\left(1-x.-\dfrac{1}{3}\right)-{\left(-\dfrac{1}{3}\right)}^2.1$ \\
$ =\dfrac{29}{9}$
	\end{enumerate}}
\end{ex}

	.Cho ba số $a,b,c$ đôi một khác nhau. Chứng minh giá trị của biểu thức sau không phụ thuộc vào $a,b,c$ .
$P=\dfrac{a^2}{\left(a-b\right)\left(a-c\right)}+\dfrac{b^2}{\left(b-a\right)\left(b-c\right)}+\dfrac{c^2}{\left(c-a\right)\left(c-b\right)}$
	\loigiai{
$\dfrac{a^2}{\left(a-b\right)\left(a-c\right)}+\dfrac{b^2}{\left(b-a\right)\left(b-c\right)}+\dfrac{c^2}{\left(c-a\right)\left(c-b\right)}$ \\
$ =\dfrac{a^2}{\left(a-b\right)\left(a-c\right)}-\dfrac{b^2}{\left(a-b\right)\left(b-c\right)}+\dfrac{c^2}{\left(a-c\right)\left(b-c\right)}$ \\
$ =\dfrac{a^2\left(a-b\right)-b^2\left(a-c\right)+c^2\left(a-b\right)}{\left(a-b\right)\left(a-c\right)\left(b-c\right)}$ \\
$ =\dfrac{a^2b-a^2c-b^2a+b^2c+c^2a-c^2b}{\left(a-b\right)\left(a-c\right)\left(b-c\right)}$ \\
$ =\dfrac{ab\left(a-b\right)-c\left(a^2-b^2\right)+c^2\left(a-b\right)}{\left(a-b\right)\left(a-c\right)\left(b-c\right)}$ \\
$ =\dfrac{\left(a-b\right)\left(ab-ac-bc+c^2\right)}{\left(a-b\right)\left(a-c\right)\left(b-c\right)}$ \\
$=\dfrac{\left(a-b\right)\left[b\left(a-c\right)-c\left(a-c\right)\right]}{\left(a-b\right)\left(a-c\right)\left(b-c\right)}$ \\
$ =\dfrac{\left(a-b\right)\left(a-c\right)\left(b-c\right)}{\left(a-b\right)\left(a-c\right)\left(b-c\right)}=1$ \\
	
	}
\end{ex}
