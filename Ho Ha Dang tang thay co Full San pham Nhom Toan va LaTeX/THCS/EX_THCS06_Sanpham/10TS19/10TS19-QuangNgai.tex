\begin{ex}%[Tuyển sinh 10 Quãng Ngãi 2019]%[Nguyễn Anh Tuấn, dự án EXTHCS-06]%[9D4B2]
	Cho Parabol $ (P):y=x^2 $ và đường thẳng $(d):y=-x+2 $.
	
	\begin{enumerate}
		\item Tìm tọa độ giao điểm của $ (P) $ và $ (d) $.
		\item Xác định $ m $ để $ (P) $, $ (d) $ và đường thẳng $ (d'): y=5mx+6 $ cùng đi qua một điểm. 
	\end{enumerate}	
	\loigiai{
		\begin{enumerate}
			\item Tọa độ giao điểm của $ (P) $ và $ (d) $ là nghiệm của hệ phương trình sau 
			$$\begin{cases}
			y=x^2\\y=-x+2
			\end{cases} \Leftrightarrow 
			\begin{cases}
			x^2+x-2=0\\y=-x+2
			\end{cases} \Leftrightarrow 
			\begin{cases}
			\left[\begin{aligned} x&=1 \\x&=-2  \end{aligned}\right. \\ y=x^2
			\end{cases}\Leftrightarrow 
			\left[\begin{aligned} (x;y)&=(1;1) \\ (x;y)&=(-2;4).
			\end{aligned}\right.$$
			Tọa độ các giao điểm là $ (1;1) $, $ (-2;4) $.
			\item Để $ (P) $, $ (d) $ và $ (d') $ đi qua cùng một điểm khi và chỉ khi
			$$\left[\begin{aligned}1&=5m\cdot 1+6  \\ 4&=5m \cdot (-2)+6 \end{aligned}\right. \Leftrightarrow \left[\begin{aligned} m&=-1 \\ m&=\dfrac{1}{5}. \end{aligned}\right.$$
			Tất cả các giá trị $ m $ cần tìm là $ m \in \left\{-1;\dfrac{1}{5}\right\} $.
		\end{enumerate}	
	}
\end{ex}
\begin{ex}%[Tuyển sinh 10 Quãng Ngãi 2019]%[Nguyễn Anh Tuấn, dự án EXTHCS-06]%[9D4K6]
	Cho phương trình $ x^2-2mx+2m-3=0 $ với $ m $ là tham số.	
	\begin{enumerate}
		\item Chứng minh rằng phương trình luôn có hai nghiệm phân biệt với mọi $ m $.
		\item Gọi $ x_1, x_2 $ là hai nghiệm của phương trình đã cho. Tìm các giá trị nguyên của $ m $ để biểu thức $ \dfrac{1}{x_1}+\dfrac{1}{x_2} $ nhận giá trị là một số nguyên.
	\end{enumerate}	
	\loigiai{
		\begin{enumerate}
			\item Ta có $ \Delta=4m^2-4(2m-3)=4m^2-8m+12=4(m^2-2m+3)=4[(m-1)^2+2] >0, \quad \forall m $. 
			
			Vậy phương trình có hai nghiệm phân biệt với mọi giá trị của $ m $.
			\item Theo định lí Vi-ét có $$ \begin{cases}
			x_1+x_2= 2m\\ x_1\cdot x_2=2m-3.
			\end{cases} $$
			Xét $ m\in \mathbb{Z} $. Khi đó, $ \dfrac{1}{x_1}+\dfrac{1}{x_2}=\dfrac{x_1+x_2}{x_1\cdot x_2}=\dfrac{2m}{2m-3} =\dfrac{2m-3+3}{2m-3}=1+\dfrac{3}{2m-3}$.
			
			Biểu thức $ \dfrac{1}{x_1}+\dfrac{1}{x_2} $ nhận giá trị là một số nguyên khi và chỉ khi $ 2m-3 \in \{\pm 1;\pm 3\} $.
			
			Xét các trường hợp sau
			\begin{itemize}
				\item $2m-3=1 \Leftrightarrow m=2$.
				\item $2m-3=-1 \Leftrightarrow m=1$.
				\item $2m-3=3 \Leftrightarrow m=3$.
				\item $2m-3=-3\Leftrightarrow m=0$.
			\end{itemize}
			Vậy $ m\in \{0;1;2;3\} $.
	\end{enumerate}}
\end{ex}
\begin{ex}%[Tuyển sinh 10 Quãng Ngãi 2019]%[Nguyễn Anh Tuấn, dự án EXTHCS-06]%[9D3B5]
	Một trường học A có tổng số giáo viên là $ 80 $. Hiện tại, tuổi	trung bình của giáo viên là $ 35 $. Trong đó, tuổi trung bình của giáo viên nữ là $ 32 $ và tuổi trung bình của giáo viên nam là $ 38 $. Hỏi trường học đó có bao nhiêu giáo viên nữ và bao nhiêu giáo viên nam?
	\loigiai{\begin{itemize}
			\item Gọi số giáo viên nam của trường là $ x $ (giáo viên), $ (0<x<80, x\in \mathbb{N}) $.
			\item Gọi số giáo viên nữ của trường là $ y $ (giáo viên), $ (0<y<80, x\in \mathbb{N}) $.
		\end{itemize}
		Khi đó, ta có $ x+y=80 $. \tagEX{1}
		
		Tuổi trung bình của giáo viên nam là $ 38 $ nên tổng số tuổi của $ x $ giáo viên nam là $ 38x $ tuổi.
		
		Tuổi trung bình của giáo viên nữ là $ 32 $ nên tổng số tuổi của $ y $ giáo viên nữ là $ 32y $ tuổi.
		
		Tổng số tuổi của giáo viên toàn trường là $ 80 \cdot 35=2800 $ (tuổi).
		
		Theo đề bài ta có phương trình $ 38x+32y=2800 \Leftrightarrow 19x+16y=1400$. \tagEX{2}
		
		Từ $ (1) $ và $ (2) $ ta có hệ phương trình $$\begin{cases}
		x+y=80\\19x+16y=1400
		\end{cases} \Leftrightarrow \begin{cases}
		x=40\\y=40.
		\end{cases}$$
		Vậy trường đó có $ 40 $ giáo viên nam và $ 40 $ giáo viên nữ.}
\end{ex}

		\item Kẻ $ AD $ cắt cung $ BC $ tại $ M $. Chứng minh $ D $ là trung điểm của $ MH $.

\immini{a) Vì $ BF \perp CF $ và $ BD \perp AD $ nên $$ \widehat{HFB} + \widehat{BDH}=180^{\circ}. $$ Do đó, tứ giác $ BFHD $ nội tiếp đường tròn.

	\item Xét tam giác $ BHD $ và tam giác $ BEC $ có:
	$$\widehat{EBC} \text { chung} ; \quad \widehat{BDH}=\widehat{BEC}=90^{\circ}.$$
	Vậy $ \triangle BHD \backsim \triangle BCE $. Suy ra $ \dfrac{BH}{BC}=\dfrac{BD}{BE} $ hay $ BD \cdot BC=BH \cdot BE $.\\

\widehat{EBC}+\widehat{ACB}=90^{\circ}\\\widehat{DAC}+\widehat{ACB}=90^{\circ}

Mà $\widehat{DAC}=\widehat{MBC}$ (hai góc nội tiếp cùng chắn cung $ MC $). Suy ra $\widehat{MBC}=\widehat{EBC}$.\\
