\end{dang}

\subsection{KIẾN THỨC CƠ BẢN}
\subsubsection{Định nghĩa nửa mặt phẳng}
Hình gồm đường thẳng $a$ và một phần của mặt phẳng bị chia ra bởi $a$ được gọi là một nửa mặt phẳng bờ $a$.  Trên hình $36$ có hai nửa mặt phẳng:
\begin{itemize}
	\item Nửa mặt phẳng (I) là nửa mặt phẳng bờ $a$ chứa điểm $M$, điểm $N$.
	\item Nửa mặt phẳng (II) là nửa mặt phẳng bờ $a$ chứa điểm $P$.
	\item Hai nửa mặt phẳng (I) và (II) có chung bờ $a$ được gọi là hai nửa mặt phẳng đối nhau.
\end{itemize}

\subsubsection{ Mỗi đường thẳng $a$ chia mặt phẳng thành hai phần (h. 36)}
\begin{itemize}
	\item Nếu hai điểm $M$, $N$ thuộc một phần thì đường thẳng $a$ không cắt đoạn thẳng $MN$.
	\item Nếu hai điểm $M$, $P$ thuộc hai phần khác nhau thì đường thẳng $a$ cắt đoạn thẳng $MP$.
	\end{itemize}
\subsubsection{Dấu hiệu 1 (Dấu hiệu T1-nhận biết tia nằm giữa hai tia)}
Cho ba tia $Ox$, $Oy$ và $Oz$ chung gốc $O$. Nếu có điểm $M$ thuộc tia $Ox$, điểm $N$ thuộc tia $Oy$ ($M\neq O,\ N\neq O$), mà tia $Oz$ cắt đoạn thẳng $MN$ tại $I$ nằm giữa hai điểm $M$ và $N$ thì tia $Oz$ nằm giữa hai tia $Ox$ và $Oy$ (h. 37).

\subsubsection{Dấu hiệu 2 (Dấu hiệu T2)}
 Nếu hai tia $Ox,\ Oy$ đối nhau thì mọi tia $Oz$ khác $Ox,\ Oy$ đều nằm giữa hai tia $Ox$ và $Oy$.
 \subsubsection{Dấu hiệu (đường thẳng cắt đoạn thẳng)}
 Có ba điểm $A$, $B$, $C$ không thẳng hàng và không có điểm nào thuộc đường thẳng $a$. Nếu đường thẳng $a$ cắt đoạn thẳng $BC$ thì đường thẳng $a$ hoặc cắt đoạn thẳng $AB$ hoặc cắt đoạn thẳng $AC$.\\
 Thật vậy, đường thẳng $a$ chia mặt phẳng thành hai nửa mặt phẳng (I) và (II). Đường thẳng $a$ cắt đoạn $BC$ nên giả sử nửa mặt phẳng (I) chứa điểm $B$ và nửa mặt phẳng (II) chứa điểm $C$ (h.38).

 Vì điểm $A$ không thuộc đường thẳng $a$, nên hoặc $A$ thuộc nửa mặt phẳng (I) hoặc $A$ thuộc nửa mặt phẳng (II).\\
 Nếu $A$ thuộc nửa mặt phẳng (I) chứa điểm $B$ thì đường thẳng $a$ không cắt đoạn thẳng $AB$ mà cắt đoạn thẳng $AC$, do hai điểm $A$ và $C$ thuộc hai nửa mặt phẳng đối nhau (h.38a).\\
 Tương tự, nếu $A$ thuộc nửa mặt phẳng (II) chứa điểm $C$ thì đường thẳng $a$ không cắt đoạn thẳng $AC$ mà cắt đoạn thẳng $AB$ (h.38b).\\
 Vậy, đường thẳng $a$ cắt đoạn thẳng $BC$ thì đường thẳng $a$ hoặc cắt đoạn thẳng $AB$ hoặc cắt đoạn thẳng $AC$. (Dấu hiệu này là định lí Pasch về tam giác).

Thật vậy, xét trường hợp $Ox$ và $Oz$ không là hai tia đối của nhau (h.39). Lấy điểm $A$ thuộc tia $Ox$ và điểm $C$ thuộc tia $Oz$ ($A$, $C$ khác $O$). Xét hai trường hợp
\begin{itemize}
\item Trường hợp tia $Om$ nằm giữa hai tia $Oy$ và $Oz$ (h.39a):\\
Vì tia $Oy$ nằm giữa hai tia $Ox$ và $Oz$ nên tia $Oy$ cắt đoạn thẳng $AC$ tại một điểm, chẳng hạn tại điểm $B$ nằm giữa hai điểm $A$ và $C$. \hfill (1)\\
Lại có tia $Om$ nằm giữa hai tia $Oy$ và $Oz$ nên tia $Om$ cắt đoạn $BC$ tại một điểm, chẳng hạn tại điểm $M$ nằm giữa hai điểm $B$ và $C$.\hfill (2)\\
Từ (1) và (2) suy ra điểm $M$ nằm giữa hai điểm $A$ và $C$, suy ra tia $Om$ nằm giữa hai tia $Ox$, $Oz$.
\item Trường hợp tia $Om$ nằm giữa hai tia $Oy$ và $Ox$ (h.39b).\\
Giải thích một cách tương tự, ta cũng có tia $Om$ nằm giữa hai tia $Ox$ và $Oz$.\\
Khi hai tia $Ox$ và $Oz$ là hai tia đối nhau, kết quả trên hiển nhiên đúng.

\end{itemize}

\subsection{Kiến thức cơ bản}
\begin{enumerate}
	\item Định nghĩa: Góc là một hình gồm hai tia chung gốc (h.41).
	\begin{itemize}
		\item Gốc chung của hai tia gọi là đỉnh của góc, hai tia gọi là hai cạnh của góc.
		\item Có nhiều các kí hiệu một góc, như: góc $xOy$, góc $MON$, góc $O$, $\angle xOy$, $ \widehat{  xOy}$, $ \widehat{ MON}$.
	\end{itemize}

\end{enumerate}
\subsection{Ví dụ minh họa} 

\begin{vd}%[Đỗ Đức Thảo, dự án 6SA19-NC1]

	\item Có $4,\ 5$ tia chung gốc $OA$, $OB$, $OC$, $OD$ và $OE$ theo thứ tự đó?

	\item Có $n$ tia chung gốc?

\item Với $3$ tia ta có $3$ góc là $ \widehat{ AOB},\ \widehat{AOC},\ \widehat{ BOC}$ (h.42).
\item Với $4$ tia chung gốc ta có $6$ góc là $$ \widehat{ AOB},\ \widehat{ AOC},\ \widehat{ AOD},\ \widehat{ BOC},\ \widehat{ BOD},\ \widehat{ COD}.$$
Với $5$ tia chung gốc ta có $10$ góc:
$$ \widehat{ AOB},\ \widehat{ AOC},\ \widehat{ AOD},\ \widehat{ AOE},\ \widehat{ BOC},\ \widehat{ BOD},\ \widehat{ BOE},\ \widehat{ COD},\ \widehat{ COE},\ \widehat{ DOE}.$$
\item Giả sử $100$ tia chung gốc được kí hiệu lần lượt là $OA_1,\ OA_2,\ OA_3, \ldots,\ OA_{100}.$\\
\textbf{\textit{Cách 1:}}\\ Tia $OA_1$ tạo với $99$ tia khác có $99$ góc.\\
Tia $OA_2$ tạo với $98$ tia khác (trừa tia $OA_1$) có $98$ góc.\\
Tia $OA_3$ tạo với $97$ tia khác (trừ các tia $OA_1$ và $OA_2$) có $97$ góc.\\
$\cdots \cdots \cdots \cdots \cdots \cdots$\\
Tia $OA_{99}$ tạo với $1$ tia khác có $1$ góc.\\
Vậy ta có 
$$S = 99 + 98 + 97 + \cdots + 2 + 1 = \dfrac{(99 + 1).99}{2}= 4950\ \text{góc}.$$
\textbf{\textit{Cách 2:}} Tia $OA_1$ tạo với $99$ tia còn lại được $99$ góc. Có $100$ tia tạo được $99\cdot 100$ góc, nhưng mỗi góc được tính $2$ lần (như góc $AOB$ và góc $BOA$ chỉ là một góc).\\
Vậy số góc được tạo thành là $\dfrac{99\cdot 100}{2}=4950\ \text{(góc)}.$
\item Với $n$ tia chung gốc thì số góc tạo thành bằng $\dfrac{n(n-1)}{2}.$
\end{enumerate}	}
\end{vd}

			\item Đường thẳng $a$ chia mặt phẳng thành hai nửa nặt phẳng bờ $a$.

				\item  TH1: $A$, $B$, $C$ cùng thuộc một nửa mặt phẳng bờ $a$ thì đường thẳng $a$ không cắt đoạn thẳng nào trong ba đoạn thẳng $AB$, $BC$, $CA$.
				\begin{center}

					\tkzLabelPoints[below ](a)
					\tkzLabelPoints[above](A,B,C)
					\tkzDrawPoints(A,B,C)
					\end{tikzpicture}
				\end{center}
				\item TH2: $A$, $B$, $C$ không cùng thuộc nửa mặt phẳng bờ $a$. Chẳng hạn $A$, $B$ cùng thuộc nửa mặt phẳng bờ $a$ và $C$ thuộc nửa mặt phẳng bờ $a$ còn lại. Khi đó, đường thẳng $a$ cắt hai đoạn thẳng $AC$, $BC$.\\

					\tkzLabelPoints[below ](a,C)
					\tkzLabelPoints[above](A,B)
					\tkzDrawPoints(A,B,C)
					\end{tikzpicture}
				\end{center}
			\end{itemize}
			\item Đường thẳng $a$ không cắt các đoạn thẳng $AB$, $AC$ nên $A$, $B$, $C$ cùng thuộc một nửa mặt phẳng bờ $a$ nên $a$ không thể cắt đoạn thẳng $BC$.
		\end{enumerate}
