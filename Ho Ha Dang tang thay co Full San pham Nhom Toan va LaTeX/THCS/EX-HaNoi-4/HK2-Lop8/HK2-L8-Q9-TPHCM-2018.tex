\begin{ex}%[8D3Y3]%[8D4B5]%[8D3K5]
Giải các phương trình sau:
\begin{enumerate}[a)]
\item $2(3x-1)=8+x$;
\item $|2-x|+2=3x$;
\item $\dfrac{x-1}{x-2}+\dfrac{x-2}{x-1}-\dfrac{x^2}{(x-1)(x-2)}=0$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Phương trình đã cho tương đương
$$ 6x-2=8+x\Leftrightarrow 5x=10\Leftrightarrow x=2.$$
Vậy tập nghiệm của phương trình là $S=\{2\}$.
\item 
\begin{itemize}
\item Với $x \leq 2$ thì $|2-x|=2-x$.\\
Phương trình trở thành $2-x+2=3x \Leftrightarrow 4x=4 \Leftrightarrow x=1$ (thỏa).
\item Với $x>2$ thì $|2-x|=x-2$.\\
Phương trình trở thành $x-2+2=3x \Leftrightarrow 2x=0 \Leftrightarrow x=0$ (loại).
\end{itemize}
Vậy tập nghiệm của phương trình là $S=\{1\}$.
\item ĐKXĐ: $x\ne 2$ và $x\ne 1$.\\
Phương trình đã cho tương đương\\ 
$\dfrac{(x-1)^2}{(x-1)(x-2)}+\dfrac{(x-2)^2}{(x-1)(x-2)}-\dfrac{x^2}{(x-1)(x-2)}=0\\
\Leftrightarrow (x-1)^2+(x-2)^2-x^2=0\\
\Leftrightarrow x^2-2x+1+x^2-4x+4-x^2=0\\
\Leftrightarrow x^2-6x+5=0\\
\Leftrightarrow (x-1)(x-5)=0\\ \Leftrightarrow \hoac{&x=5\quad \text{(nhận)}\\&x=1\quad \text{(loại)}.}$\\
Vậy tập nghiệm của phương trình là $S=\{5\}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[8D4Y4]
	Giải bất phương trình và biểu diễn tập hợp nghiệm trên trục số
	$$\dfrac{x-3}{3}-\dfrac{x-4}{4}<\dfrac{2x-3}{6}.$$
	\loigiai{
	\begin{equation*}
	\resizebox{\hsize}{!}{
		$\dfrac{x-3}{3}-\dfrac{x-4}{4}<\dfrac{2x-3}{6}
		\Leftrightarrow
		\dfrac{4x-12}{12}-\dfrac{3x-12}{12}<\dfrac{4x-6}{12}
		\Leftrightarrow
		4x-12-3x+12<4x-6
		\Leftrightarrow
		3x>6
		\Leftrightarrow
		x>2$.}
	\end{equation*}
	Vậy tập nghiệm của bất phương trình là $S=\{x|x>2 \}$.\\
	Từ đó, tập hợp nghiệm được biểu trên trục số như sau

}
\end{ex}

\begin{ex}%[8D3B6]
	Một khu vườn hình chữ nhật có chiều dài hơn chiều rộng $5$ m. Nếu bớt chiều dài $2$ m và tăng chiều rộng $3$ m thì chu vi khu vườn là $60$ m. Tính diện tích của khu vườn lúc đầu.
	\loigiai{
	Gọi $a$ (m) là chiều rộng của khu vườn. Điều kiện: $a>0$.\\
	Khi đó chiều dài của khu vườn là $a+5$ (m). \\
	Chiều dài của khu vườn sau khi thay đổi: $a+5-2=a+3$ (m).\\
	Chiều rộng của khu vườn sau khi thay đổi: $a+3$ (m).\\
	Chu vi của khu vườn sau khi thay đổi : $a+3+a+3$ (m).\\
	Theo đề bài ta có: $a+3+a+3=60 \Leftrightarrow a=27$.\\
	Vậy khu vườn có chiều rộng là $27$ m và chiều dài là $32$ m.
}
\end{ex}

\begin{ex}%[8H3K7]%[8H3G3]
Cho $\triangle ABC$ vuông tại $A$ có đường cao $AH$ ($H\in BC$).
\begin{enumerate}[a)]
\item Chứng minh: $\triangle HBA\backsim \triangle ABC$. \hfill (1đ)
\item Tia phân giác góc $\widehat{ABC}$ cắt $AH$, $AC$ lần lượt tại $D,E$ ($D\in AH, E\in AC$). Chứng minh: $\triangle ABE\backsim \triangle HBD$ và $AD=AE$. \hfill (1đ)
\item Chứng minh: $AB\cdot AD=HD\cdot BC$. \hfill (1đ)
\end{enumerate}
\loigiai{
\immini{
\begin{enumerate}[a)]
	\item Xét $\triangle HBA$ và $\triangle ABC$:\\
	$\widehat{AHB}=\widehat{CAB}(=90^\circ)$\\
	$\widehat{HBA}$ chung\\
	$\Rightarrow \triangle HBA\backsim \triangle ABC$ (g.g).
	\item Xét $\triangle ABE$ và $\triangle HBD$\\
	$\widehat{HBD}=\widehat{EBA}$($BE$ là tia phân giác của $\widehat{CBA}$)\\
	$\widehat{DHB}=\widehat{EAB}=90^\circ$\\
	$\Rightarrow \triangle ABE\backsim \triangle HBD$ (g.g)\\
	$\Rightarrow \widehat{HDB}=\widehat{AEB}$\\
	Mà $\widehat{HDB}=\widehat{EDA}$ (hai góc đối đỉnh)\\
	Nên $\widehat{AED}=\widehat{EDA}\Rightarrow \triangle ADE $ cân tại $A$\\
	$\Rightarrow AE=AD$.
	\item $\triangle HBA\backsim \triangle ABC$ (chứng minh trên)\\
	$\Rightarrow \dfrac{HB}{AB}=\dfrac{AB}{BC}$\\
	Do $BD$ là phân giác trong của $\triangle BAH$\\
	Nên $\dfrac{DH}{AD}=\dfrac{HB}{AB}$\\
	$\Rightarrow \dfrac{HD}{AD}=\dfrac{AB}{BC}\\
	\Rightarrow AB\cdot AD=HD\cdot BC$.
\end{enumerate}
}{

\begin{ex}%[8D3K6]%[8H3T8]
\begin{enumerate}[a)]
\item Trong kì thi học sinh giỏi cấp Quận Giải truyền thống Lương Thế Vinh năm học 2017 - 2018, một trường THCS ở Quận 9 có $105$ học sinh tham dự, nhà trường đã tổ chức xe đưa đón học sinh dự thi bằng $3$ xe ô tô. Biết rằng xe thứ $I$ chở nhiều hơn xe thứ $III$ là $12$ học sinh, xe thứ $II$ chở nhiều hơn xe thứ $I$ là $6$ học sinh. Hỏi mỗi xe chở bao nhiêu học sinh (không có học sinh nào đi xe riêng).
\item Một cột đèn cao $7$ m, có bóng trên mặt đất dài $4$ m, cùng thời điểm đó gần cột đèn có một tòa nhà cao tầng có bóng trên mặt đất dài $60$ m (hình vẽ). Hỏi tòa nhà đó có bao nhiêu tầng (không kể tầng hầm), biết rằng mỗi tầng của tòa nhà cao $3{,}5$ m.

\loigiai{
\begin{enumerate}[a)]
\item Gọi số học sinh xe thứ nhất chở là $x$ học sinh, $(x\in \mathbb{N}^*, x>12)$.\\
Do xe thứ $I$ chở nhiều hơn xe thứ $III$ là $12$ học sinh nên số học sinh xe thứ $III$ chở là $x-12$.\\
Do xe thứ $II$ chở nhiều hơn xe thứ $I$ là $6$ học sinh nên số học sinh xe thứ $II$ chở là $x+6$.\\
Do nhà trường có tổng cộng $105$ học sinh nên ta có phương trình:
$$x+x-12+x+6=105\Leftrightarrow x=37.$$
Vậy số học sinh xe thứ $I$ chở là $37$ học sinh; xe thứ $II$ chở là $43$ học sinh; xe thứ $III$ chở là $25$ học sinh.
\item 
\immini{
Gọi tòa nhà và bóng của tòa nhà trên mặt đất lần lượt là $AH$ và $HB$.\\
Gọi cột đèn và bóng của cột đèn trên mặt đất lần lượt là $CK$ và $KD$.\\
Theo đề bài, ta có $CK=7$ m, $KD=4$ m, $HB=60$ m.

Do $CD\parallel AB$ (hai tia sáng) nên $\widehat{CDK}=\widehat{ABH}$ (đồng vị).\\
Xét $\triangle CKD$ và $\triangle AHB$ có:\\
$\left. \begin{array}{l}
\widehat{CDK}=\widehat{ABH} \\ 
\widehat{CKD}=\widehat{AHB}
\end{array}  \right\}$\\
$\Rightarrow \triangle CKD \backsim \triangle AHB$ (g-g).\\
$\Rightarrow \dfrac{CK}{KD}=\dfrac{AH}{HB}\Rightarrow AH=\dfrac{CK\cdot HB}{KD}=\dfrac{7\cdot 60}{4}=105$ m.\\
Vậy tòa nhà cao $105$ m, mà mỗi tầng cao $3{,}5$ m nên số tầng của tòa nhà là $\dfrac{105}{3{,}5}=30$ tầng.
\end{enumerate}
}
