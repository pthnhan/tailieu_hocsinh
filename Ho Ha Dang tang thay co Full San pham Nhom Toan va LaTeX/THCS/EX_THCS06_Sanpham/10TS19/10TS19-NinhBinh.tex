\begin{ex}%[Đề thi tuyển sinh vào 10 năm học 2018-2019,Ninh Bình]%[Hoàng Mạnh Linh]%[9D4B4]%[9D4B6]
Cho phương trình $x^2 - mx + m - 4 = 0 \hspace{0.1cm}\text{($x$ là ẩn số và $m$ là tham số)} \refstepcounter{equation}\hfill (\theequation)\label{01}$ .
\begin{enumerate}
	\item Giải phương trình \eqref{01} khi $m = 8$.
	\item Chứng minh rằng phương trình \eqref{01} luôn có hai nghiệm phân biệt $x_1$ và $x_2$ với mọi $m$. Tìm tất cả các giá trị nguyên dương của $m$ để $(5x_1 - 1)(5x_2 - 1) < 0$.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Với $m = 8$, phương trình \eqref{01} trở thành
	$
	x^2 - 8x + 4 = 0 \Leftrightarrow \left[\begin{aligned}
	x &= 4 + 2\sqrt{3}\\ x &= 4 - 2\sqrt{3}.
	\end{aligned}\right.
	$\\
	Vậy $m = 8$ thì phương trình có tập nghiệm là $S = \{4 + 2\sqrt{3}; 4 - 2\sqrt{3}  \}$.
	\item Ta có $\triangle = (-m)^2 - 4(m - 4) = m^2 - 4m + 16 = (m - 2)^2 + 12 > 0, \forall m \in \mathbb{R}$ nên phương trình \eqref{01} luôn có hai nghiệm phân biệt $x_1, x_2$ với mọi $m$.\\
	Theo định lý Vi - ét ta có $\begin{cases}
	x_1 + x_2 = m\\ x_1x_2 = m - 4
	\end{cases}$. Suy ra
	\begin{eqnarray*}
	&& (5x_1  - 1)(5x_2 - 1) < 0\\
	&\Leftrightarrow &25x_1x_2 - 5(x_1 + x_2) + 1 < 0\\
	&\Leftrightarrow &25(m - 4) - 5m + 1 < 0\\
	&\Leftrightarrow &20m - 99 < 0\\
	&\Leftrightarrow &m < \dfrac{99}{20}\\
	&\Leftrightarrow &m < 4,95.
	\end{eqnarray*}
Vì $m$ là số nguyên dương nên $m \in \{1;2;3;4\}$. 
\end{enumerate}
}
\end{ex}




\begin{ex}%[Đề thi tuyển sinh vào 10 năm học 2018-2019,Ninh Bình]%[Hoàng Mạnh Linh]%[9D3B5]
Một hình chữ nhật có chu vi bằng $28$ cm. Tính chiều dài và chiều rộng của chữ nhật, biết rằng nếu tăng chiều dài thêm $1$ cm và tăng chiều rộng thêm $2$ cm thì diện tích hình chữ nhật đó tăng thêm $25$ cm$^2$. 
\loigiai{
Gọi chiều dài và chiều rộng của hình chữ nhật lần lượt là $x, y$ (cm). Điều kiện $0 < y < x < 14$.\\
Theo đề bài, chu vi của hình chữ nhật là $28$ cm nên ta có phương trình
\begin{align}\label{02}
2(x + y) = 28.
\end{align}
Mặt khác, nếu tăng chiều dài thêm $1$ cm và tăng chiều rộng thêm $2$ cm thì diện tích hình chữ nhật đó tăng thêm $25$ cm$^2$, nên ta có phương trình
\begin{align}\label{03}
(x + 1)(y + 2) = xy +25.
\end{align}
Từ \eqref{02} và \eqref{03} ta có hệ phương trình
\begin{align*}
\begin{cases}
2(x + y) = 28\\ (x + 1)(y + 2) = xy + 25.
\end{cases}
\Leftrightarrow
\begin{cases}
x + y = 14\\ 2x + y = 23.
\end{cases} 
\Leftrightarrow
\begin{cases}
x = 9\\ y = 5. \hspace{0.1cm}
\end{cases}
\text{(tmđk)}
\end{align*}
Vậy chiều dài của hình chữ nhật là $9$ cm, chiều rộng của hình chữ nhật là $5$ cm.
}
\end{ex}



\begin{ex}%[Đề thi tuyển sinh vào 10 năm học 2018-2019,Ninh Bình]%[Hoàng Mạnh Linh]%[9H3B7]%[9H3B7]%[9H3B7]%[9H3K4]
Cho tam giác nhọn $ABC$ có $AB < AC$ và đường cao $AK$. Vẽ đường tròn tâm $O$ đường kính $BC$. Từ $A$ kẻ các tiếp tuyến $AM, AN$ với đường tròn $(O)$ ($M, N$ là các tiếp điểm; $M$ và $B$ nằm trên nửa mặt phẳng có bờ là đường thẳng $AO$). Gọi $H$ là giao điểm của hai đường thẳng $MN$ và $AK$. Chứng minh rằng:
\begin{enumerate}
	\item Tứ giác $AMKO$ nội tiếp đường tròn.
	\item $KA$ là tia phân giác của $\widehat{MKN}$.
	\item $AN^2 = AK \cdot AH$.
	\item $H$ là trực tâm của tam giác $ABC$.
\end{enumerate}
\loigiai{
\begin{center}
	\begin{tikzpicture}[>=stealth,line join=round,line cap=round,font=\footnotesize,scale=1]
	\tkzDefPoints{0/0/B,8/0/C,2.5/7/A}
	\tkzDrawSegments(B,C)
	\tkzLabelPoints[below](B,C)
	\tkzDrawPoints[fill=black](B,C)
	\tkzDefMidPoint(B,C)\tkzGetPoint{O}
	\tkzLabelPoints[below](O)\tkzDrawPoints(O)
	\tkzDrawArc[color = black](O,C)(B)
	\tkzLabelPoints[above](A)\tkzDrawPoints(A)
	\tkzDrawPolygon(A,B,C)
	\tkzDrawAltitude(C,B)(A)\tkzGetPoint{K}\tkzLabelPoints[below](K)\tkzDrawPoints(K)
	\tkzDrawMedian[tuychon](C,B)(A)
	\tkzMarkRightAngles[size=.3](B,K,A)
	\tkzDrawAltitude(A,C)(B) \tkzGetPoint{D}\tkzLabelPoints[above](D)\tkzDrawPoints(D)
	\tkzInterLL(A,K)(B,D) \tkzGetPoint{H}\tkzDrawPoints(H)\tkzLabelPoints[](H)
	\tkzTangent[from=A](O,B) \tkzGetPoints{M}{N}\tkzDrawPoints(M,N)\tkzLabelPoints[left](M)\tkzLabelPoints[right](N)
	\tkzDrawSegments(M,N M,K K,N O,M O,N A,M A,N)
	\end{tikzpicture}
\end{center}
\begin{enumerate}
	\item Do $AM, AN$ là tiếp tuyến của đường tròn $(O)$ nên $\widehat{AMO} = \widehat{ANO} = 90^\circ$.\\
	Vì $AK$ là đường cao của tam giác $ABC$ nên $\widehat{AKO} = \widehat{AKC} = 90^\circ$.\\
	Do ba điểm $M, K, N$ cùng nhìn đoạn $AO$ dưới một góc vuông nên năm điểm $M, K, N, A, O$ thuộc đường tròn đường kính $AO$.\\
	Vậy tứ giác $AMKO$ nội tiếp đường tròn.
	\item Vì $AM, AN$ là tiếp tuyến của đường tròn $(O)$ nên $AM = AN. \refstepcounter{equation}\hfill (\theequation)\label{04}$\\
	Theo chứng minh ở câu $a)$ thì năm điểm $M, K, N, O, A$ cùng thuộc một đường tròn nên ta có tứ giác $AMKN$ nội tiếp. $\refstepcounter{equation}\hfill (\theequation)\label{05}$\\
	Từ \eqref{04} và \eqref{05} suy ra $\widehat{AKM} = \widehat{AKN}$ (các góc nội tiếp cùng chắn các cung bằng nhau thì bằng nhau).\\
	Vậy $KA$ là tia phân giác của $\widehat{MKN}$.
	\item Ta có $\begin{cases}
	\widehat{ANH} = \widehat{AKM} \\ \widehat{AKM} = \widehat{AKN}
	\end{cases} 
	\Rightarrow \widehat{AKN} = \widehat{ANH}	$.\\
Xét $\triangle AHN$ và $\triangle ANK$ có
\begin{itemize}
	\item $\widehat{AKN} = \widehat{ANH}$
	\item $\widehat{HAN} = \widehat{KAN}$
\end{itemize}
nên $\triangle AHN \backsim \triangle ANK$ (g - g).\\
Suy ra $\dfrac{AN}{AK} = \dfrac{AH}{AN}$, hay $AN^2 = AH \cdot AK. \refstepcounter{equation}\hfill (\theequation)\label{06}$.
\item Gọi $D$ là giao điểm của $AC$ và đường tròn $(O)$.\\
Xét $\triangle AND$ và $\triangle ACN$ có $\widehat{NAD} = \widehat{NAC}$, $\widehat{AND} = \widehat{ACN}$ (góc tạo bởi tiếp tuyến và dây cung và góc nội tiếp cùng chắn một cung thì bằng nhau) nên $\triangle AND \backsim \triangle ACN$ (g-g).\\ Suy ra $\dfrac{AN}{AC} = \dfrac{AD}{AN}$, hay $AN^2 = AD \cdot AC. \refstepcounter{equation}\hfill (\theequation)\label{07}$\\
Từ \eqref{06} và \eqref{07} suy ra $AH \cdot AK = AD \cdot AC$, hay $\dfrac{AH}{AC} = \dfrac{AD}{AK}$.\\
$\triangle AHD$ và $\triangle ACK$ có $\begin{cases}
\widehat{HAD} = \widehat{KAC}\\ \dfrac{AH}{AC} = \dfrac{AD}{AK}
\end{cases}$ nên $\triangle AHD \backsim \triangle ACK$ (c-g-c).\\
Suy ra $\widehat{ADH} = \widehat{AKC} = 90^\circ$. Dẫn đến $\widehat{HDC} = 90^\circ. \refstepcounter{equation}\hfill (\theequation)\label{08}$\\
Điểm $D$ thuộc đường tròn đường kính $BC$ nên $\widehat{BDC} = 90^\circ. \refstepcounter{equation}\hfill (\theequation)\label{09}$\\
Từ \eqref{08} và \eqref{09} suy ra $B,H,D$ thẳng hàng, nghĩa là $BH \perp AC$.\\ Lại có $AH \perp BC$ nên $H$ là trực tâm của tam giác $ABC$.
\end{enumerate}
}
\end{ex}



\begin{ex}%[Đề thi tuyển sinh vào 10 năm học 2018-2019,Ninh Bình]%[Hoàng Mạnh Linh]%[9D5G1]
Cho $a, b$ là hai số thực dương thỏa mãn $a + b \leq 4$. Tìm giá trị nhỏ nhất của biểu thức $$S = \dfrac{1}{a^2 + b^2} + \dfrac{25}{ab} + ab.$$
\loigiai{
\begin{align*}
S &= \dfrac{1}{a^2 + b^2} + \dfrac{25}{ab} + ab\\
&= \dfrac{1}{a^2 + b^2} + \dfrac{1}{2ab} + ab + \dfrac{16}{ab} + \dfrac{17}{2ab}.
\end{align*}
Với $x; y$ là các số dương ta có bất đẳng thức $\dfrac{1}{x} + \dfrac{1}{y} \geq \dfrac{4}{x + y}$. Dấu "=" xảy ra khi $x = y$.\\
 Từ đó ta có $\dfrac{1}{a^2 + b^2} + \dfrac{1}{2ab} \geq \dfrac{4}{(a + b)^2} \geq \dfrac{1}{4}$ (vì $a  + b \leq 4$).\\
Áp dụng bất đẳng thức Co-si cho hai số không âm ta có: $ab + \dfrac{16}{ab} \geq 2\sqrt{ab \cdot \dfrac{16}{ab}} = 8$.\\
Lại có $ab \leq \dfrac{(a+b)^2}{4} \Rightarrow ab \leq 4 \Rightarrow \dfrac{17}{2ab} \geq \dfrac{17}{8}$.\\
Vậy $S \geq \dfrac{1}{4} + 8 + \dfrac{17}{8} = \dfrac{83}{8}$.\\
Dấu "=" xảy ra khi và chỉ khi $\begin{cases}
a^2 + b^2 = 2ab\\ ab = \dfrac{16}{ab} \\ ab = \dfrac{(a + b)^2}{4} 
\end{cases}
\Leftrightarrow a = b = 2 
$.\\
Vậy giá trị nhỏ nhất của $S$ bằng $\dfrac{83}{8}$ khi $a = b = 2$.
}
\end{ex}
