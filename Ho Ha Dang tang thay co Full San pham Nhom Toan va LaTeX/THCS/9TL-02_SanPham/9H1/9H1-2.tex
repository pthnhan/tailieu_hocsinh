 \subsubsection{Tỉ số lượng giác của hai góc phụ nhau}    
   \begin{dl}
   	Nếu hai góc phụ nhau thì $\sin$ góc này bằng $\cos$ góc kia, $\tan$ góc này bằng $\cot$ góc kia.
   	\end{dl}
   \begin{hq}\immini{Cho hai góc $\alpha$ và $\beta$ với $\alpha + \beta = 90 ^ \circ$, khi đó
   \begin{itemize}
   		\begin{multicols}{2}
   			\item $\sin \alpha = \cos \beta$;\\
   			\item $\cos \alpha = \sin \beta$;\\
   			\item $\tan \alpha = \cot \beta $;\\
   			\item $\cot \alpha = \tan \beta$.\\
   		\end{multicols}
   	\end{itemize}}{

	\loigiai{
		\immini{
			Kẻ đường cao $BH$ của $\triangle ABC$.
			
			Khi đó ta có $HC^2=\left(AC-AH\right)^2$.
			
			Áp dụng định lý Py-ta-go ta có
			\begin{align*}
			BC^2&=BH^2+HC^2\\
			&=BH^2+\left(AC-AH\right)^2\\
			&=BH^2+AH^2+AC^2-2AC\cdot AH\\
			&=AB^2+AC^2-2AC\cdot AH. 
			\end{align*}
			Lại có $\widehat{BAC}=60^\circ$
			
			$\Rightarrow\cos 60^\circ =\dfrac{AH}{AB}\Rightarrow \dfrac{1}{2}=\dfrac{AH}{AB}$ hay $AH=\dfrac{AB}{2}$.

	Cho tứ giác $ABCD$ có $\alpha$ là góc nhọn tạo bởi hai đường chéo. Chứng minh rằng $S_{ABCD}=\dfrac{1}{2}AC\cdot BC\cdot \sin \alpha.$
	\loigiai{
		\immini{
			Giả sử hai đường chéo $AC,BD$ cắt nhau tại $I,\widehat{AIB}=\alpha$ là góc nhọn.
			
			Kẻ đường cao $AH$ của $\triangle ABD$ và đường cao $CK$ của $\triangle CBD.$
			
			Ta có $AH=AI\sin \alpha,CK=CI\sin \alpha$. 
			
			Diện tích $\triangle ABD$ là $S_{\triangle ABD}=\dfrac{1}{2}BD\cdot AH$. 
			
			Diện tích $\triangle CBD$ là $S_{\triangle CBD}=\dfrac{1}{2}BD\cdot CK.$
			
			Khi đó 
			\begin{align*}
			S_{ABCD}&=S_{\triangle ABD}+S_{\triangle CBD}=\dfrac{1}{2}BD\cdot(AH+CK)\\
			&=\dfrac{1}{2}BD\cdot(AI+CI)\sin \alpha\\
			&=\dfrac{1}{2}BD\cdot AC \cdot \sin \alpha.
			\end{align*}
		}{
			\begin{tikzpicture}
			\tkzInit[xmin=-1,xmax=5,ymin=-1,ymax=5.5]
			\tkzClip
			\tkzDefPoints{0/0/D,1.2/4/A,4/0/C,4.5/5/B}
			\tkzInterLL(A,C)(B,D)\tkzGetPoint{I}
			\tkzDefPointBy[projection = onto
			D--B](A) \tkzGetPoint{H}
			\tkzDefPointBy[projection = onto
			D--B](C) \tkzGetPoint{K}
			\tkzDrawPoints[fill=black](A,B,C,D,I,H,K)
			\tkzLabelPoints[left](A,D,K,I)
			\tkzLabelPoints[right](B,C,H)
			\tkzDrawSegments(A,B B,C C,D D,A A,C B,D)
			\tkzDrawSegments[dashed](C,K A,H)
			\tkzMarkAngles[size=0.3](B,I,A D,I,C)
			\tkzMarkRightAngles(A,H,B C,K,D)
			\end{tikzpicture}
		}
	}
