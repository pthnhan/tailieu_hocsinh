\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Hà Nam]%[8D1Y1]%[8D1B11]%[8D2K3]
	Thực hiện phép tính:
\begin{enumerate}
\item $-3x^2\left(5-2x+3x^3-x^4\right)$.
\item $\left(-3x^5+6x^2-4x^3+5x^4\right):2x^2$.
\item $\dfrac{3}{2x+2}+\dfrac{2-x}{x^2-1}$.
\end{enumerate}
	\loigiai{
\begin{enumerate}
\item $-3x^2\left(5-2x+3x^3-x^4\right)=-15x^2+6x^3-9x^5+12x^6$.
\item $\left(-3x^5+6x^2-4x^3+5x^4\right):2x^2=-\dfrac{3}{2}x^3+3-2x+\dfrac{5}{2}x^2$.
\item $\dfrac{3}{2x+2}+\dfrac{2-x}{x^2-1}=\dfrac{3}{2(x+1)}+\dfrac{2-x}{(x+1)(x-1)}$\\
    $=\dfrac{3(x-1)+2(2-x)}{(x+1)(x-1)}=\dfrac{x+1}{(x+1)(x-1})=\dfrac{1}{x-1}$.
\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Hà Nam]%[8D1B6]%[8D1B7]%[8D1B9]
Phân tích các đa thức sau thành nhân tử:
\begin{enumerate}
\item $25x^2-4y^2$.
\item $x^2-3xy^2-3x^2y-y^2$.
\item $x^2+3x+2$.
\end{enumerate}
	\loigiai{
\begin{enumerate}
\item  $25x^2-4y^2=(5x)^2-(2y)^2=(5x+2y)(5x-2y)$.
\item $x^2-3xy^2-3x^2y-y^2=(x^2-y^2)-(3x^2y+3xy^2)$
$=(x+y)(x-y)-3xy(x+y)$ \\
$=(x+y)(x-y-3xy)$.
\item $x^2+3x+2=x^2+x+2x+2$
$=x(x+1)+2(x+1)$ \\
$=(x+1)(x+2)$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Hà Nam]%[8D2B3]%[8D2K9]%[8D2G9]
Cho biểu thức: $A=\dfrac{2x}{x^2-9}+\dfrac{5}{3-x}-\dfrac{1}{x+3}$ (với $x\ne \pm 3$).
\begin{enumerate}
\item Rút gọn biểu thức $A$.
\item  Tính giá trị của biểu thức $A$ khi $x=-\dfrac{3}{2}$.
\item  Tìm $x$ nguyên để $A$ là số nguyên.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $A=\dfrac{2x}{x^2-9}+\dfrac{5}{3-x}-\dfrac{1}{x+3}=\dfrac{2x}{\left(x-3\right)\left(x+3\right)}-\dfrac{5}{x-3}-\dfrac{1}{x+3}$\\
$=\dfrac{2x-5\left(x+3\right)-\left(x-3\right)}{\left(x-3\right)\left(x+3\right)}=\dfrac{-4x-12}{\left(x-3\right)\left(x+3\right)}$\\
$=\dfrac{-4\left(x+3\right)}{\left(x-3\right)\left(x+3\right)}=\dfrac{4}{3-x}$.
\item  $x=-\dfrac{3}{2}\Rightarrow A=\dfrac{4}{3+\dfrac{3}{2}}=\dfrac{8}{9}$.
\item  Để $A$ nguyên thì $3-x$ là ước nguyên của $4$.\\
Ta có bảng sau:
\begin{center}
\begin{tabular}{|c|c|c|c|c|c|c|}
\hline
$3-x$&	$-4$&	$-2$&	$-1$&	$1$&	$2$&	$4$\\
\hline
$x$	&$7$&	$5$&	$4$&	$2$&	$1$	&$-1$\\
\hline
\end{tabular}
\end{center}
Vậy $x\in \left\{-1;1;2;4;5;7\right\}$ thì $A$ nguyên.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Hà Nam]%[8H2B6]%[8H1K9]%[8H1G5]
Cho tam giác $ABC$ vuông tại $A$. Kẻ đường cao $AH$. Gọi $K$ và $E$ lần lượt là các điểm đối xứng của $H$ qua $AB$, $AC$. Gọi $I$ là giao điểm của $KH$ và $AB$, $N$ là giao điểm của $EH$ và $AC$.
\begin{enumerate}
\item Chứng minh $AH=IN$.
\item Chứng minh $A$ là trung điểm của $KE$.
\item Tứ giác $BCEK$ là hình gì? Vì sao?
\end{enumerate}

	\loigiai
	{
\begin{center}
    \begin{tikzpicture}[scale=1]
\tkzInit[xmin=-2.5,ymin=-4.5,xmax=7.5,
ymax=4.5]
\tkzClip
\tkzDefPoint(0,0){A}\tkzDefPoint(7,0){C}\tkzDefPoint(0,4){B}
\tkzDefMidPoint(A,C)
\tkzGetPoint{L}
\tkzInterLC(C,B)(L,C)
\tkzGetPoints{H}{P}
\tkzDefPointBy[reflection=over A--B](H)
\tkzGetPoint{K}
\tkzDefPointBy[reflection=over A--C](H)
\tkzGetPoint{E}
\tkzInterLL(H,K)(A,B)
\tkzGetPoint{I}
\tkzInterLL(H,E)(A,C)
\tkzGetPoint{N}
\tkzMarkRightAngle[size=0.3](B,A,C)
\tkzMarkRightAngle[size=0.3](A,H,C)
\tkzMarkRightAngle[size=0.3](H,N,C)
\tkzMarkRightAngle[size=0.3](A,I,H)
\tkzLabelPoints[left](A)
\tkzLabelPoints[above right](H)
\tkzLabelPoints[right](C)
\tkzLabelPoints[below](E)
\tkzLabelPoints[above left](K,I,N,B)
\tkzDrawSegments(A,B A,B B,C A,H K,E H,E H,K A,C E,C)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item  Xét tứ giác $ANHI$ ta có:\
$\widehat{IAN}=\widehat{AIH}=\widehat{ANH}=90^\circ$.\\
$\Rightarrow ANHI$ là hình chữ nhật.\\
$\Rightarrow AH=IN$
\item Ta có: $K$ và $H$ đối xứng với nhau qua $AB$\\
$\Rightarrow AH=AK$$\Rightarrow \Delta AHK$ cân tại $A$.\\
$\Rightarrow AI$ là đường cao đồng thời là đường phân giác của $\Delta AHK$\\
$\Rightarrow \widehat{KAH}=2\widehat{IAH}~~~~(1)$\\
Lại có: $E$ và $H$ đối xứng với nhau qua $AC$\\
$\Rightarrow AH=AE$$\Rightarrow \Delta AHE$cân tại $A$.\\
$\Rightarrow AN$ là đường cao đồng thời là đường phân giác của $\Delta AHE$\\
$\Rightarrow \widehat{EAH}=2\widehat{NAH}~~~~(2)$\\
Từ $(1)$ và $(2)$ $K,A,E$ thẳng hàng.\\
Mà $AK=AE$\\
$\Rightarrow A$ là trung điểm của $KE$.
\item Tứ giác $BCEK$ là hình thang vuông.\\
Chứng minh:\\
$\Delta AKB=\Delta AHB\left(c-c-c\right)$\\
$\Rightarrow \widehat{AKB}=\widehat{AHB}=90^\circ\left(*\right)$\\
Hay $BK\bot KE~~~~(3)$\\
$\Delta AEC=\Delta AHC\left(c-c-c\right)$\\
$\Rightarrow \widehat{AEC}=\widehat{AHC}=90^\circ$\\
Hay $CE\bot KE~~~~(4)$\\
Từ $(3)$, $(4)$ suy ra $BK//CE\left(**\right)$\\
Từ $\left(*\right)$,$\left(**\right)$ suy ra tứ giác $BCEK$ là hình thang vuông.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Hà Nam]%[8D1G7]
	Tìm giá trị nhỏ nhất của biểu thức:
$A=x^2-2xy+2y^2+2x-10y+2033$.
\loigiai{
Ta có:
\begin{align*}
& A=x^2-2xy+2y^2+2x-10y+2033 \\
& A=x^2-2xy+y^2+2x-2y+1+y^2-8y+16+2016 \\
& A=(x-y)^2+2(x-y)+1+(y^2-8y+16)+2016 \\
& A=(x-y+1)^2+(y-4)^2+2016\ge 2016 \\
\end{align*}
Vậy $\min A=2016\Leftrightarrow \left\{\begin{array}{*{35}{l}}
x-y+1=0  \\
y-4=0  \\
\end{array}\Leftrightarrow \left\{\begin{array}{*{35}{l}}
x=3  \\
y=4  \\
\end{array}\right.\right.$
}
\end{ex}
