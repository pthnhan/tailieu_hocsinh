\subsubsection{Vị trí tương đối của hai đường tròn}
Xét vị trí tương đối của hai đường tròn $(O;R)$ và $(O'; r)$ với $R>r$.
\begin{enumerate}
\item Nếu $R-r<OO'<R+r$ thì $(O)$ và $(O')$ cắt nhau tại hai điểm phân biệt.

\item Nếu $OO'=R+r$ thì $(O)$ và $(O')$ tiếp xúc ngoài với nhau. \\
Nếu $OO'=R-r$ thì $(O)$ và $(O')$ tiếp xúc trong với nhau.

\noindent{\textbf{Chú ý.}}  
\begin{itemize}
	\item Nếu hai đường tròn cắt nhau thì hai giao điểm đối xứng với nhau qua đường nối hai tâm.
\item Nếu hai đường tròn tiếp xúc nhau thì tiếp điểm nằm trên đường nối hai tâm.
\end{itemize}
\subsubsection{Tiếp tuyến chung của hai đường tròn}
\begin{enumerate}
\item Tiếp tuyến chung của hai đường tròn là đường thẳng tiếp xúc với cả hai đường tròn đó.
\item Tiếp tuyến chung của hai đường tròn không cắt đoạn nối hai tâm là tiếp tuyến chung ngoài, cắt đoạn nối tâm là tiếp tuyến chung trong.
\end{enumerate}
\subsection{Các ví dụ}
\begin{vd}
Cho hai đường tròn $(O;R)$ và $(O'; r)$ với $R=12$cm, $r=5$cm, $OO'=13$cm.

\item Chứng minh $AO$ là tiếp tuyến của đường tròn $(O'; r)$.
\item Tính độ dài $AB$.
\end{enumerate}

\begin{vd}

\item Vẽ đường kính $BOD$ và $CO'E$. Chứng minh các bộ ba điểm $B$, $A$, $E$ và $C$, $A$, $D$ thẳng hàng.

\item Gọi $K$ là trung điểm của $DE$. Chứng minh đường tròn ngoại tiếp $\triangle OKO'$ tiếp xúc với $BC$.
\end{enumerate}

Mặt khác, $O'A=O'C=O'E$ nên $\triangle CAE$ vuông tại $A$ hay $\widehat{CAE}=90^\circ$. Từ đó suy ra $\widehat{BAC}=\widehat{CAE}=90^\circ$, do đó các bộ ba điểm $B$, $A$, $E$ và $C$, $A$, $D$ thẳng hàng.

\item Vì $IO'$ và $OK$ lần lượt là đường trung bình của tam giác $CBE$ và tam giác $DEB$ nên $IO'\parallel BE$, $IO'=\dfrac{1}{2}BE$ và $OK\parallel BE$, $OK=\dfrac{1}{2}BE$. Do đó $IO'=OK$ và $IO'\parallel OK$, suy ra tứ giác $OIO'K$ là hình bình hành.\\

\begin{vd}
Cho hai đường tròn $(O;R)$ và $(O';r)$ tiếp xúc ngoài với nhau tại $A$. Vẽ tiếp tuyến chung ngoài tiếp xúc với $(O)$ và $(O')$ lần lượt ở $B$ và $C$.  Đường vuông góc với $OO'$ kẻ từ $A$ cắt $BC$ ở $M$.
\begin{enumerate}
\item Tính $MA$ theo $R$ và $r$.
\item Tính diện tích tứ giác $BCO'O$ theo $R$ và $r$.
\item Tính diện tích tam giác $BAC$ theo $R$ và $r$.
\item Gọi $I$ là trung điểm của $OO'$. Chứng minh rằng $BC$ là tiếp tuyến của đường tròn $(I;IM)$.
\end{enumerate}

\item Vì $MA$ và $MB$ là tiếp tuyến của đường tròn $(O)$ nên $MO$ là tia phân giác của $\widehat{BMA}$, hay \[\widehat{OMA}=\dfrac{1}{2}\widehat{BMA}.\tag{1}\]
Do $MA$ và $MC$ là tiếp tuyến của đường tròn $(O')$ nên $MO'$ là tia phân giác của $\widehat{CMA}$, hay \[\widehat{O'MA}=\dfrac{1}{2}\widehat{CMA}.\tag{2}\]
Từ $(1)$ và $(2)$ ta có
\[\widehat{OMO'}=\widehat{OMA}+\widehat{O'MA}=\dfrac{1}{2}\widehat{BMA}+\dfrac{1}{2}\widehat{CMA}=\dfrac{1}{2}\widehat{BMC}=90^\circ.\]
Áp dụng hệ thức lượng trong tam giác vuông $OMO'$ vuông tại $M$, $MA\perp OO'$ ta có
\[MA^2=AO\cdot AO'\Rightarrow MA^2=Rr\Rightarrow MA=\sqrt{Rr}. \]

\item Áp dụng hệ thức lượng cho tam giác vuông $OAI$ ($\widehat{IAO}=90^\circ$), $AE\perp OI$ ta có 
\[IE\cdot IO==IA^2=IB^2=\dfrac{BC^2}{4}.\tag{4}\]
Áp dụng hệ thức lượng cho tam giác vuông $O'AI$ ($\widehat{IAO'}=90^\circ$), $AE\perp O'I$ ta có

\item Ta có $\triangle ABC\backsim\triangle IOO'$ (g.g) nên 
\[\dfrac{S_{\triangle ABC}}{S_{\triangle IOO'}}=\left(\dfrac{BC}{OO'}\right)^2\Rightarrow S_{\triangle ABC}=\dfrac{S_{\triangle IOO'}\cdot BC^2}{OO'^2}.\tag{6}\]
Mà $BC=2IA$; $OO'=2a$; $S_{\triangle IOO'}=\dfrac{1}{2}\cdot OO'\cdot IA=\dfrac{1}{2}\cdot 2a\cdot IA$ nên thay vào $(6)$ ta được
\[S_{\triangle ABC}=\dfrac{S_{\triangle IOO'}\cdot BC^2}{OO'^2}=\dfrac{a\cdot IA\cdot \left(2IA\right)^2}{\left(2a\right)^2}=\dfrac{IA^3}{a}.\]
Áp dụng hệ thức lượng trong tam giác vuông $OIO'$ ($\widehat{OIO'}=90^\circ$, $IA\perp OO'$) ta có
\[IA^2=AO\cdot AO'=R\cdot R'\le \left(\dfrac{R+R'}{2}\right)^2=a^2.\]
Suy ra $IA$ lớn nhất bằng $a$ khi $R=R'$. Vậy $S_{\triangle ABC}$ lớn nhất bằng $a^2$ khi $R=R'$.
\end{enumerate}
