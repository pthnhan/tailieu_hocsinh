\begin{ex}%[Đề thi vào 10, Chuyên KHTN Hà Nội vòng 2, 2017]%[9D3K4]%[9D1K8]
\hfill
\begin{enumerate}
\item[1)] Giải hệ phương trình $\heva{&x+y=\sqrt{x+3y}\\&x^2+y^2+xy=3.}$
\item[2)]  Với $a,b$ là các số thực dương thỏa mãn $ab+a+b=1$. Chứng minh rằng 
\[ \dfrac{a}{1+a^2} + \dfrac{b}{1+b^2} = \dfrac{1+ab}{\sqrt{2(1+a^2)(1+b^2)}}.\]
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[1)]   Điều kiện xác định : $x+y \geq 0 , x+3y \geq 0$. \\
            $\heva{&x+y=\sqrt{x+3y}\\&x^2+y^2+xy=3} \iff \heva{&x^2+y^2+2xy=x+3y\\&x^2+y^2+xy=3} 
            \iff \heva{&(x-3)(y-1)=0\\&x^2+y^2+xy=3} $\\
           + Với $x=3$, $y^2+3y+6=0$ (vô nghiệm). \\
           + Với $y=1$, $x^2+x-2=0 \iff x=1, x=-2$, ta loại nghiệm $x=-2$ do $x+y=-2+1 <0$. Vậy nghiệm của hệ là $(x;y)=(1;1)$.
\item[2)]   Do $a,b$ dương và $ab+a+b=1$ cho nên $0<a,b<1$ và $ab+a+b=1 \iff (a+b)^2 = (1-ab)^2$.
\begin{align*}
& \dfrac{a}{1+a^2} + \dfrac{b}{1+b^2} = \dfrac{1+ab}{\sqrt{2(1+a^2)(1+b^2)}} \\
& \iff  (a+b+ab^2+ab^2+1)\sqrt{2(1+a^2)(1+b^2)} = (1+ab)(1+a^2)(1+b^2)  \\
& \iff  (a+b)(1+ab)\sqrt{2(1+a^2)(1+b^2)} = (1+ab)(1+a^2)(1+b^2)  \\   
& \iff 2(a+b)^2 = (1+a^2)(1+b^2) \\
& \iff a^2+b^2+4ab = 1+a^2b^2 \\
& \iff (a+b)^2 = (1-ab)^2   ~    ( \textrm{ĐPCM})
\end{align*}
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên KHTN Hà Nội Vòng 2, 2017]%[9D5G2]%[9D5G1]
\hfill
\begin{enumerate}
\item[1)] Giả sử $p, q$ là hai số nguyên tố thỏa mãn đẳng thức 
\begin{center}
$p( p -1) = q(q^2-1)$  \quad   (*)
\end{center}
a. Chứng minh rằng tồn tại số nguyên dương $k$ sao cho  $p-1=kq,  q^2-1=kp$.\\
b. Tìm tất cả các số nguyên tố $p, q$ thỏa mãn đẳng thức (*).
\item[2)] Với $a, b, c$ là các số thực dương thỏa mãn $ab+bc+ca+abc=2$, tìm giá trị lớn nhất của biểu thức
\begin{center}
$M=\dfrac{a+1}{a^2+2a+2} + \dfrac{b+2}{b^2+2b+2} +\dfrac{c+2}{c^2+2c+2}$
\end{center}
\end{enumerate}
\loigiai{
\hfill
\begin{enumerate}
\item[1)] 
a. Dễ thấy $p>q \geq 2$ và $(p,q) =1$ cho nên tồn tại số nguyên dương $k$ sao cho $p-1 = kq$, 
(*) $\iff pkq = q(q^2-1) \iff q^2-1 = kp$. \\
b. Theo $a$ ta có $q^2-1 = k (1+kq) \iff q^2 -k^2q -1-k =0$ (1), giả sử $q_1,q_2$ là các nghiệm của phương trình (1),
theo định lý vi-ét ta có $q_1q_2 = -1-k$, $q_1+q_2 =-k^2$ do $|q_1|,|q_2|$ là các số nguyên lớn hơn 1 nên $|q_1q_2| \geq  |q_1|+|q_2|$ suy ra $1+k \geq  k^2 \iff \dfrac{1-\sqrt{5}}{2}  \leq k \leq \dfrac{1+\sqrt{5}}{2}$, $k$ nguyên dương cho nên $k=1 \Longrightarrow q=2$ và $p=3$.
\item[2)] Đặt $x=a+1,y=b+1,z=c+1$, $ab+bc+ca+abc=2 \iff (x-1)(y-1)+(y-1)(z-1)+(z-1)(x-1) + (x-1)(y-1)(z-1)=2 
\iff x+y+z=xyz$ với $x,y,z > 1$ \quad (*). Khi đó $M= \dfrac{x}{1+x^2} + \dfrac{y}{1+y^2} +\dfrac{z}{1+z^2}$,
tiếp tục ta đặt $u=\dfrac{1}{x}, v=\dfrac{1}{y},w=\dfrac{1}{z}$, (*) $\iff uv+vw+wu =1; 0<u,v,w<1$ và 
\begin{align*}
M&=\dfrac{u}{1+u^2} + \dfrac{v}{1+v^2} +\dfrac{w}{1+w^2} \\
&=\dfrac{u}{uv+vw+wu+u^2} + \dfrac{v}{uv+vw+wu+v^2} +\dfrac{w}{uv+vw+wu+w^2} \\
&=\dfrac{u}{(u+v)(u+w)} + \dfrac{v}{(u+v)(v+w)} +\dfrac{w}{(u+w)(v+w)} \\
&=\dfrac{u(v+w)}{(u+v)(v+w)(u+w)} +\dfrac{v(u+w)}{(u+v)(v+w)(u+w)} +\dfrac{w(u+w)}{(u+v)(v+w)(u+w)}  \\
&=\dfrac{2}{(u+v)(v+w)(u+w)}
\end{align*}
Ta có $(u+v)(v+w)(u+w) = (u+v+w)(uv+vw+uw) - uvw =u+v+w - uvw $, $(u+v+w)^2 \geq 3(uv+vw+wu) =3$,
$uv+vw+wu \geq 3\sqrt[3]{(uvw)^2} \Longrightarrow uvw \leq \dfrac{1}{3\sqrt{3}} \Longrightarrow - uvw \geq \dfrac{1}{3\sqrt{3}} \Longrightarrow (u+v)(v+w)(u+w) \geq \sqrt{3}- \dfrac{1}{3\sqrt{3}} =\dfrac{8}{3\sqrt{3}} 
\Longrightarrow M \leq \dfrac{3\sqrt{3}}{4}$.\\
Vậy $M_{\max} = \dfrac{3\sqrt{3}}{4}$, khi $a=b=c=\sqrt{3}-1$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên KHTN Hà Nội vòng 2, 2017]%[9H5G4]
Cho n là số nguyên dương, $n \geq  5$. Xét một đa giác lồi $n$ cạnh. Người ta muốn kẻ một số
đường chéo của đa giác mà các đường chéo này chia đa giác đã cho thành đúng $k$ miền, mỗi
miền là một ngũ giác lồi ( hai miền bất kỳ không có điểm trong chung). 
\begin{enumerate}
\item[a.] Chứng minh rằng ta có thể thực hiện được với $n=2018, k=672$.
\item[b.] Với $n=2017, k=672$ ta có thể thực hiện được không? Hãy giải thích.
\end{enumerate}

\loigiai{
\begin{enumerate}
\item[a.] Giả sử đa giác lồi là $A_1A_2 \dots A_{2018}$, ta kẻ các đường chéo $A_1A_5, A_1A_{5+3i}$ với $1 \leq i \leq 671$, khi đó ta có được đúng $672$ miền, mỗi miền là một ngũ giác lồi.
\item[b.] Giả sử ta có thể thực hiện được,  $672$ ngũ giác lồi có
tất cả $5 \times 672 = 3360 $ cạnh, trong các cạnh này có $2017$ cạnh của đa giác. Bởi vì cạnh nào của một ngũ giác  không phải là cạnh của đa giác sẽ được tính $2$ lần, cho nên  $3360=2017 + 2m$ với $m$ là số nguyên dương , phương trình này không có nghiệm nguyên dương, cho nên không thể thực hiện được việc chia đa giác.
\end{enumerate}
}
\end{ex}
