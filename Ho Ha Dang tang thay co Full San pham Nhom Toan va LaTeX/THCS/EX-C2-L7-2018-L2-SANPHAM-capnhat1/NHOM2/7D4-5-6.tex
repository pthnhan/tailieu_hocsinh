\begin{vd}%[7D4Y7]
	Cho đa thức $f(x)=x+7x^2-6x^3+3x^4+2x^2+6x-2x^4+1.$
	\begin{enumerate}
		\item Thu gọn, rồi sắp xếp các số hạng của đa thức theo lũy thừa giảm dần của biến $x$.
		\item Xác định bậc của đa thức, hệ số tự do, hệ số cao nhất.
		\item Tính $f(-1)$, $f(0)$, $f(1)$ và $f(-a)$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Có đa thức $f(x)=x^4-6x^3+9x^2+7x+1$.
			\item Đa thức $f(x)=x^4-6x^3+9x^2+7x+1$ có bậc $4$. Hệ số tự là là $1$ và hệ số cao nhất là $1$.
			\item Ta có:\\
			$f(-1)=(-1)^4-6\cdot (-1)^3+9\cdot (-1)^2+7\cdot (-1)+1=1+6+9-7+1=10$.\\
			$f(0)=0^4-6\cdot 0^3+9\cdot 0^2+7\cdot 0+1=0+0+0-0+1=1.$\\
			$f(1)=1^4-6\cdot 1^3+9\cdot 1^2+7\cdot 1+1=1-6+9+7+1=12.$\\
			$f(a)=(-a)^4-6(-a)^3+9(-a)^2+7(-a)+1=a^4+6a^3+9a^2-7a+1$.
		\end{enumerate}
	}
\end{vd}

\begin{vd}%[7D4B6]%[7D4K6]
	Cho các đa thức: $A=5x^2-3xy+7y^2$ và $B=6x^2-8xy+9y^2$.
	\begin{enumerate}
		\item Tính $P=A+B$ và $Q=A-B$.
		\item Tính giá trị của đa thức $M=P-Q$ tại $x=-1$ và $y=-2$.
		\item Cho đa thức $N=3x^2-16xy+14y^2$. Chứng minh đa thức $M-N$ luôn nhận giá trị không âm với mọi giá trị của $x$ và $y$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			\begin{eqnarray*}
				P=A+B&=&(5x^2-3xy+7y^2)+(6x^2-8xy+9y^2)\\
				&=&5x^2-3xy+7y^2+6x^2-8xy+9y^2\\
				&=&11x^2-11xy+16y^2.\\
				Q=A-B&=&(5x^2-3xy+7y^2)-(6x^2-8xy+9y^2)\\
				&=&5x^2-3xy+7y^2-6x^2+8xy-9y^2\\
				&=& -x^2+5xy-2y^2.
			\end{eqnarray*}
			\item Ta có
			\begin{eqnarray*}
				M=P-Q&=&(11x^2-11xy+16y^2)-(-x^2+5xy-2y^2)\\
				&=&11x^2-11xy+16y^2+x^2-5xy+2y^2\\
				&=&12x^2-16xy+18y^2.
			\end{eqnarray*}
			Giá trị của $M$ tại $x=-1$ và $y=-2$ là
			\begin{center}
				$M=12\cdot (-1)^2-16\cdot (-1)\cdot (-2)+18\cdot (-2)^2=12-32+72=52.$
			\end{center}
			Có thể tính $M$ theo cách khác
			\begin{center}
				$M=P-Q=(A+B)-(A-B)=A+B-A+B=2B=2(6x^2-8xy+9y^2)$.
			\end{center}
			\item Ta có $T=M-N=(12x^2-16xy+18y^2)-(3x^2-16xy+14y^2)=9x^2+4y^2$.\\
			Suy ra $T=9x^2+4y^2\ge 0$ với mọi $x$, $y$ và $T=0$ tại $x=y=0$.\\
			Vậy $T$ luôn nhận giá trị dương hoặc bằng $0$, hay nói cách khác là $T$ luôn nhận giá trị không âm với mọi giá trị của $x$ và $y$. 
		\end{enumerate}
	}
\end{vd}

\begin{vd}%[7D4B7]
	Cho đa thức $f(x)=1+x+x^2+\cdots+x^{2013}+x^{2014}$. Tính $f(-1)$, $f(0)$ và $f(1)$.
	\loigiai{
		Có $f(0)=1+0+0^2+\cdots+0^{2013}+0^{2014}=1$.\\
		Có $f(1)=1+(1+1^2+\cdots+1^{2013}+1^{2014})=1+2014=2015$.\\
		Có $f(-1)=1+[(-1)+(-1)^2+\cdots+(-1)^{2013}+(-1)^{2014}]=1+[-1+1-1+1-\cdots-1+1]$
		\begin{center}
			(trong ngoặc vuông có $2014$ số hạng)
		\end{center}
		$=1+[(-1+1)+(-1+1)+\cdots+(-1+1)]=1+(0+0+\cdots+0)=1.$
	}
\end{vd}

\begin{vd}%[7D4K7]
	Cho các đa thức
	$f(x)=x^{2014}-x^{2013}+x^{2012}-x^{2011}+\cdots+x^2-x+1$ và $h(x)=-1+x-x^2+x^3-\cdots-x^{2012}+x^{2013}-x^{2014}$.\\
	Biết $\varphi (x)=[f(x)-g(x)][f(x)+h(x)]$. Hỏi sau khi khai triển thì đa thức $\varphi (x)$ là đa thức bậc mấy?
	\loigiai{
		Có 
		\begin{eqnarray*}
			f(x)+h(x)&=&[x^{2014}-x^{2013}+x^{2012}-x^{2011}+\cdots+x^2-x+1]\\
			&+&[-1+x-x^2+x^3-\cdots-x^{2012}+x^{2013}-x^{2014}]=0.
		\end{eqnarray*}
		Vậy đa thức $\varphi (x)$ là đa thức $0$ và là đa thức không có bậc.
	}
\end{vd}

\begin{vd}%[7D4K6]

	\begin{center}

	\end{center}

	\begin{enumerate}
		\item Cho $u_1=a$ và $u_2=b$. Tính $u_{10}$.
		\item Cho $u_1=a$ và $u_{10}=13a$. Tính các số còn lại.
		\item Nếu biết $u_5$ là trung bình cộng của các số $u_1=a$ và $u_{10}$. Tính các số còn lại theo $a$. Tính $u_5$ nếu $u_1=-5.$

					\hline 
				\end{tabular}
			\end{center}

\begin{vd}%[7D4B7]
	Bạn BEE không biết tính bài toán sau như thế nào, vì tích lũy thừa $5$ của $-0{,}3$ rất khó. Bạn giúp BEE được không?\\
	Tính $f(-0{,}3)$, biết đa thức $f(x)=x^5+0{,}027x^2-2014$.
	\loigiai{
		Thay $0{,}3$ bằng $x$ và biến đổi đa thức ta có
		\begin{eqnarray*}
			f(-0{,}3)&=&x^5-(-0{,}027)x^2-2014=x^5-(-0{,}3)^3x^2-2014\\
			&=&x^5-x^3x^2-2014=x^5-x^5-2014=-2014.
		\end{eqnarray*}
	}
\end{vd}

\begin{vd}%[7D4K6]
	Bạn BEE khoe minh có thể áp dụng phương pháp thay số bằng biến như ví dụ 6 để giải bài toán sau:\\
	Chứng minh rằng với mọi giá trị của $x$ và $y$ mà $x-2y=2$ thì giá trị của đa thức $P=x^2-2xy-2x+xy^2-2y^2-2y^3-2014$ luôn là một hằng số. Bạn BEE giải bài toán trên bằng cách thay tất cả số 2 bằng $x-2y$:
	\begin{eqnarray*}
		P&=&x^2-(x-2y)xy-(x-2y)x+xy^2-(x-2y)y^2-(x-2y)y^3-2014\\
		&=&x^2-x^2y+2xy^2-x^2+2xy+xy^2-xy^2+2y^3-xy^3+2y^4-2014\\
		&=&-x^2y+2xy^2+2xy+2y^4-xy^3+2y^4-2014.
	\end{eqnarray*}
	Đến đây BEE không biết làm tiếp thế nào nữa. Các bạn có thể giúp BEE không?
	\loigiai{
		Không nên thay một cách máy móc, mà cần linh hoạt.
		\begin{itemize}
			\item Thay biến bằng số.
			\begin{eqnarray*}
				P&=&(x^2-2xy)-2x+(xy^2-2y^3)-2y^2-2014\\
				&=&x(x-2y)-2x+y^2(x-2y)-2y^2-2014\\
				&=&2x-2x+2y^2-2y^2-2014=-2014.
			\end{eqnarray*}
			\item Thay số bằng biến.
			\begin{eqnarray*}
				P&=&x^2-2xy-(x-2y)x+xy^2-(x-2y)y^2-2y^3-2014\\
				&=&x^2-2xy-x^2+2xy+xy^2-xy^2+2y^3-2y^3-2014=-2014.
			\end{eqnarray*}
		\end{itemize}
		Như vậy ta luôn có $P=-2014$, hay với mọi giá trị của $x$ và $y$ mà $x-2y=2$ thì giá trị của đa thức $P=x^2-2xy-2x+xy^2-2y^2-2y^3-2014$ luôn là một hằng số.
	}
\end{vd}

\begin{vd}%[7D4K7]
	Cho các đa thức $A(x)=(x-4)^2+2014$ và $B(x)=4|x-4|+4$.
	\begin{enumerate}
		\item Tính $A(-4)$, $A(4)$, $B(-4)$ và $B(4)$.
		\item Tìm giá trị nhỏ nhất của đa thức $f(x)=A(x)+B(x)-10.$
		\item Tìm giá trị nhỏ nhất của đa thức $\varphi (x)=A(x)-B(x)-14$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta lập bảng
			\begin{center}
				\begin{tabular}{|c|c|c|}
					\hline 
					x & -4 & 4\\ 
					\hline 
					$A(x)=(x-4)^2+2014$ & $2078$ & $2014$\\ 
					\hline 
					$B(x)=4|x-4|-4$ & $28$ & $-4$\\ 
					\hline 
				\end{tabular}
			\end{center}
			\item 
			\begin{eqnarray*}
				f(x)&=&A(x)+B(x)-10\\
				&=&\left[(x-4)^2+2014\right]+\left[4|x-4|-4\right]-10\\
				&=&(x-4)^2+4|x-4|+2000
			\end{eqnarray*}
			Nhận xét: $(x-4)^2\ge 0$ và $(x-4)^2=0$ khi $x=4$.\\
			$4|x-4|\ge 0$ và $4|x-4|=0$ tại $x=4$.\\
			Suy ra $f(x)\ge 2000$ và $f(x)=2000$ khi $x=4$.\\
			Vậy Min$ f(x)=2000$ tại $x=4$.
			\item
			\begin{eqnarray*}
				\varphi (x)&=&A(x)-B(x)-14\\
				&=&\left[(x-4)^2+2014\right]-\left[4|x-4|-4\right]-14\\
				&=&(x-4)^2-4|x-4|+2004.
			\end{eqnarray*}
			Đặt $a=|x-4|\Rightarrow (x-4)^2=(|x-4|)^2=a^2$.
			\begin{eqnarray*}
				\text{Có }\varphi (x)&=&A(x)-B(x)-14=a^2-4a+2014\\
				&=&a^2-4a+4+2000 = (a^2-2a)-(2a-4)+2000\\
				&=&a(a-2)-2(a-2)+2000= (a-2)(a-2)+2000=(a-2)^2+2000.
			\end{eqnarray*}
			Vì $(a-2)^2\ge 0$ và $(a-2)^2=0$ khi $a=2\Leftrightarrow |x-4|=2\Rightarrow x\in \left\{2;6\right\}$.\\
			Suy ra $\varphi (x)\ge 2000$ và $\varphi (x)=2000$ khi $x\in \left\{2;6\right\}$.\\
			Vậy Min$ \varphi (x)=2000$ tại $x\in \left\{2;6\right\}.$
		\end{enumerate}
	\begin{note}
		Nếu $f(x)\ge m$ ($m$ là một hằng số) và $f(x)=m$ tại $x=x_0$ thì Min$f(x)=m$ tại $x=x_0$.\\
		Trong phần c) nếu biến đổi $\varphi (x)=(x-4)^2-4|x-4|+2014$, dù có $(x-4)^2\ge 0$ và $4|x-4|\ge 0$ thì ta vẫn không thể kết luận $\varphi (x)\ge 0$, do giữa hai biểu thức $(x-4)^2$ và $4|x-4|$ là dấu trừ. Do đó ta phải đặt $a=|x-4|$, rồi sau đó tách $4a=2a+2a$, ghép cặp và đưa về bình phương của một biểu thức.
	\end{note}
	}
\end{vd}

	$A=3x^2-4xy+5y^2$, $B=2x^2+5xy-4y^2$, $C=4x^2-xy+3y^2$.
	\begin{enumerate}
		\item $P=A+B+C$, $Q=A+B-C$, $R=A-B+C$ và $S=-A+B+C$.
		\item Tính giá trị của đa thức $M=(P+Q)-(Q+S)$ tại $x=-1$ và $y=-1$. 
		\item Cho đa thức $N=-x^2+4xy+y^2$. Chứng minh rằng đa thức $T=N-M$ luôn không dương với mọi giá trị của $x$ và $y$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item
			\begin{itemize}
				\item Tính $P$:
				\begin{eqnarray*}
					P=A+B+C&=&3x^2-4xy+5y^2+2x^2+5xy-4y^2+4x^2-xy+3y^2\\
					&=&(3+2+4)x^2+(-4+5-1)xy+(5-4+3)y^2=9x^2+4y^2.
				\end{eqnarray*}
				\item Tính $Q$:
				\begin{eqnarray*}
					Q=A+B-C&=&(3x^2-4xy+5y^2)+(2x^2+5xy-4y^2)-(4x^2-xy+3y^2)\\
					&=&3x^2-4xy+5y^2+2x^2+5xy-4y^2-4x^2+xy+3y^2\\
					&=&(3+2-4)x^2+(-4+5+1)xy+(5-4+3)y^2=x^2+2xy+4y^2.
				\end{eqnarray*}
				\item Tính $R$:
				\begin{eqnarray*}
					R=A-B+C&=&(3x^2-4xy+5y^2)-(2x^2+5xy-4y^2)+(4x^2-xy+3y^2)\\
					&=&3x^2-4xy+5y^2-2x^2-5xy+4y^2+4x^2-xy+3y^2\\
					&=&(3-2+4)x^2+(-4-5-1)xy+(5+4+3)y^2=5x^2-10xy+12y^2.
				\end{eqnarray*}
				\item Tính $S$:
				\begin{eqnarray*}
					S=-A+B+C&=&-(3x^2-4xy+5y^2)+(2x^2+5xy-4y^2)+(4x^2-xy+3y^2)\\
					&=&-3x^2+4xy-5y^2+2x^2+5xy-4y^2+4x^2-xy+3y^2\\
					&=&(-3+2+4)x^2+(4+5-1)xy+(-5-4+3)y^2=3x^2+8xy-6y^2.
				\end{eqnarray*}
			\end{itemize}
			\item Ta có 
			\begin{eqnarray*}
				M&=&(P+Q)-(Q+S)=P+Q-Q-S=P-S=(9x^2+4y^2)-(3x^2+8xy-6y^2)\\
				&=&9x^2+4y^2-3x^2-8xy+6y^2=(9-3)x^2-8xy+(4+6)y^2=6x^2-8xy+10y^2.
			\end{eqnarray*}
			Khi $x=-1$ và $y=-1$ thì $M=6\cdot (-1)^2-8(-1)(-1)+10(-1)^2=6-8+10=8.$
			\item \begin{eqnarray*}
				T&=&N-M=-x^2+4xy+y^2-(6x^2-8xy+10y^2)\\
				&=&-x^2+4xy+y^2-6x^2+8xy-10y^2\\
				&=&(-1-6)x^2+(4+8)xy+(1-10)y^2\\
				&=&-7x^2+12xy-9y^2=-3x^2-(4x^2-12xy+9y^2)\\
				&=&-3x^2-(2x-3y)^2\le 0 (\text{ với mọi }x,\text{ }y)
			\end{eqnarray*}
			Vậy đa thức $T=N-M$ luôn không dương với mọi giá trị của $x$ và $y$.
		\end{enumerate}
	}
\end{bt}

\section{NGHIỆM CỦA ĐA THỨC MỘT BIẾN}

\subsection{Kiến thức cần nhớ}
Đa thức $f(x)=a_nx^n+a_{n-1}x^{n-1}+\cdots+a_2x^2+a_1x+a_0$ (với $a_n\ne 0$).\\
Nếu $f(a)=0$ thì $a$ là nghiệm ($f(x)=0$ tại $x=a$).\\
Tìm nghiệm là tìm các giá trị của $x$ để $f(x)=0$.\\
Một đa thức (khác đa thức không) có thể có một nghiệm, hai nghiệm,... hoặc vô nghiệm (số nghiệm không vượt quá số bậc)

\subsection{Hỏi đáp nhanh}

\begin{ex}%[7D4Y9]
	Cho đa thức $f(x)=ax^2+bx+c$ với $a-b+c=0$. Khi đó nghiệm của đa thức $f(x)$ là
	\choice
	{$x=1$}
	{$x=\pm 1$}
	{\True $x=-1$}
	{$x=0$}
	\loigiai{
		Đa thức $f(x)=a_nx^n+\cdots+a_3x^3+a_2x^2+a_1x+a_0$ mà có $a_1+a_3+\cdots=a_0+a_2+\cdots$ thì phương trình $f(x)=0$ có nghiệm $x=-1$.
	}
\end{ex}

\begin{vd}%[7D4B9]
	Cho đa thức $f(x)=-x^4+x+7x^2-6x^3+3x^4-6x^2+6x^3-2x^4-6$.
	\begin{enumerate}
		\item Thu gon, rồi sắp xếp các số hạng của đa thức theo lũy thừa giảm dần của biến $x$.
		\item Chứng minh rằng các giá trị $-2$, $1$ và $3$ không là nghiệm của đa thức $f(x)$.
		\item Tìm nghiệm của đa thức.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Có đa thức $f(x)=x^2+x-6$.
			\item Xét $x\in \left\{-2;1;3\right\}$ ta có
			\begin{center}
				\begin{tabular}{|c|c|c|c|}
					\hline
					$x$& $-2$ & $1$ & $3$\\
					\hline
					$x^2$ & $4$ & $1$ & $9$\\
					\hline
					$x$ & $-2$ & $1$ & $3$\\
					\hline
					$-6$ & $-6$ & $-6$& $-6$\\
					\hline
					$f(x)=x^2+x-6$  & $-4$ & $-4$ & $6$ \\
					\hline
				\end{tabular}
			\end{center}
			Vậy $-2$, $1$, $3$ không là nghiệm của đa thức $f(x)$.
			\item Đặt $f(x)=0\Rightarrow x^2+x-6=0\Rightarrow x^2+3x-2x-6=x(x+3)-2(x+3)=(x+3)(x-2)=0$.\\
			Trường hợp thứ nhất $x+3=0\Rightarrow x=-3$.\\
			Trường hợp thứ hai $x-2=0\Rightarrow x=2$.\\
			Vậy đa thức $f(x)=0$ có tập hợp nghiệm $S=\left\{-3;2\right\}$.\\
			\begin{note}
				Ta cũng có thể phân tích như sau
				$x^2-2x+3x-6=x(x-2)+3(x-2)=(x-2)(x+3)=0.$
				\begin{itemize}
					\item Để biết một giá trị có là nghiệm của một đa thức hay không, ta thay biến bằng giá trị đó, rồi thực hiện phép tính. Nếu kết quả nhận được khác $0$ thì giá trị đó không là nghiệm. Ngược lại, giá trị đó là nghiệm của đa thức.
					\item Tập hợp nghiệm của đa thức thường được kí hiệu bằng chữ cái $S$ và được viết là $S=\left\{x_0;x_1;\ldots\right\}$
				\end{itemize}
			\end{note}
		\end{enumerate}
		
	}
\end{vd}

\begin{vd}%[7D4B9]

	\begin{listEX}[2]

		\item $B=x^2-5x+6$.
		\item $C=x^3+x^2+x+1$.
		\item $D=x^2+\sqrt{2015}.$
	\end{listEX}
	\loigiai{
		\begin{enumerate}
			\item Ta có $x^3-3x=0\Leftrightarrow x(x^2-3)=0\Leftrightarrow \hoac{&x=0\\&x^2=3}\Leftrightarrow \hoac{&x=0\\&x=\pm \sqrt{3}}$.\\
			Vậy tập hợp nghiệm của đa thức $A=x^3-3x$ là $S=\left\{-\sqrt{3};0;\sqrt{3}\right\}$.
			\item Ta có
			\begin{eqnarray*}
				x^2-5x+6=0&\Leftrightarrow &x^2-3x-2x+6=0\Leftrightarrow x(x-3)-2(x-3)=0\\
				&\Leftrightarrow & (x-3)(x-2)=0\Leftrightarrow\hoac{&x-3=0\\&x-2=0} \Leftrightarrow \hoac{&x=3\\&x=2.} 
			\end{eqnarray*}
			Vậy tập hợp nghiệm của đa thức $B=x^2-5x+6$ là $S=\left\{2;3\right\}$.
			\item Ta có $x^3+x^2+x+1=0\Leftrightarrow x^2(x+1)+(x+1)=0\Leftrightarrow (x+1)(x^2+1)=0$.\\
			Suy ra $x+1=0\Leftrightarrow x=-1$.
			Vậy tập hợp nghiệm của đa thức $C=x^3+x^2+x+1$ là $S=\left\{-1\right\}$ hay đa thức $C=x^3+x^2+x+1$ tuy là một đa thức bậc ba nhưng lại có duy nhất một nghiệm.
			\item Nhận xét: $x^2\ge 0$ và $\sqrt{2015}>0$. Suy ra $D=x^2+\sqrt{2015}>0$ hay $D=x^2+\sqrt{2015}\ne 0$ $\forall x$, tức là đa thức $D=x^2+\sqrt{2015}$ không có nghiệm (vô nghiệm).
		\end{enumerate}
		\begin{note}
			\begin{itemize}
				\item Để tìm nghiệm của một đa thức ta đặt đa thức đó bằng $0$. Biến đổi đa thức thành tích của các nhân tử (các thừa số). Tích bằng $0$ khi một trong các thừa số bằng $0$.
				\item $x^2+1\ge 1\Rightarrow x^2+1\ne 0$ $\forall x$, hay đa thức $f(x)=x^2+1$ không có nghiệm.
			\end{itemize}
		\end{note}
	}
\end{vd}

\begin{vd}%[7D4B9]
	Xác định $m$ để đa thức $f(x)=mx^2-2x+8$ có nghiệm là $-1$.

		\begin{itemize}
			\item \textit{Cách 1}: Đa thức $f(x)=mx^2-2x+8$ có nghiệm là $-1$ suy ra
			\begin{center}
				$f(-1)=m(-1)^2-2(-1)+8=0\Rightarrow m+2+8=0\Rightarrow m= -10$.
			\end{center}
			Vậy khi $m=-10$ thì đa thức $f(x)=mx^2-2x+8$ có nghiệm là $-1$.
			\item \textit{Cách 2}: Đa thức $f(x)=mx^2-2x+8=0$ có nghiệm là $-1$, suy ra $m+8=-2\Rightarrow m=-10$.\\
			Vậy khi $m=-10$ thì đa thức $f(x)=mx^2-2x+8$ có nghiệm là $-1$.
		\end{itemize}	
		\begin{note}
			\begin{itemize}
				\item $f(x)=ax^2+bx+c$ có nghiệm là $+1\Rightarrow a+b+c=0$.
				\item $f(x)=ax^2+bx+c$ có nghiệm là $-1\Rightarrow a-b+c=0$.
			\end{itemize}
		\end{note}
	}
\end{vd}

\begin{vd}%[7D4B9]%[7D4K9]
	Cho các đa thức $A=3a^2-4a+5$, $B=2a^2+5a-4$ và $C=4a^2-a+3$.
	\begin{enumerate}
		\item Tính $P=A+B+C$, $Q=A+B-C$, $R=A-B+C$, $S=-A+B+C$.
		\item Tính giá trị của đa thức $M=P-Q+R-S$ tại $a\in \left\{-2;-3\right\}$.
		\item Chứng minh đa thức $M$ không có nghiệm.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Phương pháp cộng ngang
			\begin{center}
				$P=A+B+C=(3a^2-4a+5)+(2a^2+5a-4)+(4a^2-a+3)=9a^2+4$
			\end{center}
			Phương pháp cộng dọc:
			\begin{center}
				\begin{tabular}{|c|c|c|c|c|}
					\hline
					 & $P=A+B+C$ & $Q=A+B-C$ & $R=A-B+C$ & $S=-A+B+C$\\
					\hline
					$A=3a^2-4a+5$ & $3a^2-4a+5$ & $3a^2-4a+5$ & $3a^2-4a+5$ & $-3a^2+4a-5$\\
					\hline
					$B=2a^2+5a-4$ & $2a^2+5a-4$ & $2a^2+5a-4$ & $-2a^2-5a+4$ & $2a^2+5a-4$\\
					\hline
					$C=4a^2-a+3$ & $4a^2-a+3$ & $-4a^2+a-3$ & $4a^2-a+3$ & $4a^2-a+3$\\
					\hline
					 & $9a^2+4$ & $a^2+2a-2$ & $5a^2-10a+12$ & $3a^2+8a-6$\\
					\hline
				\end{tabular}
			\end{center}
			\item Ta có 
			\begin{eqnarray*}
				M&=&P-Q+R-S=(A+B+C)-(A+B-C)+(A-B+C)-(-A+B+C)\\
				&=&2A-2B+2C=2(A-B+C)=2r=10a^2-20a+24.
			\end{eqnarray*}
			Vậy $M(-2)=10\cdot (-2)^2-20\cdot (-2)+24=104$.\\ $M(-3)=10\cdot (-3)^2-20\cdot (-2)+24=174.$
			\item Có
			\begin{eqnarray*}
				M&=&10a^2-20a+24=10a^2-20a+10+14=10(a^2-2a+1)+14\\
				&=&10(a^2-a-a+1)+14=10\left[a(a-1)-(a-1)\right]+14=10(a-1)^2+14\ge 14.
			\end{eqnarray*}
			Do $M\ge 14\Rightarrow M\ne 0$, $\forall a\Rightarrow $ đa thức $M$ không có nghiệm.
		\end{enumerate}
	}
\end{vd}

\begin{vd}%[7D4B9]%[7D4K9]
	Cho đa thức $\varphi (x)=ax^3-bx^2+cx-d$ $(a\ne 0)$.
	\begin{enumerate}
		\item Tìm sự liên hệ giữa các hệ số $a$ và $c$, $b$ và $d$ sao cho $\varphi (x)$ có hai nghiệm là $-1$ và $1$.
		\item Biết đa thức $\varphi (x)$ có hai nghiệm là $\pm 2$, tìm nghiệm thứ ba còn lại.\\
		Bạn BEE không biết giải bài toán trên. Các bạn có giúp BEE được không?
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Vì $\pm 1$ là nghiệm của đa thức $\varphi (x)$ nên ta có 
			$\varphi (1)=a-b+c-d=0$ \hfill (1)\\
			$\varphi (-1)=-a-b-c-d=0$ hay $a+b+c+d=0$ \hfill (2)\\
			Lấy $(1)$ + $(2)\Rightarrow a+c=0\Rightarrow c=-a$.\\
			Tương tự, lấy $(2)-(1)\Rightarrow b+d=0\Rightarrow d=-b$.\\
			Vậy, khi $c=-a$ và $d=-b$ thì đa thức $\varphi (x)$ có hai nghiệm là $\pm 1$. 
			\item Vì $\pm 2$ là nghiệm của đa thức $\varphi (x)$, nên ta có
			$\varphi (2)=8a-4b+2c-d=0$ \hfill (3)\\
			$\varphi (-2)=-8a-4b-2c-d=0$ hay $8a+4b+2c+d=0$ \hfill (4)\\
			Lấy $(3)+(4)\Rightarrow 16a+4c=0\Rightarrow c=-4a.$\\
			Tương tự, lấy $(4)-(3)\Rightarrow 8b+2d=0\Rightarrow d=-4b$. \\
			Do đó, khi $c=-4a$ và $d=-4b$ thì đa thức $\varphi (x)$ có hai nghiệm là $\pm 2$.\\
			Thay $c=-4a$ và $d=-4b$, ta có:
			\begin{eqnarray*}
				\varphi (x)&=&ax^3-bx^2-4ax+4b=(ax^3-bx^2)-(4ax-4b)\\
				&=& x^2(ax-b)-4(ax-b)=(ax-b)(x^2-4).
			\end{eqnarray*}
			$\varphi (x)=0\Leftrightarrow (ax-b)(x^2-4)=0\Leftrightarrow \hoac{&x^2=4\\&ax-b=0}\Leftrightarrow \hoac{&x=\pm 2\\&x=\dfrac{b}{a}}$ (do $a\ne 0$).\\
			Vậy ngoài hai nghiệm là $\pm 2$, đa thức $\varphi (x)$ có nghiệm thứ ba là $\dfrac{b}{a}$ (nghiệm này có thể trùng nghiệm đa cho).
			
		\end{enumerate}
	}
\end{vd}

\begin{vd}%[7D4K9]
	Bạn BEE cho rằng đa thức $f(x)=x^3-1$ là một đa thức bậc ba, nên ngoài nghiệm là $1$ thì đa thức $f(x)$ còn có hai nghiệm nữa. Các bạn thấy thế nào?
	\loigiai{
		Gọi $S$ là tập hợp nghiệm của đa thức $f(x)=x^3-1$.\\
		Giả sử $a\ne 1$ là một nghiệm nào đó của đa thức $f(x)=x^3-1$.\\
		Xét các khoảng giá trị của $a$ ta thấy.
		\begin{itemize}
			\item Nếu $a\le 0$ thì $a^3\le 0$, do đó $f(a)=a^3-1=a^3+(-1)<0$ nên $f(a)\ne 0$ với $\forall a\le 0$, tức là $a\notin S$.
			\item Nếu $0<a<1$ thì $0<a^3<1$, do đó $f(a)=a^3-1<0$, nên $f(a)\ne 0$, tức là $a\notin S$.
			\item Nếu $a>1$ thì $a^3>1$, do đó $f(a)=a^3-1>1$ nên $f(a)\ne 0$, $\forall a>1$, tức là $a\notin S$.
		\end{itemize}
		Như vậy, ngoài $a=1$ thì không còn giá trị nào của $a$ nữa để $f(a)=0$, hay đa thức $f(x)=x^3-1$ chỉ có một nghiệm duy nhất là $1$, mặc dù đa thức đó là đa thức bậc ba.
		\begin{note}
			Đa thức bậc $n$ có số nghiệm không vượt quá $n$. Đa thức bậc $n$ có thể có $1$ nghiệm, $2$ nghiệm, $3$ nghiệm,$\ldots$, $n$ nghiệm hoặc vô nghiệm.
		\end{note}
	}
\end{vd}

\begin{bt}%[7D4K9]

	\loigiai{
		Gọi $S$ là tập hợp nghiệm của đa thức $f(x)=x^3+1$.\\
		Giả sử $a\ne 1$ là một nghiệm nào đó của đa thức $f(x)=x^3+1$.\\
		Xét các khoảng giá trị của $a$ ta thấy.
		\begin{itemize}
			\item Nếu $a< -1$ thì $a^3< -1$, do đó $f(a)=a^3+1<0$ nên $f(a)\ne 0$ với $\forall a\le 0$, tức là $a\notin S$.
			\item Nếu $-1<a<0$ thì $-1<a^3<0$, do đó $f(a)=a^3+1>0$, nên $f(a)\ne 0$, tức là $a\notin S$.
			\item Nếu $a\ge 0$ thì $a^3\ge 0$, do đó $f(a)=a^3+1\ge 1>0$ nên $f(a)\ne 0$, $\forall a\ge 0$, tức là $a\notin S$.
		\end{itemize}
		Như vậy, ngoài $a=-1$ thì không còn giá trị nào của $a$ nữa để $f(a)=0$, hay đa thức $f(x)=x^3-1$ chỉ có một nghiệm duy nhất là $1$, mặc dù đa thức đó là đa thức bậc ba.
	}
\end{bt}

\begin{bt}%[7D4K9]
	Cho các đa thức $f(x)=ax+b$ và $g(x)=bx+a$ ($a$, $b\ne 0$). Giả sử đa thức $f(x)$ có nghiệm là $x_0$, tìm nghiệm của đa thức $g(x)$.
	\loigiai{
		$x_0$ là nghiệm của đa thức $f(x)=0$ nên $ax_0+b=0\Rightarrow x_0=\dfrac{-b}{a}$.\\
		Gọi $x_1$ là nghiệm của $g(x)=0\Rightarrow bx_1+a=0\Rightarrow x_1=\dfrac{-a}{b}=\dfrac{1}{-\dfrac{b}{a}}=\dfrac{1}{x_0}$.\\
		Vậy nghiệm của đa thức $g(x)=0$ là $\dfrac{1}{x_0}$. 
	}
\end{bt}

\begin{bt}%[7D4G9]
	Tìm các hệ số $a$, $b$, $c$, $d$ của đa thức $ax^3+bx^2+cx+d$. Biết $f(0)=-5$, $f(1)=4$, $f(2)=31$, $f(3)=88$.
	\loigiai{

	}
