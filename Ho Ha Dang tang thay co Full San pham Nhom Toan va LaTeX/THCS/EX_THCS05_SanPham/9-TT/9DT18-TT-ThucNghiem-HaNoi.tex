\begin{ex}%[9D1Y2]%[9D1B8]%[9D1K8]
Cho biểu thức $ P=\dfrac{x\sqrt{x}-1}{x-\sqrt{x}}+\dfrac{x\sqrt{x}+1}{x+\sqrt{x}}-\dfrac{4}{\sqrt{x}} $ và $ Q=\dfrac{\sqrt{x}-1}{\sqrt{x}+1} $ (với $ x>0,x\neq 1 $).
\begin{enumerate}
\item Tính giá trị của biểu thức $ Q $ khi $ x=25 $.
\item Rút gọn biểu thức $ A=P\cdot Q $.
\item Tìm $ x $ để $ A\cdot \sqrt{x}<8 $.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Với $ x=25 $, ta có $ Q=\dfrac{\sqrt{25}-1}{\sqrt{25}+1}=\dfrac{2}{3} $.
\item Ta có
\begin{align*}
P&=\dfrac{(\sqrt{x}-1)(x+\sqrt{x}+1)}{\sqrt{x}(\sqrt{x}-1)}+\dfrac{(\sqrt{x}+1)(x-\sqrt{x}+1)}{\sqrt{x}(\sqrt{x}+1)}-\dfrac{4}{\sqrt{x}}\\
&=\dfrac{x+\sqrt{x}+1}{\sqrt{x}}+\dfrac{x-\sqrt{x}+1}{\sqrt{x}}-\dfrac{4}{\sqrt{x}}=\dfrac{2(x-1)}{\sqrt{x}}.
\end{align*}
Do đó $ A=\dfrac{2(x-1)}{\sqrt{x}}\cdot \dfrac{\sqrt{x}-1}{\sqrt{x}+1}=\dfrac{2(\sqrt{x}-1)^{2}}{\sqrt{x}} $.
\item Với $ A\cdot \sqrt{x}<8\Leftrightarrow (\sqrt{x}-1)^{2}<4\Leftrightarrow -2<\sqrt{x}-1<2\Leftrightarrow \left\{\begin{aligned}&\sqrt{x}>-1\ (\mbox{luôn đúng})\\ &\sqrt{x}<3 \end{aligned}\right.\Leftrightarrow x<9 $.\\
Kết hợp với điều kiện, ta có $ 0<x<9 $, $ x\neq 1 $.
\end{enumerate}
}
\end{ex}
\begin{ex}(\textit{Giải bài toán bằng cách lập phương trình hoặc hệ phương trình})%[9D3B6]
	\hfill\\
Một ô tô đi từ $ A $ đến $ B $ trong một thời gian dự định. Nếu vận tốc tăng thêm $ 14 $ km/h thì đến $ B $ sớm hơn $ 2 $ giờ so với dự định. Nếu vận tốc giảm đi $ 4 $ km/h thì sẽ đến $ B $ chậm hơn $ 2 $ giờ so với dự định. Tính khoảng cách $ AB $, vận tốc và thời gian dự định của ô tô.
\loigiai{
Gọi $ x $, $ y $ lần lượt là vận tốc và thời gian dự định của ô tô ($ x $ km/h, $ y $ giờ, $ x>4 $, $ y>2 $)\\
$ \Rightarrow $ quãng đường $ AB $ là $ xy $ km.\\
Nếu vận tốc ô tô tăng thêm $ 14 $ km/h thì thời gian ô tô đi hết quãng đường $ AB $ là $ y-2 $ giờ\\
$\Rightarrow $ ta có phương trình $ xy=(x+14)(y-2)\Leftrightarrow -x+7y=14 $.\hfill($ 1 $)\\
Nếu vận tốc ô tô giảm đi $ 4 $ km/h thì thời gian ô tô đi hết quãng đường $ AB $ là $ y+2 $ giờ\\
$ \Rightarrow $ ta có phương trình $ xy=(x-4)(y+2)\Leftrightarrow x-2y=4 $.\hfill($ 2 $)\\
Từ ($ 1 $) và ($ 2 $) ta có hệ phương trình $ \left\{\begin{aligned}& -x+7y=14\\ &x-2y=4\end{aligned}\right.\Leftrightarrow \left\{\begin{aligned}& x=\dfrac{56}{5}\\ &y=\dfrac{18}{5}\end{aligned}\right. $ (thỏa mãn điều kiện).\\
Vậy quãng đường $ AB $ là $ 40{,}32 $ km.\\
Thời gian dự định đi hết quãng đường $ AB $ của ô tô là $ \dfrac{18}{5}=3{,}6 $ giờ.\\
Vận tốc dự định của ô tô là $ \dfrac{56}{5}=11{,}2 $ km/h.
}
\end{ex}
\begin{ex}%[9D3K5]%[9D3K3]
\hfill
\begin{enumerate}
\item Giải hệ phương trình $ \left\{\begin{aligned}&\dfrac{1}{x+2}+2\sqrt{y-3}=7\\ &\dfrac{2}{x+2}-3\sqrt{y-3}=-7 \end{aligned}\right. $.
\item Cho hệ phương trình $ \left\{\begin{aligned}&(m+1)x-y=3\\ &mx+y=m \end{aligned}\right. $. Tìm $ m $ để hệ có nghiệm duy nhất thỏa mãn điều kiện $ 2x+y>0 $.
\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Điều kiện xác định: $ x\neq -2 $, $ y\geq 3 $.\\
		Đặt $ a=\dfrac{1}{x+2} $, $ b=\sqrt{y-3} $ ($ b\geq 0 $). Ta có hệ phương trình
		\[ 
		\left\{\begin{aligned}&a+2b=7\\ &2a-3b=-7 \end{aligned}\right.
		\Leftrightarrow \left\{\begin{aligned}&b=3a\\ &7a=7 \end{aligned}\right.
		\Leftrightarrow \left\{\begin{aligned}&a=1\\ &b=3 \end{aligned}\right. .
		\]
		Với $ a=1 $, $ b=3 $ ta có $ \left\{\begin{aligned}&x+2=1\\ &\sqrt{y-3}=3 \end{aligned}\right.\Leftrightarrow \left\{\begin{aligned}&x=-1\\ &y=12 \end{aligned}\right. $ (thỏa mãn điều kiện).\\
		Vậy hệ phương trình đã cho có nghiệm là $ (x;y)=(-1;12) $.
		\item Để hệ có nghiệm duy nhất thì $ \dfrac{m+1}{m}\neq \dfrac{-1}{1}\Leftrightarrow m\neq -\dfrac{1}{2} $.\\
		Ta có $ \left\{\begin{aligned}&(m+1)x-y=3\\ &mx+y=m \end{aligned}\right. \Leftrightarrow \left\{\begin{aligned}&(2m+1)x=m+3\\ &mx+y=m\end{aligned}\right.
		\Leftrightarrow \left\{\begin{aligned}&x=\dfrac{m+3}{2m+1}\\ &y=\dfrac{m^{2}-2m}{2m+1} \end{aligned}\right.$.\\
		Thay vào bài toán, ta có\\
		$ 2x+y>0\Leftrightarrow \dfrac{2(m+3)}{2m+1}+\dfrac{m^{2}-2m}{2m+1}>0\Leftrightarrow \dfrac{m^{2}+6}{2m+1}>0\Leftrightarrow 2m+1>0\Leftrightarrow m>-\dfrac{1}{2} $ (vì $ m^{2}+6>0 $).
	\end{enumerate}
}
\end{ex}

\item Chứng minh tứ giác $ CDHE $ là tứ giác nội tiếp.
\item Kéo dài $ AD $ cắt $ (O) $ tại $ N $. Chứng minh $ \triangle AHF $ cân và $ C $ là điểm chính giữa cung $ NF $.
\item Gọi $ M $ là trung điểm của cạnh $ AB $. Chứng minh $ ME $ là tiếp tuyến của đường tròn ngoại tiếp $ \triangle CDE $.
\item Cho $ B $, $ C $ là các điểm cố định và $ BC=R\sqrt{3} $. Hãy xác định vị trí của $ A $ trên $ (O;R) $ để $ DH\cdot DA $ lớn nhất.

\begin{enumerate}
\item Vì $ AD $, $ BE $ là các đường cao của tam giác $ ABC $ nên $ \widehat{ADC}=\widehat{BEC}=90^\circ$\\

\item Ta có $ \widehat{AFB}=\widehat{ACB} $ (vì cùng chắn cung $ \wideparen{AB} $).\hfill ($ 1 $)\\

\begin{ex}%[9D0G7]
Cho các số dương $ x $, $ y $, $ z $ thỏa mãn điều kiện $ xy+yz+zx=3 $. Tìm giá trị lớn nhất của biểu thức $ P=\dfrac{x}{\sqrt{x^{2}+3}}+\dfrac{y}{\sqrt{y^{2}+3}}+\dfrac{z}{\sqrt{z^{2}+3}} $.
\loigiai{
Xét
\begin{align*}
\dfrac{x}{\sqrt{x^{2}+3}}&=\dfrac{x}{\sqrt{x^{2}+xy+yz+zx}}=\dfrac{x}{\sqrt{(x+y)(x+z)}}\\
&\leq \dfrac{x}{2}\left(\dfrac{1}{x+y}+\dfrac{1}{x+z} \right)=\dfrac{1}{2}\left(\dfrac{x}{x+y}+\dfrac{x}{x+z} \right).
\end{align*}
Tương tự ta cũng có
\begin{align*}
\dfrac{y}{\sqrt{y^{2}+3}}&\leq\dfrac{1}{2}\left(\dfrac{y}{y+x}+\dfrac{y}{y+z} \right).\\
\dfrac{z}{\sqrt{z^{2}+3}}&\leq\dfrac{1}{2}\left(\dfrac{z}{z+x}+\dfrac{z}{z+y} \right).
\end{align*}
Khi đó ta có 
\[ 
P\leq \dfrac{1}{2}\left(\dfrac{x}{x+y}+\dfrac{x}{x+z}+\dfrac{y}{y+x}+\dfrac{y}{y+z}+\dfrac{z}{z+x}+\dfrac{z}{z+y} \right)=\dfrac{3}{2}.
\]
Dấu bằng xảy ra khi và chỉ khi $ x=y=z=1 $.\\
Vậy $ P_{\max}=\dfrac{3}{2} $ khi $ x=y=z=1 $.
}
\end{ex}
