\begin{ex}%[Đề thi vào 10, Chuyên Tây Ninh, 2016]%[9H3K4]
    Cho tam giác $ABC$ vuông tại $A$ có đường cao $AH$ ($H$ thuộc cạnh $BC$), cho $BH=2$ và $CH=m$. Xác định $m$ để đường thẳng $BC$ tiếp xúc với đường tròn có tâm là điểm $A$ và có bán kính $R=4$. Khi đó tính độ dài các cạnh $AB$ và $AC$.
    \loigiai{
    Để $BC$ tiếp xúc với đường tròn $(A,R=4)$ thì $AH=4$.\\
    Mặt khác tam giác $ABC$ vuông tại $A$ nên ta có $AH^2=BH.CH\Leftrightarrow 4^2=2.m\Leftrightarrow m=8$.
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Tây Ninh, 2016]%[9D4K6]
    Tìm $m$ để phương trình $x^2-2(m-1)x+m^2+2=0$ có nghiệm $x_1$, $x_2$ sao cho $T=x_1+x_2-(x_1^2+x_2^2)$ đạt giá trị lớn nhất.
    \loigiai{
    Để phương trình có nghiệm $x_1$, $x_2$ thì $\Delta'=(m-1)^2-m^2-2=-2m-1\ge 0\Leftrightarrow m\le -\dfrac{1}{2}$.\hfill $(1)$\\
    Theo định lí Vi-et ta có $\heva{x_1+x_2=2(m-1)\\ x_1.x_2=m^2+2}$.\hfill $(2)$\\
    Ta có $T=(x_1+x_2)-(x_1^2+x_2^2)=(x_1+x_2)-\left((x_1+x_2)^2-2x_1x_2\right)=-2(m-\dfrac{5}{2})^2+\dfrac{21}{2}\le \dfrac{21}{2}$ (theo $(2)$).\\
    Vậy $T_{\max}=\dfrac{21}{2}$ khi $m=\dfrac{5}{2}$.
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Tây Ninh, 2016]%[9D4G3]
    Cho hai số thực $m$, $n$ khác $0$ thỏa mãn $\dfrac{1}{m}+\dfrac{1}{n}=\dfrac{1}{2}$. Chứng minh rằng phương trình $(x^2+mx+n)(x^2+nx+m)=0$ luôn có nghiệm.
    \loigiai{
    Phương trình $x^2+mx+n=0$ có $\Delta_1=m^2-4n$.\\
    Phương trình $x^2+nx+m=0$ có $\Delta_2=n^2-4m$.\\
    Theo bài ra ta có $\dfrac{1}{m}+\dfrac{1}{n}=\dfrac{1}{2}\Leftrightarrow mn=2(m+n).$ \hfill $(*)$ \\
    Thay $(*)$ vào $\Delta_1+\Delta_2=m^2+n^2-4(m+n)$ ta được
    $$\Delta_1+\Delta_2=m^2+n^2-2mn=(m+n)^2\ge 0.$$
    Vậy ít nhất một trong hai $\Delta_1$, $\Delta_2$ phải lớn hơn hoặc bằng $0$, có nghĩa là phương trình đã cho luôn có nghiệm.
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Tây Ninh, 2016]%[9D4G7]
	Giải hệ phương trình $\heva{x^3+2x=y^3-8y\\ 3(x^2+1)=y^2-3.}$
	\loigiai{
	Hệ phương trình tương đương với $\heva{x(x^2+2)=y(y^2-8)\quad (1)\\ 3(x^2+2)=y^2.\quad (2)}$\\
	Từ phương trình $(2)$ ta thấy $y\ne 0$, nên chia tương ứng hai vế phương trình $(1)$ cho hai vế phương trình $(2)$ ta được
	$$\dfrac{x}{3}=y-\dfrac{8}{y}\Leftrightarrow xy=3y^2-24\Leftrightarrow y(3y-x)=24.\quad\quad\quad\quad\quad\quad\quad(3)$$
	Từ đó ta cũng có
	$$9(x^2+2)=3y^2=24+xy\Rightarrow 9x^2-xy=6\Leftrightarrow x(9x-y)=6.\quad (4)$$
	Từ $(4)$ ta thấy $x\ne 0$ và $9x-y\ne 0$, nên chia $(3)$ cho $(4)$ ta được
	$$\dfrac{3y^2-xy}{9x^2-xy}=4\Leftrightarrow 36x^2-3xy-3y^2=0\Leftrightarrow \hoac{y&=3x\\ y&=-4x.}$$
	\begin{itemize}
	\item Với $y=3x$, thay vào phương trình $(2)$ ta có $3x^2+6=9x^2\Leftrightarrow x^2=1\Leftrightarrow\hoac{x&=-1\\x&=1.}$
	\item Với $y=-4x$, thay vào phương trình $(2)$
ta có $3x^2+6=16x^2\Leftrightarrow x^2=\dfrac{6}{13}\Leftrightarrow\hoac{x&=-\sqrt{\dfrac{6}{13}}\\x&=\sqrt{\dfrac{6}{13}}}$	
	\end{itemize}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Tây Ninh, 2016]%[9H3G7]

Gọi $M$ là một điểm bất kỳ trên cung nhỏ $\wideparen{AC}$ của đường tròn $(O)$ ($M$ khác $A$ và $C$), trên tia $BM$ lấy một điểm $E$ sao cho $ME=MC$ ($E$ ở ngoài đoạn $BM$). Chứng minh rằng đường tròn tâm $A$ bán kính $AE$ luôn đi qua $B$ và $C$.

	\tkzDrawPoints(A,B,C,D,O,M,E)

	\quad $AM$ chung,
	
	\quad $\widehat{AMC}=\widehat{AME}$ (theo $(1)$),
	
	\quad $MC=ME$ (gt)\\
	Suy ra $\triangle AMC=\triangle AME\Rightarrow AE=AC=AB$.\\
	Vậy đường tròn tâm $A$ bán kính $AE$ luôn đi qua $B$ và $C$.

\begin{ex}%[Đề thi vào 10, Chuyên Tây Ninh, 2016]%[9H3K4]%[9H3G7]
	Từ một điểm $M$ nằm ngoài đường tròn $(O)$ kẻ hai tiếp tuyến $MA$, $MB$ với $(O)$ ($A$, $B$ là các tiếp điểm). Gọi $C$ là giao điểm của $MO$ và $AB$, lấy $D$ thuộc đoạn thẳng $AC$ ($D$ khác $A$ và $C$). Đường thẳng $MD$ cắt $(O)$ tại hai điểm $E$ và $F$ ($ME<MF$).
	\begin{enumerate}
	\item Chứng minh rằng $MA^2=ME.MF$.
	\item Chứng minh rằng bốn điểm $E$, $C$, $O$, $F$ cùng nằm trên một đường tròn.
	\end{enumerate}
	\loigiai{
	\begin{center}
	\begin{tikzpicture}
	\tkzDefPoints{0/0/M,4/0/O}
	\coordinate (m) at ($(M)!0.4!(O)$);
	\coordinate (n) at ($(M)!0.5!(O)$);
	\tkzDrawCircle[radius](O,m)
	\tkzDefCircle[radius](n,M)
	\tkzInterCC(n,M)(O,m)\tkzGetPoints{A}{B}
	\tkzInterLL(A,B)(M,O)\tkzGetPoint{C}
	\coordinate (D) at ($(A)!0.7!(C)$);
	\tkzInterLC(M,D)(O,m)\tkzGetPoints{E}{F}
	
	\tkzDrawSegments(M,A M,B M,O O,A O,B A,B M,F E,A A,F E,C O,F)
	\tkzDrawPoints(O,M,A,B,C,D,E,F)
	\tkzLabelPoints(O,C,D)
	\tkzLabelPoints[above left](A,E)
	\tkzLabelPoints[left](M)
	\tkzLabelPoints[right](F)
	\tkzMarkRightAngles(M,A,O A,C,O)
	\end{tikzpicture}
	\end{center}
	\begin{enumerate}
	\item Xét hai tam giác $AME$ và $FMA$ có
	
	\quad $\widehat{AME}=\widehat{FMA}$
	
	\quad $\widehat{MAE}=\widehat{MFA}$ (cùng chắn cung $\wideparen{AE}$)\\
	Vậy $\triangle AME\backsim \triangle FMA$ (g-g), suy ra $\dfrac{AM}{FM}=\dfrac{ME}{MA}\Leftrightarrow MA^2=ME.MF$.\hfill ($1$)
	\item Do tam giác $MAO$ vuông tại $A$ nên ta có $MA^2=MC.MO$.\hfill ($2$)\\
	Xét hai tam giác $MCE$ và $MFO$ có
	
	\quad $\widehat{CME}=\widehat{FMO}$
	
	\quad $\dfrac{ME}{MO}=\dfrac{MC}{MF}$ (do $(1)$ và $(2)$)
	
	Vậy $\triangle MCE\backsim\triangle MFO$ (c-g-c), suy ra $\widehat{MCE}=\widehat{MFO}$, suy ra tứ giác $COFE$ nội tiếp.
	
	
	\end{enumerate}	    
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Tây Ninh, 2016]%[9D5G1]
	Cho $a$, $b$, $c$ là các số thực dương thỏa mãn $a+b+c=1$. Tìm giá trị nhỏ nhất của biểu thức $P=\dfrac{9}{1-(ab+bc+ca)}+\dfrac{1}{4abc}$.
	\loigiai{
	Ta có 
	$$1-(ab+bc+ca)=(a+b+c)^2-(ab+bc+ca)=a^2+b^2+c^2+ab+bc+ca.$$
	Áp dụng bất đẳng thức Cauchy-Schwarz ta có:
	\begin{align*}
	P&=\dfrac{9}{\sum a^2+\sum ab}+\dfrac{\sum a}{4abc}=\dfrac{9}{\sum a^2+\sum ab}+\dfrac{1}{4}\sum\dfrac{1}{ab}\\
	&\ge \dfrac{9}{\sum a^2+\sum ab}+\dfrac{1}{4}\dfrac{9}{\sum ab}=\dfrac{9}{2}\left(\dfrac{1}{\sum a^2+\sum ab}+\dfrac{1}{\sum a^2+\sum ab}+\dfrac{1}{2\sum ab}\right)\ge \dfrac{9}{2}.\left(\dfrac{9}{2\left(\sum a\right)^2}\right)\\
	&=\dfrac{81}{4}.
	\end{align*}
	Dấu "$=$" xảy ra khi và chỉ khi $a=b=c=\dfrac{1}{3}$.
	}
\end{ex}
