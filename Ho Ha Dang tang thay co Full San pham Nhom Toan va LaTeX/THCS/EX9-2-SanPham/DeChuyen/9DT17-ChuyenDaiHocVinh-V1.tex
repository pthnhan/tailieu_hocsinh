\begin{ex}%[Đề thi vào 10, Chuyên Đại Học Vinh, Vòng 1, 2017]%[9D1K8]
Cho các số thực dương $a$, $b$ thỏa mãn $a+b=5$, $ab=2$. Tính giá trị của biểu thức $$A=\left(\dfrac{a\sqrt{a}+b\sqrt{b}}{\sqrt{a}+\sqrt{b}}-\sqrt{ab}\right)\cdot\left(\dfrac{a\sqrt{a}-b\sqrt{b}}{\sqrt{a}-\sqrt{b}}+\sqrt{ab}\right).$$
\loigiai{\begin{align*}
	A&=\left(a-\sqrt{ab}+b-\sqrt{ab}\right)\left(a+\sqrt{ab}+b+\sqrt{ab}\right)\\&=\left(\sqrt{a}-\sqrt{b}\right)^2\cdot\left(\sqrt{a}+\sqrt{b}\right)^2\\&=(a-b)^2=(a+b)^2-4ab.
	\end{align*}
Thay $a+b=5$, $ab=2$ vào $A$ ta được $A=17$}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Đại Học Vinh, Vòng 1, 2017]%[9D4K6]
Tìm tất cả các giá trị của tham số $m$ để phương trình $x^2-2mx+m-1=0$ có hai nghiệm phân biệt $x_1$, $x_2$ thỏa mãn $2\left(x_1+x_2\right)+x_1^2x_2^2=0$.
\loigiai{Phương trình có hai nghiệm phân biệt $\Leftrightarrow$ $\Delta'=m^2-m+1>0$. Vì $m^2-m+1>0$ với mọi $m\in\mathbb{R}$ nên phương trình luôn có hai nghiệm phân biệt, kí hiệu $x_1$, $x_2$.

Áp dụng định lí Vi-ét ta có$\heva{x_1+x_2=2m,\\x_1x_2=m-1.}$

Do đó, $2\left(x_1+x_2\right)+x_1^2x_2^2=0$ $\Leftrightarrow 4m+(m-1)^2=0$ $\Leftrightarrow m=-1$.}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Đại Học Vinh, Vòng 1, 2017]%[9D5G5]%[9D4G7]
    \hfill
    \begin{enumerate}
        \item Giải hệ phương trình $\heva{x^2+xy&=36\\y^2+yx&=45.}$
        \item Giải phương trình $2\sqrt{x+1}+\sqrt{2x+3}=\sqrt{2x^2+11x-2}.$
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item Cộng hai phương trình theo vế của hệ ta được $x^2+2xy+y^2=81$ $\Leftrightarrow (x+y)^2=81$ $\Leftrightarrow\hoac{x+y&=9\\x+y&=-9.}$
        
        * Với $x+y=9\Leftrightarrow y=9-x$ thay vào phương trình $x^2+xy=36$ ta được $x^2+x(9-x)=36\Leftrightarrow x=4\Rightarrow y=5$.
        
        * Tương tự, với $x+y=-9$ ta được $x=-4, y=-5$.
        
      Vậy hệ có hai nghiệm $(x;y)$ là $(4;5)$ và $(-4;-5)$.
        \item Điều kiện $\heva{&x\geq -1,\\ &2x^2+11x-2\geq 0.}$ Với điều kiện đó, phương trình đã cho tương đương $$4\sqrt{2x^2+5x+3}=2x^2+5x-9\Leftrightarrow t^2-4t-12=0\,(t=\sqrt{2x^2+5x+3}\geq 0)\Leftrightarrow\hoac{t&=6\\t&=-2(\mbox{loại})}.$$
        
        Với $t=6\Leftrightarrow 2x^2+5x-33=0\Leftrightarrow\hoac{x&=3\\x&=-\dfrac{11}{2}}$. Đối chiếu điều kiện, ta có nghiệm của phương trình là $x=3$.
    \end{enumerate}
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Đại Học Vinh, Vòng 1, 2017]%[9H3B7]%[9H3K3]%[9H3G3]
    Cho tam giác $ABC$ có ba góc nhọn, nội tiếp đường tròn $(O)$. Gọi $H$ là trực tâm của tam giác $ABC$, vẽ hình bình hành $BHCQ$.
    \begin{enumerate}
        \item Chứng minh rằng $ABQC$ là tứ giác nội tiếp đường tròn.
        \item Chứng minh $\widehat{BAH}=\widehat{CAO}$.
        \item Gọi $M$ là giao điểm của $HQ$ và $BC$, $G$ là giao điểm của $AM$ với $OH$. Chứng minh rằng $G$ là trọng tâm của tam giác $ABC$.
        \item Chứng minh rằng nếu $OG$ song song với $BC$ thì $\tan\widehat{AQB}\cdot\tan\widehat{AQC}=3$.
    \end{enumerate}
\loigiai{\begin{center}
	\begin{tikzpicture}[scale=1]
	\tkzInit[xmin=-3.5,xmax=6 ,ymin=-6 ,ymax=4] 
	\tkzClip[space=.5]
	\tkzDefPoints{0/3/A, -3/-2/B, 5/-2/C}
	\tkzDrawPolygon(A,B,C)
	\tkzDrawCircle[circum](A,B,C)
	\tkzCircumCenter(A,B,C)\tkzGetPoint{O}
	\tkzDrawAltitude(B,C)(A)\tkzGetPoint{K}
	\tkzDrawAltitude(A,C)(B)\tkzGetPoint{y}
	\tkzDrawAltitude(A,B)(C)\tkzGetPoint{z}
	\tkzInterLL(A,K)(B,y)\tkzGetPoint{H}
	\tkzDefMidPoint(C,B)\tkzGetPoint{M}
	\tkzInterLL(O,H)(A,M)\tkzGetPoint{G}
	\tkzDefPointBy[translation = from H to C](B)\tkzGetPoint{Q}
	\tkzDrawSegments(B,Q C,Q A,M O,H A,Q H,Q)
	\tkzDrawPoints[fill=black](O,G,H,M)
	\tkzMarkRightAngle(A,K,C)
	\tkzMarkRightAngle(B,y,C)
	\tkzMarkRightAngle(C,z,A)
	\tkzLabelPoints[above](A)
	\tkzLabelPoints[right](O)
	\tkzLabelPoints[below left](B,G)
	\tkzLabelPoints[left](H)
	\tkzLabelPoints[below](K)
	\tkzLabelPoints[below right](C,Q,M)
	\tkzMarkSegments[mark=||](M,B M,C)
	\end{tikzpicture}
	\end{center}
	\begin{enumerate}
			\item Vì tứ giác $BHCQ$ là hình bình hành nên $BH\parallel QC$, $CH\parallel BQ$ do đó, $AC\perp CQ$, $AB\perp BQ$. Từ đó suy ra $ABQC$ nội tiếp đường tròn đường kính $AQ$ hay $A,B,C,Q$ nằm trên đường tròn $(O)$.
			\item Ta có: $\widehat{ABC}=\widehat{AQC}$ (hai góc nội tiếp cùng chắn cung $\wideparen{AC}$) mà $\widehat{BAH}+\widehat{ABC}=\widehat{CAO}+\widehat{AQC}=90^\circ$ suy ra $\widehat{BAH}=\widehat{CAO}$. 
			\item Vì $BHCQ$ là hình bình hành nên $M$ là trung điểm $BC$ và $HQ$. Tam giác $AHQ$ có $M$ là trung điểm $HQ$, $O$ là trung điểm $AQ$, $OH$ cắt $AM$ tại $G$ nên nhận $G$ làm trọng tâm. Suy ra $AG=\dfrac{2}{3}AM$ mà $M$ là trung điểm $BC$ do đó $G$ cũng là trọng tâm tam giác $ABC$.
			\item Ta có: $\tan\widehat{AQB}=\tan\widehat{ACB}=\dfrac{AK}{KC}$, $\tan\widehat{AQC}=\tan\widehat{ABC}=\dfrac{AK}{KB}$.
			
			Lại có, $\Delta HBK\backsim \Delta CAK$ (g.g) nên $\dfrac{HB}{AC}=\dfrac{HK}{CK}=\dfrac{BK}{AK}$ suy ra $AK.HK=BK.CK$.
			
			Do đó, $\tan\widehat{AQB}\cdot\tan\widehat{AQC}=\dfrac{AK}{HK}=\dfrac{AM}{GM}=3$.
	\end{enumerate}}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Đại Học Vinh, Vòng 1, 2017]%[9D5G1]
    Cho các số thực dương $x$, $y$, $z$ thỏa mãn $x^2+y^2+z^2=3$. Chứng minh rằng $$5(x+y+z)+\dfrac{1}{x^3}+\dfrac{1}{y^3}+\dfrac{1}{z^3}\geq 18.$$
\loigiai{Áp dụng bất đẳng thức Cô-si ta được $x+\dfrac{1}{x^3}\geq\dfrac{2}{x}$. Ta chỉ cần chứng minh $4x+\dfrac{2}{x}\geq x^2+5$. Thật vậy, $4x+\dfrac{2}{x}-x^2-5\geq 0$ $\Leftrightarrow (x-1)^2(x-2)\leq 0$ (đúng, vì $x^2<3$ nên $x<2$).}
\end{ex}
