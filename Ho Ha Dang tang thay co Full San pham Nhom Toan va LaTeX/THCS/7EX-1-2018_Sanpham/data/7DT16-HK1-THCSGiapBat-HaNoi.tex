		{\True $2$}
		{$0{,}5$}
		{$\sqrt{2}$}

		}

	\noindent\textbf{II. PHẦN TỰ LUẬN}
	\setcounter{ex}{0}
	\begin{ex}%[Đề kiểm tra học kỳ I, Giap Bat, 2015 - 2016]%[Lê Thị Oanh, dự án 7EX-1-2018]%[7D1K4]
			Thực hiện phép tính \textit{(Tính nhanh nếu có thể)}:
			\begin{listEX}[2]
			\item $\dfrac{-3}{5}+0{,}8 \cdot \left(-7\dfrac{1}{2}\right).$
			\item $8 \dfrac{2}{3} \cdot \dfrac{7}{5} - 3\dfrac{2}{3}: \dfrac{5}{7}.$
			\item $\dfrac{3}{\sqrt{125}} + \left|-2\dfrac{1}{5}\right| : \left(\dfrac{-1}{2}\right)^3 - 1^{2008}.$
		\end{listEX}
			\loigiai
			{
				Ta có
				\begin{enumerate}
					\item  $\dfrac{-3}{5}+0{,}8 \cdot \left(-7\dfrac{1}{2}\right) = \dfrac{-3}{5}-6 = \dfrac{-3}{5}+\dfrac{4}{5}=\dfrac{-33}{5}.$
					\item  $8 \dfrac{2}{3} \cdot \dfrac{7}{5} - 3\dfrac{2}{3}: \dfrac{5}{7}
						= 8\dfrac{2}{3} \cdot \dfrac{7}{5}-3\dfrac{2}{3} \cdot \dfrac{7}{5}
						=\dfrac{7}{5}\left(8\dfrac{2}{3}-3\dfrac{2}{3}\right)
						=\dfrac{7}{5} \cdot 5=7.$
					\item  $\dfrac{3}{\sqrt{25}} + \left|-2\dfrac{1}{5}\right| : \left(\dfrac{-1}{2}\right)^3 - 1^{2008}
					=\dfrac{3}{5}+\dfrac{11}{5}:\dfrac{-1}{8}-1
					=\dfrac{3}{5}+\dfrac{11}{5} \cdot 8-1=\dfrac{86}{5}.$
			\end{enumerate}
		}
		\end{ex}
%		
		\begin{ex}%[Đề kiểm tra học kỳ I, THCS Giap Bat, 2015 - 2016]%[Lê Thị Oanh, dự án 7EX-1-2018]%[7D1B4]%[7D1K11]%[7D1K6]
			Tìm $x$ biết
			\begin{listEX}[2]
			\item  $\dfrac{4}{9}-\dfrac{11}{72} \cdot x=\dfrac{3}{4}.$
			\item  $|x+1|-6{,}3=-\sqrt{1{,}69}.$
			\item  $3^{x+3}-3^{x-1}=2160.$
			\end{listEX}
			\loigiai
			{
				Ta có
				
				\begin{enumerate}
					\item  
						\allowdisplaybreaks
						\begin{align*}
							&\dfrac{4}{9}-\dfrac{11}{72}.x=\dfrac{3}{4}\\
							&x = \left(\dfrac{4}{9} - \dfrac{3}{4}\right): \dfrac{11}{72}\\
							&x = -2.
						\end{align*}
					\item 
						\allowdisplaybreaks
						\begin{align*}
							&|x+1|-6{,}3=-\sqrt{1{,}69}\\
							&|x+1|-6{,}3=-1{,}3\\
							&|x+1|=-1{,}3+6{,}3\\
							&|x+1|=5.
						\end{align*}
						\begin{itemize}
							\item TH1: $x+1=5$, ta được $x = 4$.
							\item TH2: $x + 1 = -5$, ta được $x = -6$
						\end{itemize}
						Vậy $x = 4$ hoặc $x = -6$.
					\item Ta có
						\allowdisplaybreaks
						\begin{align*}
							&3^{x+3}-3^{x-1}=2160\\
							&3^x \cdot 3^3 - 3^x : 3 = 2160\\
							&3^x \cdot 3^3 - 3^x \cdot \dfrac{1}{3} = 2160\\
							&3^x \left(27 - \dfrac{1}{3}\right) = 2160\\
							&3^x \cdot \dfrac{80}{3} = 2160\\
							&3^x = 2160:\dfrac{80}{3}\\
							&3^x = 81 = 3^4\\
							&x = 4.
						\end{align*}
				\end{enumerate}
			}
		\end{ex}
	
	\begin{ex}%[Đề kiểm tra học kỳ I, THCS Giap Bat, 2015 - 2016]%[Lê Thị Oanh, dự án 7EX-1-2018]%[7D1B8]
		Số kg giấy vụn của ba lớp 7A, 7B, 7C thu được lần lượt tỉ lệ với các số $7; 3; 5$. Tính số kg giấy vụn thu được của mỗi lớp, biết rằng số kg giấy vụn mà 2 lớp 7A và 7C thu được nhiều hơn lớp 7B là $180$ kg.
		\loigiai
		{
			Gọi số kg giấy vụn của 3 lớp 7A, 7B, 7C thu được lần lượt là  $x, y, z~(x, y, z >0)$; đơn vị là kg.\\
			Ta có tỉ lệ thức
			\begin{align}
				&\dfrac{x}{7}=\dfrac{y}{3}=\dfrac{z}{5} \tag{1}\\
				\text{và } & x-y+z=180. \tag{2}
			\end{align}
			Từ (1) và (2), áp dụng tính chất của dãy tỉ số bằng nhau ta có
			\[\dfrac{x}{7}=\dfrac{y}{3}=\dfrac{z}{5}=\dfrac{x-y+z}{7-3+5}=\dfrac{180}{9}=20.\]
			Ta tìm được $x = 140$, $y = 60$ và $z = 100$ (đều thỏa mãn).\\
			Vậy số kg giấy vụn của 3 lớp 7A, 7B, 7C lần lượt là $140$ kg, $60$ kg, $100$ kg.
		}
	\end{ex}

		\begin{ex}%[Đề kiểm tra học kỳ I, THCS Giap Bat, 2015 - 2016]%[Lê Thị Oanh, dự án 7EX-1-2018]%[7H2B4]%[7H2K4]%[7H2G4]
			Cho $\triangle ABC$. Gọi $I$ là trung điểm đoạn thẳng $AC$. Trên tia đối của tia $IB$ lấy điểm $N$ sao cho $IB=IN.$
			\begin{enumerate}
			\item  Chứng minh $\triangle IBC = \triangle INA$.
			\item  Chứng minh $AN \parallel BC$.
			\item  Gọi $K$ là trung điểm của đoạn thẳng $AB$. Trên tia $CK$ lấy điểm $M$ sao cho $KM=KC$. Chứng minh $3$ điểm $M, A, N$ thẳng hàng.
		\end{enumerate}
			\loigiai
			{
				\begin{center}
					\begin{tikzpicture}[line join = round, line cap = round,>=stealth,font=\footnotesize,scale=.9]
						\tkzDefPoints{0/0/B}
						\coordinate (C) at ($(B)+(6,0)$);
						\tkzDefShiftPoint[B](70:4){A}
						\coordinate (I) at ($(A)!.5!(C)$);
						\tkzDefPointBy[symmetry = center I](B)    \tkzGetPoint{N}
						\tkzMarkSegments[mark=|](I,A I,C)
						\tkzMarkSegments[mark=||](I,N I,B)
						\coordinate (K) at ($(A)!.5!(B)$);
						\tkzDefPointBy[symmetry = center K](C)    \tkzGetPoint{M}
						\tkzMarkSegments[mark=x](K,A K,B)
						\tkzMarkSegments[mark=|||](K,C K,M)
						%
						\pgfresetboundingbox
						\tkzDrawSegments(A,B B,C C,A B,N N,M C,M)
						\tkzDrawPoints[fill=black](A,B,C,I,N,K,M)
						\tkzLabelPoints[above](A,I,N,M)
						\tkzLabelPoints[left](B)
						\tkzLabelPoints[right](C)
						\tkzLabelPoints[above right](K)
					\end{tikzpicture}
				\end{center}
				\begin{enumerate}
					\item Xét $\triangle AIN$ và $\triangle CIB$  có
						$\heva{& IA = IC \text{ (vì } I \text{ là trung điểm } AC\text{)} \\ & \widehat{AIN} = \widehat{BIC} \text{ (đối đỉnh)} \\ & BI = IN \text{ (giả thiết)}}$\\
						$\Rightarrow \triangle AIN = \triangle CIB$ (c.g.c).
					\item Theo câu a) ta có $\triangle AIN = \triangle CIB \Rightarrow \widehat{IAN}=\widehat{ICB}$ (hai góc tương ứng).\\
						Mà $\widehat{IAN}=\widehat{ICB}$ (so le trong) nên  suy ra $AN \parallel BC$ (đpcm).
					\item  Xét $\triangle KAM$ và $\triangle KBC $ có
						$\heva{& KA=KB \text{ (vì }K \text{ là trung điểm }AB \text{)} \\ & \widehat{AKM} = \widehat{BKC} \text{ (đối đỉnh)} \\ &KC = KM \text{ (giả thiết)}}$\\
						$\Rightarrow \triangle KAM = \triangle KBC$ (c.g.c) $\Rightarrow \widehat{MAK}=\widehat{CBK}$ (hai góc tương ứng).\\
						Mà $\widehat{MAK}=\widehat{CBK}$ (so le trong) nên suy ra  $AM \parallel BC$.\\
						Mặt khác theo câu b) có $AN \parallel BC$.\\
						Vậy ba điểm $M, A, N$ thẳng hàng.
				\end{enumerate}
			}
		\end{ex}

		\begin{ex}%[Đề kiểm tra học kỳ I, THCS Giap Bat, 2015 - 2016]%[Lê Thị Oanh, dự án 7EX-1-2018]%[7D1G6]
			Tìm giá trị lớn nhất của biểu thức
										\[A=-x^2+6x-\dfrac{37}{4}.\]
		\loigiai
		{
			Ta có
			\allowdisplaybreaks
			\begin{align*}
				A&= -x^2+6x-\dfrac{37}{4}\\
				&= -(x^2-3x-3x+9)-\dfrac{1}{4}\\
				&= -\left[ x(x-3)-3(x-3) \right] -\dfrac{1}{4}\\
				&=-(x-3)(x-3) -\dfrac{1}{4}\\
				&=-(x-3)^2 -\dfrac{1}{4} \le -\dfrac{1}{4}, \text{ với mọi }x.
			\end{align*}
			Dấu ``$=$'' xảy ra khi $x-3=0$ hay $x = 3$.
			Vậy giá trị lớn nhất của $A$ bằng $-\dfrac{1}{4}$ khi $x=3$.
		}
		\end{ex}
