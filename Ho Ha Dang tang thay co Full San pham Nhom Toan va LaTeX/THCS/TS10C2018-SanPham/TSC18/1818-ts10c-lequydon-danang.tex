    Cho biểu thức $A=\dfrac{2(x-1)}{\sqrt{x}-1}+\dfrac{x^2-\sqrt{x}}{x+\sqrt{x}+1}-\dfrac{2x+\sqrt{x}}{\sqrt{x}}$, với $x>0$; $x\ne 1$.\\ Chứng minh rằng $A\ge \dfrac{3}{4}$.
	\loigiai{
    	\begin{eqnarray*}
	    	A &= &\dfrac{2(x-1)}{\sqrt{x}-1}+\dfrac{x^2-\sqrt{x}}{x+\sqrt{x}+1}-\dfrac{2x+\sqrt{x}}{\sqrt{x}}\\
	    	&= &2(\sqrt{x}+1)+\sqrt{x}(\sqrt{x}-1)-(2\sqrt{x}+1)\\
	    	&= &x-\sqrt{x}+1\\
	    	&= &\left(\sqrt{x}-\dfrac{1}{2}\right)^2+\dfrac{3}{4} \ge \dfrac{3}{4}, \quad \forall x>0, x\ne 1.
    	\end{eqnarray*}
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, Đà Nẵng, năm 2018]%[Phạm An Bình, Dự án EX-TS10C-18]%[9D0G5]
	Tìm tất cả các cặp số nguyên $(x;y)$ thỏa mãn hệ thức $y=\dfrac{1}{x+1}+\dfrac{8}{x-4}$.
	\loigiai{
		$y=\dfrac{1}{x+1}+\dfrac{8}{x-4}=\dfrac{9x+4}{(x+1)(x-4)}$.\\
		Ta thấy $x+1$ và $x-4$ là hai số cách nhau $5$ đơn vị, suy ra $(x+1)(x-4)$ là tích của một số chẵn và một số lẻ. Suy ra $(x-4)(x+1)$ là số chẵn.\\
		Như vậy để $y\in \mathbb{Z}$ thì $\heva{&\hoac{&9x+4=0\text{ (loại vì \begin{math} x\in \mathbb{Z} \end{math})} \\&\left| 9x+4 \right| \ge \left| (x+1)(x-4) \right|} \\&9x+4 \text{ là số chẵn.}}$\\
		$9x+4$ là số chẵn khi $x=2k$, $k\in \mathbb{Z}$.
		\begin{eqnarray*}
			\left| 9x+4 \right| \ge \left| (x+1)(x-4) \right| &\Leftrightarrow &\left( x^2-3x-4 \right)^2 - (9x+4)^2 \le 0\\
			&\Leftrightarrow &(x^2-12x-8)(x^2+6x) \le 0\\
			&\Leftrightarrow &\heva{&x^2-12x-8 \le 0 \\&x^2+6x \ge 0} \text{ hoặc } \heva{&x^2-12x-8 \ge 0 \\&x^2+6x \le 0}\\
			&\Leftrightarrow &\hoac{&0 \le x \le 6+2\sqrt{11} \\&-6 \le x \le 6-2\sqrt{11}.}
		\end{eqnarray*}
		Mà $x\in \mathbb{Z}$ nên $0\le x\le 12$ hoặc $-6 \le x \le -1$.\\
		Mặt khác $x=2k$ nên $x\in \lbrace -6; -4; -2; 0; 2; 4; 6; 8; 10; 12 \rbrace$.\\
		Thử lại ta thấy $x=-6$ và $x=0$ thỏa.\\
		Với $x=-6$ thì $y=-1$.\\
		Với $x=0$ thì $y=-1$.\\
		Vậy các cặp số nguyên cần tìm là $(-6;-1)$; $(0;-1)$.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, Đà Nẵng, năm 2018]%[Phạm An Bình, Dự án EX-TS10C-18]%[9D0G2]
	Chứng minh rằng phương trình $(ax^2+2bx+c)(bx^2+2cx+a)(cx^2+2ax+b)=0$ luôn có nghiệm với mọi số thực $a$, $b$, $c$.
	\loigiai{
		$(ax^2+2bx+c)(bx^2+2cx+a)(cx^2+2ax+b)=0\Leftrightarrow \hoac{&ax^2+2bx+c=0 &(1) \\&bx^2+2cx+a=0 &(2) \\&cx^2+2ax+b=0 &(3).}$\\
		Xét các biệt thức $\Delta'$ của ba phương trình trên $\heva{&\Delta_1'=b^2-ca \\&\Delta_2'=c^2-ab \\&\Delta_3'=a^2-bc}$.\\
		Ta chứng minh bằng phương pháp phản chứng. Giả sử phương trình đề bài không có nghiệm với mọi số thực $a$, $b$, $c$, nghĩa là\\
		$\heva{&\Delta_1'=b^2-ca<0 \\&\Delta_2'=c^2-ab<0 \\&\Delta_3'=a^2-bc<0} \Rightarrow a^2+b^2+c^2-ab-bc-ca<0 \Rightarrow (a-b)^2+(b-c)^2+(c-a)^2<0$. (vô lý)\\
		Vậy ta có điều phải chứng minh.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, Đà Nẵng, năm 2018]%[Phạm An Bình, Dự án EX-TS10C-18]%[9H0G7]
	Trên cùng một hệ trục tọa độ $Oxy$, cho parabol $(P) \colon y=x^2$ và đường thẳng $(d) \colon y=mx+2m$, với $m$ là tham số. Gọi $A$ và $B$ lần lượt là giao điểm của $(d)$ với trục hoành và trục tung. Tìm tất cả các giá trị của $m$ để $(d)$ cắt $(P)$ tại hai điểm $C$ và $D$ nằm về hai phía trục tung sao cho điểm $C$ có hoành độ âm và $BD=2AC$.
	\loigiai{
		Tọa độ của $A(-2;0)$ và $B(0;2m)$.\\
		Phương trình hoành độ giao điểm của $(d)$ và $(P)$ là $x^2-mx-2m=0$.\\
		$(d)$ cắt $(P)$ tại hai điểm phân biệt $C(c;mc+2m)$ và $D(d;md+2m)$ khi $\Delta>0\Leftrightarrow \hoac{&m>0\\&m<-8.}$\\
		Theo đề bài, có $c<0$; $d>0$ và $c+d=m$; $cd=-2m$ (theo Vi-ét).
		\begin{eqnarray*}
			BD=2AC &\Leftrightarrow &BD^2=4AC^2\\
			&\Leftrightarrow &d^2+(md+2m-2m)^2=4[(c+2)^2+(mc+2m)^2]\\
			&\Leftrightarrow &(m^2+1)d^2=4(m^2+1)(c+2)^2\\
			&\Leftrightarrow &d^2=4(c+2)^2\\
			&\Leftrightarrow &\hoac{&d=2c+4 \\&d=-2c-4.}
		\end{eqnarray*}
		\begin{itemize}
			\item Với $d=2c+4$ và $c+d=m$ thì $c=\dfrac{m-4}{3}$, $d=\dfrac{2m+4}{3}$.\\
			$cd=-2m\Leftrightarrow \dfrac{m-4}{3}\cdot \dfrac{2m+4}{3}=-2m\Leftrightarrow 2m^2+14m-16=0\Leftrightarrow \hoac{&m=1\\&m=-8 &\text{(loại).}}$\\
			Thử lại thấy $m=1$ thỏa.
			\item Với $d=-2c-4$ và $c+d=m$ thì $c=-m-4$, $d=2m+4$.\\
			$cd=-2m\Leftrightarrow (-m-4)(2m+4)=-2m\Leftrightarrow 2m^2+10m+16=0$ (vô nghiệm).
		\end{itemize}
		Vậy $m=1$ thỏa đề bài.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, Đà Nẵng, năm 2018]%[Phạm An Bình, Dự án EX-TS10C-18]%[9D0G2]
	Giải phương trình $5x(x+1)=3\left(x\sqrt{2x^2+1}+4\right)$.
	\loigiai{
		\begin{eqnarray*}
			5x(x+1)=3\left(x\sqrt{2x^2+1}+4\right) &\Leftrightarrow &5x^2+5x-12-3x\sqrt{2x^2+1}=0\\
			&\Leftrightarrow &10x^2+10x-24-6x\sqrt{2x^2+1}=0\\
			&\Leftrightarrow &(\sqrt{2x^2+1}-3x)^2=(x-5)^2\\
			&\Leftrightarrow &\hoac{&\sqrt{2x^2+1}=4x-5 &(1) \\&\sqrt{2x^2+1}=2x+5 &(2).}
		\end{eqnarray*}
		$(1) \Leftrightarrow \heva{&2x^2+1=16x^2-40x+25 \\&4x-5\ge 0} \Leftrightarrow \heva{&\hoac{&x=2 \\&x=\dfrac{6}{7}} \\&x\ge \dfrac{5}{4}} \Leftrightarrow x=2$.\\
		$(2) \Leftrightarrow \heva{&2x^2+1=4x^2+20x+25 \\&2x+5\ge 0} \Leftrightarrow \heva{&\hoac{&x=-5-\sqrt{13} \\&x=-5+\sqrt{13}} \\&x\ge -\dfrac{5}{2}} \Leftrightarrow x=-5+\sqrt{13}$.\\
		Vậy nghiệm của phương trình là $x=2$; $x=-5+\sqrt{13}$.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, Đà Nẵng, năm 2018]%[Phạm An Bình, Dự án EX-TS10C-18]%[9D0G2]
	Giải hệ phương trình $\heva{&4\sqrt{x+2}+2\sqrt{3(x+4)}=3y(y-1)+10\\ &(x+2)^3+x=y(y^2+1)-2.}$
	\loigiai{
		$\heva{&4\sqrt{x+2}+2\sqrt{3(x+4)}=3y(y-1)+10 &(1)\\ &(x+2)^3+x=y(y^2+1)-2 &(2).}$\\
		Điều kiện $x\ge -2$.
		\begin{eqnarray*}
			(2) &\Leftrightarrow &(x+2)^3-y^3+x+2-y=0\\
			&\Leftrightarrow &(x+2-y)\left[(x+2)^2+(x+2)y+y^2\right]+x+2-y=0\\
			&\Leftrightarrow &(x+2-y)\left[(x+2+\dfrac{1}{2}y)^2+\dfrac{3}{4}y^2+1\right]=0\\
			&\Leftrightarrow &x+2-y=0\\
			&\Leftrightarrow &x+2=y.
		\end{eqnarray*}
		Thay $x+2=y$ vào $(1)$ ta được $4\sqrt{y}+2\sqrt{3(y+2)}=3y^2-3y+10\quad (3)$.\\
		Áp dụng bất đẳng thức Cauchy ta có $\heva{&4\sqrt{y}\le 2(y+1) \\&2\sqrt{3(y+2)}\le y+5} \Rightarrow \text{VT} \le 3y+7$.\\
		Mặt khác $3(y-1)^2 \ge 0 \Leftrightarrow 3y^2-3y+10 \ge 3y+7$.\\
		Vậy $(3)$ xảy ra khi và chỉ khi $y=1$, suy ra $x=-1$.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, Đà Nẵng, năm 2018]%[Phạm An Bình, Dự án EX-TS10C-18]%[9H0G2][9H0G1]
	Cho tam giác $ABC$ nội tiếp đường tròn tâm $O$ có $\widehat{BAC}$ tù, $AB<AC$ và $H$ là trực tâm. Các đường thẳng $AH$, $BH$, $CH$ lần lượt cắt các đường thẳng $BC$, $CA$, $AB$ tại các điểm $D$, $E$, $F$.

			\tkzDrawSegments(F,B B,C C,E B,H C,H D,H A,O E,F)
			\tkzMarkRightAngles(O,A,t)
			\end{tikzpicture}
		}
		\item Chứng minh rằng $EF$ là đường trung trực của đoạn $AK$.\\
		\textbf{Bài toán phụ.} Cho điểm $M$ không thuộc đường tròn $(O)$. Qua $M$ kẻ hai đường thẳng lần lượt cắt $(O)$ tại $A$, $B$ và $C$, $D$. Chứng minh $MA\cdot MB=MC\cdot MD$ (đẳng thức này được gọi là phương tích của điểm $M$ đối với $(O)$).

			TH 1. $M$ nằm bên ngoài $(O)$.\\

			$\Rightarrow \triangle MAC \backsim \triangle MDB$ (g-g).\\

		Có $\widehat{NBF}=\widehat{NMF}=90^\circ$, suy ra $BMFN$ nội tiếp đường tròn đường kính $FN$ (tứ giác có hai đỉnh liên tiếp nhìn một cạnh dưới hai góc bằng nhau).\\
		$\Rightarrow AF\cdot AB=AN\cdot AM$ (phương tích của điểm $A$ đối với $(BMFN)$). (3)\\
		Từ (1), (2) và (3) ta có $AK\cdot AO=AN\cdot AM$.\\
		$\Rightarrow AK=\dfrac{AN\cdot AM}{AO}=2AM \Rightarrow MK=AM$.\\
		$\Rightarrow M$ là trung điểm của $AK$.\\
		Mà $EF\perp AK$ tại $M$ nên $EF$ là đường trung trực của đoạn $AK$.
		\end{listEX}
	}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, Đà Nẵng, năm 2018]%[Phạm An Bình, Dự án EX-TS10C-18]%[9H3G8]
	Cho hai đường tròn $(I;r)$ và $(J;R)$ tiếp xúc ngoài nhau tại $E$ (với $r<R$) và đường thẳng $d$ là tiếp tuyến chung tại $E$ của hai đường tròn đó. Trên $d$ lấy hai điểm $A$ và $C$ sao cho $E$ nằm giữa $A$, $C$ và $R<EA<EC$. Các tiếp tuyến thứ hai của đường tròn $(I)$ vẽ từ $A$ và $C$ cắt nhau tại $B$, các tiếp tuyến thứ hai của đường tròn $(J)$ vẽ từ $A$ và $C$ cắt nhau tại $D$. Chứng minh rằng tồn tại một điểm cách đều bốn đường thẳng $AB$, $BC$, $CD$, $DA$.
	\loigiai{
		\immini{
			Gọi $M$, $P$ lần lượt là tiếp điểm của $(I)$ với $AB$ và $BC$.\\
			Gọi $N$, $Q$ lần lượt là tiếp điểm của $(J)$ với $AD$ và $CD$.\\
			Lấy $H$ thuộc $BC$ sao cho $BH=BA$; $K$ thuộc $CD$ sao cho $DK=DA$.\\
			Do tính chất hai tiếp tuyến cắt nhau nên ta có\\ $\heva{&AM=AE=AN\\&BM=BP\\&CP=CE=CQ\\&DQ=DN.}$
		}{
			\begin{tikzpicture}[scale=0.5]
			\tkzDefPoints{3/0/J, -2/0/I, 0/0/E, 0/4/A, 0/-6/C}
			\tkzDrawCircle(J,E) \tkzDrawCircle(I,E)
			\tkzTangent[from=A](I,E)  \tkzGetFirstPoint{M}
			\tkzTangent[from=A](J,E)  \tkzGetSecondPoint{N}
			\tkzTangent[from=C](I,E)  \tkzGetSecondPoint{P}
			\tkzTangent[from=C](J,E)  \tkzGetFirstPoint{Q}
			\tkzFindAngle (A,B,C)  \tkzGetAngle{ang}
			\tkzDefPointBy[rotation=center B angle \ang](A)\tkzGetPoint{H}
			\tkzFindAngle (A,D,C)  \tkzGetAngle{tang}
			\tkzDefPointBy[rotation=center D angle \tang](A)\tkzGetPoint{K}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, Đà Nẵng, năm 2018]%[Phạm An Bình, Dự án EX-TS10C-18]%[9D0G4]
	Cho $x$, $y$ là các số tự nhiên thỏa mãn $3y^2+1=4x^2$. Chứng minh rằng $x$ là tổng bình phương của hai số tự nhiên liên tiếp.
	\loigiai{
		Ta có $$3y^2=(2x-1)(2x+1).$$
		Do $2x-1$ và $2x+1$ nguyên tố cùng nhau nên $2x-1=3m^2$, $2x+1=n^2$ hoặc $2x-1=m^2$, $2x+1=3n^2$ với $m$, $n\in \mathbb{N}$.\\
		Trường hợp thứ nhất loại do $n^2=3m^2+2\equiv 2\pmod 3$.\\
		Trường hợp thứ hai, do $m$ lẻ nên ta đặt $m=2k+1$ với $k\in \mathbb{N}$. Suy ra $2x-1=(2k+1)^2$. Vậy ta có $x=k^2+(k+1)^2$ (đpcm).
	}
\end{ex}
