\begin{ex}%[Đề thi vào 10, Sở giáo dục Bình Dương, 2017]%[9D1B6]%[9D1B2]
Rút gọn các biểu thức sau:
    \hfill
    \begin{enumerate}[1)]
        \begin{multicols}{2}
        \item $A=3\sqrt{3}+2\sqrt{12}-\sqrt{27}$;	
        \item $B=\sqrt{{{(3-\sqrt{5})}^{2}}}+\sqrt{6-2\sqrt{5}}$.
    	\end{multicols}
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}[1)]
        \item $A=3\sqrt{3}+2\sqrt{12}-\sqrt{27}=3\sqrt{3}+4\sqrt{3}-3\sqrt{3}=4\sqrt{3}$.
        \item $B=\sqrt{\left(3-\sqrt{5} \right)^2}+\sqrt{6-2\sqrt{5}}=\sqrt{\left(3-\sqrt{5} \right)^2}+\sqrt{(\sqrt{5}-1)^2}=3-\sqrt{5}+\sqrt{5}-1=2$.
    \end{enumerate}
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Bình Dương, 2017]%[9D3B2]%[9D4K6]
    \hfill
\begin{enumerate}[1)]
        \item Giải hệ phương trình: $\heva{&2x-y=5\\&x+5y=-3}$. Tính $P=(x+y)^{2017}$ với $x, y$ vừa tìm được.
        \item Cho phương trình $x^2-10mx+9m=0 \ (1)$ ($m$ là tham số).
\begin{enumerate}[a)]
\item Giải phương trình $(1)$ khi $m = 1$.

\item Tìm các giá trị của tham số $m$ để phương trình $(1)$ có hai nghiệm phân biệt $x_1, x_2$ thỏa điều kiện $x_1-9x_2=0$.
    \end{enumerate}
    \end{enumerate}
\loigiai{
\begin{enumerate}[1)]
    \item $\heva{&2x-y=5\\ &x+5y=-3}\Leftrightarrow \heva{&10x-5y=25\\&x+5y=-3}\Leftrightarrow \heva{&11x=22\\&x+5y=-3}\Leftrightarrow \heva{&x=2\\&2+5y=-3}\Leftrightarrow \heva{& x=2\\&y=-1}$

$P=(2-1 )^{2017}=1$.
    \item
    \begin{enumerate}[a)]
        \item Khi $m=1$ phương trình đã cho trở thành $x^2-10x+9=0$.\\
Ta có $a+b+c=1-10+9=0$ nên phương trình có $2$ nghiệm phân biệt $x_1=1,x_2=\dfrac{c}{a}=9$;
        \item Điều kiện $(1)$ có $2$ nghiệm phân biệt là $25m^2-9m>0$ \hfill (*)\\
Theo định lý Vi-ét, ta có:\\
$\heva{&x_1+x_2=10m\\
 &x_1-9x_2=0\\&x_1x_2=9m}\Leftrightarrow \heva{&10x_2=10m\\&x_1-9x_2=0\\&x_1x_2=9m}\Leftrightarrow \heva{&x_2=m\\&x_1=9m\\&9m^2-9m=0}\Leftrightarrow \heva{&x_2=m \\
 &x_1=9m\\&\hoac{& m=0\\&m=1}}$

 Với $m=0$ ta có $25m^2-9m=0$, suy ra $m=0$ không thỏa mãn $(*)$.\\
 Với $m=1$ ta có $25m^2-9m=25.1^2-9.1=16>0$, suy ra $m=1$ thỏa mãn $(*)$.\\
 Vậy $m=1$ là giá trị cần tìm.
    \end{enumerate}
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Bình Dương, 2017]%[9D3B5]
 Hai đội công nhân đắp đê ngăn triều cường. Nếu hai đội cùng làm thì trong $6$ ngày xong việc. Nếu làm riêng thì đội $\mathrm{I}$ hoàn thành công việc chậm hơn đội $\mathrm{II}$ là $9$ ngày. Hỏi nếu làm riêng thì mỗi đội đắp xong đê trong bao nhiêu ngày?
\loigiai{
Gọi $x$ (ngày) là thời gian làm một mình xong việc của đội $\mathrm{I}$ ($x > 9$), $y$ (ngày) là thời gian làm một mình xong việc của đội $\mathrm{II}$ ($y > 6$). Ta có phương trình $x-y=9$.\\
Trong $1$ ngày lượng công việc làm được của đội $\mathrm{I}$ là $\dfrac{1}{x}$, đội $\mathrm{II}$ là $\dfrac{1}{y}$. Ta có phương trình $\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{1}{6}$.\\
Giải hệ $\heva{& x-y=9\\& \dfrac{1}{x}+\dfrac{1}{y}=\dfrac{1}{6}}\Leftrightarrow \heva{& x=9+y\\&\dfrac{1}{9+y}+\dfrac{1}{y}=\dfrac{1}{6}}\Leftrightarrow \heva{& x=9+y\\
 &y^2-3y-54=0}\Leftrightarrow \heva{& x=9+y\\&\hoac{& y=9 \\
 &y=-6 \text{ (loại)}}}\Leftrightarrow \heva{&x=18\\&y=9}.$\\
Vậy thời gian làm một mình xong việc của đội $\mathrm{I}$ là $18$ (ngày), đội $\mathrm{II}$ là $9$ (ngày).

}
\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Bình Dương, 2017]%[9H2B2]%[9H3B3]%[9H3G4]
Tam giác $AMB$ cân tại $M$ nội tiếp trong đường tròn $(O; R)$. Kẻ $MH$ vuông góc $AB$ $(H\in AB), MH$ cắt đường tròn tại $N$. Biết $MA = 10cm$, $AB = 12cm$.
    \begin{enumerate}[1)]
        \item Tính $MH$ và bán kính $R$ của đường tròn.
        \item Trên tia đối tia $BA$ lấy điểm $C$. Tia $MC$ cắt đường tròn tại $D$, $ND$ cắt $AB$ tại $E$. Chứng minh tứ giác $MDEH$ nội tiếp và chứng minh các hệ thức sau: $NB^2=NE.ND$ và $AC.BE=BC.AE$.
        \item Chứng minh $NB$ tiếp xúc với đường tròn ngoại tiếp tam giác $BDE$.
    \end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
    \clip (-3,-3.5) rectangle (7,3.5);
    \tkzDefPoints{0/0/O, 0/3/M, 0/-2/H, 6/-2/C, 0/-3/N}
    \tkzDrawCircle[radius](O,M)
    \tkzInterLC(H,C)(O,M)
    \tkzGetPoints{B}{A}
    \tkzInterLC(M,C)(O,M)
    \tkzGetPoints{M}{D}
    \tkzInterLL(A,B)(N,D)
    \tkzGetPoint{E}
    \tkzDrawPoints(A,B,D,E)
    \tkzMarkRightAngle(M,H,B)
    \tkzMarkRightAngle(M,B,N)
    \tkzMarkRightAngle(M,D,N)
    \tkzCircumCenter(B,E,D)
    \tkzGetPoint{I}
    \tkzDrawCircle[circum](B,E,D)
    \tkzDrawPoints(O,M,H,C,N,I)
    \tkzDrawSegments[thick](A,C A,M B,M M,C M,N N,B N,A N,D B,D A,D E,I)
    \tkzLabelPoints[below left](A,H)
    \tkzLabelPoints[below](E,N,B,C)
    \tkzLabelPoints[above](M)
    \tkzLabelPoints[right](D,O)
     \tkzLabelPoints[right](I)
\end{tikzpicture}
\end{center}
\begin{enumerate}[1)]
    \item Theo tính chất đường kính và dây cung suy ra $H$ là trung điểm $AB$ và $AH = 6\text{ cm}.$\\
$\Delta AMH$ vuông tại $H \Rightarrow MH = \sqrt{AM^2-AH^2}=\sqrt{{10^2}-6^2}=8\text{ cm}.$\\
$\Delta AMN$ vuông tại $A$, đường cao $AH$, do đó$AH^2=HM.HN\Rightarrow HN=\dfrac{AH^2}{MH}=\dfrac{36}{8}=4,5\text{ cm}$.\\
Bán kính $R=\dfrac{MN}{2}=\dfrac{MH+HN}{2}=\dfrac{8+4,5}{2}=6,25\text{ cm}.$
    \item $\widehat{MDN}={90^{\circ}}$ (góc nội tiếp chắn nửa đường tròn), $\widehat{MHE}={90^{\circ}}$ ($MH\perp AB$). Từ đó suy ra $\widehat{MDE}+\widehat{MHE}={180^{\circ}}$, do đó tứ giác $MDEH$ nội tiếp.\\
Xét các tam giác $\Delta NBE$ và $\Delta NDB$ có góc $N$ chung, $\widehat{NBE}=\widehat{NDB}$ (cùng chắn hai cung bằng nhau là cung $NA, NB$).\\
Suy ra $\Delta NBE\backsim \Delta NDB$, do đó $\dfrac{NB}{ND}=\dfrac{NE}{NB}\Rightarrow NB^2=NE.ND$.\\
Ta có cung $NA$ bằng cung $NB$ (tính chất đường kính và dây cung), suy ra $\widehat{ADE}=\widehat{EDB} \Rightarrow DE$ là phân giác trong của $\Delta ABD$.\\
Vì $ED\perp DC \Rightarrow DC$ là phân giác ngoài của $\Delta ABD$.\\
Từ đó suy ra: $\dfrac{DA}{DB}=\dfrac{EA}{EB}=\dfrac{CA}{CB}\Rightarrow AC.BE=BC.AE$.
    \item Kẻ $EI \| AM \ (I\in BM) \Rightarrow \Delta AMB \backsim \Delta EIB \Rightarrow \Delta EIB$ cân tại $I$ $\Rightarrow IE = IB$.\\
Gọi $(O')$ là đường tròn tâm $I$ ngoại tiếp $\Delta EBD'$. Ta có $NB\perp BM$ (góc nội tiếp chắn nửa đường tròn tâm $O$), từ đó suy ra $BN\perp BI \Rightarrow BN$ là tiếp tuyến đường tròn $(O') \Rightarrow\widehat{EBN}=\widehat{ED'B}$ (cùng chắn cung $BE$).\\
Mặt khác trên đường tròn $(O)$, $\widehat{EBN}=\widehat{EDB}$ (cùng chắn hai cung bằng nhau $NA, NB$) $\Rightarrow D$ nằm trên đường tròn $(O')$.\\
Vậy $NB$ tiếp xúc với đường tròn ngoại tiếp tam giác $BDE$.
\end{enumerate}
}
\end{ex}
