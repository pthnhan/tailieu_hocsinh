\begin{bt}%[Đề thi HK1, Hoàng Hoa Thám, HCM, 2018]%[Võ Thị Quỳnh Trang, dự án 11-EX-HCM]%[1D1Y3-3]
	Giải phương trình $\cos 2x+\sin 2x=\dfrac{\sqrt{2}}{2}.$
	\loigiai{
	Chia cả hai vế của phương trình cho $\sqrt{2}$ ta có:
	\begin{align*}
		\dfrac{1}{\sqrt{2}} \cos 2x +\dfrac{1}{\sqrt{2}}\sin 2x = \dfrac{1}{2} & \Leftrightarrow \cos \dfrac{\pi}{4} \cos 2x+ \sin \dfrac{\pi}{4} \sin 2x=\dfrac{1}{2} \Leftrightarrow \cos \left( 2x-\dfrac{\pi}{4} \right) =\dfrac{1}{2} \\
		& \Leftrightarrow \hoac{& 2x-\dfrac{\pi}{4}=\dfrac{\pi}{3}+k2\pi \\ & 2x-\dfrac{\pi}{4}=-\dfrac{\pi}{3}+k2\pi } (k \in \mathbb{Z})\\
		& \Leftrightarrow \hoac{& 2x=\dfrac{7\pi}{12}+k2\pi \\ & 2x=-\dfrac{\pi}{12}+ k2\pi} (k \in \mathbb{Z})\\
		& \Leftrightarrow \hoac{& x=\dfrac{7\pi}{24}+k\pi \\ & x=-\dfrac{\pi}{24}+ k\pi.} (k \in \mathbb{Z})
	\end{align*}
	Vậy tập nghiệm của phương trình là $S=\left\lbrace \dfrac{7\pi}{24}+k\pi , -\dfrac{\pi}{24}+k\pi |k\in \mathbb{Z} \right\rbrace. $
}
\end{bt}

\begin{bt}%[Đề thi HK1, Hoàng Hoa Thám, HCM, 2018]%[Võ Thị Quỳnh Trang, dự án 11-EX-HCM]%[1D2B2-1]
	Từ các chữ số $0$, $1$, $2$, $3$, $4$, $5$ lập được bao nhiêu số tự nhiên chẵn có $4$ chữ số khác nhau?
	\loigiai{Gọi số tự nhiên có $4$ chữ số khác nhau lập được từ các chữ số $0$, $1$, $2$, $3$, $4$, $5$ là $\overline{abcd}.$\\
	Ta xét các trường hợp sau:\\
	TH1: Chữ số $0$ đứng ở cuối, tức là $d=0$. Khi đó số cách chọn $\overline{abc}$ là $\mathrm{A}_5^3.$\\
	TH2: Chữ số $2$ hoặc $4$ đứng ở cuối, tức là $d=2$ hoặc $d=4$. Khi đó $a$ có $4$ cách chọn (do $a \neq 0$). Với $a$ đã chọn, số cách chọn $\overline{bc}$ là $\mathrm{A}_4^2$.\\
	Vậy số các số thỏa mãn đầu bài là $\mathrm{A}_5^3+ 2 \cdot 4 \cdot \mathrm{A}_4^2=156.$
}
\end{bt}

\begin{bt}%[Đề thi HK1, Hoàng Hoa Thám, HCM, 2018]%[Võ Thị Quỳnh Trang, dự án 11-EX-HCM]%[1D2K3-2]
	Tìm số hạng không chứa $x$ trong khai triển $\left( \dfrac{x}{2} -\dfrac{4}{x} \right) ^{18}. $
	\loigiai{
	Theo công thức khai triển nhị thức Newton ta có
	$$ \left( \dfrac{x}{2} -\dfrac{4}{x} \right) ^{18} = \sum_{k=0}^{18} \mathrm{C}_{18}^k  \left( \dfrac{x}{2} \right)^{18-k} (-1)^k \left( \dfrac{4}{x} \right)^k = \sum_{k=0}^{18} (-1)^k \mathrm{C}_{18}^k \dfrac{x^{18-k}}{2^{18-k}}\cdot \dfrac{4^k}{x^k}= \sum_{k=0}^{18} (-1)^k \mathrm{C}_{18}^k 2^{3k-18} x^{18-2k}.$$
	Số hạng không chứa $x$ tương đương với số hạng chứa $k$ thỏa mãn $18-2k=0 \Leftrightarrow k=9$.\\
	Vậy số hạng không chứa $x$ là $-\mathrm{C}_{18}^9 2^9.$
}
\end{bt}

\begin{bt}%[Đề thi HK1, Hoàng Hoa Thám, HCM, 2018]%[Võ Thị Quỳnh Trang, dự án 11-EX-HCM]%[1D2B5-2]
	Trong phòng máy vi tính có $20$ máy tính, trong đó có $9$ máy bị hỏng. Chọn ngẫu nhiên $3$ máy, tính xác suất để trong $3$ máy được chọn có ít nhất một máy chạy tốt.
	\loigiai{
	Số cách chọn ngẫu nhiên $3$ máy tính từ $20$ máy tính trong phòng máy là $\mathrm{C}_{20}^3=1140.$\\
	Số cách chọn ngẫu nhiên sao cho cả $3$ máy đều hỏng là $\mathrm{C}_9^3=84$. \\
	Vậy số cách chọn ra $3$ máy sao cho có ít nhất một máy chạy tốt là $1140 - 84=1056.$\\
	Xác suất cần tính là $\dfrac{1056}{1140}=\dfrac{88}{95}.$
}
\end{bt}

\begin{bt}%[Đề thi HK1, Hoàng Hoa Thám, HCM, 2018]%[Võ Thị Quỳnh Trang, dự án 11-EX-HCM]%[1D3B2-3]
	Xét tính tăng giảm của dãy số $(u_n)$ biết $u_n=\dfrac{5n-3}{2n+1} ~ \forall n \ge 1.$
	\loigiai{
	Dựa vào công thức xác định của dãy số $(u_n)$, ta có $$u_{n-1}=\dfrac{5(n-1)-3}{2(n-1)+1}=\dfrac{5n-8}{2n-1} \text{ với } n \geq 2.$$
	Khi đó $$u_n- u_{n-1}=\dfrac{5n-3}{2n+1} - \dfrac{5n-8}{2n-1}= \dfrac{(5n-3)(2n-1)-(5n-8)(2n+1)}{(2n+1)(2n-1)}=\dfrac{11}{4n^2-1} \ge 0 ~ \forall n \ge 2.$$
	Vậy $u_n > u_{n-1} ~ \forall n \ge 2$. Do đó $(u_n)$ là dãy tăng.	
}
\end{bt}

\begin{bt}%[Đề thi HK1, Hoàng Hoa Thám, HCM, 2018]%[Võ Thị Quỳnh Trang, dự án 11-EX-HCM]%[1D3K3-2]
	Tìm số hạng tổng quát của cấp số cộng $(u_n)$ biết $\heva{&u_1+u_2+u_3=27\\ &u_1^2+u_2^2+u_3^2=275.}$
		\loigiai{
		Gọi công sai của cấp số cộng $(u_n)$ là $d$.\\
		Từ giả thiết ta có: 
		
	\begin{align*}
		\heva{&u_1+u_1+d+u_1+2d=27 \\ & u_1+ (u_1+d)^2+(u_1+2d)^2=275} & \Leftrightarrow \heva{&3u_1+3d=27 \\ &3u_1^2+6u_1d+5d^2=275}\\ & \Leftrightarrow \heva{&u_1=9-d \\ &3(9-d)^2+6(9-d)d+5d^2=275} \\ &\Leftrightarrow \heva{&u_1=9-d \\ & 2d^2=32 } \Leftrightarrow \heva{&u_1=9-d \\ &\hoac{&d=4\\ &d=-4.}}  
	\end{align*}
	Ta xét 2 trường hợp sau\\
	TH1: $d=4 \Rightarrow u_1 = 5$. Khi đó công thức tổng quát của cấp số cộng $(u_n)$ là $$u_n=u_1+(n-1)d=5+(n-1)4=4n+1 ~ \forall n \ge 1. $$
	TH1: $d=-4 \Rightarrow u_1 = 13$. Khi đó công thức tổng quát của cấp số cộng $(u_n)$ là
	\begin{center}
		$u_n=u_1+(n-1)d=13+(n-1)(-4)=-4n+17 ~ \forall n \ge 1. $
	\end{center}
}
\end{bt}

\begin{bt}%[Đề thi HK1, Hoàng Hoa Thám, HCM, 2018]%[Võ Thị Quỳnh Trang, dự án 11-EX-HCM]%[1D3B3-6]
	Bạn An làm việc tại công ty A với mức lương khởi điểm là $5$ triệu đồng một tháng. Cứ sau $12$ tháng làm việc thì lương lại tăng thêm $1$ triệu đồng. Tính tổng số tiền An nhận được khi làm đủ $30$ năm tại công ty A.
	\loigiai{
	Với mỗi số nguyên dương $n$, kí hiệu $u_n$ (triệu đồng) là mức lương hàng tháng của bạn An ở năm thứ $n$ tại công ty. Theo giả thiết của bài toán, ta có
	$$u_1=5 \text{ và } u_{n+1}=u_n+1 \text{ với mọi } n \ge 1.$$
	Do đó, dãy số $(u_n)$ là một cấp số cộng với công sai $d=1$.\\
	Vì một năm có $12$ tháng nên tổng số tiền bạn An nhận được sau $30$ năm là
	$$ S=12(u_1+u_2+u_3+ \cdots + u_{30})=12\cdot \dfrac{(u_1+u_{30})30}{2}=180 (2u_1+29 \cdot d)=6120. $$ 
	Vậy tổng số tiền An nhận được khi làm đủ $30$ năm ở công ty A là $6120$ triệu đồng.
}
\end{bt}

\begin{bt}%[Đề thi HK1, Hoàng Hoa Thám, HCM, 2018]%[Võ Thị Quỳnh Trang, dự án 11-EX-HCM]%[1H2B4-2]%[1H2B3-2]
	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình thang, đáy lớn là $AD$ và $AD=2BC$. Gọi $M$, $N$, $I$, $K$ lần lượt là trung điểm của $SA$, $AB$, $SD$, $AD$; $E$ là giao điểm của $BD$ và $CK$.
