
\section{Phương pháp đặt ẩn phụ}

\begin{dang}{Đặt ẩn phụ hoàn toàn}
\noindent Ta cần giải phương trình $ F(x)=0 $.
	
		\begin{description}
			\item[Bước 1] Đặt $ t=\sqrt{f(x)} $ hoặc $ t=\sqrt{f(x)+k} $ hoặc $ t=\sqrt{f(x)}\pm \sqrt{g(x)} $.
			\item[Bước 2] Đưa phương trình ban đầu về dạng $ G(t)=0 $ẩn $ t $ đã biết cách giải.
			\item[Bước 3] Với mỗi nghiệm $ t _0$ của phương trình $ G(t)=0 $,  giải phương trình $ t_0=\sqrt{f(x)} $ hoặc $ t_0=\sqrt{f(x)+k} $ hoặc $ t_0=\sqrt{f(x)}\pm \sqrt{g(x)} $. Sau đó đối  chiếu với điều kiện của phương trình ban đầu để kết luận nghiệm.
		\end{description}
	
\end{dang}

\begin{note}
	Phương trình ẩn phụ tập nghiệm của nó sẽ chứa tập nghiệm của phương trình ban đầu. Vì thế, nếu phương trình ẩn phụ vô nghiệm thì phương trình ban đầu cũng vô nghiệm.
\end{note}
\begin{bt}%[]%[Hoàng Tựu]
Giải phương trình $ x^2+\sqrt{x^2+1}-5=0. $
\loigiai{
Đặt $ t= \sqrt{x^2+1}$ với $ t\ge 1. $ \\Phương trình trở thành:
$ t^2-1+t-5=0\Leftrightarrow t^2+t-6=0\Leftrightarrow \hoac{t&=2\Rightarrow \pm \sqrt{2^2-1}=\pm \sqrt{3}\\t&=-3 (\text{loại})}. $	
\\
Vậy phương trình có nghiệm $ x=\pm \sqrt{3} .$
}
\end{bt}
%
\begin{bt}%[Hoàng Tựu]
	Giải phương trình $ 3\sqrt{x+2}-6\sqrt{2-x}+4\sqrt{4-x^2}=10-3x. $
	\loigiai{
		Điều kiện: $ -2\le x\le 2 $.\\
	Pt tương đương $  3(\sqrt{x+2}-2\sqrt{2-x})+4\sqrt{4-x^2}=10-3x$.	\\
	Đặt $ t=\sqrt{x+2}-2\sqrt{2-x}\Rightarrow t^2=10-3x-4\sqrt{4-x^2} $.\\
	Phương trình trở thành:\\
	$ 3t-t^2=0\Leftrightarrow \hoac{t&=0\\t&=3} $.\\
	Với $ t=0\Leftrightarrow  \sqrt{x+2}=2\sqrt{2-x}\Leftrightarrow x+2=4(2-x)\Leftrightarrow  x=\dfrac{6}{5}$.\\
	Với $ t=3 \Leftrightarrow  \sqrt{x+2}=3+2\sqrt{2-x}\Leftrightarrow x+2=9+12\sqrt{2-x}+8-4x\Leftrightarrow 5x-15=12\sqrt{2-x}$.\\
	Nhận xét rằng $ x\le 2 $ thì $ 5x-15\le 0 $ và phương trình không có nghiệm bằng $ 2 $ nên $ \sqrt{2-x}<0 $. 
	Vậy phương trình có nghiệm $ x=\dfrac{6}{5} $ .
	}
\end{bt}

\begin{bt}%[Hoàng Tựu]
	Giải phương trình $ x^2-2x+4+\sqrt{x(x^2+4)}=0. $
	\loigiai{
		Phương trình tương đương $ x^2+4-2x+\sqrt{x(x^2+4)}=0. $.\\
		Dễ thấy $ x=0 $ không phải nghiệm của phương trình nên ta chia hai vế phương trình cho $ x $, ta thu được.\\
		$ \dfrac{x^2+4}{x}-2+\sqrt{\dfrac{x^2+4}{x}}=0. $\\
		Đặt $ t=\sqrt{\dfrac{x^2+4}{x}},\; t\ge 0 $.\\
		Phương trình trở thành $ t^2+t-2=0\Leftrightarrow \hoac{t&=1\\t&=-2\; (L)} $.\\
		Với $ t=1\Rightarrow  \sqrt{\dfrac{x^2+4}{x}}=1\Leftrightarrow \dfrac{x^2+4}{x}=1\Leftrightarrow x^2-x+4=0\;(\text{phương trình vô nghiệm})$.\\
		Vậy phương trình vô nghiệm.
	}
\end{bt}

\begin{dang}{Đặt ẩn phụ không hoàn toàn}
	Ta cần giải phương trình $ F(x)=0 $.
	
		\begin{description}
			\item[Bước 1] Đặt $ t=f(x)$.
			\item[Bước 2] Đưa phương trình ban đầu về dạng $ G(t,x)=0 $ ẩn $ t $ đã biết cách giải.
			\item[Bước 3] Với mỗi nghiệm $ t _0$ của phương trình $ G(t)=0 $,  giải phương trình $ t_0=f(x)$. Sau đó đối  chiếu với điều kiện của phương trình ban đầu để kết luận nghiệm.
		\end{description}

\end{dang}

\begin{note}
	Phương trình ẩn phụ có thể coi là ẩn $ x $ hoặc ẩn $ t $.
\end{note}
\begin{bt}%[Hoàng Tựu]
Giải phương trình $ \sqrt{2x-1}+x^2-3x+1=0. $
\loigiai{
	ĐK: $ x\ge \dfrac{1}{2} $.\\
Đặt $t=\sqrt{2x-1}$\\
Phương trình đã cho tương đương
$
x^2-(2x-1)-x+\sqrt{2x-1}=0\\
\Leftrightarrow x^2-t^2+t-x=0
\Leftrightarrow \left( x-t \right)\left( x+t-1 \right)=0
\Leftrightarrow\hoac{&x=t\\&t=1-x    }\\
\Leftrightarrow\hoac{&x^2-2x+1=0\\&\heva{&x\le 1\\& 2x-1=(1-x)^2 } }
\Leftrightarrow\hoac{&x=1\\&\heva{&x\le 1\\ &x^2-4x+2=0} }
\Leftrightarrow\hoac{&x=1\\&\heva{&x\le 1\\ &\hoac{x&=2+\sqrt{2}\; (L)\\x&=2-\sqrt{2}}} }
$\\
Vậy phương trình có hai nghiệm $ x=1 $ hoặc $ x=2-\sqrt{2} $.
}
\end{bt}
%
\begin{bt}%[Hoàng Tựu]
	Giải phương trình $ x^2+2\sqrt{2-2x}=2. $
	\loigiai{	
Đặt $y=\sqrt{2-2x}\Rightarrow 2=y^2+2x$.\\
Phương trình trở thành:
$
x^2+2y-2x-y^2=0(*)\\
\Leftrightarrow  (x-y)(x+y-2)=0
\Leftrightarrow \heva{x&=y \\ y&=2-x   }
\Leftrightarrow  \heva{  x^2+2x-2=0 \\ 
x^2-2x+2=0  }
\Leftrightarrow \heva{ x&=-1+\sqrt{3}(tm) \\ 
	x&=-1-\sqrt{3}<0 .     }
$\\
Vậy phương trình có nghiệm $ x=- 1+\sqrt{3}.$
}
\end{bt}

\begin{dang}{Đặt ẩn phụ bằng hằng số}
	\noindent Ta cần giải phương trình $ F(x)=0 $.\\
	Về mặt ý tưởng, chúng ta đảo vai trò ẩn số thành hệ số và hệ số làm ẩn. \\Ví dụ như $ (2x+1)(1-(x^2+2))=0 $ thì ta có $ \hoac{& 2x=1 \\ & x^2+2=1} \Leftrightarrow \hoac{& 1=2x\\ & 1=x^2+2} $. Do đó, một cách tự nhiên ta coi $ 1 $ là ẩn số. Đặt $ y=1 $, ta có phương trình $ y^2-(x^2-2x+2)-2x(x^2+2)=0 $. Điều này có rất có ý nghĩa ở chỗ chuyển từ một phương trình bậc cao về phương trình bậc hai. Từ phân tích như vậy, ta khái quát thành phương pháp như sau:
		\begin{description}
			\item[Bước 1] Đặt $ K=f(x)$ với $ K $ là hằng số.
			\item[Bước 2] Đưa phương trình ban đầu về dạng $ G(K,x)=0 $ ẩn $ K $ đã biết cách giải.
			\item[Bước 3] Với mỗi nghiệm $ K$ của phương trình $ G(K,x)=0 $,  giải phương trình $ K=f(x)$. Sau đó đối  chiếu với điều kiện của phương trình ban đầu để kết luận nghiệm.
		\end{description}
\end{dang}

\begin{bt}%[HSG Bắc Từ Liêm, 2017-2018]%[Hoàng Tựu]
	Giải phương trình $x^2+2\sqrt{2-2x}=2$.
	\loigiai{
		ĐK: $ x\le 1 $.
Đặt $y=2$ và phương trình trở thành: 
$
2\sqrt{y-2x}=y-{{x}^{2}}\\
\Leftrightarrow\begin{cases}
-\sqrt{2}\le x\le 1 \\ 
{{y}^{2}}-2\left( {{x}^{2}}+2 \right)y+{{x}^{4}}+8x=0 \\ 
\end{cases}\\
\Leftrightarrow\begin{cases}
\hoac{   y={{x}^{2}}+2+2x-2={{x}^{2}}+2x \\ 
	y={{x}^{2}}+2-2x+2={{x}^{2}}-2x+4   }\\
\\
-\sqrt{2}\le x\le 1 \\
\end{cases}\\
\Leftrightarrow \begin{cases}
-\sqrt{2}\le x\le 1 \\
\hoac{ x^2+2x-2=0\\
	x^2-2x+2=0  }
\end{cases}\\
\Leftrightarrow x=-1+\sqrt{3}.
$
 Vậy phương trình có nghiệm $ =-1+\sqrt{3} $.
}
\end{bt}
\begin{bt}%[]%[Hoàng Tựu]
Giải phương trình $x^2-3x+1+\sqrt{2x-1}=0$.\\

\loigiai{
Điều kiện: $x\ge \dfrac{1}{2}$.\\
Đặt $ y=1 $. Phương trình trở thành
\begin{align*}
&x^2-3x+y=\sqrt{2x-y}\\
\Leftrightarrow & \begin{cases}
x^2-3x+y\ge 0 \\ 
(x^2-3x+y)^2=2x-y \\ 
\end{cases}\\
\Leftrightarrow & \begin{cases}
x^2-3x+y\ge 0 \\ 
y^2+\left( 1+2x^2-6x \right)y+x^4+9x^2-6x^3-2x=0(*). \\ 	
\end{cases}
\end{align*}
Phương trình (*) có:\\
$\Delta =(1+2x^2-6x)^2-4(x^4+9x^2-6x^3-2x)=4x^2-4x+1=(2x-1)^2$

$ \Rightarrow  \hoac{y&=\dfrac{6x-2x^2-1+2x-1}{2}=-x^2+4x-1 \\ 
	y&=\dfrac{6x-2x^2-1-2x+1}{2}=2x-x^2  }\\
\Leftrightarrow \hoac{ x^2-4x+2=0\\ x^2-2x+1=0}\Leftrightarrow \hoac{x&=2\pm \sqrt{2}\\x&=1}.\\
	$
Thử lại, ta thấy $ x=1 $ hoặc $ x=2-\sqrt{2} $ là nghiệm của phương trình.
}
\end{bt}

\begin{dang}{Đặt ẩn phụ đưa về phương trình đẳng cấp}
 Phương trình đẳng cấp bậc là phương trình dạng $ au^2+buv+cv^2=0 $.
  		\begin{description}
  			\item[Bước 1] Đặt $ u=f(x), v=g(x)$.
  			\item[Bước 2] Đưa phương trình ban đầu về dạng $ au^2+buv+cv^2=0  $ hoặc $ au+bv=c\sqrt{uv} $ hoặc $ au+bv=\sqrt{cu^2+dv^2} $.
  			\item[Bước 3] Thay nghiệm thu được ở bước 2 vào $ u $ hoặc $ v $. Giải phương trình, sau đó đối  chiếu với điều kiện của phương trình ban đầu để kết luận nghiệm.
  		\end{description}
  \end{dang}

\begin{bt}%[Hoàng Tựu]
  	Giải phương trình $ 5\sqrt{x^3+1}=2(x^2+1). $
  	\loigiai{
  Điều kiện: $ x\ge-1. $	\\
  \noindent
 $
  	5\sqrt{x^3+1}=2(x^2+1)
  	\Leftrightarrow 5\sqrt{(x+1)(x^2-x+1)}=2(x^2+1)
  $\\
  Đặt $ u=\sqrt{x+1}, v=\sqrt{x^2-x+1} $. \\Phương trình trở thành $ 2(u+v)=5\sqrt{uv} \Leftrightarrow 4u^2-17uv+4v^2=0\Leftrightarrow (u-4v)(4u-v)=0$.\\
  Nếu $ u=4v\Leftrightarrow \sqrt{x+1}=4\sqrt{x^2-x+1}\Leftrightarrow 16x^2-17x+15=0 $ (pt vô nghiệm).\\
 Nếu $ 4u=v\Leftrightarrow 4\sqrt{x+1}=\sqrt{x^2-x+1}\Leftrightarrow x^2-17x-15=0 \Leftrightarrow \hoac{x&=\dfrac{17+\sqrt{349}}{2}\\ x&=\dfrac{17-\sqrt{349}}{2}(L)}.$
\\
Vậy phương trình có nghiệm $ x= \dfrac{17+\sqrt{349}}{2}$.
}
\end{bt}

\begin{bt}%[Hoàng Tựu]
  	Giải phương trình $ x^2+3\sqrt{x^2-1}=\sqrt{x^4-x^2+1}. $
  		\loigiai{
  		Điều kiện: $ x\ge1. $ hoặc $ x\le -1 $.	\\
  		\noindent
  		
  		Đặt $ u=x^2, v=\sqrt{x^2-1} $. \\Phương trình trở thành $ u+3v=\sqrt{u^2-v^2}\Leftrightarrow v^2+6uv=0\Leftrightarrow \hoac{ v&=0\\v&=-6u}\Leftrightarrow \hoac{\sqrt{x^2-1}&=0\Leftrightarrow x=\pm 1\\ \sqrt{x^2-1}&=-6x^2\;(VN)}.$
  		\\
  		Vậy phương trình có nghiệm $ x= \pm 1.$  		
  	}
\end{bt}

\begin{dang}{Đặt ẩn phụ đưa về  hệ phương trình }	
Ta cần giải phương trình $ F(x)=0 $.	
		\begin{description}
			\item[Bước 1] Đặt $ u=f(x), v=g(x)$.
			\item[Bước 2] Đưa phương trình ban đầu về hệ phương trình.
			\item[Bước 3] Thay nghiệm thu được ở bước 2 vào $ u $ hoặc $ v $. Giải phương trình, sau đó đối  chiếu với điều kiện của phương trình ban đầu để kết luận nghiệm.
		\end{description}	
\end{dang}

\begin{bt}%[Hoàng Tựu]
	Giải phương trình $ x^2+2\sqrt{2-2x}=2. $
	\loigiai{
		Điều kiện: $ x\le 1. $\\
	Đặt $y=\sqrt{2-2x}\,\,,y\ge 0$.
$
\Rightarrow \begin{cases}
	{{x}^{2}}+2y=2 \\ 
	{{y}^{2}}+2x=2 \\ 
	\end{cases}	
	\Leftrightarrow \begin{cases}
	y=2-x \\ 
	x=y 
	\end{cases}\\
	\Leftrightarrow \begin{cases}
	\begin{cases}
	2-2x=(2-x)^2\\
	x\le 2
	\end{cases}\\\\
	\begin{cases}
	x\ge 0\\ 
	x^2=(2-2x)^2\\
	\end{cases}
	\end{cases}
	\Leftrightarrow  \begin{cases}
	\begin{cases}
	x^2-2x+2=0\\
	x\le 2
	\end{cases}\\\\
	\begin{cases}
	x\ge 0\\ 
	x^2+2x-2=0\\
	\end{cases}
	\end{cases}
	\Leftrightarrow  x=-1+\sqrt{3}(\text{thỏa mãn}).
	$\\
	Vậy phương trình có nghiệm $ x=-1+\sqrt{3}. $
}
\end{bt}

\begin{bt}%[Hoàng Tựu]
	Giải phương trình $ 2\sqrt[3]{3x-2}+3\sqrt{6-5x}-8=0. $\\
	\loigiai{   
		Điều kiện: $ x\le \dfrac{6}{5}. $\\
		Đặt $u=\sqrt[3]{3x-2},\; v=\sqrt{6-5x}\,\,,v\ge 0$.\\
		Ta có $ \begin{cases}
		u^3=3x-2\\v^2=6-5x
		\end{cases} \Leftrightarrow 5u^3+3v^2=8.
		$
		\\
		Vậy phương trình trở thành\\
		$
		 \begin{cases}
		2u+3v=8 \\ 5u^3+3v^2=8
		\end{cases}	
		\Leftrightarrow \begin{cases}
		v=\dfrac{8-2u}{3} \\ 
		5u^3+3.\left( \dfrac{8-2u}{3}\right)^2=8 
		\end{cases}
		\Leftrightarrow  \begin{cases}
		\begin{cases}
		v=\dfrac{8-2u}{3} \\
		15u^3+4u^2-32u+40=0
		\end{cases}\\\\
		\begin{cases}
		v=\dfrac{8-2u}{3} \\ 
		(u+2)(15u^2-26u+20)=0\\
		\end{cases}
		\end{cases}\\
		\Leftrightarrow
		\begin{cases}
		v=\dfrac{8-2u}{3} \\
		u=-2\Rightarrow x=-2\;(\text{thỏa mãn}) 
		\end{cases}.\\
		$
	\\
		Vậy phương trình có nghiệm $ x=-2. $
	}		
\end{bt}
