\begin{ex}%[Hà Lê]%[9D1K6]%[9D1K8]
    \hfill
    \begin{enumerate}
        \item Không dùng máy tính bỏ túi, hãy rút gọn biểu thức $A=\left( \sqrt{26}+5\sqrt{2} \right)\sqrt{19-5\sqrt{13}}$.
        \item Cho biểu thức $B=\dfrac{x^2-\sqrt{x}}{x+\sqrt{x}+1}-\dfrac{3x-\sqrt{x}}{\sqrt{x}}$.\\
Rút gọn $B$ và tìm $x$ để $B=1$.
    \end{enumerate}
\loigiai
    {
 \begin{enumerate}
        \item Ta có \\
$
\begin{aligned}
A&=\sqrt{2} \left( \sqrt{13}+5 \right) \sqrt{19-5\sqrt{13}}= \left( \sqrt{13}+5 \right) \sqrt{38-10\sqrt{13}}\\ 
& =\left( \sqrt{13}+5 \right) \sqrt{\left( 5- \sqrt{13} \right)^2}=\left( \sqrt{13}+5 \right)\left( 5-\sqrt{13} \right) =25-13=12.
\end{aligned}.$

       \item Điều kiện: $ (x > 0)$, ta có \\
$\begin{aligned}
B &= \dfrac{\sqrt{x} \left( x\sqrt{x}-1 \right)}{x+\sqrt{x}+1} - \left( 3 \sqrt{x}-1\right) =\dfrac{\sqrt{x} \left( \sqrt{x}-1 \right) \left(x+\sqrt{x} +1 \right)}{x+\sqrt{x}+1}-\left( 3 \sqrt{x}-1\right)\\
&= x -\sqrt{x} -3\sqrt{x}+1=x - 4\sqrt{x}+1.
\end{aligned}$
\\
Để $B=1$ thì $x^2-4\sqrt{x}=0$ suy ra $\hoac{&x=0\, (\text{loại}) \\ & x=4}$.

\end{enumerate}
    }
\end{ex}


\begin{ex}%[Hà Lê]%[9D4K3]%[9D4K3]
\hfill
\begin{enumerate}
\item Cho parabol $(P): y=ax^2$. Tìm hệ số $a$ để đường thẳng $(d): y=2 $ cắt $(P)$ tại hai điểm $A$ và $B$ sao cho tam giác $OAB$ vuông (với $O$ là gốc tọa độ).
\item Tìm tham số $m$ để phương trình $x^2+(m-2)x-(m-1)(2m-3)=0$ có hai nghiệm phân biệt sao cho nghiệm này bằng bình phương nghiệm kia.
\end{enumerate}

\loigiai{
\begin{enumerate}
\item Phương trình hoành độ giao điểm của $(d)$ và $(P)$ là
\[
ax^2=2 \Leftrightarrow x^2=\dfrac{2}{a}
\]
Phương trình có 2 nghiệm phân biệt khi $a>0$. Khi đó, $A\left( \sqrt{\dfrac{2}{a}},2 \right)$ và $B\left( -\sqrt{\dfrac{2}{a}},2 \right)$.\\
Do $A, B$ đối xứng qua $Oy$ nên để tam giác $OAB$ vuông tại $O$ thì $\sqrt{\dfrac{2}{a}}=2 $ suy ra $a=\dfrac{1}{2}$.
\item Ta có $\Delta = \left( 3m-4 \right)^2$. Để phương trình có hai nghiệm phân biệt thì $m \ne \dfrac{4}{3}$.\\
Hai nghiệm của phương trình là $\hoac{& x_1=m-1 \\&x_2=3-2m}$. Ta có hai trường hợp sau\\
{\bf TH1.} $x_1=x_2^2$\\
Suy ra, $m-1=(3-2m)^2 \Leftrightarrow 4m^2-13m+10=0 \Leftrightarrow \hoac{& m=2\\&m=\dfrac{5}{2}}$.\\
{\bf TH2.} $x_2=x_1^2$\\
Suy ra, $3-2m=(m-1)^2 \Leftrightarrow m^2=2 \Leftrightarrow m = \pm \sqrt{2}$.\\
Vậy, $m \in \left\{2; \dfrac{5}{4};\pm \sqrt{2} \right\}$.
\end{enumerate}
}
\end{ex}

\begin{enumerate}
\item Giải phương trình $\left( \sqrt{x-3}+2 \right)^2+x=13$.
\item Giải hệ phương trình $\heva{& \dfrac{1}{x-2}+\dfrac{3}{y+1}=-2 \\& \dfrac{5}{x-2} -\dfrac{2}{y+1}=7.}$
\end{enumerate}

\loigiai{
\begin{enumerate}
\item Điều kiện: $x \ge 3$. Phương trình tương đương với
\[
2x+4\sqrt{x-3}-12=0 \Leftrightarrow 2(x-3)+4\sqrt{x-3}-6=0
\]
Đặt $t=\sqrt{x-3}\ge 0$ phương trình trở thành $2t^2+4t-6 \Leftrightarrow \hoac{&t=1 \\&t=-3 (\text{loại})}$.\\
Với $t=1$ suy ra $x=4$. Vậy $\mathcal{S}=\{4\}$.
\item Đặt $u=\dfrac{1}{x-2}; v=\dfrac{1}{y+1}$, hệ đã cho trở thành $\heva{& u+3v=-2 \\ & 5u-2v=7} \Leftrightarrow \heva{&u=1 \\ &v= -1}$.\\
Nghiệm của hệ phương trình là $(x;y)=(3;-2)$.
\end{enumerate}

}
\end{ex}

Cho tam giác $ABC (AB > AC)$ ngoại tiếp đường tròn tâm $I$, gọi $D, E, F$ lần lượt là các tiếp điểm của đường tròn $I$ với các cạnh $BC, CA$ và $AB$. Các đường thẳng $DE, DF$ lần lượt cắt tia $AI$ tại $K$ và $L$, gọi $H$ là đường cao hạ từ $A$ xuống $BC$.
\begin{enumerate}
\item Giả sử số đo góc $\widehat{BAC}=\alpha$, hãy tính số đo góc $\widehat{BIC}$ theo $\alpha$.
\item Chứng minh $BK \parallel EF$.
\item Gọi $M$ là trung điểm của $BC$, chứng minh tứ giác $KMLH$ nội tiếp.
\end{enumerate}
\loigiai{
\immini{
\begin{enumerate}
\item  Ta tìm được $\widehat{BIC}=90^\circ+\dfrac{\alpha}{2}$.
\item Ta có \\
$ \begin{aligned}
\widehat{BIK}&=180^\circ -\widehat{AIB}\\
&=90^\circ -\dfrac{\widehat{C}}{2}\\
&=\widehat{CDE}=\widehat{BDK}.
\end{aligned}$\\
Suy ra tứ giác $BIDK$ nội tiếp, suy ra $\widehat{BKI}=90^\circ $ suy ra $BK \parallel EF$.
\item Chứng minh được $AHKB$ và $ALHC$ là tứ giác nội tiếp. Suy ra, 
\[\widehat{LHM}=\widehat{LAC}=\widehat{LAB}=\widehat{MHK} (1)\]

Từ $(1)$ và $(2)$ suy ra tứ giác $MLHK$ nội tiếp.
\end{enumerate}

\begin{ex}%[Hà Lê]%[9D5G1]
Cho hai số thực $x, y$ thỏa mãn $0<x \le 1$, $0<y \le 1$ và $x+y=3xy$. Tìm giá trị lớn nhất và giá trị nhỏ nhất của biểu thức $P=x^2+y^2-4xy$.
\loigiai
{
Ta có $x+y \ge 2\sqrt{xy}$ suy ra $3xy \ge 2 \sqrt{xy}$ suy ra $xy \ge \dfrac{4}{9}$.\\
Do $x, y \le 1$ suy ra $(1-x)(1-y) \ge 0$ suy ra $x+y-xy \le 1 \Leftrightarrow xy \le \dfrac{1}{2}$.\\
Đặt $xy=t \left( \dfrac{4}{9} \le t \le \dfrac{1}{2}\right)$ suy ra $P=9t^2-6t$.\\
Ta có $$P=(9t-4)\left(t-\dfrac{2}{9}\right)-\dfrac{8}{9}$$
Do $t \ge \dfrac{4}{9}$ nên $P \ge -\dfrac{8}{9}$. 
Dấu bằng xảy ra khi $t=\dfrac{4}{9}$.\\
Vậy $\min{P}=-\dfrac{8}{9}$ khi $\heva{&xy=\dfrac{4}{9} \\&x+y=\dfrac{4}{3}} \Leftrightarrow \heva{&x=\dfrac{2}{3} \\ &y=\dfrac{2}{3}}$.\\
Ta có 
\[P=9t^2-6t=\left(9t- \dfrac{3}{2}\right) \left( t-\dfrac{1}{2}\right)-\dfrac{3}{4} \]
Do $\dfrac{4}{9} \le t \le \dfrac{1}{2}$ nên $\left(9t- \dfrac{3}{2}\right) \left( t-\dfrac{1}{2}\right) \le 0$. Suy ra $P \le -\dfrac{3}{4}$.\\
Dấu bằng xảy ra khi $t=\dfrac{1}{2} \Leftrightarrow (x;y) \in \left\{ \left( 1; \dfrac{1}{2}\right); \left( \dfrac{1}{2}; 1 \right)\right\}$.\\
Vậy $\max{P}=-\dfrac{3}{4}$.
}
\end{ex}
