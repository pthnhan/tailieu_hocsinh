\begin{ex}%[WTT-218cau-ham-an-2D1, Bùi Anh Tuấn]%[2D1G5-5]
	Cho hàm số $y=f(x)$. Biết $f(x)$ có đạo hàm $f'(x)$ và hàm số $y=f'(x)$ có đồ thị như hình vẽ.
	\begin{center}
	\begin{tikzpicture}[scale=0.7,>=stealth]
	\draw[->](-2,0)--(6.0,0)node[above]{$x$};
	\draw[->](0,-3.1)--(0,4)node[right]{$y$};
	\draw[smooth,samples=100,domain=0.58:3]plot(\x,{-(3)*((\x)^2-4*(\x)+3)});
	\draw[smooth,samples=100,domain=3:5.5]plot(\x,{(3)*((\x)^2-8*(\x)+15)});
	\draw[dashed](2,0)--(2,3);
	\draw[dashed](4,0)--(4,-3);
	\fill (0,0)node[below left]{$O$}circle(1.2pt) (1,0)node[below]{$1$}circle(1.2pt) (2,0)node[below]{$2$}circle(1.2pt) (3,0)node[below]{$3$}circle(1.2pt) (4,0)node[above]{$4$}circle(1.2pt) (5,0)node[below right]{$5$}circle(1.2pt);
	\end{tikzpicture}
	\end{center}
	 Hàm số $g(x)=f(x-1)$ đạt cực đại tại điểm nào dưới đây?
	\choice
	{$x=2$}
	{\True $x=4$}
	{$x=3$}
	{$x=1$}
	\loigiai{
		Cách 1: Ta có
		\begin{eqnarray*}
		g'(x)&=&f'(x-1)=0 \\
		&\Leftrightarrow & \left[\begin{aligned}
		&x-1=1\\
		&x-1=3\\
		&x-1=5\\
		\end{aligned}\right.\\
		& \Leftrightarrow & \left[\begin{aligned}
		&x=2\\
		&x=4\\
		&x=6.\\
		\end{aligned}\right. \\ 	
	g'(x)&=&f'(x-1)>0 \\
		& \Leftrightarrow & \left[\begin{aligned}
			&1<x-1<3\\
			&x-1>5\\
		\end{aligned}\right.\\
		& \Leftrightarrow& \left[\begin{aligned}
			&2<x<4\\
			&x>6.\\
		\end{aligned}\right.
		\end{eqnarray*}
		Ta chọn đáp án B.\\
		Cách 2: Đồ thị hàm số $g'(x)=f'(x-1)$ là phép tịnh tiến đồ thị hàm số $y=f'(x)$ theo phương trục hoành sang phải 1 đơn vị. Hình vẽ
		\begin{center}
		\begin{tikzpicture}[scale=0.7,>=stealth]
		\draw[->](-2,0)--(7.0,0)node[above]{$x$};
		\draw[->](0,-3.1)--(0,4)node[right]{$y$};
		\draw[smooth,samples=100,domain=0.58:3]plot(\x,{-(3)*((\x)^2-4*(\x)+3)});
		\draw[smooth,samples=100,domain=3:5.5]plot(\x,{(3)*((\x)^2-8*(\x)+15)});
		\draw[smooth,samples=100,domain=1.58:4]plot(\x,{-(3)*((\x-1)^2-4*(\x-1)+3)});
		\draw[smooth,samples=100,domain=4:6.5]plot(\x,{(3)*((\x-1)^2-8*(\x-1)+15)})  node[below right]{$y=f'(x-1)$};
		\draw[dashed](2,0)--(2,3); \draw[dashed](3,0)--(3,3);
		\draw[dashed](4,0)--(4,-3); \draw[dashed](5,0)--(5,-3);
		\fill (0,0)node[below left]{$O$}circle(1.2pt) (1,0)node[below]{$1$}circle(1.2pt) (2,0)node[below]{$2$}circle(1.2pt) (3,0)node[below]{$3$}circle(1.2pt) (4,0)node[above]{$4$}circle(1.2pt) (5,0)node[below right]{$5$}circle(1.2pt) (6,0)node[below right]{$6$}circle(1.2pt);
		\end{tikzpicture}
		\end{center}
		Đồ thị hàm số $g'(x)=f'(x-1)$ cắt trục hoành tại các điểm có hoành độ $x=2$; $x=4$; $x=6$ và giá trị hàm số $g'(x)$ đổi dấu từ dương sang âm khi qua điểm $x=4$.}
\end{ex}

	\begin{center}
	\begin{tikzpicture}[scale=0.7,>=stealth] 
	\draw[->](-2.0,0)--(4,0)node[above]{$x$};
	\draw[->](0,-5)--(0,2)node[right]{$y$};

	\end{center}
	Hàm số $y=g(x)=f(x)+4x$ có bao nhiêu điểm cực trị?

	{\True 1}
	{2}
	{3}
	{4}
	\loigiai{
		Số cực trị của hàm $g(x)$ bằng số nghiệm bội lẻ của phương trình\\
		$g'(x)=f'(x)+4=0 \Leftrightarrow f'(x)=-4$ \\
		Dựa vào đồ thị của hàm $f'(x)$ ta thấy phương trình trên có một nghiệm đơn.}
