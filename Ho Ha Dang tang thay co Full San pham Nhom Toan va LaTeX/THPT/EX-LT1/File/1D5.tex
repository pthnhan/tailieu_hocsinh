\begin{ex}%[GT11-Chương 5]%[Đào Anh Dương]%[1D5B1-1]
	Hàm số $f(x_0)$ có đạo hàm tại $x_0$ là $f'(x_0)$. Chọn khẳng định \textbf{sai}.
	\choice
	{$f'(x_0) = \lim\limits_{x\to x_0} \dfrac{f(x) - f(x_0)}{x-x_0}$}
	{$f'(x_0) = \lim\limits_{\Delta x\to 0} \dfrac{f(x_0 + \Delta x) - f(x_0)}{\Delta x}$}
	{$f'(x_0) = \lim\limits_{h\to 0} \dfrac{f(x_0 + h) - f(x_0)}{h}$}
	{\True $f'(x_0) = \lim\limits_{x\to x_0} \dfrac{f(x + x_0)-f(x_0)}{x-x_0}$}
	\loigiai{
		\begin{itemize}
			\item Theo định nghĩa đạo hàm tại một điểm SGK ĐS-GT 11, ta có
			$f'(x_0) = \lim\limits_{x\to x_0} \dfrac{f(x) - f(x_0)}{x-x_0}.$
			\item Đặt $\Delta x = x - x_0$, ta có
			$$f'(x_0) = \lim\limits_{x\to x_0} \dfrac{f(x) - f(x_0)}{x-x_0} = \lim\limits_{\Delta x\to 0} \dfrac{f(x_0 + \Delta x) - f(x_0)}{x_0 + \Delta x - x_0} = \lim\limits_{x\to x_0} \dfrac{f(x) - f(x_0)}{x-x_0}.$$
			\item Đặt $h = x - x_0$, ta có
			$$f'(x_0) = \lim\limits_{x\to x_0} \dfrac{f(x) - f(x_0)}{x-x_0} = \lim\limits_{h \to 0} \dfrac{f(x_0 + h) - f(x_0)}{x_0 + h - x_0} = \lim\limits_{h\to 0} \dfrac{f(x_0 + h) - f(x_0)}{h}.$$
			\item Như vậy, đáp án sai là $f'(x_0) = \lim\limits_{x\to x_0} \dfrac{f(x + x_0)-f(x_0)}{x-x_0}$.
		\end{itemize}	
	}
\end{ex}

			\end{align*}

\begin{ex}%[GT11-Chương 5]%[Nguyễn Tiến Thùy]%[1D5B4-1]
	Cho hàm số $y=f(x)$ xác định và có đạo hàm trên khoảng $(a;b)$. Giả sử $\Delta x$ là số gia của $x$. Khi đó, nhận định nào dưới đây {\bf sai}?
	\choice
	{$\mathrm{d}f(x)=f'(x)\cdot \Delta x$}
	{$\mathrm{d}y=f'(x)\mathrm{\,d}x$}
	{$\dfrac{\mathrm{d}y}{\mathrm{d}x}=f'(x)$}

	\loigiai{
		Theo định nghĩa vi phân $\mathrm{d}f(x)=f'(x)\cdot \Delta x$.\\
		Suy ra $\mathrm{d}x=1\cdot\Delta x=\Delta x$, từ đó suy ra $\mathrm{d}y=\mathrm{d}f(x)=f'(x)\mathrm{\,d}x\Rightarrow \dfrac{\mathrm{d}y}{\mathrm{d}x}=f'(x)$.
	}
\end{ex}

\begin{ex}%[GT11-Chương 5]%[Phan Anh Tiến]%[1D5K5-2]
	Cho $f(x)$ là một hàm đa thức bậc ba. Trong các mệnh đề sau có bao nhiêu mệnh đề đúng?
	\begin{enumerate}[(1)]
		\item $f^{(3)}(x)=C$ với mọi $x\in\mathbb{R}$ ($C$ là hằng số).
		\item $f^{(5)}(1)=5!$.
		\item $f^{(6)}(2)=0$.
		\item Phương trình $f''(x)=0$ luôn có nghiệm thực.
	\end{enumerate}
	\choice
	{$1$}
	{\True $3$}
	{$2$}
	{$4$}
	\loigiai{
		Giả sử $f(x)=ax^3+bx^2+cx+d \,(a\ne 0)$ nên $f''(x)=6ax+2b$; $f^{(3)}(x)=6a$ và  $f^{(n)}(x)\equiv 0, \forall n \ge 4.$\\
		Do đó $f^{(5)}(1)=5!$ là đáp án sai.
	}
\end{ex}

\begin{ex}%[GT11-Chương 5]%[Phan Anh Tiến]%[1D5B5-2]

\begin{ex}%[GT11-Chương 5]%[Phan Anh Tiến]%[1D5K5-1]
	Cho hàm số $y=f(x)$ có đạo hàm cấp hai tại $x_0.$ Trong số các khẳng định sau có bao nhiêu khẳng định đúng?
	\begin{enumerate}[(1)]
		\item $f''(x_0)=(f'(x_0))'.$
		\item $f''(x_0)=\mathop {\lim }\limits_{x \to x_0}\dfrac{f'(x)-f'(x_0)}{x-x_0}.$
		\item Hàm số $y=f''(x)$ liên tục tại $x_0.$
	\end{enumerate}
	\choice
	{$0$}
	{$2$}
	{\True $1$}
	{$3$}
	\loigiai{
		Theo định nghĩa ta có $f''(x_0)=\mathop {\lim }\limits_{x \to x_0}\dfrac{f'(x)-f'(x_0)}{x-x_0}.$
	}
\end{ex}

\begin{ex}%[GT11-Chương 5]%[Phan Anh Tiến]%[1D5K5-1]
	Cho hàm số $y=u \cdot v$ trong đó $u$ có đạo hàm cấp $n>2$ còn $v''\equiv 0$. Tính  $y^{(n)}$.
	\choice
	{$y^{(n)}\equiv 0$}
	{$y^{(n)}=u^{(n)}$}
	{ \True $y^{(n)}=u^{(n)} \cdot v+n \cdot u^{(n-1)} \cdot v'$}
	{$y^{(n)}=u^{(n)} \cdot v+u^{(n-1)} \cdot v'$}
	\loigiai{
		Dùng công thức đạo hàm của một tích, rồi quy nạp ta được $$y^{(n)}=\mathrm{C}_n^0 \cdot u^{(n)}+ \mathrm{C}_n^1 \cdot u^{(n-1)} \cdot v'+\cdots +\mathrm{C}_n^n \cdot v^{(n)}.$$
		Mà theo giả thiết $v''\equiv 0$ nên $y^{(n)}=u^{(n)} \cdot v+n \cdot u^{(n-1)} \cdot v'$
	}
