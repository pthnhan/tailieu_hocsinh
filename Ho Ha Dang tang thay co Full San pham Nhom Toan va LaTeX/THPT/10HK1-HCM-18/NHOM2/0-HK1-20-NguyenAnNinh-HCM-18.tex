\begin{bt}%[Đề HK1, Nguyễn An Ninh, Hồ Chí Minh 17-18]%[Nguyễn Kim Đông]%[0D2B3-2]
Tìm phương trình của parabol $(P):y=ax^2+2x+c$, biết $(P)$ có đỉnh $I(2;2)$.
\loigiai{  
Vì $(P)$ có đỉnh $I(2;2)$ nên ta có $\heva{&\dfrac{-2}{2a}=2\\&4a+4+c=2 }\Leftrightarrow \heva{&a=-\dfrac{1}{2}\\&c=0.}$\\
Do đó, parabol cần tìm có phương trình $(P): y=-\dfrac{1}{2}x^2+2x$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn An Ninh, Hồ Chí Minh 17-18]%[Nguyễn Kim Đông]%[0D3B2-3]
Giải phương trình $\dfrac{3x+4}{x-2}-\dfrac{1}{x+2}=\dfrac{34}{x^2-4}+4$.
\loigiai{
Điều kiện xác định: $x \neq \pm 2$.\\
Với điều kiện trên, phương trình đã cho tương đương với
\begin{equation*}
	(3x+4)(x+2)-(x-2)=34+4(x^2-4) \Leftrightarrow x^2-9x+8=0 \Leftrightarrow \hoac{&x=1 \text{ (thỏa điều kiện)}\\&x=8 \text{ (thỏa điều kiện).}}
\end{equation*}
Vậy $\mathcal{S}=\left\{1;8\right\}$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn An Ninh, Hồ Chí Minh 17-18]%[Nguyễn Kim Đông]%[0D3K2-4]
Giải phương trình $\sqrt{x}+\sqrt{2x-1}=2$.
\loigiai{
Điều kiện xác định: $x \geq \dfrac{1}{2}$.\\
Phương trình đã cho tương đương với 
\begin{equation*}
	\heva{&\sqrt{x}\leq 2\\&2x-1=\left(2-\sqrt{x}\right)^2} \Leftrightarrow \heva{&0 \leq x \leq 4\\&5-x=4\sqrt{x}} \Leftrightarrow \heva{&0 \leq x \leq 4\\&x^2-26x+25=0} \Leftrightarrow \heva{&0 \leq x \leq 4\\&\hoac{&x=1 \text{ (thỏa )}\\&x=25 \text{ (loại).}}}
\end{equation*}
Kết hợp với điều kiện, ta có $x=1$ là nghiệm của phương trình.\\
Vậy $\mathcal{S}=\left\{1\right\}$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn An Ninh, Hồ Chí Minh 17-18]%[Nguyễn Kim Đông]%[0D4K1-2]
	Cho $a \geq 1$, $b \geq 1$. Chứng minh rằng $a\sqrt{b-1}+b\sqrt{a-1}\leq ab$. Khi nào xảy ra dấu của đẳng thức?
	\loigiai{Vì $a \geq 1$ và $b \geq 1$ nên bằng cách chia 2 vế bất đẳng thức cần chứng minh cho $ab$, ta có bất đẳng thức tương đương
	\begin{equation*}
		\dfrac{\sqrt{b-1}}{b}+\dfrac{\sqrt{a-1}}{a}\leq 1 \Leftrightarrow \sqrt{\dfrac{1}{b} \left(1-\dfrac{1}{b}\right)}+\sqrt{\dfrac{1}{a}\left(1-\dfrac{1}{a}\right)} \leq 1.
\end{equation*}
Áp dụng bất đẳng thức Cauchy cho hai số dương $\dfrac{1}{b}$ và $1-\dfrac{1}{b}$, ta có $\sqrt{\dfrac{1}{b} \left(1-\dfrac{1}{b}\right)} \leq  \dfrac{\dfrac{1}{b}+1-\dfrac{1}{b}}{2}=\dfrac{1}{2}$.\\
Chứng minh tương tự, ta có $\sqrt{\dfrac{1}{a} \left(1-\dfrac{1}{a}\right)} \leq  \dfrac{\dfrac{1}{a}+1-\dfrac{1}{a}}{2}=\dfrac{1}{2}$.\\
Cộng vế theo vế hai bất đẳng thức trên, ta có điều phải chứng minh.\\
Dấu đẳng thức xảy ra khi và chỉ khi $\heva{&\dfrac{1}{a}=1-\dfrac{1}{a}\\ &\dfrac{1}{b}=1-\dfrac{1}{b}\\& a \geq 1\\&b \geq 1}\Leftrightarrow a=b=2.$}
\end{bt}
\begin{bt}%[Đề HK1, Nguyễn An Ninh, Hồ Chí Minh 17-18]%[Nguyễn Kim Đông]%[0H1B2-2]
	Cho tam giác $ABC$. Gọi $M, N, P$ lần lượt là trung điểm của các cạnh $BC, CA, AB$. Chứng minh rằng $\vec{AM}+\vec{BN}+\vec{CP}=\vec{0}$.
\loigiai{
	\immini{
		Ta có
		$\heva{&\vec{AM}=\vec{AC}+\vec{CM}\\ &\vec{BN}=\vec{BA}+\vec{AN}\\&\vec{CP}=\vec{CB}+\vec{BP}}$
		
		$
		\begin{aligned}
		\Rightarrow \vec{AM}+\vec{BN}+\vec{CP}&=\left(\vec{AC}+\vec{CB}+\vec{BA}\right)+\left(\vec{CM}+\vec{BP}+\vec{AN}\right)\\
		&=\vec{0}+ \vec{CM}+\vec{BP}+\vec{AN}.\\
		\end{aligned}
		$
		
		Lại có $\heva{&\vec{BP}=\vec{MN}\\ &\vec{AN}=\vec{NC}}$
		
		$\Rightarrow \vec{AM}+\vec{BN}+\vec{CP}= \vec{CM}+\vec{MN}+\vec{NC} =\vec{0}$.
	}
	{
		\begin{tikzpicture}[scale=0.7]
		\tkzInit[xmin=-3, ymin=-3,xmax=3,ymax=2]
		\tkzDefPoints{-1/2/A, -3/-3/B,3/-3/C}
		\tkzLabelPoints[above left](A)
		\tkzLabelPoints[above left](B)
		\tkzLabelPoints[above right](C)
		\tkzDrawSegments[color=black](A,B B,C C,A)
		\tkzDefMidPoint(A,B)
		\tkzGetPoint{P}
		\tkzDefMidPoint(C,B)
		\tkzGetPoint{M}
		\tkzDefMidPoint(A,C)
		\tkzGetPoint{N}
		\tkzLabelPoints[above left](P)
		\tkzLabelPoints[ below left](M)
		\tkzLabelPoints[above right](N)
		\tkzDrawPoints(A,B,C,P,M,N)
		\tkzDrawSegments[color=black](M,N N,P P,M)
		\end{tikzpicture}
	}
}
\end{bt}
\begin{bt}%[Đề HK1, Nguyễn An Ninh, Hồ Chí Minh 17-18]%[Nguyễn Kim Đông]%[0H1B4-1]%[0H2B2-1]%[0H2B2-2]
	Trong mặt phẳng $Oxy$, cho tam giác $ABC$ với $A(2;-2)$, $B(6;6)$, $C(-2;2)$. 
	\begin{enumerate}
		\item Tìm tọa độ điểm $D$ để tứ giác $ABDC$ là hình bình hành.
		\item Tìm tọa độ trực tâm $H$ của tam giác $ABC$.
		\item Tìm tọa độ tâm $I$ của đường tròn ngoại tiếp tam giác $ABC$.   	
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Giả sử $D(x_D;y_D)$. Khi đó, $\vec{AB}=(4;8)$, $\vec{CD}=(x_D+2; y_D-2)$.\\
	Tứ giác $ABDC$ là hình bình hành khi và chỉ khi $\vec{AB}=\vec{CD}$ $\Leftrightarrow \heva{&x_D+2=4\\&y_D-2=8} \Leftrightarrow \heva{&x_D=2\\ &y_D=10.}$\\
	Suy ra $D(2;10)$.
	\item  Giả sử trực tâm $H$ có tọa độ $H(x_H;y_H)$.Ta có $\vec{AH}=\left(x_H-2;y_H+2\right)$, $\vec{BH}=\left(x_H-6;y_H-6\right)$, $\vec{AC}=(-4;4)$, $\vec{BC}=(-8;-4)$. \\
	 Vì $H$ là trực tâm tam giác $ABC$ nên ta có $\heva{&\vec{AH} \perp \vec{BC}\\ & \vec{BH} \perp \vec{AC}} \Leftrightarrow \heva{& \vec{AH}\cdot \vec{BC}=0\\&\vec{BH}\cdot \vec{AC}=0.}$ \\
	Khi đó, ta có
		\begin{equation*}
			\heva{&-8\cdot \left(x_H-2\right)-4 \cdot \left(y_H+2\right)=0\\&-4\cdot \left(x_H-6\right)+4 \cdot \left(y_H-6\right)=0} \Leftrightarrow \heva{&-8x_H-4y_H=-8\\&-4x_H+4y_H=0} \Leftrightarrow \heva{&x_H=\dfrac{2}{3}\\&y_H=\dfrac{2}{3}.}
		\end{equation*}
		Do đó, $H\left(\dfrac{2}{3};\dfrac{2}{3}\right).$
	\item Gọi $I(x_I;y_I)$ là tâm đường tròn ngoại tiếp của tam giác $ABC$, ta có $\heva{&IA=IB\\&IA=IC}\Leftrightarrow \heva{&IA^2=IB^2\\&IA^2=IC^2.}$\\
	Khi đó, ta có
	\begin{equation*}
		\heva{&\left(2-x_I\right)^2+\left(-2-y_I\right)^2=\left(6-x_I\right)^2+\left(6-y_I\right)^2\\&\left(2-x_I\right)^2+\left(-2-y_I\right)^2=\left(-2-x_I\right)^2+\left(2-y_I\right)^2} \Leftrightarrow \heva{&8x_I+16y_I=64\\&8x_I-8y_I=0} \Leftrightarrow \heva{&x_I=\dfrac{8}{3}\\ &y_I=\dfrac{8}{3}.}
	\end{equation*}
	Do đó, $I\left(\dfrac{8}{3};\dfrac{8}{3}\right)$.
\end{enumerate}}
\end{bt}
