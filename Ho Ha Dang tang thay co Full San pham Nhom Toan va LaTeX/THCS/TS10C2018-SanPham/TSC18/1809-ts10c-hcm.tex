	Cho $ a,b,c $ là ba số thực thỏa điều kiện $ a+b+c = 0 $ và $ a^2 = 2(a+c+1)(a+b-1) $. Tính giá trị của biểu thức $ A = a^2 + b^2 + c^2 $.
	\loigiai{
		Thay $ a = -(b+c);\ a+c=  -b $ và $ a+b = -c $ vào điều kiện thứ hai ta có
		\begin{eqnarray*}
			&&(b+c)^2 = 2(-b+1)(-c-1)\\& \Leftrightarrow &b^2 + c^2 + 2bc = 2bc +2b - 2c - 2\\
			& \Leftrightarrow &(b-1)^2 + (c+1)^2 = 0 \Leftrightarrow \heva{&b = 1\\&c=-1.} 
		\end{eqnarray*}
		Suy ra $ a = 0 $.\\ Vậy $ A = a^2 + b^2 + c^2 = 2 $.
	}

\begin{ex}%[Đề thi vào 10,toán chuyên, thành phố Hồ Chí Minh, năm 2018]%[Nguyễn Thế Út, Dự án EX-ts10c-18]%[9D0G1]
	 Chứng minh rằng $ x^4 -x+\dfrac{1}{2} > 0 $ với mọi số thực $ x $.
	\loigiai{
	\textbf{Cách 1.} Ta có phân tích
			\[ x^4 -x + \dfrac{1}{2} = x^4 - x^2 + \dfrac{1}{4} + x^2 - x + \dfrac{1}{4} = \left(x^2 - \dfrac{1}{2} \right)^2 + \left(x-\dfrac{1}{2} \right)^2 \ge 0, \forall x\in \mathbb{R}. \]
			Đẳng thức xảy ra khi và chỉ khi $ \heva{&x^2 = \dfrac{1}{2}\\& x = \dfrac{1}{2}}$, hệ này vô nghiệm.\\
			Do đó đẳng thức không xảy ra, nghĩa là $ x^4 -x+\dfrac{1}{2} > 0 $ với mọi số thực $ x $.\\
			\textbf{Cách 2.} Ta có đánh giá bởi $ AM-GM $ như sau:
			\[ x^4 + \dfrac{1}{\sqrt[3]{256}} + \dfrac{1}{\sqrt[3]{256}} + \dfrac{1}{\sqrt[3]{256}} \ge 4\sqrt[4]{x^4 \cdot \dfrac{1}{\sqrt[3]{256}} \cdot \dfrac{1}{\sqrt[3]{256}} \cdot \dfrac{1}{\sqrt[3]{256}}} = |x| \ge x  \]
			hay
			\[ x^4 - x \ge - \dfrac{3}{\sqrt[3]{256}} = - \sqrt[3]{\dfrac{27}{256}} > - \sqrt[3]{\dfrac{32}{256}} = \dfrac{1}{2} .\]
			Do đó $ x^4 - x  + \dfrac{1}{2} > 0 .$
	}
\end{ex}	
\begin{ex}%[Đề thi vào 10,toán chuyên, thành phố Hồ Chí Minh, năm 2018]%[Nguyễn Thế Út, Dự án EX-ts10c-18]%[9D0G7]
 Cho $ x,y $ là các số thực thỏa mãn điều kiện $ x^2 - xy + y^2 = 3 $. Tìm giá trị lớn nhất và giá trị nhỏ nhất của biểu thức $ P = x^2 + y^2 $.
	\loigiai{ \textbf{Cách 1.} Ta có đánh giá sau
			\[ x^2y^2 \le \left(\dfrac{x^2+y^2}{2}\right)^2 \Leftrightarrow -\dfrac{x^2+y^2}{2} \le xy \le  \dfrac{x^2+y^2}{2} .\]
			Từ điều kiện ta có $ xy = x^2+y^2 - 3 $, thay vào BĐT trên ta có
			\[  -\dfrac{x^2+y^2}{2} \le x^2+y^2 - 3 \le  \dfrac{x^2+y^2}{2} \Leftrightarrow 2 \le x^2 + y^2 \le 6. \]
			Đẳng thức của đánh giá đầu tiên xảy ra khi $ x^2 = y^2 \Leftrightarrow x=\pm y $.\\
			Do đó $ \max P = 6 $ khi $ x = y =\pm \sqrt{3} $ và $ \min P = 2 $ khi $ x = -y = \pm 1 $.\\	
			\textbf{Cách 2.} Ta xét hai trường hợp sau
			\begin{enumerate}[\textbf{TH }1.]
				\item $ y = 0 $. Khi đó $ x^2 = 3 $. Vậy $ P = 3 $.
				\item $ y \ne 0 $. Ta có
				\[ (x^2-xy+y^2) P = 3(x^2+y^2) \Leftrightarrow (t^2 - t+1)P = 3t^2 + 3 \text{ chia hai vế cho $ y^2 $ và đặt $ t = \dfrac{x}{y} $}\]
				Khi $ P \ne 3 $, ta xem phương trình trên là bậc hai theo ẩn $ t $, nghĩa là
				\[ (P-3)t^2 - Pt + P-3 = 0 \]
				Rõ ràng phương trình này có nghiệm $ t $ do đó 
				\[ \Delta \ge 0 \Leftrightarrow P^2 - 4(P-3)^2 \ge 0 \Leftrightarrow 2 \le P \le 6 \]
			\end{enumerate}
			Kết hợp hai trường hợp ta có $ 2 \le P \le 6 $.
			\begin{itemize}
				\item Khi $ P = 2 $ thì $ t = -1 $, khi đó $ x = -y = \pm 1 $.
				\item Khi $ P = 6 $ thì $ t = 1 $, khi đó $ x = y = \pm \sqrt{3} $.
			\end{itemize}	
	}
\end{ex}

\begin{ex}%[Đề thi vào 10,toán chuyên, thành phố Hồ Chí Minh, năm 2018]%[Nguyễn Thế Út, Dự án EX-ts10c-18]%[9D0G8]
	Các số nguyên dương từ $ 1 $ đến $ 2018 $ được tô màu theo quy tắc sau: Các số mà khi chia cho $24$ dư $17$ được tô màu xanh; Các số mà khi chia cho $40$ dư $7$ được tô màu đỏ; Các số còn lại được tô màu vàng.
	\begin{enumerate}
		\item Chứng tỏ rằng không có số nguyên dương nào được tô cả hai màu xanh và đỏ. Hỏi có bao nhiêu số được tô màu vàng?
		\item Có bao nhiêu cặp số $ (a;b) $ sao cho $ a $ được tô màu xanh, $ b $ được tô màu đỏ và $ |a-b|  =2 $?
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Từ giả thiết ta suy ra:
			\begin{itemize}
				\item Số màu xanh sẽ có dạng $ a = 24k + 17,\ k\in \mathbb{Z} $, với $$ 1\le a\le 2018 \Leftrightarrow 1\le 24k + 17\le 2018 \Leftrightarrow -\dfrac{2}{3} \le k \le \dfrac{667}{8} \Leftrightarrow 0 \le k \le 83 .$$
				Vậy có $ 84 $ số được tô màu xanh.
				\item Số màu đỏ sẽ có dạng $ b = 40l + 7,\ l\in \mathbb{Z} $, với $$ 1\le b\le 2018 \Leftrightarrow 1\le 40l + 7\le 2018 \Leftrightarrow -\dfrac{3}{20} \le l \le \dfrac{2011}{40} \Leftrightarrow 0 \le l \le 50 .$$
				Vậy có $ 51 $ số được tô màu đỏ.
			\end{itemize}
			Giả sử có số được tô cả màu xanh và đỏ, khi đó tồn tại $ k_0,l_0 \in \mathbb{Z} $ sao cho
			\[ 24k_0 + 17 = 40l_0 + 7 \Leftrightarrow 12k_0 + 5 = 20l_0 \text{, mâu thuẫn với tính chẵn lẻ.} \]
			Vậy số nguyên dương nào được tô cả hai màu xanh và đỏ. Do đó số lượng số được tô màu vàng là $ 2018 - 84 - 51 = 1883 $.
			\item Ta xét hai trường hợp:
			\begin{itemize}
				\item $ a - b = 2 $. Thay $ a = 24k + 17,\ k\in \mathbb{Z} $ và $ b = 40l + 7,\ l\in \mathbb{Z} $ ta có
				\[ 24k + 17 - 40l - 7 = 2 \Leftrightarrow 3k - 5l + 1= 0 .\]
				Dễ thấy $ k = 3 $ và $ l = 2 $ là một nghiệm nguyên riêng của phương trình, vậy ta có nghiệm nguyên của phương trình là $ \heva{&k = 3+5t\\&l = 2+3t}, t\in \mathbb{Z} $.\\
				Vì $ 0 \le k \le 83 \Leftrightarrow 0 \le 3 +5t \le 83 \Leftrightarrow \dfrac{-3}{5} \le t \le 16 $. Vậy có $ 17 $ giá trị nguyên của $ t $ do đó trường hợp này có 17 cặp $ (a;b) $ thỏa đề bài.
				\item $ a - b = -2 $. Tương tự ta thu được phương trình nghiệm nguyên
				\[ 6k - 10l + 3 =0 . \]
				Dễ thấy $ UCLN(6,10) = 2 $ không là ước của 3, do đó phương trình này không có nghiệm nguyên.
			\end{itemize}
			Vậy có tất cả $ 17 $ cặp $ (a;b) $ thỏa đề bài.
		\end{enumerate}	
	}
\end{ex}
