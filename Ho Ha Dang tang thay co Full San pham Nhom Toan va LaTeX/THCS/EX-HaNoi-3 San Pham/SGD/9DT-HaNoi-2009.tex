\begin{ex}[9D1B8][9D1B8][9D1B8]
Cho biểu thức $A=\dfrac{x}{x-4}+\dfrac{1}{\sqrt{x}-2}+\dfrac{1}{\sqrt{x}+2}$ (với $x\ne 4,\ x\ge 0$).
\begin{enumerate}
	\item 	Rút gọn biểu thức $A$.
	\item	Tính giá trị của biểu thức $A$ khi $x=25$. 
	\item 	Tìm giá trị của $x$ để $A=\dfrac{-1}{3}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $A=\dfrac{x+(\sqrt{x}+2)+(\sqrt{x}-2)}{x-4}=\dfrac{x+2\sqrt{x}}{(\sqrt{x}+2)(\sqrt{x}-2)}=\dfrac{\sqrt{x}}{\sqrt{x}-2}$.
\item Với $x=25\Rightarrow A=\dfrac{5}{3}$.
\item $A=-\dfrac{1}{3}\Leftrightarrow \dfrac{\sqrt{x}}{\sqrt{x}-2}=-\dfrac{1}{3}\Leftrightarrow 3\sqrt{x}=-\sqrt{x}+2\Leftrightarrow \sqrt{x}=\dfrac{1}{2}\Leftrightarrow x=\dfrac{1}{4}$ (thỏa mãn).
\end{enumerate}
}
\end{ex} 
 
\begin{ex}[9D4B8]
 \textit{Giải bài toán sau bằng cách lập phương trình hoặc hệ phương trình:}\\
	Hai tổ sản xuất cùng may một loại áo. Nếu tổ thứ nhất may trong 3 ngày, tổ thứ hai may trong 5 ngày thì cả hai tổ may được 1310 chiếc áo. Biết rằng trong một ngày tổ thứ nhất may được nhiều hơn tổ thứ hai là 10 chiếc áo, hỏi mỗi tổ trong một ngày may được bao nhiêu chiếc áo?
\loigiai{
Gọi số áo tổ $2$ may được trong một ngày là $x$ ($x\in \mathbb N^*$, áo).\\
Số áo tổ $1$ may được trong một ngày là $x+10$ (áo).\\
Trong $3$ ngày tổ thứ nhất may được $3(x+10)$ (áo).\\
Trong $5$ ngày tổ thứ hai may được $5x$ (áo).\\
Tổ $1$ may trong $3$ ngày và tổ $2$ may trong $5$ ngày được $1310$ chiếc áo nên ta có phương trình\\
$3(x+10)+5x=1310\Leftrightarrow 8x=1280\Leftrightarrow x=160$ (thỏa mãn).\\
Vậy mỗi ngày tổ $1$ may được $160+10=170$ chiếc áo.\\
Mỗi ngày tổ $2$ may được $160$ chiếc áo.
}
\end{ex}  
\begin{ex}[9D4B4][9D4B6]
Cho phương trình $x^2-2(m+1)x+m^2+2=0$ ($1$).
\begin{enumerate}
\item Giải phương trình khi $m=1$. 
\item Tìm giá trị của $m$ để phương trình có hai nghiệm phân biệt $x_1$, $x_2$ thỏa mãn $x_1^2+x_2^2=10$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Khi $m=1$ ta có phương trình $x^{2}-4x+3=0\Leftrightarrow \hoac{&x=1\\ &x=3.}$
\item Điều kiện để phương trình $(1)$ có hai nghiệm phân biệt là $\Delta'> 0$\\
$\Leftrightarrow (m+1)^{2}-(m^{2}+2)>0\Leftrightarrow 2m-1>0\Leftrightarrow m>\dfrac{1}{2}$.\\
Áp dụng Viet ta có $\heva{&x_{1}+x_{2}=2(m+1)\\ &x_{1}x_{2}=m^{2}+2}$.\\
Ta có $x_1^2+x_2^2=10\Leftrightarrow (x_{1}+x_{2})^{2}-2x_{1}x_{2}=10\Leftrightarrow 4(m+1)^{2}-2(m^{2}+2)=10$\\
$\Leftrightarrow m^{2}+4m-5=0\Leftrightarrow\hoac{&m=1\ (\mbox{thỏa mãn})\\ &m=-5\ (\mbox{loại}).}$\\
Vậy $m=1$ thỏa mãn yêu cầu bài toán.
\end{enumerate}
}
\end{ex}  

\begin{ex}[9H3B7][9H2B6][9H2B6][9H3K7]
	Cho đường tròn $(O; R)$ và điểm $A$ nằm bên ngoài đường tròn. Kẻ tiếp tuyến $AB$, $AC$ với đường tròn ($B$, $C$ là các tiếp điểm).
\begin{enumerate}
\item Chứng minh tứ giác $ABOC$ là tứ giác nội tiếp.
\item Gọi $E$ là giao điểm của $BC$ và $OA$. Chứng minh $BE$ vuông góc với $OA$ và $OE \cdot OA=R^2$.
\item	Trên cung nhỏ $BC$ lấy điểm $K$ bất kỳ ($K$ khác $B$ và $C$). Tiếp tuyến tại $K$ của đường tròn $(O; R)$ 	cắt $AB$, $AC$ theo thứ tự tại $P$ và $Q$. Chứng minh tam giác $APQ$ có diện tích không đổi khi $K$ chuyển động trên cung nhỏ $BC$.
\item Đường thẳng qua $O$ và vuông góc với $OA$ cắt $AB$, $AC$ theo thứ tự tại $M$, $N$. Chứng minh 	$PM+QN \geq MN$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[line cap=round,line join=round,>=stealth,x=1.0cm,y=1.0cm, scale=0.7, thick]
\tkzDefPoint(0,0){O}
\def\r{4}
\tkzDefShiftPoint[O](0:1.7*\r){A}
\tkzTangent[from with R=A](O,\r cm)\tkzGetPoints{B}{C}
\tkzDefLine[orthogonal=through O](O,A)\tkzGetPoint{o}
\tkzInterLL(A,B)(O,o)\tkzGetPoint{M}
\tkzInterLL(A,C)(O,o)\tkzGetPoint{N}
\coordinate (K) at ($(O)+(10:\r)$);
\tkzTangent[at=K](O)\tkzGetPoint{k}
\tkzInterLL(A,B)(K,k)\tkzGetPoint{P}
\tkzInterLL(A,C)(K,k)\tkzGetPoint{Q}
\tkzInterLL(B,C)(O,A)\tkzGetPoint{E}
\tkzDrawSegments(A,N A,M M,N B,C P,Q O,B O,C O,K O,A B,K O,P O,Q)
\tkzDrawCircle[R](O,\r cm)
\tkzDrawPoints[size=1](O)
\pgfresetboundingbox
\tkzLabelPoints[above](M,P,B)
\tkzLabelPoints[below](N,C,Q)
\tkzLabelPoints[left](O)
\tkzLabelPoints[right](A,K)
\tkzLabelPoints[below right](E)
\tkzMarkRightAngles(O,B,M O,C,N O,K,Q O,E,C A,O,M)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Vì $AB$, $AC$ là tiếp tuyến của đường tròn $(O)$ nên $AB\perp OB$, $AC\perp OC$\\
$\Rightarrow ABOC$ là tứ giác nội tiếp.
\item Vì $AB=AC$ (tính chất tiếp tuyến) và $OB=OC\Rightarrow OA$ là đường trung trực của $BC$\\
$\Rightarrow BC\perp OA$ tại $E$.\\
Xét tam giác $OBA$ có $\widehat{OBA}=90^{\circ}$ và đường cao $BE\Rightarrow OE \cdot OA=OB^{2}=R^2$.
\item Vì $PK$, $PB$ là các tiếp tuyến của $(O)$ cắt nhau tại $P$ nên $PK=PB$. Tương tự, $QK=QC$.\\
Ta có $AP+PQ+QA=AP+PB+AQ+QC=AB+AC$ (không đổi).
\item Vì $ABC$ là tam giác cân tại $A$ và $MN\parallel BC\Rightarrow AMN$ cũng là tam giác cân tại $A$\\
$\Rightarrow \widehat{AMN}=\widehat{ANB}\Rightarrow \widehat{MAN}+2\widehat{AMN}=180^{\circ}$.\\
Vì $ABOC$ là tứ giác nội tiếp nên $\widehat{BAC}+\widehat{BOC}=180^{\circ}$. Mà $\widehat{BOC}=2\widehat{POQ}\Rightarrow \widehat{AMN}=\widehat{POQ}$.\\
Xét tam giác $POM$ có $\widehat{PON}=\widehat{PMO}+\widehat{OPM}=\widehat{POQ}+\widehat{QON}\Rightarrow \widehat{QON}=\widehat{OPM}$.\\
Ta có $\Delta OMP\backsim \Delta QNO\ (\mbox{g-g})\Rightarrow \dfrac{OM}{QN}=\dfrac{PM}{ON}\Rightarrow PM\cdot QN=OM^{2}$.\\
Áp dụng bất đẳng thức Côsi, ta có $PM+QN\geq 2\sqrt{PM\cdot QN}=2OM=MN$.
\end{enumerate}
}
\end{ex}  
\begin{ex}[9D5G5]
Giải phương trình $\sqrt{x^2-\dfrac{1}{4}+\sqrt{x^2+x+\dfrac{1}{4}}}=\dfrac{1}{2}\left(2x^3+x^2+2x+1 \right)$.
\loigiai{
\begin{align*}
&\sqrt{x^2-\dfrac{1}{4}+\sqrt{x^2+x+\dfrac{1}{4}}}=\dfrac{1}{2}\left(2x^3+x^2+2x+1 \right)\\ 
\Rightarrow&\sqrt{x^{2}-\dfrac{1}{4}+\sqrt{\left(x+\dfrac{1}{2}\right)^{2}}}=\dfrac{1}{2}[x^{2}(2x+1)+(2x+1)]\\ 
\Rightarrow&\sqrt{\left(x-\dfrac{1}{2}\right)\left(x+\dfrac{1}{2}\right)+\left|x+\dfrac{1}{2}\right|}=\left(x+\dfrac{1}{2}\right)(x^{2}+1).
\end{align*}
Nhận xét: để phương trình có nghiệm thì $x\geq -\dfrac{1}{2}$. Khi đó ta có
\begin{align*}
&\sqrt{\left(x-\dfrac{1}{2}\right)\left(x+\dfrac{1}{2}\right)+\left(x+\dfrac{1}{2}\right)}=\left(x+\dfrac{1}{2}\right)(x^{2}+1)\\ 
\Rightarrow&\sqrt{\left(x+\dfrac{1}{2}\right)^{2}}=\left(x+\dfrac{1}{2}\right)(x^{2}+1)\\ 
\Rightarrow&\left(x+\dfrac{1}{2}\right)=\left(x+\dfrac{1}{2}\right)(x^{2}+1)\Rightarrow \left(x+\dfrac{1}{2}\right)x^{2}=0\Rightarrow \hoac{&x=0\\ &x=-\dfrac{1}{2}.}
\end{align*}
\begin{itemize}
\item Với $x=0$ thay vào phương trình ta thấy thỏa mãn.
\item Với $x=-\dfrac{1}{2}$ thay vào phương trình ta thấy thỏa mãn.
\end{itemize}
Vậy tập nghiệm của phương trình là $S=\left\{0;-\dfrac{1}{2}\right\}$.
}
\end{ex}
