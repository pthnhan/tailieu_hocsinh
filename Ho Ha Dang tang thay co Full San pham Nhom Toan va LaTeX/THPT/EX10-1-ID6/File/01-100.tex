\begin{ex}%[Phạm Quang Tú]%[Huỳnh Văn Quy]%[0D1K1-2]

		\begin{itemize}
			\item Mệnh đề ``Một năm có tối đa $52$ ngày chủ nhật'' là mệnh đề sai.\\
			Giả sử ngày mùng một tháng một là ngày chủ nhật.\\
			Một năm bình thường có $365$ ngày, tức là $52$ tuần và lẻ $1$ ngày. Ngày cuối cùng của năm đó sẽ là một ngày chủ nhật nữa nên sẽ có $53$ ngày chủ nhật.
			\item Mệnh đề ``Các số nguyên tố đều là số lẻ'' là mệnh đề sai vì số $2$ là số nguyên tố nhưng là số chẵn.
			\item Mệnh đề ``Giải thưởng lớn nhất của Toán học là giải Nobel'' là mệnh đề sai, vì các nguyên nhân sau:\\
			Thứ nhất, không có giải Nobel Toán học.\\

			Trong toán học, giải thưởng mang về tột đỉnh vinh quang, tức là giải thưởng có giá trị tinh thần cao nhất cho một nhà Toán học là giải thưởng Fields; còn giải thưởng có giá trị tiền thưởng cao nhất là giải Abel.
			\item Mệnh đề ``Có vô số số nguyên tố'' là mệnh đề đúng.\\
			Ta chứng mình điều này bằng phản chứng như sau:\\
			Giả sử chỉ có hữu hạn các số nguyên tố là $p_1$; $p_2$; $p_3$; \ldots; $p_n$ và giả sử $p_1<p_2<p_3<\ldots<p_n$. Xét tích $A=p_1\cdot p_2\cdot p_3\cdot\ldots \cdot p_{n+1}$.\\
			Rõ ràng $A>p_n$ nên $A$ là hợp số, do đó $A$ có ít nhất một ước nguyên tố $p$. Khi đó $p_1$; $p_2$; $p_3$; \ldots; $p_n$ là tất cả các số nguyên tố nên tồn tại $i\in\{1;2;\ldots;n\}$ sao cho $p=p_i$.\\
			Như vậy $A$ chia hết cho $p$; $(p_1\cdot p_2\cdot \ldots \cdot p_n)$ chia hết cho $p$ nên $1$ chia hết cho $p$, mâu thuẫn.\\
			Do đó, giả sử chỉ có hữu hạn số nguyên tố là sai. Vậy có vô hạn các số nguyên tố.
		\end{itemize}
	}
\end{ex}

\subsection*{Cấp độ Khó}
\begin{ex}%[Nguyễn Phúc Đức]%[081-100 Phan Thanh Tâm]%[0D1K1-4]
	Biết $A$ là mệnh đề sai, còn $B$ là mệnh đề đúng. Mệnh đề nào sau đây là đúng?
	\choice
	{$B\Rightarrow A$}
	{$B\Leftrightarrow A$}
	{$\overline{B}\Leftrightarrow\overline{A}$}
	{\True $B\Rightarrow\overline{A}$}
	\loigiai{
		Ta có $ A $ sai nên $\overline{A}$ đúng, suy ra $B\Rightarrow\overline{A}$ đúng.
	}
\end{ex}
\begin{ex}%[Nguyễn Phúc Đức]%[081-100 Phan Thanh Tâm]%[0D1G1-4]
	Biết $A$ là mệnh đề đúng, $B$ là mệnh đề sai, $C$ là mệnh đề đúng. Mệnh đề nào sau đây là \textbf{sai}?
	\choice
	{$A\Rightarrow C$}
	{$\left(\overline{B}\Rightarrow C\right)\Rightarrow A$}
	{\True $C\Rightarrow\left(A\Rightarrow B\right)$}
	{$C\Rightarrow\left(A\Rightarrow\overline{B}\right)$}
	\loigiai{
		Ta có $B$ sai nên $A\Rightarrow B$ sai, suy ra $C\Rightarrow \left(A\Rightarrow B\right)$ sai.}
\end{ex}
\begin{ex}%[Nguyễn Phúc Đức]%[081-100 Phan Thanh Tâm]%[0D1G1-4]
	Cho $A, B, C$ là ba mệnh đề đúng. Mệnh đề nào sau đây là đúng?
	\choice
	{$A\Rightarrow \left(B\Rightarrow \overline{C}\right)$}
	{$C\Rightarrow \overline{A}$}
	{$B\Rightarrow \left(\overline{A\Rightarrow C}\right)$}
	{\True $C\Rightarrow \left(A\Rightarrow B\right)$}
	\loigiai{
		Ta có $A, B, C$ đúng nên $A\Rightarrow B$ đúng, suy ra $C\Rightarrow \left(A\Rightarrow B\right)$ đúng.
	}
\end{ex}
\begin{ex}%[Nguyễn Phúc Đức]%[081-100 Phan Thanh Tâm]%[0D1K1-5]
	Mệnh đề nào sau đây có mệnh đề phủ định đúng? 
	\choice
	{\True ``$\forall n\in\mathbb{R}\colon 2n\geq n$''}
	{``$\forall x\in\mathbb{R}\colon x<x+1$''}
	{``$\exists x\in\mathbb{Q}\colon x^2=2$''}
	{``$\exists x\in\mathbb{R}\colon x^2-3x+1=0$''}
	\loigiai{
		Ta có mệnh đề ``$\forall n\in\mathbb{R}\colon 2n\geq n$'' sai, vì với $ n=-1 $, $ 2n=-2<-1=n $.\\
		Suy ra mệnh đề phủ định là mệnh đề đúng.
	}
\end{ex}
\begin{ex}%[Nguyễn Phúc Đức]%[081-100 Phan Thanh Tâm]%[0D1B1-1]
	Cho mệnh đề chứa biến $P\left(x\right)\colon$ ``$x^2-3x+2=0$''. Mệnh đề $P\left(x\right)$ \textbf{đúng} khi nào?  
	\choice
	{$x=0$}

	{$x=-1$}
	{$x=-2$}
	\loigiai{
		Ta có $ x^2-3x+2=0\Leftrightarrow x=1\vee x=2 $ suy ra $ P(1) $ là mệnh đề đúng.
	}
\end{ex}
