\begin{ex}%[TT, Đại Học Ngoại Thương - Hà Nội, 2018]%[2D1B4-1]%[Dương BùiĐức, 12EX-7]
Đồ thị hàm số nào sau đây có tiệm cận đứng?
\choice
{$y=\dfrac{x^{2}+3x+2}{x+1}$}
{\True $y=\dfrac{x}{x-1}$}
{$y=\dfrac{x^{4}}{x^{4}+1}$}
{$y=\sqrt{x^{2}-4}$}
\loigiai{
\begin{itemize}
\item Xét $y=\dfrac{x^{2}+3x+2}{x+1}$ có $\lim\limits_{x\to-1}\dfrac{x^{2}+3x+2}{x+1}=\lim\limits_{x\to-1}(x+2)=3\Rightarrow$ đồ thị hàm số không có tiệm cận đứng.
\item Xét $y=\dfrac{x}{x-1}$ có $\heva{&\lim\limits_{x\to 1^{+}}\dfrac{x}{x-1}=+\infty\\ &\lim\limits_{x\to 1^{-}}\dfrac{x}{x-1}=-\infty}\Rightarrow$ đồ thị hàm số có tiệm cận đứng $x=1$.
\item Xét $y=\dfrac{x^{4}}{x^{4}+1}$ có tập xác định $\mathbb R\Rightarrow$  đồ thị hàm số không có tiệm cận đứng.
\item Xét $y=\sqrt{x^{2}-4}$ xác định trên $(-\infty;-2]\cup [2;+\infty)$.\\
Có $\lim\limits_{x\to -2^{-}}\sqrt{x^{2}-4}=0$, $\lim\limits_{x\to 2^{+}}\sqrt{x^{2}-4}=0\Rightarrow$ đồ thị hàm số không có tiệm cận đứng.
\end{itemize}
}
\end{ex}

\end{ex}

Cho hàm số $f(x)=\dfrac{a}{(x+1)^3}+bx\mathrm{e}^x$. Tìm $a$ và $b$ biết rằng $f'(0)=-22$ và $\displaystyle\int\limits_0^1 f(x) \mathrm{\,d}x=5$.
\choice
{$a=-2, b=-8$}
{$a=2,b=8$}
{\True $a=8,b=2$}
{$a=-8,b=-2$}
\loigiai{
Ta có $f'(x)=-\dfrac{3a}{(x+1)^4} +b(x+1)\mathrm{e}^x$, suy ra $-3a+b=f'(0)=-22$. Lại có
\begin{align*}
5=\displaystyle\int\limits_0^1 f(x) \mathrm{\,d}x= \displaystyle\int\limits_0^1 \left[ \dfrac{a}{(x+1)^3}+bx\mathrm{e}^x\right] \mathrm{\,d}x=\left.\left[ -\dfrac{a}{2(x+1)^2} +b(x-1)\mathrm{e}^x \right]\right|^1_0=\dfrac{3a}{8}+b
\end{align*}
nên ta có hệ phương trình
\begin{align*}
\heva{&-3a+b=-22\\ &\dfrac{3a}{8}+b=5}\Rightarrow \heva{&a=8\\ &b=2.}
\end{align*}
}
\end{ex}
