\begin{ex}%[Đề KSCL, Số 2 An Nhơn, Bình Định 2018]%[Nguyễn Thị Kiều Ngân, dự án 12EX-10]%[2D1B1-3]
	Gọi $S$ là tập hợp các số nguyên $m$ để hàm số $y=\dfrac{x+2m-3}{x-3m+2}$ đồng biến trên khoảng $(-\infty;-14)$. Tính tổng $T$ của các phần tử trong $S$.
	\choice
	{$T=-5$}
	{$T=-6$}
	{$T=-9$}
	{\True $T=-10$}
	\loigiai{
		Tập xác định $\mathscr{D}=\mathbb{R}\setminus \{3m-2\}$.\\
		$y'=\dfrac{-5m+5}{(x-3m+2)^2}$.\\
		Hàm số $y=\dfrac{x+2m-3}{x-3m+2}$ đồng biến trên khoảng $(-\infty;-14)$ khi và chỉ khi
		\begin{eqnarray*}
		& &\heva{&y'>0, \forall x \in \mathscr{D}\\&3m-2 \notin (-\infty;-14)}\\
		& \Leftrightarrow & \heva{&-5m+5>0\\&3m-2 \ge -14}\\
		& \Leftrightarrow & \heva{&m<1\\&m \ge -4} \Leftrightarrow m \in \{-4;-3;-2;-1;0\}. 
		\end{eqnarray*}
	Vậy $T=-4-3-2-1=-10$.
	}
\end{ex}

	{\True $x=\sqrt{6}$}
	{$x=1$}
	{$x=2\sqrt{6}$}

		\tkzLabelPoints[below](B,H)

		\tkzMarkSegments[mark=|](A,M B,M)
		\tkzMarkSegments[mark=||](D,N N,C)
		\end{tikzpicture}
	}	
		\noindent
		Tính:
		\begin{itemize}
			\item $CM =\sqrt{AC^2-AM^2} =\sqrt{4-\dfrac{x^2}{4}} =\dfrac{\sqrt{16-x^2}}{2}$.
			\item $MN =\sqrt{CM^2-NC^2} =\sqrt{4-\dfrac{x^2}{4}-1} =\dfrac{\sqrt{12-x^2}}{2}$.
			\item $MN \cdot CD =DH \cdot CM
			\Rightarrow h=DH=\dfrac{MN \cdot CD}{CM} =\dfrac{\dfrac{\sqrt{12-x^2}}{2} \cdot 2}{\dfrac{\sqrt{16-x^2}}{2}}
			=\dfrac{2\sqrt{12-x^2}}{\sqrt{16-x^2}}$.
			\item $S=S_{ABC}=\dfrac{1}{2}\cdot AB \cdot CM =\dfrac{1}{2} \cdot x \cdot \dfrac{\sqrt{16-x^2}}{2} =\dfrac{x\cdot \sqrt{16-x^2}}{4}$.
			\item $V=\dfrac{1}{3} \cdot S \cdot h =\dfrac{1}{3}\cdot \dfrac{x\cdot \sqrt{16-x^2}}{4} \cdot \dfrac{2\sqrt{12-x^2}}{\sqrt{16-x^2}}
			=\dfrac{1}{6} \cdot x \cdot \sqrt{12-x^2}$.
		\end{itemize}

		\Leftrightarrow V \le 1.$$

		Vậy $V$ đạt giá trị lớn nhất khi $x=\sqrt{6}$.
	}

	Cho hàm số $f(x)$ có đạo hàm không âm trên $[0;1]$ thỏa mãn $\dfrac{[f(x)]^2[f'(x)]^2}{\mathrm{e}^{2x}} =1+[f(x)]^2$ và $f(x)>0$ với $\forall x \in [0;1]$, biết $f(0)=1$. Hãy chọn khẳng định đúng trong các khẳng định sau.
	\choice
	{\True $\dfrac{5}{2}<f(1)<3$}
	{$3<f(1)<\dfrac{7}{2}$}
	{$2<f(1)<\dfrac{5}{2}$}
	{$\dfrac{3}{2}<f(1)<2$}
	\loigiai{
		Ta có
		\begin{eqnarray*}
			& & \dfrac{[f(x)]^2[f'(x)]^2}{\mathrm{e}^{2x}} =1+[f(x)]^2\\
			& \Rightarrow & \dfrac{f(x)f'(x)}{\mathrm{e}^{x}} =\sqrt{1+[f(x)]^2}\\
			& \Rightarrow & \dfrac{f(x)f'(x)}{\sqrt{1+[f(x)]^2}} =\mathrm{e}^{x}\\
			& \Rightarrow & I =\displaystyle \int \limits_0^1 \dfrac{f(x)f'(x)}{\sqrt{1+[f(x)]^2}} \mathrm{\,d}x =\displaystyle \int \limits_0^1 \mathrm{e}^{x} \mathrm{\,d}x =\mathrm{e}-1.
		\end{eqnarray*}

		=\left. t \right|_{t_1}^{t_2} =\sqrt{1+[f(1)]^2} -\sqrt{2}$.\\
		Do đó $\sqrt{1+[f(1)]^2} -\sqrt{2} =\mathrm{e}-1 \Leftrightarrow f(1) =\sqrt{(e-1+\sqrt{2})^2-1} \simeq 2{,}96$.\\
		Vậy $\dfrac{5}{2}<f(1)<3$.
	}
\end{ex}

			\item Trường hợp $m=0$: Đồ thị $(C)$ cắt trục $Ox$ tại $3$ điểm và đồ thị $(C)$ có $2$ điểm cực trị nằm phía dưới trục $Ox$, không có điểm cực trị nào nằm phía trên trục $Ox$ nên số điểm cực trị của đồ thị hàm số $y=\left| {x^4+x^3-\dfrac{1}{2}x^2+m} \right|$ là $3+2=5$ (nhận).
			\item Trường hợp $-5 \le m <0$, $m$ nguyên: Đồ thị $(C)$ cắt trục $Ox$ tại $2$ điểm và đồ thị $(C)$ có $3$ điểm cực trị nằm phía dưới trục $Ox$, không có điểm cực trị nào nằm phía trên trục $Ox$ nên số điểm cực trị của đồ thị hàm số $y=\left| {x^4+x^3-\dfrac{1}{2}x^2+m} \right|$ là $2+3=5$ (nhận). 
		\end{itemize}
		Do đó $m \in \{-5;-4;-3;-2;-1;0\}$.\\
		Vậy có $6$ giá trị của $m$ thỏa đề.	
	}
\end{ex}
