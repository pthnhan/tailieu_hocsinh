\begin{ex}%[Giữa kì 1 - Phòng GD\&ĐT TP Ninh Bình - 2018]%[9D1Y1]%[9D1B6]
Thực hiện các phép tính
\begin{enumerate}[a)]
  \item $A=3\sqrt{2}+5\sqrt{8}-2\sqrt{50}$;
  \item $B=\dfrac{1}{3+\sqrt{5}}+\dfrac{1}{3-\sqrt{5}}$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
  \item $A=3\sqrt{2}+5\sqrt{2^2.2}-2\sqrt{5^2.2}=3\sqrt{2}+10\sqrt{2}-10\sqrt{2}=3\sqrt{2}.$
  \item $B=\dfrac{3-\sqrt{5}+3+\sqrt{5}}{\left(3+\sqrt{5}\right)\left(3-\sqrt{5}\right)}=\dfrac{6}{9-5}=\dfrac{6}{4}=\dfrac{3}{2}.$
\end{enumerate}
}
\end{ex}

\begin{ex}%[Giữa kì 1 - Phòng GD\&ĐT TP Ninh Bình - 2018]%[9D1Y1]%[9D1B8]
Cho biểu thức $Q=\left(\dfrac{\sqrt{x}}{1-\sqrt{x}}+\dfrac{\sqrt{x}}{1+\sqrt{x}}\right)+\dfrac{3-\sqrt{x}}{x-1}$
\begin{enumerate}[a)]
  \item Tìm điều kiện xác định của $Q$.
  \item Rút gọn $Q$.
  \item Tìm $x$ để $Q=-1$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
  \item Điều kiện xác định $x\geq 0, x\neq 1$.
  \item Ta có \begin{align*}
                Q&=\left(\dfrac{\sqrt{x}}{1-\sqrt{x}}+\dfrac{\sqrt{x}}{1+\sqrt{x}}\right)+\dfrac{3-\sqrt{x}}{x-1} \\
                &=\dfrac{\sqrt{x}\left(1+\sqrt{x}\right)+\sqrt{x}\left(1-\sqrt{x}\right)}{\left(1+\sqrt{x}\right)\left(1-\sqrt{x}\right)}+\dfrac{3-\sqrt{x}}{x-1}\\
                &=\dfrac{2\sqrt{x}}{1-x}+\dfrac{3-\sqrt{x}}{x-1}\\
                &=\dfrac{3-3\sqrt{x}}{x-1}\\
                &=\dfrac{-3}{\sqrt{x}+1}.
                     \end{align*}
  \item Ta có $Q=-1 \Leftrightarrow  \dfrac{-3}{\sqrt{x}+1} =-1 \Leftrightarrow \sqrt{x}+1 =3 \Leftrightarrow \sqrt{x}=2 \Leftrightarrow x=4$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Giữa kì 1 - Phòng GD\&ĐT TP Ninh Bình - 2018]%[9D1B2]%[9D1B4]%[9D1K4]
Cho $\Delta ABC$ vuông tại $A$, có $AB = 6\mathrm{cm}, AC = 8\mathrm{cm}$.
\begin{enumerate}[a)]
  \item Tính số đo góc $B$, góc $C$ (làm tròn đến độ) và đường cao $AH$.
  \item Chứng minh $AB.\cos B+AC.\cos C=BC$.
  \item Trên cạnh $AC$ lấy điểm $D$ sao cho $DC = 2DA$. Vẽ $DE$ vuông góc với $BC$ tại $E$. Chứng minh rằng: $\dfrac{1}{AB^2}+\dfrac{1}{AC^2}=\dfrac{4}{9DE^2}.$
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzInit[xmin=-0.5,ymin=-0.2,xmax=4.5,ymax=4.5]
\tkzClip
\tkzDefPoints{0/0/A, 3/0/B, 0/4/C}
\coordinate (D) at ($(A)!0.33!(C)$);
\tkzDefPointBy[projection= onto B--C](A)
\tkzGetPoint{H}
\tkzDefPointBy[projection= onto B--C](D)
\tkzGetPoint{E}
\tkzDrawSegments(A,B A,C B,C A,H D,E)
\tkzLabelPoints[left](C,A,D)
\tkzLabelPoints[right](B,E,H)
\tkzMarkRightAngle(B,A,C)
\tkzMarkRightAngle(A,H,B)
\tkzMarkRightAngle(D,E,B)
\end{tikzpicture}
\end{center}
\begin{enumerate}[a)]
  \item Từ giả thiết ta có $BC^2=AB^2+AC^2=6^2+8^2=100 \Rightarrow BC=10$ (cm).\\
  Ta có $\sin B=\dfrac{AC}{AB}=\dfrac{4}{5} \Rightarrow \widehat{B}\approx 53^{\circ}$, do đó $\widehat{C}\approx 37^{\circ}$.\\
  Vì $AH.BC=AB.AC \Rightarrow AH=\dfrac{AB.AC}{BC}=\dfrac{6.8}{10}=4,8$ (cm).
  \item $\Delta HAB$ vuông tại $H$ nên $BH=AB.\cos B$.\\
  Tương tự, do $\Delta HCA$ vuông tại $H$ nên $CH=AC.\cos C$.\\
  Từ đó suy ra $BC=BH+CH=AB.\cos B+AC.\cos C$.
  \item $\Delta ABC$ vuông tại $A$ và có đơngf cao $AH$ nên $\dfrac{1}{AB^2}+\dfrac{1}{AC^2}=\dfrac{1}{AH^2}$. \hfill (1)\\
  Do $DE\parallel AH$ nên $\dfrac{DE}{AH}=\dfrac{CD}{AC}=\dfrac{2}{3}$.\\
  Từ đó suy ra $\dfrac{DE^2}{AH^2}=\dfrac{4}{9} \Rightarrow \dfrac{1}{AH^2}=\dfrac{4}{9 DE^2}$. \hfill (2)\\
  Từ (1) và (2) suy ra $\dfrac{1}{AB^2}+\dfrac{1}{AC^2}=\dfrac{4}{9DE^2}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Giữa kì 1 - Phòng GD\&ĐT TP Ninh Bình - 2018]%[9D1G7]
Tìm giá trị lớn nhất của $A=\dfrac{1}{x-4\sqrt{x-4}+3}$. Giá trị đó đạt được khi $x$ bằng bao nhiêu?
\loigiai{
$$A=\dfrac{1}{x-4\sqrt{x-4}+3}=\dfrac{1}{x-4-4\sqrt{x-4}+4+3}=\dfrac{1}{\left(\sqrt{x-4}-2\right)^2+3}\leq \dfrac{1}{3}, \ \forall x\geq 4.$$
Dấu bằng xảy ra khi và chỉ khi $x=8$.\\
Vậy giá trị lớn nhất của $A$ là $\dfrac{1}{3}$, đạt được khi $x=8$.
}
\end{ex}
