\begin{ex}%[Đề thi vào 10, Sở giáo dục Ninh Thuận, 2017]%[8D4Y3]%[8D3Y3]%[9D5Y3]
	Giải bất phương trình và các phương trình sau:
	\begin{enumerate}
		\item $4x-5>7$ 
		\item $2x+3(4x+2)=8$ 
		\item $\dfrac{1}{2}x^{2}=3x-4$ 
	\end{enumerate}
   	\loigiai{
   	\begin{enumerate}
   		\item Ta có $4x-5>7\Rightarrow$ $4x>12\Rightarrow$ $x>3$ 
   		\item $2x+3(4x+2)=8 \Leftrightarrow$ $2x+12x+6=8\Leftrightarrow$ $14x=2\Leftrightarrow$ $x=\dfrac{1}{7}$  \item $\dfrac{1}{2}x^{2}=3x-4\Leftrightarrow$ $x^{2}-6x+8=0$\\
   		Ta có: $\Delta'=1$. Vậy phương trình có hai nghiệm phân biệt $x_1=4$, $x_2=2$.        
   \end{enumerate}}
\end{ex} 

\begin{ex}%[Đề thi vào 10, Sở giáo dục Ninh Thuận, 2017]%[9D1K8]
	Rút gọn biểu thức:
	\begin{center}
		$A=\left( \dfrac{\sqrt{a}+1}{\sqrt{a}-1}-\dfrac{\sqrt{a}-1}{\sqrt{a}+1} \right).\dfrac{a-1}{4(a+1)}$, với $a\geq 0$; $a\neq 1$.
		   
	\end{center}
	  $\begin{array}{ll}
	  	$Với $ a\geq 0; a\neq 1,$ ta có:$\\
	  	A=\left( \dfrac{\sqrt{a}+1}{\sqrt{a}-1}-\dfrac{\sqrt{a}-1}{\sqrt{a}+1} \right).\dfrac{a-1}{4(a+1)}&=\dfrac{(\sqrt{a}+1)^{2}-(\sqrt{a}-1)^{2}}{(\sqrt{a}-1)(\sqrt{a}+1)}.\dfrac{a-1}{4(a+1)}\\
	  	 &=\dfrac{4\sqrt{a}}{a-1}.\dfrac{a-1}{4(a+1)}=\dfrac{\sqrt{a}}{a+1}
	  \end{array}$
\end{ex} 
\begin{ex}%[Đề thi vào 10, Sở giáo dục Ninh Thuận, 2017]%[9D4K6]
Áp dụng hệ thức Vi-ét để tìm hai số, biết tổng của chúng bằng $15$ và tích của chúng bằng $56$.
\loigiai{\\
Gọi hai số cần tìm là $x,y$. Ta có $\begin{cases}
S=x+y&=15\\
P=xy&=56	
\end{cases}$. Theo định lý Vi-ét, $x,y$ là hai nghiệm của phương trình $X^{2}-15X+56=0\Leftrightarrow$ $\left[\begin{array}{l}
X=7\\X=8
\end{array}\right.$. Vậy hai số cần tìm là $\begin{cases}
x=7\\y=8
\end{cases}$ hoặc $\begin{cases}
x=8\\y=7
\end{cases}$} 
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Ninh Thuận, 2017]%[9D4K3]
	Cho hai số thực $x,y$ thoả mãn điều kiện $x-y=2$. Tìm giá trị nhỏ nhất của biểu thức: $Q=3x^{2}+y^{2}+8$.
	\loigiai{
	Với $x-y=2$ ta có $y=x-2$. Thay vào $Q$ ta được:
	\begin{center}
	$Q=3x^{2}+(x-2)^{2}+8=4x^{2}-4x+12=(2x-1)^{2}+11\geq 11$	
\end{center}
Vậy giá trị nhỏ nhất của biểu thức $Q$ là $11$ khi $x=\dfrac{1}{2}$ và $y=-\dfrac{3}{2}$.      }   
	
\end{ex}
