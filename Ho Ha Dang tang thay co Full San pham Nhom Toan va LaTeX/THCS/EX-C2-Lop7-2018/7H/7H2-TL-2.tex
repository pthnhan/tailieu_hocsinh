\section{HAI TAM GIÁC BẰNG NHAU}
\subsection{Tóm tắt lí thuyết}

\begin{dn}
Hai tam giác bằng nhau là hai tam giác có các cạnh tương ứng bằng nhau, các góc tương ứng bằng nhau.\\
Nếu có $\triangle ABC$ bằng $\triangle A_1B_1C_1$ ta kí hiệu $\triangle ABC=\triangle A_1B_1C_1$.\\
Như vậy $\triangle ABC=\triangle A_1B_1C_1\Leftrightarrow\heva{& AB=A_1B_1,\, BC=B_1C_1,\, CA=C_1A_1\\& \widehat{A_1}=\widehat{A},\,\widehat{B_1}=\widehat{B},\,\widehat{C_1}=\widehat{C}.}$
\end{dn}

\begin{note}
Khi viết $\triangle ABC=\triangle A_1B_1C_1$ chúng ta cần hiểu ở đó có sự tương ứng giữa các đỉnh của hai tam giác với nhau, tức là không thể viết lại kí hiệu trên dưới dạng $\triangle ABC=\triangle B_1A_1C_1$ và nếu muốn đảo đỉnh thì cần đảo cả hai vế của dấu bằng $\triangle BAC=\triangle B_1A_1C_1$.
\end{note}

\subsection{Các dạng toán}

\begin{vd}%[Vương Quyền, dự án (EX-C2-L7-2018)]%[7H2Y2]
Hai tam giác trong các hình sau có bằng nhau không? Nếu có, hãy viết kí hiệu sự bằng nhau của hai tam giác đó.
\begin{center}
\begin{tikzpicture}
\tkzInit[xmin=-1,ymin=-2,xmax=8,ymax=3]
\tkzDefPoints{0/0/B, 3/0/C, 1/2/A, 4/-1/O}
\tkzDefPointBy[rotation = center O angle -90](B)\tkzGetPoint{N}
\tkzDefPointBy[rotation = center O angle -90](A)\tkzGetPoint{M}
\tkzDefPointBy[rotation = center O angle -90](C)\tkzGetPoint{P}
\tkzDrawSegments(A,B B,C C,A M,N N,P P,M)
\tkzDrawPoints(A,B,C,M,N,P)
\tkzLabelPoints[above](A,M,N)
\tkzLabelPoints[below](B,C,P)
\tkzMarkSegments[mark=|](A,C M,P)
\tkzMarkSegments[mark=||](B,C N,P)
\tkzMarkSegments[mark=x](A,B M,N)
\tkzLabelAngles[pos=0.5](B,A,C){$60^\circ$}
\tkzLabelAngles[pos=-0.5](N,M,P){$60^\circ$}
\tkzLabelAngles[pos=0.6](C,B,A){$70^\circ$}
\tkzLabelAngles[pos=0.8](M,P,N){$50^\circ$}
\end{tikzpicture}
\end{center}
\loigiai{
Theo hình vẽ
\begin{itemize}
\item Trong $\triangle ABC$, ta suy ra $\widehat{C}=180^\circ-\widehat{A}-\widehat{B}=180^\circ-60^\circ-70^\circ=50^\circ$.
\item Trong $\triangle MNP$ ta suy ra $\widehat{N}=180^\circ-\widehat{M}-\widehat{P}=180^\circ-60^\circ-50^\circ=70^\circ$.
\end{itemize}
Khi đó, với $\triangle ABC$ và $\triangle MNP$ ta có
\begin{itemize}
\item $AB=MN$, $BC=NP$, $CA=PM$ các cạnh bằng nhau.
\item $\widehat{A}=\widehat{M}$, $\widehat{B}=\widehat{N}$, $\widehat{C}=\widehat{P}$ các góc bằng nhau
\end{itemize}
Suy ra $\triangle ABC=\triangle MNP$.
}
\end{vd}

\begin{nx}
Qua ví dụ trên, ta thấy ngay được việc có thể giảm bớt một yếu tố giả thiết về góc mà hai tam giác vẫn bằng nhau. Điều này chính là lí do để chúng ta cần thiết đi xem xét ``Có bao nhiêu trường hợp bằng nhau của hai tam giác''.
\end{nx}

\begin{vd}%[Vương Quyền, dự án (EX-C2-L7-2018)]%[7H2B2]
Cho $\triangle ABC=\triangle HIK$.
\begin{enumerate}
\item Tìm cạnh tương ứng với cạnh $BC$. Tìm góc tương ứng với góc $H$.
\item Tìm các cạnh bằng nhau, tìm các góc bằng nhau.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có 
\begin{itemize}
\item Cạnh tương ứng với cạnh $BC$ là cạnh $IK$.
\item Góc tương ứng với góc $H$ là góc $A$.
\end{itemize}
\item Vì $\triangle ABC=\triangle HIK$ suy ra $AB=HI$, $BC=IK$, $AC=HK$, $\widehat{A}=\widehat{H}$, $\widehat{B}=\widehat{I}$, $\widehat{C}=\widehat{K}$.
\end{enumerate}
}
\end{vd}


\begin{vd}%[Vương Quyền, dự án (EX-C2-L7-2018)]%[7H2B2]
Cho $\triangle ABC=\triangle HIK$ trong đó $AB=2$ cm, $\widehat{B}=40^\circ$, $BC=4$ cm. Em có thể suy ra được số đo của những cạnh nào, những góc nào của $\triangle HIK$.
\loigiai{
Từ tính tương ứng ta có ngay $HI=AB=2$ cm, $\widehat{I}=\widehat{B}=40^\circ$, $IK=BC=4$ cm.
}
\end{vd}

\begin{vd}%[Vương Quyền, dự án (EX-C2-L7-2018)]%[7H2B2]
Cho $\triangle ABC=\triangle MNP$.
\begin{enumerate}
\item Viết đẳng thức trên dưới một dạng khác.
\item Biết $AB=4$ cm, $BC=6$ cm và $MP=5$ cm. Tính chu vi của mỗi tam giác nói trên.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có thể viết $\triangle ACB=\triangle MPN$ hoặc $\triangle BCA=\triangle NPM$.
\item Từ giả thiết $\triangle ABC=\triangle MNP$\\
Suy ra $MN=AB=4$ cm, $NP=BC=6$ cm và $MP=AC=5$ cm.\\
Khi đó chu vi của tam giác $\triangle ABC$ và $\triangle MNP$ lần lượt là
\begin{itemize}
\item $C_{\triangle ABC}=AB+BC+CA=4+6+5=15$ cm.
\item $C_{\triangle MNP}=MN+NP+PM=4+6+5=15$ cm.
\end{itemize}
\end{enumerate}
}
\end{vd}

\begin{nx}
Qua ví dụ trên, ta thấy ngay ``Hai tam giác bằng nhau sẽ có chu vi bằng nhau''.
\end{nx}

\begin{vd}%[Vương Quyền, dự án (EX-C2-L7-2018)]%[7H2K2]
Cho hình vẽ, chứng tỏ rằng hai tam giác $\triangle ABC$ và $\triangle ADC$ 
\immini{
bằng nhau.
}{
\begin{tikzpicture}
\tkzDefPoints{0/0/D, 3/0/C, 1/2/A}
\coordinate (B) at ($(A)+(C)-(D)$);
\tkzDrawSegments(A,B B,C C,D D,A A,C)
\tkzDrawPoints(A,B,C,D)
\tkzLabelPoints[below](D,C)
\tkzLabelPoints[above](A,B)
\tkzMarkSegments[mark=||](D,A C,B)
\tkzMarkSegments[mark=|](D,C A,B)
\tkzLabelAngles[pos=0.7](C,A,B A,C,D){$30^\circ$}
\tkzLabelAngles[pos=0.6](B,C,A){$80^\circ$}
\tkzLabelAngles[pos=0.7](C,D,A){$70^\circ$}
\end{tikzpicture}
}
\loigiai{
Theo hình vẽ, ta có
\begin{itemize}
\item Trong $\triangle ABC$ suy ra $\widehat{B}=180^\circ-\widehat{BAC}-\widehat{ACB}=180^\circ-30^\circ-80^\circ=70^\circ$.
\item Trong $\triangle ADC$, suy ra $\widehat{DAC}=180^\circ-\widehat{ACD}-\widehat{D}=180^\circ-30^\circ-70^\circ=80^\circ$.
\end{itemize}
Khi đó với $\triangle ABC$ và $\triangle ADC$, ta có
\begin{itemize}
\item $AB=CD$, $BC=DA$, $AC$ chung (các cạnh bằng nhau).
\item $\widehat{CAB}=\widehat{ACD}$, $\widehat{B}=\widehat{D}$, $\widehat{ACB}=\widehat{CAD}$ (các góc bằng nhau).
\end{itemize}
Suy ra $\triangle ABC=\triangle CDA$.
}
\end{vd}

\begin{vd}%[Vương Quyền, dự án (EX-C2-L7-2018)]%[7H2K2]
Cho $\triangle ABC$ có $\widehat{A}=80^\circ$. Tính số đo của các góc $\widehat{B}$, $\widehat{C}$ biết rằng tam giác $\triangle ABC=\triangle ACB$.
\loigiai{
Từ giả thiết $\triangle ABC=\triangle ACB\Rightarrow \widehat{B}=\widehat{C}$ (dựa trên sự tương ứng đỉnh).\\
Mặt khác, ta có $\widehat{A}+\widehat{B}+\widehat{C}=180^\circ\Leftrightarrow 80^\circ+\widehat{B}+\widehat{B}=180^\circ\Leftrightarrow\widehat{B}=50^\circ$.\\
Vậy $\triangle ABC$ có $\widehat{B}=\widehat{C}=50^\circ$.
}
\end{vd}

\begin{vd}%[Vương Quyền, dự án (EX-C2-L7-2018)]%[7H2B2]
Cho $\triangle ABC=\triangle MNP$. Biết có $\widehat{A}=80^\circ$, $\widehat{N}=75^\circ$. Tính số đo các góc còn lại của tam giác.
\loigiai{
Từ giả thiết $\triangle ABC=\triangle MNP$ suy ra
\begin{itemize}
\item $\widehat{A}=\widehat{M}=80^\circ$.
\item $\widehat{B}=\widehat{N}=75^\circ$.
\item $\widehat{C}=\widehat{P}=180^\circ-\widehat{M}-\widehat{N}=180^\circ-80^\circ-75^\circ=25^\circ$.
\end{itemize}
}
\end{vd}

\begin{center}
\begin{tikzpicture}

\loigiai{

\begin{bt}%[Vương Quyền, dự án (EX-C2-L7-2018)]%[7H2B2]
Hãy chỉ ra các tam giác bằng nhau trong hình sau. Hãy kí hiệu sự  bằng nhau của hai tam giác đó.
\begin{center}
\begin{tikzpicture}
\tkzDefPoints{0/0/D, 3/0/C, 1/2/A}
\coordinate (B) at ($(A)+(C)-(D)$);
\tkzInterLL(A,C)(D,B)\tkzGetPoint{O}
\tkzDrawSegments(A,B B,C C,D D,A A,C D,B)
\tkzDrawPoints(A,B,C,D,O)

\tkzLabelPoints[above](A,B)
\tkzLabelPoints[left](O)
\tkzMarkSegments[mark=||](D,A C,B)
\tkzMarkSegments[mark=|](D,C A,B)
\tkzLabelAngles[pos=0.7](C,A,B A,C,D){$30^\circ$}
\tkzLabelAngles[pos=0.6](B,C,A){$85^\circ$}
\tkzLabelAngles[pos=0.7](C,D,A){$65^\circ$}
\end{tikzpicture}
\end{center}
\loigiai{
Ta có $\triangle ABC=\triangle CDA$ vì $\heva{& AB=DC,AD=BC,AC\;\text{chung}\\& \widehat{CAB}=\widehat{ACD},\widehat{ADC}=\widehat{ABC}.}$\\
Tương tự $\triangle BAD=\triangle DCB$ vì $\heva{& AB=DC,AD=BC,AC\quad\text{chung}\\&\widehat{DAB}=\widehat{BCD},\widehat{ABD}=\widehat{CDB}.}$
}
\end{bt}

\begin{bt}%[Vương Quyền, dự án (EX-C2-L7-2018)]%[7H2B2]
Tính các góc của $\triangle ABC$ biết $\triangle ABC=\triangle ACB=\triangle BCA$.
\loigiai{
Từ giả thiết 
\begin{itemize}
\item $\triangle ABC=\triangle ACB\Rightarrow \widehat{B}=\widehat{C}$ (dựa trên sự tương ứng đỉnh).
\item $\triangle ACB=\triangle BCA\Rightarrow \widehat{A}=\widehat{B}$ (dựa trên sự tương ứng đỉnh).
\end{itemize}
Mặt khác ta có $\widehat{A}+\widehat{B}+\widehat{C}=180^\circ\Leftrightarrow\widehat{A}+\widehat{A}+\widehat{A}=180^\circ\Leftrightarrow \widehat{A}=60^\circ$.\\
Vậy $\triangle ABC$ có $\widehat{A}=\widehat{B}=\widehat{C}=60^\circ$.
}
\end{bt}

\begin{bt}%[Vương Quyền, dự án (EX-C2-L7-2018)]%[7H2B2]
Cho $\triangle ABC=\triangle MNP$. Biết $\widehat{A}=30^\circ$, $\widehat{N}=60^\circ$. Tính số đo các góc còn lại của mỗi tam giác đó.
\loigiai{
Từ giả thiết $\triangle ABC=\triangle MNP$ suy ra
\begin{itemize}
\item $\widehat{A}=\widehat{M}=30^\circ$.
\item $\widehat{B}=\widehat{N}=60^\circ$.
\item $\widehat{C}=\widehat{P}=180^\circ-\widehat{M}-\widehat{N}=90^\circ$.
\end{itemize}
}
\end{bt}

\begin{bt}%[Vương Quyền, dự án (EX-C2-L7-2018)]%[7H2K2]
Cho $\triangle ABC=\triangle MNP$.
\begin{enumerate}
\item Viết đẳng thức trên dưới một dạng khác.
\item Biết $AB=3$ cm, $BC=4$ cm và $MP=5$ cm. Tính chu vi của mỗi tam giác nói trên.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có thể viết $\triangle ACB=\triangle MPN$ hoặc $\triangle BCA=\triangle NPM$.
\item Từ giả thiết $\triangle ABC=\triangle MNP$\\
Suy ra $MN=AB=3$ cm, $NP=BC=4$ cm và $MP=AC=5$ cm.\\
Khi đó, chu vi của tam giác $ABC$ và tam giác $MNP$ là 
\begin{itemize}
\item $C_{\triangle ABC}=AB+BC+CA=12$ cm.
\item $C_{\triangle MNP}=MN+NP+PM=12$ cm.
\end{itemize}
\end{enumerate}
}
\end{bt}

\begin{bt}%[Vương Quyền, dự án (EX-C2-L7-2018)]%[7H2B2]
Cho $\triangle ABC=\triangle ADC$.
\begin{enumerate}
\item Viết đẳng thức trên dưới một dạng khác và thử vẽ hình minh họa.
\item Biết $AB=6$ cm, $DC=8$ cm và $AC=10$ cm. Tính chu vi của mỗi tam giác nói trên.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có thể viết $\triangle ACB=\triangle ACD$ hoặc $\triangle BCA=\triangle DCA$.
\immini{
Khi đó, ta có thể minh họa theo hình vẽ
}{
\begin{tikzpicture}
\tkzDefPoints{0/0/A, 4/0/C, 1/2/B, 1/-2/D}
\tkzDrawSegments(A,B B,C C,D D,A A,C)
\tkzDrawPoints(A,B,C,D)
\tkzLabelPoints[above](B)
\tkzLabelPoints[below](D,C)
\tkzLabelPoints[left](A)
\end{tikzpicture}
}
\item Từ giả thiết $\triangle ABC=\triangle ADC\Rightarrow AB=AD=6$ cm, $BC=DC=8$ cm, $AC=10$ cm.\\
Khi đó chu vi của tam giác $\triangle ABC=\triangle ADC$ là
$$C_{\triangle ADC}=C_{\triangle ABC}=AB+BC+CA=6+8+10=24\,\mathrm{cm}.$$
\end{enumerate}
}
\end{bt}
