\begin{bt}[3 điểm]
	Giải các phương trình  sau:
	\begin{enumEX}{2}
		\item $ x^2-3x+2=0. $
		\item $ 2x^2-3x-2=0. $
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item $ a=1,b=-3,c=2 $ nên $ a+b+c=0 $.\\
			 Vậy phương trình có hai nghiệm $ S=\{1,2\}. $
			\item $ \Delta=(-3)^2-4.2.(-2)=25 >0$.\\ Vậy phương trình có hai nghiệm $ x=\dfrac{3+\sqrt{25}}{2.2}=2 $ hoặc $ x= \dfrac{3-\sqrt{25}}{2.2}=-\dfrac{1}{2}$.
		\end{enumEX}
	}
\end{bt}

\begin{bt}[2 điểm]
	Giải các phương trình sau:
	\begin{enumEX}{2}
		\item $ x^4-2x^2-8=0 $.
		\item $ x+\sqrt{x}-6=0 $.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Đặt $ t=x^2, t\ge 0 $. Phương trình trở thành $ t^2-2t-8=0\Leftrightarrow \hoac{&t=4\\&t=-2\text{(loại)}.} $\\
			Với $ t=4\Leftrightarrow x^2=4\Leftrightarrow x=\pm 2 $.\\
			Vậy phương trình có hai nghiệm $ S=\{-2,2\} $.
			\item Đặt $ t=\sqrt{x}, t\ge 0 $. Phương trình trở thành $ t^2+t-6=0\Leftrightarrow \hoac{&t=2\\&t=-3\text{(loại)}.}$\\
			Với $ t=2\Leftrightarrow \sqrt{x}=2\Leftrightarrow x=4 $.\\
			Vậy phương trình có hai nghiệm $ S=\{4\} $.
		\end{enumEX}
		
	}
\end{bt}

\begin{bt}[2 điểm]
	Cho phương trình $ 7x^2+2(m-1)x-m^2=0. \hfill (1)$
	\begin{enumEX}{1}
		\item Tìm $ m $ để phương trình $ (1) $ luôn có nghiệm.
		\item Gọi $ x_1 ,\; x_2$  là hai nghiệm của phương trình $ (1) $. Tìm $ m $ thỏa mãn hệ thức $ x_1x_2=-7 $.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Ta có $ \Delta'=(1-m)^2+7m^2 $. Ta có $ (1-m)^2\ge 0 $ và $ 7m^2\ge 0 $ nên $ \Delta'\ge 0,\forall \;m\in \mathbb{R} $. \\
			Vậy phương trình $ (1) $ luôn có nghiệm.
			\item Theo hệ thức Vi-et, ta có $ x_1x_2=-\dfrac{m^2}{7} $. \\Khi đó, $x_1x_2=-7\Leftrightarrow -\dfrac{m^2}{7}=-7\Leftrightarrow m^2=49\Leftrightarrow \hoac{&m=7\\&m=-7}  $.\\ Vậy $ m=-7 $ hoặc $ m=7 $ là giá trị cần tìm.
		\end{enumEX}	
	}
\end{bt}

\begin{bt}[2 điểm]
	Tính chiều dài và chiều rộng của một  thửa ruộng hình chữ nhật. Biết chiều dài hơn chiều rộng $ 10 $ m và diện tích bằng $ 1200$  m$^2$.
	\loigiai{
		Gọi chiều rộng của thửa ruộng hình chữ nhật là $ x\; (x>0 $, m).\\
		Chiều dài hơn chiều rộng $ 10 $m nên chiều dài là $ x+10 $ (m).\\
		Diện tích thửa ruộng hình chữ nhật là 1200m$ ^2 $ nên ta có phương trình:\begin{align*}
		 &x(x+10)=1200\\
		 \Leftrightarrow &x^2+10x-1200=0\\
		 \Leftrightarrow &\hoac{&x=30\\&x=-40\text{(loại)}}.
		\end{align*} 
		Vậy chiều dài  thửa ruộng hình chữ nhật là $ 40$ m và chiều rộng thửa ruộng hình chữ nhật là $ 30 $ m.
		
	}
\end{bt}

\begin{bt}[1 điểm]
	Gọi $ x_1,\; x_2 $ là hai nghiệm của phương trình $ x^2+x-1=0. $ Lập phương trình bậc hai nhận $ 3x_1 $ và $ 3x_2 $ làm nghiệm.
	\loigiai{
		Đặt $ S=3x_1+3x_2=3(x_1+x_2) ,\; P=3x_13x_2=9x_1x_2$.\\
		Vì $ x_1, \;x_2 $ là nghiệm của phương trình $ x^2+x-1=0 $ nên ta có hệ thức Vi-et: $ \heva{&x_1+x_2=-1\\&x_1x_2=-1.} $\\
		Như vậy, $ \heva{&S=3(x_1+x_2) =-3\\&P=9x_1x_2=-9.} $ \\
		Do đó: $ 3x_1,\; 3x_2 $ là hai nghiệm của phương trình $ x^2-Sx+P=0\Leftrightarrow x^2+3x-9=0. $
	}
\end{bt}

\begin{bt}[2 điểm]
	Tìm $ m $ để phương trình $ x^2-3x+2=0 $  tương đương với phương trình $ 2x^2+mx+4=0. $
	\loigiai{
		Ta có $ x^2-3x+2=0\Leftrightarrow \hoac{x=1\\x=2.} $\\
		Vậy  phương trình $ x^2-3x+2=0$ tương đương  phương trình $ 2x^2+mx+4=0 $	khi và chỉ khi phương trình $ 2x^2+mx+4=0 $ có tập nghiệm là $ S=\{1;2\} $.\\
		Với $ x=1 $ hoặc $ x=2 $ là nghiệm thì $ \heva{&2+m+4=0\\&8+2m+4=0} \Leftrightarrow m=-6$.\\
		Thử lại, với  $ m=-6 $ thỏa  mãn.\\
		Vậy $ m=-6 $ là giá trị cần tìm.
	}
	
\end{bt}

\begin{bt}[2 điểm]
	Cho phương trình $ 2x^2+2(m-1)x-3m^2=0. \hfill(1)$
	\begin{enumEX}{1}
		\item Chứng minh  phương trình $ (1) $ luôn có hai nghiệm phân biệt.
		\item Gọi $ x_1 ,\; x_2$  là hai nghiệm của phương trình $ (1) $. Tìm giá trị nhỏ nhất của biểu thức $ A=x_1^2+x_2^2$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Xét $ \Delta'= (m-1)^2+6m^2=7m^2-2m+1=7\left( m-\dfrac{1}{7}\right) ^2+\dfrac{6}{7}$. Vậy $ \Delta'>0,\forall\; m \in \mathbb{R}$. \\
			Do đó $ (1) $ luôn có hai nghiệm phân biệt.
			\item  Theo hệ thức Vi-et, ta có $ \heva{&x_1+x_2=1-m\\&x_1\cdot x_2=-\dfrac{3m^2}{2}.}$\\
			$ A=x_1^2+x_2^2=(x_1+x_2)^2-2x_1x_2=(1-m)^2+3m^2=4m^2-2m+1=\left( 2m-\dfrac{1}{2}\right)^2 +\dfrac{3}{4}$.\\
			Do đó: $ A\ge \dfrac{3}{4} $. $ A=\dfrac{3}{4} \Leftrightarrow m=\dfrac{1}{2}$.\\\\
			Vậy giá trị nhỏ nhất của $ A=\dfrac{3}{4}$  tại $ m=\dfrac{1}{2} $.
		\end{enumEX}		
	}	
\end{bt}

\begin{bt}[2 điểm]
	Một chiếc ca nô theo dòng sông từ A đến B, rồi ngược dòng từ B về A hết $ 5 $ giờ. Tìm vận tốc riêng của ca nô (vận tốc của ca nô khi dòng nước đứng yên), biết rằng vận tốc của dòng nước là $ 4 $ km/h và khoảng cách từ A đến B là $ 48 $ km.
	\loigiai{
		Gọi vận tốc riêng của ca nô là $ x$ (km/h), $(x>4)$.\\
		Vận tốc ca nô đi từ A đến B là $ x+4 $ (km/h).\\
		Vận tốc ca nô đi từ B về A là $ x-4 $ (km/h).\\
		Thời gian ca nô đi từ A đến B là $  \dfrac{48}{x+4}$ (h).\\
		Thời gian ca nô đi từ B đến A là $  \dfrac{48}{x-4}$ (h).\\
		Tổng thời gian cả đi và về hết $ 5 $h nên ta có phương trình: \begin{align*}
			&\dfrac{48}{x+4}+ \dfrac{48}{x-4}=5\\
			\Leftrightarrow& 5x^2-96x-80=0\\
			\Leftrightarrow& \hoac{&x=20\\&x=-\dfrac{4}{5}}\text{(loại)} .
		\end{align*}
		Vậy vận tốc riêng của ca nô là $ 20 $ km/h.
	}
\end{bt}

\begin{bt}[2 điểm]
	Giải hệ phương trình $ \heva{&x^2+y^2+x+y=8\\ &xy(x+1)(y+1)=12}. $
	\loigiai{
		\begin{align*}
		&\heva{&x^2+y^2+x+y=8\\ &xy(x+1)(y+1)=12}\\ \Leftrightarrow &\heva{&(x^2+x)+(y^2+y)=8\\ &(x^2+x)(y^2+y)=12.}
		\end{align*}
		Đặt $ a=x^2+x, b=y^2+y $, hệ trở thành 
		$ 
		\heva{&a+b=8\\ &ab=12.}
		$\\
		Khi đó, $ a,\, b $ là nghiệm của phương trình $ x^2-8x+12=0$. Giải phương trình ta được hai nghiệm là $ 6 $ và $ 2 $.
		Do vai trò $ a, b $ là đối xứng nên ta xét hai trường hợp:
		\begin{enumerate}
			\item[TH1:] $ \heva{&a=x^2+x=6\\ &b=y^2+y=2} \Leftrightarrow \heva{\hoac{&x=2\\&x=-3}\\\hoac{&y=1\\ &y=-2}}.$
			\item[TH2:] $ \heva{&a=x^2+x=2\\&b= y^2+y=6} \Leftrightarrow \heva{\hoac{&x=1\\&x=-2}\\\hoac{&y=2\\ &y=-3}}.$
		\end{enumerate}
		Vậy hệ đã cho có cặp nghiệm $(x;y)$ là $ S=\{(2,1);(-3,1);(2,-2);(-3,-2), (1,2);(1,-3);(-2,2);(-2,-3 )\}. $
	}
\end{bt}

\begin{bt}[2 điểm]
	Cho phương trình $ ax^2+bx+c=0 \; (2)$ sao cho $ b>a>0  $. Chứng minh rằng: Nếu phương trình $ (2) $ vô nghiệm thì  $ \dfrac{2b+c}{a-b}<2. $
	%Gọi $ x_1 $ và $ x_2 $ tương ứng là hai nghiệm của phương trình $ ax^2+bx+c=0 $ và $ cx^2+bx+a=0 $. Chứng minh rằng x_1.x_2=1.
	\loigiai{
		Phương trình $ (2) $ vô nghiệm $\Leftrightarrow \Delta=b^2-4ac<0\Leftrightarrow 4ac>b^2 $. 
		\\
		Do đó: $ 4a+c\ge 2\sqrt{4ac}>2\sqrt{b^2}=2|b|\ge 2b $.
		\\
		Vậy $ 4a+c>2b\Leftrightarrow a+b+c>3(b-a)\Leftrightarrow \dfrac{a+b+c}{a-b}<3\Leftrightarrow 1+\dfrac{2b+c}{a-b}<3\Leftrightarrow \dfrac{2b+c}{a-b}<2. $	
	}	
\end{bt}
