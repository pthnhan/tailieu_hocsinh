\begin{bt}%[Đề HK1, Trưng Vương, Hồ Chí Minh 17-18]%[Lê Hoàng Lâm, 10EX-HK1]%[0D3Y2-4]
	
	Giải các phương trình sau:
	\begin{multicols}{2}
		\begin{enumerate}
		\item $\sqrt{2x^2+3x-1}+3x=5$.
		\item $\sqrt{2x+2}-1=\sqrt{2x-1}$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
	\begin{enumerate} 
		\item Phương trình đã cho được viết lại thành $\sqrt{2x^2+3x-1}=5-3x$. Khi đó ta có
		\begin{eqnarray*}
		& & \sqrt{2x^2+3x-1}=5-3x\\
		&\Leftrightarrow & \heva{&5-3x\ge 0 \\ &2x^2+3x-1=(5-3x)^2}\\
		&\Leftrightarrow & \heva{&x\leq\dfrac{5}{3}\\&7x^2-33x+26=0}\\
		&\Leftrightarrow & \heva{&x\leq \dfrac{5}{3}\\&\hoac{&x=1\\&x=\dfrac{26}{7}}}\\
		&\Leftrightarrow & x=1
		\end{eqnarray*}
		Vậy $S=\{1\}$.
		\item Phương trình đã cho được viết lại thành $\sqrt{2x+2}=\sqrt{2x-1}+1$
		\begin{eqnarray*}
		& & \sqrt{2x+2}=\sqrt{2x-1}+1\\
		&\Leftrightarrow &\heva{&2x-1\ge 0\\&2x+2=2x+2\sqrt{2x-1}}\\
		&\Leftrightarrow & \heva{&x\ge \dfrac{1}{2}\\&x=1}\\
		&\Leftrightarrow & x=1
		\end{eqnarray*}
		Vậy $S=\{1\}$.
		\end{enumerate}
			}
\end{bt}% giữa 2 câu phân biệt phải cách ra như mẫu này

\begin{bt}%[Đề HK1, Trưng Vương, Hồ Chí Minh 17-18]%[Lê Hoàng Lâm, 10EX-HK1]%[0D2B3-2]
	
	Cho parabol (P): $y=ax^2+bx+5$. Tìm $a,b$ biết (P) có đỉnh $I(-2;1)$.
	\loigiai{
	(P) có đỉnh là $I(-2;1)$ nên ta có
	$$\heva{&I(-2;1)\in (P)\\&x_{I}=\dfrac{-b}{2a}}
	\Leftrightarrow \heva{&4a-2b=-4\\&\dfrac{-b}{2a}=-2}
	\Leftrightarrow\heva{&4a-2b=-4\\&4a-b=0}
	\Leftrightarrow\heva{&a=1\\&b=4.}$$
			}
\end{bt}

\begin{bt}%[Đề HK1, Trưng Vương, Hồ Chí Minh 17-18]%[Lê Hoàng Lâm, 10EX-HK1]%[0D4G1-2]

	Cho ba số thực dương $a,b,c$. Chứng minh rằng:
	\begin{multicols}{2}
		\begin{enumerate}
		\item $\left( a+\dfrac{4b}{c^2}\right) \left(b+\dfrac{4c}{a^2}\right) \left(c+\dfrac{4a}{b^2}\right)\ge 64$.
		\item $\dfrac{a^3}{b}+\dfrac{b^3}{c}+\dfrac{c^3}{a} \ge ab+bc+ca$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
		\item Với $a,b,c>0$, áp dụng bất đẳng thức Cô Si cho các cặp số sau ta có:
		$$\left(a+\dfrac{4b}{c^2}\right)\ge \dfrac{4}{c}\sqrt{ab};\left(b+\dfrac{4c}{a^2}\right)\ge\dfrac{4}{a}\sqrt{bc};\left(c+\dfrac{4a}{b^2}\right)\ge\dfrac{4}{b}\sqrt{ac}$$
		Suy ra $$\left( a+\dfrac{4b}{c^2}\right) \left(b+\dfrac{4c}{a^2}\right) \left(c+\dfrac{4a}{b^2}\right)\ge\dfrac{4}{c}\sqrt{ab}\cdot\dfrac{4}{a}\sqrt{bc}\cdot\dfrac{4}{b}\sqrt{ac}=64.$$
		Dấu $``="$ xảy ra khi $a=2\sqrt{\dfrac{c}{b}};b=2\sqrt{\dfrac{a}{c}};c=2\sqrt{\dfrac{b}{a}}$.
		\item Với $a,b,c>0$, áp dụng bất đẳng thức Cô Si cho các cặp số sau ta có:
		$$\dfrac{a^3}{b}+ab\ge 2a^2;\dfrac{b^3}{c}+bc\ge 2b^2;\dfrac{c^3}{a}+ac\ge 2c^2$$
		Suy ra $$\dfrac{a^3}{b}+\dfrac{b^3}{c}+\dfrac{c^3}{a}\ge 2(a^2+b^2+c^2)-ab-bc-ac$$
		Mặt khác $$(a-b)^2+(b-c)^2+(c-a)^2\ge 0\Leftrightarrow 2(a^2+b^2+c^2)\ge 2(ab+bc+ac)$$
		Vậy $\dfrac{a^3}{b}+\dfrac{b^3}{c}+\dfrac{c^3}{a} \ge ab+bc+ca$. Dấu $``="$ xảy ra khi $a=b=c$.
		\end{enumerate}
			}
\end{bt}

\begin{bt}%[Đề HK1, Trưng Vương, Hồ Chí Minh 17-18]%[Lê Hoàng Lâm, 10EX-HK1]%[0H2Y3-1]
	
	Cho tam giác $ABC$ có $AB=3,BC=5,\widehat{ABC}=60^{\circ}$. Tính độ dài cạnh $AC$ và diện tích của tam giác $ABC$.
	\loigiai{
	\begin{itemize}
		\item $AC^2=AB^2+BC^2-2AB\cdot BC\cdot \cos B=3^2+5^2-2\cdot 3\cdot 5\cdot \cos 60^{\circ}=19\Rightarrow AC=\sqrt{19}$.
		\item $S_{\triangle ABC}=\dfrac{1}{2}AB\cdot BC\cdot \sin B=\dfrac{1}{2}\cdot 3\cdot 5\cdot \sin 60^{\circ}=\dfrac{1}{2}\cdot 3\cdot 5\cdot\dfrac{\sqrt{3}}{2}=\dfrac{15\sqrt{3}}{4}$.
	\end{itemize}
			}
\end{bt}

\begin{bt}%[Đề HK1, Trưng Vương, Hồ Chí Minh 17-18]%[Lê Hoàng Lâm, 10EX-HK1]%[0H2K2-2]
	
	Trong mặt phẳng $Oxy$, cho tam giác $ABC$ với $A(2;1),B(8;3),C(6;-3)$.
	\begin{enumerate}
		\item Gọi $I$ là trung điểm của $AB$. Tính độ dài vectơ $\vec{CI}$.
		\item Tính chu vi và diện tích tam giác $ABC$.
		\item Tìm tọa độ điểm $D$ sao cho các tam giác $ABD$ và $BCD$ là hai tam giác cân tại $D$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Ta có $I$ là trung điểm của $AB$ nên $I(5;2)$ và $|\vec{CI}|=\sqrt{(5-6)^2+(2+3)^2}=\sqrt{26}$.
		\item Ta có
	\begin{itemize}
		\item $AB=\sqrt{(8-2)^2+(3-1)^2}=2\sqrt{10}$
		\item $BC=\sqrt{(6-8)^2+(-3-3)^2}=2\sqrt{10}$
		\item $AC=\sqrt{(6-2)^2+(-3-1)^2}=4\sqrt{2}$
	\end{itemize}
		Khi đó
	\begin{itemize}
		\item Chu vi $\triangle ABC$
		$$C_{\triangle ABC}=AB+BC+AC=4\sqrt{10}+4\sqrt{2}.$$
		\item Theo công thức Hê-rông, ta có diện tích $\triangle ABC$ được tính như sau\\
		$$S_{\triangle ABC}=\sqrt{p(p-a)(p-b)(p-c)}$$
		với $a=BC, b=AC, c=AB, p$ là nửa chu vi của $\triangle ABC$ và $p=\dfrac{a+b+c}{2}$. Suy ra
		$$S_{\triangle ABC}=\sqrt{\left(2\sqrt{10}+2\sqrt{2}\right) 2\sqrt{2}\left(2\sqrt{10}-2\sqrt{2}\right) 2\sqrt{2}}=16.$$
		\end{itemize}
		\item Gọi tọa độ của điểm $D$ là $D(a;b)$. Ta có $\triangle ABD$ và $\triangle BCD$ cân tại $D$ nên suy ra
		$$\heva{&AD=BD\\ &BD=CD}\Leftrightarrow \heva{&(a-8)^2+(b-3)^2=(a-2)^2+(b-1)^2\\&(a-8)^2+(b-3)^2=(a-6)^2+(b+3)^2}\Leftrightarrow\heva{&3a+b=17\\ &a+3b=7}\Leftrightarrow \heva{&a=\dfrac{11}{2}\\ &b=\dfrac{1}{2}}$$
		Vậy $D\left(\dfrac{11}{2};\dfrac{1}{2}\right)$.
	\end{enumerate}
			}
\end{bt}

\begin{bt}%[Đề HK1, Trưng Vương, Hồ Chí Minh 17-18]%[Lê Hoàng Lâm, 10EX-HK1]%[0D3G2-4]
	
	Giải phương trình: $15x^2-27x+11+2(3x-1)\sqrt{x^2+3x-3}=0$.
	\loigiai{
	Điều kiện $x^2+3x-3\ge 0$. Đặt $t=\sqrt{x^2+3x-2}\Rightarrow t^2=x^2+3x-3$.\\
	Phương trình đã cho được viết lại thành
			\begin{eqnarray*}
			& &-(x^2+3x-3)+16x^2-24x+8+2(3x-1)\sqrt{x^2+3x-3}=0\\
			&\Leftrightarrow & -t^2+2(3x-1)t+16x^2-24x+8=0
			\Leftrightarrow \hoac{&t=-2x+2\\&t=8x-4}
			\end{eqnarray*}
	\begin{itemize}
		\item Với $t=-2x+2$, ta có
			\begin{eqnarray*}
			& &\sqrt{x^2+3x-3}=-2x+2\\
			&\Leftrightarrow & \heva{&-2x+2\ge 0\\&x^2+3x-3=(-2x+2)^2}\\
			&\Leftrightarrow & \heva{&x\le 1\\&3x^2-11x+7=0}\\
			&\Leftrightarrow & \heva{&x\le 1\\ &\hoac{&x=\dfrac{11+\sqrt{37}}{2}\\&x=\dfrac{11-\sqrt{37}}{2}}}\\
			&\Leftrightarrow & x=\dfrac{11-\sqrt{37}}{2}.
			\end{eqnarray*}
		\item Với $t=8x-4$, ta có
			\begin{eqnarray*}
			& &\sqrt{x^2+3x-3}=8x-4\\
			&\Leftrightarrow & \heva{&8x-4\ge 0\\&x^2+3x-3=(8x-4)^2}\\
			&\Leftrightarrow & \heva{&x\ge \dfrac{1}{2}\\&63x^2-67x+19=0} \quad(\text{vô nghiệm})
			\end{eqnarray*}
	\end{itemize}
	So điều kiện ban đầu ta được nghiệm của phương trình là $x=\dfrac{11-\sqrt{37}}{2}$.
	}
\end{bt}
