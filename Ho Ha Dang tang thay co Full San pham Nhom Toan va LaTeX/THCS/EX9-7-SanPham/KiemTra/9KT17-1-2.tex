\begin{ex}%[9H1K1]
Cho tam giác $ABC$ vuông tại $A$ có $\dfrac{AB}{AC}=\dfrac{3}{4}$ và đường cao $AH=9$cm. Tính độ dài đoạn thẳng $AC$.
	\choice
	{$\dfrac{45}{4}$cm}
	{$14$cm}
	{\True $15$cm}
	{$225$cm}
	\loigiai{Ta có $\dfrac{AB}{AC}=\dfrac{3}{4}$, suy ra $AB=\dfrac{3AC}{4}$. Do $AH$ là đường cao của tam giác vuông $ABC$ nên:
	$$\frac{1}{9^2} = \frac{1}{AH^2} = \frac{1}{AB^2} + \frac{1}{AC^2} = \frac{16}{9AC^2} + \frac{9}{9AC^2} = \frac{25}{9AC^2}.$$
Do đó $AC^2 = 25.9 \Rightarrow AC = 15$ (cm).}
\end{ex}

\begin{ex}%[9D1B4]%[9D1K4]
	Cho biểu thức: 
	$ P = \left( \dfrac{1}{1 - \sqrt a } - \dfrac{1}{1 + \sqrt a } \right)\left( \dfrac{1}{\sqrt a } + 1 \right) \,\, (0<a\ne 1).$
	\begin{enumerate}[$a)$]
		\item  Rút gọn biểu thức $P$.
		\item  Với những giá trị nào của $a$ thì $P > \dfrac{1}{2}$.
	\end{enumerate}
\loigiai{
\begin{enumerate}[$a)$]
\item  Với $a>0$ và $a\ne 1$, ta có:
{\allowdisplaybreaks
	\begin{align*}
	P &= \left( \frac{1}{1 - \sqrt a } - \frac{1}{1 + \sqrt a } \right)\left( \frac{1}{\sqrt a } + 1 \right)\\
	&= \frac{1 + \sqrt a  - 1 + \sqrt a }{(1 - \sqrt a )(1 + \sqrt a )} \cdot \frac{1 + \sqrt a }{\sqrt a } = \frac{2\sqrt a }{(1 - \sqrt a )(1 + \sqrt a )} \cdot \frac{1 + \sqrt a }{\sqrt a } = \frac{2}{1 - \sqrt a }.
	\end{align*}}Vậy $P = \dfrac{2}{1 - \sqrt a }$.
\item  Với $a>0$ và $a\ne 1$ thì điều kiện để $P > \dfrac{1}{2}$ là:
{\allowdisplaybreaks
	\begin{align*}
	&\frac{2}{1 - \sqrt a } > \frac{1}{2} \Leftrightarrow \frac{2}{1 - \sqrt a } - \frac{1}{2} > 0 \Leftrightarrow \frac{4 - 1 + \sqrt a }{1 - \sqrt a } > 0 \Leftrightarrow \frac{3 + \sqrt a }{1 - \sqrt a } > 0\\
	\Leftrightarrow& 1 - \sqrt a  > 0 \Leftrightarrow \sqrt a  < 1 \Leftrightarrow a < 1.
	\end{align*}}Kết hợp với điều kiện $a>0$ ta được: $0<a<1$.
\end{enumerate}
}
\end{ex}


\begin{ex}%[9D1K2]
	Tính đường chéo của một hình vuông, biết diện tích của nó bằng diện tích của
	hình chữ nhật có chiều rộng $5$ mét và chiều dài là $10$ mét.
\loigiai{Gọi cạnh của hình vuông là $x$ mét ($x>0$). Khi đó diện tích của hình vuông là ${x^2}$,  đường chéo của hình vuông  là $\sqrt {x^2 + x^2}  = \sqrt {2x^2} $.
Theo giả thiết, ta có: 
$x^2 = 10.5 \Leftrightarrow x^2 = 50$.
Vậy đường chéo của hình vuông là: 
$$\sqrt {2x^2}  = \sqrt {100}  = 10.$$
}
\end{ex}



\begin{ex}%[9D1B3]%[9D1K3]
	Cho các biểu thức: $A = \sqrt {x - 1} .\sqrt {x + 4} $ và $A = \sqrt {(x - 1)(x + 4)} $.
	\begin{enumerate}[$a)$]
		\item  Tìm $x$ để $A$ có nghĩa.
		\item  Tìm $x$ để $B$ có nghĩa.
		\item  Với giá trị nào của $x$ thì $A=B$.
	\end{enumerate}
\loigiai{
	\begin{enumerate}[$a)$]
\item  $A$ có nghĩa khi: $\left\{ \begin{array}{l}
x - 1 \ge 0\\
x + 4 \ge 0
\end{array} \right. \Leftrightarrow \left\{ \begin{array}{l}
x \ge 1\\
x \ge  - 4
\end{array} \right. \Leftrightarrow x \ge 1.$
\item  $B$ có nghĩa khi: $(x - 1)(x + 4) \ge 0$.
\begin{itemize}
	\item Trường hợp 1: $\left\{ \begin{array}{l}
	x - 1 \ge 0\\
	x + 4 \ge 0
	\end{array} \right. \Leftrightarrow \left\{ \begin{array}{l}
	x \ge 1\\
	x \ge  - 4
	\end{array} \right. \Leftrightarrow x \ge 1.$
	\item Trường hợp 2: $\left\{ \begin{array}{l}
	x - 1 \le 0\\
	x + 4 \le 0
	\end{array} \right. \Leftrightarrow \left\{ \begin{array}{l}
	x \le 1\\
	x \le  - 4
	\end{array} \right. \Leftrightarrow x \le  - 4.$
\end{itemize}
Vậy $B$ có nghĩa khi: $x \ge 1$ hoặc $ x \le  - 4.$
\item  Để $A$ và $B$ đồng thời có nghĩa thì $x\ge 1$. Khi đó $A=B$ theo định lí
khai phương một tích.
\end{enumerate}
}
\end{ex}


\begin{ex}%[9H1B2]
Xét quan hệ giữa hai góc trong biểu thức $\dfrac{\sin 32^\circ}{\cos 58^\circ}$ rồi tính giá trị của biểu thức đó.
\loigiai{
Ta có $32^\circ + 58^\circ = 90^\circ$ nên $32^\circ$ và $58^\circ$ là hai góc phụ nhau. \\

\begin{ex}%[9H1K1]%[9H1G1]
	Cho tam giác $ABC$ có $AB=5, BC=10, AC=3\sqrt 5$. Đường phân giác $BK$ của góc 
	$\widehat {ABC}$ $(K \in AC)$ cắt đường cao $AH$ $(H \in BC)$ và cắt trung tuyến $AM$  $(M \in BC)$
	của tam giác $ABC$ lần lượt tại các điểm $O$ và $T$.
	\begin{enumerate}[$a)$]
		\item  Tính $AH$.
		\item  Tính diện tích tam giác $AOT$.
	\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzDefPoints{0/0/B, 4/0/H, 10/0/C, 4/3/A, 5/0/M}
\tkzLabelPoint[above](A){$A$}
\tkzLabelPoint[below](C){$C$}
\tkzLabelPoint[below](B){$B$}
\tkzLabelPoint[below](M){$M$}
\tkzLabelPoint[below](H){$H$}
\tkzDefMidPoint(A,M)\tkzGetPoint{T}
\tkzLabelPoint[above right](T){$T$}
\tkzDrawPoint(T)
\tkzDrawSegments(A,B B,C C,A A,H A,M)
\tkzMarkRightAngle(A,H,B)
\tkzInterLL(B,T)(A,C)\tkzGetPoint{K}
\tkzDrawPoint(K)\tkzLabelPoint[above right](K){$K$}
\tkzInterLL(B,T)(A,H)\tkzGetPoint{O}
\tkzDrawPoint(O)\tkzLabelPoint[below left](O){$O$}
\tkzDrawSegment(B,K)
	\end{tikzpicture}
\end{center}
\begin{enumerate}[$a)$]
	\item Ta có: $AH^2 = AB^2 - BH^2$.
	Mặt khác: 
	$AH^2 = AC^2 - (BC - BH)^2$.\\
	Do đó: $AC^2 - AB^2=(BC - BH)^2 - BH^2$.\\
	Từ đó:
$$(10 - BH)^2 - BH^2 = (3\sqrt 5 )^2 - 5^2 \Leftrightarrow 100 - 20BH = 20 \Leftrightarrow BH = 4.$$
Ta có: $AH = \sqrt {AB^2 - BH^2}  = \sqrt {5^2 - 4^2}  = 3$.
\item  Ta có tam giác $ABM$ cân tại đỉnh $B$ (do $BA=BM=5$). Suy ra:
$$S_{ABT} = \frac{1}{2}S_{ABM} = \frac{1}{4}AH.BM = \frac{15}{4}\,\, \rm(\text{đvdt}).$$
Do $BO$ là phân giác của góc $\widehat {ABH}$ nên:
$$\frac{OH}{OA} = \frac{BH}{BA} \Rightarrow \frac{OH}{OA} + 1 = \frac{BH}{BA} + 1 \Rightarrow \frac{AH}{OA} = \frac{BH + BA}{BA} = \frac{9}{5}.$$
Suy ra: 
$$OA = \frac{5}{9}AH = \frac{5.3}{9} = \frac{5}{3} \Rightarrow S_{ABO} = \frac{1}{2}OA.BH = \frac{1}{2} \cdot \frac{5}{3} \cdot 4 = \frac{10}{3}.$$
Diện tích tam giác $AOT$ là: $S_{AOT} = S_{ABT} - S_{ABO} = \dfrac{15}{4} - \dfrac{10}{3} = \dfrac{5}{12}$ (đvdt).
\end{enumerate}
}
\end{ex}


\begin{ex}%[9D1G9]
Hãy tìm một  đa thức bậc ba, có hệ số nguyên nhận và số $1 - \sqrt[3]{2} + \sqrt[3]{4}$ làm nghiệm.
\loigiai{Xét
{\allowdisplaybreaks
	\begin{align*}
	&x = 1 - \sqrt[3]{2} + \sqrt[3]{4} \Rightarrow x\left( 1 + \sqrt[3]{2} \right) = 1 + \left( \sqrt[3]{2} \right)^3  = 3 \Rightarrow x\sqrt[3]{2} = 3 - x\\
	\Rightarrow&2x^3  = \left( 3 - x \right)^3  = 27 - 27x + 9x^2  - x^3  \Rightarrow x^3  - 3x^2  + 9x - 9 = 0.
	\end{align*}}Vậy $1 - \sqrt[3]{2} + \sqrt[3]{4}$ là nghiệm của đa thức bậc ba hệ số nguyên
\[f(x)=x^3  - 3x^2  + 9x - 9.\]
Do đó $f(x)=x^3  - 3x^2  + 9x - 9$ là một đa thức thỏa mãn các yêu cầu đề bài.}
\end{ex}
