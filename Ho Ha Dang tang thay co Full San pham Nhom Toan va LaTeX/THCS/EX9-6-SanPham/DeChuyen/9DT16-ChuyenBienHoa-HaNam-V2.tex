\begin{ex}%[Đề thi vào 10, THPT Chuyên Hà Nam, năm học 2016-2017]%[9D1B8]
Cho biểu thức
\begin{align*}
P= \left(\dfrac{10x+4}{5\sqrt{5x^3}-8} - \dfrac{\sqrt{5x}}{5x+2\sqrt{5x} +4 } \right) \cdot \left( \dfrac{1+5\sqrt{5x^3}}{1+\sqrt{5x}} - \sqrt{5x}\right) \cdot \left( \dfrac{6\sqrt{5x}-3}{\sqrt{5x}-1} - 6\right)
\end{align*}
$\left(\text{với} x\ge 0, x\ne \dfrac{1}{5}, x\ne \dfrac{4}{5} \right)$.
\begin{enumerate}
\item Rút gọn $P$.
\item Tìm các số tự nhiên $x$ lơn hơn $10$ để $P>\dfrac{7}{2}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item 	\begin{align*}
			P&=\dfrac{10x+4 - 5x + 2\sqrt{5x}}{\left(5x +2\sqrt{5x}+4\right) \left(\sqrt{5x}-2\right)} \left(5x -\sqrt{5x} + 1-\sqrt{5x}\right) \cdot \dfrac{6\sqrt{5x}-3-6\sqrt{5x}+6}{\sqrt{5x}-1}\\
	&=\dfrac{1}{\sqrt{5x}-2}\cdot \left(\sqrt{5x}-1\right)^2 \dfrac{3}{\sqrt{5x}-1}\\
			&= \dfrac{3\left(\sqrt{5x}-1\right)}{\sqrt{5x}-2}.
		\end{align*}
\item Ta có
\begin{align*}
P>7 \Leftrightarrow  \dfrac{3\left(\sqrt{5x}-1\right)}{\sqrt{5x}-2}>\dfrac{7}{2} \Leftrightarrow \dfrac{8-\sqrt{5x}}{\sqrt{5x}-2}>0 \Leftrightarrow 2< \sqrt{5x} <8 \Leftrightarrow \dfrac{4}{5}< x<\dfrac{64}{5}.
\end{align*}
Do $x\in\mathbb{N}$ và $x>10$ nên $x\in\{11;12\}$.

\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, THPT Chuyên Hà Nam, năm học 2016-2017][9D4K7]%[9D4K6]
\hfill
\begin{enumerate}
\item Giải phương trình $\sqrt{2x^2-4x-2}+(x-1)^2\sqrt{12-4x}=(8-x)\sqrt{3-x}$.
\item Cho phương trình $x^2 -2(m-1)x +2m-6=0$ ($m$ là tham số thực). Tìm tất cả các giá trị của $m$ để phương trình đã cho có hai nghiệm phân biệt $x_1$, $x_2$ thoả mãn
\begin{align*}
(x_1^2 +2x_1 +2m-5)(x_2^2 +2x_2 +2m-5) =-15.
\end{align*}
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Điều kiện của phương trình $x\le 1-\sqrt{3}$, $1+\sqrt{3}\le x \le 3$. Phương trình đã cho tương đương
\begin{align*}
&\sqrt{2x^2-4x-2} -\sqrt{3-x} +(2x^2-3-5)\sqrt{3-x}=0\\
\Leftrightarrow &\dfrac{2x^2-3x-5}{\sqrt{2x^2-4x-2}+\sqrt{3-x}}+(2x^2-3x-5)\sqrt{3-x}=0\\
\Leftrightarrow &(2x^2-3x-5) \left( \dfrac{1}{\sqrt{2x^2 -4x-2} +\sqrt{3-x}} +\sqrt{3-x}\right)=0.
\end{align*}
Vì $\dfrac{1}{\sqrt{2x^2 -4x-2} +\sqrt{3-x}} +\sqrt{3-x}>0$  nên phương trình tương đương $2x^2-3x-5=0$ hay $x=-1$ hoặc $x=\dfrac{5}{2}$.\\
Kết hợp với điều kiện ta có tập nghiệm của phương trình là $S=\left\{
-1; \dfrac{5}{2} \right\}$.

\item Phương trình đã cho có $\Delta'=m^2-4m+7=(m-2)^2+3>0$ với mọi $m$ nên phương trình luôn có hai nghiệm phân biệt $x_1$, $x_2$. Theo định lí Vi\`ete ta có $\heva{&x_1+x_2=2(m-1)\\ &x_1x_2=2m-6}.$\\
Lại có $x^2-2(m-1)x+2m-6=x^2-2x+2m-5=2mx+1$ mà $x_1$, $x_2$ là nghiệm của phương trình nên
\begin{align*}
&(x_1^2+2x_1+2m-5)(x_2^2+2x_2+2m-5)=-15\\
\Leftrightarrow &(2mx_1+1)(2mx_2+1)=-15\\
\Leftrightarrow &4m^2x_1x_1+2m(x_1+x_2)+1=-15\\
\Leftrightarrow &2m^3-5m^2-m+4=0 \Leftrightarrow (m-1)(2m^2-3m-4)=0\\
\Leftrightarrow &\hoac{&m=1\\ &m=\dfrac{3-\sqrt{41}}{4}\\ &m=\dfrac{3+\sqrt{41}}{4}}.
\end{align*}
Vậy các giá trị cần tìm của $m$ là $1$, $\dfrac{3-\sqrt{41}}{4}$, $\dfrac{3+\sqrt{41}}{4}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, THPT Chuyên Hà Nam, năm học 2016-2017]%[9D5G2]
Cho các số nguyên dương $a$ và $b$ thoả mãn $\dfrac{a+2}{b}+\dfrac{b+3}{a}$ là một số nguyên. Gọi $d$ là ước chung lớn nhất của $a$ và $b$. Chứng minh rằng $d^2 \le 2a+3b$.
\loigiai{
Do $\dfrac{a+2}{b}+\dfrac{b+3}{a}$ là số nguyên nên $ab\,|\, a^2+b^2+2a+3b$. Mà $d^2\,|\,ab$ và $d^2\,|\,a^2+b^2$ nên suy ra $d^2\,|\, 2a+3b$. Vậy $d^2\le 2a+3b$.
}
\end{ex}

\begin{ex}%[Đề thi vào 10, THPT Chuyên Hà Nam, năm học 2016-2017]%[9H3K7]%[9H3G7]
Cho đường tròn $(O,R)$ và dây $BC$ không đi qua $O$. Trên tia đối của tia $BC$ lấy điểm $A$ ($A$ khác $B$). Từ $A$ kẻ hai tiếp tuyến $AM$, $AN$ với đường tròn $(O)$ ($M$, $N$ là các tiếp điểm). Gọi $I$ là trung điểm của $BC$.
\begin{enumerate}
\item Chứng minh rằng $IA$ là tia phân giác của $\widehat{MIN}$.
\item Chứng minh rằng $\dfrac{2}{AK}=\dfrac{1}{AB}+\dfrac{1}{AC}$.
\item Lấy các điểm $E$, $P$, $F$ lần lượt trên các đoạn $AM$, $AN$, $NA$ sao cho tứ giác $AEPF$ là hình bình hành. Gọi $Q$ là điểm đối xứng của $P$ qua đường thẳng $FE$. Chứng minh rằng $O$, $P$, $Q$ thẳng hàng.
\end{enumerate}

\begin{enumerate}
\item Dễ thấy năm điểm $O$, $A$, $M$, $N$, $I$ nằm trên đường tròn đường kính $OA$. Do đó $\widehat{MIA}=\widehat{ANM}=\widehat{AMN}=\widehat{NIA}$.\\
Vậy $IA$ là tia phân giác góc $\widehat{MIN}$.
\item Ta có tứ giác $OIKH$ nội tiếp đường tròn đường kính $OK$. Suy ra $AK\cdot AI= AH\cdot AO$.\\
Trong tam giác vuông $AMO$ ta có $AM^2=AH\cdot AO$.\\
Do $AM$ là tiếp tuyến ta có $AM^2=AB \cdot AC$.\\
Từ đó ta có 
\begin{align*}
&AK\cdot AI= AB\cdot AC\\
\Rightarrow &AK\cdot \dfrac{AB+AC}{2}=AB\cdot AC\\
\Rightarrow &\dfrac{2}{AK}=\dfrac{1}{AB} + \dfrac{1}{AC}.
\end{align*}
\item Gọi $J$ là giao điểm của $EF$ và $PQ$, $T$ là trung điểm $AP$. Khi đó $TJ$ là đường trung bình của tam giác $APQ$, suy ra $AQ\parallel EF$.\\
Do tính chất của phép đối xứng nên $\widehat{EQF}=\widehat{EPF}=\widehat{EAF}$, suy ra tứ giác $AEFQ$ nội tiếp, mà $AQ\parallel EF$ nên $AEFQ$ là hình thang cân. Suy ra $\widehat{QFA}=\widehat{QEA} \Rightarrow \widehat{QFN}=\widehat{QEM}$.\\
Dễ thấy $EM=EP=AF=QE$ và $FN=FP=FQ$, kết hợp với chứng minh trên ta có $\triangle FAQ \backsim \triangle EMQ$ (c.g.c), suy ra $\widehat{FNQ}=\widehat{EMQ}$, do đó tứ giác $AMNQ$ nội tiếp. Suy ra $OQ\parallel AQ$, do đó $OQ \parallel EF$.\\
Vậy $O$, $P$, $Q$ thẳng hàng.
\end{enumerate}
}

\begin{ex}%[Đề thi vào 10, THPT Chuyên Hà Nam, năm học 2016-2017]%[9D5G1]
Cho các số thực dương $x$, $y$, $z$ thoả mãn $\dfrac{1}{x+y}+\dfrac{1}{y+z}+\dfrac{1}{z+x}=12$. Chứng minh rằng
\begin{align*}
\dfrac{1}{2x+3y+3z} +\dfrac{1}{3x+2y+3z} + \dfrac{1}{3x+3y+2z} \le 3.
\end{align*}
Dấu đẳng thức xảy ra khi nào?
\loigiai{
Áp dụng bất đẳng thức Cauchy-Schwarz, ta có
\begin{align*}
&\dfrac{1}{x+y}+\dfrac{1}{x+z}+\dfrac{1}{y+z}+\dfrac{1}{y+z}\ge \dfrac{16}{2x+3y+3z}\\
&\dfrac{1}{y+z}+\dfrac{1}{y+x}+\dfrac{1}{z+x}+\dfrac{1}{z+x}\ge \dfrac{16}{3x+2y+3z}\\
&\dfrac{1}{z+x}+\dfrac{1}{z+y}+\dfrac{1}{x+y}+\dfrac{1}{x+y}\ge \dfrac{16}{3x+3y+2z}.
\end{align*}
Cộng các bất đẳng thức trên theo vế ta được
\begin{align*}
\dfrac{16}{2x+3y+3z} +  \dfrac{16}{3x+2y+3z} + \dfrac{16}{3x+3y+2z} \le 4\left( \dfrac{1}{x+y} + \dfrac{1}{y+z} + \dfrac{1}{z+x}\right)
\end{align*}
Suy ra
\begin{align*}
\dfrac{1}{2x+3y+3z} +\dfrac{1}{3x+2y+3z} + \dfrac{1}{3x+3y+2z}\le \dfrac{4\cdot 12}{16}=3. 
\end{align*}
Dấu đẳng thức xảy ra khi $x=y=z=\dfrac{1}{8}$.
}
\end{ex}
