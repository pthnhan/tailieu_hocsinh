\begin{ex}%[HK1 THPT Hoa Lư A Ninh Bình 2018]%[Hữu Bình DA 11HK1-18]%[1D1B3-3]
	Số nghiệm của phương trình $\sin x-\sqrt{3}\cos x=2$ trong khoảng $\left(0;5\pi\right)$ là
	\choice
	{\True 3}
	{4}
	{2}
	{1}
\loigiai{
	Ta có $\sin x-\sqrt{3}\cos x=2\Leftrightarrow \sin\left(x-\dfrac{\pi}{3}\right)=1 \Leftrightarrow x-\dfrac{\pi}{3}=\dfrac{\pi}{2}+k2\pi \Leftrightarrow x=\dfrac{5\pi}{6}+k2\pi$.\\	
	Ta có $0\le \dfrac{5\pi}{6}+k2\pi \le 5\pi \Leftrightarrow -\dfrac{5}{12} \le k \le \dfrac{25}{12}$.\\
	Vì $k$ nguyên nên $k=0,1,2$.\\	
	Vậy nghiệm thuộc $(0;5\pi)$ là $x=\dfrac{5\pi}{6}, x=\dfrac{17\pi}{6}, x=\dfrac{29\pi}{6}.$
}
\end{ex}

	\choice
	{$m=2 $}
	{$m=-1 $}
	{$m=1,m=2 $}
	{\True $m=1 $}
	\loigiai{
	Theo định lí Viét: $x_1+x_2+x_3=-\dfrac{b}{a}=3$.\\
	Mặt khác 3 nghiệm lập thành một cấp số cộng nên giả sử $x_1+x_3=2x_2$.
	Do đó $3x_2=3\Leftrightarrow x_2=1$.\\
	Thay $x_2=1$ vào phương trình ta được $m=1$.\\
	Thử lại với $m=1$, ta có $x^3-3x^2+x+1=0 \Leftrightarrow x=1-\sqrt{2}, x=1, x=1+\sqrt{2}$.\\
	Vậy $m=1$.
	}
\end{ex}
