\section{ĐƯỜNG THẲNG SONG SONG VÀ ĐƯỜNG THẲNG CẮT NHAU}

\subsection{Nền tảng kiến thức}
Hai đường thẳng $y=ax+b$ ($a\neq 0$) và $y=a'x+b'$ ($a'\neq 0$)
\begin{enumEX}{1}
	\item cắt nhau khi va chỉ khi $a\neq a'$.
	\item song song với nhau khi và chỉ khi $a=a'$ và $b\neq b'$.
	\item trùng nhau khi và chỉ khi $a=a'$ và $b=b'$.
	\end{enumEX}
\textbf{Đặc biệt: } Người ta đã chứng minh được rằng nếu $a\cdot a'=-1$ thì hai đường thẳng vuông góc với nhau.
\subsection{Áp dụng giải toán}

\begin{dang}{Xét vị trí tương đối của hai đường thẳng}
Cho hai đường thẳng $d_1\colon y=ax+b$ và $d_2\colon y=a'x+b'$
\begin{enumEX}{1}
	\item Nếu $a\neq a'$ thì $d_1$ cắt $d_2$.
	\item Nếu $a=a'$ và $b\neq b'$ thì $d_1\parallel d_2$.
	\item Nếu $a=a'$ và $b=b'$ thì $d_1\equiv d_2$.
\end{enumEX}	
\end{dang}
\begin{vd}
	Xét vị trí tương đối của hai trong bốn đường thẳng sau
	\begin{multicols}{2}
		$(d_1)\colon y=-\sqrt{3}x+1$;\\
		$(d_2)\colon y=\sqrt{3}x+2$;\\
		$(d_3)\colon y=-\sqrt{3}x+2$;\\
		$(d_4)\colon y=\sqrt{3}x+2$.
	\end{multicols}
\loigiai{Ta thấy 
	\begin{itemize}
		\item $d_2\equiv d_4$ vì $a=a'=\sqrt{3}$ và $b=b'=2$;
		\item $d_1\parallel d_3$ vì $a=a'=-\sqrt{3}$ và $b=1\neq b'=2$;
		\item $d_1$ cắt $d_2$ vì $a=-\sqrt{3}\neq a'=\sqrt{3}$, đương nhiên $d_1$ cũng cắt $d_4$. Cũng từ đó suy ra $d_3$ cắt $d_2$ và $d_4$.
	\end{itemize}
}
\end{vd}
\begin{vd}
	Hai đường thẳng nào sau đây là cắt nhau? Chúng có vuông góc không?
		\begin{multicols}{3}
		$(d_1)\colon y=-2x+1$;\\
		$(d_2)\colon y=-x+2$;\\
		$(d_3)\colon y=x-1$.
		\end{multicols}
\loigiai{Ta có 
	\begin{itemize}
		\item $d_1$ cắt $d_2$ vì $a=-2\neq a'=-1$. Tuy nhiên $a\cdot a'=2\neq -1$ nên chúng không vuông góc;
		\item $d_2$ cắt $d_3$ vì $a=-1\neq a'=1$. Mặt khác $a\cdot a'=-1$ nên chúng vuông góc;
		\item $d_1$ cắt $d_3$ vì $a=-2\neq a'=1$, nhưng không vuông góc vì $a\cdot a'\neq -1$.
	\end{itemize}
}
\end{vd}
\begin{dang}{Xác định giao điểm của hai đường thẳng}
	Để tìm giao điểm của hai đường thẳng $y=ax+b$ và $y=a'x+b'$ ta xét hoành độ giao điểm của hai đồ thị thỏa mãn phương trình $ax+b=a'x+b'$ ta tìm được $x$, rồi thay vào một trong hai phương trình tìm $y$ và suy ra giao điểm.\\
	\textbf{Chú ý:} \\
	Để tìm giao điểm của đồ thị với $Ox$: cho $y=0$ suy ra $x$.\\
	Để tìm giao điểm của đồ thị với $Oy$: cho $x=0$ suy ra $y$. 
\end{dang}
\begin{vd}
	Tìm giao điểm của hai đường thẳng $y=3x-1$ và $y=x+5$. 
	\loigiai{
	Hoành độ giao điểm của hai đồ thị thỏa mãn phương trình \\
	$3x-1=x+5\Leftrightarrow x=3$ suy ra $y= 8$ (bằng cách thay $x=3$ vào $y=3x-1$ hoặc $y=x+5$).\\
	Vậy hai đường thẳng cắt nhau tại tại $A(3;8)$.
}
\end{vd}
\begin{vd}
	Tìm giao điểm của đồ thị $y=2x-4$ với $Ox$ và $Oy$.
	\loigiai{
	Đồ thị giao $Ox\colon y=0$ suy ra $2x-4=0\Leftrightarrow x=2$. Vậy đồ thị cắt $Ox$ tại $A(2;0)$.\\
	Đồ thị giao $Oy\colon x=0$ suy ra $y=-4$. Vậy đồ thị cắt $Oy$ tại $B(0;-4)$.
} 
\end{vd}
\begin{vd}
	Tìm $m$ biết đường thẳng $y=(2m-1)x-2m+2$
	\begin{enumEX}{1}
		\item cắt trục hoành tại điểm có hoành độ bằng $3$;
		\item cắt trục tung tại điểm có tung độ bằng $-1$.
	\end{enumEX}
\loigiai{
		\begin{enumEX}{1}
		\item Đồ thị hàm số cắt trục hoành tại điểm có hoành độ bằng $3$ nên đồ thị đi qua $A(3;0)$. Thay $x=3$, $y=0$ vào phương trình của đồ thị ta được\\ $0=(2m-1)\cdot 3-2m+2\Leftrightarrow 6m-3-2m+2=0\Leftrightarrow m=\dfrac{1}{4}$.
		\item Tương tự. Thay $x =0$, $y=-1$ vào phương trình của đồ thị ta được\\
		$-1=(2m-1)\cdot 0-2m+2\Leftrightarrow m=\dfrac{3}{2}$.
	\end{enumEX}
}
\end{vd}
\begin{dang}{Xác định hàm số thỏa mãn điều kiện cho trước}
Căn cứ vào giả thiết để xác định hệ số $a$, $b$ rồi từ đó suy ra phương trình của đường thẳng $y=ax+b$.\\
\textbf{Một số trường hợp phổ biến cần lưu ý:}
\begin{itemize}
	\item Hai đường thẳng $y=ax+b$ và $y=a'x+b'$ cắt nhau thì $a\neq a'$;
	\item Hai đường thẳng $y=ax+b$ và $y=a'x+b'$ song song thì $a=a'$ và $b\neq b'$;
	\item Hai đường thẳng $y=ax+b$ và $y=a'x+b'$ trùng nhau thì $a=a'$ và $b=b'$;
	\item Hai đường thẳng $y=ax+b$ và $y=a'x+b'$ vuông góc thì $a\cdot a'=-1$;
	\item Đường thẳng $y=ax+b$ đi qua $A(x_0;y_0)$ thì ta luôn có phương trình $y_0=a\cdot x_0+b$;
	\item Hai đường thẳng $y=ax+b$ và $y=a'x+b'$ cắt nhau tại điểm có hoành độ bằng $x_0$ thì ta luôn có $ax_0+b=a'x_0+b'$;
	\item Hai đường thẳng $y=ax+b$ và $y=a'x+b'$ cắt nhau tại điểm có tung độ bằng $y_0$ thì thường thay $y_0$ vào một trong hai đường thẳng để tìm $x_0$ và thay $x_0$ và $y_0$ vào đường thẳng còn lại để tìm yếu tố yêu cầu.
\end{itemize}	
\end{dang}
\begin{vd}
	Xác định hệ số $a$ của đường thẳng $y=ax+\sqrt{2}$ trong mỗi trường hợp sau:
	\begin{enumEX}{1}
		\item Đường thẳng $y=ax+\sqrt{2}$ song song với đường thẳng $y=-\dfrac{1}{2}x+1$;
		\item Đường thẳng $y=ax+\sqrt{2}$ cắt đường thẳng $y=2x+1$ tại điểm có hoành độ bằng $\sqrt{2}$.
	\end{enumEX}
\loigiai{
\begin{enumEX}{1}
	\item Rõ ràng ta thấy $b=\sqrt{2}\neq b'=1$. Mặt khác vì đường thẳng $y=ax+\sqrt{2}$ song song với đường thẳng $y=-\dfrac{1}{2}x+1$ nên ta có $a=-\dfrac{1}{2}$.\\
	Vậy $a=-\dfrac{1}{2}$ thỏa mãn yêu cầu bài toán.
	\item Vì đường thẳng $y=ax+\sqrt{2}$ cắt đường thẳng $y=2x+1$ tại điểm có hoành độ bằng $x=\sqrt{2}$ nên ta có $y=2\sqrt{2}+1$. Tọa độ giao điểm là $A\left(\sqrt{2};2\sqrt{2}+1\right)$.\\
	Đường thẳng $y=ax+\sqrt{2}$ đi qua điểm $A\left(\sqrt{2};2\sqrt{2}+1\right)$ nên ta có phương trình\\ 
	$a\sqrt{2}+\sqrt{2}=2\sqrt{2}+1\Rightarrow a=\dfrac{2+\sqrt{2}}{2}$.
\end{enumEX}
}
\end{vd}
\begin{vd}
	Xác định hàm số có đồ thị là đường thẳng đi qua điểm có tọa độ $(2;-3)$ và 
	\begin{enumEX}{1}
		\item song song với đường thẳng $y=5x+1$;
		\item cắt đường thẳng $y=2x-1$ tại điểm có tung độ bằng $5$.
	\end{enumEX}
\loigiai{Giả sử đường thẳng cần tìm là $y=ax+b$.\\
	Vì đường thẳng $y=ax+b$ đi qua điểm $(2;-3)$ nên ta có $2a+b=-3\quad (1)$.
	\begin{enumEX}{1}
		\item Vì dường thẳng $y=ax+b$ song song với đường thẳng $y=5x+1$ nên ta có $a=5$, thay vào $(1)$ suy ra $b=-13$.\\
		Vậy đường thẳng cần tìm là $y=5x-13$ thỏa mãn yêu cầu bài toán.
		\item Vì đường thẳng $y=ax+b$ cắt đường thẳng $y=2x-1$ tại điểm có tung độ $y=5$ nên hoành độ giao điểm là nghiệm của phương trình $5=2x-1\Rightarrow x=3$. Tọa độ giao điểm là $A\left(3;5\right)$.\\
		Đường thẳng $y=ax+b$ đi qua điểm $A\left(3;5\right)$ nên ta có phương trình\\ 
		$3a+b=5\Rightarrow b=5-3a$, thay vào $(1)$ suy ra $a=8\Rightarrow b=5-3\cdot 8=-19$.\\
		Vậy đường thẳng cần tìm là $y=8x-19$ thỏa yêu cầu bài toán.
	\end{enumEX}
}
\end{vd}
\begin{vd}
	Xác định hàm số biết đồ thị của nó là đường thẳng song song với đường thẳng $y=2x-1$ và cắt đường thẳng $y=3x+2$ tại điểm có hoành độ bằng $1$.
	\loigiai{
	Vì đường thẳng cần tìm song song với đường thẳng $y=2x-1$ nên phương trình có dạng $y=2x+b\quad (1)$.\\
	Mặt khác hoành độ giao điểm của nó với đường thẳng $y=3x+2$ là $x=1$ nên tung độ giao điểm là $y=3\cdot 1+2=5$, hay đường thẳng $y=2x+b$ đi qua điểm $A(1;5)$ nên từ $(1)$ ta có $5=2\cdot 1+b\Rightarrow b=3$.\\
	Vậy đồ thị hàm số cần tìm là đường thẳng có phương trình $y=2x+3$.
}
\end{vd}
\begin{dang}{Xác định giá trị của tham số $m$ để đường thẳng $y=ax+b$ thỏa mãn điều kiện cho trước}
Sử dụng kiến thức cần nhớ, thiết lập mối liên hệ thông qua điều kiện cho trước để tìm tham số $m$.	
\end{dang}
\begin{vd}
	Cho hai hàm số với biến $x$\\
	$y=(m+1)x-(2m+1)$ và $y=(2m-1)x+3m$.\\
	Tìm giá trị của $m$ sao cho đồ thị của các hàm số đó là
	\begin{enumEX}{1}  
		\item hai đường thẳng cắt nhau;
		\item hai đường thẳng song song;
		\item hai đường thẳng trùng nhau.
	\end{enumEX}
\loigiai{
\begin{enumEX}{1}  
	\item Hai đường thẳng đã cho cắt nhau khi thỏa mãn các điều kiện sau\\
	$\heva{&m+1\neq 0\\&2m-1\neq 0\\&m+1\neq 2m-1}\Leftrightarrow \heva{&m\neq -1\\&m\neq \dfrac{1}{2}\\&m\neq 2.}$\\
	Vậy với $m\neq -1$, $m\neq \dfrac{1}{2}$ và $m\neq 2$ thì hai đường thẳng cắt nhau.
	\item Hai đường thẳng song song khi thỏa mãn các điều kiện sau\\
	$\heva{&m+1\neq 0\\&2m-1\neq 0\\&m+1=2m-1\\&-(2m+1)\neq 3m}\Leftrightarrow \heva{&m\neq -1\\&m\neq \dfrac{1}{2}\\&m=2\\&m\neq -\dfrac{1}{5}.}$\\
	Vì giá trị $m=2$ đều khác các giá trị $-1$, $\dfrac{1}{2}$, $-\dfrac{1}{5}$ nên $m=2$ là giá trị cần tìm.
	\item Hai đường thẳng trùng nhau khi $\heva{&m+1=2m-1\\&-(2m+1)= 3m}\Leftrightarrow \heva{&m=2\\&m=-\dfrac{1}{5}}$ (vô lý).\\
	Vậy không có giá trị nào của $m$ để hai đường thẳng trùng nhau. Nói cách khác hai đường thẳng trên không bao giờ trùng nhau.
\end{enumEX}
}
\end{vd}
\begin{vd}
	Cho hai đường thẳng $(d)\colon y=-x+m+2$ và $\left(d'\right)\colon y=(m^2-2)x+3$. Tìm $m$ để $(d)$ và $\left(d'\right)$ song song với nhau
	\loigiai{
		Điều kiện để hai đường thẳng song song là\\
		$\heva{&m^2-2=-1\\&m+2\neq 3}\Leftrightarrow \heva{&m=\pm 1\\&m\neq 1}\Leftrightarrow m=-1$.\\
		Vậy $m=-1$ là giá trị cần tìm.
	}
\end{vd}
\begin{vd}
	Tìm giá trị của tham số $k$ để đường thẳng $(d_1)\colon y=-x+2$ cắt đường thẳng $(d_2)\colon y=2x+3-k$ tại một điểm nằm trên trục hoành.
	\loigiai{
		Ta thấy hai đường thẳng $(d_1)$; $(d_2)$ luôn cắt nhau.
		\begin{itemize}
			\item Đường thẳng $(d_1)$ cắt trục hoành tại điểm $A\left(2; 0\right)$;
			\item Đường thẳng $(d_2)$ cắt trục hoành tại điểm $B\left(\dfrac{k-3}{2}; 0\right)$.
		\end{itemize}
		Để hai đường thẳng $(d_1)$; $(d_2)$ cắt nhau tại một điểm trên trục hoành thì $\dfrac{k-3}{2}=2\Leftrightarrow k=7$. 
	}
\end{vd}

\begin{vd}
	Tìm giá trị của $m$ để hai đường thẳng $(d_1): mx+y=1$ và $(d_2): x-my=m+6$ cắt nhau tại một điểm $M$ thuộc đường thẳng $(d)\colon x+2y=8$.
	\loigiai{
		Để hai đường thẳng $(d_1)$, $(d_2)$ cắt nhau thì $ \dfrac{m}{1}\neq \dfrac{1}{-m}\Rightarrow {m}^{2}\neq -1$ luôn thỏa mãn với mọi $m$.\\
		$(d)\colon x+2y=8\Rightarrow x=8-2y\quad (1)$;\\
		$(d_1)\colon mx+y=1\Rightarrow m=\dfrac{1-y}{x}\quad $;\\
		$(d_2)\colon x-my=m+6\Rightarrow m=\dfrac{x-6}{1+y}\quad (2)$.\\
		Do đó $ \dfrac{1-y}{x}=\dfrac{x-6}{1+y}\Rightarrow 1-{y}^{2}={x}^{2}-6x$
		$ \Leftrightarrow {x}^{2}-6x+{y}^{2}-1=0 \quad (3)$.\\
		Thay $(1)$ vào $(3)$ ta được tung độ giao điểm $M$ là nghiệm phương trình\\
		$\left(8-2y\right)^2-6\left(8-2y\right)+y^2=1\Leftrightarrow 5y^2-20y+15=0\Rightarrow {y}_{1}=1$ hoặc ${y}_{2}=6$.
		\begin{itemize}
			\item Với $y_1=1\Rightarrow x_1=6$ thay $(6; 1)$ vào $(2)$ ta được $m=0$ (thỏa mãn).
			\item Với $y_2=3\Rightarrow x_2=2$ thay $(2;3)$ vào $(2)$ ta được $m=-1$ (thỏa mãn)
		\end{itemize}
		Vậy với $m=0$ hoặc $m=-1$ thì hai đường thẳng $\left(d_1\right)$ và $(d_2$) cắt nhau tại một điểm $M$ thuộc đường thẳng $(d)$. 
	}
\end{vd}
\subsection{Luyện tập}
\begin{bt}
	Xét vị trí tương đối của các cặp đường thẳng sau
	\begin{enumEX}{2}
		\item $y=2x+1$ và $y=3x+2$;
		\item $y=5x+1$ và $y=5x+2$;
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item $y=2x+1$ và $y=3x+2$ cắt nhau vì $a=2\neq a'=3$;
			\item $y=5x+1$ và $y=5x+2$ song song với nhau vì $a=a'=5$ và $b=1\neq b'=2$.
		\end{enumEX}
	}
\end{bt}
\begin{bt}
	Tìm giao điểm của đường thẳng $y=2x+3$ với
	\begin{enumEX}{1}
		\item đường thẳng $y=3x+1$;
		\item trục $Ox$;
		\item trục $Oy$.
	\end{enumEX}
\loigiai{	
	\begin{enumEX}{1}
		\item Phương trình hoành độ giao điểm của đường thẳng $y=2x+3$ với đường thẳng $y=3x+1$ là\\
		$2x+3=3x+1\Leftrightarrow x=2$. Thay $x=2$ vào một trong hai phương trình của hai đường thẳng suy ra $y=7$.\\
		Vậy giao điểm của hai đường thẳng là điểm $A(2;7)$.
		\item Vì giao điểm của đường thẳng $y=2x+3$ với trục $Ox$ nằm trên trục hoành nên $y=0$ thay vào phương trình $y=2x+3$ ta có $2x+3=0\Rightarrow x=-\dfrac{3}{2}$. \\
		Vậy giao điểm của đường thẳng $y=2x+3$ với trục hoành là điểm có hoành độ bằng $-\dfrac{3}{2}$.
		\item Cho $x=0$, thay vào phương trình $y=2x+3\Rightarrow y=2\cdot 0+3=3$.\\
		Vậy giao điểm của đường thẳng $y=2x+3$ với trục tung là điểm $A(0;3)$.
\end{enumEX}
}
\end{bt}
\begin{bt}	
	Viết phương trình đường thẳng $(d)$ song song với đường thẳng $y=3x+1$ và cắt trục tung tại điểm có tung độ bằng $4$
	\loigiai{
		Đường thẳng $(d)$ song song với đường thẳng $y=3x+1$ nên $(d)$ có dạng $ y=3x+b$ ($b\in \mathbb{R}$).\\
		Đường thẳng $(d)$ cắt trục tung tại điểm có tung độ bằng $ 4$ nên $(d)$ đi qua điểm $ A\left(0; 4\right)$ hay $ 4=3\cdot 0+b\Leftrightarrow b=4$.\\
		Vậy phương trình đường thẳng $(d)$ là $y=3x+4$.
	}
\end{bt}

\begin{bt}
	Xác định hàm số trong mỗi trường hợp sau, biết đồ thị của nó là một đường thẳng đi qua gốc tọa độ và 
	\begin{enumEX}{1}
		\item song song với đường thẳng $y=2x+5$;
		\item cắt đường thẳng $y=2x-1$ tại điểm có hoành độ bằng $5$.
	\end{enumEX}
	\loigiai{Đường thẳng cần tìm đi qua gốc tọa độ nên phương trình có dạng $y=ax$.
		\begin{enumEX}{1}
			\item Vì dường thẳng $y=ax$ song song với đường thẳng $y=2x+5$ nên ta có $a=2$.\\
