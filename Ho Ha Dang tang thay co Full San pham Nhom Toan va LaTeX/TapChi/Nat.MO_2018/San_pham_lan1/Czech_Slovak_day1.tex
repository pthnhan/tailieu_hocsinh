\begin{ex}%[HSG Czech and Slovak năm 2018, vòng 1]%[Nguyễn Anh Quốc, Dự án EX-Nat.MO-18]%[0D0G1]
	Gọi $x$, $y$, $z$ là các số thực sao cho 
	\begin{center}
		$\dfrac{1}{\left|x^2+2zy\right|}, \dfrac{1}{\left|y^2+2zx\right|},\dfrac{1}{\left|z^2+2xy\right|}$
	\end{center}
	là độ dài các cạnh của một tam giác (không suy biến). Tìm tất cả các giá trị của $xy+yz+zx.$\\
	Michal Rolinek
	\loigiai{
		Nếu $x=y=z=t>0$ thì ba phân số là ba cạnh của một tam giác đều và $xy+yz+zx=3t^2$, do đó $xy+yz+zx$ là tất cả các giá trị dương.\\
		Tương tự, vì $x=y=t>0$ và $z=-2t$ thì ba phân số  $\dfrac{1}{3}t^{-2}, \dfrac{1}{3}t^{-2},\dfrac{1}{6}^{-2}$ là các số dương là độ dài các cạnh của tam giác cân $\left(\dfrac{1}{6}<\dfrac{1}{3}+\dfrac{1}{3} \right)$. Vậy $xy+yz+zx=-3t^2$ thỏa mãn với mọi giá trị âm.\\
		Tiếp theo chúng ta thấy rằng $xy+yz+zx$ không thể bằng $0$. Theo giả thiết, các số thực $x,y,z$ phân biệt: Nếu nói $x$ và $y$ bằng nhau thì mẫu số của các phân số bằng nhau $\left|x^2+2yz\right|=\left|xy+(yz+xz)\right|=0$ điều này không thể xảy ra.\\
		Ta thấy rằng, các phân số không có giá trị tuyệt đối, hiệu $xy+yz+zx=0$ từ  mỗi mẫu số, chúng ta có\\
		\begin{eqnarray*}
			&&\dfrac{1}{x^2+2yz}+\dfrac{1}{y^2+2zx}+\dfrac{1}{z^2+2xy}\\
			&=&\dfrac{1}{(x-y)(x-z)}+\dfrac{1}{(y-z)(y-x)}+\dfrac{1}{(z-x)(z-y)}\\
			&=&\dfrac{(z-y)+(x-z)+(y-x)}{(x-y)(y-z)(z-x)}=0.
		\end{eqnarray*}
		Điều này có nghĩa là trong số các mẫu số chung (giá trị tuyệt đối) là tổng của hai số còn lại. Vậy các phân số không thỏa mãn bất đẳng thức tam giác và chúng ta thấy sự mâu thuẫn.\\
		Trả lời: Vậy giá trị có thể là tất cả các số thực, ngoại trừ số $0$.}
\end{ex}

\begin{ex}%[HSG Czech and Slovak năm 2018, vòng 1]%[Nguyễn Anh Quốc, Dự án EX-Nat.MO-18]%[9H0G3]
	Cho tam giác $ABC$, đường phân giác góc $A$ cắt $BC$ tại $D$. Gọi $E$, $F$ lần lượt là tâm của đường tròn ngoại tiếp tam giác $ABD$ , $ACD$. Biết rằng tâm của đường tròn ngoại tiếp tam giác $AEF$ nằm trên $BC$. Tìm tất cả các giá trị có thể của góc $\widehat{BAC}$. \quad (Patrik Bak).	
	\loigiai{
		\immini{
			Gọi $O$ là tâm của đường tròn ngoại tiếp tam giác $AEF$ và kí hiệu $\alpha=\widehat{BAC}$. Vì $\widehat{BAD}$ và $\widehat{ACD}$ là các góc nhọn, các điểm $E$, $F$ nằm trong nửa mặt phẳng $BCA$ và theo định lí góc nội tiếp ta có
			\begin{center}
				$\widehat{BED}=2\cdot\widehat{BAD}=\alpha=2\cdot\widehat{DAC}=\widehat{DFC}$.
			\end{center}
			Các tam cân $BED$ và $DFC$ đồng dạng, dễ thấy $\widehat{EDF}=\alpha$ và $BC$ là đường phân giác ngoài góc $D$ trong tam giác $DEF$.\\
			Điểm $O$ nằm trên $BC$ và đường trung trực của đoạn $EF$. Xét tam giác $DEF$, $O$ nằm trên đường phân giác ngoài góc $D$ và nằm trên đường trung trực của $EF$. Vì vậy nó là trung điểm của cung $EDF$ và $\widehat{EOF}=\widehat{EDF}=\alpha$.\\
			Ta có tứ giác $AEDF$ có $\widehat{EAF}=\alpha$. Tuy nhiên, đường thẳng $EF$ không đi qua điểm $A$ và $O$. Vì vậy theo định lí góc nội tiếp ta có độ lớn  góc $EOF$ bằng hai lần góc $EAF$. Từ đây ta có $360^\circ \alpha=2\cdot \alpha \Rightarrow \alpha =120^0$.
			
		}
		{\begin{tikzpicture}[scale=1, line join=round, line cap=round]
			\tkzDefPoints{0/0/O,4/0/C,-5/0/B}
			\tkzDefTriangle[two angles = 30 and 20](B,C)
			\tkzGetPoint{A}
			\tkzDrawBisector(B,A,C)
			\tkzGetPoint{D}
			\tkzDrawPolygon(A,B,C)
			\tkzCircumCenter(A,B,D)\tkzGetPoint{E}
			\tkzCircumCenter(A,D,C)\tkzGetPoint{F}
			\tkzDrawPoints[size=5,fill=black](A,B,C,O,E,F,D)
			\tkzDrawSegments(E,A E,B E,D F,A F,D F,C E,F A,D)
			\tkzDrawCircle[radius](O,A)
			\tkzLabelPoints[below](B,C,O,D)
			\tkzLabelPoints[above](A,F,E)
			\tkzMarkAngles[size=0.3cm,arc=l](B,A,C)
			\tkzMarkAngles[size=0.3cm,arc=l](D,A,C)
			\tkzMarkAngles[size=0.3cm,arc=ll](B,E,D)
			\tkzMarkAngles[size=0.3cm,arc=ll](B,E,D)
			\tkzMarkAngles[size=0.3cm,arc=ll](F,D,E)
			\tkzMarkAngles[size=0.3cm,arc=ll](D,F,C)
			\end{tikzpicture}}	
	}
\end{ex}
\begin{ex}%[HSG Czech and Slovak năm 2018, vòng 1]%[Nguyễn Anh Quốc, Dự án EX-Nat.MO-18]%[0D0G5]
	Xét những số nguyên dương $a$,$b$,$c$ là cạnh của các tam giác không suy biến sao cho GCD$(a,b,c)=1$ và các phân thức 
	\begin{center}
		$\dfrac{a^2+b^2-c^2}{a+b-c}$,$\dfrac{b^2+c^2-a^2}{b+c-a}$,$\dfrac{c^2+a^2-b^2}{c+a-b}$
	\end{center}
	đều là số tự nhiên. Chứng minh rằng tích các mẫu thức của ba phân thức trên là một bình phương hoặc hai lần bình phương của một số nguyên.
	\loigiai{
		Đặt $z=a+b-c$, $x=b+c-a$, $y=c+a-b$ là các mẫu số (dương). Khi đó $a=\dfrac{y+z}{2}$,$b=\dfrac{x+z}{2}$, $c=\dfrac{x+y}{2}$ và \\
		$a^2+b^2-c^2=\dfrac{1}{4}\left((y+z)^2+(x+z)^2-(x+y)^2 \right)=\dfrac{1}{2}\left( z(z+x+y)-xy\right)$, vì $z|xy$, $y|xz$, $x|yz$.\\
		Vì $p$ là số nguyên tố, đặt $i_p$ là số mũ lớn nhất sao cho $p^{i_p}|xyz$. Điều đó đủ để chỉ ra rằng tất cả các số nguyên tố lẻ tương ứng với $i_p$ là số chẵn. Nếu $i_2$ cũng là số chẵn thì $xyz$ là một bình phương. Ngược lại là hai lần bình phương.
	}
\end{ex}
