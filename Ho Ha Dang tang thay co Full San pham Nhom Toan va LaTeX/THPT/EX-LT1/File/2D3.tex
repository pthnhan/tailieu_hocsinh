\subsection{Nguyên hàm}
\begin{ex}%[Nguyễn Thành Khang]%[2D3B1-1]
	Cho hai mệnh đề sau:
	\begin{enumerate}[(1)]
		\item Hàm số $f(x)=|x|$ không có đạo hàm tại $x=0$ nên $\displaystyle \int f(x)\mathrm{\,d}x$ không tồn tại.
		\item Hàm số $y=g(x)$ xác định trên $\mathbb{R}$ thoả mãn $g(x)\ge 0, \forall x\in\mathbb{R}$ thì $\displaystyle \int g(x)\mathrm{\,d}x\ge 0,\forall x\in\mathbb{R}$.
	\end{enumerate}
	Nhận xét nào sau đây là đúng?
	\choice
	{Mệnh đề 1 đúng, mệnh đề 2 đúng}
	{Mệnh đề 1 đúng, mệnh đề 2 sai}
	{Mệnh đề 1 sai, mệnh đề 2 đúng}
	{\True Mệnh đề 1 sai, mệnh đề 2 sai}
	\loigiai{
		Ta có định lý: ``Mọi hàm số liên tục trên $\mathbb{R}$ đều có nguyên hàm trên $\mathbb{R}$'' nên mệnh đề 1 sai (có thể chỉ ra $\displaystyle \int f(x)\mathrm{\,d}x=\dfrac{x|x|}{2}+C$ cũng dẫn tới mệnh đề 1 sai).\\
		Nếu $g(x)=3x^2$ thì $G(x)=x^3$ là một nguyên hàm của $g(x)$ nhưng ta không có $G(x)\ge 0, \forall x\in\mathbb{R}$, nên mệnh đề 2 sai.
	}
\end{ex}

\begin{ex}%[Nguyễn Thành Khang]%[2D3B1-1]
	Cho $f(x),\ g(x)$ là các hàm số có đạo hàm trên $\mathbb{R}$. Cho các mệnh đề sau:
	\begin{enumerate}[(1)]
		\item $\displaystyle \int kf(x)\mathrm{\,d}x=k\int f(x)\mathrm{\,d}x$, với mọi $k\in\mathbb{R}$.
		\item $\displaystyle \int \left[f(x)+g(x)\right]\mathrm{\,d}x=\int f(x)\mathrm{\,d}x+\int g(x)\mathrm{\,d}x$.
		\item Nếu $f(x)\ge g(x),\forall x\in\mathbb{R}$ thì $\displaystyle \int f(x)\mathrm{\,d}x\ge \int g(x)\mathrm{\,d}x,\forall x\in\mathbb{R}$.
	\end{enumerate}
	Số mệnh đề đúng là
	\choice
	{$0$}
	{\True $1$}
	{$2$}
	{$3$}
	\loigiai{
		Mệnh đề (1) sai khi $k=0$.\\
		Mệnh đề (2) đúng.\\
		Mệnh đề (3) sai nếu chọn $f(x)=x^2$ và $g(x)=0$.
	}
\end{ex}

	}

\begin{ex}%[2D3Y2-1]%[Nguyễn Vân Trường EX-LT1]
	Cho $f(x)$ là hàm số liên tục trên $[a;b]$ ($a <b$) và $F(x)$ là một nguyên hàm của $f(x)$. Khẳng định nào sau đây {\bf sai}?
	\choice
	{$\displaystyle\int\limits_a^b f(x) \mathrm{\,d}x =F(b)-F(a)$}
	{$\displaystyle\int\limits_a^b f(x) \mathrm{\,d}x = F(x) \Big|_a^b $}
	{$\displaystyle\int\limits_a^b f(x) \mathrm{\,d}x =\left.\left( \displaystyle\int\limits f(x) \mathrm{\,d}x \right)\right|_a^b$}
	{\True $\displaystyle\int\limits_a^b F(x) \mathrm{\,d}x =f(b)-f(a)$}
	\loigiai{
		Do $f(x)$ không phải là một nguyên hàm của $F(x)$ nên công thức $\displaystyle\int\limits_a^b F(x) \mathrm{\,d}x =f(b)-f(a)$ sai.
	}
\end{ex}

\begin{ex}%[2D3Y2-1]%[Nguyễn Vân Trường EX-LT1]
	Cho $f(x),g(x)$ là hàm số liên tục trên $[a;b]$ ($a <b$). Khẳng định nào sau đây {\bf sai}?
	\choice
	{$\displaystyle\int\limits_a^b [f(x)+g(x)] \mathrm{\,d}x =\displaystyle\int\limits_a^b f(x) \mathrm{\,d}x+\displaystyle\int\limits_a^b g(x) \mathrm{\,d}x$}
	{$\displaystyle\int\limits_a^b [f(x)-g(x)] \mathrm{\,d}x =\displaystyle\int\limits_a^b f(x) \mathrm{\,d}x-\displaystyle\int\limits_a^b g(x) \mathrm{\,d}x$}
	{$\displaystyle\int\limits_a^b [f(x)\cdot g(x)] \mathrm{\,d}x =\displaystyle\int\limits_a^b f(x) \mathrm{\,d}x\cdot \displaystyle\int\limits_a^b g(x) \mathrm{\,d}x$}
	{$\displaystyle\int\limits_a^b [k g(x)] \mathrm{\,d}x =k\displaystyle\int\limits_a^b f(x) \mathrm{\,d}x$ với mọi $k \in \mathbb{R}$ và $k\ne0$}
	\loigiai{
		Do $\displaystyle\int\limits_0^1 x \mathrm{\,d}x = \dfrac{1}{2}$, $\displaystyle\int\limits_0^1 x^2 \mathrm{\,d}x = \dfrac{1}{3}$. Suy ra
		$\displaystyle\int\limits_0^1 x \mathrm{\,d}x\cdot \displaystyle\int\limits_0^1 x \mathrm{\,d}x \ne \displaystyle\int\limits_0^1 x^2 \mathrm{\,d}x$. \\Vậy 
		công thức$\displaystyle\int\limits_a^b [f(x)\cdot g(x)] \mathrm{\,d}x =\displaystyle\int\limits_a^b f(x) \mathrm{\,d}x\cdot \displaystyle\int\limits_a^b g(x) \mathrm{\,d}x$ sai.
	}
\end{ex}

\begin{ex}%[2D3B3-5]%[Nguyễn Vân Trường EX-LT1]
	Một vật chuyển động với vận tốc thay đổi theo thời gian $v=f(t)$. Khẳng định nào sau đây {\bf sai}?
	\choice
	{Quãng đường mà vật đi được trong khoảng thời gian từ thời điểm $a$ đến thời điểm $b$ là  $\displaystyle\int\limits_a^b f(x) \mathrm{\,d}x$}
	{\True Gia tốc của vật tại thời điểm $t_0$ là $f^{''}(t_0)$}
	{Vận tốc của vật tại thời điểm $t_0$ là $f(t_0)$}
	{Quãng đường mà vật đi được sau khi xuất phát được một quãng thời gian  $t_0$ là  $\displaystyle\int\limits_0^{t_0} f(x) \mathrm{\,d}x$}
	\loigiai{
		Gia tốc của vật tại thời điểm $t_0$ là $f^{'}(t_0)$. Do đó mệnh đề \lq\lq Gia tốc của vật tại thời điểm $t_0$ là $f^{''}(t_0)$\rq\rq \, là mệnh đề sai.

\begin{ex}%[2D3B2-2]%[Nguyễn Vân Trường EX-LT1]
	Cho $f(x)$ là hàm số chẵn liên tục trên $[-a;a]$ ($a >0$). Khẳng định nào sau đây đúng
	\choice
	{$\displaystyle\int\limits_{-a}^a f(x) \mathrm{\,d}x = 0$ }
	{$\displaystyle\int\limits_{-a}^a f(x) \mathrm{\,d}x = \displaystyle\int\limits_{0}^a f(x) \mathrm{\,d}x $ }
	{$\displaystyle\int\limits_{-a}^a f(x) \mathrm{\,d}x = \displaystyle\int\limits_{a}^{-a} f(x) \mathrm{\,d}x $ }
	{\True $\displaystyle\int\limits_{-a}^a f(x) \mathrm{\,d}x = 2 \displaystyle\int\limits_{0}^a f(x) \mathrm{\,d}x $ }
	\loigiai{
		Ta có $\displaystyle\int\limits_{-a}^a f(x) \mathrm{\,d}x = \displaystyle\int\limits_{-a}^0 f(x) \mathrm{\,d}x+\displaystyle\int\limits_{0}^a f(x) \mathrm{\,d}x.$ \\
		Xét tích phân $ \displaystyle\int\limits_{-a}^0 f(x) \mathrm{\,d}x$. Đặt $t=-x$. \\
		Đổi cận: $ x = -a \Rightarrow t =a; x = 0 \Rightarrow t = 0$.\\
		Đổi vi phân: $  \mathrm{\,d}t =  \mathrm{\,d}(-x) \Rightarrow  \mathrm{\,d}x =-  \mathrm{\,d}t$.\\
		Suy ra  $ \displaystyle\int\limits_{-a}^0 f(x) \mathrm{\,d}x = - \displaystyle\int\limits_{t}^0 f(t) \mathrm{\,d}t =\displaystyle\int\limits_{0}^t f(t) \mathrm{\,d}t$. Do đó $ \displaystyle\int\limits_{-a}^0 f(x) \mathrm{\,d}x=\displaystyle\int\limits_{0}^a f(x) \mathrm{\,d}x$.\\
		Vậy $\displaystyle\int\limits_{-a}^a f(x) \mathrm{\,d}x = 2 \displaystyle\int\limits_{0}^a f(x) \mathrm{\,d}x $.
	}
\end{ex}
