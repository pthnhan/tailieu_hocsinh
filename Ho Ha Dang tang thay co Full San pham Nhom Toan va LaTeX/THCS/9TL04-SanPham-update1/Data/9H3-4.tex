		Gọi $(O)$ là đường tròn cần tìm. Kéo dài $DA$ cắt đường tròn $(O)$ tại $E$.\\
		Xét $\triangle DAM$ và $\triangle DME$ có
		\begin{center}
			$\heva{&\widehat{D} \text{ chung}\\&\widehat{DMA}=\widehat{DEM} \text{ cùng chắn cung } MA}\Rightarrow \triangle DAM\backsim \triangle DME.$
		\end{center}

		Xét $\triangle OKA$ vuông tại $K$ có $R=OA=\sqrt{OK^2+KA^2}=\sqrt{1^2+3^2}=\sqrt{10}$ $\mathrm{cm}$.
	}
\end{bt}

\begin{bt}

	\begin{enumerate}
		\item Chứng minh rằng $\dfrac{IB}{IC}=\dfrac{AB^2}{AC^2}.$
		\item Tính $IA$, $IC$ biết rằng $AB=20$ $\mathrm{cm}$, $AC=28$ $\mathrm{cm}$, $BC=24$ $\mathrm{cm}.$
	\end{enumerate}
	\loigiai{

			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Xét $\triangle BAI$ và $\triangle ACI$ có
			\begin{center}
				$\heva{&\widehat{I} \text{ chung}\\&\widehat{BAI}=\widehat{ICA}\text{ cùng chắn cung } AB}\Rightarrow \triangle BAI\backsim \triangle ACI$ (g.g). 
			\end{center}
			Từ đó suy ra $ \dfrac{AB}{AC}=\dfrac{IB}{IA}\Rightarrow \dfrac{AB^2}{AC^2}=\dfrac{IB^2}{IA^2}.$\\
			Vì $\triangle ABI\backsim \triangle ACI\Rightarrow \dfrac{IB}{IA}=\dfrac{IA}{IC}\Rightarrow IA^2=IB\cdot IC$ nên $\dfrac{AB^2}{AC^2}=\dfrac{IB^2}{IB\cdot IC}=\dfrac{IB}{IC}.$
			\item Đặt $IA=x$, $IC=y$. Ta có
			\begin{center}
				$\triangle ABI\backsim \triangle ACI\Rightarrow \dfrac{AI}{CI}=\dfrac{BI}{AI}=\dfrac{AB}{CA}\Rightarrow \dfrac{x}{y}=\dfrac{y-24}{x}=\dfrac{20}{28}=\dfrac{5}{7}.$
			\end{center}
			Từ đó ta có hệ $\heva{&7x=5y\\&5x=7(y-24)}\Leftrightarrow \heva{&35x=25y\\&35x=49(y-24)}\Leftrightarrow \heva{&7x=5y\\&y=49}\Leftrightarrow \heva{&x=35\\&y=49.}$\\
			Vậy $IA=35$ $\mathrm{cm}$, $IC=49$ $\mathrm{cm}.$
		\end{enumerate}
	}
\end{bt}

	}
\end{bt}
