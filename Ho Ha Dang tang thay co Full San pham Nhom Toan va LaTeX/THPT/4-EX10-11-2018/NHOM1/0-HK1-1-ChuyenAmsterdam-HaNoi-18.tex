{$0$}
{$1$}
{$2$}
{\True $4$}

\end{ex}

\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[HKI, Chuyên Hà Nội - Amsterdam, 2018]%[0D2B3]%[0D2K3]
	Cho hàm số $y=x^2-3mx+m^2+1\quad (1)$, $m$ là tham số.
		\begin{enumerate}
			\item Khảo sát sự biến thiên và vẽ đồ thị của hàm số $(1)$ khi $m=1$.
			\item Cho đường thẳng $(d)$ có phương trình $y=mx+m^2$. Tìm giá trị của tham số $m$ để đồ thị của hàm số $(1)$ cắt đường thẳng $(d)$ tại hai điểm phân biệt có hoành độ $x_1, x_2$ thỏa mãn $\left|\sqrt{x_1}-\sqrt{x_2}\right|=1$.
		\end{enumerate}

	\loigiai{
		\begin{enumerate}
			\item Khi $m=1$ thì $y=x^2-3x+2$.
                     \begin{itemize}
\item Tọa độ đỉnh: $I\left(\dfrac{3}{2};-\dfrac{1}{4}\right)$.
\item Phương trình trục đối xứng: $x=\dfrac{3}{2}$.
\item Bảng biến thiên:
\begin{center}
\begin{tikzpicture}
\tkzTabInit[lgt=1.5,espcl=2.0,deltacl=.5]
{$x$ /1.0,$y$ /2.0}
{$-\infty$ , $\dfrac{3}{2}$ , $+\infty$}
\tkzTabVar{+/$+\infty$ ,-/ $-\dfrac{1}{4}$ , +/$+\infty$}
\end{tikzpicture}
\end{center}
\item Bảng giá trị:
\begin{center}
\renewcommand{\arraystretch}{1.25}
\begin{tabular}{|c|c|c|c|c|c|}
\hline
$x$&$0$&$1$&$\frac{3}{2}$&$2$&$3$\\ 
\hline
$y$&$2$&$0$&$-\frac{1}{4}$&$0$&$2$\\ 
\hline
\end{tabular}
\end{center}
\item Đồ thị:
\begin{center}
\begin{tikzpicture}[scale=0.7,thick,>=stealth]
\draw[->] (-1.5,0) -- (4,0) node [above left] {\tiny $x$};
\draw[->] (0,-1.0) -- (0,4.5) node [right] {\tiny $y$};
\draw[dashed,thin] (0,2)node[above right]{\tiny 2};
\draw[dashed,thin] (1.5,-1.0)--(1.5,4.5);
\draw[dashed,thin](1,0)node[below]{\tiny 1};
\draw[dashed,thin](2,0)node[below]{\tiny 2};
\draw (-0.3,0) node[above] {\tiny $O$};
\clip (-4.5,-4.5) rectangle (4.5,4.5);
\draw[smooth,samples=100,domain=-1.5:4] plot(\x,{(\x)^2-3*(\x)+2});
\end{tikzpicture}
\end{center}
\end{itemize}
			\item Phương trình hoành độ giao điểm: $x^2-3mx+m^2+1=mx+m^2\Leftrightarrow x^2-4mx+1=0\quad (*)$.\\Đồ thị hàm số $(1)$ cắt $(d)$ tại hai điểm phân biệt có hoành độ $x_1, x_2\Leftrightarrow (*)$ có hai nghiệm phân biệt $x_1, x_2\Leftrightarrow \Delta'=4m^2-1>0\Leftrightarrow \left|m\right|>\dfrac{1}{2}$. \\
Ta có $\left|\sqrt{x_1}-\sqrt{x_2}\right|=1\Leftrightarrow \left(\sqrt{x_1}-\sqrt{x_2}\right)^2=1\Leftrightarrow x_1+x_2-2\sqrt{x_1x_2}=1$.\\Áp dụng định lý Vi-et ta có $4m-2\cdot 1=1\Leftrightarrow m=\dfrac{3}{4}$ (thỏa điều kiện).\\Thử lại với $m=\dfrac{3}{4}$ thì $(*)$ thành $x^2-3x+1=0\Leftrightarrow x=\dfrac{3\pm\sqrt{5}}{2}$ thỏa $\left|\sqrt{x_1}-\sqrt{x_2}\right|=1$.
		\end{enumerate}
