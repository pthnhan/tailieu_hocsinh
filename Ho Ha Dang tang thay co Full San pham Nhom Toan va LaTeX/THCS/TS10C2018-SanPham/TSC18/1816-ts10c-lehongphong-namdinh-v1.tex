\begin{enumerate}
\item Rút gọn biểu thức $P$.
\item Chứng minh rằng với mọi $x>0$, $x\neq 1$ thì $P>4$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Với $x>0;x\neq 1$ ta có
\begin{eqnarray*}
P & = & \dfrac{x-1}{\sqrt{x}}:\dfrac{\left(\sqrt{x}-1\right)\left(\sqrt{x}+1\right)+1-\sqrt{x}}{\sqrt{x}\left(\sqrt{x}+1\right)}\\
& = & \dfrac{x-1}{\sqrt{x}}\cdot \dfrac{\sqrt{x}\left(\sqrt{x}+1\right)}{x-\sqrt{x}}=\dfrac{\left(\sqrt{x}+1\right)^2}{\sqrt{x}}.
\end{eqnarray*}
\item Với $x>0;x\neq 1$ ta có các biến đổi tương đương
\begin{eqnarray*}
P>4 & \Leftrightarrow & \dfrac{\left(\sqrt{x}+1\right)^2}{\sqrt{x}}>4\\
& \Leftrightarrow & x+2\sqrt{x}+1>4\sqrt{x}\\
& \Leftrightarrow & \left(\sqrt{x}-1\right)^2>0\,\,(\textrm{luôn đúng, với mọi}\,\, x>0;x\neq 1).
\end{eqnarray*}
Vậy $P>4$ với mọi $x>0$, $x\neq 1$.
\end{enumerate}
}
\end{ex}

\loigiai{
Đặt $\sqrt{2x-y-1}=a$, $\sqrt{x+2y}=b$ ($a,b\ge 0$). Phương trình $(1)$ trở thành
$$\left(2b^2-1\right)a=\left(2a^2-1\right)b\Leftrightarrow (a-b)(2ab+1)=0\Leftrightarrow a=b.$$
Do đó $\sqrt{2x-y-1}=\sqrt{x+2y}\Leftrightarrow x=3y+1$. Thế vào phương trình $(2)$ ta được
\begin{eqnarray*}
&& x^2+8x+5-2(x+1)\sqrt{3x+1}=2\sqrt{2x^2+5x+2}\\
&\Leftrightarrow & \left(x+1-\sqrt{3x+1}\right)^2+\left(\sqrt{2x+1}-\sqrt{x+2}\right)^2=0\\
&\Leftrightarrow & \heva{& x+1-\sqrt{3x+1}=0\\ & \sqrt{2x+1}-\sqrt{x+2}=0}\Leftrightarrow x=1\Rightarrow y=0.
\end{eqnarray*}
Thử lại thấy $(x;y)=(1;0)$ thỏa mãn hệ phương trình. Vậy nghiệm hệ là $(x;y)=(1;0)$.
}

$$Q=\dfrac{11a+11b+12c}{\sqrt{8a^2+56}+\sqrt{8b^2+56}+\sqrt{4c^2+7}}.$$
\loigiai{
Áp dụng bất đẳng thức Cô-si (Cauchy, AM-GM) ta có
\begin{align*}

= &\,\, \sqrt{(4a+4b)(2a+4c)}\\
\le &\,\, \dfrac{4a+4b+2a+4c}{2}=3a+2b+2c.\tag{1}
\end{align*}
Tương tự ta cũng có 
\[\sqrt{8b^2+56}\le 2a+3b+2c\tag{2}\]
\[\sqrt{4c^2+7}\le \dfrac{a+b+4c}{2}\tag{3}\]
Từ (1), (2) và (3) suy ra 
$$Q\ge \dfrac{11a+11b+12c}{3a+2b+2c+2a+3b+2c+\tfrac{a+b+4c}{2}}=2.$$
