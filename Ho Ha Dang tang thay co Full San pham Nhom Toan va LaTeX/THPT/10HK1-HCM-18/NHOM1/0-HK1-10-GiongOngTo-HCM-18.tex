\begin{bt}%[Đề KT HKI THPT Giồng Ông Tố, HCM, 2018]%[Đỗ Đức Thảo, 10EX-HK1]%[0D3B2-3]
	Giải phương trình
	$\dfrac{2x - 5}{x-1} - 1 = \dfrac{5x-3}{3x+5}- \dfrac{26}{(x-1)(3x+5)} $.	
	\loigiai{
		ĐKXĐ: $x \ne 1, x \ne -\dfrac{5}{3}$. \\
		Với điều kiện trên, ta có
		\begin{eqnarray*}
			&& \dfrac{2x - 5}{x-1} - 1 = \dfrac{5x-3}{3x+5}- \dfrac{26}{(x-1)(3x+5)} \\
			\Leftrightarrow && \dfrac{(2x - 5)(3x+5)}{(x-1)(3x+5)} - \dfrac{(x-1)(3x+5)}{(x-1)(3x+5)} = \dfrac{(5x-3)(x-1)}{(x-1)(3x+5)} - \dfrac{26}{(x-1)(3x+5)} \\
			\Rightarrow && (2x - 5)(3x+5) - (x-1)(3x+5) = (5x-3)(x-1) - 26 \\
			\Leftrightarrow && -2x^2 + x + 3 = 0 \\
			\Leftrightarrow && \hoac{&x = -1 \text{ (nhận)} \\ &x = \dfrac{3}{2} \text{ (nhận)}.}
		\end{eqnarray*}
		Vậy tập nghiệm của phương trình là $S = \left\{-1 ; \dfrac{3}{2} \right\}$.
	}
	
\end{bt}
\begin{bt}%[Đề KT HKI THPT Giồng Ông Tố, HCM, 2018]%[Đỗ Đức Thảo, 10EX-HK1]%[0D3B2-4]
	Giải phương trình $2x - \sqrt{4x - x^2} = 2$.
	\loigiai{
	Điều kiện xác định	$4x - x^2 \ge 0 \Leftrightarrow 0 \le x \le 4$.\\
		Với điều kiện trên, ta có
		\begin{eqnarray*}
			&& 2x - \sqrt{4x - x^2} = 2 \\
			\Leftrightarrow &&  2x - 2 = \sqrt{4x - x^2}\\
			\Rightarrow     &&  4x^2 - 8x + 4 = 4x - x^2\\
			\Leftrightarrow && 5x^2 - 12x + 4 = 0 \\
			\Leftrightarrow && \hoac{&x = 2 \\ &x = \dfrac{2}{5}.}
		\end{eqnarray*}
		Cả hai giá trị này đều thỏa điều kiện xác định. Thử lại, ta thấy chỉ có $x = 2$ là nghiệm.\\
		Vậy tập nghiệm của phương trình là $S = \{ 2\}$.	
	}
\end{bt}

\begin{bt}%[Đề KT HKI THPT Giồng Ông Tố, HCM, 2018]%[Đỗ Đức Thảo, 10EX-HK1]%[0D3B3-4]
	Giải hệ phương trình $\heva{&2x + y = 4 &(1) \\ &x^2 + xy + y^2 = 7 &(2)}$	
	\loigiai{
		Từ phương trình (1) ta có $y = 4 - 2x$. Thế vào phương trình (2) ta thu được
		$$x^2 + x(4 - 2x) + (4 - 2x)^2 = 7 \Leftrightarrow 3x^2 - 12x + 9 = 0 \Leftrightarrow \hoac{&x = 1 \\&x = 3.} $$
		Với $x = 1$, thay vào $ (1) $ ta thu được $y = 4 - 2\cdot 1 = 2$. \\
		Với $x = 3$, thay vào $ (1) $ ta thu được $y = 4 - 2 \cdot 3 = -2$. \\
		Vậy hệ có hai nghiệm $(x;y) = (1;2)$ và $(x;y) = (3;-2)$.
	}
	
\end{bt}

\begin{bt}%[Đề KT HKI THPT Giồng Ông Tố, HCM, 2018]%[Đỗ Đức Thảo, 10EX-HK1]%[0D3B2-6]%[0D3K2-5]
		\begin{enumerate}
			\item Cho phương trình $x^2-2(2m-3)x+4m^2+4=0$ ($m$ là tham số). Tìm $m$ để phương trình có $2$ nghiệm phân biệt $x_1,x_2$.
			\item Cho phương trình $x^2-2(m+1)x+2m+1=0$ ($m$ là tham số). Tìm $m$ để phương trình có hai nghiệm phân biệt $x_1,x_2$ sao cho tổng bình phương của hai nghiệm đạt giá trị nhỏ nhất.
		\end{enumerate}
	
		\loigiai{
			\begin{enumerate}
				\item Ta có $\Delta '=(2m-3)^2-(4m^2+4)=4m^2-12m+9-4m^2-4=-12m+5$.\\
				Phương trình đã cho có $2$ nghiệm phân biệt $\Leftrightarrow \Delta '>0\Leftrightarrow -12m+5>0
				\Leftrightarrow m<\dfrac{5}{12}$.
				\item Ta có $\Delta = 4(m+1)^2-4(2m+1)=4m^2$.\\
				Phương trình có $2$ nghiệm phân biệt 
				$\Leftrightarrow 4m^2>0\Leftrightarrow m\neq 0$. \quad $(*)$\\
				Áp dụng định lí Vi-et, ta có $\heva{&x_1+x_2=2(m+1)\\&x_1\cdot x_2=2m+1.}$\\	$A=x_1^2+x_2^2=(x_1+x_2)^2-2x_1\cdot x_2=4(m+1)^2-2(2m+1)=4m^2+4m+2=(2m+1)^2+1\geqslant 1$.\\
				Dấu {\lq\lq}$=${\rq\rq} xảy ra $\Leftrightarrow m=-\dfrac{1}{2}$ (thỏa $(*)$).\\
				Vậy với $m=-\dfrac{1}{2}$ thì tổng bình phương của hai nghiệm đạt giá trị nhỏ nhất.
			\end{enumerate}
		}
	\end{bt}
	
	
\begin{bt}%[Đề KT HKI THPT Giồng Ông Tố, HCM, 2018]%[Đỗ Đức Thảo, 10EX-HK1]%[0H2B1-4]%[0H2B3-1]
	\begin{enumerate}[a)]
	\item Cho tam giác $ABC$ có $AB=5$, $BC=6$ và góc $ABC =45^\circ$. Tính $\vv{AB}\cdot \vv{BC}$.
	\item Cho tam giác $ABC$ đều cạnh $a\sqrt{2}$. Tính $\vv{AB}\cdot \vv{AC}$.
	\item Cho 2 vectơ $\vv{a}, \vv{b}$ thỏa mãn: $|\vv{a}|=1$, $|\vv{b}|=2$, $|\vv{a}-2\vv{b}|=\sqrt{15}$. Tính $\vv{a}\cdot \vv{b}$.
	\item Cho hình thoi $ABCD$ cạnh $a\sqrt{5}$, góc $BAD=120^\circ$. Tính $(\vv{AB}+\vv{AD})\cdot (\vv{AC}-\vv{BD})$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item Ta có $\vec{AB}\cdot \vec{BC}=-\vec{BA}\cdot \vec{BC} = -AB\cdot BC \cdot \cos \widehat{ABC}=-5\cdot 6 \cdot \dfrac{\sqrt{2}}{2}=-15\sqrt{2}$.
			\item Do $\triangle ABC$ đều nên $\widehat{BAC}=60^\circ$.\\
			Ta có $\vec{AB}\cdot \vec{AC}=AB\cdot AC \cdot \cos \widehat{BAC}=a\sqrt{2}\cdot a\sqrt{2} \cdot \cos 60^\circ=a^2$.
			\item Ta có $ \vec{a}^2= \left|\vec{a} \right|^2=1$; $ \vec{b}^2= \left|\vec{b} \right|^2=4$.\\
			Do đó $15=\left| \vec{a}-2\vec{b} \right| ^2=\left( \vec{a}-2\vec{b} \right) ^2=\vec{a}^2-4\vec{a}\cdot\vec{b}+4\vec{b}^2=17-4\vec{a}\cdot\vec{b} \Rightarrow \vec{a}\cdot \vec{b}=\dfrac{1}{2}$.\\
			Vậy $\vec{a}\cdot \vec{b}=\dfrac{1}{2}$.
			\item Vì $ABCD$ là hình thoi nên $\widehat{BAC}=\dfrac{\widehat{BAD}}{2}=60^\circ$.
		\immini{Tam giác $ABC$ có $AB=BC$, $\widehat{BAC}=60^\circ$ nên tam giác $ABC$ là tam giác đều. Do đó $AB=AC=a\sqrt{5}$.

			&=2\cdot (a\sqrt{5})^2 =10a^2\text{ (Vì $\vv{AC}\perp  \vv{BD}$ nên $\vv{AC}\cdot \vv{BD}=0$)}
