\section{Tính chất cơ bản của phân số}
\section{Rút gọn phân số}
\subsection{Trọng tâm lý thuyết}
\begin{enumerate}
	\item Nếu ta nhân cả tử và mẫu của một phân số với cùng một số nguyên khác $0$ thì ta được một phân số bằng phân số đã cho.
	\[ \dfrac{a}{b}=\dfrac{a\cdot m}{b\cdot m} \text{ với } m \in \mathbb{Z} \text{ và } m \neq 0. \]
	Nếu ta chia cả tử và mẫu của một phân số cho cùng một ước chung của chúng thì ta được một phân số bằng phân số đã cho.
	\[ \dfrac{a}{b}=\dfrac{a:n}{b:n} \text{ với } n\in \text{ƯC}(a,b). \]
	\item Muốn rút gọn phân số, ta chia cả tử và mẫu của phân số cho một ước chung (khác $1$ và $-1$) của chúng.
	\item Phân số tối giản là phân số mà tử và mẫu chỉ có ước chung là $1$ và $-1$.\\
	Chỉ cần chia tử và mẫu của phân số cho ƯCLN của chúng, ta sẽ được một phân số tối giản.
\end{enumerate}

\subsection{Các dạng toán và phương pháp giải}

\begin{dang}{Viết các phân số bằng nhau}
Áp dụng tính chất cơ bản của phân số.
\end{dang}

\begin{vd}%[Võ Thị Quỳnh Trang, dự án EX-C2-6-2018]%[6D3B3]
Viết $5$ phân số bằng phân số $\dfrac{-4}{7}$.
\loigiai{
Ta nhân cả tử và mẫu của phân số $\dfrac{-4}{7}$ lần lượt với $2$; $3$; $4$; $5$; $6$ ta sẽ được các phân số bằng phân số $\dfrac{-4}{7}$. \\
Đó là các phân số
\[ \dfrac{-8}{14}; ~ \dfrac{-12}{21}; ~\dfrac{-16}{28}; ~\dfrac{-20}{35}; ~\dfrac{-24}{42}. \]
}
\end{vd}

\begin{vd}%[Võ Thị Quỳnh Trang, dự án EX-C2-6-2018]%[6D3K3]
Viết dạng tổng quát của các phân số bằng phân số
\begin{enumEX}{2}
		\item $\dfrac{-3}{5}$
		\item $\dfrac{12}{18}$
\end{enumEX}
\loigiai{
\begin{enumEX}{1}
	\item $\dfrac{-3}{5}$ là phân số tối giản nên dạng tổng quát của các phân số bằng phân số $\dfrac{-3}{5}$ là $$\dfrac{-3 \cdot n}{5 \cdot n} ~ (n \in \mathbb{Z} \text{ và } n \neq 0).$$
	\item $\dfrac{12}{18}$ chưa phải là phân số tối giản nên ta cần rút gọn
	\[ \dfrac{12}{18}=\dfrac{12:6}{18:6}=\dfrac{2}{3}. \]
	Do đó dạng tổng quát của các phân số bằng phân số $\dfrac{12}{18}$ là $\dfrac{2\cdot n}{3 \cdot n}$ với $n\in \mathbb{Z}$ và $n\neq 0$.
\end{enumEX}
}
\end{vd}

\begin{vd}%[Võ Thị Quỳnh Trang, dự án EX-C2-6-2018]%[6D3K3]
	Giải thích tại sao khi đổi dấu cả tử và mẫu của một phân số thì được một phân số bằng phân số đã cho?\\
	\textit{Áp dụng:} Viết các phân số sau thành các phân số bằng chúng và có mẫu dương
	\[ \dfrac{4}{-11}; ~ \dfrac{-9}{-10}; ~\dfrac{m}{n} ~ (m,n \in \mathbb{Z}; n<0). \]
	\loigiai{Ta có $\dfrac{a}{b}=\dfrac{a\cdot(-1)}{b\cdot (-1)}=\dfrac{-a}{-b}.$\\
	Đổi dấu cả tử và mẫu của một phân số chính là nhân cả tử và mẫu của một phân số với $-1$ nên ta được một phân số bằng phân số đã cho.\\
	\textit{Áp dụng:} $\dfrac{4}{-11}=\dfrac{-4}{11}$; $\dfrac{-9}{-10}=\dfrac{9}{10}$; $\dfrac{m}{n}=\dfrac{-m}{-n}.$
	}
\end{vd}

\begin{vd}%[Võ Thị Quỳnh Trang, dự án EX-C2-6-2018]%[6D3G3]
	Tìm các số nguyên $x$ và $y$ biết
	\begin{enumEX}{2}
			\item $\dfrac{x}{y}=\dfrac{4}{5}.$
			\item $\dfrac{x}{7}=\dfrac{y}{3}.$
	\end{enumEX}
	\loigiai{
	\begin{enumEX}{1}
		\item Ta có $\dfrac{x}{y}=\dfrac{4}{5}=\dfrac{4n}{5n} ~ (n \in \mathbb{Z}; n\neq 0).$\\
		Suy ra $x=4n$; $y=5n.$
		\item $\dfrac{x}{7}=\dfrac{y}{3}$ suy ra $x\cdot 3=7 \cdot y$.\\
		Do đó $\dfrac{x}{y}=\dfrac{7}{3}=\dfrac{7 \cdot n}{3 \cdot n} ~ (n \in \mathbb{Z}; n\neq 0).$\\
		Vậy $x=7n$; $y=3n.$
	\end{enumEX}
}
\end{vd}

\begin{dang}{Rút gọn phân số}
	Chia cả tử và mẫu của phân số $\dfrac{a}{b}$ cho một ước chung của $a$ và $b$ hoặc cho ƯCLN($|a|,|b|$). 
\end{dang}

\begin{vd}%[Võ Thị Quỳnh Trang, dự án EX-C2-6-2018]%[6D3B4]
	Rút gọn các phân số sau:
	\begin{enumEX}{2}
			\item $\dfrac{-42}{105};$
			\item $\dfrac{-75}{-135}.$
	\end{enumEX}
	\loigiai{
	\begin{enumEX}{1}
		\item $\dfrac{-42}{105}=\dfrac{-42:21}{105:21}=\dfrac{-2}{5}.$
		\item $\dfrac{-75}{-135}=\dfrac{(-75):15}{(-135):15}=\dfrac{-5}{-9}=\dfrac{5}{9}.$
	\end{enumEX}	
}
\end{vd}

\begin{vd}%[Võ Thị Quỳnh Trang, dự án EX-C2-6-2018]%[6D3K4]
	Rút gọn các phân số sau
	\begin{enumEX}{2}
			\item $\dfrac{72 \cdot 75}{125 \cdot 108}.$
			\item $\dfrac{2^3 \cdot 3^3 \cdot 35}{2^4 \cdot 3^2 \cdot 21}.$
	\end{enumEX}
	\loigiai{
	\begin{enumEX}{1}
		\item $\dfrac{72 \cdot 75}{125 \cdot 108}=\dfrac{(36 \cdot 2) \cdot (25 \cdot 3)}{(25\cdot 5)\cdot (36 \cdot 3)}=\dfrac{(36\cdot 25 \cdot 3) \cdot 2}{(36 \cdot 25 \cdot 3)\cdot 5}=\dfrac{2}{5}.$
		\item $\dfrac{2^3 \cdot 3^3 \cdot 35}{2^4 \cdot 3^2 \cdot 21}=\dfrac{2^3 \cdot 3^3 \cdot 5 \cdot 7}{2^4 \cdot 3^2 \cdot 3 \cdot 7}=\dfrac{(2^3 \cdot 3^3 \cdot 7) \cdot 5}{2 \cdot (2^3 \cdot 3^3 \cdot 7)}=\dfrac{5}{2}.$
	\end{enumEX}	
}
\end{vd}

\begin{vd}%[Võ Thị Quỳnh Trang, dự án EX-C2-6-2018]%[6D3K4]
	Rút gọn các phân số
	\begin{enumEX}{2}
			\item $\dfrac{11\cdot 3 +11 \cdot 4}{7 \cdot 9+7\cdot 13}.$
			\item $\dfrac{18 \cdot 13-13\cdot 3}{15\cdot 40-80}.$
	\end{enumEX}
	\loigiai{ 
	\begin{enumEX}{1}
		\item $\dfrac{11\cdot 3 +11 \cdot 4}{7 \cdot 9+7\cdot 13}=\dfrac{11\cdot (3+4)}{7\cdot (9+13)}=\dfrac{11 \cdot 7}{7\cdot 22}=\dfrac{1}{2}.$
		\item $\dfrac{18 \cdot 13-13\cdot 3}{15\cdot 40-80}=\dfrac{13\cdot (18-3)}{40\cdot (15-2)}=\dfrac{13 \cdot 15}{40 \cdot 13}=\dfrac{3}{8}.$
	\end{enumEX}
}
\end{vd}

\begin{vd}%[Võ Thị Quỳnh Trang, dự án EX-C2-6-2018]%[6D3B4]
	Đổi $48$ phút ra giờ và viết kết quả dưới dạng phân số tối giản.
	\loigiai{
	$48$ phút $=$ $\dfrac{48}{60}$ giờ $= \dfrac{4}{5}$ giờ. 	
}
\end{vd}

\begin{vd}%[Võ Thị Quỳnh Trang, dự án EX-C2-6-2018]%[6D3B4]
	Đổi $375$ cm$^2$ ra mét vuông và viết kết quả dưới dạng phân số tối giản.
	\loigiai{
	$375$ cm$^2$ $=\dfrac{375}{10000}$ m$^2$ $=\dfrac{375:125}{10000:125}$ m$^2$ $=\dfrac{3}{80}$ m$^2$.
}
\end{vd}

\begin{dang}{Nhận biết phân số tối giản}
	Nếu phân số $\dfrac{a}{b}$ có ƯCLN$(|a|,|b|)=1$ thì phân số $\dfrac{a}{b}$ là phân số tối giản.
\end{dang}

\begin{vd}%[Võ Thị Quỳnh Trang, dự án EX-C2-6-2018]%[6D3B4]
	Trong các phân số sau, phân số nào tối giản?
	$$ \dfrac{14}{-35};~\dfrac{9}{34};~\dfrac{-10}{48};~\dfrac{-12}{55}. $$
	\loigiai{
	ƯCLN$(14,35)=7$;\\
	ƯCLN$(9,34)=1$;\\
	ƯCLN$(10,48)=2$;\\
	ƯCLN$(12,55)=1$.\\
	Vậy các phân số tối giản là $\dfrac{9}{34};~\dfrac{-12}{55}.$
}
\end{vd}

\begin{vd}%[Võ Thị Quỳnh Trang, dự án EX-C2-6-2018]%[6D3G4]
	Tìm số tự nhiên $m$ nhỏ nhất khác $1$ để các phân số $\dfrac{15}{m};~\dfrac{28}{m}$ đều tối giản.
	\loigiai{
	Xét phân số $\dfrac{15}{m}$, có $15=3 \cdot 5$ nên phân số $\dfrac{15}{m}$ tối giản khi $m\neq 3k;~m \neq 5k$ ($k$ nguyên dương).\\
	Xét phân số $\dfrac{28}{m}$, có $28=2^2 \cdot 7$ nên phân số $\dfrac{28}{m}$ tối giản khi $m\neq 2k;~m \neq 7k$ ($k$ nguyên dương).\\	
	Vậy các phân số $\dfrac{15}{m};~\dfrac{28}{m}$ cùng tối giản khi $m\neq 2k;~m \neq 3k;~m\neq 5k;~m \neq 7k.$\\
	Mặt khác, $m$ là số tự nhiên nhỏ nhất khác $1$ nên ta chọn $m=11.$ 
}
\end{vd}

\begin{vd}%[Võ Thị Quỳnh Trang, dự án EX-C2-6-2018]%[6D3G4]
	Chứng minh rằng phân số $\dfrac{n}{n+1}$ với $n \in \mathbb{N}^*$ là phân số tối giản.
	\loigiai{
	Gọi $d$ là ƯCLN$(n;n+1)$.\\
	Ta có $n~\vdots ~ d;~n+1 ~ \vdots ~ d.$\\
	Suy ra $\left[ (n+1)-n \right]~\vdots ~ d$ hay $1 ~ \vdots ~d.$\\
	Do đó $d=1$. \\
	Vậy phân số $\dfrac{n}{n+1}$ là phân số tối giản.	
}
\end{vd}

\begin{bt}%[Võ Thị Quỳnh Trang- EX-C2-6-2018]%[6D3B4]

\loigiai{
	Rút gọn phân số $ \dfrac{39}{52}=\dfrac{39:13}{52:13}=\dfrac{3}{4} $.
	\\Do $ 4\cdot 5=20 > 18 $ nên ta nhân cả tử và mẫu của phân số $ \dfrac{3}{4} $ lần lượt với $ 1 $; $ 2 $; $ 3 $; $ 4 $ để được các phân số cần tìm. Đó là các phân số:
	$$\dfrac{3}{4}; \dfrac{6}{8}; \dfrac{9}{12};\dfrac{12}{16}.$$
}
\end{bt}

\begin{bt}%[Võ Thị Quỳnh Trang- EX-C2-6-2018]%[6D3B4]

			 $ \dfrac{230}{552}= \dfrac{230:46}{552:46}=\dfrac{5}{12} $.
			 	$ \dfrac{374}{612}= \dfrac{374:34}{612:34}=\dfrac{11}{18} $.

	\end{enumerate}
}
\end{bt}

\begin{bt}%[Võ Thị Quỳnh Trang- EX-C2-6-2018]%[6D3B4]
Rút gọn phân số mà:
\begin{enumEX}{1}
	\item Tử là số nhỏ nhất có ba chữ số giống nhau, còn mẫu là số lớn nhất có ba chữ số.
	\item Mẫu là số nhỏ nhất có bốn chữ số khác nhau, còn tử là số lớn nhất có ba chữ số khác nhau.
\end{enumEX}

\loigiai{
	\begin{enumEX}{1}
		\item Số nhỏ nhất có ba chữ số giống nhau là $ 111 $.
		\\ Số lớn nhất có ba chữ số là $ 999 $.
		\\ Vậy phân số cần rút gọn là $ \dfrac{111}{999} $.
		\\ $ \dfrac{111}{999} = \dfrac{111:111}{999:111}= \dfrac{1}{9} $.
		\item Số nhỏ nhất có bốn chữ số khác nhau là $ 1023 $.
		\\ Số lớn nhất có ba chữ số khác nhau là $ 987 $.
		\\ Vậy phân số cần rút gọn là $ \dfrac{987}{1023} $.
		\\$ \dfrac{987}{1023}=\dfrac{987:3}{1023:3}=\dfrac{329}{341} $.
\end{enumEX}	
}
\end{bt}

\begin{bt}%[Võ Thị Quỳnh Trang- EX-C2-6-2018]%[6D3B4]
Có bao nhiêu phân số tối giản trong các phân số sau?
$$\dfrac{-1}{30}; \dfrac{51}{100}; \dfrac{-27}{42}; \dfrac{35}{91}; \dfrac{18}{25}.$$
\loigiai{
		ƯCLN$(-1,30)=1$;\\
		ƯCLN$(51,100)=1$;\\
		ƯCLN$(-27,42)=3$;\\
		ƯCLN$(35,91)=7$.\\
		ƯCLN$(18,25)=1$.\\
		Vậy các phân số tối giản là $\dfrac{-1}{30};~\dfrac{51}{100}$ và $\dfrac{18}{25}.$
	}
\end{bt}

\begin{bt}%[Võ Thị Quỳnh Trang- EX-C2-6-2018]%[6D3K4]
Tìm các số nguyên $ b $ ($ 21\le b\le 31 $) sao cho các phân số $ \dfrac{7}{b} $; $ \dfrac{10}{b}$; $ \dfrac{11}{b}$ đều là phân số tối giản.
\loigiai{
	Xét phân số $\dfrac{7}{b}$, có $7=1 \cdot 7$ nên phân số $\dfrac{7}{b}$ tối giản khi $b \neq 7k$ ($k$ nguyên dương). \quad (1)\\
	Xét phân số $\dfrac{10}{b}$, có $10=2 \cdot 5$ nên phân số $\dfrac{10}{b}$ tối giản khi $b\neq 2k;~b \neq 5k$ ($k$ nguyên dương). \quad (2)\\
	Xét phân số $\dfrac{11}{b}$, có $11=1 \cdot 11$ nên phân số $\dfrac{11}{b}$ tối giản khi $b \neq 11k$ ($k$ nguyên dương). \quad (3)\\
	Từ (1), (2), (3) và kết hợp với điều kiện $ 21\le b\le 31 $ ta có $ b\in \left\{23; 27; 29; 31\right\} $	để các phân số $ \dfrac{7}{b} $; $ \dfrac{10}{b}$; $ \dfrac{11}{b}$ đều là phân số tối giản.
}
\end{bt}
