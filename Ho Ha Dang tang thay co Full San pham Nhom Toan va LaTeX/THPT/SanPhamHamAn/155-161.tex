\begin{ex}%[WTT-218cau-ham-an-2D1, Phạm Doãn Lê Bình]%[2D1K5-3]
\immini{	
	Cho hàm số $y=f(x)=ax^3+bx^2+cx+d$ $(a\neq 0)$ có đồ thị như hình vẽ. Phương trình $f(f(x))=0$ có bao nhiêu nghiệm thực?
	\choice
	{$3$}
	{$7$}
	{\True $9$}
	{$5$}
}{
\begin{tikzpicture}[scale=0.7,line join=round, line cap=round,>=stealth]
\tikzset{label style/.style={font=\footnotesize}}
\draw[->] (-2.3,0)--(3.1,0) node[below left] {\tiny $x$};
\draw[->] (0,-2.3)--(0,3.1) node[below left] {\tiny $y$};
\draw[dashed] (0,2)--(-1,2);
\draw[dashed] (0,-2)--(1,-2);
\draw (0,0) node [below left] {\tiny $O$};
\foreach \x in {-2,-1,1,2}
	\draw[thin] (\x,1pt)--(\x,-1pt) node [below] {\tiny $\x$};
\foreach \y in {-2,2}
	\draw[thin] (1pt,\y)--(-1pt,\y) node [left] {\tiny $\y$};
\begin{scope}
\clip (-2.2,-2.2) rectangle (3,3);
\draw[samples=200,domain=-2:2,smooth,variable=\x] plot (\x,{1*((\x)^3)+0*((\x)^2)+-3*(\x)+0});
\end{scope}
\end{tikzpicture}
}
	\loigiai{
		Đặt $t=f(x)$, phương trình $f(f(x))=0$ trở thành $f(t)=0(*)$ (số nghiệm phương trình $(*)$ là số giao điểm của đồ thị hàm số $y=f(x)$ với trục $Ox$). Nhìn vào đồ thị ta thấy phương trình $(*)$ có 3 nghiệm $t$ thuộc khoảng $(-2;2)$, với mỗi giá trị $t$ như vậy phương trình $f(x)=t$ có 3 nghiệm phân biệt. Vậy phương trình $f(f(x))=0$ có 9 nghiệm.\\
		Lưu ý: khi $t$ có 3 giá trị thuộc $(-2;2)$ thì nghiệm phương trình $f(x)=t$ là giao điểm của đồ thị hàm số $y=f(x)$ và đường thẳng $y=t$, $t\in (-2;2)$ (là hàm hằng song song trục $Ox$).}
\end{ex}

\begin{ex}%[WTT-218cau-ham-an-2D1, Phạm Doãn Lê Bình]%[2D1K5-3]
\immini{	
	Cho hàm số $y=f(x)$ có đồ thị như hình vẽ. Hỏi có bao nhiêu điểm trên đường tròn lượng giác biểu diễn nghiệm của phương trình $f\left(f\left(\cos 2x\right)\right)=0$ ?
	\choice
	{$3$}
	{\True $4$}
	{$2$}
	{$1$}
}{
	\begin{tikzpicture}[line join=round, line cap=round,>=stealth]
\tikzset{label style/.style={font=\footnotesize}}
\draw[->] (-2.5,0)--(2.5,0) node[below left] {\tiny $x$};
\draw[->] (0,-1.1)--(0,1.6) node[below left] {\tiny $y$};
\draw (0,0) node [below left] {\tiny $O$};
\draw[dashed] (-1,0)--(-1,1)--(1,1)--(1,0);
\foreach \x in {-2,-1,1,2}
	\draw[thin] (\x,1pt)--(\x,-1pt) node [below] {\tiny $\x$};
\foreach \y in {1}
	\draw[thin] (1pt,\y)--(-1pt,\y) node [left] {\tiny $\y$};
\begin{scope}
\clip (-3,-1) rectangle (3,2);
\draw[samples=200,domain=-2:2,smooth,variable=\x] plot (\x,{-1*((\x)^4)+0*((\x)^3)+2*((\x)^2)+0*(\x)});
\end{scope}
\end{tikzpicture}
}
	\loigiai{
		Từ đồ thị ta có $f(x)\leqslant 1,\forall x\in \mathbb{R}$ và suy ra được $f\left(\cos 2x\right)=\pm a\left(a>1\right)$ hoặc $f\left(\cos 2x\right)=0$ \\
		TH1: Nếu $f\left(\cos 2x\right)=a>1$ thì phương trình này vô nghiệm.\\
		TH2: Nếu $f\left(\cos 2x\right)=-a<-1$ thì $\left|\cos 2x\right|>1$, phương trình này vô nghiệm.\\
		TH3: Nếu $f\left(\cos 2x\right)=0 \Leftrightarrow \cos 2x=\pm a$ (vô nghiệm) và $\cos 2x=0$ có 4 điểm trên vòng tròn lượng giác.\\
		Vậy có $4$ điểm.}
\end{ex}
