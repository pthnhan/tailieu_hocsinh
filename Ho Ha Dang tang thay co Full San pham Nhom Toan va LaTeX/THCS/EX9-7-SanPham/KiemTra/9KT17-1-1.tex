\begin{ex}%[9D1B2]
Cho $a,b\in\mathbb{R}$, $a\neq b$. Các căn bậc hai của $(a-b)^2$ là
\choice{$|a-b|$}
{$a-b$}
{$b-a$}
{\True $a-b$ và $b-a$}
\loigiai{
Ta có $(a-b)^2>0$ có hai căn bậc hai là $\sqrt{(a-b)^2}=|a-b|$ và $-\sqrt{(a-b)^2}=-|a-b|$ đó chính là $a-b$ và $b-a$.
}
\end{ex}

\begin{ex}%[9D1B7]
Cho $\dfrac{2}{\sqrt{3}+\sqrt{5}}=\sqrt{a}-\sqrt{b}$ với $a,b$ là các số nguyên dương. Khi đó giá trị $a-b$ bằng?

\begin{ex}%[9H1K2]
Cho góc nhọn $\alpha$ có $\cot\alpha=2$, $P=\dfrac{\sin\alpha-\cos\alpha}{\sin\alpha+\cos\alpha}$ bằng?

{$\dfrac{1}{3}$}
{\True $-\dfrac{1}{3}$}
{$-\dfrac{1}{2}$}
\loigiai{

\begin{ex}%[9D1B8]
Cho biểu thức $A=\dfrac{3(x+2\sqrt{x})}{x+\sqrt{x}-2}-\dfrac{\sqrt{x}+2}{\sqrt{x}-1}-\dfrac{\sqrt{x}+1}{\sqrt{x}+2}$\quad với $x\ge 0$, $x\neq 1$. 
\begin{enumerate}
\item Chứng minh rằng $P=\dfrac{\sqrt{x}+3}{\sqrt{x}+2}$.
\item Chứng minh rằng nếu $x\geq 0$; $x\neq 1$ thì $P\leq \dfrac{3}{2}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Điều kiện $x\ge 0$, $x\neq 1$.
\begin{eqnarray*}
A & = & \dfrac{3x+6\sqrt{x}-(\sqrt{x}+2)^2-(\sqrt{x}+1)(\sqrt{x}-1)}{(\sqrt{x}-1)(\sqrt{x}+2)}\\
& = & \dfrac{x+2\sqrt{x}-3}{(\sqrt{x}-1)(\sqrt{x}+2)}=\dfrac{(\sqrt{x}-1)(\sqrt{x}+3)}{(\sqrt{x}-1)(\sqrt{x}+2)}=\dfrac{\sqrt{x}+3}{\sqrt{x}+2}.
\end{eqnarray*}
%Vậy $P=\dfrac{\sqrt{x}+3}{\sqrt{x}+2}$.
\item Điều kiện $x\ge 0$, $x\neq 1$.
Ta có $$P\le\dfrac{3}{2}\Leftrightarrow \dfrac{\sqrt{x}+3}{\sqrt{x}+2}\le \dfrac{3}{2}\Leftrightarrow 2(\sqrt{x}+3)\le 3(\sqrt{x}+2)\Leftrightarrow -\sqrt{x}\le 0\,\,\text{(luôn đúng)}.$$
Vậy $P\le \dfrac{3}{2}$ với $x\ge 0$, $x\neq 1$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[9D1B4]%[9D1K2]
Giải các phương trình sau
\begin{enumerate}
\item $\sqrt{2x-1}+\sqrt{\dfrac{8x-4}{9}}=\dfrac{5x}{3}$.
\item $\sqrt{x^2+2x+1}+\sqrt{x^2+4x+4}=3x$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Điều kiện: $x\ge \dfrac{1}{2}$. Ta có phương trình tương đương
\begin{eqnarray*}
&& \sqrt{2x-1}+\dfrac{2}{3}\sqrt{2x-1}=\dfrac{5}{3}x\\
& \Leftrightarrow & \dfrac{5}{3}\sqrt{2x-1}=\dfrac{5}{3}x
\Leftrightarrow \sqrt{2x-1}=x\\
& \Leftrightarrow & 2x-1=x^2\Leftrightarrow (x-1)^2=0\Leftrightarrow x=1\,\, \text{(thỏa mãn)}.
\end{eqnarray*}
Vậy nghiệm của phương trình là $x=1$.
\item Ta có phương trình tương đương
$$\sqrt{(x+1)^2}+\sqrt{x+2)^2}=3x
\Leftrightarrow |x+1|+|x+2|=3x.\qquad (1)$$
Vì $|x+1|+|x+2|\ge 0$ nên $3x\ge 0\Leftrightarrow x\ge 0$
$\Rightarrow x+1>0$ và $x+2>0$. Do đó
$$(1)\Leftrightarrow x+1+x+2=3x\Leftrightarrow x=3\,\,\text{(thỏa mãn)}.$$
Vậy nghiệm của phương trình là $x=3$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[8H3B8]%[9H1B2]%[9H1G1]
Cho $\triangle ABC$, đường cao $AH$. Biết $AH^2=HB.HC$
\begin{enumerate}
\item Chứng minh rằng $\triangle ABC$ là tam giác vuông.
\item Nếu $\widehat{HAB}=30^{\circ}$ và $HC=2$ cm, tính $S_{\triangle ABC}$.
\item Kẻ $HD\perp AB$ và $HE\perp AC$. Chứng minh rằng $AH^3=BC.HD.HE$.
\end{enumerate}
\loigiai{

\begin{enumerate}
\item 
\immini{
Ta có $AH^2=HB.HC\Leftrightarrow \dfrac{AH}{HB}=\dfrac{HC}{AH}$.\\
Xét $\triangle AHC$ và $\triangle BHA$ có
\begin{itemize}
\item $\widehat{CHA}=\widehat{BHA}=90^{\circ}$;
\item $\dfrac{AH}{HB}=\dfrac{HC}{AH}$.
\end{itemize}
$\Rightarrow \triangle AHC\backsim \triangle BHA\Rightarrow \widehat{HAC}=\widehat{HBA}$.\\
Mà $\widehat{HAB}+\widehat{HBA}=90^{\circ}\Rightarrow \widehat{HAB}+\widehat{HAC}=90^{\circ}$.\\
$\Rightarrow \widehat{BAC}=90^{\circ}$ hay $\triangle ABC$ vuông tại $A$.
}
{
\begin{tikzpicture}[scale=1.6]
\tkzInit[xmin=-0.5,xmax=2.5,ymin=-0.5, ymax=3.5] \tkzClip[space=.1]
\tkzDefPoints{0/0/A, 2/0/B, 0/3/C}

\tkzDefMidPoint(A,B) \tkzGetPoint{x}
\tkzInterLC(B,C)(x,A) \tkzGetPoints{H}{B}

\tkzDefMidPoint(H,B) \tkzGetPoint{y}
\tkzInterLC(B,A)(y,B) \tkzGetPoints{B}{D}

\tkzDefMidPoint(H,C) \tkzGetPoint{z}
\tkzInterLC(C,A)(z,C) \tkzGetPoints{E}{C}

\tkzMarkRightAngles(C,A,B H,D,B H,E,C A,H,B)

\tkzDrawSegments(A,B B,C C,A A,H H,D H,E)

\tkzDrawPoints[fill=white,size=8](A,B,C,D,H,E)
\tkzLabelPoints[left](C,A, E)
\tkzLabelPoints[right](B,H)
\tkzLabelPoints[below](D)
\end{tikzpicture}
}
\item Ta có $\widehat{ACB}=\widehat{HAB}$ (Cùng phụ với $\widehat{ABC}$).\\
$\Rightarrow \widehat{ACB}=\widehat{ACH}=30^{\circ}$. Mà $\triangle AHC$ vuông tại $H$ nên
$$\cos\widehat{ACH}=\dfrac{HC}{AC}\Leftrightarrow \cos 30^{\circ}=\dfrac{2}{AC}\Rightarrow AC=\dfrac{2}{\cos 30^{\circ}}=\dfrac{4}{\sqrt{3}}\,\,\mathrm{(cm)}.$$
$\triangle ABC$ vuông tại $A$ nên
$$\tan \widehat{ACB}=\dfrac{AB}{AC}\Leftrightarrow \tan 30^{\circ}=\dfrac{AB}{AC}\Rightarrow AB=AC.\tan 30^{\circ}=\dfrac{4}{\sqrt{3}}\cdot \dfrac{1}{\sqrt{3}}=\dfrac{4}{3}\,\,\mathrm{(cm)}.$$
Diện tích $\triangle ABC$ là $S=\dfrac{1}{2}\cdot AB\cdot AC=\dfrac{8}{3\sqrt{3}}=\dfrac{8\sqrt{3}}{9}\,\, (\mathrm{cm}^2)$.
\item Áp dụng các hệ thức lượng trong tam giác vuông ta được
\begin{eqnarray*}
AH.BC.HD.HE & = & (AH.BC).HD.HE\\
& = & (AB.AC).HD.HE = (AB.HD)(AC.HE)\\
& = & HB.AH.HC.AH=(HB.HC).AH^2=AH^4.
\end{eqnarray*}
$\Rightarrow BC.HD.HE=AH^3$.
\end{enumerate}
}

\end{ex}

\begin{ex}%[9D1G2]
Tìm tất cả các giá trị $x,y,z$ sao cho $\sqrt{x}+\sqrt{y-z}+\sqrt{z-x}=\dfrac{1}{2}(y+3)$.
\loigiai{
Điều kiện $y\ge z\ge x\ge 0$. Ta có các biến đổi tương đương
\begin{eqnarray*}
&& \sqrt{x}+\sqrt{y-z}+\sqrt{z-x}=\dfrac{1}{2}(y+3)\\
& \Leftrightarrow & 2\sqrt{x}+2\sqrt{y-z}+2\sqrt{z-x}=y+3\\
& \Leftrightarrow & (\sqrt{x}-1)^2+(\sqrt{y-z}-1)^2+(\sqrt{z-x}-1)^2=0\\
& \Leftrightarrow & \sqrt{x}-1=\sqrt{y-z}-1=\sqrt{z-x}-1=0.
\end{eqnarray*}
Vậy $(x;y;z)=(1;3;2)$.
}
\end{ex}
