\section{Góc ở tâm. Số đo cung}
\subsection{Tóm tắt lí thuyết}
\begin{dn}
	Góc có đỉnh trùng với tâm đường tròn được gọi là góc ở tâm.
\end{dn}

Trong hình vẽ trên $\widehat{AOB}$ là một góc ở tâm, $\wideparen{AmB}$ là cung nhỏ, $\wideparen{AnB}$ là cung lớn.
\begin{dn}
\begin{itemize}
	\item 	Số đo của cung nhỏ bằng số đo của góc ở tâm chắn cung đó.
	\item Số đo của cung lớn bằng hiệu giữa $360^{\circ}$ và số đo của cung nhỏ (có chung hai mút với cung lớn).
	\item Số đo của nửa đường tròn bằng $180^{\circ}$.
\end{itemize}
\end{dn}
\begin{note} Chú ý
	\begin{itemize}
		\item Cung nhỏ có số đo nhỏ hơn $180^{\circ}$.
		\item Cung lớn có số đo lớn hơn $180^{\circ}$.
		\item Khi hai mút của cung trùng nhau, ta có \lq\lq cung không \rq\rq      với số đo $0^{\circ}$ và cung cả đường tròn có số đo $360^{\circ}$.
	\end{itemize}
\end{note}
\begin{dn}
	Trong một đường tròn hay trong hai đường tròn bằng nhau: 
	\begin{itemize}
		\item Hai cung được gọi là bằng nhau nếu chúng có số đo bằng nhau.
		\item Trong hai cung, cung nào có số đo lớn hơn được gọi là cung lớn hơn.
	\end{itemize}

	\loigiai{

	Theo bài toán trên, vì $DC\parallel AE \Rightarrow \ \wideparen{AD}=\wideparen{CE}\Rightarrow \widehat{{{O}_{1}}}=\widehat{{{O}_{2}}}$.\\

$\Rightarrow \widehat{{{O}_{1}}}=\widehat{OEK}$.\\
