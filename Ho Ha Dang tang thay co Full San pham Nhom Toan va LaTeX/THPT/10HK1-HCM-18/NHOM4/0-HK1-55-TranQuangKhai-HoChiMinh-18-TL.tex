	\begin{bt}%[Đề HK1, Trần Quang Khải, Hồ Chí Minh 17-18]%[Nguyễn Thị Hồng Nhung]%[0D3B2-4]%[0D3B2-2]
		Giải các phương trình sau:
		\begin{enumerate}
			\item $|x^2+5x-9|=2x+1$.
			\item $\sqrt{x^2-3x+3}=3x-2$.
			\item $4x^2+x+4\sqrt{4x^2+x-4}-9=0$.
		\end{enumerate}
		\loigiai{
			\begin{enumerate}
				\item Ta có
				\begin{eqnarray*}|x^2+5x-9|=2x+1 &\Leftrightarrow &\heva {&2x+1\geq 0 \\ & \hoac{&x^2+5x-9=2x+1\\&x^2+5x-9=-2x-1}}\\
					&\Leftrightarrow &\heva {&x\geq -\dfrac{1}{2}\\& \hoac{&x^2+3x-10=0\\&x^2+7x-8=0}}\\
					&\Leftrightarrow &\heva {&x\geq -\dfrac{1}{2}\\&\hoac{&x=2\\&x=-5\\&x=1\\&x=-8}}\\
					&\Leftrightarrow &\hoac{&x=1\\&x=2.}
				\end{eqnarray*}
				Vậy phương trình có hai nghiệm phân biệt là $x=1$ và $x=2$.
				\item Ta có
				\begin{eqnarray*}
					\sqrt{x^2-3x+3}=3x-2 &\Leftrightarrow &\heva {&3x-2\geq 0\\&x^2-3x+3=(3x-2)^2}\\
					&\Leftrightarrow & \heva {&x\geq \dfrac{2}{3}\\& x^2-3x+3=9x^2-12x+4}\\
					&\Leftrightarrow & \heva {&x\geq \dfrac{2}{3} \\&8x^2-9x+1=0}\\
					&\Leftrightarrow &\heva{&x\geq \dfrac{2}{3} \\&\hoac{&x=1\\&x=\dfrac{1}{8}}}\Leftrightarrow x=1.
				\end{eqnarray*}
				Vậy phương trình có nghiệm nghiệm duy nhất là $x=1$.
				\item $4x^2+x+4\sqrt{4x^2+x-4}-9=0$.\\
				Điều kiện: $x\geq \dfrac{-1+\sqrt{65}}{8}$, hoặc $x\le \dfrac{-1-\sqrt{65}}{8}$\\
				Đặt $t=\sqrt{4x^2+x-4}$, $t\geq 0$, phương trình đã cho trở thành
				\begin{eqnarray*}
					& &  t^2+4t-5=0\\
					&\Leftrightarrow & \hoac{&t=1\text{ (thỏa mãn)}\\&t=-5\text{ (loại)}.}
				\end{eqnarray*}
				Từ đó, ta có
				\begin{eqnarray*}	
					\sqrt{4x^2+x-4}=1 &\Leftrightarrow &4x^2+x-4=1\\
					&\Leftrightarrow &4x^2+x-5=0\\
					&\Leftrightarrow &\hoac{&x=1\\&x=-\dfrac{5}{4}.}
				\end{eqnarray*}
				Vậy phương trình có hai nghiệm phân biệt là $x=1$ và $x=-\dfrac{5}{4}$.	 
			\end{enumerate}
		}
	\end{bt}% giữa 2 câu phân biệt phải cách ra như mẫu này
	
	\begin{bt}%[Đề HK1, Trần Quang Khải, Hồ Chí Minh 17-18]%[Nguyễn Thị Hồng Nhung]%[0D3G2-5]
		Cho phương trình $x^2-2mx+3m-2=0 \quad (1)$, ($m$ là tham số).
		\begin{enumerate}
			\item Tìm $m$ để phương trình $(1)$ có hai nghiệm trái dấu.
			\item  Tìm $m$ để phương trình $(1)$ có hai nghiệm phân biệt $x_1, x_2$ thỏa mãn%[0D4G5-2] $$x_{1}^2+x_{2}^2=4+x_1+x_2.$$
		\end{enumerate}
		\loigiai {
			\begin{enumerate}
				\item 
				Phương trình đã cho có hai nghiệm trái dấu  $\Leftrightarrow ac<0 \Leftrightarrow 3m-2<0 \Leftrightarrow m<\dfrac{2}{3}$.
				\item Xét phương trình bậc hai $x^2-2mx+3m-2=0 \quad (1)$, ta có $\Delta' = m^2-3m+2$.\\
				Phương trình đã cho có hai nghiệm phân biệt $x_1, x_2$			 
				\begin{eqnarray*}
					\Leftrightarrow \Delta' >0 &\Leftrightarrow &m^2-3m+2> 0 \\
					&\Leftrightarrow & \hoac {&m<1 \\&m>2.}
				\end{eqnarray*}
				Với $\hoac {&m<1 \\&m>2}$, áp dụng định lý Vi-et, ta có $ \heva{& x_1+x_2=2m\\&x_1x_2=3m-2.}$ \\
				Ta có 
				\begin{eqnarray*}
					x_{1}^2+x_{2}^2=4+x_1+x_2&\Leftrightarrow &\left(x_1+x_2\right)^2=4+x_1+x_2+2x_1x_2\\
					&\Leftrightarrow &4m^2=4+2m+6m-4 \\
					&\Leftrightarrow &4m^2-8m=0 \\
					&\Leftrightarrow & \hoac {&m=2 \,\,\text{(loại)}\\&m=0\,\,\text{(thỏa mãn).}}
				\end{eqnarray*}
				Vậy $m=0$ là giá trị cần tìm.
			\end{enumerate}
		}
	\end{bt}
	
	\begin{bt}%[Đề HK1, Trần Quang Khải, Hồ Chí Minh 17-18]%[Nguyễn Thị Hồng Nhung]%[0D4K1-1]
		Với $a>0$, $b>0$, chứng minh bất đẳng thức $\dfrac{1}{a}+\dfrac{1}{b}\geq\dfrac{4}{a+b}$.
		\loigiai {
			Vì $a>0$, $b>0$ nên
			\begin{eqnarray*}
				\dfrac{1}{a}+\dfrac{1}{b}\geq\dfrac{4}{a+b}&\Leftrightarrow &\dfrac{a+b}{ab}\geq\dfrac{4}{a+b}\\
				&\Leftrightarrow &\left(a+b\right)^2\geq 4ab \\
				&\Leftrightarrow &\left(a-b\right)^2 \geq 0 \,\, \text{(luôn đúng).}
			\end{eqnarray*}
		}
	\end{bt}
	
\begin{bt}%[Đề HK1, Trần Quang Khải, Hồ Chí Minh 17-18]%[Nguyễn Thị Hồng Nhung]%[0H2B2-1]
	Trong mặt phẳng tọa độ $Oxy$, cho tam giác $ABC$, với $A(1;2)$, $B(-2;6)$, $C(9;8)$.
	\begin{enumerate}
		\item Chứng minh tam giác $ABC$ vuông tại A.
		\item Tìm tọa độ trung điểm $M$ của cạnh $AC$ và tính độ dài trung tuyến $BM$ của tam giác $ABC$.
		\item $N$ là điểm trên cạnh $BC$ sao cho $\vec{BN}=3\vec{NC}$. Tính diện tích tam giác $ABN$.
	\end{enumerate}
	
	\loigiai{\immini{
			\begin{enumerate}
				\item Ta có $\heva {& \vec{AB}=(-3;4)\\&\vec{AC}=(8;6)} \Rightarrow \vec{AB}\cdot \vec{AC}= -3\cdot 8+4\cdot 6=0$.\\
				Do đó $\vec{AB}\perp\vec{AC}$, suy ra tam giác $ABC$ vuông tại A.
				\item Vì $M$ là trung điểm của $AC$, ta có $M(5;5)$.\\Ta có 
				$\vec{BM}=(7;-1)\Rightarrow  BM=\sqrt{7^2+\left(-1\right)^2}=5\sqrt{2}$.\\
				\item Vì $\vec{BN}=3\vec{NC} \Leftrightarrow \vec{BN}=\dfrac{3}{4}\vec{BC}\Rightarrow BN=\dfrac{3}{4}BC\Rightarrow \mathrm{S}_{\triangle ABN}=\dfrac{3}{4}\mathrm{S}_{\triangle ABC}$.\\
				Ta có $\heva {&AB=5\\&AC=10} \Rightarrow \mathrm{S}_{\triangle ABC}=\dfrac{1}{2}AC\cdot AB= \dfrac{1}{2}\cdot 5 \cdot 10=25$.\\
				$\Rightarrow \mathrm{S}_{\triangle ABN}=\dfrac{3}{4}\mathrm{S}_{\triangle ABC}=\dfrac{3}{4}\cdot 25=\dfrac{75}{4}$.\\
				Do đó, diện tích $\triangle ABN $ bằng $\dfrac{75}{4}$.
			\end{enumerate}
			
		}{
			\begin{tikzpicture}[>=stealth,scale=0.7]
			\tkzDefPoints{0/0/A, 0/10/C, 5/0/B, 0/5/M}
			\tkzDrawPoints(A,B,C,M)
			\tkzLabelPoints[below](A)
			\tkzLabelPoints[right](B)
			\tkzLabelPoints[right](C)
			\tkzLabelPoints[left](M)
			\tkzDrawSegments(A,B B,C C,A M,B)
			\tkzMarkSegments[mark=x](M,C M,A)
			\tkzMarkRightAngle(B,A,C)
			\end{tikzpicture}
		}
		
	}
\end{bt}
