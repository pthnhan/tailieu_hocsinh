 Rút gọn biểu thức $P=\dfrac{\sqrt{x+2\sqrt{x-1}}+\sqrt{x-2\sqrt{x-1}}}{\sqrt{x+\sqrt{2x-1}}-\sqrt{x-\sqrt{2x-1}}},x\geq 2$.
\loigiai{
Ta có $\sqrt{x+2\sqrt{x-1}}=\sqrt{(x-1)+2\sqrt{x-1}+1}=\sqrt{\left(\sqrt{x-1}+1\right)^2}=\sqrt{x-1}+1$.\\
Tương tự ta cũng có
{\allowdisplaybreaks $$\begin{aligned}
&\sqrt{x-2\sqrt{x-1}}=\sqrt{x-1}-1,\\
&\sqrt{x+\sqrt{2x-1}}=\dfrac{1}{\sqrt{2}}\left(\sqrt{2x-1}+1\right),\\ 
&\sqrt{x-\sqrt{2x-1}}=\dfrac{1}{\sqrt{2}}\left(\sqrt{2x-1}-1\right)
\end{aligned}$$}
Suy ra $P=\dfrac{\left(\sqrt{x-1}+1+\sqrt{x-1}-1\right)\sqrt{2}}{\sqrt{2x-1}+1-\sqrt{2x-1}+1}=\sqrt{2}\sqrt{x-1}$.
}
\end{ex}

Cho $x$ là số thực dương thỏa mãn điều kiện $x^2+\dfrac{1}{x^2}=7$. Tính giá trị của biểu thức $A=x^5+\dfrac{1}{x^5};B=x^7+\dfrac{1}{x^7}$.
\loigiai{
 Từ $x^2+\dfrac{1}{x^2}=7\Leftrightarrow \left(x+\dfrac{1}{x}\right)^2-2=7\Leftrightarrow x+\dfrac{1}{x}=3$, (do $x>0$)\\

&\left(x^2+\dfrac{1}{x^2}\right)\left(x^5+\dfrac{1}{x^5}\right)=x^7+\dfrac{1}{x^7}+x^3+\dfrac{1}{x^3}\Rightarrow B=x^7+\dfrac{1}{x^7}=7\cdot 123-18=843.

\begin{ex}%[Đề thi HSG 9, tỉnh Bắc Ninh, năm 2018]%[Nguyễn Chính Tâm, Dự án EX-9G-18]%[9D4K6]
 Cho phương trình $x^2+(m^2+1)x+m-2=0\;\;(1)$, $m$ là tham số. Tìm $m$ để phương trình $(1)$ có $2$ nghiệm phân biệt $x_1,x_2$ thỏa mãn $$\dfrac{2x_1-1}{x_2}+\dfrac{2x_2-1}{x_1}=x_1x_2+\dfrac{55}{x_1x_2}.$$

\loigiai{
 Phương trình $(1)$ có hai nghiệm phân biệt khi và chỉ khi\\
$(m^2+1)^2-4(m-2)>0\Leftrightarrow m^4+2m^2-4m+9>0\Leftrightarrow m^4+2(m-1)^2+7>0,\forall m\in\mathbb{R}$.\\
Do đó phương trình có hai nghiệm phân biệt với mọi $m$.\\
Theo định lý Vi-ét ta có $x_1+x_2=-(m^2+1)$ và $x_1x_2=m-2$.\\
Từ điều kiện, suy ra $m\neq 2$. Theo bài ra ta có
{\allowdisplaybreaks $$\begin{aligned}
&\dfrac{2x_1-1}{x_2}+\dfrac{2x_2-1}{x_1}=x_1x_2+\dfrac{55}{x_1x_2}\\ 
\Leftrightarrow& 2(x_1^2+x_2^2)-(x_1+x_2)=(x_1x_2)^2+55\\
\Leftrightarrow& 2(x_1+x_2)^2-4x_1x_2-(x_1+x_2)=(x_1x_2)^2+55\\
\Leftrightarrow& 2(x_1+x_2)^2-(x_1+x_2)=(x_1x_2+2)^2+51\\
\Leftrightarrow& 2(m^2+1)^2+(m^2+1)=m^2+51\Leftrightarrow 2m^4+4m^2-48=0\\
\Leftrightarrow&\hoac{&m^2=4\\&m^2=-6}\Leftrightarrow\hoac{&m=2\;\;(\text{loại})\\&m=-2\;\;(\text{thỏa mãn}).}
\end{aligned}$$}
Vậy $m=-2$ là giá trị cần tìm.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Bắc Ninh, năm 2018]%[Nguyễn Chính Tâm, Dự án EX-9G-18]%[9D0G2]
 Giải hệ phương trình $\heva{&(x+1)^2+y=xy+4\\&4x^2-24x+35=5(\sqrt{3y-11}+\sqrt{y})}$.
\loigiai{
 Điều kiện: $y\geq \dfrac{11}{3}$. Từ phương trình thứ nhất ta có
{\allowdisplaybreaks $$\begin{aligned}
(x+1)^2+y=xy+4\Leftrightarrow& (x+1)^2-4-y(x-1)=0\\ 
\Leftrightarrow& (x+3)(x-1)-y(x-1)=0\\
\Leftrightarrow& (x-1)(x-y+3)=0\Leftrightarrow\hoac{&x=1,\\&y=x+3.}
\end{aligned}$$}
\textbf{Trường hợp 1.} Với $x=1$ thay vào phương trình thứ hai của hệ ta được
{\allowdisplaybreaks $$\begin{aligned}
15=5(\sqrt{3y-11}+\sqrt{y})\Leftrightarrow&\sqrt{3y-11}+\sqrt{y}=3\Leftrightarrow 4y-11+2\sqrt{y(3y-11)}=9\\ 
\Leftrightarrow& \sqrt{y(3y-11)}=10-2y
\Leftrightarrow \heva{&10-2y\ge 0\\&3y^2-11y=100-40y+4y^2}\\
\Leftrightarrow& \heva{&y\le 5\\&y^2-29y+100=0}\Leftrightarrow\hoac{&y=25, \;\;\text{(Loại)}\\&y=4.\;\;\text{(Thỏa mãn)}}
\end{aligned}$$}
\textbf{Trường hợp 2.} Với $y=x+3$ thay vào phương trình  thứ hai của hệ ta được
{\allowdisplaybreaks $$\begin{aligned}
&4x^2-24x+35=5(\sqrt{3x-2}+\sqrt{x+3})\\ 
\Leftrightarrow& 4x^2-28x+24=5\sqrt{3x-2}-(3x+2)+5\sqrt{x+3}-(x+9)\\
\Leftrightarrow& 4x^2-28x+24=\dfrac{25(3x-2)-(3x+2)^2}{5\sqrt{3x-2}+3x+2}+\dfrac{25(x+3)-(x+9)^2}{5\sqrt{x+3}+x+9}\\
\Leftrightarrow& 4x^2-28x+24=\dfrac{-9x^2+63x-54}{5\sqrt{3x-2}+3x+2}+\dfrac{-x^2+7x-6}{5\sqrt{x+3}+x+9}\\
\Leftrightarrow& \hoac{&x^2-7x+6=0&\qquad(1)\\&4=\dfrac{-9}{5\sqrt{3x-2}+3x+2}+\dfrac{-1}{5\sqrt{x+3}+x+9}&\qquad(2)}
\end{aligned}$$}
Từ điều kiện $y\geq \dfrac{11}{3}\Rightarrow x\geq \dfrac{2}{3}$.
\begin{itemize}
\item Phương trình $(1)$ có nghiệm $x=1$ hoặc $x=9$.\\
Với $x=1$ suy ra $y=6$, với $x=6$ suy ra $y=6$.
\item Phương trình $(2)$ có $5\sqrt{3x-2}+3x+2>0$ và $5\sqrt{x+3}+x+9>0$, suy ra vế phải âm, vế trái dương. Do đó phương trình (2) vô nghiệm.
\end{itemize}
Vậy hệ đã cho có nghiệm là $(x;y)\in \left\{(1;4),(6;9)\right\}$.
}
\end{ex}


\begin{ex}%[Đề thi HSG 9, tỉnh Bắc Ninh, năm 2018]%[Nguyễn Chính Tâm, Dự án EX-9G-18]%[9D0G4]
 Tìm tất cả các cặp số nguyên dương $(m; n)$ thỏa mãn $$(m+n^2)\,\vdots\, (m^2-n)\;\text{và}\;(n+m^2)\,\vdots\, (n^2-m).$$
\loigiai{
Ta có 
	{\allowdisplaybreaks \[\begin{aligned}
\heva{&(m+n^2) \ \vdots \ (m^2-n)\\&(n+m^2) \ \vdots \ (n^2-m)}\Leftrightarrow&\heva{&m+n^2\geq m^2-n\\&n+m^2\geq n^2-m}\\ 
\Leftrightarrow&\heva{&(n-m+1)(m+n)\geq 0\\&(m-n+1)(m+n)\geq 0}\\
\Leftrightarrow&\heva{&m-n+1\geq 1\\&n-m+1\geq 1}\Leftrightarrow -1\leq m-n\leq 1.\qquad(1)
\end{aligned}\]}
Do $m,n$ là các số nguyên dương nên $(1)\Leftrightarrow\hoac{&m-n=1\\&m-n=0\\&m-n=-1}$.\\
\textbf{Trường hợp 1.} $m=n+1$. Ta có
$$\heva{&(m+n^2) \,\vdots \, (m^2-n)\\&(n+m^2) \, \vdots \, (n^2-m)}\Leftrightarrow\heva{&(n^2+n+1) \, \vdots \, (n^2+n+1)\\&(n^2+3n+1) \, \vdots \, (n^2-n-1)}\Leftrightarrow (n^2+3n+1) \, \vdots \, (n^2-n-1).$$
Xét $(n^2+3n+1) \ \vdots \ (n^2-n-1)$ ta có $\dfrac{n^2+3n+1}{n^2-n-1}=1+\dfrac{4n+2}{n^2-n-1}$.
Do đó $$(n^2+3n+1) \ \vdots \ (n^2-n-1)\Leftrightarrow (4n+2) \ \vdots \ (n^2-n-1).$$
Do $n^2-n-1=n(n-1)-1$, mà $n(n-1)$ chẵn, suy ra $n(n-1)-1$ lẻ, suy ra
$$\begin{aligned}
(2n+1) \ \vdots \ (n^2-n-1)&\Rightarrow 2n+1\geq n^2-n-1\\&\Leftrightarrow n^2-3n-2\leq 0 
\Leftrightarrow \dfrac{3-\sqrt{17}}{2}\leq n\leq \dfrac{3+\sqrt{17}}{2}.
\end{aligned}$$
Do $n\in\mathbb{N}^*$, suy ra $1\leq n\leq \dfrac{3+\sqrt{17}}{2}\Rightarrow n\in\{1;2;3\}$. 
Thử lại ta được $(m,n)\in\left\{(2,1),(3,2)\right\}$.\\
\textbf{Trường hợp 2.} $n=m+1$. Ta có\\
$$\heva{&(m+n^2) \ \vdots \  (m^2-n)\\&(n+m^2) \ \vdots \ (n^2-m)}\Leftrightarrow\heva{&(m^2+3m+1)\ \vdots \ (m^2-m-1)\\&(m^2+m+1) \ \vdots \ (m^2+m+1)}\Leftrightarrow (m^2+3m+1) \ \vdots \ (m^2-m-1).$$
Xét $(m^2+3m+1) \ \vdots \ (m^2-m-1)\Rightarrow ((m^2-m-1)+4m+2) \ \vdots \ (m^2-m-1)\Rightarrow (4m+2) \ \vdots \ (m^2-m-1).$\\
Do $m^2-m-1=m(m-1)-1$, mà $m(m-1)$ chẵn, suy ra $m(m-1)-1$ lẻ, suy ra
$$\begin{aligned}
(mn+1) \ \vdots \ (m^2-m-1)&\Rightarrow 2m+1\geq m^2-m-1\\ &\Leftrightarrow m^2-3m-2\leq 0\Leftrightarrow \dfrac{3-\sqrt{17}}{2}\leq m\leq \dfrac{3+\sqrt{17}}{2}.
\end{aligned}$$
Do $m\in\mathbb{N}^*$, suy ra $1\leq m\leq \dfrac{3+\sqrt{17}}{2}\Rightarrow m\in\{1;2;3\}$. 
Thử lại ta được $(m,n)\in\left\{(1,2),(2,3)\right\}$.\\
\textbf{Trường hợp 3.} $m=n$. Ta có
$$\begin{aligned}
\heva{&(m+n^2) \ \vdots \  (m^2-n)\\&(n+m^2) \ \vdots \  (n^2-m)}\Leftrightarrow& \heva{&(m^2+m) \ \vdots \  (m^2-m)\\&(m^2+m) \ \vdots \ (m^2-m)}\\ 
\Leftrightarrow& (m^2+m) \ \vdots \ (m^2-m)\Rightarrow (m+1) \ \vdots \ (m-1)\\
\Rightarrow& (m-1+2) \ \vdots \ (m-1)\Rightarrow 2 \ \vdots \ (m-1)\Rightarrow \hoac{&m=2,\\&m=3.}
\end{aligned}$$
Thử lại ta được các cặp $(m,n)\in \left\{(2,2),(3,3)\right\}$.\\
Vậy $(m;n)\in \left\{(1;2),(2;1),(2;2),(2;3),(3;2),(3;3)\right\}$.
}
\end{ex}

 Cho tập $A$ gồm $16$ số nguyên dương đầu tiên. Hãy tìm số dương $k$ nhỏ nhất có tính chất: Trong mỗi tập con gồm $k$ phần tử của $A$ đều tồn tại $2$ số phân biệt $a, b$ sao cho $a^2+b^2$ là số nguyên tố.

\begin{ex}%[Đề thi HSG 9, tỉnh Bắc Ninh, năm 2018]%[Nguyễn Chính Tâm, Dự án EX-9G-18]%[9D0G7]
 Cho $x, y, z$ là các số thực không âm thỏa mãn $x+y+z=3$ và $xy+yz+zx \neq 0$. Chứng minh rằng: $\dfrac{x+1}{y+1}+\dfrac{y+1}{z+1}+\dfrac{z+1}{x+1}\leq \dfrac{25}{3\sqrt[3]{4(xy+yz+zx)}}$.

\loigiai{
 Không mất tính tổng quát, giả sử $y$ là số ở giữa hai số $x$ và $z$.\\
Suy ra $x(x-y)(y-z)\geq 0\Leftrightarrow x^2y+xyz\geq xy^2+x^2z$. Do đó\\
$$\begin{aligned}
x^2z+y^2x+z^2y&\leq x^2y+xyz+z^2y=y(x^2+xz+z^2)\\
&\leq y(x+z)^2=\dfrac{1}{2}(2y)(x+z)^2\leq\dfrac{1}{2}\left(\dfrac{2y+x+z+x+z}{3}\right)^3=4.
\end{aligned}$$
Ta có
$$\begin{aligned}
&\dfrac{x+1}{y+1}+\dfrac{y+1}{z+1}+\dfrac{z+1}{x+1}\\
=&\dfrac{(x+1)^2(z+1)+(y+1)^2(x+1)+z+1)^2(y+1)}{(x+1)(y+1)(z+1)}\\
=&\dfrac{(x^2+y^2+z^2+2xy+2yz+2xz)+(x^2z+y^2x+z^2y)+3(x+y+z)+3}{xyz+(x+y+z)+(xy+yz+xz)+1}\\
=&\dfrac{(x+y+z)^2+3(x+y+z)+3+(x^2y+y^2x+z^2y)}{xyz+xy+yz+xz+4}\\
=&\dfrac{21+(x^2z+y^2x+z^2y)}{xyz+xy+yz+xz+4}\leq \dfrac{21+4}{2+2+(xy+yz+xz)}\leq\dfrac{25}{3\sqrt[3]{4(xy+yz+xz)}}.
\end{aligned}$$
Đẳng thức xảy ra khi $(x;y;z)=(0;1;2)$ và các hoán vị.
}
\end{ex}

 Kẻ đường thẳng qua $A$ vuông góc với $BX$ cắt $CD$ tại $S$, suy ra $X$ là trực tâm tam giác $SAB$, suy ra $AX\perp SB$.\\
Gọi $E$ là giao điểm của $BX$ và $SA$, $F$ là giao điểm của $AX$ và $SB$. Ta có

Tương tự $\widehat{BKS}=90^\circ$.\\
Theo hệ thức lượng trong tam giác vuông ta có $SK^2=SF\cdot SB, ST^2=SE\cdot SA$.\\
Mà $SE\cdot SA=SF\cdot SB\Rightarrow ST=SK$.\\
Xét hai tam giác vuông $KSM$ và $TSM$, có
$SK=ST$ và  $SM$ chung, suy ra $\triangle KSM=\triangle TSM$, do đó $MK=MT$.
