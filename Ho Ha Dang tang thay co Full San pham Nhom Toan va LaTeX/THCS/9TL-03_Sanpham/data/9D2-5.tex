	}

\begin{bt}%[Lê Minh An, Dự án 9-TL3]%[9D2B3]

\begin{bt}%[Lê Minh An, Dự án 9-TL3]%[9D2B2]

	{$y=\sqrt{3}x+1-2x$}
	{\True $y=\dfrac{-x+1}{-2}$}
	{$y=\dfrac{x+1}{-2}$}

			\item $y=x+1-2x\Leftrightarrow y=-x+1$ có $a=-1<0$ nên hàm số nghịch biến trên $\mathbb{R}$.
			\item $y=\sqrt{3}x+1-2x\Leftrightarrow y=\left(\sqrt{3}-2\right)x+1$ có $a=\sqrt{3}-2<0$ nên hàm số nghịch biến trên $\mathbb{R}$.
			\item $y=\dfrac{-x+1}{-2}\Leftrightarrow y=\dfrac{1}{2}x-\dfrac{1}{2}$ có $a=\dfrac{1}{2}>0$ nên hàm số đồng biến trên $\mathbb{R}$.
			\item $y=\dfrac{x+1}{-2}\Leftrightarrow y=-\dfrac{1}{2}x-\dfrac{1}{2}$ có $a=-\dfrac{1}{2}<0$ nên hàm số nghịch biến trên $\mathbb{R}$.
		\end{itemize}
	}
\end{bt}

\begin{bt}%[Lê Minh An, Dự án 9-TL3]%[9D2B2]

\begin{bt}%[Dương BùiĐức, Dự án 9-TL3]

\begin{bt}%[Dương BùiĐức, Dự án 9-TL3]

\begin{bt}%[9D2-5-Ôn tập chương 2]%[Mai Sương 9TL-03]
	Cho đường thẳng $d \colon y=mx+2m-1$. Xác định $m$ để khoảng cách từ $O$ đến đường thẳng $d$ là lớn nhất.
	\loigiai{
		Gọi $M(a,b)$ là điểm cố định mà đường thẳng $d$ đi qua.\\ 
		Ta có $b=ma+2m-1$ đúng với mọi $m$\\
		Do đó $(a+2)m-b-1=0$ đúng với mọi $m$\\
		$\Leftrightarrow \heva{&a+2=0\\&-b-1=0} \Leftrightarrow \heva{&a=-2\\&b=-1.}$\\
		Suy ra $M(-2;-1).$\\
		Gọi $OH$ là khoảng cách từ $O$ đến đường thẳng $d$, ta có $OH \le OM.$
		Khoảng cách từ $O$ đến đường thẳng $d$ lớn nhất khi $H \equiv M$ hay $d \perp OM$.\\
		Đường thẳng $OM$ có dạng $y=ax$ đi qua $M(-2;-1)$ nên thay $x=-2;y=-1$, ta có
		$$-1=a \cdot (-2) \Rightarrow a=\dfrac{1}{2}.$$
		Suy ra phương trình đường thẳng $OM$ là $y= \dfrac{1}{2} x.$\\
		Ta có $d \perp OM \Leftrightarrow m \cdot \dfrac{1}{2}=-1 \Leftrightarrow m=-2.$\\
		Vậy $m=-2$.
		
	}
	
\end{bt}
