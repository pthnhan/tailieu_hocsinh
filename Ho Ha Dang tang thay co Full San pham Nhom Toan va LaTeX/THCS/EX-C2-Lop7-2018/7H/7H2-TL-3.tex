

\section{Hai tam giác bằng nhau cạnh - cạnh - cạnh}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Vẽ tam giác biết ba cạnh}
\textbf{\textit{Bài toán:}} Vẽ $\triangle ABC$ biết $AB = c$, $BC = a$, $AC = b$.\\
\textit{Cách vẽ}\\
Ta lần lượt thực hiện:
\immini{
\begin{itemize}
\item Vẽ đoạn thẳng $AB = c$.
\item Trên nửa mặt phẳng bờ $AB$, vẽ cung tròn tâm $A$ bán kính $b$, vẽ cung tròn tâm $B$, bán kính $a$. Hai cung tròn này cắt nhau ở $C$.
\item Nối $AC$, $BC$ ta nhận được $\triangle ABC$ thỏa mãn giả thiết.
\end{itemize}
}{
\begin{tikzpicture}
\tkzDefPoints{0/0/A,4/0/B,1/3/C}
\tkzDrawPolygon(A,B,C)
\tkzCompass[color=red,length=1.5,dashed](A,C)
\tkzCompass[color=black,length=1,dashed](B,C)
%\tkzCompass[color=red,delta=100](A,B)
\tkzDrawPoints(A,B,C)
\tkzLabelPoints[above](C)
\tkzLabelPoints[below](A,B)
\tkzLabelSegment[below](A,B){$c$}
\tkzLabelSegment[left](A,C){$b$}
\tkzLabelSegment[right](C,B){$a$}
\end{tikzpicture}
}
\subsubsection{Trường hợp bằng nhau cạnh – cạnh – cạnh}
\begin{dl}
Nếu ba cạnh của tam giác này bằng ba cạnh của tam giác kia thì hai tam giác đó bằng nhau.
\end{dl}
\immini{
Như vậy, nếu hai tam giác $\triangle ABC$ và $\triangle {A_1}{B_1}{C_1}$ thỏa mãn:\\
$AB = {A_1}{B_1}$; $BC = {B_1}{C_1}$; $CA = {C_1}{A_1}$ thì \break $\triangle ABC = \triangle {A_1}{B_1}{C_1}$\\
và khi đó ta có ngay: $\widehat {A_1} = \widehat A$; $\widehat {B_1} = \widehat B$; $\widehat {C_1} = \widehat C$
}{\begin{tikzpicture}[scale=0.7]
\tkzDefPoints{0/0/B,4/0/C,1/2/A,6/0/B_1,10/0/C_1,7/2/A_1}
\tkzDrawPolygon(A,B,C)
\tkzDrawPolygon(A_1,B_1,C_1)
\tkzDrawPoints(A,B,C,A_1,B_1,C_1)
\tkzLabelPoints[above](A,A_1)
\tkzLabelPoints[below](C,B,B_1,C_1)
\end{tikzpicture}}
\begin{nx}
\hfill
\begin{itemize}
	\item Như vậy, để chứng tỏ hai tam giác bằng nhau chúng ta chỉ cần khẳng định ba cặp cạnh bằng nhau mà không cần phải nêu đủ $6$ yếu tố bằng nhau ($3$ yếu tố cạnh và $3$ yếu tố góc).
	\item Bằng việc khẳng định được sự bằng nhau của hai tam giác chúng ta sẽ bắt đầu làm quen với việc chứng minh các tính chất trong tam giác.
\end{itemize}
\end{nx}


\subsection{Các dạng toán}
\begin{dang}{Chứng minh hai tam giác bằng nhau}
\end{dang}
\begin{vd}%[Tuấn Nguyễn, dự án (EX-C2-L7-2018)]%[7H2Y3]
Cho đoạn thẳng $AB$. Vẽ hai cung tròn tâm $A$, tâm $B$ bán kính $AB$, chúng cắt nhau tại $C$ và $D$. Chứng minh rằng:
\begin{listEX}[2]
	\item $\triangle ABC = \triangle ABD$. 
	\item $\triangle ACD = \triangle BCD$.
\end{listEX}
\loigiai{
Theo cách dựng của giả thiết, ta có: $AB = AC = BC = AD = BD$.
\immini{
\begin{enumerate}
\item  Xét $\triangle ABC$ và $\triangle ABD$ có
\begin{itemize}
	\item $AB$ là cạnh chung;
	\item $AC=AB$ (cmt);
	\item $BC=BD$ (cmt).
\end{itemize}
$ \Rightarrow \triangle ABC = \triangle ABD$ (c–c–c).
\item  Xét $\triangle ACD$ và $\triangle BCD$ có
\begin{itemize}
	\item $CD$ là cạnh chung;
	\item $AC=BC$ (cmt);
	\item $AD=BD$ (cmt).
\end{itemize}
$ \Rightarrow \triangle ACD = \triangle BCD$ (c–c–c).
\end{enumerate}	
}{

\tkzDefPoints{0/0/A,4/0/B}

\tkzInterCC(A,B)(B,A)\tkzGetPoints{C}{D}
\tkzDrawPoints(A,B,C,D)
\tkzLabelPoints[right](B)
\tkzLabelPoints[left](A)
\tkzLabelPoints[above](C)
\tkzLabelPoints[below](D)
\draw (A)--(D)--(B)--(C)--(A) (A)--(B) (C)--(D);
\end{tikzpicture}

\begin{nx}
\hfill
\begin{itemize}
\item Như vậy, để chứng tỏ hai tam giác bằng nhau chúng ta chỉ cần khẳng định ba cặp cạnh bằng nhau mà không cần phải nêu ra đủ $6$ yếu tổ bằng nhau nữa ($3$ yếu tố cạnh và $3$ yếu tố góc).
\item Bằng việc khẳng định được sự bằng nhau của hai tam giác chúng ta sẽ bắt đầu làm quen với việc chứng minh các tính chất trong tam giác, các ví dụ sau sẽ minh họa điều này.
\end{itemize}

% Ví dụ 2
\begin{vd}%[Tuấn Nguyễn, dự án (EX-C2-L7-2018)]%[7H2B3]
Cho $\triangle ABC$ có $AB = AC$. Gọi $M$ là trung điểm của $BC$. Chứng minh rằng $AM$ là đường trung trực của $BC$.
\loigiai{
Để chứng minh $AM$ là đường trung trực của $BC$, ta chỉ cần chứng minh $AH \perp BC$.\\
\immini{
Xét hai tam giác $\triangle ABM$ và $\triangle ACM$, ta có:
\begin{itemize}
\item $AB = AC$ (giả thiết);
\item $BM = CM$ (vì $M$ là trung điểm của $BC$);
\item $AM$ chung.
\end{itemize}
}{
\begin{tikzpicture}
\tkzDefPoints{0/0/B,6/0/C,3/0/M,3/3/A}
\tkzDrawPoints(A,B,C,M)
\tkzLabelPoints[right](C)
\tkzLabelPoints[left](B)
\tkzLabelPoints[above](A)
\tkzLabelPoints[below](M)
\draw (A)--(B)--(C)--(A) (A)--(M);
\tkzLabelAngle[pos=0.6,rotate=0](B,A,M){\footnotesize $1$}
\tkzLabelAngle[pos=0.6,rotate=0](M,A,C){\footnotesize $2$}
\tkzLabelAngle[pos=0.3,rotate=0](A,M,C){\footnotesize $2$}
\tkzLabelAngle[pos=0.3,rotate=0](B,M,A){\footnotesize $1$}
\end{tikzpicture}}
\noindent Suy ra: $\triangle ABM = \triangle ACM$ (c.c.c) $ \Rightarrow \widehat {M_1} = \widehat {M_2}$.\\
Mặt khác, ta lại có:
	$$\widehat {M_1} + \widehat {M_2} = 180^\circ  \Leftrightarrow \widehat {M_1} + \widehat {M_1} = 180^\circ  \Leftrightarrow 2\widehat {M_1} = 180^\circ 
	 \Leftrightarrow \widehat {M_1} = 90^\circ  \Leftrightarrow AH \perp BC.$$
Vậy $AM$ là đường trung trực của $BC$.

\begin{nx}
\hfill
\begin{itemize}
\item Cũng từ kết quả $\triangle ABM = \triangle ACM$, ta suy ra được: 
$\widehat {A_1} = \widehat {A_2} \Rightarrow AM$ là đường phân giác của góc $\widehat A$.
\item Như vậy, trong $\triangle ABC$ có $AB = AC$ ($\triangle ABC$ cân tại $A$) thì $AM$ vừa là trung tuyến, vừa là đường cao, vừa là đường trung trực, vừa là đường phân giác – \textit{Đây chính là tính chất cơ bản của tam giác cân.}
\end{itemize}
\end{nx}
}
\end{vd}
