\begin{ex}%[Đề kiểm tra học kỳ I, Đan Phượng - Hà Nội, 2015 - 2016]%[Tuấn Nguyễn, dự án 7EX-1-2018]%[7D1B3]%[7D1B5]%[7D1B11]
Thực hiện phép tính
\begin{enumerate}
	\item $\dfrac{-23}{7}\cdot \dfrac{3}{10}+\dfrac{13}{7}\cdot \dfrac{3}{10}$.
	\item $(-0,375)\cdot 4\dfrac{1}{3}\cdot (-2)^3$.
	\item $\left|-\dfrac{15}{6} \right| -\left|\dfrac{3}{18} \right|\cdot \sqrt{81}+\sqrt{\dfrac{9}{4}}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $\dfrac{-23}{7}\cdot \dfrac{3}{10}+\dfrac{13}{7}\cdot \dfrac{3}{10}=\dfrac{3}{10}\left(\dfrac{-23}{7}+\dfrac{13}{7}\right)=\dfrac{3}{10}\cdot \dfrac{-10}{7}=-\dfrac{3}{7}$.
	\item $(-0,375)\cdot 4\dfrac{1}{3}\cdot (-2)^3=\left[(-0,375)\cdot (-2)^3\right]\cdot 4\dfrac{1}{3}=3\cdot \dfrac{13}{3}=13$.
	\item $\begin{aligned}[t]
	\left|-\dfrac{15}{6} \right| -\left|\dfrac{3}{18} \right|\cdot \sqrt{81}+\sqrt{\dfrac{9}{4}} &= \dfrac{15}{6}-\dfrac{3}{18}\cdot 9+\dfrac{3}{2} \\
	&= \dfrac{15}{6}-\dfrac{3}{2}+\dfrac{3}{2}\\
	&= \dfrac{15}{6}.
	\end{aligned}$
\end{enumerate}	
}
\end{ex}

\begin{ex}%[Đề kiểm tra học kỳ I, Đan Phượng - Hà Nội, 2015 - 2016]%[Tuấn Nguyễn, dự án 7EX-1-2018]%[7D1B3]
Tìm $x$ biết
 $\dfrac{2}{5}x-\dfrac{1}{3}=\dfrac{2}{3}$.
\loigiai{
$\begin{aligned}[t]
	\dfrac{2}{5}x-\dfrac{1}{3} &= \dfrac{2}{3} \\
	\dfrac{2}{5}x&= \dfrac{2}{3}+\dfrac{1}{3}\\
	\dfrac{2}{5}x&= 1\\
	x&=1:\dfrac{2}{5}\\
	x&=\dfrac{5}{2}.
	\end{aligned}$
}
\end{ex}


\begin{ex}%[Đề kiểm tra học kỳ I, Đan Phượng - Hà Nội, 2015 - 2016]%[Tuấn Nguyễn, dự án 7EX-1-2018]%[7D1K4]
	Tìm $x$ biết
 $\left|x-\dfrac{1}{6}\right|+\dfrac{2}{3}=\dfrac{3}{4} $.
	\loigiai{
		$\begin{aligned}[t]
			\left|x-\dfrac{1}{6}\right|+\dfrac{2}{3} &= \dfrac{3}{4} \\
			\left|x-\dfrac{1}{6}\right|&=  \dfrac{3}{4}-\dfrac{2}{3}\\
			\left|x-\dfrac{1}{6}\right|&= \dfrac{1}{12}\\
			x-\dfrac{1}{6}&=\dfrac{1}{12} \;\text{hoặc}\; x-\dfrac{1}{6}=-\dfrac{1}{12}\\
			x&=\dfrac{1}{12}+\dfrac{1}{6}\;\text{hoặc}\; x=-\dfrac{1}{12}+\dfrac{1}{6}\\
			x&=\dfrac{1}{4}\;\text{hoặc}\; x=\dfrac{1}{12}.
			\end{aligned}$
	}
\end{ex}

\begin{ex}%[Đề kiểm tra học kỳ I, Đan Phượng - Hà Nội, 2015 - 2016]%[Tuấn Nguyễn, dự án 7EX-1-2018]%[7D1B8]
Tìm $x, y$ biết  $x:2=y:(-5)$ và $x-y=-7$.
\loigiai{
$x:2=y:(-5)$ suy ra $\dfrac{x}{2}=\dfrac{y}{-5}$.\\
Áp dụng tính chất dãy tỉ số bằng nhau\\
\[\dfrac{x}{2}=\dfrac{y}{-5}=\dfrac{x-y}{2+5}=\dfrac{-7}{7}=-1.\]
Suy ra $x=-2; y=5.$
}
\end{ex}

\begin{ex}%[Đề kiểm tra học kỳ I, Đan Phượng - Hà Nội, 2015 - 2016]%[Tuấn Nguyễn, dự án 7EX-1-2018]%[7D1K8]
Tính diện tích của một hình chữ nhật biết rằng tỉ số giữa hai cạnh của nó bằng $\dfrac{3}{5}$ và chu vi bằng $64$ cm.
\loigiai{
Nửa chu vi hình chữ nhật là $64:2=32$ cm.\\
Gọi $a, b$ (cm) là độ dài hai cạnh của hình chữ nhật ($a,b>0$).\\
Theo đề ta có $a:b=3:5$ và $a+b=32$.\\
Ta có $a:b=3:5$ suy ra $\dfrac{a}{3}=\dfrac{b}{5}=\dfrac{a+b}{3+5}=\dfrac{32}{8}=4$.\\
Suy ra $\heva{&a=12\\&b=20.}$\\
Vậy diện tích hình chữ nhật là $a\cdot b=12\cdot 20=240$ cm$^2$.
}
\end{ex}

\begin{ex}%[Đề kiểm tra học kỳ I, Đan Phượng - Hà Nội, 2015 - 2016]%[Tuấn Nguyễn, dự án 7EX-1-2018]%[7H2Y1]%[7H2B4]%[7H2B5]%[7H2K5]
Cho tam giác $ABC$ vuông tại $A$. Tia phân giác của góc $B$ cắt cạnh $AC$ tại $D$.
\begin{enumerate}
	\item Cho biết góc $\widehat{ACB}=40^\circ$. Tính số đo góc $ABD$.
	\item Trên cạnh $BC$ lấy điểm $E$ sao cho $BE=BA$. Chứng minh rằng $\triangle BAD=\triangle BED$ và $DE \perp BC$.
	\item Gọi $F$ là giao điểm của $BA$ và $ED$. Chứng minh rằng $\triangle ABC=\triangle EBF$.
	\item Vẽ $CK$ vuông góc với $BD$ tại $K$. Chứng minh rằng ba điểm $K, F, C$ thẳng hàng.
\end{enumerate}
\loigiai{
	\begin{center}
	\begin{tikzpicture}[scale=1, line join=round, line cap=round]
		\clip (-0.5,-3) rectangle (5.5,4.7);
		\tkzDefPoints{0/0/A,5/0/C}
		\tkzDefTriangle[two angles = 90 and 40](A,C)    \tkzGetPoint{B}
		\tkzDrawBisector(A,B,C)\tkzGetPoint{D}
		\tkzCalcLength[cm](A,B) \tkzGetLength{rAB}
		\tkzInterLC[R](B,C)(B,\rAB cm) \tkzGetPoints{I2}{E}
		\tkzInterLL(A,B)(E,D) \tkzGetPoint{F}
		\tkzDrawAltitude(D,B)(C)\tkzGetPoint{K}
		\tkzDrawSegments(A,B B,C C,A A,F F,K D,K F,E)
		\tkzDrawPoints(A,B,C,D,E,F,K)
		\tkzLabelPoints[above](B)
		\tkzLabelPoints[left](A,F)
		\tkzLabelPoints[right](C,E)
		\tkzLabelPoints[below](K)
		\draw (D) node[shift={(-40:.4)}]{$D$};
		\tkzMarkRightAngles(B,A,C C,K,B)
		\tkzMarkAngles[size=0.7cm,arc=l,mark=|](A,B,D D,B,C)
		\tkzMarkAngles[size=0.7cm,arc=ll](B,C,A)
		\tkzLabelAngles[pos=1,rotate=30](B,C,A){$40^\circ$}
		\end{tikzpicture}
	\end{center}
\begin{enumerate}
	\item Vì $\triangle ABC$ vuông tại $A$ nên $\widehat{B}+\widehat{C}=90^\circ$ suy ra $\widehat{B}=50^\circ$.\\
	Vì $BD$ là tia phân giác của góc $B$ nên $\widehat{ABD}=\dfrac{\widehat{B}}{2}=\dfrac{50}{2}=25^\circ$.
	\item Xét $\triangle BAD$ và $\triangle BED$ có:
		\begin{itemize}
			\item $BA=BE$;
			\item $BD$ là cạnh chung;
			\item $\widehat{ABD}=\widehat{EBD}$ (vì $BD$ là tia phân giác của góc $B$).
		\end{itemize}
	Vậy $\triangle BAD=\triangle BED$ (c.g.c),
	suy ra $\widehat{DEB}=\widehat{DAB}$ (hai góc tương ứng), mà $\widehat{DAB}=90^\circ$ suy ra $\widehat{DEB}=90^\circ$ hay $DE \perp BC$.
	\item Xét tam giác $EBF$ vuông tại $E$ và tam giác $ABC$ vuông tại $A$ có
		\begin{itemize}
			\item $BE=BA$ (gt);
			\item $\widehat{ABC}$ là góc chung.
		\end{itemize}
	Vậy $\triangle ABC=\triangle EBF$ (cạnh góc vuông - góc nhọn kề).
	\item Xét tam giác $BFC$ có $BA$ và $FE$ là hai đường cao cắt nhau tại $D$ nên $D$ là trực tâm của $\triangle BFC$, do đó $BK$ là đường cao thứ ba của tam giác hay $BK \perp FC$.\\
	Mặt khác, $CK \perp BK$ suy ra $F, K, C$ thẳng hàng.
\end{enumerate}	
}
\end{ex}

Cho $A=1-\dfrac{3}{4}+\left(\dfrac{3}{4}\right)^2-\left(\dfrac{3}{4}\right)^3+\left(\dfrac{3}{4}\right)^4-\cdots - \left(\dfrac{3}{4}\right)^{2009}+\left(\dfrac{3}{4}\right)^{2010}$. Chứng tỏ $A$ không phải là số nguyên.
\loigiai{
\begin{eqnarray*}
	A&=&1-\dfrac{3}{4}+\left(\dfrac{3}{4}\right)^2-\left(\dfrac{3}{4}\right)^3+\left(\dfrac{3}{4}\right)^4-\cdots - \left(\dfrac{3}{4}\right)^{2009}+\left(\dfrac{3}{4}\right)^{2010}\\
	\dfrac{3}{4}A&=&\dfrac{3}{4}-\left(\dfrac{3}{4}\right)^2+\left(\dfrac{3}{4}\right)^3-\left(\dfrac{3}{4}\right)^4+\left(\dfrac{3}{4}\right)^5-\cdots-\left(\dfrac{3}{4}\right)^{2010}+\left(\dfrac{3}{4}\right)^{2011}.
\end{eqnarray*}
Suy ra $\dfrac{7}{4}A=\left(\dfrac{3}{4}\right)^{2011}+1$ hay $A=\dfrac{4}{7}\cdot \left(\dfrac{3}{4}\right)^{2011}+\dfrac{4}{7}$.\\
Vậy $A$ không phải là số nguyên.
}
\end{ex}
