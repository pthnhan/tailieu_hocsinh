\begin{ex}%[Đề thi vào 10, chuyên Tiền Giang, tỉnh Tiền Giang, năm 2018]%[Lục Hán Tường, Dự án EX-TS10C-18]%[9D0K1]
	Tính giá trị của biểu thức 
	\begin{align*}
		E = \left(x^5 + 3x^2 - 3x - 2\right)^{2018}
	\end{align*}
	biết $ x = \sqrt{\sqrt{3} - \sqrt{1 - \sqrt{21 - 12\sqrt{3}}}} $.
	\loigiai{
	Ta có 
	\begin{align*}
	x & = \sqrt{\sqrt{3} - \sqrt{1 - \sqrt{21 - 12\sqrt{3}}}}
	= \sqrt{\sqrt{3} - \sqrt{1 - \sqrt{\left(2\sqrt{3} - 3 \right)^2}}}\\
	& = \sqrt{\sqrt{3} - \sqrt{4 - 2\sqrt{3}}}
	= \sqrt{\sqrt{3} - \sqrt{\left(\sqrt{3}-1\right)^2}}\\
	& = \sqrt{\sqrt{3} - \sqrt{3} + 1} = 1.
	\end{align*}
	Suy ra
	\begin{align*}
		E = \left(x^5 + 3x^2 - 3x - 2\right)^{2018} = \left(1^5 + 3 \cdot 1^2 - 3 \cdot 1 - 2\right)^{2018} = \left(-1\right)^{2018} = 1.
	\end{align*}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Tiền Giang, tỉnh Tiền Giang, năm 2018]%[Lục Hán Tường, Dự án EX-TS10C-18]%[9D0G2]
	Giải phương trình 
	\begin{align*}
		5\sqrt{x^3 + 8} = 2\left(x^2 - x + 6\right).
	\end{align*}
	\loigiai{
	Điều kiện $	x^3 + 8 \geqslant 0 \Leftrightarrow x \geqslant -2 $.\\
	Với mọi $ x $ thỏa điều kiện, đặt $ \heva{&u = \sqrt{x + 2}\\&v = \sqrt{x^2 - 2x + 4}} $. Phương trình đã cho tương đương
	\begin{align*}
		& 5\sqrt{\left(x + 2\right)\left(x^2 - 2x + 4\right)} = 2\left(x^2 - x + 6\right) \\
		\Leftrightarrow & 5uv = 2\left(u^2 + v^2\right)
		\Leftrightarrow \left(u - 2v\right)\left(2u - v\right) = 0
		\Leftrightarrow \hoac{u = 2v\\v = 2u}
	\end{align*}
	Với $ u = 2v $, ta có
	\begin{align*}
		u = 2v \Leftrightarrow u^2 = 4v^2 & \Leftrightarrow x + 2 = 4\left(x^2 - 2x + 4\right)\\
		& \Leftrightarrow 4x^2 - 9x + 14 = 0 \quad {\text{(Vô nghiệm)}}
	\end{align*}
	Với $ 2u = v $, ta có
	\begin{align*}
		2u = v \Leftrightarrow 4u^2 = v^2 & \Leftrightarrow 4\left(x + 2\right) = x^2 - 2x + 4 \\
		& \Leftrightarrow x^2 - 6x - 4 = 0 \\
		& \Leftrightarrow \hoac{& x = 3 + \sqrt{13}\\& x = 3 - \sqrt{13}} \quad \text{(thỏa)}
	\end{align*}
	Vậy phương trình đã cho có hai nghiệm $ x = 3 + \sqrt{13} $ và $ x = 3 - \sqrt{13} $.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Tiền Giang, tỉnh Tiền Giang, năm 2018]%[Lục Hán Tường, Dự án EX-TS10C-18]%[9D0G2]
	Giải hệ phương trình 
	\begin{align*}
		\heva{& x^2 + y^2 + 2\left(x + y\right) = 7 \\
		& y\left(y - 2x\right) - 2x = 10.}
	\end{align*}
	\loigiai{
	Ta có
	\begin{align}
		\heva{& x^2 + y^2 + 2\left(x + y\right) = 7 \\& y\left(y - 2x\right) - 2x = 10}
		\Leftrightarrow \heva{& \left(x + 1\right)^2 + \left(y + 1\right)^2 = 9 \\& \left(y - 2x - 1\right)\left(y + 1\right) = 9.} \label{eq:1836.1}
	\end{align}
	Đặt $ \heva{& x + 1 = u\\& y + 1 = v} \Rightarrow y - 2x - 1 = v - 2u $. Suy ra hệ \eqref{eq:1836.1} tương đương
	\begin{align*}
		\heva{& u^2 + v^2 = 9\\& \left(v - 2u\right)v = 9} \Leftrightarrow \heva{& u^2 + v^2 = 9\\& v^2 - 2uv = 9.}
	\end{align*}
	Lấy phương trình trên trừ phương trình dưới ta được
	\begin{align*}
		u^2 + 2uv = 0 \Leftrightarrow \hoac{& u = 0\\& u = -2v.}
	\end{align*}
	Với $ u = 0 $, ta có
	\begin{align*}
		\heva{& u = 0\\ & v^2 = 9} \Leftrightarrow \heva{& u = 0\\ &\hoac{&v = 3\\&v = -3}} \Leftrightarrow \heva{& x = -1\\ &\hoac{&y = 2\\&y = -4}} \Leftrightarrow \hoac{\heva{& x = -1\\& y = 2}\\\heva{& x = -1\\& y = -4.}}
	\end{align*}
	Với $ u = -2v $, ta có
	\begin{align*}
		\heva{& u = -2v\\ & u^2 + v^2 = 9} \Leftrightarrow \heva{& u = -2v\\ & v^2 = \dfrac{9}{5}} \Leftrightarrow \hoac{& \heva{&u = -\dfrac{6\sqrt{5}}{5}\\&v = \dfrac{3\sqrt{5}}{5}}\\ &\heva{&u = \dfrac{6\sqrt{5}}{5}\\&v = -\dfrac{3\sqrt{5}}{5}}} \Leftrightarrow \hoac{& \heva{&x = -\dfrac{6\sqrt{5}}{5} - 1\\&y = \dfrac{3\sqrt{5}}{5} - 1}\\ &\heva{&x = \dfrac{6\sqrt{5}}{5} - 1\\&y = -\dfrac{3\sqrt{5}}{5} - 1.}}
	\end{align*}
	Vậy hệ đã cho có $ 4 $ nghiệm $ \left(-1;2\right) $, $ \left(-1;-4\right) $, $ \left(-\dfrac{6\sqrt{5}}{5} - 1;\dfrac{3\sqrt{5}}{5} - 1\right) $ và $ \left(\dfrac{6\sqrt{5}}{5} - 1;-\dfrac{3\sqrt{5}}{5} - 1\right) $.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Tiền Giang, tỉnh Tiền Giang, năm 2018]%[Lục Hán Tường, Dự án EX-TS10C-18]%[9D0G2]
	Cho phương trình 
	\begin{align*}
		x^2 - \left(2m + 4\right)x + 3m + 2 = 0.
	\end{align*}
	Xác định $ m $ để phương trình có hai nghiệm phân biệt $ x_1 $, $ x_2 $ thỏa mãn $ x_2 - 2x_1 = 3 $.
	\loigiai{
	Ta có $ \Delta = 4m^2 + 4m + 8 = \left(2m + 1\right)^2 + 7 > 0 $ với mọi $ x $. Nên phương trình đã cho luôn có hai nghiệm phân biệt $ x_1 $, $ x_2 $ thỏa
	\begin{align}
		& \heva{& x_1 + x_2 = 2m + 4\\
		& x_1x_2 = 3m + 2.} \label{eq:1836.2}
	\end{align}
	Suy ra
	\begin{align*}
		3\left(x_1 + x_2 - 4\right) = 2\left(x_1x_2 - 2\right).
	\end{align*}
	Lại có $ x_2 - 2x_1 = 3 $, nên
	\begin{align*}
		3\left(x_1 + x_2 - 4\right) = 2\left(x_1x_2 - 2\right)
		\Leftrightarrow  4x_1^2 - 3x_1 - 1 = 0 \Leftrightarrow \hoac{& x_1 = 1\\ & x_1 = -\dfrac{1}{4}.}
	\end{align*}
	Với $ x_1 = 1 \Rightarrow x_2 = 5 $. Thế vào hệ \eqref{eq:1836.2} ta được
	\begin{align*}
		\heva{& 2m + 4 = 6\\& 3m + 2 = 5} \Leftrightarrow m = 1.
	\end{align*}
	Với $ x_1 = -\dfrac{1}{4} \Rightarrow x_2 = \dfrac{5}{2} $. Thế vào hệ \eqref{eq:1836.2} ta được
	\begin{align*}
		\heva{& 2m + 4 = \dfrac{9}{4}\\& 3m + 2 = -\dfrac{5}{8}} \Leftrightarrow m = -\dfrac{7}{8}.
	\end{align*}
	Vậy tìm được $ 2 $ giá trị $ m $ thỏa yêu cầu bài toán là $ m = 1 $ và $ m = -\dfrac{7}{8} $.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Tiền Giang, tỉnh Tiền Giang, năm 2018]%[Lục Hán Tường, Dự án EX-TS10C-18]%[9D0G2]
	Trong hệ trục tọa độ $ Oxy $, cho Parabol $ (P) \colon y = \dfrac{1}{4}x^2 $. Tìm tọa độ hai điểm $ A $, $ B $ trên $ (P) $ sao cho $ A $, $ B $ đối xứng với nhau qua đường thẳng $ (d) \colon y = -2x + \dfrac{9}{2} $.
	\loigiai{
	Gọi $ A\left(a;\dfrac{a^2}{4}\right) $ và $ B\left(b;\dfrac{b^2}{4}\right) $, $ \left(a \neq b \right)$ là hai điểm phân biệt trên $ (P) $. Gọi $ M\left(\dfrac{a + b}{2};\dfrac{a^2 + b^2}{8}\right) $ là trung điểm $ AB $.\\
	Gọi $ (\Delta) $ là đường thẳng qua $ A $, $ B $. Ta có phương trình đường thẳng $ (\Delta) $
	\begin{align*}
		\dfrac{x - a}{b - a} = \dfrac{y - \dfrac{a^2}{4}}{\dfrac{b^2}{4} - \dfrac{a^2}{4}} \Leftrightarrow y = \left(\dfrac{a + b}{4}\right)x - ab
	\end{align*}
	Suy ra, để $ A $, $ B $ đối xứng qua đường thẳng $ d $ thì
	\begin{align*}
		\heva{& M \in (d)\\& (\Delta) \perp (d)} \Leftrightarrow \heva{& \dfrac{a^2 + b^2}{8} = -2\left(\dfrac{a + b}{2}\right) + \dfrac{9}{2}\\& \left(\dfrac{a + b}{4}\right)\left(-2\right) = -1} \Leftrightarrow \heva{&a^2 + b^2 = -8\left(a + b\right) + 36 \\& a + b = 2} \Leftrightarrow \heva{& ab = -8\\& a + b = 2}
	\end{align*}
	Do đó $ a $, $ b $ là nghiệm phương trình bậc $ 2 $
	\begin{align*}
		X^2 - 2X - 8 = 0 \Leftrightarrow \hoac{& X = 4\\& X = -2}
	\end{align*}
	Vậy hai điểm $ A $, $ B $ thỏa yêu cầu bài toán có tọa độ là $ A\left(4;4\right) $ và $ B\left(-2;1\right) $.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Tiền Giang, tỉnh Tiền Giang, năm 2018]%[Lục Hán Tường, Dự án EX-TS10C-18]%[9D0G4]
	Chứng minh rằng
	\begin{align*}
		M = a^4 + 6a^3 + 11a^2 + 30a
	\end{align*}
	chia hết cho $ 24 $ với mọi số nguyên $ a $.
	\loigiai{
	Chứng minh bài toán phụ: tích của $ 4 $ số nguyên liên tiếp thì chia hết cho $ 24 $.\\
	Thật vậy, gọi tích bốn số tự nhiên liên tiếp là $ P $
	\begin{listEX}
		\item Trường hợp 1: một trong $ 4 $ số bằng $ 0 $. Ta có $ P = 0 $ nên $ P \ \vdots \ 24 $.
		\item Trường hợp 2: với 4 số đều khác $ 0 $.\\
		Trong $ 4 $ số tự nhiên liên tiếp luôn tồn tai $ 2 $ số chẵn, một số chia hết cho $ 2 $ và một số chia hết cho $ 4 $. Do đó $ P \ \vdots \ 8 $.\\
		Lại có, trong $ 4 $ số tự nhiên liên tiếp luôn tồn tai một số chia hết cho $ 3 $, nên $ P \ \vdots \ 3 $.\\
		Mà $ 8 $ và $ 3 $ là hai số tự nhiên nguyên tố cùng nhau, nên $ P \ \vdots \ 24 $.
	\end{listEX}
	Áp dụng bài toán phụ, suy ra
	\begin{align*}
		a\left(a + 1\right)\left(a + 2\right)\left(a + 3\right)\ \vdots \ 24 .
	\end{align*}
	Do đó
	\begin{align*}
		a^4 + 6a^3 + 11a^2 + 30a = a\left(a + 1\right)\left(a + 2\right)\left(a + 3\right) + 24a \ \vdots \ 24 \Leftrightarrow M \ \vdots \ 24.
	\end{align*}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Tiền Giang, tỉnh Tiền Giang, năm 2018]%[Lục Hán Tường, Dự án EX-TS10C-18]%[8H0B3]%[9H0K2]%[9H0K5]
	Cho tam giác nhọn $ ABC $ có $ AB < AC $ và nội tiếp đường tròn tâm $ O $. Đường tròn tâm $ K $ đường kính $ BC $ cắt các cạnh $ AB $, $ AC $ lần lượt tại $ E $, $ F $. Gọi $ H $ là giao điểm của $ BF $ và $ CE $.
	\begin{enumerate}
		\item Chứng minh tam giác $ AEF $ và tam giác $ ACB $ đồng dạng.
		\item Gọi $ A' $ là điểm đối xứng của $ A $ qua $ O $. Chứng minh $ AA' $ vuông góc với $ EF $.
		\item Từ $ A $ dựng các tiếp tuyến $ AM $, $ AN $ đến đường tròn $ (K) $ với $ M $, $ N $ là các tiếp điểm. Chứng minh ba điểm $ M $, $ H $, $ N $ thẳng hàng.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Chứng minh tam giác $ AEF $ và tam giác $ ACB $ đồng dạng.
		\immini{
		Ta có tứ giác $ BEFC $ nội tiếp
		\begin{align*}
			\Rightarrow & \heva{& \widehat{AEF} + \widehat{BEF} = 180^\circ  \\& \widehat{BCF} + \widehat{BEF} = 180^\circ \ (\text{tứ giác }BEFC \text{ nội tiếp})}\\
			\Rightarrow & \widehat{AEF} = \widehat{BCF}.
		\end{align*}
		Xét $ \triangle AEF $ và $ \triangle ACB $, có
		\begin{align*}
			\heva{& A \text{ chung}\\& \widehat{AEF} = \widehat{BCF}} \Rightarrow \triangle AEF \backsim \triangle ACB .
		\end{align*}
		}
		{	
			\begin{tikzpicture}
			
				% Định nghĩa Điểm
				\tkzDefPoints{0/0/B,5/0/C,2/4/A}
				\tkzDefMidPoint(B,C)\tkzGetPoint{K}
				\tkzDrawCircle(K,B)
				\tkzInterLC(A,B)(K,B)\tkzGetSecondPoint{E}
				\tkzInterLC(A,C)(K,C)\tkzGetFirstPoint{F}
				
				% Vẽ điểm
				\tkzDrawPoints(A,B,C,K,E,F)
				\tkzLabelPoints[above](A)
				\tkzLabelPoints[left](B)
				\tkzLabelPoints[above left](E)
				\tkzLabelPoints[right](C)
				\tkzLabelPoints[above right](F)
				\tkzLabelPoints[below](K)
				
				% Vẽ đoạn thẳng, đường thẳng
				\tkzDrawSegments(A,B A,C B,C E,F)
				
				% Vẽ Góc
				\tkzMarkAngles[size=0.5cm](A,C,B F,E,A)
				
			\end{tikzpicture}
		}	
		
		\item Gọi $ A' $ là điểm đối xứng của $ A $ qua $ O $. Chứng minh $ AA' $ vuông góc với $ EF $.
		\immini{
		Do tứ giác $ ABA'C $ nội tiếp nên $ \widehat{AA'B} = \widehat{ACB} \Rightarrow \widehat{AA'B} = \widehat{AEF} $.\\
		Gọi $ I = AA' \cap EF $.\\
		Xét $ \triangle AEI $ và $ \triangle AA'B $, có
		\begin{align*}
			& \heva{& A \text{ chung}\\& \widehat{AEI} = \widehat{AA'B}} \Rightarrow \triangle AEI \backsim \triangle AA'B.\\
			& \Rightarrow \widehat{AIE} = \widehat{ABA'} = 90^\circ \Leftrightarrow AA' \perp EF.
		\end{align*}
		}
		{
			\begin{tikzpicture}
			
				% Định nghĩa Điểm
				\tkzDefPoints{0/0/B,5/0/C,2/4/A}
				\tkzDefMidPoint(B,C)\tkzGetPoint{K}
				\tkzDefCircle[circum](A,B,C)\tkzGetPoint{O}
				\tkzDrawCircle(O,B)
				\tkzInterLC(A,B)(K,B)\tkzGetSecondPoint{E}
				\tkzInterLC(A,C)(K,C)\tkzGetFirstPoint{F}
				\tkzInterLC(A,O)(O,A)\tkzGetSecondPoint{A'}
				\tkzInterLL(A,A')(E,F)\tkzGetPoint{I}
				
				% Vẽ điểm
				\tkzDrawPoints(A,B,C,E,F,O,A',I)
				\tkzLabelPoints[above](A)
				\tkzLabelPoints[left](B)
				\tkzLabelPoints[above left](E)
				\tkzLabelPoints[right](C,O)
				\tkzLabelPoints[above right](F,I)
				\tkzLabelPoints[below](A')
				
				% Vẽ đoạn thẳng, đường thẳng
				\tkzDrawSegments(A,B A,C B,C E,F A,A' B,A')
				
				% Vẽ Góc
				\tkzMarkAngles[size=0.5cm](A,C,B F,E,A A,A',B)
				\tkzMarkRightAngles(A',B,A A,I,E)
				
			\end{tikzpicture}
		}
		
		\item Từ $ A $ dựng các tiếp tuyến $ AM $, $ AN $ đến đường tròn $ (K) $ với $ M $, $ N $ là các tiếp điểm. Chứng minh ba điểm $ M $, $ H $, $ N $ thẳng hàng.
		\begin{center}
			\begin{tikzpicture}
			
				% Định nghĩa Điểm
				\tkzDefPoints{0/0/B,6/0/C,2/5/A}
				\tkzDefMidPoint(B,C)\tkzGetPoint{K}
				\tkzDefMidPoint(A,K)\tkzGetPoint{L}
				
				\tkzDrawCircle(K,B)
				\tkzInterLC(A,B)(K,B)\tkzGetSecondPoint{E}
				\tkzInterLC(A,C)(K,C)\tkzGetFirstPoint{F}
				\tkzInterCC(L,A)(K,C)\tkzGetSecondPoint{M}
				\tkzInterCC(L,A)(K,C)\tkzGetFirstPoint{N}
				\tkzInterLL(C,E)(B,F)\tkzGetPoint{H}
				\tkzInterLL(A,H)(B,C)\tkzGetPoint{D}
				
				\tkzDrawCircle[circum](A,K,M)
				
				% Vẽ điểm
				\tkzDrawPoints(A,B,C,E,F,M,N,H,K,D)
				\tkzLabelPoints[above](A)
				\tkzLabelPoints[left](B,M)
				\tkzLabelPoints[above left](E,H)
				\tkzLabelPoints[right](C,N)
				%\tkzLabelPoints[above right](F)
				\tkzLabelPoints[below](K,D,F)
				
				% Vẽ đoạn thẳng, đường thẳng
				\tkzDrawSegments(A,B A,C B,C A,M A,N B,F M,N B,M N,C F,N A,D A,K D,N K,M K,N)
				
				% Vẽ Góc
				\tkzMarkAngles[size=0.3cm](N,D,A N,K,A A,K,M)
				
			\end{tikzpicture}
		\end{center}
		Gọi $ D $ là chân đường cao kẻ $ A $ của tam giác $ ABC $, ta có $ \widehat{AMK} = \widehat{ANK} = \widehat{ADK} = 90^\circ $. Suy ra $ 5 $ điểm $ A $, $ M $, $ D $, $ K $, $ N $ cùng thuộc một đường tròn đường kính $ AK $.
		\begin{align*}
			\Rightarrow \heva{& \widehat{ANM} = \widehat{AKM}\\& \widehat{ADN} = \widehat{AKN}.}
		\end{align*}
		Lại có, do $ AM $, $ AN $ là tiếp tuyến. Suy ra
		\begin{align}
			\widehat{AKN} = \widehat{AKM} \Rightarrow \widehat{ADN} = \widehat{ANM}. \label{eq:1836.3}
		\end{align}
		Mặt khác, xét $ \triangle AFN $ và $ \triangle ANC $, có
		\begin{align}
			\heva{& A \text{ chung}\\ & \widehat{ACN} = \widehat{ANF} = \dfrac{1}{2} } \Rightarrow \triangle AFN \backsim \triangle ANC \Rightarrow \dfrac{AN}{AC} = \dfrac{AF}{AN} \Leftrightarrow AN^2 = AF \cdot AC .\label{eq:1836.4}
		\end{align}
		Lại xét $ \triangle ADC $ và $ \triangle AFH $, có
		\begin{align}
			\heva{& A \text{ chung}\\ & \widehat{AFH} = \widehat{ADC} = 90^\circ } \Rightarrow \triangle AFH \backsim \triangle ADC \Rightarrow \dfrac{AF}{AD} = \dfrac{AH}{AC} \Leftrightarrow AF \cdot AC = AH \cdot AD .\label{eq:1836.5}
		\end{align}
		Từ \eqref{eq:1836.4} và \eqref{eq:1836.5} suy ra
		\begin{align}
			AN^2 = AH \cdot AD \Rightarrow \dfrac{AN}{AH} = \dfrac{AD}{AN} \Rightarrow \triangle AND \backsim \triangle AHN \Rightarrow \widehat{ADN} = \widehat{ANH}. \label{eq:1836.6}
		\end{align}
		Từ \eqref{eq:1836.3} và \eqref{eq:1836.6} suy ra
		\begin{align*}
			\widehat{ANH} = \widehat{ANM}.
		\end{align*}
		Do đó ba điểm $ M $, $ H $, $ N $ thẳng hàng.
	\end{enumerate}
	}
