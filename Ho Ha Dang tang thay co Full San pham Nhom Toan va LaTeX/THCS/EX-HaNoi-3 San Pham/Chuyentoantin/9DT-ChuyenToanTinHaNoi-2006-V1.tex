\begin{ex}[Đề thi Chuyên Hà Nội năm 2005 - 2006, Vòng 1][9D1K8]
Cho biểu thức $P = \dfrac{x\sqrt{x} - 1}{x - \sqrt{x}} - \dfrac{x\sqrt{x} + 1}{x + \sqrt{x}} + \dfrac{x + 1}{\sqrt{x}}$.
\begin{enumerate}[a)]
	\item Rút gọn biểu thức $P$.
	\item Tìm $x$ để $P = \dfrac{9}{2}$.
\end{enumerate}
\loigiai{
	\begin{enumerate}[a)]
		\item Điều kiện để $P$ có nghĩa là $x > 0, x \ne 1$.
		\begin{align*}
			P&= \dfrac{(\sqrt{x}-1)(x+ \sqrt{x}+1)}{\sqrt{x}(\sqrt{x}-1)}- \dfrac{(\sqrt{x}+1)(x- \sqrt{x}+1)}{\sqrt{x}(\sqrt{x}+1)}+ \dfrac{x+1}{\sqrt{x}}\\
			&= \dfrac{x+ \sqrt{x}+1}{\sqrt{x}}- \dfrac{x- \sqrt{x}+1}{\sqrt{x}}+ \dfrac{x+1}{\sqrt{x}}= \dfrac{x+ \sqrt{x}+1- x + \sqrt{x} - 1+ x +1}{\sqrt{x}}\\
			&= \dfrac{(\sqrt{x}+1)^2}{\sqrt{x}}.
		\end{align*}
		\item $P = \dfrac{9}{2} \Leftrightarrow \dfrac{(\sqrt{x}+1)^2}{\sqrt{x}}= \dfrac{9}{2} \Leftrightarrow 2x - 5\sqrt{x} + 2= 0$. Phương trình này có hai nghiệm là $x =4, x= \dfrac{1}{4}$.
	\end{enumerate}
}
\end{ex}

\begin{ex}[Đề thi Chuyên Hà Nội năm 2005 - 2006, Vòng 1][8D4K4]
Cho bất phương trình $3\left(m - 1\right)x  + 1>  2m + x$ ($m$ là tham số).
\begin{enumerate}[a)]
	\item Giải bất phương trình với $m = 1 - 2\sqrt{2}$.
	\item Tìm $m$ để bất  phương trình nhận mọi giá trị $x > 1$ là nghiệm.
	\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
	\item Với $m = 1 - 2\sqrt{2}$ bất phương trình tương đương với
\[3\left(1- 2\sqrt{2} - 1\right)x + 1> 2(1- 2\sqrt{2})+ x \Leftrightarrow (-6\sqrt{2}- 1)x> 1- 4\sqrt{2} \Leftrightarrow x< \dfrac{4\sqrt{2}-1}{1+ 6\sqrt{2}}.\]
Vậy với $m = 1 - 2\sqrt{2}$ thì bất phương trình có nghiệm $x< \dfrac{4\sqrt{2}-1}{1+ 6\sqrt{2}}$.
	\item $3\left(m - 1\right)x  + 1>  2m + x \Leftrightarrow (3m -4)x > 2m-1$. \hfill (*)
		\begin{itemize}
			\item Nếu $3m-4=0 \Leftrightarrow m = \dfrac{4}{3}$ thì (*) $\Leftrightarrow 0 \cdot x> \dfrac{5}{3} \text{ (vô lý)}$.
			\item Nếu $3m-4<0 \Leftrightarrow m < \dfrac{4}{3}$ thì (*) $\Leftrightarrow x < \dfrac{2m-1}{3m-4}$. Khi đó không tồn tại giá trị của $m$ để (*) nhận mọi giá trị $x>1$ là nghiệm.
			\item Nếu $3m-4>0 \Leftrightarrow m > \dfrac{4}{3}$ thì (*) $\Leftrightarrow x > \dfrac{2m-1}{3m-4}$. Khi đó
				\[\text{YCBT} \Leftrightarrow \dfrac{2m-1}{3m-4} \le 1 \Leftrightarrow \dfrac{2m-1}{3m-4} -1 \le 0 \Leftrightarrow \dfrac{-m+3}{3m-4} \le 0 \Leftrightarrow \hoac{&m<\dfrac{4}{3} \text{ (loại)} \\ &m \ge 3 \text{ (chọn)}.}\]
		\end{itemize}
	Vậy với $m \ge 3$ thì bất phương trình nhận mọi giá trị $x>1$ là nghiệm.
\end{enumerate}	
}
\end{ex}

\begin{ex}[Đề thi Chuyên Hà Nội năm 2005 - 2006, Vòng 1][9D4K6]
Trên mặt phẳng tọa độ $Oxy$,  cho đường thẳng $(d): 2x - y - a^2 = 0$ và Parabol $(P): y = ax^2$ ($a$ là tham số dương).
	\begin{enumerate}[a)]
		\item Tìm $a$ để $(d)$ cắt $\left(P\right)$ tại hai điểm phân biệt $A$, $B$. Chứng minh rằng khi đó $A$, $B$ nằm về bên phải trục tung.
		\item Gọi $u$, $v$ theo thứ tự là hoành độ của $A$, $B$. Tìm giá trị nhỏ nhất của biểu thức $T = \dfrac{4}{u + v} + \dfrac{1}{uv}.$ 
	\end{enumerate}
\loigiai{
	\begin{enumerate}[a)]
	\item $(d): 2x - y - a^2 = 0 \Leftrightarrow y= 2x - a^2$.\\
Để $d$ cắt $P$ tại hai điểm phân biệt $A, B$ thì phương trình hoành độ có hai nghiệm phân biệt, tức là $2x - a^2= ax^2 \Leftrightarrow ax^2 - 2x+ a^2=0$. \hfill (1) \\
Phương trình (1) có hai nghiệm phân biệt khi và chỉ khi 
\[\heva{&a \ne 0\\ & \Delta^{'}= 1- a^3>0} \Leftrightarrow 0<a<1 \text{ (do } a>0).\]
Từ đó giả sử $u, v$ là hoành độ của $A, B$, ta có hệ thức Vi-ét $\heva{&u+v= \dfrac{2}{a}>0\\ &u \cdot v=a>0.}$\\
Suy ra phương trình (1) có hai nghiệm dương phân biệt hay hai điểm $A, B$ nằm về bên phải trục tung.
	\item Ta có $T = \dfrac{4}{u + v} + \dfrac{1}{uv}= 2a+ \dfrac{1}{a} \geq 2\sqrt{2a \cdot \dfrac{1}{a}}= 2\sqrt{2}.$ Vậy giá trị nhỏ nhất của $T$ là $2\sqrt{2}$ đạt được khi $a = \dfrac{\sqrt{2}}{2}.$ 
\end{enumerate}	
}
\end{ex}

\begin{ex}[Đề thi Chuyên Hà Nội năm 2005 - 2006, Vòng 1][9H3K8][9H3G8]
Cho  đường tròn tâm $O$ có dây cung $AB$ cố định và điểm $I$ là điểm chính giữa của cung lớn $AB$. Lấy điểm $M$ bất kỳ trên cung lớn $AB$, dựng tia $Ax$ vuông góc với đường thẳng $MI$ tại $H$ và cắt tia $BM$ tại $C$. 
\begin{enumerate}[a)]
\item Chứng minh rằng $\triangle AIB$ và $\triangle AMC$ là các tam giác cân.
\item Khi điểm $M$  di động trên cung lớn $AB$, chứng minh rằng điểm $C$ di chuyển trên một cung tròn cố định.
\item Xác định vị trí điểm $M$ để chu vi $\triangle AMC$ đạt giá trị lớn nhất.
\end{enumerate}
\loigiai{
	\immini 
	{
		\begin{enumerate}
			\item[a)] Do $I$ là điểm chính giữa của cung $AB$ nên $IA= IB$ hay tam giác $IAB$ cân tại $I$.\\
			Nếu $M$ thuộc cung $IA; \text{(không chứa B)}$ thì $\widehat{CMH}= \widehat{IMB}; \text{(đối đỉnh)}.$\\
			Mặt khác $\widehat{AMI}= \dfrac{1}{2}\text{sđ}\wideparen{AI}\Rightarrow \widehat{AMH}= 180^{\circ}- \widehat{AMI}= \dfrac{1}{2}\text{sđ}(\wideparen{AM}+ \wideparen{AI})= \dfrac{1}{2}\text{sđ}\ \wideparen{IA}= \dfrac{1}{2}\text{sđ}\ \wideparen{IB}.$\\
			Hay $\widehat{AMH}= \widehat{IMB}= \widehat{HMC}$. Suy ra $HM$ vừa là đường cao, vừa là đường phân giác của tam giác $AMC$. Suy ra $\triangle AMC$ cân tại $M$.\\
			Nếu $M$ thuộc cung $IB; \text{(không chứa A)}$ thì lập luận tương tự ta cũng có $\triangle AMC$ cân tại $M$.
			
		\end{enumerate}
	}
	{
		\begin{tikzpicture}[scale=0.45]
		\tkzInit[ymin=-7.1,ymax=8,xmin=-7.3,xmax=7]
		%\tkzAxeXY
		\tkzClip
		\tkzDefPoints{0/0/I,-3/6/A,3/6/B,0/3.75/O,-2.55/1/M,3/1.5/M_0}
		\tkzDrawCircle[circum](I,A,B)
		\tkzDefPointBy[projection= onto M--I](A)
		\tkzGetPoint{H}
		\tkzInterLL(A,H)(B,M)
		\tkzGetPoint{C}
		\tkzInterLL(A,I)(B,M_0)
		\tkzGetPoint{C_0}
		\tkzMarkRightAngle[size=0.5,fill=gray!50](A,H,M)
		\tkzDrawCircle[radius](I,C_0)
		\tkzDrawSegments(A,B A,I I,B M,I B,I A,H A,M M,H B,M H,C M,C A,O O,M_0 B,M_0 I,C_0 M_0,C_0)
		\tkzDrawPoints(A,I,B,O,M,H,C,M_0,C_0)
		\tkzLabelPoints[above](A,B,O)
		\tkzLabelPoints[above right](M_0)
		\tkzLabelPoints[above left](H)

	}
	\begin{enumerate}
		\item[b)] Do $IH$ là trung trực của $AC$ nên $IA= IC$, mà $IA$ không đổi nên $C$ nằm trên đường tròn tâm $I$ bán kính $IA$. Khi $M \equiv A$ thì $C \equiv A$, còn khi $M \equiv B$ thì $C \equiv C_1$ (trong đó $C_1$ là giao điểm của đường thẳng qua $A$ vuông góc với $IB$ với đường tròn tâm $I$ bán kính $IA$). Do đó khi $M$ chuyển động trên cung lớn $AB$ thì $C$ chuyển động trên $\wideparen{AC_1}$ của đường tròn $(I;IA)$.
		\item[c)] Giả sử $C_0$ là giao điểm của $AI$ với đường tròn tâm $I$, bán kính $IA$. Còn $M_0$ là điểm đối xứng của $A$ qua $O$. Nhận thấy rằng, khi tia $Ax$ trùng với tia $AI$ thì $M$ trùng $M_0$, lúc đó $B, M_0, C_0$ thẳng hàng. Từ mối liên hệ giữa độ dài dây cung và độ dài đường kính ta có $AM \leq AM_0, AC \leq AC_0$.\\
		Suy ra $p(\triangle AMC)\leq p(\triangle AM_0C_0)$. Vậy chu vi $\triangle AMC$ lớn nhất khi $AM$ là đường kính của đường tròn tâm $O$.
	\end{enumerate}
}
\end{ex}

\begin{ex}[Đề thi Chuyên Hà Nội năm 2005 - 2006, Vòng 1][9H1G4]
Cho $\triangle ABC$ vuông ở $A$ có $AB < AC$ và trung tuyến  $AM$, góc $\widehat{ACB} = \alpha$ và $\widehat{AMB} = \beta$.  Chứng minh rằng $\left(\sin \alpha + \cos\alpha \right)^2 = 1 + \sin\beta$.
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=0.7]
\tkzInit[ymin=-1,ymax=6,xmin=-2,xmax=9]
\tkzClip
\tkzDefPoints{0/0/A,0/5/B,8/0/C}
\tkzDefMidPoint(B,C)
\tkzGetPoint{M}
\tkzDefPointBy[projection= onto B--C](A)
\tkzGetPoint{H}
\tkzMarkRightAngles[size=0.4,fill=gray!50](B,A,C B,H,A)
\tkzDrawSegments(A,B B,C C,A A,M A,H)
\tkzMarkAngles[size=0.8cm](B,C,A)
\tkzMarkAngles[arc=ll,size=0.7cm](H,M,A)
\tkzDrawPoints(A,B,C,H,M)
\tkzLabelPoints[above](B)
\tkzLabelPoints[left](A)
\tkzLabelPoints[right](C)
\tkzLabelPoints[above](H)
\tkzLabelPoints[above](M)
\end{tikzpicture}
\end{center}
Dựng $AH \perp BC$, do $AB< AC$ nên $H \in BM$. Ta có $AH= AM \cdot \sin \beta= \dfrac{1}{2}BC \cdot
 \sin \beta$. \hfill (1) \\
 Mặt khác $AH= AC \cdot \sin \alpha= BC \cdot \sin \alpha \cdot \cos \alpha$. \hfill (2) \\
 Từ (1) và (2) suy ra $\sin \beta= \sin \alpha \cdot \cos \alpha \Rightarrow 1+ \sin \beta= (\sin \alpha + \cos \alpha)^2.$
}
\end{ex}
