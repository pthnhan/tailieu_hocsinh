\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Đăk Lăk, 2016]%[9D1K8]%[9D5G5]
\begin{enumerate}[1)]
\item Cho đa thức $P(x)=x^9-17x^8+m$. Tìm $m$ biết rằng $a=\sqrt{3}-\sqrt{3-\sqrt{13-2\sqrt{12}}}$ là một nghiệm của $P(x)$.
\item Cho $2016$ số dương $a_1,a_2,\ldots,a_{2015},a_{2016}$ thỏa mãn $\dfrac{a_1}{a_2}=\dfrac{a_2}{a_3}=\ldots=\dfrac{a_{2016}}{a_1}$. Hãy tính giá trị của biểu thức $A=\dfrac{a_1^2+a_2^2+\ldots+a_{2016}^2}{\left(a_1+a_2+\ldots+a_{2016}\right)^2}$.
\end{enumerate}
\loigiai{
\begin{enumerate}[1)]
\item Ta có $a=\sqrt{3}-\sqrt{3-\sqrt{13-2\sqrt{12}}}=\sqrt{3}-\sqrt{3-\sqrt{\left(2\sqrt{3}-1\right)^2}}=\sqrt{3}-\sqrt{\left(\sqrt{3}-1\right)^2}=1$.\\
Vì $a=1$ là một nghiệm của $P(x)$, nên ta có: $P(1)=0\Leftrightarrow 1^9-17.1^8+m=0\Leftrightarrow m=16$.
\item Ta có $\dfrac{a_1}{a_2}=\dfrac{a_2}{a_3}=\ldots=\dfrac{a_{2016}}{a_1}\Rightarrow \dfrac{a_1}{a_2}=\dfrac{a_2}{a_3}=\ldots=\dfrac{a_{2016}}{a_1}=\dfrac{a_1+a_2+\ldots+a_{201}}{a_1+a_2+\ldots+a_{201}}=1\\
\Rightarrow a_1=a_2=\ldots=a_{2016}=k>0$.\\
Do đó $A=\dfrac{a_1^2+a_2^2+\ldots+a_{2016}^2}{\left(a_1+a_2+\ldots+a_{2016}\right)^2}=\dfrac{2016k^2}{(2016k)^2}=\dfrac{1}{2016}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Đăk Lăk, 2016]%[9D4K7]%[9D3K4]%[9D5G1]
\begin{enumerate}[1)]
\item Giải phương trình: $\sqrt{2x-3}+x^2-5x+5=0$.
\item Giải hệ phương trình: $\heva{&2(x+y)=3xy\\&6(y+z)=5yz\\&3(x+z)=4xz.}$
\item Cho $x,y,z$ là các số thực dương thỏa mãn $x\geq y\geq z$ và $x+y+z=3$. Tìm giá trị nhỏ nhất của biểu thức $B=\dfrac{x}{z}+\dfrac{z}{y}+3y$.
\end{enumerate}
\loigiai{
\begin{enumerate}[1)]
\item Điều kiện: $\heva{&2x-3\geq 0\\&-x^2+5x-5\geq 0}\Leftrightarrow\heva{&x\geq \dfrac{3}{2}\\&-x^2+5x-5\geq 0}$\hfill(*)\\
$\sqrt{2x-3}+x^2-5x+5=0\Leftrightarrow \sqrt{2x-3}=-x^2+5x-5\Leftrightarrow 2x-3=x^4+25x^2+25-10x^3+10x^2-50x\\
\Leftrightarrow x^4-10x^3+35x^2-52x+28=0\Leftrightarrow (x-2)^2(x^2-6x+7)=0\Leftrightarrow \hoac{&x=2\\&x=3-\sqrt{2}\\&x=3+\sqrt{2}}$.\\
Nhận thấy chỉ có $x=2$ và $x=3-\sqrt{2}$ thỏa mãn (*).
\item Rõ ràng $x=0,y=0,z=0$ là một nghiệm của hệ.\\
Với $x\neq 0, y\neq 0,z\neq 0$, ta có\\
$\heva{&2(x+y)=3xy\\&6(y+z)=5yz\\&3(x+z)=4xz}\Leftrightarrow\heva{&\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{3}{2}\\&\dfrac{1}{y}+\dfrac{1}{z}=\dfrac{5}{6}\\&\dfrac{1}{z}+\dfrac{1}{x}=\dfrac{4}{3}}\Leftrightarrow \heva{&2\left(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\right)=\dfrac{11}{3}\\&\dfrac{1}{y}+\dfrac{1}{z}=\dfrac{5}{6}\\&\dfrac{1}{z}+\dfrac{1}{x}=\dfrac{4}{3}}\Leftrightarrow \heva{&\dfrac{1}{x}=1\\&\dfrac{1}{y}=\dfrac{1}{2}\\&\dfrac{1}{z}=\dfrac{1}{3}}\Leftrightarrow \heva{&x=1\\&y=2\\&z=3}$.\\
Vậy hệ có hai nghiệm $(0;0;0)$ và $(1;2;3)$.
\item Vì $x\geq y\geq z>0\Rightarrow (x-z)(y-z)\geq 0\Rightarrow xy+z^2\geq xz+yz\Rightarrow \dfrac{x}{z}+\dfrac{z}{y}\geq \dfrac{x}{y}+1$ (do $yz>0$).\\
Do đó $P\geq \dfrac{x}{y}+3y+1=\dfrac{x+2y}{y}+3y-1\geq 2\sqrt{3(x+2y)}-1\geq 2\sqrt{3(x+y+z)}-1=2\sqrt{3^2}-1=5$.\\
Vậy $\min P=5\Leftrightarrow \heva{&x\geq y\geq z>0\\&x+y+3=3\\&(x-z)(y-z)=0\\&\dfrac{x+2y}{y}=3y}\Leftrightarrow x=y=z=1$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Đăk Lăk, 2016]%[9D5G2]%[9D5G2]
\begin{enumerate}[1)]
\item Tìm cặp số nguyên tố $(m,n)$ sao cho: $m^2-2n^2-1=0$.
\item Cho hai số tự nhiên $a,b$ sao cho $a^2+b^2+ab$ chia hết cho $10$. Chứng minh rằng $a^2+b^2+ab$ chia hết cho $100$.
\end{enumerate}
\loigiai{
\begin{enumerate}[1)]
\item Ta có $m^2-2n^2-1=0\Rightarrow (m-1)(m+1)=2n^2\vdots 2\Rightarrow m$ lẻ, suy ra $(m-1)(m+1)\vdots 4\\
\Rightarrow 2n^2\vdots 4\Rightarrow n\vdots 2\Rightarrow n=2$. Do $n$ là số nguyên tố.\\
Khi đó $m^2=2.2^2+1\Rightarrow m=3$. Vậy cặp số nguyên tố cần tìm là $(m,n)=(3,2)$.
\item Ta có $a^2+b^2+ab\vdots 10\Rightarrow (a-b)(a^2+b^2+ab)=(a-b)^3\vdots 10\Rightarrow a^3\equiv b^3(mod \,10)$\\
suy ra $a\equiv b(mod \,10)\Rightarrow ab\equiv a^2(mod 10), a^2\equiv b^2(mod\, 10)\Rightarrow a^2+b^2+ab\equiv 3a^2(mod \,10)\\
\Rightarrow 3a^2\equiv 0(mod 10)\Rightarrow a^2\vdots 10$ (vì $(3;10)=1$), suy ra $a\vdots 10$.\\
Suy ra $a^2\vdots 100\Rightarrow a^2+b^+ab\equiv 3a^2(mod \,100)$.\\
Vậy $a^2+b^2+ab$ chia hết cho $100$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Đăk Lăk, 2016]%[8H3K9]
Cho hình chữ nhật $ABCD$, biết $AD=\dfrac{2}{3}AB$. Trên cạnh $BC$ lấy điểm $M$ sao cho đường thẳng $AM$ cắt đường thẳng $CD$ tại $I$. Lấy điểm $P$ thuộc cạnh $AB$, điểm $Q$ thuộc cạnh $CD$ sao cho $PQ$ vuông góc với $AM$. Đường phân giác của góc $\widehat{MAD}$ cắt $CD$ tại $H$. Chứng minh rằng:
\begin{multicols}{2}
\begin{enumerate}[a)]
\item $PQ=\dfrac{2}{3}BM+DH$.
\item $\dfrac{1}{AB^2}=\dfrac{1}{AM^2}+\dfrac{2}{9AI^2}$.
\end{enumerate}
\end{multicols}
\loigiai{
\immini{
\begin{enumerate}[a)]
\item Chứng minh $PQ=\dfrac{2}{3}BM+DH$\\
Kẻ $HK\parallel PQ, (K\in AB), PK\parallel HQ, (AB\parallel CD)\\
\Rightarrow PQ=HK$.\\
Lại có $HK\parallel PQ, PQ\perp AI$ (giả thiết)\\
Suy ra $HK\perp AI$ (tại $E, E$ là giao điểm của $HK$ và $AI$)\\
$\triangle ADH=\triangle AEH$ (cạnh huyền, góc nhọn)\\
suy ra $AD=AE, DH=EH$\\
Lại có $\triangle AEK\backsim \triangle ABM$ (g-g). \\
Suy ra $\dfrac{EK}{BM}=\dfrac{AE}{AB}=\dfrac{AD}{AB}$, mà $AD=\dfrac{2}{3}AB$, \\
suy ra $EK=\dfrac{2}{3}BM$.\\
Do đó $PQ=HK=EK+EH=\dfrac{2}{3}BM+DH$.
\item Chứng minh $\dfrac{1}{AB^2}=\dfrac{1}{AM^2}+\dfrac{4}{9AI^2}$.\\
Kẻ $AF\perp AM$ ($F$ thuộc đường thẳng $BC$).\\
Xét $\triangle ABF$ và $\triangle ADI$, ta có: $\widehat{ABF}=\widehat{ADI}=90^\circ,\\
\widehat{BAF}=\widehat{DAI}$ (cùng phụ với $\widehat{BAM}$).\\
Vậy $\triangle ABF\backsim \triangle ADI\Rightarrow \dfrac{AI}{AF}=\dfrac{AD}{AB}=\dfrac{2}{3}\\
\Rightarrow \dfrac{1}{AF^2}=\dfrac{4}{9AI^2}$.\\
Xét $\triangle AMF: \widehat{MAF}=90^\circ\,(gt), AB\perp MF\Rightarrow \dfrac{1}{AB^2}=\dfrac{1}{AM^2}+\dfrac{1}{AF^2}=\dfrac{1}{AM^2}+\dfrac{4}{9AI^2}$ (đpcm)
\end{enumerate}
}{
\begin{tikzpicture}[>=stealth,scale=1,font=\footnotesize]
\tkzDefPoints{0/0/A, 4/0/B, 0/-3/D, 4/-3/C}
\coordinate (M) at ($(B)!4/5!(C)$);
\tkzInterLL(A,M)(C,D)\tkzGetPoint{I}
\coordinate (P) at ($(A)!2/3!(B)$);
\tkzDefLine[perpendicular=through P](A,M)\tkzGetPoint{p}
\tkzInterLL(P,p)(C,D)\tkzGetPoint{Q}
\tkzDefLine[bisector](M,A,D)\tkzGetPoint{a}
\tkzInterLL(A,a)(C,D)\tkzGetPoint{H}
\tkzDefLine[parallel=through H](P,Q)\tkzGetPoint{h}
\tkzInterLL(H,h)(A,B)\tkzGetPoint{K}
\tkzInterLL(H,K)(A,M)\tkzGetPoint{E}
\tkzDefLine[perpendicular=through A](A,M)\tkzGetPoint{a'}
\tkzInterLL(A,a')(B,C)\tkzGetPoint{F}
\tkzDrawSegments(A,B B,C D,I D,A A,I P,Q A,H H,K A,F B,F)
\tkzDrawPoints(A,B,C,D,M,P,Q,H,K,F,E)
\tkzLabelPoints[left](A,D)
\tkzLabelPoints[right](B,I,M,F)
\tkzLabelPoints[below](Q,H,C,E)
\tkzLabelPoints[above](P,K)
\tkzMarkRightAngles(F,A,M K,E,M)
\tkzMarkAngle[mkpos=.2,size=.4](D,A,H)
\tkzMarkAngle[mkpos=.2,size=.5](H,A,M)
\end{tikzpicture}
}
}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Đăk Lăk, 2016]%[9H3G8]
Gọi $I$ là tâm đường tròn nội tiếp tam giác $MNP$ ($MP<MN$), đường thẳng vuông góc với $MI$ tại $I$ cắt $NP$ kéo dài tại $Q$. Gọi $H$ là hình chiếu vuông góc của $I$ trên $MQ$.
\begin{enumerate}[a)]
\item Chứng minh $\widehat{PIQ}=\widehat{INP}$.
\item Chứng minh điểm $H$ nằm trên đường tròn ngoại tiếp tam giác $MNP$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\immini{
\item Chứng minh $\widehat{PIQ}=\widehat{INP}$\\
Kẻ $MK\perp NP\,(K\in NP)$.\\
Tứ giác $MIKQ$ có: $\widehat{MIQ}=\widehat{MKQ}=90^\circ$, nên tứ giác $MIKQ$ là tứ giác nội tiếp. \\
Suy ra $\widehat{IQK}=\widehat{IMK}=\widehat{IMP}-\widehat{KMP}\\
=\dfrac{\widehat{NMP}}{2}-\left(90^\circ-\widehat{MPN}\right)$.\\
}{
\begin{tikzpicture}[>=stealth,scale=1,font=\footnotesize]
\tkzDefPoints{0/0/P, 4/0/N, .5/2.5/M}
\tkzInCenter(M,N,P)\tkzGetPoint{I}
\tkzDefLine[perpendicular=through I](M,I)\tkzGetPoint{i}
\tkzInterLL(I,i)(N,P)\tkzGetPoint{Q}
\tkzDefPointBy[projection=onto M--Q](I)\tkzGetPoint{H}
\tkzDefPointBy[projection=onto N--P](M)\tkzGetPoint{K}
\tkzDrawSegments(M,N N,P M,P M,Q Q,P H,P H,I I,P I,K I,N I,M M,K Q,I)
\tkzDrawPoints(M,N,P,H,K,Q,I)
\tkzLabelPoints[left](H,Q)
\tkzLabelPoints[right](N,I,M)
\tkzLabelPoints[below](P,K)
\tkzMarkRightAngles(I,H,M M,K,N M,I,Q)
\end{tikzpicture}
}
Lại có $\dfrac{\widehat{MPN}}{2}=\widehat{IPN}=\widehat{PIQ}+\widehat{IQK}$ (góc ngoài $\triangle IPQ$)\\
Suy ra $\widehat{PIQ}=\dfrac{\widehat{MPN}}{2}-\widehat{IQK}=\dfrac{\widehat{MPN}}{2}-\left[\dfrac{\widehat{NMP}}{2}-\left(90^\circ-\widehat{MPN}\right) \right]\\
=90^\circ-\dfrac{\widehat{MPN}+\widehat{NMP}}{2}=\dfrac{\widehat{MNP}}{2}=\widehat{INP}$ (đpcm)
\item Chứng minh $H$ nằm trên đường tròn ngọa tiếp tam giác $MNP$.\\
Xét $\triangle PIQ$ và $\triangle INQ$, ta có $\widehat{PIQ}=\widehat{INQ}, \widehat{Q}$ chung.\\
Vậy $\triangle PIQ\backsim \triangle INP\Rightarrow \dfrac{QI}{QN}=\dfrac{QP}{QI}\Rightarrow QI^2=QP.QN$.\\
Xét $\triangle MIQ:\widehat{MIQ}=90^\circ, IH\perp MQ\Rightarrow QI^2=QH.QM$.\\
Do đó $QP.QN=QH.QM\Rightarrow \dfrac{QP}{QH}=\dfrac{QM}{QN}$, nên $\triangle QPH\backsim \triangle QMN$ (c-g-c)/\\
Suy ra $\widehat{QPH}=\widehat{QMN}$\\
Vậy tứ giác $MNPH$ là tứ giác nội tiếp, nên $H$ nằm trên đường tròn ngoại tiếp tam giác $MNP$. 
\end{enumerate}
}
\end{ex}
