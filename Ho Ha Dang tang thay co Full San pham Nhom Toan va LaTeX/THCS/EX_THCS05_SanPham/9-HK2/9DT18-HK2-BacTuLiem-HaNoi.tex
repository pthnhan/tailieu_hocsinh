\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Bắc Từ Liêm, Hà Nội]%[9D1Y8]%[9D1B8]
	Cho hai biểu thức $A=\dfrac{4\sqrt{x}}{x-1}$ và $B=\dfrac{1}{\sqrt{x}+1} +\dfrac{\sqrt{x}}{\sqrt{x}-1}+\dfrac{2}{x-1}$ với $x \geq 0; x\neq 1.$\\
	\begin{enumerate}
		\item Tính giá trị của biểu thức $A$ khi $x=4$.
		\item Rút gọn biểu thức $B$.
		\item Tìm các giá trị của $x$ để $A=\dfrac{3}{2}.$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item  Với $x=4$ thay vào biểu thức $A$ ta có $A= \dfrac{4\sqrt{4}}{4-1}=\dfrac{8}{3}.$\\
			Vậy $A= \dfrac{8}{3}$ khi $x=4.$
			\item Xét  $B=\dfrac{1}{\sqrt{x}+1} +\dfrac{\sqrt{x}}{\sqrt{x}-1}+\dfrac{2}{x-1}$ với $x \geq 0; x\neq 1.$\\
			Ta có 
			\begin{eqnarray*}
				B &= & \dfrac{\sqrt{x}-1}{(\sqrt{x}-1)(\sqrt{x}+1)}+ \dfrac{\sqrt{x}(\sqrt{x}+1)}{(\sqrt{x}-1)(\sqrt{x}+1)}+ \dfrac{2}{(\sqrt{x}-1)(\sqrt{x}+1)}\\
				&= & \dfrac{\sqrt{x}-1+x+\sqrt{x}+2}{(\sqrt{x}-1)(\sqrt{x}+1)}\\
				&=& \dfrac{x+2\sqrt{x}+1}{(\sqrt{x}-1)(\sqrt{x}+1)}\\
				&=& \dfrac{(\sqrt{x}+1)^2}{(\sqrt{x}-1)(\sqrt{x}+1)}=\dfrac{\sqrt{x}+1}{\sqrt{x}-1}.
			\end{eqnarray*}
		Vậy $B=\dfrac{\sqrt{x}+1}{\sqrt{x}-1}$ với $x \geq 0; x \neq 1.$
		\item Với điều kiện $x\geq 0,\,\, x\neq 1$, ta có
		\begin{eqnarray*}
	 A=\dfrac{3}{2}	&\Leftrightarrow & 3(x-1)=8\sqrt{x}\\
		&\Leftrightarrow & 3x-8 \sqrt{x}-3 =0\\
		& \Leftrightarrow & (3\sqrt{x}+1)(\sqrt{x}-3)=0\\
		&\Leftrightarrow & \hoac {& 3\sqrt{x}+1=0 \,\, \text{(vô nghiệm )}\\& \sqrt{x}-3=0 \Leftrightarrow x=9\,\,\text{(thỏa mãn)}. } 
		\end{eqnarray*}
	Vậy với $x=9$ thì $A=\dfrac{3}{2}.$
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Bắc Từ Liêm, Hà Nội]%[9D4K8]
	{\it Giải bài toán bằng cách lập phương trình hoặc hệ phương trình }\\
	Một tổ công nhân dự định làm xong $240$ sản phẩm trong một thời gian nhất định. Nhưng thực tế khi thực hiện, nhờ cải tiến kĩ thuật nên mỗi ngày tổ làm tăng thêm $10$ sản phẩm so với dự định. Do đó, tổ đã hoàn thành công việc sớm hơn dự định $2$ ngày. Hỏi theo dự định mỗi ngày tổ làm được bao nhiêu sản phẩm.
	 
	\loigiai{
	Gọi số sản phẩm tổ công nhân dự định làm trong một ngày là $x$ ($ x <240, \,\, x \in \mathbb{Z}^*$).\\
	Do tổ công nhân dự định làm xong $240$ sản phẩm nên số ngày tổ công nhân dự định phải làm là $\dfrac{240}{x}$ (ngày).\\
	Khi thực hiện mỗi ngày họ làm thêm được $10$ sản phẩm nên số sản phẩm làm được là $x+10$ (sản phẩm).\\
	Khi đó, số ngày mà tổ công nhân đã làm là $\dfrac{240}{x+10}$ (ngày).\\
	Theo đề bài, do cải tiến kĩ thuật, đội công nhân đó hoàn thành sớm hơn dự định $2$ ngày nên ta có phương trình $$\dfrac{240}{x} -\dfrac{240}{x+10}=2$$
	\begin{eqnarray*}
	&\Leftrightarrow &120(x+10)-120x=x(x+10)\\
	&\Leftrightarrow & x^2+10x-1200=0\\
	&\Leftrightarrow & \hoac {&x= -40 \quad \text{(không thỏa mãn)}\\& x=30 \quad \text{(thỏa mãn).}}
	\end{eqnarray*}
Vậy mỗi ngày tổ công nhân đã dự định làm $30$ sản phẩm.
	}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Bắc Từ Liêm, Hà Nội]%[9D4B4]%[9D4K6]
	 Cho phương trình $x^2-mx+m-1=0\quad (1)$
	\begin{enumerate}
		\item  Chứng tỏ rằng phương trình có hai nghiệm với mọi giá trị của $m$.
		\item  Tìm tham số $m$ để hai nghiệm $x_1, x_2$ của phương trình $(1)$ thỏa mãn $x_1+x_2-3\sqrt{x_1x_2}=1$.
	\end{enumerate}
	\loigiai
	{ \begin{enumerate}
			\item Ta có $\Delta =m^2-4(m-1)=m^2-4m+4 =(m-2)^2 \geq 0 \quad \forall m \in \mathbb{R}.$ \\
			Vậy phương trình $(1)$ có $2$ nghiệm với mọi $m$.
			\item Để phương trình có hai nghiệm sao cho $x_1 \cdot x_2 \geq 0$ thì 
			$$\heva {&\Delta \geq 0\\& \dfrac{c}{a} \geq 0} \Leftrightarrow \heva {&(m-2)^2 \geq 0 \\& m-1 \geq 0 } \Leftrightarrow m\geq 1.$$ 
			Với $m\geq 1$, áp dụng định lý Vi-ét, ta có $\heva {& x_1+x_2=m\\& x_1x_2 =m-1}$. \\ Ta có 
			\begin{eqnarray*}
		&&	x_1+x_2-3\sqrt{x_1x_2}=1\\	&\Leftrightarrow & m -3 \sqrt{m-1} =1\\
				&\Leftrightarrow & 3\sqrt{m-1} =m-1\\
				&\Leftrightarrow & \hoac {&\sqrt{m-1}=0\\& \sqrt{m-1}=3}\\
				&\Leftrightarrow &\hoac {&m=1\\&m=10} (\text{thỏa mãn}).
			\end{eqnarray*}
			
		\end{enumerate}
		
	}

\end{ex}

\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Bắc Từ Liêm, Hà Nội]%[9H3B4]%[9H3G7]
	Cho $A$ là một điểm thuộc đường tròn $(O;R)$. Kẻ tiếp tuyến $Ax$ của đường tròn $(O)$. Lấy điểm $B$ thuộc tia $Ax$ sao cho $AB <2R$. Gọi $M$ là trung điểm của đoạn thẳng $AB$, đường thẳng vuông góc với $AB$ tại $M$ cắt đường tròn $(O)$ tại $H$ và $K$ ($H$ nằm giữa $M$ và $K$).
	\begin{enumerate}
		\item Chứng minh $\widehat{MKA} =\widehat{MAH}$. Từ đó chứng minh $\triangle MKA$ và $\triangle MAH$ đồng dạng.
		\item Kẻ $HI \perp AK $ tại $I$. Chứng minh tứ giác $AMHI$ nội tiếp một đường tròn.
		\item Kéo dài $AH$ cắt $BK$ tại $D$. Chứng minh $AD \perp KB$.
		\item Lấy $C$ đối xứng với $B$ qua $AK$. Chứng minh điểm $C$ thuộc đường tròn $(O;R)$.
	\end{enumerate}
\loigiai {
	
	\begin{center}
		\begin{tikzpicture}
		\tkzInit[ymin=-5,ymax=5,xmin=-6,xmax=7]
		\tkzClip
		\tkzDefPoints{-2/4/B,-2/0/A,5/0/E, 6/2/N}
		\tkzDefMidPoint(A,E)\tkzGetPoint{O}
		\tkzDrawCircle(O,A)
		\tkzDefMidPoint(A,B)\tkzGetPoint{M}
		\tkzInterLC(M,N)(O,A)
		\tkzGetPoints{H}{K}
		
		\tkzInterLL(A,H)(B,K) \tkzGetPoint{D}
		\tkzDefPointBy[reflection = over
		A--K](B) \tkzGetPoint{C}
		\tkzInterLL(B,C)(A,K) \tkzGetPoint{I}
		\tkzLabelPoints[below](C)
			\tkzLabelPoints[left](M,A,B)
			\tkzLabelPoints[above](K,O,D)
			\tkzLabelPoints[above right](I)
			\tkzLabelPoints[above left](H)
		\tkzDrawPoints[fill=black](A,B,H,K,O,M,D,I,C)
		\tkzDrawSegments(A,B M,N K,A A,D B,K B,C A,O)
		\tkzMarkRightAngles(H,I,A H,M,A)
		\tkzMarkSegments[mark=|](B,M A,M)
		\end{tikzpicture}
	\end{center}
\begin{enumerate}
	\item Xét $(O;R)$ ta có $\widehat{MKA} =\widehat{MAH}$ (tính chất góc tạo bởi tiếp tuyến và dây cung).\\
	Xét  $\triangle MKA$ và $\triangle MAH$ có $\heva {& \widehat{AMK} \, \text{chung}\\& \widehat{MKA} =\widehat{MAH}} \Rightarrow \triangle MKA \backsim  \triangle MAH$ (g.g) 
	\item Xét tứ giác $AMHI$ có $\widehat{AMH}+ \widehat{AIH} =180^\circ$ mà hai góc  này ở vị trí đối diện nhau nên tứ giác $AMHI$ nội tiếp.
	\item  Ta có $\triangle KBM = \triangle KMA$ (c-g-c) nên $\widehat{MKA} =\widehat{MKB}$ mà $\widehat{MKA } =\widehat{MAH}$ nên $\widehat{MAH} = \widehat{MKB} =\widehat{MKD}$. Do đó tứ giác $MAKD$ nội tiếp. Từ đó suy ra $\widehat{AMK} =\widehat{ADK} =90^\circ \Rightarrow AD \perp BK.$
	\item Dễ thấy $H$ là trực tâm $\triangle ABK$ nên $BH  \perp KA$, lại có $IH \perp KA$ nên $B, H, I$ thẳng hàng.\\
	Tứ giác $BMHD$ nội tiếp nên $\widehat{ABK}+\widehat{MHD} =180^\circ$.\\
	Mà $\widehat{ABK} =\widehat{ACK}$ (do $C$ đối xứng với $B$ qua $AK$), $\widehat{MHD} =\widehat{AHK}$ (đối đỉnh)\\
	Suy ra $\widehat{ACK} +\widehat{AHK} =180^\circ$, do đó tứ giác $AHCK$ nội tiếp.\\
	Lại có $A,H,K$ cùng thuộc đường tròn $(O;R)$ nên $C$ thuộc $(O;R)$.
	
\end{enumerate}
}

\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Bắc Từ Liêm, Hà Nội]%[9D4G7]
	Giải phương trình $\sqrt{x} +\sqrt{x+7}+2 \sqrt{x^2+7x} +2x =35.$
	\loigiai {Điều kiện xác định $x \geq 0.$\\
		Đặt $t= \sqrt{x}+\sqrt{x+7} $ (điều kiện $t \geq \sqrt{7}$). Ta có $t^2=2x+7 +2\sqrt{x^2+7x} \Leftrightarrow 2x+2\sqrt{x^2+7x} =t^2-7$.\\
		Phương trình đã cho trở thành $t^2+t-42=0 \Leftrightarrow \hoac {&t=-7\\&t=6} \Leftrightarrow t=6.$\\
		Với $t=6$, ta có
		\begin{eqnarray*}
		 2x+7+2\sqrt{x^2+7x} =36 &\Leftrightarrow & 2\sqrt{x^2+7x}= 29-2x\\
		 &\Leftrightarrow & \heva {&29-2x \geq 0\\& 4(x^2+7x)=(29-2x)^2}\\
		 &\Leftrightarrow & \heva {&x \leq \dfrac{29}{2}\\& 4x^2+28x=841-116x+4x^2}\\
		 & \Leftrightarrow & \heva {&x \leq \dfrac{29}{2}\\& x= \dfrac{841}{144}}
		 \Leftrightarrow x=\dfrac{841}{144} \quad \text{(thỏa mãn)}.
		\end{eqnarray*}
	Vậy phương trình có nghiệm duy nhất $x=\dfrac{841}{144}$.

}
\end{ex}
