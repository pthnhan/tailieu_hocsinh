\begin{ex}%[HK2, Toán 8, Ba Đình - Hà Nội, năm 2018]%[8D3Y3]%[Đỗ Viết Lân, ex-Hanoi-4]
    Giải phương trình $4x - (3 - 5x) = 2x + 11$.
	\loigiai{
    	Ta có
    	\begin{center}
    	$ \begin{array}{lrcl}
		&4x - (3 - 5x) &=& 2x + 11\\
		\Leftrightarrow & 4x - 3 + 5x &=& 2x + 11\\
		\Leftrightarrow & 4x + 5x - 2x &=& 11 + 3\\
		\Leftrightarrow & 7x & =& 14\\
		\Leftrightarrow & x &=& \dfrac{14}{7}\\
		\Leftrightarrow & x &=& 2
		\end{array} $
    	\end{center}
    	Vậy nghiệm của phương trình là $x = 2$.
    }
\end{ex}

\begin{ex}%[HK2, Toán 8, Ba Đình - Hà Nội, năm 2018]%[8D3K5]%[Đỗ Viết Lân, ex-Hanoi-4]
	Giải phương trình $\dfrac{x+2}{x-2} + \dfrac{1}{x} = -\dfrac{8}{2x - x^2}$.
	\loigiai{
		Điều kiện $\heva{& x - 2 &\neq 0\\&x &\neq 0\\&2x - x^2 &\neq 0} \Leftrightarrow \heva{&x\neq 0\\& x \neq 2}$.\\
		Quy đồng và bỏ mẫu với mẫu chung $2x - x^2 = x(2-x)$ ta có 
		\begin{center}
		$\begin{array}{lrcl}
			& \dfrac{x+2}{x-2} + \dfrac{1}{x} &=& -\dfrac{8}{2x - x^2}\\
			\Leftrightarrow &  \dfrac{-x(x+2)}{x(2-x)} + \dfrac{2-x}{x(2-x)} & =& -\dfrac{8}{x(2-x)}\\
			\Leftrightarrow & -x(x+2) + 2 - x  &=& -8 \\
			\Leftrightarrow & -x^2 - 2x + 2 - x + 8 &=&0\\
			\Leftrightarrow & -x^2 - 3x + 10 &=& 0\\
			\Leftrightarrow & -x^2 - 5x + 2x + 10 &=& 0\\
			\Leftrightarrow & -x(x+5) + 2(x+5) &=& 0\\
			\Leftrightarrow & (x+5)(2-x) &=& 0\\
			\Leftrightarrow & x+5 =0 &\text{hoặc}& 2-x = 0\\
			\Leftrightarrow & x=-5 &\text{hoặc}& x=2 \text{ (loại)}.
		\end{array}$
		\end{center}
		Vậy phương trình có nghiệm $x = -5$.
	}
\end{ex}

\begin{ex}%[HK2, Toán 8, Ba Đình - Hà Nội, năm 2018]%[8D4K5]%[Đỗ Viết Lân, ex-Hanoi-4]
	Giải phương trình $|x-7| - 3x = 5$.
	\loigiai{
		\begin{itemize}
			\item Nếu $x \geq 7$ thì $|x-7| = x - 7$. Do đó phương trình đã cho trở thành
					\begin{center}
					$\begin{array}{lrcl}
						& x-7 - 3x &=& 5\\
						\Leftrightarrow& -2x &=& 5+7\\
						\Leftrightarrow& -2x &=& 12\\
						\Leftrightarrow& x &=& -2 \text{ (loại)}.
					\end{array}$
					\end{center}
			\item Nếu $x < 7$ thì $|x-7| = 7-x$. Do đó phương trình đã cho trở thành
					\begin{center}
					$\begin{array}{lrcl}
						& 7 - x - 3x &=& 5\\
						\Leftrightarrow& -4x &=& 5 - 7\\
						\Leftrightarrow& -4x &=& -2\\
						\Leftrightarrow& x &=& \dfrac{1}{2} \text{ (nhận)}.
					\end{array}$
					\end{center}
		\end{itemize}
		Vậy nghiệm phương trình là $x = \dfrac{1}{2}$.
	}
\end{ex}

	Giải bất phương trình sau và biểu diễn nghiệm trên trục số: $$\dfrac{3x + 1}{6} - \dfrac{x-3}{2} \leq \dfrac{2x-1}{3} + 5.$$
	\loigiai{
		Quy đồng bỏ mẫu với mẫu chung là $6$. Ta có
		\begin{center}
			$\begin{array}{lrcl}
				&\dfrac{3x + 1}{6} - \dfrac{x-3}{2} &\leq& \dfrac{2x-1}{3} + 5\\
				\Leftrightarrow& \dfrac{3x + 1}{6} - \dfrac{3x-9}{6}  & \leq & \dfrac{4x-2}{6} + \dfrac{30}{6}\\
				\Leftrightarrow& 3x + 1 + 9 - 3x & \leq & 4x-2 + 30\\
				\Leftrightarrow & 3x - 3x -4x & \leq  & -2 + 30 -1 -9\\
				\Leftrightarrow & -4x & \leq  & 18\\
				\Leftrightarrow & x & \geq  & -\dfrac{9}{2}.
			 \end{array}$
		\end{center}
		Biểu diễn trên trục số\\
		\begin{center}
		\begin{tikzpicture}
		\draw[->](0,0)->(6,0);%Vẽ trục số
		\IntervalGLF{}{}{\big[}{-\frac{9}{2}}
		\end{tikzpicture}
		\end{center}
	}

\begin{ex}%[HK2, Toán 8, Ba Đình - Hà Nội, năm 2018]%[8D3K6]%[Đỗ Viết Lân, ex-Hanoi-4]
	Giải bài toán bằng cách lập phương trình.\\
	Một người đi xe máy dự định đi từ A đến B trong thời gian nhất định. Sau khi đi được nửa đường với vận tốc $30$ km/h thì người đó đi tiếp nửa quãng đường còn lại với vận tốc $36$ km/h, do đó đến B sớm hơn dự định $10$ phút. Tính quãng đường AB và thời gian dự định đi hết quãng đường AB?
	\loigiai{
		Gọi $x$ (km) là độ dài quãng đường AB. (Điều kiện $x > 0$)\\
		Thời gian người đó đi nửa quãng đường đầu là $\dfrac{\dfrac{x}{2}}{30} = \dfrac{x}{60}$ (h).\\
		Thời gian người đó đi nửa quãng đường sau là $\dfrac{\dfrac{x}{2}}{36} = \dfrac{x}{72}$ (h).\\
		Thời gian dự định đi quãng đường AB là $\dfrac{x}{30}$ (h).\\
		Đến sớm hơn dự định $10$ phút = $\dfrac{1}{6}$ giờ nên ta có phương trình
		\begin{center}
		$ \begin{array}{lrcl}
			&\dfrac{x}{60} + \dfrac{x}{72} + \dfrac{1}{6} &=& \dfrac{x}{30}\\
			\Leftrightarrow &\dfrac{x}{60} + \dfrac{x}{72} - \dfrac{x}{30} &=& - \dfrac{1}{6}\\
			\Leftrightarrow & - \dfrac{x}{360} &=& - \dfrac{1}{6}\\
			\Leftrightarrow & x &=& 60 \text{ (thỏa mãn điều kiện)}.
		  \end{array}$
		\end{center}
		Vậy quãng đường cần tìm AB dài $60$ km, và thời gian đi dự định là $\dfrac{60}{30} = 2$ giờ.
	}
\end{ex}

\begin{ex}%[HK2, Toán 8, Ba Đình - Hà Nội, năm 2018]%[8H3Y8]%[8H3B8]%[8H3G8]%[8H3K3]%[Đỗ Viết Lân, ex-Hanoi-4]
    Cho tam giác $ABC$ vuông tại $A$ có $AB > AC$. Lấy $M$ là một điểm tùy ý trên cạnh $BC$. Qua $M$ kẻ đường thẳng vuông góc với $BC$ và cắt đoạn thẳng $AB$ tại điểm $I$, cắt đường thẳng $AC$ tại điểm $D$.
    \begin{enumerate}
    	\item Chứng minh $\triangle ABC$ đồng dạng với $\triangle MDC$.
    	\item Chứng minh rằng $BI \cdot BA = BM \cdot BC$.
    	\item Chứng minh $\widehat{BAM} = \widehat{ICB}$. Từ đó chứng minh $AB$ là phân giác của $\widehat{MAK}$ với $K$ là giao điểm của $CI$ và $BD$.
    	\item Cho $AB = 8$ cm, $AC = 6$ cm. Khi $AM$ là đường phân giác trong tam giác $ABC$, hãy tính diện tích tứ giác $AMBD$.
    \end{enumerate}
\loigiai
    {
    \begin{center}
    \begin{tikzpicture}
	\tkzDefPoints{0/2.82/A,-4/0/B,2/0/C,-0.5/0/M,-0.5/1/H}
	\tkzInterLL(M,H)(A,B)\tkzGetPoint{I}
	\tkzInterLL(M,H)(A,C)\tkzGetPoint{D}
	\tkzInterLL(C,I)(B,D)\tkzGetPoint{K}
	\tkzMarkRightAngle(B,A,C)
	\tkzMarkRightAngle(C,M,D)
	\tkzMarkRightAngle(B,K,C)
	\tkzMarkAngles[size=0.4cm](K,A,B)
	\tkzMarkAngles[size=0.5cm](B,A,M)
	\tkzMarkAngles[size=0.4cm](M,A,C)
	\tkzMarkAngles[size=0.4cm](I,C,B)
	\tkzDrawSegments(A,B B,C C,A M,I A,I A,M C,I A,K B,K D,A D,I K,I D,K)
	\tkzDrawPoints[fill=black,size=1pt](A,B,C,M,I,D,K)
	\tkzLabelPoints[below](B,C,M)
	\tkzLabelPoints[above](D)
	\tkzLabelPoints[below left](I)
	\tkzLabelPoints[right](A)
	\tkzLabelPoints[left](K)
	\end{tikzpicture}
    \end{center}
    \begin{enumerate}
        \item Xét hai tam giác $ABC$ và $MDC$ lần lượt vuông tại $A$ và $M$, có $\widehat{ACB}$ là góc chung\\
        $\Rightarrow \triangle ABC \backsim \triangle MDC$ (g.g).
        \item Xét hai tam giác $ABC$ và $MBI$ lần lượt vuông tại $A$ và $M$, có $\widehat{ABC}$ là góc chung\\
        $\Rightarrow \triangle ABC \backsim \triangle MBI$ (g.g) $\Rightarrow \dfrac{BA}{BM} = \dfrac{BC}{BI} \Rightarrow BI \cdot BA = BM \cdot BC$.
        \item Xét $\triangle BAM $ và $\triangle BCI$ có
        \begin{itemize}
        \item $\widehat{ABC}$ chung;
        \item $\dfrac{BA}{BC} = \dfrac{BM}{BI}$ (do câu b).
        \end{itemize}
        Suy ra $\triangle BAM \backsim \triangle BCI$ (c.g.c) $\Rightarrow \widehat{BAM} = \widehat{ICB}.\hfill (1)$\\
        Mặt khác ta có $BA$, $DM$ là hai đường cao của $\triangle DBC$ nên $CK$ là đường cao thứ ba.\\
        Xét hai tam giác $IKB$ và $IAC$ lần lượt vuông tại $K$ và $A$ có, $\widehat{BIK}=\widehat{CIA}$ (đối đỉnh)\\
        $\Rightarrow \triangle IKB \backsim \triangle IAC$ (g.g) $\Rightarrow \dfrac{IK}{IA} = \dfrac{IB}{IC} \Rightarrow \dfrac{IK}{IB} = \dfrac{IA}{IC}$.\\
        Từ đây suy ra $\triangle IKA \backsim \triangle IBC$ (c-g-c) $\Rightarrow \widehat{IAK} = \widehat{ICB}\hfill (2)$\\
        Từ $(1)$ và $(2)$ ta có $\widehat{IAK} = \widehat{BAM}$. Do đó $AB$ là phân giác của góc $\widehat{MAK}$.
        \item Áp dụng định lý Pytago cho $\triangle ABC$ ta có $BC^2 = AB^2 +AC^2 = 6^2 + 8^2 = 100 \Rightarrow BC = 10$.\\
        Theo tính chất đường phân giác ta có $\dfrac{BM}{AB} = \dfrac{CM}{AC}$\\
        $\Rightarrow \dfrac{BM}{8} = \dfrac{CM}{6} = \dfrac{BM + CM}{8+6} = \dfrac{10}{14} = \dfrac{5}{7}$\\
        $\Rightarrow BM =  8 \cdot \dfrac{5}{7} = \dfrac{40}{7}$ và $CM = 6 \cdot \dfrac{5}{7} = \dfrac{30}{7}$.\\
        Kẻ $AH \perp BC$ ta có $AH \cdot BC = AB \cdot AC$ (vì cùng bằng hai lần $S_{ABC}$)\\
        $\Rightarrow AH = \dfrac{AB \cdot AC}{BC} = \dfrac{48}{10} = \dfrac{24}{5}$.\\
        và $HC^2 = AC^2 - AH^2 = 6^2 - \left(\dfrac{24}{5}\right)^2 = \dfrac{324}{25} \Rightarrow HC = \dfrac{18}{5}$.\\
        Ta có $AH \perp BC$ và $DM \perp BC$ nên $DM \parallel AH$.\\
        Áp dụng hệ quả định lý Ta-lét trong $\triangle CMD$ với $AH \parallel DM$ ta có\\
        $\dfrac{AH}{DM} = \dfrac{CH}{CM} \Rightarrow DM = \dfrac{AH \cdot CM}{CH} = \dfrac{40}{7}$.\\
        Ta có $S_{AMBD} = S_{DBC} - S_{AMC} = \dfrac{1}{2}\cdot DM \cdot BC - \dfrac{1}{2}\cdot AH \cdot MC = \dfrac{1}{2} \cdot \dfrac{40}{7} \cdot 10 - \dfrac{1}{2} \cdot \dfrac{24}{5} \cdot \dfrac{30}{7} = \dfrac{128}{7}$.
    \end{enumerate}
    }
\end{ex}
