\begin{ex}%[Đề thi Toán 9 Khảo sát chất lượng, THCS Gia Khánh, Bình Xuyên]%[9D3B4]%[9D4B7]
   Giải hệ phương trình và phương trình sau:
   \begin{enumerate}
    	\item $\begin{cases} 3x + 2y = 11 \\ x - 2y = 1. \end{cases}$
   		\item $4x^4 + 9x^2 - 9 = 0$.
   \end{enumerate} 
	\loigiai{
		\begin{enumerate}
			\item Ta có $\begin{cases} 3x + 2y = 11 \\ x - 2y = 1 \end{cases} \Leftrightarrow \begin{cases} 4x = 12\\ x - 2y = 1 \end{cases}\Leftrightarrow \begin{cases}x = 3\\ y = 1.\end{cases}$\\
					Vậy hệ phương trình có nghiệm duy nhất $\left(x; y\right) = \left(3; 1\right)$.
			\item Đặt $t = x^2$, điều kiện $t \ge 0$, Ta được phương trình:
					$$4t^2 + 9t - 9 = 0 \Leftrightarrow \left[\begin{aligned} &t = \dfrac{3}{4} &&\text{(thỏa mãn điều kiện)}\\ & t = -3 &&\text{(không thỏa mãn điều kiện).} \end{aligned}\right.$$
					Với $t = \dfrac{3}{4}$, ta có: $x^2 = \dfrac{3}{4} \Leftrightarrow x = \pm\dfrac{\sqrt{3}}{2}\cdot$\\
					Vậy phương trình có hai nghiệm: $x = \pm\dfrac{\sqrt{3}}{2}\cdot$
		\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Khảo sát chất lượng, THCS Gia Khánh, Bình Xuyên]%[9D5G5]
	Cho biểu thức: $P = \left(\dfrac{a\sqrt{a} - 1}{a - \sqrt{a}} - \dfrac{a\sqrt{a} + 1}{a + \sqrt{a}}\right) : \dfrac{a + 2}{a - 2}$ với $a > 0$, $ a \ne 1$, $a \ne 2$.
	\begin{enumerate}
		\item Rút gọn $P$.
		\item Tính giá trị nguyên của $a$ để $P$ có giá trị nguyên.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
	     	\item
	     		$P=\left(\dfrac{(\sqrt{a}-1)(a+\sqrt{a}+1)}{\sqrt{a}(\sqrt{a}-1)}-\dfrac{(\sqrt{a}+1)(a-\sqrt{a}+1)}{\sqrt{a}(\sqrt{a}+1)}\right):\dfrac{a+2}{a-2} = 2:\dfrac{a+2}{a-2}=\dfrac{2(a-2)}{a+2}$
	     \item
	     $P=\dfrac{2a+4-8}{a+2}=2-\dfrac{8}{a+2}$\\
	     $P$ nguyên  $\Leftrightarrow8 \vdots  (a+2)\Leftrightarrow \left[\begin{aligned}a+2 & =4\\a+2 & =8. \end{aligned}\right.$ (Vì $ a>0$) $\Leftrightarrow\left[\begin{aligned}a & =2\\a & =6. \end{aligned}\right.$
	    \end{enumerate}
	    }
\end{ex}

\begin{ex}%[Đề thi Toán 9 Khảo sát chất lượng, THCS Gia Khánh, Bình Xuyên]%[9D5G5]%[9D5G1]
     \begin{enumerate}
     \item
     Giải hệ phương trình: $\begin{cases} x^{3}+1=2y (1)\\ y^{3}+1=2x (2). \end{cases}$
     \item
     Các số thực $x, a, b, c$ thay đổi, thỏa mãn hệ:\\
     $\begin{cases} x+a+b+c=7 (1)\\ x^{2}+a^{2}+b^{2}+c^{2}=13 (2) \end{cases}$\\
     Tìm $\max x, \min x$.
     \end{enumerate} 
      \loigiai{
           \begin{enumerate}
           \item
           Lấy $(1)-(2)$ ta được:\\
           $x^3-y^3=2y-2x \Leftrightarrow (x-y)(x^{2}+xy+y^{2}+2)=0 \Leftrightarrow \left[\begin{aligned} & x=y (3)\\ & x^{2}+xy+y^{2}+2=0 (4) \end{aligned}\right.$\\
           $(4)$ vô nghiệm vì $x^{2}+xy+y^{2}+2>0$ với mọi $x, y$\\
           Thay $(3)$ vào $(1)$ ta được: $x^{3}-2x+1=0\Leftrightarrow \left[\begin{aligned} & x=1\\ & x=\dfrac{1\pm\sqrt{5}}{2} \end{aligned}\right.$\\
           Vậy hệ có ba nghiệm: $\left(x; y\right)=\left(1; 1\right), \left(x; y\right)=\left(\dfrac{1+\sqrt{5}}{2}; \dfrac{1+\sqrt{5}}{2}\right); \left(x; y\right)=\left(\dfrac{1\sqrt{5}}{2}; \dfrac{1-\sqrt{5}}{2}\right).$
           \item
           $(1)\Rightarrow a+b+c=7-x$\\
           $\Rightarrow (7-x)^{2}=(a+b+c)^{2} \le 3(a^{2}+b^{2}+c^{2}) (3)$\\
           $(2)\Rightarrow a^{2}+b^{2}+c^{2}=13-x^{2}$\\
           Thay $(2)$ vào $(3)$ ta được:\\
           $(7-x)^{2}\le 3(13-x^{2}) \Leftrightarrow 4x^{2}-14x+10\le 0\Leftrightarrow 1\le x \le \dfrac{5}{2}$\\
           Vậy $\max x =\dfrac{5}{2}, \min x=1$.
           \end{enumerate} 
        } 
\end{ex}
