Thực hiện các phép tính sau
\begin{listEX}[3]
	\item $ x\left(5 - x\right)	+ \left(x - 3\right)\left(x + 3\right) $;
	\item $ \left(x - 3\right)^2	+ \left(x + 2\right)\left(1 - x\right) $;
	\item $ \dfrac{1}{x - 1} + \dfrac{1}{x + 1} $.
\end{listEX}
\loigiai{
\begin{listEX}[2]
	\item
	$ \begin{aligned}[t]
	& x\left(5 - x\right)	+ \left(x - 3\right)\left(x + 3\right)\\
	& = 5x - x^2 + x^2 - 9\\
	& = 5x - 9.
	\end{aligned} $	
	\item
	$ \begin{aligned}[t]
	& \left(x - 3\right)^2	+ \left(x + 2\right)\left(1 - x\right)\\
	& = x^2 - 6x + 9 + x + 2 - x^2 - 2x\\
	& = -7x + 11.
	\end{aligned} $
	\item
	$ \begin{aligned}[t]
	& \dfrac{1}{x - 1} + \dfrac{1}{x + 1}\\
	& = \dfrac{x + 1}{\left(x - 1\right)\left(x + 1\right)} + \dfrac{x - 1}{\left(x - 1\right)\left(x + 1\right)}\\
	& = \dfrac{x + 1 + x - 1}{\left(x - 1\right)\left(x + 1\right)}\\
	& = \dfrac{2x}{\left(x - 1\right)\left(x + 1\right)}.
	\end{aligned} $
\end{listEX}
}
\end{bt}

Phân tích đa thức thành nhân tử
\begin{listEX}[3]
	\item $ 2a^3b + 4a^2b^2 + 6ab^3 $;
	\item $ x^2 - 2xy + y^2 - 25 $;
	\item $ 4x^3 - 7x^2 - 2x $.
\end{listEX}	
\loigiai{
\begin{listEX}[3]
	\item $ \begin{aligned}[t]
	& 2a^3b + 4a^2b^2 + 6ab^3 \\
	& = 2ab\left(a^2 + 2ab + 3b^2\right)
	\end{aligned} $.
	\item $ \begin{aligned}[t]
		& x^2 - 2xy + y^2 - 25 \\
		& = \left(x^2 - 2xy + y^2\right) - 25 \\
		& = \left(x - y\right)^2 - \left(5\right)^2\\
		& = \left(x - y + 5\right)\left(x - y - 5\right).
	\end{aligned} $	
	\item $ \begin{aligned}[t]
	& 4x^3 - 7x^2 - 2x \\
	& = x\left(4x^2 - 7x - 2\right)\\
	& = x\left(4x^2 - 8x + x - 2\right)\\
	& = x\left[\left(4x^2 - 8x\right) + \left(x - 2\right)\right]\\
	& = x\left[4x\left(x - 2\right) + \left(x - 2\right)\right]\\
	& = x\left(x - 2\right)\left(4x + 1\right).
	\end{aligned} $
\end{listEX}
		
}
\end{bt}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Gò Vấp, thành phố Hồ Chí Minh]%[Lục Hán Tường, dự án 8DT18]%[8D1B1]%[8D1K9]
Tìm $ x $
\begin{enumerate}
	\item $ \left(x + 4\right)\left(x - 4\right) + x\left(6 - x\right) = 2 $.
	\item $ \left(x + 3\right)\left(x - 2\right) - \left(x^3 - 3x^2 + 3x - 1\right):\left(x - 1\right) = 2 $.
\end{enumerate}
\loigiai{
\begin{listEX}[1]
	\item $ \begin{aligned}[t]
	& \left(x + 4\right)\left(x - 4\right) + x\left(6 - x\right) = 2\\
	& \Leftrightarrow x^2 - 16 + 6x - x^2 = 2 \\
	& \Leftrightarrow 6x = 18\\
	& \Leftrightarrow x = 3.
	\end{aligned} $	
	\item $ \begin{aligned}[t]
	&\left(x + 3\right)\left(x - 2\right) - \left(x^3 - 3x^2 + 3x - 1\right):\left(x - 1\right) = 2\\
	& \Leftrightarrow \left(x + 3\right)\left(x - 2\right) - \left(x - 1\right)^3:\left(x - 1\right) = 2\\
	& \Leftrightarrow \left(x + 3\right)\left(x - 2\right) - \left(x - 1\right)^2 = 2\\
	& \Leftrightarrow x^2 + 3x - 2x - 6 - x^2 + 2x - 1 = 2 \\
	& \Leftrightarrow 3x = 9\\
	& \Leftrightarrow x = 3
	\end{aligned} $
\end{listEX}	
}
\end{bt}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Gò Vấp, thành phố Hồ Chí Minh]%[Lục Hán Tường, dự án 8DT18]%[8D1Y9]%[8D1G1]
Cho $ A = 8 - x^3 - 2x + x^2 $; $ B = 2 - x \quad (B \neq 0) $
\begin{enumerate}
	\item Tìm $ Q $, biết $ Q = A : B $.
	\item Chứng minh $ Q $ dương với mọi $ x $ là số thực.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có
	\begin{align*}
	A & = 8 - x^3 - 2x + x^2\\
	& = \left(2^3 - x^3\right) - \left(2x - x^2\right)\\
	& = \left(2 - x\right)\left(4 + 2x + x^2\right) - x\left(2 - x\right)\\
	& = \left(2 - x\right)\left(x^2 + x + 4\right).
	\end{align*}
	Suy ra
	\begin{align*}
	Q = A : B = x^2 + x + 4
	\end{align*}
	
	\item Ta có
	\begin{align*}
	Q  = x^2 + x + 4
	 = \left[x^2 + 2 \cdot x \cdot \dfrac{1}{2} + \left(\dfrac{1}{2}\right)^2\right] - \left(\dfrac{1}{2}\right)^2 + 4
	 = \left(x + \dfrac{1}{2}\right)^2 + \dfrac{15}{4}
	\end{align*}
	Do $ \left(x + \dfrac{1}{2}\right)^2 \geqslant 0 \ \forall x \in \mathbb{R} \Rightarrow Q \geqslant \dfrac{15}{4} > 0 \ \forall x \in \mathbb{R} $.
\end{enumerate}	
}
\end{bt}
