\section{ÔN TẬP CHƯƠNG I}
\subsection{Tóm tắt lý thuyết}
\begin{enumEX}[1)]{1}
\item Điểm, đường thẳng, tia, đoạn thẳng;
\item Ba điểm thẳng hàng, đường thẳng đi qua hai điểm;
\item Vẽ độ dài đoạn thẳng, khi nào thì $AM+MB=AB$?
\item Vẽ đoạn thẳng cho biết độ dài;
\item Trung điểm của đoạn thẳng.
\end{enumEX}
\subsection{Các dạng toán}
\begin{dang}{Nhận biết khái niệm điểm, đường thẳng, tia, đoạn thẳng, nằm cùng phía, nằm khác phía}
\begin{itemize}
\item Những điểm nằm trên đường thẳng thì thuộc đường thẳng ấy và đường thẳng ấy đi qua những điểm đó.
\item Để nhận biết tia cần để ý tới gốc và phần đường thẳng được chia bởi gốc.
\item Trong ba điểm thẳng hàng chỉ có một điểm nằm giữa hai điểm còn lại.
\end{itemize}
\end{dang}

\begin{dang}{Điểm nằm giữa hai điểm khác}
\begin{itemize}
\item Nếu $AM+MB=AB$ thì điểm $M$ nằm giữa hai điểm $A$ và $B$.
\item Nếu trên tia $Ox$ có hai điểm $M$ và $N$ mà $OM<ON$ thì điểm $M$ nằm giữa hai điểm $O$ và $M$.
\end{itemize}
\end{dang}
\begin{vd}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 6]%[6H1K2]
Ba điểm $A$, $B$, $C$ có thẳng hàng không? Nếu biết:
\begin{enumEX}{1}
\item $AB=3{,}1$ cm; $AC=2{,}9$ cm; $BC=6$ cm;
\item $AB=4$ cm; $AC=6$ cm; $BC=3$ cm.
\end{enumEX}
\loigiai{
\begin{enumEX}{1}
\item  Ta có $AB+AC=3{,}1+2{,}9=6$ cm, suy ra $AB+AC=BC$. Do đó ba điểm $A$, $B$, $C$ thẳng hàng.
\item Ta có $AB+AC\ne BC$ (vì $4+6\ne 3$) nên $A$ không nằm giữa $B$ và $C$.\\
Ta có $AB+BC\ne AC$ (vì $4+3\ne 6$) nên $B$ không nằm giữa $A$ và $C$.\\
Ta có $BC+AC\ne AB$ (vì $3+6\ne 4$) nên $C$ không nằm giữa $A$ và $B$.\\
Do đó ba điểm $A$, $B$, $C$ không thẳng hàng.
\end{enumEX}
}
\end{vd}

\begin{dang}{Tính độ dài đoạn thẳng}
\begin{itemize}
\item Nếu điểm $M$ nằm giữa hai điểm $A$ và $B$ thì $AM+MB=AB$.
\item Nếu điểm $M$ là trung điểm của đoạn thẳng $AB$ thì $MA=MB=\dfrac{AB}{2}$.
\end{itemize}
\end{dang}

Trên tia $Ox$ lấy hai điểm $A$, $B$ sao cho $OA=9$ cm; $OB=3$ cm. Gọi $M$ là trung điểm của đoạn thẳng $AB$.
\begin{enumEX}{1}
\item Trong ba điểm $O$, $A$, $B$ điểm nào nằm giữa hai điểm còn lại?
\item  Tính độ dài đoạn thẳng $MB$.
\end{enumEX}
\loigiai{
	\begin{center}

\tkzInit[ymin=-0.5,ymax=0.5,xmin=-1,xmax=11]

\item $A$, $B$ thuộc tia $Ox$ mà $OB<OA$ ($3<9$) nên $B$ nằm giữa $O$ và $A$.

$$AB+OB=OA\Rightarrow AB+3=9 \Rightarrow AB=6\;\text{cm}.$$
Mà $M$ là trung điểm của đoạn thẳng $AB$ nên $MB=\dfrac{1}{2}AB=3\;\text{cm}.$
\end{enumEX}

\end{vd}

\begin{enumEX}{1}
\item Tính độ dài đoạn $AB$.
\item Trên tia đối của tia $OA$ lấy điểm $M$ sao cho $OM=1$ cm. Xác định trung điểm của đoạn $MB$.
\item Lấy $N$ là trung điểm của đoạn thẳng $OA$. Chứng tỏ $O$ là trung điểm của đoạn thẳng $MN$.
\end{enumEX}
\loigiai{
	\begin{center}
		\begin{tikzpicture}
\tkzInit[ymin=-0.5,ymax=0.5,xmin=-3,xmax=7]
\tkzClip
\tkzDefPoints{0/0/O,2/0/A,5/0/B,6.5/0/x,-1/0/M,-2.5/0/y}
\tkzDefMidPoint(A,O)\tkzGetPoint{N}
\tkzDrawSegments(y,x)
\tkzDrawPoints[fill=black](A,B,O,M,N)
\tkzLabelPoints[below](A,B,O,M,N)
\tkzLabelPoints[below left](x)
\tkzLabelPoints[below right](y)
\end{tikzpicture}
	\end{center}
\begin{enumEX}{1}
\item  Vì điểm $A$ nằm giữa hai điểm $O$ và $B$ nên 
$$OA+AB=OB\Rightarrow 2+AB=5 \Rightarrow AB=3\;\text{cm}.$$
\item  Vì điểm $O$ nằm giữa hai điểm $A$ và $M$ nên 
$$MA=MO+OA=1+2=3\;\text{cm}.$$
Suy ra $MA=AB=3$ cm. Mặt khác, $A$ nằm giữa $M$ và $B$ nên $A$ là trung điểm của đoạn thẳng $MB$.
\item Vì $N$ là trung điểm của  $OA$ nên $ON=\dfrac{1}{2}OA=\dfrac{1}{2}\cdot 2=1$ cm. Suy ra $ON=OM=1$ cm. Mặt khác, $O$ nằm giữa $M$ và $N$ nên $O$ là trung điểm của đoạn thẳng $MN$.
\end{enumEX}
}
\end{vd}

\begin{dang}{Số đoạn thẳng, số đường thẳng}
\begin{itemize}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 6]%[6H1G6]
\item Với $n$ điểm phân biệt cho trước ($n\in\mathbb{N}$, $n\ge 2$), trong đó không có ba điểm nào thẳng hàng, cứ qua hai điểm vẽ một đường thẳng thì số đường thẳng vẽ được là $\dfrac{n(n-1)}{2}$.
\item Với $n$ điểm phân biệt cho trước ($n\in\mathbb{N}$, $n\ge 2$), cứ qua hai điểm vẽ một đoạn thẳng thì số đoạn thẳng vẽ được là $\dfrac{n(n-1)}{2}$.
\item Mỗi điểm trên một đường thẳng là gốc chung của hai tia đối nhau.
\end{itemize}
\end{dang}
\begin{vd}
Cho trước $50$ điểm phân biệt. Vẽ các đoạn thẳng đi qua các cặp điểm. Hỏi vẽ được bao nhiêu đoạn thẳng?
\loigiai{
Số đoạn thẳng vẽ được là: $\dfrac{50(50-1)}{2}=1225$ (đoạn thẳng).
}
\end{vd}

\begin{vd}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 6]%[6H1G1]
Cho trước $21$ điểm phân biệt, trong đó có $10$ điểm thẳng hàng, ngoài ra không có ba điểm nào thẳng hàng. Kẻ các đường thẳng đi qua từng cặp điểm. Hỏi vẽ được bao nhiêu đường thẳng?
\loigiai{
Gọi $10$ điểm thẳng hàng là $A_1;A_2;\ldots; A_{10}$ và $11$ điểm còn lại là $B_1;B_2;\ldots; B_{11}$.\\
Qua $10$ điểm thẳng hàng chỉ kẻ được $1$ đường thẳng.\\
Số đường thẳng kẻ được từ $11$ điểm $B_1;B_2;\ldots; B_{11}$ là $\dfrac{11(11-1)}{2}=55$ (đường thẳng).\\
Từ mỗi điểm $A_1;A_2;\ldots; A_{10}$ kẻ đến $11$ điểm $B_1;B_2;\ldots; B_{11}$ ta được $11$ đường thẳng. Số đường thẳng kẻ được là $$11\times 10=110\;\text{(đường thẳng)}.$$
Vậy tổng số đường thẳng kẻ được là $$1+55+110=166\;\text{(đường thẳng)}.$$
}
\end{vd}

Cho điểm $M$ nằm giữa hai điểm $A$ và $B$, biết $AM=3$ cm; $MB=4$ cm. Tên tia đối của tia $MA$ lấy điểm $N$ sao cho $MN=7$ cm. So sánh $AM$ và $BN$.

\begin{bt}%[Nguyễn Chính Tâm, dự án EX-C2-Lớp 6]%[6H1G6]
Cho $n$ điểm phân biệt ($n\ge 2$). Cứ qua hai điểm ta kẻ một đoạn thẳng.
\begin{enumEX}{1}
\item Nếu $n=100$ thì có bao nhiêu đoạn thẳng được tạo thành?
\item Nếu có $105$ đoạn thẳng tạo thành thì $n$ là bao nhiêu?
\end{enumEX}
\loigiai{
\begin{enumEX}{1}
\item Khi $n=100$ thì số đoạn thẳng được tạo thành là $\dfrac{100\cdot (100-1)}{2}=4950$ (đoạn thẳng).
\item Ta có $\dfrac{n(n-1)}{2}=105\Rightarrow n(n-1)=210\Rightarrow n(n-1)=15\cdot 14\Rightarrow n=15$.
\end{enumEX}
}
\end{bt}
