
\section{Phép chia phân số}
\subsection{Trọng tâm kiến thức}
\subsubsection{Số nghịch đảo}
\begin{dn}
Hai số gọi là nghịch đảo của nhau nếu tích của chúng bằng $ 1 $.

Với $ a,b \in \mathbb{Z};a \ne 0; b \ne 0$ thì số nghịch đảo của $ \dfrac{a}{b} $ là $ \dfrac{b}{a} $.
\end{dn}
\subsubsection{Phép chia phân số}
\begin{note}
Muốn chia một phân số hay một số nguyên cho một phân số, ta nhân số bị chia với số nghịch đảo của số chia.
$$\dfrac{a}{b} : \dfrac{c}{d}=\dfrac{a}{b} \cdot \dfrac{d}{c}=\dfrac{a\cdot d}{b\cdot c}.$$
\end{note}


\subsection{Các dạng bài tập và phương pháp giải}
\begin{dang}{Tìm số nghịch đảo của một số cho trước}
\textbf{\textit{Phương pháp giải}}$\colon$ Số nghịch đảo của $ \dfrac{a}{b} $ là $ \dfrac{b}{a} $;

Số nghịch đảo của $ a $ là $ \dfrac{1}{a} $ với $ (a \ne 0) $. 
\end{dang}
\begin{vd}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3Y12]
Tìm số nghịch đảo của các số sau$\colon$
$$\dfrac{3}{4};\dfrac{-5}{9};-4;\dfrac{m}{10} (m \in \mathbb{Z};m\ne 0).$$
\loigiai{
Số nghịch đảo của $\dfrac{3}{4};\dfrac{-5}{9};-4;\dfrac{m}{10}$ theo thứ tự là $ \dfrac{4}{3};\dfrac{-9}{5};\dfrac{-1}{4};\dfrac{10}{m} $.
}
\end{vd}
\begin{vd}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3B12]
Trong các phân số sau, những phân số nào là nghịch đảo của nhau?
$$\dfrac{7}{15};15;\dfrac{-2}{18};\dfrac{-15}{7};-9.$$
\loigiai{
Ta thấy $ \dfrac{-2}{18}\cdot (-9)=\dfrac{18}{18}=1. $

\noindent Vậy hai số $ \dfrac{-2}{18} $ và $ -9 $ là hai số nghịch đảo nhau. Trong các số còn lại, không còn hai số nào có tích bằng $ 1 $ nên không còn hai số nào là nghịch đảo nhau.
}
\end{vd}
\begin{vd}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3B12]
	Tìm một số nguyên sao cho số này bằng số nghịch đảo của nó.
	\loigiai{
Gọi số nguyên cần tìm là $ a $ thì số nghịch đảo của nó là $ \dfrac{1}{a} $ với $ a\ \ne 0 $.

\noindent Theo đề bài ta có $ a=\dfrac{1}{a} $ suy ra $ a^2=1  $ hay $ a=\pm 1 $.\\
 Vậy số nguyên cần tìm là $ a=1 $ hoặc $ a=-1 $.	}
\end{vd}
\begin{vd}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3B12]
Cho $ a $ là số nghịch đảo của $ 1 $; $ b $ là nghịch đảo của $ 2 $. Tính nghịch đảo của tổng $ a+b $.
\loigiai{Vì $ a $ là số nghịch đảo của $ 1 $ nên $ a=1 $.\\
Vì $ b $ là số nghịch đảo của $ 2 $ nên $ b=\dfrac{1}{2} $.\\ Ta có $ a+b=1+\dfrac{1}{2}=\dfrac{3}{2} $. Do đó nghịch đảo của tổng $ a+b $ là $ \dfrac{2}{3} $. }
\end{vd}
\begin{dang}{Chia phân số}
	\textbf{\textit{Phương pháp giải}}$\colon$ Áp dụng quy tắc $ \dfrac{a}{b} : \dfrac{c}{d}=\dfrac{a}{b} \cdot \dfrac{d}{c}=\dfrac{a\cdot d}{b \cdot c} $;
	
	Rút gọn kết quả đến phân số tối giản.
\end{dang} 
\setcounter{vd}{0}
\begin{vd}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3Y12]
	Tính$\colon$
	\begin{enumEX}{2}
		\item $ \dfrac{-13}{27} : \dfrac{26}{81} $;
		\item $\dfrac{11}{14} : \left(-\dfrac{55}{49}\right)$.  
	\end{enumEX}
\loigiai{a) $ \dfrac{-13}{27} : \dfrac{26}{81} =\dfrac{-13}{27} : \dfrac{26}{81}=\dfrac{-13}{27} \cdot \dfrac{81}{26}=\dfrac{-1}{1} \cdot \dfrac{3}{2}=\dfrac{-3}{2}$;\\
	\\
b) $ \dfrac{11}{14} : \left(-\dfrac{55}{49}\right) =\dfrac{11}{14} : \left(-\dfrac{49}{55}\right)=\dfrac{1}{2} \cdot \left(-\dfrac{7}{5}\right)=-\dfrac{7}{10}$.	
}	
\end{vd}
\begin{vd}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3Y12]
	Tính$\colon$
	\begin{enumEX}{2}
		\item  $ 10 : \dfrac{-25}{3} $;
		\item $ \dfrac{-4}{15}:8 $.
	\end{enumEX}
\loigiai{\begin{enumEX}{2}
		\item  $ 10 : \dfrac{-25}{3} =10 \cdot \dfrac{-3}{25}=\dfrac{-6}{5}$;
 \item $ \dfrac{-4}{15}:8 =\dfrac{-4}{15} \cdot \dfrac{1}{8}=-\dfrac{1}{30}$.
\end{enumEX}}	
\end{vd}
\begin{dang}{Tìm một thành phần chưa biết trong phép nhân, phép chia}
\textbf{\textit{Phương pháp giải}}$\colon$ \begin{enumerate}
	\item Muốn tìm một thừa số của tích ta lấy tích chia cho thừa số đã biết.
	\item Muốn tìm một số bị chia ta lấy thương nhân với số chia.
	\item Muốn tìm số chia ta lấy số bị chia chia cho thương.
\end{enumerate}
\end{dang} 
\setcounter{vd}{0}
\begin{vd}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3B12] 
	Tìm $ x $, biết:
	\begin{enumEX}{2} 
		\item $\dfrac{-12}{19}\cdot x=\dfrac{9}{76}$;
		\item $ x \cdot \dfrac{-15}{28}=\dfrac{-3}{20} $. 
	\end{enumEX}	
\loigiai{a) $\dfrac{-12}{19}\cdot x=\dfrac{9}{76}$ suy ra
$$\begin{aligned}
	x&=\dfrac{9}{76} :\dfrac{-12}{19}\\
	x&=\dfrac{9}{76} \cdot \dfrac{-19}{12}=\dfrac{-3}{16};
\end{aligned}$$
b) $x \cdot \dfrac{-15}{28}=\dfrac{-3}{20}$ suy ra
$$\begin{aligned}
x&=\dfrac{-3}{20} :\dfrac{-15}{28}\\
x&=\dfrac{-3}{20} \cdot \dfrac{-28}{15}=\dfrac{7}{25}.
\end{aligned}$$}	
\end{vd}
\begin{vd}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3B12]
Tìm $ x $, biết$\colon$ 
\begin{enumEX}{2}
	\item $\dfrac{7}{78}: x=\dfrac{35}{52}$;
	\item $ x : \dfrac{7}{39}=13 $. 
\end{enumEX}	
\loigiai{a) Ta có  $\dfrac{7}{78} : x=\dfrac{35}{52}$ suy ra
	$$\begin{aligned}
	x&=\dfrac{7}{78} :\dfrac{35}{52}\\
	x&=\dfrac{7}{78} \cdot \dfrac{52}{35}=\dfrac{7 \cdot 52}{35 \cdot 78}=\dfrac{1}{5} \cdot \dfrac{2}{3}=\dfrac{2}{15};
	\end{aligned}$$
	b) $x : \dfrac{7}{39}=13$ suy ra
	$$\begin{aligned}
	x&=\dfrac{7}{39} \cdot 13=\dfrac{7}{39} \cdot \dfrac{13}{1}=\dfrac{7}{1} \cdot \dfrac{13}{39}\\
	x&=\dfrac{7}{3}.
	\end{aligned}$$}	
\end{vd}
\begin{vd}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3B12]
Tìm $ x $, biết$\colon$
	\begin{enumEX}{2}
		\item $x \cdot \dfrac{3}{4}=\dfrac{7}{8} : \dfrac{21}{5}$;
		\item $ \left(1+\dfrac{3}{7}\right) \cdot x=1-\dfrac{3}{7}$.
	\end{enumEX}	
\loigiai{a) $x \cdot \dfrac{3}{4}=\dfrac{7}{8} : \dfrac{21}{5}$ suy ra
	$$\begin{aligned}
	x&=\dfrac{7}{8} \cdot \dfrac{5}{21} :\dfrac{3}{4}=\dfrac{7}{8} \cdot \dfrac{5}{21}\cdot \dfrac{4}{3}=\dfrac{1 \cdot 5 \cdot1}{2 \cdot 3 \cdot 3} \\
	x&=\dfrac{5}{18};
	\end{aligned}$$
b) $ \left(1+\dfrac{3}{7}\right) \cdot x=1-\dfrac{3}{7}$ suy ra
$$\begin{aligned}
\dfrac{10}{7} \cdot x&=\dfrac{4}{7} \\
x&=\dfrac{4}{7} : \dfrac{10}{7}=\dfrac{4}{7} \cdot \dfrac{7}{10}\\
x&=\dfrac{2}{5}.
\end{aligned}$$ }
\end{vd}
\begin{dang}{Các bài toán dẫn đến phép chia phân số}
\textbf{\textit{Phương pháp giải}}:	Căn cứ vào đề bài, lập phép chia phân số thích hợp.
\end{dang}

\begin{vd}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3B12]
	Một ô tô chạy từ $A$ đến $B$ với vận tốc $50$ km/h thì hết $ \dfrac{3}{5} $ giờ. Lúc từ $B$ về $A$ xe chạy hết $ \dfrac{3}{4} $ giờ. Tính vận tốc lúc về.
\loigiai{Quãng đường $ AB $ dài là: $ 50 \cdot \dfrac{3}{5}=30 $ (km).\\
	Vận tốc của ô tô lúc về là: $ 30 : \dfrac{3}{4}=40 $(km/h).
}
\end{vd}
\begin{vd}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3K12]
	Hai vòi nước cùng chảy vào một bể cạn. Nếu chảy một mình thì vòi $ I $ cần $ 45 $ phút mới đầy bể; vòi $ II $ cần $ 30 $ phút mới đầy bể. Nếu mở cả hai vòi cùng một lúc thì sau bao lâu sẽ đầy bể?
\loigiai{Trong $1 $ phút vòi $ I $ chảy được $ \dfrac{1}{45} $ bể.\\
	Trong $1 $ phút vòi $ II $ chảy được $ \dfrac{1}{30} $ bể.\\
	Trong $1 $ phút cả hai vòi chảy được $ \dfrac{1}{45} +\dfrac{1}{30}=\dfrac{1}{18}$ bể.\\
	Nếu mở cả hai vòi cùng một lúc thì thời gian chảy đầy bể là: $ 1 : \dfrac{1}{18}=18 $(phút).
}
\end{vd}
\begin{dang}{Tính giá trị của biểu thức}
	\textbf{\textit{Phương pháp giải}}: \begin{itemize}
		\item Thứ tự thực hiện các phép tính đối với phân số cũng tương tự như đối với số tự nhiên.
		\item Áp dụng các tính chất cơ bản của phép cộng, phép nhân phân số khi có thể.
	\end{itemize}
\end{dang}
\setcounter{vd}{0}
\begin{vd}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3K12]
Tính giá trị của các biểu thức sau$\colon$
\begin{enumEX}{2}
	\item $ M=\dfrac{5}{12}-\dfrac{21}{40} : \dfrac{14}{5} $;
	\item  $ N=\left(\dfrac{41}{75}+\dfrac{17}{100}\right) :\dfrac{129}{80} $.
\end{enumEX}
\loigiai{\begin{enumerate}[a)]%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%
		\item $ M=\dfrac{5}{12}-\dfrac{21}{40} : \dfrac{14}{5}=\dfrac{5}{12}-\dfrac{21 \cdot 5}{40 \cdot 14} =\dfrac{5}{12}-\dfrac{3}{16}=\dfrac{20}{48}-\dfrac{9}{48}=\dfrac{11}{48}$;
		\item  $ N=\left(\dfrac{41}{75}+\dfrac{17}{100}\right) :\dfrac{129}{80}=\left(\dfrac{164}{300}+\dfrac{51}{300}\right)\cdot \dfrac{80}{129}= \dfrac{215}{300} \cdot \dfrac{80}{129}=\dfrac{4}{9}$.
		\end{enumerate}
}
\end{vd}
\begin{vd}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3K12]
	Rút gọn các biểu thức sau$\colon$
	\begin{enumEX}{2}
		\item $ P=\dfrac{2-\dfrac{1}{3}+\dfrac{1}{4}}{2+\dfrac{1}{6}-\dfrac{1}{4}}$;
		\item  $ Q=\dfrac{3+\dfrac{1}{6}-\dfrac{2}{5}}{5-\dfrac{1}{6}+\dfrac{7}{10}}$.
	\end{enumEX}
	\loigiai{\begin{enumerate}[a)]%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%
			\item $ P=\dfrac{2-\dfrac{1}{3}+\dfrac{1}{4}}{2+\dfrac{1}{6}-\dfrac{1}{4}}=\left(2-\dfrac{1}{3}+\dfrac{1}{4}\right) : \left(2+\dfrac{1}{6}-\dfrac{1}{4}\right)=\dfrac{24-4+3}{12} : \dfrac{24+2-3}{12}=\dfrac{23}{12} : \dfrac{23}{12}=1$;
			\item  $ Q=\dfrac{3+\dfrac{1}{6}-\dfrac{2}{5}}{5-\dfrac{1}{6}+\dfrac{7}{10}}= \dfrac{30 \cdot \left(3+\dfrac{1}{6}-\dfrac{2}{5}\right)}{30 \cdot \left(5-\dfrac{1}{6}+\dfrac{7}{10}\right)}=\dfrac{90+5-12}{150-5+21}=\dfrac{83}{166}=\dfrac{1}{2}$.
		\end{enumerate}
	}
\end{vd}

Cho $ m=\dfrac{1}{3} - \dfrac{1}{5} $. Tính số nghịch đảo của $ m $.
\loigiai{ Ta có $ m=\dfrac{1}{3} - \dfrac{1}{5}=\dfrac{2}{15} $, số nghịch đảo của $ m $ là $ \dfrac{15}{2} $.
}
\end{bt}
\begin{bt}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3Y12]
Tính$\colon$ \vspace{0.3cm}
\begin{enumEX}{2}
	\item $\dfrac{8}{3} : \dfrac{2}{9}$;
	\item  $ (-10) : \dfrac{90}{11} $.
\end{enumEX}
\loigiai{
\begin{enumEX}{2}
	\item $\dfrac{8}{3} : \dfrac{2}{9}=\dfrac{8}{3} \cdot \dfrac{9}{2}=12$;
	\item  $ (-10) : \dfrac{90}{11} =(-10) \cdot \dfrac{11}{90}=-\dfrac{11}{9}$.
\end{enumEX}
}
\end{bt}
\begin{bt}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3B12]
	Thực hiện các phép tính$\colon$
	\begin{enumEX}{2}
		\item $\dfrac{11}{15} : \dfrac{7}{5} - \dfrac{12}{35} : 6$;
		\item  $ \left(\dfrac{14}{15}+\dfrac{-3}{20} : \dfrac{1}{4}\right) : \left(\dfrac{5}{21}- \dfrac{8}{35}\right)$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
				\item $\dfrac{11}{15} : \dfrac{7}{5} - \dfrac{12}{35} : 6=\dfrac{11}{15} \cdot \dfrac{5}{7} - \dfrac{12}{35} \cdot \dfrac{1}{6}=\dfrac{11}{21}-\dfrac{2}{35}=\dfrac{55}{105}-\dfrac{6}{105}=\dfrac{49}{105}=\dfrac{7}{15}$; \vspace{8pt}
			\item  $$\begin{aligned}
			 \left(\dfrac{14}{15}+\dfrac{-3}{20} : \dfrac{1}{4}\right) : \left(\dfrac{5}{21}- \dfrac{8}{35}\right) & =\left(\dfrac{14}{15}+\dfrac{-3}{20} \cdot4 \right) : \left(\dfrac{25}{105}-\dfrac{24}{105}\right)\\
			 & =\left(\dfrac{14}{15}-\dfrac{12}{20}\right) \cdot \dfrac{105}{1}\\ 
			 & =\dfrac{1}{3} \cdot 105 =35. 
			 \end{aligned}$$
		\end{enumEX}
	}
\end{bt}
\begin{bt}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3K12]

\begin{bt}%[Nguyễn Anh Tuấn, dự án EX-C2-6-2019]%[6D3G12]
	Hai người đi xe đạp vòng quanh một cái hồ. Người thứ nhất đi một vòng hết $ 15 $ phút, người thứ hai đi một vòng hết $ 40 $ phút. Nếu hai người khởi hành cùng một lúc từ cùng một điểm và đi cùng chiều thì sau bao lâu người thứ nhất lại đuổi kịp người thứ hai?
\loigiai{Trong $ 1 $ phút người thứ nhất đi được $ \dfrac{1}{15} $ (vòng hồ). \\
	Trong $ 1 $ phút người thứ hai đi được $ \dfrac{1}{40} $ (vòng hồ).\\
	Trong $ 1 $ phút người thứ nhất đi hơn người thứ hai là $\colon$ $ \dfrac{1}{15} -\dfrac{1}{40} =\dfrac{1}{24}$ (vòng hồ).\\
	Khi người thứ nhất đuổi kịp người thứ hai thì người thứ nhất đã đi hơn người thứ hai đúng $ 1 $ vòng hồ. Do đó thời gian người thứ nhất đuổi kịp người thứ hai là$\colon$ $$1 : \dfrac{1}{24}=24 \text{ phút }.$$
}
\end{bt}
