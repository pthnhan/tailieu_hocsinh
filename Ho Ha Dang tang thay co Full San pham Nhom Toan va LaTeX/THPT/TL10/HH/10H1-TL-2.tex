\subsection{Tóm tắt lí thuyết}
\subsubsection{Định nghĩa tổng và hiệu hai véc-tơ}
\begin{dn}[Phép cộng]
	Cho hai véc-tơ $\overrightarrow{a}$ và $\overrightarrow{b}$. 
	Với điểm $A$ bất kỳ, dựng $\overrightarrow{AB}=\overrightarrow{a}$, dựng $\overrightarrow{BC}=\overrightarrow{b}$. Khi đó, véc-tơ $\overrightarrow{AC}$ được gọi là véc-tơ tổng của $\overrightarrow{a}$ và $\overrightarrow{b}$. \\Ta ký hiệu: $\overrightarrow{a}+\overrightarrow{b}$, tức là: $\overrightarrow{a}+\overrightarrow{b}=\overrightarrow{AB}+\overrightarrow{BC}=\overrightarrow{AC}$.

\begin{dn}[Phép trừ]
	Cho hai véc-tơ $\overrightarrow{a}$ và $\overrightarrow{b}$. Phép phép trừ của  $\vec{a}$ với $\vec{b}$ được định nghĩa là phép cộng của $\vec{a}$ với $-\vec{b}$. \\Ký hiệu $\vec{a}-\vec{b}=\vec{a}+(-\vec{b})$.
\end{dn}

\subsubsection{Các tính chất của phép cộng, trừ hai véc-tơ}
\begin{tc}
(giao hoán và kết hợp) \begin{multicols}{2}
\begin{enumerate}
	\item $\vec{a}+\vec{b}=\vec{b}+\vec{a}$,
	\item  $\vec{a}+(\vec{b}+\vec{c})=(\vec{a}+\vec{b})+\vec{c}$.
\end{enumerate}
\end{multicols}
\end{tc}
\begin{tc}(véc-tơ đối)
\begin{multicols}{3}
\begin{enumerate}
	\item  $-\vec{0}=\vec{0}$
	\item  $\vec{a}-\vec{b}=-(\vec{b}-\vec{a})$,
	\item $-\vec{AB}=\vec{BA}$.
\end{enumerate}
\end{multicols}
\end{tc}
\begin{tc}
(cộng với véc-tơ $\vec{0}$) $\vec{a}+\vec{0}=\vec{0}+\vec{a}=\vec{a}$.
\end{tc}
\begin{tc}
	 Cho $ 3 $ điểm $A,B,C$ ta có: \begin{multicols}{2}
\begin{enumerate}
	\item $\vec{AB}+\vec{BC}=\vec{AC}$ (quy tắc 3 điểm),  
	\item $\vec{AB}-\vec{AC}=\vec{CB}$ (quy tắc trừ).
\end{enumerate}
	 \end{multicols}
\end{tc}
\begin{tc}
		\begin{enumerate}
		\item (quy tắc trung điểm) $ I $ là trung điểm $ AB $ $\Leftrightarrow \overrightarrow{IA}+\overrightarrow{IB}=\overrightarrow{0}$,
		\item (quy tắc trọng tâm) $ G $ là trọng tâm $\triangle ABC$ $\Leftrightarrow \overrightarrow{GA}+\overrightarrow{GB}+\overrightarrow{GC}=\overrightarrow{0}$.
	\end{enumerate}
\end{tc}

\begin{bt}%[Hồ Hà Đặng]%[0H1B2]
	Cho hình bình hành $ABCD$ có tâm $O$. Xác định các véc-tơ đối của các véc-tơ sau đây:
\begin{multicols}{2}
	\begin{enumerate}
	\item $\vec{OA}$, $\vec{DO}$.
	\item $\vec{AC}$, $\vec{DA}$.
\end{enumerate}
\end{multicols}
	\loigiai{
		\begin{multicols}{2}
			\begin{enumerate}
				\item $-\vec{OA}=\vec{AO}=\vec{OC}$, $-\vec{DO}=\vec{OD}=\vec{BO}$.
				\item $-\vec{AC}=\vec{CA}$, $-\vec{DA}=\vec{AD}=\vec{BC}$.
			\end{enumerate}
		\end{multicols}
	}
\end{bt}
\begin{bt}%[Hồ Hà Đặng]%[0H1B2]
Cho hình bình hành $ABCD$ có tâm $O$. Xác định các véc-tơ sau đây:
\begin{multicols}{2}
\begin{enumerate}
	\item $\vec{OA}+\vec{OB}+\vec{OC}+\vec{OD}$.
	\item $\vec{OA}+\vec{BO}+\vec{CO}+\vec{DO}$.
	\item $\vec{AC}+\vec{BD}+\vec{BA}+\vec{DA}$.
	\item $\vec{OA}+\vec{CB}+\vec{OC}+\vec{AD}$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
	\item $\vec{OA}+\vec{OB}+\vec{OC}+\vec{OD}=\vec{0}$.
	\item $\vec{OA}+\vec{BO}+\vec{CO}+\vec{DO}=\vec{CO}+\vec{OA}+\vec{BO}+\vec{DO}=\vec{CA}$.
	\item $\vec{AC}+\vec{BD}+\vec{BA}+\vec{DA}=\vec{BA}+\vec{AC}+\vec{BD}+\vec{DA}=\vec{AC}+\vec{BA}=\vec{BC}$.
	\item $\vec{OA}+\vec{OC}+\vec{CB}+\vec{AD}=\vec{0}$.
\end{enumerate}
}
\end{bt}
\begin{bt}%[Hồ Hà Đặng]%[0H1K2]
Cho tam giác $ABC$. Tìm véc-tơ $\vec{x}$ trong các trường hợp:
\begin{multicols}{2}
\begin{enumerate}
	\item $\vec{x}+\vec{BC}=\vec{AC}+\vec{BA}$.
	\item $\vec{CA}-\vec{x}-\vec{CB}=\vec{AB}$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
	\item $\vec{x}=\vec{AC}+\vec{BA}+\vec{CB}=\vec{0}$.
	\item $\vec{x}=\vec{CA}-\vec{CB}+\vec{BA}=\vec{BA}+\vec{BA}=\vec{BE}$, với $\vec{AE}=\vec{BA}$.
\end{enumerate}
}
\end{bt}
\begin{bt}%[Hồ Hà Đặng]%[0H1K2]

	\begin{multicols}{2}

			\item $\vec{PB}+\vec{MC}+\vec{NA}$.
			\item $\vec{BA}+\vec{PA}+\vec{CM}$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
	\item $\vec{PB}+\vec{MC}+\vec{NA}=\vec{AP}+\vec{PN}+\vec{NA}=\vec{0}$.
	\item $\vec{BA}+\vec{PA}+\vec{CM}=\vec{BA}+\vec{NP}+\vec{PA}=\vec{BA}+\vec{NA}=\vec{ND}$ (dựng thêm điểm $D$ sao cho $\vec{AD}=\vec{BA}$).
\end{enumerate}
	}
\end{bt}

\begin{bt}%[Hồ Hà Đặng]%[0H1K2]
Cho tam giác $ABC$, gọi $M$ là trung điểm $AC$ và $N$ là điểm đối xứng của $B$ qua $M$. Xác định các véc-tơ sau đây:
\begin{multicols}{2}
\begin{enumerate}
	\item $\vec{AB}+\vec{AN}$.
	\item $\vec{BA}+\vec{CN}$.
	\item $\vec{AB}+\vec{MC}+\vec{MN}$.
	\item $\vec{BA}+\vec{BC}-\vec{MN}$.
\end{enumerate}
\end{multicols}
	\loigiai{
Ta có, tứ giác $BANC$ là hình bình hành.
\begin{enumerate}
	\item $\vec{AB}+\vec{AN}=\vec{AC}$ (tính chất hình bình hành $BANC$).
	\item $\vec{BA}+\vec{CN}=\vec{BE}$ (dựng $\vec{AE}=\vec{CN}$).
	\item $\vec{AB}+\vec{MC}+\vec{MN}=\vec{AB}+\vec{AM}+\vec{MN}=\vec{AB}+\vec{AN}=\vec{AC}$.
	\item $\vec{BA}+\vec{BC}-\vec{MN}=\vec{BN}+\vec{NM}=\vec{BM}$.
\end{enumerate}
	}
\end{bt}
\begin{bt}%[Hồ Hà Đặng]%[0H1K2]
	Cho hình lục giác đều $ABCDEF$, gọi $M$, $N$, $P$, $Q$, $R$, $S$ lần lượt là trung điểm $AB$, $BC$, $CD$, $DE$, $EF$, $FA$.
	Xác định các véc-tơ sau đây:
	\begin{multicols}{2}
		\begin{enumerate} 
			\item $\vec{AD}+\vec{BE}+\vec{CF}-\vec{AE}-\vec{BF}-\vec{CD}$.
			\item $\vec{MQ}+\vec{RN}+\vec{PS}$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate} 
	\item $\vec{AD}+\vec{BE}+\vec{CF}-\vec{AE}-\vec{BF}-\vec{CD}=\vec{ED}+\vec{FE}+\vec{DF}=\vec{0}$.
	\item $\vec{MQ}+\vec{RN}+\vec{PS}=\vec{BD}+\vec{FB}+\vec{DF}=\vec{0}$.
\end{enumerate}
	}
\end{bt}
\begin{bt}%[Hồ Hà Đặng]%[0H1K2]
	Cho tam giác $ABC$. Gọi $D,E,F$ lần lượt nằm trên cạnh $BC,AC,AB$ sao cho $BD=\dfrac{1}{3}BC$, $CE=\dfrac{1}{3}CA$, $AF=\dfrac{1}{3}AB$. Xác định các véc-tơ sau đây:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\vec{AF}+\vec{BD}+\vec{CE}$
			\item $\vec{AD}+\vec{BE}+\vec{CF}$
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item Lấy thêm các điểm $P,Q$ về phía ngoài cạnh $AB,AC$ sao cho $\vec{CE}=\vec{AP}$, $\vec{QA}=\vec{AF}$. Theo đó, tam giác $APQ$ đồng dạng tam giác $ACB$ nên ta có $\vec{PQ}=\vec{BD}$.
			Khi đó, $\vec{AF}+\vec{BD}+\vec{CE}=\vec{QA}+\vec{PQ}+\vec{AP}=\vec{0}$.
			\item $\vec{AD}+\vec{BE}+\vec{CF}=\vec{BD}-\vec{BA}+\vec{CE}-\vec{CB}+\vec{AF}-\vec{AC}=(\vec{BD}+\vec{CE}+\vec{AF})-(\vec{BA}+\vec{CB}+\vec{AC})=\vec{0}$.
		\end{enumerate}
	}
\end{bt}

\begin{dang}{Xác định điểm thỏa đẳng thức véc-tơ cho trước}
	Để xác định điểm $M$ thỏa đẳng thức véc-tơ cho trước, ta làm như sau:
	\\$\circ$ \textbf{HƯỚNG 1:}\\
	$-$  Biến đổi đẳng thức véc-tơ đã cho về dạng $\vec{AM}=\vec{v}$, trong đó $A$ là điểm cố định và $\vec{v}$ là véc-tơ cố định.\\
