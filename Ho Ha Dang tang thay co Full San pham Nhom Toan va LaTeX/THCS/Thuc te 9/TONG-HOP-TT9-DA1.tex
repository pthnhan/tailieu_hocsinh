		\item Giải hệ phương trình bằng phương pháp thế hoặc cộng đại số.

\begin{vd}%[SGK-9-tr20]%[Cao Thành Thái]%[9D3G5]
	Tìm số tự nhiên có hai chữ số, biết rằng hai lần chữ số hàng đơn vị lớn hơn chữ số hàng chục $ 1 $ đơn vị, và nếu viết hai chữ số ấy theo thứ tự ngược lại thì được một số mới (có hai chữ số) bé hơn số cũ $ 27 $ đơn vị.
	\loigiai{
		Gọi số tự nhiên có hai chữ số là $x=\overline{ab}, a,b\in\overline{1,9}$. Ta có $x=10a+b$.
		\\
		Do hai lần chữ số hàng đơn vị lớn hơn chữ số hàng chục $ 1 $ đơn vị nên
		\begin{equation}
			2b=a+1\Leftrightarrow a-2b=-1
		\end{equation}
		Khi viết ngược lại thì được một số mới (có hai chữ số) bé hơn số cũ $ 27 $ đơn vị tức là \begin{equation}
			\overline{ba}=\overline{ab}-27\Leftrightarrow 10b+a=10a+b-27\Leftrightarrow 9a-9b=27\Leftrightarrow a-b=3.
		\end{equation}
		Ta có hệ phương trình \begin{equation*}
			\heva{&a-2b=-1\\&a-b=3}\Leftrightarrow \heva{&a=7\\&b=4.}
		\end{equation*}
		\ding{71} Vậy số cần tìm là $74$.
	}
\end{vd}

\begin{vd}%[SGK-9-tr21]%[Cao Thành Thái]%[9D3T5]
	Một chiếc xe tải đi từ Sài Gòn đến Cần Thơ, quãng đường dài $ 189 $ km. Sau khi xe tải xuất phát $ 1 $ giờ, một chiếc xe khách bắt đầu đi từ Cần Thơ về Sài Gòn và gặp xe tải sau khi đã đi được $ 1 $ giờ $ 48 $ phút. Tính vận tốc (trung bình) của mỗi xe, biết rằng mỗi giờ xe khách đi nhanh hơn xe tải $13$ km.
	\loigiai{
		Gọi vận tốc của xe tải và xe khách lần lượt là $x,y$ (km/h). Ta có $1$ giờ $48$ phút $=1{,}8$ h.
		\\
		Mỗi giờ xe khách đi nhanh hơn xe tải $13$ km nên
		\begin{equation}
			y=x+13\label{dang1.3}
		\end{equation}
		Thời gian xe tải đi là $1+1{,}8=2{,}8$ h. Quãng đường xe tải đi là $2{,}8x$.  Thời gian xe khách đi là $1{,}8$ h. Quãng đường xe khách đi là $1{,}8y$. Vì tổng quãng đường hai xe bằng khoảng cách Sài Gòn - Cần Thơ nên
		\begin{equation}
			2{,}8x+1{,}8y=189
		\end{equation}
		Ta có hệ phương trình \begin{equation*}
			\heva{&x-y=-13\\&2{,}8x+1{,}8y=189}\Leftrightarrow \heva{&x=36\\&y=49.}
		\end{equation*}
		\ding{71} Vậy vận tốc của xe tải là $36$ km/h và xe khách là $49$ km/h.
	}
\end{vd}


\begin{vd}%[SGK-9-tr22]%[Cao Thành Thái]%[9D3T6]
	Hai đội công nhân $A$ và $B$ cùng làm một đoạn đường trong $24$ ngày thì xong. Mỗi ngày, phần việc của đội $A$ nhiều gấp rưỡi đội $B$. Hỏi nếu làm một mình thì mỗi đội làm xong đoạn đường đó trong bao lâu?
	\loigiai{
		Gọi số ngày mà đội $A$ và đội $B$ một mình làm xong đoạn đường lần lượt là $a,b$ (ngày). 
		\\
		Ta coi công việc bằng $1$, khi đó mỗi ngày đội $A$ làm được $\dfrac{1}{a}$ công việc và đội $B$ làm được $\dfrac{1}{b}$ công việc. Và phần việc của đội $A$ nhiều gấp rưỡi đội $B$ nên
		\begin{equation}
			\dfrac{1}{a}=1{,}5\cdot\dfrac{1}{b}\Leftrightarrow \dfrac{1}{a}-\dfrac{3}{2}\cdot\dfrac{1}{b}=0\label{dang1.5}
		\end{equation}
		Mỗi ngày, hai đội cùng làm thì hết $\dfrac{1}{24}$ công việc (do hoàn thành trong $ 24 $ ngày) nên
		\begin{equation}
			\dfrac{1}{a}+\dfrac{1}{b}=\dfrac{1}{24}\label{dang1.6}
		\end{equation}
		Từ \eqref{dang1.5} và \eqref{dang1.6} ta có hệ phương trình \begin{equation*}
			\heva{&\dfrac{1}{a}-\dfrac{3}{2}\cdot\dfrac{1}{b}=0\\&\dfrac{1}{a}+\dfrac{1}{b}=\dfrac{1}{24}}\Leftrightarrow \heva{&\dfrac{1}{a}=\dfrac{1}{40}\\&\dfrac{1}{b}=\dfrac{1}{60}}\Leftrightarrow \heva{&a=40\\&b=60.}
		\end{equation*}
		\ding{71} Vậy đội $A$ làm một mình cần $40$ ngày và đội $B$ cần $60$ ngày.
	}
\end{vd}

	{
		Gọi $x$ là số lớn, $y$ là số nhỏ.\\

		Do tổng của hai số tự nhiên bằng $1006$ nên
		\begin{equation}
			x + y = 1006 \label{dang1.7}
		\end{equation}
		Do số lớn chia cho số nhỏ được thương là $2$ và số dư là $124$ nên
		\begin{equation}
			x = 2y + 124 \label{dang1.8}
		\end{equation}
		Từ \eqref{dang1.7} và \eqref{dang1.8} ta có hệ phương trình
		\begin{equation*}
			\heva{& x + y = 1006 \\& x - 2y = 124} \Leftrightarrow \heva{& x = 712 \\& y = 294.} \text{ (thỏa điều kiện).}
		\end{equation*}
		\ding{71} Vậy hai số cần tìm là $712$ và $294$.
	}
\end{vd}

\begin{vd}%[SGK-9-tr22]%[Cao Thành Thái]%[9D3T5]
	Một ô-tô đi từ $A$ và dự định đến $B$ lúc $12$ giờ trưa. Nếu xe chạy với vận tốc $35$ km/h thì sẽ đến $B$ chậm $2$ giờ so với dự định. Nếu xe chạy với vận tốc $50$ km/h thì sẽ đến $B$ sớm hơn $1$ giờ so với dự định. Tính độ dài quãng đường $AB$ và thời điểm xuất phát của ô-tô tại $A$.
	\loigiai
	{
		Gọi $x$ (km) là độ dài quãng đường $AB$ và $y$ (giờ) là thời gian dự định để đi từ $A$ đến $B$ đúng lúc $12$ giờ trưa.\\
		Điều kiện: $x > 0$, $y > 1$ (do ô-tô đến $B$ sớm hơn $1$ giờ).\\
		Do xe chạy với vận tốc $35$ km/h sẽ đến $B$ chậm $2$ giờ so với dự định nên
		\begin{equation}
			\dfrac{x}{35} - y = 2 \label{dang1.11}
		\end{equation}
		Do xe chạy với vận tốc $50$ km/h sẽ đến $B$ sớm hơn $1$ giờ so với dự định nên
		\begin{equation}
			y - \dfrac{x}{50} = 1 \label{dang1.12}
		\end{equation}
		Từ \eqref{dang1.11} và \eqref{dang1.12} ta có hệ phương trình
		\begin{equation*}
			\heva{& \dfrac{x}{35} - y = 2 \\& \dfrac{x}{50} - y = -1} \Leftrightarrow \heva{& x = 350 \\& y = 8.} \text{ (thỏa điều kiện).}
		\end{equation*}
		\ding{71} Vậy quãng đường $AB$ là $350$ km và thời điểm xuất phát của ô-tô tại $A$ là $12 - 8 = 2$ giờ.
	}
\end{vd}


\begin{vd}%[SGK-9-tr23]%[Cao Thành Thái]%[9D3G6]
	Tính độ dài hai cạnh góc vuông của một tam giác vuông, biết rằng nếu tăng mỗi cạnh lên $3$ cm thì diện tích tam giác đó sẽ tăng thêm $36$ c m$^2$, và nếu một cạnh giảm đi $2$ cm, cạnh kia giảm đi $4$ cm thì diện tích của tam giác giảm đi $26$ c m$^2$.
	\loigiai
	{
		Gọi $x$ (cm), $y$ (cm) là độ dài hai cạnh góc vuông của tam giác vuông.\\
		Điều kiện: $x > 0$ và $y > 0$.\\
		Do tăng mỗi cạnh lên $3$ cm thì diện tích tam giác đó sẽ tăng thêm $36$ c m$^2$ nên
		\begin{equation}
			\dfrac{(x+3)(y+3)}{2} = \dfrac{xy}{2} + 36 \label{dang1.13}
		\end{equation}
		Do một cạnh giảm đi $2$ cm, cạnh kia giảm đi $4$ cm thì diện tích của tam giác giảm đi $26$ c m$^2$ nên
		\begin{equation}
			\dfrac{(x-2)(y-4)}{2} = \dfrac{xy}{2} - 26 \label{dang1.14}
		\end{equation}
		Từ \eqref{dang1.13} và \eqref{dang1.14} ta có hệ phương trình
		\begin{equation*}
			\heva{& \dfrac{(x+3)(y+3)}{2} = \dfrac{xy}{2} + 36 \\& \dfrac{(x-2)(y-4)}{2} = \dfrac{xy}{2} - 26} \Leftrightarrow \heva{& (x+3)(y+3) = xy + 72 \\& (x-2)(y-4) = xy - 52} \Leftrightarrow \heva{& 3x + 3y = 63 \\& 4x + 2y = 60} \Leftrightarrow \heva{& x = 9 \\& y = 12.} \text{ (thỏa đk).}
		\end{equation*}
		\ding{71} Vậy độ dài hai cạnh góc vuông của tam giác vuông là $9$ cm và $12$ cm.
	}
\end{vd}


\begin{vd}%[SGK-9-tr23]%[Cao Thành Thái]%[9D3T6]
	Hai vòi nước cùng chảy vào một bể nước cạn (không có nước) thì sau $4\dfrac{4}{5}$ giờ đầy bể. Nếu lúc đầu chỉ mở vòi thứ nhất và $9$ giờ sau mới mở thêm vòi thứ hai thì sau $\dfrac{6}{5}$ giờ nữa mới đầy bể. Hỏi nếu ngay từ đầu chỉ mở vòi thứ hai thì sau bao lâu mới đầy bể?
	\loigiai
	{
		Gọi $x$ (giờ), $y$ (giờ) lần lượt là thời gian để chỉ riêng vòi thứ nhất, vòi thứ hai chảy đầy bể.\\
		Điều kiện: $x > 0$ và $y > 0$.\\
		Trong $1$ giờ vòi thứ nhất chảy được $\dfrac{1}{x}$ bể, vòi thứ hai chảy được $\dfrac{1}{y}$ bể.\\
		Cả hai vòi cùng chảy thì đầy bể sau $4\dfrac{4}{5} \text{ giờ } = \dfrac{24}{5} \text{ giờ}$ nên trong $1$ giờ cả hai vòi cùng chảy được $\dfrac{5}{24}$ bể.\\
		Ta được:
		\begin{equation}
			\dfrac{1}{x} + \dfrac{1}{y} = \dfrac{5}{24} \label{dang1.15}
		\end{equation}
		Trong $9$ giờ vòi thứ nhất chảy được $\dfrac{9}{x}$ bể.\\
		Trong $\dfrac{6}{5}$ giờ cả hai vòi chảy được: $\dfrac{6}{5} \left( \dfrac{1}{x} + \dfrac{1}{y} \right)$ bể.\\
		Theo đề bài vòi thứ nhất được mở chảy $9$ giờ sau mới mở thêm vòi thứ hai thì sau $\dfrac{6}{5}$ giờ nữa mới đầy bể nên
		\begin{equation}
			\dfrac{9}{x} + \dfrac{6}{5} \left( \dfrac{1}{x} + \dfrac{1}{y} \right) = 1 \Leftrightarrow \dfrac{51}{x} + \dfrac{6}{y} = 5 \label{dang1.16}
		\end{equation}
		Từ \eqref{dang1.15} và \eqref{dang1.16} ta có hệ phương trình
		\begin{equation*}
			\heva{& \dfrac{1}{x} + \dfrac{1}{y} = \dfrac{5}{24} \\& \dfrac{51}{x} + \dfrac{6}{y} = 5} \Leftrightarrow \heva{& \dfrac{1}{x} = \dfrac{1}{12} \\& \dfrac{1}{y} = \dfrac{1}{8}} \Leftrightarrow \heva{& x = 12 \\& y = 8.} \text{ (thỏa điều kiện).}
		\end{equation*}
		\ding{71} Vậy nếu ngay từ đầu chỉ mở vòi thứ hai thì sau $8$ giờ bể sẽ được bơm đầy nước.
	}
\end{vd}


\begin{vd}%[SGK-9-tr24]%[Cao Thành Thái]%[9D3T6]
	Hai người thợ cùng làm một công việc trong $16$  giờ thì xong. Nếu người thứ nhất làm $3$ giờ và người thứ hai làm $6$ giờ thì chỉ hoàn thành được $25 \%$ công việc. Hỏi nếu làm riêng thì mỗi người hoàn thành công việc đó trong bao lâu?
	\loigiai
	{
		Gọi $x$ (giờ), $y$ (giờ) lần lượt là thời gian để người thứ nhất, người thứ hai một mình hoàn thành công việc.\\
		Điều kiện: $x > 0$ và $y > 0$.\\
		Trong $1$ giờ người thứ nhất làm được $\dfrac{1}{x}$ công việc, người thứ hai làm được $\dfrac{1}{y}$ công việc, cả hai cùng làm chung thì được $\dfrac{1}{16}$ công việc.\\

		\end{equation}

	}

	Một đoàn xe ô tô cần chở $30$ tấn hàng theo dự định, mỗi xe chở như nhau. Khi bắt đầu khởi hành thì có thêm $2$ ô tô đến nữa nên mỗi xe chở ít hơn $0{,}5$ tấn so với dự định. Hỏi lúc đầu đoàn xe có bao nhiêu ô tô?
	\loigiai{
		Gọi $x$ là số xe ô tô lúc đầu đoàn xe có $(x \in \mathbb{N}^*)$. Suy ra số xe ô tô lúc khởi hành là $x + 2$.\\
		Lúc đầu mỗi xe dự định chở $\dfrac{30}{x}$ tấn và khi khởi hành mỗi xe chở $\dfrac{30}{x + 2}$ tấn.\\
		Theo đề bài, ta có phương trình:
		$$\dfrac{30}{x} = \dfrac{30}{x + 2} + \dfrac{1}{2} \Leftrightarrow 60(x + 2) = 60x + x(x + 2) \Leftrightarrow x^2 + 2x - 120 = 0 \Leftrightarrow \hoac{&x = 10 \\ &x = -12.}$$
		Vậy lúc đầu đoàn xe có $10$ ô tô.
	}
\end{vd}

\begin{vd}%[Thanh Hoàng Đinh]%[Bài 32]
	Một phân xưởng dự định dệt $3000$ tấm thảm. Trong $8$ ngày đầu, họ đã thực hiện đúng kế hoạch đề ra mỗi ngày, những ngày còn lại họ dệt vượt mức mỗi ngày $10$ tấm thảm nên hoàn thành kế hoạch trước $2$ ngày. Hỏi theo kế hoạch, phân xưởng phải dệt mỗi ngày bao nhiêu tấm thảm?
	\loigiai{
		Gọi $x$ là số tấm thảm mỗi ngày phân xưởng phải dệt theo kế hoạch $(x \in \mathbb{N}^*)$.\\
		Số ngày phân xưởng phải dệt theo kế hoạch là $\dfrac{3000}{x}$ ngày.\\
		Số ngày phân xưởng dệt vượt mức là $\dfrac{3000 - 8x}{x + 10}$.\\
		Theo đề bài, ta có phương trình:
		$$\dfrac{3000}{x} - 2 = 8 + \dfrac{3000 - 8x}{x + 10} \Leftrightarrow 3000(x + 10) = 10x(x + 10) + (3000 - 8x)x$$ 								$$\Leftrightarrow 2x^2 + 100x - 30000 = 0 \Leftrightarrow \hoac{&x = 100 \\ &x = -150.}$$
		Vậy theo kế hoạch, phân xưởng phải dệt mỗi ngày $100$ tấm thảm.
	}
\end{vd}

\begin{vd}%[Thanh Hoàng Đinh]%[Bài 33]
	Lúc $7$ giờ $30$ phút, một ô tô khởi hành từ $A$. Đến $B$, ô tô nghỉ $30$ phút rồi đi tiếp, đến $C$ lúc $9$ giờ $30$ phút. Biết quãng đường $AB$ dài $30$ km, quãng đường $BC$ dài $50$ km và vận tốc đi trên quãng đường $AB$ lớn hơn vận tốc đi trên quãng đường $BC$ là $10$ km/giờ. Tính vận tốc ô tô đi trên các quãng đường $AB$, $BC$.
	\loigiai{
		Gọi $x$( km/h) là vận tốc ô tô đi trên quãng đường $AB$ $(x > 10)$. Suy ra vận tốc ô tô đi trên quãng đường $BC$ là $x - 10$.\\
		Thời gian ô tô đi hết quãng đường $AB$ là $\dfrac{30}{x}$ và thời gian ô tô đi hết quãng đường $BC$ là $\dfrac{50}{x - 10}$.\\
		Theo đề bài, ta có phương trình:
		$$\dfrac{30}{x} + \dfrac{50}{x - 10} + \dfrac{1}{2} = 2 \Leftrightarrow 60(x - 10) + 100x + x(x - 10) = 4x(x - 10)$$
		$$\Leftrightarrow 3x^2 -190x + 600 = 0 \Leftrightarrow \hoac{&x = 60 \\ &x = \dfrac{10}{3}.}$$
		Vậy vận tốc ô tô đi trên các quãng đường $AB$, $BC$ lần lượt là $60$ km/h, $50$ km/h.
	}
\end{vd}

\begin{vd}%[Thanh Hoàng Đinh]%[Bài 34]
	Một người đi xe đạp từ tỉnh $A$ đến tỉnh $B$ dài $36$ km. Lúc về người ấy tăng tốc thêm $3$ km/h, do đó thời gian về ít hơn thời gian đi là $36$ phút. Tính vận tốc lúc đi của người đó.
	\loigiai{
		Gọi $x$ ( km/h) là vận tốc lúc đi của người đi xe đạp $(x > 0)$. Khi đó vận tốc lúc về của người đi xe đạp là $x + 3$.\\
		Thời gian lúc đi là $\dfrac{36}{x}$ và thời gian lúc về là $\dfrac{36}{x + 3}$.\\
		Vì thời gian về ít hơn thời gian đi là $36$ phút nên ta có phương trình:
		$$\dfrac{36}{x} = \dfrac{36}{x + 3} + \dfrac{3}{5} \Leftrightarrow 60(x + 3) = 60x + x(x + 3) \Leftrightarrow x^2 + 3x - 180 = 0 \Leftrightarrow \hoac{&x = 12 \\ &x = -15.}$$
		Vậy vận tốc của xe lúc đi là $12$ km/h.
	}
\end{vd}
