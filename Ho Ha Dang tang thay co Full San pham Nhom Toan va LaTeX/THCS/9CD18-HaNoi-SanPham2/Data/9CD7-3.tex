
\section{Phương pháp liên hợp}

\begin{note}
Mục đích chính là khử căn thức và đặt nhân tử chung, muốn vậy ta có thể \underline{\textbf{nhẩm nghiệm}}, \textbf{\underline{thêm bớt (tách)}} hoặc \underline{\textbf{nhóm số hạng}} để nhân liên hợp. Một số hằng đẳng thức thường được sử dụng:
\[
\begin{cases}
x^{2}-y^{2}&=(x-y)(x+y)\\
x^{3}-y^{3}&=(x-y)(x^{2}-xy+y^{2})\\
...\\
x^{n}-y^{n}&=(x-y)(x^{n-1}+x^{n-2}y+...+xy^{n-2}+y^{n-1})
\end{cases}
\Rightarrow
\begin{cases}
\sqrt{x}\pm\sqrt{y}&=\dfrac{x-y}{\sqrt{x}\mp\sqrt{y}}\\
\sqrt[3]{x}\pm\sqrt[3]{y}&=\dfrac{x-y}{\sqrt[3]{x^{2}}\mp\sqrt[3]{xy}+\sqrt[3]{y^{2}}}\\
\sqrt[4]{x}\pm\sqrt[4]{y}&=\dfrac{x-y}{(\sqrt[4]{x}\mp\sqrt[4]{y})(\sqrt{x}+\sqrt{y})}
\end{cases}
\]
\underline{\textbf{Nhận xét:}} muốn tìm ra số (hoặc biểu thức) để thêm bớt, ta có thể dự đoán (nhẩm nghiệm) bằng cách dùng chức năng Shift Solve của Casio.
\end{note}

\begin{dang}{Phương trình có nghiệm đơn}
	Giả sử phương trình có nghiệm đơn $ x=x_{0} $ thì ta cần tạo ra nhân tử chung $ x-x_{0} $.\\
	Muốn tìm lượng cần thêm (bớt), ta thay nghiệm $ x_{0} $ vào từng căn thức. Khi đó ta lấy căn thức đó trừ đi kết quả vừa tìm được.
\end{dang}

\begin{bt}%[Đề thi Đại học Khối A năm 2007]%[Dương BùiĐức]
	Giải phương trình $\sqrt{2x-3}-\sqrt{x}=2x-6$.
	\loigiai{
		Điều kiện $ x\geq \dfrac{3}{2} $.\\
		Nhận xét $ x=3 $ là nghiệm của phương trình.\\
		Ta có
		\[ 
		\dfrac{x-3}{\sqrt{2x-3}+\sqrt{x}}=2(x-3).
		\]
		Vì $ x\geq \dfrac{3}{2} $ nên $ \sqrt{2x-3}+\sqrt{x}\geq \sqrt{\dfrac{3}{2}}\Rightarrow \dfrac{1}{\sqrt{2x-3}+\sqrt{x}}<1 $. Do đó phương trình có nghiệm duy nhất $ x=3 $.
	}
\end{bt}

\begin{bt}%[Đề dự bị Đại học Khối B năm 2008]%[Dương BùiĐức]
	Giải phương trình $\sqrt{10x+1}+\sqrt{3x-5}=\sqrt{9x+4}+\sqrt{2x-2}$.
	\loigiai{
		Điều kiện xác định $ x\geq \dfrac{5}{3} $.\\
		Ta có
		\begin{eqnarray*}
			&&\sqrt{10x+1}-\sqrt{9x+4}+\sqrt{3x-5}-\sqrt{2x-2}=0\\
			&\Leftrightarrow& \dfrac{x-3}{\sqrt{10x+1}+\sqrt{9x+4}}+\dfrac{x-3}{\sqrt{3x-5}+\sqrt{2x-2}}=0\\
			&\Leftrightarrow& (x-3)\left( \dfrac{1}{\sqrt{10x+1}+\sqrt{9x+4}}+\dfrac{1}{\sqrt{3x-5}+\sqrt{2x-2}}\right) =0.
		\end{eqnarray*}
	Vì $ \dfrac{1}{\sqrt{10x+1}+\sqrt{9x+4}}+\dfrac{1}{\sqrt{3x-5}+\sqrt{2x-2}}>0 $ với $ x\geq \dfrac{5}{3} $ nên phương trình có nghiệm duy nhất $ x=3 $.
	}
\end{bt}

\begin{bt}%[Dương BùiĐức]
	Giải phương trình $(\sqrt{x+2}-\sqrt{x-1})(\sqrt{2-x}+1)=1$.
	\loigiai{
Điều kiện xác định $ 1\leq x\leq 2 $.\\
Ta có
\begin{eqnarray*}
	&&(\sqrt{x+2}-\sqrt{x-1})(\sqrt{2-x}+1)=1\\
	&\Leftrightarrow&\dfrac{3}{\sqrt{x+2}+\sqrt{x-1}}=\dfrac{1}{\sqrt{2-x}+1}\\
	&\Leftrightarrow& (\sqrt{x+2}-2)+(\sqrt{x-1}-1)=3\sqrt{2-x}\\
	&\Leftrightarrow& \dfrac{x-2}{\sqrt{x+2}+2}+\dfrac{x-2}{\sqrt{x+1}+1}=3\sqrt{2-x}\\
	&\Leftrightarrow& \sqrt{x-2}\left( \dfrac{\sqrt{x-2}}{\sqrt{x+2}+2}+\dfrac{\sqrt{x-2}}{\sqrt{x+1}+1}-3\right) =0.
\end{eqnarray*}
Vì $ \dfrac{\sqrt{x-2}}{\sqrt{x+2}+2}+\dfrac{\sqrt{x-2}}{\sqrt{x+1}+1}-3<0 $ với $ 1\leq x\leq 2 $ nên phương trình có nghiệm duy nhất $ x=2 $.
}
\end{bt}

\begin{bt}%[Dương BùiĐức]
	Giải phương trình $\sqrt{x^{2}+12}+5=3x+\sqrt{x^{2}+5}$.
	\loigiai{
		Điều kiện xác định $ x\in\mathbb{R} $.\\
		Nhận xét $x=2$ là nghiệm của phương trình. Ta có
		\begin{eqnarray*}
			&&\sqrt{x^{2}+12}-4=3(x-2)+(\sqrt{x^{2}+5}-3)\\
			&\Leftrightarrow& (x-2)\left[\dfrac{x+2}{\sqrt{x^{2}+12}+4}-\dfrac{x+2}{\sqrt{x^{2}+5}+3}-3\right]=0.
		\end{eqnarray*}
	Vì $ \sqrt{x^{2}+12}>\sqrt{x^{2}+5}$ nên để phương trình có nghiệm thì $ 3x>5\Rightarrow x>\dfrac{5}{3} $.\\
	Lại có $ \sqrt{x^{2}+12}+4>x+2 $ và $ \sqrt{x^{2}+5}+3>x+2 $ nên $ \dfrac{x+2}{\sqrt{x^{2}+12}+4}-\dfrac{x+2}{\sqrt{x^{2}+5}+3}-3<0 $, do đó phương trình có nghiệm duy nhất $ x=2 $.
	}
\end{bt}

\begin{bt}%[Đề thi học sinh giỏi Lâm Đồng năm 2008]%[Dương BùiĐức]
	Giải phương trình $\sqrt{3x^{2}-5x+1}-\sqrt{x^{2}-2}=\sqrt{3(x^{2}-x-1)}-\sqrt{x^{2}-3x+4}$.
	\loigiai{
		Ta có
		\begin{eqnarray*}
			&&\sqrt{3x^{2}-5x+1}-\sqrt{3(x^{2}-x-1)}=\sqrt{x^{2}-2}-\sqrt{x^{2}-3x+4}\\
			&\Leftrightarrow&(x-2)\left(\dfrac{2}{\sqrt{3x^{2}-5x+1}+\sqrt{3(x^{2}-x-1)}}+\dfrac{3}{\sqrt{x^{2}-2}+\sqrt{x^{2}-3x+4}}\right) =0.
		\end{eqnarray*}
	Vì $ \dfrac{2}{\sqrt{3x^{2}-5x+1}+\sqrt{3(x^{2}-x-1)}}+\dfrac{3}{\sqrt{x^{2}-2}+\sqrt{x^{2}-3x+4}}>0 $ nên phương trình có nghiệm duy nhất $ x=2 $.
	}
\end{bt}

\begin{bt}%[Dương BùiĐức]
	Giải phương trình $x^{3}+3x^{2}-3\sqrt[3]{3x+5}=1-3x$.
	\loigiai{
		Nhận xét $ x=1 $ là nghiệm của phương trình.\\
		Ta có $(x+1)^{3}-8=3\left(\sqrt[3]{3x+5}-2\right)\Leftrightarrow (x-1)[(x+1)^{2}+2(x+1)+4]=\dfrac{9(x-1)}{\sqrt[3]{(3x+5)^{2}}+2\sqrt[3]{3x+5}+4}$.\\
		Vì $
		\begin{cases}
		(x+1)^{2}+2(x+1)+4&\geq 3\\
		\sqrt[3]{(3x+5)^{2}}+2\sqrt[3]{3x+5}+4&\geq 3
		\end{cases}$ nên 
		\[ 
		[(x+1)^{2}+2(x+1)+4]\left(\sqrt[3]{(3x+5)^{2}}+2\sqrt[3]{3x+5}+4\right)\geq 9.
		\]
		Do đó phương trình có nghiệm duy nhất $ x=1 $.
	}
\end{bt}

\begin{bt}%[Đề thi Đại học Khối B năm 2010]%[Dương BùiĐức]
	Giải phương trình $\sqrt{3x+1}-\sqrt{6-x}+3x^{2}-14x-8=0$.
	\loigiai{
		Điều kiện $ -\dfrac{1}{3}\leq x\leq 6 $.\\
		Nhận xét $ x=5 $ là nghiệm của phương trình.\\
		Ta có
		\begin{eqnarray*}
			&&(\sqrt{3x+1}-4)+(1-\sqrt{6-x})+3x^{2}-14x-5=0\\
			&\Leftrightarrow&\dfrac{3(x-5)}{\sqrt{3x+1}+4}+\dfrac{x-5}{1+\sqrt{6-x}}+(3x+1)(x-5)=0\\
			&\Leftrightarrow&(x-5)\left( \dfrac{3}{\sqrt{3x+1}+4}+\dfrac{1}{1+\sqrt{6-x}}+(3x+1)\right)=0.
		\end{eqnarray*}
		Vì $ -\dfrac{1}{3}\leq x\leq 6 $ nên $ \dfrac{3}{\sqrt{3x+1}+4}+\dfrac{1}{1+\sqrt{6-x}}+(3x+1)>0 $. Do đó phương trình có nghiệm duy nhất $ x=5 $.
	}
\end{bt}

\begin{dang}{Phương trình có nghiệm kép}
	Giả sử $ x=x_{0} $ là nghiệm kép thì nhân tử chung trong phương trình là $ (x-x_{0})^{2} $.
\end{dang}

\begin{bt}%[Dương BùiĐức]
	Giải phương trình $\sqrt{2x-1}+x^{2}-3x+1=0$.
	\loigiai{
		Điều kiện xác định $ x\geq \dfrac{1}{2} $.\\
		Ta có
		\[
		\sqrt{2x-1}-x+x^{2}-2x+1=0\Leftrightarrow (x-1)^{2}\left[\dfrac{-1}{\sqrt{2x-1}+x}+1\right]=0.
		\]
		Xét $ \dfrac{-1}{\sqrt{2x-1}+x}=0\Leftrightarrow \sqrt{2x-1}+x=-1 $ (vô lí).\\
		Vậy phương trình có nghiệm duy nhất $ x=1 $.
	}
\end{bt}

\begin{bt}%[Dương BùiĐức]
	Giải phương trình $9(x+1)^{2}=(3x+7)(1-\sqrt{3x+4})^{2}$.
	\loigiai{
		Điều kiện xác định $ x\geq -\dfrac{4}{3} $.\\
		Ta có
		\begin{eqnarray*}
			&&9(x+1)^{2}=(3x+7)(1-\sqrt{3x+4})^{2}\\
			&\Leftrightarrow&9(x+1)^{2}(1+\sqrt{3x+4})^{2}=9(3x+7)(x+1)^{2}\\
			&\Leftrightarrow&9(x+1)^{2}\left[(1+\sqrt{3x+4})^{2}-(3x+7)\right]=0.
		\end{eqnarray*}
		Xét $ (1+\sqrt{3x+4})^{2}-(3x+7)=0\Leftrightarrow 2\sqrt{3x+4}-2=0\Leftrightarrow 3x+4=1\Leftrightarrow x=-1 $.\\
		Vậy phương trình có nghiệm $ x=\pm 1 $.
	}
\end{bt}

\begin{dang}{Phương trình có hai nghiệm phân biệt}
	Giả sử $ x_{1},\ x_{2} $ là hai nghiệm phân biệt thì nhân tử chung trong phương trình là $ (x-x_{1})(x-x_{2}) $.\\
	Lượng liên hợp thường có dạng $ mx+n $, muốn tìm ra giá trị $ m,\ n $ thì ta thay giá trị $ x_{1},\ x_{2} $ vào từng căn thức, khi đó sẽ có mối quan hệ giữa $ m $ và $ n $.
\end{dang}

\begin{bt}%[Dương BùiĐức]
	Giải phương trình $3\sqrt[3]{x^{2}}+\sqrt{x^{2}+8}-2=\sqrt{x^{2}+15}$.
	\loigiai{
		Nhận xét: $x=\pm 1$ là nghiệm của phương trình nên ta sẽ tạo ra nhân tử chung là $ x^{2}-1 $.
		\begin{eqnarray*}
			&&3\left(\sqrt[3]{x^{2}}-1\right)+\left(\sqrt{x^{2}+8}-3\right)=\sqrt{x^{2}+15}-4\\
			&\Leftrightarrow&(x^{2}-1)\left[\dfrac{1}{\sqrt[3]{x^{4}}+\sqrt[3]{x^{2}}+1}+\dfrac{1}{\sqrt{x^{2}+8}+3}-\dfrac{1}{\sqrt{x^{2}+15}+4}\right]=0.
		\end{eqnarray*}
	Vì $ \sqrt{x^{2}+8}+3<\sqrt{x^{2}+15}+4 $ nên $ \dfrac{1}{\sqrt[3]{x^{4}}+\sqrt[3]{x^{2}}+1}+\dfrac{1}{\sqrt{x^{2}+8}+3}-\dfrac{1}{\sqrt{x^{2}+15}+4}>0 $. Do đó phương trình có nghiệm $ x=\pm 1 $.
	}
\end{bt}

\begin{bt}%[Dương BùiĐức]
	Giải phương trình $\sqrt{2x^{2}-x+3}-\sqrt{21x-17}+x^{2}-x=0$.
	\loigiai{
		Điều kiện $ x\geq \dfrac{17}{21} $.\\
		Nhận xét: phương trình có hai nghiệm $x=1,\ x=2$ nên ta sẽ tạo nhân tử chung $ (x-1)(x-2)=x^{2}-3x+2 $.\\
		Tìm lượng liên hợp cho $\sqrt{2x^{2}-x+3}$ ở dạng $ mx+n $: 
		\begin{itemize}
			\item Với $x=1\Rightarrow m+n=2$;
			\item Với $x=2\Rightarrow 2m+n=3$.
		\end{itemize}
	Giải hệ phương trình với ẩn $ m,\ n $ ta được $ m=1,\ n=1 $.\\
	Do đó
	\begin{eqnarray*}
		&&\sqrt{2x^{2}-x+3}-\sqrt{21x-17}+x^{2}-x=0\\
		&\Leftrightarrow&(\sqrt{2x^{2}-x+3}-x-1)-(\sqrt{21x-17}-3x+1)+(x^{2}-3x+2)=0\\
		&\Leftrightarrow&(x^{2}-3x+2)\left( \dfrac{1}{\sqrt{2x^{2}-x+3}+x+1}+\dfrac{9}{\sqrt{21x-17}+3x-1}+1\right)=0.
	\end{eqnarray*}
Vì $ \dfrac{1}{\sqrt{2x^{2}-x+3}+x+1}+\dfrac{9}{\sqrt{21x-17}+3x-1}+1>0 $ nên phương trình có hai nghiệm $x=1,\ x=2$.
	}
\end{bt}

\begin{bt}
Giải các phương trình sau
	\begin{enumEX}{2}
		\item $\sqrt[3]{2-x}+\sqrt{x-1}=1$.
		\item $\sqrt{5-x}+\sqrt{x-1}=-x^{2}+2x+1$.
		\item $\sqrt{x^{2}+x+1}-\sqrt{-2x^{2}+3x-1}=\sqrt{3}$.
		\item $\sqrt{4x+1}-\sqrt{3x-2}=x-5$.
		\item $\sqrt{2x^{2}+16x+18}+\sqrt{x^{2}-1}=2x+4$.
		\item $\dfrac{3x}{\sqrt{3x+10}}=\sqrt{3x+1}-1$.
		\item $\sqrt{x^{2}-3x+3}+\sqrt{x^{2}-3x+6}=3$.
		\item $\sqrt{2x^{2}+3x+5}+\sqrt{2x^{2}-3x+5}=3x$.
		\item $\sqrt{4x+1}-\sqrt{3x-2}=\dfrac{x+1}{3}$.
		\item $\sqrt[3]{x^{2}-1}+x=\sqrt{x^{3}-1}$.
		\item $\sqrt{3x+1}+\sqrt{5x+4}=3x^{2}-x+3$.
		\item $\sqrt{4x+1}-\sqrt{3x-2}=\dfrac{x+5}{5}$.
		\item $\sqrt{x^{2}-x+1}+\sqrt{x^{2}+x+1}=2$.
	\end{enumEX}
\end{bt}
