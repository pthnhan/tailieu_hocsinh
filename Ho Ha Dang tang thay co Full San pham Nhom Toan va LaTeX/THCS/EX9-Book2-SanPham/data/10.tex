		Ta có $P(n)=n^4-14n^3+71n^2-154n+120=(n-2)(n-3)(n-4)(n-5)$ là tích $4$ số tự nhiên liên tiếp chia hết cho $3;8$ mà $3;8 \geq 1$ nên chia hết cho $24$  (đpcm)  }

\begin{ex}[Đề thi vào 10 chuyên Thái Nguyên, 2017]
	Cho số tự nhiên 
	$$A=\underbrace {\overline {777\dots 7} }_{\text{n chữ số 7}}-18+2n \,\, \text{(với }n \in \mathbb{N}, n \ge 2)$$
Chứng minh rằng $A$ chia hết cho $9.$ 
	\loigiai{
		Với số tự nhiên $ a\in \mathbb{N^*} $, ta đặt $ S(a) $ là tổng các chữ số của $ a $. Khi đó ta luôn có $ a-S(a) $ chia hết cho $9$.
		Ta có: 
		$$ A=\underset{n \text{ chữ số 7}}{\underbrace{\overline{777\dots7}}}-18+2n=7.\underset{n \text{ chữ số 1}}{\underbrace{\overline{111\dots1}}}-18+2n=7.(\underset{n \text{ chữ số 1}}{\underbrace{\overline{111\dots1}}}-n) +9(n-2).$$
		Tổng các chữ số của $ \underset{n \text{ chữ số 1}}{\underbrace{\overline{111\dots1}}} $ là $ n $ nên ta có $ \underset{n \text{ chữ số 1}}{\underbrace{\overline{111\dots1}}}-n $ chia hết cho 9. \\Vì vậy $ 7(\underset{n \text{ chữ số 1}}{\underbrace{\overline{111\dots1}}}-n) $ chia hết cho 9.\\
		Hiển nhiên $ 9(n-2) $ chia hết cho 9.\\
		Do đó $ A \  \vdots \ 9.$
	}
\end{ex}
\begin{ex}[Thi vào 10, Chuyên Bạc Liêu, 2017]
	\hfill
Cho $n=2018.2017^{2016}-11^{2017}-6^{2016}$. Chứng minh $n$ chia hết cho $17$.
	\loigiai{
		 Ta có 
			$2017 \equiv 11  \mod 17 \Rightarrow 2017^{2016} \equiv 11^{2016} \mod 17$.\\
			Mặt khác: $2018\equiv 12 \mod 17$.\\
			Suy ra $2018.2017^{2016}\equiv 12.11^{2016} \mod 17$.\\
			Hay $2018.2017^{2016} \equiv (11+1).11^{2016}\equiv 11^{2017}+11^{2016} \mod 17$.\\
			Suy ra $2018.2017^{2016}-11^{2017} \equiv 11^{2016} \mod 17.$\\
			Suy ra $n=2018.2017^{2016}-11^{2017}-6^{2016} \equiv 11^{2016}-6^{2016} \mod 17$.\\
			Ta lại có:
			\begin{itemize}
				\item $11^{2} \equiv 2 \mod 17$, suy ra $11^{2016}=(11^2)^{1008} \equiv 2^{2008} \mod 17.$
				\item $6^{2} \equiv 2 \mod 17$, suy ra $6^{2016}=(6^2)^{1008} \equiv 2^{2008} \mod 17$.
			\end{itemize}
			Do đó ta có $n \equiv 2^{1008}-2^{1008} =0 \mod 17$.\\
			Vậy $n$ chia hết cho $17$.}
	\end{ex}

		Cho $a$, $b$, $c$ là các số nguyên. Chứng minh rằng nếu $a^{2014}+b^{2015}+c^{2016}$ chia hết cho $6$ thì $a^{2016}+b^{2017}+c^{2018}$ chia hết cho $6$.

			Ta có: $B-A=a^{2014}(a^2-1)+b^{2015}(b^2-1)+c^{2016}(c^2-1)$.\\
			Nhận thấy rằng $a(a^2-1)=(a-1)a(a+1)\  \vdots \ 6\Rightarrow a^{2014}(a^2-1)\  \vdots \ 6$.\\
			Tương tự ta có $b^{2014}(b^2-1)\  \vdots \ 6$, $c^{2014}(c^2-1)\  \vdots \ 6$ nên ta suy ra $B-A\  \vdots \ 6$.\\
			Mà $A\ \vdots \ 6$ nên $B\  \vdots \ 6$.}
	\end{ex}
\begin{ex}[Đề thi vào 10, Phổ thông năng khiếu Hồ Chí Minh, 2017]
	Cho $x, y$ là hai số nguyên với $x>y>0$.
	\begin{enumerate}    
		\item Chứng minh rằng nếu $x^3-y^3$ chia hết cho $3$ thì $x^3-y^3$ chia hết cho $9$.
		\item Tìm tất cả những giá trị $k$ nguyên dương sao cho $x^k – y^k$ chia hết cho $9$ với mọi $x, y$ mà $xy$ không chia hết cho $3$.
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
			\item Ta có $x^3-y^3=(x-y)^3-3xy(x-y)$ chia hết cho $3$ nên $(x-y)^3$ chia hết cho $3$. Mà $3$ là số nguyên tố nên $x-y$ chia hết cho $3$. Suy ra $\heva{& (x-y)^3 \  \vdots \ 9 \\ & 3xy(x-y) \  \vdots \ 9} \Rightarrow x^3-y^3 \  \vdots \ 9$.
			\item  Cho $x=2, y=1$ suy ra $xy$ không chia hết cho $3$.\\
			Ta có $x^k - y^k=2^k-1 \  \vdots \ 9 \Rightarrow 2^k -1 \  \vdots \ 3$.\\
			Do $2 \equiv -1 (\text{mod} 3) \Rightarrow 2^k-1 \equiv  (-1)^k-1 (\text{mod} 3) $ nên $k$ là số chẵn. \\
			Ta chứng minh $k=6n \,\,\, (n \in \mathbb{N}^*)$.\\
			Thật vậy, \\
			-	Với $k=6n+2 \Rightarrow 2^k -1 = 2^{6n+2}-1 \equiv 3 (\text{mod} 9) $. Suy ra $k=6n+2$ không thỏa mãn.\\
			-	Với $k=6n+4 \Rightarrow 2^k -1 = 2^{6n+4}-1 \equiv 6 (\text{mod} 9) $. Suy ra $k=6n+4$ không thỏa mãn.\\
			Vậy $k=6n$. \\
			Ta lại có $x^k – y^k=x^{6n} – y^{6n}=(x^6)^n – (y^6)^n \  \vdots \ (x^6 – y^6)$. Do $xy$ không chia hết cho $3$ nên cả $x, y$ đều không chia hết cho $3$.\\
			-	{\bf TH1.} $x \equiv y (\text{mod} 3)  \Rightarrow x^3 –y^3 \  \vdots \ 3$. Theo câu (1) suy ra $x^3 –y^3 \  \vdots \ 9 \Rightarrow x^k-y^k \  \vdots \ 9$.\\
			-	{\bf TH2.} $x, y$ không cùng số dư khi chia chia cho $3$. \\
			Không mất tính tổng quát, giả sử $\heva{& x = 3a+1 \\& y=3b+2}$.\\
			Ta có $x^3+y^3=(3a+1)^3+(3b+2)^3=27a^3+27a^2+9a+27b^3+27b^2+9b+9 \  \vdots \ 9$.\\
			Suy ra $x^6-y^6=(x^3-y^3)(x^3+y^3) \  \vdots \ 9 \Rightarrow x^k-y^k \  \vdots \ 9$.\\
			Vậy tập tất cả các số thỏa mãn đề bài là $k=6n$ với $n$ là số tự nhiên.
			
		\end{enumerate}
	}
\end{ex}
\begin{ex}[Đề thi vào 10, SoGiaoDucHaNoi-ChuyenTin, 2017]

	Cho các số nguyên dương $a,b$ thỏa mãn $a^2+b^2=c^2$. Chứng minh $ab$ chia hết cho $a+b+c$.
	\loigiai
	{
		 Đặt $t=a+b+c \\ \Rightarrow t^2=(a+b+c)^2=a^2+b^2+c^2+2ab+2bc+2ca=2c^2+2(ab+bc+ca) \Rightarrow t \  \vdots \ 2$\\
			Mặt khác: $a^2+b^2=c^2 \Leftrightarrow a^2+b^2=(t-a-b)^2\\
			\Leftrightarrow a^2+b^2=t^2+b^2+c^2-2ta-2tb+2ab\\ \Leftrightarrow 0=t^2-2ta-2tb+2ab\\ \Leftrightarrow ab=\dfrac{-t^2+2ta+2tb}{t}=t\left(\dfrac{-t+2a+2b}{2}\right)=(a+b+c)\left(\dfrac{-t+2a+2b}{2}\right)$\\
			Vậy $ab$ chia hết cho $a+b+c$.
	}
\end{ex}
\begin{ex}[Đề thi vào 10, Chuyên Trần Hưng Đạo, Bình Thuận, 2017]
	Cho $n$ số nguyên $a_1,a_2,\ldots,a_n$ thỏa mãn $S=a_1+a_2+\ldots+a_n$ chia hết cho $6$. Chứng minh $P=a_1^3+a_2^3+\ldots+a_n^3$ cũng chia hết cho $6$.
	\loigiai{
		Ta có 
		\begin{eqnarray*}
			P-S &=&(a_1^3-a_1)+(a_2^3-a_2)+\ldots+(a_n^3-a_n)\\
			&=&(a_1-1)a_1(a_1+1)+(a_2-1)a_2(a_2+1)+\ldots+(a_n-1)a_n(a_n+1).
		\end{eqnarray*}
		Mỗi số hạng có dạng $(a_k-1)a_k(a_k+1)$ đều chia hết cho $6$ (là tích $3$ số nguyên liên tiếp), do đó $P-S$ chia hết cho $6$, mà $S$ chia hết cho $6$ nên $P$ chia hết cho $6$.
	}
\end{ex}
\begin{ex}[Đề thi vào 10, Sở giáo dục Vĩnh Long, 2017]
Tìm tất cả số nguyên $x$ sao cho $2x^2 + x - 2$ chia hết cho $x^2 + 1$.
		
	\loigiai
	{
		
		 Ta có: $\dfrac{{2{x^2} + x - 2}}{{{x^2} + 1}} = 2 + \dfrac{{x - 4}}{{{x^2} + 1}}$\\
			Xét $x=4\Rightarrow 2x^2+x-2$ chia hết cho $x^2+1$\\
			Xét $x\neq 4$ để $(x-4)\  \vdots \ (x^2+1)\Rightarrow |x-4|\geq x^2+1$\\
			$ \Leftrightarrow \left[ {\begin{array}{*{20}{c}}
				{x - 4 \ge {x^2} + 1}\\
				{x - 4 \le  - {x^2} - 1}
				\end{array}} \right. \Leftrightarrow \left[ {\begin{array}{*{20}{c}}
				{{x^2} - x + 5 \le 0\left( 1 \right)}\\
				{{x^2} + x - 3 \le 0\left( 2 \right)}
				\end{array}} \right.$\\
			Ta nhận thấy $(1)$ vô nghiệm còn $(2)$ kết hợp với điều kiện $x$ là số nguyên suy ra $x\in \{-2,-1,0,1\}$\\
			Thử lại ta nhận giá trị $x=0$.\\
			Vậy $x\in \{0;4\}$ thì $2x^2+x-6$ chia hết cho $x^2+1$
	}
\end{ex}
\begin{ex}[Đề thi vào 10 chuyên, Sở giáo dục Bắc Giang, 2016]%[9D5G2]%[9D5G5]
	 Tìm số tự nhiên có bốn chữ số biết rằng khi chia số đó cho $120$ được số dư là $88$ và khi chia cho $61$ được số dư là $39$. 
		
	\loigiai{
	Gọi số tự nhiên có 4 chữ số phải tìm là $x, 1000\le x\le 9999.$\\ 
			Theo giả thiết ta có $x=120n+88=61k+39$ với $n, k\in \mathbb{N}$.\\
			Từ đó ta có $120n-60k=k-49$ suy ra $k-49$ chia hết cho 60.\\
			Mặt khác ta có:\\ 
			$ 1000\le x\le 9999 \Rightarrow 1000 \le 61k+39 \le 9999  \Rightarrow \dfrac{961}{61}\le k\le \dfrac{9960}{61}$\\
			$\Rightarrow 15<k<164  \Rightarrow -34<k-49<115.$\\
			Mà $k-49$ chia hết cho 60 nên $k-49$ chỉ có thể là $0; 60$.\\
			Với $k=49 \Rightarrow x=3028$ (chia $120$ dư $28$).\\
			Với $k=109 \Rightarrow x=6688$ (chia $120$ dư $88$).\\
			Vậy số cần tìm là $6688$. 
	}
\end{ex}
\begin{ex}[Đề thi vào 10 chuyên Bình Phước 2016]

\begin{ex}[Đề thi vào 10, Chuyên Nguyễn Trãi, Hải Dương, v2 , 2016]
 Tìm dạng tổng quát của số nguyên dương $n$ biết $M=n.4^n+3^n $  chia hết cho $7$. 
	
	\loigiai{
		\begin{itemize}
			\item $n=2k$ ($k$ nguyên dương): $M = 2k.16^k+9^k.$ Do $16^k, 9^k$ có cùng số dư với $2^k$ khi chia cho $7,$ nên $M$ cùng dư với $2k.2^k+2^k = (2k+1)2^k$ khi chia cho $7.$  Do đó, 
			$M \  \vdots \ 7 \Leftrightarrow 2k+1 \  \vdots \ 7 \Leftrightarrow k = 7q+3,$
			
			suy ra $n=14q+6$ ($q \in \mathbb{N}$). 
			\item $n=2k+1$ ($k$ nguyên dương): $M=4(2k+1).16^k+3.9^k$ cùng dư với $(k+4).2^k+3.2^k = (k+7)2^k$ khi chia cho $7$. 
			Do đó, 
			$M \  \vdots \ 7 \Leftrightarrow k+7 \  \vdots \ 7 \Leftrightarrow k = 7p,$
			
			suy ra $n=14p+1$ ($n=14p+1$.)
		\end{itemize}
		
		Vậy $n=14q+6$ hoặc $n = 14p+1$ ($p,q \in \mathbb{N}.$)
	}
\end{ex}

\begin{ex}[Đề thi vào 10, Chuyên Sư Phạm Hà Nội, 2016]
	Cho $S$ là tập hợp các số nguyên dương $n$ có dạng $n=x^2+3y^2$ trong đó $x,y$ là các số nguyên. Chứng minh rằng:
	\begin{enumerate}
		\item Nếu $a,b \in S$ thì $ab \in S$.
		\item Nếu $N \in S$ và $N$ chẵn thì $N$ chia hết cho $4$ và $\dfrac{N}{4} \in S$.
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
			\item Nếu $a,b\in S$ thì $a=m^2+3n^2$ và $b=p^2+3q^2$. \\
			Ta có  $ab=(m^2+3n^2)(p^{2} + 3q^{2}) = (mp + 3nq)^{2} + 3(mq - pn)^{2}$.\\
			Vậy $ab\in S$.
			\item Nếu $N\in S$ thì $N = x^{2} + 3y^{2}$. Dĩ nhiên nếu $2\mid x$ thì từ $N$ chẵn, ta có $2\mid y$, trường hợp này hiển nhiên $\dfrac{N}{4} \in {S}$.\\
			Xét $x, y$ đều lẻ, ta có $x^{2} - 9y^{2} \equiv 1 - 9 \equiv 0\pmod{8}$, do đó $4\mid x - 3y$ hoặc $4\mid x + 3y$. Giả sử $4\mid x - 3y$ (trường hợp còn lại chứng minh tương tự).
			Khi đó: $$\dfrac{N}{4} = \left(\dfrac{x - 3y}{4}\right)^{2} + 3\left(\dfrac{x + y}{4}\right)^{2}$$
			Vậy $\dfrac{N}{4}\in S$
		\end{enumerate}
	}
\end{ex}
\begin{ex}[Đề thi vào 10, PTNK, 2016]
	Cho $x$, $y$ là hai số nguyên dương mà $x^2+y^2+10$ chia hết cho $xy$.
	\begin{enumerate}    
		\item Chứng minh rằng $x$, $y$ là hai số lẻ và nguyên tố cùng nhau.
		\item Chứng minh rằng $k=\dfrac{x^2+y^2+10}{xy}$ chia hết cho $4$ và $k\geq 12$.
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
			\item Giả sử $x$ là số chẵn. Vì $x^2+y^2+10$ chia hết cho $xy$ nên $y$ là số chẵn. Từ đó ta có, $x^2+y^2$ chia hết cho $4$ và $x^2+y^2+10$ chia hết cho $4$, suy ra $10$ chia hết cho $4$ (vô lí). Vậy $x$, $y$ là hai số lẻ. Bây giờ, ta giả sử $(x,y)=d$, $d\neq 1$. Suy ra $d^2\mid(x^2+y^2)$ và theo giả thiết thì $d^2\mid 10$ (vô lí). Tóm lại, $x$, $y$ là hai số lẻ và $(x,y)=1$.
			
			\item Đặt $x=2a+1$, $y=2b+1$ suy ra $k=\dfrac{4(a^2+b^2+a+b+3)}{(2a+1)(2b+1)}$. Do đó, $k$ chia hết cho $4$. Giả sử $x$ chia hết cho $3$. Lúc này, $3\mid(y^2+10)$ (vô lí). Vậy $x$ và $y$ cùng không chia hết cho 3. Ta có: $3\mid x^2+y^2+10=kxy$ suy ra $k$ chia hết cho 3. Từ đó ta có $k\geq 12$.
		\end{enumerate}
	}
\end{ex}	
\subsection{Số nguyên tố và hợp số}
   \begin{ex}[Đề thi vào chuyên 10, Sở giáo dục Hà Nội, 2017]
   	\hfill 
   	\begin{enumerate}    
   		\item Cho $p$ là một số nguyên tố lớn hơn $3$. Chứng minh $2017 -p^2$ chia hết cho $24$.
   		\item Cho $a,b,c$ là các số nguyên dương. Chứng minh $a+b+2\sqrt{ab+c^2}$ không phải là số nguyên tố. 
   	\end{enumerate}
   	\loigiai
   	{
   		\begin{enumerate}
   			\item Vì $p$ là số nguyên tố lớn hơn $3$ nên $p$ là số nguyên tố lẻ. Đặt $p=2k+1$. \\
   			Ta có $2017-p^2=2017-(2k+1)^2=2016-4k(k+1)$. \\
   			Một trong hai số $k$, $k+1$ là số chẵn nên $4k(k+1)  \  \vdots \  8$. Do đó $(2017-p^2)  \  \vdots \  8$ $(1)$.
   			\\ Ta có $2017-p^2=(44^2-p^2) +81 =(44-p)(44+p)+81$. 
   			\\ Vì $p$ là số nguyên tố lớn hơn $3$ nên $p$ không chia hết cho $3$.
   			\begin{itemize}
   				\item[-] Nếu $p$ chia $3$ dư $1$ thì $44+p$ chia hết cho $3$.

		Vì thế từ $(3)$ suy ra: $(a+1)(k+1-ka)\geq 0$.\\
		Lại do $a>0$ nên suy ra: $k+1-ka\geq 0\Rightarrow 1\geq k(a-1)$.\\
		Vì $a-1>0,\ k>0$ nên $1\geq k(a-1)\geq 0$ và $k(a-1)\in\mathbb{N^*}$\\
		$\Rightarrow \left[\begin{array}{l}
		k(a-1)=0\\k(a-1)=1
		\end{array}\right.\Leftrightarrow \left[\begin{array}{l}
		a=1\\ \begin{cases} a=2\\ k=1\end{cases}
		\end{array}\right.$\\
		Với $a=1$. Thay vào $(3)$ ta được:
		\begin{align*}
			& (m-1)(b-1)=2\\
			\Leftrightarrow\ &\left[\begin{array}{l}
				\left\{\begin{array}{l}
					m-1=2\\ b-1=1
				\end{array}\right.\\
				\left\{\begin{array}{l}
					m-1=1\\ b-1=2
				\end{array}\right.
			\end{array}\right. \Leftrightarrow\left[\begin{array}{l}
				b=2\Rightarrow ka^2=5\Rightarrow a=1\\
				b=3\Rightarrow ka^2=5\Rightarrow a=1
			\end{array}\right.
		\end{align*}
		Vậy, trường hợp này ta được hai cặp $a=1,\ b=2$ và $a=1,\ b=3$.\\
		Với $a=2$ và $k=1$. Thay vào $(3)$ ta có:
		$$(m-1)(b-1)=0\Leftrightarrow \left[\begin{array}{l}
		b=1\\ m=1
		\end{array}\right.$$
		Khi $b=1$, ta được $a=2,\ b=1$.\\
		Khi $m=1$. Từ $(1)$ suy ra $a+k=b\Rightarrow b=3$. Khi đó $a=2,\ b=3$.\\
		Vậy có bốn cặp số $(a;b)$ thỏa mãn là: $(1;2),\ (1;3),\ (2;3),\ (2;1)$.
	}
\end{bt}

			&=1+(n+1)(n+2)(n+3)(n+4)= \left(n^2+5n+5\right)^2.

	Tìm tất cả các cặp số nguyên $(x,y)$ thỏa mãn $1+x+x^2+x^3+x^4=y^2$.
	\loigiai
	{
		Tìm tất cả các cặp số nguyên $(x,y)$ thỏa mãn $1+x+x^2+x^3+x^4=y^2$\\
		Ta có $4+4x+4x^2+4x^3+4x^4=(2y)^2$
		$\Rightarrow 4+4x+4x^2+4x^3+4x^4=(2x^2+x)^2+2x^2+(x+2)^2$ \\
		$\Rightarrow (2x^2+x)^2<(2y)^2$.\\
		Với $x=0$ ta được $y=1$ ta được cặp nghiệm nguyên $(0;1), (0;-1)$.\\
		Với $x \ne 0$, ta có $4+4x+4x^2+4x^3+4x^4 \leq 4+4x+4x^2+4x^3+4x^4 +4x^2=(2x^2+x+2)^2$.\\
		Do đó $(2x^2+x)^2 \leq (2y)^2 \leq (2x^2+x+2)^2$\\
		$\Rightarrow (2y)^2=(2x^2+x+1)^2$\\
		$\Rightarrow 4+4x+4x^2+4x^3+4x^4=(2x^2+x+1)^2$ \\
		$\Rightarrow x^2-2x-3=0$ $\Rightarrow x=-1$ hoặc $x=3$.
		Do đó, ta được các cặp nghiệm nguyên là $(-1,1); (-1,-1); (3,11); (3,-11)$.\\
		Vậy các cặp nghiệm nguyên cần tìm là: $(0,1); (0,-1); (-1,1); (-1,-1); (3,11); (3,-11)$.  
	}

	Tìm các số nguyên  $x$, $y$  sao cho $x^{3}y-x^{3}-1=2x^{2}+2x+y$.
	\loigiai
	{
		\begin{align*}
			&x^3y - x^3 - 1 = 2x^2 + 2x + y \\
			\Leftrightarrow& (x^3 - 1)y = x^3 + 1 + 2x(x + 1)\\
			\Leftrightarrow& (x - 1)(x^2 + x + 1)y = (x + 1)(x^2 - x + 1) + 2x(x + 1)\\                         
			\Leftrightarrow& (x - 1)(x^2 + x + 1)y = (x + 1)(x^2 + x + 1).\tag{1}
		\end{align*}.
		Vì $x^2 + x + 1 = \left( x + \dfrac{1}{2} \right)^2 + \dfrac{3}{4} > 0,\forall x \in \mathbb{Z}$ nên 
		$(1) \Leftrightarrow (x - 1)y = x + 1$.\hfill$(2)$\\
		Ta thấy $x = 1$ không thỏa phương trình $(2)$. Xét $x \in \mathbb{Z}$ và $x \ne 1$. Khi đó:
		\[(2) \Leftrightarrow y = \dfrac{x + 1}{x - 1} = 1 + \dfrac{2}{x - 1}.\]
		Ta có $y \in \mathbb{Z}$ khi và chỉ khi $\left( x - 1 \right)$ là ước của $2$.

	}

	\loigiai{
		\begin{itemize}
			\item Nếu $\heva{& b=0 \\ & c=0}$ thì $n= \overline{a00} = 100 \cdot a.$\\
			Do đó $\dfrac{n}{a+b+c}= \dfrac{100a}{a}=100.$
			\item Nếu $b\cdot c \neq 0$ thì $b+c>0 \Rightarrow b+c \geq 1$ do $b+c \in \mathbb{N}^* \Rightarrow a+b+c \geq a+1.$ Như vậy $$\dfrac{n}{a+b+c} < \dfrac{100(a+1)}{a+b+c} \leq  \dfrac{100(a+1)}{a+1}=100.$$ 
		\end{itemize}
		Đẳng thức xảy ra khi $n$ là số có $3$ chữ số và tròn trăm.
	}
\end{bt}
\begin{bt}
	Cho $a, b, c \in \mathbb{N}^*$ thỏa mãn điều kiện $a^2+b^2=c^2(1+ab).$ Chứng minh rằng $a\geq c$ và $ b\geq c.$
	\loigiai{
		Từ giả thiết ta có $$ a^2 -bc^2 \cdot a +b^2-c^2=0, \quad (1).$$
		Xét phương trình bậc hai ẩn $x$ sau: $x^2-bc^2x+b^2-c^2=0, \quad (2).$\\
		Từ (1), suy ra $a$ là nghiệm của phương trình (2). Theo định lý Vi-et thì phương trình (2) còn nghiệm $a'$ nữa thỏa mãn $$ \heva{& a+a'=bc^2, (3) \\ & aa'=b^2-c^2, (4)}$$
		Do $a, b, c $ nguyên, nên từ (3) suy ra $a'$ cũng là số nguyên.\\ Mặt khác vì $a'$  là nghiệm của (2) nên ta có $$ a'^2-bc^2a'+b^2-c^2=0 \Rightarrow c^2(ba'+1)=b^2+a'^2 >0.$$
		Từ đó, ta có $$ ba'+1>0 \Rightarrow a'> \dfrac{-1}{b}.$$
		Do $b$  là số nguyên dương nên $ - \dfrac{1}{b} \geq -1 \Rightarrow a' >-1.$ Vì $a'$ là số nguyên nên $a' \geq 0.$\\
		Kết hợp với $a>0 \Rightarrow aa' \geq 0.$ Từ (4) suy ra $b \geq c.$ Do $a$ và $b$ có vai trò bình đẳng nên ta cũng có $a \geq c.$ Điều phải chứng minh. 
		
	}
\end{bt}
\subsection{Bất đẳng thức phân số và căn bậc hai}
\begin{bt}
	Chứng minh bất đẳng thức $\left( 1+\dfrac{1}{2}\right) \left( 1+ \dfrac{1}{2^2}\right) \cdots \left( 1+ \dfrac{1}{2^n}\right) <3, \quad \forall n \in \mathbb{N}^*$
	\loigiai{
		Đặt vế trái của BĐT là $A$.\\
		\begin{itemize}
			\item Với $n=1$ thì $A=1+ \dfrac{1}{2}= \dfrac{3}{2}<3$ (BĐT đã cho đúng).
			\item Với $n \geq 2$ thì $A= \dfrac{3}{2} \cdot \left( 1+ \dfrac{1}{2^2}\right) \cdots \left( 1+ \dfrac{1}{2^n}\right).$\\
			Nhận xét: Với $ m \in \mathbb{N}^*, m \geq 2$ thì $$ 1+\dfrac{1}{2^m}< \dfrac{1}{2} \cdot \dfrac{2^m-1}{2^{m-1}-1} \quad (*).$$
			Thật vậy xét hiệu $$ \dfrac{1}{2} \cdot \dfrac{2^m-1}{2^{m-1}-1} -\left( 1+ \dfrac{1}{2^m} \right) = \dfrac{2^m-1}{2^m-2}-1- \dfrac{1}{2^m} =\dfrac{1}{2^m (2^{m-1}-1)}>0 \quad \text{ vì} m \geq 2.$$
			Do đó BĐT (*) được chứng minh.\\
			Áp dụng BĐT (*) với $m =2,3, ..., n$ ta được
			$$ A< \dfrac{3}{2} \left( \dfrac{1}{2} \cdot \dfrac{2^2-1}{2-1}\right) \left( \dfrac{1}{2} \cdot  \dfrac{2^3-1}{2^2-1} \right) \cdots \left( \dfrac{1}{2} \cdot \dfrac{2^n-1}{2^{n-1}-1} \right) = 3 \cdot \dfrac{2^n-1}{2^n} < 3.$$
			Điều phải chứng minh.
		\end{itemize}
	}
\end{bt} 
\begin{note}
	Ta có thể sử dụng phương pháp quy nạp để chứng minh BĐT trên.
\end{note}

	Chứng minh rằng với mọi số nguyên dương $n$, ta có $$\dfrac{1}{n+1}+\dfrac{1}{n+2}+\dfrac{1}{n+3}+...+\dfrac{1}{3n+1}>1.$$
	\loigiai{
		Trong tổng $\dfrac{1}{n+1}+ \dfrac{1}{n+2}+ \dfrac{1}{n+3}+...+\dfrac{1}{3n+1}$ có $2n+1$ số hạng. Ghép thành $n$ cặp số $\left\lbrace \dfrac{1}{2n+1-k}\mbox{ và } \dfrac{1}{2n+1-k}\right\rbrace $.\\
		Ta có $\dfrac{1}{2n+1}+\dfrac{1}{2n+1+k}=\dfrac{4n+2}{(2n+1)^2-k^2}>\dfrac{4n+2}{(2n+1)^2}=\dfrac{2}{2n+1}$.\\
		Do đó, $$\dfrac{1}{n+1}+\dfrac{1}{n+2}+\dfrac{1}{n+3}+...+\dfrac{1}{3n+1}=\sum\limits^{n}_{k=1}{\left( \dfrac{1}{2n+1-k}+\dfrac{1}{2n+1+k}\right) }+\dfrac{1}{2n+1}>\dfrac{2n}{2n+1}+\dfrac{1}{2n+1}=1.$$
	}
\end{bt} 


\begin{bt}[Đề thi vào 10, THPT Chuyên Hà Nam, năm học 2016-2017][9D5G1]
	Cho các số thực dương $x$, $y$, $z$ thoả mãn $\dfrac{1}{x+y}+\dfrac{1}{y+z}+\dfrac{1}{z+x}=12$. Chứng minh rằng
	\begin{align*}
		\dfrac{1}{2x+3y+3z} +\dfrac{1}{3x+2y+3z} + \dfrac{1}{3x+3y+2z} \le 3.
	\end{align*}
	Dấu đẳng thức xảy ra khi nào?
	\loigiai{
		Áp dụng bất đẳng thức Cauchy-Schwarz, ta có
		\begin{align*}
			&\dfrac{1}{x+y}+\dfrac{1}{x+z}+\dfrac{1}{y+z}+\dfrac{1}{y+z}\ge \dfrac{16}{2x+3y+3z}\\
			&\dfrac{1}{y+z}+\dfrac{1}{y+x}+\dfrac{1}{z+x}+\dfrac{1}{z+x}\ge \dfrac{16}{3x+2y+3z}\\
			&\dfrac{1}{z+x}+\dfrac{1}{z+y}+\dfrac{1}{x+y}+\dfrac{1}{x+y}\ge \dfrac{16}{3x+3y+2z}.
		\end{align*}
		Cộng các bất đẳng thức trên theo vế ta được
		\begin{align*}
			\dfrac{16}{2x+3y+3z} +  \dfrac{16}{3x+2y+3z} + \dfrac{16}{3x+3y+2z} \le 4\left( \dfrac{1}{x+y} + \dfrac{1}{y+z} + \dfrac{1}{z+x}\right)
		\end{align*}
		Suy ra
		\begin{align*}
			\dfrac{1}{2x+3y+3z} +\dfrac{1}{3x+2y+3z} + \dfrac{1}{3x+3y+2z}\le \dfrac{4\cdot 12}{16}=3. 
		\end{align*}
		Dấu đẳng thức xảy ra khi $x=y=z=\dfrac{1}{8}$.
	}
\end{bt}

\begin{bt}[Đề thi vào 10, Chuyên Lam Sơn - Vòng 2, 2016]%[9D5G2]
	Một số tự nhiên gọi là \textit{số thú vị} nếu số đó có $10$ chữ số khác nhau và là bội của số $11111$. Hỏi có bao nhiêu số thú vị?
	\loigiai{
		Gọi $A$ là số thú vị $\Rightarrow$ tổng các chữ số của $A$ bằng $45\Rightarrow$ $A\  \vdots \ 99999$. Gọi $A=99999k$ (với $k$ là số tự nhiên) $\Rightarrow 10235\leq k\leq 98766$. Như vậy, ta đi tìm số $k$ sao cho khi nhân $k$ với $99999$ thì các chữ số từ $0$ đến $9$, mỗi số có mặt đúng một lần.\\
		Mặt khác ta có kết quả: khi lấy $k=\overline{a_{1}a_{2}a_{3}a_{4}a_{5}}$ nhân với $99999$ thì ta thu được tích bằng giá trị của $k-1$ và viết thêm phần bù $9$ của $k-1$ vào sau số $k-1$.\\
		Ví dụ: với $k=10235\Rightarrow k-1=10234\Rightarrow$ bù $9$ của $k-1$ là $89765\Rightarrow 10235.99999=1023489765$.\\
		Chú ý rằng với $k=\overline{a_{1}a_{2}a_{3}a_{4}a_{5}}$ thì $k-1$ là số mà không tồn tại hai chữ số có tổng bằng $9$. Như vậy, bài toán được quy về việc đi tìm số các số tự nhiên $k$ thỏa mãn $10235\leq k\leq 98766$ và $k-1$ là số tự nhiên không tồn tại hai chữ số có tổng bằng $9$.\\
		Vì $k-1$ là số tự nhiên không tồn tại hai chữ số có tổng bằng $9$ nên nếu chữ số $a_{i}$ xuất hiện trong $k-1$ thì chữ số $9-a_{i}$ không thể xuất hiện trong $k-1$, do vậy ta xét các cặp số $(0;9)$, $(1;8)$, $(2;7)$, $(3;6)$, $(4;5)$.\\
		Ta lập số $k-1$ bằng cách duy nhất mỗi chữ số trong mỗi cặp (giả sử xét luôn chữ số $0$ là chữ số hàng chục nghìn) $\Rightarrow$ số lượng số $k-1$ là $2^{5}.5!=3840$. Vì số lượng các số có chữ số hàng chục nghìn tương ứng với $0$, $1$, $2$, .., $9$ là bằng nhau $\Rightarrow$ số lượng số có số $0$ đứng đầu là $384$ $\Rightarrow$ số lượng số $k-1$ cần tìm là $3840-384=3456$ số.\\
		Vậy có $3456$ số thú vị.
	}
\end{bt}

\begin{bt}[Đề thi vào 10 Chuyên, TPHCM, 2016]
	Nam cắt một tờ giấy ra làm $4$ miếng hoặc $8$ miếng, rồi lấy một số miếng nhỏ đó cắt ra làm $4$ miếng hoặc $8$ miếng nhỏ hơn và Nam cứ tiếp tục thực hiện việc cắt như thế nhiều lần. Hỏi với việc cắt như vậy, Nam có thể cắt được $2016$ miếng lớn, nhỏ hay không? Vì sao?
	\loigiai{
		Gọi $x$ là số miếng giấy Nam có được sau $k$ lần cắt $(x; k \in \mathbb{N}^*)$. Vì lúc đầu Nam có $1$ miếng giấy và mỗi lần cắt một miếng giấy ra làm $4$ miếng hoặc $8$ miếng nhỏ hơn nên sau mỗi lần cắt,  số miếng giấy tăng thêm $3$ miếng hoặc $7$ miếng, do đó ta có $x \equiv 1\left( {\bmod \,\,3} \right)$ hoặc $x \equiv 1\left( {\bmod \,\,7} \right)$.  Vì $2016 \equiv 0\left( {\bmod \,\,3} \right)$ và $2016 \equiv 0\left( {\bmod \,\,7} \right)$ nên $x\ne 2016$. Vậy sau một số lần cắt, số miếng giấy Nam có được không thể bằng $2016$.}
\end{bt}
\begin{bt}[Sưu tầm]
	Một hộp đựng $ 52 $ viên bi, trong đó có 13 viên bi màu xanh, $ 13 $ viên bi màu đỏ, $ 13 $ viên bi màu vàng, $ 13 $ viên bi màu trắng. Cần phải lấy ít nhất bao nhiêu viên bi (mà không nhìn trước) để chắc chắn trong số đó có không ít hơn $ 7 $ viên bi cùng màu? 
	\loigiai{Nếu lấy $ 6 \cdot 4 + 1 = 25 $ viên bi thì chắc chắn có ít nhất $ 7  $ viên bi cùng màu.}
\end{bt}

\begin{bt}[Sưu tầm]
	Trong phòng có $ 100 $ người. Mỗi người quen ít nhất là $ 67 $ người khác. Chứng minh rằng trong phòng phải có $ 4 $ người từng đôi một quen nhau.
	\loigiai{Xét $ A $ là một người bất kỳ trong phòng. Bởi vì $ A $ quen ít nhất $ 67 $ người khác, nên nếu ta mời tất cả những ai không quen $ A $ ra ngoài phòng thì người phải ra nhiều nhất là $ 32 $.\\
		Khi đó trong phòng còn $ A $ và $ 67 $ người quen $ A $, tức là trong phòng còn ít nhất $ 68  $ người.\\
		Gọi $ B $ là người khác $ A $ trong $ 68 $ người trong phòng ta mời những ai không quen $ B $ ra ngoài. Khí đó trong phòng còn lại ít nhất là $ 68 - 32  = 36 $ người\\
		Lại gọi $ C $ là người khác không quen $ A $ và $ B $ trong số $ 36 $ người còn lại. Mời tất cả người không quen $ C $ ra ngoài, trong phòng còn lại ít nhất là $ 36 - 32 = 4 $ người\\
		Nghĩa là ngoài $ A $, $ B $, $ C $ còn lại ít nhất một người, giả sử là $ D $. Khi đó $ A $,$ B $,$ C $,$ D $ đôi một quen nhau.
	}
\end{bt}

\begin{bt}[Sưu tầm]
	Có ba chiếc hộp: một hộp đựng hai trái cam, một hộp đựng hai trái quýt và một hộp đựng một trái cam và một trái quýt.\\
	Nhưng khi đóng kín các hộp, người ta đã dán nhầm các nhãn CC, CQ, QQ cho nên các nhãn ở ngoài hộp đều không đúng với trái đựng trong hộp. Làm thế nào để chỉ cần lấy một trái trong một hộp mà không nhìn vào trong hộp có thể biết được chính xác các trái đựng trong ba hộp?
	\loigiai{Nếu lấy $ 1 $ quả trong hộp bên ngoài dán nhãn CQ.
		\begin{itemize}
			\item Nếu quả lấy ra là cam thì hộp có nhãn $ CQ $ đựng $ 2 $ trái cam, hộp $ CC $ hai trái quýt và hộp QQ đựng $ 1 $ trái cam và $ 1 $ trái quýt.
			\item Nếu quả lấy ra là quýt thì hộp có nhãn $ CQ $ đựng $ 2 $ trái quýt, hộp  QQ đựng hai trái cam, hộp CC đựng $ 1 $ cam và $ 1 $ quýt.
		\end{itemize}
		
	}
\end{bt}
