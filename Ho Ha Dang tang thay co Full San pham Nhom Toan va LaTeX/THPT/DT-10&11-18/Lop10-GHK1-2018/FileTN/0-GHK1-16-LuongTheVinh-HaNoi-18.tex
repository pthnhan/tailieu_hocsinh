	Một học sinh giải phương trình $x+ \sqrt{x+3}+2= \sqrt{x+3} \, (*)$  như sau:
	\begin{enumerate}[Bước 1.]
		\item $(*) \Rightarrow x + 2 = 0$;
		\item $\Leftrightarrow x= -2$;
		\item Thử lại ta thấy phương trình có nghiệm $x=-2$.
	\end{enumerate}
	Lời giải trên đúng hay \textbf{sai}, nếu sai thì sai bắt đầu từ bước nào?
	\choice
	{\True Lời giải đúng}
	{Lời giải sai từ bước $ 1 $}
	{Lời giải sai từ bước $ 2 $}
	{Lời giải sai từ bước $ 3 $}
	\loigiai{
Ta có $ x+ \sqrt{x+3}+2= \sqrt{x+3}\Rightarrow x+2=0\Rightarrow x=-2 $.\\
Với $ x=-2 $, thay vào phương trình $ (*) $ ta thấy thỏa mãn.\\
Vậy $ x=-2 $ là nghiệm của phương trình.
}

	Tìm khẳng định {\bf sai} trong các khẳng định dưới đây?
	\choice
	{\True $\sqrt{1-x^2}+x=\sqrt{1-x^2}+2 \Leftrightarrow x=2$}
	{$\sqrt{x}=\sqrt{-x} \Leftrightarrow x=0$}
	{ $x^2=1 \Leftrightarrow x=\pm 1$}
	{$(x-2)^2=0 \Leftrightarrow x=2$}
	\loigiai{
\begin{enumerate}

	Vậy phương trình vô nghiệm.
	\item Xét phương trình $ \sqrt{x}=\sqrt{-x} $.\\
	Điều kiện xác định $ \heva{&x\geq 0\\ &-x\geq 0}\Leftrightarrow x=0 $.\\
	Thay $ x=0 $ vào phương trình ta thấy thỏa mãn.\\
	Vậy $ x=0 $ là nghiệm của phương trình.
	\item Xét phương trình $ x^{2}-1=0\Leftrightarrow x=\pm 1 $.
	\item Xét phương trình $ (x-2)^{2}=0\Leftrightarrow x=2 $.
\end{enumerate}
}
\end{ex}

\begin{ex}%[GHK1, THPT Lương Thế Vinh CS1 - Hà Nội, 2019]%[Dương BùiĐức, 10-EX-1-19]%[0D2B1-4]
	Đồ thị của hàm số nào sau đây nhận trục tung làm trục đối xứng?
	\choice
	{$y=x^2-3x+2$}
	{\True $y=x^2+4$}
	{$y=-x^2+2x$}
	{$y=-3x^2-x+1$}
	\loigiai{
\begin{enumerate}
	\item Xét $y=x^2-3x+2$.\\
	Tập xác định $ \mathscr{D}=\mathbb{R} $, do đó với $ x\in\mathscr{D} $ thì $ -x\in\mathscr{D} $.\\
	Ta có $ y(-x)=(-x)^{2}-3(-x)+2=x^{2}+3x+2\neq y(x)\Rightarrow y=x^2-3x+2 $ không là hàm chẵn nên đồ thị hàm số không nhận trục tung làm trục đối xứng.
	\item Xét $y=x^2+4$.\\
	Tập xác định $ \mathscr{D}=\mathbb{R} $, do đó với $ x\in\mathscr{D} $ thì $ -x\in\mathscr{D} $.\\
	Ta có $ y(-x)=(-x)^{2}+4=x^{2}+4=y(x)\Rightarrow y=x^2+4 $ là hàm chẵn nên đồ thị hàm số nhận trục tung làm trục đối xứng.
	\item Xét $y=-x^2+2x $.\\
	Tập xác định $ \mathscr{D}=\mathbb{R} $, do đó với $ x\in\mathscr{D} $ thì $ -x\in\mathscr{D} $.\\
	Ta có $ y(-x)=-(-x)^{2}+2(-x)=-x^{2}-2x\neq y(x)\Rightarrow y=-x^2-2x $ không là hàm chẵn nên đồ thị hàm số không nhận trục tung làm trục đối xứng.
	\item Xét $y=-3x^2-x+1$.\\
	Tập xác định $ \mathscr{D}=\mathbb{R} $, do đó với $ x\in\mathscr{D} $ thì $ -x\in\mathscr{D} $.\\
	Ta có $ y(-x)=-3(-x)^{2}-(-x)+1=-3x^{2}+x+1\neq y(x)\Rightarrow y=x^2-3x+2 $ không là hàm chẵn nên đồ thị hàm số không nhận trục tung làm trục đối xứng.
\end{enumerate}	
}
\end{ex}

	Cho phương trình $f(x)=g(x)$ xác định với mọi $x>0$. Trong các phương trình dưới đây, phương trình nào {\bf không} tương đương với phương trình đã cho?
	\choice
	{$k\cdot f(x)=k\cdot g(x)$, với mọi số thực $k \ne 0$}

	\loigiai{
\begin{enumerate}
	\item Xét $k\cdot f(x)=k\cdot g(x)$.\\
	Vì $ k\neq 0 $ nên $k\cdot f(x)=k\cdot g(x)\Leftrightarrow f(x)=g(x)$.
	\item Xét $(x^{2}+1)\cdot f(x)=(x^{2}+1)\cdot g(x)$.\\

	\item Xét $\sqrt{x^2+2x+3}\cdot f(x)=\sqrt{x^2+2x+3}\cdot g(x)$.\\
	Vì $ x^{2}+2x+3>0 $ với mọi $ x $ nên $\sqrt{x^2+2x+3}\cdot f(x)=\sqrt{x^2+2x+3}\cdot g(x)\Leftrightarrow f(x)=g(x)$.
	\item Xét $\dfrac{f(x)}{\sqrt{-x}}=\dfrac{g(x)}{\sqrt{-x}}$.\\
	Điều kiện xác định $ x<0 $.\\
	Ta có $\dfrac{f(x)}{\sqrt{-x}}=\dfrac{g(x)}{\sqrt{-x}}\Leftrightarrow f(x)=g(x)$ với $ x<0 $.
\end{enumerate}	
}
\end{ex}
