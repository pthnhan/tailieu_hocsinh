\section{BỘI CHUNG NHỎ NHẤT}
\subsection{Tóm tắt lý thuyết}
\subsubsection{Bội chung nhỏ nhất}
\begin{dn}
	Bội chung nhỏ nhất (BCNN) của hai hay nhiều số là số nhỏ nhất khác $0$ trong tập hợp các bội chung của các số đó.
\end{dn}
\subsubsection{Tìm BCNN bằng cách phân tích các số ra thừa số nguyên tố}
\begin{itemize}
	\item \textit{Bước 1}: Phân tích mỗi số ra thừa số nguyên tố.
	\item \textit{Bước 2}: Chọn ra các thừa số nguyên tố chung và riêng.
	\item \textit{Bước 3}: Lập một tích các thừa số đã chọn, mỗi thừa số lấy với số mũ lớn nhất của nó. Tích đó là BCNN phải tìm.
\end{itemize}
\begin{note}{Chú ý}
	\begin{itemize}
		\item Nếu các số đã cho từng đôi một nguyên tố cùng nhau thì BCNN của chúng là tích của các số đó.
		\item Trong các số đã cho, số lớn nhất là bội của các số còn lại thì BCNN của các số đã cho chính là số lớn nhất đó.
	\end{itemize}
\end{note}
\subsubsection{Cách tìm bội chung thông qua BCNN}
Để tìm bội chung của các số đã cho, ta có thể tìm bội của BCNN của các số đó.
\subsection{Các dạng toán}
\begin{dang}{Tìm BCNN của hai hay nhiều số}
	\textit{\textbf{Phương pháp giải}}
	\begin{itemize}
		\item Thực hiện theo ba bước trong quy tắc tìm BCNN.
		\item Trong trường hợp đơn giản, có thể nhẩm BCNN của hai hay nhiều số bằng cách nhân số lớn nhất lần lượt với $1; 2; 3; \ldots$ cho đến khi được kết quả là một số chia hết cho các số còn lại.
	\end{itemize}
\end{dang}
\begin{vd}
	Tìm BCNN của các số sau:
	\begin{enumEX}{2}
		\item $24$ và $80$;
		\item $90$; $99$ và $84$.
	\end{enumEX}
\end{vd}
\loigiai{
\begin{enumEX}{1}
	\item Ta có $24=2^3\cdot 3$ và $80=2^4\cdot 5$. Khi đó, BCNN$\left(24; 80\right)=2^4\cdot 3\cdot 5=240$.
	\item $90=2\cdot 3^2\cdot 5$; $99=3^2\cdot 11$ và $84=2^2\cdot 3\cdot 7$. Khi đó, BCNN$\left(90; 99; 84\right)=2^2\cdot 3^2\cdot 5\cdot 7\cdot 11=13860$.
\end{enumEX}	
}
\begin{vd}
	Tìm BCNN của các số sau:
	\begin{enumEX}{2}
		\item $270$; $135$ và $90$;
		\item $13$; $20$ và $1$.
	\end{enumEX}
\loigiai{
\begin{enumEX}{1}
	\item Vì $270\,\vdots\,135$ và $270\,\vdots\,90$ nên BCNN$(270; 135; 90)=270$.
	\item Vì hai số $13$ và $20$ nguyên tố cùng nhau nên BCNN$(13; 20; 1)=13\cdot 20=260$.
\end{enumEX}
}
\end{vd}

\begin{bt}%[Nguyễn Ngọc Hóa, dự án EX-C2-L6-2019]%[6D1B18]
Tìm BCNN của các số sau:
	\begin{enumEX}{2}
		\item $25$ và $150$;
		\item $25$ và $12$;
		\item $20$ và $56$;
		\item $100$; $270$ và $315$.
	\end{enumEX}
\loigiai{
\begin{enumEX}{1}
	\item Vì $150\,\vdots\,25$ nên BCNN$(25; 150)=150$.
	\item Do hai số $25$ và $12$ nguyên tố cùng nhau nên BCNN$(25; 12)=25\cdot 12=300$.
	\item Ta có $20=2^2\cdot5$ và $56=2^3\cdot7$. Khi đó BCNN$(20; 56)=2^3\cdot5\cdot7=280$.
	\item Ta có $100=2^2\cdot5^2$; $270=2\cdot3^3\cdot5$ và $315=3^2\cdot5\cdot7$.\\
	Khi đó, BCNN$(100; 270; 315)=2^2\cdot3^3\cdot5^2\cdot7=18900$.
\end{enumEX}
}
\end{bt}

\begin{dang}{Tìm các bội chung thỏa mãn điều kiện cho trước}
	\textit{\textbf{Phương pháp giải}}
	\begin{itemize}
		\item Tìm BCNN của các số cho trước.
		\item Tìm các bội của BCNN này.
		\item Chọn ra các số thỏa mãn điều kiện cho trước.
	\end{itemize}
\end{dang}
\begin{vd}
	Tìm các bội chung của $20$; $25$ và $75$.
\loigiai{
Ta có $20=2^2\cdot5$; $25=5^2$ và $75=3\cdot5^2$. Khi đó BCNN$(20; 25; 75)=2^2\cdot3\cdot5^2=300$.\\
Vậy BC$(20; 25; 75)=$ B$(300)=\left\lbrace 0; 300; 600; 900; \ldots\right\rbrace $.
}
\end{vd}
\begin{vd}
	Tìm các bội chung nhỏ hơn $1000$ của $45$ và $25$.
\loigiai{
Ta có $45=3^2\cdot5$ và $25=5^2$. Khi đó, BCNN$(45; 25)=3^2\cdot5^2=225$.\\
Suy ra  BC$(45; 25)=$ B$(225)=\left\lbrace 0; 225; 450; 675; 900; 1125; \ldots\right\rbrace $.\\
Vậy tập hợp các bội chung nhỏ hơn $1000$ của $45$ và $25$ là $\left\lbrace 0; 225; 450; 675; 900\right\rbrace$.
}
\end{vd}
\begin{vd}
	Tìm số tự nhiên $x$ sao cho: $x\,\vdots\,72$; $\,\vdots\,108$ và $500<x<1000$.
\loigiai{
Vì $x\,\vdots\,72$ và $x\,\vdots\,108$ nên $x\in$ BC$(72; 108)$.
Ta có $72=2^3\cdot3^2$; $108=2^2\cdot3^3$.\\
Khi đó BCNN$(72; 108)=2^3\cdot3^3=216$.\\
Suy ra BC$(72; 108)=$ B$(216)=\left\lbrace 0; 216; 432; 648; 864; 1080; \ldots\right\rbrace $.\\
Vì $500<x<1000$ nên $x\in \{648; 864\}$.
}
\end{vd}

\begin{dang}{Bài toán đưa đến việc tìm BCNN của hai hay nhiều số}
	\textit{\textbf{Phương pháp giải}}: Phân tích đề bài, suy luận đưa về việc tìm BCNN của hai hay nhiều số.
\end{dang}
\begin{vd}
	Tìm số tự nhiên nhỏ nhất khác $0$ mà $x\,\vdots\,147$ và $x\,\vdots\,105$.
\loigiai{
	Vì $x\,\vdots\,147$ và $x\,\vdots\,105$ nên $x\in $ BC$(20; 24; 36)$; mà $x$ nhỏ nhất khác $0$ nên $x$ là BCNN$(147; 105)$.\\
Ta có $147=3\cdot7^2$; $105=3\cdot5\cdot7$. Do đó BCNN$(147; 105)=3\cdot5\cdot7^2\cdot5=735$.\\
Vậy $x=735$.
}
\end{vd}

\begin{vd}
	Tìm số tự nhiên $x$ nhỏ nhất, lớn hơn $200$ mà khi chia $x$ cho $4$, cho $5$, cho $6$ đều dư $3$.
	\loigiai{
		Vì $x$ chia cho $4$, cho $5$, cho $6$ đều dư $3$ nên $(x-3)\,\vdots\,4$; $(x-3)\,\vdots\,5$ và $(x-3)\,\vdots\,6$.\\
		Suy ra $(x-3)\in$ BC$(4; 5; 6)$.\\
		Ta có $4=2^2$; $6=2\cdot3$. Khi đó, BCNN$(4; 5; 6)=2^2\cdot3\cdot5=60$.\\
		Suy ra BC$(4; 5; 6)=$ B$(60)=\{0; 60; 120; 180; 240; \ldots\}$.\\
		Suy ra $(x-3)\in \{0; 60; 120; 180; 240; \ldots\}$ hay $x\in \{3; 63; 123; 183; 243; \ldots\}$;\\
		Mà $x$ nhỏ nhất và lớn hơn $200$ nên $x=243$.\\
		Vậy $x=243$.
	}
\end{vd}

\begin{vd}
	Tìm số tự nhiên nhỏ nhất chia cho $5$ dư $2$, chia cho $8$ dư $5$.
\loigiai{
Gọi số phải tìm là $x$ $(x\in \mathbb{N})$. Vì $x$ chia cho $5$ dư $2$ nên $(x-2)\,\vdots\,5$ hay $(x+3)\,\vdots\,5$; $x$ chia cho $8$ dư $5$ nên $(x-5)\,\vdots\,8$ hay $(x+3)\,\vdots\,8$; suy ra $(x+3)$ là bội chung của $5$ và $8$.\\
Do BCNN$(5; 8)=40$ nên BC$(5; 8)=$ B$(40)$ hay $x+3\in \{0; 40; 80; 120; \ldots\}$.\\
Mà $x$ nhỏ nhất nên ta chọn $x+3=40$ hay $x=37$.\\
Vậy số cần tìm là $37$.
}
\end{vd}

\begin{vd}
	Số học sinh khối 6 của một trường khoảng gần 500 học sinh. Biết rằng nếu xếp hàng $5$, hàng $8$, hàng $12$ đều thiếu $1$. Tính số học sinh khối $6$ của trường đó.
\loigiai{
Gọi số học sinh khối $6$ của trường đó là $a$ $(a\in \mathbb{N})$.\\
Vì xếp hàng $5$, hàng $8$, hàng $12$ đều thiếu $1$ nên $(a+1)\,\vdots\,5; (a+1)\,\vdots\,8; (a+1)\,\vdots\,12$.\\
Do đó $(a+1)\in$ BC$(5; 8; 12)$.\\
Ta có BCNN$(5; 8; 12)=120$ nên BC$(5; 8; 12)=$ B$(120)$. Do đó, $a+1\in \{0; 120; 240; 360; 480; 600; \ldots\}$.\\
Mà số học sinh khoảng gần $500$ nên ta chọn $a+1=480$ suy ra $a=479$.\\
Vậy số học sinh khối $6$ của trường là $479$.
}
\end{vd}

\begin{vd}
	Các cột điện trước đây cách nhau $60$m, nay trồng lại, cách nhau $45$m. Hỏi sau cột đầu tiên không trồng lại thì cột gần nhất không phải trồng lại là cột thứ mấy?
\loigiai{
Khoảng cách gần nhất giữa hai cột không phải trồng lại phải chia hết cho $60$, cho $45$ nên khoảng cách này là BCNN$(60; 45)$. Ta có BCNN$(60; 45)=180$.\\
Vậy cột đầu tiên không phải trồng lại sau cột đầu là cột thứ $180 :  60+1=4$.
}
\end{vd}

\end{center}
\begin{bt}%[Nguyễn Ngọc Hóa, dự án EX-C2-L6-2019]%[6D1B18]
	Tìm số tự nhiên $x$ nhỏ nhất khác $0$ sao cho $x\,\vdots\,20$; $x\,\vdots\,24$ và $x\,\vdots\,36$.
	\loigiai{
		Vì $x\,\vdots\,20$; $x\,\vdots\,24$ và $x\,\vdots\,36$ nên $x\in $ BC$(20; 24; 36)$; mà $x$ nhỏ nhất khác $0$ nên $x$ là BCNN$(20; 24; 36)$.\\
		Ta có $20=2^2\cdot5$; $24=2^3\cdot3$ và $36=2^2\cdot3^2$.\\
		Khi đó BCNN$(20; 24; 36)=2^3\cdot3^2\cdot5=360$.\\
		Vậy $x=360$.
	}
\end{bt}

\begin{bt}%[Nguyễn Ngọc Hóa, dự án EX-C2-L6-2019]%[6D1K18]
	Tìm số tự nhiên $x<500$ biết rằng khi chia $x$ cho $6$, cho $7$, cho $8$ đều dư $2$.
	\loigiai{
		Vì $x$ chia cho $6$, cho $7$, cho $8$ đều dư $2$ nên $(x-2)\,\vdots\,6$; $(x-2)\,\vdots\,7$ và $(x-2)\,\vdots\,8$ suy ra $(x-2)\in$BC$(6; 7; 8)$.\\
		Ta có $6=2\cdot3$; $8=2^3$. Khi đó, BCNN$(6; 7; 8)=2^3\cdot3\cdot7=168$.\\
		Suy ra BC$(6; 7; 8)=$ B$(168)=\{0; 168; 336; 504; \ldots\}$.
		Suy ra $(x-2)\in \{0; 168; 336; 504; \ldots\}$ hay $x\in \{2; 170; 338; 506; \ldots\}$;\\
		Mà $x<500$ nên $x\in \{2; 170; 338\}$.\\
		Vậy $x\in \{2; 170; 338\}$.
	}
\end{bt}

\begin{bt}%[Nguyễn Ngọc Hóa, dự án EX-C2-L6-2019]%[6D1K18]
Một đội đồng diễn thể dục có hơn $200$ học sinh nhưng chưa đến $300$ học sinh. Khi xếp hàng $8$, hàng $10$, hàng $12$ đều không thừa học sinh nào. Hỏi đội đồng diễn đó có bao nhiêu học sinh?
\loigiai{
Gọi số học sinh của đội đồng diễn thể dục là $x$, $(200<x<300)$.\\
Vì đội đồng diễn thể dục khi xếp hàng $8$, hàng $10$, hàng $12$ đều không thừa học sinh nào nên $x\,\vdots\,8$, $x\,\vdots\,10$, $x\,\vdots\,12$ hay $x\in$ BC$(8; 10; 12)$.\\
Ta có $8=2^3$; $10=2\cdot5$; $12=2^2\cdot3$ suy ra BCNN$(8; 10; 12)=2^3\cdot3\cdot5=120$.\\
Do đó BC$(8; 10; 12)=$ BC$(120)=\{0; 120; 240; 360; 480; \ldots\}$ suy ra $x\in \{0; 120; 240; 360; 480; \ldots\}$.\\
Mà $200<x<300$ nên $x=240$. Vậy đội đồng diễn có $240$ học sinh.
}
\end{bt}

\begin{bt}*%[Nguyễn Ngọc Hóa, dự án EX-C2-L6-2019]%[6D1G18]
Tìm số tự nhiên $x$ nhỏ nhất sao cho chia $x$ cho $7$ được số dư là $4$; chia $x$ cho $11$ được số dư là $6$.
\loigiai{
Vì $x$ chia cho $7$ dư $4$ nên $x=7\cdot m+4$ $(m\in \mathbb{N})$, suy ra $2\cdot x=14\cdot m+8=7\cdot (2\cdot m+1)+1$, chia cho $7$ dư $1$.\\
Tương tự, $x$ chia cho $11$ dư $6$ nên $x=11\cdot n+6$ $(6\in \mathbb{N})$, suy ra $2\cdot x$ chia cho $11$ dư $1$.\\
Do đó $(2\cdot x-1)\in$ BC$(7; 11)$.\\
Vì BCNN$(7; 11)=77$ nên $\left(2\cdot x-1\right)\in \{0; 77; 154; 231; \ldots\}$.\\
Để $x$ nhỏ nhất thì $2\cdot x-1=0\Rightarrow 2\cdot x=1$ (không có số tự nhiên $x$ nào).\\
Vậy $2\cdot x-1=77\Rightarrow 2\cdot x=78\Rightarrow x=39$.
}
\end{bt}

\section{ÔN TẬP CHƯƠNG I}
\subsection{Tóm tắt lý thuyết}
\begin{enumEX}[1.]{1}
	\item Tập hợp - Tập hợp con.
	\item Các phép tính về số tự nhiên (Cộng, trừ, nhân, chia, lũy thừa).
	\item Tính chất chia hết của một tổng. Dấu hiệu chia hết cho $2$, cho $5$, cho $3$, cho $9$.
	\item Số nguyên tố, hợp số. Phân tích một số ra thừa số nguyên tố.
	\item ƯCLN và BCNN của hai hay nhiều số.
\end{enumEX}
\subsection{Các dạng toán}
\begin{dang}{Xác định số phần tử của một tập hợp}
	\textbf{\textit{Phương pháp giải}}
	\begin{itemize}
		\item Nếu tập hợp có ít phần tử thì ta có thể đếm trực tiếp.
		\item Nếu tập hợp $A$ gồm các phần tử là các số tự nhiên liên tiếp từ $a$ đến $b$ $(a<b)$, mà hai phần tử liền kề cách nhau $k$ đơn vị thì tập hợp $A$ có $(b-a):k+1$ phần tử.		
	\end{itemize}
\end{dang}
\begin{vd}
	Hãy viết tập hợp gồm các chữ cái của mỗi từ sau đây. Tập hợp nào có ít số phần tử nhất? (không viết dấu)
	\begin{enumEX}{2}
		\item QUẢNG NINH;
		\item QUẢNG NAM;
		\item QUẢNG NGÃI;
		\item BÌNH ĐỊNH.
	\end{enumEX}
\loigiai{
\begin{itemize}
	\item Tập hợp các chữ cái trong từ QUẢNG NINH là
	\{Q, U, A, N, G, I, H\}. Số phần tử là $7$.
	\item Tập hợp các chữ cái trong từ QUẢNG NAM là
		\{Q, U, A, N, G, M\}. Số phần tử là $6$.
	\item Tập hợp các chữ cái trong từ QUẢNG NGÃI là
		\{Q, U, A, N, G, I\}. Số phần tử là $6$.
	\item Tập hợp các chữ cái trong từ BÌNH ĐỊNH là
		\{B, I, N, H, Đ\}. Số phần tử là $5$.
\end{itemize}
Vậy tập hợp \{B, I, N, H, Đ\} có ít chữ cái nhất.
}
\end{vd}

\begin{vd}
	Tính số phần tử của tập hợp các số lẻ có ba chữ số.
\loigiai{
Tập hợp các số lẻ có ba chữ số là $\{101; 103; 105; \ldots; 997; 999\}$.\\
Số phần tử của tập hợp là $\left(999-101\right) :  2+1=450$ (số).
}
\end{vd}

\begin{vd}
	Tập hợp các ước của số $2520$ có bao nhiêu phần tử?
\loigiai{
Ta có $2520=2^3\cdot 3^2\cdot 5\cdot 7$.\\
Số lượng các ước của số $2520$ là $(3+1)\cdot(2+1)\cdot(1+1)\cdot(1+1)=48$.\\
Vậy tập hợp các ước của số $2520$ có $48$ phần tử.
}
\end{vd}

\begin{dang}{Nhận biết và viết tập hợp con của một tập hợp cho trước}
	\textbf{\textit{Phương pháp giải}}
	\begin{itemize}
		\item Dựa vào định nghĩa tập hợp con của một tập hợp: với mọi $x\in \mathrm{A}$ mà $x\in \mathrm{B}$ thì $\mathrm{A}\subset\mathrm{B}$.
		\item Để viết tập hợp con của một tập hợp, trước tiên ta viết tập hợp rỗng, rồi viết các tập hợp con có một phần tử, các tập hợp con có hai phần tử,$\ldots$ cuối cùng là chính tập hợp đã cho.
	\end{itemize}
\end{dang}

\begin{vd}
	Cho $\mathrm{E}=\{1; 3; 3; 5; 7; 9\}$; $\mathrm{F}=\{1; 2; 3; 4; 5\}$.
\begin{enumEX}{1}
	\item Hỏi $\mathrm{E}$ có phải là tập hợp con của $\mathrm{F}$ không?
	\item Hỏi $\mathrm{F}$ có phải là tập hợp con của $\mathrm{E}$ không?
	\item Tìm $\mathrm{E}\cap\mathrm{F}$.
\end{enumEX}
\loigiai{
\begin{enumEX}{1}
	\item Vì $9\in \mathrm{E}$ nhưng $9\not \in \mathrm{F}$ nên $\mathrm{E}$ không phải là tập hợp con của $\mathrm{F}$.
	\item Vì $4\in \mathrm{F}$ nhưng $4\not \in\mathrm{E}$ nên $\mathrm{F}$ không phải là tập hợp con của $\mathrm{E}$.
	\item Tìm $\mathrm{E}\cap\mathrm{F}=\{1; 3; 5\}$.
\end{enumEX}
}
\end{vd}

\begin{vd}
	Viết tập hợp $\mathrm{M}$ sao cho $\{5; 6\}\subset\mathrm{M}\subset\{5; 6; 7; 8\}$.
\loigiai{
$\mathrm{M}_1=\{5; 6\}$; $\mathrm{M}_2=\{5; 6; 7\}$; $\mathrm{M}_3=\{5; 6; 8\}$; $\mathrm{M}_4=\{5; 6; 7; 8\}$.
}
\end{vd}

\begin{bt}%[Nguyễn Ngọc Hóa, dự án EX-C2-L6-2019]%[6D1Y2]

\begin{vd}
	Cho biểu thức A$=\dfrac{n+3}{n}$ $(n\in \mathbb{N}^*)$. Tìm giá trị của $n$ để
	\begin{enumEX}{1}
		\item Biểu thức $A$ có giá trị lớn nhất.
		\item Biểu thức $A$ có giá trị là một số tự nhiên.
	\end{enumEX}

	Vậy giá trị lớn nhất của $A$ là $4$ khi $n=1$.
