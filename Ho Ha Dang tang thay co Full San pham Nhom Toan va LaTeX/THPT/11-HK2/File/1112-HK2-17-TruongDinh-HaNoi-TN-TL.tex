\begin{ex}%[HK2 (2016-2017), THPT Trương Định, Hà Nội]%[1D4B2-6]%[Dương BùiĐức]
Tính giới hạn $\lim\limits_{x\to+\infty}(5x^{2}-x^{3}+1)$.
\choice
{\True $-\infty$}
{$-1$}
{$5$}
{$+\infty$}
\loigiai{
$\lim\limits_{x\to+\infty}(5x^{2}-x^{3}+1)=\lim\limits_{x\to+\infty}\left[x^{3}\left(-1+\dfrac{5}{x}+\dfrac{1}{x^{3}}\right)\right]=-\infty$ vì $\heva{&\lim\limits_{x\to+\infty}x^{3}=+\infty\\ &\lim\limits_{x\to+\infty}\left(-1+\dfrac{5}{x}+\dfrac{1}{x^{3}}\right)=-1.}$
}
\end{ex}
\begin{ex}%[HK2 (2016-2017), THPT Trương Định, Hà Nội]%[1D4B2-3]%[Dương BùiĐức]

\choice
{$\dfrac{7}{6}$}
{\True $\dfrac{7}{2}$}
{$3$}
{$\dfrac{1}{2}$}
\loigiai{
$\lim\limits_{x\to-1}\dfrac{3x^{2}-x-4}{x^{2}-1}=\lim\limits_{x\to-1}\dfrac{(3x-4)(x+1)}{(x-1)(x+1)}=\lim\limits_{x\to-1}\dfrac{3x-4}{x-1}=\dfrac{7}{2}$.
}
\end{ex}

\begin{ex}%[HK2 (2016-2017), THPT Trương Định, Hà Nội]%[1D4K3-5]%[Dương BùiĐức]
Tìm giá trị của $m$ để hàm số $f(x)=\heva{&\dfrac{4-x}{\sqrt{x+5}-3}&\mbox{ khi }x\neq 4\\ &1-m&\mbox{ khi }x=4}$ liên tục tại $x=4$.
\choice
{\True $m=7$}
{$m=0$}
{$m=2$}
{$m=-5$}
\loigiai{
Ta có $\lim\limits_{x\to 4}\dfrac{4-x}{\sqrt{x+5}-3}=\lim\limits_{x\to 4}\dfrac{(4-x)(\sqrt{x+5}+3)}{x-4}=-\lim\limits_{x\to 4}(\sqrt{x+5}+3)=-6$.\\
Để hàm số liên tục tại $x=4$ thì $-6=1-m\Leftrightarrow m=7$.
}
\end{ex}

\begin{ex}%[HK2 (2016-2017), THPT Trương Định, Hà Nội]%[1D5K2-3]%[Dương BùiĐức]

{$m=-\dfrac{1}{2}$}
{$m=-3$}
{$m=1$}

\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[HK2 (2016-2017), THPT Trương Định, Hà Nội]%[1D5B2-1]%[1D4B3-6]%[1D5B2-3]%[Dương BùiĐức]
Cho hàm số $f(x)=-x^{3}+3x^{2}-3$ ($1$).
\begin{enumerate}
\item Giải bất phương trình $f'(x)\leq -9$.
\item Chứng minh rằng phương trình $f(x)=0$ có ít nhất hai nghiệm.
\item Viết phương trình tiếp tuyến của đồ thị hàm số $(1)$, biết hệ số góc của tiếp tuyến bằng $-24$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $f'(x)=-3x^{2}+6x\leq -9\Leftrightarrow x^{2}-2x-3\geq 0\Leftrightarrow\hoac{&x\geq 3\\ &x\leq -1.}$
\item Nhận xét: $f(x)$ xác định trên $\mathbb R$ nên $f(x)$ liên tục trên $\mathbb R$.\\
Ta có $f(0)=-3$, $f(-1)=1$, $f(1,5)=\dfrac{3}{8}\Rightarrow \heva{&f(0)\cdot f(1,5)<0\\ &f(0)\cdot f(-1)<0}\Rightarrow f(x)$ có ít nhất hai nghiệm nằm trong các khoảng $(-1;0)$ và $\left(0;\dfrac{3}{2}\right)$.
\item Xét $-3x^{2}+6x=-24\Leftrightarrow x^{2}-2x-8=0\Leftrightarrow\hoac{&x=4\\ &x=-2.}$
\begin{itemize}
\item Với $x=4\Rightarrow f(4)=-19\Rightarrow$ phương trình tiếp tuyến $y=-24(x-4)-19=-24x+77$.
\item Với $x=-2\Rightarrow f(-2)=17\Rightarrow$ phương trình tiếp tuyến $y=-24(x+2)+17=-24x-27$.
\end{itemize}
\end{enumerate}
}
\end{bt}

\end{enumerate}

\begin{center}

\item Vì tam giác $ABC$ đều cạnh $a$, $BE$ là trung tuyến nên $BE=\dfrac{a\sqrt{3}}{2}$.\\
Trong mặt phẳng $(ABC)$, kẻ đường thẳng $b$ đi qua $B$ và song song với $AC$. Qua $A$ kẻ đường thẳng $a\parallel BE$ cắt $b$ tại $K\Rightarrow AK\perp BK\Rightarrow BK\perp (SAK)$.\\
Trong mặt phẳng $(SAK)$, kẻ $AH\perp SK\Rightarrow AH\perp (SBK)$\\
$\Rightarrow AH=d(A,(SBK))=d(AC,SB)$.\\
Ta có $AH=BE=\dfrac{a\sqrt{3}}{2}$, $SA=a\Rightarrow \dfrac{1}{AH^{2}}=\dfrac{1}{a^{2}}+\dfrac{4}{3a^{2}}=\dfrac{7}{3a^{2}}\Rightarrow AH=\dfrac{a\sqrt{3}}{7}$.\\
Vậy $d(AC,SB)=\dfrac{a\sqrt{3}}{7}$.
