\section{Cộng độ dài các đoạn thẳng}

\subsection{Các dạng toán}
\begin{dang}{Tính chất cộng độ dài đoạn thẳng}
	\begin{itemize}
		\item Tính chất cộng các đoạn thẳng: Nếu điểm $M$ nằm giữa hai điểm $A$ và $B$ thì $AM+MB=AB$.
		\item \textbf{Dấu hiệu 4} (Dấu hiệu Đ4): Nếu có đẳng thức $AM+MB=AB$ thì điểm $M$ nằm giữa hai điểm $A$ và $B$.
		\item \textbf{Dấu hiệu 5} (Dấu hiệu Đ5): Nếu các điểm $A$ và $B$ thuộc tia $Ox$ và $OA<OB$ thì điểm $A$ nằm giữa hai điểm $O$ và $B$.
	\end{itemize}
\end{dang}

\begin{vd}
	Cho đoạn thẳng $AB = 5$ cm. Lấy điểm $M$ thuộc đường thẳng $AB$ mà $BM = 2$ cm. Tính độ dài đoạn thẳng $AM$.
	\loigiai
	{
		Biết điểm $M$ thuộc đường thẳng $AB$ và $BM = 2$ cm nên điểm $M$ không xác định duy nhất trên đường thẳng $AB$. Ta xét hai trường hợp sau:
		\begin{itemize}
			\item Điểm $M$ thuộc tia $BA$ (hình a):
				\begin{center}
				\begin{tikzpicture}[line join = round, line cap = round,>=stealth,font=\footnotesize,scale=1]
				\draw (0,0) node[below]{$A$}--(3,0) node[below]{$M$}--(5,0) node[below]{$B$};
				\fill(0,0) circle (1.5pt) (3,0) circle (1.5pt) (5,0) circle (1.5pt);
				\path (0,0)--(5,0) node[below = .5cm,midway]{a)};
				\end{tikzpicture}
				\end{center}
				Có $BM = 2$ cm, $BA = AB = 5$ cm nên $BM<BA$, mà điểm $M$ thuộc tia $BA$, do đó điểm $M$ nằm giữa hai điểm $A$ và $B$.\\
				Suy ra $AM+MB=AB$, hay $AM+2=5$, nên $AM=5-2=3$ (cm).
			\item Điểm $M$ thuộc tia đối của tia $BA$ (hình b):
				\begin{center}
					\begin{tikzpicture}[line join = round, line cap = round,>=stealth,font=\footnotesize,scale=1]
					\draw (0,0) node[below]{$A$}--(3,0) node[below]{$B$}--(5,0) node[below]{$M$};
					\fill(0,0) circle (1.5pt) (3,0) circle (1.5pt) (5,0) circle (1.5pt);
					\path (0,0)--(5,0) node[below = .5cm,midway]{b)};
					\end{tikzpicture}
				\end{center}
				Có hai tia $BA$ và $BM$ đối nhau nên điểm $B$ nằm giữa hai điểm $A$ và $M$.\\
				Suy ra $AM = AB+BM$, hay $AM = 5+2=7$ (cm).
		\end{itemize}
		Vậy độ dài đoạn thẳng $AM$ bằng $3$ cm hoặc $7$ cm.
	}
\end{vd}

\begin{vd}
	Cho đoạn thẳng $AB = a$. Lấy điểm $M$ thuộc đường thẳng $AB$ mà $BM=b$. Tính độ dài đoạn thẳng $AM$ theo $a$ và $b$.
	\loigiai
	{
		Biết điểm $M$ thuộc đường thẳng $AB$ và $BM=b$ nên điểm $M$ không xác định duy nhất trên đường thẳng $AB$. Ta xét hai trường hợp sau
		\begin{center}
			\begin{tikzpicture}[line join = round, line cap = round,>=stealth,font=\footnotesize,scale=1]
			\draw(0,0) node[below]{$A$}--(5,0) node[below]{$B$}--(7,0) node[below]{$M$};
			\draw(0,-1) node[below]{$A$}--(3,-1) node[below]{$M$}--(5,-1) node[below]{$B$};
			\draw(0,-2) node[below]{$M$}--(2,-2) node[below]{$A$}--(7,-2) node[below]{$B$};
			\draw(-.7,0) node[left]{a)} (-.7,-1) node[left]{b)} (-.7,-2) node[left]{c)};
			\foreach \toado in {(0,0),(5,0),(7,0),(0,-1),(3,-1),(5,-1),(0,-2),(2,-2),(7,-2)} \fill \toado circle (1.5pt);
			\end{tikzpicture}
		\end{center}
		\begin{itemize}
			\item Điểm $M$ thuộc tia đối của tia $BA$ (hình a):\\
				Vì hai tia $BA$ và $BM$ đối nhau nên điểm $B$ nằm giữa hai điểm $A$ và $M$. Ta có $AB+BM=AM$, do đó $AM=a+b$.
			\item Nếu điểm $M$ thuộc tia $BA$, ta xét các khả năng sau:
				\begin{itemize}
					\item Nếu $b<a$ thì $BM<BA$ nên điểm $M$ nằm giữa hai điểm $A$ và $B$ (hình b). Ta có $AM+MB=AB$, do đó $AM=a-b$.
					\item Nếu $b=a$ thì $BM=BA$ nên điểm $M$ trùng với điểm $A$. Ta không xét khả năng này do độ dài đoạn thẳng là một số dương.
					\item Nếu $b>a$ thì $BM>BA$ nên điểm $A$ nằm giữa hai điểm $M$ và $B$. Ta có $MA+AB=MB$, do đó $AM=b-a$.
				\end{itemize}
		\end{itemize}
		\emph{Kết luận:} Độ dài đoạn thẳng $AM$ có thể bằng $a+b$, hoặc $|a-b|$ với $a \ne b$.
	}
\end{vd}

\begin{dang}{Nhận biết ba điểm thẳng hàng}
	Nhắc lại \textbf{Dấu hiệu 4} (Dấu hiệu Đ4 - Về thứ tự ba điểm trên một đường thẳng): Nếu có đẳng thức $AM+MB=AB$ thì điểm $M$ nằm giữa hai điểm $A$ và $B$.
\end{dang}
\begin{vd}
	Cho $C$ là một điểm thuộc đoạn thẳng $AB$ và không trùng với hai điểm $A$ và $B$. Hỏi điểm $A$ có nằm giữa hai điểm $B$ và $C$ không?
	\loigiai
	{
		Vì $C$ là một điểm thuộc đoạn thẳng $AB$ nên ba điểm $A$, $B$, $C$ thẳng hàng. Lại có $C$ không trùng với hai điểm $A$ và $B$ nên điểm $C$ nằm giữa hai điểm $A$ và $B$.\\
		Trong ba điểm có một và chỉ một điểm nằm giữa hai điểm còn lại, do đó điểm $A$ không thể nằm giữa hai điểm $B$ và $C$.
	}
\end{vd}

\begin{vd}
	Hỏi ba điểm $A$, $B$ và $C$ có thẳng hàng không, nếu
	\begin{enumerate}
		\item $AB=2$ cm, $BC = 7$ cm và $AC = 5$ cm?
		\item $AB=3$ cm, $BC = 7$ cm và $AC = 5$ cm?
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
			\item Do $AB+AC=2+5=7$ (cm) mà $BC = 7$ (cm) nên $AB+AC=BC$ (hay $BA+AC=BC$).\\
			Vậy điểm $A$ nằm giữa hai điểm $B$ và $C$ (dấu hiệu Đ4).
			\item Giả sử ba điểm $A$, $B$, $C$ thẳng hàng thì phải có một điểm nằm giữa hai điểm còn lại. Ta thấy
				\begin{itemize}
					\item Nếu điểm $A$ nằm giữa hai điểm $B$ và $C$ thì $BA+AC=BC$.\\
					Nhưng $BA+AC=3+5=8$ (cm) mà $BC = 7$ (cm), do đó $BA+AC \ne BC$.\\
					Vậy điểm $A$ không nằm giữa hai điểm $B$ và $C$.
					\item Nếu điểm $B$ nằm giữa hai điểm $A$ và $C$ thì $AB+BC=AC$.\\
					Nhưng $AB+BC=3+7=10$ (cm) mà $AC=5$ (cm), do đó $AB+BC \ne AC$.\\
					Vậy điểm $B$ không nằm giữa hai điểm $A$ và $C$.
					\item Nếu điểm $C$ nằm giữa hai điểm $A$ và $B$ thì $AC+CB=AB$.\\
					Nhưng $AC+CB=5+7=12$ (cm) mà $AB=3$ (cm), do đó $AC+CB \ne AB$.\\
					Vậy điểm $C$ không nằm giữa hai điểm $A$ và $B$.
				\end{itemize}
				Như vậy trong ba điểm $A$, $B$, $C$ không có điểm nào nằm giữa hai điểm còn lại nên ba điểm $A$, $B$, $C$ không thẳng hàng.
		\end{enumerate}
	}
\end{vd}

\begin{vd}
	Cho độ dài ba đoạn thẳng $AB=c$, $BC=a$ và $CA=b$. Hỏi điểm nào nằm giữa hai điểm còn lại, biết $0<b<a$, $0<c<a$ nhưng $a<b+c$?
	\loigiai
	{
		Có $a<b+c$ nên $BC<CA+AB$ hay $BC \ne CA+AB$. Vậy điểm $A$ không nằm giữa hai điểm $B$ và $C$.\\
		Có $b<a$ nên $b<a+c$. Suy ra $CA<CB+BA$ hay $CA \ne CB+BA$. Vậy điểm $B$ không nằm giữa hai điểm $C$ và $A$.\\
		Có $c<a$ nên $c<a+b$. Suy ra $AB<BC+CA$ hay $AB \ne BC+CA$. Vậy điểm $C$ không nằm giữa hai điểm $A$ và $B$.\\
		Vậy trong ba điểm $A$, $B$, $C$ không có điểm nào nằm giữa hai điểm còn lại.
	}
\end{vd}

\subsection{BÀI TẬP}
\begin{bt}%[Nguyễn Tấn Linh, dự án 6SA19-NC1]%[Đỗ Đức Thảo, dự án 6SA19-NC1]
	Cho bốn điểm $A,B,C,D$ theo thứ tự đó cùng thuộc một đường thẳng $xy$.
	\begin{enumerate}
		\item Trên đường thẳng $xy$ có bao nhiêu đoạn thẳng? Liệt kê các đoạn thẳng đó.
		\item Chỉ ra các đoạn thẳng là tổng các đoạn thẳng khác.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Trên $xy$ có $6$ đoạn thẳng: $AB,AC,AD,BC,BD$ và $CD$.
			\begin{center}
				\begin{tikzpicture}[line join = round, line cap = round,>=stealth,thick,font=\footnotesize,scale=1]
				\tkzDefPoints{0/0/x,2/0/A,5/0/B,6.5/0/C,9/0/D,11/0/y}
				\tkzDrawSegments(x,y)
				\tkzDrawPoints[fill=black,size=3](A,B,C,D)
				\tkzLabelPoints[below](x,A,B,C,D,y)
				\end{tikzpicture}
			\end{center}
			\item Ta có: $AC=AB+BC$. Ngoài ra:
			\allowdisplaybreaks
			\begin{eqnarray*}
				AD&=&AC+CD\\
				&=& AB+BD\\
				&=& (AB+BC)+CD\\
				&=& AB+(BC+CD)\\
				BD&=&BC+CD
			\end{eqnarray*}
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Nguyễn Tấn Linh, dự án 6SA19-NC1]
	Cho ba điểm $A$, $B$, $C$ mà độ dài của $3$ đoạn thẳng thỏa mãn $AB+BC>AC$. Vậy có thể kết luận là ba điểm $A$, $B$, $C$ không thẳng hàng được không?
	\loigiai
	{
		Không thể kết luận được vì: Chẳng hạn ta chọn $AB=2$, $BC=7$ và $CA=5$ thì $AB+BC>AC$ $(9>5)$ nhưng $AB+CA=BC$ $(2+5=7)$ nên có điểm $A$ nằm giữa hai điểm $B$ và $C$.
	}
\end{bt}
\begin{bt}%[Nguyễn Tấn Linh, dự án 6SA19-NC1]
	Hỏi ba điểm $A$, $B$, $C$ có thẳng hàng không, nếu
	\begin{enumerate}
		\item $AB=\dfrac{1}{2}$, $BC=\dfrac{1}{3}$, $CA=\dfrac{1}{6}$ (cm)?
		\item $AB=5$, $BC=11$, $CA=7$ (cm)?
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
			\item Có $\dfrac{1}{2}=\dfrac{1}{3}+\dfrac{1}{6}\Rightarrow BC+CA=AB\Rightarrow$ điểm $C$ nằm giữa hai điểm $A$ và $B$. Vậy ba điểm $A$, $B$, $C$ thẳng hàng.
			\item Ta có:
			\begin{eqnarray*}
				AB+BC=11>7=CA\\
				AB+AC=12>11=BC\\
				BC+CA=18>5=AB
			\end{eqnarray*}
			Do đó không có ba điểm nào thẳng hàng.
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Nguyễn Tấn Linh, dự án 6SA19-NC1]
	Cho ba điểm thẳng hàng $A$, $B$, $C$. Hỏi điểm nào nằm giữa hai điểm còn lại, nếu:
	\begin{enumerate}
		\item $AB=2$, $BC=13$, $CA=11$ (cm)?
		\item $AC=7$ và $BC=11$ (cm)?
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
			\item Ta có: $AB+CA=BC$. Do đó điểm $A$ nằm giữa hai điểm $B$ và $C$.
			\item Có hai khả năng xảy ra: hoặc $C$ nằm giữa hai điểm $B$ và $A$, hoặc $A$ nằm giữa hai điểm $B$ và $C$.
			\begin{center}

				\tkzDefPoints{0/0/A,3/0/C,7/0/B}

				\draw[dashed] (C) to[in=150,out=30]node[above,midway]{$11$} (B) ;
				\tkzDrawSegments(A,B)
				\tkzLabelPoints[below](A,C,B)
				\tkzDrawPoints[fill=black,size=3](A,B,C)
				\end{tikzpicture}
				
				\begin{tikzpicture}[line join = round, line cap = round,>=stealth,thick,font=\footnotesize,scale=1]
				\tkzDefPoints{0/0/B,3/0/A,7/0/C}
				\draw[dashed] (B) to[in=-150,out=-30]node[below,midway]{$11$} (C) ;
				\draw[dashed] (A) to[in=150,out=30]node[above,midway]{$7$} (C) ;
				\tkzDrawSegments(B,C)
				\tkzLabelPoints[below](A,C,B)
				\tkzDrawPoints[fill=black,size=3](A,B,C)
				\end{tikzpicture}
			\end{center}
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Nguyễn Tấn Linh, dự án 6SA19-NC1]
	Cho bốn điểm $A$, $B$, $C$, $D$ thẳng hàng theo thứ tự đó.
	\begin{enumerate}
		\item So sánh các đoạn thẳng $AB$ và $CD$ biết $AC=BD$.
		\item So sánh các đoạn thẳng $AC$ và $BD$ biết $AB=CD$.
	\end{enumerate}
	\loigiai
	{
		\begin{center}
			\begin{tikzpicture}[line join = round, line cap = round,>=stealth,thick,font=\footnotesize,scale=1]
			\tkzDefPoints{0/0/A,3/0/B,5/0/C,8/0/D}
			\tkzDrawSegments(A,D)
			\tkzDrawPoints[fill=black,size=3](A,B,C,D)
			\tkzLabelPoints[below](A,B,C,D)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Ta có:
			\begin{center}
				$\begin{aligned}
				AC=AB+BC\\
				DB=DC+BC
				\end{aligned}$. Mà $AC=BD$ nên $AB=CD$.
			\end{center}
			\item Ta có:
			\begin{center}
				$\begin{aligned}
				AC=AB+BC\\
				DB=DC+BC
				\end{aligned}$. Mà $AB=CD$ nên $AC=BD$.
			\end{center}
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Nguyễn Tấn Linh, dự án 6SA19-NC1]
	Ba điểm $O$, $A$, $B$ thuộc đường thẳng $xy$. Tính độ dài đoạn thẳng $AB$ biết $OA+OB=a$ và $OA-OB=b$, $(0<b<a)$.
	\loigiai
	{
		Xét ba trường hợp:
		\begin{itemize}
			\item \immini
			{
				Điểm $O$ nằm giữa hai điểm $A$ và $B$.\\
				Ta có: $AB=OA+OB=a$.
			}
			{
				\begin{tikzpicture}[line join = round, line cap = round,>=stealth,thick,font=\footnotesize,scale=1]
				\tkzDefPoints{0/0/A,3/0/O,5/0/B}
				\tkzDrawSegments(A,B)
				\tkzDrawPoints[fill=black,size=3](A,O,B)
				\tkzLabelPoints[below](A,O,B)
				\end{tikzpicture}
			}
			\immini
			{
				\item Điểm $B$ nằm giữa hai điểm $O$ và $A$.\\
				Ta có $OB+AB=OA$ nên $AB=OA-OB=b$.
			}
			{
				\begin{tikzpicture}[line join = round, line cap = round,>=stealth,thick,font=\footnotesize,scale=1]
				\tkzDefPoints{0/0/O,3/0/B,5/0/A}
				\tkzDrawSegments(O,A)
				\tkzDrawPoints[fill=black,size=3](A,O,B)
				\tkzLabelPoints[below](A,O,B)
				\end{tikzpicture}
			}
			\item Điểm $A$ nằm giữa hai điểm $O$ và $B$, khi đó $OA<OB$, trái với đề bài là $OA-OB>0$. Do đó trường hợp này không xảy ra.
		\end{itemize}
	}
\end{bt}
\begin{bt}%[Nguyễn Tấn Linh, dự án 6SA19-NC1]
	Cho đoạn thẳng $AB=5$ (cm). Trên tia $BA$ lấy điểm $M$ sao cho $AM=2$ (cm).
	\begin{enumerate}
		\item Trong $3$ điểm $A$, $B$, $M$, điểm nào nằm giữa hai điểm còn lại?
		\item Tính độ dài đoạn thẳng $MB$.
		\item Lấy điểm $N$ thuộc tia đối của tia $BA$ sao cho $BN=1$ (cm). Tính $MN$.
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
			\item Có hai trường hợp:\\
			Trường hợp 1: Điểm $M$ nằm giữa hai điểm $A$ và $B$.
			\begin{center}
				\begin{tikzpicture}[line join = round, line cap = round,>=stealth,thick,font=\footnotesize,scale=1]
				\tkzDefPoints{0/0/B,5/0/A,3/0/M,6/0/x}
				\tkzDrawSegments(B,x)
				\tkzLabelPoints[below](A,B,M)
				\tkzDrawPoints[fill=black,size=3](A,B,M)
				\end{tikzpicture}
			\end{center}
			Trường hợp 2: Điểm $A$ nằm giữa hai điểm $B$ và $M$.
			\begin{center}
				\begin{tikzpicture}[line join = round, line cap = round,>=stealth,thick,font=\footnotesize,scale=1]
				\tkzDefPoints{0/0/B,5/0/A,7/0/M,8/0/x}
				\tkzDrawSegments(B,x)
				\tkzLabelPoints[below](A,B,M)
				\tkzDrawPoints[fill=black,size=3](A,B,M)
				\end{tikzpicture}
			\end{center}
			\item Trường hợp 1: Điểm $M$ nằm giữa hai điểm $A$ và $B$. Khi đó $BM+MA=AM\Rightarrow MB=AB-MB=3$ cm.\\
			Trường hợp 2: Điểm $A$ nằm giữa hai điểm $B$ và $M$. Khi đó $BM=BA+AM=7$ cm.
			\item Trường hợp 1: Điểm $M$ nằm giữa hai điểm $A$ và $B$. Khi đó: $MN=MB+BN=4$ cm.
			\begin{center}
				\begin{tikzpicture}[line join = round, line cap = round,>=stealth,thick,font=\footnotesize,scale=1]
				\tkzDefPoints{0/0/B,5/0/A,3/0/M,6/0/x,-1/0/N}
				\tkzDrawSegments(N,x)
				\tkzLabelPoints[below](A,B,M,N)
				\tkzDrawPoints[fill=black,size=3](A,B,M,N)
				\end{tikzpicture}
			\end{center}
			Trường hợp 2: Điểm $A$ nằm giữa hai điểm $B$ và $M$. Khi đó $MN=MA+AB+BN=8$ cm.
			\begin{center}
				\begin{tikzpicture}[line join = round, line cap = round,>=stealth,thick,font=\footnotesize,scale=1]
				\tkzDefPoints{0/0/B,5/0/A,7/0/M,8/0/x,-1/0/N}
				\tkzDrawSegments(N,x)

\begin{bt}%[Nguyễn Tấn Linh, dự án 6SA19-NC1]
	Cho đoạn thẳng $AB=7$ (cm) và điểm $M$ nằm giữa hai điểm $A$, $B$ sao cho $BM=5$ (cm). Trên tia đối của tia $MA$ lấy điểm $N$ sao cho $MN=7$ (cm). Giải thích vì sao $AM$ bằng $BN$.
	\loigiai
	{
		\begin{center}
			\begin{tikzpicture}[line join = round, line cap = round,>=stealth,thick,font=\footnotesize,scale=1]
			\tkzDefPoints{0/0/A,7/0/B,2/0/M,9/0/N}
			\tkzDrawSegments(A,N)
			\tkzDrawPoints[fill=black,size=3](A,B,M,N)
			\tkzLabelPoints[below](A,B,M,N)
			\end{tikzpicture}
		\end{center}
		Ta có: $AM+MB=AB\Rightarrow AM=AB-BM=2$ cm.\\
		Ngoài ra, $MB+BN=MN\Rightarrow BN=MN-MB=2$ cm.\\
		Do đó $AM=BN(=2)$ cm.
	}
\end{bt}
