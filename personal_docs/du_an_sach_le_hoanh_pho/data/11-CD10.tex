\begin{bt}%[Sách bồi HSG, tác giả Lê Hoành Phò]%[Nguyễn Trung Kiên]%[1D5B2-1]
    Cho hàm số $f(x)=\sqrt{x^2-2x}$. Giải bất phương trình
    \begin{listEX}[2]
        \item $f'(x)<0$.
        \item $f'(x)\leq f(x)$.
    \end{listEX}
    \loigiai{
    Hàm số $f(x)=\sqrt{x^2-2x}$ có tập xác định $\mathscr{D}=(-\infty;0] \cup [2;+\infty)$.\\
    Ta có $f'(x)=\dfrac{\left(x^2-2x\right)'}{2\sqrt{x^2-2x}} = \dfrac{x-1}{\sqrt{x^2-2x}},\, \forall x\in (-\infty;0) \cup (2;+\infty)$.\\
    Điều kiện của các bất phương trình là $x<0$ hoặc $x>2$.
    \begin{enumerate}
        \item $f'(x)<0 \Leftrightarrow \dfrac{x-1}{\sqrt{x^2-2x}}<0 \Leftrightarrow \heva{&x-1<0\\&x^2-2x>0} \Leftrightarrow x<0$.
        \item Bất phương trình $f'(x)\leq f(x)$
        $$\Leftrightarrow \dfrac{x-1}{\sqrt{x^2-2x}}\leq \sqrt{x^2-2x} \Leftrightarrow \heva{&\hoac{&x<0\\&x>2}\\&x-1\leq x^2-2x} \Leftrightarrow \heva{&\hoac{&x<0\\&x>2}\\&\hoac{&x\leq \dfrac{3-\sqrt{5}}{2}\\&x\geq \dfrac{3+\sqrt{5}}{2}}} \Leftrightarrow \hoac{&x<0\\&x\geq \dfrac{3+\sqrt{5}}{2}.}$$
    \end{enumerate}}
\end{bt}

        \[(m+1)^2+(-m)^2\geq (m+2)^2 \Leftrightarrow  m^2-2m-3\geq 0 \Leftrightarrow \hoac{&m\leq -1\\&m\geq 3.}\]
