\begin{ex}%[Đề thi vào 10, chuyên Vinh vòng 2, 2017]%[9D4K7]
 Giải phương trình
\begin{itemize}
\item[a)]  $(x-1)^2+\dfrac{2}{x^2-2x+3} = \dfrac{5}{2}.$
\item [b)] $3x\sqrt{x^2+x+1} = 3x^2+x+1.$
\end{itemize}
\loigiai{
a) Phương trình đã cho tương đương với phương trình $x^2-2x+1+\dfrac{2}{x^2-2x+3} = \dfrac{5}{2}.$ Đặt $t = x^2-2x+3 \ge 2.$ Phương trình trở thành $t^2-\dfrac{9}{2}t+2=0 \Leftrightarrow \hoac{&t = 4, \\& t=\dfrac{1}{2} \text{ (không thỏa mãn)}}.$ Suy ra $x^2-2x+3 =4 \Leftrightarrow x = 1\pm \sqrt{2}.$

b)

 $\begin{array}{ll}
\text{Phương trình đã cho} &   \Leftrightarrow 3x\sqrt{x^2+x+1}=2x^2+(x^2+x+1)\\ 
& \Leftrightarrow  2\left(\dfrac{x}{\sqrt{x^2+x+1}}\right)^2-3\dfrac{x}{\sqrt{x^2+x+1}}+1=0.
\end{array}$

 Đặt $t=\dfrac{x}{\sqrt{x^2+x+1}},$  phương trình trở thành $2t^2-3t+1=0 \Leftrightarrow \hoac{& t= 1\\& t= \dfrac{1}{2}}.$

Với $t=1,$ ta có $x=\sqrt{x^2+x+1}$  (Vô nghiệm).

Với $t=\dfrac{1}{2},$ ta có $2x=\sqrt{x^2+x+1} \Leftrightarrow\heva{&x \ge 0 \\ & 4x^2=x^2+x+1} \Leftrightarrow x=\dfrac{1+\sqrt{13}}{6}$.}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Vinh vòng 2, 2017]%[9D5G5]
 Giải hệ phương trình 
 $\heva{ & x+\dfrac{2}{x} = 2y+\dfrac{1}{y} \\ & 
\sqrt{x-1}+\sqrt{2-y} = 1}$.
\loigiai{ Điều kiện: $x \ge 1, y \le 2, y \ne 0.$
Phương trình đầu tương đương với 
\begin{center}
$x^2y+2y=2y^2+x \Leftrightarrow (x-2y)(xy-1)=0 \Leftrightarrow \heva{& x-2y=0 \\ & xy-1=0}.$
\end{center}
Với $x=2y$, thay vào phương trình còn lại ta được
\begin{center}
$\sqrt{2y-1} +\sqrt{2-y} = 1 \Leftrightarrow 2\sqrt{(2y-1)(2-y)} = -y$ (Vô nghiệm).
\end{center} 

Với $y=\dfrac{1}{x},$ thay vào phương trình còn lại ta được $\sqrt{x-1}+\sqrt{2-\dfrac{1}{x}} = 1.$ Vì $x \ge 1$ nên $VT \ge 1 =VP,$ do đó  $x=1.$

Vậy hệ có nghiệm $(x;y) = (1;1)$.}

\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Vinh vòng 2, 2017]%[9D5G5]
Tìm tất cả các bộ số nguyên dương $(a;b;c)$ thỏa mãn $a<b<c$ sao cho đa thức $P(x) = x(x+a)(x+b)(x+c) +1$ là bình phương của một đa thức có hệ số nguyên.

\loigiai{ Do đa thức $P(x)$ có  bậc cao nhất là $4$ và hệ số của nó là $1$, nên ta có thể giả sử $P(x) = (x^2+mx+n)^2$ với $m,n \in \mathbb{Z}.$

Từ $P(0) = 1,$ suy ra $n=\pm 1.$

Với $n=1,$ ta có $x(x+a)(x+b)(x+c)+1=(x+mx+1)^2 \Leftrightarrow x(x+a)(x+b)(x+c) = x(x+m)(x^2+mx+2)$. Suy ra $m>0$ và phương trình $x^2+mx+2 = 0$ có hai nghiệm nguyên âm phân biệt $\Leftrightarrow \Delta = m^2-8 = k^2, k \in \mathbb{N}^* \Leftrightarrow (m+k)(m-k) = 8.$ Do $m>k>0$ và cùng chẵn nên 

$\heva{&m+k=4\\& m-k=2}. $ Suy ra $m=3$.
Vậy $P(x) = x(x+1)(x+2)(x+3).$

Với $n=-1$, tương tự ta có 
\begin{center}
$x(x+a)(x+b)(x+c) = x(x+m)(x^2+mx-2)$
\end{center}
Vì phương trình $x^2+mx-2 =0$ có nghiệm nguyên âm nên trường hợp này không thỏa mãn yêu cầu đề bài.

Vậy $(a;b;c) = (1;2;3).$}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Vinh vòng 2, 2017]%[9H3B8]%[9H3K7]%[9H3G8]
Cho nửa đường tròn tâm $O$  đường kính $AB =2R$  và điểm 
$M$  nằm trên nửa đường tròn đó ($M$  khác  $A$  và $B$). 
 Gọi  $H$  là hình chiếu vuông góc của  $M$  lên $AB$.  
Gọi  $I,I_1,I_2$  lần lượt là tâm đường tròn nội tiếp của các tam giác $MAB, MAH, MBH$. Các đường thẳng 
$MI_1,MI_2$  cắt  $AB$  tương ứng tại  $E$ và $F.$
  
a)  Chứng minh rằng $AI_1 \perp MI_2$ và $MI \perp I_1I_2.$
  
b)  Chứng minh rằng  $EFI_2I_1$  là tứ giác nội tiếp đường tròn và các đường thẳng $MH,EI_2,FI_1$ đồng quy. 

c)  Tìm giá trị lớn nhất của  $I_1I_2$ theo $R$ khi $M$ thay đổi trên nửa đường tròn đã cho.
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1]

	\tkzClip[space=.5]

\end{center}

a) Gọi $K = AI_1 \cap MF.$ Vì $\widehat{MAB} = \widehat{BMH}$ (cùng phụ với $\widehat{MBA}$), suy ra $\widehat{MAK} = \widehat{BMF}.$ Mà $\widehat{BMF}+\widehat{AMF} = 90^{\circ},$  suy ra $\widehat{MAK}+\widehat{AMF} = 90^{\circ} \Rightarrow \widehat{AKM} =90^{\circ}$ hay $AI_1 \perp MI_2$. 

Tương tự  $BI_2 \perp MI_1,$ suy ra $I$ là trực tâm của tam giác $MI_1I_2$ hay  $MI \perp I_1I_2.$

b) Vì có $AK$ vừa là đường cao vừa là phân giác nên  tam giác $AMF$ cân tại $A$. Suy ra $K$ là trung điểm của $MF$. Vì $\widehat{EMF} =\dfrac{1}{2}\widehat{AMB} = 45^\circ$ và $ \widehat{I_1KM} = 90^\circ$ nên tam giác $MI_1F$ vuông cân tại $K$. 

Suy ra $MI_1=\sqrt{2}MK = \dfrac{MF}{\sqrt{2}}$. Tương tự ta có $MI_2 = \dfrac{ME}{\sqrt{2}}.$ Suy ra $\dfrac{MI_1}{MF} = \dfrac{MI_2}{ME} = \dfrac{1}{\sqrt{2}} \Rightarrow \Delta MI_1I_2 \backsim \Delta MFE$ (c.g.c)
$\Rightarrow \widehat{MI_1I_2} = \widehat{MFE} \Rightarrow $ tứ giác $EFI_2I_1$ nội tiếp.

Vì $\dfrac{MI_1}{MF} = \dfrac{1}{\sqrt{2}}$ và $\widehat{I_1MF} = 45^\circ$ nên tam giác $MI_1F$ vuông cân tại $I_1 \Rightarrow \widehat{MI_1F} = 90^\circ \Rightarrow \widehat{FI_1E} = 90^\circ$. Tương tự ta cũng có $\widehat{EI_2F} = 90^\circ.$ Suy ra $MH,EI_2,FI_1$ là các đường cao của tam giác $MEF$, nên $MH,EI_2,FI_1$ đồng quy.

c) Vì tam giác $AMF$ cân tại $A$ nên $AF=AM$. Tương tự $BE = BM.$ Suy ra
$EF = AF+BE-AB = AM +BM -AB \le \sqrt{2(AM^2+BM^2)}-AB = \sqrt{2AB^2}-AB = 2R(\sqrt{2}-1).$ Từ câu b) ta có  $\dfrac{I_1I_2}{EF}=\dfrac{MI_1}{MF}=\dfrac{1}{\sqrt{2}}.$

Suy ra $I_1I_2 = \dfrac{EF}{\sqrt{2}} \le \dfrac{2R(\sqrt{2}-1)}{\sqrt{2}}=(2-\sqrt{2})R.$

Dấu đẳng thức xảy ra khi $AM = BM$ hay $M$ là điểm chính giữa $AB$. Vậy giá  trị lớn nhất của $I_1I_2$ là $(2-\sqrt{2})R.$
}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Vinh vòng 2, 2017]%[9D5G1]%
Cho các số thực $a,b,c$ thỏa mãn $1\le a,b,c \le 2$. Tìm giá trị lơn nhất của biểu thức 
\begin{center}
$P=\dfrac{a^2+b^2+c^2}{ab+bc+ca}+\dfrac{(a+b+c)^3}{3(a^3+b^3+c^3)}.$
\end{center}
\loigiai{
Ta có $3(a^3+b^3+c^3) - (a+b+c)(a^2+b^2+c^2) = (a-b)(a^2-b^2)+(b-c)(b^2-c^2)+(c-a)(c^2-a^2) \ge 0.$ Suy ra 
$3(a^3+b^3+c^3)  \ge  (a+b+c)(a^2+b^2+c^2).$ Do đó

\begin{center}
$\begin{array}{ll}
P & \le \dfrac{a^2+b^2+c^2}{ab+bc+ca}+\dfrac{(a+b+c)^3}{(a+b+c)(a^2+b^2+c^2)} \\ 
&=\dfrac{a^2+b^2+c^2}{ab+bc+ca}+\dfrac{2(ab+bc+ca)}{a^2+b^2+c^2}+1 \\
\end{array}$
\end{center}
Đặt $t = \dfrac{a^2+b^2+c^2}{ab+bc+ca}, t \ge 1,$ khi đó $P \le t+\dfrac{2}{t}+1.$

Vì $a,b,c \in [1,2],$ nên $a+b \ge c, b+c \ge a, c+a \ge b.$ Suy ra $c(a+b)+a(b+c)+b(c+a) \ge a^2+b^2+c^2$, hay $ t \le 2.$ Từ $t \in [1;2] $  ta nhận được $(t-1)(t-2)\le 0 \Leftrightarrow t+\dfrac{2}{t}+1 \le 4.$ Dấu đẳng thức khi $a=b=c.$ Vậy giá trị lớn nhất của $P$ là $4.$
}
\end{ex}
