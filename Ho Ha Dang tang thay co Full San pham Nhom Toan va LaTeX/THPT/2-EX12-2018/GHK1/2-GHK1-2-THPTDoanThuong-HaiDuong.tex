Do hoàn cảnh kinh tế không được tốt nên gia đình rất lo lắng về việc đóng học phí cho Việt, gia đình em đã quyết định bán một phần mảnh đất hình chữ nhật có chu vi $50$ m, lấy tiền lo cho việc học của Việt cũng như tương lai của em. Mảnh đất còn lại sau khi bán là một hình vuông cạnh bằng chiều rộng của mảnh đất hình chữ nhật ban đầu. Tìm số tiền lớn nhất mà gia đình Việt nhận được khi bán đất, biết giá tiền $1 m^2$ đất khi bán là $1500000$ VN đồng.
	\choice
	{ $115687500$ VN  đồng}
	{ $114187500$ VN  đồng}
	{\True $117187500$ VN  đồng}
	{ $112687500$ VN  đồng}
	
	\loigiai{
		Gọi $x$ là chiều rộng mảnh đất hình chữ nhật ban đầu (đơn vị mét), $0 <x <25$.
		\\ Khi đó $25-x$ m là độ dài cạnh thứ hai của mảnh đất ban đầu.
		\\ Diện tích mảnh đất ban đầu bằng $x(25-x) $ , diện tích phần đất còn lại sau khi bán là $x^2$ (đơn vị mét vuông).
		\\ Do đó diện tích phần đất đã bán là $$x(25-x)-x^2=-2(x^2-\dfrac{25}{2} x)=-2(x^2-2.\dfrac{25}{4}x+(\dfrac{25}{4})^2-(\dfrac{25}{4})^2)=-2(x-\dfrac{25}{4})^2+\dfrac{625}{8} \geqslant \dfrac{625}{8}.$$
		Suy ra số tiền lớn nhất mà gia đình Việt nhận được là $1500000.\dfrac{625}{8}=117187500$ VN đồng.
		
	}
\end{ex}

\begin{ex}%[Đề KSCL lần 1 2017-2018, Đoàn Thượng, Hải Dương]%[2D1K5]
	Gọi $d$ là tiếp tuyến với đồ thị hàm số $(C): y=\dfrac{1}{x^2-1}$ song song với trục hoành. Tìm hoành độ tiếp điểm $x_0$ của $d$ và $(C)$.
	\choice
	{$x_0=1$}
	{$x_0=2$}
	{$x_0=-1$}
	{\True $x_0=0$}
	\loigiai{
		$y'=\dfrac{-2x}{{\left( x^2-1\right) }^2}$\\
	Ta có: $d: y=x \dfrac{-2x_0}{{\left( {x_0}^2-1\right) }^2} +\dfrac{2{x_0}^2}{\left( {x_0}^2-1\right)^2} +\dfrac{1}{{x_0}^2-1}$\\
	$d$ song song với trục hoành nên $\dfrac{-2x_0}{{\left( {x_0}^2-1\right) }^2} =0$ \\
	Suy ra $x_0=0$}
\end{ex}
\begin{ex}%[Đề KSCL lần 1 2017-2018, Đoàn Thượng, Hải Dương]%[2D1K4]
	Cho hàm số $y=\dfrac{\sqrt{x^2-1}}{x}$. Hãy chọn mệnh đề \textbf{đúng} trong các mệnh đề sau:
	\choice
	{\True Đồ thị hàm số có hai tiệm cận ngang là $y=1$ và $y=-1$}
	{Đồ thị hàm số có tiệm cận ngang là $y=-1$, có tiệm cận đứng là $x=0$}
	{Đồ thị hàm số có tiệm cận ngang là $y=1$, có tiệm cận đứng là $x=0$}
	{Đồ thị hàm số có hai tiệm cận ngang là $y=1,y=-1$, có tiệm cận đứng là $x=0$}
	\loigiai{
		$\displaystyle \lim_{x \to +\infty }\dfrac{\sqrt{x^2-1}}{x} =\displaystyle \lim_{x \to +\infty }\dfrac{\sqrt{1-\dfrac{1}{x^2}}}{1}=1$\\
		$\displaystyle \lim_{x \to -\infty }\dfrac{\sqrt{x^2-1}}{x} =\displaystyle \lim_{x \to -\infty }\dfrac{\sqrt{1-\dfrac{1}{x^2}}}{-1}=-1$\\
		$\displaystyle \lim_{x \to 0^+ }\dfrac{\sqrt{x^2-1}}{x}$ không xác định.}
\end{ex}
