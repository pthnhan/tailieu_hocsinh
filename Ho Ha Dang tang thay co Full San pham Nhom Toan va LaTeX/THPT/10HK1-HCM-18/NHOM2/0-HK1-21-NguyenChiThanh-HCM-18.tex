\begin{bt}%[Đề HKI Nguyễn Chí Thanh-HCM-2017-2018]%[Hồ Minh Hòa, 10HKI-HCM]%[0D2B1-4]
	Xét tính chẵn, lẻ của hàm số sau $y=f(x)=\dfrac{\big|x-2018\big|-\big|x+2018\big|}{\sqrt{x^2-9}}$.
	\loigiai{
		Tập xác định $\mathscr{D}=(-3;3)$ là tập đối xứng.\\
		$\begin{aligned}
		\text{Khi đó\;} \forall x \in \mathscr{D} \text{\;ta có\;}
		f(-x)&=\dfrac{\big|-x-2018\big|-\big|-x+2018\big|}{\sqrt{(-x)^2-9}}\\
		&=\dfrac{\big|x+2018\big|-\big|x-2018\big|}{\sqrt{x^2-9}}\\
		&=-\dfrac{\big|x-2018\big|-\big|x+2018\big|}{\sqrt{x^2-9}}\\
		&=-f(x).	
		\end{aligned}$\\
		Vậy hàm số đã cho là hàm số lẻ.
	}
\end{bt}
\begin{bt}%[Đề HKI Nguyễn Chí Thanh-HCM-2017-2018]%[Hồ Minh Hòa, 10HKI-HCM]%[0D2B3-2]
	Xác định parabol $(P)\colon y=ax^2+bx+c$, biết $(P)$ có trục đối xứng $x=2$ và đi qua các điểm $A(4;2)$, $B(1;-1)$.
	\loigiai{
	Do $(P)$ đi qua $A(4;2)$, $B(1;-1)$ nên ta có $\heva{&16a+4b+c=2&(1)\\&a+b+c=-1 &(2)}$\\
	Mặt khác, $(P)$ có trục đối xứng là $x=2$ nên $-\dfrac{b}{2a}=2\Rightarrow 4a+b=0\quad(3)$\\
	Giải hệ $(1)$, $(2)$ và $(3)$ ta được $\heva{&a=1\\&b=-4\\&z=2.}$\\
	Vậy parabol $(P)$ có phương trình $y=x^2-4x+2$.
}
\end{bt}
\begin{bt}%[Đề HKI Nguyễn Chí Thanh-HCM-2017-2018]%[Hồ Minh Hòa, 10HKI-HCM]%[0D3B2-4]
	Giải phương trình $\sqrt{x^2-2x+6}+1=2x$.
	\loigiai{
$\begin{aligned}
\text{Ta có\;}	
	\sqrt{x^2-2x+6}+1=2x &\Leftrightarrow \heva{&2x-1\geq 0\\ &\sqrt{x^2-2x+6}=2x-1}\\
	&\Leftrightarrow \heva{&x\geq \dfrac{1}{2}\\ &x^2-2x+6=(2x-1)^2}\\
	&\Leftrightarrow \heva{&x \geq \dfrac{1}{2}\\ & -3x^2+2x+5=0}\\
	&\Leftrightarrow \heva{&x \geq \dfrac{1}{2}\\ & \hoac{&x=\dfrac{5}{3}\\&x=-1}}\\
	&\Leftrightarrow x=\dfrac{5}{3}.
\end{aligned}$\\
	Vậy tập nghiệm của phương trình là $S=\left\{\dfrac{5}{3}\right\}$.
}
\end{bt}
\begin{bt}%[Đề HKI Nguyễn Chí Thanh-HCM-2017-2018]%[Hồ Minh Hòa, 10HKI-HCM]%[0D3B3-4]
	Giải hệ phương trình $\heva{&x+y+8=0\\&x^2+y^2+6x+2y=0}$
	\loigiai{
	Từ phương trình thứ nhất của hệ suy ra $x=-y-8$ thay vào phương trình thứ hai ta được
	$$(-y-8)^2+y^2+6(-y-8)+2y=0\Leftrightarrow 2y^2+12y+16=0 \Leftrightarrow \hoac{&y=-2\\&y=-4.}$$
	Với $y=-2\Rightarrow x=-6$.\\
	Với $y=-4\Rightarrow x=-4$.\\
	Vậy hệ phương trình đã cho có tập nghiệm $S=\left\{(-6;-2),(-4;-4)\right\}$.
}
\end{bt}
\begin{bt}%[Đề HKI Nguyễn Chí Thanh-HCM-2017-2018]%[Hồ Minh Hòa, 10HKI-HCM]%[0D3B3-5]
	Tính tuổi của hai cha con hiện nay, biết rằng trước đây hai năm thì tuổi cha gấp bảy lần tuổi con và sau ba năm nữa thì tuổi của cha chỉ còn gấp bốn lần tuổi con.
	\loigiai{
	Gọi $x, y$ lần lượt là số tuổi của cha và con hiện nay $(x>y>0; \; x,y\in \mathbb{N^*})$.\\
	Theo đề bài ta có hệ phương trình
	$$\heva{&x-2=7(y-2)\\&x+3=4(y+3)}\Leftrightarrow \heva{&x-7y=-12\\&x-4y=9}\Leftrightarrow \heva{&x=37\\&y=7.}$$
	Vậy tuổi cha là $37$ tuổi  và tuổi con là $7$ tuổi.
}
\end{bt}
\begin{bt}%[Đề HKI Nguyễn Chí Thanh-HCM-2017-2018]%[Hồ Minh Hòa, 10HKI-HCM]%[0D4K1-2]
	Chứng minh bất đẳng thức $(a+1)\sqrt{b}+(b+1)\sqrt{a}\leq 1+a+b+ab$; $\forall a;b\neq 0$.
	\loigiai{
		Nhận thấy $a$, $b$ là hai số dương nên theo bất đẳng thức Cô-si ta có
		$$(a+1)\sqrt{b\cdot 1}\leq (a+1)\cdot \dfrac{b+1}{2}$$
		$$(b+1)\sqrt{a\cdot 1}\leq (b+1)\cdot \dfrac{a+1}{2}$$
		Lấy vế cộng vế ta được 
		$$(a+1)\sqrt{b}+(b+1)\sqrt{a}\leq (a+1)(b+1)\leq 1+a+b+ab. $$ 
		Dấu bằng xảy ra khi $a=b=1$.
	}
\end{bt}
\begin{bt}%[Đề HKI Nguyễn Chí Thanh-HCM-2017-2018]%[Hồ Minh Hòa, 10HKI-HCM]%[0H2K3-1]%[0H2K3-1]
Cho tam giác $ABC$ có $AB=5$, $AC=8$, $\widehat{BAC}=60^{\circ}$.
\begin{enumerate}
\item Tính độ dài cạnh $BC$ và bán kính đường tròn ngoại tiếp tam giác $ABC$.
\item Tính diện tích tam giác $ABC$ và bán kính đường tròn nội tiếp tam giác $ABC$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Theo định lí cô-sin trong tam giác $ABC$ ta có $$BC=\sqrt{AB^2+AC^2-2AB\cdot AC\cdot\cos\widehat{BAC}}=\sqrt{25+64-2\cdot 5\cdot 8\cdot\dfrac{1}{2}}=2\sqrt{13}.$$
Theo định lí sin trong tam giác $ABC$ ta có $\dfrac{BC}{\sin\widehat{BAC}}=2R\Rightarrow R=\dfrac{BC}{2\sin\widehat{BAC}}=\dfrac{2\sqrt{39}}{3}$.
\item Diện tích tam giác $ABC$ là $S=\dfrac{1}{2}AB\cdot AC \cdot \sin\widehat{BAC}=10\sqrt{3}$.\\
Mà ta lại có $S=\dfrac{(AB+BC+AC)\cdot r}{2}\Rightarrow r=\dfrac{2S}{AB+BC+CA}=\dfrac{20\sqrt{3}}{13+2\sqrt{13}}$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HKI Nguyễn Chí Thanh-HCM-2017-2018]%[Hồ Minh Hòa, 10HKI-HCM]%[0H3K1-5]
Trong mặt phẳng $(Oxy)$, cho hai điểm $A(1;3)$, $B(4;2)$. Chứng minh tam giác $OAB$ vuông cân và tính diện tích tam giác $OAB$.
\loigiai{
Ta có $OA=\sqrt{1^2+3^2}=\sqrt{10}$, $AB=\sqrt{(4-1)^2+(2-3)^2}=\sqrt{10}$ và $OB=\sqrt{4^2+2^2}=2\sqrt{5}$.\\
Vì $OA=AB$ và $OA^2+AB^2=OB^2$ nên tam giác $OAB$ vuông cân tại $A$.\\
Diện tích tam giác $OAB$ là $S=\dfrac{1}{2}OA\cdot AB=5$ (đvdt).
}
\end{bt}
