$\Leftrightarrow \left|3A-4B\right|=3\sqrt{A^2+B^2}$\\

    \item Với $7B=24A$, chọn $A=7,B=24$, khi đó đường thằng $d$ là $7x+24y-134=0$.
\end{enumerate}
