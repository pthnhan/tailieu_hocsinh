\subsection{Tóm tắt lý thuyết}
\subsubsection{Vẽ đoạn thẳng trên tia}
\textbf{\textit{Cách 1:}} Dùng thước đo có chia khoảng (tương tự như đo đoạn thẳng).\\
\textbf{\textit{Cách 2:}} Dùng compa.\\
\textbf{\textit{Nhận xét:}} Trên tia $Ox$ bao giờ cũng vẽ được một và chỉ một điểm $M$ sao cho $OM=a$ (đơn vị dài).
\subsubsection{Dấu hiệu nhận biết một điểm nằm giữa hai điểm khác}

\subsection{Các dạng toán và phương pháp giải}

\begin{dang}{Xác định điểm nằm giữa hai điểm khác}

\end{dang}

		$C$ nằm giữa $O$ và $B$ nên tia $CO$ và $CB$ đối nhau;\\
		Do đó tia $CB$ và $CA$ đối nhau nên $C$ nằm giữa hai điểm $A$ và $B$.
	\end{enumEX}	
	}
\end{vd}

\section{TRUNG ĐIỂM CỦA ĐOẠN THẲNG}
\subsection{Tóm tắt lý thuyết}
\begin{enumEX}{1}
	\item 
		Trung điểm $M$ của đoạn thẳng $AB$ là điểm nằm giữa $A,B$ và cách đều $A,B$ ($MA=MB$).
		
	\begin{center}
			\begin{tikzpicture}
		\tkzDefPoints{0/0/A,2/0/M,4/0/B}	
		\tkzDrawPoints[fill=black](M,A,B)
		\tkzDrawSegments(A,B)
		\tkzLabelPoints[below](A,B,M)
		\end{tikzpicture}
	\end{center}
	\item Nếu $M$ là trung điểm của đoạn thẳng $AB$ thì $MA=MB=\dfrac{AB}{2}$.
\end{enumEX}
\subsection{Các dạng toán và phương pháp giải}
\begin{dang}{Nhận biết một điểm là trung điểm của đoạn thẳng}
\textbf{\textit{Phương pháp giải}}\\
Nếu $M$ nằm giữa $A$ và $B$ đồng thời $MA=MB$ thì $M$ là trung điểm của $AB$.
\end{dang}
\begin{vd}%[Hưng Nguyễn, dự án EX-C2-Lớp 6]%[6H1B10]
	Trên tia $Ox$ lấy hai điểm $A$ và $B$ sao cho $OA=3$ cm; $OB=6$ cm.
	\begin{enumEX}{1}
		\item Điểm $A$ có nằm giữa hai điểm $O$ và $B$ không?
		\item So sánh $OA$ và $AB$.
		\item Điểm $A$ có là trung điểm của đoạn thẳng $OB$ không? Vì sao?
	\end{enumEX}
	\loigiai{
	\begin{center}
		\begin{tikzpicture}
		\tkzInit[ymin=-1,ymax=0.1,xmin=-1,xmax=8]
		\tkzClip
		\tkzDefPoints{0/0/O,3/0/A,6/0/B,7/0/x}
		
		\tkzDrawPoints[fill=black](O,A,B)
		\tkzDrawSegments(O,x)
		\tkzLabelPoints[below](O,A,B)
		\tkzLabelPoints[below left](x)
		\end{tikzpicture}
	\end{center}
	\begin{enumEX}{1}
		\item Vì $A,B$ thuộc tia $Ox$ mà $OA=3$ cm; $OB=6$ cm nên $OA<OB$ suy ra điểm $A$ nằm giữa hai điểm $O$ và $B$.
		\item Điểm $A$ nằm giữa $O$ và $B$ nên \begin{eqnarray*}
		OA+AB & = & OB\\
		3+AB & = & 6\\
		AB & = & 3\text{ cm}.
		\end{eqnarray*}
		Suy ra $AB=OA\;(=3\text{ cm})$.
		\item $A$ nằm giữa $O$ và $B$ đồng thời $OA=AB$ nên $A$ là trung điểm của $OB$.
	\end{enumEX}	
	}
\end{vd}

\begin{vd}%[Hưng Nguyễn, dự án EX-C2-Lớp 6]%[6H1B10]

		\item Trên tia đối của $Cx$ lấy điểm $E$ sao cho $CE=2$ cm. Hỏi điểm $C$ có là trung điểm của $AE$ không? Vì sao?
	\end{enumEX}

		\item $A$ nằm giữa $C$ và $D$ nên

		CA+AD & = & CD\\
		2+AD & = & 5\text{ do đó }AD=3\text{ cm}.
		\end{eqnarray*}

	\end{enumEX}
	}
\end{vd}
\begin{dang}{Tính độ dài đoạn thẳng}
	\textbf{\textit{Phương pháp giải}}\\
	Dựa vào nhận xét: Khoảng cách từ trung điểm của đoạn thẳng đến mỗi đầu đoạn thẳng bằng nửa độ dài của đoạn thẳng ấy.
\end{dang}
\begin{vd}%[Hưng Nguyễn, dự án EX-C2-Lớp 6]%[6H1K10]
	Cho đoạn thẳng $AB=8$ cm. Gọi $M$ là trung điểm của $AB$. Trên đoạn $AB$ lấy hai điểm $C,D$ sao cho $AC=BD=3$ cm.
	\begin{enumEX}{1}
		\item Tính độ dài đoạn $CD$.
		\item Điểm $M$ có phải là trung điểm của đoạn thẳng $CD$ không? Vì sao?
	\end{enumEX}
	\loigiai{
	\begin{center}
		\begin{tikzpicture}
		\tkzInit[ymin=-1,ymax=0.1,xmin=-1,xmax=9]
		\tkzClip
		\tkzDefPoints{0/0/A,3/0/C,5/0/D,8/0/B,4/0/M}
		
		\tkzDrawPoints[fill=black](D,M,A,B,C)
		\tkzDrawSegments(A,B)
		\tkzLabelPoints[below](D,M,A,B,C)
		\end{tikzpicture}
	\end{center}
	\begin{enumEX}{1}
		\item Điểm $D$ nằm giữa $A,B$ nên 
		\begin{eqnarray*}
		AD+BD & = & AB\\
		AD+3 & = & 8\text{ nên }AD=5\text{ cm}.
		\end{eqnarray*}
		$C$ và $D$ cùng thuộc tia $AB$ mà $AC=3$ cm, $AD=5$ cm nên $AC<AD$ hay $C$ nằm giữa $A$ và $D$ ta có 
		\begin{eqnarray*}
		AC+CD & = & AD\\
		3+CD & = & 5\text{ nên } CD=2\text{ cm}.
		\end{eqnarray*}
		\item $M$ là trung điểm của $AB$ nên $AM=\dfrac{AB}{2}=\dfrac{8}{2}=4$ (cm).\\
		Trên tia $AB$ có $AC<AM$ (vì $3<4$) nên $C$ nằm giữa $A$ và $M$ nên 
		\begin{eqnarray*}
		AC+CM & = & AM\\
		3+CM & = & 4\text{ hay } CM=1\text{ cm}.
		\end{eqnarray*}
		Trên tia $AB$ có $AM<AD$ (vì $4<5$) nên $M$ nằm giữa $A$ và $D$ nên 
		\begin{eqnarray*}
		AM+MD & = & AD\\
		4+MD & = & 5\text{ hay } MD=1\text{ cm}.
		\end{eqnarray*}
		Ta có $M$ nằm giữa $C$ và $D$ vì $MC+MD=CD$ ($1+1=2$) đồng thời $CM=MD$ nên $M$ là trung điểm của đoạn $CD$.
	\end{enumEX}	
	}
\end{vd}

\begin{vd}%[Hưng Nguyễn, dự án EX-C2-Lớp 6]%[6H1K10]
	Trên tia $Ox$ lấy hai điểm $M,N$ sao cho $OM=6$ cm; $ON=10$ cm. Gọi $I,K$ là trung điểm của $ON$ và $MN$. Tính độ dài $IK$.
	\loigiai{
	\begin{center}
		\begin{tikzpicture}
		\tkzInit[ymin=-1,ymax=0.1,xmin=-1,xmax=12]
		\tkzClip
		\tkzDefPoints{0/0/O,5/0/I,6/0/M,8/0/K,10/0/N,11/0/x}
		
		\tkzDrawPoints[fill=black](O,I,M,K,N)
		\tkzDrawSegments(O,x)
		\tkzLabelPoints[below](O,I,M,K,N)
		\tkzLabelPoints[below left](x)
		\end{tikzpicture}
	\end{center}
	$I$ là trung điểm của $ON$ nên $IN=\dfrac{1}{2}ON=\dfrac{10}{2}=5$ (cm).\\
	$OM<ON$ nên $M$ nằm giữa $O$ và $N$ suy ra $MN=ON-OM=10-6=4$ (cm).\\
	$K$ là trung điểm của $MN$ thì $KN=\dfrac{1}{2}MN=\dfrac{4}{2}=2$ cm.\\
	Vì $K$ nằm giữa $N,I$ nên $IK=IN-KN=5-2=3$ cm.	
	}
\end{vd}

\begin{vd}%[Hưng Nguyễn, dự án EX-C2-Lớp 6]%[6H1K10]
	Cho đoạn thẳng $AB=8$ cm. $C$ là điểm nằm giữa $A$ và $B$. Gọi $M,N$ lần lượt là trung điểm của các đoạn thẳng $AC,CB$. Tính độ dài $MN$.
	\loigiai{
	\begin{center}
		\begin{tikzpicture}
		\tkzInit[ymin=-1,ymax=0.1,xmin=-1,xmax=9]
		\tkzClip
		\tkzDefPoints{0/0/A,8/0/B,3/0/C}
		\tkzDefMidPoint(A,C)\tkzGetPoint{M}
		\tkzDefMidPoint(C,B)\tkzGetPoint{N}
		
		\tkzDrawPoints[fill=black](M,N,A,B,C)
		\tkzDrawSegments(A,B)
		\tkzLabelPoints[below](M,N,A,B,C)
		\end{tikzpicture}
	\end{center}
	$M$ là trung điểm của đoạn thẳng $AC$ nên $MC=\dfrac{AC}{2}$.\\
	$N$ là trung điểm của đoạn thẳng $CB$ nên $CN=\dfrac{CB}{2}$.\\
	Mà $C$ nằm giữa $M$ và $N$ nên $MN=MC+CN=\dfrac{AC}{2}+\dfrac{CB}{2}$\\
	$\Rightarrow MN=\dfrac{AC+CB}{2}=\dfrac{AB}{2}=\dfrac{8}{2}=4$ (cm).\\
	\textbf{\textit{Nhận xét:}} Bài toán không thể và không cần thiết tính cụ thể số đo đoạn thẳng $AC,CB$ mà có thể chỉ cần biểu thị được $MC=\dfrac{AC}{2}$ và $CN=\dfrac{CB}{2}$. Sau đó tính $MN$.\\
	Dựa vào lời giải trên, ta có mối liên hệ sau: $MN=\dfrac{AB}{2}$ với mọi điểm $C$ thuộc đoạn $AB$. Như vậy nếu cho độ dài $AB$ thì tính được $MN$ và ngược lại.	
	}
\end{vd}

	Vì $I$ là trung điểm của $CD$ nên $CI=\dfrac{CD}{2}$, mà $CI=4$ cm nên $CD=8$ cm.
	}

\begin{bt}%[Hưng Nguyễn, dự án EX-C2-Lớp 6]%[6H1G10]
	Cho đoạn thẳng $AB=6$ cm. Lấy điểm $C$ thuộc tia $AB$ sao cho $AC=2$ cm.
	\begin{enumEX}{1}
		\item Tính $BC$.
		\item Vẽ điểm $D$ thuộc tia đối của tia $BC$ sao cho $BD=3$ cm. Tính $CD$.
		\item Gọi $I,K$ lần lượt là trung điểm của $BC,BD$. Tính $IK$.
	\end{enumEX}
	\loigiai{
	\begin{center}
		\begin{tikzpicture}
		\tkzInit[ymin=-1,ymax=0.1,xmin=-1,xmax=10]
		\tkzClip
		\tkzDefPoints{0/0/A,6/0/B,2/0/C,9/0/D}
		\tkzDefMidPoint(C,B)\tkzGetPoint{I}
		\tkzDefMidPoint(D,B)\tkzGetPoint{K}
		
		\tkzDrawPoints[fill=black](A,B,C,D,I,K)
		\tkzDrawSegments(A,D)
		\tkzLabelPoints[below](A,B,C,D,I,K)
		\end{tikzpicture}
	\end{center}
	\begin{enumEX}{1}
		\item Trên tia $AB$ có hai điểm $B,C$ mà $AC<AB$ ($2$ cm $<6$ cm) nên $C$ nằm giữa $A$ và $B$. Do đó
		\begin{eqnarray*}
		AC+BC & = & AB\\ 2+BC & = & 6\Rightarrow BC=4\text{ cm}.
		\end{eqnarray*}
		\item Vì $BC$ và $BD$ là hai tia đối nhau nên $B$ nằm giữa $C$ và $D$. Do đó
		\begin{eqnarray*}
			CD & = & BC+BD\\ CD & = & 4+3\Rightarrow CD=7\text{ cm}.
		\end{eqnarray*}
		\item Vì $I$ là trung điểm của $BC$ nên $IB=\dfrac{BC}{2}=\dfrac{4}{2}=2$ (cm).\\
		Vì $K$ là trung điểm của $BD$ nên $BK=\dfrac{BD}{2}=\dfrac{3}{2}=1{,}5$ (cm).\\
		$I$ nằm trên tia $BC$, $K$ nằm trên tia $BD$ mà $BC$ và $BD$ là hai tia đối nhau nên $B$ nằm giữa $I$ và $K$. Do đó
		\begin{eqnarray*}
			IK & = & IB+BK\\ IK & = & 2+1{,}5\Rightarrow IK=3{,}5\text{ cm}.
		\end{eqnarray*}
	\end{enumEX}	
	}
\end{bt}
\begin{bt}%[Hưng Nguyễn, dự án EX-C2-Lớp 6]%[6H1G10]
	Cho đoạn thẳng $AB=8$ cm. Lấy điểm $C$ nằm giữa $A,B$ sao cho $AC=3BC$.
	\begin{enumEX}{1}
		\item Tính độ dài $AC$ và $BC$;
		\item Trên tia $BA$ lấy điểm $D$ sao cho $BD=5$ cm. Chứng tỏ $D$ là trung điểm của đoạn thẳng $AC$;
		\item Vẽ điểm $E$ sao cho $D$ là trung điểm của đoạn thẳng $BE$. So sánh độ dài $AB$ và $CE$.
	\end{enumEX}
	\loigiai{
	\begin{center}
		\begin{tikzpicture}
		\tkzInit[ymin=-1,ymax=0.1,xmin=-3,xmax=10]
		\tkzClip
		\tkzDefPoints{0/0/A,8/0/B,6/0/C,3/0/D,-2/0/E}
		
		\tkzDrawPoints[fill=black](A,B,C,D,E)
		\tkzDrawSegments(E,B)
		\tkzLabelPoints[below](A,B,C,D,E)
		\end{tikzpicture}
	\end{center}
	\begin{enumEX}{1}
		\item Vì $C$ nằm giữa $A$ và $B$ nên ta có
		\begin{eqnarray*}
			AB & = & AC+BC\\ AB & = & 3BC+BC=4BC\Rightarrow BC=\dfrac{AB}{4}=\dfrac{8}{4}=2\text{ (cm)}.
		\end{eqnarray*}
		Vậy $AC=AB-BC=8-2=6$ (cm).
		\item Trên tia $BA$ có hai điểm $C,D$ mà $BC<BD$ ($2$ cm $<5$ cm) nên $C$ nằm giữa $B$ và $D$. Do đó
		\begin{eqnarray*}
			BC+CD & = & BD\\ 2+CD & = & 5\Rightarrow CD=3\text{ cm}.
		\end{eqnarray*}
		Trên tia $BA$ có hai điểm $A,D$ mà $BD<BA$ ($5$ cm $<8$ cm) nên $D$ nằm giữa $B$ và $A$. Do đó
		\begin{eqnarray*}
			BD+DA & = & BA\\ 5+DA & = & 8\Rightarrow DA=3\text{ cm}.
		\end{eqnarray*}
		Ta có $AD+DC=AC$ ($3+3=6$) nên $D$ nằm giữa $A$ và $C$, đồng thời $AD=DC=3$ cm nên $D$ là trung điểm của đoạn thẳng $AC$.
		\item Vì $D$ là trung điểm của $BE$ nên $BD=\dfrac{BE}{2}\Rightarrow BE=2BD=10$ (cm).\\
		Trên tia $BD$ có hai điểm $C,E$ mà $BC<BE$ ($2$ cm $<10$ cm) nên $C$ nằm giữa $B$ và $E$. Do đó
		\begin{eqnarray*}
			BC+CE & = & BE\\ 2+CE & = & 10\Rightarrow CE=8\text{ cm}.
		\end{eqnarray*}
		Vậy $CE=AB$ ($=8$ cm).
	\end{enumEX}
	}
\end{bt}
\begin{bt}%[Hưng Nguyễn, dự án EX-C2-Lớp 6]%[6H1K10]
	Cho đoạn thẳng $CD=9$ cm. Lấy $M$ thuộc đoạn thẳng $CD$ sao cho $CM=5$ cm.
	\begin{enumEX}{1}
		\item Tính $MD$.
		\item Gọi $A$ là trung điểm của $MD$. Tính $MA$.
		\item Lấy điểm $B$ thuộc tia đối của tia $DC$ sao cho $DB=2DA$. Hỏi $D$ có là trung điểm của đoạn thẳng $BM$ không? Vì sao?
	\end{enumEX}
	\loigiai{
	\begin{center}
		\begin{tikzpicture}
		\tkzInit[ymin=-1,ymax=0.1,xmin=-3,xmax=14]
		\tkzClip
		\tkzDefPoints{0/0/C,9/0/D,5/0/M,13/0/B}
		\tkzDefMidPoint(M,D)\tkzGetPoint{A}
		
		\tkzDrawPoints[fill=black](M,C,D,A,B)
		\tkzDrawSegments(C,B)
		\tkzLabelPoints[below](M,C,D,A,B)
		\end{tikzpicture}
	\end{center}
	\begin{enumEX}{1}
		\item Vì $M$ thuộc đoạn thẳng $CD$ nên $M$ nằm giữa $C$ và $D$. Do đó
		\begin{eqnarray*}
			CM+MD & = & CD\\ 5+MD & = & 9\Rightarrow MD=4\text{ cm}.
		\end{eqnarray*}
		\item Vì $A$ là trung điểm $MD$ nên $MA=AD=\dfrac{MD}{2}=\dfrac{4}{2}=2$ (cm).
		\item Ta có $DB=2DA=4$ (cm).\\
		Lại có $DM$ và $DC$ là hai tia trùng nhau và $B$ thuộc tia đối của tia $DC$ nên $DM$ và $DB$ là hai tia đối nhau hay $D$ nằm giữa $M$ và $B$.\\
		Mà $DM=DB$ ($=4$ cm) nên $D$ là trung điểm của đoạn thẳng $BM$.
	\end{enumEX}	
	}
\end{bt}
