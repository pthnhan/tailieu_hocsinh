	Qua phép quay tâm $O$ góc quay $-90^\circ$ đường thẳng $\Delta\colon 3x+4y-12=0$ biến thành đường thẳng?
	\choice
	{$\Delta\text{'}\colon 3x+4y+12=0$}
	{$\Delta\text{'}\colon 3x+4y-12=0$}
	{\True{$\Delta\text{'}\colon 4x+3y-12=0$}}
	{$\Delta\text{'}\colon 4x+3y+12=0$}
	\loigiai{
		$\mathrm{Q}_{(0,-90^\circ)}$ đường thẳng $(\Delta)$ biến thành $(\Delta\text{'}).$\\
		$(\Delta)$ có véc-tơ pháp tuyến $\overrightarrow{n}_{\Delta}=(3;-4)$ $\Rightarrow$ $\overrightarrow{n}_{\Delta\text{'}}=(4;3).$\\
		Nên $(\Delta\text{'}): 4x+3y+m=0.$
		Lấy $M(-4;0)\in (\Delta) \Rightarrow M\text{'}(0;4)\in (\Delta\text{'}).$\\
		Suy ra $4\cdot0+3\cdot4+m=0 \Leftrightarrow m=-12$.
		Do đó: $(\Delta\text{'})\colon 4x+3y-12=0.$
	}
\end{ex}

\begin{ex}%[HK1, THPT Xuân Hòa, Vình Phúc 2017-2018]%[Nguyễn Ngọc Hóa, Dự án 11HK1-2018]%[1D1K3-5]
	Một nghiệm của phương trình lượng giác: $\sin^{2}x+\sin^{2}2x+\sin^{2}3x=2$ là:
	\choice
	{$\dfrac{\pi}{12}$}
	{$\dfrac{\pi}{8}$}
	{\True {$\dfrac{\pi}{6}$}}
	{$\dfrac{\pi}{3}$}
	\loigiai{
		Tập xác định: $\mathscr{D}=\mathbb{R}.$
		\begin{eqnarray*}
		& &\sin^{2}x+\sin^{2}2x+\sin^{2}3x=2\\
		&\Leftrightarrow  &1-\cos 2x + 1-\cos4x + 1-\cos6x=4\\
		&\Leftrightarrow &\cos 2x + \cos4x + \cos6x=-1\\
		&\Leftrightarrow &2\cos4x\cos2x + \cos4x=-1\\
		&\Leftrightarrow &\cos4x\left(2\cos2x +1\right)=-1\\
		&\Leftrightarrow &\left(2\cos^2 2x - 1\right) \left(2\cos2x +1\right)=-1\\
		&\Leftrightarrow &4\cos^3 2x+2\cos^2 2x - 2\cos2x=0\\
		&\Leftrightarrow &2\cos2x \left(2\cos^2 2x +\cos2x-1\right)=0\\
		&\Leftrightarrow &\hoac{\cos2x&=0\\2\cos^2 2x+\cos 2x-1 &=0}
		\Leftrightarrow \hoac{\cos2x&=0\\\cos2x&=-1\\\cos2x&=\dfrac{1}{2}}
		\end{eqnarray*}
	\begin{itemize}
	\item $\cos 2x = 0 \Leftrightarrow 2x=\dfrac{\pi}{2}+k\pi \Leftrightarrow x=\dfrac{\pi}{4}+k\dfrac{\pi}{2}(k\in \mathbb{Z})$.
	\item $\cos 2x = -1 \Leftrightarrow 2x=\pi+k2\pi \Leftrightarrow x=\dfrac{\pi}{2}+k\pi(k\in \mathbb{Z})$.
	\item $\cos 2x = \dfrac{1}{2} \Leftrightarrow 2x=\pm \dfrac{\pi}{3}+k2\pi \Leftrightarrow x=\pm \dfrac{\pi}{6}+k\pi (k\in \mathbb{Z})$.
	\end{itemize}
}
\end{ex}

\begin{ex}%[HK1, THPT Xuân Hòa, Vình Phúc 2017-2018]%[Nguyễn Ngọc Hóa, Dự án 11HK1-2018]%[1D1B2-1]
	Giải phương trình sau: $2\sin^2 x-3\sin x+1=0$.
	\loigiai{
			\begin{itemize}
			\item Đặt $t=\sin x$, $-1\leq t \leq 1$.
			\item Phương trình trở thành
			$$2t^2-3t+1=0 \Leftrightarrow \hoac{&t=1 &(\text{nhận})\\ &t=\dfrac{1}{2} &(\text{nhận})}$$
			\item Với $t=1$, suy ra $\sin x=1\Leftrightarrow x=\dfrac{\pi}{2}+k2\pi,k\in \mathbb{Z}$.
			\item Với $t=\dfrac{1}{2}$, suy ra $\sin x=\dfrac{1}{2}\Leftrightarrow \sin x=\sin \dfrac{\pi}{6}\Leftrightarrow \hoac{&x=\dfrac{\pi}{6}+k2\pi\\
			&x=\dfrac{5\pi}{6}+k2\pi}$, $k\in \mathbb{Z}$.
			\end{itemize}
		Vậy phương trình có 3 họ nghiệm $\hoac{&x=\dfrac{\pi}{2}+k2\pi\\ &x=\dfrac{\pi}{6}+k2\pi\\ &x=\dfrac{5\pi}{6}+k2\pi}$, $k\in \mathbb{Z}$.
	}
\end{ex}

\begin{ex}%[HK1, THPT Xuân Hòa, Vình Phúc 2017-2018]%[Nguyễn Ngọc Hóa, Dự án 11HK1-2018]%[1D1B3-3]
	Giải phương trình sau: $\sqrt{3}\cdot\sin x+\cos x=1$.
	\loigiai{
			\begin{eqnarray}
			\sqrt{3}\cdot\sin x+\cos x=1 &\Leftrightarrow &\dfrac{\sqrt{3}}{2}\cdot \sin x+\dfrac{1}{2}\cdot \cos x=\dfrac{1}{2} \notag\\
			&\Leftrightarrow &\cos\dfrac{\pi}{6}\cdot \sin x+\sin\dfrac{\pi}{6}\cdot \cos x=\sin\dfrac{\pi}{6} \notag\\
			&\Leftrightarrow &\sin\left( x+\dfrac{\pi}{6}\right)=\sin\dfrac{\pi}{6} \notag\\
			&\Leftrightarrow &\hoac{&x+\dfrac{\pi}{6}=\dfrac{\pi}{6}+k2\pi\\ &x+\dfrac{\pi}{6}=\dfrac{5\pi}{6}+k2\pi}, k\in \mathbb{Z} \notag\\
			&\Leftrightarrow &\hoac{&x=k2\pi\\ &x=\dfrac{2\pi}{3}+k2\pi}, k\in \mathbb{Z}. \notag
			\end{eqnarray}
			Vậy phương trình có 2 họ nghiệm $\hoac{&x=k2\pi\\ &x=\dfrac{2\pi}{3}+k2\pi}$, $k\in \mathbb{Z}$.
	}
\end{ex}
