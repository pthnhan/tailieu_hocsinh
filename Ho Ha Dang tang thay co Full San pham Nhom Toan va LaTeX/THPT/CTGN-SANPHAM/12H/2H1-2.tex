\begin{dang}{Khối tứ diện có đỉnh là đỉnh của hình hộp}
Cho khối hộp $ ABCD.A'B'C'D' $
\begin{itemize}
	\item \textbf{Loại 1. }Khối tứ diện tạo bởi hai đường chéo hai mặt đối diện có thể tích là $ V_{td1} = \dfrac{V}{3}. $\\
	(ví dụ các khối tứ diện: $ ACB'D' $, $ AB'CD' $, $AD'CB'  $, \ldots)
	\item \textbf{Loại 2. }Khối tứ diện tạo bởi $ 4 $ đỉnh bất kỳ của khối hộp (không gồm hai đường chéo hai mặt đối diện) có thể tích là $ V_{td2}=\dfrac{V}{6}. $\\
	(ví dụ các khối tứ diện: $ B'ABC $, $ BA'C'D' $, $ ACC'D' $, \ldots)
\end{itemize}
\end{dang}

\begin{vd}
	Cho hình lăng trụ tam giác $ABC.A'B'C'$ có thể tích $V$ cạnh bên bằng $2a$. Gọi $M,N,P$ lần lượt là các điểm trên cạnh $AA',BB',CC'$ thỏa mãn $MA=\dfrac{1}{2}AA', NB=\dfrac{1}{3}BB', PC=\dfrac{1}{3}CC'$.$V_1$ là thể tích khối đa diện $ABC.MNP$. Tính tỉ số $\dfrac{V_1}{V}$
	\choice
	{$\dfrac{7}{6}$}
	{\True $\dfrac{7}{18}$}
	{$\dfrac{1}{18}$}
	{$\dfrac{1}{54}$}
	\loigiai{
		Áp dụng công thức 
		$\dfrac{V_{ABC.MNP}}{V_{ABC.A'B'C'}}=\dfrac{1}{3} \left(\dfrac{AM}{AA'}+\dfrac{BN}{BB'}+\dfrac{CP}{CC'}\right)
		\Rightarrow \dfrac{V_1}{V}=\dfrac{1}{3}\left(\dfrac{1}{2}+\dfrac{1}{3}+\dfrac{1}{3}\right)=\dfrac{7}{18}$.
	}

	\choice
	{\True$\dfrac{1}{2} $}
	{$2$}
	{$\dfrac{1}{18}$}
	{$\dfrac{1}{72}$}
	\loigiai{
		Áp dụng công thức $\dfrac{V_{ABCD.MNPQ}}{V_{ABCD.A'B'C'D'}}=\dfrac{1}{4} \left(\dfrac{AM}{AA'}+\dfrac{BN}{BB'}+\dfrac{CP}{CC"}+\dfrac{DQ}{DD'} \right)$.
