\begin{bt}%[HK1,THSP HCM,2018]%[1D1B2]%[1D1K3]
	Giải phương trình sau:
	\begin{enumerate}
	\item $2 \sin2x -1 =0$.
	\item $\sqrt{3}\cot (2x- 10^ \circ ) = 1$.
	\item $\sqrt{3} \cos2x + 3 \sin2x = -1$.
	\end{enumerate}
	\loigiai{		
	\begin{enumerate}
	\item $2 \sin2x -1 =0$\\
	$\Leftrightarrow \sin 2x = \dfrac{1}{2}$\\
	$\Leftrightarrow \hoac {2x &= \dfrac{\pi}{6}+ k2 \pi\\ 2x &= \dfrac{5 \pi}{6} +k2 \pi }$  $  (k \in \mathbb{Z})$\\
	$\Leftrightarrow \hoac {x &= \dfrac{\pi}{12}+ k \pi\\ x &= \dfrac{5 \pi}{12} +k \pi }$ $  (k \in \mathbb{Z})$.
	\item$\sqrt{3}\cot (2x- 10^ \circ ) = 1$ \\
		$\Leftrightarrow \cot (2x - 10^\circ) = \dfrac{1}{\sqrt{3}}$\\
	$\Leftrightarrow 2x = 70^ \circ + k180^ \circ$ $  (k \in \mathbb{Z})$\\
	$\Leftrightarrow x = 35^ \circ + k90^ \circ$ $  (k \in \mathbb{Z})$.

	\item $\sqrt{3} \cos2x + 3 \sin2x = -1$\\
	Chia hai vế phương trình cho $\sqrt{(\sqrt{3})^2+3^2}=2\sqrt{3}$, ta có:
	\begin{align*}
	pt &\Leftrightarrow \sin \dfrac{\pi}{6} \cos 2x + \sin 2x \cos \dfrac{\pi}{6}=\dfrac{-1}{2\sqrt{3}}\\
	&\Leftrightarrow \sin \left(2x + \dfrac{\pi}{6} \right) = \dfrac{-1}{2\sqrt{3}}\\
	&\Leftrightarrow \hoac {2x + \dfrac{\pi}{6} &= \arcsin \left(\dfrac{-1}{2\sqrt{3}} \right) + k2 \pi \\ 2x + \dfrac{\pi}{6} &= \pi -\arcsin  \left(\dfrac{-1}{2\sqrt{3}} \right) + k2 \pi } (k \in \mathbb{Z})\\
	&\Leftrightarrow \hoac {x &= - \dfrac{\pi}{12} + \dfrac{1}{2} \arcsin \left(\dfrac{-1}{2\sqrt{3}}\right) + k \pi \\ x  &= \dfrac{5\pi}{12} - \dfrac{1}{2}\arcsin \left( \dfrac{-1}{2\sqrt{3}} \right) + k \pi}   (k \in \mathbb{Z})  .
	\end{align*}
	\end{enumerate}
		}
\end{bt}

	\begin{enumerate}
	\item Viết $4$ số hạng đầu của dãy số $(u_n)$.
	\item Xét tính tăng, giảm của dãy số $(u_n)$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item 
	$$u_1 = \dfrac{3}{2};u_2 = \dfrac{5}{3};u_3 = \dfrac{7}{4};u_4=\dfrac{9}{5}.$$
	\item 
	\begin{align*}
	u_{n+1} - u_n &= \dfrac{2(n+1)+1}{(n+1)+1}- \dfrac{2n+1}{n+1}\\
	&= \dfrac{2n^2+5n+3-2n^2-5n-2}{(n+2)(n+1)}\\
	&= \dfrac{1}{(n+2)(n+1)} > 0.
	\end{align*}

	\end{enumerate}
	}

\begin{bt}%[HK1,THSP HCM,2018]%[1D2B5]
	Một tổ có $9$ học sinh nam và $7$ học sinh nữ. Chọn ngẫu nhiên $3$ học sinh vào đội văn nghệ. Tính xác suất sao cho trong $3$ học sinh đó:
	\begin{enumerate}
	\item Có đúng $1$ nữ.
	\item Có ít nhất $1$ nam.
	\end{enumerate}
	\loigiai{
		\begin{enumerate} 
			\item Có đúng $1$ nữ.\\
			$n(\Omega)= \mathrm{C}_{16}^3=560$.\\
			Gọi $A$ là biến cố  '' Có đúng $1$ nữ ''.\\
			$n(A) = \mathrm{C}_7^1 \cdot \mathrm{C}_9^2 = 252$.\\
			$P(A)= \dfrac{252}{560} = \dfrac{9}{20}$.
			\item Có ít nhất $1$ nam.\\
			Gọi $B$ là biến cố '' Có ít nhất $1$ nam ''.\\
			$n(B)= \mathrm{C}_9^1 \cdot \mathrm{C}_7^2 + \mathrm{C}_9^2 \cdot \mathrm{C}_7^1 +\mathrm{C}_9^3 \cdot \mathrm{C}_7^0 = 525 $.\\
			$P(B) = \dfrac{525}{560} = \dfrac{15}{16}$.
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[HK1,THSP HCM,2018]%[1D2B3]
	Tìm hệ số của số hạng chứa $x^3$ trong khai triển $\displaystyle \left(2x - \frac{1}{x^3} \right)^{11}$. 
	\loigiai{
	Số hạng tổng quát trong khai triển:
	$$T_{k+1}= \mathrm{C}_{11}^{k}(2x)^{11-k}(-\dfrac{1}{x^3})^k = \mathrm{C}_{11}^k 2^{11-k} (-1)^k x^{11-4k}$$
	Số hạng chứa $x^3$ tương ứng
	$$11-4k=3 \Leftrightarrow k=2$$
	Hệ số của số hạng chứa $x^3$ là: $$T_3=28160 .$$	
	}
\end{bt}

\begin{bt}%[HK1,THSP HCM,2018]%[1D3B3]
Tìm số hạng đầu $u_1$ và công sai $d$ của cấp số cộng $(u_n)$ biết $\heva {&5u_1+10u_5=0\\  &S_4=14}$.
\loigiai{
Ta có:
\begin{eqnarray*}
&& \heva{&5u_1+10u_5 =0\\& S_4=14}\\
& \Leftrightarrow &\heva{&5u_1+10u_1+40d=0\\& 4u_1+6d =14}\\
& \Leftrightarrow &\heva{&15u_1+40d =0\\  &4u_1+6d=14}\\
& \Leftrightarrow &\heva{&u_1=8\\  &d=-3}
\end{eqnarray*}
Vậy $u_1=8;d=-3$.
}
\end{bt}

\begin{bt}%[HK1,THSP HCM,2018]%[1D3K4]
Tế bào E.Coli trong điều kiện thích hợp cứ $20$ phút lại nhân đôi một lần. Hỏi nếu có $10^6$ tế bào thì sau $2$ giờ sẽ phân chia thành bao nhiêu tế bào?
\loigiai{
$$2 \text{ giờ} = 120 \text{ phút}$$
Số chu kỳ nhân đôi $120 : 20 = 6$\\
Số tế bào E.Coli có được sau $2$ giờ là:
$$10^6 \cdot 2^6= 64 \cdot 10^6 \text{    (tế bào)}.$$
}
\end{bt}
\begin{bt}%[HK1,THSP HCM,2018]%[1H1B1]%[1H1K2]%[1H1K3]%[1H1G2]
Cho hình chóp $S.ABCD$ có đáy là hình bình hành tâm $O$. Gọi $I$, $K$, $M$ lần lượt là trung điểm của các đoạn thẳng $SA$, $SC$, $OD$.
\begin{enumerate}
\item Xác định giao tuyến của các cặp mặt phẳng: $(SAC)$ và $(SBD)$; $(SAD)$ và $(SBC)$.
\item Xác định giao điểm $E$ của $SD$ với mặt phẳng $(BIK)$.
\item Chứng minh rằng: $SD \parallel (IKM)$.
\item Mặt phẳng $(IKM)$ cắt các cạnh $AD$, $DC$, $SB$ lần lượt tại $Q$, $P$, $L$. Xác định các điểm $Q$, $P$, $L$ và thiết diện của mặt phẳng $(IKM)$ và hình chóp $S.ABCD$.
\end{enumerate}

\begin{enumerate}
\item Xác định giao tuyến của các cặp mặt phẳng: $(SAC)$ và $(SBD)$; $(SAD)$ và $(SBC)$.\\
Ta có:\\ $\heva{S \in (SAC)\\ S \in (SBD)}$$
$$\Rightarrow S \text{ là điểm chung}$.\\
$\heva{O \in AC \subset (SAC)\\ O \in BD \subset (SBD)}$
$\Rightarrow O \text{ là điểm chung}$.\\
Vậy giao tuyến của mặt phẳng:$(SAC)$ và $(SBD)$ là $SO$.\\
Ta có:\\ $\heva{S \in (SAD)\\ S \in (SBC)}$
$\Rightarrow S \text{ là điểm chung}$.\\
$\heva {&AD \subset (SAD)\\ &BC \subset (SBC)\\ &AD \parallel BC}$.\\
Vậy giao tuyến của $(SAD)$ và $(SBC)$ là đường thẩng $d$ qua $S$ và song song với $AD$.

\item Chứng minh rằng: $SD \parallel (IKM)$.\\

$\heva {&M \in (MIK) \\ & M \in (ABCD)} \Rightarrow M$ là điển chung của $(IMK)$ và $(ABCD)$.\\
$\heva{&IK \subset (IMK)\\ &AC \subset (ABCD)\\ &IK \parallel AC}$\\

$\Rightarrow MQ \parallel AO$\\
$\Rightarrow Q$ là trung điểm $AD$.\\
$\Rightarrow QI \parallel SD$\\
$\Rightarrow SD \parallel (IMK)$.

$$\heva {&P \in QM \subset (IMK)\\ &P \in DC}$$
$\Rightarrow P$ là giao điểm của $DC$ và $(IMK)$.\\
Gọi $L = HM \cap SB$, ta có:
$$\heva {&L \in HM \subset (IMK)\\ &L \in SB}$$
$\Rightarrow L$ là giao điểm của $SB$ và $(IMK)$.\\
Vậy thiết diện của $(IMK)$ và hình chóp $S.ABCD$ là ngũ giác $QILKP$.
\end{enumerate}
