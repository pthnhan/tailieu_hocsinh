\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[Đề thi học kỳ 2, Trường THPT Hà Huy Tập, Hà Tĩnh, 2017]%[Vũ Văn Trường]%[1D4B1-1]%[1D4B2-3]
Tính các giới hạn sau:
\begin{enumerate}
  \item $\lim\dfrac{3n+1}{9n-2}$
  \item $\lim\limits_{x\to 1}\dfrac{x^2-5x+4}{x-1}$
\end{enumerate}
\loigiai{
\begin{enumerate}
  \item $\lim\dfrac{3n+1}{9n-2}=\lim\dfrac{3+\dfrac{1}{n}}{9-\dfrac{2}{n}}=\dfrac{1}{3}$.
  \item $\lim\limits_{x\to 1}\dfrac{x^2-5x+4}{x-1}=\lim\limits_{x\to 1}\dfrac{(x-1)(x-4)}{x-1}=\lim\limits_{x\to 1}(x-4)=-3$.
\end{enumerate}
