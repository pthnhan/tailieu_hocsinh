	Cho hàm số $f(x)$ có đạo hàm $f'(x)$ liên tục trên $\mathbb{R}$ và thỏa mãn $f'(x)\in[-1;1]$ với $\forall x\in (0;2).$ Biết $f(0)=f(2)=1.$ Đặt $I=\displaystyle\int\limits_0^2 f(x) \mathrm{\,d}x,$ phát biểu nào sau đây là đúng ?
	\choice
	{$I\in(-\infty;0]$}
	{$I\in(0;1]$}
	{\True $I\in [1;+\infty)$}
	{$I\in (0;1)$}
	\loigiai{
	\begin{enumerate}
		\item Cách 1: Chọn hàm\\
		Chọn $f(x)=1, \forall x $ ta được $I=\displaystyle\int\limits_0^2 f(x) \mathrm{\,d}x = 2$ nên $I\in [1;+\infty).$
		\item Cách 2: Tự luận\\
		$I=\displaystyle\int\limits_0^2 f(x) \mathrm{\,d}x =\displaystyle\int\limits_0^1 f(x) \mathrm{\,d}x +\displaystyle\int\limits_1^2 f(x) \mathrm{\,d}x=A+B$\\
		Đặt $\heva{&u=f(x)\\&\mathrm{\,d}v=\mathrm{\,d}x}\Rightarrow \heva{&\mathrm{\,d}u=f'(x)\mathrm{\,d}x\\&v=x-1.}$\\

		Vậy $I\geq1$.
	\end{enumerate}
		
	}
