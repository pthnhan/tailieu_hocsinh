\begin{ex}%[Đề thi vào 10, Sở Hà Nội, 2006]%[9D1B8]
Cho biểu thức $P=\left[\dfrac{a+3\sqrt{a}+2}{(\sqrt{a}+2)(\sqrt{a}-1)}-\dfrac{a+\sqrt{a}}{a-1}\right]:\left(\dfrac{1}{\sqrt{a}+1}+\dfrac{1}{\sqrt{a}-1}\right)$.
\begin{enumerate}
\item Rút gọn biểu thức $P$.
\item Tìm $a$ để $\dfrac{1}{P}-\dfrac{\sqrt{a}+1}{8}\ge 1$.
\end{enumerate}
\loigiai{
	\begin{enumerate}
	\item Điều kiện xác định: $a\geq 0$ và $a\neq 1$.\\
			Ta có $P=\left[\dfrac{a+3\sqrt{a}+2}{(\sqrt{a}+2)(\sqrt{a}-1)}-\dfrac{a+\sqrt{a}}{a-1}\right]:\left(\dfrac{1}{\sqrt{a}+1}+\dfrac{1}{\sqrt{a}-1}\right)$\\
			\phantom{Ta có $P$} $=\left[\dfrac{(\sqrt{a}+1)(\sqrt{a}+2)}{(\sqrt{a}+2)(\sqrt{a}-1)}-\dfrac{\sqrt{a}(\sqrt{a}+1)}{(\sqrt{a}-1)(\sqrt{a}+1)}\right]:\left(\dfrac{\sqrt{a}-1+\sqrt{a}+1}{(\sqrt{a}+1)(\sqrt{a}-1)}\right)$\\
			\phantom{Ta có $P$} $=\left(\dfrac{\sqrt{a}+1}{\sqrt{a}-1}-\dfrac{\sqrt{a}}{\sqrt{a}-1}\right)\cdot\dfrac{(\sqrt{a}+1)(\sqrt{a}-1)}{2\sqrt{a}}$\\
			\phantom{Ta có $P$} $=\dfrac{1}{\sqrt{a}-1}\cdot\dfrac{(\sqrt{a}-1)(\sqrt{a}+1)}{2\sqrt{a}}$\\
			\phantom{Ta có $P$} $=\dfrac{\sqrt{a}+1}{2\sqrt{a}}$.
	\item Ta có $\dfrac{1}{P}-\dfrac{\sqrt{a}+1}{8}\ge 1$ $\Leftrightarrow \dfrac{2\sqrt{a}}{\sqrt{a}+1}-\dfrac{\sqrt{a}+1}{8}\ge 1$\\
			\phantom{Ta có $\dfrac{1}{P}-\dfrac{\sqrt{a}+1}{8}\ge 1$} $\Leftrightarrow 16\sqrt{a}-(\sqrt{a}+1)^2 \geq 8(\sqrt{a}+1)$\\
			\phantom{Ta có $\dfrac{1}{P}-\dfrac{\sqrt{a}+1}{8}\ge 1$} $\Leftrightarrow a-6\sqrt{a}+9\leq 0$\\
			\phantom{Ta có $\dfrac{1}{P}-\dfrac{\sqrt{a}+1}{8}\ge 1$} $\Leftrightarrow (\sqrt{a}-3)^2\leq 0$\\
			\phantom{Ta có $\dfrac{1}{P}-\dfrac{\sqrt{a}+1}{8}\ge 1$} $\Leftrightarrow \sqrt{a}-3=0$\\
			\phantom{Ta có $\dfrac{1}{P}-\dfrac{\sqrt{a}+1}{8}\ge 1$} $\Leftrightarrow a=9$ (thỏa mãn điều kiện).\\
			Vậy $a=9$.
	\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở Hà Nội, 2006]%[9D4B8]
Một ca nô xuôi dòng trên một khúc sông từ $A$ đến $B$ dài $80$km, sau đó lại ngược dòng đến địa điểm $C$ cách bến $B$ $72$km. Thời gian ca nô xuôi dòng ít hơn thời gian ngược dòng là $15$ phút. Tính vận tốc riêng của ca nô biết vận tốc của dòng nước là $4$km/h.
\loigiai{
	Gọi $x$(km/h) là vận tốc riêng của ca nô (Điều kiện $x>4$).\\
	Thời gian ca nô đi từ $A$ đến $B$ là $\dfrac{80}{x+4}$ và thời gian ca nô đi từ $B$ đến $C$ là $\dfrac{72}{x-4}$.\\
	Vì thời gian ca nô xuôi dòng ít hơn thời gian ngược dòng là $15$ phút nên ta có phương trình\\
	$\dfrac{80}{x+4}+\dfrac{1}{4}=\dfrac{72}{x-4} \Leftrightarrow 320(x-4)+(x+4)(x-4)=288(x+4) \Leftrightarrow x^2+32x-2448=0 \Leftrightarrow \hoac{&x=36 \\ &x=-68.}$
	Vậy vận tốc riêng của ca nô là $36$km/h
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở Hà Nội, 2006]%[9D4B4]
Tìm toạ độ giao điểm của $A$ và $B$ của đồ thị hàm số $y=2x+3$ và $y=x^2$. Gọi $D$ và $C$ lần lượt là hình chiếu vuông góc của $A$ và $B$ trên trục hoành. Tính diện tích tứ giác $ABCD$.
\loigiai{
	Phương trình hoành độ giao điểm của hai đường $y=2x+3$ và $y=x^2$ là
	$$x^2 = 2x + 3 \Leftrightarrow \hoac{&x = -1 \Rightarrow y = 1 \\ &x = 3 \Rightarrow y = 9.}$$
	Suy ra $A(-1; 1)$ và $B(3; 9)$.\\
	Vì $D$ và $C$ lần lượt là hình chiếu vuông góc của $A$ và $B$ trên trục hoành nên ta có $D(-1; 0)$ và $C(3; 0)$.\\
	$ABCD$ là hình thang vuông tại $C$ và $D$ nên có diện tích là
	$$S_{ABCD} = \dfrac{(AD + BC)\cdot CD}{2} = \dfrac{(1 + 9)\cdot 4}{2} = 20 \,\text{(đvdt).}$$
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở Hà Nội, 2006]%[9H3B7]%[9H3B3]%[9H3G3]
Cho đường tròn $(O)$ có đường kính $AB=2R$, $C$ là trung điểm của $OA$ và dây $MN$ vuông góc với $OA$ tại $C$. Gọi $K$ là điểm tuỳ ý trên cung nhỏ $\wideparen{BM}$, $H$ là giao điểm của $AK$ và $MN$.
\begin{enumerate}
\item Chứng minh rằng tứ giác $BCHK$ là tứ giác nội tiếp;
\item Tính tích $AH.AK$ theo $R$;
\item Xác định vị trí của điểm $K$ để tổng $(KM+KN+KB)$ đạt giá trị lớn nhất và tính giá trị lớn nhất đó.
\end{enumerate}
\loigiai{
	\begin{center}
		\begin{tikzpicture}
			\tkzInit[xmin=-4,ymin=-4,xmax=4,ymax=4] \tkzClip
			\tkzDefPoints{-3/0/A,0/0/O,3/0/B}
			\tkzDefMidPoint(O,A)\tkzGetPoint{C}
			\tkzDefLine[perpendicular=through C](A,B) \tkzGetPoint{c}
			\tkzDefCircle[radius](O,A)
			\tkzInterLC(C,c)(O,A) \tkzGetPoints{N}{M}
			\tkzDefPointBy[rotation = center O angle 40](B) \tkzGetPoint{K}
			\tkzDefCircle[radius](K,B)
			\tkzInterLC(K,N)(K,B) \tkzGetPoints{D'}{D}
			\tkzInterLL(A,K)(M,N) \tkzGetPoint{H}
			\tkzDrawCircle[radius](O,A)
			\tkzDrawSegments(A,B M,N A,K B,K K,M K,N B,M B,N B,D M,A M,O)
			\tkzDrawPoints(O,A,B,C,M,N,K,H,D)				
			\tkzLabelPoints[above](M)
			\tkzLabelPoints[below](O,N)
			\tkzLabelPoints[left](A)
			\tkzLabelPoints[right](B)
			\tkzLabelPoints[above left](H,D)
			\tkzLabelPoints[below left](C)
			\tkzLabelPoints[above right](K)
			\tkzMarkRightAngles(B,C,H B,K,H)	
		\end{tikzpicture}
	\end{center}
	\begin{enumerate}
		\item Tứ giác $BCHK$ có $\widehat{BCH} = 90^\circ$ (gt) và $\widehat{BKH} = 90^\circ$ (góc nội tiếp chắn nửa đường tròn).\\
				Suy ra $BCHK$ là tứ giác nội tiếp.
		\item Hai tam giác $ACH$ và $AKB$ có $\widehat{ACH} = \widehat{AKB} = 90^\circ$ và $\widehat{BAK}$ chung.\\
				$\Rightarrow \triangle ACH \backsim \triangle AKB$ $\Rightarrow \dfrac{AC}{AK} = \dfrac{AH}{AB} \Rightarrow AH.AK = AB.AC = 2R \cdot \dfrac{R}{2} = R^2$.
		\item Trên đoạn $KN$ lấy điểm $D$ sao cho $KD = KB$.\\
				Dễ thấy hai tam giác $BMN$ và $KBD$ là các tam giác đều.\\
				Ta có $\widehat{BMK} = \widehat{BNK}$ $(1)$ (góc nội tiếp cùng chắn cung $KB$).\\
				Ta lại có $\widehat{NBD} = \widehat{MKB} = 120^\circ$ $(2)$.\\
				Từ $(1)$ và $(2)$ suy ra $\widehat{MBK} = \widehat{BND}$ (tổng các góc trong của một tam giác bằng $180^\circ$).\\
				Hai tam giác $MBK$ và $NBD$ có $BN = BM$, $\widehat{MBK} = \widehat{BND}$, $BK = BD$.\\
				$\Rightarrow \triangle MBK = \triangle NBD$ (c-g-c) $\Rightarrow MK = ND$.\\
				Do đó, ta có $KM+KN+KB = DN + DK + KN = 2KN$.\\
				Suy ra tổng $(KM+KN+KB)$ đạt giá trị lớn nhất khi $KN$ đạt giá trị lớn nhất $\Leftrightarrow KN$ là đường kính.\\
				Vậy tổng $(KM+KN+KB)$ đạt giá trị lớn nhất là $4R$ khi $K$ là điểm đối xứng của $N$ qua $O$ hay $K$ là điểm chính giữa của cung nhỏ $BC$.
	\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở Hà Nội, 2006]%[9D5G1]   
Cho hai số dương $x, y$ thoả mãn điều kiện $x+y =2$. Chứng minh : $x^2y^2(x^2+y^2) \le 2$.
\loigiai{
	Ta có $x^2y^2(x^2+y^2) = \dfrac{1}{2}xy\left[2xy(x^2 + y^2)\right] \leq \dfrac{1}{2}xy\left(\dfrac{x^2 + 2xy + y^2}{2}\right)^2 = 2xy \leq 2\left(\dfrac{x + y}{2}\right)^2 = 2$.
}
\end{ex}
