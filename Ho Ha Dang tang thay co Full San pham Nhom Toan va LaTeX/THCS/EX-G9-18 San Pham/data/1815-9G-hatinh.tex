\textbf{I. PHẦN GHI KẾT QUẢ} (\textit{thí sinh chỉ cần ghi kết quả vào tờ giấy thi})
\begin{ex}%[Đề thi HSG 9, tỉnh Hà Tĩnh, năm 2018]%[Nguyễn Thế Út, Dự án EX-9G-18]%[9H0B8]
	Tìm số cạnh của đa giác lồi có $27$ đường chéo.
	\loigiai{
		Gọi số cạnh của đa giác lồi là $n$ với $n\in\mathbb{N}$, $n>3.$ Ta có $\dfrac{n(n-3)}{2}=27$. Suy ra $n=9$.
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Hà Tĩnh, năm 2018]%[Nguyễn Thế Út, Dự án EX-9G-18]%[8D0K4]
	Cho $a_1=2017$ và $a_{n+1}=a_n+2017$ với mọi $n\ge 1$, $n\in\mathbb{N}$. Tìm $a_{2018}$.
	\loigiai{
		Ta có $a_2=a_1+2017=2.2017$, $a_3=a_2+2017=3.2017$, $\ldots$ Do đó $a_{2018}=2018.2017=4070306$.
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Hà Tĩnh, năm 2018]%[Nguyễn Thế Út, Dự án EX-9G-18]%[9D0K1]
	Cho $4a^2+b^2=5ab$ với $b>2a>0$. Tính giá trị của biểu thức $P=\dfrac{5ab}{3a^2+2b^2}.$
	\loigiai{
		Ta có $4a^2+b^2=5ab\Leftrightarrow (a-b)(4a-b)=0$. Do $b>2a>0$ nên $b=4a$. Suy ra $P=\dfrac{20a^2}{3a^2+3a^2}=\dfrac{4}{7}.$
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Hà Tĩnh, năm 2018]%[Nguyễn Thế Út, Dự án EX-9G-18]%[9D3K8]
	Hai vật chuyển động trên một đường tròn có chu vi $200$m, vận tốc vật thứ nhất là $4$m/s; vật thứ hai là $6$m/s. Hai vật xuất phát cùng một thời điểm tại một vị trí và chuyển động cùng chiều. Hỏi trong $16$ phút vật thứ hai vượt lên trước vật thứ nhất mấy lần? (không kể lúc xuất phát).
	\loigiai{
		Gọi $t$ là thời gian để hai vật gặp nhau tính từ lúc xuất phát. Quãng đường để mỗi vật đi được đến lúc gặp nhau là $S_1=v_1t=4t$, $S_2=v_2t=6t$.\\
		Vì hai vật đi cùng chiều nên ta có $S_2-S_1=S\Rightarrow 6t-4t=200\Rightarrow t=100$ (giây).\\
		Do đó cứ sau $100$ giây chúng gặp nhau một lần. Vì vậy sau $16$ phút ($960$ giây) thì chúng sẽ gặp nhau số lần là $\left[\dfrac{960}{100}\right]=9$ lần, hay vật thứ hai vượt lên trước vật thứ nhất $9$ lần.
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Hà Tĩnh, năm 2018]%[Nguyễn Thế Út, Dự án EX-9G-18]%[9H0K8]
	Có bao nhiêu tam giác khác nhau mà độ dài các cạnh là các số tự nhiên (cùng đơn vị đo) thuộc tập hợp $\left\{1,2,3,4,5,6,7\right\}$.
	\loigiai{
		Số tam giác khác nhau là $$\left[\dfrac{(n+1)(n+3)(2n+1)}{24}\right]=\left[\dfrac{8\cdot10\cdot15}{24}\right]=50.$$
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Hà Tĩnh, năm 2018]%[Nguyễn Thế Út, Dự án EX-9G-18]%[9D0K2]
	Giải phương trình: $\sqrt[3]{1-x}+\sqrt{x+3}=2$.
	\loigiai{
		Điều kiện xác định: $x\ge 3$.\\ Đặt $\sqrt[3]{1-x}=a$, $\sqrt{x+3}=b\ge 0$, ta có $$\heva{ a+b=2,\\ a^3+b^2=4}\Rightarrow a(a^2+a-4)=0.$$
		Ta có $a=0\Rightarrow x=1$; $a=\dfrac{-1\pm \sqrt{17}}{2}\Rightarrow x=\dfrac{15\pm 5\sqrt{17}}{2}$.\\
		Vậy phương trình có tập nghiệm là $S=\left\{1;\dfrac{15\pm 5\sqrt{17}}{2}\right\}.$
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Hà Tĩnh, năm 2018]%[Nguyễn Thế Út, Dự án EX-9G-18]%[9D0G2]
	Cho các số $a$, $b$ thỏa mãn $a^3+8b^3=1-6ab$. Tính $a+2b$.
	\loigiai{
		Ta có $x^3+y^3+z^3=0\Rightarrow \hoac{ & x+y+z=0 \\ & x=y=z.}$\\
		Do đó $a^3+8b^3=1-6ab\Leftrightarrow a^3+(2b)^3+(-1)^3=3a(2b)(-1)\Rightarrow \hoac{& a+2b-1=0,\\& a=2b=-1.} \Rightarrow \hoac{&a+2b=1,\\ &a+2b=-2.}$
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Hà Tĩnh, năm 2018]%[Nguyễn Thế Út, Dự án EX-9G-18]%[9D0G5]
	Tìm các số nguyên dương $a$, $b$, $c$ ($b>c$) thỏa mãn: $\heva{
	&b^2+c^2=a^2\\
	&2(a+b+c)=bc.}$
	\loigiai{
		Ta có $b^2+c^2=a^2\Rightarrow (b+c)^2-4(a+b+c)=a^2\Rightarrow (b+c-2)^2=(a+2)^2$. Vì $b>c\ge 1$ nên $b+c-2\ge 1$, do đó $b+c-2=a+2$, suy ra $$a=b+c-4\Rightarrow b^2+c^2=(b+c-4)^2\Rightarrow (b-4)(c-4)=8.$$ Vì $b-4>c-4\ge -3$ nên ta có các trường hợp sau
		\begin{itemize}
			\item Trường hợp 1: $\heva{& b-4=8\\ &c-4=1 }\Rightarrow \heva{&b=12\\& c=5}\Rightarrow a=13.$
			\item Trường hợp 2: $\heva{& b-4=4\\ &c-4=2 }\Rightarrow \heva{&b=8\\& c=6}\Rightarrow a=10.$
		\end{itemize}
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Hà Tĩnh, năm 2018]%[Nguyễn Thế Út, Dự án EX-9G-18]%[8H0K2]
	Biết khoảng cách từ trọng tâm tam giác $ABC$ đến các cạnh tỉ lệ với các số $2$; $3$; $4$ và chu vi của tam giác $ABC$ là $26$. Tính độ dài các cạnh của tam giác $ABC$.
	\loigiai{
		Gọi độ dài các cạnh $BC=a$, $CA=b$, $AB=c$; độ dài các đường cao kẻ từ đỉnh $A$, $B$, $C$ lần lượt là $x$, $y$, $z$. Khoảng cách từ trọng tâm tam giác $ABC$ đến các cạnh tỉ lệ với các số $2$; $3$; $4$ nên ta có $\dfrac{x}{2}=\dfrac{y}{3}=\dfrac{z}{4}=k$. Mặt khác $ax=by=cz=2S_{ABC}$ nên ta có $$\dfrac{a}{\dfrac{1}{x}}=\dfrac{b}{\dfrac{1}{y}}=\dfrac{c}{\dfrac{1}{z}}=\dfrac{a}{\dfrac{1}{2k}}=\dfrac{b}{\dfrac{1}{3k}}=\dfrac{c}{\dfrac{1}{4k}}=\dfrac{a+b+c}{\dfrac{13}{12k}}=24k.$$
		Từ đó suy ra $a=12$, $b=8$, $c=6$.
	}
\end{ex}

\textbf{II. PHẦN TỰ LUẬN} (\textit{thí sinh trình bày lời giải vào tờ giấy thi})
\begin{ex}%[Đề thi HSG 9, tỉnh Hà Tĩnh, năm 2018]%[Nguyễn Thế Út, Dự án EX-9G-18]%[9D0G2]
	Giải hệ phương trình: $\heva{&2y^2-x^2=1&(1)\\ &2(x^3-y)=y^3-x.&(2)}$
	\loigiai{
		Thay $1=2y^2-x^2$ vào phương trình $(2)$ ta được
		\begin{align*}
		&2x^3-2y(2y^2-x^2)=y^3-x(2y^2-x^2)\\ 
		\Rightarrow&x^3-5y^3+2x^2y+2xy^2=0.\tag{3}
		\end{align*}
		Đặt $y=tx$ thì phương trình $(3)$ trở thành $x^3(5t^3-2t^2-2t-1)=0$. 
		\begin{itemize}
			\item Nếu $x=0$, thay vào phương trình $(2)$ ta được $y=0$, không thỏa mãn phường trình $(1)$.
			\item Nếu $5t^3-2t^2-2t-1=0\Leftrightarrow (t-1)(5t^2+3t+1)=0\Rightarrow t=1$ (vì $5t^2+3t+1>0$). Suy ra $y=x$. Do đó ta có hệ phương trình $\heva{&x^2=1\\&x(x^2-1)=0.}\Rightarrow x=\pm 1.$
		\end{itemize}
		Vậy hệ phương trình đã cho có nghiệm là $(x;y)\in\left\{(-1;-1), (1;1)\right\}$.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Hà Tĩnh, năm 2018]%[Nguyễn Thế Út, Dự án EX-9G-18]%[9H0G7]
	Cho tam giác $ABC$ vuông tại $A$ có $AB<AC$, ngoại tiếp đường tròn tâm $O$. Gọi $D$, $E$, $F$ lần lượt là tiếp điểm của $(O)$ với các cạnh $AB$, $AC$, $BC$; $I$ là giao điểm của $BO$ và $EF$; $M$ là điểm di động trên đoạn $CE$. Gọi $H$ là giao điểm của $BM$ và $EF$.
\begin{enumerate}
		\item  Chứng minh rằng nếu $AM=AB$ thì các tứ giác $BDHF$, $ABHI$ nội tiếp.
	    \item  Gọi $N$ là giao điểm của $BM$ với cung nhỏ $EF$ của $(O)$; $P$ và $Q$ lần lượt là hình chiếu của $N$ trên $DE$, $DF$. Chứng minh $PQ\leq EF$.
\end{enumerate}
	\loigiai{
		Gọi $K$ là giao điểm của $BO$ với $DF$. Ta có $\triangle IKF$ vuông tại $K$. Hình chữ nhật $ADOE$ có $OD=OE$ nên $ADOE$ là hình vuông. Từ đó suy ra $\widehat{DFE}=\dfrac{1}{2}\widehat{DOE}=45^\circ\Rightarrow \widehat{BIF}=45^\circ$.
		\begin{center}
			{\begin{tikzpicture}[scale=2.2]
				\tkzInit[ymin=-0.5,ymax=3.25,xmin=-1,xmax=6]
				\tkzClip
				\tkzDefPoints{0/0/A,0/3/B,5.75/0/C,3/0/M} 
				\tkzDrawCircle[in](C,A,B)\tkzGetPoint{O}
				\tkzDefPointBy[projection=onto A--B](O)\tkzGetPoint{D}
				\tkzDefPointBy[projection=onto A--C](O)\tkzGetPoint{E}
				\tkzDefPointBy[projection=onto C--B](O)\tkzGetPoint{F}
				\tkzInterLL(B,O)(E,F)\tkzGetPoint{I}
				\tkzInterLL(B,M)(E,F)\tkzGetPoint{H}
				\tkzInterLL(B,O)(D,F)\tkzGetPoint{K}
				\tkzInterLC(M,B)(O,D)\tkzGetSecondPoint{N}
				\tkzDefPointBy[projection=onto D--E](N)\tkzGetPoint{P}
				\tkzDefPointBy[projection=onto D--F](N)\tkzGetPoint{Q}
				\tkzLabelPoints[right](O,H,N)
				\tkzLabelPoints[left](A,D,B,Q,K)
				\tkzLabelPoints[below right](I)
				\tkzLabelPoints[below](C,P,M,E)
				\tkzLabelPoints[above](F)
				\tkzDrawPoints(A,O,B,C,D,E,F,M,N,P,Q,K,H,I)
				\tkzDrawSegments(A,O A,B A,C O,B O,D O,E O,F B,C M,B D,P D,F E,F I,B N,E N,P N,Q N,F N,D I,A P,Q H,D H,O)
				\tkzMarkRightAngles(C,A,B)
				\end{tikzpicture}}
		\end{center}
	\begin{enumerate}
			\item  Khi $AM=AB$ thì $\triangle ABM$ vuông cân tại $A$ $\Rightarrow \widehat{DBH}=45^\circ=\widehat{DFH}$. Do đó tứ giác $BDHF$ nội tiếp một đường tròn. Mà tứ giác $BDOF$ nội tiếp nên năm điểm $B$, $D$, $O$, $H$, $F$ cùng thuộc đường tròn đường kính $BO$. Khi đó $\widehat{BFO}=\widehat{BHO}=90^\circ$ $\Rightarrow OH\perp BM$, mà $\triangle ABM$ vuông cân có $AH$ là phân giác nên $AH\perp BM$, suy ra ba điểm $A$, $O$, $H$ thẳng hàng. Từ đó ta có $\widehat{BAH}=\widehat{BIH}=45^\circ$, nên tứ giác $ABHI$ nội tiếp một đường tròn.\\
		    \item  Ta có tứ giác $PNQD$ nội tiếp đường tròn nên $\widehat{NPQ}=\widehat{NDQ}=\widehat{NEF}$. Tương tự ta có $\widehat{NQP}=\widehat{NDP}=\widehat{NFE}$. Do đó $\triangle NEF\backsim\triangle NPQ$ $\Rightarrow \dfrac{PQ}{EF}=\dfrac{NP}{NE}\le 1 \Rightarrow PQ\le EF$. Đẳng thức xảy ra khi điểm $P$ trùng với điểm $F$ và điểm $Q$ trùng với điểm $E$ hay $DN$ là đường kính của đường tròn $(O)$.
	\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Hà Tĩnh, năm 2018]%[Nguyễn Thế Út, Dự án EX-9G-18]%[9D0G7]
	Cho $x,y$ là các số nguyên không đồng thời bằng $0$. Tìm giá trị nhỏ nhất của biểu thức: $$F=\left | 5x^{2}+11xy-5y^{2} \right |.$$
	\loigiai{
		Đặt $F=\left | 5x^{2}+11xy-5y^{2} \right |=f(x;y)$ và $m$ là giá trị nhỏ nhất của biểu thức $F$. Ta có $m$ là số nguyên và $f(0;1)=f(1;0)=5$ nên $m\le 5$.\\
		Vì $x,y$ là các số nguyên không đồng thời bằng $0$ nên $5x^{2}+11xy-5y^{2}\ne 0$ hay $F\ne 0$.\\
		Xét $x=2n$; $y=2k$ ta có $f(x;y)=f(2n;2k)=4f(n;k)$ nên giá trị $f(2n;2k)$ không thể là giá trị nhỏ nhất của $F$. Do đó giá trị nhỏ nhất của $F$ xảy ra khi $x,y$ không cùng chẵn, vì vậy $m$ là số lẻ.\\
		Nếu $m=1$, suy ra tồn tại $x,y$ để 
		{\allowdisplaybreaks
			\begin{align*}
			&\left | 5x^{2}+11xy-5y^{2} \right |=1\\
			\Leftrightarrow& 100x^2+220xy-100y^2=\pm 20\\
			\Leftrightarrow& (10x+11y)^2-221y^2=\pm 20\\
			\Leftrightarrow& (10x+11y)^2\pm 20=221y^2\,\vdots\,13,
			\end{align*}}
	\noindent	suy ra $(10x+11y)^2$ chia cho $13$ dư $6$ hoặc dư $7$, mà số chính phương khi chia cho $13$ có dư là $0$, $1$, $3$, $4$, $9$, $10$, $12$ nên điều này vô lí.\\
		Nếu $m=3$, suy ra tồn tại $x,y$ để 
		{\allowdisplaybreaks
			\begin{align*}
			&\left | 5x^{2}+11xy-5y^{2} \right |=3\\
			\Leftrightarrow& 100x^2+220xy-100y^2=\pm 60\\
			\Leftrightarrow& (10x+11y)^2-221y^2=\pm 60\\
			\Leftrightarrow& (10x+11y)^2\pm 60=221y^2\,\vdots\,13,
			\end{align*}}
		suy ra $(10x+11y)^2$ chia cho $13$ dư $5$ hoặc dư $8$, mà số chính phương khi chia cho $13$ có dư là $0$, $1$, $3$, $4$, $9$, $10$, $12$ nên điều này vô lí.\\
		Vậy giá trị nhỏ nhất của $F$ là $5$, đạt được khi $(x;y)\in\left\{(1;0), (0;1)\right\}$.
	}
\end{ex}
