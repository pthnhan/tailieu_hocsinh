\begin{ex}%[Câu 16 Đề tham khảo - Kì thi THPT Quốc gia năm 2018]%[2D1B4]

		\end{itemize}
	}
\end{ex}

\begin{ex}%[Câu 50, đề tham khảo 2018 - Bộ giáo dục và đào tạo]%[2H1K3]
	Cho hàm số $f(x)$ có đạo hàm liên tục trên đoạn $[0;1]$ thỏa mãn $$f(1)=0, \, \displaystyle \int \limits_{0}^{1}[f'(x)]^2 \mathrm{\,d}x=7, \, \displaystyle \int \limits_{0}^{1}x^2 f(x) \mathrm{\,d}x=\dfrac{1}{3}.$$
	Tích phân $\displaystyle \int \limits_{0}^{1}f(x) \mathrm{\,d}x$ bằng
	\choice
	{\True $\dfrac{7}{5}$}
	{$1$}
	{$\dfrac{7}{4}$}
	{$4$}
	\loigiai{\noindent{\bf Cách 1.}
		Đặt  $\begin{cases} u=f(x)       \\ \mathrm{\,d}v=x^2 \mathrm{\,d}x  \end{cases}$
		$\Rightarrow$  $\begin{cases} \mathrm{\,d}u=f'(x)\mathrm{\,d}x     \\  v=\dfrac{x^3}{3}. \end{cases}$\\
		Khi đó: $\dfrac{1}{3}=$ $\displaystyle \int \limits_{0}^{1}x^2 f(x) \mathrm{\,d}x=\left[\dfrac{x^3}{3}f(x)\right]\Bigg|_0^1$ $-$ $\displaystyle \int \limits_{0}^{1} \dfrac{x^3}{3}f'(x)\mathrm{\,d}x$. \\
		Suy ra  $\displaystyle \int \limits_{0}^{1}x^3 f'(x) \mathrm{\,d}x=-1$.\hfill$(1)$\\
		Mặt khác, do
		$\displaystyle \int \limits_{0}^{1} [f'(x)]^2 \mathrm{\,d}x=7$ nên \[\displaystyle \int \limits_{0}^{1}   \left[[f'(x)]^2 +14x^3f'(x)+49x^6 \right] \mathrm{\,d}x=0 = 7-14+7=0.\]
		Hay $\displaystyle \int \limits_{0}^{1}\left(f'(x)+7x^3\right)^2 \mathrm{\,d}x=0$.\hfill$(2)$\\
		Suy ra $f'(x)+7x^3=0 \Rightarrow$ $f'(x)=-7x^3 \Rightarrow f(x)=-\dfrac{7}{4}x^4+C$.\\
		Mà $f(1)=0$ nên $C=\dfrac{7}{4}$, suy ra $f(x)=\dfrac{7}{4}(1-x^4)$. Khi đó
		$\displaystyle \int \limits_{0}^{1}f(x) \mathrm{\,d}x=\dfrac{7}{5}.$\\
		\noindent{\bf Lưu ý.} Có thể  giải thích vì sao từ $(2)$ suy ra $f'(x) + 7{x^3} = 0,\,\forall x \in \left[ {0;1} \right]$ như sau:
		Theo giả thiết ta có hàm số $y = {\left( {f'(x) + 7{x^3}} \right)^2}$ liên tục và không âm trên đoạn $[0; 1]$ (do đó, đồ thị hàm số này là một đường nét liền trên đoạn $[0; 1]$ và không có điểm nào nằm bên dưới trục $Ox$). 
		Tích phân ở $(2)$ có giá trị bằng diện tích hình phẳng giới hạn bởi đồ thị hàm số $y = {\left( {f'(x) + 7{x^3}} \right)^2}$, trục hoành, đường thẳng $x=0$, đường thẳng $x=1$.
		Mà theo $(2)$ thì hình phẳng này có diện tích bằng $0$ nên $f'(x) + 7{x^3} = 0,\,\forall x \in \left[ {0;1} \right]$.\\
		\noindent{\bf Cách 2 (tiếp nối từ (1)).} Ta có 
		bất đẳng thức  Bunhiacovski đối với tích phân: Nếu hai hàm số $f(x)$, $g(x)$ liên tục trên đoạn $[a; b]$ thì ta luôn có
		$${\left[ {\int\limits_a^b {f(x).g(x)\mathrm{d}x} } \right]^2} \le \left( {\int\limits_a^b {{f^2}(x)\mathrm{d}x} } \right).\int\limits_a^b {{g^2}(x)\mathrm{d}x}.$$
		Dấu "=" xảy ra khi và chỉ khi $g(x) = kf(x),\,\forall x \in \left[ {a;b} \right]$. Trở lại bài toán. 
		Từ $(1)$, ta có:
		$$1 = {\left( {\int\limits_0^1 {{x^3}f'(x)\mathrm{d}x} } \right)^2} \le \int\limits_0^1 {{x^6}\mathrm{d}x}  \cdot \int\limits_0^1 {{{\left[ {f'(x)} \right]}^2}\mathrm{d}x}  = \frac{1}{7} \cdot 7 = 1.$$
		Như vậy dấu "=" xảy ra, tức là $f'(x) = k{x^3}$. Thay trở lại vào $(1)$, ta được:
		$$k\int\limits_0^1 {{x^6}\mathrm{d}x}  =  - 1 \Rightarrow \frac{k}{7} =  - 1 \Rightarrow k =  - 7.$$
		Vậy $\displaystyle f'(x) =  - 7{x^3} \Rightarrow f(x) =  - \frac{7}{4}{x^4} + C\mathop  \Rightarrow \limits^{\text{do}\,\,f(1) = 0} f(x) =  - \frac{7}{4}{x^4} + \frac{7}{4}.$\\
		Do đó $\displaystyle \int \limits_{0}^{1}f(x) \mathrm{\,d}x=\dfrac{7}{5}.$}
\end{ex}
