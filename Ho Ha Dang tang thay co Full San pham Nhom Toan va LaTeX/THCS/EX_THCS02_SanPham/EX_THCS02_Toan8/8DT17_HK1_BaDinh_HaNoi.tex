\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Đống Đa, Hà Nội]%[9D3B3]
	Hãy phân tích các đa thức sau thành nhân tử:
	\begin{enumerate}
		\item $6x^{2}-3xy$
		\item $x^{2}-y^{2}-6x-9$	
		\item $x^{2}+5x-6$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $6x^{2}-3xy=3x(2x-y)$.
			\item $x^{2}-y^{2}-6x+9=(x^{2}-6x+9)-y^{2}=(x-3)^{2}-y^{2}=(x-y-3)(x+y-3)$.
			\item $x^{2}+5x-6=x^{2}-x+6x-6=x(x-1)+6(x-1)=(x+6)(x-1)$.
		\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Đống Đa, Hà Nội]%[9D4Y4]
Thực hiện phép tính:
\begin{enumerate}
	\item $(x+2)^2-(x-3)(x+1)$
	\item $(x^3-2x^2+5x-10) : (x-2)$
\end{enumerate}
	\loigiai
	{
	\begin{enumerate}
		\item 
		$(x+2)^{2}-(x-3)(x+1)=x^{2}+4x+4-(x^{2}-2x-3)=x^{2}+4x+4-x^2+2x+3=6x+7$.
		\item $(x^{3}-2x^{2}+5x-10) : (x-2)=(x-2)(x^{2}+5) : (x-2)=(x^{2}+5)$.
	\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Đống Đa, Hà Nội]%[8H2G3]
	Cho 2 biểu thức
	\[A=\dfrac{x-5}{x-4} ~ \text{và} ~ B = \dfrac{x+5}{2x} - \dfrac{x-6}{5-x} - \dfrac{2x^2-2x-50}{2x^2-10x} ~ \text{với} ~ x \neq 0; x \neq 5; x \neq 4\]
	\begin{enumerate}
		\item Tính giá trị của $A$ khi $x^{2} - 3x = 0$.
		\item Rút gọn $B$.
		\item Tìm giá trị nguyên của x để $P = \dfrac{A}{B}$ có giá trị nguyên.
	\end{enumerate}
	\loigiai
	{
	\begin{enumerate}
		\item Ta có $x^2-3x=0 \Leftrightarrow x(x-3)=0 \Leftrightarrow \left[\begin{aligned}
			&x=0 \\
			&x=3
		\end{aligned}\right.$.
			\begin{itemize}
				\item $x=3$ ta có:
				$A=\dfrac{x-5}{x-4}$$=\dfrac{3-5}{3-4}=\dfrac{-2}{-1}=2$
				\item $x=0$ ta có: $A=\dfrac{0-5}{0-4}$$=\dfrac{-5}{-4}=\dfrac{5}{4}$
			\end{itemize}
		\item
			\begin{align*}
				B &= \dfrac{x+5}{2x}-\dfrac{x-6}{5-x}-\dfrac{2x^{2}-2x-50}{2x^{2}-10x}\\
				&= \dfrac{x+5}{2x}+\dfrac{x-6}{x-5}-\dfrac{2x^{2}-2x-50}{2x^{2}-10x}\\
				&= \dfrac{(x+5)(x-5)+2x(x-6)-(2x^{2}-2x-50)}{2x^{2}-10x}\\
				&= \dfrac{x^{2}-25+2x^{2}-12x-2x^{2}+2x+50}{2x^{2}-10x}\\
				&= \dfrac{x^{2}-10x+25}{x^{2}-10x} = \dfrac{(x-5)^{2}}{2x(x-5)} = \dfrac{x-5}{2x}
			\end{align*}
		\item Ta có \[P = \dfrac{A}{B}=\dfrac{x-5}{x-4} : \dfrac{x-5}{2x} =\dfrac{2x}{x-4}
		=2+\dfrac{8}{x-4}\]
		Để $P$ nguyên thì $x-4 \in \text{Ư}(8) = \left\{\pm1;\pm2;\pm4;\pm8 \right\} $
		\begin{center}
		\begin{tabular}{|c|c|c|c|c|c|c|c|c|}
			\hline 
			$x-4$ & $-8$ &$-4$&$-2$&$-1$&$1$&$2$&$4$&$8$\\ 
			\hline 
			$x$ &$-4$&$0$&$2$&$3$&$5$&$6$&$8$&$12$\\
			\hline 
		\end{tabular}
		\end{center}
		Mặt khác $x\neq 0, x\neq5, x\neq4$ nên $x \in \{-4;2;3;6;8;12\}$.
	\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Đống Đa, Hà Nội]%[8H2G3]
		Cho $\triangle ABC$ cân tại $A$, đường cao $AD$, $O$ là trung điểm của $AC$, điểm $E$ đối xứng với $D$ qua điểm $O$.
	\begin{enumerate}
		\item Chứng minh rằng tứ giác $AECD$ là hình chữ nhật.
		\item Gọi $I$ là trung điểm của $AD$. Chứng tỏ rằng $I$ là trung điểm của $BE$.
		\item Cho $AB = 10cm, BC = 12cm$, tính diện tích $\triangle OAD$.
		\item Đường thẳng $OI$ cắt $AB$ tại $K$. Tìm điều kiện của $\triangle ABC$ để tứ giác $AEDK$ là hình thang cân.
	\end{enumerate}
\end{ex}
	\loigiai
	{
	\begin{center}
		\begin{tikzpicture}[scale = 1]
		\tkzDefPoints{-3/0/B,3/0/C}
		\tkzDefMidPoint(B,C)	\tkzGetPoint{D}
		\tkzDefPointWith[orthogonal,K=1.3](D,C)
		\tkzGetPoint{A}
		\tkzDefMidPoint(A,C)	\tkzGetPoint{O}
		\tkzDefMidPoint(A,D)	\tkzGetPoint{I}
		\tkzDefMidPoint(A,B)	\tkzGetPoint{K}
		\tkzDefPointBy[symmetry = center O](D)
		\tkzGetPoint{E}
		\tkzMarkRightAngle[size=0.4,fill=gray!50](A,D,C)
		\tkzMarkSegments[mark=|](A,O O,C)
		\tkzMarkSegments[mark=||](D,O O,E)
		\tkzMarkSegments[mark=x](A,I I,D)
		\tkzDrawPolygon(A,E,C,B)
		\tkzDrawSegments(A,C D,E A,D K,O K,D B,E)
		\tkzDrawPoints(B,C,D,A,O,I,K,E)
		\tkzLabelPoints[above](A,E)
		\tkzLabelPoints[below](B,D,C)
		\tkzLabelPoints[left](K)
		\tkzLabelPoints[right](O)
		\tkzLabelPoints(I)
		\end{tikzpicture}
	\end{center}
			\begin{enumerate}
			\item Tứ giác $AECD$ có $O$ là trung điểm của $AC$ và $DE$ (gt) nên suy ra tứ giác $AECD$ là hình bình hành. Lại có $AD \perp BC$ nên tứ giác $AECD$ là hình chữ nhật.
			\item Xét tứ giác $ABDE$ có $AE=BD$ (vì cùng bằng $DC$), hơn nữa $AE \parallel BD$ nên suy ra tứ giác $ABDE$ là hình bình hành.\\
			Mặt khác, $I$ là trung điểm của $AD$ nên $I$ cũng là trung điểm của $BE$ (tính chất hình bình hành).
			\item Xét $\triangle ABC$ có $AB = 10 cm, BC = 12cm$, khi đó $DB=6cm$. Theo định lý Pi-ta-go trong $\triangle ABD$ ta có \[AD=\sqrt{AB^2 - BD^2} = \sqrt{10^2 - 6^2} = 8(cm)\]
			Vậy $S_{ADCE} = 6 \times 8 = 48$. Khi đó $S_{OAD}=\dfrac{1}{4}S_{ADCE}$$=12cm$.
			\item Tứ giác $AEDK$ là hình thang cân nên $DK=AE$, suy ra $DC=DO=CO$ và do đó $AC=BC$. Vậy $\triangle ABC$ là tam giác đều.
		\end{enumerate}
}
\begin{ex}%[Đề thi Toán 9 Học kỳ 1 năm học 2017-2018, Đống Đa, Hà Nội]%[9D3B3]
Cho $abc=2016$. Tính giá trị của biểu thức sau:
	\[P = \dfrac{2bc - 2016}{3c-2bc+2016}-\dfrac{2b}{3-2b+ab}+\dfrac{4032-3ac}{3ac-4032+2016a}\]
\loigiai
	{
		Ta có 
		\begin{align*}
			P &= \dfrac{2bc - 2016}{3c-2bc+2016}-\dfrac{2b}{3-2b+ab}+\dfrac{4032-3ac}{3ac-4032+2016a} \\
			&= \dfrac{2bc - 2016}{3c-2bc+2016}-\dfrac{2bc}{3c-2bc+abc}+\dfrac{2abc-3ac}{3ac-2abc+2016a} \\
			&= \dfrac{2bc - 2016}{3c-2bc+2016}-\dfrac{2bc}{3c-2bc+2016}+\dfrac{2bc-3c}{3c-2bc+2016} \\
			&= \dfrac{2bc - 2016}{3c-2bc+2016}-\dfrac{2bc}{3c-2bc+2016}+\dfrac{2bc-3c}{3c-2bc+2016} \\
			&= \dfrac{2bc - 2016-2bc+2bc-3c}{3c-2bc+2016} \\
			&= \dfrac{-3c+2bc - 2016}{3c-2bc+2016}= -1
		\end{align*}
	}
\end{ex}
