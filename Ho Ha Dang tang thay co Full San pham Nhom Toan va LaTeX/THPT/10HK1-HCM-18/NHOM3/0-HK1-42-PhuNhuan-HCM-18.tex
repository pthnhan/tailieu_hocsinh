\begin{bt}%[Đề HK1, THPT Phú Nhuận, Hồ Chí Minh 17-18]%[Phan Anh]%[0D2B3-3]
Cho parabol $(P)\colon y=2x^2-(m+1)x+m$.
\begin{enumerate}
\item Khi $m=3$, xác định tọa độ đỉnh, trục đối xứng và vẽ parabol đó.
\item Với giá trị nào của $m$ thì parabol nhận đường thẳng $x=-1$ làm trục đối xứng.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item \immini{Khi $m=3$, parabol $(P)\colon y=2x^2-4x+3$
$$a=2,\, b=-4,\, c=3.$$
Ta có đỉnh của $(P)$ là $I\left(-\dfrac{b}{2a};-\dfrac{\Delta}{4a}\right)=(1;1)$.\\
Trục đối xứng của $(P)$ là đường thẳng $x=-\dfrac{b}{2a}=1$.\\
Ta có bảng giá trị là \\
\begin{tabular}{c|c|c|c|c|c}
	$x$ & $-1$ & $0$ & $1$ & $2$ & $3$\\
	\hline
	$y=2x^2-4x+3$ & $9$ & $3$ & $1$ & $3$ & $9$
\end{tabular}}
{\begin{tikzpicture}[scale=0.6,>=stealth]
\draw[->](-2,0)--(4,0)node[below]{$x$};
\draw[->](0,-1)--(0,10)node[right]{$y$};
\draw[smooth,samples=100,domain=-1.12:3.12]plot(\x,{2*(\x)^2-4*\x+3});
\draw[dashed](1,0)node[below]{$1$}--(1,1)--(0,1)node[above left]{$1$};
\draw[dashed](2,0)node[below]{$2$}--(2,3)--(0,3)node[left]{$3$};
\draw[dashed](3,0)node[below]{$3$}--(3,9)--(0,9)node[above left]{$9$}--(-1,9)--(-1,0)node[below]{$-1$};
\draw[fill=black](0,0)circle(1pt)node[above left]{$O$};
\foreach \i in {-1,0,...,3}
	{\draw[fill=black](\i,0)circle(1pt);}
\foreach \i in {1,3,9}
	{\draw[fill=black](0,\i)circle(1pt);}
\end{tikzpicture}}
\item Để parabol nhận đường thẳng $x=-1$ làm trục đối xứng thì $$-\dfrac{b}{2a}=-1\Leftrightarrow\dfrac{m+1}{4}=-1\Leftrightarrow m=-5.$$
\end{enumerate}}
\end{bt}

\begin{bt}%[Đề HK1, THPT Phú Nhuận, Hồ Chí Minh 17-18]%[Phan Anh]%[0D3B2-4]
	Giải phương trình $\sqrt{2x^2-x-3}=x+1$.
	\loigiai{Ta có
		\begin{eqnarray*}
			\sqrt{2x^2-x-3}=x+1&\Leftrightarrow&\heva{&x+1\ge0\\&2x^2-x-3=x^2+2x+1}\\
			&\Leftrightarrow&\heva{&x\ge-1\\&x^2-3x-4=0}\Leftrightarrow\heva{&x\ge-1\\&\hoac{&x=-1\\&x=4}}\Leftrightarrow\hoac{&x=-1\\&x=4.}
		\end{eqnarray*}
		Vậy phương trình có hai nghiệm là $x=-1$ và $x=4$.}
\end{bt}
\begin{bt}%[Đề HK1, THPT Phú Nhuận, Hồ Chí Minh 17-18]%[Phan Anh]%[0D3K3-4]
Giải hệ phương trình $\heva{&xy-2x=y+10\\&x^2+y^2-2x-4y=20}$.
\loigiai{Ta có $\heva{&xy-2x=y+10\\&x^2+y^2-2x-4y=20}\Leftrightarrow\heva{&2xy-4x=2y+20&\quad(1)\\&x^2+y^2-2x-4y=20&\quad(2)}$\\
Lấy $(1)$ cộng với $(2)$ ta suy ra $(x+y)^2-6(x+y)-40=0\Leftrightarrow\hoac{&x+y=-4\\&x+y=10.}$\\
\begin{itemize}
	\item Với $x+y=-4\Leftrightarrow y=-4-x$.\\
	$(1)\Leftrightarrow x(-4-x)-2x=-4-x+10\Leftrightarrow x^2+5x+6=0\Leftrightarrow\heva{&x=-2\Rightarrow y=-2\\&x=-3\Rightarrow y=-1.}$
	\item Với $x+y=10\Leftrightarrow y=10-x$.\\
	$(1)\Leftrightarrow x(10-x)-2x=10-x+10\Leftrightarrow x^2-9x+20=0\Leftrightarrow\heva{&x=5\Rightarrow y=5\\&x=4\Rightarrow y=6.}$
\end{itemize}
Vậy hệ có bốn nghiệm là $(-2;-2)$, $(-3;-1)$, $(5;5)$, $(4;6)$.}
\end{bt}

\begin{bt}%[Đề HK1, THPT Phú Nhuận, Hồ Chí Minh 17-18]%[Phan Anh]%[0D3K2-5]
	Xác định các giá trị của tham số $m$ để phương trình $x^2-2(m-4)x+m^2-3m-25=0$ có hai nghiệm phân biệt thỏa $x_1^2+x_2^2=34$.
	\loigiai{Để phương trình có hai nghiệm phân biệt thì $$\Delta'>0\Leftrightarrow(m-4)^2-\left(m^2-3m-25\right)>0\Leftrightarrow-5m+41>0\Leftrightarrow m<\dfrac{41}{5}.$$
	Theo định lý Vi-et ta có $\heva{&x_1+x_2=-\dfrac{b}{a}=2(m-4)\\&x_1x_2=\dfrac{c}{a}=m^2-3m-25.}$\\
Ta có \begin{eqnarray*}
x_1^2+x_2^2=34&\Leftrightarrow&\left(x_1+x_2\right)^2-2x_1x_2=34\,\Leftrightarrow\,4(m-4)^2-2\left(m^2-3m-25\right)=34\\
&\Leftrightarrow&2m^2-26m+80=0\,\Leftrightarrow\,\hoac{&m=5\\&m=8.}
\end{eqnarray*}
So điều kiện, ta suy ra giá trị $m$ thỏa yêu cầu bài toán là $m=5$ và $m=8$.}
\end{bt}

\begin{bt}%[Đề HK1, THPT Phú Nhuận, Hồ Chí Minh 17-18]%[Phan Anh]%[0D4K1-2]
Chứng minh rằng $4x+\dfrac{9}{x-2}\ge20$, $\forall x>2$.
	\loigiai{Ta có VT$=4x+\dfrac{9}{x-2}=4(x-2)+\dfrac{9}{x-2}+8$.\\
	Áp dụng bất đẳng thức Cô-si cho hai số $4(x-2)$ và $\dfrac{9}{x-2}$, ta được $$4(x-2)+\dfrac{9}{x-2}\ge2\sqrt{4(x-2)\cdot\dfrac{9}{x-2}}=12$$
	$$\Rightarrow\text{VT}=4(x-2)+\dfrac{9}{x-2}+8\ge12+8=20.$$
Vậy ta suy ra bất đẳng thức là đúng.}
\end{bt}

\begin{bt}%[Đề HK1, THPT Phú Nhuận, Hồ Chí Minh 17-18]%[Phan Anh]%[0H2B3-1]
Cho tam giác $ABC$ có $AB=2\sqrt{2}$, $BC=3$, góc $\widehat{ABC}=45^\circ$.
	\begin{enumerate}
		\item Tính độ dài cạnh $AC$ và diện tích tam giác $ABC$.
		\item Tính độ dài đường cao $AH$ và trung tuyến $AM$ của tam giác $ABC$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $AC^2=AB^2+BC^2-2\cdot AB\cdot BC\cdot\cos\widehat{ABC}=8+9-2\cdot2\sqrt{2}\cdot3\cdot\dfrac{\sqrt{2}}{2}=5\Rightarrow AC=\sqrt{5}$.\\
			Diện tích tam giác $ABC$ là $S_{ABC}=\dfrac{1}{2}\cdot AB\cdot BC\cdot\sin\widehat{ABC}=\dfrac{1}{2}\cdot2\sqrt{2}\cdot3\cdot\dfrac{\sqrt{2}}{2}=3$.
			\item Ta có $S_{ABC}=\dfrac{1}{2}\cdot AH\cdot BC\Rightarrow AH=\dfrac{2S_{ABC}}{BC}=\dfrac{2\cdot3}{3}=2$.\\
			Độ dài trung tuyến $AM$ là $AM=\sqrt{\dfrac{AB^2+AC^2}{2}-\dfrac{BC^2}{4}}=\sqrt{\dfrac{8+5}{2}-\dfrac{9}{4}}=\dfrac{\sqrt{17}}{2}$.
	\end{enumerate}}
\end{bt}

\begin{bt}%[Đề HK1, THPT Phú Nhuận, Hồ Chí Minh 17-18]%[Phan Anh]%[0H2B2-3]%[0H2B2-5]
Trong mặt phẳng tọa độ $Oxy$, cho $A(2;3)$, $B(2;1)$, $C(-1;1)$.
	\begin{enumerate}
		\item Chứng minh rằng $A$, $B$, $C$ tạo thành một tam giác vuông. Tính diện tích tam giác $ABC$.
		\item Tìm tọa độ điểm $M$ trên trục $Ox$ sao cho $\cos\left(\overrightarrow{AM},\overrightarrow{AB}\right)$ đạt giá trị lớn nhất.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $\overrightarrow{AB}=(0;-2)$, $\overrightarrow{BC}=(-3;0)$, suy ra $\overrightarrow{AB}\cdot\overrightarrow{BC}=0$ nên $AB\perp BC$.\\
			Vậy tam giác $ABC$ vuông tại $B$ và diện tích tam giác $ABC$ là $S=\dfrac{1}{2}\cdot AB\cdot BC=\dfrac{1}{2}\cdot2\cdot3=3$.
			\item Gọi $M(m,0)$ thuộc $Ox$, $\overrightarrow{AM}=(m-2;-3)$. Ta có  $$\cos\left(\overrightarrow{AM},\overrightarrow{AB}\right)=\dfrac{\overrightarrow{AM}\cdot\overrightarrow{AB}}{\left|\overrightarrow{AM}\right|\cdot\left|\overrightarrow{AB}\right|}=\dfrac{6}{\sqrt{(m-2)^2+9}\cdot\sqrt{4}}\le\dfrac{6}{\sqrt{0+9}\cdot2}=1$$
			Vậy $\cos\left(\overrightarrow{AM},\overrightarrow{AB}\right)$ đạt giá trị lớn nhất khi $m-2=0\Leftrightarrow m=2$\\
			$\Rightarrow M(2;0)$ là điểm cần tìm.
	\end{enumerate}}
\end{bt}
