Trên cạnh huyền $BC$ của tam giác vuông $ABC$, lấy các điểm $D$ và $E$ sao cho $BD = BA$, $CE = CA$. Tính góc $\widehat{DAE}$.
\loigiai
{
\immini{$\triangle{BAD}$ cân tại $B$, $\triangle{CAE}$ cân tại $C$. Ta có \\
\[ \widehat{BAD} = \dfrac{1}{2} \left( 180^\circ - \widehat{B} \right);\, \widehat{CAE} = \dfrac{1}{2} \left( 180^\circ - \widehat{C} \right). \]
mà $\widehat{BAD} + \widehat{CAE} = \widehat{EAD} + 90^\circ$.\\
nên
\begin{eqnarray*}
\widehat{DAE} &=& \widehat{BAD} + \widehat{CAE} - 90^\circ \\
&=& \dfrac{1}{2} \left( 180^\circ - \widehat{B} \right) + \dfrac{1}{2} \left( 180^\circ - \widehat{C} \right) - 90^\circ \\
&=& 90^\circ - \dfrac{1}{2} \left( \widehat{B} + \widehat{C} \right)\\
&=& 90^\circ - 45^\circ = 45^\circ.
\end{eqnarray*}}{
