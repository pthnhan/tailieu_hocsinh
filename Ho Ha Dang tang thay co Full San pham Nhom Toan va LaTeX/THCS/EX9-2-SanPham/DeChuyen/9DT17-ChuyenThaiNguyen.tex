\begin{ex}%[Đề thi vào 10 chuyên Thái Nguyên, 2017]%[9D1K6]
    Không dùng máy tính cầm tay hãy rút gọn: $A=\dfrac{\sqrt{3-\sqrt{5}} \left( 3+\sqrt{5}\right) }{\sqrt{10}+\sqrt{2}}.$
    \loigiai{
    	Ta có:
    	\begin{align*}
    		A&= \dfrac{\sqrt{3-\sqrt{5}}(3+\sqrt{5})}{\sqrt{10}+\sqrt{2}}=\dfrac{\sqrt{3-\sqrt{5}}(3+\sqrt{5})}{\sqrt{2}(\sqrt{5}+1)}\\
    		&=\dfrac{\sqrt{6-2\sqrt{5}}(3+\sqrt{5})}{2(\sqrt{5}+1)}=\dfrac{\sqrt{(\sqrt{5}-1)^2}(3+\sqrt{5})}{2(\sqrt{5}+1)}\\
    		&=\dfrac{(\sqrt{5}-1)(3+\sqrt{5})}{2(\sqrt{5}+1)} 		=\dfrac{3\sqrt{5}+5-3-\sqrt{5}}{2(\sqrt{5}+1)}\\
    		&=\dfrac{2\sqrt{5}+2}{2\sqrt{5}+2}=1.
    	\end{align*}}
\end{ex}

	Giải hệ phương trình sau: 
	\[ \heva{x^2+y^2-xy+4y+1&=0\\ y(7-x^2-y^2+2xy)&=2(x^2+1)}.
	\]

	&\heva{&x^2+y^2-xy+4y+1=0\\	&y(7-x^2-y^2+2xy)=2(x^2+1)} \tag{1}\\
	\Leftrightarrow  
	&\heva{&x^2+1=-y^2+xy-4y\\ &y(7-x^2-y^2+2xy)=2(-y^2+xy-4y)
	} \notag\\ 
	\Leftrightarrow &\heva{&x^2+1=-y^2+xy-4y&\\
		&y[15-(x-y)^2-2(x-y)]=0.
	}\tag{2}\\
	\end{align*}}Do $y=0$ không thỏa mãn hệ đã cho nên từ (2), ta có:
{\allowdisplaybreaks
	\begin{align*}
	&\begin{cases}

	\end{cases}\\
\Leftrightarrow &\begin{cases}
 	\hoac{x-y=-5\\ x-y=3} \\x^2+1=-y^2+xy-4y
	\end{cases}\\
	\Leftrightarrow & \hoac{\begin{cases}
		x-y=-5\\x^2+1=-y^2+xy-4y
		\end{cases}\\\begin{cases}

		\end{cases}}
	\end{align*}}Ta có:
{\allowdisplaybreaks

	&\begin{cases}

	\end{cases}\\
	\Leftrightarrow&\begin{cases}

	\end{cases} \tag{\text{hệ vô nghiệm}}
\end{align*}}
Ta có:
{\allowdisplaybreaks
\begin{align*}
	&\begin{cases}
	x-y=3\\x^2+1=-y^2+xy-4y
	\end{cases}

	\end{cases}\\

\begin{ex}%[Đề thi vào 10 chuyên Thái Nguyên, 2017]%[9D5G2]
	Cho số tự nhiên $A=\underbrace {\overline {777...7} }_{\text{n chữ số 7}}-18+2n$ với $n \in \mathbb{N}$, $n \ge 2$.\\
	\indent Chứng minh rằng $A$ chia hết cho $9.$ 
\loigiai{
Với số tự nhiên $ a\in \mathbb{N^*} $, ta đặt $ S(a) $ là tổng các chữ số của $ a $. Khi đó ta luôn có $ a-S(a) $ chia hết cho $9$.
Ta có: 
$$ A=\underset{n \text{ chữ số 7}}{\underbrace{\overline{777\dots7}}}-18+2n=7.\underset{n \text{ chữ số 1}}{\underbrace{\overline{111\dots1}}}-18+2n=7.(\underset{n \text{ chữ số 1}}{\underbrace{\overline{111\dots1}}}-n) +9(n-2).$$
Tổng các chữ số của $ \underset{n \text{ chữ số 1}}{\underbrace{\overline{111\dots1}}} $ là $ n $ nên ta có $ \underset{n \text{ chữ số 1}}{\underbrace{\overline{111\dots1}}}-n $ chia hết cho 9. \\Vì vậy $ 7(\underset{n \text{ chữ số 1}}{\underbrace{\overline{111\dots1}}}-n) $ chia hết cho 9.\\
Hiển nhiên $ 9(n-2) $ chia hết cho 9.\\
Do đó $ A $ $\vdots$ $ 9.$
}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Thái Nguyên, 2017]%[9D5G1]
Xét các số dương $a$, $b$, $c$  thỏa mãn $a+b+c \leq \sqrt{3}.$
	Tìm giá trị lớn nhất của biểu thức 
	\centerline{$P= \dfrac{a}{\sqrt{a^2+1}} + \dfrac{b}{\sqrt{b^2+1}} + \dfrac{c}{\sqrt{c^2+1}}$.}
\loigiai{
Ta có:\\
$$ (a+b+c)^2\geq 3(ab+ac+bc)\Rightarrow 3\geq 3(ab+bc+ac)\Rightarrow ab+bc+ac \leq 1. $$
Do đó ta có:
$$ ab+bc+ac \leq 1 \Rightarrow a^2+ab+bc+ac \leq a^2+1 \Rightarrow (a+b)(a+c) \leq a^2 +1. $$
Tương tự ta có:
$$ (a+b)(b+c) \leq b^2 +1, (a+c)(b+c) \leq c^2 +1.$$
Từ đó ta có:
{\allowdisplaybreaks
\begin{align*}
	&P\leq \dfrac{a}{\sqrt{(a+b)(a+c)}}+\dfrac{b}{\sqrt{(a+b)(b+c)}}+\dfrac{c}{\sqrt{(a+c)(b+c)}}\\
	\Rightarrow &P \leq \sqrt{\dfrac{a^2}{(a+b)(a+c)}}+\sqrt{\dfrac{b^2}{(a+b)(b+c)}}+\sqrt{\dfrac{c^2}{(a+c)(b+c)}}.
\end{align*}}Theo bất đẳng thức Cauchy ta có:
{\allowdisplaybreaks
\begin{align*}
	\sqrt{\dfrac{a^2}{(a+b)(a+c)}}&\leq \dfrac{1}{2}(\dfrac{a}{a+b}+\dfrac{a}{a+c})\\
	\sqrt{\dfrac{b^2}{(a+b)(b+c)}}&\leq \dfrac{1}{2}(\dfrac{b}{a+b}+\dfrac{b}{b+c})\\
	\sqrt{\dfrac{c^2}{(a+c)(b+c)}}&\leq \dfrac{1}{2}(\dfrac{c}{a+c}+\dfrac{c}{b+c}).
\end{align*}}Do đó:
\begin{align*}
	P\leq \dfrac{1}{2}(\dfrac{a}{a+b}+\dfrac{a}{a+c}+\dfrac{b}{a+b}+\dfrac{b}{b+c}+\dfrac{c}{a+c}+\dfrac{c}{b+c})=\dfrac{3}{2}.
\end{align*}
Dấu đẳng thức xảy ra khi:
$\begin{cases}
	a=b=c>)\\
	a+b+c=\sqrt{3}\\
	\dfrac{a}{a+b}=\dfrac{a}{a+c}\\
	\dfrac{b}{a+b}=\dfrac{b}{b+c}\\
	\dfrac{c}{a+c}=\dfrac{c}{b+c}
\end{cases}\Leftrightarrow a=b=c=\dfrac{\sqrt{3}}{3}.
$\\
Vậy $P$ có giá trị lớn nhất bằng $ \dfrac{3}{2} $ khi và chỉ khi $ a=b=c=\dfrac{\sqrt{3}}{3}$.
}
	
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Thái Nguyên, 2017]%[9D5G5]
	Với mỗi số nguyên dương $n$ ta kí hiệu $a_n$ là số nguyên gần $\sqrt{n}$ nhất.\\
	Ví dụ: $a_1=1$, $a_2=1$, $a_3=2$, $a_4=2$, $a_5=2$, $a_6=2$, $a_7=3$. Tính giá trị của tổng
	\begin{center}
		$S=\dfrac{1}{a_1} + \dfrac{1}{a_2} +\dfrac{1}{a_3}+\cdots+\dfrac{1}{a_{2017}}+\dfrac{1}{a_{2018}}. $
	\end{center}
\loigiai{
Gọi $ k $ là số nguyên dương gần $ \sqrt{n} $ nhất. Khi đó ta có:
{\allowdisplaybreaks
	\begin{align*}
& k-\dfrac{1}{2}<\sqrt{n}<k+\dfrac{1}{2}\Leftrightarrow k^2-k+\dfrac{1}{4}<n<k^2+k+\dfrac{1}{4} \\
 \Rightarrow& k^2-k+1\leq n<k^2+k+1.
 \end{align*}}Từ đó ta sẽ có $ 2k $ số $ a_n=k $. Vậy:
\begin{align*}
	S=\underset{2\text{ số hạng}}{\underbrace{(\dfrac{1}{1}+\dfrac{1}{1}})} +\underset{4\text{ số hạng}}{\underbrace{(\dfrac{1}{2}+\cdots+
			\dfrac{1}{2})}}+\cdots+\underset{88\text{ số hạng}}{\underbrace{(\dfrac{1}{44}+\cdots+\dfrac{1}{44}})}+\underset{38\text{ số hạng}}{\underbrace{(\dfrac{1}{45}+\cdots+\dfrac{1}{45}})}
\end{align*}
(từ $ a_{1981}=a_{1982}=\dots=a_{2018}=45 $)\\
$ S=2.44+38.\dfrac{1}{45}=\dfrac{3998}{45}. $
} 
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Thái Nguyên, 2017]%[9H3G7]
	Cho hai đường tròn tâm $O_1$ và $O_2$ nằm ngoài nhau. Đoạn thẳng $O_1O_2$ cắt đường tròn $(O_1)$ tại điểm $A$. Phần kéo dài của đoạn thẳng  $ O_1O_2 $ cắt $(O_2)$ tại $B$. Dựng đường tròn tâm $O$ tiếp xúc ngoài với đường tròn $(O_1)$ tại điểm $D$ và tiếp xúc trong với đường tròn $(O_2)$ tại $C$ (điểm $O$ không nằm trên $O_1O_2$). Chứng minh rằng các điểm $A$, $B$, $C$, $D$ cùng nằm trên một đường tròn.
	\loigiai{
		\begin{center}

Đặt $ \widehat{COD}=\alpha; \widehat{AO_1D}=\beta $. Do $\Delta O_2BC$ cân tại $O_2$ nên:	\\

Mặt khác: $ \widehat{CO_2B}=\widehat{OO_2O_1}=180^{\circ}-(\alpha+\beta) $.\hfill$(**)$\\

\[\widehat{CDA}=180^{\circ}-(\widehat{ADO_1}+\widehat{CDO})=180^{\circ}-(90^{\circ}-\dfrac{\beta}{2}+90^{\circ}-\dfrac{\alpha}{2})=\dfrac{\alpha+\beta}{2}.\tag{2}\]
Từ $(1)$ và $(2)$ suy ra $ \widehat{CBA}=\widehat{CDA}=\dfrac{\alpha+\beta}{2} $.\\
Do đó các điểm $A$, $B$, $C$, $D$ cùng nằm trên một đường tròn.
}

\begin{ex}%[Đề thi vào 10 chuyên Thái Nguyên, 2017]%[9H3G7]
	Cho ba điểm $ A $, $ B $, $ C $ phân biệt thẳng hàng theo thứ tự đó sao cho $AB<BC $. Trong một nửa mặt phẳng có bờ là đường thẳng $ AC $ dựng các hình vuông $ ABDE $ và $ BCFK $. Gọi $ I $ là trung điểm của đoạn thẳng $ EF $. Đường thẳng qua $ I $ vuông góc với $ EF $ cắt các đường thẳng $ BD $ và $ AB $ lần lượt tại $ M $ và $ N $. Chứng minh rằng:
	\begin{enumerate}
		\item[a.] Tứ giác $ AEIN $ và tứ giác $ EMID $ nội tiếp đường tròn.
		\item[b.] Ba điểm $ A $, $ I $, $ D $ thẳng hàng và các điểm $ B $, $ N $, $ F $, $ M $, $ E $ nằm trên cùng một đường tròn.
		\item[c.] Ba đường thẳng $AK$, $EF$, $CD$ đồng quy.
	\end{enumerate}
\loigiai{
	\begin{center}
	\begin{tikzpicture}[scale=0.8]
	\tkzInit[xmin=-5,xmax=6,ymin=-1,ymax=8]
	\tkzClip
	\tkzDefPoint(-2,0){A} \tkzDefPoint(0,0){B} 
	\tkzDefPoint(-2,2){E} 
	\tkzDefPoint(0,2){D}
	\tkzDefPoint(5,0){C}
	\tkzDefPoint(0,5){K}
	\tkzDefPoint(5,5){F}
	\tkzDefPoint(-1,1){O}
	\tkzDrawPoints(A,B,C,E,D,K,F,O)
	\tkzLabelPoint[below](A){$A$}
	\tkzLabelPoint[below](B){$B$}
	\tkzLabelPoint[below](C){$C$}
	\tkzLabelPoint[above left](E){$E$}
	\tkzLabelPoint[above right](D){$D$}
	\tkzLabelPoint[above left](K){$K$}
	\tkzLabelPoint[above right](F){$F$}
	\tkzLabelPoint[below](O){$O$}
	
	\tkzInterLL(A,K)(C,D) \tkzGetPoint{G}\tkzLabelPoint[above left](G){$G$}\tkzDrawPoint(G)
	\tkzDefMidPoint(E,F) \tkzGetPoint{I}\tkzLabelPoint[above right](I){$I$}\tkzDrawPoint(I)
	
	\tkzDrawSegments(A,B B,C A,E E,D D,B B,K K,F F,C E,B A,D B,F C,D A,K C,K)
	\tkzDrawSegments(D,G E,F)
	\tkzDefLine[perpendicular=through I,K=1](E,F)\tkzGetPoint{x}
	\tkzInterLL(I,x)(B,K) \tkzGetPoint{M}\tkzLabelPoint[above](M){$M$}\tkzDrawPoint(M)
	\tkzInterLL(I,x)(B,C) \tkzGetPoint{N}\tkzLabelPoint[below](N){$N$}\tkzDrawPoint(N)
	\tkzDrawSegments(M,N M,K E,M M,F)
	\tkzMarkRightAngles(D,E,A E,I,N)
	\tkzMarkAngle[size=0.5cm,% 
	opacity=.5](B,A,D)
	\tkzMarkAngle[size=0.5cm,% 
	opacity=.5](C,B,F)
	\end{tikzpicture}
	\end{center}
\begin{enumerate}
	\item[$a.$] Ta có: \\
	$ \widehat{EAN}=\widehat{EIN}=90^{\circ} $ nên tứ giác $ AEIN $ nội tiếp đường tròn đường kính $EN$.\\
	$ \widehat{EDM}=\widehat{EIM}=90^{\circ} $ nên tứ giác $ EMID $ nội tiếp đường tròn đường kính $EM$.
	\item[$b.$] Ta có $ \widehat{DAB}=\widehat{FBC}=45^{\circ} $ nên $ AD//BF $.\hfill$(1)$\\
	Gọi $ O $ là giao điểm của hai đường thẳng $ BE $ và $ AD $. Khi đó $ O $ là trung điểm của $ BE $ và $ AD $.
	Mà $ I $ là trung điểm của $ EF $ nên $ OI $ là đường trung bình của tam giác $ EBF $ ứng với cạnh $ BF $.
	Do đó $ OI//BF$.\hfill$(2)$\\
	Từ $(1)$ và $(2)$ ta có $ A,I,D $ cùng nằm trên đường thẳng qua $ O $ và song song với $ BF $.
	Vậy $ A,I,D $ thẳng hàng.

	\item[$c.$] Chứng minh tương tự ta có ba điểm $ C,I,K $ thẳng hàng.\\

	Do đó $ G $ nằm trên đường thẳng $ EF. $\\
	Vậy ba đường thẳng $ AK, CD, EF $ đồng quy.
\end{enumerate}
}
