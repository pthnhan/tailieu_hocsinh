\begin{bt}%[PTTH-MUONGBI-2017]%[1D4B2-3]%[1D4B2-7]%[Phạm Tuấn-HK2-11]
	Tính các giới hạn sau
	\begin{enumerate}
		\item[a)]  $\displaystyle \lim_{x \to 2} \dfrac{2x^2-x-6}{x-2}.$
		\item[b)] $\displaystyle \lim_{x \to +\infty} \dfrac{x-2}{3x-1-\sqrt{x^2+3}}.$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item[a)] $\displaystyle \lim_{x \to 2} \dfrac{2x^2-x-6}{x-2}= \lim_{x \to 2} \dfrac{(x-2)(2x+3)}{x-2} = \lim_{x \to 2} (2x+3) = 7.$ 
			\item[b)] $\displaystyle \lim_{x \to +\infty} \dfrac{x-2}{3x-1-\sqrt{x^2+3}}= \lim_{x \to +\infty} \dfrac{1-\dfrac{2}{x}}{3-\dfrac{1}{x} - \sqrt{1+\dfrac{3}{x^2}}} = \dfrac{1}{2}.$
		\end{enumerate}
	}
\end{bt}
