\subsection{TRỌNG TÂM KIẾN THỨC}
\subsubsection{Mô tả hình lăng trụ đứng.}
\immini{
	\begin{enumerate}[*] 
\item Hai mặt phẳng chứa hai đáy là hai mặt phẳng song song
$mp\left(ABC\right)\parallel    mp\left(A'B'C'\right)$.
\item Hai đáy là hai đa giác có cùng số cạnh.
\item Các cạnh bên vuông góc với hai mặt phẳng đáy. Độ dài của một cạnh bên được gọi là chiều cao: $ AA'\perp mp(A'B'C')$.
\item Các mặt bên vuông góc với hai mặt phẳng đáy.
\end{enumerate}
	}{

{\it Đặc biệt:}\\
Hình lăng trụ đứng có đáy là hình bình hành được gọi là hình hộp đứng.\\
Hình lăng trụ đứng có đáy là hình chữ nhật là hình hợp chữ nhật.
\subsubsection{Diện tích xung quanh của hình lăng trụ đứng}
\begin{enumerate}[*]
\item $S_{\text{xq}}=2\cdot p\cdot h$ ($p$ là nửa chu vi đáy, h là chiều cao).
\item $S_{\text{tp}}=S_{xq}+2\cdot S$ ($S$ là diện tích đáy).
\end{enumerate}
\subsubsection{ Thể tích của hình lăng trụ đứng}
$V=S\cdot h$ ($S$ là diện tích đáy, $h$ là chiều cao)
