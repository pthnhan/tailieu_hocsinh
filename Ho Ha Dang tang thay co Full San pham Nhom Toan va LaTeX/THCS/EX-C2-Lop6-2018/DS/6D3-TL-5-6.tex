\section{Quy đồng mẫu nhiều phân số}
\section{So sánh phân số}
\subsection{Trọng tâm kiến thức}
\subsubsection{Muốn quy đồng mẫu nhiều phân số ta làm theo ba bước:}
	\begin{itemize}
		\item Bước 1: Tìm một bội chung của các mẫu (thường là BCNN) để làm mẫu số chung.
		\item Bước 2: Tìm thừa số phụ của mỗi mẫu (bằng cách chia mẫu chung cho từng mẫu).
		\item Bước 3: Nhân tử và mẫu của mỗi phân số với thừa số phụ tương ứng.
	\end{itemize}
	\subsubsection{Trong hai phân số có cùng mẫu dương, phân số nào có tử lớn hơn thì lớn hơn.}
	\subsubsection{Muốn so sánh hai phân số không cùng mẫu, ta viết chúng dưới dạng hai phân số có cùng mẫu dương rồi so sánh tử số với nhau. Phân số nào có tử lớn hơn thì lớn hơn.}
\subsection{Các dạng bài tập và phương pháp giải}	
\begin{dang}{Quy đồng mẫu các phân số cho trước}
\textbf{\textit{Phương pháp giải:}}
\begin{itemize}
	\item Viết các phân số dưới dạng phân số có mẫu dương.
	\item Rút gọn các phân số (nếu có thể).
	\item Áp dụng quy tắc quy đồng mẫu số gồm ba bước.
\end{itemize}
\end{dang}
\begin{vd}%[Phan Minh Tâm]%[EX-C2-L6-2018]%[6D3Y5]
Quy đồng mẫu số các phân số sau:
\begin{listEX}[2]
	\item $ \dfrac{7}{30} $; \ $ \dfrac{8}{45} $; \ $ \dfrac{-11}{90} $.
	\item $ \dfrac{-4}{5} $; \ $ \dfrac{1}{6} $; \ $ \dfrac{-9}{7} $.
\end{listEX}
\loigiai{
\begin{enumerate}
	\item Ta có $ \mathrm{BCNN}(30;45;90)=90 $.\\
	Các thừa số phụ: $ 3 $; $ 2 $; $ 1 $.\\
	Do đó
	\begin{itemize}
		\item $ \dfrac{7}{30}=\dfrac{7\cdot 3}{30\cdot 3}=\dfrac{21}{90} $.
		\item $ \dfrac{8}{45}=\dfrac{8\cdot 2}{45\cdot 2}=\dfrac{16}{90} $.
		\item $ \dfrac{-11}{90} $ giữ nguyên.
	\end{itemize}
	\item Ta có $ \mathrm{BCNN}(5;6;7)=210 $.\\
	Các thừa số phụ: $ 42 $, $ 35 $; $ 30 $.\\
	Do đó
	\begin{itemize}
		\item $ \dfrac{-4}{5}=\dfrac{-4\cdot 42}{5\cdot 42}=\dfrac{-168}{210} $.
		\item $ \dfrac{1}{6}=\dfrac{1\cdot 35}{6\cdot 35}=\dfrac{35}{210} $.
		\item $ \dfrac{-9}{7}=\dfrac{-9\cdot 30}{7\cdot 30}=\dfrac{-270}{210} $.
	\end{itemize}
\end{enumerate}	
}
\end{vd}
\begin{vd}%[Phan Minh Tâm]%[EX-C2-L6-2018]%[6D3B5]
Quy đồng mẫu các phân số sau:
\begin{multicols}{2}
	\begin{enumerate}
		\item $ \dfrac{5}{14} $; \ $ \dfrac{13}{21} $; \ $ \dfrac{20}{49} $.
		\item $ \dfrac{-7}{-24} $; \ $ \dfrac{11}{12} $; \ $ \dfrac{23}{-36} $.
	\end{enumerate}	
\end{multicols}
\loigiai{
\begin{enumerate}
	\item $ \mathrm{BCNN}(14;21;49)=294 $.\\
	Các thừa số phụ: $ 21 $; $ 14 $; $ 6 $.\\
	Do đó
	\begin{itemize}
		\item $ \dfrac{5}{14}=\dfrac{5\cdot 21}{14\cdot 21}=\dfrac{105}{294} $.
		\item $ \dfrac{13}{21}=\dfrac{13\cdot 14}{21\cdot 14}=\dfrac{182}{294} $.
		\item $ \dfrac{20}{49}=\dfrac{20\cdot 6}{49\cdot 6}=\dfrac{120}{294} $.
	\end{itemize}
	\item Viết các phân số đã cho dưới dạng mẫu dương, ta được $$ \dfrac{7}{24};\ \dfrac{11}{12},\ \dfrac{-23}{36}. $$
	$ \mathrm{BCNN}(24; 12; 36)=72 $.\\
	Các thừa số phụ: $ 3 $; $ 6 $; $ 2 $.\\
	Do đó
	\begin{itemize}
		\item $ \dfrac{7}{24}=\dfrac{7\cdot 3}{24\cdot 3}=\dfrac{21}{72} $.
		\item $ \dfrac{11}{12}=\dfrac{11\cdot 6}{12\cdot 6}=\dfrac{66}{72} $.
		\item $ \dfrac{-23}{36}=\dfrac{-23\cdot 2}{36\cdot 2}=\dfrac{-46}{72} $.
	\end{itemize}
\end{enumerate}
}
\end{vd}
\begin{vd}%[Phan Minh Tâm]%[EX-C2-L6-2018]%[6D3K5]
	Quy đồng mẫu số các phân số 
	\begin{eqnarray*}
		\dfrac{51}{90};\ \dfrac{22}{-75};\ -5.
	\end{eqnarray*}
\loigiai{
	Rút gọn phân số chưa tối giản và viết lại các phân số đã cho dưới dạng mẫu dương, ta được $$ \dfrac{17}{30};\ \dfrac{-22}{75};\ \dfrac{-5}{1}. $$
	$ \mathrm{BCNN}(30;75)=150 $.\\
	Các thừa số phụ: $ 5 $; $ 2 $; $ 150 $.\\
	Do đó
	\begin{itemize}
		\item $ \dfrac{17}{30}=\dfrac{17\cdot 5}{30\cdot 5}=\dfrac{85}{150} $.
		\item $ \dfrac{-22}{75}=\dfrac{-22\cdot 2}{75\cdot 2}=\dfrac{-44}{150} $.
		\item $ \dfrac{-5}{1}=\dfrac{-5\cdot 150}{1\cdot 150}=\dfrac{-750}{150} $.
	\end{itemize}
}
\end{vd}
\begin{dang}{So sánh các phân số}
	\textbf{\textit{Phương pháp giải:}}
	\begin{center}
		Quy đồng mẫu số các phân số rồi so sánh các tử số.
	\end{center}
\end{dang}
\setcounter{vd}{0}
\begin{vd}%[Phan Minh Tâm]%[EX-C2-L6-2018]%[6D3Y6]
	So sánh các phân số sau
	\begin{multicols}{2}
		\begin{enumerate}
			\item $ \dfrac{5}{8} $ và $ \dfrac{7}{10} $.
			\item $ \dfrac{-13}{16} $ và $ \dfrac{-11}{12} $.
		\end{enumerate}
			
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item Quy đồng mẫu số hai phân số ta được $$ \dfrac{5}{8}=\dfrac{25}{40};\ \dfrac{7}{10}=\dfrac{28}{40}. $$
			Vì $ \dfrac{25}{40}<\dfrac{28}{40} $ nên $ \dfrac{5}{8}<\dfrac{7}{10} $.
			\item Quy đồng mẫu số hai phân số ta được $$ \dfrac{-13}{16}=\dfrac{-39}{48};\ \dfrac{-11}{12}=\dfrac{-44}{48}. $$
			Vì $ \dfrac{-39}{48}>\dfrac{-44}{48} $ nên $ \dfrac{-13}{16}>\dfrac{-11}{12} $.
		\end{enumerate}
	}
\end{vd}
\begin{vd}%[Phan Minh Tâm]%[EX-C2-L6-2018]%[6D3B6]
	Sắp xếp các phân số sau theo thứ tự tăng dần
	\begin{eqnarray*}
		\dfrac{8}{15};\ \dfrac{2}{3};\ \dfrac{4}{9};\ \dfrac{3}{5}.
	\end{eqnarray*}
	\loigiai{
		Quy đồng mẫu các phân số đã cho ta được
		$$ \dfrac{24}{45};\ \dfrac{30}{45};\ \dfrac{20}{45};\ \dfrac{27}{45}. $$
		Sắp theo thứ tự tăng dần
		\begin{eqnarray*}
		\dfrac{20}{45}<\dfrac{24}{45}<\dfrac{27}{45}<\dfrac{30}{45}.	
		\end{eqnarray*}  
		Do đó $ \dfrac{4}{9}<\dfrac{8}{15}<\dfrac{3}{5}<\dfrac{2}{3} $.
	}
\end{vd}
\begin{vd}%[Phan Minh Tâm]%[EX-C2-L6-2018]%[6D3B6]
	Sắp xếp các phân số theo thứ tự giảm dần 
	\begin{eqnarray*}
		\dfrac{-7}{12};\ \dfrac{-5}{8};\ \dfrac{-3}{5}.
	\end{eqnarray*}
		\loigiai{
		Quy đồng mẫu các phân số đã cho ta được
	\begin{eqnarray*}
		\dfrac{-70}{120};\ \dfrac{-75}{120};\ \dfrac{-72}{120}. 
	\end{eqnarray*}	
		Sắp xếp theo thứ tự giảm dần
		$$ \dfrac{-70}{120}>\dfrac{-72}{120}>\dfrac{-75}{120}. $$
		Do đó $ \dfrac{-7}{12}>\dfrac{-3}{5}>\dfrac{-5}{8} $.
	}
\end{vd}
\begin{vd}%[Phan Minh Tâm]%[EX-C2-L6-2018]%[6D3K6]
	Tìm các phân số lớn hơn $ \dfrac{1}{4} $ nhưng nhỏ hơn $ \dfrac{3}{5} $ và có mẫu là $ 20 $.
	\loigiai{
		Gọi phân số tối giản cần tìm là $ \dfrac{x}{20} $ với $ x\in\mathbb{Z} $.\\
		Ta có $ \dfrac{1}{4}<\dfrac{x}{20}<\dfrac{3}{5} $.\\
		Quy đồng mẫu các phân số ta được
		$$ \dfrac{5}{20}<\dfrac{x}{20}<\dfrac{12}{20} $$
		Suy ra $ x\in\lbrace 6; 7; 8; 9; 10; 11 \rbrace $.
		Vậy ta tìm được $ 6 $ phân số là $ \dfrac{6}{20};\ \dfrac{7}{20};\ \dfrac{8}{20};\ \dfrac{9}{20};\ \dfrac{10}{20};\ \dfrac{11}{20} $.
	}
\end{vd}
\begin{vd}%[Phan Minh Tâm]%[EX-C2-L6-2018]%[6D3K6]
	Cho $ E=\lbrace 4; 7; 9 \rbrace $, $ F=\lbrace 5; 8; 11 \rbrace $. Tìm $ x\in E $ và $ y\in F $ sao cho phân số $ \dfrac{x}{y} $ có giá trị nhỏ nhất.
	\loigiai{
		Phân số với tử và mẫu dương có giá trị nhỏ nhất khi tử số nhỏ nhất và mẫu số lớn nhất. Tử số nhỏ nhất là $ 4 $, mẫu số lớn nhất là $ 11 $. Vậy khi $ x=4 $; $ y=11 $ thì phân số $ \dfrac{x}{y}=\dfrac{4}{11} $ có giá trị nhỏ nhất.
	}
\end{vd}
\begin{dang}{So sánh hai phân số mà không quy đồng mẫu, không quy đồng tử}
	\textbf{\textit{Phương pháp giải:}}
	\begin{center}
		Dùng một phân số thứ ba làm trung gian.
	\end{center}
\end{dang}
\setcounter{vd}{0}
\begin{vd}%[Phan Minh Tâm]%[EX-C2-L6-2018]%[6D3B6]
	So sánh hai phân số $ \dfrac{-7}{15} $ và $ \dfrac{-341}{-562} $.
	\loigiai{
		Ta có
		\begin{itemize}
			\item $ \dfrac{-7}{15}<0 $ (vì tử và mẫu khác dấu);
			\item $ \dfrac{-341}{-562}>0 $ (vì tử và mẫu cùng dấu).
		\end{itemize}
	Do đó $ \dfrac{-7}{15}<\dfrac{-341}{-562}. $
	}
\end{vd}
\begin{vd}%[Phan Minh Tâm]%[EX-C2-L6-2018]%[6D3B6]
	So sánh các phân số sau
	\begin{multicols}{2}
		\begin{enumerate}
			\item $ \dfrac{147}{145} $ và $ \dfrac{718}{719} $;
			\item $ \dfrac{16}{51} $ và $ \dfrac{25}{72} $.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			\begin{itemize}
				\item $ \dfrac{147}{145}>1 $ (vì tử lớn hơn mẫu và tử, mẫu đều là số dương);
				\item $ \dfrac{718}{719}<1 $ (vì mẫu lớn hơn tử và tử, mẫu đều là số dương).
			\end{itemize}
			Do đó $ \dfrac{147}{145}>\dfrac{718}{719} $.
			\item Ta có
			\begin{itemize}
				\item $ \dfrac{16}{51}<\dfrac{17}{51}=\dfrac{1}{3} $;
				\item $ \dfrac{25}{72}>\dfrac{24}{72}=\dfrac{1}{3} $.
			\end{itemize}
			Do đó $ \dfrac{16}{51}<\dfrac{25}{72} $.
		\end{enumerate}
	}
\end{vd}
\begin{vd}%[Phan Minh Tâm]%[EX-C2-L6-2018]%[6D3K6]
	So sánh các phân số sau bằng cách hợp lý nhất
	\begin{multicols}{2}
		\begin{enumerate}
			\item $ \dfrac{29}{33} $; $ \dfrac{22}{37} $; $ \dfrac{29}{37} $.
			\item $ \dfrac{163}{257} $; $ \dfrac{163}{221} $; $ \dfrac{149}{257} $.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item Phân số $ \dfrac{29}{37} $ có tử bằng tử của phân số $ \dfrac{29}{33} $ và có mẫu bằng với mẫu của phân số $ \dfrac{22}{37} $ nên ta dùng phân số $ \dfrac{29}{37} $ làm trung gian để so sánh.\\
			Ta có $ \dfrac{29}{33}>\dfrac{29}{37} $ và $ \dfrac{29}{37}>\dfrac{22}{37} $ nên $ \dfrac{29}{33}>\dfrac{29}{37}>\dfrac{22}{37} $.
			\item Phân số $ \dfrac{163}{257} $ có tử bằng tử của phân số $ \dfrac{163}{221} $ và có mẫu bằng mẫu của phân số $ \dfrac{149}{257} $ nên ta dùng phân số $ \dfrac{163}{257} $ làm trung gian để so sánh.\\
			Ta có $ \dfrac{163}{221}>\dfrac{163}{257} $ và $ \dfrac{163}{257}>\dfrac{149}{257} $ nên $ \dfrac{163}{221}>\dfrac{163}{257}>\dfrac{149}{257} $.
		\end{enumerate}
	}
\end{vd}
\begin{vd}%[Phan Minh Tâm]%[EX-C2-L6-2018]%[6D3B6]
	So sánh các phân số
	\begin{multicols}{2}
		\begin{enumerate}
			\item $ \dfrac{25}{26} $ và $ \dfrac{89}{90} $;
			\item $ \dfrac{111}{115} $ và $ \dfrac{67}{71} $.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item Ta có $ \dfrac{25}{26}=1-\dfrac{1}{26} $; $ \dfrac{89}{90}=1-\dfrac{1}{90} $.\\
			Mà $ \dfrac{1}{26}>\dfrac{1}{90} $ nên $ \dfrac{25}{26}<\dfrac{89}{90} $.
			\item Ta có $ \dfrac{111}{115}=1-\dfrac{4}{115} $; $ \dfrac{67}{71}=1-\dfrac{4}{71} $.\\ Mà $ \dfrac{4}{115}<\dfrac{4}{71} $ nên $ \dfrac{111}{115}>\dfrac{67}{71} $.
		\end{enumerate}
	}
\end{vd}

\begin{bt}%[Phan Minh Tâm]%[EX-C2-L6-2018]%[6D3K6]
	Tìm các phân số có mẫu số là $ 150 $ mà lớn hơn $ \dfrac{7}{15} $ nhưng nhỏ hơn $ \dfrac{12}{25} $.
\loigiai{
	Gọi phân số phải tìm là $ \dfrac{x}{150} $ $ (x\in\mathbb{Z}) $.\\
	Ta có $ \dfrac{7}{15}<\dfrac{x}{150}<\dfrac{12}{25} $.\\
	Quy đồng mẫu ta được $ \dfrac{70}{150}<\dfrac{x}{150}<\dfrac{72}{150} $.\\
	Suy ra $ 70<x<72 $.\\
	Mặt khác $ x\in\mathbb{Z} $ nên $ x=71 $.\\
	Phân số phải tìm là $ \dfrac{71}{150} $.
}
\end{bt}
