\begin{ex}%[[Đề thi Toán 9 HK2 - 2017-2018, Thủ Đức, HCM]%[Phan Hoàng Anh]%[9D4B4]
Khu vườn nhà kính trồng rau sạch hình chữ nhật có diện tích $6600$ m$^2$ với chiều dài hơn chiều rộng $50$ m. Để tìm kích thước của khu vườn, em hãy giải phương trình $x^2+50x-6600=0$ với ẩn $x$ là độ dài chiều rộng của khu vườn.
	\loigiai{Xét phương trình $x^2+50x-6600=0$.\\
	Có $\Delta'=25^2+6600=7225$, $\sqrt{\Delta'}=85$.\\
Phương trình trên có hai nghiệm là $\left[\begin{aligned}
&x=\dfrac{-25+85}{1}=60\\&x=\dfrac{-25-85}{1}=-110.
\end{aligned}\right.$\\
Vậy chiều rộng của khu vườn là $x=60$ m, chiều dài của khu vườn là $y=x+50=110$ m.}
\end{ex}
\begin{ex}%[[Đề thi Toán 9 HK2 - 2017-2018, Thủ Đức, HCM]%[Phan Hoàng Anh]%[9D3B5]
Tham quan trải nghiệm một trang trại chăn nuôi, bạn Minh hỏi anh công nhân số con gà và số con bò trong trang trại đang nuôi thì được anh công nhân cười và nói rằng: ``Tất cả có $1200$ con và $2700$ chân''. bạn hãy tính giúp Minh có bao nhiêu con gà, bao nhiêu con bò nhé?
	\loigiai{Gọi $x$ là số con gà, $y$ là số con bò trong trang trại, suy ra $x+y=1200$.\\
		Ta có số chân của gà là $2x$, số chân của bò là $4y$, nên $2x+4y=2700$.\\
	Ta có $\begin{cases}
	x+y=1200\\
	2x+4y=2700
	\end{cases}\Leftrightarrow\begin{cases}
	2x+2y=2400\\
	2x+4y=2700
	\end{cases}\Leftrightarrow\begin{cases}
	x+y=1200\\
	2y=300
	\end{cases}\Leftrightarrow\begin{cases}
	y=150\\
	x=1200-y=1050.
	\end{cases}$\\
Vậy ta có $1050$ con gà và $150$ con bò trong trang trại.}
\end{ex}

\begin{ex}%[[Đề thi Toán 9 HK2 - 2017-2018, Thủ Đức, HCM]%[Phan Hoàng Anh]%[9D4B4]%[9D4B6]%[9D4B6]
Cho phương trình: $x^2-(m-1)x+2m-6=0$, $x$ là ẩn số.
	\begin{enumerate}
		\item Chứng tỏ phương trình luôn có ngiệm $x_1$, $x_2$ với mọi $m$.
		\item Tính tổng và tích của hai nghiệm theo $m$.
		\item Không giải phương trình, tính giá trị của biểu thức $A=\dfrac{x_1}{x_2}+\dfrac{x_2}{x_1}$, khi $m=2$.
	\end{enumerate}
	\loigiai{\begin{enumerate}
			\item Có $\Delta=(m-1)^2-4\cdot(2m-6)=m^2-10m+26=(m-5)^2+1>0,\forall m\in\mathbb{R}$.\\
			Vì thế nên phương trình luôn có nghiệm $x_1$, $x_2$ với mọi giá trị của $m$.
			\item Theo định lý Vi-et ta có $\begin{cases}
			x_1+x_2=-\dfrac{b}{a}=1-m\\
			x_1x_2=\dfrac{c}{a}=2m-6.
			\end{cases}$\\
			\item Ta có 
			\begin{eqnarray*}
			A=\dfrac{x_1}{x_2}+\dfrac{x_2}{x_1}=\dfrac{x_1^2+x_2^2}{x_1x_2}=\dfrac{\left(x_1+x_2\right)^2-2x_1x_2}{x_1x_2}=\dfrac{(1-m)^2-2(2m-6)}{2m-6}=-\dfrac{5}{2}.
			\end{eqnarray*}
	\end{enumerate}}
\end{ex}
\begin{ex}%[[Đề thi Toán 9 HK2 - 2017-2018, Thủ Đức, HCM]%[Phan Hoàng Anh]%[9H3B10]
\immini{Một mảnh vườn bao gồm hình chữ nhật có độ dài cạnh là $40$ m, $20$ m và một hình vuông có độ dài cạnh là $15$ m. Bên trong mảnh vườn người ta đào một cái giếng hình tròn có bán kính $4$ m (tham khảo hình vẽ bên). Tính diện tích phần mảnh vườn còn lại sau khi đào giếng với $\pi=3{,}14$.}
{\hspace{0.5cm}\begin{tikzpicture}[scale=0.8]
\draw (0,0)--(6,0)--(6,3)--(0,3)--(0,0) (6,0)--(8,0)--(8,2)--(6,2);
\draw [fill=gray!60] (6,1) circle (0.6);
\node at (3,0.3) {$40$m};
\node at (0.6,1.5) {$20$m};
\node at (7,2.3) {$15$m};
\end{tikzpicture}}
	\loigiai{\begin{itemize}
			\item Diện tích của mảnh vườn hình chữ nhật là $S_1=20\times40=800$m$^2$.
			\item Diện tích của mảnh vườn hình vuông là $S_2=15^2=225$ m$^2$.
			\item Diện tích của cái giếng là $S_3=\pi\times R^2=3{,}14\times4^2=50{,}24$ m$^2$.
	\end{itemize}
Vậy diện tích mảnh vườn còn lại sau khi đào giếng là $$S=S_1+S_2-S_3=800+225-50{,}24=974{,}76\text{ m}^2.$$}
\end{ex}

\begin{ex}%[[Đề thi Toán 9 HK2 - 2017-2018, Thủ Đức, HCM]%[Phan Hoàng Anh]%[9H3B6]
\immini{Đội tuyển U23 đã ghi lại dấu ấn vẻ vang cho nền bóng đá Việt Nam, chúng ta đã được thưởng thức bao nhiêu quả đá phạt tuyệt vời của các cầu thủ. Vậy các em có biết ``Góc sút'' của quả phảt đền $11$m là bao nhiêu độ không? Em hãy tính góc đó, biết rằng chiều rộng cầu môn là $7{,}32$m và có bao nhiêu điểm trên sân có cùng góc sút với điểm sút quả phạt đền? (làm tròn đến chữ số thập phân thứ $3$)}
{\hspace{1cm}
	\begin{tikzpicture}
	\tkzDefPoints{0/0/O, 0/3/C, 0/-3/M}
	\tkzDrawCircle(O,C)
	\coordinate (A) at ($(O)+({3*cos(140)},{3*sin(140)})$);	
	\coordinate (B) at ($(O)+({3*cos(40)},{3*sin(40)})$);
	\tkzInterLL(A,B)(O,C)	\tkzGetPoint{H}
	\tkzDrawSegments(A,B B,M A,M M,H)
	\tkzDrawPoints(A,B,M,H,O)
	\tkzLabelPoints[below right](O,H)
	\tkzLabelPoints[above right](B)
	\tkzLabelPoints[below](M)
	\tkzLabelPoints[above left](A)
	\tkzMarkAngle[label=$2\alpha$,fill=blue!50,opacity=0.5,size=0.6](B,M,A)
	\tkzLabelSegment[pos=0.5](A,B){$7{,}32$ m}
	\tkzLabelSegment[pos=0.5,rotate=90](M,H){$11$ m}
	\tkzMarkRightAngle(A,H,M)
	\end{tikzpicture}}
\loigiai{Ta có $\triangle MAB$ cân tại $M$, suy ra $H$ là trung điểm của $AB$, nên $AH=\dfrac{AB}{2}=3{,}66$.\\
Ta có $MH\perp AB$ nên $\triangle AMH$ vuông tại $H$, suy ra $\tan\alpha=\dfrac{AH}{MH}=\dfrac{3{,}66}{11}=\dfrac{183}{550}\Rightarrow\alpha\approx18{,}404^\circ$.\\
Vậy ``Góc sút'' của quả phạt đền $11$m là $2\alpha\approx36{,}807^\circ$.\\
Có vô số vị trí có cùng ``Góc sút'' với quả phạt đền $11$m, đó là cung lớn chứa góc $2\alpha$ nhìn cạnh $AB$.}
\end{ex}
\begin{ex}%[[Đề thi Toán 9 HK2 - 2017-2018, Thủ Đức, HCM]%[Phan Hoàng Anh]%[9H3B7]%[9H3B4]%[9H3B4]
Cho điểm $A$ nằm ngoài đường tròn $(O)$. Qua $A$, vẽ hai tiếp tuyến $AB$ và $AC$ đến đường tròn ($B$, $C$ là tiếp điểm) và cát tuyến $ADE$ ($D$ nằm giữa $A$ và $E$).
\begin{enumerate}
	\item Chứng minh tứ giác $ABOC$ nội tiếp, xác định tâm đường tròn ngoại tiếp tứ giác này.
	\item Chứng minh $AB^2=AD\cdot AE$.
	\item Trường hợp cát tuyến $ADE$ đi qua tâm $O$. Chứng minh $D$ là tâm đường tròn nội tiếp tam giác $ABC$.
\end{enumerate}
\loigiai{\begin{center}
		\begin{tikzpicture}
		\tkzDefPoints{0/0/O, 0/2/M, 4/1/A}
		\tkzDrawCircle(O,M)
		\tkzTangent[from=A](O,M)	\tkzGetPoints{B}{C}
		\coordinate (D) at ($(O)+({2*cos(30)},{2*sin(30)})$);
		\tkzInterLC(A,D)(O,M)	\tkzGetFirstPoint{E}
		\tkzDrawSegments(A,B A,C A,E B,O C,O A,O B,D B,E B,C)
		\tkzDrawPoints(A,B,C,D,O,E)
		\tkzLabelPoints[below](C)
		\tkzLabelPoints[above right](A)
		\tkzLabelPoints[below left](O,D)
		\tkzLabelPoints[above](B)
		\tkzLabelPoints[above left](E)
		\tkzLabelPoints[below left](D)
		\tkzMarkRightAngle(A,B,O)
		\tkzMarkRightAngle(O,C,A)
		\tkzMarkAngle[fill=blue!50,opacity=0.5,size=0.6](D,B,A)
		\tkzMarkAngle[fill=yellow!50,opacity=0.5,size=0.6](A,E,B)
		\end{tikzpicture}
	\end{center}
	\begin{enumerate}
		\item Ta có $\begin{cases}
		\widehat{ABO}=90^\circ\text{ vì }AB\text{ là tiếp tuyến của đường tròn }(O)\\
		\widehat{ACO}=90^\circ\text{ vì }AC\text{ là tiếp tuyến của đường tròn }(O)
		\end{cases}$\\
		Từ đó suy ra $\widehat{ABO}+\widehat{ACO}=180^\circ$ nên tứ giác $ABOC$ là tứ giác nội tiếp đường tròn đường kính $AO$.\\
		Tâm của đường tròn là trung điểm đoạn thẳng $AO$.
		\item Xét đường tròn $(O)$, ta có:
		\begin{itemize}
			\item $\widehat{ABD}=\dfrac{1}{2}\text{sđ}\wideparen{BD}$ (góc tạo bởi tiếp tuyến và dây cung)
			\item $\widehat{BEA}=\dfrac{1}{2}\text{sđ}\wideparen{BD}$ (góc nội tiếp)
		\end{itemize}
		Từ đó suy ra $\widehat{ABD}=\widehat{BEA}$, mà hai tam giác $BAD$ và $EBA$ có góc $\widehat{BAE}$ chung, nên suy ra $\triangle BAD\backsim\triangle EAB$.\\
		Vậy ta suy ra $\dfrac{AB}{AE}=\dfrac{AD}{AB}\Rightarrow AB^2=AD\cdot AE$.\\
		\item \immini{Ta có $AB$ và $AC$ là tiếp tuyến của đường tròn $(O)$ được vẽ từ $A$ nên $AO$ là phân giác của góc $\widehat{BAC}$.\hfill$(1)$\\
		Mặt khác, khi $ADE$ đi qua $O$ thì ta có $D$ là trung điểm của $\wideparen{BC}$ nên $BD$ là phân giác của $\widehat{ABC}$.\hfill$(2)$\\
		Từ $(1)$ và $(2)$ ta suy ra $D$ là tâm đường tròn nội tiếp tam giác $ABC$.}
		{\begin{tikzpicture}
			\tkzDefPoints{0/0/O, 0/2/M, 4/1/A}
			\tkzDrawCircle(O,M)
			\tkzTangent[from=A](O,M)	\tkzGetPoints{B}{C}
			\tkzInterLC(A,O)(O,M)	\tkzGetPoints{D}{E}
			\tkzDrawSegments(A,B A,C A,E B,O C,O A,O B,D B,E B,C)
			\tkzDrawPoints(A,B,C,D,O,E)
			\tkzLabelPoints[below](C)
			\tkzLabelPoints[above right](A,D)
			\tkzLabelPoints[above](B)
			\tkzLabelPoints[above left](E,O)
			\tkzMarkRightAngle(A,B,O)
			\tkzMarkRightAngle(O,C,A)
			\tkzMarkAngle[fill=blue!50,opacity=0.5,size=0.6](D,B,A)
			\tkzMarkAngle[fill=yellow!50,opacity=0.5,size=0.6](C,B,D)
			\end{tikzpicture}}
\end{enumerate}}
\end{ex}
