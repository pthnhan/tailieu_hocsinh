\begin{ex}%[Đề thi vào 10, Sở giáo dục Cần Thơ, 2017]%[9D4Y4]%[9D3Y4]%[9D4K7]
Giải các phương trình và hệ phương trình sau trên tập số thực.\\
\begin{tabular}{m{0.3\textwidth} m{0.3\textwidth} m{0.3\textwidth}}
a) $2x^2-9x+10=0$ ;&
b) $\heva{& 3x-2y=9\\ &x-3y=10}$ ;&
c) $(x-1)^4-8(x-1)^2-9=0$.
\end{tabular}
\loigiai{
\begin{enumerate}[a)]
\item Ta có $\Delta= (-9)^2-4\cdot 2\cdot 10=81-80=1>0 \Rightarrow \sqrt{\Delta}=1$. Phương trình có hai nghiệm phân biệt $x_1=\dfrac{9-1}{2\cdot 2}=2$ và $x_2=\dfrac{9+1}{2\cdot 2}=\dfrac{5}{2}$.
\item \ \\\begin{tabular}{l|l}
\begin{minipage}{8cm}
\textit{Cách 1. Phương pháp thế:}\\
$\begin{aligned}
&\heva{&3x-2y=9\\ &x-3y=10} \Leftrightarrow \heva{&x= 3y+10\\ &3(3y+10)-2y=9}\\
\Leftrightarrow &\heva{& x= 3y+10\\ &7y=-21} \Leftrightarrow \heva{&x=1\\ &y=-3}.
\end{aligned}$
\end{minipage}
& \begin{minipage}{8cm}
\textit{Cách 2. Phương pháp cộng đại số:}
$\begin{aligned}
&\heva{&3x-2y=9\\ &x-3y=10} \Leftrightarrow \heva{&3x-2y=9\\ &3x-9y=30}\\
\Leftrightarrow &\heva{& 3x-2y=9\\ &7y=-21} \Leftrightarrow\heva{&x=1\\ &y=-3}.
\end{aligned}$
\end{minipage}
\end{tabular}
\item Đặt $t=(x-1)^2\ge 0$. Phương trình trở thành
$\heva{& t\ge 0\\ &t^2-8t-9=0}$ hay $t=9$. Do đó $(x-1)^2=9$ hay $x=-2$ hoặc $x=4$.\\
Vậy tập nghiệm của phương trình đã cho là $S=\{-2;4\}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Cần Thơ, 2017]%[9D4B2]%[9D4B4]
Trong mặt phẳng tọa độ $Oxy$ cho parabol $(P): y=\dfrac{1}{2}x^2$ và đường thẳng $d:y=\dfrac{1}{4}x+\dfrac{3}{2}$.
\begin{enumerate}[a)]
\item Vẽ đồ thị $(P)$;
\item Gọi $A(x_1;y_1)$, $B(x_2;y_2)$ lần lượt là các giao điểm của $(P)$ và $d$. Tính giá trị biểu thức $$T=\dfrac{x_1+x_2}{y_1+y_2}.$$
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item Ta có bảng sau một số giá trị của $x$ và $y$ tương ứng
\begin{center}
{\tabcolsep=0.75cm
\begin{tabular}{c|c|c|c|c|c}
$x$& $-2$& $-1$ & $0$& $1$& $2$
\cr\hline
$y$& $2$& $\dfrac{1}{2}$& $0$& $\dfrac{1}{2}$& $2$
\end{tabular}}
\end{center}
Đồ thị parabol $(P)$
\begin{center}
\begin{tikzpicture}[>=stealth, scale=0.75]
\draw[very thin, color=gray] (-4.2,-1.2) grid (4.2,7.2);

\draw[->, line width=1.2pt] (-4.3,0)--(4.3,0) node[above]{$x$};
\draw[->, line width=1.2pt] (0,-1.2)--(0,7.3) node[right]{$y$};
\draw [smooth, color=blue, line width=1.2pt, domain=-3.8:3.8] plot(\x,{0.5*(\x)^2});
\draw (0,0.5) node[right]{\scriptsize $\dfrac{1}{2}$};
\draw[dashed] (-1,0.5)--(1,0.5);
\fill[color=black] (0,0) circle (2pt);
\fill[color=black] (1,0.5) circle (2pt);

\end{tikzpicture}
\end{center}
\item Hoành độ giao điểm của parabol $(P)$ và đường thẳng $d$ là nghiệm của phương trình $\dfrac{1}{2}x^2=\dfrac{1}{4}x+\dfrac{3}{2}$. Phương trình trên có hai nghiệm $x_1=2$ và $x_2=-\dfrac{3}{2}$. Do đó 
$$T=\dfrac{x_1+x_2}{y_1+y_2}=\dfrac{x_1+x_2}{\dfrac{1}{4}(x_1+x_2)+3}=\dfrac{4}{25}.$$
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Cần Thơ, 2017]%[9D1K8]
Cho biểu thức $P=\left(1+ \dfrac{1}{\sqrt{x}}\right) \left(\dfrac{1}{\sqrt{x}+1} +\dfrac{1}{\sqrt{x}-1}-\dfrac{2}{x-1}\right)(x>0, x\ne 1)$. Rút gọn biểu thức $P$ và tìm các giá trị của $x$ để $P>1$.
\loigiai{
Với $x>0$ và $x\ne 1$ ta có
\begin{align*}
P&=\left(1+ \dfrac{1}{\sqrt{x}}\right) \left(\dfrac{1}{\sqrt{x}+1} +\dfrac{1}{\sqrt{x}-1}-\dfrac{2}{x-1}\right)\\
&=\dfrac{\sqrt{x}+1}{\sqrt{x}} \cdot \dfrac{\left(\sqrt{x}-1\right)+\left(\sqrt{x}+1\right)-2}{\left(\sqrt{x}+1\right)\left(\sqrt{x}-1\right)}\\
&=\dfrac{\sqrt{x}+1}{\sqrt{x}} \cdot \dfrac{2\left(\sqrt{x}-1\right)}{\left(\sqrt{x}+1\right)\left(\sqrt{x}-1\right)}\\
&=\dfrac{2}{\sqrt{x}}.
\end{align*}
Khi đó $P>1 \Leftrightarrow \dfrac{2}{\sqrt{x}}>1 \Leftrightarrow \sqrt{x}<2 \Leftrightarrow x<4$.\\
Kết hợp với điều kiện ta được $P>1$ khi và chỉ khi $0<x<4$ và $x\ne 1$.
}
\end{ex}
 
 
\begin{ex}%[Đề thi vào 10, Sở giáo dục Cần Thơ, 2017]%[9D3K5]
Để chuẩn bị tham gia Hội khỏe Phù Đổng cấp trường, thầy Thành là giáo viên chủ nhiệm lớp 9A tổ chức cho học sinh trong lớp thi đấu môn bóng bàn ở nội dung đánh đôi nam nữ (một nam kết hợp một nữ). Thầy Thành chọn $\dfrac{1}{2}$ số học sinh nam kết hợp với $\dfrac{5}{8}$ số học sinh nữ của lớp để lập thành các cặp thi đấu. Sau khi đã chọn được số học sinh tham gia thi đấu thì lớp 9A còn lại $16$ học sinh làm cổ động viên. Hỏi lớp 9A có tất cả bao nhiêu học sinh?
\loigiai{
Gọi số học sinh nam và số học sinh nữ của lớp 9A lần lượt là $x$ và $y$.\\
Thầy Thành chọn $\dfrac{1}{2}$ số học sinh nam và $\dfrac{5}{8}$ số học sinh nữ tạo thành các cặp thi đấu nên $\dfrac{x}{2}=\dfrac{5y}{8}$ hay $x=\dfrac{5y}{4}$.\\
Tổng số học sinh của lớp 9A là $x+y$, số các bạn tham gia thi đấu là $\dfrac{x}{2}+\dfrac{5y}{8}=x$. Do số các bạn ở lại làm cổ động viên là $16$ nên $x+y=x+16$ hay $y=16$. Suy ra $x=20$.\\
Vậy lớp 9A có $36$ học sinh.
}
\end{ex} 
 
 \begin{ex}%[Đề thi vào 10, Sở giáo dục Cần Thơ, 2017]%[9D4K6]
 Cho phương trình $x^2-(m+4)x-2m^2+5m+3=0$ ($m$ là tham số). Tìm các giá trị nguyên của $m$ để phương trình đã cho có hai nghiệm phân biệt sao cho tích của hai nghiệm này bằng $-30$. Khi đó, tính tổng hai nghiệm của phương trình.
 \loigiai{
 Phương trình đã cho có hai nghiệm phân biệt có tích bằng $-30$ khi
 \begin{align*}
 \heva{&\Delta=(m+4)^2-4(-2m^2+5m+3)>0\\ &-2m^2+5m+3=-30} \Leftrightarrow \heva{& (3m-2)^2>0\\ & 2m^2-5m-33=0} \Leftrightarrow m=-3. 
 \end{align*}
 Khi đó tổng hai nghiệm của phương trình là $m+4=1$.
 }
 \end{ex}
 
 
\begin{ex}%[Đề thi vào 10, Sở giáo dục Cần Thơ, 2017]%[9H3B7]%[9H3K7]%[9H2G5]%[9H1G4]
Cho tam giác $ABC$ có ba góc nhọn. Đường tròn $(O)$ đường kính $BC$ cắt các cạnh $AB$, $AC$ lần lượt tại các điểm $D$ và $E$. Gọi $H$ là giao điểm hai đường thẳng $CD$ và $BE$.
\begin{enumerate}[a)]
\item Chứng minh rằng tứ giác $ADHE$ nội tiếp trong một đường tròn. Xác định tâm $I$ của đường tròn này;
\item Gọi $M$ là giao điểm của $AH$ và $BC$. Chứng minh $CM\cdot CB= CE\cdot CA$;
\item Chứng minh $ID$ là tiếp tuyến của đường tròn $(O)$;
\item Tính theo $R$ diện tích tam giác $ABC$, biết $\widehat{ABC}=45^\circ$, $\widehat{ACB}=60^\circ$ và $BC=2R$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzDefPoints{-3/0/B, 3/0/C, -1/4/A, 0/0/O}
\tkzDrawCircle (O,B)
\tkzInterLC (A,B)(O,B)\tkzGetPoints{B}{D}
\tkzInterLC(A,C)(O,B)\tkzGetPoints{E}{C}
\tkzMarkRightAngle(B,D,C)
\tkzMarkRightAngle(B,E,C)
\tkzInterLL(B,E)(C,D)\tkzGetPoint{H}
\tkzInterLL(A,H)(B,C)\tkzGetPoint{M}
\tkzLabelPoints[above](A, E)
\tkzLabelPoints[below](M, O)
\tkzLabelPoints[left](B,D)
\tkzLabelPoints[right](C,H)
\tkzMarkRightAngle(A,M,C)
\tkzDefMidPoint(A,H)\tkzGetPoint{I}
\tkzLabelPoints[above right](I)

\tkzDrawSegments(A,B B,C C,A B,E C,D A,M I,D O,D)
\end{tikzpicture}
\end{center}
\begin{enumerate}[a)]
\item Ta có $\widehat{BDC}=\widehat{BEC}=90^\circ$ (cùng chắn nửa đường tròn $(O)$). Suy ra $\widehat{ADH}=\widehat{AEH}=90^\circ$ hay tứ giác $ADHE$ nội tiếp đường tròn đường kính $AH$. Từ đó $I$ là trung điểm của $AH$.
\item Theo câu a) thì $BE$, $CD$ lần lượt là các đường cao hạ từ đỉnh $B$, $C$ của tam giác $ABC$. Do đó $H$ là trực tâm tam giác $ABC$ và $AM \perp BC$. Hai tam giác $CMA$ và $CEB$ lần lượt vuông tại $M$ và $E$ có chung góc đỉnh $C$ nên $\triangle CMA \backsim  \triangle CEB\ (\text{g.g})$. Suy ra $\dfrac{CM}{CE}=\dfrac{CA}{CB}$ hay $CM \cdot CB =CE \cdot CA$.
\item Tam giác $IAD$ cân tại $I$ do $IA=ID$ (câu a)) nên $\widehat{IDA}=\widehat{IAD}$. Tường tự $\widehat{ODC}=\widehat{OCD}$. Lại có $\widehat{IAD}=\widehat{OCD}$ (cùng phụ góc $\widehat{ABC}$) nên $\widehat{IDA}=\widehat{ODC}$. Suy ra $$\widehat{IDO}=\widehat{IDH}+\widehat{ODC}=\widehat{IDH}+\widehat{IDA}=90^\circ.$$
Vậy $ID$ là tiếp tuyến của đường tròn $(O)$.
\item Ta có 
\begin{align*}
\heva{&AM=BM\tan 45^\circ=BM\\ &AM=CM\tan 60^\circ= \sqrt{3}CM\\ &BM+CM=2R} \Rightarrow BM\left(1+\dfrac{\sqrt{3}}{3}\right)=2R \Rightarrow AM=BM=\left(3-\sqrt{3}\right)R.
\end{align*}
Vậy $S_{ABC}=\dfrac{1}{2}\cdot AM\cdot BC= \left(3-\sqrt{3}\right)R^2$.
\end{enumerate}
} 
\end{ex}
