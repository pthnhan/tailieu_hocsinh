		\dfrac{k\cdot \mathrm{C}_n^k}{\left(k+1\right)\left(k+2\right)}&=\left(\dfrac{2}{k+2}-\dfrac{1}{k+1}\right)\mathrm{C}_n^k=\dfrac{2\cdot \mathrm{C}_n^k}{k+2}-\dfrac{\mathrm{C}_n^k}{k+1}=2\cdot \dfrac{k+1}{k+2}\cdot \dfrac{\mathrm{C}_n^k}{k+1}-\dfrac{\mathrm{C}_n^k}{k+1}\\
		&=2\cdot \left(1-\dfrac{1}{k+2}\right)\cdot \dfrac{\mathrm{C}_n^k}{k+1}-\dfrac{\mathrm{C}_n^k}{k+1}=\dfrac{\mathrm{C}_n^k}{k+1}-2\cdot \dfrac{\mathrm{C}_{n+1}^{k+1}}{\left(k+2\right)\left(n+1\right)}\\
		&=\dfrac{\mathrm{C}_{n+1}^{k+1}}{n+1}-2\cdot \dfrac{\mathrm{C}_{n+2}^{k+2}}{\left(n+1\right)\left(n+2\right)}.
		\end{align*}
		Suy ra\\
		$S=\dfrac{1}{n+1}\left(-\mathrm{C}_{n+1}^2+\mathrm{C}_{n+1}^3+\cdots +{\left(-1\right)}^n\mathrm{C}_{n+1}^{n+1}\right)-\dfrac{2}{\left(n+1\right)\left(n+2\right)}\left(-\mathrm{C}_{n+2}^3+\mathrm{C}_{n+2}^4+\cdots +{\left(-1\right)}^n\mathrm{C}_{n+2}^{n+2}\right)$.\\
		Ta có
		\begin{align*}
		-\mathrm{C}_{n+1}^2+\cdots +\left(-1\right)^n\mathrm{C}_{n+1}^{n+1}&=\left(-\mathrm{C}_{n+1}^0+\mathrm{C}_{n+1}^1-\mathrm{C}_{n+1}^2+\cdots +{\left(-1\right)}^n\mathrm{C}_{n+1}^{n+1}\right)\text{+C}_{n+1}^0-\mathrm{C}_{n+1}^1\\
		&=-\left(1-1\right)^{n+1}+1-\left(n+1\right)=-n.\\
		-\mathrm{C}_{n+2}^3+\mathrm{C}_{n+2}^4+\cdots +\left(-1\right)^n\mathrm{C}_{n+2}^{n+2}&=\left(\mathrm{C}_{n+2}^0-\mathrm{C}_{n+2}^1+\cdots +{\left(-1\right)}^n\mathrm{C}_{n+2}^{n+2}\right)-\left(\mathrm{C}_{n+2}^0-\mathrm{C}_{n+2}^1+\mathrm{C}_{n+2}^2\right)\\
		&=\left(1-1\right)^{n+1}-\left(1-\left(n+2\right)+\dfrac{\left(n+1\right)\left(n+2\right)}{2}\right)=-\dfrac{n^2+n}{2}.
		\end{align*}
		Vậy ta suy ra
		$S=\dfrac{1}{n+1}\left(-n\right)+\dfrac{2}{\left(n+1\right)\left(n+2\right)}\cdot \dfrac{n^2+n}{2}=\dfrac{-n}{\left(n+1\right)\left(n+2\right)}$.
	}
