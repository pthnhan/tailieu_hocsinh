\begin{ex}%[Đề tuyển sinh 10, Điện Biên, 2016]%[9D4Y4]%[9D4B7]%[9D3Y3]
(2,0 điểm)\\
1. Giải các phương trình sau đây:
\begin{enumerate}
\item[a.] $x^2-6x+5=0$.
\item[b.] $2x^4-3x^2-2=0$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a.] Ta có $\Delta' =3^2-1.5=4>0$. Phương trình có hai nghiệm phân biệt $x_1=2, x_2=3$.
\item[b.] Đặt $t=x^2$ điều kiện $t\geq 0$, phương trình đã cho trở thành: $2t^2-3t-2=0\Leftrightarrow\hoac{&t=2\\&t=-\dfrac{1}{2}\text{ (loại)}}$.\\
Với $t=2$ thì $x^2=2\Leftrightarrow x=\pm\sqrt{2}$.
\end{enumerate}
}
2. Giải hệ phương trình sau: $\heva{&2x-3y=8\\&-3x+y=-5}$.
\loigiai{
$\heva{&2x-3y=8\\&-3x+y=-5}\Leftrightarrow\heva{&2x-3y=8\\&-9x+3y=-15}\Leftrightarrow\heva{&2x-3y=8\\&-7x=-7}\Leftrightarrow\heva{&x=1\\&y=-2}$
}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10, Điện Biên, 2016]%[9D1B8]%[9D4B7]
(1,5 điểm)\\
Cho biểu thức $A=\left(\dfrac{\sqrt{x}}{\sqrt{x}-1}-\dfrac{2}{x-\sqrt{x}}\right):\left(\dfrac{1}{\sqrt{x}+1}+\dfrac{2}{x-1}\right)$ với $x>0;x\neq 1$.
\begin{enumerate}
\item Rút gọn biểu thức $A$.
\item Tìm giá trị của $x$ để $A=1$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Rút gọn biểu thức $A$.\\
Ta có:
$A=\left(\dfrac{\sqrt{x}}{\sqrt{x}-1}-\dfrac{2}{x-\sqrt{x}}\right):\left(\dfrac{1}{\sqrt{x}+1}+\dfrac{2}{x-1}\right)\\
=\left(\dfrac{\sqrt{x}}{\sqrt{x}-1}-\dfrac{2}{\sqrt{x}(\sqrt{x}-1)}\right):\left(\dfrac{1}{\sqrt{x}+1}+\dfrac{2}{(\sqrt{x}-1)(\sqrt{x}+1)}\right)\\
=\left(\dfrac{x-2}{\sqrt{x}(\sqrt{x}-1)}\right):\left(\dfrac{\sqrt{x}-1+2}{(\sqrt{x}-1)(\sqrt{x}+1)}\right)\\
=\dfrac{x-2}{\sqrt{x}(\sqrt{x}-1)}:\dfrac{\sqrt{x}+1}{(\sqrt{x}-1)(\sqrt{x}+1)}=\dfrac{x-2}{\sqrt{x}(\sqrt{x}-1)}:\dfrac{1}{\sqrt{x}-1}=\dfrac{x-2}{\sqrt{x}}$.
\item Tìm giá trị của $x$ để $A=1$.\\
Khi $A=1\Leftrightarrow\dfrac{x-2}{\sqrt{x}}=1\Leftrightarrow x-2=\sqrt{x}\Leftrightarrow x-\sqrt{x}-2=0\Leftrightarrow\hoac{&\sqrt{x}=-1\text{ (loại)}\\&\sqrt{x}=2}\Leftrightarrow x=4$.\\
Vậy với $x=4$ thì $A=1$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10, Điện Biên, 2016]%[9D4K8]
(1,0 điểm) Giải bài toán bằng cách lập phương trình\\
Một người đi xe đạp từ địa điểm $A$ đến địa điểm $B$ dài $30$ km. Khi đi từ $B$ về $A$ người đó tăng vận tốc thêm $3$km/h so với lúc đi nên thời gian về ít hơn thời gian đi là $30$ phút. Tính vận tốc của người đi xe đạp lúc đi từ $A$ đến $B$.
\loigiai{
Gọi vận tốc người đi xe đạp từ $A$ đến $B$ là $v$ km/h.\\
The đề ta có: $\dfrac{30}{v}-\dfrac{30}{v+3}=\dfrac{1}{2}\Leftrightarrow \dfrac{90}{v(v+3)}=\dfrac{1}{2}\Leftrightarrow v^2+3v-180=0\Leftrightarrow \hoac{&v=12\\&v=-15\text{ (loại)}}$.\\
Vậy vận tốc của người đi xe đạp lúc đi từ $A$ đến $B$ là $12$ km/h.
}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10, Điện Biên, 2016]%[9D4B4]%[9D4K6]
Cho phương trình $x^2-2(m+1)x+3m-1=0$. ($m$ là tham số)\hfill(1)
\begin{enumerate}
\item Chứng minh rằng phương trình (1) luôn có hai nghiệm phân biệt với mọi $m$.
\item Tìm $m$ để phương trình (1) có hai nghiệm $x_1,x_2$ là độ dài hai cạnh góc vuông của tam giác vuông có độ dài cạnh huyền bằng $\sqrt{26}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $\Delta'=(m+1)^2-(3m-1)=m^2-m+2=\left(m-\dfrac{1}{2}\right)^2+\dfrac{7}{4}>0,\forall m\in\mathbb{R}$.\\
Vậy phương trình đã cho luôn luôn có hai nghiệm phân biệt với mọi $m$.
\item Áp dụng định lý Vi-et ta có: $\heva{&x_1+x_2=2(m+1)\\&x_2=3m-1}$.\\
Yêu cầu bài toán: $x_1^2+x_2^2=(\sqrt{26})^2\Leftrightarrow(x_1+x_2)^2-2x_1x_2=26\Leftrightarrow 4(m+1)^2-2(3m-1)=26\\
\Leftrightarrow 4m^2+2m-20=0\Leftrightarrow\hoac{&m=2\\&m=-\dfrac{5}{2}}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10, Điện Biên, 2016]%[9H3Y7]%[9H3K4]%[9H3G8]
Cho đường tròn $(O,R)$ và một đường thẳng $d$ cố định, không cắt đường tròn $(O,R), M$ là một điểm bất kỳ trên đường thẳng $d$. Từ điểm $M$ kẻ hai tiếp tuyến $MA,MB$ với đường tròn $(O,R)$, trong đó $A,B$ là tiếp điểm.
\begin{enumerate}
\item Chứng minh tứ giác $MAOB$ nội tiếp đường tròn.
\item Gọi $I$ là trung điểm của đoạn thẳng $AB$. Chứng minh rằng tích $OM.OI$ là một số không đổi và không phụ thuộc vào vị trí điểm $M$.
\item Khi điểm $M$ di chuyển trên đường thẳng $d$, chứng minh rằng điểm $I$ nằm trên một đường tròn cố định.
\end{enumerate}
\loigiai{
\immini{
1. Ta có $MA, MB$ là tiếp tuyến của đường tròn nên $\heva{&MA\perp OA\\&MB\perp OB}\Rightarrow \widehat{MAO}+\widehat{MBO}=180^\circ$.\\
Do đó, tứ giác $MAOB$ nội tiếp đường tròn.\\
2. Xét hai tam giác $MAO$ và $AIO$ có $\widehat{I}=\widehat{A}=90^\circ$ và $\widehat{O}$ chung. Suy ra $\triangle MAO\backsim\triangle AIO$.\\
Do đó: $\dfrac{OM}{OA}=\dfrac{OA}{OI}\Rightarrow OM.OI=OA^2=R^2$ (đpcm).
}{
\begin{tikzpicture}[>=stealth,scale=1]
\tkzDefPoints{0/0/x,5/0/y,3/3/O,.5/0/M,3/0/H}
\tkzDrawCircle[R](O,2cm)
\tkzDrawLine[add 2 and 2](x,y)
\tkzTangent[from with R=M](O,2cm)\tkzGetPoints{A}{B}
\tkzTangent[from with R=H](O,2cm)\tkzGetPoints{A'}{B'}
\tkzInterLL(A,B)(O,M)\tkzGetPoint{I}
\tkzInterLL(A',B')(O,H)\tkzGetPoint{J}
\tkzDrawSegments(M,A M,B M,O O,A O,B A,B)
\tkzDrawSegments[dashed](O,H A',H B',H A',B')
\tkzDrawPoints(O,A,B,M,I,H,J,A',B')
\tkzLabelPoints[left](B,I,B')
\tkzLabelPoints[above right](O,J)
\tkzLabelPoints[below](M,A,H,A')
\tkzLabelLine[pos=1, below](x,y){$d$}
\tkzMarkRightAngles(A,I,O M,A,O M,B,O O,H,M)
\tkzMarkSegments[mark=|](I,B I,A)
\end{tikzpicture}
}
3. Gọi $H$ là hình chiếu của $O$ lên đường thẳng $d$. Từ $H$ kẻ hai tiếp tuyến $HA'$ và $HB'$, gọi $J$ là trung điểm của đoạn thẳng $A'B'$, suy ra $O,J,H$ thẳng hàng.\\
Tương tự trên ta cũng có ngay $OJ.OH=R^2$.\\
Do đó $OJ.OH=OI.OM$, suy ra $\triangle OIJ\backsim \triangle OHM$. Nên $\widehat{I}=\widehat{H}=90^\circ$.\hfill(*)\\
Mặt khác, $O, d$ cố định suy ra $H,A',B',J$ cố định. Từ (*) suy ra $I$ luôn nhìn đoạn $OJ$ dưới một góc vuông. Suy ra $I$ luôn nằm trên đường tròn đường kính $OJ$.
}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10, Điện Biên, 2016]%[9D5G1]%[9D5G2]
(1,0 điểm)
\begin{enumerate}
\item Cho $x,y$ thỏa mãn hệ thức: $x^2+2y^2+2018(x+y)+2xy+4032=0$. Hãy tìm giá trị lớn nhất, giá trị nhỏ nhất của biểu thức $A=x+y+1$.
\item Cho $n$ là số nguyên dương, sao cho $2n+1, 3n+1$ là số chính phương. Chứng minh rằng $n$ chia hết cho $40$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có\\
\textbf{Cách 1:}\\
$(x+y)^2+y^2+2018(x+y)+4032=0\Leftrightarrow (x+y)^2+2018(x+y)+4032=-y^2\leq 0\\
\Leftrightarrow -2016\leq x+y\leq -2\Leftrightarrow -2015\leq x+y+1\leq -1\Leftrightarrow -2015\leq A\leq -1$.\\
Vậy $\min A=-2015$, khi $x=-2016; y=0$.\\
$\max A=-1$, khi $x=-2; y=0$.\\
\textbf{Cách 2:}\\
$(x+y)^2+y^2+2018(x+y)+4032=0\Leftrightarrow (x+y)^2+2018(x+y)+4032=-y^2\\
\Leftrightarrow (x+y+1009)^2=-y^2+1014049\Leftrightarrow (x+y+1009)^2\leq 1014049\Leftrightarrow \left|x+y+1009\right|\leq 1007\\
\Leftrightarrow -1007\leq x+y+1009\leq 1007\Leftrightarrow -2015\leq x+y+1\leq -1\Leftrightarrow -2015\leq A\leq -1$.\\
Vậy $\min A=-2015$, khi $x=-2016; y=0$.\\
$\max A=-1$, khi $x=-2; y=0$.
\item $n\in \mathbb{N}^*$. Giả sử có $a,b\in\mathbb{N}^*$ để:\\
$2n+1=a^2$.\hfill(1)\\
$3n+1=b^2$.\hfill(2)\\
Từ (1), suy ra $a$ lẻ, đặt $a=2k+1$, ta được $2n+1=(2k+1)^2\Leftrightarrow n=2k(k+1)$. Do đó $n$ là số chẵn và chia hết cho $4$.\\
$n$ chẵn, suy ra $3n+1$ lẻ, đặt $b=2p+1$, ta được $3n+1=(2p+1)^2$.\\
Lấy (1)+(2) ta có $5n+2=4k(k+1)+1+4p(p+1)+1\Leftrightarrow 5n=4k(k+1)+4p(p+1)$. \\
Suy ra $5n$ chia hết cho $8$, nên $n$ chia hết cho $8$.\hfill(3)\\
Cần chứng minh $n$ chia hết cho $5$. Chú ý rằng các số chính phương chỉ có chữ số tận cùng là $0,1,4,5,6,9$.\\
Xét các trường hợp:\\
\textbf{TH 1:} $n=5q+1$, suy ra $a^2=2(5q+1)+1=10q+3$ có chữ số tận cùng bằng $3$, suy ra vô lý.\\
\textbf{TH 2:} $n=5q+2$, suy ra $b^2=3(5q+2)+1=15q+7$ có chữ số tận cùng bằng $7$ (vì $n$ chẵn nên $q$ chẵn), suy ra vô lý.\\
\textbf{TH 3:} $n=5q+3$, suy ra $a^2=2(5q+3)+1=10q+7$ có chữ số tận cùng bằng $7$, suy ra vô lý.\\
\textbf{TH 4:} $n=5q+4$, suy ra $b^2=3(5q+4)+1=15q+13$ có chữ số tận cùng bằng $3$, suy ra vô lý.\\
suy ra $n$ chia hết cho $5$. \hfill(4)\\
Từ (3) và (4), suy ra $n$ chia hết cho $40$.
\end{enumerate}
}
\end{ex}
