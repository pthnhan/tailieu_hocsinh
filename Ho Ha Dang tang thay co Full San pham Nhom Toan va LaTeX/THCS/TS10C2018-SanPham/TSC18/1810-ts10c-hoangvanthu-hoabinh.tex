\begin{ex}%[Đề thi vào 10, chuyên Hoàng Văn Thụ, tỉnh Hoà Bình, năm 2018]%[Cấn Thị Thanh Nhàn, Dự án EX-TS10C-18]%[9D1G3]
	Rút gọn các biểu thức sau
	\begin{enumerate}
		\item $A=\dfrac{2}{\sqrt{3}-1}-\dfrac{2}{\sqrt{3}+1}$.
		\item $B= \left(\dfrac{\sqrt{x}}{x-4}+\dfrac{1}{\sqrt{x}-2}\right)\cdot\dfrac{\sqrt{x}-2}{2}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $A=\dfrac{2}{\sqrt{3}-1}-\dfrac{2}{\sqrt{3}+1} = \dfrac{2(\sqrt{3}+1) -2(\sqrt{3}-1)}{(\sqrt{3}+1)(\sqrt{3}-1)}=\dfrac{2\sqrt{3}+2-2\sqrt{3}+2}{2}=2$.
			\item 
			Điều kiện: $x \ge 0, x \neq 4$. Khi đó
			\begin{align*}
			B&=\left(\dfrac{\sqrt{x}}{x-4}+\dfrac{1}{\sqrt{x}-2}\right)\dfrac{\sqrt{x}-2}{2}\\ 
			& = \dfrac{\sqrt{x}+\sqrt{x}+2}{(\sqrt{x}-2)(\sqrt{x}+2)} .\dfrac{\sqrt{x}-2}{2}\\
			&=\dfrac{2(\sqrt{x}+1)}{(\sqrt{x}-2)(\sqrt{x}+2)}.\dfrac{\sqrt{x}-2}{2}\\
			&=\dfrac{\sqrt{x}+1}{\sqrt{x}+2}.
			\end{align*}
			
			Vậy $B = \dfrac{\sqrt{x}+1}{\sqrt{x}+2}$ với $x \ge 0,x \neq 4$.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Hoàng Văn Thụ, tỉnh Hoà Bình, năm 2018]%[Cấn Thị Thanh Nhàn, Dự án EX-TS10C-18]%[9D2G3]
	\begin{enumerate}
		\item Giải phương trình $x^2 -2|x-3|-5 =0$.
		\item Giải hệ phương trình 
		$\heva{&x+2y+1=0\\
			   &x^2+y^2-2x+4y-3&=0.}$
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
			\item \textbf{TH1.} $x-3 \ge 0 \Leftrightarrow x\ge 3$. Khi đó phương trình tương đương
			\begin{eqnarray*}
				&&x^2 -2(x-3)-5=0\\ 
				& \Leftrightarrow& x^2-2x+1=0\\
				& \Leftrightarrow& (x-1)^2=0\\
				& \Leftrightarrow& x=1 \quad(\text{loại}).
			\end{eqnarray*}
			
			\textbf{TH2.} $x-3 < 0 \Leftrightarrow x< 3$. Khi đó phương trình tương đương
			\begin{eqnarray*}
				&& x^2 -2(3-x)-5=0\\ 
				& \Leftrightarrow & x^2+2x-11=0\\
				& \Leftrightarrow & 
					\hoac{&x=-1+2\sqrt{3} \quad\text{(nhận)} \\ &x=-1-2\sqrt{3} \quad\text{(nhận).}}
			\end{eqnarray*}
			Vậy phương trình có hai nghiệm: $x=-1+2\sqrt{3}$ hoặc $x=-1-2\sqrt{3}$.
			\item Phương trình thứ nhất tương đương $x=-2y-1$, thay vào phương trình thứ hai ta có
			\begin{eqnarray*}
				&&(-2y-1)^2 +y^2-2(-2y-1)+4y-3=0\\ 
				& \Leftrightarrow & 4y^2+4y+1+y^2+4y+2+4y-3=0\\
				& \Leftrightarrow & 5y^2+12y=0\\
				& \Leftrightarrow & y=0 \vee y = \dfrac{-12}{5}.
			\end{eqnarray*}
			\begin{itemize}
				\item $y=0 \Rightarrow x =-1$.
				\item $y=\dfrac{-12}{5} \Rightarrow x =\dfrac{19}{5}$.
			\end{itemize}
			Vậy hệ phương trình có tập nghiệm $S = \left\{(-1;0),\left(\dfrac{19}{5};\dfrac{-12}{5}\right)\right\}.$
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Hoàng Văn Thụ, tỉnh Hoà Bình, năm 2018]%[Cấn Thị Thanh Nhàn, Dự án EX-TS10C-18]%[9D4G3]
	\begin{enumerate}
		\item Cho parabol $(P): y=ax^2$ và đường thẳng $d: y =3x-2.$ Tìm $a$ để $(d)$ cắt $(P)$ tại điểm có hoành độ bằng $2$, khi đó hãy vẽ đồ thị $(P)$ trên mặt phẳng tọa độ.
		\item Quãng đường từ $A$ đến $B$ dài $72$km. Người thứ nhất đi xe đạp từ $A$ đến $B$, sau đó $1$ giờ, người thứ hai đi xe đạp từ $B$ đến $A$. Hai người gặp nhau tại $C$ cách $B$ là $30$km, vận tốc của người đi xe thứ hai lớn hơn vận tốc của người đi xe thứ nhất là $1$km. Tính thời gian của mỗi người từ lúc khởi hành đến lúc gặp nhau. 
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
			\item Phương trình hoành độ giao điểm của $(d)$ và $(P)$ là  $ax^2 -3x+2=0$.\\ 
			Vì $(d)$ cắt $(P)$ tại một điểm có hoành độ bằng $2$ nên thay $x=2$ vào phương trình ta có $$ 4a-6+2=0 \Leftrightarrow a=1. $$
			Vậy $(P) \colon y=x^2$. \\
			Bảng giá trị của $(P)$
			\begin{center}
				\renewcommand{\arraystretch}{1.25}
				\begin{tabular}{|c|c|c|c|c|c|}
					\hline
					$ x $& $ -2 $& $ -1 $& $ 0 $&$ 1 $&$ 2 $ \\ 
					\hline
					$ y $& $ 4 $& $ 1 $&$ 0 $ &$ 1 $&$ 4 $\\ 
					\hline
				\end{tabular}
			\end{center}
			Đồ thị hàm số $ (P)\colon y=x^2 $
			\begin{center}
			\begin{tikzpicture}[>=stealth,x=1cm,y=1cm,scale=1]
			\def\xmin{-2.5};\def\xmax{2.5};\def\ymin{-0.5};\def\ymax{4.5};
			\draw[->,line width = 1pt] (\xmin,0) -- (\xmax,0) node[right] {$x$};
			\draw[->,line width = 1pt] (0,\ymin) -- (0,\ymax) node[above] {$y$};
			\draw($(0,0)$)node[below left]{$O$};

			\end{center}
			\item Gọi vận tốc của người đi xe thứ nhất là $x$ (đơn vị: km / h), $ x>0 $.\\
			Khi đó, vận tốc của người đi xe thứ hai là $x+1$.\\
			Quãng đường người đi xe thứ nhất tới $C$ là: $72-30=42$ (km).\\
			Do đó, thời gian của người đi xe thứ nhất tới $C$ là: $\dfrac{42}{x}$ (km / h).\\
			Quãng đường người đi xe thứ hai tới $C$ là: $30$ (km).\\
			Do đó, thời gian của người đi xe thứ hai tới $C$ là: $\dfrac{30}{x+1}$ (km / h).\\
			Vì người đi xe thứ nhất xuất phát trước $1$ giờ so với người đi xe thứ hai nên ta có phương trình:
			\begin{eqnarray*}
				&&\dfrac{42}{x} = 1+\dfrac{30}{x+1}\\
				&\Leftrightarrow& 42(x+1) = x(x+1) +30x\\
				&\Leftrightarrow& x^2-11x-42=0\\
				&\Leftrightarrow& \hoac{&x =14 \quad(\text{nhận}) \\ &x=-3 \quad(\text{loại}).}
			\end{eqnarray*}
			Vậy thời gian người đi xe thứ nhất từ lúc khởi hành tới lúc gặp nhau là $\dfrac{42}{14} = 3$ (giờ).\\
			Thời gian người đi xe thứ hai từ lúc khởi hành tới lúc gặp nhau là $\dfrac{30}{15} = 2$ (giờ).
		\end{enumerate}
	}

	Cho đường tròn $(O; R)$ và điểm $A$ cố định sao cho $OA=2R.$ Đường kính $BC$ của $(O)$ quay quanh $O$ nhưng đường thẳng $BC$ không đi qua điểm $A.$ Vẽ đường tròn qua ba điểm $A, B, C$ cắt đường thẳng $OA$ tại điểm thứ hai là $I$. \begin{ex}%[Đề thi vào 10, chuyên Hoàng Văn Thụ, tỉnh Hoà Bình, năm 2018]%[Cấn Thị Thanh Nhàn, Dự án EX-TS10C-18]%[9D0G3]
	\begin{enumerate}
		\item Tìm số tự nhiên $n$ sao cho $(n+5)$ và $(n+30)$ đều là các số chính phương. 
		\item Hôm nay là thứ Ba, hỏi sau $(1^3+2^3+3^3+\dots+110^3+111^3)$ ngày nữa thì là ngày thứ mấy trong tuần?
	\end{enumerate} 
	\loigiai {
		\begin{enumerate}
			\item Để $n+5, n+30$ là các số chính phương thì phải tồn tại các số tự nhiên $a,b$ sao cho $$n+5 =b^2; n+30=a^2 \text{ với }  (a,b \in \mathbb{N}^*,a>b).$$
			Ta có $a^2-b^2 = 25 \Leftrightarrow (a-b)(a+b)=25$.\\
			Mặt khác, $a,b \in \mathbb{N}^*$ nên ta suy ra $(a-b),(a+b) \in \{1;5;25\}$.\\ Ta thấy $a+b>a-b$ nên $a-b =1$ và $a+b=25.$ 
			Do đó, $a= 13; b=12$. Suy ra $n=139.$
			\item 
			Ta có $n^3-n=n(n-1)(n+1) \Rightarrow n^3=(n-1)n(n+1)+n$, áp dụng ta có
			\begin{eqnarray*}
				&&  1^3+2^3+3^3+\cdots+110^3+111^3 \\
				&=& (1-1)1(1+1)+1+(2-1)2(2+1)+2+(3-1)3(3+1)+3+\cdots+ \\
				&&  +(110-1)110(110+1)+110+(111-1)111(111+1)+111 \\
				&=& (1+2+3+\cdots+110+111)+(1\cdot2\cdot3+2\cdot3\cdot4+\cdots+109\cdot110\cdot111+110\cdot111\cdot112) \\
				&=& \dfrac{111(111+1)}{2}+\dfrac{110\cdot111\cdot112\cdot113}{4} \\
				&=& 2^6\cdot3^2\cdot7^2\cdot37^2 \text{ (chia hết cho 7)}
			\end{eqnarray*}
			Vậy sau $(1^3+2^3+3^3+\dots+110^3+111^3)$ ngày nữa là ngày thứ Ba.
		\end{enumerate}
	}
\end{ex}
