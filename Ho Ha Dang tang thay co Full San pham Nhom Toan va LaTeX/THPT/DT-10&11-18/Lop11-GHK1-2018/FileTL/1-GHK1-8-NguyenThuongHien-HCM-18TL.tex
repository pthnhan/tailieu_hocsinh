\begin{bt}%[Thi GHK, Nguyễn Thượng Hiền - Hồ Chí Minh, 2018]%[Nguyễn Ngọc Hóa, EX-DT-10&11]%[1D1B1-1]
	Tìm tập xác định của hàm số $y=\dfrac{\tan 2x-1}{(2\sin x-1)(\cos 3x-1)}$.
	\loigiai{
	Hàm số xác định khi và chỉ khi
		$$\heva{&\cos 2x\ne 0\\ &2\sin x-1\ne 0\\ &\cos 3x-1\ne 0}
		\Leftrightarrow \heva{&2x\ne \dfrac{\pi}{2}+k\pi\\ &\sin x\ne \dfrac{1}{2}\\ &\cos 3x\ne 1}
		\Leftrightarrow \heva{&x\ne \dfrac{\pi}{4}+k\dfrac{\pi}{2}\\ &x\ne \dfrac{\pi}{6}+k2\pi\\ &x\ne \dfrac{5\pi}{6}+k2\pi\\ &x\ne k\dfrac{2\pi}{3}}\quad (k\in \mathbb{Z}).$$	
Vậy $\mathscr{D}=\mathbb{R}\setminus \left\lbrace \dfrac{\pi}{4}+k\dfrac{\pi}{2}; \dfrac{\pi}{6}+k2\pi; \dfrac{5\pi}{6}+k2\pi; k\dfrac{2\pi}{3}, k\in \mathbb{Z}\right\rbrace$.	
}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Thượng Hiền - Hồ Chí Minh, 2018]%[Nguyễn Ngọc Hóa, EX-DT-10&11]%[1D1B1-5]
	Tìm giá trị lớn nhất, giá trị nhỏ nhất của hàm số sau
		$$y=2-4\sin ^2x\cdot \cos ^2x.$$
	\loigiai{
		Ta có $y=2-4\sin ^2x\cdot \cos ^2x=2-\sin ^22x$.\\
		Vì $-1\leq \sin 2x\leq 1$ nên $0\leq \sin ^22x\leq 1\Leftrightarrow 2-0\ge 2-\sin ^22x\ge 2-1\Leftrightarrow 2\ge y\ge 1$.\\
		Vậy $\min y=1$ khi $\sin^22x=1\Leftrightarrow 2x=\pm \dfrac{\pi}{2}+k2\pi\Leftrightarrow x=\pm \dfrac{\pi}{4}+k\pi, k\in \mathbb{Z}$;\\
		$\max y=2$ khi $\sin^22x=0\Leftrightarrow 2x=k\pi\Leftrightarrow x=k\dfrac{\pi}{2}, k\in \mathbb{Z}$.
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Thượng Hiền - Hồ Chí Minh, 2018]%[Nguyễn Ngọc Hóa, EX-DT-10&11]%[1D1B2-1]
	Giải các phương trình sau
	\begin{enumEX}{2}
		\item $\cos\left(3x-\dfrac{\pi}{6}\right)+2\sin ^2x=1$;
		\item $\dfrac{\sin 2x(\sqrt{2}\cos x+1)}{\tan x+1}=0$.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item 
			\allowdisplaybreaks
			\begin{eqnarray}
				\cos\left(3x-\dfrac{\pi}{6}\right)+2\sin ^2x=1&\Leftrightarrow 	&\cos\left(3x-\dfrac{\pi}{6}\right)=1-2\sin ^2x\notag\\
				&\Leftrightarrow 	&\cos\left(3x-\dfrac{\pi}{6}\right)=\cos 2x\notag\\
				&\Leftrightarrow 	&\hoac{&3x-\dfrac{\pi}{6}=2x+k2\pi\\ &3x-\dfrac{\pi}{6}=-2x+k2\pi} (k\in \mathbb{Z})\notag\\
				&\Leftrightarrow 	&\hoac{&x=\dfrac{\pi}{6}+k2\pi\\ &5x=\dfrac{\pi}{6}+k2\pi} (k\in \mathbb{Z})\notag\\
				&\Leftrightarrow 	&\hoac{&x=\dfrac{\pi}{6}+k2\pi\\ &x=\dfrac{\pi}{30}+k\dfrac{2\pi}{5}} (k\in \mathbb{Z}).\notag
			\end{eqnarray}
			Vậy phương trình có hai họ nghiệm $x=\dfrac{\pi}{6}+k2\pi$ và $x=\dfrac{\pi}{30}+k\dfrac{2\pi}{5}$, $k\in \mathbb{Z}$.
			\item Điều kiện xác định: $\tan x+1\ne 0\Leftrightarrow \tan x\ne -1\Leftrightarrow x\ne -\dfrac{\pi}{4}+k\pi, k\in \mathbb{Z}$.
			\allowdisplaybreaks
			\begin{eqnarray}
				\dfrac{\sin 2x(\sqrt{2}\cos x+1)}{\tan x+1}=0&\Rightarrow &\sin 2x(\sqrt{2}\cos x+1)=0\notag\\
				&\Leftrightarrow &\hoac{&\sin 2x=0\\ &\sqrt{2}\cos x+1=0}\notag\\
				&\Leftrightarrow &\hoac{&2x=k\pi\\ &\cos x=-\dfrac{1}{\sqrt{2}}}\notag\\
				&\Leftrightarrow &\hoac{&x=k\dfrac{\pi}{2} \text{ (nhận)}\\ &x=\dfrac{3\pi}{4}+k2\pi \text{ (loại)}\\ &x=-\dfrac{3\pi}{4}+k2\pi \text{ (nhận)}}\notag
			\end{eqnarray}
			Vậy phương trình có hai họ nghiệm $x=k\dfrac{\pi}{2}$ và $x=-\dfrac{3\pi}{4}+k2\pi$, $k\in \mathbb{Z}$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Thượng Hiền - Hồ Chí Minh, 2018]%[Nguyễn Ngọc Hóa, EX-DT-10&11]%[1D1K3-3]
	Giải phương trình sau $\sqrt{3}\cos 2x-2\sin x\cos x+1=0$.
	\loigiai{
		\allowdisplaybreaks
		\begin{eqnarray*}
			\sqrt{3}\cos 2x-2\sin x\cos x+1=0&\Leftrightarrow &\sqrt{3}\cos 2x-\sin 2x=-1\notag\\
			&\Leftrightarrow &\dfrac{\sqrt{3}}{2}\cos 2x-\dfrac{1}{2}\sin 2x=-\dfrac{1}{2}\notag\\
			&\Leftrightarrow &\cos \dfrac{\pi}{6}\cdot \cos 2x-\sin \dfrac{\pi}{6}\cdot \sin 2x=\cos \dfrac{2\pi}{3}\notag\\
			&\Leftrightarrow &\cos \left(2x+\dfrac{\pi}{6}\right)=\cos \dfrac{2\pi}{3}\notag\\
			&\Leftrightarrow &\hoac{&2x+\dfrac{\pi}{6}=\dfrac{2\pi}{3}+k2\pi\\  &2x+\dfrac{\pi}{6}=-\dfrac{2\pi}{3}+k2\pi} (k\in \mathbb{Z})\notag\\
			&\Leftrightarrow &\hoac{&2x=\dfrac{\pi}{2}+k2\pi\\  &2x=-\dfrac{5\pi}{6}+k2\pi} (k\in \mathbb{Z})\notag\\
			&\Leftrightarrow &\hoac{&x=\dfrac{\pi}{4}+k\pi\\  &x=-\dfrac{5\pi}{12}+k\pi} (k\in \mathbb{Z}).\notag
		\end{eqnarray*}
	Vậy phương trình có hai họ nghiệm $x=\dfrac{\pi}{4}+k\pi, x=-\dfrac{5\pi}{12}+k\pi (k\in \mathbb{Z})$.
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Thượng Hiền - Hồ Chí Minh, 2018]%[Nguyễn Ngọc Hóa, EX-DT-10&11]%[1D1G3-2]
	Giải các phương trình sau
	$$5\sin ^22x-3\sin 4x+4\sin ^4x-4\sin ^2x+1=0.$$
	\loigiai{
\begin{align*}
			&5\sin ^22x-3\sin 4x+4\sin ^4x-4\sin ^2x+1=0\notag\\
			\Leftrightarrow &5\sin ^22x-6\sin 2x\cos 2x+(2\sin ^2x-1)^2=0\notag\\
			\Leftrightarrow &5\sin ^22x-6\sin 2x\cos 2x+\cos^22x=0.\tag{1}
		\end{align*}
	\begin{itemize}
		\item Nếu $\cos 2x=0\Rightarrow \sin 2x=\pm 1$. Thay vào phương trình $(1)$, ta được $5=0$ (vô lý) nên $\cos 2x \ne 0$.
		\item Với $\cos 2x\ne 0\Leftrightarrow 2x\ne \dfrac{\pi}{2}+k\pi\Leftrightarrow x\ne \dfrac{\pi}{4}+k\dfrac{\pi}{2}, k\in \mathbb{Z}$. Chia hai vế của phương trình $(1)$ cho $\cos^22x$, ta được
			$$5\tan^2 2x-6\tan 2x+1=0 \Leftrightarrow \hoac{&\tan 2x=1\\ &\tan 2x=\dfrac{1}{5}.}$$
		\begin{itemize}
			\item[+] $\tan2x=1\Leftrightarrow 2x=\dfrac{\pi}{4}+k\pi\Leftrightarrow x=\dfrac{\pi}{8}+k\dfrac{\pi}{2}, k\in \mathbb{Z}$.
			\item[+] $\tan2x=\dfrac{1}{5}\Leftrightarrow 2x=\arctan\dfrac{1}{5}+k\pi\Leftrightarrow x=\dfrac{1}{2}\arctan\dfrac{1}{5}+k\dfrac{\pi}{2}, k\in \mathbb{Z}$.
		\end{itemize}
Kết hợp với điều kiện, phương trình có hai họ nghiệm
$$x=\dfrac{\pi}{8}+k\dfrac{\pi}{2} \text{ và} x=\dfrac{1}{2}\arctan\dfrac{1}{5}+k\dfrac{\pi}{2}, k\in \mathbb{Z}$$
\end{itemize}
	}
\end{bt}

\begin{bt}%[Thi GHK, Nguyễn Thượng Hiền - Hồ Chí Minh, 2018]%[Nguyễn Ngọc Hóa, EX-DT-10&11]%[1D1G3-3]
	Giải các phương trình sau
		$$\sqrt{3}\sin 2x-2\cos^2x-4\sin x+4\sqrt{3}\cos x+4=0.$$
\loigiai{
	\allowdisplaybreaks
\begin{eqnarray*}
	& &\sqrt{3}\sin 2x-2\cos^2x-4\sin x+4\sqrt{3}\cos x+4=0\notag\\
	&\Leftrightarrow &\sqrt{3}\sin 2x-\cos2x-4\sin x+4\sqrt{3}\cos x+3=0\notag\\
	&\Leftrightarrow &2\left(\dfrac{\sqrt{3}}{2}\sin 2x-\dfrac{1}{2}\cos 2x\right)-8\left(\dfrac{1}{2}\sin x-\dfrac{\sqrt{3}}{2}\cos x\right)+3=0\notag\\
	&\Leftrightarrow &-2\left(\cos 2x\cdot\cos \dfrac{\pi}{3}-\sin 2x\cdot \sin \dfrac{\pi}{3}\right)+8\left(\cos x\cdot \cos \dfrac{\pi}{6}-\sin x\cdot \sin\dfrac{\pi}{6}\right)+3=0\notag\\
	&\Leftrightarrow &-2\cos\left(2x+\dfrac{\pi}{3}\right)+8\cos\left(x+\dfrac{\pi}{6}\right)+3=0\notag\\
	&\Leftrightarrow &-2\left[2\cdot \cos^2\left(x+\dfrac{\pi}{6}\right)-1\right]+8\cos\left(x+\dfrac{\pi}{6}\right)+3=0\notag\\
	&\Leftrightarrow &-4\cdot \cos^2\left(x+\dfrac{\pi}{6}\right)+8\cos\left(x+\dfrac{\pi}{6}\right)+5=0.\notag
\end{eqnarray*}
Đặt $t=\cos\left(x+\dfrac{\pi}{6}\right)$, $|t|\leq 1$.\\
Phương trình trở thành: $-4t^2+8t+5=0\Leftrightarrow \hoac{&t=\dfrac{5}{2}\text{ (loại)}\\ &t=-\dfrac{1}{2}\text{ (nhận)}.}$\\
Với $t=-\dfrac{1}{2}\Rightarrow \cos\left(x+\dfrac{\pi}{6}\right)=-\dfrac{1}{2}=\cos\dfrac{2\pi}{3}$
$$\Leftrightarrow \hoac{&x+\dfrac{\pi}{6}=\dfrac{2\pi}{3}+k2\pi\\ &x+\dfrac{\pi}{6}=-\dfrac{2\pi}{3}+k2\pi}(k\in \mathbb{Z})
\Leftrightarrow \hoac{&x=\dfrac{\pi}{2}+k2\pi\\ &x=\dfrac{-5\pi}{6}+k2\pi} (k\in \mathbb{Z}).$$
}
\end{bt}
