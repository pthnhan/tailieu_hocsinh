\begin{ex}%[Đề thi vào 10, chuyên PTNK, Hồ Chí Minh, năm 2018]%[Hoàng Hải, Dự án EX-TS10C-18]%[9D0G1]
	Biết $ 0<x\le y $ và
	$$ \left( \dfrac{(\sqrt{x}+\sqrt{y})^2+(\sqrt{x}-\sqrt{y})^2}{(\sqrt{x}+\sqrt{y})(\sqrt{x}-\sqrt{y})+2(x+2y)} \right) + \left( \dfrac{y}{\sqrt{x}(\sqrt{x}+\sqrt{y})}+\dfrac{x}{\sqrt{y}(\sqrt{x}+\sqrt{y})} \right) =\dfrac{5}{3}. \text{ Tính } \dfrac{x}{y} $$.
	\loigiai{
		Xét vế trái của đẳng thức đề bài cho, ta có
		\begin{align*}
		\text{VT}=&\left( \dfrac{(\sqrt{x}+\sqrt{y})^2+(\sqrt{x}-\sqrt{y})^2}{(\sqrt{x}+\sqrt{y})(\sqrt{x}-\sqrt{y})+2(x+2y)} \right) + \left( \dfrac{y}{\sqrt{x}(\sqrt{x}+\sqrt{y})}+\dfrac{x}{\sqrt{y}(\sqrt{x}+\sqrt{y})} \right) \\
		=& \dfrac{2(x+y)}{3(x+y)} + \dfrac{y\sqrt{y}+x\sqrt{x}}{\sqrt{xy}(\sqrt{x}+\sqrt{y})}\\
		=&\dfrac{2}{3}+\dfrac{y\sqrt{y}+x\sqrt{x}}{\sqrt{xy}(\sqrt{x}+\sqrt{y})}.
		\end{align*}
		Với $ 0<x\le y $, đặt $ a = \dfrac{x}{y}, 0<a\le 1$, suy ra $ x=ay $, ta có
		\begin{eqnarray*}
			& & \dfrac{2}{3} + \dfrac{y\sqrt{y}(1+a\sqrt{a})}{y\sqrt{y}(a+\sqrt{a})}=\dfrac{5}{3} \\
			& \Leftrightarrow &\dfrac{1+a\sqrt{a}}{a+\sqrt{a}}=1 \\
			& \Leftrightarrow & 1+a\sqrt{a}=a+\sqrt{a}\\
			& \Leftrightarrow & (a-1)(\sqrt{a}-1)=0 \\
			& \Leftrightarrow & a=1.
		\end{eqnarray*}
		Vậy $ \dfrac{x}{y}=1 $.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên PTNK, Hồ Chí Minh, năm 2018]%[Hoàng Hải, Dự án EX-TS10C-18]%[9D0K2]%[9D0G2]
	\begin{enumerate}
		\item Giải phương trình $ \dfrac{2x^2(7-x)}{\sqrt{3-x}} = x(x-7) $.
		\item Giải hệ phương trình $ \heva{&(x+3)(x-1) = (y-2)(x+3) \\ &(x-1)\sqrt{y^2-5y+8} = (y-2)^2. }$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Điều kiện $ x<3 $, khi đó phương trình tương đương
			\begin{eqnarray*}
				x(7-x)\left(\dfrac{2x}{\sqrt{3-x}}+1 \right) = 0 
				& \Leftrightarrow & \hoac{&x=0 &\qquad\text{(nhận)} \\ &x=7 &\qquad\text{(loại)} \\ &\dfrac{2x}{\sqrt{3-x}}+1 = 0 &\qquad(*).}
			\end{eqnarray*}
			Xét phương trình (*), ta có
			$$ \dfrac{2x+\sqrt{3-x}}{\sqrt{3-x}}=0 
				\Leftrightarrow \sqrt{3-x}=-2x 
				\Leftrightarrow \heva{&x\le 0 \\ & 3-x=4x^2} 
				\Leftrightarrow \heva{&x\le 0 \\ & x=-1 \vee x=\dfrac{3}{4}} \Leftrightarrow x=-1. $$
			Vậy tập nghiệm của phương trình là $ S=\{-1;0\} $.
			\item Ta có $ y^2-5y+8=\left( y-\dfrac{5}{2} \right)^2+\dfrac{7}{4}>0 $ với mọi $ y $. Xét phương trình
			$$(x+3)(x-1) = (y-2)(x+3) 
			\Leftrightarrow  (x+3)(x-y+1)=0 
			\Leftrightarrow \hoac{&x=-3 \\ &x=y-1.}
			$$
			\begin{itemize}
				\item Với $ x=-3 $, thế vào phương trình còn lại ta có
				\begin{eqnarray*}
					-4\sqrt{y^2-5y+8}=(y-2)^2 \qquad\text{(loại, vì VT$<0$ và VP$\ge 0$).}
				\end{eqnarray*}
				\item Với $ x=y-1 $, thế vào phương trình còn lại ta có
				\begin{eqnarray*}
					(y-2)\sqrt{y^2-5y+8}=(y-2)^2 
					& \Leftrightarrow & \hoac{&y-2=0 \\ &\sqrt{y^2-5y+8}=y-2} \\
					& \Leftrightarrow & \hoac{&y=2 \\ &\heva{&y\ge 2 \\ &y^2-5y+8=(y-2)^2}}\\
					& \Leftrightarrow & \hoac{&y=2 &\text{(nhận)} \\ &y=4 &\text{(nhận)}.}
				\end{eqnarray*}
			\end{itemize}
			Vậy tập nghiệm của hệ phương trình là $ S=\{(1;2),(3;4) \} $
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên PTNK, Hồ Chí Minh, năm 2018]%[Hoàng Hải, Dự án EX-TS10C-18]%[9D0B2]%[9D0K2]
	Cho phương trình $ x^2-x+3m-11=0 \quad(1) $.
	\begin{enumerate}
		\item Với giá trị nào của $ m $ thì phương trình $ (1) $ có nghiệm kép? Tìm nghiệm đó.
		\item Tìm $ m $ để $ (1) $ có hai nghiệm phân biệt $ x_1,x_2 $ sao cho $ 2017x_1+2018x_2=2019 $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $ \Delta = 1-4(3m-11) =  45-12m $. $ (1) $ có nghiệm kép khi và chi khi $ \Delta=0 \Leftrightarrow m=\dfrac{15}{4} $.\\
			Khi đó nghiệm kép $ x_1,x_2 $ của $ (1) $ là $ x_1=x_2=\dfrac{1}{2} $.
			\item Phương trình $ (1) $ có hai nghiệm phân biệt $ x_1,x_2 $ khi và chỉ khi $ \Delta>0\Leftrightarrow m<\dfrac{15}{4} $.\\
			Theo định lý Vi-ét và yêu cầu bài toán, ta có
			\begin{eqnarray*}
				\heva{&x_1+x_2=1 \\ &x_1x_2=3m-11 \\ &2017x_1+2018x_2=2019} 
				& \Leftrightarrow & \heva{&x_1+x_2=1 \\ &x_1x_2=3m-11 \\ &2017(x_1+x_2)+x_2=2019} \\
				& \Leftrightarrow & \heva{&x_1=-1 \\ &x_2=2 \\&3m-11=-2} \\
				& \Leftrightarrow & \heva{&x_1=-1 \\ &x_2=2 \\&m=3\qquad\text{(nhận).}}
			\end{eqnarray*}
			Vậy $ m=3 $ thỏa yêu cầu bài toán.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên PTNK, Hồ Chí Minh, năm 2018]%[Hoàng Hải, Dự án EX-TS10C-18]%[9D0T2]
	\begin{enumerate}
		\item Đầu tháng 5 năm 2018, khi đang vào vụ thu hoạch, giá dưa hấu bất ngờ giảm mạnh. Nông dân A cho biết vì sợ dưa hỏng nên phải bán 30\% số dưa hấu thu hoạch được với giá 1500 đồng mỗi kilogam (1500đ/kg), sau đó nhờ phong trào “giải cứu dưa hấu” nên đã may mắn bán hết số dưa còn lại với giá 3500đ/kg; nếu trừ tiền đầu tư thì lãi được 9 triệu đồng (không kể công chăm sóc hơn hai tháng của cả nhà). Cũng theo ông A, mỗi sào đầu tư (hạt giống, phân bón,..) hết 4 triệu đồng và thu hoạch được 2 tấn dưa hấu. Hỏi ông A đã trồng bao nhiêu sào dưa hấu?
		\item Một khi đất hình chữ nhật ABCD ( AB < AD) có chu vi 240 mét được chia thành hai phần gồm khu đất hình chữ nhật ABMN làm chuồng trại và phần còn lại làm vườn thả để nuôi gà (M, N lần lượt thuộc các cạnh AD, BC). Theo quy hoạch trang trại nuôi được 2400 con gà, bình quân mỗi con gà cần một mét vuông của diện tích vườn thả và diện tích vườn thả gấp ba lần diện tích chuồng trại. Tính chu vi của khu đất làm vườn thả.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Gọi $ x $ là số sào dưa hấu ông A đã trồng,  $ x \in \mathbb{N}^* $.
			Khi đó số tấn dưa hấu thu hoạch là $ 2x $.\\
			Ta có
			\begin{center}
				\renewcommand*{\arraystretch}{2}

					\hline
					& Số tiền (triệu đồng) & Số tấn dưa hấu & Giá mỗi tấn (triệu đồng)
					\\\hline
					Bán lần I & $1,5\cdot 0,6x=0,9x$ & $30\%\cdot 2x = 0,6x$ & $ 1,5 $  \\\hline
					Bán lần II & $3,5\cdot 1,4x=4,9x$ & $70\%\cdot 2x = 1,4x$ & $ 3,5 $  \\\hline
					Tổng hai lần bán & $ 5,8x $ & $ 2x $ &
					\\\hline
				\end{tabular}
			\end{center}
			Số tiền lãi là $ 9 $ triệu đồng, ta có phương trình $5,8x - 4x = 9 \Leftrightarrow  x=5$.\\
			Vậy ông A đã trồng 5 sào dưa hấu.
			\item 
			\immini{
				Ta có $\heva{&S_{MNCD}=2400 \\ &S_{ABNM}=\dfrac{1}{3}S_{MNCD}=800.}$\\
				Kết hợp đề bài ta có\\
				$\heva{&(AB+AD)\cdot 2 = 240 \\ &AB\cdot AD=2400+800} \Leftrightarrow \heva{&AB+AD=120 \\ &AB\cdot AD=3200}$\\
				Khi đó $ AB,AD $ thỏa nghiệm phương trình\\
				$x^2-120x+3200=0 \Rightarrow \heva{&AB=40 \\ &AD=80.}$
			}
			{	\begin{tikzpicture}[scale=0.8]
				\tkzDefPoints{0/0/A,8/0/D,0/-4/B,8/-4/C}
				\tkzLabelPoints[above](A)

\begin{ex}%[Đề thi vào 10, chuyên PTNK, Hồ Chí Minh, năm 2018]%[Hoàng Hải, Dự án EX-TS10C-18]%[9H0G1]
	Tứ giác $ ABCD $ nội tiếp đường tròn $ (T) $ tâm $ O $, bán kính $ R $; $ \widehat{CAD}=45^\circ $, $ AC $ vuông góc với $ BD $ và cắt $ BD $ tại $ I $. Dựng $ CK $ vuông góc với $ AD $ $ (K\in AD) $, $ CK $ cắt $ BD $ tại $ H $ và cắt $ (T) $ tại $ E $ $ (E\not\equiv C) $.
	\begin{enumerate}
		\item Tính số đó góc $ \widehat{COD} $. Chứng minh các điểm $ C,I,K,D $ cùng thuộc một đường tròn và $ AC=BD $.
		\item Chứng minh $ A $ là tâm đường tròn ngoại tiếp tam giác $ BHE $. Tính $ IK $ theo $ R $.
		\item $ IK $ cắt $ AB $ tại $ F $. Chứng minh $ O $ là trực tâm tam giác $ AIK $ và $ CK\cdot CB = CF\cdot CD $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có\immini{
				$\heva{&\widehat{CAD}=\dfrac{1}{2}\text{Sđ}(\wideparen{CD})&\text{(góc nội tiếp)} \\ &\widehat{COD}=\text{Sđ}(\wideparen{CD})&\text{(góc ở tâm)}}$ \\
				$\Rightarrow \widehat{COD}=2\cdot \widehat{CAD}=90^\circ.$\\
				Xét tứ giác $ CIKD $ có:\\
				$\heva{&\widehat{CID}=90^\circ \qquad(AC\bot BD) \\
					&\widehat{CKD}=90^\circ \qquad(CK\bot AD) \\
					&\widehat{CID},\widehat{CKD} \text{ là hai góc kề cùng nhìn cạnh } CD}$\\
				$\Rightarrow $ Tứ giác $ CIKD $ nội tiếp.\\
				$\Rightarrow  C,I,K,D $ cùng nằm trên một đường tròn. \\
				Ta có $ \widehat{DAI}=\widehat{ADI}=45^\circ $ ($ \triangle AID $ vuông tại $ I $)
				$\Rightarrow \text{Sđ}(\wideparen{CD})=\text{Sđ}(\wideparen{AB}) \Rightarrow \text{Sđ}(\wideparen{DCB})=\text{Sđ}(\wideparen{ABC})$\\ $\Rightarrow AC=BD.$
			}
			{\vspace*{-5.5cm}
				\begin{tikzpicture}[>=stealth,x=1cm,y=1cm,scale=1]
				\tkzInit[ymin=-1,ymax=5,xmin=-1,xmax=15]
				\tkzDefPoints{0/0/O,3/0/C}
				
				\tkzDrawCircle(O,C)
				\tkzDefPointBy[rotation=center O angle 90](C)\tkzGetPoint{D}
				\tkzDefPointBy[rotation=center O angle 210](C)\tkzGetPoint{A}
				\tkzDefPointBy[projection=onto A--C](D)\tkzGetPoint{I}
				\tkzInterLC(D,I)(O,C)\tkzGetPoints{}{B}
				\tkzDefPointBy[projection=onto A--D](C)\tkzGetPoint{K}
				\tkzInterLC(C,K)(O,C)\tkzGetPoints{E}{}
				\tkzInterLL(D,B)(C,K)\tkzGetPoint{H}
				\tkzInterLL(A,B)(I,K)\tkzGetPoint{F}
				
				\tkzDrawPolygon(A,B,C,D)
				\tkzDrawSegments(O,C O,D A,C B,D C,E O,K)
				\tkzDrawSegments(O,I B,E A,H K,F B,F A,E C,F)
				\tkzDrawPoints(A,B,C,D,O,I,E,K,H,F)
				\tkzLabelPoints[left](O)
				\tkzLabelPoints[right](C)
				\tkzLabelPoints[above](K,D,E)
				\tkzLabelPoints[above left]()
				\tkzLabelPoints[above right](H)
				\tkzLabelPoints[below]()
				\tkzLabelPoints[below left](I,A)
				\tkzLabelPoints[below right](B,F)
				\tkzMarkAngles[arc=l,size=0.5 cm](A,D,I E,C,A C,A,D)
				\tkzMarkRightAngles(D,I,C C,K,D)
				\end{tikzpicture}
			}
			\item Ta có $ \heva{&\widehat{DAC}=\widehat{ADB}=45^\circ &\text{($ \triangle AID $ vuông tại $ I $)} \\  &\widehat{DAC}=\widehat{ACE}=45^\circ &\text{($ \triangle ACK $ vuông tại $ K $)} } \Rightarrow \text{Sđ}(\wideparen{AB})=\text{Sđ}(\wideparen{AE})\Rightarrow AB=AE$. \\
			Xét $ \triangle HCB $ có $ CI $ vừa là tia phân giác, vừa là đường cao nên $ CI $ cũng là đường trung tuyến, do đó $ I $ là trung điểm $ BH $. Khi đó $ \triangle AIH = \triangle AIB $ (hai cạnh góc vuông bằng nhau).\\
			Vậy $ AB=AH=AE $ do đó $ A $ là tâm đường tròn ngoại tiếp $ \triangle BHE $.\\
			Chứng minh tương tự $ I $ là trung điểm BH, ta có $ K $ là trung điểm EH.\\
			Ta có $ \widehat{BOE}=2\widehat{BCE}=2\cdot 
			(\widehat{ACE}+\widehat{ACB})=180^\circ $.\\
			Do đó $ B,O,E $ thẳng hàng và $ IK=\dfrac{1}{2}BE=R $ (vì $ IK $ là đường trung bình $\triangle BHE$).
			\item Ta có $\heva{&OE=OB \\ &IH=IB} \Rightarrow OI \parallel EC$ mà $ EC\bot AD $ nên $ OI\bot AD $.\\

			\text{Vậy $ \widehat{FBC}=\widehat{FIC} \Rightarrow FBIC $ nội tiếp}
				&\Rightarrow \heva{&\widehat{BFC}=\widehat{BIC}=90^\circ \\ &\widehat{BCF}=\widehat{BIF}=\widehat{KID}=\widehat{KCD}} \\
				&\Rightarrow \triangle FCB \backsim \triangle KCD \\
				&\Rightarrow \dfrac{CF}{CK}=\dfrac{CB}{CD}\Rightarrow CK\cdot CB= CF\cdot CD.
			\end{aligned}$
			
		\end{enumerate}
	}
\end{ex}
