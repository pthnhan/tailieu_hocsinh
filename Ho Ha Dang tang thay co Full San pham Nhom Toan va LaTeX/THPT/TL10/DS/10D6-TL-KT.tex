	Rút gọn các biểu thức sau
	\begin{enumerate}
		\item $A= \sqrt{\sin^2 x (1+ \cot x) + \cos^2 x (1+ \tan x)}$;
		
		\item $B= \dfrac{2 \sin (a+b)}{\cos(a+b) + \cos(a-b)}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			
			$\begin{aligned}
			A &= \sqrt{\sin^2 x (1+ \cot x) + \cos^2 x (1+ \tan x)} = \sqrt{\sin^2 x + 2 \sin x \cos x + \cos^2 x} \\
			&= \sqrt{(\sin x + \cos x)^2} = \vert \sin x + \cos x \vert . \end{aligned}$
			\item Ta có
			
			$ \begin{aligned}
			B &= \dfrac{2 \sin a \cos b + 2 \sin b \cos a}{\cos a \cos b - \sin a \sin b + \cos a \cos b + \sin a \sin b} \\
			&= \dfrac{2 \sin a \cos b + 2 \sin b \cos a}{2 \cos a \cos b} \\
			&= \tan a + \tan b.
			\end{aligned}$
		\end{enumerate}
	}
\end{ex}

\begin{ex}[1,0 điểm]%[Trương Hữu Dũng]%[0D6G3]
	Chứng minh rằng nếu tam giác $ABC$ có ba góc $A, B, C$ thỏa mãn 
	$\sin A = \cos B + \cos C$ thì tam giác $ABC$ vuông.
	\loigiai{
		Theo đề bài ta có 
		
		$\sin A = \cos B + \cos C$
		
		$\Leftrightarrow \sin A = 2 \cos \dfrac{B+C}{2} \cos \dfrac{B-C}{2}$
		
		$\Leftrightarrow 2 \sin \dfrac{A}{2} \cos \dfrac{A}{2}= 2 \sin \dfrac{A}{2} \cos \dfrac{B-C}{2}$
		
		$\Leftrightarrow \cos \dfrac{A}{2} = \cos \dfrac{B-C}{2}$
		
		$\Leftrightarrow \left[ 
		\begin{aligned}
		\dfrac{A}{2} &= \dfrac{C-B}{2} \\
		\dfrac{A}{2} &= \dfrac{B-C}{2}
		\end{aligned} \right.$	
		
		$\Leftrightarrow \left[ 
		\begin{aligned}
		A+B &= C \\
		A+C &= B.
		\end{aligned} \right.$
		
		Do đó, tam giác $ABC$ vuông tại $B$ hoặc vuông tại $C$.
	}
\end{ex}

\begin{ex}[1,0 điểm]%[Trương Hữu Dũng]%[0D6G3]
	Cho tam giác $ABC$	có ba góc $A, B, C$ thỏa mãn $2 \sin A \sin B = 1 + \cos C$. Hỏi tam giác $ABC$ là tam giác gì?
	\loigiai{
		Ta có 
		
		$\begin{aligned} 
		& 2 \sin A \sin B = 1 + \cos C \\
		\Leftrightarrow & 2 \sin A \sin B = 1 - \cos A \cos B + \sin A \sin B \\
		\Leftrightarrow & \cos A \cos B + \sin A \sin B =1 \\
		\Leftrightarrow & \cos (A-B) =1 \\
		\Leftrightarrow & A-B =0 \\
		\Leftrightarrow & A =B. \\
		\end{aligned}$
		
		Vậy tam giác $ABC$ cân tại $C$.
		
	}
\end{ex}

\subsection{Đề số 2a}
\setcounter{bt}{0}
\begin{bt}%[Đoàn Nhật Thiện]%[0D6B3]
	[2,0 điểm]
	Cho $\tan a=\dfrac{4}{3}\  \left({\pi < a< \dfrac{3\pi}{2}}\right)$. Tính $\sin 2a; \tan \left({2a-\dfrac{\pi}{4}}\right)$. 
	\loigiai{
		Ta có
		\begin{itemize}
			\item $\dfrac{1}{\cos^2 a}=1+\tan^2 a=1+\left(\dfrac{4}{3}\right)^2=\dfrac{25}{9}\Rightarrow\cos a=\dfrac{3}{5}$  $\left(\text{ do }\pi<a<\dfrac{3\pi}{2}\right)$ \dotfill ($0.5$đ)
			\item $\sin a=\dfrac{4}{5}\Rightarrow\sin 2a=\dfrac{24}{25}$\dotfill ($0.5$đ)
			\item $\cos 2a=2\cos^2a-1=\dfrac{7}{25}\Rightarrow\tan 2a=\dfrac{24}{7}$\dotfill ($0.5$đ)
			\item $\tan \left({2a-\dfrac{\pi}{4}}\right)=\dfrac{31}{17}$\dotfill ($0.5$đ)
	\end{itemize}}
\end{bt}
\begin{bt}%[Đoàn Nhật Thiện]%[0D6K2]
	[2,0 điểm]
	Chứng minh đẳng thức: $\dfrac{\sin^2x}{\sin x-\cos x}+\dfrac{\sin x+\cos x}{1-\tan^2x}=\sin x+\cos x$. 
	\loigiai{
		VT$=	\dfrac{\sin^2x}{\sin x-\cos x}+\dfrac{\left(\sin x+\cos x\right)\cos^2x}{\cos^2x -\sin^2x}$\dotfill ($0.5$đ)\\
		$	=\dfrac{\sin^2x}{\sin x-\cos x}+\dfrac{\cos^2x}{\cos x-\sin x}$ \dotfill ($0.5$đ)\\
		$	=\dfrac{\sin^2x-\cos^2x}{\sin x-\cos x}$\dotfill ($0.5$đ)\\
		$	=\sin x+\cos x=VP$\dotfill ($0.5$đ)
		
	}
\end{bt}
\begin{bt}%[Đoàn Nhật Thiện]%[0D6K3]
	[2,0 điểm]
	Chứng minh biểu thức: $A=\sin^2x+\sin^2\left(\dfrac{\pi}{3}+x\right)+\sin^2\left(\dfrac{\pi}{3}-x\right)$ không phụ thuộc vào $x$. 
	\loigiai{
		$A=\dfrac{1-\cos 2x}{2}+\dfrac{1-\cos \left(\dfrac{2\pi}{3}+2x\right)}{2}+\dfrac{1-\cos \left(\dfrac{2\pi}{3}-2x\right)}{2}$\dotfill ($0.5$đ)\\
		$=\dfrac{3}{2}- \dfrac{1}{2}\left[\cos 2x+\cos \left(\dfrac{2\pi}{3}+2x\right)+\cos \left(\dfrac{2\pi}{3}-2x\right)\right]$\dotfill ($0.5$đ)\\
		$=\dfrac{3}{2}-\dfrac{1}{2}\left(\cos 2x+2\cos \dfrac{2\pi}{3}\cos2x\right)$\dotfill ($0.5$đ)\\
		$=\dfrac{3}{2}$ (không phụ thuộc vào $x$)\dotfill ($0.5$đ)
		
	}
\end{bt}
\begin{bt}%[Đoàn Nhật Thiện]%[0D6K3]
	[2,0 điểm]
	
	Rút gọn biểu thức: $A =\dfrac{\sin 3x+\sin 2x+\sin x}{2+\cos x-2{\sin}^2x}$.
	\loigiai{
		$A = \dfrac{\sin 3x+\sin 2x+\sin x}{2+\cos x-2{\sin}^2x}$  \\
		$=\dfrac{2\sin 2x\cdot\cos x+\sin 2x}{2(1-\sin^2x)+\cos x}$  \dotfill ($0.75$đ)\\
		$=\dfrac{4\sin x\cdot\cos^2x+2\sin x.\cos x}{2\cos^2x+\cos x}$\dotfill ($0.75$đ)\\
		$=\dfrac{2\sin x\cdot\cos x(2\cos x+1)}{\cos x(2\cos x+1)}
		=2\sin x$\dotfill ($0.5$đ)\\ 
		
	}
\end{bt}
\begin{bt}%[Đoàn Nhật Thiện]%[0D6G3]
	[2,0 điểm]
	Trong tam giác $ABC$, chứng minh rằng :\\	$\sin^2A + \cos^2B + \cos^2C = 2-2\cos A\sin B\sin C$.
	\loigiai{
		$VT = \sin^2A+1+\dfrac{1}{2}\left({\cos 2B+\cos 2C}\right)$\dotfill ($0.5$đ)\\ 
		$= 2-\cos^2A-\cos A\cos(B-C)$\dotfill ($0.5$đ)\\ 
		$= 2-\cos A[\cos A + \cos(B-C)]$\\
		$=2-2\cos A\cos \dfrac{A+B-C}{2}\cos \dfrac{A-B+C}{2}$ \dotfill ($0.5$đ)\\ 
		$=2-2\cos A\sin B\sin C = VP$ (do $ A + B + C = \pi$) \dotfill ($0.5$đ)\\ 
	}
\end{bt}

	\begin{enumerate}
		\item Tính các giá trị lượng giác của góc $\alpha$.
		\item  Tính $\cos\left(25\pi-\alpha\right)$ và $\cot\left(\dfrac{5\pi}{2}+\alpha\right)$
	\end{enumerate}

			$\cos\alpha=-\sqrt{1-\left(\sin\alpha\right)^2}=-\dfrac{2\sqrt{2}}{3}$; \\
			$\tan\alpha=\dfrac{\sin\alpha}{\cos\alpha}=\dfrac{1}{3}:\left(-\dfrac{2\sqrt{2}}{3}\right)=-\dfrac{\sqrt{2}}{4}$;\\
			$\cot\alpha=\dfrac{1}{\tan\alpha}=-2\sqrt{2}$.
			\item Ta có:\\
			$\cos\left(25\pi-\alpha\right)=\cos\left(24\pi+\pi-\alpha\right)=\cos\left(\pi-\alpha\right)=-\cos\alpha=\dfrac{2\sqrt{2}}{3}$.\\
			$\cot\left(\dfrac{5\pi}{2}+\alpha\right)=\cot\left(2\pi+\dfrac{\pi}{2}+\alpha\right)=\cot\left(\dfrac{\pi}{2}+\alpha\right)=-\tan\alpha=\dfrac{\sqrt{2}}{4}$.
