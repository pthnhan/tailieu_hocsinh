\begin{bt}%[Đề HK1, Minh Đức, HCM 17-18]%[Nguyễn Khắc Hưởng]%[0D2B2-2]
	Viết phương trình đường thẳng dạng $y=ax+b$ đi qua hai điểm $A(1;2)$ và $B(3;8)$.
	\loigiai{
		Đường thẳng $y=ax+b$ đi qua $A, B$ nên có $\heva{&2=a+b\\&8=3a+b}\Leftrightarrow\heva{&a=3\\&b=-1}$.\\
		Vậy phương trình đường thẳng cần tìm là $y=3x-1$.
	}
\end{bt}
\begin{bt}%[Đề HK1, Minh Đức, HCM 17-18]%[Nguyễn Khắc Hưởng]%[0H2B3-1]
	Cho tam giác $ABC$ vuông tại $C$, có $AC=a$, $AB=3a$, đường cao $CH$. Tính độ dài đường $CH$ theo $a$.
	\loigiai{
		\begin{itemize}
		\item $\Delta ABC$ vuông tại $C$ nên có $CB^2=AB^2-AC^2=8a^2$.
		\item Có $\dfrac{1}{CH^2}=\dfrac{1}{CA^2}+\dfrac{1}{CB^2}=\dfrac{1}{a^2}+\dfrac{1}{8a^2}=\dfrac{9}{8a^2}\Rightarrow CH=\dfrac{2\sqrt{2}a}{3}$.
		\end{itemize}
	}
\end{bt}
\begin{bt}%[Đề HK1, Minh Đức, HCM 17-18]%[Nguyễn Khắc Hưởng]%[0H2B1-4]
	Trong mặt phẳng với hệ tọa độ $Oxy$, cho ba điểm $M(1;1)$, $N(-2;4)$, $P(-3;2)$. Tính góc giữa hai véc-tơ $\vec{MN}$ và $\vec{MP}$.
	\loigiai{
		\begin{itemize}
		\item Có $\vec{MN}=(-3;3)$, $\vec{MP}=(-4;1)$.
		\item $\cos(\vec{MN},\vec{MP})=\dfrac{-3\cdot(-4)+3\cdot 1}{\sqrt{(-3)^2+3^2}\cdot\sqrt{(-4)^2+1^2}}=\dfrac{5\sqrt{34}}{34}$.
		\end{itemize}
		Vậy $\left(\vec{MN},\vec{MP}\right)=\arccos\dfrac{5\sqrt{34}}{34}$ hay $\left(\vec{MN},\vec{MP}\right) \approx 30^\circ 57'$.
	}
\end{bt}
\begin{bt}%[Đề HK1, Minh Đức, HCM 17-18]%[Nguyễn Khắc Hưởng]%[0H2B2-2]

	\loigiai{
		\begin{itemize}
		\item Có $AB=\sqrt{(-3)^2+3^2}=3\sqrt{2}$, $BC=\sqrt{(-4)^2+0^2}=4$, $CA=\sqrt{7^2+(-3)^2}=\sqrt{58}$.
		\item Chu vi $\Delta ABC$ là $Q=AB+BC+CA=3\sqrt{2}+\sqrt{58}+4$.
		\end{itemize}
		}
\end{bt}
\begin{bt}%[Đề HK1, Minh Đức, HCM 17-18]%[Nguyễn Khắc Hưởng]%[0D2K3-2]

		\end{itemize}

		}

\begin{bt}%[Đề HK1, Minh Đức, HCM 17-18]%[Nguyễn Khắc Hưởng]%[0D3K2-1]
	Giải các phương trình sau:
	\begin{enumerate}
	\item $(x+4)^2-(x-3)^2=2x-5$.
	\item $(1-2x)^2-(2x-1)(x-4)=7-3x$.
	\item $\dfrac{x-3}{4}-2x+3=-\dfrac{1}{5}$.
	\item $\dfrac{2x-5}{3}-\dfrac{5x}{2}+1=-3x$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
	\item $(x+4)^2-(x-3)^2=2x-5$.\\
	Phương trình tương đương với $12x+12=0\Leftrightarrow x=-1$.\\
	Vậy phương trình có tập nghiệm là $S=\{-1\}$.
	\item $(1-2x)^2-(2x-1)(x-4)=7-3x$.\\
	Phương trình tương đương với $2x^2+8x-10=0\Leftrightarrow \hoac{&x=1\\&x=-5.}$\\
	Vậy phương trình có tập nghiệm là $S=\{1;-5\}$.
	\item $\dfrac{x-3}{4}-2x+3=-\dfrac{1}{5}$.\\
	Phương trình tương đương với $\dfrac{7(7-5x)}{20}=0\Leftrightarrow x=\dfrac{7}{5}$.\\
	Vậy phương trình có tập nghiệm là $S=\left\{\dfrac{7}{5}\right\}$.
	\item $\dfrac{2x-5}{3}-\dfrac{5x}{2}+1=-3x$.\\
	Phương trình tương đương với $\dfrac{7x-4}{6}=0\Leftrightarrow x=\dfrac{4}{7}$.\\
	Vậy phương trình có tập nghiệm là $S=\left\{\dfrac{4}{7}\right\}$.
	\end{enumerate}
		}
\end{bt}
\begin{bt}%[Đề HK1, Minh Đức, HCM 17-18]%[Nguyễn Khắc Hưởng]%[0D3K2-2]
	Giải các phương trình sau:
	\begin{enumerate}
	\item $|3x-2|=1$.
	\item $|x^2-2x+1|=4$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
	\item $|3x-2|=1$.\\
	Phương trình tương đương với $\hoac{&3x-2=1\\&3x-2=-1}\Leftrightarrow \hoac{&x=1\\&x=\dfrac{1}{3}.}$\\
	Vậy phương trình có tập nghiệm là $S=\left\{1;\dfrac{1}{3}\right\}$.
	\item $|x^2-2x+1|=4$.\\
	Phương trình tương đương với $\hoac{&x^2-2x+1=4\\&x^2-2x+1=-4}\Leftrightarrow \hoac{&x=-1\\&x=3.}$\\
	Vậy phương trình có tập nghiệm là $S=\left\{-1;3\right\}$.
	\end{enumerate}
		}
\end{bt}
\begin{bt}%[Đề HK1, Minh Đức, HCM 17-18]%[Nguyễn Khắc Hưởng]%[0D3K2-3]
	Giải các phương trình sau:
	\begin{enumerate}
	\item $\dfrac{3}{x-3}+5=\dfrac{2}{3}$.
	\item $\dfrac{x+2}{x-2}-\dfrac{x-3}{x+3}=5$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
	\item $\dfrac{3}{x-3}+5=\dfrac{2}{3}$.\\
	Điều kiện $x\neq 3$.\\
	Phương trình tương đương với $\dfrac{13x-30}{3(x-3)}=0\Leftrightarrow x=\dfrac{30}{13}$ (thỏa mãn).\\
	Vậy phương trình có tập nghiệm là $S=\left\{\dfrac{30}{13}\right\}$.
	\item $\dfrac{x+2}{x-2}-\dfrac{x-3}{x+3}=5$.\\
	Điều kiện $\heva{&x\neq 2\\&x\neq -3.}$\\
