\begin{ex}%[Đề tuyển sinh lớp 10, chuyên Đồng Tháp, 2017]%[9D4B4]%[9D4K5]
\textbf{(1,5 điểm)}
\begin{enumerate}
\item[a.] Cho biểu thức $Q=\dfrac{(x^2-x-6)(x^2+2x-3)}{(x^2-9)(x+2)^2}$ với $x\neq -2, x\neq\pm 3$. Rút gọn $Q$ và tính giá trị biểu thức $Q$ khi $x=\dfrac{3}{2}$.
\item[b.] Giải phương trình $3\sqrt{x^2-\dfrac{1}{4}+\sqrt{x^2+x+\dfrac{1}{4}}}=\dfrac{1}{2}(2x^3+x^2+2x+1)$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a.] Ta có $Q=\dfrac{(x+2)(x-3)(x-1)(x+3)}{(x-3)(x+3)(x+2)^2}=\dfrac{x-1}{x+2}$.\\
Khi $x=\dfrac{3}{2}$ thì $Q=\dfrac{\dfrac{3}{2}-1}{\dfrac{3}{2}+2}=\dfrac{1}{7}$.
\item[b.] $3\sqrt{x^2-\dfrac{1}{4}+\sqrt{x^2+x+\dfrac{1}{4}}}=\dfrac{1}{2}(2x^3+x^2+2x+1)\\
\Leftrightarrow 3\sqrt{x^2-\dfrac{1}{4}+\sqrt{\left(x+\dfrac{1}{2}\right)^2}}=\dfrac{1}{2}(x^2+1)(2x+1)
\Leftrightarrow \heva{&2x+1\geq 0\\& 3\sqrt{x^2-\dfrac{1}{4}+x+\dfrac{1}{2}}=\dfrac{1}{2}(x^2+1)(2x+1)}\\
\Leftrightarrow \heva{&x\geq -\dfrac{1}{2}\\&3\sqrt{\left(x+\dfrac{1}{2}\right)^2}=(x^2+1)\left(x+\dfrac{1}{2}\right)}\Leftrightarrow \heva{&x\geq -\dfrac{1}{2}\\&\left(x+\dfrac{1}{2}\right)(x^2+1-3)=0}\\
\Leftrightarrow\heva{&x\geq -\dfrac{1}{2}\\&\hoac{&x=-\dfrac{1}{2}\\&x=\pm\sqrt{2}}}$.\\
Vậy phương trình có hai nghiệm $x=-\dfrac{1}{2}; x=\sqrt{2}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề tuyển sinh lớp 10, chuyên Đồng Tháp, 2017]%[9D4K6]%[9D4G6]
\textbf{(1,5 điểm)}
\begin{enumerate}
\item[a.] Giải hệ phương trình: $\heva{&x+y+xy=11\\&x^2+y^2+xy=19}$.
\item[b.] Trong mặt phẳng tọa độ $Oxy$ cho Parabol $(P): y=x^2$ và đường thẳng $d:y-2m+1=0$ ($m$ là tham số). Tìm $m$ để $(P)$ cắt $d$ tại hai điểm $A$ và $B$ sao cho tam giá $AOB$ đều và tính diện tích tam giác đó.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a.] $\heva{&x+y+xy=11\\&x^2+y^2+xy=19}\Leftrightarrow\heva{&x+y+xy=11\\&(x+y)^2-xy=19}\Leftrightarrow\heva{&x+y+xy=11\\&(x+y)^2+x+y=30}\Leftrightarrow\heva{&x+y+xy=11\\&\hoac{&x+y=-6\\&x+y=5}}$.\\
Với $\heva{&x+y=-6\\&xy=17}$, hệ vô nghiệm.\\
Với $\heva{&x+y=5\\&xy=6}\Leftrightarrow\heva{&x=3\\&y=2}$ hoặc $\heva{&x=2\\&y=3}$.
\item[b.] Phương trình hoành độ giao điểm của $(P)$ và $d$ là: $x^2=2m-1$. $d$ cắt $(P)$ tại hai điểm phân biệt khi và chỉ khi $2m-1>0\Leftrightarrow m>\dfrac{1}{2}$.\\
Giả sử $A\left(-\sqrt{2m-1};2m-1\right)$ và $B\left(\sqrt{2m-1};2m-1\right)$.\\
Ta có: $OA=\sqrt{2m-1+(2m-1)^2}=\sqrt{4m^2-2m}, OB=\sqrt{4m^2-2m}, AB=2\sqrt{2m-1}$.\\
Dễ thấy $OA=OB$, nên tam giác $AOB$ đều khi $OA^2=AB^2\Leftrightarrow 4m^2-2m=4(2m-1)\Leftrightarrow 4m^2-10m+4=0\Leftrightarrow\hoac{&m=2\text{ thỏa mãn}\\&m=\dfrac{1}{2}\text{ loại}}$.\\
Vậy $m=2$ là giá trị cần tìm.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề tuyển sinh lớp 10, chuyên Đồng Tháp, 2017]%[9D4K6]%[9D5G1]
\textbf{(2,0 điểm)}
\begin{enumerate}
\item[a.] Cho phương trình $x^2-(2m+1)x-3=0$ ($m$ là tham số). Giả sử phương trình đã cho có hai nghiệm phân biệt $x_1,x_2$ với mọi $m$ và $H=x_1^2+x_2^2-6x_1x_2$. Tìm giá trị nhỏ nhất của $H$.
\item[b.] Cho ba số dương $x,y,z$ thỏa mãn $x^3+y^3+z^3=1$. Chứng minh bất đẳng thức sau:$$\dfrac{x^2}{\sqrt{1-x^2}}+\dfrac{y^2}{\sqrt{1-y^2}}+\dfrac{z^2}{\sqrt{1-z^2}}\geq 2$$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a.] Theo định lý Vi-ét ta có: $x_1+x_2=2m+1$ và $x_1x_2=-3$.\\
Từ $H=x_1^2+x_2^2-6x_1x_2=(x_1+x_2)^2-8x_1x_2=(2m+1)^2-8(-3)=(2m+1)^2+24\geq 24$.\\
Vậy $\min H=24$ khi $2m+1=0\Leftrightarrow m=-\dfrac{1}{2}$.
\item[b.] Từ giả thiết ta có $0<x,y,z<1$. Do đó, theo bất đẳng tức Cô-si ta có:\\
$\sqrt{x^2(1-x^2)}\leq \dfrac{x^2+1-x^2}{2}=\dfrac{1}{2}$.\\
Nên $\dfrac{x^2}{\sqrt{1-x^2}}=\dfrac{x^3}{\sqrt{x^2(1-x^2)}}\geq 2x^3$. Chứng minh tương tự ta cũng có:\\
$\dfrac{y^2}{\sqrt{1-y^2}}\geq 2y^3$ và $\dfrac{z^2}{\sqrt{1-z^2}}\geq 2z^3$.\\
Cộng vế theo vế ta được:\\
$\dfrac{x^2}{\sqrt{1-x^2}}+\dfrac{y^2}{\sqrt{1-y^2}}+\dfrac{z^2}{\sqrt{1-z^2}}\geq2(x^3+y^3+z^3)=2$. Ta có điều phải chứng minh.\\
Đẳng thức xảy ra khi $x=y=z=\dfrac{\sqrt{2}}{2}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề tuyển sinh lớp 10, chuyên Đồng Tháp, 2017]%[9D3G8]
Cho tam giác $ABC$ vuông tại đỉnh $A$ có phân giác trong $AM$ ($M$ thuộc $BC$) và $\widehat{ABC}=60^\circ$. Qua $M$ vẽ đường thẳng vuông góc với cạnh $BC$ cắt đoạn thẳng $AC$ tại $N$, cắt đường thẳng $AB$ tại $P$.
\begin{enumerate}
\item Chứng minh tứ giác $PAMC$ nội tiếp trong một đường tròn và tam giác $PMC$ vuông cân.
\item Gọi $O$ là tâm đường tròn ngoại tiếp tam giác $PBC, I$ là trung điểm của đoạn $PC$. Chứng minh ba điểm $M,O,I$ thẳng hàng và $MO$ song song với $BN$.
\item Chứng minh $\widehat{PNC}=\widehat{POC}$.
\item Khi $AB =3$ cm. Hãy tính diện tích tam giác $PBC$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Chứng minh tứ giác $PAMC$ nội tiếp trong một đường tròn và tam giác $PMC$ vuông cân.\\
\immini{
\textbf{Cách 1:}Xét hai tam giác $PMB$ và $CAB$ có $\widehat{PMB}=\widehat{CAB}=90^\circ$.\hfill(1)\\
Lại có $\widehat{ACB}=90^\circ-60^\circ=30^\circ$, mà $\widehat{MNC}=\widehat{ANP}=60^\circ\Rightarrow \widehat{NPA}=30^\circ=\widehat{ACB}$.\hfill(2)\\
Góc $B$ chung.\hfill(3)\\
Từ (1), (2) và (3) ta có $\Delta PMB\backsim\Delta CAB\Rightarrow \dfrac{BP}{BC}=\dfrac{BM}{BA}\Leftrightarrow BM.BC = BP.BA$.\\
Suy ra: Tứ giác $PAMC$ nội tiếp đường tròn.\\
\textbf{Cách 2:} Ta có $\widehat{PMC}=\widehat{PAC}=90^\circ$. Do đó tứ giác $PAMC$ nội tiếp đường tròn tâm $I$ là trung điểm của $PC$. Suy ra $\widehat{PCM}=\widehat{MAB}=45^\circ$.\\
Vậy tam giác $PMC$ vuông cân tại $M$.
}{
\begin{tikzpicture}[scale=.7]
\tkzInit[xmin=-3,xmax=3.5,ymin=-1,ymax=5.5]
\tkzClip
\tkzDefPoints{0/0/A,3/0/B,0/5/C}
\tkzDefLine[bisector](C,A,B)\tkzGetPoint{x}
\tkzInterLL(A,x)(B,C)\tkzGetPoint{M}
\tkzDefLine[perpendicular=through M](B,C)\tkzGetPoint{y}
\tkzInterLL(M,y)(A,B)\tkzGetPoint{P}
\tkzInterLL(M,y)(C,A)\tkzGetPoint{N}
\tkzDefMidPoint(P,C)\tkzGetPoint{I}
\tkzCircumCenter(P,B,C)\tkzGetPoint{O}
\tkzDrawPoints(A,B,C,M,N,P,I,O)
\tkzDrawSegments(P,B B,C C,A A,M M,P C,P I,M B,N)
\tkzMarkAngle[mark=|,size=.4,mkpos=.5](M,A,C)
\tkzMarkAngle[mark=|,size=.4,mkpos=.5](B,A,M)
\tkzMarkAngle[mkpos=.2,size=.4](C,B,A)
\tkzLabelAngle[pos=.6](A,B,C){\footnotesize $60^\circ$}
\tkzMarkRightAngles(C,M,P C,A,P)
\tkzLabelPoints[left](C,N,I)
\tkzLabelPoints[below](A,B,P,O)
\tkzLabelPoints[right](M)
\end{tikzpicture}
}
\item Gọi $O$ là tâm đường tròn ngoại tiếp tam giác $PBC, I$ là trung điểm của đoạn $PC$. Chứng minh ba điểm $M,O,I$ thẳng hàng và $MO$ song song với $BN$.\\
Theo câu trên $\Delta MPC$ vuông cân tại $M$ và $I$ là trung điểm $PC$ nên có ngay $MI$ là đường trung trực của $CP$. Suy ra $I,M,O$ thẳng hàng.\\
Mặt khác: $\heva{&CA\perp PB\\&PM\perp BC\\&CA\cap PM=N}\Rightarrow \heva{&BN\perp PC\\& MO\perp PC}\Rightarrow MO\parallel BN$.
\item Chứng minh $\widehat{PNC}=\widehat{POC}$.\\
Ta có:$\widehat{PCN}=15^\circ\Rightarrow \widehat{PNC}=180^\circ-(\widehat{PCN}+\widehat{CPN})=120^\circ$.\hfill(4)\\
Mặt khác, $O$ là tâm đường tròn ngoại tiếp tam giác $PBC$, nên $\widehat{PBC}=\dfrac{1}{2}\widehat{POC}\Rightarrow \widehat{POC}=2\widehat{PBC}=120^\circ$.\hfill(5)\\
Từ (4) và (5) ta có $\widehat{PNC}=\widehat{POC}$.
\item Khi $AB =3$ cm. Hãy tính diện tích tam giác $PBC$.\\
Ta có $CA=AB\tan 60^\circ=3\sqrt{3}$ và $BC=\dfrac{AB}{\cos 60^\circ}=6$.\\
Theo giả thiết thì $AM$ là đường phân giác nên\\ $\dfrac{MB}{AB}=\dfrac{MC}{AC}\Leftrightarrow \dfrac{MB}{AB}=\dfrac{BC-MB}{AC}\Leftrightarrow MB(AC+AB)=BC.AB\\
\Leftrightarrow MB=\dfrac{AB.BC}{AB+AC}=\dfrac{3.6}{3\sqrt{3}+3}=\dfrac{6}{1+\sqrt{3}}=3(\sqrt{3}-1)$.\\
Lại có $BP=\dfrac{BM}{\cos 60^\circ}=6(\sqrt{3}-1)$.\\
Vậy $S_{PBC}=\dfrac{1}{2}.3\sqrt{3}.6(\sqrt{3}-1)=9(3-\sqrt{3})$.
\end{enumerate}
}
\end{ex}
