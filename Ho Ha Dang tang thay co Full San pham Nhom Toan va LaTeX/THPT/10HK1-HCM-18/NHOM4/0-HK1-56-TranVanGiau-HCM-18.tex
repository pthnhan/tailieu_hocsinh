\begin{bt}%[Đề HK1, Trần Văn Giàu, Hồ Chí Minh 17-18][Huỳnh Thanh Tiến, HKI-10-HCM][0D2B1-2]
Tìm tập xác định của hàm số $ f(x)=\dfrac{\sqrt{x+3}-\sqrt{7-x}}{x^2-4}+\sqrt{x-1} $.
	\loigiai{
		Điều kiện xác định của hàm số:
$$ \heva{& x+3\geqslant 0\\ & 7-x\geqslant 0\\ & x^2-4\neq 0\\ & x-1\geqslant 0} 
 \Leftrightarrow 	
 \heva{& x\geqslant -3\\ & x \leqslant 7\\ & x\neq \pm 2\\ & x\geqslant 1} 
 \Leftrightarrow x\in [1;7]\setminus \{2\}. $$
	}
\end{bt}

\begin{bt}%[Đề HK1, Trần Văn Giàu, Hồ Chí Minh 17-18][Huỳnh Thanh Tiến, HKI-10-HCM][0D3K2-2][0D3K2-4] 
	Giải các phương trình sau:
	\begin{enumerate}
		\item $ \left| 3x^2-5x+1 \right| = 2x-1$.
		\item $ x^2-4x-6= \sqrt{2x^2-8x+12}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Điều kiện: $2x-1 \geq 0 \Leftrightarrow x \geq \dfrac{1}{2} $.
			Phương trình đã cho tương đương 
			$$ \hoac{& 3x^2-5x+1=2x-1 \\& 3x^2-5x+1=-2x+1} 
			\Leftrightarrow \hoac{& 3x^2-7x+2=0 \\& 3x^2-3x=0} 
			\Leftrightarrow \hoac{& x=2 \vee x=\dfrac{1}{3} \\& x=0 \vee x=1.}$$
			Kết hợp với điều kiện, nghiệm của phương trình đã cho là $x=1$ và $x=2$.
			
			\item Đặt $t=\sqrt{2x^2-8x+12},\, t \geq 0$. \\ Khi đó phương trình trở thành: 
			$t^2-2t-24=0 \Leftrightarrow \hoac{& t=6 &\text{(TMĐK)} \\& t=-4 &\text{ (loại)}.}$\\
			Tại $t=6$, ta có
			$$\sqrt{2x^2-8x+12}=6 \Leftrightarrow 2x^2-8x-24=0 \Leftrightarrow \hoac{& x=6 \\& x=-2.}$$
			Vậy phương trình đã cho có hai nghiệm là $x = 6$ và $x = -2$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề HK1, Trần Văn Giàu, Hồ Chí Minh 17-18][Huỳnh Thanh Tiến, HKI-10-HCM][0D3K2-6]
	Cho phương trình $(m-1)x^2-2mx+m+1=0$. Tìm $m$ để phương trình có hai nghiệm phân biệt $x_1, x_2$ thỏa mãn $x_2=2x_1$.
	\loigiai{Xét phương trình $(m-1)x^2-2mx+m+1=0\quad (1)$\\ 
		- Nếu $m=1\Rightarrow$ phương trình ($1$) có dạng $-2x+2=0\Leftrightarrow x=1$ (loại).\\
		- Nếu $m \neq 1$. Ta thấy $a+b+c=0$ nên ($1$) luôn có 2 nghiệm phân biệt $\hoac{&x=1\\&x=\dfrac{m+1}{m-1}.}$\\
		Theo giả thiết $x_2=2x_1$, ta có 2 trường hợp sau:
		\begin{itemize}
			\item TH1: Nếu $\dfrac{m+1}{m-1}=2\Leftrightarrow m=3$.
			\item TH2: Nếu $2\left(\dfrac{m+1}{m-1}\right)=1\Leftrightarrow m=-3$.
		\end{itemize}
		
		Vậy $m=\pm 3$ là giá trị cần tìm.}
\end{bt}

\begin{bt}%[Đề HK1, Trần Văn Giàu, Hồ Chí Minh 17-18][Huỳnh Thanh Tiến, HKI-10-HCM][0D4K1-2]
	Tìm giá trị nhỏ nhất của hàm số $y=2x+\dfrac{4}{3x-6}$, biết $x>2$.
	\loigiai{
		Ta có $y=2x+\dfrac{4}{3x-6}=2\left(x-2\right)+\dfrac{4}{3\left(x-2\right)}+4$.\\
		Mặt khác, do $x>2$ nên theo BĐT Cauchy có \begin{center}
			$2\left(x-2\right)+\dfrac{4}{3\left(x-2\right)}\ge 2\sqrt{2\left(x-2\right)\cdot \dfrac{4}{3\left(x-2\right)}}=\dfrac{4\sqrt{6}}{3}$.
		\end{center}
		Do đó: $y\ge \dfrac{4\sqrt{6}}{3}+4$.\\
		Dấu \lq\lq  = \rq\rq \
		 khi và chỉ chi $2\left(x-2\right)=\dfrac{4}{3\left(x-2\right)}\Leftrightarrow \left(x-2\right)^2=\dfrac{2}{3}\Leftrightarrow x=2+\dfrac{\sqrt{6}}{3}$ (do $x>2$).\\
	}
\end{bt}

\begin{bt}%[Đề HK1, Trần Văn Giàu, Hồ Chí Minh 17-18][Huỳnh Thanh Tiến, HKI-10-HCM][0H1B4-3]
	Cho $A(3;2)$, $B(-1;-2)$, $C(0;1)$.
	\begin{enumerate}
		\item Tính độ dài véc-tơ $\vec{v}=2\vec{BC}+\vec{CA}-3\vec{AB}$.
		\item Tìm tọa độ điểm $D$ với $D$ là điểm đối xứng của $A$ qua $B$.
		\item Xác định tọa độ trực tâm $H$ của tam giác $ ABC $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $2\vec{BC}=(2;6)$, $\vec{CA}=(3;1)$,  $-3\vec{AB}=(12;12)$ nên  $\vec{v}=2\vec{BC}+\vec{CA}-3\vec{AB}=(17;19)$.
			\item Ta có 
			$\heva{& x_D=2x_B-x_A=-2-3=-5\\ & y_D=2y_B-y_A=-4-2=-6}$ nên $D(-5;-6).$
			\item Gọi $H(a;b)$ là trực tâm tam giác $ABC$. Khi đó
			$$ \vec{AH}=(a-3;b-2),\, \vec{BH}=(a+1;b+2).$$
			Ta có
			$$ \heva{&\vec{AH}\cdot \vec{BC}=0\\ &\vec{BH}\cdot \vec{CA}=0}\Leftrightarrow \heva{&a+3b-9=0\\ &3a+b+5=0}\Leftrightarrow \heva{&a=-3\\ &b=4.} $$
			Vậy $H(-3;4)$.
		\end{enumerate}
	}
\end{bt}
