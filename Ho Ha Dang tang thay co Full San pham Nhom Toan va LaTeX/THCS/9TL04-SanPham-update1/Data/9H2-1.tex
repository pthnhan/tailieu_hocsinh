\section{Sự xác định đường tròn. Tính chất đối xứng của đường tròn}



\subsection{Tóm tắt lí thuyết}

\subsubsection{Định nghĩa đường tròn}

\begin{dn}
 \immini[0.01]{
 Đường tròn tâm $O$ bán kính $R$ (với $R>0$) là hình gồm các điểm cách đều điểm $O$ một khoảng không đổi bằng $R$.\\
 Đường tròn tâm $O$ bán kính $R$ được kí hiệu là $(O;R)$, ta cũng có thể kí hiệu là $(O)$ khi không cần chú ý đến bán kính.
 }
 {
 \begin{tikzpicture}
  \tkzDefPoints{0/0/O, 1.5/0/M}
  \tkzDrawPoints[fill=black](O)
  \tkzDrawCircle[radius](O,M)
  \tkzDrawSegments(O,M)
  \tkzLabelPoints[below](O)
  \tkzLabelLine[above](O,M){$R$}
 \end{tikzpicture}
 }
\end{dn}
\immini{{\bf Nhận xét.} Cho đường tròn $(O;R)$ và một điểm $M$. Khi đó
\begin{itemize}
 \item $M$ nằm trên $(O;R)$ khi và chỉ khi $OM=R$.
 \item $M$ nằm bên trong $(O;R)$ khi và chỉ khi $OM<R$.
 \item $M$ nằm bên ngoài $(O;R)$ khi và chỉ khi $OM>R$.
\end{itemize}
}
{
\begin{tikzpicture}[scale=0.9]
 \def\a{2}
 \tkzDefPoints{0/0/O,\a/0/A}
 \draw (A) arc(0:360:\a cm and \a cm);

 \coordinate(M_2) at ([shift={(O)}]220:1.5 cm and 1.5 cm);
 \coordinate(M_3) at ([shift={(O)}]330:2.5 cm and 2.5 cm);
 \tkzDrawPoints[fill=black](O, M_1, M_2, M_3)
 \tkzDrawSegments(O,A O,M_1 O,M_2 O,M_3)
 \tkzLabelPoints[above](M_1)
 \tkzLabelPoints[below](O,M_3)
 \tkzLabelPoints[below right](M_2)
 \tkzLabelLine[above](O,A){$R$}
\end{tikzpicture}
}
\subsubsection{Cách xác định đường tròn}
\begin{enumerate}
 \item Một đường tròn được xác định khi biết tâm và bán kính của nó.
 \item Một đường tròn được xác định khi biết một đoạn thẳng là đường kính của đường tròn đó.
 \item Qua ba điểm không thẳng hàng cho trước ta vẽ được một và chỉ một đường tròn.
\end{enumerate}

\subsubsection{Tính chất đối xứng của đường tròn}
\begin{tc}
 Đường tròn là hình có tâm đối xứng. Tâm của đường tròn là tâm đối xứng của đường tròn đó.
\end{tc}

\begin{tc}
 Đường tròn là hình có trục đối xứng. Bất kỳ đường kính nào cũng là trục đối xứng của đường tròn.
\end{tc}

\subsection{Các ví dụ}

\begin{vd}
\label{9H2-1.vd.1}
 Cho tam giác $ABC$ vuông tại $A$. Xác định tâm và bán kính đường tròn đi qua ba đỉnh của tam giác $ABC$.
 \loigiai
  {
  \immini{
  Gọi $M$ là trung điểm của $BC$.\\
  Ta có $AM$ là trung tuyến ứng với cạnh huyền nên $AM=\dfrac{BC}{2}$.\\
  Suy ra $MA=MB=MC=\dfrac{BC}{2}$.\\
  Vậy đường tròn đi qua ba đỉnh của tam giác $ABC$ có tâm là điểm $M$ và bán kính $R=\dfrac{BC}{2}$.
  }
  {
  \begin{tikzpicture}[scale=0.8]
   \tkzDefPoints{0/0/B, 3/0/C}
   \tkzDefShiftPoint[B](50:1){b}
   \tkzDefShiftPoint[C](-40:1){c}
   \tkzInterLL(B,b)(C,c)\tkzGetPoint{A}
   \tkzDefMidPoint(B,C)\tkzGetPoint{M}
   \tkzDrawPoints[fill=black](A,B,C,M)
   \tkzDrawSegments(A,B B,C C,A A,M)
   \tkzDrawCircle(M,A)
   \tkzMarkSegments[mark=|,size=0.1cm](B,M M,C)
   \tkzMarkRightAngles(B,A,C)
   \tkzLabelPoints[above](A)
   \tkzLabelPoints[below](M)
   \tkzLabelPoints[left](B)
   \tkzLabelPoints[right](C)
  \end{tikzpicture}
  }
  }
\end{vd}

\begin{vd}
 Chứng minh rằng, nếu một tam giác có một cạnh là đường kính của đường tròn đi qua ba đỉnh của tam giác đó thì tam giác đó là tam giác vuông.
 \loigiai
  {
  \immini{
  Xét tam giác $ABC$ có ba đỉnh nằm trên đường tròn $(O)$ đường kính $BC$.\\
  Ta có $OA=OB=OC$ (vì là bán kính của $(O)$).\\
  Lúc đó $AO$ là trung tuyến ứng với cạnh $BC$ và $AO=\dfrac{BC}{2}$.\\
  Vậy $ABC$ là tam giác vuông tại $A$.
  }
  {
  \begin{tikzpicture}[scale=0.8]
   \tkzDefPoints{0/0/B, 3/0/C}
   \tkzDefShiftPoint[B](50:1){b}
   \tkzDefShiftPoint[C](-40:1){c}
   \tkzInterLL(B,b)(C,c)\tkzGetPoint{A}
   \tkzDefMidPoint(B,C)\tkzGetPoint{O}
   \tkzDrawPoints[fill=black](A,B,C,O)
   \tkzDrawSegments(A,B B,C C,A A,O)
   \tkzDrawCircle(O,A)
   \tkzMarkSegments[mark=|,size=0.1cm](B,O O,C O,A)
   \tkzLabelPoints[above](A)
   \tkzLabelPoints[below](O)
   \tkzLabelPoints[left](B)
   \tkzLabelPoints[right](C)
  \end{tikzpicture}
  }
  \begin{note}
   Đường tròn qua ba đỉnh của một tam giác vuông thì nó có tâm là trung điểm của cạnh huyền và bán kính bằng phân nửa độ dài cạnh huyền. Ngược lại, một đường tròn đi qua ba đỉnh của một tam giác nhận một cạnh của tam giác đó là đường kính thì tam giác đó là tam giác vuông.
  \end{note}
  }
\end{vd}

\begin{vd}
 Cho tam giác đều $ABC$ có cạnh bằng $a$. Tính bán kính đường tròn đi qua ba đỉnh của tam giác $ABC$.
 \loigiai
  {
  \immini{
  Gọi $M$, $N$, $P$ lần lượt là trung điểm của $BC$, $CA$, $AB$.\\
  Dựng các đường trung trực của các cạnh $AB$, $BC$, $CA$, các đường trung trực này đồng quy tại $O$, suy ra $O$ là tâm của đường tròn đi qua ba đỉnh của tam giác $ABC$. Bán kính của đường tròn $(O)$ là $R=OA=OB=OC$.\\
  Vì $ABC$ là tam giác đều nên các đường trung trực này cũng là các đường trung tuyến của tam giác $ABC$. Suy ra $O$ cũng là trọng tâm của tam giác $ABC$.
  }
  {
  \begin{tikzpicture}[scale=0.8]
   \tkzInit[xmin=-0.6,xmax=3.6,ymin=-1,ymax=3.2]
   \tkzClip
   \tkzDefPoints{0/0/B, 3/0/C}
   \tkzCalcLength[cm](B,C)\tkzGetLength{r}
   \tkzDefShiftPoint[B](60:\r){A}
   \tkzDefMidPoint(B,C)\tkzGetPoint{M}
   \tkzDefMidPoint(C,A)\tkzGetPoint{N}
   \tkzDefMidPoint(A,B)\tkzGetPoint{P}
   \tkzDefCircle[circum](A,B,C)\tkzGetPoint{O}
   \tkzDrawPoints[fill=black](A,B,C,M,N,P,O)
   \tkzDrawPolygon(A,B,C)
   \tkzDrawCircle(O,A)
   \tkzDrawSegments(O,N O,P A,M)
   \tkzMarkRightAngles[size=0.15](A,M,B C,N,O A,P,O)
   \tkzLabelPoints[above](A)
   \tkzLabelPoints[below](M)
   \tkzLabelPoints[left](B,P)
   \tkzLabelPoints[right](C,N)
   \tkzLabelPoints[below right](O)
  \end{tikzpicture}
  }
  \noindent Trong tam giác $ABM$ vuông tại $M$ ta có $AM=\sqrt{AB^2 - BM^2} = \sqrt{a^2 - \left(\dfrac{a}{2}\right)^2} = \dfrac{a\sqrt{3}}{2}$.\\
  Lại có $OA=\dfrac{2}{3}AM = \dfrac{2}{3} \cdot \dfrac{a\sqrt{3}}{2} = \dfrac{a\sqrt{3}}{3}$.\\
  Vậy bán kính đường tròn đi qua ba đỉnh của tam giác $ABC$ là $R=\dfrac{a\sqrt{3}}{3}$.
  }
\end{vd}

\begin{vd}
 \label{9H2-1.vd.2}
 Cho hình chữ nhật $ABCD$ có $AB=12$ cm, $BC=5$ cm. Chứng minh rằng bốn điểm $A$, $B$, $C$, $D$ cùng thuộc một đường tròn. Xác định tâm và bán kính của đường tròn đó.
 \loigiai
  {
  \immini{
  Gọi $O$ là giao điểm $AC$ và $BD$. Khi đó $O$ là trung điểm của $AC$, $BD$.\\
  Mà $ABCD$ là hình chữ nhật nên $AC=BD$.\\
  Do đó $OA=OB=OC=OD$ hay bốn điểm $A$, $B$, $C$, $D$ cùng thuộc một đường tròn $(O)$, bán kính $R=OA=\dfrac{AC}{2}$.\\
  Tam giác $ABC$ vuông tại $B$ nên $AC=\sqrt{AB^2+BC^2} = \sqrt{12^2+5^2} = 13$.
  }
  {
  \begin{tikzpicture}[scale=0.8]
   \tkzDefPoints{0/0/A, 3/0/B}
   \tkzDefShiftPoint[B](90:1.5){C}
   \tkzDefPointBy[translation=from B to A](C)\tkzGetPoint{D}
   \tkzDefMidPoint(A,C)\tkzGetPoint{O}
   \tkzDrawPoints[fill=black](A,B,C,D,O)
   \tkzDrawPolygon(A,B,C,D)
   \tkzDrawSegments(A,C B,D)
   \tkzDrawCircle(O,A)
   \tkzMarkRightAngles[size=0.15](A,B,C A,D,C)
   \tkzLabelPoints[below](O)
   \tkzLabelPoints[left](A,D)
   \tkzLabelPoints[right](B,C)
  \end{tikzpicture}
  }
  \noindent Suy ra $R=\dfrac{AC}{2} = 6,5$ cm.\\
  Vậy bốn điểm $A$, $B$, $C$, $D$ cùng thuộc một đường tròn $(O)$ bán kính $R=6,5$ cm.
  \begin{note}
   Đường tròn qua bốn đỉnh của hình chữ nhật $ABCD$ có tâm là giao điểm của hai đường chéo và bán kính của nó bằng một nửa độ dài đường chéo của hình chữ nhật đó.
  \end{note}
  }
\end{vd}

\begin{vd}

  Lại có $AC \perp BD$.\\
  Vậy $ABCD$ là hình chữ nhật có hai đường chéo vuông góc với nhau nên $ABCD$ là hình vuông.

\begin{vd}
 Cho hình thang cân $ABCD$ với $AB \parallel CD$ và $AB > CD$. Chứng minh rằng bốn điểm $A$, $B$, $C$, $D$ cùng thuộc một đường tròn.
 \loigiai
  {
  \immini{
  Gọi $M$, $N$ lần lượt là trung điểm của $AB$, $CD$.\\
  Do $ABCD$ là hình thang cân với hai đáy $AB$, $CD$ nên $MN$ đường trung trực của $AB$, $CD$.\\
  Gọi $P$ là trung điểm của $BC$. Qua $P$ dựng đường trung trực của $BC$ cắt $MN$ tại $O$. Ta cần chứng minh $OA=OB=OC=OD$.\\
  Thật vậy, vì $O$ nằm trên đường trung trực của $AB$ nên $OA=OB$.\\
  Mà $MN$ cũng là trung trực của $CD$ nên $OC=OD$.\\
  Hơn nữa, $O$ nằm trên đường trung trực của $BC$ nên $OB=OC$.
  }

  \noindent Từ đó suy ra $OA=OB=OC=OD$.\\
  Vậy bốn điểm $A$, $B$, $C$, $D$ cùng thuộc một đường tròn $(O)$ bán kính $R=OA$.
  }
\end{vd}

\begin{vd}
 Trên mặt phẳng tọa độ $Oxy$, hãy xác định vị trí của mỗi điểm $A(-1;-1)$, $B(-1;-2)$, $C\left(\sqrt{2};\sqrt{2}\right)$ đối với đường tròn tâm $O$ bán kính $2$.
 \loigiai
  {
  \immini{
  \begin{itemize}
   \item $OA$ là cạnh huyền trong tam giác vuông cân cạnh bằng $1$ nên $OA=\sqrt{1^2+1^2}=\sqrt{2}<2$, suy ra $A$ nằm bên trong đường tròn $(O;2)$.
   \item $OB$ là cạnh huyền trong tam giác vuông có hai cạnh góc vuông là $1$; $2$ nên $OB=\sqrt{1^2+2^2}=\sqrt{5}>2$, suy ra $B$ nằm bên ngoài đường tròn $(O;2)$.
   \item $OC$ là cạnh huyền trong tam giác vuông cân cạnh bằng $\sqrt{2}$ nên $OC=\sqrt{\sqrt{2}^2+\sqrt{2}^2}=2$, suy ra $C$ nằm trên đường tròn $(O;2)$.
  \end{itemize}
  }
  {
  \begin{tikzpicture}[line join=round, line cap=round, font=\footnotesize,>=stealth, scale=0.8]
   \draw[->](-3,0)--(3,0) node[below right] {$x$};

\begin{bt}
 Cho tam giác $ABC$ có $H$ là trực tâm. Lấy $M$, $N$ thuộc tia $BC$ sao cho $MN=BC$ và $M$ nằm giữa $B$, $C$. Gọi $D$ là hình chiếu của $M$ lên $AC$ và $E$ là hình chiếu của $N$ lên $AB$. Chứng minh rằng các điểm $A$, $D$, $E$, $H$ cùng thuộc một đường tròn.
 \loigiai

  {
