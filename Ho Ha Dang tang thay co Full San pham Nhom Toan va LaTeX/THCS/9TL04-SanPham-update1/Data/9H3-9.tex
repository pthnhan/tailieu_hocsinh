\section{Độ dài đường tròn, cung tròn}
\subsection{Tóm tắt lý thuyết}
\subsubsection{Độ dài đường tròn}
\immini
{\begin{enumerate}
		\item Độ dài đường tròn (hay ``Chu vi hình tròn'') được kí hiệu là $\mathrm{C}$.
		\item Độ dài $\mathrm{C}$ của một đường tròn bán kình $R$ được tính theo công thức
		\begin{center}
			$\mathrm{C}=2\pi R$.
		\end{center}
		\item Nếu gọi $\mathrm{d}$ là đường kính của đường tròn ($\mathrm{d}=2R$) thì
		\begin{center}
			$\mathrm{C}=\pi \mathrm{d}$.
		\end{center}
	\end{enumerate}
}
{
	\begin{tikzpicture}[>=stealth,scale=1]
	\tkzDefPoints{0/0/A, 5/0/B}
	\draw[-,dashed] (0,-2.5)--(0,2.5);
	\draw[-,dashed] (5,-2.5)--(5,2.5);
	\draw[<->,dashed] (0,1)--(5,1) node at (3.3,1.2) {$d$};
	\tkzDrawSegments(A,B)
	\tkzDefMidPoint(A,B)\tkzGetPoint{O}
	\tkzLabelSegments[color=black,pos=.5](O,B){$R$}
	\tkzDrawCircle[color=black](O,B)
	\tkzDrawPoints[size=3,fill=black](A) \tkzLabelPoints[left](A)
	\tkzDrawPoints[size=3,fill=black](B) \tkzLabelPoints[right](B)
	\tkzDrawPoints[size=3,fill=black](O) \tkzLabelPoints[below left](O)
	\end{tikzpicture}
}
\bigskip
Trong đó $\pi$ đọc là ``pi'' và $\pi \approx 3{,}14$ ($\pi = 3{,}14159265\ldots$).

\subsubsection{Độ dài cung tròn}
\begin{enumerate}
	\item Đường tròn bán kính $R$ (ứng với cung $360^\circ$) có độ dài là $2\pi R$.
	\item Mỗi cung $1^\circ$ bán kính $R$ có độ dài là
	\begin{center}
		$\dfrac{2\pi R}{360}=\dfrac{\pi R}{180}$.
	\end{center}
\item Một cung $n^\circ$, bán kính $R$ có độ dài
\begin{center}
	$l=\dfrac{\pi R n}{180}$.
\end{center}
\end{enumerate}

		\item Tính độ dài cung $30^\circ$ của một đường tròn có bán kính $5$ cm.
		\item Tính chu vi của một vành xe đạp có đường kính $65$ cm.

	\end{center}

\begin{bt}%
Đường kính bánh xe của một chiếc xe đạp là $73$ cm. Hỏi
\begin{enumerate}
	\item Bánh xe đó quay được bao nhiêu vòng khi xe đi được một đoạn đường $8$ km?
	\item Xe đi được bao nhiêu ki-lô-mét nếu bánh xe quay đủ $1000$ vòng?
\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Bánh xe đạp quay được một vòng nghĩa là nó đi được một độ dài là chu vi của bánh xe.\\
		Chu vi của bánh xe đạp: $C=\pi d=73\pi$ (cm).\\
		Số vòng mà bánh xe quay được khi đi đoạn đường dài $8$ km $=800000$ cm là
		\begin{center}
			$800000 : (73\pi)\approx 3488$ (vòng).
		\end{center}
		\item Đoạn đường mà xe đi được khi bánh xe quay đủ $1000$ vòng là
		\begin{center}
			$1000\cdot 73\pi\approx 229336{,}3$ (cm).
		\end{center}
	Đổi $229336{,}3$ (cm) $\approx2{,}29$ (km).\\
	Vậy xe đi được khoảng $2{,}29$ km khi bánh xe quay đủ $1000$ vòng.
	\end{enumerate}	
	}
\end{bt}

\begin{bt}%

			\item Chu vi bánh xe trước: $C_1=\pi d=0{,}8\pi$ (m).

		\end{itemize}

		\begin{center}

		\end{center}

	}

	C_a=3\cdot \dfrac{\pi\cdot 2}{2}=3\pi \text{ (cm)}.\tag{$1$}
\end{align}
Tương tự $C_b$ là độ dài của hai nửa đường tròn đường kính $3$ cm nên
\begin{align}
	C_b=\pi\cdot 3=3\pi \text{ (cm)}.\tag{$2$}
\end{align}
$C_c$ là độ dài của một nửa đường tròn đường kính $6$ cm nên
\begin{align}
	C_c=\dfrac{\pi\cdot 6}{2}=3\pi \text{ (cm)}.\tag{$3$}
\end{align}
Từ $(1)$, $(2)$ và $(3)$ suy ra $C_a=C_b=C_c$.\\
