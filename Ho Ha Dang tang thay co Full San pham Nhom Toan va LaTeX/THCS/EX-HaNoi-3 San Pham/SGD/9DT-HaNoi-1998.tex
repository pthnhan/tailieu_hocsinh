\begin{ex}%[Đề Lớp 9 SGD Hà Nội - 1998\_1999]%[9D1K7]%[9D5G2]
	Cho biểu thức  $P=\left(\dfrac{2x+1}{\sqrt{x^3}-1}-\dfrac{1}{\sqrt{x}-1}\right):\left( 1-\dfrac{x+4}{x+\sqrt{x}+1}\right)$.
	\begin{enumerate}
		\item 	Rút gọn $P$.
		\item	Tìm giá trị nguyên của $x$ để $P$ nhận giá trị nguyên dương.
	\end{enumerate}
\loigiai{\begin{enumerate}
		\item \begin{eqnarray*}
		P&=&\left(\dfrac{2x+1}{\sqrt{x^3}-1}-\dfrac{1}{\sqrt{x}-1} \right):\left( 1-\dfrac{x+4}{x+\sqrt{x}+1} \right)\\
		&=&\left(\dfrac{2x+1}{(\sqrt{x}-1)(x+\sqrt{x}+1)}-\dfrac{1}{\sqrt{x}-1} \right):\dfrac{x+\sqrt{x}+1-x-4}{x+\sqrt{x}+1}\\
		&=&\dfrac{2x+1-x-\sqrt{x}-1}{(\sqrt{x}-1)(x+\sqrt{x}+1)}\cdot\dfrac{x+\sqrt{x}+1}{\sqrt{x}-3}\\
		&=&\dfrac{x-\sqrt{x}}{(\sqrt{x}-1)(x+\sqrt{x}+1)}\cdot\dfrac{x+\sqrt{x}+1}{\sqrt{x}-3}\\
		&=&\dfrac{\sqrt{x}(\sqrt{x}-1)}{(\sqrt{x}-1)(\sqrt{x}-3)}=\dfrac{\sqrt{x}}{\sqrt{x}-3}.
		\end{eqnarray*}
		\item Có $P=\dfrac{\sqrt{x}}{\sqrt{x}-3}=1+\dfrac{3}{\sqrt{x}-3}$.\\
		Để $P$ nhận giá trị nguyên dương thì $\sqrt{x}-3$ phải là ước của $3$\\
		$\Rightarrow\hoac{&\sqrt{x}-3=3\\&\sqrt{x}-3=1\\&\sqrt{x}-3=-1\\&\sqrt{x}-3=-3}\Leftrightarrow\hoac{&\sqrt{x}=6\\&\sqrt{x}=4\\&\sqrt{x}=2\\&\sqrt{x}=0}\Leftrightarrow\hoac{&x=36\\&x=16\\&x=4\\&x=0.}$
\end{enumerate}}	
\end{ex} 

\begin{ex}%[Đề Lớp 9 SGD Hà Nội - 1998\_1999]%[9D4K8]
	\textit{Giải bài toán sau bằng cách lập phương trình:}\\
	Một người dự định đi xe đạp từ A đến B cách nhau $36$ km trong thời gian nhất định. Sau khi đi được nửa quãng đường người đó dừng lại nghỉ 18 phút. Do đó để đến B đúng hẹn người đó đã tăng vận tốc thêm $2$ km/h trên quãng đường còn lại. Tính vận tốc ban đầu và thời gian xe lăn bánh trên đường.
	\loigiai{Gọi $x$ km/h là vận tốc ban đầu của người đó, suy ra vận tốc lúc sau là $(x+2)$ km/h.\\
	Thời gian người đó đi với vận tốc ban đầu trên nửa đoạn đường sau là $\dfrac{18}{x}$.\\
Thời gian người đó phải di chuyển khi đã tăng tốc là $\dfrac{18}{x+2}$.\\
Do người đó đến B đúng hẹn nên $\dfrac{18}{x}=\dfrac{18}{60}+\dfrac{18}{x+2}\Leftrightarrow18\left(\dfrac{1}{x}-\dfrac{1}{x+2}\right)=\dfrac{3}{10}$\\
$\Leftrightarrow320=x^2+2x\Leftrightarrow x^2+2x-120=0\Leftrightarrow\hoac{&x=10\\&x=-12.}$\\
Vậy vận tốc ban đầu của người đó là $x=10$ km/h.\\
Tổng thời gian xe lăn bánh là $T=\dfrac{18}{x}+\dfrac{18}{x+2}=3{,}3$ h.
}
\end{ex}  

\begin{ex}%[Đề Lớp 9 SGD Hà Nội - 1998\_1999]%[9H3B3]%[9H2K1]
	Cho tam giác $ABC$ vuông tại $A$, đường cao $AH$. Đường tròn đường kính $AH$ cắt các cạnh $AB$, $AC$ lần lượt tại $E$ và $F$.
	\begin{enumerate}
		\item 	Chứng minh rằng tứ giác $AEHF$ là hình chữ nhật.
		\item 	Chứng minh $AE \cdot AB=AF \cdot AC$.
		\item	Đường thẳng qua $A$ vuông góc với $EF$ cắt cạnh $BC$ tại $I$. Chứng minh $I$ là trung điểm của $BC$. 
		\item 	Chứng minh nếu diện tích tam giác $ABC$ gấp đôi diện tích hình chữ nhật $AEHF$ thì tam giác $ABC$ vuông cân.
	\end{enumerate}
\loigiai{\begin{center}
		\begin{tikzpicture}
		\tkzDefPoints{0/0/A, 3/0/C, 0/5/B}
		\tkzDefPointBy[projection = onto B--C](A)	\tkzGetPoint{H}
		\coordinate (J) at ($(A)!0.5!(H)$);
		\coordinate (I) at ($(B)!0.5!(C)$);
		\tkzDrawCircle(J,A)
		\tkzInterLC(A,B)(J,A)	\tkzGetSecondPoint{E}
		\tkzInterLC(A,C)(J,A)	\tkzGetFirstPoint{F}
		\tkzInterLL(E,F)(A,I)	\tkzGetPoint{T}
		\tkzDrawSegments(A,B A,C A,H B,C E,F A,I E,H H,F)
		\tkzDrawPoints(A,B,C,I,H,E,F)
		\tkzLabelPoints[below left](A)
		\tkzLabelPoints[below right](C,F)
		\tkzLabelPoints[above left](B,E)
		\tkzLabelPoints[above right](H,I)
		\tkzMarkRightAngle(E,T,A)
		\tkzMarkRightAngle(B,H,A)
		\end{tikzpicture}
	\end{center}
	\begin{enumerate}
			\item Do $E$ và $F$ thuộc đường tròn đường kính $AH$ nên $\widehat{AEH}=\widehat{AFH}=90^\circ$.\\
			Mà $\widehat{EAF}=90^\circ$ nên tứ giác $AEHF$ là hình chữ nhật (tứ giác có ba góc vuông).
			\item Tam giác vuông $ABH$ có $HE$ là đường cao nên $AE\cdot AB=AH^2$.\\
			Tam giác vuông $ACH$ có $HF$ là đường cao nên $AF\cdot AC=AH^2$.\\
			Vậy $AE\cdot AB=AF\cdot AC$.
			\item Ta có $AE\cdot AB=AF\cdot AC\Rightarrow\dfrac{AE}{AF}=\dfrac{AC}{AB}$\\
			Mà góc $A$ chung nên $\triangle AEF\backsim\triangle ACB\Rightarrow\widehat{AFE}=\widehat{ABI}$.\\
			Mà $\widehat{AFE}=\widehat{BAI}$ (cùng phụ với góc $\widehat{AEF}$)\\
			nên $\widehat{BAI}=\widehat{ABI}$, suy ra $\triangle IAB$ cân tại $I$, vậy $IA=IB$.\\
			Chứng minh tương tự ta có $IA=IC$, nên $I$ là trung điểm của $BC$
			\item Giả sử $S_{ABC}=2S_{AEHF}\Rightarrow\dfrac{S_{AEF}}{S_{ACB}}=\dfrac{1}{4}=\dfrac{EF}{BC}\Rightarrow EF=\dfrac{1}{2}BC$.\\
			Mà ta có $AI=\dfrac{1}{2}BC$ và $EF=AH$, nên $AH=AI$. Suy ra $H\equiv I$.\\
			Vậy $\triangle ABC$ vuông cân tại $A$.
	\end{enumerate}}
\end{ex}
