\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Marie Curie, Hà Nội]%[Bùi Mạnh Tiến]%[9D1B8]%[9D1K8]
	Cho biểu thức $A=\left(\dfrac{1}{x-\sqrt{x}}+\dfrac{1}{\sqrt{x}-1}\right):\dfrac{\sqrt{x}+1}{\left(\sqrt{x}-1\right)^2}$
	\begin{enumerate}[a)]
		\item Rút gọn $A$.
		\item Tính giá trị của $x$ để $A=\dfrac{1}{3}$.
		\item Tính giá trị của biểu thức $A$ với $x=3+2\sqrt{2}$.
		\item Tìm giá trị lớn nhất của biểu thức $P=A-9\sqrt{x}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Điều kiện xác định : $\heva{&x>0\\&x\ne 1.}$
			\begin{align*}
				A=\left(\dfrac{1}{x-\sqrt{x}}+\dfrac{1}{\sqrt{x}-1}\right):\dfrac{\sqrt{x}+1}{\left(\sqrt{x}-1\right)^2}&=\left(\dfrac{1}{\sqrt{x}(\sqrt{x}-1)}+\dfrac{1}{\sqrt{x}-1}\right)\cdot \dfrac{\left(\sqrt{x}-1\right)^2}{\sqrt{x}+1}\\
				&=\dfrac{1+\sqrt{x}}{\sqrt{x}\left(\sqrt{x}+1\right)}\cdot \dfrac{\left(\sqrt{x}-1\right)^2}{\sqrt{x}+1}\\
				&=\dfrac{\sqrt{x}-1}{\sqrt{x}}.
			\end{align*}
			\item 
			\[A=\dfrac{1}{3}\Leftrightarrow \dfrac{\sqrt{x}-1}{\sqrt{x}}=\dfrac{1}{3}\Leftrightarrow 3\sqrt{x}-3=\sqrt{x}\Leftrightarrow \sqrt{x}=\dfrac{3}{2}\Leftrightarrow x=\dfrac{9}{4}.\]
			\item 
				\begin{align*}
					A&=\dfrac{\sqrt{x}-1}{\sqrt{x}}=1-\dfrac{1}{\sqrt{x}}=1-\dfrac{1}{\sqrt{3+2\sqrt{2}}}=1-\dfrac{1}{\sqrt{\left(\sqrt{2}+1\right)^2}}\\
					&=1-\dfrac{1}{\sqrt{2}+1}=\dfrac{\sqrt{2}}{\sqrt{2}+1}=\sqrt{2}\left(\sqrt{2}-1\right)=2-\sqrt{2}.
				\end{align*}
			\item 
			\begin{center}
				$P=A-9\sqrt{x}=1-\dfrac{1}{\sqrt{x}}-9\sqrt{x}=1-\left(9\sqrt{x}+\dfrac{1}{\sqrt{x}}\right)\le 1-2\sqrt{9\sqrt{x}\cdot \dfrac{1}{\sqrt{x}}}=-5$.
			\end{center}
		Vậy giá trị lớn nhất của $A$ là $-5$. \\
		Dấu bằng xảy ra khi và chỉ khi $9\sqrt{x}=\dfrac{1}{\sqrt{x}}\Leftrightarrow x=\dfrac{1}{9}$.
		\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Marie Curie, Hà Nội]%[Bùi Mạnh Tiến]%[9D4B4]%[9D4K6]
	Cho phương trình bậc hai ẩn $x$ với tham số $m$:
	\[x^2-\left(m+1\right)x+2m-1=0. \tag{1}\]
	\begin{enumerate}
		\item Giải phương trình khi $m=-2$.
		\item Tìm biểu thức liên hệ giữa hai nghiệm $x_1;x_2$ của phương trình $(1)$ không phụ thuộc vào $m$.
		\item Tìm $m$ để nghiệm của phương trình thỏa mãn biểu thức $x_1^2+x_2^2+3x_1x_2=12$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item 
			Khi $m=-2$ thì $(1)\Leftrightarrow x^2+x-5=0$.\\
			$\Delta = 1^2-4\cdot 1\cdot (-5)=21>0$.\\
			Phương trình có hai nghiệm $\hoac{&x_1=\dfrac{-1+\sqrt{21}}{2}\\&x_2=\dfrac{-1-\sqrt{21}}{2}.}$
			\item 
				$\Delta = \left[-\left(m+1\right)\right]^2-4\cdot 1\cdot (2m-1)=m^2+2m+1-8m+4=m^2-6m+5$.\\
				Để phương trình có hai nghiệm thì $m^2-6m+5>0.$\quad(*)\\
				Với $m$ thỏa mãn điều kiện (*) thì theo định lý Vi-ét ta có 
				\begin{center}
					$\heva{&x_1+x_2=m+1\\&x_1x_2=2m-1}\Rightarrow \heva{&x_1+x_2=m+1\\&x_1x_2=2(x_1+x_2)-3.}$
				\end{center}
			Vậy hệ thức giữa $x_1,x_2$ không phụ thuộc vào $m$ là $x_1x_2=2(x_1+x_2)-3.$
			\item 
			Từ ý b) ta có
			\begin{eqnarray*}
				x_1^2+x_2^2+3x_1x_2=12&\Leftrightarrow& (x_1+x_2)^2+x_1x_2=12\Leftrightarrow (m+1)^2+2m-1-12=0\\
				&\Leftrightarrow& m^2+4m-12=0\Leftrightarrow \hoac{&m=-6\\&m=2.}
			\end{eqnarray*}
			Thử $m=-6$ và $m=2$ vào điều kiện (*) ta thấy chỉ có $m=-6$ thỏa mãn.
		\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Marie Curie, Hà Nội]%[Bùi Mạnh Tiến]%[9D4K8]
	Một tổ công nhân được phân công may $100$ chiếc áo. Khi bắt đầu công việc thì một người phải điều đi làm công việc khác, nên để may xong số áo đó đúng tiến độ thì mỗi người còn lại phải may thêm $5$ áo so với dự định. Hỏi lúc đầu tổ công nhân có bao nhiêu người.
	\loigiai
	{
		Gọi $x$ là số công nhân của tổ (Điều kiện $x\in \mathbb{N^*}$).\\
		Ban đầu mỗi người cần may $\dfrac{100}{x}$(chiếc áo).\\
		Khi một người bị điều đi làm công việc khác thì mỗi người còn lại phải may $\dfrac{100}{x-1}$(chiếc áo).\\
		Do mỗi người phải may thêm $5$ chiếc nên 
		\begin{eqnarray*}
			&&\dfrac{100}{x-1}-\dfrac{100}{x}=5
			\Leftrightarrow \dfrac{20}{x-1}-\dfrac{20}{x}=1\\
			&\Leftrightarrow &\dfrac{20x-20(x-1)}{x\left(x-1\right)}=1
			\Leftrightarrow 20=x(x-1)\\ 
			&\Leftrightarrow &x^2-x-20=0
			\Leftrightarrow \hoac{&x=5\\&x=-4\text{ (loại).}}
		\end{eqnarray*}
		Vậy số công nhân của tổ là $5$ người.
	}

\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Marie Curie, Hà Nội]\%[9H3B7]%[9H3K7]%[Bùi Mạnh Tiến]%[9H3K6]
	Cho hình vuông $ABCD$, điểm $E$ thuộc cạnh $BC$. Qua $B$ kẻ đường thẳng vuông góc với đường thẳng $DE$ tại $H$, cắt đường thẳng $DC$ ở $K$.
	\begin{enumerate}
		\item Chứng minh rằng $BHCD$ là tứ giác nội tiếp.
		\item Tính số đo $\widehat{CHK}$.
		\item Chứng minh hệ thức $KC\cdot KD=KH\cdot KB$.
		\item Khi $E$ di chuyển trên cạnh $BC$ thì $H$ di chuyển trên đường nào?
	\end{enumerate}
	\loigiai
	{
		\immini{
			a) Theo giả thiết ta có $\widehat{BHC}=\widehat{BCD}=90^\circ$ (cùng chắn cung $BD$)
			Suy ra $BHCD$ là tứ giác nội tiếp.\\
			b) Vì $ABCD$ là hình vuông nên $\triangle BCD$ vuông cân tại $C\Rightarrow \widehat{CBD}=45^\circ$.\\
			Vì $BHCD$ là tứ giác nội tiếp nên $\widehat{CHD}=\widehat{CBD}=45^\circ$ (cùng chắn cung CD)\\
			c) Xét $\triangle KCB$ và $\triangle KHD$ có
			\begin{center}
				$\heva{&\text{Góc K chung}\\&\widehat{KHD}=\widehat{KCB}=90^\circ}\Rightarrow \triangle KCB\backsim \triangle KHD.$
			\end{center}
		}
		{
			\begin{tikzpicture}
				\tkzDefPoints{0/4/A, 0/0/B, 4/0/C, 4/4/D}
				\tkzDefPointWith[linear,K=0.3](C,B)\tkzGetPoint{E}
				\tkzDefPointBy[projection=onto D--E](B)\tkzGetPoint{H};
				\tkzInterLL(B,H)(D,C)\tkzGetPoint{K}
				\tkzDefMidPoint(C,A) \tkzGetPoint{O}
				\tkzDrawSegments(A,B B,C C,D D,A D,E B,H B,K E,H C,K H,C B,D A,C)
				\tkzDrawCircle(O,B)
				\tkzDrawPoints(A,B,C,D,E,H,K,O)
				\tkzMarkRightAngles(D,H,B B,C,D);
				\draw (A) node[left] {$A$} (B) node[left] {$B$} (C) node[right] {$C$} (D) node[right] {$D$} (H) node[below] {$H$} (K) node[right] {$K$} (E) node[above left] {$E$} (O) node[below] {$O$};
			\end{tikzpicture}
		}
	\flushleft
	Suy ra $\dfrac{KC}{KH}=\dfrac{KB}{KD}\Rightarrow KC\cdot KD=KH\cdot KB$.\\
	d) Vì $\widehat{BHD}=90^\circ$ nên $H$ nằm trên đường tròn đường kính $BD$.\\
	Khi $E\equiv B\Rightarrow H\equiv B$. Khi $E\equiv C\Rightarrow H\equiv C$.\\
	Vậy điểm $H$ nằm trên cung tròn $BC$ của đường tròn đường kính $BD$.
	}
	
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Marie Curie, Hà Nội]%[Bùi Mạnh Tiến]%[9D4G7]
	Giải phương trình $\sqrt{5x^2+10x+1}=7-x^2-2x$.
	\loigiai
	{
		Điều kiện : $5x^2+10x+1\ge 0 \quad(*)$.\\
		Đặt $t=\sqrt{5x^2+10x+1}$ $(t\ge 0)$.\\
		Ta có $t^2=5x^2+10x+1\Rightarrow x^2+2x=\dfrac{t^2-1}{5}$.\\
		Phương trình đã cho trở thành $t=7-\dfrac{t^2-1}{5}\Leftrightarrow t^2+5t-36=0\Leftrightarrow \hoac{&t=4\\&t=-9\text{ (loại)}.}$\\
		Với $t=4\Leftrightarrow 5x^2+10x+1=16\Leftrightarrow x^2+2x-3=0\Leftrightarrow \hoac{&x=1\\&x=-3}$ thỏa mãn $(*)$.\\
		Vậy phương trình có hai nghiệm $x=1$ và $x=-3$.
	}
\end{ex}
