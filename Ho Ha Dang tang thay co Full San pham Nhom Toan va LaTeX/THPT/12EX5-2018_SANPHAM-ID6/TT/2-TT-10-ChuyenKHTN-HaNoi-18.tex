	Kết hợp với điều kiện của $m$ ở trên ta được $m=7$ (do $m$ nguyên).
	}
\end{ex}

\begin{ex}%[Thi thử lần 1 THPT chuyên KHTN - 2018]%[2D2G4-4]
	Cho các số $a,b>1$ thỏa mãn $\log_2 a+\log_3 b=1$. Tìm giá trị lớn nhất của $P=\sqrt{\log_3 a}+\sqrt{\log_2 b}$.
	\choice
	{\True $\sqrt{\log_2 3+\log_3 2}$}
	{$\sqrt{\log_3 2}+\sqrt{\log_2 3}$}
	{$\dfrac{1}{2}\left(\log_2 3+\log_3 2\right)$}
	{$\dfrac{2}{\sqrt{\log_2 3+\log_3 2}}$}
	\loigiai{
	Ta có $P=\sqrt{\log_3 a}+\sqrt{\log_2 b}=\sqrt{\log_3 2}\sqrt{\log_2 a}+\sqrt{\log_2 3}\sqrt{\log_3 b}$. Áp dụng bất đẳng thức Cauchy-Schwarz ta có $P^2\leq (\log_3 2+\log_2 3)(\log_2 a+\log_3 b)=\log_3 2+\log_2 3$. Suy ra $P\leq \sqrt{\log_2 3+\log_3 2}$.
	}
\end{ex}

\begin{ex}%[Thi thử lần 1 THPT chuyên KHTN - 2018]%[1D5K2-5]
	Viết phương trình tiếp tuyến của đồ thị hàm số $y=\dfrac{x+2}{2x+3}$ biết tiếp tuyến đó cắt trục tung và trục hoành tại hai điểm phân biệt $A,B$ sao cho tam giác $OAB$ cân.
	\choice
	{\True $y=-x-2$}
	{$y=x+2$}
	{$y=x-2$}
	{$y=-x+2$}
	\loigiai{
	Từ giả thiết suy ra tiếp tuyến có hệ số góc bằng $1$ hoặc $-1$.\\
	Ta có $y'=-\dfrac{1}{(2x+3)^2}<0$ nên $y'=-1\Leftrightarrow x=-1$ hoặc $x=-2$.\\
	Tại $x=-1$ ta có phương trình tiếp tuyến là $y=-x$ (loại vì đường thẳng này đi qua $O$).\\
	Tại $x=-2$ ta có phương trình tiếp tuyến là $y=-x-2$.
	}

\begin{ex}%[Thi thử lần 1 THPT chuyên KHTN - 2018]%[2D3G3-1]
\immini{
	Cho hàm số $y=ax^4+bx^2+c$ có đồ thị $(C)$, biết rằng $(C)$ đi qua điểm $A(-1;0)$, tiếp tuyến $d$ tại $A$ của $(C)$ cắt $(C)$ tại hai điểm có hoành độ lần lượt là $0$ và $2$. Diện tích hình phẳng giới hạn bởi $d$, đồ thị $(C)$ và hai đường thẳng $x=0$, $x=2$ có diện tích bằng $\dfrac{28}{5}$ (phần gạch chéo trong hình vẽ). Tính diện tích giới hạn bởi $d$, đồ thị $(C)$ và hai đường thẳng $x=-1$, $x=0$.
		\choice
		{$\dfrac{2}{5}$}
		{$\dfrac{1}{4}$}
		{$\dfrac{2}{9}$}
		{\True $\dfrac{1}{5}$}}
	{\begin{tikzpicture}[thick,>=stealth,scale=0.6] 
			\draw[->] (-3,0) -- (3.5,0) node[above] {\small $x$};
			\draw[->] (0,-1.3) -- (0,7.4) node[right] {\small $y$};
		 	\draw[thick,smooth,samples=100,domain=-2.05:2.05] plot(\x,{(\x)^4-3*(\x)^2+2});
			\draw[thick,smooth,samples=100,domain=0:2, name path=A] plot(\x,{(\x)^4-3*(\x)^2+2});
			\draw[thick,smooth,samples=100,domain=-1.5:2.6] plot(\x,{2*(\x)+2});		
			\draw[thick,smooth,samples=100,domain=0:2,name path=B] plot(\x,{2*(\x)+2});	
			\draw[fill=white] (-1,0) node [below]{\footnotesize $-1$} circle (0.07);
			\draw[fill=white] (0,0) node [below right]{\footnotesize $O$} circle (0.07);			
			\draw (2,0) node [below]{\footnotesize $2$}circle (0.07);
			\draw[fill=white] (0,2) circle (0.07);			
			\draw[fill=white] (2,6) circle (0.07);	
			\draw[dashed] (2,0)--(2,6);
			\tikzfillbetween[of=A and B,split]{pattern=north west lines};
			\end{tikzpicture}}
	\loigiai{
	Ta có $y'=4ax^3+2bx$. \\
	Phương trình tiếp tuyến $d$ tại $A(-1;0)$ là $d: y=y'(-1)(x+1)+0=(-4a-2b)(x+1)$.\\
	Phương trình hoành độ giao điểm của $d$ và $(C)$ là $(-4a-2b)(x+1)=ax^4+bx^2+c$.\\
	Theo giả thiết, $x=0$ và $x=2$ là hai nghiệm của phương trình này, lần lượt thay $x=0$ và $x=2$ vào ta được 
	$$\heva{& -4a-2b=c\\ & -12a -6b=16a+4b+c}\Leftrightarrow \heva{& 4a+2b+c=0 & (1)\\ & 28a +10b+c=0 & (2)}$$

	\begin{alignat*}{3}
	&\dfrac{28}{5}&=&\displaystyle\int_{0}^{2}\left[(-4a-2b)(x+1)-(ax^4+bx^2+c)\right]\mathrm{\,d}x\\
	&&=& \left[(-4a-2b)\left(\dfrac{x^2}{2}+x\right)-\left(\dfrac{ax^5}{5}+\dfrac{bx^3}{3}+cx\right)\right]\Big\vert^2_0\\
	&&=& (-4a-2b)\cdot 4 - \left(\dfrac{32}{5}a+\dfrac{8}{3}b+2c\right)
	\end{alignat*}
