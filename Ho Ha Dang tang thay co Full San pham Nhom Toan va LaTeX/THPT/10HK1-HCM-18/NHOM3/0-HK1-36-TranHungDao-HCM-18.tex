\begin{bt}%[Đề thi HK1, 2017 - 2018 trường THPT Trần Hưng Đạo, Hồ Chí Minh]%[Hữu Bình]%[0D3B2-6]
	Tìm tham số $m$ để phương trình $(m^2-m)x=2x+m^2-1$ nghiệm đúng với $\forall x\in \mathbb{R}$.
	\loigiai{
		$$(m^2-m)x=2x+m^2-1 \Leftrightarrow (m^2-m-2)x=m^2-1.$$
		Phương trình đã cho nghiệm đúng với $\forall x\in \mathbb{R}$ khi và chỉ khi
		$$\heva{& m^2-m-2=0 \\ &m^2-1=0} \Leftrightarrow \heva{& m=2\vee m=-1 \\
			& m=1\vee m=-1 } \Leftrightarrow m=-1.$$
	Vậy $m=-1$.	
	}
\end{bt}

\begin{bt}%[Đề thi HK1, 2017 - 2018 trường THPT Trần Hưng Đạo, Hồ Chí Minh]%[Hữu Bình]%[0D2B3-2]
	Xác định $a,b,c$ biết parapol $y=ax^2+bx+c$ đi qua điểm $D(3;4)$ và có đỉnh là $I(1;-8)$.
	\loigiai{
		Theo giả thiết ta có
		$$ \left\{\begin{aligned}& 9a+3b+c=4 \\
		& a+b+c=-8\\
		& \dfrac{-b}{2a}=1 
		\end{aligned}\right.\Leftrightarrow \left\{\begin{aligned}& a=3 \\
		& b=-6 \\
		& c=-5.
		\end{aligned}\right.$$
		Vậy $a=3, b=-6, c=-5$.
	}
\end{bt}

\begin{bt}%[Đề thi HK1, 2017 - 2018 trường THPT Trần Hưng Đạo, Hồ Chí Minh]%[Hữu Bình]%[0D3K2-5]
	Tìm các giá trị của tham số $m$ để phương trình $mx^2-\left(2m+3\right)x+m=0$ có hai nghiệm phân biệt $x_1,x_2$ sao cho $\left|x_1-x_2\right|=3$.
	\loigiai{
	Phương trình	$mx^2-\left({2m+3}\right)x+m=0$ có 2 nghiệm phân biệt $x_1,x_2$$\Leftrightarrow \left\{\begin{aligned}& a\ne 0 \\
		& \Delta '>0 
		\end{aligned}\right.\Leftrightarrow \left\{\begin{aligned}& m\ne 0 \\
		& m>-\dfrac{3}{4}.
		\end{aligned}\right.$ \\
		Khi đó $S=x_1+x_2=\dfrac{2m+3}{m}$, $P=x_1x_2=1$.\\
		$\left|{x_1-x_2}\right|=3 \Leftrightarrow \left({x_1+x_2}\right)^2-4x_1x_2=9 \Leftrightarrow \left({\dfrac{2m+3}{m}}\right)^2-4=9\Leftrightarrow 3m^2-4m-3=0\Leftrightarrow m=\dfrac{2\pm \sqrt{{13}}}{3}$.\\
		Đối chiếu điều kiện, ta được $m=\dfrac{2\pm \sqrt{{13}}}{3}$.
	}
\end{bt}

\begin{bt}%[Đề thi HK1, 2017 - 2018 trường THPT Trần Hưng Đạo, Hồ Chí Minh]%[Hữu Bình]%[0D3K2-2]
	Giải các phương trình $\left|{x^2-x+2}\right|=3x-1$.
	\loigiai{
		\begin{align*}
		&\left|{x^2-x+2}\right|=3x-1 \Leftrightarrow \left\{\begin{aligned}& 3x-1\geqslant 0 \\
		& x^2-x+2=3x-1\vee x^2-x+2=-3x+1 
		\end{aligned}\right.\\
		&\Leftrightarrow \left\{\begin{aligned}& 3x-1\geqslant 0 \\
		& x=-1\vee x=1\vee x=3 
		\end{aligned}\right.\Leftrightarrow x=1\vee x=3 .		
		\end{align*}
	Vậy nghiệm của phương trình là $x=1, x=3$.
	}
\end{bt}

\begin{bt}%[Đề thi HK1, 2017 - 2018 trường THPT Trần Hưng Đạo, Hồ Chí Minh]%[Hữu Bình]%[0D3K2-4]
	Giải các phương trình $(2x+2)(x-2)-\sqrt{{x^2-x+1}}+5=0$.
	
	\loigiai{
		$$(2x+2)(x-2)-\sqrt{{x^2-x+1}}+5=0 \Leftrightarrow 2x^2-2x+1 - \sqrt{{x^2-x+1}}=0 $$
		Đặt $t=\sqrt{{x^2-x+1}}$ $(t\geqslant 0)$.	Phương trình đã cho trở thành
		$$2t^2-t-1=0 \Leftrightarrow \left[\begin{aligned}& t=1 \\
		& t=-\dfrac{1}{2} \text{ (loại). }
		\end{aligned}\right.$$
		Với $t=1$, ta có $x^2-x=0$$\Leftrightarrow x=0\vee x=1$.}
\end{bt}

\begin{bt}%[Đề thi HK1, 2017 - 2018 trường THPT Trần Hưng Đạo, Hồ Chí Minh]%[Hữu Bình]%[0D3B3-5]
	Một nhà máy đã sử dụng hết $130$ kg đồng và $80$ kg chì để sản xuất hai loại thiết bị điện loại A và loại B. Biết sản xuất mỗi thiết bị điện loại A cần $3$ kg đồng và $2$ kg chì, sản xuất mỗi thiết bị điện loại B cần $2$ kg đồng và $1$ kg chì. Tính số thiết bị điện mỗi loại mà nhà máy đã sản xuất.
	\loigiai{
		Gọi $x, y$ lần lượt là số thiết bị điện loại A, loại B. Điều kiện $x\ge 0, y\ge 0$.  
		Ta có hệ
		$$\left\{\begin{aligned}& 3x+2y=130 \\
		& 2x+y=80 
		\end{aligned}\right.\Leftrightarrow \left\{\begin{aligned}& x=30 \\
		& y=20.
		\end{aligned}\right.$$  
		Vậy nhà máy sản xuất được 30 thiết bị điện loại A và 20 thiết bị điện loại B. 
	}
\end{bt}
\begin{bt}%[Đề thi HK1, 2017 - 2018 trường THPT Trần Hưng Đạo, Hồ Chí Minh]%[Hữu Bình]%[0H1B4-3]
	Trong mặt phẳng $Oxy$, cho cho ba điểm $A(0;-2)$, $B(5;0)$, $C(3;5) $.
	\begin{enumerate}
		\item Tìm tọa độ điểm $E$ để $ABEC$ là hình bình hành.
		\item Tìm tọa độ điểm  $N\in Oy$  để ba điểm $A,C,N$ tạo thành tam giác vuông tại $N$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Gọi  $E(x;y)$ . Ta có  $\overrightarrow{AC}(3;7)$, $\overrightarrow{AB}(5;2)$, $\overrightarrow{CE}(x-3;y-5)$.\\
			Vì  $3.\cdot2\ne 5\cdot7$   nên $ \overrightarrow{AC},\overrightarrow{AB}$  không cùng phương.\\ 
			$ABEC$  là hình bình hành $\Leftrightarrow \overrightarrow{AB}=\overrightarrow{CE} \Leftrightarrow \left\{\begin{aligned}& x-3=5 \\& y-5=2 \end{aligned}\right.\Leftrightarrow \left\{\begin{aligned}& x=8 \\& y=7. \end{aligned}\right.$\\
			Vậy $E(7;8)$.
			\item Gọi  $N(0;y)\in Oy$ . Ta có  $\overrightarrow{AN}(0;y+2)$, $\overrightarrow{CN}(-3;y-5)$.\\ 
			$A,C,N$ tạo thành tam giác vuông tại $N$ $\Leftrightarrow A\not\equiv N$ và $\overrightarrow{AN}\perp \overrightarrow{CN}$ hay\\
			$$ \left\{\begin{aligned}& y\ne -2 \\& (y+2)(y-5)=0 \end{aligned}\right. \Leftrightarrow \left\{\begin{aligned}& y\ne -2 \\& y=-2\vee y=5 \end{aligned}\right. \Leftrightarrow y=5$$.\\
			Vậy $N(0;5)$. 
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề thi HK1, 2017 - 2018 trường THPT Trần Hưng Đạo, Hồ Chí Minh]%[Hữu Bình]%[0H2K2-1][0H1B3-4]
	Cho tam giác  $ABC$  có  $AB=5$ , $AC=6$ và  $\widehat{BAC}={60}^\circ$ .
	Gọi  $M$ là điểm thỏa  $\overrightarrow{BM}=2\overrightarrow{MC}$.
	\begin{enumerate}
		\item Tính  $\overrightarrow{AB}\cdot\overrightarrow{AC}$.
		\item  Phân tích véc-tơ  $\overrightarrow{AM}$  theo  $\overrightarrow{AB}$  và  $\overrightarrow{AC}$ .
		\item Tính  $\overrightarrow{AM}\cdot\overrightarrow{BC}$.
	\end{enumerate} 
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=0.7] 
			\tkzInit[ymin=-1,ymax=6,xmin=-1,xmax=6] \tkzClip
			\tkzDefPoints{0/0/A,5/0/B,6/0/D}
			\tkzDefPointBy[rotation = center A angle 60](D) \tkzGetPoint{C}
			\tkzLabelAngle[pos=1.4](B,A,C){$60^\circ$}
			\tkzMarkAngle[size=0.8cm](B,A,C)
			\coordinate (M) at ($(B)!0.6667!(C)$);
			\tkzDrawPoints(A,B,C,M)
			\tkzLabelPoints[right](B,C,M)
			\tkzLabelPoints[left](A)
			\tkzDrawSegments(A,B B,C C,A A,M)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item $\overrightarrow{AB}\cdot\overrightarrow{AC}=AB\cdot AC\cdot \cos \left(\overrightarrow{AB},\overrightarrow{AC}\right)=5\cdot 6\cdot \cos 60{}^\circ=15$.
			\item $\overrightarrow{BM}=2\overrightarrow{MC}$  
			$\Leftrightarrow \overrightarrow{AM}-\overrightarrow{AB}=2\left(\overrightarrow{AC}-\overrightarrow{AM}\right)  \Leftrightarrow \overrightarrow{AM}=\dfrac{1}{3}\overrightarrow{AB}+\dfrac{2}{3}\overrightarrow{AC}$.
			\item   $\overrightarrow{AM}\cdot \overrightarrow{BC}=\left(\dfrac{1}{3}\overrightarrow{AB}+\dfrac{2}{3}\overrightarrow{AC}\right)\left(\overrightarrow{AC}-\overrightarrow{AB}\right)=-\dfrac{1}{3}A{B}^{2}-\dfrac{1}{3}\overrightarrow{AB}\cdot \overrightarrow{AC}+\dfrac{2}{3}A{C}^{2}=-\dfrac{25}{3}-\dfrac{15}{3}+\dfrac{72}{3}=\dfrac{32}{3}$.
		\end{enumerate}
	}
\end{bt}

	Cho tam giác  $PQR$  vuông tại  $P$ , có  $PR=4$ . Tính  $\overrightarrow{RP}\cdot \overrightarrow{RQ}$.
	\loigiai{
		Tam giác  $PQR$  vuông tại  $P$  $\Rightarrow \cos \widehat{PRQ}=\dfrac{PR}{QR}$.\\ 
		Ta có $\overrightarrow{RP}\cdot\overrightarrow{RQ}=RP.RQ\cdot\cos \left(\overrightarrow{RP},\overrightarrow{RQ}\right)=PR\cdot QR\cdot \dfrac{PR}{QR}$  $=P{R}^{2}=16$.
	}
\end{bt}
