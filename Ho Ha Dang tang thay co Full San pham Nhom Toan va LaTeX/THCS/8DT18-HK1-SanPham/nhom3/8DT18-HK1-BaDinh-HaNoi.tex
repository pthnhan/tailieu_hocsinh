\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Ba Đình, Hà Nội]%[Nguyễn Việt Hưng, dự án 8DT18-HK1]%[8D2B6]%[8D2B9]%[8D2K9]
	Cho biểu thức $A=\dfrac{5}{x+3}-\dfrac{2}{3-x}-\dfrac{3x^2-2x-9}{x^2-9}$ với $x\neq-3$ và $x\neq3$.
	\begin{enumerate}
		\item Rút gọn biểu thức $A$.
		\item Tính giá trị của $A$ khi $|x-2|=1$.
		\item Tìm giá trị nguyên của $x$ để $A$ có giá trị nguyên.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Với $x\ne \pm 3$, ta có
		\begin{eqnarray*}
			A&= & \dfrac{5}{x+3}-\dfrac{2}{3-x}-\dfrac{3x^2-2x-9}{x^2-9}\\
			&= & \dfrac{5(x-3)+2(x+3)-3x^2+2x+9}{(x+3)(x-3)}\\
			&= & \dfrac{-3x^2+9x}{(x+3)(x-3)}\\
			&=& \dfrac{-3x(x-3)}{(x+3)(x-3)}\\
			&=& \dfrac{-3x}{x+3}.
		\end{eqnarray*}
		\item $|x-2|=1\Leftrightarrow \hoac{& x-2=1\\ & x-2=-1}\Leftrightarrow \hoac{& x=3\text{ (không thỏa mãn)}\\ & x=1 \text{ (thỏa mãn).}}$\\
		Thay $x=1$ vào biểu thức, ta được $A=\dfrac{-3}{4}$.
		\item Ta có $\dfrac{-3x}{x+3}=\dfrac{-3(x+3)+9}{x+3}=-3+\dfrac{9}{x+3}$.\\
		Với giá trị nguyên của $x$ thì $A$ có giá trị nguyên khi và chỉ khi $x+3\in\text{Ư}(9)=\{-9;-3;-1;1;3;9\}$.\\
		Khi đó, ta có bảng sau
		\begin{longtable}{|p{1.5 cm}|p{1.5 cm}|p{1.5 cm}|p{1.5 cm}|p{1.5 cm}|p{1.5 cm}|p{1.5 cm}|}
			\hline
			\centerline {$x+3$} & \centerline{$-9$} & \centerline{$-3$} & \centerline{$-1$} & \centerline{$1$} & \centerline{$3$} & \centerline{$9$}\\
			\hline
			\centerline{$x$} & \centerline{$-12$} & \centerline{$-6$} & \centerline{$-4$} & \centerline{$-2$} & \centerline{$0$} & \centerline{$6$}\\
			\hline
		\end{longtable}
		Các giá trị này đều thỏa mãn điều kiện xác định nên $A$ có giá trị nguyên khi $$x\in\{-12;-6;-4;-2;0;6\}.$$
	\end{enumerate}	
	}
\end{bt}

	Cho tam giác $ABC$ vuông tại $A$, $M$ là trung điểm của $AC$. Gọi $D$ là điểm đối xứng với $B$ qua $M$.
	\begin{enumerate}
		\item Chứng minh tứ giác $ABCD$ là hình bình hành.
		\item Gọi $N$ là điểm đối xứng với $B$ qua $A$. Chứng minh tứ giác $ACDN$ là hình chữ nhật.
		\item Kéo dài $MN$ cắt $BC$ tại $I$. Vẽ đường thẳng qua $A$ song song với $MN$ cắt $BC$ ở $K$. Chứng minh $KC=2BK$.
		\item Qua $B$ kẻ đường thẳng song song với $MN$ cắt $AC$ tại $E$. Tam giác $ABC$ cần có thêm điều kiện gì để tứ giác $EBMN$ là hình vuông.
	\end{enumerate}
	\loigiai{
		\vspace*{-1cm}
		\immini
		{
			\begin{enumerate}
			\item Vì $D$ đối xứng với $B$ qua $M$ nên $MB=MD$.\\
			Xét tứ giác $ABCD$ có $M$ là trung điểm của $BD$ và $M$ là trung điểm của $AC$ nên $ABCD$ có hai đường chéo cắt nhau tại trung điểm mỗi đường. Suy ra $ABCD$ là hình bình hành (dấu hiệu nhận biết).
			\item Vì $ABCD$ là hình bình hành nên $AB\parallel CD$ và $AB=CD$ (tính chất hình bình hành).\\
			Do $N$ là điểm đối xứng với $B$ qua $A$ nên $AB=AN$. Suy ra $AN=CD$.\\
			Xét tứ giác $ACDN$ có $AN=CD$ và $AN\parallel CD$ nên $ACDN$ là hình bình hành (dấu hiệu nhận biết).\\
			Mà $\widehat{CAN}=90^\circ$, suy ra $ACDN$ là hình chữ nhật (dấu hiệu nhận biết).
			\end{enumerate}
		}

		}	
		\begin{enumerate}
			\item[c)] Xét tam giác $AKC$ có $M$ là trung điểm $AC$ và $MI\parallel AK$ nên $I$ là trung điểm $CK$ hay $CI=IK$.\\
			Xét tam giác $ABI$ có $A$ là trung điểm $BN$ và $AK\parallel NI$ nên $K$ là trung điểm của $BI$ hay $IK=KB$.\\
			Vậy $CI=IK=KB$, suy ra $KC=2BK$.
			\item[d)] Vì $EB\parallel MN$ nên $\widehat{EBA}=\widehat{ANM}$ (hai góc so le trong).\\
			Vậy $\triangle AEB=\triangle AMN$ (g-c-g), suy ra $EB=MN$.\\
			Xét tứ giác $MNEB$ có $EB\parallel MN$ và $EB=MN$ nên $MNEB$ là hình bình hành.\\
			Mà $ME\perp NB$ nên $MNEB$ là hình thoi.\\
			Do đó, hình thoi $MNEB$ là hình vuông khi và chỉ khi $AM=AB$ hay tam giác $ABC$ có thêm tính chất $AC=2AB$.
		\end{enumerate}

	Cho $a$ thỏa mãn $a^2-5a+2=0$. Tính giá trị của biểu thức
	$$P=a^5-a^4-18a^3+9a^2-5a+2017+(a^4-40a^2+4):a^2.$$
	\loigiai{
	Ta có
	\begin{eqnarray*}
		a^5-a^4-18a^3+9a^2-5a&= & a^5-5a^4+2a^3+4a^4-20a^3+8a^2+a^2-5a+2-2\\
		&= & a^3(a^2-5a+2)+4a^2(a^2-5a+2)+a^2-5a+2-2\\
		&= & (a^2-5a+2)(a^3+4a^2+1)-2\\
		&=& 0-2\\
		&=& -2.
	\end{eqnarray*}
	Vì $a^2-5a+2=0$ nên $a^2+2=5a\Rightarrow (a^2+2)^2=(5a)^2$.\\
	$\Rightarrow a^4+4a^2+4=25a^2\Rightarrow a^4-21a^2+4=0\Rightarrow a^4-40a^2+4=-19a^2$.\\
	Vậy $P=-2+2017-19a^2:a^2=2015-19=1996$.
	}
\end{bt}
