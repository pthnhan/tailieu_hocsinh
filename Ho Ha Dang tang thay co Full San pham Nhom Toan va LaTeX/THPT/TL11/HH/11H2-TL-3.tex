\section{ĐƯỜNG THẲNG VÀ MẶT PHẲNG SONG SONG}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Vị trí tương đối của đường thẳng và mặt phẳng}
Đường thẳng $d$ và mặt phẳng $(\alpha)$ có ba vị trí tương đối:
\begin{itemize}
	\item \textit{$d$ và $(\alpha)$ không có điểm chung}. 
	\begin{center}
		\begin{tikzpicture}
		\tkzDefPoints{0/0/A, 5/0/B, 1/2/C, 0.5/2.7/E, 6/2.7/F}
		\tkzDefPointBy[translation=from A to B](C)\tkzGetPoint{D}
		\tkzDrawSegments(A,B A,C C,D D,B E,F)
		\tkzMarkAngle[size=0.6](B,A,C)
		\draw ($(A)!0.08!(C)$)node[right]{$\alpha$};
		\draw($(F)!0.1!(E)$)node[above]{$d$};
		\end{tikzpicture}\hspace*{0.2cm}
	\end{center}
	Khi đó ta nói $d$ \textit{song song} với  $(\alpha)$ hay $(\alpha)$ \textit{song song} với $d$ và kí hiệu là $d \parallel (\alpha)$ hay $(\alpha) \parallel d$.
	\item \textit{$d$ và $(\alpha)$ có một điểm chung duy nhất $M$}. 
	\begin{center}
		\begin{tikzpicture}%1H23-2.39 SGK/tr60
		\tkzDefPoints{0/0/A, 5/0/B, 1/2/C, 2/3/E, 4/-1/F}
		\tkzDefPointBy[translation=from A to B](C)\tkzGetPoint{D}
		\tkzMarkAngle[size=0.6](B,A,C)
		\draw ($(A)!0.08!(C)$)node[right]{$\alpha$};
		\tkzInterLL(E,F)(A,D)\tkzGetPoint{M}
		\tkzInterLL(M,F)(A,B)\tkzGetPoint{N}
		\tkzDrawSegments(A,B B,D D,C C,A E,M N,F)
		\tkzDrawSegments[dashed](M,N)
		\tkzLabelPoints[above right](M)
		\tkzDrawPoints(M)
		\draw($(E)!0.3!(M)$)node[above right]{$d$};
		\end{tikzpicture}
	\end{center}
	Khi đó ta nói $d$ và $(\alpha)$ \textit{cắt nhau} tại điểm $M$ và kí hiệu là $d \cap (\alpha) = \{M\}$ hay $d \cap (\alpha) = M$.
	\item \textit{$d$ và $(\alpha)$ có từ hai điểm chung trở lên.} 
	\begin{center}
		\begin{tikzpicture}
		\tkzDefPoints{0/0/A, 5/0/B, 1/2/C, 1/0.6/E, 5/1.5/F}
		\tkzDefPointBy[translation= from A to B](C)\tkzGetPoint{D}
		\tkzMarkAngle[size=0.6](B,A,C)
		\draw ($(A)!0.08!(C)$)node[right]{$\alpha$};
		\coordinate (M) at ($(E)!0.2!(F)$);
		\coordinate (N) at ($(E)!0.4!(F)$);
		\coordinate (P) at ($(E)!0.6!(F)$);
		\coordinate (Q) at ($(E)!0.8!(F)$);
		\draw($(E)!0.9!(F)$)node[above]{$d$};
		\tkzDrawPoints(M, N, P, Q)
		\tkzDrawSegments(A,B B,D D,C C,A E,F)
		\end{tikzpicture}
	\end{center}
	Khi đó ta nói $d$ nằm trong $(\alpha)$ hay $(\alpha)$ chứa $d$  và kí hiệu là $d \subset (\alpha)$ hay $(\alpha) \supset d$.
	\end{itemize}
\subsubsection{Tính chất}
\begin{dl}
	Nếu đường thẳng $d$ không nằm trong mặt phẳng $(\alpha)$ và $d$ song song với đường thẳng $d'$ nằm trong mặt phẳng $(\alpha)$ thì $d$ song song với $(\alpha)$. 
\end{dl}
\begin{note}
	Tóm tắt định lí
	\immini
	{\vspace*{0.5cm}
		$\heva{&d \not \subset (\alpha) \\&d \parallel d'\\&d' \subset (\alpha)} \Rightarrow d \parallel (\alpha). $	
	}

	\tkzDrawLine[add = 0.2 and 0.2](M,H)

	\tkzMarkAngles[size=0.7,](B,A,D)
	\tkzLabelAngle[pos=0.4](B,A,D){ $\alpha$}
	\end{tikzpicture}
	}
\end{note}
\begin{dl}
	Cho đường thẳng $a$ song song với mặt phẳng $(\alpha)$. Nếu mặt phẳng $(\beta)$ chứa $a$ và cắt $(\alpha)$ theo giao tuyến $b$ thì $b$ song song với $a$. 
\end{dl}
\begin{note}
	Tóm tắt định lí
	\immini
	{\vspace*{0.5cm}
		$\heva{&a \parallel (\alpha) \\&a \subset (\beta)\\&(\alpha) \cap (\beta)=b} \Rightarrow a \parallel b.$

	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng]%[1H2K3]

\begin{bt}%[Đinh Thanh Hoàng]%[1H2B3]%[1H2B3]%[1H2K3]
	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình bình hành. Trên các cạnh $SA$, $SB$, $AD$ lần lượt lấy các điểm $M$, $N$, $K$ sao cho $\dfrac{SM}{SA}=\dfrac{SN}{SB}=\dfrac{DK}{DA}$. Chứng minh rằng:
	\begin{enumerate}
		\item Đường thẳng $MN$ song song với mặt phẳng $(ABCD)$.
		\item Đường thẳng $SD$ song song với mặt phẳng $(MNK)$.
