\section{PHƯƠNG TRÌNH TÍCH}
\subsection{Trọng tâm kiến thức}
\begin{dn}
	Phương trình tích là phương trình có dạng $A(x)\cdot B(x)\cdots C(x) = 0$, trong đó $A(x), B(x), C(x)$ là những biếu thức của biến $x$.
	\end{dn}
\begin{note}
 Các bước giải một phương đưa được vè phương trình tích.\\
	Bước 1. Chuyển tất cả các hạng tử sang vế trái ròi phân tích vế này thành nhân tử, đưa phương trình về dạng $A(x) \cdots B(x) \cdots C(x) = 0.$\\
	Bước 2. Giải các phương trình $A(x) = 0$, $B(x) = 0$, $\ldots C(x) = 0$ rồi lấy tất cả các nghiệm của chúng.
\end{note}

\subsection{Các dạng toán}
\begin{dang}{Giải các phương trình tích}
Phương pháp giải:\\
Giải phương trình $A(x) = 0$ và $B(x) = 0$ rồi lấy tất cả các nghiệm.
\end{dang}
\begin{vd}%[8D3B4]
%vd 1
Giải các phương trình sau:\\
	\begin{listEX}[2]
		\item $(x + 5)(2x - 4) = 0;$
		\item $(x^2 + 1)(5x + 3) = 0;$
		\item $\left(\dfrac{3}{4}x - 1\right)\left(\dfrac{5}{3}x + 2\right) = 0$.
\end{listEX}
\loigiai{
\begin{listEX}[2]
%	\allowdisplaybreaks
	\item 	$\begin{aligned}[t]
	 &(x + 5)(2x - 4) = 0  \\
	 \Leftrightarrow &\hoac{&x + 5 = 0\\&2x - 4 = 0}\\
	\Leftrightarrow & \hoac{&x = -5\\&x = 2.}\\
	\end{aligned}$\\
		Tập nghiệm $S = \{-5;2\}$.
	\item $(x^2 + 1)(5x + 3) = 0 \quad (1)$\\
	Vì $x^2 + 1 > 0$ nên phương trình\\
	$(1) \Leftrightarrow 5x + 3 = 0 \Leftrightarrow x = -\dfrac{3}{5}. $\\
		Tập nghiệm $S = \left\lbrace - \dfrac{3}{5} \right\rbrace $.
	
	\item  $ \begin{aligned}[t]
	&\left( \dfrac{3}{4}x - 1 \right) \left( \dfrac{5}{3}x + 2 \right) = 0\\ 
	\Leftrightarrow &\hoac{& \dfrac{3}{4}x - 1 = 0 \\&\dfrac{5}{3}x + 2 = 0 }\\ 
	\Leftrightarrow &\hoac{& x = \dfrac{4}{3} \\& x = -\dfrac{6}{5}.}\\
	\end{aligned}$\\
	Tập nghiệm $S = \left\{ \dfrac{4}{3};-\dfrac{6}{5} \right\}.$
\end{listEX}


}
\end{vd}

			\Leftrightarrow \ & x^3 + x^2 + x + 1 = 0\\
			\Leftrightarrow\  &(x +  1)(x^2 + 1) = 0.\\
			\end{aligned}$\\

			Tập nghiệm $S = \left\{- 1\right\}.$

			&x^3 + (x - 5)(x + 8) = 2x^2 - 37 \\

			\Leftrightarrow\ & x^3 - x^2 + 3x - 3 = 0\\
			\Leftrightarrow\ & x^3 - x^2 + 3x - 3 = 0\\

			\Leftrightarrow\ & (x - 1)(x^2 + 3) = 0.\\
		\end{aligned}$\\
			Vì $x^2 + 3 > 0$ nên $x - 1 = 0 \Leftrightarrow x = 1.$\\
			Tập nghiệm $S = \left\{1\right\}.$
			
		\end{listEX}
	}

\begin{vd}%[8D3K4]
Cho phương trình $(x + m)^2 - (x - 3m)^2 = 0$ trong đó $m$ là một số cho trước.
\begin{enumerate}
	\item Tìm các giá trị của $m$ để phương trình có một trong các nghiệm là $x = 2.$
	\item Với các giá trị của $m$ tìm được ở câu a), hãy giải phương trình đã cho.	
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Thay $x = 2$ vào phương trình đã cho ta được phương trình 
	\begin{eqnarray*}
	&&(2 + m)^2-(2 - 3 m)^2= 0\\
	&\Leftrightarrow & (2 + m - 2 + 3 m)(2 + m + 2 - 3 m)= 0\\
    &\Leftrightarrow & 4 m(- 2 m + 4) = 0\\
    &\Leftrightarrow & \hoac{&m = 0 \\ & - 2 m + 4 = 0}\\
    &\Leftrightarrow & \hoac{&m = 0 \\ & m = 2.}
\end{eqnarray*}
	\item \textit{Trường hợp 1}. Với $m = 0$ thì phương trình đã cho trở thành\\
	$$x^2 - x^2 = 0 \Leftrightarrow 0x^2 = 0.$$
	Phương trình nghiệm đúng với mọi $x$.\\
	\textit{Trường hợp 2}. Với $m = 2$ thì phương trình đã cho trở thành 
   \begin{eqnarray*}
    & &(x + 2)^2 - (x + 6)^2 = 0 \\
   &\Leftrightarrow & (x + 2 - x + 6)(x + 2 + x - 6) = 0\\
   &\Leftrightarrow & 8(2x - 4) = 0\\
   &\Leftrightarrow & x  = 2.
     \end{eqnarray*}
\end{enumerate}

}
\end{vd}

\begin{vd}%[8D3K4]
Cho phương trình $x^3 - x^2 - 9x - 9m = 0$ trong đó $m$ là một số cho trước. Biết $x = 3$ là một nghiệm của phương trình. Tìm tất cả các nghiệm còn lại.
\loigiai{
	 \allowdisplaybreaks
Thay $x = 3$ vào phương trình ta được \\
$$27 - 9 - 27 - 9m = 0 \Leftrightarrow m = -1.$$
Với $m = -1$ thì phương trình đã cho trở thành
  \begin{eqnarray*}
  && x^3- x^2- 9 x + 9 = 0\\
  &\Leftrightarrow & x^2(x - 1) - 9(x - 1)= 0\\
  & \Leftrightarrow & (x - 1)(x^2- 9)= 0\\
  & \Leftrightarrow & (x - 1)(x + 3)(x - 3) = 0 \\
 & \Leftrightarrow & \hoac{&x = 1\\&x = -3\\ & x = 3.}
\end{eqnarray*}
Vậy tất cả các nghiệm còn lại là $x = 1, x = -3.$

}	
	\end{vd}

	Cho phương trình: $2x^3 + 5x^2 - 8x - 4m = 0 \quad(1)$\\
	Biết $x = -2$ là một nghiệm của phương trình $(1)$. Tìm các nghiệm còn lại của phương trình đó.
	
	\loigiai{
		Thay $x = -2$ vào phương trình đã cho ta được  
	$$- 16 + 20 + 16 - 4m = 0 \Leftrightarrow 4m = 20 \Leftrightarrow  m = 5.$$
			Với $m = 5$ phương trình đã cho trở thành 
				
				$$2x^3 + 5x^2 - 8x - 20 = 0$$
		     	$$ \Leftrightarrow (2x + 5)(x + 2)(x - 2) = 0$$
					\shortsol{
				Tập nghiệm $S= \left\{ -\dfrac{5}{2}; -2; 2 \right\}$.
			}
	}
\end{bt}
\section{PHƯƠNG TRÌNH CHỨA ẨN Ở MẪU}
\subsection{Trọng tâm kiến thức}
\begin{enumerate}[1)]
	\item \textbf{Điều kiện xác định của một phương trình} là điều kiện của ẩn để tất cả các mẫu trong phương trình đều khác $0$.
	\item Cách giải phương trình chứa ẩn ở mẫu
	\begin{itemize}
		\item \textit{Bước 1.} Tìm điều kiện xác định của phương trình.
		\item \textit{Bước 2.} Quy đồng mẫu hai vế của phương trình rồi khử mẫu.
		\item \textit{Bước 3.} Giải phương trình vừa nhận được.
		\item \textit{Bước 4.} Kết luận.
	\end{itemize}
\end{enumerate}
\subsection{Các dạng bài tập và phương pháp giải}
\begin{dang}{Tìm điều kiện xác định của một phương trình}
	\begin{itemize}
		\item Đặt điều kiện cho tất cả các mẫu trong phương trình có giá trị khác $0$.
		\item Giải các điều kiện này và lấy tất cả các kết quả tìm được.
	\end{itemize}
\end{dang}
\begin{vd}%[8D3Y5]
	Tìm điều kiện xác định của các phương trình:
	\begin{enumEX}{2}
		\item $\dfrac {7x} {x + 4} - \dfrac {x - 3} {x - 1} = \dfrac {x - 5} {8};$	
		\item $\dfrac {x + 6} {5 (x - 2)} - \dfrac {x - 1} {3(x + 2)} = \dfrac {4}{x^2 - 4}.$
	\end{enumEX}
	\loigiai{
		\begin{listEX}[2]
			\item ĐKXĐ là $\heva{&x + 4 \neq 0\\& x - 1 \neq 0} \Leftrightarrow \heva{&x \neq  - 4\\& x \neq 1.}$\\
			Vậy ĐKXĐ là $x \neq -4$ và $x \neq 1$.
			\item Ta có $x^{2} - 4 = (x - 2) (x + 2)$.\\
			ĐKXĐ là $\heva{&x - 2 \neq 0 \\& x + 2 \neq 0} \Leftrightarrow \heva{&x \neq 2 \\& x \neq -2.}$\\
			Vậy ĐKXĐ là $x \neq \pm 2$.
		\end{listEX}	
	}
\end{vd}
\begin{vd}%[8D3Y5]
	Tìm điều kiện xác định của các phương trình:
	\begin{enumEX}{2}
		\item $\dfrac{5 - x}{x^2 + 6x + 9} = \dfrac{3x - 2}{x^2 + 6x + 8}$;
		\item $\dfrac{x - 7}{x^2 + 1} = \dfrac{x + 6}{x^2 - x + 1}$.
	\end{enumEX}
	\loigiai{\begin{enumEX}{2}
					\item 	\begin{align*}
				&x^{2} + 6x + 9 = (x + 3)^2 \neq 0 \ \text{khi} \  x \neq -3.\\
				&x^2 + 6x + 8 = (x + 2) (x + 4) \neq 0 \Leftrightarrow  \heva{& x \neq -2\\& x \neq -4}.
				\end{align*}
				Vậy ĐKXĐ là $x \neq -3$; $x \neq -2$; $ x \neq -4$.
					
			\item Ta có $x^2 + 1 > 0$ với mọi giá trị của $x$.\\
			$x^{2} - x + 1 = \left( 1 - \dfrac {1}{2} \right) ^ {2} + \dfrac {3}{4} > 0$ với mọi giá trị của $x$.\\
			Vậy ĐKXĐ là mọi $x$.
				\end{enumEX}
			
		}
\end{vd}

\begin{vd}%[8D3B5]
	\begin{enumEX}{2}
		\item $\dfrac{15x - 10}{x^2 + 3} = 0$;
		\item $\dfrac{6}{x} - 1 = \dfrac{2x - 3}{3}$;
		\item $\dfrac{3x - 1}{x + 1} = \dfrac{2x + 1}{x - 1}$.
	\end{enumEX}
	\loigiai{ \allowdisplaybreaks
		\begin{enumEX}{2}
			\item $\dfrac{15x - 10}{x^2 + 3} = 0 \quad (1)$\\
			Vì $x^2 + 3 > 0$ với mọi giá trị của $x$ nên ta có ĐKXĐ là mọi $x$.\\
			$(1) \Rightarrow 15x - 10 = 0 \Leftrightarrow 15 = 10 \Leftrightarrow x = \dfrac{2}{3}$.\\ 
			Tập nghiệm $S = \left\lbrace \dfrac{2}{3} \right\rbrace$.
			\item $\dfrac{6}{x} - 1 = \dfrac{2x - 3}{3} \quad (2)$  \\
			ĐKXĐ: $x \neq 0.$\\
			$\begin{aligned}[t]
			(2)& \Rightarrow 18 - 3x = x(2x - 3)\\
			\Leftrightarrow\ &18 - 3x = 2x^2 - 3x\\
			\Leftrightarrow\ & 2x^2 - 18 = 0\\
			\Leftrightarrow \ & x^2 - 9 = 0\\
			\Leftrightarrow\ & (x - 3)(x + 3) = 0\\
			\Leftrightarrow\ & \hoac{&x - 3 = 0\\&x + 3 = 0}\\
			\Leftrightarrow\ & \hoac{&x = 3\\ &x = -3} \text{(thỏa mãn ĐKXĐ)}.
			\end{aligned}$\\
			Tập nghiệm $S = \left\{ -3; 3 \right\}$. 
			\item $\dfrac{3x - 1}{x + 1} = \dfrac{2x + 1}{x - 1} \quad (3)$.  \\
			ĐKXĐ: $x \neq \pm 1$.
			\begin{align*}
			(3) &\Rightarrow (3x - 1)(x - 1) = (2x + 1)(x + 1)\\
			&\Leftrightarrow 3x^2 - 3x - x + 1 = 2x^2 + 2x + x + 1\\
			&\Leftrightarrow x^2 - 7x = 0\\
			&\Leftrightarrow x(x - 7) = 0\\
			&\Leftrightarrow \hoac{&x = 0 \\&x = 7} \text{(thỏa mãn ĐKXĐ).}
			\end{align*}
		Tập nghiệm $S = \left\{ 0; 7 \right\}$.	
		\end{enumEX}
	}
\end{vd}
\begin{vd}%[8D3B5]
	Giải các phương trình:
	\begin{enumEX}{2}
		\item $\dfrac {x^2 - 4x - 5}{x - 5} = 0;$
		\item $\dfrac {x^2 - 5x + 6}{x - 2} = 1;$
		\item $\dfrac {x - 2}{x + 8} + \dfrac {1}{x} = \dfrac{8}{x(x + 8)}$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{2}
			\item $\dfrac {x^{2} - 4x - 5}{x - 5} = 0 \quad(1)$ \\
			ĐKXĐ: $x \neq 5$.\\
	    	$\begin{aligned}[t]
			(1) &\Rightarrow x^2 - 4x - 5 = 0\\
			\Leftrightarrow \ &x^{2} - 5x + x - 5 = 0\\
			\Leftrightarrow\ &x (x - 5) + (x - 5) = 0\\
			\Leftrightarrow\ &( x - 5 ) ( x + 1 ) = 0\\
		     \Leftrightarrow\ &\hoac{&x = 5\ \text{(loại)}\\& x = -1\ \text{(thỏa mãn)}.}\\
		     \end{aligned}$\\
			Tập nghiệm $S = \left\{-1\right\}.$
			\item  $\dfrac {x^{2} - 5x + 6}{x - 2} = 1 \quad(2)$ \\
			ĐKXĐ: $ x \neq 2$.\\
			$\begin{aligned}[t]
			(2) &\Rightarrow x^{2} - 5x + 6 = x - 2\\
		   \Leftrightarrow\ & x^2 -6x + 8 = 0\\
			\Leftrightarrow\ & (x - 4)(x - 2) = 0\\
			\Leftrightarrow\ & \hoac{&x - 4 = 0\\ & x - 2 = 0 }\\
		   \Leftrightarrow\ & \hoac{&x = 4\ \text{(thỏa mãn)}\\ & x = 2\ \text{(loại)}.}\\
			\end{aligned}$\\
				Tập nghiệm $S = \{4\}$.
			\item $\dfrac {x - 2}{x + 8} + \dfrac {1}{x} = \dfrac{8}{x(x + 8)} \quad(3)$\\
			ĐKXĐ: $\heva{&x + 8 \neq 0 \\& x \neq 0} \Leftrightarrow \heva{&x \neq - 8\\& x \neq 0.}$\\
			$\begin{aligned}[t]
			(3) &\Rightarrow x(x - 2) + (x + 8) = 8\\
			\Leftrightarrow\ &x^2 - 2x + x + 8 = 8\\
			\Leftrightarrow\ &x^2 - x = 0\\
			\Leftrightarrow\ &x(x - 1) = 0\\
		     \Leftrightarrow\ & \hoac{&x = 0\\&x - 1 = 0}\\
			  \Leftrightarrow\ & \hoac{&x = 0\ \text{(loại)}\\&x = 1\ \text{(thỏa mãn)}.}\\
			  	\end{aligned}$\\
				Tập nghiệm $ S = \left\{1\right\} $.
				\end{enumEX}
	}
\end{vd}
\begin{vd}%[8D3B5]
	Giải các phương trình sau:
	\begin{enumEX}{2}
		\item $\dfrac{x+5}{3(x-1)} + 1 = \dfrac{3x + 7}{5(x-1)}$;
		\item $\dfrac{3x - 1}{3x + 1} - \dfrac{3x + 1}{3x - 1} = \dfrac{4}{9x^2 - 1}$;
		\item $\dfrac{3x - 1}{x - 1} - \dfrac{2x + 5}{x + 3} - \dfrac{8}{x^2 + 2x - 3} = 1$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{2}
			\item $\dfrac{x+5}{3(x-1)} + 1 = \dfrac{3x + 7}{5(x-1)} \quad(1)$\\
			ĐKXĐ: $x \neq 1$.
			\begin{align*}
			(1) &\Rightarrow 5(x + 5) + 15(x - 1) = 3(3x + 7)\\
			\Leftrightarrow\ &5x + 25 + 15x - 15 = 9x + 21\\
			\Leftrightarrow\ &5x + 15x - 9x = 21 - 25 + 15\\
			\Leftrightarrow\ &11x = 11\\
			\Leftrightarrow\ &x = 1\ \text{(loại)}.
			\end{align*}
			Tập nghiệm $S = \varnothing $.
			\item $\dfrac{3x - 1}{3x + 1} - \dfrac{3x + 1}{3x - 1} = \dfrac{4}{9x^2 - 1} \quad(2)$\\
			ĐKXĐ: $x \neq \pm \dfrac{1}{3}$.
			\begin{align*}
			(2) &\Rightarrow \left( 3x - 1 \right)^2 - \left( 3x + 1 \right)^2 = 4\\
			&\Leftrightarrow  9x^2 - 6x + 1 - 9x^2 - 6x - 1 = 4\\
			&\Leftrightarrow -12x = 4\\
			&\Leftrightarrow x = - \dfrac{1}{3} \hspace{0.2cm} \text{(loại)}.
			\end{align*}
			Tập nghiệm $S = \varnothing$.
			\item $\dfrac{3x - 1}{x - 1} - \dfrac{2x + 5}{x + 3} - \dfrac{8}{x^2 + 2x - 3} = 1 \quad(3)$ \\
			ĐKXĐ là $\heva{& x - 1 \neq 0\\& x +3 \neq 0}   \Leftrightarrow \heva{&x \neq 1\\&x \neq -3.}$\\
			\begin{align*}
			(3) &\Rightarrow (3x - 1)(x + 3) - (2x + 5)(x - 1) - 8 = x^2 + 2x - 3\\
			&\Leftrightarrow 3x^2 + 9x - x - 3 - 2x^2 + 2x - 5x + 5 - 8 = x^2 + 2x -3\\
			&\Leftrightarrow 3x = 3 \\
			&\Leftrightarrow x = 1\ \text{(loại)}.
			\end{align*}
			Tập nghiệm $S = \varnothing$.
		\end{enumEX}
	}
\end{vd}
\begin{vd}%[8D3B5]
	Giải các phương trình sau:
	\begin{enumEX}{2}
		\item $\dfrac{x+3}{x+1} + \dfrac{x-2}{x} = \dfrac{2(x^2 + x - 1)}{x(x+1)}$;
		\item $\dfrac{2}{x - 2} - \dfrac{x}{x+3} = \dfrac{5x}{(x - 2)(x +3)} - 1$;
		\item $\dfrac{1}{x^2 + x + 1} - \dfrac{1}{x^2 - x + 1} = \dfrac{1 - 2x}{x^4 + x^2 + 1}$.
	\end{enumEX}
	\loigiai{\allowdisplaybreaks
		\begin{enumEX}{2}
			\item $\dfrac{x+3}{x+1} + \dfrac{x-2}{x} = \dfrac{2(x^2 + x - 1)}{x(x+1)} \quad(1)$\\
			ĐKXĐ: $x \neq 0$, $x \neq -1$.
			\begin{align*}
			(1)& \Rightarrow x(x + 3) + (x - 2)(x + 1) = 2(x^2 + x - 1)\\
			&\Leftrightarrow x^2 + 3x +x^2 - x - 2 = 2x^2 + 2x -2\\
			&\Leftrightarrow 0x = 0.
			\end{align*}
			Phương trình này có nghiệm đúng với mọi $x$.\\
			Vậy phương trình đã cho có nghiệm với mọi $x$ trừ giá trị $x = 0$ và $x = -1$.
			\item $\dfrac{2}{x - 2} - \dfrac{x}{x+3} = \dfrac{5x}{(x - 2)(x +3)} - 1 \quad(2)$\\
			ĐKXĐ: $x \neq 2$; $x \neq -3$.\\
			Từ phương trình (2)
			\begin{align*}
			&\Rightarrow 2x + 6 - x(x-2) = 5x-(x-2)(x + 3)\\
			&\Leftrightarrow 2x + 6 - x^2 + 2x = 5x - x^2 - x + 6\\
			&\Leftrightarrow 0x = 0
			\end{align*}
			Phương trình này có nghiệm đúng với mọi $x$.\\
			Phương trình đã cho có nghiệm với mọi $x$ trừ giá trị $x = 2$ và $x = -3$.
			\item 	$\dfrac{1}{x^2 + x + 1} - \dfrac{1}{x^2 - x + 1} = \dfrac{1 - 2x}{x^4 + x^2 + 1} \quad (3)$\\
			Ta có:
			\begin{align*}
			&x^4 + x^2 + 1 = (x^2 + x + 1) (x^2 - x + 1)\\
			&x^2 + x + 1 = \left( x + \dfrac{1}{2} \right)^2 + \dfrac{3}{4} > 0\\ 
			&x^2 - x + 1 = \left( x - \dfrac{1}{2} \right)^2 + \dfrac{3}{4} > 0.
			\end{align*}
			Do đó ĐKXĐ là mọi $x$.
			\begin{align*}
			(3) &\Rightarrow x^2 - x + 1 - x^2 - x - 1 = 1 - 2x\\
			&\Leftrightarrow 0x = 1.
			\end{align*}
			Phương trình này vô nghiệm.\\
			Vậy phương trình đã cho vô nghiệm.
		\end{enumEX}
	}
\end{vd}
\begin{dang}{Tìm giá trị của biến để giá trị của hai biểu thức có mối liên quan nào đó}
	Từ mối liên quan giữa giá trị của hai biểu thức, ta lập một phương trình rồi giải phương trình này.
\end{dang}
\begin{vd}
	Cho hai biểu thức $A = \dfrac{3}{3x + 1} + \dfrac{2}{1 - 3x}$, $B = \dfrac{x - 5}{9x^2 - 1}$ với giá trị nào của $x$ thì hai biểu thức $A$ và $B$ có cùng một giá trị?
	\loigiai{
		$$A = B \Leftrightarrow 	\dfrac{3}{3x + 1} + \dfrac{2}{1 - 3x} = \dfrac{x - 5}{9x^2 - 1}. \quad(1)$$
		ĐKXĐ: $x \neq \pm \dfrac{1}{3}$.
		\begin{align*}
		(1) &\Leftrightarrow \dfrac{3}{3x + 1} - \dfrac{2}{3x - 1} = \dfrac{x - 5}{9x^2 - 1}\\
		&\Rightarrow 3(3x - 1) - 2(3x + 1) = x - 5\\
		&\Leftrightarrow 9x - 3 - 6x - 2 - x + 5 = 0\\
		&\Leftrightarrow 2x = 0\\
		&\Leftrightarrow x = 0 \ \text{(thỏa mãn ĐKXĐ).}
		\end{align*}
		Vậy khi $x = 0$ thì $A = B$.
	}
\end{vd}
\begin{vd}%[8D3B5]
	Cho ba biểu thức $A = \dfrac{2}{5x - 2}$; $B = \dfrac{4}{1 - 5x}$; $C = \dfrac{2}{(5x - 2)(5x - 1)}$. \\
	Tìm các giá trị của $x$ để tổng $A + B$ có giá trị bằng giá trị của biểu thức $C$.
	\loigiai{
		$$A + B = C \Leftrightarrow \dfrac{2}{5x - 2} + \dfrac{4}{1 - 5x} = \dfrac{2}{(5x - 2)(5x - 1)} \quad(1) $$\\
		ĐKXĐ: $x \neq \dfrac{2}{5}$; $x \neq \dfrac{1}{5}$.
		\begin{align*}
		(1) &\Leftrightarrow \dfrac{2}{5x - 2} - \dfrac{4}{5x - 1} =  \dfrac{2}{(5x - 2)(5x - 1)}\\
		&\Leftrightarrow 2(5x - 1) - 4(5x - 2) = 2\\
		&\Leftrightarrow 10x - 2 - 20x + 8 = 2\\
		&\Leftrightarrow -10x = -4\\
		&\Leftrightarrow x = \dfrac{2}{5}\ \text{(loại).} 
		\end{align*}
		Vậy không có giá trị nào của $x$ để tổng $A + B$ có giá trị bằng giá trị của biểu thức $C$.
	}
\end{vd}
\begin{vd}%[8D3K5]
	Cho hai biểu thức $P = \dfrac{x - 2}{x - 1} + \dfrac{x + 2}{x + 1}$ và $Q = \dfrac{2 - x^2}{1 - x^2}$. Với giá trị nào của $x$ thì giá trị của biểu thức $P$ bằng $2$ lần giá trị của biểu thức $Q$?
	\loigiai{
		$$ P = 2Q \Leftrightarrow \dfrac{x - 2}{x - 1} + \dfrac{x + 2}{x + 1} = \dfrac{2(2 - x^2)}{1 - x^2}. \quad(1)$$\\
		ĐKXĐ: $x \neq \pm 1$.
		\begin{align*}
		(1) &\Leftrightarrow \dfrac{x - 2}{x - 1} + \dfrac{x + 2}{x + 1} = \dfrac{2(x^2 - 2)}{x^2 - 1}\\
		&\Rightarrow (x - 2)(x + 1) + (x + 2)(x - 1) = 2(x^2 - 2)\\
		&\Leftrightarrow x^2 - x - 2 + x^2 + x - 2 = 2x^2 - 4\\
		&\Leftrightarrow 0x = 0.
		\end{align*}	
		Phương trình này nghiệm đúng với mọi $x$.\\
		Vậy $P = 2Q$ với mọi giá trị của $x$ trừ $x = \pm 1$.
	}
\end{vd}
\begin{dang}{Biết phương trình tham số $m$ có một trong các nghiệm là $x = x_0$, tìm nghệm còn lại}
	\begin{itemize}
		\item Thay $x = x_0$ vào phương trình đã cho rồi giải phương trình để tìm giá trị của tham số $m$.
		\item Thay giá trị vừa tìm được của $m$ vào phương trình đã cho rồi giải phương trình để tìm tất cả các nghiệm.
	\end{itemize}
\end{dang}
\begin{vd}%[8D3B5]
	Cho phương trình $\dfrac{x - 2}{x - 4} + \dfrac{x - 3}{x - 2} = \dfrac{m}{3}$ trong đó $m$ là một số cho trước. Biết $x = 5$ là một trong các nghiệm của phương trình, tìm các nghiệm còn lại.
	\loigiai{
		\begin{itemize}
			\item Thay $x = 5$ vào phương trình đã cho ta được
			$$\dfrac{3}{1} + \dfrac{2}{3} = \dfrac{m}{3} \Leftrightarrow m = 11.$$
			\item Thay $m = 11$ vào phương trình đã cho ta được
			$$\dfrac{x - 2}{x - 4} + \dfrac{x - 3}{x - 2} = \dfrac{11}{3} \quad(1)$$\vspace{0.2cm}\\
			ĐKXĐ: $x \neq 4$; $x \neq 2$.
			\begin{align*}
			(1) &\Rightarrow 3(x - 2)^2 + 3(x - 3)(x - 4) = 11(x - 4)(x - 2)\\
			&\Leftrightarrow 3(x^2 - 4x + 4) + 3(x^2 - 7x + 12) = 11(x^2 - 6x + 8)\\
			&\Leftrightarrow 3x^2 - 12x + 12 + 3x^2 - 21x + 36 = 11x^2 - 66x + 88\\
			&\Leftrightarrow 5x^2 - 33x + 40 = 0\\
			&\Leftrightarrow (x - 5)(5x - 8) = 0\\
			&\Leftrightarrow \heva{&x = 5 \\&x = \dfrac{8}{5}} \text{(thỏa mãn ĐKXĐ).}
			\end{align*}
		\end{itemize}	
		Vậy nghiệm còn lại là $x = \dfrac{8}{5}$.
	}
\end{vd}
\begin{vd}%[8D3K5]
	Cho phương trình $\dfrac{2x + m}{x - 1} = \dfrac{5(x - 1)}{x + 1}$.\\
	Chứng minh rằng nếu $x = \dfrac{1}{3}$ là một nghiệm của phương trình thì phương trình còn có một nghiệm nguyên.
	\loigiai{
		\begin{itemize}
			\item Thay $x = \dfrac{1}{3}$ vào phương trình đã cho ta được
			$$\dfrac{\dfrac{2}{3} + m}{\dfrac{1}{3} - 1} = \dfrac{5\cdot \left( \dfrac{1}{3} - 1 \right) }{\dfrac{1}{3} + 1} \Leftrightarrow \dfrac{- (2 + 3m)}{2} = \dfrac{-5}{2} \Leftrightarrow m = 1.$$
			\item Thay $m = 1$ vào phương trình đã cho ta được
			$$\dfrac{2x + 1}{x - 1} = \dfrac{5(x - 1)}{x + 1} \quad(1)$$
			ĐKXĐ: $x \neq \pm 1$.
			\begin{align*}
			(1) &\Rightarrow (2x + 1)(x + 1) = 5(x - 1)^2\\
			&\Leftrightarrow 2x^2 + 2x + x + 1 = 5x^2 - 10x + 5\\
			&\Leftrightarrow 3x^2 - 13x + 4 = 0\\
			&\Leftrightarrow (3x - 1)(x - 4) = 0\\
			&\Leftrightarrow \hoac{&x = \dfrac{1}{3}\\&x = 4} \text{(thỏa mãn ĐKXĐ).}
			\end{align*}
		\end{itemize}	
		Vậy phương trình đã cho có nghiệm $x = \dfrac{1}{3}$ và nghiệm nguyên $x = 4$.
	}
\end{vd}
\subsection{Bài tập tự luyện}
\begin{bt}%[8D3B5]
	Giải các phương trình sau:
	\begin{enumEX}{2}
		\item $\dfrac{2x - 1}{x^2 - 1} = \dfrac{6}{3x + 1}$;
		\item $\dfrac{1}{x + 3} + \dfrac{1}{x - 1} = \dfrac{10}{(x + 3)(x - 1)}$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{2}
			\item $\dfrac{2x - 1}{x^2 - 1} = \dfrac{6}{3x + 1}\quad(1)$\\
			ĐKXĐ: $x \neq \pm 1$ và $x \neq \dfrac{-1}{3}$.
			\begin{align*}
			(1) &\Rightarrow (2x - 1)(3x + 1) = 6(x^2 - 1)\\
			&\Leftrightarrow 6x^2 - x - 1 = 6x^2 - 6\\
			&\Leftrightarrow x = 5 \hspace{0.2cm} \text{(thỏa mãn)}.
			\end{align*}
			Tập nghiệm $S = \left\lbrace 5 \right\rbrace $.
			\item $\dfrac{1}{x + 3} + \dfrac{1}{x - 1} = \dfrac{10}{(x + 3)(x - 1)}\quad(2)$\vspace{0.2cm}\\
			ĐKXĐ: $x \neq -3$ và $x \neq 1$.
			\begin{align*}
			(2) &\Rightarrow x - 1 + x + 3 = 10\\
			&\Leftrightarrow 2x = 8\\
			&\Leftrightarrow x = 4 \hspace{0.2cm} \text{(thỏa mãn)}.
			\end{align*}
			Tập nghiệm $S = \left\{4 \right\}$.
		\end{enumEX}
	}
\end{bt}
\begin{bt}%[8D3B5]
	Giải các phương trình sau:
	\begin{enumEX}{2}
		\item $\dfrac{1}{x + 1} - \dfrac{4}{x^2 - x + 1} = \dfrac{2x^2 + 1}{x^3 + 1}; \quad(1)$
		\item $\dfrac{3x + 1}{x + 1} - \dfrac{2x - 5}{x - 3} = 1 - \dfrac{4}{(x + 1)(x - 3)}. \quad (2)$
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item ĐKXĐ: $x \neq -1$.
			\begin{align*}
			(1) &\Rightarrow x^2 - x + 1 - 4(x + 1) = 2x^2 + 1\\
			&\Leftrightarrow x^2 - 5x + 4 = 0\\
			&\Leftrightarrow (x + 1)(x + 4)  = 0\\
			&\Leftrightarrow \heva{&x = -1\ \text{(loại)}\\&x = -4\ \text{(thỏa mãn).}}
			\end{align*}
			Tập nghiệm $S = \left\{ -4 \right\} $.
			\item ĐKXĐ: $x \neq -1$ và $x \neq 3$.\\
			Từ phương trình (2)
			\begin{align*}
	 &\Rightarrow (3x + 1)(x - 3)-(2x - 5)(x + 1) = (x + 1)(x - 3) - 4 \\
		&\Leftrightarrow 3x^2 - 8x - 3 - 2x^2 + 3x + 5 = x^2 + x - 3x - 4\\
		&\Leftrightarrow -3x = -9\\
		&\Leftrightarrow x = 3\ \text{(loại)}.  
			\end{align*}
			Tập nghiệm $S = \varnothing$.
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[8D3K5]
	Giải các phương trình sau:
	\begin{enumEX}{2}
		\item $\dfrac{6}{x^2 - 9} - \dfrac{1}{2x - 7} = \dfrac{13}{(x + 3)(2x - 7)}$;
		\item $\dfrac{2(x - 5)}{x^2 + 4x +3} = \dfrac{x - 5}{x^2 + 5x + 6}$.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item $\dfrac{6}{x^2 - 9} - \dfrac{1}{2x - 7} = \dfrac{13}{(x + 3)(2x - 7)} \quad(1)$ \\
			ĐKXĐ: $x \neq \pm 3$ và $x \neq \dfrac{7}{2}$.
			\begin{align*}
			(1) &\Rightarrow 6(2x - 7) - (x^2 - 9) = 13(x - 3)\\
			&\Leftrightarrow 12x - 42 - x^2 + 9 - 13x + 39 = 0\\
			&\Leftrightarrow x^2 + x - 6 = 0 \\
			&\Leftrightarrow (x - 2)(x + 3) = 0\\
			&\Leftrightarrow \hoac{&x = 2\ \text{(thỏa mãn)}\\&x = -3\ \text{(loại).}}
			\end{align*}
			Tập nghiệm $S = \left\{ 2 \right\}$.
			\item $\dfrac{2(x - 5)}{x^2 + 4x +3} = \dfrac{x - 5}{x^2 + 5x + 6} \quad(2)$ \\
			ĐKXĐ: $x \neq -1$, $x \neq -2$, $x \neq -3$.
			\begin{align*}
			(2) &\Leftrightarrow (x - 5)\left[\dfrac{1}{(x + 2)(x + 3)}-\dfrac{2}{(x + 1)(x + 3)}\right] = 0\\
			&\Leftrightarrow (x - 5)(-x - 3) = 0 \\
			&\Leftrightarrow \hoac{&x = 5\ \text{(thỏa mãn)}\\&x = -3\ \text{(loại).}}
			\end{align*}
			Tập nghiệm $S = \left\{ 5\right\}$. 
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[8D3K5]
	Cho phương trình $\dfrac{1}{x + 1}-\dfrac{2 x^2- m}{x^3+ 1}=\dfrac{4}{x^2- x + 1}$. Biết $x = 0$ là một nghiệm của phương trình. Tìm các nghiệm còn lại.
	\loigiai{
		\begin{itemize}
			\item Thay $x = 0$ vào phương trình, ta được:
			$$\dfrac{1}{1}-\dfrac{0 - m}{0 + 1}=\dfrac{4}{0 - 0 + 1} \Leftrightarrow 1 + m = 4 \Leftrightarrow m = 3.$$
			\item Thay $m = 3$ vào phương trình ta được:
			$$\begin{aligned}[t]
			&\dfrac{1}{x + 1}-\dfrac{2 x^2- 3}{x^3+ 1}=\dfrac{4}{x^2- x + 1} \hspace{0.2cm} (\text{ĐKXĐ}\ x \neq -1)\\
			\Rightarrow\ &x^2 - x + 1 - 2x^2 + 3 = 4(x + 1)\\
			\Leftrightarrow\ & -x^2 - 5x = 0\\
			\Leftrightarrow\ & \hoac{&x = 0\\&x = -5} \text{(thỏa mãn ĐKXĐ).}
			\end{aligned}$$
			Vậy $x = -5$ là nghiệm còn lại của phương trình.
		\end{itemize}
	}
\end{bt}
\begin{bt}%[8D3K5]
	Cho hai biểu thức $A =\dfrac{x + 1}{2 x - 3}$; $B =\dfrac{3 x}{x^2- 4}$, với giá trị nào của $x$ thì hai biểu thức $A$ và $B$ có cùng một giá trị?
	\loigiai{
		$$A = B\Leftrightarrow\dfrac{x + 1}{2 x - 3}=\dfrac{3 x}{x^2- 4}. $$
		ĐKXĐ: $x \neq \pm 2$ và $x \neq \dfrac{3}{2}$. Biến đổi phương trình về dạng:\\
		$$\begin{aligned}[t]
	    	&x^3- 5 x^2+ 5 x - 4 = 0\\
		\Leftrightarrow\ & x^3- 4 x^2- x^2+ 4 x + x - 4 = 0\\
		\Leftrightarrow\ &(x - 4)\left(x^2- x + 1\right) = 0\\
		\Leftrightarrow\ &x = 4\ \text{(thỏa mãn ĐKXĐ)}.\\
		\end{aligned}$$
		Đáp số: $x = 4$.
	} 
\end{bt}
