\begin{ex}%[Tuyển sinh 10 Yên Bái 2018]%[Triệu Minh Hà, dự án EXTHCS-06]%[8D1B8]
	Phương trình $\dfrac{3}{1 - 4x} = \dfrac{2}{4x + 1} - \dfrac{8 + 6x}{16x^2 - 1}$ có bao nhiêu nghiệm?
	\choice
	{$2$ }
	{$0$ }
	{\True $1$ }
	{$3$ }
	\loigiai{
		\begin{eqnarray*}
		& &\dfrac{3}{1 - 4x} = \dfrac{2}{4x + 1} - \dfrac{8 + 6x}{16x^2 - 1}\\
		&\Leftrightarrow &\dfrac{-3}{4x - 1} = \dfrac{2}{4x + 1} - \dfrac{8 + 6x}{16x^2 - 1}\\
		&\Leftrightarrow &\dfrac{8 + 6x}{16x^2 - 1} = \dfrac{2}{4x + 1} + \dfrac{3}{4x - 1}\\
		&\Leftrightarrow &\dfrac{8 + 6x}{16x^2 - 1} = \dfrac{8x - 2 + 12x + 3}{16x^2 - 1}\\
		&\Leftrightarrow &8 + 6x = 20x + 1\\
		&\Leftrightarrow &14x = 7 \\
		&\Leftrightarrow &x = \dfrac{1}{2}. 
		\end{eqnarray*}
		Phương trình đã cho có $1$ nghiệm.
	}

\begin{ex}%[Tuyển sinh 10 Yên Bái 2018]%[Triệu Minh Hà, dự án EXTHCS-06]%[8D3K6]%[9D3K5]
	Tìm số tự nhiên $n$ có hai chữ số, biết rằng tổng các chữ số của nó bằng $9$ và nếu cộng thêm vào số đó $63$ đơn vị thì được một số mới cũng viết bằng hai chữ số đó nhưng theo thứ tự ngược lại. 
	\choice
	{$n = 36$ }
	{\True $n = 18$ }
	{$n = 45$ }
	{$n = 27$ }
	\loigiai{
		Gọi chữ số hàng chục là $a$ và chữ số hàng đơn vị là $b$, điều kiện ($0 <  a < 9$ và $0< b < 9$).\\ Số ban đầu có dạng $n = 10a + b$.\\
		Biết tổng các chữ số ở hàng đơn vị và hàng chục là $9$ nên ta có phương trình
		\begin{align}\label{01}
		a + b = 9
		\end{align}
		Vì cộng thêm vào số đó $63$ đơn vị thì được một số mới cũng viết bằng hai chữ số đó nhưng theo thứ tự ngược lại nên ta có phương trình
		\begin{align}\label{02}
		10a + b + 63 = 10b + a
		\end{align} 
		Từ \eqref{01} và \eqref{02} ta có hệ phương trình
		\begin{align*}
		\begin{cases}a + b  = 9\\10a + b + 63 = 10b + a\end{cases} \Leftrightarrow \begin{cases}a + b = 9\\ 9b - 9a = 63\end{cases} \Leftrightarrow \begin{cases}a + b = 9\\ b - a = 7\end{cases} \Leftrightarrow \begin{cases}a = 1 \\ b= 8. \end{cases}
		\end{align*}
		Vậy số tự nhiên cần tìm là $18$.
	}
\end{ex}
