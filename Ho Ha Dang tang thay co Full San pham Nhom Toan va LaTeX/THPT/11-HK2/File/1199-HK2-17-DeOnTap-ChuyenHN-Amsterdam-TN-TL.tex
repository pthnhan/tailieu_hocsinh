\begin{ex}%[Đề cương HK2 (2017-2018), THPT Chuyên Amsterdam, Hà Nội]%[1D3B4-3]%[Đinh Thanh Hoàng]
	Cho dãy số $(u_n)$ với $u_n = 3^n$. Khẳng định nào dưới đây đúng?
	\choice
	{$\dfrac{u_1 + u_3}{2} = u_2$}
	{\True $u_1 \cdot u_2 \cdots u_{150} = u_{11325}$}
	{$\dfrac{u_{k-1} \cdot u_{k+1}}{2} = u_k$}
	{$u_1 + u_2 + \cdots + u_{200} = \dfrac{u_{200} - 1}{2}$}
	\loigiai{
		Ta có $u_n = 3^n \Rightarrow u_{n+1} = 3 \cdot 3^n \Rightarrow \dfrac{u_{n+1}}{u_n} = 3 \Rightarrow u_{n+1} = 3u_n$.\\
		Vậy $u_u$ là một cấp số nhân với số hạng đầu $u_1 = 3$ và công bội $q = 3$ $\Rightarrow u_{11325} = 3^{11325}$.\\
		Mặt khác ta có $u_1 \cdot u_2 \cdots u_{150} = 3^1 \cdot 3^2 \cdots 3^{150} = 3^{11325}$.\\
		Vậy $u_1 \cdot u_2 \cdots u_{150} = u_{11325}$.
	}

\begin{ex}%[Đề cương HK2 (2017-2018), THPT Chuyên Amsterdam, Hà Nội]%[1D4B2-4]%[Đinh Thanh Hoàng]

	{$L= + \infty$}
	{$L= - \infty$}
	{\True $L= 2$}
	{$L= -2$}

			L &= \lim\limits_{x \to - \infty} \dfrac{-4x-1}{\sqrt{x^2 - 7x +1}+\sqrt{x^2-3x+2}}\\

			&= \lim\limits_{x \to - \infty} \dfrac{-4-\dfrac{1}{x}}{- \sqrt{1-\dfrac{7}{x}+\dfrac{1}{x^2}}-\sqrt{1-\dfrac{3}{x}+\dfrac{2}{x^2}}}\\
			&=2.
		\end{align*}
	}
\end{ex}

\begin{ex}%[Đề cương HK2 (2017-2018), THPT Chuyên Amsterdam, Hà Nội]%[1D4K2-3]%[Đinh Thanh Hoàng]
	Tính $L=\lim\limits_{x \to 1} \dfrac{x^m -x^n}{x-1}$ với $m,n \in \mathbb{N}^*$.
	\choice
	{$L=+\infty$}
	{$L=0$}
	{\True $L=m-n$}
	{$L=m+n$}
	\loigiai{
		Ta có $L=\lim\limits_{x \to 1} \dfrac{x^m -x^n}{x-1} =  \lim\limits_{x \to 1} \dfrac{x^m -1}{x-1} -\lim\limits_{x \to 1} \dfrac{x^n -1}{x-1}$.\\
		$\lim\limits_{x \to 1} \dfrac{x^m -1}{x-1}= \lim\limits_{x \to 1} \dfrac{(x-1)(x^{m-1}+x^{m-2} + \cdots + 1 )}{x-1}=\lim\limits_{x \to 1} (x^{m-1}+x^{m-2} + \cdots + 1 ) =m$.\\
		$\lim\limits_{x \to 1} \dfrac{x^n -1}{x-1}= \lim\limits_{x \to 1} \dfrac{(x-1)(x^{n-1}+x^{n-2} + \cdots + 1 )}{x-1}=\lim\limits_{x \to 1} (x^{n-1}+x^{n-2} + \cdots + 1 ) =n$.\\
		Vậy $L= \lim\limits_{x \to 1} \dfrac{x^m -x^n}{x-1}=m-n$.
	}
\end{ex}

\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[Đề cương HK2 (2017-2018), THPT Chuyên Amsterdam, Hà Nội]%[1D4B2-7]%[1D4B2-5]%[Đinh Thanh Hoàng]
	Tìm các giới hạn sau:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\lim\limits_{x \to +\infty} \dfrac{\sqrt{x^2 + 1} + x}{5 - 2x}$.
			\item $\lim\limits_{x \to 0^+} \dfrac{\sqrt{x^2 + x + 1} - \sqrt{x^2 - x + 1}}{x^2}$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item $\lim\limits_{x \to +\infty} \dfrac{\sqrt{x^2 + 1} + x}{5 - 2x} = \lim\limits_{x \to +\infty} \dfrac{x\sqrt{1 + \dfrac{1}{x^2}} + x}{5 - 2x} = \lim\limits_{x \to +\infty} \dfrac{\sqrt{1 + \dfrac{1}{x^2}} + 1}{\dfrac{5}{x} - 2} = -1$.
			\item $\lim\limits_{x \to 0^+} \dfrac{\sqrt{x^2 + x + 1} - \sqrt{x^2 - x + 1}}{x^2} = \lim\limits_{x \to 0^+} \dfrac{2x}{x^2\left(\sqrt{x^2 + x + 1} + \sqrt{x^2 - x + 1}\right)}$\\
					\phantom{$\lim\limits_{x \to 0^+} \dfrac{\sqrt{x^2 + x + 1} - \sqrt{x^2 - x + 1}}{x^2}$} $= \lim\limits_{x \to 0^+} \dfrac{2}{x\left(\sqrt{x^2 + x + 1} + \sqrt{x^2 - x + 1}\right)}$.\\
					Ta có: $\lim\limits_{x \to 0^+} x\left(\sqrt{x^2 + x + 1} + \sqrt{x^2 - x + 1}\right) = 0$ và $x\left(\sqrt{x^2 + x + 1} + \sqrt{x^2 - x + 1}\right) > 0$ khi $x \to 0^+$.\\
					Vậy $\lim\limits_{x \to 0^+} \dfrac{\sqrt{x^2 + x + 1} - \sqrt{x^2 - x + 1}}{x^2} = +\infty$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề cương HK2 (2017-2018), THPT Chuyên Amsterdam, Hà Nội]%[1D4B3-4]%[1D5B2-3]%[Đinh Thanh Hoàng]
	\hfill
	\begin{enumerate}
		\item Tìm giá trị của tham số $a$ để hàm số $f(x) = \heva{&ax^2 &&\mbox{khi } x\leq 2 \\ &x^2 + x - 1 &&\mbox{khi } x > 2}$ liên tục trên $\mathbb{R}$.
		\item Cho hàm số $y = \dfrac{3x + 5}{x + 3}$ có đồ thị $(C)$. Viết phương trình tiếp tuyến của $(C)$ biết tiếp tuyến này vuông góc với đường thẳng $(d): 4x + y - 3 = 0$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Tập xác định của hàm số là $D = \mathbb{R}$.\\
					Khi $x > 2$ thì $f(x) = x^2 + x - 1$ liên tục trên khoảng $(2; +\infty)$.\\
					Khi $x < 2$ thì $f(x) = ax^2$ liên tục trên khoảng $(-\infty; 2)$.\\
					Hàm số đã cho liên tục trên $\mathbb{R}$ khi và chỉ khi
					$$\lim\limits_{x \to 2} f(x) = f(2) \Leftrightarrow \lim\limits_{x \to 2^+} f(x) = \lim\limits_{x \to 2^-} f(x) = f(2) \Leftrightarrow 4a = 5 \Leftrightarrow a = \dfrac{5}{4}\cdot$$
					Vậy $a = \dfrac{5}{4}$ thì hàm số liên tục trên $\mathbb{R}$.
			\item Ta có $y' = \dfrac{4}{(x + 3)^2}\cdot$\\
					Do tiếp tuyến vuông góc với đường thẳng $(d): y = -4x + 3$ nên tiếp tuyến có hệ số góc $k = \dfrac{1}{4}\cdot$\\
					Gọi $(x_0; y_0)$ là tiếp điểm, ta có: $y'(x_0) = k \Leftrightarrow \dfrac{4}{(x_0 + 3)^2} = \dfrac{1}{4} \Leftrightarrow \hoac{&x_0 = 1 \Rightarrow y_0 = 2 \\ &x_0 = -7 \Rightarrow y_0 = 4.}$\\
					\begin{itemize}
						\item Phương trình tiếp tuyến của đồ thị $(C)$ tại điểm $(1; 2)$ là: $y = \dfrac{1}{4}(x - 1) + 2 = \dfrac{1}{4}x + \dfrac{7}{4}\cdot$
						\item Phương trình tiếp tuyến của đồ thị $(C)$ tại điểm $(-7; 4)$ là: $y = \dfrac{1}{4}(x + 7) + 4 = \dfrac{1}{4}x + \dfrac{23}{4}\cdot$
					\end{itemize}
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề cương HK2 (2017-2018), THPT Chuyên Amsterdam, Hà Nội]%[1H3B4-2]%[1H3B3-2]%[1H3B4-3]%[1H3K3-4]%[Đinh Thanh Hoàng]
	Cho hình chóp tứ giác $S.ABCD$ có đáy $ABCD$ là hình thoi tâm $O$, cạnh $a$, góc $ABC$ bằng $60^\circ$. Biết góc giữa đường thẳng $SC$ và mặt phẳng $(ABCD)$ bằng $60^\circ$, $SO \perp (ABCD)$.
	\begin{enumerate}
		\item Chứng minh rằng $(SAC) \perp (SBD)$. Tính độ dài đoạn $SO$ theo $a$.
		\item Trên các cạnh $SB$, $SD$ theo thứ tự lấy các điểm $E$, $F$ sao cho $\dfrac{SE}{SB} = \dfrac{SF}{SD} = \dfrac{2}{3}\cdot$ Chứng minh $EF \perp (SAC)$ và $SC \perp (AEF)$.
		\item Tính góc giữa hai mặt phẳng $(AEF)$ và $(ABCD)$.
		\item Gọi $I$ là giao điểm của đường thẳng $SC$ và mặt phẳng $(AEF)$. Tính diện tích tứ giác $AEIF$.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=1]
				\tkzDefPoints{0/0/O, -4/-1/B, 1/-1/C, 0/6/S}
				\tkzDefPointBy[symmetry = center O](C) 	\tkzGetPoint{A}
				\tkzDefPointBy[symmetry = center O](B) 	\tkzGetPoint{D}
				\tkzDefPointBy[homothety = center S ratio 0.6666667](B) \tkzGetPoint{E}
				\tkzDefPointBy[homothety = center S ratio 0.6666667](D) \tkzGetPoint{F}
				\tkzInterLL(S,O)(E,F) \tkzGetPoint{H}
				\tkzInterLL(A,H)(S,C) \tkzGetPoint{I}
				\tkzDrawPolygon[pattern=dots,dashed](A,E,F)						
				\tkzDrawSegments[dashed](S,A S,O A,B A,C A,D B,D A,I)
				\tkzDrawSegments(S,B S,C S,D B,C C,D I,E I,F)
				\tkzMarkRightAngles[size=.3](S,O,A S,O,D A,O,D)				
				\tkzDrawPoints(O,H,I,E,F)				
				\tkzLabelPoints[above](S)
				\tkzLabelPoints[below](O)
				\tkzLabelPoints[left](E)
				\tkzLabelPoints[right](D,F)
				\tkzLabelPoints[above left](A)
				\tkzLabelPoints[below left](B)
				\tkzLabelPoints[above right](I)
				\tkzLabelPoints(C,H)		
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Ta có $SO \perp (ABCD)$ nên $SO \perp BD$ \hfill $(1)$.\\
					Mặt khác $ABCD$ là hình thoi nên $BD \perp AC$ \hfill $(2)$.\\
					Từ $(1)$ và $(2)$ ta có $BD \perp (SAC) \Rightarrow (SAC) \perp (SBD)$.\\
					Theo giả thiết $ABCD$ là hình thoi tâm $O$, cạnh $a$, $\widehat{ABC} = 60^\circ$ nên $ABC$ là tam giác đều cạnh $a$.\\
					$\Rightarrow OA = OC = \dfrac{a}{2}\cdot$\\
					Mặt khác $SO \perp (ABCD)$ nên $\widehat{\left(SC, (ABCD)\right)} = \widehat{SCO} = 60^\circ$.\\
					Xét tam giác $SOC$ có $SO = OC\cdot\tan 60^\circ = \dfrac{a\sqrt{3}}{2}\cdot$
			\item Xét tam giác $SBD$ có $\dfrac{SE}{SB} = \dfrac{SF}{SD} = \dfrac{2}{3}$ nên $EF \parallel BD$.\\
					Theo chứng minh trên ta có $BD \perp (SAC) \Rightarrow EF \perp (SAC) \Rightarrow EF \perp SC$ \hfill $(3)$.\\
					Gọi $H = SO \cap EF$, do $\dfrac{SE}{SB} = \dfrac{SF}{SD} = \dfrac{SH}{SO} = \dfrac{2}{3}$ nên $H$ là trọng tâm của tam giác $SAC$.\\
					Mặt khác ta lại có tam giác $SAC$ cân tại $S$, $SA = SC = \dfrac{OC}{\cos 60^\circ} = a$\\
					$\Rightarrow$ tam giác $SAC$ đều $\Rightarrow AH \perp SC$ \hfill $(4)$.\\
					Từ $(3)$ và $(4)$ ta có $SC \perp (AEF)$.
			\item Ta có $\widehat{\left((ABCD), (AEF)\right)} = \widehat{HAO}$.\\
					Xét tam giác vuông $HAO$ ta có $\tan\widehat{HAO} = \dfrac{HO}{OA} = \dfrac{SO}{3AO} = \dfrac{\sqrt{3}}{3} \Rightarrow \widehat{HAO} = 30^\circ$.\\
					Vậy $\widehat{\left((ABCD), (AEF)\right)} =  30^\circ$.
			\item Trong mặt phẳng $(SAC)$ gọi $I = AH \cap SC \Rightarrow I = SC \cap (AEF)$.\\
					Ta có $S_{AEIF} = S_{AEF} + S_{IEF}$.\\
					Do $EF \perp (SAC) \Rightarrow EF \perp AI$.\\
					Mặt khác ta lại có $AH = \dfrac{2}{3}AI = \dfrac{a\sqrt{3}}{3}$, $HI = \dfrac{a\sqrt{3}}{6}$, $BD = 2BO = \dfrac{a\sqrt{3}}{2}\cdot$\\
					Do đó $S_{AEIF} = S_{AEF} + S_{IEF} = \dfrac{1}{2}EF(AH + IH) = \dfrac{1}{2}a\sqrt{3}\cdot\dfrac{a\sqrt{3}}{2} = \dfrac{3a^2}{4}\cdot$
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề cương HK2 (2017-2018), THPT Chuyên Amsterdam, Hà Nội]%[1D4K3-6]%[Đinh Thanh Hoàng]
	Cho phương trình $(m^2 + 1)x^3 - 2m^2x^2 - 4x + m^2 + 1 = 0$ ($m$ là tham số). Chứng minh rằng phương trình luôn có ba nghiệm phân biệt với mọi giá trị của tham số $m$.
	\loigiai{
		$(m^2 + 1)x^3 - 2m^2x^2 - 4x + m^2 + 1 = 0 \Leftrightarrow m^2(x^3 - 2x^2 + 1) + x^3 - 4x + 1 = 0$.\\
		Xét hàm số $f(x) = m^2(x^3 - 2x^2 + 1) + x^3 - 4x + 1$ liên tục trên $\mathbb{R}$.\\
		Ta có $f\left(\dfrac{1 + \sqrt{5}}{2}\right) = 1 - \sqrt{5} < 0$, $f\left(\dfrac{1 - \sqrt{5}}{2}\right) = 1 + \sqrt{5} > 0$.\\
		Do $\lim\limits_{x \to -\infty} f(x) = -\infty \Rightarrow$ phương trình đã cho có một nghiệm $x \in \left(-\infty;  \dfrac{1 - \sqrt{5}}{2}\right)$.\\
		Tương tự $\lim\limits_{x \to +\infty} f(x) = +\infty \Rightarrow$ phương trình đã cho có một nghiệm $x \in \left(\dfrac{1 + \sqrt{5}}{2}; +\infty\right)$.\\
		Mặt khác $f\left(\dfrac{1 - \sqrt{5}}{2}\right)\cdot f\left(\dfrac{1 + \sqrt{5}}{2}\right) = -4 < 0$ nên phương trình có một nghiệm $x \in \left(\dfrac{1 - \sqrt{5}}{2}; \dfrac{1 + \sqrt{5}}{2}\right)$.\\
		Vậy phương trình đã cho có ba nghiệm phân biệt với mọi giá trị của tham số $m$.
	}
\end{bt}
