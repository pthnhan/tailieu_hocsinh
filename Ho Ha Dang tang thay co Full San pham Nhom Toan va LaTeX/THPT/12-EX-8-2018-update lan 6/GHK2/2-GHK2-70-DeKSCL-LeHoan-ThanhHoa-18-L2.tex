	\choice
	{\True $6$}
	{$4$}
	{$12$}
	{$7$}
	\loigiai{
		Đặt $\heva{&u=x\\&\mathrm{\,d}v=f'(2x)} \Rightarrow \heva{&\mathrm{\,d}u=\mathrm{\,d}x\\& v = \dfrac{1}{2} f(2x)}$.\\

		Tính $I_1 = \displaystyle\int\limits_0^1  f(2x)\mathrm{\,d}x$.\\
		Đặt $t=2x \Rightarrow \mathrm{\,d}t = 2 \mathrm{\,d}x$. Khi đó $I_1 =\dfrac{1}{2} \displaystyle\int\limits_0^2  f(t)\mathrm{\,d}t = 4$. Do đó $I=8-2=6$.
		
	}
\end{ex}

		Do đó $V_{AB'C'D'}\geq \dfrac{27}{64}V_{ABCD}$. \\
		Đẳng thức xảy ra khi và chỉ khi $\dfrac{AB}{AB'}=\dfrac{AC}{AC'}=\dfrac{AD}{AD'}=\dfrac{4}{3}$.\\
		Vì $B', C', D'$ lần lượt nằm trên  các cạnh $AB, AC, AD$ nên 
		$$\vec{AB'}=\dfrac{3}{4}\vec{AB} \Rightarrow B'\left(\dfrac{7}{4};\dfrac{1}{4};\dfrac{7}{4}\right).$$
		Lúc đó mặt phẳng $(B'C'D')$ đi qua $B'$ và song song mặt phẳng $(BCD)$ nên có phương trình là $$16x+40y-44z+39=0.$$
	}
\end{ex}

\begin{ex}%[Đề KSCL lần 2, Lê Hoàn, Thanh Hóa, 18]%[2D1G3-5]%[Trần Quang Thạnh (12EX-8)]
