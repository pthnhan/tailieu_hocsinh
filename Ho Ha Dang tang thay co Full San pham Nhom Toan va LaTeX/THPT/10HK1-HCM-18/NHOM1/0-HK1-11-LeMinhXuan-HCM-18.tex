		\begin{bt}%[Đề HK1, Lê Minh Xuân, Hồ Chí Minh 17-18]%[QK Trương]%[0D2B2-2]
		Xác định các hệ số $a$, $b$ sao cho đồ thị (d) của hàm số $y=ax + b$ đi qua hai điểm $A(- 1; 1), B(2; - 3)$.
		\loigiai{
			Vì đường thẳng $(d)$ đi qua hai điểm $A(- 1; 1), B(2; - 3)$ nên ta có hệ phương trình là\\
			$$\heva{ & - a + b=1 \\   & 2a + b= - 3}   \Leftrightarrow \heva{ & a= - \dfrac{4}{3} \\   & b= - \dfrac{1}{3}.}$$\\
			Vậy $a= - \dfrac{4}{3}$ và $b= - \dfrac{1}{3}$.		
		}
	\end{bt}
	\begin{bt}%[Đề HK1, Lê Minh Xuân, Hồ Chí Minh 17-18]%[QK Trương]%[0D3B2-4]
		Giải phương trình $\sqrt{x^2+6x+5}=\sqrt{x+5}$.
		\loigiai{
				Phương trình tương đương $\heva{& x+5 \ge 0\\
					& x^2+6x+5=x+5} \Leftrightarrow \heva{&x \ge -5\\ & x^2+5x=0} \Leftrightarrow \heva{&x \ge -5\\ & x=0 \vee x=-5.}$\\
				Vậy tập nghiệm $ S=\left\lbrace -5;0\right\rbrace $.
		}
	\end{bt}
\begin{bt}%[Đề HK1, Lê Minh Xuân, Hồ Chí Minh 17-18]%[QK Trương]%[0D3B2-2]
		Giải phương trình $\left| x^2-x+1 \right| = x+1$.
		\loigiai{
				Phương trình tương đương\\ $\heva {& x + 1\ge 0 \\   & \hoac{& x^2-x+1=x+1 \\ & x^2-x+1=-x-1}} \Leftrightarrow \heva {& x \ge -1 \\   & \hoac{& x^2-2x = 0 \\ & x^2 = -2}} \Leftrightarrow \heva {& x \ge -1 \\   & \hoac{& x = 0 \\ & x = 2}} \Leftrightarrow \hoac{& x = 0 \\ & x = 2.}$\\
				Vậy tập nghiệm $ S=\left\lbrace 0;2\right\rbrace $.		
		}
	\end{bt}
	\begin{bt}%[Đề HK1, Lê Minh Xuân, Hồ Chí Minh 17-18]%[QK Trương]%[0D3K2-6]
		Cho phương trình $ \colon mx^2 +2\left( m -1\right) x + m + 3 =0 $. Tìm $ m $ để phương trình có hai nghiệm phân biệt.
		\loigiai{
			Phương trình có hai nghiệm phân biệt $ \Leftrightarrow \heva{& a \neq 0 \\ & \Delta > 0} \Leftrightarrow  \heva{& m \neq 0 \\ & -20m + 4 > 0} \Leftrightarrow \heva{& m \neq 0 \\ & m < \dfrac{1}{5}.}$\\
			Vậy $ 0 \neq m < \dfrac{1}{5} $.
		}
	\end{bt}
	\begin{bt}%[Đề HK1, Lê Minh Xuân, Hồ Chí Minh 17-18]%[QK Trương]%[0D3B3-3]
		Giải hệ phương trình $ \heva{& \dfrac{3}{x-1} - y + 2 = 0\\ & - \dfrac{1}{x-1} - y = 6} $.
		\loigiai{
			Hệ phương trình tương đương $ \heva{& \dfrac{3}{x-1} - y = - 2\\ & - \dfrac{1}{x-1} - y = 6} \Leftrightarrow \heva{& \dfrac{1}{x-1}= -2 \\ & y = -4} \Rightarrow \heva{& x = \dfrac{1}{2} \\ & y = -4.}$\\
			Vậy hệ phương trình có nghiệm $ \heva{& x= \dfrac{1}{2} \\ & y = -4.} $
		}
	\end{bt}
	\begin{bt}%[Đề HK1, Lê Minh Xuân, Hồ Chí Minh 17-18]%[QK Trương]%[0H2B2-2]%[0H1B4-3]%[0H2B2-1]
		Trong mặt phẳng tọa độ $ Oxy $ cho ba điểm $ A\left( -2; 1\right) , B\left( 3; -2\right) , C\left( 0; 3\right) . $
		\begin{enumerate}
			\item Tính độ dài trung tuyến $ BM $ của tam giác $ ABC $.
			\item Tìm tọa độ điểm $ D $ sao cho $ ABCD $ là hình bình hành.
			\item Tính $ \cos \left( \vec{AB} , \vec{AC}\right) $.
		\end{enumerate}
		\loigiai{
			\begin{enumerate}
				\item Tính độ dài trung tuyến $ BM $ của tam giác $ ABC $.\\
				Vì $ M $ là trung điểm của cạnh $ AC $ nên ta suy ra $ M\left( -1; 2\right) $ và $ \vec{BM} = \left( -4; 4\right) $.\\
				Vậy độ dài trung tuyến $ BM = \sqrt{16+16} = 4\sqrt{2}$.
				\item Tìm tọa độ điểm $ D $ sao cho $ ABCD $ là hình bình hành.\\
				Ta có $ ABCD $ là hình bình hành $ \Leftrightarrow \vec{AB} = \vec{DC} \Leftrightarrow \heva{& x_B - x_A = x_C - x_D \\ & y_B - y_A = y_C - y_D} \Leftrightarrow \heva{& x_D = -5 \\ & y_D = 6.}$\\
				Vậy $ D\left( -5; 6\right) $.
				\item Tính $ \cos \left( \vec{AB} , \vec{AC}\right) $.\\
				Ta có $ \vec{AB}=\left( 5; -3\right)$ và  $\vec{AC}=\left( 2; 2\right) $.\\
				$ \cos \left( \vec{AB} , \vec{AC}\right) =\dfrac{ 5 \cdot 2 + (-3) \cdot 2  }{\sqrt{5^2+3^2} \cdot \sqrt{2^2+2^2}}=\dfrac{\sqrt{17}}{17} $.
			\end{enumerate}
		}
	\end{bt}
	\begin{bt}%[Đề HK1, Lê Minh Xuân, Hồ Chí Minh 17-18]%[QK Trương]%[0D4K1-2]
		Cho $ a, b, c $ là các số thực dương. Chứng minh rằng$ \colon $
		$$ \left( 1+\dfrac{a}{b}\right) \left( 1+\dfrac{b}{c}\right) \left( 1+\dfrac{c}{a}\right) \ge 8. $$
		\loigiai{
			Vì $ a, b, c $ là các số thực dương nên theo bất đẳng thức Cô-si ta suy ra
			\begin{eqnarray*}
				& & 1 + \dfrac{a}{b} \ge 2\sqrt{\dfrac{a}{b}} \quad (1)\\
				& & 1 + \dfrac{b}{c} \ge 2\sqrt{\dfrac{b}{c}} \quad (2)\\
				& & 1 + \dfrac{c}{a} \ge 2\sqrt{\dfrac{c}{a}} \quad (3)
			\end{eqnarray*}
			Từ (1), (2), (3) ta suy ra $ \left( 1+\dfrac{a}{b}\right) \left( 1+\dfrac{b}{c}\right) \left( 1+\dfrac{c}{a}\right) \ge 8\sqrt{\dfrac{a}{b} \cdot \dfrac{b}{c} \cdot \dfrac{c}{a}} = 8$. Dấu bằng xảy ra khi $a=b=c>0$.
		}
	\end{bt}
