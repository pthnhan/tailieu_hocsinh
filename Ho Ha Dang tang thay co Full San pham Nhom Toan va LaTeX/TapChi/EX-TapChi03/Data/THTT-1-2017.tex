	chia hết cho $4$ nên $2p$ chia hết cho $4$.
\end{itemize}	
\end{nx}}

	Cho $ p,q $ là hai số nguyên tố cùng nhau. Đặt $ T=\left[\dfrac{p}{q}\right] +\left[\dfrac{2p}{q}\right] +\cdots +\left[\dfrac{(q-1)p}{q}\right] $.
	\begin{enumerate}
		\item Tính $ T $.
		\item Xác định giá trị $ p,q $ để $ T $ là số nguyên tố, trong đó kí hiệu $ [a] $ là số nguyên lớn nhất không vượt quá $ a $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item  Ta có:	
			{\allowdisplaybreaks
				\begin{align*}
				T&=\left[\dfrac{p}{q}\right] +\left[\dfrac{2p}{q}\right] +\cdots +\left[\dfrac{(q-1)p}{q}\right] \\
				&=\dfrac{p}{q}-\left\{\dfrac{p}{q}\right\}+\dfrac{2p}{q}-\left\{\dfrac{2p}{q}\right\}+\cdots+\dfrac{(q-1)p}{q}-\left\{\dfrac{(q-1)p}{q}\right\}\\
				&=\left( {\sum\limits_{m = 1}^{q - 1} m } \right)\cdot \dfrac{p}{q}- \sum\limits_{k = 1}^{q - 1} \left\{\dfrac{pk}{q}\right\}=\dfrac{q(q-1)}{2}\cdot\dfrac{p}{q}-\sum\limits_{k = 1}^{q - 1} \left\{\dfrac{pk}{q}\right\}\\
				&=\dfrac{p(q-1)}{2}-\sum\limits_{k = 1}^{q - 1} \left\{\dfrac{pk}{q}\right\}. \tag{1}
				\end{align*}}
			Mặt khác, với  $ \forall k=\overline{1,q-1}$ thì
			$\dfrac{pk}{q}+\dfrac{p(q-k)}{q}=\dfrac{pq}{q}=p \in \mathbb{Z} $. Do đó
			$$\left\{\dfrac{pk}{q}\right\}+\left\{\dfrac{p(q-k)}{q}\right\}=\dfrac{pk}{q}+\dfrac{p(q-k)}{q}-\left[\dfrac{pk}{q}\right]-\left[\dfrac{p(q-k)}{q}\right]\in \mathbb{Z}.$$	
			Ta lại có $ \dfrac{pk}{q} \notin \mathbb{Z}, \dfrac{p(q-k)}{q}\notin \mathbb{Z} $ nên
			$ 0<\left\{\dfrac{pk}{q}\right\}<1; \,\, 0<\left\{\dfrac{p(q-k)}{q}\right\}<1 $.\\
			Suy ra 
			$$  0<\left\{\dfrac{pk}{q}\right\}+\left\{\dfrac{p(q-k)}{q}\right\}<2 \Rightarrow \left\{\dfrac{pk}{q}\right\}+\left\{\dfrac{p(q-k)}{q}\right\}=1, \,\, \forall  k=1,2, \dots ,q-1.$$
			Do đó 
			\[\sum\limits_{k = 1}^{q - 1} \left\{\dfrac{pk}{q}\right\}+\sum\limits_{k = 1}^{q - 1} \left\{\dfrac{p(q-k)}{q}\right\}=q-1 \Rightarrow 2   \sum\limits_{k = 1}^{q - 1} \left\{\dfrac{pk}{q}\right\}=q-1 \Rightarrow  \sum\limits_{k = 1}^{q - 1} \left\{\dfrac{pk}{q}\right\}=\dfrac{q-1}{2}.\tag{2}\]
			Từ $ (1) $ và $ (2) $ suy ra
			$$ T=\dfrac{p(q-1)}{2}-\dfrac{q-1}{2}=\dfrac{(p-1)(q-1)}{2}. $$
			\item Để $ T= \dfrac{(p-1)(q-1)}{2}$ là một số nguyên tố thì một trong hai số $ p-1 $ và $ q-1 $ phải là 2 hoặc 1.
			Nếu $ p-1=2 \Leftrightarrow p=3$ thì $ T $ nguyên tố khi và chỉ khi $ q=r+1 $ với $ r $ nguyên tố sao cho $ r+1 $ không chia hết cho 3.
			Nếu $ p-1=1 \Leftrightarrow p=2$ thì $ T $ nguyên tố khi và chỉ khi $ q=2r+1 $ với $ r $ nguyên tố.
			Do ở $ T $, hai số $ p,q $ có vai trò bình đẳng nên các cặp số $ (p,q) $ để $ T $ nguyên tố là:
			\begin{itemize}
				\item $ (p,q) =(3,r+1)$ hoặc $ (p,q)=(r+1,3) $ với $ r $ nguyên tố sao cho $ r+1 $ không chia hết cho 3.
				\item $ (p,q) =(2,2r+1)$ hoặc $ (p,q)=(2r+1,2) $ với $ r $ nguyên tố.
			\end{itemize}
		\end{enumerate}
	}
