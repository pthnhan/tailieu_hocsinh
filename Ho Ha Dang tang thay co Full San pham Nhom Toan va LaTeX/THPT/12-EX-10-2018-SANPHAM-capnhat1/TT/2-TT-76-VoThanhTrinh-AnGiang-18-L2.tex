\begin{ex}%[Thi thử L2, Võ Thành Trinh An Giang, 2018]%[Trần Như Ngọc, dự án (12EX-10)]%[1D1B1-3]

		\end{itemize}

	}

\begin{ex}%[Thi thử L2, Võ Thành Trinh An Giang, 2018]%[Trần Như Ngọc, dự án (12EX-10)]%[2D1B4-1]
	Đồ thị hàm số $y=\dfrac{x^2-1}{x^2+3x+2}$ có tất cả bao nhiêu đường tiệm cận?
	\choice
	{$1$}
	{\True $2$}
	{$3$}
	{$4$}
	\loigiai{
		Tập xác định $\mathscr{D}=\mathbb R\setminus\{-2;-1\}$.\\

		\begin{itemize}
			\item $\lim\limits_{x\to\pm\infty}y=\lim\limits_{x\to\pm\infty}\dfrac{x-1}{x+2}=\lim\limits_{x\to\pm\infty}\dfrac{1-\frac{1}{x}}{1+\frac{2}{x}}=1.$
			\item $\lim\limits_{x\to-2^+}y=\lim\limits_{x\to-2^+}\dfrac{x-1}{x+2}=+\infty.$
			\item $\lim\limits_{x\to-1}y=\lim\limits_{x\to-1}\dfrac{x-1}{x+2}=-2.$
		\end{itemize}
		Vậy đồ thị hàm số có tất cả 2 đường tiệm cận.
	}
