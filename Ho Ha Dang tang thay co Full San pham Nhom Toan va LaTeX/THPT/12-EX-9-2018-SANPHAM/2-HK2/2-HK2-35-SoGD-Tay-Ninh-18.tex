	\begin{ex}%[Đề thi HK2 Sở GD\&ĐT Tây Ninh 2018]]%[MyNguyen, Dự án (12EX-9)]%[2D3K3-1]
		Tính diện tích hình phẳng giới hạn bởi các đường $y=x^2-4x+3; y=0; x=0$ và $x=4$.
		\choice
		{\True $4$}
		{$\dfrac{3}{4}$}
		{$\dfrac{1}{4}$}
		{$\dfrac{4}{3}$}
		\loigiai{
		Xét phương trình $x^2-4x+3=0\Leftrightarrow \hoac{& x=1\\ & x=3}$.	Diện tích hình phẳng cần tìm là 
		$$S= \displaystyle\int\limits_{0}^{4}|x^2-4x+3| \mathrm{\,d}x=\displaystyle\int\limits_{0}^{1}(x^2-4x+3) \mathrm{\,d}x+\displaystyle\int\limits_{1}^{3}(-x^2+4x-3) \mathrm{\,d}x+\displaystyle\int\limits_{3}^{4}(x^2-4x+3) \mathrm{\,d}x $$
		$$=\left(\dfrac{1}{3}x^3-2x^2+3x\right)\Big|_{0}^{1}-\left(\dfrac{1}{3}x^3-2x^2+3x\right)\Big|_{1}^{3}+\left(\dfrac{1}{3}x^3-2x^2+3x\right)\Big|_{3}^{4} =\dfrac{4}{3}+\dfrac{4}{3}+\dfrac{4}{3} = 4.$$
		}
	\end{ex}
