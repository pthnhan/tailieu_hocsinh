\begin{ex}%[Đề KSCL TSĐH Lần 2, Phan Châu Trinh, Đà Nẵng 2018]%%[Dương Xuân Lợi-12Ex10]%[2D3K2-3]
	Giá trị thực dương của tham số $m$ sao cho $\displaystyle\int\limits_0^m {x\mathrm{e}^{\sqrt {x^2+ 1}}} \mathrm{\, d}x = 2^{500} \mathrm {e}^{\sqrt {m^2+ 1}}$ là
	\choice
	{$m=2^{250}\sqrt{2^{500}-2}$} 	
	{$m=\sqrt{{2^{1000}}-1}$} 
	{$m=\sqrt{2^{1000}+1}$}
	{\True $m=2^{250}\sqrt{2^{500}+2}$}
	\loigiai{
		Đặt $t=\sqrt{x^2+1}$ $\Rightarrow t^2=x^2+1$ $\Rightarrow t\mathrm{\,d}t=x\mathrm{\,d}x$.\\
		Đổi cận: $x=0\Rightarrow t=1$, $x=m\Rightarrow t=\sqrt{m^2+1}$.
		$$\Rightarrow I=\displaystyle\int\limits_0^m{x\mathrm{e}^{\sqrt{x^2+1}}}\mathrm{\,d}x
		=\displaystyle\int\limits_1^{\sqrt{m^2+1}}{t\mathrm{e}^t}\mathrm{\,d}t.$$
		Đặt $\heva{&u=t\\&\mathrm{\,d}v=\mathrm{e}^t\mathrm{\,d}t}$ 
		$\Rightarrow \heva{&\mathrm{\,d}u=\mathrm{\,d}t\\&v=\mathrm{e}^t.}$
		\begin{eqnarray*}
			I &=& \left.t\mathrm{e}^t\right|_1^{\sqrt{m^2+1}}-\displaystyle\int\limits_1^{\sqrt{m^2+1}} {\mathrm{e}^t}\mathrm{\,d}t\\
			&=& \sqrt{m^2+1}\cdot{\mathrm{e}^{\sqrt{m^2+1}}}-\mathrm{e}-\left. \mathrm{e}^t \right|_1^{\sqrt{{m^2}+1}}\\
			&=& \sqrt{m^2+1}\cdot\mathrm{e}^{\sqrt{m^2+1}}-\mathrm{e}-\mathrm{e}^{\sqrt{m^2+1}}+\mathrm{e}\\
			&=& \left(\sqrt{m^2+1}-1\right)\mathrm{e}^{\sqrt{m^2+1}}.
		\end{eqnarray*}
		Theo giả thuyết 
		\begin{eqnarray*}
			& & \left(\sqrt{m^2+1}-1\right)\mathrm{e}^{\sqrt{m^2+1}}=2^{500}\mathrm{e}^{\sqrt{m^2+1}}\\
			& \Leftrightarrow & \sqrt{m^2+1}-1=2^{500}\\
			& \Leftrightarrow & m^2+1=\left(2^{500}+1\right)^2\\
			& \Leftrightarrow & m=\sqrt{\left(2^{500}+1\right)^2 - 1}\\
			& \Leftrightarrow & m=\sqrt{\left(2^{500}+1+1\right)\left(2^{500}+1-1\right)}\\
			& \Leftrightarrow & m=2^{250}\sqrt{2^{500}+2}.
		\end{eqnarray*}
	}
\end{ex}

\begin{ex}%[Đề KSCL TSĐH Lần 2, Phan Châu Trinh, Đà Nẵng 2018]%%[Dương Xuân Lợi-12Ex10]%[2D2G6-5]
	Giải bất phương trình $\log_3\dfrac{5x+1}{(x-1)^2}\ge 3x^2-11x+ 3$ ta được tập nghiệm $S$. Biết rằng $S$ có dạng $[a;b]\backslash \{1\}$. Hãy tính $T=(a+ b)-ab$.
	\choice
	{$\dfrac{23}{3}$}
	{$\dfrac{11}{3}$}
	{\True $3$}
	{$\dfrac{10}{3}$}
	\loigiai{
		Điều kiện: $\heva{&x>-\dfrac{1}{5}\\&x\ne1}$.\\
		\begin{eqnarray*}
			& &\log_3\dfrac{5x+1}{(x-1)^2}\ge3x^2-11x+3\\
			& \Leftrightarrow &\log_3(5x+1)-\log _3(x-1)\ge 3(x-1)^2-(5x+1)+1\\
			& \Leftrightarrow &\log_3(5x+1)+(5x+1)\ge\log_3(x-1)^2+3(x-1)^2+\log_3 3\\
			& \Leftrightarrow &\log_3(5x+1)+(5x+1)\ge\log_3\left[3(x-1)^2\right]+3(x-1)^2\\
			&\Leftrightarrow &f(5x+1)\ge f\left[3(x-1)^2\right].
		\end{eqnarray*}
		Xét hàm $f(t)=\log_3t+t$, tới $t>0$.\\
		$f'(t)=\dfrac{1}{t\ln3}+1>0$, với $\forall t>0$ nên hàm số đồng biến trên $(0;+\infty)$.\\
		Nên 
		\begin{eqnarray*}
			& &f(5x+1)\ge f\left[3(x-1)^2\right]\\
			&\Leftrightarrow &5x+1>3(x-1)^2\\
			&\Leftrightarrow &3x^2-11x+2\le 0\\
			&\Leftrightarrow &\dfrac{11-\sqrt{97}}{6} \le x \le \dfrac{11+\sqrt{97}}{6}.
		\end{eqnarray*}
		Kết hợp với điều kiện suy ra tập nghiệm của bất phương trình là $\left[\dfrac{11-\sqrt{97}}{6};\dfrac{11+\sqrt{97}}{6} \right]\backslash \{1\}$.\\
		Vậy $T=(a+b)-ab=3$.
	}
\end{ex}
