		Giải phương trình $x^2+3x+8=(x+5)\sqrt{x^2+x+2}$.
		\loigiai{Ta có $x^2+3x+8= \left(x+ \dfrac{3}{2}\right)^2 +\dfrac{23}{4}>0$. Do đó $x+5>0 \Leftrightarrow x>-5$.
			\\ Phương trình đã cho tương đương với 
			\begin{align*}
			x^2+x-2 = (x+5) \left( \sqrt{x^2+x+2} -2 \right) & \Leftrightarrow 
			x^2+x-2 = (x+5) \cdot \dfrac{x^2+x-2}{\sqrt{x^2+x+2}+2} \\
			& \Leftrightarrow (x^2+x-2) \left( 1- \dfrac{x+5}{\sqrt{x^2+x+2}+2}\right) =0\\
			&\Leftrightarrow \hoac{&x^2+x-2=0 \\ &\dfrac{x+5}{\sqrt{x^2+x+2}+2} =1.}
			\end{align*}
			TH1. $x^2 +x-2=0 \Leftrightarrow \hoac{&x=1 \\ &x=-2}.$ Cả hai nghiệm này đều thỏa mãn.
			\\ TH2. $\dfrac{x+5}{\sqrt{x^2+x+2}+2}=1 \Leftrightarrow \sqrt{x^2+x+2}+2=x+5 \Leftrightarrow \sqrt{x^2+x+2}= x+3$.
			\\ Bình phương hai vế ta có $$x^2+x+2 =x^2+6x+9 \Leftrightarrow x=-\dfrac{7}{5}. \text{ (thỏa mãn)}$$
			Vậy phương trình đã cho có tập nghiệm $S=\left\{ -2;-\dfrac{7}{5};1 \right\}$.
		}
	\end{ex}

		Giải hệ phương trình $\heva{&y^2-2xy=8x^2-6x+1\\&y^2=x^3+8x^2-x+1.}$
		\loigiai{Phương trình thứ nhất tương đương
			\begin{align*}
			(y-x)^2=(3x-1)^2 \Leftrightarrow \hoac{&y-x=3x-1 \\ &y-x=1-3x} \Leftrightarrow \hoac{&y=4x-1 \\ &y=1-2x.}
			\end{align*}
			TH1. $y=4x-1$, thay vào phương trình thứ hai ta có 
			\begin{align*}
			(4x-1)^2 =x^3+8x^2 -x+1 &\Leftrightarrow 16x^2 -8x+1 =x^3 +8x^2 -x+1 \\
			&\Leftrightarrow x^3-8x^2+7x =0 \\
			&\Leftrightarrow x(x-7)(x-1) =0 \\
			&\Leftrightarrow \hoac{&x=0 \\ &x=1 \\&x=7.}
			\end{align*}
			Với $x=0$ thì $y=-1$. Với $x=1$ thì $y=3$. Với $x=7$ thì $y=27$.
			\\ TH2. $y=1-2x$, thay vào phương trình thứ hai ta có
			\begin{align*}
			(1-2x)^2 = x^3 +8x^2 -x+1 &\Leftrightarrow 4x^2-4x+1 = x^3 +8x^2 -x+1 \\
			& \Leftrightarrow x^3 +4x^2 +3x=0 \\
			&\Leftrightarrow x(x+3)(x+1)=0 \\
			&\Leftrightarrow \hoac{&x=0 \\ &x=-1 \\ &x=-3.}
			\end{align*}
			Với $x=0$ thì $y=1$. Với $x=-1$ thì $y=3$. Với $x=-3$ thì $y=7$.
			\\ Vậy hệ phương trình có tập nghiệm $S = \left\{(0;-1), (1;3), (7;27), (0;1), (-1;3) , (-3;7) \right\}$.}

		Cho các số nguyên dương $a$, $b$, $c$, $d$ thỏa mãn $a<b \le c <d$; $ad=bc$ và $\sqrt{d}-\sqrt{a} \le 1$.
		\begin{enumerate}
			\item Chứng minh $a+d>b+c$.
			\item Chứng minh $a$ là một số chính phương.
		\end{enumerate}
		\loigiai{
			\begin{enumerate}
				\item Ta có $a(a+d) -a(b+c) =a^2+ad -ab-ac =a^2+bc -ab-ac =(a-b)(a-c) > 0$.\\
				Do đó $a(a+d) > a(b+c) \Leftrightarrow a+d >b+c$.
				\item Ta có $1 \geq \left(\sqrt{d} -\sqrt{a}\right)^2 =a+d -2\sqrt{ad} \Rightarrow 2\sqrt{ad} \geq a+d-1$.
				\\ Mặt khác $2\sqrt{ad} = 2\sqrt{bc} \leq b+c$.
				\\ Suy ra $b+c \geq a+d -1 >b+c-1$. Do $a+d-1$ là số nguyên nên $a+d-1=b+c$.
				\\ Do đó $2\sqrt{ad} =a+d-1 \Leftrightarrow \sqrt{d} - \sqrt{a} =1 \Leftrightarrow \sqrt{d} =\sqrt{a}+1$.
				\\ Bình phương hai vế ta có $d=a+2\sqrt{a}+1 \Leftrightarrow \sqrt{a} = \dfrac{d-a-1}{2}.$
				\\ Do đó $\sqrt{a}$ là số hữu tỷ. Mà $a$ là số nguyên dương nên $\sqrt{a}$ là số nguyên. Vì vậy $a$ là số chính phương.
		\end{enumerate}}
	\end{ex}

		Với $x$, $y$, $z$ là các số thực thỏa mãn $xyz=1$, chứng minh 
		$$\dfrac{1}{xy+x+1}+\dfrac{1}{yz+y+1} + \dfrac{1}{zx+z+1}=1. $$
		\loigiai{Ta có 
			\begin{align*}
			\dfrac{1}{yz+y+1} &= \dfrac{xyz}{yz+y+xyz} = \dfrac{xz}{xz+z+1} =\dfrac{xz}{xz+z+xyz} =\dfrac{x}{xy+x+1}, \\
			\dfrac{1}{xz+z+1} &= \dfrac{xyz}{xz+z+xyz} = \dfrac{xy}{xy+x+1}.
			\end{align*}
			Từ đây suy ra $$\dfrac{1}{xy+x+1}+\dfrac{1}{yz+y+1} + \dfrac{1}{zx+z+1}= \dfrac{1}{xy+x+1} + \dfrac{x}{xy+x+1} + \dfrac{xy}{xy+x+1} =1. $$}

	\begin{ex}%[Đề thi vào 10 chuyên, thành phố Hà Nội, năm 2018]%[Nguyễn Hữu Nhân, Dự án EX-TS10C-18]%[9D0G7]
		Với $x$, $y$, $z$ là các số thực dương thay đổi và thỏa mãn $\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=3$, tìm giá trị lớn nhất của biểu thức $P=\dfrac{1}{\sqrt{2x^2+y^2+3}} + \dfrac{1}{\sqrt{2y^2+z^2+3}}+ \dfrac{1}{\sqrt{2z^2+x^2+3}}$.
		\loigiai{
			\textbf{Bổ đề: }Cho $a,b,c>0$. Ta có $\dfrac{1}{a} +\dfrac{1}{b} +\dfrac{1}{c} \geq \dfrac{9}{a+b+c}$.
			\\ Thật vậy, áp dụng bất đẳng thức Cauchy - Schwarz ta có $$\left(a+b+c\right) \left( \dfrac{1}{a} + \dfrac{1}{b} + \dfrac{1}{c}\right) \geq 9 \Leftrightarrow  \dfrac{1}{a} +\dfrac{1}{b} +\dfrac{1}{c} \geq \dfrac{9}{a+b+c}.$$
			Dấu đẳng thức xảy ra khi $a=b=c$.
			\\ \textbf{Quay lại bài toán.} 
			\\	Áp dụng bất đẳng thức Cauchy ta có 
			$$2x^2+y^2+3 = 2(x^2+1)+ (y^2+1) \geq 4x +2y.$$
			Do đó $\dfrac{1}{\sqrt{2x^2+y^2+3}} \leq \dfrac{1}{\sqrt{4x+2y}}. \refstepcounter{equation}\hfill (\theequation)\label{1808.1}$
			\\ Áp dụng bất đẳng thức Cauchy ta có 
			\begin{align}
			\dfrac{1}{4x+2y} + \dfrac{1}{6} \geq \dfrac{2}{\sqrt{6(4x+2y)}}. \label{1808.2}
			\end{align}
			\\ Sử dụng bổ đề ở trên ta có $\dfrac{1}{x} +\dfrac{1}{2y}= \dfrac{1}{2x} +\dfrac{1}{2x} +\dfrac{1}{2y} \geq \dfrac{9}{4x+2y}. \refstepcounter{equation}\hfill (\theequation)\label{1808.3}$
			\\ Từ \eqref{1808.1}, \eqref{1808.2} và \eqref{1808.3}, ta có $\dfrac{1}{\sqrt{2x^2+y^2+3}} \leq \dfrac{\sqrt{6}}{2} \left( \dfrac{1}{9x} +\dfrac{1}{18y} +\dfrac{1}{6}\right)$.
			\\ Do đó $P \leq \dfrac{\sqrt{6}}{2} \left( \dfrac{1}{6x} +\dfrac{1}{6y} +\dfrac{1}{6z} +\dfrac{1}{2}\right) = \dfrac{\sqrt{6}}{2}.$
			\\ Dấu đẳng thức xảy ra khi $x=y=z=1$. Vậy giá trị lớn nhất của $P$ là $\dfrac{\sqrt{6}}{2}$.
		}
	\end{ex}

	\begin{ex}%[Đề thi vào 10 chuyên, thành phố Hà Nội, năm 2018]%[Nguyễn Hữu Nhân, Dự án EX-TS10C-18]%[9D0G4]%[9D0G8]
		Cho tập hợp $S=\{ x \in \mathbb{Z}|1 \le x \le 50 \}$. Xét $A$ là một tập hợp con bất kì của tập hợp $S$ và có tính chất: Không có ba phần tử nào của tập hợp $A$ là số đo độ dài ba cạnh của một tam giác vuông.
		\begin{enumerate}
			\item Tìm một tập hợp $A$ có đúng 40 phần tử và thỏa mãn điều kiện đề bài.
			\item Có hay không một tập hợp $A$ có đúng $41$ phần tử và thỏa mãn điều kiện đề bài? Giải thích câu trả lời.
		\end{enumerate}
		\loigiai{
			\begin{enumerate}
				\item \textbf{Bổ đề:} Cho ba số nguyên dương $a,b,c$ thỏa mãn $a^2+b^2=c^2$. Khi đó, $abc$ chia hết cho $5$.
				Ta giả sử $a,b,c$ đều không chia hết cho $5$.
				\\ Một số chính phương khi chia cho $5$ chỉ có thể có số dư là $0$, $1$ hoặc $4$. Do $a,b,c$ không chia hết cho $5$ nên $a^2,b^2,c^2$ chia $5$ có số dư là $1$ hoặc $4$.
				\\ Nếu $a^2$ và $b^2$ có cùng số dư khi chia cho $5$ thì $a^2+b^2$ chia $5$ có số dư là $2$ hoặc $3$ (mâu thuẫn). Vậy $a^2$ và  $b^2$ chia $5$ có số dư khác nhau, do đó $a^2+b^2$ chia hết cho $5$. Suy ra $c$ chia hết cho $5$ (mâu thuẫn).
				\\ Do đó, giả sử phản chứng sai. Vì vậy $abc$ chia hết cho $5$.
				\\ \textbf{Quay lại bài toán}
				\\ Xét $A = \{ x \in \mathbb{Z}| 1\leq x \leq 50 \text{ và } x \ \vdots \ 5\}$. Sử dụng bổ đề trên, ta suy ra tập $A$ không chứa ba phần tử nào tạo thành số đo ba cạnh của một tam giác vuông.
				\item Ta xét $A=S \setminus \{5;20;30;35;50;8;9;24;36\}$.
				\\ Các số chia hết cho $5$ nằm trong $A$ là $10$, $15$, $25$, $40$, $45$.
				\begin{enumerate}
