		&=&-15.
		\end{eqnarray*}

	Giải  phương trình $\sqrt{x}-2=5$.
	\loigiai{
		Điều kiện $x\geq 0$.\\
		$\sqrt{x}-2=5 \Leftrightarrow \sqrt{x}=7\Leftrightarrow x=49$.\\
		Vậy phương trình có nghiệm $x=49$.
	}

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Đông Anh, Hà Nội]%[Tô Ngọc Thy, dự án 9ĐT19-HK1]%[9D1B6]
	Giải phương trình 
	 $\sqrt{25x+25}+3\sqrt{x+1}-\dfrac{1}{2}\sqrt{4x+4}=42$.
	\loigiai{
			Điều kiện $x\geq -1$.\\
			\begin{eqnarray*}
				\sqrt{25x+25}+3\sqrt{x+1}-\dfrac{1}{2}\sqrt{4x+4}&=&42\\
				\Leftrightarrow 5\sqrt{x+1}+3\sqrt{x+1}-\sqrt{x+1}&=&42\\
				\Leftrightarrow 7\sqrt{x+1}&=&42\\
				\Leftrightarrow\sqrt{x+1}&=&6\\
				\Leftrightarrow x+1&=&36\\
				\Leftrightarrow x&=&35.
			\end{eqnarray*}
			Vậy phương trình có nghiệm $x=35$.
	}
\end{bt}
\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Đông Anh, Hà Nội]%[Tô Ngọc Thy, dự án 9ĐT19-HK1]%[9D1K8]
	Cho biểu thức $P=\left( \dfrac{\sqrt{x}}{\sqrt{x}-1}-\dfrac{1}{x-\sqrt{x}}\right) \colon\left( \dfrac{1}{\sqrt{x}+1}+\dfrac{2}{x-1}\right)$.
	\begin{enumerate}
		\item Tìm điều kiện xác định và rút gọn $P$.
		\item Tìm các giá trị của $x$ để $P<0$.
		\item Tìm $m$ để $x$ thỏa mãn $P\cdot\sqrt{x}=m-\sqrt{x}$.
	\end{enumerate} 
	\loigiai{
	\begin{enumerate}
		\item Điều kiện $\heva{&x\geq 0\\&\sqrt{x}-1\neq 0\\&x-\sqrt{x}\neq 0\\&x-1\neq 0}\Rightarrow\heva{&x> 0\\&x\neq 1.}$\\
		\begin{eqnarray*}
		P&=&\left( \dfrac{\sqrt{x}}{\sqrt{x}-1}-\dfrac{1}{x-\sqrt{x}}\right) \colon\left( \dfrac{1}{\sqrt{x}+1}+\dfrac{2}{x-1}\right)\\
		&=&\dfrac{\sqrt{x}\cdot\sqrt{x}-1}{\sqrt{x}\left( \sqrt{x}-1\right)}\colon\dfrac{\sqrt{x}-1+2}{\left(\sqrt{x}-1\right) \left( \sqrt{x}+1\right)}\\
		&=& \dfrac{\left(\sqrt{x}-1\right) \left( \sqrt{x}+1\right)}{\sqrt{x}\left( \sqrt{x}-1\right)}\cdot\dfrac{\left(\sqrt{x}-1\right) \left( \sqrt{x}+1\right)}{\left( \sqrt{x}+1\right)}\\
		&=& \dfrac{\left(\sqrt{x}-1\right) \left( \sqrt{x}+1\right)}{\sqrt{x}}\\
		&=&\dfrac{x-1}{\sqrt{x}}.
		\end{eqnarray*}
		\item Ta có $P<0\Leftrightarrow\dfrac{x-1}{\sqrt{x}}<0\Leftrightarrow\heva{& x-1<0\\&x>0}\Leftrightarrow 0<x<1$.\\
		Vậy $0<x<1$ thì $P<0$.
		\item Ta có 
		\begin{eqnarray*}
		P\cdot\sqrt{x}=m-\sqrt{x}
		&\Leftrightarrow & \dfrac{x-1}{\sqrt{x}}\cdot\sqrt{x}=m-\sqrt{x}\\
		&\Leftrightarrow &x-1=m-\sqrt{x}\\
		&\Leftrightarrow &x+\sqrt{x}=m+1\\
		&\Leftrightarrow &x+\sqrt{x}+\dfrac{1}{4}=m+1+\dfrac{1}{4}\\
		&\Leftrightarrow &\left( \sqrt{x}+\dfrac{1}{2}\right)^2=m+\dfrac{5}{4}.
		\end{eqnarray*}
		Để có số $x$ thỏa mãn $P\cdot\sqrt{x}=m-\sqrt{x}$ thì $m+\dfrac{5}{4}>\dfrac{1}{4}$ hay $m>-1$.\\
		Vì $x\neq 1$ nên $m\neq 1$.\\
		Vậy $m>-1$, $m\neq 1$ thì ta có $x$ thỏa mãn $P\cdot\sqrt{x}=m-\sqrt{x}$.
	\end{enumerate} 	
	}
\end{bt}

	Cho hàm số $y=2x+3$ có đồ thị là $(d)$ và hàm số $y=-0{,}5x-2$ có đồ thị là $(d')$.
	\begin{enumerate}
		\item Vẽ $(d)$ và $(d')$ trên cùng một mặt tọa độ.
		\item Xác định tọa độ giao điểm của hai đồ thị bằng phương pháp đại số.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item \immini{Đồ thị hàm số $y=2x+3$ là đường thẳng $(d)$ cắt trục $Ox$ tại $A(-1{,}5;0)$ và cắt trục $Oy$ tại $B(0;3)$.\\
	Đồ thị hàm số $y=-0{,}5x-2$ là đường thẳng $(d')$ cắt trục $Ox$ tại $C(-4;0)$ và cắt trục $Oy$ tại $D(0;-2)$.}
	{\begin{tikzpicture}[scale=0.7, font=\footnotesize, line join=round, line join=round, line cap = round,>=stealth]
		\draw[->] (-5,0)--(0,0) %
		node[below right]{$O$} circle (1pt) --(2,0) node[below]{$x$};
		\draw[->] (0,-4) --(0,5.5) node[right]{$y$};
		\foreach \x in {-4,-1.5}{
			\draw (\x,0) node[below]{$\x$} circle (1pt);}
		\foreach \y in {-2,3}{
			\draw (0,\y) node[left]{$\y$} circle (1pt);}
		\draw [domain=-3:1, samples=100] %
		plot (\x, {2*(\x)+3}) node[right]{$(d)$};
		\draw [domain=-5:1.5, samples=100] %
		plot (\x, {-0.5*(\x)-2}) node[right]{$(d')$};
		\end{tikzpicture}}
	\item Gọi tọa độ giao điểm của hai đường thẳng $(d)$ và $(d')$ là $H\left( x_H;y_H\right)$.\\
	$H\left( x_H;y_H\right)$ thuộc đường thẳng $(d)\colon y=2x+3$ nên $y_H=2x_H+3$.\\
	$H\left( x_H;y_H\right)$ thuộc đường thẳng $(d')\colon y=0{,}5x-2$ nên $y_H=0{,}5x_H-2$.\\
	Suy ra $2x_H+3=0{,}5x_H-2\Leftrightarrow x_H=-2$.\\
	Thay $x_H=-2$ vào $y_H=2x_H+3 \Rightarrow y_H=-1$.\\
	Vậy $H(-2;-1)$. 
	\end{enumerate}	
	}

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Đông Anh, Hà Nội]%[Tô Ngọc Thy, dự án 9ĐT19-HK1]%[9H2K5]
	Cho nửa đường tròn $(O)$, đường kính $AB$ và điểm $C$ thuộc nửa đường tròn đó. Từ $C$ kẻ $CH$ vuông góc với $AB \left( H\in AB\right) $. Gọi $M$ là hình chiếu của $H$ trên $AC$, $N$ là hình chiếu của $H$ trên $BC$.
	\begin{enumerate}
		\item Chứng minh tứ giác $HMCN$ là hình chữ nhật.
		\item Chứng minh $MN$ là tiếp tuyến của đường tròn đường kính $BH$.
		\item Chứng minh $MN$ vuông góc với $CO$.
		\item Xác định vị trí của điểm $C$ trên nửa đường tròn đường kính $AB$ để đoạn thẳng $MN$ có độ dài lớn nhất?
	\end{enumerate}
	\loigiai{
		\begin{center}
				\begin{tikzpicture}[scale=0.8, font=\footnotesize, line join=round, line join=round, line cap = round,>=stealth]
			\tkzSetUpPoint[fill=black,size=4]
			\tkzInit[ymin=-4,ymax=5,xmin=-0.5,xmax=10.8]
			\tkzClip
			\tkzDefPoints{0/0/A,10/0/B,5/0/O,2/4/C}
			\draw (10,0) arc [radius=5cm, start angle=0, end angle= 180];
			\tkzDefPointBy[projection=onto A--O](C)\tkzGetPoint{H}
			\tkzDefPointBy[projection=onto A--C](H)\tkzGetPoint{M}
			\tkzDefPointBy[projection=onto C--B](H)\tkzGetPoint{N}
			\tkzInterLL(M,N)(C,H)\tkzGetPoint{E}
			\tkzDefMidPoint(H,B)\tkzGetPoint{F}
			\tkzDrawCircle(F,H)
			\tkzLabelPoints[below](O,F)
			\tkzLabelPoints[below left](H)
			\tkzLabelPoints[left](A,E)
			\tkzLabelPoints[right](B)
			\tkzLabelPoints[above](C,N,M)
			\tkzDrawPoints(A,B,O,C,H,M,N,E,F)
			\tkzDrawSegments(A,B O,C C,H A,C B,C H,M H,N M,N F,N F,E)
			\end{tikzpicture}
		\end{center}
	\begin{enumerate}
		\item Điểm $C$ thuộc nửa đường tròn đường kính $AB$ nên $\widehat{ACB}=90^\circ\Rightarrow\widehat{MCN}=90^\circ\quad (1)$.\\
		$M$ là hình chiếu của $H$ trên $AC$ nên $\widehat{HMC}=90^\circ\quad (2)$.\\
		$N$ là hình chiếu của $H$ trên $CB$ nên $\widehat{HNC}=90^\circ\quad (3)$.\\
		Từ $(1)$, $(2)$, $(3)$ suy ra tứ giác $HMCN$ là hình chữ nhật.
		\item Gọi $F$ là trung điểm của $HB$.\\
		$\triangle HNB$ vuông tại $N$ nên $FH=FB=FN=\dfrac{BH}{2}$.\\
		Suy ra $N$ thuộc đường tròn tâm $F$, đường kính $BH\quad (4)$.\\
		Gọi $MN$ giao với $CH$ tại $E$.\\
		Mà tứ giác $HMCN$ là hình chữ nhật nên $EH=EN=EC=EM$.\\
		Xét $\triangle EHF$ và $\triangle ENF$ có
		$$EH=EN \mbox{ (cmt)}$$
		$$EF \mbox{ cạnh chung}$$
		$$FH=FN \mbox{ (cmt)}$$
		Suy ra $\triangle EHF=\triangle ENF$ (c.c.c).\\
		$\Rightarrow \widehat{EHF}=\widehat{ENF}$.\\
		Mà $\widehat{EHF}=90^\circ$ (gt) $\Rightarrow\widehat{ENF}=90^\circ$.\\
		$\Rightarrow MN\perp FN \quad (5)$.\\
		Từ $(4)$ và $(5)$ suy ra $MN$ là tiếp tuyến của đường tròn đường kính $BH$.
		\item  Vì $\triangle FNB$ cân tại $F$ nên $\widehat{FNB}=\widehat{FBN}$.\\
		Vì $\triangle COB$ cân tại $O$ nên $\widehat{OCB}=\widehat{OBC}$.\\
		Suy ra $\widehat{OCB}=\widehat{FNB}$.\\
		Mà $\widehat{OCB}$ và $\widehat{FNB}$ ở vị trí đồng vị $\Rightarrow CO \Arrowvert FN$.\\
		Mặt khác $MN\perp FN$ ($MN$ là tiếp tuyến tại $N$ của đường tròn đường kính $BH$).\\
		Suy ra $MN\perp CO$.
		\item Vì tứ giác $HMCN$ là hình chữ nhật nên $MN=CH$.\\
		$MN$ lớn nhất khi và chỉ khi $CH$ lớn nhất.\\
		$\triangle CHO$ vuông tại $H$ nên $CH\leq CO$.\\
		$CH$ lớn nhất khi và chỉ khi $CH=CO$.\\
		Suy ra $H$ trùng với $O$.\\
		Khi đó $C$ là giao điểm của đường thẳng (qua $O$ và vuông góc với $AB$) với đường tròn tâm $O$ đường kính $AB$.
		
	\end{enumerate}	
	}
