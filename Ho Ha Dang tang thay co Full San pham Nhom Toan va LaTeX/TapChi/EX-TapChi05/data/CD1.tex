
	
\section{Kĩ thuật sử dụng nguyên lí Canto trong toán sơ cấp}
\begin{center}
\textbf{\color{violet}{Trần Phương (Hà Nội)}}
\end{center}	
\noindent Thế kỉ XIX nhà Toán học Đức Canto (Cantor, 23.8.1829-10.4.1920) cha đẻ của ``Lý thuyết Tập hợp'' đã phát biểu nguyên lí nổi tiếng về tính giao khác rỗng của các tập hợp Compăc (hoặc lồi) lồng nhau mà một trường hợp đặc biệt của nó là: ``Mọi dãy các đoạn thẳng lồng nhau trong $\mathbb{R}$ có giao khác rỗng tức là: Nếu có $\mathbb{R}\supset\Delta_1\supset\Delta_2\supset\ldots\supset\Delta_n\supset\Delta_{n+1}\supset\ldots$ thì $\bigcap\limits_{n=1}^{+\infty}\Delta_n\ne\varnothing$.''\\
Nếu có thêm điều kiện thắt nhau tức là $\lim\limits_{n\to+\infty}|\Delta_n|=0$ thì $\bigcap\limits_{n=1}^{+\infty}\Delta_n$ là $1$ điểm duy nhất.\\
Ví dụ: Nếu $\Delta_n=\left[0;1+\dfrac{1}{n}\right]$ thì $\bigcap\limits_{n=1}^{+\infty}\Delta_n=[0;1]$; Nếu $\Delta_n=\left[1-\dfrac{1}{n};1+\dfrac{1}{n}\right]$ thì $\bigcap\limits_{n=1}^{+\infty}\Delta_n=\{1\}$.\\
Nguyên lí Canto tuy đơn giản song được sử dụng rất hiệu quả trong các bài toán chứng minh sự tồn tại. Bài viết này giới thiệu kĩ thuật sử dụng Nguyên lí Canto với hy vọng nó sẽ là một công cụ tốt giúp các bạn học sinh trong các kì thi học sinh giỏi. Để cho tiện ở toàn bộ bài ta quy ước: ``đoạn'' tức là ``đoạn thẳng trong $\mathbb{R}$''.
\subsection{Một số ví dụ giải toán}
\begin{vd}
Chứng minh rằng $\exists\alpha\in\mathbb{R}$ sao cho $\dfrac{1}{3}\le\{\alpha^n\}\le \dfrac{2}{3},\ \forall n\in\mathbb{N}$ (với $\{a\}$ là phần lẻ của số thực  $a\colon 0\le\{a\}=a-[a]<1$).
\loigiai{
\textit{Nhận xét:} Một đoạn $\Delta$ có độ dài $\ge 1$ luôn chứa một đoạn con $\Delta'$ nào đó sao cho 
$$\dfrac{1}{3}\le \{\alpha\}\le\dfrac{2}{3},\ \forall\alpha\in\Delta'.$$
Với $n=1$. Lấy $\Delta_1=\left[\dfrac{10}{3};\dfrac{13}{3}\right]$ thì $\dfrac{1}{3}\le\{\alpha\}\le\dfrac{2}{3},\ \forall\alpha\in\Delta_1$. Ta sẽ xây dựng $\Delta_2\subset\Delta_1$ sao cho $\dfrac{1}{3}\le\{\alpha^2\}\le\dfrac{2}{3},\ \forall\alpha\in\Delta_2$. Vì $\alpha\in\Delta_1=\left[\dfrac{10}{3};\dfrac{13}{3}\right]$ nên $\dfrac{100}{9}\le\alpha^2\le\dfrac{169}{9}$. Độ dài của đoạn $\left[\dfrac{100}{9};\dfrac{169}{9}\right]$ là $\dfrac{69}{9}>1$ nên theo nhận xét trên,  $\exists\Delta_2\subset\Delta_1\colon\dfrac{1}{3}\le\{\alpha^2\}\le\dfrac{2}{3},\ \forall\alpha\in\Delta_2$.  Giả sử ta đã xây dựng được $$\Delta_1\supset\Delta_2\supset\ldots\supset\Delta_n$$
sao cho $\dfrac{1}{3}\le\{\alpha^k\}\le\dfrac{2}{3},\ \forall k=\overline{1,n}$. Ta sẽ xây dựng đoạn $\Delta_{n+1}$ sao cho $$\Delta_{n+1}\subset\Delta_n, \,  \dfrac{1}{3}\le\{\alpha^{n+1}\}\le\dfrac{2}{3},\ \forall\alpha\in\Delta_{n+1}.$$
Thật vậy, vì $\alpha\in\Delta_n$ nên $\exists m_n\in\mathbb{N}\colon m_n+\dfrac{1}{3}\le\alpha^n\le m_n+\dfrac{2}{3}$, tương đương với
$$\left(m_n+\dfrac{1}{3}\right)\alpha\le\alpha^{n+1}\le\left(m_n+\dfrac{2}{3}\right)\alpha.$$
Độ dài đoạn $\left[\left(m_n+\dfrac{1}{3}\right)\alpha;\left(m_n+\dfrac{2}{3}\right)\alpha\right]$ là $\dfrac{\alpha}{3}\ge\dfrac{10}{9}>1$ nên $\exists\Delta_{n+1}\subset\left[\left(m_n+\dfrac{1}{3}\right)\alpha;\left(m_n+\dfrac{2}{3}\right)\alpha\right]$ sao cho
$$\dfrac{1}{3}\le\left\{\alpha^{n+1}\right\}\le\dfrac{2}{3},\ \forall\alpha\in\Delta_{n+1}.$$
Như vậy theo nguyên lí quy nạp ta đã xây dựng được dãy vô hạn các đoạn lồng nhau: $\Delta_1\supset\Delta_2\supset\ldots\supset\Delta_n\supset\Delta_{n+1}\supset\ldots$ và trên mỗi $\Delta_n$ ta có $\dfrac{1}{3}\le\{\alpha^n\}\le\dfrac{2}{3},\ \forall\alpha\in\Delta_n$. Do đó theo nguyên lí Canto thì $\exists\alpha_0\in\bigcap\limits_{n=1}^{+\infty}\Delta_n$. Khi đó $\dfrac{1}{3}\le\{\alpha_0^n\}\le\dfrac{2}{3},\ \forall n\in\mathbb{N}$.
}
\end{vd}

\begin{vd}
Cho $\{a_n\}\colon a_1=2,a_{n+1}=\left[\dfrac{3a_n}{2}\right],\ \forall n\ge 2$. Chứng minh rằng $$\exists\alpha\in\mathbb{R}:a_n=\left[\alpha\cdot\left(\dfrac{3}{2}\right)^n\right]+1,\ \forall n\ge 1.$$
\loigiai{
Điều kiện $a_n=\left[\alpha\cdot\left(\dfrac{3}{2}\right)^n\right]+1\Leftrightarrow a_n-1\le\left(\dfrac{3}{2}\right)^n\cdot\alpha<a_n\Leftrightarrow\dfrac{a_n-1}{\left(\dfrac{3}{2}\right)^n}\le\alpha<\dfrac{a_n}{\left(\dfrac{3}{2}\right)^n}$.\\
Đặt $\Delta_n=\left[\dfrac{a_n-1}{\left(\dfrac{3}{2}\right)^n};\dfrac{a_n}{\left(\dfrac{3}{2}\right)^n}\right]$. Ta sẽ chứng minh $\Delta_{n+1}\subset\Delta_n$.\\
Theo giả thiết $a_{n+1}=\left[\dfrac{3a_n}{2}\right]$ nên $a_{n+1}-1\ge\dfrac{3}{2}(a_n-1)$ và $a_{n+1}\le\dfrac{3}{2}a_n$. Do đó
$$\Delta_{n+1}=\left[\dfrac{a_{n+1}-1}{\left(\dfrac{3}{2}\right)^{n+1}};\dfrac{a_{n+1}}{\left(\dfrac{3}{2}\right)^{n+1}}\right]\subset\Delta_n=\left[\dfrac{a_n-1}{\left(\dfrac{3}{2}\right)^n};\dfrac{a_n}{\left(\dfrac{3}{2}\right)^n}\right].$$
Vậy ta có $\Delta_1\supset\Delta_2\supset\ldots\supset\Delta_n\supset\Delta_{n+1}\supset\ldots$ và trên mỗi $\Delta_n$ ta có
$$\dfrac{a_n-1}{\left(\dfrac{3}{2}\right)^n}\le\alpha<\dfrac{a_n}{\left(\dfrac{3}{2}\right)^n}.$$
Theo nguyên lí Canto thì $\exists\alpha_0\in\bigcap\limits_{n=1}^{+\infty}\Delta_n$ tức là $\dfrac{a_n-1}{\left(\dfrac{3}{2}\right)^n}\le\alpha_0<\dfrac{a_n}{\left(\dfrac{3}{2}\right)^n},\ \forall n\in\mathbb{N}$ hay $a_n=\left[\alpha_0\cdot\left(\dfrac{3}{2}\right)^n\right]+1,\ \forall n$.\\
Qua hai ví dụ trên ta có nhận xét sau:
\begin{nx}
Để chứng minh dạng bài toán: ``$\exists x\in\mathbb{R}$ sao cho mệnh đề $A_n(x)$ đúng $\forall n\in\mathbb{N}$.'' Ta đi xây dựng một dãy vô hạn các đoạn lồng nhau $\Delta_1\supset\Delta_2\supset\ldots\supset\Delta_n\supset\Delta_{n+1}\supset\ldots$ sao cho $\forall x\in\Delta_n$ thì $A_n(x)$ đúng $\forall n\in\mathbb{N}$. Khi đó $x_0\in\bigcap\limits_{n=1}^{+\infty}\Delta_n$ là giá trị để $A_n(x)$ đúng $\forall n\in\mathbb{N}$. Ta cần chú ý mệnh đề $A_n(x)$ hoặc chính là kết luận hoặc được suy ra từ kết luận của bài toán. Có hai cách xây dựng dãy các đoạn lồng nhau: Hoặc xây dựng theo quy nạp; hoặc xây dựng đoạn tổng quát $\Delta_n$ và chứng minh $\Delta_{n+1}\subset\Delta_n,\ \forall n\in\mathbb{N}$.\\
Ở ví dụ 1, mệnh đề $A_n(\alpha)$ là: ``$\alpha\in\mathbb{R}\colon\dfrac{1}{3}\le\{\alpha^n\}\le\dfrac{2}{3}$'': Xây dựng $\Delta_n$ theo quy nạp.\\
Ở ví dụ 2, mệnh đề $A_n(\alpha)$ là: ``$\alpha\in\mathbb{R}\colon\dfrac{a_n-1}{\left(\dfrac{3}{2}\right)^n}\le\alpha<\dfrac{a_n}{\left(\dfrac{3}{2}\right)^n}$'': Xây dựng ngay $\Delta_n$.
\end{nx}
Dựa vào nhận xét này ta có thể sử dụng Nguyên lí Canto trong các ví dụ phong phú hơn:
}
\end{vd}

\begin{vd}
Tồn tại hay không hàm $f:\mathbb{R}\to\mathbb{N}^+$ sao cho $\forall x\in\mathbb{Q}$, $\forall y\notin\mathbb{Q}$ và $|x-y|<2^{-n}$ thì $f(x)\cdot f(y)\ \vdots\  2^n$?
\loigiai{
Giả sử $\exists f$, ta sẽ chứng minh $\exists x_0$ để $f(x_0)\ \vdots\  2^n\ \forall n$ và do đó $f(x_0)=0\notin\mathbb{N}^+$: Vô lí.\\
Từ đó suy ra không tồn tại hàm $f$ thỏa mãn đề bài. Trước hết ta xác định mệnh đề $A_n(x)$ là: ``$x\in\mathbb{R}\colon f(x)\ \vdots\  2^n$.''\\
Ta sẽ xây dựng bằng quy nạp dãy các đoạn lồng nhau $\Delta_0\supset\Delta_1\supset\ldots\supset\Delta_n\supset\ldots$ sao cho $\forall x\in\Delta_n$ ta có $f(x)\ \vdots\  2^n$, ở đây $n=0,1,2,\ldots$ Với $n=0$, lấy bất kì đoạn $\Delta_0\subset\mathbb{R}$. Khi đó $f(x)\ \vdots\  2^0=1\ \forall x\in\Delta_0$. Giả sử ta đã xây dựng được dãy các đoạn $\Delta_0\supset\Delta_1\supset\ldots\supset\Delta_n$. Ta đi xây dựng $\Delta_{n+1}$.\\
Lấy $x_0$ vô tỉ tùy ý trong $\Delta_n$ và gọi $m$ là số mũ của $2$ trong khai triển $f(x_0)$ ra thừa số nguyên tố. Khi đó $\forall y$ hữu tỉ $\in \Delta_n$ sao cho $|x_0-y|<2^{-(m+n+1)}$ ta có $f(x_0)\cdot f(y)\ \vdots\  2^{m+n+1}$. Suy ra $$f(y)\ \vdots\  2^{n+1}\quad \forall y\in \mathbb{Q}_n=\left(\mathbb{Q}\cap \Delta_n\right)\cap \left(x_0-2^{-(m+n+1)},\ x_0+2^{-(m+n+1)}\right).$$
Bây giờ lấy $y_0\in\mathbb{Q}_n$ và gọi $l$ là số mũ của $2$ trong khai triển $f(x_0)$ ra thừa số nguyên tố. Khi đó $\forall x$ vô tỉ $\in \mathbf{I}_n=\Delta_n\setminus \mathbb{Q}_n$ sao cho
$$|x-y_0|<2^{-(l+n+1)}\quad \text{thì}\quad f(x)\cdot f(y_0)\ \vdots\  2^{l+n+1}.$$ 
Suy ra
\begin{align*}
&f(x)\ \vdots\  2^{n+1}\ \forall x\in\left(\Delta_n\setminus \mathbb{Q}_n\right)\cap \left(y_0-2^{-(l+n+1)},\ y_0+2^{-(l+n+1)}\right).\tag{2}
\end{align*}
Từ $(1)$ và $(2)$ suy ra $\forall x$ vô tỉ hay hữu tỉ mà $$x\in \Delta_n\cap \left(x-2^{-(m+n+1)},\ x_0+2^{-(m+n+1)}\right)\cap \left(y_o-2^{-(l+n+1)},\ y_0+2^{-(l+n+1)}\right)=\Delta_{n+1}$$
ta đều có $f(x)\ \vdots\  2^{n+1}$. Rõ ràng $\Delta_{n+1}\subset \Delta_n$ nên theo nguyên lí quy nạp ta có
$$\Delta_0\supset \Delta_1\supset \ldots\supset \Delta_n\supset\Delta_{n+1}\supset \ldots$$
và $\forall x\in\Delta_n$ ta có $f(x)\ \vdots\  2^n$, ở đây $n=0,1,2,\ldots$\\
Theo nguyên lí Canto ta lấy $x_0\in \displaystyle\bigcap\limits_{n=1}^{+\infty}\Delta_n$ thì 
$$f(x_0)\ \vdots\  2^n\quad \forall n\longrightarrow f(x_0)=0 \not\in \mathbb{N}^+.\ \text{Vô lí}.$$
Vậy $\exists f$.
}
\end{vd}
\subsection{Bài tập tự luyện}

\begin{bt}
	Chứng minh rằng không tồn tại hàm $f\colon \mathbb{R}\longrightarrow \mathbb{R^+}$ sao cho 
	$$f(x)\cdot f(y)\leq |x-y|\quad \forall x\in\mathbb{Q},\ \forall y\not\in \mathbb{Q}.$$
	\textit{(Gợi ý: $A_n(x)$ là mệnh đề \lq\lq $f(x)\leq\dfrac{1}{n}$\rq\rq).}
\end{bt}

\begin{bt}
	Chứng minh rằng không tồn tại hàm $f\colon \mathbb{R^+}\longrightarrow\mathbb{R^+}$ sao cho $\forall x\in\mathbb{Q}$, $\forall y\not\in \mathbb{Q}$ thì $$\left[f(x)\right]^{f(y)}\leq |x-y|.$$
\end{bt}


\begin{vd}
	Chứng minh rằng nếu mệnh đề \lq\lq\textit{Giữa hai số chính phương tồn tại một số nguyên tố}\rq\rq\ là đúng thì mệnh đề \lq\lq\textit{Tồn tại một số $\alpha\in\mathbb{R}$ để $\left[\alpha^{2^n}\right]$ nguyên tố $\forall n$}\rq\rq\ cũng đúng.
	\loigiai{
		Mệnh đề $A_n(\alpha)$ là \lq\lq$\alpha\in\mathbb{R}:\left[\alpha^{2^n}\right]$ nguyên tố\rq\rq. Ta sẽ xây dựng đoạn $\Delta_n$ để trên $\Delta_n$ mệnh đề $A_n(\alpha)$ là đúng. Vì đoạn trên $\mathbb{R}$ có tính liên tục nên ta phải xây dựng $\Delta_n$ sao cho $\left[\alpha^{2^n}\right]$ bằng $1$ số nguyên tố cố định $P_n\ \forall \alpha\in \Delta_n$.\\
		Với $n=0$, ta lấy $\Delta_0=\left[2;\dfrac{5}{2}\right]$, suy ra
		$$\left[\alpha^{2^0}\right]=\left[\alpha\right]=2\quad\forall \alpha\in\Delta_0=\left[2;\dfrac{5}{2}\right].$$
		Giả sử ta xây dựng được dãy các đoạn $\Delta_0\supset \Delta_1\supset \ldots \supset \Delta_n$. Ta sẽ xây dựng $\Delta_{n+1}$.\\
		Theo giả thiết qui nạp thì $\forall \alpha\in\Delta_n$ luôn $\exists P_n$ nguyên tố để 
		$$P_n=\left[\alpha^{2^n}\right]\quad \forall\alpha\in \Delta_n\Leftrightarrow P_n^2\leq \alpha^{2^{n+1}}<(P_n+1)^2.$$
		Theo giả thiết mệnh đề thì luôn $\exists P_{n+1}$ nguyên tố mà $P_n^2<P_{n+1}<(P_n+1)^2$. Do đó ta lấy $$\Delta_{n+1}=\left[\sqrt[2^{n+1}]{P_{n+1}};\sqrt[2^{n+1}]{P_{n+1}+\dfrac{1}{2}}\right]\subset \Delta_n.$$
		Khi đó $\forall \alpha\in\Delta_{n+1}$ ta có $\left[\alpha^{2^{n+1}}\right]=P_{n+1}$. Theo nguyên lí qui nạp ta có 
		$$\Delta_0\supset \Delta_1\supset \ldots\supset \Delta_n\supset \Delta_{n+1}\supset \ldots$$
		Lấy $\displaystyle\alpha_0\in\bigcap\limits_{n=0}^{+\infty}\Delta_n$ thì $\left[\alpha_0^{2^n}\right]$ nguyên tố $\forall n$.
	}
\end{vd}


\begin{vd}
	Chứng minh rằng $\forall k$ nguyên dương luôn $\exists\ r>1$ và $r\not\in \mathbb{Z}$ sao cho $\left[r^n\right]\ \vdots\  k\quad \forall n\in\mathbb{N}$.
	\loigiai{
		\textit{Nhận xét:} Một đoạn có độ dài $>k$ luôn chứa đoạn con $\Delta'$ không có số nguyên và $\left[r\right]=k\quad \forall r\in\Delta'$.\\
		Mệnh đề $A_n(r)$ là \lq\lq $r\not\in \mathbb{Z}: \left[r^n\right]\ \vdots\  k$\rq\rq.\\
		Xây dựng $\Delta_1=\left[k+\dfrac{1}{3};k+\dfrac{1}{2}\right]\Rightarrow [r]=k\quad \forall r\in \Delta_1\ (r\not\in\mathbb{Z})$.\\
		Giả sử đã xây dựng $\Delta_1\supset\Delta_2\supset \ldots\supset\Delta_n$. Ta sẽ xây dựng $\Delta_{n+1}$.\\
		Theo cách xây dựng $\Delta_n$ thì luôn $\exists m_n\in \mathbb{N}:\left[r^n\right]=k\cdot m_n\quad \forall r\in\Delta_n$. Suy ra
		$$k\cdot m_n\cdot r\leq r^{n+1}<(k\cdot m_n+1)r.$$
		Độ dài đoạn $\left[k\cdot m_n\cdot r;(k\cdot m_n+1)r\right]$ là $r>k$ (vì $r\in\Delta_1=\left[k+\dfrac{1}{3};k+\dfrac{1}{2}\right]$) nên tồn tại đoạn con $\Delta_{n+1}\subset \Delta_n$ sao cho $\Delta_{n+1}\cap \mathbb{Z}=\varnothing$ và $\left[r^{n+1}\right]=k\cdot m_{n+1}\ \vdots\  k\quad \forall r\in \Delta_{n+1}.$\\
		Vậy ta có dãy các đoạn $\Delta_1\supset \Delta_2\supset \ldots\supset \Delta_n\supset\Delta_{n+1}\supset\ldots$\\
		Lấy $r_0\in \displaystyle\bigcap\limits_{n=1}^{+\infty}\Delta_n$ thì $1<r_0\not\in \mathbb{Z}$ và $\left[r^n\right]\ \vdots\  k\quad \forall n\in\mathbb{N}$ (đpcm).
	}
\end{vd}

\subsection{Bài tập rèn luyện}

\begin{bt}
	Chứng minh rằng $\forall k,\ h\in\mathbb{N^+}$ luôn $\exists\ r>1$ và $r\not\in\mathbb{Z}$ sao cho $\left[r^n\right] \equiv k$ (mod $h$) $\forall n\in\mathbb{N}$.
\end{bt}

\begin{bt}
	Chứng minh rằng $\forall a<b$ luôn $\exists\ x\in[a;b]$ và một dãy nguyên dương $\{n_k\}\longrightarrow +\infty$ với $n_k$ thỏa mãn $|\sin n_k!\pi x|\geq \dfrac{1}{2}\quad \forall n_k$.
\end{bt}

\begin{bt}
	Cho dãy số $\{x_n\}_{n=1}^{+\infty}$. Chứng minh rằng: Nếu $\exists\ c\in\mathbb{R}$ sao cho $\forall \alpha >1$ ta luôn có $$\displaystyle\lim\limits_{n\to +\infty}x_{[\alpha^n]}=c\quad \text{thì}\quad \displaystyle\lim\limits_{n\to+\infty}x_n=c.$$
\end{bt}
