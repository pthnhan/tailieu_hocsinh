\begin{ex}%[Đề tuyển sinh vào 10, SGD Lào Cai, Năm học 2018 - 2019]%[9D1K8]
	Cho biểu thức $P=\left(\dfrac{x-6}{x+3\sqrt{x}}-\dfrac{1}{\sqrt{x}}+\dfrac{1}{\sqrt{x}+3}:\dfrac{2\sqrt{x}-6}{x+1}\right)$ với $x>0$, $x\ne 9$.
	\begin{enumerate}[a)]
		\item Rút gọn biểu thức $P$.
		\item Tìm $x$ để $P=1$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]	
	\item 	Điều kiện: $x>0$, $x\ne9$.
	\begin{align*}
	P&=\left(\dfrac{x-6}{x+3\sqrt{x}}-\dfrac{1}{\sqrt{x}}+\dfrac{1}{\sqrt{x}+3}\right):\dfrac{2\sqrt{x}-6}{x+1}\\
	&=\left(\dfrac{x-6}{\sqrt{x}(\sqrt{x}+3)}-\dfrac{1}{\sqrt{x}}+\dfrac{1}{\sqrt{x}+3}\right):\dfrac{2(\sqrt{x}-3)}{x+1}\\
	&=\dfrac{x-6-(\sqrt{x}+3)+\sqrt{x}}{\sqrt{x}(\sqrt{x}+3)}\cdot \dfrac{x+1}{2(\sqrt{x}-3)}\\
	&=\dfrac{x-6-\sqrt{x}-3+\sqrt{x}}{\sqrt{x}(\sqrt{x}+3)}\cdot\dfrac{x+1}{2(\sqrt{x}-3)}\\
	&=\dfrac{(x-9)(x+1)}{2\sqrt{x}(x-9)}=\dfrac{x+1}{2\sqrt{x}}
	\end{align*}	
	\item Tìm giá trị của $x$ để $P=1$.\\
	Điều kiện: $x>0$, $x\ne 9$.
	\begin{align*}
	P=1&\Leftrightarrow \dfrac{x+1}{2\sqrt{x}}=1\Leftrightarrow x+1=2\sqrt{x}\Leftrightarrow x-2\sqrt{x}+1=0\\
	&\Leftrightarrow \left(\sqrt{x}-1\right)^2=0\Leftrightarrow \sqrt{x}-1=0\Leftrightarrow \sqrt{x}=1\Leftrightarrow x=1 \:(\text{thỏa mãn})
	\end{align*}
	Vậy $x=1$ thì $P=1$.
	\end{enumerate}

}
\end{ex}

\begin{ex}%[Đề tuyển sinh vào 10, SGD Lào Cai, Năm học 2018 - 2019]%[9D2B5]%[9D4K2]
Cho đường thẳng $(d)\colon y=-\dfrac{1}{2}x+2$
		\begin{enumerate}[a)]
			\item Tìm $m$ để đường thẳng $(\Delta)\colon y=(m-1)x+1$ song song với đường thẳng $(d)$.
			\item Gọi $A$, $B$ là giao điểm của $(d)$ với parabol $(P)\colon y=\dfrac{1}{4}x^2$. Tìm tọa độ điểm $N$ nằm trên trục hoành sao cho $NA+NB$ nhỏ nhất.
		\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
	\item Tìm $m$ để đường thẳng $(\Delta)\colon y=(m-1)x+1$ song song với đường thẳng $(d)$.\\
	Đường thẳng $(d)\parallel(\Delta)\Leftrightarrow\begin{cases}
	m-1=-\dfrac{1}{2}\\
	1\ne2 
	\end{cases}
	\Leftrightarrow m=\dfrac{1}{2}$.\\
	Vậy $m=\dfrac{1}{2}$
	\item Gọi $A$, $B$ là giao điểm của $(d)$ với parabol $(P)\colon y=\dfrac{1}{4}x^2$. Tìm tọa độ điểm $N$ nàm trên trục hoành sao cho $NA+NB$ nhỏ nhất.\\
	Hoành độ giao điểm của đường thẳng $(d)$ là $(P)$ là nghiệm của phương trình: $\dfrac{1}{4}x^2=-\dfrac{1}{2}x+2\Leftrightarrow x^2+2x-8=0$\\
	$\Leftrightarrow (x-2)(x+4)=0\Leftrightarrow\left[\begin{array}{c}
	x-2=0\\
	x+4=0
	\end{array}\Leftrightarrow \left[
	\begin{array}{l}
	x=1\Rightarrow y=1\Rightarrow A(2;1)\\
	x=-4\Rightarrow y=4\Rightarrow B(-4;4).
	\end{array}
	 \right.\right.$\\
	 Khi đó $A(2;1)$, $B(-4;4)$ là hai giao điểm của hai đồ thị hàm số.\\
	 Gọi$A'$ là điểm đối xứng với $A$ qua $Ox$ thì $A'(2;-1)$.\\
	 Khi đó ta có: $NA=NA'$ nên $(NA+NB)_{\min}\Leftrightarrow (NA'+NB)_{\min}$\\
	 Mà $A'$, $B$ nằm khác phía với trục $Ox$.\\
	 Nên để $(NA'+NB)_{\min}$ thì $A'$, $B$, $N$ thẳng hàng.\\
	 Từ đó suy ra điểm $N$ cần tìm là giao điểm của đường thẳng $A'B$ với trục hoành: $N(n;0)$\\
	 Gọi phương trình đường thẳng $(d')$ đi qua hai điểm $A'$, $B$ là $y=ax+b$.\\
	 Do $A'$, $B$ thuộc đường thẳng $(d')$ nên ta có hệ phương trình:
	 \[\begin{cases}
	 2a+b=-1\\
	 -4a+b=4
	 \end{cases}
	 \Leftrightarrow
	 \begin{cases}
	 a=-\dfrac{5}{6}\\
	 b=\dfrac{2}{3}
	 \end{cases}
	 \]
	 Ta có phương trình đường thẳng $(d')$ là: $y=-\dfrac{5}{6}x+\dfrac{2}{3}$.\\
	 Khi đó điểm $N$ thuộc đường thẳng $(d')$ và $N\left(\dfrac{4}{5};0\right)$.\\
	 Vậy khi $N\left(\dfrac{4}{5};0\right)$ thì $(NA+NB)_{\min}=A'B=\sqrt{(-4-2)^2+(4+1)^2}=\sqrt{61}$.
\end{enumerate}}
\end{ex}


\begin{ex}%[Đề tuyển sinh vào 10, SGD Lào Cai, Năm học 2018 - 2019]%[9D3B4]%[9D3K4]
	Cho hệ phương trình: $\begin{cases}
	x+ay=3a\\
	-ax+y=2-a^2
	\end{cases}$\qquad (*), với $a$ là tham số.
	\begin{enumerate}[a)]
		\item Giải hệ phương trình $(*)$ khi $a=1$.
		\item Tìm $a$ để hệ phương trình $(*)$ có nghiệm duy nhất $(x;y)$ thỏa mãn $\dfrac{2y}{x^2+3}$ là số nguyên.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}[a)]
		\item Thay $a=1$ vào hệ phương trình ta được:
		\[(I)\Leftrightarrow\begin{cases}
		x+y=3\\
		-x+y=1
		\end{cases}
		\Leftrightarrow
		\begin{cases}
		2y=4\\
		x=3-y
		\end{cases}
		\Leftrightarrow
		\begin{cases}
		y=2\\
		x=3-2=1
		\end{cases}
		\Leftrightarrow \begin{cases}
		x=1\\
		y=2.
		\end{cases}
		\]
		Vậy với $a=1$ thì hệ phương trình có nghiệm duy nhất $(x;y)=(1;2)$.
		\item Tìm $a$ để hệ phương trình (*) có nghiệm duy nhất $(x;y)$ thỏa mãn $\dfrac{2y}{x^2+3}$ là số nguyên.
		\begin{itemize}
			\item Với $a=0$, ta có $(*)\Leftrightarrow \begin{cases}
			x=0\\y=2
			\end{cases}$\\
			Suy ra hệ phương trình có nghiệm duy nhất.
			\item Với $a\ne0$, hệ phương trình có nghiệm duy nhất khi và chỉ $\dfrac{1}{-a}\ne\dfrac{a}{1}$ ($a\ne0$)\\
			$\Leftrightarrow -a^2\ne1$ (luôn đúng).\\
			Vậy hệ phương trình luôn có nghiệm duy nhất với mọi $a$.\\
			Ta có: $(*)\Leftrightarrow \begin{cases}
			x=3a-ay\\-a(3a-ay)+y=2-a^2
			\end{cases}
			\Leftrightarrow
			\begin{cases}
			x=3a-ay\\
			-3a^2+a^2y+y=2-a^2
			\end{cases}$\\
			$\Leftrightarrow \begin{cases}
			x=3a-ay\\y(a^2+1)=2+2a^2
			\end{cases}
			\Leftrightarrow
			\begin{cases}
			x=3a-ay\\
			y=\dfrac{2a^2+2}{a^2+1}=2
			\end{cases}
			\Leftrightarrow
			\begin{cases}
			x=3a-2a=a\\
			y=2
			\end{cases}$.\\
			Suy ra hệ phương trình có nghiệm duy nhất $(x;y)=(a;2)$.\\
			Ta có: $\dfrac{2y}{x^2+3}=\dfrac{2\cdot 2}{a^2+3}=\dfrac{4}{a^2+3}$\\
			$\dfrac{2y}{x^2+3}\in\mathbb{Z}\Leftrightarrow \dfrac{4}{a^2+3}\in\mathbb{Z}\Leftrightarrow \dfrac{4}{a^2+3}\in\mathbb{Z}\Leftrightarrow (a^2+3)\in U(4)$\\
			Mà $U(4)=\{\pm1;\pm2;\pm4\}$.\\
			Lại có: $a^2+3\geq3\forall a\Rightarrow a^2+3=4\Leftrightarrow a^2=1\Leftrightarrow \left[\begin{array}{c}a=1\\a=-1\end{array}\right.$.\\
			Vậy $a=\pm1$ thỏa mãn điều kiện bài toán.
		\end{itemize}
	\end{enumerate}}
\end{ex}


\begin{ex}%[Đề tuyển sinh vào 10, SGD Lào Cai, Năm học 2018 - 2019]%[9D4B5]%[9D4G6]
	Cho phương trình $x^2-2x+m-3=0$\qquad$(1)$, với $m$ là tham số.
	\begin{enumerate}[a)]
		\item Giải phương trình $(1)$ khi $m=0$.
		\item Tìm tất cả các giá trị của $m$ để phương trình $(1)$ có hai nghiệm phân biệt $x_1$, $x_2$ thỏa mãn:
		\[x_1^2+12=2x_2-x_1x_2\]
	\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
	\item Giải phương trình $(1)$ khi $m=0$.\\
	Thay $m=0$ vào phương trình $(1)$ ta có:\\
	$(1)\Leftrightarrow x^2-2x-3=0\Leftrightarrow (x-3)(x+1)=0\Leftrightarrow \left[\begin{array}{c}x-3=0\\x+1=0\end{array}\right.\Leftrightarrow \left[\begin{array}{c}x=3\\x=-1\end{array}\right.$\\
	Vậy với $m=0$ thì phương rình $(1)$ có hai nghiệm phân biệt $x=-1$ và $x=3$.
	\item Tìm tất cả các giá trị của $m$ để phương trình $(1)$ có hai nghiệm phân biệt $x_1$, $x_2$ thỏa mãn:
	\[x_1^2+12=2x_2-x_1x_2\]
	Phương trình $(1)$ có hai nghiệm phân biệt $\Leftrightarrow \Delta'>0\Leftrightarrow 1-m+3>0\Leftrightarrow m<4$.\\
	Ap dụng hệ thức Vi-ét ta có: $\begin{cases}
	x_1+x_2=2\\
	x_1x_2=m-3
	\end{cases}
	\Leftrightarrow
	\begin{cases}
	x_2=2-x_1\qquad(1)\\
	x_1x_2=m-3\qquad(2)
	\end{cases}$\\
	Theo đề bài ta có: $x_1^2+12=2x_2-x_1x_2$\hfill$(3)$\\
	Thế $(1)$ vào $(2)$ ta có:
	\begin{align*}
	&x_1^2+12=2(2-x_1)-x_1(2-x_1)\\
	\Leftrightarrow&x_1^2+12=4-2x_1-2x_1+x_1^2\\
	\Leftrightarrow&-4x_1=8\Leftrightarrow x_1=-2\Rightarrow x_2=2-x_1=4\\
	\Rightarrow&x_1x_2=m-3\Leftrightarrow m-3=-8\Leftrightarrow m=-5\:(\text{thỏa mãn})
	\end{align*}
	Vậy $m=-5$.	
\end{enumerate}}
\end{ex}

\begin{ex}%[Đề tuyển sinh vào 10, SGD Lào Cai, Năm học 2018 - 2019]%[9H3B7]%[9H3K3]%[9H0K7]
	Cho đường tròn $(O)$ đường kính $AB=2R$, $C$ là trung điểm của $OA$ và dây cung $MN$ vuông góc với $OA$ tại $C$. Gọi $K$ là điểm tùy ý trên cung nhỏ $BM$ ($K$ khác $B$, $M$), $H$ là giao điểm của $AK$ và $MN$.
	\begin{enumerate}[a)]
		\item Chứng minh rằng $BCHK$ là tứ giác nội tiếp.
		\item Chứng minh $AH\cdot AK=AM^2$.
		\item Xác định vị trí điểm $K$ để $KM+KN+KB$ đạt giá trị lớn nhất và tính giá trị lớn nhất đó.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\tkzDefPoints{0/0/O, 4/0/B, -4/0/A}	
			\tkzDrawCircle[radius](O,A)
			\coordinate (C) at ($(O)!0.5!(A)$);
			\tkzDefPointBy[rotation = center O angle 120](B)
			\tkzGetPoint{M}	
			\tkzDefPointBy[rotation = center O angle -120](B)
			\tkzGetPoint{N}	
			\tkzDefPointBy[rotation = center O angle 70](B)
			\tkzGetPoint{K}	
			\tkzInterLL(A,K)(C,M)\tkzGetPoint{H}
			\coordinate (I) at ($(M)!0.5!(K)$);
			\tkzInterLL(I,O)(N,K)\tkzGetPoint{E}
			\pgfresetboundingbox
			\tkzDrawSegments(A,B M,N M,A M,B N,B N,K M,K B,K A,K C,M E,M)
			\tkzDrawPoints(O,A,B,C,M,N,K,H,E)
			\tkzLabelPoints[above](K,M)
			\tkzLabelPoints[below](N)
			\tkzLabelPoints[below left](C)
			\tkzLabelPoints[right](B,E)
			\tkzLabelPoints[left](A,H)
			\tkzLabelPoints(O)
			\tkzMarkRightAngles[size=0.3](M,C,O)
			\end{tikzpicture}
		\end{center}
	\begin{enumerate}[a)]
		\item Ta có $\widehat{AKB}=90^{\circ}$ (góc nội tiếp chắn nửa đường tròn).\\
		Xét tứ giác $BCHK$ có: $\widehat{BCH}+\widehat{BKH}=90^{\circ}+90^{\circ}=180^{\circ}$.\\
		Suy ra tứ giác $BCHK$ là tứ giác nội tiếp (Tứ giác có tổng hai góc đối bằng $180^{\circ})$.
		\item Ta có $OA\perp MN$ tại $C$\\
		Suy ra $C$ là trung điểm của $MN$ (quan hệ vuông góc giữa đường kính và dây cung)\\
		$\Rightarrow \triangle AMN$ có $AC$ là đường cao đồng thời là trung tuyến.\\
		$\Rightarrow \triangle AMN$ cân tại $A$. Suy ra $AM=AN$\\
		$\Rightarrow \textrm{sđ}\wideparen{AM}=\textrm{sđ}\wideparen{AN}$.\\
		$\Rightarrow \widehat{AMN}=\widehat{AKM}$ (hai góc nội tiếp chắn hai cung bằng nhau)\\
		Xét tam giác $AMN$ và $AKM$ có: $\begin{cases}
		\widehat{MAK}\:\text{chung}\\
		\widehat{AMN}=\widehat{AKM}\quad (\text{chứng minh trên})
		\end{cases}$\\
		$\Rightarrow \triangle AMH\backsim \triangle AKM$ (g.g)\\
		$\Rightarrow \dfrac{AM}{AK}=\dfrac{AH}{AM}\Rightarrow AM^2=AH\cdot AK$.
		\item Lấy điểm $E$ thuộc $KN$ sao cho $KM=KE$.\\
		Xét tam giác vuông $AMB$ có $AM^2=AC\cdot AB=\dfrac{R}{2}\cdot2R=R^2\Rightarrow AM=R$ (hệ thức lượng trong tam giác vuông)\\
		$\Rightarrow \sin\widehat{ABM}=\dfrac{AM}{AB}=\dfrac{R}{2R}=\dfrac{1}{2}\Rightarrow \widehat{ABM}=30^{\circ}\Rightarrow \widehat{MBN}=60^{\circ}$ (tính đối xứng)\\
		$\Rightarrow \widehat{MKE}=\widehat{MKN}=\widehat{MBN}=60^{\circ}$ (hai góc nội tiếp cùng chắn cung $\wideparen{MN}$).\\
		$\Rightarrow \triangle AKE$ đều $\Rightarrow KM=KE=ME$.\\
		Ta có $AB$ là trung trực của $MN$ $\Rightarrow BM=BN$\\
		Lại có $\widehat{MBN}=60^{\circ}$ (cmt) $\Rightarrow\triangle BMN$ đều $\Rightarrow MB=MN=BN$ và $\widehat{BMN}=60^{\circ}$\\
		$\Rightarrow \widehat{BMN}=\widehat{KME}=60^{\circ}$\\
		$\Rightarrow \widehat{BMN}-\widehat{BME}=\widehat{KME}-\widehat{BME}$\\
		$\Leftrightarrow \widehat{EMN}=\widehat{KMB}$\\
		Xét tam giác $KMB$ và tam giác $EMN$ có:\\
		$\begin{cases}
		KM=EM;\\
		MB=MN;\\
		\widehat{EMN}=\widehat{KMB}\:(\textrm{cmt})
		\end{cases}\Rightarrow \triangle KMB=\triangle EMN$ (c.g.c)\\
		$\Rightarrow KB=EN$ (hai cạnh tương ứng)\\
		$\Rightarrow S=KM+KN+KB=KE+(KE+EN)+EN=2(KE+EN)=2KN$.\\
		$KN$ lớn nhất khi và chỉ khi $KN$ là đường kính của đường tròn $O$, khi đó $KN=2R$ và $S_{\max}=4R$.\\
		$KN$ nhỏ nhất khi và chỉ khi $K\equiv M\Rightarrow KN=MN\Rightarrow S_{\min}=2MN$.\\
		Xét tam giác vuông $AMB$ có $MC^2=AC\cdot BC=\dfrac{R}{2}\cdot \dfrac{3R}{2}=\dfrac{3R^2}{4}\Rightarrow MC=\dfrac{R\sqrt{3}}{2}\Rightarrow MN=R\sqrt{3}\Rightarrow S_{\min}=2R\sqrt{3}$.\\
		Vậy $(KM+KN+KB)_{\max}=4R$ và $(KM+KN+KB)_{\min}=2R\sqrt{3}$.		
	\end{enumerate}}
\end{ex}
