\begin{ex}%[Đề thi tuyển sinh vào lớp 10 năm học 2018 - 2019 tỉnh Nghệ An]%[Nguyễn Hữu Nhân, dự án EXTHCS-06]%[9D1Y6]%[9D1Y8]%[9D2Y3]
	\begin{enumerate}
		\item So sánh $2\sqrt{3}+\sqrt{27}$ và $\sqrt{74}$.
		\item Chứng minh đẳng thức $\left( \dfrac{1}{\sqrt{x}-2} - \dfrac{1}{\sqrt{x}+2} \right) \cdot \dfrac{x-4}{4}=1 $, với $x \ge 0$ và $x \neq 4$.
		\item Tìm giá trị của $m$ để đồ thị hàm số $y=3x+m$ đi qua điểm $A(1;2)$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có
	\begin{align*}
	&(2\sqrt{3}+\sqrt{27})^2=(2\sqrt{3})^2+4\sqrt{3\cdot 27}+(\sqrt{27})^2=12+36+27=75.\\
	&(\sqrt{74})^2=74.
	\end{align*}
	Do $ 75>74$ nên $2\sqrt{3}+\sqrt{27} >\sqrt{74} $.
	\item \begin{eqnarray*}
	& \left( \dfrac{1}{\sqrt{x}-2} - \dfrac{1}{\sqrt{x}+2} \right) \cdot \dfrac{x-4}{4}&=\dfrac{\sqrt{x}+2-\sqrt{x}+2}{x-4}\cdot \dfrac{x-4}{4}\\
	&&=\dfrac{4}{x-4}\cdot \dfrac{x-4}{4}\\
	&&=1.
	\end{eqnarray*}
	\item Để đồ thị hàm số $y=3x+m$ đi qua điểm $ A(1;2) $ thì
	$$2=3\cdot 1+m \Leftrightarrow m=2-3=-1.$$
	Vậy $ m=-1 $.
\end{enumerate}}
\end{ex}

\begin{ex}%[Đề thi tuyển sinh vào lớp 10 năm học 2018 - 2019 tỉnh Nghệ An]%[Nguyễn Hữu Nhân, dự án EXTHCS-06]%[9D4Y3]%[9D4B6]
	Cho phương trình $x^2+2x+m-1=0$ (*), trong đó $m$ là tham số.
	\begin{enumerate}
		\item Giải phương trình (*) khi $m=-2$.
		\item Tìm $m$ để phương trình (*) có hai nghiệm phân biệt $x_1$ và $x_2$ thỏa mãn điều kện $x_1=2x_2$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Khi $ m=-2 $ phương trình trở thành:
\\	$ x^2+2x-3=0 \Leftrightarrow (x+3)(x-1)=0 \Leftrightarrow \hoac{&x=-3\\ &x=1.} $
\\ Vậy phương trình có nghiệm $ x=-3 $ hoặc $ x=1 $ khi $ m=-2 $.
	\item Xét phương trình (*) có:
	$ \Delta'= 1^2-(m-1)=2-m$.
	\\ Để phương trình có hai nghiệm phân biệt $ x_1 $ và $ x_2 $ thì $ \Delta'>0$ hay $2-m>0 \Leftrightarrow m<2 $.
	\\ Khi đó, áp dụng Vi-ét ta được $ \heva{&x_1+x_2=-2\\ &x_1\cdot x_2=m-1.} $
	\\ Theo đề bài $ x_1=2x_2 $, suy ra $ 3x_2=-2 \Leftrightarrow x_2=\dfrac{-2}{3} $.
	\\ Do đó $ m-1=x_1\cdot x_2=2x_2^2=2\cdot \left(\dfrac{-2}{3}\right)^2=\dfrac{8}{9} \Rightarrow m=\dfrac{17}{9}$ (thỏa mãn $ m<2 $).
	\\ Vậy $ m=\dfrac{17}{9} $.
\end{enumerate}}
\end{ex}

\begin{ex}%[Đề thi tuyển sinh vào lớp 10 năm học 2018 - 2019 tỉnh Nghệ An]%[Nguyễn Hữu Nhân, dự án EXTHCS-06]%[9D3B5]
	Nhân ngày sách Việt Nam, $120$ học sinh khối $8$ và $100$ học sinh khối $9$ cùng tham gia phong trào xây dựng "Tủ sách nhân ái". Sau một thời gian phát động, tổng số sách cả hai khối đã quyên góp được là $540$ quyển. Biết rằng mỗi học sinh khối $9$ quyên góp nhiều hơn mỗi học sinh khối $8$ một quyển. Hỏi mỗi khối đã quyên góp được bao nhiêu quyển sách? (Mỗi học sinh trong cùng một khối quyên góp số lượng sách như nhau).
	\loigiai{
	Gọi $ x, y $ lần lượt là số quyển sách mà khối 8 và khối 9 quyên góp được (đơn vị: quyển sách; điều kiện: $ x>0; y>0 $).
\\Vì tổng số sách cả hai khối quyên góp được là $ 540 $ quyển nên ta có $ x+y=540 $ (quyển sách).
\\ Mỗi học sinh khối 8 quyên góp được $ \dfrac{x}{120} $ (quyển sách).
\\ Mỗi học sinh khối 9 quyên góp được $ \dfrac{y}{100} $ (quyển sách).
\\ Theo đề ra ta có $ \dfrac{y}{100}- \dfrac{x}{120}=1 $ (quyển sách).
\\Vậy ta có hệ phương trình
$$\heva{&x+y=540\\ &\dfrac{y}{100}- \dfrac{x}{120}=1} \Leftrightarrow \heva{&x=240 \quad \text{(TM)}\\ &y=300 \quad \text{(TM)}.}$$
Vậy khối 8 đã quyên góp được $ 240 $ quyển sách, khối 9 đã quyên góp được $ 300 $ quyển sách.}
\end{ex}

\begin{ex}%[Đề thi tuyển sinh vào lớp 10 năm học 2018 - 2019 tỉnh Nghệ An]%[Nguyễn Hữu Nhân, dự án EXTHCS-06]%[9D3G3]
	Giải hệ phương trình $\heva{&x(2x-2y+1)=y\\&y+2\sqrt{1-x-2x^2}=2(1+y^2)}.$
	\loigiai{Điều kiện: $1-x-2x^2 \geq 0 \Leftrightarrow -1 \leq x \leq \dfrac{1}{2}$. \\
	Phương trình đầu tiên tương đương với
	\begin{align*}
	2x^2-2xy+x-y =0 \Leftrightarrow (2x+1)(x-y)=0 \Leftrightarrow \left[\begin{aligned} &x=-\dfrac{1}{2} \\ &x=y. \end{aligned}\right.
	\end{align*}
	Nếu $x=-\dfrac{1}{2}$, thay vào phương trình thứ hai ta có $y+2=2(1+y^2) \Leftrightarrow y=2y^2 \Leftrightarrow \left[ \begin{aligned} &y=0 \\ &y=\dfrac{1}{2}.	\end{aligned}\right.$
	\\ Nếu $x=y$, thay vào phương trình thứ hai ta có $y+2\sqrt{1-y-2y^2} =2+2y^2. \hfill (*)$
	\\ Áp dụng bất đẳng thức Cô-si ta có $2\sqrt{1-y-2y^2} \leq (1-y-2y^2)+1=2-y-2y^2$.
	\\ Do đó $y+ 2\sqrt{1-y-2y^2} \leq 2-2y^2 \leq 2+2y^2$.
	\\ Dấu đẳng thức xảy ra khi $y=0$.
	\\ Vì vậy phương trình $(*)$ có nghiệm $y=0$, và khi đó $x=0$ (thỏa mãn).
	\\ Vậy hệ phương trình đã cho có nghiệm $(x;y)$ thuộc tập $S=\left\{ (0;0), \left( -\dfrac{1}{2};0\right), \left( -\dfrac{1}{2}; \dfrac{1}{2}\right) \right\}$.
}
\end{ex}
