\section{LÀM QUEN VỚI SỐ NGUYÊN ÂM}
\section{TẬP HỢP CÁC SỐ NGUYÊN}
\subsection{Tóm tắt lý thuyết}
\subsubsection{Các ví dụ}
\begin{itemize}
\item Nhiệt độ của nước đá đang tan là $0^\circ$C.\\
Nhiệt độ dưới $0^\circ$C được viết với dấu ``$-$'' đằng trước. Chẳng hạn nhiệt độ dưới $0^\circ$C $3$ độ được viết là $-3^\circ$C.
\item Độ cao của mực nước biết là $0$ m.\\
Độ cao thấp hơn mực nước biển được viết với dấu ``$-$'' đằng trước. Chẳng hạn độ cao của đáy vịnh Cam Ranh là $-30$ m.
\end{itemize}
\subsubsection{Trục số}
Ta biểu diễn các số nguyên âm trên tia đối của tia số và ghi các số $-1$, $-2$, $-3$,\ldots\\
Hình dưới đây là một trục số.
\begin{center}
\begin{tikzpicture}[scale=1,>=stealth,x=1cm]
\draw[->] (-4.5,0)node[below]{$x'$}--(5.5,0)node[below]{$x$};
\foreach \x in {-3,-2,-1,0,1,2,3,4}
{\draw (\x,-.1)--(\x,.1) node[below=5pt]{$\x$};}
\end{tikzpicture}
\end{center}
Điểm $0$ (không) gọi là điểm gốc của trục số. Chiều theo mũi tên gọi là chiều dương. Chiều ngược lại gọi là chiều âm.
\subsubsection{Tập hợp các số nguyên}
Tập hợp $\{\ldots; -3; -2; -1; 0; 1; 2; 3;\ldots\}$ gồm các số nguyên âm, số $0$ và các số nguyên dương là tập hợp các số nguyên, kí hiệu là $\mathbb{Z}$.
\subsubsection{Số đối}
\begin{itemize}
\item Số $-1$ và số $1$ là hai số đối nhau.
\item Số $2$ có số đối là $-2$.
\item Số đối của $a$ kí hiệu là $-a$.
\end{itemize}
\subsection{Các dạng toán và phương pháp giải}
\begin{dang}{Biểu thị các đại lượng có hai hướng ngược nhau}
Cần nắm vững quy ước về ý nghĩa của các số mang dấu ``$-$'' và các số mang dấu ``$+$''.
\end{dang}

\begin{vd}%[6D2Y1]
Điền vào chỗ trống $(\ldots)$ trong các câu sau:
\begin{enumEX}{1}
\item Nếu $-30$ m là biểu diễn độ sâu của vịnh Cam Ranh là $30$ m dưới mực nước biển thì $+10$ m biểu diễn độ cao của mực nước sông Hồng về mua mưa là \ldots
\item Nếu $+5$ triệu đồng biểu diễn số tiền lãi thì $-3$ triệu đồng biểu diễn\ldots
\end{enumEX}
\loigiai{
\begin{enumEX}{1}
\item \ldots $10$ m trên mực nước biển.
\item \ldots số tiền lỗ là $3$ triệu đồng.
\end{enumEX}
}
\end{vd}

\begin{vd}%[6D2Y1]
Trong các câu sau, câu nào đúng, câu nào sai?
\begin{enumerate}[(A)]
\item Nếu $-10$ biểu diễn năm thứ $10$ trước công nguyên thì $+2015$ biểu diễn năm $2015$ sau công nguyên.
\item Nếu $+4$ đi-ốp cho biết loại kính dành cho người viễn thị $4$ đi-ốp thì $-3$ đi ốp cho biết loại kính dành cho người cận thị $3$ đi-ốp.
\item Nhiệt độ tại ngăn làm đá của tủ lạnh là $-4^\circ$C cho biết nhiệt độ tại ngăn này là $4^\circ$C dưới $0^\circ$C. Nhiệt độ trong phòng là $+28^\circ$C cho biết nhiệt độ trong phòng là $28^\circ$C trên $0^\circ$C.
\end{enumerate}
\loigiai{
Cả ba câu (A), (B), (C) đều đúng.
}
\end{vd}

\begin{dang}{Biểu diễn số nguyên trên trục số}
Trên trục số thì các điểm biểu diễn số nguyên âm nằm ở bên trái gốc $0$, các điểm biểu diễn số nguyên dương nằm ở bên phải gốc $0$.
\end{dang}

\begin{vd}%[6D2B1]
Trên trục số $x'x$, vẽ các điểm $A$, $B$, $C$ lần lượt biểu diễn các số $-4$, $-1$ và $2$.
\loigiai{
Xem hình dưới
\begin{center}
\begin{tikzpicture}[scale=1,>=stealth,x=1cm]
\draw[->] (-5,0)node[below]{$x'$}--(4,0)node[below]{$x$};
\foreach \x in {-4,-3,-2,-1,0,1,2}{\draw (\x,-.1)--(\x,.1) node[below=5pt]{$\x$};}
\draw (-4,0)node[above]{$A$};
\draw (-1,0)node[above]{$B$};
\draw (2,0)node[above]{$C$};
\end{tikzpicture}
\end{center}
}
\end{vd}

\begin{vd}%[6D2B1]
Trong hình dưới đây mỗi điểm $E$, $F$, $G$, $H$ nằm trên trục số biểu diễn số nào?
\begin{center}
\begin{tikzpicture}[scale=1,>=stealth,x=1cm]
\draw[->] (-6,0)node[below]{$x'$}--(4,0)node[below]{$x$};
\foreach \x in {-5,-4,-3,-2,-1,0,1,2,3}{\draw (\x,-.1)--(\x,.1);}
\draw (-5,0)node[above]{$H$};
\draw (-2,0)node[above]{$G$};
\draw (-3,0)node[above]{$E$};
\draw (3,0)node[above]{$F$};
\draw (-1,0)node[below=5pt]{$-1$};
\draw (0,0)node[below=5pt]{$0$};
\draw (1,0)node[below=5pt]{$1$};
\end{tikzpicture}
\end{center}
\loigiai{
\begin{multicols}{2}
\begin{itemize}
\item Điểm $G$ biểu diễn số $-2$;
\item Điểm $E$ biểu diễn số $-3$;
\item Điểm $H$ biểu diễn số $-5$;
\item Điểm $F$ biểu diễn số $+3$.
\end{itemize}
\end{multicols}
}
\end{vd}

\begin{dang}{Đọc và sử dụng các kí hiệu $\in$; $\notin$; $\subset$; $\mathbb{N}$; $\mathbb{Z}$}
\begin{itemize}
\item Kí hiệu $\in$ ($\notin$) cho biết một số là phần tử (hoặc không là phần tử) của một tập hợp nào.
\item Kí hiệu $\subset$ cho biết một tập hợp là tập con của một tập hợp.
\item Kí hiệu $\mathbb{N}$ chỉ tập hợp các số tự nhiên, kí hiệu $\mathbb{Z}$ chỉ tập hợp các số nguyên.
\end{itemize}
\end{dang}

\begin{vd}%[6D2Y2]
Đọc những điều ghi sau đây và cho biết điều đó đúng hay sai?
\begin{center}
	$-5\in\mathbb{Z};\qquad -5\in\mathbb{N};\qquad \mathbb{N}\subset \mathbb{Z}.$
\end{center}
\loigiai{
\begin{itemize}
\item $-5\in\mathbb{Z}$ đọc là: $-5$ là số nguyên (đúng).
\item $-5\in\mathbb{N}$ đọc là: $-5$ là số tự nhiên (sai).
\item $\mathbb{N}\subset \mathbb{Z}$ đọc là: Tập hợp các số tự nhiên là tập hợp con của tập hợp các số nguyên (đúng).
\end{itemize}
}
\end{vd}

\begin{vd}%[6D2B2]
Điền các kí hiệu thích hợp vào ô trống
\begin{enumEX}{2}
\item $3\boxEX[0.5cm]{\phantom{$\in$}}\mathbb{N}$ và $3\in\boxEX[0.5cm]{\phantom{$\mathbb{Z}$}}$;
\item $-7\in\boxEX[0.5cm]{\phantom{$\mathbb{Z}$}}$ nhưng $-7\boxEX[0.5cm]{\phantom{$\notin$}}\mathbb{N}$;
\item $\mathbb{Z}\cap\mathbb{N}=\boxEX[0.5cm]{\phantom{$\mathbb{N}$}}$.
\end{enumEX}
\loigiai{
\begin{enumEX}{2}
\item $3\boxEX[0.5cm]{$\in$}\mathbb{N}$ và $3\in\boxEX[0.5cm]{$\mathbb{Z}$}$;
\item $-7\in\boxEX[0.5cm]{$\mathbb{Z}$}$ nhưng $-7\boxEX[0.5cm]{$\notin$}\mathbb{N}$;
\item $\mathbb{Z}\cap\mathbb{N}=\boxEX[0.5cm]{$\mathbb{N}$}$.
\end{enumEX}
}
\end{vd}

\begin{vd}%[6D2Y2]
Trong các khẳng định sau, khẳng định nào đúng, khẳng định nào sai?
\begin{enumEX}{4}
\item $-100\in\mathbb{N}$.
\item $-21\in\mathbb{Z}$.
\item $7\notin\mathbb{N}$.
\item $0\in\mathbb{Z}$.
\end{enumEX}
\loigiai{
\begin{enumEX}{2}
\item $-100\in\mathbb{N}$ là khẳng định sai.
\item $-21\in\mathbb{Z}$ là khẳng định đúng.
\item $7\notin\mathbb{N}$ là khẳng định sai.
\item $0\in\mathbb{Z}$ là khẳng định đúng.
\end{enumEX}
}
\end{vd}

\begin{dang}{Tìm số đối của một số cho trước}
Hai số đối nhau chỉ khác nhau về dấu. Số đối của $0$ là $0$.
\end{dang}

\begin{vd}%[6D2Y2]
Tìm số đối của các số sau: $7$; $-11$; $0$; $-(-5)$.
\loigiai{
\begin{itemize}
\item Số đối của $7$ là $-7$.
\item Số đối của $-11$ là $11$.
\item Số đối của $0$ là $0$.
\item Ta có $-(-5)=5$ nên số đối của $-(-5)$ là $-5$.
\end{itemize}
}
\end{vd}

\begin{vd}%[6D2B2]
Trong các câu sau, câu nào đúng, câu nào sai?
\begin{enumEX}{1}
\item $-8$ và $+8$ là hai số đối nhau.
\item $10$ có số đối là $-10$.
\item $-20$ có số đối là $20$.
\item Lấy số đối của một số nguyên rồi lấy số đối của kết quả thì được một số dương.
\end{enumEX}
\loigiai{
\begin{itemize}
\item Các câu a), b), c) là các câu đúng.
\item Câu d) sai, ví dụ số đã cho là $-1$. Lấy số đối của nó được $1$, lấy số đối của số này lại được $-1$ là một số âm.
\end{itemize}
}
\end{vd}

\section{THỨ TỰ TRONG TẬP HỢP CÁC SỐ NGUYÊN}
\subsection{Tóm tắt lý thuyết}
\subsubsection{So sánh hai số nguyên}
\begin{itemize}
\item Khi biểu diễn trên trục số (nằm ngang), điểm $a$ nằm bên trái điểm $b$ thì số nguyên $a$ nhỏ hơn số nguyên $b$.
\item Số nguyên $b$ gọi là \textit{số liền sau} của số nguyên $a$ nếu $a<b$ và không có số nguyên nào nằm giữa $a$ và $b$. Khi đó ta cũng nói $a$ là \textit{số liền trước} của $b$.
\end{itemize}
\subsubsection{Giá trị tuyệt đối của số nguyên}
\begin{itemize}
\item Khoảng cách từ điểm $a$ đến điểm $0$ trên trục số là giá trị tuyệt đối của số nguyên $a$.
\item Nhận xét:
\begin{itemize}
\item Giá trị tuyệt đối của số $0$ là số $0$.
\item Giá trị tuyệt đối của một số nguyên dương là chính nó.
\item Giá trị tuyệt đối của một số nguyên âm là số đối của nó.
\item Trong hai số nguyên âm, số nào có giá trị tuyệt đối nhỏ hơn thì lớn hơn.
\item Hai số đối nhau có giá trị tuyệt đối bằng nhau.
\end{itemize}
\end{itemize}
\subsection{Các dạng toán và phương pháp giải}
\begin{dang}{Tìm giá trị tuyệt đối của một số cho trước và ngược lại}
Dựa vào phần nhận xét ở trên.
\end{dang}

\begin{vd}%[6D2Y3]
Tìm giá trị tuyệt đối của các số sau:
$2; 7; 0; -5; -11.$
\loigiai{
Ta có $|2|=2$; $|7|=7$; $|0|=0$; $|-5|=5$; $|-11|=11$.
}
\end{vd}

\begin{vd}%[6D2B3]
Tìm $x$ biết
\begin{enumEX}{3}
\item $|x|=0$;
\item $|x|=9$;
\item $|x|=-3$.
\end{enumEX}
\loigiai{
\begin{enumEX}{1}
\item $|x|=0$ suy ra $x=0$.
\item $|x|=9$ suy ra $x=\pm 9$ (vì $|9|=|-9|=9$).
\item $|x|=-3$. Không có giá trị nguyên nào của $x$ thỏa mãn điều kiện này (vì giá trị tuyệt đối của một số nguyên phải là một số tự nhiên). 
\end{enumEX}
}
\end{vd}

\begin{vd}%[6D2K3]
Với $x$ là một số nguyên bất kì, cho biết tính đúng, sai của các khẳng định sau:
\begin{enumEX}{1}
\item $|x|\ge 0$;
\item $|x|\in\mathbb{N}$;
\item $|x|\ge x$.
\end{enumEX}
\loigiai{
Cả ba khẳng định trên đều đúng.
}
\end{vd}

\begin{dang}{So sánh các số nguyên}
Dựa vào các nhận xét sau:
\begin{itemize}
\item Số nguyên dương lớn hơn số $0$.
\item Số nguyên âm nhỏ hơn số $0$ và nhỏ hơn số dương.
\item Trong hai số nguyên âm, số nào có giá trị tuyệt đối lớn hơn thì số ấy nhỏ hơn.
\end{itemize}
\end{dang}

\begin{vd}%[6D2B3]
Sắp xếp các số nguyên sau theo thứ tự tăng dần:
$7;-3;31;-45;0.$
\loigiai{
Ta có $-45<-3<0<7<31$.
}
\end{vd}

\begin{vd}%[6D2B3]
Sắp xếp các số nguyên sau theo thứ tự giảm dần:
$-7;6;0;-12;-1;10.$
\loigiai{
Ta có $10>6>0>-1>-7>-12$.
}
\end{vd}

\begin{vd}%[6D2K3]
Viết các số sau:
\begin{enumEX}{1}
\item Số nguyên âm lớn nhất có hai chữ số;
\item Số nguyên âm nhỏ nhất có ba chữ số.
\end{enumEX}
\loigiai{
\begin{enumEX}{1}
\item Số nguyên âm lớn nhất có hai chữ số là $-10$.
\item Số nguyên âm nhỏ nhất có ba chữ số là $-999$.
\end{enumEX}
}
\end{vd}

\begin{vd}%[6D2B3]
Cho $a$ là số nguyên dương, $b$ là số nguyên âm. Khi đó, cách viết nào dưới đây đúng?
\choice{\True $b<0<a$}{$0<b<a$}{$0<a<b$}{$b<a<0$}
\loigiai{
Vì $b$ là số nguyên âm nên $b<0$, $a$ là số nguyên dương nên $a>0$. Do đó cách viết đúng là $b<0<a$.
}
\end{vd}

\begin{vd}%[6D2K3]
Viết bốn số nguyên liên tiếp, trong đó có số $0$.
\loigiai{
Có bốn trường hợp:
\begin{multicols}{4}
\begin{itemize}
\item $-3;-2;-1;0$.
\item $-2;-1;0;1$.
\item $-1;0;1;2$.
\item $0;1;2;3$.
\end{itemize}
\end{multicols}

}
\end{vd}

\begin{dang}{Tìm các số nguyên thuộc một khoảng cho trước}
Dựa vào quy tắc so sánh các số nguyên để chọn ra các số nguyên thích hợp.
\end{dang}

\begin{vd}%[6D2B3]
Viết tập hợp $M=\{ x\in\mathbb{Z}\,| -8<x<3\}$ bằng cách liệt kê các phần tử của nó.
\loigiai{
$$M=\{-7;-6;-5;-4;-3;-2;-1;0;1;2\}.$$
}
\end{vd}

\begin{vd}%[6D2B3]
Cho $E=\{ x\in\mathbb{Z}\,| -10<x\le 10\}$; $F=\{ x\in\mathbb{Z}\,| x\le -5\}$. Viết tập hợp $E\cap F$.
\loigiai{
Ta có
\begin{eqnarray*}
E & = & \{-9;-8;-7;-6;-5;-4;\ldots ;10\};\\
F & = & \{-5;-6;-7;-8;-9;-10;\ldots\}. 
\end{eqnarray*}
Do đó $E\cap F=\{-9;-8;-7;-6;-5\}$.
}
\end{vd}

\begin{vd}%[6D2K3]
Tìm các số nguyên $x$ sao cho $1<|x|\le 5$.
\loigiai{
Vì $1<|x|\le 5$ nên $|x|\in \{2;3;4;5\}$.
Do đó $x\in\{\pm 2;\pm 3;\pm 4; \pm 5\}$.
}
\end{vd}
\subsection{Bài tập tự luyện}
\begin{bt}%[6D2Y3]
Tìm giá trị tuyệt đối của các số sau: $0$; $-3$; $5$; $-2$; $4$.
\loigiai{
Giá trị tuyệt đối của các số $0$; $-3$; $5$; $-2$; $4$ lần lượt là $0$; $3$; $5$; $2$; $4$.
}
\end{bt}

\begin{bt}%[6D2B3]
Sắp xếp các số nguyên
\begin{enumEX}{1}
\item Theo thứ tự tăng dần: $7$; $-12$; $3$; $-5$; $-1$;
\item Theo thứ tự giảm dần: $-4$; $6$; $0$; $-7$; $-1$.
\end{enumEX}
\loigiai{
Ta có
\begin{enumEX}{1}
\item $-12<-5<-1<3<7$;
\item $6>0>-1>-4>-7$.
\end{enumEX}
}
\end{bt}

\begin{bt}%[6D2B3]
Viết tập hợp các số nguyên $x$ sao cho
\begin{enumEX}{2}
\item $-5<x<2$;
\item $-3\le x\le 3$.
\end{enumEX}
\loigiai{
\begin{enumEX}{1}
\item Tập hợp các  số nguyên thỏa mãn $-5<x<2$ là
$$\{-4;-3;-2;-1;0;1\}.$$
\item Tập hợp các số nguyên thỏa mãn $-3\le x\le 3$ là
$$\{-3;-2;-1;0;1;2;3\}.$$
\end{enumEX}
}
\end{bt}

\begin{bt}%[6D2K3]
Tìm các số nguyên $x$ biết
\begin{enumEX}{3}
\item $|x|=5$;
\item $|x|=-4$;
\item $|x|<3$.
\end{enumEX}
\loigiai{
\begin{enumEX}{1}
\item $|x|=5$ suy ra $x=\pm 5$ (vì $|5|=|-5|=5$).
\item $|x|=-3$. Không có giá trị nguyên nào của $x$ thỏa mãn điều kiện này (vì giá trị tuyệt đối của một số nguyên phải là một số tự nhiên).
\item $|x|<3$ suy ra $|x|\in\{0;1;2\}$. Do đó $x\in\{ 0;\pm 1;\pm 2\}$.
\end{enumEX}
}
\end{bt}

\begin{bt}%[6D2G3]
Tìm các cặp số nguyên $x$, $y$, biết rằng $|x|+|y|=1$.
\loigiai{
Ta có $|x|$; $|y|$ là các số tự nhiên và $|x|+|y|=1$ nên $|x|=0$; $|y|=1$ hoặc $|x|=1$; $|y|=0$.
\begin{itemize}
\item Với $|x|=0$ và $|y|=1$ ta được $x=0$ và $|y|=\pm 1$.\\
Do đó ta được hai cặp số $(0;1)$, $(0;-1)$.
\item Với $|x|=1$ và $|y|=0$ ta được $x=\pm 1$ và $y=0$.\\
Do đó ta được hai cặp số $(1;0)$, $(-1;0)$.
\end{itemize}
Tóm lại ta được bốn cặp số là $(0;1)$, $(0;-1)$, $(1;0)$, $(-1;0)$.
}
\end{bt}
