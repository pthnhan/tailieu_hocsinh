
\section{PHƯƠNG TRÌNH MŨ}

\begin{vd}
Tìm nghiệm tự nhiên của phương trình $2^x+3=y^2$.
\loigiai{
$\bullet$ Nếu $x \geq 2$ thì $2^x~\vdots~ 4$, do đó vế trái chia $4$ dư $3$, còn  $y$ lẻ nên vế phải chia $4$ dư $1$, không thỏa mãn. Vậy $x \in  \{0;1\}$.\\
$\bullet$ Nếu $x=0$ thì $y^2=4$, nên  $y=2$.\\
$\bullet$ Nếu $x=1$ thì $y^2=5$, không có nghiệm tự nhiên. \\
Đáp số: Nghiệm $(x;y)$ là $(0;2)$. 
 }
\end{vd}
\textbf{Kinh nghiệm giải toán} \\
$\bullet$ Trong ví dụ trên, trước hết ta tìm số tự nhiên $k$ để với $x \geq k$ thì phương trình không có nghiệm nguyên. Sau đó xét $x \in  \{0;1; \dots ; k-1\}$. \\
$\bullet$    Khi tìm số dư trong phép chia lũy thừa của một số nguyên cho một số nguyên, ta thường dùng bổ đề sau với $a,b$ là các số nguyên. \\ 
\textbf{Bổ đề 1.} $(a^n-b^n) ~\vdots~ (a-b)$ với $n$ là số tự nhiên.\\
\textbf{Bổ đề 2.} $(a^n+b^n) ~\vdots~ (a+b)$ với $n$ là số tự nhiên lẻ.\\
\textbf{Bổ đề 3.} $(a+b)^n=ak+b^n$ với $n$ là số tự nhiên, $k$ là số nguyên nào đó. \\
\begin{vd}
Giải phương trình với nghiệm tự nhiên 
\begin{center}
$2^x+57=y^2$
\end{center}
\loigiai{
Xét hai trường hợp: \\
a) $x$ lẻ. Đặt $x=2n+1~(n \in \mathbb{N})$. Ta có 
\[2^x =2^{2n+1} =2 \cdot 4^n =2(3+1)^n =2(3k+1) =6k+2~(k \in \mathbb{Z}) \]
Khi đó vế trái của phương trình đã cho là số chia hết cho $3$ dư $2$, còn vế phải là số chình phương, chia cho $3$ không dư $2$, loại. \\
b) $x$ chẵn. Đặt $x=2n~(n \in \mathbb{N})$. Ta có 
\[y^2-2^{2n} =57 \Leftrightarrow (y-2^n)(y+2^n)=3 \cdot 19\]
Ta thấy $y+2^n >0$ nên $y-2^n >0$ và $y+2^n >y-2^n$. \\
Từ đó có bảng giá trị sau 
\begin{center}
\begin{tabular}{|c|c|c|}
\hline 
$y+2^n$ & $57$ & $19$ \\ 
\hline 
$y-2^n$ & $1$ & $3$ \\ 
\hline 
$2^n$ & $28$, loại & $8$ \\ 
\hline 
$n$ &  & $3$ \\ 
\hline 
$y$ &  & $11$ \\ 
\hline 
$x=2n$ &  & $6$ \\ 
\hline 
\end{tabular} 
\end{center}
Ta có $2^6 +57 =11^2$. \\
Đáp số: nghiệm $(x;y)$ của phương trình là $(6;11)$.\\
\textbf{Lưu ý.} Sử dụng bổ  đề $2$ để chứng minh $2^x+57$ chia hết cho $3$ dư $2$ trong trường hợp $x$ là số lẻ như sau. \\
Vì $x$ là số lẻ nên $(2^x+1)~\vdots~(2+1)$, do đó $2^x +57 =(2^x+1) +56$ chia cho $3$ dư $2$.
}
\end{vd}
 

\begin{vd}
Tìm nghiệm tự nhiên của phương trình 
\[8^x-37=y^3 \]
\loigiai{
$8^x-37=y^3 \Leftrightarrow (2^x)^3-y^3=37 \Leftrightarrow (2^x-y)(2^{2x} + y \cdot 2^x+y^2) =37 \quad (1)$ \\
Do $2^{2x} + y \cdot 2^x+y^2 >0$ nên $2^x-y$ là ước tự nhiên của $37$ và $2^x-y < 2^{2x} + y \cdot 2^x+y^2$. Do đó $2^x -y=1$. Thay vào $(1)$ được $(y+1)^2+y(y+1) +y^2 =37 \Leftrightarrow y^2+y=12 \Leftrightarrow y(y+1)=12$. Do đó $y=4$, $x=2$.  \\
Đáp số: Nghiệm $(x;y)$ là $(2;3)$. 
}
\end{vd}

\begin{vd}
Tìm nghiệm tự nhiên của phương trình 
\[2^y = 1+x+x^2+x^3 \tag{1}\]
\loigiai{
\textbf{Cách 1}. $(1) \Leftrightarrow 2^y = (x+1)(x^2+1)$.\\
$x+1$ và $x^2+1$ là các ước tự nhiên của $2^x$ nên 
\[ \heva{&x+1=2^m  \quad &(2)\\&x^2+1=2^n \quad &(3)\\& m+n=y} \quad \textrm{với $m,n \in \mathbb{N}$}\] 
Rút $x$ từ $(2)$ và thay vào $(3)$  được 
\begin{align*}
(2^m-1)^2+1 =2^n & \Leftrightarrow 2^{2m} -2 \cdot 2^n +1+1=2^n \\
                               & \Leftrightarrow 2^{2m}-2^{m+1} +2 =2^n \tag{4}
\end{align*}
$\bullet$ Nếu $m \geq 2$ thì $2^{2m}$ và $2^{m+1}$ chia hết cho $4$ nên vế trái của $(4)$ chia hết cho $4$ dư $2$. Mặt khác $m \geq 2$ nên từ $(2)$ suy ra $x \geq 3$, từ $(3)$ suy ra $2^n=x^2+1 \geq 3^2+1 =10$ nên $n \geq 4$, do đó vế phải của $(4)$ chia hết cho $4$, không thỏa mãn. \\
$\bullet$ Nếu $m=1$ thì từ $(2)$ suy ra $x=1$. Thay vào $(1)$ được $2^x=4$ nên $y=2$. \\
$\bullet$ Nếu $m=0$ thì từ $(2)$ suy ra $x=0$. Thay vào $(1)$ được $2^x=1$ nên $y=0$. \\
\textbf{Cách 2}. $(1) \Leftrightarrow 2^y = (x+1)(x^2+1)$ suy ra $x^2+1=2^n~(5)$. Với $n \in \mathbb{N}$.\\
$\bullet$ Nếu $n=0$ thì $x=0$ nên $y=0$. \\
$\bullet$ Nếu $n=1$ thì $x=1$ nên $y=2$. \\
$\bullet$ Nếu $n \geq 2$ thì vế phải của $(5)$ chia hết chia $4$, còn vế phải của $(2)$ chia $4$ dư $1$ hoặc dư $2$. không thỏa mãn. \\
Đáp số: Nghiệm $(x;y)$ là $(1;2)$ hoặc $(0;0)$.
}
\end{vd}

\begin{vd}
Giải phương trình sau với nghiệm tự nhiên
\[2^x+2^y+2^z=1024 \tag{1}\]
\loigiai{
Do vai trò của $x,y,z$ như nhau, ta giả sử $x  \leq y \leq z$. \\
Chia  hai vế của $(1)$ cho $2^x \neq 0$ ta được  
\[1+2^{y-x} +2^{z-x} =2^{10-x} \tag{2}\]
Do $2^{10-x}$ nên $2^{10-x}$ là bội của $2$. Ta lại có $z>x$, vì nếu $z=x$ thì $x=y=z$, khi đó $(2)$ trở thành 
$1+2^0+2^0=2k$ với $k$ nguyên, loại. Từ đó $2^{10-x}$ là bội của $2$, suy ra $1+2^{y-x}$ là bội của $2$. Do đó $2^{y-x} =1$, vậy $y=x$. \\
Thay vào $(2)$ được 
\begin{align*}
1+1+2^{z-x} =2^{10-x} & \Leftrightarrow 2+2^{z-x} = 2^{10-x}  \\
                                      & \Leftrightarrow 2(1+2^{z-x-1}) =2^{10-x} \Leftrightarrow 1+2^{z-x-1} =2^{9-x}
\end{align*}  
Do $2^{9-x} >1$ nên $2^{9-x}$ là bội của $2$. Do đó $2^{z-x-1} =1$ và $2=2^{9-x}$. Từ đó $x=8;y=8;z=9$. \\
Đáp số: Nghiệm $(x;y;z)$ là $(8;8;9), (8;9;8), (9;8;8)$. \\
\textbf{Lưu ý}. \\
a) Do $2^{10}$ là lũy thừa của $2$ có số mũ không quá lớn nên có thể giải ví dụ trên bằng cách xét các lũy thừa của $2$ với số mũ từ $0$ đến $0$, đó là: $1,2,4,8,16,32,64,128,256,512$ rồi bằng lập luận chọn ra  $256+256+512=1024$, tức là $2^8+2^8+2^9=1024$. \\
b) Ta có bài toán tổng quát hơn ví dụ $51$. \\
Giải phương trình sau với nghiệm tự nhiên 
\[2^x+2^y+2^z=2^n\]
trong đó $n$ là số tự nhiên cho trước $(n\geq 2)$. \\
Giải tương tự như trên, ta được $x=y=n-2$, $z=n-1$. 
}
\end{vd}

\begin{bt}
Tìm các số tự nhiên $x$ sao cho  
\[3^x+4^x=5^x.\]
\loigiai{
Phương trình không có nghiệm $x=0$, $x=1$. \\
Phương trình có nghiệm $x=2$. \\
Với $x \geq 3$, viết phương trình dưới dạng $\left (\dfrac{3}{5} \right )^x + \left (\dfrac{4}{5} \right )^x =1$.\\
Ta có: $\left (\dfrac{3}{5} \right )^x  < \left (\dfrac{3}{5} \right )^2 $;   $\left (\dfrac{4}{5} \right )^x  < \left (\dfrac{4}{5} \right )^2 $ suy ra $\left (\dfrac{3}{5} \right )^x +\left (\dfrac{4}{5} \right )^x < \left (\dfrac{3}{5} \right )^2  +\left (\dfrac{4}{5} \right )^2  =1$. \\
Vậy phương trình có một nghiệm duy nhất $x=2$. 
}
\end{bt}

\begin{bt}
Giải mỗi phương trình trình sau với nghiệm tự nhiên
\begin{multicols}{3}
\begin{enumerate}
\item $5^x+48=y^2.$ 
\item $3^x+8=y^2.$
\item $2^x-1=y^2.$  
\item $4^x+5=y^2.$  
\item $2^x+45=y^2.$  
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item Với $x=0$ thì $y^2=49$ nên $y=7$.\\
Với $x \geq 1$ thì vế trái tận cùng bằng $3$ nên $y^2$ tận cùng bằng $3$ loại. \\
Đáp số: Nghiệm $(x;y)$ là $(0;7)$.
\item  Với $x=0$ thì $y=3$. \\
Với $x \geq 1$ thì $3^x+8$ chia $3$ dư $2$, không là số chính phương, loại. \\
Đáp số : Nghiệm $(x;y)$ là $(0;3)$. 
\item Lần lượt xét $x=0$, $x=1$, $x \geq 2$.\\
Đáp số: Nghiệm $(x;y)$ là $(0;0)$, $(1;1)$. 
\item Lần lượt xét $x=0$, $x=1$, $x \geq 2$.\\
Chú ý rằng với $x \geq 2$ thì vế trái chia cho $8$ dư $5$, do đó vế phải là số lẻ và chia $8$ dư $1$. \\
Đáp số: Nghiệm $(x;y)$  là $(1;3)$.
\item Lần lượt xét $x=0$, $x=1$, $x=2$, $x \geq 3$. \\
Đáp số: Nghiệm $(x;y)$ là $(2;7)$. 
\end{enumerate}
}
\end{bt}

\begin{bt}
Giải mỗi phương trình trình sau với nghiệm tự nhiên
\begin{multicols}{2}
\begin{enumerate}
\item $x^4=x^y.$ 
\item $2^x+2^y=2^{x+y}.$
\item $(2^x+1)(2^x+2)+3^y=307.$  
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item Nếu $x=0$, ta có $0=0^y$, đúng với mọi số $y$ nguyên dương (chú ý $0^0$ không có nghĩa). \\
Nếu $x=1$, ta có $1=1^y$, đúng với mọi số tự nhiên $y$. \\
Nếu $x \geq 2$, ta có $y=4$. \\
Đáp số: Nghiệm $(x;y)$ là $(0;t)$; $(1;k)$; $(m;4)$ với $t \in \mathbb{N}^*$, $k \in \mathbb{N}$, $m \in \mathbb{N}$ và $m \geq 2$. 
\item Giả sử $x \geq y$. Chia cả hai vế cho $2^y$ được 
\[2^{x-y} +1 =2^x \tag{1}\]
Nếu $x=y$ thì $2=2^x$ nên $x=1$. \\
Nếu $x>y$ thì vế trái của $(1)$ lẻ, còn vế phải chẵn. Điều này không xảy ra. \\
Đáp số: Nghiệm $(x;y)$ là $(1;1)$.
\item $(2^x+1)(2^x+2) +3^y =307 \quad (1)$\\
Xét tích ba số tự nhiên liên tiếp $2^x, 2^x +1,2^x+2$, có ít nhất một số chia hết cho $3$, mà $2^x$ không chia hết cho $3$ nên $(2^x+1)(2^x+2)~\vdots ~ 3$. \\
Nếu $y \geq 1$ thì $3^y ~\vdots ~ 3$ nên vế  trái của $(1)$ chia hết cho $3$ còn vế phải không chia hết cho $3$, loại. \\
Nếu $y=0$ thì $3^y=1$ nên $(2^x+1)(2^x+2) =307-1=306=17 \cdot 18$. Vậy $2^x+1=17$, suy ra $x=4$. \\
Đáp số: Nghiệm $(x;y)$ là $(4;0)$.
\end{enumerate}
}
\end{bt}

\begin{bt}
Tìm số tự nhiên $n$ để $3^n+1$ là số chính phương.
\loigiai{
Đặt $3^n+1=a^2$ với $a \in \mathbb{N}$ \quad $(1)$ \\
Thử thấy $a=0$, $a=1$ không thỏa mãn $(1)$ nên $a \geq 2$. \\
Viết $(1)$ dưới dạng  \[3^n = (a-1)(a+1).\]
$a-1$ và $a+1$ là các ước tự nhiên của $3^n$. Các ước tự nhiên của $3^n$ là $3^0;3^1;3^2; \dots ; 3^n$. \\
Vì $(a+1)-(a-1)=2$ nên chỉ có thể là  $\heva{&a-1=3^0\\&a+1=3^1=3}$ nên $a=2$. Khi đó $n=1$. \\
Đáp số: $n=1$.
}
\end{bt}

\begin{bt}
Giải mỗi phương trình trình sau với nghiệm tự nhiên
\begin{multicols}{3}
\begin{enumerate}
\item $2^x+33=y^2.$ 
\item $3^x+55=y^2.$
\item $5^x+51=y^2.$  
\item $7^x-1=2^y.$  
\item $3^x+1=2^y.$  
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item $2^x+33=y^2$. \\
Xét hai trường hợp: $x$ lẻ và $x$ chẵn. Giải tương tự như ví dụ $48$. \\
Ta tìm được $x=4$ và $x=8$. Khi đó 
\[2^4+33-7^2 ; 2^8+33=17^2\]
\item $3^x+55=y^2$ \quad $(1)$\\
Xét hai trường hợp sau \\
Nếu $x$ lẻ, thì theo bổ đề $2$, ta có $(3^x+1) ~\vdots ~ (3+1)$ nên $3^x+55 =(3^x+1) +54$ chia cho $4$ dư $2$. \\
Vế trái của $(1)$ chia cho $4$ dư $2$ còn vế phải chia cho $4$ chỉ có thể dư $0$ hoặc $1$, vậy đẳng thức trên không xảy ra. \\
Nếu $x$ chẵn, đặt $x=2n$ $(n \in \mathbb{N})$, ta có \\
$(1) \Leftrightarrow 3^{2n} +55 =y^2 \Leftrightarrow y^2 -(3^n)^2 =55 \Leftrightarrow (y+3^n)(y-3^n)=55$. \\
Ta có bảng giá trị 
\begin{center}
\begin{tabular}{|c|c|c|}
\hline 
$y+3^n$ & $55$ & $11$ \\ 
\hline 
$y-3^n$ & $1$ & $5$ \\ 
\hline 
$3^n$ & $27$ & $3$ \\ 
\hline 
$n$ & $3$ & $1$ \\ 
\hline 
$x$ & $6$ & $2$ \\ 
\hline 
\end{tabular} 
\end{center}
Ta tìm được $x=6$ và $x=2$. Khi đó $3^6+55=28^2$; $3^2+55=8^2$. 
\item $5^x+51=y^2$ \quad $(1)$. \\
Nếu $x$ lẻ, thì $(5^x+1) ~\vdots~(5+1)$ nên $5^x+51=(5^x+1) +50$ chia cho $3$ dư $2$. Vế trái của $(1)$ chia cho $3$ dư $2$ nên không là số chính phương, loại. \\
Nếu $x$ chẵn, đặt $x=2n$ $(n \in \mathbb{N})$, đưa về $(y+5^x)(y-5^x)=51$. \\
Xét các ước của $51$, ta tìm được $5^n =25 \Leftrightarrow n=2$. \\
Khi đó $x=4$ và $5^4+51=26^2$.
\item Sử dụng bổ đề $1$. Phương trình không có nghiệm nguyên vì vế trái chia hết cho $3$, vế phải không chia hết cho $3$. 
\item Dễ thấy $y>0$. Với $y=1$ thì $x=0$; với $y=2$ thì $x=1$. \\
Chú ý rằng với $y \geq 3$ thì vế phải chia hết cho $8$, còn vế trái chia hết cho $8$ dư $2$ nếu $x$ là số chẵn, hoặc dư $4$ nếu $x$ là số lẻ (sử dụng bổ đề $3$). \\
Đáp số: Nghiệm $(x;y)$ là $(0;1)$, $(1;2)$. 
\end{enumerate}
}
\end{bt}

\begin{bt}
Giải mỗi phương trình trình sau với nghiệm tự nhiên
\begin{multicols}{2}
\begin{enumerate}
\item $3^x+7=y^3.$ 
\item $8^x+61=y^3.$
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item Lần lượt xét $x=0$, $x=1$, $x \geq 2$. \\
Chú ý rằng với $x \geq 2$ thì vế trái chia $9$ dư $7$, còn vế phải chia $9$ dư $0,1,8$ (xem giải thích ở lời giải của bài $12$). \\
Đáp số: Nghiệm $(x;y)$ là $(0;2)$. 
\item Giải tương tự ví dụ $49$. Nghiệm $(x;y)$ là $(2;5)$. 
\end{enumerate}
}
\end{bt}

\begin{bt}
Giải mỗi phương trình trình sau với nghiệm tự nhiên
\begin{multicols}{2}
\begin{enumerate}
\item $3^y=5x^3-317.$ 
\item $10^y=81x+1.$
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item $3^y=5x^3-317.$  \quad $(1)$\\
Với $y=0$ thì vế phải không chia hết cho $5$, loại. \\
Với $y=1$ thì $x=4$, thỏa mãn. \\
Với $y \geq 2$. Xét vế phải của $(1)$ có $x^3$ chia $9$ dư $0,1,8$ (xem giải thích ở lời giải của bài $12$) nên $5x^3$ chia cho $9$ dư $0,5,4$ còn vế phải của $(1)$  có $3^y ~\vdots~ 9$ nên $3^y+317$ chia cho $9$ dư $2$. Đẳng thức $(1)$ không thể xảy ra. \\
Đáp số: Nghiệm $(x;y)$ là $(4;1)$. 
\item $10^y=81x+1 \Leftrightarrow 10^y-1=81x$. \\
Nếu $y=0$ thì $x=0$. \\
Nếu $y \geq 1$, ta có $\underbrace{99 \dots 9}_{y \textrm{ chữ số}} =81x$. \\
Đặt $t=\underbrace{99 \dots 9}_{y \textrm{ chữ số}} = 9 \dots \underbrace{11 \dots 1}_{y \textrm{ chữ số}}$
thì $t~\vdots~81 \Leftrightarrow y ~\vdots~9$. \\
Chia số gồm toàn chữ số $9$ cho $81$, lấy đến chín chữ số $9$ đem chia ta được thương là $A=123456789$ và  số dư bằng $0$.\\
Đáp số : Nghiệm $(x;y)$ là $(0;0)$, $(\underbrace{123456789 \dots 123456789}_{k~ \textrm{lần} ~123456789}; 9k)$ với $k \in \mathbb{N}^*$
\end{enumerate}
}
\end{bt}

\begin{bt}
Tìm các số tự nhiên $n$ để mỗi biểu thức sau là lập phương của một số tự nhiên
\begin{multicols}{2}
\begin{enumerate}
\item $3^n+5.$ 
\item $3^n-1.$
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item Đặt $3^n+5=a^3$, $a \in \mathbb{N}$. \\
Lần lượt xét $n=1$, $n=1$, chỉ có $n=1$ thỏa mãn: $3^1+5=2^3$. \\
Nếu $n \geq 2$ thì vế trái chia $9$ dư $5$, còn vế  phải $a^3$ chia $9$ dư $0,1,8$ (xem giải thích ở lời giải của bài $12$) nên đẳng thức trên không xảy ra.\\
Đáp số: $n=1$. 
\item Đặt $3^n-1=a^3$, $a \in \mathbb{N}$. \\
Ta có $3^n=a^3+1 =(a+1)(a^2-a+1)$. \\
Các số nguyên dương $a+1$ và $a^3-a+1$ là ước của $3^n$ nên 
\[\heva{&a+1=3^x & (2) \\ &a^2-a+1=3^y &(3) \\
&x+y=a} \quad  \textrm{với } x,y \in \mathbb{N} 
\]
Rút $a$ từ $(2)$ thay vào $(3)$ được
\begin{align*}
(3^x-1)^2-(3^x-1)+1=3^y &\Leftrightarrow 3^{2x}-2 \cdot 3^x+1 -3^x+1+1=3^y \\
&\Leftrightarrow  3^{2x} -3^{x+1}+3=3^y  \tag{4}
\end{align*}
Nếu $x \geq 2$ thì $(3^{2x} - 3^{x+1})~ \vdots~ 9$ nên vế trái của $(4)$ chia cho $9$ dư $3$. Do $x \geq 2$ nên $3^{2x} -3^{x+1} +3 = 3^x (3^x-3) +3 \geq 9\cdot 6 +3=57$ nên $y \geq 4$, do đó vế phải của $(4)$ chia hết cho $9$. Đẳng thức $(4)$ không xảy ra. \\
Nếu $x=1$ thì từ $(2)$ suy ra $a=2$. Khi đó $3^n=a^3+1=9$ nên $n=2$. \\
Nếu $x=0$ thì từ $(2)$ suy ra $a=0$. Khi đó $3^n=a^3+1=1$ nên $n=0$. \\
Đáp số $n=2$ và $n=0$.  
\end{enumerate}
}
\end{bt}

\begin{bt}
Giải mỗi phương trình trình sau với nghiệm tự nhiên
\begin{multicols}{2}
\begin{enumerate}
\item $3^y=x^2-5x+7.$ 
\item $3^y=x^3+x^2+x+1.$ 
\item $2^y=x^3+1.$ 
\item $2^y=x^4+x^3+x+1.$ 
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item $3^y=x^2-5x+7.$  \\
Nếu $x \geq 2$ thì vế trái chia hết cho $9$, ta sẽ chứng minh vế phải không chia hết cho $9$.Thật vậy, giả sử $(x^2-5x+7)~\vdots ~ 9 $ $ (1)$ thì $(x^2-5x+7) ~\vdots ~ 3 \Rightarrow (x^2-2x+1) ~\vdots ~ 3 \Rightarrow (x-1)^2~\vdots ~ 3 \Rightarrow (x-1)~\vdots ~ 3$ (vì $3$ là số nguyên tố). \\
Đặt $x=3k+1$, $k \in \mathbb{N}$. Khi đó $x^2-5x+7=(3k+1)^2=5(3k+1)+7=9k^2-9k+3$, không chia hết cho $9$, mâu thuẫn với $(1)$. \\
Xét $y=0$, ta được $x=2$ và $x=3$. \\
Xét $y=1$, ta được $x=1$ và $x=4$. \\
Đáp số: Nghiệm $(x;y)$ là $(2;0)$, $(3;0)$, $(1;1)$, $(4;1)$. 
\item $3^x=x^3+x^2+x+1 \Leftrightarrow 3^y=(x+1)(x^2+1).$ \\
Giải tương tự cách $2$ của ví dụ $50$. Suy ra $x^2+1 =3^n$ $(1)$. \\
Xét $n=0$ thì $x=0$, $y=0$. \\
Xét $ n \geq 1$ thì vế phải của $(1)$ chia hết cho $3$, còn vế trái chia hết cho $3$ dư $1$ hoặc dư $2$, loại. \\
Đáp số: Nghiệm $(x;y)$ là $(0;0)$.
\item Giải tương tự cách $1$ của ví dụ $50$ 
\[2^y=(x+1)(x^2-x+1)\]
$x+1$ và $x^2-x+1$ là các ước tự nhiên của $2^y$ nên 
\[ \heva{&x+1 & (1)\\ &x^2-x+1=2^y &(2) \\& m+n=y} \quad  \textrm{với } m,n \in \mathbb{N}\]
Rút $x$ từ $(1)$ rồi thay vào $(2)$ được \\
\begin{align*}
(2^m-1)^2-(2^m-1)+1=2^n & \Leftrightarrow 2^{2m} -2 \cdot 2^m +1-2^m+1+1=2^n \\
&\Leftrightarrow 2^m(2^m-3) +3 =2^n \quad (3)
\end{align*}
Nếu $ m \geq 2$ thì vế trái của $(3)$ chia $4$ dư $3$. Ta có  
\[ 2^m(2^m-3)+3 \geq 4 \cdot 1 +3=7 \textrm{ nên } 2^n \geq 7\] 
do đó vế phải của $(3)$ chia hết cho $4$ loại. \\
Xét $m=0$, ta được $x=0$, $y=0$. \\
Xét $m=1$, ta được $x=1$, $y=1$. \\
Đáp số: Nghiệm $(x;y)$  là $(0;0)$, $(1;1)$.
\item $2^y = x^4+x^3+x+1 \Leftrightarrow  2^y=(x+1)(x^3+1)$ $(1)$ \\
$x+1$ và $x^3+1$ là các ước tự nhiên của $2^y$ nên 
\[ \heva{&x+1 & (2)\\ &x^3+1=2^y &(3) \\& m+n=y} \quad  \textrm{với } m,n \in \mathbb{N}\]
Rút $x$ từ $(1)$ rồi thay vào $(2)$ được \\
\begin{align*}
(2^m-1)^3+1=2^n & \Leftrightarrow 2^{3m} -3 \cdot 2^{2m} +3 \cdot 2^m=2^n \\
&\Leftrightarrow 2^m(2^{2m} -3 \cdot 2^{m} +3)=2^n
\end{align*}
Nếu $ m \geq 2$ thì
$2^{2m} -3 \cdot 2^{m} +3=2^m(2^m-3) +3  \geq 4 \cdot 1 +3=7$. \\
Khi đó  $2^m(2^m-3) +3$ là số lẻ lớn hơn $1$ nên không thể là ước của $2^n$, loại. \\
Xét $m=0$, ta được $x=0$, $y=0$. \\
Xét $m=1$, ta được $x=1$, $y=2$. \\
Đáp số: Nghiệm $(x;y)$  là $(0;0)$, $(1;2)$.
\end{enumerate}
}
\end{bt}


\begin{bt}
Giải mỗi phương trình trình sau với nghiệm tự nhiên
\begin{multicols}{2}
\begin{enumerate}
\item $7^x+13^y=19^z.$ 
\item $2^x+2^y=2^z.$ 
\item $2^x+2^y+2^z=552 ~ (x<y<z)$ 
\item $x^y+1=z^2 ~ (x \textrm{ là số nguyên tố})$ 
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item Phương trình không có nghiệm tự nhiên. Hãy xét số dư khi chia cho $3$ của mỗi số $7^x,13^y,19^z$.
\item Giả sử $x \geq y$. Chia hai vế cho $2^y \neq 0$ được. 
\[2^{x-y} +1 =2^{z-y} \tag{1}\]
Từ đề bài suy ra $x=y=k$ ($k$ là số tự nhiên). Thay vào $(1)$ được 
\[2=2^{z-k} \Leftrightarrow 1=z-k \Leftrightarrow z=k+1\]
Đáp số: Nghiệm $(x;y;z)$ là $(k;k;k+1)$ với $k$ là số tự nhiên tùy ý. 
\item $2^x+2^y+2^z=552 \Leftrightarrow 2^x(1+2^{y-x} + 2^{z-x}) = 2^3 \cdot 3 \cdot 23$. \\
Do $y>x$, $z>x$ nên $1+2^{y-x} +2^{z-x}$ là số lẻ, suy ra $2^x=2^3 \Leftrightarrow x=3$. \\
Tự giải tiếp. \\
Đáp số: Nghiệm $(x;y;z)$ là $(3;5;9)$. 
\item $x^y = (z-1)(z+1)$ \\
Do đó $x$ là số nguyên tố nên $z+1$, $z-1$ là các lũy thừa của $x$. Đặt $z-1=x^m$, $z+1=x^{m+n}$ với $m,n \in \mathbb{N}$. Ta có 
\begin{align*}
(z+1)-(z-1) = x^{m+n} -x^m &\Leftrightarrow 2= x^m(x^n-1) \\
& \Leftrightarrow \heva{&x^m=2\\& x^n-1=1}  \textrm{hoặc} \heva{&x^m=1\\& x^n-1=2} 
\end{align*}
Từ đó $x=2$, $m=1$, $n=1$ hoặc $x=3$, $m=2$, $n=1$. \\
Đáp số: Nghiệm $(x;y;z)$ là $(2;3;3)$, $(3;1;2)$. 
\end{enumerate}
}
\end{bt}

\begin{bt}
Tìm các số nguyên dương $x,y,z,t$ thỏa mãn mỗi điều kiện sau
\begin{multicols}{2}
\begin{enumerate}
\item $x^y+x^z=x^t.$ 
\item $x^x+y^y+z^z=t^t.$ 
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item $x^y+x^z=x^t$ \quad $(1)$. \\
Do vai trò của $y,z$ như nay, ta giả sử $ y \leq z <t$. Chia cả hai vế cho $x^y$ được 
 \[ 1+x^{z-y} =x^{t-y} \tag{2}\]
Nếu $x>y$ thì từ $(2)$ suy ra $ 1 ~\vdots ~x$, do đó $x=1$, không thỏa mãn $(1)$. Vậy $z=y$. Đặt $z=y=k$ ($k$ là số nguyên dương). \\
Thay vào $(2)$ được $2=x^{t-k}$ nên $x=2$, $t-k=1$. \\
Đáp số: Nghiệm $(x;y;z;t)$ là $(2;k;k;k+1)$ với số nguyên dương $k$ tùy ý. 
\item $x^x+y^y+z^z=t^t.$ \quad $(1)$ \\
Giả sử $1 \leq x \leq y \leq z <t$. \\
Nếu $z=1$ thì $x=y=1$. Ta có $3=t^3$, loại. \\
Nếu $z=2$ thì $t \geq 3$. Khi đó $t^t \geq 3^3 > 3 \cdot 2^3 \geq x^x +y^y+z^z$, trái với $(1)$. \\
Nếu  $z \geq 3$ thì $t^t \geq (z+1)^{z+1} < z^{z+1} =z \cdot z^z \geq 3 \cdot z^z \geq x^x+y^x+z^z$, trái với $(1)$. \\
Vậy phương trình không có nghiệm nguyên dương. 
\end{enumerate}
}
\end{bt}

\begin{bt}
Tìm các số nguyên dương $x$ và $y$ khác nhau sao cho
\[x^y=y^x.\]
\loigiai{
Giả sử $1 \leq x <y$. Chia cả hai vế của phương trình cho $x^x$ được 
\[x^{y-x} = \dfrac{y^x}{x^x}\]
Ta có $y^x ~\vdots ~ x^x$ mà $x$ là số nguyên dương nên $y ~\vdots ~ x$. Đặt $y=kx$ với $ k \in \mathbb{N}$, $k \geq 2$. Theo đề bài thì $x^{kx} = (kx)^x \Leftrightarrow (x^k)^x = (kx)^x \Leftrightarrow x^k = kx \Leftrightarrow x^{k-1} =k$ $(1)$ \\
Ta thấy $x \geq 2$ vì nếu $x=1$ thì $k=1$, loại. Do đó $x^{k-1} \geq 2^{k-1}$. \\
Từ $(1)$ và $(2)$ suy ra $k \geq 2^{k-1}$, do đó $2k \geq 2^k$ $(3)$. \\
Ta thấy với $k \geq 3$ thì bất đẳng thức $(3)$ không thể xảy ra (có thể chứng minh điều này bằng quy nạp toán học) \\
Do đó $k=2$. Thay $k=2$ vào $(1)$ được $x=2$. \\
Suy ra $y=kx=2 \cdot 2 =4$. \\
Thử lại có $2^4=4^2$. \\
Đáp số: Nghiệm $(x;y)$ là $(2;4)$ hoặc $(4;2)$. 
}
\end{bt}
