\begin{ex}%[9D4Y2]%[9D4Y4]%[9D3Y4]%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018 Q.Phú Nhuận, Tp. Hồ Chí Minh] %[Đào Anh Dương]
	Giải các phương trình và hệ phương trình sau.
	\begin{enumerate}
		\item $\sqrt{2}x^2-\sqrt{8}=0$.
		\item $2x^2+3x-2=0$.
		\item $\begin{cases} -x+y=-5\\ 3x+5y=-1 \end{cases}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $\sqrt{2}x^2-\sqrt{8}=0 \Leftrightarrow x^2=4 \Leftrightarrow x=2 \quad \text{hoặc} \quad x=-2$.
			\item Ta có $\Delta = 9+16 = 25 \Rightarrow \sqrt{\Delta}=5$. Do đó\\
			$$x_1=\dfrac{-b+\sqrt{\Delta}}{2a} = \dfrac{-3+5}{4} = \dfrac{1}{2} \quad \text{và} \quad x_2=\dfrac{-b-\sqrt{\Delta}}{2a} = \dfrac{-3-5}{4} = -2.$$
			\item $\begin{cases} -x+y=-5\\ 3x+5y=-1 \end{cases} \Leftrightarrow \begin{cases} -3x+3y=-15\\ 3x+5y=-1 \end{cases} \Leftrightarrow \begin{cases} 8y=-16\\ 3x+5y=-1 \end{cases} \Leftrightarrow \begin{cases} y=-2\\ x=3 \end{cases}$.
		\end{enumerate}
		}
\end{ex}

	Cho hàm số $y=-\dfrac{x^2}{2}$ có đồ thị là $(P)$ và hàm số $y=\dfrac{3}{2}x-2$ có đồ thị là $(d)$.
	\begin{enumerate}
		\item Vẽ đồ thị $(P)$ và $(d)$ trên cùng hệ trục tọa độ $Oxy$.
		\item Tìm tọa độ giao điểm của $(P)$ và $(d)$ bằng phép toán.
	\end{enumerate}
	\loigiai{

				\hline

				\hline 
				\end{tabular}

		Suy ra $x^2+3x-4=0$, giải ra ta được $x=1$ hoặc $x=-4$.\\
		Ta có $x=1 \Rightarrow y=-\dfrac{1}{2}$ và $x=-4 \Rightarrow y=-8$.\\
		Vậy, toạ độ giao điểm của $(P)$ và $(d)$ là $\left (1,-\dfrac{1}{2} \right )$ và $(-4,-8)$.

\begin{ex}%[9D4K6]%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018 Q.Phú Nhuận, Tp. Hồ Chí Minh]%[Đào Anh Dương]
	Cho phương trình bậc hai $x^2+(2m-3)x-m+1=0. \hfill (1)$
	\begin{enumerate}
		\item Chứng minh rằng phương trình luôn có hai nghiệm phân biệt $x_1$, $x_2$ với mọi giá trị của $m$.
		\item Tìm giá trị của $m$ thoả mãn hệ thức $(x_1-3)(x_2-3)=5$.
	\end{enumerate}
	\loigiai
	{
	\begin{enumerate}
		\item Ta có $\Delta = (2m-3)^2 -4(-m+1) = 4(m-1)^2 + 1 > 0, \forall m.$\\
		Do đó, phương trình luôn có hai nghiệm phân biệt với mọi giá trị của $m$.
		\item Theo hệ thức Vi-et ta có
		$\begin{cases} x_1+x_2=-2m+3\\ x_1x_2=-m+1 \end{cases}$.\\
		Từ yêu cầu bài toán ta có
		\begin{eqnarray*}
			& &(x_1-3)(x_2-3)= 5\\
			&\Leftrightarrow &x_1x_2 - 3(x_1+x_2)+4=0\\
			&\Leftrightarrow &(-m+1) - 3(-2m+3)+4=0\\
			&\Leftrightarrow &5m=4 \Leftrightarrow m=\dfrac{4}{5}.
		\end{eqnarray*}	
		Vậy, $m=\dfrac{4}{5}$ là giá trị cần tìm.
	\end{enumerate}
}
\end{ex}

\begin{ex}%[9D4K8]%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018 Q.Phú Nhuận, Tp. Hồ Chí Minh]%[Đào Anh Dương]
	Cần phải pha bao nhiêu lít nước ở nhiệt độ $40^{\circ
	}$C vào $8$ lít nước ở nhiệt độ $70^{\circ}$C để được lượng nước ở nhiệt độ $60^{\circ}$C.
	\loigiai{
		Gọi số lít nước ở nhiệt độ $40^{\circ
		}$C là $x$ (lít); $x>0$.\\
		Giả sử $c$ là nhiệt dung riêng của nước.\\
		Ta có nhiệt lượng do $8$ lít nước ở nhiệt độ $70^{\circ}$C tỏa ra là: $Q_1= 8c(70-60)= 80c$.\\
		Nhiệt lượng do $x$ lít nước ở nhiệt độ $40^{\circ}$C thu vào là: $Q_2= xc(60-40)=20xc$.\\
		Do nhiệt lượng tỏa ra bằng nhiệt lượng thu vào nên ta có phương trình:\\
		$80c= 20xc \Leftrightarrow x= 4\ \text{(thỏa mãn)}$.\\
		Vậy cần pha thêm $4$ lít nước ở nhiệt độ $40^{\circ}$C.
		
		
	}
\end{ex}
\begin{ex}%[9H3K9]%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018 Q.Phú Nhuận, Tp. Hồ Chí Minh]%[Đào Anh Dương]
	Bạn Nam đi học từ nhà tới trường bằng xe đạp có đường kính bánh xe là $700$ mm. Tính quãng đường từ nhà tới trường, biết rằng bánh xe đạp quay tất cả $875$ vòng (giả sử bạn Nam đạp xe chạy thẳng từ nhà tới trường trên một đường thẳng và kết quả được làm tròn đến chữ số thập phân thứ nhất).
	\loigiai{
		Đổi $700$ mm = $0{,}7$ m.\\
		Chu vi của bánh xe là: $0{,}7 \cdot 3{,}14= 2{,}198$ (m).\\
		Quãng đường từ nhà tới trường là: $2,198 \cdot 875= 1923{,}25$ (m) $\approx 1{,}9$ (km).\\
		Vậy quãng đường từ nhà Nam tới trường dài khoảng $1{,}9$ km.
	}
\end{ex}

	\begin{ex}%[9H3K7]%[9H3G7]%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018 Q.Phú Nhuận, Tp. Hồ Chí Minh] %[Đào Anh Dương]
	Từ điểm $S$ nằm ngoài đường tròn $(O;R)$ vẽ hai tiếp tuyến $SA,SB$ đến $(O)$ ($A,B$ là hai tiếp điểm) và cát tuyến $SCD$ ($C$ nằm giữa $S$ và $D$; tia $SD$ nằm trong góc $ASO$).
	\begin{enumerate}
		\item Chứng minh tứ giác $SAOB$ nội tiếp và $SA^2 = SD \cdot SC$.
		\item Gọi $H$ là giao điểm của $AB$ với $OS$. Chứng minh $\widehat{DCO} = \widehat{SHC}$.
		\item Gọi $I$ là trung điểm của $CD$. Chứng minh $\triangle IAC \backsim \triangle ICB$.
	\end{enumerate}
	\loigiai 
	{
		\begin{center}
			\begin{tikzpicture}[scale=0.8]
			\tkzInit[xmin=-8.7,xmax=3.2,ymin=-4.1,ymax=4.1]
			%\tkzAxeXY
			\tkzClip
			\tkzDefPoints{0/0/O,3/0/a,-8/0/S}
			\tkzDrawCircle[radius](O,a)
			\tkzTangent[from=S](O,a)
			\tkzGetPoints{B}{A}
			\tkzDefPointBy[rotation = center O angle 50](a)
			\tkzGetPoint{D}
			\tkzInterLC(S,D)(O,a)
			\tkzGetPoints{C}{D}
			\tkzInterLL(A,B)(O,S)
			\tkzGetPoint{H}
			\coordinate (I) at ($(C)!0.5!(D)$);
			\coordinate (K) at ($(S)!0.5!(O)$);
			\tkzDrawCircle[radius](K,O)
			\tkzMarkRightAngles[size=0.3,fill=gray!50](S,A,O S,B,O A,H,O)
			\tkzMarkAngles[size=0.8cm,arc=ll](O,C,D C,H,S)
			\tkzMarkSegments[mark=x](I,C I,D)
			\tkzDrawSegments(S,A S,B S,D O,A O,B S,O A,B C,A C,B C,O C,H I,A I,B O,I)
			\tkzDrawPoints(O,S,A,B,C,D,H,I)
			\tkzLabelPoints[above](A,D,I)
			\tkzLabelPoints[below](B)
			\tkzLabelPoints[left](S)
			\tkzLabelPoints[right](O)
			\tkzLabelPoints[above left](C)
			\tkzLabelPoints[below left](H)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Tứ giác $SAOB$ có $\widehat{SAO} + \widehat{SBO} = 90^\circ + 90^\circ = 180^\circ$ nên $SAOB$ là tứ giác nội tiếp.\\
			Xét hai tam giác $SAC$ và $SDA$ có
			$\begin{cases}
			\widehat{ASC}=\widehat{DSA} \\ \widehat{SAC}=\widehat{SDA} \,(\text{cùng chắn cung} \, \wideparen{AC}).
			\end{cases}$\\
			Do đó $\triangle SAC \backsim \triangle SDA$, suy ra 
			$$\dfrac{SA}{SD} = \dfrac{SC}{SA} \Rightarrow SA^2 = SD \cdot SC.$$
			\item Trong tam giác vuông $ASO$ có $AH$ là đường cao nên $SA^2 = SH \cdot SO$.\\
			 Mặt khác ta lại có $SA^2 = SC \cdot SD$.\\
			  Suy ra $SH \cdot SO = SC \cdot SD \Rightarrow \dfrac{SC}{SO} = \dfrac{SH}{SD}$.\\
			  Do đó $\triangle SCH \backsim \triangle SOD \Rightarrow \widehat{SCH} = \widehat{SOD}$.\\
			  Mà  $\widehat{SCH} + \widehat{HCD} = 180^\circ$.\\
			  Suy ra  $\widehat{SOD} + \widehat{HCD} = 180^\circ$ nên tứ giác $OHCD$ nội tiếp.\\
			 Khi đó ta có $\widehat{SHC}=\widehat{CDO} \,(\text{cùng bù với} \, \widehat{CHO}).$\\
			Mà $\widehat{CDO} = \widehat{DOC} \,(\triangle OCD \, \text{cân}).$\\
			Vậy $\widehat{DCO} = \widehat{SHC}$.
			\item Ta có $A,I,B$ cùng nhìn $SO$ dưới một góc vuông nên ngũ giác $SAIOB$ nội tiếp.\\
			Suy ra $\begin{cases}
			\widehat{AIS}=\widehat{AOS} \,(\text{cùng chắn cung} \, \wideparen{SA}) \\ \widehat{BIS}=\widehat{BOS} \,(\text{cùng chắn cung} \, \wideparen{SB})
			\end{cases}.$\\
			Mà $\widehat{AOS} = \widehat{BOS}$ nên $\widehat{AIS} = \widehat{BIS}  \hfill(1)$.\\
			Ta lại có $\begin{cases}
			\widehat{ASC}=\widehat{ABI} \,(\text{cùng chắn cung} \, \wideparen{AI}) \\ \widehat{CAS}=\widehat{CBA} \,(\text{cùng chắn cung} \, \wideparen{AC})
			\end{cases}.$\\
			Suy ra $\widehat{ASC} + \widehat{CAS} = \widehat{ABI} + \widehat{CBA}$\\
			Tương đương với $\widehat{ACI} = \widehat{CBI} \hfill(2).$\\
			Từ $(1)$ và $(2)$ suy ra $\triangle IAC \backsim \triangle ICB$.
		\end{enumerate}
	}
\end{ex}
