\chapter{Đa giác. Diện tích đa giác}
\section{Đa giác}
\subsection{Tóm tắt lí thuyết}
Đa giác $A_1A_2A_3\ldots A_n$ là hình gồm $n$ đoạn thẳng $(n\ge 3)$, trong đó bất kì hai đoạn thẳng nào có điểm chung cũng không cùng nằm trên một đường thẳng.\\
Đa giác lồi là đa giác nằm trong một nửa mặt phẳng mà bờ là đường thẳng chứa bất kì cạnh nào của đa giác đó.\\
Khi nói đến đa giác mà không chú thích gì thêm, ta hiểu đó là đa giác lồi.\\
Tổng số đo các góc của đa giác $n$ cạnh là $(n-2)\cdot 180^\circ$.\\
Đa giác đều là đa giác có tất cả các cạnh bằng nhau và tất cả các góc bằng nhau.
\begin{vd}
Tổng các góc của một đa giác $n$ cạnh trừ đi góc $A$ của nó bằng $570^\circ$. Tính $n$ và~$\widehat{A}$.
\loigiai{
Ta có $(n-2)\cdot 180^\circ -\widehat{A}=570^\circ$ nên $\widehat{A}=(n-2)\cdot 180^\circ -570^\circ$.\\
Do $0^\circ <\widehat{A}<180^\circ$ nên $0<(n-2)\cdot 180^\circ -570^\circ <180^\circ\Leftrightarrow 0<n-2-\dfrac{570}{180}<1\Leftrightarrow 5\dfrac{1}{6}<n<6\dfrac{1}{6}$.\\
Do $n$ là số tự nhiên nên $n=6$ và $\widehat{A}=(6-2)\cdot 180^\circ -570^\circ =150^\circ$.
}
\end{vd}
\begin{vd}
Ngũ giác đều $ABCDE$ có các đường chéo $AC$ và $BE$ cắt nhau ở $K$. Chứng minh rằng $CKED$ là hình thoi.
\loigiai{
\immini{
Góc của ngũ giác đều bằng $\dfrac{(5-2)\cdot 180^\circ}{5}=108^\circ$.\\

Do đó $\widehat{A_2}=\widehat{C_2}=108^\circ -36^\circ =72^\circ$.\\
Ta có $\widehat{C_2}+\widehat{D}=72^\circ +108^\circ =180^\circ$ nên $AC\parallel DE$.\\
Chứng minh tương tự, $BE\parallel CD$.\\
Do đó $CKED$ là hình bình hành.\\
Ta lại có $CD=DE$ nên $CKED$ là hình thoi.

\begin{bt}%[Lê Nguyễn Viết Tường, EX-C2-L8-2018]%[8H2K1]
Tính số cạnh của một đa giác, biết rằng đa giác đó có:
\begin{enumerate}
\item Tổng các góc trong bằng tổng các góc ngoài (tại mỗi đỉnh của đa giác chỉ kể một góc ngoài);
\item Số đường chéo gấp đôi số cạnh;
\item Tổng các góc trong trừ đi một góc của đa giác bằng $2570^\circ$.
\end{enumerate}
\loigiai{
Gọi số cạnh của đa giác là $n$.
\begin{enumerate}
\item Ta có $(n-2)\cdot 180^\circ =2\cdot 180^\circ$. Ta tìm được $n=4$.
\item Ta có $\dfrac{n(n-3)}{2}=2n$. Ta tìm được $n=7$.
\item Ta có $(n-2)\cdot 180^\circ - \widehat{A}=2570^\circ$ nên $\widehat{A}=(n-2)\cdot 180^\circ - 2570^\circ$.\\
Do $0^\circ <\widehat{A}<180^\circ$ nên $0<(n-2)\cdot 180 -2570<180\Leftrightarrow 16\dfrac{5}{18}<n<17\dfrac{5}{18}$.\\
Do $n$ là số tự nhiên nên $n=17$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Lê Nguyễn Viết Tường, EX-C2-L8-2018]%[8H2G1]
\begin{enumerate}
\item Tìm số $n$ sao cho mặt phẳng có thể được phủ kín bởi các đa giác đều bằng nhau có $n$ cạnh.
\item Có tồn tại các ngũ giác bằng nhau (không yêu cầu đều) để phủ kín mặt phẳng không?
\item Số đo các góc của đa giác đều $n$ cạnh là số tự nhiên. Có bao nhiêu giá trị của $n$ thỏa mãn bài toán?
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Theo đề bài ta có $360\,\vdots\,\dfrac{(n-2)\cdot 180^\circ}{n}\Leftrightarrow n\in \{3;4;6\}$.
\item Có. Chẳng hạn ngũ giác là nửa của lục giác đều.
\item Ta có $\dfrac{(n-2)\cdot 180^\circ}{n}\in\mathbb{N}\Rightarrow 180-\dfrac{360}{n}\in \mathbb{N}\Rightarrow 360\,\vdots\,n$.\\
Phân tích $360$ ra thừa số nguyên tố ta được $360=2^3\cdot 3^2\cdot 5$.\\
Số $360$ có $24$ ước tự nhiên. Do $n\ge 3$ nên ta loại các số $1$ và $2$.\\
Vậy có $22$ giá trị của $n$ thỏa mãn bài toán.
\end{enumerate}
}
\end{bt}

\loigiai{
\immini{Gọi $F$ là trung điểm của $GC$.
\\ Ta có $$KG\parallel DF, BG=\dfrac{4}{5}BF$$ nên chứng minh được $$KG=\dfrac{4}{5}DF=\dfrac{2}{5}AG, BK=\dfrac{4}{5}BD.$$
}{
\begin{tikzpicture}[line join=round, line cap=round, font=\footnotesize, scale=0.7]
\tikzset{label style/.style={font=\footnotesize}} 
\tkzDefPoints{0/0/B,6/0/C,2/4/A,2/0/E,4/0/G,5/0/F,1/2/H,4/2/D} 
\tkzDrawSegments[dashed](D,F)
\tkzInterLL(A,G)(B,D) 
\tkzGetPoint{K}
\tkzInterLL(H,G)(B,D) 
\tkzGetPoint{I}
\tkzDrawSegments(A,B B,C C,A A,G B,D H,G I,E H,D)
\tkzLabelPoints[below](C,B,E,G,F,C)
\tkzLabelPoints[left](H)
\tkzLabelPoints[right](D)
\tkzLabelPoints[below right](K)
\tkzLabelPoints[above](A,I)
\tkzDrawPoints(I,K,A,B,C,D,H,G,E,F) 
\end{tikzpicture}
}
\noindent Lại có $HD\parallel BC$ nên $$\dfrac{ID}{IB}=\dfrac{HD}{BC}=\dfrac{\dfrac{1}{2}BC}{\dfrac{2}{3}BG}=\dfrac{3}{4}\Rightarrow ID=\dfrac{3}{4}IB.$$
Suy ra $$ID=\dfrac{3}{7}BD,IB=\dfrac{4}{7}BD, IK=\dfrac{8}{35}BD.$$
Đặt 
\begin{equation}
S_{IKG}=a.
\end{equation} 
thì \begin{equation}
S_{AIG}=\dfrac{5}{2}a, S_{BIG}=\dfrac{5}{2}a, S_{IEG}=\dfrac{5}{4}a.
\end{equation}
Từ $(1)$, $(2)$ suy ra 
\begin{equation}
S_{EIKG}=\dfrac{5}{4}a+a=\dfrac{9}{4}a.
\end{equation} 
Lại có $S_{BKG}=\dfrac{5}{2}a+a=\dfrac{7}{4}a$ nên
\begin{equation}
S_{ABG}=\dfrac{7}{2}a\cdot \dfrac{5}{2}=\dfrac{35}{4}a, S_{ABC}=\dfrac{35}{4}a\cdot\dfrac{3}{2}=\dfrac{105}{8}a.
\end{equation}
Từ $(3)$, $(4)$ suy ra $S_{EIKG}:S_{ABC}=\dfrac{9}{4}:\dfrac{105}{8}=\dfrac{6}{35}$ hay $S_{EIKG}=\dfrac{6}{35}S_{ABC}$.
}
\end{bt}

\begin{bt}%[Lê Thanh Hải, EX-C2-L8-2018]%[8H2K3]
Chứng minh rằng tam giác có một đỉnh là giao điểm hai cạnh đối của một tứ giác, hai đỉnh kia là trung điểm hai đường chéo của tứ giác đó có diện tích bằng $\dfrac{1}{4}$ diện tích tứ giác. 	
\loigiai{
\immini{
Gọi $M,N$ là trung điểm các đường chéo $BD, AC$ của tứ giác $ABCD$, $E$ là giao điểm của $AD, BC$. Ta có
\begin{eqnarray*}
S_{EMN}& =&S_{EDC}-S_{EMD}-S_{ENC}-S_{DMC}-S_{MNC}
\\ &  =& S_{EDC}-\dfrac{1}{2}S_{EBD}-\dfrac{1}{2}S_{EAC}-\dfrac{1}{2}S_{DBC}-\dfrac{1}{2}S_{AMC}
\\& = &\dfrac{1}{2}\left(S_{EDC}-S_{EBD}-S_{DBC} \right)+\dfrac{1}{2}\left(S_{EDC}-S_{EAC}-S_{AMC}\right)
\\ & =& 0+\dfrac{1}{2}\left(S_{ADM}+S_{CDM} \right)=\dfrac{1}{4}S_{ABCD}.
\end{eqnarray*}

\begin{note}
Nếu tứ giác $ABCD$ có $F$ là giao điểm của $AB$ và $CD$ thì từ bài toán trên ta cũng có $S_{FMN}=\dfrac{1}{4}S_{ABCD}$. Suy ra $E$ và $F$ cách đều đường thẳng $MN$. Do đó đường thẳng $MN$ đi qua trung điểm của $EF$. Đường thẳng đi qua $M,N$ và trung điểm của $EF$ gọi là đường thẳng Gau-xơ.
\end{note}	
}

\subsubsection{Dựng hình}

\begin{bt}%[Lê Thanh Hải, EX-C2-L8-2018]%[8H2K3]
Cho tam giác $ABC$. Dựng các điểm $D$ và $F$ nằm trên cạnh $AB$, $E$ nằm trên cạnh $AC$ sao cho đường gấp khúc $CDEF$ chia tam giác $ABC$ ra bốn phần có diện tích bằng nhau.	
\loigiai{
\immini{Lấy điểm $D$ trên $AB$ sao cho $BD=\dfrac{1}{4}BA$. Suy ra $$S_{BCD}=\dfrac{1}{4}S_{ABC}.$$
Lấy điểm $E$ trên $AC$ sao cho $CE=\dfrac{1}{3}AC$. Suy ra $$S_{CDE}=\dfrac{1}{3}S_{CDA}=\dfrac{1}{4}S_{ABC}.$$
}{
\begin{tikzpicture}[line join=round, line cap=round, font=\footnotesize, scale=0.6]
\tikzset{label style/.style={font=\footnotesize}}  
\tkzDefPoints{0/0/A,6/0/B,4/5/C} 
\tkzLabelPoints[below](A,B,D,F)
\tkzLabelPoints[above](C)
\tkzLabelPoints[left](E)
\coordinate (D) at ($(B)!0.25!(A)$);
\coordinate (E) at ($(C)!0.33!(A)$);
\coordinate (F) at ($(A)!0.5!(D)$);
\tkzDrawSegments(A,B A,C B,C C,D D,E E,F)
\tkzDrawPoints[fill=black](A,B,C,E,F,D)
\end{tikzpicture}
}	
\noindent Lấy điểm $F$ trên $AD$ sao cho $AF=\dfrac{1}{2}AD$. Suy ra $S_{DEF}=S_{EFA}=\dfrac{1}{2}S_{DEA}=\dfrac{1}{3}S_{CDA}=\dfrac{1}{4}S_{ABC}.$\\
Vậy với cách lấy các điểm $D,E,F$ như trên thỏa được yêu cầu bài toán.	
}
\end{bt}

\begin{bt}%[Lê Thanh Hải, EX-C2-L8-2018]%[8H2K3]

\loigiai{
