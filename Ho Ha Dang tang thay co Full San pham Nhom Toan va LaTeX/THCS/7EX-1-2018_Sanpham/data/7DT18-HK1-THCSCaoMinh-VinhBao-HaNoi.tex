\begin{ex}%[Đề kiểm tra học kỳ I, THCS Cao Minh - Vinh Bảo - Hà Nội, 2017 - 2018]%[Trần Thịnh, dự án 7EX-1-2018]%[7D1B3]%[7D1B6]%[7D1B4]%[7D1B9]
	Thực hiện phép tính:
	\begin{enumEX}{2}
		\item $\dfrac{3}{5}-\dfrac{8}{15}\cdot \dfrac{21}{4}$;
		\item $\dfrac{5}{2}+\left(\dfrac{-1}{2}\right)^3$;
		\item $\dfrac{2}{5}-\left(0,{,}5+1\dfrac{3}{4}\right)$;
		\item $\dfrac{2}{3} : \left(\dfrac{1}{3}-\sqrt{4}\right)$.
	\end{enumEX}	
\loigiai{
	\begin{enumEX}{2}
		\item $\dfrac{3}{5}-\dfrac{8}{15}\cdot \dfrac{21}{4}
		=\dfrac{3}{5}-\dfrac{14}{5}=\dfrac{-11}{5}$.
		\item $\dfrac{5}{2}+\left(\dfrac{-1}{2}\right)^3=\dfrac{5}{2}+\dfrac{-1}{8}=\dfrac{19}{8}$.
		\item $\dfrac{2}{5}-\left(0,{,}5+1\dfrac{3}{4}\right)\\
		=0{,}4-(0{,}5+1{,}75)\\
		=0{,}4-2{,}25=-1{,}85$.
		\item $\dfrac{2}{3} : \left(\dfrac{1}{3}-\sqrt{4}\right) =\dfrac{2}{3}:\left(\dfrac{1}{3}-2\right)\\
		=\dfrac{2}{3}:\left(\dfrac{-5}{3}\right)=\dfrac{-2}{5}$.
	\end{enumEX}	
}
\end{ex}

\begin{ex}%[Đề kiểm tra học kỳ I, THCS Cao Minh - Vinh Bảo - Hà Nội, 2017 - 2018]%[Trần Thịnh, dự án 7EX-1-2018]%[7D1Y7]%[7D1B8]
	Tìm $x$, $y$ biết:
	\begin{enumEX}{2}
		\item $\dfrac{3}{5}=\dfrac{6}{x}$;
		\item $x : y=3 : 8$ và $x-y=30$.
	\end{enumEX}
\loigiai{
	\begin{enumEX}{1}
		\item $\dfrac{3}{5}=\dfrac{6}{x}
		\Leftrightarrow 3x=5\cdot 6 =30
		\Leftrightarrow x=30 :3 =10$.
		\item Ta có $x : y=3 : 8\Rightarrow \dfrac{x}{3}=\dfrac{y}{8}$.\\
		Áp dụng tính chất của dãy tỉ số bằng nhau ta có
		$\dfrac{x}{3}=\dfrac{y}{8}=\dfrac{x-y}{3-8}=\dfrac{30}{-5}=-6$\\ 
		$\Rightarrow x=-6\cdot 3=-18$; $y=-6\cdot 8=-48$.
	\end{enumEX}	
	}
\end{ex}

\begin{ex}%[Đề kiểm tra học kỳ I, THCS Cao Minh - Vinh Bảo - Hà Nội, 2017 - 2018]%[Trần Thịnh, dự án 7EX-1-2018]%[7D2B7]
	Cho hàm số $y=ax$ ($a\neq 0$) biết đồ thị hàm số đi qua điểm $A(1;3)$.
	\begin{enumEX}{2}
		\item Xác định hệ số $a$;
		\item Tìm $y$ biết $x=-2$;
		\item Tìm $x$ biết $y=9$;
		\item Vẽ đồ thị hàm số đó.
	\end{enumEX}	
\loigiai{
	\begin{enumerate}
		\item Vì đồ thị hàm số đi qua điểm $A(1;3)$ nên ta có $3=a\cdot 1 \Leftrightarrow a=3$. Vậy hệ số $a=3$.
		\item Với $a=3$ ta có hàm số $y=3x$. Thay $x=-2$ vào ta được $y=3\cdot (-2) =-6$.
		\item Với $a=3$ ta có hàm số $y=3x$. Thay $y=9$ vào ta được $9=3\cdot x \Leftrightarrow x= 3$.
		\immini{
		\vspace*{-0.2cm}
		\item 
		Xét hàm số $y=3x$.\\
		Khi $x=1$ thì $y=3$, ta được $M(1; 3)$.\\
		Đồ thị hàm số $y=3x$ là đường thẳng $OM$.}{
		\begin{tikzpicture}[>=stealth,scale=0.7, line join = round, line cap = round]
		%\draw [dotted, step=1, black, thin, help lines](-2,-2) grid (2,4);
		\draw [->](-2,0) -- (3,0) node [below]{\footnotesize $x$};
		\draw [->](0,-2) -- (0,4) node [left]{\footnotesize $y$};
		\draw (0,0) node[below left] {\footnotesize $O$};
		\foreach \X / \x in {-2/, -1/, 1/1, 2/}
			\draw (\X,2pt) -- (\X,-2pt) node[below] {\footnotesize $\x$};
		\foreach \Y / \y in {-2/, -1/, 1/, 2/,3/3}
			\draw (2pt,\Y) -- (-2pt,\Y) node[left] {\footnotesize $\y$};
		\draw [dashed] (1,0)--(1,3)--(0,3);
		\fill (0,0) circle (1.5pt) (1,3) circle (1.5pt) node[right] {\footnotesize $M$}; 
		\begin{scope}
		\clip (-2,-2) rectangle (3,4);
		\draw[samples=300,domain=-2:2,smooth,variable=\x] plot (\x,{3*\x});
		\end{scope}
		\end{tikzpicture}
		}
	\end{enumerate}	
	
	}
\end{ex}

\begin{ex}%[Đề kiểm tra học kỳ I, THCS Cao Minh - Vinh Bảo - Hà Nội, 2017 - 2018]%[Trần Thịnh, dự án 7EX-1-2018]%[7H2Y1]%[7H2K4]
	Cho tam giác $ABC$ vuông tại $A$ và $\widehat{ABC}-\widehat{ACB}=30^\circ$. Trên cạnh $BC$ lấy điểm $D$ sao cho $BD=BA$. Vẽ tia phân giác $BE$ của $\widehat{ABC}$ ($E\in AC$).
	\begin{enumerate}
		\item Tính $\widehat{ABC}$ và $\widehat{ACB}$;
		\item Chứng minh $\triangle ABE =\triangle DBE$;
		\item Chứng minh $ED \perp BC$;
		\item So sánh $\widehat{ABC}$ và $\widehat{DEC}$.
	\end{enumerate}	
\loigiai{
	\begin{enumerate}
		\item Theo giả thiết ta có $\heva{&\widehat{ABC}-\widehat{ACB}=30^\circ\\ &\widehat{ABC}+\widehat{ACB}=90^\circ} \Rightarrow \heva{&\widehat{ABC}=60^\circ \\& \widehat{ACB}=30^\circ .}$
		\item $\triangle ABE$ và $\triangle DBE$ có $BA=BD$ (giả thiết), $BE$ là cạnh chung, $\widehat{ABE}=\widehat{DBE}$ ($BD$ là phân giác $\widehat{ABC}$).
		Do đó $\triangle ABE =\triangle DBE$ (c.g.c).
		\immini{
		\item Theo câu trên ta có $\triangle ABE =\triangle DBE$,\\
		suy ra $\widehat{BDE}=\widehat{BAE}=90^\circ$. Do đó $ED \perp BC$.
		\item  Tam giác $DEC$ vuông tại $D$ nên\\
		$\widehat{DEC}=90^\circ - \widehat{DCE}=90^\circ -30^\circ =60^\circ$.\\
		Vậy $\widehat{ABC} = \widehat{DEC}$.}{
		\begin{tikzpicture}[>=stealth,scale=1, line join = round, line cap = round]
		\tkzDefPoints{0/0/B,5/0/C}		
		\coordinate (I) at ($(B)!0.5!(C)$);
		\tkzDefPointBy[rotation = center I angle -60](B)\tkzGetPoint{A}
		\tkzDefPointBy[rotation = center B angle -60](A)\tkzGetPoint{D}
		\tkzMarkSegments[mark=|](B,A D,B)
		\coordinate (e) at ($(A)!0.5!(D)$);
		\tkzInterLL(e,B)(C,A)    \tkzGetPoint{E}
		
		\tkzMarkRightAngles[size=0.2](B,A,C)
		\tkzMarkAngles[size=0.5cm,arc=l,mark=|](D,B,E E,B,A)
		\tkzDrawSegments(A,B B,C C,A B,E E,D)
		\tkzDrawPoints[fill=black](A,B,C,D,E)
		\pgfresetboundingbox
		\tkzLabelPoints[above](A) 
		\tkzLabelPoints[above right](E) 
		\tkzLabelPoints[right](C)
		\tkzLabelPoints[left](B)
		\tkzLabelPoints[below](D) 
		\end{tikzpicture}
		}
	\end{enumerate}	
	}
\end{ex}
