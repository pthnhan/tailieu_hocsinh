					\item Ngược lại nếu cho hai góc bằng nhau và có một cặp cạnh là hai tia đối của nhau, bằng cách tính toán ta có thể chứng tỏ được cặp cạnh thứ hai cũng là hai tia đối nhau để suy ra hai góc đó là hai góc đối đỉnh.
				\end{itemize}
		\end{enumerate}
	\end{note}

\begin{vd}
	Cho đường thẳng $AB$ và điểm $O$ trên đường thẳng đó. Trên cùng một nửa mặt phẳng bờ $AB$ vẽ hai tia $OC$ và $OD$ sao cho $\widehat{AOC} = \widehat{BOD} = 50^\circ$.
	\begin{enumerate}
		\item Hai góc $\widehat{AOC}$ và $\widehat{BOD}$ có phải là hai góc đối đỉnh không? Vì sao?
		\item Trên nửa mặt phẳng bờ $AB$ không chứa tia $OD$, vẽ tia $OE$ sao cho tia $OA$ là tia phân giác của góc $\widehat{COE}$. Hai góc $\widehat{BOD}$ và $\widehat{AOE}$ có phải là hai góc đối đỉnh không? Vì sao?
	\end{enumerate}

			\tkzLabelAngles[pos=1.1](C,O,A){$50^\circ$}

		\end{center}
		\begin{enumerate}
			\item Hai góc $\widehat{AOC}$ và $\widehat{BOD}$ có một cặp cạnh $OA$ và $OB$ là hai tia đối nhau, cặp cạnh còn lại $OC$ và $OD$ không phải là hai tia đối nhau (vì cùng nằm trên một nửa mặt phẳng bờ $AB$) nên hai góc đó không phải là hai góc đối đỉnh.
			\item Ta có $\widehat{DOC} = 180^\circ - \left(\widehat{COB} + \widehat{COA}\right) = 180^\circ - \left(50^\circ + 50^\circ\right) = 80^\circ$, $\widehat{COA} = \widehat{AOE} = 50^\circ$ (vì $OA$ là tia phân giác của góc $\widehat{COE}$). Suy ra

			Vậy $\widehat{DOE}$ là góc bẹt nên $OD$ và $OE$ là hai tia đối nhau.\\
			Mặt khác ta có $OA$ và $OB$ cũng là hai tia đối nhau. Do đó $\widehat{BOD}$ và $\widehat{AOE}$ là hai góc đối đỉnh.
		\end{enumerate}
	}
\end{vd}

\begin{vd}

	{
		\immini
		{

			Mặt khác vì $\widehat{yOz} + \widehat{xOz} = 180^\circ$ (kề bù) nên suy ra 
			\[\widehat{xOz} = 180^\circ - \widehat{yOz} = 180^\circ - 90^\circ = 90^\circ.\] 
			Khi đó ta có $\widehat{yOt} = \widehat{xOz} = 90^\circ$ (đối đỉnh).
		}
		{
			\begin{tikzpicture}[>=stealth,scale=0.8, line join = round, line cap = round]
