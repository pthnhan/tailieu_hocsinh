\begin{bt}%[Phan Minh Tâm, 9TL-03]%[9D3B4]
	Biết $ (a;b) $ là nghiệm của hệ phương trình $ \heva{&7x+4y=18\\&3x-4y=2}. $ Tính $ S=a+b. $
	\choice
	{$ S=-1 $}
	{$ S=-3 $}
	{\True $ S = 3 $}
	{$ S = 5 $}
	\loigiai{
		\begin{eqnarray*}
			\heva{&7x+4y=18\\&3x-4y=2} \Leftrightarrow \heva{&10x=20\\&3x-4y=2} \Leftrightarrow \heva{&x=2\\&y=1.}	
		\end{eqnarray*}
		Vậy $ S = 2+1 = 3. $
	}
\end{bt}

	\end{eqnarray*}	}
\end{bt}

\begin{dang}{Giải hệ phương trình.}
	\begin{itemize}
		\item  Giải hệ phương trình bằng phương pháp thế. 
		\item  Giải hệ phương trình bằng phương pháp cộng đại số.
		\item  Giải hệ bằng phương pháp đặt ẩn phụ.
	\end{itemize}
	
\end{dang}
\begin{bt}
	%Phuong pháp the
	Giải các hệ phương trình	
	\begin{enumEX}{2}  
		\item $\heva{&2x+y=1\\&3x+4y=-1.}$
		\item $\heva{&5x+6y=17\\&9x-y=7.}$
		\item $\begin{cases}
		2x-3y = 7 \\
		3x + 2y = 4. 
		\end{cases}$  
		\item $\heva{&\dfrac{1}{2x - 1} + \dfrac{4}{y + 5} = 3\\&\dfrac{3}{2x - 1} - \dfrac{2}{y + 5} =  - 5.}$
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item $\heva{&2x+y=1\\&3x+4y=-1}\Leftrightarrow \heva{&y=1-2x\\&3x+4(1-2x)=-1}\Leftrightarrow \heva{&y=1-2x\\&-5x=-5}\Leftrightarrow \heva{&x=1\\&y=-1.}$\\
			Vậy hệ phương trình có nghiệm là $(x;y)=(1;-1)$.
			\item $\heva{&5x+6y=17\\&9x-y=7}\Leftarrow \heva{&5x+6(9x-7)=17\\&y=9x-7}\Leftrightarrow \heva{&59x=59\\&y=9x-7}\Leftrightarrow \heva{&x=1\\&y=2.}$\\
			Vậy hệ phương trình có nghiệm là $(x;y)=(1;2)$.
			\item Ta có hệ phương trình tương đương với $$\begin{cases}
			6x-9y = 21 \\
			6x + 4y = 8 
			\end{cases}
			\Leftrightarrow 
			\begin{cases}
			13y=-13 \\
			3x + 2y = 4 
			\end{cases}
			\Leftrightarrow 
			\begin{cases}
			y=-1 \\
			x = 2. 
			\end{cases}$$
			Vậy hệ phương trình có nghiệm duy nhất $(x; y)=\left(2; -1\right)$.
			\item Xét hệ
			$\heva{&\dfrac{1}{2x - 1} + \dfrac{4}{y + 5} = 3\\
				&\dfrac{3}{2x - 1} - \dfrac{2}{y + 5} =  - 5}$ $(I)$\\
			Điều kiện $\heva{&x \ne \dfrac{1}{2}\\&y \ne  - 5.}$\\
			Khi đó, $\left( I \right) \Leftrightarrow \left\{ \begin{array}{l}
			\dfrac{1}{2x - 1} + \dfrac{4}{y + 5} = 3\\
			\dfrac{6}{2x - 1} - \dfrac{4}{y + 5} =  - 10
			\end{array} \right.\,\, \Leftrightarrow \left\{ \begin{array}{l}
			\dfrac{1}{2x - 1} + \dfrac{4}{y + 5} = 3\\
			\dfrac{7}{2x - 1} =  - 7
			\end{array} \right. \Rightarrow \left\{ \begin{array}{l}
			x = 0\\
			y =  - 4.
			\end{array} \right.$\\
			So với điều kiện, thỏa mãn.\\
			Vậy hệ phương trình có nghiệm $(x;y)=(0;-4)$ .
		\end{enumerate}
	}
\end{bt}

\begin{bt}
	%Phuong pháp cong dai so

	\begin{enumEX}{2}

\begin{bt}
	Giải các hệ phương trình	
	\begin{enumEX}{2}  
		\item $\heva{&\dfrac{-5}{x-1}+\dfrac{1}{y-1}=10\\&\dfrac{1}{x-1}+\dfrac{3}{y-1}=-18.}$
		\item $\heva{\sqrt{x}+2\sqrt{y-1}=5\\ 4\sqrt{x}-\sqrt{y-1}=2.}$
		\item $\begin{cases} 2\sqrt{x-2}+\dfrac{y}{y+3}=1\\ 4\sqrt{x-2}-\dfrac{3y}{y+3}=7.\end{cases}$
		\item $
		\left\{\begin{aligned} & \dfrac{2}{x-2}+\dfrac{1}{y+1}=3 \\& \dfrac{3}{x-2}-\dfrac{2}{y+1}=8.\end{aligned}\right. $
		\item $\heva{&\vert x+5\vert -\dfrac{2}{\sqrt{y}-2}=4\\&\vert x+5\vert -\dfrac{1}{\sqrt{y}-2}=3.}$
		\item $\begin{cases}\dfrac{9}{\sqrt{2x-1}}-\dfrac{3}{y+1}=2\\ \dfrac{4}{\sqrt{2x-1}}-\dfrac{1}{y+1}=1.\end{cases}$
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item Điều kiện $x\neq 1$ và $y\neq 1$.\\
			Đặt $\dfrac{1}{x-1}=u$; $\dfrac{1}{y-1}=v$. Hệ phương trình đã cho trở thành
			$$\heva{&-5u+v=10\\&u+3v=-18}\Leftrightarrow \heva{&v=5u+10\\&u+3(5v+10)=-18}\Leftrightarrow \heva{&v=5u+10\\&16u=-48}\Leftrightarrow \heva{&u=-3\\&v=-5.}$$
			Với $u=-3$; $v=-5$ ta có hệ
			$$\heva{&\dfrac{1}{x-1}=-3\\&\dfrac{1}{y-1}=-5}\Leftrightarrow \heva{&-3(x-1)=1\\&-5(y-1)=1}\Leftrightarrow \heva{&-3x=-2\\&-5y=-4}\Leftrightarrow \heva{&x=\dfrac{2}{3}\\&y=\dfrac{4}{5}.}$$
			Đối chiếu với điều kiện, hệ phương trình có nghiệm $(x;y)=\left(\dfrac{2}{3};\dfrac{4}{5}\right)$.
			\item Điều kiện: $ x \ge 0$, $y\ge 1.$
			
			Đặt $\heva{&a=\sqrt{x}\\&b=\sqrt{y-1}}$. Điều kiện $a;b\ge 0$. Khi đó hệ phương trình ban đầu trở thành
			
			$\heva{&a+2b=5\\&4a-b=2} \Leftrightarrow \heva{&a=5-2b\\&4(5-2b)-b=2} \Leftrightarrow \heva{&a=5-2b\\&-9b=-18} 
			\Leftrightarrow
			\heva{&a=1\\&b=2.}$\\
			Do đó $\heva{&\sqrt{x}=1\\&\sqrt{y-1}=2} \Leftrightarrow \heva{&x=1\\&y-1=4} \Leftrightarrow \heva{&x=1\\&y=5.}$\\
			Vậy hệ phương trình có nghiệm  $(x;y)=(1;5).$
			\item Điều kiện $x\geq 2,\ y\neq -3$. Đặt $a=\sqrt{x-2},\ b=\dfrac{y}{y+3}$. Thu được\\
			$\begin{aligned}
			& \begin{cases}2a+b=1\\ 4a-3b=7\end{cases}\Leftrightarrow \begin{cases}4a+2b=2\\ 4a-3b=7\end{cases}\Leftrightarrow \begin{cases}5b=-5\\ 2a+b=1\end{cases}\Leftrightarrow \begin{cases}b=-1\\ a=1.\end{cases}
			\end{aligned}$\\
			Từ đó suy ra\\
			$\begin{cases}\sqrt{x-2}=1\\ \dfrac{y}{y+3}=-1\end{cases}\Leftrightarrow \begin{cases}x-2=1\\ y=-y-3\end{cases}\Leftrightarrow \begin{cases}x=3\\ y=-\dfrac{3}{2}.\end{cases}$\\
			Vậy hệ có nghiệm $(x;y)=\left(3;-\dfrac{3}{2}\right)$.
			\item Điều kiện xác định $x \neq 2$ và $y \neq -1$.\\
			Đặt $u=\dfrac{1}{x-2}$ và $v=\dfrac{1}{y+1}$, hệ phương trình đã cho trở thành
			\begin{equation*}
			\left\{\begin{aligned} & 2u+v=3 \\& 3u-2v=8 \end{aligned}\right.
			\Leftrightarrow  \left\{\begin{aligned} & 4u+2v=6 \\& 3u-2v=8\end{aligned}\right.
			\Leftrightarrow  \left\{\begin{aligned} & 7u=14 \\& v=3-2u\end{aligned}\right.
			\Leftrightarrow  \left\{\begin{aligned} & u=2 \\& v=-1.\end{aligned}\right.
			\end{equation*}
			Như vậy ta được
			\begin{equation*}
			\left\{\begin{aligned} & \dfrac{1}{x-2}=2 \\& \dfrac{1}{y+1}=-1\end{aligned}\right. \Leftrightarrow \left\{\begin{aligned} & x=\dfrac{5}{2} \\& y=-2.\end{aligned}\right. \text{ (thỏa mãn điều kiện xác định)}
			\end{equation*}
			Vậy hệ phương trình đã cho có nghiệm duy nhất $(x;y)=\left(\dfrac{5}{2};-2\right)$. 
			\item Điều kiện $y\geq 0;y\ne 4$.\\
			Đặt $a=\vert x+5\vert;b=\dfrac{1}{\sqrt{y}-2}(a\geq 0)$.\\
			Ta có hệ $\heva{&a-2b=4\\&a+b=3}\Leftrightarrow \heva{&a=\dfrac{10}{3}\\&b=-\dfrac{1}{3}.}$\\
			Ta có $\sqrt{y}-2=-3\Leftrightarrow\sqrt{y}=-1$(vô nghiệm).\\
			Vậy hệ phương trình vô nghiệm.
			\item Điều kiện $x>\dfrac{1}{2}$, $y\ne -1$.\\
			Đặt $\dfrac{1}{\sqrt{2x-1}}=a$ và $\dfrac{1}{y+1}=b$, hệ phương trình đã cho trở thành $$\begin{cases}9a-3b=2\\ 4a-b=1\end{cases}\Rightarrow \begin{cases}a=\dfrac{1}{3}\\ b=\dfrac{1}{3}\end{cases}\Rightarrow \begin{cases}\dfrac{1}{\sqrt{2x-1}}=\dfrac{1}{3}\\ \dfrac{1}{y+1}=\dfrac{1}{3}\end{cases}\Rightarrow \begin{cases}x=5\\ y=2.\end{cases}$$
			Vậy nghiệm của hệ phương trình đã cho là $(x;y)=(5;2).$
		\end{enumerate}
	}
\end{bt}
\begin{dang}{Giải và biện luận hệ phương trình}
	Biến đổi từ hệ phương trình về phương trình theo ẩn $x$ hoặc $y$. Từ đó giải và biện luận phương trình một ẩn.
\end{dang}
\begin{bt}
	Giải và biện luận hệ phương trình $\heva{&x-my=2\\&mx-4y=m-2.}$
	\loigiai{
		$(I) \heva{&x-my=2 \quad&(1)&\\&mx-4y=m-2\quad &(2)&.}$\\
		Từ $(1)$ ta có $x=my+2$ thay vào $(2)$ ta được 
		$$m(my+2)-4y=m-2\Leftrightarrow (m-2)(m+2)y=-(m+2)\quad (3).$$
		$\bullet$ Nếu $m=2$ phương trình $(3)$ trở thành $0y=-4$. Phương trình $(3)$
		vô nghiệm, do đó hệ phương trình $(I)$
		vô nghiệm.\\
		$\bullet$ Nếu $m=-2$ phương trình $(3)$ trở thành $0y=0$. Phương trình $(3)$
		có vô số nghiệm $y\in \mathbb{R}$, do đó hệ phương trình $(I)$
		có vô số nghiệm $\heva{&x=-2y+2\\&y\in \mathbb{R}.}$\\
		$\bullet$ Nếu $m \neq \pm 2$ thì phương trình $(3)$  có nghiệm duy nhất $y=\dfrac{-1}{m-2}$.\\
		Do đó hệ phương trình $(I)$ có nghiệm duy nhất $\heva{&x=\dfrac{4-m}{2-m}\\&y=\dfrac{-1}{m-2}.}$\\
		Kết luận\\
		$\bullet$ Với $m\neq \pm 2$ hệ phương trình $(I)$ có nghiệm duy nhất $\heva{&x=\dfrac{4-m}{2-m}\\&y=\dfrac{-1}{m-2}.}$\\
		$\bullet$ Với $m=-2$ hệ phương trình $(I)$ có vô số nghiệm $\heva{&x=-2y+2\\&y\in \mathbb{R}.}$\\
		$\bullet$ Với $m=2$ hệ phương trình $(I)$ vô nghiệm.
		\begin{bt}
			Giải và biện luận hệ phương trình $\heva{&mx+y=4\\&x-my=1.}$
			\loigiai{
				Từ phương trình thứ nhất có $y=4-mx$, thay vào phương trình thứ hai ta được
				$$x-m(4-mx)=1\Leftrightarrow (1+m^2)x=1+4m\quad (*).$$
				Với mọi $m$, phương trình $(*)$ có nghiệm duy nhất $x=\dfrac{1+4m}{1+m^2}\Rightarrow y=4-\dfrac{m(1+4m)}{1+m^2}=\dfrac{4-m}{1+m^2}$.\\
				Kết luận: Với mọi $m$, hệ có nghiệm duy nhất $\heva{&x=\dfrac{1+4m}{1+m^2}\\&y=\dfrac{4-m}{1+m^2}.}$
			}
		\end{bt}
		
		\begin{bt}
			Giải và biện luận hệ phương trình $\heva{&mx+y=2m &\quad (1)&\\&x+my=m+1 &\quad(2)&.}$
			\loigiai{
				Từ $(1)$ ta có $y=2m-mx$ thay vào $(2)$ ta được
				\begin{eqnarray*}
					x+m(2m-mx)=m+1 &\Leftrightarrow& (1-m^2)x=-2m^2+m+1\\
					&\Leftrightarrow& (m-1)(m+1)x=(m-1)(2m+1)\quad (3).
				\end{eqnarray*}
				$\bullet$ Nếu $m=1$ phương trình $(3)$ trở thành $0x=0$. Phương trình $(3)$ có vô số nghiệm $x\in \mathbb{R}$, do đó hệ có vô số nghiệm $\heva{&x\in \mathbb{R}\\& y=2-x.}$\\
				$\bullet$ Nếu $m=-1$ phương trình $(3)$ trở thành $0x=2$. Phương trình $(3)$ có vô  nghiệm , do đó hệ  vô nghiệm.\\
				$\bullet$ Nếu $m=\pm1$ phương trình $(3)$ có nghiệm duy nhất $x=\dfrac{2m+1}{m+1}$.\\ Do đó hệ có nghiệm duy nhất $\heva{&x=\dfrac{2m+1}{m+1}\\&y=\dfrac{m}{m+1}.}$\\
				Kết luận\\
				$\bullet$ Với $m=\pm1$  hệ có nghiệm duy nhất $\heva{&x=\dfrac{2m+1}{m+1}\\&y=\dfrac{m}{m+1}.}$\\
				$\bullet$ Với $m=1$  hệ có vô số nghiệm $\heva{&x\in \mathbb{R}\\& y=2-x.}$\\
				$\bullet$ Với $m=-1$  hệ  vô nghiệm.
			}
		\end{bt}
}\end{bt}
\begin{dang}{Xác định tham số để hệ có nghiệm thỏa mãn điều kiện đề bài}
\end{dang}
\begin{bt}
	Cho hệ phương trình $\heva{&ax+y=1\\&ax+by=-5}$ ($a, b$ là tham số).\\	
	Tìm $a$, $b$ để hệ có nghiệm $(2;-3)$. 
	\loigiai{
		Hệ phương trình có nghiệm $x=2$ và $y=-3$ khi $\heva{&2a-3=1\\&2a-3b=-5}\Leftrightarrow \heva{&a=2\\&b=3.}$\\
		Vậy $a=2, b=3$ thì hệ có nghiệm $(2;-3)$.
	}
\end{bt}		
\begin{bt}
	Cho hệ phương trình $\heva{&x=2\\&mx+y=m^2+3}$ ($m$ là tham số).\\	
	Tìm $m$ sao cho hệ có nghiệm  thỏa mãn $x+y$ là nhỏ nhất. 
	\loigiai{
		Ta có $\heva{&x=2\\&mx+y=m^2+3}\Leftrightarrow \heva{&x=2\\&m\cdot 2+y=m^2+3}\Leftrightarrow \heva{&x=2\\&y=m^2-2m+3.}$\\
		Vậy với mọi $m$ hệ luôn có nghiệm duy nhất $x=2, y=m^2-2m+3$.\\
		Có $x+y=m^2-2m+5=(m-1)^2+4\ge 4,\forall m\in \mathbb{R}$.\\
		Suy ra $x+y$ nhỏ nhất bằng $4$ khi $m=1$.\\
		Kết luận: $m=1$ thì hệ có nghiệm $(x;y)$ thỏa mãn tổng $x+y$ là nhỏ nhất.
	}
\end{bt}	
\begin{bt}
	Cho hệ phương trình $\heva{&x+my=3\\&mx+y=-3.}$\\
	Tìm $m$ để hệ có nghiệm duy nhất $(x;y)$ thỏa mãn $x>0$?
	\loigiai{
		Từ phương trình thứ hai ta được $y=-3-mx$, thay vào phương trình thứ nhất ta được
		$$x+m(-3-mx)=3\Leftrightarrow (1-m^2)x=3m+3\Leftrightarrow (1-m^2)x=3(m+1)\quad (*).$$
		Hệ có nghiệm duy nhất khi phương trình $(*)$ có nghiệm duy nhất $\Leftrightarrow 1-m^2\neq 0\Leftrightarrow m\neq \pm 1.$\\
		Khi đó $(*)$ có nghiệm duy nhất $x=\dfrac{3(1+m)}{1-m^2}=\dfrac{3}{1-m}$.\\
		Do điều kiện $x>0\Leftrightarrow \dfrac{3}{1-m}>0\Leftrightarrow 1-m>0\Leftrightarrow m<1$.\\
		Kết luận\\
		Với $m<1$ và $m\neq -1$ thì hệ có nghiệm duy nhất $(x;y)$ thỏa mãn $x>0$.}
\end{bt}	
\begin{bt}
	Cho hệ phương trình $\heva{&2x+3y=m\\&5x-y=1.}$\\
	Tìm $m$ đê hệ có nghiệm $(x;y)$ thỏa mãn $x>0; \; y<0$?
	\loigiai{
		Ta có $\heva{&2x+3y=m\\&5x-y=1}\Leftrightarrow \heva{&2x+3(5x-1)=m\\&y=5x-1}\Leftrightarrow \heva{&x=\dfrac{m+3}{17}\\&y=\dfrac{5m-2}{17}.}$\\
		Hệ có nghiệm thỏa mãn $\heva{&x>0\\&y<0}\Leftrightarrow \heva{&\dfrac{m+3}{17}>0\\&\dfrac{5m-2}{17}<0}\Leftrightarrow \heva
		{&m>-3\\&m<\dfrac{2}{5}}\Leftrightarrow -3<m<\dfrac{2}{5}$.
	}
	
\end{bt}
\begin{bt}
	Cho hệ phương trình $\heva{&x+2y=3m+3 &\quad(1)&\\&4x-3y=m-10 &\quad(2).}$\\
	Tìm $m$ để hệ có nghiệm duy nhất $(x;y)$ thỏa mãn $x^2+y^2$ đạt giá trị nhỏ nhất?
	\loigiai{
		Từ $(1)$ suy ra $x=3m+3-2y\quad(3)$, thay vào $(2)$ ta được
		\begin{eqnarray*}
			4(3m+3-2y)-3y=m-10 &\Leftrightarrow& 12m+12-8y-3y=m-10\\
			&\Leftrightarrow& -11y=-11m-22 \Leftrightarrow y=m+2.
		\end{eqnarray*}
		Thay $y=m+2$ vào $(3)$ ta được $x=m-1$.\\
		Có $x^2+y^2=(m-1)^2+(m+2)^2=2m^2+2m+5=2\left(m+\dfrac{1}{2}\right)^2+\dfrac{9}{2}\ge \dfrac{9}{2}\,\forall m\in \mathbb{R}$.\\
		Đẳng thức xảy ra khi $m=-\dfrac{1}{2}$.\\
		Vậy $m=-\dfrac{1}{2}$ thì hệ có nghiệm duy nhất $(x;y)$ thỏa mãn $x^2+y^2$ đạt giá trị nhỏ nhất.
	}
\end{bt}
\subsubsection{Giải toán bằng cách lập hệ phương trình}
Các bước làm tổng quát.
\begin{enumerate}[Bước 1.]
	\item Lập hệ phương trình.
	\begin{itemize}
		\item Chọn ẩn (thường là các đại lượng cần tìm) và đặt điều kiện thích hợp cho chúng.
		\item Biểu diễn các đại lượng chưa biết theo các ẩn và các đại lượng đã biết.
		\item Lập hệ phương trình biểu thị mối quan hệ giữa các đại lượng.
	\end{itemize}
	\item Giải hệ phương trình vừa lập.
	\item Kiểm tra xem các nghiệm của hệ có thỏa mãn điều kiện đặt ra hay không, rồi trả lời yêu cầu bài toán.
\end{enumerate}

\begin{dang}{Toán số học, phần trăm}
	\begin{itemize}
		\item Biểu diễn của một số tự nhiên. Ví dụ số tự nhiên có $3$ chữ số có biểu diễn là $\overline{abc}=100a+10b+c$, với điều kiện $a\in \mathbb{N}^*$; $b$, $c\in \mathbb{N}$; $a$, $b$, $c\le 9$.
		\item Phép chia có dư $a=b\cdot q+r$. Trong đó $a$ là số bị chia, $b$ là số chia, $q$ là thương và $r$ là số dư.
		\item Phép tính cộng, trừ, nhân, chia với phần trăm.
	\end{itemize}
\end{dang}
\begin{bt}
	Tìm số tự nhiên có hai chữ số, biết hiệu giữa chữ số hàng chục và chữ số hàng đơn vị là $5$. Nếu lấy số đã cho chia cho số viết theo thứ tự ngược lại ta được thương là $3$ và số dư là $13$.
	\loigiai{
		Gọi số phải tìm là $\overline{xy}=10x+y$ ($x\in \mathbb{N}^*$; $y\in \mathbb{N}$; $x$, $y\le 9$).\\
		Ta có hệ phương trình sau
		$$\heva{&x-y=5\\&10x+y=3(10y+x)+13} \Leftrightarrow \heva{&x=6\\&y=1.}$$
		So với điều kiện ta thấy thỏa mãn.\\
		Vậy số cần tìm là $61$.
	}
\end{bt}
\begin{bt}
	Tìm một số có hai chữ số, biết rằng tổng các chữ số của số đó bằng $9$ và viết các chữ số theo thứ tự ngược lại thì được một số bằng $\dfrac{2}{9}$ số ban đầu.
	\loigiai{
		Gọi số phải tìm là $\overline{xy}=10x+y$ ($x\in \mathbb{N}^*$; $y\in \mathbb{N}$; $x$, $y\le 9$).\\
		Ta có hệ phương trình sau
		$$\heva{&x+y=9\\&10y+x=\dfrac{2}{9} (10x+y)} \Leftrightarrow \heva{&x=8\\&y=1.}$$
		So với điều kiện ta thấy thỏa mãn.\\
		Vậy số cần tìm là $81$.
	}
\end{bt}
\begin{bt}
	Tháng thứ nhất hai tổ sản xuất được $900$ chi tiết máy. Tháng thứ hai tổ một vượt mức $15\%$ và tổ hai vượt mức $10\%$ so với tháng thứ nhất. Vì vậy hai tổ đã sản xuất được $1010$ chi tiết máy. Hỏi tháng thứ hai mỗi tổ sản xuất được bao nhiêu chi tiết máy?
	\loigiai{
		Gọi số chi tiết máy tổ một làm trong tháng thứ nhất là $x$, số chi tiết máy tổ hai làm trong tháng thứ hai là $y$ ($x$, $y\in \mathbb{N}^*$).\\
		Số chi tiết máy tổ một làm trong tháng thứ hai là $x+15\% x=1{,}15x$.\\
		Số chi tiết máy tổ hai làm trong tháng thứ hai là $y+10\% y=1{,}1y$.\\
		Ta có hệ phương trình sau
		$$\heva{&x+y=900\\&1{,}15x+1{,}1y=1010} \Leftrightarrow \heva{&x=400\\&y=500.}$$
		So với điều kiện ta thấy thỏa.\\
		Vậy trong tháng hai, tổ một làm được $1{,}15x=460$ chi tiết máy, tổ hai làm được $1{,}1y=550$ chi tiết máy.
	}
\end{bt}

\begin{dang}{Toán năng suất công việc}
	Năng suất công việc bằng khối lượng công việc chia cho thời gian hoàn thành.
\end{dang}
\begin{bt}
	Hai anh em nông dân cùng cày trên một cánh đồng. Mỗi ngày người anh cày hơn người em $10$ m$^2$ đất. Sau ba ngày làm việc, cả hai anh em cày được $930$ m$^2$ đất. Hỏi năng suất mỗi người làm trong một ngày là bao nhiêu?
	\loigiai{
		Gọi năng suất người anh làm trong một ngày là $x$, năng suất người em làm trong một ngày là $y$ ($x$, $y>0$).\\
		Ta có hệ phương trình sau
		$$\heva{&x-y=10\\&3x+3y=930} \Leftrightarrow \heva{&x=160\\&y=150.}$$
		So với điều kiện ta thấy thỏa.
		Vậy năng suất người anh làm trong một ngày là $160$ m$^2$, năng suất người em làm trong một ngày là $150$ m$^2$.
		
	}
\end{bt}

\begin{dang}{Toán chuyển động}
	\begin{itemize}
		\item Công thức vận tốc, quãng đường, thời gian.\\
		$v=\dfrac{s}{t}$ hoặc $s=v\cdot t$ hoặc $t=\dfrac{s}{v}$.\\
		Trong đó $v$ là vận tốc, $s$ là quãng đường, $t$ là thời gian.
		\item Các đại lượng phải đổi đơn vị cho phù hợp.
	\end{itemize}
\end{dang}
\begin{bt}
	Một khách du lịch đi trên ô tô $5$ giờ, sau đó đi tiếp bằng tàu hỏa trong $7$ giờ được quãng đường dài $720$ km. Hỏi vận tốc của tàu hỏa và ô tô, biết rằng mỗi giờ tàu hỏa đi nhanh hơn ô tô $12$ km.
	\loigiai{
		Gọi vận tốc của ô tô là $x$ (km/h), vận tốc của tàu hỏa là $y$ (km/h) ($x>0$, $y>0$).\\
		Quãng đường khách du lịch đi bằng ô tô là $5x$ (km).\\
		Quãng đường khách du lịch đi bằng tàu hỏa là $7x$ (km).\\
		Theo giả thiết ta có $5x+7y=720$.\\
		Do vận tốc tàu hỏa hơn ô tô $12$ km nên ta có $y-x=12$.\\
		Ta có hệ phương trình sau $$\heva{&y-x=12\\&5x+7y=720} \Leftrightarrow \heva{&x=53\\&y=65.}$$
		So với điều kiện ta thấy thỏa.\\
		Vậy vận tốc tàu hỏa là $65$ km/h; vận tốc của ô tô là $53$ km/h.
	}
\end{bt}
\begin{bt}

		Gọi thời gian vòi thứ nhất chảy một mình đầy bể là $x$ (giờ), thời gian vòi thứ hai chảy một mình đầy bể là $y$ (giờ) ($x>0$, $y>0$).\\
		Trong một giờ, vòi thứ nhất chảy được $\dfrac{1}{x}$ bể.\\

		So với điều kiện ta thấy thỏa mãn.\\
		Vậy vòi thứ nhất chảy một mình đầy bể trong $2$ giờ, vòi thứ hai chảy một mình đầy bể trong $3$ giờ.
	}

\begin{bt}
	Giải các hệ phương trình sau:
	\begin{enumEX}{2}
		\item $\heva{&x+y+z=11\\&2x-y+z=5\\&3x+2y+z=14.}$
		\item $\heva{&x+y+z+t=4\\&x+y-z-t=8\\&x-y+z-t=12\\&x-y-z+t=16.}$
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item $\heva{&x+y+z=11\\&2x-y+z=5\\&3x+2y+z=14} \Leftrightarrow \heva{&z=11-x-y\\&x-2y=-6\\&2x+y=3} \Leftrightarrow \heva{&z=11-x-y\\&-2x+4y=12\\&2x+y=3} \Leftrightarrow \heva{&z=11-x-y\\&5y=15\\&2x+y=3} \\ \Leftrightarrow
			\heva{&z=8\\&y=3\\&x=0.}$
			\item $\heva{&x+y+z+t=4\\&x+y-z-t=8\\&x-y+z-t=12\\&x-y-z+t=16} \Leftrightarrow \heva{&x+y+z+t=4\\&x+y-z-t=8\\&x-y+z-t=12\\&4x=40} \Leftrightarrow \heva{&y+z+t=-6\\&y-z-t=-2\\&-y+z-t=2\\&x=10} \Leftrightarrow \heva{&y+z+t=-6\\&2y=-8\\&-y+z-t=2\\&x=10}\\ \Leftrightarrow \heva{&z+t=-2\\&y=-4\\&z-t=-2\\&x=10} \Leftrightarrow \heva{&z=-2\\&y=-4\\&t=0\\&x=10.}$
		\end{enumEX}
	}
\end{bt}

\begin{dang}{Giải toán bằng cách lập hệ phương trình}
	Phân tích kĩ từng ý và áp dụng các bước làm tổng quát.
\end{dang}
\begin{bt}
	Lúc $7$ giờ, An khởi hành từ A để đến gặp Bích tại B lúc $9$ giờ $30$ phút. Nhưng đến $9$ giờ, An được biết Bích bắt đầu đi từ B để đến C (không nằm trên quãng đường AB) với vận tốc bằng $3{,}25$ lần vận tốc của An. Ngay lúc đó An tăng thêm vận tốc $1$ km/h và khi tới B, An đã đi theo đường tắt đến C chỉ dài bằng $\dfrac{1}{3}$ quãng đường mà Bích đi từ B đến C, do đó An và Bích đến C cùng một lúc. Nếu Bích cũng đi theo đường tắt như An thì Bích đến B trước An là $2$ giờ. Tính vận tốc lúc đầu của An.
	\loigiai{
		Gọi vận tốc lúc đầu của An là $x$ (km/h) ($x>0$).\\
		Vận tốc của Bích là $3{,}25x$ (km/h).\\
		Gọi chiều dài quãng đường tắt BC là $y$ (km) ($y>0$).\\
		Quãng đường Bích đi từ B đến C là $3y$ (km).\\
		Gọi $N$ là vị trí mà An bắt đầu tăng vận tốc.\\
		Quãng đường NB mà An dự định đi trong $\dfrac{1}{2}$ giờ là $0{,}5x$ (km).\\
		An đi từ N đến C (theo đường tắt) được $0{,}5x+y$ (km) với vận tốc $x+1$ (km/h) hết $\dfrac{0{,}5x+y}{x+1}$ (giờ).\\
		Trong thời gian đó, Bích đi $3y$ (km) với vận tốc $3{,}25x$ (km/h).\\
		Ta có phương trình thứ nhất $$\dfrac{0{,}5x+y}{x+1}=\dfrac{3y}{3{,}25x}.$$
		Nếu Bích đi đường tắt từ B đến C thì chỉ hết $\dfrac{y}{3{,}25x}$ (giờ), ít hơn so với đi theo đường vòng $2$ giờ. Ta có phương trình thứ hai $$\dfrac{3y}{3{,}25x}-\dfrac{y}{3{,}25x}=2 \Leftrightarrow y=3{,}25x.$$
		Thay $y=3{,}25x$ vào phương trình thứ nhất ta được $x=4$.\\
		So với điều kiện ta thấy thỏa.\\
		Vậy vận tốc lúc đầu của An là $4$ km/h.
	}
\end{bt}
