\section{PHÂN TÍCH ĐA THỨC THÀNH NHÂN TỬ BẰNG PHƯƠNG PHÁP NHÓM HẠNG TỬ}
\subsection{Tóm tắt lí thuyết}
Nhóm các số hạng một cách thích hợp để có thể dùng phương pháp đặt nhân tử chung hoặc dùng hằng đẳng thức đối với mỗi nhóm. Sau đó tiếp tục đặt nhân tử chung hoặc dùng hằng đẳng thức.

\subsection{Các dạng toán}
\begin{dang}[Phân tích đa thức thành nhân tử bằng phương pháp nhóm các hạng tử]
Nhóm các số hạng của đa thức thành từng nhóm rồi phân tích từng nhóm thành nhân tử. Tiếp tục phân tích đến khi được một tích của các đa thức.
\end{dang}

\begin{vd}
Phân tích đa thức sau thành nhân tử
\begin{listEX}[2]
\item $A = ax - bx + ab - x^2$;
\item $B = x^2 - y^2 + 4x + 4$.
\end{listEX}	
	\loigiai{
\begin{enumerate}
\item Nhóm các hạng tử đầu và cuối với nhau, nhóm hai hạng tử còn lại với nhau:
$$A = ax - x^2 - bx + ab = x(a - x) + b(a - x) = (a - x)(x-b).$$
\item  Nhóm các hạng tử thứ nhất, thứ ba, thứ tư với nhau
$$B = x^2 + 4x + 4 - y^2 = (x + 2)^2 - y^2 = (x + 2 - y)(x + 2 + y).$$
\end{enumerate}
\begin{note}
Lưu ý:
\begin{itemize}
	\item Ở câu a), ta có thể nghĩ đến cách nhóm thành hai nhóm, mỗi nhóm gồm hai hạng tử liền nhau:
$$A = (ax - bx) + \left(ab - x^2\right) = x(a - b) + \left( ab - x^2\right).$$
Đến đây gặp bế tắc.
\item 	Ở câu b), ta cũng có thể nghĩ đến cách nhóm như trên:
$$B = \left(x^2 - y^2\right) + (4x + 4y) =(x + y)(x - y) + 4(x + 1).$$
Đến đây gặp bế tắc.
\item 	Các ví dụ trên cho thấy: Cần tìm cách nhóm các hạng tử một cách thích hợp. cách thích hợp đó là nhóm các hạng tử nhằm làm xuất hiện nhân tử chung  $a - x$  (câu a), hoặc xuất hiện dạng  $a^2 - b^2$  (câu b).
\end{itemize}
\end{note}		
	}
\end{vd}

\begin{vd}
Phân tích đa thức sau thành nhân tử	
\begin{listEX}[2]
	\item $x^3+x^2+x+1$;
	\item $x^2y+xy^2-x-y$;
	\item $x^2-xy+5x-5y$;
	\item $2x^2-x-6xy+3y$.
\end{listEX}
	\loigiai{
\begin{enumerate}
	\item Ta có
	{\allowdisplaybreaks \begin{eqnarray*}
		x^3+x^2+x+1&=&\left(x^3+x^2\right)+(x+1)\\
		&=&x^2(x+1)+(x+1)\\
		&=&(x+1)\left(x^2+1\right).
	\end{eqnarray*}}
	\item Ta có
	{\allowdisplaybreaks \begin{eqnarray*}
		x^2y+xy^2-x-y&=&\left(x^2+xy^2\right)-(x+y)\\
		&=&xy(x+y)-(x+y)\\
		&=&(x+y)(xy-1).
	\end{eqnarray*}}
	\item Ta có
	{\allowdisplaybreaks \begin{eqnarray*}
		x^2-xy+5x-5y&=&\left(x^2-xy\right)+5(x-y)\\
		&=&x(x-y)+5(x-y)\\
		&=&(x-y)(x+5).
	\end{eqnarray*}}
	\item Ta có
	{\allowdisplaybreaks \begin{eqnarray*}
		2x^2-x-6xy+3y&=&\left(2x^2-x\right)-(6xy-3y)\\
		&=&x(2x-1)-3y(2x-1)\\
		&=&(2x-1)(x-3y).
	\end{eqnarray*}}
\end{enumerate}
	}
\end{vd}




\begin{vd}
Phân tích đa thức sau thành nhân tử
\begin{listEX}[2]
\item $x^2+7x+7y-y^2$;
\item $x^2-2x-9y^2+6y$;
\item $x^2+2xy-25+y^2$;
\item $x^2y^2-x^2+8x-16$;
\item $x^2-6xy+9y^2+4x-12y$;
\item $x^2-xy+x^3-3x^2y+3xy^2-y^3$.
\end{listEX}
\loigiai{
\begin{enumerate}
\item Ta có
{\allowdisplaybreaks \begin{eqnarray*}
	x^2+7x+7y-y^2&=&\left(x^2-y^2\right)+(7x+7y)\\
	&=&(x-y)(x+y)+7(x+y)\\
	&=&(x+y)(x-y+7).
\end{eqnarray*}}
\item Ta có
{\allowdisplaybreaks \begin{eqnarray*}
	x^2-2x-9y^2+6y&=&\left(x^2-9y^2\right)-(2x-6y)\\
	&=&(x-3y)(x+3y)-2(x-3y)\\
	&=&(x-3y)(x+3y-2).
\end{eqnarray*}}
\item Ta có
{\allowdisplaybreaks \begin{eqnarray*}
	x^2+2xy-25+y^2&=&\left(x^2+2xy+y^2\right)-25\\
	&=&\left(x+y\right)^2-5^2\\
	&=&(x+y-5)(x+y+5).
\end{eqnarray*}}
\item Ta có
{\allowdisplaybreaks \begin{eqnarray*}
	x^2y^2-x^2+8x-16&=&x^2y^2-\left(x^2-8x+16\right)\\
	&=&\left(xy\right)^2-\left(x-4\right)^2\\
	&=&(xy-x+4)(xy+x-4).
\end{eqnarray*}}
\item Ta có
{\allowdisplaybreaks \begin{eqnarray*}
	x^2-6xy+9y^2+4x-12y&=&\left(x^2-6xy+9y^2\right)+(4x-12y)\\
	&=&\left(x-3y\right)^2+4(x-3y)\\
	&=&(x-3y)(x-3y+4).
\end{eqnarray*}}
\item Ta có
{\allowdisplaybreaks \begin{eqnarray*}
	x^2-xy+x^3-3x^2y+3xy^2-y^3&=&\left(x^2-xy\right)+\left(x^3-3x^2y+3xy^2-y^3\right)\\
	&=&x(x-y)+\left(x-y\right)^3\\
	&=&(x-y)\left[x+\left(x-y\right)^2\right]\\
	&=&(x-y)\left(x+x^2-2xy+y^2\right).
\end{eqnarray*}}
\end{enumerate}
}
\end{vd}

\begin{vd}
Tính nhanh
\begin{listEX}[2]
\item $45^2+33^2-22^2+90\cdot 33$;
\item $111^2-137^2-48^2+96\cdot 137$.
\end{listEX}
\loigiai{
\begin{enumerate}
\item 
{\allowdisplaybreaks \begin{eqnarray*}
45^2+33^2-22^2+90\cdot 33&=&\left(45^2+2\cdot 45\cdot 33+33^2\right)-22^2\\
&=&(45+33)^2-22^2=78^2-22^2=100\cdot 56=5600.
\end{eqnarray*}}
\item 
{\allowdisplaybreaks \begin{eqnarray*}
111^2-137^2-48^2+96\cdot 137&=&111^2-\left(137^2-2\cdot 48\cdot 137+48^2\right)\\
&=&111^2-89^2=4400.
\end{eqnarray*}}
\end{enumerate}
}
\end{vd}

\begin{vd}
	Tính giá trị của biểu thức $A=2^{10}-2^9-2^8-2^7-\cdots -2^2-2-1$.	
	\loigiai{
		{\allowdisplaybreaks \begin{eqnarray*}	
			A&=&\left(2^{10}-2^9\right)-2^8-2^7-\cdots-2^2-2-1\\
			&=&2^9(2-1)-2^8-2^7-\cdots-2^2-2-1\\
			&=&2^9-2^8-2^7-\cdots-2^2-2-1\\
			&=&\left(2^9-2^8\right)-2^7-\cdots-2^2-2-1\\
			&=&2^8(2-1)-2^7-\cdots-2^2-2-1\\
			&=&2^8-2^7-\cdots-2^2-2-1\\
			&=&\cdots \\
			&=&2(2-1)-1=2-1=1.
		\end{eqnarray*}}	
	}
\end{vd}



\begin{dang}[Tìm $x$]
\begin{itemize}
\item Chuyển tất cả các số hạng về vế trái, vế phải bằng $0$.
\item Phân tích vế trái thành nhân tử, đưa về dạng $A\cdot B=0$.
\item Áp dụng $A\cdot B=0$ suy ra $A=0$ hoặc $B=0$.
\end{itemize}

\begin{vd}
Tìm $x$ biết
\begin{listEX}[2]
\item $x^2+3x-(2x+6)=0$;
\item $5x+20-x^2-4x=0$;
\item $3x^2-3x+2x^3-2x^2=0$;
\item $x^3+27=-x^2+9$.
\end{listEX}
\loigiai{
\begin{enumerate}
	\item Ta có
	{\allowdisplaybreaks \begin{eqnarray*}
	x^2+3x-(2x+6)&=&0\\
	x(x+3)-2(x+3)&=&0\\
	(x+3)(x-2)&=&0
	\end{eqnarray*}}Suy ra $ x+3=0 $ hoặc $ x-2=0 $.\\
	Do đó $ x=-3; x=2 $.
	\item Ta có
	{\allowdisplaybreaks \begin{eqnarray*}
		5x+20-x^2-4x&=&0\\
		5(x+4)-x(x+4)&=&0\\
		(x+4)(5-x)&=&0
	\end{eqnarray*}}Suy ra $ x+4=0 $ hoặc $ 5-x=0 $.\\
	Do đó $ x=-4; x=5 $.
	\item Ta có
	{\allowdisplaybreaks \begin{eqnarray*}
		3x^2-3x+2x^3-2x^2&=&0\\
		3x(x-1)+2x^2(x-1)&=&0\\
		x(x-1)(3+2x)&=&0
	\end{eqnarray*}}Suy ra $ x=0 $ hoặc $ x-1=0 $ hoặc $ 3+2x=0 $.\\
	Do đó $ x=0; x=1; x=-1,5 $.
	\item Ta có
	{\allowdisplaybreaks \begin{eqnarray*}
		x^3+27&=&-x^2+9\\
		x^3+27+x^2-9&=&0\\
		(x+3)\left(x^2-3x+9\right)+(x-3)(x+3)&=&0\\
		(x+3)\left(x^2-2x+6\right)&=&0.
	\end{eqnarray*}}Ta thấy $ x^2-2x+6=(x-1)^2+5>0 $.\\
	Do đó $ x+3=0 \Rightarrow x=-3 $.
\end{enumerate}
}
\end{vd}

\begin{vd}
Tìm $x$ biết
\begin{listEX}[2]
	\item $x^3-5x^2+x-5=0$;
	\item $x^4-2x^3+10x^2-20x=0$.
\end{listEX}	
	\loigiai{
\begin{enumerate}
	\item 
	{\allowdisplaybreaks \begin{eqnarray*}
		x^3-5x^2+x-5&=&0\\
		\left(x^3-5x^2\right)+(x-5)&=&0\\
		x^2(x-5)+(x-5)&=&0\\
		(x-5)\left(x^2+1\right)&=&0
	\end{eqnarray*}}
	Vì $x^2+1>0$ với mọi $x$ nên $x-5=0$.\\
	Vậy $x=5$.
		\item 
	{\allowdisplaybreaks \begin{eqnarray*}
		x^4-2x^3+10x^2-20x&=&0\\
		\left(x^4-2x^3\right)+\left(10x^2-20x\right)&=&0\\
		x^3(x-2)+10x(x-2)&=&0\\
		x(x-2)\left(x^2+10\right)&=&0
	\end{eqnarray*}}Vì $x^2+10>0$ với mọi $x$ nên $x=0$ hoặc $x-2=0$.\\
	Vậy $x=0$; $x=2$.
\end{enumerate}			
	}
\end{vd}
\begin{vd}[Trường THCS\&THPT Nguyễn Tất Thành, Năm học 2012-2013]
Tìm $x$ biết $5(x+4)-x^2-4x=0$.	
	\loigiai{
{\allowdisplaybreaks \begin{eqnarray*}
5(x+4)-x^2-4x&=&0\\
5(x+4)-\left(x^2+4x\right)&=&0\\
5(x+4)-x(x+4)&=&0\\
(x+4)(5-x)&=&0
\end{eqnarray*}}Do đó $x+4=0$ hoặc $5-x=0$.\\
Vậy $x=-4$; $x=5$.
	}
\end{vd}

\begin{dang}[Chứng minh giá trị của biểu thức $A$ chia hết cho số $k$]
Dùng phương pháp đặt nhân tử chung để phân tích biểu thức đã cho thành nhân tử: $A=k \cdot B$ ($k \ne 0$). Khi đó $A \;\vdots \; k$.
\end{dang}
\begin{note}
Kiến thức bổ sung:
{\allowdisplaybreaks \begin{eqnarray*}
&&a^n-b^n=(a-b)\left(a^{n-1}+a^{n-2}b+a^{n-3}b^2+\cdots+a^2b^{n-3}+ab^{n-2}+b^{n-1}\right).\\
&&a^n+b^n=(a+b)\left(a^{n-1}-a^{n-2}b+a^{n-3}b^2-\cdots+a^2b^{n-3}-ab^{n-2}+b^{n-1}\right) \;\text{với}\; n\;\text{lẻ}.	
\end{eqnarray*}}
\end{note}


\begin{vd}
Chứng minh rằng $ 49^n+77^n-29^n-1 $ chia hết cho $ 48 $.	
	
	\loigiai
	{
Ta có
{\allowdisplaybreaks \begin{eqnarray*}
49^n+77^n-29^n-1&=&\left(77^n-29^n\right)+\left(49^n-1\right)\\
&=&(77-29)\cdot B_1+(49-1)\cdot B_2\\
&=&48\cdot B_1+48\cdot B_2\\
&=&48(B_1+B_2)\;\vdots \;48.
\end{eqnarray*}}		
	}
\end{vd}


\begin{vd}
Chứng minh rằng
\[A=2^0+2^1+2^2+2^3+2^4+2^5+2^6+\cdots+2^{97}+2^{98}+2^{99} \;\text{chia hết cho}\;31.\]	
	\loigiai{
$A=\left(2^0+2^1+2^2+2^3+2^4\right)+\left(2^5+2^6+2^7+2^8+2^9\right)+\cdots+\left(2^{95}+2^{96}+2^{97}+2^{98}+2^{99}\right)$	\\
$A=(1+2+4+8+16)+2^5(1+2+4+8+16)+\cdots+2^{95}(1+2+4+8+16)$	\\
$A=31\cdot \left(1+2^5+\ldots+2^{95}\right) \;\vdots\;31$.
	}
\end{vd}

\begin{vn}
Phân tích đa thức sau thành nhân tử
\begin{listEX}[2]
	\item $ 3x^3-x^2-21x+7 $;
	\item $ x^3-4x^2+8x-8 $;
	\item $ x^3-5x^2-5x+1 $;
	\item $ x^2y-xz+z-y $;
	\item $ x^4-x^3+x^2-1 $;
	\item $ x^4-x^2+10x-25 $.
\end{listEX}
\loigiai{
\begin{enumerate}
\item Ta có
{\allowdisplaybreaks \begin{eqnarray*}
	3x^3-x^2-21x+7&=&\left(3x^3-x^2\right)-(21x-7)\\
	&=&x^2(3x-1)-7(3x-1)\\
	&=&(3x-1)\left(x^2-7\right).
\end{eqnarray*}}Vậy $3x^3-x^2-21x+7=(3x-1)\left(x^2-7\right)$.
\item Ta có
{\allowdisplaybreaks \begin{eqnarray*}
	x^3-4x^2+8x-8&=&\left(x^3-8\right)-\left(4x^2-8x\right)\\
	&=&(x-2)\left(x^2+2x+4\right)-4x(x-2)\\
	&=&(x-2)\left(x^2-2x+4\right).
\end{eqnarray*}}Vậy $x^3-4x^2+8x-8=(x-2)\left(x^2-2x+4\right)$.
\item Ta có
{\allowdisplaybreaks \begin{eqnarray*}
	x^3-5x^2-5x+1&=&\left(x^3+1\right)-\left(5x^2+5x\right)\\
	&=&(x+1)\left(x^2-x+1\right)-5x(x+1)\\
	&=&(x+1)\left(x^2-6x+1\right).
\end{eqnarray*}}Vậy $x^3-5x^2-5x+1=(x+1)\left(x^2-6x+1\right)$.
\item Ta có
{\allowdisplaybreaks \begin{eqnarray*}
	x^2y-xz+z-y&=&\left(x^2y-y\right)-(xz-z)\\
	&=&y\left(x^2-1\right)-z(x-1)\\
	&=&y(x+1)(x-1)-z(x-1)\\
	&=&(x-1)(xy+y-z).
\end{eqnarray*}}Vậy $	x^2y-xz+z-y=(x-1)(xy+y-z)$.
\item Ta có
{\allowdisplaybreaks \begin{eqnarray*}
	x^4-x^3+x^2-1&=&\left(x^4-x^3\right)+\left(x^2-1\right)\\
	&=&x^3(x-1)+(x-1)(x+1)\\
	&=&(x-1)\left(x^3+x+1\right).
\end{eqnarray*}}Vậy $x^4-x^3+x^2-1=(x-1)\left(x^3+x+1\right)$.
\item Ta có
{\allowdisplaybreaks \begin{eqnarray*}
	x^4-x^2+10x-25&=&\left(x^2\right)^2-\left(x^2-10x+25\right)\\
	&=&\left(x^2\right)^2-(x-5)^2\\
	&=&\left(x^2-x+5\right)\left(x^2+x-5\right).
\end{eqnarray*}}Vậy $x^4-x^2+10x-25=\left(x^2-x+5\right)\left(x^2+x-5\right)$.

\end{enumerate}
}
\end{vn}

\begin{vn}Phân tích đa thức sau thành nhân tử:
\begin{listEX}[2]
\item $ax+ay-3x-3y$;  	
\item $xy+1-x-y$; 
\item $x^3-2x^2+2x-4$;  	
\item $x^2+ab+ax+bx$;  
\item $xy-4+2x-2y$;  	
\item $ax+bx-cx+a+b-c$; 
\item  $x^2-4xy+4y^2-4$;  	
\item  $16-x^2+2xy-y^2$. 
\end{listEX}
\loigiai{ 
\begin{enumerate}
 \item $ ax+ay-3x-3y=(ax+ay)-(3x+3y)=a(x+y)-3(x+y)=(x+y)(a-3) $.
 \item $xy+1-x-y=(xy-x)+(1-y)=x(y-1)-(y-1)=(y-1)(x-1)$.
 \item $x^3-2x^2+2x-4=(x^3-2x^2)+(2x-4)=x^2(x-2)+2(x-2)=(x-2)\left(x^2+2\right)$.
 \item $x^2+ab+ax+bx=(x^2+ax)+(ab+bx)=x(x+a)+b(a+x)=(x+a)(x+b)$.
 \item $xy-4+2x-2y=(xy+2x)-(4+2y)=x(y+2)-2(y+2)=(y+2)(x-2)$.
 \item $ax+bx-cx+a+b-c=(ax+bx-cx)+(a+b-c)=x(a+b-c)+(a+b-c)=(a+b-c)(x+1)$.
 \item $x^2-4xy+4y^2-4=\left(x^2- 4xy+4y^2\right)-4=(x-2y)^2-2^2=(x-2y-2)(x+2y+2)$.
 \item $16-x^2+2xy-y^2=16-\left(x^2-2xy+y^2\right)=4^2-(x-y)^2=(4-x+y)(4+x-y)$.
\end{enumerate}}
\end{vn}


\begin{vn}

	\begin{listEX}[2]
		\item $4x^2 -y^2+8(y-2)$;
		\item $x^2+2x-4y^2-4y$;  	
		\item $x^4-6x^3+54x-81$;  
		\item $ax^2+ax-bx^2-bx-a+b$.  	
	\end{listEX}
	\loigiai{
		\begin{enumerate}

				&=&(2x)^2-(y-4)^2=(2x-y+4)(2x+y-4).
			\end{eqnarray*}}

\begin{vn}
	Tìm $x$ biết
	\begin{listEX}[2]
		\item $x^5+x^4+x+1=0$;
		\item $x^4+3x^3-x-3=0$;
		\item $ 3x^3-7x^2+6x-14=0 $;
		\item $ 6x^3+16x^2-150x-400=0 $;
		\item $ 5(x+4)-x^2-4x=0 $;
		\item $ 5x(x-3)-x+3=0 $.
	\end{listEX}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			{\allowdisplaybreaks \begin{eqnarray*}
				x^5+x^4+x+1&=&0\\
				x^4(x+1)+(x+1)&=&0\\
				(x+1)\left(x^4+1\right)&=&0.
			\end{eqnarray*}}Ta thấy $ x^4+1>0 $ với mọi $ x $.\\
		Do đó $ x+1=0 \Rightarrow x=-1 $.
			\item Ta có
			{\allowdisplaybreaks \begin{eqnarray*}
				x^4+3x^3-x-3&=&0\\
				x^3(x+3)-(x+3)&=&0\\
				(x+3)\left(x^3-1\right)&=&0.
			\end{eqnarray*}}Suy ra $ x+3=0 $ hoặc $ x^3-1=0 $.\\
			Do đó $ x=-3; x=1 $.
			\item Ta có
			{\allowdisplaybreaks \begin{eqnarray*}
				3x^3-7x^2+6x-14&=&0\\
				x^2(3x-7)+2(3x-7)&=&0\\
				(3x-7)\left(x^2+2\right)&=&0.
			\end{eqnarray*}}Ta thấy $ x^2+2>0 $ với mọi $ x $.\\
		Do đó $ 3x-7=0 \Rightarrow x=\dfrac{7}{3} $.
			\item Ta có
			{\allowdisplaybreaks \begin{eqnarray*}
				6x^3+16x^2-150x-400&=&0\\
		        2x^2(3x+8)-50(3x+8)&=&0\\
				2(3x+8)\left(x^2-25\right)&=&0\\
				2(3x+8)(x-5)(x+5)&=&0.
			\end{eqnarray*}}Suy ra $ 3x+8=0 $ hoặc $ x-5=0 $ hoặc $ x+5=0 $.\\
			Do đó $ x=-\dfrac{8}{3}; x=5; x=-5 $.
			\item Ta có
			{\allowdisplaybreaks \begin{eqnarray*}
				5(x+4)-x^2-4x&=&0\\
				5(x+4)-\left(x^2+4x\right)&=&0\\
				5(x+4)-x(x+4)&=&0\\
				(x+4)(5-x)&=&0.
			\end{eqnarray*}}Suy ra $ x+4=0 $ hoặc $ 5-x=0 $.\\
			Do đó $ x=-4; x=5 $.
			\item Ta có
			{\allowdisplaybreaks \begin{eqnarray*}
				5x(x-3)-x+3&=&0\\
				5x(x-3)-(x-3)&=&0\\
				(x-3)(5x-1)&=&0.
			\end{eqnarray*}}Suy ra $ x-3=0 $ hoặc $ 5x-1=0 $.\\
			Do đó $ x=3; x=\dfrac{1}{5} $.
		\end{enumerate}
	}
\end{vn}

\begin{vn}
