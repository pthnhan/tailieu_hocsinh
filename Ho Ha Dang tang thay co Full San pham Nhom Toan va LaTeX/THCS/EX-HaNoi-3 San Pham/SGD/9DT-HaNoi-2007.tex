\begin{ex}%[Thi vào 10, Sở GD&ĐT Hà Nội, 2007 - 2008]%[0D1B7]%[0D1K7]%[Đào Anh Dương]  
	Cho biểu thức  $P=\dfrac{\sqrt{x}}{\sqrt{x}-1}+\dfrac{3}{\sqrt{x}+1}-\dfrac{6\sqrt{x}-4}{x-1}$.
	\begin{enumerate}
		\item 	Rút gọn $P$.
		\item	Tìm các giá trị của $x$ để $P <\dfrac{1}{2}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
		\item Điều kiện $0 \le x \ne 1$.
		\begin{align*}
		P&=\dfrac{\sqrt{x}}{\sqrt{x}-1}+\dfrac{3}{\sqrt{x}+1}-\dfrac{6\sqrt{x}-4}{x-1}\\
		&=\dfrac{\sqrt{x}(\sqrt{x}+1) + 3(\sqrt{x}-1) - 6\sqrt{x} + 4}{x-1}\\
		&=\dfrac{x - 2\sqrt{x} + 1}{x-1}\\
		&=\dfrac{\sqrt{x}-1}{\sqrt{x}+1}.
		\end{align*}	
		\item	Để $P = \dfrac{\sqrt{x}-1}{\sqrt{x}+1} < \dfrac{1}{2} \Leftrightarrow 2(\sqrt{x}-1) \le \sqrt{x}+1$
		$\Leftrightarrow \sqrt{x} \le 3 \Leftrightarrow x \le 9$.\\
		Kết hợp điều kiện ta được $0 \le x < 9$ và $x \ne 1$.
	\end{enumerate}
	}
\end{ex} 

\begin{ex}%[Thi vào 10, Sở GD&ĐT Hà Nội, 2007 - 2008]%[0D4B8]%[Đào Anh Dương]   
	\textit{Giải bài toán sau bằng cách lập phương trình}\\
	Một người đi xe đạp từ $A$ đến $B$ cách nhau $24$ km. Khi từ $B$ trở về $A$ người đó tăng vận tốc thêm $4$ km/h so với lúc đi, vì vậy thời gian về ít hơn thời gian đi $30$ phút. Tính vận tốc của người đi xe đạp khi đi từ $A$ đến $B$.
	\loigiai{
	Gọi vận tốc lúc đi là $x$ (km/h), $x>0$.\\
	Khi đó, vận tốc lúc về là $x+4$ (km/h).\\
	Theo đề bài ta có phương trình $\dfrac{24}{x} - \dfrac{24}{x+4} = \dfrac{1}{2}$.\\
	Phương trình tương đương với $x^2 + 4x - 192 = 0$.\\
	Giải ra ta được $x=12$ và $x=-16 \, \text{(loại)}$.\\
	Vậy, vận tốc người đi xe đạp khi đi từ $A$ đến $B$ là $12$ km/h.	
	}
\end{ex} 
 
\begin{ex}%[Thi vào 10, Sở GD&ĐT Hà Nội, 2007 - 2008]%[0D4Y5]%[0D4K6]%[Đào Anh Dương]   
	Cho phương trình $x^2 +bx+c=0$.
	\begin{enumerate}
		\item Giải phương trình khi $b=-3$, $c=2$. 
		\item Tìm $b$, $c$ để phương trình có hai nghệm phân biệt và tích bằng $1$.
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Khi $b=-3$, $c=2$ ta có tổng các hệ số $a+b+c = 0$ nên phương trình có hai nghiệm $x_1 = 1$, $x_2 = 2$.
		\item Phương trình có hai nghệm phân biệt và tích bằng $1$ khi\\
		$\heva{&\Delta > 0 \\&P = \dfrac{c}{a} = 1} \Leftrightarrow \heva{&b^2 - 4c > 0 \\&c = 1} \Leftrightarrow \heva{&b>2 \,\, \text{hoặc} \,\, b<-2  \\&c=1}$.
	\end{enumerate}
	}
	
\end{ex}  

\begin{ex}%[Thi vào 10, Sở GD&ĐT Hà Nội, 2007 - 2008]%[0H3B4]%[0H3K7]%[Đào Anh Dương]    
	Cho đường tròn $(O;R)$ tiếp xúc với đường thẳng $d$ tại $A$. Trên đường thẳng $d$ lấy điểm $H$ ($H$ khác $A$) và $AH<R$. Qua $H$ kẻ đường thẳng vuông góc với $d$ cắt đường tròn tại hai điểm phân biệt $E, B$ ($E$ nằm giữa $B$ và $H$).
	\begin{enumerate}
		\item 	Chứng minh $\widehat {ABE} = \widehat  {EAH}$ và $\triangle ABH \backsim \triangle EAH$.
		\item 	Lấy điểm $C$ trên đường thẳng $d$ sao cho $H$ là trung điểm của $AC$, đường thẳng $CE$ cắt $AB$ tại $K$. 	Chứng minh tứ giác $AHEK$ nội tiếp.
		\item	Xác định vị trí của điểm $H$ để $AB = R\sqrt{3}$.
	\end{enumerate}
	\loigiai{
	\begin{center}
		\begin{tikzpicture}[scale=1]
			\tkzInit[xmin=-6.5,xmax=3.2,ymin=-3.7,ymax=3.2]
			\tkzClip
			\tkzDefPoints{0/0/O,0/-3/A}
			\tkzDrawCircle(O,A)
			\tkzDefLine[perpendicular=through A,K=2](A,O)
			\tkzGetPoint{d}
			\tkzDrawLine[add=0.5 and 0](A,d)
			\coordinate (H) at ($(A)!0.4!(d)$);
			\tkzDefLine[perpendicular=through H,K=1.3](d,A)
			\tkzGetPoint{v}
			\tkzInterLC(H,v)(O,A)
			\tkzGetPoints{E}{B}
			\coordinate (C) at ($(A)!2!(H)$);
			\tkzInterLL(C,E)(A,B)
			\tkzGetPoint{K}
			\coordinate (M) at ($(E)!0.5!(B)$);
			\tkzMarkRightAngles[size=0.25,fill=gray!50](O,A,C O,M,B A,H,B)
			\tkzMarkSegments[mark=||](H,C H,A)
			\tkzDrawSegments(B,H C,K O,A O,B O,E A,B O,M)
			\tkzDrawPoints(O,A,H,E,B,C,K,M)
			\tkzLabelPoints[above](B)
			\tkzLabelPoints[below](d,C,H,A)
			\tkzLabelPoints[left](E,M)
			\tkzLabelPoints[right](O,K)
		\end{tikzpicture}
	\end{center}
	\begin{enumerate}
		\item 	Ta có $\widehat {ABE} = \widehat  {EAH}$ (cùng chắn cung $AE$).\\
		$\triangle ABH$ và $\triangle EAH$ là hai tam giác vuông góc có $\widehat {ABE} = \widehat  {EAH}$ nên $\triangle ABH \backsim \triangle EAH$.
		\item 	Vì $H$ là trung điểm của $AC$ và $EH \perp AC$ nên $\triangle AEC$ cân tại $E$.\\
		Suy ra 
		$\widehat{ECA} = \widehat{EAC} = \widehat{ABH}.$\\
		Mà $\widehat{ABH} + \widehat{BAH} = 90^\circ$.\\
		$\Rightarrow \widehat{ECA} + \widehat{BAH} = 90^\circ$.\\
		$\Rightarrow \widehat{EKA} = 90^\circ$.\\
		Tứ giác $AHEK$ có $\widehat{EHA} + \widehat{EKA} = 180^\circ$ nên là tứ giác nội tiếp.
		\item	Gọi $M$ là trung điểm của $EB$ thì $OM \perp EB$ và $OM = AH$.\\
		Ta có $AB = R\sqrt{3} \Rightarrow \widehat{AOB} = 120^\circ \Rightarrow \widehat{BOM} = 30^\circ \Rightarrow \triangle OBE$ đều cạnh $R$.\\
		Vậy $OM = AH = \dfrac{R\sqrt{3}}{2}$. 
	\end{enumerate}
	}
\end{ex} 
 
\begin{ex}%[Thi vào 10, Sở GD&ĐT Hà Nội, 2007 - 2008]%[0D2K5]%[Đào Anh Dương]   
	Cho đường thẳng $y = (m-1)x+2$. Tìm $m$ để khoảng cách từ gốc toạ độ $O$ tới đường thẳng đó lớn nhất.
	\loigiai{
	\begin{center}
		\begin{tikzpicture}[>=stealth,scale=1,x=1cm,y=1cm]
			\draw[->](-3,0)--(0,0) node[below right]{$O$} -- (5,0) node[below]{$x$};
			\draw[->](0,-2)--(0,4) node[right]{$y$};
			\tkzDefPoints{0/2/A,3.5/0/B}
			\tkzDefPointBy[projection = onto A--B](O)
			\tkzGetPoint{H}
			\tkzDrawLine[add = 0.7 and 0.4](A,B)
			\tkzMarkRightAngles[size=0.3,fill=gray!50](O,H,B)
			\tkzDrawSegments(O,H)
			\tkzDrawPoints(O,A,B,H)
			\tkzLabelPoints[above](B,H)
			\tkzLabelPoints[above right](A)
			\draw(A) node[below left]{$2$};
		\end{tikzpicture}
	\end{center}
	Dễ thấy $A(0;2)$ là điểm cố định của đường thẳng. Gọi $B$ là giao điểm của đường thẳng với trục hoành. Trong tam giác vuông $OAB$ kẻ $OH \perp AB$, $H \in AB$ thì $OH$ chính là khoảng cách từ gốc tọa độ $O$ tới đường thẳng.\\
	Vì $OH$, $OA$ lần lượt là đường vuông góc và đường xiên kẻ từ $O$ đến $AB$ nên $OH \le OA$.\\
	Do đó, khoảng cách từ $O$ đến đường thẳng lớn nhất khi $H$ trùng với $A$, nghĩa là đường thẳng đi qua $A$ và song song với trục hoành.\\
	Suy ra $m-1 = 0 \Leftrightarrow m = 1$.
	}
\end{ex}
