	\item \textbf{Tính chất phân giác của một góc}

			\Rightarrow 2\widehat{MOA} & =&2\widehat{MOB}=\widehat{AOB}\\

		\end{eqnarray*}

		\item Biết các góc $ AOx $ và $ BOx $ có số đo bằng $ 30^\circ $. Chứng tỏ rằng tia $ Ox $ là tia phân giác của góc $ AOB $.

\subsection{VÍ DỤ MINH HỌA}

\begin{vd}
	Cho hai góc $AOx$ và $BOx$ \textit{không kề nhau}
	\begin{enumerate}
		\item Vẽ hình biết số đo góc $AOx=38^\circ$ và góc $BOx=112^\circ$. Trong ba tia $OA,OB,Ox$, tia nào nằm giữa hai tia còn lại? Vì sao?
		\item Tính góc $AOB$.
		\item Vẽ tia phân giác $OM$ của góc $AOB$. Tính góc $MOx$.
		\item Cho $\widehat{AOx}=\alpha$ và $\widehat{BOx}=\beta$, trong đó $0^\circ<\alpha + \beta < 180^\circ$ và $\alpha \neq \beta$. Tìm điều kiện giữa $\alpha$ và $\beta$ để tia $OA$ nằm giữa hai tia $OB$ và $Ox$. Tính số đo góc $MOx$ theo $\alpha$ và $\beta$. 
	\end{enumerate}  
	\loigiai
	{
		\begin{enumerate}
			\item   \immini
			{
				Do hai góc $AOx$ và $BOx$ là hai góc \textit{không} kề nhau mà có chung                 cạnh $Ox$ nên hai tia $OA$ và $OB$ cùng nằm trong một nửa mặt phẳng bờ chứa tia $Ox$ (h.55).\\
				Mặt khác $\widehat{AOx}<\widehat{BOx}$ (vì $38^\circ < 112^\circ$) nên tia $OA$ nằm giữa hai tia $OB$ và $Ox$.
				\item Do tia $OA$ nằm giữa hai tia $OB$ và $Ox$ nên ta có\\ $\widehat{AOx}+\widehat{AOB}=\widehat{BOx}$.\\
				$\Rightarrow 38^\circ +\widehat{AOB}=112^\circ \Rightarrow \widehat{AOB}=74^\circ$.
			}
			{
				\begin{tikzpicture}[line join = round, line cap = round,>=stealth,thick,font=\footnotesize,scale=0.9]
				\clip (-2,-1.5-0.1) rectangle (3+0.1,3+0.1);

				\tkzDefPointBy[rotation = center O angle 38](x)    \tkzGetPoint{A}
				\tkzDefPointBy[rotation = center O angle 112](x)    \tkzGetPoint{B}
