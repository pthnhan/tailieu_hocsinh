\begin{ex}%[Đề thi thử 2018 lần 1, THPT Thanh Miện 1 – Hải Dương]%[Phạm Văn Đức, 12EX-6]%[Tuấn Nguyễn]%[1D2B3-2]
	Hệ số của $x^{10}$ trong khai triển của biểu thức $(2x-3x^2)^5$ bằng
	\choice
	{$357$}
	{$243$}
	{$628$}
	{\True $-243$}
	\loigiai{
	
	Số hạng tổng quát trong  khai triển của biểu thức $(2x-3x^2)^5$ là $$T=\mathrm{C}_5^k\cdot (2x)^{5-k}\cdot (-3x^2)^k=(-3)^k\cdot 2^{5-k}\cdot \mathrm{C}_5^k\cdot x^{5+k}.$$

	}

\begin{ex}%[Đề thi thử 2018 lần 1, THPT Thanh Miện 1 – Hải Dương]%[Phạm Văn Đức, 12EX-6]%[Tuấn Nguyễn]%[2D2B6-1]
	Bất phương trình $\log_2\dfrac{x^2-6x+8}{4x-1}\ge 0$ có tập nghiệm là $T=\left(\dfrac{1}{4};a \right]\cup [b;+\infty)$. Tính $M=a+b$.
	\choice
	{$M=12$}
	{$M=8$}
	{$M=9$}
	{\True $M=10$}
	\loigiai{
	Điều kiện xác định: $\dfrac{x^2-6x+8}{4x-1}> 0 \quad(*).$
	Ta có 
	\begin{align*}
		\log_2\dfrac{x^2-6x+8}{4x-1}\ge 0&\Leftrightarrow \dfrac{x^2-6x+8}{4x-1}\ge 1 \;(\text{thỏa mãn điều kiện } (*))\\
		&\Leftrightarrow \dfrac{x^2-10x+9}{4x-1}\ge 0\\
		&\Leftrightarrow x\in \left(\dfrac{1}{4};1 \right]\cup [9;+\infty).
	\end{align*}
	Vậy $T=a+b=10$.
	}
\end{ex}

Dấu bằng đạt được khi $\heva{& B=C \\  & \sin A=1 \\ & \sin (\dfrac{A}{2}+\dfrac{\pi}{4})=1}$ $\Leftrightarrow \heva{& A=\dfrac{\pi}{2} \\  & B=C=\dfrac{\pi}{4}}$ $\Rightarrow \heva{& x=1 \\  & y=\sqrt{2}-1 \\ & z=\sqrt{2}+1.}$}
\end{ex}
