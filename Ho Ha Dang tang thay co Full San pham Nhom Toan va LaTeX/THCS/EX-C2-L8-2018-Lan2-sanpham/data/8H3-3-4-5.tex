\section{Tính chất đường phân giác của tam giác}
\subsection{Trọng tâm kiến thức}
\begin{dl}
	\immini
	{
		Trong tam giác, đường phân giác của một góc trong chia cạnh đối diện thành hai đoạn thẳng tỉ lệ với hai cạnh kề hai đoạn ấy.
		\begin{equation*}
		\left.\begin{aligned}
		& \triangle ABC \\ & \widehat{BAD} = \widehat{CAD}
		\end{aligned}\right\}
		\Rightarrow \dfrac{DB}{DC} = \dfrac{AB}{AC}.
		\end{equation*}
	}
	{
		\begin{tikzpicture}[line join = round, line cap = round,>=stealth,font=\footnotesize,scale=.7]
		\tkzDefPoints{0/0/B}
		\coordinate (C) at ($(B)+(6,0)$);
		\tkzDefShiftPoint[B](70:4){A}
		\tkzDefLine[bisector](B,A,C)    \tkzGetPoint{d}
		\tkzInterLL(A,d)(B,C)    \tkzGetPoint{D}
		%
		\pgfresetboundingbox
		\tkzMarkAngles[size=.7cm,arc=l,mark=|](B,A,D)
		\tkzMarkAngles[size=.9cm,arc=l,mark=|](D,A,C)		
		\tkzDrawSegments(A,B B,C C,A A,D)
		\tkzDrawPoints[fill=black](A,B,C,D)
		\tkzLabelPoints[above](A)
		\tkzLabelPoints[below](D)
		\tkzLabelPoints[left](B)
		\tkzLabelPoints[right](C)
		\end{tikzpicture}
	}
\end{dl}
\begin{note}
	Định lý vẫn đúng với đường phân giác góc ngoài của tam giác.
	\immini
	{
		\begin{equation*}
		\left.\begin{aligned} &\triangle ABC~(AB \ne AC) \\ & \widehat{BAE} = \widehat{C'AE} \end{aligned}\right\} \Rightarrow \dfrac{EB}{EC} = \dfrac{AB}{AC}.
		\end{equation*}
	}
	{
		\begin{tikzpicture}[line join = round, line cap = round,>=stealth,font=\footnotesize,scale=.6]
		\tkzDefPoints{0/0/B}
		\coordinate (C) at ($(B)+(5,0)$);
		\tkzDefShiftPoint[B](100:4){A}
		\coordinate (C') at ($(C)!1.3!(A)$);
		\tkzDefLine[bisector](C',A,B)    \tkzGetPoint{e}
		\tkzInterLL(A,e)(B,C)    \tkzGetPoint{E}
		\tkzMarkAngles[size=.5cm,arc=l,mark=|](C',A,E)
		\tkzMarkAngles[size=.7cm,arc=l,mark=|](E,A,B)
		%
		\pgfresetboundingbox
		\tkzDrawSegments(A,E E,C C,C' A,B)
		\tkzDrawPoints[fill=black](A,B,C,E)
		\tkzLabelPoints[above](A,C')
		\tkzLabelPoints[below](E,B,C)
		\end{tikzpicture}
	}
	Các định lý trên có định lý đảo
	\begin{align*}
		&\dfrac{DB}{DC} = \dfrac{AB}{AC} \Rightarrow AD \text{ là đường phân giác trong của tam giác.}\\
		&\dfrac{EB}{EC} = \dfrac{AB}{AC} \Rightarrow AE \text{ là đường phân giác ngoài của tam giác.}
	\end{align*}
\end{note}

\subsection{Các dạng bài tập và phương pháp giải}

\begin{vd}%[8H3B3]
	Cho $\triangle ABC$ có $AD$ là đường phân giác. Trên $AB$ lấy điểm $M$, trên $AC$ lấy điểm $N$ sao cho $BM = BD$, $CN = CD$. Biết $AB = 7$ cm, $AC = 8$ cm, $BC = 12$ cm. Tính chu vi $\triangle AMN$.
	\loigiai
	{
		\immini
		{
			Theo tính chất đường phân giác ta có $\dfrac{BD}{CD} = \dfrac{AB}{AC}$
			 nên $\dfrac{BD}{BC} = \dfrac{AB}{AB + AC}$ hay $\dfrac{BD}{12} = \dfrac{7}{7+8}$. 
			 Suy ra $BD = \dfrac{7 \cdot 12}{15} = 5{,}6$ (cm).\\
			Do đó $CD = 12 - 5{,}6 = 6{,}4$ (cm).\\
			Suy ra $BM = 5{,}6$ (cm), $CN = 6{,}4$ (cm) nên $AM = 7 - 5{,}6 = 1{,}4$ (cm), $AN = 8 - 6{,}4 = 1{,}6$ (cm).
		}
		{
			\begin{tikzpicture}[line join = round, line cap = round,>=stealth,font=\footnotesize,scale=.7]
			\tkzDefPoints{0/0/B}
			\coordinate (C) at ($(B)+(6,0)$);

			\tkzDefLine[bisector](B,A,C)    \tkzGetPoint{d}
			\tkzInterLL(A,d)(B,C)    \tkzGetPoint{D}
			\tkzDuplicateLen(B,D)(B,A)    \tkzGetPoint{M}
			\tkzDuplicateLen(C,D)(C,A)    \tkzGetPoint{N}
			%
			\pgfresetboundingbox
			\tkzMarkAngles[size=.7cm,arc=l,mark=|](B,A,D)
			\tkzMarkAngles[size=.9cm,arc=l,mark=|](D,A,C)

			\tkzLabelPoints[below](D)
			\tkzLabelPoints[left](B,M)
			\tkzLabelPoints[right](C,N)
			\end{tikzpicture}
		}
		Từ $\dfrac{BD}{CD} = \dfrac{AB}{AC}$ suy ra $\dfrac{BD}{AB} = \dfrac{CD}{AC}$ dẫn tới $\dfrac{BM}{AB} = \dfrac{CN}{AC}$ (do $BM = BD$, $CN = CD$), suy ra $MN \parallel BC$.\\
		Theo hệ quả định lý Ta-lét ta có $\dfrac{MN}{BC} = \dfrac{AM}{AB}$ hay $\dfrac{MN}{12} = \dfrac{1{,}4}{7} \Rightarrow MN = 2{,}4$ (cm).\\
		Vậy chu vi $\triangle AMN$ là $1{,}4 + 1{,}6 + 2{,}4 = 5{,}4$ (cm). 
	}
\end{vd}

\begin{vd}%[8H3K3]
	Cho $\triangle ABC$ cân tại $A$ có $BM,CN$ là các đường phân giác. Chứng minh rằng
	\begin{listEX}[2]
		\item $MN \parallel BC$.
		\item $\dfrac{1}{BC} + \dfrac{1}{AB} = \dfrac{1}{MN}$.
	\end{listEX}
	\loigiai
	{
		\immini
		{
			\begin{enumerate}
				\item Áp dụng tính chất đường phân giác, ta có $\dfrac{AM}{MC} = \dfrac{AB}{BC}$ và $\dfrac{AN}{NB} = \dfrac{AC}{BC}$.\\
				Mà $AB = AC$ nên $\dfrac{AM}{MC} = \dfrac{AN}{NB} \Rightarrow MN \parallel BC$ (Định lý Ta-lét đảo).
				\item $MN \parallel BC \begin{aligned}[t]
				&\Rightarrow \widehat{MNC} = \widehat{NCB} \Rightarrow \widehat{MNC} = \widehat{MCN}\\
				&\Rightarrow \triangle MNC \text{ cân } \Rightarrow MN = MC.
				\end{aligned}$\\
				$MN \parallel BC \Rightarrow \dfrac{MN}{BC} = \dfrac{AM}{AC}$ (hệ quả định lý Ta-lét)
			\end{enumerate}
		}
		{
				\begin{tikzpicture}[line join = round, line cap = round,>=stealth,font=\footnotesize,scale=.9]
			\tkzDefPoints{0/0/B}
			\coordinate (C) at ($(B)+(4,0)$);
			\tkzDefTriangle[two angles = 70 and 70](B,C)    \tkzGetPoint{A}
			\tkzDrawBisector(C,B,A)  \tkzGetPoint{M}
			\tkzDrawBisector(A,C,B)  \tkzGetPoint{N}
			\tkzMarkAngles[size=.7cm,arc=l](C,B,M)
			\tkzMarkAngles[size=.9cm,arc=l](M,B,A)
			\tkzMarkAngles[size=.7cm,arc=l,mark=|](A,C,N)
			\tkzMarkAngles[size=.9cm,arc=l,mark=|](N,C,B)

			\tkzLabelPoints[above](A)
			\tkzLabelPoints[left](B,N)
			\tkzLabelPoints[right](C,M)
			\end{tikzpicture}
		}
		$\Rightarrow \dfrac{MN}{BC} = \dfrac{AC - MN}{AC}$ (vì $MC = MN$)
		hay $\dfrac{MN}{BC} = 1 - \dfrac{MN}{AC} \Rightarrow \dfrac{MN}{BC} + \dfrac{MN}{AC} = 1$.
		\par Vậy $\dfrac{1}{BC} + \dfrac{1}{AB} = \dfrac{1}{MN}$.
	}
\end{vd}

\subsection{Các dạng bài tập và phương pháp giải}
\begin{dang}{Tìm tỉ số đồng dạng của hai tam giác}
	\emph{Phương pháp giải:} Sử dụng định nghĩa hoặc định lý hai tam giác đồng dạng.

\begin{vd}%[8H3B4]
	Cho $\triangle ABC$, lấy điểm $M$ thuộc $AB$ sao cho $\dfrac{AM}{MB} = \dfrac{2}{3}$. Kẻ hai đường thẳng qua $M$ lần lượt song song với $AC$ và $BC$ cắt $BC$ và $AC$ lần lượt tại $D$ và $F$.

		\item Nêu tất cả các cặp tam giác đồng dạng.
		\item Với mỗi cặp tam giác đồng dạng, hãy viết tỉ số đồng dạng tương ứng.
	\end{enumerate}

	{
		\immini
		{

				&MD \parallel AC \text{ nên } \triangle MBD \backsim \triangle ABC \tag{2}

			\end{enumerate}

\begin{vd}%[8H3B4]
	Cho $\triangle ABC \backsim \triangle DEF$, tỉ số đồng dạng bằng $\dfrac{2}{3}$. Biết chu vi $\triangle ABC$ là $24$ cm. Tính chu vi $\triangle DEF$.
	\loigiai
	{
		$\triangle ABC \backsim \triangle DEF$, tỉ số đồng dạng bằng $\dfrac{2}{3}$ nên $\dfrac{AB}{DE} = \dfrac{BC}{EF} = \dfrac{CA}{FD} = \dfrac{2}{3}$.\\
		Suy ra $\dfrac{AB+BC+CA}{DE+EF+FD} = \dfrac{2}{3}$ hay $\dfrac{24}{DE+EF+FD} = \dfrac{2}{3}$.\\
		Vậy chu vi $\triangle DEF$ là $\dfrac{24\cdot 3}{2} = 36$ (cm).	
	}
\end{vd}

\begin{dang}{Tính độ dài đoạn thẳng}

	\begin{itemize}
		\item Viết hai tam giác đồng dạng.
		\item Lập tỉ số các cặp cạnh tương ứng rồi sử dụng tỉ lệ thức.
	\end{itemize}
\end{dang}
\begin{vd}%[8H3B4]
	Cho $\triangle ABC \backsim \triangle A'B'C'$, biết $AB = 8$ cm, $AC = 6$ cm, $BC = 10$ cm. Cạnh lớn nhất của $\triangle A'B'C'$ là $25$ cm. Tính cạnh nhỏ nhất của $\triangle A'B'C'$.
	\loigiai 
	{
		$\triangle ABC$ có cạnh $BC = 10$ cm là lớn nhất. Mà $\triangle ABC \backsim \triangle A'B'C'$ nên $B'C'$ là cạnh lớn nhất của $\triangle A'B'C'$, do đó $B'C' = 25$ cm.\\
		$AC = 6$ cm là cạnh nhỏ nhất của $\triangle ABC$ nên $A'C'$ là cạnh nhỏ nhất của $\triangle A'B'C'$. Suy ra 
		\begin{eqnarray*}
			&&\dfrac{AC}{A'C'} = \dfrac{BC}{B'C'}\\
			&\Rightarrow& \dfrac{6}{A'C'} = \dfrac{10}{25}\\
			&\Rightarrow& A'C' = \dfrac{6 \cdot 10}{25} = 15 \text{ cm.}
		\end{eqnarray*}
	}
\end{vd}

\begin{dang}{Chứng minh hai tam giác đồng dạng}
	\emph{Phương pháp giải:} 
	\begin{itemize}
		\item Xếp các cạnh của hai tam giác theo cùng một thứ tự (chẳng hạn từ nhỏ tới lớn).
		\item Lập ba tỉ số, nếu chúng bằng nhau thì hai tam giác đồng dạng.
	\end{itemize}
\end{dang}
\begin{vd}%[8H3B5]
	Cho $\triangle ABC$ có $AB = 3$ cm, $BC = 5$ cm, $CA = 7$ cm và $\triangle A'B'C'$ có $A'B' = 4{,}5$ cm, $B'C' = 7{,}5$ cm, $C'A' = 10{,}5$ cm. Hỏi $\triangle ABC$ và $\triangle A'B'C'$ có đồng dạng với nhau không? Tại sao?
	\loigiai
	{
		Ta có $\dfrac{3}{4{,}5} = \dfrac{5}{7{,}5} = \dfrac{7}{10{,}5} \left(=\dfrac{1}{1{,}5}\right)$ nên $\dfrac{AB}{A'B'} = \dfrac{BC}{B'C'} = \dfrac{CA}{C'A'}$.\\
		Do đó $\triangle ABC \backsim \triangle A'B'C'$ (c.c.c).
	}
\end{vd}

\begin{vd}%[8H3G5]
	Cho điểm $M$ nằm trong $\triangle ABC$. Gọi $G_1,G_1,G_3$ lần lượt là trong tâm các tam giác $MBC, MCA,MAB$. Chứng minh rằng $\triangle G_1G_2G_3 \backsim \triangle ABC$.

			$\Rightarrow \dfrac{MG_2}{ME} = \dfrac{MG_1}{MF} = \dfrac{2}{3}$\\
			$\Rightarrow G_1G_2 \parallel EF \Rightarrow \dfrac{G_1G_2}{EF} = \dfrac{MG_2}{ME} \Rightarrow \dfrac{G_1G_2}{EF} = \dfrac{2}{3}$. \\
			Mặt khác ta có $EF = \dfrac{1}{2}AB \Rightarrow G_1G_2 = \dfrac{1}{3}AB$.\\
			Tương tự ta có $G_2G_3 = \dfrac{1}{3}BC$ và $G_1G_3 = \dfrac{1}{3}AC$.
		}

\begin{bt}%[8H3B5]
	Cho $\triangle ABC$ có $AB = 6$ cm, $AC = 9$ cm, $BC = 12$ cm và $\triangle DEF$ có $DE = 24$ cm, $EF = 18$ cm, $DF = 12$ cm. $\triangle ABC$ có đồng dạng với $\triangle DEF$ hay không? Vì sao?
	\loigiai
	{
		Ta có $\dfrac{AB}{FD} = \dfrac{AC}{FE} = \dfrac{BC}{DE} = \dfrac{1}{2}$.\\
		Suy ra $\triangle ABC \backsim \triangle FDE$ (c.c.c).
	}
\end{bt}

\begin{bt}%[8H3K5]
	Cho tứ giác $ABCD$ có $AB = 2$ cm, $BC = 10$ cm, $CD = 12{,}5$ cm, $AD = 4$ cm, $BD = 5$ cm. Chứng minh rằng tứ giác $ABCD$ là hình thang.
	\loigiai
	{
		\immini
		{
			Xét $\triangle ABD$ và $\triangle BDC$ có
			\begin{equation*}
			\dfrac{AB}{BD} = \dfrac{AD}{BC} = \dfrac{BD}{DC} \left(\text{vì } \dfrac{2}{5} = \dfrac{4}{10} = \dfrac{5}{12{,}5}\right)

			Vậy tứ giác $ABCD$ là hình thang.

\begin{bt}%[8H3B5]
	Cho $\triangle ABC$ vuông tại $A$ có $AB = 3$ cm, $AC = 4$ cm và $\triangle DEF$ vuông tại $D$ có $DE = 15$ cm, $EF = 25$ cm.
	\begin{enumerate}
		\item Tính độ dài các đoạn thẳng $BC, DF$.
		\item $\triangle ABC$ và $\triangle DEF$ có đồng dạng với nhau không? Vì sao?
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
			\item $BC = 5$ cm, $DF = 20$ cm.

		\end{enumerate}
	}
\end{bt}

\begin{bt}%[8H3G5]
	Cho $\triangle ABC$. Trên cạnh $AB$ lấy các điểm $I,K$ sao cho $AI = IK = KB$. Trên cạnh $BC$ lấy các điểm $E$ và $D$ sao cho $BD = DE = EC$. Trên cạnh $AC$ lấy điểm $F$ và $G$ sao cho $AF = FG = GC$. Gọi $M$ là giao điểm của $AD$ và $BF$, $N$ là giao điểm của $BG$ và $CK$,$P$ là giao điểm của $AE$ và $CI$. Chứng minh $\triangle ABC \backsim \triangle NPM$.
	\loigiai
	{
		\immini
		{
			$\dfrac{AF}{AC} = \dfrac{BD}{BC} = \dfrac{1}{3} \Rightarrow DF \parallel AB$.\\
			$DF \parallel AB \Rightarrow \dfrac{DF}{AB} = \dfrac{CD}{CB} \Rightarrow \dfrac{DF}{AB} = \dfrac{2}{3}$.\\
			$DF \parallel AB \Rightarrow \dfrac{AM}{MD} = \dfrac{AB}{DF} \Rightarrow \dfrac{AM}{MD} = \dfrac{3}{2} \Rightarrow \dfrac{AM}{AD} = \dfrac{3}{5}$.\\
			Tương tự ta có $\dfrac{AP}{AE} = \dfrac{3}{5} \Rightarrow \dfrac{AM}{AD} = \dfrac{AP}{AE}$ hay $MP \parallel DE$ hay $MP \parallel BC$.\\
			Do đó $\dfrac{MP}{DE} = \dfrac{AM}{AD} \Rightarrow \dfrac{MP}{DE} = \dfrac{3}{5}$.\\
			Mặt khác $DE = \dfrac{1}{3}BC \Rightarrow \dfrac{MP}{BC} = \dfrac{1}{5}$. \hfill (1)\\
			Tương tự ta có $\dfrac{NP}{AB} = \dfrac{1}{5}$ và $\dfrac{MN}{AC} = \dfrac{1}{5}$. \hfill (2)\\
			Từ (1) và (2) suy ra $\triangle ABC \backsim \triangle NPM$ (c.c.c).
		}
		{
			\begin{tikzpicture}[line join = round, line cap = round,>=stealth,font=\footnotesize,scale=.9]

			\coordinate (C) at ($(B)+(6,0)$);
