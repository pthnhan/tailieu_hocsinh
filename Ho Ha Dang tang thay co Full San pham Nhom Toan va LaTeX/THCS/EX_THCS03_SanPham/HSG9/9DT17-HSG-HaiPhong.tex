\begin{ex}%[HSG toán 9, Hải Phòng, 2016 - 2017]%[9D1K8]
Cho $x=\dfrac{\sqrt[3]{10+6\sqrt{3}}\Big(\sqrt{3}-1\Big)}{\sqrt{6+2\sqrt{5}}-\sqrt{5}}$. Tính giá trị của biểu thức: $P=(12x^2+4x-55)^{2017}$.
\loigiai{
Ta có:
\begin{align*}
& \sqrt[3]{10+6\sqrt{3}}\Big(\sqrt{3}-1\Big)=\sqrt[3]{(\sqrt{3}+1)^3}\Big(\sqrt{3}-1\Big)=\Big(\sqrt{3}+1\Big)\Big(\sqrt{3}-1\Big)=2.\\
& \sqrt{6+2\sqrt{5}}-\sqrt{5}=\sqrt{(\sqrt{5}+1)^2}-\sqrt{5}=\sqrt{5}+1-\sqrt{5}=1.
\end{align*}
Vậy $x=2$. Thay giá trị $x$ vào $P$ ta được: $$P=(12.2^2+4.2-55)^{2017}=1^{2017}=1.$$
}
\end{ex}

\begin{ex}%[HSG toán 9, Hải Phòng, 2016 - 2017]%[9D1K8]
Cho biểu thức $M=\dfrac{a+1}{\sqrt{a}}+\dfrac{a\sqrt{a}-1}{a-\sqrt{a}}+\dfrac{a^2-a\sqrt{a}+\sqrt{a}-1}{\sqrt{a}-a\sqrt{a}}$ với $a>0,\ a\neq 1$. Với những giá trị nào của $a$ thì biểu thức $N=\dfrac{6}{M}$ nhận giá trị nguyên?
\loigiai{
Với điều kiện $a>0,\ a\neq 1$ thì:\\
\begin{align*}
M & = \dfrac{a+1}{\sqrt{a}}+\dfrac{\Big(\sqrt{a}-1\Big)\Big(a+\sqrt{a}+1\Big)}{\sqrt{a}\Big(\sqrt{a}-1\Big)}-\dfrac{\Big(\sqrt{a}-1\Big)\Big(\sqrt{a}+1\Big)\Big(a-\sqrt{a}+1\Big)}{\sqrt{a}\Big(\sqrt{a}-1\Big)\Big(\sqrt{a}+1\Big)}\\
& = \dfrac{a+1}{\sqrt{a}}+\dfrac{a+\sqrt{a}+1}{\sqrt{a}}-\dfrac{a-\sqrt{a}+1}{\sqrt{a}}=\dfrac{\Big(\sqrt{a}+1\Big)^2}{\sqrt{a}}.\\
\Rightarrow N& =\dfrac{6}{M}=\dfrac{6\sqrt{a}}{\Big(\sqrt{a}+1\Big)^2}>0.
\end{align*}
Khi đó ta thấy với $0<a\neq 1$ thì 
$$a-\sqrt{a}+1>0\Leftrightarrow \Big(\sqrt{a}+1\Big)^2>3\sqrt{a}\Leftrightarrow \dfrac{6\sqrt{a}}{\Big(\sqrt{a}+1\Big)^2}<2.$$
Do $0<N<2$ nên để $N$ có giá trị nguyên thì $N=1$. Điều này xảy ra khi và chỉ khi
\begin{align*}
&\dfrac{6\sqrt{a}}{a+2\sqrt{a}+1}=1\Leftrightarrow a-4\sqrt{a}+1=0\Leftrightarrow \Big(\sqrt{a}-2\Big)^2=3\\
\Leftrightarrow & \left[\begin{array}{l}
\sqrt{a}=\sqrt{3}+2\\ \sqrt{a}=-\sqrt{3}+2
\end{array}\right. \Leftrightarrow  \left[\begin{array}{l}
a=7+4\sqrt{3}\ (\text{thỏa mãn})\\ a=7-4\sqrt{3}\ (\text{thỏa mãn})
\end{array}\right.
\end{align*}
Vậy $a=7\pm 4\sqrt{3}$.
}
\end{ex}

\begin{ex}%[HSG toán 9, Hải Phòng, 2016 - 2017]%[9D4G3]
Cho phương trình: $x^2-2mx+m^2-m-6=0$ (với $m$ là tham số). Với giá trị nào của $m$ thì phương trình có hai nghiệm $x_1,\ x_2$ sao cho $|x_1|+|x_2|=8$?
\loigiai{
Phương trình $x^2-2mx+m^2-m-6=0$ có hai nghiệm thì
$$\Delta'=m^2-(m^2-m-6)=m+6\geq 0\Leftrightarrow m\geq -6.$$
Theo hệ thức Viét, ta có: $$\left\{\begin{array}{l}
x_1+x_2=2m\\ x_1x_2=m^2-m-6
\end{array}\right.$$
Ta có
\begin{align*}
|x_1|+|x_2|=8\Leftrightarrow x_1^2+x_2^2+2|x_1x_2|=64 \Leftrightarrow (x_1+x_2)^2-2x_1x_2+2|x_1x_2|=64.\tag{1}
\end{align*}
{\bf Trường hợp 1:} Nếu $x_1$ và $x_2$ cùng dấu thì:
\begin{align*}
x_1x_2\geq 0\Leftrightarrow \left\{\begin{array}{l}
m\geq -6\\ m^2-m-6=(m+2)(m-3)\geq 0
\end{array}\right.\Leftrightarrow\left[\begin{array}{l}
-6\leq m\leq -2\\ m\geq 3
\end{array}\right. \tag{*}
\end{align*}
Khi đó
$$(1)\Leftrightarrow (x_1+x_2)^2=64\Leftrightarrow 4m^2=64\Leftrightarrow m=\pm 4\ (\text{thỏa mãn (*)})$$
{\bf Trường hợp 2:} Nếu $x_1$ và $x_2$ trái dấu thì:
\begin{align*}
x_1x_2<0\Leftrightarrow m^2-m-6=(m+2)(m-3)<0\Leftrightarrow -2<m<3. \tag{**}
\end{align*}
Khi đó
\begin{align*}
(1)\Leftrightarrow&\ (x_1+x_2)^2-4x_1x_2=64\Leftrightarrow 4m^2-4(m^2-m-6)=64\\
\Leftrightarrow&\ m+6=16\Leftrightarrow m=10\ \text{(không thỏa mãn điều kiện (**))}
\end{align*}
Kết luận: $m=\pm 4$.
}
\end{ex}

\begin{ex}%[HSG toán 9, Hải Phòng, 2016 - 2017]%[9D4G3]
Cho hệ phương trình $\left\{\begin{array}{l}
x^3y^2-2x^2y-x^2y^2+2xy+3x-3=0\\ y^2+x^{2017}=y+3m
\end{array}\right.$ (với $m$ là tham số).\\
Tìm các giá trị của $m$ để hệ phương trình có hai nghiệm phân biệt $(x_1;y_1)$ và $(x_2;y_2)$ thỏa mãn điều kiện $(x_1+y_2)(x_2+y_1)+3=0$.
\loigiai{
Ta có
\begin{align*}
& x^3y^2-2x^2y-x^2y^2+2xy+3x-3=0\\
\Leftrightarrow\ & (x-1)(x^2y^2-2xy+3)=0\\
\Leftrightarrow\ & \left[\begin{array}{l}
x=1\\(xy-1)^2+2=0\ (\text{vô nghiệm})
\end{array}\right.
\end{align*}
Thay $x=1$ vào phương trình dưới của hệ ta được $y^2-y-3m+1=0.\quad (1)$\\
Để phương trình $(1)$  có hai nghiệm phân biệt thì
$$\Delta=1+4(3m-1)>0\Leftrightarrow 12m-3>0\Leftrightarrow m>\dfrac{1}{4}$$
Theo đề bài và chú ý rằng $x_1=x_2=1$:
$$(x_1+y_2)(x_2+y_1)+3=0\Leftrightarrow 4+y_1+y_2+y_1y_2=0.\quad (2)$$
Với $m>\dfrac{1}{4}$, theo hệ thức Viét cho phương trình $(1)$ ta có
$$\left\{\begin{array}{l}
y_1+y_2=1\\ y_1y_2=1-3m
\end{array}\right.$$
Thay vào $(2)$ ta được
$$5+1-3m=0\Leftrightarrow m=2\ (\text{thỏa mãn}).$$
Kết luận $m=2$.
}
\end{ex}

\begin{ex}%[HSG toán 9, Hải Phòng, 2016 - 2017] %[9D5G2]
Tìm tất cả các số nguyên dương $a,\ b$ sao cho $a+b^2$ chia hết cho $a^2b-1$.
\loigiai{
Ta có 
\begin{align*}
& (a+b^2)\ \vdots\ (a^2b-1)\Rightarrow a+b^2=k(a^2b-1),\ k\in\mathbb{N^*}\\
\Leftrightarrow\ & a+k=b(ka^2-b)\ \text{hay}\ mb=a+k\quad (1)\ \text{với}\ m=ka^2-b\in \mathbb{N^*}\\
\Leftrightarrow\ & m+b=ka^2.\quad (2)
\end{align*}
Từ $(1)$ và $(2)$ suy ra: 
$$mb-m-b+1=a+k-ka^2+1\Leftrightarrow (m-1)(b-1)=(a+1)(k+1-ka).\quad (3)$$
Do $m,\ b\in\mathbb{N^*}\Rightarrow (m-1)(b-1)\geq 0$.\\
Vì thế từ $(3)$ suy ra: $(a+1)(k+1-ka)\geq 0$.\\
Lại do $a>0$ nên suy ra: $k+1-ka\geq 0\Rightarrow 1\geq k(a-1)$.\\
Vì $a-1>0,\ k>0$ nên $1\geq k(a-1)\geq 0$ và $k(a-1)\in\mathbb{N^*}$\\
$\Rightarrow \left[\begin{array}{l}
k(a-1)=0\\k(a-1)=1
\end{array}\right.\Leftrightarrow \left[\begin{array}{l}
a=1\\ \begin{cases} a=2\\ k=1\end{cases}
\end{array}\right.$\\
Với $a=1$. Thay vào $(3)$ ta được:
\begin{align*}
& (m-1)(b-1)=2\\
\Leftrightarrow\ &\left[\begin{array}{l}
\left\{\begin{array}{l}
m-1=2\\ b-1=1
\end{array}\right.\\
\left\{\begin{array}{l}
m-1=1\\ b-1=2
\end{array}\right.
\end{array}\right. \Leftrightarrow\left[\begin{array}{l}
b=2\Rightarrow ka^2=5\Rightarrow a=1\\
b=3\Rightarrow ka^2=5\Rightarrow a=1
\end{array}\right.
\end{align*}
Vậy, trường hợp này ta được hai cặp $a=1,\ b=2$ và $a=1,\ b=3$.\\
Với $a=2$ và $k=1$. Thay vào $(3)$ ta có:
$$(m-1)(b-1)=0\Leftrightarrow \left[\begin{array}{l}
b=1\\ m=1
\end{array}\right.$$
Khi $b=1$, ta được $a=2,\ b=1$.\\
Khi $m=1$. Từ $(1)$ suy ra $a+k=b\Rightarrow b=3$. Khi đó $a=2,\ b=3$.\\
Vậy có 4 cặp số $(a;b)$ thỏa mãn là: $(1;2),\ (1;3),\ (2;3),\ (2;1)$.
}
\end{ex}

\begin{ex}%[HSG toán 9, Hải Phòng, 2016 - 2017]% [9D5G1]
Cho ba số thực $a,\ b,\ c$ dương. Chứng minh rằng:
$$\sqrt{\dfrac{a^3}{a^3+(b+c)^3}}+\sqrt{\dfrac{b^3}{b^3+(c+a)^3}}+\sqrt{\dfrac{c^3}{c^3+(a+b)^3}}\geq 1$$
\loigiai{
Với $x$ là số dương, áp dụng bất đẳng thức Cauchy ta có:
\begin{align*}
& \sqrt{x^3+1}=\sqrt{(x+1)(x^2-x+1)}\leq \dfrac{x+1+x^2-x+1}{2}=\dfrac{x^2+2}{2}\\
\Rightarrow\ & \sqrt{\dfrac{1}{x^3+1}}\geq \dfrac{2}{x^2+2} \tag{*}\\
\end{align*}
Dấu ``='' xảy ra khi  $x=2$.\\
Áp dụng bất đẳng thức $(*)$ ta được
$$\sqrt{\dfrac{a^3}{a^3+(b+c)^3}}=\sqrt{\dfrac{1}{1+\bigg(\dfrac{b+c}{a}\bigg)^3}}\geq \dfrac{2}{\bigg(\dfrac{b+c}{a}\bigg)^2+2}=\dfrac{2a^2}{(b+c)^2+2a^2}.$$
Suy ra 
\begin{align*}
\sqrt{\dfrac{a^3}{a^3+(b+c)^3}}\geq \dfrac{2a^2}{2\Big(b^2+c^2\Big)+2a^2}=\dfrac{a^2}{a^2+b^2+c^2}.\tag{1}
\end{align*}
Tương tự ta có
\begin{align*}
\sqrt{\dfrac{b^3}{b^3+(c+a)^3}}\geq&\ \dfrac{b^2}{a^2+b^2+c^2}.\tag{2}\\
\sqrt{\dfrac{c^3}{c^3+(a+b)^3}}\geq&\ \dfrac{c^2}{a^2+b^2+c^2}.\tag{3}
\end{align*}
Cộng theo vế của ba bất đẳng thức $(1),\ (2)$ và $(3)$ ta được:
$$\sqrt{\dfrac{a^3}{a^3+(b+c)^3}}+\sqrt{\dfrac{b^3}{b^3+(c+a)^3}}+\sqrt{\dfrac{c^3}{c^3+(a+b)^3}}\geq 1.$$
Dấu ``='' xảy ra khi $a=b=c$.
}
\end{ex}

\begin{ex}%[HSG toán 9, Hải Phòng, 2016 - 2017]%[9H3K3]
Cho ba điểm $A,\ B,\ C$ cố định nằm trên một đường thẳng $d$ (điểm $B$ nằm giữa điểm $A$ và điểm $C$). Vẽ đường tròn tâm $O$ thay đổi nhưng luôn đi qua điểm $B$ và điểm $C$ (điểm $O$ không thuộc đường thẳng $d$). Kẻ $AM$ và $AN$ là các tiếp tuyến với đường tròn tâm $O$ (với $M$ và $N$ là các tiếp điểm). Đường thẳng $BC$ cắt $MN$ tại điểm $K$. Đường thẳng $AO$ cắt $MN$ tại điểm $H$ và cắt đường tròn tại các điểm $P$ và $Q$ ($P$ nằm giữa $A$ và $Q$).
\begin{enumerate}
\item[a)] Chứng minh điểm $K$ cố định khi đường tròn tâm $O$ thay đổi.
\item[b)] Gọi $D$ là trung điểm của $HQ$, từ $H$ kẻ đường thẳng vuông góc với $MD$ cắt đường thẳng $MP$ tại $E$. Chứng minh rằng $P$ là trung điểm của $ME$.
\end{enumerate}
\loigiai{\\
\begin{center}
\begin{tikzpicture}
	\clip (-6,-5) rectangle (7,3);
    \coordinate (A) at (-5,0);
    \coordinate (B) at (0,0);
    \coordinate (C) at (6,0);
    \coordinate (O) at (3,-1.5);
    \coordinate (I) at ($(B)!0.5!(C)$);
    \tkzDrawPoints(A,B,C,I,O)
    \draw (A) -- (C) (O) -- (I); 
    \tkzDrawCircle(O,B) 
    \tkzTangent[from=A](O,B) \tkzGetSecondPoint{M}
    \tkzTangent[from=A](O,B) \tkzGetFirstPoint{N}
    \tkzDrawPoints(M,N)
    \draw (A) -- (M) -- (N) -- (A) (B) -- (N) -- (C) (N) -- (O) (A) -- (O);
    \tkzInterLL(A,C)(M,N) \tkzGetPoint{K} \tkzDrawPoint(K)
    \tkzInterLL(A,O)(M,N) \tkzGetPoint{H} \tkzDrawPoint(H)
    \tkzInterLC(A,O)(O,B) \tkzGetPoints{P}{Q} \tkzDrawPoints(P,Q)
    \tkzDrawSegments(O,Q M,Q) \tkzLabelPoints[right](Q)
    \coordinate (D) at ($(H)!0.5!(Q)$); \tkzDrawPoint(D) \tkzDrawSegment(M,D)
    \coordinate (F) at ($(M)!(H)!(D)$); \tkzDrawPoint(F)
    \tkzInterLL(M,P)(F,H) \tkzGetPoint{E} \tkzDrawPoint(E)
    \tkzDrawSegments(E,M E,F)
    \tkzLabelPoints[below left](A,N,E)   \tkzLabelPoints[above right](C)
    \tkzLabelPoints[above left](B,K,I,H,P)
    \tkzLabelPoints[above](M)
    \tkzLabelPoints[below](O,D)
\end{tikzpicture}
\end{center}
a) Gọi $I$ là trung điểm của $BC$, suy ra $IO\perp BC$.\\
$\triangle ABN$ đồng dạng với $\triangle ANC$ (vì $\widehat{ANB}=\widehat{ACN}$, $\widehat{CAN}$ chung)
$$\Rightarrow \dfrac{AB}{AN}=\dfrac{AN}{AC}\Rightarrow AB.AC=AN^2$$
$\triangle ANO$ vuông tại $N$, đường cao $NH$ nên $AH.AO=AN^2$. Suy ra
\begin{align*}
& AB.AC=AH.AO \tag{1}
\end{align*}
$\triangle AHK$ đồng dạng với $\triangle AIO\ (g.g)$ nên
\begin{align*}
\dfrac{AH}{AI}=\dfrac{AK}{AO}\Rightarrow AI.AK=AH.AO \tag{2}
\end{align*}
Từ $(1)$ và $(2)$ suy ra $$AI.AK=AB.AC\Rightarrow AK=\dfrac{AB.AC}{AI}$$
Ta có $A, B, C$ cố định nên $I$ cũng cố định, suy ra $AK$ không đổi. Mà $A$ cố định, $K$ là giao điểm của $BC$ và $MN$ nên $K$ thuộc tia $AB$, suy ra $K$ cố định.\\
b) Xét hai tam giác $MHE$ và $QDM$. Có\\
$\widehat{NME}=\widehat{MQP}$\\
$\widehat{NHE}=90^\circ-\widehat{AHE}=\widehat{MDH}$. Suy ra $\widehat{MHE}=180^\circ-\widehat{NHE}=180^\circ-\widehat{MDH}=\widehat{MDQ}$\\
Vậy $\triangle MHE$ đồng dạng với $\triangle QDM$, suy ra $\dfrac{ME}{MQ}=\dfrac{MH}{DQ}$.\\
Mặt khác, $\triangle PMH$ đồng dạng với $\triangle MQH$ (g.g), suy ra $\dfrac{MP}{MQ}=\dfrac{MH}{QH}=\dfrac{MH}{2DQ}$.\\
Từ hai kết quả trên suy ra 
$$\dfrac{MP}{MQ}=\dfrac{1}{2}.\dfrac{ME}{MQ}\Rightarrow ME=2MP$$
Từ đó suy ra $P$ là trung điểm của $ME$.
}
\end{ex}

\begin{ex}%[HSG toán 9, Hải Phòng, 2016 - 2017]% [9D5G5]
Cho tập hợp $A$ gồm $21$ phần tử là các số nguyên khác nhau thỏa mãn tổng của $11$ phần tử bất kì lớn hơn tổng của $10$ phần tử còn lại. Biết các số $101$ và $102$ thuộc tập hợp $A$. Tìm tất cả các phần tử của tập hợp $A$.
\loigiai{
Giả sử $A=\{a_1;\ a_2;\ a_3;\ \cdots;\ a_{21}\}$ với $a_1,\ a_2,\ a_3,\ \cdots,\ a_{21}\in\mathbb{Z}$ và $a_1<a_2<a_3<\cdots<a_{21}$.\\
Theo giả thiết ta có
\begin{align*}
& a_1+a_2+a_3+\cdots+a_{11}>a_{12}+a_{13}+ \cdots +a_{21}\\
\Leftrightarrow a_1>\ & (a_{12}-a_2)+(a_{13}-a_3)+\cdots+(a_{21}-a_{11})\tag{1}
\end{align*}
Mặt khác với $x,\ y\in\mathbb{Z}$ và nếu $x<y$ thì $y\geq x+1$. Suy ra
\begin{align*}
a_{12}-a_2\geq 10,\ a_{13}-a_3\geq 10,\ \cdots,\ a_{21}-a_{11}\geq 10\tag{2}
\end{align*}
Nên từ $(1)$ suy ra
$$a_1>10+10+\cdots+10=100$$
Mà $a_1$ nhỏ nhất và $101\in A$ nên $a_1=101$.\\
Ta có
\begin{align*}
& 101>(a_{12}-a_2)+(a_{13}-a_3)+\cdots+(a_{21}-a_{11})\geq 100\\
\Rightarrow & (a_{12}-a_2)+(a_{13}-a_3)+\cdots+(a_{21}-a_{11}) =100
\end{align*}
Kết hợp với $(2)$, suy ra
\begin{align*}
& a_{12}-a_2=a_{13}-a_3=\cdots=a_{21}-a_{11}=10\tag{3}\\
\Rightarrow\ & 10=a_{12}-a_2=(a_{12}-a_{11})+(a_{11}-a_{10})+\cdots+(a_3-a_2)\geq 10\\
\Rightarrow\ & a_{12}-a_{11}=a_{11}-a_{10}=\cdots=a_3-a_2=1\tag{4}
\end{align*}
Ta có $a_1=101$ mà $102\in A$ suy ra $a_2=102$.\\
Kết hợp với $(3)$ và $(4)$ suy ra $A=\{101;\ 102;\ \cdots;\ 121\}$.
}
\end{ex}
