\begin{ex}%[Đề thi vào 10, Sở giáo dục Hưng Yên, 2017]%[9D1B8]%[9D2K3]%[9D3B2]
	\begin{enumerate} [a)]
		\item Rút gọn biểu thức $A=\sqrt{3}+\sqrt{(2-\sqrt{3})^2}+6$.
		\item Tìm $m$ để đồ thị hàm số $y=mx+3$ cắt trục hoành tại điểm có hoành độ bằng $ 3 $.
		\item Giải hệ phương trình $\heva{&x+3y=9\\
			&x-y=1}$.
\end{enumerate}
\loigiai{
	\begin{enumerate} [a)]
		\item $A=\sqrt{3}+\sqrt{(2-\sqrt{3})^2}+6=\sqrt{3}+|2-\sqrt{3}|+6=\sqrt{3}+2-\sqrt{3}+6=2+6=8$.
		\item Đồ thị hàm số cắt trục hoành tại điểm có hoành độ bằng $5$ nên ta có phương trình: $$0=5m+3\Leftrightarrow m=\dfrac{-3}{5}$$
		\item Lấy phương trình dưới nhân với $-1$ rồi cộng với phương trình trên ta được phương trình: $4y=8\Leftrightarrow y=2\Rightarrow x=3$.\\
		Hoặc rút $x=9-3y$ thay vào phương tình dưới ta được phương trình:\\ $9-3y-y=1\Rightarrow y=2\Rightarrow x=9-3.2=3$.
	\end{enumerate}
}
\end{ex}


\begin{ex}%[Đề thi vào 10, Sở giáo dục Hưng Yên, 2017]%[9D4B3]%[9D4K6]
	Cho phương trình $x^2-2x-m=0$ ($m$ là tham số).
	\begin{enumerate} [a)]
		\item Giải phương trình với $m=3$.
		\item Tìm các giá trị của $m$ để phương trình có hai nghiệm phân biệt $x_1, x_2$ thỏa mãn điều kiện: $(x_1 x_2+1)^2-2(x_1+x_2)=0$.
\end{enumerate}
\loigiai{ 
	\begin{enumerate} [a)]
		\item Thay $m=3$ vào phương trình ta có $x^2-2x-3=0$.\\
		Ta có ${\Delta}'={b'}^2-a.c=1+3=4$.\\
		Theo công thức nghiệm, ta có:
		$\left[ \begin{array}{ll} 
		x_1=\dfrac{-b'-\sqrt{{\Delta'}}}{a}=\dfrac{1-2}{1}=-1\\ 
		x_2=\dfrac{-b'+\sqrt{{\Delta'}}}{a}=\dfrac{1+2}{1}=3
		\end{array} \right.$.
		\item Phương trình có 2 nghiệm phân biệt $\Leftrightarrow {\Delta}'>0$ $\Leftrightarrow 1+m>0$ $ \Leftrightarrow m>-1$.\\
		Theo định lý Vi-ét $\heva{
		&x_1+x_2=\dfrac{-b}{a}=2\\
		&x_1 x_2=\dfrac{c}{a}=-m\\
		}$.\\
		Thay vào điều kiện ta có:\\
		$(-m+1)^2-2.2=0\Leftrightarrow (1-m)^2-4=0\Leftrightarrow$
		$\left[ \begin{array}{ll} 
		m=3 &\text{(nhận)}\\ 
		m=-1 &\text{(loại)}
		\end{array} \right.$.
	\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Hưng Yên, 2017]%[9H3G7]
	Cho tứ giác $ABCD$ nội tiếp đường tròn đường kính $AB$. Hai đường chéo $AC$ và $BD$ cắt nhau tại $E,F$ là hình chiếu vuông góc của $E$ trên $ AB $.
	\begin{enumerate} [a)]
		\item Chứng minh tứ giác $ADEF$ nội tiếp.
		\item Gọi $N$ là giao điểm của $CF$ và $BD$. Chứng minh $BN.ED=BD.EN$.	
\end{enumerate}
\loigiai{ 
\immini{a) $\triangle ADB$ nội tiếp đường tròn và có cạnh $AB$ là đường kính nên $\triangle ADB$ vuông tại $D\Rightarrow \widehat{ADB}=90^0$. \\
	Mặt khác $EF\perp AB\Rightarrow \widehat{AFE}=90^0$.\\
	Xét tứ giác $ADEF$ có: $\widehat{ADE}+\widehat{AFE}=90^0+90^0=180^0$. Vậy tứ giác này nội tiếp đường tròn.}{
	\begin{tikzpicture}[scale = 1]
	\tkzInit[xmin = -3.5,xmax=3.5,ymin=-3.5,ymax=3.5]
	\tkzClip
	\tkzDefPoints{0/0/O,-3/0/A,3/0/B,1/1.5/E}
	\tkzDrawCircle(O,A)
	\tkzInterLC(E,B)(O,A) \tkzGetPoints{D}{D'} 
	\tkzInterLC(A,E)(O,A) \tkzGetPoints{C'}{C}
	\tkzDrawPolygon(A,B,C,D)
	\tkzDefLine[orthogonal=through E](B,A) \tkzGetPoint{f} 
	\tkzInterLL(E,f)(A,B) \tkzGetPoint{F}
	\tkzInterLL(F,C)(E,B) \tkzGetPoint{N}
	\tkzDrawSegments(A,C B,D E,F F,C)
	\tkzMarkRightAngles(A,D,B A,C,B E,F,B) 	
	\tkzMarkAngles[size=0.7, fill=green, opacity=0.5](D,C,E E,B,F)
	\tkzMarkAngles[size=0.5, fill=green, opacity=0.5](E,C,F)
	\tkzDrawPoints(A,B,C,D,E,F,N) 
	\tkzLabelPoints[above](C,D,E,N)
	\tkzLabelPoints[below](F) 
	\tkzLabelPoints[left](A)
	\tkzLabelPoints[right](B)
	\end{tikzpicture}}	
b) Xét tứ giác $ABCD$ có $\widehat{DCA}=\widehat{ABD}$ $\color{red}{(1)}$ do cùng chắn cung $AD$.\\
Chứng minh tương tự câu a) cho tứ giác $FECB$ thì tứ giác nãy cũng nội tiếp đường tròn $\Rightarrow \widehat{ACF}=\widehat{ABD}$ $\color{red}{(2)}$ do cùng chắn cung $EF$.\\
Từ $\color{red}(1)$ và ${\color{red}(2)}\Rightarrow \widehat{DCA}=\widehat{ACF}\Rightarrow CA$ là tia phân giác của $\widehat{DCF}$.\\
Xét $\triangle DCN$ có $CE$ là đường phân giác của góc $C$. Theo tính chất đường phân giác $\Rightarrow \dfrac{CN}{CD}=\dfrac{EN}{ED}$ $\color{red}(3)$.\\
Mặt khác, ta có $CA\perp CB$ ($\triangle ABC$ nội tiếp đường tròn có đường kính $AB$) mà $CA$ là tia phân giác trong $\Rightarrow CB$ là tia phân giác ngoài của $\triangle DCN$, theo tính chất của tia phân giác ngoài $\Rightarrow \dfrac{BN}{BD}=\dfrac{CN}{CD}$ $\color{red}(4)$.\\
Từ $\color{red}(3)$ và ${\color{red}(4)} \Rightarrow \dfrac{EN}{ED}=\dfrac{BN}{BD}\Rightarrow BN.ED=EN.BD$ (đpcm).	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Hưng Yên, 2017]%[9D5G1]
	Cho hai số thực dương $x,y$ thỏa mãn điều kiện $x+y\leq 4$. Tìm giá trị nhỏ nhất của biểu thức $P=\dfrac{2}{x^2+y^2}+\dfrac{35}{xy}+2xy.$
\loigiai{
	Ta có: 
	\begin{align*}
	P &=\dfrac{2}{x^2+y^2}+\dfrac{35}{xy}+2xy  \\
	&=\dfrac{2}{x^2+y^2}+\dfrac{x^2+y^2}{32}+\dfrac{35}{xy}+\dfrac{35xy}{16}+2xy-\dfrac{x^2+y^2}{32}-\dfrac{35xy}{16}\\
	&=\dfrac{2}{x^2+y^2}+\dfrac{x^2+y^2}{32}+\dfrac{35}{xy}+\dfrac{35xy}{16}-\dfrac{x^2+y^2}{32}-\dfrac{xy}{16}-\dfrac{xy}{8}\\
	&=\dfrac{2}{x^2+y^2}+\dfrac{x^2+y^2}{32}+\dfrac{35}{xy}+\dfrac{35xy}{16}-\dfrac{(x+y)^2}{32}-\dfrac{xy}{8}
	\end{align*}
	Sử dụng Cauchy cho các cặp sau $\dfrac{2}{x^2+y^2};\dfrac{x^2+y^2}{32}$ và $\dfrac{35}{xy};\dfrac{35xy}{16}$, ta có:\\
	\begin{align*}
	&\dfrac{2}{x^2+y^2}+\dfrac{x^2+y^2}{32}\geq 2\sqrt{\dfrac{2}{32}}=\dfrac{1}{2}\\
	&\dfrac{35}{xy}+\dfrac{35xy}{16}\geq 2\sqrt{\dfrac{35^2}{16}}=\dfrac{35}{2}
	\end{align*}
	Mặt khác $x+y\leq 4\Rightarrow x.y\leq 4$ nên $ \dfrac{xy}{8}\leq\dfrac{1}{2}$ , $\dfrac{(x+y)^2}{32}\leq\dfrac{1}{2}$.\\
	Vậy ta có $\min P=\dfrac{1}{2}+\dfrac{35}{2}-\dfrac{1}{2}-\dfrac{1}{2}=17$. 
	Dấu "$=$" xảy ra $\Leftrightarrow x=y=2$ .\\
	\\
	Cách khác:\begin{align*}
	P &=\dfrac{2}{x^2+y^2}+\dfrac{35}{xy}+2xy \\
	&=\dfrac{2}{x^2+y^2}+\dfrac{2}{2xy}+\dfrac{34}{xy}+\dfrac{17xy}{8}-\dfrac{xy}{8}\\
	&=2\left(\dfrac{1}{x^2+y^2}+\dfrac{1}{2xy}\right) +\left(\dfrac{34}{xy}+\dfrac{17xy}{8}\right)-\dfrac{xy}{8}\\
	&\geq \dfrac{2.4}{x^2+y^2+2xy}+2\sqrt{\dfrac{34.17}{8}}-\dfrac{(x+y)^2}{32}\\
	&\geq \dfrac{8}{16}+17-\dfrac{1}{2}= 17
	\end{align*}
	Dấu "$=$" xảy ra $\Leftrightarrow x=y=2$ .
}
\end{ex}
