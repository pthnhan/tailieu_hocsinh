\begin{bt}%[Thi HK1 Phan Đăng Lưu, HCM 2018]%[Lê Quốc Hiệp, dự án 10HK1-HCM]%[0D2B1-2]
	Tìm tập xác định của hàm số $y=\dfrac{3-x}{\sqrt{2x+1}}-\sqrt{1-x}$.
	\loigiai{Điều kiện xác định của hàm số đã cho là
		$$\heva{&2x+1>0\\&1-x\ge0}\Leftrightarrow\heva{&x>-\dfrac{1}{2}\\&x\le1}\Leftrightarrow -\dfrac{1}{2}<x\le1.$$
		Vậy tập xác định của hàm số đã cho là $\mathscr{D}=\left(-\dfrac{1}{2};1\right]$.}
\end{bt}

\begin{bt}%[Thi HK1 Phan Đăng Lưu, HCM 2018]%[Lê Quốc Hiệp, dự án 10HK1-HCM]%[0D3B2-4]%[0D3B2-2]%[0D3K2-4]
	Giải các phương trình sau
	\begin{enumerate}
		\item $\sqrt{3x^2-10x-44}+x=8$.
		\item $\left|7-4x\right|=\left|2x\right|$.
		\item $\sqrt{3x+4}-\sqrt{x-3}=3$.
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
			\item Ta có 
			\begin{eqnarray*}
				& &\sqrt{3x^2-10x-44}+x=8 \\
				& \Leftrightarrow & \sqrt{3x^2-10x-44}=8-x\\
				& \Leftrightarrow & \heva{&3x^2-10x-44=(8-x)^2\\&8-x\ge0}\\
				& \Leftrightarrow & \heva{&2x^2+6x-108=0\\&x\le8}\\
				& \Leftrightarrow & \heva{&x=-9 \text{ hoặc }x=6\\&x\le8}\\
				& \Leftrightarrow & x=-9 \text{ hoặc }x=6.
			\end{eqnarray*}
			Vậy tập hợp nghiệm của phương trình là $S=\{-9;6\}$.
			\item Ta có
			\[\left|7-4x\right|=\left|2x\right|\Leftrightarrow\hoac{&7-4x=2x\\&7-4x=-2x}\Leftrightarrow\hoac{&6x=7\\&2x=7}\Leftrightarrow\hoac{&x=\dfrac{7}{6}\\&x=\dfrac{7}{2}.}\]
			Vậy tập hợp nghiệm của phương trình là $S=\left\lbrace\dfrac{7}{6};\dfrac{7}{2} \right\rbrace$.
			\item Điều kiện xác định của phương trình là 
			\[\heva{&3x+4\ge0\\&x-3\ge0}\Leftrightarrow\heva{&x\ge-\dfrac{4}{3}\\&x\ge3}\Leftrightarrow x\ge3.\]
			Phương trình đã cho tương đương với 
			\begin{eqnarray*}
				& & \sqrt{3x+4}=3+\sqrt{x-3}\\
				& \Leftrightarrow & 3x+4=\left(3+\sqrt{x-3}\right)^2\\
				& \Leftrightarrow & 3x+4=9+6\sqrt{x-3}+x-3\\
				& \Leftrightarrow & 3\sqrt{x-3}=x-1\\
				& \Leftrightarrow & 9(x-3)=(x-1)^2 \quad\text{(do $x\ge3$)}\\
				& \Leftrightarrow & x^2-11x+28=0\\
				& \Leftrightarrow & \hoac{&x=7 \quad\text{(nhận)}\\&x=4.\quad\text{(nhận)}}
			\end{eqnarray*}
			Vậy tập hợp nghiệm của phương trình là $S=\{4;7\}$.
		\end{enumerate}
		
	}
\end{bt}

\begin{bt}%[Thi HK1 Phan Đăng Lưu, HCM 2018]%[Lê Quốc Hiệp, dự án 10HK1-HCM]%[0D4B1-1]
	Cho $a,~b$ là các số thực, chứng minh rằng \[a^2+b^2+2\ge2(a+b). \]
	
	\loigiai
	{
		Ta có 
		\begin{eqnarray*}
			& & a^2+b^2+2\ge2(a+b)\\
			& \Leftrightarrow & a^2+b^2+2-2(a+b)\ge0\\
			& \Leftrightarrow & a^2-2a+1+b^2-2b+1\ge0\\
			& \Leftrightarrow & (a-1)^2+(b-1)^2\ge0.\qquad (1)
		\end{eqnarray*}
		Vì (1) luôn đúng nên bài toán đã được chứng minh.\\
		Dấu ``$=$'' của bài toán xảy ra khi $a=b=1$.
	}
\end{bt}

\begin{bt}%[Thi HK1 Phan Đăng Lưu, HCM 2018]%[Lê Quốc Hiệp, dự án 10HK1-HCM]%[0D3K2-5]
	Tìm $m$ để phương trình $x^2-2(m+1)x+m^2-1=0$ có hai nghiệm $x_1,~x_2$ thỏa mãn 
	\[x_1x_2+x_1+x_2=0.\]
	
	\loigiai
	{
		Ta có 
		\[\Delta'=(m+1)^2-(m^2-1)=2m+2.\]
		Để phương trình đã cho có hai nghiệm $x_1,~x_2$ thì
		\[\Delta'\ge0\Leftrightarrow m\ge-1.\qquad(1)\]
		Theo định lí Vi-ét ta có 
		\[\heva{&x_1+x_2=2(m+1)\\&x_1x_2=m^2-1.}\]
		Do đó, 
		\begin{eqnarray*}
			& &x_1x_2+x_1+x_2=0 \\
			& \Leftrightarrow & m^2-1+2(m+1)=0 \\
			& \Leftrightarrow & m^2+2m+1=0 \\
			& \Leftrightarrow & m=-1 \text{ (thỏa mãn (1)).}
		\end{eqnarray*}
		Vậy giá trị $m=-1$ thỏa mãn đề bài.
	}
\end{bt}

\begin{bt}%[Thi HK1 Phan Đăng Lưu, HCM 2018]%[Lê Quốc Hiệp, dự án 10HK1-HCM]%[0H2B2-1]%[0H1B4-3]
	Trong mặt phẳng tọa độ $Oxy$, cho tam giác $ABC$ có $A(-2;2),~B(2;1)$ và $C(-1;-2)$.
	\begin{enumerate}
		\item Chứng minh $\triangle ABC$ cân, tính chu vi $\triangle ABC$.
		\item Tìm tọa độ điểm $D$ để tứ giác $ABCD$ là hình bình hành.
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
			\item Ta có 
			\begin{eqnarray*}
				& AB&=\sqrt{(2+2)^2+(1-2)}=\sqrt{17},\\
				& AC&=\sqrt{(-1+2)^2+(-2-2)^2}=\sqrt{17}, \\
				& BC&=\sqrt{(-1-2)^2+(-2-1)^2}=3\sqrt{2}.
			\end{eqnarray*}
			Do đó tam giác $ABC$ cân tại $A$ và có chu vi bằng
			\[AB+BC+AC=\sqrt{17}+3\sqrt{2}+\sqrt{17}=2\sqrt{17}+3\sqrt{2}.\]
			\item 
			\immini
			{
				Gọi $D(x;y)$, vì $ABCD$ là hình bình hành nên ta có 
				\[\vec{AD}=\vec{BC}\qquad(1).\]
				Ta có
				\begin{eqnarray*}
					&\vec{AD}&=(x+2;y-2),\\
					&\vec{BC}&=(-3;-3).
				\end{eqnarray*}
			}
			{\begin{tikzpicture}[line cap=round,line join=round,x=1.0cm,y=1.0cm,>=stealth,scale=0.75]
				\tkzDefPoints{-2/2/A,2/1/B,-1/-2/C}
				\coordinate (D) at ($(A)+(C)-(B)$);
				\fill (D) node[shift={(180:2ex)}]{$D$} circle(1pt);
				\fill (A) node[shift={(90:2ex)}]{$A$} circle(1pt);
				\fill (C) node[shift={(-90:2ex)}]{$C$} circle(1pt);
				\fill (B) node[shift={(0:2ex)}]{$B$} circle(1pt);
				\tkzDrawSegments(A,B D,C)
				\draw[->] (A)--(D);
				\draw[->] (B)--(C);
				\end{tikzpicture}}
			\noindent
			Từ (1) ta có hệ phương trình 
			\[\heva{&x+2=-3\\&y-2=-3}\Leftrightarrow\heva{&x=-5\\&y=-1.}\]
			Vậy $D(-5;-1)$.
		\end{enumerate}
	\vspace{-3ex}
	}
\end{bt}

\begin{bt}%[Thi HK1 Phan Đăng Lưu, HCM 2018]%[Lê Quốc Hiệp, dự án 10HK1-HCM]%[0H2B3-1]%[0H2B3-1]%[0H2B3-1]
	Cho tam giác $ABC$ có $AB=8,~AC=5$ và $BC=7$. Tính
	\begin{enumerate}
		\item Số đo góc $\widehat{BAC}$.
		\item Diện tích $\triangle ABC$. Đường cao $AH$ của $\triangle ABC$.
		\item Bán kính đường tròn ngoại tiếp $\triangle ABC$.
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
				\item Ta có 
				\immini{
					\vspace{-5ex}
				\begin{eqnarray*}
					&\cos A&=\dfrac{AC^2+AB^2-BC^2}{2\cdot AC\cdot AB}\\
					& &= \dfrac{5^2+8^2-7^2}{2\cdot5\cdot8}\\
					& &= \dfrac{1}{2}.
				\end{eqnarray*}}
			{\begin{tikzpicture}[line cap=round,line join=round,x=1.0cm,y=1.0cm,>=stealth,scale=0.75]
				\tkzDefPoints{0/0/A}
				\tkzDefShiftPoint[A](0:8){B}
				\tkzInterCC[R](A,5 cm)(B,7 cm) \tkzGetPoints{C}{}
				\coordinate (H) at ($(B)!(A)!(C)$);
				\tkzMarkAngle[size=.5](B,A,C)
				\draw (A)--(H);
				\draw (A)--(B) node[below,pos=0.5] {$8$};
				\draw (A)--(C) node[left,pos=0.5] {$5$};
				\draw (B)--(C) node[right,pos=0.5] {$7$};
				\tkzMarkRightAngles[size=0.2](A,H,B)
				\tkzLabelPoints[](A,B)
				\tkzLabelPoints[above](C,H)
				\tkzDrawPoints[fill=black](A,B,C,H)
				\end{tikzpicture}}
				\vspace{-3.5cm}	
				Suy ra $\widehat{BAC}=\widehat{A}=60^\circ$. 
				\item Ta có diện tích $S$ của tam giác $ABC$ là
				\[S=\dfrac{1}{2}AC\cdot AB \cdot \sin A=\dfrac{1}{2}\cdot5\cdot8\cdot\sin60^\circ=10\sqrt{3}.\]
				Ta có 
				\allowdisplaybreaks
				\begin{eqnarray*}
					& &S=\dfrac{1}{2}AH\cdot BC \\
					& \Leftrightarrow & AH=\dfrac{2S}{BC}\\
					& \Leftrightarrow & AH=\dfrac{2\cdot10\sqrt{3}}{7}=\dfrac{20\sqrt{3}}{7}.
				\end{eqnarray*}
				Vậy $AH=\dfrac{20\sqrt{3}}{7}$.
				\item Gọi $R$ là bán kính đường tròn ngoại tiếp tam giác $ABC$.\\
				Ta có 
				\allowdisplaybreaks
				\begin{eqnarray*}
					& &S=\dfrac{AB\cdot AC \cdot BC}{4R}\\
					& \Leftrightarrow & R=\dfrac{AB\cdot AC \cdot BC}{4S}\\
					& \Leftrightarrow & R=\dfrac{8\cdot 5 \cdot 7}{4\cdot10\sqrt{3}}=\dfrac{7\sqrt{3}}{3}.
				\end{eqnarray*}
				Vậy $R=\dfrac{7\sqrt{3}}{3}$.
		\end{enumerate}
			
	}
\end{bt}
