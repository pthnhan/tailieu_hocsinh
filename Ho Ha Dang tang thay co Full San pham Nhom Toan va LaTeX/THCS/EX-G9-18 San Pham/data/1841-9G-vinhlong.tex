\begin{ex}%[Đề thi HSG 9, tỉnh Vĩnh Long, năm 2018]%[Nguyễn Bình Nguyên, Dự án EX-9G-18]%[9D0B1]
\begin{enumerate}
	\item Tính giá trị của biểu thức $ A=\left(\dfrac{3+\sqrt{5}}{2}\right)^3+\left(\dfrac{3-\sqrt{5}}{2}\right)^3 $.
	\item Cho biểu thức $ P=\dfrac{\sqrt{x}-3}{\sqrt{x}-2}-\dfrac{2\sqrt{x}-1}{\sqrt{x}-1}+\dfrac{x-2}{x-3\sqrt{x}+2} $. Tìm $ x $ để $ P\geq 2 $.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Áp dụng hằng đẳng thức $ a^3+b^3=(a+b)^3-3ab(a+b) $ ta tính được \begin{align*}
	A&=\left(\dfrac{3+\sqrt{5}}{2}+\dfrac{3-\sqrt{5}}{2}\right)^3-3\cdot\dfrac{3+\sqrt{5}}{2}\cdot \dfrac{3-\sqrt{5}}{2}\cdot\left(\dfrac{3+\sqrt{5}}{2}+\dfrac{3-\sqrt{5}}{2}\right)\\
	&=3^3-3\cdot\dfrac{9-5}{4}\cdot 3=27-9=18.
	\end{align*}  
	Vậy $ A=18 $.
	\item Điều kiện $ \heva{&x\geq 0\\ & x\neq 4\\ &x\neq 1.} $\\
	Khi đó, $\begin{aligned}[t]
	P&=\dfrac{\left(\sqrt{x}-3\right)\left(\sqrt{x}-1\right)-\left(2\sqrt{x}-1\right)\left(\sqrt{x}-2\right)+x-2}{\left(\sqrt{x}-2\right)\left(\sqrt{x}-1\right)}\\
	&=\dfrac{x-4\sqrt{x}+3-2x+5\sqrt{x}-2+x-2}{\left(\sqrt{x}-2\right)\left(\sqrt{x}-1\right)}\\
	&=\dfrac{\sqrt{x}-1}{\left(\sqrt{x}-2\right)\left(\sqrt{x}-1\right)}\\
	&=\dfrac{1}{\sqrt{x}-2}.
	\end{aligned}$\\
	Nếu $ \sqrt{x}-2<0 $ thì $ P=\dfrac{1}{\sqrt{x}-2}<0 $. Do đó,  \begin{align*}
	P\geq 2& \Leftrightarrow\heva{& \sqrt{x}-2>0\\ & \dfrac{1}{\sqrt{x}-2}\geq 2}\Leftrightarrow\heva{&\sqrt{x}>2\\& \sqrt{x}-2\leq\dfrac{1}{2}}\\
	& \Leftrightarrow\heva{&\sqrt{x}>2\\ &\sqrt{x}\leq\dfrac{5}{2}}\Leftrightarrow\heva{& x>4\\ & x\leq \dfrac{25}{4}}\Leftrightarrow 4<x\leq\dfrac{25}{4}.
	\end{align*}
	Đối chiếu với điều kiện, giá trị $ x $ cần tìm là $ 4<x\leq\dfrac{25}{4} $.
\end{enumerate}	
}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Vĩnh Long, năm 2018]%[Nguyễn Bình, Dự án EX-9G-18]%[9D0G2]
\begin{enumerate}
	\item Giải phương trình $ \dfrac{x^2+2x+7}{x^2+2x+3}=x^2+2x+4 $.\\
	\item Giải hệ phương trình $ \heva{& x^2-xy+y^2=1\\ & x^2+2xy-y^2-3x-y=-2.} $
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có $ x^2+2x+3=(x+1)^2+2\geq 2\ \forall x\in\mathbb{R} $.\\
	Đặt $ t= x^2+2x+3$, điều kiện $ t\geq 2 $. Phương trình đã cho trở thành
	\begin{align*}
	\dfrac{t+4}{t}=t+1\Leftrightarrow t+4=t^2+t\Leftrightarrow t^2=4\Leftrightarrow\hoac{& t=2\\ &t=-2\ (\text{loại, vì}\ t\geq 2).}
	\end{align*}
	Với $ t=2 $ thì $ x^2+2x+3=2\Leftrightarrow x^2+2x+1=0\Leftrightarrow x=-1 $.\\
	Vậy phương trình đã cho có duy nhất nghiệm $x=-1$.
	\item Giải hệ phương trình $ \heva{& x^2-xy+y^2=1 & (1)\\ & x^2+2xy-y^2-3x-y=-2. &(2)} $\\
	Cộng phương trình $ (1) $ với phương trình $ (2) $ vế theo vế ta được \begin{align*}
	&\;2x^2+xy-3x-y=-1\\
	\Leftrightarrow &\; (2x^2-2x)+(xy-y)-(x-1)=0\\
	\Leftrightarrow &\; (x-1)(2x+y-1)=0\\
	\Leftrightarrow &\;\hoac{& x=1\\ & y=1-2x.}
	\end{align*}
	Với $ x=1 $ thì từ $ (1) $ ta có $ y^2-y=0\Leftrightarrow\hoac{& y=0\\ & y=1.} $\\
	Với $ y=1-2x $ thì từ $ (1) $ ta có  $ 7x^2-5x=0  \Leftrightarrow\hoac{& x=0&&\Rightarrow y=1\\ &x=\dfrac{5}{7}&&\Rightarrow y=-\dfrac{3}{7}.} $\\
	Vậy hệ đã cho có 4 nghiệm $ \heva{&x=1\\& y=0};\heva{&x=1\\& y=1};\heva{&x=0\\& y=1};\heva{&x=\dfrac{5}{7}\bigskip\\& y=-\dfrac{3}{7}} $.
\end{enumerate}		
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Vĩnh Long, năm 2018]%[Nguyễn Bình, Dự án EX-9G-18]%[9D0K2]
Cho phương trình $ x^2-5x+m+4=0$ ($ m $ là tham số). Tìm các giá trị của $ m $ để phương trình có hai nghiệm phân biệt $ x_1, x_2 $ và thỏa mãn $ x_1\left(1-3x_2\right)+x_2\left(1-3x_1\right)=m^2-23 $.
\loigiai{
Ta có $ \Delta=25-4(m+4)=9-4m $.\\
Phương trình có hai nghiệm phân biệt khi và chỉ khi $ \Delta>0\Leftrightarrow 9-4m>0\Leftrightarrow m<\dfrac{9}{4} $.\\
Khi đó, theo Vi-et thì $ \heva{& x_1+x_2=5\\ & x_1x_2=m+4.} $\\
Do đó, $ \begin{aligned}[t]
 &\; x_1\left(1-3x_2\right)+x_2\left(1-3x_1\right)=m^2-23\\
\Leftrightarrow &\; x_1+x_2-6x_1x_2=m^2-23\\
\Leftrightarrow  &\; 5-6(m+4)=m^2-23\\
\Leftrightarrow &\; m^2+6m-4=0\\
\Leftrightarrow  &\; m=-3\pm \sqrt{13} \ \left(\text{thỏa mãn điều kiện}\ m<\dfrac{9}{4} \right).
\end{aligned} $\\
Vậy giá trị $ m $ cần tìm là $ m=-3\pm \sqrt{13} $.	
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Vĩnh Long, năm 2018]%[Nguyễn Bình, Dự án EX-9G-18]%[9D0K7]
Với mọi số thực $ x, y, z $.
\begin{enumerate}
	\item Chứng minh rằng $ 3\left(x^2+y^2+z^2\right)\geq (x+y+z)^2 $.
	\item  Tìm giá trị nhỏ nhất của $ T=x^4+y^4+z^4 $ với $ xy+yz+zx=1 $.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có $ \begin{aligned}[t]
	&\; 3\left(x^2+y^2+z^2\right)\geq (x+y+z)^2\\
	\Leftrightarrow &\; 3x^2+3y^2+3z^2\geq x^2+y^2+z^2+2xy+2yz+2zx\\
	\Leftrightarrow &\; (x-y)^2+(y-z)^2+(z-x)^2\geq 0\ (\text{luôn đúng}).
	\end{aligned} $\\
	Đẳng thức xảy ra khi và chỉ khi $ x=y=z $.
	\item Từ $\begin{aligned}[t]
	&\; (x-y)^2+(y-z)^2+(z-x)^2\geq 0\\
	\Leftrightarrow &\; 2x^2+2y^2+2z^2-2xy-2yz-2zx\geq 0\\
	\Leftrightarrow &\; x^2+y^2+z^2\geq xy+yz+zx.
	\end{aligned}  $\\
	Đẳng thức xảy ra khi và chỉ khi $ x=y=z $.\\
	Do đó, $ T=x^4+y^4+z^4\geq x^2y^2+y^2z^2+z^2x^2\geq\dfrac{1}{3}(xy+yz+zx)^2=\dfrac{1}{3} $.\\
	Đẳng thức xảy ra khi $ x=y=z=\dfrac{\sqrt{3}}{3} $ hoặc $ x=y=z=-\dfrac{\sqrt{3}}{3} $ .\\
	Vậy giá trị nhỏ nhất của $ T $ là $ T_{\min}=\dfrac{1}{3} $ khi $ x=y=z=\pm\dfrac{\sqrt{3}}{3} $.
\end{enumerate}		
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Vĩnh Long, năm 2018]%[Nguyễn Bình, Dự án EX-9G-18]%[9D1K4]
	\begin{enumerate}
		\item Tìm tất cả các số nguyên dương $ m $ sao cho $ 70+4n-n^2 $ là số chính phương.
		\item Cho $ 5 $ số tự nhiên phân biệt sao cho tổng của ba số bất kỳ trong chúng lớn hơn tổng của hai số còn lại và tất cả $ 5 $ số đã cho đều không nhỏ hơn $ 5 $. Tìm tất cả các bộ gồm $ 5 $ số thỏa mãn đề bài mà tổng của chúng nhỏ hơn $ 40 $.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Đặt $ 70+4n-n^2=k^2 $, với $ k \in \mathbb{Z}$. Ta có $ 74=k^2+n^2-4n+4 \Leftrightarrow k^2+\left(n-2\right)^2=74$.\\
		Suy ra $\heva{&0\leq k^2\\&(n-2)^2\leq 74^2}$ với $ k \in \mathbb{Z}$ và $ n \in \mathbb{N^*}$.\\
		Các số chính phương bé hơn $ 74 $ là $ 0$, $1$, $4$, $9$, $16$, $25$, $36$, $49$, $64$.\\
		Vì $ k^2+(n-1)^2=74 $ nên ta có các trường hợp sau
		\begin{itemize}
			\item Trường hợp 1: $ \heva{&k^2=49\\&(n-2)^2=25}\Leftrightarrow \heva{&k=7\\&n=7}$, nhận.
			\item Trường hợp 2: $ \heva{&k^2=25\\&(n-2)^2=49}\Leftrightarrow \heva{&k=5\\&n=9}$, nhận.
		\end{itemize}
	\item Gọi $a$, $b$, $c$, $d$, $e$ là các số đã cho và giải sử $ a<b<c<d<e$.
	\begin{itemize}
		\item Nếu $ a\geq 6$ thì $b\geq a+1\geq 7 \Rightarrow c\geq 8 \Rightarrow d \geq9 \Rightarrow e\geq10$. Từ đó ta có $ a+b+c+d+e\geq 40$, (mâu thuẫn).
		\item Nếu $ a<6 $, mà $ a\geq5$ thì $a=5$. Ta có $ 5+b+c\geq d+e+1\Rightarrow b+c \geq d+e-4 $.\\
		Mà $ d-2\geq b $ và $ e-2\geq c$ suy ra $ d+e-4 \geq b+c $.\\
		Do đó $ \heva{&b=d-2\\&c=e-2}\Rightarrow a+b+c+d+e=5+2b+2c+4<40$\\
		$\Rightarrow b+c <\dfrac{31}{2}\Rightarrow b+(b+1)\leq b+c\leq \dfrac{31}{2} \Rightarrow b\leq 7.$ Suy ra $b=6$ hoặc $b=7$.\\
		Nếu $ b=6 $ thì $ d=b+2=8 $. Vì $ b<c<d $ nên $ c=7\Rightarrow e=c+2=9 $.\\
		Nếu $ b=7$ thì $ d=b+2=9 $. Vì $ b<c<d $ nên $ c=8\Rightarrow e=c+2=10$.
	\end{itemize}
Vậy có hai bộ số thoả mãn yêu cầu bài toán là $ \left(5;6;7;8;9 \right)$ và $ \left(5;7;8;9;10 \right) $. 
	\end{enumerate}	
		
	}
\end{ex}
