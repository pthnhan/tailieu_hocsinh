\section{HỆ PHƯƠNG TRÌNH VỚI NGHIỆM NGUYÊN}

\begin{vd}\label{hpt1}
    Tìm nghiệm nguyên của hệ phương trình
    \begin{center}
    
    $\heva{&x+y+z=3\\&x^3+y^3+z^3=3.}$
    \end{center}
    \loigiai{
        Sử dụng hằng đẳng thức $(x+y+z)^3-(x^3+y^3+z^3)=3(x+y)(y+z)(z+x)$ ta có
        $$27-3=3(x+y)(y+z)(z+x) \Leftrightarrow 8=(x+y)(y+z)(z+x)$$
        Đặt $x+y=c,y+z=a,z+x=b$ thì $abc=8$.\\
        Giả sử $x\le y\le z$ thì $a\ge b\ge c$.\\
        Ta có $a+b+c=2(x+y+z)=6$ nên $a\ge 2$.\\
        Ta lại có $abc=8$ nên $a\in \{2;4;6\}$.\\
        Xét các trường hợp sau:
        \begin{enumerate}
        \item[•] Với $a=2$ ta có $\heva{&b+c=4\\&bc=4}$\\
        Suy ra $b=c=2$. Ta được $x=y=z=1$.
        \item[•] Với $a=4$ ta có $\heva{&b+c=2\\&bc=2}$\\
        Không có nghiệm nguyên.
        \item[•] Với $a=8$ ta có $\heva{&b+c=-2\\&bc=1}$\\
        Suy ra $b=c=-1$. Ta được $x=-5;y=4;z=4$.\\
        \textit{Đáp số}: Nghiệm $(x;y;z)$ là $(1;1;1),(-5;4;4),(4;4;-5),(4;-5;4)$.
        
        \end{enumerate}
        
    }
\end{vd}

\begin{center}
    \textbf{BÀI TẬP}
\end{center}

\begin{bt}
    Tìm nghiệm nguyên của mỗi hệ phương trình sau:
    \begin{enumerate}
    \item $\heva{&2x+3y=8\\&5y+3z=1}$.
    \item $\heva{&3x+5y=2\\&2x-3z=4}$.
    \end{enumerate}
    \loigiai{
    \begin{enumerate}
\item $\heva{&x+y+z=3&~(1)\\&x^3+y^3+z^3=9&~(2)}$\\
Từ (1) có $y$ là số chẵn. Đặt $y=2k(k$ là số nguyên) được $x=4-3k$.\\
Thay $y=2k$ vào (2) được $10k+3z=1$ nên $z=\dfrac{1-10k}{3}=-3k+\dfrac{1-k}{3}$.\\
Đặt $\dfrac{1-k}{3}=t$ ($t$ là số nguyên) thì $k=1-3t$. Khi đó\\

$x=4-3(1-3t)=1+9t$\\
$y=2(1-3t)=2-6t$\\
$z=-3(1-3t)+t=10t-3$.

\textit{Đáp số}: Nghiệm $(x;y;z)$ là $(1+9t;2-6t;10t-3)$ với $t$ là số nguyên tùy ý.
\item $\heva{&x=15t-1\\&y=1-9t\\&z=10t-2} $ với $t$ là số nguyên tùy ý.
\end{enumerate} 
    }
\end{bt}
\begin{bt}
Tìm nghiệm nguyên của hệ phương trình $\heva{&x=\dfrac{1}{2}\left(y+\dfrac{1}{y}\right)\\&y=\dfrac{1}{2}\left(z+\dfrac{1}{z}\right)\\&z=\dfrac{1}{2}\left(x+\dfrac{1}{x}\right)}$.
\loigiai{
Biến đổi:\\
$\heva{&2x=y+\dfrac{1}{y}~(1)\\&2y=z+\dfrac{1}{z}~(2)\\&2z=x+\dfrac{1}{x}~(3)}$\\
Từ $(1)$ suy ra $\dfrac{1}{y}$ nguyên, do đó $y=1$ hoặc $y=-1$.\\
Tương tự $z=1$ hoặc $z=-1$; $x=1$ hoặc $x=-1$.\\
Chú ý rằng $x,y,z$ cùng dấu.\\
\textit{Đáp số}: Nghiệm $(x;y;z)$ là $(1;1;1),(-1;-1;-1)$.
}
\end{bt}
\begin{bt}
Tìm nghiệm nguyên của mỗi hệ phương trình sau:
\begin{enumerate}
\item $\heva{&x+y+z=3\\&x^3+y^3+z^3=9}$.
\item $\heva{&x+y+z=25\\&x^2+y^2+z^2=209}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Giải tương tự ví dụ \ref{hpt1} ở trang \pageref{hpt1}.\\
Đưa về $6=abc$ với $a=y+z,b=x+z,c=x+y$.\\
Ta tìm được $a=3;b=2;c=1$.\\
Suy ra $x=0;y=1;z=2$.\\
\textit{Đáp số}: Nghiệm $(x;y;z)$ là $(0;1;2)$ và các hoán vị của nó.
\item $\heva{&x+y+z=25&~(1)\\&x^2+y^2+z^2=209&~(2)}$\\
\textit{Cách 1}. Rút $z$ từ (1) được $z=25-x-y$.\\
Thay vào (2) được $x^2+y^2+(25-x-y)^2=209$.\\
Rút gọn được $x^2+(y-25)x+(y^2-25y+208)=0$.\hfill$(3)$\\
$\triangle \ge 0 \Leftrightarrow (y-25)^2-4(y^2-25y+208)\ge 0 \\ \Leftrightarrow -3y^2+50y-207 \ge 0\\
\Leftrightarrow \dfrac{23}{3}\le y\le 9$.\\
Do $y$ là số nguyên nên $y=8$ hoặc $y=9$.\\
$\bullet$ Với $y=8$, thay vào $(3)$ được $x^2-17x+72=0$ nên có $x_1=8;x_2=9$.\\
$\bullet$ Với $y=9$, thay vào $(3)$ được $x^2-16x+64=0$ nên có $x_3=8$.\\
\textit{Đáp số}: Nghiệm $(x;y;z)$ là $(8;8;9),(9;8;8),(8;9;8)$.\\
\textit{Cách 2}. Từ $(1)$ suy ra $(x+y+z)^2=625$.\\
Trừ đi $(2)$ được $2(xy+yz+zx)=416$.\hfill$(4)$\\
Nhân $(2)$ với $2$ được $2(x^2+y^2+z^2)=418$.\hfill$(5)$\\
Lấy $(5)$ trừ $(4)$ được $(x-y)^2+(y-z)^2+(z-x)^2=2$.\\
Ba số chính phương có tổng bằng $2$ nên tồn tại một số bằng $0$.\\
Bạn đọc tự giải tiếp.
\end{enumerate}

}
\end{bt}
\begin{bt}
Tìm nghiệm nguyên của hệ phương trình $\heva{&xy+2zt=0\\&xt-yz=1}$.
\loigiai{
Ta có $\heva{&xy+2zt=0&~(1)\\&xt-yz=1&~(2)}$.\\
Suy ra $(xy+2zt)^2+2(xt-yz)^2=2$.\\
Rút gọn được $(x^2+2z^2)(y^2+2t^2)=2$. Xét hai trường hợp:
\begin{enumerate}
\item $\heva{&x^2+2z^2=1\\&y^2+2t^2=2}$\\
Suy ra $x=\pm 1;z=0;y=0;t=\pm 1$.\\
Từ $(2)$ có $xt=yz+1=1$ nên $x$ và $t$ cùng dấu.\\
Ta được nghiệm $(x;y;z;t)$ là $(1;0;0;1)$ và $(-1;0;0;-1)$.
\item $\heva{&x^2+2z^2=2\\&y^2+2t^2=1}$\\
Suy ra $x=0;z=\pm 1;y=\pm 1;t=0$.\\
Từ $(2)$ có $yz=xt-1=-1$ nên $y$ và $z$ trái dấu.\\
Ta được nghiệm $(x;y;z;t)$ là $(0;1;-1;0)$ và $(0;-1;1;0)$.
\end{enumerate}
}
\end{bt}
