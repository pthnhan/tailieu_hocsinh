\begin{ex}%[Đề TT, Sở Tiền Giang 2018]%[Phạm Lâm, dự án 12EX-10]%[2D3G2-4]
Cho hàm số $y=f(x)>0$, $\forall x\ge 0$, thỏa mãn $\heva{f''(x)&\cdot f(x)-2[f'(x)]^2+xf^3(x)=0,\\f'(0)&=0;f(0)=1.}$ Tính $f(1).$
\choice
{$\dfrac{2}{3}$}
{$\dfrac{3}{2}$}
{\True $\dfrac{6}{7}$}
{$\dfrac{7}{6}$}
\loigiai{
	\begin{eqnarray*}
		& & f''(x)\cdot f(x)-2[f'(x)]^2+xf^3(x)=0\Leftrightarrow f''(x)\cdot f^2(x)-2[f'(x)]^2 f(x)=-xf^4(x)\\
		&\Leftrightarrow & \dfrac{f''(x)\cdot f^2(x)-2[f'(x)]^2\cdot f(x)}{f^4(x)}=-x\Leftrightarrow \left[ \dfrac{f'(x)}{f^2(x)}\right]'=-x\\
		&\Leftrightarrow &  \displaystyle\int\limits  \left[ \dfrac{f'(x)}{f^2(x)}\right]'\mathrm{\,d}x = \displaystyle\int\limits (-x)\mathrm{\,d}x \Leftrightarrow \dfrac{f'(x)}{f^2(x)}=-\dfrac{x^2}{2}+C.
	\end{eqnarray*}
	
Với $f'(0)=0$; $f(0)=1$ suy ra $C=0$ và
$$\displaystyle\int\limits_0^1 \dfrac{f'(x)}{f^2(x)} \mathrm{\,d}x=
\displaystyle\int\limits_0^1 \left(-\dfrac{x^2}{2} \right)  \mathrm{\,d}x
\Leftrightarrow \displaystyle\int\limits_0^1 \dfrac{\mathrm{\,d}(f(x))}{f^2(x)} 
=-\dfrac{x^3}{6}\bigg|_0^1
\Leftrightarrow -\dfrac{1}{f(x)}\bigg|_0^1=-\dfrac{1}{6}
\Leftrightarrow \dfrac{1}{f(0)}-\dfrac{1}{f(1)}=-\dfrac{1}{6}
$$
Suy ra $1-\dfrac{1}{f(1)}=-\dfrac{1}{6}\Leftrightarrow \dfrac{1}{f(1)}=\dfrac{7}{6}\Leftrightarrow f(1)=\dfrac{6}{7}.$
 }\end{ex}
