\begin{ex}%[Đề thi vào 10, Sở giáo dục Nam Định, 2017]%[9D1K8]
Cho biểu thức $P=\dfrac{1}{x^2-\sqrt{x}}:\dfrac{\sqrt{x}+1}{x\sqrt{x}+x+\sqrt{x}}$ với $x>0$ và $x\neq 1$.
\begin{enumerate}
\item Rút gọn biểu thức $P$.
\item Tìm các giá trị $x$ sao cho $3P=1+x$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có \\
$
\begin{aligned}
 P& =\dfrac{1}{x^2-\sqrt{x}}:\dfrac{\sqrt{x}+1}{x\sqrt{x}+x+\sqrt{x}}\\
 & =\dfrac{1}{\sqrt{x}(x\sqrt{x}-1)}.\dfrac{x\sqrt{x}+x+\sqrt{x}}{\sqrt{x}+1}\\
& =\dfrac{1}{\sqrt{x}(\sqrt{x}-1)(x+\sqrt{x}+1)}.\dfrac{\sqrt{x}(x+\sqrt{x}+1)}{\sqrt{x}+1}=\dfrac{1}{x-1}.
\end{aligned}$
\item $3P=1+x\Leftrightarrow \dfrac{3}{x-1}=1+x\Leftrightarrow x^2-1=3\Leftrightarrow x^2=4\Leftrightarrow x=2$ vì $x>0$ và $x\neq 1$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Nam Định, 2017]%[9D4K4]%[9D4K6]
Cho phương trình $x^2-x+m+1=0$ $(1)$ ($m$ là tham số)
\begin{enumerate}
\item Tìm các giá trị của $m$ để phương trình $(1)$ có hai nghiệm phân biệt.
\item Gọi $x_1$, $x_2$ là hai nghiệm phân biệt của phương trình $(1)$. Tìm các giá trị của $m$ sao cho $x_1^2+x_1x_2+3x_2=7$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Có $\Delta=-4m-3$. Phương trình có hai nghiệm phân biệt $\Leftrightarrow \Delta>0\Leftrightarrow m<-\dfrac{3}{4}$.
\item Áp dụng hệ thức vi-ét, ta có $\heva{& x_1+x_2=1\\ & x_1.x_2=m+1}$.\\
Khi đó 
$ x_1^2+x_1x_2+3x_2=7\Leftrightarrow x_1(x_1+x_2)+3x_2=7\Leftrightarrow x_1+3x_2=7$.\\
Từ đó ta có hệ phương trình $\heva{& x_1+x_2=1\\ & x_1+3x_2=7}\Leftrightarrow \heva{& x_1=-2\\& x_2=3}$\\ 
$\Rightarrow x_1x_2=-2.3=m+1\Rightarrow m=-7$.\\
Vậy $m=-7$ thỏa mãn điều kiện đề bài.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Nam Định, 2017]%[9D3K4]
Giải hệ phương trình $\heva{& 2x+3y=xy+5\\ & \dfrac{1}{x}+\dfrac{1}{y+1}=1}$
\loigiai{
Điều kiện $\heva{& x\neq 0\\& y\neq -1}$. Khi đó \\
$\heva{& 2x+3y=xy+5\\ & \dfrac{1}{x}+\dfrac{1}{y+1}=1}
\Leftrightarrow \heva{& 2x+3y=xy+5\\ & y+1=xy}\Leftrightarrow
\heva{& 2x+2y=6\\ & y+1=xy}\Leftrightarrow 
\heva{& x=3-y\\ & y+1=y(3-y)}\\\Leftrightarrow
\heva{& x=3-y\\ & y+1=y(3-y)}
\Leftrightarrow \heva{& x=3-y\\ & y^2-2y+1=0}\Leftrightarrow
\heva{& x=2\\& y=1}
$\\
Vậy hệ có nghiệm duy nhất $(x;y)=(2;1)$.
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Nam Định, 2017]%[9H2K2]%[9H3K3]
Cho tam giác $ABC$ vuông tại $A$ đường cao $AH$. Đường tròn tâm $E$ đường kính $BH$ cắt $AB$ tại $M$ ($M$ khác $B$), đường tròn tâm $F$ đường kính $HC$ cắt $AC$ tại $N$ ($N$ khác $C$).
\begin{enumerate}
\item Chứng minh $AM.AB=AN.AC$ và $AN.AC=MN^2$
\item Gọi $I$ là trung điểm của $EF$, $O$ là giao điểm của $AH$ và $MN$. Chứng minh $IO$ vuông góc với đường thẳng $MN$.
\item Chứng minh $4(EN^2+FM^2)=BC^2+6AH^2$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1.5]
%\tkzInit[xmin=-1,xmax=8,ymin=-3.5,ymax=5]
%\tkzClip
\tkzDefPoints{0/0/A, 0/3/B, 4/0/C}
\tkzDrawAltitude (B,C)(A)\tkzGetPoint{H}
\tkzDefMidPoint (H,B) \tkzGetPoint{E}
\tkzDrawCircle[radius](E,H)
\tkzInterLC (A,B)(E,B)\tkzGetPoints{M}{M'}
\tkzDefMidPoint (H,C)\tkzGetPoint{F}
\tkzDrawCircle[radius](F,H)
\tkzInterLC (A,C)(F,H)\tkzGetPoints{N'}{N}
\tkzDefMidPoint (E,F)\tkzGetPoint{I}
\tkzInterLL (A,H)(M,N)\tkzGetPoint{O}
\tkzDrawPoints(A, B, C, H, E, F, M, N, I, O)
\tkzDrawSegments(A,C A,B B,C M,E N,F  M,H N,H M,N I,O)
\tkzMarkRightAngles(A,H,B B,A,C)
\tkzLabelPoints[below](A, C, N, O)
\tkzLabelPoints[above](B, E, F, H, I)
\tkzLabelPoints[left](M)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Ta có $\widehat{BMH}=\widehat{HNC}=90^\circ$ (các góc nội tiếp chắn nửa đường tròn)\\
$\Rightarrow HM\perp AB$, $HN\perp AC$. Áp dụng hệ thức lượng vào các tam giác vuông $AHB$ và $AHC$, có $AH^2=AM.AB$ và $AH^2=AN.AC\Rightarrow AM.AB=AN.AC$.\\
Mặt khác, tứ giác $AMHN$ có ba góc vuông nên là hình chữ nhật khi đó $AH=MN\Rightarrow AN.AC=MN^2$.
\item Tứ giác $AMHN$ là hình chữ nhật, có $O$ là giao điểm của $AH$ và $MN$.\\
$\Rightarrow O$ là trung điểm của $AH$ và $MN$. Khi đó $\triangle EMO=\triangle EHO$ $(c.c.c)\Rightarrow \widehat{EMO}=\widehat{EHO}=90^\circ\Rightarrow EM\perp MN$.\\
Chứng minh tương tự ta có $FN\perp MN$.\\
$\Rightarrow ME\parallel NF\Rightarrow MEFN$ là hình thang vuông. Mà $OI$ là đường trung bình của hình thang vuông $MEFN$ nên $OI\perp MN$.
\item Đặt $MN=AH=h$, $x$, $y$ lần lượt là bán kính của đường tròn $(E)$ và $(F)$. Ta có\\
$4(EN^2+FM^2)=4[(ME^2+MN^2)+(ME^2+MN^2)]=4(x^2+y^2+2h^2)$\\
$BC^2+6AH^2=(HB+HC)^2+6h^2=HB^2+HC^2+2HB.HC+6h^2=4x^2+4y^2+2h^2+6h^2=4(x^2+y^2+2h^2)$.\\
Vậy $4(EN^2+FM^2)=BC^2+6AH^2$.
\end{enumerate}

}
\end{ex}


\begin{ex}%[Đề thi vào 10, Sở giáo dục Nam Định, 2017]%[9D1G7]
Giải phương trình $\sqrt{5x^2+4x}-\sqrt{x^2-3x-18}=5\sqrt{x}$.
\loigiai{
Điều kiện: $x\geq 6$.\\
$
\begin{aligned}
& \sqrt{5x^2+4x}-\sqrt{x^2-3x-18}=5\sqrt{x}\\
&\Leftrightarrow \sqrt{5x^2+4x}-5\sqrt{x}=\sqrt{x^2-3x-18}\\
&\Leftrightarrow 5x^2+4x+25x-10x\sqrt{5x+4}=x^2-3x-18\\
&\Leftrightarrow 6(5x+4)-10x\sqrt{5x+4}+4x^2+2x-6=0
\end{aligned}
$\\
Đặt $\sqrt{5x+4}=t$, Phương trình trên trở thành: $6t^2-10xt+4x^2+2x-6=0\Rightarrow\Delta =(x-6)^2\geq 0$. Khi đó 
$\hoac{& t=\dfrac{5x+|x-6|}{6}\\& t=\dfrac{5x-|x-6|}{6}}\Rightarrow\hoac{& t=x-1\\& t=\dfrac{2x+3}{3}}$\\
Với $t=x-1\Rightarrow x-1=\sqrt{5x+4}\Rightarrow x^2-7x-3=0\Leftrightarrow x=\dfrac{7+\sqrt{61}}{2}$ vì $x\geq 6$.\\
Với $t=\dfrac{2x+3}{3}\Rightarrow 2x+3=3\sqrt{5x+4}\Leftrightarrow 4x^2-33x-27=0\Leftrightarrow x=9$ vì $x\geq 6$.\\
Vậy tập nghiệm của phương trình $S=\left\{\dfrac{7+\sqrt{61}}{2};9\right\}$.
}
\end{ex}
