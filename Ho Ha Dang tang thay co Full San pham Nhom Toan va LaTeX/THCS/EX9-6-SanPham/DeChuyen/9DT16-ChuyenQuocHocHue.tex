\begin{ex}%[Đề thi vào 10, Chuyên Quốc học Huế, vòng 2, năm 2016]%[9D1K8]
\hfill

    Cho biểu thức $P = \dfrac{x^2-\sqrt{x}}{x+\sqrt{x}+1}-\dfrac{2x+\sqrt{x}}{\sqrt{x}}+\dfrac{2(x-1)}{\sqrt{x}-1}$.
    \begin{enumerate}
    \item Tìm $x$ để $P(x)$ xác định và rút gọn $P(x)$.
    \item Tìm các giá trị của $x$ để biểu thức $Q(x) = \dfrac{2\sqrt{x}}{P(x)}$ nhận giá trị nguyên.
    \end{enumerate}
    
\loigiai{
\begin{enumerate}
\item Điều kiện: $x > 0; x\ne 1$

Ta có 

$
\begin{aligned}
P &= \dfrac{\sqrt{x}(\sqrt{x}-1)(x+\sqrt{x}+1)}{x+\sqrt{x}+1} - \dfrac{\sqrt{x}(2\sqrt{x}+1)}{\sqrt{x}}+\dfrac{2(x-1)}{\sqrt{x}-1}\\
 &= x -\sqrt{x}+1.
\end{aligned}$

\item Với $Q = \dfrac{2\sqrt{x}}{x -\sqrt{x}+1}$

Với $x > 0; x\ne 1$ suy ra $\heva{2\sqrt{x}>0 \\ x-\sqrt{x}+1>0}$ \; $\Rightarrow Q > 0$.

Ta có $Q = \dfrac{2\sqrt{x}}{x -\sqrt{x}+1}
 = \dfrac{2}{\sqrt{x}+\dfrac{1}{\sqrt{x}}-1}$.

Vì $\sqrt{x}+\dfrac{1}{\sqrt{x}}-1 >1$ (với $x > 0; x\ne 1$) suy ra $Q < 1$ 

Vậy $0<Q<1$ mà $Q\in \mathbb{Z}$ $\Rightarrow Q = 1$

Với $Q = 1$ thay vào ta được $\hoac{x = \dfrac{7+3\sqrt{5}}{2}\\ x =  \dfrac{7-3\sqrt{5}}{2}}.$
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Quốc học Huế, vòng 2,  năm 2016]%[9D4K2]%[9D4G7]
\hfill

Trong mặt phẳng tọa độ $Oxy$ cho parabol $(P): y = mx^2 \; (m > 0)$ và đường thẳng $(d): y = 2x-m^2$.
\begin{enumerate}
\item Tìm $m$ để $(d)$ cắt $(P)$ tại hai điểm phân biệt $A$ và $B$. Khi đó chứng minh $A$ và $B$ cùng nằm về một phía với trục tung.
\item Với $m$ tìm được ở câu $a)$. Gọi $x_A, x_B$ theo thứ tự là hoành độ các điểm $A$ và $B$. Tìm $m$ để biểu thức $K = \dfrac{2}{x_A+x_B}+\dfrac{1}{4x_Ax_B+1}$ đạt giá trị nhỏ nhất.
\end{enumerate}

\loigiai{
\begin{enumerate}
\item Xét phương trình hoành độ giao điểm của $(d)$ và $(P)$, ta được $mx^2-2x+m^2 = 0$.

Để $d$ cắt $(P)$ tại hai điểm phân biệt thì $\Delta =  1-m^3 > 0 \Rightarrow m < 1$.

Kết hợp điều kiện $m > 0 \Rightarrow 0 < m < 1$ thì $d$ cắt $P$ tại hai điểm phân biệt

Áp dụng hệ thức vi-ét $x_1.x_2 = m > 0$. 

Vậy $(d)$ cắt $(P)$ tại hai điểm phân biệt nằm cùng 1 phía với trục tung.
\item Ta có $K = m+\dfrac{1}{4m+1} = \dfrac{1}{4}(4m+1)+\dfrac{1}{4m+1} - \dfrac{1}{4}\geq \dfrac{3}{4}$.

Dấu $=$ xảy ta khi $m = \dfrac{1}{4}$
\end{enumerate}
}
\end{ex}

\begin{enumerate}
\item Giải phương trình: $\sqrt{x^2+3x+2} +\sqrt{x^2-1}+6 = 3\sqrt{x+1}+2\sqrt{x+2}+2\sqrt{x-1}$.
\item Giải hệ phương trình: $\heva{x^3+2xy^2+12y& = 0\\ 8y^2+x^2 &= 12}$.
\end{enumerate}

\loigiai{
\begin{enumerate}
\item Đặt $a = \sqrt{x+1}; b = \sqrt{x+2}; c = \sqrt{x-1}$ ($x > 1$).

Phương trình tương đương với $ab+ac +6 = 3a+2b+2c \Leftrightarrow \hoac{a = 2\\ b+c-3 = 0}.$

suy ra $\hoac{\sqrt{x+1} = 2\\ \sqrt{x+2}+\sqrt{x-1} = 3} \Leftrightarrow \hoac{x = 3\\ x  =2.}$
\item Phương trình tương đương với $x^3+2xy^2+8y^3+x^2y = 0$.

Vì $y = 0$ không phải là nghiệm của phương trình nên ta có:
$$\dfrac{x^3}{y^3}+\dfrac{x^2}{y^2}+2\dfrac{x}{y}+8 = 0.$$

$\Rightarrow x = -2y$, thay  vào ta được $(x, y) = (-2; 1)\mbox{ hoặc } (2; -1)$.
\end{enumerate}
}

Cho hai đường tròn $(O_1)$ và $(O_2)$ có bán kính khác nhau, cắt nhau tại hai điểm $A$ và $B$ sao cho $O_1, O_2$ thuộc hai nửa mặt phẳng có bờ là đường thẳng $AB$. Đường tròn $(O)$ ngoại tiếp tam giác $BO_1O_2$ cắt $(O_1)$ và $(O_2)$ lần lượt tại $K$ và $L$ khác $A,\; B$. Đường thẳng $AO$ cắt $(O_1)$ và $(O_2)$ lần lượt tại $M$ và $N$ (khác $A$). Hai đường thẳng $MK$ và $NL$ cắt nhau tại $P$ sao cho $P$ và $B$ thuộc hai nửa mặt phẳng có bờ là đường thẳng $KL$. Chứng minh rằng
\begin{enumerate}
\item Tứ giác $BKPL$ nội tiếp đường tròn.
\item Điểm $A$ cách đều hai đường thẳng $BK$ và $BL$.
\item Điểm $P$ thuộc đường thẳng $AB$ khi và chỉ khi tam giác $PKL$ cân.
\end{enumerate}

\begin{enumerate}
\item Ta có $
\widehat{PKB} = 180^\circ - \widehat{MAB}= \widehat{BAN} = \widehat{BLN} = 180^\circ - \widehat{BLP}
$

$\Rightarrow \widehat{PKB} + \widehat{BLP} = 180^\circ$ suy ra tứ giác $BLKP$ nội tiếp.
\item Ta có 

$
\begin{aligned}
\widehat{BKO_2} & =  \widehat{BO_1O_2} \mbox{ (chắn cung }\wideparen{BO_2})\\
& = \dfrac{1}{2}\;\widehat{BOA} = \widehat{BKA}
\end{aligned}
$

$\Rightarrow K, A, O_2$ thẳng hàng.

Chứng minh tương tự ta được $L, A, O_1$ thẳng hàng.

Do đó $\widehat{KBA} = \dfrac{1}{2}\;\widehat{KO_1A} = \dfrac{180^\circ - \widehat{KAO_1}}{2}=  \dfrac{180^\circ - \widehat{LAO_2}}{2} =\dfrac{\widehat{LO_2A}}{2} = \widehat{LBA}$.

Suy ra $BA$ là phân giác $\widehat{KBL}$ hay $A$ cách đều $BK$ và $BK$.
\item Ta có 

$
			\begin{aligned}
			\widehat{PMN}&= \widehat{KBA} \left(=\dfrac{1}{2}\;\mbox{sđ}\;\wideparen{KA}\right)\\
			& = \widehat{LBA} = \widehat{PNM}\left(=\dfrac{1}{2}\;\mbox{sđ}\;\wideparen{AL}\right) 
			\end{aligned}
			$
			
Suy ra $\triangle PMN$ cân tại $P$.

Nếu $P\in AB \Rightarrow \widehat{PMA} = \widehat{PBK}  \left(=\dfrac{1}{2}\;\mbox{sđ}\;\wideparen{AK}\right) $

$\Rightarrow \triangle PMA \backsim \triangle PBK \Rightarrow PM.PK = PA.PB$.

Chứng minh tương tự $PA.PB = PN.PM$

Suy ra $PM.PK = PA.PB = PN.PM \Rightarrow \triangle PKL \backsim \triangle PNM \Rightarrow \triangle PKL$ cân.

Nếu $PK = PL \Rightarrow PK.PM = PL.PN$

Gọi $A'$ và $A"$ là giao điểm của $PB$ với $(O_1)$ và $(O_2)$.

Suy ra $PK.PM = PA'.PB = PL.PN = PA".PB$

Suy ra $A'$ trùng $A"$ trùng $A$ $\Rightarrow P\in AB$. 
\end{enumerate}
}

\begin{ex}%[Đề thi vào 10, Chuyên Quốc học Huế, vòng 2,  năm 2016]%[9D5G1]%[9D5G3]
\hfill
\begin{enumerate}
\item Cho $x, y > 0$ và $x+ y\geq 3$. Tìm giá trị nhỏ nhất của $M = 6x^2+4y^2+10xy+\dfrac{4x}{y}+\dfrac{3y}{x}+2016$.
\item Tìm các bộ số nguyên dương $(x, y, z)$ biết $\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z} = 1$ và $\sqrt{x-y+z} = \sqrt{x}-\sqrt{y}+\sqrt{z}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có 

$
\begin{aligned}
M &= \left ( x+y \right )\left ( 6x+4y+\dfrac{4}{y}+\dfrac{3}{x} \right )+2009 \\
& \geq 3(3x+\dfrac{3}{x}+y+\dfrac{4}{y}+3x+3y)+2009 \\
&\geq 3(6+4+9)+2009=2066
\end{aligned}
 $
 
 Dấu $=$ xảy ra khi $x = 1; y = 2$.
 \item $\sqrt{x-y+z}=\sqrt{x}-\sqrt{y}+\sqrt{z}\Leftrightarrow y-\sqrt{xy}-\sqrt{yz}+\sqrt{zx}=0\\ \Leftrightarrow (\sqrt{y}-\sqrt{x})(\sqrt{y}-\sqrt{z})=0$ \;  $\Leftrightarrow \hoac{y=z\\ y =x}$
 
 TH1: $x = y = z$ suy ra $(x,y,z) = (3;3;3)$.

TH2: $y = z \ne x$ ta có $\dfrac{1}{x}+\dfrac{1}{y} = 1 \Leftrightarrow (x-1)(y - 2) = 2$ suy ra $(x, y,z) = (2; 4; 4)$.

TH3: $x= y \ne z$ tìm được $(x, y,z) = (4;4;2)$.

Vậy có 3 bộ thỏa mãn $(3;3;3)(2;4:4)(4;4;2)$.
\end{enumerate}
}

\end{ex}
