\noindent \textbf{I. Phần chung}	
\begin{bt}%[HK1,Chuyên Trần Đại Nghĩa HCM,2018]%[Nguyễn Tài Tuệ]%[0D3B3-3]
	Giải phương trình $ x|x-1| =4x-6.$
	\loigiai{
\begin{itemize}
	\item Nếu $ x\ge 1 $ thì phương trình đã cho trở thành 
	$$ x^2-x=4x-6\Leftrightarrow x^2-5x+6=0\Leftrightarrow \hoac{&x=2 \text{ (thỏa mãn)}\\&x=3 \text{ (thỏa mãn)}.} $$
	\item Nếu $ x<1 $ thì phương trình đã cho trở thành 
	$$ x-x^2 =4x-6\Leftrightarrow x^2+3x -6=0\Leftrightarrow \hoac{&x=\dfrac{-3+\sqrt{33} }{2} \text{ ( loại )}\\&x=\dfrac{-3-\sqrt{33} }{2} \text{ (thỏa mãn)}.} $$
\end{itemize}
Vậy phương trình đã cho có 3 nghiệm $ x=2,x=3, x=\dfrac{-3-\sqrt{33}}{2}.$
	}
\end{bt}
\begin{bt}%[HK1,Chuyên Trần Đại Nghĩa HCM,2018]%[Nguyễn Tài Tuệ]%[0D3K3-4]
	Giải hệ phương trình $ \begin{cases} xy+x+y =-7\\ x^2 +y^2 -2x-2y =15.\end{cases} $
	\loigiai{
		Hệ phương trình đã cho tương đương với
	{\allowdisplaybreaks	\begin{align*}
		&\heva{&x+y+xy =-7\\&(x+y)^2 -2(x+y +xy) =15} \\
		\Leftrightarrow& \heva{&x+y+xy =-7\\
		&(x+y)^2 =1}\\ \Leftrightarrow& \heva{&x+y+xy =-7\\
		&\hoac{& x+y=1\\&x+y=-1}}\\
		 \Leftrightarrow& \hoac{& \heva{&x+y=1\\&xy =-8} \hfill (1)\\&\heva{&x+y=-1\\&xy=-6} \hfill(2)}
			\end{align*} }
		Xét hệ (1) $  \heva{&x+y=1\\&xy =-8} $.\\
		Ta có $ x,y $ là nghiệm của phương trình $ X^2-X-8=0 \Leftrightarrow X=\dfrac{1\pm \sqrt{33}}{2}.$\\
		Do đó $ \left (\dfrac{1+\sqrt{33}}{2},\dfrac{1-\sqrt{33}}{2}\right ), \left (\dfrac{1-\sqrt{33}}{2},\dfrac{1+\sqrt{33}}{2}\right ) $ là nghiệm của hệ phương trình.\\
		Xét hệ (2)	$\heva{&x+y=-1\\&xy=-6.}$\\
		Ta có $ x,y $ là nghiệm của phương trình $ t^2+t-6 =0\Leftrightarrow \hoac{&t=2\\&t=-3} $.\\		
		Suy ra   $(2;-3), (-3;2)$ là nghiệm của hệ phương trình là.\\
		Vậy hệ phương trình đã cho có 4 nghiệm $\left (\dfrac{1+\sqrt{33}}{2},\dfrac{1-\sqrt{33}}{2}\right ), \left (\dfrac{1-\sqrt{33}}{2},\dfrac{1+\sqrt{33}}{2}\right ),(2;-3), (-3;2).$
	}
\end{bt}

\noindent \textbf{II. Phần riêng}

\noindent\textbf{A. Tự nhiên}
\begin{bt}%[HK1,Chuyên Trần Đại Nghĩa HCM,2018]%[Nguyễn Tài Tuệ]%[0D3B3-1]
	Cho phương trình $ (x-1)(x^2 +2mx+m^2 -4)=0 $ ($ x $ là ẩn số, $ m $ là tham số). Tìm $ m $ để phương trình có $ 3 $ nghiệm phân biệt $ x_1, x_2,x_3 $ thỏa mãn điều kiện $ x_1^2+x_2^2+x_3^2 =25.$
	\loigiai{ 
	}
\end{bt}
Ta có \begin{align*}
&(x-1)(x^2 +2mx+m^2 -4)=0\\
\Leftrightarrow&\hoac{&x=1\\&x^2 +2mx+m^2-4=0 \,~~~~ (1)}
\end{align*}
Phương trình đã cho có $ 3 $ nghiệm phân biệt khi và chỉ khi phương trình $ (1) $ có hai nghiệm phân biệt khác $ 1 $\\
$ \Leftrightarrow \heva{&m^2-m^2+4>0\\&1+2m+m^2-4\ne 0}\Leftrightarrow m^2+2m-3\ne 0\Leftrightarrow \heva{& m\ne1\\&m\ne -3.} $\\
Giả sử $ x_3=1 $, khi đó ta có $ x_1,x_2 $ là nghiệm của phương trình $ (1) $ và 
\begin{align*}
&x_1^2+x_2^2+x_3^2=25\\
\Leftrightarrow & (x_1+x_2)^2 -2x_1x_2 =24\\
\Leftrightarrow & 4m^2-2(m^2-4)=24\\
\Leftrightarrow & 2m^2=16\\
 \Leftrightarrow &m=\pm 2\sqrt{2} \text{ (thỏa mãn điều kiện)}
\end{align*}
Vậy $ m=\pm 2\sqrt{2} $ là giá trị cần tìm.
\begin{bt}%[HK1,Chuyên Trần Đại Nghĩa HCM,2018]%[Nguyễn Tài Tuệ]%[0H2K2-3]
	Cho tam giác $ ABC $ đều cạnh $ a $. $ M,N $ là hai điểm thỏa mãn $ \vec{AM} =\dfrac{1}{3}\vec{AB}, \vec{NA}=-\dfrac{1}{5}\vec{AC} $. Chứng minh $ BN\perp CM.$
	\loigiai{
	\immini{
		Ta có $ \vec{BN} =\vec{AN}-\vec{AB} =\dfrac{1}{5}\vec{AC}-\vec{AB}$
		và $ \vec{CM}=\vec{AM}-\vec{AC} =\dfrac{1}{3}\vec{AB} -\vec{AC}. $\\
		Nên
		\begin{align*}
		\vec{BN}\cdot\vec{CM}=&\left(\dfrac{1}{5}\vec{AC}-\vec{AB}\right) \cdot \left(\dfrac{1}{3}\vec{AB} -\vec{AC}\right)\\
		= &\dfrac{16}{15}\vec{AB}\cdot \vec{AC} -\dfrac{1}{5}\vec{AC}^2-\dfrac{1}{3}\vec{AB}^2\\
		=&\dfrac{16}{15}a^2\cos 60^\circ -\dfrac{1}{5}a^2-\dfrac{1}{3}a^2 \\
		=&0.
		\end{align*} 
		Do đó $ BN\perp CM. $}{
		\begin{tikzpicture}

\noindent \textbf{B. XÃ HỘI}

\begin{bt}%[HK1,Chuyên Trần Đại Nghĩa HCM,2018]%[Nguyễn Tài Tuệ]%[0D3K3-3]
	Cho phương trình $ (x+5)(-x^2+2mx-m+3)=0 $ ( $ m  $ là tham số, $ x $ là ẩn số). Tìm các giá trị của $ m $ để phương trình có ba nghiệm phân biệt, trong đó chỉ có một nghiệm dương và tổng các nghiệm là số dương.
	\loigiai{
		Phương trình đã cho tương đương với $ \hoac{&x=-5\\&x^2-2mx+m-3=0 ~~~(2)} $\\
		Phương trình đã cho có $ 3 $ nghiệm phân biệt khi và chỉ khi phương trình $ (2) $ có hai nghiệm phân biệt khác $ -5 $ $ \Leftrightarrow \heva{&m^2-m+3 >0 \text{( luôn đúng)}\\&25 +10m+m-3 \ne 0}\Leftrightarrow m\ne -2. $\\
		Khi đó phương trình đã cho có đúng $ 1 $ nghiệm dương và tổng các nghiệm là số dương khi và chỉ khi phương trình $ (2) $ có hai nghiệm $ x_1,x_2 $ trái dấu  và $ x_1+x_2-5>0 $ khi và chỉ khi
		$$ \heva{&m-3<0\\&2m-5>0}\Leftrightarrow \heva{&m<3\\& m>\dfrac{5}{2}. }\Leftrightarrow \dfrac{5}{2}<m<3.\text{ (thỏa mãn )} $$
		Vậy $ m\in \left(\dfrac{5}{2};3\right).  $ 
	}
\end{bt}

\begin{bt}%[HK1,Chuyên Trần Đại Nghĩa HCM,2018]%[Nguyễn Tài Tuệ]%[0D3B3-2]
	Cho hệ phương trình $ \begin{cases}
	mx+2y=m+1\\ 2x+my =2m+5
	\end{cases}, $ với $ m $ là tham số. Tìm $ m\in \mathbb{Z} $ để hệ phương trình có nghiệm nguyên duy nhất.
	\loigiai{
		Ta có các định thức $ D=m^2-4, D_x =m^2-3m-10,D_y=2m^2+3m-2.$\\
		Hệ Phương trình đã cho có nghiệm  duy nhất khi và chỉ khi $ D=m^2-4 \ne 0 \Leftrightarrow m\ne \pm 4. $\\
		Với $ m\ne \pm 4 $ hệ phương trình đã cho có nghiệm duy nhất $ \heva{& x=\dfrac{D_x}{D}=\dfrac{m-5}{m-2}\\&y=\dfrac{D_y}{D} =\dfrac{2m-1}{m-2}} $\\
		hay $ \heva{&x=1-\dfrac{3}{m-2}\\&y=2+\dfrac{3}{m-2}.} $\\
		Hệ phương trình đã cho có nghiệm nguyên khi và chỉ khi $ m-2 \in \text{Ư}(3)$ tương đương với
		$$ \hoac{&m-2=-1\\&m-2=1\\& m-2=-3\\&m-2=3}\Leftrightarrow \hoac{&m=1\\&m=3\\& m=-1\\&m=5}.$$
		Đối chiếu với điều kiện, suy ra $ m\in \{-1,1,3,5\} $ là các giá trị cần tìm.
	}
\end{bt}


\noindent\textbf{ C. CHUYÊN TOÁN}
\begin{bt}%[HK1,Chuyên Trần Đại Nghĩa HCM,2018]%[Nguyễn Tài Tuệ]%[0D4G1-3]
	Cho $ a,b,c, d $ là bốn số dương. Chứng minh $$ \dfrac{a^2}{b+c+d}+\dfrac{b^2}{c+d+a}+\dfrac{c^2}{d+a+b}+\dfrac{d^2}{a+b+c}\ge \dfrac{2\sqrt{a^2+b^2+c^2+d^2} }{3}.$$ 
	\loigiai{
		Không mất tính tổng quát giả sử $ a\ge b\ge c\ge d $. Khi đó\\
		$ a^2 \ge b^2\ge c^2\ge d^2 $ và $ \dfrac{1}{b+c+d}\ge \dfrac{1}{c+d+a}\ge \dfrac{1}{d+a+b}\ge \dfrac{1}{a+b+c}. $\\
		Áp dụng bất đẳng thức Chebyshev và bất đẳng thức Cauchy schwarz ta có 
		\begin{align*}
		&4\cdot\left(\dfrac{a^2}{b+c+d}+\dfrac{b^2}{c+d+a}+\dfrac{c^2}{d+a+b}+\dfrac{d^2}{a+b+c}\right)  \\ 
	\ge &	\left( a^2+b^2+c^2+d^2\right)\left(\dfrac{1}{b+c+d}+ \dfrac{1}{c+d+a}+ \dfrac{1}{d+a+b}+ \dfrac{1}{a+b+c}\right)  \hfill \text{ ( theo bất đẳng thức Chebyshev) } \\
	\ge &\dfrac{16\left( a^2+b^2+c^2+d^2\right) }{3(a+b+c+d)} \hfill \text{ ( theo bất đẳng thức Cauchy schwarz) }\\
	=& \dfrac{4\sqrt{4(a^2+b^2+c^2+d^2)}\cdot \sqrt{4(a^2+b^2+c^2+d^2)} }{3(a+b+c+d)}\\
	\ge& \dfrac{4\sqrt{4(a^2+b^2+c^2+d^2)}}{3}.
		\end{align*}
		Do đó $$ \dfrac{a^2}{b+c+d}+\dfrac{b^2}{c+d+a}+\dfrac{c^2}{d+a+b}+\dfrac{d^2}{a+b+c}\ge \dfrac{2\sqrt{a^2+b^2+c^2+d^2} }{3}.$$
		Đẳng thức xảy ra khi $ a=b=c=d. $
	}
\end{bt}
