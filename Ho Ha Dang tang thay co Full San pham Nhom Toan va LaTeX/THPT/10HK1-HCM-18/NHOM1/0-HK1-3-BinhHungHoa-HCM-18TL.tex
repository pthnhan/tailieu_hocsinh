\begin{bt}%[Đề HK1, Bình Hưng Hòa, Hồ Chí Minh, 2017]%[Đỗ Đường Hiếu, 10EX-HK1]%[0D4K5-6]%[0D3B2-3]
Giải các phương trình sau
\begin{enumerate}
\item $\sqrt{x^2-2x+6}=2x-3$.
\item $x^2-3x +3\sqrt{x^2-3x+1}=3$.
\item $\dfrac{x+2}{4}-\dfrac{1}{x-2}=\dfrac{x-6}{4(x-2)}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có
\begin{eqnarray*}
\sqrt{x^2-2x+6}=2x-3 & \Leftrightarrow & \heva{&2x-3\ge 0\\& \left(\sqrt{x^2-2x+6}\right)^2=\left(2x-3\right)^2}\\
& \Leftrightarrow & \heva{&x\ge \dfrac{3}{2}\\& x^2-2x+6=4x^2-12x+9}\\
& \Leftrightarrow & \heva{&x\ge \dfrac{3}{2}\\& 3x^2-10x+3=0}\\
& \Leftrightarrow & \heva{&x\ge \dfrac{3}{2}\\& x=\dfrac{1}{3}\; \text{hoặc}\; x=3}\\
& \Leftrightarrow & x=3.
\end{eqnarray*}
Vậy, phương trình có một nghiệm duy nhất $x=3$.
\item Ta có
\begin{eqnarray*}
x^2-3x +3\sqrt{x^2-3x+1}=3 \Leftrightarrow \left(x^2-3x+1\right)+3\sqrt{x^2-3x+1}-4=0
\end{eqnarray*}
Đặt $t=\sqrt{x^2-3x+1}$ với $t\ge 0$, ta được phương trình
$$t^2+3t-4=0 \Leftrightarrow \hoac{&t=1\\&t=-4 <0 \; (\text{loại})}.$$
Với $t=1$ ta có 
\begin{eqnarray*}
\sqrt{x^2-3x+1}=1 &\Leftrightarrow & x^2-3x+1 =1\\
&\Leftrightarrow & x^2-3x =0\\
&\Leftrightarrow & x =0\; \text{hoặc} \; x=3.
\end{eqnarray*}
Vậy, tập nghiệm của phương trình là $S=\{0; 3\}$.
\item Điều kiện xác định: $x\ne 2$.\\
Với điều kiện đó, ta có\\
\begin{eqnarray*}
\dfrac{x+2}{4}-\dfrac{1}{x-2}=\dfrac{x-6}{4(x-2)} &\Leftrightarrow & \dfrac{x+2}{4}=\dfrac{x-6}{4(x-2)}+\dfrac{1}{x-2}\\
&\Leftrightarrow & \dfrac{x+2}{4}=\dfrac{x-6}{4(x-2)}+\dfrac{4}{4(x-2)}\\
&\Leftrightarrow & \dfrac{x+2}{4}=\dfrac{x-2}{4(x-2)}\\
&\Leftrightarrow & \dfrac{x+2}{4}=\dfrac{1}{4}\\
&\Leftrightarrow & x+2=1\\
&\Leftrightarrow & x=-1.
\end{eqnarray*}
Đối chiếu điều kiện, ta suy ra phương trình có nghiệm duy nhất $x=-1$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Bình Hưng Hòa, Hồ Chí Minh, 2017]%[Đỗ Đường Hiếu, 10EX-HK1]%[0D3K2-5]
Tìm tất cả các giá trị của tham số $m$ để phương trình $x^2-2x+2m+5=0$ có hai nghiệm dương phân biệt.
\loigiai{
Phương trình $x^2-2x+2m+5=0$ có hai nghiệm dương phân biệt khi và chỉ khi
\begin{eqnarray*}
\heva{&\Delta'>0\\& P>0\\& S>0} &\Leftrightarrow & \heva{&(-1)^2-1\cdot (2m+5)>0\\& 2m+5>0\\& 2>0}\\
& \Leftrightarrow & \heva{&m<-2\\& m>-\dfrac{5}{2}}\\
& \Leftrightarrow & -\dfrac{5}{2}<m<-2.
\end{eqnarray*}
Vậy, tập hợp tất cả các giá trị của tham số $m$ để phương trình $x^2-2x+2m+5=0$ có hai nghiệm dương phân biệt là $T=\left(-\dfrac{5}{2};-2\right)$.
}
\end{bt}


\begin{bt}%[Đề HK1, Bình Hưng Hòa, Hồ Chí Minh, 2017]%[Đỗ Đường Hiếu, 10EX-HK1]%[0D3K2-5]
Tìm tất cả các giá trị của tham số $m$ để phương trình $x^2+2(m-1)x+m^2-3=0$ có hai nghiệm phân biệt $x_1$, $x_2$ thỏa mãn $x_1^2+x_2^2=4$.
\loigiai{
Phương trình $x^2+2(m-1)x+m^2-3=0$ có hai nghiệm phân biệt $x_1$, $x_2$ khi và chỉ khi
$$\Delta' = (m-1)^2-1\cdot (m^2-3)>0 \Leftrightarrow -2m+4>0 \Leftrightarrow m<2\quad (*).$$
Khi đó ta có $x_1+x_2=-2(m-1)$ và $x_1x_2=m^2-3$.\\
Bởi vậy
\begin{eqnarray*}
x_1^2+x_2^2=4 &\Leftrightarrow & \left(x_1+x_2\right)^2-2x_1x_2=4\\
&\Leftrightarrow &\left[-2 \left(m-1\right)\right]^2-2(m^2-3)=4\\
&\Leftrightarrow &m^2-4m+3=0 \\
&\Leftrightarrow & \hoac{&m=1\\&m=3.}
\end{eqnarray*}
Đối chiếu với điều kiện $(*)$, ta tìm được $m=1$.
}
\end{bt}

\begin{bt}%[Đề HK1, Bình Hưng Hòa, Hồ Chí Minh, 2017]%[Đỗ Đường Hiếu, 10EX-HK1]%[0D4K1-2]
Chứng minh rằng $\dfrac{a}{bc}+\dfrac{b}{ac}+\dfrac{c}{ab}\ge \dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}$ với mọi $a, b, c >0$. 
\loigiai{
Áp dụng bất đẳng thức Cô-si cho hai số dương, ta có
\begin{eqnarray*}
&\dfrac{a}{bc}+\dfrac{b}{ac}\ge 2\sqrt{\dfrac{a}{bc}\cdot\dfrac{b}{ac}}=2\sqrt{\dfrac{1}{c^2}}=\dfrac{2}{c}\quad &(1);\\
&\dfrac{b}{ac}+\dfrac{c}{ab}\ge 2\sqrt{\dfrac{b}{ac}\cdot\dfrac{c}{ab}}=2\sqrt{\dfrac{1}{a^2}}=\dfrac{2}{a}\quad &(2);\\
&\dfrac{c}{ab}+\dfrac{a}{bc}\ge 2\sqrt{\dfrac{c}{ab}\cdot\dfrac{a}{bc}}=2\sqrt{\dfrac{1}{b^2}}=\dfrac{2}{b}\quad &(3).
\end{eqnarray*}
Cộng vế với vế các bất đẳng thức $(1)$, $(2)$ và $(3)$ và rút gọn, ta được
 $$\dfrac{a}{bc}+\dfrac{b}{ac}+\dfrac{c}{ab}\ge \dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}.$$
Dấu đẳng thức xảy ra khi và chỉ khi
\begin{eqnarray*}
\heva{&\dfrac{a}{bc}=\dfrac{b}{ac}\\&\dfrac{b}{ac}=\dfrac{c}{ab}\\&\dfrac{c}{ab}=\dfrac{a}{bc}}\Leftrightarrow a=b=c.
\end{eqnarray*}
}
\end{bt}

\begin{bt}%[Đề HK1, Bình Hưng Hòa, Hồ Chí Minh, 2017]%[Đỗ Đường Hiếu, 10EX-HK1]%[0H1B4-5]%[0H2K2-3]%[0H2K2-2]
Trong mặt phẳng hệ tọa độ $Oxy$, cho ba điểm $A(-1;3)$, $B(-4;-5)$ và $C(1;-2)$.
\begin{enumerate}
\item Chứng tỏ $A$, $B$, $C$ là ba đỉnh của một tam giác và tìm tọa độ trọng tâm $G$ của tam giác $ABC$.
\item Tìm tọa độ trực tâm $H$ của tam giác $ABC$.
\item Tìm tọa độ điểm $M$ thuộc trục hoành sao cho $\left|2\overrightarrow{MA}+\overrightarrow{MC}\right|$ đạt giác trị nhỏ nhất.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $\overrightarrow{AB}=(-3;-8)$, $\overrightarrow{AC}=(2;-5)$.\\
Vì $\dfrac{-3}{2}\ne \dfrac{-8}{-5}$ nên hai véc-tơ $\overrightarrow{AB}$ và $\overrightarrow{AC}$ không cùng phượng. Từ đó suy ra  $A$, $B$, $C$ là ba đỉnh của một tam giác.\\
Tọa độ trọng tâm $G$ của tam giác $ABC$ là
\begin{eqnarray*}
\heva{&x_G=\dfrac{x_A+x_B+x_C}{3}=\dfrac{(-1)+(-4)+(1)}{3}=-\dfrac{4}{3}\\&y_G=\dfrac{y_A+y_B+y_C}{3}=\dfrac{(3)+(-5)+(-2)}{3}=-\dfrac{4}{3}}
\end{eqnarray*}
Vậy: $G\left(-\dfrac{4}{3};-\dfrac{4}{3}\right)$.
\item Giả sử $H(x;y)$ là trực tâm của tam giác $ABC$.\\
Ta có $\overrightarrow{BH}=(x+4;y+5)$, $\overrightarrow{CH}=(x-1;y+2)$, $\overrightarrow{AB}=(-3;-8)$, $\overrightarrow{AC}=(2;-5)$.\\
Vì $H$ là trực tâm tam giác $ABC$ nên
\begin{eqnarray*}
\heva{&\overrightarrow{AB}\perp\overrightarrow{CH}\\&\overrightarrow{AC}\perp\overrightarrow{BH}}
&\Leftrightarrow & \heva{&\overrightarrow{AB}\cdot\overrightarrow{CH}=0\\&\overrightarrow{AC}\cdot\overrightarrow{BH}=0}\\
&\Leftrightarrow & \heva{&(-3)\cdot(x-1)+(-8)\cdot (y+2)=0\\&2\cdot (x+4)+(-5)\cdot (y+5)=0}\\
&\Leftrightarrow & \heva{&3x+8y=-13\\&2x-5y=17}\\
&\Leftrightarrow & \heva{&x=\dfrac{71}{31}\\&y=-\dfrac{77}{31}.}
\end{eqnarray*}
Vậy: $H\left(\dfrac{71}{31};-\dfrac{77}{31}\right)$.
\item Gọi $M(x;0)$ là điểm thuộc trục hoành, ta có
\begin{eqnarray*}
\heva{&\overrightarrow{MA}=(-1-x;3)\\&\overrightarrow{MC}=(1-x;-2)}\Rightarrow 2\overrightarrow{MA}+\overrightarrow{MC}=(-1-3x;4).
\end{eqnarray*}
Nên
\begin{eqnarray*}
 \left|2\overrightarrow{MA}+\overrightarrow{MC}\right|& =&\sqrt{(-1-3x)^2+4^2}\\ 
& =&\sqrt{(1+3x)^2+4^2}\ge 4.
\end{eqnarray*}
Dấu ``$=$'' xảy ra khi và chỉ khi $1+3x=0 \Leftrightarrow x=-\dfrac{1}{3}$. Khi đó $M\left(-\dfrac{1}{3};0\right)$.
\end{enumerate}
}
\end{bt}


\begin{bt}%[Đề HK1, Bình Hưng Hòa, Hồ Chí Minh, 2017]%[Đỗ Đường Hiếu, 10EX-HK1]%[0D4G5-6]
Giải phương trình $4x^2+12x\sqrt{x+1}=27\left(x+1\right)$.
\loigiai{
Điều kiện xác định: $x\ge -1$.\\
Đặt $t=\sqrt{x+1}$, với $t\ge 0$. Ta có phương trình
\begin{eqnarray*}
4x^2+12tx-27t^2=0 &\Leftrightarrow &(2x+9t)(2x-3t)=0\\
& \Leftrightarrow & \hoac{&2x=-9t\\&2x=3t}.
\end{eqnarray*}
\begin{itemize}
\item Với $2x=-9t$, ta có
\begin{eqnarray*}
2x=-9\sqrt{x+1}&\Leftrightarrow & \heva{&x\le 0\\&4x^2=81(x+1)}\\
&\Leftrightarrow & \heva{&x\le 0\\&4x^2-81x-81=0}\\
&\Leftrightarrow & \heva{&x\le 0\\&x=\dfrac{81+9\sqrt{97}}{8}\; \text{hoặc}\;x=\dfrac{81-9\sqrt{97}}{8}}\\
&\Leftrightarrow & x=\dfrac{81-9\sqrt{97}}{8}.
\end{eqnarray*}
\item Với $2x=3t$, ta có
\begin{eqnarray*}
2x=3\sqrt{x+1}&\Leftrightarrow & \heva{&x\ge 0\\&4x^2=9(x+1)}\\
&\Leftrightarrow & \heva{&x\ge 0\\&4x^2-9x-9=0}\\
&\Leftrightarrow & \heva{&x\ge 0\\&x=3\; \text{hoặc}\;x=-\dfrac{3}{4}}\\
&\Leftrightarrow & x=3.
\end{eqnarray*}
\end{itemize}
Đối chiếu điều kiện ban đầu, suy ra phương trình có hai nghiệm phân biệt là $x=\dfrac{81-9\sqrt{97}}{8}$ và $x=3$.
}
\end{bt}
