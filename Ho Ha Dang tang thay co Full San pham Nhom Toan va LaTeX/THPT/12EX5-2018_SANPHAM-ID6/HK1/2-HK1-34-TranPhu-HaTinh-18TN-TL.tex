 \noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[Đề KSCL học kì 1, 2017 - 2018, THPT Trần Phú, Hà Tĩnh]%[Nguyễn Vân Trường, dự án 12EX5-2018]%[2D2Y5-1] 
Giải phương trình $2^{x+1}=8$.
\loigiai{
$2^{x+1}=8 \Leftrightarrow 2^{x+1}=2^3 \Leftrightarrow x+1=3 \Leftrightarrow x=2.$ Vậy $x=2$. 
}
\end{bt}

\begin{bt}%[Đề KSCL học kì 1, 2017 - 2018, THPT Trần Phú, Hà Tĩnh]%[Nguyễn Vân Trường, dự án 12EX5-2018]%[2D2B5-1] 
Giải phương trình $\log_2(2x-6) = 3$.
\loigiai{
Điều kiện xác định $2x-6 > 0 \Leftrightarrow x > 3. $\\
$\log_2(2x-6) = 3 \Leftrightarrow 2x-6 = 2^3 \Leftrightarrow  x =7.$ Vậy $x =7$.
}
\end{bt}

\begin{bt}%[Đề KSCL học kì 1, 2017 - 2018, THPT Trần Phú, Hà Tĩnh]%[Nguyễn Vân Trường, dự án 12EX5-2018]%[2D1B2-1]
 Tìm cực trị của hàm số $y = x^3-3x^2+4$.
\loigiai{
Tập xác định $\mathscr{D} = \mathbb{R}.$\\
$y' = 3x^2-6x, y' = 0 \Leftrightarrow \hoac{& x = 0 \\ & x = 2.}$\\
$y'' = 6x -6.$\\
 Với $x = 0$, ta có $6 \cdot 0-6 < 0$, hàm số đạt cực đại tại $x = 0 \Rightarrow y = 4.$\\
Với $x = 2$, ta có $6 \cdot 2-6> 0$, hàm số đạt cực tiểu tại $x = 2 \Rightarrow y=0$.
}
\end{bt}

 Tìm giá trị lớn nhất và nhỏ nhất của hàm số $y = \dfrac{x-1}{x+4}$ trên $[0;2]$.
\loigiai{
Điều kiện xác định $ x \ne -4$. $y' = \dfrac{5}{(x+4)^2}> 0$ với mọi $x \ne 0$. Suy ra hàm số đồng biến trên đoạn $[0;2]$. Ta có $y(0) = -\dfrac{1}{4}$ và $y(2) = \dfrac{1}{6}.$ Vậy $\underset{[0;2]}{\min} y =  -\dfrac{1}{4}$ và  $\underset{[0;2]}{\max } y =  \dfrac{1}{6}$. 
}
\end{bt}
