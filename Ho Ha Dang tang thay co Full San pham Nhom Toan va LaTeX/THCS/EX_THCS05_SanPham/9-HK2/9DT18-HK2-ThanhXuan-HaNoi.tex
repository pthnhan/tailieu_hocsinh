	\item Chứng minh rằng $ P < 0 $ với mọi $ x \ne 4, x> 0. $
	\item Tìm những giá trị của $ x $ để $ P = - \dfrac{1}{15}. $
\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $ P = \dfrac{1}{\sqrt x (\sqrt x - 2)} - \dfrac{2}{(\sqrt x - 2)(\sqrt x + 2)} $\\
			$  P = \dfrac{(\sqrt x +2)}{\sqrt x(\sqrt x -2)(\sqrt x + 2)} - \dfrac{2\sqrt x}{\sqrt x (\sqrt x -2)(\sqrt x + 2)} $\\
			
			$ P = \dfrac{-1}{\sqrt x (\sqrt x + 2)} $.
			\item  Với mọi $ x> 0 $ ta có $ \begin{cases} \sqrt x > 0 \\ \sqrt x + 2 > 0 \end{cases} \Rightarrow \sqrt x (\sqrt x + 2) > 0 $.\\
			Mà $ -1 < 0 $ nên $ P = \dfrac{-1}{\sqrt x (\sqrt x + 2)} < 0$ với mọi $ x \ne 4, x > 0. $
			\item Với $ x\ne 4; \  x> 0 $ thì $ P = -\dfrac{1}{15} \Leftrightarrow  \dfrac{-1}{\sqrt x (\sqrt x +2)} = \dfrac{-1}{15} \Leftrightarrow \sqrt x(\sqrt x + 2) = 15 \\\Leftrightarrow x + 2\sqrt x - 15 =0 \Leftrightarrow \left[\begin{aligned} \sqrt x & =3 \quad \text{(nhận)}\\\sqrt  x &=-5 \quad \text{(loại)}. \end{aligned}\right. $
			 \\

			Vậy  $ P = \dfrac{-1}{15} $ khi $x =9. $
		\end{enumerate}

Một người đi ô tô từ $ A $ đến $ B $ cách nhau $ 90$km . Khi đi từ $ B $ trở về $ A $ người đó tăng tốc độ $ 5$km/h  so với tốc độ lúc đi, vì vậy thời gian về ít hơn thời gian đi là $ 15 $ phút. Tính tốc độ của ô tô lúc đi từ $ A $ đến $ B $.
	\loigiai{Đổi $ 15$ phút = $\dfrac{1}{4} $h.\\
		Gọi vận tốc của ô tô lúc đi từ $ A $ đến $ B $ là $ x $ (km/h), $ x> 0 $.\\
	Vận tốc của ô tô đi từ $ B $ trở về $ A $ là $ x + 5 $ (km/h).\\
Thời gian ô tô đi từ $ A  $ đến $ B $ là $ \dfrac{90}{x} $ (h).\\
Thời gian ô tô đi từ $ B $ trở về $ A $ là $ \dfrac{90}{x+5} $ (h).\\
 Thời gian về ít hơn thời gian đi là $ 15 $ phút nên ta có phương trình :\\ $ \dfrac{90}{x} - \dfrac{90}{x+5} = \dfrac{1}{4} \Leftrightarrow \dfrac{360(x+5)}{4x(x+5)} - \dfrac{360x}{4x(x+5)} = \dfrac{x(x+5)}{4x(x+5)}$ $\Rightarrow 360(x+5) - 360x = x(x+5)\\ \Leftrightarrow x^2 + 5x - 1800 = 0 \Leftrightarrow \left[\begin{aligned} x & =40 \quad \text{(nhận)}\\ x &=-45 \quad \text{(loại)}. \end{aligned}\right.   $ \\
Vậy vận tốc của ô tô lúc đi từ $ A $ đến $ B $ là $ 40$km/h.}

 	Cho hai số thực $ m $ và $ n $ khác $ 0 $ thỏa mãn $ \dfrac{1}{m} + \dfrac{1}{n} = \dfrac{1}{2} $. Chứng minh rằng trong hai phương trình $ x^2 +mx +n = 0 $ và $ x^2 + nx +m = 0 $ có ít nhất một phương trình có nghiệm.

	\loigiai
	{ Với $ m, n \ne 0 $.\\
		$ \dfrac{1}{m} + \dfrac{1}{n} = \dfrac{1}{2} \Rightarrow mn = 2(m+n) .$\\
		Phương trình $ x^2 +mx +n =0 \quad (1) $ có $ \Delta_1 = m^2 -4n. $\\
		Phương trình $ x^2 +nx +m =0 \quad (2) $ có $ \Delta_2 = n^2 - 4m. $\\
		$ \Delta_1 + \Delta_2 = m^2 +n^2 -4(m+n) \ge 2mn - 4(m+n) $ (Theo BĐT cô-si).\\
		Mà: $ mn= 2(m+n) $
		$ \Rightarrow \Delta_1 + \Delta_2 \ge 4(m+n) -4(m+n) = 0 .$\\
		Suy ra có ít nhất một $ \Delta \ge 0$.\\
		Vậy $ 2 $ phương trình đã cho có ít nhất một phương trình có nghiệm.
