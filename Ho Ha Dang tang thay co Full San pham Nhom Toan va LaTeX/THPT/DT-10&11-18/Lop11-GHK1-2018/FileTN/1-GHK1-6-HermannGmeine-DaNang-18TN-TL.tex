\begin{bt}%[Thi GHK1, THPT HermannGmeine- Đà Nẵng, 2018]%[Nguyễn Anh Tuấn, EX-DT-10&11]%[1D1B2-1]
Giải phương trình $ \sin\left(\dfrac{x}{2}+10^{\circ}\right)=-\dfrac{1}{2} $.
\loigiai{Ta có
$ \sin\left(\dfrac{x}{2}+10^{\circ}\right)=-\dfrac{1}{2} \Leftrightarrow  \sin\left(\dfrac{x}{2}+10^{\circ}\right)=\sin (-30^{\circ})\Leftrightarrow \hoac{&x=-80^{\circ}+k720^{\circ}\\&x=400^{\circ}+k720^{\circ}}(k\in \mathbb{Z}).$
}
\end{bt}
\begin{bt}%[Thi GHK1, THPT HermannGmeine- Đà Nẵng, 2018]%[Nguyễn Anh Tuấn, EX-DT-10&11]%[1D1B3-1]
	Giải phương trình $ 3\cot^2 x+3\cot x-2 =0 $.
\loigiai{Ta có
 $ 3\cot^2 x+3\cot x-2 =0 \Leftrightarrow \hoac{&\cot x=\dfrac{-3+\sqrt{33}}{6}\\&\cot x=\dfrac{-3-\sqrt{33}}{6}}\Leftrightarrow \hoac{&x=\mathrm{arccot} \dfrac{-3+\sqrt{33}}{6}+k\pi\\&x= \mathrm{arccot}\dfrac{-3-\sqrt{33}}{6}+k\pi .} (k\in \mathbb{Z})$
}
\end{bt}
\begin{bt}%[Thi GHK1, THPT HermannGmeine- Đà Nẵng, 2018]%[Nguyễn Anh Tuấn, EX-DT-10&11]%[1D1K3-1]
	Giải phương trình $ \sin 2x+2\cot x=3 $.
\loigiai{
Điều kiện xác định $ \sin x \ne 0 $.\\
Ta có $ \sin 2x=2\sin x\cdot \cos x=\dfrac{2\cdot \dfrac{\cos x}{\sin x}}{\dfrac{1}{\sin^2 x}} =\dfrac{2\cot x}{\cot^2 x+1}$. \\
Phương trình đã cho viết lại như sau $$\dfrac{2\cot x}{\cot^2 x+1}+2\cot x=3 \Leftrightarrow 2\cot^3 x-3\cot^2 x+4\cot x-3=0$$
$$\Leftrightarrow \hoac{&\cot x=1\\&2\cot^2 x-\cot x+3=0}\Leftrightarrow \cot x=1\Leftrightarrow x=\dfrac{\pi}{4}+k\pi, k\in \mathbb{Z}.$$
Đối chiếu với điều kiện, tập nghiệm của phương trình là $ \left\{\dfrac{\pi}{4}+k\pi, k\in \mathbb{Z}\right\} $.
}
\end{bt}

\begin{bt}%[Thi GHK1, THPT HermannGmeine- Đà Nẵng, 2018]%[Nguyễn Anh Tuấn, EX-DT-10&11]%[1D1K3-3]
Giải phương trình $ \sqrt{3}\cos 5x+\sin 5x=2\cos 3x $.
\loigiai{
Chia hai vế của phương trình cho $ 2 $, ta được
$$\begin{aligned}
&\dfrac{\sqrt{3}}{2}\cos 5x+\dfrac{1}{2}\sin 5x=\cos 3x\\
\Leftrightarrow &\cos \left(5x-\dfrac{\pi}{6}\right)=\cos 3x\\
\Leftrightarrow & \hoac{&5x-\dfrac{\pi}{6}=3x+k2\pi\\&5x-\dfrac{\pi}{6}=-3x+k2\pi} (k\in \mathbb{Z}) \Leftrightarrow \hoac{&x=\dfrac{\pi}{12}+k\pi\\&x=\dfrac{\pi}{48}+\dfrac{k\pi}{4} }(k\in \mathbb{Z}).
\end{aligned}$$}
\end{bt}
\begin{bt}%[Thi GHK1, THPT HermannGmeine- Đà Nẵng, 2018]%[Nguyễn Anh Tuấn, EX-DT-10&11]%[1D1B2-1]
Tìm nghiệm thuộc $ [0;2\pi] $ của phương trình $ \tan \left(\dfrac{x}{2}-\dfrac{\pi}{4}\right)=\tan \dfrac{\pi}{8} $.
\loigiai{
Phương trình đã cho có nghiệm là 
$ \dfrac{x}{2}-\dfrac{\pi}{4}=\dfrac{\pi}{8} + k\pi , k\in \mathbb{Z}\Leftrightarrow x=\dfrac{3\pi}{4}+k2\pi, k\in \mathbb{Z}  $.\\
Giải bất phương trình $ 0 \leq  \dfrac{3\pi}{4}+k2\pi \leq 2\pi$ với $ k\in \mathbb{Z} $, ta được $ x=\dfrac{3\pi}{4} $.
}

\end{bt}
\begin{bt}%[Thi GHK1, THPT HermannGmeine- Đà Nẵng, 2018]%[Nguyễn Anh Tuấn, EX-DT-10&11]%[1D1K1-5]
Tìm giá trị lớn nhất và giá trị nhỏ nhất của hàm số $ y=\dfrac{\sin x-\cos x}{2\sin x+\cos x-3} $.
\loigiai{
Gọi $ T $ là tập giá trị của hàm số đã cho và $ y_0 \in T $. \\
Ta có $ 2\sin x+\cos x-3=2(\sin x-1)+(\cos x-1) < 0, \forall x\in \mathbb{R}$ nên $ 2\sin x+\cos x-3 \ne 0 $.\\
Lại có $$ y_0=\dfrac{\sin x-\cos x}{2\sin x+\cos x-3}  \Leftrightarrow (2y_0-1)\sin x+(y_0+1)\cos x=3y_0 \Leftrightarrow \sin \left(x+\alpha\right)=\dfrac{3y_0}{\sqrt{5y_0^2-2y_0+2}}.$$
Trong đó $ \cos \alpha=\dfrac{2y_0-1}{\sqrt{5y_0^2-2y_0+2}} $, $ \sin \alpha =\dfrac{y_0+1}{\sqrt{5y_0^2-2y_0+2}} $.\\
Điều kiện cần và đủ để phương trình $ (*) $ có nghiệm là 
$$\dfrac{|3y_0|}{\sqrt{5y_0^2-2y_0+2}}\leq 1\Leftrightarrow 4y_0^2+2y_0-2\leq 0\Leftrightarrow -1\leq y_0\leq \dfrac{1}{2}.$$
Với $ y_0=-1 $, suy ra $ \sin x=1\Leftrightarrow x=\dfrac{\pi}{2}+k2\pi, k\in \mathbb{Z} $.\\
Với $ y_0=\dfrac{1}{2} $, suy ra $ \cos x=1\Leftrightarrow x=k2\pi, k\in \mathbb{Z} $.\\
Vậy $ \max \limits_{\mathbb{R}}y=\dfrac{1}{2} $, $ \min \limits_{\mathbb{R}}y=-1 $.
}
\end{bt}
