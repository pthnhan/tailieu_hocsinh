		 \end{align*}

		\item Giải hệ phương trình  $\heva{&x^2-2y^2=7x,\\&y^2-2x^2=7y.}$

		\begin{enumerate}	
		\item Trừ các phương trình của hệ, theo vế, ta được
		$$x^2-y^2+2(x^2-y^2)=7(x-y)$$ $$ \Leftrightarrow (x-y)(3x+3y)=7(x-y)$$
		$$ \Leftrightarrow (x-y)(3x+3y-7)=0$$
		$$\Leftrightarrow \hoac{&x-y=0\\&3x+3y-7=0}$$
		$$ \Leftrightarrow \hoac{&x=y\\&y=\dfrac{7-3x}{3}.}$$
		\begin{enumerate}[TH 1.]
		\item Với $x=y$, thay vào phương trình đầu của hệ, ta được
		$$x^2-2x^2=7x \Leftrightarrow x^2+7x=0 \Leftrightarrow \hoac{&x=0\\&x=-7.}$$ Trường hợp này, hệ có các nghiệm $(x,y)=(0,0), (-7,-7).$
		\item Với $y=\dfrac{7-3x}{3}$, thay vào phương trình đầu của hệ, ta được
		$$x^2-2\left(\dfrac{7-3x}{3}\right)^2=7x \Leftrightarrow 9x^2-21x+98=0 \text{ (phương trình này vô nghiệm).}$$
		Vậy hệ phương trình có hai nghiệm, là $(0;0)$ và $(-7;-7).$
		\end{enumerate}
		\item Điều kiện: $x\geq\dfrac{1}{2}$.\\
		Với điều kiện trên, ta có
		\begin{align*}
		&3x^2+65=2x\left(17-\sqrt{2x-1}\right)\\
		&\Leftrightarrow -3x^2+34x-65-2x\sqrt{2x-1}=0\\
		&\Leftrightarrow (2x-1)-(3x-8-x+8)\sqrt{2x-1}-3x^2+32x-64=0\\
		&\Leftrightarrow (2x-1)-(3x-8-x+8)\sqrt{2xx-1}+(8-3x)(x-8)=0\\
		&\Leftrightarrow \left(\sqrt{2x-1}-3x+8\right)\cdot\left(\sqrt{2x-1}+x-8\right)=0\\
		&\Leftrightarrow \hoac{&\sqrt{2x-1}-3x+8=0\\&\sqrt{2x-1}+x-8=0}\\
		&\Leftrightarrow \hoac{&\sqrt{2x-1}=3x-8,\\&\sqrt{2x-1}=8-x.}
		\end{align*}
		\begin{enumerate}[TH 1.]
			\item Xét $\sqrt{2x-1}=3x-8\\ \Leftrightarrow \heva{&3x-8\geq0\\&2x-1=(3x-8)^2} \Leftrightarrow \heva{&x\geq\dfrac{8}{3}\\&9x^2-50x+65=0}$\\
$$	\Leftrightarrow \heva{&x\geq\dfrac{8}{3}\\&\hoac{&x=\dfrac{25+2\sqrt{10}}{9},\ \text{(thỏa mãn),}\\&x=\dfrac{25-2\sqrt{10}}{9}\ \text{(loại).}}}$$
			\item Xét $\sqrt{2x-1}=8-x \Leftrightarrow \heva{&8-x\geq0\\&2x-1=(8-x)^2} \Leftrightarrow \heva{&x\leq8\\&x^2-18x+65=0}$\\
			$$\Leftrightarrow \heva{&x\leq8,\\&\hoac{&x=13\ \text{(loại)}\\&x=5\ \text{(thỏa mãn).}}}$$
		\end{enumerate}
		Vậy phương trình có tập nghiệm là $S=\left\{5;\dfrac{25+2\sqrt{10}}{9}\right\}.$
		\end{enumerate}
			}

			\begin{enumerate}
				\item Trên mặt phẳng lấy $21$ điểm bất kì trong đó không có ba điểm nào thẳng hàng; mỗi điểm được tô bởi một trong bốn màu đỏ, cam, vàng và lục. Xét các tam giác có ba đỉnh thuộc các điểm đã cho, chứng minh tồn tại tam giác có ba đỉnh cùng màu và ba cạnh cùng màu.
				\item Giả sử $n\in \mathbb{N}, n\geq2.$ Xét các số tự nhiên dạng $a_n=\overline{11\cdots1}$ được viết bởi $n$ chữ	số $1$. Chứng minh rằng nếu $a_n$ là một số nguyên tố thì $n$ là ước của $a_n-1.$

			\loigiai{
			\begin{enumerate}
			\item Vì có $21$ điểm và được tô bởi $4$ màu mà $21=4\cdot5+1$ nên theo nguyên lí Dirichlet sẽ tồn tại ít nhất $6$ điểm được tô cùng một màu.\\
			Gọi $6$ điểm cùng màu đó là $A$,$B$,$C$,$D$,$E$,$F$. Từ điểm $A$ ta kẻ với $5$ điểm còn lại được $5$ đoạn thẳng, $5$ đoạn thẳng này được tô $2$ màu thì sẽ có ít nhất $3$ đoạn được tô cùng màu.\\
			Không mất tính tổng quát giả sử các đoạn $AB$, $AC$, $AD$ được tô cùng màu tím. \\
			Trong các đoạn nối $3$ điểm $B$, $C$ và $D$ nếu có một đoạn màu tím, giả sử là $BD$ thì tam giác $ABD$ là tam giác cần tìm.  Nếu trong các đoạn nối $3$ điểm $B$, $C$, $D$ không có đoạn nào màu tím thì tam giác $BCD$ là tam giác cần tìm.

			\end{enumerate}
