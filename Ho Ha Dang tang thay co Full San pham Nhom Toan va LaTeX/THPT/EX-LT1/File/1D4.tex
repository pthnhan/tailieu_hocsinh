	Cho dãy số $(u_n)$ thỏa mãn $\lim\limits u_n=L$. Khẳng định nào sau đây là đúng?
	\choice {Nếu $u_n>0\ \forall n$ thì $L>0$}{ Nếu $L\geq 0$ thì $u_n \geq 0\ \forall n$}{$\lim\limits \sqrt{u_n}=\sqrt{L}$}{\True $\lim\limits u_n^2=L^2$}
	\loigiai{\hfill
		\begin{itemize}
			\item Xét $u_n=\dfrac{1}{n}$, ta có $u_n>0\ \forall n$, nhưng  $\lim\limits u_n=0$ nên \textbf{A} sai.
			\item Xét dãy $u_n=1-\dfrac{100}{n}$, ta có $\lim\limits u_n=1>0$, nhưng $u_n <0 \\text{với} \forall n<100.$ Do đó \textbf{B} sai.
			\item Khi $L<0$ thì $\sqrt{L}$ không tồn tại. Do đó \textbf{C} sai.
			\item Ta có $\lim\limits u_n^2=\lim\limits(u_n\cdot u_n)=L\cdot L=L^2.$ Do đó \textbf{D} đúng.
		\end{itemize}
	}
