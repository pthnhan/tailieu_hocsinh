\subsection{Hỏi đáp nhanh}
\begin{ex}%[7D2Y2]
	Điền vào chỗ chấm (...)
	\begin{enumerate}
		\item Chu vi một hình vuông tỉ lệ thuận với cạnh hình vuông, hệ số tỉ lệ là \sh{4}.
		\item Số hàng mua được tỉ lệ thuận với \sh{số tiền} nếu giá hàng không thay đổi.
		\item Chu vi một hình tròn tỉ lệ thuận với \sh{đường kính} với hệ số tỉ lệ là $\pi$.
		\item Diện tích một tam giác có đáy là $a$ (hằng số khác $0$) tỉ lệ thuận với đường cao theo hệ số tỉ lệ là \sh{$\dfrac{a}{2}$}.
	\end{enumerate}
\end{ex}

	\begin{enumerate}

		\item \TF{S} Nếu ta cùng thêm  một số vào tất cả các giá trị của hai đại lượng tỉ lệ thuận.
	\end{enumerate}
\end{ex}

\begin{vd}%[7D2B2]
	Cho các giá trị tương ứng của $x$ và $y$ trong bảng dưới.
	
	\begin{center}
		\setlength\extrarowheight{5pt}
		\newcolumntype{M}[1]{>{\centering\arraybackslash}m{#1}}
		\newcolumntype{N}{@{}m{0pt}@{}}
		\begin{tabular}{|M{2cm}|M{1cm}|M{1cm}|M{1cm}|M{1cm}|M{1cm}|M{1cm}|N}
			\hline
			\multirow{2}*{Bảng I} & $x$ & $0{,}65$ & $2{,}75$ & $0{,}6$ & $1{,}34$ & $37$ \\
			\cline{2-7}
			& $y$ & $5{,}2$ & $22$ & $4{,}8$ & $10{,}72$ & $296$  \\\hline
		\end{tabular}
	\end{center}
	\begin{center}
		\setlength\extrarowheight{5pt}
		\newcolumntype{M}[1]{>{\centering\arraybackslash}m{#1}}
		\newcolumntype{N}{@{}m{0pt}@{}}
		\begin{tabular}{|M{2cm}|M{1cm}|M{1cm}|M{1cm}|M{1cm}|M{1cm}|M{1cm}|N}
			\hline
			\multirow{2}*{Bảng II} & $x$ & $-2^5$ & $-2^3$ & $-2^1$ & $2^2$ & $2^4$ \\
			\cline{2-7}
			& $y$ & $128$ & $32$ & $4$ & $-16$ & $-64$  \\ \hline
		\end{tabular}
	\end{center}
	\begin{enumerate}
		\item Trong mỗi bảng, các đại lượng $x$ và $y$ có tỉ lệ thuận với nhau không?
		\item Nếu các đại lượng $x$ và $y$ tỉ lệ thuận với nhau, hãy chỉ ra hệ số tỉ lệ và viết công thức biểu thức biểu thị sự tương quan đó.
	\end{enumerate}
	\loigiai{
		\textbf{Bảng I.}
		\begin{enumerate}
			\item Ta có
			\[ \dfrac{5{,}2}{0{,}65}=\dfrac{22}{2{,}75}=\dfrac{4{,}8}{0{,}6}=\dfrac{10{,}72}{1{,}34}=\dfrac{296}{37}=8.\]
			Vậy các đại lượng $y$ và $x$ tỉ lệ thuận với nhau.
			\item ta có $\dfrac{y}{x}=\dfrac{5{,}2}{0{,}65}=8 \Rightarrow y=8x$ và hệ số tỉ lệ là $8$.
		\end{enumerate}
		\begin{note}
			Nếu tất cả các tỉ số giá trị tương ứng của $x$ và $y$ đều bằng nhau thì hai đại lượng $x$ và $y$ tỉ lệ thuận với nhau.
		\end{note}
		\textbf{Bảng II.\\}
		Ta có $\dfrac{4}{-2^1} \ne \dfrac{128}{-2^5}$ ($-2\ne-4$) và $\dfrac{128}{-2^5}=\dfrac{32}{-2^3}=\dfrac{-16}{2^2}=\dfrac{-64}{2^4}=-4$.\\
		Vậy các đại lượng $y$ và $x$ không tỉ lệ thuận với nhau.
	}
\end{vd}

\begin{vd}%[7D2K2]
	Cho $x$ và $y$ là hai đại lượng tỉ lệ thuận. Kí hiệu $x_1$ và $x_2$ là hai giá trị của đại lượng $x$ mà $x_1=-1$ và $x_2=-3$. Gọi $y_1$ và $y_2$ là hai giá trị tương ứng của đại lượng $y$ mà $y_1-y_2=-2$.
	\begin{enumerate}
		\item Tìm các giá trị $y_1$ và $y_2$.
		\item Đại lượng $x$ và $y$ liên hệ với nhau theo công thức nào?
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Biết $x$ và $y$ là hai đại lượng tỉ lệ thuận, suy ra $y=kx$ ($k\ne 0$).\\
			Theo tính chất của hai đại lượng tỉ lệ thuận,ta có
			\[ \dfrac{y_1}{-1}=\dfrac{y_2}{-3}=\dfrac{y_1-y_2}{(-1)-(-3)}=\dfrac{-2}{2}=-1 \]
			Suy ra $y_1=(-1) \cdot (-1)=1$ và $y_2=(-1) \cdot (-3)=3$.
			\item Vậy hai đại lượng $x$ và $y$ liên hệ với nhau theo công thức $y=(-1) \cdot x=-x$.
		\end{enumerate}
		\begin{note}
			Với hai đại lượng tỉ lệ thuận
			\begin{itemize}
				\item Tỉ số hai giá trị tương ứng của chúng không đổi.
				\item Tỉ số hai giá trị của đại lượng này bằng tỉ số hai giá trị tương ứng của đại lượng kia.
			\end{itemize}
		\end{note}
		\textbf{Mở rộng.\\}
		\textit{Việt Nam có nền kinh tế nông nghiệp từ hàng nghìn năm nay. Nền nông nghiệp của nước ta không chỉ sản xuất ra đủ một lượng lớn lương thực đáp ứng nhu cầu trong nước mà còn xuất khẩu sang nhiều nước trên thế giới. Ngành trồng lúa ở nước ta là một trong những ngành sản xuất lương thực vô cùng quan trọng và đạt được những thành tựu đáng kể đưa Việt Nam trở thành nước xuất khẩu gạo lớn nhất trên thế giới.}
	}
\end{vd}

\begin{vd}%[7D2K2]
	Cứ $100$ kg thóc cho $65$ kg gạo. Chất bột chứa trong gạo là $ 80\% $.
	\begin{enumerate}
		\item Hỏi trong $30$ kg thóc có bao nhiêu ki lô gam chất bột?
		\item Từ $1$ kg gạo người ta làm ra được $2{,}2$ kg bún tươi. Hỏi để làm ra $14{,}3$ kg bún tươi cần bao nhiêu ki-lô-gam thóc?
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Vì $100$ kg thóc được $65$ kg gạo, suy ra tương quan giữa hai đại lượng thóc và gạo là tỉ lệ thuận.\\
			Ta có
			\begin{center}
				\setlength\extrarowheight{5pt}
				\newcolumntype{M}[1]{>{\centering\arraybackslash}m{#1}}
				\newcolumntype{N}{@{}m{0pt}@{}}
				\begin{tabular}{|M{3cm}|M{3cm}|M{3cm}|N}
					\hline
					& Tiêu chuẩn (kg) & Hiện thực (kg) \\ [5pt]\hline
					Khối lượng thóc & $100$ & $30$ \\ [5pt]\hline
					Khối lượng gạo & $65$ & $x$ \\ [5pt]\hline
				\end{tabular}
			\end{center}
			Theo tính chất đại lượng tỉ lệ thuận, ta có $\dfrac{30}{x}=\dfrac{100}{65}\Rightarrow x=\dfrac{65 \cdot 30}{100}=19,5$(kg gạo).\\
			Chất bột chứa trong gạo là $80\%$, suy ra tương quan giữa hai đại lượng gạo và chất bột là tương quan tỉ lệ thuận.\\
			Ta có
			\begin{center}
				\setlength\extrarowheight{5pt}
				\newcolumntype{M}[1]{>{\centering\arraybackslash}m{#1}}
				\newcolumntype{N}{@{}m{0pt}@{}}
				\begin{tabular}{|M{3cm}|M{3cm}|M{3cm}|N}
					\hline
					& Tiêu chuẩn (kg) & Hiện thực (kg) \\ [5pt]\hline
					Khối lượng thóc & $100$ & $19{,}5$ \\ [5pt]\hline
					Khối lượng gạo & $80$ & $y$ \\ [5pt]\hline
				\end{tabular}
			\end{center}
			Theo tính chất đại lượng tỉ lệ thuận, ta có 
			\[ \dfrac{19,5}{y}=\dfrac{100}{80}\Rightarrow y=\dfrac{19{,}5 \cdot 80}{100}=15{,}6 \text{ (kg chất bột)}. \]
			Vậy trong $30$ kg thóc có $15{,}6$ kg chất bột.
			\item Từ $1$ kg gạo làm được $2{,}2$ kg bún tươi, suy ra tương quan giữa gạo và bún tươi tỉ lệ thuận.\\
			Gọi khối lượng gạo cần là $x$, ta có $\dfrac{x}{14{,}3}=\dfrac{1}{2{,}2}\Rightarrow x=\dfrac{1{,}4 \cdot 3{,}1}{2{,}2}=6{,}5$ (kg gạo).\\
			Gọi khối lượng thóc phải có là $y$, ta có $\dfrac{y}{6{,}5}=\dfrac{100}{65}\Rightarrow y=\dfrac{100 \cdot 6{,}5}{65}=10$ (kg thóc).\\
			Vậy để sản xuất ra $14{,}3$ kg bún tươi cần có $10$ kg thóc.
		\end{enumerate}
		\textbf{Cách khác.\\}
		Cứ $100$ kg thóc có $65$ kg gạo suy ra trong thóc có $65\% $ gạo.\\
		Chất bột trong gạo là $80\%$, ta có $80\%\left(=\dfrac{4}{5}\right)$ của $65\%$ bằng $\dfrac{4}{5} \cdot 65\%=52\% $.\\
		Suy ra, trong thóc có $52\%$ chất bột.\\
		Từ $30$ kg thóc có $52\% $ của $30$ bằng $ 30 \cdot 52\%=15{,}6$ (kg chất bột).
	}
\end{vd}

\begin{vd}%[7D2K2]
	Một xe đạp và một xa máy cùng đi một lúc từ thành phố $A$ đến thành phố $B$. Vì vận tốc của xe đạp nhỏ hơn vận tốc của xe máy là $18$ km/h, nên khi xe máy tới $B$ thì xe đạp mới tới $C$, cách $B$ một quãng đường bằn $0{,}6$ quãng đường $AB$. Tìm vận tốc mỗi xe.
	\loigiai{
		Kí hiệu các quãng đường $AC$ là $S_1$, và $AB$ là $S_2$ ta có
		$\dfrac{CB}{AB}=0,6=\dfrac{3}{5}$, do đó $\dfrac{S_1}{S_2}=\dfrac{AC}{AB}=\dfrac{2}{5}$.\\
		Gọi vận tốc và thời gian người xe đạp từ $A$ đến $C$ là $v_1$ và $t_1$.\\
		Gọi vận tốc và thời gian người xe máy từ $A$ đến $B$ là $v_2$ và $t_2$.\\
		Theo đề bài ta có $v_2-v_1=18$ (km/h).\\
		Hai xe đi cùng một lúc từ $A$, một xe tới $C$ và một xe tới $B$, vì cùng thời gian nên quãng đường và vận tốc là hai đại lượng tỉ lệ thuận.\\
		Ta có $\dfrac{v_1}{v_2}=\dfrac{S_1}{S_2}=\dfrac{2}{5}$ , suy ra $\dfrac{v_1}{2}=\dfrac{v_2}{5}=\dfrac{v_2-v_1}{5-2}=\dfrac{18}{3}=6$.\\
		Vậy vận tốc xe đạp là $v_1=6 \cdot 3=12$ (km/h) và vận tốc xe máy là $v_2=6 \cdot 5=30$ (km/h).\\
		\begin{note}
			\textbf{Chú ý.}
			\begin{itemize}
				\item Cùng thời gian thì QUÃNG ĐƯỜNG và VẬN TỐC là hai đại lượng tỉ lệ thuận.
				\item Đã có hiệu (hoặc tổng) của hai đại lượng nào đó hãy đi tìm tỉ số của chúng.
				\item Ngược lại đã có tỉ số của hai đại lượng nào đó hãy đi tìm tổng (hoặc hiệu) của chúng.
			\end{itemize}
		\end{note}
	}
\end{vd}
\begin{vd}%[7D2K2]
	Chia số $38$ thành ba số sao cho số thứ nhất và số thứ hai tỉ lệ theo $ 0{,}8:0{,}375$, còn số thứ hai và số thứ ba tỉ lệ theo $0{,}25$ và $1{,}75$.
	\loigiai{
		Gọi số thứ nhất, thứ hai và thứ ba theo thứ tự là $x$, $y$, $z$.\\
		Theo đề bài ta có $x+y+z=38$.\\
		Biết
		\[ x:y=0{,}8:0{,}375=32:15. \tag{1} \]
		và
		\[ y:z=0{,}25:1{,}75=1:7=15:105. \tag{2} \]
		Từ $(1)$ và $(2)$ suy ra
		\[ \dfrac{x}{32}=\dfrac{y}{15}=\dfrac{z}{105}=\dfrac{x+y+z}{32+15+105}=\dfrac{38}{152}=\dfrac{1}{4}=0{,}25.\]
		Vậy số thứ nhất bằng $8$, số thứ hai bằng $3{,}75$ và số thứ ba bằng $26{,}25$.
	}
\end{vd}

\begin{vd}%[7D2K2]
	Một cửa hàng có ba súc vải cùng khổ và có tổng độ dài là $86{,}1$ m. Khi bán $28\%$ súc vải thứ nhất, $40\%$ súc vài thứ hai và $64\%$ súc vải thứ ba thì chiều dài bs súc vải còn lại đều bằng nhau. Hỏi chiều dài mỗi súc vải khi chưa bán?
	\loigiai{
		Gọi chiều dài của ba súc vải khi chưa bán là $ x,y,z$(m), với $ x,y,z > 0$. ta có $ x+y+z=86{,}1$.\\
		Sau khi bán, chiều dài các súc vải còn lại bằng nhau: $72\% \cdot x=60\% \cdot y=36\% \cdot z$.\\
		Suy ra
		\[ x:y:z=\dfrac{1}{72\%}:\dfrac{1}{60\%}:\dfrac{1}{36\%}=5:6:10 .\]
		Áp dụng tính chất của dãy tỉ số bằng nhau
		\[ \dfrac{x}{5}=\dfrac{y}{6}=\dfrac{z}{10}=\dfrac{x+y+z}{5+6+10}=\dfrac{86{,}1}{21}=4{,}1.\]
		Chiều dài ba súc vải lần lượt là $20{,}5$ m; $24{,}6$ m; $41$ m.\\
		\textbf{Mở rộng.\\}
		\textit{Vải được cuộn lại thành từng súc để gọn và dễ chuyên chở. Các súc vải có chiều rộng gọi là khổ vải (các khổ có quy định kích thước, như khổ $1$ m; $1{,}2$ m).}
	}
\end{vd}

\begin{vd}%[7D2K2]
	Một nông trường trồng rừng phòng hộ vào ba lô đất. Biết diện tích lô thứ nhất bằng $40\% $ diện tích cả ba lô. Còn diện thích lô thứ hai và thứ ba tỉ lệ theo $1{,}5$ và $ 1{,}(3)$. Nếu diện tích lô thứ nhất lớn hơn diện tích lô thứ ba lg $1{,}2$ ha, thì diện tích của cả ba lô là bao nhiêu hecta?
	\loigiai{
		Gọi diện tích ba lô đất lần lượt là $x$, $y$ và $z$ (ha). Điều kiện $x,y,z > 0$.\\
		Theo đề bài ta có $x=40\%\left(x+y-z\right)$; $y:z=1{,}5:1{,}(3)=9:8$ và $x-z=12$ (ha).\\
		Suy ra
		\[x=\dfrac{2}{5}.\left(x+y+z\right)\Rightarrow\dfrac{x}{2}=\dfrac{x+y+z}{5}=\dfrac{y+z}{3}. \tag{1}\]
		\[ \dfrac{y}{9}=\dfrac{z}{8}=\dfrac{y+z}{17}. \tag{2}\]
		Từ $(1)$ và $(2)$ suy ra
		\[ \dfrac{x}{34}=\dfrac{y+x}{51}=\dfrac{y}{27}=\dfrac{z}{24} .\]
		Ta có
		\[ \dfrac{x}{34}=\dfrac{y}{27}=\dfrac{z}{24}=\dfrac{x+y+z}{85}=\dfrac{x-z}{10}=\dfrac{12}{10}=1{,}2. \]
		Vậy diện tích của cả ba lô đất bằng $x+y+z=1{,}2 \cdot 85=102$ (ha).
		\begin{note}
			\textbf{Nhận xét.\\}
			Cần biến đổi và kết nối để có được một dãy tỉ số bằng nhau trên cơ sở các tỉ lệ đã cho.
		\end{note}
	}
\end{vd}

\begin{vd}%[7D2K2]
	Anh hơn em $3$ tuổi. Tìm tuổi anh và tuổi em, biết tuổi anh hiện nay bằng $2$ lần tuổi em khi tuổi anh bằng tuổi em hiện nay.
	\loigiai{
		Gọi tuổi anh và em hiện nay là $x$ và $y$.\\
		Điều kiện và $x > y$, $x,y \in \mathbb{N^*}$.\\
		Anh hơn em $3$ tuổi, nên ta có $x=y+3$.\\
		Khi tuổi anh bằng tuổi em hiện nay, thì tuổi anh là $y$ và tuổi em là $y-3$.\\
		Ta có bảng so sánh
		\begin{center}
			\setlength\extrarowheight{5pt}
			\newcolumntype{M}[1]{>{\centering\arraybackslash}m{#1}}
			\newcolumntype{N}{@{}m{0pt}@{}}
			\begin{tabular}{|M{3cm}|M{4cm}|M{4cm}|N}
				\hline
				& Hiện nay & Khi tuổi anh bằng tuổi em hiện nay \\ [5pt]\hline
				Tuổi anh & $x$ & $y$ \\ [5pt]\hline
				Tuổi em & $y$ & $y-3$ \\ [5pt]\hline
			\end{tabular}
		\end{center}
		Biết tuổi anh hiện nay bằng $2$ lần tuổi em khi tuổi anh bằng tuổi em hiện nay, ta có tỉ lệ
		\[ \dfrac{x}{2}=\dfrac{y-3}{1}.\]
		Mà
		\[ x=y+3\Rightarrow\dfrac{y+3}{2}=\dfrac{y-3}{1}.\]
		Áp dụng tính chất dãy tỉ số bằng nhau
		\[ \dfrac{y+3}{2}=\dfrac{y-3}{1}=\dfrac{6}{1}=6\Rightarrow y=9,x=12. \]
		Vậy tuổi anh bằng $12$ và tuổi em bằng $9$.
	}
\end{vd}

\begin{bt}%[7D2G2]
	Hai bạn, một cao và một thấp, cùng nhau đi từ một nhà đến trường. Bước chân của bạn cao bằng $120\%$ bước chân của bạn thấp nhưng trong cùng một thời gian, bạn thấp bước nhiều hơn bạn cao $20\%$ số bước. Hỏi ai sẽ đến trường trước?
	\loigiai{
		Gọi độ dài mỗi bước chân của bạn thấp là $x$ (m) thì độ dài bước chân của bạn cao sẽ là \break $120\% \times x = 1{,}2x$ (m).\\
		Gọi số bước chân bạn cao bước đi được trong một giờ là $y$ (bước) thì số bước chân bạn thấp bước đi được trong một giờ là $(100\%+20\%) \times y = 120\% \times y=1{,}2y$ (bước).\\
		Trong một giờ quãng đường bạn thấp đi được là $x \times 1{,}2y=1{,}2xy$ (m).\\
		Trong một giờ quãng đường bạn cao đi được là $1{,}2x \times y =1{,}2xy $ (m).\\
		Vậy quãng đường đi được trong một giờ của hai bạn là như nhau hay vận tốc của hai bạn giống nhau, suy ra hai bạn đến trường cùng lúc.
	}
\end{bt}

\begin{bt}%[7D2G2]
	Biết rằng $7$ con mèo ăn hết $14$ con chuột trong thời gian là $21$ phút. Hỏi cần bao nhiêu con mèo để ăn hết $100$ con chuột trong một thời gian là $150$ phút?
	\loigiai{
		\textbf{Nhận xét.} Với cùng một số mèo, thời gian và số chuột mà mèo ăn hết là hai đại lượng tỉ lệ thuận.\\
		Mà $\dfrac{14}{21}=\dfrac{100}{150}\left(=\dfrac{2}{3}\right)$.\\
		Do đó để ăn hết $100$ con chuột trong một thời gian là $150$ phút cần $7$ con mèo.
	}
\end{bt}
