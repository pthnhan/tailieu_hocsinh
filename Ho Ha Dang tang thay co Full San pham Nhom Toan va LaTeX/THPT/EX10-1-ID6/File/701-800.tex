\begin{ex}%[Nguyễn Trung Hiếu]%[Huỳnh Văn Quy]%[0D2B3-2]
	Tìm các số thực $a$, $b$ sao cho parabol $(P)\colon y=ax^2+bx-2$ đi qua hai điểm $M(1;2)$ và $N(-2;-4)$.
	\choice
	{$a=-1; b=5$}
	{\True $a=1; b=3$}
	{$a=-2; b=6$}
	{$a=2; b=2$}
	\loigiai{Điểm $M(1;2)\in(P)\Leftrightarrow 2=a\cdot 1^2+b\cdot 1-2\Leftrightarrow a+b=4$\hfill $(1)$\\
		Điểm $N(-2;-4)\in(P)\Leftrightarrow -4=a\cdot (-2)^2+b\cdot (-2)-2\Leftrightarrow 4a-2b=-2$\hfill$(2)$\\

		a+b=4\\
		4a-2b=-2
		\end{cases}
		\Leftrightarrow
		\begin{cases}
		a=1\\
		b=3.
		\end{cases}$}
\end{ex}

\begin{ex}%[Thầy Lâm Hữu Phước]%[Huỳnh Văn Quy]%[0D2K3-2]
	Cho hàm số $y=ax^2+bx+c$ có đồ thị $(P)$. Biết $(P)$ là một parabol quay bề lõm xuống dưới và có đỉnh nằm ở góc phần tư thứ ba của hệ trục $Oxy$. Tính chất nào sau đây \textbf{đúng}?
	\choice{$a<0,b>0,c<0$}{$a>0,b<0,c>0$}{\True $a<0,b<0,c<0$}{$a<0,b<0,c>0$}
	\loigiai{
		Vì $(P)$ có bề lõm quay xuống nên $a<0$, hoành độ đỉnh $x=-\dfrac{b}{2a}$ âm (góc phần tư thứ ba) nên $a$ và $b$ cùng dấu, tức là $b<0$. Do tung độ đỉnh âm (góc phần tư thứ ba) nên giá trị của hàm số tại $x=0$ hiển nhiên âm, tức là $c<0$.
	}
\end{ex}
