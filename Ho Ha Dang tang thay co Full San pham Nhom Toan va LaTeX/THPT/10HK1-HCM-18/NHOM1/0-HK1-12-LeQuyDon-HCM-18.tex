\begin{bt}%[Đề HK1, Lê Quý Đôn, Hồ Chí Minh 17-18]%[Nguyện Ngô, 10EX-HK1]%[0D2B1-4]
Xét tính chẵn, lẻ của hàm số: $f(x)=\dfrac{2x^2+3}{|x+2|-|x-2|}$.
\loigiai{Ta có 
$$|x+2|-|x-2|=0\Leftrightarrow |x+2|=|x-2|\Leftrightarrow\hoac{&x+2=x-2\\&x+2=-x+2}\Leftrightarrow x=0.$$
Suy ra tập xác định của hàm số: $\mathscr{D}=\mathbb{R}\setminus\{0\}$.\\
Ta có 
$\forall x \in \mathscr{D}$ thì $-x\in \mathscr{D}$ và $f(-x)= \dfrac{2(-x)^2+3}{|-x+2|-|-x-2|}=-f(x)$.\\
Vậy hàm số đã cho là hàm số lẻ.
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Quý Đôn, Hồ Chí Minh 17-18]%[Nguyện Ngô, 10EX-HK1]%[0D2K3-2]
Xác định parabol $(P)\colon f(x)=ax^2+bx+2$, biết rằng parabol đi qua $B(-1; 6)$ và có tung độ đỉnh là $-\dfrac {1}{4}$.
\loigiai{Điều kiện: $a\neq 0$.
Parabol đi qua $B(-1; 6)$ nên  $a-b+2=6\Leftrightarrow a-b=4$.
Mặt khác tung độ đỉnh là $-\dfrac {1}{4}$ nên ta có: $b^2-8a=a\Leftrightarrow b^2=9a$.
Do đó ta có hệ phương trình:
$$\heva{&a-b=4\\&b^2=9a}\Leftrightarrow\heva{&a-b=4\\&b^2-9b-36=0}\Leftrightarrow\hoac{&a=1, b=-3\\&a=16, b=12.}$$
Vậy có $2$ parabol thỏa mãn đề bài là: $y=x^2-3x+2$ và $y=16x^2+12x+2$.
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Quý Đôn, Hồ Chí Minh 17-18]%[Nguyện Ngô, 10EX-HK1]%[0D3B2-4]
Giải phương trình $\sqrt{2x^2+7x+5}=x+1$.
\loigiai{
\begin{align*}
\sqrt{2x^2+7x+5}=x+1&\Leftrightarrow\heva{&x+1\geq 0\\&2x^2+7x+5=(x+1)^2}\\
&\Leftrightarrow\heva{&x\geq -1\\&x^2+5x+4=0}\\
&\Leftrightarrow\heva{&x\geq -1\\&\hoac{&x=-1\\&x=-4}}\Leftrightarrow x=-1.
\end{align*}
Vậy phương trình có tập nghiệm $S=\{-1\}$.
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Quý Đôn, Hồ Chí Minh 17-18]%[Nguyện Ngô, 10EX-HK1]%[0D3K2-2]
Giải phương trình $2x-|x^2-4x+5|=5$.
\loigiai{
\begin{align*}
2x-|x^2-4x+5|=5&\Leftrightarrow |x^2-4x+5|=2x-5\\
& \Leftrightarrow \heva{&2x-5\geq 0\\ &\hoac{&x^2-4x+5=2x-5\\&x^2-4x+5=-2x+5}}\\
&\Leftrightarrow \heva{&x\geq\dfrac{5}{2}\\ &\hoac{&x=0\\&x=2}}\quad\text{vô nghiệm}.
\end{align*}
Vậy phương trình có tập nghiệm $S=\varnothing$.
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Quý Đôn, Hồ Chí Minh 17-18]%[Nguyện Ngô, 10EX-HK1]%[0D4G5-2]
Cho $x$, $y$ là các số thực thỏa mãn $x>1$, $y>1$. Chứng minh rằng:\\
$\dfrac {xy}{\sqrt{(x-1)(y-1)}}\geq 4$.
\loigiai{
 Với $x>1$, $y>1$, ta có
 \begin{align*}
 \dfrac {xy}{\sqrt{(x-1)(y-1)}}\geq 4&\Leftrightarrow x^2y^2 \geq 16(x-1)(y-1)\\
 &\Leftrightarrow  x^2y^2-16xy+16x+16y-16 \geq0\\
 &\Leftrightarrow  y^2x^2-16(y-1)x+16(y-1) \geq0.\tag{$*$}
 \end{align*}
Coi vế trái của bất đẳng thức $(*)$ là tam thức bậc hai ẩn $x$, có $y$ là tham số. Khi đó ta có
$$\heva{&a=y^2>0\\&\Delta' =64(y-1)^2-16y^2(y-1)=-16(y-1)(y-2)^2 \leq 0,\quad\forall y>1.}$$
Suy ra $(*)$ đúng với mọi $x>1$, $y>1$. Bất đẳng thức được chứng minh.
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Quý Đôn, Hồ Chí Minh 17-18]%[Nguyện Ngô, 10EX-HK1]%[0H2K3-1]
Cho tam giác $ABC$ có $AB=9$, $AC=12$, $\widehat{BAC}=120^\circ$. Tính diện tích tam giác $ABC$, độ dài cạnh $BC$, độ dài trung tuyến $AM$ và bán kính đường tròn nội tiếp tam giác $ABC$.
\loigiai{
Ta có\\
$BC^2=AB^2+AC^2-2AB\cdot AC\cdot\cos\widehat{BAC}=9^2+12^2-2\cdot 9\cdot 12\cdot\cos120^\circ=333\Rightarrow BC=3\sqrt{37}$.\\
$AM^2=\dfrac{AB^2+AC^2}{2}-\dfrac{BC^2}{4}=\dfrac{9^2+12^2}{2}-\dfrac{333}{4}=\dfrac{117}{4}\Rightarrow  AM =\dfrac{3\sqrt{13}}{2}$.\\
Diện tích tam giác $ABC$ là $S=\dfrac{1}{2}AB\cdot AC\cdot\sin\widehat{BAC}=\dfrac{1}{2}\cdot 9\cdot 12\cdot\dfrac{\sqrt3}{2}=27\sqrt{3}$.\\
Nửa chu vi $p=\dfrac{9+12+3\sqrt{37}}{2}$. Suy ra bán kính đường tròn nội tiếp\\ 
$r=\dfrac{S}{p} =\dfrac{54\sqrt3}{9+12+3\sqrt{37}}\approx 2{,}38$.
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Quý Đôn, Hồ Chí Minh 17-18]%[Nguyện Ngô, 10EX-HK1]%[0H1B4-3]%[0H2K2-4]
Trong mặt phẳng tọa độ $Oxy$ cho tam giác $ABC$ với $A(1; 3)$, $B(-3; 0)$, $C(0; -2)$.
\begin{enumerate}
\item Tìm tọa độ điểm $M$ sao cho tứ giác $ABCM$ là hình bình hành.
\item Tìm tọa độ điểm $D$ thuộc trục $Oy$, sao cho  $\left|\vec{AD}+\vec{BD}-\vec{CD}\right|=2\sqrt5$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Tứ giác $ABCM$ là hình bình hành khi và chỉ khi
$$\vec{MC}=\vec{AB}\Leftrightarrow\heva{&-x_M=-4\\&-2-y_M=-3}\Leftrightarrow\heva{&x_M=4\\&y_M=1.}$$
Vậy điểm $M(4; 1)$.			
\item Vì $D\in Oy$ nên $D(0; d)$. Ta có $\vec{AD}=(-1; d-3)$, $\vec{BC}=(3; -2)$, suy ra $\vec{AD}+\vec{BC}=(2; d-5)$. Khi đó
\begin{align*}
\left|\vec{AD}+\vec{BD}-\vec{CD}\right|=2\sqrt5&\Leftrightarrow \left|\vec{AD}+\vec{BC}\right|=2\sqrt5\\
&\Leftrightarrow \sqrt{4+(d-5)^2}=2\sqrt{5}\\
&\Leftrightarrow d^2-10d+9=0 \Leftrightarrow\hoac{&d=1\\&d=9.}
\end{align*}
Vậy có hai điểm $D$ thỏa mãn đề bài là: $D_1(0; 9)$, $D_2(0; 1)$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Quý Đôn, Hồ Chí Minh 17-18]%[Nguyện Ngô, 10EX-HK1]%[0D3K2-3]
Xác định tất cả các giá trị của tham số $m$ để phương trình $\dfrac{x-m}{x+1}= m+1$ có nghiệm.
\loigiai{
Ta có
$$\dfrac{x-m}{x+1}= m+1\Leftrightarrow\heva{&x\neq-1\\&x-m=(m+1)x+m+1}\Leftrightarrow\heva{&x\neq-1\\&mx+2m+1=0.\quad\quad (*)}$$
\begin{itemize}
\item Với $m=0$, $(*)$ trở thành: $1=0$ vô nghiệm.
\item Với $m\neq 0$, $(*)\Leftrightarrow x=-\dfrac{2m+1}{m}$. Phương trình đã cho có nghiệm khi và chỉ khi\\
$-\dfrac{2m+1}{m}\neq-1\Leftrightarrow m\neq-1$.
\end{itemize}
Vậy với $m\in\mathbb{R}\setminus\{-1; 0\}$ thì phương trình đã cho có nghiệm.
}
\end{bt}
