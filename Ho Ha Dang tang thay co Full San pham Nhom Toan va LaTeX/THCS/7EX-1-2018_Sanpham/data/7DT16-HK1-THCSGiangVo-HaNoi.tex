	{\True $\dfrac{2xy}{7}=3$}
\loigiai{
	Ta có : 
	\begin{itemize}
		\item $xy=a\Leftrightarrow y=\dfrac{a}{x}$. Nếu $a=0$ thì $x$ và $y$ không phụ thuộc nhau; nếu $a\neq 0$ thì $x$ và $y$ tỉ lệ nghịch.
		\item $\dfrac{x}{y}=4\Leftrightarrow x=4y$ nên $x$, $y$ tỉ lệ thuận.
		\item $x+y=5\Leftrightarrow y=-x+5$ nên $x$ giảm thì $y$ giảm nhưng không tỉ lệ.
		\item $\dfrac{2xy}{7}=3\Leftrightarrow x=\dfrac{\dfrac{21}{2}}{y}$ nên $x$ và $y$ tỉ lệ nghịch.
	\end{itemize}
	}
\end{ex}

Thực hiện phép tính: $A=\left(1\text{,}2-\sqrt{\dfrac{1}{4}}\right):1\dfrac{1}{20}+\left| \dfrac{3}{4}-1\text{,}25\right|-{\left(-\dfrac{3}{2}\right)}^2$.
\loigiai{

	\begin{eqnarray*}
		A&=&\left(\dfrac{12}{10}-\dfrac{1}{2}\right):\dfrac{21}{20}+\left| \dfrac{3}{4}-\dfrac{5}{4}\right|-\dfrac{9}{4}\\
		&=&\left(\dfrac{12}{10}-\dfrac{5}{10}\right)\cdot \dfrac{20}{21}+\left|\dfrac{2}{4}\right|-\dfrac{9}{4}\\
		&=&\dfrac{7}{10}\cdot\dfrac{20}{21}+\dfrac{2}{4}-\dfrac{9}{4}\\

	Ba lớp $7A$, $7B$, $7C$ đã đóng góp một số sách để hưởng ứng việc xây dựng mỗi lớp có một thư viện riêng. Biết số sách góp được của lớp $7A$, $7B$, $7C$ tỉ lệ với $6$, $4$, $5$ và tổng số sách góp được của lớp $7A$ và lớp $7B$ hơn số sách của lớp $7C$ là $40$ quyển. Tính số sách mỗi lớp đóng góp được. 
\loigiai{
	Gọi số sách của lớp $7A$, $7B$, $7C$ lần lượt là $a$, $b$, $c$. Theo bài ra ta có $$\dfrac{a}{6}=\dfrac{b}{4}=\dfrac{c}{5}\Rightarrow \dfrac{a}{6}=\dfrac{b}{4}=\dfrac{-c}{-5}=\dfrac{a+b-c}{6+4-5}=\dfrac{40}{5}=8.$$
	Vậy số sách của lớp $7A$ là $8\cdot 8=48$ (quyển sách), số sách của lớp $7B$ là $4\cdot 8=32$ (quyển sách), số sách của lớp $7C$ là $5\cdot 8=40$ (quyển sách).
	}
\end{ex}

	Cho góc nhọn $\widehat{xOy}$, lấy điểm $A$ thuộc tia $Ox$ và điểm $B$ thuộc tia $Oy$ sao cho $OA=OB$. Gọi $C$ là trung điểm của đoạn thẳng $AB$.

\begin{ex}%[Đề kiểm tra học kỳ I, THCS Giảng Võ-Hà Nội, 2015 - 2016]%[Nguyễn Phúc Đức, dự án 7EX-1-2018]%[7D1G8]
	Cho $a$, $b$, $c \neq 0$ và $a+b+c=\dfrac{a+2b-c}{c}=\dfrac{b+2c-a}{a}=\dfrac{c+2a-b}{b}$. 
Tính $$P=\left(2+\dfrac{a}{b}\right)\left(2+\dfrac{b}{c}\right)\left(2+\dfrac{c}{a}\right).$$
\loigiai{
	Xét hai trường hợp sau:
	\begin{itemize}
		\item Nếu $a+b+c=0$, ta có 
		\begin{eqnarray*}
			&&a+b+c=\dfrac{a+2b-c}{c}=\dfrac{b+2c-a}{a}=\dfrac{c+2a-b}{b}=0\\
			&\Rightarrow&a+2b-c=0;\, b+2c-a=0;\, c+2a-b=0\\
			&\Rightarrow&c=a+2b;\, a=b+2c;\, b=c+2a
		\end{eqnarray*}
		Khi đó \begin{eqnarray*}
			P&=&\left(2+\dfrac{a}{b}\right)\left(2+\dfrac{b}{c}\right)\left(2+\dfrac{c}{a}\right)\\
			&=&\dfrac{2b+a}{b}\cdot \dfrac{2c+b}{c}\cdot \dfrac{2a+c}{a}\\
			&=&\dfrac{c}{b}\cdot\dfrac{a}{c}\cdot\dfrac{b}{a}\\
			&=&\dfrac{abc}{abc}=1.
		\end{eqnarray*}
	\item Nếu $a+b+c\neq 0$, ta có 
	\begin{eqnarray*}
		&&a+b+c=\dfrac{a+2b-c}{c}=\dfrac{b+2c-a}{a}=\dfrac{c+2a-b}{b}\\
		&\Rightarrow&a+b+c=\dfrac{a+2b-c+b+2c-a+c+2a-b}{a+b+c}=\dfrac{2(a+b+c)}{a+b+c}=2\\
		&\Rightarrow&\dfrac{a+2b-c}{c}=\dfrac{b+2c-a}{a}=\dfrac{c+2a-b}{b}=2\\
		&\Rightarrow&c+2a=3b;\, b+2c=3a;\, a+2b=3c\\
	\end{eqnarray*}
	Khi đó \begin{eqnarray*}
		P&=&\left(2+\dfrac{a}{b}\right)\left(2+\dfrac{b}{c}\right)\left(2+\dfrac{c}{a}\right)\\
		&=&\dfrac{2b+a}{b}\cdot \dfrac{2c+b}{c}\cdot \dfrac{2a+c}{a}\\
		&=&\dfrac{3c}{b}\cdot\dfrac{3a}{c}\cdot\dfrac{3b}{a}\\
		&=&\dfrac{9abc}{abc}=9.
	\end{eqnarray*}
	\end{itemize}
	Vậy theo giả thiết đã cho thì $P=1$ hoặc $P=9$.
	}
\end{ex}
