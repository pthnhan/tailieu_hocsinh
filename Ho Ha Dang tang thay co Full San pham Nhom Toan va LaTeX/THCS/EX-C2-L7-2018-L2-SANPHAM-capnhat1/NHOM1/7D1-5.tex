	\end{enumerate}

		\begin{enumerate}
			\item Ta có

			\[ y=3\sqrt 7+5\sqrt 2=\sqrt{63}+\sqrt{50}<\sqrt{64}+\sqrt{64}=8+8=16 \Rightarrow y < 16. \]

			\item Xét $\left| x\right|=\dfrac{1}{\sqrt{29+5}}$ và $\left| y\right|=\dfrac{1}{\sqrt{29}+\sqrt 5}$.\\
			Ta có $\sqrt{29+5}=\sqrt{34}<\sqrt{36}=6$ và $\sqrt{29}+\sqrt 5 >\sqrt{25}+\sqrt 4=5+2=7$.\\
			Suy ra $0 <\sqrt{29+5}<\sqrt{29}+\sqrt 5\Rightarrow\dfrac{1}{\sqrt{29+5}}>\dfrac{1}{\sqrt{29}+\sqrt 5}> 0$.\\
			Vậy $\dfrac{-1}{\sqrt{29+5}}<\dfrac{-1}{\sqrt{29}+\sqrt 5}$.
		\end{enumerate}
	}
\end{vd}

	\end{enumerate}

\begin{bt}%%[7D1B12]
	Tính diện tích hình tròn và chu vi đường tròn. Lấy số $\pi\approx\dfrac{22}{7}$.
	\begin{enumerate}
		\item Một hình vuông nằm bên trong một hình tròn (hình bên). Biết đường chéo hình vuông bằng $\sqrt{18}$ cm. Tính diện tích phần hình tròn không bị hình vuông phủ (chính xác tới chữ số thập phân thứ hai).
		\item Một vệ tinh bay trên quỹ đạo vòng tròn quanh Trái Đất. Biết quỹ đạo của vệ tinh có độ dài là $66000$ km. Hỏi độ dài quỹ đạo của về tinh giảm bao nhiêu ki-lô-mét nếu bán kính của quỹ đạo giảm $70$ km?
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $S_{\text{(hình tròn)}} =\pi.R^2=\dfrac{22}{7}.\left(\dfrac{\sqrt{18}}{2}\right)^2=\dfrac{99}{7}=14\dfrac{1}{7}$ (cm$^2$).\\
			$S_{\text{(hình vuông)}} = 3 \cdot 3=9$ (cm$^2$).\\
			$S_{\text{(phần hình tròn không bị hình vuông phủ)}}=14\dfrac{1}{7}-9=5\dfrac{1}{7}\approx 5{,}14$ (cm$^2$).\\
			\item Bán kính quỹ đạo lúc đầu $R=66000:(2\pi)=10500$ (km).\\
			Bán kính quỹ đạo lúc sau $R'=10500-70=10430$ (km).\\
			Quỹ đạo vệ tinh lúc sau $C'=2\pi R'=2 \cdot \dfrac{22}{7} \cdot 10430=65560$ (km).\\
			Quỹ đạo giảm so với quỹ đạo ban đầu $C-C'=66000-65560=440$ (km).
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[7D1G12]
	Chứng minh một số là số vô tỉ.\\
	\textbf{Bạn CHICKEN hỏi:} "$\sqrt 2 $ là số vô tỉ hay số hữu tỉ?"
	\loigiai{
		\textbf{Bạn EGG chứng minh.\\}
		Giả sử $\sqrt 2 $ là một số hữu tỉ, thì $\sqrt 2=\dfrac{m}{n}$ (và $\left(m,n\right)=1$).\\
		Suy ra $m=\sqrt 2 \cdot n \Rightarrow m^2=2n^2 \Rightarrow m^2 \ \vdots \ 2 \Rightarrow m \ \vdots \ 2$ ($2$ là số nguyên tố).\\
		Từ $m \ \vdots \ 2 \Rightarrow m=2k$ ($k \in \mathbb{N^*}$) suy ra $n^2=m^2:2=2k^2 \Rightarrow n^2 \ \vdots \ 2 \Rightarrow n \ \vdots \ 2$ (vì $2$ là số nguyên tố).\\
		Do đó $m$ và $n$ đều chia hết cho $2$, trái với điều kiện $\left(m,n\right)=1$.\\
		Vậy điều giả sử là sai, hay $\sqrt 2 $ là một số vô tỉ.\\
		\textbf{Suy ngẫm.\\}
		Các số $\sqrt 2$; $\sqrt 3$; $\sqrt 5$; $\sqrt 6$; $\sqrt 7$; $\sqrt 8$; $\sqrt{10}$; $\ldots$ là các số vô tỉ.\\
		Các số $\sqrt 1$; $\sqrt 4$; $\sqrt 9$; $\sqrt {16}$; $\sqrt{25}$; $\ldots$ là các số hữu tỉ.\\
		Như vậy, với số nguyên dương $a$ nếu $a$ là số chính phương thì $\sqrt a $ là số hữu tỉ, còn nếu $a$ không là số chính phương thì $\sqrt a $ là số vô tỉ.\\
		\textbf{Giải thích.\\}
		Thật vậy, giả sử $a$ không là số chính phương nhưng $\sqrt a $ là một số hữu tỉ.\\
		Ta có $\sqrt a \in \mathbb{Q} \Rightarrow \sqrt a = \dfrac{m}{n}$ với $m,n \in \mathbb{N}$, $n>1$ và $(m,n)=1 \Rightarrow m^2=a \cdot n^2$.\\
		Gọi $p$ là một ước nguyên tố của $n$ $\left(p\ne 1\right)$ thì $n \ \vdots \ p \Rightarrow m^2 \ \vdots \ p \Rightarrow m \ \vdots \ p$ (vì $p$ là số nguyên tố) $\Rightarrow\left(m,n\right)=p\ne 1$, trái với điều giả sử.\\
		Vậy $\sqrt a $ là số vô tỉ.\\
		\textbf{Bài toán.\\}
		Bạn hãy tìm hai số hữu tỉ $a$ và $b$, biết $a+b \cdot \sqrt p=0$, trong đó $p$ là số nguyên tố.
	}
\end{bt}
