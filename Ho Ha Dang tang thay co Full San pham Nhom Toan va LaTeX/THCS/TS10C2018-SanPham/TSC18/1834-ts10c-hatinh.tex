	Cho $x, y, z$ là các số hữu tỷ, thỏa mãn $\dfrac{1}{x} + \dfrac{1}{y} = \dfrac{1}{z}$.
	Chứng minh $\sqrt {{x^2} + {y^2} + {z^2}}$ là một số hữu tỷ.
	\loigiai{
	Từ giả thiết $\dfrac{1}{x} + \dfrac{1}{y} = \dfrac{1}{z} \Leftrightarrow zx + yz = xy \Leftrightarrow 2xy - 2yz - 2zx = 0$\\
	$\begin{array}{lll}
	\mbox{Do đó }\sqrt {{x^2} + {y^2} + {z^2}} & = & \sqrt {{x^2} + {y^2} + {z^2} + 2xy - 2yz - 2zx}\\
	& = & \sqrt {{{\left( {x + y - z} \right)}^2}}\\
	& = & \left| {x + y - z} \right| \mbox{ là số hữu tỷ.}
	\end{array}$

	}
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Hà Tĩnh, tỉnh Hà Tĩnh, năm 2018]%[Tô Ngọc Thy, Dự án EX-TS10C-18]%[9D0G2]
    \begin{listEX}
   	\item Giải phương trình $4{x^2} - 3x - 2 = \sqrt {x + 2}$.
   	\item Giải hệ phương trình   	
   	$\left\{ \begin{array}{l}
   	xy - x - y =  - 5\\
   	\dfrac{1}{{{x^2} - 2x}} + \dfrac{1}{{{y^2} - 2y}} = \dfrac{2}{3}.
   	\end{array} \right.$
   \end{listEX}
\loigiai{
  \begin{listEX}
  	\item ĐKXĐ: $x \geq - 2$. Ta có phương trình tương đương $16{x^2} - 12x - 8 = 4\sqrt {x + 2}$\\
    $ \Leftrightarrow 16{x^2} - 8x + 1 = 4\left( {x + 2} \right) + 4\sqrt {x + 2}  + 1 \Leftrightarrow {\left( {4x - 1} \right)^2} = {\left( {2\sqrt {x + 2}  + 1} \right)^2}$\\
    Xét $4x - 1 =  - \left( {2\sqrt {x + 2}  + 1} \right) \Leftrightarrow \sqrt {x + 2}  =  - 2x \Leftrightarrow \left\{ \begin{array}{l}
    4{x^2} - x - 2 = 0\\
    - 2 \le x \le 0
    \end{array} \right. \Rightarrow x = \dfrac{{1 - \sqrt {33} }}{8}$\\
    Xét $4x - 1 = 2\sqrt {x + 2}  + 1 \Leftrightarrow \sqrt {x + 2}  = 2x - 1 \Leftrightarrow \left\{ \begin{array}{l}
    4{x^2} - 5x - 1 = 0\\
    x \ge \dfrac{1}{2}
    \end{array} \right. \Rightarrow x = \dfrac{{5 + \sqrt {41} }}{8}$\\
    Vậy tập nghiệm của phương trình $S = \left\{ {\dfrac{{1 - \sqrt {33} }}{8};\dfrac{{5 + \sqrt {41} }}{8}} \right\}$.
  	\item ĐKXĐ: $x, y \neq 0; x, y \neq 2$. Từ phương trình $xy - x - y =  - 5 \Leftrightarrow \left( {x - 1} \right)\left( {y - 1} \right) =  - 4$.\\
  	Đặt $\left\{ \begin{array}{l}
  	x - 1 = a\\
  	y - 1 = b
  	\end{array} \right. \Rightarrow \left\{ \begin{array}{l}
  	{x^2} - 2x = {a^2} - 1\\
  	{y^2} - 2y = {b^2} - 1
  	\end{array} \right.$ và $ab=-4$. Ta có\\
    $\dfrac{1}{{{x^2} - 2x}} + \dfrac{1}{{{y^2} - 2y}} = \dfrac{2}{3} \Leftrightarrow \dfrac{{{a^2} + {b^2} - 2}}{{17 - {a^2} - {b^2}}} = \dfrac{2}{3} \Rightarrow {a^2} + {b^2} = 8 \Rightarrow {\left( {a + b} \right)^2} - 2ab = 8 \Rightarrow a + b = 0$\\
    $ \Rightarrow a =  - b$. Do đó $ab =  - 4 \Leftrightarrow  - {a^2} =  - 4 \Leftrightarrow a =  \pm 2$.\\
    TH1: $\left\{ \begin{array}{l}
    a =  - 2\\
    b = 2
    \end{array} \right. \Rightarrow \left\{ \begin{array}{l}
    x - 1 =  - 2\\
    y - 1 = 2
    \end{array} \right. \Rightarrow \left\{ \begin{array}{l}
    x =  - 1\\
    y = 3
    \end{array} \right.$\\
    TH2: $\left\{ \begin{array}{l}
    a = 2\\
    b =  - 2
    \end{array} \right. \Rightarrow \left\{ \begin{array}{l}
    x - 1 = 2\\
    y - 1 =  - 2
    \end{array} \right. \Rightarrow \left\{ \begin{array}{l}
    x = 3\\
    y =  - 1
    \end{array} \right.$\\
    Vậy hệ phương trình có nghiệm $(x;y)$ là $(- 1;3), (3;- 1)$.
  \end{listEX}
    }
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Hà Tĩnh, tỉnh Hà Tĩnh, năm 2018]%[Tô Ngọc Thy, Dự án EX-TS10C-18]%[9D0K7]
	Cho phương trình ${x^2} + 2mx - 1 - 2m = 0 \left( m \mbox{ là tham số} \right) $. Chứng minh phương trình luôn có hai nghiệm $x_1, x_2$ với mọi $m$. Tìm $m$ để biểu thức $P = \dfrac{{2{x_1}{x_2} + 1}}{{x_1^2 - 2m{x_2} + 1 - 2m}}$ đạt giá trị nhỏ nhất.
	\loigiai{
	 Ta có $\Delta ' = {m^2} + 2m + 1 = {\left( {m + 1} \right)^2} \ge 0$ nên phương trình luôn có hai nghiệm với mọi $m$.\\
	Theo Vi-et ta có $\left\{ \begin{array}{l}
	{x_1} + {x_2} =  - 2m\\
	{x_1}{x_2} =  - 2m - 1
	\end{array} \right.$\\
	$\begin{array}{lll}
		\mbox{Do đó } P = \dfrac{{ - 4m - 1}}{{x_1^2 + \left( {{x_1} + {x_2}} \right){x_2} + 1 - 2m}} & = & \dfrac{{ - 4m - 1}}{{x_1^2 + {x_1}{x_2} + x_2^2 + 1 - 2m}}\\
	& = & \dfrac{{ - 4m - 1}}{{{{\left( {{x_1} + {x_2}} \right)}^2} - {x_1}{x_2} + 1 - 2m}}\\
	& = & \dfrac{{ - 4m - 1}}{{4{m^2} + 2}} = \dfrac{{ - 4m - 1}}{{4{m^2} + 2}} + 1 - 1 \\
	& = & \dfrac{{{{\left( {2m - 1} \right)}^2}}}{{4{m^2} + 2}} - 1 \ge  - 1

	\end{array}$\\
	Vậy GTNN của $P$ bằng $- 1$ khi $m = \dfrac{1}{2}$.
	}
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Hà Tĩnh, tỉnh Hà Tĩnh, năm 2018]%[Tô Ngọc Thy, Dự án EX-TS10C-18]%[9D0G7]
	Cho $3$ số dương $x, y, z$ thỏa mãn $x + y + z = 1$.
	Chứng minh: $$\sqrt {\dfrac{{xy}}{{xy + z}}}  + \sqrt {\dfrac{{yz}}{{yz + x}}}  + \sqrt {\dfrac{{zx}}{{zx + y}}}  \le \dfrac{3}{2}$$.
	\loigiai{
	Áp dụng BĐT CauChy ta có\\ $\sqrt {\dfrac{{xy}}{{xy + z}}}  = \sqrt {\dfrac{{xy}}{{xy + z\left( {x + y + z} \right)}}}  = \sqrt {\dfrac{{xy}}{{\left( {z + x} \right)\left( {y + z} \right)}}}  \le \dfrac{1}{2}\left( {\dfrac{x}{{z + x}} + \dfrac{y}{{y + z}}} \right)$\\
	Tương tự ta cũng có
	 $$\sqrt {\dfrac{{yz}}{{yz + x}}}  \le \dfrac{1}{2}\left( {\dfrac{y}{{x + y}} + \dfrac{z}{{z + x}}} \right)$$
	$$\sqrt {\dfrac{{zx}}{{zx + y}}}  \le \dfrac{1}{2}\left( {\dfrac{z}{{y + z}} + \dfrac{x}{{x + y}}} \right)$$\\
	Do đó $P \le \dfrac{1}{2}\left( {\dfrac{x}{{z + x}} + \dfrac{y}{{y + z}} + \dfrac{y}{{x + y}} + \dfrac{z}{{z + x}} + \dfrac{z}{{y + z}} + \dfrac{x}{{x + y}}} \right) = \dfrac{3}{2}$.\\
	Dấu $''=''$  xảy ra khi và chỉ khi $x = y = z = \dfrac{1}{3}$.	
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Hà Tĩnh, tỉnh Hà Tĩnh, năm 2018]%[Tô Ngọc Thy, Dự án EX-TS10C-18]%[9D0G4]
	Cho $a_1 <a_2 <a_3 < \ldots <a_n < \ldots$ với $\left( n \in N^* \right)$ là những số nguyên dương và không có hai số nào liên tiếp. Đặt $S_n = a_1 + a_2 + \ldots + a_n$. Chứng minh rằng luôn tồn tại ít nhất một số chính phương $b$ thỏa mãn $S_n\leq b \leq S_{n+1}$.
	\loigiai{
	Vì $S_n = a_1 + a_2 + \ldots + a_n$ nên $ S_{n+1} = S_n + a_{n+1}$.\\
	Ta có $\sqrt{S_{n+1}} - \sqrt{S_n} \geq 1 \Leftrightarrow S_{n+1} \geq \left( \sqrt{S_n} + 1\right)^2 \Leftrightarrow S_n + a_{n+1} \geq \left( \sqrt{S_n} + 1\right)^2$  	\\
	$\Leftrightarrow S_n + a_{n+1} \geq S_n + 2\sqrt{S_n} + 1 \Leftrightarrow a_{n+1} \geq 2\sqrt{S_n} + 1 \hspace{1cm}(*)$\\
	Vì $a_1 <a_2 <a_3 < \ldots <a_n < \ldots$ và không có hai số nào liên tiếp nên ta có\\
	$ a_{n+1} \geq a_n + 2$\\
	$ a_n \geq a_{n-1} + 2 \Leftrightarrow a_{n+1} \geq a_{n-1} + 2.2$\\
	$ a_{n-1} \geq a_{n-2} + 2 \Leftrightarrow a_{n+1} \geq a_{n-2} + 3.2$\\
	$\cdots$\\
	$ a_2 \geq a_1 + 2 \Leftrightarrow a_{n+1} \geq a_1 + n.2$\\
	Cộng theo vế các bất đẳng thức trên được\\
	$na_{n+1} \geq a_n + a{n-1} + \cdots + a_1 + 2\left( 1 + 2 + \cdots + n\right) = S_n + n(n+ 1) \Leftrightarrow na_{n+1} - n(n+1) \geq S_n$\\
	$\Leftrightarrow 2\sqrt{na_{n+1} - n(n+1)} + 1 \geq 2\sqrt{S_n} + 1$. Ta chứng minh $a_{n+1} \geq 2\sqrt{na_{n+1} - n(n+1)} + 1$.\\
	Thậy vậy $a_{n+1} \geq 2\sqrt{na_{n+1} - n(n+1)} + 1 \Leftrightarrow a_{n+1}^2 - 2a_{n+1} + 1 \geq 4na_{n+1} - 4n(n+1)$\\
	$ \Leftrightarrow a_{n+1}^2 - 2(2n+1)a_{n+1} + (2n+1)^2 \geq 0 \Leftrightarrow (a_{n+1} - 2n - 1)^2 \geq 0$ luôn đúng, do đó $(*)$ đúng.\\
	Vậy với $n \in N^*$ thì ta luôn có $\sqrt{S_{n+1}} - \sqrt{S_n} \geq 1$ suy ra luôn tồn tại ít nhất $1$ số nguyên dương thỏa mãn $\sqrt{S_n} \leq k \leq \sqrt{S_{n+1}} \Leftrightarrow S_n \leq k^2 \leq S_{n+1}$ hay luôn tồn tại ít nhất một số chính phương $b = k^2$ với $(k \mbox{nguyên dương})$ thỏa mãn $S_n\leq b \leq S_{n+1}$.
	}
