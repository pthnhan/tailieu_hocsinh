	\loigiai{
	 Điều kiện để hai đường thẳng cắt nhau là $m^2+1\neq 1 \Leftrightarrow m \neq 0.$\\
	  	Thay $x=-1$ vào hàm số $y=x-3$ ta được $y=-4.$\\
	  	Thay $x=-1;\ y =-4$ vào hàm số $y=(m^2+1)x=2m-3$ ta được
	  	\begin{align*}
	  		-4=-m^2-1+2m-3 \Leftrightarrow m(m-2)=0 \Leftrightarrow 
	  		\left[\begin{aligned}
	  			&m=0\ (\text{loại})\\
	  			&m=2\ (\text{nhận})
	  		\end{aligned}\right.
	  	\end{align*}
	  	Vậy, với $m=2$ thì hai đường thẳng $y=x-3$ và $y=(m^2+1)x+2m-3$ cắt nhau tại điểm có hoành độ bằng $-1$.  		  
	}

\begin{ex}%[Đề thi tuyển sinh lớp 10  năm học 2018-2019 tỉnh Hải Dương]%[Nguyễn Quang Hiệp, dự án  EX-THCS06]%[9D4B8]
		Quãng đường tỉnh Hải Dương  - Hạ Long dài $100$ km. Một ô tô đi từ tỉnh Hải Dương  đến Hạ Long rồi nghĩ ở đó $8$ giờ $20$ phút, sau đó trở về tỉnh Hải Dương  hết tất cả $12$ giờ. Tính vận tốc của ô tô lúc đi, biết vận tốc ô tô lúc về nhanh hơn vận tốc ô tô lúc đi $10$ km/h.	
		\loigiai{
			Gọi vận tốc lúc đi của ô tô là $x$ km/h $(x>0)$.\\
			Vận tốc lúc về là $x+10$ km/h.\\
			Thời gian lúc đi là $\dfrac{100}{x}$h.\\
			Thời gian lúc về là $\dfrac{100}{x+10}$h. \\
			Theo đề ta có phương trình
			\begin{align*}
					&\dfrac{100}{x}+\dfrac{100}{x+10}+\dfrac{25}{3}=12\\ 
					\Leftrightarrow &\dfrac{100x+1000}{x(x+10)}+\dfrac{100x}{x(x+10)}=\dfrac{11}{3}\\
					\Rightarrow &600x+3000=11x^2+110x\\
					\Leftrightarrow &11x^2-490x-3000=0\\
					\Leftrightarrow &x=1\ \text{ (thỏa mãn) hoặc}\ x= -\dfrac{60}{11}\ \text{(không thỏa mãn)}.
			\end{align*}  
	Vậy vận tốc của ô tô lúc đi là $50$km/h.
		}	

\end{ex}

		Tìm $m$ để phương trình $x^2-2mx+m^2-2=0$ ($x$ là ẩn, $m$ là tham số) có hai nghiệm phân biệt $x_1,x_2$ thỏa mãn điều kiện $| x_1^3-x^3_2|=10\sqrt{2}$.
		\loigiai{
		   Ta có $\Delta' = m^2-m^2+2=2>0$ do đó phương trình có hai nghiệm phân biệt với mọi $m$.\\
		   Theo hệ thức Vi-ét ta có 
		   $\left\lbrace\begin{aligned}
		   		&x_1+x_2=2m\\
		   		&x_1x_2=m^2-2
		    \end{aligned}\right. \Rightarrow |x_1-x_2|=2 \sqrt{2}$.\\
		Ta có 
		$|x_1^3-x_2^3|=10 \sqrt{2}\Leftrightarrow |x_1-x_2|\left[(x_1+x_2)^2-x_1x_2 \right] = 10 \sqrt{2}$\\ 
		$\Rightarrow  3m^2+2=5\Leftrightarrow m=\pm 1.$ \\
		Vậy giá trị $m$ thỏa mãn yêu cầu bài toán là $m=\pm 1.$
		
		   	
		}
\end{ex}

\begin{ex}%[Đề thi tuyển sinh lớp 10  năm học 2018-2019 tỉnh Hải Dương]%[Nguyễn Quang Hiệp, dự án  EX-THCS06]%[9D4G6]
	Cho phương trình $ax^2+bx+c=0$ ($a\neq 0$) có hai nghiệm $x_1,x_2$ thỏa mãn $0 \leq x_1 \leq x_2 \leq 2$. Tìm giá trị nhỏ nhất của biểu thức $L=\dfrac{3a^2-ab+ac}{5a^2-3ab+b^2}$.
	\loigiai{
	   Theo hệ thức Vi-ét ta có 
	   $\left\lbrace\begin{aligned}
	   		&x_1+x_2=-\dfrac{b}{a}\\
	   		&x_1x_2=\dfrac{c}{a}.
	   	\end{aligned}\right.$\\

	   	Ta có $L=\dfrac{3a^2-ab+ac}{5a^2-3ab+b^2}=\dfrac{3-\dfrac{b}{a}+\dfrac{c}{a}}{5-3\dfrac{b}{a}+{\left(\dfrac{b}{a}\right)}^2}=\dfrac{3+x_1+x_2+x_1x_2}{5+3x_1+3x_2+{\left(x_1+x_2\right)}^2}$\\
	   	Theo giả thiết ta có $0\leq x_1 \leq x_2 \leq 2 \Leftrightarrow 
	   	\left\lbrace\begin{aligned}
	   	 	&x_1^2 \leq 2x_1\\
	   	 	&x_2^2 \leq 2x_2\\
	   	 	&x_1-2 \leq 0\\
	   	 	&x_2-2 \leq 0 
	   	\end{aligned}\right.
	   	\Rightarrow
	   	\left\lbrace\begin{aligned}
	   	 	&x_1^2+x_2^2\leq 2(x_1+x_2)\\
	   	 	&(x_1-2)(x_2-2)\geq 0 
	   	\end{aligned}\right.
	   	$.\\
	   	\begin{align*}
	   		\dfrac{1}{L}&=\dfrac{5+3(x_1+x_2)+x_1^2+x_2^2+2x_1x_2}{3+x_1+x_2+x_1x_2}\\
	   					&\leq \dfrac{5+3(x_1+x_2)+2(x_1+x_2)+2x_1x_2}{3+x_1+x_2+2x_1x_2}\\
	   					&=\dfrac{3(3+x_1+x_2+x_1x_2)-(x_2-2)x_1+2(x_2-2)}{3+x_1+x_2+x_1x_2}\\
	   					&=3- \dfrac{(x_1-2)(x_2-2)}{3+x_1+x_2+x_1x_2}\leq 3 \ (\text{do}\ 0\leq x_1\leq x_2 \leq 2)
	   	\end{align*}
	   	Đẳng thức xảy ra khi và chỉ khi 
	   	$\left\lbrace\begin{aligned}
	   	    &x_1^2 = 2x_1\\
	   		&x_2^2 = 2x_2\\
	   		&(x_1-2)(x_2-2)=0 
	    \end{aligned}\right. 
	    \Leftrightarrow
	    \left[\begin{aligned}
	    	&\left\lbrace\begin{aligned}
	    	 	&x_1=0\\
	    	 	&x_2=2	 
	    	\end{aligned}\right.\\
	    	&\left\lbrace\begin{aligned}
	    	 	&x_1=2\\
	    	 	&x_2=0	 
	    	\end{aligned}\right.\ (\text{loại vì}\ x_1 \leq x_2) \\
	    	&\left\lbrace\begin{aligned}
	    	 	&x_1=2\\
	    	 	&x_2=2	 
	    	\end{aligned}\right.
	    \end{aligned}\right.
	    $\\

	   	Vậy $L_{\min}=\dfrac{1}{3}$ khi $(x_1;x_2)=\lbrace (0;2),\ (2;2) \rbrace$
	}
\end{ex}
