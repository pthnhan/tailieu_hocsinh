\begin{ex}%[ChuyenSPHN-2013-V2]%[9D5K2]
\begin{enumerate}
\item Cho các số thực $a, b,c$ thỏa mãn đồng thời hai đẳng thức sau 
\begin{enumerate}
\item[i)] $\left( a + b \right)\left( b + c \right)\left( c +a \right) = abc$. 
\item[ii)] $\left( a^3 + b^3 \right)\left( c^3 + b^3 \right)\left( a^3 + c^3 \right) = a^3b^3c^3$.
\end{enumerate}
Chứng minh rằng $abc=0$.
\item  Cho các số thực dương a,b thỏa mãn  $ab > 2013a + 2014b$. Chứng minh bất đẳng thức \[a + b > \left( \sqrt {2013}  + \sqrt {2014}  \right)^2.\]
\end{enumerate}
\loigiai{
\begin{enumerate}
\item 
\begin{align*}
&\left( a^3 + b^3 \right)\left( c^3 + b^3 \right)\left( a^3 + c^3 \right) = a^3b^3c^3 \\
&\Leftrightarrow  (a+b)(b+c)(c+a)(a^2+b^2-ab)(b^2+c^2-bc)(c^2+a^2-ca)=a^3b^3c^3 \tag{1}
\end{align*}
$\bullet$ $abc=0$ dễ thấy tồn tại $a,b,c$ thỏa mãn hai điều kiện đã cho. \\
$\bullet$ $abc \ne 0$,    $(1) \Leftrightarrow (a^2+b^2-ab)(b^2+c^2-bc)(c^2+a^2-ca)=a^2b^2c^2$. \\
Ta có  $a^2+b^2-ab \geq a^2+b^2 -|ab| \geq 2|ab| -|ab| =|ab|$. \\
Chứng minh tương tự $b^2+c^2 \geq |bc|$ và $c^2+a^2 \geq |ac|$. \\
Suy ra $(a^2+b^2-ab)(b^2+c^2-bc)(c^2+a^2-ca) \geq a^2b^2c^2$, dấu $=$ xảy ra khi $a=b=c$, thay vào hai điều kiện đã cho không thỏa mãn. \\
Vậy $abc=0$. 
\item $ab > 2013a + 2014b \Leftrightarrow 1> \dfrac{2014}{a} + \dfrac{2013}{b}$. \\
Theo bất đẳng thức $B-C-S$ ta có $a+b > (a+b) \left ( \dfrac{2014}{a} + \dfrac{2013}{b}\right ) \geq \left( \sqrt {2013}  + \sqrt {2014}  \right)^2$. 
\end{enumerate}
}
\end{ex}

\begin{ex}%[ChuyenSPHN-2013-V2]%[9D3K4]
Tìm tất cả các số hữu tỷ $(x;y)$ thỏa mãn hệ phương trình  
\[\heva{&x^3-2y^3=x+4y\\&6x^2-19xy+15y^2=1.}\]
\loigiai{
Dễ thấy $y=0$ không là nghiệm của hệ. \\
$x^3-2y^3=x+4y \Leftrightarrow x^3-2y^3 =(x+4y)(6x^2-19xy+15y^2)  \Leftrightarrow 5 x^3 + 5 x^2 y - 61 x y^2 + 62 y^3 =0 \Leftrightarrow 5\left(  \dfrac{x}{y} \right)^3 +  5\left(  \dfrac{x}{y} \right)^2 - 61\left(  \dfrac{x}{y} \right) +62 =0 \Leftrightarrow \dfrac{x}{y}=2 \Leftrightarrow x=2y$. Thay vào phương trình thứ hai của hệ ta có $y^2=1$.  \\
Vậy các cặp $(x;y)$ thỏa mãn $(2;1)$, $(-2;-1)$.
}
\end{ex}

\begin{ex}%[ChuyenSPHN-2013-V2]%[9D5G2]
Với mỗi số nguyên dương $n$, kí hiệu $S_n$ là tổng $n$ số nguyên tố đầu tiên 
$(S_1=2;S_2=2+3; S_3=2+3+5,\cdots) $. Chứng minh rằng trong dãy số $S_1; S_2; S_3;\cdots$ không tồn tại hai số hạng liên tiếp đều là số chính phương. 
\loigiai{
Kí hiệu $p_m$ là số nguyên tố thứ $m$. Giả sử tồn tại các số nguyên dương $k,l$ sao cho $S_{m-1} = k^2$, $S_m=l^2$ $(m>5)$. Ta có $l^2-k^2=p_m \Leftrightarrow (l-k)(l+k)=p_m$, $p_m$ là số nguyên tố suy ra 
\[\heva{& l-k=1\\&l+k = p_m} \Rightarrow  2l=1+p_m \Rightarrow 4S_m = (1+p_m)^2\]
Do $m >4$ nên $4S_m < 4(1+3+5+7+9+  \cdots + p_m) = 4\left ( \dfrac{1+p_m}{2}\right )^2 = (1+p_m)^2$ (mâu thuẫn).
}
\end{ex}

\begin{ex}%[ChuyenSPHN-2013-V2]%[9H3K3]%[9H3K7]
Cho tam giác không cân $ABC$ nội tiếp đường tròn $(O)$, $BD$ là phân giác góc $\widehat{ABC}$. Đường thẳng $BD$ cắt $(O)$ tại điểm thứ hai $E$. Đường tròn $(O_1)$ đường kính $DE$ cắt đường tròn $(O)$ tại điểm thứ hai $F$. 
\begin{enumerate}
\item Chứng minh rằng đường thẳng đối xứng với đường thẳng  $BF$ qua đường thẳng $BD$ đi qua trung điểm  của cạnh $AC$
\item Biết tam giác $ABC$  vuông tại $B$ góc $\widehat{BAC}=60^\circ$  và bán kính đường tròn $(O)$ bằng $R$.  Tính bán kính đường tròn  $(O_1)$ theo $R$. 
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzInit[xmin=0,xmax=8,ymin=-2,ymax=5.2]
\tkzClip
\tkzDefPoints{2/4/B,1/1/A,7/1/C}
\tkzCircumCenter(A,B,C) \tkzGetPoint{O}
\tkzDefLine[bisector](A,B,C) \tkzGetPoint{b}
\tkzInterLL(B,b)(A,C) \tkzGetPoint{D}
\tkzInterLC(B,b)(O,A) \tkzGetPoints{E}{E'}
\tkzDefMidPoint(D,E) \tkzGetPoint{O1}
\tkzDefMidPoint(A,C) \tkzGetPoint{M}
\tkzDefPointBy[symmetry=center O](E) \tkzGetPoint{G}
\tkzInterLC(G,D)(O1,D) \tkzGetPoints{F}{F'}
\tkzDrawSegments(A,B B,C C,A B,E G,F G,E B,M B,F B,G)
\tkzDrawPoints[fill=black,size=7](A,B,C,D,E,F,M,O,O1,G)
\tkzDrawCircle(O1,D)
\tkzDrawCircle(O,A)
\tkzLabelPoints[above](B,G)
\tkzLabelPoints[above right](M)
\tkzLabelPoints[below](E)
\tkzLabelPoints[below left](F,A)
\tkzLabelPoint[below left](O1){$O_1$}
\tkzLabelPoints[below right](C)
\tkzLabelPoints[right](O)
\tkzLabelPoints[above left](D)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Gọi $M$ là trung điểm của $AC$, dễ thấy $EM$ vuông góc với $AC$. Gọi $G$ là giao điểm của $EM$
và $(O)$, $\widehat{GBE} = \widehat{GMD} = 90^\circ$ cho nên tứ giác $BGMD$ nội tiếp suy ra $\widehat{MGD} =\widehat{MBD}$, mặt khác $\widehat{MGD} =\widehat{FBE} \Rightarrow \widehat{MBD} = \widehat{FBE}$, suy ra $BF$ và $BM$ đối xứng qua $BD$.
\item Từ giả thiết suy ra $M$ là tâm đường trìn ngoại tiếp tam giác $ABC$ và $AB=R$, $BC=R\sqrt{3}$. Theo tính chất phân giác $\dfrac{DA}{DC} =  \dfrac{R}{R\sqrt{3}}  \Rightarrow DC=DA\sqrt{3}$. \\
$DA+DC=2R$ suy ra $DA=(\sqrt{3}-1)R \Rightarrow MD=R-DA = (2-\sqrt{3})R \Rightarrow DE=2\sqrt{2-\sqrt{3}} R$. Vậy bán kính đường tròn ngoại tiếp $(O_1)$ bằng $\sqrt{2-\sqrt{3}} R$. 
\end{enumerate}
}
\end{ex}

\begin{ex}%[ChuyenSPHN-2013-V2]%[9D5G2]
Độ dài ba cạnh của tam giác $ABC$ là ba số nguyên tố. Chứng minh rằng diện tích của tam giác $ABC$ không thể là số nguyên.
\loigiai{
Gọi $a,b,c$ là các số nguyên tố và là độ các cạnh của tam giác $ABC$. Đặt $P=a+b+c$, kí hiệu $S$ là diện tích tam giác $ABC$. Ta có \[16S^2 = (a+b+c)(a+b-c)(a+c-b)(b+c-a)=P(P-2a)(P-2b)(P-2c) \tag{1}\]
$S$ là số nguyên dương, từ $(1)$ suy ra $P=a+b+c$ chẵn. \\
Trường hợp 1: Nếu $a,b,c$ cùng chẵn thì $a=b=c=2$ suy ra $S=\sqrt{3}$.  \\
Trường hợp 2: Nếu $a,b,c$ có một số chẵn và hai số lẻ, giả sử $a$ chẵn thì $a=2$. \\
Nếu $b \ne c$ thì $|b-c| \geq 2=a$ vô lý. \\
Nếu $b=c$ thì $S^2=b^2-1 \Rightarrow (b-S)(b+S)=1$, đẳng thức không thể xảy ra vì $b,S$ là các số nguyên dương. \\
Vậy diện tích của tam giác $ABC$ không thể là số nguyên.  
}
\end{ex}

\begin{ex}%[ChuyenSPHN-2013-V2]%[9D5G1]
Cho $a_1,a_2,\cdots,a_{11}$ là các số nguyên dương lớn hơn hoặc bằng $2$, đôi một khác nhau và thỏa mãn $a_1+a_2+\cdots+a_{11}=407$. Tồn tại hay không số nguyên dương $n$ sao cho tổng các số dư của các phép chia $n$ cho $22$ số $a_1,a_2,\cdots,a_{11} , 4a_1,4a_2,\cdots,4a_{11}$ bằng $2012$.
\loigiai{
Ta chứng minh không tồn tại  thỏa mãn đề bài. \\
Giả sử ngược lại tồn tại số tự nhiên $n$, ta luôn có tổng các số dư trong phép chia $n$  cho $a_1,a_2,\cdots,a_{11}$  không thể vượt quá $407 - 11 = 396$ và  tổng các số dư trong phép chia  $n$ cho  $4a_1,4a_2,\cdots,4a_{11}$ không thể vượt quá $4.407 - 11 = 1617$. Suy ra tổng các số dư trong phép chia $n$  cho $a_1,a_2,\cdots,a_{11} $, $ 4a_1,4a_2,\cdots,4a_{11}$  không thể vượt quá $396 + 1617 = 2013$. Kết hợp với giả thiết tổng các số dư bằng $2012$, suy ra khi chia $n$ cho $22$ số trên thì có $21$ phép chia có số dư lớn nhất và phép chia có số dư nhỏ hơn số chia $2$ đơn vị. \\
Suy ra tồn tại $a_k$  sao cho $a_k;4a_k$ thỏa mãn điều kiện trên. Khi đó $1$ trong $2$ số  $n+ 1 ;   n+ 2$ chi hết cho $a_k$  và số còn lại chia hết cho $4a_k$ . Suy ra $(n+ 1 ;   n+ 2) \geq a_k \geq 2$ điều này không đúng. Vậy không tồn tại $n$ thỏa mãn đề ra.
}
\end{ex}
