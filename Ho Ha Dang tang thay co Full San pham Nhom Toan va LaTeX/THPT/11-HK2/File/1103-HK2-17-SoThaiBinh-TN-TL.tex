Tìm các giá trị của tham số $m$ để hàm số $f(x)= \begin{cases}
    \dfrac{\sqrt{x^2+5}-3}{x-2} \quad \text{khi} \,\,x\neq 2\\ mx \quad \text{khi} \,\,x=2
\end{cases} \text{liên tục tại }x=2.$

\loigiai{
    $\bullet f(2)=2m$.\\
    $\bullet {\underset{x\to 2}{\lim} {\dfrac{\sqrt{x^2+5}-3}{x-2}}}={\underset{x\to 2}{\lim}{\dfrac{x^2-4}{(x-2)(\sqrt{x^2+5}+3)}}}={\underset{x\to 2}{\lim}\dfrac{x+2}{\sqrt{x^2+5}+3}}=\dfrac{2}{3}$.\\
    $\bullet$ Hàm số liên tục tại $x=2\Leftrightarrow {\underset{x\to 2}{\lim}{f(x)}=f(2)}\Leftrightarrow 2m=\dfrac{2}{3}\Leftrightarrow m=\dfrac{1}{3}$.    
}
\end{bt}

\begin{enumerate}
    \item Chứng minh $AE \perp SC$.
    \item Chứng minh $MD \perp (SAM)$.
    \item Tính $SA$, biết khoảng cách từ $C$ đến mặt phẳng $(SDM)$ bằng $\dfrac{a}{2}$.
\end{enumerate}

                BC \perp BA\\
                BC \perp SA

                AE \perp SB
                \end{cases} \Rightarrow AE \perp (SBC) \Rightarrow AE \perp SC$.

                DM \perp MA\\
                DM \perp SA
                \end{cases} \Rightarrow DM \perp (SAM)$.
    \end{enumerate}

\loigiai{
\vspace*{-1cm}
    \immini{Kẻ $AK \perp SM$, suy ra $AK$ là khoảng cách từ $A$ đến $(SDM)$.\\

Xét tam giác  $SAM$ có $\dfrac{1}{SA^2}+\dfrac{1}{AM^2}=\dfrac{1}{SK^2}\\
