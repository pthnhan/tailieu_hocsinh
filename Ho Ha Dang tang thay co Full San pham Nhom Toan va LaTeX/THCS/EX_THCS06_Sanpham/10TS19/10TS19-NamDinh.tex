\begin{ex}%[Đề tuyển sinh 10 năm học 2018 - 2019 Nam Định]%[Lê Minh An, dự án EXTHCS-06]%[9D2B4]

\begin{ex}%[Đề tuyển sinh 10 năm học 2018 - 2019 Nam Định]%[Lê Minh An, dự án EXTHCS-06]%[9H3B9]

}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018 - 2019 Nam Định]%[Lê Minh An, dự án EXTHCS-06]%[9D1B8]
Cho biểu thức $M=\left(\dfrac{4x}{\sqrt{x}-1}-\dfrac{\sqrt{x}-2}{x-3\sqrt{x}+2}\right)\cdot\dfrac{\sqrt{x}-1}{x^2}$, với $x>0$, $x\neq 1$, $x\neq 4$.
\begin{enumerate}
\item Rút gọn $M$.
\item Tìm $x$ để $M<4$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Điều kiện $x>0$, $x\neq 1$, $x\neq 4$. Ta có
\begin{eqnarray*}
M & = & \left(\dfrac{4x}{\sqrt{x}-1}-\dfrac{\sqrt{x}-2}{\left(\sqrt{x}-1\right)\left(\sqrt{x}-2\right)}\right)\cdot\dfrac{\sqrt{x}-1}{x^2}\\
& = & \dfrac{4x-1}{\sqrt{x}-1}\cdot\dfrac{\sqrt{x}-1}{x^2}=\dfrac{4x-1}{x^2}.
\end{eqnarray*}
\item Ta có $x>0$ nên $x^2>0$ khi đó
$$M<4\Leftrightarrow \dfrac{4x-1}{x^2}<4\Leftrightarrow 4x^2-4x+1>0\Leftrightarrow (2x-1)^2>0\Leftrightarrow x\neq \dfrac{1}{2}.$$
Vậy với $x>0$, $x\neq \dfrac{1}{2}$, $x\neq 1$, $x\neq 4$ thì $M<4$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018 - 2019 Nam Định]%[Lê Minh An, dự án EXTHCS-06]%[9D4Y6]%[9D4B6]
Cho phương trình $x^2-mx-4=0\quad (1)$ (với $m$ là tham số).
\begin{enumerate}
\item Giải phương trình (1) với $m=3$.
\item Chứng minh rằng với mọi giá trị của $m$ thì phương trình (1) luôn có hai nghiệm phân biệt $x_1$, $x_2$. Tìm $m$ để $x_1^2+x_2^2=17$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Với $m=3$ phương trình (1) trở thành $x^2-3x-4=0$.\\
Ta có $a-b+c=0$ nên phương trình có hai nghiệm $x=-1$, $x=4$.
\item Ta có $\Delta=m^2+16>0$, $\forall m$ nên phương trình (1) luôn có hai nghiệm phân biệt $x_1$, $x_2$.\\
Theo định lí Vi-ét ta có $\begin{cases}
x_1+x_2=m\\ x_1x_2=-4
\end{cases}$.\\
Mà
$$x_1^2+x_2^2=17\Leftrightarrow \left(x_1+x_2\right)^2-2x_1x_2=17\Leftrightarrow m^2+8=17\Leftrightarrow m=\pm 3.$$
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018 - 2019 Nam Định]%[Lê Minh An, dự án EXTHCS-06]%[9D3K3]
Giải hệ phương trình $\left\{\begin{array}{lr}
\sqrt{xy}-\dfrac{4}{\sqrt{xy}}=3 & (1)\\ x(1-y)+15=0. & (2)
\end{array}\right.$
\loigiai{
Điều kiện $xy>0$.\\
Đặt $\sqrt{xy}=t$, $t>0$. Phương trình $(1)$ trở thành
$$t-\dfrac{4}{t}=3\Leftrightarrow t^2-3t-4=0\Leftrightarrow \left[\begin{array}{lr}
t=-1 & \text{(loại)}\\ t=4.
\end{array}\right.$$
Với $t=4$, suy ra $\sqrt{xy}=4\Leftrightarrow xy=16$.\\
Biến đổi phương trình $(2)$ ta được
$$x-xy+15=0\Leftrightarrow x-16+15=0\Leftrightarrow x=1\Rightarrow y=16.$$
Vậy nghiệm hệ là $(1;16)$.
}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018 - 2019 Nam Định]%[Lê Minh An, dự án EXTHCS-06]%[9H3B7]%[9H3K7]
Cho đường tròn $(O)$ đường kính $AB$ và điểm $C$ trên đường tròn ($C\neq A$, $C\neq B$). Gọi $D$ là một điểm trên cung nhỏ $CB$ ($D\neq C$, $D\neq B$). $E$ là giao điểm của $AD$ và $BC$, $I$ là hình chiếu vuông góc của $E$ lên $AB$, $M$ là giao điểm thứ hai của đường thẳng $DI$ và đường tròn $(O)$.
\begin{enumerate}
\item Chứng minh $BDEI$ là tứ giác nội tiếp và $CM\perp AB$.
\item Gọi $K$ là giao điểm của $BC$ và $DM$. Chứng minh $BK\cdot CE=BC\cdot EK$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1]
\tkzDefPoints{0/0/A, 7/0/B}
\clip (-0.5,-4) rectangle (7.5,4);
\coordinate (O) at ($(A)!.5!(B)$);
\tkzDefPointBy[rotation = center O angle 35](B)\tkzGetPoint{D}
\tkzDefPointBy[rotation = center O angle 115](B)\tkzGetPoint{C}
\tkzDrawCircle[radius](O,A)
\tkzDefLine[perpendicular=through C](A,B)\tkzGetPoint{c}
\tkzInterLC(C,c)(O,A)\tkzGetPoints{M}{C}
\tkzInterLL(B,C)(A,D)\tkzGetPoint{E}
\tkzInterLL(B,C)(D,M)\tkzGetPoint{K}
\tkzInterLL(A,B)(D,M)\tkzGetPoint{I}
\draw (C)node[above]{$C$}--(M)node[below]{$M$}--(D)node[above right]{$D$}--(C)--(B)node[right]{$B$}--(A)node[left]{$A$}--(D)--(B);
\draw (E)node[above]{$E$}--(I)node[below right]{$I$};
\draw (K)node[right]{$K$};
\draw (O)node[below]{$O$};
\tkzDrawPoints[fill=black](A,B,O,C,D,M,E,K,I)
\tkzMarkRightAngles(E,I,B A,D,B)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item
\begin{itemize}
\item Xét tứ giác $BDEI$ có
\begin{itemize}
\item $\widehat{EIB}=90^\circ$ (Vì $EI\perp AB$).
\item $\widehat{EDB}=\widehat{ADB}=90^\circ$ (góc nội tiếp chắn nửa đường tròn).
\end{itemize}
Nên $\widehat{EIB}+\widehat{EDB}=180^\circ$, mà $\widehat{EIB}$, $\widehat{EDB}$ là hai góc đối nhau.\\
Vậy tứ giác $BDEI$ là tứ giác nội tiếp.
\item Tứ giác $BDEI$ là tứ giác nội tiếp nên $\widehat{IEB}=\widehat{IDB}$ (cùng nhìn $IB$).\\
Mà $\widehat{IDB}=\widehat{MCB}$ (hai góc nội tiếp cùng chắn $\wideparen{BM}$).\\
Suy ra $\widehat{IEB}=\widehat{MCB}$. Mà $\widehat{IEB}$, $\widehat{MCB}$ ở vị trí đồng vị, nên $CM\parallel EI$.\\
Mà $IE\perp AB$. Vậy $CM\perp AB$.
\end{itemize} 
\item Ta có $AB\perp CM$ nên $AB$ là đường trung trực của $CM$. Suy ra $AC=AM$ hay $\wideparen{AM}=\wideparen{AC}$.\\
Lại có $\widehat{ADC}=\dfrac{1}{2}\text{sđ}\wideparen{AC}$ và $\widehat{ADM}=\dfrac{1}{2}\text{sđ}\wideparen{AM}$.\\
Suy ra $\widehat{ADM}=\widehat{ADC}$ hay $DE$ là tia phân giác trong của $\triangle DCK$,
nên $\dfrac{CE}{EK}=\dfrac{DC}{DK}$.\hfill (1)\\
Mà $DB\perp DE$, nên $DB$ là tia phân giác ngoài đỉnh $D$ của $\triangle DCK$,
nên $\dfrac{BC}{BK}=\dfrac{DC}{DK}$.\hfill (2)\\
Từ (1) và (2) suy ra $\dfrac{CE}{EK}=\dfrac{BC}{BK}\Leftrightarrow BK\cdot CE=BC\cdot EK$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018 - 2019 Nam Định]%[Lê Minh An, dự án EXTHCS-06]%[9D4G8]
Hai đoàn đại biểu của trường A và trường B cùng tham dự một buổi hội thảo. Mỗi đại biểu của trường A lần lượt bắt tay với với từng đại biểu của trường B một lần. Tính số đại biểu của mỗi trường, biết số cái bắt tay bằng ba lần số đại biểu của cả hai trường và số đại biểu trường A nhiều hơn số đại biểu trường B.
\loigiai{
Gọi số đại biểu hai trường A, B lần lượt là $x,y$ ($x,y>0$, $x,y\in\mathbb{N}$, $x>y$).\\
Mỗi đại biểu của trường A lần lượt bắt tay với với từng đại biểu của trường B một lần, nên số cái bắt tay là $xy$.\\
Mà số cái bắt tay bằng ba lần số đại biểu của cả hai trường nên
$$xy=3(x+y)\Leftrightarrow y(x-3)=3x\,\, (>0)\Leftrightarrow y=\dfrac{3x}{x-3}=3+\dfrac{9}{x-3}.$$
Vì $x,y\in\mathbb{N}$ nên $x-3\in\text{Ư}(9)=\{\pm 1;\pm 3;\pm 9\}$.
Ta có bảng sau
\begin{center}
\begin{tabular}{|c|c|c|c|c|c|c|}
\hline 
$x-3$ & $-9$ & $-3$ & $-1$ & $1$ & $3$ & $9$ \\ 
\hline 
$x$ & $-6$ & $0$ & $2$ & $4$ & $6$ & $12$ \\ 
\hline 
$y$ & \textbf{} & \textbf{} & $-6$ & $12$ & $6$ & $4$ \\ 
\hline 
Đối chiếu ĐK & loại & loại & loại & loại & loại & thỏa mãn \\ 
\hline 
\end{tabular} 
\end{center}
Vậy số đại biểu trường A là $12$ và số đại biểu trường B là $4$.
}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018 - 2019 Nam Định]%[Lê Minh An, dự án EXTHCS-06]%[9D0G7]
Cho $x,y,z$ là các số dương. Chứng minh rằng
$$\dfrac{2\sqrt{x}}{x^3+y^2}+\dfrac{2\sqrt{y}}{y^3+z^2}+\dfrac{2\sqrt{z}}{z^3+x^2}\le\dfrac{1}{x^2}+\dfrac{1}{y^2}+\dfrac{1}{z^2}.$$
\loigiai{
Theo BĐT Cô-si (Cauchy, AM-GM) ta có
$$x^3+y^2\ge 2\sqrt{x^3y^2}=2\sqrt{x}\cdot xy;$$
$$y^3+z^2\ge 2\sqrt{y^3z^2}=2\sqrt{y}\cdot yz;$$
$$z^3+x^2\ge 2\sqrt{z^3x^2}=2\sqrt{z}\cdot zx.$$
Do đó
\[\dfrac{2\sqrt{x}}{x^3+y^2}+\dfrac{2\sqrt{y}}{y^3+z^2}+\dfrac{2\sqrt{z}}{z^3+x^2}\le\dfrac{1}{xy}+\dfrac{1}{yz}+\dfrac{1}{zx}.\tag{1}\]
Cũng theo BĐT Cô-si ta có
$$\dfrac{1}{x^2}+\dfrac{1}{y^2}\ge \dfrac{2}{xy};\quad \dfrac{1}{y^2}+\dfrac{1}{z^2}\ge \dfrac{2}{yz};\quad \dfrac{1}{z^2}+\dfrac{1}{x^2}\ge \dfrac{2}{zx}.$$
Cộng vế suy ra
\[\dfrac{1}{xy}+\dfrac{1}{yz}+\dfrac{1}{zx}\le \dfrac{1}{x^2}+\dfrac{1}{y^2}+\dfrac{1}{z^2}.\tag{2}\]
Từ (1) và (2) suy ra điều phải chứng minh. Dấu bằng xảy ra khi $x=y=z=1$.
}
\end{ex}
