\subsection{Các ví dụ}
\begin{vd}
Cho hình tròn $(I,1\,\mathrm{cm})$ nội tiếp hình vuông $ABCD$.
\begin{enumerate}
\item Tính thể tích và diện tích của hình cầu tạo thành khi quay hình tròn $(I,1\,\mathrm{cm})$ quanh một đường kính của nó.
\item Tính thể tích và diện tích toàn phần của hình trụ tạo thành khi quay hình vuông $ABCD$ quanh $OO'$, với $O$, $O'$ lần lượt là trung điểm $BC$ và $AD$. 
\end{enumerate}

\begin{enumerate}
\item Hình cầu tạo thành khi quay hình tròn $(I,1\,\mathrm{cm})$ quanh một đường kính của nó cũng có tâm là $I$ và bán kính $R=1$ cm. Do đó, thể tích của khối cần là $V=\dfrac{4}{3}\pi R^3=\dfrac{4\pi}{3}$ cm$^3$ và diện tích mặt cầu là $S=4\pi R^2=4\pi$ cm$^2$.
\item Hình trụ tạo thành khi quay hình vuông $ABCD$ quanh $OO'$ có hai đáy là hai hình tròn $(O,OB)$ và $(O',O'A)$.\\
Vì hình vuông $ABCD$ ngoại tiếp đường tròn $(I,1\,\,\mathrm{cm})$ nên $AB=BC=2$ cm.\\ Do đó $OB=1$ cm.\\
Suy ra, thể tích hình trụ là $V=\pi \cdot OB^2\cdot AB=2\pi$ cm$^3$. Diện tích toàn phần của hình trụ là $S_{\mathrm{tp}}=2\pi\cdot OB\cdot AB+2\pi\cdot OB^2=6\pi$ cm$^2$.
\end{enumerate}
}
\end{vd}

\begin{vd}%[Lê Minh An, Dự Án 9TL-04]
Cho $\triangle ABC$ đều cạnh $a$, đường cao $AH$, nội tiếp đường tròn tâm $O$. 
\begin{enumerate}
\item Tính thể tích hình nón và hình cầu tạo thành khi quay $\triangle ABC$ và đường tròn $(O)$ quanh trục $AH$, biết $a=2$ cm.
\item Tính tỉ số diện tích xung quanh hình nón và diện tích mặt cầu tạo thành khi quay $\triangle ABC$ và đường tròn $(O)$ quanh trục $AH$.
\end{enumerate}
\loigiai{
\begin{center}

\draw (0,4) node[above]{$A$};

\draw (0,0) [fill=black] circle (1pt);

\draw (-4,0)[dashed] arc (180:0:4cm and 0.87cm);

\draw (-3.45,-1.97)[dashed] arc (180:0:3.45cm and 0.6cm);
\draw [dashed] (0,4)--(-3.45,-1.97);
\draw [dashed] (0,4)--(3.45,-1.97)  circle(1.5pt) node[right]{$C$};
\draw [dashed] (0,-2)--(3.45,-1.97);
\draw [dashed] (0,-2)--(-3.45,-1.97)circle(1.5pt) node[left]{$B$};
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Hình nón tạo thành khi quay $\triangle ABC$ quanh trục $AH$ tạo thành hình nón có đáy là hình tròn tâm $O$ bán kính $HB$, chiều cao $AH$.\\
Hình cầu tạo thành khi quay hình tròn tâm $O$ ngoại tiếp $\triangle ABC$ quanh trục $AH$ là hình cầu tâm $O$ bán kính $OA$.\\
Lại có $a=2$ cm, $AH=AB\sin 60^\circ=\dfrac{a\sqrt{3}}{2}=\sqrt{3}$ cm, $HB=\dfrac{BC}{2}=\dfrac{a}{2}=1$ cm. Do $\triangle ABC$ đều nên $O$ là tâm đường tròn ngoại tiếp đồng thời là trọng tâm $\triangle ABC$, suy ra $OA=\dfrac{2}{3}AH=\dfrac{2\sqrt{3}}{3}$ cm.\\
Khi đó thể tích hình nón là $$V_{\text{nón}}=\dfrac{1}{3}\cdot AH\cdot \pi\cdot HB^2=\dfrac{\pi\sqrt{3}}{3}\,\,\mathrm{cm}^3.$$ Thể tích hình cầu $$V_{\text{cầu}}=\dfrac{4}{3}\pi\cdot OA^3=\dfrac{32\pi\sqrt{3}}{27}\,\,\mathrm{cm}^3.$$
\item Đường sinh của hình nón là $AB=a$. Diện tích xung quanh hình nón là
$$S_1=\pi\cdot HB\cdot AB=\dfrac{a^2\pi}{2}.$$
Diện tích mặt cầu là
$$S_2=4\pi\cdot OA^2=\dfrac{4a^2\pi}{2}.$$
Do đó tỉ số diện tích xung quanh hình nón và diện tích mặt cầu là
$$\dfrac{S_1}{S_2}=\dfrac{a^2\pi}{2}:\dfrac{4a^2\pi}{2}=\dfrac{1}{4}.$$
\end{enumerate}
}
\end{vd}

\begin{vd}%[Lê Minh An, Dự Án 9TL-04]
Cho hình trụ $(T)$ có hai đáy là hình tròn $(O;R)$ và $(O',R)$ và hình nón $(N)$ có đỉnh là $O'$, đáy là hình tròn $(O,R)$.
\begin{enumerate}
\item Từ miếng xốp hình trụ $(T)$, người ta gọt bỏ để tạo thành khối xốp hình nón $(N)$. Tính thể tích phần bị gọt bỏ đi. Biết $R=3$ cm và $OO'=4$ cm.
\item Nếu tăng gấp đôi bán kính $R$ thì thể tích hình trụ $(T)$ và hình nón $(N)$ thay đổi như nào?
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[>=stealth,scale=1.2]
			\coordinate (A) at (0,0);
			\coordinate (B) at (3,0);
			\coordinate (C) at (3,2);
			\coordinate (D) at ($(A)+(C)-(B)$);
			\coordinate[label=below:{$O$}] (O) at ($(A)!1/2!(B)$);
			\coordinate[label=above:{$O'$}] (O') at ($(C)!1/2!(D)$);
			\draw (A)--(D) (C)--(B);
			\draw (O') ellipse (1.5 and 0.5);
			\draw[dashed] (A)..controls (0.1,0.7) and (2.9,0.7)..(B) (A)--(B)--(O')circle(1.2pt)--(A) (O)circle(1.2pt)--(O');
			\draw (A)..controls (0.1,-0.7) and (2.9,-0.7)..(B);
			%\node[rotate=90] at (1.3,1) {$r\sqrt{3}$};
			\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Thể tích khối xốp hình trụ là $V_{\text{trụ}}=OO'\cdot \pi\cdot R^2=36\pi$ cm$^3$.\\
Thể tích khối xốp hình nón là $V_{\text{nón}}=\dfrac{1}{3}\cdot OO'\cdot \pi\cdot R^2=12\pi$ cm$^3$.\\
Vậy thể tích phần xốp bị gọt bỏ là $V=V_{\text{trụ}}-V_{\text{nón}}=24\pi$ cm$^3$.
\item Thể tích hình trụ với bán kính $R$ là $V_1=OO'\cdot \pi\cdot R^2$.\\
Thể tích hình trụ với bán kính $R'=2R$ là $V_1=OO'\cdot \pi \cdot (2R)^2=4\cdot OO'\cdot \pi\cdot R^2$.\\
Khi đó ta có $\dfrac{V_1}{V_1'}=\dfrac{1}{4}$.\\
Vậy khi tăng gấp đôi bán kính $R$ thì thể tích hình trụ tăng lên $4$ lần.\\
Thể tích hình nón với bán kính $R$ là $V_2=\dfrac{1}{3}\cdot OO'\cdot \pi\cdot R^2$.\\
Thể tích ình nón với bán kính $R'=2R$ là $V_2=\dfrac{1}{3}\cdot OO'\cdot \pi \cdot (2R)^2=\dfrac{4}{3}\cdot OO'\cdot \pi\cdot R^2$.\\
Khi đó ta có $\dfrac{V_2}{V_2'}=\dfrac{1}{4}$.\\
Vậy khi tăng gấp đôi bán kính $R$ thì thể tích hình nón tăng lên $4$ lần.
\end{enumerate}
}
\end{vd}

\begin{vd}%[Lê Minh An, Dự Án 9TL-04]

\begin{bt}%[Lê Minh An, Dự Án 9TL-04]

\begin{enumerate}
\item Khi quay hình chữ nhật $ABCD$ quanh trục $MN$ thì được khối gì? Tính thể tích của khối đó.
\item Khi quay $\triangle NAB$ quanh trục $MN$ thì được khối gì? Tính diện tích xung quanh của khối đó.
\end{enumerate}

$$S=\pi\cdot AM\cdot AN=\dfrac{\pi\sqrt{17}}{4}\,\,\mathrm{cm}^2.$$ 
\end{enumerate}

\begin{bt}%[Lê Minh An, Dự Án 9TL-04]
Cho hình tròn $(O,R)$ có diện tích bằng $4\pi$. Quay hình tròn quanh một đường kính ta được hình cầu tâm $O$ bán kính $R$.
\begin{enumerate}
\item Tính thể tích hình cầu.
\item Nếu diện tích hình tròn giảm một nửa thì diện tích của mặt cầu sẽ thay đổi như nào?
\end{enumerate}
\loigiai{
\begin{enumerate}
\item 
\immini{
Diện tích hình tròn là 
$$\pi\cdot R^2=4\pi\Leftrightarrow R=2.$$
Do đó thể tích hình cầu là
$$V=\dfrac{4}{3}\cdot \pi\cdot R^3=\dfrac{32\pi}{3}.$$
}{
\begin{tikzpicture}[scale=1,>=stealth]
\tkzDefPoints{0/0/O, 2/0/D, -2/0/C}
\clip (-2.5,-2.5) rectangle (2.5,2.5);
\tkzDrawCircle[radius](O,C)
\tkzDefLine[perpendicular=through O](C,D)\tkzGetPoint{o}
\tkzInterLC(O,o)(O,C)\tkzGetPoints{A}{B}
\draw[dashed] (A)node[below]{$A$}--(B)node[above]{$B$};
\draw[dashed] (D) arc (0:180:2cm and 0.5cm);
\draw (O)node[right]{$O$};
\draw (D) arc (0:-180:2cm and 0.5cm);
\tkzDrawPoints[fill=black](A,B,O)
\end{tikzpicture}
}
\item Diện tích mặt cầu là 
$$S=4\pi\cdot R^2=16\pi.$$
Nếu diện tích hình tròn giảm một nửa thì được tròn bán kính $R'$ và 
$$\pi\cdot R'^2=2\pi\Leftrightarrow R'=\sqrt{2}.$$
Khi đó diện tích của mặt cầu mới là
$$S'=4\pi\cdot R'^2=8\pi.$$
Suy ra $\dfrac{S}{S'}=2$. Vậy diện tích mặt cầu cũng giảm đi một nửa.
\end{enumerate}
}
