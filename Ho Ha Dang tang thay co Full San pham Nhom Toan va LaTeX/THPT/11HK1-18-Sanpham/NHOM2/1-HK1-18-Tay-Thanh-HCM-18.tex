	Giải phương trình lượng giác $ 2\cos\left( 2x+\dfrac{\pi}{4} \right)+1=0  $.
	\loigiai{
	Ta có
	\begin{align*}
	2\cos\left( 2x+\dfrac{\pi}{4} \right) +1=0&\Leftrightarrow\cos\left( 2x+\dfrac{\pi}{4} \right) =-\dfrac{1}{2}\\&\Leftrightarrow\cos\left( 2x+\dfrac{\pi}{4} \right) =\cos\dfrac{2\pi}{3}\\&\Leftrightarrow\hoac{&2x+\dfrac{\pi}{4}=\dfrac{2\pi}{3}+k2\pi\\&2x+\dfrac{\pi}{4}=-\dfrac{2\pi}{3}+k2\pi\\}\\&\Leftrightarrow\hoac{&x=\dfrac{5\pi}{24}+k\pi\\&x=-\dfrac{11\pi}{24}+k\pi\\}\ (k\in\mathbb{Z}).
	\end{align*}
	Vậy tất cả các nghiệm của phương trình là $x=\dfrac{5\pi}{24}+k\pi$, $x=-\dfrac{11\pi}{24}+k\pi$ với $k\in\mathbb{Z}$.
	}
\end{bt}

\begin{bt}%[HK1 THPT Tây Thạnh, TPHCM]%[Phan Minh Quốc Vinh - DA 11HK1-18]%[1D1K3-2]
	Giải phương trình lượng giác $ 4\sin^2x-10\sin x\cos x-\cos 2x=5  $.
	\loigiai{
	Ta có $ 4\sin^2x-10\sin x\cos x-\cos 2x=5\Leftrightarrow 2\sin^2 x-5\sin x\cos x-\cos^2x=2\ \ (1)  $
	\begin{itemize}
		\item Trường hợp $ \cos x=0\Leftrightarrow x=\dfrac{\pi}{2}+k\pi $, thay $ x $ vào phương trình $ (1) $, ta được $$ 2=2  \mbox{ (luôn đúng)}$$ suy ra $ x=\dfrac{\pi}{2}+k\pi $ là nghiệm của phương trình $ (1) $.
		\item Trường hợp $ \cos x\ne 0 $, chia hai vế phương trình $ (1) $ cho $ \cos^2x $, ta được $$ 2\tan^2x-5\tan x-1=2(1+\tan^2x)\Leftrightarrow \tan x=-\dfrac{3}{5}\Leftrightarrow x=\arctan \left(-\dfrac{3}{5} \right)+k\pi.  $$
	\end{itemize}
	Vậy nghiệm của phương trình $ (1) $ là $ x=\dfrac{\pi}{2}+k\pi $, $ x=\arctan\left(-\dfrac{3}{5} \right)+k\pi$ với $k\in\mathbb{Z}$.
	}
\end{bt}

\begin{bt}%[HK1 THPT Tây Thạnh, TPHCM]%[Phan Minh Quốc Vinh - DA 11HK1-18]%[1D1G3-6]
	Giải phương trình $ \dfrac{\sqrt{3}\cos x+\sqrt{3}(5-2\sin^2x)-\sin 2x-7\sin x}{2\cos x-1}=0 $.
	\loigiai{
	Ký hiệu $ \dfrac{\sqrt{3}\cos x+\sqrt{3}(5-2\sin^2x)-\sin 2x-7\sin x}{2\cos x-1}=0$ là phương trình $ (*) $.\\
	Điều kiện $ 2\cos x-1\ne0\Leftrightarrow x\ne\pm \dfrac{\pi}{3}+k2\pi $.\\
	Ta có
	\begin{align*}
	(*)&\Leftrightarrow \sqrt{3}\cos x-2\sin x\cos x-10\sin x+5\sqrt{3}-2\sqrt{3}\sin^2x+3\sin x=0\\&\Leftrightarrow\left( \sqrt{3}-2\sin x \right)\left( \cos x+\sqrt{3}\sin x+5 \right)=0\\&\Leftrightarrow\hoac{&\sqrt{3}-2\sin x=0\\&\cos x+\sqrt{3}\sin x=-5,\mbox{ (phương trình vô nghiệm vì }a^2+b^2<c^2)\\}\\&\Leftrightarrow\sin x=\dfrac{\sqrt{3}}{2}\\&\Leftrightarrow\hoac{&x=\dfrac{\pi}{3}+k2\pi\mbox{ (loại)}\\&x=\dfrac{2\pi}{3}+k2\pi\mbox{ (nhận)}}\ (k\in\mathbb{Z}).
	\end{align*}
	Vậy nghiệm của phương trình $ (*) $ là $ x=\dfrac{2\pi}{3}+k2\pi $ với $k\in\mathbb{Z}$.
	}
\end{bt}

	Từ các chữ số $ 0 $, $ 1 $, $ 2 $, $ 3 $, $ 4 $, $ 5 $, $ 6 $, $ 7 $ có thể lập được bao nhiêu số tự nhiên gồm $ 4 $ chữ số khác nhau?
	\loigiai{

\begin{bt}%[HK1 THPT Tây Thạnh, TPHCM]%[Phan Minh Quốc Vinh - DA 11HK1-18]%[1D2K1-1]
	Trong một lớp học gồm có $ 28 $ học sinh nam và $ 15 $ học sinh nữ. Có bao nhiêu cách gọi $ 4 $ học sinh lên bảng giải bài tập sao cho có cả học sinh nam và học sinh nữ?
	\loigiai{
		Chọn $ 4 $ học sinh sao cho có cả học sinh nam và học sinh nữ, có $ 3 $ trường hợp như sau
		\begin{itemize}
			\item Chọn được $ 1 $ nam và $ 3 $ nữ, có $ \mathrm{C}_{28}^1\cdot\mathrm{C}_{15}^3 $.
			\item Chọn được $ 2 $ nam và $ 2 $ nữ, có $ \mathrm{C}_{28}^2\cdot\mathrm{C}_{15}^2 $.
			\item Chọn được $ 3 $ nam và $ 1 $ nữ, có $ \mathrm{C}_{28}^3\cdot\mathrm{C}_{15}^1 $.
		\end{itemize}
	Vậy có $ \mathrm{C}_{28}^1\cdot\mathrm{C}_{15}^3+\mathrm{C}_{28}^2\cdot\mathrm{C}_{15}^2+\mathrm{C}_{28}^3\cdot\mathrm{C}_{15}^1=101570 $ cách chọn $ 4 $ học sinh có cả học sinh nam và học sinh nữ.
	}
\end{bt}

	\loigiai{
		Không gian mẫu $ n(\Omega)=6^5 $.\\
		Có $ 5^5 $ khả năng không lần nào mặt sáu chấm xuất hiện và có $ \mathrm{C}_5^15^4 $ khả năng mặt sáu chấm xuất hiện đúng $ 1 $ lần.\\
		Suy ra số khả năng mặt sáu chấm xuất hiện ít nhất hai lần là $ 6^5-5^5-\mathrm{C}_5^15^4=1526 $ hay $ n(A)=1526 $.\\
		Vậy xác suất cần tìm là $ P(A)=\dfrac{n(A)}{n(\Omega)}=\dfrac{1526}{7776}=\dfrac{763}{3888} $.

		Số hạng tổng quát trong khai triển của nhị thức là $$ T_{k+1}=\mathrm{C}_{21}^k\left( 2x^3 \right)\left(-\dfrac{1}{x^2} \right)^k=\mathrm{C}_{21}^k2^{21-k}(-1)^kx^{63-5k},\ \ (0\le k\le21).   $$
		$ T_{k+1} $ chứa $ x^{23} $ khi chỉ khi $ 63-5k=23\Leftrightarrow k=8 $ (nhận)\\

\begin{bt}%[HK1 THPT Tây Thạnh, TPHCM]%[Phan Minh Quốc Vinh - DA 11HK1-18]%[1D2K2-6]

	\loigiai{
		Với $ n\in\mathbb{N}^{*} $, ta có
		\begin{align*}
		\dfrac{3}{2}\mathrm{C}_{n+4}^3-\mathrm{A}_{n+2}^2=12(n+2)&\Leftrightarrow \dfrac{3}{2}\dfrac{(n+4)!}{3!(n+1)!}-\dfrac{(n+2)!}{n!}=12(n+1)\\&\Leftrightarrow \dfrac{(n+2)(n+3)(n+4)}{4}-(n+1)(n+2)=12(n+2)\\&\Leftrightarrow n^2+3n-40=0\\&\Leftrightarrow \hoac{&n=5\mbox{ (nhận)}\\&n=-8\mbox{ (loại)}.\\}
		\end{align*}
		Vậy $ n=5 $.
