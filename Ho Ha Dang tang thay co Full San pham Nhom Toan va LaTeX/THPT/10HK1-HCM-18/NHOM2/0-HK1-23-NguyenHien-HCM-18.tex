\begin{bt}%[Đề HK1, Nguyễn Hiền, Hồ Chí Minh 17-18]%[Hiếu Nguyễn]%[0D2B3-2]
Cho parabol $(P):y=ax^2+bx+c$. Xác định $a,b,c$ biết $(P)$ có trục đối xứng $x=\dfrac{3}{2}$ và đi qua hai điểm $M(-1;0), N(2;6)$.
\loigiai{
Từ giả thiết ta có hệ phương trình $\heva{&-\dfrac{b}{2a}=\dfrac{3}{2}\\ & a-b+c=0\\& 4a+2b+c=6} \Leftrightarrow \heva{&3a+b=0 \\& a-b+c=0 \\ &4a+2b+c=6}\Leftrightarrow \heva{&a=-1\\& b=3\\& c=4.}$
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Hiền, Hồ Chí Minh 17-18]%[Hiếu Nguyễn]%[0D3B2-2]
Giải phương trình $|3x+1|=|x-1|$.
\loigiai{
$|3x+1|=|x-1|\Leftrightarrow \hoac{&3x+1=x-1 \\ &3x+1=1-x} \Leftrightarrow \hoac{&2x=-2\\&4x=0}\Leftrightarrow  \hoac{&x=-1\\&x=0.}$
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Hiền, Hồ Chí Minh 17-18]%[Hiếu Nguyễn]%[0D3K2-4]
Giải phương trình $(x-2)\sqrt{1-2x}=x^2-4$.
\loigiai{
Với điều kiện $x\le \dfrac{1}{2}$ thì phương trình đã cho tương đương với 
$$(x-2)\sqrt{1-2x}-(x-2)(x+2)=0 \Leftrightarrow (x-2)\left(\sqrt{1-2x}-x-2\right)=0 \Leftrightarrow \hoac{&x-2=0 & (1)\\&\sqrt{1-2x}=x+2. &(2)}$$
Do $x\le \dfrac{1}{2}\Rightarrow x-2<0$ nên phương trình $(1)$ vô nghiệm.\\
Phương trình $(2)\Leftrightarrow \heva{&x+2\ge 0 \\& 1-2x=x^2+4x+4}\Leftrightarrow \heva{&x\ge -2\\&x^2+6x+3=0}\Leftrightarrow \heva{&x\ge -2 \\ &\hoac{&x=-3+\sqrt{6}\\&x=-3-\sqrt{6}.}} $\\
So sánh với điều kiện ta thấy $x=-3+\sqrt{6}$ thỏa mãn.\\
Vậy phương trình đã cho có đúng một nghiệm là $x=-3+\sqrt{6}$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Hiền, Hồ Chí Minh 17-18]%[Hiếu Nguyễn]%[0D3B3-4]
Giải hệ phương trình $\heva{&x^2+xy+y^2=7\\&x+y=3.}$
\loigiai{
Hệ đã cho tương đương với
$$\heva{&(x+y)^2-xy=7\\&x+y=3}\Leftrightarrow \heva{&9-xy=7\\&x+y=3} \Leftrightarrow \heva{&x+y=3\\&xy=2.}$$
Suy ra $x,y$ là hai nghiệm của phương trình $t^2-3t+2=0 \Leftrightarrow \hoac{&t=1\\&t=2.}$\\
Vậy hệ có nghiệm là $\heva{&x=1\\&y=2}$; $\heva{&x=2\\&y=1.}$
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Hiền, Hồ Chí Minh 17-18]%[Hiếu Nguyễn]%[0D3B2-5]
Cho phương trình $x^2 +2(m-1)x+m^2-3m=0.$
\begin{enumerate}
\item Tìm $m$ để phương trình có một nghiệm bằng $0$. Tìm nghiệm còn lại.
\item Tìm m để phương trình có hai nghiệm phân biệt $x_1, x_2$ thỏa mãn $x_1^2 + x_2^2 = 16 $.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Phương trình có một nghiệm bằng $0$ suy ra 
			$$0^2+2(m-1)0+m^2-3m=0 \Leftrightarrow m^2-3m=0 \Leftrightarrow \hoac{&m=0\\&m=3.}$$
			\begin{itemize}
			\item Với $m=0$ ta được phương trình $x^2-2x=0 \Leftrightarrow \hoac{&x=0\\&x=2.}$
			\item Với $m=3$ ta được phương trình $x^2+4x=0 \Leftrightarrow \hoac{&x=0\\&x=-4.}$
			\end{itemize}	
\item Phương trình có hai nghiệm phân biệt khi và chỉ khi $$\Delta'=(m-1)^2-m^2+3m=m+1 >0 \Leftrightarrow m>-1.$$
	Theo định lí Vi-ét ta có $\heva{&x_1+x_2 = 2(1-m)\\& x_1x_2=m^2-3m.}$\\
	Khi đó 
	\begin{eqnarray*}
	&& x_1^2 + x_2^2 = 16 \Leftrightarrow (x_1+x_2)^2 -2x_1x_2=16 \\
	&\Leftrightarrow & 4(1-m)^2-2(m^2-3m)=16\Leftrightarrow 2(m^2-2m+1)-m^2+3m=8\\
	&\Leftrightarrow & m^2-m-6=0 \Leftrightarrow \hoac{&m=-2\\&m=3.}
	\end{eqnarray*}		
	Đối chiếu với điều kiện  $m>-1$ ta được $m=3.$

\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Hiền, Hồ Chí Minh 17-18]%[Hiếu Nguyễn]%[0D4B1-1]
Chứng minh rằng với mọi số thực $a , b ,c$ ta luôn có 
$$2 a^2 + b^2 + c^2 \ge 2a(b + c).$$
\loigiai{
Điều phải chứng minh tương đương với
$$2 a^2 + b^2 + c^2 \ge 2a(b + c) \Leftrightarrow a^2-2ab+b^2 +a^2-2ac+c^2 \ge 0 \Leftrightarrow (a-b)^2 + (a-c)^2 \ge 0 \text{ (luôn đúng).}$$
Đẳng thức xảy ra khi và chỉ khi $a=b=c.$
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Hiền, Hồ Chí Minh 17-18]%[Hiếu Nguyễn]%[0H2B2-1]%[0H1B3-2]%[0H2K2-3]
Cho tam giác $ABC$ đều cạnh $a$. Trên 3 cạnh $AB$ , $BC$ , $AC$ lần lượt lấy ba điểm $M$ ,$ N $, $P$ sao cho $\overrightarrow {AM}  = \dfrac{1}{2}\overrightarrow {AB} $, $\overrightarrow {BN}  = \dfrac{1}{3}\overrightarrow {BC} $, $\overrightarrow {AP}  = x\cdot \overrightarrow {AC} $ với $x>0 $.
\begin{enumerate}
\item Tính $\overrightarrow {AB} \cdot \overrightarrow {AC} $.
\item Chứng minh rằng : $\overrightarrow {AN}  = \dfrac{2}{3}\overrightarrow {AB}  + \dfrac{1}{3}\overrightarrow {AC} $ .
\item Tìm $x$ để $MP$ vuông góc với $AN$ .
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Theo định nghĩa tích vô hướng thì $$\overrightarrow{AB}\cdot \overrightarrow{AC}=AB\cdot AC \cdot \cos \left(\overrightarrow{AB},\overrightarrow{AC}\right)=a\cdot a\cdot \cos 60^\circ =\dfrac{a^2}{2}.$$
\item Từ giả thiết $\overrightarrow {BN}  = \dfrac{1}{3}\overrightarrow {BC} $ suy ra
$$3\left(\overrightarrow{AN}-\overrightarrow{AB}\right)=\overrightarrow{AC}-\overrightarrow{AB}\Leftrightarrow 3\overrightarrow{AN}=2\overrightarrow{AB}+\overrightarrow{AC}\Leftrightarrow \overrightarrow{AN}=\dfrac{2}{3}\overrightarrow{AB}+\dfrac{1}{3}\overrightarrow{AC}.$$
\item Ta có $\overrightarrow {AP}  = x\cdot \overrightarrow {AC} \Leftrightarrow \overrightarrow {AM}+\overrightarrow {MP}=x\cdot \overrightarrow {AC}\Leftrightarrow \overrightarrow {MP}=x\cdot \overrightarrow {AC} - \dfrac{1}{2}\overrightarrow{AB}$.\\
Do đó
$$MP\perp AN  \Leftrightarrow \left(x\cdot \overrightarrow {AC} - \dfrac{1}{2}\overrightarrow{AB}\right)\left(\dfrac{2}{3}\overrightarrow {AB}  + \dfrac{1}{3}\overrightarrow {AC} \right)=0 \Leftrightarrow \dfrac{x}{3}AC^2 - \dfrac{1}{3}AB^2 + \left(\dfrac{2x}{3}-\dfrac{1}{6}\right)\cdot \overrightarrow{AB}\cdot\overrightarrow{AC}=0.$$
Mặt khác $\overrightarrow {AB} \cdot \overrightarrow {AC} =\dfrac{a^2}{2}$ nên
$$\dfrac{x-1}{3}a^2  + \dfrac{4x-1}{12}a^2=0\Leftrightarrow \dfrac{8x-5}{12}a^2=0\Leftrightarrow x= \dfrac{5}{8}.$$
Vậy $x=\dfrac{5}{8}$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Hiền, Hồ Chí Minh 17-18]%[Hiếu Nguyễn]%[0H2B2-3]%[0H2K2-3]
Trong mặt phẳng tọa độ $ Oxy$ cho $A( 5 ; 0 )$, $B( 0 ; 1 )$ và $C(3;3)$. 
\begin{enumerate}
\item Chứng minh tam giác $ABC$ vuông. Tính diện tích tam giác $ABC$.
\item Tìm tọa độ tâm đường tròn ngoại tiếp tam giác $ABC$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Có $\overrightarrow{AB}=(-5;1)$, $\overrightarrow{AC}=(-2;3)$, $\overrightarrow{BC}=(3;2)$.\\
Ta thấy $\overrightarrow{BC}\cdot \overrightarrow{AC} = -2\cdot 3 + 3\cdot 2 = 0 \Rightarrow BC \perp AC\Rightarrow \triangle ABC$ vuông tại $C$.\\
Tam giác $ABC$ vuông tại $C$ và $AC=BC=\sqrt{2^2+3^2}=\sqrt{13}$ nên có diện tích là\\
$$S_{ABC}=\dfrac{1}{2}AC\cdot BC = \dfrac{1}{2}\cdot\sqrt{13}\cdot \sqrt{13}=\dfrac{13}{2}.$$
\item Do tam giác $ABC$ vuông tại $C$ nên tâm đường tròn ngoại tiếp tam giác là trung điểm của $AB$ và có tọa độ là $\left(\dfrac{5}{2};\dfrac{1}{2}\right).$
\end{enumerate}
}
\end{bt}
