\chapter{Tam giác đồng dạng}
\section{Các trường hợp đồng dạng của tam giác}
\subsection{Tóm tắt lí thuyết}
\begin{dl}
	Hai tam giác đồng dạng với nhau nếu
	\begin{itemize}
		\item Ba cạnh của tam giác này tương ứng tỉ lệ với ba cạnh của tam giác kia (trường hợp cạnh - cạnh - cạnh).
		\item Hai cạnh của tam giác này tương ứng tỉ lệ với hai cạnh của tam giác kia và các góc xen giữa hai cạnh ấy bằng nhau (trường hợp cạnh - góc - cạnh).
		\item Hai góc của tam giác này lần lượt bằng hai góc của tam giác kia (trường hợp góc - góc).
	\end{itemize}
\end{dl}

\begin{bt}%[Phạm An Bình, dự án EX-C2-8-2018]%[8H3K5]

			Gọi độ dài ba đường cao kẻ từ $A$, $B$, $C$ của $\triangle ABC$ lần lượt là $h_a$, $h_b$, $h_c$.\\
			Gọi tam giác $\triangle DEF$ là tam giác có độ dài các cạnh bằng độ dài ba đường cao của $\triangle ABC$.\\
			Ta có $\dfrac{1}{2}\cdot a\cdot h_a=\dfrac{1}{2}\cdot b\cdot h_b=\dfrac{1}{2}\cdot c\cdot h_c$ (cùng là diện tích $\triangle ABC$).
		}{
			\begin{tikzpicture}[scale=1, line join=round, line cap=round,>=stealth, font=\footnotesize]
			%\tkzInit[xmin=-.5, xmax=5.2, ymin=-.7, ymax=2.7]\tkzClip

		Mà $bc=a^2$ (giả thiết) nên $\dfrac{a\cdot h_a}{a^2}=\dfrac{h_b}{c}=\dfrac{h_c}{b}$.\\
		$\Rightarrow \dfrac{h_a}{a}=\dfrac{h_b}{c}=\dfrac{h_c}{b}$.\\

		$\dfrac{AB}{DE}=\dfrac{AC}{DF}=\dfrac{BC}{EF}$ (vì $\dfrac{h_a}{a}=\dfrac{h_b}{c}=\dfrac{h_c}{b}$).\\

		Vậy tam giác $ABC$ đồng dạng với tam giác có độ dài các cạnh bằng độ dài ba đường cao của tam giác $ABC$.
	}
\end{bt}

\begin{dang}{Trường hợp góc - góc}
\end{dang}
\begin{vd}%[Phạm An Bình, dự án EX-C2-8-2018]%[8H3B7]
	Cho tam giác $ABC$, đường phân giác $AD$. Giả sử $AC=b$, $AB=c$, $DB=m$, $DC=n$. Kẻ tia $Cx$ sao cho $\widehat{DCx}=\widehat{BAD}$ (tia $Cx$ khác phía với $A$ đối với $BC$).
	\begin{enumerate}
		\item Chứng minh rằng $AD\cdot DI=mn$.
		\item Chứng minh rằng $AD^2=bc-mn$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Chứng minh rằng $AD\cdot DI=mn$.
			\immini{
				Xét $\triangle ABD$ và $\triangle CID$ ta có\\
				$\widehat{BAD}=\widehat{ICD}$ (giả thiết),\\
				$\widehat{BDA}=\widehat{IDC}$ (đối đỉnh).\\
				$\Rightarrow \triangle ABD \backsim \triangle CID$ (g-g).\\
				$\Rightarrow \widehat{B}=\widehat{I}$ và $\dfrac{AD}{BD}=\dfrac{CD}{ID}$.\\
				$\Rightarrow AD\cdot DI=DB\cdot DC=mn$.\qquad (1)
			}{
				\begin{tikzpicture}[scale=.9, line join=round, line cap=round,>=stealth, font=\footnotesize]
				\tkzInit[xmin=-.5, xmax=3.2, ymin=-1.7, ymax=2.5]\tkzClip
				\tkzDefPoints{0/0/B, 1/2/A, 3/0/C}
				\tkzDefLine[bisector](B,A,C) \tkzGetPoint{a}
				\tkzInterLL(A,a)(B,C) \tkzGetPoint{D}
				\tkzFindAngle(B,A,D) \tkzGetAngle{bad}
				\tkzDefPointBy[rotation=center C angle \bad](B) \tkzGetPoint{x}
				\tkzInterLL(C,x)(A,a) \tkzGetPoint{I}
				
				\tkzDrawPolygon(A,B,C)
				\tkzDrawSegments(A,I C,x)
				\tkzLabelSegment[left](A,B){$c$}
				\tkzLabelSegment[right](A,C){$b$}
				\tkzLabelSegment(C,D){$n$}
				\tkzLabelSegment(B,D){$m$}
				\tkzDrawPoints[fill=black](A,B,C,D,I)
				\tkzLabelPoints[above](A,x)
				\tkzLabelPoints[below](B,C,I)
				\tkzLabelPoints[below left](D)
				\tkzMarkAngles[size=0.6 cm](B,A,D D,C,I)
				\tkzMarkAngles[size=0.7 cm](D,A,C)
				\end{tikzpicture}
			}
			\item Chứng minh rằng $AD^2=bc-mn$.\\
			Xét $\triangle ABD$ và $\triangle AIC$ ta có\\
			$\widehat{BAD}=\widehat{IAC}$ (giả thiết),\\
			$\widehat{B}=\widehat{I}$ (chứng minh trên).\\
			$\Rightarrow \triangle ABD \backsim \triangle AIC$ (g-g).\\
			$\Rightarrow \dfrac{AB}{AI}=\dfrac{AD}{AC}$.\\
			$\Rightarrow AD\cdot AI=AC\cdot AB=bc$.\qquad (2)\\
			Từ (1) và (2) suy ra $AD\cdot AI - AD\cdot DI=bc-mn$.\\
			$\Rightarrow AD(AI - DI)=bc-mn\Rightarrow AD^2=bc-mn$.
		\end{enumerate}
	}
\end{vd}
