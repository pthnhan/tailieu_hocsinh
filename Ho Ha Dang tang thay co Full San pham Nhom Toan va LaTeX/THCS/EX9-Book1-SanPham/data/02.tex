
\section{PHƯƠNG PHÁP DÙNG BẤT ĐẲNG THỨC}

\subsection{Phương pháp sắp thứ tự các ẩn}

\begin{vd}
   Tìm ba số nguyên dương sao cho tổng của chúng bằng tích của chúng.
    \loigiai{
        \begin{itemize}
        \item \textbf{Cách 1.} Gọi các số nguyên dương phải tìm là $x, y, z$. Ta có:
        $$x+y+z=xyz.$$
        Chú ý rằng các ẩn $x, y, z$ có vai trò bình đẳng trong phương trình nên có thể sắp thứ tự giá trị các ẩn, chẳng hạn:
        $$1\leq x\leq y\leq z.$$
        Do đó $xyz=x+y+z\leq 3$.\\
        Chia hai vế của bất đẳng thức $xyz\leq 3z$ cho số dương $z$, ta được $xy\leq 3$. Do đó $xy\in \lbrace 1;2;3\rbrace.$
        \begin{itemize}
        \item [+] Với $xy=1$, ta có $x=1$, $y=1$. Thay vào $x+y+z=xyz$ được $2+z=z$, loại.
        \item [+] Với $xy=2$, ta có $x=1$, $y=2$. Thay vào $x+y+z=xyz$ ta được $z=3$.
        \item [+] Với $xy=3$, ta có $x=1$, $y=3$. Thay vào $x+y+z=xyz$ được $z=2$, loại vì trái với sắp xếp $y\leq z$.
        \end{itemize}
        Vậy ba số phải tìm là $1; 2; 3$.
        \item \textbf{Cách 2.} Chia hai vế của $x+y+z=xyz$ cho $xyz\neq 0$, ta được
        $$\dfrac{1}{yz}+\dfrac{1}{xz}+\dfrac{1}{xy}=1.$$
        Giả sử $1\leq x\leq y\leq z$. Ta có:
        $$1=\dfrac{1}{yz}+\dfrac{1}{xz}+\dfrac{1}{xy}\leq \dfrac{1}{x^2}+\dfrac{1}{x^2}+\dfrac{1}{x^2}=\dfrac{3}{x^2}.$$
        Suy ra $\dfrac{3}{x^2}\geq 1$, do đó $x^2\leq 3$ nên $x=1$. Thay $x=1$ vào $x+y+z=xyz$ ta được
        $$\begin{aligned}
        &  1+y+z=yz\\
        \Leftrightarrow & yz-y-z=1\\
        \Leftrightarrow & y(z-1)-(z-1)=2\\
        \Leftrightarrow & (z-1)(y-1)=2.
        \end{aligned}$$
        Ta có $z-1\geq y-1\geq 0$ nên $z-1=2$ và $y-1=1$. Suy ra $(y; z)=(2; 3)$. Ba số phải tìm là $1; 2; 3$.
\end{itemize}   
\textbf{Lưu ý:} Ở cách 1, từ $xy\leq 3$ còn có thể suy ra $x^2\leq xy\leq 3$ nên $x=1$.        
    }
\end{vd}
\subsection{Phương pháp xét từng khoảng giá trị của ẩn}
\begin{vd}
Tìm nghiệm nguyên dương của phương trình
$$\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{1}{3}.$$
\loigiai{
Do vai trò bình đẳng của $x$ và $y$, giả sử $x\geq y$. Ta sẽ dùng bất đẳng thức để giới hạn khoảng giá trị của số nhỏ hơn (là $y$).\\
Hiển nhiên ta có $\dfrac{1}{y}< \dfrac{1}{3}$ nên $y>3$.\\
Mặt khác, do $x\geq y\geq 1$ nên $\dfrac{1}{x}\leq \dfrac{1}{y}$. Do đó
$$\dfrac{1}{3}=\dfrac{1}{x}+\dfrac{1}{y}\leq \dfrac{1}{y}+\dfrac{1}{y}=\dfrac{2}{y} \Rightarrow \dfrac{2}{y}\geq \dfrac{1}{3} \Rightarrow y\leq 6.$$
Do $y$ là số nguyên nên từ $y>3$ và $y\leq 6$ suy ra $y\in \lbrace 4;5;6\rbrace$.
\begin{itemize}
\item Với $y=4$ ta được $\dfrac{1}{x}=\dfrac{1}{3}-\dfrac{1}{4}=\dfrac{1}{12}$ nên $x=12$.
\item Với $y=5$ ta được $\dfrac{1}{x}=\dfrac{1}{3}-\dfrac{1}{5}=\dfrac{2}{15}$, loại vì $x$ không là số nguyên.
\item Với $y=6$ ta được $\dfrac{1}{x}=\dfrac{1}{3}-\dfrac{1}{6}=\dfrac{1}{6}$ nên $x=6$.
\end{itemize}
\textit{Đáp số:} Các nghiệm nguyên dương $(x;y)$ của phương trình đã cho là $(4; 12)$, $(12; 4)$; $(6;6)$.\\
\textbf{Lưu ý:}
\begin{itemize}
\item [a)] Để giới hạn $y\leq 6$, có thể lập luận
$$y\leq x \Rightarrow \dfrac{1}{y}\geq \dfrac{1}{x} \Rightarrow \dfrac{1}{y}\geq \left(\dfrac{1}{x}+\dfrac{1}{y}\right):2=\dfrac{1}{3}:2=\dfrac{1}{6}.$$ Vậy $y\leq 6$.
\item [b)] Cách giải đưa về phương trình ước số:
$$\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{1}{3} \Leftrightarrow \dfrac{x+y}{xy}=\dfrac{1}{3} \Leftrightarrow xy-3x-3y=0 \Leftrightarrow (x-3)(y-3)=9.$$
Sau đó, xét các ước của $9$.
\end{itemize}
\textbf{Kinh nghiệm giải toán:} Khi các ẩn trong phương trình có vai trò bình đẳng, ta thường sắp thứ tự các ẩn, sau đó dùng bất đẳng thức để giới hạn khoảng giá trị của số nhỏ.
}
\end{vd}

\subsection{Phương pháp chỉ ra nghiệm nguyên}
Phương pháp xét từng khoảng giá trị của ẩn còn được thể hiện dưới dạng chỉ ra một hoặc một vài số là nghiệm của phương trình, rồi chứng minh phương trình không còn nghiệm nào khác.
\begin{vd}
Tìm các số tự nhiên $x$ sao cho
$$2^x+3^x=5^x.$$
\loigiai{
Viết phương trình dưới dạng
$$\left(\dfrac{2}{5}\right)^x+\left(\dfrac{3}{5}\right)^x=1.$$
\begin{itemize}
\item Với $x=0$ thì vế trái phương trình trên bằng 2, loại.
\item Với $x=1$ thì vế trái phương trình trên bằng 1, đúng.
\item Với $x\geq 2$ thì $\left(\dfrac{2}{5}\right)^x<\dfrac{2}{5}$, $\left(\dfrac{3}{5}\right)^x< \dfrac{3}{5}$ nên
$$\left(\dfrac{2}{5}\right)^x+\left(\dfrac{3}{5}\right)^x<\dfrac{2}{5}+\dfrac{3}{5}=1, \text{loại}.$$
\end{itemize}
Vậy nghiệm duy nhất của phương trình là $x=1$.
}
\end{vd}


\subsection{Phương pháp sử dụng điều kiện để phương trình bậc hai có nghiệm}
Ở những phương trình có thể đưa về phương trình bậc hai đối với một ẩn, ta sử dụng điều kiện phương trình có nghiệm là $\Delta \geq 0$.
\begin{vd}
Tìm nghiệm nguyên của phương trình
$$x^2-xy+y^2=2x-y.$$
\loigiai{
Viết phương trình đã cho thành phương trình bậc hai đối với $x$ ta được
$$x^2-(y+2)x+(y^2+y)=0.$$
Điều kiện để phương trình bậc hai ẩn $x$ có nghiệm là $\Delta \geq 0$. \\Ta có $\Delta =(y+2)^2-4(y^2+y)=-3y^2+4$.
$$\Delta \geq 0 \Leftrightarrow 3y^2\leq 4.$$
Do $y\in \mathbb{Z}$ nên $y\in \lbrace 0; 1; -1\rbrace$.
\begin{itemize}
\item Với $y=0$, thay vào phương trình $x^2-xy+y^2=2x-y$ ta được $x^2-2x=0$. Ta có $x_1=0$; $x_2=2$.
\item Với $y=1$, thay vào phương trình $x^2-xy+y^2=2x-y$ ta được $x^2-3x+2=0$. Ta có $x_3=1$; $x_4=2$.
\item Với $y=-1$, thay vào phương trình $x^2-xy+y^2=2x-y$ ta được $x^2-x=0$. Ta có $x_5=0$; $x_6=1$.
\end{itemize}
\textit{Đáp số:} Phương trình có nghiệm $(x; y)$ là $(0; 0)$, $(2; 0)$; $(1; 1)$; $(2; 1)$; $(0; -1)$; $(1; -1)$.
}
\end{vd}
\textbf{Kinh nghiệm giải toán:}
Biệt số $\Delta \geq 0$ là điều kiện cần và đủ  để phương trình bậc hai có nghiệm, nhưng chỉ là điều kiện cần (chứ chưa đủ) để phương trình có nghiệm nguyên. Tuy nhiên các giá trị tìm được nói trên đều là các số nguyên nên chúng là nghiệm nguyên của phương trình. Không đòi hỏi phải thử chúng vào phương trình đã cho.
\begin{vd}
Tìm nghiệm nguyên của phương trình
$$x^2+xy+y^2=x+y.$$
\loigiai{
Viết phương trình đã cho dưới dạng phương trình bậc hai đối với $x$, ta được:
$$x^2+(y-1)x+(y^2-y)=0.$$
Điều kiện để phương trình bậc hai theo ẩn $x$ có nghiệm là $\Delta \geq 0$. \\
Ta có $\Delta =(y^2-1)^2-4(y^2-y)=-3y^2+2y+1$.
$$\Delta \geq 0 \Leftrightarrow 3y^2-2y-1\leq 0 \Leftrightarrow (3y+1)(y-1)\geq 0 \Leftrightarrow -\dfrac{1}{3}\leq y\leq 1.$$
Do $y\in \mathbb{Z}$ nên $y\in \lbrace 0; 1\rbrace$.
\begin{itemize}
\item Với $y=0$, thay vào phương trình $x^2+(y-1)x+(y^2-y)=0$ ta được $x^2-x=0$, ta có $x_1=0; x_2=1$.
\item Với $y=1$, thay vào phương trình $x^2=0$, ta có $x_3=0$.
\end{itemize}
\textit{Đáp số:} Phương trình có nghiệm $(x; y)$ là $(0; 0)$, $(0; 1)$, $(1; 0)$.\\
\textbf{Kinh nghiệm giải toán:}
Khi giải bất phương trình $\Delta \geq 0$, ta phải giải bất phương trình bậc hai
$$3y^2-2y-1\leq 0.$$
Trong lời giải trên, ta  biến đổi tương đương để đưa về bất phương trình tích $(3y+1)(y-1)\leq 0$.\\
Có nhiều cách khác để giải bất phương trình $3y^2-2y-1\leq 0$:
\begin{itemize}
\item \textbf{Cách 1:} $$3y^2-2y-1\leq 0 \Leftrightarrow 9y^2-6y-3\leq 0 \Leftrightarrow (3y-1)^2\leq 4 \Leftrightarrow -2\leq 3y-1\leq 2 \Leftrightarrow -\dfrac{1}{3}\leq y\leq 1.$$
Suy ra $y\in \lbrace 0; 1\rbrace$.
\item \textbf{Cách 2:} $3y^2-2y-1\leq 0 \Leftrightarrow y(3y-2)\leq 1$.
\begin{itemize}
\item [+] Nếu $y\geq 2$ thì $y(3y-2)\geq 2\cdot 4=8$, loại.
\item [+] Nếu $y\leq -1$ thì $y(3y-2)\geq (-1)(-5)=5$, loại
\end{itemize}
Do $y\in \mathbb{Z}$ nên $y\in \lbrace 0; 1\rbrace$.
\end{itemize}
}
\end{vd}

\begin{bt}
   Tìm nghiệm nguyên dương của phương trình $$\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{1}{4}.$$
    \loigiai{
Giả sử $1 \leq x \leq y$ thì $\dfrac{1}{x} \geq \dfrac{1}{y}$. Suy ra
$$\dfrac{1}{4}=\dfrac{1}{x}+\dfrac{1}{y} \leq \dfrac{1}{x} \Rightarrow x \leq 8.$$
Mặt khác, ta có $\dfrac{1}{x}<\dfrac{1}{4} \Rightarrow x>4$. Cho $x$ các giá trị từ $5$ đến $8$ để tìm $y$.
\begin{center}
\begin{tabular}{|c|c|c|c|c|}
\hline 
$x$ & $5$ & $6$ & $7$ & $8$ \\ 
\hline 
$y$ & $20$ & $12$ & loại & $8$ \\ 
\hline 
\end{tabular} 
\end{center}
Đáp số: $(x;y)=(5;20),\ (20;5),\ (6;12),\ (12;6),\ (8;8).$
    }
\end{bt}
\begin{bt}
Tìm ba số nguyên dương sao cho tích  của chúng gấp đôi tổng của chúng.
\loigiai{
\textbf{\textit{Cách 1.}} Ta có $xyz=2(x+y+z)$ \hfill (1)\\
Giả sử $x\leq y\leq z.$ Ta có $xyz=2(x+y+z) \leq 6z.$ Suy ra $xy \leq 6$.
\begin{itemize}
\item Xét $xy=1$, có $x=1,\ y=1$.  Thay vào (1) ta được $z=-4$, loại.
\item Xét $xy=2$, có $x=1,\ y=2$.  Thay vào (1), loại.
\item Xét $xy=3$, có $x=1,\ y=3$.  Thay vào (1) ta được $z=8$.
\item Xét $xy=4$, với $x=1,\ y=4$,  thay vào (1) ta được $z=5$; với $x=y=2$, thay vào (1) ta được $z=4$.
\item Xét $xy=5$, có $x=1,\ y=5$.  Thay vào (1) ta được $z=4$, loại vì trái với $y\leq z$.
\item Xét $xy=6$, với $x=1,\ y=6$, thay vào (1) ta được $z=\dfrac{7}{2}$, loại. Với $x=y=3$, thay vào (1) ta có $z=\dfrac{7}{2}$ loại.
\end{itemize}
\textit{Kết luận:} Bộ ba số phải tìm là $(1;\ 3;\ 8),\ (1;\ 4;\ 5),\ (2;\ 2;\ 4).$\\
\textbf{\textit{Cách 2.}}
Từ (1) ta có $\dfrac{1}{xy}+\dfrac{1}{yz}+\dfrac{1}{zx}=\dfrac{1}{2}.$ \hfill (2)\\
Giả sử $1\leq x \leq y\leq z$, từ (2) ta có $\dfrac{3}{x^2} \geq \dfrac{1}{2}$ nên $x^2 \leq 6$. Vậy $x \in \{1;\ 2\}$
\begin{itemize}
\item Với $x=1$, thay vào (1) ta được $2(1+y+z)=yz$.\\ Biến đổi về tích ta được
$z-2)(y-2)=6$. Từ đây ta tìm được $y=3,\ z=8$ và $y=4,\ z=5$.
\item Với $x=2$, thay vào (1) ta được $2+y+z=yz$.\\ Đưa về tích ta được: $(z-1)(y-1)=3.$
Giải ra ta được $y=2,\ z=4$.
\end{itemize}
 Từ đó ta được kết quả như trên.

}
\end{bt}

\textit{Tìm nghiệm nguyên của các phương trình sau $($bài $\ref{c1hu}$, $\ref{c2hu}$, $\ref{c3hu})$:}
\begin{bt}\label{c1hu}
$x^2+xy+y^2=2x+y.$
\loigiai{
Biến đổi phương trình về dạng $$x^2+(y-2)x+y^2-y=0.$$ 
Ta có $\Delta=(y-2)^2-4(y^2-y)=-3y^2+4$. Vì $\Delta \geq 0$, nên $|y| \leq \dfrac{2}{\sqrt{3}}$. Suy ra $y\in \{-1,\ 0,\ 1\}$.
\begin{itemize}
\item Với $y=-1$, ta có $x^2-3x+2=0 \Leftrightarrow x=1,\ x=2.$
\item Với $y=0$, ta có $x^2-2x=0 \Leftrightarrow x=0,\ x=2.$
\item Với $y=1$, ta có $x^2-x=0 \Leftrightarrow x=1,\ x=0.$
\end{itemize}
Vậy nghiệm của phương trình là $(x;y)=(1;-1),\ (2;-1),\ (0;0),\ (2;0),\ (0;1),\ (1;1).$
}
\end{bt}   
\begin{bt}\label{c2hu}
$x^2-3xy+3y^2=3y.$
\loigiai{
Ta viết phương trình về dạng $x^2-3xy+3y^2-3y=0.$ Ta có $\Delta =-3y(y-4).$\\ Vì $\Delta \geq 0$ nên $y\in \{0;\ 1;\ 2;\ 3;\ 4 \}$.
\begin{itemize}
\item $y=0$, ta có $x=0$.
\item $y=1$, ta có $x^2-3x=0 \Leftrightarrow x=0,\ x=3$.
\item $y=2$, ta có $x^2-6x+6=0$, phương trình này không có nghiệm nguyên.
\item $y=3$, ta có $x^2-9x+18=0\Leftrightarrow x=3,\ x=6.$
\item $y=4$, ta có $x^2-12x+36=0 \Leftrightarrow x=6.$ 
\end{itemize}
Vậy nghiệm $(x;y)$ của phương trình là $(0;0),\ (0;1),\ (3;1),\ (3;3),\ (6;3),\ (6;4).$
}
\end{bt}   
\begin{bt}\label{c3hu}
$x^2-2xy+5y^2=y+1.$
\loigiai{
Ta viết phương trình dưới dạng $x^2-2xy+5y^2-y-1=0$, ta có $\Delta'=-4y^2+y+1$. Vì $\Delta \geq 0$ nên ta có $y=0$. Thay vào phương trình ta được: $x^2-1=0 \Leftrightarrow x=\pm 1$.\\
Vậy nghiệm của phương trình là $(x;y)=(1;0),\ (-1;0).$
}
\end{bt}   
\begin{bt}
Tìm các số tự nhiên $x$ sao cho $$2^x+3^x=35.$$
\loigiai{
Ta thấy $x=3$ là một nghiệm của phương trình.
\begin{itemize}
\item Với $x>3$, ta có $\begin{cases}2^x>2^3=8 \\3^x>3^3=27 \end{cases} \Rightarrow 2^x+3^x>35$. Nên trường hợp này phương trình vô nghiệm.
\item Với $x<3$, chứng minh tương tự phương trình cũng vô nghiệm.
\end{itemize}
Vậy nghiệm của phương trình là $x=3.$
}
\end{bt}   
\begin{bt}
Tìm nghiệm nguyên dương của phương trình
$$x!+y!=(x+y)!$$
(Kí hiệu $x!$ là tích các số tự nhiên từ $1$ đến $x$).
\loigiai{
Giả sử $x \leq y$. Ta có 
$$y!(y+1)\cdots (y+x)=(x+y)!=x!+y! \leq 2y! \Rightarrow (y+1)(y+2) \cdots (y+x) \leq 2.$$
Từ đây, ta có $x=y=1.$\\
Thử lại ta thấy $x=y=1$ là nghiệm của phương trình.
}
\end{bt}   
\begin{bt}
Chứng minh rằng phương trình sau không có nghiệm nguyên dương
$$x^{17}+y^{17}=19^{17}.$$
\loigiai{
Giả sử $1\leq x \leq y<19$. Ta có 
$$19^{17}\geq (y+1)^{17}=y^{17}+17y^{16}+\cdots >y^{17}+17y^{16},$$
suy ra 
$$x^{17}+y^{17}>y^{17}+17y^{16} \geq y^{17}+17x^{16} \Rightarrow x>17.$$
Mặt khác $x^{17}<19^{17},$ nên $x<19$. Từ đó, ta có $x=y=18$. Tuy nhiên, khi đó $$x^{17}+y^{17}=2.18^{17}<19^{17}.$$
Do đó, phương trình đã cho không có nghiệm nguyên dương.
}
