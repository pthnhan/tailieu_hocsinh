\section{SỐ VÔ TỈ - CĂN BẬC HAI - SỐ THỰC}
\subsection{Tóm tắt lí thuyết}
\begin{enumerate}[\tickEX]
	\item Số vô tỉ là số viết được dưới dạng số thập phân vô hạn không tuần hoàn.
	\item Số vô tỉ và số hữu tỉ gọi chung là \textit{số thực}. Tập hợp các số thực kí hiệu là $\mathbb{R}$.\\
	Trong tập hợp các số thực cũng có các phép toán với các tính chất tương tự như các phép toán trong tập hợp các số hữu tỉ.
	\item Căn bậc hai của số không âm $a$ là số $x$ sao cho $x^2=a$.\\
	Số dương $a$ có đúng hai căn bậc hai là $\sqrt a$ và $-\sqrt a$.\\
	Số $0$ chỉ có một căn bậc hai là số $0$.
\end{enumerate}
\subsection{Các dạng toán}
\begin{dang}[Tìm căn bậc hai của một số không âm]
	Sử dụng định nghĩa: Căn bậc hai của một số $a$ không âm là số $x$ sao cho $x^2=a$.
	\begin{itemize}
	\item Với $a>0$ thì $a$ có hai căn bậc hai là $\sqrt a$ và $-\sqrt a$.
	\item Với $a=0$ thì $a$ chỉ có một căn bậc hai là $0$.
	\item Với $a<0$ thì $a$ không có căn bậc hai nào.
\end{itemize}
\end{dang}

\begin{vd}
Tìm căn bậc hai của
\begin{listEX}[4] 
	\item $25$
	\item $15$
	\item $(-1)^2$
	\item $-25$
\end{listEX}
\loigiai
{
\begin{enumerate}
	\item Vì $25>0$ và $5^2=25$,  $(-5)^2=25$ nên $25$ có hai căn bậc hai là $5$ và $-5$.
	\item Vì $15>0$ nên $15$ có hai căn bậc hai là $\sqrt{15}$ và $-\sqrt{15}$.
	\item $(-1)^2$ có hai căn bậc hai là $1$ và $(-1)$.
	\item Vì $-25<0$ nên $-25$ không có căn bậc hai nào.
\end{enumerate}
}
\end{vd}
\begin{vd}
Tìm $x$ biết $x+\sqrt{x}=0$.
\loigiai
{
Ta có $\sqrt{x}$ có nghĩa khi và chỉ khi $x\ge 0$. Suy ra: $x+\sqrt{x}\ge 0$. Dấu $``="$ xảy ra khi và chỉ khi $x=0$. Vậy $x=0$ là giá trị duy nhất thỏa mãn yêu cầu bài toán.
}
\end{vd}
\begin{vd}
Tính và so sánh
\begin{listEX}[2]
\item $\sqrt{9}\cdot \sqrt{16}$ và $\sqrt{9\cdot 16}$.
\item $\sqrt{144+25}$ và $\sqrt{144}+\sqrt{25}$.
\end{listEX}
\loigiai{
\begin{enumerate}
	\item Vì $\sqrt{9} \cdot \sqrt{16} = 3 \cdot 4 = 12$ và $\sqrt{9 \cdot 16} = \sqrt{144}=12$ nên $ \sqrt{9} \cdot \sqrt{16}= \sqrt{9 \cdot 16}$.
	\item Vì $\sqrt{144+25} = \sqrt{169} = 13$ và $\sqrt{144} + \sqrt{25} = 12+5 = 17$ nên $\sqrt{144+25} < \sqrt{144} + \sqrt{25}$.
\end{enumerate}
}
\end{vd}
\begin{vd}
	Hãy so sánh $\sqrt{10+3}$ và $\sqrt{10} + \sqrt{3}$.
	\loigiai{Ta có $\sqrt{10+3} = \sqrt{13} < \sqrt{16} = 4$ và $\sqrt{10} + \sqrt{3} > \sqrt{9} + \sqrt{1} = 3+1 = 4$.\\
	Do đó $\sqrt{10+3} < \sqrt{10} + \sqrt{3}$.}
\end{vd}
\begin{vd}
Liệu rằng có thể kết luận $\sqrt{a}$ và $\sqrt{b}$ có nghĩa nếu $\sqrt{ab}$ có nghĩa không? Vì sao?
\loigiai{
Không thể kết luận $\sqrt a$ và $\sqrt b$ có nghĩa khi $\sqrt{ab}$ có nghĩa. Vì khi $a<0$ và $b<0$ thì $\sqrt{ab}$ có nghĩa nhưng $\sqrt a$ và $\sqrt b$ không có nghĩa.

}
\end{vd}
\begin{dang}[Số vô tỉ]
Để chứng minh $a$ là một số vô tỉ ta có thể chứng minh rằng $a$ không thể là số hữu tỉ. Cụ thể, ta giả sử $a = \dfrac{m}{n}$, trong đó $m, \, n \in \mathbb{Z},\, n \ne 0$ và $(m,n)=1$, từ đó lập luận để chứng tỏ rằng giả thiết này là không thể xảy ra.
\end{dang}
\begin{vd}
Chứng minh rằng $\sqrt{2}$ là số vô tỉ.
\loigiai
{
Giả sử $\sqrt{2}$ là số hữu tỉ, tức là tồn tại $m, \, n \in \mathbb{Z}, n\ne 0$ và $(m,n)=1$ sao cho $\sqrt 2 = \dfrac{m}{n}$.\\
Khi đó $2 = \dfrac{m^2}{n^2}$.\\
Suy ra $m^2 = 2n^2$. Vì $(m,n)=1$ nên $m^2$ chia hết cho $2$. Từ đây ta có $m$ chia hết cho $2$. Đặt $m = 2k$.\\
Khi đó $n^2 = 2k^2$. Từ đây ta lại có $n$ chia hết cho $2$, vô lí vì $(m,n)=1$.\\
Vậy $\sqrt 2$ không thể là số hữu tỉ. Do đó $\sqrt 2$ là số vô tỉ.
}
\end{vd}
\begin{vd}
Chứng tỏ rằng tổng của một số hữu tỉ với một số vô tỉ là một số vô tỉ.
\loigiai
{
Giải sử tồn tại số hữu tỉ $q$ và số vô tỉ $i$ sao cho $q+i = p$ là số hữu tỉ. Khi đó $i = p-q$ là hiệu của hai số hữu tỉ nên là số hữu tỉ, vô lí. Vậy ta có điều phải chứng minh.
}
\end{vd}
\begin{vd}
	Cho $A = \dfrac{2}{\sqrt{x} + 1}$. Tìm số nguyên $x$ để $A$ nhận giá trị nguyên.
	\loigiai{Điều kiện: $x \ge 0$.\\
$A \in \mathbb{Z}   \Leftrightarrow \left( \sqrt{x} + 1\right) | 2 \Leftrightarrow \left( \sqrt{x} +1 \right) \in \left\{ \pm 1; \pm 2 \right \} $.\\
Hơn nữa, vì $\sqrt{x} + 1 \ge 1$ nên $\left( \sqrt{x} + 1 \right) \in \left\{ 1;2 \right\}$.
\begin{itemize}
	\item $\sqrt{x} +1 = 1 \Leftrightarrow \sqrt{x} = 0 \Leftrightarrow x = 0$.
	\item $\sqrt{x} +1 = 2 \Leftrightarrow \sqrt{x} = 1 \Leftrightarrow x = 1$.
\end{itemize}
}
\end{vd}
\begin{dang}[So sánh số thực]
	Để so sánh hai số thực $a$, $b$ ta có thể sử dụng một số kết quả sau:
	\begin{itemize}
		\item Nếu $a>m$ và $m>b$ thì $a>b$.
		\item Nếu $a$, $b$ là hai số thực dương thì:
		\begin{enumerate}[+]
			\item $a > b \Leftrightarrow \sqrt{a} > \sqrt{b} $.
			\item $a > b\Leftrightarrow a^2 > b^2$.
		\end{enumerate}
	\end{itemize}
\end{dang}
\begin{vd}
	Hãy sắp xếp các số sau theo thứ tự tăng dần: $-\sqrt 2$; $1{,}5$; $1\dfrac{1}{4}$; $\pi$.
	\loigiai{
	Ta có: $-\sqrt 2 < 0 < 1 \dfrac{1}{4} <1  \dfrac{1}{2} = 1{,}5 < 3 < \pi$.
	Vậy $-\sqrt 2 < 1\dfrac{1}{4} < 1{,}5 < \pi$.
}
\end{vd}
\begin{vd}
	Hãy so sánh $a = 3 \sqrt{2}$ và $b = 2 \sqrt{3}$.
	\loigiai{Vì  $a^2 = \left( 3 \sqrt{2} \right) ^2=3^2 \left( \sqrt{2} \right) ^2 = 9 \cdot 2=18$ và $b^2 = \left( 2\sqrt{3} \right)^2 =2^2 \left( \sqrt{3} \right)^2=4\cdot 3 = 12$ nên $a^2 > b^2$.\break Suy ra $a > b$.}
\end{vd}
\begin{vd}
	Hãy so sánh $x = \sqrt{3} + \sqrt{6}$ và $y = \sqrt{2} + \sqrt{7}$.
	\loigiai{Ta có $x^2=\left( \sqrt{3}+ \sqrt{6} \right)^2=\left( \sqrt{3}+ \sqrt{6} \right)\cdot \left(\sqrt{3} + \sqrt{6} \right)= 9 + 2 \sqrt{18}$ và tương tự: $y^2 = 9+ \sqrt{14}$.\\
	Vì $x^2 > y^2$ nên $x > y$. }
\end{vd}

\begin{vn}
Chứng minh rằng $\sqrt{3}$ là số vô tỉ.
\loigiai{Giả sử $\sqrt{3}$ là số hữu tỉ, tức là tồn tại $m,n \, \in \mathbb{Z}$, $(m,n)=1$ sao cho $\sqrt{3} =  \dfrac{m}{n}$.\\
	Khi đó $3 = \dfrac{m^2}{n^2}$.\\
	Suy ra $m^2 = 3n^2$. Vì $(m,n)=1$ nên $m^2$ chia hết cho $3$. Từ đây ta có $m$ chia hết cho $3$. Đặt $m=3k$.\\
	Khi đó $n^2 = 3k^2$. Từ đây ta lại có $n$ chia hết cho $3$, vô lí vì $(m,n)=1$.\\
	Vậy $\sqrt{3}$ không thể là số hữu tỉ. Do đó $\sqrt{3}$ là số vô tỉ.
}
\end{vn}
\begin{vn}
	Trong các khẳng định sau, có bao nhiêu khẳng định đúng? Hãy chứng minh.
	\begin{enumerate}
	\item Tổng hai số vô tỉ là một số vô tỉ.
	\item Tích một số vô tỉ với một số hữu tỉ khác $0$ là một số vô tỉ.
	\item Thương của một số vô tỉ và một số hữu tỉ khác $0$ là một số vô tỉ.
	\end{enumerate}
\loigiai{\begin{enumerate}
		\item ``Tổng hai số vô tỉ là một số vô tỉ'' là một khẳng định sai. Chẳng hạn $\sqrt{2} + 1$ và $-\sqrt{2} + 1$ là hai số vô tỉ nhưng $\left( \sqrt{2} + 1 \right) + \left( -\sqrt{2} + 1 \right) = 2$ là số hữu tỉ.
		\item ``Tích một số vô tỉ với một số hữu tỉ khác $0$ là một số vô tỉ'' là một khẳng định đúng. Thật vậy, giả sử tồn tại số vô tỉ $i$ và số hữu tỉ $q$ (khác $0$) sao cho $i \cdot q = p$ là số hữu tỉ. Khi đó, $x = \dfrac{p}{q}$ là số hữu tỉ, vô lí. Vậy ta có điều phải chứng minh.
		\item ``Thương của một số vô tỉ và một số hữu tỉ khác $0$ là một số vô tỉ'' là một khẳng định đúng. Thật vậy, giả sử tồn tại số vô tỉ $i$ và số hữu tỉ $q$ (khác $0$) sao cho $ \dfrac{i}{q} = p$ là số hữu tỉ. Khi đó, $i = p \cdot q$ là số hữu tỉ, vô lí. Vậy ta có điều phải chứng minh.
\end{enumerate}}
\end{vn}

\begin{vn}

	\loigiai{

}
\end{vn}
\begin{vn}
	Cho $A=2018+\sqrt{2019-x}$.
	\begin{enumerate}
		\item Tìm điều kiện của $x$ để $A$ có nghĩa.
		\item Tìm giá trị của $x$ để $A$ đạt giá trị nhỏ nhất. Hãy tìm giá trị nhỏ nhất đó.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $A$ có nghĩa khi và chỉ khi $\sqrt{2019-x}$ có nghĩa. Tức là $2019-x \ge 0\Leftrightarrow x \le 2019$.
	\item Với $x \le 2019$ thì $\sqrt{2019-x} \ge 0$. Suy ra $A = 2018 + \sqrt{2019-x} \ge 2018$. Vậy giá trị nhỏ nhất của $A$ là $2018$ đạt được khi và chỉ khi $x = 2019$.
\end{enumerate}
}
\end{vn}
