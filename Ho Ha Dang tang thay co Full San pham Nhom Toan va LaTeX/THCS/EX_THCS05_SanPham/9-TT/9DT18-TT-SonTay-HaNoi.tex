\begin{ex}%[Đề thi thử Toán 9 năm học 2017-2018, Sơn Tây, Hà Nội]%[9D1B8]%[9D1B2]%[9D1K8]
	Cho biểu thức $A=\left[ \dfrac{2\sqrt{x}-9}{x-5\sqrt{x}+6}-\dfrac{\sqrt{x}+3}{\sqrt{x}-2}-\dfrac{2\sqrt{x}+1}{3-\sqrt{x}}\right] \cdot\dfrac{x-3\sqrt{x}}{x\sqrt{x}+1}$ với $x\ge 0;\ x\ne 4;\ x\ne 9$.
	\begin{enumerate}
		\item Rút gọn biểu thức $A$.
		\item Tính giá trị của biểu thức $A$ khi $x=\sqrt{3+2\sqrt{2}}+\sqrt{11-6\sqrt{2}}+12$.
		\item Tính giá trị lớn nhất của $A$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Với $x\ge 0;\ x\ne 4;\ x\ne 9$, ta có:\\
		$\begin{aligned}
			A & =\left[ \dfrac{2\sqrt{x}-9}{(\sqrt{x}-2)(\sqrt{x}-3)}-\dfrac{\sqrt{x}+3}{\sqrt{x}-2}+\dfrac{2\sqrt{x}+1}{\sqrt{x}-3}\right] \cdot\dfrac{\sqrt{x}(\sqrt{x}-3)}{(\sqrt{x}+1)(x-\sqrt{x}+1)}\\
			& = \dfrac{2\sqrt{x}-9-(x-9)+(2x-3\sqrt{x}-2)}{(\sqrt{x}-2)(\sqrt{x}-3)}\cdot\dfrac{\sqrt{x}(\sqrt{x}-3)}{(\sqrt{x}+1)(x-\sqrt{x}+1)}\\
			& = \dfrac{x-\sqrt{x}-2}{(\sqrt{x}-2)(\sqrt{x}-3)}\cdot\dfrac{\sqrt{x}(\sqrt{x}-3)}{(\sqrt{x}+1)(x-\sqrt{x}+1)}\\
			& = \dfrac{(\sqrt{x}+1)(\sqrt{x}-2)}{(\sqrt{x}-2)(\sqrt{x}-3)}\cdot\dfrac{\sqrt{x}(\sqrt{x}-3)}{(\sqrt{x}+1)(x-\sqrt{x}+1)}\\
			& = \dfrac{\sqrt{x}}{x-\sqrt{x}+1}.
		\end{aligned}$
		\item Ta có:\\
		$\begin{aligned}
			x & =\sqrt{3+2\sqrt{2}}+\sqrt{11-6\sqrt{2}}+12=\sqrt{(1+\sqrt{2})^2}+\sqrt{(3-\sqrt{2})^2}+12\\
			& =1+\sqrt{2}+3-\sqrt{2}+12=16\text{ (thỏa mãn).}
		\end{aligned}$\\
		Vậy giá trị của $A$ khi $x=16$ là $A=\dfrac{\sqrt{16}}{16-\sqrt{16}+1}=\dfrac{4}{13}.$
		\item Khi $x=0$, ta có $A=0$.\\
		Khi $x>0;\ x\ne 4;\ x\ne 9$, ta có $A=\dfrac{\sqrt{x}}{x-\sqrt{x}+1}=\dfrac{1}{\sqrt{x}-1+\dfrac{1}{\sqrt{x}}}.$\\
		Áp dụng bất đẳng thức Cô-si, ta có:
		$$\sqrt{x}+\dfrac{1}{\sqrt{x}}\ge 2\Rightarrow \sqrt{x}+\dfrac{1}{\sqrt{x}}-1\ge 1\Rightarrow A\le 1.$$
		Đẳng thức xảy ra khi và chỉ khi $\sqrt{x}=\dfrac{1}{\sqrt{x}}\Leftrightarrow x=1$ (thỏa mãn).\\
		Vậy $\max A=1$ khi và chỉ khi $x=1$.
	\end{enumerate}	
	}
\end{ex}
\begin{ex}%[Đề thi thử Toán 9 năm học 2017-2018, Sơn Tây, Hà Nội]%[9D3B6]
	\textit{Giải bài toán bằng cách lập phương trình hoặc hệ phương trình}\\
	Một ô tô dự định đi từ $A$ đến $B$ trong khoảng thời gian nhất định. Biết rằng, nếu vận tốc giảm đi $10$ km/h thì ô tô đến $B$ chậm hơn $96$ phút so với dự định. Nếu vận tốc tăng thêm $20$ km/h thì ô tô đến sớm hơn dự định $2$ giờ. Tính độ dài quãng đường $AB$.
	\loigiai{
	Gọi thời gian dự định đi từ $A$ đến $B$ là $x$ (giờ) ($x>2$).\\
	Gọi vận tốc dự định là $y$ (km/h) ($y>10$).\\
	Vậy quãng đường $AB$ là $xy$ (km).\\
	Nếu vận tốc giảm đi $10$ km/h thì ô tô đến $B$ chậm hơn $96$ phút $=\dfrac{8}{5}$ giờ so với quy định nên ta có phương trình $$(y-10)\left( x+\dfrac{8}{5}\right) =xy.$$
	Nếu vận tốc tăng thêm $20$ km/h thì ô tô đến sớm hơn dự định $2$ giờ nên ta có phương trình $$(y+20)(x-2)=xy.$$
	Vậy ta có hệ phương trình\\
	$\begin{cases}
	(y-10)\left( x+\dfrac{8}{5}\right) =xy\\ (y+20)(x-2)=xy
	\end{cases}\Leftrightarrow \begin{cases}
	-10x+\dfrac{8}{5}y=16\\ 20x-2y=40
	\end{cases}\Leftrightarrow \begin{cases}
	x=8\\ y=60.
	\end{cases}$ (thỏa mãn điều kiện).\\
	Vậy quãng đường $AB$ là $480$ km.
	}
\end{ex}
\begin{ex}%[Đề thi thử Toán 9 năm học 2017-2018, Sơn Tây, Hà Nội]%[9D3B4]
	Giải hệ phương trình sau $\begin{cases}
	\dfrac{2}{x-1}+\sqrt{y+1}&=0\\ \dfrac{3}{x-1}-2\sqrt{y+1}&=-7.
	\end{cases}$
	\loigiai{
	Điều kiện xác định: $\begin{cases}
	x\ne 1\\ y\ge -1.
	\end{cases}$\\
	Đặt $a=\dfrac{1}{x-1};\ b=\sqrt{y+1}$.\\
	Hệ đã cho $\Leftrightarrow \begin{cases}
		2a+b=0\\ 3a-2b=-7
		\end{cases}\Leftrightarrow \begin{cases}
		a=-1\\ b=2
	\end{cases}\Leftrightarrow \begin{cases}
		\dfrac{1}{x-1}=-1\\ \sqrt{y+1}=2
	\end{cases}\Leftrightarrow \begin{cases}
		x=0\\ y=3
	\end{cases}$ (thỏa mãn).\\
	Vậy hệ đã cho có nghiệm duy nhất $(x;y)$ là $(0;3)$.
	}
\end{ex}
\begin{ex}%[Đề thi thử Toán 9 năm học 2017-2018, Sơn Tây, Hà Nội]%[9D4B2]%[9D4K6]
	Trong mặt phẳng tọa độ $Oxy$ cho $(P):y=x^2$ và $(d):y=2(m-1)x-m^2+3m$.
	\begin{enumerate}
		\item Với $m=3$, tìm tọa độ giao điểm của $(P)$ và $(d)$.
		\item Tìm $m$ để $(d)$ cắt $(P)$ tại hai điểm phân biệt có hoành độ lần lượt là chiều dài và chiều rộng của hình chữ nhật có diện tích bằng $\dfrac{7}{4}$.
	\end{enumerate}
	\loigiai{
	Xét phương trình hoành độ giao điểm của $(P)$ và $(d)$, ta có $$x^2=2(m-1)x-m^2+3m\Leftrightarrow x^2-2(m-1)x+m^2-3m=0\quad (1)$$
	\begin{enumerate}
		\item Với $m=3$, phương trình $(1)$ trở thành $x^2-4x=0\Leftrightarrow \left[\begin{aligned}
			& x=0\Rightarrow y=0\\ & x=4\Rightarrow y=16.
		\end{aligned}\right.$\\
		Vậy khi $m=3$ thì $(d)$ cắt $(P)$ tại hai điểm phân biệt $O(0;0)$ và $A(4;16)$.
		\item $(d)$ cắt $(P)$ tại hai điểm phân biệt có hoành độ lần lượt là chiều dài và chiều rộng của hình chữ nhật có diện tích bằng $\dfrac{7}{4}$ khi và chỉ khi phương trình $(1)$ có hai nghiệm dương phân biệt $x_1,x_2$ thỏa mãn $x_1x_2=\dfrac{7}{4}$.\\
		Phương trình $(1)$ có hai nghiệm dương phân biệt\\
		$\Leftrightarrow \begin{cases}
			(m-1)^2-(m^2-3m)>0\\ 2(m-1)>0\\ m^2-3m>0
		\end{cases}\Leftrightarrow \begin{cases}
			m+1>0\\ m>1\\ m(m-3)>0
		\end{cases}\Leftrightarrow \begin{cases}
			m>1\\ m>3
		\end{cases}\Leftrightarrow m>3.$\\
		Với $m>3$, áp dụng hệ thức Vi-ét ta có $x_1x_2=m^2-3m$.\\
		Vậy $x_1x_2=\dfrac{7}{4}\Leftrightarrow m^2-3m=\dfrac{7}{4}\Leftrightarrow 4m^2-12m-7=0\Leftrightarrow \left[\begin{aligned}
			& m=\dfrac{7}{2}\text{ (thỏa mãn)}\\ & m=\dfrac{-1}{2}\text{ (loại)}.
		\end{aligned}\right.$\\
		Vậy $m=\dfrac{7}{2}$ là giá trị cần tìm.
	\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi thử Toán 9 năm học 2017-2018, Sơn Tây, Hà Nội]%[9H3Y7]%[9H3B3]%[9H3K3]%[9H3G4]
	Cho đường tròn $(O)$ đường kính $AB$ cố định, điểm $I$ nằm giữa $A$ và $O$ sao cho $AI=\dfrac{2}{3}AO$. Kẻ dây cung $MN$ vuông góc với $AB$ tại $I$, gọi $C$ là điểm tùy ý thuộc cung lớn $MN$ sao cho $C$ không trùng với $M,\ N$ và $B$. Nối $AC$ cắt $MN$ tại $E$.
	\begin{enumerate}
		\item Chứng minh $4$ điểm $I,E,C,B$ cùng thuộc một đường tròn.
		\item Chứng minh hai tam giác $AME$ và $ACM$ đồng dạng.
		\item Chứng minh $AE\cdot AC-AI\cdot IB=AI^2$.
		\item Hãy xác định vị trí điểm $C$ sao cho khoảng cách từ $N$ đến tâm đường tròn ngoại tiếp tam giác $CME$ là nhỏ nhất.
	\end{enumerate}
	\loigiai{
	\begin{center}
		\begin{tikzpicture}[scale=.83]
		\tkzInit[ymin=-4,ymax=4,xmin=-1,xmax=7]
		\tkzClip
		\tkzDefPoints{0/0/A,6/0/B}
		\tkzDefMidPoint(A,B)\tkzGetPoint{O}
		\tkzDrawCircle[radius](O,A)
		\tkzDefPointBy[homothety=center A ratio 2/3](O)\tkzGetPoint{I}
		\tkzDefLine[perpendicular=through I](B,A) \tkzGetPoint{d}
		\tkzInterLC(d,I)(O,A) \tkzGetPoints{N}{M}
		\tkzDefPointBy[rotation= center O angle 30](B) \tkzGetPoint{C}
		\tkzInterLL(A,C)(M,N) \tkzGetPoint{E}
		\tkzCircumCenter(C,M,E)\tkzGetPoint{J}
		\tkzDrawCircle[radius](J,M)
		\tkzDefLine[perpendicular=through N](B,M) \tkzGetPoint{e}
		\tkzInterLL(N,e)(M,B) \tkzGetPoint{H}
		
		\tkzLabelPoints[above](M,O) \tkzLabelPoints[below](N,J)
		\tkzLabelPoints[left](A) \tkzLabelPoints[right](B,H)
		\tkzLabelPoints[above right](C,E)
		\tkzLabelPoints[below left](I)
		\tkzDrawPoints[fill=black](A,B,I,M,N,O,E,C,J,H)
		\tkzDrawSegments(A,B M,N A,M A,C M,C B,C M,B N,H)
		\tkzMarkRightAngles[size=0.2](M,I,B N,H,B)
		\end{tikzpicture}
	\end{center}
	\begin{enumerate}
		\item Vì $\widehat{ACB}$ là góc nội tiếp chắn nửa đường tròn nên $\widehat{ACB}=90^\circ$.\\
		Xét tứ giác $EIBC$ có $\widehat{EIB}+\widehat{ECB}=180^\circ$.\\
		Mà hai góc này ở vị trí đối nhau nên $EIBC$ là tứ giác nội tiếp hay $4$ điểm $I,E,C,B$ cùng thuộc một đường tròn.
		\item Vì $MN\perp AB$ và $AB$ là đường kính của $(O)$ nên $A$ là điểm chính giữa cung nhỏ $MN$ hay $\wideparen{AM}=\wideparen{AN}$.\\
		Suy ra $\widehat{AMN}=\widehat{ACM}$ (hai góc nội tiếp của $(O)$ chắn hai cung bằng nhau).\\
		Xét $\triangle AME$ và $\triangle ACM$ có:\\
		$\begin{cases}
		\widehat{AME}=\widehat{ACM}\\ \widehat{A}\text{ chung}
		\end{cases}$\\
		$\Rightarrow \triangle AME\backsim\triangle ACM (g-g)$.
		\item Vì $\triangle AME\backsim\triangle ACM\Rightarrow \dfrac{AE}{AM}=\dfrac{AM}{AC}\Rightarrow AE\cdot AC=AM^2$.\\
		Do $\widehat{AMB}$ là góc nội tiếp chắn nửa đường tròn nên $\widehat{AMB}=90^\circ$.\\
		Xét $\triangle AMB$ vuông tại $M$ có $MI\perp AB$ nên $AI\cdot IB=MI^2$.\\
		Vậy $AE\cdot AC-AI\cdot IB=AM^2-MI^2=AI^2$ (theo định lí Pi-ta-go cho $\triangle AMI$).
		\item Gọi $J$ là tâm đường tròn ngoại tiếp tam giác $CME$.\\
		Ta có $\triangle AME\backsim\triangle ACM\Rightarrow \widehat{AME}=\widehat{ACM}$.\\
		Mà $\widehat{MCE}$ là góc nội tiếp chắn $\wideparen{ME}$ của đường tròn $(J)$ nên $\widehat{MCE}=\dfrac{1}{2}$sđ$\wideparen{ME}$.\\
		$\Rightarrow \widehat{AME}=\dfrac{1}{2}$sđ$\wideparen{ME}$.\\
		Vậy $AM$ là tiếp tuyến của đường tròn $(J)$ (theo định lý đảo về góc tạo bởi tiếp tuyến và dây cung).\\
		Suy ra $AM\perp JM$.\\
		Mà $AM\perp BM$ nên $J\in BM$.\\
		Gọi $H$ là hình chiếu vuông góc của $N$ lên $BM$.\\
		Ta có $NJ\ge NH$ nên $NJ$ nhỏ nhất khi $J\equiv H$.\\
		Khi đó $C$ là giao điểm thứ hai của đường tròn $(O)$ và $(H;HM)$.
	\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi thử Toán 9 năm học 2017-2018, Sơn Tây, Hà Nội]%[9D5G1]
	Cho các số dương $a,b,c$ thỏa mãn $a^2+b^2+c^2=1$. Tìm giá trị nhỏ nhất của biểu thức $$P=\dfrac{ab}{c}+\dfrac{bc}{a}+\dfrac{ca}{b}.$$
	\loigiai{
	Ta có $P=\dfrac{ab}{c}+\dfrac{bc}{a}+\dfrac{ca}{b}\Rightarrow P^2=\dfrac{a^2b^2}{c^2}+\dfrac{b^2c^2}{a^2}+\dfrac{c^2a^2}{b^2}+2(a^2+b^2+c^2).$\\
	Áp dụng bất đẳng thức Cô-si với các số dương ta có:
	$$\begin{cases}
	\dfrac{a^2b^2}{c^2}+\dfrac{b^2c^2}{a^2}\ge 2b^2\\[0.2cm] \dfrac{b^2c^2}{a^2}+\dfrac{c^2a^2}{b^2}\ge 2c^2\\[0.2cm] \dfrac{c^2a^2}{b^2}+\dfrac{a^2b^2}{c^2}\ge 2a^2.
	\end{cases}$$
	Cộng vế với vế các bất đẳng thức trên ta có
	$$P^2\ge 3(a^2+b^2+c^2)=3\Rightarrow P\ge \sqrt{3} \text{ (do } P>0).$$
	Đẳng thức xảy ra khi và chỉ khi $a=b=c=\dfrac{\sqrt{3}}{3}$.\\
	Vậy $\min P=\sqrt{3}$ khi và chỉ khi $a=b=c=\dfrac{\sqrt{3}}{3}$.
	}
\end{ex}
