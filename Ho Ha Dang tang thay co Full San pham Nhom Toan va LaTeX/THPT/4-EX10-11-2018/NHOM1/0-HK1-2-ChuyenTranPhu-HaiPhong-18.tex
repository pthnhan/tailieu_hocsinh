\begin{ex}%[Đề HK1 T10, Chuyên Trần Phú, Hải Phòng 2017]%[0D3K2]
Cho phương trình $|x-3m+1|=|(m+1)x-3|$. Khẳng định nào dưới đây là đúng?
\choice
{Khi $m=0$ phương trình vô nghiệm}
{Khi $m=-2$ phương trình vô nghiệm}
{\True Khi $m\ne0$ và $m\ne-2$ phương trình có 2 nghiệm}
{Khi $m\ne0$ phương trình có nghiệm duy nhất}
\loigiai{
Ta có: $|x-3m+1|=|(m+1)x-3|\Leftrightarrow \hoac{&x-3m+1=(m+1)x-3\\& x-3m+1=-(m+1)x+3}\Leftrightarrow \hoac{& mx=4-3m\\& (m+2)x=3m+2}$.
Nếu $\heva{&m\ne0\\& m\ne-2}$ thì ta có nghiệm $\hoac{&x=\dfrac{4-3m}{m}\\& x=\dfrac{3m+2}{m+2}}$.
}
\end{ex}

\begin{ex}%[Đề HK1 T10, Chuyên Trần Phú, Hải Phòng 2017]%[0D3K2]
Cho phương trình $\dfrac{3m(x-1)}{x+1}=5m+1$. Khẳng định nào dưới đây là đúng?
\choice
{Khi $m\ne-\dfrac{1}{8}$ phương trình có nghiệm bằng $0$}
{Khi $m\ne-\dfrac{1}{2}$ phương trình có nghiệm duy nhất $x=-\dfrac{8m+1}{2m+1}$}
{\True Khi $\heva{&m\ne-\dfrac{1}{2}\\& m\ne0}$ phương trình có nghiệm duy nhất $x=-\dfrac{8m+1}{2m+1}$}
{Khi $m=-\dfrac{1}{2}$ phương trình có tập nghiệm $S=\mathbb R$}
\loigiai{
Điều kiện: $x\ne-1$.
$$\dfrac{3m(x-1)}{x+1}=5m+1 \Leftrightarrow 3m(x-1)=(5m+1)(x+1)\Leftrightarrow (2m+1)x=-8m-1\ (1)$$
Khi $m\ne-\dfrac{1}{2}$, phương trình $(1)$ có nghiệm $x=-\dfrac{8m+1}{2m+1}$.\\
Do $x\ne-1$ nên $-\dfrac{8m+1}{2m+1}\ne-1\Leftrightarrow 8m+1\ne 2m+1 \Leftrightarrow m\ne 0$.\\
Vậy khi $\heva{&m\ne-\dfrac{1}{2}\\& m\ne0}$ phương trình có nghiệm duy nhất $x=-\dfrac{8m+1}{2m+1}$.
}
\end{ex}

\choice
{Đường trung trực $BC$}
{Đường tròn tâm $I$, bán kính $R=\dfrac{AB}{2}$ với $I$ là đỉnh của hình bình hành $ABIC$}
{Đường thẳng song song với $BC$}
{\True Đường tròn tâm $I$, bán kính $R=\dfrac{AB}{2}$ với $I$ là đỉnh của hình bình hành $ABCI$}
\loigiai{
Gọi $I$ là điểm thỏa: $\vv{IA}+\vv{BC}=\vv{0}\Leftrightarrow \vv{IA}=\vv{CB}$, suy ra $I$ là đỉnh thứ 4 của hình bình hành $ABCI$ nên $I$ cố định.\\

Suy ra tập hợp điểm $M$ là đường tròn tâm $I$, bán kính $R=\dfrac{AB}{2}$, với $I$ là đỉnh thứ 4 của hình bình hành $ABCI$.

{\True $4$}
{$3$}
{$1$}
{$2$}

\begin{bt} (1.0 điểm) %[Đề HK1 T10, Chuyên Trần Phú, Hải Phòng 2017]%[0D3K2]%[0D3K2]
\begin{enumerate}
\item Tìm $m$ để phương trình $x^2+2x+m=0$ có 2 nghiệm phân biệt $x_1,x_2$ thỏa $x_1^2+x_2^2=6$.
\item Giải phương trình: $\sqrt{x^2+2x-6}=2x-3$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Tìm $m$ để phương trình $x^2+2x+m=0\ (*)$ có 2 nghiệm phân biệt $x_1,x_2$ thỏa $x_1^2+x_2^2=6$.
\item[] Ta có: $\Delta'=(b')^2-ac=1-m$.
\item[] Phương trình $(*)$ có 2 nghiệm phân biệt $\Leftrightarrow\heva{&a\ne0\\&\Delta'>0}\Leftrightarrow \heva{&1\ne0\\& 1-m>0}\Leftrightarrow m<1$.
\item[] Khi đó, theo định lí Vi-et ta có: $\heva{&x_1+x_2=-\dfrac{b}{a}=-2\\& x_1x_2=\dfrac{c}{a}=m}$.
\item[] Ta có: $x_1^2+x_2^2=6 \Leftrightarrow (x_1+x_2)^2-2x_1x_2=6\Leftrightarrow 4-2m=6\Leftrightarrow m=-1$ (nhận).
\item Giải phương trình: $\sqrt{x^2+2x-6}=2x-3$.
\item[] $\begin{aligned}  \sqrt{x^2+2x-6}=2x-3 & \Leftrightarrow \heva{&2x-3\ge0\\& x^2+2x-6=(2x-3)^2}  \Leftrightarrow\heva{& x\ge\dfrac{3}{2}\\& 3x^2-14x+15=0} \\& \Leftrightarrow \heva{&x\ge\dfrac{3}{2}\\& \hoac{&x=3\ \text{(nhận)}\\& x=\dfrac{5}{3}\ \text{(nhận)}}} \Leftrightarrow \hoac{&x=3\\& x=\dfrac{5}{3}}\end{aligned}$
\item[] Vậy $S=\left\{\dfrac{5}{3};3\right\}$.
\end{enumerate}
