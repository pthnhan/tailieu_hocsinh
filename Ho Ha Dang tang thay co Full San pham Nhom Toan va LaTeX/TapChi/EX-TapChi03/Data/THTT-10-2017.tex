	Tính tổng
	\[ A = 3+4+6+9+13+18+\cdots + 4953 \]
	(các số hạng trong tổng $ A $ được xác định theo công thức $ a_{n+1}  = a_n + n $ với $ n\in \mathbb{N}^* $).
	\loigiai{
		Theo giả thiết có
		{\allowdisplaybreaks
		\begin{align*}
			&a_2 = a_1 + 1\\
			&a_3 = a_2 + 2\\
			&a_4 = a_3 + 3\\
			&\ldots\\
			&a_n = a_{n-1} + n-1\\
			&a_{n+1} = a_n + n.
		\end{align*}}Cộng theo vế các đẳng thức trên được
		\[ a_2 + a_3 + \cdots + a_n + a_{n+1} = a_1 + a_2 + \cdots + a_{n-1} + a_n + 1 + 2 + \cdots + n .\]
		Suy ra $ a_{n+1} = a_1 + 1 + 2 + \cdots + n =  3 + \dfrac{n(n+1)}{2} \hfill (*)$ \\  Mặt khác $ a_{n+1} = 4953 $, từ đó giải ra $ n = 99 $. Do đó tổng $ A $ có $ 100 $ số hạng.\\
		Từ $ (*) $ cho $ n $ lần lượt bằng $ 1, 2, 3, 4, \cdots, 99 $ ta có $$ 2a_2 = 6 + 1\cdot 2; 2a_3 = 6 + 2\cdot 3; \cdots; 2a_{99} = 6 + 98\cdot 99; 2a_{100} = 6 + 99\cdot 100 .$$
		Do đó
		\[ 2A = 2a_1 + 2a_2 + \cdots + 2a_{99} + 2a_{100} = 100\cdot 6 + 1\cdot 2 + 2\cdot 3+ \cdots + 98\cdot 99 + 99\cdot 100. \]
		Đặt $ B = 1\cdot 2 + 2\cdot 3+ \cdots + 98\cdot 99 + 99\cdot 100 $ suy ra $ 3B = 1\cdot 2\cdot3 + 2\cdot 3\cdot3+ \cdots + 98\cdot 99\cdot3 + 99\cdot 100\cdot3 $. Phân tích ta thu được
		\[ 3B = 1\cdot 2\cdot3 + 2\cdot 3(4-1)+ \cdots + 98\cdot 99(100-97) + 99\cdot 100(101-98)  = 99\cdot100\cdot101. \]
		Suy ra $ B = 33\cdot 100\cdot101 $. Thay vào biểu thức $ 2A $ được $ 2A = 600 + B = 333900 $.\\ Vậy $ A = 166 950 $.
	}
\end{bt}
\begin{bt}[T2/484-Toán học \& Tuổi trẻ, tháng 10 năm 2017]%[Nguyễn Thế Út,Dự án EX_TapChi03]%[7D0K6]
	Tìm các số tự nhiên $ x $ và $ y $ thỏa mãn phương trình: $ (1+x!)\cdot (1+y!) = (x+y)! \quad (1) $ (trong đó ký hiệu $ n! = 1\cdot2\cdot3\cdots n $ ).
	\loigiai{
		\begin{itemize}
			\item Với $ x = 0 $ thay vào $ (1) $ ta có:
			\[ 2(1+y!) = y! \Rightarrow y! = -2 \]
			(loại vì $ y \in \mathbb{N} $).
			\item Với $ x = 1 $ thay vào $ (1) $ ta có:
			\[ 2(1+y!) = (y+1)! \Rightarrow y!(y-1) = 2 \Rightarrow y = 2. \]
			\item Với $ x \ge 2 $ thì $ (x+y)! $ là số chẵn, $ 1+x! $ là số lẻ. Từ $ (1) $ suy ra $ 1+y! $ phải là số chẵn, hay $ y! $ là số lẻ nên $ y = 1 $. Khi đó ta có $ 2(1+x!) = (x+1)! \Rightarrow x = 2 $.
		\end{itemize}	
		Vậy phương trình có nghiệm là $ (1;2) $ và $ (2;1) $.
	}
\end{bt}

\loigiai{
	Đánh số các hàng và các cột bởi $ 1,2,\ldots ,8 $ và giả sử không mất tính tổng quát thì ô $ (1,1) $ là ô màu đen.\\

	Ký hiệu $ C $ là số các viên sỏi trong các ô có tung độ lẻ và hoành độ là số chẵn.\\

	Chứng minh rằng với mọi số nguyên dương $n$ ta luôn có

	Từ đó suy ra 
	$$\lim\limits_{n\to+\infty }\left ( \dfrac{1}{2}+\dfrac{1}{3} +\cdots+\dfrac{1}{n}\right )=+\infty. $$
	\loigiai{ Áp dụng bất đẳng thức Cauchy ta có:
		$$\sqrt[k]{{2.\underbrace {1.1\ldots 1}_{k - 1\; \text{số }1}}} \le \dfrac{{2 + \overbrace {1 + 1 + \cdots + 1}^{k - 1\; \text{số }1}}}{k} = \dfrac{{k + 1}}{k}.$$
		Nhưng đẳng thức không xảy ra nên 
		$$\sqrt[k]{2}<\dfrac{k+1}{k}. $$
		Do đó 
		{\allowdisplaybreaks
	\begin{align*}
	\prod_{k=2}^{n}\sqrt[k]{2}<\prod_{k=2}^{n}\frac{k+1}{k}&\Leftrightarrow 2^{\tfrac{1}{2}+\tfrac{1}{3}+\cdots+\tfrac{1}{n}}<\dfrac{n+1}{2}\\&\Leftrightarrow \dfrac{1}{2}+\dfrac{1}{3}+\cdots+\dfrac{1}{n}<\log_{2}\dfrac{n+1}{2}.\tag{1}
	\end{align*}}
		Mặt khác sử dụng kết quả quen thuộc $\ln\left ( 1+x \right )<x,\;\forall x>0, $ ta có
		{\allowdisplaybreaks
	\begin{align*}
	\sum_{k=2}^{n}\ln\left ( 1+\dfrac{1}{k} \right )<\sum_{k=2}^{n}\dfrac{1}{k} 
	\quad&\Leftrightarrow \ln\prod_{k=2}^{n}\dfrac{k+1}{k}<\dfrac{1}{2}+\dfrac{1}{3}+\cdots+\dfrac{1}{n}\\ &
	\Leftrightarrow \ln\dfrac{n+1}{2}<\dfrac{1}{2}+\dfrac{1}{3}+\cdots+\dfrac{1}{n}.\tag{2}
	\end{align*}	
	}Từ $(1)$ và $(2)$ ta có điều phải chứng minh.\\
		Từ chứng minh trên và do $\lim \limits_ {n\to+\infty}\ln\dfrac{n+1}{2}=+\infty$ nên $\lim \limits_ {n\to+\infty}\left ( \dfrac{1}{2}+\dfrac{1}{3}+...+\dfrac{1}{n} \right )=+\infty.$
		
		\begin{nx}
			Ngoài cách giải trên, ta có thể giải cách khác bằng cách dựa vào hai kết quả sau:
			\[\ln\left ( 1+x \right )<x,\;\forall x>0;\tag{3}\]
			và \[ x<\log_{2}\left ( 1+x \right ),\;\forall x\in\left (0;\frac{1}{3}  \right].\tag{4}\]
			Để chứng minh $(3)$ và $(4),$ ta xét các hàm $f\left ( x \right )=x-\ln\left ( 1+x \right )$ và $g\left ( x \right )=x-\log_{2}\left ( 1+x \right ),$ rồi chứng tỏ $f'\left ( x \right )>0,\;\forall x>0;g'\left ( x \right )<0,\;\forall x\in\left( 0;\dfrac{1}{3} \right].$\\
			Từ $(3)$ và $(4)$ ta suy ra được bất đẳng thức cần chứng minh.
		\end{nx}
	}

	Cho các số thực dương $x,\;y,\;z$ thỏa mãn $\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=\dfrac{16}{x+y+z} .$ Tìm giá trị nhỏ nhất của biểu thức $$ P=\frac{x}{y}+\frac{y}{z}+\frac{z}{x}.$$
	\loigiai{ Đặt $a=\dfrac{y}{x},\;b=\dfrac{z}{y},\;c=\dfrac{x}{z}. $ ta có $a,\;b,\;c>0,\;abc=1$ và $P=\dfrac{x}{y}+\dfrac{y}{z}+\dfrac{z}{x}=bc+ca+ab. $\\
		Từ giả thiết 
		\begin{align*}
			16=\left ( x+y+z \right )\left ( \dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z} \right )&=1+\dfrac{y}{x}+\dfrac{z}{x}+\dfrac{x}{y}+1+\dfrac{z}{y}+\dfrac{x}{z}+\dfrac{y}{z}+1\\ &=1+a+ab+bc+1+b+c+ca+1. \tag{1}
		\end{align*}
		Suy ra 
		\begin{align*}
			15&=abc+ab+bc+ca+a+b+c+1\\
			&=\left ( a+1 \right )\left ( b+1 \right )\left ( c+1 \right )\\
			&=\left ( a+1 \right )\left [ \left ( \sqrt{bc} +1\right )^{2}+\left ( \sqrt{b}-\sqrt{c} \right )^{2} \right ]\\
			&\geq \left ( a+1 \right )\left ( \sqrt{bc} +1\right )^{2}=\left ( a+1 \right )\left ( \frac{1}{\sqrt{a}}+1 \right )^{2}.
		\end{align*}
		Suy ra 
		\begin{align*}
			&15a\geq \left ( a+1 \right )\left ( 1+\sqrt{a} \right )^{2}=\left ( a+1 \right )\left [\left ( a+1  \right )+2\sqrt{a} \right ]\\
			&\Rightarrow 16a\geq \left ( a+1 \right )^{2}+2\left ( a+1 \right )\sqrt{a}+a=\left ( a+1+\sqrt{a} \right )^{2}\\
			&\Rightarrow 4\sqrt{a}\geq a+1+\sqrt{a}\Rightarrow \frac{5}{4}\geq a-3\sqrt{a}+\frac{9}{4}=\left ( \sqrt{a} -\frac{3}{2}\right )^{2}\\
			&\Rightarrow \frac{\sqrt{5}}{2}\geq \sqrt{a}-\frac{3}{2}\Rightarrow q\geq a;\;q=\left ( \frac{3+\sqrt{5}}{2} \right )^{2}=\frac{7+3\sqrt{5}}{2}.
		\end{align*}
		Do vai trò của $a,\;b,\;c$ trong $(1)$ như nhau nên ta có $q\ge b;\;q\ge c.$ Do đó 
		\begin{align*}
			&\left ( q-a \right )\left ( q-b \right )\left ( q-c \right )\geq 0\\
			&\Rightarrow q^3-q^2\left ( a+b+c \right )+q\left ( ab+bc+ca \right )-abc\geq 0\\
			&\Rightarrow q^3-q^2\left ( a+b+c+ab+bc+ca \right )+\left ( q^2+q \right )\left ( ab+bc+ca \right )-1\geq 0\\
			&\Rightarrow q^3-13q^2+\left ( q^2+q \right )P-1\geq 0\\
			&\Rightarrow P\geq \frac{1+13q^2-q^3}{q+q^2}=\frac{13-\sqrt{5}}{2}.
		\end{align*}
		Vậy $P=\dfrac{13-\sqrt{5}}{2} $ khi và chỉ khi trong ba số $a,b,c$ có một hằng số bằng $q=\dfrac{7+3\sqrt{5}}{2} $ và hai số còn lại bằng $\dfrac{1}{\sqrt{q}}=\dfrac{3-\sqrt{5}}{2} $. Vậy khi đó ta được nghiệm là $\left ( x,y,z \right )=\left (\dfrac{3+\sqrt{5}}{2}t;\;t;\;\dfrac{3-\sqrt{5}}{2}t  \right ) $ và hai hoán vị vòng quanh, trong đó $t$ là số thực dương tùy ý.
		\begin{nx}
			Đây là bài đại số khó, có nhiều biến đổi hay. Ngoài cách này, ta có thể dùng cách khác là dùng đạo hàm để khảo sát hàm số.
		\end{nx}}
\end{bt}

\begin{bt}[T10/484-Toán học \& Tuổi trẻ, tháng 10 năm 2017]%[Hoang Trung Dung,Dự án EX_TapChi03]%[0D0G5]
	Với mỗi số nguyên dương $n$, kí hiệu $f(n)$ là tổng bình phương của các ước số nguyên dương của $n$. Hãy tìm tất cả các số nguyên dương $n$ sao cho 
	$$ \sum_{k=1}^{n}f\left ( k \right )\geq \frac{10n^3+15n^2+2n}{24}. $$
	\loigiai{
		Với mỗi $k\in\left [1,n \right] $, trong các số từ $1$ đến $n$ có đúng $\left[ \dfrac{n}{k} \right] $ số là bội của $k.$ Do đó trong tổng $\displaystyle\sum\limits_ {k=1}^{n}f\left ( k \right ) $ mỗi số $k^2$ được tính $\left[ \dfrac{n}{k} \right] $ lần. Thành thử ta tìm tất cả $n$ sao cho 
		$$T_{n}=\sum_{k=1}^{n}f\left ( k \right )=\sum_{k=1}^{n}k^{2}\left[ \dfrac{n}{k} \right]\geq \frac{10n^3+15n^2+2n}{24}. $$
		Giả sử $n=q_{k}\cdot k+r_{k},\;\left ( 0\leq r_{k} <k\right ). $ Khi đó $ \left[ \dfrac{n}{k} \right]=q_{k}$ và $k^2\left[ \dfrac{n}{k} \right]=k^2\left( \dfrac{n-r_{k}}{k} \right)=n\cdot k-k\cdot r_{k}.$ \\ Vậy khi đó ta có 
		$$T_{n}=\sum_{k=1}^{n}n\cdot k-\sum_{k=1}^{n}k\cdot r_{k}=\dfrac{n^2\left ( n+1 \right )}{2}-S_{n},\;\left ( S_n=\sum_{k=1}^{n} k\cdot r_k\right ). $$
		Ta xét hai trường hợp sau:
	\begin{itemize}
			\item  Nếu $n$ lẻ: $n=2m+1,\; \left ( m\geq 1 \right )$ (dễ thấy $m=0$ không thỏa mãn đề bài). Ta có với $ k=m+1,\;m+2,\;\ldots,\;2m$ thì $r_k$ tương ứng là $m,\;m-1,\ldots ,\;1.$ Do đó
		\begin{align*}
		S_n&\geq \sum_{k=m+1}^{2m}k\cdot r_k=\left ( m+1 \right )m+\left ( m+2 \right )\left ( m-1 \right )+\cdots +2m\cdot 1\\
		&=\sum_{k=1}^{m}k\left ( 2m+1-k \right )=\left ( 2m+1 \right )\sum_{k=1}^{m}k-\sum_{k=1}^{m}k^2\\
		&=\frac{\left ( 2m+1 \right )m\left ( m+1 \right )}{2}-\frac{m\left ( m+1 \right )\left ( 2m+1 \right )}{6}\\
		&=\frac{m\left ( m+1 \right )\left ( 2m+1 \right )}{3}=\frac{n^3-n}{12}.
		\end{align*}
		Thành thử ta có 
		$$ T_{n}\leq \dfrac{n^2\left ( n+1 \right )}{2}-\frac{n^3-n}{12}=\frac{5n^3+6n^2+n}{12}<\frac{10n^3+15n^2+2n}{24}. $$
		\item  Nếu $n$ chẵn: $n=2m.$\\
		Với $m=1\Leftrightarrow n=2$ thì thay vào ta thấy thỏa mãn.\\
		Với $m>1$ và $k=m+1,\ldots ,2m-1$ thì $r_k$ tương ứng là $m-1,\;m-2,\ldots ,1.$ Khi đó ta có
		$$ S_n\geq \sum_{k=m+1}^{2m-1}k\cdot r_k=\sum_{k=1}^{m-1}k\left ( 2m-k \right )=2m\sum_{k=1}^{m-1}k-\sum_{k=1}^{m-1}k^2=\frac{n\left ( n-2 \right )\left ( 2n+1 \right )}{24}. $$
	\end{itemize}
		Vậy ta được
		$$T_n\leq \frac{n^2\left ( n+1 \right )}{2}-\frac{n\left ( n-2 \right )\left ( 2n+1 \right )}{24}=\frac{10n^3+15n^2+2n}{24}. $$
		Đẳng thức xảy ra khi $r_k=0$ với mọi $k=1,2,\ldots ,m.$\\
		Tức là $2m$ chia hết cho $1,2,\ldots ,m.$\\
		Với $m=2,\;n=4$ (thỏa mãn).\\
		Với $m=3,\;n=6$(thỏa mãn).\\
		Với $m>3,$ do $ n\vdots \left ( m-1 \right ),\;n\vdots \left ( m-2 \right )$ và $\left ( m-1,m-2 \right )=1 $ nên $n\vdots \left ( m-1 \right )\left ( m-2 \right ). $  Suy ra $$n\geq \left ( m-1 \right )\left ( m-2 \right )\Leftrightarrow 2m\geq \left ( m-1 \right )\left ( m-2 \right )\Leftrightarrow m^2-5m+2\leq 0\Rightarrow m<5\Rightarrow m=4\Rightarrow n=8.$$ Thử lại thấy không thỏa mãn vì $8$ không chia hết cho $3$. Vậy các số cần tìm là $2,4,6.$
	}
\end{bt}

\begin{bt}[T11/484-Toán học \& Tuổi trẻ, tháng 10 năm 2017]%[Nguyễn Quang Tân,Dự án EX_TapChi03]%[1D0G1]
	Cho dãy số $(x_n)$ thỏa mãn các điều kiện $x_1=1$, $x_2=\dfrac{1}{2}$, $x_{n+2}x_n=x_{n+1}^2+4^{-n}$, $n\in\mathbb{N}^*$. Tính $\lim\limits_{x\to +\infty}(3-\sqrt{5})^n x_n$.
	\loigiai{
		\setcounter{equation}{0}
		Từ giả thiết, ta có
		\begin{equation}
		x_{n+2}x_n=x_{n+1}^2+4^{-n},\; n\in \mathbb{N}^*
		\label{t11.484.1}
		\end{equation}
		và 
		\begin{equation*}
		x_{n+1}x_{n-1}=x_n^2+4^{-n+1},\; 1<n\in \mathbb{N}^*
		\end{equation*}
		hay 
		\begin{equation}
		\dfrac{1}{4}x_{n+1} x_{n-1}=\dfrac{1}{4}x_n^2+4^{-n}, \; 1<n\in \mathbb{N}^*.
		\label{t11.484.2}
		\end{equation}
		Trừ theo vế (\ref{t11.484.1}) và (\ref{t11.484.2}), ta thu được
		\[x_{n+1}\left(\dfrac{1}{4}x_{n-1}+x_{n+1}\right)=x_n\left(\dfrac{1}{4}x_n + x_{n+2}\right), 1<n\in \mathbb{N}^*.\]
		Hay 	
		\begin{equation}
		\dfrac{x_{n+1}}{x_n}=\dfrac{\dfrac{1}{4}x_n+x_{n+2}}{\dfrac{1}{4}x_{n-1}+x_{n+1}}, \; 1<n\in \mathbb{N}^*. \label{t11.184.3}
		\end{equation} 
		Xét dãy $(y_n)$ xác định bởi $y_n=\dfrac{1}{4}x_n+x_{n+2}, \; n\in\mathbb{N}^*$ thì $y_1=\dfrac{1}{4}x_1+x_3=\dfrac{3}{4}.$\\
		Từ (\ref{t11.184.3}) ta có 
		\begin{equation*}
		\dfrac{x_{k+1}}{x_k}=\dfrac{y_k}{y_{k-1}},\; 1<k\in\mathbb{N}^*.
		\end{equation*}
		Suy ra 
		\begin{eqnarray}
		\prod\limits_{k=2}^n \dfrac{x_{k+1}}{x_k}=\prod\limits_{k=2}^n \dfrac{y_k}{y_{k-1}}&\Leftrightarrow & \dfrac{x_{n+1}}{x_2}=\dfrac{y_n}{y_1}
		\Leftrightarrow \dfrac{x_{n+1}}{x_2}=\dfrac{\dfrac{1}{4}x_n+x_{n+2}}{y_1}\nonumber \\
		&\Leftrightarrow & 4x_{n+2}-6x_{n+1}+x_n=0,\; n\in \mathbb{N}^*. 
		\label{t11.184.4}
		\end{eqnarray}
		Phương trình sai phân (\ref{t11.184.4}) với điều kiện $x_1=1, x_2=\dfrac{1}{2}$, có nghiệm là
		\[x_n=\dfrac{10-4\sqrt{5}}{5}\left(\dfrac{3+\sqrt{5}}{4}\right)^n+\dfrac{10+4\sqrt{5}}{5}\left(\dfrac{3-\sqrt{5}}{4}\right)^n \]
		hay
		\[ (3-\sqrt{5})^n x_n=\dfrac{10-4\sqrt{5}}{5}+\dfrac{10+4\sqrt{5}}{5}\left(\dfrac{3-\sqrt{5}}{2}\right)^{2n},\; 1<k\in \mathbb{N}^*. \]
		Suy ra
		\[\lim\limits_{n\to +\infty} (3-\sqrt{5})^n x_n=\lim\limits_{n\to +\infty} \left(\dfrac{10-4\sqrt{5}}{5}+\dfrac{10+4\sqrt{5}}{5}\left(\dfrac{3-\sqrt{5}}{2}\right)^{2n}\right)=\dfrac{10-4\sqrt{5}}{5}.\]
		Kết luận $\lim\limits_{n\to +\infty} (3-\sqrt{5})^n x_n=\dfrac{10-4\sqrt{5}}{5}.$
	}
\end{bt}
