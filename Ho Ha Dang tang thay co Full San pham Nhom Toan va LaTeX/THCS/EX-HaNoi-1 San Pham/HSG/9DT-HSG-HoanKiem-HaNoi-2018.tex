\begin{ex} %[Đề chọn đội tuyển HSG quận Hoàn Kiếm năm 2018]%[9D5G2]%[9D5G3]
    \hfill
    \begin{itemize}
    \item[a)] Tìm ba số nguyên tố sao cho tích của chúng gấp $5$ lần tổng của chúng.
    \item[b)] Tìm bộ số $(x;y)$ nguyên thỏa mãn đẳng thức $1+\sqrt{x+y+3}=\sqrt{x}+\sqrt{y}$.
    \end{itemize}
    \loigiai{
    a) Gọi ba số cần tìm là $a,b,c$. Theo đề ta có $abc=5(a+b+c)$.\\
    Dễ thấy $abc\ \vdots\ 5$ nên trong ba số $a,b,c$ có ít nhất một số chia hết cho $5$. \\
    Không mất tính tổng quát, giả sử $a\ \vdots\ 5$. Vì $a$ nguyên tố nên $a=5$, từ đó $$5bc=5(5+b+c)\Leftrightarrow (b-1)(c-1)=6.$$
    Kết hợp với điều kiện $b,c$ nguyên tố ta thu được $b=2, c=7$ hoặc $b=7,c=2$.\\
    Vậy có $6$ bộ số $(a;b;c)$ thỏa mãn yêu cầu là $(5;2;7)$ và các hoán vị.\\
    b) Từ giả thiết suy ra $\sqrt{x+y+3}=\sqrt{x}+\sqrt{y}-1$. Bình phương hai vế và thu gọn ta có $$\sqrt{xy}-\sqrt{x}-\sqrt{y}-1=0\Leftrightarrow (\sqrt{x}-1)(\sqrt{y}-1)=2 \qquad (*)$$
    Không mất tính tổng quát, giả sử $x\geq y$.
    \begin{itemize}
    \item Xét $y\geq 6$ thì $(\sqrt{x}-1)(\sqrt{y}-1)\geq (\sqrt{6}-1)^2>2$ (mâu thuẫn).
    \item Xét $0\leq y<6$, vì $y\in\mathbb{Z}$ nên $y\in\{0;1;2;3;4;5\}$.\\
    Thay vào $(*)$ tìm được $x=9,y=4$.
    \end{itemize}
    Vậy có $2$ bộ số $(x;y)$ thỏa mãn yêu cầu là $(9;4)$ và $(4;9)$.
    }
\end{ex}

\begin{ex}%[Đề chọn đội tuyển HSG quận Hoàn Kiếm năm 2018]%[9D4G7]
	\hfill
	\begin{itemize}
	\item[a)] Giải phương trình $x^2+2x\sqrt{x-\dfrac{1}{x}}=3x+1$.
	\item[b)] Giải hệ phương trình $\heva{& 6y+2=\dfrac{x}{y}-\sqrt{x-2y}\\
	& \sqrt{x+\sqrt{x-2y}}=x+3y-2}$.
	\end{itemize}
	\loigiai{
	a) Điều kiện: $x-\dfrac{1}{x}\geq 0$.\\
	Do $x\neq 0$ nên chia hai vế của phương trình cho $x$ ta thu được $$x-\dfrac{1}{x}+2\sqrt{x-\dfrac{1}{x}}-3=0.$$
	Tìm được $\sqrt{x-\dfrac{1}{x}}=1$ hoặc $\sqrt{x-\dfrac{1}{x}}=-3$ (loại).\\
	Từ $\sqrt{x-\dfrac{1}{x}}=1$ ta tìm được $x=\dfrac{1\pm \sqrt{5}}{2}$ (thỏa mãn).\\
	Vậy nghiệm của phương trình là $x=\dfrac{1\pm \sqrt{5}}{2}$.\\
	b) Điều kiện $\heva{& y\neq 0\\ & x-2y\geq 0\\ & x+\sqrt{x-2y}\geq 0}.$\\
	Quy đồng mẫu số phương trình đầu của hệ ta thu được $$x-2y-y\sqrt{x-2y}-6y^2=0.$$
	Từ đó suy ra được $\sqrt{x-2y}=-2y$ hoặc $\sqrt{x-2y}=3y$.\\
	\textbf{Trường hợp 1.} Xét $\sqrt{x-2y}=-2y$. Thay vào phương trình sau ta có $$\sqrt{x-2y}=x+3y-2 \Rightarrow -2y=x+3y-2 \Leftrightarrow x=2-5y.$$
	Do đó, $\sqrt{x-2y}=-2y \Leftrightarrow \sqrt{2-7y}=-2y\Leftrightarrow \heva{& y\leq 0\\ & 2-7y=4y^2}\Leftrightarrow y=-2\Rightarrow x=12$.\\
	\textbf{Trường hợp 2.} Xét $\sqrt{x-2y}=3y$. Thay vào phương trình sau ta có $$\sqrt{x+3y}=x+3y-2\Leftrightarrow \sqrt{x+3y}=-1\ \text{(loại)}\ \text{hoặc}\ \sqrt{x+3y}=2\Leftrightarrow x=4-3y.$$
	Do đó, $\sqrt{x-2y}=3y\Leftrightarrow \sqrt{4-5y}=3y\Leftrightarrow \heva{& y\geq 0\\ & 4-5y=9y^2} \Leftrightarrow y=\dfrac{4}{9}\Rightarrow x=\dfrac{8}{3}$.\\
	Thử lại ta có các nghiệm của hệ là $(12;-2)$ và $\left(\dfrac{8}{3};\dfrac{4}{9}\right)$.
	}
\end{ex}

\begin{ex}%[Đề chọn đội tuyển HSG quận Hoàn Kiếm năm 2018]%[9D5G1]
	Cho các số thực không âm $x,y,z$ thỏa mãn $x^2+y^2+z^2=3$.
	\begin{itemize}
		\item[a)] Chứng minh $xy^2+yz^2+zx^2\leq 2+xyz$.
		\item[b)] Tìm giá trị lớn nhất và giá trị nhỏ nhất của biểu thức $P=\dfrac{x}{2+y}+\dfrac{y}{2+z}+\dfrac{z}{2+x}$.
	\end{itemize}
	\loigiai{
	a) Không mất tính tổng quát, giả sử $x\leq y\leq z$. Khi đó, 
	\begin{alignat*}{2}
		& x(y-z)(y-x)\leq 0\\
		 \Leftrightarrow \ & xy^2+zx^2\leq x^2y+xyz\\
		 \Leftrightarrow \ & xy^2+yz^2+zx^2\leq y(x^2+z^2)+xyz=y(3-y^2)+xyz.
	\end{alignat*}
	Xét hiệu $2-y(3-y^2)=y^3-3y+2=(y-1)^2(y+2)\geq 0$.\\
	Suy ra $y(3-y^2)\leq 2\Rightarrow xy^2+yz^2+zx^2\leq 2+xyz$.\\
	Dấu $``=''$ xảy ra khi và chỉ khi $(x;y;z)=(1;1;1)$ hoặc $(0;1;\sqrt{2})$ và các hoán vị của chúng.\\
	b) Nhận xét: Biến đổi $P\leq 1\Leftrightarrow xy^2+yz^2+zx^2\leq 2+xyz$.\\
	Từ đó tìm được $P_{\max}=1\Leftrightarrow (x;y;z)=(1;1;1)$ hoặc $(0;1;\sqrt{2})$ và các hoán vị của chúng.\\
	Ta có $P=\dfrac{x^2}{2x+xy}+\dfrac{y^2}{2y+yz}+\dfrac{z^2}{2z+zx}\geq \dfrac{(x+y+z)^2}{2(x+y+z)+xy+yz+zx}=Q$ \\
	(bất đẳng thức Cauchuy-Schwarz).\\
	Đặt $t=x+y+z$ thì $xy+yz+zx=\dfrac{t^2-3}{2}$ với $\sqrt{3}\leq t\leq 3$. \\
	Khi đó, $Q=\dfrac{2t^2}{t^2+4t-3}$ với $\sqrt{3}\leq t\leq 3$.\\
	Ta có $Q\geq\dfrac{\sqrt{3}}{2}\Leftrightarrow \dfrac{2t^2}{t^2+4t-3}\geq\dfrac{\sqrt{3}}{2} \Leftrightarrow (t-\sqrt{3})\left[(4-\sqrt{3})t-3\right]\geq 0$ (luôn đúng với $t\geq\sqrt{3}$).\\
	Từ đó tìm được $P_{\min}=\dfrac{\sqrt{3}}{2}\Leftrightarrow (x;y;z)=(\sqrt{3};0;0)$ và các hoán vị của nó.
	}
\end{ex}

\begin{ex}%[Đề chọn đội tuyển HSG quận Hoàn Kiếm năm 2018]%[9H3K4]%[9H3K6]%[9H3K7]
	Cho tam giác nhọn $ABC$ ($AB<AC$) nội tiếp đường tròn $(O)$, các đường cao $BD, CE$ cắt nhau tại $H$. Đường tròn tâm $I$ đường kính $AH$ cắt đường tròn $(O)$ tại điểm thứ hai $P$.
	\begin{itemize}
	\item[a)] Chứng minh hai tam giác $PBC$ và $PED$ đồng dạng.
	\item[b)] Gọi $M$ là trung điểm của $BC$. Chứng minh ba điểm $P,H,M$ thẳng hàng.
	\item[c)] Các tiếp tuyến tại $A$ và $P$ của đường tròn $(I)$ cắt nhau tại $N$. Chứng minh rằng các đường thẳng $MN, DE, AH$ đồng quy.
	\end{itemize}
	\loigiai{
	\begin{center}
	\begin{tikzpicture}
		\tkzDefPoints{0/0/B, 7/0/C, 1.5/5/A}
		\tkzDefMidPoint(B,C) \tkzGetPoint{M}
		\tkzDefCircle[circum,/tikz/overlay](A,B,C) 
		\tkzGetPoint{O} \tkzGetLength{r}
		\tkzDrawCircle[R](O,\r pt)
		\tkzDefLine[perpendicular=through B](A,C) 
		\tkzInterLL(B,tkzPointResult)(A,C) \tkzGetPoint{D}
		\tkzDefLine[perpendicular=through C](A,B) 
		\tkzInterLL(C,tkzPointResult)(A,B) \tkzGetPoint{E}
		\tkzInterLL(B,D)(C,E) \tkzGetPoint{H}
		\tkzDefCircle[circum,/tikz/overlay](A,D,H) 
		\tkzGetPoint{I} \tkzGetLength{i}
		\tkzDrawCircle[R](I,\i pt)
		\tkzInterCC[/tikz/overlay,R](O,\r pt)(I,\i pt) \tkzGetFirstPoint{P}
		\tkzDefLine[perpendicular=through P](P,I) \tkzGetPoint{p} 
		\tkzDefLine[perpendicular=through A](A,I) \tkzGetPoint{a}
		\tkzInterLL(P,p)(A,a) \tkzGetPoint{N}
		\pgfresetboundingbox
		\coordinate (K) at ($(A)!2!(O)$);
		\tkzInterLL(A,P)(B,C) \tkzGetPoint{F}
		\tkzDrawSegments(A,B F,C C,A B,D C,E A,H I,P I,N A,N P,N P,B E,D A,K I,M A,F D,F F,I P,K B,K N,M) 
		\tkzDrawPoints[fill=white, size=10pt](O,A,B,C,D,E,H,I,P,N,M,K,P,F)
		\tkzLabelPoints[left](P)
		\tkzLabelPoints[right](D,O)
		\tkzLabelPoints[below](H,M,E)
		\tkzLabelPoints[below left](F,B)
		\tkzLabelPoints[below right](C,K)
		\tkzLabelPoints[above](A)
		\tkzLabelPoints[above left](N)
	\end{tikzpicture}
	\end{center}
	a) Ta có $\widehat{PCB}=\widehat{PAB}=\widehat{PDE}$ và $\widehat{PBC}=180^\circ - \widehat{PAC}=\widehat{PED}$.\\
	Do đó $\Delta PBC \backsim \Delta PED$ (g-g).\\
	b) Kẻ đường kính $AK$, ta có $\widehat{APK}=90^\circ$, chú ý $\widehat{APH}=90^\circ$ từ đó ba điểm $P,H,K$ thẳng hàng.\\
	Mặt khác, $BH\parallel KC$, $CH\parallel KB$ nên $BHCK$ là hình bình hành.\\
	Từ đó $H,M,K$ thẳng hàng. Suy ra $P,H,M$ thẳng hàng.\\
	c) Gọi $F$ là giao điểm của $AP$ và $BC$, ta có $\widehat{AEP}=\widehat{AHP}=\widehat{BFP}$, từ đó $BEPF$ là tứ giác nội tiếp, suy ra $\widehat{FEP}=\widehat{FBP}=\widehat{PAC}$.\\
	Suy ra $\widehat{FEP}+\widehat{DEP}=\widehat{PAC}+\widehat{DEP}=180^\circ$ từ đó ba điểm $D,E,F$ thẳng hàng, chú ý rằng $IM$ là trung trực của $DE$, do đó $DE\perp IM$.\\
	Ta có $\widehat{NIP}=\widehat{AHP}=\widehat{MFP}$, từ đó $\Delta PNI\backsim PMF$ (g-g) $(*)$\\
	Từ $(*)$ dễ dàng suy ra $\Delta IPF \backsim NPM$ (c-g-c), suy ra $\widehat{PIF}=\widehat{PNM}\Rightarrow FI\perp MN$.\\
	Chú ý $AH\perp FM$, từ đó $MN,DE,AH$ là ba đường cao của $\Delta IMF$ nên chúng đồng quy.
	}
\end{ex}

\begin{ex}%[Đề chọn đội tuyển HSG quận Hoàn Kiếm năm 2018]%[9D5G2]
	Một số số nguyên dương được viết trên một bảng đen. Tổng của hai số bất kì trong chúng là lũy thừa của $2$. Hỏi trên bảng có thể có nhiều nhất bao nhiêu số phân biệt?
	\loigiai{
	Ta sẽ chứng minh rằng số các số nguyên dương phân biệt không thể lớn hơn $2$.\\
	Thật vậy, giả sử có ba số $a<b<c$ trên bảng (trường hợp $4$ số trở lên vẫn trích ra được bộ ba số như thế).\\
	Dễ thấy theo đề bài, $a+b; b+c; c+a$ là ba lũy thừa phân biệt của $2$.\\
	Mặt khác, $a+b<a+c<b+c$ suy ra $b+c\geq 2(a+c)$ hay $b\geq 2a+c$ (vô lí).\\
	Từ đó suy ra có nhiều nhất hai số nguyên dương thỏa mãn yêu cầu đề bài.\\
	Ta có thể lấy một trường hợp cụ thể là $1$ và $3$.
	}
\end{ex}
