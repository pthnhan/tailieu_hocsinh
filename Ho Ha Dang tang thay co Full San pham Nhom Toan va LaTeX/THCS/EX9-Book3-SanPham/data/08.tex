\section{BẤT ĐẲNG THỨC VÀ CỰC TRỊ DẠNG PHÂN THỨC}

\begin{center}
 \large \textbf{TỔNG QUAN VỀ CHUYÊN ĐỀ}
\end{center}
Chuyên đề này bao gồm hai nội dung:
\begin{itemize}
	\item Bất đẳng thức dạng phân thức, trong đó có bất đẳng thức về cộng mẫu số.
	\item Tìm giá trị nhỏ nhất, giá trị lớn nhất của các biểu thức có dạng phân thức, trong đó có bài toán tìm cực trị mà biến nằm trong một khoảng nhất định, khi đó cần dự đoán giá trị của biến để xảy ra cực trị.
\end{itemize}
\textbf{Tính nhẩm và đời sống}
\begin{center}
	\textbf{CHIA NHANH}
\end{center}
Bạn Vinh cần làm phép chia $1:1{,}03$ và làm tròn đến hàng phần trăm. Vinh đang loay hoay tính toán vì không có máy tính bỏ túi trong tay. Hà nói với Vinh:\\
- Bạn chỉ cần lấy $1$ trừ đi $0{,}03$ là được $!$ ($0{,}03$ là \textit{phần hơn} của số chia so với $1$).\\
	Vinh rất ngạc nhiên khi sau đó kiểm tra lại bằng máy tính:\\
	$1:1{,}03=0{,}9708\cdots\simeq 0{,}97(=1-0{,}03)$.\\ 
	Vinh thử một vài trường hợp khác cũng thấy đúng:\\
$1:1{,}04=0{,}9615\cdots\simeq 0{,}96$.\\	
$1:1{,}07=0{,}9345\cdots\simeq 0{,}93$.
\begin{enumerate}
	\item Bạn hãy giải thích điều đó.
	\item Có phải kết quả tính theo cách trên luôn nhỏ hơn kết quả đúng hay không?
\end{enumerate}
\textbf{Giải}
\begin{enumerate}
	\item Giả sử cần tính $\dfrac{1}{1+a}$ trong đó $a$ rất nhỏ so với $1$.\\ Khi đó $1\approx 1-a^2$ nên $\dfrac{1}{1+a}\approx \dfrac{1-a^2}{1+a}=1-a$.
	\item  Do $1>1-a^2$ nên $\dfrac{1}{1+a}>\dfrac{1-a^2}{1+a}=1-a$.\\
	Vậy kết quả tính được theo cách trên luôn nhỏ hơn kết quả đúng.
\end{enumerate}

\subsection{Bất đẳng thức dạng phân thức}

Ngoài các bất đẳng thức về liên hệ giữa trung bình cộng và trung bình nhân (bất đẳng thức Cô-si), liên hệ giữa tổng các bình phương và bình phương của tổng (bất đẳng thức Bu-nhi-a-cốp-ski) được nêu ở Chuyên đề $7$, cần nhớ các bất đẳng thức về liên hệ giữa tổng các số và tổng các nghịch đảo của chúng:\\
$\dfrac{1}{x}+\dfrac{1}{y}\ge \dfrac{4}{x+y}$ với $x,y>0$.\hfill (1)\\
$\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\ge \dfrac{9}{x+y+z}$ với $x$,$y$,$z$$>0$.              \hfill (2)\\
Bạn đọc cũng nên biết các bất đẳng thức tổng quát của các bất đẳng thức trên và chứng minh chúng để khi cần, có thể sử dụng chúng như những bổ đề:\\
$\dfrac{a^2}{x}+\dfrac{b^2}{y}\ge \dfrac{(a+b)^2}{x+y}$ với $x,y>0$. \hfill (3)\\
$\dfrac{a^2}{x}+\dfrac{b^2}{y}+\dfrac{x^2}{z}\ge \dfrac{(a+b+c)^2}{x+y+z}$ với $x,y,z>0$. \hfill (4)\\
Bất đẳng thức $(1)$ là trường hợp đặc biệt của bất đẳng thức $(3)$ khi $a=b=1$.\\
Bất đẳng thức $(2)$ là trường hợp đặc biệt của bất đẳng thức $(4)$ khi $a=b=c=1$. Ta gọi các bất đẳng thức trên là các bất đẳng thức về cộng mẫu số.
\setcounter{vd}{98}
\begin{vd}
    Chứng minh các bất đẳng thức $(3)$ và $(4)$ nói trên.
    \loigiai{
     \begin{enumerate}
     	\item Chứng minh bất đẳng thức $(3)$ như sau:\\
     	$\dfrac{a^2}{x}+\dfrac{b^2}{y}\geq \dfrac{(a+b)^2}{x+y}\Leftrightarrow \dfrac{a^2y+b^2x}{xy}\geq \dfrac{(a+b)^2}{x+y}$\\
     	$\Leftrightarrow\left(a^2y+b^2x\right)\left(x+y\right)\geq xy\left(a^2+2ab+b^2\right)$\\
     	$\Leftrightarrow a^2y^2+b^2x^2\geq2abxy \Leftrightarrow \left(ay-bx\right)^2\geq 0$, đúng.\\
     	Xảy ra đẳng thức khi và chỉ khi $\dfrac{a}{x}=\dfrac{b}{y}$.
     	\item Chứng minh bất đẳng thức $(4)$ như sau:\\
     	\textbf{Cách 1} Áp dụng bất đẳng thức $(3)$ hai lần, ta có\\
     	$$\left(\dfrac{a^2}{x}+\dfrac{b^2}{y}\right)+\dfrac{c^2}{z}\geq \dfrac{(a+b)^2}{x+y}+ \dfrac{c^2}{z} \geq \dfrac{(a+b+c)^2}{x+y+z}.$$
     	\textbf{Cách 2} Áp dụng bất đẳng thức Bu-nhi -a-cốp-xki, ta có \\
     	$$\left[\left(\dfrac{a}{\sqrt{x}}\right)^2+\left(\dfrac{b}{\sqrt{y}}\right)^2+\left(\dfrac{c}{\sqrt{z}}\right)^2\right]\cdot\left[\left(\sqrt{x}\right)^2+\left(\sqrt{y}\right)^2+\left(\sqrt{z}\right)^2\right]\geq (a+b+c)^2$$
     	$\Rightarrow \left(\dfrac{a^2}{x}+\dfrac{b^2}{y}+\dfrac{c^2}{z}\right)(x+y+z)\geq (a+b+c)^2$\\
     	$\Rightarrow \dfrac{a^2}{x}+\dfrac{b^2}{y}+\dfrac{c^2}{z}\geq \dfrac{(a+b+c)^2}{x+y+z}$.\\
     	Xảy ra đẳng thức khi và chỉ khi $\dfrac{a}{x}=\dfrac{b}{y}=\dfrac{c}{z}$.
     \end{enumerate}  
    }
\end{vd}

\loigiai{
Theo bất đẳng thức Cô-si, ta có\\
$(x+y)^2\geq 4xy>0$ \hfill (1)\\

Nhân $(1)$ với $(2)$ theo từng vế được\\
$(x+y)^2\left(\dfrac{1}{x^2}+\dfrac{1}{y^2}\right)\geq 8 \Rightarrow \dfrac{1}{x^2}+\dfrac{1}{y^2}\geq \dfrac{8}{(x+y)^2}$
}
\end{vd}
\begin{vd}
	Cho $A=\dfrac{a^2}{b^2}+\dfrac{b^2}{a^2}$ và $B=\dfrac{a}{b}+\dfrac{b}{a}$. Chứng minh rằng:
\begin{multicols}{2}
	\item [a)] $A\geq2$;
	\item [b)] $A\geq B$.
\end{multicols}
\loigiai{
\begin{enumerate}[a)]
	\item Đặt $x=\dfrac{a}{b}$ và $y=\dfrac{b}{a}$ thì $xy=1$.\\
	$A=x^2+y^2\geq 2 \left| xy\right| =2$ nên $A\geq2$. \hfill (1)
	\item Ta có $x^2+1\geq 2x$ và $y^2+1\geq 2y$ nên\\
	$x^2+y^2+2\geq 2(x+y)\Rightarrow A+2 \geq 2B$.  \hfill (2)\\
	Từ $(1)$ suy ra $A+A\geq A+2$. Kết hợp với $(2)$ có $2A\geq 2B$ nên $A\geq B$.
\end{enumerate}
}
\end{vd}
\begin{vd}
Cho $A=\dfrac{a^2}{b^2}+\dfrac{b^2}{c^2}+\dfrac{c^2}{a^2}$ và $B=\dfrac{a}{b}+\dfrac{b}{c}+\dfrac{c}{a}$. Chứng minh rằng:
\begin{enumerate}[a)]
	\item $A\geq3$;
	\item $A\geq B$.
\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Đặt $x=\dfrac{a}{b}; y=\dfrac{b}{c}; z=\dfrac{c}{a}$ thì $xyz=1$  \\
			Xét $A+1=(x^2+y^2)+(z^2+1) \geq 2 |xy| +2|z| \geq 2\sqrt{|4xyz|}=4$  \\
			Suy ra $A \geq 3$.  \hfill (1) 
			\item Ta có $x^2+1 \geq 2x; y^2+1 \geq 2y; z^2+1 \geq 2z $ nên \\ $\dfrac{a^2}{b^2}+\dfrac{b^2}{c^2}+\dfrac{c^2}{a^2} +3 \geq 2(x+y+z) \Rightarrow A+3 \geq 2B $  \hfill (2)\\
			Từ (1) suy ra $A+A \geq A+3 $ nên $ 2A \geq A+3$ \\  
			Kết hợp với (2) ta có $2A \geq 2B$  nên $A \geq B$.
		\end{enumerate}
	}
\end{vd}

\begin{vd} 
	Cho $x=\dfrac{a+b}{a-b}, y=\dfrac{b+c}{b-c}, z=\dfrac{c+a}{c-a} $\\ 
	Chứng minh rằng:
	\begin{enumerate}
		\item $xy+yz+zx=-1$. 
		\item $x^2+y^2+z^2 \geq 2 $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $x+1 = \dfrac{a+b}{a-b}+1=\dfrac{2a}{a-b}$  . Từ đó\\ \indent 
			$(x+1)(y+1)(z+1) = \dfrac{2a}{a-b}.\dfrac{2b}{b-c}.\dfrac{2c}{c-a}$ \hfill  (1) \\ 
			Ta có $x-1 = \dfrac{a+b}{a-b}-1=\dfrac{2b}{a-b}$  . Từ đó\\ \indent 
			$(x-1)(y-1)(z-1) = \dfrac{2b}{a-b}.\dfrac{2c}{b-c}.\dfrac{2a}{c-a}$ \hfill  (2) \\
			Từ (1) và (2) suy ra  $(x+1)(y+1)(z+1) =(x-1)(y-1)(z-1)$\\
			Nhân và rút gọn ta được $xy+yz+zx=-1$.\\
			\item 
			Ta có $x^2+y^2+z^2 = (x+y+z)^2 - 2(xy+yz+zx) = (x+y+z)^2 +2$ (theo câu a).\\
			Vậy $x^2+y^2+z^2 \geq 2 $.
		\end{enumerate}
	}
\end{vd}

\begin{vd} 
	Chứng minh bất đẳng thức
	\begin{center}
		$ \dfrac{a}{a+b}+\dfrac{b}{b+c}+\dfrac{c}{c+a} \geq \dfrac{3}{2}$    với $a \geq b \geq c >0$.
	\end{center}
	\loigiai{
		Gọi vế trái là $A$ . Ta có
		$A-\dfrac{3}{2}=
		\left( \dfrac{a}{a+b}-\dfrac{1}{2} \right)
		+\left(\dfrac{b}{b+c}-\dfrac{1}{2} \right)
		+\left(\dfrac{c}{c+a} -\dfrac{1}{2} \right) 
		\\ \indent
		=\dfrac{a-b}{2(a+b)} +\dfrac{b-c}{2(b+c)} + \dfrac{c-a}{2(c+a)}    \\ \indent
		=\dfrac{a-b}{2(a+b)} +\dfrac{(b-a)+(a-c)}{2(b+c)} + \dfrac{c-a}{2(c+a)} \\ \indent
		=\dfrac{a-b}{2} . \left( \dfrac{1}{a+b} -\dfrac{1}{b+c}   \right) 
		+\dfrac{a-c}{2} . \left( \dfrac{1}{b+c} - \dfrac{1}{c+a}   \right)  \\ \indent
		=\dfrac{a-b}{2} .  \dfrac{c-a}{(a+b)(b+c)} 
		+\dfrac{a-c}{2} .  \dfrac{a-b}{(b+c)(c+a)}   \\ \indent
		=\dfrac{(a-b)(a-c)}{2(b+c)} .  \left( -\dfrac{1}{a+b} +\dfrac{1}{c+a}   \right)  \\ \indent
		=\dfrac{(a-b)(a-c)(b-c)}{2(a+b) (b+c) (c+a)} \geq 0$. 
		Vậy $A \geq \dfrac{3}{2} $.\\
		Đẳng thức xảy ra khi và chỉ khi có ít nhất hai số bằng nhau.
	}
\end{vd}

\begin{vd}
	Chứng minh bất đẳng thức
	\begin{center}
		$ \dfrac{a^3+b^3}{2ab}+\dfrac{b^3+c^3}{2bc} +\dfrac{c^3+a^3}{2ca}\geq a+b+c$   \qquad  với $a ,b,c $ dương.
	\end{center}
	\loigiai{
		Do $a \geq b \geq c >0$ nên $a^3+b^3 = (a+b)(a^2-ab+b^2) \geq (a+b)ab$ . \\
		$\Rightarrow  \dfrac{a^3+b^3}{2ab} \geq \dfrac{a+b}{2} $ . Từ đó ta có\\
		$$\dfrac{a^3+b^3}{2ab}+\dfrac{b^3+c^3}{2bc}+\dfrac{c^3+a^3}{2ca}\geq \dfrac{a+b}{2}+\dfrac{b+c}{2}+\dfrac{c+a}{2}.$$
		Dấu bằng xảy ra khi và chỉ khi $a=b=c$.
	}
\end{vd}

\begin{vd}
	Cho các số thực dương $a,\ b,\ c$ thỏa mãn $a \leq b,\ a\leq c$ và $abc=1$. Chứng minh bất đẳng thức $$a+b^2+c^2 \geq \dfrac{1}{a}+\dfrac{1}{b^2}+\dfrac{1}{c^2}.$$
	\loigiai{
		Ta có $a\leq b$ và $a\leq c$ nên $a^3\leq abc=1$, do đó $a \leq 1$.
		Bất đẳng thức cần chứng minh tương đương với 
		$$M=\left(a-\dfrac{1}{a} \right)+b^2+c^2 -\left(\dfrac{1}{b^2}+\dfrac{1}{c^2}  \right) \geq 0. \eqno (1)$$
		Ta có $$a-\dfrac{1}{a}=\dfrac{a^2-1}{a}=(a^2-1)bc\  (\text{vì}\ \dfrac{1}{a}=bc )$$ và $$b^2+c^2-\dfrac{b^2+c^2}{b^2c^2}=(b^2+c^2)\left(1-\dfrac{1}{b^2c^2} \right)=(b^2+c^2)(1-a^2)\ (\text{vì}\ \dfrac{1}{b^2c^2}=a^2),$$
		Nên $$M=(a^2-1)bc+(b^2+c^2)(1-a^2)=(1-a^2)(b^2-bc+c^2)=(1-a)(1+a)(b^2-bc+c^2).$$
		Do $a,\ b,\ c>0$ và $a \leq 1$ nên $1+a>0,\ 1-a \geq 0,\ b^2-bc+c^2 \geq 0$, do đó $M \geq 0$.
		Bất đẳng thức đã cho được chứng minh. Đẳng thức xảy ra khi và chỉ khi $a=b=c=1$.
	}
\end{vd}
\begin{vd}
	Chứng minh bất đẳng thức $\dfrac{1}{x^2+1}+\dfrac{1}{y^2+1} \geq \dfrac{2}{xy+1}$ với $xy \geq 1.$
	\loigiai{
		\textit{Cách 1:} Bất đẳng thức cần chứng minh tương đương với
		$$\begin{aligned} &\dfrac{x^2+1+y^2+1}{(x^2+1)(y^2+1)} \geq \dfrac{2}{xy+1} \\& \Leftrightarrow  (x^2+y^2+2)(xy+1) \geq 2(x^2y^2+x^2+y^2+1) \\ & \Leftrightarrow xy(x^2+y^2)+x^2+y^2+2xy+2-2x^2y^2-2x^2-2y^2-2 \geq 0\\& \Leftrightarrow  xy(x^2+y^2-2xy)-(x^2+y^2-2xy) \geq 0 \\& \Leftrightarrow (x-y)^2(xy-1) \geq 0. \end{aligned}$$
		Bất đẳng thức cuối đúng vì $xy \geq 1$.\\
		Bất đẳng thức đã cho được chứng minh. Xảy ra đẳng thức khi và chỉ khi $x=y$ hoặc $xy=1$.\\
		\textit{Cách 2:}  Vế trái 
		$$\begin{aligned} A&=\dfrac{1}{1+x^2}+\dfrac{1}{1+y^2}=\dfrac{x^2+y^2+2}{x^2y^2+x^2+y^2+1} \\ & = \dfrac{(x^2+y^2+1+x^2y^2)+(1-x^2y^2)}{x^2y^2+x^2+y^2+1} =1+ \dfrac{1-x^2y^2}{x^2y^2+x^2+y^2+1}. \ \ \ (1)\end{aligned}$$ 
		Ta có $x^2y^2+x^2+y^2+1 \geq x^2y^2 +2xy+1= (xy+1)^2. $\\
		Suy ra $\dfrac{1}{x^2y^2+x^2+y^2+1} \leq \dfrac{1}{(xy+1)^2}.$
		Nhân hai vế với $1-x^2y^2 \leq 0$ ta được 
		$$\dfrac{1-x^2y^2}{x^2y^2+x^2+y^2+1} \geq \dfrac{1-x^2y^2}{(1+xy)^2}=\dfrac{1-xy}{1+xy}. \eqno (2)$$
		Từ (1) và (2) ta có $$A \geq 1+\dfrac{1-xy}{1+xy}=\dfrac{2}{1+xy}.$$
	}
\end{vd}
\subsection{Cực trị dạng phân thức}
\begin{vd}
	Tìm giá trị nhỏ nhất của $A=\dfrac{x^2+1}{x}$ với 
	\begin{multicols}{2}
		\begin{itemize}
			\item[a)] $0<x \leq \dfrac{1}{2}$
			\item[b)] $x \geq 2$
		\end{itemize}
	\end{multicols}
	\loigiai{
		\begin{enumerate}[a)]
			\item Với dự đoán cực trị xảy ra khi $x=\dfrac{1}{2},$ tức là $x^2=\dfrac{1}{4}$, ta biến đổi như sau
			$$A=\dfrac{x^2+\dfrac{1}{4}+\dfrac{3}{4}}{x}=\dfrac{x^2+\dfrac{1}{4}}{x}+\dfrac{3}{4x}.$$
			Do $x>0$ nên $\dfrac{x^2+\dfrac{1}{4}}{x} \geq \dfrac{2\cdot \dfrac{1}{2}x}{x}=1$, 
			$0<x \leq \dfrac{1}{2}$ nên $\dfrac{3}{4x} \geq \dfrac{3}{2}$. Do đó $A \geq 1+\dfrac{3}{2}=\dfrac{5}{2}$.\\ Vậy $\min A=\dfrac{5}{2}$, đạt được khi $x=\dfrac{1}{2}$.
			\item Với dự đoán cực trị xảy ra khi $x=2 \Leftrightarrow x^2=4$ nên ta viết
			$$A=\dfrac{x^2+4-3}{x}=\dfrac{x^2+4}{x}-\dfrac{3}{x}.$$
			Do $x \geq 2$ nên $ \dfrac{x^2+4}{x} \geq \dfrac{2\cdot 2x}{x}=4$ và $\dfrac{3}{x} \leq \dfrac{3}{2}$.  Suy ra $A \geq 4-\dfrac{3}{2}=\dfrac{5}{2}.$\\
			Vậy $\min A=\dfrac{5}{2}$, đạt được khi $x=2.$
			
		\end{enumerate}
	}
\end{vd}
\begin{vd}
	Tìm giá trị nhỏ nhất của $A=\dfrac{x^2+1}{x}$ với 
	\begin{multicols}{2}
		\begin{itemize}
			\item[a)] $0<x \leq \dfrac{1}{4}$
			\item[b)] $x \geq 3$
		\end{itemize}
	\end{multicols}
	\loigiai{
		\begin{enumerate}[a)]
			\item Với dự đoán cực trị xảy ra khi $x=\dfrac{1}{4},$ tức là $x^2=\dfrac{1}{16}$, ta biến đổi như sau
			$$A=\dfrac{x^2+\dfrac{1}{16}+\dfrac{15}{16}}{x}=\dfrac{x^2+\dfrac{1}{16}}{x}+\dfrac{15}{16x}.$$
			Do $x>0$ nên $\dfrac{x^2+\dfrac{1}{16}}{x} \geq \dfrac{2\cdot \dfrac{1}{4}x}{x}=\dfrac{1}{2}$, 
			$0<x \leq \dfrac{1}{4}$ nên $\dfrac{15}{16x} \geq \dfrac{15}{4}$. Do đó $A \geq \dfrac{1}{2}+\dfrac{15}{4}=\dfrac{17}{4}$.\\ Vậy $\min A=\dfrac{17}{4}$, đạt được khi $x=\dfrac{1}{4}$.
			\item Với dự đoán cực trị xảy ra khi $x=3$, hay $x^2=9$ nên ta viết
			$$A=\dfrac{x^2+9-8}{x}=\dfrac{x^2+9}{x}-\dfrac{8}{x}.$$
			Do $x \geq 3$ nên $ \dfrac{x^2+9}{x} \geq \dfrac{2\cdot 3x}{x}=6$ và $\dfrac{8}{x} \leq \dfrac{8}{3}$.  Suy ra $A \geq 6-\dfrac{8}{3}=\dfrac{10}{3}.$\\
			Vậy $\min A=\dfrac{10}{3}$, đạt được khi $x=3.$
		\end{enumerate}
	}
\end{vd}

		Tổng quát: Với $0<x \leq \dfrac{1}{n}$ thì $\min A=n\dfrac{1}{n}$ (hỗn số).
	\end{itemize}
\end{note}
\begin{vd}
	Tìm giá trị nhỏ nhất của biểu thức $A=\dfrac{x^2+y^2}{2xy}$ với $x,\ y>0$ và $x \geq 2y.$
	\loigiai{
