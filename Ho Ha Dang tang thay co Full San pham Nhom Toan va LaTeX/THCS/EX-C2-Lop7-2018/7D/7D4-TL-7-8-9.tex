\section{ĐA THỨC MỘT BIẾN}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Đa thức một biến}
	Đa thức một biến là tổng của những đơn thức của cùng một biến.\\
	Để chỉ rõ:
	\begin{itemize}
		\item $A$ là đa thức của biến $x$ người ta viết $A(x)$;
		\item $B$ là đa thức của biến $y$ người ta viết $B(y)$.
	\end{itemize}
	Khi đó, giá trị của đa thức $A(x)$ tại $x = 8$ được kí hiệu là $A(8)$, $\ldots$
	\begin{note}
		Mỗi số được coi là một đa thức một biến.
	\end{note}
\subsubsection{Sắp xếp một đa thức}
	Để thuận lợi cho việc tính toán đối với các đa thức một biến, người ta sắp xếp các hạng tử của chúng theo lũy thừa tăng hoặc giảm của biến.
\subsubsection{Hệ số}
	Xét đa thức: $P(x) = 2x^3 + cx - 3$ đó là một đa thức đã thu gọn, trong đó:
	\begin{itemize}
		\item $2$ là hệ số của lũy thừa bậc $3$;
		\item $c$ là hệ số của lũy thừa bậc $1$;
		\item $-3$ là hệ số của lũy thừa bậc $0$ (còn gọi là \textit{hệ số tự do}).
	\end{itemize}
	Ngoài ra, vì $P(x)$ có bậc bằng $3$ nên hệ số của lũy thừa bậc $3$ còn gọi là \textit{hệ số cao nhất}.
	\begin{note}
		Có thể viết đa thức $P(x)$ đầy đủ từ lũy thừa bậc cao nhất đến lũy thừa bậc $0$ như sau: $P(x) = 2x^3 + 0x^2 + cx - 3$.\\
		Do đó, ta nói hệ số của lũy thừa bậc $2$ của $P(x)$ bằng $0$.
	\end{note}
\subsection{Các dạng toán}
\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-L7-2018]%[7D4B7]
	Cho đa thức $A(x) = x^2 - 8x + 19$.
	\begin{enumEX}{2}
		\item Tìm bậc của đa thức $A(x)$;
		\item Tính $A(4)$, $A(-1)$.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item Đa thức $A(x)$ có bậc $2$.
			\item Ta có $A(4) = 4^2 - 8 \cdot 4 + 19 = 3$; $A(-1) = (-1)^2 - 8 \cdot (-1) + 19 = 28$.
		\end{enumerate}
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-L7-2018]%[7D4B7]
	Viết một đa thức một biến và đưa ra dạng tổng quát của nó trong các trường hợp sau:
	\begin{enumerate}
		\item Có hai hạng tử mà bậc cao nhất là $8$ và hệ số tự do là $-5$;
		\item Có ba hạng tử mà hệ số cao nhất là $7$ và hệ số tự do là $-2$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Đa thức một biến có hai hạng tử mà bậc cao nhất là $8$ và hệ số tự do là $-5$. Chẳng hạn: $x^8 - 5$, $3x^8 - 5$, $\ldots$\\
					Vậy dạng tổng quát của đa thức trên là: $m \cdot x^8 - 5$, với $m \neq 0$.
			\item Đa thức một biến có ba hạng tử mà hệ số cao nhất là $7$ và hệ số tự do là $-2$. Chẳng hạn: $7x^2 + 5x - 2$, $7x^5 + 8x^2 - 2$, $\ldots$\\
					Vậy dạng tổng quát của đa thức trên là: $7x^m + ax^n - 2$, với $m > n \geq 1$, $a \neq 0$ và $a < 7$.
		\end{enumerate}
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-L7-2018]%[7D4B7]
	Hãy sắp xếp các hạng tử của các đa thức sau: $P(x) = 7 + x^3 + 9x - 27x^2$.
	\loigiai{
		Khi sắp xếp các hạng tử của chúng theo:
		\begin{itemize}
			\item Lũy thừa tăng, ta được: $P(x) = 7 + 9x - 27x^2 + x^3$;
			\item Lũy thừa giảm, ta được: $P(x) = x^3 - 27x^2 + 9x + 7$.
		\end{itemize}
	}
\end{vd}

\begin{note}
	Để sắp xếp các hạng tử của một đa thức, trước hết phải thu gọn đa thức đó.
\end{note}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-L7-2018]%[7D4B7]
	Hãy sắp xếp các hạng tử của đa thức sau: $Q(x) = 2 - 3x^2 + x^4 - 6x^3 + 9x + 3x^3 - x - 7$.
	\loigiai{
		Thu gọn đa thức bằng cách:
		$$Q(x) = (2 - 7) - 3x^2 + x^4 - (6x^3 - 3x^3) + (9x - x) = -5 - 3x^2 + x^4 - 3x^3 + 8x.$$
		Khi sắp xếp các hạng tử của chúng theo:
		\begin{itemize}
			\item Lũy thừa tăng, ta được: $Q(x) = -5 + 8x - 3x^2 - 3x^3 + x^4$;
			\item Lũy thừa giảm, ta được: $Q(x) = x^4 - 3x^3 - 3x^2 + 8x - 5$.
		\end{itemize}
	}
\end{vd}

\begin{nx}
	Mọi đa thức bậc hai của biến $x$, sau khi sắp xếp các hạng tử của chúng theo lũy thừa giảm của biến đều có dạng: $ax^2 + bx + c$, với $a \neq 0$.\\
	Như vậy, ta có một biểu thức đại số trong đó $x$ là biến còn $a$, $b$, $c$ đại diện cho các số xác định cho trước. Để phân biệt với biến, người ta gọi những chữ như vậy là \textit{hằng số} (gọi tắt là \textit{hằng}).
\end{nx}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-L7-2018]%[7D4B7]
	Cho đa thức $Q(x) = 2x^5 - 3x^2 - 3 + x^4 - 2 + 6x^3 + 8x - 6x^3 + 5 - 2x^5$.
	\begin{enumerate}
		\item Sắp xếp các hạng tử của $Q(x)$ theo lũy thừa giảm của biến.
		\item Viết đa thức $Q(x)$ đầy đủ từ lũy thừa bậc cao nhất đến lũy thừa bậc $0$.
		\item Chỉ ra các hệ số của $Q(x)$.
		\item Tính $Q(-2)$, $Q(1)$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Thu gọn $Q(x)$, ta được:
					$$Q(x) = (2x^5 - 2x^5) - 3x^2 - (3 + 2 - 5) + x^4 + (6x^3 - 6x^3) + 8x = -3x^2 + x^4 + 8x.$$
					Khi đó, $Q(x)$ được sắp xếp theo lũy thừa giảm của biến là:
					$$Q(x) = x^4 - 3x^2 + 8x.$$
			\item Dạng đầy đủ của $Q(x)$ là: $Q(x) = x^4 - 0x^3 - 3x^2 + 8x + 0$.
			\item Như vậy, $Q(x)$ có:
					\begin{itemize}
						\item $1$ là hệ số của lũy thừa bậc $4$;
						\item $0$ là hệ số của lũy thừa bậc $3$;
						\item $-3$ là hệ số của lũy thừa bậc $2$;
						\item $8$ là hệ số của lũy thừa bậc $1$;
						\item $0$ là hệ số của lũy thừa bậc $0$.
					\end{itemize}
			\item Ta có: $Q(-2) = (-2)^4 - 3 \cdot (-2)^2 + 8 \cdot (-2) = 16 - 12 - 16 = -12$.\\
					\phantom{Ta có:} $Q(1) = 1^4 - 3 \cdot 1^2 + 8 \cdot 1 = 1 - 3 + 8 = 6$.
		\end{enumerate}
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-L7-2018]%[7D4B7]
	Cho đa thức $Q(x) = x^3 + 2x^4 - 6x^2 + 9 - 5x^3 + x^3 + 11$.
	\begin{enumerate}
		\item Sắp xếp các hạng tử của $Q(x)$ theo lũy thừa giảm của biến.
		\item Viết đa thức $Q(x)$ đầy đủ từ lũy thừa bậc cao nhất đến lũy thừa bậc $0$.
		\item Chỉ ra các hệ số của $Q(x)$.
		\item Tính $Q(-3)$, $Q(2)$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Thu gọn $Q(x)$, ta được:
					$$Q(x) = (1 - 5 + 1)x^3 + 2x^4 - 6x^2 + 11 + 9 = -3x^3 + 2x^4 - 6x^2 + 20.$$
					Khi đó, $Q(x)$ được sắp xếp theo lũy thừa giảm của biến là:
					$$Q(x) = 2x^4 - 3x^3 - 6x^2 + 20.$$
			\item Dạng đầy đủ của $Q(x)$ là: $Q(x) = 2x^4 - 3x^3 - 6x^2 + 0x + 20$.
			\item Như vậy, $Q(x)$ có:
					\begin{itemize}
						\item $2$ là hệ số của lũy thừa bậc $4$;
						\item $-3$ là hệ số của lũy thừa bậc $3$;
						\item $-6$ là hệ số của lũy thừa bậc $2$;
						\item $0$ là hệ số của lũy thừa bậc $1$;
						\item $20$ là hệ số của lũy thừa bậc $0$.
					\end{itemize}
			\item Ta có: $Q(-3) = 2 \cdot (-3)^4 - 3 \cdot (-3)^3 - 6 \cdot (-3)^2 + 20 = 209$.\\
					\phantom{Ta có:} $Q(2) = 2 \cdot 2^4 - 3 \cdot 2^4 - 6 \cdot 2^2 + 20 = 4$.
		\end{enumerate}
	}
\end{vd}

\begin{vd}%[Đinh Thanh Hoàng, Dự án EX-C2-L7-2018]%[7D4K7]
	Tính giá trị của đa thức sau:
	\begin{enumerate}
		\item $x + x^2 + x^3 + x^4 + \cdots + x^{50}$ tại $x = -1$.
		\item $ax^3 + bx^2 + cx + d$ tại $x = 1$ ($a$, $b$, $c$, $d$ là \textit{hằng số}).
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Thay $x = -1$ vào $x + x^2 + x^3 + x^4 + \cdots + x^{50}$, ta được:
					$$(-1) + (-1)^2 + (-1)^3 + (-1)^4 + \cdots + (-1)^{50} = -1 + 1 - 1 + \cdots + 1 = 0.$$
			\item Thay $x = 1$ vào $ax^3 + bx^2 + cx + d$, ta được:
					$$a \cdot 1^3 + b \cdot 1^2 + c \cdot 1 + d = a + b + c + d.$$
		\end{enumerate}
	}
\end{vd}

\subsection{Bài tập tự luyện}
\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-L7-2018]%[7D4B7]
	Viết một đa thức một biến và đưa ra dạng tổng quát của nó trong các trường hợp sau:
	\begin{enumerate}
		\item Có ba hạng tử mà bậc cao nhất là $5$ và hệ số tự do là $9$.
		\item Có bốn hạng tử mà hệ số cao nhất là $11$ và hệ số tự do là $6$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Đa thức một biến có ba hạng tử mà bậc cao nhất là $5$ và hệ số tự do là $9$. Chẳng hạn: $x^5 + x^2 + 9$, $2x^5 + 3x^4 + 9$, $\ldots$\\
					Vậy dạng tổng quát của đa thức trên là: $ax^5 + bx^n + 9$ với $a$, $b$, $n \neq 0$ và $1 \leq n < 5$.
			\item Đa thức một biến có bốn hạng tử mà hệ số cao nhất là $11$ và hệ số tự do là $6$. Chẳng hạn: $11x^8 + 3x^2 - 9x + 6$, $11x^3 - x^2 + 10x + 6$, $\ldots$\\
					Vậy dạng tổng quát của đa thức trên là: $11x^m + ax^n + bx^p + 6$ với $m > n > p \geq 1$; $a$, $b \neq 0$ và $a$, $b < 11$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-L7-2018]%[7D4B7]
	Cho đa thức $P(x) = 4x^2 + x^4 - x^2 + 50 + 2x^3 + 6x - 2x^3 + 2x + 4$.
	\begin{enumerate}
		\item Sắp xếp các hạng tử của $P(x)$ theo lũy thừa giảm của biến.
		\item Viết đa thức $P(x)$ đầy đủ từ lũy thừa bậc cao nhất đến lũy thừa bậc $0$.
		\item Chỉ ra các hệ số của $P(x)$.
		\item Tính $P(-2)$, $P(1)$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Thu gọn $P(x)$, ta được:
					$$P(x) = (4 - 1)x^2 + x^4 + (50 + 4) + (2 - 2)x^3 + (6 + 2)x = 3x^2 + x^4 + 54 + 8x.$$
					Khi đó, $P(x)$ được sắp xếp theo lũy thừa giảm của biến là:
					$$P(x) = x^4 + 3x^2 + 8x + 54.$$
			\item Dạng đầy đủ của $P(x)$ là: $P(x) = x^4 + 0x^3 + 3x^2 + 8x + 54$.
			\item Như vậy, $P(x)$ có:
					\begin{itemize}
						\item $1$ là hệ số của lũy thừa bậc $4$;
						\item $0$ là hệ số của lũy thừa bậc $3$;
						\item $3$ là hệ số của lũy thừa bậc $2$;
						\item $8$ là hệ số của lũy thừa bậc $1$;
						\item $54$ là hệ số của lũy thừa bậc $0$.
					\end{itemize}
			\item Ta có: $P(-2) = (-2)^4 + 3 \cdot (-2)^2 + 8 \cdot (-2) + 54 = 66$.\\
					\phantom{Ta có:} $P(1) = 1^4 + 3 \cdot 1^2 + 8 \cdot 1 + 54 = 66$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-L7-2018]%[7D4B7]
	Cho đa thức $Q(x) = 8 + 3x - x^2 + 9x^3 - 3x - x^2 - x^3 - 6$.
	\begin{enumerate}
		\item Sắp xếp các hạng tử của $Q(x)$ theo lũy thừa giảm của biến.
		\item Viết đa thức $Q(x)$ đầy đủ từ lũy thừa bậc cao nhất đến lũy thừa bậc $0$.
		\item Chỉ ra các hệ số của $Q(x)$.
		\item Tính $Q(-4)$, $Q(3)$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Thu gọn $Q(x)$, ta được:
					$$Q(x) = (8 - 6) + (3 - 3)x - (1 + 1)x^2 + (9 - 1)x^3 = 2 - 2x^2 + 8x^3.$$
					Khi đó, $Q(x)$ được sắp xếp theo lũy thừa giảm của biến là:
					$$Q(x) = 8x^3 - 2x^2 + 2.$$
			\item Dạng đầy đủ của $Q(x)$ là: $Q(x) = 8x^3 - 2x^2 + 0x + 2$.
			\item Như vậy, $Q(x)$ có:
					\begin{itemize}
						\item $8$ là hệ số của lũy thừa bậc $3$;
						\item $-2$ là hệ số của lũy thừa bậc $2$;
						\item $0$ là hệ số của lũy thừa bậc $1$;
						\item $2$ là hệ số của lũy thừa bậc $0$.
					\end{itemize}
			\item Ta có: $Q(-4) = 8 \cdot (-4)^3 - 2 \cdot (-4)^2 + 2 = -542$.\\
					\phantom{Ta có:} $Q(3) = 8 \cdot 3^3 - 2 \cdot 2^2 + 2 = 200$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-L7-2018]%[7D4K7]
	Cho đa thức $R = -4xy + x^2 + 2y^2$.
	\begin{enumerate}
		\item Sắp xếp các hạng tử của $R(x)$ theo lũy thừa giảm của biến. Chỉ ra các hệ số của $R(x)$. Tính $R(-3)$.
		\item Sắp xếp các hạng tử của $R(y)$ theo lũy thừa giảm của biến. Chỉ ra các hệ số của $R(y)$. Tính $R(2)$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $R(x)$ được sắp xếp theo lũy thừa giảm của biến là: $R(x) = x^2 - 4xy + 2y^2$.\\
					Như vậy, $R(x)$ có:
					\begin{itemize}
						\item $1$ là hệ số của lũy thừa bậc $2$;
						\item $-4y$ là hệ số của lũy thừa bậc $1$;
						\item $2y^2$ là hệ số của lũy thừa bậc $0$.
					\end{itemize}
					Ta có ngay $R(-3) = (-3)^2 - 4 \cdot (-3) \cdot y + 2y^2 = 9 + 12y + 2y^2$.
			\item $R(y)$ được sắp xếp theo lũy thừa giảm của biến là: $R(x) = 2y^2 - 4xy + x^2$.\\
					Như vậy, $R(y)$ có:
					\begin{itemize}
						\item $2$ là hệ số của lũy thừa bậc $2$;
						\item $-4x$ là hệ số của lũy thừa bậc $1$;
						\item $x^2$ là hệ số của lũy thừa bậc $0$.
					\end{itemize}
					Ta có ngay $R(2) = 2 \cdot 2^2 - 4x \cdot 2 + x^2 = 8 - 8x + x^2$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đinh Thanh Hoàng, Dự án EX-C2-L7-2018]%[7D4K7]
	Cho đa thức: $P(x) = ax^2 + 5x^4 - 8x + 9 - x^2 + ax$ ($a$ là hằng số).
	\begin{enumerate}

		\item Chỉ ra các hệ số của $P(x)$.
		\item Tính $P(-2)$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Thu gọn $P(x)$, ta được:
					$$P(x) = (a - 1)x^2 + 5x^4 + (a - 8)x + 9.$$
					Khi đó, $P(x)$ được sắp xếp theo lũy thừa giảm của biến là:
					$$P(x) = 5x^4 + (a - 1)x^2 + (a - 8)x + 9.$$
			\item Như vậy, $P(x)$ có:
					\begin{itemize}
						\item $5$ là hệ số của lũy thừa bậc $4$;
						\item $a - 1$ là hệ số của lũy thừa bậc $3$;
						\item $a - 8$ là hệ số của lũy thừa bậc $1$;
						\item $9$ là hệ số của lũy thừa bậc $0$.
					\end{itemize}
			\item Ta có $P(-2) = 5 \cdot (-2)^4 + (a - 1) \cdot (-2)^2 + (a - 8) \cdot (-2) + 9 = 101 + 2a$.
		\end{enumerate}

\section{Nghiệm của đa thức một biến}
\subsection{Tóm tắt lí thuyết}
\begin{dn}
	Nếu tại $ x = a $, đa thức $ P(x) $ có giá trị bằng $ 0 $ thì ta nói $ a $ (hoặc $ x = a $) là một nghiệm của đa thức đó.
\end{dn}
\begin{note}
	Một đa thức (khác đa thức $ 0 $) có thể có $ 1 $ nghiệm, $ 2 $ nghiệm, $ \dots $ hoặc không có nghiệm (gọi là vô nghiệm).
\end{note}
\begin{dl}
	Một đa thức (khác đa thức $ 0 $) có số nghiệm không vượt quá bậc của nó.
\end{dl}

	Chứng tỏ rằng nếu $ a + b +c = 0 $ thì $ x = 1 $ là nghiệm của đa thức $ f(x) = ax^2 + bx + c $. Ngoài ra nếu $ a \ne 0 $ thì $ x = \dfrac{c}{a} $ cũng là nghiệm của đa thức $ f(x) $.
	\loigiai{
		Vì $ x =1 $ là nghiệm của $ f(x) $ suy ra $ f(1) = 0 \Leftrightarrow  a\cdot 1^2 + b\cdot 1 + c = 0 $, điều này nghĩa là $ a + b + c = 0 $.\\	
		Với $ a \ne 0 $. Xét $ f\left(\dfrac{c}{a} \right) = a\cdot\left(\dfrac{c}{a}\right)^2  + b\cdot \dfrac{c}{a} + c = \dfrac{c}{a}\left(c+b+a\right) = 0\ \text{(vì $ a +b + c = 0 $)}   $.
	}
\end{vd}
\begin{note}
	Áp dụng kết quả trên, ta có thể tìm được nghiệm của các đa thức $ f(x) = ax^2 + bx + c $ có $ a + b +c = 0 $. Ví dụ sau đây sẽ minh họa cho điều này.
\end{note}

	Tìm các nghiệm của đa thức $ f(x) = 3x^2 - 7x + 4 $.
	\loigiai{
		Ta có $ 3 - 7  +4 =0 $, do đó đa thức $ f(x) $ có các nghiệm là $ x = 1, x = \dfrac{4}{3} $.
	}
\end{vd}
\begin{vd}%[Lê Thanh Hải, dự án EX-C2-L7-2018]%[7D4K9]

	\loigiai{
		Để $ x = 1 $ là nghiệm của đa thức $ f(x) $ thì cần có
		\[ f(1) = 0 \Leftrightarrow a+b+c+d = 0.\]
		Vậy với $ a +b +c +d = 0 $ thì $ f(x) $ nhận $ x = 1 $ làm nghiệm.
	}
\end{vd}
\begin{vd}%[Lê Thanh Hải, dự án EX-C2-L7-2018]%[7D4K9]
	Chứng tỏ rằng đa thức $ x^2 + 2x  +3 $ không có nghiệm.
	\loigiai{
		Ta có $ x^2 + 2x + 3 = x^2 + x + x + 1 +2 = x(x+1) + (x+1) +2 = (x+1)^2 +2 > 0 $ (với mọi $ x $).\\
		 Do đó đa thức $ x^2 + 2x  +3 $ không có nghiệm.
	}
	\end{vd}

\begin{bt}%[Lê Thanh Hải, dự án EX-C2-L7-2018]%[7D4Y9]
	Cho đa thức: $Q(x)=x^2-8x+7.$\\ Kiểm nghiệm rằng đa thức $Q(x)$ có hai nghiệm $x=1,x=7$.
	\loigiai{
		Ta có:\\
		$Q(1)=1^2-8\cdot 1+7=1-8+7=0\Rightarrow x=1$ là một nghiệm của đa thức $Q(x)$.\\
		$Q(7)=7^2-8\cdot 7+7=49-56+7=0\Rightarrow x=7$ là một nghiệm của đa thức $Q(x)$.
	}
\end{bt}
\begin{bt}%[Lê Thanh Hải, dự án EX-C2-L7-2018]%[7D4Y9]

	\begin{enumEX}{2}
		\item $x+8$;
		\item $3x-7$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Nghiệm của đa thức $x+8$ thỏa mãn $x=8=0 \Leftrightarrow x=-8.$\\ Vậy $x=-8$ là nghiệm của đa thức $x+8$.
			\item Nghiệm của đa thức $3x-7$ thỏa mãn $3x-7=0\Leftrightarrow 3x=7 \Leftrightarrow x=\dfrac{7}{3}$.\\
			Vậy $x=\dfrac{7}{3}$ là nghiệm của đa thức $3x-7$.
		\end{enumEX}
	}
\end{bt}
\begin{bt}%[Lê Thanh Hải, dự án EX-C2-L7-2018]%[7D4B9]
	Tìm nghiệm của các đa thức:
	\begin{enumEX}{2}
		\item $(x-2)(2x+8)$;
		\item $(3x-9)(2x+5)$.
	\end{enumEX}
	\loigiai{\begin{enumEX}{1}
			\item Nghiệm của đa thức $(x-2)(2x+8)$ thỏa mãn:\\

			Vậy $x=2$ hoặc $x=-4$ là nghiệm của đa thức $(x-2)(2x+8)$.
			\item Nghiệm của đa thức $(3x-9)(2x+5)$ thỏa mãn:\\
			$(3x-9)(2x+5)=0 \Leftrightarrow 3x-9=0$ hoặc $2x+5=0 \Leftrightarrow x=3$ hoặc $x=-\dfrac{5}{2}$.\\
			Vậy $x=3$ hoặc $x=-\dfrac{5}{2}$ là nghiệm của đa thức $(3x-9)(2x+5)$.
		\end{enumEX}

\end{bt}

\begin{bt}%[Lê Thanh Hải, dự án EX-C2-L7-2018]%[7D4B9]
	Chứng tỏ rằng nếu $a-b+c=0$ thì $x=-1$ là nghiệm của đa thức:
	$$f(x)=ax^2+bx+c.$$
	Ngoài ra nếu $a\ne 0$ thì $x=-\dfrac{c}{a}$ cũng là nghiệm của đa thức $f(x)$.
	\loigiai{
		Ta có: $f(-1)=a\cdot (-1)^2+b\cdot (-1)+c=a-b+c=0$\\
		$\Leftrightarrow x=-1$ là nghiệm của đa thức $f(x)$.\\
		Với $a\ne 0$, ta có: $f(-\dfrac{c}{a})=a\cdot \left(-\dfrac{c}{a}\right)^2+b\cdot \left(-\dfrac{c}{a}\right)+c=\dfrac{c^2}{a}-\dfrac{bc}{a}+c=\dfrac{c}{a}(c-b+a)=0$\\
		$\Leftrightarrow x=-\dfrac{c}{a}$ là nghiệm của đa thức $f(x)$.
		
	}
\end{bt}

	}
