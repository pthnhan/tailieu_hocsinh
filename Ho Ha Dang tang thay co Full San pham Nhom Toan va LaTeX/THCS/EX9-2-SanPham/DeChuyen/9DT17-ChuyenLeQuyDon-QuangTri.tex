Rút gọn biểu thức $A=\dfrac{\sqrt{3+\sqrt{5-\sqrt{13+\sqrt{48}}}}}{\sqrt{6}+\sqrt{2}}$.
\loigiai{\begin{align*}
	A&=\dfrac{\sqrt{3+\sqrt{5-\sqrt{\left(2\sqrt{3}+1\right)^2}}}}{\sqrt{6}+\sqrt{2}}=\dfrac{\sqrt{3+\sqrt{4-2\sqrt{3}}}}{\sqrt{6}+\sqrt{2}}\\&=\dfrac{\sqrt{3+\sqrt{\left(\sqrt{3}-1\right)^2}}}{\sqrt{6}+\sqrt{2}}=\dfrac{1}{2}.

\begin{ex}%[Đề thi vào 10, Chuyên Lê Quý Đôn, Quảng Trị, 2017]%[9D5G1]
Cho biểu thức $P=\dfrac{x^2}{y}+\dfrac{y^2}{x}+\dfrac{1}{x+y}$. Tìm giá trị nhỏ nhất của biểu thức $P$ trong các trường hợp sau.
\begin{enumerate}
	\item $x$, $y$ là các số thực dương.
	\item $x$, $y$ là các số nguyên dương.
\end{enumerate}
\loigiai{\begin{enumerate}
		\item Ta có: $\dfrac{x^2}{y}+y\geq 2x$ suy ra $\dfrac{x^2}{y}\geq 2x-y$. Từ đó ta có $\dfrac{x^2}{y}+\dfrac{y^2}{x}\geq x+y$ hay $P\geq x+y+\dfrac{1}{x+y}\geq 2$. Dấu \lq\lq=\rq\rq  xảy ra khi $x=y=\dfrac{1}{2}$.
		\item Từ giả thiết ta có $x+y\geq 2$. Vậy $P\geq x+y+\dfrac{1}{x+y}=\dfrac{1}{4}(x+y)+\dfrac{1}{x+y}+\dfrac{3}{4}(x+y)\geq \dfrac{5}{2}$. Dấu bằng xảy ra khi $x=y=1.$
\end{enumerate}}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Lê Quý Đôn, Quảng Trị, 2017]%[9D5G5]%[9D4G7]
    \hfill
    \begin{enumerate}
        \item Giải phương trình $2\sqrt{3-x}+\sqrt{2+x}=5$.
        \item Giải hệ phương trình $\heva{&x^3+y^3+1=3xy\\&x^2+2xy+2y^2=5}$.
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item Đặt $t=\sqrt{2+x}$, ta được phương trình theo $t$: $t^2-2t+1=0\Rightarrow t=1$. Đối chiếu điều kiện thấy $t=1$ thỏa mãn, do đó $t=1\Leftrightarrow x=-1$.
        \item Ta có: $x^3+y^3+1=3xy\Leftrightarrow \left(x+y+1\right)\left[(x-y)^2+(x-1)^2+(y-1)^2\right]=0\Leftrightarrow\hoac{&x+y+1=0\\&x=y=1}$.
        
        + Rõ ràng $x=y=1$ là một nghiệm của hệ.
        
        + Với $x+y+1=0$ ta có $y=-x-1$ thay vào phương trình còn lại ta được $x^2+2x-3=0.$ Giải ra ta được $x=1$ và $x=-3$. 
        
        Vậy hệ có 3 nghiệm $(1;1)$, $(1;-2)$, $(-3,2)$.
    \end{enumerate}
    }
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên Lê Quý Đôn, Quảng Trị, 2017]%[9D5G2]%[9D4G3]
	\hfill
	\begin{enumerate}
		\item Tìm chữ số tận cùng của $a=2017^{6^4}$.
		\item Tìm tất cả các nghiệm nguyên của phương trình $7\left(x+y\right)=3\left(x^2+xy+y^2\right)$.
	\end{enumerate}
	\loigiai{\begin{enumerate}
			\item $2017^{6^4}\equiv 7^{6^4}\equiv 3^{6^4}\equiv 81^{4\cdot 81}\equiv 1 \pmod {10}$.
			\item $7(x+y)=3(x^2+xy+y^2)\Rightarrow 3\mid x+y$. Từ giả thiết suy ra $0\leq x+y\leq 3$ do đó $x+y=0$ hoặc $x+y=3$.
			
			+ Với $x+y=0$ ta được $x=y=0$.
			
			+ Với $x+y=3$ suy ra $y=3-x$ thay vào phương trình ta được $x^2-3x+2=0$. Giải ra ta được $x=1$, $x=2$.
			
			Vậy ta được 3 nghiệm nguyên $(0;0)$, $(1;2)$, $(2;1)$.
	\end{enumerate}}
\end{ex}
