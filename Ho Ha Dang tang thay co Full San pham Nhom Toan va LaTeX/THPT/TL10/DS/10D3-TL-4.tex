\section{HỆ PHƯƠNG TRÌNH HAI ẨN}
\subsection{Hệ phương trình gồm các phương trình bậc nhất và bậc hai}
Để giải các hệ phương trình dạng này, ta chủ đạo sử dụng phương pháp thế và phương pháp cộng đại số thông thường, đôi khi kết hợp thêm giải pháp đặt ẩn phụ để làm gọn bài toán.
\begin{vd}%[Phan Hoàng Anh]%[0D3B4]
Giải các hệ phương trình sau:
\begin{multicols}{2}
\begin{enumerate}
	\item $\heva{&x+2y=1\\&x^2-y^2=2x-1-y}$
	\item $\heva{&3x^2+2x-y^2=1\\&y^2+4x=8}$
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
	\item $\heva{&x+2y=1&(1)\\&x^2-y^2=2x-1-y&(2)}$\\
	$(1)\Leftrightarrow x=1-2y$.\\
	$(2)\Leftrightarrow (1-2y)^2-y^2=2(1-2y)-1-y\Leftrightarrow 3y^2+y=0\Rightarrow\hoac{&y=0\Rightarrow x=1\\&y=-\dfrac{1}{3}\Rightarrow x=\dfrac{5}{3}}$.\\
	Vậy hệ có hai nghiệm là $\heva{&x=1\\&y=0}$, $\heva{&x=\dfrac{5}{3}\\&y=-\dfrac{1}{3}}$.
	\item $\heva{&3x^2+2x-y^2=1&(1)\\&y^2+4x=8&(2)}$\\
	Lấy $(1)+(2)$ ta được $3x^2+6x-9=0\Leftrightarrow\hoac{&x=-3\\&x=1}$.\\
	Với $x=-3$, $(2)\Leftrightarrow y^2=8-4x=20\Leftrightarrow\hoac{&y=-2\sqrt{5}\\&y=2\sqrt{5}}$.\\
	Với $x=1$, $(2)\Leftrightarrow y^2=8-4x=4\Leftrightarrow\hoac{&y=-2\\&y=2}$.\\
	Vậy hệ có bốn nghiệm là $\heva{&x=-3\\&y=\pm2\sqrt{5}}$, $\heva{&x=1\\&y=\pm2}$.
\end{enumerate}
}
\end{vd}
\begin{vd}%[Phan Hoàng Anh]%[0D3K4]
Giải các hệ phương trình sau:
\begin{multicols}{2}
	\begin{enumerate}
		\item $\heva{&2x^3+4x^2+x^2y=9-2xy\\&x^2+y=6-4x}$
		\item $\heva{&x-\dfrac{1}{x}=y-\dfrac{1}{y}\\&2x^2-xy-1=0}$
	\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
	\item $\heva{&2x^3+4x^2+x^2y=9-2xy&(1)\\&x^2+y=6-4x&(2)}$\\
	$(2)\Leftrightarrow y=-x^2-4x+6$\\
	$(1)\Leftrightarrow 2x^3+4x^2+x^2(-x^2-4x+6)=9-2x(-x^2-4x+6)$\\
	$\Leftrightarrow x^4+4x^3-2x^2-12x+9=0\Leftrightarrow(x^2+2x-3)^2=0\Leftrightarrow\hoac{&x=1\Rightarrow y=1\\&x=-3\Rightarrow y=9}$.\\
	Vậy hệ phương trình có hai nghiệm là $\heva{&x=1\\&y=1}$, $\heva{&x=-3\\&y=9}$.
	\item $\heva{&x-\dfrac{1}{x}=y-\dfrac{1}{y}&(1)\\&2x^2-xy-1=0&(2)}$\\
	Điều kiện: $x\ne0$ và $y\ne0$.\\
	$(1)\Leftrightarrow x-y-\dfrac{1}{x}+\dfrac{1}{y}=0\Leftrightarrow(x-y)\left(1+\dfrac{1}{xy}\right)=0\Leftrightarrow\hoac{&x=y\\&xy=-1}$.\\
	Trường hợp 1: $x=y$, $(2)\Leftrightarrow2x^2-x^2-1=0\Leftrightarrow x^2=1\Leftrightarrow\hoac{&x=1\Rightarrow y=1\\&x=-1\Rightarrow y=-1}$.\\
	Trường hợp 2: $xy=-1$, $(2)\Leftrightarrow2x^2+1-1=0\Leftrightarrow x=0$ (loại).\\
	Vậy hệ phương trình có hai nghiệm là $\heva{&x=1\\&y=1}$, $\heva{&x=-1\\&y=-1}$.
\end{enumerate}
}
\end{vd}
\begin{vd}%[Phan Hoàng Anh]%[0D3K4]
Xác định các giá trị của $m$ để hệ phương trình $\heva{&x-2y=m\\&x^2+2xy-y^2=2m}$ có hai nghiệm phân biệt.
	\loigiai{Ta có $(1)\Leftrightarrow x=2y+m$.\\
		Thế vào $(2)$ ta được $(2)\Leftrightarrow(2y+m)^2+2y(2y+m)-y^2=2m\Leftrightarrow7y^2+6my+m^2-2m=0(*)$.\\
		Để hệ có hai nghiệm phân biệt thì $(*)$ phải có hai nghiệm phân biệt\\ $\Leftrightarrow\Delta'>0\Leftrightarrow2m^2+14m>0\Leftrightarrow\hoac{&m<-7\\&m>0}$.
	}
\end{vd}

\subsection{Hệ phương trình đối xứng loại $1$}
\begin{dn} Hệ phương trình đối xứng loại $1$ của hai ẩn $x, y$ là hệ mà khi ta thay thế $x$ bởi $y$ và $y$ bởi $x$ thì ta được hệ mới không thay đổi (thứ tự các phương trình trong hệ giữ nguyên).
\end{dn}
Phương pháp giải:\\
Bước 1: Đặt điều kiện nếu cần;\\
Bước 2: Đặt $x+y=S; xy=P$ ($S^2\geq 4P$). Khi đó ta đưa về hệ mới của 2 ẩn $S,P$;\\
Bước 3: Giải hệ ta tìm được $S,P$;\\
Bước 4: $x,y$ là  nghiệm của phương trình $X^2-SX+P=0$.\\
\begin{note}
Chú ý:
\begin{center}
$x^2+y^2=S^2-2P$;
$x^3+y^3=S^3-3SP$.
\end{center}
\end{note}

\begin{vd}%[Nguyễn Thị Thu]%[0D3B4]
Giải hệ phương trình sau: $\heva{&x+y+xy=5,\\ &x^2+y^2-3xy=-1.}$
\loigiai{
Hệ đã cho có thể viết lại là:
\begin{center}
$\heva{&S+P=5\\ &S^2-5P=-1}\Leftrightarrow
\heva{&P=5-S\\ &S^2+5S-24=0}\Leftrightarrow
\hoac{&S=-8\\&S=3}.$
\end{center}
TH1:\\
$\heva{&S=3\\ &P=2}\Leftrightarrow
\heva{&x+y=3\\ &xy=2}\Leftrightarrow
\hoac{&\heva{&x=1\\ &y=2}\\&\heva{&x=2\\ &y=1}}.$\\
TH2:\\
$\heva{&S=-8\\ &P=13}\Leftrightarrow
\heva{&x+y=-8\\ &xy=13}\Leftrightarrow
\hoac{&\heva{&x=-4+\sqrt3\\ &y=-4-\sqrt{3}}\\& \heva{&x=-4-\sqrt3\\ &y=-4+\sqrt3}}.$\\
Vậy hệ phương trình đã cho có 4 nghiệm là: $(1; 2), (2; 1), (-4+\sqrt{3}; -4-\sqrt{3} ), (-4-\sqrt{3}; -4+\sqrt{3})$
}
\end{vd}

\begin{note} Chú ý:
1. Đối với hệ đối xứng của hai ẩn $x,y$ thì nếu $(x_0; y_0)$ là nghiệm thì $(y_0; x_0)$ cũng là nghiệm của hệ.\\
2. Có một số hệ phương trình không phải là hệ đối xứng loại 1, tuy nhiên ta có thể chọn biến phù hợp để đổi biến đưa về hệ đối xứng loại 1.
\end{note}


\begin{vd}%[Nguyễn Thị Thu]%[0D3K4]
Giải hệ phương trình sau: $\heva{&x^2+x=\dfrac{14}{3}y,\\ &x^3=5y^2.}$
\loigiai{Xét hệ $\heva{&x^2+x=\dfrac{14}{3}y &(1),\\ &x^3=5y^2 &(2).}$\\
$\bullet$ Nếu $y=0\Rightarrow x=0\Rightarrow (0; 0)$ là nghiệm của hệ.\\
$\bullet$ Nếu $y\neq 0$. Chia 2 vế của phương trình  $(1)$ cho $y$ ta có: $\dfrac{x^2}{y}+ \dfrac{x}{y}=\dfrac{14}{3}$.\\
Chia 2 vế của phương trình $(2)$ cho $y^2$ ta được $\dfrac{x^2}{y}\dfrac{x}{y}=5$.\\
Vậy ta có hệ phương trình:
$\heva{&\dfrac{x^2}{y}+ \dfrac{x}{y}=\dfrac{14}{3},\\ &\dfrac{x^2}{y}\dfrac{x}{y}=5}.$\\
Đặt $\dfrac{x^2}{y}=u$; $\dfrac{x}{y}=v$ ta được hệ mới là:
\begin{center}
$\heva{&u+v=\dfrac{14}{3}\\&uv=5}
\Leftrightarrow\hoac{&\heva{&u=3\\&v=\dfrac{5}{3}}\\&\heva{&v=\dfrac{5}{3}\\&v=3}}.$
\end{center}
TH1: $\heva{&u=3\\&v=\dfrac{5}{3}}\Leftrightarrow
\heva{&\frac{x^2}{y}=3\\&\dfrac{x}{y}=\dfrac{5}{3}}\Leftrightarrow
\heva{&x=\dfrac{9}{5}\\&y=\dfrac{27}{25}}.$\\
TH2: Giải tương tự ta có\\
$\heva{&x=\frac{27}{25}\\ &y=\dfrac{9}{5}}.$\\
Vậy hệ phương trình đã cho có 2 nghiệm là:
$\left(\dfrac{9}{5};\dfrac{27}{25}\right), \left(\dfrac{27}{25}; \dfrac{9}{5}\right)$.
}
\end{vd}


\begin{vd}%[Nguyễn Thị Thu]%[0D3G4]
Tìm tất cả các giá trị thực của tham số m để  hệ phương trình sau có nghiêm:
 $\heva{&\sqrt{x-2}+\sqrt{y+1}=4,\\ &x+y=2m+1.}$
\loigiai{
Đặt $u=\sqrt{x-2}; v=\sqrt{y+1}\quad (u,v \geq0)$.
Khi đó hệ phương trình được viết lại là:
$\heva{&u+v=4\\ &u^2+v^2=2m}\Leftrightarrow
\heva{&u+v=4\\ &uv=8-m}$\\
$\Leftrightarrow u; v$ là 2 nghiệm của phương trình: $x^2-4x+8-m=0$.
Để hệ phương trình đã cho có nghiệm thì phương trình trên phải có hai nghiệm không âm\\
$\Leftrightarrow \heva{&\Delta\geq0 \\&S\geq0\\&P\geq0}
\Leftrightarrow \heva{&m-4\geq0\\&8-m\geq0}\Leftrightarrow4\leq m\leq8$.\\
Vậy tất cả các giá trị m cần tìm là: $4\leq m\leq8$.
}
\end{vd}

\begin{vd}%[Nguyễn Thị Thu]%[0D3G4]
Tìm tất cả các giá trị của tham số m để  hệ phương trình sau có nghiêm thực:
 $\heva{&\sqrt{x}+\sqrt{y}=1,\\ &x\sqrt{x}+y\sqrt{y}=1-6m.}$
\loigiai{
Điều kiện $x\geq0; y\geq0$.
Đặt $\sqrt{x}+\sqrt{y}=S;\sqrt{x}\sqrt{y}=P\quad (S,P \geq0; S^2\geq4P)$.
Khi đó hệ phương trình được viết lại là:
\begin{center}
$\heva{& S=1\\ &S^3-3PS=1-6m}\Leftrightarrow
\heva {&S=1\\ &P=2m}.$
\end{center}
Khi đó $S; P$ là 2 nghiệm của phương trình: $x^2-x+2m=0$.\\
Để hệ phương trình đã cho có nghiệm thì phương trình trên phải có hai nghiệm không âm\\
$\Leftrightarrow$
$\heva{&\delta =1-4m \geq0 \\ &P=2m\geq0\\&S=1\geq0 \quad\text{(luôn đúng)}}\Leftrightarrow 0\leq m\leq\dfrac{1}{4}$.
Vậy tất cả các giá trị m cần tìm là: $0\leq m\leq\dfrac{1}{4}$.
}
\end{vd}

\end{center}

}
\end{bt}

\begin{bt}%[Nguyễn Thị Thu]%[0D3K4]
Giải hệ phương trình sau: $\heva{&x^2+y^2+xy=4y-1,\\ &x^3+x^2y+x-3y=0.}$
\loigiai{
Hệ đã cho có thể viết lại là:
$\heva{&x^2+1+y^2+xy=4y\quad &(1),\\ &(x^2+1)(x+y)=4y\quad &(2).}$ \\
Dễ thấy $y=0$ không thỏa mãn hệ phương trình. Chia cả hai vế phương trình $(1), (2)$ cho $y$ ta được hệ:
$\heva{&\dfrac{x^2+1}{y}+(x+y)=4,\\&\dfrac{x^2+1}{y}(x+y)=4.}$\\
Đặt $\dfrac{x^2+1}{y}=u; x+y=v$ ta được hệ phương trình:
$\heva{&u+v=4\\ &uv=4}\Leftrightarrow \heva{&u=2\\ &v=2}\Leftrightarrow \heva{&\dfrac{x^2+1}{y}=2,\\&x+y=2.}$\\
Giải hệ ta suy ra tập nghiệm của hệ là:
$S=\{(1;1);(-3;5)\}$.
}
\end{bt}



\begin{bt}%[Nguyễn Thị Thu]%[0D3G4]
	Tìm tập giá trị thực của tham số $m$ đề hệ sau có nghiệm $\heva{&x^2+y^2+2x+2y=2,\\&xy(x+2)(y+2)=m.}$
	\loigiai{
		Đặt $u=x^2+2x\geq -1; v=y^2+2y\geq -1\Rightarrow\heva{&u+v=2\\&uv=m}\quad (*).$\\
		Hệ trên có nghiệm $\Leftrightarrow$ hệ $(*)$ có hai nghiệm $\geq -1\Leftrightarrow $ phương trình $t^2-2t+m=0$ có hai nghiệm thỏa mãn
$-1\leq x_1\leq x_2
\Leftrightarrow\heva{&\Delta'\geq0\\&(x_1+1)(x_2+1)\geq 0\\&(x_1+1)+(x_2+1)\geq 0 }
\Leftrightarrow\heva{&m\leq 1\\&m+3\geq 0\\&4\geq 0 \quad\text{(luôn đúng)}}\Leftrightarrow-3\leq m\leq 1.$
	}
\end{bt}







\subsection{Hệ phương trình đối xứng loại $2$}
\begin{dn}
Hệ phương trình đối xứng loại $2$ là hệ phương trình có dạng
	$\heva{
		& f(x,y) = 0\\
		& f(y,x) = 0.
	}$
\end{dn}
\begin{note}
	Nếu hệ phương trình có nghiệm là $(a,b)$ thì nó cũng có nghiệm $(b,a)$.
\end{note}
\begin{dang}{Giải hệ phương trình đối xứng loại $2$.}
Phương pháp giải hệ phương trình đối xứng loại $2$:\\
 $f(x,y) - f(y,x) = 0 \Leftrightarrow (x-y) h(x,y) 	= 0  \Leftrightarrow
	\hoac{
		&x = y\\
		&h(x,y) = 0.
	}$
\end{dang}
\begin{note}
	Thường thì $h(x,y)$ là những phương trình dễ dàng tìm ra mối liên hệ giữa $x$ và $y$; hoặc $h(x,y)$ là phương trình vô nghiệm.
\end{note}
\begin{vd}%[Mai Hà Lan]%[0D3K4]
Giải hệ phương trình
$\heva{
	& x^2 - 2018 x = 2017 y \\
 	& y^2 - 2018 y = 2017 x.
}$
\loigiai{
	$\heva{
		& x^2 - 2018 x = 2017 y \quad \text{(1)}\\
		& y^2 - 2018 y = 2017 x. \quad \text{(2)}
	}$\\
	Lấy $(1)$ trừ $(2)$ vế theo vế ta được
	\begin{eqnarray*}
		&& (x^2 - y^2) - 2018 (x-y) = 2017 (y-x)\\
		&\Leftrightarrow& (x-y) (x+y-1) = 0\\
		&\Leftrightarrow& \hoac{& y = x \\ & y = -x + 1.}
	\end{eqnarray*}
	Vậy hệ phương trình đã cho tương đương với
	\begin{eqnarray*}
		(I) \heva{& y = x\\ & x^2 - 2018 x = 2017 y}
		\quad \text{hoặc} \quad (II)
		\heva{& y = -x + 1\\ & x^2 - 2018 x = 2017 y}
	\end{eqnarray*}
	\begin{itemize}
		\item Giải $(I)$:
			$(I) \Leftrightarrow \hoac{& x = y = 0\\ & x = y = 4035.} $
		\item Giải $(II)$:
		$
		(II) \Leftrightarrow  \heva{& x = \dfrac{1 + \sqrt{8069}}{2} \\ & y = \dfrac{1 - \sqrt{8069}}{2} .}
		\quad \text{hoặc} \quad
		\heva{& x = \dfrac{1 - \sqrt{8069}}{2} \\ & y = \dfrac{1 + \sqrt{8069}}{2} .} 	
	 	$
	\end{itemize}
	Kết luận, hệ phương trình có bốn nghiệm: $(0;0)$, $(4035;4035)$, $\left( \dfrac{1 - \sqrt{8069}}{2}; \dfrac{1 + \sqrt{8069}}{2} \right)$,\\ $\left( \dfrac{1 + \sqrt{8069}}{2}; \dfrac{1 - \sqrt{8069}}{2} \right)$.
}
\end{vd}
\begin{vd}%[Mai Hà Lan]%[0D3G4]

$\heva{
	& x^3 + 2 = 3y \\
	& y^3 + 2 = 3x.
}$
\loigiai{
	$\heva{
		& x^3 + 2 = 3y \quad \text{(1)}\\
		& y^3 + 2 = 3x. \quad \text{(2)}
	}$\\

		&& (x - y)(x^2 + xy + y^2) = 3 (y-x)\\

	\end{eqnarray*}

		\item Giải $(I)$:

\end{vd}

\begin{vd}%[Mai Hà Lan]%[0D3G4]
	Tìm điều kiện của tham số $m$ để hệ phương trình sau có nghiệm duy nhất
	$$\heva{
		& x - 2 y  = m \dfrac{y}{x}\\
		& y - 2 x  = m \dfrac{x}{y}.
	}$$
	\loigiai{
		Vì hệ phương trình đối xứng nên giả sử nghiệm của hệ là $(x;y)$ thì $(y;x)$ cũng là nghiệm của hệ, vậy để hệ có nghiệm duy nhất thì $x = y$. Suy ra $(1)$ trở thành\\
		$$ x - 2 x  = m \Leftrightarrow - x = m \Leftrightarrow x = -m.$$
		Để hệ phương trình có nghiệm duy nhất thì $x = y = - m \ne 0 \text{, suy ra } m \ne 0$.\\
		Thử lại, với $m \ne 0$, $x \ne 0, y \ne 0$ thì hệ phương trình tương đương với
		$\heva{
			& x^2 - 2 xy  = m y \quad \text{(1)}\\

		}$\\

		\begin{eqnarray*}
			\Leftrightarrow
			(I) \heva{& y = x\\ & x^2 - 2 xy  = m y}
			\quad \text{hoặc} \quad (II)
			\heva{& y = -x -m\\ & x^2 - 2 xy  = m y}
		\end{eqnarray*}
		\begin{itemize}
			\item Giải $(I)$:
			$(I) \Leftrightarrow \hoac{& x = y = 0 \, \text{ (loại) }\\ & x = y = - m.} $
			\item Giải $(II)$: Từ hệ $(II)$ ta được phương trình $3x^2 + 3mx  + m^2 = 0$. Có $\Delta = -3m^2 <0, \forall m \ne 0$. Nên hệ phương trình $(II)$ vô nghiệm.
		\end{itemize}
		Kết luận, hệ phương trình có nghiệm duy nhất khi và chỉ khi $m \ne 0$.
	}
\end{vd}

\subsection{Hệ phương trình đẳng cấp}
\begin{dn}
	Biểu thức $ f(x,y) $ được gọi là biểu thức đẳng cấp bậc $ 2 $ nếu $ f(mx,my)=m^2f(x,y) $.
\end{dn}
\begin{dn}
	Biểu thức $ f(x,y) $ được gọi là biểu thức đẳng cấp bậc $ 3 $ nếu $ f(mx,my)=m^3f(x,y) $.
\end{dn}
\begin{dn}
	Hệ phương trình đẳng cấp bậc $ 2 $ theo $ x,y $ có dạng tổng quát:
\begin{align}\label{he1}
	\begin{cases}
		a_1x^2+b_1xy+c_1y^2=d_1 \\
		a_2x^2+b_2xy+c_2y^2=d_2.
	\end{cases}
\end{align}
\textit{(Mỗi phương trình trong hệ \eqref{he1} là các biểu thức đẳng cấp bậc 2)}\\
Phương pháp giải:
\begin{itemize}
	\item Xét $ x=0 $. Thay $ x=0 $ vào \eqref{he1} để tìm $ y $. Nếu không tìm được $ y $ thì hệ vô nghiệm trong trường hợp này.
	\item Xét $ x\neq0 $.
	\begin{enumerate}[(i)]
		\item \label{1} Nếu $ d_1=0 $ hoặc $ d_2=0 $, chẳng hạn, $ d_1=0 $ thì ta chia cả hai vế phương trình thứ nhất cho $ x^2 $ ta được phương trình có dạng: 	$ c_1(\frac{y}{x})^2+b_1(\frac{y}{x})+a_1 =0 $. Giải phương trình này ta tìm được  tỉ số $ \frac{y}{x} $, sau đó thế vào phương trình còn lại để tìm nghiệm $ x,y $.
		\item Nếu $ d_1\neq0 $ và $ d_2\neq0 $ thì ta có thể tạo ra một phương trình đẳng cấp bậc $ 2 $ thuần nhất (\textit{phương trình có hệ số tự do bằng $ 0 $)} bằng cách:
		\\$d_2( a_1x^2+b_1xy+c_1y^2-d_1)-
		d_1(a_2x^2+b_2xy+c_2y^2 -d_2)=0 $. Sau đó giải giống \eqref{1}.
	\end{enumerate}
\end{itemize}
\end{dn}

\subsection{Hệ phương trình hai ẩn khác}

Giải hệ phương trình $\heva{&4x^2+y^2=5,\\&8x^3+y^3=5y+4x^2y.}$
\loigiai{Xét hệ $\heva{&4x^2+y^2=5\quad &(1),\\&8x^3+y^3=5y+4x^2y\quad &(2).}$\\
$\bullet$ \textbf{Cách Thế biểu thức:} Để ý thấy $(1)\Leftrightarrow 4x^2=5-y^2$, $(2)\Leftrightarrow 8x^3-4x^2y=y(5-y^2)$, nên ta có thể giải bằng phương pháp thế biểu thức $5-y^2=4x^2$ vào $(2)$ và nhận được phương trình:
\begin{center}
$8x^3-4x^2y=y.4x^2\Leftrightarrow 8x^3-8x^2y=0\Leftrightarrow\hoac{& x=0\\& x=y}.$
\end{center}
+/ Với $x=0\Rightarrow y=\pm\sqrt{5}$.\\
+/ Với $x=y\Rightarrow 5x^2=5\Leftrightarrow x=\pm 1\Rightarrow y=\pm 1$.\\
Vậy hệ có tập nghiệm $S=\{(0; \sqrt{5}), (0; -\sqrt{5}), (1; 1), (-1; -1)\}$.\\
 $\bullet$ \textbf{Cách Thế hằng số:} Để ý thấy có hệ số $5$ ở cả hai phương trình. Ở phương trình $(2)$, hệ số $5$ nằm ở số hạng có bậc $1$, trong khi tất cả các số hạng khác có bậc $3$.  \begin{center}

 \end{center}

					  	$\Leftrightarrow \hoac{&2m+m\sqrt{m}=0 \text{ (vô nghiệm) }   \\&2m+m\sqrt{m}=1\\&2m-m\sqrt{m}=0\\&2m-m\sqrt{m}=1.} \Leftrightarrow   \hoac{&m=\dfrac{3-\sqrt{5}}{2}\\&m=4\\&m=1 \text{ hoặc } m=\dfrac{3+\sqrt{5}}{2}.} $
					\end{enumerate}
			\end{enumerate}
			\item Vậy $ m\in\left\lbrace 1;4;\dfrac{3-\sqrt{5}}{2};\dfrac{3+\sqrt{5}}{2}\right\rbrace  $	.	
	\end{itemize}
	}
\end{bt}

Giải hệ phương trình $\heva{&x^3(2+3y)=8,\\&x(y^3-2)=6.}$
\loigiai{Xét hệ $\heva{&x^3(2+3y)=8\quad &(1),\\&x(y^3-2)=6\quad &(2).}$\\
+/ Trường hợp $x=0$ không thỏa mãn.\\
+/ Trường hợp $x\neq 0$, hệ đã cho tương đương với $\heva{&2+3y=\left(\dfrac{2}{x}\right)^3,\\&y^3-2=\dfrac{6}{x}.}$\\

Lấy $(3)+(4)$ ta được phương trình $y^3+3y=u^3+3u\Leftrightarrow (y-u)(y^2+yu+u^2+3)=0\Leftrightarrow y=u$.\\
Giải ra tập nghiệm $S=\{(-2; -1), (1; 2)\}$.
}
\end{bt}
\begin{bt}%[NguyệnNgô][0D3G4]
Giải hệ phương trình $\heva{&10x^2y-16xy^2+24y^3-2(x+2y)=0,\\&2xy(x^2+4y^2)+2=(x+2y)^2.}$

\begin{center}

 \begin{center}

\begin{bt}%[NguyệnNgô]%[0D3G4]
Giải phương trình $7x^2+7x=\sqrt{\dfrac{4x+9}{28}}$.
\loigiai{$\bullet$ Bài này có thể sử dụng phương pháp đặt ẩn phụ không hoàn toàn để đưa về hệ phương trình. Cách xây dựng ẩn phụ như sau: Quan sát vế trái có bậc hai nên ta sẽ đặt $\alpha t+\beta=\sqrt{\dfrac{4x+9}{28}}\quad (*)$, sao cho sau khi bình phương, rút gọn thu được phương trình dạng:
\begin{center}
$7t^2+7t=\alpha x+\beta\quad (**)$.
\end{center}
Từ $(*), (**)$ ta được hệ
\begin{center}
$\heva{&x=7\alpha^2t^2+14\alpha\beta t+7\beta^2-\dfrac{9}{4}\\&x=\dfrac{7}{\alpha}t^2+\dfrac{7}{\alpha}t-\dfrac{\beta}{\alpha}}$.
\end{center}
Suy ra
\begin{center}
$7\alpha^2t^2+14\alpha\beta t+7\beta^2-\dfrac{9}{4}=\dfrac{7}{\alpha}t^2+\dfrac{7}{\alpha}t-\dfrac{\beta}{\alpha}$.
\end{center}
Đồng nhất hệ số hai vế ta có $\alpha=1, \beta=\dfrac{1}{2}$.\\
$\bullet$ Cách giải: Xét phương trình $7x^2+7x=\sqrt{\dfrac{4x+9}{28}}\quad (1)$.\\
Đặt $t+\dfrac{1}{2}=\sqrt{\dfrac{4x+9}{28}}\quad (2), t\geq-\dfrac{1}{2}$. Suy ra $7t^2+7t=x+\dfrac{1}{2}\quad (3)$. Từ $(1), (2), (3)$ ta có hệ đối xứng loại 2 với điều kiện $ t\geq-\dfrac{1}{2}$:
\begin{center}
$\heva{&7x^2+7x=t+\dfrac{1}{2},\\&7t^2+7t=x+\dfrac{1}{2}.}$
\end{center}
Giải hệ này ta được hai nghiệm $\heva{&x=\dfrac{-6+5\sqrt{2}}{14}\\&t=\dfrac{-6+5\sqrt{2}}{14}}$ và $\heva{&x=\dfrac{-8-\sqrt{46}}{14}\\&t=\dfrac{-8+\sqrt{46}}{14}}$.
Vậy phương trình ban đầu có tập nghiệm $S=\left\{\dfrac{-8-\sqrt{46}}{14}; \dfrac{-6+5\sqrt{2}}{14}\right\}$.
}
\end{bt}
