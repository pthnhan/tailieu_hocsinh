\section{Hai đường thẳng song song}
\subsection{Tóm tắt lí thuyết}
\begin{dn}
	Hai đường thẳng $ a $ và $ b $ gọi là song song với nhau nếu chúng không có điểm chung.\\
	Kí hiệu $ a\parallel b $.
\end{dn}
\begin{tc}
	Hai đường thẳng song song thì
	\immini{
\begin{enumerate}
	\item các góc đồng vị bằng nhau;
	\item các góc so le trong bằng nhau;
	\item các góc so le ngoài bằng nhau;
	\item các góc trong cùng phía bù nhau.
\end{enumerate}	
}{

\end{tc}\noindent
\textbf{Dấu hiệu nhận biết:}
\begin{enumerate}
	\item Nếu có một cặp góc so le trong (hoặc đồng vị) bằng nhau hoặc trong cùng phía bù nhau thì hai đường thẳng tương ứng song song;
	\item Hai đường thẳng phân biệt cùng song song với đường thẳng thứ ba thì song song với nhau.
\end{enumerate}
\begin{dn}
	Hai góc có cạnh tương ứng song song là hai góc có đường thẳng chứa mỗi cạnh của góc này tương ứng song song với đường thẳng chứa một cạnh của góc kia.
\end{dn}
\subsection{Hỏi đáp nhanh}
\begin{ex}%[7H1Y4]
	Cho đường thẳng $ c $ cắt hai đường thẳng $ a,\ b $. Nếu có một cặp góc so le trong bằng nhau thì ta có kết luận được các cặp góc so le còn lại bằng nhau và các cặp góc đồng vị cũng bằng nhau hay không? Vì sao?
	\loigiai{
	Nếu có một cặp góc so le trong bằng nhau thì $ a\parallel b $, do đó các cặp góc so le còn lại bằng nhau và các cặp góc đồng vị cũng bằng nhau.
}
\end{ex}

\begin{ex}%[7H1Y4]
	Cho hai đường thẳng $ a$ và $ b $ song song với nhau, đường thẳng $ c $ cắt $ a,\ b $ tương ứng tại $ A,\ B $. Biết rằng hai góc trong cùng phía bằng nhau, hỏi có tính được các góc còn lại với đỉnh $ A $ và $ B $ hay không?
	\loigiai{

			\widehat{A}_{1}+\widehat{B}_{1}&=&180^{\circ}\\
			2\widehat{A}_{1}&=&180^{\circ}\\
			\widehat{A}_{1}&=&90^{\circ}.
		\end{eqnarray*}
	Vậy các góc còn lại của đỉnh $ A $ và $ B $ đều bằng $ 90^{\circ} $.

			Vì $ \widehat{A}_{2}$ và $\widehat{B}_{1} $ là hai góc so le trong nên $ a\parallel b $.
			\item Ta có $ \widehat{B}_{1}+\widehat{B}_{2}=180^{\circ} $ (hai góc kề bù). Mà $ \widehat{A}_{1}+\widehat{B}_{1}=180^{\circ} $ (giả thiết) nên $ \widehat{A}_{1}=\widehat{B}_{2} $.\\
			Vì $ \widehat{A}_{1}$ và $\widehat{B}_{2} $ là hai góc đồng vị nên $ a\parallel b $.
			\item Ta có $ \widehat{A}_{1}=\widehat{A}_{3} $ (hai góc đối đỉnh). Mà $ \widehat{A}_{1}+\widehat{B}_{1}=180^{\circ} $ (giả thiết) nên $ \widehat{A}_{3}+\widehat{B}_{1}=180^{\circ} $.\\
			Vì $ \widehat{A}_{3}$ và $\widehat{B}_{1} $ là hai góc trong cùng phía nên $ a\parallel b $.
		\end{enumerate}
	}
\end{vd}
\begin{note}
	Muốn chứng minh hai đường thẳng song song, ta có thể chứng minh một trong các khẳng định sau:
	\begin{enumerate}
		\item Hai góc đồng vị bằng nhau;
		\item Hai góc so le trong bằng nhau;
		\item Hai góc trong cùng phía bù nhau.
	\end{enumerate}
\end{note}

		Ta tính các góc tại đỉnh $ M $ và $ N $:
		\begin{enumerate}[TH1.]
			\item Tại $ M $, ta có $ \widehat{M}_{2}=\widehat{M}_{4}=100^{\circ} $ (hai góc đối đỉnh). Mà $ \widehat{M}_{2}+\widehat{M}_{3}=180^{\circ} $ (hai góc kề bù) nên 
			\begin{eqnarray*}
				\widehat{M}_{3}&=&180^{\circ}-\widehat{M}_{2}\\
				\widehat{M}_{3}&=&80^{\circ}.
			\end{eqnarray*}

			\item Tại $ N $, ta có $ \widehat{N}_{2}=\widehat{M}_{4}=100^{\circ} $ (hai góc so le trong), do đó $ \widehat{N}_{4}=\widehat{N}_{2}=100^{\circ} $ (hai góc đối đỉnh). Vì $ \widehat{N}_{2} $ và $ \widehat{3} $ là hai góc kề bù nên
			\begin{eqnarray*}
				\widehat{N}_{2}+\widehat{N}_{3}&=&180^{\circ}\\
				\widehat{N}_{3}&=&180^{\circ}-\widehat{N}_{2}\\
				\widehat{N}_{3}&=&80^{\circ}.
			\end{eqnarray*}
			Vì $ \widehat{N}_{1}$ và $\widehat{N}_{3} $ là hai góc đối đỉnh nên $ \widehat{N}_{1}=\widehat{N}_{3}=80^{\circ} $.
		\end{enumerate}
	}
\end{vd}
