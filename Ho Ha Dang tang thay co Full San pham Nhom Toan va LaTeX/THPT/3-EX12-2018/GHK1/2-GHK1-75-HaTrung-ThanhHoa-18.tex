		$\Rightarrow 2a_2=1\cdot \left(1+2+\cdots+2017\right)+2\cdot \left(1+2+\cdots+2017\right)+\cdots+2017\cdot (1+2+\cdots+2017)-\left(1^2+2^2+\cdots +2017^2\right).$

		$T=a_2+\dfrac{1}{2}\left(1^2+2^2+\cdots+2017^2\right)=\dfrac{2a_2+\left(1^2+2^2+\cdots+2017^2\right)}{2}=$\\
		$=\dfrac{1}{2}\cdot \left[1\cdot \left(1+2+\cdots+2017\right)+2\cdot \left(1+2+\cdots+2017\right)+\cdots+2017\cdot \left(1+2+\cdots+2017\right)\right]=$\\
		$=\dfrac{1}{2}\cdot \dfrac{2017.2018}{2}\cdot \dfrac{2017\cdot 2018}{2}=\dfrac{1}{2}{\left(\dfrac{2017\cdot 2018}{2}\right)}^2.$}
\end{ex}
