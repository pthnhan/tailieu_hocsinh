\begin{listEX}[2]
	\item $5xy^2+10y$.
	\item $4x^2-y^2+5\left(2x-y\right)$.   
	\item $25-a^2+2ab-b^2$. 
	\item $x^2+x-30$.   
\end{listEX}	
\loigiai{
\begin{listEX}[2]
	\item $\begin{aligned}[t]
	&&& 5xy^2+10y\\
	=&&& 5y(xy+2). 
	\end{aligned}$
	\item $\begin{aligned}[t]
	&&& 4x^2-y^2+5\left(2x-y\right)\\
	=&&& \left(2x+y\right)\left(2x-y\right)+5\left(2x-y\right)\\
	=&&&\left(2x-y\right)\left(2x+y+5\right). 
	\end{aligned}$
	\item $\begin{aligned}[t]
	&&& 25-a^2+2ab-b^2\\
	=&&& 5^2-\left(a^2-2ab+b^2\right)\\
	=&&&5^2-\left(a-b\right)^2\\
	=&&&\left(5+a-b\right)\left(5-a+b\right).
	\end{aligned}$
	\item $\begin{aligned}[t]
	&&&x^2+x-30 \\
	=&&&x^2+6x-5x-30\\
	=&&&x\left(x+6\right)-5x\left(x+6\right)\\
	=&&&\left(x+6\right)\left(x-5\right).
	\end{aligned}$
\end{listEX}
	}

Tìm $x$ biết
\begin{listEX}[2]
	\item $\left(x-5\right)^2-\left(x+3\right)\left(x-3\right)=14$.       
	\item $\left(x+7\right)^2-3x-21=0$.       
\end{listEX}
	\loigiai{
\begin{enumerate}
	\item \begin{eqnarray*}
		\left(x-5\right)^2-\left(x+3\right)\left(x-3\right)&=&14 \\     
		x^2-10x+25-x^2+9&=&14\\
		-10x+34&=&14\\
		-10x&=&14-34\\
		-10x&=&-20\\
		x&=&2.
	\end{eqnarray*}
	\item \begin{eqnarray*}
		\left(x+7\right)^2-3x-21&=&0\\
		(x+7)^2-3(x+7)&=&0\\
		(x+7)(x+7-3)&=&0\\
		(x+7)(x+4)&=&0\\
		x+7=0 &\;\text{hoặc}\;& x+4=0\\
		x=-7&\;\text{hoặc}\;& x=-4.
	\end{eqnarray*}
\end{enumerate}	
	}
\end{bt}

\begin{bt}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Tân Bình, HCM]%[Tuấn Nguyễn, dự án 8DT18-HK1]%[8H1B9]%[8H1B4]%[8H1B9]%[8D1G9]
Cho tam giác $ABC$ vuông tại $A$ ($AB<AC$), đường cao $AH$. Gọi $O$ là trung điểm của $BC$, $D$ là điểm đối xứng của $A$ qua $O$.
\begin{enumerate}
	\item Chứng minh tứ giác $ABDC$ là hình chữ nhật.
	\item Trên tia đối của tia $HA$ lấy điểm $E$ sao cho $HE=HA$. Chứng minh $ED=2HO$.
	\item Chứng minh tứ giác $BEDC$ là hình thang cân.
	\item Gọi $M$, $N$ lần lượt là hình chiếu của điểm $E$ lên hai đường thẳng $BD$ và $CD$. Chứng minh ba điểm  $H, M, N$ thẳng hàng.
\end{enumerate}	
	\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1.2, line join=round, line cap=round]
\clip (-0.5,-0.5) rectangle (5.7,5);
\tkzDefPoints{0/0/A,0/3/B,5/0/C}
\tkzDrawAltitude(C,B)(A)\tkzGetPoint{H}
\tkzDefMidPoint(C,B) \tkzGetPoint{O}
\tkzDefPointBy[symmetry = center O](A) \tkzGetPoint{D}
\tkzDefPointBy[symmetry = center H](A) \tkzGetPoint{E}
\tkzDrawAltitude(D,B)(E)\tkzGetPoint{M}
\tkzDrawAltitude(D,C)(E)\tkzGetPoint{N}
\tkzInterLL(M,N)(E,D) \tkzGetPoint{I}
\tkzInterLL(E,M)(A,D) \tkzGetPoint{K}
\tkzDrawPoints(A,B,C,H,O,D,E,M,N,I,K)
\tkzDrawSegments(A,B B,C C,A H,E B,D D,C N,D A,D E,D H,N M,K)
\tkzLabelPoints[below](O,I)
\tkzLabelPoints[right](C,D,N)
\tkzLabelPoints[left](A,B,H,E)
\tkzLabelPoints[below right](M,K)
\tkzMarkRightAngles(A,H,C E,M,D E,N,D B,A,C)
\tkzMarkSegments[mark=|](O,B O,C O,A O,D)
\tkzMarkSegments[mark=||](A,H H,E)
\end{tikzpicture}
\end{center}
\begin{enumerate}
	\item Chứng minh tứ giác $ABDC$ là hình chữ nhật.\\
	Tứ giác $ABDC$ có $O$ vừa là trung điểm của $BC$ vừa là trung điểm của $AD$ nên tứ giác $ABDC$ là hình bình hành.\\
	Mặt khác $\widehat{BAC}=90^\circ$ (vì $ABC$ vuông tại $A$)\\
	Suy ra tứ giác $ABDC$ là hình chữ nhật. 
	\item Trên tia đối của tia $HA$ lấy điểm $E$ sao cho $HE=HA$. Chứng minh $ED=2HO$.\\
	Trong tam giác $AED$ có $\heva{&O \;\text{là trung điểm } AD\\&H\;\text{là trung điểm }AE}$ suy ra $HO$ là đường trung bình tam giác $AED$
	suy ra $HO=\dfrac{1}{2}ED$ hay $ED=2HO$.
	\item Chứng minh tứ giác $BEDC$ là hình thang cân.\\
	Vì $HO$ là đường trung bình tam giác $AED$ suy ra $HO \parallel ED$ hay $BC \parallel ED$ suy ra tứ giác $BEDC$ là hình thang \hfill (1).\\
	Tam giác $CAE$ có $CH$ vừa là đường cao vừa là đường trung tuyến nên  $\triangle ACE$ cân tại $C$
	suy ra $EC = AC$.\\
	Mà $BD = AC$ (do tứ giác $ABDC$ là hình chữ nhật) suy ra  $EC = BD$ \hfill (2).\\
	Từ (1) và (2) suy ra tứ giác $BEDC$ là hình thang cân.
	\item Gọi $M, N$ lần lượt là hình chiếu của điểm $E$ lên hai đường thẳng $BD$ và $CD$. Chứng minh ba điểm  $H, M, N$ thẳng hàng. \\
	Tứ giác $MEND$ có $\widehat{EMD}=\widehat{END}=\widehat{NDM}=90^\circ$ nên tứ giác $MEND$ là hình chữ nhật.\\
	Gọi $I$  là giao điểm $ED$ và $MN$
	suy ra $I$ là trung điểm $ED$ và $I$ là trung điểm $MN$.\\
	Vì $HI$ là đường trung bình tam giác $AED$ suy ra $HI \parallel AD$ \hfill (3)\\
	Gọi $K$  là giao điểm $EM$ và $AD$. $\triangle EDK$ cân tại $D$ có $DM$ là đường cao suy ra $DM$ là đường trung tuyến của $\triangle EDK$ suy ra $M$ là trung điểm $EK$.\\ 
	Ta có $MI$ là đường trung bình tam giác $EKD$
	suy ra $MI \parallel KD$.\\
	Mà $A, K, D$ thẳng hàng
	suy ra $MI \parallel AD$ \hfill (4).\\
	Từ (3) và (4) suy ra $H , M, I$ thẳng hàng.\\
	Mà $M , I, N$ thẳng hàng (do $I$ là trung điểm $MN$)
	suy ra $H , M, N$ thẳng hàng.
\end{enumerate}	
	}

Gần tết Bác An có một phòng cần thay đổi gạch lót sàn. Biết chiều dài cần $20$ viên gạch, chiều rộng cần $10$ viên gạch. Mỗi viên gạch có kích thước $40$ cm $\times$ $40$ cm với giá là $65000$ đồng/viên gạch.
\begin{enumerate}
	\item Tính chiều dài, chiều rộng của căn phòng.
	\item Hỏi hỏi diện tích sàn của căn phòng nhà bác An là bao nhiêu mét vuông?
	\item Hãy tính tiền bác An cần mua gạch để lót sàn?
\end{enumerate}	
	\loigiai{
\begin{enumerate}
	\item Tính chiều dài, chiều rộng của căn phòng. \\
	Chiều dài của căn phòng: $20 \times 40 = 800$ (cm) = $8$ (m).\\
	Chiều dài của căn phòng: $10 \times 40 = 400$ (cm) = $4$ (m).
	\item Hỏi hỏi diện tích sàn của căn phòng nhà bác An là bao nhiêu mét vuông?\\
	Diện tích sàn của căn phòng: $8 \times 4 = 32$ (m$^2$).
	\item Hãy tính tiền bác An cần để mua gạch? \\
	Số viên gạch cần mua: $20 \times 10 = 200$ (viên gạch).\\
	Số tiền để mua gạch: $200 \times 65 000 = 13 000 000$ (đồng).
\end{enumerate}	
	}
\end{bt}
