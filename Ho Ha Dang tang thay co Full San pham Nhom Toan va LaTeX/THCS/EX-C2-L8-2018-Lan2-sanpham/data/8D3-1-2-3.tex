\section{Mở đầu về phương trình}
\section{Phương trình bậc nhất một ẩn và cách giải}
\subsection{Trọng tâm kiến thức}
\begin{enumerate}
	\item Phương trình một ẩn: Một phương trình với ẩn $x$ có dạng $A(x)=B(x)$, trong đó vế trái $A(x)$ và vế phải $B(x)$ là hai biểu thức của cùng một biến $x$.\\
	Giá trị của biến thoải mãi phương trình là một nghiệm của phương trình.\\
	Tập hợp tất cả các nghiệm của phương trình được gọi là tập nghiệm của phương trình đó, thường kí hiệu $S$.
	\item Giải phương trình là tìm tất cả các nghiệm của phương trình đó.
	\item Phương trình tương đương: Hai phương trình có cùng một tập nghiệm là hai phương trình tương đương, kí hiệu $\Leftrightarrow $.
	\item Phương trình bậc nhất một ẩn: Phương trình dạng $ax+b=0$, với $a$ và $b$ là hai số đã cho và $a\ne 0$, được gọi là phương trình bậc nhất một ẩn.
	\item Hai quy tắc biến đổi trương đương phương trình
	\begin{itemize}
		\item Quy tắc chuyển vế: Trong một phương trình ta có thể chuyển một hảng tử từ vế này sang vế kia và đổi dấu hạng tử đó.
		\item Quy tắc nhân với một số: Trong một phương trình, ta có thể nhân (hoặc chia) cả hai vế với cùng một số khác $0$.
	\end{itemize}
	\item Cách giải phương trình bậc nhất một ẩn
	\[ax+b=0\Leftrightarrow ax=-b\Leftrightarrow x=\dfrac{-b}{a}.\]
	Phương trình $ax+b=0$ luôn có một nghiệm duy nhất $x=\dfrac{-b}{a}$.
\end{enumerate}
\subsection{Các dạng bài tập và phương pháp giải}
\begin{dang}{Xét xem giá trị $x=a$ có là nghiệm của phương trình không?}
Thay $x=a$ vào hai vế của phương trình. Nếu hai vế có cùng một giá trị thì $x=a$ là một nghiệm của phương trình đó.
\end{dang}
\begin{vd}%[8D3Y2]
Với mỗi phương trình sau, xét xem $x=-2$ có phải là nghiệm của nó không?
\begin{enumEX}{3}
	\item $5x+3=2x-3$.
	\item $\dfrac{1}{4}x-1=x+\dfrac{1}{2}$.
	\item $-x+5=2(1-x)$.
\end{enumEX}
\loigiai{
	\begin{enumerate}
		\item Với $x=-2$ thì 
		{\allowdisplaybreaks
			\begin{eqnarray*}
				5x+3 & = & 5 \cdot (-2)+3=-7.\\
				2x-3 & = & 2 \cdot (-2)-3=-7.
			\end{eqnarray*}
		}
		Vậy $x=-2$ là nghiệm của phương trình $5x+3=2x-3$.
		\item Với $x=-2$ thì
		{\allowdisplaybreaks
			\begin{eqnarray*}
				\dfrac{1}{4}x-1 & = & \dfrac{1}{4} \cdot (-2)-1=\dfrac{-3}{2}\\
				x+\dfrac{1}{2} & = & -2+\dfrac{1}{2}=-\dfrac{3}{2}.
			\end{eqnarray*}
		}
		Vậy $x=-2$ là nghiệm của phương trình $\dfrac{1}{4}x-1=x+\dfrac{1}{2}$
		\item Với $x=-2$ thì
		{\allowdisplaybreaks
			\begin{eqnarray*}
				-x+5 & = & -(-2)+5=7\\
				2(1-x) & = & 2\left(1+2\right)=6.
			\end{eqnarray*}
		}
		Vì $7\ne 6$ nên $x=-2$ không là nghiệm của phương trình $-x+5=2(1-x)$.
	\end{enumerate}
}
\end{vd}
\begin{vd}%[8D3Y2]
Phương trình nào dưới đây có nghiệm $x=3$?
\begin{enumEX}{3}
	\item $2x+7=10+x$.
	\item $-3x+8=4-x$.
	\item $\dfrac{1}{9}x+\dfrac{2}{3}=x-2$.
\end{enumEX}
\loigiai{
	\begin{enumerate}
		\item Với $x=3$ thì
		{\allowdisplaybreaks
			\begin{eqnarray*}
				2x+7 & = & 2 \cdot 3+7=13\\
				10+x & = & 10+3=13.
			\end{eqnarray*}
		}
		Vậy $x=3$ là nghiệm của phương trình $2x+7=10+x$
		\item Với $x=3$ thì
		{\allowdisplaybreaks
			\begin{eqnarray*}
				-3x+8 & = & -3 \cdot 3+8=-1\\
				4-x & = & 4-3=1.
			\end{eqnarray*}
		}
		Vì $-1\ne 1$ nên $x=3$ không là nghiệm của phương trình $-3x+8=4-x$
		\item Với $x=3$ thì
		{\allowdisplaybreaks
			\begin{eqnarray*}
				\dfrac{1}{9}x+\dfrac{2}{3} & = & \dfrac{1}{9} \cdot 3+\dfrac{2}{3}=1\\
				x-2 & = & 3-2=1.
			\end{eqnarray*}
		}
		Vậy $x=3$ là nghiệm của phương trình $\dfrac{1}{9}x+\dfrac{2}{3}=x-2$.
	\end{enumerate}
}
\end{vd}
\begin{vd}%[8D3Y2]
Với giá trị nào của $m$ thì phương trình $\dfrac{2}{5}mx=4x+2$ nhận $x=-5$ là nghiệm?
\loigiai{
$x=-5$ là nghiệm của phương trình $\dfrac{2}{5}mx=4x+2$ khi $\dfrac{2}{5} \cdot m \cdot \left(-5\right)=4 \cdot \left(-5\right)+2$.\\
Hay
\[-2m=-18 \Leftrightarrow m=(-18):(-2) \Leftrightarrow m=9.\]
}
\end{vd}
\begin{vd}%[8D3Y2]
Thử lại để thấy rằng phương trình $mx-2=-x+3m+1$ luôn nhận $x=3$ là nghiệm dù $m$ lấy bất kì giá trị nào.
\loigiai{
Khi $x=3$ thì vế trái có giá trị là $3m-2$; vế phải có giá trị $3m-2$\\
Cả hai vế đều có cùng giá trị là $3m-2$.\\
Vậy $x=3$ là nghiệm của phương trình $mx-2=-x+3m+1$ với mọi giá trị của $m$.
}
\end{vd}
\begin{dang}{Xét xem hai phương trình có tương đương không?}
	\begin{itemize}
		\item Dựa vào định nghĩa: Nếu hai phương trình có cùng một tập nghiệm thì hai phương trình đó tương đương.
		\item Dùng quy tắc chuyển vế hai quy tắc nhân để biến đổi phương trình này thành phương trình kia.
	\end{itemize}
\end{dang}

\begin{vd}%[8D3B2]
	Cho hai phương trình $3x=0$ và $x(x+3)=0$. Hai phương trình trên có tương đương không? Vì sao?
	\loigiai{
		Phương trình $3x=0$ có tập nghiệm $S_1=\left\{ 0\right\}$.\\
		Phương trình $x(x+3)=0$ có tập nghiệm là $S_2=\left\{{0;-3}\right\}$\\
		Vì $S_1\ne{S_2}$ nên hai phương trình đã cho không tương đương.
	}
\end{vd}
\begin{vd}%[8D3B2]
	Cho hai phương trình $5x=10$ và $x-1=1$. Hai phương trình trên có tương đương không? Vì sao?
	\loigiai{
		Ta có
		{\allowdisplaybreaks
			\begin{eqnarray*}
				5x=10 & \Leftrightarrow & x=10:5\Leftrightarrow x=2\\
				x-1=1 & \Leftrightarrow & x=1+1\Leftrightarrow x=2.
			\end{eqnarray*}
		}
		Hai phương trình có cùng một tập nghiệm là $S=\left\{ 2\right\}$ nên chúng tương đương.
	}
\end{vd}
\begin{vd}%[8D3B2]
	Cho hai phương trình $x^2+10=1$ và $\dfrac{5}{x}=0$. Hai phương trình trên có tương đương không? Vì sao?
	\loigiai{
		Ta có $x^2+10=1\Leftrightarrow{x^2}=1-10\Leftrightarrow{x^2}=-9$.\\
		Không có số nào có bình phương là một số âm, suy ra $S_1= \varnothing $.\\
		Lại có $\dfrac{5}{x}=0\Rightarrow{S_2}=\varnothing $.\\
		Suy ra hai phương trình $x^2+10=1$ và $\dfrac{5}{x}=0$ có cùng một tập nghiệm.\\
		Vậy hai phương trình $x^2+10=1$ và $\dfrac{5}{x}=0$ tương đương với nhau.
	}
\end{vd}
\begin{vd}%[8D3B2]
	Chứng minh rằng hai phương trình $5x-1=2x+20$ và $\dfrac{3x-1}{4}=5$ tương đương.
	\loigiai{
		Xét phương trình $5x-1=2x+20$\\
		Chuyển $2x$ sang vế trái ta được $5x-2x-1=20$ hay $3x-1=20$\\
		Chia cả hai vế cho $4$ ta được $\dfrac{3x-1}{4}=5$.\\
		Vậy các phương trình $5x-1=2x+20$ và $\dfrac{3x-1}{4}=5$ tương đương.
	}
\end{vd}

\begin{dang}{Nhận dạng phương trình bậc nhất một ẩn số}
	Nếu phương trình có dạng $ax+b=0$ với $a$, $b$ là hai số đã cho $a\ne 0$ thì phương trình là phương trình là phương trình bậc nhất một ẩn.
\end{dang}

\begin{vd}%[8D3B2]
	Trong các phương trình sau, phương trình nào là phương trình bậc nhất một ẩn?
	\begin{enumEX}{3}
		\item $x+5=0$.
		\item $\dfrac{1}{2}x-3=0$.
		\item $2x^2+7=0$.
		\item $-\dfrac{5}{8}x=0$.
		\item $\dfrac{4}{x}+9=0$.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item Phương trình $x+5=0$ có dạng $ax+b=0$, trong đó $a=1$; $b=5$ nên phương trình bậc nhất một ẩn.
			\item Phương trình $\dfrac{1}{2}x-3=0$ có dạng $ax+b=0$, trong đó $a=\dfrac{1}{2}$; $b=-3$ nên phương trình bậc nhất một ẩn.
			\item Phương trình $2x^2+7=0$ không có dạng $ax+b=0$ nên không là phương trình bậc nhất một ẩn.
			\item Phương trình $\dfrac{-5}{8}x=0$ có dạng $ax+b=0$ trong đó $a=-\dfrac{5}{8}$; $b=0$ nên là phương trình bậc nhất một ẩn.
			\item Phương trình $\dfrac{4}{x}+9=0$ không có dạng $ax+b=0$ nên không là phương trình bậc nhất một ẩn.
		\end{enumerate}
	}
\end{vd}
\begin{vd}%[8D3B2]
	Với giá trị nào của $m$ thì mỗi phương trình sau là phương trình bậc nhất một ẩn?
	\begin{enumEX}{3}
		\item $\dfrac{1}{5}x+m-1=0$.
		\item $\left(m+3\right)x-2=0$.
		\item $\left(x-3\right)m+4=0$.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item Phương trình $\dfrac{1}{5}x+m-1=0$ với mọi giá trị của $m$ luôn có dạng $ax+b=0$ trong đó $a=\dfrac{1}{5}$; $b=m-1$ nên luôn là phương trình bậc nhất một ẩn.
			\item Phương trình $\left(m+3\right)x-2=0$ là phương trình bậc một ẩn khi $m+3\ne 0$ hay $m\ne-3$.
			\item Phương trình $\left(x-3\right)m+4=0$ hay $mx-3m+4=0$ là phương trình bậc nhất một ẩn khi $m\ne 0$.
		\end{enumerate}
		
	}
\end{vd}

\begin{dang}{Giải phương trình bậc nhất một ẩn}
	Áp dụng quy tắc chuyển về và quy tắc nhân.
\end{dang}

\begin{vd}%[8D3Y2]
	Giải phương trình
	\begin{enumEX}{3}
		\item $8x-24=0$.
		\item $15-3x=0$.
		\item $\dfrac{1}{7}x+\dfrac{2}{21}=0$.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			\[8x-24=0 \Leftrightarrow 8x=24 \Leftrightarrow x=24:8 \Leftrightarrow x=3.\]
			\item Ta có
			\[15-3x=0 \Leftrightarrow -3x=-15 \Leftrightarrow x=(-15):(-3) \Leftrightarrow x=5.\]
			\item Ta có
			\[\dfrac{1}{7}x+\dfrac{2}{21}=0 \Rightarrow \dfrac{1}{7}x=-\dfrac{2}{21} \Leftrightarrow x=\dfrac{-2}{21}:\dfrac{1}{7} \Leftrightarrow x=\dfrac{-2}{3}.\]
		\end{enumerate}
	}
\end{vd}
\begin{vd}%[8D3Y2]
	Giải các phương trình
	\begin{enumEX}{3}
		\item $4x+5=1$.
		\item $-5x+2=14$.
		\item $6x-3=8x+9$.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			\[4x+5=1 \Leftrightarrow 4x=1-5 \Leftrightarrow 4x=-4 \Leftrightarrow x=(-4):4 \Leftrightarrow x=-1.\]
			\item 
			\[-5x+2=14 \Leftrightarrow -5x=14-2 \Leftrightarrow -5x=12 \Leftrightarrow x=\dfrac{-12}{5}.\]
			\item
			\[6x-3=8x+9 \Leftrightarrow -2x=12 \Leftrightarrow x=12:(-2) \Leftrightarrow x=-6.\]
		\end{enumerate}
		
	}
\end{vd}
\begin{vd}%[8D3B2]
	Cho phương trình $\left(m^2-9\right)x-3=m$. Giải phương trình trong các trường hợp sau
	\begin{enumEX}{3}
		\item $m=2$.
		\item $m=3$.
		\item $m=-3$.
	\end{enumEX}
	
	\loigiai{
		\begin{enumerate}
			\item Khi $m=2$, phương trình đã cho trở thành
			\[-5x-3=2 \Leftrightarrow -5x=2+3 \Leftrightarrow -5x=5 \Leftrightarrow x=-1.\]
			\item Khi $m=3$, phương trình đã cho trở thành: $0x-3=3\Leftrightarrow 0x=6$.\\
			Phương trình này vô nghiệm.
			\item Khi $m=-3$, phương trình đã cho trở thành: $0x-3=-3\Leftrightarrow 0x=0$.\\
			Phương trình này nghiệm đúng với mọi $x$.
		\end{enumerate}
	}
\end{vd}
\subsection{Bài tập tự luyện}
\begin{bt}%[8D3Y2]
	Trong các phương trình dưới đây, phương trình nào có nghiệm $x=4$; phương trình nào có nghiệm $x=-4$?
	\begin{enumEX}{3}
		\item $-\dfrac{1}{2}x+5=3\left(x-3\right)$.
		\item $3x-2=x+4$.
		\item $x^2-15=1$.
	\end{enumEX}
	\loigiai{
		$x=4$ là nghiệm của phương trình $-\dfrac{1}{2}x+5=3\left(x-3\right)$ và phương trình $x^2-15=1$.\\
		$x=-4$ là nghiệm của phương trình $x^2-15=1$.
	}
\end{bt}
\begin{bt}%[8D3B2]
	Tìm giá trị của $m$ để phương trình $7x+4=x-2m$ nhận $x=-6$ là nghiệm?
	\loigiai{
		Phương trình $7x+4=x-2m$ nhận $x=-6$ là nghiệm khi
		\[7 \cdot (-6)+4=(-6)-2m \Leftrightarrow 2m=32 \Leftrightarrow m=16.\]
	}
\end{bt}
\begin{bt}%[8D3B2]
	Với giá trị nào của $m$ thì phương trình $5x-mx=2m+1$ là phương trình bậc nhất một ẩn.
	\loigiai{
		Thu gọn phương trình đã cho ta được $(5-m) \cdot x-(2m+1)=0$.\\
		Phương trình này là phương trình bậc nhất một ẩn khi $5-m\ne 0$ hay $m\ne 5$.
	}
\end{bt}
\begin{bt}%[8D3B2]
	Với giá trị nào của $m$ thì phương trình $4x+9-m=0$ có nghiệm khác $0$.
	\loigiai{
		\[4x+9-m=0 \Leftrightarrow 4x=m-9 \Leftrightarrow x=\dfrac{m-9}{4}.\]
		Ta có $x\ne 0$ khi $\dfrac{m-9}{4}\ne 0\Leftrightarrow m\ne 9$
	}
\end{bt}
\begin{bt}%[8D3Y2]
	Cho các phương trình
	\begin{enumEX}{3}
		\item $\dfrac{1}{3}x+4=11$.
		\item $-x+9=3$.
		\item $3x-5=x+7$.
	\end{enumEX}
	Hãy giải các phương trình trên rồi cho biết hai phương trình nào tương đương với nhau?
	\loigiai{
		Ta có
		{\allowdisplaybreaks
			\begin{eqnarray*}
				\dfrac{1}{3}x+4=11 & \Leftrightarrow & x=21\\
				-x+9=3 & \Leftrightarrow & x=6\\
				3x-5=x+7 & \Leftrightarrow & x=6.
			\end{eqnarray*}
		}
		Vậy phương trình $-x+9=3$ tương đương với phương trình $3x-5=x+7$.
	}
\end{bt}
\begin{bt}%[8D3B2]
	Tìm $m$ để phương trình $-2x+1=0$ và $mx-1=3$ tương đương với nhau.
	\loigiai{
		Phương trình $-2x+1=0$ có tập hợp nghiệm là $S_1=\left\{\dfrac{1}{2}\right\}$\\
		Phương trình $-2x+1=0$ tương đương với phương trình $mx-1=3$ khi $S_2=\left\{{\dfrac{1}{2}}\right\}$, tức là khi $m \cdot \dfrac{1}{2}-1=3\Rightarrow m=8$
	}
\end{bt}
\begin{bt}%[8D3B2]
	Cho phương trình $\left(m-1\right)\left(m-2\right)=-m+2$.\\
	Hãy giải phương trình trong các trong các trường hợp sau
	\begin{enumEX}{3}
		\item $m=1$.
		\item $m=2$.
		\item $m=0$.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item Khi $m=1$ thì phương trình đã cho trở thành $0x=1$, vô nghiệm.
			\item Khi $m=2$ thì phương trình đã cho thành $0x=0$, nghiệm đúng với mọi $x$.
			\item Khi $m=0$ thì phương trình đã cho trở thành $2x=2\Leftrightarrow x=1$.
		\end{enumerate}
	}
\end{bt}
\section{Phương trình đưa được về dạng $ax+b=0$}
\subsection{Trọng tâm kiến thức}
\subsubsection{Các bước giải}
	\begin{itemize}
		\item Quy đồng mẫu ở hai vế.
		\item Khử mẫu (bằng cách nhân hai vế với mẫu chung).
		\item Bỏ dấu ngoặc.
		\item Chuyển các hạng tử chứa ẩn sang một vế, các hằng số sang vế kia.
		\item Thu gọn và giải phương trình nhận được.
	\end{itemize}
\subsubsection{Chú ý}
Quá trình giải có thể dẫn đến trường hợp đặc biệt là hệ số của ẩn bằng $0$;
\begin{itemize}
	\item Trường hợp $0x=m$ với $m \ne 0$, phương trình vô nghiệm.
	\item Trường hợp $0x=m$, phương trình nghiệm đúng với mọi $x$.
\end{itemize}

\subsection{Các dạng bài tập và phương pháp giải}
\begin{dang}{Giải phương trình}
	Theo các bước nêu ở trên.
\end{dang}

\begin{vd}%[8D3Y3]
	Giải các phương trình sau
	\begin{enumerate}

		{\allowdisplaybreaks

				A=B & \Leftrightarrow & \dfrac{11x+3}{12}=\dfrac{3\left(x+5\right)}{8}\\
				& \Leftrightarrow & \dfrac{2 \cdot \left(11x+3\right)}{24}=\dfrac{9\left(x+5\right)}{24}\\
				& \Leftrightarrow & 22x+6=9x+45 \Leftrightarrow 22x-9x=45-6\\
