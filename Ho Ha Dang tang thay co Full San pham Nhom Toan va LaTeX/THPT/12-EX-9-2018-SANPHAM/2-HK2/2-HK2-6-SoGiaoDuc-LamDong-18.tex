\begin{ex}%[HK2 (2017-2018), Sở Giáo Dục Lâm Đồng]%[Lê Quốc Hiệp dự án EX9]%[2H3G2-8]
	Trong không gian $Oxyz$, cho hai mặt cầu $(S_1),~(S_2)$ có phương trình lần lượt là $(x-2)^2+(y-1)^2+(z-1)^2=16$ và $(x-2)^2+(y-1)^2+(z-5)^2=4$. Gọi $(P)$ là mặt phẳng thay đổi tiếp xúc với cả hai mặt cầu $(S_1),~(S_2)$. Tính khoảng cách lớn nhất từ gốc tọa độ $O$ đến mặt phẳng $(P)$.
	\choice
	{$\dfrac{9}{2}-\sqrt{15}$}
	{$\sqrt{15}$}
	{\True $\dfrac{9+\sqrt{15}}{2}$}
	{$\dfrac{8\sqrt{3}+\sqrt{5}}{2}$}
	\loigiai
	{
		\immini
		{
			Mặt cầu $(S_1)$ có tâm $I(2;1;1)$ và bán kính $R_1=4$.\\
			Mặt cầu $(S_2)$ có tâm $J(2;1;5)$ và bán kính $R_2=2$.\\
			Gọi $A,~B$ lần lượt là hai tiếp điểm của $(S_1),~(S_2)$ với mặt phẳng (P).\\
			Gọi $M$ là giao điểm của $IJ$ với mặt phẳng $(P)$.
			Ta có  
			\[\dfrac{MI}{MJ}=\dfrac{IA}{JB}=2.\]
			
		}
		{\begin{tikzpicture}[line cap=round,line join=round,x=1.0cm,y=1.0cm,>=stealth,scale=1]
			\tkzDefPoints{0/0/A,2/0/B,0/4/I,2/2/J}
			\tkzInterLL(A,B)(I,J)\tkzGetPoint{M}
			\tkzDrawSegments(J,B)
			\tkzDrawPolygon(I,A,M)
			\tkzMarkRightAngles[size=0.2](I,A,B J,B,M)
			\tkzLabelPoints[](A,B,M)
			\tkzLabelPoints[right](I,J)
			\tkzDrawPoints[fill=black](A,B,I,J,M)
			\tkzMarkSegments[mark=|](I,J J,M A,B B,M)
			\end{tikzpicture}}
		\noindent Suy ra $J$ là trung điểm của $IM$, do đó $M(2;1;9)$.\\
		Gọi véc-tơ pháp tuyến của mặt phẳng $(P)$ là $\vec{n}=(a;b;c)~(a^2+b^2+c^2>0)$, khi đó phương trình của mặt phẳng $(P)$ là
		\[a(x-2)+b(y-1)+c(z-9)=0.\]
		Ta có
		\begin{eqnarray*}
			& &\mathrm{d}(I,(P))=4 \\
			& \Leftrightarrow & \dfrac{|8c|}{\sqrt{a^2+b^2+c^2}}=4 \\
			& \Leftrightarrow & \dfrac{|c|}{\sqrt{a^2+b^2+c^2}}=\dfrac{1}{2}\\
			& \Leftrightarrow & a^2+b^2=3c^2\\
			& \Leftrightarrow & \left(\dfrac{a}{c}\right)+\left(\dfrac{b}{c}\right)^2=3.\qquad(1)
		\end{eqnarray*}
		Mặt khác, 
		\[\mathrm{d}(O,(P))=\dfrac{|2a+b+9c|}{\sqrt{a^2+b^2+c^2}}=\dfrac{|2a+b+9c|}{2c}=\dfrac{1}{2}\left|\dfrac{2a}{c}+\dfrac{b}{c}+9\right|.\qquad(2)\]
		Áp dụng bất đẳng thức Bunhiacopxki ta có
		\[\left(\dfrac{2a}{c}+\dfrac{b}{c}\right)^2\le\left(2^2+1^2\right)\left[\left(\dfrac{a}{c}\right)^2+\left(\dfrac{b}{c}\right)^2\right].\qquad (3)\]
		Từ (1) và (3) ta có
		\[\left(\dfrac{2a}{c}+\dfrac{b}{c}\right)^2\le15\Leftrightarrow-\sqrt{15}\le\dfrac{2a}{c}+\dfrac{b}{c}\le\sqrt{15}.\qquad(4)\]
		Từ (2) và (4) suy ra
		\[\dfrac{9-\sqrt{15}}{2}\le\mathrm{d}(O,(P))\le\dfrac{9+\sqrt{15}}{2}.\]
		Vậy khoảng cách lớn nhất từ gốc tọa độ $O$ đến mặt phẳng $(P)$ bằng $\dfrac{9+\sqrt{15}}{2}$.
	}
