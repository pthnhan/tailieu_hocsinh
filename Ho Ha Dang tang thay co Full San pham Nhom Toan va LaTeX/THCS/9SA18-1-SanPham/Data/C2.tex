\section{Góc ở tâm. Liên hệ giữa cung và dây cung}


\begin{bt}%[Bài 16]%[Thầy Hóa]
	Cho đường tròn $(O;R)$; $AB$ và $AD$ là hai dây cung sao cho tia $AO$ nằm giữa hai tia $AB$ và $AD$, $AB=R\sqrt{2}$; $AD=R$. Vẽ dây $BC$ song song với $AD$.
	\begin{enumerate}
		\item Tính số đo các cung $AD$, $AB$;
		\item Tứ giác $ABCD$ là hình gì? Vì sao?
		\item Các tiếp tuyến tại $B$ và $C$ cắt nhau tại $E$. Chứng minh rằng tam giác $EBC$ đều.
	\end{enumerate}

\begin{bt}%[Bài 17]%[Thầy Hóa]
	Trên dây cung $AB$ của đường tròn $(O;R)$ $(AB\ne 2R)$ có hai điểm $C$, $D$ sao cho $AC=CD=DB$. Các tia $OC$, $OD$ cắt cung $AB$ lần lượt ở $E$, $F$. Chứng minh:
	\begin{enumerate}
		\item $\wideparen{AE}=\wideparen{BF}$;
		\item Tứ giác $AEFB$ là hình thang cân;
		\item $\wideparen{AE}<\wideparen{EF}$.
	\end{enumerate}
	\loigiai{
		\allowdisplaybreaks
		\immini
		{
			\begin{enumerate}
				\item Tam giác $OAB$ có $OA=OB=R$ nên $\triangle OAB$ cân tại $O$, suy ra $\widehat{OAC}=\widehat{OBD}$.\\

				\begin{itemize}
					\item $OA=OB=R$;
					\item $\widehat{OAC}=\widehat{OBD}$ (chứng minh trên);
					\item $AC=BD$ (giả thiết).
				\end{itemize}
				$\Rightarrow \triangle OAC=\triangle OBD$ (cạnh - góc - cạnh)\\
				$\Rightarrow \widehat{AOC}=\widehat{BOD}$ (góc tương ứng)\\
				$\Rightarrow \wideparen{AE}=\wideparen{BF}$.
			\end{enumerate}
		}

			Do đó $CD\parallel EF$ (định lý Ta-lét đảo) hay $EF\parallel AB \Rightarrow AEFB$ là hình thang.\\
			Mặt khác $AEFB$ nội tiếp đường tròn $(O)$ nên $AEFB$ là hình thang cân.
			\item[c)] Gọi $I$ là trung điểm của $OD$. Khi đó $CI$ là đường trung bình của $\triangle ADO \Rightarrow CI\parallel OA$.\\

			Mà $\widehat{ICO}=\widehat{COA}$ (hai góc so le trong) nên $\widehat{AOE}<\widehat{EOF}$.\\ Vậy $\wideparen{AE}<\wideparen{EF}$.
		\end{enumerate}
	}
\end{bt}

\section{Góc nội tiếp}

\begin{bt}%[Bài 19]%[YouTuTu]
	Cho tam giác $ABC$ có ba góc nhọn, ở bên ngoài tam giác vẽ hai nửa đường tròn có đường kính $AB$, $AC$. Một đường thẳng $d$ qua $A$ và cắt hai nửa đường tròn lần lượt tại $M$, $N$ (khác $A$).
	\begin{enumEX}{1}
		\item Chứng minh tứ giác $BCNM$ là hình thang vuông.
		\item Chứng minh đường trung trực của $MN$ đi qua một điểm cố định.
		\item Chứng minh rằng $M$, $N$ cách đều một điểm cố định.
		\item Chứng minh trung điểm của $MN$ thuộc một đường cố định.
		\item Giả sử tam giác $ABC$ cân tại $A$. Xác định vị trí của $d$ để diện tích tứ giác $BCMN$ lớn nhất.
		\item Giả sử tam giác $ABC$ vuông tại $A$. Xác định hai điểm $M$, $N$ sao cho chu vi tứ giác $BCMN$ lớn nhất.
	\end{enumEX}
	\loigiai{
		\immini

				\item Vì $AB$ và $AC$ là đường kính nên $\widehat{BMA}=\widehat{CNA}=90^\circ$.\\

				\item Gọi $F$ là trung điểm của $MN$, $K$ là giao điểm của đường trung trực $MN$ với $BC$. Khi đó $FK$ là đường trung bình của hình thang vuông $BCNM$, suy ra $K$ là trung điểm của $BC$ (là một điểm cố định).
			\end{enumerate}
		}

		\begin{enumerate}
			\item[c)] Vì $KF$ là trung trực của $MN$ nên $KM=KN$, hay $M$, $N$ cách đều điểm $K$-cố định.
			\item[d)] Trung điểm $F$ của $MN$ thuộc đường tròn đường kính $AK$-cố định.
			\item[e)] Kẻ $BD$ vuông góc với $CN$. Khi đó $MN=BD\le BC$.\\

\begin{bt}%[Bài 20]%[YouTuTu]
