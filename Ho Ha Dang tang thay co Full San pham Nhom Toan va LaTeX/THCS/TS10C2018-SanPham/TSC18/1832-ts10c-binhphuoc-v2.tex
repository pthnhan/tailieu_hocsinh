	\begin{enumerate}
		\item Rút gọn biểu thức $T=\left(\dfrac{\sqrt{a}+1}{\sqrt{ab}+1}+\dfrac{\sqrt{ab}+\sqrt{a}}{\sqrt{ab}-1}\right):\left(\dfrac{\sqrt{a}+1}{\sqrt{ab}+1}-\dfrac{\sqrt{ab}+\sqrt{a}}{\sqrt{ab}-1}+1\right)$.
		\item Cho $x+\sqrt{3}=2$. Tính giá trị biểu thức:\[H=x^5-3x^4-3x^3+6x^2-20x+2023.\]
	\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
	\item Điều kiện: $\begin{cases}
	a\geq0\\
	b\geq0\\
	ab\ne1
	\end{cases}$\\
	Ta có: \[\dfrac{\sqrt{a}+1}{\sqrt{ab}+1}+\dfrac{\sqrt{ab}+\sqrt{a}}{\sqrt{ab}-1}=\dfrac{2\sqrt{ab}(\sqrt{a}+1)}{ab-1}.\]
	và \[\dfrac{\sqrt{a}+1}{\sqrt{ab}+1}-\dfrac{\sqrt{ab}+\sqrt{a}}{\sqrt{ab}-1}+1=\dfrac{-2(\sqrt{a}+1)}{ab-1}.\]
	Nên \[T=\dfrac{2\sqrt{ab}(\sqrt{a}+1)}{ab-1}:\dfrac{-2(\sqrt{a}+1)}{ab-1}=-\sqrt{ab}.\]
	\item Ta có: $x+\sqrt{3}=2\Leftrightarrow 2-x=\sqrt{3}\Rightarrow (2-x)^2=3\Leftrightarrow 4-4x+x^2=3\Leftrightarrow x^2-4x+1=0$,\\
	$H=(x^5-4x^4+x^3)+(x^4-4x^3+x^2)+5(x^2-4x+1)+2018$.\\
	Suy ra $H=x^3(x^2-4x+1)+x^2(x^2-4x+1)+5(x^2-4x+1)+2018$.\\
	Do $x^2-4x+1=0$ nên $H=2018$.
\end{enumerate}}

\begin{ex}%[Đề TS vào 10 chuyên, sở GD Bình Phước, 2018 - 2019]%[9D4G2]
	Cho Parabol $(P)\colon y=\dfrac{1}{2}x^2$ và đường thẳng $(d)\colon y=(m+1)x-m^2-\dfrac{1}{2}$ ($m$ là tham số).\\
	Với giá trị nào của $m$ thì đường thẳng $(d)$ cắt Parabol $(P)$ tại hai điểm $A(x_1;y_1)$, $B(x_2;y_2)$ sao cho biểu thức $T=y_1+y_2-x_1x_2$ đạt giá trị nhỏ nhất.
	\loigiai{Phương trình hoành độ giao điểm:
	\begin{equation}
		\dfrac{1}{2}x^2=(m+1)x-m^2-\dfrac{1}{2}\Leftrightarrow x^2-2(m+1)x+2m^2+1=0\tag{1}
	\end{equation}
	Để $(d)$ cắt $(P)$ tại hai điểm $A(x_1;y_1)$, $B(x_2;y_2)$ thì phương trình $(1)$ có hai nghiệm.\\
	$\Leftrightarrow \Delta'\geq0\Leftrightarrow (m+1)^2-2m^2-1=2m-m^2\geq0\Leftrightarrow 0\leq m\leq 2$.\\
	Vậy với $0\leq m\leq 2$ thì đường thẳng $(d)$ cắt Parabol $(P)$ tại hai điểm $A(x_1;y_1)$, $B(x_2;y_2)$.\\
	Khi đó theo định lí Vi-ét thì
	\[\begin{cases}
	x_1+x_2=2(m+1)\\
	x_1x_2=2m^2+1
	\end{cases}\]
	Ta có 
	\[y_1=(m+1)x_1-m^2-\dfrac{1}{2}\]
	\[y_2=(m+1)x_2-m^2-\dfrac{1}{2}\]
Do đó
\begin{align*}
T&=y_1+y_2-x_1x_2=(m+1)(x_1+x_2)-2m^2-1-x_1x_2\\
&=2(m+1)^2-4m^2-2=-2m^2+4m=2-2(m-1)^2,\forall m\in[0;2].
\end{align*}
Đặt $t=m-1$. Do $m\in[0;2]\Rightarrow t\in[-1;1]\Rightarrow t^2\in[0;1]$.\\
Nên $T=2-2(m-1)^2=2-2t^2\geq0$.\\
Vậy giá trị nhỏ nhất của $T$ bằng $0$ đạt được khi $t^2=1\Leftrightarrow (m-1)^2=1\Leftrightarrow m=0;\:m=2$.
}
\end{ex}

	&\Leftrightarrow\left[\begin{array}{l}
	x-3=0\\

	\end{array}
	\right.\\
	&\Leftrightarrow \left[
	\begin{array}{l}
	x=3\\

	\end{array}
	\right.
	\end{align*}

VT(*)<\dfrac{7}{2}\\
VP(*)>\dfrac{16}{3}
\end{cases} \left(\forall x\geq\dfrac{7}{3}\right)$. Suy ra phương trình $(*)$ vô nghiệm.\\

\begin{ex}%[Đề TS vào 10 chuyên, sở GD Bình Phước, 2018 - 2019]%[9D0G2]

	(x^2+1)(y^2+1)=10\\(x+y)(xy-1)=3

	\begin{cases}

	\Leftrightarrow
	\begin{cases}
	x^2y^2+x^2+y^2+1=10\\(x+y)(xy-1)=3
	\end{cases}
	\Leftrightarrow
	\begin{cases}
	(x+y)^2+(xy-1)^2=10\\(x+y)(xy-1)=3
	\end{cases}\tag{1}
	\end{equation}

x+y=u\\
xy-1=v
\end{cases}$.\\

\begin{align*}
(1)&\Leftrightarrow\begin{cases}
u^2+v^2=10\\
uv=3
\end{cases}
\Leftrightarrow
\begin{cases}
(u+v)^2-2uv=10\\
uv=3
\end{cases}
\Leftrightarrow
\begin{cases}
(u+v)^2=16\\
uv=3
\end{cases}\\
&\Leftrightarrow \left[
\begin{array}{l}
\begin{cases}
u+v=4\\
uv=3
\end{cases}\\
\begin{cases}
u+v=-4\\
uv=3
\end{cases}
\end{array}
\right.
\Leftrightarrow
\left[
\begin{array}{l}
\begin{cases}
u=1\\v=3
\end{cases}\\
\begin{cases}
u=3\\v=1
\end{cases}\\
\begin{cases}
u=-1\\v=-3
\end{cases}\\
\begin{cases}
u=-3\\v=-1
\end{cases}
\end{array}
\right.
\end{align*}

	u=1\\v=3

	\end{cases} \:(\text{Hệ pt vô nghiệm})$
	\item Với $\begin{cases}
	u=3\\v=1
	\end{cases}\Leftrightarrow\begin{cases}

	x+y=-3\\xy=0
	\end{cases}\Leftrightarrow\begin{cases}
	x=0\\y=-3
	\end{cases}\vee
	\begin{cases}
	x=-3\\y=0
	\end{cases}$
\end{itemize}
Vậy hệ phương trình có các nghiệm là: $(1;2)$, $(2;1)$, $(1;-2)$, $(-2;1)$, $(0;-3)$, $(-3;0)$.}

	\begin{enumerate}[a)]
		\item Tìm tất cả các số nguyên tố $p$ sao cho $16p+1$ là lập phương của số nguyên dương.
		\item Tìm tất cả các bộ số nguyên $(a,b)$ thỏa mãn $3(a^2+b^2)-7(a+b)=-4$.

\loigiai{
\begin{enumerate}[a)]
	\item Vì $16p+1$ là lẻ và lớn hơn $1$ nên có thể đặt $16p+1=(2n+1)^3$, $\forall n\in\mathbb{N}^{*}$.\\
	Ta có: 
	\[16p+1=(2n+1)^3\Leftrightarrow 8p=n(4n^2+6n+3)\]
	Vì $4n^2+6n+3$ là số lẻ lớn hơn $1$ và không phân tích được thành tích của hai số nguyên nên từ trên suy ra $\begin{cases}
	n=8\\
	4n^2+6n+3=p
	\end{cases}$\\
	Từ đó, ta có $p=307$. Thử lại ta thấy thỏa mãn.\\
	Vậy $p=307$ là số nguyên tố duy nhất thỏa mãn yêu cầu.

	\right.
	\]
\end{itemize}
Kết hợp với giả thuyết và nhận xét ở trên, ta có các bộ số $(a,b)$ thỏa mãn là $\{(0,1);(1,0);(2,2)\}$.
\end{enumerate}}
\end{ex}

\begin{ex}%[Đề TS vào 10 chuyên, sở GD Bình Phước, 2018 - 2019]%[9D0G7]
\begin{enumerate}[a)]
\item Cho $x$, $y$ là hai số dương. Chứng minh rằng: $\dfrac{x^2}{y}+\dfrac{y^2}{x}\geq x+y$.
\item Xét các số thực $a$, $b$, $c$ với $b\ne a+c$ sao cho phương trình bậc hai $ax^2+bx+c=0$ có hai nghiệm thực $m$, $n$ thỏa mãn $0\leq m,n\leq 1$. Tìm giá trị lớn nhất và giá trị nhỏ nhất của biểu thức
\[M=\dfrac{(a-b)(2a-c)}{a(a-b+c)}\]
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
	\item Với $x$, $y$ là hai số dương, ta có:
	\[\dfrac{x^2}{y}+\dfrac{y^2}{x}\geq x+y\Leftrightarrow x^3+y^3\geq xy(x+y).\]
	\[\Leftrightarrow (x+y)(x^2-xy+y^2)\geq xy(x+y).\]
	\[x^2-xy+y^2\geq xy\Leftrightarrow x^2-2xy+y^2\geq 0\Leftrightarrow (x-y)^2\geq 0\:(\text{Hiển nhiên}). \]
	Vậy bất đẳng thức được chứng minh. Đẳng thức xảy ra khi và chỉ khi $x=y>0$.
	\item Giả thiết phương trình $ax^2+bx+c=0$ có hai nghiệm $m,n\:(0\leq m\leq 1, 0\leq n\leq 1)$ nên $a\ne0$.\\
	Theo định lí Vi-ét ta có: $m+n=-\dfrac{b}{a}$ và $m\cdot n=\dfrac{c}{a}$.\\
	Từ đó suy ra
	\[
	M=\dfrac{(a-b)(2a-c)}{a(a-b+c)}=\dfrac{\left(1-\dfrac{b}{a}\right)\left(2-\dfrac{c}{a}\right)}{1-\dfrac{b}{a}+\dfrac{c}{a}}=\dfrac{(1+m+n)(2-mn)}{1+m+n+mn}.
	\]
	Vì $2-mn\leq 2$ và $mn\geq0$ nên
	\[M\leq \dfrac{(1+m+n)\cdot2}{1+m+n}=2\]
	Vậy giá trị lớn nhất của $M$ là $2$ đạt được khi $mn=0$ hay $c=0$.\\
	Do $0\leq m\leq 1$, $0\leq n\leq 1$ nên $mn\leq 1$, suy ra:
	\[
	m(n-1)+n(m-1)+(mn-1)\leq 0\Leftrightarrow mn\leq \dfrac{1}{3}(1+m+n).
	\]
	Do đó:
	\[
	M\geq \dfrac{1+m+n}{1+m+n+\dfrac{1}{3}(1+m+n)}=\dfrac{3}{4}
	\]
	Vậy giá trị nhỏ nhất của $M$ là $\dfrac{3}{4}$ đạt được khi $m=n=1$ hay $a+b+c=0$ và $a=c$.
\end{enumerate}}
\end{ex}
