\begin{bt}%[Nguyễn Cao Cường]%[EX-TS10-ToanTT-2019]%[7D1T9]%[6D3T9]
	Bạn Bình gửi ngân hàng $3{,}5$ triệu theo định kì $3$ năm và sau $3$ năm bạn nhận được $4{,}235$ triệu.
	\begin{enumerate}
		\item Tính lãi suất ngân hàng.
		\dapso{$21\%$}
		\item Với lãi suất $7\%$/năm  thì đến năm $18$ tuổi bạn nhận được tổng cộng bao nhiêu tiền.
		Biết năm nay bạn Bình vừa tròn $15$ tuổi.
		\dapso{$4235000\;\text{đồng}$}
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item \textbf{Cách 1.} Số tiền lãi mà bạn Bình nhận được sau $3$ năm là  $4235000-3500000=735000$ đồng.\\
			Vậy lãi suất ngân hàng là $735000:3500000\cdot 100\%=21\%.$\\
			\textbf{Cách 2.}
			Gọi $x$ là lãi suất cần tìm. Theo đề bài ta có phương trình
			\begin{eqnarray*}
				&&	3500000(1+x)=4235000 \\
				&\Leftrightarrow& 1+x=\dfrac{4235000}{3500000}\\
				&\Leftrightarrow& 1+x=1{,}21\\
				&\Leftrightarrow& x=0{,}21=21\%.
			\end{eqnarray*}
			
			\item Lãi suất $7\%$ một năm, vậy một kì hạn $3$ năm có lãi suất là $7\%\cdot 3=21\%$.\\
			Đến năm $18$ tuổi bạn đã gửi được $3$ năm, tức $1$ kì hạn, khi đó số tiền lãi bạn nhận được là $3500000\cdot 21\%=735000$ đồng.\\
			Vậy đến năm $18$ tuổi, bạn nhận cả vốn lẫn lãi là $3500000+735000=4235000$ đồng.
			
		\end{enumerate}
	}
\end{bt}

	suất ngân hàng là $8\%$ một năm. Hỏi bạn gửi bao nhiêu tiền?
	\dapso{$4000000$ đồng}	
	\loigiai{
		\textbf{Cách 1.} Lãi suất $8\%$ một năm, vậy kì hạn $3$ năm có lãi suất là $8\%\cdot 3=24\%$.\\
		Vậy sau $3$ năm số tiền lãi và vốn là $124\%$ tương ứng với $4960000$ đồng.\\
		Vậy số tiền ban đầu ($100\%$) là $\dfrac{100\%\cdot 4960000}{124\%}=4000000$ đồng.\\
		\textbf{Cách 2.}
		Gọi $x$ là số tiền bạn Tân gửi, lãi suất $8\%$ một năm, vậy kì hạn $3$ năm có lãi suất là $8\%\cdot 3=24\%$. Khi đó số tiền lãi bạn nhận được là $x\cdot 24\%$.\\
		Vậy sau $3$ năm bạn nhận được cả vốn lẫn lãi là $x+x\cdot 24\%=x(1+24\%)$. Theo đề bài ta có vốn lẫn lãi là $4960000$ đồng nên
		\begin{eqnarray*}
			&&x(1+24\%)=4960000 \\
			&\Leftrightarrow& x=\dfrac{4960000}{1+24\%}\\
			&\Leftrightarrow& x=4000000.
		\end{eqnarray*}
		Vậy số tiền Tân gửi là $4000000$ đồng.
	}
\end{bt}

			Khi đó số tiền lãi sinh ra là $x\cdot 35\%$.\\

			\begin{eqnarray*}
				&&x(1+35\%)=6750000 \\
				&	\Leftrightarrow & x=\dfrac{6750000}{1+35\%}\\
				&\Leftrightarrow& x=5000000.
			\end{eqnarray*}
			Vậy số tiền bạn Linh gửi là $5000000$ đồng.
			\item \textbf{Cách 1.} Số tiền lãi bạn Linh nhận được sau $1$ kì ($5$ năm) là $6050000-5000000=1050000$ đồng.\\
			Phần trăm lãi suất Linh nhận được là $1050000:5000000\cdot 100\%=21\%$.\\
			Vậy số năm trong $1$ kì Linh đã gửi là $21\%:7\%=3$ năm.\\	
			\textbf{Cách 2.} Gọi $y$ là số năm của một định kì, khi đó số lãi sinh ra là $5000000\cdot y\cdot 7\%$.\\
			Vậy sau định kì cả vốn lẫn lãi là $5000000+5000000\cdot y\cdot 7\%=5000000(1+y\cdot 7\%)$. Theo đề bài, ta có
			\begin{eqnarray*}
				&&	5000000(1+y\cdot 7\%)=6050000\\
				& \Leftrightarrow &1+y\cdot 7\%=\dfrac{6050000}{5000000} \\
				& \Leftrightarrow &1+y\cdot 7\%=1{,}21\\
				&\Leftrightarrow& y=3.
			\end{eqnarray*}
			Vậy sau định kì $3$ năm thì Bạn linh nhận được số tiền là $6050000$.
		\end{enumerate}
	}
	
\end{bt}

\begin{bt}%[Võ Tấn Đạt]%[EX-TS10-ToanTT-2019]%[8D1B3]

$0{,}5$ (giờ).

\[S=50\cdot 2 = 100 \text{km }.\]

\[V_{\text{thường}} = 50 \cdot 120\% = 60 \text{ km/h}.\]

\[V_{\text{cao tốc}} = 60 \cdot 125\% = 75 \text{  km/h}.\]

\[T= \dfrac{50}{60} + \dfrac{50}{75} = 1{,}5 \text{ giờ }.\]
Vậy xe đi nhanh hơn dự định $2-1{,}5=0{,}5$  giờ.
}
\end{bt}

	Ông Hùng đi mua một chiếc tivi ở siêu thị điện máy. Nhân dịp $30/4$ nên siêu thị điện máy giảm giá $15\%$. Vì ông có thẻ vàng của siêu thị điện máy nên được giảm tiếp $20\%$ giá của chiếc tivi sau khi đã được giảm $15\%$, vì vậy ông Hùng chỉ phải trả $13.328.000$ đồng. Hỏi giá bán ban đầu của chiếc tivi là bao nhiêu?
	\dapso{$19{,}6$ triệu đồng}
	\loigiai{
		Gọi $x$ (đồng) là giá bán ban đầu của chiếc tivi (điều kiện $x>0$).\\
		Nhân dịp $30/4$ siêu thị điện máy giảm giá $15\%$ nên giá chiếc tivi còn $x-0{,}15x=0{,}85x$ đồng.\\
		Ông Hùng có thẻ vàng của siêu thị nên được giảm tiếp $20\%$ giá của chiếc tivi sau khi đã được giảm $15\%$ nên giá của chiếc tivi lúc này còn là $0{,}85x-0{,}85x\cdot 0{,}2=0{,}68x$ đồng.\\
		Theo đề, ta có $0{,}68x=13.328.000 \Leftrightarrow x=19.600.000$ đồng.\\
		Vậy ban đầu giá chiếc tivi là $19{,}6$ triệu đồng.
	}
\end{bt}

\begin{bt}%[Tăng Lâm Tường Vinh]%[EX-TS10-ToanTT-2019]%[8D4T2]
	Một xe tải có chiều rộng $2{,}4$ m và chiều cao $2{,}5$ m muốn đi qua một cái cổng có hình parabol. Biết khoảng cách giữa hai chân cổng là $4$ m và khoảng cách từ đỉnh cổng (đỉnh parabol) tới mỗi chân cổng là $2\sqrt{5}$ m (bỏ qua độ dày của cổng).
	\begin{enumerate}
		\item Trong mặt phẳng tọa độ $Oxy$ gọi parabol $(P)$: $y=ax^2$ với $a<0$ hình biểu diễn cổng mà xe tải muốn đi qua. Chứng minh $a=-1$.
		\item Hỏi xe tải có thể đi qua cổng có được không? Tại sao?
	\end{enumerate}
	\dapso{Qua được cổng.} 
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=1, font=\footnotesize, line join=round, line cap=round, >=stealth]
			\def\xmin{-4}\def\xmax{4}\def\ymin{-4}\def\ymax{3}
			\draw[thick,->] (\xmin-0.6,0)--(\xmax+0.6,0) node[below] {\footnotesize $x$};
			\draw[thick,->] (0,\ymin-0.6)--(0,\ymax+0.6) node[right] {\footnotesize $y$};
			
			\foreach \x in {\xmin,...,\xmax}{\ifthenelse{\not \x=0}{
					\draw (\x,0.1)--(\x,-0.1) node [below] {\footnotesize $\x$};}{}}
			\foreach \y in {-3,...,\ymax} {
				\ifthenelse{\not \y=0}{\draw (0.1,\y)--(-0.1,\y) node [left] {\footnotesize $\y$};}{} }
			\draw (-0.1,-4) node[below left]{$-4$};
			
			\draw (0,0) node [above left]{$O$};
			
			\clip (\xmin-0.5,\ymin-0.5) rectangle (\xmax+0.5,\ymax+0.5);
			\draw[smooth,domain=-2.3:2.3,samples=50] plot (\x,{(-(\x)^2)});
			\draw (-4,-4) -- (4,-4);
			\draw (-4,-1.5) -- (4,-1.5);
			\filldraw (-2,-4) circle (1pt) node[above left]{$A$}; 
			\filldraw (2,-4) circle (1pt) node[above right]{$B$}; 
			\filldraw (0,-4) circle (1pt) node[above right]{$H$}; 
			\def\a{sqrt{1.5}};
			\filldraw (-\a,-1.5) circle (1pt) node[above left]{$C$}; 
			\filldraw (\a,-1.5) circle (1pt) node[above right]{$D$}; 
			\filldraw (0,-1.5) circle (1pt) node[above right]{$M$};
			\draw (0,0) -- (-2,-4);	
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Đỉnh cổng là đỉnh của parabol $y=ax^2$ $(a<0)$, đỉnh cổng là $O(0;0)$.\\
			Gọi hai chân cổng là $A$, $B$. $AB$ cắt $Oy$ tại $H$. Ta có $AH=HB=\dfrac{AB}{2}$, $OA=OB=2\sqrt{5}$ ($A$, $B$, $H$ nằm dưới trục hoành).\\
			$\Delta HOA$ vuông tại $H$, suy ra $OH^2+AH^2=OA^2$ (định lý Py-ta-go) $$\Rightarrow OH^2=\left(2\sqrt{5}\right)^2-2^2=16\Rightarrow OH=4.$$
			Do đó $H(0;-4)$, nên $A(-2;4)$, $B(2;-4)$. $A\in(P)$ nên $-4=a(-2)^2\Leftrightarrow a=-1.$
			\item Gọi giao điểm của đường thẳng đi qua điểm cao nhất của xe tải, song song với trục hoành với $(P)$ là $C$, $D$; $CD$ cắt $Oy$ tại $M$.\\
			Phương trình đường thẳng $CD$ là $y=1{,}5$.\\
			Phương trình hoành độ giao điểm của $(P)$ và $CD$ là
			$$-x^2=1{,}5\Leftrightarrow x=\pm \dfrac{\sqrt{6}}{2}.$$
			Do đó $CD=\left|\dfrac{\sqrt{6}}{2}-\left(\dfrac{-\sqrt{6}}{2}\right)\right| = \sqrt{6}$. Mà $\sqrt{6}>2{,}4$. Tại độ cao $2{,}5$ m thì chiều rộng của cổng là $\sqrt{6}$ lớn hơn $2{,}4$ m là chiều rộng của xe tải. Như vậy xe tải có thể qua cổng được.
		\end{enumerate}
	}
\end{bt}
