\begin{ex}%[Đề thi 10TS19, Phú Yên]%[9D3B2]

	{$\begin{cases} y=2x-3 \\ y=x+5 \end{cases}$}
	{\True $\begin{cases} y=2x-3 \\ y=2x+1 \end{cases}$}
	{$\begin{cases} y=2x-3 \\ 2y=4x-6 \end{cases}$}

		Xét trong các đáp án ta thấy hệ phương trình có dạng $\begin{cases} y=a_1x+b_1 \quad (d_1) \\ y=a_2x+b_2 \quad (d_2) \end{cases}$.\\

		Đề bài yêu cầu hệ phương trình vô nghiệm nên $(d_1) \parallel (d_2)\Leftrightarrow \begin{cases} a_1=a_2 \\ b_1\ne b_2 \end{cases}$. Nhìn vào các đáp án, chỉ có đáp án $\begin{cases} y=2x-3 \\ y=2x+1 \end{cases}$ thỏa.
	}
\end{ex}

\begin{ex}%[Đề thi 10TS19, Phú Yên]%[9D1B1]%[9D4B7]
	\begin{enumerate}
		\item So sánh $5$ và $2\sqrt{6}$.
		\item Giải phương trình $x^4-4x^2-5=0$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $5=\sqrt{25}$ và $2\sqrt{6}=\sqrt{2^2\cdot 6}=\sqrt{24}$.\\
			Vì $25>24$ nên $\sqrt{25} > \sqrt{24}$. Suy ra $5 > 2\sqrt{6}$.
			\item $x^4-4x^2-5=0$.\\
			Đặt $t=x^2$ ($t\ge 0$), phương trình trở thành\\
			$t^2-4t-5=0 \Leftrightarrow (t-5)(t+1)=0 \Leftrightarrow t=5 \text{ (nhận)} \text{ hay } t=-1 \text{ (loại)}$.\\
			Với $t=5$ thì $x^2=5 \Leftrightarrow x=-\sqrt{5} \text{ hay } x=\sqrt{5}$.\\
			Vậy tập nghiệm của phương trình là $S=\left\lbrace -\sqrt{5}; \sqrt{5} \right\rbrace$.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi 10TS19, Phú Yên]%[9D4B4]%[9D4B6]
	Cho phương trình $4x^2-2(m+1)x+m^2=0$ ($m$ là tham số).
	\begin{enumerate}
		\item Với giá trị nào của $m$ thì phương trình có nghiệm kép?
		\item Trong trường hợp phương trình có nghiệm, dùng hệ thức Vi-ét, hãy tính tổng các bình phương hai nghiệm của phương trình.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $\Delta ' = (m+1)^2-4m^2=-3m^2+2m+1$.\\
			Phương trình có nghiệm kép $\Leftrightarrow \Delta '=0 \Leftrightarrow -3m^2+2m+1=0 \Leftrightarrow m=1 \text{ hoặc } m=-\dfrac{1}{3}$.
			\item Phương trình có nghiệm $\Leftrightarrow \Delta ' \ge 0 \Leftrightarrow -3m^2+2m+1 \ge 0$.\\
			Theo hệ thức Vi-ét ta có $\begin{cases} x_1 + x_2 = \dfrac{m+1}{2} \\ x_1x_2=\dfrac{m^2}{4} \end{cases}$.\\
			Tổng bình phương các nghiệm của phương trình là\\
			$S=x_1^2+x_2^2=(x_1+x_2)^2-2x_1x_2=\dfrac{(m+1)^2}{4}-\dfrac{2m^2}{4}=\dfrac{-m^2+2m+1}{4}$.\\
			Vậy $S=\dfrac{-m^2+2m+1}{4}$ với điều kiện $-3m^2+2m+1 \ge 0$.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi 10TS19, Phú Yên]%[9D3K6]
	\textit{Giải bài toán bằng cách lập phương trình hoặc hệ phương trình.}\\
	Nếu mở cả hai vòi nước chảy vào một bể cạn thì sau $3$ giờ bể đầy nước. Nếu chỉ mở một vòi thì vòi thứ nhất làm đầy bể nhanh hơn vòi thứ hai là $2$ giờ $30$ phút. Hỏi nếu mở riêng từng vòi thì mỗi vòi chảy bao lâu đầy bể?
	\loigiai{
		Gọi thời gian vòi thứ nhất chảy một mình đầy bể là $x$ (giờ) ($x>0$).\\
		Gọi thời gian vòi thứ hai chảy một mình đầy bể là $y$ (giờ) ($y>0$).\\
		Khi đó, mỗi giờ vòi thứ nhất chảy được $\dfrac{1}{x}$ bể và vòi thứ hai chảy được $\dfrac{1}{y}$ bể.\\
		Vì nếu mở cả hai vòi nước chảy vào một bể cạn thì sau $3$ giờ bể đầy nên mỗi giờ cả hai vòi chảy được $\dfrac{1}{3}$ bể. Do đó ta có phương trình $\dfrac{1}{x}+\dfrac{1}{y}=\dfrac{1}{3}\quad (1)$.\\
		Vòi thứ nhất làm đầy bể nhanh hơn vòi thứ hai là $2$ giờ $30$ phút $=\dfrac{5}{2}$ (giờ) nên ta có phương trình $x+\dfrac{5}{2}=y \quad (2)$.\\
		Thay $(2)$ vào $(1)$ ta có $\dfrac{1}{x}+\dfrac{1}{x+\dfrac{5}{2}}=\dfrac{1}{3} \Leftrightarrow x=5 \text{ (nhận) hay } x=-\dfrac{3}{2} \text{ (loại)}$.\\ Suy ra $y=5+\dfrac{5}{2}=7{,}5$ (nhận).\\
		Vậy thời gian vòi thứ nhất chảy một mình đầy bể là $5$ giờ, thời gian vòi thứ hai chảy một mình đầy bể là $7{,}5$ giờ.
	}
\end{ex}

\begin{ex}%[Đề thi 10TS19, Phú Yên]%[9H3B7]%[8H3B7]%[9H3K6]
	Cho đường tròn $(O;R)$ đường kính $AB$. Gọi $d$ là tiếp tuyến của đường tròn tại $A$. $C$ là điểm chuyển động trên đường thẳng $d$. $BC$ cắt $(O)$ tại $D$ ($D \ne B$). Gọi $E$ là trung điểm của $BD$.
	\begin{enumerate}
		\item Chứng minh rằng $OACE$ là tứ giác nội tiếp.
		\item Chứng minh rằng $BE\cdot BC = 2R^2$.
		\item Tìm tập hợp các tâm $I$ của đường tròn ngoại tiếp tam giác $AEC$.
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
		\item Chứng minh rằng $OACE$ là tứ giác nội tiếp.	
		\immini{
			Vì $E$ là trung điểm của $BD$ nên $OE \perp BD$ (quan hệ đường kính và dây cung).\\
			Xét tứ giác $OACE$ có $\widehat{OAC}+\widehat{OEC}=90^\circ + 90^\circ = 180^\circ \\ \Rightarrow$ Tứ giác $OACE$ nội tiếp (tứ giác có tổng hai góc đối bằng $180^\circ$).
			
			\item Chứng minh rằng $BE\cdot BC = 2R^2$.\\
			Xét $\triangle BOE$ và $\triangle BCA$ có
			$\begin{cases} \widehat{ABC} \text{ chung}\\\widehat{BEO}=\widehat{BAC}=90^\circ \end{cases}$.\\
			$\Rightarrow \triangle BOE \backsim \triangle BCA$ (g-g)\\
			$\Rightarrow \dfrac{BE}{BA}=\dfrac{BO}{BC} \Rightarrow BE\cdot BC=BA\cdot BO=2RR=2R^2$.
		}{
			\begin{tikzpicture}[scale=0.7]
			\tkzInit[ymin=-2,ymax=5,xmin=-3,xmax=3]
			\tkzClip
			\tkzDefPoints{-2/0/A, 2/0/B, -2/3/C}
			\tkzDrawCircle[diameter](A,B)\tkzGetPoint{O}
			\tkzDefLine[orthogonal=through A](A,B) \tkzGetPoint{d}
			\tkzDrawLine[end = $d$](A,d)
			\tkzInterLC(C,B)(O,A)\tkzGetFirstPoint{D}
			\tkzDefMidPoint(D,B)\tkzGetPoint{E}
			\tkzDrawPoints[size = 5pt](A,B,O,D,C,E)
			\tkzLabelPoints[above](D,E)
			\tkzLabelPoints[below right](B,O)
			\tkzLabelPoints[below left](A,C)
			\tkzDrawSegments(A,B C,B O,E A,E)
			\tkzMarkSegments[mark=s||](D,E E,B)
			\tkzMarkRightAngles(O,A,d)
			\end{tikzpicture}
		}
		
		\item Tìm tập hợp các tâm $I$ của đường tròn ngoại tiếp tam giác $AEC$.\\
		Ta có tứ giác $OACE$ nội tiếp $\Rightarrow$ đường tròn ngoại tiếp tam giác $ACE$ chính là đường tròn ngoại tiếp tứ giác $OACE$.\\
		$\Rightarrow$ Tâm $I$ thuộc đường trung trực của $OA$.\\
		Mà $OA$ cố định nên trung trực của $OA$ cố định.\\
		Suy ra khi $C$ di chuyển trên đường thẳng $d$ thì tâm $I$ của đường tròn ngoại tiếp tam giác $ACE$ di chuyển trên trung trực của $OA$.\\
		Vậy tập hợp các điểm $I$ cần tìm là đường trung trực của $OA$.
	\end{enumerate}
