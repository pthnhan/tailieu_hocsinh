\begin{bt}%[Đề HK1, Lê Hồng Phong, Hồ Chí Minh 17-18]%[Hoàng Hải, 11EX-HK1]%[1D1B2-1]%[1D1B3-3]
Giải các phương trình sau
\begin{enumerate}
	\item $ \sin 2x = \sqrt{2}\sin x $.
	\item $ \sqrt{3}\cot x = 8\cos 2x \cos x - 1 $.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Phương trình tương đương
	\begin{eqnarray*}
		& & 2\sin x \cos x - \sqrt{2}\sin x = 0 \\
		& \Leftrightarrow & \sin x(2\cos x-\sqrt{2}) = 0 \\
		& \Leftrightarrow & \hoac{&\sin x = 0 \\ &\cos x = \dfrac{\sqrt{2}}{2}} \\
		& \Leftrightarrow & \hoac{&x = k\pi \\ &x = \pm \dfrac{\pi}{4}+k2\pi} \quad(k\in\mathbb{Z}).
	\end{eqnarray*}
	Vậy nghiệm của phương trình là $ \hoac{&x = k\pi \\ &x = \pm \dfrac{\pi}{4}+k2\pi} \quad(k\in\mathbb{Z}) $.
	\item Điều kiện $ \sin x \ne 0 $, nhân hai vế phương trình cho $ \sin x $ ta có
	\begin{eqnarray*}
		& & \sqrt{3}\cos x = 2\sin 4x - \sin x \\
		& \Leftrightarrow & \sqrt{3}\cos x + \sin x = 2\sin 4x \\
		& \Leftrightarrow & \dfrac{\sqrt{3}}{2} \cos x + \dfrac{1}{2}\sin x = \sin 4x \\
		& \Leftrightarrow & \sin\dfrac{\pi}{3}\cos x + \cos\dfrac{\pi}{3}\sin x = \sin4x \\
		& \Leftrightarrow & \sin \left( \dfrac{\pi}{3}+x \right) = \sin 4x \\
		& \Leftrightarrow & \hoac{&\dfrac{\pi}{3}+x = 4x + k2\pi \\ &\dfrac{\pi}{3}+x = \pi - 4x + k2\pi} \quad(k\in\mathbb{Z}) \\
		& \Leftrightarrow & \hoac{&x=\dfrac{\pi}{9}-k\dfrac{2\pi}{3}\quad\text{(nhận)} \\ &x=\dfrac{2\pi}{15}+k\dfrac{2\pi}{5}\quad\text{(nhận).}}
	\end{eqnarray*}
	Vậy nghiệm của phương trình là $ \hoac{&x=\dfrac{\pi}{9}+k\dfrac{2\pi}{3} \\ &x=\dfrac{2\pi}{15}+k\dfrac{2\pi}{5}} \quad(k\in\mathbb{Z}) $.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Hồng Phong, Hồ Chí Minh 17-18]%[Hoàng Hải, 11EX-HK1]%[1D2B2-2]%[1D2B5-2]
\begin{enumerate}
	\item Có bao nhiêu số tự nhiên gồm 7 chữ số khác nhau trong đó có đúng 4 chữ số chẵn.
	\item Để chuẩn bị cho lễ kỷ niệm 90 năm Petrus Ký - Lê Hồng Phong, nhà trường cần lập một đội tuyển tình nguyện gồm $ 10 $ em học sinh qua đơn đăng ký. Qua đăng ký có 20 em học sinh muốn tham gia đội tình nguyện trong đó có $ 15 $ em là học sinh giỏi. Để đảm bảo công bằng nhà trường quyết định chọn ngẫu nhiên $ 10 $ em trong $ 20 $ em đăng ký nói trên. Tính xác suất để trong đội hình tình nguyện số học sinh giỏi chiếm $ 80\% $. 
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta xét các trường hợp
	\begin{itemize}
		\item Số các số tự nhiên gồm $ 7 $ chữ số khác nhau (bao gồm cả chữ số $ 0 $ đứng đầu), trong đó có đúng $ 4 $ chữ số chẵn là $ \mathrm{C}_5^4 \cdot \mathrm{C}_5^3 \cdot 7! $
		\item Số các số tự nhiên gồm $ 7 $ chữ số khác nhau mà chữ số $ 0 $ đứng đầu, trong đó có đúng $ 4 $ chữ số chẵn là $ \mathrm{C}_4^3 \cdot \mathrm{C}_5^3 \cdot 6! $
	\end{itemize}
	Vậy số các số tự nhiên thỏa đề bài là $ \mathrm{C}_5^4 \cdot \mathrm{C}_5^3 \cdot 7! - \mathrm{C}_4^3 \cdot \mathrm{C}_5^3 \cdot 6! = 223200 $ (số).
	
	\item Ta có
	\begin{itemize}
		\item Số cách chọn một đội hình tình nguyện gồm $ 10 $ em trong $ 20 $ đơn đăng ký là $ \mathrm{C}_{20}^{10} $.
		\item Để chiếm $ 80\% $ đội hình tình nguyện, thì số học sinh giỏi cần là $ 80\% \cdot 10 = 8 $ (em). \\
		Vậy số cách chọn một đội hình tình nguyện mà số học sinh giỏi chiếm $ 80\% $ là $ \mathrm{C}_{15}^{8} \cdot \mathrm{C}_5^2 $.
	\end{itemize}
	Xác suất để trong đội hình tình nguyện số học sinh giỏi chiếm $ 80\% $ là
	$$ \dfrac{\mathrm{C}_{15}^{8} \cdot \mathrm{C}_5^2}{\mathrm{C}_{20}^{10}} =  \dfrac{225}{646} \approx 0,35.$$
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Lê Hồng Phong, Hồ Chí Minh 17-18]%[Hoàng Hải, 11EX-HK1]%[1D2B3-2]
Biết $ n $ là số nguyên dương thỏa $ 3\mathrm{C}_{n+1}^2 - 4\mathrm{A}_n^2 = -22n $. Hãy tìm số hạng chứa $ x^2 $ trong khai triển của $ P(x) = \left( 2x^2 - \dfrac{1}{x^3} \right)^n. $	
\loigiai{
	Ta có
	\begin{eqnarray*}
		& & 3\mathrm{C}_{n+1}^2 - 4\mathrm{A}_n^2 = -22n \quad(n\ge 2)\\
		& \Leftrightarrow & 3\mathrm{C}_n^2 + 3\mathrm{C}_n^1 -4\cdot2!\mathrm{C}_n^2 = -22n \\
		& \Leftrightarrow & 25n - 5\mathrm{C}_n^2 = 0 \\
		& \Leftrightarrow & 25n - 5\cdot\dfrac{n!}{2!(n-2)!} = 0 \\
		& \Leftrightarrow & 25n - \dfrac{5}{2}n(n-1) = 0 \\
		& \Leftrightarrow & 5n\left( 5-\dfrac{1}{2}n +\dfrac{1}{2} \right) = 0 \\
		& \Leftrightarrow & \hoac{&n=0 \qquad\text{(loại)} \\ &n=11 \quad\text{(nhận).}}
	\end{eqnarray*}
	Số hạng tổng quát của $ P(x) $ là $ \mathrm{T}_{k+1} = \mathrm{C}_{11}^{k} (2x^2)^{11-k} \left( -\dfrac{1}{x^3} \right)^k = \mathrm{C}_{11}^{k} 2^{11-k} (-1)^k x^{22-5k} $.\\
	Số hạng chứa $ x^2 $ ứng với $ 22-5k = 2 \Leftrightarrow k=4 $, vậy số hạng đó là
	$ \mathrm{C}_{11}^{4} 2^7 x^2 $.
}\end{bt}

\begin{bt}%[Đề HK1, Lê Hồng Phong, Hồ Chí Minh 17-18]%[Hoàng Hải, 11EX-HK1]%[1D3T1-1]
Trong phòng thí nghiệm có một loài vi khuẩn có quy luật sinh sản như nhau: cứ một giờ thì số vi khuẩn nhân đôi và tự chết đi 2 con. Theo quy  luật đó hỏi sau 24 giờ thì số vi khuẩn trong phòng thí nghiệm là bao nhiêu con, biết rằng ban đầu cho phòng thí nghiệm có 10 con vi khuẩn.
\loigiai{
	Gọi $ T_n $ là số vi khuẩn trong phòng thí nghiệm sau $ n $ giờ, đặt $ a=2 $. Ta có
	\begin{eqnarray*}
		 T_0 & = & 10 \\
		 T_1 & = & 2T_0-a \\
		 T_2 & = & 2T_1-a = 4T_0-3a\\
		 T_3 & = & 2T_2-a = 8T_0-7a\\
		 &&\cdots   \\
		 T_n & = &2T_{n-1}-a.
	\end{eqnarray*}
	Ta dự đoán $ T_n = 2^nT_0-(2^n-1)a  $, nếu điều này đúng thì $ T_{n+1} = 2^{n+1}T_0-(2^{n+1}-1)a $. \\
	Thật vậy, ta có 
	\begin{eqnarray*}
		T_{n+1} &=& 2T_n-a \\
				&=& 2[2^nT_0-(2^n-1)a]-a \\
				&=& 2^{n+1}T_0 - 2^{n+1}a +2a - a \\
				&=& 2^{n+1}T_0 -(2^{n+1}-1)a.
	\end{eqnarray*}
	Vậy số vi khuẩn sinh ra sau $ n $ giờ là $ T_n = 10\cdot 2^n - 2(2^n-1). $ \\
	Sau 24 giờ thì số vi khuẩn trong phòng thì nghiệm là $ T_{24} = 10\cdot 2^{24}-2(2^{24}-1)= 8\cdot 2^{24} + 2. $
}
\end{bt}
	
\begin{bt}%[Đề HK1, Lê Hồng Phong, Hồ Chí Minh 17-18]%[Hoàng Hải, 11EX-HK1]%[1H2B1-2]%[1H2B3-2]%[1H2K1-3]
Cho hình chóp $ S.ABCD $ có $ ABCD $ là hình bình hành tâm $ O $. Lấy $ M $ trên cạnh $ SA $ sao cho $ MA=2MS $ và $ N $ trên cạnh $ BC $ sao cho $ NB=2NC $.
\begin{enumerate}
	\item Tìm giao tuyến hai mặt phẳng $ (SAD) $ và $ (MBN) $.
	\item Chứng minh: $ MN \parallel (SCD) $.
	\item Tìm giao điểm $ P $ của mặt phẳng $ (MNO) $ với $ SB $. Tính tỉ số $ \dfrac{S_{\triangle SCP}}{S_{\triangle SBC}} $.
\end{enumerate}
\loigiai{
\begin{center}
	\begin{tikzpicture}[scale=1.5]

\begin{enumerate}
	\item Ta có 
	\begin{itemize}
		\item $ \heva{&M\in SA \subset (SAD) \\ &M\in (MBN)} 
		\Rightarrow M\in (SAB) \cap (MBN)  $.
		\item $ \heva{&AD \subset (SAD) \\ &BC \subset (MBN) \\ &AD \parallel BC} 
		\Rightarrow  (SAD) \cap (MBN) =MK $ với $ K\in SD$ sao cho $MK \parallel AD \parallel BC $.
	\end{itemize}

	\item Ta có 
	\begin{eqnarray*}
		&&\heva{&\text{Trong $ \triangle SAD $ ta có $ MK \parallel AD \Rightarrow 	 
			     \dfrac{MK}{AD} = \dfrac{SM}{SA} = \dfrac{1}{3} $} \\
		        &NB=2NC \Rightarrow \dfrac{CN}{CB} = \dfrac{1}{3}} \\
	    & \Rightarrow & \dfrac{MK}{AD} = \dfrac{CN}{CB} \Rightarrow MK = NC \quad\text{($AD=CB,ABCD$ là hình bình hành)}.
	\end{eqnarray*}
	Mặt khác ta có $ MK \parallel NC $ do đó $ MKCN $ là hình bình hành 
	$ \Rightarrow MN \parallel CK $.\\
	Vậy ta có 
	$ \heva{&MN \parallel CK \\ &CK \subset (SCD) \\ &MN \not\subset (SCD)} \Rightarrow MN \parallel (SCD) $. 
	
	\item 
	\textbf{Cách 1.} Dùng định lý Ta-lét. \\
	Trong $ (ABCD) $ kẻ $ NO $ cắt $ AB $	tại $ Q $, kẻ $ AE $ song song với $ BC $ cắt $ QC $ tại $ E $, khi đó $ E\in AD $. Áp dụng định lý Ta-lét ta có
	\begin{eqnarray*}
		& \dfrac{QA}{QB} = \dfrac{AE}{BN}

	\begin{eqnarray*}
		& & \dfrac{QA}{QB} = \dfrac{AF}{BP}

						   = \dfrac{1}{k} \cdot \dfrac{AM}{SM} 
						   \quad\text{(với $BP=kSP$)} \\
		& \Rightarrow & \dfrac{1}{2} = \dfrac{1}{k} \cdot \dfrac{2}{1} 
		\Rightarrow k=4.
	\end{eqnarray*}
	Vậy	 $ \dfrac{S_{\triangle SCP}}{S_{\triangle SBC}} = \dfrac{SP}{SB} = \dfrac{1}{5} $.

	\begin{eqnarray*}
		& & \dfrac{QA}{QB} \cdot \dfrac{NB}{NC} \cdot \dfrac{OC}{OA} = 1 \\
		& \Leftrightarrow & \dfrac{QA}{QB} \cdot \dfrac{2}{1} \cdot \dfrac{1}{1} = 1 \\
		& \Leftrightarrow & \dfrac{QA}{QB} = \dfrac{1}{2}.
	\end{eqnarray*}

	\begin{eqnarray*}
		& & \dfrac{QA}{QB} \cdot \dfrac{PB}{PS} \cdot \dfrac{MS}{MA} = 1 \\
		& \Leftrightarrow & \dfrac{1}{2} \cdot \dfrac{PB}{PS} \cdot \dfrac{1}{2} = 1 \\
		& \Leftrightarrow & \dfrac{PB}{PS} = 4.
	\end{eqnarray*}
	Vậy	 $ \dfrac{S_{\triangle SCP}}{S_{\triangle SBC}} = \dfrac{SP}{SB} = \dfrac{1}{5} $.
\end{enumerate}
