\begin{ex}%[Thi thử L2, Sở Bắc Giang, 2018]%[2D1B4-1]%[Nguyễn Tiến Thùy, dự án 12EX-10]
Đồ thị hàm số nào sau đây có tiệm cận ngang?
\choice
{\True $y=\dfrac{x}{x^2+1}$}
{$y=\dfrac{x^2}{x+1}$}
{$y=\dfrac{x^2-3x+2}{x-1}$}
{$y=\dfrac{\sqrt{4-x^2}}{1+x}$}
\loigiai{
Ta có 
\begin{itemize}
\item $\lim\limits_{x\rightarrow\pm\infty}\dfrac{x}{x^2+1}=\lim\limits_{x\rightarrow\pm\infty}\dfrac{\dfrac{1}{x}}{1+\dfrac{1}{x^2}}=0$.\\
Từ đó suy ra đồ thị hàm số $y=\dfrac{x}{x^2+1}$ có tiệm cận ngang.
\item $\lim\limits_{x\rightarrow\pm\infty}\dfrac{x^2}{x+1}=\lim\limits_{x\rightarrow\pm\infty}\dfrac{1}{\dfrac{1}{x}+\dfrac{1}{x^2}}=\pm\infty$, suy ra đồ thị hàm số không có tiệm cận ngang.
\item $\lim\limits_{x\rightarrow\pm\infty}\dfrac{x^2-3x+2}{x-1}=\lim\limits_{x\rightarrow\pm\infty}\dfrac{1-\dfrac{3}{x}+\dfrac{2}{x^2}}{\dfrac{1}{x}-\dfrac{1}{x^2}}=\pm\infty$, suy ra đồ thị hàm số không có tiệm cận ngang.
\item Hàm số $y=\dfrac{\sqrt{4-x^2}}{x+1}$ có tập xác định $\mathscr{D}=[-2;2]\setminus\{-1\}$, suy ra hàm số không tồn tại giới hạn tại vô tận, từ đó suy ra đồ thị hàm số không có tiệm cận ngang.
\end{itemize}
}
\end{ex}

\begin{ex}%[Thi thử L2, Sở Bắc Giang, 2018]%[2D1G3-1]%[Nguyễn Tiến Thùy, dự án 12EX-10]
