\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 2, TP Hồ Chí Minh]%[8D1B2]
Rút gọn biểu thức:
\begin{multicols}{2}
	\begin{enumerate}[a)]
	\item $3(x^2-7)-x(3x+5)$
	\item $(x+3)(x-3)-x(23+x)$
	\item $(x-2)(x+5)-(x+7)(x+2)$
	\end{enumerate}
\end{multicols}
\loigiai
	{
	\begin{enumerate}[a)]
	\item $3(x^2-7)-x(3x+5) = 3x^2 - 21 - (3x^2+5x) = 3x^2 - 21 - 3x^2 - 5x = -21 - 5x$
	\item $(x + 3)(x - 3) - x(23 + x) = x^2 - 9 - (23x + x^2) = x^2 - 9 - 23x - x^2 = -9 - 23x $
	\item $(x - 2)(x + 5) - (x + 7)(x+2) = x^2 + 3x - 10 - (x^2 + 9x + 14) =  x^2 + 3x - 10 - x^2 - 9x - 14 = -24 - 6x$
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 2, TP Hồ Chí Minh]%[8D1B9]
Phân tích đa thức thành nhân tử:
\begin{multicols}{2}
\begin{enumerate}[a)]
\item $x^3y + 2x^2y^2 + xy^3$
\item $x^2 + 2x - y^2 + 2y$
\item $25x^2 - 4y^2 + 9 +30x$
\item $x^2 - x - 56$
\end{enumerate}
\end{multicols}
\loigiai
	{
	\begin{enumerate}[a)]
	\item $x^3y + 2x^2y^2 + xy^3 = xy(x^2 + 2xy + y^2) = xy(x + y)^2$
	\item $x^2 + 2x - y^2 + 2y = (x^2 - y^2) + (2x + 2y) = (x - y)(x + y) + 2(x + y) = (x + y)(x - y + 2)$
	\item $25x^2 - 4y^2 + 9 +30x = (25x^2 + 30x + 9) - 4y^2 = (5x + 3)^2 - 4y^2 = (5x + 3 - 2y)(5x + 3 + 2y)$
	\item $x^2 - x - 56 = x^2 - 8x + 7x - 56 = (x^2 - 8x) + (7x - 56) = x(x - 8) + 7(x - 8) = (x - 8)(x + 7)$
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 2, TP Hồ Chí Minh]%[8D2K6]
Rút gọn biểu thức:
\begin{multicols}{2}
\begin{enumerate}[a)]
\item $\dfrac{x^2}{x + 1} + \dfrac{2x + 1}{x + 1} + \dfrac{x}{x + 1} + \dfrac{1}{x + 1}$
\item $\dfrac{x}{x - 1} - \dfrac{x + 1}{x} + \dfrac{x - 2}{x^2 - x}$
\end{enumerate}
\end{multicols}
\loigiai
	{
	\begin{enumerate}[a)]
	\item $\dfrac{x^2}{x + 1} + \dfrac{2x + 1}{x + 1} + \dfrac{x}{x + 1} + \dfrac{1}{x + 1} = \left(\dfrac{x^2 + 2x + 1}{x + 1}\right) + \left( \dfrac{x + 1}{x + 1}  \right) = \dfrac{(x + 1)^2}{x + 1} + 1 = x + 1 + 1 = x + 2$.
	\item $\dfrac{x}{x - 1} - \dfrac{x + 1}{x} + \dfrac{x - 2}{x^2 - x} = \dfrac{x^2}{x(x - 1)} - \dfrac{(x + 1)(x - 1)}{x(x - 1)} + \dfrac{x - 2}{x(x - 1)} = \dfrac{x^2 - (x^2 - 1) + x - 2}{x(x - 1)} = \dfrac{1}{x}$
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 2, TP Hồ Chí Minh]%[8H1B9]%[8H1K7]%[8H1K11]%[8H1G9]
Cho $\triangle ABC$ vuông tại $A (AB  < AC)$. Gọi $M$ là trung điểm $BC$. Vẽ $MD \perp AB$ tại $D$ và $ME \perp AC$ tại $E$.
\begin{enumerate}[a)]
\item Chứng minh: Tứ giác $ADME$ là hình chữ nhật.
\item Chứng minh: Tứ giác $BMED$ là hình bình hành.
\item Gọi $F$ là điểm đối xứng với $M$ qua $E$. Chứng minh: Tứ giác $AMCF$ là hình thoi.
\item Gọi $N$ là điểm đối xứng với $E$ qua $M$. Vẽ $EK$ vuông góc $BC$ tại $K$. Chứng minh: $AK \perp NK$.
\end{enumerate}
\begin{center}
\begin{tikzpicture}[scale = 1.5]
\tkzDefPoint(0,0){A}
\tkzDefPoint(0,3){B}
\tkzDefPoint(4,0){C}
\tkzLabelPoints[left](A,B)
\tkzLabelPoints[right](C)
\tkzDefMidPoint(B,C)\tkzGetPoint{M}\tkzLabelPoints[right](M)
\tkzDrawSegments[](A,B A,C B,C)
\tkzDrawAltitude(A,B)(M)\tkzGetPoint{D}\tkzLabelPoints[left](D)
\tkzDrawAltitude(A,C)(M)\tkzGetPoint{E}\tkzLabelPoints[below right](E)
\tkzDrawSegments[](A,M D,E)
\tkzDrawAltitude(B,C)(E)\tkzGetPoint{K}\tkzLabelPoints[right](K)
\tkzDrawSegment[](A,K)
\tkzDefPointBy[symmetry=center E](M)
\tkzGetPoint{F}\tkzLabelPoints[below](F)
\tkzDefPointBy[symmetry=center M](E)
\tkzGetPoint{N}\tkzLabelPoints[above](N)
\tkzInterLL(A,N)(B,E)\tkzGetPoint{O}
\tkzLabelPoint[below](O){O}
\tkzMarkRightAngles[size=0.2](B,A,C A,E,M A,D,M E,K,B)
\tkzMarkSegments[mark=|,size=3](B,M M,C)
\tkzMarkSegments[mark=|||,size=3](E,F E,M M,N)
\tkzDrawSegments[](A,F E,F F,C K,N M,N A,N B,E)
\tkzDrawPoints(A,B,C,M,D,E,K,F,N,O)
\end{tikzpicture}
\end{center}
\loigiai
	{
	\begin{enumerate}[a)]
	\item Xét tứ giác $ADME$ có:
	\begin{enumerate}[+]
	\item $\widehat{DAE} = 90^\circ$ (Tam giác $ABC$ vuông tại $A$)
	\item $\widehat{MDA} = 90^\circ$ ($MD \perp AB$)
	\item $\widehat{AEM} = 90^\circ$ ($ME \perp AC$)
	\end{enumerate}
	Suy ra tứ giác $ADME$ là hình chữ nhật.
	\item Ta có $\left. \begin{aligned}
	ME \perp AC \\
	AB \perp AC
\end{aligned}\right\} \Rightarrow ME  \parallel AB$\\ 
$ME$ là đường thẳng đi qua trung điểm $BC$ và $ME \parallel AB$, suy ra $ME$ là đường trung bình của $\triangle ABC$.\\
 Vậy $E$ là trung điểm của $AC$.\\
Ta có $\left. \begin{aligned}
	MD \perp AB \\
	AC \perp AB
\end{aligned}\right\} \Rightarrow MD  \parallel AC$ \\
$MD$ là đường thẳng đi qua trung điểm $BC$ và $MD  \parallel AC$, suy ra $MD$ là đường trung bình của $\triangle ABC$.\\
Vậy $D$ là đường trung điểm của $AB$. Suy ra $DE$ là đường trung bình của $\triangle ABC$ 
\begin{align}\label{01}
\Rightarrow DE \parallel BC \Rightarrow DE  = \dfrac{1}{2}BC
\end{align}
 Vì $M$ là trung điểm của $BC$
\begin{align}\label{02}
\Rightarrow MB = MC = \dfrac{1}{2}BC
\end{align} 
Từ \eqref{01} và \eqref{02} ta có $\Rightarrow \begin{cases} DE \parallel BM \\ DE = BM \end{cases}$.\\
Suy ra tứ giác $BDME$ là hình bình hành.
\item Xét tứ giác $AMCF$ có:\\
$E$ là trung điểm $MF$ (vì $M$ đối xứng với $F$ qua $E$)\\
Mà $E$ là trung điểm của $AC$ (chứng minh trên)\\
Nên tứ giác $AMCF$ là hình bình hành.\\
Ta có $AC \perp MF$ (vì $ME \perp AC$)\\
Do đó tứ giác $AMCF$ là hình thoi.
\item Gọi $O$ là giao điểm của hai đường chéo $AN$ và $BE$ của hình chữ nhật $ABNE$.\\ Xét tam giác vuông $BKE$ có $KO$ là đường trung tuyến ứng với cạnh huyển $BE$ nên 
\[KO = \dfrac{BE}{2}\]
mà $BE = AN$ (đường chéo hình chữ nhật) nên
\[KO = \dfrac{AN}{2} \]
Xét tam giác $AKN$ có trung tuyến $KO$ bẳng nửa cạnh $AN$ nên tam giác $AKN$ vuông tại $K$.\\
Vậy $AK \perp KN$.
  \end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi Toán 8 Học kỳ 1 năm học 2017-2018, Quận 2, TP Hồ Chí Minh]%[8H2K2]

\begin{enumerate}[a)]
\item Tính số tiền mua gạch.
\item Ước lượng một viên gạch giá bao nhiêu ?
\end{enumerate}
\loigiai
	{
	\begin{enumerate}[a)]
	\item Số tiền lát gạch là: $4,8 \times 15 \times 160000 = 11520000$
	\item Số viên gạch là $200$, giá một viên là $57600$ đồng.
	\end{enumerate}
	}
