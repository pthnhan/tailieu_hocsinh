\begin{bt}%[Đề HK1, Lý Thường Kiệt, Hồ Chí Minh 17-18]%[Ngô Quang Anh]%[0D2B1-4]
Xét tính chẵn, lẻ của hàm số $f(x)=\dfrac{\left| x^2-x\right|-\left| x^2+x\right|}{9-x^2}\cdot$
	\loigiai{
	Tập xác định $\mathscr{D}=\mathbb{R}\backslash \{-3;3\}$ thỏa mãn $\forall x\in \mathscr{D}\Rightarrow-x\in \mathscr{D}$ và\\
	$f(-x)=\dfrac{\left| x^2+x\right|-\left| x^2-x\right|}{9-x^2}=-f(x),\; \forall x\in \mathscr{D}.$\\
	Vậy hàm số đã cho là hàm số lẻ.	
	}
\end{bt}

\begin{bt}%[Đề HK1, Lý Thường Kiệt, Hồ Chí Minh 17-18]%[Ngô Quang Anh]%	[0D3B2-2]
	Giải phương trình $\left| x^2-6x+8\right|=x+2$
	\loigiai{
		\begin{eqnarray*}
		PT \Leftrightarrow  \heva{
			& x+2\ge 0 \\ 
			& \hoac{
				& x^2-6x+8=x+2 \\ 
				& x^2-6x+8=-(x+2)} 
		}
		 \Leftrightarrow &\heva{
			& x\ge-2 \\ 
			& \hoac{
				& x^2-7x+6=0 \\ 
				& x^2-5x+10=0 }
		}
		 \Leftrightarrow &\heva{
			& x\ge-2 \\ 
			& \hoac{
				& x=1 \\ 
				& x=6.} 
		}
	\end{eqnarray*}
		Vậy $x=1$ hoặc $x=6$ là nghiệm phương trình.		
	}
\end{bt}

\begin{bt}%[Đề HK1, Lý Thường Kiệt, Hồ Chí Minh 17-18]%[Ngô Quang Anh]%[0D3B2-4]
Giải phương trình $\sqrt{2x^2+3x-4}+6=7x.$
\loigiai{{\allowdisplaybreaks 
	\begin{align*}
	Pt &\Leftrightarrow \sqrt{2x^2+3x-4}=7x-6 \Leftrightarrow \heva{
		& 7x-6\ge 0 \\ 
		& 2x^2+3x-4=49x^2-84x+36 \\ 
	}\\
	&\Leftrightarrow \heva{
		& x\ge \dfrac{6}{7}\\ 
		& 47x^2-87x+40=0 \\ 
	}
	\;\;\Leftrightarrow \heva{
		& x\ge \dfrac{6}{7} \\ 
		& x=1; x=\dfrac{40}{47} \\ 
	} \Leftrightarrow x=1.
	\end{align*}}
	Vậy phương trình đã cho có nghiệm $x=1$.
}
\end{bt}

\begin{bt}%[Đề HK1, Lý Thường Kiệt, Hồ Chí Minh 17-18]%[Ngô Quang Anh]%[0D3B2-5]
	Tìm $m$ để phương trình  $\left(m-1\right)x^2-\left(2m+4\right)x+m-1=0$ có 2 nghiệm  $x_1, x_2$ thỏa hệ thức   $\dfrac{x_1}{x_2}+\dfrac{x_2}{x_1}=62$.
\loigiai{Pt có 2 nghiệm $\Leftrightarrow \heva{
		& a\ne 0 \\ 
		& {\Delta}'\ge 0 \\ 
	}\Leftrightarrow \heva{
		& m\ne 1 \\ 
		& 6m+3\ge 0 \\ 
	}\Leftrightarrow \heva{
		& m\ne 1 \\ 
		& m\ge-\dfrac{1}{2}\cdot  \\ 
	}$\\
	Theo định lí Viét:\;
	$x_1+x_2=\dfrac{2m+4}{m-1};x_1x_2=\dfrac{m-1}{m-1}=1.$ \\
	$\dfrac{x_1}{x_2}+\dfrac{x_2}{x_1}=62\Leftrightarrow x_1^2+x_2^2=62x_1x_2\Leftrightarrow {\left(x_1+x_2\right)}^2=64x_1x_2\Leftrightarrow {\left(\dfrac{2m+4}{m-1}\right)}^2=64\\
	\Leftrightarrow 15m^2-36m+12=0$ $\Leftrightarrow \hoac{
		& m=2\quad\;(N) \\ 
		& m=\dfrac{2}{5}\quad\;(N). \\ 
	}$\\
	Vậy $m=2$ hay $m=\dfrac{2}{5}\cdot $ 
}	
\end{bt}

\begin{bt}%[Đề HK1, Lý Thường Kiệt, Hồ Chí Minh 17-18]%[Ngô Quang Anh]%[0D4K1-2]
	Cho $x,y\in \mathbb{R}$ thỏa mãn $2x^2+3y^2=5$. Chứng minh rằng $2x+3y\le 5.$
\loigiai{Áp dụng BĐT Cauchy, ta có\\
	$2x^2+2\ge 2\sqrt{2x^2\cdot 2}=4\left| x\right|$ \\
	$3y^2+3\ge 2\sqrt{3y^2\cdot 3}=6\left| y\right|$\\
	$\Rightarrow 4\left| x\right|+6\left| y\right|\le 2x^2+3y^2+5=10\Leftrightarrow 2\left| x\right|+3\left| y\right|\le 5.$\\
	Mặt khác $2\left| x\right|+3\left| y\right|\ge 2x+3y.$\\
	Vậy $2x+3y\le 5.$
}	
\end{bt}

\begin{bt}%[Đề HK1, Lý Thường Kiệt, Hồ Chí Minh 17-18]%[Ngô Quang Anh]%[0H1B4-3]
Trong mặt phẳng tọa độ $Oxy$, cho ba điểm $M(-1;1)$, $N\left(1;3\right)$, $P(-2;5).$ 
Tìm tọa độ điểm $E$ biết $\vec{PE}=2\vec{MN}$.
\loigiai{Ta có $\vec{MN}=(2;2)$.\\
	$\vec{PE}=2\vec{MN}\Leftrightarrow \heva{
		& x_E+2=4 \\ 
		& y_E-5=4 \\ 
	}\Leftrightarrow \heva{
		& x_E=2 \\ 
		& y_E=9. \\}$\\
	Vậy $E(2;9)$.
}	
\end{bt}
\begin{bt}%[Đề HK1, Lý Thường Kiệt, Hồ Chí Minh 17-18]%[Ngô Quang Anh]%[0H2B2-3]
Trong mặt phẳng tọa độ $Oxy$, cho $A\left(1;2\right)$, $B\left(3;1\right)$. Chứng minh rằng tam giác $AOB$ là vuông cân tại $A$. Tính diện tích của tam giác $AOB.$
\loigiai{Ta có $\vec{AO}=(-1;-2),\vec{AB}=(2;-1).$\\
	$\vec{AO}\cdot \vec{AB}=0$ và $AO=AB=\sqrt{5}\Rightarrow $ Tam giác $AOB$ vuông cân tại $A$.\\
	Diện tích tam giác $AOB$ là $S=\dfrac{1}{2}\cdot AO\cdot AB=\dfrac{5}{2}\cdot $
}
	
\end{bt}
\begin{bt}%[Đề HK1, Lý Thường Kiệt, Hồ Chí Minh 17-18]%[Ngô Quang Anh]%[0H2B2-3]
	Trong mặt phẳng tọa độ $Oxy$, cho tam giác $ABC$ với $A(-2;6)$, $B(1;-3)$,  
	$C\left(5;5\right)$. Tìm tọa độ trực tâm $H$ của tam giác $ABC$. 
\loigiai{Gọi $H\left(x;y\right)$ $\Rightarrow \vec{AH}(x+2;y-6),\; \vec{BC}(4;8)$, $\vec{BH}(x-1;y+3), \;\vec{AC}(7;-1)$.\\
	$H$ là trực tâm tam giác $ABC$ 
	$\Leftrightarrow \heva{
		& \vec{AH}\cdot \vec{BC}=0 \\ 
		& \vec{BH}\cdot \vec{AC}=0 \\ 
	}$
	$\Leftrightarrow \heva{
		& 4x+8y=40 \\ 
		& 7x-y=10. \\ 
	}$ \\
	Vậy $H\left(2;4\right)$.
}	
	
\end{bt}
