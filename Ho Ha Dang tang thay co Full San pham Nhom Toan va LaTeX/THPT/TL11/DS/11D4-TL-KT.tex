				\begin{aligned}
				S_{n+1}&=S_n +a_n\cdot b_{n+1}^2=S_{n-1}+a_{n-1}\cdot b_n^2+a_n\cdot b_{n+1}^2=\cdots\\

				&=2-\left(\dfrac{5}{9}\right)^{n+1}.
				\end{aligned}
				$$
