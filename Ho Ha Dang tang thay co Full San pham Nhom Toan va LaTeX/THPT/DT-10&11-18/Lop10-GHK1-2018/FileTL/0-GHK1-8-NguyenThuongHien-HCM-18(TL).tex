\begin{bt}%[Đề kiểm tra GHKI, Nguyễn Thượng Hiền - Hồ Chí Minh, 2018]%[Bùi Quốc Hoàn, EX-DT-10&11]%[0D2B3-1]%[0D2B1-2]
Tìm tập xác định của  các hàm số sau
\begin{enumEX}{2}
	\item $y = \dfrac{5 - x^2}{\left(x + 2\right)\sqrt{-x - 1}}$.
	\item $y = \dfrac{\sqrt{5 - 2\sqrt{x - 1}}}{\left\vert 3 - x \right\vert + x - 3}$.
\end{enumEX}
	\loigiai{	\begin{enumerate}
			\item Để hàm số xác định khi\\
			$\heva{&\left(x + 2\right)\sqrt{- x - 1}\neq 0\\ &- x - 1\geq 0}
			 \Leftrightarrow \heva{&x + 2\neq 0\\ &- x - 1 > 0}\Leftrightarrow \heva{&x\neq -2\\ &x < - 1}.$\\
			 	Vậy tập xác định của hàm số là $\mathscr{D} = \left(-\infty;-1\right)\setminus\left\{- 2\right\}$.
			\item Để hàm số xác định khi\\  
				$\heva{&5 -  2\sqrt{x - 1}\geq 0\\ & x - 1\geq 0\\&\left\vert 3 - x \right\vert + x - 3\neq 0}\Leftrightarrow \heva{&\sqrt{x - 1}\leq \dfrac{5}{2}\\&x -1\geq 0\\ &\left\vert 3 - x \right\vert \neq 3 - x}\Leftrightarrow \heva{&x - 1\leq \dfrac{25}{4}\\&x - 1\geq 0\\ &3 - x < 0}
				\Leftrightarrow \heva{&x\leq \dfrac{29}{4}\\&x\geq 1\\ &x > 3}\Leftrightarrow 3 < x\leq \dfrac{29}{4}.$
		Vậy tập xác định của hàm số là $\mathscr{D} = \left(3; \dfrac{29}{4}\right]$.
			\end{enumerate}
	}
\end{bt}
\begin{bt}%[Đề kiểm tra GHKI, Nguyễn Thượng Hiền - Hồ Chí Minh, 2018]%[Bùi Quốc Hoàn, EX-DT-10&11]%[0D2B1-4]
	Xét tính chẵn lẻ của hàm số $f\left(x\right) = \dfrac{x^4 - 2x^2 + 3}{\left\vert x\right\vert\left(x^3 + x\right)}$. 
		\loigiai{Tập xác định của hàm số $\mathscr{D} = \mathbb{R}\setminus\left\{0\right\}$.\\
	Dễ thấy $x\in \mathscr{D}$ thì $- x\in \mathscr{D}$.\\
	Mặt khác 
		\begin{eqnarray*}
	f\left(- x\right) &=& \dfrac{\left(- x\right)^4 - 2\left(- x\right)^2 + 3}{\left\vert - x\right\vert\left(\left(- x\right)^3 + \left(- x\right)\right)}\\ 
	&=& \dfrac{x^4 - 2x^2 + 3}{\left\vert x\right\vert\left(- x^3 - x\right)} = - \dfrac{x^4 - 2x^2 + 3}{\left\vert x\right\vert\left(x^3 + x\right)}\\
	 &=&  -f\left(x\right).
	\end{eqnarray*}
Do đó hàm số $y = f\left(x\right)$ là hàm lẻ.
		}
\end{bt}

\begin{bt}%[Đề kiểm tra GHKI, Nguyễn Thượng Hiền - Hồ Chí Minh, 2018]%[Bùi Quốc Hoàn, EX-DT-10&11]%[0D2B1-3]
	Xét sự biến thiên của  hàm số $f\left(x\right) = \dfrac{2x +  5}{x - 4}$ trên khoảng $\left(4; +\infty\right)$. 
	\loigiai{Xét hàm số $f\left(x\right) = \dfrac{2x +  5}{x - 4}$ trên khoảng $\left(4; +\infty\right)$.\\
	Giả sử $x_{1}$, $x_{2}$ bất kỳ thuộc $\left(4; +\infty\right)$ và  $x_{1} < x_{2}$.\\
	Khi đó 
		\begin{eqnarray*}
		f\left(x_{2}\right) - f\left(x_{1}\right) &=& \dfrac{2x_{2} +  5}{x_{2} - 4}- \dfrac{2x_{1} +  5}{x_{1} - 4}\\
		 &=&  \left(2  + \dfrac{13}{x_{2} - 4}\right) - \left(2 + \dfrac{13}{x_{1} - 4}\right)\\
		  &=& \dfrac{13}{x_{2} - 4} - \dfrac{13}{x_{1} - 4}
		 = \dfrac{13\left(x_{1} - x_{2}\right)}{\left(x_{2} - 4\right)\left(x_{1} - 4\right)}.
	\end{eqnarray*}
	Do giả thiết ta có $x_{1} - x_{2} < 0$ và $\left(x_{2} - 4\right)\left(x_{1} - 4\right)$ do đó $f\left(x_{2}\right) - f\left(x_{1}\right) < 0\Leftrightarrow f\left(x_{2}\right) < f\left(x_{1}\right)$.\\
	Vậy hàm số nghịch biến trên khoảng $\left(4; +\infty\right)$.
}
\end{bt}


\begin{bt}%[Đề kiểm tra GHKI, Nguyễn Thượng Hiền - Hồ Chí Minh, 2018]%[Bùi Quốc Hoàn, EX-DT-10&11]%[0D2B3-1]%[0D2K3-2]
	\begin{enumEX}{1}
		\item Lập bảng biến thiên và vẽ đồ thị hàm số $y = - 2x^2 + 4x  - 3$
		\item Xác định parabol $y = ax^2 + bx  +c$, biết rằng parabol đó đi qua $2$ điểm $A\left(2; 0\right)$, $B\left(- 2; - 8\right)$ và tung độ đỉnh là $1$.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item - Tập xác định của hàm số: $\mathscr{D} = \mathbb{R}$.\\
		Do $a = - 2$ ta suy ra \\ 
		  hàm số nghịch biến trên khoảng $\left(-1; +\infty\right)$,\\
		  hàm số đồng biến trên khoảng $\left(-\infty; 1\right)$.\\
		  Tọa độ đỉnh $P\left(1;-1\right)$.\\
		  - Bảng biến thiên:
		  	\begin{center}
		  \begin{tikzpicture}[scale=1, font=\footnotesize, line join=round, line cap=round, >=stealth]
		  \tkzTabInit[nocadre=false,lgt=1.2,espcl=2.5,deltacl=0.6]
		  {$x$/0.6,$f(x)$/2}
		  {$-\infty$,$1$,$+\infty$}
		  \tkzTabVar{-/$-\infty$,+/$-1$,-/$-\infty$}
		  \end{tikzpicture}
		  \end{center}
		 -  Đồ thị:\\
		  Khi $x = 2$ suy ra $y = - 3$;\\
		  Khi  $x = 0$ suy ra $y = - 3$. 
		  	\begin{center}
		  	\begin{tikzpicture}[scale=1, font=\footnotesize, line join=round, line cap=round, >=stealth]
		  	\draw[->] (-3,0)--(0,0) node[below left]{$O$}--(4,0) node[below]{$x$};
		  \draw[->] (0,-5.5) --(0,2) node[right]{$y$};
		   \draw (-1,0) node[below]{$-1$} circle (1pt);
		   \draw (-2,0) node[below]{$-2$} circle (1pt);
		  \draw (1,0) node[below left]{$1$} circle (1pt);
		   \draw (2,0) node[below]{$2$} circle (1pt);
		    \draw (3,0) node[below]{$3$} circle (1pt);
		  \draw (0,1) node[left]{$1$} circle (1pt);
		  \draw (0,-1) node[left]{$-1$} circle (1pt);
		  \draw (0,-2) node[left]{$-2$} circle (1pt);

		    \draw [black, thick, domain=-0.5:2.5, samples=100] %
		  plot (\x, {-2*(\x)^2 + 4*(\x) - 3});

		   \draw[dashed] (2,0)--(2,-3)--(0,-3);

		\end{center}
	  		\item Do giả thiết đồ thị hàm số  đi qua $2$ điểm $A\left(2; 0\right)$, $B\left(- 2; - 8\right)$ và tung độ đỉnh là $1$ ta có hệ phương trình\\
	  		$\heva{& 4a + 2b + c = 0\\ &4a - 2b + c = - 8\\&- \dfrac{b^2 - 4ac}{4a} = 1}\Leftrightarrow \heva{& 4a + 2b + c = 0\quad (1)\\ &4a - 2b + c = - 8\quad (2)\\&b^2 - 4ac = - 4a\quad(3)}$\\
	  	Từ $(1)$ và $(2)$ suy ra $4b  = 8\Leftrightarrow b = 2\quad (4)$.\\
	  	Khi đó thay vào $(1)$ suy ra $c = - 4 - 4a\quad (5)$.\\
	  Thay $(4)$, $(5)$ vào $(3)$ ta có\\
	  $1 - a\left(- 4 - 4a\right) = - a \Leftrightarrow 4a^2 - 5a + 1 = 0 \Leftrightarrow \hoac{& a = 1 \\ & a = \dfrac{1}{4}}.$\\
  		Vậy các parabol thỏa mãn có phương trình là $y = x^2 + 2x - 8$ và $y = \dfrac{1}{4}x^2 + 2x - 5$. 
  				\end{enumerate}
		}
