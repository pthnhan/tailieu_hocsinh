\begin{ex}%[Đề TS Lớp 10, Tỉnh Bạc Liêu 2018]%[9D1B8]
	Rút gọn biểu thức
	     \begin{enumerate}
                \item $A=\sqrt{45}+\sqrt{20}-2\sqrt{5}$.
                \item $B=\dfrac{a+2\sqrt{a}}{\sqrt{a}+2}-\dfrac{a-4}{\sqrt{a}-2}$, (với $a\ge 0; a\neq 4$).
              \end{enumerate}
	\loigiai{
	    \begin{enumerate}
                \item \begin{eqnarray*}
                           A &=&\sqrt{45}+\sqrt{20}-2\sqrt{5}\\
                             &=& \sqrt{3^2\cdot5}+\sqrt{2^2\cdot 5}-2\sqrt{5}\\
                             &=& 3\sqrt{5}+2\sqrt{5}-2\sqrt{5}\\
                             &=&3\sqrt{5}.
                       \end{eqnarray*}
                 \item \begin{eqnarray*}
                           B &=&\dfrac{a+2\sqrt{a}}{\sqrt{a}+2}-\dfrac{a-4}{\sqrt{a}-2}\\
                             &=& \dfrac{\sqrt{a}(\sqrt{a}+2)}{\sqrt{a}+2}-\dfrac{(\sqrt{a}+2)(\sqrt{a}-2)}{\sqrt{a}-2}\\
                             &=&\sqrt{a}-(\sqrt{a}+2)\\
                             &=&-2.
                       \end{eqnarray*}
              \end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề TS Lớp 10, Tỉnh Bạc Liêu 2018]%[9D4B2]
	Cho hàm số $y=\dfrac{1}{2}x^2$ có đồ thị $(P)$ và đường thẳng $d\colon y=x+2m$. 
	 \begin{enumerate}
                \item Vẽ đồ thị $(P)$.
                \item Tìm tất cả các giá trị của $m$ sao cho $d$ cắt $(P)$ tại điểm có hoành độ bằng $-1$.
              \end{enumerate}

	\begin{enumerate}

\end{longtable}

\item Hoành độ giao điểm của đường thẳng $d$ và parabol $(P)$ là nghiệm của phương trình
$$\dfrac{1}{2}x^2=x+2m\Leftrightarrow \dfrac{1}{2}x^2-x-2m=0.\quad(1)$$
Để $d$ cắt $(P)$ tại điểm có hoành độ bằng $-1\Leftrightarrow x=-1$ là nghiệm của phương trình $(1)$
\begin{eqnarray*}
    &\Leftrightarrow& \dfrac{1}{2}\cdot(-1)^2-(-1)-2m=0\\
    &\Leftrightarrow& 2m=\dfrac{3}{2}\\
    &\Leftrightarrow& m=\dfrac{3}{4}.
\end{eqnarray*}
Vậy $m=\dfrac{3}{4}$.
\end{enumerate}
		
	}

\begin{ex}%[Đề TS Lớp 10, Tỉnh Bạc Liêu 2018]%[9D4B6]%[9D4B4]%[9D4K6]
	Cho phương trình $x^2+4x+m+1=0\quad (1)$ (với $m$ là tham số).
	\begin{enumerate}
                \item Giải phương trình $(1)$ với $m=2$.
                \item Tìm điều kiện của $m$ để phương trình $(1)$ có nghiệm.
                \item Tìm tất cả các giá trị của $m$ sao cho phương trình $(1)$ có hai nghiệm $x_1$, $x_2$ thỏa mãn điều kiện $$\dfrac{x_1-1}{2x_2}+\dfrac{x_2-1}{2x_1}=-3.$$
              \end{enumerate}
	\loigiai{
		\begin{enumerate}
                \item Thay $m=2$ vào $(1)\colon x^2+4x+3=0$\\
                Ta có $a-b+c=1-4+3=0\Leftrightarrow$ phương trình có hai nghiệm 
                $\begin{cases} x_1=-1 \\ x_2=-3. \end{cases}$\\
               Vậy, với $m=2$ thì phương trình có hai nghiệm $x_1=-1$, $x_2=-3$.
                \item $\Delta'=2^2-(m+1)=3-m$.\\
                Để phương trình $(1)$ có nghiệm khi $\Delta'\ge 0\Leftrightarrow 3-m\ge 0\Leftrightarrow m\le 3$.
                \item Để phương trình có hai nghiệm phân biệt $\Leftrightarrow \Delta'>0\Leftrightarrow3-m>0\Leftrightarrow m<3$.\\
                Áp dụng định lý Vi-ét, ta có  $\begin{cases} x_1+x_2=-4 \\ x_1x_2=m+1 \end{cases}\quad (*)$\\
               Ta có 
                \begin{eqnarray*}
                  & &  \dfrac{x_1-1}{2x_2}+\dfrac{x_2-1}{2x_1}=-3\\
                  &\Leftrightarrow& x_1(x_1-1)+x_2(x_2-1)=-6x_1x_2,\,(x_1\neq 0,x_2\neq 0)\\
                  &\Leftrightarrow& (x_1^2+x_2^2)-(x_1+x_2)=-6x_1x_2\\
                  &\Leftrightarrow& (x_1+x_2)^2-2x_1x_2-2(x_1+x_2)=-6x_1x_2\\
                   &\Leftrightarrow& (x_1+x_2)^2+4x_1x_2-(x_1+x_2)=0.\quad (**)\\
                   \text{Thay $(*)$ vào $(**)$ ta được}\\
                  & & 16+4(m+1)+4=0\Leftrightarrow m=-6\,\text{(thỏa mãn)}.
              \end{eqnarray*}
             Vậy $m=-6$.
              \end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề TS Lớp 10, Tỉnh Bạc Liêu 2018]%[9H3B7]
	Cho nửa đường tròn tâm $O$ đường kính $AB$. Vẽ bán kính $CO$ vuông góc với $AB$, $M$ là một điểm bất kỳ trên cung $AC$ ($M$ khác $A$, $C$ và điểm chính giữa cung $AC$); $BM$ cắt $AC$ tại $H$. Gọi $K$ là chân đường vuông góc kẻ từ $H$ đến $AB$.
	\begin{enumerate}
                \item Chứng minh tứ giác $BCHK$ là tứ giác nội tiếp.
                \item Chứng minh $CA$ là phân giác góc $MCK$.
                \item Kẻ $CP$ vuông góc với $BM$ ($P\in BM$) và trên đoạn $BM$ lấy điểm $E$ sao cho $BE=AM$. Chứng minh $ME=2CP$.
              \end{enumerate}
	\loigiai{
		\begin{center}
                       \begin{tikzpicture}[scale=1]
                        \tkzDefPoints{-4/0/A, 4/0/B, 0/0/O, 0/4/C}
                        \tkzDefPointBy[rotation=center O angle -32](A)
                        \tkzGetPoint{M}
                        \tkzInterLL(A,C)(M,B)
                        \tkzGetPoint{H}
                        \tkzDefPointBy[rotation=center C angle 90](M)
                        \tkzGetPoint{E}
                        \tkzDefMidPoint(M,E)
                        \tkzGetPoint{P}
                        \tkzDefPointBy[projection=onto A--B](H)
                        \tkzGetPoint{K}
                        \pgfresetboundingbox
                        \draw (B) arc[radius=4,start angle=0, end angle =180];
                        \draw (A)--(B)--(C)--(A) (B)--(M); 
                        \draw (A)--(M)--(C) (H)--(K)--(C)--(P) (C)--(O) (C)--(E); 
                        \tkzLabelPoints[below](E,A,K,O,B,P)
                        \tkzLabelPoints[above](H,C)
                        \tkzLabelPoints[left](M)
                        \tkzMarkAngle[arc=ll,size=.9cm](H,C,K)
                        \tkzMarkAngle[arc=ll,size=.9cm](E,B,O)
                        \tkzMarkAngle[arc=ll,size=.9cm](M,C,A)
                        \tkzMarkSegments[mark=||](A,M B,E)
                        \tkzMarkRightAngle(B,M,A)
                        \tkzMarkRightAngle(B,K,H)
                        \tkzMarkRightAngle(B,P,C)
                        \tkzMarkRightAngle(B,C,H)
                        \tkzDrawPoints(A,B,C,O,K,H,M,P,E)
                        \tkzMarkAngle[arc=l,size=.6cm](C,A,M)
                        \tkzMarkAngle[arc=l,size=.6cm](C,B,M)
                        \end{tikzpicture}
                         \end{center}
                \begin{enumerate}
                \item Xét tứ giác $BCHK$ có
                        $\begin{cases} \widehat{BCH}=\widehat{BCA}=90^\circ \,\text{(góc nội tiếp chắn nửa đường tròn)} \\  \widehat{BKH}=90^\circ \,\text{(do $HK\perp AB$)}\end{cases}$\\ 
                        $\Rightarrow BCHK$ là tứ giác nội tiếp đường tròn đường kính $BH$.
                \item Xét $(O)$ có 
                        $\widehat{ABM}=\widehat{ACM}$                   \quad $(1)$ (hai góc nội tiếp cùng chắn cung $AM$).\\
                        Xét đường tròn ngoại tiếp tứ giác $BCHK$ có $\widehat{KBH}=\widehat{KCH}$\quad $(2)$ (hai góc nội tiếp cùng chắn cung $HK$).\\
                        Từ $(1)$, $(2)\Rightarrow \widehat{KCH}=\widehat{ACM}\Rightarrow CA$ là tia phân giác góc $\widehat{MCK}$.
                \item Xét $\triangle ACM$ và $\triangle BEC$ có
                        $$\begin{cases} \widehat{MAC}=\widehat{EBC}\, \text{(hai góc nối tiếp cùng chắn cung MC)}\\ AM=BE\, \text{(giả thiết)}\\AC=BC\, \text{(giả thiết)}\end{cases}$$
                        $\Rightarrow \triangle AMC=\triangle BEC\Rightarrow MC=EC\Rightarrow \triangle MCE$ cân tại $C\Rightarrow P$ là trung điểm đoạn $ME$.\\
                        Lại có $\widehat{CME}=\widehat{CMB}=\dfrac{1}{2}\text{sđ}\wideparen{BC}=45^\circ \Rightarrow \triangle MCE$ vuông cân tại $C$. Vậy $CP=\dfrac{1}{2}ME$ (tính chất đường trung tuyến trong tam giác vuông). Hay $ME=2CP$.
                
                \end{enumerate}
	}
\end{ex}
