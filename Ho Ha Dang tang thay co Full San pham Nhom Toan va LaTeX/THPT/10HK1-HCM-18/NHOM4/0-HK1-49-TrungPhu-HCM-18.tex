\begin{bt}%[Đề HK1, Trung Phú, Hồ Chí Minh 17-18]%[Trần Nhân Kiệt]%[0D2B1-2]%[0D2B1-4]
\begin{enumerate}
\item Tìm tập xác định của $y=\dfrac{\sqrt{5-2x}}{x^2-9}+\dfrac{1}{3x+1}.$
\item Xét tính chẵn lẻ của hàm số $f(x)=3|x|\cdot x^2-2|x|$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Điều kiện $\heva{&5-2x\ge 0\\&x^2-9\neq 0\\&3x+1\neq 0}\Leftrightarrow \heva{&x\le \dfrac{5}{2}\\&x\neq \pm 3\\&x\neq -\dfrac{1}{3}}\Leftrightarrow x\in \left(-\infty;\dfrac{5}{2}\right]\setminus \left\{-3;-\dfrac{1}{3}\right\}.$\\
Tập xác định là $\mathscr{D}=\left(-\infty;\dfrac{5}{2}\right]\setminus \left\{-3;-\dfrac{1}{3}\right\}.$
\item Tập xác định là $\mathscr{D}=\Bbb{R}.$\\
Ta có $\forall x\in \mathscr{D}\Rightarrow -x\in\mathscr{D}.$\\
Xét $f(-x)=3|-x|\cdot (-x)^2-2|-x|=3|x|\cdot x^2-2|x|.$\\
Suy ra $f(-x)=f(x)$, $\forall x\in \mathscr{D}.$ \\
Vậy hàm số đã cho là hàm số chẵn.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Trung Phú, Hồ Chí Minh 17-18]%[Trần Nhân Kiệt]%[0D2B3-3]%[0D2B3-4]
Cho Parabol $(P)\colon y=x^2-4x+2.$
\begin{enumerate}
\item Khảo sát sự biến thiên và vẽ $(P).$
\item Biện luận theo $m$ số nghiệm của phương trình $x^2-4x=4m+1.$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Tập xác định $\mathscr{D}=\Bbb{R}.$
\immini
{
\begin{itemize}
\item Tọa độ đỉnh $I(2;-2), $ trục đối xứng $x=2.$
\item Bảng biến thiên

{$x$ /1,$y$ /2}

\end{center}
Hàm số đồng biến trên khoảng $(2;+\infty)$ và nghịch biến trên khoảng $(-\infty;2).$
\item Hệ $a=1>0\Rightarrow (P)$ quay bề lõm lên trên.
\end{itemize}
}

\item Xét phương trình $x^2-4x=4m+1 \quad (1).$
Ta có $(1)\Leftrightarrow x^2-4x+2=4m+3.$\\
Đặt $y=x^2-4x+2$ có đồ thị là Parabol $(P)$ và $d\colon y=4m+3$ với $d\parallel Ox$ hoặc $d\equiv Ox.$\\
Số giao điểm của $d$ và $(P)$ chính là số nghiệm của phương trình $(1).$\\
Dựa vào đồ thị Parabol $(P)$ suy ra 
\begin{itemize}
\item $4m+3<-2\Leftrightarrow m<-\dfrac{5}{4}\Rightarrow d$ và $(P)$ không có điểm chung $\Rightarrow (1)$ vô nghiệm.
\item $4m+3=-2\Leftrightarrow m=-\dfrac{5}{4}\Rightarrow d$ và $(P)$ có 
$1$ điểm chung $\Rightarrow (1)$ có $1$ nghiệm.
\item $4m+3>-2\Leftrightarrow m>-\dfrac{5}{4}\Rightarrow d$ và $(P)$ có $2$ điểm chung $\Rightarrow (1)$ có $2$ nghiệm.
\end{itemize}
\end{enumerate}
}

\begin{bt}%[Đề HK1, Trung Phú, Hồ Chí Minh 17-18]%[Trần Nhân Kiệt]%[0D3B2-3]
Giải phương trình $\dfrac{x+3}{x^2-3x+2}+\dfrac{x+1}{x-2}=1.$
\loigiai{
Điều kiện $\heva{&x^2-3x+2\neq 0\\&x-2\neq 0}\Leftrightarrow \heva{&x\neq 1\\&x\neq 2.}$\\
Đặt $\dfrac{x+3}{x^2-3x+2}+\dfrac{x+1}{x-2}=1\quad (1).$
\begin{align*}
(1)&\Leftrightarrow \dfrac{x+3+(x+1)(x-1)}{x^2-3x+2}=1\\
&\Leftrightarrow x^2+x+2=x^2-3x+2\\
&\Leftrightarrow x=0 \, (\text{thỏa mãn}).
\end{align*}
Tập nghiệm của phương trình là $S=\{0\}.$
}
\end{bt}


\begin{bt}%[Đề HK1, Trung Phú, Hồ Chí Minh 17-18]%[Trần Nhân Kiệt]%[0D2B2-6]
Tìm $m$ để phương trình $(m-2)x^2-(2m+5)x+m+9=0$ vô nghiệm.
\loigiai{
\begin{itemize}
\item Xét $m-2=0\Leftrightarrow m=2.$\\
Thay $m=2$ vào phương trình $\Rightarrow -9x+11=0\Leftrightarrow x=\dfrac{11}{9}\Rightarrow m=2$ (không thỏa mãn).
\item Xét $m-2\neq 0\Leftrightarrow m\neq 2.$\\
Phương trình đã cho vô nghiệm khi và chỉ khi 
\begin{align*}
&\heva{&m\neq 2\\&\Delta <0}\\
\Leftrightarrow &\heva{&m\neq 2\\& (2m+5)^2-4(m-2)(m+9)<0}\\
\Leftrightarrow &\heva{&m\neq 2\\& -8m+97<0}\\
\Leftrightarrow &\heva{&m\neq 2\\& m>\dfrac{97}{8}}\\
\Leftrightarrow & m>\dfrac{97}{8}.
\end{align*}
Vậy $m\in\left(\dfrac{97}{8};+\infty\right)$ thì phương trình đã cho vô nghiệm.
\end{itemize}
}
\end{bt}


\begin{bt}%[Đề HK1, Trung Phú, Hồ Chí Minh 17-18]%[Trần Nhân Kiệt]%[0D3K3-4]
Giải hệ phương trình $\heva{&x^2-2xy+3x-4y=-2\\&y^2-2xy+3y-4x=-2.}$
\loigiai{
Đặt $\heva{&x^2-2xy+3x-4y=-2\quad (1)\\&y^2-2xy+3y-4x=-2\quad (2)}.$ Lấy $(1)$ trừ $(2)$ theo vế, ta được\\
$$x^2-y^2+7x-7y=0\Leftrightarrow (x-y)(x+y+7)=0\Leftrightarrow \hoac{&x-y=0\\&x+y+7=0.} $$
\begin{itemize}
\item Xét $x-y=0\Leftrightarrow y=x.$ Thay vào $(1)$, ta được $-x^2-x+2=0\Leftrightarrow \hoac{&x=1\Rightarrow y=1\\&x=-2\Rightarrow y=-2.}$
\item Xét $x+y+7=0\Leftrightarrow y=-x-7.$ Thay vào $(1),$ ta được
$$x^2-2x\cdot(-x-7)+3x-4\cdot(-x-7)=-2\Leftrightarrow x^2+7x+10=0\hoac{&x=-2\Rightarrow y=-5\\&x=-5\Rightarrow y=-2.} $$
\end{itemize}
Vậy hệ đã cho có $4$ nghiệm là $(1;1)$, $(-2;-2)$, $(-2;-5)$, $(-5;-2).$
}
\end{bt}


\begin{bt}%[Đề HK1, Trung Phú, Hồ Chí Minh 17-18]%[Trần Nhân Kiệt]%[0D4G1-1]
Chứng minh rằng $$12xy+8x-14< 4x^4y^4-3x^2y^2+16x^2,\quad\forall x, y\in \Bbb{R}.$$
\loigiai{
Ta có
\begin{align*}
& 12xy+8x-14< 4x^4y^4-3x^2y^2+16x^2\\
\Leftrightarrow &\left(4x^4y^4-8x^2y^2+4\right)+\left(5x^2y^2+16x^2-12xy-8x+10\right)> 0\\
\Leftrightarrow & 4\left(x^2y^2-1\right)^2+\left(5y^2+16\right)x^2-2(6y+4)x+10> 0.
\end{align*}
Xét $f(x)=\left(5y^2+16\right)x^2-2(6y+4)x+10.$\\
Coi $f(x)$ là tam thức bậc hai ẩn $x$, hệ số $a=5y^2+16>0.$\\
Ta có $\Delta'=(6y+4)^2-10(5y^2+16)=-14y^2+48y-144=-14\left(y-\dfrac{12}{7}\right)^2-\dfrac{720}{7}<0$, $\forall y\in \Bbb{R}.$\\
Suy ra $f(x)>0$, $\forall x, y\in \Bbb{R}.$
Mặt khác $4\left(x^2y^2-1\right)^2\ge 0 $, $\forall x, y\in \Bbb{R}.$\\
Vậy $4\left(x^2y^2-1\right)^2+\left(5y^2+16\right)x^2-2(6y+4)x+10> 0$, $\forall x, y\in \Bbb{R}.$
}
\end{bt}

Trong mặt phẳng $(Oxy)$, cho tam giác $ABC$ có $A(1;3)$, $B(3;4)$, $C(7;1).$
\begin{enumerate}
\item Tìm tọa độ điểm $D$ để $ABCD$ là hình bình hành.
\item Tìm tọa độ trực tâm $H$ của tam giác $ABC.$
\end{enumerate}
\loigiai{
\immini
{
\begin{enumerate}
\item Gọi $D(x;y)\Rightarrow \vec{AD}=(x-1;y-3)$, $\vec{BC}=(4;-3).$\\
Vì $ABCD$ là hình bình hành nên $\vec{AD}=\vec{BC}$\\
$\Leftrightarrow \heva{&x-1=4\\&y-3=-3}\Leftrightarrow \heva{&x=5\\&y=0}\Rightarrow D(5;0).$
\item Gọi $H(x;y)$ là trực tâm tam giác $ABC.$\\
Ta có $\vec{AH}=(x-1;y-3)$, $\vec{BC}=(4;-3)$,\\ $\vec{BH}=(x-3;y-4)$, $\vec{AC}=(6;-2)$.\\
Vì $H$ là trực tâm $\triangle ABC$ nên 
\begin{align*}
& \heva{&\vec{AH}\cdot \vec{BC}=0\\&\vec{BH}\cdot \vec{AC}=0}
\Leftrightarrow \heva{&4\cdot(x-1)-3\cdot(y-3)=0\\&6\cdot(x-3)-2\cdot(y-4)=0}\\
\Leftrightarrow &\heva{&4x-3y=-5\\&3x-y=5}\Leftrightarrow \heva{&x=4\\&y=7.}
\end{align*}
Vậy $H(4;7).$
\end{enumerate}
}

\begin{bt}%[Đề HK1, Trung Phú, Hồ Chí Minh 17-18]%[Trần Nhân Kiệt]%[0H2K2-1]
Cho tam giác $ABC$ biết $AB=7$, $BC=8$ và góc $\widehat{B}=120^\circ$. Điểm $M$ thuộc cạnh $AC$  thỏa mãn $7AM-4MC=0.$ Tính độ dài đoạn thẳng $BM.$
\loigiai{
\immini
{
Vì $M$ thuộc cạnh $AC$ và $7AM-4MC=0$ nên $AM=\dfrac{4}{11}AC.$\\
Ta có $\vec{BM}=\vec{BA}+\vec{AM}=\vec{BA}+\dfrac{4}{11}\vec{AC}=\vec{BA}+\dfrac{4}{11}\left(\vec{BC}-\vec{BA}\right)$\\
$\Rightarrow \vec{BM}=\dfrac{7}{11}\vec{BA}+\dfrac{4}{11}\vec{BC}$\\
$\Rightarrow BM^2=\left(\dfrac{7}{11}\vec{BA}+\dfrac{4}{11}\vec{BC}\right)^2$\\
$\Rightarrow BM^2=\dfrac{49}{121}BA^2+\dfrac{16}{121}BC^2+\dfrac{56}{121}\vec{BA}\cdot \vec{BC}$\\
}
{
\begin{tikzpicture}
\tkzDefPoints{0/0/B,3/0/C,-1/2/A}
\tkzDefPointBy[homothety = center A ratio 4/11](C)
\tkzGetPoint{M}
\tkzDrawPolygon(A,B,C)
\tkzDrawSegments(B,M)
\tkzDrawPoints(A,B,C,M)
\tkzLabelPoint[above](A){\footnotesize $A$}
\tkzLabelPoint[below](B){\footnotesize $B$}
\tkzLabelPoint[below](C){\footnotesize $C$}
\tkzLabelPoint[right](M){\footnotesize $M$}
\tkzMarkAngles[size=.7,opacity=.8](C,B,A)

\tkzLabelSegment[below](B,C){ $8$}

\end{tikzpicture}
}
\noindent $\Rightarrow BM^2=\dfrac{49}{121}\cdot 7^2+\dfrac{16}{121}\cdot 8^2+\dfrac{56}{121}\cdot 7\cdot 8 \cdot \cos 120^\circ=\dfrac{4993}{121}\Rightarrow BM=\dfrac{\sqrt{4993}}{11}.$
}
\end{bt}


\begin{bt}%[Đề HK1, Trung Phú, Hồ Chí Minh 17-18]%[Trần Nhân Kiệt]%[0D4K1-2]
Cho số thực $x\ge 3.$ Chứng minh rằng $x+\dfrac{1}{x}\ge \dfrac{10}{3}.$
\loigiai{
Áp dụng bất đẳng thức {\bf Cô-si} với $2$ số dương là $\dfrac{x}{9}$ và $\dfrac{1}{x}.$
\begin{align*}
\dfrac{x}{9}+\dfrac{1}{x}&\ge 2\sqrt{\dfrac{x}{9}\cdot \dfrac{1}{x}}\\
\Rightarrow \dfrac{x}{9}+\dfrac{1}{x}&\ge\dfrac{2}{3}.
\end{align*}
Ta có $$x+\dfrac{1}{x}=\dfrac{8x}{9}+\left(\dfrac{x}{9}+\dfrac{1}{x}\right)\ge \dfrac{8}{9}\cdot 3+\dfrac{2}{3}=\dfrac{10}{3}.$$
Dấu ``$=$'' xảy ra khi $x=3.$
}
\end{bt}
