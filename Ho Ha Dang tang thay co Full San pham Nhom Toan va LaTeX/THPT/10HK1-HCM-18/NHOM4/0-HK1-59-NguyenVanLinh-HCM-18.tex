\begin{bt}%[Đề HK1, Nguyễn Văn Linh, Hồ Chí Minh 17-18]%[Nguyễn Sỹ]%[0D1Y1-2]%[0D1B1-2]
Tìm tập xác định của các hàm số sau
\begin{multicols}{2}
\begin{enumerate}
\item $y=\dfrac{1-2x}{2x^2-3x-9}$.
\item $y=\dfrac{2x^3}{\sqrt{2x-1}}-3\sqrt{x+2}$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item Hàm số xác định $\Leftrightarrow 2x^2-3x-9 \neq 0 \Leftrightarrow \heva {& x\neq 3\\&x\neq -\dfrac{3}{2}.}$\\
Vậy tập xác định của hàm số là $\mathscr{D}= \mathbb{R} \setminus \left \{ -\dfrac{3}{2};3\right \}.$
\item Hàm số xác định $\Leftrightarrow \heva {&2x-1>0\\&x+2\geq 0} \Leftrightarrow x>\dfrac{1}{2}$. \\
Vậy tập xác định của hàm số là $\mathscr{D}= \left( \dfrac{1}{2};+\infty \right)$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Văn Linh, Hồ Chí Minh 17-18]%[Nguyễn Sỹ]%[0D3B2-4]
Giải các phương trình sau
\begin{enumerate}
\item $\sqrt{2x^2-x+8}=x+2$.
\item $\sqrt{x+3}-2\sqrt{3x-1}=0$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có
\begin{eqnarray*}
 \sqrt{2x^2-x+8}=x+2 &\Leftrightarrow &\heva {&x+2\geq 0\\& 2x^2-x+8=(x+2)^2} \\
 &\Leftrightarrow & \heva {&x\geq-2\\& x^2-5x+4=0}\\
 & \Leftrightarrow & \heva {&x\geq -2\\ & \hoac {& x=1\\&x=4}} \Leftrightarrow \hoac {&x=1\\&x=4.}
\end{eqnarray*}
Vậy phương trình có hai nghiệm phân biệt là $x=1$ và $x=4$.
\item Ta có phương trình:
\begin{eqnarray*}
 & &\sqrt{x+3}-2\sqrt{3x-1}=0\\
&\Leftrightarrow & \sqrt{x+3} =2\sqrt{3x-1}\\
&\Leftrightarrow & \heva {& x+3\geq 0\\&x+3 =4(3x-1)}\\
&\Leftrightarrow & \heva {&x\geq -3\\&x=\dfrac{7}{11}} \Leftrightarrow x=\dfrac{7}{11}.
\end{eqnarray*}
Vậy phương trình có nghiệm duy nhất $x=\dfrac{7}{11}$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Văn Linh, Hồ Chí Minh 17-18]%[Nguyễn Sỹ]%[0D3B2-3]
	Giải phương trình $\dfrac{x+1}{x-1}+\dfrac{2-x}{x+3}=3$.
	\loigiai{
 Xét phương trình $\dfrac{x+1}{x-1}+\dfrac{2-x}{x+3}=3$.\\
Điều kiện xác định $\heva {&x\neq 1\\&x \neq -3.}$\\
Phương trình đã cho tương đương với 
\begin{eqnarray*}
	& & (x+1)(x+3)+(2-x)(x-1)=3(x-1)(x+3)\\
	&\Leftrightarrow & 3x^2-x-10=0\\
	&\Leftrightarrow & \hoac {&x=2\\& x= -\dfrac{5}{3} }  \,\, \text{ (thỏa mãn)}.
\end{eqnarray*}
Vậy phương trình có hai nghiệm phân biệt là $x=2$; $x=-\dfrac{5}{3}$.	
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Văn Linh, Hồ Chí Minh 17-18]%[Nguyễn Sỹ]%[0D3K2-5]
	Cho phương trình $x^2-(2m+1)x+m^2-m+1=0 \quad (1)$, ($m$ là tham số).
	\begin{enumerate}
		\item Tìm $m$ để phương trình $(1)$ có nghiệm kép. Tính nghiệm kép đó.
		\item  Xác định $m$ để phương trình $(1)$ có hai nghiệm phân biệt $x_1, x_2$ thỏa mãn $$x_1+x_2-3x_1x_2+2=0.$$
	\end{enumerate}
\loigiai {
\begin{enumerate}
	\item Xét phương trình bậc hai $x^2-(2m+1)x+m^2-m+1=0 \quad (1)$, ta có $\Delta = 8m-3$.\\
	Phương trình đã cho có nghiệm kép $\Leftrightarrow \Delta =0 \Leftrightarrow 8m-3=0 \Leftrightarrow m=\dfrac{3}{8}$.\\
	Với $m=\dfrac{3}{8}$, phương trình đã cho có nghiệm kép là $x_1=x_2= -\dfrac{b}{2a}=\dfrac{2m+1}{2}=\dfrac{7}{8}$.
	\item Phương trình đã cho có hai nghiệm phân biệt $x_1, x_2 \Leftrightarrow \Delta >0 \Leftrightarrow m> \dfrac{3}{8}.$\\
	Với $m> \dfrac{3}{8}$, áp dụng định lý Vi-et, ta có $ \heva{& x_1+x_2=2m+1\\&x_1x_2=m^2-m+1.}$ \\
	Ta có 
	\begin{eqnarray*}
	x_1+x_2-3x_1x_2+2=0 &\Leftrightarrow &(2m+1)-3(m^2-m+1)+2=0 \\
	&\Leftrightarrow &-3m^2+5m=0 \\
	&\Leftrightarrow & \hoac {&m=0 \,\,\text{(loại)}\\&m=\dfrac{5}{3}\,\,\text{(thỏa mãn).}}
		\end{eqnarray*}
	Vậy $m=\dfrac{5}{3}$ là giá trị cần tìm.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Văn Linh, Hồ Chí Minh 17-18]%[Nguyễn Sỹ]%[0H2B2-1]%[0H1B4-3]
Trong mặt phẳng tọa độ $Oxy$, cho tam giác $ABC$ có $A(1;3), B(5;1), C(9;-5)$.
\begin{enumerate}
	\item  Gọi $I$ là trung điểm của đoạn $AC$. Tính $\vec{BI}\cdot \vec{AB}$.
	\item  Tính chu vi của $\triangle ABC$.
	\item  Tìm tọa độ điểm $D$ sao cho $\vec{AD}= 2 \cdot \vec{BC}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Gọi $I$ là trung điểm của $AC$, ta có $I(5;-1).$\\
	Ta có $\heva {& \vec{BI}=(0;-2)\\&\vec{AB}=(4;-2)} \Rightarrow \vec{BI}\cdot \vec{AB}= 0\cdot 4+(-2)\cdot (-2)=4.$
	\item Ta có 
	\begin{eqnarray*}
	\vec{AB}=(4;-2)&\Rightarrow & AB=2\sqrt{5}\\
	\vec{AC}=(8;-8)&\Rightarrow & AC=8\sqrt{2}\\
	\vec{BC}=(4;-6)&\Rightarrow & BC =2\sqrt{13}.
	\end{eqnarray*}
	Do đó, chu vi $\triangle ABC $ bằng $2\sqrt{5}+8\sqrt{2}+2\sqrt{13}.$
	\item Gọi tọa độ điểm $D(x;y)$, ta có $\vec{AD}=(x-1;y-3)$ và $\vec{BC}=(4;-6).$\\
	Do đó $\vec{AD}=2\vec{BC} \Leftrightarrow \heva {&x-1=8\\&y-3=-12} \Leftrightarrow \heva {&x=9\\&y=-9.}$\\
	Vậy $D(9;-9)$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Văn Linh, Hồ Chí Minh 17-18]%[Nguyễn Sỹ]%[0D3K2-6]
	Cho phương trình $(m+1)x^2-2(m+1)x+m+3=0$, (với $m$ là tham số). Tìm $m$ để phương trình đã cho vô nghiệm.
	\loigiai {
\begin{itemize}
	\item Với $m=-1$ phương trình trở thành $0\cdot x = -2$, do đó phương trình đã cho vô nghiệm.
	\item Với $m\neq -1$, phương trình đã cho là phương trình bậc hai, do đó phương trình đã cho vô nghiệm $\Leftrightarrow \Delta '<0 \Leftrightarrow (m+1)^2-(m+1)(m+3) <0 \Leftrightarrow -2(m+1)<0 \Leftrightarrow m>-1.$
\end{itemize}	
Vậy với $m \geq -1$ thì phương trình đã cho vô nghiệm.
}
\end{bt}
