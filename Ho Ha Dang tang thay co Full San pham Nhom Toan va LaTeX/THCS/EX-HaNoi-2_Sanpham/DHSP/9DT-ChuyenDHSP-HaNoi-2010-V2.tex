\begin{ex}%[Đề thi tuyển sinh 10 trường THPT chuyên ĐH Sư phạm Hà Nội năm học 2010 - 2011, vòng 2]%[9D5G5]
   Giả sử $a$ và $b$ là hai số dương khác nhau và thỏa mãn $a-b=\sqrt{1-b^2}-\sqrt{1-a^2}$. Chứng minh rằng $a^2+b^2=1$.
    \loigiai{
    $a-b=\sqrt{1-b^2}-\sqrt{1-a^2} \Leftrightarrow a+ \sqrt{1-a^2} = b + \sqrt{1-b^2} \Leftrightarrow a\sqrt{1-a^2} = b\sqrt{1-b^2}$\\
    $\Rightarrow a^2 - a^4 = b^2 - b^4 \Leftrightarrow a^4 - b^4 - (a^2 - b^2) = 0 \Leftrightarrow (a^2 - b^2) (a^2 + b^2 - 1) = 0.$\\
    Theo đề bài ta có $a\ne b$ nên $a^2 - b^2 \ne 0$, suy ra $a^2 + b^2 - 1 = 0$ hay $a^2 + b^2 = 1$.
    }
\end{ex}

\begin{ex}%[Đề thi tuyển sinh 10 trường THPT chuyên ĐH Sư phạm Hà Nội năm học 2010 - 2011, vòng 2]%[8D1G3]
   Chứng minh rằng $\sqrt{2009^2+2009^2\times 2010^2+2010^2}$ là một số nguyên dương.
    \loigiai{
    Đặt $a = 2009$, ta có
    \begin{align*}
    & 2009^2+2009^2\times 2010^2+2010^2 = a^2 + a^2 (a+1)^2 + (a+1)^2 \\
    = & a^4 +2a^3 + 3 a^2 + 2a + 1 = a^4 + a^2 + 1 + 2a^3 + 2a^2 + 2a = (a^2+a+1)^2.
    \end{align*}
    Do đó $\sqrt{2009^2+2009^2\times 2010^2+2010^2} = \sqrt{(a^2 + a+ 1)^2} = a^2 + a + 1$ là một số nguyên dương. 
    }
\end{ex}

\begin{ex}%[Đề thi tuyển sinh 10 trường THPT chuyên ĐH Sư phạm Hà Nội năm học 2010 - 2011, vòng 2]%[9D4G6]
   Giả sử bốn số thực $a, b, c, d$ đôi một khác nhau và thỏa mãn đồng thời hai điều kiện sau
	\begin{enumerate}[i)]
	\item Phương trình $x^2-2cx-5d=0$ có hai nghiệm là $a$ và $b$.
	\item Phương trình $x^2-2ax-5b=0$ có hai nghiệm là $c$ và $d$.
	\end{enumerate}
	Chứng minh rằng
	\begin{enumerate}
	\item $a-c=c-b=d-a$.
	\item $a+b+c+d=30$.
	\end{enumerate}
    \loigiai{
    \begin{enumerate}
    \item Theo định lý Vi-ét, ta có $\begin{cases} a + b = 2c & (1) \\ ab = -5d & (2) \end{cases}$ và $\begin{cases} c + d = 2a & (3) \\ cd = -5b & (4). \end{cases}$\\
    Từ $(1)$ và $(3)$ ta suy ra $c-b=a-c=d-a$.
    \item Đặt $c-b=a-c=d-a = m$, suy ra $\begin{cases} c = a - m \\ b = c-m = a-2m \\ d = a+ m. \end{cases}$\\
    Từ đó ta có $a+b+c+d = 4a - 2m$.\\
    Từ $(2)$ suy ra $a (a - 2m) = -5 (a+m) \Leftrightarrow a^2 - 2am = -5a -5m$ $(5)$.\\
    Từ $(4)$ suy ra $(a-m)(a+m) = -5(a-2m) \Leftrightarrow a^2 - m^2 = -5a + 10m$ $(6)$.\\
    Từ $(5)$ và $(6)$ suy ra $m^2 - 2am = - 15m$. Theo giả thiết $a\ne c$ nên $m \ne 0$, suy ra $m - 2a = -15$. Từ đó ta có $a+b+c+d = 30$.
   	\end{enumerate}    
    }
\end{ex}

\begin{ex}%[Đề thi tuyển sinh 10 trường THPT chuyên ĐH Sư phạm Hà Nội năm học 2010 - 2011, vòng 2]%[9D5G5]
   Giả sử $m$ và $n$ là những số nguyên dương với $n>1$. Đặt $S=m^2n^2-4m+4n$. Chứng minh rằng
	\begin{enumerate}
	\item Nếu $m>n$ thì $(mn^2-2)^2<n^2S<m^2n^4$.
	\item Nếu $S$ là số chính phương thì $m=n$.
	\end{enumerate}
    \loigiai{
    \begin{enumerate}
    \item $(mn^2-2)^2<n^2S \Leftrightarrow m^2 n^4 -4mn^2 + 4 < m^2 n^4 -4mn^2 + 4n^3 \Leftrightarrow n^3 > 1 \Leftrightarrow n > 1$ (đúng theo giả thiết).\\
    $n^2S<m^2n^4 \Leftrightarrow m^2 n^4 -4mn^2 +4n^3 <m^2 n^4 \Leftrightarrow m>n$ (đúng theo giả thiết).
    \item Giả sử ngược lại $m \ne n$, xét hai trường hợp\\
    {\bf TH1:} $m>n$, theo ý $(1)$ và do $S$ chính phương nên suy ra \\
    	$n^2 S = (mn^2-1)^2 \Rightarrow m^2n^4 -4mn^2 +4n^3 =m^2n^4 -2mn^2 + 1 \Rightarrow 4n^3 = 2mn^2 + 1$ (\textbf{Sai} vì vế trái là số chẵn và vế phải là số lẻ với $m,n$ nguyên dương).\\
    {\bf TH2:} $m<n$, khi đó
    	\begin{itemize}
    	\item Nếu $m\ge 2$ thì $2mn + 1 \ge 4n + 1 > 4n$, suy ra 
    		$$ (mn)^2 < (mn)^2 - 4m + 4n < (mn)^2 + 4n < (mn)^2 + 2mn + 1 = (mn+1)^2$$
    	suy ra $(mn)^2 < S < (mn+1)^2$ (mâu thuẫn với $S$ là số chính phương).
    	\item Nếu $m = 1$ thì $S = n^2 + 4n - 4$ và $n \ge 2$. \\
    		Với $n > 2$ thì $n^2 + 2n + 1 < n^2 + 2n + 2n - 4 < n^2 + 4n + 4$ nên $(n+1)^2 < S < (n+2)^2$ (mâu thuẫn với $S$ là số chính phương).\\
    		Với $n = 2$ thì  $S=8$ không phải số chính phương. 
    	\end{itemize}
    Vậy nếu $S$ là số chính phương thì $m=n$.
    \end{enumerate}
    }
\end{ex}

   Cho tam giác $ABC$ với $AB>AC,AB>BC$. Trên cạnh $AB$ lấy các điểm $M$ và $N$ sao cho $BC=BM$ và $AC=AN$.
	\begin{enumerate}
	\item Chứng minh điểm $N$ nằm trong đoạn thẳng $BM$.
	\item Qua $M$ và $N$ kẻ $MP$ song song với $BC$ và $NQ$ song song với $CA$ $(P\in CA,Q\in CB)$. Chứng minh rằng $CP=CQ$.
	\item Cho $\widehat{ACB}=90^\circ,\widehat{CAB}=30^\circ$ và $AB=a$. Hãy tính diện tích của tam giác $MCN$ theo $a$.

    $MP \parallel BC$ nên suy ra $\widehat{BCM} = \widehat{PMC}$ (sole trong). Suy ra $\widehat{BMC} = \widehat{PMC}$. Vậy $MC$ là phân giác $\widehat{BMP}$. Gọi $P_1$ là điểm đối xứng của $P$ qua $MC$ thì $P_1 \in AB$.\\

\begin{ex}%[Đề thi tuyển sinh 10 trường THPT chuyên ĐH Sư phạm Hà Nội năm học 2010 - 2011, vòng 2]%[9D5G4]
   Trên một bảng đen ta viết ba số $\sqrt{2}, 2, \dfrac{1}{\sqrt{2}}$. Ta bắt đầu thực hiện một trò chơi như sau: Mỗi lần chơi ta xóa hai số nào đó trong ba số trên bảng, giả sử là $a$ và $b$, rồi viết vào hai vị trí vừa xóa hai số mới là $\displaystyle \frac{a+b}{\sqrt{2}}$ và $\displaystyle\frac{|a-b|}{\sqrt{2}}$, đồng thời giữ nguyên số còn lại. Như vậy sau mỗi lần chơi trên bảng luôn có ba số. Chứng minh rằng dù ta có chơi bao nhiêu lần đi chăng nữa thì trên bảng không thể có đồng thời ba số $\displaystyle\frac{1}{2\sqrt{2}};\sqrt{2};1+\sqrt{2}$.
    \loigiai{
    Do $\left( \dfrac{a+b}{\sqrt{2}} \right)^2 + \left( \dfrac{|a-b|}{\sqrt{2}} \right)^2 = a^2 + b^2$ nên tổng bình phương ba số không thay đổi sai mỗi lần chơi.\\
    Tổng bình phương ba số ban đầu là $\left( \sqrt{2} \right)^2  + 2^2 + \left( \dfrac{1}{\sqrt{2}} \right)^2 = \dfrac{13}{2}$.\\
    Tổng bình phương ba số cần biến đổi ra là $\left( \dfrac{1}{2\sqrt{2}} \right)^2 + \left( \sqrt{2} \right)^2 + \left( 1 + \sqrt{2} \right)^2 = \dfrac{41}{8} + 2\sqrt{2}$.\\
    Như thế ta không thể nào có được đồng thời ba số $\displaystyle\frac{1}{2\sqrt{2}};\sqrt{2};1+\sqrt{2}$.
    }
\end{ex}
