	{$2$}
	{$3$}
	{$0$}
	{\True $1$}
	\loigiai{

	}

\begin{bt}%[HK1, Nguyễn Sỹ Sách, Nghệ An 2017]%[1D1K3]
	Giải phương trình $2\sin\left(2x+\dfrac{\pi}{6}\right)+4\cos x +1=0$.
	\loigiai{
	Ta có $2\sin\left(2x+\dfrac{\pi}{6}\right)+4\cos x +1=0 \Leftrightarrow \sqrt{3}\sin 2x + \cos 2x +4\cos x+1=0$\\
	$\Leftrightarrow 2\sqrt{3}\sin x\cos x+2\cos^2 x+4\cos x=0 \Leftrightarrow 2\cos x (\sqrt{3}\sin x+\cos x+2)=0$\\
	$\Leftrightarrow\hoac{&\cos x=0 & (1)\\&\sqrt{3}\sin x+\cos x+2=0 & (2)}$.\\
	Ta có:
	\begin{itemize}
		\item $(1) \Leftrightarrow x=\dfrac{\pi}{2}+k\pi, k \in \mathbb{Z}$.
		\item $(2) \Leftrightarrow \dfrac{1}{2}\cdot\cos x +\dfrac{\sqrt{3}}{2}\cdot\sin x=-1\Leftrightarrow\cos\left(x-\dfrac{\pi}{3}\right)=-1 \Leftrightarrow x-\dfrac{\pi}{3}=\pi+k\pi \\\Leftrightarrow x=\dfrac{4\pi}{3}+k\pi , k \in \mathbb{Z}$.
	\end{itemize}
Vậy phương trình có nghiệm $x=\dfrac{\pi}{2}+k\pi$, $x=\dfrac{4\pi}{3}+k\pi , k \in \mathbb{Z}$.
	}
\end{bt}
