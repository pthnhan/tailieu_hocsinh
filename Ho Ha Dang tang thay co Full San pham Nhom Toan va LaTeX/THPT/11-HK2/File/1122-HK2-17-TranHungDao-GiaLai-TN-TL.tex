Cho hình chóp đều $S.ABC$ có cạnh đáy bằng $a$ và cạnh bên bằng $\dfrac{a\sqrt{13}}{6}$. Tính góc giữa hai mặt phẳng $(SBC)$ và $(ABC)$.

{$45^\circ$}
{$30^\circ$}
{$90^\circ$}
{\True $60^\circ$}

\begin{center}
\begin{tikzpicture}

\tkzDefShiftPoint[A](-45:2){B}

\tkzDefMidPoint(B,C)\tkzGetPoint{M}
\tkzDefMidPoint(B,A)\tkzGetPoint{N}

\tkzDefShiftPoint[O](90:4){S}

\tkzLabelPoints[above](S)
\tkzLabelPoints[below](O,B)
\tkzLabelPoints[below right](M)
\tkzLabelPoints[above right](C)

\tkzLabelPoints[below left](N)

\end{center}
Gọi $O$ là tâm của đáy và $M$ là trung điểm của cạnh $BC$.\\

Dễ thấy $AM=\dfrac{a\sqrt{3}}{2}$ suy ra $OM=\dfrac{a\sqrt{3}}{6}$.\\
Mặt khác tam giác $SCM$ vuông tại $M$ suy ra $SM^2=SC^2-CM^2=\dfrac{13a^2}{36}-\dfrac{a^2}{4}=\dfrac{4a^2}{36}$ ,từ đó suy ra $SM=\dfrac{a}{3}$.\\
Trong tam giác vuông $SOM$ ta có $\sin\widehat{SMO}=\dfrac{OM}{SM}=\dfrac{\sqrt{3}}{2}$ suy ra $\widehat{SMO}=60^\circ$.
}
\end{ex}

\tkzLabelPoints[above right](D)
\tkzLabelPoints[right](H)
\tkzLabelPoints[below,yshift=-6pt](O)
\tkzMarkRightAngles(S,A,B S,A,D B,O,C H,O,D O,H,S)
\end{tikzpicture}
\end{center}
Trong mặt phẳng $(SAC)$ kẻ $OH\perp SC$ tại $H$.\hfill $(1)$\\
Theo bài ra ta có $SA\perp(ABCD)$ suy ra $SA\perp BD$.\hfill $(2)$\\
Mặt khác tứ giác $ABCD$ là hình vuông suy ra $BD\perp AC$.\hfill $(3)$\\
Từ $(2)$, $(3)$ suy ra $BD\perp (SAC)\Rightarrow BD\perp OH$.\hfill $(4)$\\
Từ $(1)$, $(4)$ suy ra $OH$ là đoạn vuông góc chung của $BD$ và $SC$.\\
Xét tam giác vuông $SAC$ có $SC=\sqrt{SA^2+AC^2}=\sqrt{4a^2}=2a$.\\
Xét hai tam giác $SAC$ và $OHC$ có góc $\widehat{C}$ chung và $\widehat{SAC}=\widehat{OHC}=90^\circ$, suy ra hai tam giác $SAC$ và $OHC$ đồng dạng, suy ra $\dfrac{SA}{OH}=\dfrac{SC}{OC}\Rightarrow OH=\dfrac{SA\cdot OC}{SC}=\dfrac{a\sqrt{2}\cdot\dfrac{a\sqrt{2}}{2}}{2a}=\dfrac{a}{2}$.
}
\end{ex}

\begin{ex}%[HK2, THPT Trần Hưng Đạo, Gia Lai, 2016-2017]%[Nguyễn Tiến Thùy]%[1D4G2-3]
Tính giới hạn $L=\lim\limits_{x\rightarrow 1}\dfrac{\sqrt{4x+5}-\sqrt{x+8}}{\sqrt{3+x}-2}$.
\choice
{$L=1$}
{$L=0$}
{\True $L=2$}
{$L=+\infty$}
\loigiai{
Ta có 
\begin{align*}
L&=\lim\limits_{x\rightarrow 1}\dfrac{\left(\sqrt{4x+5}-\sqrt{x+8}\right)\left(\sqrt{4x+5}+\sqrt{x+8}\right)\left(\sqrt{3+x}+2\right)}{\left(\sqrt{3+x}-2\right)\left(\sqrt{3+x}+2\right)\left(\sqrt{4x+5}+\sqrt{x+8}\right)}\\
&=\lim\limits_{x\rightarrow 1}\dfrac{(3x-3)\left(\sqrt{3+x}+2\right)}{(x-1)\left(\sqrt{4x+5}+\sqrt{x+8}\right)}\\
&=\lim\limits_{x\rightarrow 1}\dfrac{3\left(\sqrt{3+x}+2\right)}{\left(\sqrt{4x+5}+\sqrt{x+8}\right)}=2.
\end{align*}
}
\end{ex}

\begin{bt}%[HK2, THPT Trần Hưng Đạo, Gia Lai, 2016-2017]%[Nguyễn Tiến Thùy]%[1D4K3-3]
Xét tính liên tục của hàm số $f(x)=\heva{&\dfrac{x^2-4x+3}{x-3}& \mbox{ khi } x\ne 3\\ &2& \mbox{ khi }x=3}$ tại $x_0=3$.
\loigiai{
Ta có $f(3)=2$ và $\lim\limits_{x\rightarrow 3}f(x)=\lim\limits_{x\rightarrow 3}\dfrac{x^2-4x+3}{x-3}=\lim\limits_{x\rightarrow 3}\dfrac{(x-1)(x-3)}{x-3}=\lim\limits_{x\rightarrow 3}(x-1)=2$.\\
Từ đó ta có $f(3)=\lim\limits_{x\rightarrow 3}f(x)$, suy ra hàm số liên tục tại $x_0=3$.
}
\end{bt}

\begin{bt}%[HK2, THPT Trần Hưng Đạo, Gia Lai, 2016-2017]%[Nguyễn Tiến Thùy]%[1H3B3-2]%[1H3G5-4]
Cho hình chóp $S.ABC$ có đáy $ABC$ là tam giác vuông cân tại $B$, $AB=BC=2a$, $SA$ vuông góc với mặt phẳng $(ABC)$ và $SA=3a$.
\begin{enumerate}
\item Chứng minh rằng $BC\perp (SAB)$.
\item Tính khoảng cách giữa hai đường thẳng chéo nhau $SC$ và $AB$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzDefPoints{0/0/A}
\tkzDefShiftPoint[A](-45:2){B}
\tkzDefShiftPoint[A](0:4){C}
\tkzDefShiftPoint[A](90:3){S}
\coordinate (D) at ($(S)+(B)-(A)$);
\coordinate (H) at ($(C)!0.666!(D)$);

\tkzDrawPoints[fill=black](A,B,C,S,D,H)
\tkzDrawSegments[](S,A S,B S,C A,B B,C S,D D,B C,D B,H)
\tkzDrawSegments[dashed](A,C)

\tkzLabelPoints[left](A,D)
\tkzLabelPoints[below](B)
\tkzLabelPoints[below left](H)
\tkzLabelPoints[right](C)
\tkzLabelPoints[above](S)

\tkzMarkSegments[mark=||](A,B B,C)
\tkzMarkRightAngles(S,A,B S,A,C A,B,C B,H,C)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Ta có $SA\perp (ABC)$ suy ra $SA\perp BC$.\hfill $(1)$\\
Mặt khác $BC\perp AB$ (giả thiết).\hfill $(2)$\\
Từ $(1),(2)$ suy ra $BC\perp (SAB)$.
\item Dựng điểm $D$ sao cho tứ giác $ABDS$ là hình chữ nhật, suy ra $SD\parallel AB$ và $SD\perp BD$.\hfill $(3)$\\
Theo câu a) ta có $BC\perp (SAB)$ suy ra $BC\perp SD$.\hfill $(4)$\\
Từ $(3)$ suy ra $AB\parallel (SCD)$ suy ra $d(AB;SC)=d(AB;(SCD))=d(B;(SCD))$.\\
Trong mặt phẳng $(BCD)$ kẻ $BH\perp CD$ tại $H$.\hfill $(5)$\\
Từ $(3),(4)$ suy ra $SD\perp(BCD)\Rightarrow SD\perp BH$.\hfill $(6)$\\
Từ $(5)$ và $(6)$ suy ra $BH\perp(SCD)$, từ đó ta có $d(AB;SC)=BH$.\\
Trong tam giác vuông $BCD$ có 
$$\dfrac{1}{BH^2}=\dfrac{1}{BD^2}+\dfrac{1}{BC^2}$$
$$\Rightarrow BH=\sqrt{\dfrac{BD^2\cdot BC^2}{BD^2+BC^2}}=\sqrt{\dfrac{36a^4}{13a^2}}=\dfrac{6\sqrt{13}a}{13}.$$
\end{enumerate}
}
\end{bt}
