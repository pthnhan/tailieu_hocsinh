	hình trụ bằng bao nhiêu để diện tích toàn phần $(S_{tp})$ nhỏ nhất?
	Ta có $V_0=h\cdot \pi \cdot r^2 \Rightarrow h= \dfrac{V_0}{\pi r^2}$.\\
Mà $S_{tp}= 2\pi r h+2\pi r^2= 2\pi (rh+r^2)= 2\pi \left( \dfrac{V_0}{\pi r}+r^2\right).$\\

$$ S_{tp}=2 \pi \left( \dfrac{V_0}{2\pi r} + \dfrac{V_0}{2\pi r}+r^2\right) \geq 2\pi \cdot 3 \sqrt[3]{\dfrac{V_0^2}{4\pi}} = 3\cdot \sqrt[3]{2\pi V_0^2}.$$

		\tkzMarkAngle[size=0.7cm, label=$\alpha$](M,O,N)
		\tkzLabelSegment[left](N,C){$R$}
		\end{tikzpicture}

	Đặt trục $Ox$ như hình vẽ. Mặt phẳng vuông góc với trục $Ox$ cắt trục $Ox$ tạo thành hình chữ nhật $EFGH.$\\
	Ta có $OI=x\Rightarrow NI=R-x\Rightarrow FG=2IG=2\sqrt{NG^2-NI^2}=2\sqrt{R^2-(R-x)^2}.$\\
	Mặt khác $IJ=OI\tan \alpha=x\tan \alpha.$\\
	Như vậy $S(x)=S_{EFGH}=IJ\cdot FG=2\tan \alpha \cdot x\cdot \sqrt{R^2-(R-x)^2}.$\\ 
	Thể tích khối hình nêm là $V= \displaystyle\int\limits_0^R 2\tan \alpha \cdot x\cdot \sqrt{R^2-(R-x)^2} \mathrm{\,d}x.$\\
	Đặt $R-x=R\sin t\Rightarrow x=R-R\sin t$, $\mathrm{\,d}x=-R\cos t\mathrm{\,d}t.$\\
	Khi $x=0\Rightarrow t=\dfrac{\pi}{2}$, khi $x=R\Rightarrow t=0.$ Khi đó
	\begin{eqnarray*}
		V&=&2\tan \alpha \displaystyle\int\limits_{\frac{\pi}{2}}^0 (R-R\sin t) \cdot \sqrt{R^2-R^2\sin ^2 t}\cdot (-R\cos t) \mathrm{\,d}t\\
		&=&  2R^3\tan \alpha \displaystyle\int\limits_0^{\frac{\pi}{2}} (1-\sin t)\cdot \cos ^2 t\mathrm{\,d}t\\
		&=& 2R^3\tan \alpha \displaystyle\int\limits_0^{\frac{\pi}{2}} \dfrac{1+\cos 2t}{2}\mathrm{\,d}t + 2R^3\tan \alpha \displaystyle\int\limits_0^{\frac{\pi}{2}} \cos ^2 t\mathrm{\,d}(\cos t)\\
		&=&  2R^3\tan \alpha \left(\dfrac{t}{2}+\dfrac{\sin 2t}{4}+\dfrac{\cos t}{3}\right)\bigg|_0^{\frac{\pi}{2}}\\
		&=&(\dfrac{\pi}{2}-\dfrac{2}{3})R^3\tan \alpha.
	\end{eqnarray*}
\end{dang}

\begin{dang}{Thể tích hình nêm loại 2}
	Một hình trụ được cắt bởi mặt phẳng $(P)$ đi qua đường kính của đáy và tạo với đáy góc $\alpha$. Hình tạo thành là một hình nêm(xem hình minh họa bên dưới). Biết đáy của hình trụ có bán kính $R$. Tính thể tích hình nêm tạo thành là $V=\dfrac{2R^3}{3}\tan \alpha.$
	
	Ta chọn trục $Ox$ như hình vẽ. Thiết diện tại điểm có tọa độ $x$ và vuông góc với trục $Ox$ cắt hình nêm theo thiết diện là tam giác vuông $MNP$.\\
	Ta có $OM=x\Rightarrow MN=\sqrt{ON^2-OM^2}=\sqrt{R^2-x^2}$.\\
	Mặt khác dễ thấy rằng $\triangle MNP\backsim \triangle OAB\Rightarrow \widehat{PMN}=\widehat{BOA}=\alpha.$\\
	Trong $\triangle MNP$ có $NP=MN\tan \alpha = \sqrt{R^2-x^2}\cdot \tan \alpha .$\\
	Thể tích của hình nêm là
	\begin{eqnarray*}
		V&=& \displaystyle\int\limits_{-R}^{R} S(x) \mathrm{\,d} x\\
		&=&	\displaystyle\int\limits_{-R}^{R} \dfrac{1}{2}\cdot \sqrt{R^2-x^2}\cdot \sqrt{R^2-x^2} \tan \alpha \mathrm{\,d} x\\
		&=&\tan \alpha \displaystyle\int\limits_{0}^{R} (R^2-x^2) \mathrm{\,d} x \\
		&=&\tan \alpha \cdot \left(R^2x-\dfrac{x^3}{3}\right)\bigg|_0^R\\
		&=& \dfrac{2R^3}{3}\tan \alpha.
	\end{eqnarray*}
\end{dang}
