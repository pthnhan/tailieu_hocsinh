\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2016-2017, huyện Phú lộc, Thừa Thiên Huế]%[9D1Y1]%[9D1B8]%[9D1K2]
Cho biểu thức $A=\left(\dfrac{3x + \sqrt{9x} - 3}{x + \sqrt{x} - 2} + \dfrac{1}{\sqrt{x} - 1} + \dfrac{1}{\sqrt{x} + 2} - 2\right): \dfrac{1}{x - 1}$.
\begin{enumerate}
\item Tìm điều kiện của x để biểu thức $A$ có nghĩa. 
\item  Rút gọn biểu thức $A$. 
\item  Tìm giá trị của $x$ để $\dfrac{2}{A}$ là số tự nhiên. 
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Điều kiện: $\heva{& x\ge 0 \\ & x\ne 1.}$
\item $A=\left(\dfrac{3x + \sqrt{9x} - 3}{x + \sqrt{x} - 2} + \dfrac{1}{\sqrt{x} - 1} + \dfrac{1}{\sqrt{x} + 2} - 2\right): \dfrac{1}{x - 1} \\  
=\dfrac{x + 3\sqrt{x} + 2}{\left(\sqrt{x} - 1\right)\left(\sqrt{x} + 2\right)}\cdot (x - 1) \\  
=\dfrac{\left(\sqrt{x} + 1\right)\left(\sqrt{x} + 2\right)}{\left(\sqrt{x} - 1\right)\left(\sqrt{x} + 2\right)}\cdot \left(\sqrt{x} - 1\right)\left(\sqrt{x} + 1\right) \\  ={\left(\sqrt{x} + 1\right)}^2$. 
\item Với điều kiện: $\heva{& x\ge 0 \\ & x\ne 1.}$\\
Ta có:  $A =\left(\sqrt{x} + 1\right)^2$.\\
Vì A = $\left(\sqrt{x} + 1\right)^2 \ge 1$ với mọi $x \ge 0$ nên  $0 \le \dfrac{2}{\left(\sqrt{x} + 1\right)^2} \le 2$.\\
Do đó: $\dfrac{2}{A}=\dfrac{2}{\left(\sqrt{x} + 1\right)^2}\in \mathbb{N}$ khi $\left(\sqrt{x} + 1\right)^2=1$ hoặc $\left(\sqrt{x} + 1\right)^2=2$.\\
Mà $\sqrt{x} + 1 >0$ nên $\sqrt{x} + 1=1$  hoặc $\sqrt{x} + 1 = \sqrt{2}$.\\
Do đó: $x=0$ hoặc $x=\left(\sqrt{2} - 1\right)^2=3 - 2\sqrt{2}$.\\
Vậy $\dfrac{2}{A}$ là số tự nhiên khi $x=0$ hoặc $x=3 - 2\sqrt{2}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2016-2017, huyện Phú lộc, Thừa Thiên Huế]%[9D1G2]
Giải phương trình: $x^2 - 10x + 27=\sqrt{6 - x} + \sqrt{x - 4}$.
\loigiai{
Điều kiện:  $4 \le x \le 6$.
\begin{itemize}
\item $VT=x^2 - 10x + 27=(x - 5)^2 + 2\ge 2$, dấu $"="$ xảy ra $\Leftrightarrow x=5$.
\item $VP=\sqrt{6 - x} + \sqrt{x - 4}\le \sqrt{\left(1^2 + 1^2\right)\left[{\left(\sqrt{6 - x}\right)}^2 + {\left(\sqrt{x - 4}\right)}^2\right]}\Leftrightarrow VP\le 2$.\\
dấu $"="$ xảy ra $\Leftrightarrow \dfrac{1}{\sqrt{6 - x}}=\dfrac{1}{\sqrt{x - 1}}\Rightarrow 6 - x=x - 4\Leftrightarrow x=5$.\\
$VT=VP\Leftrightarrow x=5$ (TMĐK). Vậy nghiệm của phương trình là $x=5$.
\end{itemize}
}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2016-2017, huyện Phú lộc, Thừa Thiên Huế]%[9D5G5]
Tìm giá trị lớn nhất, giá trị nhỏ nhất của biểu thức: $A=\dfrac{x + 1}{x^2 + x + 1}$.
\loigiai{
Ta có: $x^2 + x + 1=\left(x + \dfrac{1}{2}\right)^2 + \dfrac{3}{4}>0, \forall x\in \mathbb{R}$.\\
$A=\dfrac{x + 1}{x^2 + x + 1}=\dfrac{x^2 + x + 1 - x^2}{x^2 + x + 1}=1 - \dfrac{x^2}{x^2 + x + 1}\le 1$ (vì $\dfrac{x^2}{x^2 + x + 1}\ge 0, \forall x\in \mathbb{R}$).\\
Đẳng thức xảy ra khi $x = 0$, suy ra: $\max A = 1$ khi $x = 0$.\\
$A=\dfrac{x + 1}{x^2 + x + 1}\Rightarrow 3A=\dfrac{3x + 3}{x^2 + x + 1}=\dfrac{x^2 + 4x + 4 - \left(x^2 + x + 1\right)}{x^2 + x + 1}$.\\
$=\dfrac{{(x + 2)}^2}{x^2 + x + 1} - 1\ge - 1$ (vì $\dfrac{{(x + 2)}^2}{x^2 + x + 1}\ge 0, \forall x\in \mathbb{R}$).\\
Suy ra: $A\ge - \dfrac{1}{3}$, đẳng thức xảy ra khi $x + 2=0\Leftrightarrow x= - 2$.\\
Suy ra:  $\min A = - \dfrac{1}{3}$, khi $x= - 2$.
}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2016-2017, huyện Phú lộc, Thừa Thiên Huế]%[9D2Y3]%[9D2K3]
Cho hai đường thẳng: $y = x + 3 \quad (d_1)$; $y = 3x + 7 \quad (d_2)$.
\begin{enumerate}
\item Gọi $A$ và $B$ lần lượt là giao điểm của $(d_1)$ và $(d_2)$ với trục $Oy$. Tìm tọa độ trung điểm $I$ của đoạn thẳng $AB$. 
\item Gọi $J$ là giao điểm của $(d_1)$ và $(d_2)$. Tam giác $OIJ$ là tam giác gì? Tính diện tích của tam giác đó. 
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Tìm được $A(0; 3); B(0; 7)$. Suy ra $I(0; 5)$.
\item Hoành độ giao điểm $J$ của $(d_1)$ và $(d_2)$ là nghiệm của phương trình: $x + 3 = 3x + 7$\\
$\Rightarrow x=-2 \Rightarrow  y_J = 1 \Rightarrow J(- 2; 1)$.\\
Suy ra: $OI^2 = 0^2 + 5^2 = 25; OJ^2 = 2^2 + 1^2 = 5; IJ^2 = 2^2 + 4^2 = 20$.\\
$\Rightarrow OJ^2 + IJ^2 = OI^2 \Rightarrow$ tam giác $OIJ$ là tam giác vuông tại $J$.\\
$\Rightarrow S_{\triangle OIJ}=\dfrac{1}{2}OI\cdot OJ=\dfrac{1}{2}\cdot \sqrt{5}\cdot\sqrt{20}=5$ (đvdt).
\end{enumerate}
}
\end{ex}

Cho đường tròn $(O; R)$ đường kính $AB$. Gọi $M$ là điểm nằm giữa $A$ và $B$. Qua $M$ vẽ dây $CD$ vuông góc với $AB$, lấy điểm $E$ đối xứng với $A$ qua $M$.
\begin{enumerate}
\item Tứ giác $ACED$ là hình gì? Vì sao?
\item Gọi $H $ và $K$ lần lượt là hình chiếu của$ M $ trên $AC$ và $BC$. Chứng minh rằng: 
$$\dfrac{HM}{HK}\cdot \dfrac{MK}{MC}=\dfrac{CD}{4R}.$$
\item Gọi $C'$ là điểm đối xứng với $C$ qua $A$. Chứng minh rằng $C'$ nằm trên một đường tròn cố định khi $M$ di chuyển trên đường kính $AB$ ($M$ khác $A$ và $B$).

\begin{enumerate}
\item Vì  $CD \perp  AB \Rightarrow CM = MD$.\\

Mà  $AE \perp  CD \Rightarrow$ tứ giác $ACED$ là hình thoi.
\item Vì tam giác $ABC$ có $AB$ là đường kính $(O)$ nên $\triangle ABC$ vuông tại $C$, suy ra tứ giác $CHMK$ là hình chữ nhật.\\
Áp dụng hệ thức lượng vào các tam giác vuông ta có: 
 $$MH. AC = MA. MC \Rightarrow MH = \dfrac{MA. MC}{AC}.$$
Tương tự ta có: $MK = \dfrac{MB. MC}{BC}\Rightarrow MH. MK = \dfrac{MA. MB. MC^2}{AC. BC}$.\\
Mà $MA. MB = MC^2; AC. BC = MC. AB$ (do $\triangle ABC$ vuông tại $C$).\\
$\Rightarrow MH. MK = \dfrac{MC^2. MC^2}{MC. AB}=\dfrac{MC^3}{AB}\Rightarrow \dfrac{MH. MK}{MC^2}=\dfrac{MC}{AB}$.\\
Mà $MC = MK$ (do $CHMK$ là hình chữ nhật)\\
$\Rightarrow \dfrac{MH. MK}{HK. MC}=\dfrac{MC}{AB}=\dfrac{2MC}{2AB}=\dfrac{CD}{4R}.$\\

Do đó $O'C' = OC = R$ không đổi.\\
Suy ra $C'$ nằm trên đường tròn $(O'; R')$ cố định khi $M$ di chuyển trên đường kính $AB.$

Cho $a, b, c$ là ba số dương thỏa mãn: $a + b + c = 1$. Chứng minh rằng: 
$$\dfrac{c + ab}{a + b} + \dfrac{a + bc}{b + c} + \dfrac{b + ac}{a + c}\ge 2.$$
\loigiai{
Vì $a + b + c = 1$ nên
$$c + ab = c(a + b + c) + ab = (c + a)(c + b)$$
$$a + bc = a(a + b + c) + bc = (b + a)(b + c)$$
$$b + ac = b(a + b + c) + ac = (a + b)(a + c)$$
do đó BĐT cần chứng minh tương đương với: 
$\dfrac{(c + a)(c + b)}{a + b} + \dfrac{(b + a)(b + c)}{a + c} + \dfrac{(a + b)(a + c)}{b + c}\ge 2 \\ 
\Leftrightarrow \sqrt{\left(\dfrac{(c + a)(c + b)}{a + b}\right)^2} + \sqrt{\left(\dfrac{(b + a)(b + c)}{a + c}\right)^2} + \sqrt{\left(\dfrac{(a + b)(a + c)}{b + c}\right)^2}\ge 2.$\\
Mặt khác dễ thấy: $x^2 + y^2 + z^2\ge xy + yz + zx$, với mọi $x, y, z$ (*)\\
Áp dụng (*) ta có: $VT \ge b + c + a + b + c + a=2$.\\
Dấu 
$"="$ xảy ra khi và chỉ khi  $a = b =c =\dfrac{1}{3}$.
}
\end{ex}
