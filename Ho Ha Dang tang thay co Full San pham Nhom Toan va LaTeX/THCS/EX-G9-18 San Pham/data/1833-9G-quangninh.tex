\begin{ex}%[Đề thi HSG 9, tỉnh Quảng Ninh, năm 2018]%[Hưng Nguyễn, Dự án EX-9G-18]%[9D1K9]
	Rút gọn biểu thức $\dfrac{\sqrt[3]{2}+\sqrt{7+2\sqrt{10}}+\sqrt[3]{3\sqrt[3]{4}-3\sqrt[3]{2}-1}}{\sqrt{5}+\sqrt{2}+1}$.
	\loigiai{
	Ta có\\
	$\begin{aligned}
		\dfrac{\sqrt[3]{2}+\sqrt{7+2\sqrt{10}}+\sqrt[3]{3\sqrt[3]{4}-3\sqrt[3]{2}-1}}{\sqrt{5}+\sqrt{2}+1}& =\dfrac{\sqrt[3]{2}+\sqrt{\left( \sqrt{5}+\sqrt{2}\right) ^2}+\sqrt[3]{\left( 1-\sqrt[3]{2}\right) ^3}}{\sqrt{5}+\sqrt{2}+1}\\
		& =\dfrac{\sqrt[3]{2}+\sqrt{5}+\sqrt{2}+1-\sqrt[3]{2}}{\sqrt{5}+\sqrt{2}+1}=1.
	\end{aligned}$	
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Quảng Ninh, năm 2018]%[Hưng Nguyễn, Dự án EX-9G-18]%[9D0G7]
	Cho hai số dương $x,y$ thỏa mãn $x^3+y-x\sqrt[3]{y}=\dfrac{-1}{27}$. Tính giá trị biểu thức $\dfrac{x}{y}$.
	\loigiai{
	Áp dụng bất đẳng thức Cô-si, ta có
	$$x^3+y+\dfrac{1}{27}\ge 3\sqrt[3]{\dfrac{x^3y}{27}}=x\sqrt[3]{y}\Leftrightarrow x^3+y-x\sqrt[3]{y}\ge \dfrac{-1}{27}.$$
	Đẳng thức xảy ra khi và chỉ khi $x^3=y=\dfrac{1}{27}\Leftrightarrow \begin{cases}
		x=\dfrac{1}{3}\\ y=\dfrac{1}{27}.
	\end{cases}$\\
	Vậy $\dfrac{x}{y}=9$.
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Quảng Ninh, năm 2018]%[Hưng Nguyễn, Dự án EX-9G-18]%[9D0G4]
	Với mọi số nguyên $n$, chứng minh rằng $n(n+2)(73n^2-1)\;\vdots\;24$.
	\loigiai{
	Ta có $n(n+2)(73n^2-1)=72n^3(n+2)+(n-1)n(n+1)(n+2)$.\\
	Mà $72n^3(n+2)\;\vdots\;24\;\forall n$ nên ta chỉ cần chứng minh $(n-1)n(n+1)(n+2)\;\vdots\;24$.\\
	Thật vậy, trong bốn số nguyên liên tiếp $n-1,n,n+1,n+2$ có hai số chẵn liên tiếp có dạng $2k$ và $2k+2$ ($k\in \mathbb{Z}$).\\
	Do $2k(2k+2)=4k(k+1)\;\vdots\; 8$ (vì $k(k+1)\;\vdots\; 2$) nên $(n-1)n(n+1)(n+2)\;\vdots\;8\;\forall n$.\\
	Lại có $(n-1)n(n+1)(n+2)\;\vdots\;3\;\forall n$ và $\text{ƯCLN}(3,8)=1$.\\
	Suy ra $(n-1)n(n+1)(n+2)\;\vdots\;24\;\forall n$.\\
	Vậy ta có điều phải chứng minh.
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Quảng Ninh, năm 2018]%[Hưng Nguyễn, Dự án EX-9G-18]%[9D0G4]
	Tìm số tự nhiên $n$ để $2^4+2^7+2^n$ là số chính phương.
	\loigiai{
	Đặt $2^4+2^7+2^n=k^2$ với $k\in \mathbb{N}^*$.\\
	Ta có $16+128+2^n=k^2\Leftrightarrow 2^n=(k-12)(k+12).$\\
	Khi đó $\begin{cases}
	k+12=2^x\\ k-12=2^y
	\end{cases}$ với $x,y\in\mathbb{N}$, $x>y$ và $x+y=n$.\\
	Từ đó suy ra $2^x-2^y=24\Leftrightarrow 2^y(2^{x-y}-1)=24$.\\
	Mà $2^{x-y}-1$ là số lẻ nên $2^{x-y}-1\in \{1;3\}$.\\
	Thay vào ta có $\begin{cases}
	2^{x-y}-1=3\\ 2^y=8
	\end{cases}\Leftrightarrow \begin{cases}
		x=5\\ y=3.
	\end{cases}\Rightarrow n=8.$\\
	Khi đó $2^4+2^7+2^8=400=20^2$ thỏa mãn bài toán.
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Quảng Ninh, năm 2018]%[Hưng Nguyễn, Dự án EX-9G-18]%[9D0G2]
	Giải phương trình $\sqrt{2-3x}=-3x^2+7x-1$.
	\loigiai{
	ĐKXĐ: $x\le \dfrac{2}{3}$.\\
	$\begin{aligned}
		& \sqrt{2-3x}=-3x^2+7x-1\Leftrightarrow 3x^2-7x+2+\sqrt{2-3x}-1=0\\
		\Leftrightarrow & (3x-1)(x-2)+\dfrac{1-3x}{\sqrt{2-3x}+1}=0 \Leftrightarrow (3x-1)\left( x-2-\dfrac{1}{\sqrt{2-3x}+1}\right) \quad (1)
	\end{aligned}$\\
	Do $x\le \dfrac{2}{3}\Rightarrow x-2<0\Rightarrow x-2-\dfrac{1}{\sqrt{2-3x}+1}<0$.\\
	Vậy $(1)\Leftrightarrow 3x-1=0\Leftrightarrow x=\dfrac{1}{3}$ (thỏa mãn điều kiện).\\
	Do đó phương trình đã cho có nghiệm duy nhất $x=\dfrac{1}{3}$.
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Quảng Ninh, năm 2018]%[Hưng Nguyễn, Dự án EX-9G-18]%[9D0G2]
	Giải hệ phương trình $\begin{cases}
	3x-y^2-2\sqrt{(x-2)(y+1)}=-5\\ -2x+y^2+y=6.
	\end{cases}$
	\loigiai{
	ĐKXĐ: $(x-2)(y+1)\ge 0$.\\
	Cộng theo vế hai phương trình của hệ ta được
	$$x-2-2\sqrt{(x-2)(y+1)}+y+1=0\quad (1)$$
	\begin{itemize}
		\item Trường hợp 1: $x=2$.\\
		$(1)\Leftrightarrow y=-1$.\\
		Thử lại ta có $(x;y)=(2;-1)$ không phải là nghiệm của hệ.
		\item Trường hợp 2: $y=-1$.\\
		$(1)\Leftrightarrow x=2$.\\
		Đây cũng không phải là nghiệm của hệ.
		\item Trường hợp 3: $\begin{cases}
		x>2\\ y>-1.
		\end{cases}$\\
		$(1)\Leftrightarrow \left( \sqrt{x-2}-\sqrt{y+1}\right) ^2=0\Leftrightarrow \sqrt{x-2}=\sqrt{y+1}\Leftrightarrow x=y+3$.\\
		Thay $x=y+3$ vào phương trình thứ hai của hệ ta được
		$$y^2-y-12=0\Leftrightarrow (y-4)(y+3)=0\Leftrightarrow \left[\begin{aligned}
		& y=4 \text{ (thỏa mãn)}\\ & y=-3 \text{ (không thỏa mãn)}.
		\end{aligned}\right.$$
		Suy ra $x=7$ (thỏa mãn).
		\item Trường hợp 4: $\begin{cases}
		x<2\\ y<-1.
		\end{cases}$\\
		Khi đó $x-2-2\sqrt{(x-2)(y+1)}+y+1<0$ nên $(1)$ vô nghiệm.\\
		Vậy hệ phương trình đã cho có nghiệm duy nhất $(x;y)=(7;4)$.
	\end{itemize}	
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Quảng Ninh, năm 2018]%[Hưng Nguyễn, Dự án EX-9G-18]%[9H3K7]%[9H3K7]%[9H3K7]
	Cho đoạn thẳng $AB$, điểm $C$ nằm giữa hai điểm $A$ và $B$. Trên cùng một nửa mặt phẳng bờ là đường thẳng $AB$, vẽ nửa đường tròn đường kính $AB$ và nửa đường tròn đường kính $BC$. Lấy điểm $M$ thuộc nửa đường tròn đường kính $BC$ ($M\ne B;\ M\ne C$). Kẻ $MH$ vuông góc với $BC$ ($H\in BC$), đường cao $MH$ cắt nửa đường tròn đường kính $AB$ tại $K$. Hai đường thẳng $AK$ và $CM$ giao nhau tại $E$.
	\begin{enumerate}
		\item Chứng minh rằng $\widehat{HKB}=\widehat{CEB}$ và $BE^2=BC\cdot AB$.
		\item Từ $C$ kẻ $CN\perp AB$ ($N$ thuộc nửa đường tròn đường kính $AB$), đường thẳng $NK$ cắt $CE$ tại $P$. Chứng minh rằng $NP=PE$.
		\item Chứng minh rằng khi $NE$ là tiếp tuyến của nửa đường tròn đường kính $AB$ thì $NE=2NC$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item \immini
		{
			Ta có $\widehat{BME}=\widehat{BKE}=90^\circ$ nên $BMKE$ nội tiếp $\Rightarrow \widehat{BKM}=\widehat{BEM} \Rightarrow \widehat{HKB}=\widehat{CEB}$.\\
			Mà $\widehat{HKB}=\widehat{BAE}$ (do cùng phụ với $\widehat{AKH}$) nên $\widehat{CEB}=\widehat{EAB}$.\\
			Xét $\triangle BEC$ và $\triangle BAE$ có $\begin{cases}
			\widehat{B} \text{ chung}\\ \widehat{CEB}=\widehat{EAB}.
			\end{cases}$\\
			$\Rightarrow \triangle BEC\backsim\triangle BAE$ (g.g).\\
			$\Rightarrow \dfrac{BE}{BA}=\dfrac{BC}{BE}\Rightarrow BE^2=BC\cdot BA$.
			
		}
		{
			\begin{tikzpicture}[scale=1]
			\tkzInit[ymin=-1,ymax=6,xmin=-1,xmax=7]
			\tkzClip
			\tkzDefPoints{0/0/A,6/0/B,1/0/C}
			\tkzDefMidPoint(A,B)\tkzGetPoint{O}
			\tkzDefMidPoint(C,B)\tkzGetPoint{j}
			\tkzDrawArc[towards](O,B)(A)
			\tkzDrawArc[towards](j,B)(C)
			\tkzDefPointBy[rotation= center j angle 120](B) \tkzGetPoint{M}
			\tkzDefPointBy[projection=onto B--C](M)\tkzGetPoint{H}
			\tkzInterLC(M,H)(O,B) \tkzGetSecondPoint{K}
			\tkzInterLL(A,K)(M,C)\tkzGetPoint{E}
			\tkzDefLine[perpendicular=through C](B,A) \tkzGetPoint{d}
			\tkzInterLC(C,d)(O,B) \tkzGetSecondPoint{N}
			\tkzInterLL(N,K)(C,E)\tkzGetPoint{P}
			\tkzInterLL(B,P)(N,E)\tkzGetPoint{Q}
			
			\tkzDrawPoints(M,A,B,C,H,K,E,N,P,O,Q)
			\tkzDrawSegments(A,B K,H C,E A,E N,C N,P K,B E,B B,N N,E N,A B,Q O,N B,M)
			\tkzLabelPoints[above](E,N,K,Q) \tkzLabelPoints[below](C,H,A,B,O)
			\tkzLabelPoints[right](P,M)
			\tkzMarkRightAngles(N,C,B K,H,B)
			\end{tikzpicture}
		}
		\item Xét $\triangle ABN$ vuông tại $N$ có $NC\perp AB\Rightarrow BN^2=BC\cdot BA\Rightarrow BN=BE$.\\
		Vậy $\triangle BNE$ cân tại $B\Rightarrow \widehat{BNE}=\widehat{BEN}\quad (1)$.\\
		Vì $ABKN$ là tứ giác nội tiếp nên $\widehat{BAK}=\widehat{BNK}$.\\
		Mà theo câu a thì $\widehat{BAK}=\widehat{BEP}$ nên $\widehat{BNP}=\widehat{BEP}\quad (2)$.\\
		Từ $(1)$ và $(2)$ suy ra $\widehat{PNE}=\widehat{PEN}$.\\
		Vậy $\triangle PNE$ cân tại $P$ hay $NP=PE$.
		\item Gọi $O$ là trung điểm của $AB$, $Q$ là giao điểm của $BP$ và $NE$.\\
		Vì $\begin{cases}
		BN=BE\\ PN=PE
		\end{cases}$ nên $BP$ là trung trực của $EN\Rightarrow\begin{cases}
			BQ\perp EN\\ QN=QE=\dfrac{NE}{2}.
		\end{cases}$\\
		Mà $ON\perp NE$ (tính chất tiếp tuyến) $\Rightarrow ON\parallel BQ\Rightarrow \widehat{ONB}=\widehat{NBQ}$.\\
		Lại có $\widehat{ONB}=\widehat{OBN}$ (do $\triangle ONB$ cân tại $O$).\\
		$\Rightarrow \widehat{CBN}=\widehat{QBN}\Rightarrow \triangle CBN=\triangle QBN$ (cạnh huyền - góc nhọn).\\
		$\Rightarrow NC=QN=\dfrac{NE}{2}\Rightarrow NE=2NC$.
	\end{enumerate}	
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Quảng Ninh, năm 2018]%[Hưng Nguyễn, Dự án EX-9G-18]%[9D0G7]
	Cho $a,b$ là các số dương thỏa mãn $a+b+2ab=12$.\\
	Tìm giá trị nhỏ nhất của biểu thức $A=\dfrac{a^2+ab}{a+2b}+\dfrac{b^2+ab}{2a+b}$.
	\loigiai{
	Ta có\\
	$\begin{aligned}
	P & =(a+b)\left( \dfrac{a}{a+2b}+\dfrac{b}{2a+b}\right) =(a+b)\left( \dfrac{a^2}{a^2+2ab}+\dfrac{b^2}{2ab+b^2}\right) \\
	& \ge \dfrac{(a+b)^3}{a^2+b^2+2ab+2ab}\ge \dfrac{(a+b)^3}{(a+b)^2+\dfrac{(a+b)^2}{2}}=\dfrac{2(a+b)}{3}.
	\end{aligned}$\\
	Mặt khác $12=a+b+2ab\le a+b+\dfrac{(a+b)^2}{2}$.\\
	Suy ra $(a+b)^2+2(a+b)-24\ge 0\Leftrightarrow (a+b+6)(a+b-4)\ge 0\Leftrightarrow a+b\ge 4$ (do $a,b>0$).\\
	Do đó $P\ge \dfrac{8}{3}$. Đẳng thức xảy ra khi và chỉ khi $a=b=2$.\\
	Vậy giá trị nhỏ nhất của $P$ là $\dfrac{8}{3}$ khi $a=b=2$.
	}
\end{ex}
