\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2010-2011, Lâm Đồng]%[9D1K3]
	Rút gọn $A=\sqrt{127-48\sqrt{7}}-\sqrt{127+48\sqrt{7}}$
	\loigiai{
		Ta có \begin{eqnarray*}
		A&=&\sqrt{64-2.8.3\sqrt{7}+63}+\sqrt{64+2.8.3\sqrt{7}+63}\\
		 &=&\sqrt{(8-3\sqrt{7})^2}+\sqrt{(8+3\sqrt{7})^2}\\
		 &=&\left| 8-3\sqrt{7}\right| +\left| 8+3\sqrt{7}\right|\\
		 &=& 8-3\sqrt{7}+8+3\sqrt{7}=16
		\end{eqnarray*}
Vậy $A=16$.

\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2010-2011, Lâm Đồng]%[9D2K2]
	Cho hàm số $y=f\left(x\right)=\left(3m^2-7m+5\right)x-2011\quad\left(\ast\right)$. Chứng minh hàm số $\left(\ast\right)$ luôn đồng biến trên $\mathbb{R}$ với mọi $m$.
	\loigiai{
		Ta có
		\begin{eqnarray*}
		3m^2-7m+5=3\left(m^2-\dfrac{7}{3}m\right)+5&=&3\left( m^2-2.\dfrac{7}{6}m+\dfrac{49}{36}-\dfrac{49}{36}\right)+5\\&=&3\left(m-\dfrac{7}{6}\right)^2-\dfrac{49}{12}+5\\&=&3\left(m-\dfrac{7}{6}\right)^2+\dfrac{11}{12}>0 \text{ với mọi }m
		\end{eqnarray*}
		 Do đó hàm số $\left(\ast\right)$ luôn đồng biến trên $\mathbb{R}$ với mọi $m$.
		 }
\end{ex}
\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2010-2011, Lâm Đồng]%[8D1B3]
	Cho hai số dương $x,y$ thõa mãn điều kiện $3x+y-1=0$.\\Tìm giá trị nhỏ nhất của biểu thức $B=3x^2+y^2$.
	\loigiai{
		Ta có $3x+y-1=0\Rightarrow y=1-3x$. Do đó
		\begin{eqnarray*}
		B=3x^2+(1-3x)^2=12x^2-6x+1&=&12\left[ \left(x-\dfrac{1}{4}\right)^2+\dfrac{1}{48}\right]\\
		                          &=&12\left(x-\dfrac{1}{4}\right)^2+\dfrac{1}{4}\geq \dfrac{1}{4}  
		\end{eqnarray*}
Vậy GTNN của $B$ là $\dfrac{1}{4}$ khi $x=\dfrac{1}{4}$ và $y=\dfrac{1}{4}$.		 
		 }
\end{ex}
\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2010-2011, Lâm Đồng]%[9H3K4]
	Cho hai đường tròn $(O)$ và $(O')$ cắt nhau tại hai điểm $A$ và $B$. Trên đường thẳng $AB$ lấy điểm $M$ sao cho $A$ nằm giữa $M$ và $B$. Từ $M$ kẻ cát tuyến $MCD$ với đường tròn $(O)$ và tiếp tuyến $MT$ với đường tròn $(O')$($T$ là tiếp điểm). Chứng minh $MC.MD=MT^2$.
	\loigiai{\begin{center}
	\begin{tikzpicture}[line cap=round,line join=round,scale=1,>=stealth]   
\tkzDefPoints{0/0/O,4/0/O'}
\tkzDrawCircle[R](O,2cm)
\tkzDrawCircle[R](O',3cm)
\tkzInterCC[R](O,2cm)(O',3cm) \tkzGetPoints{A}{B}
\coordinate (M) at ($(A)!-1.4!(B)$);
\coordinate (D) at ($(O)+(-145:2)$);
\tkzInterLC[R](M,D)(O,2cm) \tkzGetPoints{}{C}
\tkzTangent[from with R=M](O',3cm) \tkzGetPoints{}{T}
\tkzDrawSegments(M,B M,D M,T O',T B,C D,A A,T B,T)
\tkzLabelPoints[left](A)
\tkzLabelPoints[above left](C)
\tkzLabelPoints[above](M,T)
\tkzLabelPoints[below](O,O',B,D)
\tkzDrawPoints[fill=white,size=4pt](O,O',A,B,M,D,C,T)
\tkzMarkAngles[size=.8,fill=yellow!30!blue,opacity=0.4](A,B,C A,D,C)
\tkzMarkAngles[size=.6,fill=yellow!30!blue,opacity=0.4](D,M,B)
\tkzMarkAngles[size=.8,fill=red!30!blue,opacity=0.4](M,T,A T,B,M)
\tkzMarkRightAngles[size=0.2](M,T,O')
\end{tikzpicture}
\end{center}
		Xét $\triangle MCB$ và $\triangle MAD$, ta có:\\
$\left.\begin{aligned}
&\widehat{DMB} \text{ chung }	\\ &\widehat{MDA}=\widehat{MBC}(=\dfrac{1}{2}\text{sđ}\wideparen{AC})\end{aligned}\right\rbrace \Rightarrow\triangle MCB\backsim\triangle MAD \Rightarrow \dfrac{MC}{MA}=\dfrac{MB}{MD}$ hay $MC.MD=MA.MB \quad(1)$\\
Xét $\triangle MTA$ và $\triangle MBT$, ta có:\\
$\left.\begin{aligned}
&\widehat{BMT} \text{ chung }	\\ &\widehat{MTA}=\widehat{MBT}(=\dfrac{1}{2}\text{sđ}\wideparen{AT})\end{aligned}\right\rbrace \Rightarrow\triangle MTA\backsim\triangle MBT \Rightarrow \dfrac{MT}{MB}=\dfrac{MA}{MT}$ hay $MT^2=MA.MB \quad(2)$\\
		Từ $(1)$ và $(2)\Rightarrow MC.MD=MT^2$(đccm)
		}
\end{ex}

\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2010-2011, Lâm Đồng]%[9D5G5]
	Giải hệ phương trình $\begin{cases} x+y+z=2\\2xy-z^2=4. \end{cases}$
	\loigiai{
		Ta có
		\begin{eqnarray*}
		\begin{cases} x+y+z=2\\2xy-z^2=4 \end{cases}\Leftrightarrow \begin{cases}z=2-x-y\\z^2=2xy-4 \end{cases}&\Leftrightarrow &\begin{cases} (2-x-y)^2=2xy-4\\z=2-x-y \end{cases}\\&\Leftrightarrow &\begin{cases} (x-2)^2+(y-2)^2=0\\z=2-x-y \end{cases}\Leftrightarrow \begin{cases} x=y=2\\z=-2. \end{cases}
\end{eqnarray*}		 }
\end{ex}
\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2010-2011, Lâm Đồng]%[9D5G5]
	Chứng minh $D=n(n+1)(n+2)(n+3)$ không phải là số chính phương với mọi $n\in \mathbf{N^{\ast}}$.
	\loigiai{
		\begin{eqnarray*}
		D&=&n(n+1)(n+2)(n+3)\\
		 &=&(n^2+3n)(n^2+3n+2)\\
		 &=&(n^2+3n)^2+2(n^2+3n)\\
		 &\Rightarrow &(n^2+3n)^2<D<(n^2+3n)^2+2(n^2+3n)+1\\
		 &\Rightarrow &(n^2+3n)^2<D<(n^2+3n+1)^2
		\end{eqnarray*}
		Do đó $D$ không phải là số chính phương vì $(n^2+3n)^2$ và $(n^2+3n+1)^2$ là hai số chính phương liên tiếp.
		 }
\end{ex}
\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2010-2011, Lâm Đồng]%[9D5G1]
	Cho hai số dương $a$ và $b$. Chứng minh $\dfrac{1}{a}+\dfrac{1}{b}\geq \dfrac{4}{a+b}$.
	\loigiai{Ta có
		\begin{eqnarray*}
		(a-b)^2\geq 0\Leftrightarrow a^2+b^2\geq 2ab\Leftrightarrow (a+b)^2\geq 4ab&\Leftrightarrow &\dfrac{a+b}{ab}\geq \dfrac{4}a+b (\text{Vì }(a+b)ab>0)\\&\Leftrightarrow &\dfrac{1}{a}+\dfrac{1}{b}\geq \dfrac{4}{a+b} (\text{đfcm})
		\end{eqnarray*}}
\end{ex}
\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2010-2011, Lâm Đồng]%[9D5G3]
	Tìm nghiệm tự nhiên của phương trình $2x^2-xy-y^2-8=0$.
	\loigiai{
		Ta có 
		\begin{eqnarray*}
		2x^2-xy-y^2-8=0
		\Leftrightarrow (2x+y)(x-y)=8
		&\Leftrightarrow &\begin{cases} 2x+y=8\\x-y=1 \end{cases} \text{ hoặc }\begin{cases} 2x+y=4\\x-y=2 \end{cases}\\
		&\Leftrightarrow &\begin{cases} x=3\\y=2 \end{cases} \text{ hoặc }\begin{cases} x=2\\y=0 \end{cases}
		\end{eqnarray*}
		Vậy các nghiệm tự nhiên của hệ là $(3;2)$ và $ (2;0)$.
		}
\end{ex}

\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2010-2011, Lâm Đồng]%[9H1G1]
	Cho tam giác $MPN$ cân tại $M (\widehat{M}<90^{\circ})$. Gọi $D$ là giao điểm các đường phân giác trong của tam giác $MPN$. Biết $DM=2\sqrt{5}$cm, $DN=3$cm. Tính độ dài $MN$.

		}
