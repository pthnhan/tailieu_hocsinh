\noindent\textbf{I. PHẦN TRẮC NGHIỆM}
\begin{ex}%[HK2,THPT Nguyễn Huệ, Ninh Thuận, 2016-2017]%[0D5B1]%[Đức Nguyễn]
Điểm thi học kỳ $2$ của một lớp $10$ được ghi trong bảng sau:	
	\begin{center}
		%\renewcommand\arraystretch{2}
		\tabcolsep = 5mm
		\begin{tabular}{|c|c|c|c|c|c|c|c|c|}
		\hline 
		8&6,5&7&9&8&8&5&5&4\\ 	\hline 
		7&4,5&10&7&7&6&6&6&6\\ 	\hline
		5&6&2,5&8&8&7&4&10&6\\ 	\hline
		9&6,5&9&7,5&8&6&9&3&7\\ 	\hline
		6&9&2&7&5,5&6&4&6&7\\ 	\hline
	\end{tabular}	
	\end{center}
Trong bảng trên, tần suất của điểm $7$ là
	\choice
	{$20\%$}
	{\True $17,8\%$}
	{$16\%$}
	{$15,6\%$}
	\loigiai{Tần số của điểm $7$ là $8$.\\
		Vậy tần suất của điểm $7$ là $f=\dfrac{8}{45}\cdot 100\%=17,8\%$.
	}
\end{ex}
\begin{ex}%[HK2,THPT Nguyễn Huệ, Ninh Thuận, 2016-2017]%[0D4B5]%[Đức Nguyễn]

	\choice
	{\True $\mathbb{R}$}
	{$\left[\dfrac{1}{3};\dfrac{2}{3}\right]$}
	{$\left(-\infty;2\right)$}
	{$\left[3;+\infty\right)$}

		Vậy tập nghiệm của bất phương trình đã cho là $\mathrm{S}=\mathbb{R}$.
	}
\end{ex}

\begin{ex}%[HK2,THPT Nguyễn Huệ, Ninh Thuận, 2016-2017]%[0H3B1]%[Đức Nguyễn]

\begin{ex}%[HK2,THPT Nguyễn Huệ, Ninh Thuận, 2016-2017]%[0H3B1]%[Đức Nguyễn]

	\choice
	{$\vv{a}=\left(6;-8\right)$}
	{$\vv{b}=\left(-3;4\right)$}
	{$\vv{c}=\left(3;-4\right)$}

		$\vv{x}$ không cùng phương với các véc-tơ còn lại nên không phải là véc-tơ chỉ phương của $d$.
	}
\end{ex}

\begin{ex}%[HK2,THPT Nguyễn Huệ, Ninh Thuận, 2016-2017]%[0H3B2]%[Đức Nguyễn]
	Trong các phương trình dưới đây, phương trình nào là phương trình của một đường tròn?
	\choice
	{$x^2+y^2-x-y+9=0$}
	{\True $x^2+y^2-x=0$}
	{$x^2+y^2-2xy-1=0$}
	{$x^2-y^2-2x+3y-1=0$}
	\loigiai{Phương trình $x^2+y^2-2xy-1=0$ có chứa hạng tử $xy$ nên không thể là phương trình của đường tròn.\\
		Phương trình $x^2-y^2-2x+3y-1=0$ có hệ số của $x^2$ và $y^2$ không bằng nhau nên không thể là phương trình đường tròn.\\
		Phương trình $x^2+y^2-x-y+9=0\Leftrightarrow x^2-x+\dfrac{1}{4}+y^2-y+\dfrac{1}{4}+\dfrac{34}{4}=0$\\
		$\Leftrightarrow \left(x-\dfrac{1}{2}\right)^2+\left(y-\dfrac{1}{2}\right)^2=-\dfrac{34}{4}$ nên không là phương trình đường tròn.\\
		$x^2+y^2-x=0\Leftrightarrow \left(x-\dfrac{1}{2}\right)^2+y^2=\dfrac{1}{4}$ là phương trình của đường tròn tâm $I\left(\dfrac{1}{2};0\right)$, bán kính $R=\dfrac{1}{2}$.
	}
\end{ex}

\begin{ex}%[HK2,THPT Nguyễn Huệ, Ninh Thuận, 2016-2017]%[0H3B1]%[Đức Nguyễn]
Viết phương trình tổng quát của đường thẳng đi qua $2$ điểm $A\left(3;-1\right)$ và $B\left(1;5\right)$.	
	\choice
	{$3x-y-2=0$}
	{\True $3x+y-8=0$}
	{$-x+3y+6=0$}
	{$-x+3y-14=0$}
	\loigiai{Một véc-tơ chỉ phương của đường thẳng $AB$ là $\vv{u}=\dfrac{1}{2}\vv{AB}=\dfrac{1}{2}\left(-2;6\right)=\left(-1;3\right)$.\\
		Một véc-tơ pháp tuyến của đường thẳng $AB$ là $\vv{n}=\left(3;1\right)$.\\
		Vậy phương trình của đường thẳng đi qua $A,B$ có dạng: \\
		$3(x-1)+(y-5)=0\Leftrightarrow 3x+y-8=0$.
	}

\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[HK2,THPT Nguyễn Huệ, Ninh Thuận, 2016-2017]%[0D4B3]%[0D4K6]%[Đức Nguyễn]
	Giải các bất phương trình sau 
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\left|5-2x\right|<x-1$.
			\item $\dfrac{\left(x^2-x-2\right)\left(3-x\right)}{-x^2-3x+4}<0$.
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item Ta có $\left|5-2x\right|<x-1\Leftrightarrow \heva{&x-1\geq 0\\&1-x<5-2x<x-1}$\\
			$\Leftrightarrow \heva{&x-1\geq 0\\&5-2x<x-1\\&5-2x>1-x}\Leftrightarrow \heva{&x\geq 1\\&x>2\\&x<4}\Leftrightarrow 2<x<4$.\\
			Vậy tập nghiệm của bất phương trình là $\mathrm{S}=\left(2;4\right)$.
			\item Ta có: \\
			$x^2-x-2=0\Leftrightarrow x=-1;x=2$\\
			$3-x=0\Leftrightarrow x=3$\\
			$-x^2-3x+4=0\Leftrightarrow x=1;x=-4$.\\
			Bảng xét dấu vế trái
				\begin{center}
				\begin{tikzpicture}
				\tkzTabInit[lgt=1.2,espcl=2.1]
				{$x$  /1.2, $VT$ /1.2}
				{$-\infty$ , $-4$,$-1$,$1$,$2$,$3$,$+\infty$}
				\tkzTabLine{,-,d,+,z,-,d,+,z,-,z,+,}
				\end{tikzpicture}
			\end{center}
			Vậy tập nghiệm của bất phương trình là $\mathrm{S}=\left(-\infty;-4\right)\cup\left(-1;1\right)\cup\left(2;3\right)$.
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[HK2,THPT Nguyễn Huệ, Ninh Thuận, 2016-2017]%[0D6B2]%[Đức Nguyễn]
Cho $\cos \alpha=\dfrac{2}{5}$ và $\dfrac{3\pi}{2}<\alpha<2\pi$. Hãy tính $\sin \alpha$, $\tan \alpha$, $\cot \alpha$.
	\loigiai{
		Ta có $\sin^2 \alpha=1-\cos^2 \alpha=1-\left(\dfrac{2}{5}\right)^2=\dfrac{21}{25}\Rightarrow \sin \alpha=\pm\dfrac{\sqrt{21}}{5}$.\\
		Vì $\dfrac{3\pi}{2}<\alpha<2\pi\Rightarrow \sin \alpha=-\dfrac{\sqrt{21}}{5}$.\\
		$\tan \alpha=\dfrac{\sin \alpha}{\cos \alpha}=-\dfrac{\sqrt{21}}{2}$.\\
		$\cot \alpha=\dfrac{1}{\tan \alpha}=-\dfrac{2\sqrt{21}}{21}$.
	}
\end{bt}
\begin{bt}%[HK2,THPT Nguyễn Huệ, Ninh Thuận, 2016-2017]%[0D6B3]%[Đức Nguyễn]
	Chứng minh hệ thức sau $\dfrac{1-\cos 2\alpha}{\sin 2\alpha}=\tan \alpha$.
	\loigiai{
	Ta có : $\dfrac{1-\cos 2\alpha}{\sin 2\alpha}=\dfrac{2\sin^2 \alpha}{2\sin \alpha\cdot\cos \alpha}=\dfrac{\sin \alpha}{\cos \alpha}=\tan \alpha$ (điều phải chứng minh).
	}
\end{bt}
\begin{bt}%[HK2,THPT Nguyễn Huệ, Ninh Thuận, 2016-2017]%[0H3B2]%[Đức Nguyễn]
Hãy xác định tâm và bán kính của đường tròn $(C):x^2+y^2-2x+4y-1=0$.
	\loigiai{
	Ta có : $x^2+y^2-2x+4y-1=0\Leftrightarrow (x^2-2x+1)+(y^2+4y+4)=6\Leftrightarrow \left(x-1\right)^2+\left(y+2\right)^2=6$.\\
	Vậy đường tròn đã cho có tâm là $I\left(1;-2\right)$, bán kính $R=\sqrt{6}$.
	}
\end{bt}
\begin{bt}%[HK2,THPT Nguyễn Huệ, Ninh Thuận, 2016-2017]%[0H3B1]%[0H3B2]%[Đức Nguyễn]
Trong mặt phẳng tọa độ $Oxy$, cho điểm $A\left(2;3\right)$ và đường thẳng $(\Delta):2x-y+3=0$.
\begin{enumerate}
	\item Viết phương trình đường thẳng $d$ đi qua $A$ và có hệ số góc $k=-2$.
	\item Viết phương trình đường tròn tâm $A$ và tiếp xúc với đường thẳng $\Delta$.
\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Phương trình của đường thẳng $d$ có dạng $y=-2(x-2)+3\Leftrightarrow y=-2x+7$.
			\item Gọi $(C)$ là đường tròn cần tìm. \\
			Vì $(C)$ tiếp xúc với đường thẳng $\Delta$ nên bán kính của đường tròn $(C)$ là: \\
			$R=d\left(A;\Delta\right)=\dfrac{\left|2\cdot 2-3+3\right|}{\sqrt{2^2+(-1)^2}}=\dfrac{4}{\sqrt{5}}$.\\
			Vậy phương trình của $(C)$ là $(x-2)^2+(y-3)^2=\dfrac{16}{5}$.
		\end{enumerate}
	}
\end{bt}
