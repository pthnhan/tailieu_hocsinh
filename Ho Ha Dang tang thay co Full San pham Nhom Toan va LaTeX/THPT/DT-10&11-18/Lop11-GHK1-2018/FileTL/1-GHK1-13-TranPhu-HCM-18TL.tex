\begin{bt}%[Thi GHK, THPT Trần Phú, HCM, 2018-2019]%[Nhật Thiện]%[1D1B1-1]
	Tìm tập xác định của hàm số $y=\sqrt{\dfrac{2+\sin x}{1-\cos x}}$.
	\loigiai{
	Hàm số xác định khi $\heva{&\dfrac{2+\sin x}{1-\cos x}\ge 0\\&1-\cos x\ne 0}\Leftrightarrow 1-\cos x\ne 0\Leftrightarrow \cos x\ne 1\Leftrightarrow x\ne k2\pi$, $k\in\mathbb{Z}$.\\
	Vậy tập xác định $\mathscr{D}=\mathbb{R}\setminus \left\{k2\pi,k\in\mathbb{Z}\right\}$.
}
\end{bt}
\begin{bt}%[Thi GHK, THPT Trần Phú, HCM, 2018-2019]%[Nhật Thiện]%[1D1B1-5]
	Tìm giá trị nhỏ nhất, giá trị lớn nhất của hàm số $y=\sqrt{4-2\sin^5(2x)}-8$.
	\loigiai{Ta có 
	\begin{eqnarray*}
		& &-1\le \sin (2x)\le 1, \forall x\\
		&\Leftrightarrow& -1\le \sin^5 (2x)\le 1, \forall x\\
		&\Leftrightarrow& 2\le 4-2\sin^5 (2x)\le 6, \forall x\\
		&\Leftrightarrow& \sqrt{2}-8\le \sqrt{4-2\sin^5 (2x)}-8\le \sqrt{6}-8.
	\end{eqnarray*}
Vậy $\min y=\sqrt{2}-8$ khi $\sin 2x=-1\Leftrightarrow x=-\dfrac{\pi}{4}+k\pi$, $k\in\mathbb{Z}$.\\
$\max y=\sqrt{6}-8$ khi $\sin 2x=1\Leftrightarrow x=\dfrac{\pi}{4}+k\pi$, $k\in\mathbb{Z}$.
}
\end{bt}
\begin{bt}%[Thi GHK, THPT Trần Phú, HCM, 2018-2019]%[Nhật Thiện]%[1D1B2-1]
	Giải phương trình $(\sin x+1)\left(\sin x-\sqrt{2}\right)=0$.
	\loigiai{
	\begin{eqnarray*}
		& &(\sin x+1)\left(\sin x-\sqrt{2}\right)=0\Leftrightarrow\hoac{&\sin x=-1\Leftrightarrow x=-\dfrac{\pi}{2}+k2\pi\\&\sin x=\sqrt{2}\text{ (phương trình vô nghiệm)}},k\in\mathbb{Z}.
	\end{eqnarray*}
}
\end{bt}
\begin{bt}%[Thi GHK, THPT Trần Phú, HCM, 2018-2019]%[Nhật Thiện]%[1D1K3-5]
	Giải phương trình $\sin 5x-1=\sin x-2\sin^2 \dfrac{3x}{2}$.
	\loigiai{
	\begin{eqnarray*}
		& &\sin 5x-1=\sin x-2\sin^2 \dfrac{3x}{2}\\
		&\Leftrightarrow &\sin 5x-\sin x=1-2\sin^2 \dfrac{3x}{2}\\
		&\Leftrightarrow& 2\cos 3x\sin 2x=\cos 3x\\
		&\Leftrightarrow &\cos 3x(2\sin 2x-1)=0\\
		&\Leftrightarrow&\hoac{&\cos 3x=0\\&\sin 2x=\dfrac{1}{2}=\sin \dfrac{\pi}{6}}\\
		&\Leftrightarrow& \hoac{&x=\dfrac{\pi}{6}+k\dfrac{\pi}{3}\\&x=\dfrac{\pi}{12}+k\pi\\&x=\dfrac{5\pi}{12}+k\pi}\qquad(k\in\mathbb{Z}).
	\end{eqnarray*}
}
\end{bt}
\begin{bt}%[Thi GHK, THPT Trần Phú, HCM, 2018-2019]%[Nhật Thiện]%[1D1K3-5]
	Giải phương trình $\cos x+\cos 5x=\cos 3x+\cos 7x$.
	\loigiai{
	\begin{eqnarray*}
		& &\cos x+\cos 5x=\cos 3x+\cos 7x\\
		&\Leftrightarrow &2\cos 3x\cos 2x=2\cos 5x\cos 2x\\
		&\Leftrightarrow &\hoac{&\cos 2x=0\\&\cos 3x=\cos 5x}\\
		&\Leftrightarrow & \hoac{&x=\dfrac{\pi}{4}+k\dfrac{\pi}{2}\\&x=k\pi\\&x=k\dfrac{\pi}{4}}\qquad(k\in\mathbb{Z}).
	\end{eqnarray*}
}
\end{bt}
\begin{bt}%[Thi GHK, THPT Trần Phú, HCM, 2018-2019]%[Nhật Thiện]%[1D1K3-6]
	Giải phương trình $\dfrac{(1-2\cos x)(1+\cos x)}{(1+2\cos x)\sin x}=1$.
	\loigiai{
	Điều kiện $\heva{&\sin x\ne 0\\&\cos x\ne -\dfrac{1}{2}}\Leftrightarrow \heva{&x\ne m\pi\\& x\ne \pm\dfrac{2\pi}{3}+n2\pi}$.\\
	Ta có phương trình trở thành \begin{eqnarray*}
		& &(1-2\cos x)(1+\cos x)=(1+2\cos x)\sin x\\
		&\Leftrightarrow& 1+\cos x-2\cos x-2\cos^2x=\sin x+2\sin x\cos x\\
		&\Leftrightarrow& 1-\cos x-2\cos^2x=\sin x+2\sin x\cos x\\
		&\Leftrightarrow& \sin 2x+\cos 2x+\sin x+\cos x=0\\
		&\Leftrightarrow& \sin \left(2x+\dfrac{\pi}{4}\right)+\sin \left(x+\dfrac{\pi}{4}\right)=0\\
		&\Leftrightarrow& \sin \left(2x+\dfrac{\pi}{4}\right)=\sin \left(-x-\dfrac{\pi}{4}\right)\\
		&\Leftrightarrow& \hoac{&2x+\dfrac{\pi}{4}=-x-\dfrac{\pi}{4}+k2\pi\\&2x+\dfrac{\pi}{4}=\pi+x+\dfrac{\pi}{4}+k2\pi}\\
		&\Leftrightarrow& \hoac{&x=-\dfrac{\pi}{6}+k\dfrac{2\pi}{3}&\text{(nhận)}\\& x=\pi+k2\pi &\text{(loại)}}, (k\in\mathbb{Z}).
	\end{eqnarray*}
}
\end{bt}
\begin{bt}%[Thi GHK, THPT Trần Phú, HCM, 2018-2019]%[Nhật Thiện]%[1D2B2-1]
	Trong vườn nhà bạn Lan có $10$ bông hồng vàng, $8$ bông hồng đỏ và $5$ bông hồng trắng (xem như các bông hoa khác nhau). Bạn Lan muốn chọn $5$ bông hoa để tặng mẹ. Hỏi bạn Lan có bao nhiêu cách chọn để số bông hoa được chọn có ít nhất $2$ màu.
	\loigiai{
	\begin{itemize}
		\item Chọn màu tùy ý có $\mathrm{C}_{23}^5$ cách.
		\item Trường hợp 1: Chọn toàn màu vàng có $\mathrm{C}_{10}^5$ cách.
		\item Trường hợp 2: Chọn toàn màu đỏ có $\mathrm{C}_8^5$ cách.
		\item Trường hợp 3: Chọn toàn màu trắng có $\mathrm{C}_5^5$ cách.
	\end{itemize}
\noindent Vậy để số bông hoa được chọn có ít nhất $2$ màu là 
$$\mathrm{C}_{23}^5-\mathrm{C}_{10}^5-\mathrm{C}_8^5-\mathrm{C}_5^5=33340\text{ cách}.$$
}
\end{bt}
\begin{bt}%[Thi GHK, THPT Trần Phú, HCM, 2018-2019]%[Nhật Thiện]%[1H1B7-2]
	Trong mặt phẳng $Oxy$, cho điểm $I(1;-1)$ và đường thẳng $\Delta'\colon x+2y-1=0$. Viết phương trình đường thẳng $\Delta$. Biết $\Delta'$ là ảnh của $\Delta$ qua phép vị tự tâm $I$ tỉ số $k=-\dfrac{1}{2}$.
	\loigiai{
	Gọi $M(x,y)\in \Delta$. Gọi $M'(x',y')$ là ảnh của $M$ qua phép vị tự tâm $I$ tỉ số $k$. Ta có \begin{eqnarray*}
		& &\mathrm{V}_{\left(I,-\dfrac{1}{2}\right)}(M)=M'(x',y')\Leftrightarrow \vec{IM'}=-\dfrac{1}{2}\vec{IM}\\
		&\Leftrightarrow& \heva{&x'-1=-\dfrac{1}{2}(x-1)\\&y'+1=-\dfrac{1}{2}(y+1)}\Leftrightarrow \heva{&x'=1-\dfrac{1}{2}(x-1)\\&y'=-1-\dfrac{1}{2}(y+1).}
	\end{eqnarray*}
Ta có $\mathrm{V}_{\left(I,-\dfrac{1}{2}\right)}(\Delta)=\Delta '$ suy ra $M'\in \Delta'$. Khi đó
\begin{eqnarray*}
	x'+2y'-1=0&\Leftrightarrow& 1-\dfrac{1}{2}(x-1)+2\left[-1-\dfrac{1}{2}(y+1)\right]-1=0\\
	&\Leftrightarrow& 2-(x-1)+2(-2-(y+1))-2=0\\
	&\Leftrightarrow& x+2y+5=0.
\end{eqnarray*}
Vậy $\Delta$ có phương trình là $x+2y+5=0$.
}
\end{bt}
