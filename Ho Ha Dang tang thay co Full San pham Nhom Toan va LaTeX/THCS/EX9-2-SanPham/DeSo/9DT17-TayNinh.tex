\begin{ex}%[Đề thi vào 10, Sở giáo dục Tây Ninh, 2017]%[9D4Y4]

\loigiai{
Ta có $\Delta=(-5)^2-4.1.(-14)=25+56=81=9^2>0\Rightarrow\sqrt{\Delta}=9.$ Phương trình đã cho có hai nghiệm phân biệt $x_1=\dfrac{5+9}{2}=7$ và $x_2=\dfrac{5-9}{2}=-2$.

\begin{ex}%[Đề thi vào 10, Sở giáo dục Tây Ninh, 2017]%[9D2B5]
	Tìm $ m $ để đường thẳng $(d):y=(2m-1)x+3$ song song với đường thẳng  $(d'):y=5x+6$.
	\loigiai{
Đường thẳng $(d)$ có $a=2m-1,b=3$ và $(d')$ có $a'=5$, $b'=6$. 

Khi đó, $(d)\parallel (d')\Leftrightarrow\heva{&a=a'\neq0\\&b\neq b'}\Leftrightarrow \heva{&2m-1=5\\&3\neq 6\,(\text{đúng})}\Leftrightarrow m=3$.	
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Tây Ninh, 2017]%[9D3B4]
	Tìm $ a $ và $ b $ biết hệ phương trình $\heva{
  & ax+y=1 \\ 
 & ax+by=-5 
}$ có một nghiệm là $ (2;-3) $.
\loigiai{
Xét hệ phương trình $\heva{
	& ax+y=1 \quad(1)\\ 
	& ax+by=-5 \quad(2)
}$. Thay nghiệm $(2;-3)$ vào $(1)$ ta được $2a-3=1\Leftrightarrow a=2$. Thay $a=2$ và nghiệm $(2;-3)$ vào $(2)$ ta được $2.2+b(-3)=-5\Leftrightarrow b=3$. Vậy ta có $a=2,b=3$.
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Tây Ninh, 2017]%[9H1B1]
	Cho tam giác $ ABC $ vuông tại $ A $ có đường cao $ AH $ ($ H $ thuộc cạnh $ BC $) biết 
            $ AB = a, BC = 2a $. Tính theo $ a $ độ dài $ AC $ và $ AH $.
\loigiai{
Theo Định lý Pytago ta có $AC^2=BC^2-AB^2=(2a)^2-a^2=3a^2\Rightarrow AC=a\sqrt{3}$. Áp dụng hệ thức lượng trong tam giác  $ABC$ vuông tại $A$, đường cao $AH$ thì $AH.BC=AB.AC\Leftrightarrow AH=\dfrac{AB.AC}{BC}=\dfrac{a.a\sqrt{3}}{2a}=\dfrac{a\sqrt{3}}{2}$.
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Tây Ninh, 2017]%[9D4K6]
	Tìm $ m $ để phương trình $x^2+x-m+2=0$ có hai nghiệm phân biệt $ x_1, x_2 $  thỏa $x_1^3+x_2^3+x_1^2x_2^2=17$.
	\loigiai{
Ta có $\Delta = 1^2-4(-m+2)=4m-7$ và hệ số $a=1\neq0$. Phương trình có hai nghiệm phân biệt khi và chỉ khi $\Delta>0\Leftrightarrow 4m-7>0\Leftrightarrow m>\dfrac{7}{4}$. Theo Định lý Viete, ta có $\heva{&S=x_1+x_2=-1\\&P=x_1x_2=-m+2}$. Theo yêu cầu bài toán:
\begin{eqnarray*}
x_1^3+x_2^3+x_1^2x_2^2=17&\Leftrightarrow& (x_1+x_2)^3-3x_1x_2(x_1+x_2)+x_1^2x_2^2=17\Leftrightarrow S^3-3PS+P^2=17\\&\Leftrightarrow&(-1)^3-3(-m+2)(-1)+(-m+2)^2=17
\\&\Leftrightarrow&  -1-3m+6+m^2-4m+4-17=0\Leftrightarrow m^2-7m-8 =0\\
&\Leftrightarrow&m=-1 \,(\text{loại}) \text{ hoặc } m=8 \,(\text{nhận}).
\end{eqnarray*}
Vậy $m=8$ thỏa yêu cầu bài toán.

}
\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Tây Ninh, 2017]%[9D3K6]
	Một mảnh đất hình chữ nhật có chiều dài hơn chiều rộng $ 6 $ m và độ dài đường chéo bằng  $\dfrac{\sqrt{65}}{4}$ lần chiều rộng. Tính diện tích của mảnh đất hình chữ nhật đã cho.
\loigiai{
Gọi $x,y$ (m) lần lượt là chiều dài và chiều rộng của mảnh đất (điều kiện: $x>y>0$). Khi đó, độ dài đường chéo được tính theo Định lý Pytago là $\sqrt{x^2+y^2}$. Theo đề bài ta có hệ phương trình 
\begin{eqnarray*}
\heva{& x-y=6\\&\sqrt{x^2+y^2}=\dfrac{\sqrt{65}}{4}.y}&\Leftrightarrow& \heva{& x-y=6\\& 16(x^2+y^2)=65y^2}\Leftrightarrow\heva{& x=y+6\\& 16(y+6)^2-49y^2=0}
\\&\Leftrightarrow&\heva{& x=y+6\\& -33y^2 +192y+576=0\quad(*)}
\end{eqnarray*}
Ta có $(*)\Leftrightarrow y=8\,\text{(nhận)}\text{ hoặc } y=-\dfrac{24}{11}\,\text{(loại)}$, khi đó, $x=y+6=14$.

Vậy chiều dài của mảnh đất là $14$ m và chiều rộng là $8$ m.
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Tây Ninh, 2017]%[9H3K7]
	Cho tam giác $ ABC $ có  $\widehat{BAC}$ tù. Trên $ BC $ lấy hai điểm $ D $ và $ E $, trên $ AB $ lấy điểm $ F $, trên $ AC $ lấy điểm $ K $ sao cho $ BD = BA $, $ CE = CA $, $ BE = BF $, $ CK = CD $. Chứng minh bốn điểm $ D $, $ E $, $ F $ và $ K $ cùng nằm trên một đường tròn.
	\loigiai{
		\\$\circ$ Ta có, $BA=BD,BE=BF$ nên $\dfrac{BF}{BA}=\dfrac{BE}{BD}$,  suy ra được $EF\parallel AD$, $\widehat{BAD}=\widehat{BDA}$, do đó, tứ giác $FEDA$ là hình thang cân và là tứ giác nội tiếp $ (1) $.
		\\$\circ$ Một cách tương tự, $CA=CE,CK=CD$ nên $\dfrac{CK}{CA}=\dfrac{CD}{CE}$,  suy ra được $KD\parallel AE$, $\widehat{CEA}=\widehat{CAE}$, do đó, tứ giác $AEDK$ là hình thang cân và là tứ giác nội tiếp $ (2) $.

\begin{ex}%[Đề thi vào 10, Sở giáo dục Tây Ninh, 2017]%[9H3G8]
	Cho tam giác $ ABC $ ($ AB < AC $), nội tiếp đường tròn đường kính $ BC $, có đường cao $ AH $ ($ H $ thuộc cạnh $ BC $), đường phân giác của góc $ A $ trong tam giác $ ABC $ cắt đường tròn đó tại $ K $ ($ K $ khác $ A $), Biết $\dfrac{AH}{HK}=\dfrac{\sqrt{15}}{5}$. Tính $\widehat{ACB}$.
\loigiai{
	\\$\circ$ Gọi $R$ là bán kính của đường tròn $(BC)$. Do $AK$ là phân giác góc $BAC$ nên $K$ là trung điểm của cung $\overset{\frown}{BC}$, do đó, $IK\perp BC$ với $I$ là trung điểm $BC$. Khi đó, $IB=IC=IK=R$. 
	\\$\circ$ Ta có $\dfrac{AH}{HK}=\dfrac{\sqrt{15}}{5}\Leftrightarrow \dfrac{AH^2}{HK^2}=\dfrac{3}{5}\Leftrightarrow\dfrac{HB.HC}{IH^2+IK^2}=\dfrac{3}{5}\Leftrightarrow 5(R+IH)(R-IH)=3(IH^2+R^2)\Leftrightarrow 5(R^2-IH^2)=3(IH^2+R^2)\Leftrightarrow 8IH^2=2R^2\Leftrightarrow IH=\dfrac{R}{4}$. 
	\\$\circ$ Từ đó $HC=\dfrac{3R}{4},AH=\dfrac{\sqrt{3}R}{4}$. Tam giác $AHC$ vuông tại $H$ có $\tan ACH=\dfrac{AH}{CH}=\sqrt{3}$ hay $\widehat{ACB}=\widehat{ACH}=30^{\circ}$. 
		
	\begin{center}
		\begin{tikzpicture}[scale=.9]
		\tkzInit[ymin=-3,ymax=3,xmin=-2.5,xmax=4.5]
		\tkzClip
		\tkzDefPoints{0/2/A,-1/0/B,4/0/C,0/0/H}
		\tkzDefMidPoint(B,C)\tkzGetPoint{I}
		\tkzDrawCircle[circum](A,B,C)
		\tkzDefLine[bisector](B,A,C) \tkzGetPoint{c}
		\tkzInterLL(B,C)(A,c)\tkzGetPoint{E}
		\tkzInterLC(A,c)(I,B)\tkzGetFirstPoint{K}
		\tkzDrawSegments(A,B B,C C,A A,H A,E B,K K,I A,K)
		\tkzMarkRightAngles(C,H,A B,A,C)
		\tkzDrawPoints(A,B,C,H,K,E,I)
		\tkzLabelPoints[above](A,I)
		\tkzLabelPoints[below](H,K)
		\tkzLabelPoints[right](C)
		\tkzLabelPoints[left](B)
		\tkzMarkAngles[arc=l,size=.5cm](B,A,K)
		\tkzMarkAngles[arc=l,size=.6cm](K,A,C)
		\end{tikzpicture}
	\end{center}
}
\end{ex}
