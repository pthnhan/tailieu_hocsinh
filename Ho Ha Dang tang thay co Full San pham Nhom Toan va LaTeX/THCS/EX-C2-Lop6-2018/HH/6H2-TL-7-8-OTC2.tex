\section{Đường tròn}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Đường tròn và hình tròn}
\begin{dn}
	Đường tròn tâm $O$, bán kính $R$ là hình gồm các điểm cách $O$ một khoảng bằng $R$, kí hiệu $(O;R).$
\end{dn}
\begin{dn}
	Hình tròn là hình gồm các điểm nằm trên đường tròn và các điểm nằm trong đường tròn đó.
\end{dn}
\subsubsection{Cung và dây cung}
\immini{
	Hai điểm $C,\ D$ của một đường tròn chia đường tròn thành hai \textit{cung}. Đoạn thẳng nối hai mút của một cung gọi là \textit{dây cung}.\medskip
	
	Dây cung đi qua tâm là \textit{đường kính}.\medskip
	
	Trong hình bên, đoạn thẳng $CD$ gọi là \textit{dây cung}; đoạn thẳng $AB$ gọi là \textit{đường kính}.
}{
	\begin{tikzpicture}[scale=0.8]
	\tkzInit[xmin=-1,xmax=4.5,ymin=-2.5,ymax=2.5]
	\tkzClip
	\tkzDefPoints{0/0/A,2/0/O,4/0/B,-5/0/M,5/2.1/N}
	\tkzDrawCircle(O,A)
	\tkzInterLC(M,N)(O,A)\tkzGetFirstPoint{C}
	\tkzInterLC(M,N)(O,A)\tkzGetSecondPoint{D}
	\tkzDrawPoints[fill=black](O,A,B,C,D)
	\tkzLabelPoints[left](A,C)
	\tkzLabelPoints[right](B)
	\tkzLabelPoints[above right](D)
	\tkzLabelPoints[below](O)
	\tkzDrawSegments[](A,B C,D)
	\end{tikzpicture}
}
\subsection{Các dạng toán}
\begin{dang}{Nhận biết vị trí của một điểm đối với đường tròn}
	\textbf{PHƯƠNG PHÁP GIẢI:}
	\begin{list}{•}{}
		\item Điểm $A$ thuộc đường tròn $(O;R)$ nếu $OA=R$.
		\item Điểm $A$ nằm trong đường tròn $(O;R)$ nếu $OA<R$.
		\item Điểm $A$ nằm ngoài đường tròn $(O;R)$ nếu $OA>R$.
	\end{list}
\end{dang}
\begin{vd}%[Lê Nguyễn Viết Tường, C2-L6-2018]%[6H2Y8]
	Vẽ đường tròn $(O;1{,}5\text{ cm})$. Lấy $3$ điểm $A,B,C$ sao cho $OA=1$ cm, $OB=1{,}5$ cm, $OC=2$ cm. Trong $3$ điểm $A,B,C$, điểm nào nằm trong đường tròn, ngoài đường tròn, điểm nào thuộc đường tròn?
	\loigiai{
		\immini{
			Vì $OA=1$ cm $<1{,}5$ cm nên điểm $A$ nằm trong đường tròn.\medskip
			
			$OB=1{,}5$ cm nên điểm $B$ nằm trên đường tròn.\medskip
			
			$OC=2$ cm $>1{,}5$ cm nên điểm $C$ nằm ngoài đường tròn.
		}{
			\begin{tikzpicture}[scale=1]
			\tkzDefPoints{0/0/O}
			\tkzDefShiftPoint[O](-20:1){A}
			\tkzDefShiftPoint[O](5:1.5){B}
			\tkzDefShiftPoint[O](30:2){C}
			\tkzDrawCircle(O,B)
			\tkzDrawPoints[fill=black](O,A,B,C)
			\tkzLabelPoints[below left](O)
			\tkzLabelPoints[right](B,C)
			\tkzLabelPoints[below](A)
			\tkzDrawSegments[](O,A O,B O,C)
			\end{tikzpicture}
		}
	}
\end{vd}
\begin{vd}%[Lê Nguyễn Viết Tường, C2-L6-2018]%[6H2B8]%[6H2K8]
	Cho đoạn thẳng $AB=3$ cm. Vẽ đường tròn $(A;2\text{ cm})$ cắt $AB$ tại điểm $M$. Vẽ đường tròn $(B;1{,}5\text{ cm})$ cắt $AB$ tại $N$. Hai đường tròn này cắt nhau tại $I, K$.
	\begin{listEX}[]
		\item Tính $AI,BK$.
		\item Chứng tỏ $N$ là trung điểm của $AB$.
	\end{listEX}
	\loigiai{
		\immini{
			\begin{listEX}[]
				\item Vì điểm $I$ thuộc đường tròn $(A;2\text{ cm})$\\ nên $AI=2$ cm.\medskip
				
				Vì điểm $K$ thuộc đường tròn $(B;1{,}5\text{ cm})$\\ 
				nên $BK=1{,}5$ cm.\medskip
				\item Vì điểm $N$ thuộc đường tròn $(B;1{,}5\text{ cm})$\\
				nên $BN=1{,}5$ cm.\medskip
				
				Vì $N$ nằm giữa $A$ và $B$ nên $AN+NB=AB$.\medskip
				
				Lại có $BN=1{,}5$ cm, $AB=3$ cm.\medskip
				
				Nên $AN=1{,}5$ cm.\\
			\end{listEX}
		}{
			\begin{tikzpicture}[scale=1.2]
			\tkzDefPoints{0/0/A,2/0/M,1.5/0/N,3/0/B}
			\tkzDrawCircle(A,M)
			\tkzDrawCircle(B,N)
			\tkzInterCC(A,M)(B,N)\tkzGetFirstPoint{I}
			\tkzInterCC(A,M)(B,N)\tkzGetSecondPoint{K}
			\tkzDrawPoints[fill=black](A,B,M,N,I,K)
			\tkzLabelPoints[above=.1](I)
			\tkzLabelPoints[below left](A,N)
			\tkzLabelPoints[below right](B)
			\tkzLabelPoints[above right](M)
			\tkzLabelPoints[below=.1](K)
			\tkzDrawSegments[](A,I B,K A,B)
			\end{tikzpicture}
		}
		Khi đó ta có $AN=NB$ do đó $N$ là trung điểm của $AB$.
	}
\end{vd}\medskip

\begin{dang}{Đếm số dây cung, số cung của đường tròn}
	\textbf{PHƯƠNG PHÁP GIẢI}\medskip
	
	Đoạn thẳng nối $2$ điểm phân biệt trên đường tròn tạo thành $1$ dây cung. Hai điểm này chia đường tròn thành $2$ phần, mỗi phần gọi là một cung.\medskip
	
	Trên đường tròn có $n$ điểm phần biệt. Nối mỗi cặp điểm ta được một dây cung thì số dây cung là $\dfrac{n\cdot (n-1)}{2}$ và số cung tạo thành là $n\cdot (n-1).$
\end{dang}
\setcounter{vd}{0}
\begin{vd}%[Lê Nguyễn Viết Tường, C2-L6-2018]%[6H2B8]%[6H2B8]
	Trên đường tròn $(O)$ lấy $9$ điểm phân biệt. Nối mỗi cặp điểm ta được một dây cung.
	\begin{listEX}[]
		\item Hỏi vẽ được bao nhiêu dây cung?
		\item Hỏi trên đường tròn có bao nhiêu cung được tạo thành?
	\end{listEX}
	\loigiai{
		\begin{listEX}[]
			\item Số dây cung là $\dfrac{9\cdot (9-1)}{2}=36$ (dây cung).
			\item Số cung trên đường tròn là $9\cdot (9-1)=72$ (cung).
		\end{listEX}
	}
\end{vd}
\begin{vd}%[Lê Nguyễn Viết Tường, C2-L6-2018]%[6H2K8]
	Trên đường tròn tâm $O$ ta lấy $n$ điểm. Số cung tròn tạo thành là $132$. Hỏi $n$ là số nào?
	\loigiai{
		Số cung tạo thành từ $n$ điểm là $n\cdot (n-1)$ nên $n\cdot (n-1)=132=12\cdot 11.$\\
		Suy ra $n=12.$
	}
\end{vd}

\section{Tam giác}
\subsection{Tóm tắt lí thuyết}
\immini{
	
	\begin{dn}
		Tam giác $ABC$ là hình gồm ba đoạn thẳng $AB,\ BC,\ CA$ khi $A,\ B,\ C$ không thẳng hàng.
	\end{dn}
	
	Tam giác $ABC$ kí hiệu là $\triangle ABC.$\medskip
	
	Ba điểm $A,B,C$ là ba \textit{đỉnh} của tam giác.\medskip
	
	Ba đoạn thẳng $AB,BC,CA$ là ba \textit{cạnh} của tam giác.\medskip
	
	Ba góc $\widehat{A},\widehat{B},\widehat{C}$ là ba \textit{góc} của tam giác.
	
}{
	\begin{tikzpicture}[scale=.8]
	\tkzDefPoints{0/0/B,2/3.5/A,5/0/C}
	\tkzDrawPoints[fill=black](A,B,C)
	\tkzLabelPoints[below left](B)
	\tkzLabelPoints[below right](C)
	\tkzLabelPoints[above](A)
	\tkzDrawSegments(A,B B,C C,A)
	\end{tikzpicture}
}
\subsection{Các dạng toán}
\begin{dang}{Nhận dạng tam giác và các yếu tố}
	\textbf{PHƯƠNG PHÁP GIẢI}\medskip
	
	Đối chiếu với định nghĩa của tam giác.
\end{dang}
\begin{vd}%[Lê Nguyễn Viết Tường, C2-L6-2018]%[6H2Y9]
	\immini{
		Cho hình vẽ bên. Hãy dùng kí hiệu ghi:
		\begin{listEX}
			\item Tên các tam giác khác nhau trong hình.
			\item Tên các góc trong mỗi tam giác.
			\item Những tam giác chung cạnh $AM$.
		\end{listEX}
	}{
		\begin{tikzpicture}
		\tkzDefPoints{0/0/B,0.75/3/A,2/0/M,5/0/C}
		\tkzDrawPoints[fill=black](A,B,C,M)
		\tkzDrawSegments(A,B B,C C,A A,M)
		\tkzLabelPoints[above](A)
		\tkzLabelPoints[below left](B)
		\tkzLabelPoints[below](M)
		\tkzLabelPoints[below right](C)
		\end{tikzpicture}
	}
	\loigiai{
		\begin{listEX}
			\item Tên các tam giác khác nhau trong hình là $\triangle ABC,\triangle ABM,\triangle ACM.$
			\item Tên các góc trong $\triangle ABC$ là $\widehat{ABC},\widehat{BCA},\widehat{CAB}.$\\
			Tên các góc trong $\triangle ABM$ là $\widehat{ABM},\widehat{BMA},\widehat{MAB}.$\\
			Tên các góc trong $\triangle ACM$ là $\widehat{ACM},\widehat{CMA},\widehat{MAC}.$
			\item Tam giác chung cạnh $AM$ là $\triangle ABM$ và $\triangle AMC$.
		\end{listEX}
	}

\begin{dang}{Vẽ tam giác}
	\textbf{PHƯƠNG PHÁP GIẢI}\medskip
	
	Để vẽ một tam giác biết độ dài ba cạnh, ta vẽ một cạnh trước. Sau đó vẽ hai cung tròn có tâm là hai đỉnh đã vẽ, hai bán kính lần lượt bằng độ dài hai cạnh còn lại. Giao điểm hai cung tròn là đỉnh thứ ba của tam giác.
\end{dang}

\begin{dang}{Tính số tam giác tạo thành}
	\textbf{PHƯƠNG PHÁP GIẢI}\medskip
	
	Cho $n$ điểm $A_1;\ A_2;\ldots;\ A_n$ theo thứ tự trên đường thẳng $xy$ và điểm $M$ nằm ngoài đường thẳng $xy$. Nối $M$ với $n$ điểm đó. Số tam giác đếm được là $\dfrac{n\cdot (n-1)}{2}$.
\end{dang}
\setcounter{vd}{0}
\begin{vd}%[Lê Nguyễn Viết Tường, C2-L6-2018]%[6H2B9]
	Trên đường thẳng $xy$ lấy $6$ điểm $A,\ B,\ C,\ D,\ E,\ F$ theo thứ tự đó. Lấy điểm $M$ nằm ngoài đường thẳng $xy$. Nối $M$ với các điểm $A,\ B,\ C,\ D,\ E,\ F$. Số tam giác tạo thành là bao nhiêu?
	\loigiai{
		Số tam giác nhận được đúng bằng số đoạn thẳng tạo thành từ $6$ điểm trên đường thẳng $xy$. \\
		Mà số đoạn thẳng lập được là $\dfrac{6\cdot(6-1)}{2}=15$.\\
		Suy ra số tam giác tạo thành là $15$ tam giác.
	}
\end{vd}\medskip

\begin{vd}%[Lê Nguyễn Viết Tường, C2-L6-2018]%[6H2K9]
	Cho $n$ điểm $A_1;\ A_2;\ldots;\ A_n$ theo thứ tự trên đường thẳng $xy$ và điểm $M$ nằm ngoài đường thẳng $xy$. Nối $M$ với $n$ điểm đó. Số tam giác đếm được là $55$ thì giá trị của $n$ là bao nhiêu?
	\loigiai{
		Số tam giác tạo thành bằng số đoạn thẳng lập được từ các điểm trên đường thẳng $xy$ nên ta có $\dfrac{n\cdot (n-1)}{2}=55$ hay $n\cdot (n-1)=110=11\cdot 10.$\\
		Suy ra $n=11.$

\item Vẽ tia $ Ot $ là tia phân giác của góc $ \widehat{yOz} $. Tính số đo góc $ \widehat{zOt} $ và $ \widehat{tOx} $.
\end{listEX}
\loigiai{
\immini{
\begin{listEX}
\item Vì hai tia $ Oy $, $ Oz $ cùng nằm về nửa mặt phẳng bờ chứa tia $ Ox $ và $ \widehat{xOy}=30^{\circ} < \widehat{xOz}=110^{\circ} $ nên tia $ Oy $ nằm giữa hai tia $ Ox $ và $ Oz $.
\item Ta có
\begin{eqnarray*}
	\widehat{xOy}+\widehat{yOz}&=&\widehat{xOz}\\
	30^{\circ}+\widehat{yOz}&=&110^{\circ}\\
	\widehat{yOz}&=&80^{\circ}.
\end{eqnarray*}
\end{listEX}

\widehat{zOt}=\widehat{tOy}=\dfrac{80^{\circ}}{2}=40^{\circ}.

\begin{bt}%[Dương BùiĐức, C2-L6-2018]%[6H2K4]
Trên cùng nửa mặt phẳng bờ chứa tia $ Ox $, vẽ hai tia $ Oy $ và $ Oz $ sao cho $ \widehat{xOz}=50^{\circ} $, $ \widehat{xOy}=120^{\circ} $. Gọi tia $ Oa $, $ Ob $ thứ tự là phân giác của góc $ \widehat{xOy} $ và $ \widehat{yOz} $. Tính số đo của góc $ \widehat{aOb} $.
\loigiai{
\immini{
	Ta có 
	\begin{eqnarray*}
		\widehat{xOz}+\widehat{zOy}&=&\widehat{xOy}\\
		50^{\circ}+\widehat{zOy}&=&120^{\circ}\\
		\widehat{zOy}&=&70^{\circ}.
	\end{eqnarray*}
	Vì tia $ Oa $, $ Ob $ thứ tự là phân giác của góc $ \widehat{xOy} $ và $ \widehat{yOz} $ nên
	\begin{eqnarray*}
		\widehat{xOa}&=&\dfrac{\widehat{xOy}}{2}=60^{\circ}.\\
		\widehat{zOb}&=&\dfrac{\widehat{zOy}}{2}=35^{\circ}.
	\end{eqnarray*}
}{

Mà 
\begin{eqnarray*}
	\widehat{xOz}+\widehat{zOb}&=&\widehat{xOb}\\
	50^{\circ}+35^{\circ}&=&\widehat{xOb}\\

\end{eqnarray*}
Lại có
\begin{eqnarray*}
	\widehat{xOa}+\widehat{aOb}&=&\widehat{xOb}\\
	60^{\circ}+\widehat{aOb}&=&85^{\circ}\\
	\widehat{aOb}&=&25^{\circ}.
\end{eqnarray*}

\textbf{Nhận xét.} Bạn có thể chứng tỏ được bài toán tổng quát sau:\\
``Trên nửa mặt phẳng bờ $ Ox $ vẽ hai tia $ Oy $ và $ Oz $ sao cho $ \widehat{xOy}<\widehat{xOz} $. Gọi tia $ Oa $, $ Ob $ thứ tự là phân giác của góc $ \widehat{xOy} $ và $ \widehat{yOz} $ thì $ \widehat{aOb}=\dfrac{\widehat{xOz}}{2}$''.
\end{bt}
