\begin{ex}%[Đề HK1 Toán 11, Võ Thành Trinh, An Giang, 2017-2018]%[1H1K2]

 \Rightarrow 2(x'-1)-3(y'+2)+1=0 \Leftrightarrow 2x'-3y'-7=0$\\
 Vậy $d':2x-3y-7=0$.
  }
\end{ex}
