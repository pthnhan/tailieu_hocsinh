	\begin{enumEX}[1)]{2}
		\item $ x^{m}\cdot x^{n}=x^{m+n} $.
		\item $ x^{m}: x^{n}=x^{m-n} $.
		\item $ (x\cdot y)^{m}=x^{m}\cdot y^{m} $.
		\item $ (x: y)^{m}=x^{m}: y^{m} $.
	\end{enumEX}
	\item Quy tắc nhân đơn thức với đa thức: muốn nhân đơn thức với đa thức, ta nhân đơn thức này với từng hạng tử của đa thức kia rồi cộng các tích lại với nhau.
\end{enumerate}
\subsection{Các dạng toán}
\begin{dang}[Thực hiện phép nhân đơn thức với đa thức]
\begin{enumerate}[\tickEX]
	\item Xác định dấu nhờ quy tắc về dấu.
	\item Nhân các hệ số tương ứng.
	\item Nhân các biến số tương ứng nhờ quy tắc lũy thừa.
\end{enumerate}
\end{dang}
\begin{vd}
	Thực hiện phép tính sau:
	\begin{listEX}[2]
		\item $\dfrac{3}{5}xy(xy^{2}-5x+10y)$;
		\item $ \left(3xy-2x^{2}+xy^{2}\right)\cdot \dfrac{2}{3}x^{2}y $.
	\end{listEX}
	\loigiai{
		\begin{enumerate}
			\item $\dfrac{3}{5}xy(xy^{2}-5x+10y)=\dfrac{3}{5}x^{2}y^{3}-3x^{2}y+6xy^{2}$.
			\item $ \left(3xy-2x^{2}+xy^{2}\right)\cdot \dfrac{2}{3}x^{2}y=2x^{3}y^{2}-\dfrac{4}{3}x^{4}y+\dfrac{2}{3}x^{3}y^{3} $.
		\end{enumerate}
	}
\end{vd}
\begin{vd}
	Thực hiện phép tính và xác định bậc của đa thức:
	\begin{listEX}[2]
		\item $ 2x^{3}\left(5x^{3}-x-\dfrac{1}{2}\right) $;
		\item $\dfrac{2}{3}x^{2}y(6xy-x^{2}+1)$.
	\end{listEX}
	\loigiai{
		\begin{enumerate}
			\item $ 2x^{3}\left(5x^{3}-x-\dfrac{1}{2}\right)=10x^{6}-2x^{4}-x^{3}\Rightarrow $ đa thức có bậc $ 6 $.
			\item $\dfrac{2}{3}x^{2}y(6xy-x^{2}+1)=4x^{3}y^{2}-\dfrac{2}{3}x^{4}y+\dfrac{2}{3}x^{2}y\Rightarrow$ đa thức có bậc $ 5 $.
		\end{enumerate}
	}
\end{vd}

\begin{vd}
	Cho $ n\in\mathbb{N}^{*}$. Tìm hệ số của hạng tử bậc lớn nhất của đa thức
	\[ 
	A=3x^{n}(4x^{n-1}-1)-2x^{n+1}(6x^{n-2}-1).
	\]
	\loigiai{
		Ta có
{\allowdisplaybreaks
\begin{align*}
3x^{n}(4x^{n-1}-1)-2x^{n+1}(6x^{n-2}-1)&=12x^{2n-1}-3x^{n}-12x^{2n-1}+2x^{n+1}\\
&=2\cdot 2x^{n}-3x^{n}=x^{n}.
\end{align*}}Do đó hệ số của hạng tử bậc lớn nhất của đa thức $ A $ bằng $ 1 $.
	}
\end{vd}
\begin{dang}[Tìm giá trị của biến $ x $]
	\begin{enumerate}[\tickEX]
		\item Thực hiện phép nhân đơn thức với đa thức.
		\item Thực hiện phép cộng (trừ) các đơn thức đồng dạng.
		\item Áp dụng quy tắc tìm một số hạng khi biết tích và số hạng còn lại.
	\end{enumerate}
\end{dang}
\begin{vd}
	Tìm giá trị của $ x $ biết:
	\begin{listEX}[2]
		\item $ 3x(8x-4)-6x(4x-3)=30 $;
		\item $ 3x(5-2x)+2x(3x-5)=20 $.
	\end{listEX}
	\loigiai{
		\begin{enumerate}
			\item Ta có
		{\allowdisplaybreaks
		\begin{align*}
		&3x(8x-4)-6x(4x-3)=30\\
		\Rightarrow& 24x^{2}-12x-24x^{2}+18x=30
		\Rightarrow 6x=30\Rightarrow x=5.
	\end{align*}}Vậy $ x=5 $.
			\item Ta có
		{\allowdisplaybreaks
		\begin{align*}
			&3x(5-2x)+2x(3x-5)=20\\
			\Rightarrow& 15x-6x^{2}+6x^{2}-10x=20
			\Rightarrow 5x=20\Rightarrow x=4.
		\end{align*}}Vậy $ x=4 $.
		\end{enumerate}
	}
\end{vd}
\begin{vd}
	Tìm giá trị của $ x $ biết:
	\begin{listEX}[2]
		\item $ 3x^{2}-3x(-2+x)=36 $;
		\item $ x^{2}+8=0 $.
	\end{listEX}
	\loigiai{
		\begin{enumerate}
			\item Ta có $ 3x^{2}-3x(-2+x)=36\Leftrightarrow 3x^{2}+6x-3x^{2}=36\Leftrightarrow x=6 $.\\
			Vậy $ x=6 $.
			\item Ta có $ x^{2}\geq 0 $ với mọi $ x $ nên $ x^{2}+8>0 $ với mọi $ x $, do đó không tồn tại $ x $ thỏa mãn đề bài.
		\end{enumerate}
	}
\end{vd}
\begin{vd}
	Tìm giá trị của $ x $ biết:
	\begin{enumerate}
		\item $ 2x^{4}+x(-2x^{2}+x^{3})-3(x^{4}-6)=26 $;
		\item $ -4x^{5}-3x^{3}(2x^{2}-1)+5(2x^{5}-x^{3})+2(x^{3}-x^{2}+2)+5=0 $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			\[ 
			2x^{4}+x(-2x^{2}+x^{3})-3(x^{4}-6)=26\Leftrightarrow 2x^{4}-2x^{3}+x^{4}-3x^{4}+18=26\Leftrightarrow x^{3}=8\Leftrightarrow x=2.
			\]
			Vậy $ x=2 $.
			\item Ta có
		{\allowdisplaybreaks\begin{eqnarray*}
				&&-4x^{5}-3x^{3}(2x^{2}-1)+5(2x^{5}-x^{3})+2(x^{3}-x^{2}+2)+5=0\\
				&\Leftrightarrow& -4x^{5}-6x^{5}+3x^{3}+10x^{5}-5x^{3}+2x^{3}-2x^{2}+4+5=0\\
				&\Leftrightarrow& x^{2}=9\Leftrightarrow x=\pm 3.
			\end{eqnarray*}}
			Vậy $ x=\pm 3 $.
		\end{enumerate}
	}
\end{vd}

\begin{vd}
Rút gọn các biểu thức sau:
\begin{enumEX}{2}
\item $ xy(x^{5}-y^{3})-x^{2}y(x^{4}-y^{3}) $;
\item $ x^{3}y^{4}(x^{2}-y^{3})-x^{3}y^{3}(x^{4}-y^{4}) $.
\end{enumEX}
\loigiai{
\begin{enumerate}
	\item $ xy(x^{5}-y^{3})-x^{2}y(x^{4}-y^{3})=x^{6}y-xy^{4}-x^{6}y+x^{2}y^{4}=x^{2}y^{4}-xy^{4} $.
	\item $ x^{3}y^{4}(x^{2}-y^{3})-x^{3}y^{3}(x^{4}-y^{4})=x^{5}y^{4}-x^{3}y^{7}-x^{7}y^{3}+x^{3}y^{7}=x^{5}y^{4}-x^{7}y^{3} $.
\end{enumerate}
}
\end{vd}
\begin{vd}
	Rút gọn rồi tính giá trị các biểu thức sau:
	\begin{enumerate}
		\item $ A=x^{3}(x^{2}-y^{3})-y^{3}(y^{2}-x^{3}) $ với $ x=2,\ y=-2 $;
		\item $ B=(-x^{2}y)(x^{2}y^{2}-2x)+xy^{3}(x^{3}-xy^{3}) $ với $ x=3,\ y=2 $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			\[ 
			A=x^{3}(x^{2}-y^{3})-y^{3}(y^{2}-x^{3})=x^{5}-x^{3}y^{3}-y^{5}+x^{3}y^{3}=x^{5}-y^{5}.
			\]
			Thay $ x=2,\ y=-2 $ vào $ A $, ta được 
			\[ 
			A=2^{5}-(-2)^{5}=64.
			\]
			\item Ta có
			\[ 
			B=(-x^{2}y)(x^{2}y^{2}-2x)+xy^{3}(x^{3}-xy^{3})=-x^{4}y^{3}+2x^{3}y+x^{4}y^{3}-x^{2}y^{6}=2x^{3}y-x^{2}y^{6}.
			\]
			Thay $ x=3,\ y=2 $ vào $ B $, ta được
			\[ 
			B=2\cdot 3^{3}\cdot 2-3^{2}\cdot 2^{6}=-468.
			\]
		\end{enumerate}
	}
\end{vd}
\begin{vd}
Tính giá trị của biểu thức $ A=x^{5}-70x^{4}-70x^{3}-70x^{2}-70x+29 $ bằng cách hợp lí tại $ x=71 $.
\loigiai{
	\begin{phantich}
		Ta thấy giá trị của $ x=71 $ nên $ x-71=0 $ và trong biểu thức có hệ số $ 70=71-1 $, do đó ta có thể đưa ra lời giải như sau.
	\end{phantich}
Ta có
{\allowdisplaybreaks\begin{eqnarray*}
A&=&x^{5}-70x^{4}-70x^{3}-70x^{2}-70x+29\\
&=&x^{5}-(71-1)x^{4}-(71-1)x^{3}-(71-1)x^{2}-(71-1)x+29\\
&=&x^{5}-71x^{4}+x^{4}-71x^{3}+x^{3}-71x^{2}+x^{2}-71x+x+29\\
&=&x^{4}(x-71)+x^{3}(x-71)+x^{2}(x-71)+x(x-71)+x+29.
\end{eqnarray*}}Thay $ x=71 $ vào, ta được $ A=100 $.
\begin{note}
	\begin{enumerate}[1)]
		\item Một cách tổng quát ``nếu $ x=m $ thì $ x-m=0 $''.
		\item Đây là dạng toán nâng cao, đòi hỏi học sinh phải tìm ra được quan hệ đặc biệt trong bài toán, cụ thể mối quan hệ giữa hệ số của biểu thức cần tính và giá trị của biến số, từ đó sẽ có cách tính hợp lí.
	\end{enumerate}
\end{note}
}
\end{vd}
\begin{dang}[Bài toán chứng minh]
	\begin{enumerate}[\tickEX]
		\item Với bài toán chứng minh đẳng thức: ta thường biến đổi từ vế phức tạp về vế đơn giản, hoặc tìm biểu thức trung gian để so sánh.
		\item Với bài toán chứng minh giá trị của biểu thức không đổi: ta thường rút gọn để chứng minh biểu thức đó bằng một hằng số nào đó.
	\end{enumerate}
\end{dang}
\begin{vd}
Chứng minh rằng giá trị của biểu thức sau không phụ thuộc vào giá trị của các biến:
\begin{enumEX}{2}
\item $ A=10-5x(x-1{,}2)+2x(2{,}5x-3) $;
\item $ B=x(2x+1)-x^{2}(x+2)+x^{3}-x+3 $.
\end{enumEX}
\loigiai{
	\begin{enumerate}
	\item Ta có
{\allowdisplaybreaks\begin{eqnarray*}
		A&=&10-5x(x-1{,}2)+2x(2{,}5x-3)\\
		&=&10-5x^{2}+6x+5x^{2}-6x\\
		&=&10\ (\text{đpcm}).
	\end{eqnarray*}}
	\item Ta có
	{\allowdisplaybreaks\begin{eqnarray*}
		B&=&x(2x+1)-x^{2}(x+2)+x^{3}-x+3\\
		&=&2x^{2}+x-x^{3}-2x^{2}+x^{3}-x+3\\
		&=&3\ (\text{đpcm}).
	\end{eqnarray*}}
	\end{enumerate}
}
\end{vd}
\begin{vd}
	Cho biểu thức $ A=x(1+x)-x^{2}(1-x)+x^{3}(x^{2}-1) $. Chứng minh rằng với hai giá trị đối nhau của $ x $ thì biểu thức $ A $ có hai giá trị đối nhau.
	\loigiai{
		Ta có
		{\allowdisplaybreaks\begin{eqnarray*}
			A&=&x(1+x)-x^{2}(1-x)+x^{3}(x^{2}-1)\\
			&=&x+x^{2}-x^{2}+x^{3}+x^{5}-x^{3}
			=x^{5}+x.
		\end{eqnarray*}}
	Xét với giá trị $ x $ và $ -x $ tùy ý, ta có
	{\allowdisplaybreaks\begin{eqnarray*}
		A(x)&=&x^{5}+x,\\
		A(-x)&=&(-x)^{5}+(-x)=-x^{5}-x=-A(x)\ (\text{đpcm}).
	\end{eqnarray*}}
	}
\end{vd}
\begin{vd}
Cho $ m $ số, mỗi số bằng $ 3n+1 $ và $ n $ số, mỗi số bằng $ 9-3m $. Biết rằng tổng tất cả các số đó bằng $ 5 $ lần tổng $ m+n $. Chứng minh rằng $ m=n $.
\loigiai{
Giá trị của $ m $ số $ 3n+1 $ là $ m(3n+1) $, giá trị của $ n $ số $ 9-3m $ là $ n(9-3m) $.\\
Theo đề bài ta có
{\allowdisplaybreaks\begin{eqnarray*}
&&m(3n+1)+n(9-3m)=5(m+n)\\
&\Leftrightarrow&3mn+m+9n-3mn=5m+5n\\
&\Leftrightarrow&9n-5n=5m-m\\
&\Leftrightarrow&4n=4m\\
&\Leftrightarrow&n=m\ (\text{đpcm}).
\end{eqnarray*}}
}
\end{vd}

\begin{vd}
	Chứng minh các đẳng thức sau:
	\begin{enumerate}
		\item $ 4(x-6)-x^{2}(2+3x)+x(5x-4)+3x^{2}(x-1)+24=0 $;
		\item $ x^{n}(y^{n}-xy^{n-1}+1)-y^{n}(x^{n}-yx^{n-1}+1)+x^{n-1}y^{n-1}(x^{2}-y^{2})=x^{n}-y^{n} $.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có 
	{\allowdisplaybreaks\begin{eqnarray*}
		VT&=&4(x-6)-x^{2}(2+3x)+x(5x-4)+3x^{2}(x-1)+24\\
		&=&4x-24-2x^{2}-3x^{3}+5x^{2}-4x+3x^{3}-3x^{2}+24\\
		&=&-24+24\\
		&=&0.
	\end{eqnarray*}}
	Suy ra điều phải chứng minh.
	\item Ta có 
	{\allowdisplaybreaks\begin{eqnarray*}
		VT&=&x^{n}(y^{n}-xy^{n-1}+1)-y^{n}(x^{n}-yx^{n-1}+1)+x^{n-1}y^{n-1}(x^{2}-y^{2})\\
		&=&x^{n}y^{n}-x^{n+1}y^{n-1}+x^{n}-y^{n}x^{n}+y^{n+1}x^{n-1}-y^{n}+x^{n+1}y^{n-1}-x^{n-1}y^{n+1}\\
		&=&x^{n}-y^{n}=VP.
	\end{eqnarray*}}
	Suy ra điều phải chứng minh.
\end{enumerate}
}
