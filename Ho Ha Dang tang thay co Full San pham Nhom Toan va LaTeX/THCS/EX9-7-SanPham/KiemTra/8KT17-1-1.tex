\begin{ex}%[8D1Y1]%[8D1B11]
Thực hiện phép tính:
\begin{multicols}{2}
\begin{enumerate}
	\item $2x^2(3x^2-xy-\dfrac{3}{2}y^2)$
	\item $(16x^4y^3-20x^2y^3-4x^4y^4):(4x^2y^2)$
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item Ta có $2x^2(3x^2-xy-\dfrac{3}{2}y^2)=2x^2\cdot 3x^2-2x^2\cdot xy-2x^2\cdot\dfrac{3}{2}y^2=6x^4-2x^3y-3x^2y^2$.
\item Ta có $(16x^4y^3-20x^2y^3-4x^4y^4):(4x^2y^2)=\dfrac{16x^4y^3}{4x^2y^2}-\dfrac{20x^2y^3}{4x^2y^2}-\dfrac{4x^4y^4}{4x^2y^2}=4x^2y-5y-x^2y^2$
\end{enumerate}
}
\end{ex}

\begin{ex}%[8D1B6]%[8D1K8]%[8D1B8]
Phân tích đa thức thành nhân tử:
\begin{multicols}{3}
\begin{enumerate}
	\item $5x(3-2x)-7(2x-3)$.
	\item $x^3-4x^2+4x$.
	\item $x^2+5x+6$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}
\item $5x(3-2x)-7(2x-3)=5x(3-2x)+7(3-2x)=(3-2x)(5x+7)$.
\item $x^3-4x^2+4x=x(x^2-4x+4)=x(x-2)^2$.
\item $x^2+5x+6=x^2+2x+3x+6=x(x+2)+3(x+2)=(x+2)(x+3)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[8D2B9]
Cho biểu thức $M=(4x + 3)^2-2x(x + 6)-5(x-2)(x +2)$
\begin{enumerate}
	\item Thu gọn biểu thức $M$.
	\item Tính giá trị biểu thức tại $x=-2$.
	\item Chứng minh biểu thức $M$ luôn dương.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $M=16x^2+24x+9-2x^2-12x-5x^2+20=9x^2+12x+29$.
\item Thay $x=-2$ vào biểu thức $M$ ta được:\\
$M=9(-2)^2+12(-2)+29=41$.
\item Ta có $M=9x^2+12x+29=9x^2+2\cdot 3x\cdot 2+4+25=(3x+2)^2+25\geq 25>0$.\\
Do đó $M>0$ với mọi $x$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[8H1K7]%[8H1B9]%[8H1K8]%[8H1B10]
Cho tam giác $ABC$, trực tâm $H$. Các đường thẳng vuông góc với $AB$ tại $B$, vuông góc với $AC$ tại $C$ cắt nhau ở $D$. Chứng minh rằng:
\begin{enumerate}
	\item $BDCH$ là hình bình hành.
	\item $\widehat{BAC}+\widehat{BDC}=180^\circ$.
	\item $H, M, D$ thẳng hàng ($M$ là trung điểm của $BC$).
	\item $OM = \dfrac{1}{2}AH$ ($O$ là trung điểm của $AD$).
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[>=stealth,scale=1.5,font=\footnotesize]
\tkzInit[xmin=-1,xmax=5,ymin=-1.5,ymax=3.5]
\tkzClip
\tkzDefPoints{0/0/B,4/0/C,1/3/A}
\tkzDefPointBy[projection=onto A--B](C)\tkzGetPoint{E}
\tkzDefPointBy[projection=onto A--C](B)\tkzGetPoint{F}
\tkzDefLine[perpendicular=through B](A,B)\tkzGetPoint{i}
\tkzDefLine[perpendicular=through C](A,C)\tkzGetPoint{j}
\tkzInterLL(C,E)(B,F)\tkzGetPoint{H}
\tkzInterLL(B,i)(C,j)\tkzGetPoint{D}
\tkzDefMidPoint(B,C)\tkzGetPoint{M}
\tkzDefMidPoint(A,D)\tkzGetPoint{O}
\tkzDrawSegments(A,B B,C C,A A,H B,D C,E H,D A,D B,F O,M)
\tkzDrawPoints(A,B,C,D,E,H,M,F,O)
\tkzLabelPoints[left](A,E,B)
\tkzLabelPoints[right](D,C,F,O)
\tkzLabelPoints[below](M,H)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Chứng minh $BDCH$ là hình bình hành.\\
Ta có $BH\parallel DC$  (cùng vuông góc với $AC$)\\
$CH\parallel DB$ (cùng vuông góc với $AB$) nên $BDCH$ là hình bình hành.
\item Chứng minh$\widehat{BAC}+\widehat{BDC}=180^\circ$.\\
Tứ giác $ABCD$ có $\widehat{ABD}+\widehat{ACD}=180^\circ$. \\
Nên $\widehat{BAC}+\widehat{ACD}=180^\circ$.
\item Chứng minh $H, M, D$ thẳng hàng.\\
Do $M$ là trung điểm của đường chéo $BC$ của hình bình hành $BHCD$ nên $M$ cũng là trung điểm của $HD$ do đó $H, M , D$ thẳng hàng.
\item Chứng minh $OM = \dfrac{1}{2}AH$.\\
Vì $O$ là trung điểm của $AD$ và $M$ là rung điểm của $HD$, suy ra $OM$ là đường trung bình của tam giác $AHD$. \\
Do đó $OM=\dfrac{1}{2}AH$.
\end{enumerate}
}
\end{ex}
