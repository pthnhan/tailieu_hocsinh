\begin{ex}%[Thi HK2, THPT Lý Thái Tổ, Hà Nội , 2018] %[2D3K2-4]%[Trần Tuấn Việt, dự án(12EX-8)
Cho $f(x)$ là hàm số liên tục trên $\left[a; b\right]$ thỏa mãn $\displaystyle \int_a^b f(x) \mathrm{d}x = 7$. Tính $I = \displaystyle \int_a^b f(a + b - x) \mathrm{d}x$
\choice
{$I = a + b + 7$}
{$I = a + b - 7$}
{$I = 7 - a - b$}
{\True $I = 7$}
\loigiai{ Gọi $F(x)$ là nguyên hàm của hàm số $f(x)$ ta có
$\displaystyle \int_a^b f(x) \mathrm{d}x = F(x) \big| _a^b = F(b) - F(a) = 7$. \medskip

Xét tích phân \begin{align*}
\displaystyle \int_a^b f(x) \mathrm{d}x & = \int_a^b f(a + b - x) \mathrm{d}x \\
& = - F(a + b - x)\big|_a^b \\
& = - F(a + b - b) + F(a + b - a) \\
& = F(b) - F(a) \\
& = 7.
\end{align*}
}
\end{ex}
