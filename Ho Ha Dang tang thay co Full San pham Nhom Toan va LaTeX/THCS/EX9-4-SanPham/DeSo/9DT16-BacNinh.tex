\begin{ex}%[Đề thi vào 10, Sở giáo dục Bắc Ninh, 2016]%[9D1Y2]%[9D4B4]%[9D1K8]
\hfill
\begin{enumerate}
\item Tính giá trị của biểu thức $P=\sqrt{x+1}$ với $x=8$.
\item Giải phương trình $x^2-4x+3=0$.
\item Rút gọn biểu thức $A=\left(\dfrac{1}{\sqrt{x}-1}+\dfrac{1}{\sqrt{x}+1}\right).\dfrac{\sqrt{x}-1}{2}$ với $x \ge 0$, $x \ne 1$.
\end{enumerate}

\loigiai{
\begin{enumerate}
\item Với $x=8$ thì $P=\sqrt{8+1}=\sqrt{9}=3$.
\item Ta có $a+b+c=1-4+3=0$ nên phương trình có hai nghiệm $x=1$ và $x=3$.
\item $A=\dfrac{\sqrt{x}+1+\sqrt{x}-1}{\left(\sqrt{x}-1\right)\left(\sqrt{x}+1\right)}.\dfrac{\sqrt{x}-1}{2}=\dfrac{2\sqrt{x}}{\left(\sqrt{x}-1\right)\left(\sqrt{x}+1\right)}.\dfrac{\sqrt{x}-1}{2}=\dfrac{\sqrt{x}}{\sqrt{x}+1}.$
\end{enumerate}
}
\end{ex}


\begin{ex}%[Đề thi vào 10, Sở giáo dục Bắc Ninh, 2016]%[9D3B2]%[9D4G6]
Cho hệ phương trình $\heva{&x+2y=m\\&2x+5y=1}$ với $m$ là tham số.
\begin{enumerate}
\item Giải hệ phương trình khi $m=0$.
\item Tìm $m$ để hệ phương trình có nghiệm duy nhất $(x, y)$ sao cho $x$ và $y$ là hai nghiệm của phương trình $t^2-(3m-1)t+m^4+9m-13=0$ với $t$ là ẩn số.
\end{enumerate}

\loigiai{
\begin{enumerate}
\item Khi $m=0$ thì hệ trở thành $\heva{&x+2y=0\\&2x+5y=1}$\\
$\Leftrightarrow \heva{&2x+4y=0\\&2x+5y=1} \Leftrightarrow \heva{&y=1\\&x=-2.}$
\item $\heva{&x+2y=m\\&2x+5y=1}$ $\Leftrightarrow \heva{&2x+4y=2m\\&2x+5y=1} \Leftrightarrow \heva{&x=5m-2\\&y=1-2m.}$\\
Vì  $x$ và $y$ là hai nghiệm của phương trình $t^2-(3m-1)t+m^4+9m-13=0$  nên theo hệ thức Vi-et ta có:\\
$x.y=m^4+9m-13 \Leftrightarrow (5m-2).(1-2m)=m^4+9m-13 \Leftrightarrow m^4+10m^2-11=0$
$ \Leftrightarrow \hoac{&m^2=1\\&m^2=-11 \;\text{(loại)}}$ $ \Leftrightarrow \hoac{&m=1\\&m=-1.}$\\
Thử lại, $m=1$, $m=-1$ thỏa bài toán.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Bắc Ninh, 2016]%[9D4K8]
Quãng đường từ Bắc Ninh đi Hà Nội dài $30$ km. Một ô tô đi từ Bắc Ninh đi Hà Nội, rồi từ Hà Nội về Bắc Ninh. Biết vận tốc lúc đi lớn hơn vận tốc lúc về là $10$ km/h. Do đó, thời gian về nhiều hơn thời gian đi là $9$ phút. Tính vận tốc của ô tô khi đi từ Bắc Ninh ra Hà Nội.
\loigiai{
Gọi $x$(km/h) là vận tốc của ô tô đi từ Bắc Ninh ra Hà Nội $(x>0)$.\\
Suy ra vận tốc ô tô đi từ Hà Nội về Bắc Ninh là $x+10$ (km/h).\\
Đổi $9$ phút $=\dfrac{3}{20}$ (h).\\
Theo đề, ta có phương trình: $\dfrac{30}{x}-\dfrac{30}{x+10}=\dfrac{3}{20}$\\
$$\Leftrightarrow x^2+10x-2000=0 \Leftrightarrow \hoac{&x=40\\&x=-50\;\text{(loại)}.}$$
Vậy vận tốc của ô tô khi đi từ Bắc Ninh ra Hà Nội là $40$ (km/h).
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Bắc Ninh, 2016]%[9H3B7]%[9H3G4]
Cho tam giác $ABC$ có ba góc đều nhọn, góc $\widehat{BAC}=45^\circ$. Gọi $O$ là tâm  đường tròn ngoại tiếp tam giác $ABC$. Các đường cao $BD$, $CE$ ($D$ thuộc $AC$, $E$ thuộc $AB$) cắt nhau tại $H$.
\begin{enumerate}
\item Chứng minh rằng tứ giác $ADHE$ nội tiếp một đường tròn.
\item Chứng minh rằng tam giác $HDC$ vuông cân tại $D$.
\item Tính tỉ số $\dfrac{DE}{BC}$.
\item Chứng minh rằng $OA$ vuông góc với $DE$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1]
	\tkzInit[xmin=-1,xmax=7,ymin=-3,ymax=5]
	\clip (-1,-3) rectangle (7,5); 
	\tkzDefPoint[label=right:$B$](0:6){B}
	\tkzDefPoint[label=above:$C$](4,4){C}
	\tkzDefPoint[label=left:$A$](0,0){A}
	\tkzDefShiftPoint[A](102:3){x}
	\tkzCircumCenter(A,B,C)\tkzGetPoint{O} 
	\tkzDrawCircle(O,A)
	\tkzDefPointBy[projection=onto A--B](C)
	\tkzGetPoint{E}
	\tkzDefPointBy[projection=onto A--C](B)
	\tkzGetPoint{D}
	\tkzInterLL(C,E)(B,D) \tkzGetPoint{H}
	\tkzDefLine[perpendicular=through A,K=3](A,O)
	\tkzGetPoint{M}
	\tkzDrawLines(A,M)
	\tkzDrawSegments(A,B B,C C,A C,E B,D O,A D,E)
	\tkzMarkRightAngles[fill=green!60!cyan,size=0.2,opacity=1](C,E,B B,D,C O,A,M)
	\tkzMarkAngles[arc=ll,size=0.5cm,mkcolor=red](A,C,E D,B,A E,A,D C,H,D E,H,B)
	\tkzDrawPoints[color=red](A,B,C,D,E,H,O)
	\tkzLabelPoints[above left](D)
	\tkzLabelPoints[above right](H)
	\tkzLabelPoints[below](E,O,x)
	\end{tikzpicture}   	
	  		
\end{center}
\begin{enumerate}
\item Vì $\heva{&BD \perp AC \Rightarrow  \widehat{HDA}=90^\circ\\& CE \perp AB \Rightarrow \widehat{HEA}=90^\circ}$ $\Rightarrow \widehat{HDA}+\widehat{HEA}=180^\circ$.\\
Vậy tứ giác $ADHE$ nội tiếp.
\item Tam giác $ACE$ vuông tại $E$, có $\widehat{A}=45^\circ$ $\Rightarrow \widehat{ACE}=45^\circ$ $(1)$.\\
Mặt khác, tứ giác $ADHE$ nội tiếp nên $\widehat{BHE}=\widehat{A}$ hay $\widehat{BHE}=90^\circ$.\\
Ta có $\widehat{DHC}=\widehat{BHE}$ (đđ) $\Rightarrow \widehat{DHC}=45^\circ$ $(2)$.\\
Từ $(1)$ và $(2)$ suy ra $ \widehat{DHC}= \widehat{DCH}$, hay $\triangle DCH$ vuông cân tại $D$.
\item $\triangle ADB$ vuông cân tại $D$ nên $\widehat{DBA}=45^\circ$ $(3)$.\\
Từ $(1)$ và $(3)$ $\Rightarrow$ $\widehat{DCE}=\widehat{DBE}=45^\circ$ $\Rightarrow$ tứ giác $CDEB$ nội tiếp.\\
$\Rightarrow \widehat{ADE}=\widehat{CBE}$ $\Rightarrow \triangle ADE \backsim \triangle ABC$ $\Rightarrow \dfrac{DE}{BC}=\dfrac{AD}{AB}=\dfrac{\sqrt{2}}{2}$.
\item Qua $A$ vẽ tiếp tuyến $Ax$.\\
Ta có $\widehat{CAx}=\widehat{ABC}$ (góc tạo bởi tiếp và dây cung cùng chắn cung $\wideparen{AC}$).\\
Mặt khác: $\widehat{ABC}=\widehat{ADE}$ $\Rightarrow \widehat{CAx}=\widehat{ADE} \Rightarrow Ax \parallel DE$ hay $AO \perp DE$ (đpcm).
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Sở giáo dục Bắc Ninh, 2016]%[9D4G7]%[9D5G1]
\hfill
\begin{enumerate}
\item Giải phương trình $x^2+2x-3=4\sqrt{2x+3}$.
\item Cho ba số thực $a$, $b$, $c$ thỏa mãn $a+b=4c$. Chứng minh rằng
$$ 2\sqrt{a^2-ab+b^2}+\sqrt{a^2-2ac+4c^2}+\sqrt{b^2-2bc+4c^2}\ge 8c. $$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Điều kiện $x \ge -\dfrac{3}{2}$.\\
PT $\Leftrightarrow \left(x+2\right)^2=\left(\sqrt{2x+3}+2\right)^2$\\
$\Leftrightarrow \hoac{&\sqrt{2x+3}+2=x+2 &(1)\\&\sqrt{2x+3}+2=-x-2 & (2)}$\\
$(1) \Leftrightarrow \sqrt{2x+3}=x \Leftrightarrow \heva{&x \ge 0\\ &2x+3=x^2} \Leftrightarrow \heva{&x \ge 0\\&x^2-2x-3=0} \Leftrightarrow \heva{&x \ge 0\\& \hoac{&x=-1\\&x=3}} \Leftrightarrow x=3$.\\
$(2) \Leftrightarrow \sqrt{2x+3}=-x-4 \Leftrightarrow \heva{&x \le -4\\ &2x+3=(x+4)^2} \Leftrightarrow \heva{&x \le -4\\&x^2+6x+13=0} (\text{hệ vô nghiệm}).$\\
Vậy phương trình có một nghiệm $x=3$.
\item Với $c \le 0$ thì bất đẳng thức đúng.\\
Với $c>0$ thì chia hai vế của bất đẳng thức cho $c$, ta được:
$$P = 2\sqrt{\left(\dfrac{a}{c}\right)^2-\dfrac{a}{c}.\dfrac{b}{c}+\left(\dfrac{b}{c}\right)^2}+\sqrt{\left(\dfrac{a}{c}\right)^2-2\dfrac{a}{c}+4}+\sqrt{\left(\dfrac{b}{c}\right)^2-2\dfrac{b}{c}+4} \geq 8 (*)$$
Áp dụng bất đẳng thức $a^2+b^2 \ge \dfrac{(a+b)^2}{2}$, ta có $\left(\dfrac{a}{c}\right)^2+\left(\dfrac{b}{c}\right)^2 \ge 8.$\\
Suy ra 
$$P \geq 2\sqrt{8-\dfrac{a}{c}.\dfrac{b}{c}}+\sqrt{\left(\dfrac{a}{c}-1\right)^{2}+3}+\sqrt{\left(\dfrac{b}{c}-1\right)^{2}+3}$$
Ta sẽ chứng minh $2\sqrt{8-\dfrac{a}{c}.\dfrac{b}{c}}+\sqrt{\left(\dfrac{a}{c}-1\right)^{2}+3}+\sqrt{\left(\dfrac{b}{c}-1\right)^{2}+3} \ge 8  \;\;(1)$\\
Ta có $a+b=4c \Leftrightarrow \dfrac{a}{c}+\dfrac{b}{c}=4$.\\
Theo bất đẳng thức Cô-si thì $\dfrac{a}{c}.\dfrac{b}{c} \le \dfrac{\left(\dfrac{a}{c}+\dfrac{b}{c}\right)^2}{4}=4$.\\
$\Rightarrow 2\sqrt{8-\dfrac{a}{c}.\dfrac{b}{c}} \ge 4\;\; (2)$\\
và $\sqrt{\left(\dfrac{a}{c}-1\right)^{2}+3}+\sqrt{\left(\dfrac{b}{c}-1\right)^{2}+3} \ge \sqrt{\left(\dfrac{a}{c}-1+\dfrac{b}{c}-1\right)^{2}+(\sqrt{3}+\sqrt{3})^{2}}=4 \;\;(3)$.\\
Từ $(2)$ và $(3)$ suy ra $(1)$ đúng, nên $(*)$ đúng. Dấu $``="$ xảy ra khi $a=b=2c$.
\end{enumerate}
}
\end{ex}
