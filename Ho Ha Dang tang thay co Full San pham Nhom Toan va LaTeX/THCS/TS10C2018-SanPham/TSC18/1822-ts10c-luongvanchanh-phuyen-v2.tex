\begin{ex}%[Đề thi vào 10, chuyên Lương Văn Chánh, tỉnh Phú Yên, năm 2018]%[Huỳnh Xuân Tín, Dự án EX-TS10C-18]%[9D0K1]

	Cho biểu thức $$P=\left[ \dfrac{2\sqrt{a}+1}{2\sqrt{ab}+1}+\dfrac{2\left( \sqrt{ab}+\sqrt{a}\right) }{2\sqrt{ab}-1}-1\right]:\left[ \dfrac{2\sqrt{a}+1}{2\sqrt{ab}+1}-\dfrac{2\left( \sqrt{ab}+\sqrt{a}\right) }{2\sqrt{ab}-1}+1\right]. $$
		\begin{enumerate} 
		\item Rút gọn biểu thức $P$.
		\item  Tìm giá trị nhỏ nhất của $P$ biết $2\sqrt{a}+\sqrt{b}=1.$
	\end{enumerate}

	\loigiai{
		\begin{enumerate} 
		\item Điều kiện $a\ge0, b\ge 0, ab\not=\dfrac{1}{4}$.
		Ta có
		\begin{eqnarray*}
			\dfrac{2\sqrt{a}+1}{2\sqrt{ab}+1}+\dfrac{2\left( \sqrt{ab}+\sqrt{a}\right) }{2\sqrt{ab}-1}-1&=& \dfrac{2\sqrt{a}+1}{2\sqrt{ab}+1}+ \dfrac{2\sqrt{a}+1}{2\sqrt{ab}-1}\\
			&=& \left( 2\sqrt{a}+1\right) \left(\dfrac{1}{2\sqrt{ab}+1}+ \dfrac{1}{2\sqrt{ab}-1} \right) \\
			&=& \dfrac{4\sqrt{ab}\left(2\sqrt{a}+1 \right) }{4ab-1}.
			\end{eqnarray*}
	\begin{eqnarray*}
		\dfrac{2\sqrt{a}+1}{2\sqrt{ab}+1}-\dfrac{2\left( \sqrt{ab}+\sqrt{a}\right) }{2\sqrt{ab}-1}+1&=& \dfrac{2\sqrt{a}+1}{2\sqrt{ab}+1}- \dfrac{2\sqrt{a}+1}{2\sqrt{ab}-1}\\
		&=& \left( 2\sqrt{a}+1\right) \left(\dfrac{1}{2\sqrt{ab}+1}- \dfrac{1}{2\sqrt{ab}-1} \right) \\
		&=& \dfrac{-2\left(2\sqrt{a}+1 \right) }{4ab-1}.
	\end{eqnarray*}
Do đó $$P=\dfrac{4\sqrt{ab}\left(2\sqrt{a}+1 \right) }{4ab-1} \times \dfrac{4ab-1}{-2\left(2\sqrt{a}+1 \right)}=-2\sqrt{ab}.$$
		\item  Ta có $\left(2\sqrt{a}-\sqrt{b} \right)^2=\left(2\sqrt{a}+\sqrt{b} \right)^2 -8\sqrt{ab}\ge0$\\
		Hay $-2\sqrt{ab}\ge -\dfrac{1}{4}\left(2\sqrt{a}+\sqrt{b} \right)^2=-\dfrac{1}{4}.$\\
		Dấu ' ' = ' ' xảy ra khi và chỉ khi $$\heva{&2\sqrt{a}+\sqrt{b}=1\\
		&2\sqrt{a}=\sqrt{b}}\Leftrightarrow \heva{&a=\dfrac{1}{16}\\
	&b=\dfrac{1}{4}.}$$
Vậy $\min P=-\dfrac{1}{4}$ khi và chỉ khi $a=\dfrac{1}{16}, 
b=\dfrac{1}{4}.$
	\end{enumerate}
	
  }
\end{ex}

    Giải các phương trình, hệ phương trình
   %\begin{listEX}%[3]% Bỏ trống tùy chọn số cột thì sẽ ra 1 cột
    		\begin{enumerate} 
        \item $\dfrac{3x}{x^2+x+1}+\dfrac{3x}{x^2-x+1}=4$;
        \item $\heva{&x^2+y^2=5\\

\loigiai
    {
  	\begin{enumerate} 
        \item $\dfrac{3x}{x^2+x+1}+\dfrac{3x}{x^2-x+1}=4.$\\
        Vì $x=0$ không phải là nghiệm, nên ta chia cả tử và mẫu mỗi phân thức cho $x$ ta được
        $$\dfrac{3}{x+\dfrac{1}{x}+1}+\dfrac{3}{x+\dfrac{1}{x}-1}=4.$$
       Đặt $y=x+\dfrac{1}{x},$  $|y|\ge2 $.
    \\
    Phương trình trở thành $$\dfrac{3}{y+1}+ \dfrac{3}{y-1}=4\Leftrightarrow 2y^2-3y-2=0 \Leftrightarrow\hoac{&y=2\\
    &y=-\dfrac{1}{2}(\mbox{loại.})}$$
Với $y=2$, ta có $x+\dfrac{1}{x}=2\Leftrightarrow x^2-2x+1=0\Leftrightarrow x=1.$
\\
Vậy phương trình có nghiệm duy nhất $x=1$.    
        \item $\heva{&x^2+y^2=5\\
        	&(x+y)^3+x^2+y^2=x+y-2xy+1.}$\\
        Hệ phương trình tương đương $\heva{&(x+y)^2-2xy=5\\
        	&(x+y)^3+(x+y)^2-(x+y)-1=0.}$\\
Đặt $S=x+y, P=xy$ thì hệ trở thành
$$\heva{&S^2-2P=5\qquad (1)\\
&S^3+S^2-S-1=0\qquad (2).}$$
Từ (2) $\Leftrightarrow (S-1)(S+1)^2=0\Leftrightarrow S=\pm 1.$\\
Thế $S=\pm 1$ vào (1) ta được $P=-2$.\\
Với $S=1, P=-2$ ta có phương trình $X^2-X-2=0 \Leftrightarrow\hoac{& X=-1\\ &X=2.}$
\\
Ta thu được các nghiệm $(-1;2), (2;-1)$.\\
Với $S=-1, P=-2$ ta có phương trình $X^2+X-2=0 \Leftrightarrow\hoac{& X=1\\ &X=-2.}$
\\
Ta thu được các nghiệm $(1;-2), (-2;1)$.\\
Vậy các nghiệm của hệ phương trình là $(-1;2), (2;-1), (1;-2), (-2;1). $
\end{enumerate}
    }
\end{ex}

	\begin{enumerate} 
		\item Chứng minh rằng $a\not=0$, từ đó suy ra $\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=0.$
		\item Chứng minh rằng $\dfrac{x}{z}+\dfrac{z}{y}+\dfrac{y}{x}=-3.$
	\end{enumerate} 
	\loigiai{
		\begin{enumerate} 
			\item Ta có $a=x(x-y)=y(y-z)=z(z-x)$ \qquad (1).\\
Do		 $x,y,z$ khác $0$, đôi một khác nhau	 suy ra $a\not=0$.\\
Từ (1) ta có $\dfrac{a}{x}=x-y; \dfrac{a}{y}=y-z; \dfrac{a}{z}=z-x$.\\
Suy ra $a\left( \dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\right) =x-y+y-z+z-x=0.$ Mà $a\not=0$ nên $\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=0.$
			\item Từ $\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=0$ suy ra $xy+yz+zx=0\Leftrightarrow xy=-yz-zx$ \qquad (2).\\
			 Vì thế  $$\dfrac{x}{z}+\dfrac{z}{y}+\dfrac{y}{x}=\dfrac{x^2y+y^2z+z^2x}{xyz}=\dfrac{-x(yz+zx)+y^2z+z^2x}{xyz}=\dfrac{-xyz-z\left(x^2-y^2-zx \right) }{xyz} \quad(3).$$
			 Theo giả thiết $x^2-xy=y^2-yz\rightarrow x^2-y^2=xy-yz$ \qquad (4).\\
			 Kết hợp (2), (3), (4): $\dfrac{x}{y}+\dfrac{z}{y}+\dfrac{y}{x}=\dfrac{-xyz-z\left(xy--yz-zx \right) }{xyz}=-3.$
		\end{enumerate}

\begin{ex}%[Đề thi vào 10, chuyên Lương Văn Chánh, tỉnh Phú Yên, năm 2018]%[Huỳnh Xuân Tín, Dự án EX-TS10C-18]%[9D0G7]
	Cho $x,y,z$ là các số thực dương thỏa mãn $x+y+z=2$. Chứng minh rằng $$\left(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z} \right) (2+9xyz)\ge21.$$ Đẳng thức xảy ra khi nào?
	\loigiai{Bất đẳng thức cần chứng minh tương đương với
		\begin{eqnarray*}
			& & 2\left(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z} \right)+ 9xyz\left(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z} \right)\ge 21\\
			&\Leftrightarrow& (x+y+z)\left(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z} \right)+ 9\left(xy+yz+zx \right)\ge 21\\
				&\Leftrightarrow&\left(\dfrac{x}{y}+\dfrac{y}{x}\right)+ \left(\dfrac{y}{z}+\dfrac{z}{y}\right)+ \left(\dfrac{z}{x}+\dfrac{x}{z}\right)+ 9\left(xy+yz+zx \right)\ge 18\\
					&\Leftrightarrow& \dfrac{(x+y)^2}{xy}+\dfrac{(z+y)^2}{zy}+\dfrac{(x+z)^2}{xz}+ 9\left(xy+yz+zx \right)\ge 24 \qquad(1).
			\end{eqnarray*}
		Áp dụng bất đẳng thức Cauchy ta có 
		\\
	$\dfrac{(x+y)^2}{xy}+9xy\ge 2\sqrt{\dfrac{(x+y)^2}{xy}\cdot 9xy}= 6(x+y)$	\qquad (2).\\
	$\dfrac{(z+y)^2}{zy}+9yz\ge 2\sqrt{\dfrac{(z+y)^2}{zy}\cdot 9yz}= 6(z+y)$	\qquad (3).\\
		$\dfrac{(z+x)^2}{zx}+9xz\ge 2\sqrt{\dfrac{(z+x)^2}{zx}\cdot 9xz}= 6(z+x)$	\qquad (4).\\
		Cộng (2), (3) và (4) vế theo vế ta được
	$$\dfrac{(x+y)^2}{xy}+\dfrac{(z+y)^2}{zy}+\dfrac{(x+z)^2}{xz}+ 9\left(xy+yz+zx \right)\ge 12(x+y+z)=24.$$
	Vậy (1) đúng. Bất đẳng thức đã cho được chứng minh.\\
	Đẳng thức xảy ra khi
	$$\heva{&\dfrac{(x+y)^2}{xy}=9xy\\&\dfrac{(z+y)^2}{zy}=9yz\\ &\dfrac{(z+x)^2}{zx}=9zx}\Leftrightarrow \heva{&\dfrac{x+y}{xy}=3\\ &\dfrac{z+y}{zy}=3\\&\dfrac{x+z}{xz}=3 }\Leftrightarrow \dfrac{1}{x}=\dfrac{1}{y}=\dfrac{1}{z}\Leftrightarrow x=y=z=\dfrac{2}{3}.$$
	
}
\end{ex}
