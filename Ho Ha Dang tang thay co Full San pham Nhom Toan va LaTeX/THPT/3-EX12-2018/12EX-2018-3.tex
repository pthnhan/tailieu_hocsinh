				\setcounter{x}{-1}
				\foreach \a in {#1} {

						\addtocounter{y}{1}
						\setcounter{z}{-1}
						\foreach \c in {1,...,\b} {
							\addtocounter{z}{1}
							\cube{\value{x}}{\value{y}}{\value{z}}
						}
					}
				}
			}
