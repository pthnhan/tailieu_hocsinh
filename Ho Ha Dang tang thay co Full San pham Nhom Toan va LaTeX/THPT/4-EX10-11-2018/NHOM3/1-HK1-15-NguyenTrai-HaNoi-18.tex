\begin{bt}%[Đề HK1, lớp 11, THPT Nguyễn Trãi, Hà Nội 2017]%[1D1B3]%[1D1B3]%[1D1K3]
Giải các phương trình sau
		\begin{enumerate}
			\item $\sin^2x+5\sin x\cos x+6\cos^2x=6$.
			\item $\sqrt{3}\sin x+\cos x=2$.
			\item $\cos3x-\sin2x-\cos x=0$.
		\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item \begin{itemize}
				\item $\cos x=0\Leftrightarrow\heva{&x=\dfrac{\pi}{2}+k\pi\\&\sin^2x=1-\cos^2x=1}$, khi đó pt$\Leftrightarrow1=6$ (vô lý)\\
				Loại nghiệm $x=\dfrac{\pi}{2}+k\pi$.
				\item $\cos x\ne0$, chia hai vế phương trình cho $\cos^2x$, ta được\\
				pt$\Leftrightarrow \tan^2x+5\tan x+6=6(1+\tan^2x)\Leftrightarrow5\tan^2x-5\tan x=0$\\
				$\Leftrightarrow\hoac{&\tan x=0\\&\tan x=-1}\Leftrightarrow\hoac{&x=k\pi\\&x=-\dfrac{\pi}{4}+k\pi}$, $k\in\mathbb{Z}$.
			\end{itemize}
			\item pt$\Leftrightarrow\dfrac{\sqrt{3}}{2}\sin x+\dfrac{1}{2}\cos x=1\Leftrightarrow\sin\left(x+\dfrac{\pi}{6}\right)=1\Leftrightarrow x+\dfrac{\pi}{6}=\dfrac{\pi}{2}+k2\pi\Leftrightarrow\dfrac{\pi}{3}+k2\pi$, $k\in\mathbb{Z}$.
			\item pt$\Leftrightarrow\cos3x-\cos x-\sin2z=0\Leftrightarrow-2\sin2x\sin x-\sin2x=0\Leftrightarrow-\sin2x(2\sin x+1)=0$\\
			$\Leftrightarrow\hoac{&\sin2x=0\\&\sin x=-\dfrac{1}{2}}\Leftrightarrow\hoac{&2x=k\pi\\&x=-\dfrac{\pi}{6}+k2\pi\\&x=\pi+\dfrac{\pi}{6}+k2\pi}\Leftrightarrow\hoac{&x=k\dfrac{\pi}{2}\\&x=\dfrac{\pi}{6}+k2\pi\\&x=\dfrac{7\pi}{6}+k2\pi}$, $k\in\mathbb{Z}$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề HK1, lớp 11, THPT Nguyễn Trãi, Hà Nội 2017]%[1H2B3]%[1H2B2]%[1H2K1]
	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình bình hành. Gọi $M$, $N$ lần lượt là trung điểm của $SA$, $SB$, $P$ là trọng tâm của tam giác $BCD$.
		\begin{enumerate} 
			\item Chứng minh rằng đường thẳng $MN$ song song với mặt phẳng $(SCD)$.
			\item Tìm giao tuyến của mặt phẳng $(MNP)$ và mặt phẳng $(ABCD)$.
			\item Tìm giao điểm $G$ của đường thẳng $SC$ với mặt phẳng $(MNP)$. Tính tỉ số $\dfrac{SC}{SG}$.
		\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\tkzDefPoints{0/0/A, -2/-1.5/D, 7/0/B, 0.5/6/S}
			\coordinate (C) at ($(B)+(D)-(A)$);
			\tkzInterLL(A,C)(B,D)	\tkzGetPoint{O}
			\coordinate (P) at ($(C)!0.67!(O)$);
			\coordinate (M) at ($(S)!0.5!(A)$);
			\coordinate (N) at ($(S)!0.5!(B)$);
			\tkzDefLine[parallel=through P](B,A)	\tkzGetPoint{p}
			\tkzInterLL(A,D)(P,p)	\tkzGetPoint{E}
			\tkzInterLL(D,S)(P,p)	\tkzGetPoint{F}
			\tkzInterLL(S,C)(M,P)	\tkzGetPoint{G}
			\tkzInterLL(M,G)(C,D)	\tkzGetPoint{T}
			\tkzDrawSegments(S,B B,C C,D D,S S,G F,p T,G)
			\tkzDrawSegments[dashed](S,A A,B A,C A,D B,D M,N N,P P,F M,T)
			\tkzDrawPoints(S,A,B,C,D,M,N,P,E,G)
			\tkzLabelPoints[below](G)
			\tkzLabelPoints[above left](A,S,M)
			\tkzLabelPoints[above right](B,N)
			\tkzLabelPoints[below left](C,D,P)
			\tkzLabelPoints[above](E)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate} 
			\item Ta có $MN$ là đường trung bình của $\triangle SAB$ nên $MN\parallel AB$, mà $AB\parallel CD$ nên $MN\parallel CD$.\\
			Có $\heva{&MN\parallel CD\\&CD\subset(SCD)\\&MN\not\subset(SCD)}\Rightarrow MN\parallel(SCD)$.
			\item Có $\heva{&MN\parallel AB\\&MN\subset(MNP)\\&AB\subset(ABCD)\\&P\in(MNP)\cap(ABCD)}\Rightarrow(MNP)\cap(ABCD)=Pt\parallel MN\parallel AB$.

			Vậy ta suy ra $\dfrac{SC}{SG}=\dfrac{1}{2}$.
		\end{enumerate}
	}
