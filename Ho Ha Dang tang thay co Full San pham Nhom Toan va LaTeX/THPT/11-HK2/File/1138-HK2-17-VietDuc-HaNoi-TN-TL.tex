	Tìm các giới hạn sau:
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\lim\limits_{x \rightarrow -\infty} \left(4x-\sqrt{16x^2+5}\right).$	
			\item $\lim\limits_{x \rightarrow -1} \dfrac{\sqrt{9x^2+7}-4}{-x^2+3x+4}.$
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item $\lim\limits_{x \rightarrow -\infty} \left (4x-\sqrt{16x^2+5}\right )=\lim\limits_{x \rightarrow -\infty} x\left (4+\sqrt{16+\dfrac{5}{x^2}}\right )=-\infty$\\
			vì $\heva{ &\lim\limits_{x \rightarrow -\infty} x = -\infty\\ &\lim\limits_{x \rightarrow -\infty} 4+\sqrt{16+\dfrac{5}{x^2}} = 8>0.}$  
			
			\item Ta có  \begin{align*}
			\lim\limits_{x \rightarrow -1} \dfrac{\sqrt{9x^2+7}-4}{-x^2+3x+4} & = \lim\limits_{x \rightarrow -1} \dfrac{(\sqrt{9x^2+7}-4)(\sqrt{9x^2+7}+4)}{(-x^2+3x+4)(\sqrt{9x^2+7}+4)} \\
			&= \lim\limits_{x \rightarrow -1} \dfrac{9x^2-9}{(-x^2+3x+4)(\sqrt{9x^2+7}+4)}\\
			&= \lim\limits_{x \rightarrow -1} \dfrac{9(x-1)(x+1)}{-(x+1)(x-4)(\sqrt{9x^2+7}+4)}\\
			&= \lim\limits_{x \rightarrow -1} \dfrac{9(x-1)}{-(x-4)(\sqrt{9x^2+7}+4)}\\
			&= \dfrac{-18}{-(-5) \cdot 8}=\dfrac{-9}{20}.
			\end{align*}
		\end{enumerate}
	}

	\begin{enumerate}
		\item Giải phương trình $f'(x)+g''(x)=0.$\\
		\item Viết phương trình tiếp tuyến của đồ thị hàm số $y=\dfrac{1}{x \sqrt{2}}$ tại giao điểm của đồ thị hai hàm số $y=f(x)$ và $y=g(x)$.
	\end{enumerate}

		\begin{enumerate}
			\item Ta có $f'(x)=\left(\dfrac{1}{x\sqrt{2}} \right)'=-\dfrac{1}{\sqrt{2}x^2}.$\\

			Do đó $f'(x)+g''(x)=0 \Leftrightarrow -\dfrac{1}{\sqrt{2}x^2}+\sqrt{2}=0 \Leftrightarrow x^2=\dfrac{1}{2} \Leftrightarrow x=\pm \dfrac{1}{\sqrt{2}}.$\\
			\item Xét phương trình hoành độ giao điểm $\dfrac{1}{x\sqrt{2}}= \dfrac{x^2}{\sqrt{2}} \Leftrightarrow x=1 \Rightarrow y=\dfrac{1}{\sqrt{2}}.$\\

			Vậy phương trình tiếp tuyến của đồ thị hàm số $y=\dfrac{1}{x \sqrt{2}}$ tại điểm $\left(1;\dfrac{1}{\sqrt{2}}\right)$ là    $$y=-\dfrac{1}{\sqrt{2}}(x-1)+\dfrac{1}{\sqrt{2}}.$$
		\end{enumerate}
	}
\end{ex}

			SI \in (SCD), SI \perp CD
			\end{cases} 
			\Rightarrow SI \perp (ABCD).$
			\item  Ta có $\widehat{ABI}=\widehat{ABC}-\widehat{IBC}=120^\circ-30^\circ=90^\circ$, suy ra $AB \perp IB.$\\
			Ta có
			$\begin{cases}
			AB \perp SI \text{ (Vì } SI \perp (ABCD)) \\
			AB \perp IB \text{ (cmt) }
			\end{cases}
			\Rightarrow AB \perp (SIB) \Rightarrow AB \perp SB.$\\
			Suy ra $\bigtriangleup SAB$ vuông tại $B$.
			\item Ta có $IB$ là hình chiếu của $SB$ trên $(ABCD)$ nên góc giữa $SB$ và mặt phẳng $(ABCD)$ là $\widehat{SBI}.$\\
			$\bigtriangleup SCD$ là tam giác đều cạnh $a$ suy ra $SI=\dfrac{\sqrt{3}a}{2}.$\\
			$\triangle BCD$ là tam giác đều cạnh $a$ suy ra $IB=\dfrac{\sqrt{3}a}{2}.$\\
			Do đó $\triangle SBI$ vuông cân tại $I$\\ 
			$\Rightarrow \widehat{SBI}=45^{\circ}.$\\
			Vậy góc giữa đường thẳng $SB$ và mặt phẳng $(ABCD)$ bằng $45^{\circ}.$
		\end{enumerate}
	}
\end{ex}

	Cho hình lăng trụ đứng $ABC.A'B'C'$, có đáy $ABC$ là tam giác vuông tại $C$. Biết $CB=CA=a, A A'=\dfrac{a \sqrt{6}}{2}$. Tính khoảng cách giữa hai đường thẳng $CA'$ và $C'B$.
