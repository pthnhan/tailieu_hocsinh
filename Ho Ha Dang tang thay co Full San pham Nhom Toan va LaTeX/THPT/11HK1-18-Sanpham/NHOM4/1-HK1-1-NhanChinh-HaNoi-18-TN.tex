\begin{ex}%[HK1 2017-2018 THPT Nhân Chính, Hà Nội]%[Lê Mạnh Thắng - DA 11HK1-18]%[1D2G3-2]
	Phương trình $0!+1\cdot 1!+2\cdot 2!+3\cdot 3!+ \cdots + n\cdot n!=362880$ có nghiệm $n$. Khi đó hệ số của $x^n$ trong khai triển thành đa thức của $P(x)=(x^2-x^3+1)^n$ là
	\choice
	{$756$}
	{\True $238$}
	{$328$}
	{$765$}
	\loigiai{
		Đặt $S=0!+1\cdot 1!+2\cdot 2!+3\cdot 3!+ \cdots + n\cdot n!$, Khi đó
		\begin{eqnarray*}
		&& S+1!+2!+\cdots+n!=0!+(1\cdot 1!+1!)+(2\cdot 2!+2!)+\cdots + (n\cdot n!+n!)\\
		&\Leftrightarrow& S+1!+2!+\cdots+n!=1!+2!+\cdots+(n+1)!\\
		&\Leftrightarrow& S=(n+1)!
		\end{eqnarray*}
	Theo đề, $S=362880 \;\Leftrightarrow\; (n+1)!= 362880 \;\Leftrightarrow\; n=8$.\\
	Ta có\vspace*{-9pt}
	\begin{eqnarray*}
		(x^2-x^3+1)^8&=&\displaystyle\sum_{k=0}^{8}\mathrm{C}_{8}^{k}\left(x^2\right)^{8-k}\left(1-x^3\right)^k\,=\,
		\sum_{k=0}^{8}\sum_{l=0}^{k}\mathrm{C}_{8}^{k}\,\mathrm{C}_{k}^{l}\,1^{k-l}\,x^{16-2k}\left(-x^3\right)^l\\
		&=&\sum_{k=0}^{8}\sum_{l=0}^{k}\mathrm{C}_{8}^{k}\,\mathrm{C}_{k}^{l}\,(-1)^l\,x^{16-2k+3l}.
	\end{eqnarray*}
Cho $16-2k+3l=8 \;\Leftrightarrow\; \heva{&k=7\\&l=2}$ hoặc $\heva{&k=4\\&l=0}$.\\
Suy ra, hệ số chứa $x^8$ trong khai triển của $P(x)$ là $a_8=\mathrm{C}_{8}^{7}\cdot\mathrm{C}_{7}^{2}\,(-1)^2+\mathrm{C}_{8}^{4}\cdot\mathrm{C}_{4}^{0}\,(-1)^0=238$.
}
\end{ex}

\end{itemize}
}
