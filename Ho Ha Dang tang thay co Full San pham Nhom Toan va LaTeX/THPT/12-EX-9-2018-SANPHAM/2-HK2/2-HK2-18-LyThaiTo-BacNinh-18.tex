\begin{ex}%[Đề HK2 Khối 12, Lý Thái Tổ, Bắc Ninh 2018]%[Nguyễn Thị Kiều Ngân, dự án 12EX9]%[2D1B5-6]
	Tìm tọa độ của tất cả các điểm $M$ trên đồ thị $(C)$ của hàm số $y=\dfrac{x-1}{x+1}$ sao cho tiếp tuyến của $(C)$ tại điểm $M$ song song với đường thẳng $d \colon x-2y+7=0$.
	\choice
	{$M(1;0)$ và $M(-3;2)$}
	{$M(0;1)$ và $M(2;-3)$}
	{\True $M(1;0)$}
	{$M(-3;2)$}
\loigiai{
	Tập xác định: $\mathscr{D}=\mathbb{R}\setminus \{-1\}$.\\
	Đạo hàm $y'=\dfrac{2}{(x+1)^2}$. Gọi $M(x_0;y_0)$ là tiếp điểm.\\
	$d\colon x-2y+7=0 \Leftrightarrow y=\dfrac{1}{2}x +\dfrac{7}{2}$.\\
	Tiếp tuyến của $(C)$ tại điểm $M$ song song với đường thẳng $d$ nên hệ số góc của tiếp tuyến là $\dfrac{1}{2}$ $\Rightarrow y'(x_0)=\dfrac{1}{2} \Leftrightarrow \dfrac{2}{(x_0 +1)^2} =\dfrac{1}{2} \Leftrightarrow (x_0 +1)^2 = 4 \Leftrightarrow \hoac{&x_0 =1\\&x_0 =-3.}$
	\begin{itemize}
	\item $x_0 =1 \Rightarrow y_0 =0$ $\Rightarrow$ Phương trình tiếp tuyến là $y=\dfrac{1}{2}(x-1)$ (nhận).
	\item $x_0 =-3 \Rightarrow y_0 =2$ $\Rightarrow$ Phương trình tiếp tuyến là $y=\dfrac{1}{2}(x+3) +2 \Rightarrow y=\dfrac{1}{2}x +\dfrac{7}{2}$ (loại).
	\end{itemize}
	Vậy $M(1;0)$.
	}
\end{ex}

	Ta có $\displaystyle \int \limits_0^1 \left( \dfrac{1}{x+1} -\dfrac{1}{x+2} \right) \mathrm{\,d}x 
	=\left( \ln |x+1| -\ln|x+2| \right)\Big|_0^1
	=2\ln 2 -\ln 3$.\\
	Mà $\displaystyle \int \limits_0^1 \left( \dfrac{1}{x+1} -\dfrac{1}{x+2} \right) \mathrm{\,d}x =a\ln 2 +b\ln 3$ nên $a=2$, $b=-1$.\\
