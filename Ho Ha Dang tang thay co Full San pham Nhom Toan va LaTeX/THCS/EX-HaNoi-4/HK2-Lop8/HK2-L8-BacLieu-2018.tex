\begin{ex}%[HK2, Toán 8, Bạc Liêu, năm 2017]%[8D3Y4]%[8D3K5]
Giải các phương trình sau: 
\hfill
    \begin{enumerate}    
        \item $4x - 8 = 0$
        \item $\dfrac{2x - 1}{x - 1} + 1 = \dfrac{1}{x - 1}$
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item[a)] $4x - 8 = 0 \Leftrightarrow 4x = 8 \Leftrightarrow x = 2$.\\
        Vậy phương trình đã cho có nghiệm $x = 2$.
        \item[b)] Điều kiện xác định: $x \neq 1$.
Phương trình đã cho tương đương\medskip

        $\begin{aligned}
        && \dfrac{2x - 1}{x - 1} + 1 - \dfrac{1}{x - 1} & = 0\\
        &\Leftrightarrow & \dfrac{2x - 1}{x - 1} + \dfrac{x - 1}{x - 1} - \dfrac{1}{x - 1} & = 0\\
        &\Leftrightarrow & \dfrac{(2x - 1) + (x - 1) - 1}{x - 1} & = 0\\
        & \Leftrightarrow & 3x - 3 & = 0\\
        & \Leftrightarrow & x & = 1 \quad (\text{Không TMĐK}).
        \end{aligned}$\\
        Vậy phương trình đã cho vô nghiệm.
    \end{enumerate}}
\end{ex}

\begin{ex}%[HK2, Toán 8, Bạc Liêu, năm 2017]%[8D3Y4]%[8D3K5]
Giải phương trình
\hfill
    \begin{enumerate}
        \item $2x\left(3x - 6\right) = 0$
        \item $2x^3 - 5x^2 - 3x = 0$
    \end{enumerate}
\loigiai
    {
    \begin{enumerate}
        \item[a)] Ta có: $ 2x\left(3x - 6\right) = 0
        \Leftrightarrow \heva{&2x = 0\\&3x - 6 = 0}
        \Leftrightarrow  \heva{&x = 0\\ &x = 2}.$\\
        Vậy phương trình đã cho có hai nghiệm là $x = 0$ và $x = 2$.
    \end{enumerate}
        \item[b)] Phương trình đã cho tương đương với \smallskip
        
    $\begin{aligned}
       & x\left( 2x^2 - 5x - 3 \right) & = 0 \\
       \Leftrightarrow \quad & x\left( (2x^2 - 6x) + (x - 3)\right) & = 0\\
       \Leftrightarrow \quad & x\left(x - 3\right)\left(2x + 1\right) & = 0\\
       \Leftrightarrow \quad & \heva{&x = 0\\&x = 3\\&x = -\dfrac{1}{2}}.
        \end{aligned}$ \\
        Vậy phương trình đã cho có 3 nghiệm là $x = 0, x = 3$ và $x = -\dfrac{1}{2}$.
    }
\end{ex}

\begin{ex} Cho tam giác $ABC$ vuông tại $A$, có $AB = 6$ cm, $AC = 8$ cm, và $AH$ là đường cao. 
\begin{enumerate}
\item Chứng minh $\Delta HBA$ đồng dạng với $\Delta ABC$. 
\item Chứng minh $AB^2 = HB.BC$
\item Kẻ tia phân giác góc $A$ cắt $BC$ tại $I$. Tính độ dài cạnh $BI$. 
\end{enumerate}  

\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzInit[ymin=-3,ymax=2.5,xmin=-3.5,xmax=6]
\tkzClip
\tkzDefPoints{0/2/A,-3/-2/B,5.5/-2/C,0/-2/H}
\tkzDefLine[bisector](B,A,C) \tkzGetPoint{d};
\tkzInterLL(A,d)(B,C)\tkzGetPoint{I};
\tkzLabelPoints[above left](A)
\tkzLabelPoints[below left](B)
\tkzLabelPoints[below](C)
\tkzLabelPoints[below](H)
\tkzLabelPoints[below](I)
\tkzMarkRightAngles(A,H,B B,A,C)
%\tkzDrawPoints(A,B,C)
\tkzDrawSegments(A,B B,C C,A A,H A,I)
\end{tikzpicture}
\end{center}

\begin{enumerate}
\item Xét $\Delta HBA$ và $\Delta ABC$ có\\
Góc $\widehat{CBA}$ chung\\
$\widehat{BHA} = \widehat{BAC} (=90^o)$\\
Suy ra $\Delta HBA \sim \Delta ABC$ $(g.g)$
\item Vì $\Delta HBA \sim \Delta ABC$ nên ta có $\dfrac{HB}{AB} = \dfrac{BA}{BC} \Leftrightarrow AB.BA = HB.BC \Leftrightarrow AB^2 = HB.BC.$
\item Theo giả thiết $AB = 6$ cm, $AC = 8$ cm, nên $BC = \sqrt{AB^2 + AC^2} = \sqrt{6^2 + 8^2} = 10$ cm. \smallskip

Theo tính chất đường phân giác trong tam giác ta có $\dfrac{BI}{CI} = \dfrac{AB}{AC} \Leftrightarrow \dfrac{BI}{BC-BI} = \dfrac{6}{8} \Leftrightarrow \dfrac{BI}{10 - BI} = \dfrac{3}{4} \Leftrightarrow 4BI = 30 - 3BI \Leftrightarrow 7BI = 30 \Leftrightarrow BI = \dfrac{30}{7}$ cm.\smallskip

Vậy cạnh $BI$ có độ dài $\dfrac{30}{7}$ cm.
\end{enumerate}
}
\end{ex}

\begin{ex}%[HK2, Toán 8, Bạc Liêu, năm 2017]%[8H4B6]
Cho hình hộp chữ nhật có chiều dài $6$ cm, chiều rộng $3$ cm, chiều cao $5$ cm. Hãy tính thể tích của hình hộp chữ nhật trên. 

\loigiai{Thể tích của hình hộp chữ nhật được tính bằng công thức $V = (\text{chiều dài})\times (\text{rộng})\times (\text{cao})$. Do đó thể tích của hình hộp chữ nhật theo đề bài cho là $V = 6. 3. 5 = 90$ cm$^3$.}

\end{ex}
