\begin{bt}%[Đề TS10, Tiền Giang, 2017-2018]%[Phan Hoàng Anh]%[9D1B7]
Tính giá trị của biểu thức $A=\sqrt{4-2\sqrt{3}}-\dfrac{1}{2}\sqrt{12}$.
\loigiai{Ta có \begin{eqnarray*}
		A&=&\sqrt{4-2\sqrt{3}}-\dfrac{1}{2}\sqrt{12}=\sqrt{\left(\sqrt{3}-1\right)^2}-\dfrac{1}{2}\cdot2\sqrt{3}\\
		&=&\left|\sqrt{3}-1\right|-\sqrt{3}=\sqrt{3}-1-\sqrt{3}=-1.
\end{eqnarray*}}
\end{bt}	
\begin{bt}%[Đề TS10, Tiền Giang, 2017-2018]%[Phan Hoàng Anh]%[9D4B7]
Giải phương trình $x^4+x^2-20=0$.
\loigiai{Đặt $t=x^2$, $t\ge0$, phương trình được viết lại là 
	$$t^2+t-20=0\Leftrightarrow\left[\begin{aligned}
	&t=-5\,\text{(L)}\\&t=4\,\text{(N)}
	\end{aligned}\right.\Leftrightarrow x^2=4\Leftrightarrow\left[\begin{aligned}
	&x=2\\&x=-2.
	\end{aligned}\right.$$
Vậy phương trình có hai nghiệm là $x=2$, $x=-2$.}
\end{bt}	
\begin{bt}%[Đề TS10, Tiền Giang, 2017-2018]%[Phan Hoàng Anh]%[9D3B4]
Giải hệ phương trình $\begin{cases}
3x-y=11\\
2x+y=9.
\end{cases}$
\loigiai{Ta có $\begin{cases}
	3x-y=11\\
	2x+y=9.
	\end{cases}\Leftrightarrow\begin{cases}
	5x=20\\2x+y=9
	\end{cases}\Leftrightarrow\begin{cases}
	x=4\\y=9-2x=1.
	\end{cases}$\\
Vậy hệ phương trình có nghiệm là $(4;1)$.}
\end{bt}	
\begin{bt}%[Đề TS10, Tiền Giang, 2017-2018]%[Phan Hoàng Anh]%[9D4B6]
Cho phương trình $x^2-2x-5=0$ có hai nghiệm $x_1$, $x_2$. Không giải phương trình, hãy tính giá trị các biểu thức $B=x_1^2+x_2^2$ và $C=x_1^3+x_2^3$.
	\loigiai{Theo định lý Vi-ét ta có $\begin{cases}
			x_1+x_2=-\dfrac{b}{a}=2\\x_1x_2=\dfrac{c}{a}=-5.
			\end{cases}$\\
			\begin{itemize}
				\item $B=x_1^2+x_2^2=\left(x_1+x_2\right)^2-2x_1x_2=2^2-2(-5)=4+10=14$.
				\item $C=x_1^3+x_2^3=\left(x_1+x_2\right)^3-3x_1x_2\left(x_1+x_2\right)=2^3-3(-5)(2)=38$.
			\end{itemize}}
\end{bt}
\begin{bt}%[Đề TS10, Tiền Giang, 2017-2018]%[Phan Hoàng Anh]%[9D4B2]%[9D4B5]%[9D4B5]
Trong mặt phẳng tọa độ $Oxy$, cho parabol $(P)\colon y=\dfrac{1}{2}x^2$ và đường thẳng $d\colon y=x+m$.
\begin{enumerate}
	\item Vẽ $(P)$ và $d$ trên cùng một mặt phẳng tọa độ khi $m=2$.
	\item Định các giá trị của $m$ để $d$ cắt $(P)$ tại hai điểm phân biệt $A$ và $B$.
	\item Tìm các giá trị của $m$ để độ dài đoạn thẳng $AB=6\sqrt{2}$.
\end{enumerate}
	\loigiai{\begin{enumerate}
			\item 
			\immini{Khi $m=2$ thì $d\colon y=x+2$. Ta có\\
				\begin{tabular}{c|c|c|c}
				$x$ & $-2$ & $0$ & $2$\\
				\hline
				$y=x+2$ & $0$ & $2$ & $4$
			\end{tabular}\\
			\begin{tabular}{c|c|c|c|c|c}
				$x$ & $-4$ & $-2$ & $0$ & $2$ & $4$\\
				\hline
				$y=\dfrac{1}{2}x^2$ & $8$ & $2$ & $0$ & $2$ & $8$
			\end{tabular}}
		{\begin{tikzpicture}[>=stealth,scale=0.6]
		\draw[->](-5,0)--(5,0)node[below]{$x$};
		\draw[->](0,-1)--(0,9)node[above]{$y$};
		\draw[smooth,samples=100,domain=-4.2:4.2]plot(\x,{(\x)^2/2})node[above]{$(P)$};
		\draw[domain=-3:5]plot(\x,{\x+2})node[below]{$d$};
		\draw[dashed](-4,0)node[below]{$-4$}--(-4,8)--(0,8)node[above left]{$8$}--(4,8)--(4,0)node[below]{$4$};
		\draw[dashed](-2,0)node[below]{$-2$}--(-2,2)--(0,2)node[above left]{$2$}--(2,2)--(2,0)node[below]{$2$};
		\draw[dashed](2,0)--(2,4)--(0,4)node[above left]{$4$};
		\draw[fill](0,0)node[below left]{$O$} circle(1pt) (2,0)circle(1pt) (-2,0)circle(1pt) (0,2)circle(1pt) (0,4)circle(1pt) (0,8)circle(1pt) (4,0)circle(1pt) (-4,0)circle(1pt) (-4,8)circle(1pt) (4,8)circle(1pt) (-2,2)circle(1pt) (2,2)circle(1pt) (2,4)circle(1pt);
		\end{tikzpicture}}
			\item Phương trình hoành độ giao điểm của $(P)$ và $d$ là
			$$\dfrac{1}{2}x^2=x+m\Leftrightarrow x^2-2x-2m=0.\quad(*)$$
			Để $(P)$ cắt $d$ tại hai điểm phân biệt $A$ và $B$ thì $(*)$ có hai nghiệm phân biệt\\
			$\Leftrightarrow\Delta'>0\Leftrightarrow1^2-(-2m)>0\Leftrightarrow1+2m>0\Leftrightarrow m>-\dfrac{1}{2}$.
			\item Gọi $A(x_1;y_1)$, $B\left(x_2;y_2\right)$, khi đó ta có 
			\begin{center}
				$\begin{cases}
				x_1+x_2=-\dfrac{b}{a}=2\\
				x_1x_2=\dfrac{c}{a}=-2m
				\end{cases}$ và $\begin{cases}
				y_1=x_1+m\\
				y_2=x_2+m.
				\end{cases}$
			\end{center}
		\begin{eqnarray*}
		AB&=&\sqrt{\left(x_2-x_1\right)^2+\left(y_2-y_1\right)^2}=\sqrt{2\left(x_2-x_1\right)^2}\\
		&=&\sqrt{2\left[\left(x_2+x_1\right)^2-4x_1x_2\right]}=\sqrt{2\left[2^2-4(-2m)\right]}=\sqrt{2(4+8m)}.
		\end{eqnarray*}
	Yêu cầu bài toán là $AB=6\sqrt{2}\Leftrightarrow\sqrt{2(4+8m)}=6\sqrt{2}\Leftrightarrow8+16m=72\Leftrightarrow m=4$.\\
	So điều kiện tồn tại hai nghiệm, ta suy ra $m=4$ là giá trị cần tìm.
	\end{enumerate}}
\end{bt}
\begin{bt}%[Đề TS10, Tiền Giang, 2017-2018]%[Phan Hoàng Anh]%[9D4B8]
Hai bến sông A và B cách nhau $60$ km. Một ca-nô đi xuôi dòng từ A đến B rồi ngược dòng từ B về A. Thời gian đi xuôi dòng ít hơn thời gian đi ngược dòng là $20$ phút. Tính vận tốc ngược dòng của ca-nô, biết vận tốc xuôi dòng lớn hơn vận tốc ngược dòng của ca-nô là $6$ km/h.
	\loigiai{Gọi $x$ km/h là vận tốc ngược dòng của ca-nô.\\
	Suy ra vận tốc ca-nô khi xuôi dòng là $(x+6)$ km/h.\\
	Thời gian ca-nô chạy xuôi dòng từ A đến B là $\dfrac{60}{x+6}$ giờ.\\
	Thời gian ca-nô chạy ngược dòng từ B về A là $\dfrac{60}{x}$ giờ.\\
	Do thời gian đi xuôi dòng ít hơn đi ngược dòng là $20$ phút nên
	$$\dfrac{60}{x}-\dfrac{60}{x+6}=\dfrac{1}{3}\Leftrightarrow x^2+6x-1080=0\Leftrightarrow\left[\begin{aligned}
	&x=30\,\text{(N)}\\&x=-36\,\text{(L).}
	\end{aligned}\right.$$
Vậy vận tốc ca-nô khi ngược dòng là $30$ km/h.}
\end{bt}
\begin{bt}%[Đề TS10, Tiền Giang, 2017-2018]%[Phan Hoàng Anh]%[9H3B7]%[9H3B3]%[9H3B3]%[9H3B3]
	Cho tam giác $ABC$ có ba góc nhọn $(AB<AC)$, các đường cao $AF$, $BD$ và $CE$ cắt nhau tại $H$.
	\begin{enumerate}
		\item Chứng minh tứ giác $BEDC$ nội tiếp trong một đường tròn.
		\item Chứng minh $AE\cdot AB=AD\cdot AC$.
		\item Chứng minh $FH$ là phân giác của góc $\widehat{EFD}$.
		\item Gọi $O$ là trung điểm của đoạn thẳng $BC$. Chứng minh rằng $\widehat{DOC}=\widehat{FED}$.
	\end{enumerate}
	\loigiai{
		\immini{
			\begin{enumerate}
				\item Theo giả thiết ta có $BD\perp AC$, $CE\perp AB$, suy ra hai điểm $D$, $E$ cùng nhìn $BC$ dưới một góc vuông, từ đó suy ra tứ giác $BEDC$ nội tiếp một đường tròn.
				\item Xét hai tam giác $ABC$ và $ADE$ có\\
				$\widehat{ABC}=\widehat{ADE}$ (do tứ giác $BCDE$ nội tiếp),\\
				$\widehat{ACB}=\widehat{AED}$ (do tứ giác $BCDE$ nội tiếp),\\
				suy ra $\triangle ABC\backsim\triangle ADE$ (g-g).\\
				Từ đó suy ra $\dfrac{AB}{AD}=\dfrac{AC}{AE}\Leftrightarrow AB\cdot AE=AD\cdot AC$.
			\end{enumerate}
		}{
			\begin{tikzpicture}[scale=1.2, line join=round, line cap=round]
			\tkzDefPoints{0/0/A,-2/-4/B,3/-4/C}
			\tkzDrawAltitude(B,C)(A)\tkzGetPoint{F}
			\tkzDrawAltitude(C,A)(B)\tkzGetPoint{D}
			\tkzDrawAltitude(A,B)(C)\tkzGetPoint{E}
			\pgfresetboundingbox
			\tkzInterLL(A,F)(B,D)\tkzGetPoint{H}
			\coordinate (O) at ($(C)!0.5!(B)$);
			
			\tkzDrawPoints(A,B,C,D,E,F,H,O)
			\tkzDrawSegments(A,B B,C C,A E,D D,F E,F O,D)
			
			\tkzLabelPoints[above](A)
			\tkzLabelPoints[below left](B)
			\tkzLabelPoints[below right](C)
			\tkzLabelPoints[xshift=-2pt,yshift=-4pt](H)
			\tkzLabelPoints[below](F,O)
			\tkzLabelPoints[above right](D)
			\tkzLabelPoints[above left](E)
			
			\tkzMarkRightAngles[size=0.2](B,E,C B,D,C A,F,B A,F,C)
			\end{tikzpicture}
		}
		\begin{enumerate}
			\item[c)] Do tứ giác $BCDE$ nội tiếp, suy ra $\widehat{DBE}=\widehat{DCE}$ (cùng chắn cung $\wideparen{DE}$).\hfill $(1)$\\
			Theo giả thiết ta có $\widehat{BEH}+\widehat{BFH}=180^\circ$ (do $AF$ và $CE$ là đường cao), suy ra tứ giác $BEHF$ nội tiếp, từ đó suy ra $\widehat{DBE}=\widehat{EFH}$ (cùng chắn cung $\wideparen{EH}$).\hfill $(2)$\\
			Tương tự, ta chứng minh được tứ giác $CDHF$ nội tiếp, suy ra $\widehat{DFH}=\widehat{DCE}$.\hfill $(3)$\\
			Từ $(1)$, $(2)$ và $(3)$ suy ra $\widehat{DBE}=\widehat{EFH}=\widehat{DFH}=\widehat{DCE}$, từ đó suy ra $AF$ là phân giác góc $\widehat{EFD}$.
			\item[d)] Tương tự ý c) ta chứng minh được $CE$ là phân giác góc $\widehat{DEF}$, suy ra $\widehat{DEF}=2\widehat{HEF}$.\hfill $(4)$\\
			Theo chứng minh trên, tứ giác $BEHF$ nội tiếp, suy ra $\widehat{HEF}=\widehat{HBF}$ (cùng chắn cung $\wideparen{HF}$).\hfill $(5)$\\
			Mặt khác, do tam giác $BCD$ vuông tại $D$ và $O$ là trung điểm cạnh huyền $BC$ nên ta có $OB=OC=OD$, suy ra tam giác $OBD$ cân tại $O$, suy ra $\widehat{DOC}=2\widehat{DBO}=2\widehat{HBF}$.\hfill $(6)$\\
			Từ $(4)$, $(5)$ và $(6)$ suy ra $\widehat{DOC}=\widehat{FED}$.
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Đề TS10, Tiền Giang, 2017-2018]%[Phan Hoàng Anh]%[9H4B1]
	Một hình trụ có diện tích xung quanh bằng $256\pi$ cm$^2$ và bán kính đáy bằng $\dfrac{1}{2}$ đường cao. Tính bán kính đáy và thể tích của hình trụ.
	\loigiai{
		Giả sử bán kính đáy hình trụ là $R$ ($R>0$), suy ra chiều cao của hình trụ là $h=2R$.\\
		Theo bài ra ta có diện tích xung quanh của hình trụ là 
		\begin{center}
			$S_{\mathrm{xq}}=2\pi Rh=4\pi R^2=256\pi\Rightarrow R=8$ cm.
		\end{center}
		Từ đó suy ra thể tích của hình trụ là $V=h\pi R^2=2\cdot8\cdot\pi\cdot8^2=1024$ cm$^3$. 
	}
\end{bt}
