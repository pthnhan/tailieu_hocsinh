\begin{ex}%[Đề thi HSG 9, tỉnh ĐakLak, năm 2018]%[Nguyễn Quang Hiệp, Dự án EX-9G-18]%[9D4G1]%[9D4G6]
	\begin{enumerate}
		\item Cho phương trình $x^2+2(2m-3)x+m^2=0,$ với $m$ là tham số. Tìm tất cả các giá trị của $m$ để phương trình có hai nghiệm khác không $ x_1,x_2$ (chúng có thể trùng nhau) và biểu thức $\dfrac{1}{x_1}+\dfrac{1}{x_2}$ đạt giá trị nhỏ nhất.
		\item Cho parabol $(P):\ y=ax^2.$ Tìm điều kiện của $a$ để trên $(P)$ có điểm $A(x_0;y_0)$ với hoành độ dương thỏa mãn điều kiện $\sqrt{x^2_0+1}-\sqrt{y_0+4}=x_0- \sqrt{y_0+3}.$
	\end{enumerate}

	\loigiai{
	\begin{enumerate}
		\item Phương trình có hai nghiệm khác $0 \Leftrightarrow \heva{&(2m-3)^2-m^2 \geq 0 \\ &m^2 \neq 0} \Leftrightarrow \heva{&(m-3)(m-1) \geq 0 \\ &m \neq 0}$ \\
		$\Leftrightarrow \heva{&\hoac{&m \geq 3 \\ & m \leq 1}\\ &m \neq 0} \Leftrightarrow \hoac{&\heva{&m \leq 1 \\ & m \neq 0}\\&m \geq 3.} \hfill (\ast)$\\
		Theo hệ thức Vi-ét ta có $\heva{&x_1+x_2=-4m+6\\&x_1x_2=m^2.}$\\ \smallskip
		Khi đó \\
		$\begin{aligned}
			\dfrac{1}{x_1}+\dfrac{1}{x_2}=\dfrac{x_1+x_2}{x_1x_2} 
			&=\dfrac{-4m+6}{m^2}=\dfrac{-12m+18}{3m^2}\\
			&=\dfrac{-2m^2+2(m^2-6m+9)}{3m^2}=-\dfrac{2}{3}+\dfrac{2(m-3)^2}{3m^2} \geq -\dfrac{2}{3}.
		\end{aligned}$\\
		Đẳng thức xảy ra khi $m=3$ (TMĐK $(\ast)$).\\
		Vậy $\min \left(\dfrac{1}{x_1} +\dfrac{1}{x_2}\right)=-\dfrac{2}{3}$ khi $m=3.$

		\item  Ta có \\
		$\begin{aligned}
			\sqrt{x^2_0+1} - \sqrt{y_0+4} = x_0 - \sqrt{y_0+3}
			& \Leftrightarrow \sqrt{x^2_0+1} - x_0 = \sqrt{y_0+4} - \sqrt{y_0+3} \\
			& \Leftrightarrow \dfrac{1}{\sqrt{x^2_0+1} + x_0} = \dfrac{1}{\sqrt{y_0+4} + \sqrt{y_0+3}}\\
			& \Leftrightarrow \sqrt{x^2_0+1} + x_0 = \sqrt{y_0+4} + \sqrt{y_0+3}
		\end{aligned}$\\
		Từ đó ta có $\heva{&\sqrt{x^2_0+1}-x_0=\sqrt{y_0+4} - \sqrt{y_0+3}\\ &\sqrt{x^2_0+1} + x_0 = \sqrt{y_0+4} + \sqrt{y_0+3}}$\\
		$ \Rightarrow \sqrt{x^2_0+1} = \sqrt{y_0+4} \Leftrightarrow x^2_0+1 = y_0 +4$\\
		$ \Rightarrow (1-a)x^2_0=3 \Rightarrow x^2_0 = \dfrac{3}{1-a}>0 \Rightarrow a<1 $.
	\end{enumerate}}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh ĐakLak, năm 2018]%[Nguyễn Quang Hiệp, Dự án EX-9G-18]%[9D0G5]%[9D0G4]
	\begin{enumerate}
		\item Tìm tất cả các cặp số nguyên dương $(x;y)$ thỏa mãn: $x^2-y^2+4x-2y=18$
		\item Tìm tất cả các cặp số $(a;b)$ nguyên dương thỏa mãn hai điều kiện:
		\begin{enumerate}[i)]
			\item $a,b$ đều khác $1$ và ước số chung lớn nhất của $a,b$ là $1.$
			\item Số $N=ab(ab+1)(2ab+1)$ có đúng $16$ ước số nguyên dương.
		\end{enumerate}
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $x^2-y^2 +4x-2y=18 \Leftrightarrow (x^2+4x+4)-(y^2+2y+1)=21 \Leftrightarrow (x-y+1)(x+y+3)=21.$\\
			Do $x,y$ nguyên dương nên ta có các trường hợp sau
			\begin{enumerate}[+]
				\item $\heva{&x-y+1=1\\&x+y+3=21} \Leftrightarrow \heva{&x-y=0\\&x+y=18} \Leftrightarrow \heva{&x=9\\&y=9.}$
				\item $\heva{&x-y+1=3\\ &x+y+3=7} \Leftrightarrow \heva{&x-y=2\\ &x+y=4} \Leftrightarrow \heva{&x=3\\ &y=1.}$
			\end{enumerate}
			Vậy các nghiệm nguyên cần tìm là $S=\left \lbrace  (9;9),(3;1) \right \rbrace .$
			\item Ta có $N=ab(ab+1)(2ab+1)$ chia hết cho các số: $1; a; b(ab+1)(2ab+1);b;a(ab+1)(2ab+1);$\\ $ab+1;ab(2ab+1);2ab+1;ab(ab+1);N;ab;(ab+1)(2ab+1);b(ab+1);a(2ab+1);a(ab+1);$\\ $b(2ab+1)$ có $16$ ước dương.\\
			Nên để $N$ chỉ có đúng $16$ ước dương thì $a;b;ab+1;2ab+1$ là số nguyên tố.\\ Do $a.b > 1 \Rightarrow ab+1 >2.$
			\begin{enumerate}[+]
				\item Nếu $a;b$ cùng lẻ thì $ab+1$ chia hết cho $2$ nên là hợp số (vô lý). Do đó không mất tính tổng quát, giả sử $a$ chẵn $b$ lẻ, suy ra $a=2.$
				\item Ta cũng có nếu $b$ không chia hết cho $3 \Leftrightarrow \hoac{&b=3k+1\\&b=3k+2.}$\\
				\begin{itemize}
					\item $b=3k+1 \Rightarrow ab+1 = 2b+1=2(3k+1)+1=6k+3\ \vdots\ 3$ là hợp số (vô lý).
					\item $b=3k+2 \Rightarrow 2ab+1 = 4b+1 = 4(3k+2)+1=12k+9\vdots\ 3$ là hợp số (vô lý).
				\end{itemize}
			\end{enumerate}
 			Vậy $a=2;b=3.$
		\end{enumerate}

	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh ĐakLak, năm 2018]%[Nguyễn Quang Hiệp, Dự án EX-9G-18]%[9D0G7]
	Với $x,y$ là hai số thực thỏa mãn $y^3+3y^2+5y+3=11 \sqrt{9-x^2}- \sqrt{9x^4-x^6}.$ Tìm giá trị lớn nhất và giá trị nhỏ nhất của biểu thức $T=x-y+2018.$
	\loigiai{
	 	Điều kiện $-3 \leq x \leq 3.$\\  
	 	$y^3+3y^2+5y+3=11 \sqrt{9-x^2} - \sqrt{9x^4-x^6}
	 		\Leftrightarrow (y+1)^3 +2(y+1) = \left(\sqrt{9-x^2}\right)^3 + 2\sqrt{9-x^2}$\\
	 		Đặt $a=y+1; b=\sqrt{9-x^2}$. Khi đó ta có \\
	 	$a^3 +2a = b^3 + 2b \Leftrightarrow (a^3-b^3)+2(a-b)=0 \Leftrightarrow (a-b)(a^2+ab+b^2+2)=0.$\\
	 	Ta thấy $a^2+ab + b^2 + 2 = \left(a+\dfrac{1}{2}b \right)^2 +\dfrac{3}{4}b^2 +2 >0.$ Do đó\\
	 	$a-b=0 \Leftrightarrow a = b \Leftrightarrow y +1 - \sqrt{9-x^2} = 0 \Leftrightarrow y = \sqrt{9-x^2}-1 $\\
	 	$  \Leftrightarrow x-y=x- \sqrt{9-x^2}+1 = 4 - (3-x + \sqrt{9-x^2})$\\
	 	Vì $ -3 \leq x \leq 3$ nên $3-x \geq 0; \sqrt{9-x^2} \geq 0 \Rightarrow 3 - x + \sqrt{9-x^2} \geq 0. $ \\
	 	Vậy $x - y \leq 4 \Rightarrow T = x-y + 2018 \leq 4 + 2018 = 2022.$\\
	 	Đẳng thức xảy ra $\Leftrightarrow \heva{&3-x=0\\ &9-x^2 = 0} \Leftrightarrow x = 3 \Rightarrow y=-1.$\\
	 	Vậy $T_{\max}=2022 \Leftrightarrow x=3;y=-1.$\\
	 	Ta chứng minh $x-y \geq 1-3 \sqrt{2}.$ Thật vậy\\
	 	$x-y \geq 1- 3 \sqrt{2} \Leftrightarrow x - \sqrt{9-x^2}+1 \geq 1 - 3 \sqrt{2} \Leftrightarrow x + 3 \sqrt{2} \geq \sqrt{9-x^2} \Leftrightarrow x^2 +6 \sqrt{2}x +18 \geq 9-x^2$\\
	 	$\Leftrightarrow 2x^2=6 \sqrt{2}x +9 \geq 0 \Leftrightarrow (\sqrt{2}x+3)^2$ (luôn đúng).\\
	 	$\Rightarrow T = x - y +2018 \geq 1 - 3 \sqrt{2}+2018=2019- 3 \sqrt{2}.$\\
	 	Đẳng thức xảy ra $\Leftrightarrow x = -\dfrac{3 \sqrt{2}}{2}\ (TM) \Rightarrow y = \dfrac{3 \sqrt{2}-2}{2}$\\
	 	Vậy $T_{\min} = 2019-3 \sqrt{2} \Leftrightarrow x=-\dfrac{3\sqrt{2}}{2};y=\dfrac{3 \sqrt{2}}{2}.$}
\end{ex}
