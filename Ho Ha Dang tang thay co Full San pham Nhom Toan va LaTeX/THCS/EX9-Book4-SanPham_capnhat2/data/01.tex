\begin{center}
 \large \textbf{TỔNG QUAN VỀ CHUYÊN ĐỀ}
\end{center}
\noindent{Các} hình tam giác, tứ giác được biết đến từ các lớp dưới. Với những kiến thức hình học của lớp 7, lớp 8, chúng ta sẽ phát hiện ra rất nhiều tính chất thú vị về độ dài, về góc, về tính song song, vuông góc, thẳng hàng... từ những hình tưởng như đơn giản.
Các bài toán trong chuyên đề này gồm đủ các dạng như: tính toán, chứng minh, dựng hình, tìm giá trị nhỏ nhất, giá trị lớn nhất. Chúng ta đòi hỏi phải vẽ thêm đường phụ, điều đó sẽ mang đến cho chúng ta nhiều cảm hứng và lợi ích trong giải toán.

\subsection{TAM GIÁC}
\begin{vd}
    Cho tam giác $ABC$ vuông tại $A$, $\widehat{B}=60^\circ$, điểm $M$ thuộc cạnh $BC$ cùng với điểm $A$ chia chu vi tam giác $ABC$ thành hai phần bằng nhau (tức là $AB+BM=AC+CM$). Tính góc $AMB$.

\subsubsection{Đa giác}
\begin{bt}
	Tính các góc của một đa giác có số đo các góc tăng đều từ $90^ \circ$ đến $126^ \circ$.
	\loigiai{
		Xét đa giác $n$ cạnh có số đo các góc tăng dần là $a_1,a_2,\cdots,a_n$. Tổng các góc trong bằng $\dfrac{(a_1+a_n)n}{2}=\dfrac{(90^{\circ}+126^{\circ})n}{2}=108^{\circ}n$.\\
		Giải phương trình $(n-2)180^{\circ}=108^{\circ}n$ ta được $n=5$.\\
		Gọi $d$ là độ tăng dần của các góc. Do $n=5$ nên $90^{\circ}+4d=126^{\circ}$ nên $d=9^{\circ}$.\\
		Số đo các góc của đa giác là $90^{\circ},99^{\circ},108^{\circ},117^{\circ},126^{\circ}$.
	}
\end{bt}
\begin{bt}
