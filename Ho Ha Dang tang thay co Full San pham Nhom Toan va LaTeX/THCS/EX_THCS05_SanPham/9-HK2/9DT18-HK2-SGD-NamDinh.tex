\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Sở Giáo Dục Nam Định]%[9D1B8]% [9D1K8]
Cho biểu thức $P=\dfrac{1}{2\left(1+\sqrt{x}\right)}+\dfrac{1}{2\left(1-\sqrt{x}\right)}-\dfrac{x^2+2}{1-x^3}$ với $x\geq 0,\ x\neq 1.$
\begin{enumerate}
\item Rút gọn $P.$
\item Chứng minh rằng với $x \geq 0,\ x\neq 1$ thì $P<0$.
\end{enumerate}
\loigiai{\hfill
\begin{enumerate}
\item Ta có $$\begin{aligned} P&=\dfrac{1-\sqrt{x}+1+\sqrt{x}}{2\left(1-\sqrt{x}\right)\cdot \left(1+\sqrt{x} \right)}-\dfrac{x^2+2}{1-x^3}\\&= \dfrac{1}{1-x}-\dfrac{x^2+2}{(1-x)(1+x+x^2)}\\&= \dfrac{1+x+x^2-(x^2+2)}{(1-x)(1+x+x^2)}=-\dfrac{1}{x^2+x+1}. \end{aligned}$$
\item Ta có $$x^2+x+1=\left(x+\dfrac{1}{2}\right)^2+\dfrac{3}{4}>0\ \forall x.$$ Suy ra $P<0.$
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Sở Giáo Dục Nam Định]%[9D4Y4]%[9D4K3]%[9D4K6]
Cho phương trình $x^2-(m+6)x+3m+9=0\ \ (1)$, với $m$ là tham số.
\begin{enumerate}
\item Giải phương trình (1) khi $m=1.$
\item Tìm tất cả các giá trị của $m$ để phương trình (1) nhận $x=1+\sqrt{2}$ là một nghiệm.
\item Giả sử phương trình (1) có hai nghiệm là $x_1,\ x_2$. Chứng minh rằng giá trị của biểu thức $ x_1^2+(m+6)x_2-m^2-9m$ là một hằng số không phụ thuộc vào $m$.
\end{enumerate}
\loigiai{\hfill
\begin{enumerate}
\item Khi $m=1$ thì phương trình (1) trở thành: 
$$x^2-7x+12=0 \Leftrightarrow (x-3)(x-4)=0 \Leftrightarrow x=3,\ x=4.$$
Vậy phương trình có hai nghiệm $x=3,\ x=4.$
\item Phương trình có một nghiệm $x=1+\sqrt{2}$ khi và chỉ khi 
$$\begin{aligned} &\left(1+\sqrt{2} \right)^2-(m+6)(1+\sqrt{2})+3m+9=0 \\&\Leftrightarrow 6-4\sqrt{2}+m(2-\sqrt{2})=0\\&\Leftrightarrow m=-\dfrac{6-4\sqrt{2}}{2-\sqrt{2}}=-\dfrac{(2-\sqrt{2})^2}{2-\sqrt{2}}=-2+\sqrt{2}.\end{aligned}$$
\item Theo định lí Vi-ét, ta có $x_1+x_2=m+6$. Mặt khác $x_1$ là nghiệm của phương trình (1) nên $x_1^2=(m+6)x_1-3m-9.$ Do đó
$$\begin{aligned} x_1^2+(m+6)x_2-m^2-9m&=(m+6)x_1-3m-9+(m+6)x_2-m^2-9m \\&= (m+6)(x_1+x_2)-m^2-12m-9 \\&=(m+6)^2-m^2-12m-9=27. \end{aligned}$$
Ta có điều phải chứng minh.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Sở Giáo Dục Nam Định]%[9D4K7]
Giải hệ phương trình $\begin{cases} \dfrac{x}{y}-\dfrac{y}{x}=\dfrac{5}{6} \\ x^2-y^2=5. \end{cases}$
\loigiai{Điều kiện $xy\neq 0.$\\
Hệ đã cho tương đương với 
$$\begin{cases}x^2-y^2=\dfrac{5}{6}xy \\x^2-y^2=5  \end{cases}\Leftrightarrow \begin{cases} xy=6\\ x^2-y^2=5 \end{cases}\Leftrightarrow \begin{cases} y=\dfrac{6}{x}\\ x^2-\left(\dfrac{6}{x}\right)^2=5. \end{cases}  $$
Ta có $$x^2-\left(\dfrac{6}{x}\right)^2=5 \Leftrightarrow x^4-5x^2-36=0 \Leftrightarrow (x^2-9)(x^2+4)=0 \Leftrightarrow x^2=9 \Leftrightarrow x=\pm 3.$$
Vậy nghiệm của hệ là $(x;y)=(3;2),\ (-3;-2).$
}
\end{ex}

Cho tam giác $ABC$ vuông ở $A,$ kẻ đường cao $AH$. Đường tròn tâm $I$ đường kính $AH$ cắt các cạnh $AB$, $AC$ lần lượt tại $M$ và $N$ ($M,\ N$ không trùng $A$). Gọi $D$ là trung điểm $HC$.

\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2017-2018, Sở Giáo Dục Nam Định]%[9D1G1]%[9D5G1]
\hfill
\begin{enumerate}
\item Tìm giá trị nhỏ nhất của biểu thức $\dfrac{x^2+5}{\sqrt{x^2+4}}.$
\item Cho các số thực $a,\ b,\ c$ thỏa mãn $a\neq 0$; $c>0$; $a-b+c<0$. Chứng minh phương trình $ax^2+bx+c=0$ (ẩn $x$) có hai nghiệm phân biệt.
\end{enumerate}
\loigiai{\hfill
\begin{enumerate}
\item Ta có $$P=\dfrac{x^2+5}{\sqrt{x^2+4}}=\dfrac{(x^2+4)+1}{\sqrt{x^2+4}}=\sqrt{x^2+4}+\dfrac{1}{\sqrt{x^2+4}}=\left(\sqrt{x^2+4}+\dfrac{4}{\sqrt{x^2+4}}\right)-\dfrac{3}{\sqrt{x^2+4}}.$$
Áp dụng bất đẳng thức Cô si, ta có $$\sqrt{x^2+4}+\dfrac{4}{\sqrt{x^2+4}} \geq 2\sqrt{\sqrt{x^2+4}\cdot \dfrac{4}{\sqrt{x^2+4}}}=4.$$
Do $\sqrt{x^2+4} \geq 2$, nên $\dfrac{3}{\sqrt{x^2+4}} \leq \dfrac{3}{2}$.\\
Suy ra $P \geq 4-\dfrac{3}{2}=\dfrac{5}{2}$. Đẳng thức xả ra khi $x=0.$\\
Vậy GTNN của $P$ bằng $\dfrac{5}{2}.$
\item Từ giả thiết ta suy ra $ac-bc+c^2<0 \Rightarrow ac <bc-c^2$. Suy ra 
$$\Delta =b^2-4ac>b^2-4(bc-c^2)=(b-2c)^2 \geq 0.$$
Suy ra phương trình đã cho luôn có hai nghiệm phân biệt.
\end{enumerate}
}
\end{ex}
