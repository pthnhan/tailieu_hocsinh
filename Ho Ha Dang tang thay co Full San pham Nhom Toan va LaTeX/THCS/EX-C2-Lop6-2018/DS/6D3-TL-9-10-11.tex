\section{Phép trừ phân số}
\subsection{Trọng tâm kiến thức}
\subsubsection{Số đối}
\begin{dn}
	Hai số gọi là đối nhau nếu tổng của chúng bằng $0$.
\end{dn}
Số đối của $\dfrac{a}{b}$ kí hiệu là $-\dfrac{a}{b}$.
\begin{note}
	Ta có $-\dfrac{a}{b}=\dfrac{-a}{b}=\dfrac{a}{-b}$.
\end{note}
\subsubsection{Phép trừ phân số}
\begin{dl}
Muốn trừ một phân số cho một phân số, ta cộng số bị trừ với số đối của số trừ.
$$\dfrac{a}{b}-\dfrac{c}{d}=\dfrac{a}{b}+\left(-\dfrac{c}{d}\right).$$
\end{dl}
\subsection{Các dạng bài tập và phương pháp giải}
\begin{dang}{Tìm đối số của số cho trước}
Số đối của $\dfrac{a}{b}$ là $\dfrac{-a}{b}$.\\
Số đối của $\dfrac{-a}{b}$ là $\dfrac{a}{b}$.
\end{dang}
\begin{vd}%[Nguyễn Cao Cường, dự án EX-C2-6-2019]%[6D3B9]
Tìm số đối của các số sau
$\dfrac{-7}{10};\dfrac{-3}{-8}; -\dfrac{5}{2}; \dfrac{1}{6};0$.
	\loigiai{
		\begin{itemize}
			\item Số đối của $\dfrac{-7}{10}$ là  $\dfrac{7}{10}$;
				\item Số đối của $\dfrac{-3}{-8}$ là  $\dfrac{-3}{8}$;
					\item Số đối của $-\dfrac{5}{2}$ là  $\dfrac{5}{2}$;
						\item Số đối của $\dfrac{1}{6}$ là  $-\dfrac{1}{6}$ (hoặc $\dfrac{-1}{6}$ hoặc $\dfrac{1}{-6}$);
							\item Số đối của $0$ là  $0$.
		\end{itemize}
	}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường, dự án EX-C2-6-2019]%[6D3B9]
	Trong các phân số sau, những phân số nào đối nhau?
	$$-\dfrac{9}{21}; -\dfrac{5}{11}; \dfrac{-5}{11}; \dfrac{3}{7}$$
	\loigiai{
	Ta có $\dfrac{-9}{21}=\dfrac{-9:3}{21:3}=\dfrac{-3}{7}$.\\ Do $\dfrac{-3}{7}$ và $\dfrac{3}{7}$ là hai số đối của nhau nên $\dfrac{-9}{21}$ là số đối của $\dfrac{3}{7}$.\\
	Còn lại hai phân số $-\dfrac{5}{11}$ và $\dfrac{-5}{11}$ không đối nhau mà chúng bằng nhau.
	}
\end{vd}

\begin{dang}{Trừ phân số}
	Muốn lấy một phân số trừ đi một phân số, ta có thể làm theo các bước
	\begin{itemize}
		\item Quy đồng mẫu các phân số.
		\item Lấy tử thứ nhất trừ đi tử thứ hai và giữ nguyên mẫu chung.
	\end{itemize}
\end{dang}
\begin{vd}%[Nguyễn Cao Cường, dự án EX-C2-6-2019]%[6D3B9]
	Tính
	\begin{enumEX}{2}
		\item $\dfrac{45}{90}-\dfrac{1}{6}$;
		\item $\dfrac{8}{15}-\dfrac{17}{60}$.
		\end{enumEX}
	\loigiai{
	\begin{enumEX}{2}
		\item $\dfrac{45}{90}-\dfrac{1}{6}=\dfrac{1}{2}-\dfrac{1}{6}=\dfrac{3}{6}-\dfrac{1}{6}=\dfrac{2}{6}=\dfrac{1}{3}$;
		\item $\dfrac{8}{15}-\dfrac{17}{60}=\dfrac{32}{60}-\dfrac{17}{60}=\dfrac{15}{60}=\dfrac{1}{4}$.
	\end{enumEX}
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường, dự án EX-C2-6-2019]%[6D3B9]
	Tính 
	\begin{enumEX}{2}
		\item $\dfrac{-25}{56}-\dfrac{16}{21}$;
		\item $9-\dfrac{12}{15}$.
		\end{enumEX}
	\loigiai{
	\begin{enumEX}{2}
		\item $\dfrac{-25}{56}-\dfrac{16}{21}=\dfrac{-75}{168}-\dfrac{128}{168}=\dfrac{-203}{168}=\dfrac{-29}{24}$;
		\item $9-\dfrac{12}{15}=\dfrac{9}{1}-\dfrac{4}{5}=\dfrac{45}{5}-\dfrac{4}{5}=\dfrac{41}{5}$.
	\end{enumEX}
}
\end{vd}

\begin{dang}{Thực hiện một dãy các phép tính cộng và trừ phân số}
	Viết các phân số có mẫu âm thành các phân số bằng nó nhưng có mẫu dương.
	\begin{itemize}
		\item Thay các phép trừ bằng phép cộng với số đối.
		\item Thực hiện các phép cộng và rút gọn kết quả (nếu có thể).
	\end{itemize}
\begin{note}
	Tùy theo đặc điểm của các số hạng mà có thể áp dụng các tính chất giao hoán, kết hợp để nhóm các số hạng thích hợp thành từng nhóm rồi cộng cho hợp lí.
\end{note}
	\end{dang}
\begin{vd}%[Nguyễn Cao Cường, dự án EX-C2-6-2019]%[6D3B9]
	Thực hiện các phép tính $\dfrac{11}{-16}-\dfrac{7}{20}+\dfrac{43}{80}$.
	\loigiai{
	Ta có
	\begin{eqnarray*}
		& & \dfrac{11}{-16}-\dfrac{7}{20}+\dfrac{43}{80}=\dfrac{-11}{16}+\dfrac{-7}{20}+\dfrac{43}{80}=\dfrac{-55}{80}+\dfrac{-28}{80}+\dfrac{45}{80}=\dfrac{-40}{80}=-\dfrac{1}{2}.
		\end{eqnarray*}
}
\end{vd}
\begin{vd}%[6D3B9]
	Thực hiện các phép tính
	\begin{enumEX}{2}
		\item $1-\dfrac{-1}{4}+\dfrac{-5}{9}-\dfrac{7}{12}$;
		\item $\dfrac{25}{48}+\dfrac{7}{12}-\dfrac{9}{16}-\dfrac{1}{8}$.
		\end{enumEX}
	\loigiai{
	\begin{enumEX}{1}
		\item $1-\dfrac{-1}{4}+\dfrac{-5}{9}-\dfrac{7}{12}=\dfrac{36}{36}+\dfrac{9}{36}+\dfrac{-20}{36}+\dfrac{-21}{36}=\dfrac{4}{36}=\dfrac{1}{9}$;
		\item $\dfrac{25}{48}+\dfrac{7}{12}-\dfrac{9}{16}-\dfrac{1}{8}=\dfrac{25}{48}+\dfrac{28}{48}+\dfrac{-27}{48}+\dfrac{-6}{48}=\dfrac{20}{48}=\dfrac{5}{12}$.
	\end{enumEX}
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường, dự án EX-C2-6-2019]%[6D3B9]
	Tính bằng cách hợp lí nhất
	\begin{enumEX}{2}
		\item $\dfrac{7}{15}-\left(\dfrac{2}{15}-\dfrac{12}{18}\right)$;
		\item $\left(\dfrac{7}{41}-\dfrac{4}{9}\right)-\left(\dfrac{3}{19}+\dfrac{7}{41}\right)+\left(\dfrac{4}{9}-\dfrac{16}{19}\right)$.
		\end{enumEX}
	\loigiai{
	\begin{enumEX}{1}
		\item Ta có  $$\dfrac{7}{15}-\left(\dfrac{2}{15}-\dfrac{12}{18}\right)=\dfrac{7}{15}-\dfrac{2}{15}+\dfrac{12}{18}=\dfrac{5}{15}+\dfrac{12}{18}=\dfrac{1}{3}+\dfrac{2}{3}=1;$$
		\item Ta có
		\begin{eqnarray*}
&&\left(\dfrac{7}{41}-\dfrac{4}{9}\right)-\left(\dfrac{3}{19}+\dfrac{7}{41}\right)+\left(\dfrac{4}{9}-\dfrac{16}{19}\right)=\dfrac{7}{41}-\dfrac{4}{9}-\dfrac{3}{19}-\dfrac{7}{41}+\dfrac{4}{9}-\dfrac{16}{19}\\
&=& \left(\dfrac{7}{41}-\dfrac{7}{41}\right)+\left(-\dfrac{4}{9}+\dfrac{4}{9}\right)+\left(-\dfrac{3}{19}-\dfrac{16}{19}\right)= 0+0+(-1)=-1.
		\end{eqnarray*}
	\end{enumEX}

}
\end{vd}
\begin{dang}{Tìm số hạng chưa biết một tổng một hiệu}
	\begin{itemize}
		\item Một số hạng của tổng bằng tổng trừ đi số hạng kia.
		\item Số bị trừ bằng hiệu cộng với số trừ.
		\item Số trừ bằng số bị trừ trừ đi hiệu.
	\end{itemize}
	\end{dang}
\begin{vd}%[Nguyễn Cao Cường, dự án EX-C2-6-2019]%[6D3B9]
	Tìm $x$, biết
	\begin{enumEX}{2}
		\item $\dfrac{-13}{24}+x=\dfrac{9}{20}$;
		\item $x+\dfrac{13}{15}=-\dfrac{2}{5}$.
		\end{enumEX}
	\loigiai{
	\begin{enumEX}{1}
		\item Ta có $\dfrac{-13}{24}+x=\dfrac{9}{20}$, suy ra
		\begin{eqnarray*}
		&x&=\dfrac{9}{20}-\dfrac{-13}{24}\\
		&x&=\dfrac{54}{120}+\dfrac{65}{120}=\dfrac{119}{120};
		\end{eqnarray*}
	\item Ta có $x+\dfrac{13}{15}=-\dfrac{2}{5}$, suy ra
	\begin{eqnarray*}
	&x&=-\dfrac{2}{5}-\dfrac{13}{15}\\
	&x&=-\dfrac{19}{15}.
	\end{eqnarray*}
		\end{enumEX}
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường, dự án EX-C2-6-2019]%[6D3B9]
	Tìm $x$, biết
	\begin{enumEX}{2}
		\item $\dfrac{29}{30}-x=\dfrac{43}{60}$;
		\item $x-\dfrac{30}{75}=\dfrac{16}{30}$.
		\end{enumEX}
	\loigiai{
	\begin{enumEX}{1}
	\item Ta có $\dfrac{29}{30}-x=\dfrac{43}{60}$, suy ra
	\begin{eqnarray*}
	&x&=\dfrac{29}{30}-\dfrac{43}{60}\\
	&x&=\dfrac{58}{60}-\dfrac{43}{60}=\dfrac{15}{60}=\dfrac{1}{4};
	\end{eqnarray*}	
\item Ta có $x-\dfrac{30}{75}=\dfrac{16}{30}$ suy ra $x-\dfrac{2}{5}=\dfrac{4}{5}$ nên
$$x=\dfrac{4}{5}+\dfrac{2}{5}=\dfrac{6}{5}.$$
		\end{enumEX}
}
\end{vd}
\begin{vd}%[Nguyễn Cao Cường, dự án EX-C2-6-2019]%[6D3B9]
	Tìm $x\in \mathbb{Z}$, biết $\dfrac{1}{9}-\dfrac{x}{6}=\dfrac{-13}{18}$.
	\loigiai{
	Ta có $\dfrac{1}{9}-\dfrac{x}{6}=\dfrac{-13}{18}$, suy ra 
	\begin{eqnarray*}
	&\dfrac{x}{6}&=\dfrac{1}{9}-\dfrac{-13}{18}\\
	&\dfrac{x}{6}&=\dfrac{2}{18}+\dfrac{13}{18}\\
	&\dfrac{x}{6}&=\dfrac{15}{18}.
	\end{eqnarray*}
Do đó $x=\dfrac{6\cdot 15}{18}=5$.
}
\end{vd}
\begin{dang}{Các bài toán dẫn đến phép trừ phân số}
Căn cứ vào đề bài, lập phép trừ phân số thích hợp.
\end{dang}
\begin{vd}%[Nguyễn Cao Cường, dự án EX-C2-6-2019]%[6D3B9]
	Một đội công nhân sửa đường trong ba ngày. Ngày đầu sửa được $\dfrac{9}{10}$ km, ngày thứ hai sửa hơn ngày đầu $\dfrac{1}{4}$ km, ngày thứ ba sửa ít hơn ngày đầu là $\dfrac{1}{5}$ km. Tính quãng đường đội công nhân đó đã sửa trong ba ngày.
	\loigiai{
	Quãng đường đội công nhân đã sửa trong ba ngày là
	\begin{eqnarray*}
	&&\dfrac{9}{10}+\left(\dfrac{9}{10}+\dfrac{1}{4}\right)+\left(\dfrac{9}{10}-\dfrac{1}{5}\right)=\dfrac{18}{20}+\dfrac{18}{20}+\dfrac{5}{20}+\dfrac{18}{20}-\dfrac{4}{20}=\dfrac{55}{20}=\dfrac{11}{4}\; \text{km}.
	\end{eqnarray*}
}
\end{vd}
\begin{dang}{Tính tổng các phân số theo quy luật}
	Viết mỗi số hạng thành một hiệu của hai phân số, số trừ của nhóm trước bằng số bị trừ của nhóm sau rồi khử liên tiếp.
	\end{dang}
\begin{vd}%[Nguyễn Cao Cường, dự án EX-C2-6-2019]%[6D3K9]
	Chứng tỏ rằng
	$$\dfrac{1}{1}-\dfrac{1}{2}=\dfrac{1}{1\cdot 2};\; \dfrac{1}{2}-\dfrac{1}{3}=\dfrac{1}{2\cdot 3};\; \dfrac{1}{3}-\dfrac{1}{4}=\dfrac{1}{3\cdot 4}.$$
	Áp dụng: Tính tổng
	$$S=\dfrac{1}{1\cdot 2}+\dfrac{1}{2\cdot 3}+\dfrac{1}{3\cdot 4}+\dots+\dfrac{1}{49\cdot 50}.$$
	\loigiai{
	Ta có
	\begin{eqnarray*}
&&\dfrac{1}{1}-\dfrac{1}{2}=\dfrac{2}{2}-\dfrac{1}{2}=\dfrac{1}{2}=\dfrac{1}{1\cdot 2};\\
&&\dfrac{1}{2}-\dfrac{1}{3}=\dfrac{3}{6}-\dfrac{2}{6}=\dfrac{1}{6}=\dfrac{1}{2\cdot 3};\\
&&\dfrac{1}{3}-\dfrac{1}{4}=\dfrac{4}{12}-\dfrac{3}{12}=\dfrac{1}{12}=\dfrac{1}{3\cdot 4}.
	\end{eqnarray*}
Áp dụng
\begin{eqnarray*}
	&S&=\dfrac{1}{1\cdot 2}+\dfrac{1}{2\cdot 3}+\dfrac{1}{3\cdot 4}+\dots+\dfrac{1}{49\cdot 50}\\
	&&=\left(\dfrac{1}{1}-\dfrac{1}{2}\right)+\left(\dfrac{1}{2}-\dfrac{1}{3}\right)+...+\left(\dfrac{1}{49}-\dfrac{1}{50}\right)\\
	&&=\dfrac{1}{1}-\dfrac{1}{50}=\dfrac{49}{50}.
\end{eqnarray*}
}
\end{vd}
\begin{vd}%[Nhật Thiện, dự án EX-C2-6-2019]%[6D3K9]
	Chứng tỏ rằng
	$\dfrac{1}{3}-\dfrac{1}{5}=\dfrac{2}{3\cdot 5}$; $\dfrac{1}{5}-\dfrac{1}{7}=\dfrac{2}{5\cdot 7}$; $\dfrac{1}{7}-\dfrac{1}{9}=\dfrac{2}{7\cdot 9}$.\\
	Áp dụng: Tính tổng $S=\dfrac{2}{3\cdot 5}+\dfrac{2}{5\cdot 7}+\dfrac{2}{7\cdot 9}+\cdots+\dfrac{2}{97\cdot 99}.$
	\loigiai{
		Ta có 
		$$\dfrac{1}{3}-\dfrac{1}{5}=\dfrac{5-3}{15}=\dfrac{2}{15}=\dfrac{2}{3\cdot 5};$$
		$$\dfrac{1}{5}-\dfrac{1}{7}=\dfrac{7-5}{35}=\dfrac{2}{35}=\dfrac{2}{5\cdot 7};$$
		$$\dfrac{1}{7}-\dfrac{1}{9}=\dfrac{9-7}{63}=\dfrac{2}{63}=\dfrac{2}{7\cdot 9}.$$
		Áp dụng: Ta có \begin{eqnarray*}
			S&=&\dfrac{2}{3\cdot 5}+\dfrac{2}{5\cdot 7}+\dfrac{2}{7\cdot 9}+\cdots+\dfrac{2}{97\cdot 99}.\\
			S&=&\left(\dfrac{1}{3}-\dfrac{1}{5}\right)+\left(\dfrac{1}{5}-\dfrac{1}{7}\right)+\left(\dfrac{1}{7}-\dfrac{1}{9}\right)+\cdots+\left(\dfrac{1}{97}-\dfrac{1}{99}\right)\\
			S&=&\dfrac{1}{3}-\dfrac{1}{99}=\dfrac{32}{99}.
		\end{eqnarray*}
	}
\end{vd}

\begin{bt}%[Nhật Thiện, dự án EX-C2-6-2019]%[6D3B9]
	Tìm $x$, biết: 
	\begin{enumEX}{2}
		\item $x-\dfrac{2}{65}=\dfrac{4}{39}$;
		\item $\dfrac{1}{150}-x=\dfrac{-4}{25}$.
	\end{enumEX}
	\loigiai{
		\begin{listEX}
			\item Ta có $x-\dfrac{2}{65}=\dfrac{4}{39}$.\\
			Suy ra \begin{eqnarray*}
				x&=&\dfrac{2}{65}+\dfrac{4}{39}\\
				x&=&\dfrac{2\cdot 39+4\cdot 65}{65\cdot 39}\\
				x&=&\dfrac{338}{2535}\\
				x&=&\dfrac{2}{15}.
			\end{eqnarray*}
			Do đó $x=\dfrac{2}{15}$.
			\item Ta có $\dfrac{1}{150}-x=\dfrac{-4}{25}$.\\
			Suy ra \begin{eqnarray*}
				-x&=&\dfrac{-4}{25}-\dfrac{1}{150}\\
				-x&=&\dfrac{-24}{150}-\dfrac{1}{150}\\
				-x&=&\dfrac{-25}{150}\\
				-x&=&\dfrac{-1}{6}\\
				x&=&\dfrac{1}{6}.
			\end{eqnarray*}
			Do đó $x=\dfrac{1}{6}$.
		\end{listEX}
	}
\end{bt}
\begin{bt}%[Nhật Thiện, dự án EX-C2-6-2019]%[6D3K9]
	Cho phân số $P=\dfrac{6\cdot n-2}{8\cdot n}$ $(n\in \mathbb{N*})$. Tìm giá trị của $n$ để phân số $P$ có giá trị nhỏ nhất. Tìm giá trị nhỏ nhất đó của $P$.
	\loigiai{
		Ta có $P=\dfrac{6\cdot n-2}{8\cdot n}=\dfrac{6\cdot n}{8\cdot n}-\dfrac{2}{8\cdot n}=\dfrac{3}{4}-\dfrac{1}{4\cdot n}$.\\
		Phân số $P$ có giá trị nhỏ nhất khi và chỉ khi $\dfrac{1}{4\cdot n}$ có giá trị lớn nhất, tức là khi $4\cdot n$ có giá trị là số tự nhiên khác $0$ nhỏ nhất.\\
		Do đó $n=1$. Khi đó giá trị nhỏ nhất của $P$ là $\dfrac{3}{4}-\dfrac{1}{4}=\dfrac{1}{2}$.
	}
\end{bt}

\section{Phép nhân phân số}
\section{Tính chất cơ bản của phép nhân phân số}
\subsection{Trọng tâm kiến thức}
\begin{itemize}
	\item Muốn nhân hai phân số, ta nhân các tử với nhau và nhân các mẫu với nhau.
	$$\dfrac{a}{b}\cdot \dfrac{c}{d}=\dfrac{a\cdot c}{b\cdot d}.$$
	\item Phép nhân các phân số có các tính chất cơ bản tương tự tính chất phép nhân các số nguyên:
	\begin{enumEX}{1}
		\item Tính chất giao hoán;
		\item Tính chất kết hợp;
		\item Nhân với số $1$;
		\item Tính chất phân phối của phép nhân đối với phép cộng.
	\end{enumEX}
\end{itemize}
\subsection{Các dạng bài tập và phương pháp giải}
\begin{dang}{Nhân hai hay nhiều phân số}
	\begin{itemize}
		\item Vận dụng quy tắc $\dfrac{a}{b}\cdot \dfrac{c}{d}=\dfrac{a\cdot c}{b\cdot d}$.
		\item \begin{note}
			Nên rút gọn kết quả khi vận dụng kết quả trên.
		\end{note}
	\end{itemize}
\end{dang}
\begin{vd}%[Nhật Thiện, dự án EX-C2-6-2019]%[6D3Y10]
	Nhân các phân số sau
	\begin{enumEX}{2}
		\item $\dfrac{7}{9}\cdot \dfrac{4}{5}$;
		\item $\dfrac{-3}{8}\cdot 15$.
	\end{enumEX}
	\loigiai{
		\begin{listEX}
			\item Ta có: $\dfrac{7}{9}\cdot \dfrac{4}{5}=\dfrac{7\cdot 4}{9\cdot 5}=\dfrac{28}{49}$;
			\item Ta có: $\dfrac{-3}{8}\cdot 15=\dfrac{-3\cdot 15}{8\cdot 1}=\dfrac{-45}{8}$.
		\end{listEX}
	}
\end{vd}
\begin{vd}%[Nhật Thiện, dự án EX-C2-6-2019]%[6D3B10]
	Làm tính nhân:
	\begin{enumEX}{2}
		\item $\dfrac{-3}{8}\cdot \dfrac{10}{21}$;
		\item $\left(-\dfrac{12}{25}\right)\cdot \left(-\dfrac{35}{18}\right)$.
	\end{enumEX}
	\loigiai{
		\begin{listEX}

\begin{bt}%[Nhật Thiện, dự án EX-C2-6-2019]%[6D3K10]
