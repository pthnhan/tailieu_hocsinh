\subsection{Tóm tắt lí thuyết}
\subsubsection{Định nghĩa, sự xác định véc-tơ}
\begin{dn}[Véc-tơ]
\textit{Véc-tơ} là một đoạn thẳng có hướng.
\end{dn}
\begin{minipage}{0.65\textwidth}Véc-tơ có điểm đầu (gốc) $A$, điểm cuối (ngọn) $B$ được kí hiệu là $\protect\overrightarrow{AB}$.\\
Véc-tơ còn được kí hiệu là $\protect\overrightarrow{a}$, $\protect\overrightarrow{b}$, $\protect\overrightarrow{x}$, $\protect\overrightarrow{y}$,\ldots khi không cần chỉ rõ điểm đầu và điểm cuối của nó.\\
Một véc-tơ hoàn toàn được xác định khi biết điểm đầu và điểm cuối của nó.\\
\begin{note}
Với hai điểm phân biệt $A$ và $B$ ta chỉ có một đoạn thẳng ($AB$ hoặc $BA$), nhưng có hai véc-tơ khác nhau là $\protect\overrightarrow{AB}$ và $\protect\overrightarrow{BA}$.
\end{note}\end{minipage}\hspace*{10pt}

\begin{dn}[Độ dài véc-tơ]
Độ dài của đoạn thẳng $AB$ là độ dài (hay mô-đun) của véc-tơ $\protect\overrightarrow{AB}$, kí hiệu là $\left|\overrightarrow{AB}\right|$.
Tức là $\left|\overrightarrow{AB}\right|=AB$.\\
Đương nhiên $\left|\protect\overrightarrow{AB}\right|=\left|\protect\overrightarrow{BA}\right|$.
\end{dn}
\begin{dn}[Véc-tơ-không]
Véc-tơ-không là véc-tơ có điểm đầu và điểm cuối trùng nhau. Véc-tơ-không được kí hiệu là $\protect\overrightarrow{0}$.
\end{dn}
Ta có $\protect\overrightarrow{0}=\protect\overrightarrow{AA}=\protect\overrightarrow{BB}=\ldots$

\subsubsection{Hai véc-tơ cùng phương, cùng hướng}
\begin{dn}[Giá véc-tơ]
Giá của một véc-tơ khác $\protect\overrightarrow{0}$ là đường thẳng chứa điểm đầu và điểm cuối của véc-tơ đó.
\end{dn}
\begin{dn}[Phương, hướng véc-tơ] Hai véc-tơ được gọi là cùng phương nếu giá của chúng song song hoặc trùng nhau.
\end{dn}
Trên hình 1.3a) ta có các véc-tơ $\protect\overrightarrow{AB}$, $\protect\overrightarrow{CD}$, $\protect\overrightarrow{EF}$ cùng phương. Trên hình 1.3b) ta có $\protect\overrightarrow{AB}$ và $\protect\overrightarrow{MN}$ cùng phương, còn $\protect\overrightarrow{AB}$ và $\protect\overrightarrow{MP}$ không cùng phương.\\
Hai véc-tơ cùng phương có thể cùng hướng hoặc ngược hướng. Chẳng hạn $\protect\overrightarrow{AB}$ và $\protect\overrightarrow{CD}$ cùng hướng, $\protect\overrightarrow{AB}$ và $\protect\overrightarrow{EF}$ ngược hướng (hình 1.3a).\\

\subsubsection{Hai véc-tơ bằng nhau}
\begin{dn}[Véc-tơ bằng nhau]
Hai véc-tơ gọi là bằng nhau nếu chúng có cùng hướng và cùng độ dài.
\end{dn}

	Cho ngũ giác $ABCDE$. Có bao nhiêu véc-tơ khác véc-tơ $\vec{0}$, có điểm đầu và điểm cuối là đỉnh của ngũ giác.
	\loigiai{
		Từ hai điểm phân biệt, chẳng hạn $A$, $B$, ta xác định được hai véc-tơ khác véc-tơ-không là $\vec{AB}$, $\vec{BA}$. Mà từ năm đỉnh $A$, $B$, $C$, $D$, $E$ của ngũ giác ta có $10$ cặp điểm phân biệt, do đó có $20$ véc-tơ thỏa mãn yêu cầu bài toán.
	}
\end{bt}
\begin{bt}%[Đỗ Vũ Minh Thắng]%[0H1B1]
	Cho hình bình hành $ABCD$ có tâm là $O$. Tìm các véc-tơ từ $5$ điểm $A$, $B$, $C$, $D$, $O$
	\begin{enumerate}[a)]
		\item Bằng véc-tơ $\vec{AB}$; $\vec{OB}$.
		\item Có độ dài bằng $\left |\vec{OB}\right |$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item $\vec{AB}=\vec{DC}$; $\vec{OB}=\vec{DO}$.
			\item $\vec{BO}$, $\vec{DO}$, $\vec{OD}$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đỗ Vũ Minh Thắng]%[0H1B1]
	Cho ba điểm $A$, $B$, $C$ phân biệt thẳng hàng.
