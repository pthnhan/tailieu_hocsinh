\begin{bt}%[Tăng Lâm Tường Vinh]%[EX-TS10-ToanTT-2019]%[9D2T2]
	Với sự phát triển của khoa học kỹ thuật hiện nay, người ta tạo ra nhiều mẫu xe lăn đẹp và tiện dụng cho người khuyết tật. Công ty $A$ đã sản xuất ra những chiếc xe lăn cho người khuyết tật với số vốn ban đầu là $500$ triệu đồng. Chi phí sản xuất ra một chiếc xe lăn là $2.500.000$ đồng. Giá bán ra mỗi chiếc là $3.000.000$ đồng.
	\begin{enumerate}
		\item Viết hàm biểu diễn tổng số tiền đã đầu tư đến khi sản xuất ra được $x$ chiếc xe lăn (gồm vốn ban đầu và chi phí sản xuất) và hàm số biểu diễn số tiền thu được khi bán ra $x$ chiếc lăn.
		\dapso{$y=500+2{,}5x$; $y=3x$.}
		\item Công ty $A$ phải bán bao nhiêu chiếc xe mới có thể thu hồi vốn ban đầu.
		\dapso{$1000$ chiếc xe.}
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Tổng chi phí vốn cố định và vốn sản xuất ra $x$ chiếc xe lăn (đơn vị triệu đồng): $y=500+2{,}5x$.\\
			Hàm số biểu diễn số tiền thu được khi bán ra $x$ chiếc xe lăn là $y=3x$.
			\item Để số tiền bán được và số vốn đầu tư ban đầu bằng nhau, ta có
			$$500+2{,}5x=3x\Leftrightarrow 0{,}5x=500\Leftrightarrow x=1000.$$
			Vậy công ty $A$ phải bán $1000$ chiếc xe lăn mới thu hồi được vốn ban đầu.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Lê Vũ Hải]%[EX-TS10-ToanTT-2019]%[9D2T2]
	Với sự phát triển khoa học kỹ thuật hiện nay, người ta tạo ra nhiều mẫu xe lăn đẹp và tiện dụng cho người khuyết tật. Công ty $A$ đã sản xuất ra những chiếc xe lăn cho người khuyết tật với số vốn ban đầu là $500$ triệu. Chi phí để sản xuất một chiếc xe lăn là $2.500.000$ VNĐ. Giá bán ra mỗi chiếc là $3.000.000$ VNĐ.
	\begin{enumerate}
		\item Viết hàm số biểu diễn tổng số tiền đã đầu tư đến khi sản xuất được $x$ chiếc xe lăn (gồm vốn ban đầu và chi phí sản xuất) và hàm số biểu diễn số tiền thu được khi bán ra $x$ chiếc xe lăn.
		\dapso{$y=500+2{,}5x$ và $y=3x$.}
		\item Công ty $A$ phải bán bao nhiêu chiếc xe mới có thể thu hồi vốn ban đầu.
		\dapso{$1000$ chiếc xe.}
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Gọi $x$ là số chiếc xe lăn đã sản xuất, điều kiện $x\in \mathbb{N}$.\\
			Chi phí để sản xuất $x$ chiếc xe là $2{,}5x$ (triệu đồng).\\
			Chi phí vốn ban đầu là $500$ triệu đồng.\\
			Tổng số tiền phải bỏ ra đến khi sản xuất được $x$ chiếc xe được biểu diễn theo hàm số là $y = 500+2{,}5x$ (triệu đồng).\\
			Hàm số biểu diễn số tiền thu được khi bán ra $x$ chiếc xe là $y=3x$ (triệu đồng).
			\item Để thu hồi được vồn thì số tiền thu được từ bán xe phải bằng số tổng số tiền đã đầu tư, do đó ta có
			\begin{eqnarray*}
				& & 500 + 2{,}5x = 3x \\
				& \Leftrightarrow & x = 1000.
			\end{eqnarray*}
			Vậy công ty phải bán được $1000$ chiếc xe mới có thể thu hồi vốn ban đầu.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Lê Ngọc Hiếu]%[EX-TS10-ToanTT-2019]%[9D2T2]
Giá bán của ấm đun nước được xác định bởi hàm số $C=\dfrac{11R}{8}+50$  (trong đó: $C$ là giá bán, đơn vị tính: nghìn đồng; $R$ là bán kính của đáy ấm, đơn vị tính cm). Bạn Mai mua $2$ cái ấm có bán kính của đáy ấm lần lượt là $24$ cm và $32$ cm. Khi đi mua hàng, Mai mang theo $200\;000$ đồng. Hỏi Mai có đủ tiền để trả không?  	
	\dapso{Bạn Mai đủ tiền để trả.}
	\loigiai{
		
		Số tiền bạn Mai phải trả được tính theo công thức $C=\dfrac{11R}{8}+50.$
		
		Thay $R_1=24$ và $R_2=32$ vào $C$ ta có
		
		$$C_1=\dfrac{11R_1}{8}+50=\dfrac{11\cdot 24}{8}+5=83 
		\text{ và } C_2=\dfrac{11\cdot 32}{8}+50=94.$$
		 
		Số tiền bạn Mai phải trả là
		$C=C_1+C_2=177 .$
		
		Vậy bạn Mai đủ tiền để trả.
	}
\end{bt}

\begin{bt}%[Lê Ngọc Hiếu]%[EX-TS10-ToanTT-2019]%[9D2T2]
Để tăng thu nhập phụ giúp bố mẹ, bạn Lan nhận gia công sản phẩm thủ công. Vì thời gian trong ngày chủ yếu dành cho việc học nên Lan dự định mỗi ngày chỉ hoàn thành $50$ sản phẩm. Vì khéo tay nên mỗi ngày Lan hoàn thành hơn $20\%$ so với dự định. Hỏi bạn Lan cần thời gian bao nhiêu ngày để hoàn thành hết $1800$ sản phẩm đã nhận. 	
	\dapso{$30$ ngày}
	\loigiai{
		
	Gọi $x$ là số lượng sản phẩm bạn Lan hoàn thành trong $1$ ngày.
		
	Thời gian bạn Lan hoàn thành xong $1800$ sản phẩm là $y=\dfrac{1800}{x}$ (ngày).
		
	Ta có lượng sản phẩm thực tế bạn Lan làm trong một ngày là $x= 50+50\cdot 20\%=60$ sản phẩm.
	
	Thay $x=60$, ta có $y=\dfrac{1800}{60}=30.$
	
	Vậy thời gian bạn Lan hoàn thành xong $1800$ sản phẩm là $30$ ngày. 
	}
\end{bt}

\begin{bt}%[Võ Tấn Đạt]%[EX-TS10-ToanTT-2019]%[9D2T2]
	Công ty đồ chơi Bingbon vừa cho ra đời một đồ chơi tàu điện điều khiển từ xa. Trong điều kiện phòng thí nghiệm, quãng đường $s$ cm đi được của đoàn tàu đồ chơi là một hàm số của thời gian $t$ giây, hàm số đó là $s=6t+9.$ Trong điều kiện thực tế người ta thấy rằng nếu đoàn tàu đồ chơi di chuyển quãng đường $12$ cm thì mất $2$ giây, và cứ trong mỗi $10$ giây thì nó đi được $52$ cm. 
	\begin{enumerate}
		\item Trong điều kiện phòng thí nghiệm, sau $5$ giây đoàn tàu đồ chơi di chuyển được bao nhiêu cm? 
		\dapso{$39$ cm.}
		\item Mẹ  bé An mua đồ chơi này về cho bé chơi, bé ngồi cách mẹ $2$ mét. Hỏi cần bao nhiêu giây để đoàn tàu đồ chơi đi từ chỗ  mẹ tới chỗ bé? 
		\dapso{$t=39{,}6$ giây.}
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Trong điều kiện phòng thí nghiệm, quãng đường đoàn tàu đồ chơi di chuyển được sau $5$ giây là $$s(5)=6\cdot 5+9= 39\ \text{cm}.$$
			\item Gọi quãng đường $s'$ cm đi được của đoàn tàu đồ chơi trong thực tế, thế thì $$s'=at+b.$$
			Do trong điều kiện thực tế, người ta thấy rằng nếu đoàn tàu đồ chơi di chuyển quãng đường $12$ cm thì mất $2$ giây, và cứ trong mỗi $10$ giây thì nó đi được $52$ cm nên ta được hệ
			$$\Rightarrow \heva{& 12=2a+b \\& 52=10a+b }\Leftrightarrow
			\heva{& a=5 \\& b=2 .}$$
			Vậy $s'=5t+2$. Bé ngồi cách mẹ $2$ m$=200$ cm. Suy ra $$200=5t+2\Leftrightarrow t=39{,}6\ \text{giây}.$$
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Lê Mạnh Thắng]%[EX-TS10-ToanTT-2019]%[9D2T2]
	Do các hoạt động công nghiệp thiếu kiểm soát của con người làm cho nhiệt độ Trái Đất tăng dần một cách đáng lo ngại. Các nhà khoa học đưa ra công thức dự báo nhiệt độ trung bình trên bề mặt Trái Đất: $T=0{,}02t+15$ trong đó $T$ là nhiệt độ trung bình mỗi năm ($^\circ$C), t là số năm kể từ năm $1950$. Hãy tính nhiệt độ Trái Đất vào các năm $1950$ và $2020$.

	\loigiai{
		Nhiệt độ Trái Đất năm $1950$ (tương ứng với $t=0$) là $T=0{,}02t+15=0{,}02\cdot 0+15=15^\circ$C.\\
		Nhiệt độ Trái Đất năm $2020$ (tương ứng với $t=70$) là $T=0{,}02\cdot 70+15=16{,}4^\circ$C.\\

\begin{bt}%[Lê Mạnh Thắng]%[EX-TS10-ToanTT-2019]%[9D2T2]
	Khi nuôi cá trong hồ, các nhà sinh vật học đã ước tính rằng nếu trên mỗi mét vuông hồ cá có $n$ con cá thì khối lượng trung bình của mỗi con cá sau một vụ cân nặng $T = 500 -  200n$ (gam). Sau khi nuôi vụ đầu tiên thì cân nặng trung bình của mỗi con cá là $200$ gam. Biết rằng diện tích của hồ là $150$ m$^2$. Hãy tính số lượng cá được nuôi trong hồ.
	\dapso{$225$ con}
	\loigiai{
		Số lượng cá trên mỗi mét vuông là $n=\dfrac{500-T}{200}=\dfrac{500-200}{200}=1{,}5$ con.\\
		Diện tích hồ là $150\,$m$^2$ nên số lượng cá trong hồ là $150\cdot 1{,}5=225$ con.
	}
\end{bt}

\begin{bt}%[Lê Minh An]%[EX-TS10-ToanTT-2019]%[9D2B3]
	\immini{
	Tiền vốn và lãi bán hàng của một cửa hàng kinh doanh 6 tháng đầu năm được biểu thị bằng một đoạn thẳng, với vốn ban đầu là $15$ triệu đồng (hình).
	Hãy viết phương trình đồ thị trên, từ đó suy ra tiền vốn và lãi ở tháng thứ tư là bao nhiêu?
	\dapso{$y=2x+15$, $23$ triệu đồng}
	}{
	\begin{tikzpicture}[line join=round, line cap=round,>=stealth,scale=0.7]
	\tikzset{label style/.style={font=\footnotesize}}
	\def \xmin{-1}
	\def \xmax{7}
	\def \ymin{-1}
	\def \ymax{6.7}
	\draw[->] (\xmin,0)--(\xmax,0) node[above] {$x$ (tháng)};
	\draw[->] (0,\ymin)--(0,\ymax) node[right] {$y$ (triệu đồng)};
	\draw (0,0) node [below left] {$O$};
	\tkzDefPoints{0/3/A,5/5/B}
	\tkzDrawLine[add = 0 and 0.2](A,B)
	\draw[dashed] (5,0)node[below]{$5$}--(5,5)node[above]{$B$}--(0,5)node[left]{$25$};
	\draw (0,3)node[left]{$15$}node[below right]{$A$};
	\end{tikzpicture}
	}
	\loigiai{
		Đường thẳng là đồ thị hàm số bậc nhất $y=ax+b$.\\
		Mà $A(0;15)$ thuộc đồ thị nên $15=a\cdot 0+b\Rightarrow b=15$.\\
		Lại có $B(5;25)$ thuộc đồ thị nên $25=a\cdot 5+15\Leftrightarrow a=2$.\\
		Vậy hàm số cần tìm là $y=2x+15$.\\
		Tiền vốn và lãi ở tháng thứ tư là $y=2\cdot 4+15=23$ triệu đồng.
	}
\end{bt}
