\begin{ex}{\it{(3,0 điểm)}}[9D5G2][9D5G3]
\begin{enumerate}
		\item Tìm các số nguyên dương $n$ để $A = \dfrac{\big(n - 8\big)^2 - 48}{n + 5}$ có giá trị là số nguyên dương.
		\item Tìm các số nguyên dương $x$, $y$  thỏa mã đẳng thức $x^2 + y\big(y^2 + y - 3x\big) = 0$.
		\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $A=n-21+\dfrac{121}{n+5}$ nên $n+5$ là ước của $121$ và $n+5\geq 6.$ Nên 
\[\left[\begin{array}{rl}n+5&=11\\ n+5&=121\end{array}\right.\Leftrightarrow\left[\begin{array}{rl}n&=6\\ n&=116.\end{array}\right.\]


\item Biến đổi phương trình $x^2-3yx+y^3+y^2=0.$ Coi $x$ là ẩn, tính biệt thức 
\[\Delta=9y^2-4y^3-4y^2=y^2(5-4y).\]
Do $\Delta\geq 0$ nên $y\leq \dfrac{5}{4}.$ Vì $y$ là số nguyên dương nên $y=1.$

Thay vào phương trình ban đầu $x^2-3x+2=0\Leftrightarrow x=1 \text{ hoặc } x=2.$

Vậy phương trình đã cho có 2 nghiệm $(x;y)=(1;1)$ và $(x;y)=(2;1).$
\end{enumerate}
}
\end{ex}
\begin{ex}{\it{(2,0 điểm)}}[9D5G5]\\
Giải hệ phương trình:   $\begin{cases}
\big(x^2 + 1\big)y  = 2x^2\\
\big(y^2 + 1\big)z  = 2y^2\\
\big(z^2 + 1\big)x  = 2z^2.
\end{cases}$
\loigiai{Xét 2 trường hợp:

TH1: Nếu $x=0$ thì $y=0$ và $z=0.$

TH2: Nếu $x\neq 0$ thì $y> 0, z> 0, x>0.$

Nhân từng vế 3 phương trình với nhau, thu được: \begin{equation}
(x^2+1)(y^2+1)(z^2+1)=8xyz.\label{2009pt1}
\end{equation}

Ta có $x^2+1\geq 2x, \quad y^2+1\geq 2y, \quad z^2+1\geq z$ nên $(x^2+1)(y^2+1)(z^2+1)\geq 8xyz.$

Nên phương trình (\ref{2009pt1}) tương đương với $x=y=z=1.$

Từ đó thu được các nghiệm $(x,y,z)$ là $(0;0;0),(1;1;1).$
}
\end{ex}
\begin{ex}{\it{(3,0 điểm)}}[9H3G7]\\
Cho tam giác $ABC$ có ba góc nhọn nội tiếp  đường tròn $\big(O\big)$. Gọi $BD$ và $CE$ là hai đường cao của tam giác $ABC$.
\begin{enumerate}	
\item Chứng minh $AD\cdot AC  = AE\cdot AB$.
\item  Tia $AO$ cắt $BC$ tại $A_1$ và cắt cung nhỏ $BC$ tại $A_2$, tia $BO$ cắt $AC$ tại $B_1$ và cắt cung nhỏ $AC$ tại $B_2$. Tia $CO$ cắt $AB$ tại $C_1$ và cắt cung nhỏ $AB$ tại $C_2$. Chứng minh 
\[\dfrac{A_1A_2}{A_1A} + \dfrac{B_1B_2}{B_1B} + \dfrac{C_1C_2}{C_1C} = 1. \]
\item Từ $A$ vẽ tia $Ax$ vuông góc với $DE$. Cho $BC$ cố định, điểm $A$ di động trên cung lớn $BC$ sao cho tam giác $ABC$ luôn có $3$ góc nhọn. Chứng minh tia $Ax$ luôn đi qua một điểm cố định.
\end{enumerate}

Suy ra $\widehat{ADE}=\widehat{ABC}$ nên $\triangle ADE\backsim \triangle ABC$ ta được \[\dfrac{AD}{AB}=\dfrac{AE}{AC}\Rightarrow AD\cdot AC=AE\cdot AB.\]

\item Ta có $\dfrac{A_1A_2}{A_1A}=\dfrac{S_{CA_1A_2}}{S_{CA_1A}}=\dfrac{S_{BA_1A_2}}{S_{BA_1A}}=\dfrac{S_{CA_1A_2}+S_{BA_1A_2}}{S_{CA_1A}+S_{BA_1A}}=\dfrac{S_{A_2BC}}{S_{ABC}}.$

Gọi $H=BD\cap CE$ ta có $CH\perp AB$ và $BA_2\perp AB$ nên $CH\parallel BA_2.$ 

Tương tự $BH\parallel CA_2$ nên $BHCA_2$ là hình bình hành. 

Suy ra $S_{A_2BC}=S_{HBC}.$

\begin{ex}{\it{(1,0 điểm)}}[8D1G9]\\
Cho đa thức $P\big(x\big) = x^4 +  ax^3 + bx^2 + cx + d$ ($a$, $b$, $c$, $d$ là hằng số). Biết rằng $P\big(1\big) = 10$, $P\big(2\big) = 20$, $P\big(3\big) = 30$. Hãy tính giá trị của biểu thức $\dfrac{P\big(12\big) + P\big(- 8\big)}{10} + 25.$   
\loigiai{
Đa thức $P(x)-10x$ có các nghiệm $1$, $2$ và $3$, nên tồn tại đa thức $Q$ sao cho \[P(x)-10x=(x-1)(x-2)(x-3)Q(x).\]
Đa thức $P(x)-10x$ có bậc $4$ và hệ số bậc cao nhất bằng $1$, nên $Q(x)=x+e$ với $e$ là hằng số.

\noindent Do đó, $P(12)+P(-8)=10\cdot(12-8)+11\cdot10\cdot9\cdot(12+e)+9\cdot10\cdot11\cdot(8-e).$  

$\Rightarrow P(12)+P(-8)=10\cdot4+11\cdot10\cdot9\cdot(12+e+8-e)=40+990\cdot20=19840$.

\noindent Vậy $\dfrac{P\big(12\big) + P\big(- 8\big)}{10} + 25=2009.$  
}
\end{ex}
\begin{ex}{\it{(1,0 điểm)}}[9D5G1]\\
Chứng minh rằng nếu ba điểm $A$, $B$, $C$ không có điểm nào nằm ngoài đường tròn $\big(O\big)$ sao cho tam giác $ABC$ có $3$ góc nhọn thì chu vi của đường tròn ngoại tiếp tam giác $ABC$ không lớn hơn chu vi của đường tròn $\big(O\big).$

Giả sử các tia $AB,BC,CA$ cắt đường tròn $(O)$ lần lượt tại các điểm $A_1,B_1,C_1$.
Và gọi $R$ và $r$ lần lượt là bán kính đường tròn $(O)$ và đường tròn ngoại tiếp tam giác $ABC.$
Trong các góc $\widehat{AC_1B_1},\widehat{CB_1A_1},\widehat{BA_1C_1}$, không mất tính tổng quát ta giả sử góc nhỏ nhất là $\widehat{AC_1B_1}.$

Khi đó $\widehat{BAC}=\widehat{AC_1A_1}+\widehat{AA_1C_1}\geq \widehat{AC_1A_1}+\widehat{AC_1B_1}=\widehat{A_1C_1B_1}.$

Suy ra $\sin \widehat{BAC}\geq \sin \widehat{A_1C_1B_1}.$

Vì góc $\widehat{ABC}$ nhọn nên góc $\widehat{A_1BB_1}$ là góc tù nên $A_1B_1\geq BB_1\geq BC.$

Áp dụng định lý sin ta có: $A_1B_1=2R\sin \widehat{A_1C_1B_1}$ và $BC=2r\sin \widehat{BAC}.$

Suy ra $2R\sin \widehat{A_1C_1B_1}\geq 2r\sin \widehat{BAC}$ nên $\dfrac{R}{r}\geq \dfrac{\sin \widehat{BAC}}{\sin \widehat{A_1C_1B_1}}\geq 1.$

Vậy $R\geq r.$
}
\end{ex}
