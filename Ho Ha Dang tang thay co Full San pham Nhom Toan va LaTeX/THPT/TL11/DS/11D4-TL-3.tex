\section{Hàm số liên tục}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Hàm số liên tục tại một điểm}
\begin{dn}
Cho hàm số $y=f(x)$ xác định trên khoảng $K$ và $x_0\in K$. Hàm số $y=f(x)$ được gọi là \textbf{liên tục} tại $x_0$ nếu $\lim\limits_{x\to x_0}f(x)=f(x_0)$.
\end{dn}

\begin{note}
Hàm số $y=f(x)$ không liên tục tại $x_0$ được gọi là \textbf{gián đoạn} tại điểm đó.
\end{note}

\subsubsection{Hàm số liên tục trên một khoảng}

\begin{dn}
Hàm số $y=f(x)$ được gọi là \textbf{liên tục trên một khoảng} nếu nó liên tục tại mọi điểm của khoảng đó.
\end{dn}

\begin{dn}
Hàm số $y=f(x)$ được gọi là \textbf{liên tục trên đoạn} $[a;b]$ nếu nó liên tục trên khoảng $(a;b)$ và 
$\lim\limits_{x\to a^+}f(x)=f(a)$, $\lim\limits_{x\to b^-}f(x)=f(b)$.
\end{dn}

\begin{note}
Khái niệm hàm số liên tục trên nửa khoảng, như $(a;b]$, $[a;+\infty)$, $\ldots$ được định nghĩa một cách tương tự.
\end{note}

\begin{note}

\subsubsection{Một số định lí cơ bản}
\begin{dl}
\hfill
\begin{enumerate}
\item  Hàm số đa thức liên tục trên toàn bộ tập số thực $\mathbb{R}$.
\item Hàm số phân thức hữu tỉ (thương của hai đa thức) và các hàm số lượng giác liên tục trên từng khoảng của tập xác định của chúng.
\end{enumerate}
\end{dl}

\begin{dl}
Giả sử $y=f(x)$ và $y=g(x)$ là hai hàm số liên tục tại điểm $x_0$. Khi đó
\begin{enumerate}
\item Các hàm số $y=f(x)+g(x)$, $y=f(x)-g(x)$ và $y=f(x).g(x)$ liên tục tại $x_0$.
\item Hàm số $y=\dfrac{f(x)}{g(x)}$ liên tục tại $x_0$ nếu $g(x_0)\neq 0$.
\end{enumerate}
\end{dl}

\begin{dl}
Nếu hàm số $y=f(x)$ liên tục trên đoạn $[a;b]$ và $f(a)f(b)<0$, thì tồn tại ít nhất một điểm $c\in (a;b)$ sao cho $f(c)=0$.
\end{dl}

\begin{note}
Nếu hàm số $y=f(x)$ liên tục trên đoạn $[a;b]$ và $f(a)f(b)<0$ thì phương trình $f(x)=0$ có ít nhất một nghiệm nằm trong khoảng $(a;b)$.
\end{note}


\subsection{Các dạng toán}

\begin{dang}{Xét tính liên tục của hàm số tại một điểm}
Cho hàm số $y=f(x)$ xác định trên tập $D$. Để xét tính liên tục của hàm số $y=f(x)$ tại điểm $x_{0}\in D$, ta thực hiện các bước sau:
\begin{enumerate}[Bước 1.]
	\item Tính $f(x_0)$.
	\item Tìm $\lim\limits_{x\to x_0}f(x)$.
	\item So sánh và rút ra kết luận.
	\begin{itemize}
		\item Nếu $\lim\limits_{x\to x_0}f(x)=f(x_0)$ thì hàm số $f(x)$ liên tục tại điểm $x_0$.
		\item Nếu $\lim\limits_{x\to x_0}f(x)\ne f(x_0)$ thì hàm số $f(x)$ không liên tục (gián đoạn) tại điểm $x_0$.
	\end{itemize}
\end{enumerate}
\end{dang}
\begin{vd}%[Huỳnh Văn Quy]%[1D4B3]
	Cho hàm số: $f(x)=\heva{&\dfrac{x^2-1}{x-1}&\text{nếu}&\quad x\ne  1 \\ 
		& a&\text{nếu}&\quad x=1}$ với $a$ là hằng số. \\
	Xét tính liên tục của hàm số tại $x_0=1$.
\loigiai{Ta có:
	\begin{itemize}
		\item $f(1) = a$.
		\item $\lim\limits_{x\to 1}f(x)=\lim\limits_{x\to 1}\dfrac{x^2-1}{x-1}=\lim\limits_{x\to 1}(x+1)=2$.
		\begin{itemize}
			\item Nếu $a=2$ thì hàm số $f(x)$ liên tục tại điểm $x_0=1$.
			\item Nếu $a\ne2$ thì hàm số $f(x)$ gián đoạn tại điểm $x_0=1$.
		\end{itemize}
	\end{itemize}}	
\end{vd}
\begin{vd}%[Huỳnh Văn Quy]%[1D4B3]

	\end{cases}$.\\

	\loigiai{Ta có:
	\begin{itemize}
		\item $f(0)=0$.
		\item $\lim\limits_{x\to0^{+}}f(x)=\lim\limits_{x\to0^{+}}(x^2+1)=1$.
		\item $\lim\limits_{x\to0^{-}}f(x)=\lim\limits_{x\to0^{-}}x=0$.
\end{itemize}
	Ta có: $f(0)=\lim\limits_{x\to0^{-}}f(x)\ne\lim\limits_{x\to0^{+}}f(x)$. Vậy hàm số $f(x)$ gián đoạn tại điểm $x=0$.
}
\end{vd}
\begin{vd}%[Huỳnh Văn Quy]%[1D4B3]

	-2&\text{nếu}\: x=1
	\end{cases}$.\\ 
	Xét tính liên tục của hàm số $f(x)$ tại điểm $x_0=1$.
	\loigiai{Ta có: $f(1)=-2$.\\
		$\lim\limits_{x\to1}f(x)=\lim\limits_{x\to1}\dfrac{x^2-6x+5}{x^2-1}=\lim\limits_{x\to1}\dfrac{(x-5)(x-1)}{(x-1)(x+1)}=\lim\limits_{x\to1}\dfrac{x-5}{x+1}=-2=f(1)$.\\
		Vậy hàm số $f(x)$ liên tục tại $x=1$.}
\end{vd}
\begin{vd}%[Huỳnh Văn Quy]%[1D4B3]

	\end{cases}$
 tại điểm $x_0=2$.
 \loigiai{Ta có:
 \begin{itemize}
 	\item $f(2)=1$.
 	\item $\lim\limits_{x\to 2}f(x)=\lim\limits_{x\to 2}\dfrac{1-\sqrt{2x-3}}{2-x}=\lim\limits_{x\to 2}\dfrac{1-(2x-3)}{(2-x)(1+\sqrt{2x-3})}=\lim\limits_{x\to 2}\dfrac{2(2-x)}{(2-x)(1+\sqrt{2x-3})}\\
 	=\lim\limits_{x\to 2}\dfrac{2}{1+\sqrt{2x-3}}=1=f(2)$
\end{itemize}
Vậy hàm số $f(x)$ liên tục tại $x_0=2$.}
\end{vd}

\begin{vd}%[Huỳnh Văn Quy]%[1D4K3]

	\end{cases}$. Tìm $a$ để hàm số liên tục tại $x_0=2$.

\begin{vd}%[Huỳnh Văn Quy]%[1D4K3]
	Cho hàm số $f(x)=\begin{cases}

	\end{cases}$. Tìm $m$ để hàm số liên tục tại $x_0=2$.
	\loigiai{Ta có: $f(2)=m^2+3m$.\\
		$\lim\limits_{x\to 2}f(x)=\lim\limits_{x\to 2}\dfrac{x^2-4}{x-2}=\lim\limits_{x\to 2}\dfrac{(x-2)(x+2)}{(x-2)}=\lim\limits_{x\to 2}(x+2)=4$.\\
		Để hàm số liên tục tại điểm $x=2$ thì $\lim\limits_{x\to 2}f(x)=f(2)\Leftrightarrow 4=m^2+3m\Leftrightarrow \left[\begin{array}{l}
		m=1\\
		m=-4
		\end{array}
		\right.$.
	}
\end{vd}

\begin{vd}%[Huỳnh Văn Quy]%[1D4K3]
	Tìm $m$ để hàm số 
	$f(x)=\begin{cases}
	\dfrac{\sqrt{1-x}-\sqrt{1+x}}{x}&\text{nếu}\: x<0\\
	m+\dfrac{x^3-3x+1}{x+2}&\text{nếu}\: x\geq 0
	\end{cases}$ 
	liên tục tại $x_0=0$.
	\loigiai{Ta có: $f(0)=m+\dfrac{1}{2}$.
		\begin{align*}
		\lim\limits_{x\to0^{-}}f(x)&=\lim\limits_{x\to0^{-}}\dfrac{\sqrt{1-x}-\sqrt{1+x}}{x}=\lim\limits_{x\to0^{-}}\dfrac{\left(\sqrt{1-x}-\sqrt{1+x}\right)\left(\sqrt{1-x}+\sqrt{1+x}\right)}{x\left(\sqrt{1-x}+\sqrt{1+x}\right)}\\
		&=\lim\limits_{x\to0^{-}}\dfrac{-2x}{x\left(\sqrt{1-x}+\sqrt{1+x}\right)}=\lim\limits_{x\to0^{-}}\dfrac{-2}{\sqrt{1-x}+\sqrt{1+x}}=-1.

	}
\end{vd}

	}
\end{vd}

\begin{bt}%[Huỳnh Văn Quy]%[1D4G3]

\begin{bt}%[KV Thanh]%[1D4K3]

$\lim\limits_{x\to 0}f(x)=\lim\limits_{x\to 0}\dfrac{\sqrt[3]{8+x}-\sqrt{4-x}}{x}=\lim\limits_{x\to 0}\dfrac{\left(\sqrt[3]{x+8}-2\right)-(\sqrt{4-x}-2)}{x}$.
\begin{align*}
L_1&=\lim\limits_{x\to 0}\dfrac{\sqrt[3]{x+8}-2}{x}=\lim\limits_{x\to 0}\dfrac{(x+8)-8}{x\left(\sqrt[3]{(x+8)^2}+2\sqrt[3]{x+8}+4\right)}\\
&=\lim\limits_{x\to 0}\dfrac{1}{\sqrt[3]{(x+8)^2}+2\sqrt[3]{x+8}+4}=\dfrac{1}{12}.\\
L_2&=\lim\limits_{x\to 0}\dfrac{\sqrt{4-x}-2}{x}=\lim\limits_{x\to 0}\dfrac{(4-x)-4}{x\left(\sqrt{4-x}+2\right)}\\
&=\lim\limits_{x\to 0}\dfrac{-1}{\sqrt{4-x}+2}=-\dfrac{1}{4}.
\end{align*}
Suy ra $\lim\limits_{x\to 0}f(x)=L_1-L_2=\dfrac{1}{12}+\dfrac{1}{4}=\dfrac{1}{3}=f(0)$.\\

}

\begin{bt}%[KV Thanh]%[1D4G3]
Xét tính liên tục của hàm số $f(x)=\heva{& \dfrac{(1+2017x)^{2018}-(1+2018x)^{2017}}{x^2}, &\text{ với } x\neq 0\\& 2017\cdot 2018, &\text{ với } x=0.}$ trên tập số thực $\mathbb{R}$.
\loigiai{
TXĐ của hàm số $\mathscr{D}=\mathbb{R}$.\\

Xét tại $x=0$:\\
$(1+2017x)^{2018}=1+\mathrm{C}_{2018}^1(2017x)+\mathrm{C}_{2018}^2(2017x)^2+\mathrm{C}_{2018}^3(2017x)^3+\cdots +\mathrm{C}_{2018}^{2018}(2017x)^{2018}$.\\
$(1+2018x)^{2017}=1+\mathrm{C}_{2017}^1(2018x)+\mathrm{C}_{2017}^2(2018x)^2+\mathrm{C}_{2017}^3(2018x)^3+\cdots +\mathrm{C}_{2017}^{2017}(2018x)^{2017}$.\\

$(1+2017x)^{2018}-(1+2018x)^{2017}=\left(\mathrm{C}_{2018}^22017^2-\mathrm{C}_{2017}^22018^2\right)x^2+a_3x^3+a_4x^4+\cdots +a_{2018}x^{2018}$, trong đó $a_k=\mathrm{C}_{2018}^k2017^k-\mathrm{C}_{2017}^k2018^k, 3\leq k\leq 2017$ và $a_{2018}=2017^{2018}$.\\
$\mathrm{C}_{2018}^22017^2-\mathrm{C}_{2017}^22018^2=\dfrac{2018\cdot 2017}{2}2017^2-\dfrac{2017\cdot 2016}{2}2018^2=\dfrac{2017\cdot 2018}{2}$\\
Suy ra 
\begin{align*}
