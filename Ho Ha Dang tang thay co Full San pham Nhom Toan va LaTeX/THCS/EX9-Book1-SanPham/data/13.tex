\begin{vd}
    Tìm nghiệm nguyên của phương trình$$y=\sqrt{x+2\sqrt{x-1}}+\sqrt{x-2\sqrt{x-1}}.$$
    \loigiai{
       Điều kiện: $x\ge 1.$
       \begin{eqnarray*}
       y&=&\sqrt{(x-1)+1+2\sqrt{x-1}}+\sqrt{(x-1)+1-2\sqrt{x-1}}\\
        &=&\left|\sqrt{x-1}+1\right|+\left|\sqrt{x-1}-1\right|\\
        &=&\sqrt{x-1}+1+\left|\sqrt{x-1}-1\right|.
       \end{eqnarray*}
       Xét hai trường hợp:           
\begin{enumerate}[a)]
\item Với $x=1$ thì $y=2$.
\item Với $x\ge 2$ thì $y=\sqrt{x-1}+1+\sqrt{x-1}-1=2\sqrt{x-1}.$\\Do đó $y^2=4(x-1)$. Do $x\ge 2$ nên có thể đặt $x-1=t^2$ với $t$ là số nguyên dương.\\
Ta có $\heva{&x=t^2+1\\&y=2t.}$ 
\end{enumerate} 
\textit{Kết luận:} Nghiệm nguyên $(x;y)$ của phương trình là $(1;2)$ và $(t^2+1;2t))$ với $t$ là số nguyên dương tùy ý.   
    }
\end{vd}

\begin{vd}
Tìm nghiệm nguyên của phương trình $$\sqrt{x}+\sqrt{x+3}=y.$$
\loigiai{
Điều kiện: $x\ge 0, y\ge 0.$ Khi đó thì
\begin{align*}
 &\sqrt{x}+\sqrt{x+3}=y\tag{1}\\
\Leftrightarrow & x+x+3+2\sqrt{x(x+3)}=y^2 \Leftrightarrow 2\sqrt{x(x+3)}=y^2-2x-3
\end{align*}
Đặt $y^2-2x-3=m$ (với $m\in \mathbb{N}$) thì $2\sqrt{x(x+3)}=m.$\hfill (2)
\begin{itemize}
\item Xét $m=0$, từ $(2)$ suy ra $x=0$ (vì $x\in \mathbb{N}$), thay vào $(1)$ được $y=\sqrt{3}$ (loại).
\item Xét $m>0$. Bình phương hai vế của $(2)$ được
\begin{eqnarray*}
4x(x+3)=m^2&\Leftrightarrow &(2x+3)^2-m^2=9\\&\Leftrightarrow &(2x+3+m)(2x+3-m)=9.
\end{eqnarray*}
Vì $2x+3+m$ và $2x+3-m$ là các ước tự nhiên của $9$ và $2x+3+m>2x+3-m$ nên\\
\begin{center}
$\heva{&2x+3+m=9\\&2x+3-m=1}$
\end{center}
Ta có $2x+3=5\Leftrightarrow x=1$. Khi đó $y=3$.
\end{itemize}
\textit{Đáp số:} Nghiệm $(x;y)$ là $(1;3).$
}
\end{vd}

\begin{vd}
Tìm nghiệm nguyên của phương trình \begin{align*}
\sqrt{9x^2+16x+96}=3x-16y-24 \tag{1}
\end{align*}
\loigiai{
Đặt $m=3x-16y-24$ với $m\in \mathbb{N}$. Ta có
{\allowdisplaybreaks
\begin{align*}
(1)&\Leftrightarrow  9x^2+16x+96=m^2\\&\Leftrightarrow  81x^2+9\cdot 16x+864=9m^2\\&\Leftrightarrow (9x+8)^2-(3m)^2=-800.\tag{2}
\end{align*}}Thay $m=3x-16y-24$ vào $(2)$ rồi rút gọn được
\begin{eqnarray*}
& &(18x-48y-64)(48y+80)=-800\\
&\Leftrightarrow &(9x-24y-32)(3y+5)=-25.
\end{eqnarray*}
Ta thấy $3y+5$ là ước của $25$ và chia $3$ dư $2$ nên có bảng giá trị sau\\
\begin{center}
\begin{tabular}{|c|c|c|c|}
\hline 
$3y+5$ & $-1$ & $5$ & $-25$ \\ 
\hline 
$y$ & $-2$ & $0$ & $-10$ \\ 
\hline 
$x$ & $1$ & $3$ & $-23$ \\ 
\hline 
\end{tabular} 
\end{center}
Loại $x=3, y=0$ vì khi đó $m<0$.\\ Với $x=1, y=-2$ và $x=-23, y=-10$ thì $m>0$.\\
\textit{Đáp số:} Nghiệm $(x;y)$ là $(1;-2),(-23;-10).$\\
$\blacktriangleright\blacktriangleright$ \textbf{\textit{ Kinh nghiệm giải toán}}
\begin{itemize}


\item Ở ví dụ $53$ ta đặt $y^2-2x-3=m$ để đưa đến phương trình $4x(x+3)=m^2$ và được $(2x+3+m)(2x+3-m)=9$. Giải phương trình này gọn hơn so với cách thay $m$ bởi $y^2-2x-3$.\\
\item Ở ví dụ $54$, ta đặt $3x-16y-24=m$ để đưa đến phương trình $9x^2+16x+96=m^2$ và được $(9x+8)^2-(3m)^2=-800$. Giải phương trình này phức tạp vì phải xét nhiều trường hợp. Ta đã thay $m=3x-16y-24$ để đưa về phương trình đơn giản hơn: $(9x-24y-32)(3y+5)=-25.$\\
\item Như vậy, khi giải toán cần biết vận dụng linh hoạt cách đặt ẩn phụ cho phù hợp với từng bài toán cụ thể.
\end{itemize}
}
\end{vd}

\begin{vd}\label{laoth1}
Tìm nghiệm nguyên của phương trình
\begin{align*}
\sqrt{x}+\sqrt{y}=\sqrt{1980}\tag{1}
\end{align*}
\loigiai{
\begin{align*}
\sqrt{x}=\sqrt{1980}-\sqrt{y}\tag{2}
\end{align*}
Với điều kiện $0\le x\le 1980, 0\le y\le 1980$, ta có
$$(2)\Leftrightarrow x=1980+y-2\sqrt{1980y}\Leftrightarrow x=1980+y-12\sqrt{55y}.$$
Do $x,y$ nguyên nên $12\sqrt{55y}$ nguyên.\\ Ta biết rằng với $y$ nguyên thì $\sqrt{55y}$ hoặc là số nguyên, hoặc là số vô tỉ. Chỉ có thể $\sqrt{55y}$ là số nguyên, tức là $55y$ là số chính phương nên $11\cdot 5\cdot y=k^2 (k\in \mathbb{N})$.\\
Do đó $y=11\cdot 5\cdot a^2=55a^2 (a\in \mathbb{N})$. Tương tự $x=55b^2 (b\in \mathbb{N}).$\\ Thay vào $(1)$ được $$a\sqrt{55}+b\sqrt{55}=6\sqrt{55}\Leftrightarrow a+b=6.$$
Giả sử $y\le x$ thì $a\le b$. Ta có bảng giá trị sau
\begin{center}
\begin{tabular}{|m{1.5cm}|m{1.5cm}|m{2cm}|m{2cm}|}
\hline 
$a$ & $b$ & $x=55a^2$ & $y=55b^2$ \\ 
\hline 
$0$ & $6$ & $0$ & $1980$ \\ 
 
$1$ & $5$ & $55$ & $1375$ \\ 

$2$ & $4$ & $220$ & $880$ \\ 
 
$3$ & $3$ & $495$ & $495$ \\ 
\hline
\end{tabular} 
\end{center}
Có bảy nghiệm $(x;y)$ là $(0;1980),(1980;0),(55;1375),(1375;55),(220;880),(880;220),(495;495).$\\
\textbf{Lưu ý.} Ta có nhận xét: Nếu các số $\sqrt{x}, \sqrt{y}$ với $x,y\in \mathbb{N}$ có tổng là số vô tỉ $6\sqrt{55}$ thì $\sqrt{x}=b\sqrt{55}, \sqrt{y}=a\sqrt{55}$ với $a,b\in\mathbb{N}$ và $a+b=6.$
}
\end{vd}
\begin{vd}\label{ptvt1}
Tìm nghiệm nguyên của phương trình$$\sqrt{x+\sqrt{x+\sqrt{x+\sqrt{x}}}}=y.$$
\loigiai{Điều kiện: $x\ge 0;y\ge 0.$\\
Bình phương hai vế rồi chuyển vế được $$\sqrt{x+\sqrt{x+\sqrt{x}}}=y^2-x=k (k\in \mathbb{N}).$$
Lại bình phương hai vế rồi chuyển vế được $$\sqrt{x+\sqrt{x}}=k^2-x=m (m\in\mathbb{N}).$$
Lại bình phương hai vế được$$x+\sqrt{x}=m^2.$$
Ta biết rằng $x$ nguyên thì $\sqrt{x}$ hoặc là số nguyên hoặc là số vô tỉ. Do $x+\sqrt{x}=m^2 (m\in\mathbb{N})$ nên $\sqrt{x}$ không là số vô tỉ. Do đó $\sqrt{x}$ là số nguyên và là số tự nhiên.\\
Ta có $\sqrt{x}(\sqrt{x}+1)=m^2.$ Hai số tự nhiên liên tiếp $\sqrt{x}$ và $\sqrt{x}+1$ có tích là một số chính phương nên số nhỏ bằng $0$ tức là $\sqrt{x}=0$.\\Suy ra $x=0;y=0,$ thõa mãn phương trình đã cho.\\Nghiệm $(x;y)$ của phương trình là $(0;0).$

}
\end{vd}
\begin{vd}\label{laoth2}
Tìm nghiệm nguyên dương của phương trình 
\begin{align*}
y=\sqrt[3]{2+\sqrt{x}}+\sqrt[3]{2-\sqrt{x}}\tag{1}
\end{align*}
\loigiai{Áp dụng hằng đẳng thức $(a+b)^3=a^3+b^3+3ab(a+b)$, ta lập phương hai vế của $(1)$ được
{\allowdisplaybreaks
\begin{align*}
&y^3=2+\sqrt{x}+2-\sqrt{x}+3\cdot \sqrt[3]{4-x}\cdot y\tag{2}\\
\Leftrightarrow & y^3=4+3y\sqrt[3]{4-x}\\
\Leftrightarrow & y(y^2-3\sqrt[3]{4-x})=4.\tag{3}
\end{align*}}Với $x$ nguyên thì $3\sqrt[3]{4-x}$ là số nguyên hoặc là số vô tỉ. Từ $(3)$ ta thấy $3\sqrt[3]{4-x}$ không thể là số vô tỉ nên phải là số nguyên. Do đó từ $(3)$ suy ra $y$ là ước của $4.$ Do $y>0$ nên ta có bảng giá trị sau
\begin{center}
\begin{tabular}{|m{3cm}|m{1cm}|m{1cm}|m{2cm}|}
\hline 
$y$ & $1$ & $2$ & $4$ \\ 
\hline 
$y^2-3\sqrt[3]{4-x}$ & $4$ & $2$ & $1$ \\ 
\hline 
$3\sqrt[3]{4-x}$ & $-3$ & $2$ & $15$ \\ 
\hline 
$\sqrt[3]{4-x}$ & $-1$ & loại & $5$ \\ 
\hline 
$4-x$ & $-1$ &  & $125$ \\ 
\hline 
$x$ & $5$ &  & $-121$ (loại) \\ 
\hline 
\end{tabular} 
\end{center}
Thử lại, giá trị $x=5, y=1$ thỏa mãn phương trình đã cho.\\
\textit{Đáp số:} Nghiệm $(x;y)$ là $(5;1).$\\
\textbf{Lưu ý:} Ở phương trình $(2)$, ta đã thay $\sqrt[3]{2+\sqrt{x}}+\sqrt[3]{2-\sqrt{x}}$ nên $(1)$ và $(2)$ có thể không tương đương. Vì thế sau khi tìm được nghiệm $(x;y)=(5;1)$ của phương trình $(3)$, ta vẫn thử lại vào phương trình $(1).$\\
$\ast$ \textbf{ Kinh nghiệm giải toán}\\
Khi giải phương trình vô tỉ với nghiệm nguyên, ta thường biến đổi để trong phương trình chỉ chứa một căn thức của một số nguyên, từ đó dẫn đến căn thức ấy cũng là số nguyên (xem các ví dụ $55, 56, 57$). 
}
\end{vd}

\begin{bt}
Tìm nghiệm nguyên của mỗi phương trình sau:
\begin{multicols}{2}
\begin{enumerate}[a)]
\item $x^3+1=\sqrt{3x+1}$;
\item $x^3+3=\sqrt{15x+1}$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}[a)]
\item Do $x$ nguyên nên để $\sqrt{3x+1}$ có nghĩa, ta phải có $x\geq 0$.\\
\textit{Cách 1.} Với $x\geq 2$ thì
$$\sqrt{3x+1}<\sqrt{3x+x}=2\sqrt{x}<2x<x^3<x^3+1,\; \text{loại}.$$
Xét $x=0,x=1$ đều thỏa mãn.\\
\textit{Cách 2.} $x^3+1=\sqrt{3x+1}\Leftrightarrow \left( x^3+1\right) ^2=3x+1$,\\
đưa về $x(x-1)\left( x^4+x^3+x^2+3x+3\right)=0$.\\
Do $x\geq 0$ nên nghiệm của phương trình là $0$ và $1$.
\item Giải tương tự như câu a. \textit{Đáp số: $x=1$.}
\end{enumerate}
\textbf{\textit{Kinh nghiệm giải toán}}\\
Nghiệm nguyên (nếu có) của các phương trình $x^3+1=\sqrt{3x+1}$ và $x^3+3=\sqrt{15x+1}$ chỉ có thể lấy một vài giá trị tự nhiên nhỏ, vì khi $x$ tăng thì vế trái của các phương trình trên tăng rất nhanh, trong khi vế phải chỉ tăng từ từ. Nhận xét đó giúp ta có cách giải $1$ nói trên.
}
\end{bt}

\begin{bt}
    Tìm nghiệm nguyên của mỗi phương trình sau:
    \begin{multicols}{2}
   \noindent a) $\sqrt{x^2-4x+7}=x-y$;\\c) $2y^2=1+\sqrt{49-x^2-2x}$\vfill \noindent b) $y^2=1+\sqrt{9-x^2-4x}$;\\ d) $x^2-y^2=\sqrt{y+1}.$
    \end{multicols}
    \loigiai{
        \begin{enumerate}[a)]
        \item $\sqrt{x^2-4x+7}=x-y.$\\ Đặt $x-y=m\ge 0$, đưa về $(x-2)^2-m^2=-3.$\\
        \textit{Đáp số:} Nghiệm $(x;y)$ là $(1;-1), (3;1).$
        \item $y^2=1+\sqrt{13-(x^2+4x+4)}=1+\sqrt{13-(x+2)^2}\le 1+\sqrt{13}.$\\ Suy ra $1\le y^2\le 4.$
        \begin{itemize}
        \item Với $y^2=1$ thì $13=(x+2)^2$, không có nghiệm nguyên.
        \item Với $y^2=4$ thì $3=\sqrt{13-(x+2)^2}\Leftrightarrow (x+2)^2=4\Leftrightarrow \hoac{&x=0\\&x=-4}$
        \end{itemize}
        \textit{Đáp số:} Nghiệm $(x;y)$ là $(0;2),(0;-2),(-4;2),(-4;-2).$
        \item $2y^2=1+\sqrt{50-(x+1)^2}\le 1+\sqrt{50}<1+8=9.$\\ Suy ra $y^2\le 4.$ Do $y^2\neq 0$ nên $y^2\in \{1;4\}.$
        \begin{itemize}
        \item Xét $y^2=1$, ta tìm được $x_1=6, x_2=-8.$
        \item Xét $y^2=4$, ta tìm được $x_3=0, x_4=-2.$
        \end{itemize}
     \textit{Đáp số:} Nghiệm $(x;y)$ là $(6;1), (6;-1), (-8;1), (-8;-1), (0;2), (0;-2), (-2;2), (-2;-2).$ 
     \item  Xét phương trình $x^2-y^2=\sqrt{y+1}$.\hfill$(1)$\\
     Điều kiện $y+1\ge 0,$ nên $y\ge -1.$
     \begin{itemize}
     \item Xét $y=-1$, thay vào $(1)$ được $x=\pm 1.$
     \item Xét $y=0$, thay vào $(1)$ được $x=\pm 1.$
     \item Xét $y\ge 1$ thì $y+1\ge 2.$ Ta biết rằng với $a\ge 2$ thì $\sqrt{a}<a$ nên $\sqrt{y+1}<y+1.$ Từ $(1)$ có \[x^2=y^2+\sqrt{y+1}<y^2+y+1<(y+1)^2.\tag{2}\]
     Mặt khác từ $(1)$ có $x^2>y^2$.\hfill$(3)$\\
     Từ $(2)$ và $(3)$ suy ra $y^2<x^2<(y+1)^2$, loại.
\end{itemize}
\textit{Đáp số:} Nghiệm $(x;y)$ là $(1;-1), (-1;-1), (1;0), (-1;0).$      
\end{enumerate}            
    }
\end{bt}

\begin{bt}
Tìm nghiệm nguyên của mỗi phương trình sau:
\begin{multicols}{2}
\begin{enumerate}[a)]
\item $2\sqrt{x}-3\sqrt{y}=\sqrt{48}$
\item $\sqrt{x}+\sqrt{x+4}=y$
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}[a)]
\item Xét phương trình $2\sqrt{x}-3\sqrt{y}=4\sqrt{3}$.\hfill$(1)$\\
Điều kiện: $y\ge  0; x\ge 0$.\\
Rút $3\sqrt{y}$ từ $(1)$ được $3\sqrt{y}=2\sqrt{x}-4\sqrt{3}$\\
Với điều kiện $x\ge 12$, bình phương hai vế được $9y=4x+48-16\sqrt{3x}$\\
 Suy ra $3x$ là số chính phương nên $x=3a^2$ $(a\in \mathbb{N})$.\\      
 Rút $2\sqrt{x}$ từ $(1)$ và cũng lập luận tương tự như trên ta được $y=3b^2$ $(b\in \mathbb{N})$.\\
Thay vào $(1)$ ta được $2a\sqrt{3}-3b\sqrt{3}=4\sqrt{3} \Leftrightarrow 2a-3b=4$.\\
Ta thấy $b \vdots 2$. Đặt $b=2t$ $(t\in \mathbb{N})$ thì $a-3t=2$ nên $a=3t+2$.\\
Khi đó \\
$x=3a^2=3(3t+2)^2$ (thỏa mãn $x\ge 12$)\\
$y=3b^2=12t^2$ với $t\in \mathbb{N}$.\\
Đáp số $\left\{\begin{array}{l}
x=3(3t+2)^2\\
y=12t^2
\end{array}\right.$ với $t$ là số tự nhiên tùy ý.
\item Ta có:
{\allowdisplaybreaks
\begin{align*}
&\sqrt{x}+\sqrt{x+4}=y\tag{1}\\
\Leftrightarrow&2x+4+2\sqrt{x(x+4)}=y^2\Leftrightarrow 2\sqrt{x(x+4)}=y^2-2x-4.
\end{align*}}Đặt $y^2-2x-4=m$ với $(m\in \mathbb{N})$ ta có\\
$4x(x+4)=m^2 \Leftrightarrow (2x+4)^2-m^2=16 \Leftrightarrow (2x+4+m)(2x+4-m)=16$\\
Ta tìm được $x=0$, thay vào $(1)$ được $y^2=4$.\\
 Đáp số: Nghiệm $(x;y)=(0;2)$\\
 \textbf{Lưu ý.} Có thể giải phương trình $4x(x+4)=m^2$ với $x\ge 1$ như sau:\\
 Từ $4x(x+4)=m^2$, suy ra $x(x+4)$ là số chính phương. Điều này không xảy ra đối với $x\ge 1$, vì khi đó $(x+1)^2<x^2+4x<(x+2)^2$. 
\end{enumerate}    
}
\end{bt}

\begin{bt}
Phương trình sau có bao nhiêu nghiệm nguyên?
\[\sqrt{x}+\sqrt{y}=\sqrt{2000}.\tag{1}\]
\loigiai{
Lập luận như ví dụ \ref{laoth1} (ở trang \pageref{laoth1}),  ta đưa về 
\[a\sqrt{5}+b\sqrt{5}=20\sqrt{5} \; (a;b\in \mathbb{N} \Leftrightarrow a+b=20.\tag{2}\]
Ứng với một cặp số tự nhiên $(a;b)$ có một cặp số tự nhiên $(x;y)$ thỏa mãn $(1)$. Ta thấy $a$ nhận các giá trị $0;1;2; \ldots ;20$, tương ứng $b$ nhận các giá trị $20;19;18; \ldots; 0$.\\
Phương trình $(2)$ có $21$ nghiệm nguyên, do đó phương trình $(1)$ có $21$ nghiệm nguyên.
}
\end{bt}

\begin{bt}
	Tìm nghiệm nguyên của mỗi phương trình sau:
	\begin{enumerate}[a)]
		\item $ y=x+ \sqrt{x+2+2 \sqrt{x+1}}. $
		\item $ y \sqrt2 = \sqrt{x+ \sqrt{2x-1}}+ \sqrt{x- \sqrt{2x-1}}. $
		\item $ y= \sqrt{x-1-2 \sqrt{x-2}}+ \sqrt{x+2-4 \sqrt{x-2}}. $
	\end{enumerate}	
\loigiai{
\begin{enumerate}[a)]
	\item Đặt $ t= \sqrt{x+1} $ ($ t \in \mathbb{N} $) thì $ x=t^2-1. $ 
	
	Ta có $ y=t^2-1+ \sqrt{t^2+1+2t}=t^2-1+t+1=t^2+t. $ 
	
	\textit{Đáp số}: Nghiệm $ (x;y) $ là $ (t^2-1;t^2+t) $ với $ t $ là số tự nhiên tùy ý.
	\item $ 2y = \sqrt{2x+ 2 \sqrt{2x-1}}+ \sqrt{2x- 2 \sqrt{2x-1}} = \sqrt{2x-1} +1+ \left|  \sqrt{2x-1}-1 \right|. $
	
	Điều kiện: $2x-1 \geq 0 \Leftrightarrow x \geq \dfrac{1}{2}. $ Do $x$ là số nguyên nên $ x \geq 1. $ 
	
	Do đó $ 2y= \sqrt{2x-1}+1+ \sqrt{2x-1}-1=2 \sqrt{2x-1} $ nên $ y= \sqrt{2x-1}. $
	
	Đặt $2x-1=(2t+1)^2$ ($ t \in \mathbb{N} $) thì $ x=2t^2+2t+1. $
	
	\textit{Đáp số}: Nghiệm $(x;y)$ là $(2t^2+2t+1;2t+1)$ với $ t \in \mathbb{N}. $
	\item Điều kiện: $x \geq 2.$ Rút gọn được $ y= \left| \sqrt{x-2}-1 \right| + \left|  \sqrt{x-2}-2 \right|. $
	
	Ta thấy $\sqrt{x-2} \geq 1 \Leftrightarrow x \geq 3$ và $ \sqrt{x-2} \geq 2    \Leftrightarrow x \geq 6.$
	
	Xét các trường hợp sau:
	\begin{itemize}
		\item Với $ x=2 $ thì $ y=1+2=3. $
		\item Với $ 3 \leq x< 6 $ thì $ y= \sqrt{x-2}-1+2- \sqrt{x-2}=1. $ 
		
		Ta có các nghiệm $ (x;y) $ là $ (3;1), (4;1), (5;1). $ 
		\item Với $ x \geq6 $ thì $ y= \sqrt{x-2}-1+ \sqrt{x+2}-2=2 \sqrt{x-2}-3. $
		
		Đặt $ x-2=t^2$ ($t\in \mathbb{N}, t \geq 2 \text{ để } x \geq 6$) thì $ x=t^2+2; y=2t-3. $
    \end{itemize}	
		\textit{Đáp số}: Nghiệm $ (x;y) $ là $ (2;3), (3;1), (4;1), (5;1) $ và $ (t^2+2;2t-3) $ với $ t \in \mathbb{N}, t \geq 2. $ 	
\end{enumerate}
}
\end{bt}

\begin{bt}
Tìm nghiệm nguyên của mỗi phương trình sau: 
\begin{multicols}{2}
\begin{enumerate}[a)]
\item $y=\sqrt{6x+\sqrt{5x+\sqrt{4x+\sqrt{x}}}}$.
\item $y=\sqrt{x+\dfrac{1}{2}+\sqrt{x+\dfrac{1}{4}}}$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}[a)]
\item $y=\sqrt{6x+\sqrt{5x+\sqrt{4x+\sqrt{x}}}}$.\\ 
Điều kiện $x\geqslant 0, y\geqslant 0$.\\ 
Phương trình tương đương $\sqrt{5x+\sqrt{4x+\sqrt{x}}}=y^2-6x$.\\ 
Bình phương hai vế rồi chuyển vế ta được $\sqrt{4x+\sqrt{x}}=k^2-5x$.\\ 
Đặt $\sqrt{4x+\sqrt{x}}=m\ (m\in \mathbb{N})$. Suy ra $4x+\sqrt{x}=m^2$.\\ 
Đặt $\sqrt{x}=a\ (a\in \mathbb{N})$. Đưa về phương trình ẩn $a$ ta được $4a^2+a-m^2=0$.\\ 
Ta có $\Delta =1+4m^2$.\\ 
$\Delta$ là số chính phương $\Leftrightarrow 1+4m^2=n^2\ (n\in \mathbb{N})\Leftrightarrow (n+2m)(n-2m)=1$.\\ 
Ta tìm được $m=0, a=0, x=0, y=0$.\\ 
Đáp số: Nghiệm $(x;y)$ là $(0;0)$.
\item  $y=\sqrt{x+\dfrac{1}{2}+\sqrt{x+\dfrac{1}{4}}}$ \hfill $(1)$\\ 
Đặt $\sqrt{x+\dfrac{1}{4}}=k$ (với $k\geqslant 0$) thì $x+\dfrac{1}{4}=k^2$. Thay vào $(1)$ được \\
$y=\sqrt{k^2+\dfrac{1}{4}+k}=k+\dfrac{1}{2}$. \hfill $(2)$\\
Để $y\in \mathbb{N}$ thì $k=t+\dfrac{1}{2}$ với $t\in \mathbb{N}$.\\ 
Khi đó $y=t+1, x=k^2-\dfrac{1}{4}=\left(t+\dfrac{1}{2}\right)^2-\dfrac{1}{4}=t^2+t=t(t+1)$.\\ 
Đáp số: $\heva{& x=t(t+1)\\& y=t+1}$ với $t$ là số tự nhiên tùy ý. 
\end{enumerate}
}
\end{bt}

\begin{bt}
Tìm nghiệm nguyên của phương trình $\sqrt{x+4\sqrt{x+4\sqrt{x+...+4\sqrt{x+4\sqrt{5x}}}}}=x$ trong đó vế trái có $100$ dấu căn.
\loigiai{
$\sqrt{x+4\sqrt{x+4\sqrt{x+\ldots +4\sqrt{x+4\sqrt{5x}}}}}=x$\hfill $(1)$\\
Ta thấy $x\geqslant 0$. Xét các trường hợp sau:
\begin{itemize}
\item $x=0$ thì $(1)$ được nghiệm đúng.
\item $x=5$ thì $(1)$ được nghiệm đúng.
\item $0<x<5$. Ta có $x^2<5x\Rightarrow x<\sqrt{5x} \Rightarrow 4x< 4\sqrt{5x} \Rightarrow 5x < x+4\sqrt{5x} \Rightarrow \sqrt{5x}< \sqrt{x+4\sqrt{5x}}$. Cứ như thế thì ở phương trình $(1)$ giá trị vế trái lớn hơn $\sqrt{5x}$, mà $x<\sqrt{5x}$ nên giá trị vế phải nhỏ hơn giá trị vế trái. Loại.
\item $x>5$. Lập luận tương tự, ở phương trình $(1)$ giá trị vế trái nhỏ hơn $\sqrt{5x}$, mà $x>\sqrt{5x}$ nên giá trị vế phải lớn hơn giá trị vế trái. Loại.\\ 
Đáp số: $x=0$ và $x=5$.
\end{itemize}     
}
\end{bt}

\begin{bt}
Tìm nghiệm nguyên dương của mỗi phương trình sau:
\begin{enumerate}[a)]
\item $2\sqrt{x}+2\sqrt{y-3}+2\sqrt{z}=x+y+z$.
\item $\sqrt{x}+\sqrt{y}=\sqrt{z+2\sqrt{2}}$. \end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item Đưa về $(\sqrt{x}-1)^2+(\sqrt{y-3}-1)^2+(\sqrt{z}-1)^2=0$.\\ 
Đáp số: Nghiệm $(x;y;z)$ là $(1;4;1)$.
\item Giả sử $x\geqslant y$. Từ $\sqrt{x}+\sqrt{y}=\sqrt{z+2\sqrt{2}}$ ta có \\
$x+y+2\sqrt{xy}=z+2\sqrt{2}\Leftrightarrow 2\sqrt{xy}=z-x-y+2\sqrt{2}$\hfill $(1)$\\
Với điều kiện $z-x-y+2\sqrt{2}\geqslant 0$ \hfill $(2)$ \\
thì $(1)\Leftrightarrow 4xy=(z-x-y)^2+8+4\sqrt{2}(z-x-y)$.\\
Do $z-x-y$ nguyên nên chỉ có thể $z-x-y=0$ thỏa mãn điều kiện $(2)$.\\ 
Khi đó $4xy=8\Leftrightarrow xy=2$. Do $x\geqslant y$ nên $x=2,y=1$.\\ 
Đáp số: Nghiệm $(x;y;z)$ là $(2;1;3),(1;2;3)$.
\end{enumerate}
}
\end{bt}
