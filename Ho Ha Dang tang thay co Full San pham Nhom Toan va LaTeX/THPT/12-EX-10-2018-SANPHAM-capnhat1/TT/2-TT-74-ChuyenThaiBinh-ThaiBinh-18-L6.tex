	\choice
	{$T=2$}
	{\True $T=-16$}
	{$T=-2$}
	{$T=16$}
	\loigiai{
		Đặt $\heva{&u=\ln (x^2+16)\\& \mathrm{\,d}v=x\mathrm{\,d}x}\Rightarrow \heva{&\mathrm{\,d}u=\dfrac{2x}{x^2+16}\mathrm{\,d}x\\& v=\dfrac{1}{2}(x^2+16).}$\\
		$\begin{aligned}
			\displaystyle \int\limits_{0}^{3}x\ln (x^2+16)\mathrm{\,d}x&=\left[\dfrac{1}{2}(x^2+16)\cdot \ln (x^2+16)\right] \bigg|_0^3-\int\limits_{0}^{3}\dfrac{2x}{x^2+16}\cdot\dfrac{1}{2}(x^2+16) \mathrm{\,d}x\\&=\dfrac{1}{2}\cdot 25\cdot \ln 25-\dfrac{1}{2}\cdot 16\cdot \ln 16-\displaystyle\int\limits_{0}^{3}x\mathrm{\,d}x\\&=25\ln 5-16\ln 4-\dfrac{x^2}{2}\bigg|_0^3\\&=25\ln 5-32\ln 2-\dfrac{9}{2}.
		\end{aligned}$ \\
		$\Rightarrow a=25,b=-32,c=-9 \Rightarrow T=a+b+c=25-32-9=-16$.
	}
