	Cho biểu thức $A=\dfrac{2\sqrt{x}}{x-9}-\dfrac{2}{\sqrt{x}+3}$ và $B=\dfrac{6}{x-3\sqrt{x}}$ với $x>0,x\ne 9$.

	\item Rút gọn biểu thức $A$.

	\end{enumerate}

	\item $A=\dfrac{6}{x-9}$.

\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2016-2017, Đống Đa, Hà Nội]%[9D4K8]
	Giải bài toán bằng cách lập phương trình hoặc hệ phương trình.\\
Hai vòi nước cùng chảy vào một bể thì sau 6 giờ thì đầy bể. Nếu mở vòi 1 chảy một mình trong 3 giờ rồi khóa lại, mở vòi 2 chảy tiếp trong 4 giờ thì lượng nước chảy được bằng $60\%$ bể. Hỏi nếu mỗi vòi chảy riêng thì trong bao lâu sẽ đầy bể.
	\loigiai{
	Gọi $x,y$ (giờ) là thời gian vòi 1, vòi 2 tương ứng chảy một mình đầy bể.\\
	Ta có hệ phương trình $\begin{cases}6\left(\dfrac{1}{x}+\dfrac{1}{y}\right)=1\\
	3\cdot\dfrac{1}{x}+4\cdot\dfrac{1}{y} = \dfrac{3}{5}\end{cases}$. Giải được $x=15$ và $y=10$.\\
	Vậy nếu chảy riêng thì vòi 1 chảy trong 15 giờ thì đầy bể và vòi 2 chảy trong 10 giờ thì đầy bể.
		}

\begin{ex}%[Đề thi Toán 9 Học kỳ 2 năm học 2016-2017, Đống Đa, Hà Nội]%[9D3B3]
	Cho đường tròn tâm $O$ bán kính$R$ có hai đường kính $AB$ và $CD$ vuông góc với nhau. Lấy điểm $M$ bất kì thuộc đoạn $OA$ ($M$ khác $O,A$). Tia $DM$ cắt đường tròn $(O)$ tại $N$.
	\begin{enumerate}
	\item Chứng minh rằng tứ giác $OMNC$ nội tiếp được trong một đường tròn.
	\item Chứng minh rằng $DM\cdot DN=DO\cdot DC=2R^2$.
	\item Tiếp tuyến tại $C$ với đường tròn $(O)$ cắt tia $DM$ tại $E$, đường tròn ngoại tiếp tam giác $CDE$ cắt $BC$ tại $F$. Chứng minh $DF//AN$.
	\item Nối $B$ với $N$ cắt $OC$ tại $P$. Tìm vị trí của điểm $M$ để $\dfrac{OM}{AM}+\dfrac{OP}{CP}$ đạt giá trị nhỏ nhất.
	\end{enumerate}
	\loigiai{
\begin{tikzpicture}
 \tkzInit[ymin=-3,ymax=5,xmin=-6,xmax=5]
\tkzDefPoints{-2/0/A,3/0/B}
\tkzDrawPoints(A,B)
\tkzDefMidPoint(A,B)\tkzGetPoint{O}
\tkzDrawPoints(O)
\tkzDrawCircle(O,B)
\tkzLabelPoints(A,B,O)
\tkzDrawSegments(A,B)
\tkzDefPointBy[rotation=center O angle 90](B)
\tkzGetPoint{C}
\tkzDefPointBy[rotation=center O angle 90](A)
\tkzGetPoint{D}
\tkzDrawSegments(C,D)
\tkzDefPoints{-0.5/0/M}
\tkzInterLC(D,M)(O,B)\tkzGetSecondPoint{N}
\tkzDrawSegments(N,D N,C B,C)
\tkzDrawPoints(C,D,M,N)
\tkzLabelPoints(C,D,M,N)
\tkzDefPoints{-3/2.5/X,4/2.5/Y}
\tkzDrawSegments(X,Y)
\tkzInterLL(D,M)(X,Y)\tkzGetPoint{E}
\tkzCircumCenter(C,E,D)\tkzGetPoint{I}
\tkzDrawCircle(I,C)
\tkzInterLC(B,C)(I,C)\tkzGetSecondPoint{F}
\tkzDrawSegments(N,E A,N D,F)
\tkzDrawPoints(E,F)
\tkzLabelPoints(E,F)
\end{tikzpicture}
\begin{enumerate}
\item Tứ giác $OMNC$ có $\widehat{MOC}+\widehat{MNC}=180^o$ nên nội tiếp được trong một đường tròn.
\item Xét $\triangle DON$ và $\triangle DMC$ có:\\
$\widehat{D}$ chung\\
$\widehat{DCM}=\widehat{DNO}$\\
Suy ra $\triangle DON \backsim \triangle DMC$\\
Suy ra $\dfrac{DN}{DC}=\dfrac{DO}{DM}$\\
Vậy $DM\cdot DN=DO\cdot DC=2R^2$.
\item $\widehat{AND}=\widehat{NDF}=45^o$.
\item $\dfrac{AM}{MO}=\sqrt{2}$.
\end{enumerate}
	}
\end{ex}
