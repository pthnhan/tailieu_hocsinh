\begin{ex}%[Đề thi vào 10, Sở giáo dục Thanh Hóa, 2016]%[9D2B2]%[9D3B3]
	\hfill
	\begin{enumerate}[1)]
		\item Giải các phương trình sau:
		
		a) $x-5=0$.
		
		b) $x^2-4x+3=0$.
		
		\item Giải hệ phương trình: $\begin{cases}
		2x-y=1\\3x+y=4.
		\end{cases}$
		
	\end{enumerate}
	\loigiai{
	\begin{enumerate}[1)]
		\item 
		\hfill
		\begin{enumerate}[a)]
			\item $x-5=0 \Leftrightarrow x=5$.  Vậy tập nghiệm của phương trình là $\{5\}$.
			\item $x^2-4x+3=0$. Có $a+b+c=1-4+3=0$ nên phương trình có hai nghiệm $x=1,x=3$.
			
			Vậy tập nghiệm của phương trình là $\{1;3\}$.
		\end{enumerate}
		\item $\begin{cases}
		2x-y=1\\3x+y=4.
		\end{cases} \Leftrightarrow \begin{cases}
		5x=5 \\ 3x+y=4
		\end{cases} \Leftrightarrow \begin{cases}
		x=1 \\ 3.1+y=4
		\end{cases} \Leftrightarrow \begin{cases}
		x=1\\y=1.
		\end{cases}$
		
		Hệ có nghiệm duy nhất $(1;1)$.
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Thanh Hóa, 2016]%[9D2B2]%[9D3B3]
	Cho biểu thức $A = \left( \displaystyle \frac{x\sqrt{x}-1}{x- \sqrt{x}}- \frac{x \sqrt{x}+1}{x+\sqrt{x}}\right):\displaystyle \frac{2\left(x-2\sqrt{x}+1\right)}{x-1}$ (với $x >0$ và $x \ne 1$). 
	
	\begin{enumerate}[1)]
		\item Rút gọn biểu thức $A$.
		
		\item Tìm các số nguyên $x$ để biểu thức $A$ có giá trị nguyên.
	\end{enumerate}
\loigiai
	{
	\begin{enumerate}[1)]
		\item Có
		
		$A= \left[
		\displaystyle \frac{\left(\sqrt{x}-1\right) \left( x + \sqrt{x} +1\right)}{\sqrt{x}\left(\sqrt{x}-1\right)}- 
		\frac{\left(\sqrt{x}+1\right) \left( x - \sqrt{x} +1\right)}{\sqrt{x}\left(\sqrt{x}+1\right)}
		\right]:
		\displaystyle \frac{2 \left(\sqrt{2}-1\right)^2}{\left(\sqrt{x}-1\right) \left( \sqrt{x}+1\right)}$
		
		$= \left(
		\displaystyle \frac{ x + \sqrt{x} +1}{\sqrt{x}}- 
		\frac{ x - \sqrt{x} +1}{\sqrt{x}}
		\right):
		\displaystyle \frac{2 \left(\sqrt{2}-1\right)}{\left( \sqrt{x}+1\right)}$
		
		$\displaystyle =\frac{2 \sqrt{x}}{\sqrt{x}} \cdot \frac{\sqrt{x}+1}{2\left(\sqrt{x}-1\right)}$
		
		$\displaystyle =\frac{\sqrt{x}+1}{\sqrt{x}-1}$.
		
		\item $A = \displaystyle \frac{\sqrt{x}-1+2}{\sqrt{x}-1} =1 + \frac{2}{\sqrt{x}-1}$.
		
		Vì $x$ nguyên nên ta có $A$ nguyên $\Leftrightarrow \displaystyle \frac{2}{\sqrt{x}-1}$ nguyên $\Leftrightarrow \sqrt{x}-1$ là ước của $2$.
		
		Mặt khác $x>0, x \ne 1$ nên $\sqrt{x}-1 > -1$. Do đó 
		
		Vậy $x=4$ hoặc $x=9$ thỏa mãn đề bài.
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Thanh Hóa, 2016]%[9D2B2]%[9D3B3]
	Trong mặt phẳng tọa độ $Oxy$, cho đường thẳng $(d): y =mx+1$ và parabol $(P): y=2x^2$.
	\begin{enumerate}[1)]
		\item Tìm $m$ để đường thẳng $(d)$ đi qua điểm $A(1;3)$.
		
		\item Chứng minh rằng đường thẳng $(d)$ luôn cắt parabol $(P)$ tại hai điểm phân biệt $A(x_1;y_1), B(x_2;y_2)$. Hãy tính giá trị của biểu thức $T=x_1x_2+y_1y_2.$
	\end{enumerate}
\loigiai{
	\begin{enumerate}[1)]
		\item Đường thẳng $(d)$ đi qua điểm $A(1;3) \Leftrightarrow 3=m . 1+1 \Leftrightarrow m=2.$
		
		Vậy $m=2$.
		
		\item Xét phương trình hoàng độ giao điểm của $(d)$ và $(P)$:
		
		$2x^2=mx+1 \Leftrightarrow 2x^2-mx-1=0$\hfill  (1)
		
		Có $\Delta = m^2-4.2.(-1)=m^2+8$. Vì $m^2 \ge 0, \forall m \Rightarrow m^2+8 \ge 8 >0, \forall m$.
		
		Suy ra phương trình (1) luôn có $2$ nghiệm phân biệt với mọi $m$ $\Rightarrow$ $(d)$ luôn cắt $(P)$ tại $2$ điểm phân biệt $A(x_1;y_1), B(x_2;y_2) \forall m$, trong đó $x_1,x_2$ là hai nghiệm của (1) và $y_1=2x_1^2, y_2=2x_2^2$.
		
		Theo định lý Viét ta có: $x_1x_2= \displaystyle -\frac{1}{2} \Rightarrow T =x_1x_2 + 2x_1^2.2x_2^2 = x_1x_2+4(x_1x_2)^2=\frac{1}{2}$
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Thanh Hóa, 2016]%[9D2B2]%[9D3B3]
	Cho tứ giác $ABCD$ nội tiếp đường tròn đường kính $AD$. Hai đường chéo $AC$ và $BD$ cắt nhau tại $E$. Gọi $F$ là điểm thuộc đường thẳng $AD$ sao cho $EF$ vuông góc với $AD$. Đường thẳng $CF$ cắt đường tròn đường kính $AD$ tại điểm thứ hai là $M$. Gọi $N$ là giao điểm của $BD$ và $CF$. Chứng minh rằng:
	\begin{enumerate}[1)]
		\item Tứ giác $CEFD$ nội tiếp đường tròn.
		\item $FA$ là đường phân giác của góc $\widehat{BFM}$.
		\item $BD \cdot NE = BE \cdot ND$.
	\end{enumerate}
\loigiai{
	\begin{center}
		\begin{tikzpicture}
		\draw (0,0) circle  (4cm);
		\tkzDefPoints{0/0/O, -4/0/A, 4/0/D, -2/3.464/B, 1/3.873/C}
		\coordinate (E) at ($(A)!.685!(C)$);
		\coordinate (F) at ($(O)!.15!(A)$);
		\coordinate (N) at ($(F)!.55!(C)$);
		\coordinate (P) at ($(O)!.3!(D)$);
		\tkzDrawSegments[thick](A,D A,B B,D C,D B,C A,C E,F B,F C,F N,P)
		\tkzMarkRightAngle(E,F,O)
		\tkzMarkRightAngle[size=.1](A,C,D)
		\tkzMarkAngles[size=5mm](N,P,O)
		\tkzLabelPoints[below](O)
		\tkzLabelPoints[below left](A)
		\tkzLabelPoints[above left](B)
		\tkzLabelPoints[above right](C)
		\tkzLabelPoints[below right](D)
		\tkzLabelPoints[above](E)
		\tkzLabelPoints[below left](F)
		\tkzLabelPoints[above right](N)
		\tkzLabelPoints[below](P)
		\end{tikzpicture}
	\end{center}
	\begin{enumerate}[1)]
		\item Có $\widehat{ACD}=90^{\circ}$ (góc nội tiếp chắn nửa đường tròn), hay $\widehat{ECD}=90^{\circ}$.
		
		Mặt khác, $EF \perp AD$ nên $\widehat{EFD}=90^{\circ}$.
		
		Suy ra $\widehat{ECD}+\widehat{EFD}=180^{\circ}$, do đó $CEFD$ là tứ giác nội tiếp.
		
		\item Vì $CEFD$ là tứ giác nội tiếp (cmt) nên $\widehat{CFD}=\widehat{CED}$ (do hai góc nội tiếp cùng chắn cung $CD$). \hfill (1)
		
		Chứng minh tương tự có tứ giác $ABEF$ nội tiếp, suy ra $\widehat{BFA}=\widehat{BEA}$ (do hai góc nội tiếp cùng chắn cung $AB$).\hfill (2)
		
		Có $\widehat{BEA}=\widehat{CED}$, $\widehat{AFM}=\widehat{CFD}$.\hfill (3)
		
		Từ (1), (2) và (3) suy ra $\widehat{BFA}=\widehat{AFM}$, do đó $FA$ là tia phân giác của $\widehat{BFM}$.
		
		\item Vẽ $NP \parallel BF (P \in AD)$.
		
		Ta có $\widehat{NPF}=\widehat{BFA}$ (đồng vị); $\widehat{BFA}=\widehat{NFP} \Rightarrow \widehat{NPF}=\widehat{NFP} \Rightarrow \triangle NFD \text{ cân tại } N \Rightarrow NP=NF$.
		
		Vì $NP \parallel BF$ nên $\dfrac{NP}{BF}=\dfrac{DN}{DB} \Rrightarrow \dfrac{NF}{BF}=\dfrac{DN}{DB}$. \hfill (4)
		
		Vì $\widehat{BFA}=\widehat{NFP}$ nên $\widehat{EFB}=\widehat{EFN}$ (cùng phụ với hai góc bằng nhau).
		
		Suy ra $FE$ là phân giác $\widehat{BFN}$ của $\triangle BFN$. Theo định lí đường phân giác ta có $\dfrac{NF}{BF}=\dfrac{NE}{BE}$ \hfill (5)
		
		Từ (4) và (5) $\Rightarrow \dfrac{DN}{DB}=\dfrac{NE}{BE} \Rightarrow BD.NE=BE.DN$ (đpcm).
	\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Thanh Hóa, 2016]%[9D2B2]%[9D3B3]
	Cho $a,b,c$ là các số thực dương thỏa mãn: $a^2+2b^2 \le 3c^2$. Chứng minh rằng: $ \displaystyle \frac{1}{a}+ \frac{2}{b} \ge \frac{3}{c}.$
	\loigiai{		
		Áp dụng bất đẳng thức Bunhiacopxki cho hai bộ số, ta có
		$$(a+2b)^2=(1.a+\sqrt{2}.\sqrt{2}b)^2 \leq (1+2)(a^2+2b^2) \leq 3.3c^2=9c^2 \Rightarrow a+2b \leq 3c.$$		
		Với mọi $x, y, z >0$, áp dụng bất đẳng thức Côsi cho 3 số dương ta có
		$$(x+y+z)\left(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\right) \geq 3 \sqrt[3]{xyz}.3 \sqrt[3]{\dfrac{1}{xyz}}=9 \Rightarrow \dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z} \geq \dfrac{9}{x+y+z}.$$		
		Áp dụng bất đẳng thức trên, ta có 
		$\dfrac{1}{a}+\dfrac{2}{b}=\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{b} \geq \dfrac{9}{a+b+c}=\dfrac{9}{a+2b} \geq \dfrac{9}{3c}=\dfrac{3}{c}$ (đpcm).
		
		Dấu "=" xảy ra $\Leftrightarrow a=b=c$.
	}
\end{ex}
