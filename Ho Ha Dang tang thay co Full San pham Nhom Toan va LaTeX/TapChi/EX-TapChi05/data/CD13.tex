\section{Phương tích, trục đẳng phương}
\begin{center}\textbf{\color{violet}{Trường THPT chuyên Thái Nguyên}}\end{center}
\noindent Phương tích, trục đẳng phương là một trong những công cụ đã rất quen thuộc trong giải toán hình học phẳng. Kiến thức liên quan đến chúng khá đơn giản nhưng lại có nhiều ứng dụng mạnh trong các bài toán hình học phẳng, có thể xử lý các bài toán khó với cách xử lí đẹp và ấn tượng. 
Trong bài viết này, tác giả đề cập đến một số ứng dụng cơ bản của phương tích, trục đẳng phương như bài toán chứng minh tứ giác nội tiếp; vuông góc, song song; thẳng hàng, đồng quy; các yếu tố cố định... nhằm giúp các em học sinh nắm bắt được phần nào ứng dụng của chúng.

\subsection{Kiến thức cơ sở}	

\subsubsection{Phương tích của một điểm đối với một đường tròn}
\begin{dl}
	Cho đường tròn $(O; R)$ và điểm $P$ cố định, $OP = d$. Một đường thẳng thay đổi qua $P$ cắt đường tròn tại hai điểm $M$ và $N$. Khi đó $\overline{PA}.\overline{PB}=PO^2-R^2=d^2-R^2$.
\end{dl}
\begin{center}
	\begin{tikzpicture}[>=stealth,line join=round,line cap=round,font=\footnotesize,scale=1]
	\tkzDefPoints{0/0/O,2.5/0/I,-5/0/P}

	\tkzDefPointBy[symmetry = center O](M')    \tkzGetPoint{M}

	\tkzDrawSegments(P,N M,M')
	
	\tkzDrawPoints[fill=black](O,M',M,P,N)
	\tkzLabelPoints[below](O)
	\tkzLabelPoints[left](P)
	\tkzLabelPoints[above left](M)
	\tkzLabelPoints[above right](N)
	\tkzLabelPoints[right](M')	
	\end{tikzpicture}
\end{center}
\noindent{\bf Chứng minh.} 
Gọi $M'$ là điểm đối xứng của $M$ qua $O$. Ta có $MN\perp NM'$. Khi đó: {\allowdisplaybreaks \begin{align*}
	\overline{PM}.\overline{PN}&\, =\,\vec{PM}.\vec{PN}=\overline{PM'}\overline{PN}=\left(\overline{PO}+\overline{OM}\right)\left(\overline{PO}+\overline{OM'}\right)\\ &\,=\,\left(\overline{PO}+\overline{OM}\right)\left(\overline{PO}-\overline{OM}\right)=PO^2-OM^2=PO^2-R^2 \end{align*}}

\begin{dn}
	Phương tích của điểm $M$ đối với đường tròn $(O;R)$, kí hiệu $\mathscr{P}_{P/(O)}$, được xác định bởi $\mathscr{P}_{P/(O)}=OP^2-R^2$.
\end{dn}

\noindent{\bf Các tính chất.}
\begin{tc}\text{}
	\begin{itemize}
		\item Nếu $A, B$ cố định và $\overline{AB}.\overline{AM}$ bằng hằng số  thì $M$ cố định.
		\item 	Điểm $P$ nằm ngoài $(O; R)$ $\Leftrightarrow \mathscr{P}_{P/(O)}>0$.
		\item 	Điểm $P$ nằm trong $(O; R)$ $\Leftrightarrow \mathscr{P}_{P/(O)}<0$.
		\item 	Điểm $P$ nằm trên $(O; R)$ $\Leftrightarrow \mathscr{P}_{P/(O)}=0$.
	\end{itemize}
\end{tc}
\begin{tc}
	Cho đường tròn $(O; R)$ và một điểm $M$ nằm trên $(O)$. Qua $M$ kẻ cát tuyến $MAB$ và tiếp tuyến $MT$ tới $(O)$. Khi đó $\overline{MA}.\overline{MB}={\overline{MT}}^2=MO^2-R^2.$ 
\end{tc}

\begin{tc}
	Cho hai đường thẳng $AB,CD$ phân biệt, cắt nhau tại $M$ ($M$ khác $A,B,C,D$). Khi đó, nếu $\overline{MA}.\overline{MB}=\overline{MC}.\overline{MD}$ thì bốn điểm $A,B,C,D$ cùng nằm trên một đường tròn.
\end{tc}

\begin{tc}
	Cho hai đường thẳng $AB,MT$ phân biệt cắt nhau tại $M$($M$ khác $A,B,T$). Khi đó, nếu $\overline{MA}.\overline{MB}=MT^2$ thì đường tròn ngoại tiếp tam giác $ABT$ tiếp xúc với $MT$ tại $T$.
\end{tc}

\subsubsection{Trục đẳng phương của hai đường tròn}
\begin{dl}
	Cho hai đường tròn không đồng tâm $\left(O_1;R_1\right)$, $\left(O_2;R_2\right)$. Tập hợp các điểm $M$ có phương tích đối với hai đường tròn bằng nhau là một đường thẳng. Đường thẳng này gọi là trục đẳng phương của hai đường tròn đã cho.
\end{dl}
\noindent{\bf Chứng minh.} 
Giả sử điểm $M$ có phương tích đến hai đường tròn bằng nhau. Gọi $H$ là hình chiếu của $M$ trên $O_1O_2$, $I$ là trung điểm của $O_1O_2$. Ta có:
{\allowdisplaybreaks \begin{align*}
	&\,\mathscr{P}_{M/(O_1)}=\mathscr{P}_{M/(O_2)}\Leftrightarrow MO_1^2-R_1^2=MO_2^2-R_2^2\\ &\,\Leftrightarrow MO_1^2-MO_2^2=R_1^2-R_2^2\\
	&\,\Leftrightarrow \left(MH^2+HO_1^2\right)-\left(MH^2+HO_2^2\right)=R_1^2-R_2^2\\ &\,\Leftrightarrow HO_1^2-HO_2^2=R_1^2-R_2^2\\
	&\, \Leftrightarrow \left(\overline{HO_1}-\overline{HO_1}\right)\left(\overline{HO_1}+\overline{HO_2}\right)=R_1^2-R_2^2\\&\,\Leftrightarrow \overline{O_2O_1}.2.\overline{HI}=R_1^2-R_2^2\\&\,\Leftrightarrow \overline{IH}=\dfrac{R_1^2-R_2^2}{\overline{O_1O_2}}\end{align*}}
Từ đây suy ra $H$ cố định, suy ra $M$ thuộc đường thẳng qua $H$ và vuông góc với $O_1O_2$.

\noindent{\bf Các tính chất.}
\begin{itemize}
	\item Trục đẳng phương của hai đường tròn vuông góc với đường nối tâm.
	\item	Nếu hai đường tròn cắt nhau tại $A$ và $B$ thì $AB$ chính là trục đẳng phương.
	\item	Nếu điểm $M$ có cùng phương tích với hai đường tròn thì đường thẳng qua $M$ và vuông góc với đường nối tâm là trục đẳng phương.
	\item	Nếu hai điểm $M, N$ có cùng phương tích đối với hai đường tròn thì đường thẳng $MN$ là trục đẳng phương.
	\item	Nếu ba điểm có cùng phương tích với hai đường tròn thì chúng thẳng hàng.
	\item	Nếu $\left(O_1\right),\left(O_2\right)$ cắt nhau tại $A$ thì đường thẳng qua $A$ vuông góc với $O_1O_2$ là trục đẳng phương.	
\end{itemize}
\noindent{\bf Cách xác định trục đẳng phương của hai đường tròn không đồng tâm.}\\
Cho hai đường tròn $\left(O_1\right)$ và $\left(O_2\right)$ không cắt nhau, ta có cách dựng trục đẳng phương của hai đường tròn như sau:
\begin{enumerate}
	\item[-] Dựng đường tròn $\left(O_3\right)$ cắt cả hai đường tròn $\left(O_1\right)$ và $\left(O_2\right)$ cắt nhau lần lượt tại $A,B$ và $C,D$.
	\item[-] Đường thẳng $AB$ và $CD$ cắt nhau tại $M$.
	\item[-] Đường thẳng qua $M$ vuông góc với $O_1O_2$ chính là trục đẳng phương của $\left(O_1\right)$ và $\left(O_2\right)$.
\end{enumerate}

\noindent{\bf Tâm đẳng phương của ba đường tròn.}
\begin{dl}
	Cho ba đường tròn $\left(O_1\right), \left(O_2\right)$ và $\left(O_3\right)$. Khi đó, $3$ trục đẳng phương của các cặp đường tròn trùng nhau hoặc song song hoặc cùng đi qua một điểm , điểm đó được gọi là tâm đẳng phương của ba đường tròn.
\end{dl}
\noindent{\bf Các tính chất}
\begin{itemize}
	\item Nếu ba đường tròn đôi một cắt nhau thì các dây cung chung cùng đi qua một điểm.
	\item Nếu ba trục đẳng phương song song hoặc trùng nhau thì tâm của ba đường tròn thẳng hàng.
	\item Nếu ba đường tròn cùng đi qua một điểm và có các tâm thẳng hàng thì trục đẳng phương trùng nhau.
\end{itemize}

\subsection{Các dạng toán}

\subsubsection{Ứng dụng giải các bài toán định lượng}
\begin{bt}[Phương tích trọng tâm]
	Cho tam giác $ABC$ nội tiếp đường tròn $(O; R)$. Xác định phương tích của trọng tâm $G$ của tam giác $ABC$ với $(O)$ theo các cạnh $BC = a$, $CA = b$, $AB = c$.
	\loigiai{Do $G$ là trọng tâm tam giác $ABC$ nên
		{\allowdisplaybreaks
			\begin{align*}
			&\vec{OA}+\vec{OB}+\vec{OC}=3\vec{OG}\\
			\Rightarrow& OA^2+OB^2+OC^2+2\vec{OA}.\vec{OB}+2\vec{OB}.\vec{OC}+2\vec{OC}.\vec{OA}=9OG^2.\tag{1}\end{align*}}Ta có 
		\[2\vec{OA}.\vec{OB}=OA^2+OB^2-AB^2=2R^2-c^2;2\vec{OB}.\vec{OC}=2R^2-a^2;2\vec{OC}.A=2R^2-b^2.\tag{2}\]
		Từ $(1)$ và $(2)$ 	Suy ra 
		$OG^2=\dfrac{1}{9}\left(9R^2-a^2-b^2-c^2\right)$.\hfill$(3)$\\
		Do $(3)$ nên $\mathscr{P}_{G/(O)}=OG^2-R^2=-\dfrac{1}{9}\left(a^2+b^2+c^2\right)$.
		
	}
\end{bt}

\begin{bt}[Phương tích trực tâm]

	\loigiai{ Xét trường hợp tam giác $ABC$ nhọn. Gọi $K, A'$ lần lượt là giao điểm của $AH$ với $BC, \left(O\right)$. Áp dụng định lí \textbf{Sin} cho tam giác $AHB$ 
		{\allowdisplaybreaks \begin{align*}&\,\dfrac{AH}{\sin \widehat{ABH}}=\dfrac{AB}{\sin\widehat{AHB}}\Rightarrow \dfrac{AH}{\cos A}=\dfrac{AB}{\sin C}\\
			&\,\Rightarrow HA=\dfrac{AB.\cos A}{\sin C}=\dfrac{c.\cos A}{\cos C}=2R.\cos A
			\end{align*}}
		Tương tự ta cũng có $HB=2R.\cos B$.\\
		Vì $\widehat{BHA'}=\widehat{C}\widehat{BA'A}\Rightarrow \Delta BHA'$ cân tại $B$ nên ta có: $$KH=KA'\Rightarrow HA'=2KH=2.BH.\cos\widehat{BHA'}=2R\cos B\cos C.$$ 
		Do tam giác $ABC$ nhọn nên $\mathscr{P}_{P/(O)}=\overline{HA}.\overline{HA'}=-HA.HA'=-8R^2\cos A\cos B\cos C$.\\
		Trường hợp tam giác $ABC$ vuông hay tù chứng minh tương tự.\\
		\noindent{\bf Nhận xét} 
		$$\mathscr{P}_{P/(O)}=OH^2-R^2=-8R^2\cos A\cos B\cos C\Rightarrow OH^2=R^2\left(1-8\cos A.\cos B\cos C\right)$$
		
	}	
\end{bt}

\begin{bt}[Hệ thức Euler]%[0H0G3]
	Cho $\triangle ABC$ nội tiếp đường tròn $(O,R)$ và ngoại tiếp đường tròn $(I,r)$. Chứng minh rằng $OI^2 = R^2 -2Rr$.

\begin{bt}%[0H0G3]
	Cho $\triangle ABC$ có $BC=a, CA=b, AB=c$. Xét đường tròn $(I)$ sao cho 
	$$\mathscr{P}_{A/(I)} +a^2= \mathscr{P}_{B/(I)} +b^2 = \mathscr{P}_{C/(I)}+c^2=0.$$
	Gọi $R,R'$ lần lượt là bán kính đường tròn ngoại tiếp tam giác $ABC$ và $(I)$, $M$ là trung điểm của đoạn $BC$.
	\begin{enumerate}
		\item Chứng minh rằng $I$ là trực tâm của $\triangle ABC$.
		\item Chứng minh rằng $R'=2R$.
		\item Tính $\mathscr{P}_{M/(I)}$.
	\end{enumerate}

		\begin{enumerate}
			\item Từ giả thiết, suy ra $\mathscr{P}_{A/(I)} = -a^2$, $\mathscr{P}_{B/(I)}= -b^2$, $\mathscr{P}_{C/(I)} = -c^2$.
			\\ Ta có 
			{\allowdisplaybreaks
				\begin{align*}
				\mathscr{P}_{A/(I)} & = IA^2 -R'^2 = -a^2 \Leftrightarrow IA^2 = R'^2 -a^2 = R'^2 - BC^2, \tag{1}\\ 
				\mathscr{P}_{B/(I)} & = IB^2 -R'^2 = -b^2 \Leftrightarrow IB^2 = R'^2 -b^2 = R'^2 - CA^2, \tag{2} \\
				\mathscr{P}_{C/(I)} & = IC^2 -R'^2 = -c^2 \Leftrightarrow IC^2 = R'^2 -c^2 = R'^2 - BA^2. \tag{3}

Mặt khác $\dfrac{I_1A}{IA}=\dfrac{r_1}{r}=\cos \alpha$ suy ra  \[I_1A =IA \cos \alpha \Rightarrow II_1=IA-I_1A=IA(1-\cos \alpha) =2IA\sin^2 \dfrac{\alpha}{2}.\]

			Theo tính chất trực tâm tam giác ta có $\mathscr{P}_{H/(BQ)}=\overline{HB} \cdot \overline{HE}=\overline{HC} \cdot \overline{HF}=\mathscr{P}_{H/(CP)}$.\\
			Vậy $HK$ là trục đẳng phương của hai đường tròn $(BQ)$  và $(CP) \Rightarrow HK \perp IJ$.   
			
			\item Áp dụng định lí Menelaus cho tam giác $KBC$ với cát tuyến $MIJ$, ta có
			$$\dfrac{\overline{MB}}{\overline{MC}} \cdot\dfrac{\overline{JC}}{\overline{JK}} \cdot \dfrac{\overline{IK}}{\overline{IB}}=1 =\dfrac{\overline{NQ}}{\overline{NP}} \cdot \dfrac{\overline{IK}}{\overline{IQ}} \cdot \dfrac{\overline{JP}}{\overline{JK}} \Rightarrow \dfrac{\overline{MB}}{\overline{MC}}=\dfrac{\overline{NQ}}{\overline{NP}}$$

			Tương tự ta cũng có, $\widehat{CAI}=\widehat{BAJ}$.\\
			Từ đó suy ra $\widehat{IAM}=\widehat{JAN}$.
		\end{enumerate}
