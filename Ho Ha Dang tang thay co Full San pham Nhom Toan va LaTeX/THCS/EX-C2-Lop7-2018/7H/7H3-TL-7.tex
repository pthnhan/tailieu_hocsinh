
\section{TÍNH CHẤT ĐƯỜNG TRUNG TRỰC CỦA MỘT ĐOẠN THẲNG}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Định lí về tính chất các điểm thuộc đường trung trực}
\begin{dl}
\textbf{(Định lí thuận)} Điểm nằm trên đường trung trực của một đoạn thẳng thì cách đều hai đầu mút của đoạn thẳng đó.
\end{dl}
Như vậy, với đoạn thẳng $AB$ và đường trung trực $d$ của nó thì \lq\lq$M\in d\Rightarrow MA=MB$\rq\rq.
\begin{note}
\textbf{Nhận xét:} Từ kết quả của định lí trên gợi ý cho chúng ta giải bài toán \lq\lq \textit{Tìm điểm $M$ thuộc hình $(H)$ cách đều hai điểm $A$ và $B$ cho trước}\rq\rq, bởi khi đó $M$ chính là giao điểm của $(H)$ với đường trung trực của đoạn thẳng $AB$. 
\end{note}
\begin{dl}
	\textbf{(Định lí đảo)} Điểm cách đều hai đầu mút của một đoạn thẳng thì nằm trên đường trung trực của đoạn thẳng đó. 
\end{dl}
Như vậy, với đoạn thẳng $AB$ và đường trung trực $d$ của nó thì
\immini
{
	$$M\in d\Leftrightarrow MA=MB.$$
	Từ hai định lý trên ta có:\\
	\textit{\lq\lq Tập hợp các điểm cách đều hai mút của một đoạn thẳng là đường trung trực của đoạn thẳng đó \rq\rq.}
}
{

	\tkzInit[xmin=-0.3,xmax=4.3,ymin=-0.8,ymax=4]

	\tkzLabelPoints[below](A,B)

	\end{tikzpicture}
}
\immini
{
	\begin{note}
	\textbf{Nhận xét:} Như vậy với $AB$ cho trước thì tập các điểm $C$ sao cho tam giác $ABC$ cân tại $C$ là đường trung trực của đoạn thẳng $AB$, trừ trung điểm $M$ của đoạn thẳng $AB$.
\end{note}
}

\subsubsection{Cách dựng đường trung trực}
\immini
{
	Để dựng đường trung trực $d$ của đoạn thẳng $AB$ cho trước bằng thước và compa, ta thực hiện theo các bước:
	\begin{itemize}
		\item \textbf{Bước 1:} Lấy điểm $A$ làm tâm vẽ cung tròn có bán kính $R>\dfrac{1}{2}AB$.\\
		Lấy điểm $B$ làm tâm vẽ cung tròn có bán kính $R$. Hai cung tròn này cắt nhau tại $P$ và $Q$.
		\item \textbf{Bước 2:} Dùng thước vẽ đường thẳng $PQ$, đó là đường trung trực của đoạn thẳng $AB$.
	\end{itemize}
}
{
	\begin{tikzpicture}[line cap = round, line join=round, scale=0.8]
	\tkzInit[xmin=-0.3,xmax=4.3,ymin=-3.2,ymax=3.2]
	\tkzClip
	\def\r{3}
	\tkzSetUpPoint[color=black,fill=white,size=6]
	\tkzDefPoints{0/0/A, 3/0/a, 4/0/B, 1/0/b, 2/2.8/d}
	\tkzDrawCircle[dashed](A,a)
	\tkzDrawCircle[dashed](B,b)
	\tkzInterCC(A,a)(B,b)\tkzGetPoints{P}{Q}
	\tkzDrawSegments(A,B)
	\tkzDrawLine(P,Q)
	\tkzDrawPoints(A,B,P,Q)
	\tkzLabelPoints[right](d)
	\tkzLabelPoints[below](A,B)
	\tkzLabelPoints[left=2mm](P,Q)
	\end{tikzpicture}
}
\begin{note}
	\textbf{Chú ý:}
	\begin{itemize}
		\item Khi vẽ hai cung tròn trên, ta phải lấy bán kính $R>\dfrac{1}{2}AB$ thì hai cung tròn đó mới có điểm chung.
		\item Giao điểm của hai đường thẳng $PQ$ với đoạn thẳng $AB$ là trung điểm của đoạn thẳng $AB$, do đó cách dựng trên cũng là cách dựng trung điểm của đoạn thẳng bằng thước thẳng và compa.
	\end{itemize}
\end{note}
\subsection{Các dạng toán}
\begin{dang}{Chứng minh tính chất đường trung trực}
	Với đoạn thẳng $AB$ và đường trung trực $d$ của nó thì $M\in d\Leftrightarrow MA=MB$.
\end{dang}
\setcounter{vd}{0}
\begin{vd}%[Thái Văn Đạt, Dự án EX Lớp 7]%[7H3B7]
	Cho đoạn thẳng $AB$. Trên đường trung trực $d$ của đoạn thẳng $AB$ lấy điểm $M$ bất kì. Chứng minh rằng $MA=MB$.
\loigiai{
\immini
{
	Nếu $M$ nằm trên đoạn thẳng $AB$ thì $M$ là trung điểm của đoạn thẳng $AB$ và hiển nhiên $MA=MB$.\\
	Ta xét trường hợp $M$ nằm ngoài đoạn thẳng $AB$ và gọi $H$ là trung điểm của đoạn thẳng $AB$.\\
	Xét hai tam giác $AHM$ và $BHM$ cùng vuông tại $H$ có $HM$ chung và $HA=HB$ nên $\triangle AHM=\triangle BHM\,(2cgv)$.\\
	Khi đó ta thu được $MA=MB$ (đpcm).
}
{

	\tkzDrawPoints(A,M,B,H)
	\tkzLabelPoints[below](A,B,H)

	\tkzMarkRightAngles(M,H,B)
	\end{tikzpicture}
}
}
\end{vd}

\begin{vd}%[Thái Văn Đạt, Dự án EX Lớp 7]%[7H3B7]
	Cho đoạn thẳng $AB$ và điểm $M$ thỏa mãn $MA=MB$. Chứng minh rằng $M$ nằm trên đường trung trực của đoạn thẳng $AB$.
\loigiai{
\immini
{
	Nếu $M\in AB$, kết hợp với $MA=MB$ thì $M$ là trung điểm của $AB$ và hiển nhiên $M$ nằm trên đường trung trực của đoạn thẳng $AB$.\\
	Ta xét trường hợp $M\notin AB$. Khi đó ta dựng $MH\perp AB$.\\
	Xét hai tam giác $AMH$ và $BMH$ cùng vuông tại $H$ có $MH$ chung và $MA=MB$ nên $\triangle AMH=\triangle BMH\, (ch-gn)$.\\
	Khi đó $HA=HB$ hay $H$ là trung điểm của đoạn thẳng $AB$.\\
	Do đó đường thẳng $MH$ là đường trung trực của đoạn thẳng $AB$ hay ta có điều phải chứng minh.
}

\begin{note}
	\textbf{Nhận xét:} Như vậy, để chứng minh đường thẳng $d$ là đường trung trực của đoạn thẳng $AB$ chúng ta có thể lựa chọn một trong hai cách sau:\\
	\textbf{Cách 1:} Chứng minh $d$ đi qua trung điểm $M$ của $AB$ và vuông góc với $AB$.\\
	\textbf{Cách 2:} Lấy hai điểm $P,Q$ nằm trên $d$. Ta đi chứng minh $PA=PB$ và $QA=QB$.
\end{note}

Do đó ta thu được $D$, $E$, $F$ thẳng hàng.
}
\end{vd}

\begin{bt}%[Thái Văn Đạt, Dự án EX Lớp 7]%[7H3B7]
	Cho tam giác $ABC$. Tìm điểm $O$ sao cho:
	\begin{enumerate}
		\item $O$ thuộc trung tuyến $AM$ và cách đều hai đỉnh $A,B$.
		\item $O$ cách đều các đỉnh $A,B,C$.
		\item $O$ cách đều các đỉnh $A,C$ và cách đều các cạnh của góc $\hat{A}$.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $O$ cách đều hai đỉnh $A,B$ nên $O$ nằm trên đường trung trực của đoạn thẳng $AB$.\\
	Do đó $O$ là giao điểm của đường trung trực đoạn thẳng $AB$ và trung tuyến $AM$.
	\item $O$ cách đều các đỉnh $A,B,C$ nên $OA=OB=OC$ hay $O$ là giao điểm của hai đường trung trực của đoạn thẳng $AB$, $AC$.
	\item $O$ cách đều các đỉnh $A,C$ nên $O$ thuộc đường trung trực của đoạn thẳng $AC$.\\
	Lại có $O$ cách đều hai cạnh của góc $\hat{A}$ nên $O$ nằm trên đường phân giác của góc $\widehat{BAC}$.\\
	Do đó $O$ là giao điểm của đường trung trực đoạn thẳng $AC$ và đường phân giác góc $\widehat{BAC}$.
\end{enumerate}
}
\end{bt}
