\loigiai{
\immini
{Ta có $\heva{&BC\perp AB\\&BC\perp SA}\Rightarrow BC\perp SB$ và $\heva{&CD\perp AD\\&CD\perp SA}\Rightarrow CD\perp SD$.\\
    Do đó $\Delta SBC$ vuông tại $B$ và $\Delta SCD$ vuông tại $D$.\\
    Mặt khác ta có $SC$ chung và $BC=CD$ nên suy ra $\Delta SBC=\Delta SCD$.\\
    Vậy $\Delta SBC$ và $\Delta SDC$ là hai tam giác vuông có chung cạnh huyền và không đồng phẳng nên hai đường cao tương ứng hạ từ điểm $B$ và điểm $D$ cùng đi qua một điểm $H$ trên cạnh $SC$ và $BH=DH$.
}
{
    \begin{tikzpicture}[>=stealth, line join=round, line cap = round,scale=0.6]
    \tkzDefPoints{0/0/A, 4/-1.5/C, -1/-1.5/B, 5/0/D,0/4/S}
   \tkzDefPointBy[homothety = center S ratio .6](C)
   \tkzGetPoint{H}
    \tkzDrawSegments(S,B S,C S,D B,C C,D B,H H,D)
   \tkzDrawSegment[dashed](A,S)
   \tkzDrawSegment[dashed](A,B)
   \tkzDrawSegment[dashed](A,D)
   \tkzDrawSegment[dashed](B,D)
   \tkzDrawPoints(S,A,B,C,D,H)
    \tkzLabelPoints[left](A)
    \tkzLabelPoints[below](B)
    \tkzLabelPoints[right](C,D)
    \tkzLabelPoints[above](S,H)
    \tkzMarkRightAngle(S,A,D)
     \tkzMarkRightAngle(S,H,D)
      \tkzMarkRightAngle(S,H,B)
    \end{tikzpicture}
}
  Ta có $\heva{&BH\subset (SBC)\\&BH\perp SC}$ và $\heva{&DH\subset (SCD)\\&DH\perp SC}$ nên góc giữa hai mặt phẳng $(SBC)$ và $(SCD)$ chính là góc giữa hai đường thẳng $BH$ và $DH$.\\
Do $\Delta SBC$ vuông tại $B$ nên $BH=\dfrac{SB\cdot BC}{\sqrt{SB^2+BC^2}}=\dfrac{a\sqrt{2}\cdot a}{\sqrt{2a^2+a^2}}=\dfrac{a\sqrt{2}}{\sqrt{3}}$.\\
  Do đó $\cos \widehat{BHD}=\dfrac{BH^2+DH^2-BD^2}{2BH\cdot DH}=\dfrac{\dfrac{2a^2}{3}+\dfrac{2a^2}{3}-2a^2}{2\cdot\dfrac{2a^2}{3}}=-\dfrac{1}{2}$.\\
Suy ra $\widehat{BHD} = 120^\circ$.\\
Vì góc giữa hai đường thẳng luôn nhỏ hơn hoặc bằng $90^\circ$, do đó góc giữa hai đường thẳng $BH$ và $DH$ bằng $180^\circ-120^\circ=60^\circ$.\\
Vậy góc giữa hai mặt phẳng $(SBC)$ và $(SDC)$ bằng $60^\circ$.
}
\end{ex}

\begin{bt}%[HK2 (2016-2017), THPT Hai Bà Trưng, Huế]%[1D5B3-1]%[Hồ Sỹ Trường]
Cho hàm số $y=\sqrt{3}\sin x-\cos x$ có đồ thị $(\mathscr C)$. Viết phương trình tiếp tuyến của đồ thị $(\mathscr C)$ tại điểm có hoành độ $x=\dfrac{\pi}{2}$.
	\loigiai{
Ta có $y'=\sqrt{3}\cos x+\sin x$.\\
Với $x=\dfrac{\pi}{2}$ ta có $y\left(\dfrac{\pi}{2}\right)=\sqrt{3}$, $y'\left(\dfrac{\pi}{2}\right)=1$.\\
Phương trình tiếp tuyến của đồ thị $(\mathscr C)$ tại điểm có hoành độ $x=\dfrac{\pi}{2}$ là
$$y=1\cdot\left(x-\dfrac{\pi}{2}\right)+\sqrt{3}=x-\dfrac{\pi}{2}+\sqrt{3}.$$
	}
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Hai Bà Trưng, Huế]%[1D5B3-1]%[Hồ Sỹ Trường]
    Cho hàm số $y=\sqrt{3}\sin x-\cos x$. Giải phương trình $y'=0$.
    \loigiai{
Ta có $y'=\sqrt{3}\cos x+\sin x$.\\
$y'=0\Leftrightarrow \sqrt{3}\cos x+\sin x=0\Leftrightarrow \sqrt{3}+\tan x=0\Leftrightarrow \tan x=-\sqrt{3}\Leftrightarrow x= -\dfrac{\pi}{3}+k\pi,\, k\in\mathbb{Z}$.} 
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Hai Bà Trưng, Huế]%[1D5B5-2]%[Hồ Sỹ Trường]
    Cho hàm số $y=\sqrt{3}\sin x-\cos x$. Chứng minh rằng $y+y''=0$.
    \loigiai{
 Ta có $y'=\sqrt{3}\cos x+\sin x$, $y''=-\sqrt{3}\sin x+\cos x$.\\
 Do đó $y+y''=\left(\sqrt{3}\sin x-\cos x\right)+\left(-\sqrt{3}\sin x+\cos x\right)=0$.
 \hfill \rule{0.3cm}{0.3cm}       
    }
