			\tkzDrawSegments[dashed](S,A A,C B,D A,B A,D M,N S,M)
			\tkzDrawPoints(S,A,B,C,D,O,M,N)
			\tkzLabelPoints[below left](B)
			\tkzLabelPoints[below right](C)
			\tkzLabelPoints[above right](D,M)
			\tkzLabelPoints[above left](A,S)
			\tkzLabelPoints[below](O,N)

Gọi $M$, $N$ lần lượt là trung điểm của $AD$ và $BC$.\\
Ta có $MN$ là đường trung bình của hình chữ nhật $ABCD$ nên $MN\perp AD$.\\
Mặt khác $SA\perp(ABCD)\Rightarrow SA\perp MN$. Vậy $MN\perp(SAD)\Rightarrow (SMN)\perp(SAD)$.\\
Do $SO\subset(SMN)$ nên mặt phẳng $(SMN)$ chính là mặt phẳng $(P)$, và thiết diện tạo bởi $(P)$ với hình chóp $S.ABCD$ là tam giác $SMN$.\\
Có $MN=AB=a$, $AM=\dfrac{1}{2}AD=a$, $SM=\sqrt{SA^2+AM^2}=a\sqrt{2}$.\\
Do $\triangle SMN$ vuông tại $M$ nên diện tích thiết diện là $$S=S_{SMN}=\dfrac{1}{2}\cdot SM\cdot MN=\dfrac{1}{2}\cdot a\cdot a\sqrt{2}=\dfrac{a^2\sqrt{2}}{2}.$$}
\end{ex}

\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[1125-HK2-Tiên Lãng-Hải Phòng-2016\_2017]%[Phan Hoàng Anh]%[1D5K5-1]
Cho hàm số $y=\sin^2x$.
\begin{enumerate}
	\item Tính giá trị của $y''\left(\dfrac{\pi}{2}\right)$.
	\item Rút gọn biểu thức $A=4(y+1)+y''$.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $y'=(\sin^2x)'=2\sin x\cdot(\sin x)'=2\sin x\cos x=\sin2x$.\\
	$y''=(y')'=(\sin2x)'=2\cos2x\Rightarrow y''\left(\dfrac{\pi}{2}\right)=2\cos\pi=-2$.
	\item $A=4(y+1)+y''=4(\sin^2x+1)+2\cos2x=4\sin^2x+4+2(1-2\sin^2x)=6$.
\end{enumerate}}
\end{bt}
\begin{bt}%[1125-HK2-Tiên Lãng-Hải Phòng-2016\_2017]%[Phan Hoàng Anh]%[1H3K3-3]%[1H3K5-4]
Cho hình chóp tứ giác đều $S.ABCD$ có đáy $ABCD$ là hình vuông canh bằng $a$, $SA=a\sqrt{2}$.
\begin{enumerate}
	\item Tính góc giữa đường thẳng $SD$ và mặt phẳng $(ABCD)$.
	\item Tính khoảng cách giữa hai đường thẳng $BD$ và $SC$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzDefPoints{0/0/A,6/0/D,-2/-1.5/B}
\coordinate (C) at ($(B)+(D)-(A)$);
\coordinate (O) at ($(A)!0.5!(C)$);
\coordinate (S) at ($(O)+(0,6)$);
\coordinate (H) at ($(S)!0.68!(C)$);
\tkzDrawSegments(S,B S,C S,D B,C C,D)
\tkzDrawSegments[dashed](S,A A,B A,D S,O A,C B,D O,H)
\tkzDrawPoints(S,A,B,C,D,O,H)
\tkzLabelPoints[below left](B)
\tkzLabelPoints[below right](C)
\tkzLabelPoints[above right](D,H)
\tkzLabelPoints[above](S)
\tkzLabelPoints[below](O)
\tkzLabelPoints[above left](A)
\tkzMarkRightAngle(O,H,C)
\tkzMarkRightAngle(D,O,H)
\end{tikzpicture}
\end{center}
\begin{enumerate}
			\item Gọi $O$ là tâm của $ABCD$, do $S.ABCD$ là hình chóp tứ giác đều nên $SO\perp(ABCD)$.\\
			Vậy $OD$ là hình chiếu của $SD$ lên  $(ABCD)$.\\
			$\Rightarrow (SD,(ABCD))=(SD,OD)=\widehat{SDO}$ (vì $\triangle SDO$ vuông tại $O$).\\
			Có $ABCD$ là hình vuông cạnh $a$ nên $BD=a\sqrt{2}$.\\
			Mặt khác $SA=SB=SC=SD=a\sqrt{2}$ nên $\triangle SBD$ đều.\\
			Vậy $(SD,(ABCD))=\widehat{SDO}=60^\circ$.
			\item Gọi $H$ là hình chiếu vuông góc của $O$ lên $SC$.\\
			Có $\heva{&BD\perp AC\\&BD\perp SO}\Rightarrow BD\perp(SAC)\Rightarrow BD\perp OH$ tại $O$.\\
			Vậy $OH$ là đoạn vuông góc chung của hai đường thẳng $BD$ và $SC$ nên $d(BD,SC)=OH$.\\
			Có $OC=\dfrac{1}{2}AC=\dfrac{a\sqrt{2}}{2}$ và $SO=\dfrac{SA\sqrt{3}}{2}=\dfrac{a\sqrt{6}}{2}$ (do $\triangle SAC$ đều).\\
			Tam giác vuông $SOC$ có $\dfrac{1}{OH^2}=\dfrac{1}{SO^2}+\dfrac{1}{OC^2}=\dfrac{2}{3a^2}+\dfrac{2}{a^2}=\dfrac{8}{3a^2}\Rightarrow OH=\dfrac{a\sqrt{6}}{4}$.\\
			Vậy khoảng cách giữa hai đường thẳng $BD$ và $SC$ là $d=\dfrac{a\sqrt{6}}{4}$.
	\end{enumerate}}
\end{bt}
