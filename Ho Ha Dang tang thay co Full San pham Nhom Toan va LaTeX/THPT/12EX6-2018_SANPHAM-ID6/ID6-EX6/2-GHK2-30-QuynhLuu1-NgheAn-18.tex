\begin{ex}%[Đề thi thử THPTQG lần 1, Quỳnh Lưu 1, Nghệ An 2018]%[Sỹ Trường, Dự án EX-6]%[2D3K1-1]
	Hàm số nào dưới đây \textbf{không} là nguyên hàm của hàm số $y=\dfrac{x\left(2+x\right)}{\left(x+1\right)^2}$?
	\choice
	{\True $y=\dfrac{x^2+x-1}{x+1}$}
	{$y=\dfrac{x^2-x-1}{x+1}$}
	{$y=\dfrac{x^2+x+1}{x+1}$}
	{$y=\dfrac{x^2}{x+1}$}
	\loigiai{
		Ta có biến đổi:  $\dfrac{x\left(2+x\right)}{\left(x+1\right)^2}=\dfrac{x^2+2x}{\left(x+1\right)^2}=1-\dfrac{1}{\left(x+1\right)^2}$.\\
		Suy ra, $\displaystyle \int \dfrac{x\left(2+x\right)}{\left(x+1\right)^2}\mathrm{\,d}x=\int \left(1-\dfrac{1}{\left(x+1\right)^2}\right)\mathrm{\,d}x=x+\dfrac{1}{x+1}+C=\dfrac{x^2+\left(C+1\right)x+C+1}{x+1}$.
		\begin{itemize}
			\item Với $C=-2$, ta được $y=\dfrac{x^2-x-1}{x+1}$;
			\item Với $C=0$, ta được $y=\dfrac{x^2+x+1}{x+1}$;
			\item Với $C=-1$, ta được $y=\dfrac{x^2}{x+1}$.
		\end{itemize}
		Vậy hàm số không phải là nguyên hàm của hàm số đã cho là $y=\dfrac{x^2+x-1}{x+1}$.
	}
\end{ex}
