\begin{bt}%[KT1T, Chu Văn An - Đăk Nông, 2018-2019]%[Đặng Tân Hoài, EX-DT-10&11]%[0D1Y1-1] 
	Trong các phát biểu sau, phát biểu nào là mệnh đề?
	\begin{enumerate}
		\item  Pari là thủ đô của nước Pháp.				
		\item  Không được đi lối này!
		\item  $5$ là số nguyên tố.
		\item  $4$ là số lớn hơn $\pi$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item  Câu đã cho là khẳng định đúng nên nó là một mệnh đề.				
		\item  Câu đã cho là một câu cảm thán không biết tính đúng hay sai nên nó không phải là mệnh đề. 
		\item  Vì $5$ là số nguyên chỉ chia hết cho $1$ và chính nó nên câu đã cho là khẳng định đúng. Do đó nó là một mệnh đề.
		\item  Ta có $\pi \approx 3{,}14 \ldots$ nên câu đã cho là khẳng định đúng.  Do đó nó là một mệnh đề.		
	\end{enumerate}
	}
\end{bt}

\begin{bt}%[KT1T, Chu Văn An - Đăk Nông, 2018-2019]%[Đặng Tân Hoài, EX-DT-10&11]%[0D1B2-2] 
	Cho tập $A= \left\{ a;b;c \right\} $. Tìm tất cả các tập con của $A$.
	\loigiai{
		Theo tính chất về tập hợp con, ta có $\varnothing \subset A,~A \subset A$. Ngoài ra, còn có các tập hợp con sau:\\
		$\bullet$ Tập hợp con có $1$ phần tử gồm có $\left\{ a \right\}$, $\left\{ b \right\}$, $\left\{ c \right\}$.\\
		$\bullet$ Tập hợp con có $2$ phần tử gồm có $\left\{ a;b \right\}$, $\left\{ b;c \right\}$, $\left\{ a;c \right\}$.\\
	\underline{Nhận xét}: 	Nếu một tập hợp có $n$ phần tử (với $n \in \mathbb{N}^*$) thì số tập con của nó là $2^n$.
	}
\end{bt}

\begin{bt}%[KT1T, Chu Văn An - Đăk Nông, 2018-2019]%[Đặng Tân Hoài, EX-DT-10&11]%[0D1K1-5]
	Các mệnh đề sau đúng hay sai? Vì sao? Viết mệnh đề phủ định của chúng.
	\begin{enumerate}
		\item \lq\lq $\exists x \in \mathbb{N},~ x~\text{chia hết cho}~x+1$\rq\rq. 
		\item  \lq\lq $\forall x \in \mathbb{Z},~ x \geq -1 \Rightarrow x^2 \geq 1$\rq\rq .
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Mệnh đề đã cho đúng vì tồn tại $x=0 \in \mathbb{N}$ thì $0$ chia hết cho $1$ là điều đúng. Nó có mệnh đề phủ định là \lq\lq $\forall x \in \mathbb{N},~ x~\text{không chia hết cho}~x+1$\rq\rq. 
		\item  Mệnh đề đã cho sai vì tồn tại $x=0 \in \mathbb{Z}$ thì \lq\lq $0 \geq -1 \Rightarrow 0^2 \geq 1$\rq\rq \, là điều sai.\\
		Áp dụng kiến thức $\overline{A \Rightarrow B} = A \wedge \overline{B}$ với $A \wedge \overline{B}$ là giao của hai mệnh đề $A$ và $\overline{B}$ ta có mệnh đề phủ định của mệnh đề đã cho là \lq\lq $\exists x \in \mathbb{Z},~ x \geq -1 ~\text{và}~ x^2 \leq 1$\rq\rq .
	\end{enumerate}
	}
\end{bt}

\begin{bt}%[KT1T, Chu Văn An - Đăk Nông, 2018-2019]%[Đặng Tân Hoài, EX-DT-10&11]%[0D1B3-2]
	Cho $A= \left\{ 1;2;3;4;5 \right\} $, $B=\left\{x \mid (x^2-2x-3)(x^2-2x-8)=0\right\}$. Tìm  $A \cap B$,  $A \cup B$, $A \setminus B$.
	\loigiai{
	Ta có $(x^2-2x-3)(x^2-2x-8)=0 \Leftrightarrow \hoac{& x^2-2x-3=0\\ & x^2-2x-8=0} \Leftrightarrow \hoac{& x=-1\\ & x=3\\ & x=-2 \\ & x=4}$. Do đó $B=\left\{ -2;-1;3;4 \right\} $.\\
	Vậy $A \cap B = \left\{ 3;4 \right\}$,  $A \cup B =\left\{ -2;-1;1;2;3;4;5 \right\}$, $A \setminus B =\left\{ 1;2;5 \right\}$.
	}
\end{bt}

\begin{bt}%[KT1T, Chu Văn An - Đăk Nông, 2018-2019]%[Đặng Tân Hoài, EX-DT-10&11]%[0D1G4-1]
	Cho các tập hợp $A=[m;m+2]$ và $B=[-1;2]$, trong đó $m$ là số thực tùy ý.
	\begin{enumerate}
		\item Tìm tất cả các giá trị của $m$ để $A \subset B$.
		\item Tìm tất cả các giá trị của $m$ để $A \cap B \neq \varnothing$.
	\end{enumerate}
	\loigiai{Ta thấy $m+2>m,~\forall m \in \mathbb{R}$ nên tập hợp $A$ đã xác định.
	\immini{Tập hợp $A=[m;m+2]$ có biểu diễn là}{ 
		\begin{tikzpicture}[scale=1, font=\footnotesize, line join = round, line cap = round, >=stealth]
		\draw[thick,->] (-2,0)node[below=6pt]{$-\infty$}--(0,0) node[scale=1.5]{\bf [} node[below=6pt]{$m$}--(1.5,0) node[below=6pt]{$A$}--(3,0) node[scale=1.5]{\bf ]}node[below=6pt]{$m+2$}--(5,0)node[below=6pt]{$+\infty$};
		\foreach \i in {1,...,10} 
		\draw ($(0,0)-(.2*\i,0)$) node[scale=.6]{/};
		\foreach \i in {1,...,10} 
		\draw ($(3,0)+(.2*\i,0)$) node[scale=.6]{/};
		\end{tikzpicture}}
	\immini{Tập hợp $B=[-1;2]$ có biểu diễn là}{
		\begin{tikzpicture}[scale=1, font=\footnotesize, line join = round, line cap = round, >=stealth]
		% Vẽ (1,4]
		\draw[thick,->] (-2,0)node[below=6pt]{$-\infty$}--(1,0) node[scale=1.5]{\bf [ } node[below=6pt]{$-1$}--(2.5,0) node[below=6pt]{$B$}--(4,0) node[scale=1.5]{\bf ]}node[below=6pt]{$2$}--(5,0)node[below=6pt]{$+\infty$};
		\foreach \i in {1,...,15} 
		\draw ($(1,0)-(.2*\i,0)$) node[scale=.6,rotate=60]{/};
		\foreach \i in {1,...,5} 
		\draw ($(4,0)+(.2*\i,0)$) node[scale=.6,rotate=60]{/};
		\end{tikzpicture}}
	\begin{enumerate}
		\item Yêu cầu bài toán $\Leftrightarrow \heva{& m \geq -1\\ & m+2 \leq 2} \Leftrightarrow -1 \leq m \leq 0$.\\
		Vậy $-1 \leq m \leq 0$ thì thỏa mãn yêu cầu bài toán.
		\item \underline{Bước 1}: Ta tìm $m$ để thỏa mãn ngược yêu cầu bài toán. Tức là\\
		$A \cap B = \varnothing \Leftrightarrow \hoac{& m>2\\ & m+2<-1} \Leftrightarrow \hoac{& m>2\\ & m<-3}$.\\
		\underline{Bước 2}: Phủ định của kết quả trên ta được $A \cap B \neq \varnothing \Leftrightarrow -3 \leq m \leq 2$.\\
		Vậy $-3 \leq m \leq 2$ thì thỏa mãn yêu cầu bài toán.
	\end{enumerate}
	}
\end{bt}

\begin{bt}%[KT1T, Chu Văn An - Đăk Nông, 2018-2019]%[Đặng Tân Hoài, EX-DT-10&11]%[0D1K1-4]
	Chứng minh bằng phản chứng. Cho $n \in \mathbb{N}$, nếu $5n+5$ là số lẻ thì $n$ là số chẵn.
	\loigiai{
	Giả sử $n$ là số lẻ. Suy ra	$n=2k+1,~\forall k \in \mathbb{N}$.\\
	Khi đó $5n+5=5(2k+1)+5=10k+10=10(k+1)$ (trái giả thiết $5n+5$ là số lẻ).\\
	Vậy phải có $n$ là số chẵn.
	}
\end{bt}
