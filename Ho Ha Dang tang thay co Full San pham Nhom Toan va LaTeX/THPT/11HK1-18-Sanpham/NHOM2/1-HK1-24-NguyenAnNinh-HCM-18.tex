\begin{bt}%[HK1-THPT Nguyễn An Ninh - TP HCM - Năm 2017-2018]%[Lê Thanh Nin]%[1D1Y2-1]
	Giải phương trình sau: $\cos \left(2x+\dfrac{\pi}{3}\right)-\cos\left(x-\dfrac{\pi}{6}\right)=0.$
	\loigiai{
	\begin{eqnarray*}
		&&\cos \left(2x+\dfrac{\pi}{3}\right)-\cos\left(x-\dfrac{\pi}{6}\right)=0\\
		&\Leftrightarrow &\cos \left(2x+\dfrac{\pi}{3}\right)=\cos\left(x-\dfrac{\pi}{6}\right)\\
		&\Leftrightarrow &\hoac{&2x+\dfrac{\pi}{3}=x-\dfrac{\pi}{6}+k2\pi\\&2x+\dfrac{\pi}{3}=-x+\dfrac{\pi}{6}+k2\pi}\\
		&\Leftrightarrow &\hoac{&x=-\dfrac{\pi}{2}+k2\pi\\&x=-\dfrac{\pi}{18}+k\dfrac{2\pi}{3}} \quad (k\in \mathbb{Z}).
	\end{eqnarray*}
	Vậy phương trình có các nghiệm là $x=-\dfrac{\pi}{2}+k2\pi, x=-\dfrac{\pi}{18}+k\dfrac{2\pi}{3} \quad (k\in \mathbb{Z}).$	
	} 
\end{bt}

\begin{bt}%[HK1-THPT Nguyễn An Ninh - TP HCM - Năm 2017-2018]%[Lê Thanh Nin]%[1D1B3-1]
	Giải phương trình sau: $3\cos^22x-14\sin x\cos x=3.$	
	\loigiai{
		\begin{eqnarray*}
			&&3\cos^22x-14\sin x\cos x=3\\
			&\Leftrightarrow &3(1-\sin^22x)-7\sin 2x-3=0\\
			&\Leftrightarrow &-3\sin^22x-7\sin 2x=0\\
			&\Leftrightarrow &\hoac{&\sin 2x=0\\&\sin 2x=-\dfrac{7}{3}\quad (\text{loại})}\\
			&\Leftrightarrow &\sin2x=0\Leftrightarrow 2x=k\pi\Leftrightarrow x=k\dfrac{\pi}{2}\quad (k\in \mathbb{Z})
		\end{eqnarray*}
		Vậy phương trình có các nghiệm là $x=k\dfrac{\pi}{2}\quad (k\in \mathbb{Z}).$		
	} 
\end{bt}

\begin{bt}%[HK1-THPT Nguyễn An Ninh - TP HCM - Năm 2017-2018]%[Lê Thanh Nin]%[1D1K3-5]
	Giải phương trình sau: $(2\sin x+1)(2\sin x-4+3\cos 4x)=3-4\cos^2x.$	
	\loigiai{
	\begin{eqnarray*}
		&&(2\sin x+1)(2\sin x-4+3\cos 4x)=3-4\cos^2x\\
		&\Leftrightarrow & 4\sin^2x+6\sin x\cos4x-6\sin x-4+3\cos4x=-1+4\sin^2x\\
		&\Leftrightarrow & 6\sin x\cos4x-6\sin x+3\cos4x-3=0\\
		&\Leftrightarrow & 6\sin x(\cos4x-1)+3(\cos4x-1)=0\\
		&\Leftrightarrow & (6\sin x+3)(\cos4x-1)=0\\
		&\Leftrightarrow & \hoac{&6\sin x+3=0\\&\cos4x-1=0}\Leftrightarrow \hoac{&\sin x=-\dfrac{1}{2}\\&\cos4x=1}\\
		&\Leftrightarrow & \hoac{&x=-\dfrac{\pi}{6}+k2\pi\\&x=\dfrac{7\pi}{6}+k2\pi\\&x=k\dfrac{\pi}{2}}\quad (k\in \mathbb{Z}).\\
	\end{eqnarray*}
Vậy phương trình có các nghiệm là $x=-\dfrac{\pi}{6}+k2\pi, x=\dfrac{7\pi}{6}+k2\pi, x=k\dfrac{\pi}{2}\quad (k\in \mathbb{Z}).$	
		
	} 
\end{bt}

\begin{bt}%[HK1-THPT Nguyễn An Ninh - TP HCM - Năm 2017-2018]%[Lê Thanh Nin]%[1D2B2-1]
	Có bao nhiêu số tự nhiên gồm $4$ chữ số khác nhau và chia hết cho $10$ sao cho trong số đó phải có mặt chữ số $2$?
	\loigiai{
	Số cần tìm chia hết cho $10$ nên có dạng $\overline{abc0}.(a\ne 0)$\\
	Chọn $1$ trong $3$ vị trí $a,b,c$ để đặt số $2$ có $3$ cách. \\
	Chọn $2$ số khác nhau ( khác $0$ và $2$) đặt vào $2$ vị trí còn lại có $\mathrm{A}_8^2$ cách.\\
	Vậy có $3\mathrm{A}_8^2=168$ (số).
		
	} 
\end{bt}

\begin{bt}%[HK1-THPT Nguyễn An Ninh - TP HCM - Năm 2017-2018]%[Lê Thanh Nin]%[1D2B3-2]
	Tìm số hạng không chứa $x$ trong khai triển của biểu thức $\left(3x^3+\dfrac{2}{x}\right)^{12}$ với $x\neq 0.$
	\loigiai{
		$$\left(3x^3+\dfrac{2}{x}\right)^{12}=\displaystyle\sum_{k=0}^{12}\mathrm{C}_{12}^k\left(3x^3\right)^{12-k}\left(\dfrac{2}{x}\right)^k=\displaystyle\sum_{k=0}^{12}\mathrm{C}_{12}^k3^{12-k}2^kx^{36-4k}.$$
		Số hạng không chứa $x$ ứng với $36-4k=0\Leftrightarrow k=9$.\\ Từ đó suy ra số hạng phải tìm là $\mathrm{C}_{12}^93^32^9=3041280.$
	}

\end{bt}

\begin{bt}%[HK1-THPT Nguyễn An Ninh - TP HCM - Năm 2017-2018]%[Lê Thanh Nin]%[1D2B5-2]
	Có hai kệ sách, kệ thứ nhất có $4$ cuốn sách Toán và $5$ cuốn sách Lý, kệ thứ hai có $6$ cuốn sách Toán và $4$ cuốn sách Lý. Lấy ngẫu nhiên từ mỗi kệ một cuốn sách. Tính xác suất sao cho hai cuốn sách được lấy ra cùng môn.
	\loigiai{
		Số cách lấy ngẫu nhiên từ mỗi kệ $1$ cuốn sách là $n(\Omega)=9\cdot10=90.$\\
		Số cách lấy hai cuốn sách cùng môn là $n(A)=4\cdot6+5\cdot4=44$.\\
		Xác suất cần tìm là $\mathrm{P}(A)=\dfrac{44}{90}=\dfrac{22}{45}.$
	} 
\end{bt}
