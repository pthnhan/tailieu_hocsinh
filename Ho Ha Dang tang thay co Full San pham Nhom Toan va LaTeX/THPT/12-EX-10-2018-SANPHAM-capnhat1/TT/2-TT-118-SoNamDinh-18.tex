	{$2$}

		& &\log_3(x^3+1)=\log_3|2x-1|+\log_3(x+1)\\

		&\Leftrightarrow & x^2+x=0\\
		& \Leftrightarrow & \hoac{&x=0 \quad \text{(thỏa mãn)}\\& x=-1 \quad \text{(loại).}}
		\end{eqnarray*}

		\end{itemize}
