}
\end{ex}

\begin{ex}%[TT L4, chuyen Quang Trung, Bình Phước, 2018]%[Lê Mạnh Thắng, 12EX-10]%[2D4G3-4]
	Trong mặt phẳng phức, xét hình bình hành tạo bởi các điểm $0,z,\dfrac{1}{z}$ và $z+\dfrac{1}{z}$. Biết $z$ có phần thực dương và diện tích hình bình hành bằng $\dfrac{35}{37}$. Tìm giá trị nhỏ nhất của $\left|z+\dfrac{1}{z}\right|^2$. 
	\choice
	{$\dfrac{53}{20}$}
	{$\dfrac{60}{37}$}
	{$\dfrac{22}{9}$}
	{\True $\dfrac{50}{27}$}
	\loigiai{
	\immini{
		Gọi $O,A,B,C$ lần lượt là các điểm biểu diễn của số phức $0,z,\dfrac{1}{z},z+\dfrac{1}{z}$. Khi đó, diện tích hình bình hành $OACB$ là
		$$S=OA\cdot OB \sin{\alpha}=\left|z\right|\cdot \left|\dfrac{1}{z}\right|\sin{\alpha}=\dfrac{35}{37}\;\Leftrightarrow\; \sin{\alpha}=\dfrac{35}{37}.$$
		Suy ra, $\cos{\alpha}=\pm \sqrt{1-\sin^2{\alpha}}=\pm \dfrac{12}{37}$.
	}{
	\begin{tikzpicture}[>=stealth,x=1cm,y=1cm,scale=1,my angle/.style = {draw,very thin,angle radius=6mm,angle eccentricity=0.7,
		font=\footnotesize}]
	\draw[->,line width = .6pt] (-0.7,0)--(0,0) node[shift={(-0.3,-0.3)}]{$O$}--(3.3,0) node[below]{$x$};
	\draw[->,line width = .6pt] (0,-1.5) --(0,3) node[right]{$y$};
	\tkzDefPoints{0/0/O}
	\tkzDefShiftPoint[O](50:3){A}
	\tkzDefShiftPoint[O](-50:1.5){B}
	\tkzDefPointBy[translation=from O to A](B) \tkzGetPoint{C};
	\tkzDrawPoints(O,A,B,C)
	\tkzDrawSegments(O,A A,C C,B B,O O,C)
	\tkzLabelPoints[above](A)
	\tkzLabelPoints[below](B)
	\tkzLabelPoints[right](C)
	\pic [my angle,"$\alpha$"]{angle = O--A--C};
	\end{tikzpicture}
}
	Áp dụng định lý cô-sin trong tam giác $OAC$, ta có
		\begin{align*}
		\left|z+\dfrac{1}{z}\right|^2=OC^2&=OA^2+OB^2-2OA\cdot OB\cos{\alpha}\\
		&=\left|z\right|^2+\left|\dfrac{1}{z}\right|^2
		-2\left|z\right|\left|\dfrac{1}{z}\right|\cos{\alpha}
		\geqslant 2-2\cos{\alpha}.
		\end{align*}
\begin{itemize}
	\item Nếu $\cos{\alpha}=\dfrac{12}{37}$ thì $\left|z+\dfrac{1}{z}\right|^2\geqslant 2-2\cdot\dfrac{12}{37}=\dfrac{50}{37}.$
	\item Nếu $\cos{\alpha}=-\dfrac{12}{37}$ thì $\left|z+\dfrac{1}{z}\right|^2\geqslant 2+2\cdot\dfrac{12}{37}=\dfrac{98}{37}.$
\end{itemize}
	Suy ra, $\left|z+\dfrac{1}{z}\right|^2$ nhỏ nhất bằng $\dfrac{50}{37}$ khi $\left|z\right|=1$ và $\cos{\alpha}=\dfrac{12}{37}$.
	}
\end{ex}
