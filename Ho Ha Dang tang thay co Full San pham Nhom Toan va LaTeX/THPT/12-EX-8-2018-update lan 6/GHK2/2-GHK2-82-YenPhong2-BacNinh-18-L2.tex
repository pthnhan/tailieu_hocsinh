\begin{ex}%[KSCL L2, Yên Phong 2, Bắc Ninh 2018]%[2D3B3-3]%[Khuất Văn Thanh, 12EX-8]

\choice
{$a=-3$}
{$a=-2$}
{$a=2$}
{\True $a=\pm 2$}
\loigiai{
Hoành độ giao điểm của đồ thị hàm số với trục $Ox$ là nghiệm của $x^2-ax=0\Leftrightarrow\hoac{&x=0\\&x=a}$.\\
\begin{itemize}
\item TH$1$: Với $a>0$ thì thể tích của khối tròn xoay\\
$V=\pi\displaystyle\int\limits_0^a\left(x^2-ax\right)^2\mathrm{\,d}x=\pi\left(\dfrac{x^5}{5}-\dfrac{ax^4}{2}+\dfrac{a^2x^3}{3}\right)\Biggr|_0^a=\dfrac{a^5\pi}{30}$. Suy ra $\dfrac{a^5\pi}{30}=\dfrac{16\pi}{15}\Leftrightarrow a=2$.
\item TH$2$: Với $a<0$ thì thể tích của khối tròn xoay\\
$V=\pi\displaystyle\int\limits_a^0\left(x^2-ax\right)^2\mathrm{\,d}x=\pi\left(\dfrac{x^5}{5}-\dfrac{ax^4}{2}+\dfrac{a^2x^3}{3}\right)\Biggr|_a^0=-\dfrac{a^5\pi}{30}$. Suy ra $-\dfrac{a^5\pi}{30}=\dfrac{16\pi}{15}\Leftrightarrow a=-2$.
\end{itemize}
Vậy $a=\pm 2$.
}
\end{ex}

\begin{ex}%[KSCL L2, Yên Phong 2, Bắc Ninh 2018]%[1D5B2-2]%[Khuất Văn Thanh, 12EX-8]

{\True $y=1080x-13717$}
{$y=24x+91$}
{$y=24x-53$}
{$9x-y-15=0$}

$\Leftrightarrow -x+19=0\Leftrightarrow x=19\Rightarrow y=6803$.\\

Suy ra có một điểm duy nhất có tung độ là nghiệm của phương trình đã cho là $M(19;6803)$.\\
$y'(19)=3\cdot 19^2-3=1080$.\\

\begin{ex}%[KSCL L2, Yên Phong 2, Bắc Ninh 2018]%[1D3B4-6]%[Khuất Văn Thanh, 12EX-8]
Tính tổng của cấp số nhân lùi vô hạn $(u_n)$ biết $u_1=1$ và $u_1, u_3, u_4$ theo thứ tự là ba số hạng liên tiếp trong một cấp số cộng.

{\True $\dfrac{\sqrt{5}-1}{2}$}
{$\dfrac{\sqrt{5}+1}{2}$}
{$2$}
{$\dfrac{1}{\sqrt{5}-1}$}
\loigiai{
Gọi $q~(|q|<1)$ là công bội của cấp số nhân lùi vô hạn $(u_n)$.\\
Khi đó ta có $u_1=1, u_3=u_1q^2=q^2, u_4=u^1q^3=q^3$.\\
Do $u_1,u_3,u_4$ theo thứ tự là ba số hạng liên tiếp trong một cấp số cộng nên $$u_1+u_4=2u_3\Leftrightarrow 1+q^3=2q^2\Leftrightarrow \hoac{&q=1\\&q=\dfrac{1-\sqrt{5}}{2}\\&q=\dfrac{1+\sqrt{5}}{2}}.$$
Vì $|q|<1$ nên $q=\dfrac{1-\sqrt{5}}{2}$.\\
Vậy tổng cấp số nhân lùi vô hạn là $S=\dfrac{u_1}{1-q}=\dfrac{1}{1-\dfrac{1-\sqrt{5}}{2}}=\dfrac{\sqrt{5}-1}{2}$.

\end{ex}

\begin{ex}%[KSCL L2, Yên Phong 2, Bắc Ninh 2018]%[1D5K2-5]%[Khuất Văn Thanh, 12EX-8]
Cho hàm số $y=f(x)$ xác định và có đạo hàm trên $\mathbb{R}$, thỏa mãn $$f(2x)=4f(x)\cos x-2x,\forall x\in\mathbb{R}.$$ Viết phương trình tiếp tuyến của đồ thị hàm số đã cho tại giao điểm của đồ thị với trục tung.
\choice
{$y=2-x$}
{$y=-x$}
{\True $y=x$}
{$y=2x-1$}
\loigiai{
Gọi $M(x_0;y_0)$ là giao điểm của đồ thị hàm số đã cho với trục $Oy$. Suy ra $x_0=0$.\\
Do $f(2x)=4f(x)\cos x-2x,\forall x\in\mathbb{R}$ nên $f(0)=4f(0)\cos 0-2\cdot 0\Rightarrow f(0)=0$.\\
Từ $f(2x)=4f(x)\cos x-2x,\forall x\in\mathbb{R}\Rightarrow 2f'(2x)=4f'(x)\cos x-4f(x)\sin x-2,\forall x\in\mathbb{R}$. Suy ra $2f'(0)=4f'(0)-2\Rightarrow f'(0)=1$.\\
Vậy phương trình tiếp tuyến của đồ thị hàm số đã cho tại $M$ là $y=x$.
}
\end{ex}
