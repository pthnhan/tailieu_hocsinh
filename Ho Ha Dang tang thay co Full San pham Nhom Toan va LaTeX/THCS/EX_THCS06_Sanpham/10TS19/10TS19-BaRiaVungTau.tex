\begin{ex}%[Đề tuyển sinh 10 năm học 2018-2019, Bà Rịa Vũng Tàu]%[Lê Mạnh Thắng-10TS19]%[9D4Y2][9D4B4]
	Cho parabol $(P)\colon y=2x^2$ và đường thẳng $(d)\colon y=2x+m$ ($m$ là tham số).
	\begin{enumerate}
		\item Vẽ parabol $(P)$.
		\item Với những giá trị nào của $m$ thì $(P)$ và $(d)$ chỉ có một điểm chung. Tìm tọa độ điểm chung đó.
	\end{enumerate}
	\loigiai{
\begin{enumerate}
	\item Vẽ đồ thị parabol $(P)$.
	\immini{
		Tập xác định: $\mathscr{D}=\mathbb{R}.$\\
		Bảng giá trị của hàm số $y=2x^2$. 
		\begin{center}
			\begin{tabular}[c]{|c|c|c|c|c|c|}
				\hline
				$x$ & $-2$ & $-1$ & $0$ & $1$ & $2$\\
				\hline
				$y$ & $8$ & $2$ & $0$ & $2$ & $8$\\
				\hline 
			\end{tabular}
		\end{center}
		Vì hệ số $a=2>0$ nên hàm số nghịch biến trên $(-\infty;0)$ và đồng biến trên $(0;+\infty)$.
	}{
		\begin{tikzpicture}[>=stealth,x=1cm,y=1cm,yscale=0.7]
		\draw[->,line width = 0.7pt] (-3,0)--(0,0) node[below right]{$O$}--(3,0) node[below]{$x$};
		\draw[->,line width = .7pt] (0,-.8) --(0,9) node[right]{$y$};
		\foreach \x in {-2,-1,1,2}{\draw[-] (\x,-0.1)--(\x,0.1) node[shift={(0,-0.4)}]{$\x$};}
		\foreach \y in {8,4,2}{\draw[-] (-0.1,\y)--(0.1,\y) node[shift={(0.2,0.2)}]{$\y$};}
		\draw [line width = .7pt, domain=-2.07:2.07, samples=200] plot (\x, {2*(\x)^2});
		\draw[dashed] (-2,0)--(-2,8)--(2,8)--(2,0);
		\draw[dashed] (-1,0)--(-1,2)--(1,2)--(1,0);
		\draw (2.5,8) node[above]{$(P)$};
		\end{tikzpicture}		
	}
	\item Phương trình hoành độ giao điểm của $(P)$ và $(d)$ là
	\[2x^2=2x+m \,\Leftrightarrow\, 2x^2-2x-m=0 \tag{1}\]
	$(P)$ và $(d)$ chỉ có một điểm chung khi và chỉ khi phương trình $(1)$ có nghiệm kép
	\[\Leftrightarrow\; \Delta' =0 \,\Leftrightarrow\, 1+2m=0 \,\Leftrightarrow\, m=-\dfrac{1}{2}.\]
	Khi đó, nghiệm kép của phương trình là $x=-\dfrac{b}{2a}=\dfrac{1}{2}$, suy ra $y=\dfrac{1}{2}$.\\
	Vậy $m=-\dfrac{1}{2}$ thì $(P)$ và $(d)$ có một điểm chung có tọa độ là $\left(\dfrac{1}{2};\dfrac{1}{2}\right)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018-2019, Bà Rịa Vũng Tàu]%[Lê Mạnh Thắng-10TS19]%[9D4K8]
	Hai ô tô khởi hành cùng một lúc từ thành phố $A$ đến thành phố $B$ cách nhau $450\text{\,km}$ với vận tốc không đổi. Vận tốc xe thứ nhất lớn hơn vận tốc xe thứ hai $10\text{\,km/h}$ nên xe thứ nhất đến thành phố $B$ trước xe thứ hai $1{,}5\text{\,giờ}$. Tính vận tốc mỗi xe. 
	\loigiai{
	Gọi $x\text{\,(km/h)}$ là vận tốc của xe thứ nhất ($x>10$). Khi đó vận tốc của xe thứ hai là $x-10\text{\,(km/h)}$.\\
	Thời gian xe thứ nhất và xe thứ hai đi từ thành phố $A$ đến thành phố $B$ cách nhau $450\text{\,km}$ lần lượt là $\dfrac{450}{x}\text{\;giờ}$ và $\dfrac{450}{x-10}\text{\;giờ}$.\\
	Theo đề bài ta có phương trình:
	\begin{eqnarray*}
		\dfrac{450}{x-10}-\dfrac{450}{x}=1{,}5 &\Leftrightarrow&
		300x-300(x-10)=x(x-10)\\
		&\Leftrightarrow& x^2-10x-3000=0 
		\;\Leftrightarrow\; \left[\begin{aligned}& x=-50\text{\;(loại)} \\ & x=60. \end{aligned}\right.
	\end{eqnarray*}
Vậy vận tốc của xe thứ nhất là $60\text{\,km/h}$; vận tốc xe thứ hai là $50\text{\,km/h}$.
	}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018-2019, Bà Rịa Vũng Tàu]%[Lê Mạnh Thắng-10TS19]%[9D4K6]
	Cho phương trình $x^2-mx-1=0$ ($m$ là tham số). Tìm tất cả giá trị của $m$ để phương trình có hai nghiệm phân biệt $x_1,x_2$ thỏa $x_1<x_2$ và $\left|x_1\right|-\left|x_2\right|=6$.
	\loigiai{
	Phương trình có hai nghiệm $x_1$, $x_2$ khi và chỉ khi
			\begin{center}
				$\Delta > 0 \,\Leftrightarrow\, m^2+4>0, \forall m\in \mathbb{R}$.
			\end{center}
	Hệ thức Vi-ét: $\left\lbrace \begin{aligned}& x_1+x_2=m \\ & x_1x_2=-1 \end{aligned}\right.$.
			Vì $x_1<x_2$ và $x_1x_2=-1<0$ nên $x_1<0<x_2$. Do đó
			\[\left|x_1\right|-\left|x_2\right|=6 \,\Leftrightarrow\, -x_1-x_2=6 \,\Leftrightarrow\, x_1+x_2=-6
			\,\Leftrightarrow\, m=-6.\]
			Vậy $m=-6$ là giá trị cần tìm.
	}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018-2019, Bà Rịa Vũng Tàu]%[Lê Mạnh Thắng-10TS19]%[9H3G7]
	Cho đường tròn $(O;R)$ và điểm $A$ ở bên ngoài đường tròn đó. Kẻ cát tuyến $AMN$ không đi qua $O$ ($M$ nằm giữa $A$ và $N$). Kẻ hai tiếp tuyến $AB,AC$ với $(O;R)$ ($B,C$ là hai tiếp điểm và $C$ thuộc cung nhỏ $MN$). Đường thẳng $BC$ cắt $MN$ và $AO$ lần lượt tại $E,F$. Gọi $I$ là trung điểm của $MN$.
	\begin{enumerate}
		\item Chứng minh tứ giác $ABOC$ nội tiếp được đường tròn.
		\item Chứng minh $EB\cdot EC=EM\cdot EN$ và $IA$ là tia phân giác của $\widehat{BIC}$.
		\item Tia $MF$ cắt $(O;R)$ tại điểm thứ hai là $D$. Chứng minh $\triangle{AMF} \backsim \triangle{AON}$ và $BC \parallel DN$.
		\item Giả sử $AO=2R$. Tính diện tích tam giác $ABC$ theo $R$.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=1]
			\tkzDefPoints{0/0/O, 5/4/A, 3/0/R, -5/0/Q}
			\clip (-4,-3.5) rectangle (6,4.5);
			\tkzDrawCircle(O,R)
			\tkzDefMidPoint(O,A)\tkzGetPoint{P}
			\tkzInterCC(P,A)(O,R)\tkzGetFirstPoint{C}
			\tkzInterCC(P,A)(O,R)\tkzGetSecondPoint{B}
			\tkzInterLC(A,Q)(O,R)\tkzGetFirstPoint{N}
			\tkzInterLC(A,Q)(O,R)\tkzGetSecondPoint{M}
			\tkzInterLL(B,C)(A,Q)\tkzGetPoint{E}
			\tkzInterLL(B,C)(A,O)\tkzGetPoint{F}
			\tkzDefMidPoint(M,N)\tkzGetPoint{I}
			\tkzInterLC(M,F)(O,R)\tkzGetSecondPoint{D}
			\tkzDrawPoints[fill=black](O,A,B,C,M,N,E,F,I,D)
			\tkzDrawSegments(A,B A,C A,O B,C O,B O,C A,N O,I M,C B,N M,D 
			O,N I,B I,C D,N)
			\tkzLabelPoints[right](A,C)
			\tkzLabelPoints[above](B,M,E)
			\tkzLabelPoints[left](N,F)
			\tkzLabelPoints[shift={(-0.45,0.5)}](I)
			\tkzLabelPoints[below](O,D)
			\tkzMarkRightAngles(A,B,O A,C,O O,I,N);
			\end{tikzpicture}
		\end{center}
	\begin{enumerate}
		\item Vì $AB,AC$ là tiếp tuyến của đường tròn $(O;R)$ tại $B,C$ nên $\widehat{ABO}=\widehat{ACO}=90^\circ$. Do đó, tứ giác $ABOC$ nội tiếp.
		\item $\bullet\;$ Chứng minh $EB\cdot EC=EM\cdot EN$.\\
			Xét hai tam giác $EBN$ và $EMC$ có:\\ 
			$\widehat{BEN}=\widehat{MEC}$ (đối đỉnh)\\
			$\widehat{BNE}=\widehat{MCE}$ (cùng chắn cung $\wideparen{BM}$)\\
			Suy ra, $\triangle{EBN} \backsim \triangle{EMC}$, suy ra $\dfrac{EB}{EM}=\dfrac{EN}{EC}$ hay $EB\cdot EC=EM\cdot EN$.\\
			$\bullet\;$ Chứng minh $IA$ là tia phân giác của $\widehat{BIC}$.\\
			Ta có $OI \perp MN$ (định lý đường kính và dây cung)\\
			Suy ra, $I$ thuộc đường tròn đường kính $AO$ hay $5$ điểm $O,I,B,A,C$ cùng thuộc một đường tròn.\\
			Ta có $\widehat{AIB}=\widehat{ACB}$ (cùng chắn cung $\wideparen{AB}$)\\
			và $\widehat{AIC}=\widehat{ABC}$ (cùng chắn cung $\wideparen{AC}$)\\
			mà $\widehat{ACB}=\widehat{ABC}$ (tính chất hai tiếp tuyến cắt nhau)\\
			Suy ra, $\widehat{AIB}=\widehat{AIC}$ hay $IA$ tia phân giác của $\widehat{BIC}$.
		\item $\bullet\;$ Chứng minh $\triangle{AMF} \backsim \triangle{AON}$.\\
		Ta có $\triangle{ABN} \backsim \triangle{AMB}$ (do $\widehat{A}$ chung và $\widehat{ANB}=\widehat{ABM}=\dfrac{1}{2}\text{\,sđ\,}\wideparen{BM}$).\\
		Suy ra, $AB^2=AM\cdot AN$. \tagEX{1}
		Mặt khác, tam giác $ABO$ vuông tại $B$ có $BF$ là đường cao nên $AB^2=AF \cdot AO$.	\tagEX{2}
		Từ $(1)$ và $(2)$, suy ra $AM\cdot AN=AF \cdot AO$ hay $\dfrac{AM}{AO}=\dfrac{AF}{AN} \;\Rightarrow\; \triangle{AMF} \backsim \triangle{AON}$.\\
		$\bullet\;$ Chứng minh $BC \parallel DN$.\\
		Ta có $\widehat{AFM}=\widehat{ANO}$ (do $\triangle{AMF} \backsim \triangle{AON}$).
		Suy ra,
		\[90^\circ - \widehat{AFM}= 90^\circ - \widehat{ANO} \;\Leftrightarrow\; \widehat{MFE}=\widehat{NOI} \tag{3}\]
		Mặt khác,
		\[\widehat{MDN}=\dfrac{1}{2}\widehat{MON}=\widehat{NOI} \tag{4}\]
		Từ $(3)$ và $(4)$ $\,\Rightarrow \widehat{MFE}=\widehat{MDN}$.\\
		Mà hai góc này ở vị trí đồng vị, suy ra $BC \parallel DN$.
		\item Trong tam giác $ACO$ vuông tại $C$, $CF$ là đường cao, ta có
		\[OF\cdot OA = OC^2 \;\Rightarrow\; OF=\dfrac{OC^2}{OA}=\dfrac{R}{2}
		\;\Rightarrow\; AF=OA-OF=\dfrac{3R}{2}.\]
		Ta lại có
		\[CF=\sqrt{OC^2-OF^2}=\dfrac{R\sqrt{3}}{2} \;\Rightarrow\; BC=2CF=R\sqrt{3}.\]
		Do đó, diện tích tam giác $ABC$ là
		\[S_{ABC}=\dfrac{1}{2}AF\cdot BC=\dfrac{1}{2}\cdot \dfrac{3R}{2}\cdot R\sqrt{3}=\dfrac{3\sqrt{3}R^2}{4}.\]
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018-2019, Bà Rịa Vũng Tàu]%[Lê Mạnh Thắng-10TS19]%[9D0B2]
	Giải phương trình $2\sqrt{x}-\sqrt{3x+1}=x-1$.
	\loigiai{
	Điều kiện: $x\geqslant 0$. Phương trình đã cho tương đương với
	\begin{eqnarray*}
	4\sqrt{x}-2\sqrt{3x+1}=2x-2 &\Leftrightarrow& x+4\sqrt{x}+4=3x+1+2\sqrt{3x+1}+1\\
	&\Leftrightarrow& \left(\sqrt{x}+2\right)^2=\left(\sqrt{3x+1}+1\right)^2\\
	&\Leftrightarrow& 
	\left[ \begin{aligned} &\sqrt{x}+2= \sqrt{3x+1}+1\\&\sqrt{x}+2= -\sqrt{3x+1}-1 \text{\;(phương trình vô nghiệm)}\end{aligned}\right.\\
	&\Leftrightarrow& \sqrt{x}+1=\sqrt{3x+1}\\
	&\Leftrightarrow& x+2\sqrt{x}+1=3x+1\\
	&\Leftrightarrow& 2\sqrt{x}\left(\sqrt{x}-1\right)=0 \;\Leftrightarrow\; \left[\begin{aligned} &x=0\\&x=1.\end{aligned} \text{\;(thỏa mãn điều kiện)}\right.
	\end{eqnarray*}
Vậy phương trình đã cho có tập nghiệm $S=\{0;1\}$.
	}
\end{ex}

\begin{ex}%[Đề tuyển sinh 10 năm học 2018-2019, Bà Rịa Vũng Tàu]%[Lê Mạnh Thắng-10TS19]%[9D0B7]
	Cho hai số thực dương $a,b$ thỏa mãn $a+b+3ab=1$. Tìm giá trị lớn nhất của biểu thức
	\[P=\sqrt{1-a^2}+\sqrt{1-b^2}+\dfrac{3ab}{a+b}.\]
	\loigiai{
	Áp dụng bất đẳng thức Cô-si, ta có
		\begin{center}
			$1-a^2+\dfrac{8}{9} \geqslant 2\sqrt{\dfrac{8}{9}(1-a^2)}=\dfrac{4\sqrt{2}}{3}\sqrt{1-a^2}$\\
			$1-b^2+\dfrac{8}{9} \geqslant 2\sqrt{\dfrac{8}{9}(1-b^2)}=\dfrac{4\sqrt{2}}{3}\sqrt{1-b^2}$
		\end{center}
	Suy ra, 
	\[\dfrac{4\sqrt{2}}{3}\left(\sqrt{1-a^2}+\sqrt{1-b^2}\right) \leqslant \dfrac{34}{9}-(a^2+b^2).\]
	Từ giả thiết và $a,b$ dương nên 
	\[\dfrac{1-a-b}{3}=ab \leqslant \dfrac{(a+b)^2}{4} \,\Leftrightarrow\, 3(a+b)^2+4(a+b)-4\geqslant0 \,\Rightarrow a+b\geqslant \dfrac{2}{3}.\]
	Mặt khác,
	\[a^2+b^2\geqslant \dfrac{(a+b)^2}{2}\geqslant\dfrac{1}{2}\cdot \left(\dfrac{2}{3}\right)^2=\dfrac{2}{9}
	\,\Rightarrow\, \dfrac{34}{9}-(a^2+b^2) \leqslant \dfrac{32}{9}.\]
	Do đó, 
	\[\sqrt{1-a^2}+\sqrt{1-b^2} \leqslant \dfrac{3}{4\sqrt{2}} \cdot \dfrac{32}{9} = 4\sqrt{2}{3}.\]
	Ta lại có,
	\[\dfrac{3ab}{a+b}=\dfrac{1-a-b}{a+b}=\dfrac{1}{a+b}-1 \leqslant \dfrac{3}{2}-1=\dfrac{1}{2}.\]
	Suy ra, 
	\[P=\sqrt{1-a^2}+\sqrt{1-b^2}+\dfrac{3ab}{a+b}\leqslant 4\sqrt{2}{3} +\dfrac{1}{2}.\]
	Dấu bằng xảy ra khi $a=b=\dfrac{1}{3}$. Vậy giá trị lớn nhất của $P$ bằng $\dfrac{4\sqrt{2}}{3}+\dfrac{1}{2}$.
	}
\end{ex}
