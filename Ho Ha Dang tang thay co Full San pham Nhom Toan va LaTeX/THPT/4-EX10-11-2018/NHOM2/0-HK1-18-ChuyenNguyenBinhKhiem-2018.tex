\begin{ex}%[HK1 T10, NBK, Vĩnh Long 2017-2018]%[0D3K2]
Tìm $m$ để phương trình $x^2-2x+m-1=0$ có 2 nghiệm trái dấu $x_1$, $x_2$ sao cho $\dfrac{x_1}{x_2}+\dfrac{x_2}{x_1}<1$.
\choice
{Không có $m$ thỏa}
{\True $m<1$}
{$m>\dfrac{7}{3}$}
{$m<3$}
\loigiai
{Phương trình có hai nghiệm trái dấu khi $a\cdot c=m-1<0\Leftrightarrow m<1$.\\
	Theo Vi-et ta có $\heva{& x_1+x_2=2 \\
		& x_1\cdot x_2=m-1 \\
	}$.\\
	Ta có $\dfrac{x_1}{x_2}+\dfrac{x_2}{x_1}<1\Leftrightarrow \dfrac{x_1^2+x_2^2}{x_1.x_2}<1\Leftrightarrow \dfrac{{{(x_1+x_2 )}^2}-2x_1\cdot x_2}{x_1\cdot x_2}<1$\\
	$\Leftrightarrow \dfrac{4-2(m-1 )}{m-1}<1\Leftrightarrow -2m+6>m-1$ (vì $m-1<0$).\\
	$\Leftrightarrow m<\dfrac{7}{3}$.\\
So điều kiện suy ra $m<1$.}
\end{ex}

		\begin{enumerate}
			\item

			\\$f(-x )=\dfrac{\left| -x-1 \right|-\left| -x+1 \right|}{{{(-x )}^4}-1}=\dfrac{\left| x+1 \right|-\left| x-1 \right|}{x^4-1}=-f(x )$.
			\\Hàm số $y=f(x )=\dfrac{\left| x-1 \right|-\left| x+1 \right|}{x^4-1}$ là hàm số lẻ.
			
			\item
			Điều kiện $\heva{
				& x\ne 0 \\
				& y\ne 0 \\
			}$.
			\\Đặt $S=x+y,P=xy$. $(P\ne 0 )$
			\\Hệ phương trình $\Leftrightarrow $ $\heva{
				& {{S}^2}-2P=2PS \\

				& S=P=0 & (\text{loại}) \\

			}$

				& x\le 7 \\

					& x=5 \\
					& x=10 \\

			}\Leftrightarrow x=5$.
			\\Vậy tập nghiệm phương trình là $S=\left\{ 5 \right\}$.
			
		\end{enumerate}
	}
