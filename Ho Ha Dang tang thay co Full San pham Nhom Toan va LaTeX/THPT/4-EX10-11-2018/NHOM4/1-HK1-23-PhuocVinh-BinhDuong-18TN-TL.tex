\begin{ex}%[Đề HK1 lớp 11, THPT Phước Vĩnh, Bình Dương, 2017-2018]%[1D1B1]

\choice
{$y=\sin x+1$}
{$y=\sin x+\cos x+1$}
{\True $y=\tan 2x$}
{$y=\cos x$}
\loigiai{
Xét hàm $y=f(x)=\tan 2x$. Tập xác định $\mathscr{D}=\mathbb{R} \backslash \left\{\dfrac{\pi}{4}+\dfrac{k\pi}{2}\right\}$ có tính đối xứng.\\
Ta có $f(-x)=\tan (-2x) = -\tan 2x = -f(x)$.\\
Vậy hàm số $y=\tan x$ là hàm số lẻ.\\
Hàm số $y=\cos x$ là hàm số chẵn.\\
Các hàm số $y=\sin x+1, y=\sin x+\cos x+1$ là các hàm số không chẵn không lẻ.
}
\end{ex}

\end{enumerate}

\end{ex}

\begin{ex}%[Đề HK1 lớp 11, THPT Phước Vĩnh, Bình Dương, 2017-2018]%[1D2K2]

\begin{bt}%[Đề HK1 lớp 11, THPT Phước Vĩnh, Bình Dương, 2017-2018]%[1D1B2]
Giải phương trình $\sin \left(2x-\dfrac{\pi}{6}\right)=\dfrac{1}{2}$.
\loigiai{
$\sin \left(2x-\dfrac{\pi}{6}\right)=\dfrac{1}{2} \Leftrightarrow \hoac{&2x-\dfrac{\pi}{6}=\dfrac{\pi}{6}+k2\pi\\&2x-\dfrac{\pi}{6}=\dfrac{5\pi}{6}+k2\pi} \Leftrightarrow \hoac{&x=\dfrac{\pi}{6}+k\pi\\&x=\dfrac{\pi}{2}+k\pi} (k\in \mathbb{Z})$.
}
\end{bt}

\begin{bt}%[Đề HK1 lớp 11, THPT Phước Vĩnh, Bình Dương, 2017-2018]%[1D1B3]
Giải phương trình $2\cos \dfrac{x}{2}-2\sin \dfrac{x}{2}=\sqrt{2}$.
\loigiai{
\begin{align*}
&2\cos \dfrac{x}{2}-2\sin \dfrac{x}{2}=\sqrt{2}\\
\Leftrightarrow &\dfrac{1}{\sqrt{2}}\cos \dfrac{x}{2}-\dfrac{1}{\sqrt{2}}\sin \dfrac{x}{2} = \dfrac{1}{2}\\
\Leftrightarrow &\sin \dfrac{\pi}{4}\cos \dfrac{x}{2}-\cos \dfrac{\pi}{4}\sin \dfrac{x}{2} = \dfrac{1}{2}\\
\Leftrightarrow &\sin \left(\dfrac{\pi}{4}-\dfrac{x}{2}\right) = \dfrac{1}{2}\\
\Leftrightarrow &\hoac{&\dfrac{\pi}{4}-\dfrac{x}{2}=\dfrac{\pi}{6}+k2\pi\\&\dfrac{\pi}{4}-\dfrac{x}{2}=\dfrac{5\pi}{6}+k2\pi}\\
\Leftrightarrow &\hoac{&x=\dfrac{\pi}{6}-k4\pi\\&x=-\dfrac{7\pi}{6}-k4\pi} (k \in \mathbb{Z}).
\end{align*}
}
\end{bt}
