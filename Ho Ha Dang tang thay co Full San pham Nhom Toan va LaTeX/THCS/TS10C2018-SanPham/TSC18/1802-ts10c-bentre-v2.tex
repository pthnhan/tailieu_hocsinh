\begin{ex}%[Đề thi vào 10, chuyên Bến Tre, năm 2018]%[Hung Tran, Dự án EX-TS10C-18]%[9D1K8]
Cho biểu thức $P=\dfrac{a\sqrt{b}+\sqrt{a}-b\sqrt{a}-\sqrt{b}}{1+\sqrt{ab}}$ với $a,b$ là hai số thực dương.
\begin{enumerate}
\item Rút gọn biểu thức $P:\dfrac{1}{\left(\sqrt{a}+\sqrt{b}\right)(a+b)}$.
\item Tính giá trị của biểu thức $P$ khi $a=2019+2\sqrt{2018}$ và $b=2020+2\sqrt{2019}$.
\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item Ta có $P=\dfrac{\sqrt{ab}\left(\sqrt{a}-\sqrt{b}\right)+\left(\sqrt{a}-\sqrt{b}\right)}{1+\sqrt{ab}}=\dfrac{\left(\sqrt{a}-\sqrt{b}\right)\left(\sqrt{ab}+1\right)}{1+\sqrt{ab}}=\sqrt{a}-\sqrt{b}$.
	\begin{align*}
	\Rightarrow P:\dfrac{1}{\left(\sqrt{a}+\sqrt{b}\right)(a+b)}
	&= P\cdot \left(\sqrt{a}+\sqrt{b}\right)(a+b)\\
	&=\left(\sqrt{a}-\sqrt{b}\right)\left(\sqrt{a}+\sqrt{b}\right)(a+b)\\
	&= (a-b)(a+b)=a^2-b^2.
	\end{align*}
	\item Ta có
	\begin{align*}
	P &=\sqrt{a}-\sqrt{b}=\sqrt{2019+2\sqrt{2018}}-\sqrt{2020+\sqrt{2019}}\\
	&=\sqrt{\left(\sqrt{2018}+1\right)^2}-\sqrt{\left(\sqrt{2019}+1\right)^2}=\sqrt{2018}-\sqrt{2019}.
	\end{align*}
	\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Bến Tre, năm 2018]%[Hung Tran, Dự án EX-TS10C-18]%[9D0K4]
Cho $p$ là số nguyên tố lớn hơn $3$. Chứng minh rằng $p^2-1$ chia hết cho $24$.
	\loigiai{
	 Ta có $p^2-1=(p-1)(p+1)$.\\
	Do $p$ là số nguyên tố lớn hơn $3\Rightarrow p$ là số lẻ.\\
	$\Rightarrow p-1$ và $p+1$ là hai số chẵn liên tiếp nên $p^2-1\,\vdots\,8$.\\
	$p$ là số nguyên tố lớn hơn $3\Rightarrow \hoac{&p=3k+1\\&p=3k+2}$ với $k\ge 1$.
	\begin{itemize}
	\item $p=3k+1\Rightarrow (p-1)(p+1)=3k(p+1)\,\vdots\,3$.
	\item $p=3k+2\Rightarrow (p-1)(p+1)=3(p-1)(k+1)\,\vdots\,3$.
	\end{itemize}
	Từ đó suy ra $p^2-1$ chia hết cho $3;8$.\\
	Mà $(3;8)=1\Rightarrow p^2-1\,\vdots\, 24$.
	}
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Bến Tre, năm 2018]%[Hung Tran, Dự án EX-TS10C-18]%[9D4K6]
Cho phương trình $x^2-2mx-m-4=0$ với $m$ là tham số. Tìm các giá trị của $m$ để phương trình đã cho có hai nghiệm phân biệt $x_1,x_2$ thỏa $\dfrac{1}{x_1^2+x_2^2}$ đạt giá trị lớn nhất.
	\loigiai{
	Phương trình có hai nghiệm phân biệt khi và chỉ khi 
	$$\Delta '=m^2+m+4>0\Leftrightarrow \left(m+\dfrac{1}{2}\right)^2+\dfrac{15}{4}>0,\forall m\in\mathbb{R}. $$\\
	Theo định lí Vi-ét ta có $\heva{&x_1+x_2=2m\\&x_1x_2=-m-4.}$\\
	Ta có 
	\begin{align*}
	\dfrac{1}{x_1^2+x_2^2}&=\dfrac{1}{(x_1+x_2)^2-2x_1x_2}=\dfrac{1}{4m^2+2m+8}\\
	&=\dfrac{1}{\left(2m+\dfrac{1}{2}\right)^2+\dfrac{31}{4}}\le \dfrac{1}{\dfrac{31}{4}}=\dfrac{4}{31}.
	\end{align*}
	Vậy $\max \dfrac{1}{x_1^2+x_2^2}=\dfrac{4}{31}\Leftrightarrow m=-\dfrac{1}{4}$.
	}
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Bến Tre, năm 2018]%[Hung Tran, Dự án EX-TS10C-18]%[9D0K2]
Giải phương trình $\sqrt{x^3+1}=x^2-3x-1$.
	\loigiai{
	Điều kiện $\heva{&x^3+1\ge 0\\&x^2-3x-1\ge 0}\Leftrightarrow \heva{&x\ge -1\\&x^2-3x-1\ge 0.}$\\
	Ta có \begin{eqnarray*}
	\sqrt{x^3+1}=x^2-3x-1 &\Leftrightarrow &\sqrt{(x+1)(x^2-x+1)}=(x^2-x+1)-2(x+1).
	\end{eqnarray*}
	Đặt $a=\sqrt{x+1}; b=\sqrt{x^2-x+1}; (a\ge 0,b>0)$.\\
	Phương trình đã cho trở thành 
	\begin{eqnarray*}
	b^2-2a^2=ab &\Leftrightarrow & 2a^2+ab-b^2=0 \\
	&\Leftrightarrow &(a+b)(2a-b)=0 \\
	&\Leftrightarrow & \hoac{&a+b=0\\&2a=b}\Leftrightarrow 2a=b
	\end{eqnarray*}
	Suy ra 
	\begin{eqnarray*}
	 2\sqrt{x+1}=\sqrt{x^2-x+1} &\Leftrightarrow & x^2-5x-3=0\\
	&\Leftrightarrow & x=\dfrac{5\pm \sqrt{37}}{2} (\text{TMĐK}).
	\end{eqnarray*}
	Vậy phương trình có hai nghiệm $x=\dfrac{5\pm \sqrt{37}}{2}$.
	}
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Bến Tre, năm 2018]%[Hung Tran, Dự án EX-TS10C-18]%[9D0K2]
Giải hệ phương trình $\heva{&x^2+4y^2=2\\&(x-2y)(1-2xy)=4.}$
	\loigiai{
	Ta có 
	\begin{eqnarray*}
	\heva{&x^2+4y^2=2\\&(x-2y)(1-2xy)=4}
	&\Leftrightarrow & \heva{&x^2-4xy+4y^2=2\left(1-2xy\right)\\&(x-2y)(1-2xy)=4}\\
	&\Leftrightarrow &\heva{&(x-2y)^2=2(1-2xy)\\&(x-2y)(1-2xy)=4}.
	\end{eqnarray*}
	Đặt $\heva{&a=x-2y\\&b=1-2xy}$. Hệ phương trình trở thành \\
	\begin{align*}
	\heva{&a^2=2b\\&ab=4}\Leftrightarrow\heva{&a=2\\&b=2}\Rightarrow \heva{&x-2y=2\\&1-2xy=2}\Leftrightarrow \heva{&x=1\\&y=-\dfrac{1}{2}.}
	\end{align*}
	Vậy hệ phương trình có nghiệm duy nhất $(x;y)=\left(1;-\dfrac{1}{2}\right)$.
	}
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Bến Tre, năm 2018]%[Hung Tran, Dự án EX-TS10C-18]%[9D0G5]
Tìm các nghiệm nguyên của phương trình $x^3-xy+2=x+y$.
	\loigiai{
	Biến đổi phương trình thành
	\begin{align*}
	(x+1)(x^2-x-y)=-2\Leftrightarrow \hoac{&\heva{&x+1=-2\\&x^2-x-y=1}\\&\heva{&x+1=2\\&x^2-x-y=-1}\\&\heva{&x+1=1\\&x^2-x-y=-2}\\&\heva{&x+1=-1\\&x^2-x-y=2}}\Leftrightarrow \hoac{&\heva{&x=-3\\&y=11}\\&\heva{&x=1\\&y=1}\\&\heva{&x=0\\&y=2}\\&\heva{&x=-2\\&y=4}}.
	\end{align*}
	Vậy phương trình có các nghiệm nguyên là $(-3;11), (1;1), (0;2), (-2;4)$.
	}
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Bến Tre, năm 2018]%[Hung Tran, Dự án EX-TS10C-18]%[9D0K7]
Cho hai số thực dương $a,b$ thỏa mãn $a+b=1$. Tìm giá trị nhỏ nhất của biểu thức $T=\dfrac{4}{a}+\dfrac{1}{b}$.
	\loigiai{
	Ta có \begin{equation*}
T=\dfrac{4}{a}+\dfrac{1}{b} = \dfrac{4a+4b}{a}+\dfrac{a+b}{b}\\
= 5+\dfrac{4b}{a}+\dfrac{a}{b}\ge 5+4=9.
	\end{equation*}
	Vậy $\min T=9$ khi $a=\dfrac{2}{3}; b=\dfrac{1}{3}$.
	}
\end{ex}
\begin{ex}%[Đề thi vào 10, chuyên Bến Tre, năm 2018]%[Hung Tran, Dự án EX-TS10C-18]%[9H3B7]%[9H0B1]%[9H0K7]
Cho nửa đường tròn $(O;R)$ có đường kính $AB$. Vẽ đường thẳng $d$ là tiếp tuyến của $(O)$ tại $B$. Trên cung $AB$ lấy điểm $M$ tùy ý ($M$ khác $A,B$), tia $AM$ cắt đường thẳng $d$ tại điểm $N$. Gọi $C$ là trung điểm đoạn thẳng $AM$, tia $CO$ cắt đường thẳng $d$ tại điểm $D$.
\begin{enumerate}
\item Chứng minh tứ giác $OBNC$ là tứ giác nội tiếp.
\item Gọi $E$ là hình chiếu của $N$ trên đoạn thẳng $AD$. Chứng minh rằng ba điểm $N,O,E$ thẳng hàng và $\dfrac{NE\cdot AD}{ND}=2R$.
\item Chứng minh rằng $CA\cdot CN=CO\cdot CD$.
\item Xác định vị trí của điểm $M$ để $2AM+AN$ đạt giá trị nhỏ nhất.
\end{enumerate}

	 \begin{tikzpicture}[scale=.8]

	\begin{enumerate}
	\item Do $C$ là trung điểm của $AM$ nên $OC\perp AM$.\\
	Tứ giác $OCBN$ có $\widehat{OCM}=\widehat{OBN}=90^\circ$ nên nội tiếp đường tròn.
	\item Xét trong tam giác $ADN$ có $DC, AB$ lần lượt là các đường cao.\\
	Do đó $O$ là trực tâm tam giác $\Rightarrow N,O,E$ thẳng hàng.\\
	Ta có $S_{AND}=\dfrac{1}{2}NE\cdot AD=\dfrac{1}{2}AB\cdot DN$\\
	$\Rightarrow NE\cdot AD=2R\cdot DN\Rightarrow \dfrac{NE\cdot AD}{DN}=2R$.
	\item Xét hai tam giác vuông $CAO$ và $CDN$ ta có\\
	$\widehat{CDN}=\widehat{MBN}=\dfrac{1}{2}\ \textrm{sđ}\wideparen{MB}$  (góc tạo bởi tiếp tuyến và dây cung).\\
	$\widehat{MAB}=\dfrac{1}{2}\ \textrm{sđ}\wideparen{MB}$ (góc nội tiếp).\\
	Suy ra $\widehat{NAB}=\widehat{CDN}$ (cùng phụ với $\widehat{CND}$).\\
	Từ đó ta có $\triangle CAO\backsim \triangle CDN\Rightarrow \dfrac{CA}{CO}=\dfrac{CD}{CN}\Rightarrow CA\cdot CN=CO\cdot CD$.
	\item Áp dụng bất đẳng thức Cô-si ta có\\
	$2AM+AN\ge 2\sqrt{2AM\cdot AN}=2\sqrt{2AB^2}=4R\sqrt{2}$.\\
	Vậy $\min 2AM+AN=4R\sqrt{2}$ khi $AM=\dfrac{AN}{2}\Leftrightarrow M$ là điểm chính giữa cung $AB$.
	\end{enumerate}
	}
