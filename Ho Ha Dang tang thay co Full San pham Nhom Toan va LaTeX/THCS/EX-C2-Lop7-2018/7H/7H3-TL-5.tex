\subsubsection{Định lý về tính chất các điểm thuộc tia phân giác}

\immini{
	\begin{dl}[Định lý thuận]
		Điểm nằm trên  tia phân giác của một góc thì cách đều hai cạnh của góc đó.
	\end{dl}
	Như vậy, với góc $xOy$ có tia phân giác $Ot$ thì\\
	$M\in Ot\Rightarrow \text{kc}(M\text{,}Ox)=\text{kc}(M\text{,}Oy)\Rightarrow MH=MK$.
}{
	\begin{tikzpicture}[font=\footnotesize, scale=.8]
	\tkzInit[xmin=-1, xmax=3, ymin=-1.5, ymax=1.5]
	\tkzClip
	\tkzDefPoints{0/0/O, 2.5/1/x, 2.5/-1/y, 2/0/M}
	\tkzDefPointBy[projection = onto O--x](M) \tkzGetPoint{H}
	\tkzDefPointBy[projection = onto O--y](M) \tkzGetPoint{K}
	\tkzDrawSegments(O,x O,y M,H M,K)
	\tkzDrawLines[add=0 and .5](O,M)
	\tkzLabelPoints[left](O)
	\tkzLabelPoints[below](K,y)
	\tkzLabelPoints[above](x,H)
	\tkzLabelLine[pos=1.4, above](O,M){$t$}
	\tkzLabelPoints(M)
	\tkzMarkRightAngles(O,H,M O,K,M)
	\tkzMarkAngles[mkpos=.2, size=.6](M,O,H)
	\tkzMarkAngles[mkpos=.2, size=.5](K,O,M)
	\end{tikzpicture}
}
\begin{note}
	$M$ thuộc hình $(H)$ và cách đều hai cạnh của $\widehat{xOy}$ khi $M$ là giao điểm của $(H)$ với tia phân giác của $\widehat{xOy}$.
\end{note}

\subsection{Các dạng toán}

\end{dang}
\begin{vd}%[Phạm An Bình, EX-C2-L7-2018]%[7H3B5]%[7H3B5]

\begin{dang}{Chứng minh một tia là tia phân giác của một góc}
	Sử dụng định lí đảo.
\end{dang}
\begin{vd}%[Phạm An Bình, EX-C2-L7-2018]%[7H3B5]
	Cho $\triangle ABC$ vuông tại $A$. Vẽ $\triangle DBC$ vuông cân tại $D$ ở phía ngoài $\triangle ABC$. Chứng minh rằng $AD$ là tia phân giác của góc $A$.
	\loigiai{
		\immini{
			Kẻ $DH\perp AB$, $DK\perp AC$.\\
			Xét hai tam giác vuông $\triangle DHB$ và $\triangle DKC$ ta có\\
			$DB=DC$ ($\triangle DBC$ cân tại $D$) và $\widehat{HDB}=\widehat{KDC}$ (cùng phụ với $\widehat{CDH}$).\\
			$\Rightarrow \triangle DHB = \triangle DKC$ (ch-gn).\\
			$\Rightarrow DH=DK\Rightarrow AD$ là tia phân giác của góc $A$.
		}{
			\begin{tikzpicture}[font=\footnotesize, scale=.8]
			\tkzInit[xmin=-1, xmax=3, ymin=-1, ymax=2.5]
			\tkzClip
			\tkzDefPoints{0/0/A, 0/1.5/C, 2.5/0/B, 2/2/D, 2/0/H, 0/2/K}
			\tkzDrawPolygon(A,B,D,K)
			\tkzDrawSegments(C,D A,D D,H C,B)
			\tkzLabelPoints[left](A,C,K)
			\tkzLabelPoints[below](H,B)
			\tkzLabelPoints[right](D)
			\tkzMarkRightAngles(A,K,D D,H,A C,D,B B,A,C)
			\tkzDrawPoints[fill=black](A,B,C,D,H,K)
			\tkzMarkSegments[mark=|](D,C D,B)
			\end{tikzpicture}
		}
	}
\end{vd}
\begin{note}
	Vậy có hai cách chứng minh $Ot$ là tia phân giác của $\widehat{xOy}$
	\begin{enumerate}[Cách 1.]
		\item Chứng minh $\widehat{tOx}=\widehat{tOy}$.
		\item Lấy $M$ trên $Ot$, kẻ $MH\perp Ox$, $MK\perp Oy$. Chứng minh $MH=MK$.
	\end{enumerate}
\end{note}

\begin{dang}{Dựng tia phân giác của một góc}
\end{dang}
\begin{vd}%[Phạm An Bình, EX-C2-L7-2018]%[7H3B5]
	Cho góc $xOy$. Lấy các điểm $A$, $B$ thuộc $Ox$ sao cho $OA>OB$. Lấy các điểm $C$, $D$ thuộc $Oy$ sao cho $OC=OA$ và $OD=OB$. Gọi $E$ là giao điểm của $AD$ và $BC$. Chứng minh rằng
	\begin{enumerate}
		\item $AD=BC$.
		\item $\triangle ABE=\triangle CDE$.
		\item $OE$ là tia phân giác của góc $xOy$.
	\end{enumerate}
	\loigiai{
		\immini{
			\begin{enumerate}
				\item Chứng minh $AD=BC$.\\
				Xét $\triangle OAD$ và $\triangle OCB$ ta có\\
				$OA=OC$ (gt) và $\widehat{AOC}$ chung và $OD=OB$ (gt).\\
				$\Rightarrow \triangle OAD = \triangle OCB$ (c-g-c).\\
				$\Rightarrow AD=BC$.
				\item Chứng minh $\triangle ABE=\triangle CDE$.\\
				Xét $\triangle ABE$ và $\triangle CDE$ ta có\\
				$\widehat{A}=\widehat{C}$ và $AB=OA-OB=OC-OD=CD$ và $\widehat{ABE}=180^\circ-\widehat{OBC}=180^\circ-\widehat{ODA}=\widehat{CDE}$.\\
				$\Rightarrow \triangle ABE = \triangle CDE$ (g-c-g).
				\item Chứng minh $OE$ là tia phân giác của góc $xOy$.\\
				Xét $\triangle ABE$ và $\triangle CDE$ ta có\\
				$OA=OC$ (gt) và $\widehat{A}=\widehat{C}$ và $AE=CE$.\\
				$\Rightarrow \triangle AOE = \triangle COE$ (c-g-c).\\
				$\Rightarrow \widehat{AOE}=\widehat{COE}$.\\
				$\Rightarrow OE$ là tia phân giác của $\widehat{xOy}$.
			\end{enumerate}
		}{
			\begin{tikzpicture}[font=\footnotesize]
			\tkzInit[xmin=-1, xmax=3.5, ymin=-1.5, ymax=1.5]
			\tkzClip
			\tkzDefPoints{0/0/O, 3/1/x, 3/-1/y, 2/0/E}
			\tkzDefPointBy[projection = onto O--x](E) \tkzGetPoint{B}
			\tkzDefPointBy[projection = onto O--y](E) \tkzGetPoint{D}
			\tkzInterLL(B,E)(O,y)\tkzGetPoint{C}
			\tkzInterLL(D,E)(O,x)\tkzGetPoint{A}
			\tkzDrawSegments(O,x O,y O,E A,D B,C)
			\tkzLabelPoints[left](O)
			\tkzLabelPoints[below](D,y,C)
			\tkzLabelPoints[above](x,B,A)
			\tkzLabelPoints[right](E)
			\tkzDrawPoints[fill=black](O,A,B,C,D,E)
			\tkzMarkSegments[mark=|](O,B O,D)
			\end{tikzpicture}
		}
	}
\end{vd}
\begin{note}
	Phương pháp dựng tia phân giác của $\widehat{xOy}$
	\begin{enumerate}[Bước 1.]
		\item Chứng minh Lấy các điểm $A$, $B$ thuộc $Ox$. Lấy các điểm $C$, $D$ thuộc $Oy$ sao cho $OC=OA$ và $OD=OB$.
		\item Xác định giao điểm $I$ của $AD$ và $BC$. Khi đó $OI$ chính là tia phân giác của góc $xOy$.
	\end{enumerate}
\end{note}

\begin{dang}{Sử dụng tính chất tia phân giác của một góc để giải toán}
\end{dang}
\begin{vd}%[Phạm An Bình, EX-C2-L7-2018]%[7H3B5]
	Cho $\triangle ABC$ nhọn. Tìm điểm $D$ thuộc trung tuyến $AM$ sao cho $D$ cách đều hai cạnh của góc $B$.
	\loigiai{
		Do $D$ cách đều hai cạnh của góc $B$ nên $BD$ là tia phân giác của góc $B$.\\
		Vậy $B$ là giao điểm của trung tuyến $AM$ và tia phân giác của góc $B$.
	}
\end{vd}
\begin{vd}%[Phạm An Bình, EX-C2-L7-2018]%[7H3B5]
	Cho $\triangle ABC$. Chứng minh rằng
	\begin{enumerate}
		\item Nếu đường trung tuyến $AM$ đồng thời là đường phân giác của góc $A$ thì $\triangle ABC$ cân tại $A$.
		\item Nếu $\triangle ABC$ cân tại $A$ thì đường trung tuyến $AM$ cũng đồng thời là đường phân giác của góc $A$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item
			\immini{
				Kẻ $MH\perp AB$, $MK\perp AC$.\\
				Vì $AM$ là phân giác góc $A$ nên $MH=MK$.\\
				Xét hai tam giác vuông $\triangle HBM$ và $\triangle KCM$ ta có\\
				$MH=MK$ (cmt) và $BM=CM$ ($AM$ là trung tuyến).\\
				$\Rightarrow \triangle HBM = \triangle KCM$ (ch-cgv).\\
				$\Rightarrow \widehat{B}=\widehat{C}$.\\
				$\Rightarrow \triangle ABC$ cân tại $A$.		
			}{
				\begin{tikzpicture}
				\tkzInit[xmin=-1.5, xmax=1.5, ymin=-.5, ymax=3]\tkzClip
				\tkzDefPoints{0/0/M, 0/2.5/A, -1/0/B, 1/0/C}
				\tkzDefPointBy[projection = onto A--B](M) \tkzGetPoint{H}
				\tkzDefPointBy[projection = onto A--C](M) \tkzGetPoint{K}
				\tkzDrawSegments(A,B A,C B,C M,H M,K A,M)
				\tkzMarkAngle[mkpos=.2,size=.5](B,A,M)
				\tkzMarkAngle[mkpos=.2,size=.6](M,A,C)
				\tkzMarkRightAngles(A,H,M A,K,M)
				\tkzMarkSegments[mark=s|](B,M M,C)
				\tkzDrawPoints[fill=black](A,B,C,M,H,K)
				\tkzLabelPoints[below](B,M,C)
				\tkzLabelPoints[above](A)
				\tkzLabelPoints[left](H)
				\tkzLabelPoints[right](K)
				\end{tikzpicture}
			}
			
			\item
			\immini{
				Xét $\triangle AMB$ và $\triangle AMC$ ta có\\
				$AB=AC$ ($\triangle ABC$ cân tại $A$) và $\widehat{B}=\widehat{C}$ (vì $\triangle ABC$ cân tại $A$) và $MB=MC$ (vì $AM$ là trung tuyến).\\
				$\Rightarrow \triangle AMB = \triangle AMC$ (c-g-c).\\
				$\Rightarrow \widehat{BAM}=\widehat{CAM}$.\\
				$\Rightarrow AM$ là đường phân giác của góc $A$.	
			}{
				\begin{tikzpicture}
				\tkzInit[xmin=-1.5, xmax=1.5, ymin=-.5, ymax=3]\tkzClip
				\tkzDefPoints{0/0/M, 0/2.5/A, -1/0/B, 1/0/C}
				\tkzDrawSegments(A,B A,C B,C A,M)
				\tkzMarkSegments[mark=||](B,A A,C)
				\tkzMarkSegments[mark=s|](B,M M,C)
				\tkzDrawPoints[fill=black](A,B,C,M)
				\tkzLabelPoints[below](B,M,C)
				\tkzLabelPoints[above](A)
				\end{tikzpicture}
			}
		\end{enumerate}
	}
\end{vd}
\begin{note}
	Câu a) còn có thể được chứng minh bằng cách lấy điểm $A_1$ trên tia $AM$ sao cho $MA=MA_1$ ($A_1$ khác $A$).
\end{note}

\begin{vd}%[Phạm An Bình, EX-C2-L7-2018]%[7H3B5]
	Cho $\triangle ABC$. Chứng minh rằng hai đường phân giác của hai góc ngoài tại $B$ và $C$ và phân giác trong của góc $A$ cùng đi qua một điểm.
	\loigiai{
		\immini{
			Gọi $M$ là giao điểm của hai đường phân giác của hai góc ngoài tại $B$ và $C$. Ta cần chứng minh $M$ thuộc tia phân giác của góc $A$.\\
			Kẻ $MH\perp AB$, $MD\perp BC$, $MK\perp AC$.\\
			Vì $M$ thuộc tia phân giác góc ngoài tại $B$ nên $MH=MD$.\\
			Vì $M$ thuộc tia phân giác góc ngoài tại $C$ nên $MK=MD$.\\
			Suy ra $MH=MK\Rightarrow M$ thuộc tia phân giác của góc $A$.\\
			Vậy ta có điều phải chứng minh.
		}{
			\begin{tikzpicture}[scale=.7]
			\tkzInit[xmin=-1.5, xmax=4.7, ymin=-3.5, ymax=2.7]\tkzClip
			\tkzDefPoints{0/0/B, 1/2/A, 3/0/C}
			\tkzDefLine[bisector](B,A,C) \tkzGetPoint{a}
			\tkzDefLine[bisector out](A,B,C) \tkzGetPoint{b}
			\tkzInterLL(A,a)(B,b)\tkzGetPoint{M}
			\tkzDefPointBy[projection = onto A--B](M) \tkzGetPoint{H}
			\tkzDefPointBy[projection = onto A--C](M) \tkzGetPoint{K}
			\tkzDefPointBy[projection = onto B--C](M) \tkzGetPoint{D}
			\tkzDrawLines[add=0 and .2](A,H A,K)
			\tkzDrawSegments(B,C B,M C,M M,H M,K M,D)
			\tkzMarkAngle[mkpos=.2,size=.5](H,B,M)
			\tkzMarkAngle[mkpos=.2,size=.6](M,B,C)
			\tkzMarkAngle[size=10,arc=ll](M,C,K)
			\tkzMarkAngle[size=11,arc=ll](B,C,M)
			\tkzMarkRightAngles(M,H,B M,K,C M,D,B)
			\tkzDrawPoints[fill=black](A,B,C,M,H,K,D)
			\tkzLabelPoints[below](M)
			\tkzLabelPoints[above](A,D)
			\tkzLabelPoints[left](B,H)
			\tkzLabelPoints[right](C,K)
			\end{tikzpicture}
		}
	}
\end{vd}
\begin{note}
	Vậy, ví dụ trên dùng cả định lí thuận và đảo.
\end{note}

\subsubsection{Bài tập tự luyện}

\begin{bt}%[Phạm An Bình, EX-C2-L7-2018]%[7H3B5]
	Cho hai đường thẳng $xx'$, $yy'$ cắt nhau tại $O$.
	\begin{enumerate}
		\item Chứng minh hai tia phân giác $Ot$ và $Ot'$ của một cặp góc kề bù tạo thành một góc vuông.
		\item Chứng minh rằng nếu $M$ thuộc đường thẳng $Ot$ hoặc $Ot'$ thì $M$ cách đều hai đường thẳng $xx'$ và $yy'$.
		\item Chứng minh rằng nếu điểm $M$ cách đều hai đường thẳng $xx'$ và $yy'$ thì $M$ thuộc đường thẳng $Ot$ hoặc $Ot'$.
		\item Hãy đưa ra nhận xét về tập hợp các điểm cách đều hai đường thẳng cắt nhau $xx'$ và $yy'$.
	\end{enumerate}
	\loigiai{
		\begin{center}
		\begin{tikzpicture}
			\tkzInit[xmin=-2.5, xmax=2.2, ymin=-1, ymax=3.5]\tkzClip
			\tkzDefPoints{-2/-.5/x, -2/2/y, 2/2/x', 2/-1/y'}
			\tkzInterLL(x,x')(y,y')\tkzGetPoint{O}
			\tkzDefLine[bisector](y,O,x) \tkzGetPoint{t}
			\tkzDefLine[bisector](x',O,y) \tkzGetPoint{t'}
			\tkzDrawLines[add=0 and 0](x,x' y,y')
			\tkzDrawLines[add=0 and -.5](O,t O,t')
			\tkzLabelLine[below](O,t){$t$}
			\tkzLabelLine[right](O,t'){$t'$}
			\tkzDrawPoints[fill=black](O)
			\tkzMarkAngle[mkpos=.2,size=.5](y,O,t)
			\tkzMarkAngle[mkpos=.2,size=.6](t,O,x)
			\tkzMarkAngle[size=10,arc=ll](x',O,t')
			\tkzMarkAngle[size=11,arc=ll](t',O,y)
			\tkzLabelPoints[below](O,x,x',y)
			\tkzLabelPoints[above](y')
			%\tkzLabelPoints[above](A)
		\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Chứng minh hai tia phân giác $Ot$ và $Ot'$ của một cặp góc kề bù tạo thành một góc vuông.\\
			Ta có $\widehat{tOy}=\dfrac{\widehat{xOy}}{2}$ ($Ot$ là phân giác của $\widehat{xOy}$) và $\widehat{yOt'}=\dfrac{\widehat{yOx'}}{2}$ ($Ot'$ là phân giác của $\widehat{yOx'}$).\\
			Suy ra $\widehat{tOy}+\widehat{yOt'}=\dfrac{\widehat{xOy}+\widehat{yOx'}}{2}$.\\
			$\Rightarrow \widehat{tOt'}=90^\circ\Rightarrow Ot\perp Ot'$.\\
			Vậy ta có điều phải chứng minh.
			\item Chứng minh rằng nếu $M$ thuộc đường thẳng $Ot$ hoặc $Ot'$ thì $M$ cách đều hai đường thẳng $xx'$ và $yy'$.
			\begin{itemize}
				\item Vì $M$ thuộc $Ot$ là tia phân giác của $\widehat{xOy}$ nên $M$ cách đều $Ox$ và $Oy$.\\
				Vậy $M$ cách đều $xx'$ và $yy'$.
				\item Vì $M$ thuộc $Ot'$ là tia phân giác của $\widehat{x'Oy}$ nên $M$ cách đều $Ox'$ và $Oy$.\\
				Vậy $M$ cách đều $xx'$ và $yy'$.
			\end{itemize}
			\item Chứng minh rằng nếu điểm $M$ cách đều hai đường thẳng $xx'$ và $yy'$ thì $M$ thuộc đường thẳng $Ot$ hoặc $Ot'$.\\
			Vì $M$ cách đều hai đường thẳng $xx'$ và $yy'$ nên $OM$ là tia phân giác của góc tại đỉnh $O$ có 2 cạnh nằm trên 2 đường thẳng $xx'$ và $yy'$.\\
			Vậy $M$ thuộc đường thẳng $Ot$ hoặc $Ot'$.
			\item Hãy đưa ra nhận xét về tập hợp các điểm cách đều hai đường thẳng cắt nhau $xx'$ và $yy'$.\\
			Tập hợp các điểm cách đều hai đường thẳng cắt nhau $xx'$ và $yy'$ là 2 đường phân giác $Ot$ và $Ot'$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Phạm An Bình, EX-C2-L7-2018]%[7H3B5]
	Cho $\triangle ABC$, đường cao $AH$. Vẽ điểm $D$ sao cho $AB$ là đường trung trục của $HD$. Vẽ điểm $E$ sao cho $AC$ là đường trung trực của $HE$. Giả sử $DE$ cắt $AB$, $AC$ theo thứ tự tại $I$ và $K$. Chứng minh rằng
	\begin{enumerate}
		\item $IB$ là tia phân giác của $\widehat{HID}$.
		\item $KC$ là tia phân giác của $\widehat{HKE}$.
		\item $HA$ là tia phân giác của $\widehat{IHK}$.
		\item $IC$ là tia phân giác của $\widehat{HIK}$, từ đó suy ra $IC$ vuông góc với $AB$.
		\item $KB$ là tia phân giác của $\widehat{HKI}$, từ đó suy ra $KB$ vuông góc với $AC$.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\tkzInit[xmin=-1, xmax=3.2, ymin=-.5, ymax=2.5]\tkzClip
			\tkzDefPoints{0/0/B, 1/2/A, 3/0/C}
			\tkzDefPointBy[projection = onto B--C](A) \tkzGetPoint{H}
			\tkzDefPointBy[reflection=over A--B](H)\tkzGetPoint{D}
			\tkzDefPointBy[reflection=over A--C](H)\tkzGetPoint{E}
			\tkzInterLL(D,E)(A,B)\tkzGetPoint{I}
			\tkzInterLL(D,E)(A,C)\tkzGetPoint{K}
			\tkzDrawPolygon(A,B,C)
			\tkzDrawSegments(A,H D,E H,D H,E H,I H,K)
			\tkzDrawPoints[fill=black](A,B,C,H,D,E,I,K)
			\tkzLabelPoints[below](H,B,C)
			\tkzLabelPoints[above](A,D,I,K,E)
			\tkzMarkRightAngles(C,H,A)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Chứng minh $IB$ là tia phân giác của $\widehat{HID}$.\\
			Vì $\triangle HID$ cân tại $I$ nên $IB$ là tia phân giác của $\widehat{HID}$.
			\item Chứng minh $KC$ là tia phân giác của $\widehat{HKE}$.\\
			Vì $\triangle HKE$ cân tại $K$ nên $KC$ là tia phân giác của $\widehat{HKE}$.
			\item Chứng minh $HA$ là tia phân giác của $\widehat{IHK}$.\\
			Trong $\triangle IHK$ ta có
			\begin{itemize}
				\item Theo kết quả câu a suy ra $IA$ là phân giác ngoài của góc $I$.
				\item Theo kết quả câu b suy ra $KA$ là phân giác ngoài của góc $K$.
			\end{itemize}
			Từ đó, suy ra $HA$ là tia phân giác của $\widehat{IHK}$.
			\item Chứng minh $IC$ là tia phân giác của $\widehat{HIK}$, từ đó suy ra $IC$ vuông góc với $AB$.\\

			\begin{itemize}
				\item Theo kết quả câu b ta có $KC$ là phân giác ngoài của góc $K$.
				\item Theo kết quả câu c ta có $HA$ là phân giác của góc $H$.
				\item $HA\perp HC\Rightarrow HC$ là phân giác ngoài của góc $H$.
			\end{itemize}
			Từ đó, suy ra $IC$ là tia phân giác của $\widehat{HIK}$.\\
			Trong $\triangle IHD$ ta có

				\item $IB$ là tia phân giác của góc $I$.
				\item $IC$ là tia phân giác ngoài của góc $I$.
			\end{itemize}
			Từ đó, suy ra $IC\perp IB\Rightarrow IC\perp AB$.
			\item Chứng minh $KB$ là tia phân giác của $\widehat{HKI}$, từ đó suy ra $KB$ vuông góc với $AC$.\\

				\item $HB$ là phân giác ngoài của góc $H$.
			\end{itemize}
			Từ đó, suy ra $KB$ là tia phân giác của $\widehat{HKI}$.\\

				\item $KC$ là tia phân giác của góc $K$.
			\end{itemize}
			Từ đó, suy ra $KB\perp KC\Rightarrow KB\perp AC$.
		\end{enumerate}
