\begin{ex}%[8D1B1]%[8D1K11]%[8D1B12]
Thực hiện phép tính:
\begin{enumerate}
\item $2xy.3x^2y^3$.
\item $x.(x^2-2x+5)$.
\item $(3x^2-6x):3x$.
\item $(x^2-2x+1):(x-1)$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $2xy.3x^2y^3=6x^3y^4$.
\item $x.(x^2-2x+5)=x^3-2x^2+5x$.
\item $(3x^2-6x):3x=x-2$.
\item $(x^2-2x+1):(x-1)=x-1$
\end{enumerate}
}
\end{ex}

\begin{ex}%[8D1B6]%[8D1K7]%[8D1K6]
Phân tích các đa thức sau thành nhân tử:
\begin{enumerate}
\item $5x^2y-10xy^2$.
\item $3(x + 3)-x^2+9$.
\item $x^2-y^2+xz-yz$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $5x^2y-10xy^2=5xy(x-2y)$.
\item $3(x + 3)-x^2+9=(6-x)(x+3)$.
\item $x^2-y^2+xz-yz=(x^2-y^2)+(xz-yz)=(x-y)(x+y+z)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[8D2K1]%[8D2K3]
Cho biểu thức: $A=\dfrac{x^2}{x^2-4}-\dfrac{x}{x-2}+\dfrac{2}{x+2}$.
\begin{enumerate}
\item Với điều kiện nào của $x$ thì giá trị của biểu thức $A$ được xác định?
\item Rút gọn biểu thức $A$.
\item Tìm giá trị của biểu thức $A$ tại $x=1$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Biểu thức $A$ xác định khi $\heva{&x-2 \neq 0\\&x+2 \neq 0}$.
\item $A=\dfrac{x^2}{x^2-4}-\dfrac{x}{x-2}+\dfrac{2}{x+2}=\dfrac{x^2-x(x+2)+2(x-2)}{x^2-4}=\dfrac{-4}{x^2-4}$.
\item Thay $x=1$ vào $A$ ta được $A=\dfrac{4}{3}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[8H1B9]%[8H1K9]%[8H12G9]
Cho tam giác $MNP$ vuông tại $M$, đường cao $MH$. Gọi $D, E$ lần lượt là chân các đường vuông góc hạ từ $H$ xuống $MN$ và $MP$.
\begin{enumerate}
\item Chứng minh tứ giác $MDHE$ là hình chữ nhật.
\item Gọi $A$ là trung điểm của $HP$. Chứng minh tam giác $DEA$ vuông.
\item Tam giác $MNP$ cần có thêm điều kiện gì để $DE=2EA$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1.2]
\clip (-1,-1) rectangle (7,5);
\tkzDefPoints{0/0/M, 6/0/P, 0/4/N}
\tkzDrawAltitude(P,N)(M)
\tkzGetPoint{H}
\tkzDrawAltitude(M,P)(H)
\tkzGetPoint{E}
\tkzDrawAltitude(M,N)(H)
\tkzGetPoint{D}
\tkzDefMidPoint(H,P)
\tkzGetPoint{A}
\tkzInterLL(H,M)(E,D)
\tkzGetPoint{O}
\tkzDrawPoints(M,N,P,A,H,D,E,O)
\tkzLabelPoints[below](M,P,E)
\tkzLabelPoints[left](N,D,O)
\tkzLabelPoints[above](H,A)
\tkzDrawSegments(M,N M,P N,P D,E A,E)
\tkzMarkRightAngle(M,H,P)
\tkzMarkRightAngle(N,D,H)
\tkzMarkRightAngle(H,E,P)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Tứ giác $MDHE$ có ba góc vuông nên là hình chữ nhật.
\item $MDHE$ là hình chữ nhật nên hai đường chéo bằng nhau và  cắt nhau tại trung điểm của mỗi đường.\\
Gọi $O$ là giao điểm của $MH$ và $DE$.\\
Ta có: $OH=OE \Rightarrow \widehat{OHE}=\widehat{OEH}$.\\
$\triangle EHP$ vuông tại $E$ và $A$ là trung điểm $PH$ nên $AE = AH$.\\
Suy ra $\widehat{AHE}=\widehat{AEH}$.\\
Do đó $\widehat{AEO}=\widehat{AHO}=90^\circ$.\\
Vậy tam giác $DEA$ vuông tại $E$.
\item Ta có $DE=2EA$\\
$\Leftrightarrow OE=EA$\\
$\Leftrightarrow$ $\triangle OEA$ vuông cân tại $E$.\\
$\Leftrightarrow \widehat{EOA}=45^\circ$\\
$\Leftrightarrow \widehat{HOE}=90^\circ$\\
$\Leftrightarrow MDHE$ là hình vuông.\\
$\Leftrightarrow MH$ là phân giác góc $M$.\\
$\Leftrightarrow \triangle MNP$ vuông cân tại $M$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[8D1G5]
Cho $a+b=1$. Tính giá trị của biểu thức $M=a^3+b^3+3ab(a^2+b^2)+6a^2b^2(a+b)$.
\loigiai{
\begin{align*}
M&=a^3+b^3+3ab(a^2+b^2)+6a^2b^2(a+b)\\
&=(a+b)(a^2-ab+b^2)+3ab(a^2+b^2)+6a^2b^2(a+b)\\
&=(a+b)((a+b)^2-3ab)+3ab((a+b)^2-2ab)+6a^2b^2(a+b)\\
&=1-3ab+3ab(1-2ab)+6a^2b^2\\
&=1
\end{align*}
}
\end{ex}
