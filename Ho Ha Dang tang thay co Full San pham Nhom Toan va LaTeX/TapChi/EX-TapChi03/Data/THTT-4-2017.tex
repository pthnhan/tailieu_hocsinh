\begin{bt}[T3/478-Toán học \& Tuổi trẻ, tháng 4 năm 2017]%[Nguyễn Trường Sơn,Dự án EX_TapChi03]%[9D0K7]
Xét ba số thực dương $ a,b,c $ sao cho $ \sqrt{a}+\sqrt{b}+\sqrt{c}=1 $. Tìm giá trị lớn nhất của:  \[P=\sqrt{abc}\left( \dfrac{1}{\sqrt{(a+b)(a+c)}}+ \dfrac{1}{\sqrt{(b+c)(a+b)}}+ \dfrac{1}{\sqrt{(c+b)(a+c)}}\right).  \]
\loigiai{ \textbf{\textit{Cách 1.}} Áp dụng bất đẳng thức AM-GM cho hai số dương, ta có:
	\[a+b\geq 2\sqrt{ab},b+c \geq 2\sqrt{bc},a+c\geq 2\sqrt{ac},\]
	nên
\begin{align*}
\dfrac{\sqrt{abc}}{\sqrt{(a+b)(a+c)}}\leq\dfrac{\sqrt{abc}}{\sqrt{2\sqrt{ab}\cdot 2\sqrt{ac}}}=\dfrac{\sqrt[4]{bc}}{2}\leq \dfrac{\sqrt{b}+\sqrt{c}}{4}.\tag{1}
\end{align*}	
Đẳng thức xảy ra khi và chỉ khi $ a=b=c $. 
Chứng minh tương tự ta có:
{\allowdisplaybreaks
\begin{align*}
\dfrac{\sqrt{abc}}{\sqrt{(a+b)(b+c)}}&\leq \dfrac{\sqrt{a}+\sqrt{c}}{4},\tag{2}\\
\dfrac{\sqrt{abc}}{\sqrt{(a+c)(b+c)}}&\leq \dfrac{\sqrt{a}+\sqrt{b}}{4}.\tag{3}
\end{align*}}Từ $(1)$, $(2)$, $(3)$ và kết hợp với giả thiết $ \sqrt{a}+\sqrt{b}+\sqrt{c}=1 $ ta có:
\[ P\leq \dfrac{\sqrt{a}+\sqrt{b}}{4}+\dfrac{\sqrt{c}+\sqrt{b}}{4}+\dfrac{\sqrt{a}+\sqrt{c}}{4}=\dfrac{1}{2}. \]
Dấu bằng xảy ra khi và chỉ khi $ a=b=c=\dfrac{1}{9} $.\\
Vậy giá trị lớn nhất của biểu thức $ P $ là $ \dfrac{1}{2} $ khi $ a=b=c=\dfrac{1}{9} $.\\
\textbf{\textit{Cách 2.}} Đặt $ x=\sqrt{a},y=\sqrt{b},z=\sqrt{c}$ $ (x,y,z>0) .$
Khi đó ta có: $ x+y+z=1 $.
Áp dụng bất đẳng thức AM-GM cho hai số dương, ta có:
\[\sqrt{(x^{2}+y^{2})(x^{2}+z^{2})}\geq 2\sqrt{2xy \cdot 2xz}=2x\sqrt{yz}\]
nên
\begin{align*}
\dfrac{1}{\sqrt{(x^{2}+y^{2})(x^{2}+z^{2})}}\leq\dfrac{1}{2x\sqrt{yz}}=\dfrac{\sqrt{yz}}{2xyz}\leq \dfrac{y+z}{4xyz}.\tag{5}
\end{align*}	
Chứng minh tương tự ta có:
{\allowdisplaybreaks
\begin{align*}
\dfrac{1}{\sqrt{(z^{2}+y^{2})(x^{2}+y^{2})}}&\leq \dfrac{x+z}{4xyz}.\tag{6}\\
\dfrac{1}{\sqrt{(z^{2}+y^{2})(x^{2}+z^{2})}}&\leq \dfrac{x+y}{4xyz}.\tag{7}
\end{align*}}Từ $(5)$, $(6)$, $(7)$ ta có:
$ P\leq \dfrac{1}{2}. $
Dấu bằng xảy ra khi và chỉ khi $ a=b=c=\dfrac{1}{9} $.
Vậy giá trị lớn nhất của biểu thức $ P $ là $ \dfrac{1}{2} $ khi $ a=b=c=\dfrac{1}{9} $.
}
\end{bt}

\loigiai{
Đặt $ f(x)=y $, ta thu được hệ phương trình:
$\begin{cases}
x^{2}+2x+m=y &(1)\\
y^{2}+2y+m=x. &(2)
\end{cases}$\\
Từ (1) và (2) suy ra $ (x-y)(x+y+3)=0\Leftrightarrow \left[\begin{array}{l}
x=y \\
y=-3-x.\end{array}\right. $
\begin{itemize}
	\item Xét trường hợp $ y=x $, ta thu được phương trình $ x^{2}+x+m=0$\hfill $(3)$\\

Với $ m>\dfrac{1}{4} $ thì (3) vô nghiệm.\\
Với $ m=\dfrac{1}{4} $ thì (3) có nghiệm kép $ x=-\dfrac{1}{2} $.\\

Với $ m>\dfrac{-3}{4} $ thì (4) vô nghiệm.\\
Với $ m=\dfrac{-3}{4} $ thì (4) có nghiệm kép $ x=-\dfrac{-3}{2} $.\\
Với $ m<\dfrac{-3}{4} $ thì (4) có hai nghiệm $ x_{3,4}=\dfrac{-3 \mp \sqrt{-3-4m}}{2} $.
\end{itemize}
Kết luận:
\begin{itemize}
	\item Với $ m<\dfrac{-3}{4} $ thì phương trình có bốn nghiệm

\item Với  $ m=\dfrac{-3}{4} $ thì phương trình có ba nghiệm: hai nghiệm đơn là  $ x_{1,2}=\dfrac{-1\pm \sqrt{1-4m}}{2} $ và một nghiệm  kép $ x=-\dfrac{-3}{2} $.
\item Với $ m \in \left( \dfrac{-3}{4};\dfrac{1}{4}\right)  $ thì phương trình có hai nghiệm  $ x_{1,2}=\dfrac{-1\pm \sqrt{1-4m}}{2} $.

\item Với $ m>\dfrac{1}{4} $ thì phương trình vô nghiệm.
\end{itemize}

Vậy bài toán đúng với $ n=k+1 $.

Cho $a,b,c$ là độ dài ba cạnh của một tam giác. Xét biểu thức $$F(a,b,c)=a^3+b^3+c^3+3abc-ab(a+b)-bc(b+c)-ca(a+c)$$
Chứng minh rằng $F(a,b,c)=\min\left\lbrace F(a+b,b+c,c+a), 4a^2b^2c^2F\left(\dfrac{1}{a},\dfrac{1}{b},\dfrac{1}{c}\right) \right\rbrace $.
\loigiai{Không mất tính tổng quát của bài toán, ta giả sử $c\geq a, c\geq b$. Khi đó
	{\allowdisplaybreaks
	\begin{align*}
	F(a,b,c)&=a^3+b^3+c^3+3abc-ab(a+b)-bc(b+c)-ca(a+c) \\
	&=(a^3+b^3+c^3+2abc-a^2b-ca^2-ab^2-b^2c)+(abc-c^2a-bc^2+c^3)\\
	&=(a+b-c)(a-b)^2+c(a-c)(b-c).
	\end{align*}}Ta dễ chứng minh được $F(a+b,b+c,c+a)\geq F(b+c,c+a,a+b)$.

{\allowdisplaybreaks
\begin{align*}
F(b+c,c+a,a+b)-F(a,b,c)&=(b+c+c+a-a-b)(b+c-c-a)^2\\
&+(a+b)(b+c-a-b)(c+a-a-b)\\
&-(a+b-c)(a-b)^2-c(a-c)(b-c)\\
&=(3c-a-b)(a-b)^2+(a+b-c)(c-a)(c-b)\geq 0 \tag{1}
\end{align*}}(do $c\ge a, c\ge b$ và $a, b, c$ là độ dài ba cạnh của một tam giác).

{\allowdisplaybreaks
\begin{align*}
4a^2b^2c^2F\left(\dfrac{1}{a},\dfrac{1}{b},\dfrac{1}{c} \right)-F(a,b,c)&=4a^2b^2c^2\left[ \left(\dfrac{1}{a}+\dfrac{1}{b}-\dfrac{1}{c} \right)\left(\dfrac{1}{a}-\dfrac{1}{b}\right)^2 +\dfrac{1}{c}\left(\dfrac{1}{a}-\dfrac{1}{c}\right)\left(\dfrac{1}{b}-\dfrac{1}{c}\right) \right]\\
&-(a+b-c)(a-b)^2-c(a-c)(b-c)\\
&=\dfrac{G}{abc}.
\end{align*}}Ta sẽ đi tính $G$ như sau
{\allowdisplaybreaks
\begin{align*}
G&=4c^2(bc+ca-ab)(a-b)^2+4a^2b^2(a-c)(b-c)-abc(a+b-c)(a-b)^2-abc^2(a-c)(b-c)\\
&=c(4c^2a+4bc^2-a^2b-ab^2-3abc)(a-b)^2+ab(4ab-c^2)(a-c)(b-c)\\
&\geq (4c^3a+4bc^3-a^2bc-ab^2c-3abc^2)(a-b)^2+ab\left[4ab-(a+b)^2\right](a-c)(b-c)\\
&=\left[4c^3a+4bc^3-a^2bc-ab^2c-3abc^2-ab(ab-bc-ca+c^2)\right] (a-b)^2 \geq 0.\tag{2}
\end{align*}}Vì $4c^3a\geq 4abc^2,bc^3\geq a^2b^2$. Từ $(1)$ và $(2)$ suy ra được khẳng định bài toán. Đẳng thức xảy ra khi và chỉ khi $a=b=c$, tức là tam giác đều.

Chứng minh rằng 
$$\mathrm{BCNN}(1,2,\ldots,2n)$$
chia hết cho $\mathrm{C}^{n}_{2n}$ với mọi số nguyên dương $n$.
\loigiai{Cho số nguyên $x$ và $p$ là một số nguyên tố bất kì, ta kí hiệu $\nu_p(x)$ là số mũ cao nhất của $p$ mà chia hết $x$.
Đặt $M=\mathrm{BCNN}(1,2,\ldots,2n)$. Ta cần chứng minh $\nu_p(M)\ge \nu_p(\mathrm{C}^{n}_{2n})$, với mọi số nguyên tố $p$.
Áp dụng công thức $\nu_p(n!)=\displaystyle\sum\limits_{i=1}^{+\infty}\left[\dfrac{n}{p^i}\right]$ và $\mathrm{C}^{n}_{2n}=\dfrac{(2n)!}{n!n!}$ ta được $$\nu_p\left(\mathrm{C}^{n}_{2n}\right)=\displaystyle\sum\limits_{i=1}^{+\infty}\left[\dfrac{2n}{p^i}\right]-2\displaystyle\sum\limits_{i=1}^{+\infty}\left[\dfrac{n}{p^i}\right]\cdot$$
Đặt $m=\nu_p(\mathrm{C}^{n}_{2n})$. Khi đó ta có $p^m\le 2n$. Thật vậy, giả sử ngược lại, $p^m>2n$. Khi đó $$p^m>n\Rightarrow \left[\dfrac{2n}{p^i}\right]=\left[\dfrac{n}{p^i}\right]=0,\forall i\ge m, i\in\mathbb{Z}.$$
Suy ra \[m=\nu_p\left(\mathrm{C}^{n}_{2n}\right)=\displaystyle\sum\limits_{i=1}^{+\infty}\left[\dfrac{2n}{p^i}\right]-2\displaystyle\sum\limits_{i=1}^{+\infty}\left[\dfrac{n}{p^i}\right]=\displaystyle\sum\limits_{i=1}^{+\infty}\left(\left[\dfrac{2n}{p^i}\right]-2\left[\dfrac{n}{p^i}\right]\right)=\displaystyle\sum\limits_{i=1}^{m-1}\left(\left[\dfrac{2n}{p^i}\right]-2\left[\dfrac{n}{p^i}\right]\right).\tag{1}\]
Với $x\in\mathbb{R}$ bất kì ta có 
{\allowdisplaybreaks
\begin{align*}
\left[2x\right]&=2x-\left\{2x\right\}=2\left([x]+\left\{x\right\}\right)-\left\{2x\right\}\\ 
&=2\left[x\right]+2\left\{x\right\}-\left\{2x\right\}\le 2\left[x\right]+2\left\{x\right\}<2\left[x\right]+2.
\end{align*}}Suy ra $\left[2x\right]-2\left[x\right]\le 1 \quad \text{(do $\left[2x\right]-2\left[x\right]$ là số nguyên)}.$
Do đó, từ $(1)$ suy ra $m\le m-1$ (mâu thuẫn). Vậy $p^m\le 2n$.\hfill$(2)$\\

Do tính lớn nhất của $k$ nên từ $(2)$ suy ra $k\ge m$. Từ đó ta có điều phải chứng minh.}
