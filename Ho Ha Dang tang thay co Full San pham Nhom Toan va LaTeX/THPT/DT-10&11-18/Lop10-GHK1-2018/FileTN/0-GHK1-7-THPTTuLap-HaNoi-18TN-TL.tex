\begin{ex}%[Kiểm tra 1 tiết, Tự Lập, Hà Nội,2019]%[Trần Toàn,Dự án GK1-Toán 10-11]%[0D1B3-1]
Cho hai tập hợp $ A=\left\lbrace x\in \mathbb{R}\,\mid\,(2x-x^2)(x-1)=0 \right\rbrace$, $ B=\left\lbrace n \in \mathbb{N}\,\mid\,0<n^2<10 \right\rbrace$. Chọn mệnh đề đúng?
	\choice
	{\True $A \cap B =\left\lbrace 1;2 \right\rbrace $}
	{$A \cap B =\left\lbrace 2 \right\rbrace $}
	{ $A \cap B =\left\lbrace 0;1;2;3 \right\rbrace $}
	{$A \cap B =\left\lbrace 0;3 \right\rbrace $}
	\loigiai{
Ta có 
\begin{itemize}
	\item $ (2x-x^2)(x-1)=0\Leftrightarrow \hoac{& x=0 \\ &x=1\\&x=2 }\Rightarrow A=\left\lbrace 0;1;2 \right\rbrace  $.
	\item  $ B= \left\lbrace 1;2;3 \right\rbrace$.
\end{itemize}
Suy ra $ A\cap B= \left\lbrace 1;2 \right\rbrace$.}
\end{ex}

Cho hai tập hợp $ A=\left\lbrace x\in \mathbb{Z}\,\mid\, x^2+x-6=0 \right\rbrace$, $ B=\left\lbrace x\in \mathbb{N}\,\mid\, 2x^2-3x+1=0 \right\rbrace  $. Chọn khẳng định đúng.	
	\choice
	{$ B\setminus A=\left\lbrace 1;2 \right\rbrace  $}
	{$ A \cap B=\left\lbrace -3;1;2 \right\rbrace  $}
	{\True $ A\setminus B=A $}
	{$ A \cup B=\varnothing$}
	\loigiai
	{
Từ giả thiết ta có 
\begin{itemize}
	\item $ x^2+x-6=0\Leftrightarrow \hoac{& x=-3 \\ &x=2 }$, do $ x \in \mathbb{Z} \Rightarrow A=\left\lbrace -3;2 \right\rbrace $.
	\item  $ 2x^2-3x+1=0\Leftrightarrow \hoac{& x=1 \\ &x=\dfrac{1}{2}}$, do $ x\in \mathbb{N} \Rightarrow B= \left\lbrace 1 \right\rbrace$.
\end{itemize}
Vậy $ B\setminus A=\left\lbrace 1 \right\rbrace  $, $ A\cap B=\varnothing $, $ A\setminus B=A $, $ A \cup B= \left\lbrace -3;1;2 \right\rbrace  $.	 
	}
\end{ex}

\begin{ex}%[Kiểm tra 1 tiết, Tự Lập, Hà Nội,2019]%[Trần Toàn,Dự án GK1-Toán 10-11]%[0D1K3-3]
	Trong một lớp học có $40$ học sinh, trong đó có $30$ học sinh đạt học sinh giỏi môn Toán, $25$ học sinh đạt học sinh giỏi môn Văn. Biết rằng chỉ có $5$ học sinh không đạt danh hiệu học sinh giỏi môn nào trong cả hai môn Toán và Văn. Hỏi có bao nhiêu học sinh chỉ học giỏi một môn trong hai môn Toán hoặc Văn?
	\choice
	{$20$}
	{\True $15$}
	{$5$}
	{$10$}
	\loigiai{
		Gọi $A$ là hai tập hợp học sinh giỏi môn Toán và $B$ là hai tập hợp học sinh giỏi môn Văn.
		\begin{center}
			{
				\begin{tikzpicture}[scale=.8, font=\footnotesize, line join=round, line cap=round, >=stealth]
				\def\firstcircle{(0.5cm,0) ellipse [x radius=3cm, y radius=1.5cm, rotate=-50]}
				\def\secondcircle{(3cm,-0.5) ellipse [x radius=2cm, y radius=1.2cm, rotate=1500]}
				\colorlet{circle edge}{black!50}
				\colorlet{circle area}{black!20}
				\tikzset{filled/.style={fill=circle area, draw=circle edge, thick},
					outline/.style={draw=circle edge, thick}}
				\begin{scope}
				\clip \firstcircle;
				\fill[filled] \secondcircle;
				\end{scope}
				\draw[outline] \firstcircle node {$A$};
				\draw[outline] \secondcircle node {$B$}; 
				\end{tikzpicture}
			}
		\end{center}

		$$ \left| A\cup B \right| =40-5=35 .$$
		Số học sinh học giỏi cả hai môn Toán và Văn là
		$$ \left| A\cap B \right| =30+25-35=20 .$$
		Số học sinh chỉ học giỏi một môn trong hai môn Toán hoặc Văn là
		$$ \left| (A\cup B) \backslash (A\cap B) \right| =35-20=15. $$
	}
\end{ex}

Cho $X=\left[-3;1\right)$, $Y=\left(0;4\right)$.

\begin{bt}%[Kiểm tra 1 tiết, Tự Lập, Hà Nội,2019]%[Trần Toàn,Dự án GK1-Toán 10-11]%[0D1K4-1]
Cho tập hợp $B=\{x\in \mathbb{R}|\,-4<x\le 4\}$ và $C=\{x\in \mathbb{R}|\,x\le m\}$. Xác định tập $B\cap C$ tùy theo giá trị của $m$?
\loigiai{
	Ta có $ B=\left(-4;4 \right]$, $ C=\left(-\infty;m\right]$.
Xét các trường hợp sau
\begin{itemize}
\item $m\le -4$ thì $B\cap C=\varnothing$.
\item $-4<m< 4$ thì $B\cap C=\left(-4;m\right]$.
\item $m\ge 4$ thì $B\cap C=\left(-4;4\right]$.
\end{itemize}
}
\end{bt}
\begin{bt}%[Kiểm tra 1 tiết, Tự Lập, Hà Nội,2019]%[Trần Toàn,Dự án GK1-Toán 10-11]%[0D1B3-1]%[0D1B3-2]
Gọi $N(A)$ là số phần tử của tập $A$. Cho $N(A)=38$, $N(B)=20$, $N(A\cup B)=45$. Tính $N(A\cap B)$, $N(A\setminus B)$, $N(B\setminus A)$.
\loigiai{
Ta có $N(A\cap B)=n(A)+n(B)-N(A\cup B)=13.$\\
$N(A\setminus B)=N(A\cup B)-n(B)=25$.\\
$N(B\setminus A)=N(A\cup B)-n(A)=7$.}
\end{bt}
\begin{bt}%[Kiểm tra 1 tiết, Tự Lập, Hà Nội,2019]%[Trần Toàn,Dự án GK1-Toán 10-11]%[0D1K3-2]
Cho các tập hợp $A$, $B$ và $C$. Chứng minh $A\setminus	\left(B\cap C\right)=\left(A \setminus B\right) \cup \left(A\setminus C\right)$.
\loigiai{
Ta phải chứng minh hai điều
\begin{itemize}
\item $A\setminus\left(B\cap C\right) \subset \left(A \setminus B\right) \cup \left(A\setminus C\right)$.
Thật vậy, giả sử 
\begin{eqnarray*}
x \in A\setminus\left(B\cap C\right) &\Leftrightarrow& \heva{& x \in A \\ & x \notin B\cap C}  \Leftrightarrow  \heva{& x \in A\\ & \hoac{& x\notin B\\ & x \notin C}} \\ &\Leftrightarrow & \hoac{& \heva{& x\in A\\ & x\notin B}\\ & \heva{& x\in A\\ & x\notin C}}\Leftrightarrow \hoac{& x\in A\setminus B\\ & x \in A\setminus C}\\&\Leftrightarrow &  x \in \left(A \setminus B\right) \cup \left(A\setminus C\right).
\end{eqnarray*}
Do đó $A\setminus\left(B\cap C\right) \subset \left(A \setminus B\right) \cup \left(A\setminus C\right)$ \quad (1).
\item $\left(A \setminus B\right) \cup \left(A\setminus C\right) \subset A\setminus	\left(B\cap C\right)$.
Thật vây, giả sử 
\begin{eqnarray*}
x \in \left(A \setminus B\right) \cup \left(A\setminus C\right) &\Leftrightarrow& \hoac{& x \in \left(A \setminus B\right)\\ & x\in \left(A\setminus C\right) }\Leftrightarrow  \hoac{& \heva{& x \in A\\ & x\notin B}\\ & \heva{& x\in A\\ & x\notin C}}\\&\Leftrightarrow & \heva{& x\in A\\ & \hoac{& x\notin B\\ & x \notin C}}\Leftrightarrow \heva{& x\in A\\ & x\notin B\cap C}\\
&\Leftrightarrow & x\in A\setminus	\left(B\cap C\right).
\end{eqnarray*}
Vậy $\left(A \setminus B\right) \cup \left(A\setminus C\right) \subset A\setminus	\left(B\cap C\right)$ \quad (2).
\end{itemize}
Từ (1) và (2) ta có điều phải chứng minh.
}
\end{bt}
