\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Hưng Yên, 2017]%[9D1B8]%[9D5K1]
	Cho biểu thức $ P = \dfrac{2\sqrt{x} -1}{\sqrt{x}+1} - \dfrac{2\sqrt{x}+1}{\sqrt{x}-1} $ với $ x \ge 0, x \ne 1 $.
	\begin{enumerate} [a)]
		\item Rút gọn biểu thức $ P $.
		\item Tìm các giá trị $ x $ để $ P = \dfrac{3}{4} $.
		\item Tìm giá trị nhỏ nhất của biểu thức $ A = (\sqrt{x}-4)(x-1)P $.
	\end{enumerate}
\loigiai{ Với điều kiện $ x > 0, x \ne 1 $.
	\begin{enumerate}[a)]
		\item $ P = \dfrac{2\sqrt{x} -1}{\sqrt{x}+1} - \dfrac{2\sqrt{x}+1}{\sqrt{x}-1} = \dfrac{(2\sqrt{x}-1)(\sqrt{x}+1) - (2\sqrt{x}+1)(\sqrt{x}-1)}{x-1} = \dfrac{2\sqrt{x}}{x-1} $.
		\item Giải phương trình:
		\[ P = \dfrac{3}{4}\Leftrightarrow  \dfrac{2\sqrt{x}}{x-1} = \dfrac{3}{4} \Leftrightarrow 3x-8\sqrt{x} -3 = 0 \Leftrightarrow (\sqrt{x}-3)(3\sqrt{x}+1) = 0 \Leftrightarrow \sqrt{x} = 3 \Leftrightarrow x = 9 \]
		\item Biểu thức $ A = (\sqrt{x}-4)(x-1)P = 2\sqrt{x}(\sqrt{x}-4) = 2(\sqrt{x} - 2)^2 -8 \ge -8 $.\\
		Vậy giá trị nhỏ nhất của $ A $ là $ -8 $, khi chỉ khi $ x = 4 $.
	\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Hưng Yên, 2017]%[9D4K6]
	Cho Parabol $ (P): y =x^2 $ và đường thẳng $ (d): y = 3x + m -2 $. Tìm tham số $ m $ để $ (P) $ cắt $ (d) $ tại hai điểm phân biệt có hoành độ dương.
\loigiai{
	Phương trình hoành độ giao điểm của $ (P) $ và $ (d) $ là $ x^2 = 3x+m-2 \Leftrightarrow x^2 - 3x + 2-m = 0 \quad (1) $.\\
	Yêu cầu bài toán tương đương $ (1) $ có hai nghiệm phân biệt dương $ \Leftrightarrow  \begin{cases}
	\Delta > 0 \\
	S > 0 \\
	P > 0
	\end{cases} $. \\
	Mà ta có $ \Delta = 9 - 4(2-m) = 1 +4m $ và $ \begin{cases}
	S = 3 \\
	P = 2 -m
	\end{cases} $.\\
	Vậy $ \begin{cases}
	1 + 4m > 0 \\
	2 - m > 0 
	\end{cases} \Leftrightarrow -\dfrac{1}{4} < m < 2 $.
}
\end{ex}
\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Hưng Yên, 2017]%[9D4B3]%[9D3B4]
	\begin{enumerate}[a)]
		\item Giải hệ phương trình $ \begin{cases}
		x^2 -2xy = 2y - x \\
		x^2 + 2x = 9 - y
		\end{cases}$.
		\item Giải phương trình $ \sqrt{\dfrac{1-2x}{x}} = \dfrac{3x+x^2}{x^2+1} $.
	\end{enumerate}
\loigiai{ \hfill
	\begin{enumerate}[a)]
		\item Từ phương trình thứ nhất của hệ ta có $$  x^2 +x -2xy -2y = 0 \Leftrightarrow (x+1)(x-2y) = 0 \Leftrightarrow \left[\begin{array}{l}
		x = -1 \\ x = 2y
		\end{array} \right.  $$
		Xét $ x = -1 $ thay vào phương trình còn lại ta thu được $ y = 10 $.\\
		Xét $ x = 2y $ thay vào phương trình còn lại ta thu được $ 4y^2 + 5y - 9 = 0 \Leftrightarrow \left[\begin{array}{l}
		y = 1 \Rightarrow x = 2 \\ y = -\dfrac{9}{4} \Rightarrow x = -\dfrac{9}{2}
		\end{array} \right.  $. \\
		Vậy hệ đã cho có $ 3 $ nghiệm $ (x;y) $ là $ (-1;10),(2;1),\left(-\dfrac{9}{2};-\dfrac{9}{4}\right)  $.
		\item Điều kiện $ \dfrac{1-2x}{x} \ge 0 \Leftrightarrow x \in \left(0;\dfrac{1}{2}\right]  $.\\
		\begin{align*}
		&\sqrt{\dfrac{1-2x}{x}} = \dfrac{3x+x^2}{x^2+1} \Leftrightarrow \sqrt{\dfrac{1-2x}{x}} - 1 = \dfrac{3x+x^2}{x^2+1} - 1\\
		&\Leftrightarrow\dfrac{\dfrac{1-2x}{x}-1}{\sqrt{\dfrac{1-2x}{x}} +1} = \dfrac{3x-1}{x^2+1} \Leftrightarrow\dfrac{1-3x}{x\left (\sqrt{\dfrac{1-2x}{x}} +1\right )} = \dfrac{3x-1}{x^2+1}\\
		&\Leftrightarrow (1-3x)\left(\dfrac{1}{x\left (\sqrt{\dfrac{1-2x}{x}} +1\right )} + \dfrac{1}{1+x^2} \right) = 0 \\
		&\Leftrightarrow x = \dfrac{1}{3}. \text{ Vì } \dfrac{1}{x\left (\sqrt{\dfrac{1-2x}{x}} +1\right )} + \dfrac{1}{1+x^2} > 0, \forall x \in \left(0;\dfrac{1}{2}\right] 
		\end{align*}
	\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Hưng Yên, 2017]%[9D5G3]
	Tìm tất cả các nghiệm nguyên của phương trình $ y^3 - 2x -2 = x(x+1)^2 $.
\loigiai{
	Ta có $ y^3 = x^3 + 2x^2 + 3x +2 $. Rõ ràng $ 2x^2 + 3x + 2 = 2\left(x + \dfrac{3}{4}\right)^2 + \dfrac{7}{8} \ge \dfrac{7}{8} > 0  $. Do đó $ y^3 > x^3 $.\\
	Mặt khác, xét $ |x| > 1 $. \\
	Khi đó $ y^3 = (x+1)^3 + 1-x^2 \le (x+1)^3 $, mẫu thuẫn vì $ x^3 < y^3 < (x+1)^3 $.\\
	Vậy $ |x| \le 1 $. Khi đó $ x = \pm 1 \text{ hoặc } x = 0 $.
	\begin{enumerate}[TH1.]
		\item $ x = -1 $. Thay vào pt được $ y = 1 $.
		\item $ x = 1 $. Thay vào pt được $ y = 2 $.
		\item $ x = 0 $. Thay vào pt được $ y = \sqrt[3]{2} \notin \mathbb{Z} $, loại.
	\end{enumerate}
	Vậy phương trình có 2 nghiệm $ (x;y) $ là $ (-1;1), (1;2) $.
}
\end{ex}
\begin{ex}%[Đề thi vào 10 chuyên, Sở giáo dục Hưng Yên, 2017]%[9D5G1]
	Cho $ a,b,c $ dương và $ a^4b^4 + b^4c^4 + c^4a^4 = 3a^4b^4c^4 $. Chứng minh rằng
	\[ \dfrac{1}{a^3b+2c^2+1} + \dfrac{1}{b^3c+2a^2+1} + \dfrac{1}{c^3a+2b^2 +1} \le \dfrac{3}{4}. \]
\loigiai{ \textbf{Cách 1.} 
	Từ điều kiện đề bài ta có $ \dfrac{1}{a^4} + \dfrac{1}{b^4} + \dfrac{1}{c^4} = 3 $.\\
	Trước hết ta chứng minh BĐT sau: Với $ x,y> 0 $ ta có: \[ \dfrac{1}{x+y} \le \dfrac{1}{4}\left(\dfrac{1}{x}+\dfrac{1}{y}\right) \]
	BĐT trên tương đương với $ (x+y)\left(\dfrac{1}{x} + \dfrac{1}{y}\right) \ge 4 $. BĐT này đúng theo $ AM-GM $ như sau:
	\begin{align*}
	&x+y\ge 2\sqrt{xy} > 0 \\
	&\dfrac{1}{x} + \dfrac{1}{y} \ge \dfrac{2}{\sqrt{xy}} > 0
	\end{align*}
	Từ đó áp dụng vào ta có: 
	\[ \dfrac{1}{a^3b+2c^2+1} \le \dfrac{1}{4}\left(\dfrac{1}{a^3b+1} + \dfrac{1}{2c^2} \right) \le \dfrac{1}{4}\left(\dfrac{1}{4}\left(\dfrac{1}{a^3b} + 1 \right)  + \dfrac{1}{2c^2}\right) = \dfrac{1}{16}\left(\dfrac{1}{a^3b} + \dfrac{2}{c^2} + 1\right)  \]
	Mà theo $ AM - GM $ ta có $ \dfrac{1}{a^4} + \dfrac{1}{a^4} + \dfrac{1}{a^4} + \dfrac{1}{b^4} \ge \dfrac{4}{a^3b} $ và $ \dfrac{1}{c^4} + 1 \ge  \dfrac{2}{c^2} $.\\
	Nên $ \dfrac{1}{a^3b+2c^2+1} \le \dfrac{1}{16}\left( \dfrac{3}{4a^4} + \dfrac{1}{4b^4} + \dfrac{1}{c^4} + 2 \right)  $.\\
	Tương tự ta có $ \begin{cases}
	\dfrac{1}{b^3c+2a^2+1} \le \dfrac{1}{16} \left(\dfrac{3}{4b^4} + \dfrac{1}{4c^4} + \dfrac{1}{a^4} + 2 \right) \\
	\dfrac{1}{c^3a+2b^2+1} \le \dfrac{1}{16}\left( \dfrac{3}{4c^4} + \dfrac{1}{4a^4} + \dfrac{1}{b^4} + 2 \right)
	\end{cases} $.\\
	Cộng theo vế các BĐT trên lại ta có $ VT \le \dfrac{1}{16}\left( 2\left(\dfrac{1}{a^4} + \dfrac{1}{b^4} + \dfrac{1}{c^4}\right)  +6 \right) = \dfrac{3}{4} $.\\
	Đẳng thức xảy ra khi chỉ khi $ a = b= c = 1 $.
	
	\textbf{Cách 2.} Từ điều kiện ta có $ \dfrac{1}{a^4} + \dfrac{1}{b^4} + \dfrac{1}{c^4} = 3 $. Mặt khác: $ 3 = \dfrac{1}{a^4} + \dfrac{1}{b^4} + \dfrac{1}{c^4} \ge \dfrac{3}{\sqrt[3]{a^4b^4c^4}} \Rightarrow abc \ge 1 $.\\
	Ta có $ a^3b + 2c^2 + 1 \ge 4\sqrt[4]{a^3bc^4} \ge 4\sqrt[4]{a^2c^3} = 4\sqrt{ac}\sqrt[4]{c} $. Do đó 
	\[ \dfrac{1}{a^3b+2c^2+1} \le \dfrac{1}{4\sqrt{ac}\sqrt[4]{c}} \le \dfrac{1}{8}\left(\dfrac{1}{ac} + \dfrac{1}{\sqrt{c}}\right)  \]
	Tương tự ta có $ \begin{cases}
	\dfrac{1}{b^3c+2a^2+1} \le \dfrac{1}{8}\left(\dfrac{1}{ba} + \dfrac{1}{\sqrt{a}}\right)  \\
	\dfrac{1}{c^3a+2b^2+1} \le \dfrac{1}{8}\left(\dfrac{1}{ab} + \dfrac{1}{\sqrt{b}}\right) 
	\end{cases} $.\\
	Cộng theo vế các BĐT trên lại ta có $ VT \le \dfrac{1}{8}\left(\dfrac{1}{ab}+\dfrac{1}{bc} + \dfrac{1}{ac} + \dfrac{1}{\sqrt{a}}+\dfrac{1}{\sqrt{b}} + \dfrac{1}{\sqrt{c}} \right) $.\\
	Mà theo BĐT $ BCS $ thì $$ \dfrac{1}{a^4} + \dfrac{1}{b^4} + \dfrac{1}{c^4} \ge \dfrac{1}{3}\left( \dfrac{1}{a^2} + \dfrac{1}{b^2} + \dfrac{1}{c^2} \right)^2  \ge \dfrac{1}{27}\left(\dfrac{1}{a} + \dfrac{1}{b} + \dfrac{1}{c}\right)^4 \ge \dfrac{1}{243}\left(\dfrac{1}{\sqrt{a}} + \dfrac{1}{\sqrt{b}} + \dfrac{1}{\sqrt{c}}\right)^8 $$
	Suy ra $ \dfrac{1}{\sqrt{a}} + \dfrac{1}{\sqrt{b}} + \dfrac{1}{\sqrt{c}} \le 3 $. Tương tự:
	\[ \dfrac{1}{a^4} + \dfrac{1}{b^4} + \dfrac{1}{c^4} \ge \dfrac{1}{3}\left( \dfrac{1}{a^2} + \dfrac{1}{b^2} + \dfrac{1}{c^2} \right)^2 \ge \dfrac{1}{3}\left( \dfrac{1}{ab} + \dfrac{1}{bc} + \dfrac{1}{ca} \right)^2  \]
	Suy ra $ \dfrac{1}{ab} + \dfrac{1}{bc} + \dfrac{1}{ca} \le 3 $. Từ đó ta có $ VT \le \dfrac{3+3}{8} = \dfrac{3}{4} $.\\ 
	Đẳng thức xảy ra khi chỉ khi $ a = b= c = 1 $.
}
\end{ex}
