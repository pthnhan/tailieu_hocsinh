\begin{ex}%[Đề thi thử Toán 9 năm học 2017-2018, THPT Chuyên KHTN, Hà Nội, Vòng 1]%[9D0G2]
 Giải phương trình $$x^2+1+\sqrt{x^2+x+2}=2x+\sqrt{3x+1}.$$
	\loigiai{
Điều kiện: $x\ge -\dfrac{1}{3}.$ Khi đó, phương trình đã cho có thể viết lại thành $$x^2+x+2+\sqrt{x^2+x+2}=3x+1+\sqrt{3x+1}.$$ Đặt $u=\sqrt{x^2+x+2}\ge 0$ và $v=\sqrt{3x+1}\ge 0$, ta được $$u^2+u=v^2+v,$$ hay $$(u-v)(u+v+1)=0.$$ Từ đây, do $u+v+1>0$ nên $u-v=0$, hay $x^2+x+2=3x+1$. Giải phương trình này ta được $x=1$.\\ Đối chiếu điều kiện, ta thấy phương trình đã cho có nghiệm duy nhất $x=1$.
		
	}
\end{ex}

\begin{ex}%[Đề thi thử Toán 9 năm học 2017-2018, THPT Chuyên KHTN, Hà Nội, Vòng 1]%[9D0G2]
	 Giải hệ phương trình $$\begin{cases}x^2+4y^2=5 \\ 5x+10y+4x^2y+8y^2x=27. \end{cases}$$
	\loigiai{
Phương trình thứ hai của hệ có thể viết lại thành \[(x+2y)(5+4xy)=27.\tag{1}\]Thay $5=x^2+4y^2$ vào $(1)$, ta được $(x+2y)^3=27,$ hay $x+2y=3$. Suy ra $xy=1$.\\ Giải hệ phương trình $\begin{cases} x+2y=3\\xy=1\end{cases}$, ta được hai nghiệm $(x;y)$ là $(1;1)$ và $\left(2;\dfrac{1}{2}\right).$
	}
\end{ex}


\begin{ex}%[TT-THPT Chuyên KHTN, Hà Nội năm 2017-2018, Vòng 1]%[9D0G4]
	Tìm tích của tất cả các ước số nguyên dương phân biệt của số $n=(420)^4.$
	\loigiai{
Ta có $n=\left(2^2\cdot 3\cdot 5\cdot 7\right)^4$ nên ước của $n$ có dạng $m=2^a3^b5^c7^d$ với $0\le a\le 8$ nhận $9$ giá trị, $0\le b\le 4,$\,$0\le c\le 4,$\,$0\le d\le 4$ với mỗi số $b,c,d$ nhận $5$ giá trị suy ra $n$ có $9\cdot 5^3=1125$ ước dương. \\Ta thấy, nếu $m\neq 420^2$ thì $\dfrac{420^4}{m}$ cũng là ước suy ra $m$ và $\dfrac{420^4}{m}$ là cặp ước có tích bằng $420^4$. Ta có $\dfrac{1125-1}{2}=562$ cặp ước như vậy.\\ Do đó, tích của tất cả các ước nguyên dương phân biệt của $n$ bằng $\left(420^4\right)^{562}\cdot 420^2=420^{2250}$.
	}
\end{ex}

\begin{ex}%[TT-THPT Chuyên KHTN, Hà Nội năm 2017-2018, Vòng 1]%[9D0G7]
	 Với $a,b,c>0$ và $\min (ab,bc,ca)\ge 1.$ Chứng minh rằng $$(1+a^2)(1+b^2)(1+c^2)\le \left(1+\left(\dfrac{a+b}{2}\right)^2\right)\left(1+\left(\dfrac{b+c}{2}\right)^2\right)\left(1+\left(\dfrac{c+a}{2}\right)^2\right).$$
	\loigiai{Ta có $$(1+a^2)(1+b^2)=(ab-1)^2+(a+b)^2\le \left(\left(\dfrac{a+b}{2}\right)^2-1\right)^2+(a+b)^2=\left(\left(\dfrac{a+b}{2}\right)^2+1\right)^2.$$ Một cách tương tự, ta cũng có $$(1+b^2)(1+c^2)\le \left(\left(\dfrac{b+c}{2}\right)^2+1\right)^2.$$ và $$(1+c^2)(1+a^2)\le \left(\left(\dfrac{c+a}{2}\right)^2+1\right)^2.$$ Nhân các vế bất đẳng thức ta có điều phải chứng minh.
		
	}
\end{ex}


\begin{ex}%[Đề thi thử Toán 9 năm học 2017-2018, THPT Chuyên KHTN, Hà Nội, Vòng 1]%[9H0G1]
	Cho tam giác nhọn $ABC$ nội tiếp đường tròn $(O)$ với $BA>BC$. Phân giác ngoài góc $\angle ABC$ cắt đường thẳng qua $A$ song song với $BC$ tại $P$.
	\begin{enumerate}[1)]
	\item Chứng minh rằng $AP=AB.$
	\item Tiếp tuyến qua $A$ của $(O)$ cắt $PB$ tại $Q$. $BP$ cắt $(O)$ tại $M$ khác $B$. Chứng minh rằng $$MA^2=MQ\cdot MP.$$
	\item Gọi $R$ đối xứng với $Q$ qua $AC$. Chứng minh rằng $\angle APR=\angle CPB.$
	\end{enumerate}
	\loigiai{\begin{center}
	\begin{tikzpicture}
	\tkzInit[xmin=-4.8,ymin=-1.2,xmax=5.5,ymax=9.5]
	\tkzClip 
	\tkzDefPoints{2/2/O}
	\tkzDrawCircle[R](O,3 cm) 
	\tkzDefShiftPoint[O](100:3){A}
	\tkzDefShiftPoint[O](-130:3){B}
	\tkzDefShiftPoint[O](-45:3){C}
	\tkzFindAngle(C,B,A) \tkzGetAngle{ang}%đn số đo góc
	\tkzDefPointBy[rotation = center A angle -\ang](B)
\tkzGetPoint{P}
\tkzInterLC(P,B)(O,B)\tkzGetSecondPoint{M}
\tkzTangent[at=A](O) và \tkzGetPoint{c}
\tkzInterLL(A,c)(P,B)\tkzGetPoint{Q} 
\tkzDefPointBy[reflection = over A--C](P)
\tkzGetPoint{R}
	\tkzDrawPolygon(A,B,C)
	\tkzDrawSegments[](A,P P,B P,R C,R A,Q O,A C,Q C,P C,M A,M A,R)
	\tkzDrawLines[add=0 and 1](C,B)
	\tkzLabelLine[pos=2,below](C,B){$t$}
	\tkzDrawPoints[color=black](O,A,B,C,P,M,Q,R)
	\tkzLabelPoints[above left](A,P)
	\tkzLabelPoints[below left](B,M,Q)
	\tkzLabelPoints[](C,O,R)
	\tkzMarkRightAngles(Q,A,O)
	\end{tikzpicture}
	\end{center}
	\begin{enumerate}[1)]
	\item Ta có $\angle BPA=\angle PBt$ (so le trong). Mà $\angle PBt=\angle PBA$ ($BP$ là tia phân giác của góc ngoài $\angle ABC$). Do đó, $\angle APB=\angle PBA$ nên tam giác $APB$ cân tại $A$. Suy ra $AP=AB$ (điều phải chứng minh).
	\item Hai tam giác $MAQ$ và $MPA$ có $\angle AMP$ chung.\\ Để ý $AQ$ là tiếp tuyến của $(O)$ nên $\angle QAM=\angle ABM$ (cùng chắn cung $\wideparen{AM}$).\\Kết hợp với chứng minh trên, $\triangle APB$ cân tại $A$ ta suy ra $\angle QAM=\angle MPA.$\\ Do đó, $\triangle MAQ\backsim \triangle MPA$, suy ra $MA^2=MQ\cdot MP$ (điều phải chứng minh).
	\item Từ $R$ đối xứng $Q$ qua $AC$ nên 
	\begin{align*}
		\angle{PAR}=360^\circ-\angle{PAC}-\angle{CAR}&=180^\circ +\angle{ACB}-\angle{CAQ}\\&=180^\circ +\angle{ACB}-(\angle{CAM}+\angle{MAQ}).\tag{1}
	\end{align*}
 Để ý $\angle CAM=\angle MBt$ (tứ giác $AMBC$ nội tiếp) và $\angle MAQ=\angle MBA $ nên ta viết tiếp đẳng thức $(1)$ như sau  
	\[\angle PAR=\angle ACB +\angle MBC-\angle MBA=\angle ACB+\angle ABC=180^\circ-\angle BAC.\tag{2}\]
	Mặt khác, $\angle BAC=\angle BMC$ nên \[180^\circ-\angle BAC=180^\circ-\angle BMC=\angle PMC.\tag{3}\]
	Từ $(2)$ và $(3)$ suy ra $\angle PAR=\angle PMC.$
\\ Kết hợp với $\dfrac{AP}{AR}=\dfrac{AP}{AQ}=\dfrac{MP}{MA}=\dfrac{MP}{MC}$, ta suy ra $\triangle PAR\backsim \triangle PMC$, do đó $\angle{APR}=\angle{CPB}.$
	\end{enumerate}
		
	}
\end{ex}

\begin{ex}%[Đề thi thử Toán 9 năm học 2017-2018, THPT Chuyên KHTN, Hà Nội, Vòng 1]%[9D0G4]
	Giả sử số nguyên dương $n$ có tính chất: có tồn tại một cách xếp $a_1,a_2,\ldots ,a_{2n}$ của $2n$ số $1,1,2,2\ldots ,n,n$ sao cho với mỗi $k=1,2,\ldots ,n$ luôn tồn tại đúng $k$ số xếp giữa hai số $k$. Chứng minh rằng $n^2+n$ chia hết cho $4$.
	\loigiai{Giả sử $a_1,a_2,\ldots ,a_{2n}$ là một hoán vị thỏa mãn yêu cầu của bài toán, khi đó với mọi $k=1,2,\ldots ,n$ luôn tồn tại chỉ số $i_k$ mà $a_{i_k}=a_{i_k+k+1}=k$. Ta tính tổng các chỉ số bằng $2$ cách.
	\begin{enumerate}[\textbf{Cách }1.]
	\item $$1+2+\cdots +2n=n(2n+1).$$
	\item $$\sum_{k=1}^n (i_k+i_k+k+1)=2\sum_{k=1}^n i_k+\dfrac{n(n+3)}{2}$$
	\end{enumerate}
	Suy ra $2\displaystyle\sum_{k=1}^n i_k=n(2n+1)-\dfrac{n(n+3)}{2}=\dfrac{n(3n-1)}{2}$ nên $n(3n-1)\equiv 0\pmod 4$.\\ Ta có $n^2+n=4n^2-n(3n-1)$ suy ra $n^2+n$ chia hết cho $4$ (đpcm).
		
	}
\end{ex}
