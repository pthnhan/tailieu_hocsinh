	Cho hàm số $y=-3x^2+6x+1$ có đồ thị là parabol $(P)$. Lập bảng biến thiên của hàm số này và tìm tọa độ đỉnh $I$ của $(P)$.
	\loigiai{

	Tập xác định: $\mathscr{D}=\mathbb{R}$\\
	Tọa độ đỉnh: $I(1;2)$\\
	Chiều biến thiên: Do $a<0$ nên hàm số đồng biến trên $(-\infty;1)$ và nghịch biến trên $(1;+\infty)$.
	\\
Bảng biến thiên:

\begin{bt}%[Đề HKI, THPT Marie Curie, TP. HCM 2018]%[0D2B1]
	Tìm tập xác định của hàm số $y=\dfrac{4x-1}{(3x-x^2)(5-\sqrt{x-2})}$.
	\loigiai{
	Điều kiện: $\heva{& 3x-x^2\neq 0\\ & 5-\sqrt{x-2}\neq 0}\Leftrightarrow \heva{& x\neq 0\\& x\neq 3\\& x\neq 27\\& x\geq 2}$.\\
	Vậy tập xác định của hàm số đã cho là $\mathscr{D}=[2;+\infty)\backslash \{3;27\}$.
		}
\end{bt}
\begin{bt}%[Đề HKI, THPT Marie Curie, TP. HCM 2018]%[0D3K2]
	Giải phương trình $2x^2+\sqrt{x^2-3x+5}=6x+11$.
	\loigiai{
	$2x^2+\sqrt{x^2-3x+5}=6x+11\Leftrightarrow 2(x^2-3x+5)+\sqrt{x^2-3x+5}-21=0$ (*).\\
	Đặt $t=\sqrt{x^2-3x+5}, t\geq 0$. PT (*) trở thành $2t^2+t-21=0\Leftrightarrow \hoac{& t=3 \text{ (thỏa mãn) }\\&t=-\dfrac{7}{2}\text{ (loại) }}$.\\
Với $t=3$, ta có $\sqrt{x^2-3x+5}=3\Leftrightarrow x^2-3x-4=0\Leftrightarrow \hoac{&x=-1\\&x=4}$.\\	
	Vậy phương trình đã cho có hai nghiệm $x=-1;x=4$.
		}
\end{bt}
\begin{bt}%[Đề HKI, THPT Marie Curie, TP. HCM 2018]%[0D3K3]
	Giải hệ phương trình $\heva{&\sqrt{2y^2+x+y}=3\\&y^2+5y+2x=12}$.
	\loigiai{
	$\heva{&\sqrt{2y^2+x+y}=3\\&y^2+5y+2x=12}\Leftrightarrow \heva{&2y^2+x+y=9\quad (1)\\&y^2+5y+2x=12\quad (2)}$.\\
	$(1)\Leftrightarrow x=-2y^2-y+9$ thế vào $(2)$ ta có:\\
	$y^2+5y+2(-2y^2-y+9)=12\Leftrightarrow y^2-y-2=0\Leftrightarrow \hoac{&y=-1\\&y=2}$.\\
	Với $y=-1\Rightarrow x=8$.\\
	Với $y=2\Rightarrow x=-1$.\\
	Vậy hệ phương trình đã cho có hai nghiệm $(8;-1)$ và $(-1;2)$.
		}
\end{bt}
\begin{bt}%[Đề HKI, THPT Marie Curie, TP. HCM 2018]%[0D3K2]
	Cho phương trình $\dfrac{m^2x+1}{2}-\dfrac{m^2x+3}{3}=\dfrac{m^2-4m+9x}{6}$. Tìm tất cả giá trị của tham số $m$ để phương trình có tập nghiệm là $\mathbb{R}$.
	\loigiai{
	$\dfrac{m^2x+1}{2}-\dfrac{m^2x+3}{3}=\dfrac{m^2-4m+9x}{6}\Leftrightarrow (m^2-9)x-m^2+4m-3=0$.\\
	Suy ra phương trình đã cho có tập nghiệm là $\mathbb{R}\Leftrightarrow \heva{&m^2-9=0\\&m^2-4m+3=0}\Leftrightarrow \heva{&\hoac{&m=-3\\&m=3}\\&\hoac{&m=1\\&m=3}}\Leftrightarrow m=3$.\\ 
	Kết luận: $m=3$.
		}
\end{bt}
\begin{bt}%[Đề HKI, THPT Marie Curie, TP. HCM 2018]%[0D3G2]
	Cho phương trình $x^2-4mx-m+5=0$. Tìm tất cả các giá trị của tham số $m$ để phương trình có hai nghiệm phân biệt $x_1$ và $x_2$ thỏa mãn hệ thức $(2x_1+3)(2x_2+3)=59$.
	\loigiai{
	Phương trình đã cho có hai nghiệm phân biệt $x_1, x_2\Leftrightarrow\Delta=4m^2+m-5>0$ (*).\\
	Khi đó, theo định lý Viet, ta có $\heva{&x_1+x_2=4m\\&x_1x_2=5-m}$.\\
	Suy ra $(2x_1+3)(2x_2+3)=59\Leftrightarrow 6(x_1+x_2)+4x_1x_2-50=0\Leftrightarrow 24m+4(5-m)-50=0\Leftrightarrow m=\dfrac{3}{2}$.\\
	Kết hợp với (*), ta có $m=\dfrac{3}{2}$ là giá trị cần tìm.
		}
\end{bt}
\begin{bt}%[Đề HKI, THPT Marie Curie, TP. HCM 2018]%[0H2K2]
	Trong mặt phẳng tọa độ $Oxy$, cho hai điểm $A(-1;2)$ và $B(5;4)$.
	\begin{enumerate}
	\item[a)] Chứng minh $3$ điểm $A, B$ và $M(-4;1)$ là $3$ điểm thẳng hàng.
	\item[b)] Tìm tọa độ điểm $P$ nằm trên trục $Ox$ sao cho tam giác $ABP$ vuông tại $P$.
	\item[c)] Tìm tọa độ điểm $E$ nằm trên parabol $(P): y=x^2+5$ để $E$ cách đều hai điểm $A$ và $B$. 
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item[a)]
	Ta có $\vec{AB}=(6;2), \vec{AM}=(-3;-1)\Rightarrow \vec{AB}=-2\vec{AM}\Rightarrow \vec{AB}$ và $\vec{AM}$ cùng phương. Suy ra $A, B, M$ thẳng hàng.  
	\item[b)] 
	$P\in Ox\Rightarrow P(x;0)$.\\
	$\vec{AP}=(x+1;-2), \vec{BP}=(x-5;-4)$.\\
	Do đó $\Delta ABP$ vuông tại $P\Leftrightarrow \vec{AP}\cdot\vec{BP}=0\Leftrightarrow (x+1)(x-5)+8=0\Leftrightarrow x^2-4x+3=0$\\
	$\Leftrightarrow \hoac{&x=1\\&x=3}$. Vậy $P(1;0)$ hoặc $P(3;0)$.
	\item[c)]
	$E\in (P):y=x^2+5\Rightarrow E(x_E;x_E^2+5), \vec{AE}=(x_E+1;x_E^2+3),\vec{BE}=(x_E-5;x_E^2+1)$.\\
	$E$ cách đều $A$ và $B\Leftrightarrow AE=BE\Leftrightarrow \sqrt{(x_E+1)^2+(x_E^2+3)^2}=\sqrt{(x_E-5)^2+(x_E^2+1)^2}$\\
	$\Leftrightarrow x_E^2+2x_E+1+x_E^4+6x_E^2+9=x_E^2-10x_E+25+x_E^4+2x_E^2+1$\\
	$\Leftrightarrow x_E^2+3x_E-4=0\Leftrightarrow \hoac{&x_E=1\\&x_E=-4}$.\\
	Vậy $E(1;6)$ hoặc $E(-4;21)$.
	\end{enumerate}
		}
\end{bt}
\begin{bt}%[Đề HKI, THPT Marie Curie, TP. HCM 2018]%[0D3K2]
	Cho phương trình $(m^2+3)(x-1)-m=(3-2m)x-5$. Tìm tất cả các giá trị của tham số $m$ để phương trình có một nghiệm duy nhất và nghiệm này là số nguyên.
	\loigiai{
	$(m^2+3)(x-1)-m=(3-2m)x-5\Leftrightarrow (m^2+2m)x=m^2+m-2$.\\
	Phương trình đã cho có nghiệm duy nhất khi và chỉ khi $m^2+2m\neq 0\Leftrightarrow m\neq 0$ và $m\neq -2$ (*).\\
	Khi đó nghiệm duy nhất là $x=\dfrac{m^2+m-2}{m^2+2m}=\dfrac{(m-1)(m+2)}{m(m+2)}=\dfrac{m-1}{m}=1-\dfrac{1}{m}$.\\
	Suy ra nghiệm duy nhất này là một số nguyên khi và chỉ khi $m$ là ước của $1\Leftrightarrow m=\pm 1$ (thỏa mãn (*)). \\
	Vậy $m=-1$ hoặc $m=1$.
		}
\end{bt}
