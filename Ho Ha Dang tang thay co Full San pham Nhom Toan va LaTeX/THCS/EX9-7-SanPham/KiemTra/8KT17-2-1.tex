\begin{ex}%[Amsterdam Hà nội, 2016]%[8D2B9]
Cho biểu thức $A= \left ( \dfrac{x^3+1}{x^2-1} -\dfrac{x^2-1}{x-1} \right ) : \left ( x+\dfrac{x}{x-1}\right )$
\begin{enumerate}
\item[a)] Tìm điều kiện của $x$ để biểu thức $A$ có nghĩa. Rút gọn $A$. 
\item[b)] Tìm $x$ để $A=3$. 
\item[c)]  Tìm $x$ nguyên sao cho $A$ nhận giá trị nguyên.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a)] Điều kiện $x \neq \pm 1$ và $x \neq 0$. \\
$A= \left ( \dfrac{x^2-x+1}{x-1} -x-1 \right ) : \dfrac{x^2}{x-1} = \dfrac{2-x}{x-1} : \dfrac{x^2}{x-1} = \dfrac{2-x}{x^2}$. 
\item[b)]   
$A=3 \iff 3x^2 +x-2=0 \iff (x+1)(3x-2) =0 \iff  x= \dfrac{2}{3}$
\item[c)]   $2-x$ chia hết cho $x$ suy ra $2$ chia hết cho $x$, mặt khác $x \neq \pm 1$ cho nên $x= \pm 2$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Amsterdam Hà nội, 2016]%[8D1B9]
Phân tích các đa thức sau thành nhân tử.
\begin{enumerate}
\item[a)] $x^4-x^2-4xy-4y^2$.
\item[b)] $(x-1)(x-2)(x+7)(x+8) +8 $.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a)] $x^4-x^2-4xy-4y^2 = x^4 - (x+2y)^2 = (x^2+x+2y)(x^2-x-2y)$.
\item[b)] $(x-1)(x-2)(x+7)(x+8) +8 = (x^2+6x-16)(x^2+6x-7) +8 = (x^2+6x-7)^2 -9(x^2+6x-7) + 8$ 
$=(x^2+6x-8)(x^2+6x-15)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Amsterdam Hà nội, 2016]%[8D1B5]
    Cho các số thực $x,y$ thỏa mãn $x+y=1$ và $x^3+y^3=2$. Tính giá trị của các biểu thức.
\begin{enumerate}
\item[a)] $M=xy$.
\item[b)] $N=x^5+y^5$
\end{enumerate}
\loigiai{ 
\item[a)] $M=xy = \dfrac{(x+y)^3 - x^3-y^3}{3(x+y)} = \dfrac{1-2}{3} = -\dfrac{1}{3}$.
\item[b)] $N=x^5+y^5 = (x^2+y^2)(x^3+y^3) - x^2y^2(x+y) = 2(x^2+y^2) - \dfrac{1}{9} = 2 [(x+y)^2 -2xy] - \dfrac{1}{9} = \dfrac{29}{9}$
}
\end{ex}

\begin{ex}%[Amsterdam Hà nội, 2016]%[8H1K3]%[8H1K6]%[8H2K6]
Cho hình vuông $ABCD$ có $AC$ cắt $BD$ tại $O$. Gọi $E,F$ theo thứ tự là các điểm đối xứng với $O$ qua $AD$ và $BC$.
\begin{enumerate}
\item[a)] Chứng minh rằng các tứ giác $AODE$, $BOCF$ là hình  vuông.
\item[b)]  Nối $EC$ cắt $DF$ tại $I$. Chứng minh rằng $OI \perp CD$.
\item[c)] Biết diện tích của hình lục giác $ABFCDE$ bằng $6$. Tính độ dài cạnh của hình vuông $ABCD$. 
\item[d)] Lấy $K$ là một điểm bất kì trên cạnh $BC$. Gọi $G$ là trọng tâm của tam giác $AIK$. Chứng minh rằng điểm $G$ thuộc một đường thẳng cố định khi $K$ di chuyển trên cạnh $BC$. 
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[thick]
\tkzDefPoints{1/3/A,5/3/B,5/-1/C}
\tkzDefPointBy[translation = from B to A](C) \tkzGetPoint{D}
\tkzInterLL(A,C)(B,D) \tkzGetPoint{O} 
\tkzDefBarycentricPoint(B=3,C=1) \tkzGetPoint{K}
\tkzDefPointBy[reflection = over A--D](O) \tkzGetPoint{E}
\tkzDefPointBy[reflection = over B--C](O) \tkzGetPoint{F}
\tkzInterLL(E,C)(D,F) \tkzGetPoint{I}
\tkzDefMidPoint(A,I) \tkzGetPoint{M}
\tkzDefBarycentricPoint(M=2,B=1) \tkzGetPoint{Q}
\tkzDefBarycentricPoint(M=2,C=1) \tkzGetPoint{P}
\tkzDefBarycentricPoint(M=2,K=1) \tkzGetPoint{G}
\tkzDrawSegments(A,E E,D D,C C,F F,B B,A A,D D,B A,C C,B E,C D,F A,I I,K K,A M,C M,B M,K P,Q)
\tkzDrawPoints[fill=magenta,color=black,size=5](A,B,C,D,E,F,M,I,K,P,Q,G,O)
\tkzLabelPoints[below](D,C,P,I)
\tkzLabelPoints[above](A,B,Q)
\tkzLabelPoints[right](O,K,F)
\tkzLabelPoints[left](M,E)
\tkzLabelPoints[below right = -4pt](G)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item[a)] Ta có tam giác $ADO,BCO$ vuông cân tại $O$ và $O$ đối xứng với $E$ qua $AD$, đối xứng với $F$ qua $BC$ cho nên các tứ giác $AODE$, $BOCF$ là hình thang vuông.
\item[b)] Dễ thấy tứ giác $EDCF$ là hình thang cân với các cạnh đáy là $EF$, $DC$ cho nên các tam giác $ODC$, $IDC$ cân tại $O$ và $I$,  suy ra $IO \perp CD$. 
\item[c)] Diện tích lục giác bằng tổng diện tích của hình vuông $ABCD$ và diện tích các tam giác $ADE$, $BCF$ cho nên diện tích của hình vuông  bằng $\dfrac{2}{3}$ diện tích của lục giác, suy ra diện tích của hình vuông $ABCD$ bằng $4$.  Vậy độ dài cạnh của hình vuông $ABCD$ bằng $2$.
\item[d)] Gọi $M,P,Q$ lần lượt là trung điểm $AI$, trọng tâm các tam giác $ABI,ACI$. \\
Ta có $\dfrac{MQ}{QB} = \dfrac{MG}{GK} = \dfrac{MP}{PC} = \dfrac{1}{2}$, cho nên $P,Q,G$ thẳng hàng.
Vậy $G$ thuộc đường thẳng cố định $PQ$. 
\end{enumerate}
}
\end{ex}

\begin{ex}%[Amsterdam Hà nội, 2016]%[8D2K9]
Cho ba số $a,b,c$ đôi một khác nhau. Chứng minh rằng giá trị của biểu thức sau không phụ thuộc vào $a,b,c$.
\begin{center}
$P=\dfrac{a^2}{(a-b)(a-c)} + \dfrac{b^2}{(b-a)(b-c)} + \dfrac{c^2}{(c-a)(c-b)}$
\end{center}
\end{ex}
\loigiai{
$P=\dfrac{a^2}{(a-b)(a-c)} + \dfrac{b^2}{(b-a)(b-c)} + \dfrac{c^2}{(c-a)(c-b)} = -\dfrac{a^2(b-c)+b^2(c-a)+ c^2(a-b)}{(a-b)(b-c)(c-a)}$. Xét đa thức $Q= a^2(b-c)+b^2(c-a)+ c^2(a-b)$, nếu $a=b$ hoặc $b=c$ hoặc $c=a$ thì $Q=0$ cho nên $Q=\alpha(a-b)(b-c)(c-a)$, cho $a=1,b=2,c=3$, ta có $-2 = 2 \alpha \Longrightarrow \alpha =-1$. Thử lại ta thấy $-(a-b)(b-c)(c-a) = a^2(b-c)+b^2(c-a)+ c^2(a-b)$\\
Vậy $P= \dfrac{(a-b)(b-c)(c-a)}{(a-b)(b-c)(c-a)} = 1$.  
}
