\noindent\textbf{II. PHẦN TỰ LUẬN}
\setcounter{ex}{0}
\begin{ex}%[Đề kiểm tra học kỳ I, THCS Đại Kim, 2017 - 2018]%[Lê Minh Đức, dự án 7EX-1-2018]%[7D1Y3]%[7D1Y9]%[7D1Y3]%[7D1B11]
	Thực hiện phép tính (bằng cách hợp lí nếu có thể):
	\begin{listEX}[2]
		\item $\dfrac{1}{2} \cdot \dfrac{8}{3}-\dfrac{-1}{9}$
		\item $\dfrac{7}{13}-1\dfrac{2}{29}-\dfrac{7}{13}+1,25+\dfrac{2}{29}$
		\item $\dfrac{-13}{8} \cdot \dfrac{7}{19}+\dfrac{5}{8} \cdot \dfrac{7}{19}$
		\item $-2^2\sqrt{(-3)^2+4^2}-1\dfrac{1}{3}:2\dfrac{2}{3}$
	\end{listEX}
	\loigiai{
		Ta có
		\begin{enumerate}
			\item $\dfrac{1}{2} \cdot \dfrac{8}{3}-\dfrac{-1}{9}=\dfrac{4}{3}+\dfrac{1}{9}=\dfrac{13}{9}$.
			\item $\dfrac{7}{13}-1\dfrac{2}{29}-\dfrac{7}{13}+1,25+\dfrac{2}{29}=\left(\dfrac{7}{13}-\dfrac{7}{13}\right)+\left(\dfrac{2}{29}-1\dfrac{2}{29}\right)+1{,}25=0-1+1{,}25=0{,}25$ .
			\item $\dfrac{-13}{8} \cdot \dfrac{7}{19}+\dfrac{5}{8} \cdot \dfrac{7}{19}=\dfrac{7}{19}\left(\dfrac{-13}{8}+\dfrac{5}{8}\right)=\dfrac{-7}{19}$.
			\item $-2^2\sqrt{(-3)^2+4^2}-1\dfrac{1}{3}:2\dfrac{2}{3}=-4 \cdot \sqrt{25}-\dfrac{4}{3}:\dfrac{8}{3}=-20-\dfrac{1}{2}=\dfrac{-41}{2}$.
		\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề kiểm tra học kỳ I, THCS Đại Kim, 2017 - 2018]%[Lê Minh Đức, dự án 7EX-1-2018]%[7D1B2]%[7D1B3]%[7D1Y6]%[7D1Y7]
	Tìm $x$ biết
	\begin{listEX}[2]
		\item $ x+\dfrac{2}{5}=\dfrac{3}{10}$.
		\item $\dfrac{1}{6}x-3=\dfrac{-2}{3}$.
		\item $\left(\dfrac{1}{5}-x\right)^2=\dfrac{16}{9}$.
		\item $\dfrac{x-1}{6}=\dfrac{x+3}{5}$.
	\end{listEX}
	\loigiai{
		\begin{enumerate}
			\item \begin{eqnarray*}
				x+\dfrac{2}{5}&=&\dfrac{3}{10}\\
				x&=&\dfrac{3}{10}-\dfrac{2}{5}\\
				x&=&\dfrac{-1}{10}.
			\end{eqnarray*}
			\item \begin{eqnarray*}
				\dfrac{1}{6}x-3 & = & \dfrac{-2}{3}\\
				\dfrac{1}{6}x & = & \dfrac{-2}{3}+3\\
				\dfrac{1}{6}x & = & \dfrac{7}{3}\\
				x & = & \dfrac{7}{3}:\dfrac{1}{6}\\
				x & = & 14.
			\end{eqnarray*}
			\item $\left(\dfrac{1}{5}-x\right)^2=\dfrac{16}{9}$
			\begin{itemize}
				\item \textbf{Trường hợp 1:} $\dfrac{1}{5}-x=\dfrac{4}{3} \Leftrightarrow x=\dfrac{1}{5}-\dfrac{4}{3} \Leftrightarrow x=\dfrac{-17}{15}$.
				\item \textbf{Trường hợp 2:} $\dfrac{1}{5}-x=\dfrac{-4}{3} \Leftrightarrow x=\dfrac{1}{5}+\dfrac{4}{3} \Leftrightarrow x=\dfrac{23}{15}$.
			\end{itemize}
			\item \begin{eqnarray*}
				\dfrac{x-1}{6} & = & \dfrac{x+3}{5}\\
				(x-1) \cdot 5 & = & (x+3) \cdot 6\\
				5x-5 & = & 6x+18\\
				6x-5x & = & -5-18\\
				x & = & -23.
			\end{eqnarray*}
		\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề kiểm tra học kỳ I, THCS Đại Kim, 2017 - 2018]%[Lê Minh Đức, dự án 7EX-1-2018]%[7D2B1]
	Cho biết để hoàn thành một công việc trong $10$ giờ thì cần $48$ người làm. Hỏi nếu chỉ có $40$ người thì làm mất bao nhiêu thời gian để hoàn thành công việc đó? (Năng suất làm việc của mỗi công nhân là như nhau)
	\loigiai{
		Gọi thời gian để hoàn thành công việc với $40$ người là $x$ ($x > 10$).\\
		Vì thời gian làm việc là số người là $2$ đại lượng tỉ lệ thuận với nhau nên ta có
		\[ \dfrac{48}{40} = \dfrac{10}{x} \text{ hay } \dfrac{6}{5}=\dfrac{10}{x}. \]
		Suy ra $6x=50 \Rightarrow x=12$ (thỏa mãn điều kiện).\\
		Vậy với $40$ người thì cần $12$ giờ để hoàn thành công việc.
	}
\end{ex}
\begin{ex}%[Đề kiểm tra học kỳ I, THCS Đại Kim, 2017 - 2018]%[Lê Minh Đức, dự án 7EX-1-2018]%[7H2B5]%[7H2K5]%[7H2K5]
	Cho góc nhọn $mOn$, $Ot$ là tia phân giác của góc $mOn$. Trêm tia $Om$ và $On$ lấy điểm lần lượt hai điểm $C$ và $D$ sao cho $OC = OD$. Đoạn thẳng $CD$ cắt $Ot$ tại $P$.
	\begin{enumerate}
		\item Chứng minh $\triangle OCP = \triangle ODP$. 
		\item Chứng minh $CP = DP$ và $OP \perp CD$.
		\item Trên tia $Ot$ lấy điểm $Q$ sao cho $P$ là trung điểm $OQ$. Chứng minh $CQ\parallel OD$.
	\end{enumerate}
	\loigiai{
		\immini
		{
			\begin{enumerate}
				\item[a)] Xét $\triangle OCP$ và $\triangle ODP$ có
				\[ \heva{& OC=OD \text{ (giả thiết)}\\& \widehat {COP} = \widehat {DOP} \text{ ($Ot$ là tia phân giác)}\\& OP \text{ chung}.}\]
				Suy ra $\triangle OCP = \triangle ODP$ (cạnh - góc - cạnh), suy ra điều phải chứng minh.
			\end{enumerate}
		}
		{
			\begin{tikzpicture}[line join=round, line cap=round,>=stealth]
			\tikzset{label style/.style={font=\footnotesize}}
			\tkzDefPoints{0/0/O,5.2/0/n,2.8/0/D}
			\tkzDefPointBy[ rotation = center O angle 40](n) \tkzGetPoint{t}
			\tkzDefPointBy[ rotation = center O angle 40](t) \tkzGetPoint{m}
			\coordinate (P) at ($(O)!(D)!(t)$);
			\tkzInterLL(O,m)(D,P) \tkzGetPoint{C}
			\tkzDefLine[parallel=through C](O,D)\tkzGetPoint{4} %Vẽ đường thẳng song song
			\tkzInterLL(C,4)(O,t) \tkzGetPoint{Q}
			\tkzDrawPoints[fill=black](O,C,D,P,Q)
			\tkzDrawSegments(O,n O,t O,m D,C C,Q)
			%\tkzLabelPoints[above right](C,E)
			\tkzLabelPoints[below left](O,D,n)
			\tkzLabelPoints[left](C,m)
			\tkzLabelPoints[right](P,Q,t)
			\tkzMarkRightAngles[size=0.2](C,P,O)
			\tkzMarkSegments[mark=|,size=2,thin](O,C O,D)
			\end{tikzpicture}
		}
		\begin{enumerate}
			\item[b)] Vì $\triangle OCP = \triangle ODP$ (chứng minh trên) nên
			\[ \heva{& CP = DP \text{ (hai cạnh tương ứng)}\\& \widehat {CPO} = \widehat {DPO} \text{ (hai góc tương ứng)}.} \]
			Mà $\widehat {CPO}$ và $\widehat {DPO}$ là hai góc kề bù nên
			\[ \widehat {CPO} + \widehat {DPO} = 180^\circ. \]
			Suy ra $\widehat {CPO} = \widehat {DPO} = 90^\circ$ hay $OP \perp CD$.
			\item[c)] Xét $\triangle PCQ$ và $\triangle PDO$ có
			\[ \heva{& PQ=PO \text{ (giả thiết)}\\& \widehat {CPQ} = \widehat {DPO} \text{ (đối đỉnh)}\\& CP=DP \text{ chứng minh trên}.}\]
			Suy ra $\triangle PCQ = \triangle PDO$ (cạnh - góc - cạnh).\\
			$\triangle \widehat {PQC} = \widehat {POD}$ (hai góc tương ứng của hai tam giác bằng nhau).\\
			Mà hai góc này ở vị trí so le trong nên $CQ\parallel OD$ (dấu hiệu nhận biết hai đường thẳng song song).
		\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề kiểm tra học kỳ I, THCS Đại Kim, 2017 - 2018]%[Lê Minh Đức, dự án 7EX-1-2018]%[7D1G6]
	Cho năm số tự nhiên $a$, $b$, $c$, $d$, $e$ thỏa mãn $a^b=b^c=c^d=d^e=e^a$. Chứng minh rằng năm số $a$, $b$, $c$, $d$, $e$ bằng nhau.
	\loigiai{
		\textbf{Nhận xét.} \textit{Trong $2$ lũy thừa bằng nhau thì lũy thừa có cơ số nhỏ hơn sẽ có số mũ lớn hơn}.\\
		Xét $2$ số $a$ và $b$, nếu $a < b$ thì $c < b$ vì $a^b=b^c$.\\
		Từ $c <b$ mà $ b^c = c^d$ suy ra $ c < d$.\\
		Tương tự ta có $d<e$ và $e<a$.\\
		Suy ra $a<b<c<d<e<a$ (không thỏa mãn).\\
		Vậy $a$ không lớn hơn $b$. \hfill $(1)$\\
		Tương tự ta chứng minh được $b$ không lớn hơn $a$. \hfill $(2)$\\
		Từ $(1)$ và $(2)$ suy ra $a=b$.\\
		Chứng minh tương tự ta có $b=c$, $c=d$, $d=e$ hay $a=b=c=d=e$.\\
		Suy ra điều phải chứng minh.
	}
\end{ex}
