	\noindent\textbf{II. PHẦN TỰ LUẬN}
	\setcounter{ex}{0}
		\begin{ex}%[Đề kiểm tra học kỳ I, THCS Dịch Vọng, 2017 - 2018]%[Phạm Tuấn Hùng, dự án 7EX-1-2018]%[7D1B9]
			Thực hiện phép tính:
			\begin{listEX}[2]
				\item $\dfrac{2}{5}\cdot 15\dfrac{1}{3}-\dfrac{2}{5}\cdot 10\dfrac{1}{3}$
				\item $\left( -\dfrac{1}{2}\right)^2 +6\div \left( -\dfrac{1}{2}\right) + \sqrt{25}- \left( -2011 \right) ^0$
			\end{listEX}
			\loigiai
			{
				Ta có
				\begin{enumerate}
					\item 
								$\dfrac{2}{5}\cdot 15\dfrac{1}{3}-\dfrac{2}{5}\cdot 10\dfrac{1}{3}
								=\dfrac{2}{5}\left( 15\dfrac{1}{3}-10\dfrac{1}{3}\right)
								=\dfrac{2}{5}\left( \dfrac{46}{3}-\dfrac{31}{3}\right)
								=\dfrac{2}{5}\cdot \dfrac{15}{3}
								=\dfrac{2}{5}\cdot 5
								=2$.
					\item 
								$\left( -\dfrac{1}{2}\right)^2 +6 : \left( -\dfrac{1}{2}\right) + \sqrt{25}- \left( -2011 \right) ^0
								=\dfrac{1}{4}+6\cdot \dfrac{-2}{1}+5-1
								=\dfrac{1}{4}-12+5-1
								=\dfrac{1}{4}-8
								=\dfrac{1}{4}-\dfrac{32}{4}
								=-\dfrac{31}{4}$.	
				\end{enumerate}
			}
		\end{ex}
		
		\begin{ex}%[Đề kiểm tra học kỳ I, THCS Dịch Vọng, 2017 - 2018]%[Phạm Tuấn Hùng, dự án 7EX-1-2018]%[7D1K9]
			Tìm $x$, $y$ biết
			\begin{listEX}[3]
				\item $\dfrac{x}{7}=\dfrac{y}{8}$ và $x+y=30$;
				\item $(x-1)^3=27$;
				\item $2-\left|x+\dfrac{2}{3} \right|=0 $.
			\end{listEX}
			\loigiai
			{
				\begin{enumerate}
					\item
							Áp dụng tính chất dãy tỉ số bằng nhau ta có
							\[\dfrac{x}{7}=\dfrac{x}{8}=\dfrac{x+y}{7+8}=\dfrac{30}{15}=2.\]
							Vậy $\dfrac{x}{7}=2 \Rightarrow x=7\cdot 2=14$ và $\dfrac{y}{8}=2 \Rightarrow y=8\cdot 2=16$.
					\item 
						\begin{align*}
							(x-1)^3&=27\\
							x-1&=3\\
							x &= 4.
						\end{align*}
					\item 
						\begin{align*}
							2-\left|x+\dfrac{2}{3} \right|&=0\\
							\left|x+\dfrac{2}{3} \right|&=2.
						\end{align*}
						\begin{itemize}
							\item TH1: $x + \dfrac{2}{3} = 2$ ta suy ra $x = \dfrac{4}{3}$.
							\item TH2: $x + \dfrac{2}{3} = -2$ ta suy ra $x = -\dfrac{8}{3}$.
						\end{itemize}
				\end{enumerate}
			}
		\end{ex}
	
		\begin{ex}%[Đề kiểm tra học kỳ I, THCS Dịch Vọng, 2017 - 2018]%[Phạm Tuấn Hùng, dự án 7EX-1-2018]%[7D2B1]
		Biết số học sinh giỏi học kì I của $3$ khối $6$; $7$; $8$ của một trường THCS tỉ lệ với $8$; $9$; $10$. Hãy tính số học sinh giỏi của mỗi khối $6$; $7$; $8$. Biết số học sinh giỏi của cả $3$ khối là $270$ em. 
			\loigiai
			{
				Gọi $a$; $b$; $c$ lần lượt là số học sinh giỏi của khối $6$; $7$; $8$. ($a,b,c\in \mathbb{N}$).\\
				Theo đề bài ta có $\dfrac{a}{8}=\dfrac{b}{9}=\dfrac{c}{10}$ và $a+b+c=270$.\\
				Áp dụng tính chất của dãy tỉ số bằng nhau ta có
				\[\dfrac{a}{8}=\dfrac{b}{9}=\dfrac{c}{10}=\dfrac{a+b+c}{8+9+10}=\dfrac{270}{27}=10.\]
				Vậy $a=8\cdot 10= 80$; $b=9\cdot 10=90$ và $c=10\cdot 10=100$.\\ Vậy số học sinh các khối $6$; $7$; $8$ lần lượt là $80$; $90$; $100$. 
			}
		\end{ex}
		\begin{ex}%[Đề kiểm tra học kỳ I, THCS Dịch Vọng, 2017 - 2018]%[Phạm Tuấn Hùng, dự án 7EX-1-2018]%[7H2B9]
		Cho $\triangle ABC$, $E$ là trung điểm của $BC$. Lấy $D$ thuộc tia đối của tia $EA$ sao cho $ED=EA$.
		\begin{enumerate}
			\item Chứng minh rằng $\triangle AEB=\triangle DEC$.
			\item Chứng minh rằng $AC\parallel BD$.
			\item Kẻ $EI\perp AC$ $(I\in AC)$; $EK\perp BD$ $(K\in BD)$. Chứng minh $\triangle AIE = \triangle DKE$.
			\item Chứng minh $3$ điểm $I$, $E$, $K$ thẳng hàng.
		\end{enumerate}
		\loigiai
		{
			\immini
			{
				\begin{enumerate}
						\item Xét $\triangle AEB$ và $\triangle DEC$ có
								\begin{itemize}
									\item $BE=CE$ ($E$ là trung điểm $BC$).
									\item $\widehat{AEB}=\widehat{DEC}$ (đối đỉnh).
									\item $AE=DE$ (giả thiết).
								\end{itemize}
							Vậy $\triangle AEB=\triangle DEC$ (c-g-c).
							\item Xét $\triangle AEC$ và $\triangle DEB$ có
							\begin{itemize}
								\item $BE=CE$ ($E$ là trung điểm $BC$).
								\item $\widehat{AEC}=\widehat{DEB}$ (đối đỉnh).
								\item $AE=DE$ (giả thiết).
							\end{itemize}
				\end{enumerate}
			}
			{
			\begin{tikzpicture}[scale=1, line join=round, line cap=round, >=stealth]
			%Phan Khai Bao Mac Dinh
			\tkzInit[ymin=-4.5,ymax=4.5,xmin=-0.5,xmax=6.5]
			%\tkzAxeXY
			%\tkzGrid
			\tkzClip
			\tkzSetUpPoint[color=black,fill=black,size=6]
			%Dinh Hinh
			\tkzDefPoints{2/4/A,6/0/C,0/0/B}
			\tkzDrawPolygon(A,B,C)
			\tkzDefMidPoint(B,C)\tkzGetPoint{E}
			\tkzDefPointBy[rotation= center E angle 180](A) \tkzGetPoint{D}
			
			\tkzMarkSegments[mark=|](B,E C,E)
			\tkzMarkSegments[mark=||](A,E D,E)
			\tkzDrawAltitude(A,C)(E)\tkzGetPoint{I}
			\tkzDrawAltitude(B,D)(E)\tkzGetPoint{K}
			\tkzMarkRightAngles[fill= gray!20](E,K,D E,I,A)
			\tkzDrawSegments(A,D B,D C,D)
			\tkzDrawPoints(A,B,C,D,I,K,E)
			%GanLabel
			\tkzLabelPoints[above](A)
			\tkzLabelPoints[below](D)
			\tkzLabelPoints[left](B)
			\tkzLabelPoints[right](C)
			\tkzLabelPoints[above right](I)
			\tkzLabelPoints[above left]()
			\tkzLabelPoints[below right](E)
			\tkzLabelPoints[below left](K)
			\end{tikzpicture}
		}
			Vậy $\triangle AEC=\triangle DEB$ (c-g-c).
		Ta có $\widehat{CAE}=\widehat{BDE}$ (hai góc tương ứng trong $\triangle AEC=\triangle DEB$) mà hai góc này nằm ở vị trí sole-trong nên $AC\parallel BD$.
			\begin{enumerate}
				\setcounter{enumi}{2}
				\item Xét $\triangle AIE$ vuông tại $I$ và $\triangle DKE$ vuông tại $K$ có
				\begin{itemize}
					\item $\widehat{CAE}=\widehat{BDE}$ (hai góc tương ứng trong $\triangle AEC=\triangle DEB$) 
					\item $AE=DE$ (giả thiết).
				\end{itemize}
				Vậy $\triangle AIE=\triangle DKE$ (ch-gn).
				\item Ta có $\heva{&\widehat{AEI}+\widehat{IED}=180^\circ\\&\widehat{AEI}=\widehat{DEK}}\Rightarrow \widehat{IED}+\widehat{DEK}=180^\circ$. Vậy $I$, $E$, $K$ thẳng hàng.
			\end{enumerate}
	}
	\end{ex}

	Tìm các số $a$, $b$ biết
	\[|5a-6b+300|^{2011}+(2a-3b)^{2010}=0.\]
	\loigiai{
		Ta có $\heva{&|5a-6b+300|^{2011}\geq 0\\&(2a-3b)^{2010}\geq 0}$. Vậy $\heva{&5a-6b+300=0\\&2a-3b=0}\Rightarrow \heva{&a=-300\\&b=-200}$.
	}
\end{ex}
