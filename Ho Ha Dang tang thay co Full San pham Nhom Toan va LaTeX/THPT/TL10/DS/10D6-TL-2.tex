\subsection{Tóm tắt lí thuyết}
\subsubsection{Định nghĩa}
\immini{\begin{itemize}
\item $\sin \alpha = \overline{OK}$.
\item $\cos \alpha = \overline{OH}$.
\item $\tan \alpha = \dfrac{\sin \alpha}{\cos \alpha}$ nếu $\cos \alpha \neq 0$.
\item $\cot \alpha = \dfrac{\cos \alpha}{\sin \alpha}$ nếu $\sin \alpha \neq 0$.
\end{itemize}
Các giá trị $\sin \alpha,\cos \alpha, \tan \alpha, \cot \alpha$ được gọi là các giá trị lượng giác của cung $\alpha$.\\
Ta cũng gọi trục tung là \textbf{trục sin}, còn trục hoành là \textbf{trục cosin}.

\subsubsection{Hệ quả}
\begin{enumerate}
\item $\sin \alpha$ và $\cos \alpha$ xác định với mọi $\alpha \in \mathbb{R}$, hơn nữa
\begin{itemize}
\item $\sin (\alpha+k2\pi)= \sin \alpha, \forall k \in \mathbb{Z}$.
\item $\cos (\alpha+k2\pi)= \cos \alpha, \forall k \in \mathbb{Z}$.
\end{itemize}
\item $-1 \le \sin \alpha \le 1$ và $-1 \le \cos \alpha \le 1$.
\item Với mọi $m \in \mathbb{R}$ mà $-1 \le m \le 1$ đều tồn tại $\alpha, \beta$ sao cho $\sin \alpha = m$ và $\cos \beta = m$.
\item $\tan \alpha$ xác định với mọi $\alpha \neq \dfrac{\pi}{2}+k\pi, k \in \mathbb{Z}$.
\item $\cot \alpha$ xác định với mọi $\alpha \neq k\pi, k \in \mathbb{Z}$.

\subsubsection{Công thức lượng giác cơ bản}
\begin{itemize}
\item $\sin ^2 \alpha + \cos^2 \alpha = 1$.
\item $1+\tan^2 \alpha = \dfrac{1}{\cos^2 \alpha}, \quad \alpha \neq \dfrac{\pi}{2}+k\pi, k \in \mathbb{Z}$.
\item $1+\cot^2 \alpha = \dfrac{1}{\sin^2 \alpha}, \quad \alpha \neq k\pi, k \in \mathbb{Z}$.
\item $\tan \alpha \cdot \cot \alpha = 1, \quad \alpha \neq \dfrac{k\pi}{2}, k \in \mathbb{Z}$.
\end{itemize}

	\item $A=\sin 50^\circ \cdot \cos (-100^\circ)$.
	\item $B=\sin 195^\circ \cdot \tan \dfrac{20\pi}{7}$.

\loigiai{
\begin{enumerate}
	\item $A=\sin 50^\circ \cdot \cos (-100^\circ)$.\\
	Ta có: điểm cuối của cung $50^\circ$ thuộc góc phần tư thứ $\mathit{I}$ nên $\sin 50^\circ>0$.\\
	Điểm cuối của cung $-100^\circ$ thuộc góc phần tư thứ $\mathit{III}$ nên $\cos (-100^\circ)<0$.\\
	Do đó, $A<0$.
	\item $B=\sin 195^\circ \cdot \tan \dfrac{20\pi}{7}$.\\
	Ta có: điểm cuối của cung $195^\circ$ thuộc góc phần tư thứ $\mathit{III}$ nên $\sin 195^\circ<0$.\\
	Điểm cuối của cung $\dfrac{20\pi}{7}=\dfrac{6\pi}{7}+2\pi$ thuộc góc phần tư thứ $\mathit{II}$ nên $\tan \dfrac{20\pi}{7}<0$.\\
	Do đó, $B>0$.
\end{enumerate}
}

\begin{dang}{Rút gọn biểu thức và chứng minh đẳng thức}
Một số hệ thức hay dùng trong bài toán rút gọn biểu thức hoặc chứng minh đẳng thức:
\begin{itemize}
\item $\sin ^2 \alpha + \cos^2 \alpha = 1$.
\item $1+\tan^2 \alpha = \dfrac{1}{\cos^2 \alpha}, \quad \alpha \neq \dfrac{\pi}{2}+k\pi, k \in \mathbb{Z}$.
\item $1+\cot^2 \alpha = \dfrac{1}{\sin^2 \alpha}, \quad \alpha \neq k\pi, k \in \mathbb{Z}$.
\item $\tan \alpha \cdot \cot \alpha = 1, \quad \alpha \neq \dfrac{k\pi}{2}, k \in \mathbb{Z}$.
\end{itemize}
\end{dang}
