\section{THỨ TỰ THỰC HIỆN CÁC PHÉP TÍNH}
\subsection{Tóm tắt lý thuyết}
\subsubsection{Đối với biểu thức không có dấu ngoặc}
\begin{itemize}
	\item Nếu chỉ có cộng, trừ hoặc chỉ có nhân, chia ta thực hiện phép tính từ trái sang phải.
	\item Nếu có phép tính cộng, trừ, nhân, chia, nâng lên lũy thừa ta thực hiện phép tính nâng lên lũy thừa trước, rồi đến nhân và chia, cuối cùng là cộng và trừ: $$\text{Lũy thừa}\ \rightarrow\ \text{Nhân và chia}\ \rightarrow \text{Cộng và trừ}.$$
\end{itemize}
\subsubsection{Đối với biểu thức có dấu ngoặc}
Nếu biểu thức có các dấu ngoặc: ngoặc tròn $(\ )$, ngoặc vuông $[\ ]$, ngoặc nhọn $\{\ \}$, ta thực hiện phép tính trong dấu ngoặc tròn trước, rồi thực hiện phép tính trong dấu ngoặc vuông, cuối cùng thực hiện phép tính trong dấu ngoặc nhọn. $$(\ )\ \rightarrow\ [\ ]\ \rightarrow\ \{\ \}.$$

		\end{enumEX}
	}
\end{vd}

	\end{array}

		\end{array}
		\]
	}
\end{vd}

\begin{bt}%[Lê Thanh Hải, dự án EX-C2-L6-2018 Lần 1]%[6D1K9]
	Một quyển sách có $200$ trang. Để đánh số các trang của quyển sách này phải dùng tất cả bao nhiêu chữ số?
	\loigiai{
		Số trang có một chữ số là $9$ (trang).\\
		Số trang có hai chữ số là $99-9=90$ (trang).\\
		Số trang có ba chữ số là $200-(9+90)=101$ (trang).\\
		Số chữ số cần dùng là 
		$$9\cdot 1 +90\cdot 2+101\cdot 3=492\quad \text{(chữ số)}.$$
	}
\end{bt}

\begin{bt}%[Lê Thanh Hải, dự án EX-C2-L6-2018 Lần 1]%[6D1G9]

		$$900-(9\cdot 1+90\cdot 2)=900-189=711\quad \text{(chữ số)}.$$
		Số trang có ba chữ số là $711:3=237$ (trang).\\
		Số trang của quyển sách là $99+237=336$ (trang).
	}
\end{bt}

\section{TÍNH CHẤT CHIA HẾT CỦA MỘT TỔNG}
\subsection{Tóm tắt lý thuyết}
\begin{tc}
	Nếu $a\,\vdots\, m$; $b\,\vdots\, m$; $c\,\vdots\, m$ thì $(a+b+c)\,\vdots\, m$; $(a+b-c)\,\vdots\, m$.
\end{tc}
\begin{tc}
	Nếu $a\not\vdots\, m$; $b\,\vdots\, m$; $c\,\vdots\, m$ thì $(a+b+c)\not \,\vdots\, m$.
\end{tc}
\begin{tc}
	Nếu $a,b \in \mathbb{N}$ và $a\,\vdots\, m$ thì $a\cdot b \,\vdots\, m$.
\end{tc}
\subsection{Các dạng bài tập và phương pháp giải}
\begin{dang}{Xét tính chia hết của một tổng, một hiệu, một tích}
	Áp dụng tính chất 1, tính chất 2 hoặc tính chất 3 về tính chia hết của một tổng (hiệu) hay tích để xét.
\end{dang}

\begin{vd}%[Lê Thanh Hải, dự án EX-C2-L6-2018 Lần 1]%[6D1B10]
	Áp dụng tính chất chia hết, xét xem mỗi tổng hoặc hiệu sau có chia hết cho $6$ không?
	\begin{enumEX}{1}
		\item $60+24+36$.
		\item $48-12$.
		\item $15+42-30$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item $60\,\vdots\, 6$; $24\,\vdots\, 6$; $36 \,\vdots\, 6$. Suy ra $(60+24+36)\,\vdots\, 6$ (tính chất 1).
			\item $48\,\vdots\, 6$; $12\,\vdots\, 6$. Suy ra $(48-12)\,\vdots\, 6$ (tính chất 1).
			\item $15\not \,\vdots\, 6$; $42\,\vdots\, 6$; $30\,\vdots\, 6$. Suy ra $(15+42-30)\not \,\vdots\, 6$ (tính chất 2).
		\end{enumEX}
	}
\end{vd}
\begin{vd}%[Lê Thanh Hải, dự án EX-C2-L6-2018 Lần 1]%[6D1B10]
	Áp dụng tính chất chia hết của một tích, xét xem các tích sau có chia hết cho $8$ không ?
	\begin{enumEX}{1}
		\item $8\cdot 431$.
		\item $40\cdot 2015$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Vì $8\,\vdots\, 8$ nên $8\cdot 431 \,\vdots\, 8$ (tính chất 3).
			\item Vì $40\,\vdots\, 8$ nên $40\cdot 2015 \,\vdots\, 8$ (tính chất 3).
		\end{enumEX}
	}
\end{vd}
\begin{vd}%[Lê Thanh Hải, dự án EX-C2-L6-2018 Lần 1]%[6D1B10]
	Trong các khẳng định sau, khẳng định nào đúng, khẳng định nào sai ?
	\begin{enumEX}{1}
		\item Hiệu $94-38$ không chia hết cho $2$.
		\item Tổng $5\cdot 41 +10\cdot 13$ chia hết cho $5$.
		\item Tổng $45+37+23$ không chia hết cho $5$.
	\end{enumEX}
	\loigiai{
		\begin{enumEX}{1}
			\item Sai vì $94\,\vdots\, 2$; $38\,\vdots\, 2$ nên $(94-38)\,\vdots\, 2$.
			\item Đúng vì $5\,\vdots\, 5$ nên $5\cdot 41 \,\vdots\, 5$; $10\,\vdots\, 5$ nên $10\cdot 13\,\vdots\, 5$. Do đó $5\cdot 41+10\cdot 13$ chia hết cho $5$.
			\item Sai vì $45\,\vdots\, 5$; $(37+23)=60\,\vdots\, 5$ nên $(45+37+23)\,\vdots\, 5$.
		\end{enumEX}
	}
\end{vd}
\begin{vd}%[Lê Thanh Hải, dự án EX-C2-L6-2018 Lần 1]%[6D1B10]
	Chia một số cho $15$ được số dư là $9$. Hỏi số đó có chia hết cho $3$ không? Có chia hết cho $5$ không?
	\loigiai{
		Gọi $a$ là số chia cho $15$ được số dư là $9$. Khi đó $a=15k+9$ ($k\in \mathbb{N}$).\\
		Ta thấy $15\,\vdots\, 3\Rightarrow 15k\,\vdots\, 3$; $9\,\vdots\, 3$ nên $15k+9\,\vdots\, 3$ tức là $a\,\vdots\, 3$.\\
		Mặt khác $15k\,\vdots\, 5$; $9\not \,\vdots\, 5$ nên $(15k+9)\not \,\vdots\, 5$ tức là $a\not \,\vdots\, 5$.}
\end{vd}
%
\begin{dang}{Tìm điều kiện của một số hạng để tổng hoặc hiệu chia hết cho một số}
	Áp dụng tính chất $1$, tính chất $2$.
\end{dang}

\begin{vd}%[Lê Thanh Hải, dự án EX-C2-L6-2018 Lần 1]%[6D1B10]
	Cho tổng $S=56+32-8+x$ với $x \in \mathbb{N}$. Tìm điều kiện của $x$ để
	\begin{enumerate}[a)]
		\item $S \,\vdots\, 8$.
		\item $S \not \,\vdots\, 4$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item Vì $56 \,\vdots\, 8, 32 \,\vdots\, 8, 8\,\vdots\, 8$ nên $56+32-8+x \,\vdots\, 8 \Leftrightarrow x \,\vdots\, 8$.
			\item Vì $56 \,\vdots\, 4, 32 \,\vdots\, 4, 8\,\vdots\, 4$ nên $56+32-8+x \not \,\vdots\, 4 \Leftrightarrow x \not \,\vdots\, 4$.
			
		\end{enumerate}
	}
\end{vd}
\begin{vd}%[Lê Thanh Hải, dự án EX-C2-L6-2018 Lần 1]%[6D1B10]
	Cho tổng $S=12+30+54+x$ với $x \in \mathbb{N}$. Điều kiện để $S \,\vdots\, 6$ là$\colon$
	\choice
	{$x$ có chữ số tận cùng là $6$}
	{$x$ là số chẵn}
	{$x$ là số chia hết cho $3$}

		Vì $12 \,\vdots\, 6, 30 \,\vdots\, 6, 54 \,\vdots\, 6$ nên $(12+30+54+x) \,\vdots\, 6 \Leftrightarrow x \,\vdots\, 6$.
	}
\end{vd}

\begin{bt}%[Lê Thanh Hải, dự án EX-C2-L6-2018 Lần 1]%[6D1B10]

\begin{bt}%[Lê Thanh Hải, dự án EX-C2-L6-2018 Lần 1]%[6D1B10]

	\begin{listEX}[2]
		\item $12 \cdot 173 +36 \cdot 41 +21$;
		\item $3 \cdot 241 \cdot 4 +24 \cdot k$ với $k \in \mathbb{N}$.
	\end{listEX}
	\loigiai{
		\begin{listEX}
			\item Ta có $12 \,\vdots\, 12$ nên $(12 \cdot 173) \,\vdots\, 12$; $36 \,\vdots\, 12$ nên $(36 \cdot 41) \,\vdots\, 12$; $21 \not \,\vdots\, 12$\\
			Suy ra $ (12 \cdot 173 +36 \cdot 41 +21) \not \,\vdots\, 12$.
			\item Ta có $3 \cdot 241 \cdot 4 = (12 \cdot 241) \,\vdots\, 12$, $24 \cdot k \,\vdots\,12$, suy ra $(3 \cdot 241 \cdot 4 +24 \cdot k) \,\vdots\, 12$.
		\end{listEX}
	}
\end{bt}
