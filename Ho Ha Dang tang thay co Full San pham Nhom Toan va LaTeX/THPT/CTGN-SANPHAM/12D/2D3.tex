\begin{dang}{Tích phân hàm $(kx+b)^{\alpha}$}
$\displaystyle\int\limits(kx+b)^{\alpha}\mathrm{\,d}x =\dfrac{1}{k} \cdot \dfrac{(kx+b)^{\alpha +1}}{\alpha +1}+C$  ($k \ne 0$ và $\alpha \ne -1$). 
\loigiai{
Đặt $t = kx+b \Rightarrow \mathrm{\,d}x = \dfrac{1}{k}  \mathrm{\,d}t .$\\
Ta có $\displaystyle\int\limits(kx+b)^{\alpha}\mathrm{\,d}x = \dfrac{1}{k} \displaystyle\int\limits t^{\alpha}\mathrm{\,d}t = \dfrac{1}{k} \cdot \dfrac{t^{\alpha +1}}{\alpha +1}+C=\dfrac{1}{k} \cdot \dfrac{(kx+b)^{\alpha +1}}{\alpha +1}+C.$
}

\end{dang}

 Đặt $t = 1-2x \Rightarrow \mathrm{\,d}x = \dfrac{1}{-2}  \mathrm{\,d}t $.\\

Cách 2:

}
\end{vd}

\begin{dang}{Tích phân hàm $\dfrac{1}{x^2-a^2}$}
$\displaystyle\int\limits\dfrac{1}{x^2-a^2}\mathrm{\,d}x = \dfrac{1}{2a}\ln\left|\dfrac{x-a}{x+a}\right| +C$

\loigiai{
Ta có $\dfrac{1}{x^2-a^2} = \dfrac{1}{2a}\left(\dfrac{1}{x-a}-\dfrac{1}{x+a}\right)$. Suy ra 
\begin{eqnarray*}
& \displaystyle\int\limits\dfrac{1}{x^2-a^2}\mathrm{\,d}x  & =\dfrac{1}{2a} \displaystyle\int\limits\left(\dfrac{1}{x-a}-\dfrac{1}{x+a} \right)\mathrm{\,d}x \\ 
&& = \dfrac{1}{2a}\left(\ln|x-a|-\ln|x+a|\right) +C = \dfrac{1}{2a}\ln\left|\dfrac{x-a}{x+a}\right| +C.
\end{eqnarray*}
}
\end{dang}

\begin{dang}{Tích phân hàm sin và hàm cosin của $kx+b$}
Cho $k \ne 0$, ta có
\begin{enumerate}
\item $\displaystyle\int\limits \sin (kx+b)\mathrm{\,d}x=-\dfrac{1}{k}\cos(kx+b)+C$.
\item $\displaystyle\int\limits \cos (kx+b)\mathrm{\,d}x=\dfrac{1}{k}\sin(kx+b)+C$.
\end{enumerate}
\loigiai{

\begin{enumerate}
\item   Đặt $t = kx+b \Rightarrow \mathrm{\,d}x = \dfrac{1}{k}  \mathrm{\,d}t $.\\
Ta có $\displaystyle\int\limits\sin (kx+b)\mathrm{\,d}x = \dfrac{1}{k} \displaystyle\int\limits\sin t \mathrm{\,d}t =- \dfrac{1}{k} \cdot \cos t+C=-\dfrac{1}{k}\cos(kx+b)+C.$

\item Đặt $t = kx+b \Rightarrow \mathrm{\,d}x = \dfrac{1}{k}  \mathrm{\,d}t $.\\
Ta có $\displaystyle\int\limits\cos (kx+b)\mathrm{\,d}x = \dfrac{1}{k} \displaystyle\int\limits\cos t \mathrm{\,d}t =- \dfrac{1}{k} \cdot \sin t+C=-\dfrac{1}{k}\cos(kx+b)+C.$
\end{enumerate}
}


\end{dang}

}

\section{Nguyên hàm và tích phân hàm vô tỉ, hàm đặc biệt}
\subsection{Các dạng toán và công thức giải nhanh}
\begin{dang}{Tích phân hàm lẻ}
Nếu $f(x)$ liên tục và là hàm lẻ trên $\left[-a;a\right],\ a>0$ thì $I=\displaystyle\int\limits_{-a}^{a}f(x)\mathrm{\,d}x=0$.\\
\end{dang}
\noindent
\emph{Chứng minh}: 	Ta có $I=\displaystyle\int\limits_{-a}^{a}f(x)\mathrm{\,d}x=\displaystyle\int\limits_{-a}^{0}f(x)\mathrm{\,d}x+\displaystyle\int\limits_{0}^{a}f(x)\mathrm{\,d}x$.\\
		Xét tích phân $I_1=\displaystyle\int\limits_{-a}^{0}f(x)\mathrm{\,d}x$.\\
		Đặt $x=-t\Rightarrow \mathrm{\,d}x=-\mathrm{\,d}t$\\

		Phương trình đường parabol $(P)\colon y=-x^2+4$.\\
		Diện tích cánh cửa cổng là $S=2\displaystyle \int \limits_0^2 (-x^2+4) \mathrm{\,d}x =2\left. \left( -\dfrac{x^3}{3}+4x \right)\right|_0^2=2\cdot \dfrac{16}{3} =\dfrac{32}{3}$.\\
		\textbf{Cách 2:} Dùng công thức nhanh.\\
		Diện tích cánh cửa cổng là $S=\dfrac{4}{3}\cdot 4 \cdot 2 =\dfrac{32}{3}$.
	}
\end{vd}
\begin{vd}%[Lương Thế Vinh, Hà Nội, Lần 3 - 2018]%[Nguyễn Thị Kiều Ngân]%[2D3G3-1]
	\immini{
	Cho parabol $(P_1)\colon y=-x^2+4$ cắt trục hoành tại hai điểm $A$, $B$ và đường thẳng $d\colon y=a$ $(0<a<4)$. Xét parabol $(P_2)$ đi qua $A$, $B$ và có đỉnh thuộc đường thẳng $y=a$. Gọi $S_1$ là diện tích hình phẳng giới hạn bởi $(P_1)$ và $d$, $S_2$ là diện tích hình phẳng giới hạn bởi $(P_2)$ và trục hoành. Biết $S_1=S_2$ (tham khảo hình vẽ bên). Tính $T=a^3-8a^2+48a$.
	\choice
	{$T=99$}
	{\True $T=64$}
	{$T=32$}
	{$T=72$}
	}{
	\begin{tikzpicture}[>=stealth, scale=1]
		\draw[domain=-2.2:2.2] plot (\x,{-(\x)^2+4});

		\draw[domain=-2.2:2.2] plot (\x,{-0.43*(\x)^2+1.72});

	\loigiai{
		Phương trình hoành độ giao điểm của $(P_1)$ và $d$ là
		$$-x^2+4=a \Leftrightarrow x=\pm \sqrt{4-a}.$$
		Hình phẳng giới hạn bởi $(P_1)$ và $d$ có chiều cao $4-a$, chiều rộng $2\sqrt{4-a}$ nên diện tích 
		$$S_1=\dfrac{4}{3}\cdot \sqrt{4-a}\cdot (4-a).$$
		Hình phẳng giới hạn bởi $(P_2)$ và $d$ có chiều cao $a$, chiều rộng $4$ nên diện tích
		$$S_2=\dfrac{4}{3}\cdot a\cdot 2=\dfrac{8}{3}a.$$
		Ta có $$S_1=S_2 \Leftrightarrow \left( \sqrt{4-a} \right)^3 =2a \Leftrightarrow 64-48a+12a^2-a^3=4a^2
		\Leftrightarrow a^3-8a^2+48a=64.$$
		Vậy $T=64$.

Chọn hệ trục tọa độ như hình vẽ. Phương trình đường tròn $(I,r)$ là $x^2+(y-R)^2=r^2$.\\
Suy ra $y=R\pm\sqrt{r^2-x^2}$. Từ đó ta có thể tích cần tìm là
\begin{center}
	$\begin{array}{ll}
V&= \pi\displaystyle\int\limits_{-r}^r \left((R+\sqrt{r^2-x^2})^2-(R-\sqrt{r^2-x^2})^2\right)\mathrm{\,d}x\\
&= \pi\displaystyle\int\limits_{-r}^r4R\sqrt{r^2-x^2}\mathrm{\,d}x\\
&= \pi\displaystyle\int\limits_{-\frac{\pi}{2}}^{\frac{\pi}{2}}4Rr\cos t\cdot r\cos t \mathrm{\,d}t\quad (x=r\sin t)\\
&= \pi\displaystyle\int\limits_{-\frac{\pi}{2}}^{\frac{\pi}{2}}4r^2R\dfrac{1+\cos 2t}{2}\mathrm{\,d}t\\
&=2\pi r^2R\left(t+\dfrac{1}{2}\sin 2t\right)\bigg|_{-\frac{\pi}{2}}^{\frac{\pi}{2}}\\
&=2\pi^2r^2R.
\end{array}$
\end{center}
Ta sẽ dùng công thức này ở dạng $V=\pi r^2\cdot 2\pi R =S\cdot P$ trong đó $S$ là diện tích hình tròn và $P=2\pi R$ là chu vi đường tròn tạo ra khi cho tâm $I$ quay quanh $Ox$.
\end{dang}
