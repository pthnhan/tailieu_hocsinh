	\item Diện tích xung quanh, diện tích toàn phần của hình nón cụt
	\begin{eqnarray*}
		S_\text{xq}&=&\pi(R+r)l \\
		S_\text{tp}&=&\pi(R+r)l+\pi R^2+\pi r^2
	\end{eqnarray*}
	\centerline{$R,\ r$ lần lượt là bán kính hai đáy, $l$ là độ dài đường sinh của hình nón cụt).}
	\item Thể tích hình nón cụt: 
	\[V=\dfrac{\pi}{3}h(R^2+r^2+Rr)\]
	\centerline{($h$ là đường cao của hình nón cụt).}
\end{itemize}

\begin{note} 
	Hình khai triển mặt xung quanh của một hình nón là một hình quạt.
\end{note}
\begin{note} 
	Một hình nón được xác định khi biết $2$ trong $3$ yếu tố: bán kính đáy, chiều cao, đường sinh. 
\end{note}

\subsection{Các ví dụ}
\begin{vd}
	Một hình nón có bán kính đáy bằng $r$, diện tích xung quanh gấp đôi diện tích đáy. Tính theo $r$
	\begin{enumerate}
		\item Diện tích xung quanh của hình nón;
		\item Thể tích của hình nón.
	\end{enumerate}

	{
		\immini

		\immini

\subsection{Luyện tập}
\begin{bt} 
	Cho tam giác $ABC$ vuông tại $A$, $\widehat{B}=60^\circ$ và $BC=2a\ (\mbox{đơn vị độ dài}).$ Quay xung quanh tam giác một vòng quanh cạnh huyền $BC$. Tìm diện tích xung quanh và thể tích hình tạo thành.
	\loigiai
	{
		\immini
