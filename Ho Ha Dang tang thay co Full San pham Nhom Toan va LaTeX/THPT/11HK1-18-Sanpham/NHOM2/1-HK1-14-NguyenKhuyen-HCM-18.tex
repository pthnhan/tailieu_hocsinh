\begin{bt}%[HK1 THPT Nguyễn Khuyến, HCM]%[Đào-V- Thủy DA 11HK1-18]%[1D1B3-1]
	Giải phương trình $\sin 3x\sin x=\cos 2x$.
	\loigiai{
	Ta có 
	\begin{align*}
	\sin 3x\sin x=\cos 2x&\Leftrightarrow \dfrac{1}{2}(\cos 2x-\cos 4x)=\cos 2x\\
	&\Leftrightarrow 2\cos^2 2x+\cos 2x-1=0\\
	&\Leftrightarrow \hoac{&\cos 2x=-1\\ &\cos 2x=\dfrac{1}{2}}\Leftrightarrow \hoac{&2x=\pi+k2\pi\\ &2x=\pm\dfrac{\pi}{3}+k2\pi}\Leftrightarrow \hoac{&x=\dfrac{\pi}{2}+k\pi\\ &x=\pm\dfrac{\pi}{6}+k\pi}\; (k\in\mathbb{Z}).
	\end{align*}
	Vậy phương trình đã cho có các nghiệm là $x=\dfrac{\pi}{2}+k\pi$, $x=\pm\dfrac{\pi}{6}+k\pi$ (với $k\in\mathbb{Z}$).
}
\end{bt}	

\begin{bt}%[HK1 THPT Nguyễn Khuyến, HCM]%[Đào-V- Thủy DA 11HK1-18]%[1D2B3-2]
	Tìm số hạng không chứa $x$ trong khai triển $\left(x^3-\dfrac{3}{x^3}\right)^{12}$ (với $x\neq 0$).
	\loigiai{
	Số hạng tổng quát trong khai triển $\left(x^3-\dfrac{3}{x^3}\right)^{12}$ (với $x\neq 0$) có dạng:
	$$\mathrm{C}^k_{12}\cdot \left(x^3\right)^{12-k}\cdot\left(\dfrac{3}{x^3}\right)^k=\mathrm{C}^k_{12}\cdot 3^k\cdot x^{36-6k}.$$
	Ta có $36-6k=0\Leftrightarrow k=6$.\\
	Vậy số hạng cần tìm là $\mathrm{C}^6_{12}\cdot 3^6=673596$.
}
\end{bt}

\begin{bt}%[HK1 THPT Nguyễn Khuyến, HCM]%[Đào-V- Thủy DA 11HK1-18]%[1D2B2-1]
	Có bao nhiêu số tự nhiên có ba chữ số khác nhau tạo ra từ các chữ số $1$, $2$, $3$, $4$, $5$, $6$, $7$, $8$, trong đó chữ số hàng chục là chữ số nhỏ nhất trong ba chữ số?
	\loigiai{
	\begin{itemize}
		\item Số các số tự nhiên có ba chữ số khác nhau tạo ra từ các chữ số $1$, $2$, $3$, $4$, $5$, $6$, $7$, $8$ là $\mathrm{A}^3_8=336$ (số).
		\item Ta thấy, mỗi số được lập đều có một trong ba khả năng: chữ số hàng trăm là nhỏ nhất, chữ số hàng chục là nhỏ nhất, chữ số hàng đơn vị là nhỏ nhất.
		\item Vậy số các số thỏa mãn bài toán là $\dfrac{336}{3}=112$ (số).
	\end{itemize}
}
\end{bt}

\begin{bt}%[HK1 THPT Nguyễn Khuyến, HCM]%[Đào-V- Thủy DA 11HK1-18]%[1D2B2-1]
	Trong một đoàn có $80$ đàn ông và $60$ phụ nữ. Nếu muốn tuyển chọn một phái đoàn gồm có $1$ ông trưởng đoàn, $1$ ông phó đoàn, $2$ nữ thư ký và $3$ thành viên. Hỏi có bao nhiêu cách chọn?
	\loigiai{
	\begin{itemize}
		\item Chọn $1$ ông trưởng đoàn, có $\mathrm{C}^1_{80}$ (cách).
		\item Chọn $1$ ông phó đoàn, có $\mathrm{C}^1_{79}$ (cách).
		\item Chọn $2$ nữ thư ký, có $\mathrm{C}^2_{60}$ (cách).
		\item Chọn $3$ thành viên, có $\mathrm{C}^3_{136}$ (cách).
	\end{itemize}
Vậy số cách chọn thỏa mãn yêu cầu bài toán là $\mathrm{C}^1_{80}\cdot \mathrm{C}^1_{79}\cdot \mathrm{C}^2_{60}\cdot \mathrm{C}^3_{136}=4586871456000$ (cách).
}
\end{bt}

\begin{bt}%[HK1 THPT Nguyễn Khuyến, HCM]%[Đào-V- Thủy DA 11HK1-18]%[1D2B5-2]
	Trong vườn hoa có $12$ bông hoa khác nhau gồm: $3$ bông trắng, $4$ bông xanh và $5$ bông vàng. Cần chọn ngẫu nhiên $4$ bông hoa để cắm vào $1$ bình hoa. Tính xác suất để chọn $4$ bông gồm đủ cả ba màu.
	\loigiai{
	Số cách chọn $4$ bông từ $12$ bông là $n(\Omega)=\mathrm{C}^4_{12}=495$ (cách).\\
	Gọi $A$ là biến cố: \lq\lq chọn được $4$ bông gồm đủ ba màu\rq\rq. Có ba trường hợp sau:
	\begin{itemize}
		\item Trường hợp 1: $2$ bông trắng, $1$ bông xanh, $1$ bông vàng. Có $\mathrm{C}^2_3\cdot \mathrm{C}^1_4\cdot \mathrm{C}^1_5$ (cách).
		\item Trường hợp 2: $1$ bông trắng, $2$ bông xanh, $1$ bông vàng. Có $\mathrm{C}^1_3\cdot \mathrm{C}^2_4\cdot \mathrm{C}^1_5$ (cách).
		\item Trường hợp 3: $1$ bông trắng, $1$ bông xanh, $2$ bông vàng. Có $\mathrm{C}^1_3\cdot \mathrm{C}^1_4\cdot \mathrm{C}^2_5$ (cách).
	\end{itemize}
	Do đó, ta được $n(A)=\mathrm{C}^2_3\cdot \mathrm{C}^1_4\cdot \mathrm{C}^1_5 + \mathrm{C}^1_3\cdot \mathrm{C}^2_4\cdot \mathrm{C}^1_5 + \mathrm{C}^1_3\cdot \mathrm{C}^1_4\cdot \mathrm{C}^2_5 =270$ (cách).\\
	Vậy xác suất cần tìm là 
	$$\mathrm{P}(A)=\dfrac{n(A)}{n(\Omega)}=\dfrac{270}{495}=\dfrac{6}{11}.$$
}
\end{bt}

\begin{bt}%[HK1 THPT Nguyễn Khuyến, HCM]%[Đào-V- Thủy DA 11HK1-18]%[1D2K5-4]
	Xác suất bắn trúng mục tiêu của hai xạ thủ lần lượt là $0{,}8$ và $0{,}9$. Hai người cùng bắn vào một mục tiêu một cách độc lập, mỗi người bắn một viên. Tính xác suất để cả hai người bắn trượt.
	\loigiai{
	Gọi $A_1$ là biến cố: \lq\lq Người thứ nhất bắn trượt\rq\rq. Ta có $\mathrm{P}(A_1)=1-0{,}8=0{,}2$.\\
	$A_2$ là biến cố: \lq\lq Người thứ hai bắn trượt\rq\rq. Ta có $\mathrm{P}(A_2)=1-0{,}9=0{,}1$.\\
	$A$ là biến cố: \lq\lq Cả hai người bắn trượt\rq\rq. Ta có $A=A_1A_2$.\\
	Vì $A_1$, $A_2$ là hai biến cố độc lập nên theo công thức nhân xác suất ta được
	$$\mathrm{P}(A)=\mathrm{P}(A_1)\cdot \mathrm{P}(A_2)=0{,}2\cdot 0{,}1=0{,}02.$$
}
\end{bt}

\begin{bt}%[HK1 THPT Nguyễn Khuyến, HCM]%[Đào-V- Thủy DA 11HK1-18]%[1D2K5-2]
	Một đoàn người gồm $30$ người Việt Nam đi du lịch bị lạc tại Châu Phi, biết rằng trong đoàn có $12$ người biết tiếng Anh, có $8$ người biết tiếng Pháp và có $17$ người chỉ biết tiếng Việt. Cần chọn ra $4$ người đi hỏi đường. Tính xác suất để có đúng $2$ người biết cả hai thứ tiếng Anh và Pháp.
	\loigiai{
	Số cách chọn $4$ người đi hỏi đường từ $30$ người là $n(\Omega)=\mathrm{C}^4_{30}$.\\
	Gọi $A$ là biến cố: \lq\lq Trong $4$ người được chọn, có đúng $2$ người biết cả hai thứ tiếng Anh và Pháp\rq\rq.\\
	Theo giả thiết, số người biết cả hai thứ tiếng Anh và Pháp là $12+8+17-30=7$ (người).\\
	Do đó $n(A)=\mathrm{C}^2_7\cdot \mathrm{C}^2_{23}$.\\
	Vậy xác suất cần tìm là 
	$$\mathrm{P}(A)=\dfrac{n(A)}{n(\Omega)}=\dfrac{\mathrm{C}^2_7\cdot \mathrm{C}^2_{23}}{\mathrm{C}^4_{30}}=\dfrac{253}{1305}\approx 0{,}194.$$
}
\end{bt}

\begin{bt}%[HK1 THPT Nguyễn Khuyến, HCM]%[Đào-V- Thủy DA 11HK1-18]%[1H2B1-2]%[1H2B1-4]%[1H2B3-2]%[1H2G3-5]
	Cho hình chóp $S.ABCD$ có $ABCD$ là hình thang đáy lớn $AB=2DC$. Gọi $G$ là trọng tâm tam giác $SAB$ và $O$ là giao điểm của $AC$ và $BD$.
	\begin{enumerate}
		\item Tìm giao tuyến của các cặp mặt phẳng $(SDC)$ và $(SAB)$, $(SAD)$ và $(SBC)$.
		\item Tìm thiết diện tạo bởi mặt phẳng $(GCD)$ và hình chóp. Thiết diện là hình gì?
		\item Chứng minh $GO$ song song với mặt phẳng $(SAD)$.
		\item Gọi $E$ là trung điểm của $BC$. Tìm giao điểm $H$ của $SB$ và mặt phẳng $(GDE)$. Tính $\dfrac{HB}{HS}$.
	\end{enumerate}
\loigiai{
	\begin{center}
		\begin{tikzpicture}
		\coordinate (A) at (0,0);
		\coordinate (B) at (6,0);
		\coordinate (D) at (1,-2);
		\coordinate (C) at (4,-2);
		\coordinate (S) at (2,5);
		\coordinate (Y) at (-1,5);

		\draw (S)--(A)--(I)--(E) (H)--(S);
		\draw (C)--(S)--(D) (C)--(N) (D)--(M) (Y)--(Z) (S)--(I) (E)--(K)--(H);
		\draw[dashed] (A)--(B) (C)--(D) (M)--(N) (A)--(C) (B)--(D) (G)--(O) (D)--(G)--(E)--(D) (S)--(J)--(I) (H)--(G) (E)--(B)--(H) (B)--(K);
		\foreach \p/\pos in {S/above, A/left, B/below, C/below right, D/below left, I/below, M/left, N/right, E/below, H/above right, K/right}\fill (\p) circle(1pt) node[\pos]{\p};
		\fill (O) node[shift={(120:10pt)}] {$O$} circle(1pt);

		\end{tikzpicture}
	\end{center}
\begin{enumerate}
\item 
	\begin{itemize}
		\item Xét hai mặt phẳng $(SDC)$ và $(SAB)$ có 
		$$\heva{&S\in (SDC)\cap (SAB)\\ &DC\subset (SDC), AB\subset (SAB), DC\parallel AB.}$$
		Do đó $(SDC)\cap (SAB)=\Delta$, trong đó $\Delta$ là đường thẳng đi qua $S$ và $\Delta\parallel AB$.
		\item Gọi $I$ là giao điểm của $AD$ và $BC$. Khi đó 
		$$\heva{&S\in (SAD)\cap (SBC)\\ &I\in (SAD\cap (SBC).}$$
		Suy ra $(SAD)\cap (SBC)=SI$.
	\end{itemize}
\item Xét hai mặt phẳng $(GCD)$ và $(SAB)$ có
$$\heva{&G\in (GCD)\cap (SAB)\\ &CD\subset (GCD), AB\subset (SAB), CD\parallel AB.}$$
Suy ra $(GCD)\cap (SAB)=MN$, trong đó $M\in SA$, $N\in SB$ và $MN\parallel AB\parallel CD$.\\
Khi đó, ta có $(GCD)\cap (SAD)=DM$, $(GCD)\cap (SBC)=CN$, $(GCD)\cap (ABCD)=CD$.\\
Vậy thiết diện cần tìm là hình thang $CDMN$.
\item Gọi $J$ là trung điểm của $AB$.\\
Xét $\triangle IAB$, ta có $CD$ là đường trung bình nên $C$, $D$ lần lượt là trung điểm của $IB$ và $IA$.\\
Do đó $O$ là trọng tâm $\triangle IAB$.\\
Khi đó $\dfrac{JG}{JS}=\dfrac{JO}{JI}=\dfrac{1}{3}\Rightarrow OG\parallel SI$. Mà $SI\subset (SAD)$. Vậy $OG\parallel (SAD)$. 
\item Gọi $K=DE\cap AB$, $H=SB\cap GK$. Ta được 
$$\heva{&H\in SB\\ &H\in GK, GK\subset (GDE)\Rightarrow H\in (GDE).}$$
Do đó $H=SB\cap (GDE)$.\\
Áp dụng định lý Menenaus trong $\triangle IAB$ ta có 
$$\dfrac{KB}{KA}\cdot \dfrac{DA}{DI}\cdot \dfrac{EI}{EB}=1\Rightarrow \dfrac{KB}{KA}\cdot 1\cdot 3=1\Rightarrow \dfrac{KB}{KA}=\dfrac{1}{3}\Rightarrow \dfrac{KB}{KJ}=\dfrac{1}{2}.$$
Áp dụng định lý Menenaus trong $\triangle SJB$ ta có 
$$\dfrac{HB}{HS}\cdot \dfrac{GS}{GJ}\cdot \dfrac{KJ}{KB}=1\Rightarrow \dfrac{HB}{HS}\cdot 2\cdot 2=1\Rightarrow \dfrac{HB}{HS}=\dfrac{1}{4}.$$
\end{enumerate}
