	\begin{ex}%[Đề thi HSG 9, tỉnh Bà Rịa Vũng Tàu, năm 2018]%[Lê Quân, Dự án EX-9G-18]%[9D0K7]
		Cho biểu thức $$P = x-\left(
		\dfrac{1}{\sqrt{x}-1} + \dfrac{x-\sqrt{x}-11}{x+\sqrt{x}-2}\right):\left(\dfrac{\sqrt{x}+1}{\sqrt{x}+2}-\dfrac{x-\sqrt{x}-4}{x+\sqrt{x}-2}\right).$$
		với $x\geq 0, x\ne 1.$
		\begin{enumerate}
			\item Rút gọn biểu thức $P$.
			\item Tìm giá trị của $x$ để biểu thức $P$ đạt giá trị nhỏ nhất.
		\end{enumerate}
		\loigiai{
			\begin{enumerate} 
				\item Quy đồng và rút gọn ta có 
				\begin{eqnarray*}
					P &=& x-\dfrac{x-9}{(\sqrt{x}-1)(\sqrt{x}+2)}:\dfrac{\sqrt{x}+3}{(\sqrt{x}-1)(\sqrt{x}+2)}\\
					&=& x-\dfrac{(\sqrt{x}-3)(\sqrt{x}+3)}{\sqrt{x}+3}\\
					&=& x-\sqrt{x}+3.
				\end{eqnarray*}
				\item Ta có $P=x-\sqrt{x}+3=\left(\sqrt{x}-\dfrac{1}{2}\right)^2+\dfrac{11}{4}\geq \dfrac{11}{4}$, dấu bằng xảy ra khi $\sqrt{x}-\dfrac{1}{2}=0$ hay $x=\dfrac{1}{4}.$ Vậy giá trị nhỏ nhất của $P$ là $\dfrac{11}{4}$ khi $x=\dfrac{1}{4}.$
			\end{enumerate}
		}
	\end{ex}
	\begin{ex}%[Đề thi HSG 9, tỉnh Bà Rịa Vũng Tàu, năm 2018]%[Lê Quân, Dự án EX-9G-18]%[9D0K2]
	Giải phương trình  $x^2-3x+\sqrt{2x-1}+1=0.$
	\loigiai{
	Điều kiện $x\geq \dfrac{1}{2}$ phương trình tương đương với
	\begin{eqnarray*}
		& & x^2-x + \dfrac{1}{4} - \left[ (2x-1) - \sqrt{2x-1} + \dfrac{1}{4} \right] =0\\
		& \Leftrightarrow  & \left(x-\dfrac{1}{2}\right)^2 - \left(\sqrt{2x-1}-\dfrac{1}{2}\right)^2 = 0\\
		& \Leftrightarrow & \hoac{&x = \sqrt{2x-1}\\ &\left(\sqrt{x -\dfrac{1}{2}}\right)^2 +\sqrt{2x-1} - \dfrac{1}{2} = 0}\\
		& \Leftrightarrow & \hoac{&(x-1)^2 = 0\\ &\sqrt{x-\dfrac{1}{2}}=\dfrac{-\sqrt{2}\pm 2}{2}}\\
		& \Leftrightarrow & \hoac{&x=1 \\ & x = 2-\sqrt{2}.}
	\end{eqnarray*}
	Vậy phương trình có tập nghiệm $S=\{ 1; 2-\sqrt{2}\}.$
	}
	\end{ex}
	\begin{ex}%[Đề thi HSG 9, tỉnh Bà Rịa Vũng Tàu, năm 2018]%[Lê Quân, Dự án EX-9G-18]%[9D0G2] 
	Giải hệ phương trình $\heva{&x^3+y^2+2x=0\\& y^3-x^2+2y=0}.$
	\loigiai{
	 Trừ hai vế hệ phương trình ta được phương trình tương đương với 
		\begin{eqnarray*}
		&&	x^3-y^3+y^2+x^2+2x-2y=0\\
		&\Leftrightarrow & (x-y)(x^2+xy+y^2+2)+x^2+y^2=0\\
		&\Leftrightarrow & (x-y)=\dfrac{-(x^2+y^2)}{\left(x+\dfrac{y}{2}\right)^2+\dfrac{3y^2}{4}+2} < 0\\
		&\Rightarrow& x < y.
		\end{eqnarray*}	
		Cộng hai vế của hệ phương trình ta được phương trình tương đương với
		\begin{eqnarray*}
		& & x^3+y^3+y^2-x^2+2(x+y)=0\\
		&\Leftrightarrow & (x+y)(x^2-xy+y^2+y-x+2)=0\\
		&\Leftrightarrow & (x+y)\left[\left(x-\dfrac{y}{2}\right)^2+\dfrac{3y^2}{4}+y-x+2\right] =0 \\
		&\Leftrightarrow & \hoac{&x=-y\\&\left(x-\dfrac{y}{2}\right)^2+\dfrac{3y^2}{4}+y-x+2=0.}
		\end{eqnarray*}
 		\begin{itemize}
		\item Với $x=-y$ ta được $y^3-y^2+2y=0 \Leftrightarrow y(y^2-y+2)=0 \Leftrightarrow y = 0 \Rightarrow x = 0.$

	 	\item Vì $x<y$ theo chứng minh trên nên ta có  $\left(x-\dfrac{y}{2}\right)^2+\dfrac{3y^2}{4}+y-x+2 = 0$ vô nghiệm.
		\end{itemize}
	Vậy phương trình có tập nghiệm $S=(0;0).$
		}
	\end{ex}	
	\begin{ex}%[Đề thi HSG 9, tỉnh Bà Rịa Vũng Tàu, năm 2018]%[Lê Quân, Dự án EX-9G-18]%[9H0G1]%[9H0G4]
		Cho tam giác $ABC$ nhọn $\left(AB<AC\right)$ có $M$ là trung điểm của $BC.$ Đường tròn $(O)$ đường kính $AM$ lần lượt cắt cạnh $AB, AC$ tại $P, Q$ khác $A.$
		\begin{enumerate}
			\item Chứng minh rằng $BP\cdot BA+CQ\cdot CA=\dfrac{1}{2}BC^2.$
			\item Đường thẳng qua $M$ và vuông góc với cạnh $BC$ cắt đường tròn $(O)$ tại điểm $K$ khác $M$. Chứng minh rằng $KP\cdot MQ=KQ\cdot MP.$
			\item Gọi $N$ là trung điểm của đoạn $PQ,$ đường tròn ngoại tiếp tam giác $MNK$ cắt đường thẳng $PQ$ tại điểm $S$ khác $N.$ Chứng minh $SM, SK$ là hai tiếp tuyến của đường tròn $(O).$
		\end{enumerate}
		\loigiai{
			\begin{center}
				\begin{tikzpicture}[scale = 0.8]
				\draw [thick, cyan] (2.25,1.25) circle (2.574cm);
				\draw (0,0)--(2,5)--(7,0)--(0,0); 
				\draw (4.5,2.5)--(8.88,-5.35)--(1.48,3.71);
				\draw (0,0)--(4.5,2.5)--(1,-1)--(2.25,1.25);
				\draw (1,-1)--(8.88,-5.38);
				\draw (2.25,1.25)--(2.99,1.85);
				\fill (0,0) circle (0.05) node [below left] {$A$};
				\fill (2,5) circle (0.05) node [above] {$B$};
				\fill (7,0) circle (0.05) node [below right] {$C$};
				\fill (1,-1) circle (0.05) node [below left] {$K$};
				\fill (4.5,2.5) circle (0.05) node [above right] {$M$};
				\fill (2.99,1.85) circle (0.05) node [above right] {$N$};
				\fill (2.25,1.25) circle (0.05) node [above left] {$O$};
				\fill (1.48,3.71) circle (0.05) node [left] {$P$};
				\fill (4.5,0) circle (0.05) node [below] {$Q$};
				\fill (8.88,-5.38) circle (0.05) node [below] {$S$};	
				\fill (3.5,3.5) circle (0.05) node [above right] {$J$};	
				\end{tikzpicture}
			\end{center}
			\begin{enumerate}
				\item Gọi $J$ là giao điểm của $BM$ và đường tròn $(O)$ ta có $BP.BA=BJ.BM$ tương tự ta cũng có với $CQ\cdot CA = CM\cdot CJ$ kết hợp với $M$ là trung điểm của $BC$ nên ta có
				\begin{eqnarray*}
				&& BP\cdot BA+CQ\cdot CA = BJ\cdot BM +  CM\cdot CJ\\ &\Leftrightarrow & BP\cdot BA+CQ\cdot CA = BM(BJ+CJ)=\dfrac{1}{2}BC^2.
				\end{eqnarray*}
				\item Từ $M$ là trung điểm $BC$ và giả sử $KP\cdot MQ=KQ\cdot MP$ ta có $$ \dfrac{KP}{KQ}=\dfrac{MP}{MQ}=\dfrac{S_{\triangle AMB}/AB}{S_{\triangle AMC}/AC}=\dfrac{AC}{AB}.$$
				Ta sẽ chứng minh $\dfrac{KP}{KQ}=\dfrac{AC}{AB}=\dfrac{\sin\angle B}{\sin\angle C}$, thật vậy vì $AK\parallel BC$ nên
				\begin{equation*}
				\dfrac{KP}{KQ}=\dfrac{\sin\angle KQP}{\sin\angle KPQ}=\dfrac{\sin (180^{\circ}-\angle KAP)}{\sin\angle KAQ}=\dfrac{\sin\angle B}{\sin\angle C} .
				\end{equation*}
				Từ đó suy ra điều phải chứng minh.
				\item  Để chứng minh $SM, SK$ là tiếp tuyến thì ta cần chứng minh  $\angle SMO = \angle SKO = 90^{\circ}$ mà ta đã có $\angle ONS = 90^{\circ}$ nên ta chỉ cần chứng minh tứ giác $KONM$ nội tiếp. Áp dụng định lý Ptoleme ta có $KM\cdot PQ=KQ\cdot PM+KP\cdot QM=2KQ\cdot PM.$
				\begin{eqnarray*}
					& \Rightarrow & KM\cdot (2NQ)=2KQ\cdot PM\\
					& \Rightarrow & KM\cdot NQ=KQ.PM \\
					& \Rightarrow & \triangle KNQ\sim \triangle KPM\\
					& \Rightarrow & \angle KNQ=\angle KPM=\frac{1}{2}\angle KOM.
				\end{eqnarray*}
				Tương tự  $\angle MNQ=\frac{1}{2}\angle KOM$ và từ đó ta có $\angle KNM=\angle KOM$ suy ra tứ giác $KOMN$ nội tiếp nên từ đó suy ra điều phải chứng minh.		
			\end{enumerate}
		}
	\end{ex}	
	\begin{ex}%[Đề thi HSG 9, tỉnh Bà Rịa Vũng Tàu, năm 2018]%[Lê Quân, Dự án EX-9G-18]%[9D0G7]
		\begin{enumerate}
			\item Cho các số dương $x, y$ thỏa mãn $x^2+y^2=2$. Chứng minh rằng 
			$\dfrac{x^2}{y+1}+\dfrac{y^2}{x+1}\geq 1.$
			\item Xét ba số dương $a, b, c$ thỏa $a^2+b^2+c^2=3.$ Tìm giá trị lớn nhất của biểu thức
			$$P=\dfrac{a}{\sqrt{2-bc}}+\dfrac{b}{\sqrt{2-ca}}+\dfrac{c}{\sqrt{2-ab}}.$$
		\end{enumerate}
		\loigiai{
			\begin{enumerate}
				\item Bất phương trình cần chứng minh tương đương với 
				\begin{eqnarray*}
					& & x^3+x^2 +y^3+y^2 \geq xy + (x+y) + 1\\
					& \Leftrightarrow & (x+y)^3-3xy(x+y)+2-xy-(x+y)-1 \geq 0\\
					& \Leftrightarrow & (x+y)\left[(x+y)^2-3xy-1\right] +(1-xy)\geq 0\\
					& \Leftrightarrow & (x+y)(x^2+y^2-3xy-1) +(1-xy) \geq 0\\
					& \Leftrightarrow & (x+y)(1-xy) + (1-xy) \geq 0\\
					& \Leftrightarrow & (1-xy)(x+y+1) \geq 0,
				\end{eqnarray*}
				áp dụng BĐT Cô-Si ta có $2 = x^2+y^2 \geq 2xy\Rightarrow 1 \geq xy \Rightarrow 1-xy \geq 0.$ Mặt khác ta cũng có $x+y+1\geq 0 $ nên ta có $(1-xy)(x+y+1) \geq 0$. Từ đó suy ra điều phải chứng minh.
				
				\item Áp dụng BĐT Cô-Si ta có  
				\begin{equation*}
				\dfrac{a}{\sqrt{2-bc}}\leq \dfrac{a}{\sqrt{2-\dfrac{b^{2}+c^{2}}{2}}}=\dfrac{a}{\sqrt{2-\dfrac{3-a^{2}}{2}}} = \dfrac{a}{\sqrt{\dfrac{a^{2}+1}{2}}}\leq \dfrac{a}{\sqrt{a}}=\sqrt{a},
				\end{equation*} 
				tương tự ta cũng có	$\dfrac{b}{\sqrt{2-ac}}\leq \sqrt{b} \text{ và } \dfrac{c}{\sqrt{2-ab}}\leq \sqrt{c}$ từ đó
				\begin{eqnarray*}
					& \Rightarrow & 0\leq P\leq \sqrt{a} + \sqrt{b} + \sqrt{c}\\
					& \Rightarrow &P^2\leq (\sqrt{a} + \sqrt{b} + \sqrt{c})^2 \leq 3(a+b+c)\\
					& \Rightarrow &P^4\leq 9(a+b+c)^2 \leq 9.3(a^2+b^2+c^2)=81\\
					& \Rightarrow &P\leq 3.
				\end{eqnarray*}
				Từ đó ta có giá trị lớn nhất của $P$ là $3$ dấu bằng xảy ra khi $a=b=c=1.$
			\end{enumerate}
		}
	\end{ex}

	\begin{ex}%[Đề thi HSG 9, tỉnh Bà Rịa Vũng Tàu, năm 2018]%[Lê Quân, Dự án EX-9G-18]%[9D0G4]
		\begin{enumerate}
			\item Chứng minh rằng với mọi số nguyên $n$ thì số $A = \dfrac{n^3}{3} + \dfrac{n^2}{2} + \dfrac{n}{6}$ là một số nguyên.
			\item Cho tập hợp $A$ gồm 100 số nguyên dương tử số 1 đến số 100. Tìm số nguyên dương $k$ nhỏ nhất sao cho mọi cách chọn ra $k$ phần tử tập hợp $A$ thì trong đó luôn có ba số là độ dài ba cạnh của một tam giác.
		\end{enumerate}
		\loigiai{
			\begin{enumerate}
				\item  Ta có 
				\begin{eqnarray*}
					A=\dfrac{2n^3+3n^2+n}{6} &=& \dfrac{n(n+1)(2n+1)}{6}\\
					&=& \dfrac{n(n+1)(n+2+n-1)}{6}\\
					&=& \dfrac{n(n+1)(n+2)}{6}+\dfrac{(n-1)n(n+1)}{6}.
				\end{eqnarray*}
				Ta thấy rằng $n(n+1)(n+2)$ và $(n-1)n(n+1)$ đều là tích của ba số nguyên liên tiếp nên đều chia hết cho 6 từ đó suy ra điều phải chứng minh.
				\item Ta có $A = \{1; 2; 3;\ldots; 99; 100\} $ vì trong $k$ phần tử của $A$ luôn có ba số thực thỏa yêu cầu nên $k\geq 3, k\in\mathbb{Z^+}$. Khi $k=3$ thì bộ $\{1; 2; 3\}$ không là ba cạnh của tam giác. Khi $k=4$ thì bốn phần tử của $A$ có dạng $\{n-3; n-2; n-1; n\}$, với $n\geq 4, n\in\mathbb{Z^+}$ trong đó tồn tại bộ ba $n-2<n-1<n$ và $(n-2)+(n-1)-n=n-3>0$. Vậy $k=4$ là số nguyên dương nhỏ nhất thỏa yêu cầu bài toán.
			\end{enumerate}
			
		}
	\end{ex}
