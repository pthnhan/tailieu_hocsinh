	Giải phương trình $\sqrt{x+9}+2012\sqrt{x+6}=2012+\sqrt{(x+9)(x+6)}$.
	\loigiai{
	Điều kiện xác định: $x\geq -6$.\\
	Phương trình đã cho tương đương
	\begin{align*}
	\left( \sqrt{x+9}-2012 \right)\left( \sqrt{x+6}-1 \right)=0 \Leftrightarrow \hoac{& x=-5\text{ (thỏa mãn)}\\ & x=4048135\text{ (thỏa mãn).}}
	\end{align*}
	Vậy tập nghiệm của phương trình đã cho là $S=\{-5; 4048135\}$.
	}

\begin{ex}[Đề thi vào Chuyên KHTN Hà Nội năm 2012, vòng 1][9D3G4]
	Giải hệ phương trình $\heva{& x^2+y^2+2y=4\\ & 2x+y+xy=4.}$
	\loigiai{
	Nhân cả hai vế của phương trình thứ hai với $2$ rồi cộng theo vế với phương trình thứ nhất ta được
	\begin{align*}
	(x+y)^2+4(x+y)=12 \Leftrightarrow \hoac{&x+y=2\\ &x+y=-6.}
	\end{align*}
	\begin{itemize}
		\item Với $x+y=2$ ta có
		\begin{align*}
			\heva{&x+y=2\\ &2x+y+xy=4} \Leftrightarrow \heva{&y=2-x\\ &x^2-3x+2=0} \Leftrightarrow \hoac{&\heva{&x=2\\&y=0}\\ &x=y=1.}
		\end{align*}
		\item Với $x+y=-6$ ta có
		\begin{align*}
			\heva{&x+y=-6\\ &2x+y+xy=4} \Leftrightarrow \heva{&y=-x-6\\ &x^2+5x+10=0.}
		\end{align*}
		Hệ phương trình này vô nghiệm.
	\end{itemize}
	Vậy các nghiệm của hệ phương trình đã cho là $(2;0)$ và $(1;1)$.
	}
\end{ex}

	\setlength{\belowdisplayskip}{-12pt}
	Tìm tất cả các cặp số nguyên $(x,y)$ thỏa mãn đẳng thức
	\begin{align*}
		(x+y+1)(xy+x+y)=5+2(x+y).
	\end{align*}}
	\loigiai{
	Phương trình đã cho tương đương
	\begin{align*}
		(x+y+1)\left( x+y+xy-2 \right)=3.
	\end{align*}
	Do $x$, $y$ là các số nguyên nên $x+y+1 \in \{\pm 1, \pm 3\}$ hay $x+y\in \{-4;-2;0;2\}$.
	\begin{itemize}
		\item Nếu $x+y=-4$ thì $xy=5$ không có cặp số nguyên nào thỏa mãn.
		\item Nếu $x+y=-2$ thì $xy=1$ ta tìm được $x=y=-1$.
		\item Nếu $x+y=0$ thì $xy=5$ không có cặp số nguyên nào thỏa mãn.
		\item Nếu $x+y=2$ thì $xy=1$ ta tìm được $x=y=1$.
	\end{itemize}
	Vậy có hai cặp số nguyên $(x;y)$ thỏa mãn yêu cầu là $(1;1)$ và $(-1;-1)$.
	}

\begin{ex}[Đề thi vào Chuyên KHTN Hà Nội năm 2012, vòng 1][9D5G1]
	Giả sử $x$, $y$ là các số thực dương thỏa mãn điều kiện $\left( \sqrt{x}+1 \right)\left( \sqrt{y}+1 \right)\geq 4$. Tìm giá trị nhỏ nhất của biểu thức $P=\dfrac{x^2}{y}+\dfrac{y^2}{x}$.
	\loigiai{
	Từ giả thiết ta có
	\begin{align*}
		\sqrt{xy}+\sqrt{x}+\sqrt{y}\geq 3.
	\end{align*}
	Áp dụng bất đẳng thức Cauchy cho các số không âm ta có
	\begin{align*}
		3\leq \sqrt{xy}+\sqrt{x}+\sqrt{y} \leq \dfrac{x+y}{2}+\dfrac{x+1}{2}+\dfrac{y+1}{2}=x+y+1.
	\end{align*}
	Suy ra $x+y\geq 2$.\\
	Vậy ta có
	\begin{align*}
		P=\left( \dfrac{x^2}{y}+y\right) +\left( \dfrac{y^2}{x}+x\right) -(x+y)\geq 2x+2y-(x+y)=x+y\geq 2.
	\end{align*}
	Đẳng thức xảy ra khi và chỉ khi $x=y=1$.\\
	Vậy $\min P=2$ khi $x=y=1$.
	}
\end{ex}
\begin{ex}[Đề thi vào Chuyên KHTN Hà Nội năm 2012, vòng 1][9H3B3][9H3K7]
	Cho tam giác $ABC$ nội tiếp đường tròn tâm $O$. Gọi $M$ là một điểm trên cung nhỏ $BC$ ($M$ khác $B$, $M$ khác $C$ và $AM$ không đi qua $O$). Giả sử $P$ là một điểm thuộc đoạn thẳng $AM$ sao cho đường tròn đường kính $MP$ cắt cung nhỏ $BC$ tại điểm $N$ khác $M$.
	\begin{enumerate}
		\item Gọi $D$ là điểm đối xứng với điểm $M$ qua $O$. Chứng minh rằng ba điểm $N$, $P$, $D$ thẳng hàng.
		\item Đường tròn đường kính $MP$ cắt $MD$ tại điểm $Q$ khác $M$. Chứng minh rằng $P$ là tâm đường tròn nội tiếp tam giác $AQN$.
	\end{enumerate}
	\loigiai{
	\begin{center}
	\begin{tikzpicture}[>=stealth,thick]
	\tkzInit[ymin=-3.5,ymax=3.5,xmin=-4,xmax=4]
	\tkzClip
	\tkzDefPoints{0/0/O}
	\tkzDefShiftPoint[O](-35:3){C}
	\tkzDefShiftPoint[O](215:3){B}
	\tkzDefShiftPoint[O](120:3){A}
	\tkzDefShiftPoint[O](-95:3){M}
	\tkzDefShiftPoint[O](-115:3){N}
	\tkzDefBarycentricPoint(M=1,O=-2)\tkzGetPoint{D}
	\tkzInterLL(A,M)(D,N)\tkzGetPoint{P}
	\tkzDefMidPoint(M,P)\tkzGetPoint{a}
	\tkzInterLC(D,M)(a,M)\tkzGetPoints{Q}{M}

	\tkzDrawCircle(O,A)
	\tkzDrawSegments(A,B B,C C,A A,M D,N M,D N,Q A,D A,N A,Q P,Q)
	\tkzDrawCircle[diameter](M,P)
	\tkzDrawPoint(O)

	\tkzLabelPoints[above](A,D)
	\tkzLabelPoints[left](B)
	\tkzLabelPoints[right](C,O)
	\tkzLabelPoints[below left](N)
	\tkzLabelPoints[below](M)
	\tkzLabelPoints[above left](P)
	\draw (Q)+(0.2,-0.4) node {$Q$};
	\end{tikzpicture}
	\end{center}
	\begin{enumerate}
		\item Do $N$ nằm trên đường trên đường tròn đường kính $MP$ và $N$ nằm trên đường tròn đường kính $MD$ nên $\widehat{MNP}=\widehat{MND}=90^\circ$.\\
		Từ đó suy ra $N$, $P$, $D$ thẳng hàng.
		\item Do các tứ giác $NADM$ và $NPQM$ nội tiếp nên ta có 
		\begin{align*}
			\widehat{ANP}=\widehat{AMD}=\widehat{QNP}.\tag{1}
		\end{align*}
		Dễ thấy $APQD$ nội tiếp đường tròn đường kính $DP$ nên
		\begin{align*}
			\widehat{AQP}=\widehat{ADP}=\widehat{AMN}=\widehat{NQP}.\tag{2}
		\end{align*}
		Từ $(1)$ và $(2)$ ta có $NP$ và $QP$ là các đường phân giác trong của tam giác $ANQ$.\\
		Vậy $P$ là tâm đường tròn nội tiếp tam giác $AQN$.
	\end{enumerate}
	}
\end{ex}
\begin{ex}[Đề thi vào Chuyên KHTN Hà Nội năm 2012, vòng 1][9D5G1]
	Giả sử $a$, $b$, $c$ là các số thực dương thỏa mãn $a\leq b\leq 3\leq c$, $c\leq b+1$, $a+b\geq c$.\\
	Tìm giá trị nhỏ nhất của biểu thức $Q=\dfrac{2ab+a+b+c(ab-1)}{(a+1)(b+1)(c+1)}$.
	\loigiai{
	Ta có \begin{eqnarray*}
	Q &=& \dfrac{2ab+a+b+c(ab-1)}{(a+1)(b+1)(c+1)}\\
	&=& \dfrac{ab+a+ab+b+abc+ac+abc+bc-abc-ac-bc-c}{(a+1)(b+1)(c+1)}\\
	&=& \dfrac{a(b+1)+b(a+1)+ac(b+1)+bc(a+1)-ac(b+1)-c(b+1)}{(a+1)(b+1)(c+1)}\\
	&=& \dfrac{a(b+1)(c+1)+b(a+1)(c+1)-c(b+1)(a+1)}{(a+1)(b+1)(c+1)}\\
	&=& \dfrac{a}{a+1}+\dfrac{b}{b+1}-\dfrac{c}{c+1}.
	\end{eqnarray*}
	Ta chứng minh
	\begin{align*}
		\dfrac{a}{a+1}+\dfrac{b}{b+1}-\dfrac{c}{c+1}\geq \dfrac{1}{1+1}+\dfrac{2}{2+1}-\dfrac{3}{3+1}=\dfrac{5}{12}.\tag{1}
	\end{align*}
	Thật vậy, ta có $(1)$ tương đương với
	\begin{align*}
		&\left( \dfrac{a}{a+1}-\dfrac{1}{2} \right) + \left( \dfrac{b}{b+1} - \dfrac{2}{3} \right) + \left( \dfrac{3}{4}-\dfrac{c}{c+1} \right)\geq 0\\
		\Leftrightarrow &\dfrac{a-1}{2(a+1)}+\dfrac{b-2}{3(b+1)}+\dfrac{3-c}{4(c+1)}\geq 0\\
		\Leftrightarrow &\left( 3-c \right)\left[ \dfrac{1}{4(c+1)} -\dfrac{1}{3(b+1)} \right] + \left[ (3-c)+(b-2) \right]\left[ \dfrac{1}{3(b+1)} -\dfrac{1}{2(a+1)} \right]+\\
		&+ \left[ (3-c) +(b-2) +(a-1) \right] \dfrac{1}{2(a+1)}\geq 0\\
		\Leftrightarrow & \dfrac{(3-c)(3b-4c-1)}{12(b+1)(c+1)} + \dfrac{(b+1-c)(2a-3b-1)}{6(b+1)(a+1)} + \dfrac{a+b-c}{2(a+1)}\geq 0. \tag{2}
	\end{align*}
	Từ điều kiện $0<a\leq b\le 3\leq c$, $b+1 \leq c$, $c\leq a+b$, ta thấy mỗi phân thức ở vế trái đều không âm. Do đó $(2)$ đúng nên $(1)$ đúng.\\
	Đẳng thức xảy ra khi $a=1$, $b=2$ và $c=3$.\\
	Vậy giá trị nhỏ nhất của $Q$ là $\dfrac{5}{12}$.
	}
\end{ex}
