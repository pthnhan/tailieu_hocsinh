	Cho hàm số $f(x)=\left(m^2+3m-4\right)x^{2017}+m^2-7$. Gọi $S$ là tập hợp tất cả các giá trị của tham số $m$ để hàm số $f$ là hàm số lẻ trên $\mathbb{R}$. Tính tổng các phần tử của $S$.
	\choice
	{\True $0$}
	{$-3$}
	{$\sqrt 7 $}
	{$2\sqrt 7 $}
	\loigiai{
		Tập xác định $\mathscr{D}=\mathbb{R}$. Suy ra $\forall x\in \mathscr{D}$ thì $-x\in \mathscr{D}$.\\
		Ta có $f(-x)=-\left(m^2+3m-4\right) x^{2017}+m^2-7$.\\
		Để $f$ là hàm số lẻ thì $\forall x\in \mathscr{D}$, $f(x)=-f(-x)$.\\
		$\Rightarrow\left(m^2+3m-4\right) x^{2017}+m^2-7=\left(m^2+3m-4\right) x^{2017}-m^2+7$.\\
		$\Rightarrow m^2=7 \Leftrightarrow m=\pm\sqrt 7 $.\\
		Vậy tổng các phần tử của $S$ là $\sqrt 7+\left(-\sqrt 7\right)=0$.
	}
\end{ex}

\begin{ex}%[Thi GHK, Việt Đức - Hà Nội, 2018]%[Lê Minh Đức, EX-DT-10&11]%[0D2B1-4]
	Trong các hàm số sau, hàm số nào là hàm chẵn?
	\choice
	{\True $y=\sqrt{2-x}+\sqrt{2+x}$}
	{$y=\sqrt{x+2}+\sqrt{x-2}$}
	{$y=\left|x+2\right|-\left|x-2\right|$}
	{$y=x^4+x+1$}
	\loigiai{
		\begin{itemize}
			\item Hàm số $y=\sqrt{2-x}+\sqrt{2+x}$ có tập xác định là $\mathscr{D}=\left[-2;2\right]$.\\
			Suy ra $\forall x\in \mathscr{D}$ thì $-x\in \mathscr{D}$.\\
			Ta có $f(-x)=\sqrt{2-(-x)}+\sqrt{2+(-x)}$ $=\sqrt{2+x}+\sqrt{2-x}=f(x)$.\\
			Vậy hàm số $y=\sqrt{2-x}+\sqrt{2+x}$ là hàm số chẵn.
			\item Hàm số $y=\left|x+2\right|-\left|x-2\right|$ có tập xác định là $\mathscr{D}=\left[2;+\infty\right)$.\\
			Ta có $2\in \mathscr{D}$ nhưng $-2\notin \mathscr{D}$ nên hàm số trên không là hàm số chẵn cũng không là hàm số lẻ.
			\item Hàm số $y=\left|x+2\right|-\left|x-2\right|$ có tập xác định là $\mathscr{D}=\mathbb{R}$.\\
			Suy ra $\forall x\in \mathscr{D}$ thì $-x\in \mathscr{D}$.\\
			Ta có $f(-x)=\left|-x+2\right|-\left|-x-2\right|=\left|x-2\right|-\left|x+2\right|=-f(x)$.\\
			Vậy hàm số $y=\left|x+2\right|-\left|x-2\right|$ là hàm số lẻ.
			\item Hàm số $y=x^4+x+1$ có tập xác định là $\mathscr{D}=\mathbb{R}$.\\
			Suy ra $\forall x\in \mathscr{D}$ thì $-x\in \mathscr{D}$.\\
			Ta có $f(1)=3$ và $f(-1)\ne 1$. Do $f(1)\ne f(-1)$ và $f(1)\ne-f(-1)$ nên hàm số trên không là hàm số chẵn cũng không là hàm số lẻ.
		\end{itemize}
	}
\end{ex}

\begin{bt}%[Thi GHK, Việt Đức - Hà Nội, 2018]%[Lê Minh Đức, EX-DT-10&11]%[0D1B2-1]
	Cho hai tập hợp $A=\left\{ x \in \mathbb{Z} \ \middle | \ (2x^2-x-3)(x^2-4) = 0 \right\}$ và $B=\left\{ x \in \mathbb{N} \ \middle | \ x<4 \right\}$. Viết lại các tập hợp $A$ và $B$ bằng cách liệt kê các phần tử.
	\loigiai{
		Ta có 
		\[\left(2x^2-x-3\right)\left(x^2-4\right)=0 \Leftrightarrow \hoac{& 2x^2-x-3=0\\& x^2-4=0} \Leftrightarrow \hoac{& (x+1)(2x-3)=0\\& x^2=4} \Leftrightarrow \hoac{& x=-1\\& x=\dfrac{3}{2}\\& x=\pm 2}\]
		Do $x \in \mathbb{Z} \Rightarrow x \in \left\{ -2;-1;2 \right\}$ hay $A=\left\{ -2;-1;2 \right\}$.\\
		$B=\left\{ x \in \mathbb{N} \ \middle | \ x<4 \right\}=\left\{0;1;2;3\right\}$.
	}
\end{bt}

\begin{bt}%[Thi GHK, Việt Đức - Hà Nội, 2018]%[Lê Minh Đức, EX-DT-10&11]%[0D2K1-2]
	Gọi tập xác định của các hàm số $f(x)=\sqrt{5+x}+\sqrt{5-x};g(x)=\dfrac{3x+4}{\sqrt{x+4}}$ lần lượt là $\mathscr{D}_1$; $\mathscr{D}_2$. Hãy tìm $\mathscr{D}_1$; $\mathscr{D}_2$; $\mathscr{D}_1\cap \mathscr{D}_2$; ${\mathscr{D}_1}\cup \mathscr{D}_2$.
\loigiai{
	Điều kiện xác định của hàm số $f(x)=\sqrt{5+x}+\sqrt{5-x}$ là
	\[ \heva{& 5+x\ge 0\\& 5-x\ge 0} \Leftrightarrow -5\le x\le 5. \]
	Do đó tập xác định của $f(x)=\sqrt{5+x}+\sqrt{5-x}$ là $\mathscr{D}_1=\left[-5;5\right]$\\
	Điều kiện xác định của hàm số $g(x)=\dfrac{3x+4}{\sqrt{x+4}}$ là
	\[ \heva{& \sqrt{4+x}\ne 0\\& x+4\ge 0} \Leftrightarrow x >-4 .\]
	Do đó tập xác định của $g(x)=\dfrac{3x+4}{\sqrt{x+4}}$ là $\mathscr{D}_2=\left(-4;+\infty\right)$.\\
	Suy ra $\mathscr{D}_1\cap \mathscr{D}_2=\left(-4;5\right]$; ${\mathscr{D}_1}\cup \mathscr{D}_2=\left(-5;+\infty\right)$.
}
\end{bt}

\begin{bt}%[Thi GHK, Việt Đức - Hà Nội, 2018]%[Lê Minh Đức, EX-DT-10&11]%[0D2K1-2]

\loigiai{

Điều kiện xác định 
\[ \left|x-2\right|-1 > 0\Leftrightarrow\left|x-2\right| > 1 \Leftrightarrow \hoac{& x-2 > 1\\& x-2 <-1} \Leftrightarrow \hoac{& x > 3\\& x < 1} \Rightarrow{\mathscr{D}_1}=\left(-\infty;1\right)\cup\left(3;+\infty\right). \]

}
