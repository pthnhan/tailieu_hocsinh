\begin{bt}%[GHK1 Toán 11, Lê Quý Đôn, HCM, 2018-2019]%[Vũ Nguyễn Hoàng Anh, EX-DT-10\&11]%[1D1B2-1]
	Giải phương trình $2\sin \left(\dfrac{\pi}{3}-x\right)-\sqrt{3}=0$.
	\loigiai{
		Phương trình đã cho tương đương
		\begin{align*}
			\sin \left(\dfrac{\pi}{3}-x\right)=\sin \dfrac{\pi}{3} \Leftrightarrow\ & \hoac{&\dfrac{\pi}{3}-x=\dfrac{\pi}{3}-k2\pi\\ &\dfrac{\pi}{3}-x=\dfrac{2\pi}{3}-k2\pi}\\
			 \Leftrightarrow\ &\hoac{&x=k2\pi\\ &x=-\dfrac{\pi}{3}+k2\pi} \ (k\in\mathbb{Z}).
		\end{align*}
	}
\end{bt}

\begin{bt}%[GHK1 Toán 11, Lê Quý Đôn, HCM, 2018-2019]%[Vũ Nguyễn Hoàng Anh, EX-DT-10\&11]%[1D1B3-1]
	Giải phương trình $\cos 2x -5\cos x+3=0$.
	\loigiai{
		Phương trình đã cho tương đương
		\begin{align*}
			2\cos^2 x-5\cos x +2=0 \Leftrightarrow \hoac{&\cos x= \dfrac{1}{2}\\ &\cos x=2}
			\Leftrightarrow x=\pm \dfrac{\pi}{3}+k2\pi\ (k\in \mathbb{Z}).
		\end{align*} 
	}
\end{bt}

\begin{bt}%[GHK1 Toán 11, Lê Quý Đôn, HCM, 2018-2019]%[Vũ Nguyễn Hoàng Anh, EX-DT-10\&11]%[1D1B3-3]
	Giải phương trình $\sin 2x +\sqrt{3}\cos 2x=\sqrt{2}$.
	\loigiai{
		Phương trình đã cho tương đương
		\begin{align*}
			&\sin 2x \cos \dfrac{\pi}{3}+\cos 2x \sin \dfrac{\pi}{3}=\sin \dfrac{\pi}{4} \\
			\Leftrightarrow\ & \sin \left(2x+\dfrac{\pi}{3}\right)=\sin \dfrac{\pi}{4}\\
			\Leftrightarrow\ & \hoac{&2x+\dfrac{\pi}{3}=\dfrac{\pi}{4}+k2\pi\\ &2x+\dfrac{\pi}{3}=\dfrac{3\pi}{4}+k2\pi}\\
			\Leftrightarrow\ & \hoac{&x=-\dfrac{\pi}{24}+k\pi\\ &x=\dfrac{5\pi}{24}+k\pi}\ (k\in \mathbb{Z}).
		\end{align*} 
	}
\end{bt}

\begin{bt}%[GHK1 Toán 11, Lê Quý Đôn, HCM, 2018-2019]%[Vũ Nguyễn Hoàng Anh, EX-DT-10\&11]%[1D1K3-1]
	Giải phương trình $\cos 3x +\cos 2x +\cos x+1=0$.
	\loigiai{
		Phương trình đã cho tương đương
		\begin{align*}
			&4\cos^3x +2\cos^2 x-2\cos x=0\\
			 \Leftrightarrow\ &\hoac{&\cos x=0 \\ &\cos x=\dfrac{1}{2}\\ &\cos x=-1} \Leftrightarrow \hoac{&x=k2\pi\\ &x=\pm \dfrac{\pi}{6}+k2\pi \\ &x=\pi +k2\pi}	\\
			 \Leftrightarrow\ &\hoac{&x=k2\pi\\ &x=(2k+1)\pi\\ &x=\pm \dfrac{\pi}{6}+k2\pi}\ (k\in \mathbb{Z}).
			 \end{align*} 
	}
\end{bt}

\begin{bt}%[GHK1 Toán 11, Lê Quý Đôn, HCM, 2018-2019]%[Vũ Nguyễn Hoàng Anh, EX-DT-10\&11]%[1D2K2-1]
	Cho $11$ học sinh gồm $5$ nam và $6$ nữ, xếp thành một hàng ngang, hỏi có bao nhiêu cách sắp xếp sao cho
	\begin{enumerate}
		\item Nam riêng, nữ riêng.
		\item Nam, nữ xếp xen kẽ.
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Khi xếp nam riêng, nữ riêng thì có $2$ trường hợp: nhóm nam ở đầu hàng hoặc nhóm nữ ở đầu hàng.
		\begin{center}
			\begin{tabular}{|c|c|c|c|c|c|c|c|c|c|c|}
				\hline
				\textbf{Nam}& \textbf{Nam}&\textbf{Nam}&\textbf{Nam}&\textbf{Nam}& \textit{Nữ}& \textit{Nữ}&\textit{Nữ}&\textit{Nữ}&\textit{Nữ}&\textit{Nữ}\\
				\hline
			\end{tabular}
		
			\begin{tabular}{|c|c|c|c|c|c|c|c|c|c|c|}
				\hline
				\textit{Nữ}& \textit{Nữ}&\textit{Nữ}&\textit{Nữ}&\textit{Nữ}&\textit{Nữ}&\textbf{Nam}& \textbf{Nam}&\textbf{Nam}&\textbf{Nam}&\textbf{Nam}\\
				\hline
			\end{tabular}
		\end{center}
		Trong mỗi trường hợp có $5!$ cách xếp các bạn nam và $6!$ cách xếp các bạn nữ. Vậy có tất cả $2\times 5!\times 6!=172\,800$ cách xếp nam riêng, nữ riêng.
		\item Khi xếp nam, nữ xen kẽ, mỗi bạn nam bắt buộc ở giữa hai bạn nữ.
		\begin{center}
			\begin{tabular}{|c|c|c|c|c|c|c|c|c|c|c|}
				\hline
				\textit{Nữ} & \textbf{Nam} & \textit{Nữ} & \textbf{Nam} & \textit{Nữ} & \textbf{Nam} & \textit{Nữ} & \textbf{Nam} & \textit{Nữ} & \textbf{Nam} & \textit{Nữ}\\
				\hline
			\end{tabular}
		\end{center}
		Do đó trong cách xếp này có tất cả $5!\times 6!=86\,400$ cách xếp thỏa mãn.
	\end{enumerate} 
}
\end{bt}

\begin{bt}%[GHK1 Toán 11, Lê Quý Đôn, HCM, 2018-2019]%[Vũ Nguyễn Hoàng Anh, EX-DT-10\&11]%[1D2K1-2]
	Có bao nhiêu số tự nhiên gồm $4$ chữ số được lập thành từ các chữ số $1$, $2$, $3$, $4$, $5$, $6$, $7$, $8$, $9$ và là một số chia hết cho $15$.
	\loigiai{
		Gọi $N=\overline{abcd}$ là một số thỏa mãn đề bài. Do $N$ chia hết cho $5$ nên $d=5$. Ta chỉ việc chọn các số $a$, $b$, $c$ sao cho $a+b+c+5$ chia hết cho $3$ hay $a+b+c$ chia $3$ dư $1$. Lấy $a$, $b$ một cách tùy ý ta có $9\times 9=81$ cách. Khi đó số dư của phép chia $a+b$ cho $3$ rơi vào một trong ba trường hợp
		\begin{itemize}
			\item \textit{Trường hợp 1:} $a+b$ chia hết cho $3$, ta có $3$ cách chọn $c$ thuộc $\{2;5;8\}$.
			\item \textit{Trường hợp 2:} $a+b$ chia $3$ dư $1$, ta có $3$ cách chọn $c$ thuộc $\{1;4;7\}$.
			\item \textit{Trường hợp 3:} $a+b$ chia $3$ dư $2$, ta có $3$ cách chọn $c$ thuộc $\{3;6;9\}$.
		\end{itemize}
		Từ đó có tất cả $81\times 3=243$ số thỏa mãn đề bài.
	}
\end{bt}
