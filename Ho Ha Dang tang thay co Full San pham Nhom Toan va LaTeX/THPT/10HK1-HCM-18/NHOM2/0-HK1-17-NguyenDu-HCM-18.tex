\begin{bt}%[Đề HK1, Nguyễn Du, Hồ Chí Minh 17-18]%[Tuấn Nguyễn]%[0D2B1-2]
Tìm tập xác định của hàm số  $f(x) = \dfrac{\sqrt{1 - x}}{x^2 - 3x + 2}$.
\loigiai{
\begin{eqnarray*}
\text{Hàm số xác định}& \Leftrightarrow &\heva{&1-x \ge 0\\&x^2-3x+2 \ne 0}\\ 
  &\Leftrightarrow& \heva{&x \le 1\\&x \ne 1\\&x \ne 2}\\
 & \Leftrightarrow &x<1.
\end{eqnarray*}
Vậy tập xác định là $\mathscr{D}=(\infty; 1)$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Du, Hồ Chí Minh 17-18]%[Tuấn Nguyễn]%[0D2B3-2]
Viết phương trình  của parabol  $(P) \colon y = ax^2 + bx + c$  biết rằng đồ thị  $(P)$  đi qua điểm  $A(1; - 6)$  và có đỉnh là  $I(- 1; - 2)$.
\loigiai{
Vì $A(1;-6) \in (P)$ và  $(P)$ có đỉnh $I(-1;-2)$ nên ta có hệ
\begin{eqnarray*}
\heva{&a+b+c=-6\\&\dfrac{-b}{2a}=-1\\&a-b+c=-2} \Leftrightarrow \heva{&a=-1\\&b=-2\\&c=-3.}
\end{eqnarray*}
Vậy phương trình parabol  $(P) \colon y=-x^2-2x-3$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Du, Hồ Chí Minh 17-18]%[Tuấn Nguyễn]%[0D3B2-5]
Cho phương trình  $x^2 - 2mx + m^2 - m = 0$, trong đó $m$ là tham số.  Xác định giá trị của tham số $m$ để phương trình đã cho có hai nghiệm phân biệt  $x_1$, $x_2$  thỏa mãn  $x_1^2 + x_2^2 = 4$.
\loigiai{
Phương trình đã cho có hai nghiệm phân biệt
\begin{eqnarray*}
 &\Leftrightarrow& \Delta'>0 \\ 
& \Leftrightarrow& m^2-(m^2-m)>0\\
& \Leftrightarrow&m>0.
\end{eqnarray*}
Với $m>0$, theo hệ thức Vi-et, ta có
$$ \heva{&x_1+x_2=2m\\&x_1x_2=m^2-m.} $$
Ta có
\begin{eqnarray*}
&&x_1^2 + x_2^2 = 4\\ 
&\Leftrightarrow & (x_1+x_2)^2-2x_1x_2=4\\
&\Leftrightarrow &(2m)^2-2(m^2-m)=4\\
&\Leftrightarrow &2m^2+2m-4=0\\
&\Leftrightarrow &\hoac{&m=1\\&m=-2 &\text{(loại)}.}
\end{eqnarray*}
Vậy $m=1$ là giá trị cần tìm.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Du, Hồ Chí Minh 17-18]%[Tuấn Nguyễn]%[0D2B2-2]
Giải phương trình $\left|x^2 + 2x - 2\right| = \left|x - 8\right|$.                               
\loigiai{ 
\begin{eqnarray*}
\left|x^2 + 2x - 2\right| = \left|x - 8\right|& \Leftrightarrow &\hoac{&x^2+2x-2=x-8\\&x^2+2x-2=-x+8}\\
& \Leftrightarrow &\hoac{&x^2+x+6=0&\text{(phương trình vô nghiệm)}\\&x^2+3x-10=0}\\
& \Leftrightarrow &\hoac{&x=2\\&x=-5.}\\
\end{eqnarray*}
Vậy tập nghiệm của phương trình là $S=\{-5; 2\}$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Du, Hồ Chí Minh 17-18]%[Tuấn Nguyễn]%[0D2G2-4]
Giải phương trình $6\sqrt{x(2x + 1) + 4} + 2x^2 + x - 12 = 0$.
\loigiai{
$6\sqrt{x(2x + 1) + 4} + 2x^2 + x - 12 = 0 \Leftrightarrow 6\sqrt{x(2x+1)+4}+x(2x+1)-12=0$.\\
Đặt $t=\sqrt{2x(x+1)+4}$ $(t \ge 0)$. \\
Phương trình trở thành $6t+t^2-16=0 \Leftrightarrow \hoac{&t=2\\&t=-8&\text{(loại)}.}$\\
Với $t=2 \Rightarrow \sqrt{x(2x+1)+4}=2 \Leftrightarrow \hoac{&x=0\\&x=-\dfrac{1}{2}.}$\\
Thử lại nghiệm, thấy thỏa. Vậy tập nghiệm của phương trình là $S=\left\{\dfrac{-1}{2}; 0\right\}$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Du, Hồ Chí Minh 17-18]%[Tuấn Nguyễn]%[0D3K3-4]
Giải hệ phương trình   $\heva{&x - y =  - 1\\&x^2 + 2x + y^2 = 7.}$
\loigiai{
\begin{eqnarray*}
\text{Hệ phương trình}& \Leftrightarrow &\heva{&y=x+1\\&x^2+2x+(x+1)^2=7}\\ 
& \Leftrightarrow &\heva{&y=x+1\\&2x^2+4x-6=0}\\
& \Leftrightarrow &\heva{&y=x+1\\&\hoac{&x=1\\&x=-3}}\\
& \Leftrightarrow &\heva{&x=1\\&y=2} \;\text{hoặc}\; \heva{&x=-3\\&y=-2.}
\end{eqnarray*}
Vậy hệ phương trình có hai nghiệm $(x;y)=(1;2)$; $(x;y)=(-3;-2)$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Du, Hồ Chí Minh 17-18]%[Tuấn Nguyễn]%[0H2B2-1]%[0H2B3-3]%[0H2K2-5]
Trong hệ trục tọa độ  $Oxy$  cho tam giác  $ABC$  có tọa độ $3$ đỉnh là  $A(-1;1)$,  $B(0;\sqrt{3} + 1)$,  $C(1;1)$ và điểm $D(0; 2)$.
\begin{enumerate}
\item Tính tích vô hướng của hai véc-tơ  $\overrightarrow{AD}$  và  $\overrightarrow{AC}$. Tìm tọa độ điểm $E$ là trung điểm cạnh  $AC$.
\item  Chứng minh rằng tam giác  $ABC$  đều và tìm tọa độ tâm đường tròn ngoại tiếp tam giác  $ABC$.
\item  Tìm tọa độ điểm  $M$  nằm trên trục  $Ox$  sao cho giá trị của tích vô hướng  $\overrightarrow {AM} \cdot \overrightarrow {CM} $  đạt giá trị nhỏ nhất.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $\vec{AD}=(1;1)$, $\vec{AC}=(2;0)$ suy ra
$\vec{AD} \cdot \vec{AC}=1\cdot 2+1\cdot 0=2.$
Vì điểm $E$ là trung điểm cạnh  $AC$ nên ta có
$$ \heva{&x_E=\dfrac{x_A+x_C}{2}\\&y_E=\dfrac{y_A+y_C}{2}} \Leftrightarrow  \heva{&x_E=0\\&y_E=1.} $$
Vậy $E(0;1)$.
\item Ta có $AB=AC=BC=2$ nên tam giác $ABC$ đều.\\
Vì $\triangle ABC$ đều nên tâm đường tròn ngoại tiếp $\triangle ABC$ là trọng tâm $G$.\\
$G$ là trọng tâm của $\triangle ABC$ nên 
$$\heva{&x_G=\dfrac{x_A+x_B+x_C}{3}\\&y_G=\dfrac{y_A+y_B+y_C}{3}} \Leftrightarrow  \heva{&x_G=\dfrac{-1+0+1}{3}=0\\&y_G=\dfrac{1+\sqrt{3}+1+1}{3}=\dfrac{\sqrt{3}+3}{3}.} $$
Vậy tọa độ tâm đường tròn ngoại tiếp $\triangle ABC$ là $G\left(0;\dfrac{\sqrt{3}+3}{3}\right)$.
\item Vì $M \in Ox$ nên $M(m;0)$. Ta có $\vec{AM}=(m+1;-1)$, $\vec{CM}=(m-1;-1)$.\\
Ta có $\vec{AM} \cdot \vec{CM}=m^2 \ge 0$ $\forall m$. Suy ra giá trị $\vec{AM} \cdot \vec{CM}$ nhỏ nhất khi $m=0$.\\
Vậy $M(0; 0)$.
\end{enumerate}
}
\end{bt}
