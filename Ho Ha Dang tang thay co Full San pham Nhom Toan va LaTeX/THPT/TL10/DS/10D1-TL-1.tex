\section{MỆNH ĐỀ}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Mệnh đề}
\begin{dn} Mệnh đề logic (gọi tắt là mệnh đề) là một câu khẳng định \textbf{hoặc đúng hoặc sai}.
	\begin{itemize}
		\item Một mệnh đề không thể vừa đúng vừa sai.
		\item Một câu khẳng định đúng gọi là \textbf{mệnh đề đúng}. Một câu khẳng định sai gọi là \textbf{mệnh đề sai}.
	\end{itemize}
	
\end{dn}
\begin{note} \textbf{Những điểm cần lưu ý.}
	\begin{itemize}
		\item Các câu hỏi, câu cảm thán, câu mệnh lệnh không phải là mệnh đề.
		\item Mệnh đề thường được kí hiệu bằng các chữ cái in hoa.\\
		\hspace{3cm} Ví dụ: Q:``6 chia hết cho 3''.
		\item Một câu mà chưa thể nói đúng hay sai nhưng chắc chắn nó chỉ đúng hoặc sai, không thể vừa đúng vừa sai cũng là một mệnh đề.\\
		\hspace{3cm} Ví dụ: ``Có sự sống ngoài Trái Đất'' là mệnh đề.
		\item Trong thực tế, có những mệnh đề mà tính đúng sai của nó luôn gắn với một thời gian và địa điểm cụ thể: đúng ở thời gian hoặc địa điểm này nhưng sai ở thời gian hoặc địa điểm khác. Nhưng ở bất kì thời điểm nào, địa điểm nào cũng luôn có giá trị chân lí đúng hoặc sai.\\
		Ví dụ: Sáng nay bạn An đi học.\\
	\end{itemize}
\end{note}
\subsubsection{Mệnh đề chứa biến}
\begin{dn}
	Những câu khẳng định mà tính \textbf{đúng-sai} của chúng tùy thuộc vào giá trị của biến gọi là những \textbf{mệnh đề chứa biến.}
\end{dn}
Ví dụ: Cho $P(x): x>x^2$ với $x$ là số thực. Khi đó $P(2)$ là mệnh đề sai, $P\left(\dfrac{1}{2}\right)$ là mệnh đề đúng.
\subsubsection{Mệnh đề phủ định}
\begin{dn}
	Cho mệnh đề $P$. Mệnh đề ``\textbf{Không phải} $P$'' được gọi là mệnh đề phủ định của $P$ và kí hiệu là $\overline{P}$.
	\begin{itemize}
		\item Mệnh đề $P$ và mệnh đề phủ định $\overline{P}$ là hai câu khẳng định trái ngược nhau. Nếu $P$ đúng thì $\overline{P}$ sai, nếu $P$ sai thì $\overline{P}$ đúng.
		\item Mệnh đề phủ định của $P$ có thể diễn đạt theo nhiều cách khác nhau. Chẳng hạn, xét mệnh đề $P$: ``$2$ là số chẵn''. Khi đó, mệnh đề phủ định của $P$ có thể phát biểu là $\overline{P}$: ``$2$ không phải là số chẵn'' hoặc ``$2$ là số lẻ''.
	\end{itemize} 
\end{dn}
\subsubsection{Mệnh đề kéo theo và mệnh đề đảo}
\begin{dn}
	Cho hai mệnh đề $P$ và $Q$. Mệnh đề ``Nếu $P$ thì $Q$'' được gọi là \textbf{mệnh đề kéo theo}.
	\begin{itemize}
		\item  Kí hiệu là $P\Rightarrow Q.$
		\item Mệnh đề kéo theo chỉ sai khi $P$ đúng $Q$ sai.
		\item $P\Rightarrow Q$ còn được phát biểu là `` $P$ kéo theo $Q$'', ``$P$ suy ra $Q$'' hay ``Vì $P$ nên $Q$''.
	\end{itemize}
\end{dn}
\begin{note}
	\textbf{Chú ý}
	\begin{itemize}
		\item Trong toán học, định lí là một mệnh đề đúng, thường có dạng: $P\Rightarrow Q$.
		Khi đó ta nói $P$ là giả thiết, $Q$ là kết luận của định lí, hoặc $P$ là điều kiện đủ để có $Q$, hoặc $Q$ là điều kiện cần để có $P$.
		\item Trong logic toán học, khi xét giá trị chân lí của mệnh đề $P\Rightarrow Q$ người ta không quan tâm đến mối quan hệ về nội dung của hai mệnh đề $P$, $Q$. Không phân biệt trường hợp $P$ có phải là nguyên nhân để có $Q$ hay không mà chỉ quan tâm đến tính đúng, sai của chúng.
		
		Ví dụ: ``Nếu mặt trời quay quanh trái đất thì Việt Nam nằm ở châu Âu'' là một mệnh đề đúng. Vì ở đây hai mệnh đề $P$: ``Mặt trời quay xung quanh trái đất'' và $Q$: ``Việt Nam nằm ở châu Âu'' đều là mệnh đề sai.
	\end{itemize}
\end{note}
\begin{dn}
	Cho mệnh đề kéo theo $P\Rightarrow Q$. Mệnh đề $Q\Rightarrow P$ được gọi là \textbf{mệnh đề đảo} của mệnh đề $P\Rightarrow Q$.
\end{dn}
\begin{note}
	Mệnh đề đảo của một mệnh đề đúng không nhất thiết là một mệnh đề đúng.
\end{note}
\subsubsection{Mệnh đề tương đương}
\begin{dn}
	Cho hai mệnh đề $P$ và $Q$. Mệnh đề có dạng ``$P$ nếu và chỉ nếu $Q$'' được gọi là mệnh đề tương đương.
	\begin{itemize}
		\item Kí hiệu là $P \Leftrightarrow Q$
		\item Mệnh đề $P \Leftrightarrow Q$ đúng khi cả hai mệnh đề $P\Rightarrow Q$ và $Q \Rightarrow P$ cùng đúng hoặc cùng sai. (Hay $P \Leftrightarrow Q$ đúng khi cả hai mệnh đề $P$ và $Q$ cùng đúng hoặc cùng sai) 
		\item $P\Leftrightarrow Q$ còn được phát biểu là ``$P$ khi và chỉ khi $Q$'', ``$P$ tương đương với $Q$'', hay ``$P$ là điều kiện cần và đủ để có $Q$''.
	\end{itemize}
\end{dn}
\begin{note}
	Hai mệnh đề $P$, $Q$ tương đương với nhau hoàn toàn không có nghĩa là nội dung của chúng như nhau, mà nó chỉ nói lên rằng chúng có cùng giá trị chân lí (cùng đúng hoặc cùng sai).
	
	Ví dụ: ``Hình vuông có một góc tù khi và chỉ khi 100 là số nguyên tố'' là một mệnh đề đúng.
\end{note}
\subsubsection{Các kí hiệu $\forall$ và $\exists$}
\begin{itemize}
	\item Kí hiệu $\forall$ (với mọi): $``\forall x \in X, P(x)$'' hoặc $``\forall x \in X : P(x)$''.
	\item Kí hiệu $\exists$ (tồn tại): $``\exists x \in X, P(x)$'' hoặc $``\exists x \in X : P(x)$''.
\end{itemize}
\begin{note} \textbf{Chú ý}
	\begin{itemize}
		\item Phủ định của mệnh đề $``\forall x \in X, P(x)$'' là mệnh đề $``\exists x\in X, \overline{P(x)}$''.
		\item Phủ định của mệnh đề $``\exists x\in X, P(x)$'' là mệnh đề  $``\forall x\in X, \overline{P(x)}$''.
	\end{itemize}
\end{note}

\subsection{Các dạng toán}
\begin{dang}{Mệnh đề có nội dung đại số và số học}
	
\end{dang}
\begin{vd}
	Tìm mệnh đề phủ định của các mệnh đề sau:
	\begin{enumerate}[a) ]
		\item $A:\text{\textquotedblleft}\sqrt{6}$ là số hữu tỉ\textquotedblright.
		\item $B:\text{\textquotedblleft}n $ chia hết cho $3$ và $5$ thì $n$ chia hết cho $15$\textquotedblright.
		\item $C:\text{\textquotedblleft}\forall x\in \mathbb{N}: x^2+x+3>0$\textquotedblright.
		\item $D:\text{\textquotedblleft}\exists x \in \mathbb{N}, \exists y \in \mathbb{R}: \dfrac{x}{y}+\dfrac{y}{x}=2$\textquotedblright.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item $\overline{A}: \text{\textquotedblleft} \sqrt{6}$ không là số hữu tỉ\textquotedblright.
			\item  $\overline{B}:$ \textquotedblleft $n$ không chia hết cho $3$ hoặc $n$ không chia hết cho $5$ thì nó không chia hết cho $15$ \textquotedblright.
			\item $\overline{C}:\text{\textquotedblleft}\exists x\in \mathbb{N}: x^2+x+3\le 0$\textquotedblright.
			\item  $\overline{D}:\text{\textquotedblleft}\forall x \in \mathbb{N}, \forall y \in \mathbb{R}: \dfrac{x}{y}+\dfrac{y}{x}\neq2$\textquotedblright.
		\end{enumerate}
	}
\end{vd}
\begin{vd}
	Xét tính đúng - sai của các mệnh đề sau và tìm mệnh đề phủ định của nó:
	\begin{enumerate}[a) ]
		\item $\forall x \in \mathbb{R}: x^2+6>0$.
		\item $\exists x \in \mathbb{R}: x^2+x+1=0$.
		\item $\exists x \in \mathbb{R}: x>x^2$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a) ]
			\item Mệnh đề đúng.\\
			Phủ định là $\overline{A}: \exists x \in \mathbb{R}: x^2+6 \le 0$. 
			\item Mệnh đề sai vì phương trình $x^2+x+1=0$ vô nghiệm trong $\mathbb{R}$.\\
			Phủ định là $\overline{B}:\text{\textquotedblleft}\forall x \in \mathbb{R}:x^2+x+\neq 0$.
			\item Mệnh đề đúng, ví dụ $x=\dfrac{1}{2}$.\\
			Phủ định là 	$\forall x \in \mathbb{R}: x\le x^2$
		\end{enumerate}
	}
\end{vd}

\begin{vd}
	Điều chỉnh các mệnh đề sau để được các mệnh đề đúng:
	\begin{enumerate}[a)]
		\item $\forall x \in \mathbb{R}: 3x-1=0$.
		\item $\forall x \in \mathbb{R}: x^2-4x=0$.
		\item $\exists x \in \mathbb{R}: x^2+1<0$.
		\item $\forall x \in \mathbb{R}: x >\dfrac{1}{x}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item $\exists x \in \mathbb{R}: 3x-1=0$.
			\item $\exists x \in \mathbb{R}: x^2-4x=0$.
			\item $\exists x \in \mathbb{R}: x^2+1>0$ hoặc $\forall x \in \mathbb{R}: x^2+1>0$.
			\item $\exists x \in \mathbb{R}: x >\dfrac{1}{x}$.
	\end{enumerate}}
\end{vd}	
\begin{vd}
	Chứng minh \textquotedblleft Nếu $n^2$ là số chẵn thì $n$ là số chẵn.\textquotedblright
	\loigiai{
		\\Giả sử $n$ là số lẻ $\Rightarrow n=2k+1$, $k\in \mathbb{N}$
		\\ \phantom{Giả sử $n$ là số lẻ} $\Rightarrow n^2=4k^2+4k+1=2\left(2k^2+2k\right)+1$
		\\ \phantom{Giả sử $n$ là số lẻ} $\Rightarrow n^2$ là số lẻ (trái giả thiết).\\
		Vậy $n$ là số chẵn. }
\end{vd}
\begin{vd}
	Chứng minh rằng: \begin{enumerate}[a)]
		\item Với mọi số nguyên $n$ thì $n^3-n$ chia hết cho $3$.
		\item Với mọi số nguyên $n$ thì $n(n-1)(2n-1)$ chia hết cho $6$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item Ta có: $n^3-n=n(n^2-1)=n(n-1)(n+1)=(n-1)n(n+1)$.\\
			Do $n-1$, $n$, $n+1$ là 3 số nguyên liên tiếp nên có 1 số chia hết cho 3. \\
			Khi đó $(n-1)n(n+1)$ chia hết cho 3 hay $n^3-n$ chia hết cho $3$.
			\item Ta có $n-1$, $n$ là 2 số nguyên liên tiếp nên tích $n(n-1)(2n-1)$ chia hết cho 2.\\
			Xét 3 số nguyên liên tiếp $n-1$, $n$, $n+1$, trong 3 số này có ít nhất 1 số chia hết cho 3.
			\begin{itemize}
				\item Nếu 1 trong 2 số $n-1$, $n$ cho hết cho 3 thì tích $n(n-1)(2n-1)$ chia hết cho 3.
				\item Nếu $n+1$ chia hết cho 3 thì  $2n-1=2(n+1)-3$ cũng chia hết cho 3. Suy ra tích $n(n-1)(2n-1)$ chia hết cho 3.
			\end{itemize}
			Vậy tích $n(n-1)(2n-1)$ vừa chia hết cho 2 vừa chia hết cho 3 nên chia hết cho 6.
		\end{enumerate}
	}
	
\end{vd}

\begin{bt}%[Điệp Nguyễn]%[0D1Y1]
	Hãy xét tính đúng - sai của các mệnh đề sau đây và tìm mệnh đề phủ định của chúng:
	\begin{enumerate}[a)]
		\item $A:\text{ \textquotedblleft} \forall x \in \mathbb{R}: x^2>1\text{\textquotedblright}.$
		\item  $B:\text{ \textquotedblleft} \exists x \in \mathbb{Z}: 6x^2-13x+6=0\text{\textquotedblright}.$
		\item  $C:\text{ \textquotedblleft} \forall x \in \mathbb{N},\exists y \in \mathbb{N}:y=x+2\text{\textquotedblright}.$
		\item  $D:\text{ \textquotedblleft} \forall x \in \mathbb{R},\forall y \in \mathbb{R}:\dfrac{x}{y}+\dfrac{y}{x}\ge 0\text{\textquotedblright}.$
	\end{enumerate}
	\loigiai{ 
		\begin{enumerate}[a)]
			\item Mệnh đề sai, ví dụ như $x=0$.\\
			Phủ định là $\overline{A}:\text{\textquotedblleft}\exists x \in \mathbb{R}: x^2\le1\text{\textquotedblright}.$
			\item Mệnh đề sai vì $6x^2-13x+6=0\Leftrightarrow \left[\begin{aligned}
			&x=\dfrac{3}{2}\\
			&x=\dfrac{2}{3}
			\end{aligned}\right.$, cả hai nghiệm đều không thuộc $\mathbb{Z}$.\\
			Phủ định là $\overline{B}:\text{ \textquotedblleft} \forall x \in \mathbb{Z}: 6x^2-13x+6\neq 0\text{\textquotedblright}.$
			\item Mệnh đề đúng.\\
			Phủ định là $\overline{C}:\text{ \textquotedblleft} \exists x \in \mathbb{N},\forall y \in \mathbb{N}:y\neq x+2\text{\textquotedblright}.$
			\item Mệnh đề sai, ví dụ $x=1, y=-2$.\\
			Phủ định là $\overline{D}:\text{ \textquotedblleft} \exists x \in \mathbb{R},\exists y \in \mathbb{R}:\dfrac{x}{y}+\dfrac{y}{x}< 0\text{\textquotedblright}.$
	\end{enumerate}}
\end{bt}
\begin{bt}%[Điệp Nguyễn]%[0D1B1]
	Xét tính đúng - sai của các mệnh đề sau. Nếu mệnh đề sai hãy sửa lại cho đúng:	
	\begin{enumerate}[a)]
		\item $\forall x\in \mathbb{R}:x>4\Rightarrow x>16$.
		\item $\forall x\in \mathbb{R}:x^2>36\Rightarrow x>6$.
		\item $\left\{\begin{aligned}
		&ax^2+bx+c=0\\
		&a\neq 0
		\end{aligned}\right.$ có nghiệm kép $\Leftrightarrow \Delta = b^2-4ac=0$. 
		\item $\forall a,b,c\in \mathbb{R}: \left\{\begin{aligned}
		&a>b\\
		&b>c
		\end{aligned}\right.\Leftrightarrow a>c$.
		\item $\forall a,b \in \mathbb{Z}: \left\{\begin{aligned}
		&a\,\vdots\, 3\\
		&b\,\vdots\, 2
		\end{aligned}\right.\Leftrightarrow ab\,\vdots\, 6$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item Mệnh đề đúng.
			\item Mệnh đề sai, ví dụ $x=-7$.\\
			Sửa lại là  $\forall x\in \mathbb{R}:x>6\Rightarrow x^2>36$ hoặc  $\exists x\in \mathbb{R}:x^2>36\Rightarrow x>6$.
			\item Mệnh đề đúng.
			\item Mệnh đề $\left\{\begin{aligned}
			&a>b\\
			&b>c
			\end{aligned}\right.\Rightarrow a>c$ là đúng.\\
			Mệnh đề $a>c\Rightarrow \left\{\begin{aligned}
			&a>b\\
			&b>c
			\end{aligned}\right. $ là sai, vì dụ như $a=3, c=1, b=0$.\\
			Như vậy mệnh đề  $\left\{\begin{aligned}
			&ax^2+bx+c=0\\
			&a\neq 0
			\end{aligned}\right.$ có nghiệm kép $\Leftrightarrow \Delta = b^2-4ac=0$ là sai.\\
			Sửa lại mệnh đề đúng là $\forall a,b,c\in \mathbb{R}: \left\{\begin{aligned}
			&a>b\\
			&b>c
			\end{aligned}\right.\Rightarrow a>c$.
			\item Mệnh đề $\left\{\begin{aligned}
			&a\,\vdots\, 3\\
			&b\,\vdots\, 2
			\end{aligned}\right.\Rightarrow ab\,\vdots\, 6$ là đúng.\\
			Mệnh đề $ab\,\vdots\, 6\Rightarrow \left\{\begin{aligned}
			&a\,\vdots\, 3\\
			&b\,\vdots\, 2
			\end{aligned}\right.$ là sai, ví dụ như $a=6, b=1$.\\
			Như vậy mệnh đề $\forall a,b \in \mathbb{Z}: \left\{\begin{aligned}
			&a\,\vdots\, 3\\
			&b\,\vdots\, 2
			\end{aligned}\right.\Leftrightarrow ab\,\vdots\, 6$ là sai.\\
			Sửa lại mệnh đề đúng là $\forall a,b \in \mathbb{Z}: \left\{\begin{aligned}
			&a\,\vdots\, 3\\
			&b\,\vdots\, 2
			\end{aligned}\right.\Rightarrow ab\,\vdots\, 6$
	\end{enumerate}}
\end{bt}
\begin{bt}%[Điệp Nguyễn]%[0D1B1]
	Xét tính đúng - sai các mệnh đề sau và tìm mệnh đề phủ định của chúng:
	\begin{enumerate}[a)]
		\item $\forall a \in \mathbb{R}, \forall b \in \mathbb{R}: \left(a+b\right)^2=a^2-2ab+b^2$.
		\item $\forall a \in \mathbb{R}, \forall b \in \mathbb{R}: a^2+2>b^2+1$.
		\item $\exists a \in \mathbb{R}, \exists b \in \mathbb{R}: a+b>1$.
		\item $\exists a \in \mathbb{R}, \forall b \in \mathbb{R}: a^2<b$.
		\item $\forall a \in \mathbb{R}, \exists b \in \mathbb{R}: a^2=b+1$.
		\item $\forall a, b,c \in \mathbb{R}$ mà $a+b+c=0$ thì $-\dfrac{a^2+b^2+c^2}{2}=ab+bc+ca$.
	\end{enumerate}	
	\loigiai{
		\begin{enumerate}[a)]
			\item Mệnh đề sai vì $(a+b)^2=a^2-2ab+b^2$.\\
			Phủ định là $\exists a \in \mathbb{R}, \exists b \in \mathbb{R}: \left(a+b\right)^2\neq a^2-2ab+b^2$.
			\item Mệnh đề sai, ví dụ $a=0, b=2$.\\
			Phủ định là $\exists a \in \mathbb{R}, \exists b \in \mathbb{R}: a^2+2\le b^2+1$.
			\item  Mệnh đề đúng.\\
			Phủ định là $\forall a \in \mathbb{R}, \forall b \in \mathbb{R}: a+b\le 1$.
			\item Mệnh đề sai, ví dụ $a=3, b=1$.\\
			Phủ định là $\forall a \in \mathbb{R}, \exists b \in \mathbb{R}: a^2\ge b$.
			\item Mệnh đề đúng, số $b$ xác định bởi $b=a^2-1, \forall a \in \mathbb{R}$.\\
			Phủ định là  $\exists a \in \mathbb{R}, \forall b \in \mathbb{R}: a^2\neq b+1$.
			\item Mệnh đề đúng vì $a+b+c=0\Leftrightarrow (a+b+c)^2=0\Leftrightarrow a^2+b^2+c^2+2(ab+bc+ac)=0$\\
			$\Leftrightarrow -\dfrac{a^2+b^2+c^2}{2}=ab+bc+ca$.\\
			Phủ định là $\exists a, b,c \in \mathbb{R}$ mà $a+b+c\neq 0$ thì $-\dfrac{a^2+b^2+c^2}{2}\neq ab+bc+ca$.
	\end{enumerate}}
\end{bt}

\begin{bt}%[Điệp Nguyễn]%[0D1K1]\text{~}
	\begin{enumerate}[a)]
		\item Nếu $a+b<2$ thì một trong hai số $a$ và $b$ nhỏ hơn 1.
		\item Nếu $x\ne -1$ và $y\ne -1$ thì $x+y+xy\ne -1$.
		\item Nếu tích của hai số tự nhiên là một số lẻ thì tổng của chúng là một số chẵn.
		\item Nếu ${{x}^{2}}+{{y}^{2}}=0$ thì $x = 0$ và $y = 0$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item Giả sử $a\ge 1$ và $b \ge 1$, suy ra $a+b\ge 2$ (trái giả thiết).\\
			Vậy nếu $a+b<2$ thì một trong hai số $a$ và $b$ nhỏ hơn 1.
			\item Giả sử: $x+y+xy=1\Rightarrow x+1+y+xy=0\Rightarrow (x+1)(y+1)=0\Rightarrow \left[ \begin{aligned}
			&x=-1\\
			&y=-1
			\end{aligned} \right.$ (trái giả thiết).\\
			Vậy nếu $x\ne -1$ và $y\ne -1$ thì $x+y+xy\ne -1$.
			\item Giả sử tổng $a+b$ là số lẻ thì một trong hai số $a$, $b$ có 1 số là số lẻ còn số còn lại là số chẵn nên tích $a.b$ là số chẵn (trái giả thiết).\\
			Vậy nếu tích của hai số tự nhiên là một số lẻ thì tổng của chúng là một số chẵn.
			\item Giả sử $x \neq 0$ hoặc $y\neq 0$.
			\begin{itemize}
				\item Nếu $x\neq 0\Rightarrow x^2>0\Rightarrow x^2+y^2>0$ (trái giả thiết).
				\item Nếu $y\neq 0\Rightarrow y^2>0 \Rightarrow x^2+y^2>0 $ (trái giả thiết).
			\end{itemize}
			Vậy nếu ${{x}^{2}}+{{y}^{2}}=0$ thì $x = 0$ và $y = 0$.
		\end{enumerate}
	}
\end{bt}

			\item Giả sử $n^2+n+1$ chia hết cho $9$, khi đó $n^2+n+1=9k$, với $k$ là số nguyên. Như vậy phương trình $n^2+n+1-9k=0 \quad(1)$ sẽ có nghiệm nguyên.\\ Xét $\Delta=1-4(1-9k)=36k-3=3(12k-1)$. Ta thấy $\Delta$ chia hết cho $3$, $12k-1$ không chia hết cho $3$ nên $\Delta$ không chia hết cho $9$, do đó $\Delta$ không là số chính phương nên phương trình $(1)$ không có nghiệm nguyên (mâu thuẫn giả thiết).

Như vậy phương trình $n^2+11n+39-49k=0 \quad(1)$ sẽ có nghiệm nguyên.\\ Xét $\Delta=11^2-4(39-49k)=196k-35=7(28k-5)$. 			\\Vậy $n^2+11n+39$ không chia hết cho $49$.
		\end{enumerate}
	}
\end{bt}

		\item Tất cả bài tập trong sách này đều dễ.
		\item Có ít nhất một hình thang nội tiếp được trong đường tròn.
		\item $``\exists x\in \mathbb{R},x+3=5".$
		\item $``\forall x\in \mathbb{R},x>5".$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Tồn tại một bài tập trong sách không dễ.
			\item Mọi hình thang đều không nội tiếp được trong đường tròn.
			\item $``\forall x\in \mathbb{R},x+3 \ne 5".$
			\item $``\exists x \in \mathbb{R}, x\le 5".$
		\end{enumerate}
	}
\end{vd}

\begin{bt}%[0D1B1]%[Nguyễn Viết Sinh]

		\item Tồn tại một số thực không có nghịch đảo.

			\item $``\forall n \in \mathbb{Z}, n \in \mathbb{N}".$
			\item $``\exists n\in \mathbb{N}, n \notin \mathbb{Z}".$
			\item $``\forall n \in \mathbb{N}, n \in \mathbb{R}".$

\begin{bt}%[0D1G1]%[Nguyễn Viết Sinh]
	Xét xem các mệnh đề sau đúng hay sai và nêu các mệnh đề phủ định của chúng.
	\begin{enumerate}
		\item $``\forall x\in \mathbb{R}, x^2-7x+15>0".$
		\item $``\exists x \in \mathbb{R},x^3+2x^2+8x+16=0".$
		\item $``\forall x\in \mathbb{R},\forall y\in \mathbb{R}, 2x+3y=5".$
		\item $``\exists x\in \mathbb{R},\exists y\in \mathbb{R}, x^2+y^2-2x-4y=-1".$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có:\\ $x^2-7x+15=x^2-2.\dfrac{7}{2}.x+\dfrac{49}{4}+15-\dfrac{49}{4}=\left( x-\dfrac{7}{2}\right)^2+\dfrac{11}{4}\ge \dfrac{11}{4}>0\quad \forall x\in \mathbb{R}$.\\
			Vậy mệnh đề đúng.\\
			Mệnh đề phủ định: $``\exists x\in \mathbb{R},x^2-7x+15\le 0".$
			\item $\exists x=-2\in \mathbb{R}, (-2)^3+2.(-2)^2+8.(-2)+16=0 \Rightarrow$ Mệnh đề đúng.\\
			Mệnh đề phủ định: $``\forall x\in \mathbb{R}, x^3+2x^2+8x+16\ne 0".$
			\item $\exists x=0\in \mathbb{R},\exists y=0\in \mathbb{R}, 2.0+3.0=0\ne 0\Rightarrow$ Mệnh đề sai.\\
			Mệnh đề phủ định: $``\exists x\in \mathbb{R},\exists y\in \mathbb{R}, 2x+3y\ne 0".$
			\item $\exists x=1\in \mathbb{R}, \exists y=0 \in \mathbb{R}, 1^2+0^2-2.1-4.0=-1 \Rightarrow$ Mệnh đề đúng.\\
			Mệnh đề phủ định: $``\forall x\in \mathbb{R},\forall y\in \mathbb{R}, x^2+y^2-2x-4y=-1".$
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[0D1G1]%[Nguyễn Viết Sinh]
	Chứng minh rằng: Nếu nhốt 25 con thỏ vào 6 cái chuồng thì sẽ có ít nhất một chuồng chứa nhiều hơn 4 con thỏ.
	\loigiai{
		Ta định nghĩa mệnh đề $Q$.\\
		$Q:$ Ít nhất 1 chuồng chứa nhiều hơn 4 con thỏ.\\
		Suy ra mệnh đề $\overline{Q}:$ Tất cả các chuồng chứa ít hơn hoặc bằng 4 con thỏ.\\
		Giả sử mệnh đề $\overline{Q}$ đúng, tức là tất cả các chuồng chứa ít hơn hoặc bằng 4 con thỏ. Khi đó số thỏ sẽ có tối đa là $4.6=24$ con, mâu thuẫn với giả thiết số thỏ là 25 con.\\ Suy ra mệnh đề $\overline{Q}$ sai, do đó mệnh đề $Q$ đúng.\\
		Vậy nếu nhốt 25 con thỏ vào 6 cái chuồng thì sẽ có ít nhất 1 chuồng chứa nhiều hơn 4 con thỏ.
	}
\end{bt}

\begin{bt}%[0D1G1]%[Nguyễn Viết Sinh]
	Cho các mệnh đề chứa biến $P(n):$ ``$n$ là số chẵn'' và $Q(n):``7n+4$ là số chẵn''.
	\begin{enumerate}
		\item Phát biểu và chứng minh mệnh đề $`` \forall n \in \mathbb{N}, P(n) \Rightarrow Q(n)".$
		\item Phát biểu và chứng minh mệnh đề đảo của mệnh đề ở câu $1.$
	\end{enumerate} 
	\loigiai{
		\begin{enumerate}
			\item Với mọi số tự nhiên $n$, nếu $n$ là số chẵn thì $3n+4$ cũng là số chẵn.\\
			Chứng minh:\\
			Với mọi số tự nhiên $n$ chẵn, ta có: $3n$ và $4$ là các số chẵn. Suy ra $3n+4$ là một số chẵn.\\
			Vậy mệnh đề đúng.
			\item Với mọi số tự nhiên $n$, nếu $3n+4$ là số chẵn thì $n$ cũng là số chẵn.\\
			Chứng minh:\\
			Với mọi số tự nhiên $n$ mà $3n+4$ là số chẵn thì ta suy ra $3n$ là số chẵn (do $4$ là số chẵn). Khi đó $n$ là một số chẵn.\\
			Vậy mệnh đề đảo đúng.
		\end{enumerate}
	}
\end{bt}
