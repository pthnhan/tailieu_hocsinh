	Lập phương trình parabol có đỉnh $I(-4;19)$ và cắt đường thẳng $y=-6$ tại điểm có hoành độ bằng $1$.
	\loigiai{
	Giả sử parabol cần tìm là $(\mathscr{P})$ có phương trình là $y=ax^2+bx+c$ với $a\ne 0$. Khi đó, theo giả thiết ta có
	$$\heva{&-\dfrac{b}{2a}=-4\\& I\in (\mathscr{P})\\ &(1;-6)\in (\mathscr{P})} \Leftrightarrow \heva{&b=8a\\&16a-4b+c=19\\&a+b+c=-6}\Leftrightarrow \heva{&b=8a\\&-16a+c=19\\&9a+c=-6}\Leftrightarrow \heva{a&=-1\\b&=-8\\c&=3.}$$
	
	Vậy phương trình parabol cần tìm là $y=-x^2-8x+3$.
	}	
\end{bt}

\begin{bt}%[Kiểm tra GHKI, THPT Trần Phú - Hồ Chí Minh, 2018 - 2019]%[Phạm Đức, EX-DT-10,11]%[0D2K1-4]
	Xét tính chẵn, lẻ của hàm số $y=\dfrac{x^2+1}{|x-4|-|x+4|}$.
	\loigiai{Đặt $f(x)=\dfrac{x^2+1}{|x-4|-|x+4|}$.
	Điều kiện xác định 
	\begin{align*}
		|x-4|-|x+4|\ne 0&\Leftrightarrow |x-4|\ne |x+4|\\ &\Leftrightarrow (x-4)^2\ne (x+4)^2\\ &\Leftrightarrow -8x\ne 8x\\ &\Leftrightarrow x\ne 0
	\end{align*}
	Vậy tập xác định của hàm số là $\mathscr{D}=\mathbb{R}\setminus \{0\}$.
	\begin{itemize}
		\item Ta có $x\in \mathscr{D} \Rightarrow -x\in \mathscr{D}$.
		\item $f(-x)=\dfrac{(-x)^2+1}{|-x-4|-|-x+4|}=\dfrac{x^2+1}{|x+4|-|x-4|}=-\dfrac{x^2+1}{|x-4|-|x+4|}=-f(x)$.
	\end{itemize}
	Vậy hàm số đã cho là hàm số lẻ.
	}	
\end{bt}

\begin{bt}%[Kiểm tra GHKI, THPT Trần Phú - Hồ Chí Minh, 2018 - 2019]%[Phạm Đức, EX-DT-10,11]%[0D2K1-2]
	Tìm $a$ để hàm số $y=\dfrac{4x^2+1}{\sqrt{2a+7-x}(x-5-a^2)}-\sqrt{x+8-a}$ xác định trên nửa khoảng $(-2;5]$.
	\loigiai{
	Ta có điều kiện xác định:
	
	$$\heva{x+8-a&\ge 0\\2a+7-x&>0\\x-5-a^2&\ne 0}\Leftrightarrow \heva{x&\ge a-8 \\x&<2a+7\\x&\ne 5+a^2}\Leftrightarrow \heva{&a-8\le x<2a+7\\&x\ne 5+a^2.}$$
	Hàm số đã cho xác định trên nửa khoảng $(-2;5]$ khi và chỉ khi $$\heva{-2&\ge a-8\\ 5&< 2a+7\\ 5+a^2& \notin (-2;5]}\Leftrightarrow \heva{&a\le 6\\ &a> -1\\& \hoac{5+a^2&\le -2\\5+a^2&>5}}\Leftrightarrow\heva{-1< a\le 6\\ a\ne 0.}$$
	Vậy với $a\in (-1;6]\setminus \{0\}$ thì hàm số đã cho xác định trên $(-2;5]$.
	}	
\end{bt}

\begin{bt}%[Kiểm tra GHKI, THPT Trần Phú - Hồ Chí Minh, 2018 - 2019]%[Phạm Đức, EX-DT-10,11]%[0H1K3-4]
	Cho hình chữ nhật $ABCD$ có $AB=6$ cm, $AD=8$ cm. Gọi $I$ là trung điểm của đoạn thẳng $CD$, $G$ là trọng tâm tam giác $BCI$.
	\begin{enumerate}
		\item Phân tích vectơ $\overrightarrow{BI}$, $\overrightarrow{AG}$ theo hai vectơ $\overrightarrow{AB}$ và $\overrightarrow{AD}$.
		\item Tính $|6\overrightarrow{AG}+\overrightarrow{AD}|$.
	\end{enumerate}
	\loigiai{
	
		\begin{center}
			\begin{tikzpicture}[scale=.7, font=\footnotesize,line join=round, line cap=round,>=stealth]
		\tkzDefPoints{0/0/A, 0/-6/B, 8/0/D}
		\coordinate (C) at ($(B)+(D)-(A)$);
		\coordinate (I) at ($(C)!.5!(D)$);
		\coordinate (K) at ($(B)!.5!(I)$);
		\coordinate (G) at ($(C)!.67!(K)$);
		%\coordinate (C) at ($(B)+(D)-(A)$);
		%\coordinate (O) at ($(B)!0.5!(D)$);
		\tkzDrawSegments(A,B B,C C,D D,A K,C A,K A,I)
		\tkzDrawSegments[->](A,G B,I)
		\tkzDrawPoints[fill=black](A,B,C,D,I,K,G)
		\tkzMarkSegments[mark=|](D,I I,C)
		\tkzMarkSegments[mark=||](B,K K,I)
		\tkzLabelPoints[left](A,B)
		\tkzLabelPoints[right](D,I,C)
		\tkzLabelPoints[below](K,G)
	\end{tikzpicture}
		\end{center}
	\begin{enumerate}
		\item Phân tích vectơ $\overrightarrow{BI}$, $\overrightarrow{AG}$ theo hai vectơ $\overrightarrow{AB}$ và $\overrightarrow{AD}$.
		\begin{itemize}
			\item Ta có
			$
			\overrightarrow{BI}= \overrightarrow{BC}+\overrightarrow{CI}=\overrightarrow{AD}+\dfrac{1}{2}\overrightarrow{CD}=-\dfrac{1}{2}\overrightarrow{AB}+\overrightarrow{AD}.$\\
			Vậy $\overrightarrow{BI}=-\dfrac{1}{2}\overrightarrow{AB}+\overrightarrow{AD}.$
			\item Gọi $K$ là trung điểm $BI$. Ta có
			\begin{align*}
			\overrightarrow{AG}&= \overrightarrow{AK}+\overrightarrow{KG}\\
			&=\dfrac{1}{2}\left( \overrightarrow{AB}+\overrightarrow{AI}\right) +\dfrac{1}{3}\overrightarrow{KC}\\
			&=\dfrac{1}{2}\overrightarrow{AB}+\dfrac{1}{2}(\overrightarrow{AD}+\overrightarrow{DI})+\dfrac{1}{3}\cdot \left(- \dfrac{1}{2}\right)  \left( \overrightarrow{CI}+\overrightarrow{CB}\right) \\
			&=\dfrac{1}{2}\overrightarrow{AB}+\dfrac{1}{2}\overrightarrow{AD}+\dfrac{1}{4}\overrightarrow{AB}+\dfrac{1}{3}\cdot \dfrac{1}{2}\overrightarrow{AB}+\dfrac{1}{6}\overrightarrow{AD}\\
			&=\dfrac{5}{6}\overrightarrow{AB}+\dfrac{2}{3}\overrightarrow{AD}.
			\end{align*}
			Vậy $\overrightarrow{AG}=\dfrac{5}{6}\overrightarrow{AB}+\dfrac{2}{3}\overrightarrow{AD}.$
		\end{itemize}
		\item Tính $|6\overrightarrow{AG}+\overrightarrow{AD}|$.\\
		Theo câu a) ta có $\overrightarrow{AG}=\dfrac{5}{6}\overrightarrow{AB}+\dfrac{2}{3}\overrightarrow{AD}$ nên
		$|6\overrightarrow{AG}+\overrightarrow{AD}|=|5\overrightarrow{AB}+4\overrightarrow{AD}+\overrightarrow{AD}|=5|\overrightarrow{AB}+\overrightarrow{AD}|=5|\overrightarrow{AC}|.$\\
		Trong tam giác vuông $ABC$ có $AC^2=AB^2+BC^2=100 \Rightarrow AC=10.$ Vậy 
		$|6\overrightarrow{AG}+\overrightarrow{AD}|=5|\overrightarrow{AC}|=10.$
	\end{enumerate}}	
\end{bt}

\begin{bt}%[Kiểm tra GHKI, THPT Trần Phú - Hồ Chí Minh, 2018 - 2019]%[Phạm Đức, EX-DT-10,11]%[0H1K3-5]
	Cho tam giác $ABC$ và các điểm $M$, $N$ xác định bởi $\overrightarrow{BM}=\overrightarrow{BC}-2\overrightarrow{AB}$ và $\overrightarrow{CN}=-\dfrac{1}{2}\overrightarrow{AC}-\overrightarrow{BC}$. Chứng minh ba điểm $A$, $M$, $N$ thẳng hàng.
	\loigiai{
	Từ đẳng thức $\overrightarrow{BM}=\overrightarrow{BC}-2\overrightarrow{AB}$ ta có
	$\overrightarrow{BM}-\overrightarrow{BC}=-2\overrightarrow{AB}
	\Leftrightarrow  \overrightarrow{CM}=-2\overrightarrow{AB} \quad (1)$\\
	Từ đẳng thức $\overrightarrow{CN}=-\dfrac{1}{2}\overrightarrow{AC}-\overrightarrow{BC}$ ta có
	$\overrightarrow{BC}+\overrightarrow{CN}=-\dfrac{1}{2}\overrightarrow{AC}
	\Leftrightarrow  \overrightarrow{BN}=-\dfrac{1}{2}\overrightarrow{AC} \Leftrightarrow  2\overrightarrow{BN}=-\overrightarrow{AC} \quad (2)$.\\
	Cộng vế theo vế hai đẳng thức $(1)$ và $(2)$ ta được
	\begin{align*}
		\overrightarrow{CM}+2\overrightarrow{BN}=-2\overrightarrow{AB}-\overrightarrow{AC}
		\Leftrightarrow & 2\overrightarrow{AB}+2\overrightarrow{BN}=-\overrightarrow{CM}-\overrightarrow{AC}\\
		\Leftrightarrow & 2\overrightarrow{AN}=\overrightarrow{MC}+\overrightarrow{CA}\\
		\Leftrightarrow & 2\overrightarrow{AN}=\overrightarrow{MA}.
	\end{align*}
	Vậy ba điểm $A$, $M$, $N$ thẳng hàng.
	}	
\end{bt}
