\begin{ex}%[Đề thi chuyên Toán Tin, Sở Giáo dục Hà Nội năm 2005]%[9D5G2]
    Cho $P = \left(a + b\right)\left(b + c\right)\left(c + a\right) - abc$ với $a$, $b$, $c$ là các số nguyên. Chứng minh rằng nếu $a + b + c$ chia hết cho $4$ thì $P$ chia hết cho $4$.
    \loigiai{
        Ta có: $a+b+c\equiv 0\pmod{4}\Rightarrow a+b\equiv -c\pmod{4}$
        
        Tương tự: $b+c\equiv -a\pmod{4}, c+a\equiv -b\pmod{4}$.
        
        Suy ra: $P\equiv \left(-c\right)\cdot\left(-a\right)\cdot\left(-b\right)-abc\equiv -2abc\pmod{4}$.
        
        Mặt khác, do $a+b+c$ chia hết cho 4 nên trong 3 số $a,b,c$ có ít nhất một số chẵn.
        
        Khi đó $2abc$ chia hết cho 4, hay $P$ chia hết cho 4.
    }
\end{ex}
\begin{ex}%[Đề thi chuyên Toán Tin, Sở Giáo dục Hà Nội năm 2005]%[9D4G7]
	Cho hệ phương trình: $\heva{&\left(x+y\right)^4+13=6x^2y^2+m \\& xy\left(x^2+y^2\right)=m}$
	\begin{enumerate}
    	\item Giải phương trình với $m = -  10$.
    	\item Chứng minh rằng không tồn tại giá trị của $m$ để hệ có nghiệm duy nhất.
    \end{enumerate}
	\loigiai{
	    \begin{enumerate}
	        \item Với $m=-10$, ta nhận được hệ $\heva{&\left(x+y\right)^4+23=6x^2y^2 \\& xy\left(x^2+y^2\right)=-10}$
	        
	        Đặt $u=\left(x+y\right)^2, v=xy$, hệ phương trình trở thành: $\heva{&u^2-6v^2+23=0 &(1) \\&uv-2v^2=-10 &(2)}$
	        
	        Từ phương trình $(2)$ ta có: $u=\dfrac{2v^2-10}{v}$.
	        
	        Thay vào phương trình $(1)$ ta được:
	        
	        $\left(\dfrac{2v^2-10}{v}\right)^2-6v^2+23=0\Leftrightarrow 2v^4+17v^2-100=0\Leftrightarrow v^2=4\Leftrightarrow \hoac{&v=2 \\&v=-2}$
	        
	        Với $v=2$, ta có: $u=-1$, loại vì $u=\left(x+y\right)^2\ge0$.
	        
	        Với $v=-2$, ta có: $u=1$, khi đó ta có: $\left(x+y\right)^2=1\Leftrightarrow \heva{&x+y=1 \\&x+y=-1}$
	        
	        Giải hệ $\heva{&x+y=1 \\&xy=-2}$ ta được các nghiệm $(x;y)=(-1;2)$ và $(x;y)=(2;-1)$.
	        
	        Giải hệ $\heva{&x+y=-1 \\&xy=-2}$ ta được các nghiệm $(x;y)=(1;-2)$ và $(x;y)=(-2;1)$.
	        
	        Vậy với $m=-10$, hệ có nghiệm $(x;y)\in\left\{(-1;2),(2;-1),(1;-2),(-2;1)\right\}$.
	        \item Nhận thấy nếu $(x_0;y_0)$ là nghiệm của hệ thì $(-x_0;-y_0)$ cũng là nghiệm của hệ.
	        
	        Do đó hệ có nghiệm duy nhất thì $x_0=y_0=0$.
	        
	        Thay nghiệm $x_0=y_0=0$ vào hệ đã cho ta được $\heva{&m=13 \\&m=0}$, vô lý.
	        
	        Vậy không tồn tại giá trị của $m$ để hệ có nghiệm duy nhất.
	    \end{enumerate}
    }
\end{ex}
\begin{ex}%[Đề thi chuyên Toán Tin, Sở Giáo dục Hà Nội năm 2005]%[9D5G1]
	Cho ba  số dương $x$, $y$, $z$ thỏa mãn hệ thức $\dfrac{1}{x} + \dfrac{2}{y} + \dfrac{3}{z} = 6$. Xét hệ thức $P = x + y^2 + z^3$.
	\begin{enumerate}
		\item Chứng minh  $P \geq x + 2y + 3z - 3$.
		\item Tìm giá trị nhỏ nhất cuả $P$.
	\end{enumerate}
	\loigiai{
	    \begin{enumerate}
	        \item Áp dụng bất đẳng thức AM-GM, ta có:
	        $$y^2+1\ge 2y; z^3+1+1\ge 3z$$
	        Suy ra: $P=x+y^2+z^3\ge x+\left(2y-1\right)+\left(3z-2\right)=x+2y+3z-3$.
	        \item Áp dụng bất đẳng thức Cauchy-Schwarz ta có:
	        $$\left(x+2y+3z\right)\left(\dfrac{1}{x}+\dfrac{2}{y}+\dfrac{3}{z}\right)\ge \left(\sqrt{x}\cdot\sqrt{\dfrac{1}{x}}+\sqrt{2y}\cdot\sqrt{\dfrac{2}{y}}+\sqrt{3z}\cdot\sqrt{\dfrac{3}{z}}\right)^2=36$$
	        Khi đó: $x+2y+3z\ge 6$.
	        
	        Suy ra: $P\ge x+2y+3z-3\ge 3$.
	        
	        Dấu bằng xảy ra khi và chỉ khi: $\heva{&x=y=z \\&\dfrac{1}{x}+\dfrac{2}{y}+\dfrac{3}{z}=6}\Leftrightarrow x=y=z=1$.
	        
	        Vậy $\min P=3\Leftrightarrow x=y=z=1$.
	    \end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi chuyên Toán Tin, Sở Giáo dục Hà Nội năm 2005]%[9H3G7]%[9D5G1]
    Cho tam giác $ABC$, lấy $3$ điểm $D$, $E$, $F$ theo thứ tự trên cạnh $BC$, $CA$, $AB$ sao cho $AEDF$ là tứ giác nội tiếp. Trên tia $AD$ lấy điểm $P$ ($D$ nằm giữa $A$ và $P$)  sao cho $DA\cdot DP = DB \cdot DC$.
    \begin{enumerate}	
        \item Chứng minh tứ giác $ABPC$ nội tiếp.   
        \item Chứng minh hai tam giác $DEF$ và tam giác $PCB$ đồng dạng với nhau.  
        \item  Gọi $S$ và $S_1$ lần lượt là diện tích tam giác $ABC$  và tam giác $DEF$. Chứng minh rằng $\dfrac{S_1}{S} \leq \left(\dfrac{EF}{2AD}\right)^2$.
    \end{enumerate}
	\loigiai{
	    \begin{tikzpicture}[thick]
            \tkzInit[ymin=-3.5,ymax=7.5,xmin=-4,xmax=11]
            \tkzClip
            \tkzDefPoints{1/6/A,0/0/B,9/0/C,7/1.5/E,3/0/D}
            \tkzDrawCircle[circum](A,B,C)
            \tkzDrawCircle[circum](A,E,D)
            \tkzCircumCenter(A,E,D)\tkzGetPoint{I}
            \tkzInterLC(A,B)(I,A)\tkzGetFirstPoint{F}
            \tkzCircumCenter(A,B,C)\tkzGetPoint{O}
            \tkzInterLC(A,D)(O,A)\tkzGetFirstPoint{P}
            \tkzDrawSegments(A,B B,C C,A A,P E,F D,E D,F B,P C,P)
            \tkzDrawPoints[fill=blue](A,B,C,D,E,F,P)
            \tkzLabelPoints[right](C,E)
            \tkzLabelPoints[below left](D,F,P,B)
            \tkzLabelPoints[above left](A)
            \tkzMarkAngle[arc=l,size=1 cm,fill=red,opacity=0.5](F,A,D)
            \tkzMarkAngle[arc=l,size=1 cm,fill=red,opacity=0.5](F,E,D)
            \tkzMarkAngle[arc=l,size=1 cm,fill=red,opacity=0.5](B,C,P)
            \tkzMarkAngle[arc=l,size=1 cm,fill=yellow,opacity=0.5](D,A,E)
            \tkzMarkAngle[arc=l,size=1 cm,fill=yellow,opacity=0.5](D,F,E)
            \tkzMarkAngle[arc=l,size=1 cm,fill=yellow,opacity=0.5](P,B,C)
        \end{tikzpicture}
	    \begin{enumerate}
	        \item Từ $DA\cdot DP=DB\cdot DC$, suy ra: $\dfrac{DP}{DB}=\dfrac{DC}{DA}$
	        
	        Mà $\widehat{BDP}=\widehat{ADC}$ nên $\triangle BDP\sim \triangle ADC$, dẫn đến $\widehat{DPB}=\widehat{DCA}$ suy ra tứ giác $ABPC$ nội tiếp.
	        \item Vì các tứ giác $AEDF$ và $ABPC$ nội tiếp nên $\widehat{DEF}=\widehat{DAF}=\widehat{BCP}$ và $\widehat{DFE}=\widehat{DAE}=\widehat{CBP}$.
	        
	        Từ đó suy ra: $\triangle DEF\sim\triangle PCB$.
	        \item Vì $\triangle DEF\sim\triangle PCB$ nên $S_1=\dfrac{EF^2}{BC^2}\cdot S_{PCB}=\dfrac{EF^2}{BC^2}\cdot \dfrac{DP}{DA}\cdot S$
	        
	        Lại có: $\dfrac{1}{BC^2}=\dfrac{1}{\left(BD+DC\right)^2}\le\dfrac{1}{4BD\cdot DC}=\dfrac{1}{4DA\cdot DP}$
	        
	        Suy ra: $S_1\le \dfrac{EF^2}{4DA\cdot DP}\cdot \dfrac{DP}{DA}\cdot S\leq \left(\dfrac{EF}{2AD}\right)^2\cdot S$
	    \end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi chuyên Toán Tin, Sở Giáo dục Hà Nội năm 2005]%[9D5G4]
	Cho hình vuông $ABCD$ và $2005$ đường thẳng đồng thời thỏa mãn hai điều kiện sau:
    \begin{enumerate}	
    	\item Mỗi đường thẳng đều cắt hai cạnh đối của hình vuông.
    	\item  Mỗi đường thẳng đều chia hình vuông thành hai phần có tỉ số diện tích là $0{,}5$.
    \end{enumerate}
    Chứng minh rằng trong $2005$ đường thẳng đó có ít nhất $502$ đường đồng quy.
	\loigiai{
		\immini{
		    Đặt $AB=AD=a$.
		
		    Gọi $EF,HK$ là hai trục đối xứng của hình vuông $ABCD$ mà $EF\parallel AD$ và $HK\parallel AB$. Giả sử một đường thẳng cắt các đoạn thẳng $AD,BC,EF$ lần lượt tại $G,J,M$ sao cho $S_{CDGJ}=2S_{ABJG}$.
		    
		    Từ đó ta có: $DG+CJ=2(AG+BJ)$ hay $(a-AG)+(b-BJ)=2AG+2BJ$.
		
		    Từ đó $AG+BJ=\dfrac{2a}{3}$ hay $EM=\dfrac{a}{3}$. Suy ra $M$ là điểm cố định.
		
		    Lấy điểm $N$ trên $EF$, các điểm $P,Q$ trên $HK$ sao cho $EM=MN=NF=HP=PQ=QK=\dfrac{a}{3}$.
		}
		    {
		    \begin{tikzpicture}[thick]
		        \tkzInit[ymin=-1,ymax=7,xmin=-1,xmax=7]
                \tkzClip
                \tkzDefPoints{0/6/A,6/6/B,6/0/C,0/0/D,3/6/E,3/0/F,0/3/H,6/3/K,3/4/M,3/2/N,2/3/P,4/3/Q,-4/3/X,10/5/Y}
                \tkzInterLL(X,M)(A,D)\tkzGetPoint{G}
                \tkzInterLL(X,M)(B,C)\tkzGetPoint{J}
                \tkzDrawPoints[fill=blue](M,N,P,Q)
                \tkzDrawSegments(A,B B,C C,D D,A E,F H,K X,Y)
                \tkzLabelPoints[above](E,Q)
                \tkzLabelPoints[below](F,P)
                \tkzLabelPoints[above left](A,G,M)
                \tkzLabelPoints[above right](B)
                \tkzLabelPoints[below right](J,K,C,N)
                \tkzLabelPoints[below left](H,D)
            \end{tikzpicture}
		}
		Lập luận tương tự như trên thì các đường thẳng thỏa mãn điều kiện của đề bài phải đi qua một trong 4 điểm cố định $M,N,P,Q$.
		
		Áp dụng nguyên lý Dirichlet, từ $2005$ đường thẳng thỏa mãn điều kiện của đề bài phải có ít nhất $\left\lfloor \dfrac{2005}{4}\right\rfloor+1=502$ đưởng thẳng đi qua một trong 4 điểm trên, nghĩa là $502$ đường thẳng đó đồng quy.
	}
\end{ex}
