Cho các tập hợp $M=(- \infty;-3) \cup (3;+\infty)$; $N=(-2;5]$. Tìm $M \cup N$; $M \setminus N$.
	\loigiai{
	$M \cup N = (- \infty;-3) \cup (3;+\infty) \cup (-2;5]= (-\infty ; + \infty).$\\
	$M \setminus N = (- \infty;-3)\cup (5;+\infty).$
	}

\begin{bt}%[Thi GHK1, Trần Đại Nghĩa - Hồ Chí Minh, 2018]%[Lê Ngọc Hiếu, EX-DT-10&11]%[0D1B4-1]
Cho các tập hợp $A=[m-1;m+2]$; $B=(-3;4]$. Tìm tất cả các giá trị của $m$ để $A \cap B = \varnothing$.	
	\loigiai{
		$$A \cap B = \varnothing \Leftrightarrow \hoac{m+2 &\leq -3\\ m-1 &> 4 } \Leftrightarrow \hoac{m &\leq -5 \\ m &>5.}$$
		Vậy $\hoac{m &\leq -5 \\ m &>5}$ thỏa yêu cầu đề bài.
	}
\end{bt}

\begin{bt}%[Thi GHK1, Trần Đại Nghĩa - Hồ Chí Minh, 2018]%[Lê Ngọc Hiếu, EX-DT-10&11]%[0D2K1-2] 
	Tìm tập xác định của hàm số $y=\dfrac{\sqrt{x+2}}{|x+3|-2}-\sqrt[3]{x^2 -25}.$
	\loigiai{
	Điều kiện
	\begin{align*}
	\heva{&x+2 \geq 0 \\ &|x+3|-2 \neq 0}
	\Leftrightarrow \heva{x &\geq -2 \\ x &\neq -1 \\ x &\neq -5.}
	\end{align*}	
	Tập xác định $\mathscr{D} = \left [-2;+\infty \right ) \setminus \{-1;-5\}$.}
\end{bt}

\begin{bt}%[Thi GHK1, Trần Đại Nghĩa - Hồ Chí Minh, 2018]%[Lê Ngọc Hiếu, EX-DT-10&11]%[0H1K3-4]%[0H1K3-4]  
	Cho tam giác $ABC$ vuông tại $A$. Gọi $M$, $N$ lần lượt là trung điểm $AB$, $AC$.
	\begin{enumerate}
		\item Tính $\vec{CM}$ theo $\vec{AB}$ và $\vec{AC}$.
		\item Biết $AB=6$ cm, $\sin C = \dfrac{1}{2}$. Tính $\left |\vec{CM}+\vec{BN}\right |$.
	\end{enumerate}
	\loigiai{
		\begin{center}		
			\begin{tikzpicture}
			\tkzDefPoints{2/3/A,0/0/B,7/0/C}
			\coordinate (M) at ($(A)!1/2!(B)$);
			\coordinate (N) at ($(A)!1/2!(C)$);
			\tkzDrawSegments(A,B B,C C,A C,M)
			\tkzLabelPoints[left](B)
			\tkzLabelPoints[above right](N)
			\tkzLabelPoints[above left](M)
			\tkzLabelPoints[above](A)
			\tkzLabelPoints[below](C)
			\tkzMarkRightAngles(B,A,C)
			\tkzMarkAngles(A,C,B)
			\tkzDrawPoints(A,B,C,M,N)
			\end{tikzpicture}
		\end{center}
\begin{enumerate}
	\item Tính $\vec{CM}$ theo $\vec{AB}$ và $\vec{AC}$.\\
	$\vec{CM}=\vec{AM}-\vec{AC}=\dfrac{1}{2}\vec{AB}-\vec{AC}.$
	\item Biết $AB=6$ cm, $\sin C = \dfrac{1}{2}$. Tính $|\vec{CM}+\vec{BN}|$.\\
	$\sin C= \dfrac{1}{2}\Rightarrow \widehat{C}= 30 ^ \circ.$\\
	$AC=\dfrac{AB}{\tan C}=\dfrac{6}{\dfrac{\sqrt{3}}{3}}=6\sqrt{3}.$\\
	$\vec{BN}=\vec{AN}-\vec{AB}=\dfrac{1}{2}\vec{AC}-\vec{AB}.$\\
	$\left |\vec{CM}+\vec{BN}\right |^2=\left |-\dfrac{1}{2}\vec{AB}-\dfrac{1}{2}\vec{AC}\right |^2= \left |\dfrac{1}{2}\vec{AB}+\dfrac{1}{2}\vec{AC}\right| ^2 \\ 
	=\dfrac{1}{4}\left (AB^2+ 2 \vec{AB} \cdot \vec{AC} + AC^2 \right )=\dfrac{1}{4}\left (6^2+(6\sqrt{3})^2\right )=36$.\\
	Vậy $\left |\vec{CM}+\vec{BN}\right |=6$.
\end{enumerate}

	}
\end{bt}

\begin{bt}%[Thi GHK1, Trần Đại Nghĩa - Hồ Chí Minh, 2018]%[Lê Ngọc Hiếu, EX-DT-10&11]%[0D2K1-4] 
	Xét tính chẵn, lẻ của hàm số $y=h(x)=\sqrt{x+2}+\sqrt{2-x}+\dfrac{2}{x^2 -1}$.
	\loigiai{
	Tập xác định $\mathscr{D}=[-2;2] \setminus \{-1;1\}$.\\
	$\forall x \in \mathscr{D} \Rightarrow -x \in \mathscr{D}.$
	\\
	$h(-x)=\sqrt{-x+2}+\sqrt{2-(-x)}+\dfrac{2}{(-x)^2 -1}=\sqrt{2-x}+\sqrt{x+2}+\dfrac{2}{x^2 -1}=h(x).$\\
	Vậy $y=h(x)$ là hàm số chẵn.	
	}
\end{bt}

\begin{bt}%[Thi GHK1, Trần Đại Nghĩa - Hồ Chí Minh, 2018]%[Lê Ngọc Hiếu, EX-DT-10&11]%[0H1K3-4]%[0H1G3-5]
	Cho tam giác $ABC$ và các điểm $J$ ,$K$ lần lượt thỏa mãn $\vec{JA}+4\vec{JC}=\vec{0}$; $\vec{CB}=-\dfrac{3}{4}\vec{BK}$.
	\begin{enumerate}
		\item Biểu diễn $\vec{JK}$ theo $\vec{AB}$ và $\vec{AC}$.
		\item Gọi $I$ là trung điểm $AB$. Chứng minh $I,J,K$ thẳng hàng.
	\end{enumerate} 
	\loigiai{\begin{center}		
		\begin{tikzpicture}
\tkzDefPoints{2/3/A,0/0/B,7/0/C}
\coordinate (J) at ($(A)!4/5!(C)$);
\coordinate (K) at ($(B)!4/3!(C)$);
\coordinate (I) at ($(A)!1/2!(B)$);
\tkzDrawSegments(A,B B,C C,A C,K)
\tkzLabelPoints[left](B)
\tkzLabelPoints[right](K)
\tkzLabelPoints[above right](J)
\tkzLabelPoints[above left](I)
\tkzLabelPoints[above](A)
\tkzLabelPoints[below](C)
\tkzDrawPoints(A,B,C,J,K,I)
\end{tikzpicture}
	\end{center}
		\begin{enumerate}
			\item Biểu diễn $\vec{JK}$ theo $\vec{AB}$ và $\vec{AC}$.\\
			$\vec{JK}=\vec{JC}+\vec{CK}=\dfrac{1}{5}\vec{AC}+\dfrac{1}{3}\vec{BC}=\dfrac{1}{5}\vec{AC}+\dfrac{1}{3} (\vec{AC}-\vec{AB})=\dfrac{-1}{3}\vec{AB}+\dfrac{8}{15}\vec{AC}.$
			\item Gọi $I$ là trung điểm $AB$. Chứng minh $I,J,K$ thẳng hàng.
			$I$ là trung điểm $AB$ nên\\
			$2\vec{JI}=\vec{JA}+\vec{JB}
			=\dfrac{-4}{5}\vec{AC}+\vec{JC}+\vec{BC}
			=\dfrac{-4}{5}\vec{AC}+\dfrac{1}{5}\vec{AC}+\vec{AB}-\vec{AC}
			=\dfrac{-8}{5}\vec{AC}+\vec{AB}.$\\
			Suy ra $\dfrac{-2}{3}\vec{JI}=\dfrac{-1}{3}\vec{AB}+\dfrac{8}{15}\vec{AC}=\vec{JK}.$\\
			Vậy $I,J,K$ thẳng hàng.
		\end{enumerate} 
	}
\end{bt}

	\loigiai{
		Xét $g(x)=x^2+6x+11=(x+3)^2+2 > 0, \forall x \in (-3; + \infty)$.\\
		Tọa độ đỉnh $I(-3;2)$.\\
		Bảng biến thiên
		\begin{center}
\begin{tikzpicture}
\tkzTabInit[lgt=1.2,espcl=3]
{$x$ /1.2, $g(x)$ /2.5}
{$-3$,$+\infty$}
\tkzTabVar{-/$2$,+/$+\infty$}
\end{tikzpicture}
\end{center}
Hàm số $g(x)$ đồng biến trên $(-3;+ \infty)$.\\
Suy ra $g(x_1)> g(x_2),  \forall x_1 > x_2 > -3$\\
$\Rightarrow \sqrt{g(x_1)}> \sqrt{g(x_2)},  \forall x_1 > x_2 > -3$\\
$\Rightarrow f(x_1)> f(x_2),  \forall x_1 > x_2 > -3$\\
Vậy hàm số $f(x)$ đồng biến trên $(-3;+ \infty)$.
	}

\begin{bt}%[Thi GHK1, Trần Đại Nghĩa - Hồ Chí Minh, 2018]%[Lê Ngọc Hiếu, EX-DT-10&11]%[0D2G1-4]
	Chứng minh đồ thị của hàm số $y=h(x)=\dfrac{|x-1|-|x+1|}{2-|x|}$ nhận gốc tọa độ $O$ làm tâm đối xứng.
	\loigiai{
	Tập xác định $\mathscr{D}= \mathbb{R} \setminus \{-2;2\}$.\\
	$\forall x \in \mathscr{D} \Rightarrow -x \in \mathscr{D}$.\\
	$h(-x)=\dfrac{|-x-1|-|-x+1|}{2-|-x|}=\dfrac{|x+1|-|x-1|}{2-|x|}=-\left (\dfrac{|x-1|-|x+1|}{2-|x|}\right )=-h(x).$	\\
	$y=h(x)$ là hàm số lẻ nên nhận gốc tọa độ $O$ làm tâm đối xứng.
	}
\end{bt}

\begin{bt}%[Thi GHK1, Trần Đại Nghĩa - Hồ Chí Minh, 2018]%[Lê Ngọc Hiếu, EX-DT-10&11]%[0H1K3-4]%[0H1G3-5]
	Cho tam giác $ABC$ có $M$ là trung điểm $AB$; $N$ và $E$ lần lượt thỏa mãn $\vec{NA}+2\vec{NC}=\vec{0}$; $3\vec{EB}+4\vec{EC}=\vec{0}$.
	\begin{enumerate}
		\item Biểu diễn $\vec{AE}$ theo $\vec{AB}$ và $\vec{AC}$.
		\item Tìm điểm $I$ trên $MN$ sao cho $A,E,I$ thẳng hàng.
	\end{enumerate} 
	\loigiai{\begin{center}		
			\begin{tikzpicture}
			\tkzDefPoints{2/3/A,0/0/B,7/0/C}
			\coordinate (N) at ($(A)!2/3!(C)$);
			\coordinate (E) at ($(B)!4/7!(C)$);
			\coordinate (M) at ($(A)!1/2!(B)$);
			\coordinate (I) at ($(A)!7/12!(E)$);
			\tkzDrawSegments(A,B B,C C,A A,E M,N)
			\tkzLabelPoints[left](B)
			\tkzLabelPoints[above right](N,I)
			\tkzLabelPoints[above left](M)
			\tkzLabelPoints[above](A)
			\tkzLabelPoints[below](E,C)
			\tkzDrawPoints(A,B,C,M,N,E,I)
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Biểu diễn $\vec{AE}$ theo $\vec{AB}$ và $\vec{AC}$.\\
			$3\vec{EB}+4\vec{EC}=\vec{0}$\\
			$\Rightarrow 4\vec{EC}=3\vec{BE}$\\
			$\Rightarrow \vec{BC}=\vec{BE}+\vec{EC}=\vec{BE}+\dfrac{3}{4}\vec{BE}=\dfrac{7}{4}\vec{BE}$\\
			$\Rightarrow\vec{BE}=\dfrac{4}{7}\vec{BC}.$
			$\vec{AE}=\vec{AB}+\vec{BE}=\vec{AB}+\dfrac{4}{7}\vec{BC}=\vec{AB}+\dfrac{4}{7}(\vec{AC}-\vec{AB})=\dfrac{3}{7}\vec{AB}+\dfrac{4}{7}\vec{AC}$\\		
			$\Rightarrow \vec{AE}=\dfrac{3}{7}\vec{AB}+\dfrac{4}{7}\vec{AC}.$
			\item Tìm điểm $I$ trên $MN$ sao cho $A,E,I$ thẳng hàng.\\
			$\heva{I \in MN &\Rightarrow \exists a \neq 0 : \vec{MI}=a \vec{MN}\\
				I \in AE &\Rightarrow \exists b \neq 0 : \vec{AI}=b \vec{AE}}$\\
			$\Rightarrow \vec{AI}-\vec{AM}=a\left (\vec{AN}-\vec{AM}\right )$\\
			$\Rightarrow b \vec{AE}-\dfrac{1}{2}\vec{AB}=a\left (\dfrac{2}{3} \vec{AC}-\dfrac{1}{2}\vec{AB}\right )$\\
				$\Rightarrow b\left (\dfrac{3}{7}\vec{AB}+\dfrac{4}{7}\vec{AC}\right )-\dfrac{1}{2}\vec{AB}=\dfrac{2a}{3}\vec{AC}-\dfrac{a}{2}\vec{AB}$\\
				$\Rightarrow \heva{&\dfrac{3}{7}b-\dfrac{1}{2}=\dfrac{-1}{2}a\\
				&\dfrac{4}{7}b=\dfrac{2}{3}a} \Leftrightarrow \heva{a&=\dfrac{1}{2}\\ b&=\dfrac{7}{12}.}$
		\end{enumerate}
Vậy $I \in AE$ sao cho $\vec{AI}=\dfrac{7}{12} \vec{AE}$ thỏa yêu cầu đề bài.	}
\end{bt}
