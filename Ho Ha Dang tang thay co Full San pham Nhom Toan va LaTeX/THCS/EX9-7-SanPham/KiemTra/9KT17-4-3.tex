\begin{ex}%[9D1B1]%[9D1B7]
Cho biểu thức $A=\left(\dfrac{\sqrt{x}}{\sqrt{x}-1}-\dfrac{1}{x-\sqrt{x}}\right):\left(\dfrac{1}{\sqrt{x}+1}+\dfrac{2}{x-1}\right)$
\begin{enumerate}
\item Nêu điều kiện xác định và rút gọn $A.$
\item Tìm các giá trị của $x$ sao cho $A<0.$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Điều kiện xác định $\heva{& \sqrt{x}-1\neq 0\\& x\neq 0\\& x\geq 0}\Rightarrow$ $x>0$ và $x\neq 1.$\\
Khi đó 
\begin{align*}
& A=\left(\dfrac{\sqrt{x}}{\sqrt{x}-1}-\dfrac{1}{\sqrt{x}(\sqrt{x}-1)}\right):\left(\dfrac{1}{\sqrt{x}+1}+\dfrac{2}{(\sqrt{x}-1)(\sqrt{x}+1)}\right)
\\& =\dfrac{x-1}{\sqrt{x}(\sqrt{x}-1)}:\dfrac{\sqrt{x}-1+2}{(\sqrt{x}-1)(\sqrt{x}+1)}
\\& = \dfrac{x-1}{\sqrt{x}(\sqrt{x}-1)}.\dfrac{(\sqrt{x}-1)(\sqrt{x}+1)}{\sqrt{x}+1} =\dfrac{x-1}{\sqrt{x}}
\end{align*}
\item Với $x>0$ và $x\neq 1$, ta có $A<0\Leftrightarrow\dfrac{x-1}{\sqrt{x}}<0\Leftrightarrow x-1<0$ (vì $\sqrt{x}>0$) $\Leftrightarrow x<1$.\\
Đối chiếu với điều kiện ta có với $0<x<1$ thì $A<0.$
\end{enumerate}
}
\end{ex}

\begin{ex}%[9D3B5]
Một xe khách và một xe du lịch khởi hành đồng thời từ $A$ đến $B$ dài $100$ km. Xe du lịch có vận tốc lớn hơn vận tốc xe khách là $20$ $\mathrm{km}/\mathrm{h}$, nên nó đến $B$ trước xe khách $25$ phút. Tính vận tốc mỗi xe.
\loigiai{
Gọi vận tốc của xe khách là $x$ ($\mathrm{km}/\mathrm{h}$), điều kiện $x>0.$\\
Khi đó vận tốc của xe du lịch là $x+20$ ($\mathrm{km}/\mathrm{h}$).\\
Thời gian xe khách đi hết quãng đường $AB$ là $\dfrac{100}{x}$ h.\\
Thời gian xe du lịch đi hết quãng đường $AB$ là $\dfrac{100}{x+20}$ h.\\
Theo bài ra ta có phương trình $\dfrac{100}{x}-\dfrac{100}{x+20}=\dfrac{5}{12}\Leftrightarrow x^2+20x-4800=0\Rightarrow\heva{&x_1=60\\&x_2=-80}$\\
Vậy vận tốc xe khách là $60$ $\mathrm{km}/\mathrm{h}$. Vận tốc xe du lịch là $80$ $\mathrm{km}/\mathrm{h}$.
}
\end{ex}

\begin{ex}%[9D4B6]
Cho phương trình $x^2-(m+2)x-m-3=0$ $(1)$, với $m$ là tham số.
\begin{enumerate}
\item Giải phương trình $(1)$ khi $m=1.$
\item Tìm $m$ để phương trình $(1)$ có hai nghiệm $x_1$, $x_2$ thỏa mãn $x_1^2+x_2^2>1.$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Với $m=1$ phương trình $(1)$ trở thành $x^2-3x-4=0$.\\
Ta có $a-b+c=0$ nên phương trình có hai nghiệm $x_1=-1$, $x_2=4.$
\item Ta có $\Delta=[-(m+2)]^2-4(-m-3)=m^2+8m+16=(m+4)^2\geq 0$, $\forall m.$\\
Phương trình luôn có nghiệm với mọi $m$, giả sử $x_1$, $x_2$ là hai nghiệm của phương trình.\\
Khi đó theo Vi-ét ta có $\heva{& x_1+x_2=m+2\\& x_1 x_2=-m-3}$\\
Theo bài ra ta có $x_1^2+x_2^2>1\Leftrightarrow (x_1+x_2)^2-2x_1.x_2>1\Leftrightarrow (m+2)^2-2(-m-3)>1$\\
$\Leftrightarrow (m+3)^2>0.$\\
Do $(m+3)^2\geq 0$ $\forall m$ nên $(m+3)^2>0\Leftrightarrow m\neq 3 $\\
Vậy với $m\neq 3$ thì phương trình $(1)$ có hai nghiệm $x_1$, $x_2$ thỏa mãn $x_1^2+x_2^2>1.$
\end{enumerate}
}
\end{ex}

\begin{ex}%[9H3B7]
Cho điểm $M$ nằm ngoài $(O;R)$ vẽ các tiếp $MA$, $MB$ với $(O;R)$. Vẽ đường kính $AC$, tiếp tuyến tại $C$ của đường tròn $(O;R)$ cắt $AB$ ở $D$. Chứng minh rằng
\begin{enumerate}
\item Tứ giác $MAOB$ nội tiếp.
\item $AB.AD=4R^2.$
\item $OD$ vuông góc với $MC.$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item \immini{
Xét tứ giác $MAOB$ có $\widehat{MAO}=90^\circ$ (vì $MA$ là tiếp tuyến của $(O;R)$), $\widehat{MBO}=90^\circ$ (vì $MB$ là tiếp tuyến của $(O;R)$).\\
Do đó $\widehat{MAO}+\widehat{MBO}=180^\circ.$\\
Vậy tứ giác $MAOB$ nội tiếp.
\item Ta có $\widehat{ACD}=90^\circ$ (vì $DC$ là tiếp tuyến của $(O;R)$).\\
Góc $\widehat{ABC}=90^\circ$ (góc nội tiếp chắn nửa đường tròn $(O;R)$).\\
$\Rightarrow \triangle ACD$ vuông tại $C$ có đường cao $CB$.\\
Áp dụng hệ thức lượng về cạnh và đường cao trong tam giác vuông ta có $AB.AD=AC^2=(2R)^2=4R^2$.
}{
\begin{tikzpicture}[scale=1.3]
\tkzInit[xmin=-5,xmax=2,ymin=-3,ymax=4]
\tkzClip
\tkzDefPoints{0/0/O, 2/0/R, -4/0/M}
\tkzDrawCircle[radius](O,R)
\tkzTangent[from=M](O,R)\tkzGetPoints{A}{B}
\coordinate (C) at ($(A)!2!(O)$);
\tkzTangent[at=C](O)\tkzGetPoint{d}
\tkzInterLL(A,B)(C,d)\tkzGetPoint{D}
\tkzInterLL(M,O)(A,B)\tkzGetPoint{I}
\tkzInterLL(M,C)(O,D)\tkzGetPoint{H}
\tkzDrawSegments(M,A M,B M,O A,C A,D C,D B,O B,C O,D M,C)
\tkzDrawPoints(O,A,B,M,D,C,H,I)
\tkzLabelPoints[above](D,C,M)
\tkzLabelPoints[below](A)
\tkzLabelPoints[below right](O,H)
\tkzLabelPoints[above left](B,I)
\end{tikzpicture}
}
\item Gọi $I$ là giao điểm của $MO$ và $AB$, $H$ là giao điểm của $MC$ và $OD.$\\
 Ta có $\widehat{MAO}=\widehat{ACD}=90^\circ$, $\widehat{AMO}=\widehat{CAD}$ (cùng phụ với $\widehat{MAI}$).\\
$\Rightarrow \triangle MAO\backsim\triangle ACD$ (g.g)\\
$\Rightarrow \dfrac{MA}{AC}=\dfrac{AO}{CD}$ mà $AO=CO$ nên $\dfrac{MA}{AC}=\dfrac{CO}{CD}$\\ 
$\widehat{MAC}=\widehat{OCD}=90^\circ\Rightarrow\triangle MAC\backsim\triangle OCD$ (c.g.c).\\
$\Rightarrow\widehat{ACM}=\widehat{ODC}$ mà $\widehat{MCD}=\widehat{AMC}$ (so le trong)\\
$\Rightarrow\triangle MAC\backsim\triangle CDH$ (g.g) $\Rightarrow\widehat{DAC}=\widehat{MAC}=90^\circ\Rightarrow OD\perp MC.$
\end{enumerate}
}
\end{ex}

\begin{ex}%[9D5G1]
Cho các số dương $a$, $b$, $c$. Chứng minh bất đẳng thức $\sqrt{\dfrac{a}{b+c}}+\sqrt{\dfrac{b}{a+c}}+\sqrt{\dfrac{c}{a+b}}>2.$
\loigiai{
Theo bất đẳng thức cô-si ta có $\sqrt{\dfrac{b+c}{a}.1}\leq \left(\dfrac{b+c}{a}+1\right):2=\dfrac{b+c+a}{2a}$. Do đó $\sqrt{\dfrac{a}{b+c}}\geq \dfrac{2a}{a+b+c}$\\
Tương tự $\sqrt{\dfrac{b}{a+c}}\geq \dfrac{2b}{a+b+c}$ và $\sqrt{\dfrac{c}{a+b}}\geq \dfrac{2c}{a+b+c}.$\\
Cộng từng vế $\sqrt{\dfrac{a}{b+c}}+\sqrt{\dfrac{b}{a+c}}+\sqrt{\dfrac{c}{a+b}}\geq \dfrac{2(a+b+c)}{a+b+c}=2.$\\
Dấu $"="$ xảy ra khi và chỉ khi $\heva{& a=b+c\\& b=a+c\\& c=a+b}\Leftrightarrow a+b+c=0$, trái với giả thiết $a$, $b$, $c$ là các số dương, do đó dấu $"="$ không thể xảy ra\\
Vậy $\sqrt{\dfrac{a}{b+c}}+\sqrt{\dfrac{b}{a+c}}+\sqrt{\dfrac{c}{a+b}}>2.$
}
\end{ex}
