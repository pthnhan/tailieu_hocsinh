\begin{bt}%[Đề thi HK1, 2017 - 2018 trường Phổ thông Năng khiếu, Hồ Chí Minh]%[Nguyễn Thành Khang]%[0D3B2-6]
	Tìm $m$ để phương trình $\dfrac{m}{x}+\dfrac{1}{x-m}=1$ có nghiệm.
	\loigiai{
		Điều kiện $x\ne 0, x\ne m$.\\
		Phương trình đã cho tương đương với $$m(x-m)+x=x(x-m)\Leftrightarrow x^2-(2m+1)x+m^2=0.\qquad (*)$$
		Phương trình đã cho có nghiệm khi phương trình (*) có nghiệm khác $0$ và khác $m$
		$$ \Leftrightarrow \heva{&\Delta \ge 0 \\ &m^2\ne 0 \\ &m^2-(2m+1)m+m^2\ne 0} \Leftrightarrow \heva{&(2m+1)^2-4m^2\ge 0 \\ &m^2 \ne 0 \\&-m\ne 0} \Leftrightarrow \heva{&4m+1\ge 0 \\ &m\ne 0} \Leftrightarrow \heva{&m\ge -\dfrac{1}{4} \\ &m\ne 0.} $$
		Vậy $m\in \left[-\dfrac{1}{4};+\infty\right)\setminus \{0\}$.
	}
\end{bt}

\begin{bt}%[Đề thi HK1, 2017 - 2018 trường Phổ thông Năng khiếu, Hồ Chí Minh]%[Nguyễn Thành Khang]%[0D6B2-5]
	Cho $A=3\left(\sin^3 x\cos x+\cos^3 x\sin x\right)$, $B=\sin \left(x-\dfrac{5\pi}{2}\right)\sin(x+\pi)$.\\
	Tìm số thực $C$ biết $A-\left(\sqrt{3}\tan\dfrac{4\pi}{3}\right)B+C=0$ với mọi $x$.
	\loigiai{
		Ta có $A=3\sin x\cos x(\sin^2 x+\cos^2 x)=3\sin x\cos x$\\
		và $B=\sin \left(x-\dfrac{\pi}{2}\right)(-\sin x)=\sin \left(\dfrac{\pi}{2}-x\right)\sin x=\sin x\cos x$.\\
		Khi đó $A-\left(\sqrt{3}\tan\dfrac{4\pi}{3}\right)B+C=A-3B+C=C$.\\
		Vậy để $A-\left(\sqrt{3}\tan\dfrac{4\pi}{3}\right)B+C=0$ với mọi $x$ thì $C=0$.
}
\end{bt}

\begin{bt}%[Đề thi HK1, 2017 - 2018 trường Phổ thông Năng khiếu, Hồ Chí Minh]%[Nguyễn Thành Khang]%[0D3B3-3]
	Giải hệ phương trình: $\heva{&2\sqrt{x}+\dfrac{1}{|y-1|}=7 \\ &-\sqrt{x}+\dfrac{2}{|y-1|}=4}$.
	\loigiai{
		Điều kiện: $x\ge 0, y\ne 1$.\\
		Hệ phương trình đã cho tương đương với
		\allowdisplaybreaks{
		\begin{eqnarray*}
			\heva{&2\sqrt{x}+\dfrac{1}{|y-1|}=7 \\ &-\sqrt{x}+\dfrac{2}{|y-1|}=4} &\Leftrightarrow& \heva{&2\sqrt{x}+\dfrac{1}{|y-1|}=7 \\ &-2\sqrt{x}+\dfrac{4}{|y-1|}=8} \Leftrightarrow \heva{&2\sqrt{x}+\dfrac{1}{|y-1|}=7 \\ &\dfrac{5}{|y-1|}=15} \\
			&\Leftrightarrow& \heva{&\sqrt{x}=2 \\ &\dfrac{1}{|y-1|}=3} \Leftrightarrow \heva{&x=4 \\ &|y-1|=\dfrac{1}{3}} \Leftrightarrow \hoac{&\heva{&x=4 \\ &y=\dfrac{4}{3}} \\ &\heva{&x=4 \\ &y=\dfrac{2}{3}.}}
		\end{eqnarray*}
		}
		Vậy hệ đã cho có nghiệm $(x;y)\in\left\{\left(4;\dfrac{4}{3}\right);\left(4;\dfrac{2}{3}\right)\right\}$.
}
\end{bt}

\begin{bt}%[Đề thi HK1, 2017 - 2018 trường Phổ thông Năng khiếu, Hồ Chí Minh]%[Nguyễn Thành Khang]%[0D3B2-1]%[0D3B2-4]
	Giải phương trình: $(x-2)\left(\sqrt{x^2-4x+3}-x\right)=0$.
	\loigiai{
		Điều kiện: $x^2-4x+3\ge 0 \Leftrightarrow \hoac{&x\ge 3 \\ &x\le 1}$.\\
		Phương trình đã cho tương đương với $\hoac{&x=2 &\text{(loại)} \\ &\sqrt{x^2-4x+3}=x &(1)}$.\\
		Giải phương trình $(1)$ ta được	$$(1) \Leftrightarrow \heva{&x\ge 0 \\ &x^2-4x+3=x^2} \Leftrightarrow \heva{&x\ge 0 \\ &-4x+3=0} \Leftrightarrow x=\dfrac{3}{4}.\quad \text{(thoả mãn)}$$
		Vậy phương trình đã cho có nghiệm $x=\dfrac{3}{4}$.
}
\end{bt}

\begin{bt}%[Đề thi HK1, 2017 - 2018 trường Phổ thông Năng khiếu, Hồ Chí Minh]%[Nguyễn Thành Khang]%[0D2B3-3]
	Tìm toạ độ giao điểm của $(P)\colon y=ax^2+bx+c\ (a\ne 0)$ với trục tung biết $5a-c=0$ và $I(2;1)$ là đỉnh của $(P)$.
	\loigiai{
		Toạ độ đỉnh của $(P)$ có dạng $\left(\dfrac{-b}{2a};\dfrac{-\Delta}{4a}\right)$ nên ta có $\heva{&\dfrac{-b}{2a}=2 \\ &\dfrac{-\Delta}{4a}=1} \Leftrightarrow \heva{&b=-4a \\ &\Delta = -4a}.$\\
		Mà $c=5a$ nên ta có $\Delta=(-4a)^2-4\cdot a\cdot 5a=-4a\Leftrightarrow 16a^2-16a=0 \Leftrightarrow a=1$ (vì $a\ne 0$).\\
		Khi đó $b=-4, c=5$. Dẫn đến toạ độ giao điểm của $(P)$ với trục tung là $A(0;5)$.
}
\end{bt}

\begin{bt}%[Đề thi HK1, 2017 - 2018 trường Phổ thông Năng khiếu, Hồ Chí Minh]%[Nguyễn Thành Khang]%[0D3B3-2]
	Tìm $m$ để hệ phương trình $\heva{&(m+1)x+my=2 \\ &2mx+y=m+1}$ có nghiệm.
	\loigiai{
		Với $m=0$, hệ đã cho có nghiệm $(x;y)=(2;1)$.\\
		Với $m\ne 0$, hệ đã cho tương đương với
		\begin{eqnarray*}
			\heva{&(m+1)x+my=2 \\ &2m^2x+my=m^2+m} &\Leftrightarrow& \heva{&2mx+y=m+1 \\ &\left(2m^2-m-1\right)x=m^2+m-2} \\ &\Leftrightarrow& \heva{&2mx+y=m+1 &(1) \\ &(m-1)(2m+1)x=(m-1)(m+2). &(2)}
		\end{eqnarray*}
		\begin{itemize}
			\item Với $m=1$ thì $(2)\Leftrightarrow 0x=0$, đúng với mọi $x$, hệ có vô số nghiệm.
			\item Với $m=-\dfrac{1}{2}$ thì $(2)\Leftrightarrow 0x=-\dfrac{9}{4}$, hệ vô nghiệm.
			\item Với $m\ne 1, m\ne -\dfrac{1}{2}$ thì $(2)\Leftrightarrow x=\dfrac{m+2}{2m+1}$. Thay vào $(1)$ ta được $y=\dfrac{-m+1}{2m+1}$.
		\end{itemize}
		Vậy hệ có nghiệm khi và chỉ khi $m\ne -\dfrac{1}{2}$.
	}
\end{bt}

\begin{bt}%[Đề thi HK1, 2017 - 2018 trường Phổ thông Năng khiếu, Hồ Chí Minh]%[Nguyễn Thành Khang]%[0H2K2-2]%[0H2G2-3]
	Cho hình chữ nhật $ABCD$ nội tiếp trong đường tròn tâm $O$, bán kính $R$.
	\begin{enumerate}
		\item Chứng minh rằng với mọi $M$ thuộc đường tròn $(O)$ ta có: $\vec{MA}+\vec{MB}+\vec{MC}+\vec{MD}=4\vec{MO}$. Tính $S=\vec{MA}\cdot \vec{MC}+\left(\vec{MB}+\vec{MD}\right)\left(\vec{MA}+\vec{MB}+\vec{MC}+\vec{MD}\right)$ theo $R$.
		\item Biết $AB=2AD$ và điểm $K$ thuộc cạnh $AB$ thoả mãn $\vec{BK}=x\vec{BA}$. Tìm $x$ để $CK$ vuông góc với $BD$.
	\end{enumerate}
	\loigiai{
		\immini[0.02]{
			\vspace*{-2em}
			\begin{enumerate}
				\item Vì $ABCD$ là hình chữ nhật nên $O$ là trung điểm $AC$ và $BD$. Ta có
				\begin{eqnarray*}
				& & \vec{MA}+\vec{MB}+\vec{MC}+\vec{MD}\\
				&=& \vec{MO}+\vec{OA}+\vec{MO}+\vec{OB}+\vec{MO}+\vec{OC}+\vec{MO}+\vec{OD} \\
				&=& 4\vec{MO}+\left(\vec{OA}+\vec{OC}\right)+\left(\vec{OB}+\vec{OD}\right)=4\vec{MO}.
				\end{eqnarray*}
				Vì $AC$ là đường kính của $(O)$ nên $MA\perp MC$. Suy ra $\vec{MA}\cdot \vec{MC}=0$, dẫn tới
				\begin{eqnarray*}
					S &=& \left(\vec{MB}+\vec{MD}\right)\left(\vec{MA}+\vec{MB}+\vec{MC}+\vec{MD}\right) \\
					&=& 2\vec{MO}\cdot 4\vec{MO}=8MO^2=8R^2.
				\end{eqnarray*}
			\end{enumerate}
		}{
			\begin{tikzpicture}
				\tkzInit[xmin=-0.5, xmax=5.5, ymin=-2, ymax=4.5]
				\tkzClip
				\tkzDefPoints{0/0/A,5/0/B,5/2.5/C,0/2.5/D}
				\tkzDefMidPoint(A,C)\tkzGetPoint{O}
				\tkzDrawCircle(O,A)
				\tkzDefPointBy[rotation=center O angle 70](C)\tkzGetPoint{M}
				\tkzDefPointBy[homothety=center B ratio 0.25](A)\tkzGetPoint{K}
				\tkzInterLL(C,K)(B,D)\tkzGetPoint{H}
				\tkzMarkRightAngles(C,H,B)
				\tkzDrawPoints[fill=black](A,B,C,D,O,K,M)
				\tkzDrawSegments(A,B B,C C,D D,A A,C B,D M,A M,C C,K)
				\tkzLabelPoints[above](M)
				\tkzLabelPoints[above right](C)
				\tkzLabelPoints[above left](D)
				\tkzLabelPoints[below](O,K)
				\tkzLabelPoints[below right](B)
				\tkzLabelPoints[below left](A)
			\end{tikzpicture}
		}
		\begin{enumerate}
			\item[b)] Ta có $\vec{CK}=\vec{CB}+\vec{BK}=-\vec{AD}+x\vec{BA}$ và $\vec{BD}=\vec{BA}+\vec{AD}$.\\
			Để $CK$ vuông góc với $BD$, ta có
			\begin{eqnarray*}
				& & \vec{CK}\cdot \vec{BD}=0 \\
				&\Leftrightarrow& \left(-\vec{AD}+x\vec{BA}\right)\left(\vec{BA}+\vec{AD}\right)=0 \\
				&\Leftrightarrow& xAB^2-AD^2=0 \qquad(\text{vì }AB\perp AD\Rightarrow \vec{BA}\cdot\vec{AD}=0) \\
				&\Leftrightarrow& 4xAD^2-AD^2=0 \\
				&\Leftrightarrow& x=\dfrac{1}{4}.
			\end{eqnarray*}
			Vậy $x=\dfrac{1}{4}$.
		\end{enumerate}
	}
\end{bt}
