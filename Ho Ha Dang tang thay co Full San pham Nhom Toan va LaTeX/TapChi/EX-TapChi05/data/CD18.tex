\section{Phương pháp phân ly đẳng thức}
\begin{center}\textbf{\color{violet}{Dương Đức Lâm-Trường THPT Chuyên ĐHSP Hà Nội}}\end{center}
\noindent Khi đi tìm lời giải cho một bài toán bất đẳng thức, thông thường ta hay quan tâm đến câu hỏi \textit{``Đẳng thức xảy ra khi nào?''}. Đối với nhiều bài toán bất đẳng thức, thật không ngoa khi nói rằng, tìm được điều kiện xảy ra dấu bằng là đã giải xong nửa bài toán. Tuy nhiên công việc của một nửa còn lại vẫn thật sự còn nhiều chông gai nếu dấu bằng không xảy ra tại tâm, hoặc tại nhiều điểm, hay ``biến hóa khôn lường''. Khi đó, những đánh giá của chúng ta cũng phải ``biến hóa'' theo, bởi chỉ cần một chút ``sơ ý'', một ước lượng thiếu tinh tế thôi cũng có thể làm đổ vỡ mọi cố gắng của chúng ta ngay. Để đối phó với những bài toán như vậy, chiến lược nào giúp chúng ta xử lí chúng mà không phải bận tâm nhiều đến điều kiện xảy ra đẳng thức? Quá trình tìm tòi và tìm lời giải cho nhiều bài toán bất đẳng thức cho chúng tôi một kinh nghiệm như sau: Sử dụng các phép biến đổi tương đương, ta tách các đại lượng ``nhạy cảm'', chẳng hạn tích các biến, riêng ra rồi đánh giá với các đại lượng còn lại như bình thường (một cách tự nhiên nhất, nhưng vẫn đủ tinh tế, theo một nghĩa nào đó) mà không sợ điều kiện xảy ra đẳng thức bị ảnh hưởng. Bởi lí do đó, ta sẽ đặt tên cho phương pháp này là \textit{Phương pháp phân ly đẳng thức}. Chúng ta cùng bắt đầu tìm hiểu về phương pháp này thông qua bài toán kinh điển sau đây.

\begin{bt}%[Phương pháp phân ly đẳng thức]%[Kiều Ngân, Dự án EX_TapChi05]%[0D0G2]
	Cho $a$, $b$, $c$ là các số thực không âm và không có hai số nào đồng thời bằng không. Chứng minh rằng
	$$\dfrac{a(b+c)}{b^2+bc+c^2}+\dfrac{b(c+a)}{c^2+ca+a^2}+\dfrac{c(a+b)}{a^2+ab+b^2}\geqslant 2.$$
\loigiai{
	Bài toán đẹp đẽ này được đề xuất bởi Darij Grinberg trên diễn đàn MathLinks cách đây khoảng chục năm trước. Điều đặc biệt, và cũng làm nên độ khó của bất đẳng thức này, là đẳng thức xảy ra khi $a=b=c$ (gọi là \textit{dấu bằng tại tâm}) hoặc $a=b$, $c=0$ cùng các hoán vị (gọi là \textit{dấu bằng tại biên}). Với những bất đẳng thức có điều kiện xảy ra đẳng thức như vậy (và cũng tương đối phổ biến), các đánh giá trung gian của chúng ta đều phải thỏa mãn xảy ra đẳng thức tại ít nhất hai điểm trên. Có khá nhiều cách chứng minh bất đẳng thức này. Bây giờ chúng ta sẽ tìm một cách tiếp cận khác cho bài toán này bằng phương pháp của chúng ta. Nhận thấy rằng, đẳng thức tại biên thường ``bị'' gây ra bởi tích $abc$, nên ta sẽ tìm cách ``gạt nó sang một bên'' để dễ bề ``xử lí''! Sử dụng biến đổi
	$$\dfrac{x(y+z)}{y^2+yz+z^2}=\dfrac{x(y+z)^2}{(y+z)(y^2+yz+z^2)}=\dfrac{x}{y+z}+\dfrac{xyz}{(y+z)(y^2+yz+z^2)},$$
	ta đưa bất đẳng thức về dạng
		\[\dfrac{a}{b+c}+\dfrac{b}{c+a}+\dfrac{c}{a+b}+abc\cdot \sum \dfrac{1}{(b+c)(b^2+bc+c^2)} \geqslant 2, \tag {1}\]
	ở đó ta sử dụng kí hiệu $\sum f(a,b,c)$ để chỉ các tổng hoán vị $f(a,b,c)+f(b,c,a)+f(c,a,b)$. Đến đây ta đã tách được đại lượng ``nhạy cảm'' $abc$ ra một bên. Ta cần một ước lượng cho biểu thức $\sum \dfrac{1}{(b+c)(b^2+bc+c^2)}$. Một ước lượng tốt là 
		\[\sum \dfrac{1}{(b+c)(b^2+bc+c^2)}\geqslant \dfrac{4}{(a+b)(b+c)(c+a)}, \tag{2}\]
	hay tương đương
		$$\dfrac{(a+b)(a+c)}{b^2+bc+c^2}+\dfrac{(b+c)(b+a)}{c^2+ca+a^2}+\dfrac{(c+a)(c+b)}{a^2+ab+b^2} \geqslant 4.$$
	Bạn đọc sẽ hỏi, tại sao lại nghĩ ra ước lượng này? Câu trả lời sẽ được tìm thấy bên dưới. Bây giờ ta sẽ chứng minh bất đẳng thức $(2)$. Lưu ý rằng bất đẳng thức này chỉ có dấu bằng tại tâm, như vậy ta hi vọng là việc chứng minh không khó! Quả vậy, có nhiều cách để chứng minh cho bất đẳng thức này. Sử dụng AM-GM ta có đánh giá sau
		\[(b^2+bc+c^2)(ab+bc+ca)\geqslant \dfrac{1}{4}(b+c)^2(a+b+c)^2. \tag{3}\]
	Thiết lập hai bất đẳng thức tương tự, ta suy ra $(2)$ đúng nếu chứng minh được
		$$\dfrac{(a+b)(a+c)}{(b+c)^2}+\dfrac{(b+c)(b+a)}{(c+a)^2}+\dfrac{(c+a)(c+b)}{(a+b)^2} \geqslant \dfrac{(a+b+c)^2}{ab+bc+ca}.$$
	Theo bất đẳng thức Cauchy-Schwarz thì
		\[\dfrac{a}{b+c}+\dfrac{b}{c+a}+\dfrac{c}{a+b} \geqslant \dfrac{(a+b+c)^2}{2(ab+bc+ca)}, \tag{4}\]
	nên ta chỉ cần chứng minh
		$$\dfrac{(a+b)(a+c)}{(b+c)^2}+\dfrac{(b+c)(b+a)}{(c+a)^2}+\dfrac{(c+a)(c+b)}{(a+b)^2} \geqslant \dfrac{a}{b+c}+\dfrac{b}{c+a}+\dfrac{c}{a+b},$$
	nhưng bất đẳng thức này tương đương
		$$\dfrac{(a-b)(a-c)}{(b+c)^2}+\dfrac{(b-c)(b-a)}{(c+a)^2}+\dfrac{(c-a)(c-b)}{(a+b)^2} \geqslant 0,$$
	luôn đúng theo \textit{bất đẳng thức Vornicu-Schur}\footnote{Với $a\geq b\geq c\geq 0$ và $x,y,z\geq 0$, $x\geq y$ thì
	$$x(a-b)(a-c)+y(b-c)(b-a)+z(c-a)(c-b)\geq 0.$$
	Thật vậy, từ giả thiết $x(a-c)\geq y(b-c)\geq 0$, nên 
	$$VT=(a-b)\bigl[x(a-c)-y(b-c)\bigr]+z(a-c)(b-c)\geq 0.$$
	Đây là trường hợp thường dùng nhất của bất đẳng thức Vornicu-Schur mà chúng tôi sử dụng trong bài viết này.}, thật vậy, có thể giả sử $a\geqslant b \geqslant c$, khi đó viết lại vế trái bất đẳng thức trên thành
		$$(a-b)^2 \dfrac{a^2+ab+b^2+(a+b-c)c}{(b+c)^2(c+a)^2}+\dfrac{(c-a)(c-b)}{(a+b)^2},$$
	luôn không âm. Cuối cùng, bởi ước ượng $(2)$, bất đẳng thức ban đầu sẽ đúng nếu ta chứng minh được
		$$\dfrac{a}{b+c}+\dfrac{b}{c+a}+\dfrac{c}{a+b}+\dfrac{4abc}{(a+b)(b+c)(c+a)} \geqslant 2,$$
	nhưng đây chính là dạng phân thức của \textit{bất đẳng thức Schur}\footnote{Với $a,b,c\geq 0$ và $k>0$ thì
	$$a^k(a-b)(a-c)+b^k(b-c)(b-a)+c^k(c-a)(c-b)\geq 0.$$
	Thật vậy, không mất tính tổng quát, giả sử $a\geq b\geq c$ thì $a^k\geq b^k\geq 0$ nên 
	$$VT=(a-b)\bigl[a^k(a-c)-b^k(b-c)\bigr]+c^k(a-c)(b-c)\geq 0.$$
	Trường hợp $k=1$ (bất đẳng thức Schur bậc 3) là trường hợp thường dùng nhất của bất đẳng thức Schur mà chúng tôi sử dụng trong bài viết này.}.
}
\begin{nx}
	Mặc dù ta đã sử dụng một bất đẳng thức chỉ có dấu bằng tại tâm (bất đẳng thức $(2)$), nhưng vẫn không làm mất đi trường hợp dấu bằng tại biên của bất đẳng thức cần chứng minh. Sở dĩ như vậy vì đánh giá $(2)$ được thực hiện dưới sự ``che chắn'' của đại lượng $abc$. Một cách tổng quát, khi gặp những bất đẳng thức có dấu bằng xảy ra tương tự như trên, chúng ta có thể biến đổi bất đẳng thức về dạng
		$$A(a,b,c)+abc\cdot B(a,b,c) \geqslant 0,$$
	rồi đánh giá tiếp $B(a,b,c)\geqslant C(a,b,c)$ mà đánh giá này không nhất thiết phải đảm bảo có dấu bằng tại biên.
\end{nx}
\begin{nx}
	Việc nghĩ ra đánh giá $(2)$ cũng khá tự nhiên nếu chúng ta suy ngược từ cuối như sau. Bởi bất đẳng thức Schur dạng phân thức, $(1)$ sẽ được chứng minh nếu ta chứng minh được
		$$\sum \dfrac{a}{b+c} +abc\cdot \sum \dfrac{1}{(b+c)(b^2+bc+c^2)} \geqslant \sum \dfrac{a}{b+c}+\dfrac{abc}{(a+b)(b+c)(c+a)}.$$
	Thu gọn bất đẳng thức này ta nhận được $(2)$.
\end{nx}
\begin{nx}
	Từ lời giải trên, chúng ta đã chứng minh được một bất đẳng thức mạnh hơn bất đẳng thức ban đầu là
		$$\sum \dfrac{a(b+c)}{b^2+bc+c^2} \geqslant \dfrac{a}{b+c}+\dfrac{b}{c+a}+\dfrac{c}{a+b}+\dfrac{4abc}{(a+b)(b+c)(c+a)}.$$
	Lưu ý rằng, nếu sử dụng cách đánh giá $(3)$ và sau đó $(4)$, ta nhận được một lời giải ngắn gọn cho bất đẳng thức ban đầu (bạn đọc hãy kiểm tra!). Nhưng với bài toán làm mạnh này, nếu áp dụng trực tiếp đánh giá $(3)$, thì ta cần chứng minh bất đẳng thức sau với mọi $a$, $b$, $c$ không âm
		$$\dfrac{4(ab+bc+ca)}{(a+b+c)^2}\left(\sum \dfrac{a}{b+c} \right) \geqslant \sum \dfrac{a}{b+c}+\dfrac{4abc}{(a+b)(b+c)(c+a)}.$$
	Cho $a=0$, $b=1$, $c=2$ thì vế trái (\textit{VT}) bằng $\dfrac{20}{9}$ trong khi vế phải (\textit{VP}) bằng $\dfrac{20}{8}$, bất đẳng thức này sai! Việc làm của chúng ta là vô nghĩa? Không hẳn vậy. Nếu có một chút tò mò, thử thêm vài giá trị khác, ta vẫn nhận được $VT \leqslant VP$. Vậy phải chăng bất đẳng thức ngược lại là đúng? Câu trả lời là quả thật nó đúng, và lời giải chúng tôi xin nhường lại cho bạn đọc! Điều chúng tôi muốn nhắn tới các bạn ở đây là, việc sáng tạo các bài toán không phải là điều gì đó xa vời, mà nó ẩn giấu ngay sau chính những bài toán cũ. Vấn đề là phát hiện ra nó sớm hay muộn thôi, và tất nhiên, điều đó còn phụ thuộc vào \textit{chỉ số tò mò} của bạn!
\end{nx}
\begin{nx}
	Từ lời giải bài toán và các nhận xét trên, ta có thể rút ra một dãy các mối quan hệ lí thú sau đây giữa bất đẳng thức ban đầu với bất đẳng thức sau khi làm mạnh
	 	$$\sum \dfrac{(a+b)(a+c)}{b^2+bc+c^2} \geqslant 2\sum \dfrac{a(b+c)}{b^2+bc+c^2} \geqslant 2\sum \dfrac{a}{b+c}+\dfrac{8abc}{(a+b)(b+c)(c+a)} \geqslant 4.$$
	Ý tưởng đẩy $abc$ sang bên cạnh rồi xử lí biểu thức dưới tích $abc$ cũng sử dụng được cho việc tìm lời giải bài toán cũng rất nổi tiếng sau đây.
	 
\end{nx}
\end{bt}

\begin{bt}%[Phương pháp phân ly đẳng thức]%[Kiều Ngân, Dự án EX_TapChi05]%[0D0G2]
	Chứng minh rằng với mọi $a$, $b$, $c$ là các số thực không âm và không có hai số nào đồng thời bằng không, ta có bất đẳng thức
	$$\dfrac{1}{(a+b)^2}+\dfrac{1}{(b+c)^2}+\dfrac{1}{(c+a)^2} \geqslant \dfrac{9}{4(ab+bc+ca)}.$$
	\hfill {\textit{(Iran TST 1996)}}
\loigiai{
	Bài toán này xuất hiện lần đầu tiên trên tạp chí Crux năm $1994$ do Ji Chen\footnote{Đại học Ningbo, Trung Quốc} đề nghị, sau đó lại xuất hiện trong đề thi chọn đội tuyển IMO của Iran năm $1996$. Bất đẳng thức này nổi tiếng vì vẻ đẹp, độ khó cũng như những cảm hứng mà nó mang lại. Đến nay, người ta đã tìm được rất nhiều cách chứng minh. Ta sẽ tiếp cận nó bằng con đường phân ly đẳng thức (chú ý rằng đẳng thức xảy ra khi $a=b=c$ hoặc $a=b$, $c=0$ hoặc các hoán vị). Nhân cả hai vế của bất đẳng thức với $(a+b)(b+c)(c+a)$ và để ý $(a+b)(b+c)(c+a)=(a+b+c)(ab+bc+ca)-abc$, ta nhận được 
		$$\dfrac{(a+b)(a+c)}{b+c}+\dfrac{(b+c)(b+a)}{c+a}+\dfrac{(c+a)(c+b)}{a+b} \geqslant \dfrac{9(a+b+c)}{4}-\dfrac{9abc}{4(ab+bc+ca)},$$
	hay tương đương
		$$\dfrac{(a+b)(a+c)}{b+c}+\dfrac{(b+c)(b+a)}{c+a}+\dfrac{(c+a)(c+b)}{a+b}+\dfrac{9abc}{4(ab+bc+ca)} \geqslant \dfrac{9(a+b+c)}{4}.$$
	Như vậy ta đã tách được $abc$ ra ngoài. Dưới sự ``che chắn'' của $abc$, ta có thể đánh giá $\dfrac{9}{4(ab+bc+ca)}$ một cách khá thoải mái, cụ thể là ước lượng cơ bản
		$$ab+bc+ca \leqslant \dfrac{1}{3}(a+b+c)^2,$$
	mà không sợ mất đi điều kiện dấu bằng tại biên. Khi đó ta cần chứng minh
		$$\sum \dfrac{(a+b)(a+c)}{b+c}+\dfrac{27abc}{4(a+b+c)^2} \geqslant \dfrac{9(a+b+c)}{4}.$$
	Nhân cả hai vế với $a+b+c$, thu được 
		$$\sum \dfrac{a(a+b)(a+c)}{b+c} +\sum (a+b)(a+c)+\dfrac{27abc}{4(a+b+c)} \geqslant \dfrac{9(a+b+c)^2}{4},$$
	hay 
		$$\sum \left(\dfrac{a(a+b)(a+c)}{b+c}-2a^2\right) +\dfrac{27abc}{4(a+b+c)} \geqslant \dfrac{3}{2}(ab+bc+ca)-\dfrac{3}{4}(a^2+b^2+c^2),$$
	tương đương
		$$\sum \dfrac{a(a-b)(a-c)}{b+c}+\dfrac{3}{4}\left(\dfrac{9abc}{a+b+c}-2(ab+bc+ca)+a^2+b^2+c^2\right) \geqslant 0,$$
	luôn đúng vì $\displaystyle\sum \dfrac{a(a-b)(a-c)}{b+c}\geqslant 0$ theo Vornicu-Schur, đại lượng còn lại không âm theo Schur.\\
	Cũng với cách tiếp cận này, chúng ta nhận được một lời giải ngắn gọn cho bài toán đẹp sau đây, được để xuất bởi Vasile Cirtoaje\footnote{Đại học Ploiesti, Romania}.
}
\end{bt}


\begin{bt}%[Phương pháp phân ly đẳng thức]%[Lâm Hữu Phước,Dự án EX_TapChi05]%[0D0G2]
	Chứng minh rằng với mọi $a,b,c$ là các số thực không âm, ta có bất đẳng thức
	\[a\sqrt{a^2+3bc}+b\sqrt{b^2+3ca}+c\sqrt{c^2+3ab}\geq 2(ab+bc+ca).\]
\loigiai{
Đẳng thức cũng xảy ra tại tâm và tại biên. Trừ hai vế bất đẳng thức cho $a^2+b^2+c^2$, ta có
\[\sum a\Bigl(\sqrt{a^2+3bc}-a\Bigr)\geq 2(ab+bc+ca)-\bigl(a^2+b^2+c^2\bigr).\]
Sau khi nhân liên hợp, nhận được 
\[3abc\cdot \sum\dfrac{1}{\sqrt{a^2+3bc}+a}\geq 2(ab+bc+ca)-\bigl(a^2+b^2+c^2\bigr).\]
Tiếp theo, nhờ có nhân tử $abc$ phía trước, ta đánh giá tổng $\displaystyle\sum\dfrac{1}{\sqrt{a^2+3bc}+a}$ một cách khá nhẹ nhàng bằng Cauchy--Schwarz như sau
\[\sum\dfrac{1}{\sqrt{a^2+3bc}+a}\geq \dfrac{9}{\sum\Bigl(\sqrt{a^2+3bc}+a\Bigr)}\geq \dfrac{9}{\sqrt{3\sum a^2+9\sum bc}+\sum a}\geq \dfrac{9}{3(a+b+c)}.\]
Cuối cùng ta cần chứng minh 
\[\dfrac{9abc}{a+b+c}\geq 2(ab+bc+ca)-\bigl(a^2+b^2+c^2\bigr),\]
nhưng đây chính là bất đẳng thức Schur.
}
\begin{nx}
	Chúng ta hãy cùng so sánh bài toán này với bài toán ``anh em" với nó sau đây: Chứng minh với $a,b,c$ không âm thì
	\[a\sqrt{4a^2+5bc}+b\sqrt{4b^2+5ca}+c\sqrt{4c^2+5ab}\geq (a+b+c)^2\]
	Sau khi giới thiệu nó trên diễn đàn toán học diendantoanhoc.net, chúng tôi nhận được một lời giải rất đẹp nhưng đầy tính kỹ thuật của thầy Phan Thành Nam\footnote{Đại học Cergy-Pontoise (Pháp), người có nhiều đóng góp cho phong trào Olympic toán ở Việt Nam.} (mà vì mục đích bài viết, chúng tôi không giới thiệu nó ở đây). Rõ ràng, với bài toán này thì cách giải trên không còn áp dụng được nữa. Nhưng phương pháp phân ly đẳng thức vẫn hoạt động tốt. Mời bạn đọc thử sức.
\end{nx}
\end{bt}
\begin{bt}%[Phương pháp phân ly đẳng thức]%[Lâm Hữu Phước,Dự án EX_TapChi05]%[0D0G2]
Chứng minh rằng với mọi $a,b,c$ là các số thực không âm ta có
\[\dfrac{a^2+bc}{(a+b)(a+c)}+\dfrac{b^2+ca}{(b+c)(b+a)}+\dfrac{c^2+ab}{(c+a)(c+b)}\leq \dfrac{2\bigl(a^3+b^3+c^3\bigr)}{a^3+b^3+c^3+abc}\]
\begin{flushright}
	\textit{(Tập huấn đội tuyến IMO 2014)}
\end{flushright}
\loigiai{
Với bất đẳng thức này, đẳng thức xảy ra khi $a=b=c$ hoặc $abc=0$, tức chỉ cần một trong ba biến $a,b$ hoặc $c$ bằng không là đã có đẳng thức. Như vậy phải rất cẩn thận trong đánh giá của chúng ta, nếu không điều kiện xảy ra dấu bằng của đẳng thức sẽ bị ``lung lay" ngay! Một cách tự nhiên, ta sẽ làm cách nào đó ``loại bỏ" dấu đẳng thức tại biên, bằng cách bỏ đi đại lượng $abc$ này. Nhưng loại bỏ bằng cách nào? Bởi vì chỉ cần một trong ba biến bằng không thì hai vế bằng nhau, nên nếu lấy $VT$ trừ $VP$ chắc chắn sẽ xuất hiện nhân tử $abc$. Nhưng nếu tính toán trực tiếp $VT-VP$ thì sẽ rất cồng kềnh và mất thời gian. Bây giờ ta để ý tiếp, cho $a=0$ thì cả hai vế đều bằng 2. Vậy ta sẽ không quy đồng trực tiếp $VT-VP$ mà làm gián tiếp bằng cách trừ 2 vào cả hai vế như sau
\[2-\Biggl(\sum \dfrac{a^2+bc}{(a+b)(a+c)}\Biggr)=\dfrac{4abc}{(a+b)(b+c)(c+a)},\]
và 
\[2-\dfrac{2\bigl(a^3+b^3+c^3\bigr)}{a^3+b^3+c^3+abc}=\dfrac{2abc}{a^3+b^3+c^3+abc}.\]
Suy ra bất đẳnt thức cần chứng minh trở thành
\[\dfrac{4abc}{(a+b)(b+c)(c+a)}\geq \dfrac{2abc}{a^3+b^3+c^3+abc}.\]
Loại bỏ $abc$ ở 2 vế, ta cần chứng minh
\[2\bigl(a^3+b^3+c^3+abc\bigr)\geq (a+b)(b+c)(c+a),\]
luôn đúng vì 
\[2\bigl(a^3+b^3+c^3+abc\bigr)-(a+b)(b+c)(c+a)=\sum(b+c)(b-c)^2\geq 0.\]
Bất đẳng thức được chứng minh xong.
}
\end{bt}
\begin{bt}%[Phương pháp phân ly đẳng thức]%[Lâm Hữu Phước,Dự án EX_TapChi05]%[0D0G2]
	Cho $a,b,c$ là các số thực không âm, không có hai số nào bằng không. Chứng minh rằng
	\[1\leq \dfrac{a(b+c)}{b^2+c^2+a(b+c)}+\dfrac{b(c+a)}{c^2+a^2+b(c+a)}+\dfrac{c(a+b)}{a^2+b^2+c(a+b)}\leq 1\dfrac{4abc}{(a+b)(b+c)(c+a)}\]
	\begin{flushright}
		\textit{(Tập huấn đội tuyển IMO 2014)}
	\end{flushright}
\loigiai{
Cũng như ví dụ trên, bất đẳng thức này có đẳng thức tại biên, cụ thể, dấu bằng ở bất đẳng thức bên trái xảy ra khi $abc=0$. Còn ở bất đẳng thức bên phải là $a=b=c$ hoặc $abc=0$. Lại để ý tiếp rằng biểu thức $\dfrac{4abc}{(a+b)(b+c)(c+a)}$ đã chứa nhân tử $abc$ nên ta hy vọng có thể tách loại được nhân tử này từ $\displaystyle\sum \dfrac{a(b+c)}{b^2+c^2+a(b+c)}-1$. Nhưng chớ vội quy đồng trực tiếp biểu thức này! Hãy để ý đến đẳng thức
\[\dfrac{a}{a+b+c}+\dfrac{b}{a+b+c}+\dfrac{c}{a+b+c}=1.\]
Thật thú vị bởi vì đẳng thức tưởng chừng như rất hiển hiên này lại giúp ích nhiều cho chúng ta trong việc phân tách thành nhân tử $abc$ ở trên. Thật vậy, bởi
\[\dfrac{a(b+c)}{b^2+c^2+a(b+c)}-\dfrac{a}{a+b+c}=\dfrac{2abc}{\bigl(b^2+c^2+ab+ac\bigr)(a+b+c)},\]
cùng với hai đẳng thức tương tự, thì bất đẳng thức kép cần chứng minh lúc này đã đơn giản hơn rất nhiều
\[0\leq \sum\dfrac{2abc}{\bigl(b^2+c^2+ab+ac\bigr)(a+b+c)}\leq \dfrac{4abc}{(a+b)(b+c)(c+a)}.\]
Bất đẳng thức bên trái là hiển nhiên, còn bất đẳng thức bên phải tương đương với 
\[\sum\dfrac{1}{b^2+c^2+ab+ac}\leq \dfrac{2(a+b+c)}{(a+b)(b+c)(c+a)}.\]
Ta có
\[b^2+c^2+a(b+c)\geq \dfrac{(b+c)^2}{2}+a(b+c)=\dfrac{1}{2}(b+c)(b+c+2a),\]
suy ra
\[\sum\dfrac{1}{b^2+c^2+ab+ac}\leq \sum\dfrac{2}{(b+c)(b+c+2a)}.\]
Cuối cùng ta cần chỉ ra 
\[\sum\dfrac{2}{(b+c)(b+c+2a)}\leq \dfrac{2(a+b+c)}{(a+b)(b+c)(c+a)},\]
tương đương
\[\sum\dfrac{(a+b)(a+c)}{b+c+2a}\leq a+b+c,\]
nhưng điều này là hiển nhiên theo AM -- GM
\[\sum\dfrac{(a+b)(a+c)}{b+c+2a}\dfrac{1}{4}\sum\dfrac{(2a+b+c)^2}{(b+c+2a)}=a+b+c.\]
Bất đẳng thức được chứng minh xong.
}
\end{bt}

\begin{bt}%[Phương pháp phân ly đẳng thức]%[Nguyễn Thành Tiến, Dự án EX_TapChi05]%[0D0G2]
	Cho $a, b, c$ là các số thực không âm không đồng thời bằng không. Chứng minh rằng
	\begin{eqnarray*}
		\dfrac{a^3+2abc}{a^3+\left(b+c\right)^3}+\dfrac{b^3+2abc}{b^3+\left(c+a\right)^3}+\dfrac{c^3+2abc}{c^3+\left(a+b\right)^3}\geq 1
	\end{eqnarray*}
\begin{flushright}
\it{(Tập huấn đội tuyển IMO 2014)}
\end{flushright}
\loigiai{
Lời giải này của Võ Quốc Bá Cẩn, cũng là những ý tưởng đầu tiên trong lời giải của tác giả bài toán. Bất đẳng thức đã cho tương đương với
\begin{eqnarray*}
	2abc\sum\dfrac{1}{a^3+\left(b+c\right)^3}\geq 1-\sum\dfrac{a^3}{a^3+\left(b+c\right)^3}
\end{eqnarray*}
Sử dụng đẳng thức hiển nhiên
\begin{eqnarray*}
	1=\dfrac{a^3}{a^3+b^3+c^3}+\dfrac{b^3}{a^3+b^3+c^3}+\dfrac{c^3}{a^3+b^3+c^3}
\end{eqnarray*}
ta có thể viết lại vế phải của bất đẳng thức trên thành
\begin{eqnarray*}
	\sum\left(\dfrac{a^3}{a^3+b^3+c^3}-\dfrac{a^3}{a^3+\left(b+c\right)^3}\right)=\sum\dfrac{3a^3bc\left(b+c\right)}{\left(a^3+b^3+c^3\right)\left(a^3+\left(b+c\right)^3\right)}
\end{eqnarray*}
Như vậy, chúng ta đã ta tách loại được $abc$, chỉ cần chứng minh
\begin{eqnarray*}
	2\sum\dfrac{a^3+b^3+c^3}{a^3+\left(b+c\right)^3}\geq \sum\dfrac{3a^2\left(b+c\right)}{a^3+\left(b+c\right)^3}
\end{eqnarray*}
hay
\begin{eqnarray*}
	\sum\dfrac{2a^3-3a^2\left(b+c\right)+2b^3+2c^3}{a^3+\left(b+c\right)^3}\geq 0
\end{eqnarray*}
tương đương
\begin{eqnarray*}
	\sum\dfrac{(a-b)^3-(c-a)^3+3c^2(c-a)-3b^2(a-b)}{a^3+\left(b+c\right)^3}\geq 0
\end{eqnarray*}
ta đưa bài toán về chứng minh hai bất đẳng thức sau
\begin{eqnarray*}
	\sum\dfrac{(a-b)^3-(c-a)^3}{a^3+\left(b+c\right)^3}\geq 0
\end{eqnarray*}
và 
\begin{eqnarray*}
	\sum\dfrac{c^2\left(c-a\right)-b^2\left(a-b\right)}{a^3+\left(b+c\right)^3}\geq 0
\end{eqnarray*}
Bất đẳng thức thứ nhất tương đương với
\begin{eqnarray*}
	\sum\left(a-b\right)^3\left(\dfrac{1}{a^3+\left(b+c\right)^3}-\dfrac{1}{b^3+\left(c+a\right)^3}\right)\geq 0
\end{eqnarray*}
hay
\begin{eqnarray*}
	\sum\dfrac{3c\left(a-b\right)^4\left(a+b+c\right)}{\left(a^3+\left(b+c\right)^3\right)\left(b^3+\left(c+a\right)^3\right)}\geq 0
\end{eqnarray*}
luôn đúng. Bất đẳng thức thứ hai tương đương
\begin{eqnarray*}
	\sum\left(a-b\right)\left(\dfrac{a^2}{b^3+\left(c+a\right)^3}-\dfrac{b^2}{a^3+\left(b+c\right)^3}\right)\geq 0
\end{eqnarray*}
Để khẳng định bất đẳng thức này đúng, ta chỉ cần kiểm tra với $a\geq b$ thì
\begin{displaymath}
\dfrac{a^2}{b^3+(c+a)^3}\geq \dfrac{b^2}{a^3+(b+c)^3}
\end{displaymath}
hay
\begin{displaymath}
a^5-b^5\geq b^2\left(c+a\right)^3-a^2\left(b+c\right)^3
\end{displaymath}
tương đương
\begin{displaymath}
a^5-b^5\geq a^2b^2\left(a-b\right)+c^3\left(b^2-a^2\right)+3c^2ab\left(b-a\right)
\end{displaymath}
luôn đúng với mọi $a\geq b$ và $c\geq 0$.\\
\noindent Sau khi sử dụng bài toán này để tập huấn đội tuyển IMO năm nay, chúng tôi nhận được thêm hai lời giải khác của các em học sinh, một lời giải ngắn gọn bất ngờ còn một lời giải dù khá dài nhưng dựa trên tư tưởng phân ly đẳng thức rất đáng hoan nghênh. Chúng tôi xin giới thiệu cả hai.\\
{\bf{Lời giải 2}}. Lời giải của Vương Nguyễn Thùy Dương. Ta có phân tích sau\\
\begin{eqnarray*}
	a^3+\left(b+c\right)^3=\left(a+b+c\right)\left(a^2+(b+c)^2-a(b+c)\right).
\end{eqnarray*}
Do đó
\begin{eqnarray*}
\dfrac{a^3+2abc}{a^3+\left(b+c\right)^3}-\dfrac{a}{a+b+c}&=&\dfrac{a}{a+b+c}\left(\dfrac{a^2+2bc}{a^2+(b+c)^2-a(b+c)}-1\right)\\
&=&\dfrac{a}{a+b+c}\cdot\dfrac{b(a-b)-c(c-a)}{a^2+(b+c)^2-a(b+c)}.
\end{eqnarray*}
Bất đẳng thức cần chứng minh trở thành
\begin{displaymath}
\dfrac{1}{a+b+c}\sum ab(a-b)\left(\dfrac{1}{a^2+(b+c)^2-a(b+c)}-\dfrac{1}{b^2+(c+a)^2-b(c+a)}\right)\geq 0,
\end{displaymath}
hay
\begin{displaymath}
\dfrac{3abc}{a+b+c}\sum\dfrac{\left(a-b\right)^2}{\left(a^2+(b+c)^2-a(b+c)\right)\left(b^2+(c+a)^2-b(c+a)\right)}\geq 0
\end{displaymath}
luôn đúng.\\
{\bf{Lời giải 3}}. Lời giải của bạn Nguyễn Huy Tùng. Trước hết ta phát biểu và chứng minh một bổ đề.\\
{\bf{Bổ đề.}} Với $a,\,b,\,c$ là các số không âm không đồng thời bằng không, ta có
\begin{displaymath}
\dfrac{1}{a^3+\left(b+c\right)^3}+\dfrac{1}{b^3+\left(c+a\right)^3}+\dfrac{1}{c^3+\left(a+b\right)^3}\geq \dfrac{2}{a^3+b^3+c^3+3abc}
\end{displaymath}
{\bf{Chứng minh.}} Bất đẳng đã cho tương đương với lần lượt từng bất đẳng thức
\begin{eqnarray*}
&&\sum\dfrac{a^3+b^3+c^3+3abc}{a^3+\left(b+c\right)^3}\geq 2\\
&&\sum\dfrac{3bc\left(b+c-a\right)}{a^3+\left(b+c\right)^3}\leq 1\\
&&\sum\dfrac{3bc\left(b+c-a\right)}{a^2+\left(b+c\right)^2-a\left(b+c\right)}\leq\sum\left(b+c-a\right)\\
&&\sum\dfrac{\left(b+c-a\right)\left(a^2+b^2+c^2-ab-bc-ca\right)}{a^2+\left(b+c\right)^2-a\left(b+c\right)}\geq 0\\
&&\left(a^2+b^2+c^2-ab-bc-ca\right)\sum\dfrac{b+c-a}{a^2+\left(b+c\right)^2-a\left(b+c\right)}\geq 0
\end{eqnarray*}
Ta chỉ cần chứng minh
\begin{displaymath}
\sum\dfrac{b+c-a}{a^2+\left(b+c\right)^2-a\left(b+c\right)}\geq 0.
\end{displaymath}
Nhân hai vế bất đẳng thức với $\left(a+b+c\right)^2$ và để ý
\begin{displaymath}
\left(a+b+c\right)^2=a^2+\left(b+c\right)^2-a\left(b+c\right)+3a\left(b+c\right),
\end{displaymath}
ta được
\begin{displaymath}
\sum\left(b+c-a\right)+\sum\dfrac{3a\left(b+c\right)\left(b+c-a\right)}{a^2+\left(b+c\right)^2-a\left(b+c\right)}\geq 0
\end{displaymath}
tương đương
\begin{displaymath}
\left(a+b+c\right)+3\sum a-3\sum\dfrac{a^3}{a^2+\left(b+c\right)^2-a\left(b+c\right)}
\end{displaymath}
luôn đúng vì theo AM-GM
\begin{displaymath}
3\sum \dfrac{a^3}{a^2+\left(b+c\right)^2-a\left(b+c\right)}\leq 3\sum \dfrac{a^3}{\frac{3}{4}a^2}=4\left(a+b+c\right)
\end{displaymath}
Bổ đề được chứng minh. Đẳng thức xảy ra khi $a=b=c$. Quay trở lại bài toán, bất đẳng thức cần chứng minh tương đương
\begin{displaymath}
\dfrac{a^2}{a^3+\left(b+c\right)^3}+2abc\sum \dfrac{1}{a^3+\left(b+c\right)^3}\geq 1
\end{displaymath}
Sử dung bất đẳng thức Cauchy-Schwarz và Bổ đề, ta được
\begin{displaymath}
VT\geq \dfrac{\left(a^3+b^3+c^3\right)^2}{\left(a^3+b^3+c^3\right)^2+3abc\sum a^2\left(b+c\right)}+\dfrac{4abc}{a^3+b^3+c^3+3abc}.
\end{displaymath}
Ta chỉ cần chứng minh
\[\dfrac{\left(a^3+b^3+c^3\right)^2}{\left(a^3+b^3+c^3\right)^2+3abc\sum a^2\left(b+c\right)}+\dfrac{4abc}{a^3+b^3+c^3+3abc}\geq 1.\]
Khai triển bất đẳng thức này ta nhận được
\begin{displaymath}
abc\left[2\sum a^3-\sum a^2\left(b+c\right)\right]^2+abc\left[\sum a^3+3abc-\sum a^2\left(b+c\right)\right]\sum a^2\left(b+c\right)\geq 0,
\end{displaymath}
hiển nhiên đúng.
}

\begin{bt}%[Phương pháp phân ly đẳng thức]%[Nguyễn Thành Tiến, Dự án EX_TapChi05]%[0D0G2]

	\begin{displaymath}
	\dfrac{a^2+bc}{\left(a+b\right)\left(a+c\right)}+\dfrac{b^2+ca}{\left(b+c\right)\left(b+a\right)}+\dfrac{c^2+ab}{\left(c+a\right)\left(c+b\right)}\geq \dfrac{a^2}{a^2+bc}+\dfrac{b^2}{b^2+ca}+\dfrac{c^2}{c^2+ab}.
	\end{displaymath}
\begin{flushright}
{\it{(Tập huấn đội tuyển IMO 2014)}}
\end{flushright}
\loigiai{
Lấy 2 trừ cho hai vế của bất đẳng thức và để ý
\begin{displaymath}
2-\left(\sum \dfrac{a^2+bc}{\left(a+b\right)\left(a+c\right)}\right)=\dfrac{4abc}{\left(a+b\right)\left(b+c\right)\left(c+a\right)}.
\end{displaymath}
còn 
\begin{displaymath}
2-\sum \dfrac{a^2}{a^2+bc}=\dfrac{abc\left(a^3+b^3+c^3+abc\right)}{\left(a^2+bc\right)\left(b^2+ca\right)\left(c^2+ab\right)}.
\end{displaymath}
Ta đưa bất đẳng thức về dạng
\begin{displaymath}
\dfrac{abc\left(a^3+b^3+c^3+abc\right)}{\left(a^2+bc\right)\left(b^2+ca\right)\left(c^2+ab\right)}\geq \dfrac{4abc}{\left(a+b\right)\left(b+c\right)\left(c+a\right)}.
\end{displaymath}
Loại bỏ tích $abc$ ở hai vế, ta cần chứng minh
\begin{displaymath}
\left(a^3+b^3+c^3+abc\right)\left(a+b\right)\left(b+c\right)\left(c+a\right)\geq 4\left(a^2+bc\right)\left(b^2+ca\right)\left(c^2+ab\right).
\end{displaymath}
Mặc dù đã tách loại được $abc$ nhưng bất đẳng thức này vẫn có dấu bằng tại tâm và tại biên. Đến đây các bạn có thể khai triển hai vế, nhưng như thế quá phức tạp. Hãy để ý đến bất đẳng thức Schur
\begin{displaymath}
a^3+b^3+c^3+abc\geq ab\left(a+b\right)+bc\left(b+c\right)+ca\left(c+a\right)-2abc ,
\end{displaymath}
nên ta chỉ việc chứng minh
\begin{displaymath}
\left(ab(a+b)+bc(b+c)+ca(c+a)-2abc\right)(a+b)(b+c)(c+a)\geq 4(a^2+bc)(b^2+ca)(c^2+ab). \hfill (5)
\end{displaymath}
Vẫn chưa đơn giản hơn chút nào! Có lẽ bạn đã nghĩ thế khi đọc đến đây, nhưng hãy để ý, bây giwof chỉ cần cho hai biến bằng nhau (chẳng hạn $a=b$) thì bất đẳng thức (5) trở thành đẳng thức. Bởi tính đối xứng và bậc của bất đẳng thức (5) bằng $6$ nên nó tương đương với
\begin{displaymath}
\left(a-b\right)^2\left(b-c\right)^2\left(c-a\right)^2\geq 0
\end{displaymath}
Đó là điều phải chứng minh! Đẳng thức xảy ra khi và chỉ khi $a=b=c$ hoặc $abc=0$
