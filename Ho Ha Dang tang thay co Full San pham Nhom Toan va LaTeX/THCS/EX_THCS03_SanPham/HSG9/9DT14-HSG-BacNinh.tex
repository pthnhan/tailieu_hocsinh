\begin{ex}[Đề thi Toán 9 Học sinh giỏi năm học 2013-2014 Tỉnh Bắc Ninh][9D1K7][9D4K7]
Cho biểu thức: $P=\dfrac{x^2-\sqrt{x}}{x+\sqrt{x}+1}-\dfrac{2x+\sqrt{x}}{\sqrt{x}}+\dfrac{2(x-1)}{\sqrt{x}-1}$ \hspace{1cm} $(x>0, x\ne1)$.
\begin{enumerate}
\item  Rút gọn $P$.
\item Tìm giá trị của $x$ để $P=3$.
\end{enumerate}

\loigiai{
\begin{enumerate}
\item Ta có
\begin{align*}
P&=\dfrac{x^2-\sqrt{x}}{x+\sqrt{x}+1}-\dfrac{2x+\sqrt{x}}{\sqrt{x}}+\dfrac{2(x-1)}{\sqrt{x}-1}\\
&= \dfrac{\sqrt{x}(\sqrt{x^3}-1)}{x+\sqrt{x}+1}-\dfrac{\sqrt{x}(2\sqrt{x}+1)}{\sqrt{x}}+\dfrac{2(\sqrt{x}-1)(\sqrt{x}+1)}{\sqrt{x}-1}\\
&=\dfrac{\sqrt{x}(\sqrt{x}-1)(x+\sqrt{x}+1)}{x+\sqrt{x}+1}-2\sqrt{x}-1+2(\sqrt{x}+1)\\
&= x-\sqrt{x}+1.
\end{align*} 
\item $P=3\Leftrightarrow x-\sqrt{x}+1=3 \Leftrightarrow x-\sqrt{x-2}=0$.\\
Đặt $\sqrt{x}=t, t\geq 0$ ta được $t^2-t-2=0 \Leftrightarrow \left[\begin{aligned} & t=-1 \, (\text{loại}) \\ & t=2\end{aligned}\right.$\\
Với $t=2$ ta được $\sqrt{x}=2\Leftrightarrow x=4 $ (thỏa mãn điều kiện).\\
Vậy với $x=4$ thì $P=3$.
\end{enumerate}
}
\end{ex}

\begin{ex}[Đề thi Toán 9 Học sinh  giỏi năm học 2013-2014 Tỉnh Bắc Ninh][9D4K3][9D4K6]
Cho phương trình $x^2+(4m+1)x +2(m-4)=0$ (1) \hspace{0.5cm}($x$ là ẩn số, $m$ là tham số).
\begin{enumerate}
\item Chứng minh rằng phương trình (1) luôn có hai nghiệm phân biệt với mọi $m$.
\item Gọi $x_1; x_2$ là hai nghiệm của (1). Tìm $m$ để $\vert x_1 -x_2 \vert =17$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Có $\Delta=(4m+1)^2-8(m-4)=16m^2+8m+1-8m +32 =16m^2+33$.\\
Vì $\Delta = 16m^2+33>0, \forall m \in \mathbb{R}$ nên phương trình (1) luôn có hai nghiệm phân biệt với mọi $m$.
\item Phương trình (1) luôn có hai nghiệm phân biệt với mọi $m$ nên theo định lý Vi-ét ta có 
$$\begin{cases} x_1+x_2=-(4m+1) \\ x_1x_2=2(m-4). \end{cases}$$
Theo yêu cầu bài toán $\vert x_1 -x_2 \vert=17 \Leftrightarrow (x_1-x_2)^2=289\Leftrightarrow (x_1+x_2)^2-4x_1x_2=289\break \Leftrightarrow (4m+1)^2-8(m-4)=289\Leftrightarrow 16 m^2+33=289 \Leftrightarrow m=\pm 4$.\\
Vậy $m=\pm 4$ là giá trị cần tìm.

\end{enumerate}
}
\end{ex}

\begin{ex}[Đề thi Toán 9 Học sinh  giỏi năm học 2013-2014 Tỉnh Bắc Ninh][9D3K4][9D3G4]
\begin{enumerate}
\item Giải hệ phương trình $\begin{cases} 4x^2+y^4-4xy^3=1 \\ 2x^2+y^2-2xy=1. \end{cases}$
\item Cho các số thực $m, n, p$ thỏa mãn $n^2+np+p^2=1-\dfrac{3 m^2}{2}$. Tìm giá trị lớn nhất và giá trị nhỏ nhất của biểu thức $S=m +n+p$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $$\begin{cases} 4x^2+y^4-4xy^3=1 \\ 2x^2+y^2-2xy=1 \end{cases} \Leftrightarrow \begin{cases} 4x^2+y^4-4xy^3=1 \,(1)\\ 4x^2+2y^2-4xy=2 \, (2).\end{cases}$$
Trừ vế với vế của hai phương trình ta được\\ $$y^4-2y^2+1-4xy^3+4xy =0 \Leftrightarrow (y^2-1)(y^2-1-4xy)=0 .$$

\begin{itemize}
\item Nếu $y^2-1=0 \Leftrightarrow y=\pm 1$ thay vào (1) ta được các nghiệm $(x;y)$ là $(0;1)$, $(1;1)$, $(-1; -1)$, $(0; -1)$.

\item Nếu $y^2-1-4xy=0 \Leftrightarrow x=\dfrac{y}{4}-\dfrac{1}{4y}$ (do $y=0$ không thỏa mãn). Thay vào (2) ta được:\\
$4\left(\dfrac{y}{4}-\dfrac{1}{4y}\right) ^2+2y^2-4y\left(\dfrac{y}{4}-\dfrac{1}{4y}\right)=2 \Leftrightarrow \dfrac{y^2}{4}-\dfrac{1}{2}+\dfrac{1}{4y^2}+2y^2-y^2+1=2 \Leftrightarrow 5y^4-6y^2+1=0$.

Với $y^2=1\Rightarrow x=0$ ta được nghiệm $(x; y)$ là $(0; \pm 1)$.

Với $y^2=\dfrac{1}{5}$ ta được nghiệm $(x;y)$ là $\left( \dfrac{1}{\sqrt{5}}; \dfrac{-1}{\sqrt{5}}\right); \left(\dfrac{-1}{\sqrt{5}}; \dfrac{1}{\sqrt{5}} \right)$.
\end{itemize}

\item Có $n^2+np+p^2=1-\dfrac{3m^2}{2}$\\ 
$\Leftrightarrow (m+n+p)^2+(m-p)^2+(n-m)^2=2$\\
$\Leftrightarrow (m+n+p)^2=2-(m-p)^2-(n-m)^2\leq 2\Rightarrow S^2\leq 2$\\
 $\Leftrightarrow -\sqrt{2}\leq S\leq \sqrt{2}$.\\
Khi đó $\max S=\sqrt{2}\Leftrightarrow m=n=p=\dfrac{\sqrt{2}}{3}$; $\min S=-\sqrt{2}\Leftrightarrow m=n=p=-\dfrac{\sqrt{2}}{3}$.
\end{enumerate}
}
\end{ex}
