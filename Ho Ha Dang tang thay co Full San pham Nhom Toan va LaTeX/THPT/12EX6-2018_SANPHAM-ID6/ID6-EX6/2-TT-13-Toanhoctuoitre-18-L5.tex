	Cho các số thực $a,b$ thỏa mãn $\sqrt[3]{a^{14}}>\sqrt[4]{a^7}$, $\log_b\left(2\sqrt{a+1}\right)<\log_b\left(\sqrt{a}+\sqrt{a+2}\right)$. Khẳng định nào sau đây là đúng?
	\choice
	{$a>1,b>1$}
	{$0<a<1<b$}
	{\True $0<b<1<a$}
	{$0<a<1,0<b<1$}
	\loigiai{ Điều kiện để các căn thức có nghĩa là $a>0$.\\
		Ta có $\sqrt[3]{a^{14}}>\sqrt[4]{a^7}\Leftrightarrow a^{\tfrac{14}{3}}>a^{\tfrac{7}{4}}\Rightarrow a>1$.$\quad (1)$\\
		Xét hiệu $\left(2\sqrt{a+1}\right)^2-\left(\sqrt{a}+\sqrt{a+2}\right)^2=4a+4-\left(2a+2+2\sqrt{a(a+2)}\right)=2a+2-2\sqrt{a(a+2)}$.\\
		Vì $a>1$ nên $2a+2=a+a+2\geq 2\sqrt{a(a+2)}$, dấu $``="$ không xảy ra.\\
		Vậy \begin{eqnarray*}
			2a+2>2\sqrt{a(a+2)}&\Leftrightarrow& \left(2\sqrt{a+1}\right)^2-\left(\sqrt{a}+\sqrt{a+2}\right)^2>0\\
			&\Leftrightarrow& \left(2\sqrt{a+1}\right)^2>\left(\sqrt{a}+\sqrt{a+2}\right)^2\\
			&\Rightarrow& 2\sqrt{a+1}>\sqrt{a}+\sqrt{a+2}.
		\end{eqnarray*}
		Từ đó ta có  $\log_b\left(2\sqrt{a+1}\right)<\log_b\left(\sqrt{a}+\sqrt{a+2}\right)\Rightarrow  0<b<1$.$\quad(2)$\\
		Từ $(1)$ và $(2)$ suy ra $0<b<1<a$.
	}
\end{ex}

		$\Leftrightarrow \hoac{&u=-\dfrac{5}{6}\Rightarrow H\left( \dfrac{5}{6}; \dfrac{1}{3}; -\dfrac{5}{6} \right) \Rightarrow \overrightarrow{IH} = \left(-\dfrac{1}{6};\dfrac{1}{3}; \dfrac{1}{6} \right) \\ & u = -\dfrac{7}{6}\Rightarrow  H\left( \dfrac{7}{6}; -\dfrac{1}{3}; -\dfrac{7}{6} \right)\Rightarrow \overrightarrow{IH} = \left(\dfrac{1}{6}; -\dfrac{1}{3}; -\dfrac{1}{6} \right).}$\\
		Vì $\overrightarrow{IH}$ cùng hướng với $\overrightarrow{IN} \Rightarrow H\left( \dfrac{5}{6}; \dfrac{1}{3}; -\dfrac{5}{6} \right)$.
	}
\end{ex}

\begin{ex}[Thi thử L5, Toán học tuổi trẻ, 2018][Phan Quốc Trí, dự án 12-EX6][2D4K3-4]
	Cho các số phức $z_1,z_2$ với $z_1\ne 0$. Tập hợp các điểm biểu diễn số phức $w=z_1z+z_2$ là đường tròn tâm là gốc tọa độ và bán kính bằng $1$. Tập hợp các điểm biểu diễn số phức $z$ là đường nào sau đây?
	\choice
	{Đường tròn tâm là gốc tọa độ, bán kính bằng $\vert z_1 \vert$}
	{\True Đường tròn tâm là điểm biểu diễn số phức $-\dfrac{z_2}{z_1}$, bán kính bằng $\dfrac{1}{\vert z_1 \vert}$}
	{Đường tròn tâm là gốc tọa độ, bán kính bằng $\dfrac{1}{\vert z_1 \vert}$ }
	{Đường tròn tâm là điểm biểu diễn số phức $\dfrac{z_2}{z_1}$, bán kính bằng $\dfrac{1}{\vert z_1 \vert}$ }
	\loigiai{
		Ta có $ w=z\cdot z_1 +z_2 \Leftrightarrow z+\dfrac{z_2}{z_1}=\dfrac{w}{z_1}$. Khi đó \\
		$$\left| z+ \dfrac{z_2}{z_1} \right| = \dfrac{\vert w \vert}{\vert z_1 \vert} \Leftrightarrow \left| z- \left(- \dfrac{z_2}{z_1} \right) \right| = \dfrac{1}{\vert z_1 \vert}   \quad (*)$$
		Chú ý rằng: Tập hợp các  điểm biểu diễn số phức $z$ thỏa $\vert z - (a+bi) \vert =r$ là đường tròn tâm $I(a;b)$ bán kính $r$.\\
		Áp dụng vào biểu thức $(*)$, suy ra tập hợp các điểm biểu diễn số phức $z$ là đường tròn có tâm là điểm biểu diễn số phức $-\dfrac{z_2}{z_1}$, bán kính bằng  $\dfrac{1}{\vert z_1 \vert}$.
	}

	Tính đạo hàm cấp $n$ $(n \in \mathbb{N}^* )$ của hàm số $y=\ln \vert 2x-3 \vert$.

	\loigiai{
		Ta có $y'=\dfrac{2}{2x-3}$; \;
		$y''=-\dfrac{4}{(2x-3)^2}$; \;
		$y^{(3)}=\dfrac{16}{(2x-3)^3}$.\\
		Dự đoán $y^{(n)}=(-1)^{n-1}(n-1)! \left( \dfrac{2}{2x-3}  \right)^n $ \quad (*)\\
		Chứng minh công thức (*) bằng qui nạp: \\
		Với $n=1$, ta có $y'=\dfrac{2}{2x-3}$. (đúng)\\
		Giả sử công thức (*) đúng với $n=k$, nghĩa là $y^{(k)}=(-1)^{k-1}(k-1)! \left( \dfrac{2}{2x-3}  \right)^k $.\\
		Ta chứng minh công thức (*) đúng với $n=k+1$, tức là
		$$y^{(k+1)}=(-1)^{k}k! \left( \dfrac{2}{2x-3}  \right)^{k+1} $$
		Thật vậy
		\begin{eqnarray*}
			y^{(k+1)}&=&\left( (-1)^{k-1}(k-1)! \left( \dfrac{2}{2x-3}  \right)^k \right)'\\
			&=& (-1)^{k-1}(k-1)! \;k  \left( \dfrac{2}{2x-3} \right)^{k-1} \cdot \left( \dfrac{2}{2x-3} \right)'\\
			&=& (-1)^{k-1}(k-1)! \;k  \left( \dfrac{2}{2x-3} \right)^{k-1}\cdot \dfrac{-4}{(2x-3)^2}\\
			&=&(-1)^{k}k! \left( \dfrac{2}{2x-3}  \right)^{k+1}
		\end{eqnarray*}
	}
