\section{SỐ TỰ NHIÊN VÀ CÁC CHỮ SỐ}
\begin{center}
TỔNG QUAN VỀ CHUYÊN ĐỀ
\end{center}

Số tự nhiên và các chữ số của nó là một dạng toán số học thường gặp, trong đó yêu cầu của bài toán thường là tìm các chữ số chưa biết.

Ngoài trường hợp có thể tìm lần lượt từng chữ số, ta thường gộp các nhóm chữ số lại thành một biến mới rồi tìm giá trị của biến đó.

Để tìm các giá trị ấy, ta thường chú ý đên:
\begin{itemize}
\item Sử dụng tính chia hết: Xét xem các biến (hoặc các biểu thức chứa biến) chia hết cho số nào, hoặc là ước của số nào.
\item Sử dụng bất đẳng thức: Xét xem các biến (hoặc các biểu thức chưa biến) bị chặn trong khoảng nào.
\item Sử dụng phương trình: Xét xem các biến liên hệ với nhau bởi phương trình nào.
\end{itemize}

Tất cả các công việc trên đòi hỏi người làm toán phải biết quan sát, phát hiện đặc điểm của bài toán để đưa ra phương án giải quyết hợp lý.

\begin{vd}
Tìm số tự nhiên $\overline{abcd}$ chia hết cho tích của $\overline{ab}$ và $\overline{cd}$ (các số $a,b,c,d$ không bắt buộc khác nhau).
\loigiai{
$\overline{abcd} : \overline{ab} \cdot \overline{cd}.$\\
Đặt $\overline{ab} = x,$ $\overline{cd} = y,$ ta có $\overline{abcd} = 100x + y \ \vdots \ xy \Rightarrow y \ \vdots \ x.$\\
Đặt $y = kx$ với $k \in \mathbb{N}$ và $1 \leq k \leq 9$ thì $100x + kx \ \vdots \ xkx.$\\
$\Rightarrow 100 + k \ \vdots \ kx$ $(1)$ $\Rightarrow k \in \{ 1;2;4;5\}.$\\
Với $k=1$ thì từ $(1)$ có $101 \ \vdots \ x \Rightarrow x = 1,$ (loại).\\
Với $k=2$ thì từ $(1)$ có $102 \ \vdots \ 2x \Rightarrow 51 \ \vdots \ x.$ Ta có
\begin{center}
\begin{tabular}{|c|c|c|}
\hline
$x$&$17$&$51$\\
\hline 
$y$&$34$&$102,$ loại \\
\hline
\end{tabular}
\end{center}
Với $k=4$ thì từ $(1)$ có $104 \ \vdots \ 4x \Rightarrow 26 \ \vdots \ x$. Ta có
\begin{center}
\begin{tabular}{|c|c|c|}
\hline
$x$&$26$&$13$\\
\hline 
$y$&$104$, loại&$52$\\

\end{tabular}
\end{center}

Có hai đáp số: $1734 \ \vdots \ 17 \cdot 34$ và $1352 \ \vdots \ 13 \cdot 52.$
}
\end{vd}
\begin{vd}
Tìm các chữ số $a,$ $b,$ $c$ (không bắt buộc khác nhau) sao cho $\overline{aaaa} - \overline{bb} = \overline{cc} ^2$.
\loigiai{
$\overline{aaaa} - \overline{bb} = \overline{cc} ^2 \Rightarrow 1111a - 11b = 11^2 \cdot c^2$
$\Rightarrow 101a - b = 11 c^2$ $ \Rightarrow 99a + (2a-b) = 11 c^2$. \hfill $(1)$\\
Ta có $2a - b \ \vdots \ 11$ nên $2a - b \in \{ 0 ; 11\}$.
\begin{enumerate}
\item Xét $2a = b$ thì $a \leq 4$. Thay vào $(1)$ có $9a = c^2 \Rightarrow a \in \{ 1;4\}$\\
Ta có 
\begin{align*}
1111 - 22 &= 33^2\\
4444 - 88 &= 66^2
\end{align*}
\item Xét $2a - b =11$. Thay vào $(1)$ có $9a + 1 = c^2 \Rightarrow 9a = (c+1)(c-1)$.\\
Ta thấy $c+1$ và $c-1$ có hiệu bằng $2$ mà tích của chúng chia hết cho $9$ nên phải có một số chia hết cho $9$, đó là số $c+1$. Ta được $c+1 = 9$ nên $c = 8$. Từ đó $a = 7$.\\
Khi đó $7777 - 33 = 88^2.$\\
Đáp số: $1111 - 22 = 33^2;$ $4444 - 88 = 66^2;$ $7777 - 33 = 88^2.$
\end{enumerate}
}
\end{vd}
\begin{vd}
Tìm số tự nhiên có ba chữ số, biết rằng bình phương của nó cũng tận cùng bởi ba chữ số ấy theo thứ tự đó.
\loigiai{
Đặt $\overline{abcdeg} = \overline{deg}^2 = n^2$ $(1 \leq n \leq 999)$.\\
Ta có $1000 \cdot \overline{abc} + n = n^2$ nên $n^2 - n \ \vdots \ 1000 \Rightarrow n(n-1) \ \vdots \ 1000.$\\
Hai số $n$ và $n-1$ nguyên tố cùng nhau nhỏ hơn $1000$, có tích chia hết cho $1000$ nên có hai trường hợp.
\begin{enumerate}
\item $\heva{ n & \ \vdots \ 125 \\ n-1 & \ \vdots \ 8}$\\
$n$ là số lẻ có ba chữ số và chia hết cho $125$ nên $n \in \{ 125;375;625;875 \}$. Chỉ có $n= 625$ cho $n-1 = 624 \ \vdots \ 8$.\\
Ta được $625 ^2 = 390625.$
\item $\heva{n - 1 &\ \vdots \ 125 \\ n &\ \vdots \ 8}$\\
$n-1$ là số lẻ có ba chữ số và chia hết cho $125$ nên $n-1 \in \{ 125;375;625;875 \}$. Chỉ có $n-1 = 375$ cho $n=376 \ \vdots \ 8.$\\
Ta được $376^2 = 141376.$
\end{enumerate}
Đáp số: $625$ và $376$.
}
\end{vd}
\begin{vd}
Tìm số tự nhiên có sáu chữ số, biết rằng nó bằng bình phương hai số tạo bởi ba chữ số đầu và tạo bởi ba chữ số cuối (không thay đổi thứ tự).
\loigiai{
Gọi số cần tìm là $\overline{abcdeg}$, ta có $\overline{abcdeg} = \left( \overline{abc} + \overline{deg} \right)^2.$\\
Đặt $\overline{abc} = x$, $\overline{deg} = y$. Do $(x+y)^2$ là một số có sáu chữ số nên $x+y \leq 99$.\\
Ta có
$$1000x + y = (x+y)^2 \Leftrightarrow 999x + (x+y) = (x+y)^2.$$
Đặt $x+y = n$ thì $999x = n(n-1)$. \hfill $(1)$\\
$n(n-1) \ \vdots \ 999$. Do $n \leq 999$ nên ta xét hai trường hợp
\begin{enumerate}
\item $n=99$. Thay vào $(1)$ được $x=998$, do đó $y=1$.\\
Ta được $998001 = (998+1)^2$.
\item $n<999= 3^3 \cdot 37$. Hai số $n$ và $n-1$ nguyên tố cùng nhau có tích chia hết cho $999$ nên có hai trường hợp
$$\heva{ n & \ \vdots \ 37 \\ n-1 & \ \vdots \ 27} \text{ và } \heva{n-1 &\ \vdots \ 37 \\ n &\ \vdots \ 27}$$
Ở trường hợp đầu, ta đặt $n=37k$ (dp $n<999$ nên $k < 27$).\\
Ta có $37k \ \vdots \ 27 \Rightarrow 10k -1 \ \vdots \ 27 \Rightarrow 80k -8 \ \vdots \ 27 \Rightarrow 81k - (k+8) \ \vdots \ 27 \Rightarrow k+8 \ \vdots \ 27$.\\
Do $k<27$ nên $k=19$. Khi đó $n= 37 \cdot 19 = 703$. Thay vào $(1)$ được $x=494$, dó đó $y=703 - 494 = 209$. Ta được $494209 = (494 + 209)^2$.\\
Ở trường hợp sau, ta đặt $n= 37m+1$ (do $n<999$ nên $m<27$).
Có hai đáp số: $998001$ và $494209$.
\end{enumerate}
}
\textbf{Lưu ý: }\\
	\hspace*{1cm} Cách đặt $n = 37k$ rồi xét $37k - 1 \ \vdots \ 27$ như ở lời giải trên có ưu điểm hơn so với cách đặt $n = 27k + 1$ rồi xét $27k + 1 \ \vdots \ 37$. \\
	\hspace*{1cm} Cũng vậy, các đặt $n = 37m + 1$ rồi xét $37m + 1 \ \vdots \ 27$ như ở lời giải trên có ưu điểm hơn so với cách đặt $n = 27m$ rồi xét $27m - 1 \ \vdots \ 37$. 
\end{vd}

\begin{vd}
	Tìm số tự nhiên $\overline{abc}$ (các chữ số $a$, $b$, $c$ không bắt buộc khác nhau) sao cho $\overline{abc} = \overline{ab}^2 - c^2$.
	\loigiai{
	\textit{Cách 1.} Đặt $\overline{ab} = x$ $(10 \leq x \leq 99)$. Ta có 
	$$10x + c = x^2 - c^2 \Leftrightarrow c(c + 1) = x(x - 10).$$
	Do $c \geq 9$ nên $c(c + 1) \leq 90 \Rightarrow x(x - 10) \leq 90 \Rightarrow x \leq 15$. \\
	Xét $\overline{ab}$ từ $10$ đến $15$, ta thấy:
	\begin{center}
	\begin{tabular}{|c|c|c|c|}
		\hline
		$\overline{ab}$&$\overline{ab} \left(\overline{ab} - 10 \right)$&$c(c+1)$& $\overline{abc} = \overline{ab}^2 - c^2$\\
		\hline 
		$10$&$10 \cdot 0$&$0 = 0 \cdot 1$ & $100 = 10^2 - 0^2$\\
		\hline
		$11$&$11 \cdot 1$&$11$, loại & \\
		\hline
		$12$&$12 \cdot 2$&$24$, loại & \\
		\hline
		$13$&$13 \cdot 3$&$394$, loại & \\
		\hline
		$14$&$14 \cdot 4$&$ 56 = 7 \cdot 8$& $147 = 14^2 - 7^2$ \\
		\hline
		$15$&$15 \cdot 5$&$75$, loại & \\
		\hline
	\end{tabular}
	\end{center}
	Có hai đáp số: $100$ và $147$. \\
	\textit{Cách 2.} Đặt $\overline{ab} = x$. Ta có 
	\allowdisplaybreaks{
	\begin{align*}
	& 10x + c = x^2 - c^2 \Leftrightarrow 40x + 4c = 4x^2 - 4c^2\\
	\Leftrightarrow & 4c^2 + 4c = 4x^2 - 40x \Leftrightarrow (2c + 1)^2 = (2x - 10)^2\\
	\Leftrightarrow &(2x - 10)^2 - (2c + 1)^2 = 99 \\
	\Leftrightarrow &(2x + 2c - 9)(2x - 2c - 11) = 99. 
	\end{align*}} 
	Do $2x - 2c - 11 < 2x + 2c - 9$ nên có các trường hợp:
	\begin{center}
	\begin{tabular}{|c|c|c|c|}
		\hline
		$2x + 2c - 9$&$99$& $33$ &$11$\\
		\hline 
		$2x - 2c - 11$&$1$&$3$ & $9$\\
		\hline
		$x + c$&$54$&$21$&$10$\\
		\hline
		$x - c$&$6$&$7$&$10$\\
		\hline
		$x$&$30$&$14$&$10$\\
		\hline
		$c$&$24$, loại&$7$&$0$\\
		\hline
	\end{tabular}
	\end{center}
	Có hai đáp số $147$ và $100$.
	}
\end{vd}

	\loigiai{
	Do $\overline{ab}^2$ có bốn chữ số nên $a > 2$. Ta thấy $a$ khác $3$, $4$, $5$, $6$, $7$, $8$ vì $\overline{3b}^2 < 3000$,$\overline{4b}^2 < 4000$, $\overline{5b}^2 < 5000$, $\overline{6b}^2 < 6000$, $\overline{7b}^2 < 7000$, $\overline{8b}^2 < 8000$.\\
	Xét $\overline{9b}^2 = \overline{9cdb}$, do $b^2$ tận cùng bằng $b$ nên $b \in \{0;1;5;6 \}$. Xét các chữ số $90^2$, $91^2$, $95^2$, $96^2$ ta được hai đáp số: $95^2 = 9025$ và $96^2 = 9216$.

\begin{bt}
	Tìm số tự nhiên có hai chữ số, biết rằng bình phương của nó bằng lập phương của tổng các chữ số của nó.
	\loigiai{
	Đặt $n = \overline{ab}^2 = (a + b)^3$. \\
	Do $n$ bằng bình phương của một số tự nhiên và bằng lập phương của một số tự nhiên nên $n = x^6$ $(x \in \mathbb{N}$. Ta có \\
	$x^6 = \overline{ab}^2 = (a + b)^3$ nên $\heva{&x^3 = \overline{ab} \\&x^2 = a + b} (1)$. \\
	Do $x^3 = \overline{ab} \geq 10$ nên $x \geq 3$. \\
	Do $x^2 = a + b \leq 18$ nên $x \leq 4$. \\
	Xét $x = 3$, từ $(1)$ có $\overline{ab} = 27$ và $a + b = 9$, thỏa mãn.\\
	Xét $x = 4$, từ $(1)$ có $\overline{ab} = 64$ và $a + b = 16$, loại. \\
	\textit{Đáp số:} $27$.
	}
\end{bt}
\begin{bt}
	Tìm số tự nhiên có năm chữ số, biết rằng nó bằng lập phương của số tạo bởi hai chữ số đầu (không thay đổi thứ tự).
	\loigiai{
	Xét $\overline{abcde} = \overline{ab}^3$. Đặt $\overline{ab} = x$, $\overline{cde} = y$ với $10 \leq x \leq 99$ và $0 \leq y \leq 999$. \\
	Ta có $1000x + y = x^3$ nên $y = x \left(x^2 - 1000 \right)$. \\
	Do $y \geq 0$ nên $x^2 \geq 1000 \Rightarrow x \geq 32 \hfill(1)$.\\
	Ta có $x^2 - 1000 = \dfrac{y}{x} \geq \dfrac{999}{32} = 31 \dfrac{7}{32}$\\
	$\Rightarrow x^2 - 1000 \leq 31 \Rightarrow x^2 \leq 1031 \Rightarrow x \leq 32 \hfill(2)$. \\
	Từ $(1)$ và $(2)$ suy ra $x = 32$. Ta có $32^3 = 32748$. \\
	Vậy số cần tìm là $32748$.
	}
\end{bt}
\begin{bt}
	Tìm số tự nhiên $\overline{abcdeg}$ chia hết cho tích của $\overline{abc}$ và $\overline{deg}$ (các chữ số $a$, $b$, $c$, $d$, $e$, $g$ không bắt buộc khác nhau).
	\loigiai{
	Xét $\overline{abcdeg} \ \vdots \ \overline{abc} \cdot \overline{deg}$.\\
	Đặt $\overline{abc} = x$, $\overline{deg} = y$, ta có $1000x + y \ \vdots \ xy \Rightarrow y \ \vdots \ x$.\\
	Đặt $y = kx$ với $k \in \mathbb{N}$ và $1 \leq k  \leq 9$ thì $1000x + kx \ \vdots \ xkx \Rightarrow 1000 + k \ \vdots \ kx \hfill (1)$. \\
	$\Rightarrow 1000 \ \vdots \ k \Rightarrow k \in \{1;2;4;5;8 \}$. \\
	Với $k = 1$, thay vào $(1)$ có $1001 \ \vdots \ x \Rightarrow x = 143, y = 143$.\\
	Với $k = 2$, thay vào $(1)$ có $1002 \ \vdots \ 2x \Rightarrow 501 \ \vdots \ x$. Ta có
	\begin{center}
	\begin{tabular}{|c|c|c|}
		\hline
		$x$&$167$&$501$\\
		\hline 
		$y$&$334$&$1002,$ loại \\
		\hline
	\end{tabular}
	\end{center}

	Với $k = 5$, thay vào $(1)$ có $1005 \ \vdots \ 5x \Rightarrow 201 \ \vdots \ x \Rightarrow x = 201, y = 1005$ loại.\\

\begin{bt}
	Cho năm chữ số tự nhiên $a$, $b$, $c$, $d$, $e$, mỗi số có bốn chữ số đề gồm cả bốn chữ số $1$, $2$, $3$, $4$. Chứng minh rằng khổng thể xảy ra $a^3 + b^3 + c^3 = d^3 + e^3$.
	\loigiai{
	Giả sử $a^3 + b^3 + c^3 = d^3 + e^3$ \hfill(1). \\
	Mỗi số $a$, $b$, $c$, $d$ có tổng các chữ số bằng $1 + 2 + 3 + 4 = 10$ nên chia hết cho $9$ dư $1$. Khi $a = 9k + 1$ thì $a^3 = 9m + 1$. Không thể xảy ra (1) vì vế trái chia cho $9$ dư $3$, còn vế phải chia cho $9$ dư $2$.
	}
\end{bt}
