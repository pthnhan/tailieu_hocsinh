\begin{ex}%[Đề thi HSG 9, tỉnh Hưng Yên, năm 2018]%[Phạm Doãn Lê Bình, Dự án EX-9G-18]%[9D0G1]
	Cho hai số thực dương $a,b$ thỏa mãn $\dfrac{1}{a} + \dfrac{1}{b} = \dfrac{1}{2018}$. Chứng minh rằng $$\sqrt{a+b} = \sqrt{a-2018} + \sqrt{b-2018}.$$
	\loigiai{
Từ giả thiết ta có $\dfrac{1}{a} = \dfrac{1}{2018} - \dfrac{1}{b} < \dfrac{1}{2018} \Rightarrow a > 2018$. Tương tự ta cũng có $b > 2018$.\\
Với $a,b$ là các số thực lớn hơn $2018$, ta có
	\begin{align*}
	& \dfrac{1}{a} + \dfrac{1}{b} = \dfrac{1}{2018} \Leftrightarrow ab = 2018 a + 2018 b \Leftrightarrow ab - 2018 a - 2018 b + 2018^2 = 2018^2 \\
	\Leftrightarrow & (a-2018)(b-2018) = 2018^2  \Leftrightarrow \sqrt{(a-2018)(b-2018)} = 2018\\
	 \Leftrightarrow & 2 \sqrt{(a-2018)(b-2018)} = 2 \cdot 2018\\
	 \Leftrightarrow & a - 2018 + 2 \sqrt{(a-2018)(b-2018)} + b - 2018 = a+b\\
	 \Leftrightarrow & \left( \sqrt{a-2018} + \sqrt{b-2018} \right)^2 = a+b
	 \Leftrightarrow \sqrt{a-2018} + \sqrt{b-2018} = \sqrt{a+b}.
	\end{align*}
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Hưng Yên, năm 2018]%[Phạm Doãn Lê Bình, Dự án EX-9G-18]%[9D0G1]
    Cho $a$ là nghiệm dương của phương trình $6x^2 + \sqrt{3}x - \sqrt{3} = 0$. Tính giá trị biểu thức $A = \dfrac{a+2}{\sqrt{a^4+a+2}-a^2}$.
\loigiai
    {
    Ta có 
    \begin{align*}
   &  6a^2 +\sqrt{3}a-\sqrt{3} = 0 \Leftrightarrow 6a^2 = \sqrt{3}(1-a) \Leftrightarrow 36a^4 = 3(1-2a+a^2)\\
    \Leftrightarrow & 36a^4 +36a + 72 = 3a^2 + 30a + 75 \Leftrightarrow 36(a^4+a+2) = 3(a+5)^2 \\
    \Leftrightarrow & 6\sqrt{a^4+a+2} = \sqrt{3}a + 5\sqrt{3}.
    \end{align*}
   Mặt khác $A= \dfrac{a+2}{\sqrt{a^4+a+2}-a^2} = \dfrac{(a+2) \left( \sqrt{a^4+a+2}+a^2 \right)}{a+2} = \sqrt{a^4+a+2}+a^2.$\\
   Suy ra $6A = 6\sqrt{a^4+a+2}+ 6a^2 = 6a^2 + \sqrt{3}a + 5\sqrt{3} = 6a^2 + \sqrt{3}a -\sqrt{3} + 6\sqrt{3} = 6\sqrt{3}.$\\
   Vậy $A =\sqrt{3}.$
    }
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Hưng Yên, năm 2018]%[Phạm Doãn Lê Bình, Dự án EX-9G-18]%[9D0G2]
Giải phương trình $\left( 1 - \sqrt{1-x} \right) \sqrt[3]{2-x} = x$.
\loigiai{
Điều kiện xác định của phương trình: $x \le 1$.\\
Phương trình ban đầu trở thành $\dfrac{x \sqrt[3]{2-x}}{1+\sqrt{1-x}}=x$.\hfill (1)\\
Ta thấy $x=0$ là nghiệm của phương trình $(1)$.\\
Xét $x \ne 0$, chia hai vế cho $x$ ta được $\dfrac{\sqrt[3]{2-x}}{1+\sqrt{1-x}}=1 \Rightarrow \sqrt[3]{2-x} = 1 + \sqrt{1-x}$.\\
Đặt $\sqrt[3]{2-x} = a \Rightarrow a^3 = 2- x \Rightarrow a^3 - 1  = 1-x.$\\
Khi đó phương trình $(1)$ trở thành\\
$a = 1 + \sqrt{a^3-1} \Leftrightarrow a - 1 = \sqrt{(a-1)(a^2+a+1)} \Leftrightarrow \hoac{& \sqrt{a-1} = 0 \\ & \sqrt{a-1} = \sqrt{a^2+a+1}.}$
\begin{itemize}
\item $\sqrt{a-1} = 0 \Leftrightarrow a = 1 \Rightarrow x = 1$ (nhận).
\item $\sqrt{a-1} = \sqrt{a^2+a+1} \Leftrightarrow a - 1 = a^2 + a + 1 \Leftrightarrow a^2 + 2 = 0$ (vô nghiệm).
\end{itemize}
Vậy phương trình ban đầu có hai nghiệm $x = 0$; $x=1$.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Hưng Yên, năm 2018]%[Phạm Doãn Lê Bình, Dự án EX-9G-18]%[9D0G5]
Tìm tất cả các cặp số nguyên $(x;y)$ thỏa mãn $(x-2018)^2 = y^4 -6y^3 +11y^2 -6y$.
\loigiai{
Ta có $y^4 -6y^3 +11y^2 -6y = \left( y^2 -3y \right)^2 + 2 \left( y^2 - 3y \right)$.\\
Đặt $y^2 - 3y = a$. Phương trình ban đầu trở thành\\
$(x-2018)^2 = a^2 + 2a = (a+1)^2 - 1 \Leftrightarrow (x-2018)^2 - (a+1)^2 = - 1 \Leftrightarrow (x+a-2017)(x-a-2019)=-1.$
\begin{itemize}
\item $\heva{ & x+a-2017 = 1 \\ & x-a-2019 = -1} \Leftrightarrow \heva{ & x=2018 \\ & a = 0} \Leftrightarrow \heva{ & x=2018 \\ & y^2-3y = 0} \Leftrightarrow \heva{& x = 2018 \\ & \hoac{ & y = 0 \\ & y = 3. }}$
\item $\heva{ & x+a-2017 = -1 \\ & x-a-2019 = 1} \Leftrightarrow \heva{ & x=2018 \\ & a = -2} \Leftrightarrow \heva{ & x=2018 \\ & y^2-3y = -2} \Leftrightarrow \heva{& x = 2018 \\ & \hoac{ & y = 1 \\ & y = 2. }}$
\end{itemize}
Vậy có bốn cặp số nguyên $(x;y)$ thỏa mãn đề bài là $(2018;0)$, $(2018;1)$, $(2018;2)$, $(2018;3)$.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Hưng Yên, năm 2018]%[Phạm Doãn Lê Bình, Dự án EX-9G-18]%[9D0G2]
Giải hệ phương trình $\heva{ & \sqrt{2x+1} + \sqrt{2y + 1} = \dfrac{(x-y)^2}{2} \\ & (3x+2y)(y+1) = 4 - x^2. }$
\loigiai{
$\heva{ & \sqrt{2x+1} + \sqrt{2y + 1} = \dfrac{(x-y)^2}{2} & (1) \\ & (3x+2y)(y+1) = 4 - x^2 & (2)}$.
Điều kiện xác định: $x,y \ge - \dfrac{1}{2}$.\\
Ta có $(2) \Leftrightarrow x^2 +3(y+1)x + 2y^2 + 2y - 4 = 0$. \hfill $(2')$\\
Ta coi phương trình $(2')$ như phương trình bậc hai ẩn $x$, tham số là $y$, ta có
	$$\Delta = 9(y+1)^2 - 4(2y^2 + 2y - 4) = y^2 + 10 y + 25 = (y+5)^2.$$
Suy ra $\hoac{& x = \dfrac{-3(y+1)+y+5}{2} = -y +1 \\ &  x = \dfrac{-3(y+1)-(y+5)}{2} = -2y -4.}$\\
{\bf TH1:} $x=-y+1$, thay vào phương trình $(1)$ ta có $$\sqrt{-2y+3} + \sqrt{2y+1} = \dfrac{(-2y+1)^2}{2} = \dfrac{4y^2 - 4y +1}{2}. \quad (1')$$
Điều kiện xác định: $-\dfrac{1}{2}\le y \le \dfrac{3}{2}$. Đặt $t = \sqrt{-2y +3} + \sqrt{2y +1}$, ta có 
	\begin{align*}
	& t^2 = 4 + 2 \sqrt{(-2y+3)(2y+1)} = 4 + 2\sqrt{-4y^2 + 4y + 3}\\
	\Rightarrow & 4y^2 - 4y = 3 -\left( \dfrac{t^2 - 4}{2} \right)^2 = \dfrac{-t^4 + 8t^2 -4}{4}.
	\end{align*}
Thay vào phương trình $(1')$ ta có 
	\begin{align*}
	& t=\dfrac{\dfrac{-t^4+8t^2-4}{4}+1}{2} = \dfrac{-t^4 + 8t^2 }{8} \Leftrightarrow t^4-8t^2 +8t = 0\\
	\Leftrightarrow & t(t-2)(t^2+2t-4)=0 \Leftrightarrow \hoac{& t=0 \\ & t = 2 \\ & t = - 1 - \sqrt{5} \\ & t = -1 + \sqrt{5}.}
	\end{align*}
Mặt khác do $t\ge 0$ và $t^2 = 4 + 2\sqrt{-4y^2+4y+3} \ge 4$ nên suy ra $t \ge 2$. Kết hợp với các kết quả thu được ở trên ta suy ra $t=2 \Leftrightarrow \hoac{& y = -\dfrac{1}{2} \\& y = \dfrac{3}{2}.}$
	\begin{itemize}
	\item $ y = -\dfrac{1}{2} \Rightarrow x = \dfrac{3}{2}$;
	\item $y = \dfrac{3}{2} \Rightarrow x = - \dfrac{1}{2}$.
	\end{itemize}
{\bf TH2:} $x = -2y - 4 \Leftrightarrow x + 2y = -4$.\\
Theo điều kiện ban đầu ta có $x+2y \ge -\dfrac{3}{2}$, do đó phương trình trên vô nghiệm. \\
Vậy hệ phương trình ban đầu có 2 cặp nghiệm $(x;y)$ là $\left( - \dfrac{1}{2} ; \dfrac{3}{2} \right)$ và $\left( \dfrac{3}{2}; - \dfrac{1}{2} \right)$.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Hưng Yên, năm 2018]%[Phạm Doãn Lê Bình, Dự án EX-9G-18]%[9D0G7]
Cho ba số thực dương $x$, $y$, $z$ thỏa mãn $2\sqrt{y}+ \sqrt{z} = \dfrac{1}{\sqrt{x}}$. Chứng minh rằng $$\dfrac{3yz}{x} + \dfrac{4zx}{y} + \dfrac{5xy}{z} \ge 4.$$
\loigiai{
Ta có $\dfrac{3yz}{x} + \dfrac{4zx}{y} + \dfrac{5xy}{z} = \left( \dfrac{yz}{x} + \dfrac{zx}{y} \right) + \left( \dfrac{2yz}{x} + \dfrac{2xy}{z} \right) + \left( \dfrac{3zx}{y} + \dfrac{3xy}{z} \right) \ge 2z + 4y + 6x.$\\
Mà $2z+ 4y + 6x  = 2z + 2x + 4y + 4x \ge 4 \sqrt{xz} + 8 \sqrt{xy} = 4 \sqrt{x} \left( 2\sqrt{y} + \sqrt{x} \right) = 4 \sqrt{x} \left(\dfrac{1}{\sqrt{x}} \right)=4.$\\
Do đó ta có điều phải chứng minh.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Hưng Yên, năm 2018]%[Phạm Doãn Lê Bình, Dự án EX-9G-18]%[9H0G1]%[9H0G3]%[9H0G6]
Cho đường tròn $(O;R)$  và điểm $A$ cố định với $OA = 2R$; đường kính $BC$ quay quanh $O$ sao cho tam giác $ABC$ là tam giác nhọn. Đường tròn ngoại tiếp tam giác $ABC$ cắt đường thẳng $OA$ tại điểm thứ hai là $I$. Các đường thẳng $AB$, $AC$ cắt $(O;R)$ lần lượt tại điểm thứ hai là $D$ và $E$. Gọi $K$ là giao điểm của $DE$ với $OA$.
	\begin{enumerate}[a)]
	\item Chứng minh $AK \cdot AI = AE \cdot AC$.
	\item Tính độ dài đoạn $AK$ theo $R$.
	\item Chứng minh tâm đường tròn ngoại tiếp tam giác $ADE$ luôn thuộc một đường thẳng cố định. 
	\end{enumerate}
\loigiai{
\immini{\begin{enumerate}[a)]
	\item Ta có tứ giác $BDEC$ nội tiếp nên $\widehat{ABC} = \widehat{AED}$. \\
Xét đường tròn ngoại tiếp tam giác $ABC$, ta có $\widehat{ABC} = \widehat{AIC}$ ($2$ góc nội tiếp cùng chắn cung $AC$).\\
Suy ra $\widehat{AED} = \widehat{AIC}$. Từ đó suy ra $\triangle AEK \backsim \triangle AIC$ (g-g). Suy ra $\dfrac{AK}{AC} = \dfrac{AE}{AI}$. Từ đó suy ra $AK \cdot AI = AE \cdot AC$.
	\end{enumerate}
}
{
	\begin{tikzpicture}[>=stealth]

	\begin{enumerate}[a)]
	\setcounter{enumi}{1}
	\item Xét đường tròn $(O)$, ta có đẳng thức $AE \cdot AC = AO^2 - R^2 = 4R^2 - R^2 = 3R^2$.	\\
	Xét đường tròn ngoại tiếp tam giác $ABC$, dễ thấy $\triangle OAB \backsim \triangle OCI$ (g-g), suy ra $$\dfrac{OA}{OC} = \dfrac{OB}{OI} \Rightarrow OI = \dfrac{OB \cdot OC}{OA} = \dfrac{R}{2} \Rightarrow AI = OA + OI = \dfrac{5R}{2}.$$
	Mặt khác, ta có $AK \cdot AI = AE \cdot AC = 3R^2$ nên $AK = \dfrac{3R^2}{AI} = \dfrac{6R}{5}$.
	\item Gọi $M,N$ là hai giao điểm của đường thẳng $OA$ và đường tròn $(O)$ ($M$ nằm giữa $A$ và $K$).\\
	Gọi $J$ là giao điểm của $AO$ và đường tròn ngoại tiếp tam giác $ADE$ ($J$ không trùng với $A$).\\

	$$MK = AK - AM = \dfrac{6R}{5} - R = \dfrac{R}{5}; \quad NK = AN - AK = 3R - \dfrac{6R}{5} = \dfrac{9R}{5}.$$

\begin{ex}%[Đề thi HSG 9, tỉnh Hưng Yên, năm 2018]%[Phạm Doãn Lê Bình, Dự án EX-9G-18]%[9D0G8]
Từ $625$ số tự nhiên liên tiếp $1;2;3;\dots;625$ chọn ra $311$ số sao cho không có hai số nào có tổng bằng $625$. Chứng minh rằng trong $311$ số được chọn, bao giờ cũng có ít nhất một số chính phương.
\loigiai{
Ta chia $625$ số đó thành $313$ tập hợp là $\{1;624\}, \{2;623\}, \dots , \{625\}$. \\
Giả sử trong $311$ số ta chọn không có số chính phương. Mà có $3$ tập hợp là $\{625\}$, $\{49; 576\}$ và $\{225; 400\}$ có tất cả các phần tử là các số chính phương. Vậy $311$ số được chọn chỉ nằm trong $310$ tập hợp còn lại.\\
Theo nguyên tắc Đi-rich-let thì có ít nhất $1$ tập hợp có $2$ số trong số $311$ số được chọn mà như vậy thì tổng hai số đó bằng $625$ trái với giả thiết.\\
Vậy suy ra trong $311$ số được chọn bao giờ cũng có ít nhất một số chính phương.
}
\end{ex}
