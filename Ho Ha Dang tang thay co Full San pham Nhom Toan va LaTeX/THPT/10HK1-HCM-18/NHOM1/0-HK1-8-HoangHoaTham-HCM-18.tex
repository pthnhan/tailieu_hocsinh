\begin{bt}%[HK1, Hoàng Hoa Thám, HCM 17-18]%[Đào Anh Dương, 10-EX-HCM]%[0D6Y2-5]
	Chứng minh hệ thức $\dfrac{\sin x + \tan x}{\tan x} = 1+\sin x \cot x$.
	\loigiai{
		\begin{align*}
		\text{VT} &= \dfrac{\sin x + \tan x}{\tan x}\\
		&= 1 + \sin x \cdot \dfrac{1}{\tan x}\\
		&= 1+\sin x \cot x = \text{VP}.
		\end{align*}
	}
\end{bt}

\begin{bt}%[HK1, Hoàng Hoa Thám, HCM 17-18]%[Đào Anh Dương, 10-EX-HCM]%[0D2B1-4]
	Xét tính chẵn, lẻ của hàm số $f(x) = \dfrac{\left| 2x+3 \right| - \left| 2x-3 \right|} {x^3 \left|x \right|}$.
	\loigiai{
		\begin{itemize}
			\item Tập xác định $\mathscr{D} = \mathbb{R} \setminus \{0\}$ là tập đối xứng.
			\item Ta có	$f(-x) = \dfrac{\left| -2x+3 \right| - \left| -2x-3 \right|} {(-x)^3 \left|-x \right|} = \dfrac{\left| 2x-3 \right| - \left| 2x+3 \right|} {-x^3 \left|x \right|} = \dfrac{\left| 2x+3 \right| - \left| 2x-3 \right|} {x^3 \left|x \right|} = f(x)$.
		\end{itemize}
		Vậy, $f(x)$ là hàm số chẵn.
	}
\end{bt}

\begin{bt}%[HK1, Hoàng Hoa Thám, HCM 17-18]%[Đào Anh Dương, 10-EX-HCM]%[0D2B3-2]
	Xác định parabol $(P) \colon y = ax^2 + bx + c$, biết $(P)$ đi qua hai điểm $A(-5;0)$, $B(2;-7)$ và có trục đối xứng là đường thẳng $x = -2$.
	\loigiai{
		Yêu cầu bài toán tương đương với 
		$$\heva{&25a-5b+c=0 \\&4a+2b+c=-7 \\&-\dfrac{b}{2a}=-2} \Leftrightarrow \heva{&25a-5b+c=0 \\&4a+2b+c=-7 \\&4a-b=0} \Leftrightarrow \heva{&a=-1 \\&b=-4 \\&c=5.}$$
		Vậy, $(P) \colon y=-x^2-4x+5$ là parabol cần tìm.
	}
\end{bt}

\begin{bt}%[HK1, Hoàng Hoa Thám, HCM 17-18]%[Đào Anh Dương, 10-EX-HCM]%[0D3B2-6]
	Định $m$ để phương trình $m(x-m+3)=m(x-2)+6$ có tập nghiệm là $\mathbb{R}$.
	\loigiai 
	{
		Ta có
		\begin{eqnarray*}
			& &m(x-m+3)=m(x-2)+6, \forall x \in \mathbb{R} \\
			&\Leftrightarrow &mx-m^2+3m=mx-2m+6, \forall x \in \mathbb{R}\\
			&\Leftrightarrow &m^2-5m+6=0\\
			&\Leftrightarrow &\hoac{&m=2 \\ &m=3.}
		\end{eqnarray*}
		Vậy, $m=2$ và $m=3$ là các giá trị thỏa mãn yêu cầu bài toán.
	}
\end{bt}

\begin{bt}%[HK1, Hoàng Hoa Thám, HCM 17-18]%[Đào Anh Dương, 10-EX-HCM]%[0D3B2-2]
	Giải phương trình $\left|2x^2+x+3\right| - \left|x^2-3x+3\right| = 0$.
	\loigiai 
	{
		Ta có
		\begin{align*}
		&\left|2x^2+x+3\right| - \left|x^2-3x+3\right| = 0 \Leftrightarrow \left|2x^2+x+3\right| = \left|x^2-3x+3\right|\\
		\Leftrightarrow &\hoac{&2x^2+x+3 = x^2-3x+3 \\ &2x^2+x+3 = -x^2+3x-3} \Leftrightarrow \hoac{&x^2+4x=0 \\ &3x^2-2x+6=0} \Leftrightarrow \hoac{&x = 0 \\& x = -4.}
		\end{align*}
		Tập nghiệm của phương trình là $S=\{0;-4\}.$
	}
\end{bt}

\begin{bt}%[HK1, Hoàng Hoa Thám, HCM 17-18]%[Đào Anh Dương, 10-EX-HCM]%[0D3B2-4]%[0D3K2-4]
	Giải các phương trình sau
	\begin{multicols}{2}
		\begin{enumerate}
			\item $\sqrt{3x^2-4x-4} = \sqrt{2x+5}$.
			\item $2x^2 + x - \sqrt{x^2 + 5} - 2x \sqrt{x^2 + 5} = 7$.
		\end{enumerate}
	\end{multicols}
	\loigiai 
	{
		\begin{enumerate}
			\item Ta có
			\begin{align*}
			&\sqrt{3x^2-4x-4} = \sqrt{2x+5} \Leftrightarrow \heva{&2x+5 \ge 0 \\ &3x^2-4x-4 = 2x+5} \\
			\Leftrightarrow &\heva{&x \ge -\dfrac{5}{2} \\ &3x^2 - 6x - 9 = 0} \Leftrightarrow \heva{&x \ge -\dfrac{5}{2}\\ &\hoac{&x = -1 \\ &x = 3}} \Leftrightarrow \hoac{&x=-1 \\ &x=3.}
			\end{align*}
			Tập nghiệm của phương trình là $S=\{-1;3\}.$
			\item Ta có
			\begin{eqnarray*}
				& &2x^2 + x - \sqrt{x^2 + 5} - 2x \sqrt{x^2 + 5} = 7 \Leftrightarrow 2x \left(x - \sqrt{x^2+5}\right) + \left(x - \sqrt{x^2 + 5}\right) = 7 \\
				&\Leftrightarrow &\left(x - \sqrt{x^2 + 5}\right) \cdot (2x+1)=7 \Leftrightarrow \dfrac{-5(2x+1)}{x + \sqrt{x^2+5}} = 7\\
				&\Leftrightarrow &7\sqrt{x^2 + 5} = -17x - 5 \\
				&\Leftrightarrow &\heva{&x \le -\dfrac{5}{17} \\ &49x^2 + 245 = 289x^2 + 170x + 25} \\
				&\Leftrightarrow &\heva{&x \le -\dfrac{5}{17} \\ &240x^2 + 170x - 220 = 0}\\
				&\Leftrightarrow &\heva{&x \le -\dfrac{5}{17} \\ &\hoac{&x = \dfrac{2}{3} \\ &x = -\dfrac{11}{8}}} \Leftrightarrow x = -\dfrac{11}{8}.
			\end{eqnarray*}
			Vậy nghiệm của phương trình là $x=-\dfrac{11}{8}$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[HK1, Hoàng Hoa Thám, HCM 17-18]%[Đào Anh Dương, 10-EX-HCM]%[0H1B4-3]%[0H1K4-3]
	Trong mặt phẳng $Oxy$ cho ba điểm $A(3; 4), B(2; 1), C(6; 3)$.
	\begin{enumerate}
		\item Tìm tọa độ điểm $N$ thỏa mãn $2 \vec{NB}+ \vec{NC}- \vec{NA}= \vec{0}$.
		\item Tìm tọa độ điểm $M(m-2; 2m+1)$ sao cho tam giác $ACM$ vuông tại $A$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Giả sử $N(x; y)$.
			Ta có $\vec{NB}= (2-x; 1-y), \vec{NC}= (6-x; 3-y), \vec{NA}= (3-x; 4-y)$.\\
			Khi đó 
			\begin{align*}
			2 \vec{NB}+ \vec{NC}- \vec{NA}= \vec{0} &\Leftrightarrow \heva{&2(2-x)+(6-x)- (3-x)=0\\ &2(1-y)+ (3-y)- (4-y)=0}\\
			&\Leftrightarrow \heva{&7-2x=0\\ &1-2y=0} \Leftrightarrow \heva{&x= \dfrac{7}{2}\\ &y= \dfrac{1}{2}.}
			\end{align*}
			Vậy, $N \left(\dfrac{7}{2}; \dfrac{1}{2}\right)$ là điểm cần tìm.
			\item Ta có $\vec{AC}= (3; -1), \vec{AM}= (m-5; 2m-3)$\\
			Tam giác $ACM$ vuông tại $A$ $\Leftrightarrow \vec{AC} \cdot \vec{AM}= 0 \Leftrightarrow 
			3(m-5)-1(2m-3)=0 \Leftrightarrow m=12$.\\
			Vậy, $M(10; 25)$ là điểm cần tìm.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[HK1, Hoàng Hoa Thám, HCM 17-18]%[Đào Anh Dương, 10-EX-HCM]%[0D2K3-1]
	Tìm giá trị lớn nhất của hàm số $y= x(2- \sqrt{2}x)$, với $0 \leq x \leq 2$.
	\loigiai{
		Ta có $y= -\sqrt{2}x^2+ 2x$. Do $a= -\sqrt{2}<0$ nên ta có bảng biến thiên
		\begin{center}
			\begin{tikzpicture}[>=stealth]			
			\tkzTabInit[espcl=3,deltacl=1]
			{$x$ /1, $ y$ /3}
			{$0$ ,$1/\sqrt{2}$ , $2$}
			\tkzTabVar{ -/$0$, +/$\dfrac{1}{\sqrt{2}}$, -/$4-4\sqrt{2}$}
			\end{tikzpicture}
		\end{center}
		Dựa vào bảng biến thiên, giá trị lớn nhất của hàm số $y= x(2- \sqrt{2}x)$ là $\dfrac{1}{\sqrt{2}}$ khi $x= \dfrac{1}{\sqrt{2}}$.	
	}
\end{bt}
