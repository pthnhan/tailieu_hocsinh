\begin{bt}%[Đề HK1, Nguyễn Thị Minh Khai, Hồ Chí Minh 17-18]%[Vinh Vo, dự án 10HK1-HCM-18]%[0D3B2-4]
	Giải phương trình $ 1 - \sqrt{5 -3x + x^2} = 2x $.
	\loigiai{
	Ta thấy 
	\begin{align*}
	& \ 1 - \sqrt{5-3x + x^2} = 2x 
	\\
	\Leftrightarrow & \ \sqrt{x^2 - 3x + 5} = 1 - 2x
	\\
	\Leftrightarrow & \ \heva{ & 1-2x \geq 0 \\ & x^2 - 3x + 5 = (1-2x)^2 }
	\\
	\Leftrightarrow & \ \heva{ & x \leq \dfrac{1}{2} \\ & 3x^2 - x - 4 = 0 } 
	\\
	\Leftrightarrow & \ \heva{ & x \leq \dfrac{1}{2} \\ & \hoac{ & x = -1 \\ & x = \dfrac{4}{3} } } \\
	\Leftrightarrow & \ x = -1.
	\end{align*}
	}
\end{bt}	
\begin{bt}%[Đề HK1, Nguyễn Thị Minh Khai, Hồ Chí Minh 17-18]%[Vinh Vo, dự án 10HK1-HCM-18]%[0D3K2-4]
	Giải phương trình $ \sqrt{3x - 5 } + \sqrt{x + 1} = 4 + 4x^2 - x^3 - 3x $.
	\loigiai{
				\item Điều kiện $ \heva{ & 3x - 5 \geq 0 \\ & x + 1 \geq 0 } \Leftrightarrow \heva{ & x \geq \dfrac{5}{3} \\ & x \geq - 1 } \Leftrightarrow x \geq \dfrac{5}{3}. \hfill (*) $\\
	Với điều kiện $ (*) $, ta có 
	{\allowdisplaybreaks
		\begin{align*}
		& \ \sqrt{3x-5} + \sqrt{x+1} = 4 + 4x^2-x^3-3x \tag{1}
		\\
		\Leftrightarrow & \ \left (\sqrt{3x-5} - 2 \right ) + \left ( \sqrt{x+1} -2 \right ) = -x^3 + 4x^2 - 3x
		\\
		\Leftrightarrow & \ \dfrac{3(x - 3)}{ \sqrt{3x - 5} + 2 } + \dfrac{x - 3}{ \sqrt{x+1} + 2 } + x(x-1)(x-3) = 0
		\\
		\Leftrightarrow & \ (x - 3) \left [ \dfrac{3}{ \sqrt{3x -5} + 2 } + \dfrac{1}{ \sqrt{x+1} + 2 } + x(x-1) \right ] = 0
		\\\Leftrightarrow & \ \hoac{ & x = 3 && (2) \\ & \dfrac{3}{ \sqrt{3x -5} + 2 } + \dfrac{1}{ \sqrt{x+1} + 2 } + x(x-1) = 0. && (3)   }
		\end{align*}
	} \\[-9pt]
	Từ $ (*) $ và $ (2) $ ta nhận $ x = 3 $.\\
	Từ $ (*) $ ta được $ \heva{ & x \geq \dfrac{5}{3} \\ & x - 1 \geq \dfrac{2}{3} } \Rightarrow x(x-1) > 0.$ \hfill $ (4) $\\
	Từ $ (3) $ và $ (4) $, ta thấy $ (3) $ vô nghiệm.\\
	Vậy phương trình $ (1) $ có nghiệm $ x = 3 $.	
	}	
\end{bt}	
\begin{bt}%[Đề HK1, Nguyễn Thị Minh Khai, Hồ Chí Minh 17-18]%[Vinh Vo, dự án 10HK1-HCM-18]%[0D3B3-4]
	Giải hệ phương trình  $ \heva{ & x + y + xy = 5 \\ & x^2 + y^2 = 5. } $
	\loigiai{
		 Đặt $ \heva{ & S = x + y \\ & P = xy } $. \quad Điều kiện $ S^2 - 4P \geq 0. $ \hfill $ (*) $\\
		Ta thấy 
		\begin{align*}
		& \ \heva{ & x+ y + xy = 5 \\ & x^2 + y^2 = 5 } \tag{I}
		\\
		\Leftrightarrow 
		& \ \heva{ & x + y  + xy = 5 \\ & (x+ y )^2 - 2xy = 5 }
		\\
		\Rightarrow 
		& \ \heva{& S + P = 5 \\ & S^2 - 2P = 5} 
		\\
		\Leftrightarrow 
		& \ \heva{&  P = 5 - S \\& S^2 + 2S - 15 = 0}
		\\
		\Leftrightarrow 
		& \ \heva{& P = 5 - S \\ & \hoac{  & S = -5 \\ & S = 3 }}
		\\
		\Leftrightarrow 
		& \  \hoac{ 
				& 	\heva{& S = 3  \\& P = 2 }\quad  (\text{so với } (*) \text{ nhận}) \\
				& \heva{& S = -5 \\ & P = 10 } \quad (\text{so với } (*) \text{ loại}). 
			}
		\end{align*}
		Khi đó, $ x, y $ là nghiệm của phương trình $ X^2 - 3X + 2 = 0 \Leftrightarrow X = 1 \lor X = 2 $.\\
		Vậy hệ $ (\mathrm{I}) $ có tập nghiệm $\Big \{ (1;2),\ (2;1) \Big \} $.
		
	}	
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Thị Minh Khai, Hồ Chí Minh 17-18]%[Vinh Vo, dự án 10HK1-HCM-18]%[0D3B2-6]
Tìm giá trị tham số $ m $ sao cho phương trình $ m^2 x = 4x - 2m + m^2 $ có nghiệm tuỳ ý.
	\loigiai{
		Ta thấy 
		\begin{align*}
		& \ m^2 x = 4x - 2m + m^2 \text{ có nghiệm tuỳ ý}
		\\
		\Leftrightarrow 
		& \ \left (m^2 -  4 \right ) x - m^2 + 2m = 0 \ \forall m \in \mathbb{R}
		\\
		\Leftrightarrow 
		& \ \heva{ & m^2 - 4 = 0 \\ & m^2 - 2m = 0  }
		\\
		\Leftrightarrow 
		& \ m = 2.		
		\end{align*}
	}	
\end{bt}
\begin{bt}%[Đề HK1, Nguyễn Thị Minh Khai, Hồ Chí Minh 17-18]%[Vinh Vo, dự án 10HK1-HCM-18]%[0D3B2-5]
Tìm giá trị tham số $ m $ sao cho 	phương trình $ x^2 + 2mx  + 4 = 0 $ có hai nghiệm $ x_1, x_2 $ thỏa mãn $ \big |x_1 - x_2 \big | = 2\sqrt{2} $.
	\loigiai{
		Ta thấy 
		\begin{align*}
		& \ x^2 + 2m x + 4  = 0 	\text{ có hai nghiệm } x_1, x_2
		\\
		\Leftrightarrow 
		& \
		m^2 - 4 \geq 0 \\
		\Leftrightarrow 
		& \ m \leq -2 \lor m \geq 2. \tag{1}
		\end{align*}	
		Khi đó, theo hệ thức Vi-ét ta có $ \heva{ & x_1 + x_2 = - 2m \\ & x_1 x_2 = 4.} $ \hfill $ (2) $ \\
		Ta thấy 
		\begin{align*}
		& \ \big | x_1 - x_2 \big | = 2\sqrt{2} 
		\\ 
		\Leftrightarrow 
		& \ \left ( x_1 - x_2\right )^2 = 8 \\
		\Leftrightarrow 
		& \  \left ( x_1 + x_2 \right )^2 - 4x_1x_2 = 8\\
		\stackrel{(2)}{\Rightarrow}
		& \  (-2m)^2 - 4 \cdot 4 = 8 \\
		\Leftrightarrow 
		& \ m = \pm \sqrt{6} \quad (\text{vì } (1) \text{ nhận} ).
		\end{align*}
	}	
\end{bt}
\begin{bt}%[Đề HK1, Nguyễn Thị Minh Khai, Hồ Chí Minh 17-18]%[Vinh Vo, dự án 10HK1-HCM-18]%[0D4B1-2]
Tìm giá trị lớn nhất của hàm số $  y = x \sqrt{1 - x^2} $ với $ 0<x<1 $.	
\loigiai{ 
	Với $ 0 < x < 1 $, ta có
	\begin{align*}
		& \ x \cdot \sqrt{1 - x^2}
		\\
		\Leftrightarrow 
		& \ \sqrt{x^2} \cdot \sqrt{1 - x^2} \stackrel{Cauchy}{\leq} \dfrac{x^2 + 1 - x^2}{2}
		\\
		\Leftrightarrow 
		& \ \sqrt{x^2} \cdot \sqrt{1 - x^2} \leq \dfrac{1}{2}. \tag{1}  
	\end{align*}	
	Đẳng thức $ (1) $ xảy ra $  \Leftrightarrow \sqrt{x^2} = \sqrt{1 - x^2} \Rightarrow x = \dfrac{ \sqrt{2} }{2} $.\\
	Vậy $ \max \limits_{(0;1)} y = \dfrac{1}{2} $ khi $ x = \dfrac{ \sqrt{2}}{2} $.
}	
\end{bt}

Cho $ \triangle ABC $ có $ K $ là trung điểm $ AB $. Gọi $ I, J $ là các điểm thỏa mãn $ \overrightarrow{AI} = \dfrac{1}{3} \overrightarrow{AC}$ và $2 \overrightarrow{JB} = \overrightarrow{JC} $.	Chứng minh rằng $ K, I, J $ thẳng hàng.
	\loigiai{
		\immini{
			Vì $ K $ là trung điểm $ AB $, ta có 
			\begin{align*}
			& \ 2 \overrightarrow{ JK } = \overrightarrow{ JA } + \overrightarrow{ JB }  
			\\
			\Rightarrow 
			& \ 4 \overrightarrow{ JK  }  = 2 \overrightarrow{ JA } + 2 \overrightarrow{ JB }  
			\\
			\Leftrightarrow 
			& \ 4 \overrightarrow{ JK } = 2 \overrightarrow{ JA } + \overrightarrow{ JC }\quad (\text{do } 2 \overrightarrow{ JB} = \overrightarrow{JC } )   \\
			\Leftrightarrow 
			& \ 4 \overrightarrow{ JK } = 3   \overrightarrow{ JI } + 2 \overrightarrow{ IA } + \overrightarrow{ IC }  \\
			\Rightarrow
			&\
			4\overrightarrow{ JK } = 3 \overrightarrow{ JI } \quad  (\text{do } 2 \overrightarrow{ AI} = \overrightarrow{IC } ).
			\end{align*}
			Vậy $ I, J, K $ thẳng hàng.
		}{
			\begin{tikzpicture}
			\tkzDefPoint(0,0){B}
			\tkzDefShiftPoint[B](0:3){C}
			\tkzDefShiftPoint[B](60:3){A}
			\tkzDefShiftPoint[B](180:3){J}
			\tkzDefMidPoint(B,A) \tkzGetPoint{K}
			\tkzDefBarycentricPoint(C=1,A=2) \tkzGetPoint{I}
			
			\tkzDrawSegments(J,A A,C J,C B,A J,I)
			\tkzDrawPoints[fill = black](A,B,C,J,K,I)
			\tkzLabelPoints(B,C,J,K)
			\tkzLabelPoints[above right](A,I)

		}

\begin{bt}%[Đề HK1, Nguyễn Thị Minh Khai, Hồ Chí Minh 17-18]%[Vinh Vo, dự án 10HK1-HCM-18]%[0H1B3-6]
Cho $ \triangle ABC $. 	Tìm tập hợp các điểm $ M $ sao cho $ | 2 \overrightarrow{MA} - 3 \overrightarrow{MB} -2\overrightarrow{MC} | = | \overrightarrow{MB} - \overrightarrow{MC} | $.
	\loigiai{
		Gọi $ I $ thỏa mãn
		\begin{align*}
		& \ 2\overrightarrow{IA} - 3\overrightarrow{IB} -2 \overrightarrow{IC} = \overrightarrow{0} \\
		\Leftrightarrow 
		& \ 3 \overrightarrow{BI} = 2\overrightarrow{CA} \\
		\Leftrightarrow 
		& \ I \text{ cố định}. \tag{1}  
		\end{align*}
		Ta thấy 
		\begin{align*}
		& \ \Big| 2\overrightarrow{ MA } -3\overrightarrow{ MB } -2\overrightarrow{MC  } \Big| = \Big|\overrightarrow{  MB}-\overrightarrow{ MC }\Big| \tag{*} \\
		\Leftrightarrow 
		& \ \Big| 2 \left (\overrightarrow{MI} + \overrightarrow{IA} \right ) - 3 \left  (\overrightarrow{MI} + \overrightarrow{IB} \right  ) - 2  \left (\overrightarrow{MI} +\overrightarrow{IC}\right  )\Big| = \Big| \overrightarrow{BC}  \Big| 
		\\ 
		\Leftrightarrow 
		& \ \Big | - 3 \overrightarrow{MI} \Big | = \Big | BC \Big | \quad (\text{do } (1) ) \\
		\Leftrightarrow
		& \ IM = \dfrac{BC}{3}. \tag{2}
		\end{align*}
		Từ $ (2) $, ta được tập hợp  các điểm $ M $ thỏa mãn $ (*) $ là đường tròn tâm $ I $ bán kính $ R = \dfrac{BC}{3} $.	
	}	
\end{bt}
\begin{bt}%[Đề HK1, Nguyễn Thị Minh Khai, Hồ Chí Minh 17-18]%[Vinh Vo, dự án 10HK1-HCM-18]%[0H3B1-6]
	Trong mặt phẳng toạ độ $ Oxy $, cho $ A(-2;2)$, $B(1;0)$, $C(3;-3) $. Tính toạ độ trực tâm $ H $ của $ \triangle ABC $.
	\loigiai{
	Gọi điểm  $ H(x;y) $. Ta có $$ \heva{ & \overrightarrow{ AH } = (x+2;y-2)\\ & \overrightarrow{ BH } = (x-1;y)\\ & \overrightarrow{ BC} = (2;-3) \\ & \overrightarrow{AC } = (5;-5).  } $$
	$ H $ là trực tâm $ \triangle ABC $, ta được
	\begin{align*}
	& \ \heva{ & \overrightarrow{ AH } \cdot \overrightarrow{ BC } = 0 \\ & \overrightarrow{ BH } \cdot \overrightarrow{ AC } = 0 }\\
	\Leftrightarrow 
	& \ \heva{ & 2x - 3y = - 10\\ & 5x - 5y = 5 }\\
	\Leftrightarrow 
	& \ \heva{& x = 13 \\ &y = 12.}
	\end{align*}	
	}	
\end{bt}
\begin{bt}%[Đề HK1, Nguyễn Thị Minh Khai, Hồ Chí Minh 17-18]%[Vinh Vo, dự án 10HK1-HCM-18]%[0H3B1-6]
Trong mặt phẳng toạ độ $ Oxy $, cho $ A(-2;2)$, $B(1;0)$, $C(3;-3) $. Tính toạ độ điểm $ D $ thuộc trục $ Oy $ sao cho $ ABCD $ là hình thang có cạnh đáy lớn là $ BC $.	
	\loigiai{
		Vì $ D $ thuộc $ Oy $ nên ta có $ D(0;y) $.\\
		Ta có $ \heva{ & \overrightarrow{ AD } = ( 2; y - 2) \\ & \overrightarrow{ BC} = (2;-3).    } $	\\
		Véc-tơ $ \overrightarrow{AD }  $ cùng hướng với véc-tơ $ \overrightarrow{ BC}  $ 
			\begin{align*}
				\Leftrightarrow  & \ 
				\overrightarrow{ AD } = k \cdot \overrightarrow{ BC },\ k \in \mathbb{R}^+ \\
				\Leftrightarrow & \ 
				\heva{& 2k = 2 \\ & -3k = y - 2} \\
				\Leftrightarrow & \
				\heva{& k = 1 \\& y = - 1.}
			\end{align*}
		Ta thấy $ D(0;-1) $ nên $ AD = BC $. Vì bài toán yêu cầu xác định $ D $ để $ ABCD $ là hình thang có đáy lớn là $ BC $ nên trong trường hợp này ta thấy hình bình hành $ ABCD $ không thỏa mãn yêu cầu bài toán.
}
