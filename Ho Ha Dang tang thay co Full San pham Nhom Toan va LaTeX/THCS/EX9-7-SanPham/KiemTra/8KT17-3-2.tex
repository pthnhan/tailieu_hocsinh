\begin{ex}%[Đề thi giữa kì 2 Toán 8, Trường THCS Chu Mạnh Trinh, 2017]%[8D3B2]%[8D3B4]%[8D3K5]%[8D3K4]

a) $5x-4= 21.$

b) $2x^3+6x^2=x^2+3x.$ 

c) $\dfrac{2}{x-3}+\dfrac{3}{x+3} = \dfrac{3x+5}{x^2-9}.$

d) $(x+1)^2(x+2)+(x-1)^2(x-2) = 12.$

\loigiai{
a) $x=5.$

b) PT $\Leftrightarrow x(2x^2+5x-3) = 0 \Leftrightarrow \hoac{& x=0\\  & (2x-1)(x+3) =0 }$

x=-3 \\
x=\dfrac{1}{2}}.$ 

Vậy $ x \in \left\{0;-3;\dfrac{1}{2} \right\}.$

c) Điều kiện $x\ne \pm 1.$ PT $\Leftrightarrow 2(x+3)+3(x-3) = 3x+5 \Leftrightarrow x=5$. Vậy $x=5.$

d) PT $\Leftrightarrow x^3+5x-6=0 \Leftrightarrow(x-1)(x^2+x+6) = 0 \Leftrightarrow \hoac{x-1=0 \\ \left(x+\dfrac{1}{2}\right)^2+\dfrac{23}{4}=0} \Leftrightarrow x=1.$

trung điểm của các cạnh $AB, BC; M$ là giao điểm của $CE$ và $DF$. 

a) Chứng minh  tam giác $BEC$ và tam giác $CFD$ bằng nhau và $CE \perp DF.$  

b) Chứng minh $CM.CE = CB.CF.$ 

c) Tính diện tích tam giác $MDC.$
\loigiai{
\immini{
a) $\triangle BEC =\triangle CFD$ (c.g.c) $\Rightarrow \widehat{BEC}=\widehat{DFC}.$  Suy ra $DF \perp EC = M.$

b) Xét tam giác $CMF$ và tam giác $CBE$, có $\widehat{CMF} = \widehat{CBE} = 90^\circ$ và $\widehat{MCF}$ chung. Suy ra 

$\triangle CMF \backsim \triangle CBE$ (g.g)  $\Rightarrow \dfrac{CM}{CB} = \dfrac{CF}{CE} \Rightarrow CM.CE = CB.CF$

\begin{ex}%[Đề thi giữa kì 2 Toán 8, Trường THCS Chu Mạnh Trinh, 2017]%[8DXGX]
Cho $a, b, c$ là các số thực dương thỏa mãn $a b c = 1.$  Chứng minh rằng: 
\begin{center}
$\dfrac{2}{(a+1)^2+b^2+1}+\dfrac{2}{(b+1)^2+c^2+1}+\dfrac{2}{(c+1)^2+a^2+1} \le 1.$
\end{center}
\loigiai{
Áp dụng bất đẳng thức $x^2+y^2 \ge 2xy,$ ta có $a^2+b^2 \ge 2ab.$ Khi đó
\begin{center}
$\dfrac{2}{(a+1)^2+b^2+1} = \dfrac{2}{a^2+b^2+2a+2} \le \dfrac{2}{2ab+2a+2}=\dfrac{1}{ab+a+1}.$
\end{center}

Lập luận tương tự ta có 
\begin{center}
$\dfrac{2}{(b+1)^2+c^2+1} \le \dfrac{1}{bc+b+1}, \dfrac{2}{(c+1)^2+a^2+1}\le \dfrac{1}{ca+c+1}.$
\end{center}
Suy ra
\begin{center}
$\begin{array}{ll}
VT  & \le \dfrac{1}{ab+a+1}+\dfrac{1}{bc+b+1}+ \dfrac{1}{ca+c+1} \\
&=  \dfrac{1}{ab+a+1}+\dfrac{a}{abc+ab+a}+ \dfrac{ab}{abca+abc+ab}\\
&= \dfrac{1}{ab+a+1}+\dfrac{a}{1+ab+a}+ \dfrac{ab}{a+1+ab} =1
\end{array}$
\end{center}
Dấu bằng xảy ra khi 
$a=b=c=1.$

Vậy bất đẳng thức được chứng minh xong.
}
\end{ex}
