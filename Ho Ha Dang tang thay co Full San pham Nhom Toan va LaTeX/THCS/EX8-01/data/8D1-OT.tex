\begin{vd}
Thực hiện phép tính
\begin{enumerate}
	\item $ A=(2x-1)(3x^{2}-7x+5) $;
	\item $ B=(y-1)(y^{2}+y+1)+\left(\dfrac{1}{3}x^{2}y-y\right)(2x+y^{2}) $.
\end{enumerate}
\loigiai
{
	\begin{enumerate}
		\item Ta có $ A=6x^{3}-14x^{2}+10x-3x^{2}+7x-5=6x^{3}-17x^{2}+17x-5 $.
		\item $ B=(y^{3}-1)+\left(\dfrac{2}{3}x^{3}y-2xy+\dfrac{1}{3}x^{2}y^{3}-y^{3}\right)=\dfrac{2}{3}x^{3}y-2xy+\dfrac{1}{3}x^{2}y^{3}-1 $.
	\end{enumerate}
}
\end{vd}
\begin{vd}
	Rút gọn các biểu thức sau
	\begin{enumerate}
		\item $ A=(x-y)(x+y)(x^{2}+y^{2})(x^{4}+y^{4}) $;
		\item $ B=(x+2)(x^{2}+2x+4)-(x-2)^{3}-6(x-1)(x+1) $.
	\end{enumerate}
	\loigiai
	{
	\begin{enumerate}
		\item Ta có $ A=(x^{2}-y^{2})(x^{2}+y^{2})(x^{4}+y^{4})=(x^{4}-y^{4})(x^{4}+y^{4})=x^{8}-y^{8} $.
		\item Ta có $ B=(x^{3}+4x^{2}+8x+8)-(x^{3}-6x^{2}+12x-8)-6(x^{2}-1)=4x^{2}-4x+22 $.
	\end{enumerate}	
	}
\end{vd}

\begin{vd}
	Tính nhanh
	\begin{enumerate}
		\item $ A=39^{2}+78\cdot 61+61^{2} $;
		\item $ B=2{,}83\cdot 5{,}68-2{,}83\cdot 4{,}68+1{,}17\cdot 5{,}68-1{,}17\cdot 4{,}68 $;
		\item $ C=20^{2}+18^{2}+16^{2}+\cdots+4^{2}+2^{2}-(19^{2}+17^{2}+\cdots+3^{2}+1^{2}) $.
	\end{enumerate}
	\loigiai
	{
		\begin{phantich}
			Ta thấy
			\begin{enumerate}
				\item $ 78\cdot 61=2\cdot 39\cdot 61 $.
				\item $ 5{,}68-4{,}68=1$ và $2{,}83+1{,}17=4 $.
				\item Ta có thể nhóm hai số liên tiếp để sử dụng hằng đẳng thức dạng $ a^{2}-b^{2} $.
			\end{enumerate}
		Do đó ta có lời giải như sau
		\end{phantich}
		\begin{enumerate}
			\item Ta có $ A=39^{2}+78\cdot 61+61^{2}=39^{2}+2\cdot 39\cdot 61+61^{2}=(39+61)^{2}=100^{2}=10000 $.
			\item Ta có $ B=2{,}83(5{,}68-4{,}68)+1{,}17(5{,}68-4{,}68)=(5{,}68-4{,}68)(2{,}83+1{,}17)=1\cdot 4=4 $.
			\item Ta có
			\begin{eqnarray*}
				C&=&(20^{2}-19^{2})+(18^{2}-17^{2})+(16^{2}-15^{2})+\cdots+(4^{2}-3^{2})+(2^{2}-1^{2})\\
				&=&(20-19)(20+19)+(18-17)(18+17)+(16-15)(16+15)\\
				&&\qquad+\cdots+(4-3)(4+3)+(2-1)(2+1)\\
				&=&39+35+31+\cdots+7+3=756.
			\end{eqnarray*}
		\end{enumerate}
		\begin{note}
			Với những bài toán tính nhanh, ta cần phải biến đổi biểu thức cần tính để áp dụng các hằng đẳng thức hoặc phân tích thành nhân tử (ưu tiên tạo ra các số tròn chục, tròn trăm $\ldots$).
		\end{note}
	}
\end{vd}
\begin{vd}
	Tính nhanh
	\begin{enumEX}{2}
		\item $ A=a^{2}+\dfrac{1}{2}a+\dfrac{1}{16} $ với $ a=49{,75} $;
		\item $ B=a^{2}-b^{2}-2b-1 $ với $ a=93 $, $ b=6 $.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item Ta có $ A=a^{2}+2\cdot\dfrac{1}{4}\cdot a+\dfrac{1}{4^{2}}=\left(a+\dfrac{1}{4}\right)^{2} $.\\
			Với $ a=49{,75}=49\dfrac{3}{4} $, ta có $ A=\left(49\dfrac{3}{4}+\dfrac{1}{4}\right)^{2}=50^{2}=2500 $.
			\item Ta có $ B=a^{2}-(b^{2}+2b+1)=a^{2}-(b+1)^{2}=(a-b-1)(a+b+1) $.\\
			Với $ a=93 $, $ b=6 $ ta có $ B=(93-6-1)(93+6+1)=86\cdot 100=8600 $.
		\end{enumerate}
	}
\end{vd}
\begin{vd}
	Rút gọn rồi tính giá trị của các biểu thức sau
	\begin{enumerate}
		\item $ A=5x^{2}(3x-2)-(4x+7)(6x^{2}-x)-(7x-9x^{3}) $ với $ x=-\dfrac{3}{4} $;
		\item $ B=(2x+y)(4x^{2}-2xy+y^{2})-(2x-y)(4x^{2}+2xy+y^{2}) $ với $ y=10 $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			\begin{eqnarray*}
				A&=&5x^{2}(3x-2)-(4x+7)(6x^{2}-x)-(7x-9x^{3})\\
				&=&15x^{3}-10x^{2}-(24x^{3}-4x^{2}+42x^{2}-7x)-(7x-9x^{3})\\
				&=&15x^{3}-10x^{2}-24x^{3}+4x^{2}-42x^{2}+7x-7x+9x^{3})\\
				&=&-48x^{2}.
			\end{eqnarray*}
			Với $ x=-\dfrac{3}{4} $ thì $ A=-48\cdot \left(-\dfrac{3}{4}\right)^{2}=-48\cdot \dfrac{9}{16}=-27 $.
			\item Ta có $ B=(2x)^{2}+y^{3}-[(2x)^{3}-y^{3}]=8x^{3}+y^{3}-8x^{3}+y^{3}=2y^{3} $.\\
			Với $ y=10 $ thì $ B=2\cdot 10^{3}=2000 $.
		\end{enumerate}
}
\end{vd}
\begin{vd}
	Tính giá trị của biểu thức
	\begin{enumerate}
		\item $ A=2x^{2}+6xy-10x $ với $ x=-4 $, $ y=3 $;
		\item $ B=2x^{2}(x^{2}+y^{2})+2y^{2}(x^{2}+y^{2})+5(y^{2}+x^{2}) $ với $ x^{2}+y^{2}=1 $.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $ A=2x^{2}+6xy-10x=2x(x+3y-5)=2\cdot (-4)(-4+3\cdot 3-5)=0 $.
			\item Ta có
			\begin{eqnarray*}
				B&=&2x^{2}(x^{2}+y^{2})+2y^{2}(x^{2}+y^{2})+5(y^{2}+x^{2})\\
				&=&(x^{2}+y^{2})(2x^{2}+2y^{2}+5)\\
				&=&(x^{2}+y^{2})[2(x^{2}+y^{2})+5]\\
				&=&1\cdot (2\cdot 1+5)=7.
			\end{eqnarray*}
		\end{enumerate}
	}
\end{vd}
\begin{vd}
	Cho $ x+y=15 $ và $ xy=-100 $. Tính giá trị của biểu thức
	\begin{enumEX}{3}
		\item $ A= x^{2}+y^{2} $;
		\item $ B= x^{4}+y^{4} $;
		\item $ C= x^{2}-y^{2} $.
	\end{enumEX}
	\loigiai{
		\begin{phantich}
			\begin{enumerate}
				\item Ta thấy biểu thức cần tính trong bài toán có dạng $ a^{2}+b^{2} $ nên có thể đưa về dạng $ (a+b)^{2} $ hoặc $ (a-b)^{2} $.
				\item Chú ý $ x^{4}=(x^{2})^{2} $.
				\item Áp dụng hằng đẳng thức dạng $ a^{2}-b^{2} $.
			\end{enumerate}
		Do đó ta có lời giải sau.
		\end{phantich}
		\begin{enumerate}
			\item Ta có $ A=(x+y)^{2}-2xy=15^{2}-2\cdot (-100)=425 $.
			\item Ta có $ B=(x^{2}+y^{2})^{2}-2x^{2}y^{2}=425^{2}-2\cdot (-100)^{2}=160625 $.
			\item Ta có $ C=(x-y)(x+y) $. Mà
			\[ 
			(x-y)^{2}=(x+y)^{2}-4xy=15^{2}-4\cdot (-100)=625\Rightarrow x-y=\pm 25.
			\]
			\begin{enumerate}[+)]
				\item Với $ x-y=25 $ thì $ C=15\cdot 25=375 $.
				\item Với $ x-y=-25 $ thì $ C=15\cdot (-25)=-375 $.
			\end{enumerate}
		\end{enumerate}
		\begin{note}
			Trong bài toán cho ta biết tổng $ x+y $ và tích $ xy $ thì ta phải biến đổi để sử dụng được hằng đẳng thức dạng $ (a+b)^{2} $ hoặc $ (a-b)^{2} $.
		\end{note}
	}
\end{vd}

\begin{vd}
Chứng minh đẳng thức
\begin{enumerate}
	\item $ 3(x^{2}+y^{2}+z^{2})-(x-y)^{2}-(y-z)^{2}-(z-x)^{2}=(x+y+z)^{2} $;
	\item $ (x+y+z)^{3}-(x+y-z)^{3}-(x+z-y)^{3}-(y+z-x)^{3}=24xyz $;
	\item $ (x^{2}+y^{2})^{3}+(z^{2}-x^{2})^{3}-(y^{2}+z^{2})^{3}=3(x^{2}+y^{2})(x^{2}-z^{2})(y^{2}+z^{2}) $.
\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item $ 3(x^{2}+y^{2}+z^{2})-(x-y)^{2}-(y-z)^{2}-(z-x)^{2}=(x+y+z)^{2} $.
		\begin{enumerate}[+)]
			\item Biến đổi vế trái ta được:
			\begin{eqnarray*}
				VT&=&3(x^{2}+y^{2}+z^{2})-(x-y)^{2}-(y-z)^{2}-(z-x)^{2}\\
				&=&3x^{2}+3y^{2}+3z^{2}-(x^{2}-2xy+y^{2})-(y^{2}-2yz+z^{2})-(z^{2}-2xz+x^{2})\\
				&=&3x^{2}+3y^{2}+3z^{2}-x^{2}+2xy-y^{2}-y^{2}+2yz-z^{2}-z^{2}+2xz-x^{2}\\
				&=&x^{2}+y^{2}+z^{2}+2xy+2yz+2xz.
			\end{eqnarray*}
			\item Biến đổi vế phải ta được:
			\begin{eqnarray*}
				VP&=&(x+y+z)^{2}\\
				&=&(x+y)^{2}+2(x+y)z+z^{2}\\
				&=&x^{2}+2xy+y^{2}+2xz+2yz+z^{2}\\
				&=&x^{2}+y^{2}+z^{2}+2xy+2yz+2xz.
			\end{eqnarray*}
		\end{enumerate}
		Do đó $ VT=VP $ (đpcm).
		\item $ (x+y+z)^{3}-(x+y-z)^{3}-(x+z-y)^{3}-(y+z-x)^{3}=24xyz $.\\
		Nhận xét $ (A+B)^{3}-(A-B)^{3}=2B^{3}+6A^{2}B $ và $ (A+B)^{3}+(A-B)^{3}=2A^{3}+6AB^{2} $ nên ta có
		\begin{eqnarray*}
			VT&=&[((y+z)+x)^{3}-((y+z)-x)^{3}]-[(x+(y-z))^{3}+(x-(y-z))^{3}]\\
			&=&(2x^{3}+6(y+z)^{2}x)-(2x^{3}+6x(y-z)^{2})\\
			&=&6x(y^{2}+z^{2}+2yz)-6x(y^{2}+z^{2}-2yz)=24xyz=VP.
		\end{eqnarray*}
	Suy ra điều phải chứng minh.
		\item $ (x^{2}+y^{2})^{3}+(z^{2}-x^{2})^{3}-(y^{2}+z^{2})^{3}=3(x^{2}+y^{2})(x^{2}-z^{2})(y^{2}+z^{2}) $.\\
		Nhận xét $ y^{2}+z^{2}=(x^{2}+y^{2})+(z^{2}-x^{2}) $ nên ta có
		\begin{eqnarray*}
			VT&=&(x^{2}+y^{2})^{3}+(z^{2}-x^{2})^{3}-[(x^{2}+y^{2})+(z^{2}-x^{2})]^{3}\\
			&=&(x^{2}+y^{2})^{3}+(z^{2}-x^{2})^{3}-[(x^{2}+y^{2})^{3}+(z^{2}-x^{2})^{3}\\
			&&\quad+3(x^{2}+y^{2})(z^{2}-x^{2})[(x^{2}+y^{2})+(z^{2}-x^{2})]\\
			&=&3(x^{2}+y^{2})(x^{2}-z^{2})(y^{2}+z^{2})=VP.
		\end{eqnarray*}
		Suy ra điều phải chứng minh.
	\end{enumerate}
}
\begin{note}
Ta nhớ các kết quả sau
\begin{enumerate}[1)]
	\item $ (x+y+z)^{2}=x^{2}+y^{2}+z^{2}+2xy+2yz+2xz $.
	\item $ (A+B)^{3}-(A-B)^{3}=2B^{3}+6A^{2}B $.
	\item $ (A+B)^{3}+(A-B)^{3}=2A^{3}+6AB^{2} $.
\end{enumerate}
\end{note}
\end{vd}

\begin{vd}
Phân tích đa thức thành nhân tử bằng phương pháp đặt nhân tử chung
\begin{enumEX}{2}
\item $ 35x^{2}y^{3}-14x^{2}y^{2}+49x^{2}y $;
\item $ -18x^{4}y^{2}-27x^{3}y^{3}-45x^{2}y^{4} $.
\end{enumEX}
\loigiai{
\begin{enumerate}
	\item Ta có $ 35x^{2}y^{3}-14x^{2}y^{2}+49x^{2}y=7x^{2}y(5y^{2}-2y+7) $.
	\item Ta có $ -18x^{4}y^{2}-27x^{3}y^{3}-45x^{2}y^{4}=-9x^{2}y^{2}(2x^{2}+3xy+5y^{2}) $.
\end{enumerate}
}
\end{vd}
\begin{vd}
	Phân tích đa thức thành nhân tử bằng phương pháp dùng hằng đẳng thức
	\begin{enumEX}{3}
		\item $ x^{3}-8 $;
		\item $ 27x^{3}+125y^{3} $;
		\item $ x^{6}+216 $.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item Ta có $ x^{3}-8=(x-2)(x^{2}+2x+4) $.
			\item Ta có $ 27x^{3}+125y^{3}=(3x)^{3}+(5y)^{3}=(3x+5y)(9x^{2}-15xy+25y^{2}) $.
			\item Ta có $ x^{6}+216=\left(x^{2}\right)^{3}+6^{3}=(x^{2}+6)(x^{4}-6x^{2}+36) $.
		\end{enumerate}
	}
\end{vd}
\begin{vd}
	Phân tích đa thức thành nhân tử bằng phương pháp nhóm hạng tử
	\begin{enumEX}{2}
		\item $ x^{2}+7x+7y-y^{2} $;
		\item $ x^{2}-2x-9y^{2}+6y $.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item Ta có $ x^{2}+7x+7y-y^{2}=(x^{2}-y^{2})+7(x+y)=(x+y)(x-y)+7(x+y)=(x+y)(x-y+7) $.
			\item Ta có
			\begin{eqnarray*}
				x^{2}-2x-9y^{2}+6y&=&(x^{2}-9y^{2})-(2x-6y)\\
				&=&(x-3y)(x+3y)-2(x-3y)=(x-3y)(x+3y-2).
			\end{eqnarray*}
		\end{enumerate}
	}
\begin{note}
Khi sử dụng phương pháp nhóm hạng tử để phân tích đa thức thành nhân tử, ta chú ý việc nhóm các số hạng sao cho có thể sử dụng hằng đẳng thức hoặc đặt được nhân tử chung (chú ý các hạng tử có hệ số bằng nhau).
\end{note}
\end{vd}
\begin{vd}
	Phân tích đa thức thành nhân tử bằng phương pháp tách hạng tử
	\begin{enumEX}{2}
		\item $ x^{2}+5x+6 $;
		\item $ x^{2}-8x+15 $.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item Ta có $ x^{2}+5x+6=x^{2}+2x+3x+6=x(x+2)+3(x+2)=(x+2)(x+3) $.
			\item Ta có
			\begin{eqnarray*}
				x^{2}-8x+15&=&x^{2}-3x-5x+15=(x^{2}-3x)-(5x-15)\\
				&=&x(x-3)-5(x-3)=(x-3)(x-5).
			\end{eqnarray*}
		\end{enumerate}
	}
	\begin{note}
		Khi sử dụng phương pháp tách hạng tử ta chú ý đến kết quả sau
		\[ 
		x^{2}+(a+b)x+ab=(x+a)(x+b).
		\]
	\end{note}
\end{vd}
\begin{vd}
	Phân tích đa thức thành nhân tử bằng phương pháp thêm bớt hạng tử
	\begin{enumEX}{2}
		\item $ 4x^{4}+y^{4} $;
		\item $ 81x^{4}+4 $.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			\begin{eqnarray*}
			4x^{4}+y^{4}&=&(2x^{2})^{2}+(y^{2})^{2}=(2x^{2})^{2}+(y^{2})^{2}+4x^{2}y^{2}-4x^{2}y^{2}\\
			&=&(2x^{2}+y^{2})^{2}-(2xy)^{2}=(2x^{2}+y^{2}-2xy)(2x^{2}+y^{2}+2xy).
			\end{eqnarray*}
			\item Ta có
