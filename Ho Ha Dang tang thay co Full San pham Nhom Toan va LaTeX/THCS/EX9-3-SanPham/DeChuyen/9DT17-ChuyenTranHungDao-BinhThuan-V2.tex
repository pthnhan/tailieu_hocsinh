\begin{ex}%[Đề thi vào 10, Chuyên Trần Hưng Đạo, Bình Thuận, 2017]%[9D3K3]
Giải hệ phương trình: $\heva{&y^2+1=xy\\
&x^2+y^2+1+2(x+y)=0}$
\loigiai{
Ta có 
\begin{eqnarray*}
\heva{&y^2+1=xy\\
	&x^2+y^2+1+2(x+y)=0}&\Leftrightarrow&\heva{&y^2+1=xy\\
	&x^2+xy+2(x+y)=0}\Leftrightarrow\heva{&y^2+1=xy\\
	&(x+y)(x+2)=0}
\\&\Leftrightarrow&\heva{&y^2+1=xy\\
	&\hoac{&x=-2\\&x=-y}}\Leftrightarrow \hoac{&\heva{&x=-2\\&y=-1}\\&\heva{&x=-y\\&y^2+y+1=0\,(\text{vô nghiệm})}}
\end{eqnarray*}
Vậy nghiệm của hệ phương trình là $(-2;-1)$.
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên Trần Hưng Đạo, Bình Thuận, 2017]%[9D5G2]
Cho $n$ số nguyên $a_1,a_2,\ldots,a_n$ thỏa mãn $S=a_1+a_2+\ldots+a_n$ chia hết cho $6$. Chứng minh $P=a_1^3+a_2^3+\ldots+a_n^3$ cũng chia hết cho $6$.
\loigiai{
Ta có 
\begin{eqnarray*}
P-S &=&(a_1^3-a_1)+(a_2^3-a_2)+\ldots+(a_n^3-a_n)\\
&=&(a_1-1)a_1(a_1+1)+(a_2-1)a_2(a_2+1)+\ldots+(a_n-1)a_n(a_n+1).
\end{eqnarray*}
Mỗi số hạng có dạng $(a_k-1)a_k(a_k+1)$ đều chia hết cho $6$ (là tích $3$ số nguyên liên tiếp), do đó $P-S$ chia hết cho $6$, mà $S$ chia hết cho $6$ nên $P$ chia hết cho $6$.
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên Trần Hưng Đạo, Bình Thuận, 2017]%[9D5G1]
	Cho $x,y,z$ là các số thực dương thỏa mãn $x+y+z+xyz=4$. Chứng minh
	$$\left(1+xy+\dfrac{y}{z}\right) \left(1+yz+\dfrac{z}{x}\right) \left(1+zx+\dfrac{x}{y}\right)\geq 27. $$
Dấu ``$ = $'' xảy ra khi nào?
	\loigiai{
	Ta có\\
$\circ$ $\heva{
&1+xy+\dfrac{y}{z}  =  \dfrac{y+z+xyz}{z}=\dfrac{4-x}{z}\\
&1+yz+\dfrac{z}{x}  =  \dfrac{z+x+xyz}{x}=\dfrac{4-y}{x}\\
&1+zx+\dfrac{x}{y}  =  \dfrac{x+y+xyz}{y}=\dfrac{4-z}{y}}$\\ $\Rightarrow \left(1+xy+\dfrac{y}{z}\right) \left(1+yz+\dfrac{z}{x}\right) \left(1+zx+\dfrac{x}{y}\right)=\dfrac{(4-x)(4-y)(4-z)}{xyz}\quad(1).$
\\
$\circ$  $4=x+y+z+xyz\geq 4\sqrt[4]{x^2y^2z^2}=4\sqrt{xyz}\Leftrightarrow xyz\leq 1\quad(2)$.
\\$\circ$  Từ $(1)$ bất đẳng thức tương đương 
\begin{eqnarray*}
&&(4-x)(4-y)(4-z)\geq 27xyz 
\\&&\Leftrightarrow 64-xyz-16(x+y+z)+4(xy+yz+zx)\geq 27xyz
\\&&  \Leftrightarrow 16+(xy+yz+zx)\geq 7xyz+4(x+y+z)
\\&&\Leftrightarrow 16+(xy+yz+zx)\geq 3xyz+16
\\&&  \Leftrightarrow xy+yz+zx\geq  3xyz (*)
\end{eqnarray*}
Mặt khác, theo $(2)$ và bất đẳng thức AM$-$GM cho 3 số, ta có: $$xy+yz+zx\geq 3\sqrt[3]{x^2y^2z^2}\geq3\sqrt[3]{x^3y^3z^3}=3xyz.$$
Dấu ``$=$'' xảy ra khi và chỉ khi $x=y=z=1$.
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên Trần Hưng Đạo, Bình Thuận, 2017]%[9H3G7]%[9H3G3]
Cho tam giác $ABC$ nhọn ($AB<AC$) có $AD$ là đường cao, $H$ là trực tâm của tam giác $ ABC $. Tia $ BH $ cắt đường tròn đường kính $ AC $ tại $ E, F $ sao cho $BE<BF$, tia $ CH $ cắt đường tròn đường kính $ AB $ tại $ G,K $ sao cho $ CG<CK $. Đường tròn ngoại tiếp tam giác $EDG$ cắt $ BC $ tại điểm thứ hai là $ P $.
\begin{enumerate}
	\item Chứng minh $ A $ là tâm đường tròn ngoại tiếp tứ giác $KEGF$.
	\item Chứng minh ba điểm $P,E,K$ thẳng thàng.
	\item Chứng mình bốn điểm $K,D,P,F$ cùng thuộc một đường tròn.
\end{enumerate}
\loigiai{
	\begin{center}
	\begin{tikzpicture}[scale=1.3]
	\tkzInit[ymin=-1,ymax=5,xmin=-2.5,xmax=4.5]
	\tkzClip
	\tkzDefPoints{0/3/A,-1/0/B,3/0/C,0/0/D, 1.5/1.5/I, -.5/1.5/J}
	\tkzDrawSegments(A,B B,C C,A A,D)
	\tkzDefLine[orthogonal=through B](A,C)\tkzGetPoint{x}
	\tkzInterLL(B,x)(A,D)\tkzGetPoint{H}
	\tkzDrawCircle(I,A)	\tkzDrawCircle(J,A)
	\tkzInterLC(B,H)(I,A)\tkzGetPoints{E}{F}
	\tkzInterLC(C,H)(J,A)\tkzGetPoints{G}{K}
	\tkzCircumCenter(E,D,G)\tkzGetPoint{Z}
	\tkzInterLC(B,C)(Z,D)\tkzGetFirstPoint{P}
	\tkzInterLC(K,C)(I,A)\tkzGetSecondPoint{T}
	\tkzInterLC(B,F)(J,A)\tkzGetFirstPoint{Q}
	\tkzMarkRightAngles[size=.2](A,T,K F,Q,A)
	\tkzDrawCircle(Z,D)
	\tkzDrawCircle(A,G)
%	\tkzDrawCircle[circum](K,D,F)
	\tkzLabelPoints[above](A)
	\tkzLabelPoints[below](B,D,P,D,G,E,C)
	\tkzLabelPoints[above](A)
	\tkzLabelPoints[left](K)
	\tkzLabelPoints[below left](T)
	\tkzLabelPoints[right](F,H,Q)
	\tkzDrawSegments(B,F C,K K,E E,G G,F A,K A,F A,G A,E E,P G,P D,G K,D)
	\end{tikzpicture}
\end{center}
\begin{enumerate}
	\item Gọi $BH$ cắt đường tròn $(AB)$ tại $Q$ và $CH$ cắt đường tròn $(AC)$ tại $T$. Xét đường tròn $(AC)$ ta có $AF=AE$, đường tròn $(AB)$ ta có $AG=AK$ (tính chất đường kính dây cung). Mặt khác, $AK^2=AT.AB=AQ.AC=AF^2$ (hệ thức lượng tam giác vuông), nên suy ra $AF=AE=AG=AK$ hay $A$ là tâm đường tròn  ngoại tiếp tứ giác $KEGF$.
	\item Ta có $\widehat{PEG}=\widehat{PDG}=\widehat{BAG}=\dfrac{\widehat{KAG}}{2}\,(1)$ (do 2 tứ giác $EDPG$ và $ABDG$ nội tiếp).
	\\Mặt khác $\widehat{KEG}=\widehat{KEA}+\widehat{AEG}=180^{\circ}-\dfrac{\widehat{KAG}}{2}\,(2)$ (sử dụng tính chất của $2$ tam giác cân $AKE,AEG$).\\
	Từ $(1)$ và $(2)$ ta suy ra $\widehat{KEG}+\widehat{GEP}=180^{\circ}$ hay $3$ điểm $K,E,P$ thẳng hàng.
	\item Ta có $\widehat{KEG}+\widehat{KAB}=\widehat{KEG}+\dfrac{\widehat{KAG}}{2}=180^{\circ}$. Trong đường tròn tâm $(A)$ ngoại tiếp tứ giác $AEGF$, $\widehat{KEG}+\widehat{KFG}=180^{\circ}$. Suy ra $\widehat{KDB}=\widehat{KAB}=\widehat{KFG}=\widehat{KFP}$, hay tứ giác $KDPF$ nội tiếp.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Trần Hưng Đạo, Bình Thuận, 2017]%[9D5G5]
	 Trong ngày quốc tế thiếu nhi $ 1/6 $ vừa qua, có $ 97 $ em nhỏ đến từ $ 3 $ trường của một huyện miền núi được nhận mỗi em một món quà. Biết rằng chỉ có $ 4 $ loại quà được phát và nếu trong $ 5 $ em nhỏ bất kỳ đến từ cùng một trường, nhận cùng một loại quà thì có $ 2 $ em cùng tuổi. Chứng minh rằng, luôn có $ 3 $ em nhỏ đến từ cùng một trường, cùng tuổi và nhận cùng một loại quà.
\loigiai{
Do có $97$ học sinh đến từ $ 3 $ trường, nên luôn có ít nhất một trường có tối thiểu là $\left[\dfrac{97}{3}\right]+1=33$ học sinh. Chia quà cho $33$ học sinh thì ít nhất có một nhóm $ \left[\dfrac{33}{4}\right] +1=9$ học sinh được cùng loại quà.
\\
$\circ$ Giả sử, trong nhóm $9$ học sinh này, không có $ 3 $ học sinh nào cùng tuổi. Khi đó, theo giả thiết, ta chọn $ 5 $ học sinh ra và lấy được hai học sinh $(A,B)$ cùng tuổi, loại ra; ta chọn tiếp $ 5 $ học sinh và được hai học sinh $(C,D)$ cùng tuổi, loại ra; ta được trong $ 5 $ học sinh còn lại có hai học sinh $(E,F)$ cùng tuổi. 
\\
$\circ$ Phân học sinh thành 2 cặp $(A,C,E)$ và $(B,D,F)$ và 3 học sinh còn lại đặt là $(X,Y,Z)$, chú ý rằng, mỗi bộ $ 3 $ đều không có học sinh nào cùng tuổi nhau.
\\
$\circ$ Chọn $ 2 $ nhóm như sau: $(A,C,E,X,Y)$ và $(A,C,E,X,Z)$, khi đó, chọn được ra ít nhất $ 2 $ cặp cùng tuổi trong mỗi nhóm, dù cặp nào xảy ra cùng tuổi, ta đều ghép với các cặp ban đầu để được một bộ $ 3 $ cùng tuổi. Khi đó mâu thuẫn với điều giả sử.
\\Tóm lại, luôn có $ 3 $ em nhỏ đến từ cùng một trường, cùng tuổi và nhận cùng một loại quà. 
}
\end{ex}
