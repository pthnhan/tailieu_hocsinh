
\section{Toán chuyển động}
\subsection{Trọng tâm kiến thức}
\begin{enumerate}
	\item Quãng đường $=$ Vận tốc $\times$ Thời gian.\\
	Vận tốc $=$ Quãng đường $:$ Thời gian.\\
	Thời gian $=$ Quãng đường $:$ Vận tốc.
	\item Cùng đi một quãng đường thì thời gian đi tỉ lệ nghịch với vận tốc đi.
\end{enumerate}
\subsection{Các dạng bài tập và phương pháp giải}
\begin{dang}{Chuyển động cùng chiều}
	Trong chuyển động cùng chiều: Thời gian đuổi kịp $=\dfrac{\text{ Khoảng cách }}{\text{Hiệu vận tốc}}$.\\
	Trên đường kẻ chậm với người mau\\
	Chuyển động \textit{cùng chiều} đuổi kịp nhau\\
	Vận tốc đôi bên tìm \textit{hiệu số}\\
	Đường dài chia với, khó chi đâu !
\end{dang}
\begin{vd}%[8D3Y2]
Lúc $8$ giờ, người thứ nhất đi từ $A$ và đến $B$ lúc $12$ giờ. Lúc $8$ giờ $30$ phút, người thứ hai đi từ $A$ và đến $B$ lúc $11$ giờ $30$ phút. Hỏi người thứ hai đuổi kịp người thứ nhất lúc mấy giờ ?
\loigiai{
Lấy quãng đường $AB$ làm đơn vị\\
Người thứ nhất đi $AB$ trong $12-8=4$ (giờ).\\
Trong một giờ, người thứ nhất đi được $\dfrac{1}{4}$ quãng đường.\\
Người thứ hai đi $AB$ trong $11$ giờ $30$ phút $-$ $8$ giờ $30$ phút $=$ $3$ giờ.\\
Trong một giờ, người thứ hai đi được $\dfrac{1}{3}$ quãng đường.\\
Lúc người thứ hai khởi hành, người thứ nhất đã đi được 
\[8 \text{ giờ } 30 \text{ phút }-~8 \text{ giờ }=30 \text{ phút } =\dfrac{1}{2} \text{ giờ}.\]
Người thứ nhất đi trước người thứ hai $\dfrac{1}{4}\cdot\dfrac{1}{2}=\dfrac{1}{8}$ (quãng đường).\\
Trong một giờ, người thứ hai đi nhiều hơn người thứ nhất là $\dfrac{1}{3} - \dfrac{1}{4}=\dfrac{1}{12}$ (quãng đường).\\
Thời gian để người thứ hai đuổi kịp người thứ nhất là $\dfrac{1}{8}:\dfrac{1}{12}=\dfrac{3}{2}$ (giờ).\\
Lúc đó là $8$ giờ $30$ phút $+\dfrac{3}{2}$ giờ $=10$ giờ.
\begin{luuy}
Trong bài toán trên, ta không biết cụ thể vận tốc của hai người, nhưng nếu lấy quãng đường $AB$ làm đơn vị thì vận tốc của hai người là $\dfrac{1}{4}$ và $\dfrac{1}{3}$, còn khoảng cách giữa hai người (lúc người thứ hai khởi hành) là $\dfrac{1}{8}$.\\
Phép chia $\dfrac{1}{8}:\left(\dfrac{1}{3} - \dfrac{1}{4}\right)$ cho ta thời gian để người thứ hai đuổi kịp người thứ nhất.
\end{luuy}
}		
\end{vd}
\begin{dang}{Chuyển động ngược chiều}
Trong chuyển động ngược chiều: Thời gian gặp nhau $=\dfrac{\text{Khoảng cách}}{\text{Tổng vận tốc}}$.\\
Trên đường kẻ chậm người mau\\
Chuyển động ngược chiều muốn gặp nhau\\
Vận tốc đôi bên tìm tổng số\\
Đường dài chia với, khó chi đâu !
\end{dang}
\begin{vd}
Trên quãng đường $AB$, hai xe cùng khởi hành một lúc, xe tải đi từ $A$ đến $B$ hết $6$ giờ, xe con đi từ $B$ đến $A$ hết $4$ giờ. Hai xe gặp nhau sau bao lâu?
	\loigiai{
Lấy quãng đường $AB$ làm đơn vị.\\
Trong một giờ, xe tải đi được $\dfrac{1}{6}$ quãng đường.\\
Trong một giờ, xe con đi được $\dfrac{1}{4}$ quãng đường.\\
Trong một giờ, hai xe gần nhau hơn được $\dfrac{1}{6} + \dfrac{1}{4}=\dfrac{5}{12}$ (quãng đường).\\
Thời gian để hai xe gặp nhau là $\quad 1 :\dfrac{5}{12}=\dfrac{12}{5}$ (giờ) $=2$ giờ $24$ phút.
	}
\end{vd}
\begin{dang}{Chuyển động có vận tốc thay đổi }
Trong bài toán chuyển động có vận tốc thay đổi, thường dùng đến kiến thức sau\\
Cho $s_1=v_1t_1,~s_2=v_2t_2$. Nếu $s_1=s_2$ thì $v_1t_1=v_2t_2$ nên $\dfrac{t_1}{t_2}=\dfrac{v_2}{v_1}$, tức là\\
 Trên hai quãng đường bằng nhau thì thời gian đi tỉ lệ nghịch với vận tốc đi.
\end{dang}
\begin{vd}
Một người phải đi từ $A$ đến $B$ trong $5$ giờ. Lúc đầu, người đó đi với vận tốc $30$ km/h. Khi còn $75$ km nữa thì được nửa đường, người đó đi với vận tốc $45$ km/h để kịp $B$ đúng dự định. Tính quãng đường $AB$.
	\loigiai
	{
Gọi $C$ là địa điểm người đó thay đổi vận tốc, gọi $M$ là điểm chính giữa của $AB$ gọi $D$ là điểm trên $MB$ cách $M$ là $75$ km
\begin{center}
	\begin{tikzpicture}[>=stealth,scale=1, line join = round, line cap = round]
	\tkzDefPoints{0/0/A,1/0/C,4/0/M,7/0/D,8/0/B,2.5/0.5/E}
	\tkzDrawSegments(A,C C,M M,D D,B)
	\tkzDrawPoints[fill=black](A,C,M,D,B)
	\tkzLabelPoints[above](A,C,M,D,B) 
	\draw (2.5,-1) node at (2.5,1) {$75$km};
	\draw (5.5,-1) node at (5.5,1) {$75$km};
	\draw[-,dashed] (C) to[out=45,in=135] (M);
	\draw[-,dashed] (M) to[out=45,in=135] (D);
	\draw[->](0,-0.5)->(0.5,-0.5);		
	\draw(0,-1) node at (0,-1) {$30$km/h};
	\draw[->](1,-0.5)->(1.5,-0.5);		
	\draw(1.7,-1) node at (1.7,-1) {$45$km/h};
	\draw[->](7,-0.5)->(7.5,-0.5);		
	\draw(7.5,-1) node at (7.5,-1) {$45$km/h};
	\end{tikzpicture}
\end{center}
Thời gian đi đoạn $CD$ bằng $\dfrac{75\cdot 2}{45}=\dfrac{10}{3}$ (giờ).\\
Thời gian đi đoạn $AC$ và $DB$ là $\quad 5 - \dfrac{10}{3}=\dfrac{5}{3}$ (giờ).\\
Trên quãng đường $AC$ và $DB$ bằng nhau, thời gian đi tỉ lệ nghịch với vận tốc nên
\[\dfrac{t_{AC}}{t_{DB}}=\dfrac{v_{DB}}{v_{AC}}=\dfrac{45}{30}=\dfrac{3}{2} .\]
Thời gian đi $AC$ bằng $\dfrac{3}{5}$ thời gian đi $AC$ và $DB$ hay $\dfrac{5}{3}\cdot\dfrac{3}{5}= 1$ (giờ).\\
Quãng đường $AC$ dài $30\cdot 1=30$ (km).\\
Quãng đường $AB$ dài $(30+75)\cdot 2=210$ (km).
	}
\end{vd}
\begin{dang}{Chuyển động có dòng nước}
Trong bài toán chuyển động có dòng nước, cần chú ý đến:\\
Vận tốc xuôi $=$ Vận tốc của vật $+$ Vận tốc dòng nước,\\
Vận tốc ngược $=$ Vận tốc của vật $-$ Vận tốc dòng nước.\\
Do đó Vận tốc dòng nước $=\dfrac{\text{ Vận tốc xuôi $-$ Vận tốc ngược}}{2}$.
\end{dang}
\begin{vd}
Anh Hòa bơi xuôi dòng nước từ $A$ đến $B$ hết $6$ phút, còn bơi ngược từ $B$ về $A$ hết $10$ phút. Hỏi một cụm bèo trôi theo dòng nước từ $A$ đến $B$ trong bao lâu?
	\loigiai
	{
Trong một phút, anh Hòa bơi xuôi dòng được $\dfrac{1}{6}$ quãng sông.\\
Trong một phút, anh Hòa bơi ngược dòng được $\dfrac{1}{10}$ quãng sông.\\
Do vận tốc xuôi trừ vận tốc ngược bằng hai lần vận tốc dòng nước nên trong một phút, dòng nước trôi được $\quad \left(\dfrac{1}{6} - \dfrac{1}{10}\right) : 2 =\dfrac{1}{30}$ (quãng sông).\\
Thời gian để cụm bèo trôi từ $A$ đến $B$ là $\quad 1 :\dfrac{1}{30}= 30$ (phút).
	}
\end{vd}
\begin{dang}{Chuyển động của xe lửa có chiều dài đáng kể}
Nếu một xe lửa có chiều dài $a$ đi hết một vật có chiều dài $b$ (chẳng hạn một đường hầm, một chiếc cầu $\dots$) thì xe lửa phải đi một quãng đường bằng $a+b$.
\end{dang}
\begin{vd}
Một xe lửa chạy với vận tốc $45$ km/h. Xe lửa chui vào một đường hầm có chiều dài gấp $9$ lần chiều dài của xe lửa và cần $2$ phút để xe lửa vào và ra khỏi đường hầm. Tính chiều dài của xe lửa.
	\loigiai
	{
Trong $2$ phút xe lửa đi được $\dfrac{45\cdot 2}{60}= 1{,}5$ (km) $=150$ (m).\\
Trong $2$ phút, xe lửa đi đoạn đường dài bằng chiều dài đường hầm cộng với chiều dài xe lửa, tức là bằng
\[9+1=10\text{ lần chiều dài xe lửa.} \]
Chiều dài của xe lửa bằng $1500:10=150$ (m).
	}
\end{vd}
\begin{dang}{Vận tốc trung bình}
Nếu một vật đi quãng đường $s_1$ với vận tốc $v_1$ và thời gian $t_1$, rồi đi tiếp quãng đường $s_2$ với vận tốc $v_2$ và thời gian $t_2$ thì vận tốc trung bình $(v_{tb})$ trên cả quãng đường $s_1+s_2$ là $v_{tb}=\dfrac{s_1+s_2}{t_1+t_2}$.\\
Trong trường hợp $v_1=v_2$ thì $v_{tb}=\dfrac{v_1+v_2}{2}$, còn nói chung thì $v_{tb}\neq \dfrac{v_1+v_2}{2}$.
\end{dang}
\begin{vd}
Một ô tô đi nửa đầu của quãng đường $AB$ với vận tốc $30$ km/h và đi nửa sau với vận tốc $60$ km/h. Tính vận tốc trung bình của ô tô trên cả quãng đường $AB$.
	\loigiai
	{
Cứ đi $2$ km thì có $1$ km ô tô đi với vận tốc $30$ km/h và $1$ km ô tô đi với vận tốc $60$ km/h.\\
Đi $1$ km với vận tốc $30$ km/h hết $\dfrac{1}{30}$ giờ.\\
Đi $1$ km với vận tốc $60$ km/h hết $\dfrac{1}{60}$ giờ.\\
Cứ đi $2$ km thì hết $\dfrac{1}{30} + \dfrac{1}{60}=\dfrac{1}{20}$ (giờ).\\
Vận tốc trung bình của ô tô là $2 :\dfrac{1}{20}= 40$ (km/h).
	}
\end{vd}
\subsection{Bài tập tự luyện}
\begin{bt}
	Một chiếc tàu chạy trên sông khi còn 90 $\mathrm{km}$ nữa mới cập bến thì tàu bị thủng, cứ $5$ phút có $2$ tấn nước tràn vào tàu. Nếu có $105$ tấn nước tràn vào tàu thì tàu sẽ bị chìm. Trên tàu có một máy bơm, mỗi giờ bơm ra được $10$ tấn nước. Tàu phải chạy ít nhất với vận tốc nào để khi cập bến, tàu vẫn chưa bị chìm?
	\loigiai{
		Lượng nước tràn vào tàu trong một giờ khi đã có máy bơm nước ra là $$\dfrac{2\cdot 60}{5}-10=14 \,\text{ (tấn).}$$
		Thời gian để tàu bị chìm là $105:14=7{,}5$ (giờ).\\
		Để khi cập bến mà tàu vẫn chưa bị chìm thì vận tốc ít nhất của tàu phải là $90 : 7{,}5 =12$ ($\mathrm{km}/\mathrm{h}$).
	}
\end{bt}

\begin{bt}
	Hiện nay là $4$ giờ. Sau ít nhất bao lâu thì kim phút chập với kim giờ?
	\loigiai{
		Hiện nay kim phút cách kim giờ $\dfrac{1}{3}$ vòng.\\
		Mỗi giờ kim phút quay được $1$ vòng, kim giờ quay được $\dfrac{1}{12}$ vòng, kim phút quay nhanh hơn kim giờ $$1-\dfrac{1}{12}=\dfrac{11}{12}\,\text{ (vòng).}$$
		Thời gian để kim phút đuổi kịp kim giờ là $\dfrac{1}{3}:\dfrac{11}{12}=\dfrac{4}{11}$ (giờ).
	}
\end{bt}
\begin{bt}
	Trên quãng đường $AB$, hai ô tô khởi hành cùng một lúc, xe thứ nhất đi từ $A$ đến $B$, hết $2$ giờ, xe thứ hai đi từ $B$ đến $A$ hết $3$ giờ. Đến chỗ gặp nhau, quãng đường xe thứ nhất đã đi nhiều hơn quãng đường xe thứ hai đã đi là $30$ km. Tính quãng đường $AB$.
	\loigiai{
		Trong một giờ, xe thứ nhất đi được $\dfrac{1}{2}$ quãng đường $AB$, xe thứ hai đi được $\dfrac{1}{3}$ quãng đường $AB$.\\
		Thời gian để hai xe gặp nhau là $1: \left(\dfrac{1}{2}+\dfrac{1}{3}\right)=\dfrac{6}{5}$ giờ.\\
		Đến chỗ gặp nhau, quãng đường xe thứ nhất đã đi nhiều hơn xe thứ hai là 
		$$\dfrac{6}{5}\cdot \left(\dfrac{1}{2}-\dfrac{1}{3}\right)=\dfrac{1}{5}\,\text{ quãng đường}\, AB\,\text{ hay}\, 30 \,\mathrm{km}.$$
		Quãng đường $AB$ dài $30 : \dfrac{1}{5} =150$ (km).	
	}
\end{bt}
\begin{bt}
	Trên quãng đường $AB$ dài $300$ km, người thứ nhất đi từ $A$ đến $B$, người thứ hai đi từ $B$ đến $A$. Họ khởi hành cùng một lúc thì sẽ gặp nhau sau $5$ giờ. Nhưng người thứ hai có việc bận, đã khởi hành sau người thứ nhất $40$ phút, do đó sau $4$ giờ $42$ phút mới gặp người thứ nhất. Tính vận tốc mỗi người.
	\loigiai{
		Tổng vận tốc của hai người là $300: 5=60$ (km$/$h).\\
		Thời gian cả hai người cùng đi là $4$ giờ $42$ phút $=4{,}7$ giờ.\\
		Trong thời gian đó, hai người cùng đi được $$60\cdot 4{,}7=282 \,\text{km}.$$
		Quãng đường người thứ nhất đi trong $40$ phút $=\dfrac{2}{3}$ giờ là
		$$300-282=18\,\text{km}.$$
		Vận tốc của người thứ nhất là $18:\dfrac{2}{3}=27$ km$/$h.\\
		Vận tốc của người thứ hai là $60-27=33$ km$/$h.
	}
\end{bt}

\begin{bt}
	Hai xe khởi hành từ $A$ và từ $B$ cùng một lúc, đi ngược chiều nhau và sẽ gặp nhau sau $6$ giờ. Vận tốc của xe con bằng $\dfrac{4}{3}$ vận tốc của xe tải. Muốn gặp nhau ở chính giữa quãng đường $AB$ thì xe con phải đi sau xe tải bao lâu?
	\loigiai{
		Vận tốc xe con bằng $\dfrac{4}{3}$ vận tốc xe tải nên trong $6$ giờ cùng đi, xe con đi được $\dfrac{4}{7}$ quãng đường $AB$, xe tải đi được $\dfrac{3}{7}$ quãng đường $AB$.\\
		Xe con đi quãng đường $AB$ trong $6 : \dfrac{4}{7}=\dfrac{21}{2}$ (giờ).\\
		Xe con đi nửa quãng đường $AB$ trong $\dfrac{21}{2}:2=\dfrac{21}{4}$ (giờ).\\
		Xe tải đi quãng đường $AB$ trong $6:\dfrac{3}{7}=14$ (giờ).\\
		Xe tải đi nửa quãng đường $AB$ trong $14:2=7$ (giờ).\\
		Để gặp nhau ở chính giữa đường thì xe con phải khởi hành sau xe tải $7-\dfrac{21}{4}=1\dfrac{3}{4}$ giờ $=1$ giờ $45$ phút.	
	}
\end{bt}

\begin{bt}
	Một người đi từ $A$ đến $B$. Người đó tính rằng nếu đi với vận tốc $40$ km$/$h thì đến $B$ sau giờ hẹn là $20$ phút, còn đi với vận tốc $60$ km$/$h thì đến $B$ trước giờ hẹn là $10$ phút. Tính quãng đường $AB$.
	\loigiai{
		Cùng đi quãng đường $AB$ thì thời gian đi tỉ lệ nghịch với vận tốc nên thời gian đi với vận tốc nhỏ bằng $\dfrac{60}{40}=\dfrac{3}{2}$ thời gian đi với vận tốc lớn.\\
		Hiệu của hai thời gian trên bằng $20+10=30$ (phút) $=0{,}5$ (giờ).\\
		Từ đó tìm được thời gian đi với vận tốc lớn là $1$ giờ, quãng đường $AB$ dài $60$ km.
	}
\end{bt}

\begin{bt}
	Ba người cùng khởi hành một lúc từ $A$ đến $B$, vận tốc người $I$ là $40$ km$/$h, vận tốc của người $II$ là $60$ km$/$h. Người $III$ đến $B$ trước người $I$ là $18$ phút và sau người $II$ là $12$ phút. Tính quãng đường $AB$ và vận tốc của người $III$.
	\loigiai{ Quãng đường $AB$ dài $60$ km, vận tốc người $III$ là $50$ km$/$h.	
	}
\end{bt}

\begin{bt}
	Hai người cùng đi từ $A$ về một phía, người $I$ khởi hành lúc $7$ giờ, người $II$ khởi hành lúc $7$ giờ $30$ phút. Hai người gặp nhau lúc mấy giờ, biết rằng quãng đường người $I$ đi trong $30$ phút bằng quãng đường người $II$ đi trong $20$ phút.
	\loigiai{
		Gọi $B$ là địa điểm hai người gặp nhau, $t_1, t_2$ là thời gian người $I$ và $II$ đi $AB$.\\
		Ta có $\dfrac{t_1}{t_2}=\dfrac{30}{20}=\dfrac{3}{2}$.\\
		và $t_1-t_2=0{,}5$ (giờ).\\
		Từ đó tính được $t_2=1$ (giờ), hai người gặp nhau lúc $8$ giờ $30$ phút.	
	}
\end{bt}

\begin{bt}

\begin{bt}
	Anh Thắng đi xe máy với vận tốc $36$ km$/$h. Anh gặp một xe lửa dài $75$ m đi cùng chiều chạy song song bên cạnh mình trong $15$ giây. Tính vận tốc của xe lửa với đưn vị m$/$giây.
	\loigiai{
		Trong $15$ giây, xe máy đi được $\dfrac{36\cdot 15}{3600}=0{,}15$ (km)=150 (m).\\
		Trong $15$ giây, xe lửa đi được $75+150=225$ (m).\\
		Vận tốc của xe lửa là $225:15=15$ (m$/$giây).
		
	}
\end{bt}

\begin{bt}
	Một người đi từ $A$ đến $B$. Người đó đi $\dfrac{1}{3}$ quãng đường với vận tốc $20$ km$/$h rồi đi phần còn lại với vận tốc $10$ km$/$h. Hỏi trung bình trên cả quãng đường $AB$, người đó đi với vận tốc bao nhiêu?
	\loigiai{
		Cứ $3$ km thì có $1$ km đi với vận tốc $20$ km$/$h hết $\dfrac{1}{20}$ giờ và $2$ km đi với vận tốc $10$ km$/$h hết $\dfrac{2}{10}$ giờ.\\
		Cứ $3$ km thì đi hết $\dfrac{1}{20}+\dfrac{2}{10}=\dfrac{1}{4}$ (giờ).\\
		Vận tóc trung bình là $3:\dfrac{1}{4}=12$ (km$/$h).
		
	}
\end{bt}

\begin{bt}
	Lúc $6$ giờ một xe tải và một xe máy cùng xuất phát từ $A$ để đến $B$. Vận tốc xe tải là $50$ km$/$h, vận tóc của xe máy là $30$ km$/$h. Sau đó $2$ giờ, một xe con cũng đi từ $A$ đến $B$, vận tốc $60$ km$/$h. Đến mấy giờ thì xe con ở chính giữa xe máy và xe tải?	
	\loigiai{
		Lúc $8$ giờ, xe máy ở vị trí $C$, xe tải ở vị trí $D$ (Hình dưới). Gọi $M$ là trung điểm của $CD$.
		\begin{center}
			\begin{tikzpicture}[scale=1, line join=round, line cap=round,>=stealth]
			\tkzDefPoints{0/1/A',5/1/D'}
			\tkzDefPoints{0/0/A,10/0/B,3/0/C,5/0/D,4/0/M}
			\tkzDrawPoints[size=5,fill=black](A,B,C,D,M)
			\tkzDrawSegments(A,B)
			\tkzLabelPoints[above](B,D,M)
			\tkzLabelPoints[above left](A)
			\tkzLabelPoints[above right](C)
			\draw [<->][dashed] (A') -- (D');
			\tkzLabelSegment(A',D'){$100 \text{km}$}
			\draw[snake=brace,raise snake=5pt][dashed] (A)--(C);
			\draw(1.5,0.2) node [above]	{$60 \text{km}$};
			\draw(0,0) node [below]{$\vec{60}$};
			\draw(3,0) node [below]{$\vec{30}$};
			\draw(4,0) node [below]{$\vec{40}$};
			\draw(5,0) node [below]{$\vec{50}$};
			\end{tikzpicture}
		\end{center}
		Quãng đường $AC$ dài $30\cdot 2 =60$ (km).\\
		Quãng đường $AD$ dài $50\cdot2=100$ (km).\\
		Quãng đường $AM$ dài $60+(100-60):2=80$ (km).\\
		Giả sử có một xe thứ tự khởi hành từ $M$ lúc $8$ giờ, đi về phía $B$ với vận tốc $\dfrac{30+50}{2}=40$ (km$/$h) thì xe thứ tư luôn ở chính giữa xe máy và xe tải.\\
		Thời gian để xe con đuổi kịp xe thứ tư là $\dfrac{80}{60-40}=4$ (giờ).\\
		Xe con gặp xe thứ tư lúc $8+4=12$ (giờ).\\
		Vậy lúc $12$ giờ thì xe con ở chính giữa xe máy và xe tải.
	}
\end{bt}

\begin{bt}
	Hai xe buýt cùng khởi hành một lúc với vận tốc không đổi, xe thứ nhất đi từ $A$ đến $B$, xe thứ hai đi từ $B$ đến $A$. Xe thứ nhất đến $B$ thì quay lại ngay, xe thứ hai đến $A$ thì quay lại ngay. Hai xe gặp nhau lần thứ nhất tại $C$ cách $A$ là $5$ km và gặp nhau lần thứ hai tại $D$ cách $B$ là $4$ km. Tính quãng đường $AB$.
	\loigiai{
		Trên hình (dưới đây), quãng đường xe $I$ đi biểu thị bởi nét liền, quãng đường xe $II$ đi biểu diễn bởi nét đứt.

			\begin{tikzpicture}[scale=1, line join=round, line cap=round,>=stealth]

			\tkzDefPoints{0/-3/A_2,6/-3/D_2,10/-3/B_2}
			\tkzDrawPoints[size=5,fill=black](A,B,C,D)
			\tkzDrawSegments(A,B A',C' C_1,B_1 D_2,B_2)

			\tkzLabelPoints[above right](C,B)
			
			\draw [<->][dashed] (C') -- (B');
			\draw [<->][dashed] (A_2) -- (D_2);

			\draw[snake=brace,raise snake=5pt][dashed] (A)--(C);
			\draw(2.5,0.2) node [above]	{$5$};
			\draw[snake=brace,raise snake=5pt][dashed] (D)--(B);

	}
\end{bt}
