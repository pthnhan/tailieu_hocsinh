\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, tỉnh Khánh Hòa, năm 2018]%[Dương Xuân Lợi, Dự án EX-TS10C-18]%[9D0K2]
	Giải phương trình $x^2+2x+2=3x\sqrt{x+1}$.
	\loigiai{
		Điều kiện $x\ge-1$, ta có 
		$$x^2+2x+2=3x\sqrt{x+1}\Leftrightarrow x^2-3x\sqrt{x+1}+2(x+1)=0.$$
		Đặt $\heva{&u=x\\&v=\sqrt{x+1}\ge 0}$, phương trình đã cho trở thành
		$$u^2-3uv+2v^2=0\Leftrightarrow (u-v)(u-2v)=0\Leftrightarrow\hoac{&u-v=0\\&u-2v=0}\Leftrightarrow\hoac{&u=v\\&u=2v.}$$
		\begin{itemize}
			\item Với $u=v$ suy ra $\sqrt{x+1}=x\Leftrightarrow\heva{&x\ge 0\\&x+1=x^2}\Leftrightarrow x=\dfrac{1+\sqrt{5}}{2}$.
			\item Với $u=2v$ suy ra $2\sqrt{x+1}=x\Leftrightarrow\heva{&x\ge 0\\&4(x+1)=x^2}\Leftrightarrow x=2+2\sqrt{2}$.
		\end{itemize}
		Kết hợp với điều kiện ta có nghiệm của phương trình đã cho là $x=\dfrac{1+\sqrt{5}}{2}$, $x=2+2\sqrt{2}$.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, tỉnh Khánh Hòa, năm 2018]%[Dương Xuân Lợi, Dự án EX-TS10C-18]%[9D0G8]
	Có bao nhiêu số tự nhiên có ba chữ số $\overline{abc}$ sao cho $a$, $b$, $c$ là ba cạnh của một tam giác cân?
	\loigiai{
		Ta có số cần lập chỉ một trong hai trường hợp sau:
		\begin{itemize}
			\item Trường hợp $1$. $a$, $b$, $c$ là ba cạnh tam giác đều $a=b=c>0$, có $9$ số lập được.
			\item Trường hợp $2$. $a$, $b$, $c$ là ba cạnh tam giác cân (không đều).\\
			Xét $a=b \ne c$. Vì $a+b>c$ nên\\
			Với bộ số $(2,2,c)$ có $2$ cách chọn $c$, lập được $2$ số.\\
			Với bộ số $(3,3,c)$ có $4$ cách chọn $c$, lập được $4$ số.\\
			Với bộ số $(4,4,c)$ có $6$ cách chọn $c$, lập được $6$ số.\\
			Bắt đầu từ $a=b\ge 5$ trở đi thì $a+b\ge 10$ thì dù chọn $c$ là số bất kỳ từ $1$ đến $9$ (không tính trường hợp trùng với $a$, $b$) thì ta đều có $a+b>c$. Chọn $a$, $b$ có $5$ cách chọn, chọn $c$ có $8$ cách chọn. Nên có $8\cdot 5=40$ cách chọn, lập được $40$ số.\\
			Vì vai trò $a$, $b$, $c$ như nhau nên ta có $(2+4+6+40)\cdot 3=156$.\\
			Vậy có $9+156=165$ số cần tìm.
		\end{itemize}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, tỉnh Khánh Hòa, năm 2018]%[Dương Xuân Lợi, Dự án EX-TS10C-18]%[9D0B1]%[9D0G1]
	\begin{listEX}
		\item Chứng minh rằng với mọi số thực $a$, $b$, $c$ ta luôn có $(a+b+c)^2=a^2+b^2+c^2+2(ab+bc+ca)$.
		\item Cho ba số $x$, $y$, $z$ khác $0$ đồng thời thỏa mãn $x+y+z=\dfrac{1}{2}$, $\dfrac{1}{x^2}+\dfrac{1}{y^2}+\dfrac{1}{z^2}+\dfrac{1}{xyz}=4$ và $\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}>0$. Tính giá trị biểu thức $Q=\left(y^{2017}+z^{2017}\right)\left(z^{2019}+x^{2019}\right)\left(x^{2021}+y^{2021}\right)$.
	\end{listEX}
	\loigiai{
		\begin{listEX}
			\item Ta có 
			\begin{eqnarray*}
				VT&=&(a+b+c)^2=\left[(a+b)+c\right]^2\\
				&=&(a+b)^2+2(a+b)c+c^2=a^2+b^2+c^2+2(ab+bc+ca)\\
				&=&VP.
			\end{eqnarray*}
			\item Ta có 
			\begin{eqnarray*} 			4&=&\dfrac{1}{x^2}+\dfrac{1}{y^2}+\dfrac{1}{z^2}+\dfrac{1}{xyz}=\dfrac{1}{x^2}+\dfrac{1}{y^2}+\dfrac{1}{z^2}+\dfrac{2(x+y+z)}{xyz}\\
				&=&\dfrac{1}{x^2}+\dfrac{1}{y^2}+\dfrac{1}{z^2}+2\left(\dfrac{1}{xy}+\dfrac{1}{yz}+\dfrac{1}{zx}\right)= \left(\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}\right)^2.
			\end{eqnarray*}
			Mà $\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}>0 \Rightarrow \dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=2$.\hfill $(1)$\\
			Mặt khác $x+y+z=\dfrac{1}{2} \Rightarrow \dfrac{1}{x+y+z}=2$. \hfill $(2)$\\
			Từ $(1)$ và $(2)$ suy ra 
			\begin{eqnarray*}
				& &\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=\dfrac{1}{x+y+z}\Leftrightarrow (xy+yz+zx)(x+y+z)=xyz\\
				&\Leftrightarrow &(x+y)(y+z)(z+x)=0	\Leftrightarrow \hoac{&x=-y\\&y=-z\\&z=-x}\\
				&\Leftrightarrow &\hoac{&x^{2017}=-y^{2017}\\&y^{2019}=-z^{2019}\\&z^{2021}=-x^{2021}}\Rightarrow Q=0.
			\end{eqnarray*}
		\end{listEX}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, tỉnh Khánh Hòa, năm 2018]%[Dương Xuân Lợi, Dự án EX-TS10C-18]%[9H0B2]%[9H0K3]%[9H0G6]
	Cho đường tròn $(O)$ đường kính $BC$ và $H$ là một điểm trên đoạn thẳng $BO$ (điểm $H$ không trùng với hai điểm $B$ và $O$). Qua $H$ vẽ đường thẳng vuông góc với $BC$, cắt đường tròn $(O)$ tại $A$ và $D$. Gọi $M$ là giao điểm của $AC$ và $BD$, qua $M$ kẻ đường thẳng vuông góc với $BC$ tại $N$.
	\begin{listEX}
		\item Chứng minh rằng $MNBA$ là tứ giác nội tiếp.
		\item Tính giá trị của $P=2\left(\dfrac{BO}{AB}\right)^2-\dfrac{OH}{BH}$.
		\item Từ $B$ kẻ tiếp tuyến với đường tròn $(O)$, cắt hai đường thẳng $AC$ và $AN$ lần lượt tại $K$ và $E$. Chứng minh rằng đường thẳng $EC$ luôn đi qua trung điểm của đoạn thẳng $AH$ khi điểm $H$ di động trên đoạn thẳng $BO$.
	\end{listEX}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=0.9, line join=round, line cap=round]
			\tkzInit[ymin=-3,ymax=5,xmin=-6,xmax=4]
			\tkzClip
			\tkzDefPoints{-3/0/B,3/0/C,0/0/O,-1.7/0/H,-1.7/1/W,-3/1/X}
			\tkzDrawCircle[radius](O,C)
			\tkzInterLC(H,W)(O,C)\tkzGetPoints{D}{A}
			\tkzInterLL(A,C)(B,D)\tkzGetPoint{M}
			\tkzDefLine[orthogonal=through M](B,C)\tkzGetPoint{Y}
			\tkzInterLL(B,C)(M,Y)\tkzGetPoint{N}
			\tkzInterLL(A,C)(B,X)\tkzGetPoint{K}
			\tkzInterLL(A,N)(B,K)\tkzGetPoint{E}
			\tkzInterLL(A,D)(C,E)\tkzGetPoint{I}		
			\tkzDrawSegments(A,B A,D A,O M,C M,D M,N N,C A,N B,K C,E)
			\tkzDrawPoints[fill=black,size=4](A,B,C,D,O,H,M,N,K,E,I)
			\tkzLabelPoints[above](A,M,K)
			\tkzLabelPoints[below](O,D)
			\tkzLabelPoints[left](N,E)
			\tkzLabelPoints[right](C)
			\tkzLabelPoints[below left](B,I,H)
			\tkzMarkRightAngle[size=0.2](A,H,C)
			\tkzMarkRightAngle[size=0.2](M,N,C) 
			\tkzMarkRightAngle[size=0.2](K,B,N)
			\end{tikzpicture}
		\end{center}
		\begin{listEX}
			\item Ta có $\widehat{BAC}=90^{\circ}$ (góc nội tiếp chắn nửa đường tròn) suy ra $\widehat{BAM}=90^{\circ}$. Mặt khác $\widehat{MNB}=90^{\circ}$ suy ra tứ giác $MNBA$ có $\widehat{BAM}+\widehat{MNB}=90^{\circ}+90^{\circ}=180^{\circ}$. Vậy tứ giác $MNBA$ là tứ giác nội tiếp đường tròn đường kính $MB$.
			\item Ta có tam giác $ABC$ vuông tại $A$, nên $BH=\dfrac{AB^2}{BC}=\dfrac{AB^2}{2BO}$. Mặt khác $OH=BO-BH=BO-\dfrac{AB^2}{BC}=\dfrac{2BO^2-AB^2}{2BO}$ suy ra $\dfrac{OH}{BH}=\dfrac{2BO^2-AB^2}{AB^2}=2\left(\dfrac{BO}{AB}\right)^2-1$. Vậy $P=1$.
			\item Ta thấy $\widehat{MBN}=\widehat{DBC}$ (hai góc đối đỉnh); $\widehat{DBC}=\widehat{DAC}$ (tứ giác $DBAC$ nội tiếp) suy ra $\widehat{MBN}=\widehat{DAC}\Rightarrow \widehat{NMB}=\widehat{BCA}$. \hfill$(1)$
			
			Tứ giác MNBA nội tiếp nên ta có $\widehat{NMB}=\widehat{NAB}$.\hfill$(2)$
			
			Tam giác $OAC$ cân tại $O$ suy ra $\widehat{BCA}=\widehat{OAC}$.\hfill$(3)$
			
			Từ $(1)$, $(2)$ và $(3)$ suy ra $\widehat{NAB}=\widehat{OAC}\Rightarrow \widehat{OAC}+\widehat{BAO}=\widehat{NAB}+\widehat{BAO}\Leftrightarrow \widehat{BAC}=\widehat{NAO}$.\\
			Mà $\widehat{BAC}=90^{\circ}$ nên $\widehat{NAO}=90^{\circ}$, suy ra $NA$ là tiếp tuyến của $(O)$.\\
			Theo tính chất của tiếp tuyến ta có $EA=EB$ và $\widehat{EAB}=\widehat{EBA}$.\\
			Trong tam giác vuông $KAB$, ta có $\widehat{EAB}=\widehat{EBA}\Rightarrow \widehat{BKA}=\widehat{EAK}$ (phụ với hai góc bằng nhau). Suy ra tam giác $KAE$ cân tại $E \Rightarrow AE=KE\Rightarrow EB=KE$.\\
			Mặt khác, $AH$ song song BK (vì cùng vuông góc với $AB$).\\
			$AI\parallel KE\Rightarrow \dfrac{CI}{CE}=\dfrac{AI}{KE}$ (định lí Ta-lét).\\
			$HI\parallel EB \Rightarrow \dfrac{CI}{CE}=\dfrac{HI}{BE}$ (định lí Ta-lét) $\Rightarrow \dfrac{AI}{KE}=\dfrac{HI}{BE}$.\\
			Mà $KE=EB$, suy ra $AI=HI$ nên $I$ là trung điểm của đoạn thẳng $AH$.

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, tỉnh Khánh Hòa, năm 2018]%[Dương Xuân Lợi, Dự án EX-TS10C-18]%[9H0G7]
	Với $a,b,c$ là các số thực dương thỏa mãn điều kiện $a+b+c=abc$. Chứng minh rằng $\dfrac{\sqrt{1+a^2}}{a}+\dfrac{\sqrt{1+b^2}}{b}-\sqrt{1+c^2}<1$.
	\loigiai{
		Ta có $a+b+c=abc \Leftrightarrow \dfrac{1}{ab}+\dfrac{1}{bc}+\dfrac{1}{ca}=1$.\\
		Đặt $\dfrac{1}{a}=x$, $\dfrac{1}{b}=y$, $\dfrac{1}{c}=z$, khi đó $x,y,z>0$ và $xy+yz+zx=1$.
		\begin{eqnarray*}
			& &\dfrac{\sqrt{1+a^2}}{a}+\dfrac{\sqrt{1+b^2}}{b}-\sqrt{1+c^2}<1\Leftrightarrow  \sqrt{1+x^2}+\sqrt{1+y^2}-\dfrac{\sqrt{1+z^2}}{z}<1\\
			& \Leftrightarrow & \left(\sqrt{1+x^2}-1\right)\left(\sqrt{1+y^2}-1\right)-\sqrt{1+x^2}\sqrt{1+y^2}+\dfrac{\sqrt{1+z^2}}{z}>0\\
			& \Leftrightarrow & \left(\sqrt{1+x^2}-1\right)\left(\sqrt{1+y^2}-1\right)+\dfrac{\sqrt{1+z^2}-z\sqrt{1+x^2}\sqrt{1+y^2}}{z}>0.\qquad\hfill(*)
		\end{eqnarray*}	
		Ta lại có $\sqrt{1+x^2}\sqrt{1+y^2}=\sqrt{1+x^2+y^2+x^2y^2}=\sqrt{(1-xy)^2+(x+y)^2}=(x+y)\sqrt{1+z^2}$.\\
		Suy ra 	
		\begin{eqnarray*} 
			(*) &\Leftrightarrow &\left(\sqrt{1+x^2}-1\right)\left(\sqrt{1+y^2}-1\right)+\dfrac{\sqrt{1+z^2}-z(x+y)\sqrt{1+z^2}}{z}>0\\
			&\Leftrightarrow &\left(\sqrt{1+x^2}-1\right)\left(\sqrt{1+y^2}-1\right)+\dfrac{\sqrt{1+z^2}-(zx+yz)\sqrt{1+z^2}}{z}>0\\ 
			&\Leftrightarrow &\left(\sqrt{1+x^2}-1\right)\left(\sqrt{1+y^2}-1\right)+\dfrac{\sqrt{1+z^2}-(1-xy)\sqrt{1+z^2}}{z}>0\\
			&\Leftrightarrow &\left(\sqrt{1+x^2}-1\right)\left(\sqrt{1+y^2}-1\right)+\dfrac{xy\sqrt{1+z^2}}{z}>0, \forall x,y,z>0. 
		\end{eqnarray*}
		Vậy ta có điều phải chứng minh.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Lê Quý Đôn, tỉnh Khánh Hòa, năm 2018]%[Dương Xuân Lợi, Dự án EX-TS10C-18]%[9H0G8]
	Để tiết kiệm chi phí vận hành đồng thời đưa du khách tham quan hết $18$ danh lam thắng cảnh của tính K, Công ty Du lịch lữ hành KH đã thiết lập các tuyến một chiều như sau: nếu có tuyến đi từ $A$ đến $B$ và từ $B$ đến $C$ thì không có tuyến đi từ $A$ đến $C$. Hỏi có bao nhiêu cách thiết lập để đi hết $18$ địa điểm trên?
	\loigiai{
		Gọi $A$ là địa điểm có nhiều tuyến đường nhất (gồm cả đường xuất phát từ $A$ và đường đi đến $A$). Ta chia các địa điểm còn lại thành $3$ loại:\\
		Loại $1$: Các tuyến đường xuất phát từ $A$ có $n(1)=m$ tuyến đường.\\
		Loại $2$: Các tuyến đường đi đến $A$ có $n(2)=n$ tuyến đường.\\
		Loại $3$: Không có tuyến đi và đến từ $A$ có $n(3)=p$ tuyến đường.\\
		Ta có $m+n+p=17$.\\
		Số tuyến đường liên quan đến $A$ có $m+n$ tuyến.\\
		Số tuyến đường không liên quan đến $A$ không vượt quá $p(m+n)$.\\
		Số tuyến đường liên quan loại $1$ và $2$ không vượt quá $mn$.\\
		Vì $(a+b+c)^2\ge 3(ab+bc+ca) \Leftrightarrow (a-b)^2+(b-c)^2+(c-a)^2\ge 0$, với mọi số thực $a,b,c$.
		\\Do đó $ab+bc+ca\le \dfrac{(a+b+c)^2}{3}$.\\
		Gọi $S$ là tổng các tuyến đường một chiều qua $18$ địa điểm trên, ta có\\
		$$S\le m+n+p(m+n)+mn=mn+(p+1)m+n(p+1)\le \dfrac{(m+n+p+1)^2}{3}=108.$$
		Dấu bằng xảy ra khi và chỉ khi $m=n=p+1=\dfrac{18}{3}=6\Leftrightarrow \heva{&m=6\\&n=6\\&p=5.}$\\
		Vậy có thể thiết lập tối đa $108$ tuyến đường một chiều để đi qua hết $18$ địa điểm đã cho.
	}
\end{ex}
