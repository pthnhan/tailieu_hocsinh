\begin{bt}%[Đề HK1, Nguyễn Chí Thanh HCM, 2018]%[Trần Chiến, dự án HK1 2017-2018]%[1D1B3-1]
	Giải phương trình $2\cos^2 2x-3\sin 2x+3=0$.
	\loigiai{
	Ta có 
	\begin{eqnarray*}
	&& 2\cos^2 2x-3\sin 2x+3=0\\
	& \Leftrightarrow & 2(1-\sin^2 2x)-3\sin 2x+3=0\\
	& \Leftrightarrow & 2\sin^2 2x+3\sin 2x-5=0\\
	& \Leftrightarrow & \hoac{& \sin 2x=1 &&  \\ & \sin 2x=-\dfrac{5}{2} &&\text{(vô nghiệm)}}\\
	& \Leftrightarrow & 2x=\dfrac{\pi}{2}+k2\pi \ (k\in\mathbb{Z})\\
	& \Leftrightarrow & x=\dfrac{\pi}{4}+k\pi\ (k\in\mathbb{Z}).
	\end{eqnarray*}
	Vậy $x=\dfrac{\pi}{4}+k\pi\ (k\in\mathbb{Z})$.}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Chí Thanh HCM, 2018]%[Trần Chiến, dự án HK1 2017-2018]%[1D1B3-2]
	Giải phương trình $2\sin^2 x-\sqrt{3}\sin 2x+\cos^2 x=3$.
	\loigiai{
	Xét phương trình $2\sin^2 x-\sqrt{3}\sin 2x+\cos^2 x=3\Leftrightarrow 2\sin^2 x-2\sqrt{3}\sin x\cos x+\cos^2 x=3$, ta thấy $\cos x=0$ không thỏa mãn. Do đó, $\cos x\neq 0$. Chia hai vế của phương trình cho $\cos^2 x$ ta được
		\begin{eqnarray*}
		&& 2\tan^2 x-2\sqrt{3}\tan x+1=3(1+\tan^2 x)\\
		& \Leftrightarrow  & \tan^2 x+2\sqrt{3}\tan x+2=0\\
		& \Leftrightarrow & \hoac{& \tan x=1-\sqrt{3} \\ & \tan x=-1-\sqrt{3}}\\
		& \Leftrightarrow & \hoac{& x=\arctan(1-\sqrt{3})+k\pi\\ & x=\arctan(-1-\sqrt{3})+k\pi}\ (k\in\mathbb{Z}).
		\end{eqnarray*}
	Vậy $x=\arctan(1-\sqrt{3})+k\pi$, $x=\arctan(-1-\sqrt{3})+k\pi\ (k\in\mathbb{Z})$.
	}
\end{bt}
	
\begin{bt}%[Đề HK1, Nguyễn Chí Thanh HCM, 2018]%[Trần Chiến, dự án HK1 2017-2018]%[1D2B3-2]
	Tìm số hạng chứa $x^{10}$ trong khai triển $\left(3x-\dfrac{2}{x^2}\right)^{25}$ với $x\neq 0$.
	\loigiai{
	Ta có $\left(3x-\dfrac{2}{x^2}\right)^{25}=\displaystyle\sum_{k=0}^{25}\mathrm{C}^k_{25}(3x)^{25-k}\left(-\dfrac{2}{x^2}\right)^k= \displaystyle\sum_{k=0}^{25}\mathrm{C}^k_{25} 3^{25-k} (-2)^k x^{25-3k}$.\\
	Theo giả thiết ta có $25-3k=10\Leftrightarrow k=5$.\\
	Vậy số hạng cần tìm là $\mathrm{C}^5_{25}3^{20} (-2)^5 x^{10}$.
	}
\end{bt}
	
\begin{bt}%[Đề HK1, Nguyễn Chí Thanh HCM, 2018]%[Trần Chiến, dự án HK1 2017-2018]%[1D2K5-4]
	Trong đợt cứu trợ lũ lụt ở miền Trung vừa qua, người ta chở một lô hàng có $20$ sản phẩm, trong đó có $3$ sản phẩm nước giải khát. Lấy tùy ý $6$ sản phẩm từ lô hàng đó, tính xác suất để trong các sản phẩm lấy ra, có không quá $2$ sản phẩm nước giải khát.
	\loigiai{
	Lấy tùy ý $6$ sản phẩm từ $20$ sản phẩm có $\mathrm{C}^6_{20}$ cách.\\
	Số cách chọn được $3$ sản phẩm nước giải khát và $3$ sản phẩm không phải nước giải khát là $\mathrm{C}^3_3\cdot\mathrm{C}^3_{17}$.\\
	Do đó, xác suất chọn được đúng $3$ sản phẩm nước giải khát là $\dfrac{\mathrm{C}^3_3\cdot\mathrm{C}^3_{17}}{\mathrm{C}^6_{20}}=\dfrac{1}{57}$.\\
	Vậy xác suất lấy được không quá $2$ sản phẩm nước giải khát bằng $1-\dfrac{1}{57}=\dfrac{56}{57}$.
	}
\end{bt}
	
\begin{bt}%[Đề HK1, Nguyễn Chí Thanh HCM, 2018]%[Trần Chiến, dự án HK1 2017-2018]%[1D3B1-1]
	Chứng minh rằng với mọi số nguyên dương $n$, ta luôn có đẳng thức $$1\cdot 2+ 2\cdot 5+3\cdot 8+\dots+n(3n-1)=n^2(n+1).$$
	\loigiai{
	Với $n=1$ thì đẳng thức hiển nhiên đúng.\\
	Giả sử đẳng thức đúng với $n=k\ (k\geq 1)$, tức $1\cdot 2+ 2\cdot 51+\dots+k(3k-1)=k^2(k+1).$\\
	Ta chứng minh nó cũng đúng với $n=k+1$. Thật vậy,
	\begin{eqnarray*}
	1\cdot 2+ 2\cdot 5+\dots+k(3k-1)+ (k+1)[3(k+1)-1] &=& k^2(k+1)+(k+1)[3(k+1)-1]\\
	&=& (k+1)(k^2+3k+2)\\
	&=& (k+1)^2(k+2)\\
	&=& (k+1)^2[(k+1)+1].
	\end{eqnarray*}
	Vậy đẳng thức đúng với $n=k+1$ do đó đúng với mọi $n\in\mathbb{N}^*$.
	}
\end{bt}
	
\begin{bt}%[Đề HK1, Nguyễn Chí Thanh HCM, 2018]%[Trần Chiến, dự án HK1 2017-2018]%[1D3B4-3]
	Tìm số hạng đầu và công bội $q$ của cấp số nhân $(u_n)$ biết $\heva{& u_1-u_2+u_3=-\dfrac{3}{2}\\ & u_2-u_3+u_4=-\dfrac{3}{4}}$.
	\loigiai{
	Ta có $\heva{& u_1-u_2+u_3=-\dfrac{3}{2}\\ & u_2-u_3+u_4=-\dfrac{3}{4}}\Leftrightarrow \heva{& u_1-u_1q+u_1q^2=-\dfrac{3}{2}\\ & u_1q-u_1q^2+u_1q^3=-\dfrac{3}{4}}\Leftrightarrow \heva{& u_1(1-q+q^2)=-\dfrac{3}{2}\\ & u_1q(1-q+q^2)=-\dfrac{3}{4}}$.\\
	Ta thấy $1-q+q^2>0$ và $q\neq 0$ (vì $q=0$ không thỏa mãn hệ phương trình) nên chia hai phương trình cho nhau ta được $q=\dfrac{1}{2}$. Từ đó ta tính được $u_1=-2$.
	}
\end{bt}
