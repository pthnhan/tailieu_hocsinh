\subsection{TÓM TẮT KIẾN THỨC}
\begin{tomtat}
Muốn trừ số nguyên $a$ cho số nguyên $b$, ta cộng $a$ với số đối của $b$.
\begin{center}
$a-b = a+ (-b)$
\end{center}
\begin{vd}
$7-15 = 7+ (-15) = -8$.\\
$(-5) - (-9) = (-5)+9 =4$.
\end{vd}
\end{tomtat}
