\begin{ex}%[Đề KSCL HK1 Toán 11, B Bình Lục, Hà Nam, 2018]%[1D1K3]
Biến đổi phương trình $\sqrt{3}\sin x-\cos x=1$ về phương trình $\sin\left(x-\dfrac{a\pi}{b}\right)=\sin\dfrac{c\pi}{d}$, với $a, b, c, d$ là các số nguyên dương thỏa mãn $c\leq a<b$ và các phân số $\dfrac{a}{b}, \dfrac{c}{d}$ tối giản. Tính tổng $S=a+b+c+d$.
	\choice
	{$S=6$}
	{$S=10$}
	{\True $S=14$}
	{$S=7$}
\loigiai{Ta có
$\sqrt{3}\sin x-\cos x=1\Leftrightarrow \dfrac{\sqrt{3}}{2}\sin x-\dfrac{1}{2}\cos x=\dfrac{1}{2}\Leftrightarrow\sin\left(x+\alpha\right)=\sin\dfrac{c\pi}{d}$\\

với $\heva{&\cos\alpha=\dfrac{\sqrt{3}}{2}\\&\sin\alpha=-\dfrac{1}{2}}$ và $\sin\dfrac{c\pi}{d}=\dfrac{1}{2}$.\\
Ta có $\heva{&\cos\alpha=\dfrac{\sqrt{3}}{2}\\&\sin\alpha=-\dfrac{1}{2}}\Leftrightarrow \alpha=-\dfrac{\pi}{6}+k2\pi=-\dfrac{(1-12k)\pi}{6}, k\in\mathbb{Z}$.\\
Vì $0<a<b$ nên $0<1-12k<6\Leftrightarrow\heva{&k<\dfrac{1}{12}\\&k>-\dfrac{5}{12}}\Leftrightarrow k=0\Rightarrow\dfrac{a}{b}=\dfrac{1}{6}$.
Vì $\dfrac{a}{b}$ tối giản nên $a=1, b=6$.\\
Ta có $0<c\leq a\Rightarrow c=1$. Khi đó $\sin\dfrac{\pi}{d}=\dfrac{1}{2}\Leftrightarrow\hoac{&\dfrac{\pi}{d}=\dfrac{\pi}{6}+k2\pi\\&\dfrac{\pi}{d}=\dfrac{5\pi}{6}+k2\pi}
\Leftrightarrow \hoac{&d=\dfrac{6}{1+12k}\quad (1)\\&d=\dfrac{6}{5+12k}\quad (2)}$.\\
Vì $k$ nguyên, $d$ nguyên dương nên loại $(2)$, từ $(1)\Rightarrow k=0, d=6$.\\
Vậy $a=1, b=6, c=1, d=6\Rightarrow S=a+b+c+d=14$.
}
\end{ex}

\begin{bt}%[Đề KSCL HK1 Toán 11, B Bình Lục, Hà Nam, 2018]%[1D1Y2]%[1D1B3]%[1D1G3]
	Giải các phương trình sau
		\begin{enumerate}
			\item $\sin x=\sin\dfrac{\pi}{5}$.
			\item $2\cos^2x+3\cos x+1=0$.
            \item $\dfrac{\sin2x\cos2x+4\sin x\cos^2x-3\sin2x+\cos2x-2\cos x+1}{\cos2x-1}=2$.
		\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $\sin x=\sin\dfrac{\pi}{5}\Leftrightarrow\hoac{&x=\dfrac{\pi}{5}+k2\pi\\&x=\pi-\dfrac{\pi}{5}+k2\pi}
(k\in\mathbb{Z})\Leftrightarrow\hoac{&x=\dfrac{\pi}{5}+k2\pi\\&x=\dfrac{4\pi}{5}+k2\pi} (k\in\mathbb{Z}).$\\
Vậy $S=\left\{\dfrac{\pi}{5}+k2\pi; \dfrac{4\pi}{5}+k2\pi, k\in\mathbb{Z}\right\}$.
			\item $2\cos^2x+3\cos x+1=0\Leftrightarrow\hoac{&\cos x=-1\\&\cos x=-\dfrac{1}{2}}\Leftrightarrow\hoac{&x=\pi+k2\pi\\&x=\pm\dfrac{2\pi}{3}+k2\pi} (k\in\mathbb{Z})$.\\
Vậy $S=\left\{\pi+k2\pi; \pm\dfrac{2\pi}{3}+k2\pi, k\in\mathbb{Z}\right\}$.
            \item Điều kiện: $\cos2x\neq 1\Leftrightarrow 2x\neq k2\pi, k\in\mathbb{Z}\Leftrightarrow x\neq k\pi, k\in\mathbb{Z}$.\\
            PT $\Leftrightarrow \sin2x\cos2x+4\sin x\cos^2x-3\sin2x-\cos2x-2\cos x+3=0$\\
            $\Leftrightarrow (\sin2x\cos2x-\cos2x)+(-3\sin2x+3)+(4\sin x\cos^2x-2\cos x)=0$\\
            $\Leftrightarrow \cos2x(\sin2x-1)-3(\sin2x-1)+2\cos x(\sin2x-1)=0$\\
            $\Leftrightarrow (\sin2x-1)(\cos2x-3+2\cos x)=0$\\
            $\Leftrightarrow (\sin2x-1)(2\cos^2x+2\cos x-4)=0$\\
            $\Leftrightarrow\hoac{&\sin2x=1\\&\cos x=1\\&\cos x=-2\quad\text{(vô nghiệm)}}\Leftrightarrow\hoac{&2x=\dfrac{\pi}{2}+k2\pi\\&x=k2\pi} (k\in\mathbb{Z})\Leftrightarrow\hoac{&x=\dfrac{\pi}{4}+k\pi\\&x=k2\pi} (k\in\mathbb{Z})$.\\
            Kết hợp điều kiện ta có $S=\left\{\dfrac{\pi}{4}+k\pi, k\in\mathbb{Z}\right\}$.
		\end{enumerate}
	}
\end{bt}

Thiết diện là tứ giác $IKFE$.\\
Theo cách dựng, ta có $IK\parallel EF$. Gọi $M$ là trung điểm của $AB$.\\
Vì $IK\parallel AB$ và $G$ là trọng tâm $\triangle SAB$ nên $\dfrac{IK}{AB}=\dfrac{SK}{SB}=\dfrac{SG}{SM}=\dfrac{2}{3}$
$\Rightarrow IK=\dfrac{2}{3}AB=2CD$\\
Mà $EF=\dfrac{CD+AB}{2}=\dfrac{CD+3CD}{2}=2CD.$
Suy ra $IK=EF$. Vậy $IKFE$ là hình bình hành.
\end{enumerate}
}
\end{bt}
