\begin{ex}%[Đề thi HSG 9, tỉnh Lào Cai, năm 2018]%[Nguyễn Quang Tân, Dự án EX-9G-18]%[9D0G5]
	
	Tìm tất cả các cặp số tự nhiên $(x;y)$ thỏa mãn phương trình: $(x+1)^4-(x-1)^4=y^3.$
	\loigiai{
		Giả sử $x,y$  là các số tự nhiên thỏa mãn:  
		$(x+1)^4-(x-1)^4=y^3.$
		Ta biến đổi:  \\
		$\begin{array}{rl}
		(x+1)^4-(x-1)^4&=[(x+1)^2-(x-1)^2]\cdot [(x+1)^2+(x-1)^2]\\
		&=4x(2x^2+2)=8x^3+8x.
		\end{array}
		$
		\\
		Vì  $x$ là số tự nhiên nên $(2x)^3\leq 8x^3+8x< (2x+1)^3.$\\ 
		Hay $(2x)^3\leq y^3<(2x+1)^3.$ Suy ra $2x\leq y<2x+1.$ \\
		Nên $y=2x.$ Suy ra: $8x^3+8x=(2x)^3 \Leftrightarrow x=0.$\\
		Vậy chỉ có một cặp số tự nhiên thỏa mãn là  $(x;y)=(0;0).$
		
	}
	\end{ex}

	 Tìm tất cả các số nguyên tố $p$ để $2p+1$ là lập phương của một số tự nhiên.
	\loigiai{
		 Giả sử $2p+1=n^3$ với $n$ là số tự nhiên lẻ.\\
			Suy ra $n=2k+1, k\in \mathbb{N}.$\\
			Khi đó, $2p+1=(2k+1)^3\Rightarrow p=k(4k^2+6k+3).$\\
			Vì $k<4k^2+6k+3$ và $p$ là số nguyên tố nên $k=1.$\\
		Vậy $ 2p+1=3^3\Leftrightarrow p=13.$
		
	
	}

\begin{ex}%[Đề thi HSG 9, tỉnh Lào Cai, năm 2018]%[Nguyễn Quang Tân, Dự án EX-9G-18]%[9D1K8]
	Cho $P(x)=\dfrac{x^2+\sqrt{x}}{x-\sqrt{x}+1}+\dfrac{x+\sqrt{x}}{\sqrt{x}}-\dfrac{x-1}{\sqrt{x}+1}$ với $x>0.$
	\begin{enumerate}
		\item Rút bọn biểu thức $P(x).$\\
		\item Với $x>0$ tìm giá trị lớn nhất của biểu thức $\dfrac{7\sqrt{x}}{P(x)}.$
	\end{enumerate}
\loigiai{
 \begin{enumerate}
	\item Ta có \\
	$\begin{array}{rl}
	P(x)&=\dfrac{\sqrt{x}(x\sqrt{x}+1)}{x-\sqrt{x}+1}+\dfrac{\sqrt{x}(\sqrt{x}+1)}{\sqrt{x}}-\dfrac{(\sqrt{x})^2-1}{\sqrt{x}+1}\\
	&=\dfrac{\sqrt{x}(\sqrt{x}+1)(x-\sqrt{x}+1)}{x-\sqrt{x}+1}+\sqrt{x}+1-(\sqrt{x}-1)\\
	&=x+\sqrt{x}+2.
	\end{array}$
	\item Biển đổi $\dfrac{7\sqrt{x}}{P(x)}=\dfrac{7\sqrt{x}}{x+\sqrt{x}+2}=\dfrac{7}{\sqrt{x}+1+\frac{2}{\sqrt{x}}}$\\
	Lại có $\sqrt{x}+1+\dfrac{2}{\sqrt{x}}=\sqrt{x}+\dfrac{2}{\sqrt{x}}+1\ge 2\sqrt{\sqrt{x}.\dfrac{2}{\sqrt{x}}}+1=2\sqrt{2}+1$\\
	$\Rightarrow \dfrac{7}{\sqrt{x}+1+\dfrac{2}{\sqrt{x}}}\le \dfrac{7}{2\sqrt{2}+1}$ . Đẳng thức xảy ra khi $\sqrt{x}=\dfrac{2}{\sqrt{x}}\Leftrightarrow x=2.$
	\\
	Vậy $\max \dfrac{7\sqrt{x}}{P(x)}=2\sqrt{2}-1$ khi $x=2.$
	
	
	
\end{enumerate}
}
	\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Lào Cai, năm 2018] %[Nguyễn Quang Tân, Dự án EX-9G-18]%[9D0G2]
    Giải hệ phương trình: $\heva{3x^3+2x^3&=1\\
        xy^3-2x&=3.}$
\loigiai{
 Ta thấy $x=0$ không thỏa mãn hệ đã cho, suy ra $x\ne 0$.\\
Hệ đã cho tương đương: $\heva{ &2+3y =\left( \dfrac{1}{x} \right)^3 \\ 
&	y^3-2 =3\dfrac{1}{x} }$\\
Đặt $\dfrac{1}{x}=u\Rightarrow $ hệ đã cho có dạng $\heva{&u^3-2&=3y \\ &y^3-2&=3u. 
}$\\
$\Rightarrow \left( u-y \right)\left( u^2+uy+y^2+3 \right)=0$
   $\Leftrightarrow \hoac{
&u=y \\	&\left( u+\frac{y}{2} \right)^2+\frac{2y^2}{4}+3=0 
}$  
 $\Leftrightarrow u=y$.
   \\
   Với $u=y\Rightarrow y^3-3y-2=0\Leftrightarrow \left( y+1 \right)\left( y^2-y-2 \right)=0$ 
   $\Leftrightarrow \hoac{&y=-1 \\ 	&y=2. }$
\\
   Ta xét hai trường hợp
  \begin{enumerate}[i)]
  	\item $y=-1\Rightarrow u=-1\Rightarrow x=-1\Rightarrow $ hệ đã cho có nghiệm $\left( x;\,y \right)=\left( -1;\,-1 \right)$

  	\item $y=2\Rightarrow u=2\Rightarrow x=\dfrac{1}{2}\Rightarrow $ hệ đã cho có nghiệm $\left( x;\,y \right)=\left( \dfrac{1}{2};\,2 \right)$
  \end{enumerate}
   	Vậy, hệ đã cho có nghiệm $\left( x;\,y \right)$ là: $\left( -1;\,-1 \right)$  và $\left( \dfrac{1}{2};\,2 \right)$.
    }
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Lào Cai, năm 2018] %[Nguyễn Quang Tân, Dự án EX-9G-18]%[9D2K3]%[9D4K2]
	Cho hàm số $y=x^2$ có đồ thị là $(P)$ và hai điểm $A$, $B$ thuộc $(P)$ có hoành độ lần lượt là:$-\sqrt{3};\dfrac{\sqrt{3}}{3}$.  Đường thẳng (d) có phương trình $y=ax+b$.
	\begin{enumerate}
		\item Tìm $a$, $b$ biết đường thẳng $(d)$ đi qua hai điểm $A$ và $B.$
		\item Tính diện tích tam giác $OAB$ (điểm $O$ là gốc tọa độ).
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Ta có $A=\left( -\sqrt{3};3 \right)$ và $B=\left( \dfrac{\sqrt{3}}{3};\dfrac{1}{3} \right).$
		\\
		Do đường thẳng $(d)$ đi qua $A,B$ nên \\
		$\heva{&-a\sqrt{3}+b=3  \\
			&a\frac{\sqrt{3}}{3}+b=\frac{1}{3} }$
		$\Leftrightarrow \heva {	&-\sqrt{3}a+b=3  \\ &\sqrt{3}a+3b=1  }$
		$\Leftrightarrow \heva { 	&	-\sqrt{3}a+b=3  \\
			&	4b=4  }$
		$\Leftrightarrow \heva {&	a=-\frac{2\sqrt{3}}{3}  \\
			&	b=1  }$
		
		\item  Gọi M, N lần lượt là hình chiếu vuông góc của A, B trên Oy.
		
		Ta có: $\tan \widehat{AOM}=\dfrac{AM}{OM}=\dfrac{\sqrt{3}}3\Rightarrow \widehat{AOM}=30^{\circ }.$\\
		$\tan \widehat{BON}=\dfrac{NB}{ON}=\dfrac{\sqrt{3}}{3}:\dfrac{1}{3}=\sqrt{3}\Rightarrow \widehat{BON}={60}^{\circ }.$\\
		$\Rightarrow OA=\dfrac{AM}{\sin 30^{\circ }}=2\sqrt{3}$ 
		và $OB=\dfrac{BN}{\sin 60^{\circ }}=\dfrac{2}{3}$\\
		Suy ra $\widehat{BOA}=30^{\circ }+60^{\circ}=90^{\circ }.$ 
		
		Diện tích tam giác OAB là $S=\dfrac{1}{2}OA.OB=\dfrac{2\sqrt{3}}{3}.$
		
		
	\end{enumerate}
}
	\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Lào Cai, năm 2018]%[Nguyễn Quang Tân, Dự án EX-9G-18]%[9D0G7]
	 Cho 3 số dương $a,b,c$ thỏa mãn: $a+b+c=3$.
	Tìm giá trị nhỏ nhất của \[P=\dfrac{1}{1+ab}+\dfrac{1}{1+bc}+\dfrac{1}{1+ca}.\]
	\loigiai{
	 Ta có: $\dfrac{1}{1+ab}=1-\dfrac{ab}{1+ab}$\\
		Vì $1+ab\ge 2\sqrt{ab}\Rightarrow \dfrac{ab}{1+ab}\le \dfrac{ab}{2\sqrt{ab}}$ $\Rightarrow 1-\dfrac{ab}{1+ab}\ge 1-\dfrac{ab}{2\sqrt{ab}}=1-\dfrac{\sqrt{ab}}{2}.$
		\\
		Vì  $\sqrt{ab}\le \dfrac{a+b}{2}\Rightarrow 1-\dfrac{\sqrt{ab}}{2}\ge 1-\dfrac{a+b}{4}$ $\Rightarrow \dfrac{1}{1+ab}\ge 1-\dfrac{a+b}{4}.$
		\\
		Tương tự ta có: $\dfrac{1}{1+bc}\ge 1-\dfrac{b+c}{4}$  và  $\dfrac{1}{1+ca}\ge 1-\dfrac{c+a}{4}$\\
		Vậy $P=\dfrac{1}{1+ab}+\dfrac{1}{1+bc}+\dfrac{1}{1+ca}\ge 3-\dfrac{a+b+c}{2}=\dfrac{3}{2}.$\\
		Đẳng thức xảy ra khi $a=b$ và $ab=1$ suy ra $a=b=c=1$.\\
		Vậy giá trị nhỏ nhất của $P$ là $\dfrac{3}{2}$ khi $a=b=c=1.$
}
	\end{ex}
