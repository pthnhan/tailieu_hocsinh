
\section{ĐỒ THỊ CỦA HÀM SỐ BẬC NHẤT}

\subsection{Nền tảng kiến thức}
Đồ thị của hàm số bậc nhất $y=ax+b$ ($a\ne 0$) là một đường thẳng:
\begin{itemize}
	\item Cắt trục tung tại điểm có tung độ bằng $b$.
	\item Song song với đường thẳng $y=ax$ nếu $b\ne 0$; trùng với đường thẳng $y=ax$ nếu $b=0$.
\end{itemize}
Hàm số $y=ax$ là trường hợp đặc biệt của đồ thị hàm số $y=ax+b$ khi $b=0.$

\subsection{Áp dụng giải toán}

\begin{dang}{Điểm thuộc đường thẳng, điểm không thuộc đường thẳng}
	Để biết điểm có thuộc đường thẳng hay không ta thay trực tiếp điểm vào phương trình đường thẳng.
\end{dang}

\begin{vd}
	Điểm nào sau đây thuộc đường thẳng $d\colon y=2x+3.$
	\begin{enumEX}{2}
		\item $A(0;3).$
		\item $B(-\dfrac{3}{2};0).$
		\item $C(1;2).$
		\item $D(-1;1).$
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
				\item Thay tọa độ điểm $A(0;3)$ vào đường thẳng $d$ ta có $3=2\cdot 0+3$ nên $A\in d.$
			\item Thay tọa độ điểm $B(-\dfrac{3}{2};0)$ vào đường thẳng $d$ ta có $0=2\left(-\dfrac{3}{2}\right)+3$ nên $B\in d$.
			\item Thay tọa độ điểm $C(1;2)$ vào đường thẳng $d$ ta có $2\ne 2\cdot 1+3$ nên $C\notin d.$
			\item Thay tọa độ điểm $D(-1;1)$ vào đường thẳng $d$ ta có $1=2\cdot (-1)+3$ nên $D\in d.$
		\end{enumerate}
	}
\end{vd}

\begin{vd}
	Cho các điểm $A(0;-5)$, $B(1;-2)$, $C(2;1)$, $D(2{,}5;2{,}5)$, $M(x;14)$, $N(-5;20)$, $P(7;-16).$
	\begin{enumerate}
		\item Chứng minh rằng bốn điểm $A$, $B$, $C$, $D$ thẳng hàng.
		\item Tìm $x$ để ba điểm $M$, $N$, $P$ thẳng hàng.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Gọi $d\colon y=ax+b$ là đường thẳng đi qua hai điểm $A$, $B$ suy ra
			\begin{center}
				$\begin{cases}
				-5=b\\
				-2=a+b
				\end{cases}\Leftrightarrow \begin{cases}
				a=3\\
				b=-5.
				\end{cases}$
			\end{center}
			Vậy đường thẳng $d$ có phương trình là $d\colon y=3x-5$.
			Ta thấy $1=2\cdot 3-5$ nên $C\in d$.\\
			Mà $2{,}5=2\cdot 2{,}5-5$ nên $D\in d$.\\
			Vậy $4$ điểm $A$, $B$, $C$, $D$ đều thuộc đường thẳng $d$ nên chúng thẳng hàng.
			\item Gọi $\Delta \colon y=ax+b$ là đường thẳng đi qua hai điểm $N$, $P$ suy ra 
			\begin{center}
				$\begin{cases}
				20=-5a+b\\
				-16=7a+b
				\end{cases}\Leftrightarrow \begin{cases}
				36=-12a\\
				-16=7a+b
				\end{cases}\Leftrightarrow \begin{cases}
				a=-3\\
				b=5.
				\end{cases}$
			\end{center}
			Vậy đường thẳng $\Delta$ có phương trình là $\Delta \colon y=-3x+5$.\\
			Ba điểm $M$, $N$, $P$ thẳng hàng khi và chỉ khi $M\in \Delta \Leftrightarrow 14=(-3)\cdot x+5 \Leftrightarrow x=-3$.\\
			Vậy $x=-3$ thì ba điểm $M$, $N$, $P$ thẳng hàng.
		\end{enumerate}
	}
\end{vd}

\begin{vd}
	Tìm điểm cố định mà mỗi đường thẳng sau đây luôn đi qua với mọi giá trị của $m.$
	\begin{multicols}{2}
		\begin{enumerate}
			\item $y=(m-2)x+3.$
			\item $y=mx+(m+2).$
		\end{enumerate}
	\end{multicols}
	\loigiai
	{
		\begin{enumerate}
			\item Gọi $A(x_0;y_0)$ là điểm cố định thuộc đồ thị hàm số, khi đó
			\begin{eqnarray*}
				y_0=(m-2)x_0+3 \text{ với mọi } m &\Leftrightarrow & mx_0-2x_0-y_0+3=0 \text{ với } \forall m \\
				&\Leftrightarrow & \begin{cases}
					x_0=0\\
					-2x_0-y_0+3=0
				\end{cases}\\
				&\Leftrightarrow & \begin{cases}
					x_0=0\\
					y_0=3.
				\end{cases}
			\end{eqnarray*}
			Vậy điểm cố định của đồ thị hàm số là $A(0;3)$.
			\item Gọi $B(x_0;y_0)$ là điểm cố định thuộc đồ thị hàm số, khi đó
			\begin{eqnarray*}
				y_0=mx_0+m+2 \text{ với }\forall m &\Leftrightarrow & m(x_0+1)+2-y_0=0 \text{ với }\forall m \\
				&\Leftrightarrow & \begin{cases}
					x_0+1=0\\
					2-y_0=0
				\end{cases}\\
				&\Leftrightarrow & \begin{cases}
					x_0=-1\\
					y_0=2.
				\end{cases}
			\end{eqnarray*}
			Vậy điểm cố định của đồ thị hàm số là $B(-1;2)$.
		\end{enumerate}
	}
\end{vd}

\begin{dang}{Xác định đường thẳng thỏa mãn tính chất nào đó}
	- Để viết phương trình đường thẳng qua hai điểm ta gọi đường thẳng cần tìm là $y=ax+b$ rồi thay lần lượt các điểm vào đường thẳng để tìm $a$, $b$.\\
	- Ta thay điểm vào đường thẳng để tìm giá trị của tham số.
\end{dang}

\begin{vd}
	Cho hàm số $y=(m-1)x+2m.$
	\begin{enumerate}
		\item Xác định giá trị của $m$ để đồ thị của hàm số cắt trục tung tại điểm có tung độ bằng $2$.
		\item Xác định giá trị của $m$ để đồ thị của hàm số cắt trục hoành tại điểm có hoành độ bằng $-3.$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Đồ thị hàm số cắt trục tung tại điểm có tung độ bằng $2$ thì 
			$$2=(m-1)\cdot 0+2m\Leftrightarrow m=1.$$
			Đường thẳng cần tìm là $y=2.$
			\item Đồ thị hàm số cắt trục hoành tại điểm có hoành độ bằng $-3$ thì 
			\begin{center}
				$0=(m-1)\cdot (-3)+2m\Leftrightarrow m=3.$
			\end{center}
			Đường thẳng cần tìm là $y=2x+6.$
		\end{enumerate}
	}
\end{vd}

\begin{vd}
	Xác định đường thẳng đi qua hai điểm $A$ và $B$ biết
	\begin{enumEX}{2}
		\item $A(-2;0)$, $B(0;1)$.
		\item $C(1;4)$, $D(3;0)$.
		\item $E(-2;2)$, $F(1;5)$.
		\item $G(2;-33)$, $H(-1;18).$
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item Gọi đường thẳng cần tìm là $y=ax+b.$\\
			Vì đường thẳng đi qua hai điểm $A(-2;0)$ và $B(0;1)$ nên
			$\begin{cases}
			0=-2a+b\\
			1=b
			\end{cases} \Leftrightarrow \begin{cases}
			a=\dfrac{1}{2}\\
			b=1.
			\end{cases}$\\ 
			Vậy đường thẳng cần tìm là $y=\dfrac{1}{2}x+1.$
			\item Gọi đường thẳng cần tìm là $y=ax+b.$\\
			Vì đường thẳng đi qua hai điểm $C(1;4)$, $D(3;0)$ nên $\begin{cases}
			4=a+b\\
			0=3a+b
			\end{cases} \Leftrightarrow \begin{cases}
			a=-2\\
			b=6.
			\end{cases}$\\
			Vậy đường thẳng cần tìm là $y=-2x+6.$
			\item Gọi đường thẳng cần tìm là $y=ax+b.$\\
			Vì đường thẳng đi qua hai điểm $E(-2;2)$, $F(1;5)$ nên
			\begin{center}
				$\begin{cases}
				2=-2a+b\\
				5=a+b
				\end{cases} \Leftrightarrow \begin{cases}
				2=-2a+b\\
				10=2a+2b
				\end{cases} \Leftrightarrow \begin{cases}
				12=3b\\
				2=-2a+b
				\end{cases}
				\Leftrightarrow \begin{cases}
				a=1\\
				b=4.
				\end{cases}$ 
			\end{center}
			Vậy đường thẳng cần tìm là $y=x+4.$
			\item Gọi đường thẳng cần tìm là $y=ax+b.$\\
			Vì đường thẳng đi qua hai điểm $G(2;-33)$, $H(-1;18)$ nên 
			\begin{center}
				$\begin{cases}
				-33=2a+b\\
				18=-a+b
				\end{cases} \Leftrightarrow \begin{cases}
				-33=2a+b\\
				36=-2a+2b
				\end{cases}
				\Leftrightarrow \begin{cases}
				3b=3\\
				a=b-18
				\end{cases}
				\begin{cases}
				b=1\\
				a=-17.
				\end{cases}$\\
			\end{center}
			Vậy đường thẳng cần tìm là $y=-17x+1.$
		\end{enumerate}
	}
\end{vd}

\begin{vd}
	Cho ba đường thẳng $(d_1)\colon y=x-2$, $(d_2)\colon y=2x-1$ và $(d_3)\colon y=(m-1)x+2m$. Tìm $m$ để ba đường thẳng đồng quy.
	\loigiai{
		Giao điểm $M$ của đường thẳng $(d_1)$ và $(d_2)$ là nghiệm của hệ
		\begin{center}
			$\heva{&y=x-2\\&y=2x-1} \Leftrightarrow \heva{&2x-1=x-2\\&y=2x-1}\Leftrightarrow \heva{&x=-1\\&y=-3.}$
		\end{center}
		Tọa độ điểm $M(-1;-3).$\\
		Để ba đường thẳng đồng quy thì đường thẳng $(d_3)$ phải đi qua giao điểm $M$ của $(d_1)$ và $(d_2)$, khi đó
		\begin{center}
			$-3=(m-1)\cdot (-1)+2m \Leftrightarrow -3=1-m+2m \Leftrightarrow m=-4.$
		\end{center}
		Vậy với $m=-4$ thì ba đường thẳng đồng quy.
	}
\end{vd}


\begin{dang}{Vẽ đồ thị hàm số bậc nhất, đồ thị hàm trị tuyệt đối}
	- Vẽ đồ thị hàm số $y=ax+b$.
	\begin{itemize}
		\item Khi $b=0$ thì $y=ax$. Đồ thị của hàm số $y=ax$ là đường thẳng đi qua gốc tọa độ $O(0;0)$ và điểm $A(1;a)$.
		\item Xét trường hợp $y=ax+b$ với $a\ne 0$ và $b\ne 0.$\\
		Ta đã biết đồ thị của hàm số $y=ax+b$ là một đường thẳng. Do đó, để vẽ đồ thị hàm số $y=ax+b$, ta chỉ cần xác định được hai điểm phân biệt nào đó thuộc đồ thị rồi vẽ đường thẳng đi qua hai điểm đó. Thường lấy hai điểm thuộc các trục tọa độ.
	\end{itemize}
	- Để vẽ đồ thị hàm trị tuyệt đối ta cần phá dấu trị tuyệt đối rồi vẽ từng đồ thị một.
\end{dang}

			\item Tọa độ điểm $A$ là nghiệm của hệ $\heva{&y=x\\&y=3-x} \Leftrightarrow \heva{&x=\dfrac{3}{2}\\&y=\dfrac{3}{2}}.$ Vậy $A\left(\dfrac{3}{2};\dfrac{3}{2}\right).$\\
			Tọa độ điểm $B$ là nghiệm của hệ $\heva{&y=2x\\&y=3-x}\Leftrightarrow \heva{&x=1\\&y=2}.$ Vậy $B(1;2).$\\
			Gọi $H$ là hình chiếu của điểm $A$ lên trục $Ox$,

			Ta có $OA=\sqrt{OH^2+HA^2}=\dfrac{3\sqrt{2}}{2}$, $AB=\sqrt{AK^2+KB^2}=\dfrac{\sqrt{2}}{2}.$\\
			Diện tích tam giác $OAB$ là $S_{\triangle OAB}=\dfrac{1}{2}OA\cdot AB=\dfrac{1}{2}\cdot \dfrac{3\sqrt{2}}{2}\cdot \dfrac{\sqrt{2}}{2}=\dfrac{3}{4}.$
		\end{enumerate}

\end{vd}

\subsection{Luyện tập}

\begin{bt}
	Cho hàm số bậc nhất $y=(a-1)x+2a-3$. Tìm $a$ biết đồ thị hàm số đi qua điểm
	\begin{multicols}{2}
		\begin{enumerate}
			\item $A(1;2)$.
			\item $B(-1;-3).$
		\end{enumerate}
	\end{multicols}
	\loigiai{
		\begin{enumerate}
			\item Vì đồ thị hàm số đi qua điểm $A(1;2)$ nên $2=(a-1)+2a-3 \Leftrightarrow 3a=6\Leftrightarrow a=2$.
			\item Vì đồ thị hàm số đi qua điểm $B(-1;-3)$ nên $-3=-(a-1)+2a-3\Leftrightarrow a=-1.$
		\end{enumerate}
	}
\end{bt}

\begin{bt}
	Cho đường thẳng $y=(2m+1)x+3-2m$.
	\begin{enumerate}
		\item Tìm $m$ để hàm số là hàm nghịch biến.
		\item Với giá trị nào của $m$ thì đồ thị hàm số cắt trục hoành tại điểm có hoành độ $x=2.$
		\item Với giá trị nào của $m$ thì đồ thị hàm số cắt trục tung tại điểm có tung độ $y=-2$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Hàm số nghịch biến khi $2m+1<0\Leftrightarrow m<-\dfrac{1}{2}.$
			\item Ta có $0=(2m+1)\cdot 2+3-2m\Leftrightarrow m=-\dfrac{5}{2}.$
			\item Ta có $-2=(2m+1)\cdot 0+3-2m\Leftrightarrow m=\dfrac{5}{2}.$
		\end{enumerate}
	}
\end{bt}

\begin{bt}
	Tìm điểm cố định của các đồ thị hàm số sau
	\begin{multicols}{2}
		\begin{enumerate}
			\item $(d_1)\colon y=(m-1)x+2m-3.$
			\item $(d_2)\colon y=(2m+1)x+3-2m.$
		\end{enumerate}
	\end{multicols}

\subsection{Thử thách}

\begin{bt}
	Chứng minh rằng nếu một đường thẳng không đi qua gốc tọa độ, cắt trục hoành tại điểm có hoành độ bằng $a(a\ne 0)$, cắt trục tung tại điểm có tung độ bằng $b(b\ne 0)$ thì đường thẳng đó có phương trình là 
	\begin{center}
		$\dfrac{x}{a}+\dfrac{y}{b}=1.$
	\end{center}
	\loigiai{
		Gọi đường thẳng cần xác định là $y=mx+n.$\\
		Đường thẳng đi qua điểm $(0;b)$ nên $b=m\cdot 0+n\Rightarrow n=b.$\\
		Đường thẳng $y=mx+b$ đi qua điểm $(a;0)$ nên $0=m\cdot a+b\Rightarrow m=-\dfrac{b}{a}$(chú ý rằng $a\ne 0$).\\
		Đường thẳng cần xác định có phương trình là $y=-\dfrac{b}{a}x+b$ hay $\dfrac{y}{b}=-\dfrac{x}{a}+1$ tức $\dfrac{x}{a}+\dfrac{y}{b}=1.$
	}
\end{bt}

\begin{bt}
	Xác định đường thẳng đi qua $A(4;3)$, cắt trục tung tại điểm có tung độ là một số nguyên dương, cắt trục hoành tại điểm có hoành độ là một số nguyên tố.
	\loigiai{
		Giả sử đường thẳng phải tìm cắt trục hoành tại điểm có hoành độ bằng $a$, cắt trục tung tại điểm có tung độ bằng $b$ thì đường thẳng có dạng $\dfrac{x}{a}+\dfrac{y}{b}=1$(xem bài trên).\\
		Điểm $A(4;3)$ thuộc đường thẳng nên $\dfrac{4}{a}+\dfrac{3}{b}=1$. Do đó $b=\dfrac{3a}{a-4}=3+\dfrac{12}{a-4}.$\\
		Do $a$ là số nguyên tố nên $a\ge 2,a-4\ge -2$.\\
		Lần lượt cho $a-4$ nhận các giá trị $-2$, $-1$, $1$, $2$, $3$, $4$, $6$, $12$ với chú ý rằng $a$ là số nguyên tố và $b>0$, ta được $\heva{&a=5\\&b=15}$ và $\heva{&a=7\\&b=7.}$\\
		Ta tìm được hai đường thẳng $\dfrac{x}{5}+\dfrac{y}{15}=1$ (hay $y=-3x+15$) và $\dfrac{x}{7}+\dfrac{y}{7}=1$ (hay $y=-x+7$).
	}
\end{bt}
