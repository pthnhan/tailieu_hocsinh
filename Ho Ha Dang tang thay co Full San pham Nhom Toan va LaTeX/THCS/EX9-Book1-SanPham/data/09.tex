\begin{vd}
Tìm nghiệm nguyên của phương trình
\begin{center}
  $x^{4}-4x^{2}+y^{2}+2x^{2}y-9=0$
\end{center}
    \loigiai{
Ta có: $x^{4} -4x^{2}+y^{2}+2x^{2}y-9=0\Leftrightarrow \left(x^{2}+y\right)^{2}-4x^{2}=0
\Leftrightarrow \left(x^{2}+y+2x\right).\left(x^{2}+y-2x\right)=9$.\\
Suy ra: $x^{2}+y+2x$ và $x^{2}+y-2x$ là ước của $9$ nên ta có bảng giá trị sau:
\begin{center}
\begin{tabular}{|l|c|c|c|c|c|c|r|}
\hline
$x^{2}+y+2x$ & 1 & 3 & 9 & -1 & -3 & -9\\
\hline
$x^{2}+y-2x$ & 9 & 3 & 1 & -9 & -3 & -1\\
\hline
$2x$ & -4 & 0 & 4 & 4 & 0 & -4\\
\hline
$x$ & -2 & 0 & 2 & 2 & 0 & -2\\
\hline
$y$ & 1 & 3 & 1 & -9 & -3 & -9\\
\hline
\end{tabular}
\end{center}
Đáp số: Nghiệm $(x;y)$ là $(-2;-1),(0;3),(2;1),(2;-9),(0;-3),(-2;-9)$.
    }
\end{vd}
\begin{vd} Tìm nghiệm nguyên của phương trình
\[x(x+1)(x+2)(x+3)=y^{2}.\tag{1}\]
\loigiai{
Nếu $y$ thỏa mãn phương trình thì $-y$ cũng thỏa mãn, do đó ta giả sử $y\geq 0$, khi đó:
$$(1)\Leftrightarrow \left(x^{2}+3x\right)\left(x^{2}+3x+2\right)=y^{2}.$$
Đặt $x^{2}+3x+1=a$, ta được:
$$(a-1)(a+1)=y^{2} \Leftrightarrow a^{2}-1=y^{2} \Leftrightarrow (a+y)(a-y)=1.$$
Suy ra $a+y=a-y$, do đó: $y=0$.\\
 Thay vào $(1)$ được: $x_{1}=0,x_{2}=-1,x_{3}=-2, x_{4}=-3.$\\
 \emph{Đáp số}: Nghiệm $(x;y)$ là: $(0;0),(-1;0),(-2;0),(-3;0).$
 }
\end{vd}
\begin{vd}
Tìm các số nguyên $x$ để biểu thức: $x^{4}+2x^{3}+2x^{2}+x+3$ là một số chính phương
\loigiai{
Đặt $x^{4}+2x^{3}+2x^{2}+x+3=y^{2}$ với $y\in \mathbb{N}.$\\
\emph{Cách 1.} Ta thấy $y^{2}=\left(x^{4}+2x^{3}+x^{2}\right)+\left(x^{2}+x+3\right)
=(x^{2}+x)^{2}+(x^{2}+x+3).$\\
Ta sẽ chứng minh $a^{2}<y^{2}<(a+2)^{2}$ với $a=x^{2}+x$.\\
Thật vậy: $y^{2}-a^{2}=x^{2}+x+3=\left(x+\dfrac{1}{2}\right)^{2}+\dfrac{11}{4}>0.$\\
          $(a+2)^{2}-y^{2}=(x^{2}+x+2)^{2}-(x^{4}+2x^{3}+2x^{2}+x+3)$\\
                         $=3x^{2}+3x+1=3\left(x+\dfrac{1}{2}\right)^{2}+\dfrac{1}{4}>0.$\\
Do $a^{2}<y^{2}<(a+2)^{2}$ nên $y^{2}=(a+1)^{2}.$\\
    hay là $x^{4}+2x^{3}+2x^{2}+x+3=(x^{2}+x+1)^{2}\Leftrightarrow x^{2}+x-2=0\Leftrightarrow\hoac{&x=1 \\&x=-2.}$\\
Vậy $x=1$ hoặc $x=-2$.\\
\emph{Cách 2.} Đưa về phương trình ước số:\\
Nhân hai vế của phương trình đã cho với $4$ ta được:
{\allowdisplaybreaks
\begin{align*}
4y^{2}&=4x^{4}+8x^{3}+8x^{2}+4x+12\\
      &=4x^{4}+8x^{3}+4x^{2}+4x^{2}+4x+1+11\\
      &=\left(2x^{2}+(2x+1)\right)^{2}=11
\end{align*}}Do đó $(2x^{2}+2x+1+2y)(2x^{2}+2x+1-2y)=-11$\\
Ta có $2x^{2}+2x+1+2y\geq 2x^{2}+2x+1-2y$ nên xảy ra hai trường hợp:\\
\textbf{TH1.} $\heva{&2x^{2}+2x+1+2y=11 \\&2x^{2}+2x+1-2y=-1}\Leftrightarrow \heva{&2x^{2}+2x+1=5 \\&2y=6} \Leftrightarrow \heva{&x^{2}+x-2=0 \\&y=3} \Leftrightarrow \heva{&\hoac{&x=1 \\&x=-2} \\&y=3}.$\\
\textbf{TH2.} $\heva{&2x^{2}+2x+1+2y=1 \\&2x^{2}+2x+1-2y=-11}\Leftrightarrow\heva{&2x^{2}+2x+1=-5 \\&2y=6} \Leftrightarrow\heva{&x^{2}+x+3=0 \\&y=3.}$\\
Hệ này vô nghiệm.\\
\emph{Đáp số:} Có hai số nguyên $1$ và $-2$ thỏa mãn.
}
\end{vd}
\begin{vd} Tìm nghiệm nguyên của phương trình:
\[\left(x^{2}-y^{2}\right)^{2}=10y+9.\tag{1}\]
\loigiai{
Ta thấy $10y+9\geq0$ nên $y\geq -\dfrac{10}{9}$. Do $y\in \mathbb{Z}$ nên $y\geq0$.
\begin{itemize}
\item Xét $y=0$ thì $(1)$ trở thành $x^{4}=9$ (loại). Ta chỉ xét $y>0$.\\
Vì phương trình $(1)$ không thay đổi khi thay $x$ bởi $-x$, ta giả sử $x\geq0$.
\item Xét $x=0$ thì $(1)$ trở thành $y^{4}=10y+9\Leftrightarrow y(y^{3}-10).$ Suy ra $9 \, \vdots \, y$.\\
Lần lượt xét $y$ bằng $1,3,9$ không có giá trị nào của $y$ thỏa mãn.
\item Xét $x\geq1$ và $y\geq1$ thì $x+y \geq 2$. Viết $(1)$ dưới dạng
\[(x+y)(x-y)^{2}=\dfrac{10y+9}{x+y}.\tag{2}\]
Do $x+y\geq2$ nên $(x+y)(x-y)^{2}\geq2(x-y)^{2}$.\hfill$(3)$\\
Ta lại có $\dfrac{10y+9}{x+y}<\dfrac{10y+10x}{x+y}=10$ (vì $x\geq1$ nên $10x\geq 10x\geq10>9$).\hfill$(4)$\\
Từ $(2),(3),(4)$ suy ra $2(x-y)^{2}<10$ nên $(x-y)^{2}<5.$\\
Ta lại có $x-y$ lẻ (vì $x-y$ là ước của $10y+9$) nên $(x-y)^{2}=1$.\\
Thay $(x-y)^{2}=1$ vào $(2)$ được $(x+y)^{2}=10y+9$.\hfill$(5)$
\begin{itemize}
\item Với $x-y=1$ thì $x=y+1$. Thay vào $(5)$ được $(2y+1)^{2}=10y+9$. \\
$\Leftrightarrow 2y^{2}-3y-4=0$, loại vì $\Delta=41$ không là số chính phương.
\item Với $x-y=-1$ thì $x=y-1$. Thay vào $(5)$ được $(2y-1)^{2}=10y+9$\\
$\Leftrightarrow 2y^{2}-7y-4=0$. Ta có $y_{1}=-\dfrac{1}{2}$ (loại); $y_{2}=4$, khi đó $x=3$.
\end{itemize}
\emph{Đáp số:} Nghiệm $(x;y)$ là $(3;4),(-3;4)$.\\
\textbf{$\blacktriangleright$Kinh nghiệm giải toán:}
\item Ta có thể nghĩ đến viết phương trình $(1)$ dưới dạng:
$$x^{4}-2x^{2}y^{2}+(y^{4}-10y-9)=0$$
rồi đặt $y^{2}=z (z\in \mathbb{N})$ được phương trình bậc hai đối với $z$ là:
$$z^{2}-2y^{2}z+(y^{4}-10y-9)=0$$
Giải các điều kiện $\Delta^{'}\geq0$ và $\Delta^{'}$ là số chính phương đều không có hiệu quả.\\
 Ta sử dụng bất đẳng thức để chặn $(x-y)^{2}$ như cách giải trên.
 \item Vì lũy thừa bậc bốn là bình phương của lũy thừa bậc hai nên ta thường tìm nghiệm nguyên của phương trình bậc bốn với hai ẩn theo các cách:
 \begin{itemize}
 	\item [$-$] Đưa về phương trình bậc hai với hai ẩn.
 \item [$-$]  Phân tích thành nhân tử để phát hiện một biểu thức là số chính phương.
\item [$-$]  Phát hiện một số chính phương nằm giữa hai số chính phương.
\end{itemize}
\end{itemize}
}
\end{vd}

\begin{bt}
Tìm nghiệm nguyên của mỗi phương trình sau:
\begin{enumerate}[a)]
	\item $(x^{2}+y)(x+y^{2})=(x+y)^{3}$;
	\item $x^{4}-2x^{2}y+7y^{2}=55$;
\item $x^{2}y^{2}-2xy=x^{2}+16y^{2}$;
\item  $3x^{2}y^{2}+x^{2}+y^{2}=5xy$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item Khai triển rồi rút gọn được $xy(3x+3y-xy-1)=0.$
\begin{itemize}
  \item Nếu $x=0$ thì $y$ là số nguyên tùy ý.
  \item Nếu $y=0$ thì $x$ la số nguyên tùy ý.
  \item Nếu $xy\neq0$ thì $3x+3y-xy-1=0 \Leftrightarrow (3-x)(y-3)=8$
\end{itemize}
\emph{Đáp số:} Nghiệm $(x;y)$ là $(11;4),(4;11),(7;5),(5;7),(2;-5),(-5;2),(1;-1),(-1;1),(0;k),(t;0)$ với $k,t$ là các số nguyên tùy ý.
\item Đặt $x^2=z (z \in \mathbb{N}),$ ta có
\begin{align*}
&z^2-2yz+(7y^2-55)=0\\
&\Delta^{'}=y^2-(7y^2-55)=-6y^2+55\\
&\Delta^{'}\geq0\Leftrightarrow 6y^2-55\leq0 \Leftrightarrow y^2\leq \dfrac{55}{6}.
\end{align*}
Do $y\in\mathbb{Z}$ nên $y\in \{0;1;-1;2;-2;3;-3\}.$\\
\emph{Đáp số:} Nghiệm $(x;y)$ là $(2;3),(-2;3).$
\item $x^2y^2-2xy=x^2+16y^2 \Leftrightarrow x^2y^2-15y^2=x^2+2xy+y^2.$
\begin{center}
$\Leftrightarrow y^2(x^2-15)=(x+y)^2.$
\end{center}
\begin{itemize}
  \item Với $y=0$ thì $x=0$.
  \item Xét $y\neq0$ thì $x^2-15$ phải là số chính phương.
\end{itemize}
Đặt $x^2-15=a^2 (a\in\mathbb{N}) \Leftrightarrow x^2-a^2=15 \Leftrightarrow (x+a)(x-a)=15$ và $x+a\geq x-a.$\\
Ta có bảng giá trị
 \begin{center}
\begin{tabular}{|l|c|c|c|c|r|}
\hline
$x+a$ & 15 & 5 & -1 & -3\\
\hline
$x-a$ & 1 & 3 & -15 & -5\\
\hline
$x$ & 8 & 4 & -8 & -4\\
\hline
$y$ & -1 & -2 & 1 & 2\\
\hline
\end{tabular}
\end{center}
\item $3x^2y^2+x^2+y^2=5xy$ \hfill$(1)$\\
$\Leftrightarrow x^2+y^2-2xy=3xy-3x^2y^2\Leftrightarrow(x-y)^2=3xy(1-xy)$ \hfill$(2)$\\
Từ $(2)$ suy ra $xy(1-xy)\geq0 \Leftrightarrow 0\leq xy\leq1.$\\
Do $xy$ là số nguyên nên $xy \in \{0;1\}.$
\begin{itemize}
  \item Với $xy=0,$ thay vào $(1)$ được $x=y=0$.
  \item Với $xy=1$ thì $x=y=1$ hoặc $x=y=-1,$ đều nghiệm đúng $(1).$
\end{itemize}
\emph{Đáp số:} Nghiệm $(x;y)$ là $(0;0),(1;1),(-1;-1).$
\end{enumerate}
}
\end{bt}
\begin{bt}
Chứng minh rằng mỗi phương trình sau không có nghiệm nguyên:
\begin{enumerate}[a)]
\item $x^{4}-5x^{2}y^{2}+4y^{4}=3;$
\item $(x+y)^{4}+x^{4}+y^{4}=3996.$
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item $x^2(x^2-y^2)-4y^2(x^2-y^2)=3 \Leftrightarrow (x-2y)(x-y)(x+y)(x+2y)=3.$  \hfill$(1)$
\begin{itemize}
  \item Nếu $y=0$ thì $x^4=3,$ không có nghiệm nguyên.
  \item Nếu $y\neg0$ thì vế trái của $(1)$ là tích của bốn thừa số nguyên khác nhau đôi một, còn vế phải chỉ phân tích được thành nhiều nhất là tích của ba thừa số nguyên khác nhau đôi một: $3=(-3).1.(-1)$.\\
      Phương trình đã cho không có nghiệm nguyên.
\end{itemize}
\item Biến đổi về dạng $2(x^2+y^2+xy)^2=2.1998 \Leftrightarrow (x^2+y^2+xy)^2=1998.$\\
Số chính phương không thể có tận cùng bằng $8$ nên phương trình đã cho không có nghiệm nguyên.
\end{enumerate}
}
\end{bt}
\begin{bt}
Tìm nghiệm nguyên dương của mỗi phương trình sau:
\begin{enumerate}[a)]
\item $(x^{2}-y)(x+y^{2})=(x+y)^{3}$
\item $(x+y)^{4}=40y+1$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item Rút gọn phương trình rồi chia hai vế cho $y\neq0$ được
\begin{align*}
&2y^2+x(3-y)y+x(3x+1)=0.\\
&\Delta=x^2(3-x)^2-8x(3x+1)\\
&=x(x^3-6x^2-15x-8)=x(x+1)^2(x-8).
\end{align*}
Do $(x+1)^2\neq0$ nên để $\Delta$ là số chính phương thì $x(x-8)$ phải là số chính phương.\\
Đặt $x(x-8)=a^2 (x\in \mathbb{N})$\\
$\Leftrightarrow(x-4)^2=a^2+16 \Leftrightarrow (x-4+a)(x-4-a)=16.$\\
Ta thấy $x-4+a$ và $x-4-a$ cùng chẵn và $x-4+a \geq x-4-a$ nên có bảng giá trị sau:
 \begin{center}
\begin{tabular}{|l|c|c|c|c|r|}
\hline
$x-4+a$ & -2 & -4 & 4 & 8\\
\hline
$x-4-a$ & -8 & -4 & 4 & 2\\
\hline
$x-4$ & -5 & -4 & 4 & 5\\
\hline
$x$ & -1,loại & 0,loại & 8 & 9\\
\hline
$x$ &  &  & 10 & 6 hoặc 21\\
\hline
\end{tabular}
\end{center}
\emph{Đáp số:} Nghiệm $(x;y)$ là $(8;10),(9;6),(9;21)$.
\item $(x+y)^4=40y+1$  \hfill$(1)$\\
Theo đề bài $x\geq1,y\geq1$. Viết $(1)$ dưới dạng
\begin{center}
  $(x+y)^3=\dfrac{40y+1}{x+y}.$
\end{center}
Ta chứng minh được
\begin{center}
$2(x+y)^2\leq(x+y)^3=\dfrac{40y+1}{x+y}<\dfrac{40y+40x}{x+y}=40.$
\end{center}
nên $(x+y)^2<20$. Suy ra $x+y\leq44.$\\
Ta lại có $x+y$ lẻ (vì $x+y$ là ước của $40y+1$) và $x+y\geq2$ nên $x+y=3$, ta tìm được $x=1.y=2.$
\end{enumerate}
}
\end{bt}
\begin{bt} Giải mỗi phương trình sau:
\begin{enumerate}[a)]
\item $x^{4}+2x^{3}+3x^{2}+2x=y^{2}+y$ với nghiệm tự nhiên;
\item $x^{4}+x^{3}+x^{2}+x=y^{2}+y$ với nghiệm nguyên.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item $x^4+2x^3+3x^2+2x=y^2+y \Leftrightarrow (x^2+x)^2+2(x^2+x)=y^2+y$ \\
$\Leftrightarrow (x^2+x+1)^2=y^2+y+1$ \hfill $(1)$\\
Với mọi $y\in \mathbb{N}$ ta có $y^2<y^2+y+1\leq y^2+2y+1$\\
nên từ $(1)$ suy ra $y^2<(x^2+x+1)^2\leq(y+1)^2.$\\
Do đó $(x^2+x+1)^2\leq(y+1)^2.$\\
Từ đó và $(1)$ có $y^2+y+1=(y+1)^2\Leftrightarrow y=0.$\\
\emph{Đáp số:} Nghiệm $(x;y)$ là $(0;0).$
\item $x^4+x^3+x^2+x=y^2+y$ \hfill$(1)$\\
$\Leftrightarrow 4x^4+4x^3+4x^2+4x+1=4y^2+4y+1$\\
$\Leftrightarrow (2x^2+x)^2+3x^2+4x+1=(2y+1)^2$.
\begin{itemize}
  \item Trước hết, ta tìm các giá trị của $x$ để xảy ra bất đẳng thức\\
  $(2x^2+x)^2<(2x^2+x)^2+3x^2+4x+1<(2x^2+x+2)^2$ \hfill$(2)$\\
$(2)\Leftrightarrow \heva{&3x^2+4x+1>0\\&(2x^2+x)^2+4(2x^2+x)^2+4-(2x^2+x)^2-3x^2-4x-1>0}$\\
$\Leftrightarrow \heva{&(x+1)(3x+1)>0\\&5x^2+3>0}\Leftrightarrow x<-1$ hoặc $x>-\dfrac{1}{3} \Leftrightarrow x \neq -1$ (do $x\in \mathbb{Z}).$
  \item Như vậy, với $x\neq -1$ thì có $(2)$, lúc đó $(1)$ xảy ra khi\\
  $(2y+1)^2=(2x^2+x+1)$ \hfill$(3)$\\
tức là $(2x^2+x)^2+3x^2+4x+1=(2x^2+x+1)^2$\\
$\Leftrightarrow (2x^2+x)^2+3x^2+4x+1=(2x^2+x)^2+2(2x^2+x)+1$\\
$\Leftrightarrow 3x^2+4x=4x^2+2x$\\
$\Leftrightarrow x^2-2x=0 \Leftrightarrow x=0$ hoặc $x=2$. Thay vào $(3)$ tìm được $y$.
\item Với $x=-1$, thay vào $(1)$ được $y=0$ hoặc $y=-1$.
\end{itemize}
\emph{Đáp số:} Nghiệm $(x;y)$ là $(0;0),(0;-1),(2;5),(2;-6),(-1;0),(-1;-1).$
\end{enumerate}
}
\end{bt}
\begin{bt} Tìm các số nguyên $x$ để mỗi biểu thức sau là số chính phương:
\begin{enumerate}[a)]
\item $x^{4}-x^{2}+2x+2$;
\item $x(x+2)(x^{2}+2x+3)$;
\item $x(x+1)(x+7)(x+8)$;
\item $x^{4}+x^{3}+x^{2}+x+1$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item $x^4-x^2+2x+2=y^2 (y\in \mathbb{N}$\\
$\Leftrightarrow (x^2-1)^2+(x+1)^2=y^2 \Leftrightarrow (x+1)^2.\left[(x-1)^2+1\right]=y^2.$
\begin{itemize}
  \item Nếu $x+1=0$, tức là $x=-1$ thì $y=0$, thỏa mãn.
  \item Nếu $x+1\neq 0$ thì $(x-1)^2+1=k^2 (k\in \mathbb{N})$\\
  $\Leftrightarrow k^2-(x-1)^2=1 \Leftrightarrow (k+x-1)(k-x+1)=1.$\\
Suy ra $k+x-1=k-x+1$ nên $x=1$. Khi đó $y=2$.
\end{itemize}
\emph{Đáp số:} $-1$ và $1$.
  \item $x(x+2)(x^2+2x+3)=y^2 (y\in \mathbb{N})$ \hfill$(1)$\\
  $\Leftrightarrow (x^2+2x)(x^2+2x+3)=y^2.$\\
Đặt $x^2+2x=k$ với $k\in \mathbb{Z}$, ta có $k(k+3)=y^2$.
\begin{itemize}
  \item Xét $y=0$, từ $(1)$ ta có $x_1=0;x_2=-2$.
  \item Xét $y>0$. Biến đổi
  \begin{align*}
  4k^2+12k=4y^2&\Leftrightarrow (2k+3)^2-4y^2=9\\
               &\Leftrightarrow (2k+3+2y)(2k+3-2k)=9
  \end{align*}
Ta thấy $2k+3+2y>2k+3-2y$ nên xảy ra hai trường hợp
  \item Trường hợp $1$: $\heva{&2k+3+2k=9\\&2k+3-2y=1}\Rightarrow 2k+3=5 \Rightarrow k=1$.\\
  Khi đó $x(x+2)=1$, không có nghiệm nguyên.
  \item Trường hợp $2$: $\heva{&2k+3+2y=-1\\&2k+3-2=-9} \Rightarrow 2k+3=-5 \Rightarrow k=-4.$\\
  Khi đó $x(x+2)=-4 \Leftrightarrow x^2+2x-4=0$, vô nghiệm.\\
  \emph{Đáp số:} $0$ và $-2$.
\end{itemize}
\item $x(x+1)(x+7)(x+8)=y^2 (y\in \mathbb{Z})$ \hfill$(1)$\\
$\Leftrightarrow (x^2+8x)(x^2+8x+7)=y^2.$\\
Đặt $x^2+8x=k$ với $k\in\mathbb{Z}$, ta có $k(k+7)=y^2$ \hfill$(2)$
\begin{itemize}
  \item Xét $y=0$, từ $(1)$ ta được $x_1=0;x_2=-1;x_3=-7;x_4=-8$.
  \item Xét $y>0$. Giải $(2)$ được $k_1=9;k_2=-16$.\\
  Với $k=9$ ta có $x^2+8x=9$ được $x_5=1;x_6=-9$.\\
  Với $k=-16$ ta có $x^2+8x=-16$ được $x_7=-4.$
\end{itemize}
\emph{Đáp số:} $0;-1;-7;-8;1;-9;-4.$
\item \emph{Cách 1.} Giả sử $y^2=x^4+x^3+x^2+x+1$ với $y\in \mathbb{N}$ \hfill$(1)$
\begin{align*}
(1)&\Leftrightarrow 4y^2=4x^2+4x^3+4x^2+4x+4\\
   &\Leftrightarrow (2y)^2=(2x^2+x)^2+3x^2+4x+4\\
   &\Leftrightarrow (2y)^2=(2x^2+x)^2+2x^2+(x+2)^2.
\end{align*}
Suy ra $(2y)^2>(2x^2+x)^2$, nên $(2y)^2 \geq (2x^2+x+1)^2$. Do đó
\begin{center}
$4x^4+4x^3+4x^2+4x+4 \geq 4x^4+x^2+1+4x^3+4x^2+2x$
\end{center}
$\Leftrightarrow 0\geq x^2+2x-3 \Leftrightarrow 0 \geq (x+1)(x-3) \Leftrightarrow -1 \leq x \leq 3$. Mà $x\in \mathbb{Z}$ nên $x\in \{-1;0;1;2;3\}.$
\begin{itemize}
  \item Thay $x=-1$ vào $(1)$ được $y^2=1=1^2.$
  \item Thay $x=0$ vào $(1)$ được $y^2=1=1^2.$
  \item Thay $x=2$ vào $(1)$ được $y^2=5$, loại.
  \item Thay $x=2$ vào $(1)$ được $y^2=31$, loại.
  \item Thay $x=3$ vào $(1)$ được $y^2=121=11^2.$
\end{itemize}
\emph{Đáp số:} $-1;0;3$.\\
\emph{Cách 2.}
Giả sử $x^4+x^3+x^2+x+1=y^2$ với $y\in \mathbb{N}$ \hfill$(2)$\\
Từ $(2)$ ta có:
\begin{align*}
4y^2&=4x^4+4x^3+x^2+4x^2+4x+4\\
    &=(2x^2+x)^2+3x^2+4x+4\\
    &=(2x^2+x)^2+2x^2+(x+2)^2>(2x^2+x)^2 \hfill(3)
\end{align*}
Đặt $A=2x^2+x$, ta sẽ chứng minh $4y^2\leq (A+2)^2.$\\
Xét $(A+2)^2-4y^2=(2x^2+x+2)^2-4y^2$\\
$=(2x^2+x)^2+4(2x^2+x)+4-(2x^2+x)^2-3x^2-4x-4$\\
$=8x^2+4x-3x^2-4x=5x^2$ \hfill$(4)$
\begin{itemize}
  \item Nếu $x=0$ thì $A=0$ và từ $(2)$ có $y^2=1$.
  \item Nếu $x\neq 0$ thì từ $(4)$ suy ra $(A+2)^2>4y^2$. Kết hợp với $(3)$ được
  \begin{center}
    $A^2<4y^2<(A+2)^2.$
  \end{center}
\end{itemize}
Suy ra $(2y)^2=(A+1)^2 \Leftrightarrow 4y^2=(2x^2+x+1)^2$\\
$\Leftrightarrow (2x^2+x)^2+3x^2+4x+4=(2x^2+x)^2+2(2x^2+x)+1$\\
$\Leftrightarrow 3x^2+4x+4=4x^2+2x+1$
$\Leftrightarrow x^2-2x-3=0.$ Ta được $x_1=3;x_2=-1$.\\
\emph{Đáp số:} $x=0;x=3;x=-1.$\\
Biểu thức đã cho theo thứ tự bằng $1;121;1.$
\end{enumerate}
}

a)&\quad x^{2}-8xy^{3}+32y^{6}=16;

\loigiai{

  Do đó $y\in \{0;-1;1\}$.\\
  \emph{Đáp số:} Nghiệm $(x;y)$ là $(4;0),(-4;0),(-4;-1),(4;1)$.

  \item Trường hợp $y^3-1=0$ cho $y=1;x=0$.\\

\end{itemize}
\end{enumerate}
