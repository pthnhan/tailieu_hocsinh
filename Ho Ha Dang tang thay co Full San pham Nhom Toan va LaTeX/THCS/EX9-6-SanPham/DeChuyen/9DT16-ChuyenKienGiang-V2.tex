\begin{ex}%[Đề thi vào 10, Chuyên Kiên Giang, 2016,V2]%[9D1K8]
    Rút gọn biểu thức $P= \dfrac{\sqrt{x}+1}{\sqrt{y}+1 +\sqrt{\sqrt{y}+1}} + \dfrac{\sqrt{x}+1 -\sqrt{\sqrt{y}+1}}{\sqrt{y}+1-\sqrt{\sqrt{y}+1}}$ với $x , y >0.$
\loigiai
    {
    Đặt $a = \sqrt{x}+1 > 0, b= \sqrt{\sqrt{y}+1} > 0, b^2= \sqrt{y}+1$, thay vào biểu thức và thực hiện phép tính rút gọn ta được kết quả $P=\dfrac{2(a-1)}{b^2-1}= \dfrac{2 \sqrt{x}}{\sqrt{y}}$ ($a-1=\sqrt{x}$ và $b^2-1=\sqrt{y}$).
    }
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Kiên Giang, 2016,V2]%[9D4K6]
Chứng minh rằng với mọi giá trị của tham số $m$ phương trình $6x^2-2(m-2)x-m-1=0$ luôn có hai nghiệm $x_1,x_2$ và giá trị của biểu thức $Q= (2x_1+1)(2x_2+1)$ không phụ thuộc vào giá trị của $m$.
\loigiai
    {Tính $\Delta'=(m+1)^2+9 > 0$. Ta có $S = \dfrac{-b}{a} =\dfrac{m-2}{3}, P= \dfrac{c}{a}= \dfrac{-m-1}{6}$. Thực hiện phép tính trên $Q$ ta được $Q=4P+2S+1 = \dfrac{-5}{3}$.
    
    }
\end{ex}

    {
    	\hfill
    	\begin{enumerate}
    		\item Tách các mẫu thành nhân tử rồi đưa về hiệu: $$\dfrac{1}{x^2+x-2}=\dfrac{1}{(x-1)(x+2)} =\dfrac{1}{3} \left(\dfrac{1}{x-1}-\dfrac{1}{x+2}\right),$$
    		$$\dfrac{1}{x^2+x-20}=\dfrac{1}{(x-4)(X+5)}=\dfrac{1}{9}\left(\dfrac{1}{x-4}-\dfrac{1}{x+5}\right).$$
    		
    		Thay vào phương trình và giản ước các hạng tử đối nhau, thu gọn ta được phương trình bậc hai $9x^2+9x-116=0 \Leftrightarrow \hoac{&x=3,82\\&x=-4,8.}$
    		\item Do $a,b,c >0$, đặt $\sqrt{a}=x,\sqrt{b}=y,\sqrt{c}=z$. Suy ra $\sqrt{ab}=xy,a=x^2,b=y^2$. Thay vào biểu thức và rút gọn ta được $2014 \left(x^2+y^2+z^2-xy-xz-yz\right) \ge 0.$ Biểu thức luôn đúng vì $x^2+y^2+z^2 \ge xy+yz+xz$.
    	\end{enumerate}
   
    }

\begin{ex}%[Đề thi vào 10, Chuyên Kiên Giang, 2016,V2]%[9H3G7]
	Cho $\triangle ABC$ nhọn ($AB < AC$) nội tiếp đường tròn ($O$). Các đường cao $AD,BE,CF$ cắt nhau tại $H$. Đường thẳng $EF$ cắt đường thẳng $BC$ tại $M$ và đường thẳng $AM$ cắt đường tròn ($O$) tại điểm $L$ ($L \ne A$). Chứng minh rằng: 
	\begin{enumerate}
		\item $H$ là tâm đường tròn nội tiếp $\triangle DEF.$
		\item $L$ thuộc đường tròn đường kính $AH.$
		\item $\dfrac{MF}{ME}= \dfrac{AC^2.DB}{AB^2.DC}.$
	\end{enumerate}
	\loigiai{
		
			\begin{enumerate}
				\immini{\item Dùng tứ giác nội tiếp ($BFHD$, $CEHD$, $BFEC$) chứng minh $AD$ là tia phân giác của góc $FDE$, $CF$ là tia phân giác của góc $EFD$. Suy ra $H$ là tâm đường tròn nội tiếp $\triangle DEF.$
				
				\item Do tứ giác $ALBC$ nội tiếp ($O$) nên ta chứng minh được $\triangle MLB \sim \triangle MCA$. Suy ra $MB.MC=ML.MA$ (1). Tương tự tứ giác $BFEC$ nội tiếp nên ta có $MB.MC=MF.ME$ (2). Từ (1) và (2) ta có $MF.ME=ML.MA$ hay $\triangle MFL \sim \triangle MAE$ nên tứ giác $ALFE$ nội tiếp đường tròn đường kính $AH$ hay $L$ thuộc đường tròn đường kính $AH$.
				
			}{
			\begin{tikzpicture}[scale=0.8]
			\tkzDefPoints{0/0/B,1/3/A,3/0/C}
			\tkzDrawPolygon(A,B,C)

			\tkzDrawAltitude(B,A)(C)
			\tkzGetPoint{F}
			\tkzDrawAltitude(B,C)(A)
			\tkzGetPoint{D}
			\tkzDrawCircle[circum](A,B,C)\tkzGetPoint{O}
			\tkzInterLL(A,D)(B,E)\tkzGetPoint{H}
			\tkzInterLL(E,F)(B,C)\tkzGetPoint{M}
			\tkzInterLC(A,M)(O,A)\tkzGetFirstPoint{L}
			\tkzDrawSegments(M,A M,B M,E L,F L,B L,H F,D)
			\tkzDrawPoints(B,C,D,F,H,M,A,L,E)
			\tkzLabelPoints[above](A)
			\tkzLabelPoints[above left](L)

				\item Do $AD$ là phân giác trong của góc $FDE.$ Mà $AD \perp BC$ nên $DM$ là phân giác ngoài của $\triangle DEF$. Ta có $\dfrac{MF}{ME}=\dfrac{DF}{DE}$ (*). Vì $\triangle BFD \sim \triangle BCA$ nên ta tính được $DF=\dfrac{AC.BD}{AB}$. Tương tự $\triangle CED \sim \triangle CBA$ nên ta có $DE =\dfrac{AB.DC}{AC}$. Thay $DE,DF$ vào (*) ta được $\dfrac{MF}{ME}=\dfrac{AC^2.DB}{AB^2.DC}.$
			\end{enumerate}
		}
