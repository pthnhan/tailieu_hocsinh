\begin{ex}%[Đề thi HSG 9, tỉnh Sơn La, năm 2018]%[Phạm Tuấn, Dự án EX-9G-18]%[9D0K1]%[9D0K3]
\begin{enumerate}
\item[1)] Cho biểu thức $P=\dfrac{x^2+\sqrt{x}}{x-\sqrt{x}+1}-\dfrac{2x+\sqrt{x}}{\sqrt{x}}+1$ 
\begin{enumerate}
\item[a)] Rút gọn $P$.
\item[b)] Biết $0 < x < 1$, hãy so sánh $P$ với $\left | P \right |$.
\item[c)] Tìm giá trị nhỏ nhất của $P$. 
\end{enumerate}
\item[2)] Cho $f(x)=\left ( 2x^3-21x-29 \right )^{2018}$. Tính $f(x)$ tại $x=\sqrt[3]{7+\sqrt{\dfrac{49}{8}}}+\sqrt[3]{7-\sqrt{\dfrac{49}{8}}}$. 
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[1)] 
\begin{enumerate}
\item[a)] Điều kiện xác định: $x \geq 0$.
\begin{align*}
P&=\dfrac{x^2+\sqrt{x}}{x-\sqrt{x}+1}-\dfrac{2x+\sqrt{x}}{\sqrt{x}}+1= \dfrac{x^2+\sqrt{x}}{x-\sqrt{x}+1} -2\sqrt{x}  \\
&=\dfrac{x^2+\sqrt{x}-2x\sqrt{x}+2x-2\sqrt{x}}{x-\sqrt{x}+1} = \dfrac{x(x-\sqrt{x}+1) - (x\sqrt{x}+\sqrt{x}-x)}{x-\sqrt{x}+1}\\
&=x-\sqrt{x}.
\end{align*} 
\item[b)] $0<x<1 \Rightarrow x-\sqrt{x} <0 \Rightarrow |P|=-P$.
\item[c)] $P= \left ( \sqrt{x}-\dfrac{1}{2}\right )^2- \dfrac{1}{4} \geq \dfrac{1}{4}$. Suy ra $P_{\min} = -\dfrac{1}{4}$ khi $x=\dfrac{1}{4}$. 
\end{enumerate}
\item[2)] Ta có 
\begin{align*}
x^3 &= 7+\sqrt{\dfrac{49}{8}} +7-\sqrt{\dfrac{49}{8}} + 3 \cdot \sqrt[3]{7+\sqrt{\dfrac{49}{8}}} \cdot \sqrt[3]{7-\sqrt{\dfrac{49}{8}}} \left ( \sqrt[3]{7+\sqrt{\dfrac{49}{8}}}+\sqrt[3]{7-\sqrt{\dfrac{49}{8}}}\right ) \\
&=14+\dfrac{21x}{2}
\end{align*}
Suy ra $2x^3-21x-29=-1$ và $f(x)=1$.
\end{enumerate}
}
\end{ex}


\begin{ex}%[Đề thi HSG 9, tỉnh Sơn La, năm 2018]%[Phạm Tuấn, Dự án EX-9G-18]%[9D0G2]
\begin{enumerate}
\item[a)] Giải phương trình $\sqrt{x-1}+\sqrt{x^3+x^2+x+1}=1+\sqrt{x^4-1}$.
\item[b)] Giải hệ phương trình  $\heva{& \sqrt{3x}\left ( 1+\frac{1}{x+y}\right )=2\\&\sqrt{7y}\left ( 1-\frac{1}{x+y} \right ) =4\sqrt{2}.}$ 
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a)] Điều kiện xác định: $x \geq 1$. 
\begin{align*}
&\sqrt{x-1}+\sqrt{x^3+x^2+x+1}=1+\sqrt{x^4-1} \\
\Leftrightarrow &\left (\sqrt{x-1}-1 \right ) \left (\sqrt{x^3+x^2+x+1}-1\right ) =0 \\
\Leftrightarrow &\hoac{&\sqrt{x-1}-1=0\\&\sqrt{x^3+x^2+x+1}-1=0} \\
\Leftrightarrow &\hoac{&x=2\\&x=0~ \textrm{(loại)}}
\end{align*}
Vậy nghiệm của phương trình là $x=2$. 
\item[b)] Điều kiện xác định: $x,y \geq 0$ và $x+y >1$.  
$$\heva{& \sqrt{3x}\left ( 1+\frac{1}{x+y}\right )=2\\&\sqrt{7y}\left ( 1-\frac{1}{x+y} \right ) =4\sqrt{2}} \Leftrightarrow \heva{&\dfrac{1}{\sqrt{3x}} + \dfrac{2\sqrt{2}}{\sqrt{7y}} =1~(1)\\& \dfrac{1}{\sqrt{3x}} - \dfrac{2\sqrt{2}}{\sqrt{7y}} =\dfrac{1}{x+y} ~(2)}$$  
Nhân tương ứng hai vế của $(1)$ và $(2)$ ta  nhận được $$\dfrac{1}{3x} - \dfrac{8}{7y}=\dfrac{1}{x+y} \Leftrightarrow 24x^2 +38xy-7y^2=0 \Leftrightarrow \hoac{&\dfrac{x}{y}=\dfrac{1}{6}\\&\dfrac{x}{y}=-\dfrac{7}{4} ~ \textrm{(loại)}}  \Leftrightarrow 6x=y.$$ 
Thay $y=6x$ vào phương trình $\sqrt{3x}\left ( 1+\dfrac{1}{x+y}\right )=2$, ta nhận được nghiệm của hệ phương trình $$(x;y)=\left ( \dfrac{11+4\sqrt{7}}{21}; \dfrac{22+8\sqrt{7}}{7}\right ).$$
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Sơn La, năm 2018]%[Phạm Tuấn, Dự án EX-9G-18]%[9D0G7]
Cho $x, y, z > 0$ thỏa mãn $\dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=3$. Chứng minh rằng $$\dfrac{x}{x^4+1+2xy}+\dfrac{y}{y^4+1+2yz}+\dfrac{z}{z^4+1+2zx}\leq \dfrac{3}{4}$$ 
\loigiai{
Ta có $x^4+1 \geq 2x^2$ nên $x^4+1+2xy \geq 2x(x+y)$, suy ra $\dfrac{x}{x^4+1+2xy} \leq \dfrac{1}{2(x+y)}\leq \dfrac{1}{8} \left (\dfrac{1}{x}+\dfrac{1}{y}\right )$. \\
Chứng minh tương tự ta có $\dfrac{y}{y^4+1+2yz} \leq \dfrac{1}{8} \left (\dfrac{1}{y}+\dfrac{1}{z}\right )$ và $\dfrac{z}{z^4+1+2zx} \leq \dfrac{1}{8} \left (\dfrac{1}{z}+\dfrac{1}{x}\right )$. \\
Suy ra  $\dfrac{x}{x^4+1+2xy}+\dfrac{y}{y^4+1+2yz}+\dfrac{z}{z^4+1+2zx}\leq \dfrac{3}{4}$.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Sơn La, năm 2018]%[Phạm Tuấn, Dự án EX-9G-18]%[9H0G1]
Cho tam giác $ABC$ nhọn trực tâm $H$, trên đoạn $BH$ lấy điểm $M$ và trên đoạn $CH$ lấy điểm $N$ sao cho $\widehat{AMC}=\widehat{ANB}=90^{0}$. Chứng minh rằng $AM = AN$.
\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzInit[xmin=-1,ymin=-1.5,xmax=7,ymax=4.5]
\tkzClip
\tkzDefPoints{2/3.5/A,0/-1/B,6/-1/C}
\tkzOrthoCenter(A,B,C) \tkzGetPoint{H}
\tkzDefMidPoint(A,B) \tkzGetPoint{I}
\tkzDefMidPoint(A,C) \tkzGetPoint{J}
\tkzInterLC(H,C)(I,A) \tkzGetPoints{N}{N'}
\tkzInterLC(H,B)(J,A) \tkzGetPoints{M}{M'}
\tkzInterLL(H,C)(A,B) \tkzGetPoint{E}
\tkzInterLL(H,B)(A,C) \tkzGetPoint{F}
\tkzDrawSegments(A,B B,C C,A C,E B,F A,M A,N M,C B,N E,F)
\tkzMarkRightAngle(A,M,C)
\tkzMarkRightAngle(B,N,A)
\tkzMarkRightAngle(C,F,B)
\tkzMarkRightAngle(C,E,B)
\tkzDrawPoints[fill,size=6pt](A,B,C,E,F,M,N,H)
\tkzLabelPoints[above](A)
\tkzLabelPoints[below](B,C)
\tkzLabelPoints[above right](N,F)
\tkzLabelPoints[above left](E)
\tkzLabelPoints[above left=-5pt](M)
\tkzLabelPoints[below](H)
\end{tikzpicture}
\end{center}
Tứ giác $BEFC$ nội tiếp  nên $AE \cdot AB = AF \cdot AC$ \hfill (1) \\
Tam giác $ABN$ và $ACM$ vuông  nên $AE \cdot AB = AN^2$ và $AF \cdot AC = AM^2$ \hfill (2) \\
Từ (1) và (2) suy ra $AM^2=AN^2 \Leftrightarrow AM=AN$. 
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Sơn La, năm 2018]%[Phạm Tuấn, Dự án EX-9G-18]%[9H0G8]
Trong một hình vuông có cạnh bằng $6$, ta có một số các đường tròn có tổng chu vi bằng $2018$. Chứng minh rằng tồn tại một đường thẳng cắt ít nhất $108$ đường tròn trong chúng.

}
