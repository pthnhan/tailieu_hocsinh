\begin{ex}%[Đề thi Toán 9 Học sinh gỏi năm học 2012 - 2013, Kiên Giang]%[9D2B2]%[9D5G1]%[8D1K5]
\begin{enumerate}
\item Tìm $m$ để hàm số $y=(m^{2}-2m)x+m^{2}-1$ nghịch biến và đồ thị của nó cắt trục tung tại điểm có tung độ bằng $3$.
\item Tìm giá trị nhỏ nhất của $M=5x^{2}+y^{2}+z^{2}-4x-2xy-z-1$.
\item Cho $x+y=-5$ và $x^{2}+y^{2}=11$. Tính $x^{3}+y^{3}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Hàm số $y=(m^{2}-2m)x+m^{2}-1$ nghịch biến khi và chỉ khi\\
$m^{2}-2m<0\Leftrightarrow \hoac{&\heva{& m>0\\ &m-2<0}\\ &\heva{& m<0\\ &m-2>0}}\Leftrightarrow \hoac{&\heva{& m>0\\ &m<2}\\ &\heva{& m<0\\ &m>2}\ (\mbox{Loại})}\Leftrightarrow 0<m<2$ (1).\\
Đồ thị hàm số cắt trục tung tại điểm có tung độ bằng $3\Leftrightarrow m^{2}-1=3\Leftrightarrow m=\pm 2$ (2).\\
Từ (1) và (2) ta thấy $m\in\varnothing$.
\item $M=5x^{2}+y^{2}+z^{2}-4x-2xy-z-1=(x-y)^{2}+(2x-1)^{2}+\left(z-\dfrac{1}{2}\right)^{2}-\dfrac{9}{4}\geq -\dfrac{9}{4}$.\\
Dấu bằng xảy ra khi và chỉ khi $x=y=z=\dfrac{1}{2}$.
\item $x^{3}+y^{3}=(x+y)(x^{2}+y^{2}-xy)=-5(11-xy)$.\\
Mà $x+y=-5\Rightarrow x^{2}+y^{2}+2xy=25\Rightarrow xy=7\Rightarrow x^{3}+y^{3}=-20$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học sinh gỏi năm học 2012 - 2013, Kiên Giang]%[9D1K8]%[8D1G9]
\begin{enumerate}
\item Rút gọn $A=\dfrac{x^{2}+5x+6+x\sqrt{9-x^{2}}}{3x-x^{2}+(x+2)\sqrt{9-x^{2}}}:2\sqrt{1+\dfrac{2x}{3-x}}$.
\item Cho $a$, $b$, $c$ thỏa mãn $\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}=\dfrac{1}{a+b+c}$. Tính giá trị của biểu thức\\
$Q=(a^{27}+b^{27})(b^{41}+c^{41})(c^{2013}+a^{2013})$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Điều kiện xác định $-3<x<3$.\\
$A=\dfrac{(x+3)(x+2)+x\sqrt{3+x}\sqrt{3-x}}{x(3-x)+(x+2)\sqrt{3+x}\sqrt{3-x}}:2\sqrt{\dfrac{3+x}{3-x}}$\\
$=\dfrac{\sqrt{3+x}}{\sqrt{3-x}}.\dfrac{(x+2)\sqrt{3+x}+x\sqrt{3-x}}{x\sqrt{3-x}+(x+2)\sqrt{3+x}}:2\sqrt{\dfrac{3+x}{3-x}}=\dfrac{1}{2}$.
\item $\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}=\dfrac{1}{a+b+c}\Rightarrow \dfrac{1}{a}+\dfrac{1}{b}=\dfrac{1}{a+b+c}-\dfrac{1}{c}\Rightarrow \dfrac{a+b}{ab}=-\dfrac{a+b}{c(a+b+c)}$\\
$\Rightarrow (a+b)[c(a+b+c)+ab]=0\Rightarrow (a+b)[c(a+c)+bc+ab]=0$\\
$\Rightarrow (a+b)(b+c)(c+a)=0\Rightarrow \hoac{&a=-b\\ &b=-c\\ &c=-a}$\\
Thay vào $Q$ ta được $Q=0$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học sinh gỏi năm học 2012 - 2013, Kiên Giang]%[8H3K5]
Cho hình thang $ABCD$ có đáy lớn là $CD$. Qua $A$ vẽ $AK\parallel BC$ ($K\in CD$) và qua $B$ kẻ $BI\parallel AD$ ($I\in CD$); $BI$ cắt $AC$ tại $F$, $AK$ cắt $BD$ tại $E$.
\begin{enumerate}
\item Chứng minh $KD=CI$ và $EF\parallel AB$.
\item Chứng minh $AB^{2}=CD.EF$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[line cap=round,line join=round,>=stealth,x=1.0cm,y=1.0cm, scale=0.7, thick]
\tkzInit[xmax=12,xmin=-2,ymax=8,ymin=-1]
\tkzDefPoint(0,0){D}
\tkzDefShiftPoint[D](0:7){C}
\tkzDefShiftPoint[D](75:3){A}
\tkzDefShiftPoint[A](0:4){B}
\tkzDefLine[parallel=through A](B,C)\tkzGetPoint{a}
\tkzInterLL(A,a)(C,D)\tkzGetPoint{K}
\tkzDefLine[parallel=through B](A,D)\tkzGetPoint{b}
\tkzInterLL(B,b)(C,D)\tkzGetPoint{I}
\tkzInterLL(A,K)(B,D)\tkzGetPoint{E}
\tkzInterLL(B,b)(C,A)\tkzGetPoint{F}
\tkzDrawSegments(A,B B,C C,D D,A A,C B,D A,K B,I E,F)
\pgfresetboundingbox
\tkzLabelPoints[above](A,B)
\tkzLabelPoints[left](E,D)
\tkzLabelPoints[right](C,F)
\tkzLabelPoints[below](I,K)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Ta có $ABID$ và $ABCK$ là các hình bình hành $\Rightarrow DI=CK=AB$.
\begin{itemize}
\item Nếu $I$ nằm giữa $D$ và $K\Rightarrow K$ nằm giữa $I$ và $C$.\\
Ta có $DK=DI+IK=CK+IK=CI$.
\item Nếu $K$ nằm giữa $D$ và $I\Rightarrow I$ nằm giữa $K$ và $C$.\\
Ta có $DK=DI-IK=CK-IK=CI$.
\end{itemize}
Xét $\Delta AEB$ và $\Delta KED$ có\\
$\widehat{EAB}=\widehat{EKD}$, $\widehat{DEK}=\widehat{AEB}\Rightarrow \Delta AEB\backsim\Delta KED\Rightarrow \dfrac{AE}{EK}=\dfrac{AB}{KD}$.\\
Tương tự $\Delta AFB\backsim\Delta CFI\Rightarrow \dfrac{AF}{FC}=\dfrac{AB}{CI}$.\\
Mà $KD=CI\Rightarrow \dfrac{AE}{EK}=\dfrac{AF}{FC}\Rightarrow EF\parallel KC$ (định lí Talet đảo).
\item Ta có $\Delta KED\backsim \Delta AEB\Rightarrow \dfrac{DK}{AB}=\dfrac{DE}{EB}\Rightarrow \dfrac{DK+AB}{AB}=\dfrac{DE+EB}{EB}$\\
$\Rightarrow \dfrac{DK+KC}{AB}=\dfrac{BD}{EB}\Rightarrow \dfrac{DC}{AB}=\dfrac{BD}{EB}$ (1).\\
Lại có $EF\parallel DI\Rightarrow \dfrac{BD}{EB}=\dfrac{DI}{EF}\Rightarrow \dfrac{BD}{EB}=\dfrac{AB}{EF}$ (2).\\
Từ (1) và (2) ta có $\dfrac{DC}{AB}=\dfrac{AB}{EF}\Rightarrow AB^{2}=CD.EF$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học sinh gỏi năm học 2012 - 2013, Kiên Giang]%[9H3B3]%[9H2K2]%[8H2G3]
Cho tam giác đều $ABC$ nội tiếp đường tròn $(O;R)$. Gọi $M$ là điểm di động trên cung nhỏ $BC$ của đường tròn đó.
\begin{enumerate}
\item Chứng minh $BM+MC=MA$.
\item Xác định vị trí của điểm $M$ để tổng $MA+MB+MC$ đạt giá trị lớn nhất.
\item Gọi $H$, $K$, $Q$ lần lượt là hình chiếu của điểm $M$ lên $AB$, $BC$, $CA$. Đặt diện tích tam giác $ABC$ là $S$ và diện tích tam giác $MBC$ là $S'$. Chứng minh rằng $MH+MK+MQ=\dfrac{2\sqrt{3}(S+2S')}{3R}$ khi $M$ chạy trên cung nhỏ $BC$.
\end{enumerate}

\tkzDefPointBy[rotation=center O angle 35](B)\tkzGetPoint{M}
\tkzDefPointBy[rotation=center B angle 60](M)\tkzGetPoint{D}
\tkzDefPointBy[projection=onto A--B](M)\tkzGetPoint{H}
\tkzDefPointBy[projection=onto C--B](M)\tkzGetPoint{K}
\tkzDefPointBy[projection=onto C--A](M)\tkzGetPoint{Q}
\tkzDefPointBy[projection=onto C--B](A)\tkzGetPoint{N}
\tkzDrawSegments(A,H B,C C,A M,H M,K M,Q M,A B,D M,B M,C A,N)

\tkzLabelPoints[left](B)

\begin{enumerate}
\item Trên tia $MA$ lấy điểm $D$ sao cho $MD=MB\Rightarrow \Delta MBD$ cân tại $M$. Mà $\widehat{BMD}=60^{\circ}$ (do chắn cung $\wideparen{AB}$ và $\Delta ABC$ đều) nên $\Delta BMD$ đều $\Rightarrow MB=BD$.\\
Lại có $\widehat{MBC}+\widehat{CBD}=60^{\circ}=\widehat{CBD}+\widehat{DBA}\Rightarrow \widehat{MBC}=\widehat{DBA}$.\\
Xét $\Delta MBC$ và $\Delta DBA$ có $MB=DB$, $\widehat{MBC}=\widehat{DBA}$, $BC=AB$ nên $\Delta MBC=\Delta DBA$\\
$\Rightarrow MC=DA\Rightarrow MA=MD+DA=MB+MC$.
\item Ta có $MA+MB+MC=2MA\leq 4R$. Dấu bằng xảy ra khi và chỉ khi $AM$ là đường kính của đường tròn $(O;R)\Leftrightarrow M$ là điểm chính giữa cung nhỏ $\wideparen{BC}$.
\item Gọi $N$ là trung điểm $BC\Rightarrow OA=\dfrac{2}{3}AN\Rightarrow AN=\dfrac{3R}{2}$. Xét tam giác $ANB$ có\\
$AB^{2}=AN^{2}+NB^{2}=\dfrac{9R^{2}}{4}+\dfrac{AB^{2}}{4}\Rightarrow AB^{2}=3R^{2}\Rightarrow AB=R\sqrt{3}$.\\
Mà $\dfrac{MH.AB}{2}+\dfrac{MK.BC}{2}+\dfrac{MQ.AC}{2}=S_{\Delta MAB}+S_{\Delta MBC}+S_{\Delta MAC}$\\
$\Rightarrow AB(MH+MK+MQ)=2(S+S')\Rightarrow MH+MK+MQ=\dfrac{2\sqrt{3}(S+2S')}{3R}$.
\end{enumerate}
}
\end{ex}
