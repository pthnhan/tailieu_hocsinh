\begin{bt}%[HK2 (2016-2017), THPT Long Mỹ, Hậu Giang]%[1D4B2-3]%[Đoàn Nhật Thiện]
	Tính giới hạn sau: $\lim \limits_{x\to 2} \dfrac{x^2-4}{2x^2-5x+2}$.
	\loigiai{
		\begin{align*}
		&\lim \limits_{x\to 2} \dfrac{x^2-4}{2x^2-5x+2}\\
		&=\lim \limits_{x\to 2} \dfrac{(x-2)(x+2)}{(x-2)(2x-1)}\\
		&=\lim \limits_{x\to 2} \dfrac{x+2}{2x-1}\\
		&=\dfrac{4}{3}.
		\end{align*}
	}
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Long Mỹ, Hậu Giang]%[1D4B3-3]%[Đoàn Nhật Thiện]
	Cho hàm số $f(x)=\left \{ \begin{aligned} &\dfrac{\sqrt{x+7}-\sqrt{3x+3}}{x-2}&\text{ khi }& x<2\\ 
	&\dfrac{2mx-3m+2}{x-3}&\text{ khi }& x\ge 2\end{aligned} \right.$. Tìm $m$ để hàm số đã cho liên tục tại điểm $x_0=2$.
	\loigiai{
		Ta có \begin{align*}
		&\lim \limits_{x\to 2^-} \dfrac{\sqrt{x+7}-\sqrt{3x+3}}{x-2}\\
		&=\lim \limits_{x\to 2^-} \dfrac{(x+7)-(3x+3)}{(x-2)(\sqrt{x+7}+\sqrt{3x+3})}\\
		&=\lim \limits_{x\to 2^-} \dfrac{-2x+4}{(x-2)\left(\sqrt{x+7}+\sqrt{3x+3}\right)}\\
		&=\lim \limits_{x\to 2^-} \dfrac{-2}{\sqrt{x+7}+\sqrt{3x+3}}=-\dfrac{1}{3}.
		\end{align*}
		Ta có $\lim \limits_{x\to 2^+}f(x)=f(2)=-m-2$.\\
		Để hàm số liên tục tại điểm $x_0=2\Leftrightarrow \lim \limits_{x\to 2^+}f(x)=\lim 
		\limits_{x\to 2^-}f(x)=f(2)\Leftrightarrow -m-2=-\dfrac{1}{3}\Leftrightarrow m=-\dfrac{5}{3}$.
	}
\end{bt}

		Khi đó góc tạo bởi $SD$ và mặt phẳng $(SAB)$ là góc $\widehat{DSA}$.\\
		Ta có $\tan \widehat{DSA}=\dfrac{AD}{SA}=\dfrac{a}{2a}=\dfrac{1}{2}\Rightarrow \widehat{DSA}=\arctan \dfrac{1}{2}$.
		\begin{enumerate}
			\item[c)] Gọi $K$ là hình chiếu của $A$ lên $SB$. Ta có $\left \{ \begin{aligned} &BC\perp AB\\&BC\perp SA\end{aligned} \right.\Rightarrow BC\perp (SAB)\Rightarrow BC\perp AK.$\\
			Ta có $\left \{ \begin{aligned} &AK\perp SB\\ &AK\perp BC\end{aligned}\right. \Rightarrow AK\perp (SBC)\Rightarrow AK=d(A,(SBC))$.\\
			Ta có $AD\parallel BC\Rightarrow AD\parallel (SBC)\Rightarrow d(D,(SBC))=d(A,(SBC))=AK$.\\
			$\dfrac{1}{AK^2}=\dfrac{1}{SA^2}+\dfrac{1}{AB^2}\Rightarrow AK^2=\dfrac{SA^2\cdot AB^2}{SA^2+AB^2}=\dfrac{(2a)^2\cdot a^2}{(2a)^2+a^2}=\dfrac{4a^2}{5}\Rightarrow AK=\dfrac{2a\sqrt{5}}{5}.$
