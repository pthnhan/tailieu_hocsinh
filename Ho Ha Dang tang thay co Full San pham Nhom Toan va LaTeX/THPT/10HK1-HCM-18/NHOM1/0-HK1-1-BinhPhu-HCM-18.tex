\begin{bt}%[Đề HK1, Bình Phú, Hồ Chí Minh 17-18]%[Đinh Thanh Hoàng, 10EX-HK1]%[0D3B3-1]
	Giải và biện luận phương trình sau theo tham số $m$: $(m-1)x=m^2-1$.
\loigiai{
	Biện luận
	\begin{itemize}
		\item Nếu $m \neq 1 \Leftrightarrow m-1 \neq 0$ thì phương trình đã cho có nghiệm duy nhất $$x=\dfrac{m^2-1}{m-1}=\dfrac{(m-1)(m+1)}{m-1}=m+1.$$
		\item Nếu $m=1 \Leftrightarrow m-1 = 0$ thì phương trình đã cho trở thành $0x=0$ $\Rightarrow$ phương trình đã cho nghiệm đúng với mọi $x \in \mathbb{R}$.
	\end{itemize}
	Kết luận
	\begin{itemize}
		\item Nếu $m \neq 1$ thì phương trình đã cho có nghiệm duy nhất $x=m+1$.
		\item Nếu $m=1$ thì phương trình đã cho nghiệm đúng với mọi $x \in \mathbb{R}$.
	\end{itemize}
}
\end{bt}

\begin{bt}%[Đề HK1, Bình Phú, Hồ Chí Minh 17-18]%[Đinh Thanh Hoàng, 10EX-HK1]%[0D3B2-4]%[0D3B2-2]%[0D3B2-4]
	Giải các phương trình sau:
	\begin{multicols}{3}
		\begin{enumerate}
			\item $\sqrt{x^2-x+5}=\sqrt{9-x}$.
			\item $3|x-1|=2x^2-4x$.
			\item $\sqrt{4-x}-1=\sqrt{x+1}$.
		\end{enumerate}
	\end{multicols}
\loigiai{
	\begin{enumerate}
		\item Ta có $\sqrt{x^2-x+5}=\sqrt{9-x} \Leftrightarrow \heva{&9-x \geq 0 \\ &x^2-x+5=9-x} \Leftrightarrow \heva{&x\leq 9 \\ &x^2=4} \Leftrightarrow x=\pm 2$.\\
				Vậy phương trình đã cho có tập nghiệm $S=\lbrace -2;2\rbrace$.
		\item \begin{itemize}
					\item Với $x \geq 1 \Leftrightarrow x-1 \geq 0$, phương trình đã cho trở thành
							$$3(x-1)=2x^2-4x \Leftrightarrow 2x^2-7x+3=0 \Leftrightarrow \hoac{&x=3 &&\mbox{(thỏa mãn)} \\ &x=\dfrac{1}{2} &&\mbox{(không thỏa mãn).}}$$
					\item Với $x<1 \Leftrightarrow x-1<0$, phương trình đã cho trở thành
							$$-3(x-1)=2x^2-4x \Leftrightarrow 2x^2-x-3=0 \Leftrightarrow \hoac{&x=-1 &&\mbox{(thỏa mãn)} \\ &x=\dfrac{3}{2} &&\mbox{(không thỏa mãn).}}$$
				\end{itemize}		
				Vậy phương trình đã cho có tập nghiệm $S=\lbrace -1;3\rbrace$.
		\item Điều kiện xác định của phương trình là $-1 \leq x \leq 4$. Ta có
				\begin{eqnarray*}
					& & \sqrt{4-x}-1=\sqrt{x+1}\\
					&\Leftrightarrow & \sqrt{4-x}=\sqrt{x+1}+1\\
					&\Rightarrow & 4-x=x+2+2\sqrt{x+1}\\
					&\Leftrightarrow & \sqrt{x+1}=1-x\\
					&\Leftrightarrow & \heva{&1-x\geq 0 \\ &x+1=1-2x+x^2}\\
					&\Leftrightarrow & \heva{&x\leq 1 \\ &x^2-3x=0}\\
					&\Leftrightarrow & \heva{&x\leq 1 \\ &\hoac{&x=0 \\ &x=3}}\\
					&\Leftrightarrow &x=0 \,\,\mbox{(thỏa mãn)}.
				\end{eqnarray*}
				Vậy phương trình đã cho có tập nghiệm $S=\lbrace 0\rbrace$.
	\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Bình Phú, Hồ Chí Minh 17-18]%[Đinh Thanh Hoàng, 10EX-HK1]%[0D3B2-6]%[0D3K2-5]
	Cho phương trình $mx^2-2(m+1)x+m+3=0$. Đinh $m$ để phương trình
	\begin{enumerate}
		\item Có nghiệm.
		\item Có hai nghiệm thực $x_1$ và $x_2$ thỏa mãn $x_1^2+x_2^2=4$.
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item \begin{itemize}
					\item Với $m=0$, phương trình đã cho trở thành $-2x+3=0 \Leftrightarrow x=\dfrac{3}{2}$.\\
							Suy ra $m=0$ thỏa mãn.
					\item Với $m \neq 0$, phương trình đã cho là một phương trình bậc hai.\\
							Ta có $\Delta' = (m+1)^2-m(m+3) = -m+1$.\\
							Phương trình đã cho có nghiệm $\Leftrightarrow \heva{&m \neq 0 \\ &-m+1\geq 0} \Leftrightarrow \heva{&m \neq 0 \\ &m \leq 1.}$\\							
				\end{itemize}
				Vậy với $m\leq 1$ thì phương trình đã cho có nghiệm.
		\item Với $m\leq 1$ và $m \neq 0$ thì phương trình đã cho có hai nghiệm $x_1$ và $x_2$.\\
				Theo định lí Vi-ét, ta có $\heva{&x_1+x_2=\dfrac{2(m+1)}{m} \\ &x_1x_2=\dfrac{m+3}{m}.}$\\
				Ta có
				\begin{eqnarray*}
					& & x_1^2+x_2^2=4\\
					&\Leftrightarrow & (x_1+x_2)^2-2x_1x_2=4\\
					&\Leftrightarrow & \dfrac{4(m+1)^2}{m^2}-\dfrac{2(m+3)}{m}=4\\
					&\Leftrightarrow & 2(m^2+2m+1)-m(m+3)=2m^2\\
					&\Leftrightarrow & m^2-m-2=0\\
					&\Leftrightarrow & \hoac{&m=-1 &&\mbox{(thỏa mãn)} \\ &m=2 &&\mbox{(không thỏa mãn).}}					
				\end{eqnarray*}					
				Vậy $m=-1$ là giá trị cần tìm.
	\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Bình Phú, Hồ Chí Minh 17-18]%[Đinh Thanh Hoàng, 10EX-HK1]%[0D4K1-2]
	Cho $a$, $b$ là hai số thực dương. Chứng minh rằng $a+b+\dfrac{2}{a}+\dfrac{2}{b} \geq 4+\sqrt{ab}$. Dấu đẳng thức xảy ra khi nào?
\loigiai{
	Áp dụng bất đẳng thức Cô-si cho hai số dương, ta được
	$$a+b+\dfrac{2}{a}+\dfrac{2}{b} \geq 2\sqrt{ab}+\dfrac{4}{\sqrt{ab}} = \sqrt{ab}+\dfrac{4}{\sqrt{ab}} + \sqrt{ab} \geq 4+\sqrt{ab}.$$
	Đẳng thức xảy ra $\Leftrightarrow \heva{&a=b>0 \\ &\sqrt{ab}=\dfrac{4}{\sqrt{ab}}} \Leftrightarrow \heva{&a=b>0 \\ &ab=4} \Leftrightarrow a=b=2.$
}
\end{bt}

\begin{bt}%[Đề HK1, Bình Phú, Hồ Chí Minh 17-18]%[Đinh Thanh Hoàng, 10EX-HK1]%[0H2B2-2]%[0H1B4-3]%[0H2B2-3]
	Trong mặt phẳng $Oxy$, cho tam giác $ABC$ có $A(-2;-1)$, $B(-1;4)$, $C(3;0)$.
	\begin{enumerate}
		\item Chứng minh tam giác $ABC$ cân. Tính diện tích tam giác $ABC$.
		\item Tìm tọa độ điểm $K$ sao cho $\overrightarrow{AK}+2\overrightarrow{BC}=\overrightarrow{KB}$.
		\item Tìm tọa độ điểm $H$ là trực tâm tam giác $ABC$.
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Ta có $AB=\sqrt{1^2+5^2}=\sqrt{26}$ và $AC=\sqrt{5^2+1^2}=\sqrt{26}$. Suy ra tam giác $ABC$ cân tại $A$.\\
				Gọi $M$ là trung điểm của $BC$, ta có $M(1;2)$ và $AM \perp BC$.\\
				Diện tích tam giác $ABC$ là $S_{ABC}=\dfrac{1}{2}AM\cdot BC=\dfrac{1}{2}\cdot\sqrt{3^2+3^2}\cdot\sqrt{4^2+4^2}=12$.
		\item Gọi $K(x;y)$, ta có $\overrightarrow{AK}=(x+2;y+1)$, $\overrightarrow{BC}=(4;-4)$, $\overrightarrow{KB}=(-1-x;4-y)$.
				$$\overrightarrow{AK}+2\overrightarrow{BC}=\overrightarrow{KB} \Leftrightarrow \heva{&x+2+8=-1-x \\ &y+1-8=4-y} \Leftrightarrow \heva{&x=-\dfrac{11}{2} \\ &y=\dfrac{11}{2}.}$$				
				Vậy $K\left(-\dfrac{11}{2};\dfrac{11}{2}\right)$.
		\item Gọi $H(x;y)$, ta có $\overrightarrow{AH}=(x+2;y+1)$, $\overrightarrow{BC}=(4;-4)$, $\overrightarrow{BH}=(x+1;y-4)$, $\overrightarrow{AC}=(5;1)$.\\
				Vì $H$ là trực tâm tam giác $ABC$ nên ta có
				$$\heva{&\overrightarrow{AH}\cdot\overrightarrow{BC}=0 \\ &\overrightarrow{BH}\cdot\overrightarrow{AC}=0} \Leftrightarrow \heva{&4x+8-4y-4=0 \\ &5x+5+y-4=0} \Leftrightarrow \heva{&x=-\dfrac{1}{3} \\ &y=\dfrac{2}{3}.}$$
				Vậy $H\left(-\dfrac{1}{3};\dfrac{2}{3}\right)$.			
	\end{enumerate}
}
\end{bt}
