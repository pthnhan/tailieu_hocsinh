	\immini{Cho hàm số $y=f(x)$ có bảng biên thiên như hình vẽ. Hàm số $g(x)=f\left(2x^2-\dfrac{5}{2}x-\dfrac{3}{2}\right)$ nghịch biến trên khoảng nào trong các khoảng sau?

	{$\left(-1;\dfrac{1}{4}\right)$ }

	{\True $\left(1;\dfrac{5}{4}\right)$ }

	\loigiai{
		Dựa vào bảng biến thiên, suy ra $f'(x)>0\Leftrightarrow \hoac{&x<-2\\&x>3}$ và $f'(x)<0\Leftrightarrow -2<x<3$.\\
		Ta có $g'(x)=\left(4x-\dfrac{5}{2}\right)f'\left(2x^2-\dfrac{5}{2}x-\dfrac{3}{2}\right)$. Xét $g'(x)<0\Leftrightarrow \hoac{&\heva{&4x-\dfrac{5}{2}>0\\&f'\left(2x^2-\dfrac{5}{2}x-\dfrac{3}{x}\right)<0}\\&\heva{&4x-\dfrac{5}{2}<0\\&f'\left(2x^2-\dfrac{5}{2}x-\dfrac{3}{2}\right)>0}}$.

			\item $\heva{&4x-\dfrac{5}{2}>0\\&f'\left(2x^2-\dfrac{5}{2}x-\dfrac{3}{x}\right)<0}\Leftrightarrow \heva{&x>\dfrac{5}{8}\\&-2<2x^2-\dfrac{5}{2}x-\dfrac{3}{x}<3}\Leftrightarrow 1<x<\dfrac{9}{4}$.
			\item $\heva{&4x-\dfrac{5}{2}<0\\&f'\left(2x^2-\dfrac{5}{2}x-\dfrac{3}{2}\right)>0}\Leftrightarrow \hoac{&\heva{&x<\dfrac{5}{8}\\&2x^2-\dfrac{5}{2}x-\dfrac{3}{2}>3}\\&\heva{&x<\dfrac{5}{8}\\&2x^2-\dfrac{5}{2}x-\dfrac{3}{2}<-2}}\Leftrightarrow \hoac{&x<-1\\&\dfrac{1}{4}<x<\dfrac{5}{8}.}$
