		\begin{enumEX}{2}
			\item Tìm $ x $ biết $ \dfrac{2}{3}x+\dfrac{1}{7}=\dfrac{5}{3} $;
			\item Tìm $ x,\ y $ biết $ \dfrac{x}{5}=\dfrac{y}{3} $ và $ x-y=16 $;
			\item! Cho hàm số $ f(x)=2x-1 $. Tính $ f(2) $ và $ f\left( -\dfrac{1}{2}\right) $.
		\end{enumEX}
	\loigiai{
		\begin{enumerate}
	\item Ta có
	\begin{eqnarray*}
		\dfrac{2}{3}x+\dfrac{1}{7}&=&\dfrac{5}{3}\\
		\dfrac{2}{3}x&=&\dfrac{5}{3}-\dfrac{1}{7}\\
		\dfrac{2}{3}x&=&\dfrac{32}{21}\\
		x&=&\dfrac{32}{21}\cdot \dfrac{3}{2}\\
		x&=&\dfrac{16}{7}.
	\end{eqnarray*}
	\item Ta có $ \dfrac{x}{5}=\dfrac{y}{3}=\dfrac{x-y}{5-3}=8 $. Do đó
	\begin{eqnarray*}
		x&=&5\cdot 8=40.\\
		7&=&3\cdot 8=24.
	\end{eqnarray*}
	\item Ta có $ f(2)=2\cdot 2-1=3 $ và $ f\left( -\dfrac{1}{2}\right)=2\cdot \left(-\dfrac{1}{2}\right)-1=-2 $.
\end{enumerate}	
}
	\end{ex}
\begin{ex}%[Đề kiểm tra học kỳ I, Huyện Đan Phượng - Hà Nội]%[Dương BùiĐức, dự án 7EX-1-2019]%[7D1K8]
	Ba đơn vị kinh doanh góp vốn theo tỉ lệ $ 2 $; $ 4 $; $ 6 $. Hỏi mỗi đơn vị được chia bao nhiêu tiền lãi nếu tổng số tiền lãi là $ 600 $ triệu đồng và tiền lãi được chia tỉ lệ thuận với số vốn đã đóng? 
	\loigiai{
Gọi số vốn của ba đơn vị kinh doanh góp lần lượt là $ a,\ b,\ c $ (triệu đồng). Theo đề bài ta có
\[ 
\dfrac{a}{2}=\dfrac{b}{4}=\dfrac{c}{6}\text{ và }a+b+c=600.
\]
Do đó
\[ 
\dfrac{a}{2}=\dfrac{b}{4}=\dfrac{c}{6}=\dfrac{a+b+c}{2+4+6}=50.
\]
Suy ra $ a=2\cdot 50=100 $, $ b=4\cdot 50=200 $, $ c=6\cdot 50=300 $.
}
\end{ex}
\begin{ex}%[Đề kiểm tra học kỳ I, Huyện Đan Phượng - Hà Nội]%[Dương BùiĐức, dự án 7EX-1-2019]%[7H2B3]%[7H2B4]%[7H2B5]%[7H2B8]%[7H2K8]
	Cho tam giác $ ABC $ vuông tại $ A $ có $ AB = AC $. Gọi $ M $ là trung điểm của cạnh $ BC $, $ D $ là trung điểm của cạnh $ AC $.
	\begin{enumerate}
		\item Chứng minh rằng $ \triangle AMB=\triangle AMC $ và $ AM\perp BC $;
		\item Từ $ A $ kẻ đường thẳng vuông góc với $ BD $, cắt $ BC $ tại $ E $. Trên tia đối của tia $ DE $ lấy điểm $ F $ sao cho $ DF=DE $. Chứng minh rằng $ \triangle ADF=\triangle CDE $. Từ đó suy ra $ AF \parallel CE $;
		\item Từ $ C $ dựng đường thẳng vuông góc với $ AC $, cắt $ AE $ tại $ G $. Chứng minh rằng $ \triangle BAD=\triangle ACG $;
		\item Chứng minh rằng $ AB=2CG $.
	\end{enumerate}
\loigiai{
	\begin{center}

		\coordinate (D) at ($(A)!1/2!(C)$);

\begin{enumerate}
	\item Xét hai tam giác $ \triangle AMB$ và $\triangle AMC $, ta có $ AB=AC $, $ AM $ chung và $ MB=MC $ (do $ M $ là trung điểm $ BC $) nên $ \triangle AMB=\triangle AMC $ (c.c.c).\\
	Vì $ \triangle AMB=\triangle AMC $ nên $ \widehat{AMB}=\widehat{AMC} $. Lại có
	\[ 
	\widehat{AMB}+\widehat{AMC}=180^{\circ}\Rightarrow \widehat{AMB}=\widehat{AMC}=90^{\circ}.
	\]
	Do đó $ AM\perp BC $.
	\item Ta có $ DF=DE $ (giả thiết), $ DA=DC $ (do $ D $ là trung điểm $ AC $) và $ \widehat{ADF}=\widehat{CDE} $ (đối đỉnh), do đó $ \triangle ADF=\triangle CDE $ (c.g.c).\\
	Vì $ \triangle ADF=\triangle CDE $ nên $ \widehat{DAF}=\widehat{DCE} $. Mà hai góc này ở vị trí so le trong nên $ AF \parallel CE $.
	\item Gọi $ H $ là giao điểm của $ AG $ và $ BD\Rightarrow \widehat{AHD}=90^{\circ} $.\\
	Ta có $ \widehat{ADH}+\widehat{HAD}=90^{\circ} $ và $ \widehat{ADB}+\widehat{ABD}=90^{\circ} $ nên $ \widehat{ABD}=\widehat{HAD}$.\\
	Xét hai tam giác $ \triangle BAD$ và $\triangle ACG $, có\\
	$ AB=AC $ (giả thiết), $ \widehat{ABD}=\widehat{HAD} $ (chứng minh trên) và $ \widehat{BAD}=\widehat{ACG}=90^{\circ} $ nên $ \triangle BAD=\triangle ACG $ (g.c.g).
	\item Vì $ \triangle BAD=\triangle ACG $ nên $ CG=AD $. Mà $ D $ là trung điểm $ AC $ nên $ AC=2AD $. Lại có $ AB=AC $ nên $ AB=2CG $.
\end{enumerate}
} 
\end{ex}

	Cho các số $ a,\ b,\ c>0 $ và $ \dfrac{a+b}{3}=\dfrac{b+c}{4}=\dfrac{c+a}{5} $. Tính giá trị của biểu thức
	\[ 
	M=10a+b-7c+2017.
	\]
	\loigiai{
Ta có $ \dfrac{a+b}{3}=\dfrac{b+c}{4}=\dfrac{c+a}{5}=\dfrac{2(a+b+c)}{12}=\dfrac{a+b+c}{6} $. Suy ra
\[ 
\dfrac{a+b}{3}=\dfrac{b+c}{4}=\dfrac{c+a}{5}=\dfrac{a+b+c}{6}=\dfrac{c}{3}=\dfrac{a}{2}=\dfrac{b}{1}.
\]
Do đó $ a=2b,\ c=3b\Rightarrow M=2017 $.
}
\end{ex}
