\section{NHÂN ĐA THỨC VỚI ĐA THỨC}
\subsection{Tóm tắt lí thuyết}
\begin{enumerate}[\tickEX]
	\item Quy tắc nhân dấu
		\begin{center}
			\begin{tabular}{|l|c|c|c|c|}
				\hline 
				Dấu của $a$ & $+$ & $+$ & $-$ & $-$ \\ 
				\hline 
				Dấu của $b$ & $+$ & $-$ & $+$ & $-$ \\ 
				\hline 
				Dấu của $a \cdot b$ &$+$ & $-$ & $-$ & $+$ \\ 
				\hline 
			\end{tabular}
		\end{center}
	\item Quy tắc nhân một số với một tổng: $A\cdot(B+C)=AB+AC.$
	\item Quy tắc nhân hai lũy thừa cùng cơ số: $x^m\cdot x^n=x^{m+n}.$
	\item Quy tắc nhân đa thức với đa thức: Muốn nhân một đa thức với một đa thức, ta nhân mỗi hạng tử của đa thức này với từng hạng tử của đa thức kia rồi cộng các tích tìm được với nhau \[(A+B)(C+D-E)=AC+AD-AE+BC+BD-BE.\]
\end{enumerate}
\subsection{Các dạng toán}
\begin{dang}[Thực hiện phép nhân đa thức với một đa thức]
	\begin{enumerate}[\tickEX]
		\item Áp dụng quy tắc nhân dấu.
		\item Áp dụng quy tắc nhân hai lũy thừa cùng cơ số.
		\item Áp dụng quy tắc nhân đơn thức với đa thức.
		\item Áp dụng quy tắc nhân đa thức với đa thức.
	\end{enumerate}
\end{dang}
\begin{vd}
	Thực hiện các phép tính sau
	\begin{listEX}[2]
		\item $\left(x+3\right)\left(x^2+3x-5\right)$;
		\item $\left(x^2-2x+3\right)\left(x-4\right)$.
	\end{listEX}
	\loigiai{
		\begin{enumerate}
			\item $\left(x+3\right)\left(x^2+3x-5\right)= x^3+3x^2-5x+3x^2+9x-15 = x^3+6x^2+4x-15.$
			\item $\left(x^2-2x+3\right)\left(x-4\right)=x^3-4x^2-2x^2+8x+3x-12=x^3-6x^2+11x-12$.
		\end{enumerate}
	}
\end{vd}
\begin{vd}
	Thực hiện các phép tính sau
	\begin{listEX}[2]
		\item $\left(x^2y^2-\dfrac{1}{2}xy+2y\right)(x-2y)$;
		\item $(x+y)(x^2-xy+y^2)$.
	\end{listEX}
	\loigiai{
		\begin{enumerate}
			\item $\left(x^2y^2-\dfrac{1}{2}xy+2y\right)(x-2y)=x^3y^2-2x^2y^3-\dfrac{1}{2}x^2y+xy^2+2xy-4y^2$.
			\item $(x+y)(x^2-xy+y^2)=x^3-x^2y+xy^2+x^2y-xy^2+y^3=x^3+y^3$.
		\end{enumerate}
	}
\end{vd}
\begin{vd}
	Thực hiện phép tính
	\begin{listEX}[2]
		\item $(x^2-x+1)(x^2+x+1)(x^4-x^2+1)$;
		\item $(x^{2n}-x^ny^n+y^{2n})(x^n+y^n)(x^{3n}-y^{3n})$.
	\end{listEX}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			{\allowdisplaybreaks
			\begin{eqnarray*}
				&& (x^2-x+1)(x^2+x+1)(x^4-x^2+1)\\
				&=& (x^4+x^3+x^2-x^3-x^2-x+x^2+x+1)(x^4-x^2+1)\\
				&=& (x^4+x^2+1)(x^4-x^2+1)\\
				&=& x^8-x^6+x^4+x^6-x^4+x^2+x^4-x^2+1\\
				&=& x^8+x^4+1.
			\end{eqnarray*}
			}
			\item Ta có
			{\allowdisplaybreaks
			\begin{eqnarray*}
				&& (x^{2n}-x^ny^n+y^{2n})(x^n+y^n)(x^{3n}-y^{3n})\\
				&=& (x^{3n}+x^{2n}y^n-x^{2n}y^n-x^ny^{2n}+x^ny^{2n}+y^{3n})(x^{3n}-y^{3n})\\
				&=& (x^{3n}+y^{3n})(x^{3n}-y^{3n})\\
				&=& x^{6n}-x^{3n}y^{3n}+x^{3n}y^{3n}-y^{6n}\\
				&=&x^{6n}-y^{6n}.
			\end{eqnarray*}
			}
		\end{enumerate}
	}
\end{vd}
\begin{dang}[Rút gọn và tính giá trị của biểu thức]
	\begin{enumerate}[\tickEX]
		\item Sử dụng quy tắc nhân đa thức với đa thức để rút gọn biểu thức.
		\item Cần phải rút gọn biểu thức trước khi tính giá trị của biểu thức.
	\end{enumerate}
\end{dang}
\begin{vd}
	Rút gọn và tính giá trị của biểu thức 
	\begin{enumerate}
		\item $P(x)=(x^2-5)(x+3)+(x+4)(x-x^2)$ tại $x=-15;$
		\item $Q(x)=(5x-7)(2x+3)-(7x+2)(x-4)$ tại $x=\dfrac{1}{3}.$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có
			{\allowdisplaybreaks
			\begin{eqnarray*}
				P(x)&=& (x^2-5)(x+3)+(x+4)(x-x^2)\\
				&=& (x^3+3x^2-5x-15)+(x^2-x^3+4x-4x^2)\\
				&=& x^3+3x^2-5x-15+x^2-x^3+4x-4x^2\\
				&=& -x-15.
			\end{eqnarray*}
			}
			Thay $x=-15$ vào biểu thức $P(x)$ ta được $P(-15)=-(-15)-15=0$.
			\item Ta có
			{\allowdisplaybreaks
			\begin{eqnarray*}
				Q(x)&=& (5x-7)(2x+3)-(7x+2)(x-4)\\
				&=& (10x^2+15x-14x-21)-(7x^2-28x+2x-8)\\
				&=& 10x^2+x-21-7x^2+26x+8\\
				&=& 3x^2+27x-13.
			\end{eqnarray*}
			}
			Thay $x=\dfrac{1}{3}$ vào biểu thức $Q(x)$ ta được $Q\left(\dfrac{1}{3}\right)=3\cdot \left(\dfrac{1}{3}\right)^2+27 \cdot \left(\dfrac{1}{3}\right)-13=
			-\dfrac{11}{3}.$
		\end{enumerate}
	}
\end{vd}
\begin{vd}
	Cho biểu thức \[M=(x-a)(x-b)+(x-b)(x-c)+(x-c)(x-a)+x^2.\] Tìm $M$ theo $a$, $b$, $c$, biết rằng $2x=a+b+c.$
	\loigiai{
		Ta có 
		{\allowdisplaybreaks
		\begin{eqnarray*}
			M&=& (x-a)(x-b)+(x-b)(x-c)+(x-c)(x-a)+x^2\\
			&=& x^2-bx-ax+ab+x^2-cx-bx+bc+x^2-ax-cx+ac+x^2\\
			&=& 4x^2-2(a+b+c)x+ab+bc+ca.
		\end{eqnarray*}
		}
		Thay $a+b+c=2x$, ta được $M=ab+bc+ca.$
	}
\end{vd}
\begin{vd}

	\loigiai{
		{\allowdisplaybreaks

			E&=& (x^2+ax+bx+ab)(x+c)\\

		\end{eqnarray*}
		}

		\begin{itemize}
			\item Với $x=2$ thì $E=0$.
			\item Với $x=-2$ thì $E=-60.$
		\end{itemize}
	}
\end{vd}

\begin{dang}[Chứng minh đẳng thức]
	\begin{enumerate}[\tickEX]
		\item Để chứng minh một đẳng thức ta có thể thực hiện biến đổi biểu thức ở vế này (thường là vế phức tạp hơn) của đẳng thức để được một biểu thức bằng biểu thức ở vế kia.
		\item Trong một số trường hợp, để chứng minh một đẳng thức ta có thể biến đổi đồng thời cả hai vế của đẳng thức sao cho chúng cùng bằng một biểu thức thứ ba, hoặc cũng có thể lấy biểu thức vế trái trừ đi biểu thức vế phải và biến đổi có kết quả bằng $0$ thì chứng tỏ đẳng thức đã cho được chứng minh.
	\end{enumerate}	
\end{dang}
\begin{vd}
	Chứng minh các đẳng thức
	\begin{enumerate}
		\item $(x+a)(x+b)=x^2+(a+b)x+ab$;
		\item $(x+a)(x+b)(x+c)=x^3+(a+b+c)x^2+(ab+bc+ca)x+abc.$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
		\item Ta có $(x+a)(x+b)=x^2+bx+ax+ab=x^2+(a+b)x+ab.$
		\item Ta có
		{\allowdisplaybreaks
			\begin{eqnarray*}
				(x+a)(x+b)(x+c)&=& (x^2+bx+ax+ab)(x+c)\\
				&=& x^3+cx^2+bx^2+bcx+ax^2+acx+abx+abc\\
				&=& x^3+(a+b+c)x^2+(ab+bc+ca)x+abc.
			\end{eqnarray*}
		}
		\end{enumerate}
	}
\end{vd}
\begin{vd}
	Cho $x^2+y^2=2$. Chứng minh đẳng thức $2(x+1)(y+1)=(x+y)(x+y+2)$.
	\loigiai{
		Xét vế trái $VT$.
		\[VT=2(x+1)(y+1)=2(xy+x+y+1)=2xy+2x+2y+2.\]
		Xét vế phải $VP$.
		{\allowdisplaybreaks
		\begin{eqnarray*}
			VP &=& (x+y)(x+y+2)\\
			&=& x^2+xy+2x+xy+y^2+2y\\
			&=& x^2+y^2+2xy+2x+2y\\
			&=& 2+2xy+2x+2y.
		\end{eqnarray*}
		}
		Vậy $VT=VP$
	}
\end{vd}
\begin{vd}
	Cho $(a-b)^2+(b-c)^2+(c-a)^2=4(a^2+b^2+c^2-ab-bc-ca)$. Chứng minh rằng $a=b=c.$
	\loigiai{
		{\allowdisplaybreaks
		\begin{eqnarray*}
			&&(a-b)^2+(b-c)^2+(c-a)^2= 4(a^2+b^2+c^2-ab-bc-ca)\\
			&\Leftrightarrow& a^2-2ab+b^2+b^2-2bc+c^2+c^2-2ac+a^2=4a^2+4b^2+4c^2-4ab-4bc-4ca\\
			&\Leftrightarrow& 2a^2+2b^2+2c^2-2ab-2bc-2ca=0\\
			&\Leftrightarrow& a^2-2ab+b^2+b^2-2bc+c^2+c^2-2ca+a^2=0\\
			&\Leftrightarrow& (a-b)^2+(b-c)^2+(c-a)^2=0\\
			&\Leftrightarrow& \heva{a-b&=0\\b-c&=0\\c-a&=0.}\\
			&\Leftrightarrow& a=b=c.
		\end{eqnarray*}	
		}
	}
\end{vd}

			}
			\item 
			Xét vế trái $VT$.
			{\allowdisplaybreaks
			\begin{eqnarray*}
				VT &=& a(a^2-bc)+b(b^2-ca)+c(c^2-ab)\\
				&=& a^3-abc+b^3-abc+c^3-abc\\
				&=& a^3+b^3+c^3-3abc.
			\end{eqnarray*}
			}
			Xét vế phải $VP$.
			{\allowdisplaybreaks
			\begin{eqnarray*}
				VP &=& (a^2+b^2+c^2-ab-bc-ca)(a+b+c)\\
				&=& a^3+ab^2+ac^2-a^2b-abc-a^2c+a^2b+b^3+bc^2-ab^2-b^2c-abc\\
				& & +a^2c+b^2c+c^3-abc-bc^2-ac^2\\
				&=& a^3+b^3+c^3-3abc.
			\end{eqnarray*}
			}
			Vậy $VT=VP.$
		\end{enumerate}
	}
\end{vn}
\begin{vn}
	Cho các số $x$, $y$, $z$ tỉ lệ với $a$, $b$, $c$. Chứng minh rằng \[(x^2+2y^2+3z^2)(a^2+2b^2+3c^2)=(ax+2by+3cz)^2.\]
	\loigiai{
		Vì $x$, $y$, $z$ tỉ lệ với $a$, $b$, $c$ nên $\dfrac{x}{a}=\dfrac{y}{b}=\dfrac{z}{c}$.\\
		Đặt $\dfrac{x}{a}=\dfrac{y}{b}=\dfrac{z}{c}=k$ suy ra $x=ka$, $y=kb$, $z=kc.$\\
		Xét vế trái $VT$.
		{\allowdisplaybreaks
		\begin{eqnarray*}
			VT &=& (x^2+2y^2+3z^2)(a^2+2b^2+3c^2)\\
			&=& (k^2a^2+2k^2b^2+3k^2c^2)(a^2+2b^2+3c^2)\\
			&=& k^2(a^2+2b^2+3c^2)(a^2+2b^2+3c^2)\\
			&=& k^2(a^2+2b^2+3c^2)^2.
		\end{eqnarray*}
		}
		Xét vế trái $VP$.
		\[VP=(ax+2by+3cz)^2=(ka^2+2kb^2+3kc^2)^2=k^2(a^2+2b^2+3c^2)^2.\]
		Vậy $VT=VP.$
	}
\end{vn}
