Tính hợp lí nếu có thể
\begin{enumerate}
	\item $\dfrac{2}{13}\cdot\left(-\dfrac{5}{3}\right)+\dfrac{11}{13}\cdot\left(-\dfrac{5}{3}\right)$.
	\item $\left(-\dfrac{1}{2}\right)^2+\left(-\dfrac{1}{4}\right)^3\cdot 64+\left(-\dfrac{2015}{2016}\right)^0$.
	\item $\left|-3\right|:\sqrt{\dfrac{81}{16}}+\left(-\dfrac{1}{3}\right)^3\cdot 108+\sqrt{\left(-6\right)^2}\cdot\sqrt{2{,}25}$.
\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item {\allowdisplaybreaks
			\begin{eqnarray*}
				\dfrac{2}{13} \cdot \left(-\dfrac{5}{3}\right)+\dfrac{11}{13}\cdot\left(-\dfrac{5}{3}\right) & = & \left(-\dfrac{5}{3}\right) \cdot \left( \dfrac{2}{13} + \dfrac{11}{13} \right) \\
				& = & \left( - \dfrac{5}{3} \right) \cdot 1 = - \dfrac{5}{3}.
			\end{eqnarray*}
		}
		\item {\allowdisplaybreaks
			\begin{eqnarray*}
				\left(-\dfrac{1}{2}\right)^2+\left(-\dfrac{1}{4}\right)^3\cdot 64+\left(-\dfrac{2015}{2016}\right)^0 & = & \dfrac{1}{4} + \left( - \dfrac{1}{4^3} \cdot 4^3 \right) + 1\\
				& = & \dfrac{1}{4} + (-1) + 1 = \dfrac{1}{4}.
			\end{eqnarray*}
		}
		\item {\allowdisplaybreaks
			\begin{eqnarray*}
				\left|-3\right|:\sqrt{\dfrac{81}{16}}+\left(-\dfrac{1}{3}\right)^3\cdot 108+\sqrt{\left(-6\right)^2}\cdot\sqrt{2{,}25} & = & 3:\dfrac{9}{4} + \left(- \dfrac{1}{3^3} \cdot 3^3 \cdot 4 \right) + 6 \cdot 1{,}5 \\
				& = & 3 \cdot \dfrac{4}{9} + (-4) + 9 = \dfrac{4}{3} - 4 + 9 = \dfrac{19}{3}.
			\end{eqnarray*}
		}
	\end{enumerate}
}

Cho $\triangle ABC$ có $AB=AC$, $M$ là trung điểm $BC$.
\begin{enumerate}
	\item Chứng minh $\triangle AMB=\triangle AMC$.
	\item Từ $M$ kẻ $ME\perp AB$ ($ M\in AB$), $MF\perp AC$ ($M\in AC$). Chứng minh rằng $AE=AF$.
	\item Chứng minh $EF\parallel BC$.
	\item Từ $B$ kẻ đường thẳng vuông góc với $AB$, từ $C$ kẻ đường thẳng vuông góc với $ AC$, hai đường thẳng này cắt nhau tại $N$. Chứng minh rằng $A$, $M$, $N$ thẳng hàng.
\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item \immini{
		Tam giác $ABC$ có $AB=AC$ (giả thiết) suy ra $\triangle ABC$ cân tại $A$.\\
		$M$ là trung điểm của $BC$ nên $BM=CM$. \\
		Xét $\triangle AMB$ và $\triangle AMC$ có
		\[ \heva{& AM \text{ chung}\\& AB = AC \text{ (giả thiết)}\\& BM=CM \text{ (chứng minh trên)}}\]
		$\Rightarrow \triangle AMB=\triangle AMC$ (c.c.c).
		}
		{\begin{tikzpicture}[scale=1, line join=round, line cap=round,>=stealth]
			\tikzset{label style/.style={font=\footnotesize}}
			\tkzDefPoints{0/0/B,3/0/M,6/0/C,3/4/A}
			\coordinate (E) at ($(A)!(M)!(B)$);
			\coordinate (F) at ($(A)!(M)!(C)$);
			\tkzDefLine[parallel=through B](M,E)\tkzGetPoint{d} %Vẽ đường thẳng song song
			\tkzInterLL(B,d)(A,M) \tkzGetPoint{N}
			\tkzDrawPoints[fill=black](A,B,C,M,E,F,N)
			\tkzDrawSegments(A,B A,C B,C M,E M,F A,N E,F N,B N,C)
			\tkzMarkRightAngles[size=0.2](A,E,M)
			\tkzMarkRightAngles[size=0.2](A,F,M)
			\tkzMarkRightAngles[size=0.2](A,B,N)
			\tkzMarkRightAngles[size=0.2](A,C,N)
			\tkzLabelPoints[left](A,B,E,N)
			\tkzLabelPoints[right](C,F)
			\tkzLabelPoints[below left](M)
			\end{tikzpicture}}
		\item Vì $\triangle AMB=\triangle AMC$ (chứng minh trên) suy ra $\widehat {BAM} = \widehat {CAM}$ ($2$ góc tương ứng của $2$ tam giác bằng nhau).\\
		Xét $\triangle AEM$ và $\triangle AFM$ có
		\[ \heva{& AM \text{ chung}\\& \widehat {BAM} = \widehat {CAM} \text{ (chứng minh trên)}\\& \widehat {AEM} = \widehat {AFM} = 90^\circ} \Rightarrow \triangle AEM = \triangle AFM \text{ (cạnh huyền - góc nhọn).}\]
		Suy ra $AE=AF$ ($2$ cạnh tương ứng của $2$ tam giác bằng nhau).
		\item Xét $\triangle AEF$ có $AE=AF$ (chứng minh trên) nên $\triangle AEF$ là tam giác cân tại $A$ suy ra $\widehat {AEF} = \widehat{AFE}$.\\
		Mà $\widehat {EAF} + \widehat{AEF} + \widehat{AFE} = 180^\circ \Rightarrow 2 \widehat{AEF} + \widehat{EAF} = 180^\circ \Leftrightarrow \widehat{AEF} = \dfrac{180^\circ - \widehat{A}}{2}$. $(1)$\\
		Tương tự $\triangle ABC$ cân tại $A$ nên $\widehat{ABC} = \dfrac{180^\circ - \widehat{A}}{2}$. $(2)$\\
		Từ $(1)$ và $(2)$ suy ra $\widehat{ABC} = \widehat{AEF}$ mà $2$ góc này là $2$ góc đồng vị nên $EF \parallel BC$, suy ra điều phải chứng minh.
		\item Vì $\widehat {BAM} = \widehat {CAM}$ (chứng minh trên) nên $AM$ là tia phân giác của $\widehat {BAC}$. $(3)$\\
		Xét $\triangle ABN$ và $\triangle ACN$ có
		\[ \heva{& AN \text{ chung}\\& AB = AC \text{ (giả thiết)}\\& \widehat {AEM} = \widehat {AFM} = 90^\circ} \Rightarrow \triangle AEM = \triangle AFM \text{ (cạnh huyền - cạnh góc vuông).}\]
		Suy ra $\widehat {BAN} = \widehat {CAN}$ ($2$ góc tương ứng của $2$ tam giác bằng nhau) hay $AN$ là tia phân giác của $\widehat {BAC}$. $(4)$\\
		Từ $(3)$ và $(4)$ suy ra $AN \equiv AM$ hay $A$, $M$, $N$ thẳng hàng.
	\end{enumerate}
}

Cho $a$, $b$, $c$, $x$, $y$, $z$ khác $0$ thỏa mãn $\dfrac{x}{a-2b+c}=\dfrac{y}{2a-b-c}=\dfrac{z}{4a+4b+c}$. Chứng minh rằng $\dfrac{a}{x+2y+z}=\dfrac{b}{z-y-2x}=\dfrac{c}{4x-4y+z}$.
\loigiai{
	Đặt $\dfrac{x}{a-2b+c}=\dfrac{y}{2a-b-c}=\dfrac{z}{4a+4b+c} = m$, suy ra $\heva{& x=m(a-2b+c)\\& y=m(2a-b-c)\\& z=m(4a+4b+c)}$.\\
	Khi đó ta có
	{\allowdisplaybreaks
		\begin{eqnarray*}
			x+2y+z & = & m(a-2b+c)+2m(2a-b-c)+m(4a+4b+c) \\
			& = & m(a-2b+c+4a-2b-2c+4a+4b+c)=m\cdot 9a=9am \qquad (1)\\
			z-y-2x & = & m(4a+4b+c)-m(2a-b-c)-2m(a-2b+c) \\
			& = & m(4a+4b+c-2a+b+c-2a+4b-2c)=m \cdot 9b = 9bm \qquad (2)\\
			4x-4y+z & = & 4m(a-2b+c)-4m(2a-b-c)+m(4a+4b+c) \\
			& = & m(4a-8b+4c-8a+4b+4c+4a+4b+c) = m \cdot 9c = 9cm \qquad (3).
		\end{eqnarray*}
	}
 	Từ $(1)$, $(2)$ và $(3)$ suy ra
 	\[ \heva{& \dfrac{a}{x+2y+z} = \dfrac{a}{9am} = \dfrac{1}{9m}\\&\dfrac{b}{z-y-2x} = \dfrac{b}{9bm} = \dfrac{1}{9m}\\&\dfrac{c}{4x-4y+z} = \dfrac{c}{9cm} = \dfrac{1}{9m}} \Rightarrow \dfrac{a}{x+2y+z}=\dfrac{b}{z-y-2x}=\dfrac{c}{4x-4y+z} .\]
 	Suy ra điều phải chứng minh.
}
