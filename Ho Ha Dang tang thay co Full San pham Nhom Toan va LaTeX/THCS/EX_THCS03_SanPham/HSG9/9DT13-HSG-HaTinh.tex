\begin{ex}%[Đề thi chọn HSG Toán 9 năm học 2012-2013, Tỉnh Hà Tĩnh]%[9D1B3]
	
	\begin{enumerate}
		\item Tính giá trị của biểu thức $M = \left(x-y \right)^3+3 \left(x-y \right) \left(xy+1 \right)$ biết\\
		$$x= \sqrt[3]{3+2 \sqrt{2}}- \sqrt[3]{3-2 \sqrt{2}};
		y= \sqrt[3]{17+12 \sqrt{2}}-\sqrt[3]{17-12\ sqrt{2}}.$$
		\item Giải phương trình $$\dfrac{2x}{x^2-x+1}-\dfrac{x}{x^2+x+1}=\dfrac{5}{3}.$$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Áp dụng biến đổi $(a-b)^3=a^3-b^3-3ab(a-b)$, ta có:\\
			$$x^3=4 \sqrt{2}-3x  \Rightarrow x^3+3x=4 \sqrt{2}$$
			$$y^3=24 \sqrt{2}-3y \Rightarrow y^3+3y=24 \sqrt{2}$$
			Trừ từng vế hai đẳng thức trên ta được: $x^3-y^3+3(x-y)=-20 \sqrt{2}.$ Suy ra $M=-20 \sqrt{2}$
			\item Ta thấy $x^2+x+1; x^2-x+1$ luôn dương và
		$x=0$ không là nghiệm của phương trình đã cho. Do đó với $t=x+ \dfrac{1}{x}$, phương trình đã cho trở thành $$ \dfrac{2}{t-1}- \dfrac{1}{t+1}=\dfrac{5}{3}$$
		$$\Leftrightarrow 5t^2-3t-14=0 ( t \neq  \pm 1 )$$
		$$ \Leftrightarrow  \left[ \begin{aligned} & t=2 \\ & t=- \dfrac{7}{5}\end{aligned} \right.$$ Từ đó ta tìm được nghiệm của phương trình đã cho là $x=1$.
		\end{enumerate}
	}
\end{ex}
\begin{ex}%Đề thi chọn HSG Toán 9 năm học 2012-2013, Tỉnh Hà Tĩnh]%[9D4G3]
	
	\begin{enumerate}
		\item Giải hệ phương trình $\begin{cases} x^2+y^2+3=4x\\ x^3+12x+y^3=6x^2+9 \end{cases}$
		\item Tìm các số tự nhiên $a, b, c $ phân biệt sao cho biểu thức sau nhận giá trị nguyên $$ P= \dfrac{\left(ab-1\right) \left(bc-1\right) \left(ca-1 \right)}{abc}.$$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item  Nhân cả hai vế phương trình thứ nhất của hệ với $3$ ta được $$9=12x-3x^2-3y^2.$$ Thế vào phương trình thứ hai của hệ ta được $$x^3+y^3=3(x^2-y^2)$$
			$$ \Leftrightarrow (x+y)(x^2-xy+y^2-3x-3y)=0$$
		*) Với $x+y=0 \Leftrightarrow x=-y$ thế vào phương trình thứ nhất của hệ ta được $$2x^2-4x+3=0$$
		$$ \Leftrightarrow 2(x-1)^2+1=0,(L)$$
		*) Với $x^2-xy+y^2-3x-3y=0$, trừ theo từng vế với phương trình thứ nhất của hệ ta được $$ xy -x-3y+3=0$$
		$$ \Leftrightarrow (x-3)(y-1)=0$$
		$$ \Leftrightarrow \left[ \begin{aligned} & x=3 \\ & y=1.\end{aligned} \right.$$ 
		Do đó hệ có hai nghiệm $(3;0)$ và $(2;1).$
		\item  Ta có $ P=abc-(a+b+c)+\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}-\dfrac{1}{abc}$.\\
		Do $a, b, c$ là các số tự nhiên nên $P$ là số nguyên khi và chỉ khi 
		$M=\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}-\dfrac{1}{abc}$ là số nguyên.\\ 
		Do $a, b, c$ có vai trò như nhau, không mất tính tổng quát ta giả thiết $a<b<c$, suy ra $a \geq 1; b \geq 2; c \geq 3;$ suy ra 
		$$ 0< \dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}-\dfrac{1}{abc}< \dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c} \leq \dfrac{1}{1}+\dfrac{1}{2}+\dfrac{1}{3} <2.$$
		$$ \Rightarrow 0<M<2 \Rightarrow M=1,(M \in \mathbb{Z}) $$
		$$ \Rightarrow \dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}-\dfrac{1}{abc}=1 $$
		$$ \Rightarrow a+b+c=(a-1)(b-1)(c-1)+2 (*)$$
		Nếu $(a-1)(b-1) \geq 4 $ thì với $a<b<c$ ta có $3c>a+b+c \Rightarrow 3c > (a-1)(b-1)(c-1+2) \Rightarrow 3c>4(c-1)+2 \Rightarrow 3c>4c-2 \Rightarrow c<2$ (Trái với ĐK $c \geq 3$)\\
		Do đó $a+b+c \geq 6$ nên từ (*) suy ra $a-1>0$, suy ra $b-1>1$, suy ra $(a-1)(b-a)$ chỉ có thể nhận giá trị là $2$ hoặc $3$. Từ đó ta tìm được bộ số $(a;b;c)$ thỏa mãn là $(2;3;5)$.\\
		Vậy các bộ số tự nhiên phân biệt thỏa mãn bài toán gồm các hoán vị của $(2;3;5)$. Khi đó $P=21$.
		\end{enumerate}}
\end{ex}

\begin{ex}%Đề thi chọn HSG Toán 9 năm học 2012-2013, Tỉnh Hà Tĩnh]%[9D5G1]
	Tam giác $ABC$ có chu vi bằng $1$; các cạnh $a, b, c$ thỏa mãn đẳng thức $$\dfrac{a}{1-a}+\dfrac{b}{1-b}+\dfrac{c}{1-c}=\dfrac{3}{2}.$$ Chứng minh tam giác $ABC$ đều.	
\loigiai {
	Với $a, b, c$ dương và $a+b+c=1$ ta có 
	$$\dfrac{a}{1-a}+\dfrac{b}{1-b}+\dfrac{c}{1-c}=\dfrac{3}{2} \Leftrightarrow \dfrac{a}{b+c}+\dfrac{b}{c+a}+\dfrac{c}{a+b}=\dfrac{3}{2} $$
	Áp dụng BĐT Nesbit ta có $$\dfrac{a}{b+c}+\dfrac{b}{c+a}+\dfrac{c}{a+b} \geq  \dfrac{3}{2}. $$ Đẳng thức xảy ra khi và chỉ khi $ a=b=c$ hay tam giác $ABC$ đều. 
}	
	
\end{ex}

\begin{ex}%Đề thi chọn HSG Toán 9 năm học 2012-2013, Tỉnh Hà Tĩnh]%[8D5G1]	
	Cho các số thực dương $x, y, z$ thỏa mãn $x+y+z=1$. Tìm giá trị nhỏ nhất của biểu thức $$F=\dfrac{x^4}{\left(
	x^2+y^2\right)\left( x+y\right)}+\dfrac{y^4}{\left( y^2+z^2\right)\left(y+z\right)}+\dfrac{z^4}{\left(z^2+x^2\right)\left(z+x\right)}.$$

\loigiai {
Ta nhận thấy $$ F =\dfrac{x^4}{\left(
	x^2+y^2\right)\left( x+y\right)}+\dfrac{y^4}{\left( y^2+z^2\right)\left(y+z\right)}+\dfrac{z^4}{\left(z^2+x^2\right)\left(z+x\right)}.$$
$$ \Leftrightarrow F= \dfrac{y^4}{\left(
	x^2+y^2\right)\left( x+y\right)}+\dfrac{z^4}{\left( y^2+z^2\right)\left(y+z\right)}+\dfrac{x^4}{\left(z^2+x^2\right)\left(z+x\right)}.$$
$$ \Leftrightarrow F = \dfrac{1}{2}\left(\dfrac{x^4+y^4}{\left(
	x^2+y^2\right)\left( x+y\right)}+\dfrac{y^4+z^4}{\left( y^2+z^2\right)\left(y+z\right)}+\dfrac{z^4+x^4}{\left(z^2+x^2\right)\left(z+x\right)}\right)$$
Áp dụng BĐT $a^2+b^2 \geq \dfrac{(a+b)^2}{2}$ ta có\\
$$ F \geq \dfrac{1}{4} \left(\dfrac{(x^4+y^4)^2}{\left(
	x^2+y^2\right)\left( x+y\right)}+\dfrac{(y^4+z^4)^2}{\left( y^2+z^2\right)\left(y+z\right)}+\dfrac{(z^4+x^4)^2}{\left(z^2+x^2\right)\left(z+x\right)}\right) $$
$$ \Leftrightarrow F \geq \dfrac{1}{4}\left( \dfrac{x^2+y^2}{x+y} \dfrac{y^2+z^2}{y+z} \dfrac{z^2+x^2}{z+x}\right) \geq \dfrac{1}{8} \left(\dfrac{(x+y)^2}{x+y}+\dfrac{(y+z)^2}{y+z}+ \dfrac{(z+x)^2}{z+x} \right)$$
$$ \Leftrightarrow F=\dfrac{1}{4} (x+y+z)= \dfrac{1}{4}.$$
Vậy $F$ đạt giá trị nhỏ nhất bằng $\dfrac{1}{4}$ khi $x=y=z= \dfrac{1}{3}.$

}
\end{ex}
