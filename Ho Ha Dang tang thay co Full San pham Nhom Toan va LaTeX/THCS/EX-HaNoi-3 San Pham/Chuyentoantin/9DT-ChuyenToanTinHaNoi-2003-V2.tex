\begin{ex}%[Đề thi chuyên Toán - Tin AMS, Hà Nội vòng 2, năm 2003-2004]%[9D5G2]
	Cho hai số tự nhiên $a$ và $b$, chứng minh rằng nếu $a^2 + b^2$ chia hết cho $3$ thì $a$ và $b$ chia hết cho $3$.
	\loigiai{
	Đặt $ a = 3k + r $ với $ r \in \{0;1;2\} $ thì ta thấy $ a^2 $ chia cho $ 3 $ dư $ 0 $ hoặc dư $ 1 $.\\
	Tương tự $ b^2 $ chia cho $ 3 $ dư $ 0 $ hoặc dư $ 1 $.\\
	Từ $ (a^2+b^2)\ \vdots\ 3 \Rightarrow a^2 $ và $ b^2 $ đều chia hết cho $ 3 $.\\
	Suy ra $ a\ \vdots\ 3 $ và $ b\ \vdots\ 3 $.	
	}
\end{ex}	

\begin{ex}%[Đề thi chuyên Toán - Tin AMS, Hà Nội vòng 2, năm 2003-2004]%[9D4G6]
	Cho phương trình: $\left(\dfrac{1}{x}\right)^2 + \left(\dfrac{1}{x + 1}\right)^2 = m$   
	\begin{enumerate}
		\item Giải phương trình khi $m = 15$.
		\item Tìm  $m$ để phương trình có $4$ nghiệm phân biệt.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Điều kiện $ x\ne 0, x\ne -1 $. Phương trình đã cho tương đương với
		\[ \left(\dfrac{1}{x} - \dfrac{1}{x+1} \right)^2 + \dfrac{2}{x(x+1)} -15 = 0 \Leftrightarrow \left(\dfrac{1}{x(x+1)} \right)^2 + \dfrac{2}{x(x+1)} - 15 = 0. \]
		Đặt $ \dfrac{1}{x(x+1)} = y $.	Khi đó phương trình trở thành
		\[ y^2 + 2y - 15 = 0 \Leftrightarrow \hoac{&y=3\\&y=-5.}\]
		Vậy phương trình đã cho có $ 4 $ nghiệm: $ \dfrac{-3\pm\sqrt{21}}{6}, \dfrac{-5\pm\sqrt{5}}{10} $.
		\item Đặt $ \dfrac{1}{x(x+1)} = y \Leftrightarrow \heva{&x(x+1)\ne 0\\&yx^2+yx-1 = 0} $.\\
		Phương trình này có hai nghiệm phân biệt khi và chỉ khi $ \heva{&y\ne 0\\&\Delta >0} \Leftrightarrow \heva{&y<-4\\&y>0} $.\\
		Phương trình $ \left(\dfrac{1}{x}\right)^2 + \left(\dfrac{1}{x + 1}\right)^2 = m \quad (2)$ có $ 4 $ nghiệm phân biệt khi và chỉ khi $ y^2 +2y - m = 0 $ có $ 2 $ nghiệm phân biệt thỏa mãn $ \hoac{&y>0\\&y<-4} $.\\
		Theo định lý Vi-ét: $ y_1 + y_2 = -2 $. Vậy buộc $ (2) $ phải thỏa $ \heva{&y_1<-4\\&y_2>0} $.\\
		Giải hệ $ \heva{&-1-\sqrt{1+m}<-4\\&-1+\sqrt{1+m}>0} $ thu được $ m > 8 $.\\
		Thử lại thấy $ m > 8 $ là các giá trị cần tìm.		
	\end{enumerate}	
	}
\end{ex}


\begin{ex}%[Đề thi chuyên Toán - Tin AMS, Hà Nội vòng 2, năm 2003-2004]%[9D5G1]
	Cho $x$, $y$ là các số nguyên dương  thỏa mãn $x + y = 2003$. Tìm giá trị lớn nhất và nhỏ nhất của biểu thức: 
	$$P = x\left(x^2 + y\right) + y\left(y^2 + x\right)$$
	\loigiai{
	Ta có 
	$$P=(x+y)^3-3xy(x+y)+2xy=2003^3-6007xy.$$
	Vì $$(x-y)^2=(x+y)^2-4xy=2003^2-4xy$$
	Nên $xy$ tăng (hoặc giảm) khi $\left| {x - y} \right|$ giảm (tăng).\\
	Với giả thiết $x,y$ là các số nguyên dương, ta thấy ngay:\\
	$P_{min}=2003^3-6007.1001.1002$ đạt được khi $(x;y)$ bằng $(1001;1002)$ hoặc $(1002;1001)$.\\
	$P_{max}=2003^3-6007.1.1002$ đạt được khi $(x;y)$ bằng $(1;1002)$ hoặc $(1002;1)$.	
	}
\end{ex}
\begin{ex}%[Đề thi chuyên Toán - Tin AMS, Hà Nội vòng 2, năm 2003-2004]%[9H3G7]
	Cho đường tròn $\left(O\right)$ với dây $BC$ cố định ($BC < 2R$) và điểm $A$ trên cung lớn $BC$ ($A$ khác $B$, $C$ và điểm chính giữa của cung). Gọi $H$ là hình chiếu của $A$ lên $BC$, $E$ và $F$ lần lượt là hình chiếu của $B$ và $C$ trên đường kính $AA'$.
	\begin{enumerate}	
		\item Chứng minh $HE\perp AC$.   
		\item Chứng minh hai tam giác $HEF$ và tam giác $ABC$ đồng dạng.
		\item Khi $A$ di chuyển, chứng minh tâm đường tròn ngoại tiếp tam giác $HEF$ cố định.
	\end{enumerate}
	\loigiai{\hfill 
			\immini{\textbf{a)} Do $\widehat{AHB}=\widehat{AEB}=90^\circ$ nên tứ giác $AEHB$ nội tiếp. Hơn nữa do $AA'$ là đường kính nên $\widehat{ACA'}=90^\circ$. Khi đó
				\begin{align*}
				\widehat{EHC}+\widehat{HCA}&=\widehat{BAA'}+\widehat{BCA}+\widehat{BCA'}+\widehat{BCA}=\widehat{ACA'}\\
				&=90^\circ.
				\end{align*}
				Vậy $HE\perp AC$.\\
			\textbf{b)} Do $\widehat{AHC}=\widehat{AFC}=90^\circ$ nên tứ giác $AHFC$ nội tiếp. Suy ra $$\widehat{EHF}=\widehat{EHC}+\widehat{CHF}=\widehat{BAE}+\widehat{EAC}=\widehat{BAC}.$$}{\begin{tikzpicture}
				\tkzInit[xmin=-1,xmax=6,ymin=-2,ymax=5]\tkzClip
				\tkzDefPoints{2/4/A, 0/0/B, 5/0/C}
				\tkzCircumCenter(A,B,C)\tkzGetPoint{O}
				\tkzDefPointBy[projection=onto B--C](A)\tkzGetPoint{H}
				\tkzDefPointBy[symmetry=center O](A)\tkzGetPoint{A'}
				\tkzDefPointBy[projection=onto A--A'](B)\tkzGetPoint{E}
				\tkzDefPointBy[projection=onto A--A'](C)\tkzGetPoint{F}
				\tkzDrawPoints(A,B,C,E,F,H,A',O)
				\tkzDrawCircle[color=blue](O,A)
				\tkzDrawSegments(A,B B,C C,A A,A' A,H B,E C,F H,E C,A' H,F)
				\tkzCircumCenter(H,E,F)\tkzGetPoint{I}
				\tkzInterLC(B,C)(I,H)\tkzGetSecondPoint{D}
				\tkzOrthoCenter(A,B,C)\tkzGetPoint{K}
				\tkzDrawCircle[color=red](I,H)
				\tkzMarkRightAngles(A,E,B A,H,B C,F,A)
				\tkzLabelPoints[above](A)
				\tkzLabelPoints[below left](B,H)
				\tkzLabelPoints[below right](C)
				\tkzLabelPoints[below](A',I)
				\tkzLabelPoints[above right](O,E,D)
				\tkzDrawPoints(I,D,K)
				\tkzLabelPoints[below right](F)
				\tkzLabelPoints[left](K)
				\tkzDrawSegments(D,E B,K)
				\end{tikzpicture}}
			Xét tam giác $HEF$ và tam giác $ABC$ có $\widehat{HEF}=\widehat{ABC}$ (do tứ giác $AEHB$ nội tiếp) và $\widehat{EHF}=\widehat{BAC}$ (cmt). Vậy hai tam giác $HEF$ và $ABC$ đồng dạng.
			\textbf{c)} Gọi $D$ là giao điểm của đường tròn $(HEF)$ và $BC$. Khi đó 
			$$\widehat{DEF}=\widehat{DHF}=\widehat{CAF}$$
			nên $DE\parallel AC$. Mà $AC\perp HE$ nên $DE\perp HE$. Do đó tam giác $HED$ vuông tại $E$. 
			\begin{align}\label{44}
			\text{Suy ra tâm $I$ của $(HEF)$ là trung điểm của $HD$.}
			\end{align}
			Gọi $K$ là trực tâm tam giác $ABC$. Khi đó do $BK$ và $A'C$ cùng vuông với $AC$; $CK$ và $A,B$ cùng vuông với $AB$ nên tứ giác $BKCA'$ là hình bình hành. Suy ra 
			\begin{align}\label{41}
			BK=CA'.
			\end{align}
			Mặt khác $$\widehat{HDF}=\widehat{HEF}=\widehat{ABC}=\widehat{AA'C}$$
			nên tam giác $CDFA'$ nội tiếp. Suy ra 
			\begin{align}\label{42}
			\widehat{CDA'}=\widehat{CFA'}=90^\circ.
			\end{align}
			Hơn nữa 
			\begin{align}\label{43}
			\widehat{KBH}=\widehat{A'CD}\text{ (cùng phụ với $\widehat{ACB}$)}.
			\end{align}
			Từ \eqref{41}, \eqref{42} và \eqref{43} suy ra $\Delta BHK=\Delta CDA'$, từ đó có $BH=CD$. Kết hợp với \eqref{44} suy ra $I$ là trung điểm của $BC$ nên $I$ cố định.
	}
\end{ex}
\begin{ex}%[Đề thi chuyên Toán - Tin AMS, Hà Nội vòng 2, năm 2003-2004]%[9D5G4]
	Lấy $4$ điểm ở miền trong của một tứ giác để cùng với $4$ đỉnh ta được $8$  điểm, trong đó không có $3$ điểm nào thẳng hàng. Biết diện tích của tứ giác là $1$, chứng minh rằng tồn tại một tam giác có $3$ đỉnh lấy từ $8$ đỉnh đã cho có diện tích không vượt quá $\dfrac{1}{10}$. Tổng quát hóa bài toán cho $n$ đa giác với $n$ điểm nằm ở miền trong của đa giác đó.
	\loigiai{
	Nối các điểm tạo thành các tam giác đôi một chỉ chung một cạnh, phủ vừa kín các tứ giác. Tổng các góc trong của các tam giác bằng tổng các góc trong của tứ giác cộng với 4 lần $360^\circ $ nên bằng $360^\circ +4.360^\circ =10.180^\circ $. \\
	Vậy có 10 tam giác mà tổng diện tích là 1 nên tồn tại tam giác có diện tích không vượt quá $\dfrac{1}{10}$.	
	}
\end{ex}
