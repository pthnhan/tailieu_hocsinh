\begin{ex}%[Đề thi vào Chuyên KHTN Hà Nội năm 2014, vòng 1]%[9D1K3]
    Giải phương trình $\left(\sqrt{1+x}+\sqrt{1-x}\right)\left(2+2\sqrt{1-x^2}\right)=8.$
    \loigiai{
Điều kiện xác định: $-1\leq x\leq 1$\\
			Đặt $\sqrt{1+x}+\sqrt{1-x}=a\geq 0$; ta có $a^2=2+2\sqrt{1-x^2}$, ta có phương trình $a^3=8\Leftrightarrow a=2$\\
			Với $a=2$ thì $\sqrt{1+x}+\sqrt{1-x}=2\Leftrightarrow \sqrt{1-x^2}=1\Leftrightarrow x^2=0\Leftrightarrow x=0.$    
    }
\end{ex}

\begin{ex}%[Đề thi vào Chuyên KHTN Hà Nội năm 2014, vòng 1]%[9D2K5]
 Giải hệ phương trình $\heva{
			& x^2-xy+y^2=1 \\ 
			& x^2+xy+2y^2=4.}$
	\loigiai{
Xét $x=0$ ta có hệ $\heva{
				& y^2=1 \\ 
				& y^2=2}$, hệ vô nghiệm.\\
			Xét y=0 ta có hệ $\heva{
				& x^2=1 \\ 
				& x^2=4}$, hệ vô nghiệm.\\
			Vậy $x$, $ y$ khác $0$ đặt $x=ty$; $t\neq 0.$\\
			Ta có hệ $\heva{
				& t^2y^2-ty^2+y^2=1 \\ 
				& t^2y^2+ty^2+2y^2=4}\Leftrightarrow \heva{
				& y^2(t^2-t+1)=1 \\ 
				& y^2(t^2+t+2)=4 }\quad(*).$\\
			Vì mỗi vế hệ $(*)$ khác $0$ ta chia 2 vế hệ $(*)$ cho nhau ta được 
			\begin{align*}
			& \dfrac{t^2-t+1}{t^2+t+2}=\dfrac{1}{4}\Leftrightarrow 4t^2-4t+4=t^2+t+2\Leftrightarrow 3t^2-5t+2=0 \\ 
			\Leftrightarrow &  \left(t-1\right)\left(3t-2\right)=0\Leftrightarrow \hoac{
				& t=1 \\ 
				& t=\dfrac{2}{3}}.
			\end{align*}
			Với $t=1$ thay vào hệ $(*)$ ta có 2 nghiệm $\left(x;y\right)\in \left\{(1;1);(-1;-1)\right\}.$\\
			Với t=$\dfrac{2}{3}$ ta có ta có 2 nghiệm $\left(x;y\right)\in \left\{\left(\dfrac{2\sqrt{7}}{9};\dfrac{3}{\sqrt{7}}\right);\left(-\dfrac{2\sqrt{7}}{9};-\dfrac{3}{\sqrt{7}}\right)\right\}.$
	}
\end{ex}

\begin{ex}%[Đề thi vào Chuyên KHTN Hà Nội năm 2014, vòng 1]%[9D5G1]
    Giả sử $x$; $y$; $z$ là các số thực dương thỏa mãn điều kiện $x+y+z=xyz.$
		Chứng minh rằng
		$$\dfrac{x}{1+x^2}+\dfrac{2y}{1+y^2}+\dfrac{3z}{1+z^2}=\dfrac{xyz(5x+4y+3z)}{(x+y)(y+z)(x+z)}.$$
		\loigiai{
		 $x+y+z=xyz\Leftrightarrow \dfrac{1}{xy}+\dfrac{1}{xz}+\dfrac{1}{yz}=1.$\\
			Nên $\dfrac{x}{1+x^2}=\dfrac{1}{x\left(1+\dfrac{1}{x^2}\right)}=\dfrac{1}{x\left(\dfrac{1}{xy}+\dfrac{1}{xz}+\dfrac{1}{yz}+\dfrac{1}{x^2}\right)}=\dfrac{xyz}{(x+y)(x+z)}\quad (1).$\\ 
			Tương tự 
\begin{align*}
    			&\dfrac{2y}{1+y^2}=\dfrac{2xyz}{(x+y)(y+z)}\quad (2);
			\\ &\dfrac{3z}{1+z^2}=\dfrac{3xyz}{(x+z)(y+z)}\quad (3).
\end{align*}			

			Từ $(1)$, $(2)$, $(3)$ ta có $\dfrac{x}{1+x^2}+\dfrac{2y}{1+y^2}+\dfrac{3z}{1+z^2}=\dfrac{xyz(5x+4y+3z)}{\left(x+y\right)\left(y+z\right)(x+z)}.$\\
			Cách khác từ giả thiết
			\begin{align*}
			& x+y+x=xyz
			\\&\Leftrightarrow x^2+xy+xz=x^2yz
			\\&\Leftrightarrow x^2+1=x^2yz-xy-xz+1=(xy-1)(xz-1) \\ 
			& \Leftrightarrow \dfrac{1}{x^2+1}=\dfrac{1}{(xy-1)(xz-1)}
			\end{align*}
			Tương tự ta cũng có kết quả.
		}
\end{ex}

\begin{ex}%[Đề thi vào Chuyên KHTN Hà Nội năm 2014, vòng 1]%[9D5G3]
 Tìm nghiệm nguyên dương của phương trình  $x^2y^2(x+y)+x+y=3+xy.$
	\loigiai{
 Tìm nghiệm nguyên dương của phương trình: $x^2y^2(x+y)+x+y=3+xy.$\\
			Đặt $x+y=a$, $xy=b$ ta có phương trình
\begin{align*}
& ab^2+a=3+b
\\&\Leftrightarrow a(b^2+1)=b+3
\\&\Leftrightarrow a=\dfrac{b+3}{b^2+1}
\end{align*}			
			Nếu $b=3$ vô nghiệm.\\ 
			Với $b\neq 3$ ta có $ a=\dfrac{b+3}{b^2+1}\Leftrightarrow a(b-3)=\dfrac{b^2-9}{b^2+1}=1-\dfrac{10}{b^2+1}$ suy ra $b^2+1\text{Ư}(10)$; $b\in \left\{2;1\right\}$ suy ra $\left(b;a\right)=\left(1;2\right);\left(2;1\right)$ giải ra $\left(x;y\right)=\left(1;1\right).$
	}
\end{ex}

\begin{ex}%[Đề thi vào Chuyên KHTN Hà Nội năm 2014, vòng 1]%[8H3K8]%[8H3K2]%[9H3G7]
Cho tam giác nhọn $ABC$ với $AB<BC$. $D$ là điểm thuộc cạnh $BC$ sao cho $AD$ là phân giác góc $\widehat{BAC}$. Đường thẳng qua $C$ song song với $AD$ cắt trung trực $AC$ tại $E$. Đường thẳng qua $B$ song song với $AD$ cắt trung trực $AB$ tại $F$.
	\begin{enumerate}
		\item Chứng minh rằng tam giác $ABF$ đồng dạng với tam giác $ACE$.
		\item Chứng minh đường thẳng $BE$, $CF$, $AD$ đồng quy tại $G$.
		\item Đường thẳng qua $G$ song song với $AE$ cắt đường thẳng $BF$ tại $Q$. Đường thẳng $QE$ cắt đường tròn ngoại tiếp tam giác $GEC$ tại $P$ khác $E$. Chứng minh rằng các điểm $A$, $P$, $G$, $Q$, $F$ cùng nằm trên một đường tròn.
	\end{enumerate}
	\loigiai{

	\tkzInterLL(e,f)(B,b)\tkzGetPoint{F}

	\tkzInterLL(C,F)(B,E)\tkzGetPoint{G}
	\tkzDefLine[parallel=through G](A,E)\tkzGetPoint{g}
	\tkzInterLL(G,g)(B,F)\tkzGetPoint{Q}
	\tkzDefCircle[circum](E,D,C)\tkzGetPoint{O}
	\tkzDefCircle[circum](G,E,C)\tkzGetPoint{E'}
	\tkzInterLC(Q,E)(E',E)\tkzGetPoints{P}{E}

\begin{enumerate}
	\item Ta có tam giác $AFB$ cân tại $F$ tam giác $AEC$ cân tại $E$ suy ra $\widehat{ABF}=\widehat{BAF}=\dfrac{1}{2}\widehat{BAC}$;\\
$\widehat{ACE}=\widehat{CAE}=\dfrac{1}{2}\widehat{BAC}$ nên $\triangle ABF\backsim\triangle ACE$  (g.g)\\
$\Rightarrow \dfrac{BF}{CE}=\dfrac{AB}{AC}\quad (1).$
\item Ta có $BF$ song song với $ CE$ vì cùng song song với $AD$.\\
Giả sử $CF$ cắt $BE$ tại $G$, áp dụng định lí Ta-lét $\dfrac{BF}{CE}=\dfrac{BG}{GE}$ $(2)$\\
Áp dụng tính chất đường phân giác $\dfrac{BD}{DC}=\dfrac{AB}{AC}$ $(3)$\\

\begin{ex}%[Đề thi vào Chuyên KHTN Hà Nội năm 2014, vòng 1]%[9D5G1]
	Giả sử $a$, $b$, $c$ là các số thực dương thỏa mãn đẳng thức $ab+bc+ca=1$. Chứng minh rằng
	$2abc(a+b+c)\leq \dfrac{5}{9}+a^4b^2+b^4c^2+c^4a^2.$
	\loigiai{
		Ta có 
		$\heva{& a^4b^2+b^4c^2\geq 2a^2b^3c\\&b^4c^2+c^4a^2\geq 2ab^2c^3\\& a^4b^2+c^4a^2\geq 2a^3bc^2}$
 		nên $a^4b^2+b^4c^2+c^4a^2\geq abc(ab^2+bc^2+ca^2).$\\
		Suy ra \begin{align*}
		a^4b^2+b^4c^2+c^4a^2+\dfrac{1}{9} &=a^4b^2+b^4c^2+c^4a^2+\dfrac{ab+bc+ca}{9}\\
	&\geq abc({{a}}b^2+bc^2+ca^2)+\dfrac{ab+bc+ca}{9}=A
		\end{align*}
		\begin{align*}
A&=abc\left(a^2b+b^2c+c^2a+\dfrac{1}{9a}+\dfrac{1}{9b}+\dfrac{1}{9c}\right)\\&
=abc\left[\left(ab^2+\dfrac{1}{9a}\right)+\left(bc^2+\dfrac{1}{9b}\right)+\left(ca^2+\dfrac{1}{9c}\right)\right] \\ 
 A&\geq abc\left(\dfrac{2}{3}a+\dfrac{2}{3}b+\dfrac{2}{3}c\right)=\dfrac{2abc}{3}(a+b+c)\\&
 \Rightarrow a^4b^2+b^4c^2+c^4a^2+\dfrac{1}{9}\geq \dfrac{2abc}{3}(a+b+c)\quad(1)
		\end{align*}
		Ta lại có 
		\begin{align*}
		& a^2b^2+b^2c^2+c^2a^2\ge ab^2c+a^2bc+abc^2\Leftrightarrow \\ 
		& a^2b^2+b^2c^2+c^2a^2+2(ab^2c+a^2bc+abc^2)\geq 3(ab^2c+a^2bc+abc^2)\\
 \Leftrightarrow 		&\left(ab+bc+ca\right)^2\geq 3abc(a+b+c)  \dfrac{1}{3}\geq abc(a+b+c)\\
\Leftrightarrow 		& \dfrac{4}{9}\geq \dfrac{4abc}{3}(a+b+c)\tag{2}
		\end{align*}
		Từ $(1)$ và $(2)$ ta có $2abc(a+b+c)\leq \dfrac{5}{9}+a^4b^2+b^4c^2+c^4a^2.$\\
		Dấu ''$=$'' xảy ra khi $a=b=c=\dfrac{1}{\sqrt{3}}$.\\
		\textbf{Cách 2} Đặt $ab = x$, $bc = y$, $ac = z$ ($x,y,z>0$)
		Vì $ab+ac+bc = 1 \Rightarrow x+y+z = 1.$\\ 
		Suy ra $2(xy+yz+xz) \leq \dfrac{5}{9}+a^4b^2+b^4c^2+c^4a^2.$\\
		Áp dụng bất đẳng thức Cô-si 
\begin{align*}
&a^4b^2+b^4c^2\geq 2a^2b^2bc=2x^2y \\
&b^4c^2+c^4a^2\geq 2b^2c^2ac=2y^2z\\&
c^4a^2+a^4b^2\geq 2a^2c^2ba=2z^2x\\
\Rightarrow & 2(xy+yz+xz)\leq \dfrac{5}{9}+(x^2y+y^2z+z^2x)\\&
\end{align*}
		Áp dụng bất đẳng thức Côsi
\begin{align*}
&x^2y+\dfrac{1}{9}y\geq \dfrac{2}{3}yx\\
&y^2z+\dfrac{1}{9}z\geq \dfrac{2}{3}yz\\
&z^2x+\dfrac{1}{9}x\geq \dfrac{2}{3}zx\\
\Rightarrow & x^2y+y^2z+z^2x+\dfrac{1}{9}(x+y+z)\geq \dfrac{2}{3}(xy+yz+xz)\\
&\dfrac{4}{9}=\dfrac{4}{9}(x+y+z)^2\geq 3.\dfrac{4}{9}(xy+yz+xz)=\dfrac{4}{3}(xy+yz+xz)\\
\Rightarrow & x^2y+y^2z+z^2x+\dfrac{5}{9}\geq 2(xy+yz+xz)\\
\end{align*}
		Dấu $=$ xảy ra khi $x= y = z = \dfrac{1}{3}$ Suy ra $a = b = c =\dfrac{1}{\sqrt{3}}\cdot$
	}
\end{ex}
