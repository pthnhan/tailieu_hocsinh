\begin{ex}%[8D2B3]%[8D2B9]%[8D2K3]
	Cho biểu thức $ P=\left(\dfrac{x+3}{x+2}-\dfrac{2-x}{3-x}-\dfrac{x-8}{x^{2}-x-6} \right):\left(1-\dfrac{x+6}{2x+4} \right)$.
	\begin{enumerate}
		\item Rút gọn $ P $ và tìm điều kiện xác định của $ P $.
		\item Tính giá trị của $ P $ biết $ |2x-3|+1=x $.
		\item Tìm $ x $ nguyên để $ P\cdot(x- 1) $ có giá trị là số tự nhiên.
	\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có
\begin{align*}
P&=\left(\dfrac{x+3}{x+2}-\dfrac{2-x}{3-x}-\dfrac{x-8}{x^{2}-x-6} \right):\left(1-\dfrac{x+6}{2x+4} \right)\\
&=\left(\dfrac{x+3}{x+2}-\dfrac{2-x}{3-x}-\dfrac{x-8}{(x+2)(x-3)} \right):\left(\dfrac{2x+4-x-6}{2x+4} \right)\\
&=\left(\dfrac{x^{2}-9+4-x^{2}-x+8}{(x+2)(x-3)} \right):\left(\dfrac{x-2}{2(x+2)} \right)\\
&=\left(\dfrac{3-x}{(x+2)(x-3)} \right)\cdot\left(\dfrac{2(x+2)}{x-2} \right)\\
&=-\dfrac{2}{x-2}.
\end{align*}
Điều kiện xác định $ \heva{&x\neq \pm 2\\ &x\neq 3} $.
\item $ |2x-3|+1=x\Leftrightarrow |2x-3|=x-1 $ (điều kiện $ x\geq 1 $)\\
$ \Leftrightarrow \hoac{&2x-3=x-1\\ &2x-3=1-x}\Leftrightarrow\hoac{&x=2\ (\mbox{loại})\\ &x=\dfrac{4}{3}\ (\mbox{thỏa mãn}).} $\\
Với $ x=\dfrac{4}{3} $ ta có $ P=3 $.
\item $ P\cdot (x-1)=-\dfrac{2(x-1)}{x-2}=-2-\dfrac{2}{x-2} $.\\
Để $ P\cdot (x-1) $ là số tự nhiên thì $ x-2\in \mathrm{U}(2)=\{\pm 1;\pm 2\} $. Ta có bảng giá trị
\begin{center}
	\begin{tabular}{|p{2cm}<{\centering}|p{2cm}<{\centering}|p{2cm}<{\centering}|p{2cm}<{\centering}|p{2cm}<{\centering}|}\hline
		$x-2$&$-2$&$-1$&$1$&$2$ \\ \hline
		$x$&$0$&$1$&$3$&$4$ \\ \hline
		Kết luận&thỏa mãn&thỏa mãn&loại&loại\\ \hline
	\end{tabular}
\end{center}
Vậy các giá trị $ x $ nguyên để $ P\cdot(x- 1) $ có giá trị là số tự nhiên là $ x\in\{0;1\} $.
\end{enumerate}
}
\end{ex}
\begin{ex}%[8D3B7]
	Một ô tô dự định đi từ Hà Nội đến Quảng Ninh với vận tốc trung bình $ 45 $ km/h. Lúc đầu ô tô đi với vận tốc đó, khi vượt quá nửa quãng đường $ 30 $ km thì người lái xe tăng vận tốc lên thêm $ 5 $ km/h trên quãng đường còn lại. Do đó ô tô đến Quảng Ninh sớm hơn $ 20 $ phút so với dự định. Tính chiều dài quãng đường từ Hà Nội đến Quảng Ninh?
	\loigiai{
Gọi quãng đường từ Hà nội đến Quảng Ninh cần tìm là $ x \Rightarrow$ quãng đường còn lại kể từ khi xe tăng tốc là $ \dfrac{x}{2}- 30 $ (km, $ x> 60 $).\\
Thời gian ô tô dự định đi trên quãng đường còn lại là $ \dfrac{x- 60}{90}$ (giờ).\\
Thời gian ô tô thực tế đi trên quãng đường còn lại là $ \dfrac{x- 60}{100} $ (giờ).\\
Vì thời gian ô tô dự định nhiều hơn thực tế đi $ \dfrac{1}{3} $ (giờ) nên ta có phương trình:
\[ \dfrac{x-60}{90}-\dfrac{x-60}{100}=\dfrac{1}{3}\Leftrightarrow x=360\ (\mbox{thỏa mãn điều kiện}). \]
Vậy quãng đường từ Hà nội đến Quảng Ninh là $ 360 $ km.
	}
\end{ex}

\begin{ex}%[8H3B7]%[8H3B7]%[8H3K7]%[8H2G3]
	Cho tam giác $ MNP $ nhọn và có các đường cao $ MA $, $ NB $, $ PC $ đồng quy tại $ H $.
	\begin{enumerate}
		\item Chứng minh $ \triangle MBN\backsim\triangle MCP $.
		\item Chứng minh $\widehat{MCB}=\widehat{MPN} $.
		\item Chứng minh $ BN $ là tia phân giác $\widehat{ABC}$.
		\item Tính $ \dfrac{MH}{MA}+\dfrac{NH}{NB}+\dfrac{PH}{PC} $.
	\end{enumerate}
\loigiai{
\begin{center}
	\begin{tikzpicture}[line cap=round,line join=round,>=stealth,x=1.0cm,y=1.0cm, scale=0.7, thick]
		\def\a{5}
	\tkzDefPoint(0,0){M}
	\tkzDefShiftPoint[M](0:1.5*\a){P}
	\tkzDefShiftPoint[M](65:1.2*\a){N}
	\tkzDrawAltitude(N,P)(M)\tkzGetPoint{A}
	\tkzDrawAltitude(M,P)(N)\tkzGetPoint{B}
	\tkzDrawAltitude(M,N)(P)\tkzGetPoint{C}
	\tkzInterLL(B,N)(C,P)\tkzGetPoint{H}
	\tkzDrawSegments(M,N N,P P,M B,C B,A)
	\tkzDrawPoints[fill=black,size=1pt](A,B,C,M,N,P,H)
	\pgfresetboundingbox
	\tkzLabelPoints[above](N)
	\tkzLabelPoints[left](C)
	\tkzLabelPoints[below](P,B,M)
	\tkzLabelPoints[right](A)
	\tkzLabelPoints[below right](H)
	\tkzMarkRightAngles(M,B,N N,A,M N,C,P)
	\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Xét tam giác $ MBN $ và tam giác $ MCP $ có\\
$ \widehat{MBN}=\widehat{MCP}=90^\circ $, $ \widehat{M} $ chung $ \Rightarrow \triangle MBN\backsim\triangle MCP $ (g-g).
\item Vì $ \triangle MBN\backsim\triangle MCP $ nên $ \dfrac{BM}{MC}=\dfrac{MN}{MP}\Rightarrow\dfrac{BM}{MN}=\dfrac{MC}{MP} $.\\
Xét tam giác $ MBC $ và tam giác $ MNP $ có\\
$ \dfrac{BM}{MN}=\dfrac{MC}{MP} $, $ \widehat{M} $ chung $ \Rightarrow \triangle MBC\backsim\triangle MNP $ (c-g-c) $ \Rightarrow \widehat{MCB}=\widehat{MPN} $.
\item Vì $ \triangle MBC\backsim\triangle MNP\Rightarrow \widehat{MBC}=\widehat{MNP} $.\hfill$ (1) $\\
Tương tự ta chứng minh được $ \triangle PBA\backsim\triangle MNP\Rightarrow \widehat{ABP}=\widehat{MNP} $.\hfill$ (2) $\\
Mà $ \widehat{MBC}+\widehat{BCN}=\widehat{ABP}+\widehat{ABN}=90^\circ $.\hfill$ (3) $\\
Từ $ (1) $, $ (2) $, $ (3) $ ta suy ra $ \widehat{NBC}=\widehat{NBA}\Rightarrow BN $ là tia phân giác góc $ \widehat{CBA} $.
\item Đặt $ S_{\triangle HNP}=S_{1} $, $ S_{\triangle HMP}=S_{2} $, $ S_{\triangle HMN}=S_{3} $, $ S_{\triangle MNP}=S $. Ta có\\
$ \dfrac{MH}{MA}=\dfrac{S_2+S_3}{S} $, $ \dfrac{NH}{NB}=\dfrac{S_1+S_3}{S} $, $ \dfrac{PH}{PC}=\dfrac{S_2+S_1}{S}\Rightarrow \dfrac{MH}{MA}+\dfrac{NH}{NB}+\dfrac{PH}{PC}=2 $.
\end{enumerate}
}
\end{ex}
\begin{ex}%[8D2G9]
	Cho $ x $, $ y $, $ z $ đôi một khác nhau và thỏa mãn $ \dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=0 $. Tính giá trị của biểu thức $ A=\dfrac{yz}{x^{2}+2yz}+\dfrac{zx}{y^{2}+2zx}+\dfrac{xy}{z^{2}+2xy} $.
	\loigiai{
Vì $ \dfrac{1}{x}+\dfrac{1}{y}+\dfrac{1}{z}=0\Rightarrow \dfrac{xy+yz+zx}{xyz}=0\Rightarrow xy+yz+zx=0\Rightarrow yz=-xy-xz $.\\
Mà $ x^{2}+2yz=x^2+yz -xy-xz=(x-y)(x-z) $.\\
Tương tự ta có $ y^{2}+2xz=(y-x)(y-z) $, $ z^{2}+2xy=(z-x)(z-y) $.\\
Do đó $ A=\dfrac{yz}{(x-y)(x-z)}+\dfrac{xz}{(y-x)(y-z)}+\dfrac{xy}{(z-x)(z-y)}=1 $.
}
\end{ex}
