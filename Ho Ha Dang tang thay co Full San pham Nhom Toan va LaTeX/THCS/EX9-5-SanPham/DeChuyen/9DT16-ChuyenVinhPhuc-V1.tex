	\begin{ex}%[Đề thi vào 10, Chuyên Vĩnh Phúc Vòng 1, 2016]%[9D4Y6]%[9D4B6] 
		Cho phương trình: $$x^2-2mx+m+2=0 \quad (m \text{ là tham số}).$$
		\begin{enumerate}
			\item Giải phương trình khi $ m=2. $
			\item Tìm tất cả các giá trị của $ m $ để phương trình có nghiệm duy nhất.
		\end{enumerate}
		\loigiai{
			\begin{enumerate}
			\item Với $ m=2 $ phương trình được viết lại là \\
							$ x^2-4x+4=0\\
					\Leftrightarrow (x-2)^2=0\\
					\Leftrightarrow x-2=0\\
					\Leftrightarrow x=2 \\
					\Rightarrow S=\left\lbrace 2 \right\rbrace.$
			\item Ta có $ \Delta ' = m^2-m-2. $	\\
			Phương trình có nghiệm duy nhất khi \\
			$ \Delta '=0 \Leftrightarrow  m^2-m-2=0 \Leftrightarrow (m+1)(m-2)=0 \Leftrightarrow  m=-1 $ hoặc $ m=2. $			
			\end{enumerate}			
		}
	\end{ex}

	\begin{ex}%[Đề thi vào 10, Chuyên Vĩnh Phúc Vòng 1, 2016]%[9D1K8] 
		Cho biểu thức 
		$$
		A=\left(\dfrac{\sqrt{x}-1}{\sqrt{x}+1}-\dfrac{\sqrt{x}+1}{\sqrt{x}-1}\right).\left(\dfrac{1}{2\sqrt{x}}-\dfrac{\sqrt{x}}{2}\right)^2 \quad (x>0; x\neq 1).
		$$
		\begin{enumerate}
			\item Rút gọn $ A. $
			\item Tìm tất cả các giá trị của $ x $ để $ \dfrac{A}{\sqrt{x}}>3. $
		\end{enumerate}
		\loigiai{
			\begin{enumerate}
				\item Ta có 
					\begin{align*}
						A &=\left(\dfrac{\sqrt{x}-1}{\sqrt{x}+1}-\dfrac{\sqrt{x}+1}{\sqrt{x}-1}\right).\left(\dfrac{1}{2\sqrt{x}}-\dfrac{\sqrt{x}}{2}\right)^2\\
						  &=\dfrac{\left(\sqrt{x}-1\right)^2-\left(\sqrt{x}+1\right)^2}{x-1}.\left(\dfrac{1-x}{2\sqrt{x}}\right)^2\\
						  &=\dfrac{-4\sqrt{x}}{x-1}.\dfrac{(1-x)^2}{4x}\\
						  &=\dfrac{1-x}{\sqrt{x}}.
					\end{align*}
				\item Ta có $ \dfrac{A}{\sqrt{x}}>3\Leftrightarrow \dfrac{1-x}{x}>3 
				\Leftrightarrow \dfrac{1-x}{x}-3>0 \Leftrightarrow \dfrac{1-4x}{x}>0 \Leftrightarrow  1-4x> 0 \Leftrightarrow x <\dfrac{1}{4}.$\\
				Vậy $ 0<x<\dfrac{1}{4}. $
			\end{enumerate}
		}
	\end{ex}

	\begin{ex}%[Đề thi vào 10, Chuyên Vĩnh Phúc Vòng 1, 2016]% [9D3B4]%[9D3K4] 
		Cho hệ phương trình 
		$$
			\heva{&(m+1)x-2y=-1\\&x+my=5}, \quad \text{với } m \text{ là tham số.}
		$$
		\begin{enumerate}
			\item Giải hệ phương trình khi $ m =2. $
			\item Tìm tất cả các giá trị của $ m $ để hệ có nghiệm duy nhất $ (x;y) $ sao cho $ 5x+y $ lớn nhất.
		\end{enumerate}
		\loigiai{
			\begin{enumerate}
				\item Với $ m=2 $ ta có\\
					$ 	\heva{&3x-2y=-1\\&x+2y=5} 
						\Leftrightarrow \heva{&4x=4\\&x+2y=5}
						\Leftrightarrow  \heva{&x=1\\&y=2.} 
					$\\
				Vậy hệ có nghiệm duy nhất là $(x;y)=(1;2). $
				\item Ta có 
				$ \heva{(&m+1)x-2y=-1 \\&x=5-my} $\\
				Thế phương trình thứ hai vào phương trình thứ nhất ta được \\ 
				$ y=\dfrac{5m+6}{m^2_m+2}\Rightarrow x=\dfrac{10-m}{m^2+m+2}. $\\
				Suy ra $ 5x+y=\dfrac{56}{m^2+m+2}. $\\
				Ta thấy $ m^2+m+2 = \left(m+\dfrac{1}{2}\right)^2 +\dfrac{7}{4} \geq \dfrac{7}{4}.$\\
				Suy ra $ 5x+y \leq 32. $ Đẳng thức xảy ra khi $ m=\dfrac{1}{2}. $\\
				Vậy  $ 5x+y $ đạt giá trị lớn nhất bằng $ 32 $ khi $ mm=\dfrac{1}{2}. $
				
			\end{enumerate}
		}
	\end{ex}

	\begin{ex}%[Đề thi vào 10, Chuyên Vĩnh Phúc Vòng 1, 2016]% [9D5G1] 
		Cho $ x,y,z $ là các số thực không âm thõa mãn $$x^2+y^2 + z^2 +xyz=4.$$
		Tìm giá trị nhỏ nhất và  giá trị lớn nhất của biểu thức $ P=x+y+z. $
		\loigiai{
			Ta chứng minh $ P \geq 2. \hfill (1) $\\
			Thật vậy : $ P \geq 2 \Leftrightarrow  (x+y+z)^2 \geq 4 \Leftrightarrow x^2+y^2+z^2+2xy+2yz+2xz \geq xyz. $\\
			Từ giả thiết suy ra $ 0 \leq x,y,z \geq 2,$ do đó $2xy +2yz +2xz \geq 2xy \geq xyz. $\\
			Vậy $ (1) $ được chứng minh. Đẳng thức xảy ra $ \Leftrightarrow (x;y;z)=(2;0;0) $ và các hoán vị.\\
			Do đó $ minP=2. $\\
			Ta chứng minh $ P \leq 3. $ Thật vậy, theo giả thiết ta có\\
			 \begin{align*}
			 	(2-x)(2-y)(2-z) &=8-xyz-4P+2(xy+yz+zx)\\
			 					&=8-4P+(x^2+y^2+z^2)+2(xy+yz+zx)\\
			 					&=4-4P+P^2
			 \end{align*}
			 Áp dụng bất đẳng thức Caychy ta được 
			 $$
			 (2-x)(2-y)(2-z)\leq \left(\dfrac{6-(x+y+z)}{3}\right)^3=\left(\dfrac{6-P}{3}\right)^3.
			 $$
			 Do đó $ \left(\dfrac{6-P}{3}\right)^3 \geq P^2-4P+4\Leftrightarrow P^3+9P^2-108 \leq 0$\\
			 $ \Leftrightarrow (P-3)(P^2+12P+36)\leq 0 \Leftrightarrow  P \leq 3. $\\
			 Vậy $ P \leq 3. $ Đẳng thức xảy ra khi $ x=y=z=1. $ Do đó $maxP=3.  $
		}
	\end{ex}
