\begin{ex}%[Thi vào 10, Chuyên Bạc Liêu, 2017]%[9D5G2]%[9D5G3]
\hfill
\begin{enumerate}[a)]
\item Cho $n=2018.2017^{2016}-11^{2017}-6^{2016}$. Chứng minh $n$ chia hết cho $17$.
\item Tìm các số nguyên $x, y$ thỏa mãn: $x^2+y^2+5x^2y^2+60=37xy$.
\end{enumerate} 
\loigiai{
\begin{enumerate}[a)]
\item Ta có 
$2017 \equiv 11  \mod 17 \Rightarrow 2017^{2016} \equiv 11^{2016} \mod 17$.\\
Mặt khác: $2018\equiv 12 \mod 17$.\\
Suy ra $2018.2017^{2016}\equiv 12.11^{2016} \mod 17$.\\
Hay $2018.2017^{2016} \equiv (11+1).11^{2016}\equiv 11^{2017}+11^{2016} \mod 17$.\\
Suy ra $2018.2017^{2016}-11^{2017} \equiv 11^{2016} \mod 17.$\\
Suy ra $n=2018.2017^{2016}-11^{2017}-6^{2016} \equiv 11^{2016}-6^{2016} \mod 17$.\\
Ta lại có:
\begin{itemize}
\item $11^{2} \equiv 2 \mod 17$, suy ra $11^{2016}=(11^2)^{1008} \equiv 2^{2008} \mod 17.$
 \item $6^{2} \equiv 2 \mod 17$, suy ra $6^{2016}=(6^2)^{1008} \equiv 2^{2008} \mod 17$.
\end{itemize}
 Do đó ta có $n \equiv 2^{1008}-2^{1008} =0 \mod 17$.\\
 Vậy $n$ chia hết cho $17$.
 \item \textbf{Cách 1.}
\begin{align*}
x^2+y^2+5x^2y^2+60=37xy &\Leftrightarrow x^2-2xy+y^2=-5x^2y^2+35xy-60\\
&\Leftrightarrow (x-y)^2=5(-x^2y^2+7xy-12). \hspace{1cm} (*)
\end{align*}
Vì $(x-y)^2 \geq 0,  \forall x, y \in \mathbb{Z}$ nên $-x^2y^2+7xy-12 \geq 0, \forall x, y \in \mathbb{Z}. \hspace{1cm} (1)$\\
Đặt $t=xy, (t \in \mathbb{Z})$. Từ $(1)$ ta có $-t^2+7t-12 \geq 0 \Leftrightarrow 3 \leq t \leq 4$.\\
Mà $t \in \mathbb{Z}$ nên ta có $t=3 \vee t=4$ hay $xy=3 \vee xy=4$.\\
Khi đó $5(-x^2y^2+7xy-12)=0$ nên từ $(*)$ ta cũng có $(x-y)^2=0 $ hay $x=y$.
\begin{itemize}
\item $\begin{cases}xy=3\\
x=y
\end{cases} \Leftrightarrow \begin{cases}
x^2=3\hspace{1cm} \text{(không tồn tại $x\in \mathbb{N})$}\\
x=y
\end{cases}$.
\item $\begin{cases}
xy=4\\
x=y
\end{cases} \Leftrightarrow$ $\begin{cases}
x=2\\
y=2
\end{cases}$
$\vee$
$\begin{cases}
x=-2\\
y=-2
\end{cases}$.
\end{itemize}
Vậy cặp số nguyên $(x;y)$ thỏa mãn phương trình trên là: $(2;2), (-2;-2)$.\\
\textbf{Cách 2.}
Xét trường hợp $x, y \geq 0$.\\
Áp dụng bất đẳng thức AM-GM cho hai số không âm $x^2, y^2$ ta có:
\begin{center}
$\begin{gathered}
  \,\,\,\,\,\,\,37xy \geqslant 2xy + 5{x^2}{y^2} + 60 \hfill \\
   \Leftrightarrow 35xy \geqslant 5{x^2}{y^2} + 60 \hfill \\
   \Leftrightarrow 7xy \geqslant {x^2}{y^2} + 12 \hfill \\
   \Rightarrow 8xy \geqslant {x^2}{y^2} + 12 \hfill \\
   \Leftrightarrow 4 \geqslant {\left( {xy - 4} \right)^2} \hfill \\
   \Leftrightarrow 2 \geqslant xy - 4 \geqslant  - 2 \hfill \\
   \Leftrightarrow 6 \geqslant xy \geqslant 2. \hfill \\ 
\end{gathered} $
\end{center}
Suy ra, $xy \in \left\{ {2;3;4;5;6} \right\}$.\\
Mặt khác, phương trình ban đầu tương đương với:
$$(x+y)^2=39xy-5x^2y^2-60.\hspace{1cm} (*)$$
Ta xét các trường hợp:
\begin{itemize}
\item TH1: $xy=2$. Từ $(*)$ suy ra $(x+y)^2=-6$ (vô lí).
\item TH2: $xy=3$. Từ $(*)$ suy ra $(x+y)^2=12$ (loại vì $(x+y)^2$ là số chính phương).
\item TH3: $xy=4$. Từ $(*)$ suy ra $(x+y)^2=16$. Giải ra ta được $x=y=2$.
\item TH4: $xy=5$. Từ $(*)$ suy ra $(x+y)^2=10$ (loại vì $(x+y)^2$ là số chính phương).
\item TH5: $xy=6$. Từ $(*)$ suy ra $(x+y)^2=-6$ (vô lí).
\end{itemize}
Thử lại ta thấy $(x;y)=(2;2)$ là nghiệm của phương trình.\\
Do $xy$ và $(-x).(-y)$ có vai trò như nhau trong phương trình ban đầu nên nghiệm của phương trình là 
$(2;2), (-2;-2)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Thi vào 10, Chuyên Bạc Liêu, 2017]%[9D1G8]%[9D3G3]%[9D4G5]
\hfill
\begin{enumerate}[a)]
\item Cho $a=\sqrt{x^2+\sqrt[3]{x^4y^2}}+\sqrt{y^2+\sqrt[3]{x^2y^4}}$. Chứng minh $\sqrt[3]{x^2}+\sqrt[3]{y^2}=\sqrt[3]{a^2}$.
\item Giải hệ phương trình: 
$\begin{cases}
x^2+2y^2-2y=x-3xy\\
2x^2+y^2-17=3xy-x
\end{cases}$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item Đặt $b=\sqrt[3]{x^2}\ge 0, c=\sqrt[3]{y^2}\ge 0$. Bài toán trở thành:\\
Cho $a=\sqrt{b^3+b^2c}+\sqrt{c^3+bc^2}$ với $b$, $c$ là hai số thực không âm. Chứng minh: $b+c=\sqrt[3]{a^2}$.\\
Ta có 
\begin{align*}
a&=\sqrt{b^3+b^2c}+\sqrt[3]{c^3+bc^2}\\
&=\sqrt{b^2(b+c)}+\sqrt{c^2(b+c)}\\
&=b\sqrt{b+c}+c\sqrt{b+c}\hspace{10pt}(\text{do } b,c\ge 0)\\
&=(b+c)\sqrt{b+c}.
\end{align*}
Suy ra $a^2=(b+c)^3$ hay $b+c=\sqrt[3]{a^2}$.
\item Phương trình thứ nhất tương đương 
$(x+y-1)(x+2y)=0 
\Leftrightarrow\left[\begin{array}{ll}
x+y-1=0&(1)\\
x+2y=0&(2)
\end{array} \right.$.
\begin{itemize}
\item $x+y-1=0 \Leftrightarrow y=1-x$, thay vào phương trình $(2)$, ta có: \begin{align*}
&3x^2-2x-8=0
 \Leftrightarrow \left[\begin{array}{ll}
x=2 &\Rightarrow y=-1\\
x=\dfrac{-4}{3}& \Rightarrow y=\dfrac{7}{3}
\end{array} \right. .
\end{align*}
\item $x+2y=0\Leftrightarrow x=-2y $, thay vào phương trình $(2)$, ta có: \begin{align*}
&15y^2-2y-17=0 
 \Leftrightarrow \left[ \begin{array}{ll}
y=\dfrac{17}{15}& \Rightarrow x=\dfrac{-34}{15}\\
y=-1& \Rightarrow x=2
\end{array}\right. .
\end{align*}
\end{itemize}
Vậy hệ đã cho có $3$ nghiệm là $(2;-1), \left( \dfrac{-4}{3}; \dfrac{7}{3}\right), \left(\dfrac{-34}{15};\dfrac{17}{15}\right)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Thi vào 10, Chuyên Bạc Liêu, 2017]%[9D4G7]%[9D5G1]
\hfill
\begin{enumerate}[a)]
\item Cho phương trình: $x^4+2(m-3)x^2+3m+9=0$ (với $m$ là tham số). Tìm tất cả các giá trị $m$ để phương trình có $4$ nghiệm phân biệt.
\item Cho các số $a,b,c$ thỏa mãn: $a\geq 1; b\geq 4; c\geq 9$. Tìm giá trị lớn nhất của biểu thức
$$M=\dfrac{bc\sqrt{a-1}+ac\sqrt{b-4}+ab\sqrt{c-9}}{abc}$$ 
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item Đặt $t=x^2 $ $(t \geq 0)$. Phương trình đã cho trở thành: $$t^2+2(m-3)t+3m+9=0. \hspace{1cm} (*)$$
Phương trình đã cho có $4$ nghiệm phân biệt khi và chỉ khi phương trình $(*)$ có 2 nghiệm dương phân biệt. Điều này đồng nghĩa với:
$$\begin{cases}
\Delta ' =(m-3)^2-3m-9>0\\
S=t_1+t_2 =-2(m-3)> 0\\
P=t_1.t_2= 3m+9>0
\end{cases} \Leftrightarrow
\begin{cases}
\left[\begin{array}{l}
m>9\\
m<0
\end{array} \right.\\
m<3\\
m>-3
\end{cases} \Leftrightarrow -3<m<0.$$
Vậy với $-3<m<0$ thì phương trình đã cho có $4 $ nghiệm dương phân biệt.
\item Ta có $$M=\dfrac{\sqrt{a-1}}{a}+\dfrac{\sqrt{b-4}}{b}+\dfrac{\sqrt{c-9}}{c}$$
Do $a\ge 1; b\ge 4$ và $c\ge 9$ nên áp dụng bất đẳng thức AM-GM với các cặp số không âm,ta có:
$$\begin{cases}
a=(a-1)+1\geq 2\sqrt{a-1} &\Rightarrow \dfrac{\sqrt{a-1}}{a} \leq \dfrac{1}{2}.\\
b=(b-4)+4\geq 4\sqrt{b-4}&\Rightarrow \dfrac{\sqrt{b-4}}{b} \leq \dfrac{1}{4}.\\
c=(c-9)+9\geq 6\sqrt{c-9}&\Rightarrow \dfrac{\sqrt{c-9}}{c} \leq \dfrac{1}{6}.
\end{cases}$$
Do đó $M \leq \dfrac{1}{2}+\dfrac{1}{4}+\dfrac{1}{6}=\dfrac{11}{12}$.\\
Dấu $``="$ xảy ra khi và chỉ khi $\begin{cases}
a-1=1\\
b-4=4 \\
c-9=9
\end{cases}\Leftrightarrow \begin{cases}
a=2\\
b=8\\
c=18
\end{cases}$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Thi vào 10, Chuyên Bạc Liêu, 2017]%[9D5G5]%[9H3G7]
Cho đường tròn $(O)$ đường kính $BC$, trên đường tròn lấy điểm $A$ $(A\neq B, C)$. Tia phân giác của góc $\widehat{BAC}$ cắt $(O)$ tại $E$, $AI$ là đường cao của tam giác $ABC$.
\begin{enumerate}[a)]
\item Xác định vị trí của điểm $A$ trên đường tròn để tam giác $AIE$ có diện tích lớn nhất.
\item Trên đường tròn lấy điểm $M$ sao cho góc $\widehat{MCE}$ là góc tù; kẻ $EH$, $CK$ lần lượt vuông góc với $MC$, $ME$ tại $H$ và $K$. Chứng minh $MO$ vuông góc với $HK$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzInit[ymin=-3.8,ymax=3.5,xmin=-3.8,xmax=4.5]
\tkzClip
%\draw[cyan,dashed] (-5,-5) grid (5,5);
\tkzDefPoints{-3/0/B, 3/0/C}
\tkzDefMidPoint(B,C) \tkzGetPoint{O}
\tkzDrawCircle(O,C)
\tkzDefPointBy[rotation = center O angle -40](B) \tkzGetPoint{A}
\tkzDefLine[bisector](C,A,B) \tkzGetPoint{i}
\tkzInterLC(A,i)(O,C) \tkzGetFirstPoint{E}
\tkzDefPointBy[projection=onto B--C](A) \tkzGetPoint{I}
\tkzDefPointBy[rotation = center O angle 150](E) \tkzGetPoint{M}
\tkzDefPointBy[projection=onto M--C](E) \tkzGetPoint{H}
\tkzDefPointBy[projection=onto M--E](C) \tkzGetPoint{K}
\tkzDrawPoints(O,C,B,A,E,I,M,H,K)
\tkzLabelPoints[below left](O)
\tkzLabelPoints[above left](A)
\tkzLabelPoints[above right](M)
\tkzLabelPoints[left](B,K)
\tkzLabelPoints[below](E)
\tkzLabelPoints[below left](I)
\tkzLabelPoints[right](C,H)
\tkzDrawSegments(B,C B,A A,C E,H M,H E,M A,O C,K K,H A,I E,C O,M E,I E,A E,O)
\tkzMarkAngles[size=5mm](B,A,E)
\tkzMarkAngles[size=7mm](E,A,C)
\tkzMarkAngles[arc=ll,size=6mm](C,H,K C,E,K)
\tkzMarkRightAngles(A,I,O E,H,M E,K,C)
\end{tikzpicture}
\end{center}
\begin{enumerate}[a)]
\item Gọi R là bán kính đường tròn ngoại tiếp $\Delta ABC$. Đặt $IO=x$ $ (0 \leq x \leq R)$.\\
Ta có: $AI=\sqrt{R^2-x^2}$.\\
Do $AE $ là phân giác của $\widehat{BAC}$ nên $E$ là điểm chính giữa của $\wideparen{BC}$, suy ra $OE \perp BC$.\\
Từ đó ta có $IE=\sqrt{R^2+x^2}$.\\
Ta có $S_{AIE}=\dfrac{1}{2}AI.IE.\sin \widehat{AIE}=\dfrac{1}{2}AI.IE.\sin (90^\circ+ \widehat{OIE})=\dfrac{1}{2}AI.IE.\cos \widehat{OIE}\\=\dfrac{1}{2}AI.IE.\dfrac{OI}{IE}=\dfrac{1}{2}AI.OI=\dfrac{1}{2}\sqrt{R^2-x^2}.x \leq \dfrac{1}{4} (R^2-x^2+x^2)=\dfrac{R^2}{4}$.\\
Dấu "$=$" xảy ra khi và chỉ khi $\sqrt{R^2-x^2}=x \Leftrightarrow x=\dfrac{R\sqrt{2}}{2}$.\\
Từ đó ta có $AI=\sqrt{R^2-x^2}=\dfrac{R\sqrt{2}}{2}$.\\
 Vậy điểm $A $ thuộc đường tròn $(O)$ sao cho khoảng cách từ $A$ đến $BC$ bằng $\dfrac{R\sqrt{2}}{2}$.
\item Ta có $\widehat{EMC}=\dfrac{1}{2} \widehat{EOC}=\dfrac{1}{2}90^\circ=45^\circ$, suy ra $\Delta MEH$ vuông cân tại $H$.\\
Tứ giác $KCHE$ nội tiếp nên $\widehat{CHK}=\widehat{CEK}$ (cùng chắn $\wideparen{KC}$).\\
Ta lại có $\widehat{OMH}=\widehat{OME}+\widehat{EMH}=\widehat{OEM}+45^\circ$. \\
(vì $\Delta OME $ cân tại M nên $\widehat{OME}=\widehat{OEM}$).\\
Suy ra $\widehat{MHK}+\widehat{OMH}=\widehat{OEM}+45^\circ+\widehat{KEC}=\widehat{OEC}+45^\circ =90^\circ$. \\
(vì $\Delta OEC$ vuông cân tại $O $ nên $\widehat{OEC}=45^\circ$).\\
Suy ra $OM \perp HK$.
\end{enumerate}
}
\end{ex}
