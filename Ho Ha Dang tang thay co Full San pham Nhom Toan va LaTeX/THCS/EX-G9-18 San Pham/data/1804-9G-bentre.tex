\begin{ex}%[HSG Lớp 9 Bến Tre, 2018]%[Lê Minh An, Dự án EX-9G-18]%[9D0K2]
Giải phương trình $2017\sqrt{2017x-2016}+\sqrt{2018x-2017}=2018$.
\loigiai{
Điều kiện $x\ge \dfrac{2017}{2018}$.\\
Phương trình tương đương với
\begin{eqnarray*}
&& 2017\left(\sqrt{2017x-2016}-1\right)+\sqrt{2018x-2017}-1=0\\
& \Leftrightarrow & \dfrac{2017\cdot 2017(x-1)}{\sqrt{2017x-2016}+1}+\dfrac{2018(x-1)}{\sqrt{2018x-2017}+1}=0\\
& \Leftrightarrow & (x-1)\left( \dfrac{2017\cdot 2017}{\sqrt{2017x-2016}+1}+\dfrac{2018}{\sqrt{2018x-2017}+1}\right)=0\\
& \Leftrightarrow & x=1\quad\left(\text{vì\,\,} \dfrac{2017\cdot 2017}{\sqrt{2017x-2016}+1}+\dfrac{2018}{\sqrt{2018x-2017}+1}>0.\right)
\end{eqnarray*}
Vậy nghiệm phương trình là $x=1$.
}
\end{ex}

\begin{ex}%[HSG Lớp 9 Bến Tre, 2018]%[Lê Minh An, Dự án EX-9G-18]%[9D0K1]
Rút gọn biểu thức $A=\dfrac{\sqrt{2}\left(3+\sqrt{5}\right)}{2\sqrt{2}+\sqrt{3+\sqrt{5}}}+\dfrac{\sqrt{2}\left(3-\sqrt{5}\right)}{2\sqrt{2}-\sqrt{3-\sqrt{5}}}$.
\loigiai{
Nhân cả tử và mẫu của hai phân thức trong biểu thức $A$ với $\sqrt{2}$ ta được
\begin{eqnarray*}
A & = & \dfrac{2\left(3+\sqrt{5}\right)}{4+\sqrt{6+2\sqrt{5}}}+\dfrac{2\left(3-\sqrt{5}\right)}{4-\sqrt{6-2\sqrt{5}}}\\
& = & \dfrac{6+2\sqrt{5}}{4+\sqrt{5}+1}+\dfrac{6-2\sqrt{5}}{4-\sqrt{5}+1}\\
& = & \dfrac{\left(\sqrt{5}+1\right)^2}{\sqrt{5}\left(\sqrt{5}+1\right)}+\dfrac{\left(\sqrt{5}-1\right)^2}{\sqrt{5}\left(\sqrt{5}-1\right)}\\
& = &\dfrac{\sqrt{5}+1}{\sqrt{5}}+\dfrac{\sqrt{5}-1}{\sqrt{5}}=2.
\end{eqnarray*}
Vậy $A=2$.
}
\end{ex}

\begin{ex}%[HSG Lớp 9 Bến Tre, 2018]%[Lê Minh An, Dự án EX-9G-18]%[9D0K2]
Giải hệ phương trình $\heva{& x^3+6x^2y=7\\ & 2y^3+3xy^2=5}$.
\loigiai{
Hệ phương trình tương đương với
$$\heva{& 5x^3+30x^2y=35\\ & 14y^3+21xy^2=35.}$$
Trừ vế với vế hai phương trình ta được
\begin{eqnarray*}
&& 5x^3+30x^2y-21xy^2-14y^3=0\\
& \Leftrightarrow & (x-y)\left(5x^2+35xy+14y^2\right)=0\Leftrightarrow\hoac{& x=y\\ & 5x^2+35xy+14y^2=0.}
\end{eqnarray*}
\begin{itemize}
\item Với $x=y$ ta được $x=y=1$.
\item Với $5x^2+35xy+14y^2=0$ ta được $\hoac{& y=\left(\dfrac{-35-3\sqrt{105}}{28}\right)x\\ & y=\left(\dfrac{-35+3\sqrt{105}}{28}\right)x}$.
\begin{itemize}
\item Với $y=\left(\dfrac{-35-3\sqrt{105}}{28}\right)x$ ta được
$$x=-\sqrt[3]{\dfrac{98}{91+9\sqrt{105}}};\quad y=\dfrac{35+3\sqrt{105}}{28}\cdot\sqrt[3]{\dfrac{98}{91+9\sqrt{105}}}.$$
\item Với $y=\left(\dfrac{-35+3\sqrt{105}}{28}\right)x$ ta được
$$x=-\sqrt[3]{\dfrac{98}{91-9\sqrt{105}}};\quad y=\dfrac{35-3\sqrt{105}}{28}\cdot\sqrt[3]{\dfrac{98}{91-9\sqrt{105}}}.$$
\end{itemize}
\end{itemize}
}
\end{ex}

\begin{ex}%[HSG Lớp 9 Bến Tre, 2018]%[Lê Minh An, Dự án EX-9G-18]%[9D0G7]
Cho $a,b,c>0$ thỏa mãn $ab+bc+ca=28$. Tìm giá trị nhỏ nhất của biểu thức
$$P=\dfrac{5a+5b+2c}{\sqrt{12\left(a^2+28\right)}+\sqrt{12\left(b^2+28\right)}+\sqrt{c^2+28}}.$$
\loigiai{
Ta có $$\sqrt{12\left(a^2+28\right)}=\sqrt{12\left(a^2+ab+bc+ca\right)}=\sqrt{12(a+b)(a+c)}.$$
Theo BĐT Cô-si ta lại có
$$\sqrt{12(a+b)(a+c)}\le\dfrac{6(a+b)+2(a+c)}{2}=4a+3b+c.$$
Suy ra $\sqrt{12\left(a^2+28\right)}$. Tương tự ta cũng có $\sqrt{12\left(b^2+28\right)}\le 4b+3a+c$, $\sqrt{c^2+28}\le\dfrac{a+b}{2}+c$.\\
Từ đó, cộng vế ta được
$$\sqrt{12\left(a^2+28\right)}+\sqrt{12\left(b^2+28\right)}+\sqrt{c^2+28}\le \dfrac{15a+15b+6c}{2}.$$
Do đó $P\ge \dfrac{2}{3}$. Vậy GTNN của $P$ là $\dfrac{2}{3}$ khi $a=b=\sqrt{\dfrac{28}{11}}$, $c=5\sqrt{\dfrac{28}{11}}$.
}
\end{ex}

\begin{ex}%[HSG Lớp 9 Bến Tre, 2018]%[Lê Minh An, Dự án EX-9G-18]%[9H0K1]%[9H3G7]%[9H0G5]
Cho $\triangle ABC$ có ba góc nhọn nội tiếp đường tròn $(O;R)$. Giả sử các điểm $B,C$ cố định và $A$ di động trên đường tròn $(O)$ sao cho $AB<AC$ và $AC<BC$. Đường trung trực của đoạn thẳng $AB$ cắt $AC$ và $BC$ lần lượt tại $P,Q$. Đường trung trực của đoạn thẳng $AC$ cắt $AB$ và $BC$ lần lượt tại $M$ và $N$.
\begin{enumerate}
\item Chứng minh $OM\cdot ON=R^2$.
\item Chứng minh $M,N,P,Q$ cùng nằm trên một đường tròn.
\item Giả sử hai đường tròn ngoại tiếp $\triangle BMN$ và $\triangle CPQ$ cắt nhau tại $S$ và $T$. Chứng minh $S,T,O$ thẳng hàng.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=1.5]
\tkzDefPoints{0/0/O, -3/-1/B, 3/-1/C}
\clip (-6,-5) rectangle (5.5,3.5);
\tkzDefPointBy[rotation = center O angle 120](C)\tkzGetPoint{A}
\tkzDrawCircle[radius](O,B)
\coordinate (I) at ($(A)!.5!(B)$);
\coordinate (J) at ($(A)!.5!(C)$);
\tkzDefLine[perpendicular=through I](A,B)\tkzGetPoint{i}
\tkzInterLL(I,i)(B,C)\tkzGetPoint{P}
\tkzInterLL(I,i)(A,C)\tkzGetPoint{Q}
\tkzDefLine[perpendicular=through J](A,C)\tkzGetPoint{j}
\tkzInterLL(J,j)(B,C)\tkzGetPoint{M}
\tkzInterLL(J,j)(A,B)\tkzGetPoint{N}
\tkzDrawCircle[circum,dashed](B,M,N)\tkzGetPoint{o}
\tkzDrawCircle[circum,dashed](C,P,Q)\tkzGetPoint{o'}
\tkzInterCC(o,B)(o',C)\tkzGetPoints{S}{T}
\tkzDrawSegments(A,N A,Q B,C I,Q J,N O,T O,B O,C)
\tkzMarkRightAngles(A,I,O A,J,O)
\tkzDrawPoints[fill=white,size=8](O,B,C,A,P,Q,M,N,S,T)
\tkzLabelPoints[above](M,O,P)
\tkzLabelPoints[above left](B,A)
\tkzLabelPoints[above right](C)
\tkzLabelPoints[left](N)
\tkzLabelPoints[right](Q,S,T)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Ta có các $\triangle OAB$, $\triangle OAC$, $\triangle OBC$ cân tại $O$ nên các cặp góc đáy bằng nhau, mà tổng ba góc trong $\triangle ABC$ bằng $180^\circ$ nên ta suy ra được $2\widehat{OBC}+2\widehat{A}=180^\circ$. Do đó $\widehat{OBC}=90^\circ-\widehat{A}$.\\
Mà $\widehat{ONB}=90^\circ-\widehat{A}$ nên $\widehat{OBM}=\widehat{ONB}$ do đó $\triangle OBM\backsim ONB$ (g.g). Từ đó ta có $OM\cdot ON=OB^2=R^2$.
\item Tương tự câu a) ta có $OP\cdot OQ=OC^2=R^2$. Do đó $OM\cdot ON=OP\cdot OQ$. Suy ra $\triangle ONP\backsim OQM$ (c.g.c) nên tứ giác $MNQP$ nội tiếp.
\item Giả sử $OS$ cắt đường tròn ngoại tiếp $\triangle BMN$ tại $T_1$ và cắt đường tròn ngoại tiếp $\triangle CPQ$ tại điểm $T_2$. Khi đó ta có $OS\cdot OT_1=OM\cdot ON$ và $OS\cdot OT_2=OP\cdot OQ$, suy ra $OS\cdot OT_1=OS\cdot OT_2$ hay $T_1\equiv T_2\equiv T$. Vậy $O,S,T$ thẳng hàng.
\end{enumerate}
}
\end{ex}

\begin{ex}%[HSG Lớp 9 Bến Tre, 2018]%[Lê Minh An, Dự án EX-9G-18]%[9D0G5]
Tìm các số nguyên $x,y$ nguyên dương thỏa mãn phương trình $16\left(x^3-y^3\right)=15xy+371$.
\loigiai{
Vì $x,y$ nguyên dương nên $16\left(x^3-y^3\right)=15xy-371>0\Rightarrow x>y$.\\
Lại có $5xy=16\left(x^3-y^3\right)-371$ là số lẻ nên $x,y$ đều là số lẻ. Suy ra $y\ge 1$, $x\ge 3$.
\begin{itemize}
\item Với $x=3$ thì $y=1$ thay vào phương trình thỏa mãn.
\item Với $x\ge 5$ ta có $x-2\ge y$ suy ra
\begin{eqnarray*}
16\left(x^3-y^3\right) & \ge & 16\left[x^3-(x-2)^3\right]\\
& = & 16\left[x^3-\left(x^3-6x^2+12x-8\right)\right]\\
& = & 16\left(6x^2-12x+8\right).
\end{eqnarray*}
Mặt khác $15xy+371\le 15x(x-2)+371=15x^2-30x+371$. Ta chứng minh $16\left(6x^2-12x+8\right)>15x^2-30x+371$. Thật vậy
\begin{eqnarray*}
&& 16\left( 6x^2-12x+8\right)>15x^2-30x+371\\
& \Leftrightarrow & 81x^2-162x-243>0\\
& \Leftrightarrow & x^2-2x-3>0\\
& \Leftrightarrow & (x+1)(x-3)>0\quad (\text{đúng với mọi\,\,} x\ge 5).
\end{eqnarray*}
Suy ra $16\left( x^3-y^3\right)>15xy+371$ với mọi $x\ge 5$. Vậy nghiệm nguyên dương của phương trình là $(3;1)$.
\end{itemize}
}
\end{ex}

\begin{ex}%[HSG Lớp 9 Bến Tre, 2018]%[Lê Minh An, Dự án EX-9G-18]%[9D0G8]
Giả sử trung tâm thành phố Bến Tre có tất cả $2019$ bóng đèn chiếu sáng đô thị, bao gồm $671$ bóng đèn ánh sáng trắng và $673$ bóng đèn ánh sáng vàng nhạt và $675$ bóng đèn ánh sáng vàng sậm. Người ta thực hiện dự án thay bóng đèn theo quy luật sau: Mỗi lần người ta tháo bỏ hai bóng đèn khác loại và thay vào đó bằng hai bóng đèn thuộc loại còn lại. Hỏi theo quy trình trên, đến một lúc nào đó người ta có thể nhận được tất cả các bóng đèn đều thuộc cùng một loại không? Vì sao?
\loigiai{
Ta có $671$ chia cho $3$ dư $2$, $675$ chia cho $3$ dư $1$, $675$ chia cho $3$ dư $0$. Ta thấy mỗi loại bóng đèn có số bóng khi chia cho $3$ được các số dư $0,1,2$.\\
Sau mỗi bước thay bóng đèn, số bóng đèn mỗi loại giảm đi $1$ hoặc tăng thêm $2$, khi đó số dư của chúng khi chia cho $3$ thay đổi như sau
\begin{itemize}
\item Số chia cho $3$ dư $0$ sau khi thay chia cho $3$ sẽ dư $2$.
\item Số chia cho $3$ dư $1$ sau khi thay chia cho $3$ sẽ dư $0$.
\item Số chia cho $3$ dư $2$ sau khi thay chia cho $3$ sẽ dư $1$.
\end{itemize}
Do đó sau mỗi bước thay bóng thì số bóng đèn mỗi loại chia cho $3$ cũng có số dư khác nhau là $0$, $1$, $2$. Vì vậy luôn chỉ có $1$ loại bóng đèn có số lượng bóng đèn chia hết cho $3$. Giả sử đến một lúc nào đó tất cả bóng đèn cùng một loại thì số bóng đèn của $2$ loại kia đều là $0$ và chia hết cho $3$ (mâu thuẫn). Vậy không thể thay bóng đèn theo quy trình trên để tất cả bóng đèn đều thuộc cùng một loại.
}
\end{ex}
