\noindent\textbf{II. PHẦN TỰ LUẬN}
\setcounter{bt}{0}
\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Châu Thành, Long An]%[Thầy Hóa, dự án 9DT19]%[9D1B7]
	Thực hiện phép tính (có trình bày các bước biến đổi)
	\begin{enumEX}{2}
		\item $4\sqrt{3} + \sqrt{27} - \sqrt{45} + \sqrt{5}$;
		\item $\sqrt{8} + 2\sqrt{\dfrac{1}{2}} - \sqrt{0{,}1}\cdot\sqrt{20}$.
	\end{enumEX}
	\loigiai{
		\begin{enumerate}
			\item 
			\allowdisplaybreaks
			\begin{eqnarray*}
				4\sqrt{3} + \sqrt{27} - \sqrt{45} + \sqrt{5}&= &4\sqrt{3} + \sqrt{9\cdot 3} - \sqrt{9\cdot 5} + \sqrt{5}\notag\\
				&= &4\sqrt{3} +3 \sqrt{3} - 3\sqrt{5} + \sqrt{5}\notag\\
				&= & 7\sqrt{3} - 2\sqrt{5}.\notag
			\end{eqnarray*}
		\item 
		\allowdisplaybreaks
		\begin{eqnarray*}
			\sqrt{8} + 2\sqrt{\dfrac{1}{2}} - \sqrt{0{,}1}\cdot\sqrt{20}&= &\sqrt{4\cdot 2} + 2\cdot \dfrac{\sqrt{2}}{2}-\sqrt{0{,}1\cdot 20}\notag\\
			&= &2\sqrt{2} + \sqrt{2}-\sqrt{2}\notag\\
			&= &2\sqrt{2}.\notag
		\end{eqnarray*}
		\end{enumerate}
	}
\end{bt}
\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Châu Thành, Long An]%[Thầy Hóa, dự án 9DT19]%[9D1K8]
	Cho biểu thức $A =\sqrt{4x + 8} - 3\sqrt{x + 2} + \dfrac{2}{5}\sqrt{25x + 50}$ với $x\geq - 2$.
	\begin{enumerate}
		\item Rút gọn $A$;
		\item Tìm $x$ sao cho $A$ có giá trị bằng $2$.
	\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Với $x\geq - 2$, ta có
		\begin{align}
			A &=\sqrt{4x + 8} - 3\sqrt{x + 2} + \dfrac{2}{5}\sqrt{25x + 50}\notag\\
			&=2\sqrt{x + 2} - 3\sqrt{x + 2} + \dfrac{2}{5}\cdot 5\sqrt{x + 2}\notag\\
			&=2\sqrt{x + 2} - 3\sqrt{x + 2} + 2\sqrt{x + 2}\notag\\
			&=\sqrt{x + 2}.\notag
		\end{align}
		\item Để biểu thức $A$ có giá trị bằng $2$ thì với $x\geq - 2$, ta có
		\[\sqrt{x + 2}=2\Rightarrow x+2=4\Leftrightarrow x=2~\text{(nhận).}\]
		Vậy với $x=2$ thì $A=2$.
	\end{enumerate}	
}
\end{bt}

\begin{bt}%[Đề thi Toán 9 Học kỳ 1 năm học 2018-2019, Châu Thành, Long An]%[Thầy Hóa, dự án 9DT19]%[9D2K4]
 Xác định $a$, $b$ để đồ thị hàm số $y = ax + b$ song song với đường thẳng $y = 2x - 3$ và đi qua điểm $A(1;3)$.

	Cho đường tròn tâm $O$, bán kính $OA=R$, dây $BC$ vuông góc với $OA$ tại trung điểm $M$ của $OA$.
	\begin{enumerate}
		\item Tứ giác $OBAC$ là hình gì? Vì sao?
		\item Tiếp tuyến của đường tròn $(O)$ tại $B$ cắt $OA$ tại $D$. Chứng minh $DC$ là tiếp tuyến của đường tròn $(O)$.
		\item Tính độ dài đoạn thẳng $DC$ theo $R$.
	\end{enumerate}
	\loigiai{
		\immini
		{

			\item Tứ giác $OBAC$ là hình thoi. Thật vậy

			Xét tứ giác $OABC$ có
			\begin{itemize}
				\item $M$ là trung điểm của $BC$ (chứng minh trên);
				\item $M$ là trung điểm của $OA$ (giả thiết);
				\item $OA\perp BC$ (giả thiết).
			\end{itemize}
		Vậy tứ giác $OBAC$ là hình thoi.
		\end{enumerate}

	\begin{enumerate}
	\item[b)] Xét $\triangle OBD$ và $\triangle OCD$ có
	\begin{itemize}
		\item $OB=OC=R$;
		\item $OD$: cạnh chung;
		\item $\widehat{COD}=\widehat{BOD}$ ($OBCA$ là hình thoi).
	\end{itemize}
	$\Rightarrow \triangle OBD=\triangle OCD$ (cạnh - góc - cạnh)\\
	$\Rightarrow \widehat{OCD}=\widehat{OBD}$ (hai góc tương ứng).\\
	Mà $\widehat{OBD}=90^\circ$ ($BD$ là tiếp tuyến của $(O)$) nên $\widehat{OCD}=90^\circ$ hay $CD\perp OC$ tại $C\in (O)$.\\
	Vậy $DC$ là tiếp tuyến của đường tròn $(O)$.

\end{enumerate}
