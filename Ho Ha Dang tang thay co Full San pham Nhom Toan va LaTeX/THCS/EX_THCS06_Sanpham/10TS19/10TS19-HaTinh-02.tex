\begin{ex}%[Tuyển sinh 10 Hà Tĩnh 2019]%[Ngọc Diệp, dự án EXTHCS-06]%[9D1Y3]
		Rút gọn biểu thức $P= \sqrt{75}-\sqrt{3}$.
\loigiai{

\begin{ex}%[Tuyển sinh 10 Hà Tĩnh 2019]%[Ngọc Diệp, dự án EXTHCS-06]%[9D1B8]
Rút gọn biểu thức $Q=\left( 1+ \dfrac{3}{\sqrt{x}-3} \right): \dfrac{x}{\sqrt{x}-3} $ với $x>0$ và $x \ne 9$.
\loigiai{
$Q= \left( \dfrac{\sqrt{x}-3 +3}{\sqrt{x}-3} \right): \dfrac{x}{\sqrt{x}-3} =  \dfrac{\sqrt{x}}{\sqrt{x}-3} \cdot \dfrac{\sqrt{x}-3}{x}= \dfrac{1}{\sqrt{x}}$.
}
\end{ex}

\begin{ex}%[Tuyển sinh 10 Hà Tĩnh 2019]%[Ngọc Diệp, dự án EXTHCS-06]%[9D4B2]
 Xác định hệ số $a$ của hàm số $y=a x^2$, $a \ne 0$, biết đồ thị hàm số đi qua điểm $M\left( \dfrac{-1}{2};1 \right) $.
\loigiai{
Thay $x= \dfrac{-1}{2}$, $y=1$ vào hàm số $y= a x^2$ ta được
$$a \cdot \left(\dfrac{-1}{2} \right)^2 = 1 \Leftrightarrow a= 4. $$
Vậy $a=4$.
}
\end{ex}

\begin{ex}%[Tuyển sinh 10 Hà Tĩnh 2019]%[Ngọc Diệp, dự án EXTHCS-06]%[9D4K6]
	Cho phương trình $x^2 - 2 (m-1)x + m^2 -3m =0$, $m$ là tham số. Tìm giá trị $m$ để phương trình đã cho có hai nghiệm $x_1$, $x_2$ thỏa mãn $\left(1+ x_1 \right)^2 +\left(1+ x_2 \right)^2 =14 $.
\loigiai{
Ta có $\Delta' = (m-1)^2 - (m^2 -3m)= m^2 \Leftrightarrow m+1 \ge 0 \Leftrightarrow m \ge -1$. \hfill(1)\\
Theo định lý Vi-ét ta có $\begin{cases}x_1 + x_2 = 2(m-1)\\x_1 x_2 = m^2 -3m\end{cases}$.\\
Theo bài ra, ta có \begin{align*}
\left(1+ x_1 \right)^2 +\left(1+ x_2 \right)^2 =14 & \Leftrightarrow (x_1 + x_2)^2 + 2(x_1 +x_2)-2 x_1 x_2  -12 =0 \\
& \Leftrightarrow 4(m-1)^2 +4(m-1)-2 (m^2 -3m) -12 =0 \\
& \Leftrightarrow m^2 +m -6 =0\\
& \Leftrightarrow \left[\begin{aligned} m & =2 \text{\quad (thỏa mãn điều kiện (1))}\\ m &=-3 \text{\quad (loại)}. \end{aligned}\right.
\end{align*}
}
Vậy giá trị của $m$ cần tìm là $m=2$.
\end{ex}

\begin{ex}%[Tuyển sinh 10 Hà Tĩnh 2019]%[Ngọc Diệp, dự án EXTHCS-06]%[9D3K6]
Hai người công nhân cùng làm chung một công việc thì hoàn thành trong $8$ giờ. Nếu người thứ nhất làm $2$ giờ và người thứ hai làm $3$ giờ thì họ làm được $\dfrac{1}{3}$ công việc. Hỏi nếu làm một mình thì mỗi người hoàn thành công việc đó trong bao lâu?	
\loigiai{
Gọi thời gian hoàn thành công việc một mình của người thứ nhất và người thứ hai lần lượt là $x$ giờ, $y$ giờ. Trong một giờ người thứ nhất làm một mình được $\dfrac{1}{x}$ công việc, người thứ hai làm một mình được $\dfrac{1}{y}$ công việc. Ta có hệ phương trình 
\begin{align*}
\begin{cases} \dfrac{1}{x} + \dfrac{1}{y}= \dfrac{1}{8}\\ \dfrac{2}{x} + \dfrac{3}{y}= \dfrac{1}{3}\end{cases} \Leftrightarrow \begin{cases} \dfrac{1}{x} = \dfrac{1}{24}\\ \dfrac{1}{y}= \dfrac{1}{12}\end{cases} \Leftrightarrow \begin{cases} x= 24\\ y= 12.\end{cases} 
\end{align*}
Vậy nếu làm một mình xong công việc thì người thứ nhất hết $24$ giờ, người thứ hai hết $12$ giờ.
}
\end{ex}

\begin{ex}%[Tuyển sinh 10 Hà Tĩnh 2019]%[Ngọc Diệp, dự án EXTHCS-06]%[9D0G7]
	Cho $x$, $y$ là các số thực thỏa mãn đẳng thức $(x+2)(y+2)= \dfrac{25}{4}$. Tìm giá trị nhỏ nhất của biểu thức $F= \sqrt{1+x^4}+ \sqrt{1+y^4}$.
\loigiai{
	Bất đẳng thức Minkowski cho $4$ số thực $a$, $b$, $x$, $y$: $\sqrt{a^2 + x^2} + \sqrt{b^2 + y^2} \ge \sqrt{\left( a+b\right)^2 + \left( x+y\right)^2  }$.\\
Áp dụng BĐT Mincopxki cho $4$ số $1$, $1$, $x$, $y$ ta có $$F= \sqrt{1+x^4}+ \sqrt{1+y^4} \ge \sqrt{(1+1)^2 + \left( x^2 +y^2 \right)^2 } = \sqrt{\left( x^2 +y^2 \right)^2 +4}.$$
Từ giả thiết ta có $2x+2y+xy = \dfrac{9}{4}$.\\
Ta có $x^2 + y^2 \ge 2xy \Leftrightarrow \dfrac{x^2 +y^2 }{2} \ge xy$. \hfill(1)\\
Lại có $\begin{cases} 4x^2 +1 \ge 4x\\ 4y^2 +1 \ge 4y \end{cases} \Rightarrow 4x^2 + 4y^2 \ge 4(x+y) -2 \Leftrightarrow 2(x^2 +y^2) \ge 2(x+y)-1$. \hfill(2)\\
Cộng theo vế các BĐT $(1)$ và $(2)$ ta được
\begin{align*}
&\dfrac{5(x^2 + y^2)}{2} \ge xy + 2(x+y) -1 \Leftrightarrow \dfrac{5(x^2 + y^2)}{2} \ge \dfrac{5}{4} \Leftrightarrow x^2 + y^2 \ge \dfrac{1}{2} \\
&\Rightarrow F \ge \sqrt{\dfrac{1}{4} + 4}= \dfrac{\sqrt{17}}{2}.
\end{align*}
$F =\dfrac{\sqrt{17}}{2} $ khi và chỉ khi $x=y= \dfrac{1}{2}$. 
Vậy giá trị nhỏ nhất của $F$ là $\dfrac{\sqrt{17}}{2}$.
}
\end{ex}
