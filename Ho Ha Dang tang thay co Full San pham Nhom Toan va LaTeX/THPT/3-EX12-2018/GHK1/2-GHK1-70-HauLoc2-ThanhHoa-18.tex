	}

\begin{ex}%[Đề thi thử THPT Quốc Gia, Hậu Lộc 2, Thanh Hóa Lần 1-2018]%[1D2B2]
	Tìm tập nghiệm của phương trình $\mathrm{C}_x^2+\mathrm{C}_x^3 =4x$.
	\choice
	{$\{0\}$}
	{$\{-5;5\}$}
	{\True $\{5\}$}
	{$\{-5;0;5\}$}
	\loigiai{
		
		Điều kiện $x \in \mathbb{N}$ và $x \ge 3$.
		
		$ \begin{aligned}
		& \mathrm{C}_x^2+\mathrm{C}_x^3 =4x 
		\Leftrightarrow \dfrac{x!}{2! (x-2)!} + \dfrac{x!}{3! (x-3)!} = 4x \\
		\Leftrightarrow &\dfrac{x(x-1)}{2} + \dfrac{x(x-1)(x-2)}{6} = 4x \\
		\Leftrightarrow & \hoac{& x=0 \text{ (loại)}\\ & x^2 = 25}
		\Leftrightarrow  \hoac{& x=-5 \text{ (loại)}\\ & x = 5}
		\Leftrightarrow  x=5.
		\end{aligned} $
		
		Vậy tập nghiệm của phương trình là $\{5\}$.
	}
\end{ex}

	{$IJ \parallel AC$}

	{$IJ \parallel AD$}

			\textbf{Nhắc lại:} Trọng tâm tứ diện là giao điểm của các đường thẳng nối trung điểm các cạnh đối của tứ diện. Trọng tâm này cũng là trung điểm của đoạn thẳng nối trung điểm các cạnh đối diện.
			
			Gọi $M, N$ lần lượt là trung điểm của $CD$ và $AB$. 
			
			Trong $(ABM)$, gọi $I$ là giao điểm của $AG$ và $BM$.
