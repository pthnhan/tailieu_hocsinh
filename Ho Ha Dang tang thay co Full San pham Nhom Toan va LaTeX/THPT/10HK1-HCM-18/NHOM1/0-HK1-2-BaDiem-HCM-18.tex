\begin{bt}%[Đề HK1, Bà Điểm, Hồ Chí Minh 17-18]%[Phan Quốc Trí, 10EX-HK1]%[0D2B1-2]
Tìm tập xác định của hàm số $y=\sqrt{x-2}+\dfrac{5}{x^2-4}$.
\loigiai{
Hàm số xác định $\Leftrightarrow \heva{&x-2\ge 0\\&x^2-4\ne 0}\Leftrightarrow \heva{&x\ge 2\\&x \ne \pm 2.}$\\
Vậy tập xác định: $\mathscr{D}=(2;+\infty)$.
}
\end{bt}

\begin{bt}%[Đề HK1, Bà Điểm, Hồ Chí Minh 17-18]%[Phan Quốc Trí, 10EX-HK1]%[0D2B3-2]
Xác định parabol $(P):y=ax^2+bx+c $ biết $(P)$ đi qua điểm $A(2;3)$ và có đỉnh $I(1;2)$.
\loigiai{
Vì $A \in (P)$ nên $4a+2b+c=3$ \quad \quad $(1)$.\\
$(P)$ có đỉnh $I(1;2)$ nên $\heva{&-\dfrac{b}{2a}=1\\&a+b+c=2} \Leftrightarrow \heva{&2a+b=0\\&a+b+c=2}$ \quad \quad (2)\\
Từ $(1)$ và $(2)$ suy ra $\heva{&a=1\\&b=-2\\&c=3.}$\\
Vậy $(P): y=x^2-2x+3$.
}
\end{bt}

\begin{bt}%[Đề HK1, Bà Điểm, Hồ Chí Minh 17-18]%[Phan Quốc Trí, 10EX-HK1]%[0D3B2-2]%[0D3B2-4]
Giải các phương trình sau
\begin{enumerate}
\item $\left| x^2-4x-5 \right| =4x-17$.
\item $\sqrt{x+3} + \sqrt{6-x} = 3 +\sqrt{(x+3)(6-x)}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item  \allowdisplaybreaks
\begin{eqnarray*}
&&\left| x^2-4x-5 \right| =4x-17\\
&\Leftrightarrow & \heva{&4x-17\ge 0\\& \hoac{&x^2-4x-5=4x-17\\&x^2-4x-5=-4x+17}}
\Leftrightarrow  \heva{&4x-17\ge 0\\& \hoac{&x^2-8x+12=0\\&x^2-22=0}}
\Leftrightarrow  \heva{&4x-17\ge 0\\& \hoac{&x=6\\&x=2\\&x=\pm \sqrt{22}}}
\Leftrightarrow  \hoac{&x=6\\&x=\sqrt{22}.}
\end{eqnarray*} 
Vậy phương trình có nghiệm $x=6$; $x=\sqrt{22}$.
\item Điều kiện $-3\le x \le 6$.\\
Đặt $t=\sqrt{x+3}+\sqrt{6-x}, t\ge 0 \Rightarrow \sqrt{(x+3)(6-x)}=\dfrac{t^2-9}{2}$\\
Phương trình trở thành $t=3+\dfrac{t^2-9}{2} \Leftrightarrow \hoac{&t=3\\&t=-1 \; \text{(loại).}}$\\
$t=3 \Leftrightarrow \sqrt{x+3}+\sqrt{6-x}=3 \Leftrightarrow (x+3)(6-x)=0 \Leftrightarrow \hoac{&x=-3\\&x=6}$ (thỏa mãn điều kiện).\\
Vậy phương trình có nghiệm $x=-3$, $x=6$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Bà Điểm, Hồ Chí Minh 17-18]%[Phan Quốc Trí, 10EX-HK1]%[0D3K2-5]
Tìm tất cả các giá trị của $m$ để phương trình $2x^2+(1-m)x-2=0$ có hai nghiệm phân biệt $x_1,x_2$ sao cho biểu thức $P= \left| x_1-x_2 \right|$ đạt giá trị nhỏ nhất.
\loigiai{
Ta có $\Delta = (m-1)^2+16 >0, \forall m \in \mathbb{R}$ nên phương trình có 2 nghiệm phân biệt với mọi $m$.\\
\begin{eqnarray*}
P=\left| x_1-x_2 \right|=\sqrt{(x_1-x_2)^2}=\sqrt{\dfrac{\Delta}{a^2}} = \sqrt{\dfrac{m^2-2m+17}{4}}=\sqrt{\dfrac{(m-1)^2+16}{4}} \ge 2, \forall m \in \mathbb{R}.
\end{eqnarray*}
$\Rightarrow P$ đạt giá trị nhỏ nhất khi $m=1$.
}
\end{bt}

\begin{bt}%[Đề HK1, Bà Điểm, Hồ Chí Minh 17-18]%[Phan Quốc Trí, 10EX-HK1]%[0D4K1-2]
Cho hai số thực dương $x,y$ thỏa $\dfrac{4}{x}+\dfrac{5}{y}\ge 23$. Tìm giá trị nhỏ nhất của $$B=8x+\dfrac{6}{x}+18y+\dfrac{7}{y}.$$
\loigiai{
Ta có $B=8x+\dfrac{6}{x}+18y+\dfrac{7}{y} = 8x+\dfrac{2}{x}+18y + \dfrac{2}{y} + \dfrac{4}{x}+\dfrac{5}{y}$.\\
Áp dụng bất đẳng thức côsi cho $2$ số dương $8x$, $\dfrac{2}{x}$ ta được $8x+\dfrac{2}{x}\ge 8$. \hfill (1)\\
Áp dụng bất đẳng thức côsi cho $2$ số dương $18y$, $\dfrac{2}{y}$ ta được $18y+\dfrac{2}{y} \ge 12$. \hfill $(2)$\\
Ta lại có $\dfrac{4}{x}+\dfrac{5}{y}\ge 23$. \hfill $(3)$\\
Từ $(1)$, $(2)$, $(3)$ suy ra $B\ge 43$.\\
Dấu $``="$ xảy ra khi và chỉ khi $\heva{&8x=\dfrac{2}{x}\\&18y=\dfrac{2}{y}\\&\dfrac{4}{x}+\dfrac{5}{y}=23} \Leftrightarrow \heva{&x=\dfrac{1}{2}\\&y=\dfrac{1}{3}.}$\\
Vậy $B$ có giá trị nhỏ nhất là $43$ khi $(x,y)=\left( \dfrac{1}{2};\dfrac{1}{3} \right)$.
}
\end{bt}

\begin{bt}%[Đề HK1, Bà Điểm, Hồ Chí Minh 17-18]%[Phan Quốc Trí, 10EX-HK1]%[0H1B4-3]
Trong mặt phẳng $Oxy$, cho  $A(1;-2)$, $B(2;3)$. 
\begin{enumerate}
\item Tìm tọa độ điểm $M$ thuộc $Ox$ sao cho $M,A,B$ thẳng hàng.
\item Tìm tọa độ điểm $D$ sao cho tam giác $OAD$ vuông cân tại $O$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Gọi $M(x;0)$. Ta có $\overrightarrow{AB}=(1;5), \overrightarrow{AM}=(x-1;2)$.\\
$M,A,B$ thẳng hàng $\Leftrightarrow \overrightarrow{AB}$ và $\overrightarrow{AM}$ cùng phương $\Leftrightarrow (x-1)\cdot 5=1\cdot2 \Leftrightarrow x=\dfrac{7}{5}$.\\
Vậy $M\left( \dfrac{7}{5};0 \right)$.
\item Gọi $D(x;y)$. Ta có $\overrightarrow{OA}=(1;-2)$ và $\overrightarrow{OD}=(x;y)$.\\
Tam giác $OAD$ vuông cân tại $O\Leftrightarrow \heva{&\overrightarrow{OA} \cdot \overrightarrow{OD}=0\\&OA=OD} \Leftrightarrow \heva{&x-2y=0\\&x^2+y^2=5} \Leftrightarrow \heva{&x=2y\\&5y^2=5}$\\
$$\Leftrightarrow \heva{&x=2\\&y=1} \vee \heva{&x=-2\\&y=-1.}$$
Vậy $D(2;1)$ hoặc $D(-2;-1)$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Bà Điểm, Hồ Chí Minh 17-18]%[Phan Quốc Trí, 10EX-HK1]%[0H2K2-1]
Cho tam giác $ABC$ có $AB=5$, $AC=8$ và $BC=7$
\begin{enumerate}
\item Tính $\overrightarrow{AB}\cdot \overrightarrow{AC}$. Suy ra số đo góc $BAC$ của $\Delta ABC$.
\item Gọi $M$ là điểm trên cạnh $BC$ sao cho $MC=2MB$. Tính độ dài đoạn $AM$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $\overrightarrow{BC}^2=\left( \overrightarrow{AC}-\overrightarrow{AB} \right)^2=\overrightarrow{AB}^2+\overrightarrow{AC}^2-2\overrightarrow{AB} \cdot \overrightarrow{AC}$\\
$\Rightarrow \overrightarrow{AB}\cdot \overrightarrow{AC} = \dfrac{\overrightarrow{AB}^2+\overrightarrow{AC}^2-\overrightarrow{BC}^2}{2}=\dfrac{25+64-49}{2}=20$.\\
Do đó $\cos A = \cos \left( \overrightarrow{AB},\overrightarrow{AC} \right) = \dfrac{\overrightarrow{AB}\cdot \overrightarrow{AC}}{AB \cdot AC} = \dfrac{1}{2} \Rightarrow \widehat{BAC} = 60^{\circ}$.
\item Ta có $\overrightarrow{MC}=-2\overrightarrow{MB}\Leftrightarrow \overrightarrow{AC}-\overrightarrow{AM}=-2\left( \overrightarrow{AB}-\overrightarrow{AM} \right) \Leftrightarrow \overrightarrow{AM}=\dfrac{2}{3}\overrightarrow{AB}+\dfrac{1}{3} \overrightarrow{AC}$.
\begin{eqnarray*}
AM^2&=&\overrightarrow{AM}^2= \left( \dfrac{2}{3}\overrightarrow{AB}+\dfrac{1}{3} \overrightarrow{AC} \right)\\
&=& \dfrac{4}{9}\overrightarrow{AB}^2+\dfrac{1}{9}\overrightarrow{AC}+\dfrac{4}{9} \overrightarrow{AB} \cdot \overrightarrow{AC}\\
&=&\dfrac{100}{9}+\dfrac{64}{9}+\dfrac{80}{9}\\
&=&\dfrac{244}{9}.
\end{eqnarray*}
$\Rightarrow AM = \dfrac{2\sqrt{61}}{3}$.
\end{enumerate}
}
\end{bt}
