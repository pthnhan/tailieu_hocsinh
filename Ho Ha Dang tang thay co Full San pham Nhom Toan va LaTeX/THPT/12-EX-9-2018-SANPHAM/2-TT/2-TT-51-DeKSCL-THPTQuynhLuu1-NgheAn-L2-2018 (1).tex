	{$M=-35$}

		\begin{longtable}{|m{1cm}|m{1.2cm}|}
			\hline
			$u$
			& $v'$ \\
			\hline
			$x^2$
			& $\cos x$ \\
			\hline
			$2x$
			& $\sin x$ \\
			\hline
			$2$
			& $-\cos x$ \\
			\hline
			$0$
			& $-\sin x$ \\
			\hline
		\end{longtable}
		$=-\left(x^2\sin x\left|_{-\tfrac{\pi}{6}}^{\tfrac{\pi}{6}}\right. +2x\cos x\left|_{-\tfrac{\pi}{6}}^{\tfrac{\pi}{6}}\right. -2\sin x\left|_{-\tfrac{\pi}{6}}^{\tfrac{\pi}{6}}\right. \right)=2-\dfrac{\pi^2}{36}-\dfrac{\pi \sqrt{3}}{3}$. \\

		Khi đó $f(x)=\displaystyle\int\limits_{\mathrm{e}^x}^{\mathrm{e}^{2x}} t\ln t\mathrm{\,d}t=F(\mathrm{e}^{2x})-F(\mathrm{e}^x) \Rightarrow f'(x)=2\mathrm{e}^{2x}F'(\mathrm{e}^{2x})-\mathrm{e}^xF'(\mathrm{e}^x)$.\\

		&2x=\ln \dfrac{1}{4}
		\end{aligned}\right. \Leftrightarrow \left[\begin{aligned}

		\end{aligned}\right. $\\

		\begin{center}
			\begin{tikzpicture}
