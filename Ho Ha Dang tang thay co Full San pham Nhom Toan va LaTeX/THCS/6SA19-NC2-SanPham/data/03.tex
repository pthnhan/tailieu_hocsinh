\end{dang}
\subsection{KIẾN THỨC CƠ BẢN}
\begin{enumerate}
	\item \begin{dn}Trung điểm $M$ của đoạn thẳng $AB$ là
điểm nằm giữa $A$, $B$ và cách đều $A$, $B$.\end{dn} 
	\begin{center}
		\begin{tikzpicture}[scale=1, line join=round, line cap=round, >=stealth]
		%Phan Khai Bao Mac Dinh
		\tkzInit[ymin=-0.5,ymax=0.5,xmin=-0.5,xmax=6.5]
		%\tkzAxeXY
		%\tkzGrid
		\tkzClip
		\tkzSetUpPoint[color=black,fill=black,size=6]
		%Dinh Hinh
		\tkzDefPoints{0/0/A,6/0/B}
		\tkzDefMidPoint(A,B)\tkzGetPoint{M}
		\tkzDrawPoints(A,B,M)
		\tkzDrawSegments(A,B)
		\tkzLabelPoints[below](A,B,M)
		\end{tikzpicture}
	\end{center}
			\begin{itemize}
				\item Điểm $M$ cách đều $A$, $B$ có nghĩa là độ dài các
đoạn thẳng $MA$ và $MB$ bằng nhau $MA=MB$.
				\item Trung điểm của đoạn thẳng $AB$ còn được gọi là điểm chính giữa của đoạn
				thẳng $AB$.
				\item Mỗi đoạn thẳng chỉ có một trung điểm duy nhất.
			\end{itemize}
	\item \begin{tc}Nếu $M$ là trung điểm của đoạn thẳng $AB$ thì $M$ nằm giữa hai điểm $A$, $B$ và $MA=MB$.
	\end{tc}
	\item Các tính chất của trung điểm.\\
	Nếu $M$ là trung diểm của đoạn thẳng $AB$ thì
			\begin{itemize}
				\item \begin{tc}(\textit{Tính chất TĐ1}) $AM=MB=\dfrac{1}{2}AB$.\end{tc}
						Thật vậy, khi $M$ là trung điểm của $AB$ thì
						\begin{itemize}
							\item $M$ nằm giữa hai điểm $A$ và $B$  $\Rightarrow AM+MB=AB$.\hfill$(1)$
							\item $MA=MB$.\hfill$(2)$
						\end{itemize}
						Từ $(1)$ và $(2)$ suy ra $2\cdot AM=AB\Rightarrow AM=\dfrac{1}{2}AB\Rightarrow AM=MB=\dfrac{1}{2}AB$.
				\item \begin{tc}(\textit{Tính chất TĐ2}) Các đoạn thẳng có chung một trung diểm.\\
					Cho bốn điểm $A$, $B$, $C$, $D$ cùng thuộc dường thẳng $xy$ theo thứ tự dó. Nếu biết các đoạn thẳng $AD$ và $BC$ có chung trung điểm thì $AB=CD$ và $AC=BD$.\end{tc}
					\begin{center}
						\begin{tikzpicture}[scale=1, line join=round, line cap=round, >=stealth]
						%Phan Khai Bao Mac Dinh
						\tkzInit[ymin=-0.5,ymax=0.5,xmin=-0.5,xmax=10.5]
						%\tkzAxeXY
						%\tkzGrid
						\tkzClip
						\tkzSetUpPoint[color=black,fill=black,size=6]
						%Dinh Hinh
						\tkzDefPoints{1/0/A,9/0/D,2.5/0/B}
						\tkzDefMidPoint(A,D)\tkzGetPoint{M}
						\tkzDefPointBy[rotation= center M angle 180](B) \tkzGetPoint{C}
						\tkzDrawPoints(A,B,C,D,M)
						\tkzDrawLines[add=0.2 and 0.2](A,D)
						\tkzLabelLine[pos=-0.16,below](A,D){$x$}
						\tkzLabelLine[pos=1.16,below](A,D){$y$}
						\tkzLabelPoints[below](A,B,C,D,M)
						\end{tikzpicture}
					\end{center}
					Thật vậy, nếu gọi $M$ là trung điểm của $AD$ thì $M$ cũng là trung điểm của $BC$ ($AD$ và $BC$ có chung một trung điểm). Có $M$ nằm giữa hai điểm $B$, $C$ và $M$ nằm giữa hai điểm $A$, $D$. Lại có $A$, $B$, $C$, $D$ thuộc đường thẳng $xy$ theo thứ tự đó, nên dễ dàng chỉ ra
					\begin{itemize}
						\item $MA=AB+BM$ và $MD=MC+CD$, mà $MA=MD$ và $MB=MC$ nên $AB=CD$.
						\item $AC=AM+MC$ và $BD=BM+MD$, mà $MA=MD$ và $MC=MB$ nên $AC=BD$.
					\end{itemize}
			\end{itemize}
	\item Dấu hiệu trung điểm (nhận biết trung điểm của một đoạn thẳng).
			\begin{itemize}
				\item Dấu hiệu trung điểm 1: Nếu trên đoạn thẳng $AB$ tồn tại một điểm $M$ sao cho $AM=\dfrac{1}{2}AB$ (hoặc $MB=\dfrac{1}{2}AB$) thì điểm $M$ là trung điểm của đoạn thẳng $AB$.\\
				Thật vậy, đặt độ dài đoạn thẳng $AB=a$ thì $AM=\dfrac{1}{2}AB=\dfrac{1}{2}a$. Nếu gọi $N$ là
trung điểm của đoạn thẳng $AB$ thì cũng có $AN=\dfrac{1}{2}AB=\dfrac{1}{2}a$ (Tính chất TĐ1). Suy ra $AM=AN$.\\
				Ta có các diểm $M$ và $N$ cùng thuộc tia $AB$ mà $AM=AN$ nẽn điểm $M$ trùng với điểm $N$ hay điểm $M$ là trung điểm của đoạn thẳng $AB$.
				\item Dấu hiệu trung điểm 2: Cho bốn điểm $A$, $B$, $C$ và $D$ thuộc đường thẳng $xy$ theo thứ tự đó.\\
				Nếu $AB=CD$ thì $AD$ và $BC$ có chung một trung điểm.\\
				Nếu $AC=BD$ thì $AD$ và $BC$ có chung một trung điểm.
\\
				Thật vậy ta có
				\begin{itemize}
					\item Trường hợp $AB=CD$.
					Giả sử điểm $M$ là trung điểm của đoạn thẳng $BC$. Điểm $M$ nằm giữa hai điểm $B$ và $C$, mà bốn điểm $A$, $B$, $C$ và $D$ thuộc đường thẳng $xy$ theo thứ tự đó nên $M$ cũng nằm giữa hai điểm $A$ và $D$. Ta cũng có điểm $B$ nằm giữa hai điểm $A$, $M$; điểm $C$ nằm giữa hai điểm $M$, $D$.\\
					Do $MB=MC$ mà $AB=CD$ nên $AB+BM=MC+CD$. Suy ra $AM=MD$, mà $M$ nằm giữa hai điểm $A$ và $D$ nên $M$ là trung điểm của $AD$.\\
					Ta biết mỗi đoạn thẳng chỉ có một trung điểm duy nhất nên hai đoạn thẳng $AD$ và $BC$ có chung một trung điểm.
					\begin{center}
						\begin{tikzpicture}[scale=1, line join=round, line cap=round, >=stealth]
						%Phan Khai Bao Mac Dinh
						\tkzInit[ymin=-0.5,ymax=0.5,xmin=-0.5,xmax=10.5]
						%\tkzAxeXY
						%\tkzGrid
						\tkzClip
						\tkzSetUpPoint[color=black,fill=black,size=6]
						%Dinh Hinh
						\tkzDefPoints{1/0/A,9/0/D,2.5/0/B}
						\tkzDefMidPoint(A,D)\tkzGetPoint{M}
						\tkzDefPointBy[rotation= center M angle 180](B) \tkzGetPoint{C}
						\tkzDrawPoints(A,B,C,D,M)
						\tkzDrawLines[add=0.2 and 0.2](A,D)
						\tkzLabelLine[pos=-0.16,below](A,D){$x$}
						\tkzLabelLine[pos=1.16,below](A,D){$y$}
						\tkzLabelPoints[below](A,B,C,D,M)
						\end{tikzpicture}
					\end{center}
					\item Trường hợp $AC=BD$.
					Do $AC=BD$ mà $AC+CD=AB+BD$ (vì cùng bằng $AD$) nên $AB=CD$. Như vậy, các điểm $A$, $B$, $C$ và $D$ theo thứ tự cùng thuộc đường thẳng $xy$ mà $AB= CD$, nên theo trường hợp 1 thì $AD$ và $BC$ có chung một trung điểm.
					
				\end{itemize}
				
				
			\end{itemize}
	
\end{enumerate}

\subsection{VÍ DỤ MINH HỌA}

\begin{vd}%[Phạm Tuấn Hùng, dự án 6SA19-NC2]%[6H1B9]
	Cho hai tia đối nhau $Ox$ và $Ox’$ như hình bên dưới.
	\begin{enumerate}
		\item Trên tia $Ox$ lấy điểm $A$ sao cho $OA=6$ cm. Trên tia $Ox’$ lấy điểm $B$ sao cho $OB=6$ cm. Chứng tỏ rằng điểm $O$ là trung điểm của đoạn thẳng $AB$.
		\item Lấy diếm $C$ thuộc tia $Ox'$ sao cho $OC=3$ cm. Hỏi điểm $C$ là trung điểm của những đoạn thẳng $OA$, $OB$ hay $AB$?
	\end{enumerate}
	\begin{center}

		\end{tikzpicture}
	\end{center}
	\loigiai{
		\begin{enumerate}
			\item Điểm $A$ thuộc tia $Ox$ và điểm $B$ thuộc tia $Ox’$ mà hai tia $Ox$ và $Ox'$ đối nhau nên điểm $O$ nằm giữa hai điểm $A$ và $B$. Mặt khác ta lại có $OA=6$ cm và $OB=6$ cm nên $OA=OB$. Theo định nghĩa ta có điểm $O$ là trung điểm của đoạn thẳng $AB$.
			\item Điểm $C$ thuộc tia $Ox'$ và điểm $A$ thuộc tia $Ox$, do dó $A$ và $C$ là hai điểm thuộc hai tia đối nhau gổc $O$, nên điểm $O$ nằm giữa hai điểm $A$ và $C$. Từ đó suy ra điểm $C$ không nằm giữa hai điểm $A$ và $O$, hay điểm $C$ không là trung điểm của đoạn thẳng $OA$.\\
			Điểm $O$ là trung điểm của đoạn thẳng $AB$ (câu a). Có $B$ và $C$ thuộc tia $Ox'$ mà $OC<OB$ (vì $3< 6$), do đó điểm $C$ nằm giữa hai điểm $O$ và $B$. Vậy điểm $C$ không trùng với điểm $O$ (trung điểm của $AB$) nên điểm $C$ không là trung điểm của doạn thẳng $AB$.\\
			Điểm $C$ nằm giữa hai điểm $O$ và $B$, lại có $OC=\dfrac{1}{2}OB$ (vì $OC=3$ cm và
$OB=6$ cm). Vây điểm $C$ là trung điểm của đoạn thẳng $OB$ (dấu hiệu TĐ1).\\
			Với 3 đoạn thẳng $OA$, $OB$ và $AB$ thì điểm $C$ chì là trung điểm của đoạn thẳng $OB$.
		\end{enumerate}
	}
\end{vd}

\subsection{KIẾN THỨC CƠ BẢN}
\begin{enumerate}
	\item Nếu điểm $M$ thuộc đoạn thẳng $AB$ thì $AM+MB=AB$.
	\item Nếu $M$ là trung điểm của đoạn thẳng $AB$ thì $AM=MB=\dfrac{1}{2}AB$.
\end{enumerate}
\subsection{VÍ DỤ MINH HỌA}
\begin{vd}%[Bui Anh Tuan, dự án 6SA19-NC2]%[6H1K10]
	Cho đoạn thẳng $AB$ và một điểm $C$ nằm giữa hai điểm $A$ và $B$. Gọi $M$ và $N$ lần lượt là trung điểm của $AC$ và $CB$ (h. $29$).
	\begin{enumerate}
		\item Biết $AB=20$ cm. Tính độ dài của đoạn thẳng $MN$.
		\item Giả sử $MN=a$. Tính độ dài của đoạn thẳng $AB$.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[line join = round, line cap = round,>=stealth,font=\footnotesize,scale=1]
			
			\tkzDefPoints{0/0/A,5/0/B,3/0/C, 3/-0.5/E}
			\coordinate (M) at ($(A)!0.5!(C)$);
			\coordinate (N) at ($(B)!0.5!(C)$);
			\tkzDrawSegments(A,B)
			\tkzDrawPoints[fill=black](A,B,C,M,N)
			\tkzLabelPoints[above](A,B,C)
			\tkzLabelPoints[below](M,N)	
			\tkzLabelPoint[below](E){\text{Hình}\hspace{0.1 cm}$29$}
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}
			\item Điểm $C$ nằm giữa hai điểm $A$, $B$ nên hai tia $CA$ và $CB$ đối nhau. \quad (1)\\
