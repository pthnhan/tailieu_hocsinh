\begin{bt}%[Đề thi Toán 9 HK1 năm học 2018-2019, Nam Định]%[Nguyễn Sỹ, dự án 9DT19-HK1]%[9D1B7]%[9D1B8]
	Rút gọn các biểu thức
	\begin{multicols}{3}
		\begin{enumerate}
			\item $A=\dfrac{1}{2-\sqrt{3}}+\dfrac{1}{2+\sqrt{3}}$.
			\item $B=\sqrt{5+2\sqrt{(1-\sqrt{2})^2}}$.
			\item $C=\dfrac{x+\sqrt{x}-2}{\sqrt{x}+2}$.
		\end{enumerate}
	\end{multicols}
	\loigiai{ 
		\begin{enumerate}
			\item Ta có $A=\dfrac{2+\sqrt{3}+2-\sqrt{3}}{(2-\sqrt{3})(2+\sqrt{3})}=\dfrac{4}{4-3}=4$.
			\item Ta có $B=\sqrt{5+2|1-\sqrt{2}|}=\sqrt{5+2(\sqrt{2}-1)}=\sqrt{3+2\sqrt{2}}=\sqrt{(\sqrt{2}+1)^2}=\sqrt{2}+1$.
			\item Điều kiện $x\geq 0$.\\
			Ta có $C=\dfrac{(\sqrt{x}+2)(\sqrt{x}-1)}{\sqrt{x}+2}=\sqrt{x}-1$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Đề thi Toán 9 HK1 năm học 2018-2019, Nam Định]%[Nguyễn Sỹ, dự án 9DT19-HK1]%[9D2K4]
	Cho hàm số $y=(2-m)x+2$ với $m$ là tham số và $m\neq 2$, có đồ thị là đường thẳng $d$.
	\begin{enumerate}
		\item Vẽ đồ thị của hàm số trên với $m=3$.
		\item Xác định giá trị của $m$ để đường thẳng $d$ cắt đường thẳng $y=2x-4$ tại một điểm nằm trên trục hoành.
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Với $m=3$, ta có $d\colon y=-x+2$.
	\immini{Cho $x=0\Rightarrow y=2\Rightarrow d$ cắt trục $Oy$ tại $(0;2)$.\\
	Cho $y=0\Rightarrow x=2 \Rightarrow d$ cắt trục $Oy$ tại $(2;0)$.}{

\draw (0,2) node [left]{$2$};

\tkzDefPoints{0/0/A,2/0/B,0/2/C}

}
\item 
\begin{itemize}
	\item Xác định tọa độ giao điểm của đường thẳng $y=2x-4$ với trục hoành\\
	Cho $y=0\Rightarrow x=2$, suy ra đường thẳng $y=2x-4$ cắt trục $Ox$ tại điểm $I(2;0)$.
	\item Đường thẳng $d$ cắt đường thẳng $y=2x-4$ tại điểm nằm trên trục hoành khi và chỉ khi  $I\in d \Leftrightarrow 0 = (2-m)2+2 \Leftrightarrow m=3$.

\begin{bt}%[Đề thi Toán 9 HK1 năm học 2018-2019, Nam Định]%[Nguyễn Sỹ, dự án 9DT19-HK1]%[9H2B6]%[9H2K6]%[9H2K6]
	Cho đường tròn $(O;R)$, đường kính $AB$. Qua $A$ vẽ đường thẳng $d$ là tiếp tuyến của đường tròn $(O;R)$. $C$ là một điểm thuộc đường thẳng $d$. Qua $C$ vẽ tiếp tuyến thứ hai của đường tròn $(O;R)$, tiếp xúc với đường tròn $(O:R)$ tại điểm $M$. Gọi $H$ là giao điểm của $AM$ và $OC$.
	\begin{enumerate}
		\item Chứng minh $AM$ vuông góc với $OC$ và $OH\cdot OC= R^2$.
		\item Chứng minh $\widehat{OBH}=\widehat{OCB}$.
		\item Qua $O$ vẽ đường thẳng vuông góc với $OC$, đường thẳng này cắt $CM$ tại $D$. Chứng minh $DB$ là tiếp tuyến của đường tròn $(O;R)$.
	\end{enumerate}
\loigiai{
\begin{center}
	\begin{tikzpicture}[scale=1, line join=round, line cap=round]
	\tkzDefPoints{0/0/A,6/0/B,0/4/C,3/0/O}
	\draw (0,-3)--(0,5);
	\tkzDrawCircle[radius](O,A)
	
	\tkzDefPointBy[reflection=over C--O](A)\tkzGetPoint{M}
	\tkzDefPointBy[rotation=center O angle 90](C)\tkzGetPoint{C'}
	\tkzInterLL(C,M)(O,C')\tkzGetPoint{D}
	\tkzInterLL(A,M)(C,O)\tkzGetPoint{H}
	\tkzDrawPoints[size=5,fill=black](A,B,C,O,D,H)
	\tkzDrawSegments(A,B C,O A,O C,M O,M A,M C,D B,D O,D C,B H,B)
	\tkzMarkRightAngles[size=0.2](B,A,C D,O,C)
	\tkzLabelPoints[left](A,C)
	\tkzLabelPoints[right](B,D)
	\tkzLabelPoints[below](O,H)
	\tkzLabelPoints[above](M)
	\tkzLabelSegment[left](A,C){$d$}
	\end{tikzpicture}
\end{center}
\begin{enumerate}
	\item Ta có $CA=CM$ (tính chất hai tiếp tuyến cắt nhau). Suy ra $C$ thuộc đường trung trực của $AM. \hfill(1)$\\
	Mà $OA=OM=R\Rightarrow O$ cũng thuộc đường trung trực của $AM. \hfill (2)$\\
	Từ $(1)$ và $(2)$, suy ra $CO$ là đường trung trực của $AM$ hay $AM\perp CO$.\\
	Ta có $AC\perp AO$. Xét $\triangle AOC$ vuông tại $C$ có đường cao là $AH$, ta có $OH\cdot OC=OA^2$.\\
	Mà $OA=R\Rightarrow OH\cdot OC= R^2$.
	\item Ta có $OH\cdot OC=R^2\Rightarrow OH\cdot OC= OB^2$\\
	Mà $\widehat{BOH}=\widehat{ COB}\Rightarrow \triangle OBH \backsim \triangle OCB\Rightarrow \widehat{OBH}=\widehat{OCB}$.
	\item Ta có $\heva{&AM \perp CO\\&OD \perp CO}\Rightarrow AM \parallel OD$, suy ra $\widehat{MOD}=\widehat{OMA}$ (so le trong).\\
	Mà $\widehat{DOB}=\widehat{MAO}$ (đồng vị), và $\widehat{OMA}=\widehat{MAO}$ ($\triangle OAM$ cân tại $O$).\\
	Suy ra $\widehat{MOD}=\widehat{DOB}\Rightarrow \triangle OMD =\triangle OBD$ (c-g-c). Suy ra $\widehat{OMD}=\widehat{OBD}$.\\
	Mà $\widehat{OMD}=90^\circ$ ($CM$ là tiếp tuyến của $(O;R)$), suy ra $\widehat{OBD}=90^\circ$ hay $DB\perp AB$ tại $B$ thuộc đường tròn $(O)$.\\
	Suy ra $DB$ là tiếp tuyến của $(O;R)$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề thi Toán 9 HK1 năm học 2018-2019, Nam Định]%[Nguyễn Sỹ, dự án 9DT19-HK1]%[9D5G1]
	Cho $x, y$ thỏa mãn điều kiện $3\left(x\sqrt{y-9}+y\sqrt{x-9}\right)=xy$. Tính giá trị của biểu thức
	$$S=(x-17)^{2018}+(y-19)^{2019}.$$
	\loigiai{
Điều kiện $\heva{&x\ge 9\\&y\ge 9.}$\\
Ta có 	
\begin{eqnarray*}
&&3\left(x\sqrt{y-9}+y\sqrt{x-9}\right)=xy\\
& \Leftrightarrow & \left(xy-6x\sqrt{y-9}\right) +\left(xy -6y\sqrt{x-9}\right)=0\\
&\Leftrightarrow & x\left(y-6\sqrt{y-9}\right)+y\left(x-6\sqrt{x-9}\right)=0\\
&\Leftrightarrow & x\left(\sqrt{y-9}-3\right)^2+y\left(\sqrt{x-9}-3\right)^2=0.\quad (*)
\end{eqnarray*}
Mà $\heva{&x\left(\sqrt{y-9}-3\right)^2 \geq 0 \\& y\left(\sqrt{x-9}-3\right)^2 \geq 0}$. Do đó $(*)$ xảy ra khi $\heva{& \sqrt{y-9}-3=0\\&\sqrt{x-9}-3=0} \Leftrightarrow \heva{&x=18\\&y=18.}$\\
Khi đó $S=(x-17)^{2018}+(y-19)^{2019}= (18-17)^{2018}+(18-19)^{2019}=0$.

}
\end{bt}
