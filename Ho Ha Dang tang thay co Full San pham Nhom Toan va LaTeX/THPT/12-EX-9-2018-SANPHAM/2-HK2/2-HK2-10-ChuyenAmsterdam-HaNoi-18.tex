\begin{ex}%[HK2, THPT Chuyên Amsterdam, Hà Nội]%[Bùi Quốc Hoàn, dự án EX9]%[2D4K5-2]
Cho số phức $z = x + y\mathrm{i}$ ($x, y\in \mathbb{R}$) có mô-đun nhỏ nhất thỏa mãn điều kiện $\left\vert z - 4 - 2\mathrm{i}\right\vert = \left\vert z - 2\right\vert$. Tính $P = x^2 + y^2$. 
\choice
	{$32$}
	{$16$}
	{\True $8$}
	{$10$}
	\loigiai{Ta có $\left\vert z - 4 - 2\mathrm{i}\right\vert  = \left\vert \left(x - 4\right) + \left(y - 2\right)\mathrm{i}\right\vert = \sqrt{\left(x - 4\right)^2 + \left(y - 2\right)^2}$.\\
Tương tự $\left\vert z - 2\right\vert  = \left\vert \left(x - 2\right) +  y\mathrm{i}\right\vert = \sqrt{\left(x - 2\right)^2 + y^2}$.\\
Do giả thiết ta có 
	 \begin{eqnarray*}
	 	\left\vert z - 4 - 2\mathrm{i}\right\vert = \left\vert z - 2\right\vert
	 	&\Leftrightarrow&\sqrt{\left(x - 4\right)^2 + \left(y - 2\right)^2} = \sqrt{\left(x - 2\right)^2 + y^2}\\
	 &\Leftrightarrow&\left(x - 4\right)^2 + \left(y - 2\right)^2 = \left(x - 2\right)^2 + y^2\\
	 &\Leftrightarrow&x^2 - 8x + 16 + y^2 - 4y +  4 =  x^2 - 4x + 4 + y^2\\
	 &\Leftrightarrow&x + y - 4  = 0 \Leftrightarrow y = 4 - x.
	 \end{eqnarray*}
	Khi đó $P = x^2 + \left(4 - x\right)^2 = 2x^2 - 8x + 16 = 2\left(x - 2\right)^2 + 8$.\\
	Vì $\left(x -2\right)^2\geq 0, \ \forall x\in\mathbb{R}$ nên $P\geq 8, \  \forall x\in\mathbb{R}$. Dấu đẳn thức xảy ra khi $x = 2$ suy ra $y = 2$. Vậy $\min P = 8$. 
	}
\end{ex}

\begin{ex}%[HK2, THPT Chuyên Amsterdam, Hà Nội]%[Bùi Quốc Hoàn, dự án EX9]%[2D3B3-1]

	\choice
	{$\dfrac{5}{6}$}

	{$1$}
	{$\dfrac{2}{3}$}

	Khi đó phương trình tiếp tuyến tại điểm $\left(1; 2\right)$ của đồ thị hàm số $y = x^2 + 1$ là
	$$y = y'\left(1\right)\left(x -1\right) + 2\Leftrightarrow y = 2x.$$
	Gọi $\Delta$ là đường thẳng có phương trình $y = 2x$. Xét phương trình tương giao của $\left(P\right)$ và $\Delta$ 
	$$x^2 - x + 2 = 2x\Leftrightarrow x^2  - 3x + 2 = 0\Leftrightarrow \hoac{&x = 1\\&x = 2}$$
Gọi $S$ là diện tích hình phẳng $\left(H\right)$ khi đó
$$S = \displaystyle\int\limits_{1}^{2}\left\vert \left(x^2 - x + 2\right) - 2x\right\vert\, \mathrm{d}x = \displaystyle\int\limits_{1}^{2}\left\vert x^2 - 3x + 2\right\vert\, \mathrm{d}x$$
Do $x^2 - 3x + 2 \leq 0,\  \forall x\in \left[1; 2\right]$ nên 
$$S = - \displaystyle\int\limits_{1}^{2}\left(x^2 - 3x + 2\right)\, \mathrm{d}x = - \left(\dfrac{x^3}{3} - \dfrac{3x^2}{2} + 2x\right)\Big\vert^{2}_{1} = - \left(\dfrac{2}{3} - \dfrac{5}{6}\right) =  \dfrac{1}{6}$$
}
\end{ex}
