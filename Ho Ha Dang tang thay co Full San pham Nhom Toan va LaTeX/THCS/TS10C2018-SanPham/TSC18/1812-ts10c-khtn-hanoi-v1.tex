\begin{ex}%[Đề thi vào 10, Chuyên KHTN Hà Nội, vòng 1, 2018]%[9D3G3] %[9D1K4]
\begin{enumerate}
\item Giải phương trình: \[x^2-x+2\sqrt{x^3+1}=2\sqrt{x+1}\]
\item Giải hệ phương trình: \[\heva{&xy+y^2=1+y\\&x^2+2y^2+2xy=4+x}\]
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Đặt $a=\sqrt{x+1}$ và $b=\sqrt{x^2-x+1}$, phương trình đã cho trở thành
\[b^2-1+2ab=2a.\]
hay 
\[(b-1)(b+1+2a)=0. \]
Từ đó, ta có $b=1$ (do $b+1+2a>0$), hay  
\[x^2-x+1=0\]
Giải ra ta được $x=0$ hoặc $x=1$. Vậy phương trình có hai nghiệm $x=0$ và $x=1$. 
\item Nhân phương trình thứ nhất với $2$ sau đó cộng với phương trình thứ hai ta  được 
\[ (x+2y)^2=x+2y+6.\]
hay \[(x+2y-3)(x+2y+2)=0.\]
Suy ra $x=3-2y$ hoặc $x=-2y-2.$
\begin{enumerate}
\item[•] Với $x=3-2y$, thay vào phương trình thứ nhất, ta được $(y-1)^2=0$. Từ đó, ta có $y=1$ và $x=1$. 
\item[•] Với $x=-2y-2$, thay vào phương trình thứ nhất ta được  $y^2+3y+1=0$. Từ đó, ta có $y=\dfrac{-3-\sqrt{5}}{2}$, $x=1+\sqrt{5}$ hoặc   $y=\dfrac{-3+\sqrt{5}}{2}$, $x=1-\sqrt{5}.$
\end{enumerate}
\end{enumerate}
}
\end{ex}


\begin{ex}%[Đề thi vào 10, Chuyên KHTN Hà Nội, vòng 1, 2018]%[9D5G3]%[9D5G1]
\begin{enumerate}
\item Tìm tất cả các cặp số nguyên $(x;y)$ thỏa mãn 
\[(x+y)(3x+2y)^2=2x+y-1.\]
\item Với $a,b$ là các số thực dương thay đổi thỏa mãn điều kiện $\sqrt{a+2b}=2+\sqrt{\dfrac{b}{3}}$, 	tìm giá trị nhỏ nhất của biểu thức  \[M=\dfrac{a}{\sqrt{a+2b}}+\dfrac{b}{\sqrt{b+2a}}.\]
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Đặt $a=3x+2y$ và $b=x+y$. Từ phương trình đã cho ta có \[ab^2=b-a-1.\]
hay \[a(b^2+1)=b-1.\]
Suy ra $b-1$ chia hết cho $b^2+1$. Từ đó, ta có $2=(b^2+1)-(b+1)(b-1)$ chia hết cho $b^2+1$. Suy ra $b^2+1 \in \{1;2\}$ hay $b \in \{0;1;-1\}$. 
\begin{enumerate}
\item[•] Với $b=0$, ta có $a=-1$. Giải ra ta được $x=2$ và $y=-3$. 
\item[•] Với $b=1$, ta có $a=0$. Giải ra ta được $x=1$ và $y=-1$. 
\item[•] Với $b=-1$, ta có $a=-1$. Giải ra ta được $x=1$ và $y=-2$. 
\end{enumerate}
\item Sử dụng BĐT Cauchy-Schwarz ta có 
\begin{align*}
M+\sqrt{\dfrac{b}{3}} &= \dfrac{a}{\sqrt{a+2b}} + \dfrac{b}{\sqrt{b+2a}} + \sqrt{\dfrac{b}{3}} \\
& = \dfrac{a^2}{a\sqrt{a+2b}} + \dfrac{b^2}{b\sqrt{b+2a}} + \dfrac{b^2}{b\sqrt{3b}} \\
& \geqslant \dfrac{(a+b+b)^2}{a\sqrt{a+2b}+b\sqrt{b+2a	}+b\sqrt{3b}}\\
& \geqslant  \dfrac{(a+2b)^2}{a\sqrt{a+2b}+b\sqrt{(1+1)(b+2a+3b)}}\\
&=\sqrt{a+2b}.
\end{align*}
Suy ra \[M \geqslant \sqrt{a+2b} -\sqrt{\dfrac{b}{3}}=2.\]
Dấu bằng xảy ra khi $a=b=3$ nên $\min M =2.$
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên KHTN Hà Nội, vòng 1, 2018]%[9H3K3]%[9H3G7]
Cho tam giác $ABC$ có đường tròn nội tiếp $(I)$ tiếp xúc với ba cạnh $BC,CA,AB$ lần lượt tại các điểm $D,E,F$. Gọi $K$ là hình chiếu vuông góc của $B$ trên đường thẳng $DE$, $M$ là trung điểm của đoạn thẳng $DF$. 
\begin{enumerate}
\item Chứng minh rằng hai tam giác $BKM$ và $DEF$ đồng dạng.
\item Gọi $L$ là hình chiếu vuông góc của $C$ trên đường thẳng $DF$, $N$ là trung điểm của đoạn thẳng $DE$. Chứng minh rằng hai đường thẳng $MK$ và $NL$ song song. 
\item Gọi $J,X$ lần lượt là trung điểm các đoạn thẳng $KL,ID$. Chứng minh rằng đường thẳng $JX$ vuông góc với $EF$.  
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}%[scale=0.8]
\tkzDefPoints{1/4/A,0/-1/B,6/-1/C}
\tkzInit[xmin=-1,xmax=7,ymin=-3.5,ymax=4.5] 
\tkzClip
\tkzInCenter(A,B,C) \tkzGetPoint{I}
\tkzDefPointBy[projection=onto B--C](I) \tkzGetPoint{D}
\tkzDefPointBy[projection=onto A--C](I) \tkzGetPoint{E}
\tkzDefPointBy[projection=onto B--A](I) \tkzGetPoint{F}
\tkzInterLL(I,B)(D,F) \tkzGetPoint{M}
\tkzInterLL(I,C)(D,E) \tkzGetPoint{N}
\tkzDefPointBy[projection=onto D--E](B) \tkzGetPoint{K}
\tkzDefPointBy[projection=onto D--F](C) \tkzGetPoint{L}
\tkzDefMidPoint(I,D) \tkzGetPoint{X}
\tkzDefMidPoint(K,L) \tkzGetPoint{J}
\tkzDrawPoints[fill=black,size=6](A,B,C,D,E,F,M,N,I,K,L,X,J)
\tkzDrawSegments(A,B B,C C,A I,D I,E I,F I,B I,C  E,K F,L B,K C,L X,J M,K K,L)
\tkzDrawCircle(I,D)
\tkzLabelPoints[below](B,C,K,J,L,D)
\tkzLabelPoints[right](N,E)
\tkzLabelPoints[left](M,F,X)
\tkzLabelPoints[above](A,I)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Ta có $BD$, $BF$ là các tiếp tuyến của đường tròn $(I)$ nên $BI$ là trung trực của $DF$, suy ra $BI$ vuông góc với $DF$ tại $M$. Từ đó ta có tứ giác $BMDK$ nội tiếp, suy ra 
\[\angle BMK = \angle BDK =\angle CDE.\]
Mà $DC$ là tiếp tuyến của đường tròn $(I)$ nên $\angle CDE = \angle DFE$. Kết hợp với kết quả ở trên, ta được $\angle BMK = \angle DFE$. Lại có  $\angle BKM = \angle BDM =\angle DEF$ nên các tam giác $BKM$ và $DEF$ đồng dạng (g.g). 
\item Ta có các tứ giác $BKDM$ và $CLDN$ nội tiếp  nên 
\[ \angle DMK = \angle DBK, 	\quad 	\angle DCN = \angle DLN.\]
Mà $BK \parallel  CN$ nên $\angle DBK = \angle DCN$. Từ đó ta có 
\[\angle DMK = \angle DLN.\]
suy ra $MK \parallel LN$ (ĐPCM).
\item Ta có \[\angle DMK =\angle DCN =90^\circ - \angle CDN =90^\circ - \angle DFE  =90^\circ - \angle DMN \]
nên $\angle KMN =90^\circ$. Từ 	đây dễ dàng suy ra tứ giác $KMNL$ là hình thang vuông. Mà $J$ là trung điểm của $KL$ nên $JM=JN$. Lại có $XM=XN=\frac{1}{2}ID$ nên $JX \perp MN$. Nói cách khác $JX \perp EF$. Đây chính là 	điều phải chứng minh.  
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên KHTN Hà Nội vòng 1, 2018]%[9H5G4]
Trên mặt phẳng cho hai điểm $P,Q$ phân biệt. Xét $10$ đường thẳng nằm trong mặt phẳng trên thỏa mãn các tính chất sau:
\begin{enumerate}
\item[a)] Không có hai đường thẳng nào song song hoặc trùng nhau.
\item[b)] Mỗi đường thẳng đi qua $P$ hoặc $Q$, không có đường thẳng nào đi qua cả $P$ và $Q$. Hỏi 	$10$ đường thẳng trên có thể chia mặt phẳng thành tối đa bao nhiêu miền? Hãy giải thích.	   
\end{enumerate}  
\loigiai{
Gọi $m,n$ 	lần lượt là số đường đi qua $P$ và $Q$, $S$ là số miền được tạo thành, khi đó từ giả thiết ta có 
\[m+n=10.\]
Nếu $m=0$ hoặc $n=0$ thì dễ thấy $S=20.$ \\
Xét trường hợp $m,n \geqslant 1$. Bắt đầu từ mặt phẳng với hai điểm $P,Q$, ta vẽ 	$m$ đường thẳng đôi một phân biệt đi qua $P$, số miền được tạo thành là $2m$.  \\
Ta lần lượt vẽ thêm các đường thẳng đi qua $Q$. Khi vẽ đường thẳng đầu tiên, nó cắt $m$ đường thẳng đi qua $P$ tại $m$ điểm phân biệt, $m$ điểm này chia 	đường thẳng vừa vẽ thành $m+1$ phần. Nói cách khác, đường thẳng vừa vẽ  đi qua đúng $m+1$ miền 	trong $2m$ miền được tạo ra, nên lúc này số miền là $2m + (m+1)$. \\ 
Kể từ đường thẳng thứ hai đến đường thẳng thứ $n$. Mỗi đường thẳng đi qua $Q$ được vẽ sẽ cắt $m$  đường thẳng đi qua $P$ tại $m$ điểm phân biệt khác $Q$. Chúng cùng với $Q$, chia đường thẳng vừa vẽ thành $m+2$ phần. Do đó, với mỗi lần vẽ, số miền tăng thêm $m+2$. Tóm lại, ta có 
$$S=2m+(m+1)+(n-1)(m+2)=mn+2m+2n-1=mn+19.$$      	   
Sử dụng BĐT AM-GM, ta có 
\[S \leqslant  \dfrac{(m+n)^2}{4}+19=44.\]
Dấu đẳng thức xảy ra khi có $5$ đường thẳng đi qua $P$ và $5$ đường thẳng đi qua $Q$. Số miền được tạo tối đa là $44$. 	 	 
}
\end{ex}
