\subsection{Kiến thức cơ sở}
\begin{dl}
Hàm số $y=f(x)$ liên tục trên đoạn $[a;b]$ và có đạo hàm đến cấp hai trên khoảng $(a;b)$. Nếu $f(a)f(b)<0$ và $f'(x)$, $f''(x)$ cùng một dấu trên khoảng $(a;b)$ thì phương trình $f(x)=0$ có nghiệm duy nhất $c\in (a;b)$ và dãy số $(x_n)$ được xác định bởi $x_0 =b, \, x_{n+1}=x_n -\dfrac{f(x_n)}{f'(x_n)}, \, \forall n\in \mathbb{N}$ hội tụ về $c$.
\end{dl}
\textbf{\textit{Chứng minh$\colon$}}
Không mất tính tổng quát, giả sử đồ thị hàm số $y=f(x)$ liên tục trên đoạn $[a;b]$ và có đạo hàm đến cấp hai trên khoảng $(a;b)$. Nếu $f(a)<0$, $f(b)>0$ và $f'(x) >0$, $f"(x)>0$ trên khoảng $(a;b)$.
\begin{enumerate}[1)]
\item Chứng minh dãy bị chặn$\colon$ $c\le x_n,\, \forall n\in\mathbb{N}.\hfill (1)$
\begin{list}{-}{}
\item Ta có $x_0 =b$ thỏa mãn $(1)$.
\item Giả sử $c<x_n$ (nếu $x_i =c \Rightarrow x_n =c, \forall n\in \mathbb{N}$)
\item Áp dụng định lý Lagrange, ta có$\colon$
\begin{align*}
x_{n+1}&=x_n+\dfrac{f(c)-f(x_n)}{f'(x_n)},\, (\text{do}\, f(c)=0)\\
&=x_n+\dfrac{(c-x_n)f'(\xi)}{f'(x_n)}, (\text{với}\, \xi \in (c;x_n))
\end{align*}
\item Mặt khác $f'(x)>0$, $f"(x)>0$, $\forall x\in (a;b) \Rightarrow 0<f'(\xi )<f'(x_n) \Rightarrow \dfrac{f'(\xi )}{f'(x_n)}<1$.\\
Và $c-x_n <0 \Rightarrow x_{n+1}>x_n +(c-x_n)\cdot a =c$. \\
Vậy $(1)$ đúng với mọi số tự nhiên.
\end{list}
\item Theo giả thiết hàm số $y=f(x)$ liên tục trên đoạn $[a;b]$, đồng biến trên khoảng $(a;b)$.

Ta suy ra $f(x_n) >f(c) =0,\, \forall n\in \mathbb{N}$ và $f'(x_n) >0, \forall n\in\mathbb{N}$, nên ta có $\dfrac{f(x_n)}{f'(x_n)}>0$.\\
Do đó từ công thức xác định dãy, ta suy ra $x_{n+1}-x_n =-\dfrac{f(x_n)}{f'(x_n)}<0\Rightarrow x_{n+1}<x_n,\forall n\in\mathbb{N}$.

Vậy dãy số $(x_n)$ là dãy số giảm.$\hfill (2)$

Từ $(1)$ và $(2)$ dãy $(x_n)$ giảm và bị chặn dưới bởi $c$, suy ra $\exists \displaystyle\lim x_n =\alpha \ge c$.\\
Từ công thức xác định dãy số cho qua giới hạn, ta có $\colon$
 $\alpha =\alpha -\dfrac{f(\alpha )}{f'(\alpha )}\Leftrightarrow f(\alpha )=0 \Leftrightarrow \alpha =c$.
\end{enumerate}

\begin{dl}
Hàm số $y=f(x)$ liên tục trên đoạn $[a;b]$ và có đạo hàm đến cấp hai trên khoảng $(a;b)$. Nếu $f(a)f(b)<0$ và $f'(x)$, $f''(x)$ trái dấu trên khoảng $(a;b)$ thì phương trình $f(x)=0$ có nghiệm duy nhất $c\in (a;b)$ và dãy số $(x_n)$ được xác định bởi $x_0 = a,\, x_{n+1}=x_n-\dfrac{f(x_n)}{f'(x_n)}, \forall n\in\mathbb{N}$ hội tụ về $c$.
\end{dl}
\textbf{\textit{Chứng minh$\colon$}}
Không mất tính tổng quát ta giả sử đồ thị hàm số $y=f(x)$ liên tục trên đoạn $[a;b]$ và có đạo hàm cấp hai trên khoảng $(a;b)$. Nếu $f(a) >0$, $f(b)<0$ và $f'(x) <0$, $f"(x) >0 $ trên khoảng $(a;b)$.
\begin{enumerate}[1)]
\item Chứng minh dãy bị chặn$\colon$ $x_n <c, \forall n\in \mathbb{N}\hfill (3)$
\begin{list}{-}{}
\item Ta có $x_0 =a$ thỏa mãn $(3)$.
\item Giả sử $x_n <c, n\in\mathbb{N}$
\item Áp dụng định lý Lagrange, ta có
\begin{align*}
x_{n+1}&=x_n +\dfrac{f(c)-f(x_n)}{f'(x_n)}, (\text{do}\, f(c) =0)\\
&=x_n +\dfrac{f(c)-f(x_n)}{(c-x_n)}\cdot\dfrac{c-x_n}{f'(x_n)}\\
&=x_n+\dfrac{(c-x_n)f'(\xi )}{f'(x_n)}, (\text{với} \,\, \xi \in (x_n; c)).
\end{align*}
\item Mặt khác $f''(x) >0$, $f'(x)<0, \forall x\in (a;b)\Rightarrow 0>f'(c)>f'(\xi )>f'(x_n) \Rightarrow \dfrac{f'(\xi )}{f'(x_n)}<1$ và $c-x_n >0 \Rightarrow x_{n+1}<x_n +(c-x_n) \cdot 1=c$.\\
Vậy $(3)$ đúng với mọi số tự nhiên $n$.
\end{list}
\item Theo giả thiết hàm số $y=f(x)$ liên tục trên đoạn $[a;b]$, nghịch biến trên khoảng $(a;b)$.

Ta suy ra $f(x_n) >f(c) =0,\forall n\in\mathbb{N}$ và $f'(x_n) <0, \forall n\mathbb{N}$, nên ta có $\dfrac{f(x_n}{f'(x_n)}<0$.\\
Do đó từ công thức xác định dãy, ta suy ra  $x_{n+1}=x_n=-\dfrac{f(x_n)}{f'(x_n)}>0\Rightarrow x_{n+1}>x_n, \forall n\in \mathbb{N}$.\\
Vậy dãy số $(x_n)$ dãy số tăng \hfill $(4)$\\
Từ $(3)$ và $(4)$ dãy số $(x_n)$ tăng và bị chặn trên bởi $c$, suy ra $\exists\displaystyle\lim x_n =\alpha \le c$.\\
Từ công thức xác định dãy số cho qua giới hạn, ta có$\colon$ $\alpha =\alpha -\dfrac{f(\alpha}{f'(\alpha )}\Leftrightarrow f(\alpha ) =0 \Leftrightarrow \alpha =c$.
\end{enumerate}

\subsection{Bài tập áp dụng}
Ứng dụng hai định lý trên, chúng ta vận dụng vào các hàm số $f(x)$ cụ thể ta sẽ xây dựng được các bài toán về dãy số hội tụ.\\
\textcolor{blue}{Ta xét hàm số $f(x) =x^2-2x, x \in \mathbb{R}$. Ta có $f'(x)=2x-2$, $f''(x)=2>0$, $f'(x) >0$ khi $x>1$; $f'(x)<0$ khi $x<1$.}
\begin{bt}%[Nguyễn Trường Sơn,Lê Xuân Hòa, Dự án EX_TapChi05]%[1D0G1]
Cho dãy số $(x_n)$ được xác định bởi $x_0=a\in\mathbb{R},\, x_{n+1}=\dfrac{x_n^2}{2(x_n-1)},\forall n\in\mathbb{N}$.
Xác định số thực $a$ để dãy số $(x_n)$ tồn tại và có giới hạn. Tìm giới hạn của dãy số $(x_n)$ với giá trị $a$ tìm được.
\loigiai{ Ta xét các trường hợp sau
\begin{enumerate}[1)]
\item Nếu $a=1$ thì dãy số $(x_n)$ không tồn tại.
\item Nếu $a<1$ thì dễ dàng chứng minh $x_{n+1}=\dfrac{x_n^2}{2(x_n-1)}\le 0,\forall n\in\mathbb{N}$.\\
Ta xét $x_{n+1}-x_n =-\dfrac{x_x(x_n-2)}{2(x_n-1)}\ge 0$, (do $x_n\le 0,\forall n\in\mathbb{N^*})\Rightarrow x_{n+1}\ge x_n, \forall n\in\mathbb{N^*}$\\
Vậy dãy số tăng và bị chặn trên bởi số $0$, nên $\exists\displaystyle\lim x_n =\alpha \in (-\infty ; 0].\hfill (5)$\\
Từ công thức xác định dãy số cho qua giới hạn, ta có$\colon$
$$\alpha =\dfrac{\alpha ^2}{2(\alpha -1)}\Leftrightarrow \hoac{&\alpha =0\\&\alpha =2}\Leftrightarrow \alpha =2\quad (\text{do} (5)).$$
\item Nếu $a>1$. Xét hàm só $f(x)=\dfrac{x^2}{2(x-1)}>0,\forall x>1$. Ta có \\
$f'(x)=\dfrac{x^2-2x}{2(x-1)^2},\forall x>1\Rightarrow f'(x) =0\Leftrightarrow x=2>1\Rightarrow f(x)\ge f(2)=2, \forall x>1$.
\end{enumerate}
Do vậy, ta có $ x_{n+1}=f(x_{n} \geq 2, \forall n \in \mathbb{N}$.\\
Xét: $x_{n+1}-x_n=-\dfrac{x_n(x_n-2)}{2(x_n-1)}\le 0$, (do $x_n \ge 2, \forall n\in \mathbb{N^*})\Rightarrow x_{n+1}\le x_n, \forall n\ge 2$.\\
Vậy dãy số giảm và bị chặn dưới bởi số $2$, nên $\exists\displaystyle\lim x_n =\alpha \ge 2.\hfill (6)$\\
Từ công thức xác định dãy cho qua giới hạn, ta có
$$\alpha =\dfrac{\alpha ^2}{2(\alpha -1)}\Leftrightarrow \hoac{&\alpha =0\\&\alpha =2}\Leftrightarrow \alpha =2\quad (\text{do} (6)).$$
Vậy $\displaystyle\lim x_n =2$.
}
\end{bt}

\begin{bt}
Cho dãy số $(x_n)$  được xác định bởi $$x_1=a>0, x_{n+1}=\dfrac{n}{2n-1}\cdot\dfrac{x_n^2+2}{x_n},\forall n\in\mathbb{N^*}.$$
Chứng minh dãy số $(x_n)$ có giới hạn. Tìm giới hạn đó

\begin{bt}%[Nguyễn Trường Sơn,Lê Xuân Hòa, Dự án EX_TapChi05]%[1D0G1]
Cho $a$ là số thực dương và dãy số $(x_n)$ được xác định bởi $$x_1=a,\, x_{n+1}=\dfrac{x_n-2\sqrt{x_n^2+3}}{x_n^2+4},\forall n\in\mathbb{N^*}$$
Chứng minh rằng dãy số $(x_n)$ có giới hạn hữu hạn. Tìm giới hạn đó.
\loigiai{ Ta có $2\sqrt{x_n^2+3}>\sqrt{x_n^2+3}>\sqrt{x_n^2}=|x_n|\ge x_n, \forall x\in\mathbb{R}\Rightarrow x_n -2\sqrt{x_n^2+3}<0, \forall x_n\in\mathbb{R}.$\\
Từ đó suy ra $x_n <0, \forall n\ge 2$. Ta chứng minh
\begin{align*}
x_{n+1}&=\dfrac{x_n-2\sqrt{x_n^2+3}}{x_n^2+4}\ge -1, \forall n\ge 2\\
&\Leftrightarrow  x_n^2+x_n+4 \ge 2\sqrt{x_n^2+3}. \tag{9}
\end{align*}
Do hai vế của $(9)$ đều dương, nên $(9)$ tương đương với \\
$$\left(x_n^2+x_n+4\right)^2 \ge \left( 2\sqrt{x_n^2+3}\right)^2\Leftrightarrow (x_n+1)(4x_n^2+2x_n+8)\ge 0\, (\text{dúng}).$$
Do đó $-1\le x_n<0, \forall n\ge 2.\hfill (10)$\\
Ta chứng minh dãy số $(x_n)$ là dãy số giảm. Áp dụng $(10)$, ta xét $$x_{n+1}-x_n = \dfrac{x_n-2\sqrt{x_n^2+3}}{x_n^2+4}-x_n =-\dfrac{x_n^3+3x_n+2\sqrt{x_n^2+3}}{x_n^2+4}$$
Ta chứng minh
\begin{align*}
& x_n^3+3x_n+2\sqrt{x_n^2+3} >0, \forall x_n \in [-1;0) \\
\Leftrightarrow \quad & x_n\sqrt{x_n^2+3}+2\ge 0\\
\Leftrightarrow \quad & 2\ge -x_n\sqrt{x_n^2+3}\\
\Leftrightarrow \quad & x_n^4+3x_n^2-4\le 0\\
\Leftrightarrow \quad &(x_n^2-1)(x_n^2+4)\le 0, \forall x_n \in [-1;0).
\end{align*}
Vậy dãy số $(x_n)$ là dãy số giảm và bị chặn dưới bởi $-1$, nên $\exists\displaystyle\lim x_n =l\in [-1;0)$.\\

\begin{align*}
&l=\dfrac{l-2\sqrt{l^2+3}}{l^2+4}\Leftrightarrow \sqrt{l^2+3}\left( l\sqrt{l^2+3}+2\right) =0\\
&\Leftrightarrow l\sqrt{l^2+3}+2=0\Leftrightarrow l=-1\in [-1;0)
\end{align*}
Vậy $\displaystyle\lim x_n =-1$.
}
\end{bt}
