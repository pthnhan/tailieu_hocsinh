\begin{ex}%[Đề thi HSG 9, tỉnh Trà Vinh, năm 2018]%[Nguyễn Trần Anh Tuấn, Dự án EX-9G-18]%[9D0G2]
	Giải hệ phương trình
	\begin{center}
		$\heva{&\dfrac{y^2-x^2}{\sqrt{x-1}}=2y-\sqrt{x-1}\\& x^2+y^2=3x-1.}$
	\end{center}
	\loigiai{
		Điều kiện: $x>1.$\\
		Ta có \begin{align*}
		\dfrac{y^2-x^2}{\sqrt{x-1}}=2y-\sqrt{x-1}&\Leftrightarrow y^2-x^2=2y\sqrt{x-1}-(x-1)\\
		&\Leftrightarrow y^2-2y\sqrt{x-1}+x-1=x^2\\
		&\Leftrightarrow \left(y-\sqrt{x-1}\right)^2=x^2\\
		&\Leftrightarrow \hoac{& y=x+\sqrt{x-1}\\& y=-x+\sqrt{x-1}.}
		\end{align*}
		Trường hợp 1. Với $y=x+\sqrt{x-1}$, thay vào phương trình thứ hai của hệ ta có
			\begin{center}
				$\begin{array}{lrcl}
				&x^2+x^2+2x\sqrt{x-1}+x-1&=&3x-1\\
				\Leftrightarrow &(2x^2-2x+2x\sqrt{x-1}&=&0\\
				\Leftrightarrow &(x-1)+\sqrt{x-1}&=&0 \text{   (vì $x>1$)}\\
				\Leftrightarrow &\sqrt{x-1}\left(\sqrt{x-1}+1\right)&=&0\\
				\Leftrightarrow &x&=&1.\text{  (loại)}\\
				\end{array}$
			\end{center}
		Trường hợp 2. Với $y=-x+\sqrt{x-1}$, thay vào phương trình thứ hai của hệ ta có
			\begin{center}
				$\begin{array}{lrcl}
				&x^2+x^2-2x\sqrt{x-1}+x-1&=&3x-1\\
				\Leftrightarrow &2x^2-2x-2x\sqrt{x-1}&=&0\\
				\Leftrightarrow &(x-1)-\sqrt{x-1}&=&0 \text{   (vì $>1$)}\\
				\Leftrightarrow &\sqrt{x-1}\left(\sqrt{x-1}-1\right)&=&0\\
				\end{array}$\\
				$\Leftrightarrow \hoac{&x=1\\&\sqrt{x-1}=1}	\Leftrightarrow \hoac{&x=1\text{  (loại)}\\&x=2\Rightarrow y=-1.}$\\
			\end{center}	
		Vậy nghiệm của hệ phương trình là $(2;-1).$	
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Trà Vinh, năm 2018]%[Nguyễn Trần Anh Tuấn, Dự án EX-9G-18]%[9D0K2]
	Dân số xã $A$ hiện nay có $10.000$ người. Người ta dự đoán sau hai năm dân số xã $A$ là $10.404$ người. Hỏi trung bình hằng năm dân số xã $A$ tăng bao nhiêu phần trăm?
	\loigiai{
		Gọi $x$ (\%) là tỉ lệ tăng dân số hằng năm của tỉnh $A$. $(0\le x\le 100)$\\
		Dân số tỉnh $A$ sau năm thứ nhất là $10.000(1+x\%)$ (người).\\
		Dân số tỉnh $A$ sau năm thứ hai là $10.000(1+x\%)(1+x\%)=10.000(1+x\%)^2$ (người).\\
		Theo đề, ta có phương trình
		\begin{center}
		$\begin{array}{lrcl}
		&10.000(1+x\%)^2&=&10.404\\
		\Leftrightarrow &(1+x\%)^2&=&1{,}0404\\
		\Leftrightarrow &1+x\%&=&1{,}02\\
		\Leftrightarrow &x\%&=&0{,}02\\
		\Leftrightarrow &x&=&2. \textrm{  (nhận)}\\
		\end{array}$
		\end{center}
		Vậy tỉ lệ tăng dân số hằng năm của tỉnh $A$ là $2\%$.	
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Trà Vinh, năm 2018]%[Nguyễn Trần Anh Tuấn, Dự án EX-9G-18]%[9D0G1]
	Cho ba số dương $x,y,z$ thỏa mãn điều kiện $xy+yz+zx=1$. Tính giá trị của biểu thức
	$$A=x\sqrt{\dfrac{(1+y^2)(1+z^2)}{1+x^2}}+y\sqrt{\dfrac{(1+z^2)(1+x^2)}{1+y^2}}+z\sqrt{\dfrac{(1+x^2)(1+y^2)}{1+z^2}}.$$
	\loigiai{
		Ta có $xy+yz+zx=1\Leftrightarrow 1+x^2=xy+yz+zx+x^2=(x+y)(x+z). \refstepcounter{equation}\hfill (\theequation)\label{1843.1}$\\
		Chứng minh tương tự ta có
		\begin{equation}\label{1843.2} 1+y^2=(x+y)(y+z), \end{equation}
		\begin{equation}\label{1843.3} 1+z^2=(x+z)(y+z). \end{equation}
		Thay \eqref{1843.1}, (\ref{1843.2}) và (\ref{1843.3}) vào $A$ ta được
		\begin{align*}
		A&=x\sqrt{\dfrac{(x+y)(x+z)(y+z)^2}{(x+y)(x+z)}}+y\sqrt{\dfrac{(x+y)(y+z)(x+z)^2}{(x+y)(y+z)}}+z\sqrt{\dfrac{(x+z)(y+z)(x+y)^2}{(x+z)(y+z)}}\\
		&=x\sqrt{(y+z)^2}+y\sqrt{(x+z)^2}+z\sqrt{(x+y)^2}\\
		&=x(y+z)+y(x+z)+z(x+y)\\
		&=2(xy+yz+xz)\\
		&=2.
		\end{align*}
	Vậy với điều kiện đã cho, giá trị của biểu thức $A$ là $2$.
	}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Trà Vinh, năm 2018]%[Nguyễn Trần Anh Tuấn, Dự án EX-9G-18]%[9D0G7]
	Cho $a,b,c$ là các số dương thỏa mãn $a+b+c=1$. Tìm giá trị lớn nhất của
	$$P=\sqrt{\dfrac{ab}{c+ab}}+\sqrt{\dfrac{bc}{a+bc}}+\sqrt{\dfrac{ca}{b+ca}}.$$
	\loigiai{
		Ta có $$\sqrt{\dfrac{ab}{c+ab}}=\sqrt{\dfrac{ab}{c^2+ac+bc+ab}}=\sqrt{\dfrac{ab}{(a+c)(b+c)}}\le \dfrac{1}{2}\left(\dfrac{a}{a+c}+\dfrac{b}{b+c}\right),$$
		$$\sqrt{\dfrac{bc}{a+bc}}=\sqrt{\dfrac{bc}{a^2+ab+ac+bc}}=\sqrt{\dfrac{bc}{(a+b)(a+c)}}\le \dfrac{1}{2}\left(\dfrac{b}{a+b}+\dfrac{c}{a+c}\right),$$
		$$\sqrt{\dfrac{ca}{b+ca}}=\sqrt{\dfrac{ca}{b^2+ab+bc+ca}}=\sqrt{\dfrac{ca}{(b+c)(a+b)}}\le \dfrac{1}{2}\left(\dfrac{a}{b+c}+\dfrac{c}{a+b}\right).$$
		Suy ra
		\begin{align*} P&=\sqrt{\dfrac{ab}{c+ab}}+\sqrt{\dfrac{bc}{a+bc}}+\sqrt{\dfrac{ca}{b+ca}}\\&\le \dfrac{1}{2}\left(\dfrac{a}{a+c}+\dfrac{b}{b+c}\right)+\dfrac{1}{2}\left(\dfrac{b}{a+b}+\dfrac{c}{a+c}\right)+\dfrac{1}{2}\left(\dfrac{c}{b+c}+\dfrac{a}{a+b}\right)\\
		&\le \dfrac{1}{2}\left(\dfrac{a}{a+c}+\dfrac{c}{a+c}+\dfrac{b}{b+c}+\dfrac{c}{b+c}+\dfrac{b}{a+b}+\dfrac{a}{a+b}\right)\\
		&\le \dfrac{3}{2}.
		\end{align*}
		Vậy giá trị lớn nhất của $P$ là $\dfrac{3}{2}$.\\
		Dấu \textquotedblleft=\textquotedblright \;xảy ra khi $a=b=c=1$.
	}
\end{ex}
\begin{ex}[9D0G5]%[Đề thi HSG 9, tỉnh Trà Vinh, năm 2018]%[Nguyễn Trần Anh Tuấn, Dự án EX-9G-18]%[9D0G5]
	Tìm nghiệm nguyên của phương trình $x^2+2y^2+3xy+3x+5y=15.$
	\loigiai{
		Ta có
		\begin{center}
			$\begin{array}{lrcl}
			&x^2+2y^2+3xy+3x+5y&=&15\\
			\Leftrightarrow &4x^2+8y^2+12xy+12x+20y&=&60\\
			\Leftrightarrow &4x^2+12(y+1)x+9(y+1)^2-y^2+2y-1&=&68\\
			\Leftrightarrow &(2x+3y+3)^2-(y-1)^2&=&68\\
			\Leftrightarrow &(2x+2y+4)(2x+4y+2)&=&68\\
			\Leftrightarrow &((x+y+2)(x+2y+1)&=&17\\
			\Leftrightarrow& \hoac{& \heva{&x+y+2=1\\&x+2y+1=17}\\&\heva{&x+y+2=17\\&x+2y+1=1}\\&\heva{&x+y+2=-1\\&x+2y+1=-17}\\&\heva{&x+y+2=-17\\&x+2y+1=-1}}		\Leftrightarrow \hoac{& \heva{&x=-18\\&y=17}\\&\heva{&x=30\\&y=-15}\\&\heva{&x=12\\&y=-15}\\&\heva{&x=-36\\&y=17.}}
			\end{array}$
		\end{center}
		Vậy các nghiệm của phương trình là $(-18;17),(30;-15),(12;-15),(-36;17).$
	}
\end{ex}
\begin{ex}[9D0G7]%[Đề thi HSG 9, tỉnh Trà Vinh, năm 2018]%[Nguyễn Trần Anh Tuấn, Dự án EX-9G-18]%[9D0G7]
	Cho $x,y$ là các số dương thỏa mãn $x+y=2$. Chứng minh $x^3y^3(x^3+y^3)\le2.$
	\loigiai{
		Ta có $x^3y^3(x^3+y^3)=x^3y^3(x+y)(x^2+y^2-xy)=2x^3y^3[(x+y)^2-3xy]=8x^3y^3-6x^4y^4$.\\
		Ta cần chứng minh: $2-x^3y^3(x^3+y^3) \ge 0 \Leftrightarrow 3x^4y^4-4x^3y^3+1 \ge 0$.\\
		Ta có
		\begin{align*}
		\text{Vế trái}&=3x^3y^3(xy-1)-(x^3y^3-1)\\
		&=(xy-1)(3x^3y^3-x^2y^2+xy-1)\\
		&=(xy-1)\left[(x^3y^3-x^2y^2)+(x^3y^3-xy)+x^3y^3-1\right]\\
		&=(xy-1)^2(x^2y^2+x^2y^2+xy+x^2y^2+xy+1)\\
		&=(xy-1)^2(3x^2y^2+2xy+1)\ge 0, \forall x,y>0.
		\end{align*}
		Vậy $x^3y^3(x^3+y^3)\le 2,$ $\forall x,y>0$.\\
		Dấu \textquotedblleft=\textquotedblright \;xảy ra khi $x=y=1$.
	}
\end{ex}
\begin{ex}[9H0G1]%[Đề thi HSG 9, tỉnh Trà Vinh, năm 2018]%[Nguyễn Trần Anh Tuấn, Dự án EX-9G-18]%[9H0G1]
	Cho tam giác $ABC$ có ba góc nhọn nội tiếp trong đường tròn $(O)$ và có $AB<AC$. Lấy điểm $M$ thuộc cung $\wideparen {BC}$ không chứa điểm $A$ của đường tròn $(O)$. Vẽ $MH$ vuông góc với $BC,$ $MK$ vuông góc với $CA,$ $MI$ vuông góc với $AB$ ($H$ thuộc $BC,$ $K$ thuộc $AC,$ $I$ thuộc AB). Chứng minh $$\dfrac{BC}{MH}=\dfrac{AC}{MK}+\dfrac{AB}{MI}.$$
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=1]
			\tkzInit[xmin=-1,xmax=8,ymin=-4,ymax=5]
			\tkzClip
			\tkzDefPoints{2/4/A, -0/0/B, 7/0/C, 3/-2.79/M}
			\tkzDrawPolygon(A,B,C)
			\tkzCircumCenter(A,B,C)		\tkzGetPoint{O}
			\tkzDrawCircle[radius](O,A)
			\tkzDrawAltitude(B,C)(M)	\tkzGetPoint{H}
			\tkzDrawAltitude(A,C)(M)	\tkzGetPoint{K}
			\tkzDrawAltitude(A,B)(M)	\tkzGetPoint{I}
			\tkzDrawAltitude(I,K)(M)	\tkzGetPoint{T}
			\tkzDrawPoints[fill=black](O,A,B,C,M,H,I,K,T)
			\tkzMarkRightAngles[size=0.3,fill=gray!50](M,H,C M,K,C M,I,B M,T,H)
			%\tkzMarkAngles[size=0.8cm,arc=ll](M,I,K M,B,C)
			\tkzLabelPoints[above](O,A,H)
			\tkzLabelPoints[below](M)
			\tkzLabelPoints[above right](K)
			\tkzLabelPoints[above left](T)
			\tkzLabelPoints[right](C)
			\tkzLabelPoints[left](B,I)
			\tkzDrawSegments(B,I I,K M,C A,M M,B)
			%\tkzMarkRightAngles(A,D,B C,F,B B,E,A B,M,O)
			\end{tikzpicture}
		\end{center}
		$\widehat{MHC}=\widehat{MKC}=90^\circ \Rightarrow $ tứ giác $MHKC$ nội tiếp $\Rightarrow \widehat{IBM}=\widehat{ACM}$ (góc ngoài bằng góc đối trong)\\
		$\widehat{MHB}=\widehat{MIB}=90^\circ \Rightarrow $ tứ giác $MIBH$ nội tiếp $\Rightarrow \widehat{IHM}=\widehat{IBM}$ (cùng nhìn cạnh $IM$).\\
		Suy ra $\widehat{IHM}=\widehat{ACM}$.\\
		Do đó: $\widehat{IHM}+\widehat{MHK}=\widehat{ACM}+\widehat{MHK}=180^\circ$ (tứ giác $MHKC$ nội tiếp).\\
		Suy ra $I$, $H$, $K$ thẳng hàng. \\
		Xét $\triangle MIK $ và $\triangle MBC$ có\\
		$\widehat{KIM}=\widehat{CBM}$ ($IBHM$ nội tiếp),\\
		$\widehat{IKM}=\widehat{BCM}$ ($MHKC$ nội tiếp),\\
		$\Rightarrow \triangle MIK \backsim \triangle MBC$ (g-g).\\
		Hạ $MT \bot IK$ ($T\in IK$). Vì $\triangle MIK \backsim \triangle MBC$, nên ta có
		$$ \dfrac{BC}{IK}=\dfrac{MH}{MT} \Leftrightarrow \dfrac{BC}{MH}=\dfrac{IK}{MY}=\dfrac{IH}{MT}+\dfrac{HK}{MT}$$
		Xét $\triangle IHM $ và $\triangle ACM$ có\\
		$\widehat{HIM}=\widehat{CAM}$ (cùng bằng $\widehat{MBC}$),\\
		$\widehat{IHM}=\widehat{ACM}$ (cmt),\\
		$\Rightarrow \triangle IHM \backsim \triangle ACM$ (g-g).\\
		Suy ra $\dfrac{IH}{AC}=\dfrac{MT}{MK} \Leftrightarrow \dfrac{IH}{MT}=\dfrac{AC}{MK}$.\\ Tương tự ta có: $\dfrac{HK}{MT}=\dfrac{AB}{MI}$.\\
		Vậy $\dfrac{BC}{MH}=\dfrac{AC}{MK}+\dfrac{AB}{MI}$.
	}
\end{ex}
