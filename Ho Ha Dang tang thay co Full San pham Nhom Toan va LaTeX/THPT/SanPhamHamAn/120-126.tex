\begin{ex}%[WTT Bài toán hàm ẩn, Nguyễn Kiều Nhã Tú]%[2D1G3-1]
	Cho hàm số $y = f(x)$ có đồ thị $y = f'(x)$ như hình vẽ. 
	
	\begin{center}
		\begin{tikzpicture}[>=stealth,scale=0.45]
		\draw[->] (-4,0) -- (4,0) node[above] {\footnotesize$x$};
		\draw[->] (0,-14) -- (0,5) node[right] {\footnotesize$y$};
		\draw[fill=black] (0,2) circle (3pt);
		\draw[fill=black] (-2.98,-13) circle (3pt);
		\draw[fill=black] (2.98,-13) circle (3pt);
		\draw[smooth,samples=100,domain=-3:3] plot(\x,{-0.31640625*(\x)^4+1.125*(\x)^2+2});
		\draw[dashed] (-2.98,0) node[above]{\footnotesize $-\sqrt{5}$} -- (-2.98,-13)-- (0,-13)node[above left]{\footnotesize $-13$} -- (2.98,-13)-- (2.98,0) node[above]{\footnotesize$\sqrt{5}$};
		\node[below right] at (0,0) {\footnotesize$O$};\
		\node[below left] at (0,2) {\footnotesize $2$};
		\end{tikzpicture}
	\end{center}
	
	\noindent Xét hàm số $g(x)$ = $2f(x) + 2x^3 - 4x - 3m - 6\sqrt{5}$, $m \in \mathbb{R}$. Để $g(x) \le 0$ với mọi $x \in \left[-\sqrt{5}; \sqrt{5} \right]$ thì điều kiện của $m$ là
	\choice
	{\True $m \ge \dfrac{2}{3} f\left( \sqrt{5} \right)$}
	{$m \le \dfrac{2}{3} f\left( \sqrt{5} \right)$}
	{$m \le \dfrac{2}{3} f(0) - 2\sqrt{5}$}
	{$m \ge \dfrac{2}{3} f\left( - \sqrt{5} \right) - 4\sqrt{5}$}
	
	\loigiai{
		Ta có $g'(x) = 2f'(x) + 6x^2 - 4$.\\
		$g'(x) = 0 \Leftrightarrow f'(x) + 3x^2 - 2 = 0$.\\
		Ta có $g(x) \le 0, \forall x \in \left[-\sqrt{5}; \sqrt{5} \right]$ khi và chỉ khi $\max\limits_{x \in \left[-\sqrt{5}; \sqrt{5} \right]} g(x) \le 0$.\\
		\begin{center}
			\begin{tikzpicture}[>=stealth,scale=0.45]
			\draw[->] (-4,0) -- (4,0) node[above] {\footnotesize$x$};
			\draw[->] (0,-14) -- (0,7) node[right] {\footnotesize$y$};
			\draw[fill=black] (0,2) circle (3pt);
			\draw[fill=black] (-2.98,-13) circle (3pt);
			\draw[fill=black] (2.98,-13) circle (3pt);
			\draw[smooth,samples=100,domain=-3:3] plot(\x,{-0.31640625*(\x)^4+1.125*(\x)^2+2});
			\draw[smooth,samples=100,domain=-2.2:2.2] plot(\x,{1.6875*(\x)^2 - 2});
			\draw[dashed] (-2.98,0) node[above]{\footnotesize $-\sqrt{5}$} -- (-2.98,-13)-- (0,-13)node[above left]{\footnotesize $-13$} -- (2.98,-13)-- (2.98,0) node[above]{\footnotesize$\sqrt{5}$};
			\node[above right] at (0,0) {\footnotesize$O$};\
			\node[below left] at (0,2) {\footnotesize $2$};
			\end{tikzpicture}
		\end{center}
		
		\noindent Dựa vào đồ thị hàm số $y = f'(x)$ và $y = 3x^2 - 2$ ta thấy $f'(x) + 3x^2 - 2 \ge 0, \forall x \in \left[-\sqrt{5}; \sqrt{5} \right]$.\\
		Vậy $g'(x) \ge 0, \forall x \left[-\sqrt{5}; \sqrt{5} \right]$, hay hàm số $g(x)$ đồng biến trên $\left[-\sqrt{5}; \sqrt{5} \right]$. \\
		Suy ra $\max\limits_{x \in \left[-\sqrt{5}; \sqrt{5} \right]} g(x) = g\left(\sqrt{5} \right) = 2f\left(\sqrt{5} \right) - 3m$.\\
		Vậy yêu cầu đề bài tương đương với $ 2f\left(\sqrt{5} \right) - 3m \le 0$, hay $m \ge \dfrac{2}{3}f\left(\sqrt{5} \right)$.
	}
\end{ex}
