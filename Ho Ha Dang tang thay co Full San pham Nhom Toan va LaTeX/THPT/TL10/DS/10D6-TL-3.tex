\subsection{Công thức cộng}
\begin{dang}{Công thức cộng}
	Để giải các bài toán liên quan đến các công thức cộng, ta thường sử dụng các công thức sau:
	\begin{enumerate}
		\begin{multicols}{2}
			\item $\sin (a\pm b)=\sin a\cos b\pm \sin b\cos a$.
			\item $\cos (a\pm b)=\cos a\cos b\mp \sin a\sin b$.
			\item $\tan (a\pm b)=\dfrac{\tan a\pm \tan b}{1\mp \tan a\tan b}$.
		\end{multicols}
	\end{enumerate}
\end{dang}
\begin{vd}%[Hồ Sỹ Trường]%[0D6Y3]
	Tính giá trị của biểu thức $P=\cos10^\circ+\cos11^\circ\cos21^\circ+\cos69^\circ\cos79^\circ$.
	\loigiai{
		Ta có $P=\cos10^\circ+\cos11^\circ\cos21^\circ+\sin11^\circ\cos21^\circ=\cos10^\circ+\cos\left(11^\circ-21^\circ\right)=2\cos10^\circ$.
	}
\end{vd}
\begin{vd}%[Hồ Sỹ Trường]%[0D6B3]
	Rút gọn các biểu thức:
	\begin{enumerate}
		\item $A=\dfrac{\sqrt{2}\cos a-2\cos \left(\dfrac{\pi}{4}+a\right)}{-\sqrt{2}\sin a+2\sin \left(\dfrac{\pi}{4}+a\right)}$.
		\item $B=(\tan a-\tan b)\cot (a-b)-\tan a\tan b$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $A=\dfrac{\sqrt{2}\cos a -2\left(\cos \dfrac{\pi}{4}\cos a - \sin\dfrac{\pi}{4}\sin a\right)}{-\sqrt{2}\sin a +2\left(\sin\dfrac{\pi}{4}\cos a +\cos\dfrac{\pi}{4}\sin a \right)}=\dfrac{\sqrt{2}\sin a}{\sqrt{2}\cos a}=\tan a$.
			\item Ta có $B=\tan\left(a-b\right)\left(1+\tan a\tan b \right)\cot\left(a-b\right)-\tan a\tan b=1$.    
		\end{enumerate}
	}
\end{vd}

\begin{vd}%[Hồ Sỹ Trường]%[0D6B3]
	Chứng minh các đẳng thức sau:
	\begin{enumerate}
		\item $\cos a+\sin a=\sqrt{2}\cos \left(\dfrac{\pi}{4}-a\right)=\sqrt{2}\sin \left(\dfrac{\pi}{4}+a\right)$.
		\item $\cos a-\sin a=\sqrt{2}\cos \left(\dfrac{\pi}{4}+a\right)=\sqrt{2}\sin \left(\dfrac{\pi}{4}-a\right)$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $\cos a+\sin a = \sqrt{2}\left(\dfrac{1}{\sqrt{2}}\cos a+\dfrac{1}{\sqrt{2}}\sin a\right)=\sqrt{2}\left(\cos a\cos\dfrac{\pi}{4}+\sin a\sin\dfrac{\pi}{4}\right)=\sqrt{2}\cos\left(\dfrac{\pi}{4}-a\right)$.\\
			Mặt khác ta cũng có $\sqrt{2}\left(\dfrac{1}{\sqrt{2}}\cos a+\dfrac{1}{\sqrt{2}}\sin a\right)=\sqrt{2}\left(\sin\dfrac{\pi}{4}\cos a+\cos\dfrac{\pi}{4}\sin 2\right)=\sqrt{2}\sin\left(\dfrac{\pi}{4}+a\right)$.    
		\end{enumerate}
	}
\end{vd}

\begin{vd}%[Hồ Sỹ Trường]%[0D6B3]
	Cho tam giác $ABC$, chứng minh rằng $\tan A + \tan B+\tan C = \tan A\tan B\tan C$.
	\loigiai{
		Ta có 
		\begin{eqnarray*}
			\tan A+\tan B+\tan C&=&\dfrac{\sin A}{\cos A}+\dfrac{\sin B}{\cos B}+\dfrac{\sin C}{\cos C}\\
			&=&\dfrac{\sin A\cos B+\sin B\cos A}{\cos A\cos B}+\dfrac{\sin C}{\cos C}\\
			&=&\dfrac{\sin\left(A+B\right)}{\cos A\cos B}+\dfrac{\sin C}{\cos C}\\
			&=&\dfrac{\sin C}{\cos A\cos B}+\dfrac{\sin C}{\cos C}\\
			&=&\sin C\left(\dfrac{1}{\cos A\cos B}+\dfrac{1}{\cos C}\right)\\
			&=&\sin C\cdot \dfrac{\cos A\cos B+\cos C}{\cos A\cos B\cos C}\\
			&=&\tan C\cdot\dfrac{\cos A\cos B-\cos\left(A+B\right)}{\cos A\cos B}\\
			&=&\tan C\cdot \dfrac{\cos A\cos B-\cos A\cos B+\sin A\sin B}{\cos A\cos B\cos C}\\
			&=&\tan A\tan B\tan C.
		\end{eqnarray*}
	}
\end{vd}

\begin{bt}%[Hồ Sỹ Trường]%[0D6B3]
	Tính
	\begin{enumerate}
		\item $\cos\left(\alpha+\dfrac{\pi}{3}\right)$, biết $\sin \alpha=\dfrac{1}{\sqrt{3}}$ và $0<\alpha<\dfrac{\pi}{2}$.
		\item $\tan\left(\alpha -\dfrac{\pi}{4} \right)$, biết $\cos \alpha=-\dfrac{1}{3}$ và $\dfrac{\pi}{2}<\alpha<\pi$.
		\item $\cos\left(a+b\right),\sin\left(a-b\right)$, biết $\sin a=\dfrac{4}{5},0^\circ<a<90^\circ$ và $\sin b =\dfrac{2}{3},90^\circ<b<180^\circ$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Do $0<\alpha<\dfrac{\pi}{2}$ nên $\cos\alpha>0$.
			Do đó $\cos\alpha=\sqrt{1-\sin^2\alpha}=\sqrt{\dfrac{2}{3}}=\dfrac{\sqrt{6}}{3}$.\\
			Ta có $\cos\left(\alpha+\dfrac{\pi}{3}\right)=\cos\alpha\cos\dfrac{\pi}{3}-\sin\alpha\sin\dfrac{\pi}{3}=\dfrac{\sqrt{6}}{3}\cdot\dfrac{1}{2}-\dfrac{1}{\sqrt{3}}\cdot\dfrac{\sqrt{3}}{2}=\dfrac{-3+\sqrt{6}}{6}$.
			\item Do $\dfrac{\pi}{2}<\alpha<\pi$ nên $\sin\alpha>0$. Do đó $\sin\alpha=\sqrt{1-\cos^2\alpha}=\dfrac{2\sqrt{2}}{3}$. Suy ra $\tan\alpha=-2\sqrt{2}$.\\
			Ta có $\tan\left(\alpha -\dfrac{\pi}{4} \right)=\dfrac{\tan\alpha-\tan\dfrac{\pi}{4}}{1+\tan\alpha\tan\dfrac{\pi}{4}}=\dfrac{\tan\alpha-1}{\tan\alpha+1}=\dfrac{-2\sqrt{2}-1}{-2\sqrt{2}+1}=\dfrac{9+4\sqrt{2}}{7}$.
			\item Có $0^\circ<a<90^\circ$ nên $\cos a=\dfrac{3}{5}$, $90^\circ<b<180^\circ$ nên $\cos b=-\dfrac{\sqrt{5}}{3}$.\\
			$\cos\left(a+b\right)=\cos a\cos b -\sin a\sin b = \dfrac{3}{5}\cdot\dfrac{-\sqrt{5}}{3}-\dfrac{4}{5}\cdot\dfrac{2}{3}=-\dfrac{8+3\sqrt{5}}{15}$.\\
			$\sin\left(a-b\right)=\sin a\cos b-\sin b\cos a=\dfrac{4}{5}\cdot\dfrac{-\sqrt{5}}{3}-\dfrac{2}{3}\cdot\dfrac{3}{5}=-\dfrac{6+4\sqrt{5}}{15}$.
		\end{enumerate}
	}
\end{bt}

			\item $\dfrac{\cos \left(a-b\right)}{\cos\left(a+b\right)}=\dfrac{\cos a\cos b+\sin a\sin b}{\cos a\cos b-\sin a\sin b}=\dfrac{\dfrac{\cos a\cos b+\sin a\sin b}{\sin a\sin b}}{\dfrac{\cos a\cos b-\sin a\sin b}{\sin a\sin b}}=\dfrac{\cot a\cot b+1}{\cot a\cot b-1}$.

\begin{bt}%[Hồ Sỹ Trường]%[0D6B3]
	Rút gọn các biểu thức
	\begin{enumerate}
		\item $A=\sin\left(a+b\right)+\sin\left(\dfrac{\pi}{2}-a\right)\sin(-b)$.
		\item $B=\cos\left(\dfrac{\pi}{4}+a\right)\cos\left(\dfrac{\pi}{4}-a\right)+\dfrac{1}{2}\sin^2a$.
		\item $C=\cos\left(\dfrac{\pi}{2}-a\right)\sin\left(\dfrac{\pi}{2}-b\right)-\sin\left(a-b\right)$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $A=\sin a\cos b +\sin b\cos a -\cos a\sin b=\sin a\cos b$.
			\item Ta có
			\begin{eqnarray*}
				B&=&\left(\cos\dfrac{\pi}{4}\cos a-\sin\dfrac{\pi}{4}\sin a\right)\left(\cos\dfrac{\pi}{4}\cos a+\sin\dfrac{\pi}{4}\sin a\right)+\dfrac{1}{2}\sin^2a\\
				&=&\left(\dfrac{\sqrt{2}}{2}\cos a-\dfrac{\sqrt{2}}{2}\sin a\right)\left(\dfrac{\sqrt{2}}{2}\cos a+\dfrac{\sqrt{2}}{2}\sin a\right)+\dfrac{1}{2}\sin^2a\\
				&=&\dfrac{1}{2}\left(\cos a-\sin a\right)\left(\cos a+\sin a\right)+\dfrac{1}{2}\sin^2a\\
				&=&\dfrac{1}{2}\left(\cos^2a-\sin^2a\right)+\dfrac{1}{2}\sin^2a=\dfrac{1}{2}\cos^2a.
			\end{eqnarray*}
			\item Ta có
			\begin{eqnarray*}
				C&=&\left(\cos\dfrac{\pi}{2}\cos a+\sin\dfrac{\pi}{2}\sin a\right)\left(\sin\dfrac{\pi}{2}\cos b-\sin b\cos\dfrac{\pi}{2}\right)-\left(\sin a\cos b -\sin b\cos a\right)\\
				&=&\sin a\cos b-\sin a\cos b+\sin b\cos a=\sin b\cos a.
			\end{eqnarray*}
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Hồ Sỹ Trường]%[0D6B3]
	Chứng minh các đẳng thức sau:
	\begin{enumerate}
		\item $\sin\left(a+b\right)\sin\left(a-b\right)=\sin^2a-\sin^2b=\cos^2b-\cos^2a$.
		\item $\cos\left(a+b\right)\cos\left(a-b\right)=\cos^2a-\sin^2b=\cos^2b-\sin^2a$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item $\sin\left(a+b\right)\sin\left(a-b\right)=\left(\sin a\cos b+\sin b\cos a\right)\left(\sin a\cos b-\sin b\cos a\right)$\\
			$=\sin^2a\cos^2b-\sin^2b\cos^2a=\sin^2a\left(1-\sin^2b\right)-\sin^2b\left(1-\sin^2a\right)=\sin^2a-\sin^2b$.\\
			Tương tự: $\sin^2a\cos^2b-\sin^2b\cos^2a=\left(1-\cos^2a\right)\cos^2b-\left(1-\cos^2b\right)\cos^2a=\cos^2b-\cos^2a$.
			\item $\cos\left(a+b\right)\cos\left(a-b\right)=\left(\cos a\cos b - \sin a\sin b\right)\left(\cos a\cos b+\sin a\sin b\right)$\\
			$=\cos^2a\cos^2b-\sin^2a\sin^2b=\cos^2a\left(1-\sin^2b\right)-\left(1-\cos^2a\right)\sin^2b=\cos^2a-\sin^2b$.\\
			Tương tự: $\cos^2a\cos^2b-\sin^2a\sin^2b=\left(1-\sin^2a\right)\cos^2b-\sin^2a\left(1-\cos^2b\right)=\cos^2b-\sin^2a$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Hồ Sỹ Trường]%[0D6B3]

	}

\begin{bt}%[Hồ Sỹ Trường]%[0D6B3]
	Cho tam giác $ABC$, chứng minh rằng $\tan\dfrac{A}{2}\tan\dfrac{B}{2}+\tan\dfrac{B}{2}\tan\dfrac{C}{2}+\tan\dfrac{C}{2}\tan\dfrac{A}{2}=1$.
	\loigiai{
		Ta có
		\begin{eqnarray*}
			\tan\dfrac{A}{2}\tan\dfrac{B}{2}+\tan\dfrac{B}{2}\tan\dfrac{C}{2}+\tan\dfrac{C}{2}\tan\dfrac{A}{2}&=&\tan\dfrac{A}{2}\tan\dfrac{B}{2}+\tan\dfrac{C}{2}\left(\tan\dfrac{A}{2}+\tan\dfrac{B}{2}\right)\\
			&=&\tan\dfrac{A}{2}\tan\dfrac{B}{2}+\tan\dfrac{C}{2}\tan\left(\dfrac{A}{2}+\dfrac{B}{2}\right)\left(1-\tan\dfrac{A}{2}\tan\dfrac{B}{2}\right)\\
			&=&\tan\dfrac{A}{2}\tan\dfrac{B}{2}+\tan\dfrac{C}{2}\cot\dfrac{C}{2}\left(1-\tan\dfrac{A}{2}\tan\dfrac{B}{2}\right)\\
			&=&\tan\dfrac{A}{2}\tan\dfrac{B}{2}+1-\tan\dfrac{A}{2}\tan\dfrac{B}{2}=1.
		\end{eqnarray*}
	}    
\end{bt}

\subsection{Công thức nhân đôi}
\begin{dl}[Công thức góc nhân đôi]
	Với mọi giá trị của góc lượng giác $\alpha$ cho trước, ta có
	\begin{itemize}
		\item $\sin2\alpha=2\sin\alpha\cos\alpha$.
		\item $\cos2\alpha=\cos^2\alpha-\sin^2\alpha=2\cos^2\alpha-1=1-2\sin^2\alpha$.
		\item $\tan2\alpha=\dfrac{2\tan\alpha}{1-\tan^2\alpha}$, $\heva{&\alpha\ne\dfrac{\pi}{4}+k\dfrac{\pi}{2}\\&\alpha\ne\dfrac{\pi}{2}+k\pi}$, $k\in\mathbb{Z}$.
	\end{itemize}
\end{dl}
\begin{hq}[Công thức hạ bậc]
	Với mọi giá trị của góc lượng giác $\alpha$ cho trước, ta có
	\begin{itemize}
		\item $\sin^2\alpha=\dfrac{1-\cos2\alpha}{2}$.
		\item $\cos^2\alpha=\dfrac{1+\cos2\alpha}{2}$.
		\item $\tan^2\alpha=\dfrac{1-\cos2\alpha}{1+\cos2\alpha}$, $\alpha\ne\dfrac{\pi}{2}+k\pi$, $k\in\mathbb{Z}$.
	\end{itemize}
\end{hq}
\begin{hq}[Công thức nhân ba]
	Với mọi giá trị của góc lượng giác $\alpha$ cho trước, ta có
	\begin{multicols}{2}
		\begin{itemize}
			\item $\sin3\alpha=3\sin\alpha-4\sin^3\alpha$.
			\item $\cos3\alpha=4\cos^3\alpha-3\cos\alpha$.
		\end{itemize}
	\end{multicols}
	\begin{center}
		\textit{(Chứng minh lại khi sử dụng trong bài tập tự luận)}
	\end{center}
	\loigiai{
		\begin{enumerate}
			\item $\sin3\alpha=\sin(\alpha+2\alpha)=\sin\alpha\cos2\alpha+\sin2\alpha\cos\alpha=\sin\alpha(1-2\sin^2\alpha)+2\sin\alpha\cos^2\alpha$\\
			$=\sin\alpha-2\sin^3\alpha+2\sin\alpha(1-\sin^2\alpha)=3\sin\alpha-4\sin^3\alpha$.
			\item $\cos3\alpha=\cos(\alpha+2\alpha)=\cos\alpha\cos2\alpha-\sin\alpha\sin2\alpha=\cos\alpha(2\cos^2\alpha-1)-2\sin^2\alpha\cos\alpha$\\
			$=2\cos^3\alpha-\cos\alpha-2(1-\cos^2\alpha)\cos\alpha=4\cos^3\alpha-3\cos\alpha$.
	\end{enumerate}}
\end{hq}

\begin{vd}%[Phan Hoàng Anh]%[0D6Y3]

		$\Leftrightarrow\cos\alpha=\dfrac{\sqrt{2+\sqrt{2}}}{2}$ vì $0^\circ<\alpha<90^\circ$.\\
		$\sin2\alpha=2\sin\alpha\cos\alpha\Leftrightarrow\sin\alpha=\dfrac{\sin2\alpha}{2\cos\alpha}=\dfrac{\sin45^\circ}{2\cos\alpha}=\dfrac{\sqrt{2-\sqrt{2}}}{2}$.\\
		$\tan2\alpha=\dfrac{\sin2\alpha}{\cos2\alpha}=\dfrac{\sqrt{2+\sqrt{2
		}}}{\sqrt{2-\sqrt{2}}}=1+\sqrt{2}$, $\cot2\alpha=\dfrac{1}{\tan2\alpha}=\sqrt{2}-1$.
	}
\end{vd}
\begin{vd}%[Phan Hoàng Anh]%[0D6Y3]

	\loigiai{Ta có $\sin^2\alpha+\cos^2\alpha=1\Rightarrow\cos^2\alpha=1-\sin^2\alpha=\dfrac{16}{25}$\\
		Do $\alpha\in\left(\dfrac{\pi}{2};\pi\right)$ nên $\cos\alpha<0\Rightarrow\cos\alpha=-\dfrac{4}{5}$.\\
		$\sin2\alpha=2\sin\alpha\cos\alpha=2\cdot\dfrac{3}{5}\cdot\dfrac{4}{5}=\dfrac{24}{25}$.\\
		$\tan\alpha=\dfrac{\sin\alpha}{\cos\alpha}=\dfrac{3}{4}\Rightarrow\tan2\alpha=\dfrac{2\tan\alpha}{1-\tan^2\alpha}=\dfrac{24}{7}$.
	}
\end{vd}

\begin{vd}%[Phan Hoàng Anh]%[0D6B3]

	\loigiai{

			Vậy $A=\dfrac{1}{8}$.
			\item $B=\cos^3x\sin x-\sin^3x\cos x=\cos x\sin x(\cos^2x-\sin^2x)=\dfrac{1}{2}\sin2x\cos2x=\dfrac{1}{4}\sin4x$.

\begin{vd}%[Phan Hoàng Anh]%[0D6B3]

		\item $\dfrac{\sin^4\alpha-\cos^4\alpha+\cos^2\alpha}{2(1-\cos\alpha)}=\cos^2\dfrac{\alpha}{2}$.

	\loigiai{

\subsection{Công thức biến đổi}
\begin{dang}{Biến đổi một biểu thức thành một tổng hoặc thành một tích}
	Đây là dạng toán cơ bản chủ yếu để tập cho học sinh áp dụng được đối với các công thức biến đổi (tổng thành tích, tích thành tổng) đã học. Dưới đây là các công thức biến đổi đó.
	\begin{enumerate}[1.]
		\item Công thức biến đổi tích thành tổng
		\begin{itemize}
			\begin{multicols}{2}
				\item $\cos a\cos b=\dfrac{1}{2}\left[\cos(a-b)+\cos(a+b) \right]$
				\item $\sin a\sin b=\dfrac{1}{2}\left[\cos(a-b)-\cos(a+b) \right]$
				\item $\sin a\cos b=\dfrac{1}{2}\left[\sin(a+b)+\sin(a-b) \right]$
				\item $\cos a\sin b=\dfrac{1}{2}\left[\sin(a+b)-\sin(a-b) \right]$
			\end{multicols}
		\end{itemize}
		\item Công thức biến đổi tổng thành tích
		\begin{itemize}
			\begin{multicols}{2}
				\item $\sin a+\sin b=2\sin \dfrac{a+b}{2}\cos\dfrac{a-b}{2}$
				\item $\sin a-\sin b=2\cos \dfrac{a+b}{2}\sin\dfrac{a-b}{2}$
				\item $\cos a+\cos b=2\cos \dfrac{a+b}{2}\cos\dfrac{a-b}{2}$
				\item $\cos a-\cos b=-2\sin \dfrac{a+b}{2}\sin\dfrac{a-b}{2}$
				\item $\tan a+\tan b=\dfrac{\sin(a+b)}{\cos a\cos b}$
				\item $\tan a-\tan b=\dfrac{\sin(a-b)}{\cos a\cos b}$
			\end{multicols}
		\end{itemize}
	\end{enumerate}
\end{dang}

			$=\dfrac{1}{2}\cdot (-2)\sin (b+a)\sin (b-a)=\sin (a+b)\sin (a-b)$.
			\item $1+\sin a+\cos a=1+\sin \left(2\cdot \dfrac{a}{2}\right)+\cos \left(2\cdot \dfrac{a}{2}\right)=1+2\sin \dfrac{a}{2}\cos \dfrac{a}{2}+2\cos^2\dfrac{a}{2}-1$ \\
			$=2\sin \dfrac{a}{2}\cos \dfrac{a}{2}+2\cos^2\dfrac{a}{2}=2\cos \dfrac{a}{2} \left(\sin \dfrac{a}{2}+\cos \dfrac{a}{2}\right)$.
			\item $1-\cos a+\sin a=1-\cos \left(2\cdot \dfrac{a}{2}\right)+\sin \left(2\cdot \dfrac{a}{2}\right)$ \\
			$=2\sin^2\dfrac{a}{2}+2\sin \dfrac{a}{2}\cos \dfrac{a}{2}=2\sin \dfrac{a}{2}\left(\sin \dfrac{a}{2}+\cos \dfrac{a}{2}\right)$.
			\item $1-2\cos x+\cos 2x=1-2\cos x+2\cos^2x-1=2\cos x\left(\cos x-1\right)$.
			\item $1+\sin x-\cos 2x=1+\sin x-\left(1-2\sin^2x\right)=\sin x\left(1+2\sin x\right)$.
			\item $\sin^2x-\sin^22x+\sin^23x=\dfrac{1-\cos 2x}{2}-\sin^22x+\dfrac{1-\cos 6x}{2}$ \\
			$=1-\sin^22x-\dfrac{1}{2}\left(\cos 6x+\cos 2x\right)=\cos^22x-\cos 4x\cos 2x$ \\
			$=\cos 2x\left(\cos 2x-\cos 4x\right)=2\cos 2x\sin x\sin 3x$.
			\item $\cos^2x+\cos^22x+\cos^23x-1=\dfrac{1+\cos 2x}{2}+\cos^22x+\dfrac{1+\cos 6x}{2}-1$ \\
			$=\dfrac{1}{2}\left(\cos 6x+\cos 2x\right)+\cos^22x=\cos 4x\cos 2x+\cos^22x$ \\
			$=\cos 2x\left(\cos 4x+\cos 2x\right)=2\cos 2x\cos 3x\cos x$.
		\end{enumerate}
	}
\end{bt}

		\item  Biến đổi, dẫn đến $\sin A = 1$ hoặc $\cos A = 0$ sẽ có $A=90^0$.
		\item  Nếu $a^2+b^2=c^2$ thì $C=90^0$.
		\item  Nếu $\sin (A - B) = 0$ hoặc $\cos (A - B) = 1$ thì $A=B$, suy ra tam giác cân.
		\item Tam giác cân mà có một góc bằng $60^0$ là tam giác đều.
	\end{itemize}
	Một số lưu ý khi giả thiết cho $A,B,C$ là ba góc của một tam giác
	\begin{itemize}
		\item $A+B+C=180^\circ \Rightarrow (A+B)$ và $C$ bù nhau, tương tự với $(B+C)$ và $A$,...
		\item $\dfrac{A}{2}+\dfrac{B}{2}+\dfrac{C}{2}=90^\circ \Rightarrow \left(\dfrac{A}{2}+\dfrac{B}{2} \right)$ và $\dfrac{C}{2}$ phụ nhau, tương tự với $\left(\dfrac{B}{2}+\dfrac{C}{2} \right)$ và $\dfrac{A}{2}$,...
		\item Các góc $A,B,C$ đều có số đo trong khoảng $\left(0^\circ;180^\circ \right)$
		\item Các góc $\dfrac{A}{2},\dfrac{B}{2},\dfrac{C}{2}$ đều là các góc nhọn nên có các giá trị lượng giác đều dương.
	\end{itemize}
\end{dang}

\begin{vd}%[Nguyễn Tài Chung]%[0D6B3]
	Chứng minh rằng $\Delta ABC$ vuông khi 
	$\sin A\sin C = \cos A\cos C$.
	\loigiai{Ta có 
		{\allowdisplaybreaks
			\begin{align*}
			&\sin A\sin C = \cos A\cos C \Leftrightarrow \cos A\cos C - \sin A\sin C = 0\\
			\Leftrightarrow& \cos (A + C) = 0 \Leftrightarrow  - \cos B = 0 \Leftrightarrow \cos B = 0 \Leftrightarrow B = {90^\circ}
			\end{align*}}Vậy tam giác $ABC$ vuông tại $B$.
	}
\end{vd}

\begin{vd}%[Nguyễn Tài Chung]%[0D6K3]

		{\allowdisplaybreaks

			&\cos (A - B) - \cos (A + B) = 1 + \cos C\\
			\Leftrightarrow& \cos (A - B) + \cos C = 1 + \cos C\\
			\Leftrightarrow& \cos (A - B) = 1 \Leftrightarrow A - B = 0 \Leftrightarrow A = B.

		{\allowdisplaybreaks
			\begin{align*}

			& = 2\sin C\cos (A - B) + 2\sin C\cos C \\

			&= 4\frac{a}{{2R}}.\frac{b}{{2R}}.\frac{c}{{2R}} = \frac{{abc}}{{4R}}.\frac{2}{{R^2 }} = \frac{{2S}}{{R^2 }}.
			\end{align*}}Ta có điều phải chứng minh.
	}
\end{vd}
\begin{vd}%[Nguyễn Tài Chung]%[0D6K3]

\begin{vd}%[Dương Phước Sang]%[0D6G3] 
	Tam giác $ABC$ là tam giác gì nếu $\sin A=\dfrac{\sin B+\sin C}{\cos B+\cos C}$ ?
	\loigiai{\\
		Ta có $\sin A=\dfrac{\sin B+\sin C}{\cos B+\cos C} \Leftrightarrow \sin A=\dfrac{2\sin \dfrac{B+C}{2} \cos \dfrac{B-C}{2}}{2\cos \dfrac{B+C}{2} \cos \dfrac{B-C}{2}}
		\Leftrightarrow \sin A=\tan \dfrac{B+C}{2}$\\
		$ \Leftrightarrow \sin \left(2\cdot \dfrac{A}{2}\right)=\tan \left(\dfrac{\pi}{2}-\dfrac{A}{2}\right) \Leftrightarrow 2\sin \dfrac{A}{2} \cos \dfrac{A}{2}=\cot \dfrac{A}{2} \Leftrightarrow 2\sin^2\dfrac{A}{2} \cos \dfrac{A}{2}=\cos \dfrac{A}{2}$\\

	}
\end{vd}

\begin{bt}%[Dương Phước Sang]%[0D6K3] 
	Cho $A,B,C$ là $3$ đỉnh của một tam giác. Chứng minh rằng
	\begin{enumerate}
		\item $\sin A+\sin B+\sin C=4\cos \dfrac{A}{2}\cos \dfrac{B}{2}\cos \dfrac{C}{2}$
		\item $\cos A+\cos B+\cos C=1+4\sin \dfrac{A}{2}\sin \dfrac{B}{2}\sin \dfrac{C}{2}$
		\item $\sin 2A+\sin 2B+\sin 2C=4\sin A\sin B\sin C$
		\item $\cos 2A+\cos 2B+\cos 2C+1=-4\cos A\cos B\cos C$
		\item $\sin^2A+\sin^2B+\sin^2C=2+2\cos A\cos B\cos C$
		\item $\cos^2A+\cos^2B+\cos^2C=1-2\cos A\cos B\cos C$
		\item $\tan A+\tan B+\tan C=\tan A\tan B\tan C$
		\item $\tan \dfrac{A}{2}\tan \dfrac{B}{2}+\tan \dfrac{B}{2}\tan \dfrac{C}{2}+\tan \dfrac{C}{2}\tan \dfrac{A}{2}=1$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Chứng minh rằng $\sin A+\sin B+\sin C=4\cos \dfrac{A}{2}\cos \dfrac{B}{2}\cos \dfrac{C}{2}$.\\
			Ta có $\sin A+\sin B+\sin C=2\sin \dfrac{A+B}{2} \cos \dfrac{A-B}{2}+\sin \left(2\cdot \dfrac{C}{2}\right)$ \\
			$=2\cos \dfrac{C}{2} \cos \dfrac{A-B}{2}+2\sin \dfrac{C}{2} \cos \dfrac{C}{2} ~\left(\text{do } \dfrac{A+B}{2}+\dfrac{C}{2}=90^\circ\right)$ \\
			$=2\cos \dfrac{C}{2} \left(\cos \dfrac{A-B}{2}+\cos \dfrac{A+B}{2}\right) ~\left(\text{do } \sin \dfrac{C}{2}=\cos \dfrac{A+B}{2}\right)$ \\
			$=2\cos \dfrac{C}{2}\cdot 2\cos \dfrac{A}{2} \cos \left(-\dfrac{B}{2}\right)=4\cos \dfrac{A}{2} \cos \dfrac{B}{2} \cos \dfrac{C}{2}$.
			\item Chứng minh rằng $\cos A+\cos B+\cos C=1+4\sin \dfrac{A}{2}\sin \dfrac{B}{2}\sin \dfrac{C}{2}$\\
			Ta có $\cos A+\cos B+\cos C=2\cos \dfrac{A+B}{2} \cos \dfrac{A-B}{2}+\cos \left(2\cdot \dfrac{C}{2}\right)$ \\
			$=2\sin \dfrac{C}{2} \cos \dfrac{A-B}{2}+1-2\sin^2\dfrac{C}{2} ~\left(\text{do } \dfrac{A+B}{2}+\dfrac{C}{2}=90^\circ\right)$ \\
			$=1+2\sin \dfrac{C}{2}\left(\cos \dfrac{A-B}{2}-\cos \dfrac{A+B}{2}\right) ~\left(\text{do } \sin \dfrac{C}{2}=\cos \dfrac{A+B}{2}\right)$ \\
			$=1+2\sin \dfrac{C}{2} (-2)\sin \dfrac{A}{2} \sin \left(-\dfrac{B}{2}\right)=1+4\sin \dfrac{A}{2}\sin \dfrac{B}{2}\sin \dfrac{C}{2}$.
			\item Chứng minh rằng $\sin 2A+\sin 2B+\sin 2C=4\sin A\sin B\sin C$\\
			Ta có $\sin 2A+\sin 2B+\sin 2C=2\sin (A+B)\cos (A-B)+2\sin C\cos C$ \\
			$=2\sin C\cos (A-B)-2\sin C\cos (A+B) ~\left(\text{do } A+B+C=180^\circ\right)$ \\
			$=2\sin C\left[\cos (A-B)-\cos (A+B)\right]$ \\
			$=2\sin C\cdot(-2)\sin A\sin (-B)=4\sin A\sin B\sin C$.
			\item Chứng minh rằng $\cos 2A+\cos 2B+\cos 2C+1=-4\cos A\cos B\cos C$.\\
			Ta có $\cos 2A+\cos 2B+\cos 2C+1=2\cos (A+B)\cos (A-B)+2\cos^2C-1+1$ \\
			$=-2\cos C\cos (A-B)-2\cos C\cos (A+B) ~\left(\text{do } A+B+C=180^\circ\right)$ \\
			$=-2\cos C\left[\cos (A-B)+\cos (A+B)\right]$ \\
			$=-2\cos C\cdot 2\cos A\cos (-B)=-4\cos A\cos B\cos C$.
			\item Chứng minh rằng $\sin^2A+\sin^2B+\sin^2C=2+2\cos A\cos B\cos C$\\
			Ta có $\sin^2A+\sin^2B+\sin^2C=\dfrac{1-\cos 2A}{2}+\dfrac{1-\cos 2B}{2}+\dfrac{1-\cos 2C}{2}$ \\
			$=\dfrac{3}{2}-\dfrac{1}{2}\left(\cos 2A+\cos 2B+\cos 2C\right)=\dfrac{3}{2}-\dfrac{1}{2}\left[2\cos (A+B)\cos (A-B)+2\cos^2C-1\right]$ \\
			$=\dfrac{3}{2}-\dfrac{1}{2}\left[-2\cos C\cos (A-B)-2\cos C\cos (A+B)-1\right]$ \\
			$=2+\cos C\left[\cos (A-B)+\cos (A+B)\right]$ \\
			$=2+\cos C\cdot2\cos A\cos (-B)=2+2\cos A\cos B\cos C$
			\item Chứng minh rằng $\cos^2A+\cos^2B+\cos^2C=1-2\cos A\cos B\cos C$\\
			Ta có $\cos^2A+\cos^2B+\cos^2C=\dfrac{1+\cos 2A}{2}+\dfrac{1+\cos 2B}{2}+\cos^2C$ \\
			$=1+\dfrac{1}{2}\left(\cos 2A+\cos 2B\right)+\cos^2C=1+\cos (A+B)\cos (A-B)+\cos^2C$ \\
			$=1-\cos C\cos (A-B)-\cos C\cos (A+B) ~\left(\text{do } A+B+C=180^\circ\right)$ \\
			$=1-\cos C\left[\cos (A-B)+\cos (A+B)\right]$ \\
			$=1-\cos C\cdot 2\cos A\cos (-B)=1-2\cos A\cos B\cos C$
			\item Chứng minh rằng $\tan A+\tan B+\tan C=\tan A\tan B\tan C$\\
			Ta có Ta có $A+B+C=180^\circ \Rightarrow A+B=180^\circ-C \Rightarrow \tan (A+B)=\tan \left(180^\circ-C\right)$ \\
			$ \Rightarrow \dfrac{\tan A+\tan B}{1-\tan A\tan B}=-\tan C \Rightarrow \tan A+\tan B=-\tan C\left(1-\tan A\tan B\right)$ \\
			$ \Rightarrow \tan A+\tan B=-\tan C+\tan A\tan B\tan C$ \\
			$ \Rightarrow \tan A+\tan B+\tan C=\tan A\tan B\tan C$.
			\item Chứng minh rằng $\tan \dfrac{A}{2}\tan \dfrac{B}{2}+\tan \dfrac{B}{2}\tan \dfrac{C}{2}+\tan \dfrac{C}{2}\tan \dfrac{A}{2}=1$\\
			Ta có $A+B+C=180^\circ \Rightarrow \dfrac{A}{2}+\dfrac{B}{2}=90^\circ-\dfrac{C}{2} \Rightarrow \tan \left(\dfrac{A}{2}+\dfrac{B}{2}\right)=\tan \left(90^\circ-\dfrac{C}{2}\right)$ \\
			$ \Rightarrow \dfrac{\tan \dfrac{A}{2}+\tan \dfrac{B}{2}}{1-\tan \dfrac{A}{2} \tan \dfrac{B}{2}}=\cot \dfrac{C}{2}=\dfrac{1}{\tan \dfrac{C}{2}} \Rightarrow \tan \dfrac{C}{2}\left(\tan \dfrac{A}{2}+\tan \dfrac{B}{2}\right)=1-\tan \dfrac{A}{2} \tan \dfrac{B}{2}$ \\
			$ \Rightarrow \tan \dfrac{C}{2} \tan \dfrac{A}{2}+\tan \dfrac{C}{2} \tan \dfrac{B}{2}=1-\tan \dfrac{A}{2} \tan \dfrac{B}{2}$ \\
			$ \Rightarrow \tan \dfrac{A}{2} \tan \dfrac{B}{2}+\tan \dfrac{B}{2} \tan \dfrac{C}{2}+\tan \dfrac{C}{2} \tan \dfrac{A}{2}=1$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}

			&= \frac{{4\sqrt 3 }}{3} + \frac{2}{{\cos 10^\circ }} = \frac{{4\sqrt 3 \cos 10^\circ  + 6}}{{3\cos 10^\circ }} = \frac{{4\sqrt 3 \left( {\cos 10^\circ  + \dfrac{6}{{4\sqrt 3 }}} \right)}}{{3\cos 10^\circ }} \\ 
			&= \frac{{4\sqrt 3 \left( {\cos 10^\circ  + \dfrac{{\sqrt 3 }}{2}} \right)}}{{3\cos 10^\circ }} = \frac{{4\sqrt 3 \left( {\cos 10^\circ  + \cos 30^\circ } \right)}}{{3\cos 10^\circ }} \\ 
			&= \frac{{4\sqrt 3 .2\cos 20^\circ \cos 10^\circ }}{{3\cos 10^\circ }} = \frac{{8\sqrt 3 }}{3}\cos 20^\circ.
