\begin{enumerate}
\item Cho biếu thức $A=\dfrac{\sqrt{x}+4}{\sqrt{x}+2}$. Tính giá trị của biểu thức $A$ khi $x=36$.
\item Rút gọn biểu thức $B=\left(\dfrac{\sqrt{x}}{\sqrt{x}+4}+\dfrac{4}{\sqrt{x}-4}\right):\dfrac{x+16}{\sqrt{x}+2}$ (với $x\ge 0$, $x\not=16$).
\item Với các biểu thức $A$ và $B$ nói trên, hãy tìm các giá trị nguyên của $x$ để biểu thức $B(A-1)$ là số nguyên.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $A=\dfrac{\sqrt{36}+4}{\sqrt{36}+2}= \dfrac{5}{4}.$
\item $B=\dfrac{\sqrt{x}(\sqrt{x}-4)+4(\sqrt{x}+4)}{(\sqrt{x}+4)(\sqrt{x}-4)} \cdot \dfrac{\sqrt{x}+2}{x+16} =\dfrac{\sqrt{x}+2}{x-16}.$
\item $B(A-1)=\dfrac{2}{x-16}\in\mathbb{Z}\Leftrightarrow x-16 \in \{-1;1;-2;2\} \Leftrightarrow x \in\{14,15,17,18\}$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[SGD-HaNoi-2012]%[9D4K8]
\textit{Giải bài toán sau bằng cách lập phương trình:}\\
Hai người cùng làm chung một công việc trong $\dfrac{12}{5}$ giờ thì xong. Nếu mỗi người làm một mình thì thời gian để người thứ nhất hoàn thành công việc ít hơn người thứ hai là $2$ giờ. Hỏi nếu làm một mình thì mỗi người phải làm trong bao nhiêu giờ để xong công việc?
\loigiai{
Gọi $x,y$ $(y>x>0)$ theo thứ tự là thời gian để người thứ nhất, người thứ hai hoàn thành công việc khi làm một mình. \\
Khi đó $\dfrac{1}{x}$ là công việc người thứ nhất hoàn thành trong $1$ giờ,  $\dfrac{1}{y}$ là công việc người thứ hai hoàn thành trong $1$ giờ. Theo giả thiết ta có hệ phương trình
$$ \heva{&y-x=2\\&\dfrac{12}{5} \left (\dfrac{1}{x}+\dfrac{1}{y}\right )=1} \Leftrightarrow \heva{&y-x=2\\&\dfrac{12}{5} \left (\dfrac{1}{x}+\dfrac{1}{x+2}\right )=1} \Leftrightarrow \heva{&x=4\\&y=6}$$ 
Vậy người thứ nhất làm một mình cần $4$ giờ, người thứ hai làm một mình cần $6$ giờ. 
}
\end{ex}

\begin{ex}%[SGD-HaNoi-2012]%[9D3B4]%[9D4B6]
\begin{enumerate}
\item Giải hệ phương trình $\heva{&
\dfrac{2}{x}+\dfrac{1}{y}=2\\&
\dfrac{6}{x}-\dfrac{2}{y}=1.
}$
\item Cho phương trình: $x^2-(4m-1)x+3m^2-2m=0$ (ẩn x). Tìm $m$ để phương trình có hai nghiệm phân biệt $x_1, x_2$ thoả mãn $x_1^2+x_2^2=7$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $\heva{&
\dfrac{2}{x}+\dfrac{1}{y}=2\\&
\dfrac{6}{x}-\dfrac{2}{y}=1 } \Leftrightarrow \heva{&\dfrac{10}{x}=5\\&\dfrac{2}{y}=\dfrac{6}{x}-1}\Leftrightarrow \heva{&x=2\\&y=1}$
\item $\Delta =4m^2+1>0$ với mọi $m$ cho nên phương trình luôn có hai nghiệm phân biệt. Theo định lý Vi-ét ta có $x_1+x_2=4m-1, x_1x_2=3m^2-2m.$ \\
$x_1^2+x_2^2=7 \Leftrightarrow (x_1+x_2)^2-2x_1x_2 -7=0 \Leftrightarrow 10 m^2 - 4 m - 6=0 \Leftrightarrow \hoac{&m=1\\&m=-\dfrac{3}{5}.}$
\end{enumerate}
}
\end{ex}
\begin{ex}%[SGD-HaNoi-2012]%[9H3B3]%[9H3B7]
Cho đường tròn $(O; R)$ và đường kính $AB$. Bán kính $CO$ vuông góc với $AB$, $M$ là điểm bất kì trên cung nhỏ $AC$ ($M$ khác $A$ và $C$), $BM$ cắt $AC$ tại $H$. Gọi $K$ là hình chiếu của $H$ trên $AB$.
\begin{enumerate}
\item Chứng minh rằng tứ giác $CBKH$ là tứ giác nội tiếp.
\item Chứng minh rằng $\widehat{ACM}=\widehat{ACK}$.
\item Trên đoạn thẳng $BM$ lấy điểm $E$ sao cho $BE=AM$. Chứng minh rằng tam giác $ECM$ là tam giác vuông cân tại $C$.
\item Gọi $d$ là tiếp tuyến của đường tròn $(O)$ tại điểm $A$. Gọi $P$ là một điểm nằm trên $d$ sao cho hai điểm $P, C$ nằm trong cùng một nửa mặt phẳng bờ $AB$ và $\dfrac{AP \cdot MB}{MA}=R$.\\
Chứng minh rằng đường thẳng $PB$ đi qua trung điểm của đoạn thẳng $HK$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
\tkzDefPoints{-1/1/A,5/1/B,2/4/C,-0.25/2.99/M,3.01/1.75/E,-1/2.14/P,0.65/1/K}
\tkzDefMidPoint(A,B) \tkzGetPoint{O}
\tkzInterLL(A,C)(B,M) \tkzGetPoint{H}
\tkzInterLL(P,B)(H,K) \tkzGetPoint{I}
\tkzDrawPoints[fill,size=6](A,B,C,O,M,P,E,H,K,I)
\tkzDrawCircle(O,A)
\tkzMarkRightAngle(P,A,B)
\tkzMarkRightAngle(A,K,H)
\tkzDrawSegments(A,B B,C C,A M,A M,B P,A P,B H,K C,E C,O C,M C,K)
\tkzLabelPoints[below](O,K)
\tkzLabelPoints[left](A,P)
\tkzLabelPoints[right](B)
\tkzLabelPoints[above](C,H)
\tkzLabelPoints[above left](M)
\tkzLabelPoints[above right](E,I)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item[a)] Ta có $\widehat{HKB} =\widehat{HCB}=90^\circ$ cho nên tứ giác $CBHK$ nội tiếp.
\item[b)] Các tứ giác $ABCM$ và $CBHK$ nội tiếp suy ra $\widehat{ACM} =\widehat{ABM}=\widehat{ACK}.$
\item[c)] Hai tam giác $MAC$ và $EBC$ có $\widehat{MAC}=\widehat{EBC}$, $MA=BE$ và $AC=BC$ cho nên hai tam giác bằng nhau, suy ra $MC=EC \quad (1)$ và $\widehat{ACM} =\widehat{BCE}$;  
$\widehat{MCE} =\widehat{ACM} +\widehat{ACE} = \widehat{ACE} +\widehat{BCE}=90^\circ \quad (2)$. \\ 
Từ $(1)$ và $(2)$ suy ra tam giác  $ECM$ vuông cân tại $C$. 
\item[d)] Ta có $\dfrac{IK}{AP} = \dfrac{KB}{2R} \Rightarrow IK = \dfrac{AP \cdot KB}{2R} = \dfrac{R \cdot MA  \cdot KB}{2R \cdot MB} \quad (1)$.  \\
Dễ thấy hai tam giác vuông $ABM$ và $HBK$ đồng dạng cho nên $\dfrac{MA}{MB} = \dfrac{HK}{KB} \quad (2)$. \\
Từ $(1)$ và $(2)$ suy ra $IK=\dfrac{HK}{2}$. 
\end{enumerate}
}
\end{ex}
\begin{ex}%[SGD-HaNoi-2012]%[9D5K2]
Với $x, y$ là các số dương thoả mãn điều kiện $x\ge 2y$, tìm giá trị nhỏ nhất của biểu thức $M=\dfrac{x^2+y^2}{xy}$.
\loigiai{
Đặt $t=\dfrac{x}{y}\ge 2\Rightarrow M=\dfrac{t^2+1}{t}$. \\
Xét hiệu $M-\dfrac{5}{2} = \dfrac{t^2+1}{t} - \dfrac{5}{2} = \dfrac{(t-2)(2t-1)}{2t} \geq 0$ với mọi $t \geq 2$. \\
Vậy giá trị nhỏ nhất của $M$ bằng $\dfrac{5}{2}$ khi $x=2y$. 
}
\end{ex}
