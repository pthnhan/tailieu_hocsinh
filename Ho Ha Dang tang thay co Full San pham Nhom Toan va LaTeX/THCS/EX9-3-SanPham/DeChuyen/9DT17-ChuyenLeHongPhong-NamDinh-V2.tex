\begin{ex}%[Chuyên Lê Hồng Phong Nam Định vòng 2, 2017]%[9D1B8]%[9D5B5]
\hfill
\begin{enumerate}
\item[a)] Tìm tất cả các số tự nhiên $x$ thỏa mãn $\left ( \dfrac{1}{\sqrt{x} } - \dfrac{2}{\sqrt{x}-1}\right )\left (\dfrac{1}{\sqrt{x}+1}-1\right ) \geq 1 $ 
\item[b)]  Với  $a,b,c$ là các số thực thỏa mãn các điều kiện $a+b+c =3$ và $\dfrac{1}{a} + \dfrac{1}{b} +\dfrac{1}{c} =\dfrac{1}{3}$. Tính giá trị của biểu thức $P=(a-3)^{2017}.(b-3)^{2017}.(c-3)^{2017}.$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a)] Điều kiện xác định:  $x \geq  2$. \\
BPT $\iff \dfrac{(\sqrt{x}+1)\sqrt{x}}{\sqrt{x}(\sqrt{x}-1)(\sqrt{x}+1)} \geq 1 \iff \dfrac{1}{\sqrt{x} -1} \geq 1 \iff \sqrt{x}  \leq 2 \iff x \leq 4 \Longrightarrow x=2,3,4$.    
\item[b)]   Ta có $(a-3)(b-3)(c-3)=abc + 9(a+b+c)-3(ab+bc+ca) -27 = abc -3(ab+bc+ca)$, mặt khác $\dfrac{1}{a} + \dfrac{1}{b} +\dfrac{1}{c} =\dfrac{1}{3} \iff abc=3(ab+bc+ca)$, cho nên $P=0$
\end{enumerate}
}
\end{ex}

\begin{ex}%[Chuyên Lê Hồng Phong Nam Định vòng 2, 2017]%[9D4K7]%[9D3G4]
\hfill
\begin{enumerate}
\item[a)] Giải phương trình $\left (\sqrt{x+5} - \sqrt{x+1}\right )\left (\sqrt{x^2+6x+5} +1\right ) =4$
\item[b)] Giải hệ phương trình $\heva{&2\sqrt{x+3y+2} - 3\sqrt{y} = \sqrt{x+2}  \\& x^2-3x-4\sqrt{y} +10 =0}$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a)] Điều kiện xác định: $x \geq -1$. \\
PT $\iff \sqrt{x^2+6x+5} +1 = \sqrt{x+5} + \sqrt{x+1} \iff \left (\sqrt{x+5} -1\right )\left (\sqrt{x+1}-1\right ) =0$ $\iff x=0$ hoặc $x=-4$ (loại).
\item[b)] Điều kiện xác định: $x \geq -2, y\geq 0$ . \\
Áp dụng BĐT Cauchy-Schwarz  ta có $\sqrt{3}.\sqrt{3y} + \sqrt{x+2} \leq \sqrt{(3+1)(3y +x+2)}$,
cho nên phương trình thứ nhất tương đương với $\sqrt{y} =\sqrt{x+2}$, thay vào phương trình thứ hai ta được 
$x^2-3x-4\sqrt{x+2} +10=0 \iff (x-2)^2 + \dfrac{(x-2)^2}{x+6 +4\sqrt{x+2}} =0$ $\iff (x-2)^2=0$ hoặc $1+\dfrac{1}{x+6+4\sqrt{x+2}} =0$ (vô nghiệm do $x+2 \geq 0$). Vậy nghiệm của hệ là $(x;y) = (2;4)$. 
\end{enumerate}
}
\end{ex}

Cho đường tròn $(O)$, từ điểm $A$ nằm ngoài đường tròn $(O)$ kẻ hai tiếp tuyến $AB$ và $AC$ với đường tròn $(O)$ ($B,C$ là các tiếp điểm). Gọi $H$ là giao điểm của  $AO$ và $BC$, $I$ là trung điểm của $BH$. Đường thẳng qua $I$ vuông góc với $OB$ cắt $(O)$ tại hai điểm $D,K$ ($D$ thuộc cung nhỏ $BC$). \item[a)] Chứng minh tứ giác $ICEF$ nội tiếp. 
\item[b)]  Chứng minh rằng: $\widehat{DBH}=2\widehat{DKH}$.
\item[c)]  Chứng minh rằng: $BD.CE=BE.CD$ và $BF.CE^2=BE.CD^2$.
\end{enumerate}

\item[b)]  Dễ thấy hai tam giác $BCD$ và $BDI$ đồng dạng cho nên $\dfrac{BD}{BI} = \dfrac{BC}{BD} \Longrightarrow BD^2 = BI.BC=BH^2 \Longrightarrow BD=BH$, suy ra $D,H,K$ thuộc đường tròn tâm $B$

\item[c)] Dễ thấy các cặp tam giác $ABE$, $ADB$ và $ACE$, $ADC$ đồng dạng cho nên $\dfrac{BE}{BD}=\dfrac{AE}{AB}$, $\dfrac{CE}{CD}=\dfrac{AE}{AC}$, kéo theo $\dfrac{BE}{BD}=\dfrac{CE}{CD} \Longrightarrow BD.CE=BE.CD$ \\
Tứ giác $ICEF$ nội tiếp cho nên $BF.BE=BI.BC=BD^2 \Longrightarrow \dfrac{BD}{BE} = \dfrac{BF}{BD} = \dfrac{CD}{CE} \Longrightarrow BD.CD=BF.CE \iff BD.CD.CE = BF.CE^2 \iff BF.CE^2=BE.CD^2.$
\end{enumerate}

\begin{ex}%[Chuyên Lê Hồng Phong Nam Định vòng 2, 2017]%[9D5K2]
\hfill
\begin{enumerate}
\item[a)] Tìm các số nguyên $x,y$ thỏa mãn phương trình $x^3+1=4y^2$.
\item[b)]  Tìm các số tự nhiên $x$ thỏa mãn biểu thức $B=x^4-x^2-10x-25$ là số nguyên tố.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item[a)] $x^3+1=4y^2 \iff x^3=(2y-1)(2y+1)$, $(2y-1,2y+1)=1$ cho nên $2y+1=a^3, 2y-1=b^3$ $(a>b)$, $a^3-b^3=2 \iff (a-b)(a^2+ab+b^2)=2$.\\
Nếu $a-b=1$ thì $a^2+b^2+ab=2$, hệ không có nghiệm nguyên. \\
Nếu $a-b=2$ thì $a^2+b^2+ab=1$, hệ có nghiệm $a=1$, $b=-1$. \\
Vậy nghiệm của phương trình  $x=-1$, $y=0$. 
\item[b)] $B=(x^2+x+5)(x^2-x-5)$, $x^2+x+5 >0$ cho nên $x^2-x-5=1 \iff x=3,x=-2$, nếu $x=3$, $B=17$ (thỏa mãn). 

\end{enumerate}
}
\end{ex}

\begin{enumerate}

\item[b)] Cho tứ giác $ABCD$ nội tiếp đường tròn $(O)$ bán kính bằng $R=4 \mathrm{cm}$ ($O$ nằm trong tứ giác $ABCD$). Xét $33$ điểm phân biệt nằm trong tứ giác $ABCD$ sao cho không có $3$ điểm nào thẳng hàng. Chứng minh rằng trong 33 điểm đó luôn tìm được $3$ điểm là $3$ đỉnh của một tam giác có diện tích nhỏ hơn $\dfrac{3\sqrt{3}}{4}\mathrm{cm}^2$.
\end{enumerate}

\loigiai{
\begin{enumerate}
\item[a)] Ta có $\dfrac{1}{a+bc}+\dfrac{1}{b+ac} \geq \dfrac{4}{a+b+bc+ac} =\dfrac{4}{(a+b)(c+1)}$, kéo theo $P \geq \dfrac{4}{(a+b)(c+1)} + 4(a+b)(1+c) + (a+b)c  $. Áp dụng AM-GM $\dfrac{4}{(a+b)(c+1)} + 4(a+b)(1+c) \geq 2\sqrt{\dfrac{4}{(a+b)(c+1)} . 4(a+b)(1+c) } =8$, $a,b,c$ không âm cho nên $(a+b)c \geq 0$. Vậy $P_{\min} =8$ khi $a=b=\dfrac{1}{2}$ và $c=0$.
\item[b)] 
Bổ đề: Trong các tam giác nội tiếp đường tròn bán kính $R$, tam giác đều có diện tích lớn nhất và diện tích lớn nhất đó bằng  $\dfrac{3\sqrt{3}R^2}{4}$.
