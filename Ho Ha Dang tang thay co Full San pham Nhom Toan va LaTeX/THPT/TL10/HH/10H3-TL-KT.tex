\begin{bt}%[Đinh Bích Hảo]%[0H3B1]%1 
	Cho $A(1,5); B(4,-1); C(-4,-5)$. 
	\begin{enumerate}
		\item Viết phương trình tổng quát của đường thẳng chứa các cạnh của tam giác.
		\item Viết phương trình đường trung trực cạnh $BC$ và $AB$.
		\item Viết phương trình đường tròn ngoại tiếp tam giác $ABC$.
	\end{enumerate}
	\loigiai{
		
		\begin{enumerate}
			\item 
			Ta có $\overrightarrow{AB}=(3,-6)$; $\overrightarrow{BC}=(-8,-4)$, $\overrightarrow{AC}=(-5,-10)$.\\
			Phương trình tổng quát của đường thẳng chứa cạnh $AB$: $2(x-1)+(y-5)=0$ $\Leftrightarrow 2x+y-7=0$\\
			.\dotfill 0,5 điểm\\
			Phương trình tổng quat đường thẳng chứa cạnh $AC$: $2(x+4)-(y+5)=0$ $\Leftrightarrow 2x-y+3=0$ \\
			.\dotfill 0,5 điểm\\
			Phương trình tổng quát đường thẳng chứa cạnh $BC$: $(x-4)-2(y+1)=0$ $\Leftrightarrow x-2y-6=0$\\
			.\dotfill 0,5 điểm\\
			
			\item Gọi $I$ là trung điểm cạnh $BC$. Khi đó $I(0;-3)$. Phương trình đường trung trực cạnh $BC$ là $2(x-0)+(y+3)=0$ $\Leftrightarrow 2x+y+3=0$
			.\dotfill{0,5 điểm} 
			Gọi $J$ là trung điểm cạnh $AB$. Khi đó $J(\frac{5}{2};2)$. Phương trình đường trung trực cạnh $AB$ là $(x-\frac{5}{2})-2(y-2)=0$ $\Leftrightarrow 2x-4y+3=0$
			.\dotfill{0,5 điểm}        
			
			\item Vì tam giác $ABC$ vuông tại $B$ nên tâm đường tròn ngoại tiếp là trung điểm $O$ của cạnh $AC$. Ta có $O(\frac{-3}{2};0)$ và bán kính $OA=\dfrac{5\sqrt{5}}{2}$; $\overrightarrow{OA}=(\frac{-5}{2};-5)$. 
			.\dotfill{0,5 điểm} \\
			Phương trình đường tròn ngoại tiếp tam giác $ABC$ là $(x+\frac{3}{2})^2+y^2=\dfrac{125}{4}$ \dotfill 0,5 điểm.
		\end{enumerate}
		
	}
\end{bt}
\begin{bt}%[Đinh Bích Hảo]%[0H2B3]%[0H3K3]

		\item Tìm trên $(E)$ những điểm $M$ sao cho $M$ nhìn đoạn thẳng nối hai tiêu điểm dưới một góc vuông.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}

			Tâm sai $e=\dfrac{c}{a}=\dfrac{\sqrt{3}}{2}$ \dotfill 0,5 điểm\\

			.\dotfill{0,5 điểm}\\

			Tọa độ điểm $M$ là nghiệm hệ phương trình $\dfrac{x^2}{4}+\dfrac{y^2}{1}=1$ và $x^2+y^2=3$. \dotfill 0,5 điểm.\\

			.\dotfill{0,5 điểm}\\

			.\dotfill{0,5  điểm}\\
		\end{enumerate}     
	}
\end{bt}

	}
