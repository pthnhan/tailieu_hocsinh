\begin{ex}%[Thi thử L3, Chuyên Đại học Vinh, Nghệ An]%[Nguyễn Đắc Giáp, 12EX-10]%[2D2G4-7]
	Biết rằng $a$ là số thực dương sao cho bất đẳng thức $3^x+a^x\geqslant 6^x+9^x$ đúng với mọi số thực $x$. Mệnh đề nào sau đây đúng?
	\choice
	{$a\in (10;12]$}
	{\True $a\in (16;18]$}
	{$a\in (14;16]$}
	{$a\in (12;14]$}
	\loigiai{
	Ta có
	\begin{eqnarray*}
	3^x+a^x \geqslant 6^x+9^x &\Leftrightarrow& a^x-18^x \geqslant 6^x+9^x-3^x-18^x\\
	&\Leftrightarrow& a^x-18^x \geqslant 3^x\left(2^x-1\right)-9^x\left(2^x-1\right)\\
	&\Leftrightarrow& a^x-18^x \geqslant -3^x\left(2^x-1\right)\left(3^x-1\right) \quad (*)\\
	\end{eqnarray*}	
	Ta thấy
	\begin{itemize}
		\item Nếu $x\geqslant 0$ thì $\heva{&2^x \geqslant 1\\&3^x \geqslant 1}\Rightarrow \left(2^x-1\right)\left(3^x-1\right)\geqslant 0 \Rightarrow -3^x\left(2^x-1\right)\left(3^x-1\right) \le 0.$
		\item Nếu $x<0$ thì $\heva{&2^x<1\\&3^x<1}\Rightarrow \left(2^x-1\right)\left(3^x-1\right)>0 \Rightarrow -3^x\left(2^x-1\right)\left(3^x-1\right)<0.$
	\end{itemize}
	Suy ra $-3^x\left(2^x-1\right)\left(3^x-1\right) \le 0$ với mọi $x\in \mathbb{R}$.\\
	Do đó, $(*)$ đúng với mọi $x\in \mathbb{R}$ khi và chỉ khi:
	$$a^x-18^x\geqslant 0, \forall x\in \mathbb{R}\Leftrightarrow \left(\dfrac{a}{18}\right)^x\geqslant 1, \forall x\in \mathbb{R}\Leftrightarrow a=18.$$
	Vậy $a=18\in (16;18]$.
}
\end{ex}
