\begin{bt}%[Đề HK1, Võ Thị Sáu, Hồ Chí Minh 17-18][Nguyễn Ngọc Thanh Sơn][1D1B3-1]
	Giải phương trình $2\cos^2{2x}+3\sqrt{2}\sin^2{x} + 2 = 3\sqrt{2}\cos^2{x}.$
	\loigiai{\begin{eqnarray*}
			2\cos^2{2x}+3\sqrt{2}\sin^2{x} + 2 = 3\sqrt{2}\cos^2{x}
			&\Leftrightarrow &2\cos^2{2x}+3\sqrt{2}\sin^2{x} -3\sqrt{2}\cos^2{x} + 2 = 0\\
			&\Leftrightarrow &2\cos^2{2x}+3\sqrt{2} \left(\sin^2{x} -\cos^2{x}\right) + 2 = 0\\
			&\Leftrightarrow &2\cos^2{2x}-3\sqrt{2} \cos{2x} + 2 = 0\\
			&\Leftrightarrow &\hoac{&\cos{2x}=\sqrt{2}\quad \text{(loại)}\\ &\cos{2x}=\dfrac{\sqrt{2}}{2}\quad \text{(nhận)}}\\
			&\Leftrightarrow &\cos{2x} = \dfrac{\sqrt{2}}{2}\\
			&\Leftrightarrow &\cos{2x} = \cos{\dfrac{\pi}{4}}\\
			&\Leftrightarrow &\hoac{& 2x = \dfrac{\pi}{4}+k2\pi\\& 2x = -\dfrac{\pi}{4}+k2\pi}\\
			&\Leftrightarrow &\hoac{& x = \dfrac{\pi}{8}+k\pi\\& x = -\dfrac{\pi}{8}+k\pi}, k\in \Bbb Z.
		\end{eqnarray*}
		Vậy nghiệm của phương trình $x = \dfrac{\pi}{8}+k\pi$, $x = -\dfrac{\pi}{8}+k\pi, k\in \mathbb{Z}$.}
\end{bt}

		Tìm số hạng chính giữa trong khai triển nhị thức $\left(1+2x\right)^{12}$.
	\loigiai{
			Tìm số hạng chính giữa trong khai triển nhị thức $\left(1+2x\right)^{12}$.\\
			Vì $n=12$ nên có $13$ số hạng\\
			$\Rightarrow$ số hạng chính giữa là số hạng thứ $7$.\\
			Ta có: $T_7=\mathrm{C}_{12}^6\left(2x\right)^6 =\mathrm{C}_{12}^6 2^6x^6$ \\
			Vậy số hạng chính giữa trong khai triển nhị thức là $T_7=\mathrm{C}_{12}^6 2^6x^6$.
	}

\begin{bt}%[Đề HK1, Võ Thị Sáu, Hồ Chí Minh 17-18][Nguyễn Ngọc Thanh Sơn][1D2K3-2] 
		Tìm hệ số của $x^6$ trong khai triển đa thức $P(x)=\left(1+x^3\right)\left(1+2x\right)^{12}$.
	\loigiai{
			Tìm hệ số của $x^6$ trong khai triển đa thức $P(x)=\left(1+x^3\right)\left(1+2x\right)^{12}$.\\
			Ta có
			\begin{eqnarray*}
				P(x)=\left(1+x^3\right)\left(1+2x\right)^{12}
				&=&\left(1+2x\right)^{12}+x^3\left(1+2x\right)^{12}\\
				&=&\displaystyle \sum_{k=0}^{12}\mathrm{C}_{12}^k(2x)^{k} + x^3\displaystyle \sum_{k=0}^{12}\mathrm{C}_{12}^k(2x)^{k}\\
				&=&\displaystyle \sum_{k=0}^{12}\mathrm{C}_{12}^k 2^{k}x^{k} + \displaystyle \sum_{k=0}^{12}\mathrm{C}_{12}^k 2^{k}x^{k+3}.
			\end{eqnarray*}
			Suy ra hệ số của $x^6$ trong khai triển đa thức trên là $\mathrm{C}_{12}^6 2^{6} + \mathrm{C}_{12}^3 2^{3}=60896.$
	}
\end{bt}
\begin{bt}%[Đề HK1, Võ Thị Sáu, Hồ Chí Minh 17-18][Nguyễn Ngọc Thanh Sơn][1D2B5-3][1D2B5-3]
	Đội văn nghệ của trường trung học phổ thông Võ Thị Sáu có $12$ học sinh gồm $5$ học sinh khối 10; $4$ học sinh khối 11 và $3$ học sinh khối 12. Cần chọn $4$ học sinh đi thi văn nghệ. Tính xác suất để chọn được
	\begin{enumerate}
		\item Bốn học sinh thuộc cùng một khối.
		\item Bốn học sinh thuộc không quá $2$ trong $3$ khối.
	\end{enumerate}
	\loigiai{
		Chọn $4$ học sinh trong $12$ học sinh nên số cách chọn là $\mathrm{C}_{12}^4$
		$\Rightarrow \vert \Omega \vert=\mathrm{C}_{12}^4$.
		\begin{enumerate}
			\item Bốn học sinh thuộc cùng một khối.\\
			Gọi $A$ là biến cố bốn học sinh thuộc cùng một khối.\\
			Trường hợp 1: Bốn học sinh thuộc cùng khối $10$ có $\mathrm{C}_{5}^4$ cách.\\
			Trường hợp 2: Bốn học sinh thuộc cùng khối $11$ có $\mathrm{C}_{4}^4$ cách.\\
			Theo quy tắc cộng: $\vert A \vert=\mathrm{C}_{5}^4+\mathrm{C}_{4}^4$.\\
			Vậy $\mathrm{P}(A)=\dfrac{\vert A \vert}{\vert \Omega \vert}=\dfrac{\mathrm{C}_{5}^4+\mathrm{C}_{4}^4}{\mathrm{C}_{12}^4}=\dfrac{6}{495}$.
			\item Bốn học sinh thuộc không quá $2$ trong $3$ khối.\\
			Gọi $B$ biến cố chọn bốn học sinh thuộc cả ba khối.\\
			Trường hợp 1: Hai học sinh thuộc cùng khối $10$, các khối còn lại mỗi khối có một học sinh.\\
			$\Rightarrow$Có $\mathrm{C}_{5}^2.\mathrm{C}_{4}^1. \mathrm{C}_{3}^1 = 120$ cách.\\
			Trường hợp 2: Hai học sinh thuộc cùng khối $11$, các khối còn lại mỗi khối có một học sinh.\\
			$\Rightarrow$Có $\mathrm{C}_{5}^1.\mathrm{C}_{4}^2. \mathrm{C}_{3}^1 = 90$ cách.\\
			Trường hợp 3: Hai học sinh thuộc cùng khối $12$, các khối còn lại mỗi khối có một học sinh.\\
			$\Rightarrow$Có $\mathrm{C}_{5}^1.\mathrm{C}_{4}^1. \mathrm{C}_{3}^2 = 60$ cách.\\
			Theo quy tắc cộng: $\vert B \vert=120+90+60=270$.\\
			$\Rightarrow \mathrm{P}(B)=\dfrac{\vert B \vert}{\vert \Omega \vert}=\dfrac{270}{495}=\dfrac{6}{11}$.\\
			Do đó xác suất bốn học sinh thuộc không quá $2$ trong $3$ khối là\\
			$\mathrm{P}(\bar B)=1-\mathrm{P}(B)=1-\dfrac{6}{11}=\dfrac{5}{11}$.
		\end{enumerate}	
	}
\end{bt}
\begin{bt}%[Đề HK1, Võ Thị Sáu, Hồ Chí Minh 17-18][Nguyễn Ngọc Thanh Sơn][1D1K3-2]
	Giải phương trình $\sin^6 {x} + \cos^6{x} + \dfrac{1}{2}\sin{4x}=0.$
	\loigiai{
		Ta có
		\begin{eqnarray*}
			\sin^6 {x} + \cos^6{x} + \dfrac{1}{2}\sin{4x}=0
			&\Leftrightarrow &\left(\sin^2 {x} + \cos^2{x}\right)\left(\sin^4 {x} -\sin^2{x}\cos^2{x}+ \cos^4{x}\right)+ \dfrac{1}{2}\sin{4x}=0\\
			&\Leftrightarrow &\left(\sin^2 {x} + \cos^2{x}\right)^2-2\sin^2{x}\cos^2{x} -\sin^2{x}\cos^2{x}+ \dfrac{1}{2}\sin{4x}=0\\
			&\Leftrightarrow &1-3\sin^2{x}\cos^2{x}+ \sin{2x}\cos{2x}=0\\
			&\Leftrightarrow &\sin^2{2x}+\cos^2{2x}-\dfrac{3}{4}\sin^2{2x}+ \sin{2x}\cos{2x}=0\\
			&\Leftrightarrow &\cos^2{2x}+\dfrac{1}{4}\sin^2{2x}+ \sin{2x}\cos{2x}=0. \quad (1)
		\end{eqnarray*}
		Trường hợp 1: $\cos{2x}=0 \Leftrightarrow x=\dfrac{\pi}{4}+k\dfrac{\pi}{2}, k\in \Bbb Z$. Thế vào $(1) \Leftrightarrow \dfrac{1}{4}=0$(vô lý).\\
		Trường hợp 2: $\cos{2x}\ne 0 \Leftrightarrow x\ne \dfrac{\pi}{4}+k\dfrac{\pi}{2}$, chia hai vế cho $\cos^2{2x}$\\
		$(1)\Leftrightarrow 1+\dfrac{1}{4}\tan^2{2x}+\tan{2x}=0\\
		\Leftrightarrow \tan{2x}=-2 \Leftrightarrow x=\dfrac{1}{2}\arctan{(-2)}+k\dfrac{\pi}{2}, k\in \Bbb Z$.\\
		Vậy nghiệm của phương trình $x=\dfrac{1}{2}\arctan{(-2)}+k\dfrac{\pi}{2}, k\in \Bbb Z$.
	}
\end{bt}

	Cho hình chóp $SABCD$ có đáy $ABCD$ là hình thang $AD \parallel BC$ và $AD = 2BC$.
	\begin{enumerate}
