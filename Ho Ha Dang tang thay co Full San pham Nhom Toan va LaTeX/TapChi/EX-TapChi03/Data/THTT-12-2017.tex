		{\it Cách 1.}\\
		Ta có
		\begin{align*}
		\overline{ab}^2+\overline{ba}^2&= \left(10a+b\right)^2+\left(10b+a\right)^2\\
		& =100a^2+20ab+b^2+100b^2+20ab+a^2\\
		& =101\left(a^2-2ab+b^2\right)+242ab\\
		&=101(a-b)^2+242ab \tag*{(1)}
		\end{align*}
		và $ \overline{abba}=1001a+110b=1111a-110(a-b) $.\hfill $ (2) $\\
		Từ giả thiết  và $ (1), (2) $ có
		\begin{align*}
		&\; 1111a-110(a-b)=101\left(a-b\right)^2+242ab +(a-b)\\
		\Leftrightarrow &\; 1111a-242ab=101\left(a-b\right)^2+111(a-b)\\
		\Leftrightarrow &\; 11a(101-22b)=(a-b)\left[101(a-b)+111\right].\tag*{(3)}
		\end{align*}
		Từ $ (3) $ nếu 11 là ước của $ a-b $ mà $ a-b\leq 9 $ thì $ a-b=0 $, suy ra $ 11a(101-22b)=0 $, điều này không xảy ra. Vậy 11 là ước của $$101(a-b)+111=99(a-b)+110+2(a-b)+1,

		Từ $\overline{abba}+b-a=1001a+110b+b-a=1000a+111b $ và theo giả thiết ta có \begin{align*}
		\overline{ab}^2+\overline{ba}^2=\overline{abbb}.\tag*{(4)}
		\end{align*}

		\begin{enumerate}[$ \bullet $]
			\item Nếu cả $ a $ và $ b $ đều chẵn thì vế trái của $ (4) $ chia hết cho 4, xét vế phải thì $ b $ chia hết cho 4.\\

			\item Nếu $ a $ chẵn và $ b $ lẻ thì vế trái của $ (4) $ chia cho 4 dư 1, xét vế phải thì $ b $ bằng 3 hoặc 7. \\

		\end{enumerate}
	}

	Tìm nghiệm nguyên của phương trình $ \sqrt{3x+4}=\sqrt[3]{y^3+5y^2+7y+4} $.\hfill (1)
	
	\loigiai{
		{\it Trước hết ta chứng minh nhận xét sau.}\\
		Với $ a\geq 0, a\in\mathbb{Z}, a\sqrt{a}\in\mathbb{Z} $ thì $ \sqrt{a}\in\mathbb{Z} $. Thật vậy
		\begin{enumerate}[$ \bullet $]
			\item Xét $ a=0 $ thì $ \sqrt{0}=0\in\mathbb{Z} $.
			\item Xét $ a $ nguyên dương, đặt $$a\sqrt{a}=b\ (b\in\mathbb{Z})\Leftrightarrow \sqrt{a}=\dfrac{b}{a}\Rightarrow a=\dfrac{b^2}{a^2}.$$
			Nếu $ \sqrt{a}\notin\mathbb{Z} $ thì $ b $ không chia hết cho $ a $, $ b^2 $ không chia hết cho $ a^2 $, do đó $ a=\dfrac{b^2}{a^2}\notin\mathbb{Z} $. Điều này mâu thuẫn với $ a\in\mathbb{Z} $. Vậy $ \sqrt{a}\in\mathbb{Z} $.
		\end{enumerate}
		{\it Trở lại bài toán.}\\
		Điều kiện $ x\geq -\dfrac{4}{3} $. Lập phương hai vế của $ (1) $ ta có \begin{align*}
		(3x+4)\sqrt{3x+4}=y^3+5y^2+7y+4. \tag*{(2)}
		\end{align*}
		Do $ x, y\in\mathbb{Z} $, kết hợp với $ (2) $ suy ra $$3x+4\in\mathbb{Z},\qquad (3x+4)\sqrt{3x+4}\in\mathbb{Z}.$$
		Sử dụng nhận xét suy ra $ \sqrt{3x+4}\in\mathbb{Z} $. Đặt $ \sqrt{3x+4}=k\ (k\geq 0, k\in\mathbb{Z}) $, thay vào $ (2) $ ta có
		\begin{align*}
		k^3=y^3+5y^2+7y+4. \tag*{(3)}
		\end{align*}
		Nhận thấy
		\begin{enumerate}[]
			\item $ y^3+5y^2+7y+4=(y+1)^3+2(y+1)^2+1>(y+1)^3 $ (do $ 2(y+1)^2+1>0 $ với mọi $ y $);
			\item $ \begin{aligned}[t]
			y^3+5y^2+7y+4&= (y+2)^3-\left(y^2+5y+4\right)\\
			&=(y+2)^3-(y+1)(y+4)<(y+2)^3
			\end{aligned} $\\
			khi $ y<-4 $ hoặc $ y>-1 $ (do $ (y+4)(y+1)>0 $ khi $ y<-4 $ hoặc $ y>-1 $).
		\end{enumerate} 
		\begin{enumerate}[$ \bullet $]
			\item Với $ y<-4 $ hoặc $ y>-1 $, kết hợp với $ (3) $ suy ra $ (y+1)^3<k^3<(y+2)^3 $, điều này không thể xảy ra.
			\item Với $ y=-4 $ thay vào $ (3) $ ta có $ k^3=-8\Leftrightarrow k=-2 $ (không thỏa mãn điều kiện $ k>0 $).
			\item Với $ y=-3 $  thay vào $ (3) $ ta có $ k^3=1\Leftrightarrow k=1 $. Suy ra $ x=-1 $ (thỏa mãn).
			\item Với $ y=-2 $  thay vào $ (3) $ ta có $ k^3=2\Leftrightarrow k=\sqrt[3]{2} $ (không thỏa mãn điều kiện $ k\in\mathbb{Z} $).
			\item Với $ y=-1 $  thay vào $ (3) $ ta có $ k^3=1\Leftrightarrow k=1 $. Suy ra $ x=-1 $ (thỏa mãn).
		\end{enumerate}
		Vậy phương trình $ (1) $ có hai nghiệm nguyên $ (x; y) $ là $ (-1; -1), (-1; -3) $.
	\begin{nx}
		Bài toán yêu cầu giải phương trình nghiệm nguyên có hai ẩn và chứa căn thức bậc hai, bậc ba, ta thường dùng tính chất của số nguyên và của bất đẳng thức để hạn chế miền nghiệm của một ẩn, từ đó tìm nghiệm của ẩn còn lại. 
	\end{nx}
}

	\loigiai{
		Điều kiện $ x\geq 2; y\geq 2; z\geq 2 $.\\
		Xét hàm số $ f(t)=t^3-7t+\sqrt{t-2},\ t\geq 2 $.\\
		Với hai số thực phân biệt $ a, b $ ta có \begin{align*}
		f(a)-f(b)&= a^3-b^3-7(a-b)+\sqrt{a-2}-\sqrt{b-2}\\
		&= (a-b)\left(a^2+ab+b^2\right)-7(a-b)+\dfrac{a-b}{\sqrt{a-2}+\sqrt{b-2}}\\
		&=(a-b)\left(a^2+ab+b^2-7+\dfrac{1}{\sqrt{a-2}+\sqrt{b-2}}\right).
		\end{align*}
		Với $\begin{aligned}[t]
		a, b\geq 2 & \Rightarrow a^2+ab+b^2\geq 8\\
		& \Rightarrow a^2+ab+b^2-7+\dfrac{1}{\sqrt{a-2}+\sqrt{b-2}}>0
		\end{aligned} $\\
		nên $ a>b\Leftrightarrow f(a)>f(b) $.\\
		Hệ phương trình trong đầu bài tương đương với $$\heva{& f(x)=y+4\\ & f(y)=z+4\\ & f(z)=x+4}.$$
		Giả sử $ x=\max (x, y, z) $.\hfill (1)\\
		Nếu $\begin{aligned}[t]
		x>y &\Rightarrow f(x)>f(y)\Rightarrow y+4>z+4\Rightarrow y>z\\
		& \Rightarrow f(y)>f(z)\Rightarrow z+4>x+4\Rightarrow z>x
		\end{aligned}$\\
		trái với giả thiết $ (1) $.\\
		Vậy $ x=y\Rightarrow f(x)=f(y)\Rightarrow y+4=z+4\Rightarrow x=y=z $.\\
		Thay vào một phương trình trong hệ, ta được
		\begin{align*}
		&\; x^3-7x+\sqrt{x-2}=x+4\\
		\Leftrightarrow &\; x^3-8x-4+\sqrt{x-2}=0\\
		\Leftrightarrow &\; x^2 (x-3)+3x(x-3)+(x-3)+\left(\sqrt{x-2}-1\right)=0\\
		\Leftrightarrow &\; (x-3)\left(x^2+3x+1+\dfrac{1}{\sqrt{x-2}+1}\right)=0\\
		\Leftrightarrow &\; x=3\ (\text{vì}\  x^2+3x+1+\dfrac{1}{\sqrt{x-2}+1}>0,\ \forall x\geq 2). 
		\end{align*}
		Vậy hệ phương trình có nghiệm duy nhất $ (x; y; z)=(3; 3; 3) $.
	\begin{nx}
		Vì hệ phương trình trong đầu bài không thay đổi khi hoán vị vòng quanh theo thứ tự $ x\to y\to z \to x $ nên có thể giả thiết $ x=\max (x, y, z) $ nghĩa là $ x\geq y; x\geq z $. Điều then chốt của bài toán là chứng minh $ x=y=z $.
	\end{nx}
