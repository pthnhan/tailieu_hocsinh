\begin{bt}[2,0 điểm]%[Trần Đình Phương]%[1D2K3]
	Tìm số nguyên dương $n$ bé nhất sao cho trong khai triển $(1+x)^n$ với số mũ giảm dần có hai hệ số liên tiếp có tỉ số bằng $3$.
	\loigiai{
		Ta có $\dfrac{\mathrm{C}^k_n}{\mathrm{C}^{k+1}_n}=\dfrac{k+1}{n-k}$ \dotfill $0,50$ điểm.\\
		Theo giả thiết: $\dfrac{\mathrm{C}^k_n}{\mathrm{C}^{k+1}_n}=3\Leftrightarrow \dfrac{k+1}{n-k}=3\Leftrightarrow n=k+\dfrac{k+1}{3}$ \dotfill $0,50$ điểm.\\
		Đặt $t=\dfrac{k+1}{3}$. Suy ra $k=3t-1$ và $n=4t-1$\\
		Vì $k\ge 0$ nên $t\ge \dfrac{1}{3}$.\dotfill $0,50$ điểm.\\
		Ta lại có $n=4t-1$ nhỏ nhất khi và chỉ khi $t$ nhỏ nhất. \\
		Điều này xảy ra khi $t=1$. Và do đó, $n=3$\dotfill$0,50$ điểm.
	}
\end{bt}

\begin{bt}[2,0 điểm]%[Trần Đình Phương]%[1D2G5]
	Trong kì kiểm tra chất lượng đầu năm ở trường Hồng Vân cho hai lớp $11B1$ và $11B2$. Lớp $11B1$ có $25\%$ học sinh có điểm môn Toán dưới $5$, $10\%$ học sinh có điểm môn Lý $5$ và $5\%$ học sinh có điểm cả môn Toán lẫn Lý dưới $5$. Tính xác suất sao cho
	\begin{enumerate}
		\item hai học sinh đó đều có điểm môn Toán dưới $5$.
		\item hai học sinh đó không có học sinh nào có điểm dưới $5$ ở cả hai môn Toán và Lý.
	\end{enumerate}
	\loigiai{
		Với $i=1,\,2$, gọi:\\
		$T_i$ là biến cố:" học sinh được chọn từ lớp $11Bi$ có điểm Toán dưới $5$.\\
		$L_i$ là biến cố: "học sinh được chọn từ lớp $11Bi$ có điểm Lý dưới $5$".
		Dễ thấy, các biến cố $T_i$ là độc lập với nhau, các biến cố $L_i$ là độc lập với nhau.
		\begin{enumerate}
			\item Gọi $A$ là biến cố: "hai học sinh được chọn có điểm Toán dưới $5$"
			\\Ta có $\mathrm{P}(A)=\mathrm{P}(T_1.T_2)=\mathrm{P}(T_1).\mathrm{P}(T_2)=\dfrac{1}{16}$\dotfill $0,50$ điểm.
			\item Gọi $B$ là biến cố: "hai học sinh được chọn không có học sinh nào có điểm dưới $5$ ở cả hai môn Toán và Lý.
			\\Ta có:\\
			$\mathrm{P}(T_1\cup L_1)=\mathrm{P}(T_1)+\mathrm{P}(L_1)-\mathrm{P}(T_1L_1)=\dfrac{3}{10}$.\dotfill $0,50$ điểm.\\
			$\mathrm{P}(T_2\cup L_2)=\mathrm{P}(T_2)+\mathrm{P}(L_2)-\mathrm{P}(T_2L_2)=\dfrac{7}{20}$.\dotfill $0,50$ điểm.\\
			$\mathrm{P}(B)=\mathrm{P}\left(\overline{T_1\cup L_1}\cap\overline{T_2\cup L_2}\right)= \mathrm{P}(\overline{T_1\cup L_1}).\mathrm{P}(\overline{T_2\cup L_2})$\\$=\left(1-\dfrac{3}{10}\right)\left(1-\dfrac{7}{20}\right)=\dfrac{91}{200}$.\dotfill $0,50$ điểm.
		\end{enumerate}
	}
\end{bt}
