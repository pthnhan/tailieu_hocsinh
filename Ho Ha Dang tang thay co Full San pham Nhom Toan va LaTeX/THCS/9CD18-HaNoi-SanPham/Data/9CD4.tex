\begin{bt}%[9D0B2]%[9D0K2]%[9D0G2]
	Cho phương trình bậc hai $x^2-x+m=0$ (với $m$ là tham số).
	\begin{enumerate}
		\item Tìm tất cả các giá trị của tham số $m$ để phương trình đã cho có hai nghiệm trái dấu.
		\item Giải phương trình với $m=\sqrt{2}-2$.
		\item Tìm tất cả các giá trị của tham số $m$ để phương trình đã cho có hai nghiệm $x_1, x_2$  thỏa mãn biểu thức $S=x_1^4+x_2^4-x_1^5-x_2^5$  đạt giá trị lớn nhất.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Phương trình đã cho có hai nghiệm trái dấu khi và chỉ khi $a\cdot c<0 \Leftrightarrow m<0$.
		\item Với $m=\sqrt{2}-2$, phương trình trở thành 
		$$x^2-x +\sqrt{2}-2=0 \Leftrightarrow (x^2-2)-(x-\sqrt{2})=0 \Leftrightarrow (x-\sqrt{2})(x+\sqrt{2}-1)=0 \Leftrightarrow \hoac{&x=\sqrt{2}\\&x=1-\sqrt{2}.}$$
		Vậy phương trình có hai nghiệm phân biệt là $x=\sqrt{2}$ , $x= 1-\sqrt{2}$.
		\item  Phương trình có 2 nghiệm  $x_1;x_2 \Leftrightarrow \Delta \geq 0 \Leftrightarrow 1-4m \geq 0 \Leftrightarrow m\leq \dfrac{1}{4}.\hfill (1)$\\
		Áp dụng hệ thức Vi-ét, ta có $\heva{&x_1+x_2=1\\&x_1x_2=m.}$\\
		Ta có
		\allowdisplaybreaks{
			\begin{eqnarray*}
			S&=&x_1^4+x_2^4-x_1^5-x_2^5\\
			&=&x_1^4(1-x_1)+x_2^4(1-x_2)\\
			&=&x_1^4\cdot x_2 +x_2^4 \cdot x_1\\
			&=& x_1x_2\left[ x_1^3+x_2^3\right] = x_1x_2 \left[(x_1+x_2)^3-3x_1x_2(x_1+x_2)\right]\\
			&=& m(1-3m) =-3\left(m-\dfrac{1}{6}\right)^2+\dfrac{1}{12}. 
			\end{eqnarray*}
		}
		Ta có $S=-3\left(m-\dfrac{1}{6}\right)^2+\dfrac{1}{12} \leq \dfrac{1}{12}$, đẳng thức xảy ra khi $m=\dfrac{1}{6}$ thỏa mãn $(1)$.\\
		Vậy giá trị lớn nhất của biểu thức $S=x_1^4+x_2^4-x_1^5-x_2^5$ bằng $\dfrac{1}{12}$, xảy ra khi $m=\dfrac{1}{6}$.
	\end{enumerate}
	}
\end{bt}

\begin{bt}%[9D0B2]%[9D0K2]%[9D0G2]
	Cho phương trình $2x^2-2mx+m^2-2=0$ ($m$ là tham số).
	\begin{enumerate}
		\item Giải phương trình khi $m=2$.
		\item Tìm $m$ để phương trình có $2$ nghiệm phân biệt $x_1;x_2$ thỏa mãn $x_1^3+x_2^3=\dfrac{5}{2}$.
		\item Tìm tham số $m$ để phương trình đã cho có hai nghiệm phân biệt $x_1, x_2$ thỏa mãn biểu thức $A=|x_1-x_2|$ đạt giá trị lớn nhất.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Khi $m=2$ ta có phương trình $x^2-2x+1=0 \Leftrightarrow (x-1)^2=0 \Leftrightarrow x=1$.\\
			Vậy với $m=2$, phương trình đã cho có nghiệm duy nhất $x=1$.
			\item Phương trình có 2 nghiệm phân biệt $x_1;x_2 \Leftrightarrow \Delta'>0 \Leftrightarrow 4-m^2>0 \Leftrightarrow -2<m<2$ $(1)$.\\
			Theo hệ thức vi-ét ta có  $\heva{&x_1+x_2=m\\&x_1x_2=\dfrac{m^2-2}{2}.}$ \\
			Ta có 
			\allowdisplaybreaks{
				\begin{eqnarray*}
					& & (x_1+x_2)\left[(x_1+x_2)^2-3x_1x_2\right]=\dfrac{5}{2}\\
					&\Rightarrow & m^3-6m+5=0\\
					&\Leftrightarrow & (m-1)(m^2+m-5)=0\\
					&\Leftrightarrow &  \hoac{&m=\dfrac{-1\pm \sqrt{21}}{2}\\&m=1.}
				\end{eqnarray*}
			}
			Đối chiếu với $(1)$ ta có $m=1;\, m=\dfrac{-1+\sqrt{21}}{2}$ thỏa mãn.
			\item Với điều kiện $-2<m<2$, ta có $\heva{&x_1+x_2=m\\&x_1x_2=\dfrac{m^2-2}{2}.}$ \\
			Ta có $A^2=(x_1+x_2)^2-4x_1x_2 = -m^2+4 \leq 4$.\\
			Suy ra $A\leq 2$, đẳng thức xảy ra khi $m=0$.\\
			Vậy $A$ đạt giá trị lớn nhất bằng $2$, khi $m=0$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[9D4B6]
	Cho phương trình $x^2-2(m+1)x+m=0$ với $m$ là tham số.
	\begin{enumerate}
		\item Chứng minh rằng phương trình đã cho luôn có hai nghiệm phân biệt với mọi giá trị của tham số $m$.
		\item Tìm các giá trị của tham số $m$ để phương trình đã cho có hai nghiệm phân biệt $x_1,x_2$ thoả mãn $\dfrac{2x_1-1}{x_2}+\dfrac{2x_2-1}{x_1}=x_1x_2+\dfrac{3}{x_1x_2}$.
		\item Tìm các giá trị của tham số $m$ để phương trình đã cho có hai nghiệm phân biệt $x_1, x_2$ thỏa mãn biểu thức $P=-\dfrac{1}{(2x_1-1)^2}-\dfrac{1}{(2x_2-1)^2}$ đạt giá trị lớn nhất.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item 	Ta có $\Delta'=(m+1)^2-m=m^2+m+1=\left(m+\dfrac{1}{2}\right)^2+\dfrac{3}{4}>0$ với mọi $m$ nên phương trình đã cho luôn có $2$ nghiệm phân biệt $x_1,x_2$.
			\item 	Áp dụng định lý Vi-ét, ta có $\heva{&x_1+x_2=2(m+1) \\ &x_1x_2=m.}$\\
			Khi đó
			\allowdisplaybreaks{
				\begin{eqnarray*}
					\dfrac{2x_1-1}{x_2}+\dfrac{2x_2-1}{x_1}=x_1x_2+\dfrac{3}{x_1x_2} &\Leftrightarrow& \dfrac{2\left(x_1^2+x_2^2\right)-\left(x_1+x_2\right)}{x_1x_2}=x_1x_2+\dfrac{3}{x_1x_2} \\ 
					&\Leftrightarrow& 2\left(x_1+x_2\right)^2-4x_1x_2-\left(x_1+x_2\right)=x_1^2x_2^2+3 \\
					&\Leftrightarrow& 8(m+1)^2-4m-2(m+1)=m^2+3 \\
					&\Leftrightarrow& 7m^2+10m+3=0 \Leftrightarrow \hoac{&m=-1 \\ &m=-\dfrac{3}{7}.}
				\end{eqnarray*}
			}
			Vậy $m\in\left\{-1;-\dfrac{7}{3}\right\}$.
			\item Ta có $(2x_1-1)\cdot(2x_2-1)=4x_1x_2-2(x_1+x_2)+1 =-1$.\\
			Áp dụng bất đẳng thức AM-GM, ta có 
			$$P=-\left[ \dfrac{1}{(2x_1-1)^2}+\dfrac{1}{(2x_2-1)^2}\right] \leq - 2\sqrt{\dfrac{1}{\left[ (2x_1-1)\cdot(2x_2-1)\right]^2}} =-2.$$
			Đẳng thức xảy ra khi $(2x_1-1)^2=(2x_2-1)^2 \Leftrightarrow \hoac{ &x_1=x_2 \quad \text{(loại)}\\& 2(x_1+x_2)=2}\Leftrightarrow m=-\dfrac{1}{2}$.\\
			Vậy $m=-\dfrac{1}{2}$ là giá trị cần tìm.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[9D4K6]
	Cho phương trình $x^2+(m^2+1)x+m-2=0\;\;(1)$, $m$ là tham số. 
	\begin{enumerate}
		\item Chứng minh rằng phương trình đã cho luôn có hai nghiệm phân biệt với mọi giá trị của tham số $m$.
		\item Tìm tham số $m$ để phương trình đã cho có hai nghiệm trái dấu.
		\item Tìm $m$ để phương trình $(1)$ có $2$ nghiệm phân biệt $x_1,x_2$ thỏa mãn $$\dfrac{2x_1-1}{x_2}+\dfrac{2x_2-1}{x_1}=x_1x_2+\dfrac{55}{x_1x_2}.$$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item 	Phương trình $(1)$ có hai nghiệm phân biệt khi và chỉ khi $\Delta >0$
			\begin{eqnarray*}
			& \Leftrightarrow & (m^2+1)^2-4(m-2)>0\\
			&\Leftrightarrow & m^4+2m^2-4m+9>0\\
			&\Leftrightarrow & m^4+2(m-1)^2+7>0,\, \text{ luôn đúng}\, \forall m\in\mathbb{R}.
			\end{eqnarray*}
			Do đó phương trình có hai nghiệm phân biệt với mọi $m$.
			\item Phương trình $(1)$ có hai nghiệm trái dấu khi và chỉ khi $a\cdot c<0 \Leftrightarrow m<2$.\\
			Vậy $m<2$ là các giá trị cần tìm.
			\item 	Theo định lý Vi-ét ta có $x_1+x_2=-(m^2+1)$ và $x_1x_2=m-2$.\\
			Từ điều kiện, suy ra $m\neq 2$. Theo bài ra ta có
			\allowdisplaybreaks{
				\begin{eqnarray*}
				& &\dfrac{2x_1-1}{x_2}+\dfrac{2x_2-1}{x_1}=x_1x_2+\dfrac{55}{x_1x_2}\\ 
				& \Leftrightarrow & 2(x_1^2+x_2^2)-(x_1+x_2)=(x_1x_2)^2+55\\
				& \Leftrightarrow & 2(x_1+x_2)^2-4x_1x_2-(x_1+x_2)=(x_1x_2)^2+55\\
				& \Leftrightarrow & 2(x_1+x_2)^2-(x_1+x_2)=(x_1x_2+2)^2+51\\
				& \Leftrightarrow & 2(m^2+1)^2+(m^2+1)=m^2+51\\
				& \Leftrightarrow & 2m^4+4m^2-48=0\\
				& \Leftrightarrow &\hoac{& m^2=4\\&m^2=-6}\Leftrightarrow\hoac{& m=2&(\text{loại})\\&m=-2&(\text{thỏa mãn}).}
				\end{eqnarray*}
			}
			Vậy $m=-2$ là giá trị cần tìm.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[9D0B2]
	Cho phương trình $(m-2)x^2-2(m-1)x+m=0$  với $m$ là tham số.
	\begin{enumerate}
		\item Tìm $m$ để phương trình đã cho có nghiệm âm.
		\item Tìm $m$ để phương trình có hai nghiệm phân biệt cùng dấu dương.
		\item 	Biết phương trình đã cho có hai nghiệm tương ứng là độ dài hai cạnh góc vuông của một tam giác vuông. Tìm $m$ để độ dài đường cao tương ứng với cạnh huyền của tam giác vuông đó bằng $\dfrac{2}{\sqrt{5}}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item 
			\begin{itemize}
				\item Nếu $m=2$ thì phương trình đã cho trở thành $-2x+2=0 \Leftrightarrow x=1$ (không thỏa mãn yêu cầu).
				\item Nếu $m\neq 2$ thì phương trình đã cho $\Leftrightarrow \hoac{&x=1\\&x=\dfrac{m}{m-2}.}$\\
				 Do đó phương trình có nghiệm âm $\Leftrightarrow \dfrac{m}{m-2}<0 \Leftrightarrow 0<m<2$.
			\end{itemize}
		\item Phương trình đã cho có hai nghiệm phân biệt cùng dấu dương khi và chỉ khi 
		$$\heva{&\dfrac{m}{m-2}\neq 1\\& \dfrac{m}{m-2}>0} \Leftrightarrow m>2 \quad \text{hoặc}\quad  m<0.$$
		\item 	Với $m\neq 2$ ta có $$(m-2)x^2-2(m-1)x+m=0 \Leftrightarrow x_1=1,\ x_2=\dfrac{m}{m-2}.$$
		Vì $x_1,\ x_2$ là độ dài hai cạnh góc vuông của một tam giác vuông, nên $x_2>0$, hay $$\dfrac{m}{m-2}>0 \Leftrightarrow \hoac{&m<0 \\ & m>2.} \eqno (1)$$
		Khi đó, đường cao của tam giác vuông ứng với cạnh huyền bằng $\dfrac{2}{\sqrt{5}}$, nên 
		$$\dfrac{1}{x_1^2}+\dfrac{1}{x_2^2}=\dfrac{5}{4} \Leftrightarrow x^2_2=4 \Leftrightarrow x_2=\pm 2 \Leftrightarrow m=4\ \text{(thỏa (1))},\ m=\dfrac{4}{3}\ \text{(không thỏa mãn (1))}.$$
		Vậy $m=4$ là giá trị cần tìm.
		\end{enumerate}
	
	} 
\end{bt}

\begin{bt}%[9D4G6]
	Cho phương trình bậc hai $x^2 - 3x + m = 0$ ($m$ là tham số).
	\begin{enumerate}
		\item Tìm $m$ để phương trình có nghiệm bằng $-2$. Tính nghiệm còn lại ứng với $m$ vừa tìm được.
		\item Gọi $x_1$, $x_2$ là hai nghiệm của phương trình đã cho. Tìm giá trị nhỏ nhất của $$A=x_1^2+x_2^2-2x_1x_2.$$
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Phương trình $x^2 - 3x + m = 0$ có nghiệm $x_1=-2$ $\Rightarrow (-2)^2-3(-2)+m=0 \Leftrightarrow m=-10$.\\
			Do $x_1+x_2=5 \Rightarrow x_2=5$.\\
			Vậy $m=-10$ thì phương trình đã cho có nghiệm $-2$ và nghiệm còn lại là $5$.
			\item Ta có $\Delta = (-3)^2 -4 \cdot m = 9-4m$.\\
			Phương trình có nghiệm khi chỉ khi $\Delta \ge 0 \Leftrightarrow 9-4m \ge 0 \Leftrightarrow m\le \dfrac{9}{4}$.\\
			Áp dụng định lý Vi-ét, ta có $x_1+x_2=3$, $x_1\cdot x_2=m$.\\
			Mà $A=x_1^2+x_2^2-2x_1x_2 = (x_1 + x_2)^2 - 5x_1x_2 = 3^2 - 5m = 9-5m$.\\
			Vì $m\le \dfrac{9}{4}$ nên $-5m \ge -\dfrac{45}{4} \Leftrightarrow 9-5m \ge 9- \dfrac{45}{4} = -\dfrac{9}{4}$.\\
			Vậy giá trị nhỏ nhất của $A$ bằng $-\dfrac{9}{4}$ khi $m=\dfrac{9}{4}$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[9D4K6]
	Cho phương trình $x^2-mx-1=0$ ($m$ là tham số). 
	\begin{enumerate}
		\item Giải phương trình với $m=2\sqrt{2}$.
		\item Tìm tham số $m$ để phương trình có hai nghiệm phân biệt $x_1, x_2$ thỏa mãn $$x_1^4+(m^3+2m)x_2=m^2+4.$$
		\item Tìm tất cả giá trị của $m$ để phương trình có hai nghiệm phân biệt $x_1,x_2$ thỏa $x_1<x_2$ và $\left|x_1\right|-\left|x_2\right|=6$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Với $m=2\sqrt{2}$, phương trình trở thành $$x^2-2\sqrt{2}x-1=0 \Leftrightarrow (x-\sqrt{2})^2=3 \Leftrightarrow \hoac{&x=\sqrt{2}+\sqrt{3}\\&x=\sqrt{2}-\sqrt{3}.}$$
			\item Phương trình có hai nghiệm phân biệt $x_1$, $x_2$ khi và chỉ khi
				$$\Delta > 0 \,\Leftrightarrow\, m^2+4>0, \forall m\in \mathbb{R}.$$
		Do $x_1$ là nghiệm của phương trình nên $x_1^2=mx_1+1$, $$\Rightarrow x_1^4=m^2x_1^2+2mx_1+1 =m^2(mx_1+1)+2mx_1+1 =(m^3+2m)x_1+m^2+1.$$
		Do đó 
		\begin{eqnarray*}
		&&	x_1^4+(m^3+2m)x_2=m^2+4 \\
		&\Leftrightarrow& (m^3+2m)(x_1+x_2)+m^2+1=m^2+4\\
		& \Leftrightarrow& m^4+2m^2-3=0 \\
		&\Leftrightarrow& m =\pm 1 \,\text{(thỏa mãn)}.
		\end{eqnarray*}
	\item Áp dụng hệ thức Vi-ét, ta có  $\left\lbrace \begin{aligned}& x_1+x_2=m \\ & x_1x_2=-1 \end{aligned}\right.$.
	Vì $x_1<x_2$ và $x_1x_2=-1<0$ nên $x_1<0<x_2$.\\
	 Do đó
	\[\left|x_1\right|-\left|x_2\right|=6 \,\Leftrightarrow\, -x_1-x_2=6 \,\Leftrightarrow\, x_1+x_2=-6
	\,\Leftrightarrow\, m=-6.\]
	Vậy $m=-6$ là giá trị cần tìm.
		\end{enumerate}		
	}
\end{bt}

\begin{bt}%[9D4B4]%[9D4K6]
	Cho phương trình $x^2+5x+m=0$\;\;\;$(*)$, ($m$ là tham số).
	\begin{enumerate}
		\item Giải phương trình $(*)$ khi $m=-3$.
		\item Tìm $m$ để phương trình $(*)$ có hai nghiệm $x_1, x_2$ thỏa mãn $9x_1+2x_2=18$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Khi $m=-3$ ta được phương trình $x^2+5x-3=0$. Ta có
			$\Delta=5^2-4(-3)=37>0$
			nên phương trình có hai nghiệm phân biệt 
			$\left[\begin{aligned} & x=\dfrac{-5+\sqrt{37}}{2} \\ & x=\dfrac{-5-\sqrt{37}}{2}.\end{aligned}\right.$
			\item Phương trình có hai nghiệm khi $\Delta=25-4m\geq 0 \Leftrightarrow m\leq\dfrac{25}{4}$.\\
			Khi đó, ta có hệ
			$$\begin{cases} x_1+x_2=-5\\9x_1+2x_2=18\\ x_1\cdot x_2=m \end{cases}
			\Leftrightarrow \begin{cases} x_1=4\\x_2=-9\\ x_1\cdot x_2=m. \end{cases}$$
			Thay $x_1$, $x_2$ vào $x_1\cdot x_2=m$ ta được $4\cdot (-9)=m \Leftrightarrow m=-36$ (thỏa mãn).
		\end{enumerate}	
	}
\end{bt}

\begin{bt}%[9D4K6]
	Cho phương trình: $x^2-2(m-1)x+m^2-2m-3=0$ với $m$ là tham số.
	\begin{enumerate}
		\item Chứng minh phương trình luôn có $2$ nghiệm phân biệt với mọi $m$.
		\item Gọi $x_1,x_2$ là $2$ nghiệm của phương trình. Tìm $m$ để $x_1+1=\sqrt{x_2}$.
	\end{enumerate}
	\loigiai{
		
		\begin{enumerate}
			\item Xét phương trình: $x^2-2(m-1)x+m^2-2m-3=0$ với $m$ là tham số.\\
			Ta có $\Delta'=(m-1)^2-\left(m^2-2m-3\right)=4>0$ nên phương trình luôn có hai nghiệm phân biệt.
			\item 	Khi đó ta tính được $x=m-3$ hoặc $x=m+1$.\\
			Xét điều kiện $x_1+1=\sqrt{x_2}\quad (*)$, ta xét $2$ trường hợp sau.
			\begin{itemize}
				\item \textbf{Trường hợp 1:} $x_1=m-3;x_2=m+1$, thay vào $(*)$ ta được
				$$ m-2=\sqrt{m+1}\Leftrightarrow \left\{\begin{aligned}&m\ge 2 \\ &m^2-5m+3=0\end{aligned}\right. \Leftrightarrow \left\{\begin{aligned}&m\ge 2 \\ &\left[\begin{aligned}&m=\dfrac{5+\sqrt{13}}{2} \\ &m=\dfrac{5-\sqrt{13}}{2}\end{aligned}\right. \end{aligned}\right. \Leftrightarrow m=\dfrac{5+\sqrt{13}}{2}. $$
				\item \textbf{Trường hợp 2:} $x_1=m+1;x_2=m-3$, thay vào $(*)$ ta được
				$$ m+2=\sqrt{m-3}\Leftrightarrow \left\{\begin{aligned}&m\ge -2 \\ &m^2+3m+7=0\end{aligned}\right.\quad\text{(vô nghiệm)}. $$
			\end{itemize}
			Vậy $m=\dfrac{5+\sqrt{13}}{2}$ là giá trị cần tìm.
		\end{enumerate}	
	}
\end{bt}

\begin{bt}%[9D4G6]
	Cho phương trình $x^2+2(m-1)x+4m-11=0,$ với $m$ là tham số. 
	\begin{enumerate}
		\item Chứng minh phương trình đã cho luôn có nghiệm với mọi giá trị của tham số $m$.

		\item Tìm các giá trị của tham số $m$ để phương trình có hai nghiệm phân biệt $x_1$, $x_2$.
		\item Tìm các giá trị của tham số $m$ sao cho $\left|x_1\right|+\left|x_2\right|=7$.
		\item Tìm các giá trị của tham số $m$ sao cho $\left|x_1-x_2\right|=7$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Phương trình có hai nghiệm $x_1,x_2$ khi và chỉ khi $$\Delta\ge 0 \Leftrightarrow (2m+1)^2-4(m^2+1)\ge 0\Leftrightarrow 4m-3\ge 0 \Leftrightarrow m\ge \dfrac{3}{4}.$$
			\item Áp dụng định lý Vi-ét ta có $\heva{&x_1+x_2=2m+1 \\ &x_1\cdot x_2=m^2+1.}$\\
			Vì $x_1\cdot x_2=m^2+1>0,\ \forall m\ge \dfrac{3}{4}$ nên $\left|x_1\cdot x_2\right|=x_1\cdot x_2=m^2+1$.\\
			Khi đó ta có
			\allowdisplaybreaks{
				\begin{eqnarray*}
					\left|x_1\right|+\left|x_2\right|=7 &\Leftrightarrow& \left(\left|x_1\right|+\left|x_2\right|\right)^2=49 \\
					&\Leftrightarrow& x_1^2+2\left|x_1\cdot x_2\right|+x_2^2=49 \\
					&\Leftrightarrow& x_1^2+2x_1\cdot x_2+x_2^2=49 \\			
					&\Leftrightarrow& \left(x_1+x_2\right)^2=49.
				\end{eqnarray*}
			}
			Mà $x_1+x_2=2m+1$ nên ta có $$(2m+1)^2=49 \Leftrightarrow \hoac{&2m+1=7 \\ &2m+1=-7} \Leftrightarrow \hoac{&m=3 &(\text{thoả mãn}) \\ &m=-4 &(\text{loại}).}$$
			Vậy $m=3$ thoả mãn đề bài.
			\item Ta có
			\allowdisplaybreaks{
				\begin{eqnarray*}
					|x_1-x_2|=7	&\Leftrightarrow& (x_1-x_2)^2=49\\
					&\Leftrightarrow& (x_1+x_2)^2-4x_1x_2=49\\
					&\Leftrightarrow& (2m+1)^2-4(m^2+1)=49\\
					&\Leftrightarrow& 4m-52=0 \\
					&\Leftrightarrow& m=13 \qquad (\text{thoả mãn}).
				\end{eqnarray*}
			}
			Vậy $m=13$ thoả mãn đề bài.
		\end{enumerate}
	}
\end{bt}
