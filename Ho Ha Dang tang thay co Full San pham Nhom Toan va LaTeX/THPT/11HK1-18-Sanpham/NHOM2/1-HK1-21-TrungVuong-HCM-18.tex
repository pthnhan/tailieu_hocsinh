\begin{bt}%[HK1 THPT TrungVuong, TP. HCM][Lê Xuân Hòa - DA 11HK1-18]%[1D1B3-2]
Giải  phương trình  $2\cos \left( 3x -\dfrac{\pi}{6}\right) +1 = 0.\quad (a)$
\loigiai{
Ta có 
\begin{eqnarray*}
& &2\cos \left( 3x -\dfrac{\pi}{6}\right) +1 = 0\\
&\Leftrightarrow & \hoac{&3x-\dfrac{\pi}{6}=\dfrac{2\pi}{3}+k2\pi \\&3x-\dfrac{\pi}{6}=-\dfrac{2\pi}{3}}\\
&\Leftrightarrow & \hoac{&x=\dfrac{5\pi}{18}+k\dfrac{2\pi}{3}\\&x=-\dfrac{\pi}{18}+k\dfrac{2\pi}{3},  k\in\mathbb{Z}}.
\end{eqnarray*}
Vậy phương trình $(a)$ có các nghiệm $\hoac{&x=\dfrac{5\pi}{18}+k\dfrac{2\pi}{3}\\&x=-\dfrac{\pi}{18}+k\dfrac{2\pi}{3}},  k\in\mathbb{Z}.$
}
\end{bt}

\begin{bt}%[HK1 THPT TrungVuong, TP. HCM][Lê Xuân Hòa - DA 11HK1-18]%[1D1B3-2]
 Giải phương trình $\sqrt{3}\sin x -\cos x =\sqrt{2}.\quad (b)$\\
 \loigiai{
Chia hai vế của phương trình $(b)$ cho $2,$ ta có
\begin{eqnarray*}
& &\sqrt{3}\sin x -\cos x =\sqrt{2}\\
&\Leftrightarrow & \dfrac{\sqrt{3}}{2}\sin x -\dfrac{1}{2}\cos x =\dfrac{\sqrt{2}}{2}\\
&\Leftrightarrow & \sin\left( x -\dfrac{\pi}{6}\right) =\dfrac{\sqrt{2}}{2}\\
&\Leftrightarrow& \hoac{&x=\dfrac{5\pi}{12}+k2\pi\\&x=-\dfrac{\pi}{12}+k2\pi}
\end{eqnarray*}
Vậy phương trình $(b)$ có các nghiệm $ \hoac{&x=\dfrac{5\pi}{12}+k2\pi\\&x=-\dfrac{\pi}{12}+k2\pi}, k\in\mathbb{Z}.$
}
\end{bt}

\begin{bt}%[HK1 THPT TrungVuong, TP. HCM]%[Lê Xuân Hòa - DA 11HK1-18]%: [1D2K1-3]
Từ các chữ số $1,2,3,4,5,6,7,8,9$ có thể lập được bao nhiêu số tự nhiên chẵn gồm ba chữ số đôi một khác nhau và không lớn hơn $789$?
\loigiai{
Gọi $\overline{abc}, a\ne b, b\ne c, c\ne a$ là số có ba chữ số thỏa yêu cầu bài toán.
\begin{itemize}
\item Nếu $a=7,b=8$ khi đó $c$ có $7$ cách lựa chọn. Do đó có $7$ số cần tìm.
\item Nếu $a=7, b< 8$ khi đó $b$ có $6 $ cách chọn, $c$ có $7$ cách chọn. Do đó có $6\cdot 7 =42$ số.
\item Nếu $a< 7$ khi đó $a$ có $6$ cách chọn, $b$ có $8$ cách chọn, $c$ có $7$ cách chọn. Do đó, có $8\cdot 7\cdot 6=336$ số.

Vậy có tất cả $7+42+336=385$ số thỏa yêu cầu bài toán.
\end{itemize}
}
\end{bt}

\begin{bt}%[HK1 THPT TrungVuong, TP. HCM]%[Lê Xuân Hòa - DA 11HK1-18]%: [1D2K5-5]
Chọn ngẫu nhiên ba thẻ từ một hộp chứa $13$ cái thẻ được đánh số từ $1$ đến $13.$ Gọi A là biến cố ``Tổng các số ghi trên ba thẻ chọn được là một số chẵn''.
\begin{itemize}
\item[a)] Tính số phần tử của không gian mẫu.
\item[b)] Tính xác suất của biến cố A.
\end{itemize}
\loigiai{
\begin{itemize}
\item[a)] Số phần tử trong không gian mẫu là số cách chọn ba thẻ trong hộp gồm $13$ thẻ  $$\vert \Omega \vert = \mathrm{C}_{13}^3=286.$$
\item[b)] Tính xác suất của biến cố ``Tổng các số ghi trên ba thẻ chọn được là một số chẵn''.
\begin{itemize}
\item Trường hợp rút được ba thẻ đều là số chẵn có $\mathrm{C}_6^3=20$ cách.
\item Trường hợp rút được hai thẻ là số lẻ và thẻ còn lại là số chẵn có $\mathrm{C}_7^2\cdot {\mathrm{C}_6^1}=126$ cách.
\end{itemize}
Do đó $\vert \Omega_{A}\vert =20+126=146.$
\end{itemize}
Vậy xác suất để biến cố A xảy ra là $\mathrm{ P}(A) = \dfrac{\vert \Omega_{A} \vert}{\vert \Omega \vert} =\dfrac{146}{286}= \dfrac{73}{143}.$
}
\end{bt}


\begin{bt}%[HK1 THPT Trưng Vương, TP. HCM]%[Lê Xuân Hòa - DA 11HK1-18]%[1D2B3-2]
Cho biểu thức $P(x)=\left( x^3-\dfrac{2}{x}\right) ^{12} , x\ne 0.$ Tìm số hạng không chứa $x$ trong khai triển của $P(x).$
\loigiai{
Ta có $P(x) =\sum\limits_{k=0}^{12}\mathrm{C}_{12}^k x^{3k} \left( -\dfrac{2}{x}\right) ^{12-k}= \sum\limits_{k=0}^{12}\mathrm{C}_{12}^k x^{4k-12}2^{12-k}.$\\
 Yêu cầu bài toán $\Leftrightarrow x^{4k-12}=x^0 \Leftrightarrow 4k-12=0 \Leftrightarrow k=3.$\\
 Vậy số hạng không chứa $x$ trong khai triển $P(x)$ là $\mathrm{C}_{12}^3 2^9 =112640.$
}
\end{bt}

\begin{bt}%[HK1 THPT TrungVuong, TP. HCM]%[Lê Xuân Hòa - DA 11HK1-18]%: [1D3K3-3]
Cho cấp số cộng thỏa $\heva{&u_1+u_6=10\\&S_{15}=165}.$ Tìm số hạng $u_{10}$ và tổng $S_{20}.$
\loigiai{
Ta có $\heva{&u_1+u_6=10\\&S_{15}=165}\Leftrightarrow \heva{&2u_1+5d=10\\&2u_1+14d=22}\Leftrightarrow \heva{&u_1=\dfrac{5}{3}\\&d=\dfrac{4}{3}}.$\\[0.3cm]
Do đó $u_{10}=u_1+9d= \dfrac{5}{3}+9\cdot \dfrac{4}{3}=\dfrac{44}{3}$\\[0.3cm]
và $S_{20}=\dfrac{20}{2}\left( 2u_1+19d\right) = 10\left( \dfrac{10}{3}+\dfrac{76}{3}\right) = \dfrac{860}{3}.$
}
\end{bt}
