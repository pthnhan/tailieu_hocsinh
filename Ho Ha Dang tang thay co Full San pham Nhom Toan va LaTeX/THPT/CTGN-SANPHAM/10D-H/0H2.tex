\begin{hq} Hệ quả định lý côsin\\
	$$\begin{aligned}
	& \cos A=\dfrac{b^2+c^2-a^2}{2bc} \\ 
	& \cos B=\dfrac{a^2+c^2-b^2}{2ac} \\ 
	& \cos C=\dfrac{a^2+b^2-c^2}{2ab}. \\
	\end{aligned}$$
\end{hq}

\begin{dl} Định lý sin\\
	$$\dfrac{a}{\sin A}=\dfrac{b}{\sin B}=\dfrac{c}{\sin C}=2R.$$
\end{dl}

\begin{dl} Đường trung tuyến trong tam giác\\
	$$\begin{aligned}
	& m_a^2 =\dfrac{b^2 +c^2 }{2} -\dfrac{a^2 }{4} =\dfrac{2\left(b^2 +c^2 \right)-a^2 }{4}  \\ 
	&m_b^2 =\dfrac{a^2 +c^2 }{2} -\dfrac{b^2 }{4} =\dfrac{2\left(a^2 +c^2 \right)-b^2 }{4}  \\ 
	&m_c^2 =\dfrac{a^2 +b^2 }{2} -\dfrac{c^2 }{4} =\dfrac{2\left(a^2 +b^2 \right)-c^2 }{4} . \end{aligned}$$
\end{dl}

\begin{dl} Công thức tính diện tích  trong tam giác\\
$$\begin{aligned}
& {S=\dfrac{1}{2} ah_a =\dfrac{1}{2} ab\sin C=\dfrac{1}{2} ac\sin B=\dfrac{1}{2} bc\sin A}. \\ 
&{S=\dfrac{abc}{4R} =pr=\sqrt{p\left(p-a\right)\left(p-b\right)\left(p-c\right)} , \,\left(p=\dfrac{1}{2} \left(a+b+c\right)\right).} 
\end{aligned}$$
\end{dl}


\subsection{Các dạng toán và công thức giải nhanh}


\begin{dang}{Tính yếu tố đường cao - đường trung tuyến - bán kính $R$ và $r$}
	\begin{enumerate}
		\item Trong tam giác $ABC$ có  $h_a,h_b,h_c$ lần lượt là các đường cao kẻ từ các đỉnh $A,B,C$ và $r$ là bán kính đường tròn nội tiếp tam giác, ta có $$\dfrac{1}{h_a}+\dfrac{1}{h_b}+\dfrac{1}{h_c}=\dfrac{1}{r}.$$
		\loigiai{Ta có $S=\dfrac{1}{2}a{h_a}\Rightarrow \dfrac{1}{h_a}=\dfrac{a}{2S}$.\\
		Tương tự $\dfrac{1}{h_b}=\dfrac{b}{2S}$ và $\dfrac{1}{h_c}=\dfrac{c}{2S}$.\\
		Suy ra $\dfrac{1}{h_a} + \dfrac{1}{h_b} + \dfrac{1}{h_c} =\dfrac{a}{2S} + \dfrac{b}{2S} + \dfrac{c}{2S} = \dfrac{a+b+c}{2S} = \dfrac{p}{S}=\dfrac{1}{r}$.}
		\item Trong tam giác $ABC$ có $h_a$ là đường cao kẻ từ đỉnh $A$ và $R$ là bán kính đường tròn ngoại tiếp tam giác, ta có
		$$h_a=2R\sin B\sin C.$$
		\loigiai{Ta có $a\cdot h_a=2S\Rightarrow h_a=\dfrac{2S}{a}$.\\
		Mà $S=\dfrac{abc}{4R}\Rightarrow h_a=\dfrac{2bc}{4R}=\dfrac{bc}{2R}\quad (1)$.\\
		Thay $b=2R\sin B$; $c=2R\sin C$ vào $(1)$ ta được\\
		$h_a=\dfrac{2R\sin B\cdot 2R\sin C}{2R}\Rightarrow h_a=2R\sin B \sin C$.}
		
		\item Cho tam giác $ABC$ có các cạnh $a,b,c$. Đường cao $h_a,h_b,h_c$ được tính bởi công thức sau:
		$$h_a=\dfrac{2\sqrt{p(p-a)(p-b)(p-c)}}{a}$$
		$$h_b=\dfrac{2\sqrt{p(p-a)(p-b)(p-c)}}{b}$$
		$$h_c=\dfrac{2\sqrt{p(p-a)(p-b)(p-c)}}{c}.$$
		\loigiai{Ta có $S=\dfrac{1}{2}ah_a=\dfrac{1}{2}bh_b=\dfrac{1}{2}ch_c=\sqrt{p\left( p-a \right)\left( p-b \right)\left( p-c \right)}$.\\
		$\Rightarrow h_a=\dfrac{2\sqrt{p\left( p-a \right)\left( p-b \right)\left( p-c \right)}}{a}$\\
		Tương tự\\
		$ h_b=\dfrac{2\sqrt{p\left( p-a \right)\left( p-b \right)\left( p-c \right)}}{b}$\\
		$h_c=\dfrac{2\sqrt{p\left( p-a \right)\left( p-b \right)\left( p-c \right)}}{c}$.}
		\item Trong tam giác $ABC$ có $m_a,m_b,m_c$ lần lượt là đường trung tuyến ứng với các cạnh $a,b,c$. Tam giác $ABC$ vuông tại A khi và chỉ khi $$5m_a^2=m_b^2+m_c^2.$$
		\loigiai{Tam giác vuông tại $A$  khi và chỉ khi 
			$b^2+c^2=a^2$.\\ 
			$5\left( \dfrac{b^2+c^2}{2}-\dfrac{ a^2}{4} \right)=\dfrac{a^2+c^2}{2}-\dfrac{ b^2}{4}+\dfrac{a^2+b^2}{2}-\dfrac{c^2}{4}$.\\
			Vậy $5 m_a^2= m_b^2+m_c^2$.}
	\end{enumerate}
\end{dang}

\begin{dang}{Tính diện tích tam giác}
	\begin{enumerate}
		\item Với $R$ là bán kính đường tròn ngoại tiếp tam giác $ABC$. Ta có
	$$S_{\triangle ABC}=2R^2\sin A\sin B\sin C.$$ 
		\loigiai{Ta có $S=\dfrac{1}{2} ab\sin C=\dfrac{1}{2} ac\sin B=\dfrac{1}{2} bc\sin A$.\\
		Suy ra $S^3=\dfrac{1}{8}\cdot a^2b^2c^2\sin A \sin B \sin C=\dfrac{1}{8}S^2R^2\sin A \sin B \sin C$.\\
		Vậy $S_{\triangle ABC}=2R^2\sin A\sin B\sin C$.}
		\item Cho tam giác $ABC$ có hai đường trung tuyến $BM,CN$ vuông góc với nhau. Ta có
		$$S_{\triangle ABC}={a^2}\tan A.$$
		\loigiai{Hai đường trung tuyến $BM$, $CN$ vuông góc với nhau thì $\left(\dfrac{2}{3}m_b\right)^2+\left(\dfrac{2}{3}m_c\right)^2=a^2$\\
		$\Leftrightarrow \dfrac{4}{9}\left(\dfrac{a^2+b^2}{2}-\dfrac{c^2}{4}\right)+\dfrac{4}{9}\left(\dfrac{a^2+c^2}{2}-\dfrac{b^2}{4}\right)=a^2$\\
		$\Leftrightarrow 5a^2=b^2+c^2$.\\
		Mặt khác ta có $a^2=b^2-2bc\cos A \Rightarrow bc=\dfrac{2a^2}{\cos A}$.\\
		Vậy $S=\dfrac{1}{2}\sin A=a^2\tan A$.
	}
	\end{enumerate}
\end{dang}
