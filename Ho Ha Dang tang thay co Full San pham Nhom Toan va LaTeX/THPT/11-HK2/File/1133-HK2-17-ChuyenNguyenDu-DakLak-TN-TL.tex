	Cho hàm số $y=-x^3+3x-2$ có đồ thị $(C)$. Viết phương trình tiếp tuyến của $(C)$ tại điểm $M$, biết $M$ là giao điểm của $(C)$ với đường thẳng có phương trình $y=-x-2$ và $x_M>0$.

	\loigiai{

\begin{ex}%[HK2 (2016-2017), THPT Chuyên Nguyễn Du, Đắk Lắk]%[1D4K2-4]%[Nguyễn Xuân Quý]
	Cho số thực $a$ thỏa mãn $\lim\limits_{x\to -\infty}\left(\sqrt{x^2+5ax-1}+x\right)=5$. Số thực $a$ thuộc khoảng nào sau đây?
	\choice
	{$(3;10)$}
	{$(-10;-5)$}
	{\True $(-3;-1)$}
	{$(1;3)$}
	\loigiai{
		- Nhân liên hợp, sau đó chia cả tử và mẫu cho $x$ ta được
		\begin{align*}
		5&=\lim\limits_{x\to -\infty}\left(\sqrt{x^2+5ax-1}+x\right)=\lim\limits_{x\to -\infty}\dfrac{x^2+5ax-1-x^2}{\sqrt{x^2+5ax-1}-x}\\
		&=\lim\limits_{x\to -\infty}\dfrac{5a-\dfrac{1}{x}}{-\sqrt{1+\dfrac{5a}{x}-\dfrac{1}{x^2}}-1}=-\dfrac{5a}{2}.
		\end{align*}
		- Suy ra $a=-2$.
	}
\end{ex}

	Tính $L=\lim\left(\dfrac{1}{1\sqrt{2}+2\sqrt{1}}+\dfrac{1}{2\sqrt{3}+3\sqrt{2}}+\cdots+\dfrac{1}{n\sqrt{n+1}+(n+1)\sqrt{n}}\right)$.
	\choice
	{$\dfrac{1}{\sqrt{2}}$}
	{$\dfrac{1}{8}$}
	{$\dfrac{1}{2}$}
	{\True $1$}
	\loigiai{
		- Ta có $$\dfrac{1}{k\sqrt{k+1}+(k+1)\sqrt{k}}=\dfrac{1}{\sqrt{k(k+1)}\left(\sqrt{k+1}+\sqrt{k}\right)}=\dfrac{\sqrt{k+1}-\sqrt{k}}{\sqrt{k(k+1)}\left(k+1-k\right)}=\dfrac{1}{\sqrt{k}}-\dfrac{1}{\sqrt{k+1}}.$$
