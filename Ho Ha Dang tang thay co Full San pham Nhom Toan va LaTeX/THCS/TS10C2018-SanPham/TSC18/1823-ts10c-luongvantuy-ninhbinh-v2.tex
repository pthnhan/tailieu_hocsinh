\begin{ex}%[TS 10C, Chuyên Lương Văn Tụy, Ninh Bình, 2018-2019]%[Đoàn Ngọc Lê, EX-TS10C-18]%[9D1K8]
Cho biểu thức $P=\dfrac{\sqrt{x}-3}{\sqrt{x}-2}-\dfrac{\sqrt{x}+3}{\sqrt{x}+2}$ với $x>0;x\ne 4$.
\vspace*{14pt}
\begin{listEX}
\item Rút gọn biểu thức $P$.
\item Tìm giá trị lớn nhất của biểu thức $A=\dfrac{x\sqrt{x}}{P}$.
\end{listEX}
\loigiai{
\begin{listEX}
\item Với $x>0$, $x\ne 4$ ta có
\begin{align*}
P&=\dfrac{\sqrt{x}-3}{\sqrt{x}-2}-\dfrac{\sqrt{x}+3}{\sqrt{x}+2}=\dfrac{\left(x-\sqrt{x}-6\right)-\left( x+\sqrt{x}-6 \right)}{x-4} =-\dfrac{2\sqrt{x}}{x-4}.
\end{align*}
\item Ta có
\begin{align*}
A=\dfrac{x\sqrt{x}}{P}=-\dfrac{x(x-4)}{2}=-\dfrac{x^4-4x+4}{2}+2=-\dfrac{(x-2)^2}{2}+2\le 2.
\end{align*}
Dấu bằng xảy ra khi $x=2$. Vậy giá trị lớn nhất của $A$ là $2$.
\end{listEX}
}
\end{ex}

\begin{ex}%[TS 10C, Chuyên Lương Văn Tụy, Ninh Bình, 2018-2019]%[Đoàn Ngọc Lê, EX-TS10C-18]%[9D0K2]
Giải phương trình $\sqrt{3x+1}+\sqrt{x}=3$.
\loigiai{
Điều kiện xác định : $x\ge 0$.\\
Đặt $u=\sqrt{3x+1}$, $v=\sqrt{x}$ ($0\le u,v\le 3$). Phương trình đã cho trở thành
\begin{align*}
\heva{&u+v=3\\ &u^2-3v^2=1} \Leftrightarrow \heva{&v=3-u\\ &u^2-9u+14=0.}
\end{align*}
Từ đó, ta tìm được $u=2$, $v=1$. Suy ra
\begin{align*}
\heva{&\sqrt{3x+1}=2\\ &\sqrt{x}=1}\Leftrightarrow x=1.
\end{align*}
Vậy phương trình có nghiệm duy nhất $x=1$.
}
\end{ex}

\begin{ex}%[TS 10C, Chuyên Lương Văn Tụy, Ninh Bình, 2018-2019]%[Đoàn Ngọc Lê, EX-TS10C-18]%[9D0K2]
Giải hệ phương trình $\heva{&\sqrt{3x-2y}-\sqrt{3y-x}=1\\&\sqrt{3x-2y}+\sqrt{3y-x}+x+4y=9. }$
\loigiai{
Điều kiện xác định: $3x-2y\ge 0$; $3y-x\ge 0$.\\
Đặt $u=\sqrt{3x-2y}\ge 1$, $v=\sqrt{3y-x}\ge 0$, hệ phương trình đã cho trở thành
\begin{align*}
\heva{&u-v=1\\ &u+v +u^2+2v^2=9}\Leftrightarrow \heva{&u=v+1\\ &3v^2+4v-7=0.}
\end{align*}
Từ đó, ta tìm được $u=2$, $v=1$. Suy ra
\begin{align*}
\heva{&\sqrt{3x-2y}=2\\ &\sqrt{3y-x}=1}\Leftrightarrow \heva{&3x-2y=4\\ &3y-x=1}\Leftrightarrow \heva{&x=2\\ &y=1.}
\end{align*}
Vậy hệ phương trình có nghiệm duy nhất $(x;y)=(2;1)$.
}
\end{ex}


\begin{ex}%[TS 10C, Chuyên Lương Văn Tụy, Ninh Bình, 2018-2019]%[Đoàn Ngọc Lê, EX-TS10C-18]%[9D0G5]
Tìm các số tự nhiên $x,y$ thỏa mãn $x^2+3y^2+4xy+4x+10y-12=0$
\loigiai{
Nếu $y=0$ thì $x^2+4x-12=0$, suy ra $x=2$.\\
Giả sử phương trình đã cho có nghiệm tự nhiên $(x;y)$ với $y\ge 1$. Phương trình đã cho tương đương
\begin{align*}
x^2+4(y+1)x+3y^2+10y-12=0.
\end{align*}
Coi $y$ là tham số, $x$ là ẩn, phương trình trên có
\begin{align*}
\Delta'=4(y+1)^2-(3y^2+10y-12)=(y-1)^2+15.
\end{align*}
Do phương trình có nghiệm tự nhiên nên tồn tại số nguyên dương $m$ sao cho $\Delta'=m^2$, suy ra
\begin{align*}
(m-y+1)(m+y-1)=15=1\cdot 3=3\cdot 5.
\end{align*}
Do $y-1\ge 0$ nên $m+y-1>m-y+1$. Suy ra
\begin{align*}
\heva{&m-y+1=1\\ &m+y-1=15}\qquad \text{hoặc}\qquad \heva{&m-y+1=3\\ &m+y-1=5.}
\end{align*}
Giải các hệ trên ta được $y=8$ hoặc $y=2$. Với cả hai trường hợp, ta đều có
\begin{align*}
x^2+4(y+1)x+3y^2+10y-12>0,
\end{align*}
với mọi $x$ tự nhiên. \\ Vậy phương trình đã cho có nghiệm tự nhiên duy nhất là $(2;0)$.
}
\end{ex}



\begin{ex}%[TS 10C, Chuyên Lương Văn Tụy, Ninh Bình, 2018-2019]%[Đoàn Ngọc Lê, EX-TS10C-18]%[9D0G7]
Cho $a,b,c$ là ba số thực dương thỏa mãn điều kiện $ab+bc+ca=3abc$. Chứng minh rằng $$\dfrac{1}{a^2+b+2}+\dfrac{1}{b^2+c+2}+\dfrac{1}{c^2+a+2}\le \dfrac{3}{4}.$$
\loigiai{
Áp dụng bất đẳng thức AM-GM, ta có
\begin{align*}
a^2+1&\ge 2a,\\
b^2+1&\ge 2b,\\
c^2+1&\ge 2c.
\end{align*}
Suy ra
\begin{align*}
\dfrac{1}{a^2+b+2}+\dfrac{1}{b^2+c+2}+\dfrac{1}{c^2+a+2}\le \dfrac{1}{2a+b+1}+\dfrac{1}{2b+c+1}+\dfrac{1}{2c+a+1}.\tag{1}
\end{align*}
Áp dụng bất đẳng thức Cauchy-Schwarz ta có
\begin{align*}
\dfrac{1}{2a+b+1}\le \dfrac{1}{16}\left( \dfrac{1}{a}+\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{1} \right)=\dfrac{1}{16}\left( \dfrac{2}{a}+\dfrac{1}{b}+1 \right).\tag{2}
\end{align*}
Tương tự, ta có
\begin{align*}
\dfrac{1}{2b+c+1}&\le \dfrac{1}{16}\left( \dfrac{2}{b}+\dfrac{1}{c}+1 \right),\tag{3}\\
\dfrac{1}{2c+a+1}&\le \dfrac{1}{16}\left( \dfrac{2}{c}+\dfrac{1}{a}+1 \right).\tag{4}
\end{align*}
Cộng $(2)$, $(3)$, $(4)$, theo vế với chú ý rằng $\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}=3$, ta được
\begin{align*}
\dfrac{1}{2a+b+1}+\dfrac{1}{2b+c+1}+\dfrac{1}{2c+a+1}\le \dfrac{3}{4}.
\end{align*}
Kết hợp với $(1)$, ta được điều phải chứng minh.
}
\end{ex}

\begin{ex}%[TS 10C, Chuyên Lương Văn Tụy, Ninh Bình, 2018-2019]%[Đoàn Ngọc Lê, EX-TS10C-18]%[9H0B1]%[9H0K2]%[9H0G6]
Cho ba điểm cố định $A,B,C$ thẳng hàng ($B$ nằm giữa $A$ và $C$). Gọi $(O)$ là một đường tròn thay đổi luôn đi qua $B$ và $C$ (tâm $O$ không thuộc đường thẳng $BC$). Từ $A$ kẻ các tiếp tuyến $AD$, $AE$ đến đường tròn $(O)$ ($D$, $E$ là các tiếp điểm và $D$, $O$ nằm cùng trên nửa mặt phẳng có bờ là đường thẳng $BC$). Gọi $K$, $H$ lần lượt là trung điểm của $BC$ và $DE$.
\begin{listEX}
\item Chứng minh $AE^2=AB\cdot AC$.
\item Trên $DE$ lấy điểm $M$ sao cho $BM$ song song với $AD$. Chứng minh tứ giác $BMKE$ nội tiếp đường tròn và $MK$ song song với $DC$.
\item Chứng minh rằng khi đường tròn $(O)$ thay đổi thì tâm đường tròn ngoại tiếp tam giác $OHK$ thuộc một đường thẳng cố định.
\end{listEX}
\loigiai{
\begin{center}
\begin{tikzpicture}[>=stealth, scale=1]
\tkzDefPoints{-6/0/A, -2.5/0/B,0/0/K, 0/3/O}
\tkzDrawCircle(O,B)
\tkzTangent[from=A](O,B)\tkzGetPoints{E}{D}
\tkzDefBarycentricPoint(K=2,B=-1)\tkzGetPoint{C}
\tkzDefMidPoint(D,E)\tkzGetPoint{H}
\tkzInterLL(D,E)(B,C)\tkzGetPoint{F}
\tkzDefMidPoint(O,F)\tkzGetPoint{I}
\tkzDefMidPoint(F,K)\tkzGetPoint{X}
\tkzDefLine[parallel=through B](A,D)\tkzGetPoint{a}
\tkzInterLL(B,a)(D,E)\tkzGetPoint{M}
\tkzDrawSegments(O,A A,C A,D A,E D,E B,M F,O O,K K,E B,E C,E K,M O,D O,E)
\tkzMarkRightAngles(O,H,D O,K,C O,D,A O,E,A)
%\tkzDrawLine(X,I)

\tkzLabelPoints[above right](O)
\tkzLabelPoints[left](A)
\tkzLabelPoints[below left](E,B,F)
\tkzLabelPoints[above left](D)
\tkzLabelPoints[below right](K,C)
%\tkzLabelPoints[below left](X)
\draw(M)+(0.3,0.1) node{$M$};
\draw(H)+(-0.4,0.1) node{$H$};

\tkzDrawPoints[fill=black](A,B,C,D,E,F,H,K,O,M)
\end{tikzpicture}
\end{center}
\begin{listEX}
\item Ta có $\triangle ABE\backsim \triangle AEC$ (g.g), suy ra $\dfrac{AB}{AE}=\dfrac{AE}{AC}$. Vậy $AE^2=AB\cdot AC$.
\item Dễ thấy, năm điểm $O$, $A$, $D$, $E$, $K$ nằm trên đường tròn đường kính $OA$. Suy ra $\widehat{DEK}=\widehat{DAK}$, mà $\widehat{DAK}=\widehat{MBK}$ (do $AD\parallel BM$), nên $\widehat{MBK}=\widehat{MEK}$. Vậy tứ giác $BMKE$ là tứ giác nội tiếp.
\item  Gọi $F$ là giao điểm của $DE$ và $AC$. Khi đó tứ giác $OHFK$ nội tiếp đường tròn đường kính $OF$. Suy ra
\begin{align*}
AF\cdot AK=AH\cdot AO=AE^2=AB\cdot AC,
\end{align*}
hay $AF=\dfrac{AB\cdot AC}{AK}$, do đó $F$ là điểm cố định. Vậy tâm đường tròn ngoại tiếp tam giác $OHK$ (cũng chính là tâm đường tròn ngoại tiếp tứ giác $OHFK$) chạy trên đường trung trực của đoạn thẳng $FK$.
\end{listEX}
}
\end{ex}

\begin{ex}%[TS 10C, Chuyên Lương Văn Tụy, Ninh Bình, 2018-2019]%[Đoàn Ngọc Lê, EX-TS10C-18]%[9H0G8]
Trong mặt phẳng cho $2018$ điểm phân biệt $A_1,A_2,\ldots,A_{2018}$ bất kì nằm trên đường tròn $(O;1)$. Chứng minh rằng luôn tìm được điểm $M$ nằm trên đường tròn $(O;1)$ thỏa mãn
\[MA_1+MA_2+\cdots+MA_{2018}>2018.\]
\loigiai{
Giả sử không tồn tại điểm nao trên đường tròn $(O;1)$ thỏa mãn bài toán. Khi đó, lấy điểm $X$ bất kì trên đường tròn $(O;1)$, ta đều có
\begin{align*}
XA_1+XA_2+XA_3+\cdots + XA_{2018}\le 2018.\tag{1}
\end{align*}
Do có tối đa $2018$ đường kình đi qua các điểm $A_1$, $A_2$, \ldots, $A_{2018}$ nên ta có thể chọn một đường kính $MN$ sao cho $M,N\ne A_i$, $i=1,2,\ldots,2018$. Khi đó
\begin{align*}
MA_1+ NA_1&> MN= 2,\\
MA_2+NA_2&> MN=2,\\
&\ldots,\\
MA_{2018}+NA_{2018}&> MN=2.
\end{align*}
Cộng các bất đẳng thức trên theo vế, ta được
\[\left(\displaystyle\sum\limits_{n=1}^{2018}MA_n\right) +\left( \displaystyle\sum\limits_{n=1}^{2018}NA_n \right)> 2\cdot 2018.\]
Tuy nhiên theo $(1)$ thì 
\[\left(\displaystyle\sum\limits_{n=1}^{2018}MA_n\right) +\left( \displaystyle\sum\limits_{n=1}^{2018}NA_n \right)\le 2\cdot 2018.\]
Ta gặp mâu thuẫn. Vậy tồn tại điểm $M$ thỏa mãn bài toán.
}
\end{ex}
