		$$-3x+3=\dfrac{1}{2}x-\dfrac{1}{2} \Leftrightarrow x=1.$$

		&-a+a^3-a^2-\dfrac{1}{3}=0 \Leftrightarrow 4a^3=a^3+3a^2+3a+1\\ \Leftrightarrow& 4a^3=(a+1)^3 \Leftrightarrow \sqrt[3]{4}a=a+1\Leftrightarrow a=\dfrac{1}{\sqrt[3]{4}-1}.

			 \Leftrightarrow & (z+1)(xy+y+x+1)=2016\\
			 \Leftrightarrow &	(x+1)(y+1)(z+1)=16.14.9.
		\end{align*}}Do $ x \geq y \geq z \geq 8$ nên  $ (x,y,z) $ cần tìm là $ (15,13,8) $.

\begin{ex}%[Đề thi vào 10, Chuyên Hưng Yên Vòng 2, 2016]%[9D3G4]%[9D1G4]
	\hfill
	\begin{enumerate}
		\item Giải hệ phương trình $\heva{&x^2y^2-2x+y^2=0\\&2x^2-4x+3=-y^3.} $
		\item Giải phương trình $ (\sqrt{2x+5}-\sqrt{2x+2})(1+\sqrt{4x^2+14x+10})=3. $
	\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có $ \heva{&x^2y^2-2x+y^2=0\\&2x^2-4x+3=-y^3}
	\Leftrightarrow \heva{&y^2=\dfrac{2x}{x^2+1}\\&y^3=-2(x-1)^2-1.}$\\
	Ta thấy $ \dfrac{2x}{x^2+1} \leq 1 \Rightarrow y^2 \leq 1 \Rightarrow -1 \leq y \leq 1. \hfill (1) $\\
	Mặt khác $ -2(x-1)^2-1 \leq -1 \Rightarrow y^3 \leq -1 \Rightarrow y \leq -1. \hfill (2) $\\
	Từ $ (1) $ và $ (2) $ suy ra $ y=-1 \Rightarrow x= 1.$ Thử lại ta thấy thỏa mãn. \\
	Vây nghiệm của hệ là $ (x;y)=(1;-1). $
	\item Điều kiện: $ x \geq -1. $ Đặt $ u=\sqrt{2x+5}$,  $v=\sqrt{2x+2}$ $(u,v\geq 0)$. Khi đó:
	{\allowdisplaybreaks
	\begin{align*}
	&(u-v)(1+uv)=3\,\,  \text{và }\,\, u^2-v^2=3 \\
	\Rightarrow& (u-v)(1+uv)=u^2-v^2 \Leftrightarrow (u-v)(1-u)(1-v)=0\\
	\Leftrightarrow& \hoac{&u-v=0\\&1-u=0\\ &1-v=0.}
\end{align*}}Khi 
	 $ u-v=0 \Rightarrow \sqrt{2x+5}=\sqrt{2x+2} \Rightarrow 2x+5=2x+2 $ (vô nghiệm).\\
	Khi $ 1-u=0 \Rightarrow \sqrt{2x+5}=1 \Rightarrow x=-2 $ (loại).\\
Khi	$ 1-v=0  \Rightarrow \sqrt{2x+2}=1\Rightarrow x=-\dfrac{1}{2}$ (thỏa mãn).\\
Vậy nghiệm của phương trình là $ x=-\dfrac{1}{2} $.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Hưng Yên Vòng 2, 2016]%[9H3K4]%[8H3K1]%[9H3G3]
	Cho hai đường tròn $ (O_1) $ và $ (O_2) $ cắt nhau tại $ A $ và $ B. $ Tiếp tuyến chung gần $ B $ của hai đường tròn lần lượt tiếp xúc với $ (O_1) $ và $ (O_2) $ tại $C$ và $ D. $  Qua $ A $ kẻ đường thẳng song song với $ CD $, lần lượt cắt $ (O_1) $ và $ (O_2) $ tại $ M $ và $ N. $ Các đường thẳng $ CM $ và $ DN $ cắt nhau tại $ E. $ Gọi $ P $ là giao điểm của $ BC $ với $ MN$, gọi $ Q$ là giao điểm vủa $ BD $ với $ MN. $ Chứng minh rằng:
	\begin{enumerate}
		\item Đường thẳng $ AE $ vuông góc với đường thẳng $ CD. $ 
		\item $ \dfrac{BD}{BQ}+\dfrac{BC}{BP}=\dfrac{MN}{PQ}. $
		\item Tam giác $ EPQ $ là tam giác cân.		
	\end{enumerate}

			Từ $ (1) $ và $ (2) $ suy ra $ \triangle EQP $ cân tại $E$.
\end{enumerate}
}
