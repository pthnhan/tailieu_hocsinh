\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2017-2018, Huyện Tiền Hải - Tỉnh Thái Bình]%[9D1B2]%[9D1B6]
Tính giá trị của các biểu thức sau:
\begin{enumerate}
  \item $A=\sqrt{4+\sqrt{10+2\sqrt{5}}}+\sqrt{4-\sqrt{10+2\sqrt{5}}}$
  \item $B=\sqrt{\dfrac{(a+bc)(b+ca)}{c+ab}}+\sqrt{\dfrac{(c+ab)(b+ca)}{a+bc}}+\sqrt{\dfrac{(c+ab)(a+bc)}{b+ca}}$, (với $a, b, c$ là các số thực dương và $a+b+c=1$).
\end{enumerate}
\loigiai{
\begin{enumerate}
  \item Ta có  $A=\sqrt{4+\sqrt{10+2\sqrt{5}}}+\sqrt{4-\sqrt{10+2\sqrt{5}}} >0$ nên
{\allowdisplaybreaks
\begin{align*}
A^2&= 4+\sqrt{10+2\sqrt{5}} + 4-\sqrt{10+2\sqrt{5}} + 2\sqrt{16 - (10+2\sqrt{5})} \\
&= 8+2 \sqrt{6-2\sqrt{5}}= 8 +2 \sqrt{(\sqrt{5} -1)^2}\\
&= 8 +2 {(\sqrt{5} -1)}= 6 + 2\sqrt{5}=(\sqrt{5} +1)^2.
\end{align*}}
Suy ra $A=\sqrt{5} +1 $ \; (do $A>0$).
  \item Vì $a,b,c >0$ và $a+b+c=1$ nên biểu thức $B$ có nghĩa và $0< a, b, c <1.$ Ta có:
{\allowdisplaybreaks
\begin{align*}
B&= \sqrt{\dfrac{(1-b-c+bc)(1-a-c+ca)}{1-a-b+ab}} + \sqrt{\dfrac{(1-a-b+ab)(1-a-c+ca)}{1-b-c+bc}} \\
& \qquad \qquad \qquad \qquad \qquad \qquad \qquad \qquad  \qquad \qquad \qquad + \sqrt{\dfrac{(1-a-b+ab)(1-b-c+bc)}{1-a-c+ca}} \\
&= \sqrt{\dfrac{(1-b)(1-c)(1-a)(1-c)}{(1-a)(1-b)}} + \sqrt{\dfrac{(1-a)(1-b)(1-a)(1-c)}{(1-b)(1-c)}}\\
&\qquad \qquad \qquad \qquad \qquad \qquad \qquad \qquad  \qquad \qquad \qquad + \sqrt{\dfrac{(1-a)(1-b)(1-b)(1-c)}{(1-a)(1-c)}}\\
&= \sqrt{(1-c)^2} + \sqrt{(1-a)^2} + \sqrt{(1-b)^2}\\
&=  \vert 1-c \vert + \vert 1-a \vert + \vert 1-b \vert \\
&= (1-c) +(1-a) +(1-b) \qquad \mbox{ (do $ 0< a,b,c <1 $).}\\
&= 2 \qquad \mbox{ (do $a+b+c = 1$).}
\end{align*}}
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2017-2018, Huyện Tiền Hải - Tỉnh Thái Bình]%[9D3K2]%[8D1G9]
\begin{enumerate}
  \item Tìm các số $a, b$ sao cho đa thức $f(x)=x^4+ax^3+bx-1$ chia hết cho đa thức $x^3-3x+2$.
  \item Chứng minh rằng $B=4x(x+y)(x+y+z)(x+z)+y^2z^2$ là một số chính phương với $x, y, z$ là các số nguyên.
\end{enumerate}
\loigiai{
\begin{enumerate}
  \item Ta có $x^2-3x+2 = (x-1) (x-2).$ Theo bài ra ta có $f(x) \vdots (x-1)(x-2). $\\
  $\Rightarrow$
$f(x) \vdots (x-1)$ và
$f(x) \vdots (x-2).$\\
Do $f(x) \vdots (x-1) \Rightarrow f(1) =0 \Rightarrow a+b=0 \Rightarrow b=-a \qquad (1)$\\
Do $f(x) \vdots (x-2) \Rightarrow f(2) =0 \Rightarrow 8a+2b=-15  \qquad \qquad \; \; (2)$\\
Từ (1) và (2) $\Rightarrow 8a + 2(-a) = -15 \Rightarrow a=-\dfrac{5}{2} \Rightarrow b = \dfrac{5}{2}.$\\
Thử lại: $\left( x^4 -\dfrac{5}{2}x^3 +\dfrac{5}{2}x -1\right) : (x^2 -3x+2)= x^2 +\dfrac{1}{2}x - \dfrac{1}{2}.$\\
Vậy $a=-\dfrac{5}{2}, \; b = \dfrac{5}{2}.$
  \item Ta có
\begin{align*}
B&= 4x(x+y)(x+y+z)(x+z)+y^2z^2\\
&= 4 (x^2 +xy +xz) (x^2 + xy + xz +yz) + y^2z^2\\
&= 4(x^2 +xy +xz)^2 + 4 (x^2 +xy +xz)yz + y^2z^2\\
&= (2x^2 + 2xy +2xz +yz)^2.
\end{align*}
Vì $x, y,z$ là các số nguyên nên $2x^2 + 2xy +2xz +yz$ là số nguyên. Do đó $B$ là số chính phương.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi chọn học sinh giỏi Toán 9 năm học 2017-2018, Huyện Tiền Hải - Tỉnh Thái Bình]%[8D3B2]%[9D1K6]%[9D3K3]
\begin{enumerate}
  \item Tìm $m$ để phương trình $\dfrac{2m-1}{x-2}=m-3$ vô nghiệm.
  \item Giải phương trình $4\sqrt{x+1}=x^2-5x+14$.
  \item Tìm nghiệm nguyên dương của phương trình $\dfrac{xy}{z}+\dfrac{yz}{x}+\dfrac{zx}{y}=3$.
\end{enumerate}
\loigiai{
\begin{enumerate}
  \item  ĐKXĐ $x \neq 2.$\\
$\dfrac{2m-1}{x-2}=m-3 \Rightarrow 2m-1 = (x-2)(m-3)\Rightarrow (m-3)x= 4m -7 \qquad (3).$\\
+) Xét $m=3,$ phương trình (3) trở thành $0x=5$ (vô lý).\\
 Do đó, với $m=3$ phương trình đã cho vô nghiệm.\\
+) Xét $m\neq 3,$ phương trình (3) có nghiệm $x=\dfrac{4m-7}{m-3}.$\\
 Do đó, để phương trình đã cho vô nghiệm thì $\dfrac{4m-7}{m-3} =2 \Rightarrow m=\dfrac{1}{2}.$\\
Vậy với $m=3,$ $m=\dfrac{1}{2}$ thì phương trình đã cho vô nghiệm.
  \item ĐKXĐ: $x\geq -1.$ Ta có
{\allowdisplaybreaks
\begin{align*}
4\sqrt{x+1}=x^2-5x+14&\Leftrightarrow x^2-5x - 4\sqrt{x+1}+14 =0\\
&\Leftrightarrow  x^2 -6x +9 +x+1  - 4\sqrt{x+1}+4 =0\\
&\Leftrightarrow  (x-3)^2 + (\sqrt{x+1} -2)^2=0\\
&\Leftrightarrow \begin{cases}
x-3=0\\
\sqrt{x+1} -2=0
\end{cases}\Leftrightarrow x=3 \quad \mbox { (thoả mãn)}.
\end{align*}}
  \item Áp dụng bất đẳng thức Cosi cho các số dương ta có
$$3= \dfrac{xy}{z}+\dfrac{yz}{x}+\dfrac{zx}{y} \geq 3 \sqrt[3]{\dfrac{xy}{z}\cdot \dfrac{yz}{x}\cdot \dfrac{zx}{y}}= 3 \sqrt[3]{xyz}\Rightarrow xyz \leq 1 \qquad (4). $$
Do $x,y,z$ là các số nguyên dương nên từ (4) suy ra $x=y=z=1.$\\
Thử lại: Với $x=y=z=1$ ta có $\dfrac{xy}{z}+\dfrac{yz}{x}+\dfrac{zx}{y} =3.$\\
Vậy nghiệm nguyên dương của phương trình là $(x, y, z) = (1,1,1).$
\end{enumerate}
}
\end{ex}

\begin{enumerate}
  \item Đặt $BH = x \mbox{ (đơn vị: cm) } \; (0<x<6).$ Ta có $BC= x+6{,}4.$\\

Cho đa thức $f(x) = x^4 + ax^3 + bx^2 + cx + d$  (với $a, b, c, d$ là các số thực).  Biết $f(1) = 10;  f(2) = 20;  f(3) = 30$. Tính giá trị biểu thức $A =f(8)+f(-4)$ .
\loigiai{
Xét đa thức $g(x)= f(x)-10x$ $\Rightarrow$ bậc của đa thức $g(x)$ bằng 4.\\
Từ giả thiết $\Rightarrow$ $g(1)= g(2)=g(3)=0.$\\
Mà $g(x)$ có bậc bằng 4 nên $g(x)=(x-1)(x-2)(x-3)(x-a)$ (với $a$ là số thực nào đó)\\
$\Rightarrow f(x)= (x-1)(x-2)(x-3)(x-a) + 10x$ \\
$\Rightarrow \begin{cases}
f(8)= 7 \cdot 6\cdot 5 \cdot (8-a) +80\\
f(-4)= (-5)(-6)(-7)(-4-a)-40= 5 \cdot 6\cdot 7 \cdot (4+a) - 40
\end{cases}$\\
$\Rightarrow f(8) + f(-4)= 5 \cdot 6\cdot 7 \cdot (8-a+4+a) +40 = 2560.$
}
