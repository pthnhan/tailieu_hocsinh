\begin{ex}%[Đề tuyển sinh lớp 10, Đắk Lắk 2018]%[Đào Anh Dương, EX-THCS6]%[9D1Y2]
Tìm $x$ biết $2\sqrt{x}=3$.
	\loigiai{
	Điều kiện: $x \ge 0$. Ta có
		$$2\sqrt{x}=3 \Leftrightarrow \sqrt{x} = \dfrac{3}{2} \Leftrightarrow x = \dfrac{9}{4}.$$
		Vậy $x = \dfrac{9}{4}$.
		}
\end{ex}

 Giải phương trình $43x^2-2018x+1975$.
 \loigiai{

	}

\begin{ex}%[Đề tuyển sinh lớp 10, Đắk Lắk 2018]%[Đào Anh Dương, EX-THCS6]%[9D4B1]
Cho hàm số $y=(a+1)x^2$. Tìm $a$ để hàm số nghịch biến khi $x<0$ và đồng biến khi $x>0$.
	\loigiai{
	 Hàm số $y=(a+1)x^2$ nghịch biến khi $x<0$ và đồng biến khi $x>0$ $\Leftrightarrow a+1>0 \Leftrightarrow a>-1$.
	}
\end{ex}

\begin{ex}%[Đề tuyển sinh lớp 10, Đắk Lắk 2018]%[Đào Anh Dương, EX-THCS6]%[9D4B5]%[9D4K6]
	Cho phương trình $x^2-2(m+1)x+m^2+2=0 \quad (1)$, $m$ là tham số.
	\begin{enumerate}
		\item Tìm $m$ để $x=2$ là nghiệm của phương trình $(1)$.
		\item Xác định $m$ để phương trình $(1)$ có hai nghiệm phân biệt $x_1$, $x_2$ thỏa mãn $x_{1}^2 + x_{2}^2 = 10$.
	\end{enumerate}	
	\loigiai{
	\begin{enumerate}
		\item Ta có $x=2$ là nghiệm của phương trình $(1)$ khi
		\begin{eqnarray*}
		& &2^2 - 2(m+1) \cdot 2 + m^2 + 2 = 0\\
		&\Leftrightarrow &m^2 - 4m + 2 = 0\\
		&\Leftrightarrow &\left[\begin{aligned}
			&m=2+\sqrt{2}\\
			&m=2-\sqrt{2}.
		\end{aligned} \right.
		\end{eqnarray*}
		\item Phương trình $(1)$ có hai nghiệm phân biệt $x_1$, $x_2$ khi 
		$$\Delta' > 0 \Leftrightarrow (m+1)^2 - (m^2+2) > 0 \Leftrightarrow 2m-1>0 \Leftrightarrow m>\dfrac{1}{2}.$$
		Theo định lí Vi-ét ta có $\begin{cases}
		x_1 + x_2=2(m+1)\\x_1x_2 = m^2+2.
		\end{cases}$\\
		Khi đó ta có
		\begin{eqnarray*}
		x_{1}^2 + x_{2}^2 = 10 &\Leftrightarrow &(x_1+x_2)^2-2x_1x_2=10\\
		&\Leftrightarrow &\left[2(m+1)\right]^2-2(m^2+2)=10\\
		&\Leftrightarrow &m^2+4m-5=0\\
		&\Leftrightarrow &\left[ \begin{aligned}
			&m-1=0\\&m+5=0
		\end{aligned} \right. \\
		&\Leftrightarrow &\left[ \begin{aligned}
		&m=1\\&m=-5 \, \text{(loại)}.
		\end{aligned} \right. 
		\end{eqnarray*}
		Vậy $m=1$ là giá trị cần tìm.
	\end{enumerate}		
	}
\end{ex}

\begin{ex}%[Đề tuyển sinh lớp 10, Đắk Lắk 2018]%[Đào Anh Dương, EX-THCS6]%[9D2B4]
	Trong mặt phẳng tọa độ $Oxy$ cho các đường thẳng có phương trình $(d_1) \colon y=x+2$, $(d_2) \colon y=-2$, $(d_3) \colon y=(k+1)x + k$. Tìm $k$ để các đường thẳng trên đồng quy.		
	\loigiai{	
	Tọa độ giao điểm của $(d_1)$ và $(d_2)$ là nghiệm của hệ
			$\begin{cases}
			y=x+2\\y=-2
			\end{cases} \Leftrightarrow
			\begin{cases}
			x=-4\\y=-2.
			\end{cases}$\\
			Do đó ba đường thẳng trên đồng quy khi $(d_3)$ đi qua điểm $(-4;-2)$.\\
			Suy ra $-2=-4(k+1)+k \Leftrightarrow 3k=-2 \Leftrightarrow k=-\dfrac{2}{3}$.
	}
\end{ex}

\begin{ex}%[Đề tuyển sinh lớp 10, Đắk Lắk 2018]%[Đào Anh Dương, EX-THCS6]%[9D1K8]
	Rút gọn và tìm giá trị lớn nhất của biểu thức
	$$A=\left( \dfrac{1}{1-\sqrt{x}} + \dfrac{x+2}{x\sqrt{x} - 1} + \dfrac{\sqrt{x}}{x+\sqrt{x} + 1} \right) : \dfrac{\sqrt{x}-1}{3} \quad (x \ge 0, x \ne 1).$$
	\loigiai{
		Ta có
		\begin{align*}
		A&=\left( \dfrac{1}{1-\sqrt{x}} + \dfrac{x+2}{x\sqrt{x} - 1} + \dfrac{\sqrt{x}}{x+\sqrt{x} + 1} \right) : \dfrac{\sqrt{x}-1}{3}\\
		&=\dfrac{-\left(x+\sqrt{x} + 1 \right) + x + 2 + \sqrt{x} (\sqrt{x} -1 )}{(\sqrt{x} -1)(x+\sqrt{x} + 1)} \cdot \dfrac{3}{\sqrt{x}-1}\\
		&=\dfrac{3\left(\sqrt{x}-1\right)^2}{\left(\sqrt{x}-1\right)^2(x+\sqrt{x} + 1)}\\
		&=\dfrac{3}{x+\sqrt{x} + 1}.
		\end{align*}
		Vì $x+\sqrt{x} + 1 \ge 1 ,\forall x \ge 0$ nên $A=\dfrac{3}{x+\sqrt{x} + 1} \le 3$.
		Dấu đẳng thức xảy ra khi $x=0$.\\
		Vậy $\max A = 3$ khi $x=0$.
	}
\end{ex}
	
\begin{ex}%[Đề tuyển sinh lớp 10, Đắk Lắk 2018]%[Đào Anh Dương, EX-THCS6]%[9H3Y7]%[9H3K7]%[7H2K6]%[7H3K8]
	Cho tam giác $ABC$ có ba góc nhọn và $\widehat{A} = 45^\circ$. Gọi $D$, $E$ lần lượt là các hình chiếu vuông góc của $B$, $C$ lên $AC$, $AB$; $H$ là giao điểm của $BD$ và $CE$.
	\begin{enumerate}
		\item Chứng minh tứ giác $BECD$ nội tiếp.
		\item Chứng minh $DE \cdot AB = BC \cdot AD$ và tính tỉ số $\dfrac{ED}{BC}$.
		\item Chứng minh $HE+HD=BE+CD$.
		\item Gọi $I$ là tâm đường tròn ngoại tiếp của tam giác $ABC$. Chứng minh $AI \perp DE$.
	\end{enumerate}
	\loigiai{
	\begin{center}
		\begin{tikzpicture}[thick]
		\tkzDefPoints{0/5/A,-2/-2/B,4/-2/C}
		\tkzCircumCenter(A,B,C)\tkzGetPoint{I}
		\tkzDefPointBy[projection= onto A--B](C)
		\tkzGetPoint{E}
		\tkzDefPointBy[projection= onto A--C](B)
		\tkzGetPoint{D}
		\tkzInterLL(C,E)(B,D)
		\tkzGetPoint{H}
		\tkzDrawCircle[radius](I,A)
		\tkzDrawSegments(A,B B,C A,C B,D C,E A,I D,E)
		\tkzDrawPoints[color=black, size=0.1cm](A,B,C,D,E,H,I)
		\tkzMarkRightAngle(B,E,C) \tkzMarkRightAngle(C,D,B)
		\tkzLabelPoints[above](A) \tkzLabelPoints[left](B,E)
		\tkzLabelPoints[right](C,I,D)\tkzLabelPoints[below](H)
		\tkzMarkAngles[label=$45^\circ$,size=.7cm](B,A,C)
		\end{tikzpicture}
	\end{center}
	\begin{enumerate}
		\item Theo giả thiết $\widehat{BDC} = \widehat{BEC} = 90^\circ$. Khi đó tứ giác $BECD$ có đỉnh $E$ và $D$ cùng nhìn cạnh $BC$ dưới hai góc bằng nhau nên tứ giác $BECD$ nội tiếp.
		\item Tứ giác $BECD$ nội tiếp nên $\widehat{AED} = \widehat{ACB}$ (cùng bù với $\widehat{BED}$).\\
		Xét $\triangle ADE$ và $\triangle ABC$ có $\widehat{AED} = \widehat{ACB}$ và $\widehat{A}$ chung nên $\triangle ADE \backsim \triangle ABC$.\\
		Do đó $\dfrac{AD}{DE} = \dfrac{AB}{BC} \Leftrightarrow DE \cdot AB = BC \cdot AD$.\\
		Từ $\dfrac{AD}{DE} = \dfrac{AB}{BC} \Rightarrow \dfrac{DE}{BC} = \dfrac{AD}{AB}$.\\
		Vì $\triangle ABD$ vuông tại $D$ nên ta có\\
		$\dfrac{DE}{BC} = \dfrac{AD}{AB} = \cos \widehat{BAD} = \cos 45^\circ = \dfrac{\sqrt{2}}{2}$.
		\item $\triangle ABD$ vuông tại $D$ và $\widehat{BAD} = 45^\circ$ nên $\widehat{ABD} = 45^\circ \Rightarrow \widehat{EBH} = 45^\circ$\\
		$\Rightarrow \triangle EBH$ vuông cân tại $E$ $\Rightarrow HE=BE \hfill (1)$.\\
		Chứng minh tương tự $\triangle CDH$ vuông cân tại $D$ $\Rightarrow HD=CD \hfill (2)$.\\
		Từ $(1)$ và $(2)$ suy ra $HE+HD=BE+CD$.
		\item Vì $I$ là tâm đường tròn ngoại tiếp $\triangle ABC$ nên $I$ là giao điểm của ba đường trung trực $\triangle ABC$. \\
		Ta có $I$ thuộc trung trực của $AC$; $E$ thuộc trung trực của $AC$ (vì $\triangle AEC$ vuông cân tại $E$) suy ra $EI \perp AC \Rightarrow EI \perp AD \hfill (3)$.\\
		Chứng minh tương tự $DI \perp AB \Rightarrow DI \perp AE \hfill (4)$.\\
		Từ $(3)$ và $(4)$ suy ra $I$ là trực tâm của $\triangle AED \Rightarrow AI \perp DE$.
	\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề tuyển sinh lớp 10, Đắk Lắk 2018]%[Đào Anh Dương, EX-THCS6]%[9D1G8]
	Cho $n$ là số tự nhiên khác $0$. Tìm giá trị nhỏ nhất của
	$$Q=\sqrt{1+\dfrac{1}{1^2}+\dfrac{1}{2^2}} + \sqrt{1+\dfrac{1}{2^2}+\dfrac{1}{3^2}} + \sqrt{1+\dfrac{1}{3^2}+\dfrac{1}{4^2}} + \cdots + \sqrt{1+\dfrac{1}{n^2}+\dfrac{1}{(n+1)^2}} + \dfrac{101}{n+1}.$$
	\loigiai{
	Ta có \begin{align*}
	\sqrt{1+\dfrac{1}{n^2}+\dfrac{1}{(n+1)^2}} &= \sqrt{1+ \left(\dfrac{1}{n} - \dfrac{1}{n+1} \right)^2 + \dfrac{2}{n(n+1)}} \\
	&= \sqrt{1+ \left[\dfrac{1}{n(n+1)} \right]^2 + \dfrac{2}{n(n+1)}}\\
	&= \sqrt{\left[1 + \dfrac{1}{n(n+1)} \right]^2}\\
	&= 1 + \dfrac{1}{n(n+1)} = 1 + \dfrac{1}{n} - \dfrac{1}{n+1}.
	\end{align*}
	Do đó \begin{align*}
	Q &=\left(1+\dfrac{1}{1}-\dfrac{1}{2}\right) + \left(1+\dfrac{1}{2}-\dfrac{1}{3}\right) +\left(1+\dfrac{1}{3}-\dfrac{1}{4}\right)+ \cdots + \left(1+\dfrac{1}{n}-\dfrac{1}{n+1}\right) + \dfrac{101}{n+1}\\
	&= n+1-\dfrac{1}{n+1}+\dfrac{101}{n+1} = n+1+\dfrac{100}{n+1}.
	\end{align*}
	Áp dụng Bất đẳng thức Cauchy ta có
	$$Q=n+1+\dfrac{100}{n+1} \ge 2 \sqrt{(n+1) \cdot \dfrac{100}{n+1}} = 20.$$
	Dấu đẳng thức xảy ra khi $n+1=\dfrac{100}{n+1} \Leftrightarrow n=9$.\\
	Vậy $\min Q = 20$ khi $n=9$.
	}	
\end{ex}
