	\section{Đường tròn và tam giác}
	\subsection{Kiến thức cơ bản}
	\begin{enumerate}
		\item Đường tròn tâm $O$, bán kính $R$ là hình gồm các điểm cách $O$ một khoảng bằng $R$, kí hiệu $(O; R)$.
		\item Hình tròn là hình gồm các điểm nằm trên đường tròn và các điểm nằm bên trong đường tròn.
		\item Hai điểm $A$ và $B$ thuộc đường tròn tâm $O$ chia đường tròn thành hai cung tròn. Đoạn thẳng nối hai điểm $A$, $B$ là dây cung. Dây đi qua tâm là đường kính.
		\item Tam giác $ABC$ là hình gồm ba đoạn thẳng $AB$, $BC$, $CA$ khi ba điểm $A$, $B$, $C$ không thẳng hàng.
	\end{enumerate}
	\subsection{Ví dụ minh họa}
	\begin{vd}
		Cho $5$ điểm bất kì thuộc đường tròn $(O)$. Hỏi trên hình vẽ có bao nhiêu dây cung, bao nhiêu cung tạo bởi $2$ điểm trong $5$ số đã cho?
		\loigiai{
		\begin{itemize}
			\item Vì $5$ điểm đã cho thuộc một đường tròn nên qua $2$ điểm một ta vẽ được một dây cung.\\
			Chọn $1$ điểm với $4$ điểm còn lại vẽ được $4$ dây cung. Vậy với $5$ điểm vẽ được $4\cdot 5=20$ (dây cung), nhưng mỗi dây cung được tính $2$ lần nên có tất cả $\dfrac{4\cdot 5}{2}=10$ (dây cung).\\
			Bài toán này tương đương với bài toán cho $5$ điểm, cứ qua $2$ điểm vẽ được $1$ đoạn thẳng và ta có được $10$ đoạn thẳng.\\
			Vậy với $n$ điểm trên một đường tròn, ta vẽ được $\dfrac{(n-1)n}{2}$ dây cung tạo bởi $2$ trong số $n$ điểm đã cho.
			\item Mỗi dây cung $AB$ tạo thành $2$ cung tròn lớn và nhỏ (trong trường hợp hai điểm $A$ và $B$ là hai đầu mút của một đường kính, ta có hai cung bằng nhau). Suy ra qua $5$ điểm thuộc một đường tròn, vẽ được $20$ cung tròn và qua $n$ điểm thuộc một đường tròn, vẽ được $(n-1)\cdot n$  cung tròn.
		\end{itemize}
	}
	\end{vd}

		Cho điểm $M$ không thuộc đường thẳng $xy$. Lấy hai điểm $A$, $B$ trên tia $xy$ thì tồn tại một tam giác có đỉnh là điểm $M$ và $2$ đỉnh còn lại là hai điểm $A$, $B$.
		\begin{enumerate}
			\item Nếu có thêm một điểm thứ ba cũng thuộc đường thẳng $xy$ thì vẽ được bao nhiêu tam giác có đỉnh là $m$ và hai đỉnh còn lại là $2$ trong số $3$ ba điểm thuộc đường thẳng $xy$?
			\item Nếu có $100$ điểm trên đường thẳng $xy$ thì vẽ được bao nhiêu tam giác có đỉnh là $M$ và hai đỉnh còn lại là hai điểm trong số $100$ điểm thuộc đường thẳng $xy$?
		\end{enumerate}
	\loigiai{
	Kí hiệu các điểm nằm trên đường thẳng $xy$ là $A_1$, $A_2$, $A_3$,\ldots
	\begin{enumerate}
		\item Với $2$ điểm $A_1$, $A_2$ thuộc đường thẳng $xy$, ta có $1$ tam giác $\triangle MA_1A_2$.\\
		Thêm \textit{điểm thứ ba} $A_3\in xy$, ta có $3$ tam giác $\triangle MA_1A_2$, $\triangle MA_2A_3$, $\triangle MA_3A_1$. So với trường hợp $2$ điểm trên $xy$, ta thêm được hai tam giác. Như vậy có $1+2=3$ (tam giác).\\
		Tương tự, thêm \textit{điểm thứ tư} $A_4\in xy$, ta có thêm $3$ tam giác $\triangle MA_1A_4$, $\triangle MA_2A_4$, $\triangle MA_3A_4$. So với trường hợp $3$ điểm trên $xy$, ta thêm được $3$ tam giác. Như vậy có $1+2+3=6$ (tam giác).\\
		Tiếp tục như vậy với $n$ điểm trên đường thẳng $xy$, ta vẽ được
		\[ 1+2+3+\cdots +(n-1)=\dfrac{(n-1)n}{2}\text{ (tam giác).} \]
		\item Khi $n=100$, ta có $\dfrac{(100-1)\cdot 100}{2}=4950$ (tam giác).
	\end{enumerate}


}

\subsection{GHI NHỚ}

\subsubsection{ĐIỂM NẰM GIỮA HAI ĐIỂM (THỨ TỰ GIỮA BA ĐIỂM)}
\begin{enumerate}
	\item \textbf{Dấu hiệu 1} (Đ1). Với $3$ điểm thẳng hàng, có một và chỉ một điểm nằm giữa hai điểm còn lại.
	\begin{itemize}
		\item \textbf{Dấu hiệu 1a}. Trong $3$ điểm thẳng hàng $A$, $B$, $C$, nếu điểm $B$ không nằm giữa hai điểm $A$ và $C$, điểm $C$ không nằm giữa hai điểm $A$ và $B$, thì điểm $A$ phải nằm giữa hai điểm $B$ và $C$.
		\item \textbf{Dấu hiệu 1b}. Với $3$ điểm $A$, $B$, $C$ thẳng hàng mà $A$, $B$ nằm cùng phía đối với điểm $C$ và $A$, $C$ nằm cùng phía đối với điểm $B$ thì điểm $A$ nằm giữa hai điểm $B$ và $C$.
	\end{itemize}
	\item \textbf{Dấu hiệu 2} (Đ2). Nếu điểm $A$ nằm giữa hai điểm $B$, $C$ mà điểm $M$ nằm giữa hai điểm $A$, $B$ và điểm $N$ nằm giữa hai điểm $A$, $C$ thì điểm $A$  nằm giữa hai điểm $M$ và $N$.
	\item \textbf{Dấu hiệu 3} (Đ3). Nếu hai tia $AB$ và $AC$ đối nhau thì điểm $A$  nằm giữa hai điểm $B$ và $C$.
	\item \textbf{Dấu hiệu 4} (Đ4). Nếu có đẳng thức $AM+MB=AB$ thì điểm $M$  nằm giữa hai điểm $A$ và $B$.
	\item \textbf{Dấu hiệu 5} (Đ5). Nếu các điểm $A$ và $B$ thuộc tia $Ox$ và $OA<OB$ thì $A$ nằm giữa hai điểm $O$ và $B$.
\end{enumerate}

\subsubsection{TIA NẰM GIỮA HAI TIA}

\begin{enumerate}
	\item \textbf{Dấu hiệu 1} (T1). Nếu có điểm $M \ne O$ thuộc tia $Ox$, điểm $N \ne O$ thuộc tia $Oy$, mà tia $Oz$ cắt đoạn thẳng $MN$ tại điểm $I$ nằm giữa hai điểm $M$ và $N$ thì tia $Oz$ nằm giữa hai tia $Ox$ và $Oy$.
	\item \textbf{Dấu hiệu 2} (T2). Nếu hai tia $Ox$, $Oy$ đối nhau thì mọi tia $Oz$ khác $Ox$, $Oy$ đều nằm giữa  hai tia $Ox$ và $Oy$.
	\item \textbf{Dấu hiệu 3} (T3). Cho hai tia $Oy$, $Oz$ cùng thuộc nửa mặt phẳng bờ chứa tia $Ox$. Biết tia $Oy$ nằm giữa hai tia $Ox$ và $Oz$. Nếu tia $Om$ nằm giữa hai tia $Oy$ và $Oz$ , hoặc tia $Om$ nằm giữa hai tia $Oy$ và $Ox$, thì tia $Om$ nằm giữa hai tia $Ox$ và $Oz$.  
	\item \textbf{Dấu hiệu 4} (T4). Nếu có đẳng thức $\widehat{xOy}+\widehat{yOz}=\widehat{xOz} $ thì tia $Oy$ nằm giữa hai tia $Ox$ và $Oz$. 
	\item \textbf{Dấu hiệu 5} (T5). Nếu các tia $Oy$, $Oz$ cùng nằm trong một nửa mặt phẳng bờ chứa tia $Ox$ mà  $\widehat{xOy}<\widehat{yOz}$ thì tia $Oy$ nằm giữa hai tia $Ox$ và $Oz$. 
	\item \textbf{Dấu hiệu 6} (T6). Nếu các tia $Oy$, $Oz$ thuộc hai nửa mặt phẳng đối nhau có bờ chứa tia $Ox$  sao cho  $\widehat{xOy}+\widehat{xOz} \le 180^\circ $ thì tia $Ox$ nằm giữa hai tia $Oy$ và $Oz$.
\end{enumerate}

\subsubsection{TRUNG ĐIỂM CỦA ĐOẠN THẲNG}
\begin{enumerate}
	\item \textbf{Dấu hiệu 1} (TĐ1). Nếu trên đoạn thẳng $AB$ có một điểm $M$ sao cho $AM=\dfrac{1}{2}AB$ (hoặc $MB=\dfrac{1}{2}AB$) thì điểm $M$ là trung điểm của đoạn thẳng $AB$.
	\item \textbf{Dấu hiệu 2} (TĐ2). Cho bốn điểm $A$, $B$, $C$ và $D$ thuộc đường thẳng $xy$ theo thứ tự đó. Nếu có $AB=CD$ thì $AD$ và $BC$ có chung một trung điểm (hoặc $AC=BD$ thì $AD$ và $BC$ có chung một trung điểm).
\end{enumerate}

\subsubsection{TIA PHÂN GIÁC CỦA MỘT GÓC}
\begin{enumerate}
	\item \textbf{Dấu hiệu 1} (PG1). Trên nửa mặt phẳng bờ chứa tia $OA$ có các tia $OB$, $OM$ mà $\widehat{AMO}=\dfrac{1}{2}\widehat{AOB}$ thì $OM$ là tia phân giác của góc $AOB$.
	\item \textbf{Dấu hiệu 2} (PG2). Có bốn tia chung gốc $OA$, $OB$, $OC$, $OD$ xếp theo thứ tự đó mà $\widehat{AOD}<180^\circ $. Nếu hai góc $\widehat{AOB} $ và $\widehat{COD} $ bằng nhau (hoặc $\widehat{AOC} =\widehat{BOD} $) thì hai góc $AOD$ và $BOC$ có chung tia phân giác.
\end{enumerate}
