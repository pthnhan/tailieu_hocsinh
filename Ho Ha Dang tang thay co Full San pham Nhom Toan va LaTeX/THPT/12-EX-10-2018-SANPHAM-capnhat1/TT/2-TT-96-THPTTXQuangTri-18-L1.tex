		Cho hàm số $f(x)$ có đạo hàm liên tục trên $\mathbb{R}$, thoả mãn $\cot x\cdot f'(x)+f(x)=2cos^3x$ với mọi $x\neq k\pi$ và $f\left(\dfrac{\pi}{4}\right)=\dfrac{9\sqrt{2}}{4}$. Mệnh đề nào dưới đây đúng?
		\choice
		{\True $f\left(\dfrac{\pi}{3}\right)\in (1; 4)$}
		{$f\left(\dfrac{\pi}{3}\right)\in (6; 10)$}
		{$f\left(\dfrac{\pi}{3}\right)\in (3; 5)$}
		{$f\left(\dfrac{\pi}{3}\right)\in (4; 8)$}
		\loigiai{Từ giả thiết ta có 
			\begin{eqnarray*}
				& &\cos x\cdot f'(x)+\sin x\cdot f(x)=2\cos^3x\cdot\sin x\\
				&\Rightarrow &\dfrac{\cos x\cdot f'(x)+\sin x\cdot f(x)}{\cos^2x}=2\sin x\cdot \cos x\\
				&\Rightarrow &\left(\dfrac{f(x)}{\cos x}\right)'=\sin 2x\\
				&\Rightarrow & \dfrac{f(x)}{\cos x}=\displaystyle\int\limits \sin 2x \mathrm{\,d}x\\
				&\Rightarrow &f(x)=-\dfrac{1}{2}\cos 2x\cdot\cos x+C\cdot \cos x.
			\end{eqnarray*}
			Do $f\left(\dfrac{\pi}{4}\right)=\dfrac{9\sqrt{2}}{4}$ nên ta có $C=\dfrac{9}{2}$ và $f(x)=-\dfrac{1}{2}\cos 2x\cdot\cos x+\dfrac{9}{2}\cdot \cos x$.\\
			Từ đó ta có $f\left(\dfrac{\pi}{3}\right)=\dfrac{19}{8}=2,375\in (1;4)$.
		}
