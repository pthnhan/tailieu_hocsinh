\begin{ex}%[Đề thi HSG 9, tỉnh Ninh Bình, năm 2018]%[Nguyễn Cao Cường, Dự án EX-9G-18]%[9D0G1]
Cho biểu thức $M=\dfrac{a+1}{\sqrt{2}}+\dfrac{a\sqrt{a}-1}{a-\sqrt{a}}+\dfrac{a^2-a\sqrt{a}+\sqrt{a}-1}{\sqrt{a}-a\sqrt{a}}$, với $a>0$, $a\neq 1$.
\begin{enumerate}
\item  Rút gọn $M$.
\item  Chứng minh rằng $M>4$.
\item  Tìm các giá trị của $a$ để biểu thức $N=\dfrac{9}{M}$ có giá trị nguyên.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item [a.] Ta có $$\begin{aligned}
M&=\dfrac{a+1}{\sqrt{a}}+\dfrac{(\sqrt{a}-a)(a+\sqrt{a}+1)}{\sqrt{a}(\sqrt{a}-1)}+\dfrac{(\sqrt{a}-1)(\sqrt{a}+1)(a-\sqrt{a}+1)}{-\sqrt{a}(\sqrt{a}-1)(\sqrt{a}+1)} \\
&= \dfrac{a+1}{\sqrt{a}}+\dfrac{a+\sqrt{a}+1}{\sqrt{a}}-\dfrac{a-\sqrt{a}+1}{\sqrt{a}}\\
&=\dfrac{a+2\sqrt{a}+1}{\sqrt{a}}.\end{aligned}$$
\item  Ta có
$M-4=\dfrac{a+2\sqrt{a}+1}{\sqrt{a}}-4=\dfrac{(\sqrt{a}-1)^2}{\sqrt{a}}>0 \Rightarrow M>4$ (vì $a>0$, $a\neq 1$).
\item  Vì $a>0$, $a\neq 1$. Ta có $M>4 \Rightarrow N=\dfrac{9}{M}<\dfrac{9}{4} \Rightarrow 0<N\leq 2$. Do đó $N=1$ hoặc $N=2$.
\begin{enumerate}
\item [+] Xét $N=1 \Rightarrow M=9 \Rightarrow \dfrac{a+2\sqrt{a}+1}{\sqrt{a}}=9 \Leftrightarrow a-7\sqrt{a}+1=0 \Rightarrow a=\dfrac{47\pm 21\sqrt{5}}{2}$ (thỏa điều kiện).
\item [+] Xét $N=2 \Rightarrow M=\dfrac{9}{2} \Rightarrow \dfrac{a+2\sqrt{a}+1}{\sqrt{a}}=\dfrac{9}{2} \Leftrightarrow 2a-5\sqrt{a}+2=0 \Rightarrow a=\dfrac{1}{4}\text{ hoặc }a=4$ (thỏa điều kiện).
\end{enumerate}
Vậy các giá trị của $a$ thỏa mãn đề bài là $a\in \left\lbrace \dfrac{47\pm 21\sqrt{5}}{2}; \dfrac{1}{4}; 4\right\rbrace$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Ninh Bình, năm 2018]%[Nguyễn Cao Cường, Dự án EX-9G-18]%[9D0K2]
 Giải phương trình $\sqrt{-x^2+4x-3}+\sqrt{-2x^2+8x+1}=x^3-4x^2+4x+4$.
\loigiai{
 ĐKXĐ: $1\leq x\leq 3$. Ta có phương trình tương đương
$$\begin{aligned}
&x^3-4x^2+4x+1-\sqrt{-x^2+4x-3}+3-\sqrt{-2x^2+8x+1}=0\\
\Leftrightarrow &\;x(x-2)^2+\dfrac{(x-2)^2}{1+\sqrt{-x^2+4x-3}}+\dfrac{2(x-2)^2}{3+\sqrt{-2x^2+8x+1}}=0\\
\Leftrightarrow &\;(x-2)^2\left[ x+\dfrac{1}{1+\sqrt{-x^2+4x-3}}+\dfrac{2}{3+\sqrt{-2x^2+8x+1}}\right]=0.
\end{aligned}$$
Vì $$1\leq x\leq 2 \Rightarrow  x+\dfrac{1}{1+\sqrt{-x^2+4x-3}}+\dfrac{2}{3+\sqrt{-2x^2+8x+1}}>0.$$
 Do đó, $$(x-2)^2=0 \Leftrightarrow x=2.$$ 
Vậy $x=2$ là nghiệm của phương trình đã cho.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Ninh Bình, năm 2018]%[Nguyễn Cao Cường, Dự án EX-9G-18]%[9D0K2]
Giải hệ phương trình $\heva{&x^2+y=4x\\&x^4+y^2=2x^2y+y-4}$.
\loigiai{
 Từ phương trình $x^2+y=4x \Rightarrow y=4x-x^2$. Từ phương trình thứ hai, ta có 
$$\begin{aligned}
&(x^2+y)^2=4x^2y+y-4 \\
\Leftrightarrow &\;16x^2=y(4x^2+1)-4\\
\Leftrightarrow &\;16x^2=(4x-x^2)(4x^2+1)-4\\
\Leftrightarrow &\;4x^4-16x^3+17x^2-4x+4=0\\
 \Leftrightarrow &\;(x-2)^2(4x^2+1)=0\\
\Leftrightarrow &\;x=2 \\
\Rightarrow &\;y=4.
\end{aligned}$$
Vậy hệ phương trình có nghiệm duy nhất $(x; y)=(2; 4)$.
}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Ninh Bình, năm 2018]%[Nguyễn Cao Cường, Dự án EX-9G-18]%[9D0K2]
Cho phương trình $x^2-2x+m-3=0$ ($m$ là tham số). Tìm $m$ để phương trình có hai nghiệm phân biệt $x_1$ và $x_2$ thỏa mãn $\dfrac{x_1}{x_1^2+2x_2}+\dfrac{x_2}{x_2^2+2x_1}=\dfrac{1}{2}$.

\loigiai{
 Để phương trình có hai nghiệm phân biệt thì $\Delta'=4-m>0 \Rightarrow m<4$. Nếu $x_1, x_2$ là hai nghiệm của phương trình thì $x_1^2-2x_1+m-3=0, x_2^2-2x_2+m-3=0$. Áp dụng hệ thức Vi-ét ta có
 $x_1+x_2=2$, khi đó
$$\begin{aligned}
&\dfrac{x_1}{x_1^2+2x_2}+\dfrac{x_2}{x_2^2+2x_1}=\dfrac{1}{2}\\
 \Leftrightarrow &\;\dfrac{x_1}{2x_1+2x_2-m+3}+\dfrac{x_2}{2x_1+2x_2-m+3}=\dfrac{1}{2}\\
 \Leftrightarrow &\;\dfrac{x_1}{7-m}+\dfrac{x_2}{7-m}=\dfrac{1}{2}\\
  \Leftrightarrow &\;\dfrac{x_1+x_2}{7-m}=\dfrac{1}{2}\\
 \Leftrightarrow &\;7-m=4 \\
 \Leftrightarrow &\;m=3 \;(\text{TMĐK}).
\end{aligned}$$
Vậy $m=3$ là giá trị cần tìm.
}
\end{ex}
\begin{ex}%[Đề thi HSG 9, tỉnh Ninh Bình, năm 2018]%[Nguyễn Cao Cường, Dự án EX-9G-18]%[9D0G5]
 Tìm nghiệm nguyên của phương trình $x^2+xy+y^2=x^2y^2$.
\loigiai{
 Ta có  $x^2+xy+y^2=x^2y^2 \Leftrightarrow xy(xy+1)=(x+y)^2$. Do $xy$ và $xy+1$ là hai số nguyên liên tiếp nên $xy=0$ hoặc $xy+1=0$. 
\begin{itemize}
\item [-] Xét $xy=0$ thì suy ra $x=y=0$.
\item [-] Xét $xy+1=0 \Rightarrow x=-y$. Thế vào phương trình được $y^2(y^2-1)=0 \Leftrightarrow y=0; y=\pm 1$.
\end{itemize}
Vậy phương trình có nghiệm nguyên $(x; y)$ là $(0; 0)$, $(-1; 1)$, $(1; -1)$.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Ninh Bình, năm 2018]%[Nguyễn Cao Cường, Dự án EX-9G-18]%[9D0G4]
 Tìm các số tự nhiên $n$ sao cho $n^2+12n+1975$ là số chính phương.
\loigiai{
 Đặt $$n^2+12n+1975=m^2 \Leftrightarrow m^2-(n+6)^2=1939 \Leftrightarrow (|m|-|n+1|)(|m|+|n+6|)=1939, (m\in \mathbb{Z}).$$
Do $|m|+|n+6|>|m|-|n+6|$ nên ta có các trường hợp sau:
\begin{itemize}
\item [-] Trường hợp 1: $\heva{&|m|+|n+6|=1939\\&|m|-|n+6|=1} \Leftrightarrow n=963$.
\item [-] Trường hợp 2: $\heva{&|m|+|n+6|=277\\&|m|-|n+6|=7} \Leftrightarrow n=129$.
\end{itemize}
Vậy $n=963$ hoặc $n=129$.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Ninh Bình, năm 2018]%[Nguyễn Cao Cường, Dự án EX-9G-18]%[9H0K2]%[9H0K2]%[9H0K5]%[9H0K7]
Cho tam giác nhọn $ABC$ nội tiếp đường tròn $(O)$. Các đường cao $AK, BD, CI$ của tam giác $ABC$ cắt nhau tại $H$. Gọi $M$ là một điểm thay đổi trên cung nhỏ $BC$ của $(O)$ ($M$ khác $B$ và $C$). Gọi $N$, $P$ lần lượt là điểm đối xứng của $M$ qua các đường thẳng $AB$ và $AC$.
\begin{enumerate}
\item  Chứng minh rằng $AO$ vuông góc với $ID$.
\item  Chứng minh rằng các tứ giác $AHCP$ và $AHBN$ nội tiếp.
\item  Chứng minh rằng ba điểm $N, H, P$ thẳng hàng.
\item  Tìm vị trí của $M$ để đoạn thẳng $NP$ có độ dài lớn nhất.
\end{enumerate}

\begin{enumerate}
\item  Qua $A$ kẻ tiếp tuyến $Ax$ với $(O) \Rightarrow Ax \perp OA$. Tứ giác $BIDC$ có $\widehat{BIC}=\widehat{BDC}=90^{\circ}$ nên nội tiếp được trong đường tròn có tâm là trung điểm $BC$, suy ra $\widehat{ADI}=\widehat{ABC}$. Mà $\widehat{ABC}=\widehat{PAx} \Rightarrow \widehat{ADI}=\widehat{PAx}\Rightarrow ID\parallel Ax \Rightarrow AO\perp ID$.
\item  Ta có $\widehat{HDC}=\widehat{HKC}=90^{\circ}$ nên $\widehat{ACB}+\widehat{DHK}=180^{\circ}$. Mà $$\widehat{DHK}=\widehat{AHB}; \widehat{ACB}=\widehat{AMB}=\widehat{ANB}\Rightarrow\widehat{AHB}+\widehat{ANB}=180^{\circ}.$$ Do đó tứ giác $AHBN$ nội tiếp. \begin{ex}%[Đề thi HSG 9, tỉnh Ninh Bình, năm 2018]%[Nguyễn Cao Cường, Dự án EX-9G-18]%[9D0G7]
\begin{enumerate}
\item  Chứng minh rằng $\sqrt{x^4+1}\geq \dfrac{1}{\sqrt{17}}(x^2+4)$ với mọi số thực $x$.
\item  Cho $a, b$ là các số thực thỏa mãn $(a+1)(b+1)=\dfrac{9}{4}$. Tìm giá trị lớn nhất của $$P=\sqrt{a^4+1}+\sqrt{b^4+1}.$$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $\sqrt{x^4+1}\geq \dfrac{1}{\sqrt{17}}(x^2+4)>0 \Leftrightarrow 17(x^4+1)\geq (x^2+4)^2 \Leftrightarrow (4x^2-1)^2\geq 0$ đúng với mọi số thực $x$. Dấu ``='' xảy ra khi và chỉ khi $x=\pm \dfrac{1}{2}$.
\item Áp dụng kết quả câu trên, ta có $P\geq \dfrac{1}{17}(a^2+b^2+8)$. Ta lại có $$\dfrac{a^2+b^2}{2}\geq ab; a^2+\dfrac{1}{4}\geq a \text{ và }b^2+\dfrac{1}{4}\geq b.$$ Cộng theo vế các BĐT này được $$\begin{aligned}
&\dfrac{3(a^2+b^2)}{2}+\dfrac{3}{2}\geq ab+a+b+1=(a+1)(b+1)=\dfrac{9}{4}\\
\Leftrightarrow &\;a^2+b^2\geq \dfrac{1}{2}\\
\Rightarrow &\;P\geq \dfrac{1}{\sqrt{17}}\left(\dfrac{1}{2}+8\right)=\dfrac{\sqrt{17}}{2}.
\end{aligned}$$
Vậy giá trị lớn nhất của $P$ là $\dfrac{\sqrt{17}}{2}$. Đạt được khi và chỉ khi $a=b=\dfrac{1}{2}$.
\end{enumerate}
}
\end{ex}
