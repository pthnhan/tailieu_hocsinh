\subsection{Đề số 1a}
\setcounter{bt}{0}
\begin{bt}%[Nguyễn Khắc Hưởng]%[2D4Y1]
	($3{,}0$ điểm) Cho hai số phức $z_1=(3+y)+(x-2y)i,$ $z_2=4+xi$, với $x,y\in \mathbb{R}$.
	\begin{enumerate}
		\item Tìm các số thực $x, y$ biết $z_1=\overline{z_2}$.
		\item Với $x, y$ vừa tìm được ở trên, tính $|z_1|$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item 
			\begin{itemize}
			\item $\overline{z_2}=4-xi$. \dotfill $1{,}0$ điểm.
			\item $z_1=\overline{z_2}\Leftrightarrow \heva{&3+y=4\\&x-2y=-x}\Leftrightarrow \heva{x=1\\y=1}$. \dotfill $1{,}0$ điểm.
			\end{itemize}
			\item Vớp $x=y=1,$ ta có $z_1=4-i$. Vậy $|z_1|=\sqrt{4^2+(-1)^2}=\sqrt{17}$.  \dotfill $1{,}0$ điểm.
		\end{enumerate}
	}
	
\end{bt}
\begin{bt}%[Nguyễn Khắc Hưởng]%[2D4B2]
	($2{,}0$ điểm) Cho số phức $z$ thỏa mãn $(1+2i)^2z+\overline{z}=4i-20$. Tìm phần ảo của $z$.
	\loigiai{
		Đặt $z=a+bi (a,b\in \mathbb{R})$.\\
		Đẳng thức đã cho trở thành $(-3+4i)(a+bi)+(a-bi)=4i-20$\dotfill $1{,}0$ điểm.\\
		 $\Leftrightarrow \heva{&a+2b=10\\&a-b=1}\Leftrightarrow \heva{a=4\\b=3}$.\\
		Vậy phần ảo của $z$ bằng $3$.\dotfill $1{,}0$ điểm.
	}
\end{bt}
\begin{bt}%[Nguyễn Khắc Hưởng]%[2D4K3]
	($2{,}0$ điểm) Trong mặt phẳng tọa độ $Oxy,$ tìm tập hợp điểm biểu diễn các số phưc$z$ thỏa mãn: $\left|\dfrac{z-i}{1+i}\right|=|z|$.
	\loigiai{
Biểu diễn số phức $z=x+yi$ bởi điểm $M(x;y)$ trong mặt phẳng tọa độ $Oxy,$ ta có\\ $\left|\dfrac{z-i}{1+i}\right|=|z|$ $\Leftrightarrow |z-i|=|1+i||z|$ $\Leftrightarrow x^2+(y-1)^2=2(x^2+y^2)$.\dotfill $1{,}0$ điểm.\\
$\Leftrightarrow x^2+y^2+2y-1=0\Leftrightarrow x^2+(y+1)^2=2$\\
Vậy tập hợp điểm $M$ biểu diễn các số phức $z$ là đường tròn có phương trình\\ $x^2+(y+1)^2=2$.\dotfill $1{,}0$ điểm.
	}
\end{bt}
\begin{bt}%[Nguyễn Khắc Hưởng]%[2D4K4]
	($2{,}0$ điểm) Gọi $z_1$ và $z_2$ là hai nghiệm phức của phương trình $z^2-2z+2=0$, trong đó $z_1$ có phần ảo dương. Tìm phần ảo của số phức $\omega=z_1^{2018}$.
	\loigiai{
Giải phương trình ta được $z_1=1+i, z_2=1-i$\\
Xét $z_1^2=(1+i)^2=1+2i+i^2=2i$.\dotfill $1{,}0$ điểm.\\
Do đó $\omega =(1+i)^{2018}=(2i)^{1009}=2^{1009}(i^4)^{252}i=2^{1009}i$.\\
Vậy phần ảo của $\omega$ bằng: $2^{1009}$.\dotfill $1{,}0$ điểm.
	}
\end{bt}
\begin{bt}%[Nguyễn Khắc Hưởng]%[2D4G2]
	($1{,}0$ điểm) Cho hai số phức $z_1$, $z_2$ thỏa mãn $|z_1+5|=5$ và $|z_2+1-3i|=|z_2-3-6i|$. Tìm giá trị nhỏ nhất của $T=|z_1-z_2|$.
	\loigiai{
Gọi $M, N$ lần lượt là điểm biểu diễn của $z_1, z_2$.\\
Giả thiết $|z_1+5|=5$ tương đương với tập hợp $M$ là đường tròn $(C)$ tâm $I(-5;0),$ bán kính $R=5$.\\
Giả thiết $|z_2+1-3i|=|z_2-3-6i|$ tương đương với tập hợp $N$ là đường thẳng $d:6x+8y-35=0$. \dotfill $0{,}5$ điểm.\\
Bài toán trở thành tìm $M$ thuộc $(C)$ và $N$ thuộc $d$ sao cho $T=MN$ ngắn nhất.\\
Thấy $d(I,d)=\dfrac{15}{2}>5=R$ nên $d$ và $(C)$ không có điểm chung.\\
Do đó $MN$ nhỏ nhất bằng $d(I,d)-R$.\\
Vậy giá trị nhỏ nhất của $T$ bằng $d(I,d)-R=\dfrac{15}{2}-5=\dfrac{5}{2}$.\dotfill $0{,}5$ điểm.
	}
\end{bt}
