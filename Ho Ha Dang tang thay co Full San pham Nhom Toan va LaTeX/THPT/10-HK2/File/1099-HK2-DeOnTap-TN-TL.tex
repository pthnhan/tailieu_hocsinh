\begin{ex}%[HK2 (2016-2017), THPT Vĩnh Cửu, Đồng Nai]%[0D6G3]%[Trần Quang Thạnh]

\begin{bt}%[HK2 (2016-2017), THPT Vĩnh Cửu, Đồng Nai]%[0D6K3]%[Nguyễn Anh Tuấn]
	 Cho $ \cos \alpha=-\dfrac{12}{13} $ và $ \dfrac{\pi}{2}<\alpha<\pi $. Tính $ \sin 2\alpha, \cos 2\alpha, \tan 2\alpha $.
	\loigiai{
		\begin{itemize}
			\item Vì $ \cos 2\alpha =2\cos^2 \alpha -1=2\cdot \left(-\dfrac{12}{13}\right)^2-1=\dfrac{119}{169}$. 
			\item Vì $ \dfrac{\pi}{2}<\alpha<\pi $ nên $ \sin \alpha >0 $. Khi đó $ \sin^2 \alpha =1-\cos^2 \alpha =1-\left(-\dfrac{12}{13}\right)^2=\dfrac{25}{169} $.\\ Suy ra $ \sin \alpha =\dfrac{5}{13} $. Vậy $ \sin 2\alpha =2\sin \alpha \cdot \cos \alpha=2\cdot \dfrac{5}{13}\cdot \left(-\dfrac{12}{13}\right)=-\dfrac{120}{169} $.
			\item Do đó, $ \tan 2\alpha=\dfrac{\sin 2\alpha}{\cos 2 \alpha}=-\dfrac{120}{119} $.
		\end{itemize}
	} 
\end{bt}
\begin{bt}%[HK2 (2016-2017), THPT Vĩnh Cửu, Đồng Nai]%[0D6G3]%[Nguyễn Anh Tuấn]
	Chứng minh rằng $\sin^6\dfrac{x}{2}-\cos^6\dfrac{x}{2}=\dfrac{1}{4}\cos x \cdot (\sin^2x-4)$.
	\loigiai{Ta có
		\begin{align*}
		\sin^6a-\cos^6a&=(\sin^2a)^3-(\cos^2a)^3\\
		&=\left(\sin^2a-\cos^2a\right)^3+3\sin^2a \cdot \cos^2a \cdot  \left(\sin^2a-\cos^2a\right)\\
		&=-\cos^3{2a}-\dfrac{3}{4}\sin^22a \cdot \cos2a=\dfrac{1}{4}\cos2a \cdot (\sin^22a-4).
		\end{align*}
		Áp dụng với $a=\dfrac{x}{2}$, ta được $\sin^6\dfrac{x}{2}-\cos^6\dfrac{x}{2}=\dfrac{1}{4}\cos x \cdot (\sin^2x-4)$.
	}
\end{bt}
