\section{BẤT PHƯƠNG TRÌNH MŨ, LÔGARIT}
\subsection{Tóm tắt lý thuyết}
\subsubsection{Bất phương trình mũ cơ bản}

\begin{dn}
	Bất phương trình mũ cơ bản có dạng $a^x>b$ 
	(hoặc $a^x\ge b,\ a^x<b,\ a^x\le b$) với $a>0,\ a\ne 1$.
\end{dn}

\begin{enumerate}
	\item Xét bất phương trình dạng $a^x>b$. (dạng $a^x\ge b$ giải tương tự)
	\begin{itemize}
		\item Nếu $b\le 0$, tập nghiệm của bất phương trình là $\mathbb{R}$.
		\item Nếu $b>0$, khi đó\\
		Với $a>1$, ta có $a^x>b \Leftrightarrow x>\log_a b$.\\
		Với $0<a<1$, ta có $a^x>b \Leftrightarrow x<\log_a b$.
	\end{itemize}
	\item Xét bất phương trình dạng $a^x\le b$. (dạng $a^x< b$ giải tương tự)
	\begin{itemize}
		\item Nếu $b\le 0$, bất phương trình vô nghiệm.
		\item Nếu $b>0$, khi đó\\
		Với $a>1$, ta có $a^x\le b \Leftrightarrow x \le \log_a b$.\\
		Với $0<a<1$, ta có $a^x \le b \Leftrightarrow x \ge \log_a b$.
	\end{itemize}
\end{enumerate}
\subsubsection{Bất phương trình logarit}
\begin{enumerate}
	\item Bất phương trình logarit cơ bản
	\begin{dn}
		Bất phương trình logarit cơ bản có dạng $\log_a x>b$ (hoặc $\log_a x \ge b$, $\log_a x<b$, $\log_a \le b$) với $a>0, a\ne 1$.
	\end{dn}
	Xét bất phương trình $\log_a x >b$. (1)
	\begin{itemize}
		\item Trường hợp $a>1$: (1) $\Leftrightarrow x>a^b$. 
		\item Trường hợp $0<a<1$: (1) $\Leftrightarrow 0<x<a^b$. 
	\end{itemize}
	\item Một số bất phương trình logarit đơn giản\\
	Một số cách giải bất phương trình logarit đơn giản.
	\begin{itemize}
		\item Đưa về bất phương trình logarit cơ bản.
		\item Đặt ẩn phụ.
		\item Mũ hóa.
		\item Sử dụng tính đơn điệu của hàm số, đánh giá, bất đẳng thức.
	\end{itemize}
\end{enumerate}

\begin{dang}{Bất phương trình mũ và logarit bằng phương pháp đặt ẩn phụ}
	Khi giải bất phương trình mũ và logarit bằng phương pháp đặt ẩn phụ ta thường sử dụng một số phương pháp sau:
	\begin{itemize}
		\item Dùng ẩn phụ chuyển bất phương trình thành một bất phương trình với một ẩn phụ.\\
		(Ở đây ta coi các bất phương trình có dạng $f(x) \ge 0$, các trường hợp khác tiến hành tương tự).\\
		* Các phép đặt ẩn phụ với bất phương trình mũ thường gặp:
		\begin{enumerate}[+)]
			\item Bất phương trình có dạng: ${A_n}{a^{nf(x)}} + {A_{n - 1}}{a^{(n - 1)f(x)}} + ... + {A_1}{a^{f(x)}} + {A_0} \ge 0.$\\
			Đặt $t=a^{f(x)} (t>0)$, ta thu được bất phương trình: ${A_n}{t^n} + {A_{n - 1}}{t^{n - 1}} + ... + {A_1}t + {A_0} \ge 0$.
			\item Bất phương trình có dạng: $A{a^{2f(x)}} + B{{(ab)}^{f(x)}} + C{b^{2f(x)}} \ge 0.$\\
			Chia cả hai vế cho $b^{2f(x)}$ rồi đặt $t=\left(\dfrac{a}{b}\right)^{f(x)}$ ta được: $A.t^2+B.t+C\ge 0.$
			\item Bất phương trình có dạng: $A{a^{f(x)}} + B{b}^{g(x)} + C \ge 0$ trong đó, $a^{f(x)}b^{g(x)}=k$.\\
			Đặt $a^{f(x)}=t (t>0) \Rightarrow b^{g(x)}=k$.\\
			Ta thu được bất phương trình mới: $At+\dfrac{Bk}{t}+C \ge 0.$
		\end{enumerate}
		* Các phép đặt ẩn phụ thường gặp với bất phương trình logarit:
		\begin{enumerate}[+)]
			\item Đặt $t=\log_a{x} (x>0)$.
			\item Khi bất phương trình xuất hiện $a^{\log_b{x}}$, ta đặt $t=\log_b{x}$.
		\end{enumerate}
		\item Dùng ẩn phụ chuyển bất phương trình thành một bất phương trình với một ẩn phụ và hệ số vẫn chứa $x$.\\
		Trong trường hợp này, ta thường thu được một bất phương trình bậc hai theo ẩn phụ (hoặc vẫn theo $x$) có $\Delta$ là một số chính phương. 
		\item Dùng ẩn phụ chuyển bất phương trình thành một hệ bất phương trình với hai ẩn phụ.\\
		Bằng việc sử dụng hai ẩn phụ, ta đưa bất phương trình đã cho về một hệ gồm có:
		\begin{enumerate}[+)]
			\item Bất phương trình có được từ bất phương trình đầu bài.
			\item Phương trình (hoặc bất phương trình) có được từ việc đánh giá mối quan hệ của hai ẩn phụ.
		\end{enumerate}
	\end{itemize}
\end{dang}

\begin{dang}{Phương pháp đặt ẩn phụ trong bất phương trình logarit}

\begin{vd}%[Vũ Nguyễn Hoàng Anh]%[2D2G6]
	Giải bất phương trình $x+\log_2 x >1$.
	\loigiai{

		\end{align*}
		Vậy tập nghiệm của bất phương trình đã cho là $S=(1:+\infty)$.
	}
\end{vd}

	\begin{enumerate}
		\item Giải bất phương trình với $m=2$.
		\item Tìm tất cả các giá trị của $m$ để bất phương trình vô nghiệm.
	\end{enumerate}
	\loigiai{
		Bất phương trình đã cho tương đương
		\begin{align*}
			2^{\left(m^2-1\right)x} +\left(m^2-1\right) x< 2^{x-4m+3} +x-4m+3.
		\end{align*}
		Đặt $f(t)=2^t+t$. Ta có $f'(t)=2^t\ln 2+1>0,\ \forall x\in \mathbb{R}$ nên $f(t)$ là hàm số đồng biến trên $\mathbb{R}$. Do đó
		\begin{align*}
			&2^{\left(m^2-1\right)x} +\left(m^2-1\right) x< 2^{x-4m+3} +x-4m+3\\
			\Leftrightarrow \ &f\left[\left(m^2-1\right)x\right] < f(x-4m+3)\\
			\Leftrightarrow\  &\left(m^2-1\right)x<x-4m+3\\
			\Leftrightarrow\ &\left(m^2-2\right)x<-4m+3.\tag{1}
		\end{align*}
		\begin{enumerate}
			\item Với $m=2$, bất phương trình trở thành $2x<-5 \Leftrightarrow x<-\dfrac{5}{2}$.
			\item Bất phương trình đã cho vô nghiệm khi và chỉ khi bất phương trình $(1)$ vô nghiệm hay
			\begin{align*}
				\heva{&m^2-2=0\\ &-4m+3\le 0} \Leftrightarrow m=\sqrt{2}.
			\end{align*}
			Vậy $m=\sqrt{2}$ là giá trị cần tìm.
		\end{enumerate}
	}
\end{bt}

\begin{bt}%[Vũ Nguyễn Hoàng Anh]%[2D2G6]
	Giải và biện luận bất phương trình $x^2-(m+3)x+3m<(m-x)\log_2 x$.
	\loigiai{
		Điều kiện của bất phương trình $x>0$. Khi đó bất phương trình tương đương
		\begin{align*}
			(x-m)\left(x-3 + \log_2 x\right)<0.
		\end{align*}
		Đặt $f(x)=x-3+\log_2 x$, với $x>0$. Dễ thấy $f(x)$ là hàm số đồng biến trên $(0;+\infty)$ và $f(2)=0$ nên
		\begin{align*}
			(1)\Leftrightarrow \hoac{&\heva{& x-m>0\\ &f(x)<0}\\ &\heva{& x-m<0\\ &f(x)>0}} \Leftrightarrow \hoac{&\heva{&x>m\\ &x<2}\\ &\heva{&x<m\\ &x>2}}
		\end{align*}
	}
	Kết hợp với điều kiện ta có
	\begin{itemize}
		\item Nếu $m\le 0$ thì bất phương trình có tập nghiệm là $(0;2)$.
		\item Nếu $0<m<2$ thì bất phương trình đã cho có tập nghiệm là $(m;2)$.
		\item Nếu $m=2$ thì bất phương trình đã cho vô nghiệm.
		\item Nếu $m>2$ bất phương trình có tập nghiệm là $(2;m)$.
	\end{itemize}
\end{bt}

\begin{bt}%[Vũ Nguyễn Hoàng Anh]%[2D2G6]
	Tìm tất cả các giá trị của $m$ để bất phương trình
	\begin{align*}
		2^{(m+1)x+4}-2^{m^2-m-2}> \ln \left(m^2-m-2\right) - \ln \left[(m+1)x+4\right]
	\end{align*}
	nghiệm đúng với mọi $x\in [0;1]$.
	\loigiai{
		Điề kiện của bất phương trình $\heva{&m^2-m-2>0\\ &(m+1)x+4>0}$. Khi đó bất phương trình đã cho tương đương
		\begin{align*}
			2^{(m+1)x+4} + \ln \left[(m+1)x+4\right] > 2^{m^2-m-2} + \ln \left(m^2-m-2\right).\tag{1}
		\end{align*}
		Đặt $f(x)=2^t +\ln t$. Dễ thấy $f(t)$ là hàm số đồng biến trên $(0;+\infty)$. Do đó
		\begin{align*}\
			(1) \Leftrightarrow &(m+1)x+4>m^2-m-2\\
			\Leftrightarrow &g(x)=(m+1)x-m^2+m+6>0.\tag{2}
		\end{align*}
		Bất phương trình đã cho nghiệm đúng với mọi $x\in [0,1]$ khi và chỉ khi bất phương trình $(2)$ nghiệm đúng với mọi $x\in[0;1]$ hay
		\begin{align*}
			&\heva{&m^2-m-2>0\\ &g(x)>0,\ \forall  x\in [0;1]} \Leftrightarrow \heva{&\hoac{&m>2\\ &m<-1}\\ &\min\limits_{[0;1]} g(x)>0} \Leftrightarrow \hoac{&\heva{&m>2\\ &g(0)>0}\\ &\heva{&m<-1\\ &g(1)>0}}\\
			\Leftrightarrow &\hoac{&\heva{& m>2\\ &m^2-m-6<0}\\ &\heva{&m<-1\\ &m^2-2m-7<0}} \Leftrightarrow \hoac{& 1-\sqrt{8}<m<-1\\ &2<m<3}.
		\end{align*}
		Vậy tập các giá trị cần tìm của $m$ là $\left(1-\sqrt{8}; -1\right) \cup (2;3)$.
	}
\end{bt}


\begin{bt}%[Vũ Nguyễn Hoàng Anh]%[2D2G6]
	Tìm tất cả các giá trị của $m$ để bất phương trình
	\begin{align*}
		2^{\sin^2 x} + 3^{\cos^2 x} \ge m\cdot 3^{\sin^2 x}
	\end{align*}
	có nghiệm.
	\loigiai{
		Bất phương trình đã cho tương đương
		\begin{align*}
			&\left(\dfrac{2}{3}\right)^{\sin^2 x} + 3^{\cos^2x - \sin^2 x}\ge m\\
			\Leftrightarrow &\left(\dfrac{2}{3}\right)^{\sin^2 x} + 3^{1-2\sin^2 x}\ge m\\
			\Leftrightarrow &\left(\dfrac{2}{3}\right)^{\sin^2 x} + 3\left(\dfrac{1}{9}\right)^{\sin^2 x}\ge m.
		\end{align*}
		Đặt $t=\sin^2 x$, $0\le t\le 1$, khi đó bất phương trình đã cho trở thành
		\begin{align*}
			f(t)=\left(\dfrac{2}{3}\right)^t + 3\left(\dfrac{1}{9}\right)^t\ge m.
		\end{align*}
		Dễ thấy $f(t)$ là hàm số nghịch biến trên $[0;1]$ nên $\max\limits_{[0;1]} f(t) =f(0)=4$. Do đó bất phương trình đã cho có nghiệm khi và chỉ khi $m\le \max\limits_{[0;1]} f(t) \Leftrightarrow m\le 4$.
	}
\end{bt}
