\textbf{A. Lý thuyết (2 điểm):}  Học sinh chọn 1 trong 2 đề\\
\textbf{Đề 1}: Phát biểu và viết dạng tổng quát của qui tắc khai phương một tích.\\
Áp dụng: $P = \dfrac{\sqrt{50}-\sqrt{8}}{\sqrt{2}}$.
\loigiai{
	\textbf{Qui tắc khai phương một tích:} Muốn khai phương một tích của các số không âm, ta có thể khai phương từng thừa số rồi nhân các kết quả với nhau.\\
	Với hai số $a$ và $b$ không âm, ta có $\sqrt{a\cdot b}=\sqrt{a}\cdot \sqrt{b}$.\\
	\textbf{Áp dụng:}\\
	$P = \dfrac{\sqrt{50}-\sqrt{8}}{\sqrt{2}}=\dfrac{5\sqrt{2}-2\sqrt{2}}{\sqrt{2}}=\dfrac{3\sqrt{2}}{\sqrt{2}}=3$.\\
}
\textbf{Đề 2}: Định nghĩa đường tròn. Chứng minh rằng đường kính là dây lớn nhất của đường tròn.
\loigiai{
	\textbf{Định nghĩa đường tròn:} Đường tròn tâm $O$ bán kính $R$ (với $R>0$) là hình gồm các điểm cách điểm $O$ một khoảng bằng $R$, kí hiệu $(O;R)$.\\
	\textbf{Chứng minh đường kính là dây lớn nhất của đường tròn:}
	\immini
	{
		Gọi $AB$ là một dây bất kì của đường tròn $(O;R)$.\\
		Nếu $AB$ là đường kính thì $AB=2R$.\\
		Nếu $AB$ không là đường kính:\\
		Xét tam giác $AOB$, có:\\
		$AB<AO+OB=R+R=2R$.\\
		Vậy ta có $AB\le 2R$ hay đường kính là dây lớn nhất của đường tròn.
	}
	{

\noindent\textbf{B. Bài tập bắt buộc (8 điểm)} 
\begin{ex}%[2,5 điểm]%[Đề thi vào 10 TP Hà Nội năm 2002]%[9D1K8]
	Cho biểu thức $P = \left(\dfrac{4\sqrt{x}}{2+\sqrt{x}}+\dfrac{8x}{4-x}\right):\left(\dfrac{\sqrt{x}-1}{x-2\sqrt{x}}-\dfrac{2}{\sqrt{x}}\right)$.
	\begin{enumerate}
		\item Rút gọn $P$.
		\item Tìm giá trị của $x$ để $P =-1$.
		\item Tìm $m$ để với mọi giá trị của $x>9$ ta có:
		$m(\sqrt{x}-3)P >x+1$.	
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item ĐKXĐ: $x>0;x\ne 4$.\\
		$\begin{aligned}
			P&=\left(\dfrac{4\sqrt{x}}{2+\sqrt{x}}+\dfrac{8x}{4-x}\right):\left(\dfrac{\sqrt{x}-1}{x-2\sqrt{x}}-\dfrac{2}{\sqrt{x}}\right)\\
			& =\dfrac{4\sqrt{x}(2-\sqrt{x})+8x}{(2+\sqrt{x})(2-\sqrt{x})}:\dfrac{\sqrt{x}-1-2(\sqrt{x}-2)}{\sqrt{x}(\sqrt{x}-2)}\\
			& =\dfrac{8\sqrt{x}+4x}{(2+\sqrt{x})(2-\sqrt{x})}\cdot \dfrac{\sqrt{x}(\sqrt{x}-2)}{-\sqrt{x}+3}\\
			& = \dfrac{4\sqrt{x}(2+\sqrt{x})}{(2+\sqrt{x})(2-\sqrt{x})}\cdot \dfrac{\sqrt{x}(2-\sqrt{x})}{\sqrt{x}-3}\\
			& =\dfrac{4x}{\sqrt{x}-3}.
		\end{aligned}$
		\item $P=-1\Leftrightarrow \dfrac{4x}{\sqrt{x}-3}=-1\Leftrightarrow 4x+\sqrt{x}-3=0\Leftrightarrow \left[\begin{aligned}
			\sqrt{x}&=-1\\ \sqrt{x}&=\dfrac{3}{4}
		\end{aligned}\right.\Leftrightarrow x=\dfrac{9}{16}$ (thỏa mãn).\\
		Vậy $P=1$ khi và chỉ khi $x=\dfrac{9}{16}.$
		\item Ta có\\
		$\begin{aligned}
		&m(\sqrt{x}-3)P >x+1\;\forall x>9\\
		\Leftrightarrow \;& m(\sqrt{x}-3)\cdot\dfrac{4x}{\sqrt{x}-3}>x+1\;\forall x>9\\
		\Leftrightarrow\; & 4mx>x+1\;\forall x>9\\
		\Leftrightarrow\; & (4m-1)x>1\;\forall x>9\\
		\Leftrightarrow\; & 4m-1>\dfrac{1}{x}\;\forall x>9\\
		\Leftrightarrow\; & 4m-1\ge \dfrac{1}{9}\\
		\Leftrightarrow\; & m\ge \dfrac{5}{18}.
		\end{aligned}$
	\end{enumerate}
	}
\end{ex} 

\begin{ex}%[2,0 điểm]%[Đề thi vào 10 TP Hà Nội năm 2002]%[9D3B6]
	\textit{Giải bài toán sau bằng cách lập phương trình hoặc hệ phương trình:}\\
	Theo kế hoạch, hai tổ sản xuất 600 sản phẩm trong một thời gian nhất định. Do áp dụng kỹ thuật mới nên tổ I đã vượt mức $18 \%$, tổ II vượt mức $21\%$, vì vậy trong thời gian quy định họ đã hoàn thành vượt mức 120 sản phẩm. Hỏi số sản phẩm được giao của mỗi tổ theo kế hoạch?
	\loigiai{
	Gọi số sản phẩm được giao của tổ I và tổ II theo kế hoạch lần lượt là $x$ và $y$ ($0<x,y<600$; $x,y\in\mathbb{N}$).\\
	Do hai tổ được giao sản xuất $600$ sản phẩm nên ta có phương trình
	\begin{align}\label{1}
	x+y=600
	\end{align}
	Do tổ I vượt mức $18\%$, tổ II vượt mức $21\%$ và hai tổ đã hoàn thành vượt mức $120$ sản phẩm nên ta có phương trình
	\begin{align}\label{2}
	x(1+18\%)+y(1+21\%)=600+120\Leftrightarrow 118x+121y=72000
	\end{align}
	Từ \eqref{1} và \eqref{2}, ta có hệ phương trình
	\begin{equation*}
	\begin{cases}
	x+y=600\\ 118x+121y=72000
	\end{cases}\Leftrightarrow \begin{cases}
	x=200\text{ (thỏa mãn)}\\ y=400\text{ (thỏa mãn).}
	\end{cases}
	\end{equation*}
	Vậy theo kế hoạch, tổ I được giao $200$ sản phẩm, tổ II được giao $400$ sản phẩm.
	}
\end{ex}  


\begin{ex}%[3,5 điểm]%[Đề thi vào 10 TP Hà Nội năm 2002]%[9H3B7]%[9H3B7]%[9H3K3]%[9H3G8]
	Cho đường tròn $(O)$, một đường kính $AB$ cố định, một điểm $I$ nằm giữa $A$ và $O$ sao cho $AI = \dfrac{2}{3}AO$. Kẻ dây $MN$ vuông góc với $AB$ tại $I$. Gọi $C$ là điểm tùy ý thuộc cung lớn $MN$, sao cho $C$ không trùng với $M$, $N$ và $B$. Nối $AC$ cắt $MN$ tại $E$.
	\begin{enumerate}
		\item Chứng minh tứ giác $IECB$ nội tiếp được trong đường tròn.
		\item Chứng minh $\triangle AME$ đồng dạng với $\triangle ACM$ và $AM^2 = AE \cdot AC$.
		\item Chứng minh $AE.AC-AI.IB =AI^2$.
		\item Hãy xác định vị trí của điểm $C$ sao cho khoảng cách từ $N$ đến tâm đường tròn ngoại tiếp tam giác $CME$ là nhỏ nhất. 
	\end{enumerate}
	\loigiai{
	\begin{center}
		\begin{tikzpicture}[scale=0.7]
		\tkzInit[ymin=-9,ymax=7,xmin=-7,xmax=9]
		\tkzClip
		\tkzDefPoints{-6/0/A,6/0/B,0/0/O,-2/0/I}
		\tkzDefLine[perpendicular=through I](B,A) \tkzGetPoint{d}
		\tkzInterLC(d,I)(O,B)\tkzGetPoints{M}{N}
		phe\tkzDefPointBy[rotation= center O angle -80](N) \tkzGetPoint{C}
		\tkzInterLL(A,C)(M,N) \tkzGetPoint{E}
		\tkzCircumCenter(E,M,C)\tkzGetPoint{J}
		
		\tkzLabelPoints(I,E,J,B) \tkzLabelPoints[above](N)\tkzLabelPoints[left](A)
		\tkzLabelPoints[below](O)\tkzLabelPoints[above right](C)
		\tkzLabelPoints[below left](M)
		
		\tkzDrawPoints(A,B,C,O,I,N,M,E,J)
		\tkzDrawSegments(A,B M,N A,C C,B A,M M,C A,N N,B M,B)
		\tkzDrawCircle[radius](O,A)
		\tkzDrawCircle[radius](J,E)
		\end{tikzpicture}
	\end{center}
	\begin{enumerate}
		\item Do $MN\perp AB$ nên $\widehat{EIB}=90^\circ$.\\
		Vì $\widehat{ACB}$ là góc nội tiếp chắn nửa đường tròn nên $\widehat{ACB}=90^\circ$.\\
		Xét tứ giác $IECB$ có $\widehat{EIB}+\widehat{ECB}=90^\circ+90^\circ=180^\circ$.\\
		Mà hai góc ở vị trí đối nhau nên tứ giác $IECB$ là tứ giác nội tiếp.
		\item Vì $IECB$ là tứ giác nội tiếp nên $\widehat{AEI}=\widehat{IBC}$.\\
		Lại có $\widehat{ABC}=\widehat{AMC}$ (hai góc nội tiếp cùng chắn cung $\wideparen{AC}$).\\
		Suy ra $\widehat{AEM}=\widehat{AMC}$.\\
		Vậy $\triangle AME\backsim\triangle ACM$ (g-g).\\
		$\Rightarrow \dfrac{AM}{AC}=\dfrac{AE}{AM} \Rightarrow AM^2=AE\cdot AC$.
		\item Xét tam giác $AEI$ và tam giác $ABC$ có:\\
		$\begin{cases}
		\widehat{A}\text{ chung} \\ \widehat{AIE}=\widehat{ACB}=90^\circ
		\end{cases}$\\
		$\Rightarrow \triangle AEI\backsim\triangle ABC$ (g-g)$\Rightarrow \dfrac{AE}{AB}=\dfrac{AI}{AC}\Rightarrow AE\cdot AC=AB\cdot AI$.\\
		$\Rightarrow AE\cdot AC-AI\cdot IB=AB\cdot AI-AI\cdot IB=AI(AB-IB)=AI^2$.
		\item Gọi $J$ là tâm đường tròn ngoại tiếp tam giác $CME$.\\
		Vì $\triangle AEM\backsim\triangle AMC$ nên $\widehat{AME}=\widehat{ACM}$.\\
		Suy ra $AM$ là tiếp tuyến tại $M$ của $(J)$ $\Rightarrow JM\perp AM$.\\
		Mà $\widehat{AMB}=90^\circ$ (góc nội tiếp chắn nửa đường tròn) nên $BM\perp AM$.\\
		Vậy $J$ luôn thuộc đường thẳng $MB$.\\
		Do đó $NJ$ nhỏ nhất khi và chỉ khi $J$ trùng hình chiếu $H$ của $N$ trên $MB$ hay khi $C$ trùng với giao điểm của đường tròn $(H;HM)$ với $(O)$.
	\end{enumerate}	
	}
\end{ex}
