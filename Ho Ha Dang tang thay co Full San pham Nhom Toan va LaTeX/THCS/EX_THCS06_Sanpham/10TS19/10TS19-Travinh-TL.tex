	\textbf{(3,0 điểm)}
	\begin{enumerate} 
	\item  Rút gọn biểu thức $2\sqrt{75} + 3\sqrt{48} - 4\sqrt{27}.$
	\item Giải hệ phương trình $\begin{cases} 2x-y =8\\3x+2y=5. \end{cases}$
	\item Giải phương trình $3x^2 -7x +2 =0.$
\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Ta có $\sqrt{75} + 3\sqrt{48} - 4\sqrt{27} = 2\cdot 5\sqrt3 + 3\cdot 4 \sqrt 3 - 4\cdot 3 \sqrt 3 = 10 \sqrt 3 + 12 \sqrt 3 - 12\sqrt 3 = 10 \sqrt 3.$
			\item Ta có  $\begin{cases} 2x-y =8\\3x+2y=5 \end{cases} \Leftrightarrow \begin{cases} 4x - 2y =16\\3x+ 2y =5  \end{cases} \Leftrightarrow \begin{cases}
		7x = 21\\3x+2y=5
			\end{cases} \Leftrightarrow \begin{cases}
			x= 3\\y = -2.
			\end{cases}$ \\
			Vậy hệ phương trình đã cho có nghiệm là $\mathcal{S} = \left\{ {(3; - 2)} \right\}$.
			\item $\begin{aligned}[t]
			3x^2 -7x +2 =0 \Leftrightarrow  3x^2 -6x-x +2=0 \Leftrightarrow & 3x(x-2) -(x-2)=0\\\Leftrightarrow& (3x-1)(x-2)=0 \Leftrightarrow \left[\begin{aligned} x & =\dfrac{1}{3}\\ x &=2. \end{aligned}\right.
			 \end{aligned}$\\
		Vậy  phương trình đã cho có nghiệm là $\mathcal{S} = \left\{ {\dfrac{1}{2}; 2} \right\}$.	
		\end{enumerate}
		
	}
\end{ex}

	Cho hai hàm số $ y = -x +2$ và $y = x^2$ có đồ thị lần lượt là $(d)$ và $(P)$.
\begin{enumerate}
	\item Vẽ $(d)$ và $P$ trên cùng hệ trục tọa độ.
	\item Bằng phép toán tìm tọa độ giao điểm của $(d)$ và $(P)$.
\end{enumerate}
	\loigiai{
		\item Vẽ parabol và đường thẳng $ (d) $ như sau:\\
		Vẽ parabol $ (P): y =x^2$.
		\begin{itemize}
			\item Tập xác định của hàm số: $ \mathbb{R} .$
			\item Bảng giá trị:
			
			\begin{center}
				\begin{tabular}{|c|c|c|c|c|c|} \hline  
				$x$& $-2$&$ -1 $&$ 0 $&$ 1$&$2$\\ \hline 
				$ y = x^2 $&$4$  & $1$	&$0$ &$1$ &$4$ \\ \hline	
			\end{tabular}
			\end{center}
		\end{itemize}
	 Vẽ đường thẳng $ (d): y =-x +2. $\\
 Đồ thị của hàm số $ y=-x +2 $ là đường thẳng đi qua điểm $ (2;0) $ và $ (0;2) $.
		\begin{center}
			\begin{tikzpicture}[>=stealth]
			\draw[->,line width = 1pt] (-5,0)--(0,0) node[below left]{$O$}--(5,0) node[below]{$x$};
			\draw[->,line width = 1pt] (0,-3) --(0,8) node[right]{$y$};
			\foreach \x in {-2,-1,1,2,4}{
				\draw (\x,0) node[below]{$\x$} circle (1pt);
				\draw (0,\x) node[left]{$\x$} circle (1pt);
			}
			\draw [ thick, domain=-2.7:2.7, samples=100] %
			plot (\x, {(\x)^2}) node[right]{$y=x^2$};
			\draw [blue, domain=-4:4] plot (\x, {-\x+2})node[below right]{$y =-x+2$};
			\draw [dashed] (1,0)--(1,1)--(0,1);
			\draw [dashed]
			(-1,0)--(-1,1)--(0,1);
			\draw [dashed] (2,0)--(2,4)--(0,4);
			\draw [dashed] (-2,0)--(-2,4)--(0,4);
			
			\end{tikzpicture}
		\end{center}
		\item Phương trình hoành độ giao điểm của $ (d) $ và $ (P) $ là\\
		$\begin{aligned}[t]
		 x^2 = -x + 2 \Leftrightarrow x^2 +x-2=0 \Leftrightarrow &x^2 -x+2x-2=0\Leftrightarrow x(x-1)+2(x-1)=0\\ \Leftrightarrow& (x-1)(x+2)= 0 \Leftrightarrow \left[\begin{aligned} x & =1 \Rightarrow y = 1\\ x &=-2\Rightarrow y = 4. \end{aligned}\right.
		\end{aligned}$
		
		Vậy $ A(1;1) $, $ B(-2;4) $ là hai giao điểm của $ (d) $ và $ (P) $.}
\end{ex}
\begin{ex}%[9D4B4][9D0G4]%[Đề tuyển sinh vào lớp 10 THPT năm học 2018-2019, Trà Vinh]%[Huỳnh Thị Sâm]
Cho phương trình $x^2 - (m+1)x +m -2 =0$ (với $m$ là tham số).
\begin{enumerate}
	\item Chứng minh phương trình luôn có hai nghiệm phân biệt với mọi $m$.
	\item Tìm các số nguyên $m$ để phương trình có nghiệm nguyên.
\end{enumerate}
	\loigiai
	{\begin{enumerate}
			\item Ta có $\Delta = (m+1)^2 - 4(m-2) = m^2 +2m +1 - 4m + 8 = m^2 -2m + 9 =(m-1)^2 + 8 > 0.$\\
			Vì $\Delta > 0$ với mọi $m$ nên phương trình đã cho luôn có hai nghiệm phân biệt với mọi $m$.
			\item Phương trình có nghiệm nguyên khi và chỉ khi $\Delta$ là số chính phương, tức là $m^2 -2m +9$ là số chính phương.\\
			Đặt $m^2 -2m+ 9 = a^2 \Leftrightarrow (m-1)^2 + 8 = a^2 \Leftrightarrow a^2 -(m-1)^2 = 8 \Leftrightarrow (a-m+1)(a+m-1)=8.$\\
			Vì $a-m +1$ và $a+ m-1$ là các số nguyên nên chúng là ước của $8$, suy ra xảy ra các trường hợp sau.
			\begin{itemize}
			\item $a-m+1 = 8$ và $a+m -1= 1.$ Suy ra $ a= \dfrac{9}{2}$ và $m = - \dfrac{5}{2}\quad \text{(loại)}$ (Vì $m$ nguyên).
				\item  $a-m+1 = -8$ và $a+m -1= -1.$ Suy ra $ a= -\dfrac{9}{2}$ và $m = - \dfrac{9}{2}\quad \text{(loại)}$ (Vì $m$ nguyên).
			\item  $a-m+1 = -1$ và $a+m -1= -8.$ Suy ra $ a= -\dfrac{9}{2}$ và $m =  -\dfrac{5}{2}\quad \text{(loại)}$ (Vì $m$ nguyên).
			\item  $a-m+1 = 1$ và $a+m -1= 8.$ Suy ra $ a= \dfrac{9}{2}$ và $m = - \dfrac{9}{2}\quad \text{(loại)}$ (Vì $m$ nguyên).
			\item  $a-m+1 = 2$ và $a+m -1= 4.$ Suy ra $ a= 3$ và $m = 2\quad \text{(nhận)}$ (Vì $m$ nguyên).
			\item  $a-m+1 = 4$ và $a+m -1= 2.$ Suy ra $ a= 0$ và $m = 3\quad \text{(loại)}$ (Vì $a\ne 0$).
			\item  $a-m+1 = -2$ và $a+m -1= -4.$ Suy ra $ a= -3$ và $m = 0\quad \text{(nhận)}$ (Vì $m$ nguyên).
			\item  $a-m+1 = -4$ và $a+m -1= -2.$ Suy ra $ a= -3$ và $m = 2\quad \text{(nhận)}$ (Vì $m$ nguyên).
				
			\end{itemize}
		Vậy $m \in \left\{ {0;2;3} \right\}$ thì phương trình đã cho có nghiệm nguyên.
			
		\end{enumerate}
	
	}
	
\end{ex}

$AB^2 = BH\cdot BC = 3{,6}\cdot 10 = 36 \Rightarrow AB = 6$ (cm).\\

	Cho tam giác $ABC$ vuông tại $A$ $(AB< AC)$, $M$ là trung điểm của cạnh $AC$. Đường tròn đường kính $MC$ cắt $BC$ tại $N$. Đường thẳng $BM$ cắt đường tròn đường kính $MC$ tại $D$.
	\begin{enumerate}
		\item Chứng minh tứ giác $BADC$ nội tiếp.
		\item Chứng minh $DB$ là phân giác của góc $ADN$.
		\item $BA$ và $CD$ kéo dài cắt nhau tại $P$. Chứng minh ba điểm $P,M,N$ thẳng hàng.
	\end{enumerate}
	\end{ex}

	\begin{center}
		\begin{tikzpicture}[scale=1.2]

\tkzLabelPoints[ below right](D)

	\end{tikzpicture}
	\end{center}
	\begin{enumerate}
			\item Xét tứ giác $BADC$ có $\widehat{BAC} = 90^\circ$ ($\triangle ABC$ vuông tại $A$) \\
			và $\widehat{BDC} = 90^\circ$ ($BM$ cắt đường tròn đường kính $MC$ tại $D$).\\
			Suy ra tứ giác $BADC$ nội tiếp ( hai góc kề bằng nhau cùng chắn cung $ BC $).
			\item Xét tứ giác $NMDC$ có\\
			$\widehat{MNC} + \widehat{MDC} = 90^\circ + 90^\circ = 180^\circ$\\
			Suy ra tứ giác $NMDC$ nội tiếp.\\
			Do đó $\widehat{MDN} = \widehat{MCN}$ (góc nội tiếp cùng chắn cung $MN$) . Hay $\widehat{BDN} = \widehat{ACB}$.  \hfill (1)\\
			 Mà tứ giác $BADC$ nội tiếp (cmt)\\
			  Suy ra $\widehat{ADB} = \widehat{ACB}$ (góc nội tiếp cùng chắn cung $AB$).\hfill(2)\\
			  Từ (1) và (2) suy ra $\widehat{BDN} = \widehat{ABD}$.\\
			  Vậy chứng tỏ rằng $DB$ là phân giác của góc $ADN$.
			  \item 	Xét $\triangle BCP$ có $AC \perp BP$ ($\triangle ABC$ vuông tại $A$) và $BD \perp DC$ (gt).\\
			  Mà $AC$ và $BD$ cắt nhau tại $M$.\\
			  Suy ra $M$ là trực tâm $\triangle BPC$.\\
			  Do đó $PM \perp BC .$  \hfill (3)\\
			  Mặt khác $MN \perp NC$ (Vì $\triangle MNC$ nội tiếp), hay $MN \perp BC. $  \hfill (4) \\
			  Từ $(3)$ và $(4)$ suy ra $3$ điểm $P,M,N$ thẳng hàng.
			  
			
\end{enumerate}}
