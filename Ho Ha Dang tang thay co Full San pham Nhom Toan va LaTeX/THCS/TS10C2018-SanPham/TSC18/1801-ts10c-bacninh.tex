\begin{ex}%[Đề thi vào 10, chuyên Bắc Ninh, tỉnh Bắc Ninh, năm 2018]%[Nguyễn Hiếu, Dự án EX-TS10C-18]%[9D0K1]
	Rút gọn biểu thức $P=\left(\dfrac{a+\sqrt{a^2-b^2}}{a-\sqrt{a^2-b^2}}-\dfrac{a-\sqrt{a^2-b^2}}{a+\sqrt{a^2-b^2}}\right):\dfrac{4\sqrt{a^4-a^2b^2}}{b^2}$ với $|a|>|b|>0$.
	\loigiai{
	Ta có \begin{eqnarray*}
	P&=&\dfrac{\left(a+\sqrt{a^2-b^2}\right)^2-\left(a-\sqrt{a^2-b^2}\right)^2}{b^2}:\dfrac{4|a|\sqrt{a^2-b^2}}{b^2}\\
	&=&4a\sqrt{a^2-b^2}:\left(4|a|\sqrt{a^2-b^2}\right)\\
	&=&\dfrac{a}{|a|}
	\end{eqnarray*}
	Do đó nếu $a>0$ thì $P=1$; nếu $a<0$ thì $P=-1$.
		}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Bắc Ninh, tỉnh Bắc Ninh, năm 2018]%[Nguyễn Hiếu, Dự án EX-TS10C-18]%[9D0K2]
	Cho phương trình $x^2+ax+b=0 \quad (1)$, với $x$ là ẩn, $a$, $b$ là tham số. Tìm $a,b$ biết phương trình $(1)$ có hai nghiệm $x_1$, $x_2$ thỏa mãn $\heva{&x_1-x_2=5\\&x_1^3-x_2^3=35.}$
	\loigiai{
	Phương trình $(1)$ có hai nghiệm khi và chỉ khi $a^2-4b\ge 0$.\\
	Theo hệ thức Vi-ét ta có $\heva{&x_1+x_2=-a\\&x_1x_2=b.}$\\
	Ta có $x_1^3-x_2^3=(x_1-x_2)^3+3x_1x_2(x_1-x_2)$. Suy ra $35=125+15b \Leftrightarrow b=-6$.\\
	Mà $x_1=x_2+5$ nên $x_2(x_2+5)=-6 \Leftrightarrow \hoac{&x_2=-2\\&x_2=-3.}$\\
	Với $x_2=-2\Rightarrow x_1=3 \Rightarrow a=-1$ (thỏa mãn).\\
	Với $x_2=-3 \Rightarrow x_1=2 \Rightarrow a=1$ (thỏa mãn).
	Vậy $ a=-1, b=-6 $ hoặc $ a=1,b=-6 $.
		}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Bắc Ninh, tỉnh Bắc Ninh, năm 2018]%[Nguyễn Hiếu, Dự án EX-TS10C-18]%[9D0G2]
	Giải phương trình $\sqrt{x+3}+\sqrt{3x+1}=x+3$.
	\loigiai{
	Điều kiện $x\ge -\dfrac{1}{3}$.\\
	Áp dụng bất đẳng thức Cauchy ta có
	$$2\sqrt{x+3}+2\sqrt{3x+1} \le \dfrac{4+x+3}{2}+\dfrac{4+3x+1}{2}=2x+6$$
	Suy ra $\sqrt{x+3}+\sqrt{3x+1}\le x+3$.\\
	Đẳng thức xảy ra khi và chỉ khi $\heva{&\sqrt{x+3}=2\\&\sqrt{3x+1}=4}\Leftrightarrow x=1$.\\
	Thử lại $x=1$ thỏa mãn phương trình đã cho.\\
	Vậy phương trình có nghiệm duy nhất $x=1$.
		}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Bắc Ninh, tỉnh Bắc Ninh, năm 2018]%[Nguyễn Hiếu, Dự án EX-TS10C-18]%[9D0G7]
	Cho các số thực $a,b,c$ thỏa mãn điều kiện $0\le a,b,c\le 2$ và $a+b+c=3$. Tìm giá trị lớn nhất và nhỏ nhất của biểu thức $P=\dfrac{a^2+b^2+c^2}{ab+bc+ca}$.
	\loigiai{
	Ta có $a^2+b^2+c^2-(ab+bc+ca)=\dfrac{1}{2}(a-b)^2+\dfrac{1}{2}(b-c)^2+\dfrac{1}{2}(c-a)^2 \ge 0$.\\
	Nên $a^2+b^2+c^2 \ge ab+bc+ca \Rightarrow P\ge 1$.\\
	Dấu ``='' xảy ra khi và chỉ khi $a=b=c=1$.\\
	Vì $0 \le a,b,c\le 2$ nên 
	$$(2-a)(2-b)(2-c)\ge 0 \Leftrightarrow abc- 2(ab+bc+ca) + 4(a+b+c)-8 \le 0.$$
	Do đó $ab+bc+ca \ge \dfrac{abc+4}{2}\ge 2.$\\
	Suy ra
	$$P=\dfrac{(a+b+c)^2-2(ab+bc+ca)}{ab+bc+ca}=\dfrac{9}{ab+bc+ca}-2 \le \dfrac{9}{2}-2=\dfrac{5}{2}.$$
	Dấu bằng xảy ra khi và chỉ khi $a=2,b=1,c=0$ và các hoán vị.
		}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Bắc Ninh, tỉnh Bắc Ninh, năm 2018]%[Nguyễn Hiếu, Dự án EX-TS10C-18]%[9D0K4]
	Tìm cặp số nguyên tố $(x,y)$ thỏa mãn phương trình $x^2-2y^2=1$.
	\loigiai{
	Vì vế phải lẻ nên $x$ là số nguyên tố lẻ.\\
	Do đó $2y^2 = x^2-1 = (x-1)(x+1) ~\vdots~ 4 \Rightarrow y^2 ~\vdots~ 2\Rightarrow y$ là số chẵn.\\
	Mặt khác $y$ là số nguyên tố nên $y=2$. Thay vào phương trình ban đầu được $x=3$.\\
	Vậy $x=3;y=2$ thỏa mãn phương trình đã cho.
		}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Bắc Ninh, tỉnh Bắc Ninh, năm 2018]%[Nguyễn Hiếu, Dự án EX-TS10C-18]%[9D0G4]
	Chứng minh rằng nếu hiệu các lập phương của hai số nguyên liên tiếp là bình phương của một số tự nhiên $n$ thì $n$ là tổng của hai số chính phương liên tiếp.
	\loigiai{
	Giả sử $n^2=(k+1)^3-k^3=3k^2+3k+1$ với $k \in \Bbb Z$, suy ra 
	$$4n^2=12k^2+12k+4 \Leftrightarrow (2n-1)(2n+1)=3(2k+1)^2.$$
	Vì $2n-1$, $2n+1$ là hai số lẻ liên tiếp nên chúng nguyên tố cùng nhau. 
	Do đó $\heva{&2n-1=3t^2\\&2n+1=s^2}$ hoặc $\heva{&2n+1=3t^2\\&2n-1=s^2}$ với $s,t$ là các số nguyên dương nguyên tố cùng nhau và thỏa mãn $st=2k+1$; $s$ không chia hết cho 3.\\
	Vì $s$ lẻ nên $s=6m+1$ hoặc $6m-1$.\\
	Trường hợp 1: $\heva{&2n-1=3t^2\\&2n+1=s^2.}$\\
	Từ $2n+1=s^2=(6m \pm 1)^2=36m^2\pm 12m+1 \Rightarrow 3t^2=2n-1=36m^2\pm 12m -1$ (vô lí).\\
	Trường hợp 2: $\heva{&2n+1=3t^2\\&2n-1=s^2.}$\\
	Từ $2n-1=s^2\Rightarrow 2n=36m^2\pm 12m +2 $ nên $n=(3m)^2+(3m\pm 1)^2$.\\
	Vậy ta được điều phải chứng minh.
		}
\end{ex}

\begin{ex}%[Đề thi vào 10, chuyên Bắc Ninh, tỉnh Bắc Ninh, năm 2018]%[Nguyễn Hiếu, Dự án EX-TS10C-18]%[9H0G2]
	Từ điểm $A$ nằm ngoài đường tròn $(O)$ kẻ hai tiếp tuyến $AB, AC$ của đường tròn $(O)$ với $B,C$ là các tiếp điểm. Gọi $H$ là giao điểm của $AO$ và $BC$. Đường tròn đường kính $CH$ cắt đường tròn $(O)$ tại điểm $D$ khác $C$. Gọi $T$ là trung điểm của $BD$.

	\item Gọi $E$ là giao điểm của đường tròn đường kính $AB$ và $AC$ ($E$ khác $A$); $S$ là giao điểm của $AO$ và $BE$. Chứng minh rằng $TS$ song song với $HD$.
	\end{enumerate}

		\begin{enumerate}
		\item Ta có $AH \perp HC$ nên $AH$ là tiếp tuyến của đường tròn đường kính $CH \Rightarrow \widehat{AHD}=\widehat{HCD}$.\\
		Mà $\widehat{HCD}=\widehat{ABD}\Rightarrow \widehat{AHD}=\widehat{ABD}$.\\
		Do đó tứ giác $ABHD$ nội tiếp đường tròn đường kính $AB$.
		\item Kéo dài $HD, BE$ cắt nhau tại $K$.\\
		Ta có $\widehat{HDC}=90^\circ$ mà $\widehat{BEA}=90^\circ \Rightarrow \widehat{CDK}=\widehat{CEK}=90^\circ$.\\
		Nên tứ giác $CDEK$ nội tiếp $\Rightarrow \widehat{ECK}=\widehat{EDK}.$\\
		Mặt khác, do tứ giác $BHDE$ nội tiếp nên $\widehat{EDK}=\widehat{KBH}$. Do đó $\widehat{ECK}=\widehat{KBH}$.\\
		Trong tam giác $BCE$ có $\widehat{EBH}+\widehat{BCE}=90^\circ \Rightarrow \widehat{ECK}+\widehat{BCE}=90^\circ$ hay $CK \perp HC$.\\

\begin{ex}%[Đề thi vào 10, chuyên Bắc Ninh, tỉnh Bắc Ninh, năm 2018]%[Nguyễn Hiếu, Dự án EX-TS10C-18]%[9D0G8]
	Cho 20 số tự nhiên, mỗi số có ước nguyên tố không vượt quá 7. Chứng minh rằng tồn tại ít nhất hai số trong số 20 số đã cho mà tích của hai số đó là số chính phương.
		\loigiai{
		Gọi $20$ số  là $n_1$, $n_2 \ldots n_{20}$.\\
		Do ước nguyên tố của mỗi số không vượt quá 7 nên $n_i=2^{a_i}\cdot 3^{b_i}\cdot 5^{c_i}\cdot 7^{d_i}$, với $i=1,2,\ldots,20.$\\
		Các số $a_i, b_i, c_i, d_i$ nhận các giá trị chẵn hoặc lẻ, ta có 16 bộ 4 số có tính chất chẵn lẻ như vậy gồm: (chẵn, chẵn, chẵn, chẵn), (chẵn, chẵn, chẵn, lẻ),...,(lẻ, lẻ, lẻ, lẻ).\\
		Với 20 số $n_i$ ta có 20 bộ số $(a_i,b_i,c_i,d_i)$. Vậy với 20 bộ số với 16 tính chất đã nêu, theo nguyên lí Dirichlet thì tồn tại ít nhật hai bộ $n_k$ và $n_l$ với hai bộ số $(a_k,b_k,c_k,d_k), (a_l,b_l,c_l,d_l)$ tương ứng cùng tính chất chẵn lẻ như trên.\\
		Khi đó $n_k\cdot n_l=2^{a_k+a_l}\cdot 3^{b_k+b_l}\cdot 5^{c_k+c_l}\cdot 7^{d_k+d_l} =2^{2a}\cdot 3^{2b}\cdot 5^{2c}\cdot 7^{2d}$ là một số chính phương.
		}
\end{ex}
