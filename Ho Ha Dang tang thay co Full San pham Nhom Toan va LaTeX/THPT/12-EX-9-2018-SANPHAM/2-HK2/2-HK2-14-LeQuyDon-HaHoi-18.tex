	Tính tích phân $I=\displaystyle \int\limits_{0}^{1} \dfrac{x}{(x+1)^5}\mathrm{\,d}x$.
	\loigiai{
	        \begin{eqnarray*}
	        I&=&\displaystyle \int\limits_{0}^{1} \dfrac{x}{(x+1)^5}\mathrm{\,d}x\\
	        &=&\displaystyle \int\limits_{0}^{1} \dfrac{(x+1)-1}{(x+1)^5}\mathrm{\,d}x\\
	        &=&\displaystyle \int\limits_{0}^{1} \dfrac{\mathrm{\,d}x}{(x+1)^4}-\displaystyle \int\limits_{0}^{1} \dfrac{\mathrm{\,d}x}{(x+1)^5}\\
	        &=&-\dfrac{1}{3}(x+1)^{-3}\big|_0^1+\dfrac{1}{4}(x+1)^{-4}\big|_0^1\\
	        &=&-\dfrac{1}{3}(2^{-3}-1)+\dfrac{1}{4}(2^{-4}-1)=\dfrac{11}{192}.
	        \end{eqnarray*}
	}
