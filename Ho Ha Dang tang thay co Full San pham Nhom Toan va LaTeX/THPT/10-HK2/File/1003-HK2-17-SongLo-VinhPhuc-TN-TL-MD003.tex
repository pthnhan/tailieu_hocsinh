\begin{ex}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc, Mã đề 003]%[0H3K1]%[Nguyễn Hồng Điệp]
Đường thẳng đi qua điểm $M(2;1)$ và song song với đường thẳng $d:2x-4y+1=0$ có phương trình tổng quát là
	\choice	
	{\True $-x+2y=0$}
	{$2x-4y+4=0$}
	{$2x+y-3=0$}
	{$x-2y+3=0$}	
	\loigiai{
Phương trình đường thẳng $\Delta$ song song với $d$ có dạng $\Delta:2x-4y+c=0$, $c\neq 1$.\\
Do $M(2;1) \in \Delta$ nên $2.2-4.1+c=0\Leftrightarrow c=0$.\\		
Khi đó $\Delta:2x-4y=0\Leftrightarrow -x+2y=0$.\\
Vậy $\Delta:-x+2y=0$		}
\end{ex}
\begin{ex}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc, Mã đề 003]%[0D4K3]%[Nguyễn Hồng Điệp]

	{$m=2$}
	{\True $m=-2$}
	{$m=2$ hoặc $m=-2$}
	{$m=1$}	
	\loigiai{
Bất phương trình có tập nghiệm là $\mathbb{R}$ khi và chỉ khi $\heva{m^2-4=0\\m-1<0}\Leftrightarrow \heva{\hoac{&m=2\\&m=-2}\\m<1}\Rightarrow m=-2$.		
	}
\end{ex}
\begin{ex}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc, Mã đề 003]%[0H3B2]%[Nguyễn Hồng Điệp]
Cho phương trình $x^2+y^2-2x-4y+1=0 \quad (*)$. Chọn phát biểu \textbf{sai}
	\choice	
	{$(*)$ là phương trình đường tròn tâm $I(1;2)$}
	{$(*)$ là phương trình đường tròn bán kính $R=2$}
	{$(*)$ đi qua $M(1;0)$}
	{\True $(*)$ cắt trục $Ox$ tại 2 điểm}	
	\loigiai{
Ta có: $\heva{-2a=-2\\-2b=-4\\c=1}\Leftrightarrow \heva{a=1\\b=2\\c=1}$.\\
Xét $a^2-b^2-c=1^2-2^2-1=4>0\Rightarrow (*)$ là phương trình đường tròn có tâm $I(1;2)$, bán kính $R=\sqrt{4}=2$.		
\\ Ta có: $1^2+0^2-2\cdot1-4\cdot0+1=0$. Vậy $M(1;0)$ nằm trên đường tròn $(*)$.\\
Ta có: $d\left(I,Ox\right)=\dfrac{\left| 2\right|}{\sqrt{0^2+1^2}}=2=R$ nên đường tròn $(*)$ cắt trục $Ox$ tại 1 điểm (tiếp xúc).	}
\end{ex}

\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc, Mã đề 003]%[0D4B3]%[0D4B2]%[0D4K2]%[Nguyễn Hồng Điệp]
Giải
\begin{enumerate}
	\item $\dfrac{2x-5}{2-x}\ge 1$
	\item $\heva{5x+3\le \dfrac{8x-3}{3}\\10-x>\dfrac{x+1}{3}}$
	\item $|2x-3|\le x+1$
\end{enumerate}
	\loigiai{
\begin{enumerate}
	\item $\dfrac{2x-5}{2-x}\ge 1\Leftrightarrow \dfrac{2x-5}{2-x}-1\ge 0\Leftrightarrow \dfrac{3x-7}{2-x}\ge 0$.\\
	Đặt $f(x)=\dfrac{3x-7}{2-x}$.\\
	Cho $3x-7=0\Leftrightarrow x=\dfrac{7}{3}$, $2-x=0\Leftrightarrow x=2$.\\
	Bảng xét dấu:
	\begin{center}
		\begin{tikzpicture}
		\tkzTabInit
		[lgt=2,espcl=1.5,nocadre] 
		{$x$/1, $3x-7$/1, $2-x$/1 ,$f(x)$/1} 
		{$-\infty$,$2$,$\dfrac{7}{3}$,$+\infty$} 
		\tkzTabLine{, - ,t  ,- , 0 , + , } 
		\tkzTabLine{, + ,0  ,- , t , - , } 
		\tkzTabLine{, - ,d  ,+ , 0 , - , } 
		\end{tikzpicture}	
	\end{center}

	Vậy tập nghiệm là $S=\left( 2;\dfrac{7}{3} \right]$.
	\item $\heva{5x+3\le \dfrac{8x-3}{3}\\10-x>\dfrac{x+1}{3}}\Leftrightarrow \heva{15x+9\le8x-3\\30-3x>x+1}\Leftrightarrow \heva{x\le -\dfrac{12}{7}\\x<\dfrac{29}{4}}$.\\
	Vậy tập nghiệm là $S=\left( -\infty;-\dfrac{12}{7} \right]$
	\item Điều kiện $x+1\ge 0\Leftrightarrow x\ge -1$. Khi đó bất phương trình tương đương với
	$$ -(x+1)\le 2x-3\le x+1$$
	$$\Leftrightarrow \heva{ 2x-3\ge -x-1\\2x-3\le x+1}$$
	$$\Leftrightarrow \heva{x\ge \dfrac{2}{3}\\x\le 4}$$
Kết hợp với điều kiện ta được tập nghiệm của bất phương trình là $S=\left(\dfrac{2}{3};4 \right]$
\end{enumerate}	
	}
\end{bt}
\begin{bt}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc, Mã đề 003]%[0D6B2]%[Nguyễn Hồng Điệp]
Cho $\sin \alpha=-\dfrac{\sqrt{3}}{7},\pi<\alpha<\dfrac{3\pi}{2}$. Tính các giá trị lượng giác còn lại của cung $\alpha$.	
	\loigiai{
Vì $\pi<\alpha<\dfrac{3\pi}{2}\Rightarrow \cos \alpha <0$.\\
Ta có: $\cos ^2\alpha =1-\sin^2\alpha=\dfrac{46}{49}\Rightarrow \cos \alpha =-\dfrac{\sqrt{46}}{7}$.	\\
$\tan \alpha =\dfrac{\sin\alpha}{\cos \alpha}=\dfrac{\sqrt{138}}{46}$, $\cot \alpha=\dfrac{1}{\tan \alpha}=\dfrac{\sqrt{138}}{3}$.	
	}
\end{bt}
\begin{bt}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc, Mã đề 003]%[0H3B1]%[0H3K2]%[Nguyễn Hồng Điệp]
Trong mặt phẳng $Oxy$, cho $\Delta ABC$ với $A(1;2)$, $B(2;-3)$, $C(3;5)$.
\begin{enumerate}
	\item Viết phương trình tổng quát cạnh $AC$, phương trình tham số cạnh $BC$.
	\item Viết phương trình đường tròn tâm $B$ và tiếp xúc với đường thẳng $AC$.
\end{enumerate}	
	\loigiai{
\begin{enumerate}
	\item Ta có: $\vec{AC}=(2;3)$. \\
	Phương trình cạnh $AC$: $\dfrac{x-1}{2}=\dfrac{y-2}{3}\Leftrightarrow 3x-2y+1=0$.\\
	Ta có: $\vec{BC}=(1;8)$.\\
	Phương trình tham số cạnh $BC$: $\heva{&x =2+t\\&y=-3+8t}$.
	\item Bán kính đường tròn $R=d\left(B,AC\right)=\dfrac{|3.2-2.(-3)+1|}{\sqrt{3^2+(-2)^2}}=\sqrt{13}$.\\
	Phương trình đường tròn: $(C):(x-2)^2+(y+3)^2=13$.
\end{enumerate}		
	}
\end{bt}
\begin{bt}%[HK2 (2016-2017), THPT Sông Lô, Vĩnh Phúc, Mã đề 003]%[0D4K5]%[Nguyễn Hồng Điệp]
Tìm $m$ để bất phương trình sau vô nghiệm
$$m(m-4)x^2-2mx-5\ge 0\, (1)$$
	\loigiai{
\begin{itemize}
	\item Nếu $m=0$ thì bất phương trình $(1)$ tương đương $-5\ge 0$: vô nghiệm $\Rightarrow m = 0$ thỏa mãn.
	\item Nếu $m=4$ thì bất phương trình $(1)$ tương đương $-8x-5\ge 0\Leftrightarrow x\le -\dfrac{5}{8}\Rightarrow m=4$ không thỏa mãn.
	\item Nếu $m\neq 0$ và $m\neq 4$ thì bất phương trình $(1)$ vô nghiệm khi và chỉ khi
	$$\heva{a<0\\\Delta <0}\Leftrightarrow \heva{m(m-4)<0\\24m^2-80m<0}\Leftrightarrow \heva{0<m<4\\0<m<\dfrac{10}{3}}\Leftrightarrow 0<m<\dfrac{10}{3}$$.
\end{itemize}		
Vậy $0\le m<\dfrac{10}{3}$ thỏa yêu cầu bài toán.
	}
\end{bt}
