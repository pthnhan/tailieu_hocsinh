\begin{ex}%[WTT-218cau-ham-an-2D1, Lê Minh Đức]%[2D1K2-1]

	\choice
	{$3$}
	{$4$}
	{\True $5$}
	{$6$}
	\loigiai{
		Ta có $g'(x)= 2\left( x - 4 \right)f'\left( x^2 - 8x \right) = 2\left( x - 4 \right)\left[ \left( x^2 - 2x \right)^2 - 2\left( x^2 - 2x \right) \right]$.\\
		\[g'(x)= 0 \Leftrightarrow 2\left( x - 4 \right)\left[ \left( x^2 - 2x \right)^2 - 2\left( x^2 - 2x \right) \right] = 0 \Rightarrow \hoac{& x - 4 = 0\\& x^2 - 2x = 0\\& x^2 - 2x = 2} \Leftrightarrow \hoac{& x = 4\\& x = 0\\& x = 2\\& x = 1 \pm \sqrt 3}.\]

		Từ bảng biến thiên ta thấy hàm số $g(x)$ có $5$ điểm cực trị.

\begin{ex}%[WTT-218cau-ham-an-2D1, Lê Minh Đức]%[2D1K2-5]
	Cho hàm số $y = f(x)$ có đạo hàm $f'(x)= \left( x^3 - 2x^2 \right)\left( x^3 - 2x \right)$ với mọi $x \in \mathbb{R}$.\\
	Hàm số $g(x)= \left| f\left( 1 - 2018x \right) \right|$ có nhiều nhất bao nhiêu điểm cực trị?
	\choice
	{\True $9$}
	{$2018$}
	{$2022$}
	{$11$}
	\loigiai{
		Ta có $f'(x)= x^3\left( x - 2 \right)\left( x^2 - 2 \right)$.
		\[ f'(x)=0 \Leftrightarrow x^3\left( x - 2 \right)\left( x^2 - 2 \right) = 0 \Leftrightarrow \hoac{& x=0\\& x=2\\&x=-\sqrt 2\\& x= \sqrt 2}.\]
		Xét hàm số $h(x)=f(1-2018x)$, ta có $h'(x)=(1-2018x)' \cdot f'(1-2018x) = -2018 \cdot {1-2018x}^3\left( 1-2018x - 2 \right)\left( {1-2018x}^2 - 2 \right)$.
		\begin{eqnarray*}
			h'(x) = 0 & \Leftrightarrow & -2018 \cdot {1-2018x}^3\left( 1-2018x - 2 \right)\left( {1-2018x}^2 - 2 \right) =0 \\
			& \Leftrightarrow & \hoac{& 1-2018x=0\\& 1-2018x=2\\&1-2018x=-\sqrt 2\\& 1-2018x= \sqrt 2} \Leftrightarrow \hoac{& x=\dfrac{1}{2018}\\& x=\dfrac{-1}{2018}\\&x=\dfrac{1+\sqrt 2}{2018}\\& x=\dfrac{1-\sqrt 2}{2018}.}
		\end{eqnarray*}
		Các nghiệm của phương trình $h'(x)=0$ đều là các nghiệm đơn nên $h(x)$ có $4$ điểm cực trị. \hfill $(1)$\\
		Mặt khác, ta có bảng biến thiên của hàm $h(x)$
		\begin{center}
			\begin{tikzpicture}
			\tkzTabInit[espcl=3,lgt=1.5,nocadre=false]
			{$x$/1.2,$h'(x)$/0.7,$h(x)$/2.1}
			{$-\infty$,$\dfrac{-1}{2018}$,$\dfrac{1-\sqrt 2}{2018}$,$\dfrac{1}{2018}$,$\dfrac{1+\sqrt 2}{2018}$,$+\infty$}
			\tkzTabLine{,-,0,+,0,-,0,+,0,-}
			\tkzTabVar{+/$+\infty$,-/,+/,-/,+/,-/$-\infty$}
			\end{tikzpicture}
		\end{center}
		Từ bảng biến thiên ta nhận thấy phương trình $h(x)=0$ có tối đa $5$ nghiệm đơn. \hfill $(2)$\\
		Từ $(1)$ và $(2)$ suy ra hàm số $g(x)=\left| f\left( 1 - 2018x \right) \right|$ (hay $g(x)=\left| h(x) \right|$) có tối đa $4+5=9$ điểm cực trị.
	}
\end{ex}
\begin{ex}%[WTT-218cau-ham-an-2D1, Lê Minh Đức]%[2D1G2-5]
	Cho hàm số $f(x)$ có đạo hàm $f'(x)= \left( x + 1 \right)^4\left( x - 2 \right)^5 \left( x + 3 \right)^3$. Số điểm cực trị của hàm số $f\left( \left| x \right| \right)$ là
	\choice
	{$5$}
	{\True $3$}
	{$1$}
	{$2$}
	\loigiai{
			\textbf{Nhận xét. } \textit{Số điểm cực trị của hàm số $f\left( \left| x \right| \right)$ là $2a + 1$, trong đó $a$ là số điểm cực trị dương của hàm số $f(x)$.}\\
			Ta có $f'(x)= 0 \Leftrightarrow \left( x + 1 \right)^3 \left( x - 2 \right)^5 \left( x + 3 \right)^3 = 0 \Leftrightarrow \hoac{&x = - 1 \\& x = 2\\& x = - 3}$.\\
			Do $f'(x)$ chỉ đổi dấu khi $x$ đi qua $x = - 3$ và $x = 2$ nên hàm số $f(x)$ có $2$ điểm cực trị $x = - 3$ và $x = 2$ trong đó chỉ có $1$ điểm cực trị dương.\\
			Do $f\left( \left| x \right| \right) = f(x)$ nếu $x \ge 0$ và $f\left( \left| x \right| \right)$ là hàm số chẵn nên hàm số $f\left( \left| x \right| \right)$ có $3$ điểm cực trị $x = 2$, $x = 0$.
	}
\end{ex}
