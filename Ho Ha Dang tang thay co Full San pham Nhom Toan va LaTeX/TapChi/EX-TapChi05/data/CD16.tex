\section{Một số dãy số sinh bởi các hàm lượng giác}
\begin{center}
	\textbf{\color{violet}{Trương Ngọc Đắc-Trường THPT Chuyên Lê Quý Đôn, Bình Định}}
\end{center}
\noindent Trong quá trình giảng dạy cho học sinh chuyên Toán về chuyên đề dãy số, tôi nhận thấy ta có thể xây dựng dãy số xác định bởi các hàm số lượng giác nhờ sử dụng các công thức lượng giác.
Các dãy số được xác định bởi các hàm số lượng giác thường cho bởi hệ thức truy hồi khá phức tạp và các số hạng của chúng lại là những số vô tỷ càng làm cho học sinh khi giải chúng khó phát hiện ra các quy luật của dãy số.\\
\indent Ta xem xét bài toán thi học sinh giỏi Toán quốc gia năm 2014.

\subsection{Một số ví dụ về xây dựng và giải bài toán dãy số sinh bởi hàm lượng giác}
\noindent Chúng ta cùng xem xét bài toán thi học sinh giỏi quốc gia năm 2014
\begin{bt}%[Linh Nguyễn, Dự án EX-TapChi05]%[1D0G1]
	Cho hai dãy số dương $(x_n),(y_n)$ xác định bởi ${x_1} = 1,{y_1} = \sqrt 3 $ và
	\[\left\{ {\begin{array}{*{20}{c}}
		x_{n + 1}y_{n + 1} - x_n &= 0\\
		x_{n + 1}^2 + y_n &= 2
		\end{array}} \right.\]
	với mọi $n = 1,2,\ldots$\\
	Chứng minh rằng hai dãy số trên hội tụ và tìm giới hạn của chúng.
	\loigiai{
		\noindent{\bf Phân tích.} 
		Đây là một bài toán cho hai dãy số thỏa mãn một hệ thức truy hồi khá phức tạp, ta tiếp cận bằng cách tính vài giá trị đặc biệt ban đầu của hai dãy số.\\
		Ta có $x_2^2 + {y_1} = 2 \Rightarrow {x_2} = \sqrt {2 - \sqrt 3 } $ và ${x_2}{y_2} - {x_1} = 0 \Rightarrow {y_2} = \dfrac{1}{\sqrt {2 - \sqrt 3 } } = \sqrt {2 + \sqrt 3 }.$ \\
		Tương tự ta cũng có ${x_3} = \sqrt {2 - \sqrt {2 + \sqrt 3 } }$, ${y_3} = \sqrt {2 + \sqrt {2 + \sqrt 3 } }$.\\
		Từ đây ta nhận thấy $x_n^2+y_n^2=4$\quad $(*)$ đúng khi $n = 1,2,3,\ldots$.\\
		 Ta có hai cách giải sau:\\
		\noindent{\bf Lời giải 1.} 
		Ta chứng minh hệ thức $(*)$ đúng với mọi số nguyên dương $n$ bằng quy nạp.\\
		Giả sử $(*)$ đúng đến $n \ge 1$, từ công thức xác định hai dãy số và giả thiết quy nạp, ta có: \[x_{n + 1}^2 = 2 - y_n,y_{n + 1}^2 = \dfrac{x_n^2}{x_{n + 1}^2} = \dfrac{4 - y_n^2}{2 - y_n} = 2 + y_n \Rightarrow x_{n + 1}^2 + y_{n + 1}^2 = 4\]
		Vậy theo nguyên lý quy nạp $(*)$ đúng với mọi n nguyên dương.\\
		Từ chứng minh trên ta suy ra: \[{y_{n + 1}} = \sqrt {2 + y_n} ,\forall n \in \mathbb{N^*}.\tag{1}\]
		Ta chứng minh dãy số $(y_n)$ tăng và bị chặn trên bởi $2$ nên có giới hạn hữu hạn và:\[\lim\limits y_n = 2 \Rightarrow \lim\limits x_n = \lim\limits \sqrt {2 - y_{n - 1}}  = 0\]
		Ta viết $(*)$ dưới dạng: $\left( {\dfrac{x_n}{2}} \right)^2 + \left( {\dfrac{y_n}{2}} \right)^2 = 1,$ khi đó cho phép ta nghĩ đến công thức lượng giác $\sin ^2x + \cos^2x = 1\,\,\forall x \in \mathbb{R}$ và ta có lời giải $2$.\\
		\noindent{\bf Lời giải 2.} 
		Nhận thấy $x_1= 1 = 2\sin \dfrac{\pi }{6}$, ${y_1} = \sqrt 3  = 2\cos\dfrac{\pi }{6}$. Ta chứng minh bằng quy nạp rằng với mọi $n$ nguyên dương, công thức xác định dãy số đã cho là:
		\[x_n = 2\sin \dfrac{\pi }{{3\cdot 2}^n},\,y = 2\cos\dfrac{\pi }{2^n}\tag{2}\]
		Thật vậy, với $n = 1$ mệnh đề $(2)$ đúng. Giả sử mệnh đề $(2)$ đúng đến $n \geq 1$. Áp dụng công thức truy hồi, ta có:
		\begin{eqnarray*}
			&&{x_{n + 1}} = \sqrt {2 - y_n}  = \sqrt {2 - 2\cos\dfrac{\pi }{{3\cdot 2}^n}}  = \sqrt {4{{\sin }^2}\dfrac{\pi }{{3\cdot 2}^{n + 1}}}  = 2\sin \dfrac{\pi }{{3\cdot 2}^{n + 1}}.\\
			&&{y_{n + 1}} = \dfrac{{x_n}}{x_{n + 1}} = \dfrac{2\sin \dfrac{\pi }{{3\cdot 2}^n}}{2\sin \dfrac{\pi }{{3\cdot 2}^{n + 1}}} = 2 \cos\dfrac{\pi }{{3\cdot 2}^{n + 1}}.
		\end{eqnarray*}
		Theo nguyên lý quy nạp toán học, ta có $(2)$ đúng với mọi n nguyên dương.\\
		Từ đây ta có \[\lim\limits x_n = \lim\limits \left( {2\sin \dfrac{\pi }{{3\cdot 2}^n}} \right) = 0,\,\,\lim\limits \left( {2 \cos\dfrac{\pi }{{3\cdot 2}^n}} \right) = 2.\]
	}	
\end{bt}
\begin{nx}
	Hai dãy số của bài toán trên được xây dựng từ hai hàm lượng giác $\sin$ và cosin và sử dụng các công thức lượng giác:
	$\sin x = 2\sin \dfrac{x}{2}\cos\dfrac{x}{2}$, $\cos^2\dfrac{x}{2} = \dfrac{1}{2}(1 +\cos x)$
	Hoàn toàn tương tự, ta cũng có thể xây dựng được các dãy số xác định bởi các hàm số lượng giác nhờ sử dụng các công thức lượng giác.
\end{nx}

\begin{bt}%[Linh Nguyễn, Dự án EX-TapChi05]%[1D0G1]
	Cho hai dãy số dương $(x_n)$, $ (y_n)$ xác định bởi ${x_0} = \dfrac{\sqrt 2 }{2},\,{y_0} = 1$ và
	\[\left\{ {\begin{array}{*{20}{c}}
		x_n{y_{n + 1}} &=& x_{n + 1}^2\\
		x_n + 2y_nx_{n + 1}^2 &=& y_n
		\end{array}} \right.\]
	với mọi $n = 0, 1, 2,\ldots$\\
	Với mỗi số tự nhiên n, đặt ${z_n} = \displaystyle\prod_{k=0}^{n} \lim\limits\dfrac{x_i}{y_i} $. Chứng minh rằng dãy $(z_n)$ có giới hạn hữu hạn và tìm giới hạn hữu hạn đó.
	\loigiai{
		
		Ta nhận thấy ${x_0} = \dfrac{\sqrt 2 }{2} = \sin \dfrac{\pi }{{4\cdot 2}^0},\,{y_0} = 1 = \tan \dfrac{\pi }{{4\cdot 2}^0}$\\
		Từ công thức xác định dãy số, ta lại có: ${x_0} + 2{y_0}x_1^2 = {y_0} \Rightarrow {x_1} = \dfrac{\sqrt {2 - \sqrt 2 } }{2} = \sin \dfrac{\pi }{4\cdot 2}$ và 
		${y_1} = \dfrac{2x_1^2}{x_0} = \sqrt 2  - 1 = \tan \dfrac{\pi }{4\cdot 2}.$\\
		Ta chứng minh bằng quy nạp với mọi n là số tự nhiên, công thức xác định dãy số đã cho là: $x_n = \sin \dfrac{\pi }{{4\cdot 2}^n},\,\,y_n = \tan \dfrac{\pi }{{4\cdot 2}^n}\,(*)$.\\ Đặt: ${\alpha _n} = \dfrac{\pi }{{4\cdot 2}^n},\,\forall n \in \mathbb{N}.$\\
		Từ công thức truy hồi của dãy số, ta có:
		\[x_{n + 1}^2 = \dfrac{1}{2} - \dfrac{x_n}{{2y_n}} = \dfrac{1}{2} - \dfrac{\sin {\alpha _n}}{2\tan {\alpha _n}} = \dfrac{1 - \cos{\alpha _n}}{2} = {\sin ^2}\dfrac{\alpha _n}{2} \Rightarrow {x_{n + 1}} = \sin \dfrac{\alpha }{2}\]
		Và:
		
		\[{y_{n + 1}} = \dfrac{2x_{n + 1}^2}{{x_n}} = \dfrac{2{{\sin }^2}\dfrac{\alpha _n}{2}}{\sin {\alpha _n}} = \dfrac{\sin \dfrac{\alpha _n}{2}}{\cos\dfrac{\alpha _n}{2}} = \tan \dfrac{\alpha _n}{2}\]
		Theo nguyên lý quy nạp toán học, ta có $(*)$ đúng với mọi n là số tự nhiên.\\
		Vậy
		${z_0} = \dfrac{x_0}{y_0} = \dfrac{{\sqrt 2 }}{2} = \cos\dfrac{\pi }{4},\ldots,{z_n} = \dfrac{{x_0}{x_1}\cdots x_n}{{y_0}{y_1}\cdots y_n} = \cos\dfrac{\pi }{{4\cdot 2}^0} \cos\dfrac{\pi }{{4\cdot 2}^1}\ldots\,\, \cos\dfrac{\pi }{{4\cdot 2}^n}.$\\
		Áp dụng công thức lượng giác, ta có:
		\[\begin{array}{l}
		{z_n}\sin \dfrac{\pi }{{4\cdot 2}^n} = \sin \dfrac{\pi }{{4\cdot 2}^n}\cos\dfrac{\pi }{{4\cdot 2}^{n - 1}}\ldots\,\cos\dfrac{\pi }{{4\cdot 2}^0}\\
		= \dfrac{1}{2}\sin \dfrac{\pi }{{4\cdot 2}^{n - 1}}\cos\dfrac{\pi }{{4\cdot 2}^{n - 1}} = \ldots\, = \dfrac{1}{2^{n + 1}}\sin \dfrac{\pi }{2} = \dfrac{1}{2^{n + 1}}
		\end{array}\]
		Suy ra\\
		${z_n} = \dfrac{1}{{2^{n + 1}}\sin \dfrac{\pi }{{4\cdot 2}^n}}$ và $\lim\limits \,{z_n} = \lim\limits \dfrac{\dfrac{\pi}{4\cdot 2^n}}{\sin \dfrac{\pi}{4\cdot 2^n}} .\dfrac{\pi }{2} = \dfrac{\pi }{2}.$
	}
\end{bt}
\begin{bt}%[Linh Nguyễn, Dự án EX-TapChi05]%[1D0G1] 
	Cho hai dãy số dương $(x_n), (y_n)$ xác định bởi $x_0 = \sqrt{2}$ và
	\[\left\{ {\begin{array}{*{20}{c}}
		x_ny_n &= 2\sqrt {4 - x_n^2} \\
		x_n &= 2\dfrac{{4 - y_{n + 1}^2}}{4 + y_{n + 1}^2}
		\end{array}} \right.\]
	với mọi $n = 0, 1, 2,\ldots$\\
	Chứng minh rằng hai dãy số trên hội tụ và tìm giới hạn của chúng.
	\loigiai{ 
		Ta nhận thấy ${x_0} = \sqrt 2  = 2\dfrac{\sqrt 2 }{2} = 2\cos\dfrac{\pi }{{4\cdot 2}^0}$\\
		Với $n = 0 \Rightarrow {x_0}{y_0} = 2\sqrt {4 - x_0^2}  \Leftrightarrow {y_0} = 2.1 = 2\tan \dfrac{\pi }{{4\cdot 2}^0}$. Từ công thức truy hồi của dãy số, ta có:
		\[\left\{ {\begin{array}{*{20}{c}}
			{x_1}{y_1} &= 2\sqrt {4 - x_1^2} \\
			{x_0} &= 2\dfrac{{4 - y_1^2}}{y_1^2}
			\end{array} \Leftrightarrow \left\{ {\begin{array}{*{20}{c}}
				x_1^2 =& \dfrac{{16}}{4 + y_1^2}\\
				y_1^2 =& 4{{\left( {\sqrt 2  - 1} \right)}^2}
				\end{array} \Leftrightarrow \left\{ {\begin{array}{*{20}{c}}
					{x_1} &= 2\dfrac{{\sqrt {2 + \sqrt 2 } }}{2} &= 2\cos\dfrac{\pi }{4\cdot 2}\\
					{y_1} &= 2(\sqrt 2  - 1) &= 2\tan \dfrac{\pi }{{4\cdot 2}}
					\end{array}} \right.} \right.} \right.\]
		Ta chứng minh bằng quy nạp rằng với mọi $n$ là số tự nhiên, công thức xác định hai dãy số đã cho là: $x_n = 2\cos\dfrac{\pi }{{4\cdot 2}^n},\,\,y = 2\tan \dfrac{\pi }{{4\cdot 2}^n}\, (*)$\\
		Đặt ${\alpha _n} = \dfrac{\pi }{{4\cdot 2}^n},\,\forall n \in \mathbb{N}.$ Từ công thức truy hồi của dãy số, ta có:
		\[\begin{array}{l}
		\left\{ {\begin{array}{*{20}{c}}
			x_{n + 1}^2 =& \dfrac{16}{4 + y_{n + 1}^2}\\
			y_{n + 1}^2 =& 4\dfrac{{2 - x_n}}{2 + x_n}
			\end{array} \Leftrightarrow \left\{ {\begin{array}{*{20}{c}}
				x_{n + 1}^2 &= \dfrac{16}{{4 + y_{n + 1}^2}} = \dfrac{{16}}{{4 + 4{{\tan }^2}\dfrac{{\alpha _n}}{2}}} =& 4\cos{^2}\dfrac{\alpha _n}{2}\\
				y_{n + 1}^2 &= 4\dfrac{2 - 2\cos{\alpha _n}}{{2 + 2\cos{\alpha _n}}} = 4\dfrac{{{\sin }^2}\dfrac{{\alpha _n}}{2}}{\cos{^2}\dfrac{{\alpha _n}}{2}} =& 4{{\tan }^2}\dfrac{\alpha _n}{2}
				\end{array}} \right.} \right.\\
		\Leftrightarrow \left\{ {\begin{array}{*{20}{c}}
			x_n &= 2\cos\dfrac{{\alpha _n}}{2}\\
			{y_{n + 1}} &= 2\tan \dfrac{\alpha _n}{2}
			\end{array}} \right.
		\end{array}\]
		Theo nguyên lý quy nạp toán học, ta có $(*)$ đúng với mọi $n$ là số tự nhiên.\\
		Từ đây ta có $\lim\limits x_n = \lim\limits \left( {2\sin \dfrac{\pi }{{4\cdot 2}^n}} \right) = 0,\,\,\lim\limits \left( {2\tan \dfrac{\pi }{{4\cdot 2}^n}} \right) = 0.$
	}
\end{bt}

\begin{bt}%[Nguyễn Gia Minh, Dự án EX-TapChi05]%[1D0G1]
Cho hai dãy số dương $(x_n)$, $(y_n)$ xác định bởi $x_0=\sqrt{2}$, $y_0=2$ và 
\[\heva{&2x^2_{n+1}=x_ny_{n+1}\\& x^2_{n+1}y_n+2(x_n-y_n)=0}\]
với mọi $n=0,\, 1,\, 2,\ldots$\\
Chứng minh rằng hai dãy số trên hội tụ và tìm giới hạn của chúng.
\loigiai{
	Ta nhận thấy $x_0=\sqrt{2}=2\dfrac{\sqrt{2}}{2}=2\sin \dfrac{\pi}{4\cdot 2^0}$ và $y_0=2=2\tan \dfrac{\pi}{4\cdot 2^0}$.\\
	Từ công thức truy hồi của dãy số với $n=0\Rightarrow x^2_1y_0+2(x_0-y_0)=0\Leftrightarrow x^2_1=2-\dfrac{2x_0}{y_0}=2-\sqrt{2}\Rightarrow x_1=2\sin \dfrac{\pi}{4\cdot 2}$ và $y_1=\dfrac{2x^2_1}{x_0}=2(\sqrt{2}-1)=2\tan \dfrac{\pi}{4\cdot 2}$.\\
	Ta chứng minh bằng quy nạp rằng với mọi $n$ là số tự nhiên, công thức xác định hai dãy số $x_n=2\sin \dfrac{\pi}{4\cdot 2^n}$, $y_n=2\tan \dfrac{\pi}{4\cdot 2^n}.\hfill (\ast)$\\
	Đặt $\alpha_n=\dfrac{\pi}{4\cdot 2^n},\forall n\in\mathbb{N}$.\\
	Từ công thức truy hồi của dãy số, ta có 
	\begin{eqnarray*}
		\heva{& x^2_{n+1}=\dfrac{16}{4+y^2_{n+1}} \\ & y^2_{n+1}=4\cdot \dfrac{2-x_n}{2+x_n}}&\Leftrightarrow&\heva{& x^2_{n+1}=2-\dfrac{2x_n}{y_n}=2-\dfrac{4\sin \alpha_n}{2\tan \alpha_n}=2-2\cos \alpha_n=4\sin^2 \dfrac{\alpha_n}{2}\\ &y_{n+1}=2\dfrac{x^2_{n+1}}{x_n}=2\dfrac{4\sin^2\dfrac{\alpha_n}{2}}{2\sin \alpha_n}=2\tan \dfrac{\alpha_n}{2} }\\
		&\Leftrightarrow& \heva{& x_{n+1}=2\sin \dfrac{\alpha_n}{2} \\ & y_{n+1}=2\tan \dfrac{\alpha_n}{2}}
	\end{eqnarray*}
Theo nguyên lí quy nạp toán học, ta có $(\ast)$ đúng với mọi $n$ là số tự nhiên.\\
Từ đây ta có $\lim\limits x_n=\lim\limits \left(2\sin \dfrac{\pi}{4\cdot 2^n}\right)=0$, $\lim\limits y_n=\lim\limits \left(2\tan \dfrac{\pi}{4\cdot 2^n}\right)=0$.
}	
\end{bt}
\begin{bt}%[Nguyễn Gia Minh, Dự án EX-TapChi05]%[1D0G1]
Cho hai dãy số dương $\left(x_n\right)$, $\left(y_n\right)$ xác định bởi $x_0=1$ và \[\heva{& x_n=\dfrac{2y_n}{\sqrt{4+y^2_n}} \\ & 4x^2_{n+1}+x_ny^2_{n+1}=2y^2_{n+1}}\]
với mọi $n=0,\, 1,\, 2,\ldots$ Với mỗi số tự nhiên $n$, đặt $z_n=\dfrac{x_n}{y_n}$.\\
Chứng minh rằng dãy $\left(z_n\right)$ có giới hạn hữu hạn và tìm giới hạn đó.	
	\loigiai{
	Ta nhận thấy $x_0=1=2\sin \dfrac{\pi}{3}$. \\
	Với $n=0\Rightarrow x_0\sqrt{4+y^2_0}=2y_0\Leftrightarrow y^2_0=\dfrac{4}{3}\Rightarrow y_0=2\dfrac{1}{\sqrt{3}}=2\cot \dfrac{\pi}{3}$.\\
	Từ công thức truy hồi của dãy số, ta có 
	$$\heva{& x_1\sqrt{4+y_1^2}=2y_1 \\ & 4x^2_1+y^2_1=2y^2_1}\Leftrightarrow \heva{& y_1=2x_1\\ & y^2_1=12}\Leftrightarrow \heva{& x_1=\sqrt{3}=2\cos \dfrac{\pi}{3\cdot 2} \\ & y_1=2\sqrt{3}=2\cot \dfrac{\pi}{3\cdot 2}.}$$
	Ta chứng minh bằng quy nạp rằng với mọi số $n$ là số tự nhiên, công thức xác định hai dãy số đã cho là $x_n=2\cos \dfrac{\pi}{3\cdot 2^n}$, $y_n=2\cot \dfrac{\pi}{3\cdot 2^n}$. $\hfill (\ast)$\\
	Đặt $\alpha_n=\dfrac{\pi}{3\cdot 2^n},\, \forall n\in\mathbb{N}$.\\
	Từ công thức truy hồi của dãy số, ta có \\
	\begin{eqnarray*}
		&&\heva{& x_{n+1}\sqrt{4+y^2_{n+1}}=2y_{n+1} \\ & 4x^2_{n+1}+2\cos \alpha_ny^2_{n+1}=2y^2_{n+1}}\Leftrightarrow \heva{& x^2_{n+1}\left(4+y^2_{n+1}\right)=4y^2_{n+1} \\ & x^2_{n+1}=\dfrac{1}{2}\left(1-\cos \alpha_n\right)y^2_{n+1}=\sin^2 \dfrac{\alpha_n}{2}y^2_{n+1}}\\
		&\Leftrightarrow& \heva{& \sin^2 \dfrac{\alpha_n}{2}\left(4+y^2_{n+1}\right)=4 \\ & x^2_{n+1}=\sin^2 \dfrac{\alpha_n}{2}y^2_{n+1}}\Rightarrow \heva{& y^2_{n+1}=4\dfrac{1-\sin^2\dfrac{\alpha_n}{2}}{\sin^2\dfrac{\alpha_n}{2}}  \\ & x^2_{n+1}=\sin^2\dfrac{\alpha_n}{2}y^2_{n+1}}\Rightarrow \heva{& x_{n+1}=2\cos \dfrac{\alpha_n}{2} \\ & y_{n+1}=2\cot \dfrac{\alpha_n}{2}.}
	\end{eqnarray*}
Vậy $x_n=2\cos \dfrac{\pi}{3\cdot 2^n}$, $y_n=2\cot \dfrac{\pi}{3\cdot 2^n}$.\\
Theo nguyên lí quy nạp toán học, ta có $(\ast)$ đúng với mọi số tự nhiên $n$.\\
Vậy $z_n=\dfrac{x_n}{y_n}=\sin \dfrac{\pi}{3\cdot 2^n}\Rightarrow \lim\limits z_n=0$.
}
\end{bt}
\begin{bt}%[Nguyễn Gia Minh, Dự án EX-TapChi05]%[1D0G1]
Cho ba dãy số dương $\left(x_n\right)$, $\left(y_n\right)$, $\left(z_n\right)$ xác định bởi $x_0=\sqrt{3}$, $y_0=1$ và 
$$\heva{& x_n=y_nz_n \\ & x_n=x_{n+1}y_{n+1}\\&x^2_{n+1}+y_n=2.}$$
với mọi $n=0,\, 1,\, 2,\ldots$\\
Chứng minh rằng ba dãy trên hội tụ và tìm giới hạn của chúng.
	\loigiai
	{
	Ta nhận thấy $x_0=\sqrt{3}=2\sin \dfrac{\pi}{3}$, $y_0=1=2\cos \dfrac{\pi}{3}$ và $z_0=\dfrac{x_0}{y_0}=\sqrt{3}=\tan \dfrac{\pi}{3}$.\\
	Từ công thức xác định dãy số, ta lại có
	$$\heva{& x_0=x_1y_1 \\ & x^2_1=2-y_0}\Leftrightarrow \heva{& x_1y_1=\sqrt{3} \\ & x^2_1=2-1}\Leftrightarrow \heva{&x_1=1=2\sin \dfrac{\pi}{3\cdot 2}\\&y_1=\sqrt{3}=2\sin \dfrac{\pi}{3\cdot 2}}$$
	và $z_1=\dfrac{x_1}{y_1}=\dfrac{1}{\sqrt{3}}=\tan \dfrac{\pi}{3\cdot 2}$.\\
	Ta chứng minh rằng với mọi số $n$ là số tự nhiên, công thức xác định ba dãy số đã cho là \\
	$x_n=2\sin \dfrac{\pi}{3\cdot 2^n}$, $y_n=2\cos \dfrac{\pi}{3\cdot 2^n}$, $z_n=\tan \dfrac{\pi}{3\cdot 2^n}$. $\hfill (\ast)$\\
	Đặt $\alpha_n=\dfrac{\pi}{3\cdot 2^n}$ với $\forall n\in\mathbb{N}$.\\
	Với $n=0$ thì $(\ast)$ đúng. Giả sử $(\ast)$ đúng với $n\geq 1$, ta phải chứng minh $(\ast)$ đúng với $n+1$.\\
