\begin{bt}%[Đề HK1, Ten Lơ Man, Hồ Chí Minh 17-18]%[Trần Tuấn Việt, dự án 10-HKI - 2018]%[0D3Y2-4]%[0D3Y2-2]%[0D3B2-4]
Giải các phương trình sau:
\begin{enumerate}
\item $\sqrt{5 - x} - x = 1$.
\item $x = \left|x^2 - x + 2\right| - 2$.
\item $\sqrt{2 + x + x^2} - \sqrt{5 - x} = 0$.
\end{enumerate}
\loigiai{\begin{enumerate}
\item Điều kiện xác định: $5 - x \geq 0$ $\Leftrightarrow$ $x \leq 5$.\\
Phương trình đã cho tương đương
\begin{align*}
\sqrt{5 - x} = x + 1 \Leftrightarrow &\heva{&x + 1 \geq 0\\&5 - x = (x + 1)^2} \Leftrightarrow \heva{&-1 \leq x \leq 5\\&x^2 + 3x - 4 = 0} \Leftrightarrow \heva{&-1 \leq x \leq 5\\&x = 1\\&x = -4} \Leftrightarrow x  = 1.
\end{align*}
Vậy phương trình đã cho có một nghiệm $x = 1$.
\item Do $x^2 - x + 2 = \left(x - \dfrac{1}{2}\right)^2 + \dfrac{7}{4} > 0$, phương trình đã cho tương đương
$$ x = (x^2 - x + 2) - 2 \Leftrightarrow x^2 - 2x = 0 \Leftrightarrow x(x - 2) = 0 \Leftrightarrow \hoac{&x = 0\\&x = 2}.$$
\noindent Vậy phương trình đã cho có hai nghiệm $x = 0$ và $x = 2$.
\item Điều kiện xác định: $\heva{&x^2 + x + 2 \geq 0\\&5 - x \geq 0} \Leftrightarrow x \leq 5$.\\
Phương trình đã cho tương đương
$$ \sqrt{x^2 + x + 2} = \sqrt{5 - x} \Leftrightarrow x^2 + x + 2 = 5 - x \Leftrightarrow x^2 + 2x - 3 = 0 \Leftrightarrow \heva{&x = 1\\&x = -3} (\text{TMĐK}).$$
\noindent Vậy phương trình đã cho có hai nghiệm là $x = 1, x = -3$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[Đề HK1, Ten Lơ Man, Hồ Chí Minh 17-18]%[Trần Tuấn Việt, dự án 10-HKI - 2018]%[0D3B2-5]
Cho phương trình $x^2 + 2x + 1 - m = 0,$ với $m$ là tham số. Định $m$ để phương trình đã cho có đúng hai nghiệm phân biệt $x_1, x_2$ thỏa: $x_1 + x_2 = x_1 x_2 + 15.$
\loigiai{Phương trình đã cho có hai nghiệm phân biệt khi và chỉ khi $\Delta = 4m > 0 \Leftrightarrow m > 0$. Gọi $x_1, x_2$ là hai nghiệm của phương trình, theo định lý Vi-ét thì $\heva{&x_1 + x_2 = -2\\&x_1 x_2 = 1 - m}$. Khi đó, ta có
$$x_1 + x_2 = x_1 x_2 + 15 \Leftrightarrow -2 = 1 - m + 15 \Leftrightarrow m = 18 \text{ (thỏa mãn)}.$$
Vậy giá trị của $m$ cần tìm là $m = 18$.
}
\end{bt}

\begin{bt}%[Đề HK1, Ten Lơ Man, Hồ Chí Minh 17-18]%[Trần Tuấn Việt, dự án 10-HKI - 2018]%[0D2B3-2]
Cho parabol $(P): y = x^2 + bx + c$ $(b \neq 0)$.\\
Tìm parabol $(P)$ biết $(P)$ luôn đi qua điểm $M(1; 1)$ và có trục đối xứng $x = -1$.
\loigiai{$\bullet$ Parabol $(P)$ đi qua điểm $M(1; 1)$ nên: $b + c = 0.$ \hspace{\fill} ($1$)\\
$\bullet$ Parabol $(P)$ có trục đối xứng $\dfrac{-b}{2} = -1 \Leftrightarrow b = 2$. Thế vào ($1$) ta được: $c = -2$.\\
Vậy $(P): y = x^2 + 2x - 2$. 
}
\end{bt}

\begin{bt}%[Đề HK1, Ten Lơ Man, Hồ Chí Minh 17-18]%[Trần Tuấn Việt, dự án 10-HKI - 2018]%[0D2B3-4]
Tìm tọa độ giao điểm của đường thẳng $d: x + y - 2 = 0$ và $(P): y = x^2$ (biết hoành độ $x \in \mathbb{N}$).
\loigiai{Phương trình hoành độ giao điểm: $- x + 2 = x^2 \Leftrightarrow x^2 + x - 2 = 0 \Leftrightarrow \hoac{&x = 1 \text{ (thỏa mãn } x \in \mathbb{N}\text{)}\\&x = -2 \text{ (không thỏa mãn } x \in \mathbb{N}\text{)}}$.\\
$\bullet$ Tại $x = 1, y = 1$.\\
Vậy $d$ và $(P)$ giao nhau tại điểm có hoành độ $x \in \mathbb{N}$ là $(1; 1)$.}
\end{bt}

\begin{bt}%[Đề HK1, Ten Lơ Man, Hồ Chí Minh 17-18]%[Trần Tuấn Việt, dự án 10-HKI - 2018]%[0D3K2-4]
Tìm các giá trị $x$ nguyên dương thỏa mãn $x + \dfrac{1}{x - 1} + \sqrt{x - 1} + \dfrac{1}{\sqrt{x - 1}} = 5$.
\loigiai{Điều kiện xác định: $x - 1 > 0 \Leftrightarrow x > 1$.\\
Phương trình đã cho tương đương $(x - 1) + \dfrac{1}{x - 1} + \sqrt{x - 1} + \dfrac{1}{\sqrt{x - 1}} = 4$. \hspace{\fill} ($1$)\\
Đặt $\sqrt{x - 1} = t$ ($t > 0$), phương trình $(1)$ trở thành\\
$\begin{array}{lccl}
& \quad \quad \left( t^2 + \dfrac{1}{t^2}\right) + \left(t + \dfrac{1}{t}\right) - 4 &=& 0\\
&\Leftrightarrow \left(t + \dfrac{1}{t}\right)^2 + \left(t + \dfrac{1}{t}\right) - 6 &=& 0\\
& \Leftrightarrow \left(t + \dfrac{1}{t} + 2\right)\left(t + \dfrac{1}{t} - 3\right) &=& 0
\end{array}$\medskip

$\hspace{2pt} \Leftrightarrow \hoac{&t + \dfrac{1}{t} = -2 \text{ (loại vì }t>0 \text{)}\\&t + \dfrac{1}{t} = 3} \Leftrightarrow t^2 - 3t + 1 = 0 \Leftrightarrow \hoac{&t = \dfrac{3 + \sqrt{5}}{2} \text{ (TMĐK)}\\&t = \dfrac{3 - \sqrt{5}}{2}\text{ (TMĐK)}}$.\\
$\bullet$ Với $t = \dfrac{3 + \sqrt{5}}{2}$, ta có $\sqrt{x - 1} = \dfrac{3 + \sqrt{5}}{2} \Leftrightarrow x = \dfrac{9 + 3\sqrt{5}}{2}$.\\
$\bullet$ Với $t = \dfrac{3 - \sqrt{5}}{2}$, ta có $\sqrt{x - 1} = \dfrac{3 - \sqrt{5}}{2} \Leftrightarrow x = \dfrac{9 - 3\sqrt{5}}{2}$.\\
Vậy phương trình đã cho có hai nghiệm là $x = \dfrac{9 - 3\sqrt{5}}{2}$ và $x = \dfrac{9 + 3\sqrt{5}}{2}$.}
\end{bt}

\begin{bt}%[Đề HK1, Ten Lơ Man, Hồ Chí Minh 17-18]%[Trần Tuấn Việt, dự án 10-HKI - 2018]%[0H2B3-1]
Cho tam giác $ABC$. Biết $AB = 7$ cm, $AC = 8$ cm, $BC = 5$ cm. Tính số đo góc $C$ của tam giác $ABC$.
\loigiai{Xét tam giác $ABC$, ta có\\
$AB^2 = AC^2 + BC^2 - 2AC \cdot BC \cdot \cos C \Leftrightarrow 7^2 = 8^2 + 5^2 - 80 \cdot \cos C \Leftrightarrow \cos C = \dfrac{7^2 - 8^2 - 5^2}{-80} = \dfrac{1}{2}\\
\Rightarrow C = 60^\circ$.}
\end{bt}

\begin{bt}%[Đề HK1, Ten Lơ Man, Hồ Chí Minh 17-18]%[Trần Tuấn Việt, dự án 10-HKI - 2018]%[0H3Y1-6]%[0H3B1-6]%[0H2B3-3]
Trong mặt phẳng tọa độ $Oxy$, cho tam giác $ABC$ biết $A(2; 5), B(6; 2), C(6; 5), K(m; m+3)$, (với $m$ nguyên dương).
\begin{enumerate}
\item Tìm tọa độ trọng tâm $G$ của tam giác $ABC$.
\item Tìm tọa độ điểm $K$ sao cho $AK = 2\sqrt{2}$.
\item Chứng minh rằng tam giác $ABC$ vuông. Tính diện tích tam giác $ABC$.
\end{enumerate} 
\loigiai{\begin{enumerate}
\item Ta có $\heva{&x_G = \dfrac{x_A + x_B + x_C}{3}\\&y_G = \dfrac{y_A + y_B + y_C}{3}} \Leftrightarrow \heva{&x_G = \dfrac{2 + 6 + 6}{3}\\&y_G = \dfrac{5 + 2 + 5}{3}} \Leftrightarrow \heva{&x_G = \dfrac{14}{3}\\&y_G = 4}.$\\
Vậy tọa độ trọng tâm $G$ của tam giác $ABC$ là $G\left(\dfrac{14}{3}; 4\right)$.
\item $AK = 2\sqrt{2} \Leftrightarrow \sqrt{(m - 2)^2 + (m + 3 - 5)^2} = 2\sqrt{2} \Leftrightarrow 2(m - 2)^2 = 8 \Leftrightarrow (m - 2)^2 = 4 \\ \Leftrightarrow \hoac{&m - 2 = 2\\&m - 2 = -2} \Leftrightarrow \hoac{& m = 4\\& m = 0 \text{ (loại vì } m \text{ nguyên dương)}}$.\\
$\bullet$ Với $m = 4$, ta có $K(4; 7)$.\\
Vậy tọa độ điểm $K$ cần tìm là $K(4; 7)$.
\item $\bullet$ Ta có $AB  = \sqrt{(6 - 2)^2 + (2 - 5)^2} = \sqrt{3^2 + 4^2} = 5$. \\
Tương tự ta cũng có $BC = 3, AC = 4$.\\
Suy ra, $AB^2 = BC^2 + AC^2$ nên $ABC$ là tam giác vuông tại đỉnh $B$. \\
$\bullet$ Diện tích tam giác $ABC$ là $S_{ABC} = \dfrac{1}{2} BC \cdot AB = \dfrac{1}{2} \cdot 4 \cdot 3 = 6$.\\
Vậy tam giác vuông $ABC$ có diện tích bằng $6$ (đvdt).
\end{enumerate}
}
\end{bt}
