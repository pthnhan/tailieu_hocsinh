\section{PHƯƠNG TRÌNH VÀ HỆ PHƯƠNG TRÌNH BẬC NHẤT NHIỀU ẨN}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Phương trình bậc nhất hai ẩn}
\begin{kn}
Phương trình bậc nhất hai ẩn $x, y$ có dạng tổng quát là 
$$ax+by=c\quad (1),$$
trong đó $a,b,c$ là các hệ số, với điều kiện $a$ và $b$ không đồng thời bằng $0$. 
\end{kn}
\begin{note}
\begin{enumerate}
\item Nếu $a=b=c=0$ thì (1) có vô số nghiệm (mọi cặp số $(x_0;y_0)$ đều là nghiệm).
\item Nếu $a=b=0, c\neq 0$ thì (1) vô nghiệm.
\item Nếu $b\neq 0$ thì (1) có dạng $y=-\dfrac{a}{b}x+\dfrac{c}{b}$ (d). Khi đó $(x_0;y_0)$ là nghiệm của (1)$\Leftrightarrow M(x_0;y_0)$ thuộc đường thẳng (d). 
\end{enumerate}
Tổng quát, phương trình bậc nhất hai ẩn $ax+by=c, (a^2+b^2\neq 0)$ luôn có vô số nghiệm. Biểu diễn hình học tập nghiệm của phương trình là một đường thẳng trong mặt tọa độ $Oxy$. 
\end{note}
\subsubsection{Hệ hai phương trình bậc nhất hai ẩn}
\begin{kn}
Hệ hai phương trình bậc nhất hai ẩn có dạng tổng quát là 
$$\heva{& a_1x+b_1y=c_1\\& a_2x+b_2y=c_2.} (2),$$
trong đó $x,y$ là hai ẩn; các chữ còn lại là hệ số. Nếu cặp số $(x_0;y_0)$ đồng thời là nghiệm của cả hai phương trình của hệ thì $(x_0;y_0)$ được gọi là nghiệm của hệ phương trình (2). Giải hệ phương trình (2) là tìm tập nghiệm của nó.
\end{kn}
\subsubsection{Hệ ba phương trình bậc nhất ba ẩn}
\begin{kn}
Hệ ba phương trình bậc nhất ba ẩn có dạng tổng quát là 
$$\heva{& a_1x+b_1y+c_1z=d_1\\& a_2x+b_2y+c_2z=d_2\\& a_3x+b_3y+c_3z=d_3.} (3),$$
trong đó $x,y,z$ là ba ẩn; các chữ còn lại là hệ số. Nếu bộ ba số $(x_0;y_0;z_0)$ nghiệm đúng cả ba phương trình của hệ thì $(x_0;y_0;z_0)$ được gọi là nghiệm của hệ phương trình (3).
\end{kn}
\subsection{Các dạng toán}
\begin{dang}{Giải hệ hai phương trình bậc nhất hai ẩn bằng phương pháp thế hoặc phương pháp  cộng đại số}
\textbf{1. Quy tắc thế}\\
Quy tắc thế dùng để biến đổi một hệ phương trình thành hệ phương trình tương đương. Quy tắc thế gồm hai bước sau:
\begin{itemize}
\item Bước 1: Từ một phương trình của hệ đã cho (coi là phương trình thứ nhất), ta biểu diễn một ẩn theo ẩn kia rồi thế vào phương trình thứ hai để được một phương trình mới (chỉ còn một ẩn).
\item Bước 2: Dùng phương trình mới để thay thế cho phương trình thứ hai trong hệ (và giữ nguyên phương trình thứ nhất).
\end{itemize}
\textbf{Tóm tắt cách giải hệ phương trình bằng phương pháp thế}
\begin{itemize}
\item Bước 1: Dùng quy tắc thế biến đổi hệ phương trình đã cho để được một hệ phương trình mới, trong đó có một phương trình một ẩn.
\item Bước 2: Giải phương trình một ẩn vừa có, rồi suy ra nghiệm của hệ đã cho.\\
\textbf{Chú ý}: Nếu thấy xuất hiện phương trình có các hệ số của hai ẩn đểu bằng 0 thì hệ phương trình đã cho có thể có vô số nghiệm hoặc vô nghiệm.
\end{itemize}
\textbf{2. Quy tắc cộng đại số}\\
Quy tắc cộng đại số dùng để biến đổi một hệ phương trình thành hệ phương trình tương đương. Quy tắc cộng đại số gồm hai bước:
\begin{itemize}
\item Bước 1: Cộng hay trừ từng vế hai phương trình của hệ phương trình đã cho để được một phương trình mới.
\item Bước 2: Dùng phương trình mới ấy thay thế cho một trong hai phương trình của hệ (và giữ nguyên phương trình kia).
\end{itemize}
\textbf{Tóm tắt cách giải hệ phương trình bằng phương pháp cộng đại số}
\begin{itemize}
\item Bước 1: Nhân các vế của hai phương trình với một số thích hợp (nếu cần) sao cho các hệ số của một ẩn nào đó trong hai phương trình của hệ bằng nhau hoặc đối nhau.
\item Bước 2: Sử dụng quy tắc cộng đại số để được hệ phương trình mới, trong đó có một phương trình mà hệ số của một trong hai ẩn bằng 0 (tức là phương trình một ẩn).
\item Bước 3: Giải phương trình một ẩn vừa thu được rồi suy ra nghiệm của hệ đã cho.
\end{itemize}
\end{dang}

\begin{vd}%[Nguyễn Thị Huyền My]%[0D3B3]
Giải hệ phương trình $\heva{& 3x-y=1\\ &2x-3y=7.}$
\loigiai{
Hệ phương trình tương đương với $$\heva{&y=3x-1\\& 2x-3(3x-1) = 7.} \Leftrightarrow \heva{&y=3x-1\\&-7x+3=7.}\Leftrightarrow \heva{& y=3x-1\\& x=-\dfrac{4}{7}.} \Leftrightarrow \heva{&x=-\dfrac{4}{7} \\ & y=-\dfrac{19}{7}.}$$
Vậy hệ phương trình có nghiệm là $(x;y)= \left(-\dfrac{4}{7};-\dfrac{19}{7}\right)$.
}
\end{vd}

\begin{vd}%[Nguyễn Thị Huyền My]%[0D3K3]
Giải hệ phương trình  $\heva{& 2\sqrt{x+5}+\sqrt{y-8} =11\\&5\sqrt{x+5}-4\sqrt{y-8}=8 .}$
\loigiai{ Điều kiện: $\heva{& x+5\geq 0 \\& y-8 \geq 0 . } \Leftrightarrow \heva{& x \geq -5 \\& y\geq 8.}$.
Đặt $\heva{& u=\sqrt{x+5}\\& v=\sqrt{y-8}.}, u,v \geq 0$. Hệ phương trình theo $u,v$: $$\heva{& 2u+v=11 \\& 5u-4v=8 .} \Leftrightarrow \heva{& v=11-2u\\& 5u-4(11-2u) =8.} \Leftrightarrow \heva{& v=11-2.4=3\\& u=4 .}$$
Suy ra $\heva{& \sqrt{x+5}=4\\& \sqrt{y-8} =3 .} \Leftrightarrow \heva{& x+5=16\\& y-8=9.} \Leftrightarrow \heva{& x=11 \\& y=17.}$ (thỏa điều kiện)
\\
Vậy hệ phương trình có nghiệm là $(x;y)= (11;17)$.
}
\end{vd}

\begin{vd}%[Nguyễn Thị Huyền My]%[0D3K3]
Ngày sinh nhật của cô giáo $A$ gồm hai chữ số, biết rằng tổng của hai chữ số đó là $8$. Nếu viết ngày sinh nhật theo thứ tự ngược lại thì được một số bằng $4$ lần số ban đầu cộng thêm $3$. Vậy ngày sinh nhật của cô giáo $A$ là bao nhiêu?
\loigiai{Gọi ngày sinh nhật là $\overline{ab}$ với $a, b \in \mathbb{N}, 10 \leq \overline{ab}\leq 31$.
\\
Theo đề bài ta có hệ phương trình
$$\heva{&a+b=8 \\& \overline{ba} = 4\overline{ab} +3.} \Leftrightarrow  \heva{& b=8-a\\ &10b+a = 4(10a+b)+3.} \Leftrightarrow \heva{& b=8-a\\ & 39a-6(8-a) =-3.} \Leftrightarrow \heva{&a=1\\&b=7.} $$
Vậy ngày sinh nhật của cô giáo $A$ là $17$.
}
\end{vd}

\begin{vd}%[Nguyễn Thị Huyền My]%[0D3B3]
Giải hệ phương trình $\heva{& 2x+5y=8 \\ & 2x-3y=0.}$
\loigiai{
Lấy phương trình thứ nhất trừ với phương trình thứ hai ta được $8y=8$. Do đó hệ phương trình trên trở thành $$\heva{&2x+5y=8\\ &8y=8.}\Leftrightarrow \heva{& x=\dfrac{3}{2} \\ &y=1.}$$
Vậy hệ phương trình có nghiệm là $(x;y)= \left(\dfrac{3}{2};1\right)$.
}
\end{vd}

\begin{vd}%[Nguyễn Thị Huyền My]%[0D3K3]
Nếu đem $\dfrac{1}{5}$ số trâu và $\dfrac{1}{3}$ số bò gộp lại thì được $25$ con. Nếu đem $\dfrac{2}{5}$ số trâu và $\dfrac{1}{4}$ số bò gộp lại thì được $30$ con. Tính số trâu và số bò.
\loigiai{
Gọi $x$ và $y$ lần lượt là số trâu và số bò ($x,y$ là các số nguyên dương). Ta có
$$\heva{& \dfrac{1}{5}x+\dfrac{1}{3}y=25 & (1)\\ & \dfrac{2}{5}x+\dfrac{1}{4}y=30 & (2)}\Leftrightarrow \heva{& \dfrac{1}{5}x+\dfrac{1}{3}y=25 \\ & \dfrac{5}{12}y=20. & \text{(Lấy (1) nhân với 2 rồi trừ cho (2))}}$$
Từ đó suy ra $x=45, y=48$. Vậy có $45$ con trâu và $48$ con bò.
}
\end{vd}

\begin{bt}%[Nguyễn Thị Huyền My]%[0D3B3]
Giải hệ phương trình $\heva{&(\sqrt{3}+1)x+y=\sqrt{3}-1\\ & 2x-(\sqrt{3}-1)y=2\sqrt{3}.}$
\loigiai{
Hệ phương trình tương đương với $$\heva{&y=-(\sqrt{3}+1)x+\sqrt{3} - 1\\& 2x+(\sqrt{3}-1)(\sqrt{3}+1)x- (\sqrt{3}-1)^2 =2\sqrt{3}.} \Leftrightarrow \heva{& y=-(\sqrt{3}+1)x+\sqrt{3} - 1\\& 2x+2x=2\sqrt{3}+(\sqrt{3}-1)^2.}$$
$$ \Leftrightarrow \heva{& y=-(\sqrt{3}+1).\dfrac{\sqrt{3}-1}{4}+\sqrt{3} - 1	\\& x= 1.}\Leftrightarrow \heva{&x= 1	\\& y=-2.}$$
Vậy hệ phương trình có nghiệm là $(x;y)= (1;-2)$.
}
\end{bt}

\begin{bt}%[Nguyễn Thị Huyền My]%[0D3B3]
Giải hệ phương trình $\heva{& 2x+3y=1\\ & 3x-4y=10.}$
\loigiai{
Lấy phương trình thứ nhất nhân với $3$, phương trình thứ hai nhân với $2$ ta được 
\begin{align*}
& \heva{& 6x+9y=3 & (a)\\ & 6x-8y=20 & (b)} \\
\Leftrightarrow & \heva{& 6x+9y=3 \\ & 17y=-17. & \text{(Lấy $(a)$ trừ cho $(b)$)}}\\
\Leftrightarrow& \heva{& x=2\\ & y=-1.}
\end{align*}
Vậy hệ phương trình có nghiệm là $(x;y)=  \left(2;-1\right)$.
}
\end{bt}

\begin{bt}%[Nguyễn Thị Huyền My]%[0D3K3]
Giải hệ phương trình $\heva{& \dfrac{10}{4x-y}+\dfrac{1}{x+y}=1\\& \dfrac{25}{4x-y}+\dfrac{3}{x+y}=2 .}$

\loigiai{
Điều kiện: $\heva{& 4x-y\neq 0 \\& x+y\neq 0.}$.
Đặt $\heva{& u =\dfrac{1}{4x-y}\\& v=\dfrac{1}{x+y}.}$. Hệ phương trình theo $u,v$:
$$\heva{&10u+v=1 \\&25u+3v=2 .} \Leftrightarrow \heva{& v=1-10u \\& 25u+3(1-10u)=2.} \Leftrightarrow \heva{& v=1\\& u=\dfrac{1}{5}.}$$
Suy ra $\heva{& \dfrac{1}{4x-y}=\dfrac{1}{5} \\& \dfrac{1}{x+y}=1.} \Leftrightarrow \heva{& 4x-y=5 \\&x+y=1.} \Leftrightarrow \heva{& y=4x-5\\& x+4x-5=1.} \Leftrightarrow \heva{&  x=\dfrac{6}{5}\\&y=-\dfrac{1}{5}.}$
\\
Vậy hệ phương trình có nhiệm là $(x;y)=  \left(\dfrac{6}{5} ;-\dfrac{1}{5} \right)$.
}
\end{bt}

\begin{bt}%[Nguyễn Thị Huyền My]%[0D3B3]
Giải hệ phương trình $\heva{& 0,4x-0,3y=0,6\\ & 0,3x-0,2y=1,3.}$
\loigiai{
Lấy phương trình thứ nhất nhân với $3$, phương trình thứ hai nhân với $4$ ta được 
\begin{align*}
& \heva{&1,2x-0,9y=1,8 \\ & 1,2x-0,8y=5,2 .}\\
\Leftrightarrow & \heva{& 1,2x-0,9y=1,8\\ & -0,1y=-3,4 .& \text{(Lấy phương trình trên trừ phương trình dưới)}}\\
\Leftrightarrow&\heva{&x=27\\&y=34.}
\end{align*}
Vậy hệ phương trình có nghiệm là $(x;y)=(27;34)$.
}
\end{bt}

\begin{bt}%[Nguyễn Thị Huyền My]%[0D3B3]
Giải hệ phương trình $\heva{& \dfrac{3}{4}x+\dfrac{3}{7}y=\dfrac{3}{5}\\ & \dfrac{2}{5}x+\dfrac{2}{7}y=\dfrac{1}{3}.}$
\loigiai{
Lấy phương trình thứ nhất nhân với $\dfrac{2}{5}$, phương trình thứ hai nhân với $\dfrac{3}{4}$ ta được
\begin{align*}
& \heva{& \dfrac{3}{10}x+\dfrac{6}{35}y=\dfrac{6}{25}\\ & \dfrac{3}{10}x+\dfrac{3}{14}y=\dfrac{1}{4}.}\\
\Leftrightarrow&\heva{&\dfrac{3}{10}x+\dfrac{6}{35}y=\dfrac{6}{25}\\ & -\dfrac{3}{70}y=-\dfrac{1}{100}.&\text{(Lấy phương trình trên trừ phương trình dưới)}}\\
\Leftrightarrow&\heva{&x=\dfrac{2}{3}\\&y=\dfrac{7}{30}.}
\end{align*}
Vậy hệ phương trình có nghiệm là $(x;y)=\left(\dfrac{2}{3};\dfrac{7}{30}\right)$.
}
\end{bt}

\begin{bt}%[Nguyễn Thị Huyền My]%[0D3K3]
Giải hệ phương trình $\heva{& \dfrac{1}{3x}+\dfrac{3}{5y}=4 \\ & \dfrac{3}{2x}+\dfrac{1}{5y}=2.}$
\loigiai{
Điều kiện: $x\neq 0, y\neq 0$.
Ta xem đây là hệ phương trình bậc nhất theo $X=\dfrac{1}{x}$ và $Y=\dfrac{1}{y}$.\\
Nhân phương trình thứ nhất với $\dfrac{3}{2}$, phương trình thứ hai với $\dfrac{1}{3}$ ta được 
\begin{center}
$\heva{&\dfrac{1}{2}X+\dfrac{9}{10}Y=6\\ & \dfrac{1}{2}X+\dfrac{1}{15}Y=\dfrac{2}{3}.}\Leftrightarrow\heva{&\dfrac{1}{2}X+\dfrac{9}{10}Y=6 \\ & \dfrac{5}{6}Y=\dfrac{16}{3}.}\Leftrightarrow\heva{&X=\dfrac{12}{25}\\&Y=\dfrac{32}{5}.}\Leftrightarrow\heva{&x=\dfrac{25}{12}\\&y=\dfrac{5}{32}.}$
\end{center}
Vậy hệ phương trình có nghiệm là $(x;y)=\left(\dfrac{25}{12};\dfrac{5}{32}\right)$.
}
\end{bt}

\begin{bt}%[Nguyễn Thị Huyền My]%[0D3K3]
Cho đường thẳng $d:y=ax+b\ (a\neq 0)$. Viết phương trình của $d$ biết nó đi qua hai điểm $A(3;2), B(1;-2)$.
\loigiai{
Vì $d$ đi qua $A$ và $B$ nên tọa độ của $A$ và $B$ thỏa mãn phương trình của $d$. Ta có hệ phương trình $$\heva{& 3a+b=2\\ &a+b=-2.}\Leftrightarrow\heva{& a=2\\ & b=-4.}$$
Vậy $d:y=2x-4$.
}
\end{bt}

\begin{bt}%[Nguyễn Thị Huyền My]%[0D3K3]
Một bài kiểm tra có $15$ câu hỏi. Mỗi câu trả lời đúng được cộng $5$ điểm. Mỗi câu trả lời sai hoặc bỏ trống bị trừ $5$ điểm. Một học sinh làm bài kiểm tra và đạt $25$ điểm. Hỏi bạn đó đã trả lời đúng bao nhiêu câu?
\loigiai{
Gọi số câu trả lời đúng là $x$, số câu trả lời sai hoặc không trả lời là $y$ ($x,y$ là các số nguyên không âm). Ta có 
$$\heva{&x+y=15 \\ & 5x-5y=25.}\Leftrightarrow \heva{& x+y=15\\ & 10x=100.}\Leftrightarrow \heva{&x=10\\ & y=5.}$$
Vậy học sinh đó trả lời đúng $10$ câu.
}
\end{bt}

\begin{bt}%[Nguyễn Thị Huyền My]%[0D3K3]
Có $2$ loại xe khách (loại I và loại II). Nếu chọn phương án vận chuyển hành khách bằng $2$ chuyến xe loại I và $5$ chuyến xe loại II thì vận chuyển được tối đa $190$ hành khách (không tính tài xế). Nếu chọn phương án vận chuyển bằng $3$ chuyến xe loại I và $3$ chuyến xe loại II thì vận chuyển được tối đa $195$ hành khách (không tính tài xế). Hỏi mỗi loại xe có thể chứa tối đa bao nhiêu hành khách (không tính tài xế)?
\loigiai{
Gọi $x$ và $y$ lần lượt là số hành khách tối đa có thể chở được của xe loại I và xe loại II ($x,y$ là các số nguyên dương). Ta có hệ phương trình $$\heva{& 2x+5y=190 \\ & 3x+3y=195.}\Leftrightarrow\heva{& x=45\\ &y=20.}$$
}
\end{bt}

\begin{bt}%[Nguyễn Thị Huyền My]%[0D3G3]
Giải hệ phương trình $\heva{& 2|x+y|+|x-y|=7\\&-|x+y|+4|x-y|=10.} $
\loigiai{Đặt $\heva{&u=|x+y| \\&v=|x-y|}, u,v \geq 0$. Hệ phương trình theo $u, v$:
$$\heva{& 2u+v=7\\&-u+4v=10 .}\Leftrightarrow \heva{& v=7-2u \\& -u+4(7-2u) =10 .}\Leftrightarrow \heva{& v= 3\\&u=2 .}$$
Suy ra $\heva{& |x+y| = 2 \\& |x-y| = 3 .}\Leftrightarrow  \heva{& \hoac{& x+y=2\\&x+y=-2}\\& |x-y|=3 .}\Leftrightarrow \hoac{&\heva{& y=2-x \\& |x-y|=3 .}\\& \heva{& y=-2-x \\&|x-y|=3 .}}$
\\
Xét hệ $\heva{& y=2-x \\& |x-y|=3 .} \Leftrightarrow \heva{& y=2-x \\& |x-(2-x)|=3 .}\Leftrightarrow \heva{& y=2-x \\&\hoac{&2x-2=3 \\&2x-2=-3.}} \Leftrightarrow \heva{& y=2-x \\&\hoac{&x=\dfrac{5}{2} \\ &x=-\dfrac{1}{2}.}} \Leftrightarrow \hoac{& \heva{&x=\dfrac{5}{2} \\& y=-\dfrac{1}{2}.}\\&\heva{&x=-\dfrac{1}{2} \\& y=\dfrac{5}{2}.}}$
\\
Xét hệ $\heva{& y=-2-x \\& |x-y|=3 .} \Leftrightarrow \heva{& y=-2-x \\& |x-(-2-x)|=3 .}\Leftrightarrow \heva{& y=-2-x \\&\hoac{&2x+2=3 \\&2x+2=-3.}} \Leftrightarrow \heva{& y=-2-x \\&\hoac{&x=\dfrac{1}{2} \\ &x=-\dfrac{5}{2}.}} \Leftrightarrow \hoac{& \heva{&x=\dfrac{1}{2} \\& y=-\dfrac{5}{2}.}\\&\heva{&x=-\dfrac{5}{2} \\& y=\dfrac{1}{2}.}}$
\\
Vậy hệ phương trình có $4$ nghiệm là $\left(\dfrac{5}{2}; -\dfrac{1}{2} \right); \left(-\dfrac{1}{2}; \dfrac{5}{2} \right); \left(\dfrac{1}{2}; -\dfrac{5}{2} \right); \left(-\dfrac{5}{2}; \dfrac{1}{2} \right)$.
}

\begin{dang}{Giải và biện luận hệ 2 phương trình bậc nhất 2 ẩn có chứa tham số (PP Crame)}
\begin{enumerate}
\item \textbf{Dạng:}
$\heva{&a_1x+b_1y=c_1\quad (a_1^2+b_1^2 \ne 0)\\&a_2x+b_2y=c_2\quad (a_2^2+b_2^2 \ne 0).}$\\
\underline{\textit{Cách giải đã biết}}: Phép thế, phép cộng...
\item \textbf{Giải và biện luận hệ phương trình:}
\begin{enumerate}[Bước 1:]
\item Tính các định thức:
\item[•] $D=\begin{vmatrix}
a_1 & b_1 \\ 
a_2 & b_2
\end{vmatrix}=a_1b_2-a_2b_1 $ (Gọi là định thức của hệ);
\item[•] $D_x=\begin{vmatrix}
c_1 & b_1 \\ 
c_2 & b_2
\end{vmatrix}=c_1b_2-c_2b_1$ (Gọi là định thức của $x$);
\item[•] $D_y=\begin{vmatrix}
a_1 & c_1 \\ 
a_2 & c_2
\end{vmatrix}=a_1c_2-a_2c_1 $ (Gọi là định thức của $y$).
\item Biện luận
\item[•] Nếu $D \ne 0$ thì hệ có nghiệm duy nhất $\heva{&x=\dfrac{D_x}{D}\\&y=\dfrac{D_y}{D}.}$
\item[•] Nếu $D=0$ và $D_x \ne 0$ hoặc $D_y \ne 0$ thì hệ vô nghiệm.
\item[•] Nếu $D=D_x=D_y=0$ thì hệ
 có vô số nghiệm (tập nghiệm của hệ là tập nghiệm của phương trình $a_1x+b_1y=c_1$).
\end{enumerate}
\end{enumerate}
\end{dang}

\begin{vd}%[Trần Văn Hiếu]%[0D3B3]
Giải và biện luận hệ phương trình:$\heva{&mx+y=m+1\\&x+my=2.}$
\loigiai{
$D=\begin{vmatrix}
m&1\\1&m
\end{vmatrix}=m^2-1;
D_x=\begin{vmatrix}
m+1&1\\2&m
\end{vmatrix}=m^2+m-2;
D_y=\begin{vmatrix}
m&m+1\\1&2
\end{vmatrix}=m-1.$
\begin{enumerate}
\item Nếu $m=1 \Rightarrow D=D_x=D_y=0$.
Hệ có vô số nghiệm $(x;y)$ thỏa $x+y=2$.
\item Nếu $m=-1 \Rightarrow \heva{&D=0\\&D_x=-2\\&D_y=-2}$. Hệ vô nghiệm.
\item Nếu $m \ne 1,m \ne -1$. Hệ có nghiệm duy nhất $\heva{&x=\dfrac{D_x}{D}=\dfrac{m^2+m-2}{m^2-1}=\dfrac{m+2}{m+1}\\&y=\dfrac{D_y}{D}=\dfrac{m-1}{m^2-1}=\dfrac{1}{m+1}.}$
\end{enumerate}
}
\end{vd}

\begin{vd}%[Trần Văn Hiếu]%[0D3K3]
Với giá trị nguyên nào của tham số $m$, hệ phương trình $\heva{&mx+4y=m+2\\&x+my=m.}$ có nghiệm duy nhất $(x;y)$ với $x,y$ là các số nguyên.
\loigiai{
$D=m^2-4=(m-2)(m+2), D_x=m^2+2m-4m=m^2-2m=m(m-2),$\\
$ D_y=m^2-m-2=(m+1)(m-2)$.\\
$D=m^2-4 \ne 0 \Leftrightarrow m \ne \pm 2$, hệ có nghiệm duy nhất $\heva{&x=\dfrac{D_x}{D}=\dfrac{m}{m+2}\\&y=\dfrac{D_y}{D}=\dfrac{m+1}{m+2}.}$\\
Ta có:\\
$x=\dfrac{m}{m+2}=1-\dfrac{2}{m+2}$ nên để $x$ nguyên thì $m+2$ phải là ước của $2$ (1);\\
$y=\dfrac{m+1}{m+2}=1-\dfrac{1}{m+2}$ nên để $y$ nguyên thì $m+2$ phải là ước của $1$ (2).\\
Từ (1),(2), suy ra $m+2$ là ước của $1\Leftrightarrow \hoac{&m+2=1\\&m+2=-1} \Leftrightarrow \hoac{m=-1\\m=-3.}$
}
\end{vd}

\begin{vd}%[Trần Văn Hiếu]%[0D3K3]
Cho hệ phương trình: $\heva{&x+my=1\\&mx-y=-m.}$
\begin{enumerate}[a)]
\item Chứng minh rằng với mọi giá trị của $m$ hệ phương trình đã cho luôn có nghiệm duy nhất.
\item Tìm các giá trị của $m$ để hệ phương trình có nghiệm $(x;y)$ thỏa mãn $x<1,y<1$.
\item Tìm hệ thức liên hệ giữa $x,y$ không phụ thuộc vào $m$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
\item $D=-1-m^2 <0, \quad \forall m\\
D_x=-1+m^2\\
D_y=-m-m=-2m$\\
Ta có: $D \ne 0, \quad \forall m$  nên hệ phương trình có nghiệm duy nhất $\heva{&x=\dfrac{D_x}{D}=\dfrac{m^2-1}{-m^2-1}\\&y=\dfrac{D_y}{D}=\dfrac{-2m}{-m^2-1}.}$
\item $x<1 \Leftrightarrow \dfrac{m^2-1}{-m^2-1}<1 \Leftrightarrow m^2-1>-m^2-1 \Leftrightarrow m^2>0 \Leftrightarrow m \ne 0$.\\
$y<1 \Leftrightarrow \dfrac{-2m}{-m^2-1}<1 \Leftrightarrow -2m > -m^2-1 \Leftrightarrow (m-1)^2>0 \Leftrightarrow m \ne 1$.\\
Vậy với $m\ne 0\wedge m\ne 1$ thì hệ có nghiệm thỏa mãn yêu cầu bài toán.
\item $x^2+y^2=\left(\dfrac{m^2-1}{m^2+1} \right)^2+\left(\dfrac{2m}{m^2+1} \right)^2=\dfrac{m^4-2m^2+1+4m^2}{m^4+2m^2+1}=\dfrac{m^4+2m^2+1}{m^4+2m^2+1}=1$.\\
Vậy $x^2+y^2=1$ không phụ thuộc vào $m$.

\end{enumerate}
}
\end{vd}

\begin{bt}%[Trần Văn Hiếu]%[0D3G3]

Tìm tất cả các số nguyên $m$ để hai đường thẳng $y=-\dfrac{2}{3}mx+\dfrac{1}{3}m$ và $y=-x+m+1$ cắt nhau tại một điểm có các thành phần tọa độ là các số nguyên.
\loigiai{
Tọa độ giao điểm của hai đường thẳng đã cho là nghiệm của hệ phương trình $$\heva{&y=-\dfrac{2}{3}mx+\dfrac{1}{3}m\\&y=-x+m+1.}\Leftrightarrow\heva{&2mx+3y=m\\&x+y=m+1.} (1)$$
Suy ra hai đường thẳng đã cho cắt nhau tại một điểm có các thành phần tọa độ là các số nguyên khi và chỉ khi hệ (1) có nghiệm duy nhất và nghiệm duy nhất đó nguyên.\\
Ta có $$D=\begin{vmatrix}
2m & 3\\
1 & 1
\end{vmatrix}=2m-3, D_x=\begin{vmatrix}
m & 3\\
m+1 & 1
\end{vmatrix}=-2m-3,D_y=\begin{vmatrix}
2m & m\\
1 & m+1
\end{vmatrix}=2m^2+m.$$
Hệ (1) có nghiệm duy nhất $\Leftrightarrow D\neq 0\Leftrightarrow m\neq \dfrac{3}{2}$. Khi đó nghiệm duy nhất của hệ (1) là 
$$\heva{&x=-\dfrac{2m+3}{2m-3}\\&y=\dfrac{2m^2+m}{2m-3}}\Leftrightarrow \heva{&x=-1-\dfrac{6}{2m-3}\\&y=m+2+\dfrac{6}{2m-3}}.$$
Nghiệm này nguyên $\Rightarrow 2m-3$ là ước của $6$
$$\Leftrightarrow \hoac{&2m-3=-1\\ &2m-3=1\\ &2m-3=-2\\ &2m-3=2\\ &2m-3=-3\\ &2m-3=3\\ &2m-3=-6\\ &2m-3=6.}\Leftrightarrow \hoac{&m=1\\ &m=2\\ &m=\dfrac{1}{2}\\ &m=\dfrac{5}{2}\\ &m=0\\ &m=3\\ &m=-\dfrac{3}{2}\\ &m=\dfrac{9}{2}}.$$
Vậy $m\in\{0;1;2;3\}$.
}
\end{bt}
\begin{bt}%[Khuất Văn Thanh]%[0D3G3]
Giải hệ phương trình $\heva{&\dfrac{3}{2x-y}-\dfrac{4}{x-2y}=-2\\ &\dfrac{5}{2x-y}+\dfrac{2}{x-2y}=14.}$
\loigiai{
Điều kiện $\heva{&2x-y\neq 0\\&x-2y\neq 0}\Leftrightarrow \heva{&x\neq \dfrac{y}{2}\\&x\neq 2y}$ (*).\\
Với điều kiện (*), đặt $u=\dfrac{1}{2x-y}, v=\dfrac{1}{x-2y}$, hệ đã cho trở thành
$$\heva{&3u-4v=-2\\&5u+2v=14}\Leftrightarrow \heva{&3u-4v=-2\\&10u+4v=28.}\Leftrightarrow \heva{&3u-4v=-2\\&13u=26}\Leftrightarrow\heva{&u=2\\&v=2.}$$
Suy ra $\heva{&2x-y=\dfrac{1}{2}\\&x-2y=\dfrac{1}{2}.}\Leftrightarrow \heva{&3x=\dfrac{1}{2}\\&x-2y=\dfrac{1}{2}}\Leftrightarrow \heva{&x=\dfrac{1}{6}\\&y=-\dfrac{1}{6}}$ (Thỏa mãn (*)).\\
Vậy nghiệm của hệ đã cho là $\heva{&x=\dfrac{1}{6}\\&y=-\dfrac{1}{6}.}$
}
\end{bt}
\begin{bt}%[Khuất Văn Thanh]%[0D3G3]
Giải và biện luận hệ phương trình $\heva{&(m+n)x+(m-n)y=2m(m^2-n^2)\\ &x-y=-4mn}$ theo hai tham số $m,n$.
\loigiai{
Ta có 
\begin{align*}
& D=\begin{vmatrix}
m+n & m-n\\
1 & -1
\end{vmatrix}=-(m+n)-(m-n)=-2m,\\
& D_x=\begin{vmatrix}
2m(m^2-n^2)& m-n\\
-4mn & -1
\end{vmatrix}=-2m(m^2-n^2)+4mn(m-n)=-2m(m-n)^2,\\
& D_y=\begin{vmatrix}
(m+n)&2m(m^2-n^2)\\
1&-4mn
\end{vmatrix}=-4mn(m+n)-2m(m^2-n^2)=-2m(m+n)^2.
\end{align*}
\begin{itemize}
\item Nếu $D\neq 0\Leftrightarrow m\neq 0$ thì hệ đã cho có nghiệm duy nhất $\heva{&x=\dfrac{-2m(m-n)^2}{-2m}=(m-n)^2\\&y=\dfrac{-2m(m+n)^2}{-2m}=(m+n)^2.}$.
\item Nếu $D=0\Leftrightarrow m=0\Rightarrow D_x=D_y=0$: Hệ có vô số nghiệm $\heva{&x\in\mathbb{R}\\&y=x.}$
\end{itemize}
}
\end{bt}
\begin{bt}%[Khuất Văn Thanh]%[0D3G3]
Tìm tất cả các giá trị của $a,b,c$ để ít nhất một trong các hệ phương trình sau có nghiệm:
$$\heva{&x+cy=b\\&cx+y=a}(1);\quad \heva{&cx+y=a\\&bx+ay=1}(2);\quad \heva{&bx+ay=1\\&x+cy=b}(3).$$
\loigiai{
Ta giải bài toán gián tiếp: Tìm tất cả các giá trị của $a,b,c$ để cả ba hệ đã cho đều vô nghiệm.\\
Xét hệ $\heva{&x+cy=b\\&cx+y=a}(1)$.\\
Ta có $D_{(1)}=\begin{vmatrix}
1&c\\
c&1
\end{vmatrix}=1-c^2; D_x=\begin{vmatrix}
b&c\\
a&1
\end{vmatrix}=b-ac; D_y=\begin{vmatrix}
1&b\\
c&a
\end{vmatrix}=a-bc.$\\
Hệ (1) vô nghiệm khi và chỉ khi $\heva{&D_{(1)}=0\\&\hoac{&D_x\neq 0\\&D_y\neq 0}}\Leftrightarrow \heva{&1-c^2=0\\&\hoac{&b-ac\neq 0\\&a-bc\neq 0}}\Leftrightarrow \hoac{&\heva{&c=1\\&b-a\neq 0}\\&\heva{&c=-1\\&b+a\neq 0}}$.
\begin{itemize}
\item Xét trường hợp $\heva{&c=1\\&b-a\neq 0}$. Thay $c=1$ vào hệ (2), ta có: $\heva{&x+y=a\\&bx+ay=1}$\\
Ta có $D_{(2)}=\begin{vmatrix}
1&1\\
b&a
\end{vmatrix}=a-b\neq 0$. Do đó (2) có nghiệm duy nhất.
\item Xét trường hợp $\heva{&c=-1\\&b+a\neq 0}$. Thay $c=-1$ vào (3) ta có $\heva{&bx+ay=1\\&x-y=b}$\\
Ta có $D_{(3)}=\begin{vmatrix}
b&a\\
1&-1
\end{vmatrix}=-b-a=-(b+a)\neq 0$. Do đó (3) có nghiệm duy nhất.
\end{itemize}
Suy ra nếu (1) vô nghiệm thì các hệ (2),(3) đều có nghiệm. Do đó không tồn tại giá trị nào của $a,b,c$ để cả ba hệ phương trình đã cho cùng vô nghiệm.\\
Vậy, với mọi giá trị của $a,b,c$ thì ít nhất một trong các hệ đã cho có nghiệm.  
}
\end{bt}
\begin{bt}%[Khuất Văn Thanh]%[0D3G3]
Tập thể giáo viên (Toán và \LaTeX) gồm $128$ người được chia thành ba nhóm ra đề kiểm tra: Nhóm $1$, Nhóm $2$ và Nhóm $3$. Sau một ngày làm việc, cả ba nhóm hoàn thành được $476$ câu trắc nghiệm và $375$ câu tự luận; trong đó,  mỗi thành viên Nhóm $1$ hoàn thành được $3$ câu trắc nghiệm và $4$ câu tự luận, mỗi thành viên Nhóm $2$ hoàn thành được $2$ câu trắc nghiệm và $5$ câu tự luận, mỗi thành viên Nhóm $3$ hoàn thành được $6$ câu trắc nghiệm. Hỏi mỗi nhóm có bao nhiêu giáo viên?           
\loigiai{
Gọi số giáo viên của Nhóm $1$, Nhóm $2$, Nhóm $3$ lần lượt là $x,y,z$ (ĐK: $x, y, z$ nguyên dương).\\
Theo đề bài, ta lập được hệ phương trình
$$\heva{&x+y+z=128\quad (1)\\ &3x+2y+6z=476\quad (2)\\ &4x+5y=375\quad (3)}$$
Nhân hai vế của (1) với $-6$ rồi cộng vào (2), ta được hệ
$$\heva{&x+y+z=128\quad (1)\\&3x+4y=292\quad (4)\\&4x+5y=375\quad (5)}$$
Từ (4) và (5) ta có $\heva{&x=40\\&y=43.}$ Thế vào (1) ta được $z=45$ (thỏa mãn ĐK).\\
Vậy số giáo viên của các nhóm $1,2,3$ lần lượt là $40,43,45$.
}
\end{bt}

\begin{bt}%[Khuất Văn Thanh]%[0D3G3]
Tìm $m$ để hệ sau có nghiêm duy nhất và tìm nghiệm duy nhất đó.
$$\heva{&mx+y+z=1\\ &x+my+z=2\\ &x+y+mz=4.} (*)$$
\loigiai{
Hệ $(*)\Leftrightarrow \heva{&mx+y+z=1\quad (1)\\ &(m-1)x+(1-m)y=-1\quad (2)\\ &(m^2-1)x+(m-1)y=m-4\quad (3)}$
}\\
Xét hệ phương trình (2), (3), ta có:
\begin{itemize}
\item 
$D=\begin{vmatrix}
m-1 & 1-m\\
m^2-1 & m-1
\end{vmatrix}=(m-1)^2-(1-m)(m^2-1)=(m-1)^2(m+2),$
\item
$D_x=\begin{vmatrix}
-1 & 1-m\\
m-4 & m-1
\end{vmatrix}=-(m-1)-(1-m)(m-4)=(m-1)(m-5),$
\item
$D_y=\begin{vmatrix}
m-1 & -1\\
m^2-1 & m-4
\end{vmatrix}=(m-1)(m-4)+(m^2-1)=(m-1)(2m-3).$
\end{itemize}
Hệ (*) có nghiệm duy nhất $\Leftrightarrow$ Hệ (2), (3) có nghiệm duy nhất $\Leftrightarrow D\neq 0\Leftrightarrow\heva{&m\neq 1\\&m\neq -2}$.\\
Khi đó nghiệm duy nhất của hệ (*) là $\heva{&x=\dfrac{D_x}{D}=\dfrac{m-5}{(m-1)(m+2)}\\ &y=\dfrac{D_y}{D}=\dfrac{2m-3}{(m-1)(m+2)}\\ &z=\dfrac{4m+1}{(m-1)(m+2)}.}$
\\
\textbf{Cách khác:} Lấy ba phương trình của hệ cộng lại, ta được $$(m+2)x+(m+2)y+(m+2)z=7$$
Hệ có nghiệm khi $m\neq -2$. Khi đó $$x+y+z=\dfrac{7}{m+2} \quad (1)$$
Lấy các phương trình của hệ trừ cho $(1)$ ta được
$$\heva{&(m-1)x=1-\dfrac{7}{m+2}\\&(m-1)y=2-\dfrac{7}{m+2}\\&(m-1)z=4-\dfrac{7}{m+2}.}$$
Từ đó suy ra, hệ có nghiệm duy nhất khi và chỉ khi $\heva{&m\neq -2\\&m\neq 1.}$ và nghiệm duy nhất của hệ là $\heva{&x=\dfrac{m-5}{(m-1)(m+2)}\\ &y=\dfrac{2m-3}{(m-1)(m+2)}\\ &z=\dfrac{4m+1}{(m-1)(m+2)}.}$
\end{bt}


\begin{bt}%[Khuất Văn Thanh]%[0D3G3]
Giải hệ phương trình $\heva{&x+ay+a^2z+a^3=2017\\&x+by+b^2z+b^3=2017\\&x+cy+c^2z+c^3=2017.}\quad (1)$, với $a,b,c$ là các tham số đôi một khác nhau.
\loigiai{
Ta có
\begin{align*}
(1)&\Leftrightarrow \heva{&x+ay+a^2z+a^3=2017\\&(b-a)y+(b^2-a^2)z+b^3-a^3=0\\&(c-a)y+(c^2-a^2)z+c^3-a^3=0.}&\Leftrightarrow & \heva{&x+ay+a^2z+a^3=2017\\&y+(b+a)z+b^2+ab+a^2=0\\&y+(c+a)z+c^2+ac+a^2=0.} \\
&\Leftrightarrow \heva{&x+ay+a^2z+a^3=2017\\&y+(b+a)z+b^2+ab+a^2=0\\&(b-c)z+b^2-c^2+a(b-c)=0.}&\Leftrightarrow &\heva{&x+ay+a^2z+a^3=2017\\&y+(b+a)z+b^2+ab+a^2=0\\&z=-(a+b+c).}\\
& &\Leftrightarrow &\heva{&x=-ay-a^2z-a^3+2017\\&y=ab+bc+ca\\&z=-(a+b+c).}
\end{align*}
Vậy hệ (1) có nghiệm là $\heva{&x=-abc+2017\\&y=ab+bc+ca\\&z=-(a+b+c).}$
}
\end{bt}
