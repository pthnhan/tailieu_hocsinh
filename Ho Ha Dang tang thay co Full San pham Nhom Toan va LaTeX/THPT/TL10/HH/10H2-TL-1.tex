\section{GIÁ TRỊ LƯỢNG GIÁC CỦA MỘT GÓC BẤT KỲ TỪ $0^\circ$ ĐẾN $180^\circ$}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Giá trị lượng giác của một góc bất kỳ từ $\mathbf{0^{\circ}}$ đến $\mathbf{180^{\circ}}$}
\begin{dn}
\immini{
Với mỗi góc $\alpha$ ($0^{\circ}\leq \alpha\leq 180^{\circ}$), ta xác định một điểm $M$ trên nửa đường tròn đơn vị sao cho $\widehat{xOM}=\alpha$ và giả sử điểm $M$ có tọa độ $M\bigl(x_0;y_0\bigr)$. Khi đó ta định nghĩa:
\begin{itemize}
	\item \textbf{sin} của góc $\alpha$ là $y_0$, ký hiệu $\sin\alpha=y_0$;
	\item \textbf{cô-sin} của góc $\alpha$ là $x_0$, ký hiệu $\cos\alpha=x_0$;
	\item \textbf{tang} của góc $\alpha$ là $\dfrac{y_0}{x_0}$ ($x_0\neq 0$), ký hiệu $\tan \alpha=\dfrac{y_0}{x_0}$;
	\item \textbf{cô-tang} của góc $\alpha$ là $\dfrac{x_0}{y_0}$ ($y_0\neq 0$), ký hiệu $\cot\alpha=\dfrac{x_0}{y_0}$.
\end{itemize}
}{
\begin{tikzpicture}[>=stealth]
	\draw[->] (-2.2,0) -- (2.4,0) node[below]{$x$};
	\draw[->] (0,-.5) -- (0,3) node[left]{$y$};
	\draw (.1,0) node[below left]{$O$}
		(-2,0) node[below]{$-1$}
		(2,0) node[below]{$1$}
		(-0.1,2) node[above right]{$1$};
	\draw (2,0) arc (0:180:2cm);
	\path (130:2cm) coordinate (M);
	\draw[dashed] (M) node[above left]{$M$} -- (M|- 0,0) node[below]{$x_0$};
	\draw[dashed] (M) -- (M -|0,0) node[right]{$y_0$};
	\draw (0,0) -- (M);
	\draw[->] (0.3,0) to [bend right,looseness=1.3] (130:0.3cm) node[above right,midway]{$\alpha$};
\end{tikzpicture}
}

Các số $\sin\alpha,\cos\alpha,\tan\alpha,\cot\alpha$ được gọi là các \emph{giá trị lượng giác của góc $\alpha$}.
\end{dn}
\begin{note}
\begin{itemize}\textbf{Chú ý.}
\item[]
\item Nếu $\alpha$ là góc tù thì $\cos\alpha<0$, $\tan\alpha<0$, $\cot\alpha<0$.
\item $\tan\alpha$ chỉ xác định khi $\alpha\neq 90^{\circ}$.
\item $\cot\alpha$ chỉ xác định khi $\alpha\neq 0^{\circ}$ và $\alpha\neq 180^{\circ}$.
\end{itemize}
\end{note}
\begin{tc}
	Về dấu của các giá trị lượng giác.
	\begin{itemize}
		\item $\sin\alpha> 0$ với $0^{\circ}< \alpha< 180^{\circ}$.
		\item $\cos\alpha> 0$ với $0^{\circ}< \alpha< 90^{\circ}$ và $\cos\alpha< 0$ với $90^{\circ}<\alpha< 180^{\circ}$.
		\item $\tan\alpha> 0$ với $0^{\circ}< \alpha< 90^{\circ}$ và $\tan\alpha< 0$ với $90^{\circ}<\alpha< 180^{\circ}$.
		\item $\cot\alpha> 0$ với $0^{\circ}< \alpha< 90^{\circ}$ và $\cot\alpha< 0$ với $90^{\circ}<\alpha< 180^{\circ}$.
	\end{itemize}
	Như vậy, $\cos\alpha,\tan\alpha,\cot\alpha$ luôn cùng dấu với $0^{\circ}< \alpha< 90^{\circ}$ và $90^{\circ}<\alpha<180^{\circ}$.
\end{tc}
\begin{tc}
	Mối quan hệ giữa hai góc bù nhau.
	\begin{itemize}
		\item $\sin \alpha=\sin(180^{\circ}-\alpha)$.
		\item $\cos \alpha=-\cos(180^{\circ}-\alpha)$.
		\item $\tan \alpha=-\tan(180^{\circ}-\alpha)$ với $\alpha\neq 90^{\circ}$.
		\item $\cot \alpha=-\cot(180^{\circ}-\alpha)$ với $\alpha\neq 0^{\circ},180^{\circ}$.
	\end{itemize}
\end{tc}
\begin{tc}
	Mối quan hệ giữa hai góc phụ nhau (với $0^{\circ}\leq \alpha\leq 90^{\circ}$).
	\begin{itemize}
		\item $\sin (90^{\circ}-\alpha)=\cos\alpha$.
		\item $\cos (90^{\circ}-\alpha)=\sin\alpha$.
		\item $\tan (90^{\circ}-\alpha)=\cot\alpha$ với $\alpha\neq 0^{\circ}$.
		\item $\cot (90^{\circ}-\alpha)=\tan\alpha$ với $\alpha\neq 90^{\circ}$.
	\end{itemize}
\end{tc}
\begin{tc}
	Các công thức cơ bản.\\
	\begin{tabular}{>{$\bullet$ $}p{0.3\textwidth}<{$}>{$\bullet$ $}p{0.3\textwidth}<{$}>{$\bullet$ $}p{0.3\textwidth}<{$}}
		\tan\alpha=\dfrac{\sin \alpha}{\cos\alpha}. & \cot\alpha=\dfrac{\cos\alpha}{\sin\alpha}. & \tan \alpha.\cot\alpha=1.\\
		\sin^2\alpha+\cos^2\alpha=1. & 1+\tan^2\alpha=\dfrac{1}{\cos^2\alpha}. & 1+\cot^2\alpha=\dfrac{1}{\sin^2\alpha}.
	\end{tabular}
\end{tc}
\subsubsection{Góc giữa hai vec-tơ}
\begin{dn}
	\immini{Cho hai vec-tơ $\vec{a}$ và $\vec{b}$ đều khác vec-tơ $\vec{0}$. Từ một điểm $O$ bất kỳ, ta vẽ $\vec{OA}=\vec{a}$ và $\vec{OB}=\vec{b}$. Góc $\widehat{AOB}$ với số đo từ $0^{\circ}$ đến $180^{\circ}$ được gọi là \emph{góc giữa hai vec-tơ} $\vec{a}$ và $\vec{b}$. Ta ký hiệu góc giữa hai vec-tơ $\vec{a}$ và $\vec{b}$ là $\bigl(\vec{a},\vec{b}\bigr)$. Nếu $\bigl(\vec{a},\vec{b}\bigr)=90^{\circ}$ thì ta nói rằng $\vec{a}$ và $\vec{b}$ vuông góc với nhau, ký hiệu là $\vec{a}\perp\vec{b}$ hoặc $\vec{b}\perp\vec{a}$.

\begin{note}
	Từ định nghĩa ta có $\bigl(\vec{a},\vec{b}\bigr)=\bigl(\vec{b},\vec{a}\bigr)$.
\end{note}
\begin{tc}
	Nếu $\vec{a}$ và $\vec{b}$ cùng hướng thì $\bigl(\vec{a},\vec{b}\bigr)=0^{\circ}$.
\end{tc}
\begin{tc}
	Nếu $\vec{a}$ và $\vec{b}$ ngược hướng thì $\bigl(\vec{a},\vec{b}\bigr)=180^{\circ}$.
\end{tc}

\begin{dang}{Tính các giá trị lượng giác}

\begin{vd}%[Lâm Hữu Phước]%[0H2B1]
Cho $\sin\alpha=\dfrac{1}{4}$. Tính $\cos\alpha$, $\tan\alpha$, $\cot\alpha$ biết $0^{\circ}<\alpha<90^{\circ}$.
\loigiai{
Ta có $\sin^2\alpha+\cos^2\alpha=1\Rightarrow \cos^2\alpha=1-\sin^2\alpha$.

Với $\sin\alpha=\dfrac{1}{4}$ thì $\cos^2\alpha=1-\dfrac{1}{16}=\dfrac{15}{16}$.

Vì $0^{\circ}<\alpha<90^{\circ}$ nên $\cos\alpha=\dfrac{\sqrt{15}}{16}$.

Từ đó suy ra $\tan\alpha=\dfrac{\sin\alpha}{\cos\alpha}=\dfrac{\sqrt{15}}{15}$, $\cot\alpha=\dfrac{\cos\alpha}{\sin\alpha}\sqrt{15}$.
}
\end{vd}
\begin{vd}%[Lâm Hữu Phước]%[0H2B1]
Cho $\cos\alpha=-\dfrac{1}{3}$. Tính các giá trị lượng giác còn lại của góc $\alpha$.
\loigiai{
Ta có $\sin^2\alpha+\cos^2\alpha=1\Rightarrow \sin^2\alpha=1-\cos^2\alpha$.

Với $\cos\alpha=-\dfrac{1}{3}$ thì $\sin^2\alpha=1-\dfrac{1}{9}=\dfrac{8}{9}$.

Vì $\sin\alpha$ luôn dương nên $\sin\alpha=\dfrac{2\sqrt{2}}{3}$.

Từ đó suy ra $\tan\alpha=\dfrac{\sin\alpha}{\cos\alpha}=-2\sqrt{2}$, $\cot\alpha=\dfrac{\cos\alpha}{\sin\alpha}=-\dfrac{\sqrt{2}}{4}$.
}
\end{vd}
\begin{vd}%[Lâm Hữu Phước]%[0H2B1]
Cho $\tan x=2$. Tính các giá trị lượng giác còn lại của góc $x$.
\loigiai{
Trước hết, ta có $\tan x.\cot x=1\Rightarrow \cot x=\dfrac{1}{\tan x}=\dfrac{1}{2}$.

Mặt khác, $1+\tan^2x=\dfrac{1}{\cos^2x}\Rightarrow \cos^2 x=\dfrac{1}{1+\tan^2x}=\dfrac{1}{1+2^2}=\dfrac{1}{5}$.

Vì $\tan x$ và $\cos x$ cùng dấu nên $\cos x=\dfrac{\sqrt{5}}{5}$.

Áp dụng công thức $\sin^2x +\cos^2 x=1\Rightarrow\sin^2x=1-\cos^2x=1-\dfrac{1}{5}=\dfrac{4}{5}$. Từ đó suy ra $\sin x=\dfrac{2\sqrt{5}}{5}$.
}
\end{vd}
\begin{vd}%[Lâm Hữu Phước]%[0H2B1]
Cho $\cot x=-3$. Tính các giá trị lượng giác còn lại của góc $x$.
\loigiai{
Trước hết ta có $\tan x.\cot x=1\Rightarrow\tan x=\dfrac{1}{\cot x}=-\dfrac{1}{3}$.

Mặt khác $1+\cot^2 x=\dfrac{1}{\sin^2 x}\Rightarrow \sin^2 x=\dfrac{1}{1+(-3)^2}=\dfrac{1}{10}$. Suy ra $\sin x=\dfrac{\sqrt{10}}{10}$.

Do $\cot x=\dfrac{\cos x}{\sin x}\Rightarrow \cos x=\sin x.\cot x=\dfrac{-3\sqrt{10}}{10}$.
}
\end{vd}

\begin{vd}%[Vũ Văn Trường]%[0H2B1]
Cho $\heva{&a=\sin x\\&b=\cos x\sin x\\&c=\cos x\cos y}$. Chứng minh rằng $a^2+b^2+c^2=1$
\loigiai{Ta có:
\begin{align*}
  a^2+b^2+c^2 & =\sin^2x+\cos^2x(1-\cos^2y)+\cos^2x\cos^2y \\
 &=\sin^2x+\cos^2x-\cos^2x\cos^2y+\cos^2x\cos^2y\\
 &=1.
\end{align*}
}
\end{vd}
\begin{vd}%[Vũ Văn Trường]%[0H2B1]
Chứng minh các đẳng thức sau:
\begin{enumerate}
  \item $\sin^4x+\cos^4x=1-2\sin^2x\cos^2x.$
  \item $\cos^4x-\sin^4x=\cos^2x-\sin^2x=1-2\sin^2x=2\cos^2x-1.$
  \item $\tan^2x-\sin^2x=\tan^2x\sin^2x.$
  \item $\dfrac{1}{1+\tan x}+\dfrac{1}{1+\cot x}=1.$
\end{enumerate}

\loigiai{
\begin{enumerate}
  \item Ta có $\sin^4x+\cos^4x=\left(\sin^2x\right)^2+\left(\cos^2x\right)^2= \left(\sin^2x+\cos^2x\right)^2-2\sin^2x\cos^2x$\\
  Do $\sin^2x+\cos^2x=1$ nên ta suy ra $\sin^4x+\cos^4x=1-2\sin^2x\cos^2x.$
  \item $\cos^4x-\sin^4x=\left(\cos^2x\right)^2-\left(\sin^2x\right)^2=\left(\cos^2x-\sin^2x\right)\left(\cos^2x+\sin^2x\right)=\cos^2x-\sin^2x$\\
  Do $\sin^2x+\cos^2x=1$ nên $\cos^2x-\sin^2x=\cos^2x+\sin^2x-2\sin^2x=1-2\sin^2x$\\
  Tương tự ta có $\cos^2x-\sin^2x=2\cos^2x-1.$
  \item $\tan^2x-\sin^2x=\dfrac{\sin^2x}{\cos^2x}-\sin^2x=\sin^2x\left(\dfrac{1}{\cos^2x}-1\right)=\sin^2x\dfrac{1-\cos^2x}{\cos^2x}=\tan^2x\sin^2x$
  \item Ta có $\dfrac{1}{1+\tan x}+\dfrac{1}{1+\cot x}=\dfrac{1+\tan x+1+\cot x}{\left(1+\tan x\right)\left(1+\cot x\right)}$.\\
  Mặt khác $\left(1+\tan x\right)\left(1+\cot x\right)=1+\tan x\cot x+\tan x+\cot x=2+\tan x+\cot x$.\\
  Từ đó suy ra $\dfrac{1}{1+\tan x}+\dfrac{1}{1+\cot x}=\dfrac{2+\tan x+\cot x}{2+\tan x+\cot x}=1.$
\end{enumerate}
}
\end{vd}
\begin{vd}%[Vũ Văn Trường]%[0H2B1]
Cho $A, B, C$ là các góc của tam giác. Chứng minh các đẳng thức sau:
\begin{enumerate}
  \item $\sin\left(A+B\right)=\sin C.$
  \item $\cos\left(A+B\right)+\cos C=0.$
  \item $\sin\dfrac{A+B}{2}=\cos\dfrac{C}{2}.$
  \item $\tan\left(A-B+C\right)=-\tan2B.$
\end{enumerate}
\loigiai{ Do $A, B, C$ là các góc của tam giác nên ta có $A+B+C=180^{\circ}$.
\begin{enumerate}
  \item Ta có $A+B+C=180^{\circ}\Leftrightarrow A+B=180^{\circ}-C.$\\
  Từ đó suy ra $\sin\left(A+B\right)=\sin \left(180^{\circ}-C\right)=\sin C.$
  \item Ta có $A+B+C=180^{\circ}\Leftrightarrow A+B=180^{\circ}-C.$\\
  Từ đó suy ra $\cos\left(A+B\right)=\cos \left(180^{\circ}-C\right)=-\cos C \Rightarrow \cos\left(A+B\right)+\cos C=0.$
  \item Ta có $A+B+C=180^{\circ}\Leftrightarrow \dfrac{A+B}{2}=\dfrac{180^{\circ}-C}{2}=90^{\circ}-\dfrac{C}{2}.$\\
  Từ đó suy ra $\sin\dfrac{A+B}{2}=\sin\left(90^{\circ}-\dfrac{C}{2}\right)= \cos\dfrac{C}{2}.$
  \item Ta có $\tan\left(A-B+C\right)=\tan\left(A+B+C-2B\right)=\tan\left(180^{\circ}-2B\right)=-\tan2B.$
\end{enumerate}
}
\end{vd}
