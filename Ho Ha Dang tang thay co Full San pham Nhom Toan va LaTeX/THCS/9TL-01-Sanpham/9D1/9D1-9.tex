\begin{bt}

		\item $\sqrt{4x+20}-3\sqrt{5+x}+\dfrac{4}{3}\sqrt{9x+45}=6$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Điều kiện $x\ge 0$. Khi đó
			\begin{eqnarray*}
				&&4\sqrt{x}-2\sqrt{9x}+ \sqrt{16x}= 5\\
				&\Leftrightarrow& 4\sqrt{x}- 6\sqrt{x}  + 4\sqrt{x} = 5\\
				&\Leftrightarrow& 2\sqrt{x}= 5\\
				&\Leftrightarrow& x= \dfrac{25}{4}. \text{\quad (thỏa điều kiện)}
			\end{eqnarray*}
			Vậy tập nghiệm của phương trình là $S=\left\{\dfrac{25}{4}\right\}$.
			\item Điều kiện $x\ge -5$. Khi đó 
			\begin{eqnarray*}
				&& \sqrt{4x+20}-3\sqrt{5+x}+\frac{4}{3}\sqrt{9x+45}=6\\
				&\Leftrightarrow& 2\sqrt{x+5}- 3\sqrt{x+5}+ 4\sqrt{x+5}= 6 \\
				&\Leftrightarrow& 3 \sqrt{5+x}=6 \\
				&\Leftrightarrow& \sqrt{x+5}= 2 \\
				&\Leftrightarrow& x+5=4\\
				&\Leftrightarrow& x=-1.  \text{\quad (thỏa điều kiện)}
			\end{eqnarray*}
			Vậy tập nghiệm của phương trình là $S=\left\{-1\right\}$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}
	Cho biểu thức $\left( \dfrac{\sqrt{x}}{\sqrt{x}-2}+\dfrac{\sqrt{x}}{\sqrt{x}+2} \right).\dfrac{x-4}{\sqrt{4x}}$.
	\begin{enumerate}
		\item Rút gọn biểu thức với $x>0$ và $x\ne 4$.
		\item Tính giá trị của $P$  khi $x = 3 - 2\sqrt{2}$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Với $x>0; x\ne 1; x\ne 4$, ta có
			\begin{eqnarray*}
				P&=&\left( \dfrac{\sqrt{x}}{\sqrt{x}-2}+\dfrac{\sqrt{x}}{\sqrt{x}+2} \right).\dfrac{x-4}{\sqrt{4x}}\\
				&=& \dfrac{\sqrt{x}(\sqrt{x}+2)+\sqrt{x}(\sqrt{x}-2)}{x-4}.\dfrac{x-4}{\sqrt{4x}}\\
				&=& 2\sqrt{x}.\sqrt{x}.\frac{1}{2\sqrt{x}}\\
				&=&\sqrt{x}.
			\end{eqnarray*}
			Vậy $P=\sqrt{x}$.
			\item Tính giá trị của $P$  khi $x = 3 - 2\sqrt{2}$.\\
			Thu gọn $ x =  3 - 2\sqrt{2}=\left(\sqrt{2}-1\right)^2$.\\
			Khi đó $P=\sqrt{x}=\sqrt{\left(\sqrt{2}-1\right)^2}=\sqrt{2}-1$.
		\end{enumerate}
	}
\end{bt}

\begin{bt}

			{\allowdisplaybreaks

				P &= \left( \dfrac{1}{1 - \sqrt{a} } - \dfrac{1}{1 + \sqrt{a} } \right)\left( \dfrac{1}{\sqrt{a} } + 1 \right)\\
				&= \dfrac{1 + \sqrt{a}  - 1 + \sqrt{a} }{(1 - \sqrt{a} )(1 + \sqrt{a} )} \cdot \dfrac{1 + \sqrt{a} }{\sqrt{a} } = \dfrac{2\sqrt{a} }{(1 - \sqrt{a} )(1 + \sqrt{a} )} \cdot \dfrac{1 + \sqrt{a} }{\sqrt{a} } = \dfrac{2}{1 - \sqrt{a} }.
				\end{align*}}
			Vậy $P = \dfrac{2}{1 - \sqrt{a} }$.
			\item $a=9+4\sqrt{2}=\left(1+2\sqrt{2}\right)^2\Rightarrow \sqrt{a}=1+2\sqrt{2}$\\
			Vậy $P=\dfrac{2}{1 - \sqrt{a}}=\dfrac{2}{1-\left(1+2\sqrt{2}\right)}=-\dfrac{\sqrt{2}}{2}$.
			\item  Với $a>0$ và $a\ne 1$ thì điều kiện để $P > \dfrac{1}{2}$ là:
			{\allowdisplaybreaks
				\begin{align*}
				&\dfrac{2}{1 - \sqrt{a} } > \dfrac{1}{2} \Leftrightarrow \dfrac{2}{1 - \sqrt{a} } - \dfrac{1}{2} > 0 \Leftrightarrow \dfrac{4 - 1 + \sqrt{a} }{1 - \sqrt{a} } > 0 \Leftrightarrow \dfrac{3 + \sqrt{a} }{1 - \sqrt{a} } > 0\\
				\Leftrightarrow& 1 - \sqrt{a}  > 0 \Leftrightarrow \sqrt{a}  < 1 \Leftrightarrow a < 1.
				\end{align*}}
			Kết hợp với điều kiện $a>0$ ta được: $0<a<1$.
		\end{enumerate}
	}
\end{bt}
