\begin{bt}[T1/481-Toán học \& Tuổi trẻ, tháng 7 năm 2017]%[8D0G6]%[Nguyễn Thị Mai Sương]
	Tìm các số nguyên không âm $a,b,c$ thỏa mãn $9^a+952=(b+41)^2$ và $a=2^b \cdot c$.
	\loigiai{
		Xét hai trường hợp đối với $a$.
		\begin{itemize}
			\item Nếu số $a$ chẵn, $a=2n$ thì $9^a+952=81^n+952$ có chữ số tận cùng là $3$, nhưng số chính phương bất kì không có chữ số tận cùng là $3$ nên bài toán vô nghiệm.
			\item Nếu số $a$ lẻ, $a=2n+1=2^b \cdot c$ thì $b=0$ và $a=c$ là số lẻ. Từ đó có $9^a+952=41^2=1681$ nên $9^a=729$, suy ra $a=3$.\\
			Bài toán có nghiệm $a=c=3,b=0$.
		\end{itemize}
		\begin{nx}
			Một số bạn biến đổi như sau:
			$$(b+41)^2-(3^a)^2=952 \Leftrightarrow (b+41-3^a)(b+41+3^a)=952.$$
			Do $(b+41-3^a)+(b+41+3^a)$ là số chẵn nên hai số nguyên $b+41-3^a$ và $b+41+3^a$ có cùng tính chẵn lẻ, đồng thời tích của chúng bằng 
			$$952=2\cdot 746=4 \cdot 238 =14 \cdot 68 =28 \cdot 34.$$
			Từ đó tính được kết quả. Cách này dài hơn lời giải trên.
		\end{nx}
	}
\end{bt}

\begin{bt}[T2/481, Toán học \& tuổi trẻ, tháng 7 năm 2017]%[6D0K8]%[Thái Văn Đạt]
	Chứng minh rằng  $$A=\dfrac{1}{1000}+\dfrac{1}{1002}+...+\dfrac{1}{2000}<\dfrac{1}{2}$$
	\loigiai
	{Ta xét:
		{\allowdisplaybreaks
			\begin{align*}
				B&=\dfrac{1}{500}+\dfrac{1}{501}+...+\dfrac{1}{1000}\\
				&=\left(\dfrac{1}{1}+\dfrac{1}{2}+...+\dfrac{1}{1000}\right)-\left(\dfrac{1}{1}+\dfrac{1}{2}+...+\dfrac{1}{499}\right)\\
				&=\left(\dfrac{1}{1}+\dfrac{1}{2}+...+\dfrac{1}{1000}\right)-\left(\dfrac{2}{2}+\dfrac{2}{4}+...+\dfrac{2}{998}\right)\\
				&=\dfrac{1}{1}-\dfrac{1}{2}+\dfrac{1}{3}-\dfrac{1}{4}+...+\dfrac{1}{997}-\dfrac{1}{998}+\dfrac{1}{999}+\dfrac{1}{1000}\\
				&=1-\left(\dfrac{1}{2}-\dfrac{1}{3}-\dfrac{1}{1000}\right)-\left(\dfrac{1}{4}-\dfrac{1}{5}\right)-\left(\dfrac{1}{6}-\dfrac{1}{7}\right)-...-\left(\dfrac{1}{998}-\dfrac{1}{999}\right)
		\end{align*}}
		Do $\dfrac{1}{2}-\dfrac{1}{3}-\dfrac{1}{1000}>0$, $\dfrac{1}{4}-\dfrac{1}{5}>0$,...,$\dfrac{1}{998}-\dfrac{1}{999}>0$ nên ta có: $B<1$.
		Vậy: $$A=\dfrac{1}{2}\left(\dfrac{1}{500}+\dfrac{1}{501}+...+\dfrac{1}{1000}\right)=\dfrac{1}{2}B<\dfrac{1}{2}$$
		\begin{nx}
			Bài toán này là một bài toán biến đổi và làm trội khá cơ bản. Cách giải trên biến đổi nhiều và chỉ sử dụng đánh giá ở bước cuối, dưới đây là một cách làm khác sử dụng làm trội ngay mà không cần biến đổi qua nhiều bước:\\
			Ta có:
			{\allowdisplaybreaks
				\begin{align*}
					B&=\dfrac{1}{1000}+\dfrac{1}{1002}+...+\dfrac{1}{1500}\\
					&<\dfrac{1}{1000}+\dfrac{1}{1000}+...+\dfrac{1}{1000}=\dfrac{251}{1000}<
					\dfrac{251}{753}=\dfrac{1}{3}\\
					C&=\dfrac{1}{1502}+\dfrac{1}{1504}+...+\dfrac{1}{2000}\\
					&<\dfrac{1}{1502}+\dfrac{1}{1502}+...+\dfrac{1}{1502}=\dfrac{250}{1502}<\dfrac{250}{1500}=\dfrac{1}{6}
			\end{align*}}
			Do đó: $A=B+C<\dfrac{1}{3}+\dfrac{1}{6}=\dfrac{1}{2}$.\\
			Việc tìm cách chia dãy ra để đánh giá làm trội cũng không quá khó khăn bằng cách làm sau đây:
			\begin{itemize}
				\item Gọi phần tử chia đoạn đánh giá là phần tử thứ $k$.
				\item Khi đó $k$ phần tử đầu sẽ đánh giá trội là $\dfrac{1}{1000}$ và $501-k$ phần tử còn lại đánh giá trội là $\dfrac{1}{1000+2(k-1)}$.
				\item Cuối cùng là chọn $k$ để có đánh giá $$\dfrac{k}{1000}+\dfrac{501-k}{1000+2(k-1)}<\dfrac{1}{2}$$.
			\end{itemize}
			Hiển nhiên đây là công việc của giáo viên để có thể chọn cách làm tối ưu và giúp ích cho việc ra bài tương tự hoặc có phần nâng cao hơn! 
		\end{nx}
	}
\end{bt}

\begin{bt}[T3/481, Toán học \& tuổi trẻ, tháng 7 năm 2017]%[8D0K6]%[Nguyễn Đức Lợi]
	Tìm nghiệm nguyên dương của phương trình $x^5+y^5+2016= (x+2017)^5 + (y-2018)^5$.
	\loigiai{
		\textbf{\textit{Cách 1.}}\\
		Với $a$ nguyên thì $a^5-a=a(a-1)(a+1)(a^2+1)$ chia hết cho $2$ nên $a^5$ và $a$ luôn cùng tính chẵn lẻ.\\
		Biến đổi phương trình đã cho thành
		\[\left[(x+2017)^5-x^5\right] - \left[y^5-(y-2018)^5\right]=2016 \tag{1}\]
		Nhận thấy với mọi số nguyên dương $x,y$ luôn có $(x+2017)$ và $x$ khác tính chẵn lẻ; $(y-2018)$ và $y$ cùng tính chẵn lẻ. Do đó $(x+2017)^5$ và $x^5$ khác tính chẵn lẻ; $(y-2018)^5$ và $y^5$ cùng tính chẵn lẻ. Suy ra $(x+2017)^5-x^5$ là số lẻ và $(y-2018)^5-y^5$ là số chẵn nên vế trái của $(1)$ là số lẻ, trong khi vế phải của $(1)$ là $2016$ (số chẵn). Vì vậy không tồn tại các số nguyên dương $x,y$ thỏa mãn $(1)$, phương trình đã cho vô nghiệm.\\
		\textbf{\textit{Cách 2.}}\\
		Trước hết ta chứng minh bổ đề 
		\begin{bode}
			Với $a$ nguyên ta luôn có $(a^5-a) ~ \vdots ~ 5$.\\
			Thật vậy, ta có
			\begin{align*}
				a^5-a & = a(a^2-1)(a^2+1)=a(a-1)(a+1)(a^2+5-4)\\
				& = 5(a-1)a(a+1) - (a-2)(a-1)a(a+1)(a+2)
			\end{align*}
			Vì $a-2, a-1, a, a+1, a+2$ là $5$ số nguyên liên tiếp nên có đúng một số chia hết cho $5$, do đó tích của chúng chia hết cho $5$; và $5(a-1)a(a+1) ~\vdots ~ 5$. Vì vậy $(a^5-a) ~ \vdots ~ 5$.
		\end{bode}
		Trở lại bài toán, phương trình đã cho được biến đổi thành 
		\[\left[(x+2017)^5-(x+2017)\right] + \left[(y-2018)^5 -(y-2018)\right]-(x^5-x) - (y^5-y)=2017 \tag{$*$}\]
		Áp dụng bổ đề, ta có vế trái của $(*)$ luôn chia hết cho $5$ với mọi $x,y$ nguyên dương. Còn vế phải của $(*)$ là $2017$ không chia hết cho $5$. Vì vậy phương trình đã cho vô nghiệm.
		\begin{nx}
			Đây là bài toán giải phương trình sử dụng tính chia hết để chứng tỏ phương trình vô nghiệm. Kết quả của bài toán vẫn đúng với yêu cầu tìm nghiệm nguyên của phương trình.
		\end{nx}
	}
\end{bt}

\begin{bt}[T5/481, Toán học \& tuổi trẻ, tháng 7 năm 2017]%[0D0K1]%[Lê Đình Mẫn]
	Giải phương trình $$(1-2x)\sqrt{2-x^2}=x-1.$$
	\loigiai{
		Điều kiện: $-\sqrt{2}\le x\le \sqrt{2}$. Đặt $\sqrt{2-x^2}=a\ (a\ge 0)\Rightarrow x^2+a^2=2$.\hfill $(1)$\\
		Phương trình đã cho trở thành $(1-2x)a=x-1\Leftrightarrow a-2ax=x-1$.\hfill $(2)$\\
		Cộng theo vế phương trình $(1)$ và $(2)$ ta được $(a-x)^2+(a-x)-1=0$.\hfill $(3)$\\
		Giải phương trình $(3)$ ta được $a-x=\dfrac{-1\pm\sqrt{5}}{2}\Rightarrow \sqrt{2-x^2}-x=\dfrac{-1\pm\sqrt{5}}{2}$.\\
		\textit{Trường hợp 1:} $\sqrt{2-x^2}-x=\dfrac{-1-\sqrt{5}}{2}\Leftrightarrow \sqrt{2-x^2}=x-\dfrac{1+\sqrt{5}}{2}$. Do $x\le\sqrt{2}$ nên $x-\dfrac{1+\sqrt{5}}{2}<0$. Trường hợp này loại.\\
		\textit{Trường hợp 2:} $\sqrt{2-x^2}-x=\dfrac{-1+\sqrt{5}}{2}\Leftrightarrow\heva{&x\ge\dfrac{1-\sqrt{5}}{2}\\& 2-x^2=\left(x-\dfrac{1-\sqrt{5}}{2}\right)^2.\hspace{4.3cm} (4)}$\\
		Giải phương trình $(4)$ ta được $x=\dfrac{1-\sqrt{5}\pm\sqrt{10+2\sqrt{5}}}{4}$.\\
		Kết hợp với điều kiện bài toán ta có $x=\dfrac{1-\sqrt{5}+\sqrt{10+2\sqrt{5}}}{4}$ là nghiệm duy nhất của phương trình.
		\begin{nx}
			Một số bạn, khi biến đổi tương đương không đối chiếu điều kiện hoặc biến đổi hệ quả mà không thử lại, nên dẫn đến kết quả thừa nghiệm. Một số bạn giải phương trình đã cho bằng phương pháp nâng lũy thừa đưa đến phương trình bậc cao, nhưng cách làm này thiếu tự nhiên và khá khó khăn trong trường hợp nghiệm lẻ.
		\end{nx}
	}
\end{bt}

\begin{bt}[T6/481-Toán học \& Tuổi trẻ, tháng 7 năm 2017]%[2D0G2]%[Nguyễn Tài Chung]
	Giải hệ phương trình 
	$$\left\{ \begin{array}{ll}
	\dfrac{x + \sqrt {1 + x^2}}{1 + \sqrt {1 + y^2}} = x^2y&(1)\\
	1 + y^4 - \dfrac{4y}{x} + \dfrac{3}{x^4} = 0.&(2)
	\end{array} \right.$$
	\loigiai{
		Điều kiện: $x\ne 0$. Do $x + \sqrt {1 + x^2}  > x + \left| x \right| \ge 0$ nên từ $(1)$ suy ra $y>0$, kết hợp với $(2)$, ta được $\dfrac{y}{x}>0$, suy ra $x>0$. Biến đổi $(1)$, ta được
		{\allowdisplaybreaks
			\begin{align*}
				&\frac{x + \sqrt {1 + x^2}}{x^2} = y\left( 1 + \sqrt {1 + y^2}  \right)\\
				\Leftrightarrow& \frac{1}{x} + \frac{1}{x}\sqrt {1 + \frac{1}{x^2}}  = y + y\sqrt {1 + y^2} \\
				\Leftrightarrow& f\left( \frac{1}{x} \right) = f(y),\,\, \text{với} \,\,f(t) = t + t\sqrt {1 + t^2}.\tag{3}
		\end{align*}}Hàm số $f(t) = t + t\sqrt {1 + t^2} $ có $f'(t) = 1 + \sqrt {1 + t^2}  + \dfrac{t^2}{\sqrt {1 + t^2}} > 0,\,\forall t > 0$ nên đồng biến trên khoảng $(0; +\infty)$. Do đó từ $(3)$ ta có $y = \dfrac{1}{x}$. Thay vào $(2)$ ta được
		Ta có
		{\allowdisplaybreaks
			\begin{align*}
				&1 + \frac{1}{x^4} - \frac{4}{x^2} + \frac{3}{x^4} = 0 \Leftrightarrow 1 - \frac{4}{x^2} + \frac{4}{x^4} = 0 \Leftrightarrow \left( 1 - \frac{2}{x^2} \right)^2 = 0\\
				\Leftrightarrow& 1 - \frac{2}{x^2} = 0 \Leftrightarrow x^2 = 2 \Leftrightarrow \left[ \begin{array}{ll}x = \sqrt 2&(\text{thỏa điều kiện}) \\
					x =  - \sqrt 2&(\text{loại})
				\end{array} \right.
		\end{align*}}Khi đó $y = \dfrac{\sqrt 2}{2}$. Vậy nghiệm của hệ là
		$\left( x;y \right) = \left( \sqrt 2 ;\dfrac{\sqrt 2 }{2} \right)$.
		\begin{nx}
			Khi biến đổi $\dfrac{1}{x}\sqrt {1 + x^2}  = \sqrt {1 + \dfrac{1}{x^2}} $, cần có điều kiện $x>0$. Để chứng minh $y = \dfrac{1}{x}$ có thể sử dụng phương pháp nhân lượng liên hợp nhưng lời giải sẽ dài hơn.
		\end{nx}	
	}
\end{bt}

\begin{bt}[T7/481-Toán học \& Tuổi trẻ, tháng 7 năm 2017]%[1D0K6]%[Nguyễn Hữu Nhanh Tiến]
	Cho số nguyên dương $n>1$. Chứng minh rằng $\dfrac{2^{2n}}{2\sqrt{n}}<\mathrm{C}_{2n}^n<\dfrac{2^{2n}}{\sqrt{2n+1}}$.
	\loigiai{Đặt $P=\dfrac{1\cdot3\cdot5 \ldots (2n-1)}{2\cdot4\cdot6 \ldots (2n)}$. 		Dễ thấy $\mathrm{C}_{2n}^n=\dfrac{(2n)!}{(n!)^2}=2^{2n}P$.
		Ta có
		$$1>\left(1-\dfrac{1}{2^2}\right)\left(1-\dfrac{1}{4^2}\right)\cdots \left(1-\dfrac{1}{(2n)^2}\right)=\dfrac{(1.3).(3.5) \ldots \left[\left(2n-1\right)\left(2n+1\right)\right]}{2^2.4^2 \ldots (2n)^2}.$$
		Suy ra $1>\left(2n+1\right)P^2$. 
		Từ đó suy ra $P<\dfrac{1}{\sqrt{2n+1}}$. Do đó
		$\mathrm{C}_{2n}^n<\dfrac{2^{2n}}{\sqrt{2n+1}}$. 
		Tương tự: 
		$$1>\left(1-\dfrac{1}{3^2}\right)\left(1-\dfrac{1}{5^2}\right)\cdots \left(1-\dfrac{1}{\left(2n-1\right)^2}\right)=\dfrac{(2.4)(4.6)\ldots \left[(2n-2)(2n)\right]}{3^2.5^2\ldots(2n-1)^2} \Rightarrow 1>\dfrac{1}{4nP^2}.$$
		Từ đó suy ra $P>\dfrac{1}{\sqrt{4n}} \Rightarrow \mathrm{C}_{2n}^n>\dfrac{2^{2n}}{2\sqrt{n}}$.
		
		\begin{nx}
			Bài này có nhiều cách giải. Chẳng hạn là có thể quy nạp khá hiển nhiên.
		\end{nx}
	}
\end{bt}

\begin{bt}[T8/481-Toán học \& Tuổi trẻ, tháng 7 năm 2017]%[9H0G4]%[Nguyễn Ngọc Dũng]
	Cho tứ giác $ABCD$ ngoại tiếp đường tròn $(O)$. $E,F,G$ theo thứ tự là giao điểm của các cặp đường thẳng $(AB,CD)$,$(AD,CB)$, $(AC,BD)$. $K$ là giao điểm của $EF$ và $OG$. Chứng minh rằng $\widehat{AKG}=\widehat{CKG}$, $\widehat{BKG}=\widehat{DKG}$.

\begin{bt}[T10/481, Toán học \& học tuổi trẻ, tháng 7 năm 2017]%[9D1K2]%[Trần Bá Huy]
	Cho dãy số $(u_n)$ thỏa mãn $\heva{& u_1=0\\ & \log_{\tfrac{1}{4}}u_{n+1}=\left(\dfrac{1}{4}\right)^{u_n}},\ \forall n\in\mathbb{N^*}.$\\
	Chứng minh rằng dãy số có giới hạn hữu hạn và tính giới hạn đó khi $n$ dần đến vô cực.
	\loigiai{
		Từ giả thiết đã cho, suy ra $u_n>0,\ n=2,3,\ldots$\\
		Tiếp theo, ta chứng minh $u_n<\dfrac{1}{2},\ \forall n\in\mathbb{N^{*}}$.\\
		Với $n=1$ thì $u_1=0<\dfrac{1}{2}$.\\
		Giả sử $u_k<\dfrac{1}{2}$ thì từ $\log_{\tfrac{1}{4}}u_{n+1}=\left(\dfrac{1}{4}\right)^{u_n}>\left(\dfrac{1}{4}\right)^{\tfrac{1}{2}}=\dfrac{1}{2}$, suy ra $u_{n+1}<\left(\dfrac{1}{4}\right)^{\tfrac{1}{2}}=\dfrac{1}{2}$. \\
		Xét hàm số $f(x)=\left(\dfrac{1}{4}\right)^x-\log_{\tfrac{1}{4}}x$, với $x\in\left(0;\dfrac{1}{2}\right]$.
		Khi đó
		$$\begin{aligned}
		f'(x)& =\left(\dfrac{1}{4}\right)^x\ln\dfrac{1}{4}-\dfrac{1}{x\ln\dfrac{1}{4}}>\ln\dfrac{1}{4}-\dfrac{1}{x\ln\dfrac{1}{4}}\\
		&>\dfrac{2}{\ln\dfrac{1}{4}}-\dfrac{1}{x\ln\dfrac{1}{4}}>\dfrac{1}{x\ln\dfrac{1}{4}}-\dfrac{1}{x\ln\dfrac{1}{4}}=0,\ \forall x\in\left(0;\dfrac{1}{2}\right].
		\end{aligned}$$
		Vậy $f(x)$ là hàm đồng biến trên $\left(0;\dfrac{1}{2}\right]$ nên $f(x)<f\left(\dfrac{1}{2}\right)=0,\ \forall x\in\left(0;\dfrac{1}{2}\right]$.
		Từ đây suy ra $u_{n+1}>u_n$, tức $(u_n)$ là dãy tăng và bị chặn trên nên có giới hạn hữu hạn, giả sử là $a$. Khi đó, chuyển qua giới hạn, ta thu được $\log_{\tfrac{1}{4}}a=\left(\dfrac{1}{4}\right)^a\Leftrightarrow a=\dfrac{1}{2}$, do $f(x)$ là hàm đồng biến trên $\left(0;\dfrac{1}{2}\right]$ nên nghiệm đó là duy nhất.
	\begin{nx}\hfill
	\begin{enumerate}
		\item Đây là dạng toán cơ bản về dãy số sinh bởi hàm mũ, cần viện trợ các kiến thức về đạo hàm của hàm số siêu việt. 
		\item Có thể tìm được giới hạn thông qua tính chất hàm ngược.
	\end{enumerate}
\end{nx}
}
\end{bt}

		\Rightarrow S\leq\dfrac{1}{6}\log_2\left(\dfrac{4}{3}\right)^3=\dfrac{1}{2}\log_2\dfrac{4}{3}.$$
