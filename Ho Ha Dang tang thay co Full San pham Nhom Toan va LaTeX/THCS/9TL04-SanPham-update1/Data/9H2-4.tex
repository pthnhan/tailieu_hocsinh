\begin{vd}
	Cho đường tròn $ (O) $, điểm $ A $ nằm bên ngoài đường tròn. Dựng đường thẳng đi qua $ A $, cắt đường tròn ở $ B $ và $ C $ sao cho tổng $ AB + AC $ có giá trị lớn nhất.
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
				\tkzInit[ymin=-3.05,ymax=3.05,xmin=-6,xmax=5]

				\tkzInterLC(A,K)(O,L)\tkzGetSecondPoint{B}
				\tkzDefPointBy[projection = onto A--K](O)\tkzGetPoint{M}
				
				\tkzDrawPoints(O,A,B,C,M)
				\tkzDrawSegments(A,K A,O O,M)
				\tkzLabelPoints[left](A)
				\tkzLabelPoints[above](O)
				\tkzLabelPoints[below](M)
				\tkzLabelPoints[below left](B)
				\tkzLabelPoints[below right](C)
				\tkzMarkRightAngles[size=0.2](O,M,A)

			\end{tikzpicture}
		\end{center}
		Gọi $ M $ là trung điểm $ BC $, ta có $ OM \perp BC $ và $ MB = MC $. Suy ra
		\begin{align*}
			AB + AC = AM - MB + AM + MC = 2AM.
		\end{align*}
		Nên $ AB + AC $ lớn nhất khi $ AM $ lớn nhất. Mà
		\begin{align*}
			AM^2 = OA^2 - OM^2.
		\end{align*}
		Nên, $ AM $ lớn nhất khi $ OM $ nhỏ nhất $ \Leftrightarrow M \equiv O $. Vậy $ AB + AC $ lớn nhất khi đường thẳng đi qua $ A $ đi qua tâm $ O $.
	}
\end{vd}

\begin{bt}
	Cho đường tròn $ (O;R) $ có dây $ AB = R $. Trên tia $ AB $ lấy điểm $ M $ sao cho $ AM = a $. Qua $ M $ vẽ đường thẳng $ xy $ vuông góc với $ AB $. Chứng minh rằng đường thẳng $ xy $ và đường tròn $ (O;R) $ chỉ có điểm chung khi $ a \le \dfrac{3R}{2} $.
	\loigiai{
		\immini{
			Gọi $ N $ là trung điểm $ AB $. Ta có $ ON \perp AM $ và $ NM = a - \dfrac{R}{2} $.\\
			Gọi $ H $ là chân đường vuông góc dựng từ $ O $ xuống $ xy $, ta có $ OH \perp xy \Rightarrow ONMH $ là hình chữ nhật, do đó
			\begin{align*}

			& EF^2 = FC^2 + CE^2 = \dfrac{a^2}{4} + \dfrac{4a^2}{9} = \dfrac{25a^2}{36}\\

			S_{\triangle AEF}
