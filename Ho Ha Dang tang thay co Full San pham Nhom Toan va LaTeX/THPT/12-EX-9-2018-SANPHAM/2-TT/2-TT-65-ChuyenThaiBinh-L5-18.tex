}

\begin{ex}%[TT lần 5, Chuyên Thái Bình, 2018]%[Vũ Nguyễn Hoàng Anh, 12EX9]%[0D4K1-5]

&\Rightarrow &\dfrac{\left[ 3-g(x_0)+f(x_0) \right]}{\left[ 3-g(x_0) \right]^2}=1.
\end{eqnarray*}
Đặt $t=3-g(x_0)$ ta có $t^2-t-f(x_0)=0$. Phương trình này có nghiệm nên $1+4f(x_0)\ge 0$ hay $f(x_0)\ge-\dfrac{1}{4}$.
}
\end{ex}

{$P_{\min}=\dfrac{27}{5}$}

\Rightarrow &\ \left[ (x+1)(y+1) \right]^{y+1}=\dfrac{3^{9+2(y+1)}}{3^{(x+1)(y+1)}}\\
\Rightarrow &\ 3^{(x+1)(y+1)}\left[ (x+1)(y+1) \right]^{y+1} =3^9\cdot 9^{(y+1)}.\tag{1}

\begin{ex}%[TT lần 5, Chuyên Thái Bình, 2018]%[Vũ Nguyễn Hoàng Anh, 12EX9]%[2D1G2-5]

\begin{ex}%[TT lần 5, Chuyên Thái Bình, 2018]%[Vũ Nguyễn Hoàng Anh, 12EX9]%[2D3G2-4]
