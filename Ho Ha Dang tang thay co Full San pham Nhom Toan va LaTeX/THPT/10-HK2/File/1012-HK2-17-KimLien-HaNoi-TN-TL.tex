\begin{bt}%[HK2 (2016-2017), THPT Kim Liên, Hà Nội]%[0D4B5]%[Đào Trung Kiên]
	Giải bất phương trình: $1-x+\sqrt{2x^2-3x-5}\geq 0$.
	\loigiai{$1-x+\sqrt{2x^2-3x-5}\geq 0\Leftrightarrow \hoac{&\heva{&x-1<0\\& 2x^2-3x-5\geq 0}\\&\heva{&x-1\geq 0\\& 2x^2-3x-5\geq (x-1)^2}}\Leftrightarrow \hoac{&x\leq -1\\& x\geq 3}.$\\
	Vậy bất phương trình có tập nghiệm $(-\infty; -1]\cup[3; +\infty)$.}
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Kim Liên, Hà Nội]%[0D6K2]%[Đào Trung Kiên]
	Cho $\cot \alpha=\dfrac{1}{2},\left(\pi<
	\alpha<\dfrac{3\pi}{2}\right)$. Tính giá trị của biểu thức:
	$$P=\sin(\pi-\alpha)\cdot\sin\left(\dfrac{3\pi}{2}+\alpha\right)\cdot\cos\left(2017\pi+\alpha\right).$$
	\loigiai{Ta có $\sin^2\alpha=\dfrac{1}{1+\cot^2\alpha}=\dfrac{4}{5}$. Vì $\left(\pi<
		\alpha<\dfrac{3\pi}{2}\right)$ nên $\sin\alpha <0\Rightarrow $ $\sin\alpha=-\dfrac{2}{\sqrt{5}}$.\\
	Vì $\sin\left(\pi-\alpha\right)=\sin\alpha$, $\sin\left(\dfrac{3\pi}{2}+\alpha\right)=-\cos\alpha$, $\cos\left(2017\pi+\alpha\right)=-\cos\alpha$ nên ta có:\\
$P=\sin\alpha\cdot \cos^2\alpha=\sin\alpha\cdot \left(1-\sin^2\alpha\right)=-\dfrac{2\sqrt{5}}{25}$. }
\end{bt}

\begin{bt}%[HK2 (2016-2017), THPT Kim Liên, Hà Nội]%[0H3K1]%[Đào Trung Kiên]
	Trong mặt phẳng toạ độ $Oxy$ cho các điểm $A(1;-1)$, $B(-1;-3)$, $I(2;4)$. Viết phương trình đường thẳng $(d)$ song song với đường thẳng $AB$ và cắt các trục $Ox, Oy$ lần lượt tại $C, D$ sao cho diện tích tam giác $OCD$ bằng $2$.
	\loigiai{Từ giả thiết suy ra đường thẳng $AB$ có phương trình $x-y-2=0$.\\
	Vì $d$ song song với $AB$ nên $d: x-y+m=0, (m\neq -2)$. Khi đó $d$ cắt $Ox, Oy$ lần lượt tại $C(-m; 0)$ và $D(0; m)$.\\
Diện tích tam giác $OCD$ bằng $2$ $\Leftrightarrow OC\cdot OD=4\Leftrightarrow m^2=4\Leftrightarrow \hoac{&m=2\\&m=-2 (\textrm{ loại})}$.\\
Vậy đường thẳng $d$ có phương trình $x-y+2=0$.}
\end{bt}
