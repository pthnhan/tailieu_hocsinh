\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2012-2013, Tp. Đà Nẵng]%[9D1G7]%[9D5G2]
	Cho biểu thức $P=\dfrac{\sqrt{n+1}-1}{\sqrt{n+1}+1}+\dfrac{\sqrt{n+1}+3}{\sqrt{n+1}-3}-\dfrac{n-\sqrt{n+1}+7}{n-2\sqrt{n+1}-2}$ với $n \in \mathbb N,n \neq 8$
    \begin{enumerate}
    \item Rút gọn biểu thức $Q = \dfrac{P}{n + 3\sqrt{n+1} + 1}$ với $n \in \mathbb N,n \neq 8$.
    \item Tìm tất cả các giá trị của $n\quad(n \in \mathbb N,n \neq 8)$ sao cho $P$ là một số nguyên tố.
    \end{enumerate}
	\loigiai{
		\begin{enumerate}
		\item Quy đồng $P = \dfrac{(\sqrt{n+1}-1)(\sqrt{n+1}-3)+(\sqrt{n+1}+3)(\sqrt{n+1}+1)-(n-\sqrt{n+1}+7)}{(\sqrt{n+1}+1)(\sqrt{n+1}-3)}$.\\
        Do đó $P = \dfrac{\sqrt{n+1}(\sqrt{n+1}+1)}{(\sqrt{n+1}+1)(\sqrt{n+1}-3)} = \dfrac{\sqrt{n+1}}{\sqrt{n+1}-3}$.\\
        Suy ra $Q = \dfrac{P}{\sqrt{n+1}(\sqrt{n+1}+3)} = \dfrac{1}{n-8}$.
        \item $P = 1 + \dfrac{3}{\sqrt{n+1} - 3}$. Để $P$ là số nguyên thì $\sqrt{n+1} - 3$ là ước của $3$.\\
        Suy ra $\sqrt{n+1} - 3 \in \{ \pm 1; \pm 3 \} \Rightarrow n \in \{-1;3;15;35\}$. Thử lại ta thấy $n = 35$ thì $P = 2$ là số nguyên tố.
		\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2012-2013, Tp. Đà Nẵng]%[9D1G1]%[9D1G1]
	\begin{enumerate}
	\item Tìm $x$ biết $2\sqrt{x+4} - 4\sqrt{2x-6} = x - 7$.
    \item Giải hệ phương trình $\begin{cases} x + 6 = 4\sqrt{y-4}\\ y + 10 = 6\sqrt{z-9} \\ z - 16 = 2\sqrt{x-1} \end{cases}$.
	\end{enumerate}
	\loigiai{
    \begin{enumerate}
    \item Điều kiện $-4 \le x \le 3$\\
    Phương trình đã cho tương đương với $(\sqrt{x+4}+1)^2 = (\sqrt{2x-6}+2)^2$.\\
    Do đó ta có $\left[\begin{aligned}
    \sqrt{x+4}+1&=\sqrt{2x-6}+2\\
    \sqrt{x+4}+1&=-\sqrt{2x-6}-2 \text{ (vô lý)}
    \end{aligned}\right.\Leftrightarrow x = 5$.
    \item Điều kiện $x\ge 1$, $y\ge 4$, $z\ge 9$.\\
    Cộng theo vế ta có $x + y + z - 2\sqrt{x-1} - 4\sqrt{y-4} - 6\sqrt{z-9} = 0$.\\
        Phương trình trên tương đương với $(\sqrt{x-1} - 1)^2 + (\sqrt{y-4}-2)^2+(\sqrt{z-9}-3)^2=0$.\\
        Do đó $\begin{cases} \sqrt{x-1} - 1 = 0 \\ \sqrt{y-4}-2 = 0 \\ \sqrt{z-9}-3 = 0 \end{cases}\Leftrightarrow \begin{cases} x = 2 \\ y = 8 \\ z = 18 \end{cases}$. Thử lại thấy thỏa mãn.
    \end{enumerate}
		
	}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2012-2013, Tp. Đà Nẵng]%[9D2G3]%[9D3G3]
	\begin{enumerate}
	\item Cho hàm số bậc nhất $y = ax+b$ có đồ thị đi qua điểm $M(1;4)$. Biết rằng đồ thị của hàm số đã cho cắt trục $Ox$ tại điểm $P$ có hoành độ dương và cắt trục $Oy$ tại điểm $Q$ có tung độ dương. Tìm $a$ và $b$ sao cho $OP + OQ$ nhỏ nhất (với $O$ là gốc tọa độ).
    \item Tìm số tự nhiên có $2$ chữ số. Biết rằng nếu lấy tổng của $2$ chữ số ấy cộng với $3$ lần tích của $2$ chữ số ấy thì bằng $17$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
		\item Giả sử $P$ có tọa độ $(p;0)$ và $Q$ có tọa độ $(0;q)$. Khi đó ta có $\begin{cases} 0 = a p + b \\ q = a.0 + b \end{cases} \Rightarrow \begin{cases} b = q \\ a = -\dfrac{q}{p} \end{cases}$.\\
        Đồ thị hàm số đã cho đi qua $M(1;4)$ nên $4 = -\dfrac{q}{p}.1 + q \Rightarrow 1 = \dfrac{1}{p} + \dfrac{4}{q} \geq \dfrac{(1+2)^2}{p+q} $.\\
        Do đó $OP + OQ = p + q \geq 9$. Dấu bằng xảy ra khi $\dfrac{1}{p} = \dfrac{4}{q} = \dfrac{1}{2} \Leftrightarrow \begin{cases}p = 2 \\ q = 8\end{cases}$.\\
        Vậy $\begin{cases}a = -4 \\ b = 8\end{cases}$
        \item Gọi số cần tìm là $\overline{ab}$ ($0<a\le 9$, $0\le b \le 9$, $a,b \in \mathbb N$).\\
        Khi đó ta có phương trình $a+b+3ab=17 \Rightarrow a = \dfrac{17-b}{1+3b}$. Thử các giá trị của $b$ từ $0$ đến $9$ ta nhận được các số tự nhiên sau: $14$ và $41$.
		\end{enumerate}
    }
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2012-2013, Tp. Đà Nẵng]%[8H3K7]%[8H3K7]
	Cho tam giác $ABC$. Gọi $I$ là tâm đường tròn nội tiếp tam giác $ABC$, qua $I$ vẽ đường thẳng vuông góc với đường thẳng $CI$, đường thẳng này cắt các cạnh $AC$, $BC$ lần lượt tại $M$, $N$.
    \begin{enumerate}
    \item Chứng minh rằng hai tam giác $IAM$ và $BAI$ đồng dạng.
    \item Chứng minh rằng $\dfrac{AM}{BN} = \left( \dfrac{AI}{BI} \right)^2$.
    \end{enumerate}
    \loigiai{
    \begin{tikzpicture}[scale=1.3]
    	\tkzDefPoints{0/0/A,-1/-2/B,3/-2/C}
        \tkzInCenter(A,B,C)\tkzGetPoint{I}
         \tkzDefPointBy[rotation=center I angle 90](C)\tkzGetPoint{E}
         \tkzInterLL(A,C)(I,E)\tkzGetPoint{M}
         \tkzInterLL(B,C)(I,E)\tkzGetPoint{N}
         \tkzLabelPoints(I)
         \tkzLabelPoints[below](B,N,C)
         \tkzLabelPoints[above](M,A)
        \tkzDrawSegments(A,B B,C C,A I,A I,B I,C M,N)
    \end{tikzpicture}
    \begin{enumerate}
    \item Ta có\\
    $\widehat{AIC} = 180^\circ - \widehat{IAC} - \widehat{ICA} = 180^\circ - \dfrac{\widehat{BAC}}{2} - \dfrac{\widehat{BCA}}{2} = \dfrac{360^\circ - \widehat{BAC} - \widehat{BCA}}{2} = \dfrac{180^\circ + \widehat{ABC}}{2}$.\\
        Suy ra $\widehat{AIC} - 90^\circ = \dfrac{\widehat{ABC}}{2} \Rightarrow \widehat{AIM} = \widehat{ABI}$. Từ đó suy ra $\triangle IAM \backsim \triangle BAI\quad(g-g)$.
    \item Tương tự ta cũng có $\triangle BIN \backsim \triangle BAI$.\\
    Từ đó suy ra $\dfrac{IN}{BN} = \dfrac{AI}{BI} \quad (1)$.\\
    Ngoài ra từ câu a) ta có $\dfrac{AM}{IM} = \dfrac{AI}{BI} \quad (2)$.\\
    Mặt khác $\triangle CMN$ có $CI$ là đường phân giác cũng là đường cao nên cân tại $C$.\\
    Suy ra $CI$ cũng là trung tuyến. Do đó $IM = IN \quad(3)$.\\
    Từ (1), (2) và (3) ta có điều cần chứng minh.
    \end{enumerate}
    }
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2012-2013, Tp. Đà Nẵng]%[9H3G7]
	Cho tam giác $ABC$ có $\widehat{BAC}$ là góc tù. Vẽ các đường cao $CD$ và $BE$ của tam giác $ABC$ ($D$ nằm trên đường thẳng $AB$, $E$ nằm trên đường thẳng $AC$). Gọi $M,N$ lần lượt là chân đường vuông góc của $B$ và $C$ lên $DE$. Biết rằng $S_1$ là diện tích tam giác $ADE$, $S_2$ là diện tích tam giác $BEM$ và $S_3$ là diện tích tam giác $CDN$. Tính diện tích tam giác $ABC$ theo $S_1$, $S_2$, $S_3$.
    \loigiai{
    \immini{Ta có tứ giác $BCDE$ nội tiếp nên $\widehat{BDM} = \widehat{BCE}$ và $\widehat{CDN} = \widehat{CBE}$.\\
        Suy ra $\triangle BDM \backsim \triangle BCE$ và $\triangle DCN \backsim \triangle BCE$.\\
        Do đó $\dfrac{S_3}{S_{EBC}} = \dfrac{CD^2}{BC^2}$ và $\dfrac{S_{BDM}}{S_{EBC}} = \dfrac{BD^2}{BC^2}$.\\
        Như vậy ta có $\dfrac{S_3+S_{BDM}}{S_{EBC}} = 1 \Rightarrow S_3 + S_{BDM} = S_{EBC}$\\
        Do đó $S_3 + S_1 + S_2 + S_{BAE} = S_{ABC} + S_{BAE}$.\\
        Nên $S_{ABC} = S_1 + S_2 + S_3$.}{\begin{tikzpicture}[scale=0.7]
    \tkzDefPoints{0/0/A,-2/-2/B,4/-2/C}
    \tkzOrthoCenter(A,B,C)\tkzGetPoint{H}
    \tkzInterLL(A,B)(C,H)\tkzGetPoint{D}
    \tkzInterLL(C,A)(B,H)\tkzGetPoint{E}
    \tkzDefPointBy[projection=onto D--E](B)\tkzGetPoint{M}
    \tkzDefPointBy[projection=onto D--E](C)\tkzGetPoint{N}
    \tkzDrawSegments(A,B B,C C,A B,E C,D A,E A,D M,N B,M C,N)
    \tkzLabelPoints[left](M,B)
    \tkzLabelPoints[right](N,C)
    \tkzLabelPoints[above](D,E)
    \tkzLabelPoints[below](A)
    \tkzMarkRightAngles(B,E,C B,D,C B,M,E D,N,C)
    \end{tikzpicture}}
    }
\end{ex}
