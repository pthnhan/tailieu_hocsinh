\begin{ex}%[KSCL,Toán12, Sở Giáo dục và Đào tạo, Quảng Nam năm 2018]%[Hồ Như Vương, EX9]%[2D4G5-2]

	\choice

	{$2+\sqrt{3}$}
	{$4+\dfrac{14}{\sqrt{15}}$}
	{$2+\dfrac{7}{\sqrt{15}}$}
	\loigiai{

		Khi đó
		$$P=2\sqrt{(x+1)^2+ y^2}+2\sqrt{(x-1)^2+y^2}+2|y-2|=2\left( \sqrt{(x+1)^2+ y^2}+\sqrt{(1-x)^2+y^2}\right) +2|y-2|.$$
		Bằng phép biến đổi tương đương với chú ý $|x|\geq x$, ta có: Với mọi số thực $a, b, c, d$,
