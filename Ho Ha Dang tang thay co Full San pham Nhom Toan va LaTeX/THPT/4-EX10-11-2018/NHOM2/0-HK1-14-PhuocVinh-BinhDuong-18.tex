\begin{bt}%[Đề KSCL T10, Phước Vĩnh, Bình Dương 2017]%[0D3K2]
	Cho phương trình $3x^2+2(3m-1)x+3m^2-m+1=0,$ $m$ là tham số. Tìm $m$ để phương trình có hai nghiệm phân biệt $x_1,x_2$ thỏa mãn $x_1^2+x_2^2=\dfrac{39}{4}.$
	
	\loigiai{Ta có:\\
		$\begin{aligned}
		\Delta'&=\left(3m-1\right)^2-3\left(3m^2-m+1\right)\\
				   &=-3m-2
		\end{aligned}$\\
		Giá trị $m$ thỏa yêu cầu bài toán khi và chỉ khi:\\
		$\heva{&\Delta ' > 0\\&x_1^2 + x_2^2 = \dfrac{34}{9}} \Leftrightarrow \heva{&- 3m - 2 > 0\\&{m^2} - m - 2 = 0} \Leftrightarrow 
		\heva{&m <  - \dfrac{2}{3}\\& \hoac{&m =2 \\&m =-1}}$\\
		Vậy, $m=-1.$
	}
\end{bt}

				&=\vec{MB}\cdot\vec{MD}=VP

\end{bt}
