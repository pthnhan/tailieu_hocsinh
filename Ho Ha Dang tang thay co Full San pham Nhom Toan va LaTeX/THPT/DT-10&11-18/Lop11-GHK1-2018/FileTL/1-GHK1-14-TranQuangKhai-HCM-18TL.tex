\begin{bt}%[Thi GHK, Trần Quang Khải - Hồ Chí Minh, 2018]%[Phạm An Bình, EX-DT-10&11]%[1D1B3-1]
	Giải phương trình sau $\cos 2x-5\sin x-3=0$.
	\loigiai{
		\begin{eqnarray*}
			\cos 2x-5\sin x-3=0&\Leftrightarrow &-2\sin^2 x-5\sin x-2=0\\
			&\Leftrightarrow &\hoac{&\sin x=-2&\text{ (loại)}\\&\sin x=-\dfrac{1}{2}}\\
			&\Leftrightarrow &\hoac{&x=-\dfrac{\pi}{6}+k2\pi \\&x=\dfrac{7\pi}{6}+k2\pi}(k\in \mathbb{Z}).
		\end{eqnarray*}
	}
\end{bt}
\begin{bt}%[Thi GHK, Trần Quang Khải - Hồ Chí Minh, 2018]%[Phạm An Bình, EX-DT-10&11]%[1D1B3-3]
	Giải phương trình sau $\sqrt{3}\sin 2x-\cos 2x=2$.
	\loigiai{
		\begin{eqnarray*}
			\sqrt{3}\sin 2x-\cos 2x=2&\Leftrightarrow &\dfrac{\sqrt{3}}{2}\sin 2x-\dfrac{1}{2}\cos 2x=1\\
			&\Leftrightarrow &\sin \left( 2x-\dfrac{\pi}{6} \right) =1\\
			&\Leftrightarrow &2x-\dfrac{\pi}{6}=\dfrac{\pi}{2}+k2\pi\ (k\in \mathbb{Z})\\
			&\Leftrightarrow &x=\dfrac{\pi}{3}+k\pi\ (k\in \mathbb{Z}).
		\end{eqnarray*}
	}
\end{bt}
\begin{bt}%[Thi GHK, Trần Quang Khải - Hồ Chí Minh, 2018]%[Phạm An Bình, EX-DT-10&11]%[1D1K3-5]
	Giải phương trình sau $9\sin x+6\cos x-3\sin 2x+\cos 2x=8$.
	\loigiai{
		\begin{eqnarray*}
			& &9\sin x+6\cos x-3\sin 2x+\cos 2x=8\\
			&\Leftrightarrow &9\sin x+6\cos x-6\sin x\cos x+2\cos^2 x-1=8\\
			&\Leftrightarrow &9(\sin x-1)+6\cos x(1-\sin x)+2(1-\sin x)(1+\sin x)=0\\
			&\Leftrightarrow &(\sin x-1)(9-6\cos x-2-2\sin x)=0\\
			&\Leftrightarrow &\hoac{&\sin x=1\\&2\sin x+6\cos x=7\ (\text{vô nghiệm vì }7>\sqrt{2^2+6^2})}\\
			&\Leftrightarrow &x=\dfrac{\pi}{2}+k2\pi\ (k\in \mathbb{Z}).
		\end{eqnarray*}
	}
\end{bt}
\begin{bt}%[Thi GHK, Trần Quang Khải - Hồ Chí Minh, 2018]%[Phạm An Bình, EX-DT-10&11]%[1D1B1-1]
	Tìm tập xác định của hàm số $y=\cos 2x+\dfrac{1}{\tan x}+5$.
	\loigiai{
		Hàm số xác định khi $\heva{&\tan x\ne 0\\&\cos x\ne 0}\Leftrightarrow \heva{&x\ne k\pi\\&x\ne \dfrac{\pi}{2}+k\pi}\Leftrightarrow x\ne \dfrac{k\pi}{2}$ ($k\in \mathbb{Z}$).\\
		Vậy tập xác định của hàm số là $\mathscr{D}=\mathbb{R}\setminus \left\{ \dfrac{k\pi}{2} \mid k\in \mathbb{Z} \right\}$.
	}
\end{bt}
\begin{bt}%[Thi GHK, Trần Quang Khải - Hồ Chí Minh, 2018]%[Phạm An Bình, EX-DT-10&11]%[1D2B1-3]
	Từ các chữ số $0$; $1$; $2$; $3$; $4$; $5$; $6$ có thể lập được bao nhiêu số tự nhiên chẵn có $3$ chữ số khác nhau?
	\loigiai{
		Gọi $\overline{abc}$ là số tự nhiên chẵn có $3$ chữ số khác nhau.
		\begin{itemize}
			\item TH1. $c=0$, suy ra có $1$ cách chọn $c$.\\
			Chọn $a$ có $6$ cách.\\
			Chọn $b$ có $5$ cách.\\
			Suy ra có $1\cdot 6\cdot 5=30$ cách chọn.
			\item TH2. $c\in \{ 2;4;6 \}$, suy ra có $3$ cách chọn $c$.\\
			Chọn $a$ có $5$ cách.\\
			Chọn $b$ có $5$ cách.\\
			Suy ra có $3\cdot 5\cdot 5=75$ cách chọn.
		\end{itemize}
		Vậy có thể lập được $30+75=105$ số tự nhiên chẵn có $3$ chữ số khác nhau.
	}
\end{bt}
\begin{bt}%[Thi GHK, Trần Quang Khải - Hồ Chí Minh, 2018]%[Phạm An Bình, EX-DT-10&11]%[1D2B2-1]
	Từ một nhóm học sinh có $6$ nữ và $7$ nam, hỏi có bao nhiêu cách chọn một tổ \lq\lq trực trường\rq\rq\ gồm $5$ học sinh, trong đó $1$ tổ trưởng, $1$ tổ phó, $1$ thư kí và $2$ tổ viên? Biết rằng tổ trưởng phải là nữ, tổ phó phải là nam.
	\loigiai{
		Chọn tổ trưởng có $6$ cách.\\
		Chọn tổ phó có $7$ cách.\\
		Chọn thư kí có $11$ cách.\\
		Chọn $2$ tổ viên có $\mathrm{C}_{10}^2$.\\
		Vậy chọn một tổ \lq\lq trực trường\rq\rq\ có $6\cdot 7\cdot 11\cdot \mathrm{C}_{10}^2 = 20790$ cách.
	}
\end{bt}
\begin{bt}%[Thi GHK, Trần Quang Khải - Hồ Chí Minh, 2018]%[Phạm An Bình, EX-DT-10&11]%[1H1B7-2]
	Trong mặt phẳng tọa độ $Oxy$, cho điểm $A(3;0)$. Tìm tọa độ điểm $A'$ là ảnh của điểm $A$ qua phép vị tự tâm $B(1;-5)$ tỉ số $k=-2$.
	\loigiai{
		\begin{eqnarray*}
			A'(a;b)=\mathrm{V}_{\left(B,-2\right)}(A)&\Leftrightarrow &\overrightarrow{BA'}=-2\overrightarrow{BA}\\
			&\Leftrightarrow &\heva{&a-1=-2(3-1)\\&b+5=-2(0+5)}\\
			&\Leftrightarrow &\heva{&a=-3\\&b=-15.}
		\end{eqnarray*}
		Vậy $A'(-3;-15)$.
	}
\end{bt}
\begin{bt}%[Thi GHK, Trần Quang Khải - Hồ Chí Minh, 2018]%[Phạm An Bình, EX-DT-10&11]%[1H1B5-3]
	Trong mặt phẳng tọa độ $Oxy$, cho đường thẳng $(d)\colon 2x-y+5=0$. Viết phương trình đường thẳng $(d')$ là ảnh của đường thẳng $(d)$ qua phép quay tâm $O$ với góc quay $90^\circ$.
	\loigiai{
		Gọi $M(x;y)\in (d)$ và $M'(x';y')=\mathrm{Q}_{\left(O,90^\circ \right)}(M)$, suy ra $M'\in (d')$ và $\heva{&x'=-y\\&y'=x}\Leftrightarrow \heva{&y=-x'\\&x=y'}$.\\
		Thay $\heva{&y=-x'\\&x=y'}$ vào hàm số $(d)$ ta có $2y'-(-x')+5=0\Leftrightarrow x'+2y'+5=0$.\\
		Vậy $(d')\colon x+2y+5=0$.
	}
\end{bt}
\begin{bt}%[Thi GHK, Trần Quang Khải - Hồ Chí Minh, 2018]%[Phạm An Bình, EX-DT-10&11]%[1H1B2-2]
	Trong mặt phẳng tọa độ $Oxy$, cho đường tròn $(C)\colon x^2+y^2-8x+6y+9=0$. Viết phương trình đường tròn $(C')$ là ảnh của đường tròn $(C)$ qua phép tịnh tiến theo vectơ $\overrightarrow{u}(2;-5)$.
	\loigiai{
		Viết lại $(C)\colon (x-4)^2+(y+3)^2=16$, suy ra $(C)$ có tâm $I(4;-3)$ và bán kính $R=4$.\\
		Suy ra $(C')$ có tâm $I'(a;b)=\mathrm{T}_{\overrightarrow{u}}(I)$ và bán kính $R'=4$.
		\begin{eqnarray*}
			I'=\mathrm{T}_{\overrightarrow{u}}(I)&\Leftrightarrow &\heva{&a=4+2\\&b=-3-5}\\
			&\Leftrightarrow &\heva{&a=6\\&b=-8.}
		\end{eqnarray*}
		Vậy $(C')\colon (x-6)^2+(y+8)^2=16$.
	}
\end{bt}
