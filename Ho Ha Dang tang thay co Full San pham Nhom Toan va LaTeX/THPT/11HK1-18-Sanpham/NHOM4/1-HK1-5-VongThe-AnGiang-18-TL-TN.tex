Giải phương trình $\sin \left(x+ \dfrac{\pi}{6}\right)=\dfrac{1}{2}.$	
	\loigiai{
	\begin{align*}
&\sin \left(x+ \dfrac{\pi}{6}\right)=\dfrac{1}{2}\\
\Leftrightarrow & \sin \left(x+ \dfrac{\pi}{6}\right)= \sin \dfrac{\pi}{6}\\
\Leftrightarrow &\hoac{x+ \dfrac{\pi}{6}&=\dfrac{\pi}{6}+ k2 \pi\\	x+ \dfrac{\pi}{6}&= \pi - \dfrac{\pi}{6}+ k2 \pi
}, k \in \mathbb{Z}\\
\Leftrightarrow & \hoac{x&=k2 \pi\\x&=\dfrac{2\pi}{3}+k2 \pi }, k \in \mathbb{Z}.
	\end{align*}
Vậy phương trình có nghiệm 	$ \hoac{x&=k2 \pi\\x&=\dfrac{2\pi}{3}+k2 \pi }, k \in \mathbb{Z}.$}
\end{bt}
