\section{Định lí Ta-lét trong tam giác}
\subsection{Trọng tâm kiến thức}
\begin{dn}[Tỉ số của hai đoạn thẳng]
Tỉ số của hai đoạn thẳng là tỉ số độ dài của chúng theo cùng một đơn vị đo.
\end{dn}
\begin{dn}[Đoạn thẳng tỉ lệ]
 Hai đoạn thẳng $AB$ và $CD$ gọi là tỉ lệ với hai đoạn thẳng $A'C'$ và $C'D'$ nếu có hệ thức
$$\dfrac{AB}{CD}=\dfrac{A'B'}{C'D'} \ \text{hay} \ \dfrac{A'B'}{AB}=\dfrac{C'D'}{CD}.$$
\end{dn}
\begin{dl}[Định lí Ta-lét trong tam giác]
\immini{
Nếu một đường thẳng song song với một cạnh của tam giác và cắt hai cạnh còn lại thì nó định ra trên hai cạnh đó những đoạn thẳng tương ứng tỉ lệ.\\
Trong hình bên\\
$\heva{& \triangle ABC\\
& B'C' \parallel BC } \Rightarrow \dfrac{AB'}{AB}=\dfrac{AC'}{AC}$; $\dfrac{AB'}{B'B}=\dfrac{AC'}{C'C}$; $\dfrac{B'B}{AB}=\dfrac{C'C}{AC}$
}{

}
\end{dl}

\begin{vd}%[8H3K1]
Cho  $\triangle ABC$ có $AD$ là đường trung tuyến, $G$ là trọng tâm. Qua $G$ kẻ đường thẳng $d$ cắt $AB$, $AC$ thứ tự tại $M$, $N$. Chứng minh
\begin{enumEX}{2}
\item $\dfrac{AB}{AM}+\dfrac{AC}{AN}=3$.
\item $\dfrac{BM}{AM}+\dfrac{CN}{AN}=1$.
\end{enumEX}
\loigiai{
\begin{enumEX}{1}
\item Kẻ $BI \parallel CK \parallel MN$ ($I, K \in AD$).
\immini{
Xét $\triangle BDI$ và $\triangle CDK$ có
\begin{align*}
&BD&=& \ CD;\\
&\widehat{IBD} &=&\ \widehat{KCD};\\
&\widehat{IDB} &=&\ \widehat{KDC}
\end{align*}
nên $\triangle BDI =\triangle CDK$ (g.c.g) $\Rightarrow DI=DK$.\\
Áp dụng định lí Ta-lét, ta có
\begin{align*}
\dfrac{AB}{AM}&=&\ \dfrac{AI}{AG} & \ (\text{vì} \ MG \parallel BI);\\
\dfrac{AC}{AN}&=&\ \dfrac{AK}{AG} & \ (\text{vì} \ GN \parallel CK).
\end{align*}
Suy ra $\dfrac{AB}{AM}+\dfrac{AC}{AN}=\dfrac{2 AD}{AG}=3$ (vì $AD=\dfrac{3}{2}AG$).
}{

Suy ra $\dfrac{BM}{AM}+\dfrac{CN}{AN}=\dfrac{GI+GK}{AG}=\dfrac{2 GD}{AG}=1$ (vì $GD=\dfrac{1}{2}AG$).\\
Vậy $\dfrac{BM}{AM}+\dfrac{CN}{AN}=1$.

Vì $ME \parallel DF$, áp dụng định lí Ta-lét, ta có\\

\end{bt}

\begin{bt}%[8H3B2]
Cho hình thang $ABCD$ ($AB \parallel CD$) có $AB<CD$. Gọi $O$ là giao điểm hai đường chéo, $S$ là giao điểm của hai đường thẳng chứa hai cạnh bên. Đường thẳng $SO$ cắt $AB$, $CD$ thứ tự tại $M$, $N$. Chứng minh rằng
\begin{enumEX}{2}
\item $\dfrac{MA}{ND}=\dfrac{MB}{NC}$; $\dfrac{MA}{NC}=\dfrac{MB}{ND}$.
\item $MA=MB$; $NC=ND$.
\end{enumEX}
\loigiai{
\begin{enumEX}{1}
\item  \immini{\noindent Vì $MA \parallel DN$ nên ta có
$\dfrac{MA}{ND}=\dfrac{SM}{SN}$ \tagEX{1}
Vì $MB \parallel NC$ nên ta có
$\dfrac{MB}{NC}=\dfrac{SM}{SN}$.  \tagEX{2}
Từ (1) và (2) suy ra $\dfrac{MA}{ND}=\dfrac{MB}{NC}$. \\
Vì $MA \parallel NC$ nên ta có
$\dfrac{MA}{NC}=\dfrac{OM}{ON}$. \tagEX{3}
Vì $MB \parallel ND$ nên ta có:
$\dfrac{MB}{ND}=\dfrac{OM}{ON}$.  \tagEX{4}
Từ (3) và (4) suy ra $\dfrac{MA}{NC}=\dfrac{MB}{ND}$.
}{
\begin{tikzpicture}[scale=0.8, line join = round, line cap = round]
\tikzset{label style/.style={font=\footnotesize}}
\tkzDefPoints{0/0/D,6/0/C,2/5/S}
\tkzDefPointWith[linear,K=0.6](S,D) \tkzGetPoint{A}
\tkzDefPointWith[linear,K=0.6](S,C) \tkzGetPoint{B}
\tkzInterLL(A,C)(B,D)    \tkzGetPoint{O}
\tkzInterLL(S,O)(A,B)    \tkzGetPoint{M}
\tkzInterLL(S,O)(D,C)    \tkzGetPoint{N}
\tkzDrawSegments(S,D S,C D,C A,B B,D A,C)
\tkzDrawSegments[dashed](S,N)
\tkzDrawPoints(S,A,B,C,D,O,M,N)
\tkzLabelPoints[above](S)
\tkzLabelPoints[below](D,C,N)
\tkzLabelPoints[right](B)
\tkzLabelPoints[left](A)
\tkzLabelPoints[above right](M,O)
\end{tikzpicture}
}
\item Áp dụng tính chất dãy tỉ số bằng nhau, ta có\\
$\dfrac{MA}{ND}=\dfrac{MB}{NC}=\dfrac{MA+MB}{ND+NC}=\dfrac{AB}{CD}$; \tagEX{5}
$\dfrac{AM}{NC}=\dfrac{MB}{ND}=\dfrac{AM+MB}{NC+ND}=\dfrac{AB}{CD}$. \tagEX{6}
Từ (5) và (6) ta có $\dfrac{MA}{ND}=\dfrac{AM}{NC} \Rightarrow NC=ND$.\\
Mà $\dfrac{MA}{NC}=\dfrac{MB}{ND} \Rightarrow MA=MB$.
\end{enumEX}
}
\end{bt}

\begin{bt}%[8H3K2]
Cho tam giác $ABC$ cố định. Các điểm $D$, $E$ di động trên các cạnh tương ứng $AB$, $AC$ sao cho $\dfrac{AD}{DB}=\dfrac{CE}{EA}$. Chứng minh trung điểm $M$ của đoạn thẳng $DE$ nằm trên đoạn thẳng cố định.
\loigiai{
Vì $\dfrac{AD}{DB}=\dfrac{CE}{EA}$ nên $\dfrac{AD}{AB}=\dfrac{CE}{CA}$.\\
\immini{
Qua $E$ kẻ đường thẳng song song với $AB$ cắt $BC$ tại $F$, khi đó $\dfrac{CE}{CA}=\dfrac{EF}{AB}$ (hệ quả định lí Ta-lét)\\
$\Rightarrow \dfrac{EF}{AB}=\dfrac{AD}{AB} \Rightarrow EF =AD$.\\
Mặt khác $EF \parallel AD$ nên $ADFE$ là hình bình hành, do đó $M$ cũng là trung điểm của $AF$.\\
Gọi $I$, $J$ là trung điểm $AB$, $AC$ thì ta có $MI$; $MJ$ là đường trung bình của $\triangle ABF$ và $\triangle ACF$.\\
Nên $MI \parallel BC$, $MJ \parallel BC$ nên $M$ nằm trên đường trung bình của $\triangle ABC$ cố định. 
}{

\tkzLabelPoints[above](A)
\tkzLabelPoints[below](B,C,F)
\tkzLabelPoints[right](J,E)

\end{bt}

\begin{bt}%[8H3G2]
Cho tam giác $ABC$, hãy dựng hình vuông $MNPQ$ nội tiếp tam giác $ABC$ ($M$ trên $AB$, $N$ trên $AC$, $P$ và $Q$ trên cạnh $AC$).
\loigiai{
\immini{
\textit{Phân tích:}\\
Giả sử đã được hình vuông $MNPQ$ nội tiếp tam giác $ABC$. Nối $BN$, trên đoạn $BN$ lấy $N'$. Dựng hình chữ nhật $M'N'P'Q'$.\\
Áp dụng hệ quả định lí Ta-lét\\
$$\dfrac{M'N'}{MN}=\dfrac{BN'}{BN}=\dfrac{N'P'}{NP} \Rightarrow M'N'=N'P'.$$
Do đó $M'N'P'Q'$ là hình vuông.\\
\textit{Cách dựng:}\\
Trên cạnh $AB$ lấy điểm $M'$ tùy ý, dựng hình vuông $M'N'P'Q'$ ($P', Q' \in BC$).

\begin{bt}%[8H3K2]
Cho tam giác $ABC$, $M$ là điểm thuộc $BC$. Chứng minh rằng $$MA \cdot BC<MC \cdot AB+ MB \cdot AC.$$
\loigiai{
\immini{
Kẻ $MN \parallel AB \Rightarrow \dfrac{MN}{AB}=\dfrac{MC}{BC} \Rightarrow MN=AB \cdot \dfrac{MC}{BC}$.\\
Vì $MN \parallel AB$ nên $\dfrac{NA}{AC}=\dfrac{MB}{BC} \Rightarrow NA=AC \cdot \dfrac{MB}{BC}$.\\
Mà $AM<MN+NA$ (bất đẳng thức tam giác),\\
 hay $AM<AB \cdot \dfrac{MC}{BC}+AC \cdot \dfrac{MB}{BC}$.\\
Vậy $MA \cdot BC<MC \cdot AB+ MB \cdot AC$.
