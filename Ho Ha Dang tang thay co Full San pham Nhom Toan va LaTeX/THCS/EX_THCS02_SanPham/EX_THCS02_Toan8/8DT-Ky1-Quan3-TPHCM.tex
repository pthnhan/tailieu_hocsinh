\begin{ex}%[8D1B11]%[8D2B4]
	\textbf{(3,0 điểm)} Thực hiện các phép tính, rút gọn:
	\begin{multicols}{2}
	\begin{enumerate}
	\item$4x(3-x)+(2x-3)^2$
	\item$\dfrac{x}{x-3}+\dfrac{2x}{x-3}-\dfrac{9}{x-3}$
	\item$15x^2y^2:3xy^2+(4x^2-y^2):(2x+y)$
	\item$\dfrac{3}{b-1}-\dfrac{3+6b}{b^3-1}+\dfrac{1-2b}{b^2+b+1}$
	\end{enumerate}
	\end{multicols}
	\loigiai{
	\begin{enumerate}
	\item$4x(3-x)+(2x-3)^2=12x-4x^2+4x^2-12x+9=9$
	\item Điều kiện:$x \neq 0$\\
	$\dfrac{x}{x-3}+\dfrac{2x}{x-3}-\dfrac{9}{x-3}=\dfrac{x+2x-9}{x-3}
	=\dfrac{3x-9}{x-3}=\dfrac{3(x-3)}{x-3}=3$
	\item Điều kiện $x \neq 0, y\neq 0, x+2y \neq 0$\\
	$15x^2y^2:3xy^2+(4x^2-y^2):(2x+y)=5x+(2x-y)(2x+y):(2x+y)=5x+2x-y=7x-y$
	\item Điều kiện $b \neq 1$\\
	$\dfrac{3}{b-1}-\dfrac{3+6b}{b^3-1}+\dfrac{1-2b}{b^2+b+1}\\
		=\dfrac{3(b^2+b+1)-(3+6b)+(1-2b)(b-1)}{(b-1)(b^2+b+1)}\\
		=\dfrac{b^2-1}{(b-1)(b^2+b+1)}\\
		=\dfrac{(b-1)(b+1)}{(b-1)(b^2+b+1)}\\
		=\dfrac{b+1}{b^2+b+1}$
	\end{enumerate}
	}
\end{ex}
\begin{ex}%[8D1B9]%[8D1K9]
	\textbf{(2,0 điểm)} Phân tích các đa thức sau thành nhân tử:
	\begin{multicols}{2}
	\begin{enumerate}
	\item$4x^3+4x^2+x$
	\item$ab-7b+3a-21$
	\item$5x(x-3)+6-2x$
	\item$2x^3+3x^2-1$
	\end{enumerate}
	\end{multicols}
	\loigiai{
	\begin{enumerate}
	\item$4x^3+4x^2+x=x(4x^2+4x+1)=x((2x)^2+2.2x.1+1)=x(2x+1)^2$
	\item$ab-7b+3a-21=b(a-7)+3(a-7)=(a-7)(b+3)$
	\item$5x(x-3)+6-2x=5x(x-3)-2(x-3)=(5x-2)(x-3)$
	\item$2x^3+3x^2-1=x^2(2x-1)+2x(2x-1)+2x+1=(x^2+2x+1)(2x-1)=(x+1)^2(2x-1)$
	\end{enumerate}
	}

\begin{ex}[6D1T6]
	\textbf{(1,0 điểm)} 
	Theo quyết định của Bộ Công Thương ban hành, giá bán lẻ điện sinh hoạt từ ngày 01/12/2017 sẽ dao động
	trong khoảng từ $1549$ đến $2701$ đồng mỗi $kWh$ tùy bậc thang.
	Dưới đây là bảng so sánh giá điện trước và sau khi điều chỉnh:\\
	 \begin{tabular}{|llr|c|c|}
	 \hline
	 \textbf{MỨC SỬ DỤNG TRONG THÁNG (kWh)}&\quad&\quad&\textbf{GIÁ CŨ}&\textbf{GIÁ MỚI}\\
	 \hline
	 Bậc 1: Cho kWh từ & $0-50$ &($50$kWh)&$1484$&$1549$\\
	 \hline
	 Bậc 2: Cho kWh từ & $51-100$ &($50$kWh)&$1533$&$1600$\\
	 \hline
	 Bậc 3: Cho kWh từ & $101-200$ &($50$kWh)&$1786$&$1858$\\
	 \hline
	 Bậc 4: Cho kWh từ & $201-300$& ($50$kWh)&$2242$&$2340$\\
	 \hline
	 Bậc 5: Cho kWh từ & $301-400$ &($50$kWh)&$2503$&$2615$\\
	 \hline
	 Bậc 6: Cho kWh từ & $401$ trở lên &\quad                &$2587$&$2701$\\
	 \hline
	 \end{tabular}\\
	 Khi không tính thuế VAT:
	 \begin{enumerate}
	 \item Trong tháng $11/2017$, hộ $A$ tiêu thụ $540$ kWh. Hỏi hộ $A$ phải trả bao nhiêu tiền?
	 \item Nếu tháng $12/2017$, hộ $A$ vẫn tiêu thụ $540$ kWh thì theo giá mới số tiền phải trả 
	 tăng lên bao nhiêu?
	 \end{enumerate}
	\loigiai{
	\begin{enumerate}
	\item Trong tháng $11$ số tiền điện sẽ tính theo giá cũ. Khi đó để trả cho $540$ kWh,
	hộ $A$ phải mất: $540\cdot 2587=1396980$ (đồng) 
	\item Trong tháng $12$ số tiền điện sẽ tính theo giá mới. Nếu vẫn tiêu thụ $540$ kWh thì
	số tiền phải trả tăng lên là: $540 \cdot (2701-2587)=61560$ (đồng)
	\end{enumerate}
	}

\begin{ex}%[8H2K2]
	\textbf{(1,0 điểm)}
	Một nền nhà hình chữ nhật có chiều dài $20m$ và chiều rộng bằng $\dfrac{1}{4}$ chiều dài. Người ta muốn lát
	gạch hình vuông cạnh $5$ dm lên nền nhà đó nên đã mua gạch với tổng số tiền là $36800000$ đồng (giả sử
	khoảng cách giữa hai viên gạch kề nhau là không đáng kể).\\
	Hỏi giá của một viên gạch là bao nhiêu?
	\loigiai
	{
	Chiều rộng của nền nhà $20 \cdot \dfrac{1}{4}=5$ (m)\\
	Diện tích nền nhà là: $20 \cdot 5=100$ ($m^2$)\\
	Diện tích một viên gạch: $5 \cdot 5=25 (dm)^2=0.25 (m^2)$\\
	Suy ra số viên gạch cần sử dụng là: $100:0.25=400$ (viên)\\
	Do đó giá một viên gạch là $36800000:400=92000$(đồng)
	}
\end{ex}
\begin{ex}%[8H1B9]%[8H1K7]%[8H2G3]
	\textbf{(3,0 điểm)}
	Cho $\triangle ABC$ vuông tại $A$ với $AB<AC$ và có đường cao $AH$. Gọi $M$ là trung điểm của đoạn thẳng
	$AB$, E là điểm đối xứng với điểm $H$ qua điểm $M$.
	\begin{enumerate}
	\item Chứng minh tứ giác AHBE là hình chữ nhật
	\item Lấy điểm $D$ trên $HC$ sao cho $H$ là trung điểm của đoạn $DB.$ Qua điểm $D$ vẽ đường thẳng vuông góc
	với $AC$ và cắt các đường thẳng $AC, MH$ lần lượt tại $K,F.$ Gọi $N$ là trung điểm của đoạn $AK.$ Chứng minh
	tứ giác $AMFD$ là hình bình hành và $HN\perp AK$.
	\item Vẽ $KV$ vuông góc với $HK$ ($V$ thuộc $CD$.) Chứng minh rằng $S_{KDC}=2S_{KDV}$
	\end{enumerate}
	\loigiai
	{
	\begin{center}
	\begin{tikzpicture}[thick]
		\draw (0,0)--(7.72,0)--(0,4)--(-1.63,0.84)--(0,0) 
		 (0,0)--(0,4) (-1.63,0.84)--(3.26,4.3)
		 (0,0)--(1.63,3.15)--(1.63,0)--(3.26,0)--(5.49,1.15) (3.26,0)--(3.26,4.3)
		 (0,0)--(3.26,2.31) (1.63,3.15)--(3.26,0);
		\draw (-0.1,-0.3) node {$A$} (0,4.3) node {$B$} (8,0) node {$C$}
		(-1.93,0.85) node {$E$}(1.63,-0.3) node {$N$} (3.1,-0.3) node {$K$} 
		 (0.2,1.8) node {$M$}(1.63,3.47) node {$H$}
		 (3.26,4.5) node {$F$}(5.69,1.35) node {$V$} (3.46,2.51) node {$D$};
	\end{tikzpicture}
	\end{center}
	\begin{enumerate}
		\item Tứ giác $AHBE$ có hai đường chéo cắt nhau tại trung điểm của mỗi đường nên
		 là hình bình hành. Mặt khác hình bình hành này có $\widehat{AHB}=90^\circ$ nên
		là hình chữ nhật.
		\item 
	\begin{align}
		MA \perp AC, FD \perp AC. \mbox{ Suy ra} AM \parallel FD\mbox{\hspace{8cm}} \\
		\mbox{Mặt khác} MH \mbox{ là đường trung bình của tam giác BAD nên} MH \parallel AD, 
		\mbox{ hay} MF \parallel AD.
	\end{align}
		Từ $(1)\text{và}(2)$ suy ra tứ giác $AMFD$ là hình bình hành.\\
		Mặt khác $HN$ là đường trung bình của hình thang $ABDK$ suy ra $HN \parallel AB$, 
		do đó $HN \perp AK$ \textit{(đpcm)}
		\item Ta chứng minh $V$ là trung điểm của $CD$. Thật vậy\\
		Xét $\triangle VDK$, ta có:
	\begin{align}
		\widehat{VDK}=\widehat{CBA} \mbox{(đồng vị)}\\
		\mbox{Mặt khác} \widehat{VKD}=\widehat{HKA} \mbox{(cùng phụ với} \widehat{DKV})\nonumber \\
		\mbox{Mà} \widehat{HKA}=\widehat{HAK}=90^\circ-\widehat{HAB}=\widehat{CBA} 
	\end{align}
	\end{enumerate}
		Từ $(3)\text{ và }(4)$ suy ra $\triangle VDK$ cân tại $V$, từ đây ta cũng có $\triangle VKC$ cân tại 
$V$ \break(do $\widehat{VKC}=\widehat{VCK}\text{ vì lần lượt phụ với hai góc bằng nhau}\widehat{VKD},\widehat{KDC})
$, do đó $VD=DC=\dfrac{1}{2}CD$ (hay $V$ là trung điểm $CD$)\\
Ta có  $\dfrac{S_{KDC}}{S_{KDV}}=\dfrac{DC}{DV}=2 \Rightarrow S_{KDC}=2S_{KDV}$ \textit{(đpcm)}
}
