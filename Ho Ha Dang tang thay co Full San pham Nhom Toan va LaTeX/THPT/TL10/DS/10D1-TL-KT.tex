			\item Xét tính đúng sai và lập mệnh đề phủ định của mệnh đề $P:'' \exists x\in \mathbb{R}: x^2=-3''$.
			\item Cho tam giác $ABC$, xét hai mệnh đề $P: AB^2+AC^2=BC^2$, $Q: \widehat{A}=90^\circ$. Hãy lập mệnh đề $P\Rightarrow Q$ và xét tính đúng sai của mệnh đề đó.
		\end{enumerate}
		\loigiai{\begin{enumerate}
				\item Mệnh đề $P$ là sai. Phủ định của $P$ là $\overline{P}: \forall x\in \mathbb{R}: x^2\neq -3$.
				\item Cho tam giác $ABC$, mệnh đề $P\Rightarrow Q$ phát biểu như sau: Trong tam giác $ABC$, nếu $AB^2+AC^2=BC^2$ thì góc $\widehat{A}=90^\circ$.\\
				Mệnh đề $P\Rightarrow Q$ là đúng.
			\end{enumerate}
		}
	\end{bt}
