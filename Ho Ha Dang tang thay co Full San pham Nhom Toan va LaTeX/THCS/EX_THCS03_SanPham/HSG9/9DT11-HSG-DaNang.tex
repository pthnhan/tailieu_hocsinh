\begin{ex}%[Đề thi Toán 9 Học sinh giỏi năm học 2011, Tp. Đà Nẵng]%[9D1K8]
	Cho biểu thức: $M=\dfrac{a+1}{\sqrt{a}}+\dfrac{a\sqrt{a}-1}{a-\sqrt{a}}+\dfrac{a^2-a\sqrt{a}+\sqrt{a}-1}{\sqrt{a}-a\sqrt{a}}$ với $a > 0, a \neq 1.$
	\begin{enumerate}
		\item Chứng minh rằng $M>4.$ 
		\item  Với những giá trị nào của $ a $ thì biểu thức $N=\dfrac{6}{M}$ nhận giá trị nguyên?
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Do $a > 0, a \neq 1$ nên: $\dfrac{a\sqrt{a}-1}{a-\sqrt{a}}=\dfrac{\sqrt{a}-1)(a+\sqrt{a}+1}{\sqrt{a}(\sqrt{a}-1)}=\dfrac{a+\sqrt{a}+1}{\sqrt{a}}$\\
			 và
			$\dfrac{a^2-a\sqrt{a}+\sqrt{a}-1}{\sqrt{a}-a\sqrt{a}}=\dfrac{(a+1)(a-1)-\sqrt{a}(a-1)}{{\sqrt{a}(1-a)}}=\dfrac{(a-1)(a-\sqrt{a}+1)}{\sqrt{a}(1-a)}=\dfrac{-a+\sqrt{a}-1}{\sqrt{a}}\\
			\Rightarrow M=\dfrac{a+1}{\sqrt{a}}+2$
			\item Ta có $0<N=\dfrac{6}{M}<\dfrac{3}{2}$ do đó $N$ chỉ có thể nhận được một giá trị nguyên là 1\\
			Mà	
				$N = 1 \Leftrightarrow \dfrac{6\sqrt{a}}{a+1+2\sqrt{a}}=1 \Leftrightarrow a-4\sqrt{a}+1=0 \Leftrightarrow (\sqrt{a}-2)^2=3$ 
			
			$\Leftrightarrow \sqrt{a}=2+\sqrt{3}$ hay $\sqrt{a}=2-\sqrt{3}$ (phù hợp)\\
			Vậy, $ N $ nguyên $\Leftrightarrow a=(2\pm \sqrt{3})^2$
		
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học sinh gỏi năm học 2011, Tp. Đà Nẵng]%[9D2K4]%[9D2G4]
	
\begin{enumerate}
	\item Cho các hàm số bậc nhất: $y=0,5x+3,\ y=6-x$ và $y=mx$ có đồ thị lần lượt là các đường thẳng $(d_1),\ (d_2)$ và $(\Delta _m).$ Với những giá trị nào của tham số m thì đường thẳng $(\Delta _m)$ cắt hai đường thẳng $(d_1)$ và $(d_2)$ lần lượt tại hai điểm $ A $ và $ B $ sao cho điểm $ A $ có hoành độ âm còn điểm $ B $ có hoành độ dương?
	\item Trên mặt phẳng tọa độ $ Oxy, $ cho $ M $ và $ N $ là hai điểm phân biệt, di động lần lượt trên trục hoành và trên trục tung sao cho đường thẳng $ MN $ luôn đi qua điểm cố định $I(1;2)$. Tìm hệ thức liên hệ giữa hoành độ của $ M $ và tung độ của $ N; $ từ đó, suy ra giá trị nhỏ nhất của biểu thức $Q=\dfrac{1}{OM^2}+\dfrac{1}{ON^2}.$
\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Điều kiện để $(\Delta _m)$ là đồ thị hàm số bậc nhất là $m\ne 0$\\
		Phương trình hoành độ giao điểm của $(d_1)$ và $(\Delta _m)$ là:\\
		$0,5x+3=mx \Leftrightarrow (m-0,5)x=3$ \\
		Điều kiên để phương trình này có nghiệm âm là $m-0,5<0$ hay $m<0,5$\\
		Phương trình hoành độ giao điểm của $(d_2)$ và $(\Delta _m)$ là:\\
		$6-x=mx \Leftrightarrow (m+1)x=6$\\
		Điều kiên để phương trình này có nghiệm dương là $m+1>0$ hay $m>-1$\\
		Vậy điều kiện cần tìm là: $-1<m<0,5;m\ne 0$
		\item Đặt $  m = x_M $ và $n = y_N \Rightarrow m.n \neq 0 $ và $ m \neq 1 \hfill (*)$ \\
		Nên đường thẳng qua ba điểm $M, I, N$ có dạng: $y = ax+b$ \\
		$ \Rightarrow  $ $\begin{cases}
			& 0=am+b \\ 
			& 2=a+b \\ 
			& n=b \end{cases}$ 
			$ \Rightarrow  $ hệ thức liên hệ giữa $ m $ và $ n $ là $2m+n=mn$\\
		Chia hai vế cho $m.n \neq 0$ ta được: $\dfrac{1}{m}+\dfrac{2}{n}=1 \hfill(**)$
		$\Rightarrow 1={\left(\dfrac{1}{m}+\dfrac{2}{n}\right)}^2=\dfrac{1}{m^2}+\dfrac{4}{n^2}+\dfrac{4}{mn}=5\left(\dfrac{1}{m^2}+\dfrac{1}{n^2}\right)-{\left(\dfrac{2}{m}-\dfrac{1}{n}\right)}^2$\\
		$\Rightarrow Q=\dfrac{1}{m^2}+\dfrac{1}{n^2}\ge \dfrac{1}{5};$\\
		dấu “=” xảy ra khi $\dfrac{2}{m}=\dfrac{1}{n};$ kết hợp $ (**) : m = 5, n = 2,5 ($ thỏa $ (*))$\\
		Vậy giá trị nhỏ nhất của Q là $\dfrac{1}{5}$
	
	\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi Toán 9 Học sinh gỏi năm học 2011, Tp. Đà Nẵng]%[9D3K4]%[9D1G3]
	\begin{enumerate}
		\item Giải hệ phương trình: 
		$ \begin{cases}
			17x +2y  &= 2011|xy|\\
			x   -2y  &= 3xy
		\end{cases} $
		\item Tìm tất cả các giá trị của $ x,\ y,\ z $ sao cho:
		\begin{align*}
			\sqrt{x}+\sqrt{y-z}+\sqrt{z-x}=\dfrac{1}{2}(y+3).
		\end{align*}
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Nếu $ xy>0$ thì khi đó hệ đã cho tương đương với\\ $\begin{cases}
			& \dfrac{17}{y}+\dfrac{2}{x}=2011 \\ 
			& \dfrac{1}{y}-\dfrac{2}{x}=3 \end{cases}\Leftrightarrow \begin{cases}
			& \dfrac{1}{y}=\dfrac{1007}{9} \\ 
			& \dfrac{1}{x}=\dfrac{490}{9} \end{cases}\Leftrightarrow \begin{cases}
			& x=\dfrac{9}{490} \\ 
			& y=\dfrac{9}{1007} \end{cases}$ (phù hợp)\\
			Nếu $ xy<0$ thì khi đó hệ đã cho tương đương với \\$\begin{cases}
				& \dfrac{17}{y}+\dfrac{2}{x}=-2011 \\ 
				& \dfrac{1}{y}-\dfrac{2}{x}=3 \end{cases}\Leftrightarrow \begin{cases}
				& \dfrac{1}{y}=\dfrac{-1004}{9} \\ 
				& \dfrac{1}{x}=-\dfrac{1031}{18} \end{cases} \Rightarrow xy>0$ (loại)\\
				Nếu $ xy=0$ thì khi đó hệ đã cho tương đương với $x= y=0$ (nhận).\\
				Vậy hệ có đúng 2 nghiệm là $(0;0)$ và $\left(\dfrac{9}{490};\dfrac{9}{1007}\right)$
			\item 
			Điều kiện $x \geq 0; y - z  \geq 0; z - x \geq 0 \Leftrightarrow y \geq z \geq x \geq 0$ \\
			Theo BĐT Cauchy: $\sqrt{x}\le \dfrac{x+1}{2};\sqrt{y-z}\le \dfrac{y-z+1}{2};\sqrt{z-x}\le \dfrac{z-x+1}{2}$\\
			$\Rightarrow VP=\sqrt{x}+\sqrt{y-z}+\sqrt{z-x}\le \dfrac{1}{2}(y+3)=VT$\\
			Do đó $\begin{cases}
					& \sqrt{x}=1 \\ 
					& \sqrt{y-z}=1 \\ 
					& \sqrt{z-x}=1 
				\end{cases} 
				\Leftrightarrow \begin{cases}
									& x=1 \\ 
									& y=3 \\ 
									& z=2 
								\end{cases}$ thỏa điều kiện
			
		\end{enumerate}
	}

\end{ex}

\begin{ex}%[Đề thi Toán 9 Học sinh gỏi năm học 2011, Tp. Đà Nẵng]%[9H2T1]%[9H2K1]%[9H2G1]
Cho đường tròn $ (\mathcal{C}) $ với tâm $ O $ và đường kính $ AB $ cố định. Gọi $ M $ là điểm di động trên $(\mathcal{C})$ sao cho $ M $ không trùng với các điểm $ A $ và $ B. $ Lấy $ C $ là điểm đối xứng của $ O $ qua $ A. $ Đường thẳng vuông góc với $ AB $ tại $ C $ cắt đường thẳng $ AM $ tại $ N. $ Đường thẳng $ BN $ cắt đường tròn $ (\mathcal{C}) $ tại điểm thứ hai là $ E. $ Các đường thẳng $ BM $ và $ CN $ cắt nhau tại $ F. $
	\begin{enumerate}
		\item Chứng minh rằng các điểm $ A,\ E,\ F $ thẳng hàng.
		\item Chứng minh rằng tích $ AM.AN $ không đổi.
		\item Chứng minh rằng $ A $ là trọng tâm của tam giác $ BNF $ khi và chỉ khi $ NF $ ngắn nhất
	\end{enumerate}

	\loigiai{
	\begin{center}
		\begin{tikzpicture}[scale=.8]
			\tkzInit[xmin=-4,ymin=-2,xmax=4,ymax=5.5]
			\tkzClip
			\tkzDefPoints{-1/1/A, 3/1/B}
			\tkzDefMidPoint(A,B)						\tkzGetPoint{O}
			\tkzDrawCircle(O,A)
			\tkzCalcLength[cm](O,A)						\tkzGetLength{rOA}
			\tkzDefShiftPoint[O](110:\rOA){M}
			\tkzDefPointBy[symmetry=center A](O) 		\tkzGetPoint{C}
			\tkzDefLine[perpendicular=through C](A,B)	\tkzGetPoint{d}
			\tkzInterLL(C,d)(A,M)						\tkzGetPoint{N}
			\tkzInterLC(B,N)(O,A)						\tkzGetPoints{}{E}
			\tkzInterLL(B,M)(C,N)						\tkzGetPoint{F}			
			\tkzDrawSegments(B,C F,N F,B B,N F,E M,N) 
			\tkzMarkSegments[mark=s|](A,C A,O)
			\tkzMarkRightAngle(A,C,F)
			\tkzDrawPoints[color=black, fill=white](A,B,O,M,C,N,E,F,N)
			\node at ($(A)+(150:5mm)$){\footnotesize $A$};
			\node at ($(B)+(-10:3mm)$){\footnotesize $B$};
			\node at ($(C)+(230:3mm)$){\footnotesize $C$};
			\node at ($(O)+(-90:3mm)$){\footnotesize $O$};
			\node at ($(E)+(-90:5mm)$){\footnotesize $E$};
			\node at ($(F)+(180:3mm)$){\footnotesize $F$};
			\node at ($(M)+(70:3mm)$){\footnotesize $M$};
			\node at ($(N)+(180:3mm)$){\footnotesize $N$};
		\end{tikzpicture}
	\end{center}	
\begin{enumerate}
	\item $MN\perp BF$ và $BC\perp NF.$\\
	$\Rightarrow A$ là trực tâm của tam giác $BNF.$ \\
	$\Rightarrow FA\perp NB.$\\
	Lại có $AE\perp NB.$\\
	Nên $A, E, F$ thẳng hàng.	
	\item $\widehat{CAN}=\widehat{MAB}$, nên hai tam giác $ACN$ và $AMB$ đồng dạng.\\
	Suy ra: $\dfrac{AN}{AB}=\dfrac{AC}{AM}$\\
	Hay $AM.AN=AB.AC=2R^2$ không đổi (với $R$ là bán kính đường tròn $(\mathcal{C})$).
	\item Ta có $BA=\dfrac{2}{3}BC$ nên $A$ là trong tâm tam giác $BNF \Leftrightarrow C$ là trung điểm $NF. \hfill(1)$\\
	Mặt khác: $\widehat{CAN}=\widehat{CFM}$, nên hai tam giác $CNA$ và $CBF$ đồng dạng.\\
	$\Rightarrow \dfrac{CN}{BC}=\dfrac{AC}{CF}\Rightarrow CN.CF=BC.AC=3R^2.$\\
	Áp dụng bất đẳng thức Cô-si, ta có: $NF=CN+CF\ge 2\sqrt{CN.CF}=2R\sqrt{3}$ không đổi.\\
	Nên: $NF$ ngắn nhất $\Leftrightarrow CN =CF \Leftrightarrow C$ là trung điểm $NF. \hfill(2)$\\
	Từ $(1)$ và $(2)$ cho ta: $A$ là trong tâm tam giác $BNF \Leftrightarrow NF$ ngắn nhất.
	
\end{enumerate}
}

\end{ex}
