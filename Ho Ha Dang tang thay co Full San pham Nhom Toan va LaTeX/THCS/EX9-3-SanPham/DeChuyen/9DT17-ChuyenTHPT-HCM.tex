		\item Cho các số thực $a$, $b$, $c$ sao cho $a+b+c=3$, $a^2+b^2+c^2=29$ và $abc=11$. Tính $a^5+b^5+c^5$.
		\item Cho biểu thức $A=(m+n)^2+3m+n$ với $m$, $n$ là các số nguyên dương. Chứng minh rằng nếu $A$ là số chính phương thì $n^3+1$ chia hết cho $m$.
	\end{enumerate}
	\loigiai{\begin{enumerate}
			\item Ta có $a+b+c=3$, $ab+bc+ca=-10$, $abc=11$. Do đó, $a$, $b$, $c$ là ba nghệm của phương trình $x^3-3x^2-10x-11=0$. Tuy nhiên, phương trình đã cho có duy nhất một nghiệm thực $x\approx 5,2$ nên không tồn tại $a$, $b$, $c$.
			\item Ta có $(m+n)^2<(m+n)^2+3m+n<(m+n+2)^2$. Từ đó, nếu $A$ là số chính phương thì $A=(m+n+1)^2$, suy ra $m=n+1$. Do đó, $m\mid n^3+1$
	\end{enumerate}}

		\item Giải phương trình $2(x+2)\sqrt{3x-1}=3x^2-7x-3$.
		\item Giải hệ phương trình $\heva{&x+\dfrac{1}{y}-\dfrac{10}{x}=-1\\&20y^2-xy-y=1.}$
	\end{enumerate}
	\loigiai{\begin{enumerate}
			\item Phương trình đã cho tương đương với $\left(\sqrt{3x-1}+x+2\right)^2=4x^2$. Giải phương trình ta được $x=\dfrac{7+\sqrt{29}}{2}$.
			\item Ta có $20y^2-xy-y=1\Leftrightarrow 20y-x-1=\dfrac{1}{y}$. Thay vào phương trình $x+\dfrac{1}{y}-\dfrac{10}{x}=-1$ ta nhận được phương trình $xy=\dfrac{1}{2}$. Vậy hệ trở thành $\heva{&xy=\dfrac{1}{2}\\&20y^2-2y-3=0}$. Giải hệ ta được nghiệm $\left(-2;-\dfrac{1}{4}\right)$, $\left(\dfrac{5}{3};\dfrac{3}{10}\right)$. 
	\end{enumerate}}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên THPT, TPHCM, 2017]%[9D5G4]
	Trên một đường tròn có 9 điểm phân biệt, các điểm này được nối với nhau bởi các đoạn thẳng màu xanh hoặc màu đỏ. Biết rằng mỗi tam giác tạo bởi 3 trong 9 điểm chứa ít nhất một cạnh màu đỏ. Chứng minh rằng tồn tại 4 điểm sao cho 6 đoạn thẳng nối chúng đều có màu đỏ.
	\loigiai{ Trường hợp 1: Có một điểm nào đó, kí hiệu $x_1$ được nối với ít nhất 4 đỉnh $x_2$, $x_3$, $x_4$, $x_5$ các cạnh màu xanh. Lúc đó, 4 đỉnh $x_2, x_3, x_4, x_5$ phải được nối với nhau màu đỏ (vì giả thiết mỗi tam giác có ít nhất 1 cạnh màu đỏ).
		
		Trường hợp 2: Mỗi điểm có không quá 3 điểm khác nối với nó bởi cạnh màu xanh. Lúc này, có ít nhất một điểm, kí hiệu $x_1$ chung cạnh màu xanh với nhiều nhất 2 điểm (vì nếu tất cả các điểm có chung 3 cạnh xanh với 3 điêm khác thì số cạnh là $9\cdot 3:2$ không nguyên). Do đó, điểm $x_1$ được nối với ít nhất 6 điểm còn lại bằng màu đỏ. Rõ ràng, trong 6 điểm này, có ít nhất 3 điểm được nối với nhau cạnh màu đỏ, kết hợp với điểm $x_1$ ta suy ra điều phải chứng minh.
	}
\end{ex}
