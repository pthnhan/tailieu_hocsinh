\chapter{Hình lăng trụ đứng. Hình chóp đều}
\section{Hình hộp chữ nhật}
\subsection{Tóm tắt lí thuyết}
\begin{dn}
	Hình hộp chữ nhật có $6$ mặt, các mặt là những hình chữ nhật.
\end{dn}
\begin{dn}
	Hình lập phương là hình hộp chữ nhật có các mặt là những hình vuông.
\end{dn}
\begin{dn}
	Cho hình hộp chữ nhật có chiều dài $a$, chiều rộng $b$, chiều cao $c$.
	\begin{itemize}
		\item Diện tích xung quanh của hình hộp chữ nhật tính bởi công thức $$S_{\mathrm{xq}}=2(a+b)c.$$
		\item Diện tích toàn phần của hình hộp chữ nhật tính bởi công thức $$S_{\mathrm{tp}}=2(a+b)c+2ab=2(ab+bc+ca).$$
		\item Thể tích của hình hộp chữ nhật tính bởi công thức $$V=abc.$$
	\end{itemize}
\end{dn}

\begin{tc}
	Mô hình của hình hộp chữ nhật cho ta hình ảnh các quan hệ không gian sau
	\begin{center}
	\begin{tikzpicture}[scale=1, line join=round, line cap=round,>=stealth, font=\footnotesize]
	%\tkzInit[xmin=-.2, xmax=4.2, ymin=-.2, ymax=3]\tkzClip
	\tkzDefPoints{0/0/A, 3/0/B, 4/1/C, 0/2/A'}
	\tkzDefPointBy[translation= from B to C](A) \tkzGetPoint{D}
	\tkzDefPointsBy[translation=from A to A'](B,C,D){B'}{C'}{D'}
	
	\tkzDrawPolygon(A,A',D',C',C,B)
	\tkzDrawSegments(A',B' B,B' B',C')
	\tkzDrawSegments[dashed](A,D C,D D,D' A,C)
	\tkzDrawPoints[fill=black](A,B,C,D,A',B',C',D')
	\tkzLabelPoints[above](C',D')
	\tkzLabelPoints[left](A')
	\tkzLabelPoints[below](A,B,D)
	\tkzLabelPoints[right](B',C)
	\end{tikzpicture}
	\end{center}
	\begin{itemize}
		\item Hai đường thẳng song song là hai đường thẳng cùng nằm trong một mặt phẳng và không có điểm chung, chẳng hạn $AB\parallel A'B'$.
		\item Hai đường thẳng phân biệt cùng song song với một đường thẳng thứ ba thì song song với nhau, chẳng hạn $AB\parallel D'C'$ vì chúng cùng song song với $A'B'$.
		\item Đường thẳng $A'B'$ không nằm trong mặt phẳng $(ABCD)$ và song song với đường thẳng $AB$ của mặt phẳng $(ABCD)$, ta có $A'B'\parallel \text{mp} (ABCD)$.
		\item Mặt phẳng $(A'B'C'D')$ chứa hai đường thẳng cắt nhau $A'B'$, $B'C'$ cùng song song với mặt phẳng $(ABCD)$, ta có mp$(A'B'C'D')\parallel (ABCD)$.
		\item Hai mặt phẳng phân biệt có một điểm chung $A$ thì chúng có chung một đường thẳng đi qua $A$, gọi là giao tuyến của hai mặt phẳng. Chẳng hạn $AB$ là giao tuyến của hai mặt phẳng $(ABCD)$ và $(ABB'A')$.
		\item Đường thẳng $A'A$ vuông góc với hai đường thẳng cắt nhau $AB$, $AD$ của mặt phẳng $(ABCD)$, ta có $A'A\perp \text{mp}(ABCD)$.
		\item Khi $A'A\perp \text{mp}(ABCD)$ thì $A'A$ vuông góc với mọi đường thẳng đi qua $A$ và nằm trong mặt phẳng $(ABCD)$, chẳng hạn $A'A\perp AC$.
		\item Mặt phẳng $(A'B'BA)$ chứa đường thẳng $A'A$ vuông góc với mặt phẳng $(ABCD)$, ta có mp$(A'B'BA)\perp \text{mp}(ABCD)$.
	\end{itemize}
\end{tc}

\subsection{Các dạng toán}
\begin{dang}{Hình hộp chữ nhật}
	Sử dụng các tính chất của hình hộp chữ nhật.
\end{dang}
\begin{vd}%[Phạm An Bình, dự án EX-C2-8-2018][8H4K3]
	Cho hình lập phương $ABCD.A'B'C'D'$. Điểm $E$ chia $DB$ theo tỉ số $1:3$, điểm $F$ chia $B'A$ theo tỉ số $1:3$.
	\begin{enumerate}
		\item Chứng minh rằng $A'B'CD$ là hình chữ nhật. Tính diện tích hình chữ nhật đó nếu cạnh hình lập phương bằng $a$.
		\item Gọi $M$ là điểm chia $DA$ theo tỉ số $1:3$. Chứng minh rằng mặt phẳng $(EMF)$ song song với mặt phẳng $(A'B'CD)$.
		\item Chứng minh rằng $EF$ song song với mặt phẳng $(A'B'CD)$.
		\item Chứng minh $EF$ song song với mặt phẳng $(A'B'CD)$ mà không sử dụng kết quả của câu b.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}
			\item Chứng minh rằng $A'B'CD$ là hình chữ nhật. Tính diện tích hình chữ nhật đó nếu cạnh hình lập phương bằng $a$.
			\immini{
				Xét tứ giác $A'B'CD$ ta có\\
				$A'B'\parallel CD$ (vì cùng song song với $AB$),\\
				$A'B'=CD$ (vì cùng bằng $AB$).\\
				Suy ra tứ giác $A'B'CD$ là hình bình hành.\\
				Ta có $DC\perp CC'$ và $DC\perp CB$ nên $DC\perp \text{mp}(BCC'B')$, suy ra $DC\perp CB'$.\\
				Hình bình hành $A'B'CD$ có $\widehat{DCB'}=90^\circ$ nên $A'B'CD$ là hình chữ nhật.\\
				Xét $\triangle B'C'C$ vuông tại $C'$ ta có\\
				$B'C^2=B'C'^2+C'C^2$ (định lí Py-ta-go).\\
				$\Rightarrow B'C=a\sqrt{2}$.
			}{
				\begin{tikzpicture}[scale=1, line join=round, line cap=round,>=stealth, font=\footnotesize]
				%\tkzInit[xmin=-.2, xmax=4.2, ymin=-.2, ymax=3]\tkzClip
				\tkzDefPoints{0/0/A', 3/0/B', 4/1.5/C'}
				\tkzDefPointBy[rotation= center A' angle 90](B') \tkzGetPoint{A}
				\tkzDefPointBy[translation= from B' to C'](A') \tkzGetPoint{D'}
				\tkzDefPointBy[translation=from A' to A](B') \tkzGetPoint{B}
				\tkzDefPointBy[translation=from A' to A](C') \tkzGetPoint{C}
				\tkzDefPointBy[translation=from A' to A](D') \tkzGetPoint{D}
				\tkzDefPointBy[homothety = center D ratio 1/4](B) \tkzGetPoint{E}
				\tkzDefPointBy[homothety = center B' ratio 1/4](A) \tkzGetPoint{F}
				\tkzDefPointBy[homothety = center D ratio 1/4](A) \tkzGetPoint{M}
				
				\tkzDrawPolygon(A',A,D,C,C',B')
				\tkzDrawSegments(A,B B',B B,C B,D A,B' M,E B',C)
				\tkzDrawSegments[dashed](A',D' C',D' D,D' A',D M,F E,F D,B')
				\tkzDrawPoints[fill=black](A,B,C,D,A',B',C',D',E,F,M)
				\tkzLabelPoints[above](C,D)
				\tkzLabelPoints[left](A,M)
				\tkzLabelPoints[below](A',B',D',F)
				\tkzLabelPoints[right](B,C',E)
				\end{tikzpicture}
			}
			Diện tích hình chữ nhật $A'B'CD$ là $S_{A'B'CD}=A'B'\cdot B'C=a^2\sqrt{2}$.
			\item Gọi $M$ là điểm chia $DA$ theo tỉ số $1:3$. Chứng minh rằng mặt phẳng $(EMF)$ song song với mặt phẳng $(A'B'CD)$.\\
			$\triangle DAB$ có $\dfrac{DM}{MA}=\dfrac{DE}{EB}=\dfrac{1}{3}$ nên $ME\parallel AB$ (định lí Ta-lét đảo).\\
			Mà $AB\parallel A'B'$ nên $ME\parallel A'B'$.\\
			Suy ra $ME\parallel \text{mp}(A'B'CD)$.\\
			$\triangle AB'D$ có $\dfrac{DM}{MA}=\dfrac{B'F}{FA}=\dfrac{1}{3}$ nên $MF\parallel B'D$ (định lí Ta-lét đảo).\\
			Suy ra $MF\parallel \text{mp}(A'B'CD)$.\\
			Mặt phẳng $(MEF)$ chứa hai đường thẳng cắt nhau cùng song song với mp$(A'B'CD)$ nên mp$(MEF)\parallel \text{mp}(A'B'CD)$.
			\item Chứng minh rằng $EF$ song song với mặt phẳng $(A'B'CD)$.\\
			Vì mp$(MEF)\parallel \text{mp}(A'B'CD)$ nên $EF\parallel \text{mp}(A'B'CD)$.
			\item Chứng minh $EF$ song song với mặt phẳng $(A'B'CD)$ mà không sử dụng kết quả của câu b.
			\immini{
				Trong mp$(ABB'A')$, gọi $K$ là giao điểm của $BF$ và $A'B'$.\\
				Ta có $AB\parallel A'B'$ nên $\dfrac{KF}{FB}=\dfrac{B'F}{FA}$ (định lí Ta-lét).\\
				$\Rightarrow \dfrac{KF}{FB}=\dfrac{1}{3}$.\\
				$\triangle BDK$ có $\dfrac{KF}{FB}=\dfrac{DE}{EB}=\dfrac{1}{3}$ nên $EF\parallel DK$ (định lí Ta-lét đảo).\\
				Mà $DK$ nằm trong mp$(A'B'CD)$ nên $EF\parallel \text{mp}(A'B'CD)$.
			}{
				\begin{tikzpicture}[scale=.8, line join=round, line cap=round,>=stealth, font=\footnotesize]
				%\tkzInit[xmin=-.2, xmax=4.2, ymin=-.2, ymax=3]\tkzClip

				\tkzDefPointBy[rotation= center A' angle 90](B') \tkzGetPoint{A}
				\tkzDefPointBy[translation= from B' to C'](A') \tkzGetPoint{D'}
				\tkzDefPointBy[translation=from A' to A](B') \tkzGetPoint{B}
				\tkzDefPointBy[translation=from A' to A](C') \tkzGetPoint{C}
				\tkzDefPointBy[translation=from A' to A](D') \tkzGetPoint{D}
				\tkzDefPointBy[homothety = center D ratio 1/4](B) \tkzGetPoint{E}
				\tkzDefPointBy[homothety = center B' ratio 1/4](A) \tkzGetPoint{F}
				\tkzInterLL(B,F)(A',B') \tkzGetPoint{K}
				
				\tkzDrawPolygon(A',A,D,C,C',B')
				\tkzDrawSegments(A,B B',B B,C B,D A,B' B',C B,K)
				\tkzDrawSegments[dashed](A',D' C',D' D,D' A',D E,F D,K)
				\tkzDrawPoints[fill=black](A,B,C,D,A',B',C',D',E,F,K)
				\tkzLabelPoints[above](C,D)
				\tkzLabelPoints[left](A)
				\tkzLabelPoints[below](A',B',D',K)
				\tkzLabelPoints[right](B,C',E,F)
				\end{tikzpicture}
			}
		\end{enumerate}
	}
\end{vd}

\subsubsection{Bài tập tự luyện}
\begin{bt}%[Phạm An Bình, dự án EX-C2-8-2018][8H4K3]
	Cho hình hộp chữ nhật $ABCD.A'B'C'D'$. Gọi $M$, $N$ theo thứ tự là trung điểm của $AD$, $DC$. Gọi $I$, $K$ theo thứ tự là trung điểm của $B'A'$, $B'C'$. Chứng minh rằng $MN$ song song với $IK$.
	\loigiai{
		\immini{
			$\triangle ACD$ có $M$, $N$ lần lượt là trung điểm của $AD$ và $CD$ nên $MN$ là đường trung bình.\\
			$\Rightarrow MN\parallel AC$.\hfill (1)\\
			$\triangle A'B'C'$ có $I$, $K$ lần lượt là trung điểm của $A'B'$ và $B'C'$ nên $IK$ là đường trung bình.\\
			$\Rightarrow IK\parallel A'C'$.\hfill (2)\\
			Xét tứ giác $ACC'A'$ ta có\\
			$AA'\parallel CC'$ (vì cùng song song với $BB'$),\\
			$AA'=CC'$ (vì cùng bằng $BB'$).
		}{
			\begin{tikzpicture}[scale=.7, line join=round, line cap=round,>=stealth, font=\footnotesize]
			%\tkzInit[xmin=-.2, xmax=4.2, ymin=-.2, ymax=3]\tkzClip
			\tkzDefPoints{0/0/A', 3/0/B', 4/1.5/C'}
			\tkzDefPointBy[rotation= center A' angle 90](B') \tkzGetPoint{A}
			\tkzDefPointBy[translation= from B' to C'](A') \tkzGetPoint{D'}
			\tkzDefPointBy[translation=from A' to A](B') \tkzGetPoint{B}
			\tkzDefPointBy[translation=from A' to A](C') \tkzGetPoint{C}
			\tkzDefPointBy[translation=from A' to A](D') \tkzGetPoint{D}
			\tkzDefMidPoint(A,D) \tkzGetPoint{M}
			\tkzDefMidPoint(D,C) \tkzGetPoint{N}
			\tkzDefMidPoint(A',B') \tkzGetPoint{I}
			\tkzDefMidPoint(B',C') \tkzGetPoint{K}
			
			\tkzDrawPolygon(A',A,D,C,C',B')
			\tkzDrawSegments(A,B B',B B,C A,C M,N)
			\tkzDrawSegments[dashed](A',D' C',D' D,D' A',C' I,K)
			\tkzDrawPoints[fill=black](A,B,C,D,A',B',C',D',M,N,I,K)
			\tkzLabelPoints[above](C,D,N)
			\tkzLabelPoints[left](A,M)
			\tkzLabelPoints[below](A',B',D',I)
			\tkzLabelPoints[right](B,C',K)
			\end{tikzpicture}
		}
		\noindent Suy ra tứ giác $ACC'A'$ là hình bình hành.\\
		Ta có $AA'\perp A'D'$ và $AA'\perp A'B'$ nên $AA'\perp \text{mp}(A'B'C'D')$, suy ra $AA'\perp A'C'$.\\
		Hình bình hành $ACC'A'$ có $\widehat{AA'C'}=90^\circ$ nên $ACC'A'$ là hình chữ nhật.\hfill (3)\\
		Từ (1), (2), (3) suy ra $MN\parallel IK$.
	}
\end{bt}

\begin{dang}{Diện tích}
	Diện tích xung quanh của hình hộp chữ nhật tính bởi công thức $$S_{\mathrm{xq}}=2(a+b)c.$$
	Diện tích toàn phần của hình hộp chữ nhật tính bởi công thức $$S_{\mathrm{tp}}=2(a+b)c+2ab=2(ab+bc+ca).$$
	Trong đó $a$ là chiều dài, $b$ là chiều rộng và $c$ là chiều cao.
\end{dang}

\subsubsection{Bài tập tự luyện}
\begin{bt}%[Phạm An Bình, dự án EX-C2-8-2018][8H4K3]
	Tính diện tích toàn phần của một hình hộp chữ nhật có chiều dài $4$ cm, chiều rộng $3$ cm, đường chéo của hình hộp bằng $13$ cm.
	\loigiai{
		\immini{
			Gọi hình hộp chữ nhật đó là $ABCD.A'B'C'D'$.\\
			Theo đề bài ta có $A'B'=4$ cm, $B'C'=3$ cm, $AC'=13$ cm.\\
			$\triangle A'B'C'$ vuông tại $B'$ có $A'C'^2=A'B'^2+B'C'^2$ (định lí Py-ta-go).\\
			$\Rightarrow A'C'=5$ (cm).\\
			$\triangle AA'C'$ vuông tại $A'$ có $AC'^2=AA'^2+A'C'^2$ (định lí Py-ta-go).\\
			$\Rightarrow AA'=12$ (cm).\\
			Diện tích toàn phần là\\ $S_{\mathrm{tp}}=2(A'B'\cdot B'C' + B'C'\cdot AA' + AA'\cdot A'B')=192$ (cm$^2$).
		}{
			\begin{tikzpicture}[scale=.7, line join=round, line cap=round,>=stealth, font=\footnotesize]
			%\tkzInit[xmin=-.2, xmax=4.2, ymin=-.2, ymax=3]\tkzClip
			\tkzDefPoints{0/0/A', 4/0/B', 5/1/C', 0/3/A}
			\tkzDefPointBy[translation= from B' to C'](A') \tkzGetPoint{D'}
			\tkzDefPointBy[translation=from A' to A](B') \tkzGetPoint{B}
			\tkzDefPointBy[translation=from A' to A](C') \tkzGetPoint{C}
			\tkzDefPointBy[translation=from A' to A](D') \tkzGetPoint{D}
			
			\tkzDrawPolygon(A',A,D,C,C',B')
			\tkzDrawSegments(A,B B',B B,C)
			\tkzDrawSegments[dashed](A',D' C',D' D,D' A',C' A,C')
			\tkzDrawPoints[fill=black](A,B,C,D,A',B',C',D')
			\tkzLabelPoints[above](C,D)
			\tkzLabelPoints[left](A)
			\tkzLabelPoints[below](A',B',D')
			\tkzLabelPoints[right](B,C')
			\end{tikzpicture}
		}
	}
\end{bt}
\begin{bt}%[Phạm An Bình, dự án EX-C2-8-2018][8H4K3]
	Một hình hộp chữ nhật có tổng độ dài các cạnh bằng $140$ cm, khoảng cách từ một đỉnh đến đỉnh xa nhất bằng $21$ cm. Tính diện tích toàn phần.
	\loigiai{
		Gọi $a$, $b$, $c$ là các kích thước của hình hộp chữ nhật, ta có
		$$\heva{&4(a+b+c)=140\\&\sqrt{a^2+b^2+c^2}=21.}\Leftrightarrow \heva{&a+b+c=35\\&a^2+b^2+c^2=441.}$$
		Diện tích toàn phần bằng
		\begin{eqnarray*}
			S_{\mathrm{tp}}= 2(ab+bc+ca)&= &(a+b+c)^2-(a^2+b^2+c^2)\\
			&= &35^2-441=784 \text{ cm}^2.
		\end{eqnarray*}
