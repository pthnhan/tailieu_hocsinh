\section{Diện tích hình tròn, hình quạt tròn}
\subsection{Tóm tắt lí thuyết}
\begin{dn}
Hình quạt tròn là một phần hình tròn giới hạn bởi một cung tròn và hai bán kính đi qua hai mút của cung đó.
\end{dn}
\begin{dn}
Hình viên phân là phần hình tròn giới hạn bởi một cung và dây căng cung ấy.
\end{dn}
\begin{dn}
Hình vành khăn là phần hình tròn nằm giưa hai đường tròn đồng tâm.
\end{dn}

\begin{vd}%Dang toán điển hình VD2
	Tính diện tích hình tròn, biết
	\begin{enumerate}
	\item Bán kính bằng $8$ cm.
	\item Đường kính bằng $12$ cm.
	\item Chu vi của đường tròn bằng $18\pi$ cm.
        \end{enumerate}

	\begin{enumerate}
	 \item Với giả thiết ta có $R=8$ cm $\Rightarrow S=\pi \cdot 8^2=64\pi$ (cm$^2$).
        \item Với giả thiết ta có $2R=8$ cm $\Leftrightarrow R=4$ cm $\Rightarrow S=\pi \cdot 4^2=16\pi$ (cm$^2$).
        \item Với giả thiết ta có $2\pi R=18\pi \Leftrightarrow R=9$ cm $\Rightarrow S=\pi \cdot 9^2=81\pi$ (cm$^2$).
        \end{enumerate}
}
\end{vd}

\begin{bt}%45 PPGT/148
	Tính diện tích hình tròn bán kính $4$ cm và các hình quạt tròn có góc ở tâm $30^\circ$ và $150^\circ$ của hình tròn đó.
	\loigiai{
	Áp dụng công thức $S_{\text{quạt}}=\dfrac{\pi R^2n}{360}$\\
	$+$ Với $R=4$ cm và $n^\circ = 30^\circ$ ta có $S_{\text{quạt}}=\dfrac{\pi \cdot 4^2\cdot 30}{360}=\dfrac{4\pi}{3}$ (cm$^2$).\\
	$+$ Với $n^\circ = 150^\circ$ ta có $S_{\text{quạt}}=\dfrac{\pi \cdot 4^2 \cdot 150}{360}=\dfrac{20\pi}{3}$ (cm$^2$).}
\end{bt}
\begin{bt}%47 PPGT/148
	Ở vườn quốc gia Cúc Phương (tỉnh Ninh Bình) có Cây Chò ngàn năm tuổi thân to đến mức $8$ người ôm mới xuể. Hãy tính thiết diện ngang của thân cây (coi như hình tròn và mỗi sải tay khoảng $1{,}5$ m).
	\loigiai{
	Gọi $d$ là đường kính thân cây. Chu vi thân cây là $\pi \cdot d= 8\cdot 1{,}5=12\Rightarrow d=\dfrac{12}{\pi} \approx 3{,}82$ (m).\\
	Vậy diện tích thiết diện ngang của thân cây là $S=\dfrac{\pi \cdot (3{,}82)^2}{4 }\approx 11{,46}$ (m$^2$).\\
	\emph{Cách khác}\\
	Trong dân gian lưu truyền một quy tắc tìm đường kính khi biết chu vi đường tròn\\
	``Quân bát, phát tam, tồn ngũ, quân nhị'', áp dụng quy tắc này ta được 
	đường kính $$d=\dfrac{5}{16}\cdot 8 \cdot 1{,}5 =3{,}75\,\, (\text{m}).$$
	Vậy $S=\dfrac{\pi \cdot 3{,}75^2}{4}\approx 11{,}46$ (m$^2$).
	
	}
\end{bt}
\begin{bt}%BT1 toán điển hình/275
	Một hình vuông và một hình tròn có cùng chu vi. Hỏi hình nào có diện tích lớn hơn?
	\loigiai{Gọi $C$ là độ dài chu vi của hình vuông và hình tròn.\\
	Gọi $a$ là độ dài cạnh hình vuông, suy ra $C=4a\Leftrightarrow a=\dfrac{C}{4}\Rightarrow S_{hv}=a^2=\dfrac{C^2}{16}$.\\
	Gọi $R$ là bán kính của hình tròn, suy ra\\
	$C=2\pi R\Leftrightarrow R=\dfrac{C}{2\pi}\Rightarrow S_{ht}=\pi R^2=\pi \cdot \dfrac{C^2}{4{\pi}^2}=\dfrac{C^2}{4\pi}>\dfrac{C^2}{16}$.\\
	Vậy hình tròn có diện tích lớn hơn diện tích hình vuông.}
\end{bt}
