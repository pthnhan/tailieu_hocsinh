\begin{ex}%[12-TN-BGD-2]%[Nguyễn Tuấn Anh]%[2D3K2-2]
	Cho hàm số $y=f(x)>0; \forall x \geq 0,$ thỏa mãn $\heva{&f''(x) \cdot f(x) -2[f'(x)]^2+xf^3(x)=0,\\&f'(0)=0;f(0)=1.}$ Tính $f(1)$.
	\choice
	{$\dfrac{2}{3}$}
	{$\dfrac{3}{2}$}
	{\True $\dfrac{6}{7}$}
	{$\dfrac{7}{6}$}
	\loigiai{
		Giả sử tồn tại hàm số $f$ thỏa bài toán, với $x \geq 0$, ta đặt:
$$f\left( x \right) = \dfrac{1}{g\left( x \right)}.$$ 
Suy ra
$$ f'\left( x \right) = \dfrac{ - g'\left( x \right)}{{g^2}\left( x \right)};f''\left( x \right) =  - \dfrac{g''\left( x \right)g\left( x \right) - 2{\left[ {g'\left( x \right)} \right]}^2}{g^3\left( x \right)}$$
Thay vào giả thiết ta được:
$$\begin{array}{l}
\dfrac{ - g''\left( x \right)g\left( x \right) + 2\left[ g'\left( x \right) \right]^2}{g^4\left( x \right)} - 2\left[ \dfrac{g'\left( x \right)}{g^2\left( x \right)} \right]^2 + \dfrac{xg\left( x \right)}{g^4\left( x \right)} = 0\\
\Leftrightarrow \dfrac{ - g''\left( x \right)g\left( x \right) + xg\left( x \right)}{g^4\left( x \right)} = 0\\
\Leftrightarrow g''\left( x \right) = x \Leftrightarrow g\left( x \right) = \dfrac{x^3}{6} + \dfrac{c_1x}{2} + c_2.
\end{array}$$
Hay nói cách khác:
$$f\left( x \right) = \dfrac{6}{x^3 + 3c_1x + 6c_2}.$$	
Vì $f'(0)=0$; $f(0)=1$ nên ta được $c_1=0;c_2=1.$ thức hàm số $f(x)=\dfrac{6}{x^3+1}$. Thử lại ta thấy hàm $f$ thỏa mãn. Do đó $f(1)=\dfrac{6}{7}$.}
\end{ex}
