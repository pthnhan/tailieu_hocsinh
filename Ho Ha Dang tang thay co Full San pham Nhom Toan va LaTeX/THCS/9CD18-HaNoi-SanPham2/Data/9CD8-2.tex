\begin{tcolorbox}
	\begin{itemize}
		\item Sử dụng các đánh giá cơ bản như $a^2\geq 0$, $-a^2\leq 0$. 
		\item Sử dụng điều kiện để phương trình bậc hai có nghiệm (phương pháp miền giá trị).
		\item Với $a,b \geq 0$, ta có $a+b\geq 2\sqrt{a\cdot b}$.
		\item Với $a,b,c \geq 0$, ta có $a+b+c\geq 3\sqrt[3]{a\cdot b\cdot c}$.
		\item Với $a,b>0$, ta có $\dfrac{1}{a}+\dfrac{1}{b}\geq\dfrac{4}{a+b}$.
		\item Với $a,b,c > 0$, ta có $\dfrac{1}{a}+\dfrac{1}{b} +\dfrac{1}{c}\geq \dfrac{9}{a+b+c}$.
	\end{itemize}
\begin{note}
	Các bài toán tìm giá trị lớn nhất, nhỏ nhất cần phải chỉ ra đẳng thức xảy ra khi nào.
\end{note}
\end{tcolorbox}

\begin{bt} Tìm giá trị nhỏ nhất của biểu thức $P=\dfrac{4x+3}{x^2+1}$.
	\loigiai
	{ Ta có $P=\dfrac{4x+3}{x^2+1}=-1+\dfrac{x^2+4x+4}{x^2+1} =-1 +\dfrac{(x+2)^2}{x^2+1} \geq -1$.\\
		Dấu `` $=$ '' xảy ra khi $x+2=0 \Leftrightarrow x=-2$. Vậy giá trị nhỏ nhất của $P$ là $-1$.
	}
\end{bt}

\begin{bt}
	Tìm giá trị lớn nhất và giá trị nhỏ nhất của $A=\dfrac{3-4x}{x^2+1}$.
	\loigiai{
		Ta có \\
		$A=\dfrac{x^2-4x+4-x^2-1}{x^2+1}=\dfrac{(x-2)^2}{x^2+1}-1\ge -1.$\\
		Vậy	$\min A=-1$ khi và chỉ khi $x=2$.\\
		Mặt khác\\
		$A=\dfrac{4x^2+4-4x^2-4x-1}{x^2+1}=4-\dfrac{(2x+1)^2}{x^2+1}\le 4$.\\
		Vậy $\max A =4$ khi và chỉ khi $x=-\dfrac{1}{2}$.
	}
\end{bt}

\begin{bt}
	Tìm giá trị nhỏ nhất của $A=\dfrac{x^4+4x+4}{x}$ với $x>0$.
	\loigiai{
		Ta có\\
		$A=\dfrac{x^4+4x+4}{x}=x+4+\dfrac{4}{x}$.\\
		Với $a,b$ là hai số dương ta có $a+b\ge 2\sqrt{ab}$. Khi đó $\dfrac{4}{x}+x\ge 2\cdot 2=4$.
		Vậy $A\ge 4+4=8$, suy ra $\min A=8 $ khi và chỉ khi $\dfrac{4}{x}=x\Leftrightarrow x= 2$.
	}
\end{bt}

\begin{bt}
	Tìm giá trị nhỏ nhất của $A=\dfrac{3x^2-8x+6}{x^2-2x+1}$.
	\loigiai{
		{\bf Cách 1} Viết $A$ dưới dạng tổng hai biểu thức không âm\\
		$A=\dfrac{2\left( x^2-2x+1\right) +\left(x^2-4x+4 \right)  }{x^2-2x+1}=2+\dfrac{(x-2)^2}{(x-1)^2}\ge 2$.\\
		Vậy $\min A=2$ khi và chỉ khi $x=2$.\\
		{\bf Cách 2} Đặt $x-1=y$ thì $x=y+1$ ta có\\
		$A=\dfrac{3(y+1)^2-8(y+1)+6}{(y+1)^2-2(y+1)+1}= \dfrac{3y^2+6y+3-8y-8+6}{y^2+2y+1-2y-2+1}$\\ $A=\dfrac{3y^2-2y+1}{y^2}=3-\dfrac{2}{y} +\dfrac{1}{y^2}= \left(\dfrac{1}{y}-1\right)^2 +2$.\\
		Vậy $\min A=2$ khi và chỉ khi $x=2$.
	}
\end{bt}

\begin{bt}
	Tìm giá trị nhỏ nhất và giá trị lớn nhất của $A=\dfrac{x}{x^2-\sqrt{x}+1}$ với $x\ge 0$.
	\loigiai{
		Ta có 
		$B=\dfrac{x}{\left(\sqrt{x}-\dfrac{1}{2}\right)^2+\dfrac{3}{4}}$.\\
		Với $x=0$ thì $B=0$.\\
		Với $x>0$, ta có $B=\dfrac{1}{\dfrac{1}{x}-\dfrac{1}{\sqrt{x}}+1}= \dfrac{1}{\left(\dfrac{1}{\sqrt{3}}-\dfrac{1}{2} \right)^2+\dfrac{3}{4}}$.\\
		Vì tử số và mẫu số là hằng số nên $B\le \dfrac{1}{\dfrac{3}{4}}=\dfrac{4}{3}$.\\
		Dấu bằng xảy ra khi và chỉ khi $\dfrac{1}{\sqrt{x}}=\dfrac{1}{2}\Rightarrow x=4.$\\
		Vậy $\max B=\dfrac{4}{3}$ khi $x=4.$
	}
\end{bt}

\begin{bt}
	Tìm giá trị lớn nhất, giá trị nhỏ nhất của biểu thức $P=\dfrac{2x+1}{x^2+6}$.
	\loigiai{
		Ta có $P=\dfrac{2x+1}{x^2+6}\Leftrightarrow Px^2-2x+6P-1=0.\qquad (\star)$.\\
		Để $P$ tồn tại giá trị lớn nhất, giá trị nhỏ nhất thì phương trình $(\star)$ phải có nghiệm. \\
		Tức là $\Delta'\ge 0\Leftrightarrow -6P^2+P+1\ge 0\Leftrightarrow -\dfrac{1}{3}\le  P \le \dfrac{1}{2}.$\\
		Với $P=\dfrac{1}{2}$ ta có $\dfrac{2x+1}{x^2+6}=\dfrac{1}{2}\Rightarrow x=2$.\\
		Với $P=-\dfrac{1}{3}$ ta có $\dfrac{2x+1}{x^2+6}=-\dfrac{1}{3}\Rightarrow x=-3$.\\
		Vậy $\max P=\dfrac{1}{2}$ khi $x=2$; $\min P=-\dfrac{1}{2}$ khi $x=-3$
	}
\end{bt}

\begin{bt}
	Tìm giá trị lớn nhất của biểu thức $S= \dfrac{3(x+1)}{x^3+x^2+x+1}.$
	
	\loigiai
	{
		Điều kiện xác định của biểu thức là $x^3+x^2+x+1\ne0\Leftrightarrow(x+1)(x^2+1)\ne0\Leftrightarrow x\ne-1.$\\
		Khi đó ta có $$S=\dfrac{3(x+1)}{x^3+x^2+x+1}=\dfrac{3(x+1)}{(x+1)(x^2+1)}=\dfrac{3}{x^2+1}.$$
		Vì tử số của $S$ bằng 3 là dương nên $S$ đạt giá trị lớn nhất khi mẫu số $x^2+1$ đạt giá trị nhỏ nhất. Ta có $x^2+1\geq1,\ \forall x\in\mathbb{R}$ nên $x^2+1$ đạt giá trị nhỏ nhất bằng 1 khi $x=0.$ Khi đó $S$ đạt giá trị lớn nhất bằng 3 khi $x=0.$
	}
\end{bt}

\begin{bt}
	Cho hai số thực dương $x,y$ thỏa mãn $xy = 1$. Tìm giá trị nhỏ nhất của biểu thức:
	\begin{center}
		$M=x^2+y^2+\dfrac{3}{x+y+1}$.
	\end{center}
	\loigiai
	{Ta có $(x+y)^2\geq 4xy=4\Rightarrow x+y\geq 2 $ (vì $x+y>0$).\\
		Đặt $t=x+y\geq 2$. Ta có\\
		$\begin{array}{ll}
		M& = x^2+y^2+\dfrac{3}{x+y+1}=(x+y)^2+\dfrac{3}{x+y+1}-2xy \\ 
		& =t^2+\dfrac{3}{t+1}-2=\dfrac{t^3+t^2-2t+1}{t+1}.
		\end{array}$\\
		$\begin{array}{ll}
		\Rightarrow M-3 &=\dfrac{t^3+t^2-2t+1}{t+1}-3=\dfrac{t^3+t^2-5t-2}{t+1} \\
		&=\dfrac{(t-2)(t^2+3t+1)}{t+1}\geq 0  \text{ (vì }t\geq 2\text{)}.\\
		\Rightarrow M \ge 3.
		\end{array}$\\
		Đẳng thức xảy ra khi $\heva{&x+y=2\\&xy=1}\Leftrightarrow x=y=1 $. Vậy $\min M=3\Leftrightarrow x=y=1.$    
	}
\end{bt}

\begin{bt}%[Đề thi vào 10, Sở giáo dục Hưng Yên, 2017]%[9D5G1]
	Cho hai số thực dương $x,y$ thỏa mãn điều kiện $x+y\leq 4$. Tìm giá trị nhỏ nhất của biểu thức $P=\dfrac{2}{x^2+y^2}+\dfrac{35}{xy}+2xy.$
	\loigiai{ Ta có
		\begin{align*}
		P &=\dfrac{2}{x^2+y^2}+\dfrac{35}{xy}+2xy \\
		&=\dfrac{2}{x^2+y^2}+\dfrac{2}{2xy}+\dfrac{34}{xy}+\dfrac{17xy}{8}-\dfrac{xy}{8}\\
		&=2\left(\dfrac{1}{x^2+y^2}+\dfrac{1}{2xy}\right) +\left(\dfrac{34}{xy}+\dfrac{17xy}{8}\right)-\dfrac{xy}{8}\\
		&\geq \dfrac{2.4}{x^2+y^2+2xy}+2\sqrt{\dfrac{34.17}{8}}-\dfrac{(x+y)^2}{32}\\
		&\geq \dfrac{8}{16}+17-\dfrac{1}{2}= 17.
		\end{align*}
		Dấu "$=$" xảy ra $\Leftrightarrow x=y=2$ .
	}
\end{bt}

\begin{bt} Cho $x\geq 2$. Tìm giá trị nhỏ nhất của biểu thức $P=x+\dfrac{1}{x^2}$.
	
	\loigiai{
		Ta có $P= \left(\dfrac{x}{8}+\dfrac{x}{8}+\dfrac{1}{x^2}\right)+\dfrac{6x}{8}$.\\
		Áp dụng bất đẳng thức AM - GM, ta có $P \geq 3\sqrt[3]{\dfrac{x}{8}\cdot \dfrac{x}{8}\cdot \dfrac{1}{x^2}} +\dfrac{6\cdot2}{8}=\dfrac{9}{4}$. \\
		Đẳng thức xảy ra khi $x=2$. Vậy giá trị nhỏ nhất của biểu thức $P$ là $\dfrac{9}{4}$.
	}
\end{bt}

\begin{bt}
	Tìm giá trị nhỏ nhất của biểu thức 	$S= x+y+\dfrac{5}{x}+\dfrac{5}{y}$, biết $x, y$ là các số thực dương và thỏa mãn $x+y\leq 4$.
	\loigiai{
		Ta có	$S= x+y+\dfrac{5}{x}+\dfrac{5}{y} =\left(x+\dfrac{4}{x}\right)+\left(y+\dfrac{4}{y}\right)+\left(\dfrac{1}{x}+\dfrac{1}{y}\right)$.\\
		Áp dụng bất đẳng thức AM-GM và bất đẳng thức Cauchy-Schwarz, ta có 
		$$S\geq 2\sqrt{x\cdot\dfrac{4}{x}}+ 2\sqrt{y\cdot \dfrac{4}{y}} +\dfrac{4}{(x+y)^2} = 8+\dfrac{4}{(x+y)}.$$
		Mà $x+y \leq 4 \Rightarrow \dfrac{4}{(x+y)}\geq 1$. Suy ra $S\geq 9$, đẳng thức xảy ra khi $x=y=2$. Vậy giá trị nhỏ nhất của $S$ bằng $9$.
	}
	
\end{bt}

\begin{bt}%[Nguyễn Sỹ]
	Tìm giá trị nhỏ nhất của biểu thức $P= \dfrac{x^2+4x+9}{\sqrt{x^2+4x+8}}$.
	\loigiai{
		\begin{itemize}
			\item Điều kiện xác định $x \in \mathbb{R}$.
			\item Đặt $t=\sqrt{x^2+4x+8}=\sqrt{(x+2)^2+4}\geq 2$, suy ra điều kiện $t\geq 2$.
			\item Biểu thức đã cho trở thành $P=\dfrac{t^2+1}{t}=t+\dfrac{1}{t} =\dfrac{t}{4}+\dfrac{1}{t}+\dfrac{3t}{4}$.\\
			Áp dụng bất đẳng thức AM - GM, ta có $P\geq 2\sqrt{\dfrac{t}{4}\cdot \dfrac{1}{t}} +\dfrac{3t}{4}\geq 1+ \dfrac{3}{2} =\dfrac{5}{2}$, vì $t\geq 2$.
		\end{itemize}
		Do $P\geq \dfrac{5}{2}$, dấu ``=" xảy ra khi $t=2 \Rightarrow x=-2$. Vậy giá trị nhỏ nhất của $P$ bằng $\dfrac{5}{2}$.
	}
\end{bt}

\begin{bt}
	Cho $x,y,z>0$ và $x+y+z=1$. Chứng minh rằng $$\dfrac{1}{x^2+2yz}+\dfrac{1}{y^2+2zx}+\dfrac{1}{z^2+2xy}\geq 9.$$
	\loigiai{
		Đặt $a=x^2+2yz$, $b=y^2+2zx$, $c=z^2+2xy$ thì $a,b,c>0$ và $a+b+c=(x+y+z)^2=1$.\\
		Ta có $(a+b+c)\left(\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}\right)\geq 3\sqrt[3]{abc}\cdot 3\sqrt[3]{\dfrac{1}{abc}}=9$.\\
		Khi đó, $\dfrac{1}{a}+\dfrac{1}{b}+\dfrac{1}{c}\geq \dfrac{9}{a+b+c}=9$, hay 
		$$\dfrac{1}{x^2+2yz}+\dfrac{1}{y^2+2zx}+\dfrac{1}{z^2+2xy}\geq 9.$$
	}
\end{bt}

\begin{bt}
	Cho hai số thực dương $x,\ y$ thỏa mãn $2x+3y\leq2.$ Tìm giá trị nhỏ nhất của biểu thức $P=\dfrac{3}{4x^2+9y^2}+\dfrac{5}{xy}.$
	\loigiai{ 
		Áp dụng bất đẳng thức quen thuộc sau: \[\dfrac{1}{a}+\dfrac{1}{b}\geq\dfrac{4}{a+b}.\]
		Ta có:
		\begin{align*}
		&P=\dfrac{3}{4x^2+9y^2}+\dfrac{5}{xy}=3\left(\dfrac{1}{(2x)^2+(3y)^2}+\dfrac{1}{12xy}\right)+\dfrac{19}{4xy}\geq\dfrac{12}{(2x+3y)^2}+\dfrac{19}{4xy}.
		\end{align*} Vì $x,\ y>0$ và $2x+3y\leq2$ nên $\dfrac{12}{(2x+3y)^2}\geq3.$\\
		Và vì $2\geq 2x+3y\geq2\sqrt{6xy}\Rightarrow xy\leq\dfrac{1}{6}\Rightarrow\dfrac{19}{4xy}\geq\dfrac{57}{2}.$
		Do đó $P\geq3+\dfrac{57}{2}=\dfrac{63}{2}.$\\
		Vậy $P_{\min}=\dfrac{63}{2}$ khi $\begin{cases}
		2x=3y\\
		2x+3y=2
		\end{cases}\Leftrightarrow \begin{cases}x=\dfrac{1}{2}\\y=\dfrac{1}{3}.\end{cases}$
		\begin{note} Bài toán này là một dạng toán cơ bản, áp dụng bất đẳng thức $\dfrac{1}{a}+\dfrac{1}{b}\geq\dfrac{4}{a+b}$ và kỹ thuật chọn điểm rơi, tách $\dfrac{5}{xy}=\dfrac{3}{12xy}+\dfrac{19}{4xy}$ sau khi xử lý được phần thứ nhất để áp dụng bất đẳng thức trên vào biểu thức $P.$
	\end{note}}
\end{bt}

\begin{bt}
	Cho $a,\ b,\ c$ là ba số dương thỏa mãn $abc=1$. Chứng minh rằng
	$$\dfrac{1}{a^2+2b^2+3}+\dfrac{1}{b^2+2c^2+3}+\dfrac{1}{c^2+2a^2+3}\leq \dfrac{1}{2}.$$
	\loigiai{
		Ta có $a^2+2b^2+3=(a^2+b^2)+(b^2+1)+2\geq 2ab+2b+2$.\\
		Tương tự $b^2+2c^2+3\geq 2bc+2c+2$, $c^2+2a^2+3\geq 2ca+2a+2$.\\
		Suy ra
		$$\dfrac{1}{a^2+2b^2+3}+\dfrac{1}{b^2+2c^2+3}+\dfrac{1}{c^2+2a^2+3}\leq\dfrac{1}{2}\left(\dfrac{1}{ab+b+1}+\dfrac{1}{bc+c+1}+\dfrac{1}{ca+a+1}\right).$$
		Mà (chú ý $abc=1$)
		\begin{align*}
		\dfrac{1}{ab+b+1}+\dfrac{1}{bc+c+1}+\dfrac{1}{ca+a+1}=\ &\dfrac{1}{ab+b+1}+\dfrac{ab}{b+1+ab}+\dfrac{b}{1+ab+b}\\
		=\ &\dfrac{ab+b+1}{ab+b+1}=1,
		\end{align*}
		nên 
		$$\dfrac{1}{a^2+2b^2+3}+\dfrac{1}{b^2+2c^2+3}+\dfrac{1}{c^2+2a^2+3}\leq\dfrac{1}{2}.$$
		Đẳng thức xảy ra khi $a=b=c=1$.
	}
\end{bt}

\begin{bt}%[Lê Thanh Hải]%[8D2G9]
	Cho $x, y$ là các số thực dương thỏa mãn $x+y=1$. Tìm giá trị nhỏ nhất của biểu thức $A=\dfrac{1}{x}+\dfrac{4}{y}$.
	\loigiai{
		\begin{note}
			Rất dễ có một đánh giá $A=\dfrac{1}{x}+\dfrac{4}{y} \geq 2\cdot \sqrt{\dfrac{1}{x}\cdot \dfrac{4}{y}}=\dfrac{4}{\sqrt{xy}} \geq \dfrac{8}{x+y}=8.$\\ Dấu bằng đạt tại $\heva{&\dfrac{1}{x}=\dfrac{4}{y}\\&x=y} \Leftrightarrow x=y=0 \, \text{(vô lý)}$.
		\end{note}
		Lời giải đúng như sau\\
		Vì $x+y=1$ nên $A=(x+y)\left(\dfrac{1}{x}+\dfrac{4}{y}\right)=5+\dfrac{4x}{y}+\dfrac{y}{x} \geq 5+2\cdot \sqrt{\dfrac{4x}{y}\cdot \dfrac{y}{x}}=9$.\\ Dấu $"="$ xảy ra khi và chỉ khi $\heva{&\dfrac{4x}{y}=\dfrac{y}{x}\\&x+y=1 }\Leftrightarrow \heva{&y=2x\\&x+y=1} \Leftrightarrow \heva{&x=\dfrac{1}{3}\\&y=\dfrac{2}{3}.}$
	}
\end{bt}

\begin{bt}%[Lê Thanh Hải]%[8D2G9]
	Cho hai số $x, y$ thỏa $x^2+y^2=1$ và $x\neq 0, y\neq 0$. Tìm giá trị lớn nhất của $$P=\dfrac{2+xy}{2x^2+3y^2-1}.$$
	\loigiai{
		Từ giả thiết ta có ngay $P=\dfrac{2x^2+xy+2y^2}{x^2+2y^2}$.
		Chia tử và mẫu cho $y^2 \neq 0$, ta được
		{\allowdisplaybreaks
			\begin{eqnarray*}
				P& = &\dfrac{2x^2+xy+2y^2}{x^2+2y^2}\\
				& = &\dfrac{2\left(\dfrac{x}{y}\right)^2+\dfrac{x}{y}+2}{\left(\dfrac{x}{y}\right)^2+2}
			\end{eqnarray*}
		}
		Đặt $\dfrac{x}{y}=t$. Khi đó $P=\dfrac{2t^2+t+2}{t^2+2} \Rightarrow t^2\left(P-2\right)-t+2P-2=0. \hfil(*)$\\
		TH1: $P=2$. Khi đó $(*)$ là phương trình bậc nhất có nghiệm $t=2$.\\
		TH2: $P\neq 2$. Khi đó $(*)$ là phương trình bậc hai ẩn $t$. Để phương trình có nghiệm thì \\ $\Delta \geq 0 \Leftrightarrow 1-4(P-2)(2P-2) \geq 0 \Leftrightarrow 8P^2-24P+15 \leq 0 \Leftrightarrow \dfrac{6-\sqrt{6}}{4} \leq P \leq \dfrac{6+\sqrt{6}}{4}$. \\
		Vậy giá trị lớn nhất của $P$ là $\dfrac{6+\sqrt{6}}{4}$.	
	}
\end{bt}


\begin{bt}%[Lê Thanh Hải]%[8D2G9]
	Tìm GTLN và GTNN của biểu thức $$A=\dfrac{x+2y+1}{x^{2}+y^{2}+7}$$
	\loigiai{
		Ta xét hai trường hợp sau đây:\\
		\textbf{Trường hợp 1:} $x+2y+1=0 \Rightarrow A=0$.\\
		\textbf{Trường hợp 2:} $x+2y+1 \neq 0$. Đặt $A= \dfrac{1}{N} (N \neq 0)$, ta có:\\
		{\allowdisplaybreaks
			\begin{eqnarray*}
				& & \dfrac{x+2y+1}{x^{2}+y^{2}+7} = \dfrac{1}{N}\\
				& \Leftrightarrow & x^{2}+ y^{2}+7 -Nx-2Ny-N = 0 \\
				& \Leftrightarrow & \left( x- \dfrac{N}{2} \right)^{2}+ \left( y-N \right)^{2}= \dfrac{5N^{2}+4N-28}{4}
			\end{eqnarray*}
		}
		
		Vì $\left( x- \dfrac{N}{2} \right)^{2} \geq 0$  và $\left( y-N \right)^{2} \geq 0$ nên suy ra:
		$$5N^{2}+4N-28 \geq 0 \Leftrightarrow \left(N-2 \right) \left( 5N+14 \right) \geq 0 \Leftrightarrow \hoac{&N \geq 2\\&N \leq - \dfrac{14}{5}}$$
		Do đó $- \dfrac{5}{14} \leq A \leq \dfrac{1}{2}$.\\
		Từ hai trường hợp trên ta suy ra:\\
		$maxA= \dfrac{1}{2} \Leftrightarrow N=2 \Leftrightarrow \heva{&x=1\\&y=2}$\\
		$minA= - \dfrac{5}{14} \Leftrightarrow N= - \dfrac{14}{5} \Leftrightarrow \heva{&x=- \dfrac{7}{5}\\&y=- \dfrac{14}{5}}$
		
	}
\end{bt}

\begin{bt}%[Lê Thanh Hải]%[8D2G9]
	Cho hai số $x, y$ khác $0$ thỏa mãn $\left( x+y \right) xy= x^{2}+y^{2}-xy$. Tìm GTLN của biểu thức $A= \dfrac{1}{x^{3}}+ \dfrac{1}{y^{3}}$? 
	\loigiai{
		
		\begin{eqnarray*}
			A &=& \dfrac{1}{x^{3}}+ \dfrac{1}{y^{3}}= \dfrac{x^{3}+y^{3}}{x^{3}y^{3}}
			= \dfrac{\left(x+y \right) \left( x^{2}+y^{2}-xy \right)}{x^{3}y^{3}}
			= \dfrac{xy \left( x+y \right)^{2}}{x^{3}y^{3}} = \left( \dfrac{x+y}{xy} \right)^{2}
			= \left( \dfrac{(x+y)^{2}}{x^{2}+y^{2}-xy}\right)^{2}\\
			&=& \left( \dfrac{x^{2}+y^{2}+2xy}{x^{2}+y^{2}-xy}\right)^{2}
			= \left(1+ \dfrac{3xy}{x^{2}+y^{2}-xy}\right)^{2}
		\end{eqnarray*}
		
		
		$	\text{Đặt } \dfrac{3xy}{x^{2}+y^{2}-xy}= \dfrac{1}{E} (E \neq 0)\\
		\Leftrightarrow  x^{2}+y^{2}-xy= 3xyE

\begin{bt}%[Lê Thanh Hải]%[Nâng cao và phát triển toán tập 2, Vũ Hữu Bình]%[9D4G7]
	Tìm giá trị lớn nhất giá trị nhỏ nhất của $A=\dfrac{x}{x^2+1}$.
	\loigiai{
		$A$ nhận giá trị $a$ khi và chỉ khi phương trình sau có nghiệm 
		\begin{equation}
		a=\dfrac{x}{x^2+1}\Leftrightarrow ax^2-x+a=0.
		\end{equation}
		\begin{itemize}
			\item Xét $a=0$ thì $(1)$ có nghiệm $x=0$.
			\item Xét $a\neq 0$ thì $(1)$ có nghiệm khi và chỉ khi $\Delta \geqslant 0$, tức là $$1-4a^2\geqslant 0\Leftrightarrow (2a-1)(2a+1)\leqslant 0\Leftrightarrow -\dfrac{1}{2}\leqslant a\leqslant \dfrac{1}{2}.$$
			Với $a=-\dfrac{1}{2}$ hay $a=\dfrac{1}{2}$ thì nghiệm của $(1)$ là $x=\dfrac{1}{2a}$.
			\\ Nên $a=-\dfrac{1}{2}$ thì $x=-1$, $a=\dfrac{1}{2}$ thì $x=1$.
		\end{itemize}
		Kết luận: Với cả hai trường hợp trên ta có $$\min A=\dfrac{-1}{2}\Leftrightarrow x=-1; \max A=\dfrac{1}{2}\Leftrightarrow x=1.$$
	}	
\end{bt}
\begin{bt}%[Lê Thanh Hải]%[Nâng cao và phát triển toán tập 2, Vũ Hữu Bình]%[9D4K7]
	Tìm giá trị lớn nhất và giá trị nhỏ nhất của $A=\dfrac{x^2-x+1}{x^2+x+1}$.
	\loigiai{
		Biểu thức $A$ nhận giá trị $a$ khi và chỉ khi phương trình ẩn $x$ sau đây có nghiệm 
		\begin{equation}
		a=\dfrac{x^2-x+1}{x^2+x+1}
		\end{equation}
		Do $x^2+x+1\neq 0$ nên
		\begin{equation}
		(1)\Leftrightarrow ax^2+ax+a=x^2-x+1 \Leftrightarrow (a-1)x^2+(a+1)x+(a-1)=0
		\end{equation}
		\begin{itemize}
			\item Xét $a=1$ thì $(2)$ có nghiệm $x=0$.
			\item Xét $a\neq 0$ thì $(2)$ có nghiệm khi và chỉ khi $\Delta \geqslant 0$, tức là 
			\begin{eqnarray*}
				(a+1)^2-4(a-1)^2\geqslant 0&\Leftrightarrow & (a+1+2a-2)(a+1-2a+2)\geqslant 0
				\\ &\Leftrightarrow & (3a-1)(a-3)\leqslant 0\Leftrightarrow \dfrac{1}{3}\leqslant a\leqslant 3\ (a\neq 1).
			\end{eqnarray*} 
			Với $a=\dfrac{1}{3}$ hoặc $a=3$ thì nghiệm của $(2)$ là $x=\dfrac{-(a+1)}{2(a-1)}=\dfrac{a+1}{2(1-a)}$.
			\\ Nên $a=\dfrac{1}{3}$ thì $x=1$, $a=3$ thì $x=-1$.
		\end{itemize}
		Kết luận: Với cả hai trường hợp trên ta có $$\min A=\dfrac{1}{3}\Leftrightarrow x=1; \max A=3 \Leftrightarrow x=-1.$$	
	}
\end{bt}
