	Tính giới hạn $\lim\limits_{x \rightarrow 0} \dfrac{\mathrm{e}^x - \cos x}{\sin x}$.
	\choice
	{$2$}
	{\True $1$}
	{$-1$}
	{$0$}
	\loigiai{
		Ta có $A=\lim\limits_{x \rightarrow 0} \dfrac{\mathrm{e}^x - \cos x}{\sin x} = \lim\limits_{x \rightarrow 0} \dfrac{\dfrac{\mathrm{e}^x-1}{x}+\dfrac{1-\cos x}{x}}{\dfrac{\sin x}{x}}$.\\
		Sử dụng các giới hạn đặc biệt của hàm số mũ và hàm số lượng giác ta có
		\begin{eqnarray*}
		\lim\limits_{x \rightarrow 0} \dfrac{\mathrm{e}^x - 1}{x} &=& 1.\\
		\lim\limits_{x \rightarrow 0} \dfrac{1-\cos x}{x} &=& \lim\limits_{x \rightarrow 0} \dfrac{2\sin^2\dfrac{x}{2}}{x} = \lim\limits_{x \rightarrow 0} \left[\dfrac{x}{2} \cdot \left( \dfrac{\sin \dfrac{x}{2}}{\dfrac{x}{2}}\right)^2\right] = 0 \cdot 1=0. \\
		\lim\limits_{x\rightarrow 0} \dfrac{\sin x}{x} &=&1.
		\end{eqnarray*}
		\noindent Do đó $A=\dfrac{1+0}{1}=1$.
	}
\end{ex}
