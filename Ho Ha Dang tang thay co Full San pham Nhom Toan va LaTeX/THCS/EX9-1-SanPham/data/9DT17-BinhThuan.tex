\begin{ex}%[Đề thi vào 10, Sở giáo dục Bình Thuận, 2017]%[9D4B6]%[9D3B4]
	\hfill
	\begin{enumerate}
		\item Giải phương trình $x^2-4x+3=0$.
		\item Giải hệ phương trình $\heva{&5x-2y=11\\&x+y=-2.}$
	\end{enumerate}
	\loigiai
	{
		\begin{enumerate}
			\item 
			Do  $a+b+c=0$ nên $x=1$ hoặc $x=3$.
			\item $\heva{&5x-2y=11\\&x+y=-2}\Rightarrow \heva{&5x-2y=11\\&2x+2y=-4}\Rightarrow\heva{&7x=7\\&x+y=-2}\Rightarrow\heva{& x=1\\ & y=-3.}$\\
			Vậy hệ phương trình có nghiệm $(x;y)=(1;-3)$.
		\end{enumerate}
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Bình Thuận, 2017]%[9D1B3]
	Cho biểu thức: $A=2\sqrt{5}+3\sqrt{45}-\sqrt{500}$ và $B=\sqrt{20}$. Tính tích $A.B?$
	
	\loigiai
	{
		Ta có $A=2\sqrt{5}+9\sqrt{5}-10\sqrt{5}=\sqrt{5}$, $B=2\sqrt{5}$. Vậy $A.B=\sqrt{5}.2\sqrt{5}=10$.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Bình Thuận, 2017]%[9D4K8]
	Một nhóm học sinh có kế hoạch nhận trồng $200$ cây tràm giúp cho gia đình bạn An. Vì có $2$ học sinh bị bệnh không tham gia được nên mỗi học sinh còn lại phải trồng thêm $5$ cây so với dự định để hoàn thành kế hoạch(Biết số cây mỗi học sinh trồng là như nhau). Tính số học sinh thực tế đã tham gia trồng cây?
	
	\loigiai
	{
		Gọi số học sinh thực tế tham gia trồng cây là $x$ ($x\in \mathbb{N^*})$, nên số học sinh ban đầu là $x+2$. Số cây dự tính mỗi học sinh phải trồng là $\dfrac{200}{x+2}.$\\ Số cây thực tế mỗi học sinh phải trồng là $\dfrac{200}{x}$.\\
		Theo bài ta có $\dfrac{200}{x}=\dfrac{200}{x+2}+5\Leftrightarrow \hoac{&x=8\\&x=-10.}$ \\
		Vì $x\in \mathbb{N^*}$ nên  số học sinh thực tế đã tham gia gia trồng cây là $8$.
	}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục Bình Thuận, 2017 2016]%[9H3B7]%[9H3K3]%[9H3K5]%[9H3G10]
	Cho tứ giác $ABCD$ nội tiếp đường tròn tâm $O$ đường kính $AD=2R$. Hai đường chéo $AC$ và $BD$ cắt nhau tại $E$. Kẻ $EF$ vuông góc với $AD$ tại $F$.
	\begin{enumerate}
		\item Chứng minh: tứ giác $ABEF$ nội tiếp.
		\item Chứng minh: $\widehat{DBC}=\widehat{DBF}$.
		\item Tia $BF$ cắt $(O)$ tại $K$. Chứng minh: ${EF}\parallel{CK}$.
		\item Giả sử $\widehat{EFB}=60^\circ$ tính theo $R$ diện tích hình giới hạn bởi dây $BC$ và cung nhỏ $BC$?
	\end{enumerate}
	\loigiai
	{\immini{\begin{enumerate}
				\item
				Ta có $\widehat{ADB}=90^\circ$(góc nội tiếp chắn nửa đường tròn) suy ra tứ giác $ABEF$ có hai góc đối bằng $90^\circ$ nên nội tiếp.
				\item Ta có $\widehat{DBC}=\widehat{DAC}=\widehat{EBF}$(góc nội tiếp cùng chắn một cung).
				\item Ta có $\widehat{EFB}=\widehat{EAB}=\widehat{CKB}$ mà góc $\widehat{EFB}$ và $\widehat{CKB}$ đồng vị nên ${EF}\parallel{CK}.$
			\end{enumerate}
		}{\begin{tikzpicture}[scale=1]
		\tkzInit[xmin=-3.5,xmax=3,ymin=-3,ymax=3]
		\tkzDefPoint[label=B](110:2){B}
		\tkzDefPoint[label=right:C](30:2){C}
		\tkzDefPoint[label=right:D](0:2){D}
		\tkzDefPoint[label=below:O](0,0){O}
		\tkzDefPoint[label=left:A](-2,0){A}
		\tkzInterLL(A,C)(B,D)\tkzGetPoint{E}
		\tkzDrawCircle(O,A)
		\tkzLabelPoints[above](E) 
		\tkzDefPointBy[projection= onto A--D](E) \tkzGetPoint{F}  
		\tkzDefPointBy[reflection= over A--D](C) \tkzGetPoint{K}  
		\tkzDefMidPoint(B,C)    \tkzGetPoint{G}
		\tkzLabelPoints[below](F,K) 
		\tkzLabelPoints[above](G)
		\tkzMarkRightAngle[fill=green!60!cyan,size=0.2,opacity=1](B,G,O)
		\tkzMarkRightAngle[fill=green!60!cyan,size=0.2,opacity=1](E,F,D)
		\tkzDrawSegments(A,B B,C C,D D,A A,C B,D O,C O,B E,F C,K B,K O,G)
		\tkzDrawPoints[fill=red](O,A,B,C,D,E,F,K,G) 
		\end{tikzpicture}}
	\begin{enumerate}
		\item[4.] Ta có $\widehat{EFB}=60^\circ$ suy ra góc $\widehat{EAB}=60^\circ$ suy ra sđ$ \stackrel\frown{AB}=120^\circ$ suy ra $BC=R\sqrt{3}.$\\
		Kẻ $OQ$ vuông góc với $BC$ suy ra $BQ=\dfrac{R\sqrt{3}}{2}$ và góc $\widehat{BOQ}=60^\circ$.\\
		nên $OQ=BQ.\cot60^\circ=\dfrac{R\sqrt{3}}{2}.\dfrac{1}{\sqrt{3}}=\dfrac{R}{2}$.\\
		Diện tích tam giác $BOC$ là:
		$S_{BOC}=\dfrac{1}{2}BC.OQ=\dfrac{R^2\sqrt{3}}{4}$.\\
		Diện tích hình quạt $BOC$ là:
		$S_{qBOC}=\dfrac{\pi.R^2.120}{360}=\dfrac{2\pi.R^2}{3}$.\\
		Vậy diện tích hình viên phân giới hạn bởi dây $BC$ và cung nhỏ $BC$ là: $$S_{vp}=S_{qBOC}-S_{BOC}= \dfrac{2\pi.R^2}{3}-\dfrac{R^2\sqrt{3}}{4}.$$
	\end{enumerate}
}
\end{ex}
