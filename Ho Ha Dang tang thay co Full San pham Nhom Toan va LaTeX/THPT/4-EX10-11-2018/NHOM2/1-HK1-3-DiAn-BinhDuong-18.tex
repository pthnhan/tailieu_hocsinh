\begin{ex}%[Thi học kỳ 1, Dĩ An, Bình Dương, 2018]%[1D2K4]
	Cho $\mathrm{P}(A)=\dfrac{1}{3}$, $\mathrm{P}(A\cup B)=\dfrac{1}{2}$; $A$, $B$ là hai biến cố độc lập. Khi đó $\mathrm{P}(B)$ bằng   
	\choice
	{\True $\dfrac{1}{4}$}
	{$\dfrac{3}{4}$}
	{$\dfrac{1}{8}$}
	{$\dfrac{1}{6}$}
	\loigiai{
		Với $A$, $B$ là hai biến bất kỳ, ta có:\\
		$\mathrm{P}(A\cup B)=\mathrm{P}(A)+\mathrm{P}(B)-\mathrm{P}(A\cdot B)$. \\
		Mặt khác vì $A$ và $B$ là biến cố độc lập nên $\mathrm{P}(A.B)=\mathrm{P}(A)\cdot \mathrm{P}(B)$\\
		$\Rightarrow \mathrm{P}(A\cup B)=\mathrm{P}(A)+\mathrm{P}(B)-\mathrm{P}(A)\cdot \mathrm{P}(B)\Rightarrow  \mathrm{P}(B)=\dfrac{\mathrm{P}(A\cup B)-\mathrm{P}(A)}{1-\mathrm{P}(A)}=\dfrac{1}{4}$.
	}
\end{ex}
