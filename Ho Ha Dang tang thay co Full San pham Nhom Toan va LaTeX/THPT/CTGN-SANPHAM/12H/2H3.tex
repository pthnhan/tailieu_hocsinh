\begin{itemize}

		\item [+] $\left\{\begin{aligned}&\overrightarrow{a} \pm \overrightarrow{b}=(a_1 \pm b_1;a_2 \pm b_2;a_3 \pm b_3)\\&k\cdot\overrightarrow{a}=(k\cdot a_1;k\cdot a_2;k\cdot a_3) \end{aligned} \right.$
		\qquad\qquad + $\overrightarrow{a}=\overrightarrow{b} \Leftrightarrow \left\{\begin{aligned}&a_1=b_1\\&a_2=b_2\\&a_3=b_3 \end{aligned} \right.$
	\end{itemize}\vspace{-0.5cm}
	\begin{itemize}
		\item [+] $\overrightarrow{a}$ cùng phương với $\overrightarrow{b} \Leftrightarrow$ tồn tại số thực $t$ sao cho $\overrightarrow{a}=t\overrightarrow{b}$ (giả sử $\overrightarrow{b} \neq \overrightarrow{0}$)
		\item [+] Đặc biệt: nếu $b_1b_2b_3 \neq 0$ thì $\overrightarrow{a}$ cùng phương với $\overrightarrow{b} \Leftrightarrow \dfrac{a_1}{b_1}=\dfrac{a_2}{b_2}=\dfrac{a_3}{b_3}$
	\end{itemize} \vspace{-0.3cm}
	\item[*] \underline{Lưu ý}: Ba điểm $A,\,B,\,C$ thẳng hàng $\Leftrightarrow \overrightarrow{AB}$ và $\overrightarrow{BC}$ cùng phương $\Leftrightarrow \overrightarrow{AB}=k\cdot \overrightarrow{BC}$.
\end{itemize}
\subsubsection{Tích vô hướng của hai véc-tơ (\textit{kết quả của phép tính là một con số})}
\hspace{0.5cm} Với $\overrightarrow{a}=(a_1;a_2;a_3)$ và $\overrightarrow{b}=(b_1;b_2;b_3)$ thì $\overrightarrow{a}\cdot \overrightarrow{b}=\left|\overrightarrow{a}\right|\cdot \left|\overrightarrow{b}\right|\cdot \cos\left(\overrightarrow{a},\overrightarrow{b} \right)=a_1b_1+a_2b_2+a_3b_3$.
\begin{itemize}
	\begin{multicols}{2}
		\item $\overrightarrow{a}^2=|\overrightarrow{a}|^2$ và $|\overrightarrow{a}|=\sqrt{a_1^2+a_2^2+a_3^2}$
		\item $\cos \left(\overrightarrow{a},\overrightarrow{b} \right)=\dfrac{\overrightarrow{a}\cdot \overrightarrow{b}}{|\overrightarrow{a}|\cdot |\overrightarrow{b}|}$, với $\left\{\begin{aligned}&\overrightarrow{a} \neq \overrightarrow{0}\\&\overrightarrow{b} \neq \overrightarrow{0}\end{aligned}\right.$
		\item $AB=\sqrt{(x_B-x_A)^2+(y_B-y_A)^2+(z_B-z_A)^2}$
		\item $\overrightarrow{a} \perp \overrightarrow{b} \Leftrightarrow \overrightarrow{a}\cdot \overrightarrow{b}=0$, với $\left\{\begin{aligned}&\overrightarrow{a} \neq \overrightarrow{0}\\&\overrightarrow{b} \neq \overrightarrow{0}\end{aligned}\right.$
	\end{multicols}
\end{itemize}

\begin{dang}{\bf Áp dụng công thức toạ độ điểm}
	Dưới đây là các ví dụ áp dụng công thức toạ độ điểm
\end{dang}
\begin{vd}
	Trong không gian $Oxyz$, cho các điểm $A,B,C$ xác định bởi các hệ thức sau đây:\\
	$$\overrightarrow{OA}=2\overrightarrow{i}+2\overrightarrow{j}-\overrightarrow{k};\;\overrightarrow{OB}=3\overrightarrow{j}-2\overrightarrow{i} \text{ và } \overrightarrow{AC}=3\overrightarrow{i}-\overrightarrow{k}$$
	Xác định toạ độ của các điểm $A,B,C$.
	\loigiai{
	\begin{enumerate}
		\item Ta có $\overrightarrow{OA}=2\overrightarrow{i}+2\overrightarrow{j}-\overrightarrow{k} \Leftrightarrow A(2;2;-1)$.
		\item Ta có $\overrightarrow{OB}=3\overrightarrow{j}-2\overrightarrow{i}\Leftrightarrow \overrightarrow{OB}=-2\overrightarrow{i}+3\overrightarrow{j}+0\overrightarrow{k} \Leftrightarrow B(-2;3;0)$.
		\item Ta có $\overrightarrow{OC}=\overrightarrow{OA}+\overrightarrow{AC}=5\overrightarrow{i}+2\overrightarrow{j}-2\overrightarrow{k}\Leftrightarrow C(5;2;-2)$.
\end{enumerate}}
\end{vd}
\begin{vd} Tìm toạ độ các điểm được cho dưới đây:
	\begin{enumerate}
		\item Tìm toạ độ điểm $H$ là hình chiếu vuông góc của điểm $A(1;3;-2)$ lên trục tung.
		\item Tìm toạ độ điểm $K$ là hình chiếu vuông góc của điểm $B(2;-1;-5)$ lên trục cao.
		\item Tìm toạ độ điểm $T$ là hình chiếu của điểm $C(0;1;4)$ lên mặt phẳng toạ độ $(Oxz)$.
		\item Tìm toạ độ điểm $M$ là hình chiếu của điểm $E(-4;1;-1)$ lên mặt phẳng toạ độ $(Oyz)$.
		\item Tìm toạ độ điểm $A'$ là điểm đối xứng với điểm $A(1;3;-2)$ qua trục tung.
		\item Tìm toạ độ điểm $E'$ là điểm đối xứng với điểm $E(-4;1;-1)$ qua mặt phẳng $(Oxy)$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Hình chiếu vuông góc của điểm $A(1;3;-2)$ lên trục $Oy$ là điểm $H(0;3;0)$ (\textit{giữ $y_A$} \textit{lại}).
		\item Hình chiếu vuông góc của điểm $B(2;-1;-5)$ lên trục $Oz$ là điểm $K(0;0;-5)$ (\textit{giữ $z_B$} \textit{lại}).
		\item Hình chiếu của điểm $C(0;1;4)$ lên $(Oxz)$ là điểm $T(0;0;4)$ (\textit{giữ} $x_C,z_C$ \textit{lại})
		\item Hình chiếu của điểm $E(-4;1;-1)$ lên $(Oyz)$ là điểm $M(0;1;-1)$ (\textit{giữ } $y_E,z_E$ \textit{lại })
		\item Để tìm điểm đối xứng $A'$ của $A(1;3;-2)$ qua $Oy$:
			\begin{itemize}
				\item Trước tiên ta tìm hình chiếu vuông góc $H$ của $A$ lên $Oy$, đáp số $H(0;3;-2)$.
				\item Sau đó cho $H$ là trung điểm của $AA'\Rightarrow \left\{\begin{aligned}
				&x_{A'}=2x_H-x_A=-1\\
				&y_{A'}=2y_H-y_A=3\\
				&z_{A'}=2z_H-z_A=2
				\end{aligned}\right. \Rightarrow A'(-1;3;2)$.
			\end{itemize}
		\item Để tìm điểm đối xứng $E'$ của $E(-4;1;-1)$ qua $(Oxy)$:
		\begin{itemize}
			\item Trước tiên ta tìm hình chiếu vuông góc $N$ của $E$ lên $(Oxy)$, đáp số $N(-4;1;0)$.
			\item Sau đó cho $H$ là trung điểm của $AA'\Rightarrow \left\{\begin{aligned}
			&x_{E'}=2x_N-x_E=-4\\
			&y_{E'}=2y_N-y_E=1\\
			&z_{E'}=2z_N-z_E=1
			\end{aligned}\right. \Rightarrow E'(-4;1;1)$.
		\end{itemize}
	\end{enumerate}}
\end{vd}

\subsection{Các dạng toán và công thức giải nhanh}
\begin{dang}{Viết phương trình mặt phẳng bằng cách giải hệ phương trình}
	Giả sử mặt phẳng $(\alpha)$ có véc-tơ pháp tuyến $\vec{n}_\alpha=(A;B;C)$, phương trình mặt phẳng $(\alpha)$ có dạng $Ax+By+Cz+D=0$, ta có một số nhận xét sau:
	\begin{itemize}
		\item Mặt phẳng $(\alpha)$ qua điểm $M(x_M,y_M,z_M)$ suy ra $$Ax_M+By_M+Cz_M+D=0.$$
		\item Mặt phẳng $(\alpha)$ song song hoặc chứa đường thẳng $(d)$ có véc-tơ chỉ phương $\vec{a}=(a_1,b_1,c_1)$ suy ra $$\vec{n}_\alpha\perp \vec{a}\Rightarrow A\cdot a_1+B\cdot b_1+C\cdot c_1=0.$$
		\item Mặt phẳng $(\alpha)$ vuông góc với mặt phẳng $(\beta)$ có véc-tơ pháp tuyến $\vec{n}_\beta=(A',B',C')$ suy ra $$\vec{n}_\alpha\perp \vec{n}_\beta\Rightarrow A\cdot A'+B\cdot B'+C\cdot C'=0.$$
	\end{itemize}
	Khi đó, với đề bài có các yếu tố trên, ta có thể lập hệ $3$ phương trình $3$ ẩn để giải tìm các hệ số $A$, $B$, $C$, $D$. Lưu ý, ta cần xét hai trường hợp hệ số tự do $D=0$ khi đó mặt phẳng có dạng $$Ax+By+Cz=0$$ và trường hợp $D\ne 0$ khi đó mặt phẳng có dạng $$\dfrac{A}{D}x+\dfrac{B}{D}y+\dfrac{C}{D}z+1=0.$$
\end{dang}

\begin{dang}{}
	{Mặt phẳng $ (P) $ qua $ M $ cắt các trục $ Ox $, $ Oy $, $ Oz $ tại $ A(a;0;0) $, $ B(0;b;0) $, $ C(0;0;c) $. Tìm tọa độ trực tâm tam giác $ ABC $}
	
	\textbf{Phương pháp:} Mặt phẳng $ (P)\colon \dfrac{x}{a}+\dfrac{y}{b}+\dfrac{z}{c}=1$.
	
	\textbf{Chứng minh.} Xét tứ diện vuông $ O.ABC $, ta có tính chất
	\begin{equation}\label{1.2}
	M \text{ là trực tâm tam giác }ABC \Leftrightarrow OM\bot(ABC).	
	\end{equation}
	Theo tính chất (\ref{1.2}), $ H $ là hình chiếu của $ O $ lên mặt phẳng $ (ABC) $. Ta có $ H\left(\dfrac{1}{a}t;\dfrac{1}{b}t;\dfrac{1}{c}t\right) \in (P)$ nên
	\[t=\dfrac{1}{\dfrac{1}{a^2}+\dfrac{1}{b^2}+\dfrac{1}{c^2}}=\dfrac{a^2b^2c^2}{a^2b^2+b^2c^2+c^2a^2}.\]
	Vậy $ H\left(\dfrac{ab^2c^2}{a^2b^2+b^2c^2+c^2a^2};\dfrac{a^2bc^2}{a^2b^2+b^2c^2+c^2a^2};\dfrac{a^2b^2c}{a^2b^2+b^2c^2+c^2a^2}\right) $.\\
	\textbf{Đặt  biệt:} 
	\begin{itemize}
		\item Phương trình đường thẳng $\Delta$ đi qua trực tâm $H$ của $\Delta ABC$ và vuông góc với mặt phẳng $(ABC)$ có dạng $ \Delta\colon ax=by=cz $.
		\item Độ dài  $ OH =\mathrm{d}(O;(ABC))= \dfrac{|abc|}{\sqrt{a^2b^2+b^2c^2+c^2a^2}} $.
	\end{itemize}
	
\end{dang}

		Do đó $\overrightarrow{OH}=(2;1;1)$ là một véc-tơ pháp tuyến của $(ABC)$ và vì $H\in (ABC)$ nên phương trình $(ABC)$ là

\begin{vd}%[2H3K2-8]
	Trong không gian với hệ tọa độ $Oxyz$, phương trình mặt phẳng $(P)$ đi qua điểm $M(1;2;3)$ và cắt các tia $Ox$, $Oy$, $Oz$ lần lượt tại các điểm $A$, $B$, $C$ sao cho $T=\dfrac{1}{OA^2}+\dfrac{1}{OB^2}+\dfrac{1}{OC^2}$ đạt giá trị nhỏ nhất là
	\choice
	{\True $x+2y+3z-14=0$}
	{$3x+2y+z-10=0$}
	{$6x+3y+2z-18=0$}
	{$6x-3y+2z-6=0$}
	\loigiai{
		\textit{Cách 1.}
		Giả sử $A(a;0;0)$, $B(0;b;0)$, $C(0;0;c)$ với $a,b,c\in\mathbb{R^*}$ do $OA$, $OB$, $OC$ khác 0.\\
		Khi đó phương trình mặt phẳng $(P)$ qua $A$, $B$, $C$ có phương trình là $\dfrac{x}{a}+\dfrac{y}{b}+\dfrac{z}{c}=1$.\\
		Mà $M\in(P)$ nên $\dfrac{1}{a}+\dfrac{2}{b}+\dfrac{3}{c}=1$, do đó theo bất đẳng thức Bunhiacopski ta có:
		$$T=\dfrac{1}{a^2}+\dfrac{1}{b^2}+\dfrac{1}{c^2}=\dfrac{1}{14}(1^2+2^2+3^2)\left(\dfrac{1}{a^2}+\dfrac{1}{b^2}+\dfrac{1}{c^2}\right)\ge \dfrac{1}{14}\left(\dfrac{1}{a}+\dfrac{2}{b}+\dfrac{3}{c}\right)^2=\dfrac{1}{14}.$$
		$T$ đạt giá trị nhỏ nhất nên ta có dấu bằng xảy ra, tức là $$\heva{&x=2y=3z\\ &\dfrac{1}{a}+\dfrac{2}{b}+\dfrac{3}{c}=1}\Leftrightarrow \heva{&a=14\\ &b=\dfrac{14}{2}\\ &c=\dfrac{14}{3}.}$$
		Vậy phương trình mặt phẳng $(P)$ là $x+2y+3z-14=0$.\\
		\textit{Cách 2.}		
		Gọi $ H $ là trực tâm tam giác $ ABC $. Ta có 
		\[T=\dfrac{1}{OA^2}+\dfrac{1}{OB^2}+\dfrac{1}{OC^2}=\dfrac{1}{OH^2}\ge\dfrac{1}{OM^2}.\]
		Vậy $ T $ nhỏ nhất $ \Leftrightarrow M\equiv H $. Vậy mặt phẳng $ P $ qua $ M(1;2;3) $ là trực tâm tam giác $ ABC $.\\
		Vậy $ (P)\colon x+2y+3z-14=0 $.
	}
\end{vd}

\begin{dang}{}
	{Cho điểm $ M(x_M;y_M;z_M) $ không thuộc các trục và các mặt phẳng tọa độ. Mặt phẳng $ (P) $ qua $ M $ cắt các tia $ Ox $, $ Oy $, $ Oz $ tại $ A $, $ B $, $ C $. Viết phương trình mặt phẳng $ (P) $ sao cho thể tích tứ diện $ OABC $ nhỏ nhất. Tính giá trị đó}
	
	\textbf{Phương pháp:} 
	\begin{itemize}
		\item Mặt phẳng $ (P)\colon \dfrac{x}{3x_M}+\dfrac{y}{3y_M}+\dfrac{z}{3z_M}=1$.
		\item $ \min V_{OABC}=\dfrac{9}{2}\cdot x_M\cdot y_M\cdot z_M$.
	\end{itemize}
	
	\textbf{Chứng minh.} Gọi $ A(a;0;0) $, $ B(0;b;0) $, $ C(0;0;c) $ với $ a,b,c>0 $. Ta có 
	\[(P)\colon\dfrac{x}{a}+\dfrac{y}{b}+\dfrac{z}{c}=1.\]
	Khi đó $ OA=a $, $ OB=b $, $ OC=c $ suy ra 
	\[V_{OABC}=\dfrac{1}{6}OA\cdot OB\cdot OC=\dfrac{1}{6}abc.\]
	Do $ M\in (P) $ suy ra 
	\begin{equation}\label{1.1}
	\dfrac{x_M}{a}+\dfrac{y_M}{b}+\dfrac{z_M}{c} = 1
	\end{equation}
	Áp dụng bất đẳng thức Cauchy cho (\ref{1.1}),  ta có
	\[1=\dfrac{x_M}{a}+\dfrac{y_M}{b}+\dfrac{z_M}{c}\ge3\sqrt[3]{\dfrac{x_M\cdot y_M\cdot z_M}{abc}}\]
	suy ra \[\dfrac{1}{27}\ge \dfrac{x_M\cdot y_M\cdot z_M}{6V}\Leftrightarrow V_{OABC}\ge\dfrac{9}{2}\cdot x_M\cdot y_M\cdot z_M.\]
	Dấu ``$ = $'' xảy ra khi và chỉ khi $ \dfrac{x_M}{a}=\dfrac{y_M}{b}=\dfrac{z_M}{c} = \dfrac{1}{3} \Leftrightarrow\heva{&a=3x_M\\&b=3y_M\\&c=3z_M.}$\\
	Vậy $ (P)\colon \dfrac{x}{3x_M}+\dfrac{y}{3y_M}+\dfrac{z}{3z_M}=1$ và $ \min V_{OABC}=\dfrac{9}{2}\cdot x_M\cdot y_M\cdot z_M$.
\end{dang}

\begin{vd}%[2H3K2-8]
	Trong không gian với hệ tọa đô $Oxyz$, cho điểm $M(1;2;4)$. Gọi $(P)$ là mặt phẳng đi qua $M$ và cắt các tia $Ox,Oy,Oz$ lần lượt tại các điểm $A,B,C$ sao cho  thể tích tứ diện $OABC$ nhỏ nhất. $(P)$ đi qua điểm nào dưới đây?
	\choice
	{$(0;1;3)$}
	{\True $(2;2;0)$}
	{$(1;1;2)$}
	{$(-1;1;4)$}
	\loigiai{
		\textbf{Cách 1:}\\
		Mặt phẳng $ (P)\colon \dfrac{x}{3x_M}+\dfrac{y}{3y_M}+\dfrac{z}{3z_M}=1\Leftrightarrow \dfrac{x}{3}+\dfrac{y}{6}+\dfrac{z}{12}=1 \Leftrightarrow 4x+2y+z-12=0$.\\
		Mặt phẳng $(P)$ đi qua điểm $(2;2;0)$.\\
		\textbf{Cách 2:}\\
		Gọi $A(a;0;0),B(0;b;0),C(0;0;c)$, với $a,b,c>0$. Phương trình mặt phẳng $(P)$ là $\dfrac{x}{a}+\dfrac{y}{b}+\dfrac{z}{c}=1.$\\
		Ta có $M \in (P)$ nên $\dfrac{1}{a}+\dfrac{2}{b}+\dfrac{4}{c}=1$. Áp dụng bất đẳng thức Cosi ta có
		\begin{center}
			$\dfrac{1}{a}+\dfrac{2}{b}+\dfrac{4}{c} \geq 3 \sqrt[3]{\dfrac{1}{a} \cdot \dfrac{2}{b} \cdot \dfrac{4}{c}} \Leftrightarrow 1 \geq 27 \cdot \dfrac{8}{abc} \Leftrightarrow abc \geq 8 \cdot 27.$
		\end{center}
		Mặt khác thể tích khối tứ diện $OABC$ là $V_{OABC}=\dfrac{1}{6}abc \geq \dfrac{1}{6} \cdot 8 \cdot 27=36.$\\

		Vậy phương trình của $(P)$ là $\dfrac{x}{3}+\dfrac{y}{6}+\dfrac{z}{12}=1 \Leftrightarrow 4x+2y+z-12=0$.\\
		Mặt phẳng $(P)$ đi qua điểm $(2;2;0)$.
	}

\begin{dang}{}
	{Cho điểm $ A $ và đường thẳng $ \Delta $ không vuông góc với $ (P) $. Viết phương trình đường thẳng $ d $  qua $ A\in(P) $ sao cho $ d $ nằm trong (hoặc song song) $ (P) $ và tạo với $ \Delta $ một góc nhỏ nhất.}
	
	\textbf{Phương pháp:} 
	$ d $ đi qua $ A $ và có VTCP $ \overrightarrow{u}_d=\left[\left[\overrightarrow{u}_\Delta,\overrightarrow{n}_P\right],\overrightarrow{n}_P\right] $.
	
	\textbf{Chứng minh.}
	Gọi $ \Delta' $ là hình chiếu vuông góc của $ \Delta $ lên mặt phẳng $ (P) $. Ta có $ \Delta' $ có VTCP $\left[\left[\overrightarrow{u}_\Delta,\overrightarrow{n}_P\right],\overrightarrow{n}_P\right] $.\\
	Ta có với mọi $ d\subset(P) $, $ \left(\widehat{d,\Delta}\right)\ge \left(\widehat{\Delta',\Delta}\right) = \left(\widehat{\Delta,(P)}\right)=\alpha$.\\
	Do đó $ \min\left(\widehat{d,\Delta}\right)=\alpha\Leftrightarrow \hoac{&d\equiv\Delta'\\&d\parallel \Delta'} $.
	Vậy $ d $ đi qua $ A $ và có VTCP $ \overrightarrow{u}_d=\left[\left[\overrightarrow{u}_\Delta,\overrightarrow{n}_P\right],\overrightarrow{n}_P\right] $.
\end{dang}

\begin{ex}%[2H3G3-8]
	Trong không gian $Oxyz$, cho đường thẳng $d\colon \dfrac{x+2}{4}=\dfrac{y-1}{-4}=\dfrac{z+2}{3}$ và mặt phẳng $(P)\colon 2x-y+2z+1=0$. Đường thẳng $\Delta $ đi qua $E(-2;1;-2)$ song song với $(P)$ đồng thời tạo với $d$ góc bé nhất. Biết rằng $ \Delta $ có một véc-tơ chỉ phương $\vec {u}=(m;n;1)$. Tính $T=m^2-n^2$.
	\choice
	{$T=-5$}
	{$T=4$}
	{$T=3$}
	{\True $T=-4$}
	\loigiai{
		\textit{Cách 1.}
		Ta có: $ \Delta \parallel (P)$ nên $\vec {u}_{\Delta}\perp \vec {n}_{(P)} \Leftrightarrow \vec {u}_{\Delta}\cdot \vec {n}_{(P)}=0 \Leftrightarrow n=2m+2 \Rightarrow \vec {u}_{\Delta}=(m;2m+2;1)$\\
		Do đó, gọi $\alpha$ góc giữa hai đường thẳng $\Delta $ và $d$, ta có $$\cos \alpha=\dfrac{\left|\vec {u}_{\Delta}\cdot \vec {u}_{(d)}\right|}{\left|\vec {u}_{\Delta}\right|\cdot \left|\vec {u}_{(d)}\right|}=\dfrac{\left|4m+5 \right|}{\sqrt{41(5m^2+8m+5)}}=\dfrac{1}{\sqrt{41}}\cdot \sqrt{\dfrac{16m^2+40m+25}{5m^2+8m+5}}.$$
		Góc $\alpha$ nhỏ nhất khi và chỉ khi $\cos \alpha$ đạt giá trị lớn nhất.\\

		$$f'(m)=\dfrac{-72m^2-90m}{\left(5m^2+8m+5\right)^2}=0\Leftrightarrow \hoac{&m=0\\&m=-\dfrac{5}{4}.}$$

	}
