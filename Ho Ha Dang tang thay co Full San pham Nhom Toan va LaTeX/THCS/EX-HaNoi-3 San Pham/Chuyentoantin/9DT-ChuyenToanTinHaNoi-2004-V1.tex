\begin{ex}[Đề thi Chuyên Hà Nội năm 2004 - 2005, Vòng 1][9D1K8]
Cho biểu thức $$P = \left(\dfrac{\sqrt{x} - 1}{\sqrt{x} +  1} - \dfrac{\sqrt{x} + 1}{\sqrt{x} - 1}\right)\cdot \left(\dfrac{1}{2\sqrt{x}} - \dfrac{\sqrt{x}}{2}\right)^2.$$
\begin{enumerate}	
	\item Rút gọn $P$.
	\item Tìm $x$ để $\dfrac{P}{\sqrt{x}} > 2$.
\end{enumerate}
\loigiai{
	\begin{enumerate}[a)]
		\item Điều kiện để $P$ có nghĩa là $x > 0, x \ne 1$.
		\begin{align*}
			P&=\left(\dfrac{\sqrt{x} - 1}{\sqrt{x} +  1} - \dfrac{\sqrt{x} + 1}{\sqrt{x} - 1}\right)\cdot \left(\dfrac{1}{2\sqrt{x}} - \dfrac{\sqrt{x}}{2}\right)^2\\
			&= \left[\dfrac{\left(\sqrt{x}-1\right)^2-\left(\sqrt{x}+1\right)^2}{x-1}\right]\cdot\dfrac{\left(1-x\right)^2}{4x}\\
            &= \dfrac{4\sqrt{x}}{1-x}\cdot\dfrac{\left(1-x\right)^2}{4x}\\
			&= \dfrac{1-x}{\sqrt{x}}.
		\end{align*}
		\item $\dfrac{P }{\sqrt{x}}>2\Rightarrow \dfrac{1-x}{\left(\sqrt{x}\right)^2}>2 \Leftrightarrow 1-3x>0\Leftrightarrow x<\dfrac{1}{3}$. Kết hợp với $x>0$ và $x\ne 1$ ta suy ra $0<x<\dfrac{1}{3}$.
	\end{enumerate}
}
\end{ex}

\begin{ex}[Đề thi Chuyên Hà Nội năm 2004 - 2005, Vòng 1][9D4K4]
Cho phương trình $x^2 - \left(m - 2\right)x -  m^2 + 3m - 4 = 0$ (với $m$ là tham số).
\begin{enumerate}	
	\item Chứng minh phương trình có hai nghiệm phân biệt với mọi $m$.
	\item Tìm $m$ để tỉ số giữa hai nghiệm  của phương trình có giá trị tuyệt đối bằng  $2$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a)]
	\item Phương trình bậc hai có $ac=1.(-m+3m-4)=-\left(m-\dfrac{3}{4}\right)^2-\dfrac{7}{4}<0$ với mọi $m$, do đó phương trình có hai nghiệm phân biệt trái dấu.
	\item Phương trình này có hai nghiệm $x_1, x_2$ trái dấu nên $x_1=-2x_2$ hoặc $x_2=-2x_1$, do đó ta có 
$\heva{
&(x_1+2x_2)(x_2+2x_1)=0\\ 
\Leftrightarrow&x_1x_2+2(x_1+x_2)^2=0\qquad (*)
}$
Theo hệ thức Vi-ét ta có $\heva{
x_1+x_2=m-2\\
x_1x_2=-m^2+3m-4}.$\\
Thay vào $(*)$ ta được: $m^2-5m+4$. Phương trình này có nghiệm $m\in\left\{-4;1\right\}$.
\end{enumerate}	
}
\end{ex}

\begin{ex}[Đề thi Chuyên Hà Nội năm 2004 - 2005, Vòng 1][9D4K6]
Trên mặt phẳng tọa độ cho đường thẳng $(d)$ có phương trình  $2kx + \left(k - 1\right)y = 2$ ($k$ là tham số).
	\begin{enumerate}	
		\item Với giá trị nào của $k$ thì đường thẳng $(d)$ song song với đường thẳng $y = x\sqrt{3}$. Khi đó hãy tính góc tạo bởi đường$(d)$ và tia $Ox$.
		\item Tìm $k$ để khoảng cách từ gốc tọa độ đến đường thẳng $(d)$ là lớn nhất. 
	\end{enumerate}
\loigiai{
	\begin{enumerate}[a)]
	\item Ta xét các trường hợp sau
		\begin{itemize}
			\item Với $k=1$, phương trình đường thẳng $(d)$ là $x=1$, $(d)$ không song song với đường thẳng $y=\sqrt{3}x$.
			\item Với $k\ne 1$, phương trình đường thẳng $(d)$ trở thành $$y=-\dfrac{2k}{k-1}\cdot x+\dfrac{2}{k-1}\qquad\quad (*)$$
           Đường thẳng $(d)$ song song với đường thẳng $y=\sqrt{3}x$ khi và chỉ khi $\dfrac{-2k}{k-1}=\sqrt{3}$ hay         $k=\sqrt{3}\left(2-\sqrt{3}\right)$. Khi đó góc nhọn $\alpha$ tạo bởi $(d)$ với tia $Ox$ có $\tan\alpha=\sqrt{3}$ nên $\alpha=60^\circ$.
		\end{itemize}
	\item Ta xét các trường hợp sau
		\begin{itemize}
			\item Với $k=1$ thì khoảng cách từ gốc tọa độ $O$ đến $(d)$ bằng $1$.
			\item Với $k=0$ thì khoảng cách từ gốc tọa độ $O$ đến $(d)$ bằng $2$.
			\item Với $k\ne 0$ và $k\ne 1$, gọi giao điểm của $(d)$ với hai trục tọa độ $Ox$, $Oy$ là $A$ và $B$.\\
            Do $A\in Ox$ nên $y=0$, khi đó từ $(*)$ ta có $x_A=\dfrac{1}{k}$, suy ra $OA=\left|\dfrac{1}{k}\right|=\dfrac{1}{|k|}$.\\
            Do $B\in Oy$ nên $x=0$, khi đó từ $(*)$ ta có $y_B=\dfrac{2}{k-1}$, suy ra $OB=\left|\dfrac{2}{k-1}\right|=\dfrac{2}{|k-1|}$.\\
            Gọi $H$ là hình chiếu của $O$ trên $AB$. Trong tam giác vuông $OAB$, ta có 
$$\begin{aligned}
&\dfrac{1}{OH^2}=\dfrac{1}{OA^2}+\dfrac{1}{OB^2}\\ 
\Rightarrow&\dfrac{1}{OH^2}=\dfrac{1}{\left(\dfrac{1}{k}\right)^2}+\dfrac{1}{\left(\dfrac{2}{k-1}\right)^2}\\
\Rightarrow&\dfrac{1}{OH^2}=k^2+\dfrac{(k-1)^2}{4}\\
\Rightarrow&OH=\dfrac{2}{\sqrt{5k^2-2k+1}}.
\end{aligned}$$
Vì $5k^2-2k+1=5\left(k-\dfrac{1}{5}\right)^2+\dfrac{4}{5}\ge \dfrac{4}{5}$ với mọi $k$ nên $OH\le \sqrt{5}$, $OH=\sqrt{5}$ khi $k=\dfrac{1}{5}$.\\
Vậy khi $k=\dfrac{1}{5}$ thì khoảng cách từ gốc tọa độ $O$ đến $(d)$ là lớn nhất.
		\end{itemize}
\end{enumerate}	
}
\end{ex}
