\subsection{Đề số 1a}
\begin{bt}%[Phan Quốc Trí]%[2D3B1]
Tính
\begin{multicols}{2}
\begin{enumerate}[a.]
\item $\displaystyle \int \left( \sqrt{1+x}+\sqrt{1-x}+\sqrt{e^x} \right) \mathrm{\,d}x$;
\item $\displaystyle \int \dfrac{x \mathrm{\,d}x}{\sqrt{2x+1}}$;
\item $\displaystyle \int (2-x)e^{\frac{x}{2}} \mathrm{\,d}x$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}[a.]
\item $
\displaystyle \int \left( \sqrt{1+x}+\sqrt{1-x}+\sqrt{e^x} \right) \mathrm{\,d}x = \dfrac{2}{3} \sqrt{(1+x)^3} -\dfrac{2}{3} \sqrt{(1-x)^3} +2\sqrt{e^x}+C$.
\item 
Đặt $u=\sqrt{2x+1}\Rightarrow x =\dfrac{u^2-1}{2} \Rightarrow \mathrm{\,d}x = u \mathrm{\,d}u $\\
$\displaystyle \int \dfrac{x \mathrm{\,d}x}{\sqrt{2x+1}}=\displaystyle \int \dfrac{u^2-1}{2}\mathrm{\,d}u=\dfrac{1}{2} \left( \dfrac{1}{3} u^3-u \right)+C=\dfrac{1}{2} \left( \dfrac{1}{3} \sqrt{(2x+1)^3}-\sqrt{2x+1} \right)+C $.
\item 
Đặt $\heva{&u=2-x\\& \mathrm{\,d}v=e^{\dfrac{x}{2}} \mathrm{\,d}x} \Rightarrow \heva{&\mathrm{\,d}u=-\mathrm{\,d}x\\&v=2e^{\dfrac{x}{2}} }$. Khi đó \\
$\displaystyle \int (2-x)e^{\dfrac{x}{2}} \mathrm{\,d}x=2(2-x)e^{\dfrac{x}{2}}+2\displaystyle \int e^{\dfrac{x}{2}} \mathrm{\,d}x = 2(2-x)e^{\dfrac{x}{2}}+4e^{\dfrac{x}{2}}+C$.
\end{enumerate}
}
\end{bt}
\begin{bt}%[Phan Quốc Trí]%[2D3K2]
Tính
\begin{multicols}{2}
\begin{enumerate}[a.]
\item $I=\displaystyle \int \limits_1^2 \dfrac{\mathrm{\,d}x}{x^2+2x}$;
\item $I=\displaystyle \int \limits_1^e \dfrac{1+\ln^3 x}{x}\mathrm{\,d}x$;
\item $I=\displaystyle \int \limits_0^{\tfrac{\pi}{4}} \dfrac{\tan x}{\left( \sin x + \cos x \right)^2} \mathrm{\,d}x$;
\item $I=\displaystyle \int \limits_0^{\tfrac{\pi}{2}} x^2 \cos x \mathrm{\,d}x$.
\end{enumerate}
\end{multicols}
\loigiai{
\begin{enumerate}[a.]
\item   $\displaystyle \int \limits_1^2 \dfrac{\mathrm{\,d}x}{x^2+2x}=\dfrac{1}{2}\displaystyle \int \limits_1^2 \left( \dfrac{1}{x}-\dfrac{1}{x+2} \right)\mathrm{\,d}x=\left. \dfrac{1}{2} \left(\ln |x|-\ln|x+2|  \right) \right|_1^2= \dfrac{1}{2}\ln \dfrac{3}{2}$.
\item Đặt $u=\ln x \Rightarrow \mathrm{\,d}u = \dfrac{1}{x} \mathrm{\,d}x$. \\
Đổi cận: 
\begin{tabular}{l|ccc}

u & $0$ &               & $1$\\ 
\end{tabular}\\
$I=\displaystyle \int\limits_0^{1} (1+u^3) \mathrm{\,d}u= \left. \left( u + \dfrac{u^4}{4} \right) \right|_0^{1}=\dfrac{5}{4}$.
\item $I=\displaystyle \int \limits_0^{\tfrac{\pi}{4}} \dfrac{\tan x}{\left( \sin x + \cos x \right)^2} \mathrm{\,d}x= \displaystyle \int \limits_0^{\tfrac{\pi}{4}} \dfrac{\tan x}{\cos^2 x \left( \tan x +1 \right)^2}\mathrm{\,d}x$\\
Đặt $u=\tan x \Rightarrow \mathrm{\,d}u = \dfrac{1}{\cos^2 x }\mathrm{\,d}x$
Đổi cận: 
\begin{tabular}{l|ccc}
x & $0$ &\hspace{0.5cm} & $\dfrac{\pi}{4}$\\ \hline 
u & $0$ &               & $1$\\ 
\end{tabular} Khi đó \\
$I= \displaystyle \int _0 ^1 \dfrac{u \mathrm{\,d}u}{(u+1)^2}=\displaystyle \int\limits_0^1 \left( \dfrac{1}{u+1}-\dfrac{1}{(u+1)^2} \right) \mathrm{\,d}u = \left. \left( \ln \vert u+1 \vert - \dfrac{1}{u+1} \right) \right|_0^1 = \ln 2 +\dfrac{1}{2}$.
\item Đặt $\heva{&u=x^2\\& \mathrm{\,d}v=\cos x \mathrm{\,d}x} \Rightarrow \heva{& \mathrm{\,d}u = 2x \mathrm{\,d}x\\& v = \sin x \mathrm{\,d}x}$. Khi đó\\
$I=\displaystyle \int \limits_0^{\tfrac{\pi}{2}} x^2 \cos x \mathrm{\,d}x = \left. x^2\sin x \right|_0^{\tfrac{\pi}{2}} - \displaystyle \int \limits_0^{\tfrac{\pi}{2}} 2x \sin x \mathrm{\,d}x= \dfrac{\pi^2}{4}-\displaystyle \int \limits_0^{\tfrac{\pi}{2}} 2x \sin x \mathrm{\,d}x$\\
Lại đặt $\heva{&u=-2x\\& \mathrm{\,d}v=\sin x \mathrm{\,d}x} \Rightarrow \heva{&\mathrm{\,d}u=-2\mathrm{\,d}x\\&v=-\cos x}$. Khi đó\\
$I=\dfrac{\pi^2}{4} + 2x \cos x \Big|_0^{\tfrac{\pi}{2}} -\displaystyle \int \limits_0^{\tfrac{\pi}{2}} 2\cos x \mathrm{\,d}x=\dfrac{\pi^2}{4}-2\sin x \Big|_0^{\tfrac{\pi}{2}}=\dfrac{\pi^2}{4}-2$.
\end{enumerate}
}
\end{bt}
\begin{bt}%[Phan Quốc Trí]%[2D3B3]
\hfill
\begin{enumerate}[a.]
\item Tính diện tích hình phẳng giới hạn bởi đồ thị các hàm số $y=x^3$ và $y=-x^2$.
\item Tính thể tích vật thể được tạo thành khi quay hình phẳng $S$  giới hạn bởi đồ thị hàm số $y=\ln x, Ox$ và $x=2$ quanh trục $Ox$.
\end{enumerate}
\loigiai{
\begin{enumerate}[a.]
\item Phương trình hoành độ giao điểm của đồ thị các hàm số
$$x^3=-x^2\Leftrightarrow x^3+x^2=0\Leftrightarrow \hoac{&x=0\\&x=-1.}$$
Diện tích hình phẳng $S=\displaystyle \int \limits_{-1}^0 \vert x^3+x^2 \vert \mathrm{\,d}x=\displaystyle \int \limits_{-1}^0 (x^3+x^2) \mathrm{\,d}x = \left( \dfrac{x^4}{4}+\dfrac{x^3}{3} \right)\Big|_{-1}^0=\dfrac{1}{12}$.
\item Phương trình hoành độ giao điểm của đồ thị hàm số $y=\ln x$ và trục $Ox$.
$$\ln x =0 \Leftrightarrow x =1$$
Thể tích $V= \pi \displaystyle \int\limits_1^{2} \ln^2 x \mathrm{\,d}x$. \\ 
Đặt $\heva{&u=\ln^2 x \\&\mathrm{\,d}v=\mathrm{\,d}x}\Rightarrow \heva{&\mathrm{\,d}u=\dfrac{2}{x} \ln x \mathrm{\,d}x\\& v=x}$. Khi đó \\
$\displaystyle \int\limits_1^{2} \ln^2 x \mathrm{\,d}x=x\ln^2 x \Big|_1^2- \displaystyle \int\limits_1^2 2 \ln x \mathrm{\,d}x=2\ln^2 2 - \displaystyle \int\limits_1^2 2 \ln x \mathrm{\,d}x $\\
Đặt $\heva{&u=-2\ln x \\& \mathrm{\,d}v=\mathrm{\,d}x}\Rightarrow \heva{&\mathrm{\,d}u=\dfrac{-2}{x}\mathrm{\,d}x\\& v=x}$. Khi đó\\
$\displaystyle \int\limits_1^{2} \ln^2 x \mathrm{\,d}x= 2\ln^2 2 -2x \ln x \Big|_1^2 +\displaystyle \int\limits_1^2 2\mathrm{\,d}x=2\ln^2 2-4\ln 2+ 2x\Big|_1^2= 2\ln^2 2-4\ln 2+2=2 \left( \ln 2 -1 \right)^2$.\\
Vậy $V=2 \pi \left( \ln 2 -1 \right)^2$.
\end{enumerate}
}
\end{bt}
