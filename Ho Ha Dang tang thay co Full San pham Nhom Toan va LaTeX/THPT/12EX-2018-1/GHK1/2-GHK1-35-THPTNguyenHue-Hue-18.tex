	\choice
	{$m_1+m_2=6 $}

		\begin{enumerate}[Trường hợp 1.]
			\item Phương trình $x^2-mx+m=0$ có đúng 1 nghiệm (nghiệm này có thể bằng $1$ hoặc khác $1$), tức là $\Delta=m^2-4m=0$. Từ đó ta có $m=0$ hoặc $m=4$.

		\end{enumerate}
	Vậy có hai giá trị $m$ cần tìm là $m_1=0$ và $m_2=4$ nên tổng $m_1+m_2=4$.
	}

		Do hộp có $4$ mặt bên và $1$ mặt đáy nên tổng diện tích các mặt cần mạ vàng $S =a^2 +4ah = a^2 + \dfrac{16}{a}.$ \\
		Để lượng vàng cần dùng ít nhất thì $S$ phải nhỏ nhất.\\
		Xét hàm số $f(x) = a^2 +\dfrac{16}{a}, a>0$, hàm số đạt giá trị nhỏ nhất là $12$ tại $a = 2 \Rightarrow h = 1.$\\
		Nhận xét, sử dụng bất đẳng thức Cô-si, $a^2 + \dfrac{16}{a} = a^2 + \dfrac{8}{a} + \dfrac{8}{a} \geq 3 \sqrt[3]{a^2.\dfrac{8}{a}.\dfrac{8}{a}}\Longleftrightarrow a^2 + \dfrac{16}{a} \geq 12, $ dấu bằng xảy ra khi $a = 2.$
	}
\end{ex}
