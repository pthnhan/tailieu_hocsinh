\begin{ex}%[Đề thi Toán 9 Học sinh gỏi năm học 2017-2018, An Giang]%[9D1K8]
	Cho $x=\dfrac{\sqrt{2+\sqrt{3}}+\sqrt{2-\sqrt{3}}}{\sqrt{2}}$. 	Tìm giá trị của biểu thức $P=\left(1+3x^3-x^5 \right) ^{2016}$.
	\loigiai{
	\begin{align*}
	x&=\dfrac{\sqrt{2+\sqrt{3}}+\sqrt{2-\sqrt{3}}}{\sqrt{2}}\\
	&=\sqrt{2} \cdot \dfrac{\sqrt{2+\sqrt{3}}+\sqrt{2-\sqrt{3}}}{2}\\
	&=\dfrac{\sqrt{4+2\sqrt{3}}+\sqrt{4-2\sqrt{3}}}{2}\\
	&=\dfrac{\sqrt{3+2\sqrt{3}+1}+ \sqrt{3-2\sqrt{3}+1}}{2}\\
	&=\dfrac{\sqrt{\left( \sqrt{3}+1 \right)^2}+\sqrt{\left( \sqrt{3}-1 \right)^2}}{2}\\
	&= \dfrac{2\sqrt{3}}{2}\\
	&=\sqrt{3}\\
	&\Rightarrow 1-3x^3+x^5=1-3\cdot \left(\sqrt{3}\right)^3+\left(\sqrt{3}\right)^5=1.
\end{align*}
Vậy $A=\left(1-3x^3+x^5 \right)^{2016}=1$.}
\end{ex}

	Điểm $M$ thuộc $(P) \Rightarrow M\left( x; \dfrac{1}{4}x^2 \right)$; $A(0;1)$.\\

	$$MN=y_M + 1 = \dfrac{1}{4}x^2 +1$$

	\begin{align*}
	AM &= \sqrt{(x_M)^2 + (y_M - y_A)^2}\\

\begin{ex}%[Đề thi Toán 9 Học sinh gỏi năm học 2017-2018, An Giang]%[9D5G3]
	Cho phương trình $x^3+bx^2+cx+1=0$ trong đó $a,b,c$ là các số nguyên. Biết phương trình có nghiệm $x_0=2+\sqrt{5}$ tìm $b,c$ và các nghiệm còn lại của phương trình.
	\loigiai
	{Phương trình $x^3 + bx^2+ cx +1 = 0$ có nghiệm $x_0 = 2 + \sqrt{5}$ nên
	\begin{align*}	
	&(2+\sqrt{5})^3+ b (2+ \sqrt{5})^2 + c (2+ \sqrt{5})+1 = 0\\
\Leftrightarrow  &(2^3 + 12 \sqrt{5} + 30 + 5\sqrt{5})+b(9+ 4 \sqrt{5})+ c(2+\sqrt{5})+1=0\\
\Leftrightarrow & (4b+c+17)\sqrt{5} + (9b+2c+39)=0
\end{align*}
Vì $\sqrt{5}$ là số vô tỉ, $b,c$ là các số nguyên nên ta có \\
$\begin{cases}
4b+c+17&=0 \\ 9b+2c+39&=0 \end{cases} \Leftrightarrow \begin{cases} 8b+2c+34&=0 \\ 9b+2c+39&=0 \end{cases} \Leftrightarrow \begin{cases} b&=-5 \\ c&= 3 \end{cases}$\\
Với $b=-5;c=3$ phương trình trở thành
$$x^3 -5x^2+3x+1=0 \\ \Leftrightarrow (x-1)(x^2-4x-1)=0$$
	Phương trình có ba nghiệm $x_0 = 2 + \sqrt{5}$; $x_1 = 2- \sqrt{5}$; $x_2 =1.$
}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học sinh gỏi năm học 2017-2018, An Giang]%[%[Đề thi Toán 9 Học sinh gỏi năm học 2017-2018, An Giang]%[9D5G3]
Tìm $x,y$ biết $$\sqrt{2\left(\sqrt{x}+y-2 \right)}=\sqrt{\sqrt{x} \cdot y}.$$
\loigiai{
ĐK $x,y \geq 0 ; \sqrt{x}+y - 2 \geq 0.$\\
Bình phương hai vế của phương trình ta có
\begin{align*}
&2\sqrt{x}+2y-4=\sqrt{x} \cdot y\\
\Leftrightarrow& \sqrt{x}(2-y)+2(y-2)=0\\
\Leftrightarrow & (y-2)(2- \sqrt{x})=0
\end{align*}
Khi đó $x=4$ hoặc $y=2$.
So với điều kiện, ta có
$x=4$; $y \geq 0 \text{ hoặc } x \geq 0$; $y=2.$
}
\end{ex}
\begin{ex}%[Đề thi Toán 9 Học sinh gỏi năm học 2017-2018, An Giang]%[8H3K8]%[8H3K8]%[8H3K8]
Từ một điểm $M$ ở ngoài đường tròn $(O)$ vẽ hai tiếp tuyến $MB,MD$ với đường tròn ($B,D$ là tiếp điểm) và một cát tuyến qua $M$ cắt đường tròn tại hai điểm phân biệt $A,C$.
\begin{enumerate}
\item Chứng minh hai tam giác $MAB$ và $MBC$ đồng dạng.
\item Chứng minh $AB \cdot CD = BC  \cdot AD.$
\item Gọi $(d)$ là đường thẳng qua $D$ và song song với $MB$, $(d)$ cắt $BA,BC$ lần lượt tại $I$ và $J$. Chứng minh $DI=DJ.$
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}
\pgfmathsetmacro{\so}{-sqrt(3)}
\tkzDefPoint(0,0){O}
\tkzDefPoint(0,\so){X}
\tkzDefShiftPoint[O](4,0){M}
\tkzDrawCircle(O,X)
\tkzTangent[from=M](O,X)\tkzGetPoints{B}{D}
\tkzDrawLines[add=0 and 0.0](M,B)
\tkzDrawLines[add=0 and 0.0](M,D)
\tkzDrawPoints(O,M)
\tkzDefPointBy[rotation = center O angle 75](B)
\tkzGetPoint{A}
\tkzInterLC(A,M)(O,A) \tkzGetPoints{A}{C}
\tkzDefLine[parallel=through D](M,B)\tkzGetPoint{c}
\tkzInterLL(D,c)(B,A) \tkzGetPoint{I}
\tkzInterLL(D,c)(B,C) \tkzGetPoint{J}
\tkzLabelPoints[right](M,J)
\tkzLabelPoints[above](B,I)
\tkzLabelPoints[below](D)
\tkzLabelPoints[below left](C,O)
\tkzLabelPoints[above left](A)
\tkzDrawSegments(O,B O,D A,M M,B I,B B,J I,J)
\tkzMarkRightAngles(O,B,M M,D,O)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Chứng minh hai tam giác $MAB$ và $MBC$ đồng dạng.\\
Hai tam giác $MAB$ và $MBC$ có\\
$\widehat{M}$ chung\\
$\widehat{MBA} = \widehat{MCB}$ (cùng chắn cung $\wideparen{AB}$)\\
Vậy $\Delta MAB \backsim \Delta MBC.$
\item Chứng minh $AB \cdot CD = BC  \cdot AD.$\\
Ta có \\
$\Delta MAB \backsim \Delta MBC \Rightarrow \dfrac{AB}{BC} = \dfrac{MB}{MC}$ $(1)$\\
Hai tam giác $MAD$ và $MDC$ có\\
$\widehat{M}$ chung\\
$\widehat{MDA} = \widehat{MCD}$ (cùng chắn cung $\wideparen{AD}$)\\
Vậy $\Delta MAD \backsim  \Delta MDC.$\\
$\Rightarrow \dfrac{AD}{DC} = \dfrac{MD}{MC}$ $(2)$\\
Mà $MB = MD$ nên từ $(1)$ và $(2)$ ta có\\
$$\dfrac{AB}{BC} = \dfrac{AD}{DC} \Leftrightarrow AB \cdot DC = AD \cdot BD.$$
\item Gọi $(d)$ là đường thẳng qua $D$ và song song với $MB$, $(d)$ cắt $BA,BC$ lần lượt tại $I$ và $J$.Chứng minh $DI=DJ.$\\
Hai tam giác $IBD$ và $DBA$ đồng dạng do\\
$\widehat{B}$ chung\\
$\widehat{BID} = \widehat{IBM}$ (so le trong)\\
$\widehat{IBM} = \widehat{BDA}$ (chắn cung  $\wideparen{AB}$)
$$\Rightarrow \dfrac{BD}{BA} =  \dfrac{ID}{DA} \Leftrightarrow \dfrac{AD}{AB} = \dfrac{ID}{BD} (3)$$
Tương tự hai tam giác $JBD$ và $DBC$ đồng dạng.
$$\Rightarrow \dfrac{BD}{BA} = \dfrac{ID}{DC} \Leftrightarrow \dfrac{DC}{BC} = \dfrac{JD}{BD} (4)$$
Theo câu $a)$
$$\dfrac{AB}{BC} = \dfrac{AD}{DC} \Rightarrow \dfrac{AD}{AB} = \dfrac{DC}{BC} (5)$$
$(3),(4),(5) \Rightarrow \dfrac{ID}{BD} = \dfrac{JD}{BD} \Rightarrow DI = DJ.$
\end{enumerate}
}
\end{ex}
