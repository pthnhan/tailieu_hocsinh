\section{Số nguyên tố. Hợp số}
\subsection{Kiến thức cơ bản}
\begin{enumerate}
	\item Nếu có số tự nhiên $a$ chia hết cho số tự nhiên $b$ thì ta nói $a$ là bội của $b$, còn $b$ là ước của $a$.
	\item Số nguyên tố là số tự nhiên lớn hơn $1$, chỉ có hai ước là $1$ và chính nó.
	\item Hợp số là số tự nhiên lớn hơn $1$, có nhiều hơn hai ước.
	\item Phân tích một số tự nhiên lớn hơn $1$ ra thừa số nguyên tố là viết số đó dưới dạng một tích các thừa số nguyên tố. Nhờ phân tích một số tự nhiên ra thừa số nguyên tố, ta dễ dàng tìm được các ước của số tự nhiên đó.
\end{enumerate}
\textbf{KIẾN THỨC BỔ SUNG}
\begin{enumerate}
	\item Nếu một số $n$ phân tích ra thừa số nguyên tố có dạng $n=a^x\cdot b^y\cdot c^z\cdots$ thì số lượng các ước của $n$ là $(x+1)(y+1)(z+1)\cdots$\\
	\textbf{\textit{Chứng minh}}\\
	Ước của $n$ là số có dạng $A\cdot B\cdot C\cdots$, trong đó
	\begin{itemize}
		\item $A$ có $x+1$ cách chọn là $a^0$, $a^1$, $a^2$, \ldots, $a^x$;
		\item $B$ có $y+1$ cách chọn là $b^0$, $b^1$, $b^2$, \ldots, $b^y$;
		\item $C$ có $z+1$ cách chọn là $c^0$, $c^1$, $c^2$, \ldots, $c^z$;\\
		\ldots
	\end{itemize}
	Do đó số lượng các ước của $n$ bằng $(x+1)(y+1)(z+1)\cdots$
	\item Khi phân tích ra thừa số nguyên tố, số chính phương chỉ chứa các thừa số nguyên tố với số mũ chẵn.\\
	Do đó
	\begin{itemize}
		\item Số chính phương chia hết cho $2$ thì phải chia hết cho $4$.
		\item Số chính phương chia hết cho $3$ thì phải chia hết cho $9$.
		\item Số chính phương chia hết cho $5$ thì phải chia hết cho $25$.
	\end{itemize}
	\item Nếu tích $ab$ chia hết cho số nguyên tố $p$ thì tồn tại một thừa số của tích chia hết cho $p$.\\
	Đặc biệt, nếu $a^n$ chia hết cho số nguyên tố $p$ thì $a$ chia hết cho $p$.
\end{enumerate}
\subsection{Ví dụ minh họa}
\begin{vd}
	Các số $30$ và $17$ chia cho số tự nhiên $a$ khác $1$ đều dư $r$. Tìm $a$ và $r$.
	\loigiai{
		Các số $30$ và $17$ chia cho $a$ có cùng số dư nên hiệu $30-17$ chia hết cho $a$, tức là $13\,\vdots\, a$.\\
		Số nguyên tố $13$ có hai ước là $1$ và $13$. Do $a\ne 1$ nên $a=13$.\\
		Ta có $30$ chia $13$ được $2$, dư $4$.\\
		Vậy $a=13$ và $r=4$.
	}
\end{vd}
\begin{vd}
	Có hơn $20$ học sinh xếp thành một vòng tròn. Khi đếm theo chiều kim đồng hồ, bắt đầu từ số $1$, thì các số $24$ và $900$ rơi vào cùng một học sinh. Hỏi ít nhất có bao nhiêu học sinh?
	\loigiai{
		Gọi số học sinh là $a$ ($a>20$). Các số $24$ và $900$ chia cho $a$ có cùng số dư nên $900-24\,\vdots\, a$ tức là $876\,\vdots\, a$.\\
		Phân tích ra thừa số nguyên tố: $876=2^2\cdot 3\cdot 73$.\\
		Ước của $876$, lớn hơn $20$ và nhỏ nhất là $73$.\\
		Vậy có ít nhất $73$ học sinh.
	}
\end{vd}
\begin{vd}
	Tìm số tự nhiên $n$, biết rằng $1+2+3+4+\cdots +n=378$.
	\loigiai{
		Ta có $1+2+3+\cdots +n=\dfrac{n(n+1)}{2}$.\\
		Do đó $\dfrac{n(n+1)}{2}=378$, tức là $n(n+1)=756$.\\
		Phân tích ra thừa số nguyên tố: $756=2^2\cdot 3^3\cdot 7$.\\
		Do đó $756=3^3\cdot (2^2\cdot 7)=27\cdot 28$.\\
		Vậy $n=27$.
	}
\end{vd}
\begin{vd}
	Cho tích $800$ số tự nhiên từ $1$ đến $800$: $A=1\cdot 2\cdot 3 \cdots 800$.
	\begin{enumerate}
		\item Dạng phân tích của số $A$ ra thừa số nguyên tố chứa thừa số $5$ với số mũ bao nhiêu?
		\item Số $A$ tận cùng bằng bao nhiêu chữ số $0$?
	\end{enumerate}
	\loigiai{
	\begin{enumerate}
		\item Trong các số tự nhiên từ $1$ đến $800$:
		\begin{itemize}
			\item Các bội của $5$ là $5$, $10$, $15$, \ldots, $800$ gồm $(800-5):5+1=160$ (số).
			\item Các bội của $5^2$ là $25$, $50$, $75$, \ldots, $800$ gồm $(800-25):25+1=32$ (số).
			\item Các bội của $5^3$ là $125$, $250$, \ldots, $750$ gồm $(750-125):125+1=6$ (số).
			\item Các bội của $5^4$ là $625$, gồm $1$ số.
			\item Không có bội của $5^5$.
		\end{itemize}
		Vậy dạng phân tích của số $A$ ra thừa số nguyên tố chứa thừa số $5$ với số mũ là $$160+32+6+1=199.$$
		\item Dạng phân tích ra thừa số nguyên tố của số $A$ chứa thừa số $5$ với số mũ $199$, chứa thừa số $2$ với số mũ lớn hơn $199$ (vì chỉ riêng các số chẵn từ $2$ đến $800$ đã có $400$ số).\\
		Mỗi cặp thừa số $2$ và $5$ làm cho tích $A$ tận cùng một chữ số $0$. Do đó $A$ tận cùng bằng $199$ chữ số $0$.
		\begin{luuy}
			Ở câu a), với nhận xét kể từ số $1$, cứ $5$ số lại có một bội của $5$, cứ $25$ số lại có một bội của $25$, \ldots ta có thể tìm được $199$ như sau:
			\begin{align*}
			&\ [800:5]+[800:5^2]+[800:5^3]+[800:5^4]\\
			= &\ 160+32+6+1=199.
			\end{align*}
		\end{luuy}
	\end{enumerate}
	}
\end{vd}
\begin{vd}
	Chứng minh rằng chỉ có duy nhất một bộ ba số nguyên tố mà hiệu của hai số liên tiếp bằng $4$.
	\loigiai{
		Giả sử $p$, $p+4$, $p+8$ là ba số nguyên tố.\\
		Ta thấy $p\ne 2$, vì nếu $p=2$ thì $p+4=6$ là hợp số.\\
		Xét $p=3$ thì $3$, $7$, $11$ là bộ ba số nguyên tố mà hiệu của hai số liên tiếp bằng $4$.\\
		Xét $p>3$ thì $p$ có dạng $3k+1$ hoặc $3k+2$ ($k\in \mathbb{N}$).\\
		Loại $p=3k+1$ vì khi đó $p+8=3k+9$ là hợp số.\\
		Loại $p=3k+2$ vì khi đó $p+4=3k+6$ là hợp số.\\
		Vậy chỉ có duy nhất bộ ba số nguyên tố $3$, $7$, $11$ thỏa mãn đề bài.
	}
\end{vd}
\begin{vd}
	Tìm số tự nhiên nhỏ nhất có $8$ ước số.
	\loigiai{
		Ta biết rằng nếu số $n$ phân tích ra thừa số nguyên tố được $a^xb^yc^z\cdots$ thì $n$ có $(x+1)(y+1)(z+1)\cdots$ ước, trong đó $x+1$, $y+1$, $z+1$\ldots đều lớn hơn $1$.\\
		Theo đề bài $(x+1)(y+1)(z+1)\cdots =8$.\\
		Số $8$ có ba cách viết thành một tích của một hay nhiều thừa số lớn hơn $1$ là $8$, $4\cdot 2$, $2\cdot 2\cdot 2$. Xét ba trường hợp:
		\begin{itemize}
			\item Nếu $x+1=8$ thì $x=7$. Ta có $n=a^7$. Để $n$ nhỏ nhất, ta chọn $n=2^7=128$.
			\item Nếu $(x+1)(y+1)=4\cdot 2$ thì $x=3$, $y=1$ (giả sử $x\ge y$).\\
			Ta có $n=a^3b$. Để $n$ nhỏ nhất, ta chọn $n=2^3\cdot 3=24$.
			\item Nếu $(x+1)(y+1)(z+1)=2\cdot 2\cdot 2$ thì $x=y=z=1$.\\
			Ta có $n=abc$. Để $n$ nhỏ nhất, ta chọn $n=2\cdot 3\cdot 5=30$.
		\end{itemize}
		So sánh ba trường hợp trên, ta thấy số nhỏ nhất có $8$ ước là $24$.
	}
\end{vd}
\begin{vd}
	Viết mỗi số sau thành một tổng của các hợp số sao cho số số hạng của tổng là nhiều nhất.
	\begin{listEX}[4]
		\item $100$
		\item $102$
		\item $101$
		\item $103$
	\end{listEX}
	\loigiai{
		Muốn có nhiều số hạng nhất thì giá trị của mỗi số hạng phải nhỏ nhất. Các hợp số chẵn nhỏ nhất là $4$ và $6$. Hợp số lẻ nhỏ nhất là $9$.
		\begin{enumerate}
			\item $100$ chia cho $4$ được $25$ lần. Ta viết $100$ thành tổng của $25$ số $4$: $$100=4+4+\cdots +4 \text{ ($25$ số hạng).}$$
			\item $102$ chia cho $4$ được $25$, dư $2$. Ta viết $102$ thành tổng của $25$ hợp số gồm một số $6$ và $24$ số $4$: $$102=6+4+4+\cdots +4\text{ ($24$ số 4).}$$
			\item $101$ chia cho $4$ được $25$, dư $1$. Ta viết $101$ thành tổng của $24$ hợp số gồm một số $9$ và $23$ số $4$: $$101=9+4+4+\cdots +4\text{ ($23$ số 4).}$$
			\item $103$ chia cho $4$ được $25$, dư $3$. Ta viết $103$ thành tổng của $24$ hợp số gồm một số $9$, một số $6$ và $22$ số $4$: $$103=9+6+4+4+\cdots +4\text{ ($22$ số 4).}$$
		\end{enumerate}
	}
\end{vd}
\subsection{Bài tập tự luyện}
\begin{bt}
	Trong một tháng, có ba ngày chủ nhật là ba số nguyên tố. Ngày $15$ của tháng đó là ngày thứ mấy?
	\loigiai{
		Nếu ngày chủ nhật đầu tiên là ngày $2$ thì các ngày chủ nhật trong tháng là các ngày $2$, $9$, $16$, $23$, $30$, chỉ có hai số nguyên tố là $2$ và $23$, loại.\\
		Vậy nếu ba ngày chủ nhật của tháng là ba số nguyên tố thì ba số đó phải lẻ và số sau hơn số trước $14$ đơn vị. Đó là các ngày $3$, $17$, $31$.\\
		Ngày $15$ của tháng đó là ngày thứ sáu.
	}
\end{bt}
\begin{bt}
	Viết liên tiếp các số tự nhiên từ $1$ đến $99$, ta được một số $A$. Số $A$ là số nguyên tố hay hợp số?
	\loigiai{
		Ta có $A=1234\ldots 979899$.\\
		Xét $99$ số $1$, $2$, $3$, \ldots, $98$, $99$. Mỗi chữ số từ $1$ đến $99$ có mặt $10$ lần ở hàng chục và $10$ lần ở hàng đơn vị nên tổng các chữ số của $A$ bằng $(1+2+3+\cdots +9)\cdot 20=900$, chia hết cho $9$.\\
		Suy ra $A$ chia hết cho $9$ và là hợp số.
	}
\end{bt}
\begin{bt}
	Các số sau có là số chính phương không?
	\begin{listEX}[2]
		\item $2+2^2+2^3+2^4+\cdots +2^{20}$
		\item $10^{15}+8$
	\end{listEX}
	\loigiai{
	\begin{enumerate}
		\item $2+2^2+2^3+2^4+\cdots +2^{20}$ chia hết cho $2$ nhưng không chia hết cho $4$ nên không là số chính phương.
		\item $10^{15}+8$ có tận cùng bằng $8$ nên không là số chính phương.
	\end{enumerate}
	}
\end{bt}
\begin{bt}
	Cho $p$ và $p+14$ là các số nguyên tố. Chứng minh rằng $p+7$ là hợp số.
	\loigiai{
		Loại $p=2$ vì khi đó $p+14=16$ không là số nguyên tố.\\
		Xét $p=3$ thì $p+14=17$ là số nguyên tố, $p+7=10$ là hợp số.\\
		Xét $p>3$ thì $p$ có dạng $3k+1$ hoặc $3k+2$ ($k\in \mathbb{N}$).
		\begin{itemize}
			\item Nếu $p=3k+1$ thì $p+14=3k+15$ không là số nguyên tố, loại.
			\item Nếu $p=3k+2$ thì $p+7=3k+9$ là hợp số.
		\end{itemize}
	}
\end{bt}
\begin{bt}%[Dựa án 6SA19-NC1, Thịnh Trần]
	Cho $p$, $p + 20$, $p + 40$ là các số nguyên tố. Chứng minh rằng $p + 80$ là số nguyên tố.
	\loigiai{
		Loại $p = 2$ vì khi đó $p + 20$ không là số nguyên tố.\\
		Xét $p = 3$ thì $p + 20 = 23$,  $p + 40 = 43$, $p + 80 = 83$ đều là các số nguyên tố.\\
		Xét $p > 3$ thì $p$ có dạng $3k + 1$ hoặc $3k + 2$ ($k \in \mathbb{N}$).
		\begin{itemize}
			\item Nếu $p = 3k + 1$ thì $p + 20 = 3k + 21$ không là số nguyên tố, loại.
			\item Nếu $p = 3k + 2$ thì $p + 40 = 3k + 42$ không là số nguyên tố, loại.	
		\end{itemize}
		Vậy nếu $p$, $p + 20$, $p + 40$ là các số nguyên tố thì $p + 80$ là số nguyên tố.
	}
\end{bt}

\begin{bt}%[Dựa án 6SA19-NC1, Thịnh Trần] 
	Tìm số nguyên tố $p$ sao cho $p + 6$,  $p + 12$, $p + 18$, $p + 24$ cũng là các số nguyên tố.
	\loigiai{
		Xét $p = 2$ thì $p+6=8$ là hợp số nên $p=2$ không thỏa mãn.\\
		Xét $p = 3$ thì $p+12=15$ là hợp số nên $p=3$ không thỏa mãn.\\
		Xét $p = 5$ thì $p + 6 = 11$, $p + 12 = 17$, $p + 18 = 23$, $p + 24 = 29$ đều là các số nguyên tố.\\
		Xét $p > 5$ thì $p$ có một trong các dạng $5k + 1$, $5k + 2$, $5k + 3$, $5k + 4$ với $k \in \mathbb{N}$. 
		\begin{itemize}
			\item Nếu $p=5k+1$ thì $p+24=5k+25$ chia hết cho $5$ nên là hợp số.
			\item Nếu $p=5k+2$ thì $p+18=5k+20$ chia hết cho $5$ nên là hợp số.
			\item Nếu $p=5k+3$ thì $p+12=5k+15$ chia hết cho $5$ nên là hợp số.
			\item Nếu $p=5k+4$ thì $p+6=5k+10$ chia hết cho $5$ nên là hợp số.
		\end{itemize}
		Vậy $p=5$.
	}
\end{bt}

\begin{bt}%[Dựa án 6SA19-NC1, Thịnh Trần] 
	Tìm số nguyên tố nhỏ hơn $200$, biết rằng khi chia nó cho $60$ thì số dư là hợp số.
	\loigiai{
		Gọi $p$ là số nguyên tố phải tìm. Ta có $p = 60k + r = 2^2\cdot 3\cdot 5k + r$ với $k$, $r\in \mathbb{N}$, $0 < r < 60$ và $r$ là hợp số.\\
		Do $p$ là số nguyên tố nên $r$ không chia hết cho $2$, $3$, $5$.\\
		Chọn các hợp số nhỏ hơn $60$, loại đi các số chia hết cho $2$, ta có $$r\in\{9; 15; 21; 25 ; 27 ; 33 ; 35 ; 39 ; 45 ; 49 ; 51 ; 55 ; 57\}.$$
		Loại đi các số chia hết cho $3$, ta có $r \in \{25 ; 35 ; 49 ; 55\}$.\\
		Loại đi các số chia hết cho $5$, ta được $r = 49$.\\
		Suy ra $p = 60k + 49$. Do $p < 200$ nên $p = 109$ hoặc $p = 169$. Loại $p = 169 =13^2$. \\Vậy $p= 109$.
	}
\end{bt}

\begin{bt}%[Dựa án 6SA19-NC1, Thịnh Trần] 
	Chứng minh rằng số $\underbrace{11\ldots 1}_{10 \text{ chữ số }1}2\underbrace{11\ldots 1}_{10 \text{ chữ số }1}$ là hợp số.
	\loigiai{
		Ta có $\underbrace{11\ldots 1}_{10 \text{ chữ số }1}2\underbrace{11\ldots 1}_{10 \text{ chữ số }1} = \underbrace{11\ldots 1}_{11 \text{ chữ số }1}\underbrace{00\ldots 0}_{10 \text{ chữ số }0}+ \underbrace{11\ldots 1}_{11 \text{ chữ số }1}$ chia hết cho $\underbrace{11\ldots 1}_{11 \text{ chữ số }1}$ nên là hợp số.
	}
\end{bt}

\begin{bt}%[Dựa án 6SA19-NC1, Thịnh Trần] 
	Tìm ba số tự nhiên liên tiếp có tích bằng $13800$.
	\loigiai{
		Phân tích ra thừa số nguyên tố, ta được $13800 = (2^3\cdot 3)\cdot 5^2\cdot 23$.\\
		Vậy ba số tự nhiên liên tiếp phải tìm là $23$, $24$, $25$.
	}
\end{bt}

\begin{bt}%[Dựa án 6SA19-NC1, Thịnh Trần] 
	Tìm số tự nhiên $n$ biết rằng:
	\begin{listEX}[2]
		\item $2n + 1$ chia hết cho $n - 3$.
		\item $n^2 + 3$ chia hết cho $n + 1$.
	\end{listEX}	
	\loigiai{
		\begin{enumerate}
			\item $2n + 1 = 2n - 6 + 7 = 2(n - 3) + 7$.\\
			Vì $(2n + 1) \,\vdots\, (n - 3)$ nên $7 \,\vdots\, (n - 3) \Rightarrow (n-3) \in \{1;7\}
			\Rightarrow n \in \{4 ; 10\}$. \\
			Vậy $n \in \{4 ; 10\}$.
			\item $n^2 + 3 = n(n + 1) - (n + 1) + 4$.\\
			Vì $(n^2 + 3)\,\vdots\, (n + 1)$ nên $4 \,\vdots\, (n + 1) \Rightarrow n+1\in \{1;2;4\}
			\Rightarrow n\in \{0; 1 ;3\}$.	\\
			Vậy $n\in \{0; 1 ;3\}$.
		\end{enumerate}	
	}
\end{bt}

\begin{bt}%[Dựa án 6SA19-NC1, Thịnh Trần] 
	Tìm số tự nhiên $n$ biết rằng
	\begin{listEX}[2]
		\item	$1 + 2 + 3 + 4 +\cdots + n = 231$;
		\item   $1 + 3 + 5 + 7 +\cdots + (2n - 1) = 169$.
	\end{listEX}	
	\loigiai{
		\begin{enumerate}
			\item Theo tính chất của dãy cách đều ta có $1 + 2 + 3 + 4 +\cdots + n=n(n+1):2$.\\
			Suy ra $n(n+1):2=231\Rightarrow n=21$.
			\\Vậy $n = 21$.
			\item Vì $1 + 3 + 5 + \cdots + (2n - 1)$ là tổng của $n$ số lẻ liên tiếp nên tổng đó bằng $[1 +(2n- 1)]\cdot n :2$.\\
			Suy ra $[1 +(2n- 1)]\cdot n :2 = 169\Rightarrow n^2=169\Rightarrow n=13$.\\
			Vậy $n = 13$.
		\end{enumerate}	
	}	
\end{bt}

\begin{bt}%[Dựa án 6SA19-NC1, Thịnh Trần]

\begin{bt}%[Dựa án 6SA19-NC1, Thịnh Trần] 
	Bạn Chi tính tổng các số tự nhiên liên tiếp từ $1$ đến $n$ và nhận thấy tổng đó chia hết cho $29$. Bạn Hoàng tính tổng các số tự nhiên liên tiếp từ $1$ đến $m$ và cũng nhận thấy tổng đó chia hết cho $29$. Tìm $m$ và $n$, biết rằng $m < n < 50$.
	\loigiai{
		Xét $1 + 2 + 3 +\cdots +k = k(k+1):2$ với $k<50$.\\
		Vì $1 + 2 + 3 +\cdots +k$ chia hết cho $29$ nên $k(k + 1)\,\vdots\, 29$.\\
		Do $k(k + 1)$ chia hết cho số nguyên tố $29$ nên $k \,\vdots\, 29$ hoặc $(k +1)\,\vdots\, 29$.\\
		Mặt khác $0 < k < 50$ nên $k = 29$ hoặc $k + 1 = 29$, tức là $k \in \{29 ; 28\}$.\\
		Vì $m<n$ nên $m = 28$ và $n = 29$.
	}
\end{bt}

\begin{bt}%[Dựa án 6SA19-NC1, Thịnh Trần] 
	Tìm số tự nhiên nhỏ nhất:
	\begin{listEX}[2]
		\item Có $7$ ước số;
		\item Có $15$ ước số.
	\end{listEX}
	\loigiai{
		Gọi $n$ là số tự nhiên cần tìm, dễ thấy $n=0$, $n=1$ không thỏa mãn.\\
		Với mọi số tự nhiên $n>1$ ta luôn phân tích được ra thừa số nguyên tố $n=a^x\cdot b^y\cdot c^z\cdots $ (trong đó $a$, $b$, $c$ $\ldots$ là các số nguyên tố; $x$, $y$, $z$, $\ldots$ là các số tự nhiên khác $0$). Số ước của $n$ được tính theo công thức $(x+1)\cdot (y+1)\cdot (z+1) \cdots $.\\
		$n$ nhỏ nhất khi $a<b<c<\ldots $ và $x\geq y\geq z\geq \ldots$.
		\begin{enumerate}
			\item 
			Ta có $7=7\cdot 1$ nên $n=2^6=64$.\\
			Vậy số tự nhiên nhỏ nhất có $7$ ước số là $2^6 = 64$.
			\item 
			Ta có $15=15\cdot 1=5\cdot 3$ nên suy ra $n=2^{14}$ hoặc $n=2^4\cdot 3^2=144$.\\
			Vậy số tự nhiên nhỏ nhất có $15$ ước số là $2^4\cdot 3^2=144$.
		\end{enumerate}	
	}
\end{bt}

\begin{bt}%[Dựa án 6SA19-NC1, Thịnh Trần] 
	Tìm số tự nhiên nhỏ nhất chỉ chứa các thừa số nguyên tố $2$ và $5$, biết rằng khi chia nó cho $2$ thì được một số chính phương.
	\loigiai{
		Gọi số phải tìm là $n$, phân tích ra thừa số nguyên tố được $n = 2^x\cdot 5^y$ ($x$, $y\in\mathbb{N}^*$).\\
		Vì $n : 2 = 2^{x-1}\cdot 5^y$ là số chính phương nên $x - 1$ và $y$ đều chẵn.\\
		Để $n$ nhỏ nhất thì $x= 1$ và $y = 2$. Khi đó $n = 2\cdot 5^2 = 50$.
	}
\end{bt}

\begin{bt}%[Dựa án 6SA19-NC1, Thịnh Trần] 
	Tìm số tự nhiên nhỏ nhất khác $0$ sao cho khi chia nó cho $2$ thì được một số chính phương, khi chia nó cho $3$ thì được lập phương của một số tự nhiên.
	\loigiai{
		Gọi số phải tìm là $n$. Ta thấy $n$ chia hết cho $2$ và $3$, nên để $n$ nhỏ nhất ta chọn $n = 2^x\cdot 3^y$ ($x$, $y\in\mathbb{N}^*$).\\
		Vì $n : 2 = 2^{x-1}\cdot 3^y = a^2$ nên $x - 1$ và $y$ đều chẵn,\\
		Vì $n : 3 = 2^x\cdot 3^{y-1} = b^3$ nên $x$ và $y - 1$ đều chia hết cho $3$.\\
		Từ $(x- 1) \,\vdots\, 2$ và $x \,\vdots\, 3$, để $n$ nhỏ nhất ta chọn $x = 3$.\\
		Từ $y \,\vdots\, 2$ và $(y - 1) \,\vdots\, 3$, để $n$ nhỏ nhất ta chọn $y = 4$.\\
		Vậy $n = 2^3\cdot 3^4 = 648$.
	}
\end{bt}

\begin{bt}%[Dựa án 6SA19-NC1, Thịnh Trần] 
	Số tự nhiên $n$ chỉ chứa hai thừa số nguyên tố. Biết rằng $n^2$ có $21$ ước số. Hỏi số $n^3$ có bao nhiêu ước số?
	\loigiai{
		Dạng phân tích ra thừa số nguyên tố của $n$ là $n = a^x\cdot b^y$ ($x$, $y\in\mathbb{N}^*$).\\
		Ta có $n^2 = a^{2x}\cdot b^{2y}$ có $(2x + 1 )(2y + 1)$ ước số nên $(2x	+1 )(2y +1) = 21$.\\
		Giả sử $x \leq y$, ta được $x = 1$ và $y = 3$.\\
		$n^3 = a^{3x}\cdot b^{3y}$ có $(3x + 1)(3y + 1)$ ước, tức là có $4\cdot 10 = 40$ (ước).
	}
\end{bt}
