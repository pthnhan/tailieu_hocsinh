\begin{ex}%[HK2 (2016-2017),THPT Sông Lô - Vĩnh Phúc]%[0H3B2]%[Lê Ngọc Hiếu]

	\choice
	{$x+y=0$}
	{$x-y+2=0$}

\begin{ex}%[HK2 (2016-2017),THPT Sông Lô - Vĩnh Phúc]%[0D6B3]%[Lê Ngọc Hiếu]
Cho $\sin \alpha = \dfrac{3}{5}, \dfrac{\pi}{2}< \alpha < \pi$. Chọn kết quả đúng
\choice
{$\cos \alpha = \dfrac{4}{5}$}
{$\tan \alpha = \dfrac{3}{4}$}
{$\tan \alpha = \dfrac{-4}{3}$}
{\True $\cos \alpha = \dfrac{-4}{5}$}
\loigiai{
Ta có
\begin{align*}
& \sin ^2 \alpha + \cos ^2 \alpha = 1\\
\Leftrightarrow & \cos ^ 2  \alpha = 1 - \sin^ 2 \alpha\\
\Leftrightarrow & \cos^2 \alpha = 1 - \left(\dfrac{3}{5}\right)^2\\
\Leftrightarrow & \cos^2 \alpha = \dfrac{16}{25}\\
\Rightarrow & \cos \alpha = - \dfrac{4}{5} \text{( vì $\dfrac{\pi}{2}< \alpha < \pi$)}.
\end{align*}
}
\end{ex}
\begin{ex}%[HK2 (2016-2017),THPT Sông Lô - Vĩnh Phúc]%[0H2K3]%[Lê Ngọc Hiếu]
Độ dài trung tuyến $AM$ của tam giác $ABC$ có $AB=48$, $AC=14$, $BC=50$ là
\choice
{\True $25$}
{$48,5$}
{$27,8$}
{$18,5$}
\loigiai{
Áp dụng công thức
\begin{align*}
AM^2 &= \dfrac{2(AB^2 + AC^2)- BC^2}{4}\\
&= \dfrac{2(48^2 + 14^2) - 50^2}{4}\\
&= 625\\
\Rightarrow AM &= \sqrt{625} = 25. 
\end{align*}
}

\noindent\textbf{II. PHẦN TỰ LUẬN}
\begin{bt}%[HK2 (2016-2017),THPT Sông Lô - Vĩnh Phúc]%[0D4K2]%[Lê Ngọc Hiếu]
\begin{enumerate}
\item Giải bất phương trình     
$$\dfrac{4-x}{2x^2 - 3x +1} \leq 0.$$
\item Giải hệ bất phương trình sau
$$\heva{2x-3 & \geq \dfrac{5x+2}{4} \\ 7 - 3x &< \dfrac{x+1}{3}}.$$
\item Giải bất phương trình $$|2x-1| \leq \sqrt{6x+7}.$$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item     
$$\dfrac{4-x}{2x^2 - 3x +1} \leq 0$$
\begin{itemize}
\item $4-x=0 \Leftrightarrow x=4$
\item $2x^2-3x+1=0 \Leftrightarrow \hoac{x&=\dfrac{1}{2}\\ x&= 1}$
\end{itemize}
Bảng xét dấu
\begin{center}
\begin{tikzpicture}
\tkzTabInit[lgt=3,espcl=1.5]%
{$x$ /1,
$4-x$ /1,
$2x^2 - 3x +1$ /1,
$\dfrac{4-x}{2x^2 - 3x +1}$ /2}
{$- \infty$, $\dfrac{1}{2}$ , $1$ , $4$ ,$+\infty$}
\tkzTabLine{ ,+,t,+,t,+,$0$,-,}
\tkzTabLine{ ,+,$0$,-,$0$,+,t,+,}
\tkzTabLine{ ,+,d,-,d,+,$0$,-,}
\end{tikzpicture}
\end{center}
Vậy tập nghiệm là $$S = \left( \dfrac{1}{2};1 \right) \cup [4; + \infty).$$
\item
\begin{align*}
&\heva{2x-3  &\geq \dfrac{5x+2}{4} \\ 7 - 3x &< \dfrac{x+1}{3}}\\
\Leftrightarrow &\heva{8x-12 &\geq 5x+2 \\ 21 - 9x &< x+1}\\
\Leftrightarrow &\heva{3x &\geq 14 \\ -10x &< -20}\\
\Leftrightarrow &\heva{x &\geq \dfrac{14}{3}\\ x &> 2}\\
\Leftrightarrow & \ \ x \geq \dfrac{14}{3}
\end{align*}
Vậy tập nghiệm $S=\left[\dfrac{14}{3}; + \infty \right).$
\item 
\begin{align*}
&|2x-1| \leq \sqrt{6x+7}\\
\Leftrightarrow & (2x-1)^2 \leq \left(\sqrt{6x+7} \right)^2\\
\Leftrightarrow & 4x^2-4x+1 \leq 6x+7\\
\Leftrightarrow & 2x^2-5x-3 \leq 0\\
\end{align*}
Ta có$$2x^2-5x-3=0 \Leftrightarrow \hoac{x&= - \dfrac{1}{3}\\ x&=3}$$
Bảng xét dấu
\begin{center}
\begin{tikzpicture}
	\tkzTabInit[nocadre=false, lgt=2, espcl=2.5]{$x$ /1,VT/1}{$-\infty$,$- \dfrac{1}{3}$,$3$,$+\infty$}
	\tkzTabLine{,+,$0$,-,$0$,+,}
	\end{tikzpicture}
\end{center}
Vậy tập nghiệm $S=\left[- \dfrac{1}{3};3 \right].$
\end{enumerate}
}
\end{bt}
\begin{bt}%[HK2 (2016-2017),THPT Sông Lô - Vĩnh Phúc]%[0D6B2]%[Lê Ngọc Hiếu]
Cho $\cos x = - \dfrac{3}{5}, \dfrac{\pi}{2}<x< \pi$. Tính các giá trị lượng giác còn lại.
\loigiai{
Ta có 
\begin{align*}
& \sin^2 x + \cos^2 x = 1\\
\Leftrightarrow & \sin^2 x = 1 - \cos^2 x\\
\Leftrightarrow & \sin^2 x = 1 - \left(\dfrac{-3}{5}\right)^2\\
\Leftrightarrow & \sin^2 x = \dfrac{16}{25}\\
\Rightarrow & \sin x = \dfrac{4}{5} \text{    (vì }\dfrac{\pi}{2}<x< \pi)\\
\tan x &= \dfrac{\sin x}{\cos x } = \dfrac{-4}{3}\\
\cot x &= \dfrac{\cos x}{\sin x} = \dfrac{-3}{4}.
\end{align*}
}
\end{bt}
\begin{bt}%[HK2 (2016-2017),THPT Sông Lô - Vĩnh Phúc]%[0H3K1]%[Lê Ngọc Hiếu]
Trong mặt phẳng tọa độ $Oxy$ cho tam giác $ABC$ với $A(1;0)$; $B(3;2)$; $C(-1;2)$.
\begin{enumerate}
\item Lập phương trình tổng quát của đường cao $AH$.
\item Lập phương trình đường tròn đường kính $BC$. Tìm giao điểm của đường thẳng $AH$ với
đường tròn.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Lập phương trình tổng quát của đường cao $AH$.\\
$\vv{BC}= (-4;0)$\\
Phương trình tổng quát $AH$
$$(AH):
-4(x-1) + 0(y-0)=0 \Leftrightarrow x-1=0.
$$
\item Lập phương trình đường tròn đường kính $BC$. Tìm giao điểm của đường thẳng $AH$ với
đường tròn.\\
Gọi $I$ là trung điểm $BC$ 
$\Rightarrow $ $I(1;2)$
$$\vv{BC} = (-4;0) \Rightarrow BC = 4 \Rightarrow R = 2.$$
Phương trình đường tròn
$$(x-1)^2+(y-2)^2=4.$$
Giao điểm của $AH$ và đường tròn là nghiệm của hệ
$$\heva{x-1&=0 \\ (x-1)^2+(y-2)^2&=4}\Leftrightarrow
\heva{x=1 \\ \hoac{y&=0 \\ y&=4}}$$
Vậy giao điểm là $M(1;0)$ và $N(1;4).$
\end{enumerate}
}
\end{bt}
