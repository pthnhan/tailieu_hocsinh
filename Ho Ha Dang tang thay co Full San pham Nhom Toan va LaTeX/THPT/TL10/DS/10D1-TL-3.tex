\section{CÁC PHÉP TOÁN TẬP HỢP}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Giao của hai tập hợp}
\begin{dn}
Tập hợp $C$ gồm các phần tử vừa thuộc tập hợp $A$,  vừa thuộc tập hợp $B$ được gọi là giao của $A$ và $B$. Kí hiệu  $C=A\cap B$.
\end{dn}
Vậy ${A\cap B=\{x| x\in A\,\,\text{và}\,\,x\in B\}.}$
\begin{center}
	\begin{venndiagram2sets}
		\fillACapB
	\end{venndiagram2sets}
\end{center}
\begin{note}
$x \in A \cap B \Leftrightarrow \left\{ \begin{array}{l}
x \in A\\
x \in B.
\end{array} \right.$
\end{note}
\subsubsection{Hợp của hai tập hợp}
\begin{dn}
	Tập hợp $C$ gồm các phần tử  thuộc tập hợp $A$ hoặc thuộc tập hợp $B$ được gọi là hợp của $A$ và $B$. Kí hiệu  $C=A\cup B$.
\end{dn}
${A\cup B=\{x| x\in A\,\,\text{hoặc}\,\,x\in B\}.}$
\begin{center}
	\begin{venndiagram2sets}
		\fillA \fillB
	\end{venndiagram2sets}
\end{center}
\begin{note}
$x \in A \cup B \Leftrightarrow \left[ \begin{array}{l}
x \in A\\
x \in B.
\end{array} \right.$
\end{note}
\subsubsection{Hiệu và phần bù của hai tập hợp}
${A\backslash B=\{x| x\in A\,\,\text{và}\,\,x\notin B\}.}$
\begin{center}

		\fillOnlyA
	\end{venndiagram2sets}
\end{center}
\noindent{$\bullet$ Phép lấy phần bù:} Cho $A\subset E$. Phần bù của $A$ trong $E$ là $\mathcal{C}_EA=E\backslash A$.
\subsection{Các dạng toán}
\begin{dang}{Tìm giao và hợp của các tập hợp}
	Dựa vào định nghĩa giao và hợp của hai tập hợp để tìm kết quả.
\end{dang}
\begin{vd}%[Huỳnh Thanh Tiến]%[0D1B3]
	Cho hai tập hợp $A=\{1;2;3;5;7\}$ và  $B=\{n\in \mathbb{N} |\, n \text{ là ước số của } 12\}$. Tìm $A\cap B$  và $A\cup B$.
	\loigiai{
		Ta có: $B=\{1;2;3;4;6;12\}$.
		Vậy: $A\cap B=\{ 1;2;3\}$ và $A\cup B=\{ 1;2;3;4;5;6;7;12\}$.
	}
\end{vd}

\begin{vd}%[Huỳnh Thanh Tiến]%[0D1K3]
	Cho tập hợp $B=\{ x\in \mathbb{Z}|\, -4< x \le 4 \}$ và $C=\{ x\in \mathbb{Z}|\, x\le a\}$.
	Tìm số nguyên $a$ để tập hợp $B\cap C=\varnothing $.
	\loigiai{
	Ta có $B=\{-3;-2;-1;0;1;2;3;4\}$, $C=\{\ldots,a-1,a\}$.\\
		Để $B\cap C=\varnothing $ thì  $a\le -4$.
	}
\end{vd}

\begin{vd}%[Huỳnh Thanh Tiến]%[0D1B3]
	Chứng minh rằng nếu $A\subset B$ thì $A\cap B=A$.
	\loigiai{
		\begin{itemize}
			\item $x\in A\cap B\Rightarrow x\in A$, suy ra $A\cap B\subset A$.
			\item $x\in A\Rightarrow \begin{cases}
			x\in A\\
			x\in B\ \left(\text{do}\ A\subset B\right)
			\end{cases}\Rightarrow x\in A\cap B$, suy ra $A\subset A\cap B$. \\
			Vậy $A\cap B=A$. 
		\end{itemize}
	}
\end{vd}

\begin{vd}%[Huỳnh Thanh Tiến]%[0D1B3]
	Cho $A$ là tập hợp học sinh lớp $12$ của trường Buôn Ma Thuột và $B$ là tập hợp học sinh của trường Buôn Ma Thuột dự kiến sẽ lựa chọn thi khối $A$ vào các trường đại học. Hãy mô tả các học sinh thuộc tập hợp sau
	\begin{enumerate}[a)]
		\begin{multicols}{2}
			\item $A\cap B$.
			\item $A\cup B$.
		\end{multicols}
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item $A\cap B$ là tập hợp các học sinh lớp 12 thi khối $A$ của trường Buôn Ma Thuột.
			\item $A\cup B$ là tập hợp các học sinh hoặc lớp 12 hoặc chọn thi khối A của trường Buôn Ma Thuột. 
		\end{enumerate}
	}
\end{vd}


\begin{vd}%[Huỳnh Thanh Tiến]%[0D1B3]
	Cho hai tập hợp $A, B$ biết : $A=\{a;b\}, B=\{a;b;c;d\}$. Tìm tập hợp $X$ sao cho $A\cup X=B$.
	\loigiai{
		$X=\{c;d\}$; $\{b;c;d\}$; $\{a;c;d\}$; $\{a;b;c;d\}$.
	}
\end{vd}
\begin{vd}%[Huỳnh Thanh Tiến]%[0D1B3]

	\loigiai{
		Ta có $A\cap B=\{x\in\mathbb{N}|\, x\text { là bội của }3 \text{ và bội của }7 \}= \{x\in\mathbb{N}|\, x\text {  là bội của  }21\} $.
	}
\end{vd}

\begin{bt}%[Huỳnh Thanh Tiến]%[0D1B3]

\end{bt}

\begin{bt}%[Huỳnh Thanh Tiến]%[0D1B3]

\begin{bt}%[Huỳnh Thanh Tiến]%[0D1K3]

	\loigiai{Ta có: $A=\left\{0; 2; -\dfrac{1}{2}\right\}, B=\left\{2; 3; 4; 5\right\}$ nên $A\bigcap B=\left\{2\right\}$.}
\end{bt}

\begin{bt}%[Huỳnh Thanh Tiến]%[0D1K3]
	Cho $a$ là số nguyên. Tìm  $a$ để  giao của hai tập hợp $$A=\{x\in\mathbb{Z}\,\big|\, x\le a\}, \,\, B=\left\{x\in\mathbb{Z}\,\big|\, x>\dfrac{3a-4}{2}\right\}$$
	bằng rỗng. 
	\loigiai{Ta có $A\cap B=\varnothing \Leftrightarrow a\le \dfrac{3a-4}{2}\Leftrightarrow a\ge 4$. 
	}
\end{bt}

\begin{bt}%[Huỳnh Thanh Tiến]%[0D1G3]
	Cho hai tập hợp bất kì $A,B.$ Chứng minh rằng $A\cup B=A\cap B\Leftrightarrow A=B.$
	\loigiai{
		\begin{itemize}
			\item Nếu $A=B$ thì $A\cap B=A, A\cup B=A$ nên $A\cup B=A\cap B.$
			\item Ngược lại, giả sử $A\cup B=A\cap B.$ Lấy một phần tử bất kì $x\in A$ ta suy ra $x\in A\cup B.$ Vì $A\cup B=A\cap B$ nên $x\in A\cap B.$ Từ đó suy ra $x\in B$ nên $A\subset B.$ Tương tự ta cũng có $B\subset A.$ Vậy $A=B.$
		\end{itemize}
	}
\end{bt}

\begin{bt}%[Huỳnh Thanh Tiến]%[0D1K3]
	Cho các tập hợp $A=\{x\in\mathbb{N}|x<8\}$ và $B=\{x\in\mathbb{Z}| -3\leq x\leq 5\}$. Tìm $A\cap B$; $A\cup B$.
	\loigiai{

	}
\end{bt}



\begin{bt}%[Huỳnh Thanh Tiến]%[0D1K3]

\begin{bt}%[Huỳnh Thanh Tiến]%[0D1K3]
	Cho các tập hợp $A=\{x \in \mathbb{Z}\big| |x-1|<4\}$, $B=\{x \in \mathbb{Z}\big| |x-1|>2\}$. Tìm $A \cap B$.
	\loigiai{
		Ta có $|x-1|<4 \Leftrightarrow -4<x-1<4 \Leftrightarrow -3<x<5$, $A=\{-2;-1;0;1;2;3;4\}$. \\
		Lại có $|x-1|>2 \Leftrightarrow x<-1\vee x>3$, $B=\{\ldots;-3;-2;4;5;6;\ldots\}$ nên $A \cap B=\{-2;4\}$.

\begin{bt}%[Huỳnh Thanh Tiến]%[0D1G3]
	Cho các tập hợp $A=\{x\in\mathbb{Z}\,|\, 2m-1<x<2m+3\}$, $B=\{x\in\mathbb{Z}\,\big|\, |x|<2\}$. Tìm $m$ để $A\cap B=\varnothing$.
	\loigiai{

		$A\cap B=\varnothing \Leftrightarrow \left[ \begin{array}{l}
		2m+2 \leq -2 \\
		2m \geq 2
		\end{array} \right. \Leftrightarrow \left[ \begin{array}{l}
		m \leq -2 \\
		m\geq 1
		\end{array} \right.$
	}
\end{bt}

\begin{bt}%[Huỳnh Thanh Tiến]%[0D1G3]
	Cho tập $S=\left\{1; 2; 3; 4; 5; 6\right\}$. Tìm các tập con $A, B$ của  tập $S$ sao cho $A\cup B=\left\{1; 2; 3; 4\right\}$  và $A\cap B=\left\{1; 2\right\}$. 
	\loigiai{
		\begin{itemize}
			\item $A$  có hai phần tử $A=\left\{1; 2\right\}\Rightarrow B=\left\{1; 2; 3; 4\right\}$.
			\item $A$  có ba phần tử $A=\left\{1; 2; 3\right\}\Rightarrow B=\left\{1; 2; 4\right\}$. 
			\item 	$A$  có ba phần tử $A=\left\{1; 2; 4\right\}\Rightarrow  B=\left\{1; 2; 3\right\}$.
			\item 	$A$  có bốn phần tử $A=\left\{1; 2; 3; 4\right\}\Rightarrow B=\left\{1; 2\right\}$ . 
		\end{itemize}
		Vậy ta có $4$ cặp tập $A$, $B$ thỏa mãn yêu cầu bài toán.
	}
\end{bt}
\begin{bt}%[Huỳnh Thanh Tiến]%[0D1G3]
	Cho tập hợp $A=\{x\in\mathbb{R}\,|\,  x^2-4x+m+2=0\}$ và tập hợp $B=\{1;2\}$. Tìm $m$ để $A\cap B=\varnothing$. 
	\loigiai{
		\begin{itemize}
			\item TH1: $A=\varnothing$ tương đương pt: $x^2-4x+m+2=0$ vô nghiệm, tức là $\Delta'<0\Leftrightarrow m>2$.
			\item TH2: $A\ne \varnothing$ tương đương pt: $x^2-4x+m+2=0$ có 2 nghiệm khác $1,2$ $\Leftrightarrow m\ne 1; m\ne 2; m\le 2$.
			\item Vậy kết hợp lại ta có $m\ne 1; m\ne 2$.
		\end{itemize}
	}
\end{bt}
\begin{dang}{Hiệu và phần bù của hai tập hợp}
	Dựa vào định nghĩa hiệu và phần bù của hai tập hợp để tìm kết quả.

\begin{note}
	Chú ý
	\begin{itemize}
		\item Nếu $A \subset B$ thì $B \backslash A=C_B A$.
		\item Nếu $A= \varnothing$ thì $A \backslash B = \varnothing$ với mọi tập hợp $B$.
	\end{itemize}
\end{note}

\begin{vd}%[Lê Phước Thật]%[0D1B3]
	Cho $A= \left\{ 1,2,3,4,5 \right\}$ và $B= \left\{ 1,3,5,7 \right\}$. Tìm các tập hợp $A \backslash B, B \backslash A$.
	\loigiai{
		Các phần tử $2,4$ thuộc tập hợp $A$ nhưng không thuộc tập hợp $B$ nên $A \backslash B = \{2,4\}$.\\
		Chỉ có phần tử $7$ thuộc tập hợp $B$ nhưng không thuộc tập hợp $A$ nên $B \backslash A = \{7\}$
	}
\end{vd}

\begin{vd}%[Lê Phước Thật]%[0D1B3]
	Cho $A$ là tập hợp các tự nhiên lẻ. Tìm phần bù của $A$ trong tập $\mathbb{N}$ các số tự nhiên.
	\loigiai{
		Các số tự nhiên chẵn thuộc tập hợp $\mathbb{N}$ nhưng không thuộc tập hợp $A$ nên phần bù của $A$ trong $\mathbb{N}$ là tập hợp các số tự nhiên chẵn. Do đó $C_{\mathbb{N}} A= \{ 2k/ k \in \mathbb{N} \}$.
	}
\end{vd}

\begin{vd}%[Lê Phước Thật]%[0D1K3]
	Chứng minh rằng $A \backslash B = \varnothing$ thì $A \subset B$.
	\loigiai{
		Lấy $x \in A$. Nếu $x \notin B$ thì $x \in A \backslash B$ (mâu thuẫn). Do đó $x \in B$. Vậy $A \subset B$.
	}
\end{vd}

\begin{vd}%[Lê Phước Thật]%[0D1K3]
	Cho các tập hợp $A=\{4,5\}$ và $B= \{n \in N | n \le a\}$ với $a$ là số tự nhiên. Tìm $a$ sao cho $A \backslash B =A$.
	\loigiai{
		Ta có $B= \{0,1,\cdots,a\}$. Để $A \backslash B =A$ thì các phần tử của $A$ không thuộc $B$. Suy ra $a \le 3$. Vậy $a \in \{0,1,2,3\}$.
	}
\end{vd}

\begin{vd}%[Lê Phước Thật]%[0D1G3]
	Cho hai tập hợp $A, B$. Biết $A \backslash B = \{1, 2\}, B \backslash A = \{3\}$ và $B= \{3,4,5\}$. Tìm tập hợp $A$.
	\loigiai{
		Ta có $A \backslash B = \{1, 2\}$ nên $1,2 \in A$.\\
		Mà $B \backslash A = \{3\}$ nên $3 \notin A$ và $4,5 \in A$.\\
		Suy ra $A= \{1,2,4,5\}$.
	}
\end{vd}

\begin{bt}%[Lê Phước Thật]%[0D1Y3]
	Cho $A$ là tập hợp các học sinh của một lớp và $B$ là tập hợp các học sinh giỏi Toán của lớp. Hãy mô tả tập hợp $C_A B$.
	\loigiai{
		$C_A B$ là tập hợp các học sinh không giỏi Toán của lớp.
	}
\end{bt}

\begin{bt}%[Lê Phước Thật]%[0D1B3]
	Cho $A$ là tập hợp các ước nguyên dương của $12$ và $B$ là tập hợp các ước nguyên dương của $18$. Tìm các tập hợp $A \backslash B$ và $B \backslash A$.
	\loigiai{
		Ta có $A= \{1,2,3,4,6,12\}$ và $B= \{1,2,3,6,9,18\}$ nên $A \backslash B = \{4,12\}, B \backslash A = \{9,18\}$.
	}
\end{bt}

\begin{bt}%[Lê Phước Thật]%[0D1K3]
	Chứng minh rằng $A \backslash B = B \backslash A$ thì $A=B$.
	\loigiai{
		Lấy $x \in A \backslash B = B \backslash A$ thì $x \in A, x \notin B$ và $x \in B, x \notin A$. Suy ra $A \backslash B = B \backslash A= \varnothing$.\\
		Suy ra $A \subset B$ và $B \subset A$. Vậy $A=B$.
	}
\end{bt}

\begin{bt}%[Lê Phước Thật]%[0D1K3]
	Cho hai tập hợp $A, B$. Biết $A \backslash B = \{a,b,c\}, B \backslash A = \{d,e\}$ và $B= \{d,e,f\}$. Tìm tập hợp $A$.
	\loigiai{
		$A=\{a,b,c,f\}$.
	}
\end{bt}

\begin{bt}%[Lê Phước Thật]%[0D1G3]
	Cho các tập hợp $A=\{n \in N | 2<n \le 7 \}$ và $B= \{n \in N | n \le a\}$ với $a$ là số tự nhiên. Tìm $a$ sao cho:
	\begin{enumerate}
		\item $A \backslash B =A$.
		\item $A \backslash B  = \varnothing$.
	\end{enumerate}
	\loigiai{

ta có $A$ và $B$ phải có chung đúng hai phần tử $1$ và $2$. Từ  
$A \cup B  = \left\{1; 2; 3; 4 \right\}$ 
suy ra hai phần tử $3$ và $4$ phải thuộc một và chỉ một trong hai tập $A$ và $B$. Do đó có bốn kết quả sau đây:
\[
\left\{ \begin{array}{l}
A = \left\{ 1;2;3 \right\} \\ 
B = \left\{ 1;2;4 \right\}, \\ 
\end{array} \right. \left\{ \begin{array}{l}
A = \left\{ 1;2;4 \right\} \\ 
B = \left\{ 1;2;3 \right\}, \\ 
\end{array} \right. \left\{ \begin{array}{l}
A = \left\{ 1;2;3;4 \right\} \\ 
B = \left\{ 1;2 \right\}, \\ 
\end{array} \right. \left\{ \begin{array}{l}
A = \left\{ 1;2 \right\} \\ 
B = \left\{ 1;2;3;4 \right\}. \\ 
\end{array} \right.
\]
\item  Vì $C \cup (A \cap B) = A \cup B$  mà $A \cup B  =  \left\{ 1; 2; 3; 4 \right\},$
$A \cap B  = \left\{ 1; 2 \right\}$  nên $3, 4\in C$. Do đó các tập $C$ thỏa mãn yêu cầu bài toán là:
\[
\left\{ 3;\,\,4 \right\},\left\{ 1;\,3;\,\,4 \right\},\left\{ \,2;\,\,3;\,\,4 \right\},\left\{ 1;\,\,2;\,\,3;\,\,4 \right\}.
\]
\end{enumerate}
}
\end{bt}
\begin{bt}%[Nguyễn Tài Chung]%[0D1G3]
	Xét $X$ và $Y$ là hai tập hợp con của tập hợp 
	$\{ 1;2;3;4;5;6;7;8;9\} $ và thỏa mãn ba  điều kiện:
	{\allowdisplaybreaks
		\begin{align*}
		&(1)\quad  X \cap Y = \{ 4;6;9\}.\\
		&(2)\quad X \cup \{ 3;4;5\}  = \{ 1;3;4;5;6;8;9\}.\\
		&(3)\quad Y \cup \{ 4;8\}  = \{ 2;3;4;5;6;7;8;9\}.
		\end{align*}}
	\begin{itemize}
		\item [$a)$] Hãy chỉ ra rằng từ điều kiện $(1)$ và $(2)$
		ta suy ra $1 \in X$ và $1 \notin Y$, $8 \in X$ và $8 \notin Y$, $7 \notin X$.
		\item [$b)$] Xác định các tập hợp $X$ và $Y$ mà thỏa mãn các điều kiện $(1),$ $(2)$ và $(3)$.
	\end{itemize}
\loigiai{
\begin{enumerate}[$a)$]
	\item  Vì $1 \in X \cup \{ 3;4;5\} $ nên $1 \in X$ và vì 
$1 \notin X \cap Y$ nên $1 \notin Y$. Tương tự ta có 
$8 \in X$ và $8 \notin Y$. Từ (3) suy ra $7 \in Y$, do đó $7 \notin X$ vì nếu $7 \in X$ thì mâu thuẫn với (1).
\item Ta có:
\begin{itemize}
	\item  $1 \in X$ và $1 \notin Y$;
	\item $2 \notin X$ (do (2)) và $2 \in Y$ (do (3));
	\item $3 \in Y$ (do (3)) và $3 \notin X$ (do (1)); 
	\item $4 \in X$  và $4 \in Y$ (do (1)); 
	\item $5 \in Y$ (do (3)) và $5 \notin X$ (do (1));
	\item $6 \in X$  và $6 \in Y$ (do (1));
	\item $7 \in Y$ (do (3)) và $7 \notin X$ (do (1));
	\item $8 \in X$  và $8 \notin Y$ (do câu $a)$);
	\item $6 \in X$  và $6 \in Y$ (do (1)).
\end{itemize}

$$X = \{ 1;4;6;8;9\} ,\,\,Y = \{ 2;3;4;5;6;7;9\}.$$
\end{enumerate}
}

\begin{bt}%[Huỳnh Thanh Tiến]%[0D1G3]
	Cho các tập hợp $A = \left\{x \in \mathbb{R}\big| x^2 +x - m = 0\right\}$, 
	$B = \left\{x \in \mathbb{R}\big| x^2 - mx +1 = 0\right\}$ ($m$ là tham số thực). Tìm tất cả các giá trị của $m$ để $A\cap B \ne \varnothing $
	\loigiai{
		Vì $A\cap B \ne \varnothing $ nên tồn tại $a\in A\cap B $. Khi đó 
		$$\begin{cases}
		a^2+a-m=0\\
		a^2-ma+1=0
		\end{cases}\Rightarrow (1+m)a-(1+m)=0\Rightarrow \left[\begin{array}{l} m=-1 \\ a=1 \end{array} \right.$$
		\begin{itemize}
			\item Nếu $m=-1$ thử lại thấy $B=\varnothing$ nên không thỏa.
			\item  Nếu $a=1$ thay vào tập $A$ tìm được $m=2.$ Thử lại khi $m=2$ thấy $A\cap B=\{1\}$. 
		\end{itemize}
		Vậy $m=2.$
	}
\end{bt}
\begin{bt}%[Nguyễn Tài Chung]%[0D1G3]
	Cho $3$ tập hợp:
	{\allowdisplaybreaks
		\begin{align*}
		A &= \left\{ x|x = 3n - 2,\,n \in \mathbb{N}^* \right\}\\
		B &= \left\{ x|x = 1003 - 2m,\,m \in \mathbb{N}^* \right\}\\
		C &= \left\{ x|x = 6p + 1,\,p \in \mathbb{Z},\,0 \le p \le 166 \right\}.
		\end{align*}}Chứng minh rằng $A \cap B = C$.
\end{bt}
\noindent{\bf Giải.} Cần chứng minh $A \cap B \subset C$ và $C \subset A \cap B$.
\begin{itemize}
	\item Giả sử $x \in A \cap B$, cần chỉ ra $x\in C$. Thực vậy, nếu 
	$x \in A \cap B$ thì $x\in A$ và $x\in B$, tức là tồn tại các số nguyên dương $m, n$ sao cho:
	$$x = 3n - 2 = 1003 - 2m.$$
	Khi đó:  $\displaystyle m = \frac{1005 - 3n}{2} \Leftrightarrow m = 502 - n - \frac{n - 1}{2}$. Vì $m, n$ là những số nguyên
	dương nên suy ra $\dfrac{n - 1}{2} = p \in \mathbb{Z}$. Từ đó $n = 2p + 1$ và
	$$m = 502 - (2p + 1) - p = 501 - 3p.$$
	Vì $m, n$ là những số nguyên dương nên
	$$\left\{ \begin{array}{l}
	2p + 1 \ge 1\\
	501 - 3p \ge 1
	\end{array} \right. \Rightarrow \left\{ \begin{array}{l}
	p \ge 0\\
	p \le \dfrac{{500}}{3}
	\end{array} \right. \Rightarrow 0 \le p \le 166.$$
	Nhưng $x = 3n - 2 = 3(2p + 1) - 2 = 6p + 1$, suy ra 
	\[x \in C \Rightarrow A \cap B \subset C.\tag{1}\]
	\item Chứng minh $C \subset A \cap B$. Giả sử $x\in C$, cần chứng minh $x \in A \cap B$. Thực vậy, nếu $x\in C$ thì tồn tại $p\in\mathbb{Z}$, 
	$0 \le p \le 166$, sao cho $x=6p+1$. Ta sẽ chỉ ra rằng $x$ có thể được viết dưới dạng $x=3n-2$, $n\in\mathbb{N}^*$.
	Ta có 
	$$x = 6p + 1 = (6p + 3) - 2 = 3(2p + 1) - 2 = 3n - 2,$$
	với $n=2p+1\in \mathbb{N}^*$, suy ra $x\in A$. Ta còn phải chứng minh $x\in B$.
	$$x = 6p + 1 = 1003 - (1002 - 6p) = 1003 - 2(501 - 3p) = 1003 - 2m,$$
	với $m = 501 - 3p$. Ta có: 
	$$0 \le p \le 166 \Rightarrow \,0 \le 3p \le 498 \Rightarrow 501 - 3p \ge 3 \Rightarrow m = 501 - 3p \in \mathbb{N}^*.$$
	Như vậy $x\in B$. Từ $x\in A$ và $x\in B$ suy ra 
	\[x \in A \cap B\Leftarrow C \subset A \cap B.\tag{2}\]
\end{itemize}
Từ (1) và (2) suy ra $A \cap B = C$, điều phải chứng minh.
