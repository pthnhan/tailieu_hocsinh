\section{CHỨNG MINH ĐẲNG THỨC ĐẠI SỐ}
\subsection{Tóm tắt lí thuyết}
\begin{enumerate}
\item Đẳng thức có dạng \[A=B\tag{1}\]
trong đó $A$, $B$ là các biểu thức đại số.
\item Muốn chứng minh đẳng thức $(1)$, ta thường dùng một số cách sau
\begin{itemize}
\item Biến đổi từ $A$ về $B$ hoặc từ $B$ về $A$.
\item Chứng minh $A=C$ và $B=C$. Từ đó suy ra $A=B$.
\end{itemize}
\end{enumerate}
\subsection{Một số ví dụ}
\begin{vd}
Cho $a$, $b$, $c$ thỏa mãn $a+b+c=0$ và $M=a(a+b)(a+c)$, $N=b(b+c)(b+a)$, $P=c(c+a)(c+b)$. Chứng minh $M=N=P$.
\loigiai
{
Vì $a+b+c=0$ nên ta có
\[\begin{cases}
M=a(a+b)(a+c)=a\cdot (-c)\cdot (-b)=abc\\
N=b(b+c)(b+a)=b\cdot (-a)\cdot (-c)=abc\\
P=c(c+a)(c+b)=c\cdot (-b)\cdot (-a)=abc
\end{cases}\Rightarrow M=N=P.\]
}
\end{vd}
\begin{vd}
Với $a$, $b$, $x$, $y$ là các số thực bất kì, chứng minh rằng
\[(a^2+b^2)(x^2+y^2)=(ax+by)^2+(ay-bx)^2.\]
Từ đó suy ra $(a^2+b^2)(x^2+y^2)\ge (ax+by)^2.$
\loigiai
{
Ta có
\begin{align*}
(ax+by)^2+(ay-bx)^2&=a^2x^2+2axby+b^2y^2+a^2y^2-2aybx+b^2x^2\\ 
&=(a^2x^2+a^2y^2)+(b^2y^2+b^2x^2)\\
&=a^2(x^2+y^2)+b^2(x^2+y^2)\\
&=(a^2+b^2)(x^2+y^2).
\end{align*}
Do đó \((a^2+b^2)(x^2+y^2)=(ax+by)^2+(ay-bx)^2.\)\hfill$(1)$\\
Vì $(ay-bx)^2\ge 0$ với mọi $a$, $b$, $x$, $y$ nên từ $(1)$ ta có
\[(a^2+b^2)(x^2+y^2)=(ax+by)^2+(ay-bx)^2\ge (ax+by)^2.\]
Đẳng thức xảy ra khi và chỉ khi $ay=bx$.
}
\end{vd}
\begin{vd}
Với $a$, $b$, $x$, $y$ là các số thực bất kì, chứng minh rằng
\[m(a^2+b^2)+nab=\dfrac{2m+n}{4}(a+b)^2+\dfrac{2m-n}{4}(a-b)^2.\]
Từ đó suy ra với $a,b\ge 0$ thì  $\sqrt{2a^2+ab+2b^2}\ge \dfrac{\sqrt{5}}{2}(a+b).$
\loigiai{
Ta có
\begin{align*}
&\dfrac{2m+n}{4}(a+b)^2+\dfrac{2m-n}{4}(a-b)^2\\
=\;&\dfrac{2m+n}{4}(a^2+2ab+b^2)+\dfrac{2m-n}{4}(a^2-2ab+b^2)\\ 
=\;&\left(\dfrac{2m+n}{4}+\dfrac{2m-n}{4}\right)a^2+\left(\dfrac{2m+n}{4}-\dfrac{2m-n}{4}\right)2ab+\left(\dfrac{2m+n}{4}+\dfrac{2m-n}{4}\right)b^2\\
=\;&m(a^2+b^2)+nab.
\end{align*}
Do đó \(m(a^2+b^2)+nab=\dfrac{2m+n}{4}(a+b)^2+\dfrac{2m-n}{4}(a-b)^2.\)\hfill$(1)$\\
Từ $(1)$, thay $m=2$ và $n=1$ ta được $2a^2+ab+2b^2=\dfrac{5}{4}(a+b)^2+\dfrac{3}{4}(a-b)^2.$\hfill$(2)$\\
Vì $\dfrac{3}{4}(a-b)^2\ge 0$ với mọi $a$, $b$ nên từ $(2)$ ta có
\[2a^2+ab+2b^2=\dfrac{5}{4}(a+b)^2+\dfrac{3}{4}(a-b)^2\ge \dfrac{5}{4}(a+b)^2.\]
Do đó $\sqrt{2a^2+ab+2b^2}\ge \dfrac{\sqrt{5}}{2}(a+b).$ Dấu bằng xảy ra khi và chỉ khi $a=b$.
}
\end{vd}

\begin{vd}
Với $a$, $b$, $c$, $x$, $y$, $z$ là các số thực bất kì, chứng minh rằng
\[(a^2+b^2+c^2)(x^2+y^2+z^2)=(ax+by+cz)^2+(ay-bx)^2+(bz-cy)^2+(cx-az)^2.\]
Từ đó suy ra $(a^2+b^2+c^2)(x^2+y^2+z^2)\ge (ax+by+cz)^2.$
\loigiai{
Ta có
\begin{align*}
&(ax+by+cz)^2+(ay-bx)^2+(bz-cy)^2+(cx-az)^2\\
=\;&(a^2x^2+b^2y^2+c^2z^2+2axby+2bycz+2czax)+(a^2y^2-2aybx+b^2x^2)\\
&\qquad\qquad\qquad +(b^2z^2-2bzcy+c^2y^2)+(c^2x^2-2cxaz+a^2z^2)\\ 
=\;&(a^2x^2+a^2y^2+a^2z^2)+(b^2x^2+b^2y^2+b^2z^2)+(c^2x^2+c^2y^2+c^2z^2)\\
=\;&a^2(x^2+y^2+z^2)+b^2(x^2+y^2+z^2)+c^2(x^2+y^2+z^2)\\
=\;&(a^2+b^2+c^2)(x^2+y^2+z^2).
\end{align*}
Do đó \((a^2+b^2+c^2)(x^2+y^2+z^2)=(ax+by+cz)^2+(ay-bx)^2+(bz-cy)^2+(cx-az)^2.\)\hfill$(1)$\\
Vì $(ay-bx)^2\ge 0$, $(bz-cy)^2\ge 0$, $(cx-az)^2\ge 0$ với mọi $a$, $b$, $c$, $x$, $y$, $z$ nên từ $(1)$ ta có
\begin{align*}
(a^2+b^2+c^2)(x^2+y^2+z^2)&=(ax+by+cz)^2+(ay-bx)^2+(bz-cy)^2+(cx-az)^2\\ 
&\ge (ax+by+cz)^2
\end{align*}
Dấu bằng xảy ra khi và chỉ khi $ay=bx$, $bz=cy$ và $cx=az$.
}
\end{vd}

\begin{vd}
Cho các số $a$, $b$, $c$ thỏa mãn $a+b+c=0$ và $a^2+b^2+c^2=14$. Chứng minh rằng $a^4+b^4+c^4=98$.
\loigiai{
Ta có
\begin{align*}
a+b+c=0&\Rightarrow (a+b+c)^2=0\\ 
&\Rightarrow a^2+b^2+c^2+2(ab+bc+ca)=0\\ 
&\Rightarrow ab+bc+ca=-7\Rightarrow (ab+bc+ca)^2=49\\ 
&\Rightarrow a^2b^2+b^2c^2+c^2a^2+2abc(a+b+c)=49\\ 
&\Rightarrow a^2b^2+b^2c^2+c^2a^2=49.\tag{1}
\end{align*}
Mặt khác
\begin{align*}
a^2+b^2+c^2=14&\Rightarrow \left(a^2+b^2+c^2\right)^2=196\\ 
&\Rightarrow a^4+b^4+c^4+2(a^2b^2+b^2c^2+c^2a^2)=196.\tag{2}
\end{align*}
Từ $(1)$ và $(2)$ suy ra $a^4+b^4+c^4=98$.
}
\end{vd}
\begin{vd}
Chứng minh rằng $a^4+b^4+(a+b)^4=2(a^2+ab+b^2)^2.$
\loigiai{
Biến đổi vế trái của đẳng thức ta được
\begin{align*}
a^4+b^4+(a+b)^4&=a^4+b^4+\left[(a+b)^2\right]^2\\ 
&=a^4+b^4+\left(a^2+2ab+b^2\right)^2\\ 
&=a^4+b^4+a^4+4a^2b^2+b^4+4a^3b+2a^2b^2+4ab^3\\ 
&=2\left(a^4+a^2b^2+b^4+2a^3b+2a^2b^2+2ab^3\right)\\ 
&=2(a^2+ab+b^2)^2.
\end{align*}
}
\end{vd}

\begin{vd}
Cho hai số $x$, $y$ thỏa mãn $x^3-3xy^2=5$ và $y^3-3x^2y=10.$ Chứng minh rằng $x^2+y^2=5.$
\loigiai{
Ta có
\begin{align*}
&x^3-3xy^2=5\Rightarrow \left(x^3-3xy^2\right)^2=25\Rightarrow x^6-6x^4y^2+9x^2y^4=25,\tag{1}\\ 
&y^3-3x^2y=10\Rightarrow \left(y^3-3x^2y\right)^2=100\Rightarrow y^6-6x^2y^4+9x^4y^2=100.\tag{2}
\end{align*}
Cộng $(1)$ và $(2)$ theo vế với vế ta được
\[\left(x^2+y^2\right)^3=125\Leftrightarrow x^2+y^2=5.\]
}
\end{vd}

\begin{vd}
Cho $x+y+z=0$. Chứng minh rằng $2(x^5+y^5+z^5)=5xyz(x^2+y^2+z^2).$
\loigiai{
Vì $x+y+z=0$ nên $x+y=-z$, $y+z=-x$, $z+x=-y$. Khi đó
\begin{align*}
x^3+y^3+z^3=3xyz&\Leftrightarrow \left(x^3+y^3+z^3\right)\left(x^2+y^2+z^2\right)=3xyz\left(x^2+y^2+z^2\right)\\ 
&\Leftrightarrow x^5+y^5+z^5-xyz(xy+yz+zx)=3xyz\left(x^2+y^2+z^2\right)\\
&\Leftrightarrow 2\left(x^5+y^5+z^5\right)-2xyz(xy+yz+zx)=6xyz\left(x^2+y^2+z^2\right).\tag{1}
\end{align*}
Ta có
\begin{align*}
x+y+z=0&\Leftrightarrow \left(x+y+z\right)^2=0\\ 
&\Leftrightarrow x^2+y^2+z^2=-2(xy+yz+zx)\\
&\Leftrightarrow xyz\left(x^2+y^2+z^2\right)=-2xyz(xy+yz+zx).\tag{2}
\end{align*}
Thay $(2)$ vào $(1)$ ta được $2(x^5+y^5+z^5)=5xyz(x^2+y^2+z^2).$
}
\end{vd}

\begin{vd}
Cho các số thực $a$, $b$, $c$ thỏa mãn $a^3-b^2-b=b^3-c^2-c=c^3-a^2-a=\dfrac{1}{3}$. Chứng minh rằng $a=b=c$.
\loigiai{
Ta có
\[a^3-b^2-b=\dfrac{1}{3}\Rightarrow a^3=b^2+b+\dfrac{1}{3}=\left(b+\dfrac{1}{2}\right)^2+\dfrac{1}{12}>0\Rightarrow a>0.\]
Tương tự $b>0$, $c>0$.\\
Nếu $a\ge b$ thì
\begin{align*}
&\Rightarrow \left(b+\dfrac{1}{2}\right)^2+\dfrac{1}{12}\ge \left(c+\dfrac{1}{2}\right)^2+\dfrac{1}{12}\Rightarrow b\ge c\Rightarrow b^3\ge c^3\\ 
&\Rightarrow \left(c+\dfrac{1}{2}\right)^2+\dfrac{1}{12}\ge \left(a+\dfrac{1}{2}\right)^2+\dfrac{1}{12}\Rightarrow c\ge a.
\end{align*}
Từ đó suy ra $a=b=c$.
}
\end{vd}

\begin{vd}
Cho bốn số dương $a$, $b$, $c$, $d$ thỏa mãn $a^4+b^4+c^4+d^4=4abcd.$ Chứng minh rằng $a=b=c=d.$
\loigiai{
Ta có
\begin{align*}
a^4+b^4+c^4+d^4=4abcd&\Leftrightarrow \left(a^2-b^2\right)^2+\left(c^2-d^2\right)^2+2\left(ab-cd\right)^2\\ 
&\Leftrightarrow a=b=c=d.
\end{align*}
}
\end{vd}

\begin{vd}
Cho đa thức $f(x)=x^4+ax^3+bx^2+cx+d$. Biết $f(1)=10$, $f(2)=20$, $f(3)=30$. Chứng minh rằng $f(-8)+f(12)=19840.$
\loigiai{
Xét đa thức $g(x)=f(x)-10x$. Ta có 
\begin{align*}
&g(1)=f(1)-10=10-10=0;\\ 
&g(2)=f(2)-20=20-20=0;\\ 
&g(3)=f(3)-30=30-30=0.
\end{align*}
Suy ra $g(x)$ nhận $x=1$, $x=2$, $x=3$ là các nghiệm và $g(x)$ có dạng đa thức bậc $4$ nên \[g(x)=(x-1)(x-2)(x-3)(x-m).\]
Do đó $f(x)=g(x)+10x=(x-1)(x-2)(x-3)(x-m)+10x.$ Từ đó suy ra\[f(-8)+f(12)=9\cdot 10\cdot 11\cdot (12-m+8+m)+40=19840.\]
}
\end{vd}

\begin{vd}[Đề thi học sinh giỏi lớp 9 tỉnh Nam Định năm học 2011-2012] Cho các số thực $a, b, c$ khác nhau từng đôi một vào thỏa mãn điều kiện
$a^{2}-b=b^{2}-c=c^{2}-a.$ 
Chứng minh rằng $(a+b+1)(b+c+1)(c+a+1)=-1.$
\loigiai{
Ta có
\begin{align*}
a^{2}-b=b^{2}-c&\Rightarrow a^2-b^2=b-c\Rightarrow (a+b)(a-b)=b-c\\ 
&\Rightarrow a+b=\dfrac{b-c}{a-b}\quad\text{(do $a\ne b$)}\\
&\Rightarrow a+b+1=\dfrac{b-c}{a-b}+1=\dfrac{b-c+a-b}{a-b}=\dfrac{a-c}{a-b}.
\end{align*}
Tương tự ta có $b+c+1=\dfrac{b-a}{b-c}$; $c+a+1=\dfrac{c-b}{c-a}$. Do đó
\[(a+b+1)(b+c+1)(c+a+1)=\dfrac{a-c}{a-b}\cdot\dfrac{b-a}{b-c}\cdot\dfrac{c-b}{c-a}=-1.\]
}
\end{vd}

\begin{vd}[Đề thi vào lớp 10 THPT chuyên ĐHSP Hà Nội năm học 2012-2013] Cho  $a,b,c$ đôi một khác nhau thỏa mãn $a^2(b+c)=b^2(c+a)=2012.$ Chứng minh rằng $M=c^2(a+b)=2012$.
\loigiai{
Vì $a^2(b+c)=b^2(c+a)=2012$ nên $abc\ne 0$.
Ta có
\begin{align*}
a^2(b+c)=b^2(c+a)&\Rightarrow a^2b+a^2c=b^2c+b^2a\\ 
&\Rightarrow \left(a^2b-b^2a\right)+\left(a^2c-b^2c\right)=0\\ 
&\Rightarrow ab\left(a-b\right)+\left(a-b\right)(ac+bc)=0\\ 
&\Rightarrow \left(a-b\right)(ab+bc+ca)=0\\ 
&\Rightarrow ab+bc+ca=0\quad\text{(do $a\ne b$)}\\ 
&\Rightarrow a(b+c)=-bc\Rightarrow a^2(b+c)=-abc\\
&\Rightarrow -abc=2012.
\end{align*}
Mặt khác, từ $ab+bc+ca=0$ ta có
\[c(a+b)=-ab\Rightarrow c^2(a+b)=-abc\Rightarrow M=c^2(a+b)=2012.\]
}
\end{vd}

\begin{vd}[Đề thi vào lớp 10 THPT chuyên ĐHSP Hà Nội năm học 2013-2014] Với các số thực $a,b,c$ thỏa mãn $a+b+c=0$, chứng minh đẳng thức \[(a^2+b^2+c^2)^2=2(a^4+b^4+c^4).\]
\loigiai{
Ta có
\begin{align*}
a+b+c=0&\Rightarrow \left(a+b+c\right)^2=0\\ 
&\Rightarrow a^2+b^2+c^2=-2(ab+bc+ca)\\
&\Rightarrow \left(a^2+b^2+c^2\right)^2=\left[-2(ab+bc+ca)\right]^2\\
&\Rightarrow a^4+b^4+c^4+2\left(a^2b^2+b^2c^2+c^2a^2\right)=4\left[a^2b^2+b^2c^2+c^2a^2+2abc(a+b+c)\right]\\
&\Rightarrow a^4+b^4+c^4=2\left(a^2b^2+b^2c^2+c^2a^2\right)\quad \text{(do $a+b+c=0$)}\\
&\Rightarrow 2(a^4+b^4+c^4)=(a^2+b^2+c^2)^2.
\end{align*}
}
\end{vd}

\begin{vd}[Đề thi vào lớp 10 THPT chuyên ĐHSP Hà Nội năm học 2013-2014] Với các số thực $a,b,c$ thỏa mãn đồng thời hai đẳng thức 
\begin{itemize}
\item[(i)]  $(a+b)(b+c)(c+a)=abc.$
\item[(ii)] $(a^3+b^3)(b^3+c^3)(c^3+a^3)=a^3b^3c^3.$
\end{itemize}
Chứng minh rằng $abc=0.$
\loigiai{
Từ (i) và (ii) suy ra \[abc(a^2-ab+b^2)(b^2-bc+c^2)(c^2-ca+a^2)=a^3b^3c^3.\tag{1}\]
Nếu $abc\ne 0$ thì từ $(1)$ ta có
 \[(a^2-ab+b^2)(b^2-bc+c^2)(c^2-ca+a^2)=a^2b^2c^2.\tag{2}\]
Mặt khác
 \[(a^2-ab+b^2)(b^2-bc+c^2)(c^2-ca+a^2)\ge |ab|\cdot |bc|\cdot |ca|=a^2b^2c^2.\tag{3}\]
Từ $(2)$ và $(3)$ suy ra $a=b=c$. Thay vào (i) ta được\[8a^3=a^3\Leftrightarrow 7a^3=0\Leftrightarrow a=0\Rightarrow abc=0,\]
điều này mâu thuẫn với $abc\ne 0$. Vậy $abc=0$.
}
\end{vd}

\begin{vd}[Đề thi vào lớp 10 THPT chuyên KHTN-ĐHQGHN năm học 2015-2016] Giả sử $a,b$ là hai số thực phân biệt thỏa mãn $a^2+3a=b^2+3b=2.$
\begin{enumerate}
\item Chứng minh rằng $a+b=-3$.
\item Chứng minh rằng $a^3+b^3=-45$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Do $a^2+3a=b^2+3b$ nên 
\begin{align*}
(a^2+3a)-(b^2+3b)=0&\Leftrightarrow (a^2-b^2)+3(a-b)=0\\ 
&\Leftrightarrow (a-b)(a+b+3)=0\Rightarrow a+b=-3\quad \text{(do $a\ne b$)}.
\end{align*}
\item Do $a^2+3a=b^2+3b=2$ nên 
\begin{align*}
(a^2+3a)+(b^2+3b)=4&\Rightarrow (a^2+b^2)+3(a+b)=4\Rightarrow a^2+b^2=13\\ 
&\Rightarrow (a+b)^2-2ab=13\Rightarrow ab=-2\\
&\Rightarrow a^3+b^3=(a+b)(a^2-ab+b^2)=-45.
\end{align*}
\end{enumerate}
}
\end{vd}

\begin{vd}[Đề thi vào lớp 10 THPT chuyên KHTN-ĐHQGHN năm học 2015-2016] Với $a,b,c$ là các số thực thỏa mãn $$(3a+3b+3c)^3=24+(3a+b-c)^3+(3b+c-a)^3+(3c+a-b)^3.$$
Chứng minh rằng $(a+2b)(b+2c)(c+2a)=1.$
\loigiai{
Đặt $x=3a+b-c$, $y=3b+c-a$, $z=3c+a-b$. Từ giả thiết ta có
\begin{align*}
&(x+y+z)^3-x^3-y^3-z^3=0\\ 
\Rightarrow \;& 3(x+y)(y+z)(z+x)=24\\
\Rightarrow \;& 24(a+2b)(b+2c)(c+2a)=24\\
\Rightarrow \;&(a+2b)(b+2c)(c+2a)=1.
\end{align*}
}
\end{vd}
\begin{vd}
\begin{enumerate}
\item Cho $x+y=a+b$, $x^2+y^2=a^2+b^2$. Chứng minh rằng $x^3+y^3=a^3+b^3.$
\item Cho $x+y=a+b$, $x^2+y^2=a^2+b^2$. Chứng minh rằng với $n\in\mathbb{N}^*$ thì $x^n+y^n=a^n+b^n.$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có 
\[2xy=(x+y)^2-(x^2+y^2)=(a+b)^2-(a^2+b^2)=2ab.\]
Do đó
\[x^3+y^3=(x+y)^3-3xy(x+y)=(a+b)^3-3ab(a+b)=a^3+b^3.\]
\item Theo giả thiết ta có
\begin{align*}
x^2+y^2=a^2+b^2&\Rightarrow  x^2-a^2=b^2-y^2\\ 
&\Rightarrow (x-a)(x+a)=(b-y)(b+y).\tag{1}
\end{align*}
Mặt khác, từ $x+y=a+b$ suy ra $x-a=b-y$. Thay vào $(1)$ ta được
\begin{align*}
&(x-a)(x+a)=(x-a)(b+y)\\ 
\Rightarrow\;&(x-a)(x+a)-(x-a)(b+y)=0\\
\Rightarrow\;&(x-a)\left[(x+a)-(b+y)\right]=0.
\end{align*}
Xét hai trường hợp sau
\begin{itemize}
\item $x-a=0$, do đó $b-y=0$, dẫn tới $x=a$ và $y=b$.\\
Khi đó $x^n+y^n=a^n+b^n$ với mọi $n\in\mathbb{N}^*$.
\item $(x+a)-(b+y)=0$, do đó $x+a=b+y$. Kết hợp với $x-a=b-y$ ta có $2x=2b$, suy ra $x=b$, do đó $y=a$.\\
Khi đó $x^n+y^n=a^n+b^n$ với mọi $n\in\mathbb{N}^*$.
\end{itemize}
Vậy $x^n+y^n=a^n+b^n$ với mọi $n\in\mathbb{N}^*$.
\end{enumerate}
\textbf{Lưu ý}. Ta có thể giải câu b) rồi suy ra câu a) bằng cách thay $n=3$.
}
\end{vd}

\begin{vd}\label{8D1G2VD}
\begin{enumerate}
\item Chứng minh rằng $(x+1)^3=x^3+3x^2+3x+1$.

\[a^2+ab+b^2=b^2+bc+c^2=c^2+ca+a^2.\]
\loigiai{
\begin{itemize}
\item Trước hết, ta đi chứng minh rằng nếu $a$, $b$, $c$ là các số thực thỏa mãn $a+b+c=0$ thì \[a^2+ab+b^2=b^2+bc+c^2=c^2+ca+a^2.\]

&\Rightarrow (a-c)(a+c)+b(a-c)=0\\
&\Rightarrow (a-c)(a+b+c)=0\\
&\Rightarrow a+b+c=0 \qquad \text{(do $a\ne c$)}.
\end{align*}
\end{itemize}
}
\end{vd}
