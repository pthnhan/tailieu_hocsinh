\begin{ex}%[Đề thi Toán 8 HKII năm học 2016 - 2017 - THCS Liên Châu - Yên Lạc]
	Tập nghiệm của phương trình $x^2-x=0$ là
	\choice
	{\{$0$\}}
	{\True \{$0;1$\}}
	{\{$1$\}}
	{\{$0;-1$\}}
	\loigiai{
	Ta có $x^2-x=0 \Leftrightarrow x(x-1)=0 \Leftrightarrow x=0$ hoặc $x-1=0 \Leftrightarrow x=0$ hoặc $x=1$.\\
	Vậy phương trình $x^2-x=0$ có tập nghiệm $S=\{0;1\}$.
	}
\end{ex}
\begin{ex}%[Đề thi Toán 8 HKII năm học 2016 - 2017 - THCS Liên Châu - Yên Lạc]
	Điều kiện xác định của phương trình $\displaystyle\frac{x+2}{x-3}=\displaystyle\frac{3x-1}{x(x-3)}+1$ là
	\choice
	{$x \ne 0$ hoặc $x \ne 3$}
	{$x \ne 0$ và $x \ne -3$}
	{\True $x \ne 0$ và $x \ne 3$}
	{$x \ne 3$}
	\loigiai{
	Điều kiện xác định của phương trình $\displaystyle\frac{x+2}{x-3}=\displaystyle\frac{3x-1}{x(x-3)}+1$ là
	$$\begin{cases}x-3 \neq 0\\x(x-3) \neq 0  \end{cases}\Leftrightarrow \begin{cases}x \neq 0 \\x\neq 3. \end{cases}$$
	}
\end{ex}

\begin{ex}%[Đề thi Toán 8 HKII năm học 2016 - 2017 - THCS Liên Châu - Yên Lạc]
	Giải các phương trình và bất phương trình sau:
	\begin{enumerate}[a)]
		\item $2x-3=0$;
		\item $\displaystyle\frac{x+3}{5}<\displaystyle\frac{5-x}{3}$;
		\item $\displaystyle\frac{1}{x-1}-\displaystyle\frac{3}{x-2}=\displaystyle\frac{-1}{(x-1)(x-2)}$.
	\end{enumerate}
	\loigiai{
	\begin{enumerate}[a)]
		\item Ta có $2x-3=0 \Leftrightarrow 2x = 3 \Leftrightarrow x=\displaystyle\frac{3}{2}$.\\
		Vậy phương trình $2x-3=0$ có tập nghiệm $S=\left\{\displaystyle\frac{3}{2}\right\}$.
		\item Ta có $\displaystyle\frac{x+3}{5}<\displaystyle\frac{5-x}{3} \Leftrightarrow \displaystyle\frac{3(x+3)}{5\times 3}<\displaystyle\frac{5(5-x)}{5\times 3} \Leftrightarrow 3(x+3)<5(5-x) \Leftrightarrow 3x+9<25-5x$
		$$\Leftrightarrow 3x+5x<25-9 \Leftrightarrow 8x<16 \Leftrightarrow x<2.$$
		Vậy bất phương trình $\displaystyle\frac{x+3}{5}<\displaystyle\frac{5-x}{3}$ có tập nghiệm $S=\{x|x<2\}$.
		\item Xét phương trình (*): $\displaystyle\frac{1}{x-1}-\displaystyle\frac{3}{x-2}=\displaystyle\frac{-1}{(x-1)(x-2)}$.\\
		Điều kiện: $x-1\ne 0$ và $x-2 \ne0 \Leftrightarrow x\ne 1$ và $x\ne 2$.\\
		Ta có (*) $\Leftrightarrow \displaystyle\frac{(x-2)-3(x-1)}{(x-1)(x-2)}=\displaystyle\frac{-1}{(x-1)(x-2)}$.\\
		\centerline{$\Rightarrow x-2-3x+3=-1 \Leftrightarrow 2x=2 \Leftrightarrow x=1$ (loại so với điều kiện $x\ne 1$).}
		Vậy phương trình đã cho vô nghiệm.
	\end{enumerate}}
\end{ex}
\begin{ex}%[Đề thi Toán 8 HKII năm học 2016 - 2017 - THCS Liên Châu - Yên Lạc]
	Một người đi xe máy từ $A$ đến $B$ với vận tốc $25$ km/h. Lúc về, người đó đi với vận tốc $30$ km/h nên thời gian đi là $20$ phút. Tính quãng đường $AB$.
	\loigiai{
	Gọi quãng đường $AB$ là $x$ km ($x>0$).\\
	Do đi từ $A$ đến $B$ với vận tốc $25$ km/h nên thời gian đi là $\displaystyle\frac{x}{25}$ (h).\\
	Do đi từ $B$ trở về $A$ với vận tốc $30$ km/h nên thời gian về là $\displaystyle\frac{x}{30}$ (h).\\
	Vì thời gian về ít hơn thời gian đi là $20$ phút $=\displaystyle\frac{1}{3}$ (h) nên ta có phương trình\\
	\centerline{$ \displaystyle\frac{x}{25}-\displaystyle\frac{x}{30}=\displaystyle\frac{1}{3} \Leftrightarrow 6x-5x=50 \Leftrightarrow x=50$ (thoả mãn $x>0$).}\vspace{0.2cm}
	Vậy quãng đường $AB$ dài $50$ km.
	}
\end{ex}
\begin{ex}%[Đề thi Toán 8 HKII năm học 2016 - 2017 - THCS Liên Châu - Yên Lạc]
	Cho tam giác $ABC$ có $AH$ là đường cao ($H\in BC$). Gọi $D$ và $E$ lần lượt là hình chiếu của $H$ trên $AB$ và $AC$. Chứng minh rằng
	\begin{enumerate}[a)]
		\item $\triangle ABH \backsim \triangle AHD$.
		\item $HE^2=AE \cdot EC$.
		\item Gọi $M$ là giao điểm của $BE$ và $CD$. Chứng minh rằng $\triangle DBM \backsim \triangle ECM$.
	\end{enumerate}
	\loigiai{
	\begin{center}
		\begin{tikzpicture}[line width=0.6pt]
		\clip (-0.5,-0.5) rectangle (5.5,5.5);
		\tkzDefPoints{0/0/B, 5/0/C, 1.8/5/A}
		\tkzDefPointBy[projection= onto B--C](A) \tkzGetPoint{H}
		\tkzDefPointBy[projection= onto A--B](H) \tkzGetPoint{D}
		\tkzDefPointBy[projection= onto A--C](H) \tkzGetPoint{E}
		\tkzInterLL(B,E)(C,D) \tkzGetPoint{M}
		\tkzLabelPoints[above](A);
		\tkzLabelPoints[below left](B)
		\tkzLabelPoints[below](H);
		\tkzLabelPoints[below right](C);
		\tkzLabelPoints[above left](D);
		\tkzLabelPoints[above right](E);
		\tkzLabelPoints[above](M);
		\tkzDrawSegments(A,B B,C C,A A,H D,H E,H B,E D,C);
		\tkzMarkRightAngle(A,D,H);
		\tkzMarkRightAngle(A,H,C);
		\tkzMarkRightAngle(A,E,H);
		\end{tikzpicture}
	\end{center}
	\begin{enumerate}[a)]
		\item \underline{Chứng minh $\triangle ABH \backsim \triangle AHD$}\\
		Ta có $\triangle ABH$ và $\triangle AHD$ là hai tam giác vuông có $\widehat{BAH}$ chung.\\
		Vậy $\triangle ABH \backsim \triangle AHD$.
		\item \underline{Chứng minh $HE^2=AE\cdot EC$}\\
		Chứng minh tương tự câu a ta được $\begin{cases}\triangle AEH \backsim \triangle AHC\\ \triangle HEC \backsim \triangle AHC \end{cases}$, từ đó
		$$\triangle AEH \backsim \triangle HEC \Rightarrow \displaystyle\frac{EH}{EC}=\displaystyle\frac{EA}{EH} \Rightarrow EH^2=EA\cdot EC.$$
		\item \underline{Chứng minh rằng $\triangle DBM \backsim \triangle ECM$}\\
		$\triangle ABH \backsim \triangle AHD \Rightarrow \displaystyle\frac{AB}{AH}=\displaystyle\frac{AH}{AD} \Rightarrow AH^2=AB\cdot AD$.\\
		$\triangle ACH \backsim \triangle AHE \Rightarrow \displaystyle\frac{AC}{AH}=\displaystyle\frac{AH}{AE} \Rightarrow AH^2=AC\cdot AE$.\\
		Từ đó ta có $AB\cdot AD = AC \cdot AE \Rightarrow \displaystyle\frac{AB}{AC}=\displaystyle\frac{AE}{AD}$.\\
		$\Rightarrow \triangle ABE \backsim \triangle ACD$ (chung góc $\widehat{BAC}$) $\Rightarrow \widehat{ABE}=\widehat{ACE} \Rightarrow \triangle DBM \backsim \triangle ECM$.
	\end{enumerate}
	}
\end{ex}
\begin{ex}%[Đề thi Toán 8 HKII năm học 2016 - 2017 - THCS Liên Châu - Yên Lạc]
	Cho phương trình ẩn $x$ sau đây $(2x+m)(x-1)-2x^2+mx+m-2=0$. Tìm các giá trị của $m$ để phương trình có các nghiệm là một số không âm.
	\loigiai{
	Ta có $(2x+m)(x-1)-2x^2+mx+m-2=0$.\\
	\centerline{$ \Leftrightarrow 2x^2-2x+mx-m-2x^2+mx+m-2=0 \Leftrightarrow (m-1)x=1$.}
	Vậy để phương trình có nghiệm là một số không âm thì $m-1>0 \Leftrightarrow m>1$.
	}
\end{ex}
