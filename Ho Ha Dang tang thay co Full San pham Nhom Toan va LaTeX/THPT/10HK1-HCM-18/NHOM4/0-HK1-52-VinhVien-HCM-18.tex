\begin{bt}%[HK1, 2018, THPT Vĩnh Viễn, HCM]%[Vũ Nguyễn Hoàng Anh, 10HK1]%[0D3B2-2]
Giải phương trình $\left| 2x^2-2x-4 \right|=2x-4$.
\loigiai{
Phương trình đã cho tương đương
\begin{align*}
\heva{&2x-4\ge 0\\ &\hoac{&2x^2-2x-4=2x-4\\ &2x^2-2x-4=-2x+4}}\Leftrightarrow \heva{&x\ge 2\\ &\hoac{&2x^2-4x=0\\ &2x^2-8=0}} \Leftrightarrow x=2.
\end{align*}
Vậy tập nghiệm của phương trình đã cho là $S=\{2\}$.
}
\end{bt}

\begin{bt}%[HK1, 2018, THPT Vĩnh Viễn, HCM]%[Vũ Nguyễn Hoàng Anh, 10HK1]%[0D3B2-4]
Giải phương trình $\sqrt{2x^2-8x-6}=x-3$.
\loigiai{
Phương trình đã cho tương đương
\begin{align*}
\heva{&x-3\ge 0\\ &2x^2-8x-6=(x-3)^2}\Leftrightarrow \heva{&x\ge 3\\ &x^2-2x-15=0}\Leftrightarrow x=5.
\end{align*}
Vậy tập nghiệm của phương trình đã cho là $S=\{5\}$.
}
\end{bt}

\begin{bt}%[HK1, 2018, THPT Vĩnh Viễn, HCM]%[Vũ Nguyễn Hoàng Anh, 10HK1]%[0D3B2-2]
Giải phương trình $\left| x^2-x-1 \right|=\left| 3x^2-2x-1 \right|$.
\loigiai{
Phương trình đã cho tương đương
\begin{align*}
\hoac{& x^2-x-1=3x^2-2x-1\\ &x^2-x-1=-3x^2+2x+1} \Leftrightarrow \hoac{& 2x^2-x=0\\ &4x^2-3x-2=0} \Leftrightarrow \hoac{&x=\dfrac{1}{2}\\ &x=0\\ &x=\dfrac{3\pm \sqrt{41}}{8}.}
\end{align*}
Vậy tập nghiệm của phương trình đã cho là $S=\left\{ \dfrac{1}{2}; 0; \dfrac{3\pm \sqrt{41}}{8}\right\}$.
}
\end{bt}

\begin{bt}%[HK1, 2018, THPT Vĩnh Viễn, HCM]%[Vũ Nguyễn Hoàng Anh, 10HK1]%[0D3K2-4]
Giải phương trình sau $\sqrt{x^2-x+1}+1=3x^2-3x$.
\loigiai{
 Đặt $t=\sqrt{x^2-x+1}$, ta có $t=\sqrt{\left( x-\dfrac{1}{2} \right)^2+\dfrac{3}{4}}\ge \dfrac{\sqrt{3}}{2}$. Khi đó phương trình đã cho trở thành
\begin{align*}
t+1=3\left( t^2-1 \right) \Leftrightarrow 3t^2-t-4=0 \Leftrightarrow t=\dfrac{4}{3}.
\end{align*}
Suy ra
\begin{align*}
\sqrt{x^2-x+1}=\dfrac{4}{3}\Leftrightarrow x^2-x-\dfrac{7}{9}=0 \Leftrightarrow x=\dfrac{3\pm \sqrt{37}}{6}.
\end{align*}
Vậy tập nghiệm của phương trình đã cho là $S=\left\{\dfrac{3+\sqrt{37}}{6};\dfrac{3-\sqrt{37}}{6} \right\}$.
}
\end{bt}


\begin{bt}%[HK1, 2018, THPT Vĩnh Viễn, HCM][Vũ Nguyễn Hoàng Anh, 10HK1]%[0D3B2-5]%[0D3K2-5]
Cho phương trình $x^2-(2m+1)x+m-8=0$. \qquad$(1)$
\begin{enumerate}
\item Chứng minh phương trình $(1)$ luôn có hai nghiệm phân biệt với mọi $m\in \mathbb{R}$.
\item Định $m$ để phương trình $(1)$ có hai nghiệm phân biệt cùng âm.
\item Định $m$ để phương trình $(1)$ có hai nghiệm phân biệt $x_1$, $x_2$ thỏa mãn
\begin{align*}
x_1^2+x_2^2 =4x_1x_2+51.
\end{align*}
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có
\begin{align*}
\Delta = (2m+1)^2-4(m-8)=4m^2+33>0,\ \forall m\in \mathbb{R}.
\end{align*}
Vậy phương trình đã cho luôn có hai nghiệm phân biệt với mọi $m\in \mathbb{R}$.
\item Phương trình $(1)$ có hai nghiệm cùng âm khi và chỉ khi
\begin{align*}
\heva{&2m+1<0\\ & m-8>0}\Leftrightarrow \heva{&m<-\dfrac{1}{2}\\ &m>8.}
\end{align*}
Vậy không tồn tại $m$ thỏa mãn đề bài.
\item Áp dụng định lí Vi\`ete, ta có
\begin{align*}
\heva{&x_1+x_2=2m+1\\ &x_1x_2=m-8.}
\end{align*}
Suy ra
\begin{eqnarray*}
&&x_1^2+x_2^2 =4x_1x_2+51\\
&\Leftrightarrow & (x_1+x_2)^2-6x_1x_2-51=0\\
& \Leftrightarrow & (2m+1)^2-6(m-8)-51=0\\
&\Leftrightarrow &4m^2-2m-2=0\\
&\Leftrightarrow &\hoac{&m=1\\ &m=-\dfrac{1}{2}.}
\end{eqnarray*}
Vậy tập các giá trị cần tìm của $m$ là $S=\left\{1;-\dfrac{1}{2}\right\}$.
\end{enumerate}
}
\end{bt}

\begin{bt}%[HK1, 2018, THPT Vĩnh Viễn, HCM]%[Vũ Nguyễn Hoàng Anh, 10HK1]%[0D3B3-4]
Giải hệ phương trình $\heva{&x^2-3xy+y^2=95\\ & 3x+y=1.}$

\loigiai{
Hệ phương trình đã cho tương đương
\begin{align*}
&\heva{&y=-3x+1\\ &x^2+3x(3x-1)+(3x-1)^2=95}\\
\Leftrightarrow\ &\heva{&y=-3x+1\\ &19x^2-9x-94=0}\\ \Leftrightarrow\ &\hoac{&x=-2, y=7\\ &x=\dfrac{47}{19},y=-\dfrac{122}{19}.}
\end{align*}
Vậy hệ phương trình đã cho có hai nghiệm là $\left( -2;7 \right)$ và $\left( \dfrac{47}{19};-\dfrac{122}{19} \right)$.
}
\end{bt}

\begin{bt}%[HK1, 2018, THPT Vĩnh Viễn, HCM]%[Vũ Nguyễn Hoàng Anh, 10HK1]%[0D3B2-4]
Giải hệ phương trình $\heva{&x^2+y^2-x+3y=0\\ &x^2+y^2+2x+2y-7=0.}$
\loigiai{
Lấy phương trình thứ hai trừ cho phương trình thứ nhất theo vế, rồi thay vào phương trình thứ nhất ta được
\begin{align*}
&\heva{&y=3x-7\\ &x^2+(3x-7)^2-x+3(3x-7)=0}\\
\Leftrightarrow \ &\heva{&y=3x-7\\ &10x^2-34x+28=0}\\
\Leftrightarrow \ &\hoac{&x=2, y=-1\\ &x=\dfrac{7}{5},y=-\dfrac{14}{5}.}
\end{align*}
Vậy hệ phương trình đã cho có hai nghiệm là $(2;-1)$ và $\left( \dfrac{7}{5}; -\dfrac{14}{5} \right)$.
}
\end{bt}

\begin{bt}%[HK1, 2018, THPT Vĩnh Viễn, HCM]%[Vũ Nguyễn Hoàng Anh, 10HK1]%[0H2B2-2]%[0H2B2-1]%[0H2B2-1]
Trong mặt phẳng $Oxy$, cho tam giác $ ABC$ với $A(5;4)$, $B(-3;0)$, $C(3;-2)$.
\begin{enumerate}
\item Chứng minh tam giác $ABC$ vuông cân tại $C$.
\item Tìm tọa độ trọng tâm $G$ của tam giác $ABC$, và tính độ dài đường trung tuyến kẻ từ $A$.
\item Tìm điểm $M$ thuộc trục hoành cách đều $A$ và $B$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Ta có $\overrightarrow{CA}=(2;6)$, $\overrightarrow{CB}=(-6;2)$. Suy ra
\begin{align*}
\heva{&\left| \overrightarrow{CA} \right|=\left| \overrightarrow{CB} \right|=2\sqrt{10}\\ & \overrightarrow{CA}\cdot \overrightarrow{CB}=0\ \text{hay}\ CA\perp CB.}
\end{align*}
Vậy tam giác $ABC$ vuông cân tại $C$.
\item  Ta có
\begin{align*}
\heva{&x_G=\dfrac{x_A+x_B+x_C}{3}=\dfrac{5}{3}\\ &y_G=\dfrac{y_A+y_B+y_C}{3}=\dfrac{2}{3}.}
\end{align*}
Vậy $G\left( \dfrac{5}{3};\dfrac{2}{3} \right)$. Lại có $\overrightarrow{GA}=\left( \dfrac{10}{3}; \dfrac{10}{3}\right)$ nên $m_a=\dfrac{3}{2}GA =5\sqrt{2}$.
\item Vì $M\in Ox$ nên $M=(m;0)$. Suy ra $\overrightarrow{AM}=(m-5;-4)$ và $\overrightarrow{BM}=(m+3;0)$. Do $M$ cách đều $A$ và $B$ nên
\begin{align*}
MA=MB \Leftrightarrow (m-5)^2+(-4)^2=(m+3)^2 \Leftrightarrow m=2.
\end{align*}
Vậy $M(2;0)$.
\end{enumerate}

}
\end{bt}

\begin{bt}%[HK1, 2018, THPT Vĩnh Viễn, HCM]%[Vũ Nguyễn Hoàng Anh, 10HK1]%[0H2B3-1]
Cho tam giác $ABC$ có $b=8$, $c=7$, $\widehat{A}=120^\circ$. Hãy tính
\begin{enumerate}
\item Độ dài cạnh $BC=a$ và diện tích tam giác $ABC$.
\item Tính độ dài đường cao $h_a$ của tam giác $ABC$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Áp dụng định lí cô-sin ta có
\begin{align*}
a=BC=\sqrt{b^2+c^2-2bc\cos A}=\sqrt{57}.
\end{align*}
Áp dụng công thức tính diện tích tam giác ta có
\begin{align*}
S_{ABC}=\dfrac{1}{2}bc\sin A=14\sqrt{3}.
\end{align*}
\item Ta có $h_a=\dfrac{2S_{ABC}}{a}=\dfrac{28\sqrt{19}}{19}$.
\end{enumerate}
}
\end{bt}
