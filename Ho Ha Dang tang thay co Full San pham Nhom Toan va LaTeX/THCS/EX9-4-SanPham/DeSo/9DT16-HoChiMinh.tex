\begin{ex}%[Đề thi vào 10, Sở giáo dục TP HCM, 2016]%[9D3B4]%[9D4B5]%[9D4B7]
	Giải các phương trình và hệ phương trình sau:
\begin{enumerate}
	\item $x^2-2\sqrt{5}x+5=0$
	\item $4x^4-5x^2-9=0$
	\item $\heva{&2x+5y=-1\\&3x-2y=8}$
	\item $x(x+3)=15-(3x-1)$.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item $x^2-2\sqrt{5}x+5=0$. Tính được $\Delta'=0$, phương trình có nghiệm kép $x=-\dfrac{b}{2a}=\sqrt{5}$.
	\item $4x^4-5x^2-9=0$. Đặt $t=x^2\geq0$, được phương trình $4t^2-5t-9=0\Leftrightarrow\hoac{&t=-1\,(\text{loại})\\&t=\dfrac{9}{4}}$. Với $t=\dfrac{9}{4}$, ta có $x=\pm\dfrac{3}{2}$.
	\item $\heva{&2x+5y=-1\\&3x-2y=8}\Leftrightarrow x=2,y=-1$. Nghiệm hệ phương trình là $(2;-1)$.
	\item $x(x+3)=15-(3x-1)\Leftrightarrow x^2+6x-16=0\Leftrightarrow x=-8,x=2$.
\end{enumerate}
}

\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục TP HCM, 2016]%[9D1K8]%[9D4K8]
\begin{enumerate}
\item Thu gọn biểu thức sau: $A=\dfrac{2-\sqrt{3}}{1+\sqrt{4+2\sqrt{3}}}+ \dfrac{2+\sqrt{3}}{1-\sqrt{4-2\sqrt{3}}}$
\item Ông Sáu gửi một số tiền vào ngân hàng theo mức lãi suất tiết kiệm với kỳ hạn $ 1 $ năm là $ 6\% $. Tuy nhiên sau thời hạn một năm ông Sáu không đến nhận tiền lãi mà để thêm một năm nữa mới lãnh. Khi đó số tiền lãi có được sau năm đầu tiên sẽ cộng dồn vào số tiền ban đầu để thành số tiền gửi cho năm kế tiếp với lãi suất cũ. Sau $ 2 $ năm ông Sáu nhận được số tiền là $ 112.360.000 $ đồng (kể cả gốc lẫn lãi). Hỏi ban đầu ông Sáu đã gửi bao nhiêu tiền?
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Ta có \begin{eqnarray*}
	A&=&\dfrac{2-\sqrt{3}}{1+\sqrt{4+2\sqrt{3}}}+ \dfrac{2+\sqrt{3}}{1-\sqrt{4-2\sqrt{3}}}=\dfrac{2-\sqrt{3}}{1+\sqrt{(\sqrt{3}+1)^2}}+ \dfrac{2+\sqrt{3}}{1-\sqrt{(\sqrt{3}-1)^2}}\\
	&=&\dfrac{2-\sqrt{3}}{1+(\sqrt{3}+1)}+ \dfrac{2+\sqrt{3}}{1-(\sqrt{3}-1)}=\dfrac{2-\sqrt{3}}{2+\sqrt{3}}+ \dfrac{2+\sqrt{3}}{2-\sqrt{3}}\\
	&=&\dfrac{(2-\sqrt{3})^2+(2+\sqrt{3})^2}{(2+\sqrt{3})(2-\sqrt{3})}=14.
	\end{eqnarray*}
	\item Gọi $P$ (đồng) là số tiền ban đầu của ông Sáu. Sau một năm, ông Sáu được số tiền cả vốn lẫn lãi là $P+P\times 6\%=P(1+6\%)$ (đồng), số tiền này coi như tiền vốn để tính lãi cho năm thứ hai. Cuối năm thứ hai, ông Sáu có số tiền là $P(1+6\%)+P(1+6\%)\times 6\%=P(1+6\%)^2$ (đồng). Theo đề bài số tiền ông Sáu rút được là $112.360.000$ đồng, nên ta có:
	$$112.360.000=P(1+6\%)^2\Leftrightarrow P=\dfrac{112.360.000}{(1+6\%)^2}=100.000.000\,(\text{đồng}).$$
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục TP HCM, 2016]%[9D4K4]
	Cho phương trình: $x^2-2mx+m-2=0\quad(1)$  ($x$ là ẩn số).
\begin{enumerate}
	\item Chứng minh phương trình $ (1) $ luôn có $ 2 $ nghiệm phân biệt với mọi giá trị $m$.
	\item Định $m$ để hai nghiệm $x_1,x_2$ của phương trình $ (1) $ thỏa mãn 
	$$(1+x_1)(2-x_2)+(1+x_2)(2-x_1)=x^2_1+x^2_2+2.$$
\end{enumerate}
\loigiai{
Xét phương trình: $x^2-2mx+m-2=0\quad(1)$  ($x$ là ẩn số)
\begin{enumerate}
	\item 
	Ta tính được $\Delta' = m^2-m+2=\left(m-\dfrac{1}{2}\right)^2+\dfrac{7}{4}>0$ (với $m\in\mathbb{R}$). Do đó, phương trình $ (1) $ luôn có $ 2 $ nghiệm phân biệt với mọi giá trị $m$.
	\item Theo Định lý Viète phương trình $(1)$ có hai nghiệm $x_1,x_2$ thỏa $\heva{&x_1+x_2=-\dfrac{b}{a}=2m\\&x_1x_2=\dfrac{c}{a}=m-2.}$
	
	Kết hợp với giả thiết, ta có
\begin{eqnarray*}
&&(1+x_1)(2-x_2)+(1+x_2)(2-x_1)=x^2_1+x^2_2+2\\&&\Leftrightarrow (x_1+x_2)^2-(x_1+x_2)-2=0
\Leftrightarrow 4m^2-2m-2=0\Leftrightarrow \hoac{&m=1\\&m=-\dfrac{1}{2}}.
\end{eqnarray*}
Vậy có $ 2 $ giá trị $m= 1,-\dfrac{1}{2}$ thỏa yêu cầu bài toán.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Sở giáo dục TP HCM, 2016]%[9H3B7]%[9H3B3]%[9H3K7]%[9H3G7]
	Cho tam giác $ABC$ $(AB<AC)$ có ba góc nhọn. Đường tròn tâm $O$ đường kính $BC$ cắt các đoạn $AC,AB$ lần lượt tại $D$, $E$. Gọi $H$ là giao điểm của $BD$ và $CE$, $F$ là giao điểm của $AH$ và $BC$.
\begin{enumerate}
	\item Chứng minh: $AF\perp BC$ và $\widehat{AFD}=\widehat{ACE}$.
	\item Gọi $ M $ là trung điểm của $ AH $. Chứng minh $MD\perp OD$ và $ 5 $ điểm $M,D,O,F,E$ cùng thuộc một đường tròn.
	\item Gọi $ K $ là giao điểm của $ AH $ và $DE$. Chứng minh $MD^2=MK.MF$ và $K$ là trực tâm của tam giác $MBC$.
	\item 
	Chứng minh: $\dfrac{2}{FK}=\dfrac{1}{FH}+\dfrac{1}{FA}$.
\end{enumerate}
	\loigiai{
			\begin{tikzpicture}
		\tkzInit[ymin=-3,ymax=5,xmin=-6,xmax=5]
		\tkzClip
		\tkzDefPoints{0/4.5/A,-2/0/B,3/0/C}
		\tkzDefMidPoint(B,C)\tkzGetPoint{O}
		\tkzDrawCircle(O,B)
		\tkzInterLC(A,B)(O,B)\tkzGetSecondPoint{E}
		\tkzInterLC(A,C)(O,B)\tkzGetFirstPoint{D}
		\tkzInterLL(B,D)(C,E)\tkzGetPoint{H}
		\tkzInterLL(B,C)(A,H)\tkzGetPoint{F}
		\tkzInterLL(E,D)(A,H)\tkzGetPoint{K}
		\tkzDefMidPoint(A,H)\tkzGetPoint{M}
		\tkzDrawCircle[dashed,circum](O,M,E)
		\tkzInterLC(M,C)(O,B)\tkzGetFirstPoint{L}
		\tkzDrawSegments(A,B B,C C,A B,D C,E A,F F,D M,D O,D E,D M,B M,C)
		\tkzDrawPoints(A,B,C,O,D,E,H,F,M,K,L)
		\tkzLabelPoints[right](C)
		\tkzLabelPoints[below left](H,K)\tkzLabelPoints[below](O,F)
		\tkzLabelPoints[left](B,E,M)\tkzLabelPoints[above](D,A)\tkzLabelPoints[above=3pt](L)
		\end{tikzpicture}
		
		\begin{enumerate}
			\item Ta có $BD\perp AC, CE\perp AB$ và $BD,CE$ cắt nhau tại $H$ nên $H$ là trực tâm $\triangle ABC$, hay $AH\equiv AF\perp BC$. 
			
			Xét tứ giác  $HFCD$ có $\widehat{HFC}=\widehat{HDC}=90^{\circ}$ nên nội tiếp đường tròn, suy ra $\widehat{AFD}=\widehat{DCH}=\widehat{ACE}$ (góc  nội tiếp).
			\item Tứ giác $AEHD$ nội tiếp đường tròn $(AH)$ tâm $M$ nên tam giác $MAD$ cân tại $M$, suy ra $\widehat{MAD}=\widehat{MDA}$ $(1)$.
			
			Tam giác $ODC$ cân tại $O$, suy ra $\widehat{ODC}=\widehat{OCD}$ $(2)$.
			
			Từ $(1),(2)$, kết hợp với $\widehat{MAD}+\widehat{DCO}=\widehat{FAC}+\widehat{ACF}=90^{\circ}$ ta được $\widehat{MDA}+\widehat{ODC}=90^{\circ}$, hay góc $\widehat{MDO}=90^{\circ}$, suy ra $MD\perp OD$.
			
			Chứng minh tương tự ta suy ra được $ME\perp OE,\widehat{MEO}=90^{\circ}$, cùng với $\widehat{MFO}=90^{\circ}$, ta suy ra  $ 5 $ điểm $M,D,O,F,E$ cùng thuộc đường tròn đường kính $OM$.
			\item 
			Xét tam giác $MDK$ và $MFD$ có $\widehat{M}$ chung, $\widehat{MDK}=\widehat{MFD}$ (vì $\widehat{MDE}=\widehat{DCE}=\widehat{HFD}$, trong các tứ giác nội tiếp). Do đó, $\triangle MDK\backsim \triangle MFD$, hay $\dfrac{MD}{MF}=\dfrac{MK}{MD}\Rightarrow MD^2=MF.MK$.
			
			Gọi $L$ là giao điểm của $MC$ và $(O)$. Ta có, $ML.MC=MD^2$ (suy ra được từ $\triangle MLD\backsim \triangle MDC$) và $MD^2=MF.MK$ (chứng minh trên), nên $ML.MC=MF.MK$.  
			
			Khi đó, $\triangle MLK\backsim \triangle MFC$, suy ra $FKLC$ là tứ giác nội tiếp, hay góc $\widehat{KFC}=\widehat{KLC}=90^{\circ}$, tức là $KL\perp LC$.
			 
			 Mặt khác $BL\perp LC$, nên $B,K,L$ thẳng hàng và $BK$ là đường cao. Ta vừa chứng minh được $BK, MK$ cùng là đường cao trong $\triangle MBC$, nên $K$ là trực tâm của tam giác $MBC$.
			\item 
			Ta có $FH.FA=FB.FC$ $(\triangle BFH\backsim\triangle AFC)$; $FK.FM=FB.FC$ $(\triangle BFK\backsim\triangle MFC)$.
			
			Suy ra $FH.FA=FK.FM\Rightarrow 2FH.FA=2FK.FM=FK.(FA+FH)$ hay ta được $\dfrac{2}{FK}=\dfrac{1}{FH}+\dfrac{1}{FA}$.
	
\end{enumerate}

}
\end{ex}
