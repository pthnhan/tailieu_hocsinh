\subsection*{PHẦN CHUNG CHO TẤT CẢ HỌC SINH}
\begin{bt}%[HKI, THPT Gia Định - TPHCM]%[1D1K3]
	Giải phương trình sau:\\
	$\sin\left(\dfrac{7x}{8}-\dfrac{\pi}{3}\right)-\sqrt{3}\cos\left(\dfrac{7x}{8}-\dfrac{\pi}{8} \right)= 2\cos\left( \dfrac{x}{8}+\dfrac{\pi}{4} \right)$.
	\loigiai{\immini{
			$\sin\left(\dfrac{7x}{8}-\dfrac{\pi}{3}\right)-\sqrt{3}\cos\left(\dfrac{7x}{8}-\dfrac{\pi}{8} \right)= 2\cos\left( \dfrac{x}{8}+\dfrac{\pi}{4} \right) \\                     \Leftrightarrow -\sin\left(\dfrac{7x}{8}-\dfrac{\pi}{3}\right)+\sqrt{3}\cos\left(\dfrac{7x}{8}-\dfrac{\pi}{8} \right)= -2\cos\left( \dfrac{x}{8}+\dfrac{\pi}{4} \right)\\
			\Leftrightarrow \sqrt{3}\cos\left(\dfrac{7x}{8}-\dfrac{\pi}{8} \right)-\sin\left(\dfrac{7x}{8}-\dfrac{\pi}{3}\right)=-2\cos\left( \dfrac{x}{8}+\dfrac{\pi}{4} \right)\\ \Leftrightarrow \dfrac{\sqrt{3}}{2} \cos\left(\dfrac{7x}{8}-\dfrac{\pi}{3}\right)-\dfrac{1}{2}\sin\left(\dfrac{7x}{8}-\dfrac{\pi}{8} \right)=- \cos\left( \dfrac{x}{8}+\dfrac{\pi}{4} \right) \\
			\Leftrightarrow \cos\left(\dfrac{\pi}{6} \right) \cos\left(\dfrac{7x}{8}-\dfrac{\pi}{3}\right)-\sin\left(\dfrac{\pi}{6} \right) \sin\left(\dfrac{7x}{8}-\dfrac{\pi}{8} \right)= -\cos\left( \dfrac{x}{8}+\dfrac{\pi}{4} \right) \\
			\Leftrightarrow  \cos\left(\dfrac{7x}{8}-\dfrac{\pi}{6}\right) = \cos\left(\pi- \dfrac{x}{8}-\dfrac{\pi}{4} \right) \Leftrightarrow \hoac{& x=\dfrac{11}{12} +k2\pi \\ &x=-\dfrac{7}{12}+k2\pi}$.
			
	}}
\end{bt}

\begin{bt}%[HKI, THPT Gia Định - TPHCM]%[1D2B3]
	Tìm số hạng không chứa biến $x$ của khai triển $\left(7x^4+\dfrac{8}{7x^5} \right)^{18} $.
	
	\loigiai{\immini{
			Số hạng tổng quát: $T_{k+1}=\mathrm{C}_{18}^k(7x^4)^{18-k}\left(\dfrac{8}{7x^5} \right)^k= \mathrm{C}_{18}^k \times 7^{18-2k}\times 8^k\times x^{72-9k}$ \\
			Vì số hạng không chứa $x$ nên: $72-9k=0 \Leftrightarrow k=8 \Rightarrow$ số hạng không chứa $x$ là:$T_{9}=\mathrm{C}_{18}^8\cdot7^2\cdot8^8 $.
			
	}}
\end{bt}\begin{bt}%[HKI, THPT Gia Định - TPHCM]%[1D2B3]
	Giải phương trình $3\mathrm{C}_n^6 =7\mathrm{C}_n^7$ với $n$ là số tự nhiên, $\mathrm{C}_n^k $ là tổ hợp chập $k$ của $n$ phần tử.
	
	\loigiai{\immini{
			Điều kiện: $\heva{&n \geq 7\\&n \in \mathrm{Z}}$\\
			$3\mathrm{C}_n^6 =7\mathrm{C}_n^7 \Leftrightarrow \dfrac{3\times n!}{3!(n-6)!}= \dfrac{7\times n!}{7!(n-7)!}\\
			\Leftrightarrow  \dfrac{3n(n-1)\ldots(n-5)(n-6)!}{3!(n-6)!}=\dfrac{7n(n-1)\ldots(n-5)(n-6)(n-7)!}{7!(n-7)!}  \Leftrightarrow \dfrac{1}{2}=\dfrac{n-6}{720}\\ \Leftrightarrow n=366 $ (nhận).
			
	}}
\end{bt}

\subsection*{PHẦN RIÊNG C} 
\begin{bt}%[HKI, THPT Gia Định - TPHCM]%[1D2K3]
	Tìm hệ số của số hạng chứa $x^8$ của khai triển $ \displaystyle\left(2-7x+8x^2 \right)\displaystyle\left(7+8x\right)^{18}$.
	
	\loigiai{\immini{
			Số hạng chứa $x^8$ của khai triển là: $T=-7x\mathrm{C}_{18}^{7} \cdot7^{11}\cdot(8x)^{7}+8x^2\mathrm{C}_{18}^{6} \cdot7^{12}\cdot(8x)^{6} +2\mathrm{C}_{18}^{8} \cdot7^{10}\cdot(8x)^{8} $. \\
			nên hệ số của số hạng chứa $x^8$ là: $T=-7\mathrm{C}_{18}^{7} \cdot7^{11}\cdot8 ^7+8\mathrm{C}_{18}^{6} \cdot7^{12}\cdot8^{6}+2\mathrm{C}_{18}^{8} \cdot7^{10}\cdot8^{8} $
			
	}}
\end{bt}
\begin{bt}%[HKI, THPT Gia Định - TPHCM]%[1D2B5]
	Một hộp có $12$ viên bi, trong đó có $5$ viên bi xanh, $4$ viên bi đỏ và $3$ viên bi vàng (các viên bi đôi một khác nhau). Lấy ngẫu nhiên $3$ viên bi từ hộp. Tính xác suất để $3$ viên bi lấy ra chỉ có tối đa hai màu. 
	
	\loigiai{\immini{
			Số phần tử của không gian mẫu: $n(\Omega)=\mathrm{C}_{12}^3$ cách \\
			Gọi  là biến cố $A$ : " lấy được 3 viên bi có đủ 3 màu " \\
			$\Rightarrow n(A)=\mathrm{C}_5^1\mathrm{C}_4^1\mathrm{C}_3^1 \Rightarrow P(A)=\dfrac{n(A)}{n(\Omega)}=\dfrac{3}{11}$\\
			Gọi biến cố $B$ : "lấy được ba viên bi có tối đa hai màu " \\
			$\Rightarrow P(B)=1-P(A)=\dfrac{8}{11}		$	
	}}
\end{bt}
