\begin{ex} %[Đề GHK I, Chu Văn An, Hà Nội 2017-2018] %[2D1B3]
	Tìm giá trị lớn nhất của hàm số $y=\sqrt{x + 1} + \sqrt{3 - x}$ trên đoạn $\left[- 1; 3\right]$.
	\choice 
	{$\mathop {\max}\limits_{x \in \left[ {-1;3} \right]} y=2\sqrt{3}$}
	{$\mathop {\max}\limits_{x \in \left[ {-1;3} \right]} y=3\sqrt{2}$}
	{\True $\mathop {\max}\limits_{x \in \left[ {-1;3} \right]} y =2\sqrt{2}$}
	{$\mathop {\max}\limits_{x \in \left[ {-1;3} \right]} y=2$}
	\loigiai{
		Hàm số liên tục và xác định trên $\left[- 1; 3\right]$.\\
		Ta có $y'=\dfrac{1}{2\sqrt{x + 1}} - \dfrac{1}{2\sqrt{3 - x}}=0\Leftrightarrow x=1$.\\
		Hàm số không có đạo hàm tại $x= - 1; x=3$.\\
		Suy ra $y\left(- 1\right)=2; y\left(3\right)=2; y\left(1\right)=2\sqrt{2}$.
		Vậy $\mathop {\max}\limits_{x \in \left[ {-1;3} \right]} y =2\sqrt{2}$.}
\end{ex}

		Đường thẳng $y=m - 2x$ cắt đồ thị hàm số $y=\dfrac{2x + 4}{x + 1}$ tại hai điểm phân biệt khi và chỉ khi phương trình $\left(1\right)$ có hai nghiệm phân biệt khác $ - 1$ \\
		$\Leftrightarrow \heva{&\left(m - 4 \right)^2 - 8\left(- m + 4\right)>0 \\ & 2 - (m - 4)\left(- 1\right) - m + 4\ne 0} \Leftrightarrow \heva{& m^2 - 16>0 \\ & 2\ne 0} \Leftrightarrow |m|>4.$}
