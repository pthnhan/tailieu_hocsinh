\begin{bt}%[HK1, THPT Đan Phượng Hà Nội, 2018]%[1D1B3]
Giải phương trình
\begin{enumerate}[a)]
\item $\sin 4x +\cos 5x=0$.
\item $\sin 3x +\cos 2x =1+2\sin x \cos 2x$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item TXĐ: $\mathscr{D}=\mathbb{R}$.
\\ $\sin 4x +\cos 5x=0\Leftrightarrow \sin \left(\dfrac{\pi}{2}- 5x \right)= \sin\left(-4x\right)$
\\ $ \Leftrightarrow \hoac{& \dfrac{\pi}{2}-5x=-4x+k2\pi \\& \dfrac{\pi}{2}-5x =\pi + 4x  k2\pi} \Leftrightarrow \hoac{& x= \dfrac{\pi}{2}+k2\pi \\& x= -\dfrac{\pi}{18} + \dfrac{k2\pi}{9}},k\in \mathbb{Z}$.\\
Vậy tập nghiệm của phương trình là $S=\left\{ \dfrac{\pi}{2}+k2\pi;-\dfrac{\pi}{18} + \dfrac{k2\pi}{9}\bigg| k\in \mathbb{Z}\right\}$.
\item TXĐ: $\mathscr{D}=\mathbb{R}$.
\\ $\sin 3x +\cos 2x =1+2\sin x \cos 2x $
\\ $\Leftrightarrow 3\sin x -4\sin^3 x + 1- 2\sin^2 x-2\sin x(1-2\sin^2 x)-1=0$
\\ $\Leftrightarrow -2\sin^2 x + \sin x=0\Leftrightarrow \hoac{& \sin x=0 \\& \sin x=\dfrac{1}{2}}\Leftrightarrow \hoac{& x=k\pi \\& x=\dfrac{\pi}{6}+k2\pi \\& x=\dfrac{5\pi}{6}+k2\pi}, k\in \mathbb{Z}.$\\
Vậy tập nghiệm của phương trình là $S=\left\{k\pi;\dfrac{\pi}{6}+k2\pi;\dfrac{5\pi}{6}+k2\pi\bigg| k\in \mathbb{Z}\right\}$.
\end{enumerate}
}
\end{bt}
\begin{bt}%[HK1, THPT Đan Phượng Hà Nội, 2018]%[1D2B3]

\begin{bt}%[HK1, THPT Đan Phượng Hà Nội, 2018]%[1D2K5]

$$\left( 1^2+1+1\right)\cdot 1!+\left( 2^2+2+1\right)\cdot 2!+\left( 3^2+3+1\right)\cdot 3!+\ldots +\left( n^2+n+1\right)\cdot n!$$
\loigiai{
Với $k\geq 1$, ta có: $\left( k^2+k+1\right)\cdot k!=\left( k^2+2k+1-k\right)\cdot k!=(k+1)^2\cdot k!-k\cdot k!=(k+1)\cdot (k+1)!-k\cdot k!$.\\
Áp dụng:\\
$\left( 1^2+1+1\right)\cdot 1!=2\cdot 2!-1\cdot 1!$\\
$\left( 2^2+2+1\right)\cdot 2!=3\cdot 3!-2\cdot 2!$\\
$\left( 3^2+3+1\right)\cdot 3!=4\cdot 4!-3\cdot 3!$\\
\hspace*{0.5cm} $\vdots$\hspace*{2cm}$\vdots$ \\
$\left( n^2+n+1\right)\cdot n!=(n+1)\cdot (n+1)!-n\cdot n!$\\
Cộng theo vế các đẳng thức, ta được: \\
$\left( 1^2+1+1\right)\cdot 1!+\left( 2^2+2+1\right)\cdot 2!+\left( 3^2+3+1\right)\cdot 3!+\ldots +\left( n^2+n+1\right)\cdot n!=(n+1)\cdot (n+1)!-1$.
}
