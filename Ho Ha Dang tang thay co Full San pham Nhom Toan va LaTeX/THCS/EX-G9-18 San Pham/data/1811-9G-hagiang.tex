\begin{ex}%[Đề thi HSG 9, tỉnh Hà Giang, năm 2018]%[Tô Ngọc Thy, Dự án EX-9G-18]%[9D0K1]

Cho $x=\sqrt{4+\sqrt{7}}-\sqrt{4-\sqrt{7}}$. Tính $A={\left(x^4-x^3-x^2+2x-1\right)}^{2017}$.
\loigiai{
	Ta có $x\sqrt{2}=\sqrt{8+2\sqrt{7}}-\sqrt{8-2\sqrt{7}}=\sqrt{7}+1-\left(\sqrt{7}-1\right)=2$$ \Rightarrow x=\sqrt{2}$. Do đó $A=1$.
}
\end{ex}


\begin{ex}%[Đề thi HSG 9, tỉnh Hà Giang, năm 2018]%[Tô Ngọc Thy, Dự án EX-9G-18]%[8D0G5]
 Cho $a,b,c$ là các số hữu tỉ đôi một khác nhau.
	Chứng minh rằng $A=\dfrac{1}{{\left(a-b\right)}^2}+\dfrac{1}{{\left(b-c\right)}^2}+\dfrac{1}{{\left(c-a\right)}^2}$ là bình phương của một số hữu tỉ.
\loigiai{
	Ta có \begin{align*}
			& {\left(\dfrac{1}{a-b}+\dfrac{1}{b-c}+\dfrac{1}{c-a}\right)}^2=\dfrac{1}{{\left(a-b\right)}^2}+\dfrac{1}{{\left(b-c\right)}^2}+\dfrac{1}{{\left(c-a\right)}^2}\\
			&+\dfrac{2}{\left(a-b\right)\left(b-c\right)}+\dfrac{2}{\left(b-c\right)\left(c-a\right)}+\dfrac{2}{\left(c-a\right)\left(a-b\right)} \\ 
			& =\dfrac{1}{{\left(a-b\right)}^2}+\dfrac{1}{{\left(b-c\right)}^2}+\dfrac{1}{{\left(c-a\right)}^2}+\dfrac{2\left(c-a+a-b+b-c\right)}{\left(a-b\right)\left(b-c\right)\left(c-a\right)} \\ 
			& =\dfrac{1}{{\left(a-b\right)}^2}+\dfrac{1}{{\left(b-c\right)}^2}+\dfrac{1}{{\left(c-a\right)}^2}.
		\end{align*}
		Vậy $A=\dfrac{1}{{\left(a-b\right)}^2}+\dfrac{1}{{\left(b-c\right)}^2}+\dfrac{1}{{\left(c-a\right)}^2}$ là bình phương của một số hữu tỉ.
}
\end{ex}


\begin{ex}%[Đề thi HSG 9, tỉnh Hà Giang, năm 2018]%[Tô Ngọc Thy, Dự án EX-9G-18]%[9D0K2]

Giải phương trình $\dfrac{2x}{2x^2-5x+3}+\dfrac{13x}{2x^2+x+3}=6$.

\loigiai{

 ĐKXĐ: $x\ne 1;x\ne \dfrac{3}{2}$. \\
		Xét $x=0$ không phải là nghiệm. \\
		Xét $x\ne 0$, ta có phương trình
		$\dfrac{2}{2x-5+\dfrac{3}{x}}+\dfrac{13}{2x+1+\dfrac{3}{x}}=6$.\\
		Đặt $2x-5+\dfrac{3}{x}=t$ được $\dfrac{2}{t}+\dfrac{13}{t+6}=6\Leftrightarrow 2t^2+7t-4=0\Leftrightarrow \hoac{
			& t=\dfrac{1}{2} \\ 
			& t=-4}.$\\
		Với $t=\dfrac{1}{2}$ suy ra $2x-5+\dfrac{3}{x}=\dfrac{1}{2}\Leftrightarrow 4x^2-11x+6=0\Leftrightarrow \hoac{
			& x=\dfrac{3}{4} \\ 
			& x=2}.$\\
		Với $t=-4$ suy ra $2x-5+\dfrac{3}{x}=-4\Leftrightarrow 2x^2-x+3=0$ vô nghiệm.\\
		Vậy phương trình có tập nghiệm $S=\left\{\dfrac{3}{4};2\right\}$.
	}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Hà Giang, năm 2018]%[Tô Ngọc Thy, Dự án EX-9G-18]%[9D0K8]
Cho $P=x^2+ax+b$ với $a,b\in \mathbb{N}$. Biết $P(1)=2017$. Tính $P(3)+P\left(-1\right)$.
\loigiai{
Ta có $P(1)=2017\Leftrightarrow 2017=1+a+b\Leftrightarrow a+b=2016$.\\
Do đó $P(3)+P\left(-1\right)=\left(9+3a+b\right)+\left(1-a+b\right)=10+2\left(a+b\right)=10+2\cdot2016=4012$.
}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Hà Giang, năm 2018]%[Tô Ngọc Thy, Dự án EX-9G-18]%[9D0G4]
Tìm các số nguyên dương $n$ sao cho $n^4+n^3+1$ là số chính phương.
\loigiai{
	Đặt $A=n^4+n^3+1$. Với $n=1$ thì $A=3$ không thỏa mãn.\\
	Với $n\ge 2$ ta có $4A=4n^4+4n^3+4$.\\
	Xét $4A-{\left(2n^2+n-1\right)}^2=3n^2+2n+3>0\Rightarrow 4A>{\left(2n^2+n-1\right)}^2$.\\
	Xét $4A-{\left(2n^2+n\right)}^2=4-n^2\le 0\Rightarrow 4A\le {\left(2n^2+n\right)}^2$.\\
	Vậy $4A={\left(2n^2+n\right)}^2\Rightarrow n=2$.\\
	Với $n=2$ thì $A=25$ thỏa mãn bài toán.}
\end{ex}

\begin{ex}%[Đề thi HSG 9, tỉnh Hà Giang, năm 2018]%[Tô Ngọc Thy, Dự án EX-9G-18]%[9D0G7]
Cho $a,b,c>0$. Chứng minh rằng $\dfrac{b^2+c^2}{a}+\dfrac{c^2+a^2}{b}+\dfrac{a^2+b^2}{c}\ge 2\left(a+b+c\right)$.
\loigiai{
	Áp dụng bất đẳng thức AM-GM ta có
	\begin{align*}
		& \dfrac{b^2+c^2}{a}+\dfrac{c^2+a^2}{b}+\dfrac{a^2+b^2}{c}\ge 2\left(\dfrac{bc}{a}+\dfrac{ca}{b}+\dfrac{ab}{c}\right) \\ 
		& =\left(\dfrac{bc}{a}+\dfrac{ca}{b}\right)+\left(\dfrac{ca}{b}+\dfrac{ab}{c}\right)+\left(\dfrac{bc}{a}+\dfrac{ab}{c}\right) \\ 
		& \ge 2\left(a+b+c\right).
	\end{align*}
	Dấu $"="$ xảy ra khi và chỉ khi $a=b=c$.}
\end{ex}



\begin{ex}%[Đề thi HSG 9, tỉnh Hà Giang, năm 2018]%[Tô Ngọc Thy, Dự án EX-9G-18]%[9H0G2][9H0G5]
Cho tam giác $ABC$ vuông cân tại $A$. Gọi $D$ là trung điểm của cạnh $BC$. Lấy điểm $M$ bất kỳ trên đoạn $AD$ ($M$ khác $A,D$). Gọi $N,P$ theo thứ tự là hình chiếu vuông góc của $M$ xuống các cạnh $AB,AC$ và $H$ là hình chiếu vuông góc của $N$ xuống đường thẳng $PD$.
\begin{enumerate}
	\item  Chứng minh rằng $AH$ vuông góc với $BH$.
	\item  Đường thẳng qua $B$ song song với $AD$ cắt đường trung trực của $AB$ tại $I$. Chứng minh rằng ba điểm $H,N,I$ thẳng hàng.
\end{enumerate}
\loigiai{
	\begin{center}
		\begin{tikzpicture}
		\tkzInit[ymin=-2.8,ymax=5.5,xmin=-1,xmax=5.5]
		\tkzClip
		\tkzDefPoints{0/0/A,5/0/B,0/5/C}
		\tkzDefMidPoint(A,B)\tkzGetPoint{J}
		\tkzDrawMedian(C,B)(A)
		\tkzGetPoint{D}
		\tkzDefPointBy[homothety=center A ratio 0.6](D)\tkzGetPoint{M}
		\tkzDrawAltitude(B,A)(M)
		\tkzGetPoint{N}
		\tkzDrawAltitude(A,C)(M)
		\tkzGetPoint{P}
		\tkzDrawAltitude(P,D)(N)
		\tkzGetPoint{H}
		\tkzDefLine[parallel=through B](A,D) \tkzGetPoint{c}
		\tkzDefLine[parallel=through B](A,C) \tkzGetPoint{e}
		% \tkzShowLine[parallel=through B](A,D)
		\tkzDefLine[orthogonal=through J](B,A) \tkzGetPoint{d}
		\tkzInterLL(B,c)(J,d) \tkzGetPoint{I}
		\tkzDrawPoints(I)
		\tkzInterLL(B,e)(P,D) \tkzGetPoint{E}
		\tkzDrawPoints(E)
		\tkzDrawSegments(A,B B,C C,A P,E D,I I,H P,N H,B E,N E,B A,H)
		\tkzDrawPoints(A,B,C,D,I,M,N,P,H)
		\tkzLabelPoints[below left](N)
		\tkzLabelPoints[below](B)
		\tkzLabelPoints[left](A,C,P)
		\tkzLabelPoints[right](M,E,I)
		\tkzLabelPoints[above](H,D)

	\begin{enumerate}
		\item  Đường thẳng qua $B$ song song với $AC$ cắt tia $PD$ tại $E$.\\
		Ta có $BE=PC=BN$$\Rightarrow \Delta BEN$ vuông cân.\\
		Do $\widehat{NBE}=\widehat{NHE}=90^\circ$ nên $B,H$ cùng thuộc đường tròn đường kính $NE$.\\
		Suy ra $\widehat{NHB}=\widehat{NEB}=45^\circ$ (1).\\
		Tương tự hai điểm $A,H$ cùng thuộc đường tròn đường kính $PN$.\\
		$\Rightarrow \widehat{AHN}=\widehat{APN}=45^\circ$ (2).\\
		Từ (1) và (2) $\Rightarrow \widehat{AHB}=90^\circ$ hay $AH\bot BH$.
		\item  Từ giả thiết suy ra $\widehat{AIB}=90^\circ$ nên $I$ là điểm chính giữa của cung $AIB$ của đường tròn đường kính $AB$.\\
		Mặt khác, theo kết quả câu a) thì tia $HN$ là tia phân giác của $\widehat{AHB}$ và $\widehat{AHB}$ là góc nội tiếp chắn cung $\overset\frown{AIB}$ của đường tròn đường kính $AB$, nên tia $HN$ phải đi qua $I$.\\
		Do đó $3$ điểm $H,N,I$ thẳng hàng.
\end{enumerate}}
\end{ex}
