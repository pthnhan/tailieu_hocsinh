\begin{ex}%[Đề thi vào 10 chuyên Thái Nguyên, 2016]%[9D3G3]
	Giải hệ phương trình: $\heva{2y(x^2-y^2)&=3x \indent (1) \\ x(x^2+y^2)&=10y \indent (2)}$ với $x, y$ cùng dấu.
	\loigiai{
	+ Nếu $y=0$ thì $x=0 \Rightarrow (0;0)$ là nghiệm của hệ.
	\\+ Nếu $y \neq 0$ thì đặt $x=ky$ $(k>0)$ ta được hệ phương trình
\begin{align*}
\begin{cases}
2y(k^2y^2-y^2)=3ky \\ ky(k^2y^2+y^2)=10y
\end{cases} \Rightarrow \dfrac{2(k^2-1)}{k(k^2+1)} =\dfrac{3k}{10} \Rightarrow 3k^4-17k^2+20=0 \\
\Rightarrow k= \pm 2 \vee k=\pm \sqrt{\dfrac{5}{3}}
\end{align*} 
Do $k>0$ nên $k=2$ hoặc $k=\sqrt{\dfrac{5}{3}}$.
 \begin{itemize}
 	\item Với $k=2 \Rightarrow x=2y$, thay vào $(1)$ ta được $y^2=1 \Rightarrow y=\pm 1$.
 	\\$\Rightarrow (2;1) \vee (-2;-1)$ là nghiệm của hệ
 	\item Với $k=\sqrt{\dfrac{5}{3}} \Rightarrow x=\sqrt{\dfrac{5}{3}}y$, thay vào $(1)$ ta được
 	\begin{align*}
 	2y\left(\dfrac{5}{3}y^2-y^2\right)=15y \Rightarrow y=\pm \sqrt{\dfrac{3\sqrt{15}}{4}} \Rightarrow x= \pm \sqrt{\dfrac{5\sqrt{15}}{4}}
 	\end{align*}
 \end{itemize}
Vậy hệ phương trình có 4 nghiệm: $\left(2;1\right), \left(-2,-1\right), \left( \sqrt{\dfrac{5\sqrt{15}}{4}}; \sqrt{\dfrac{3\sqrt{15}}{4}} \right), \left( -\sqrt{\dfrac{5\sqrt{15}}{4}}; - \sqrt{\dfrac{3\sqrt{15}}{4}}\right)$}
\end{ex}   

\begin{ex}%[Đề thi vào 10 chuyên Thái Nguyên, 2016]%[9D5G1]
	Tìm giá trị nhỏ nhất của biểu thức:
	\[ P= \sqrt{x+6\sqrt{x-9}}+\sqrt{x-6\sqrt{x-9}}.\]
	\loigiai{Điều kiện: $\begin{cases}
		x\geq 9 \\ x\geq 6\sqrt{x-9}
		\end{cases} \Leftrightarrow x\geq 9$.
	\\ Ta có $P=\sqrt{(3+\sqrt{x-9})^2} + \sqrt{(3- \sqrt{x-9})^2}=|3+\sqrt{x-9}| + |3-\sqrt{x-9}|$.
	\\ Do tính chất $|A|+|B| \geq |A+B|$. Dấu $"="$ xảy ra khi $AB\geq 0$ nên 
	$$P= |3+\sqrt{x-9}| + |3-\sqrt{x-9}| \geq |3+\sqrt{x-9} +3 -\sqrt{x-9}|=6$$ 
	Dấu "=" xảy ra khi $(3+\sqrt{x-9})(3-\sqrt{x-9})\geq 0 \Leftrightarrow 9-(x-9)\geq 0 \Leftrightarrow x\geq 18$.
\\ Vậy $\min P=6$ khi $9 \leq x \leq 18$}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Thái Nguyên, 2016]%[9D5G3]
	Tìm tất cả nghiệm nguyên $(x;y)$ của phương trình:
	\[ 2xy+x+y=87.\]
	\loigiai{
	Ta có $2xy+x+y=83 \Leftrightarrow 4xy+2x+2y=166 \Leftrightarrow (2x+1)(2y+1)=167$.
\\ Do $167$ là số nguyên tố nên $2x+1$ chỉ có thể là một trong $4$ số $\pm 1, \pm 167$.
\begin{itemize}
	\item Với $2x+1=1$ ta được $x=0,y=83$.
	\item Với $2x+1=-1$ ta được $x=-1, y=-84$.
	\item Với $2x+1=167$ ta được $x=83,y=0$.
	\item Với $2x+1=-167$ ta được $x=-84,y=-1$.
\end{itemize}
Vậy phương trình đã cho có $4$ nghiệm nguyên $(x,y)$ là $(0;83),(-1;-84),(83;0),(-84;-1)$.
}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Thái Nguyên, 2016]%[9D5G3]
	Tìm tất cả các số có 5 chữ số $\overline{abcde}$ sao cho $\sqrt[3]{\overline{abcde}}=\overline{ab}$.
	\loigiai{Ta có: $\overline{abcde}=1000\overline{ab} +\overline{cde}$.
	\\ Ta lại có $1000\overline{ab} + \overline{cde} = (\overline{ab})^3$. Đặt $m=\overline{ab},n=\overline{cde}$ ta được $$1000m+n=m^3 \Rightarrow m^3 \geq 1000m \Rightarrow m^2 \geq 1000 \Rightarrow m \geq 32 ~ (1)$$
Vì $n<1000$ nên $m^3<1000m+1000 \Rightarrow m(m^2-1000)<1000$.
\\ Nếu $m\geq 33$ thì $m(m^2-1000)\geq 33.89=2937 \geq 1000$ (vô lý).
\\ Do đó $m<33$ (2).
\\ Từ $(1)$ và $(2)$ suy ra $m=32$.
\\ Vậy $\overline{abcde}=32^3=32768$.}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Thái Nguyên, 2016]%[9D5G2]
	Cho ba số $a$, $b$, $c$ nguyên dương, nguyên tố cùng nhau và thỏa mãn điều kiện: $\dfrac{1}{a}+\dfrac{1}{b}=\dfrac{1}{c}$. Chứng minh $a+b$ là số chính phương.
	\loigiai{Từ $\dfrac{1}{a}+\dfrac{1}{b}=\dfrac{1}{c}$ suy ra $\dfrac{a+b}{ab}=\dfrac{1}{c} \Leftrightarrow c(a+b)=ab \Leftrightarrow ab-ac-bc=0$ $(1)$
	\\ Cộng hai vế của $(1)$ với $c^2$ ta được $(a-c)(b-c)=c^2 \indent (2)$
\\ Nếu $d$ là ước nguyên dương của $a-c$ và $b-c$ thì $c^2$ $ \vdots$ $ d^2$
\\ Do đó $c \vdots d$, mà $a-c$ và $b-c$ cùng chia hết cho $d$ nên $d$ là ước của $a,b,c$. Theo giả thiết $a,b,c$ nguyên tố cùng nhau nên $d=1$.
\\ Vậy $a-c$ và $b-c$ không có ước chung lớn hơn $1$, mà $(a-c)(b-c)$ là số chính phương nên $a-c$ và $b-c$ là số chính phương khác nhau.
\\ Đặt $a-c=k^2, b-c=m^2 \Rightarrow c^2=(a-c)(b-c)=k^2m^2 \Rightarrow c=km$
\\ $\Rightarrow a+b=(a-c)+(b-c)+2c=k^2+m^2+2km=(k+m)^2$
\\ Vậy $a+b$ là số chính phương.}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Thái Nguyên, 2016]%[9H3G3]
	Cho đường tròn tâm $O$ và dây cung $AB$. Từ một điểm $M$ bất kỳ trên đường tròn ($M$ khác $A$ và $B$), kẻ $MH \perp AB$ tại $H$. Gọi $E$, $F$ lần lượt là hình chiếu vuông góc của $H$ trên $MA$, $MB$. Qua $M$ kẻ đường thẳng vuông góc với $EF$, cắt dây cung $AB$ tại $D$. Chứng minh rằng: $\dfrac{MA^2}{MB^2}=\dfrac{AH}{BD}.\dfrac{AD}{BH}.$
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
		\tkzInit[xmin=-5,ymin=-3,xmax=5,ymax=3]
		\tkzClip
		\tkzDefPoints{0/0/O, -2/-1/A, 2/-1/B, -0.8/2.1/M}
		\tkzDrawPoints(O,A,B,M)
		\tkzLabelPoints[below left](O)
		\tkzLabelPoints(B)
		\tkzLabelPoints[below left](A)
		\tkzLabelPoints[above](M)
		\tkzDrawCircle[radius](O,A)
		\tkzDrawSegments(A,B)
		\tkzDrawSegments(M,A M,B)
		\tkzDrawAltitude(A,B)(M)
		\tkzDefPoints{-0.79/-1/H}
		\tkzDrawPoints(H)
		\tkzLabelPoints[below](H)
		\tkzDrawAltitude(M,A)(H)
		\tkzDrawAltitude(M,B)(H)
		\tkzDefPointBy[projection= onto A--M](H)
		\tkzGetPoint{E}
		\tkzDefPointBy[projection= onto B--M](H)
		\tkzGetPoint{F}
		\tkzDrawPoints(E,F)
		\tkzLabelPoints[left](E)
		\tkzLabelPoints[right](F)
		\tkzDrawSegments(E,F)

		\tkzDrawPoints(D)
		\tkzLabelPoints[below](D)
		\tkzDrawSegments(M,D)
		\tkzDefPointBy[projection= onto A--M](D)
		\tkzGetPoint{E'}
		\tkzDefPointBy[projection= onto B--M](D)

		\tkzDrawPoints(E',F')

		\tkzLabelPoints[right](F')

		\tkzMarkRightAngle(D,F',M)
		\tkzMarkRightAngle(H,E,M)
		\tkzMarkRightAngle(H,F,M)
		\tkzMarkRightAngle(M,H,A)
		\end{tikzpicture}

	Gọi $E',F'$ lần lượt là hình chiếu vuông góc của $D$ trên $MA,MB$.
\\ Ta có:
\begin{align*}
	&\dfrac{AH}{BD}=\dfrac{S_{MAH}}{S_{MBD}}=\dfrac{HE.MA}{DF'.MB};~ \dfrac{AD}{BH}=\dfrac{S_{MAD}}{S_{MBH}} =\dfrac{DE'.MA}{HF.MB}
	\\ & \dfrac{AD}{BD}.\dfrac{AH}{BH}=\dfrac{MA^2}{MB^2}.\dfrac{HE.DE'}{DF'.HF} ~(1)
\end{align*}
Ta có: $\widehat{EHF} = \widehat{E'DF'}$ (vì cùng bù với góc $\widehat{AMB}$)
\\ Ta lại có $\widehat{HFE}=\widehat{E'DF'}$ (hai góc có cạnh tương ứng vuông góc)
\\ Mà $\widehat{DMF'} = \widehat{DE'F'}$ (cùng chắn cung $DF'$)
\\ $\Rightarrow \widehat{HFE} =\widehat{DE'F}$.
\\ Suy ra hai tam giác $HEF$ và $DF'E'$ đồng dạng 
\\ $\Rightarrow \dfrac{HE}{HF}=\dfrac{DF'}{DE'} \Rightarrow HE.DE'=HF.DF' \Rightarrow \dfrac{HE.DE'}{HF.DF'}=1$ $(2)$
\\ Từ $(1)$ và $(2)$ ta được $\dfrac{MA^2}{MB^2}=\dfrac{AH}{BD}.\dfrac{AD}{BH}$.}
\end{ex}

\begin{ex}%[Đề thi vào 10 chuyên Thái Nguyên, 2016]%[9H3G7]%[9H2G7]
	Cho tam giác $ABC$ vuông tại $A$, đường cao $AH$. Gọi $(O)$ là đường tròn ngoại tiếp tam giác $AHC$. Trên cung nhỏ $AH$ của $(O)$ lấy điểm $M$ bất kỳ khác $A$ và $H$. Trên tiếp tuyến của $M$ tại $(O)$ lấy hai điểm $D$, $E$ sao cho $BD=BE=BA$. Đường thẳng $BM$ cắt $(O)$ tại điểm thứ hai $N$.
	\begin{enumerate}
		\item Chứng minh rằng tứ giác $BDNE$ nội tiếp được trong một đường tròn.
		\item Chứng minh đường tròn ngoại tiếp tứ giác $BDNE$ và đường tròn $(O)$ tiếp xúc nhau.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}
			\tkzInit[xmin=-1.5,ymin=-4,xmax=6,ymax=3.5]
			\tkzClip
			\tkzDefPoint(0,0){B}
			\tkzDefShiftPoint[B](0:5){C}
			\tkzDefShiftPoint[B](0:5){C}
			\tkzDefShiftPoint[B](40:4){a}
			\tkzDefPointBy[projection=onto B--a](C) \tkzGetPoint{A}
			\tkzDefPointBy[projection=onto B--C](A) \tkzGetPoint{H}
			\tkzDefCircle[circum](A,H,C) 			\tkzGetPoint{O}
			\tkzCalcLength[cm](O,A) 				\tkzGetLength{rOA}
			\tkzCalcLength[cm](B,A) 				\tkzGetLength{rBA}
			\tkzDefShiftPoint[O](170:\rOA){M}
			\tkzTangent[at = M](O) 				    \tkzGetPoint{d}
			\tkzInterLC[R](M,d)(B,\rBA cm)			\tkzGetPoints{D}{E}
			\tkzInterLC(B,M)(O,A)					\tkzGetPoints{}{N}
			\tkzDrawCircle(O,A)	
			\tkzDrawLine[add=1.5 and 4.5](M,d)
			\tkzMarkRightAngle(O,M,D)
			\tkzDrawSegments(A,B A,C B,C B,N B,D B,E O,M)
			\tkzDrawPoints[fill=white,color=black,size=8pt](A,B,C,H,M,D,E,O,N)
			\node at ($(A)+(80:3mm)$){\footnotesize $A$};
			\node at ($(B)+(200:3mm)$){\footnotesize $B$};
			\node at ($(C)+(-50:3mm)$){\footnotesize $C$};
			\node at ($(H)+(-80:3mm)$){\footnotesize $H$};
			\node at ($(M)+(240:5mm)$){\footnotesize $M$};
			\node at ($(D)+(180:3mm)$){\footnotesize $D$};
			\node at ($(E)+(-10:3mm)$){\footnotesize $E$};
			\node at ($(O)+(10:3mm)$){\footnotesize $O$};
			\node at ($(N)+(90:3mm)$){\footnotesize $N$};	
			\end{tikzpicture}
		\end{center}
	\begin{enumerate}
		\item[a.] Vì $AH \perp BC \Rightarrow \widehat{AHC} = 90^\circ$. Suy ra $AC$ là đường kính của đường tròn $(O)$. Mà $\Delta ABC$ vuông tại $A$ nên $AB$ là tiếp tuyến của $(O)$. 
		\\ Xét các tam giác $BAM$ và $BNA$ có: $\widehat{ABN}$ chung và $\widehat{BAM} = \widehat{BNA}$ (có số đo bằng nửa số đo cung $AM$). Vậy $\triangle BAM$ và $\triangle BNA$ đồng dạng $\Rightarrow \dfrac{BM}{BA}=\dfrac{BA}{BN} \Rightarrow BM.BN=BA^2$.
		\\ Theo giả thiết $BA=BD \Rightarrow BM.BN=BD^2 \Rightarrow \dfrac{BM}{BD}=\dfrac{BD}{BN}$.
		\\ Mặt khác $\widehat{DBN}$ là góc chung của $\triangle BDM$ và $\triangle BND$.
		\\ Vậy nên $\triangle BDM$ $\backsim$ $\triangle BND$  $\Rightarrow \widehat{BDM} = \widehat{BND}$.
		\\ Ta có $BD=BE \Rightarrow \triangle BDE$ cân tại $B \Rightarrow \widehat{BDM} = \widehat{BED} \Rightarrow \widehat{BND} = \widehat{BED}$.
		\\ $\Rightarrow$ Tứ giác $BDNE$ nội tiếp được trong một đường tròn.
		\item[b.] Kí hiệu đường tròn ngoại tiếp tứ giác $BDNE$ là $(O')$.
		\\ Giả sử $(O)$ và $(O')$ không tiếp xúc nhau. Vì $(O)$ và $(O')$ có điểm chung $N$ nên chúng điểm chung thứ hai là $N'$ và $BN'$ cắt $DE$ tại điểm $M'$ khác $M$.
		\\ Các tam giác $BDM'$ và $BN'D$ có: $\widehat{DBN'}$ chung, $\widehat{BN'D} =\widehat{BED}$ (hai góc nội tiếp cung chắn cung $DB$ của $(O')$), $\widehat{BED} = \widehat{BDE} \Rightarrow \widehat{BN'D}= \widehat{BDE}$
		\\ Vậy nên các tam giác $BDM'$ và $BN'D$ đồng dạng
		\\ $\Rightarrow \dfrac{BM'}{BD}=\dfrac{BD}{BN'}\Rightarrow BM'.BN'=BD^2$.
		\\ Ta có $BM.BN=BD^2$ nên $BM'.BN'=BM.BN \Rightarrow \dfrac{BM}{BN'}=\dfrac{BM'}{BN}$.
		\\ Hai tam giác $BMM'$ và $BN'N$ có $\widehat{NBN'}$ chung và $\dfrac{BM}{BN'}=\dfrac{BM'}{BN}$ nên chúng đồng dạng $\Rightarrow \widehat{BMM'}=\widehat{BN'N} \Rightarrow \widehat{BN'N} + \widehat{M'MN}=180^\circ$
		\\ $\Rightarrow$ Tứ giác $M'MNN'$ nội tiếp đường tròn.
		\\ Đường tròn ngoại tiếp $M'MNN'$ và $(O)$ có chung nhau ba điểm $M,N,N'$ nên chúng trùng nhau $\Rightarrow$ $M'$ là điểm chung thứ hai của $(O)$ và tiếp tuyến $DE$. Điều này vô lý nên giả sử sai.
		\\ Vậy đường tròn ngoại tiếp tứ giác $BDNE$ và đường tròn $(O)$ tiếp xúc nhau.
\end{enumerate}}
\end{ex}
