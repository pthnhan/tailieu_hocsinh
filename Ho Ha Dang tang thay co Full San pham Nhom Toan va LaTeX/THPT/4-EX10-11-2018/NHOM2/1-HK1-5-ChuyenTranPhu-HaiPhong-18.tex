\begin{ex}%[HK1, Chuyên Trần Phú, Hải Phòng, 2018]%[1H2B3]
	Cho các đường thẳng $a, b, c$ và các mặt phẳng $(\alpha), (\beta)$. Giả thiết nào sau đây đủ để kết luận đường thẳng $a$ song song với đường thẳng $b$?
	\choice
	{$a\cap b=\varnothing$}
	{$\heva{a&\parallel c \\ b&\parallel c}$}
	{$\heva{a&\parallel (\alpha) \\ b&\parallel (\alpha)}$}
	{\True $\heva{a\parallel (\alpha), a\parallel (\beta) \\ (\alpha)\cap (\beta)=b}$}
	\loigiai{
		Nếu $a\cap b=\varnothing$ thì $a\parallel b$ hoặc $a, b$ chéo nhau.\\
		Nếu $\heva{a&\parallel c \\ b&\parallel c}$ thì $a\parallel b$ hoặc $a\equiv b$.\\
		Nếu $\heva{a&\parallel (\alpha) \\ b&\parallel (\beta)}$ thì không kết luận được quan hệ giữa $a$ và $b$.
	}
\end{ex}
