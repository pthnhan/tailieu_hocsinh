Cho các biểu thức $M=\left(\dfrac{x+3}{x-9}+\dfrac{1}{\sqrt x+3}\right)$ và $N=\dfrac{\sqrt x}{\sqrt x-3}$ với $x>0$; $x\ne9$.
\begin{enumerate}
	\item Tính giá trị của biểu thức $N$ khi $x=4$.
	\item Rút gọn biểu thức $B=M:N$.
	\item Chứng minh $B >\dfrac{1}{3}$.
\end{enumerate}
\loigiai{
	\begin{enumerate}
		\item Thay $x=4$ thỏa mãn điều kiện vào biểu thức $N$, ta được
		\[N=\dfrac{\sqrt 4}{\sqrt 4-3}=\dfrac{2}{-1}=-2.\]
		\item Ta có
		{\allowdisplaybreaks
			\begin{eqnarray*}
				B=M:N & = & \left(\dfrac{x+3}{x-9}+\dfrac{1}{\sqrt x+3}\right):\dfrac{\sqrt x}{\sqrt x-3}\\
				& = & \left[\dfrac{x+3}{(\sqrt x-3)(\sqrt x+3)}+\dfrac{1}{\sqrt x+3}\right]:\dfrac{\sqrt x}{\sqrt x-3}\\
				& = & \dfrac{x+3+\sqrt x-3}{(\sqrt x-3)(\sqrt x+3)}:\dfrac{\sqrt x}{\sqrt x-3}\\
				& = & \dfrac{\sqrt x (\sqrt x+1)}{(\sqrt x-3)(\sqrt x+3)} \cdot \dfrac{\sqrt x-3}{\sqrt x}=\dfrac{\sqrt x+1}{\sqrt x+3}.
			\end{eqnarray*}
		}
		\item Xét hiệu
		{\allowdisplaybreaks
			\begin{eqnarray*}
				B-\dfrac{1}{3} & = & \dfrac{\sqrt x+1}{\sqrt x+3}-\dfrac{1}{3}=\dfrac{3\sqrt x+3-\sqrt x-3}{3(\sqrt x+3)}\\
				& = & \dfrac{2\sqrt x}{3(\sqrt x+3)}.
			\end{eqnarray*}
		}
		Vì $x>0$ nên $\sqrt x > 0$ và $\sqrt x + 3 >0$ suy ra $\dfrac{2\sqrt x}{3(\sqrt x+3)} > 0$ hay $B-\dfrac{1}{3}>0\Leftrightarrow B>\dfrac{1}{3}$.\\
		Vậy $B >\dfrac{1}{3}$.
	\end{enumerate}
}

	\item $\sqrt{4x^2+4x+1}=6$.
	\item $\sqrt{4x+20}+\sqrt{x+5}-\dfrac{1}{3}\sqrt{9x+45}=4$.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Điều kiện xác định $x \in \mathbb{R}$.
	{\allowdisplaybreaks
		\begin{eqnarray*}
			\sqrt{4x^2+4x+1}=6 & \Leftrightarrow & \sqrt{(2x+1)^2}=6\Leftrightarrow |2x+1|=6\\
			& \Leftrightarrow & \hoac{& 2x+1=6\\& 2x+1=-6} \Leftrightarrow \hoac{& x=\dfrac{5}{2}\\& x=\dfrac{-7}{2}.}
		\end{eqnarray*}
	}
	So với điều kiện, thỏa mãn.\newline
	Vậy tập nghiệm của phương trình là $S=\left\{\dfrac{5}{2};\dfrac{-7}{2}\right\}$.
	\item $\sqrt{4x+20}+\sqrt{x+5}-\dfrac{1}{3}\sqrt{9x+45}=4$.\newline
	Điều kiện $ x\ge-5$.
	{\allowdisplaybreaks
		\begin{eqnarray*}
			\sqrt{4x+20}+\sqrt{x+5}-\dfrac{1}{3}\sqrt{9x+45}=4 & \Leftrightarrow &  \sqrt{4(x+5)}+\sqrt{x+5}-\dfrac{1}{3}\sqrt{9(x+5)}=4 \\
			& \Leftrightarrow & 2\sqrt{x+5}+\sqrt{x+5}-\dfrac{1}{3} \cdot 3\sqrt{x+5}=4\\
			& \Leftrightarrow & 2\sqrt{x+5}=4\Leftrightarrow x+5=4 \\
			& \Leftrightarrow & x=-1 .
		\end{eqnarray*}
	}
	So với điều kiện, thỏa mãn.\\
	Vậy tập nghiệm của phương trình là $S=\left\{-1\right\}$.
\end{enumerate}
}

Cho đường thẳng $(d) \colon y=\left(k+1\right)x+k$.
\begin{enumerate}
	\item Tìm giá trị của $k$ để đường thẳng $(d)$ đi qua điểm $A(1;2)$.
	\item Tìm giá trị của $k$ để đường thẳng $(d)$ song song với đường thẳng $y=2x+3$.
	\item Tìm điểm cố định mà $(d)$ luôn đi qua với mọi $k$.
\end{enumerate}
\loigiai{
\begin{enumerate}
	\item Vì đường thẳng $(d)$ đi qua điểm $A(1;2)$ nên thay $x=1$ và $y=2$ vào phương trình $y=(k+1)x+k$ ta được
	\[(k+1) \cdot 1+k=2 \Leftrightarrow k=\dfrac{1}{2}.\]
	\item Đường thẳng $(d)$ song song với đường thẳng $ y=2x+3$ khi và chỉ khi
	\[\heva{& k+1=2\\& k\ne 3} \Leftrightarrow k=1.\]
	Vậy $k=1$ thì đường thẳng $(d)$ song song với đường thẳng $y=2x+3$.
	\item Gọi $M(x_0;y_0)$ là điểm cố định mà $(d)$ luôn đi qua.\\
	Thay $x=x_0$ và $ y=y_0$ vào phương trình $y=(k+1)x+k$, ta được
	\[(k+1)x_0+k=y_0\Leftrightarrow kx_0+x_0+k=y_0\Leftrightarrow (x_0+1)k+x_0-y_0=0. \tag{1}\]
	Để $(d)$ luôn đi qua với mọi $k$ thì $(1)$ luôn đúng với mọi $k$.
	\[\Leftrightarrow \heva{& x_0+1=0\\& x_0-y_0=0} \Leftrightarrow \heva{& x_0=-1\\& y_0=-1.}\]
	Vậy $(d)$ luôn đi qua điểm cố định $M(-1;-1)$ với mọi $k$.
\end{enumerate}
}

Cho $AC$ là đường kính của đường tròn tâm $(O;R)$. Trên tiếp tuyến tại $A$ của $(O;R)$, lấy điểm $I$ sao cho $IA$ lớn hơn $R$. Từ $I$ vẽ tiếp tuyến thứ $2$ với $(O;R)$ với tiếp điểm là $B$. Qua $O$ kẻ đường thẳng vuông góc với $AC$, cắt đường thẳng $BC$ tại $H$.

		\item Chứng minh $BC\parallel OI$.
		\item Chứng minh rằng tứ giác $AOHI$ là hình chữ nhật.
		\item Tia $OB$ cắt $IH$ tại $K$. Chứng minh tam giác $IOK$ cân.

\begin{enumerate}
	\item Xét $(O;R)$ có $AI$ và $BI$ là các tiếp tuyến cắt nhau tại $I$ nên $IA=IB$ (tính chất $2$ tiếp tuyến cắt nhau).\\
	Lại có $OA=OB(=R)$ do đó $IO$ là đường trung trực của $AB$ suy ra $AB \perp OI$. \tagEX{1}
	Vì $\triangle ABC$ nội tiếp đường tròn đường kính $AC$, tâm $O$ nên $OA=OB=OC=\dfrac{AC}{2}$ suy ra $\triangle ABC$ vuông tại $B$ (tính chất tam giác có trung tuyến bằng một nửa cạnh đối diện).\\
	Suy ra $\widehat {ABC} = 90^\circ \Leftrightarrow AB \perp BC$. \tagEX{2}
	Từ $(1)$ và $(2)$ ta có
	\[ \left. \begin{aligned}
	& AB \perp OI\\
	& AB \perp BC
	\end{aligned} \right\} \Rightarrow OI\parallel BC \text{ (từ vuông góc đến song song)}.\]
	Suy ra điều phải chứng minh.
	\item 
	Xét $\triangle AIO$ và $\triangle OHC$ có
	\[ \left. \begin{aligned}
	& \widehat{IAO}=\widehat{HOC}=90^\circ\\
	& AO=OC\left(=R\right)\\
	& \widehat{IOA}=\widehat{HCO} \text{ (so le trong)}
	\end{aligned} \right\} \Rightarrow \triangle AIO=\triangle OHC \text{ (g-c-g)}.\]
	Tứ giác $IOCH$ có
	\[ \left. \begin{aligned}
	& OI\parallel HC\\
	& OI=HC
	\end{aligned} \right\} \Rightarrow IOCH \text{ là hình bình hành}.\]
	Suy ra $IH \parallel OC\parallel AC$ mà $HO \perp AC\Rightarrow \widehat{OHI}=90^\circ$.\\ 
	Xét tứ giác $AOHI$ ta có
	\[ \left. \begin{aligned}
	& \widehat{IAO}=90^\circ \text{ (vì $AI$ là tiếp tuyến của $(O;R)$ tại $A$)}\\
	& \widehat{AOH}=90^\circ \text{ (vì $HO \perp AC$)}\\
	& \widehat{OHI}=90^\circ \text{ (chứng minh trên)}
	\end{aligned} \right\} \Rightarrow AOHI \text{ là hình chữ nhật}.\]
	Suy ra điều phải chứng minh.
	\item Vì tứ giác $AOHI$ là hình chữ nhật (chứng minh trên) suy ra $\widehat{HIA}=90^\circ $ (tính chất).\\
	Ta có $\widehat{OIK}=90^\circ-\widehat{AIO}$. \tagEX{3}
	Mà tam giác $ AOI$ vuông tại $A$ nên $\widehat{AOI}=90^\circ-\widehat{AIO}$. \tagEX{4}

	Mà $\widehat{IOK}=\widehat{AOI}$ (tính chất hai tiếp tuyến cắt nhau) do đó $\widehat{IOK}=\widehat{OIK}$.\\
	Suy ra $\triangle IOK$ cân tại $K$.

	Vì $IA$ và $IB$ là các tiếp tuyến của $(O;R)$ nên $OI$ là đường trung trực của $AB$ suy ra $AB \perp OI$ hay $AE$ là đường cao của tam giác vuông $IAO$.\\
	Áp dụng hệ thức lượng vào tam giác vuông $IAO$ có
	\[\dfrac{1}{AE^2}=\dfrac{1}{IA^2}+\dfrac{1}{AO^2}=\dfrac{1}{(2R)^2}+\dfrac{1}{R^2}=\dfrac{5}{4R^2}\Rightarrow AE=\dfrac{2R}{\sqrt 5}\Rightarrow AB=\dfrac{4R}{\sqrt 5}.\]
	Áp dụng định lý Py-ta-go vào tam giác vuông $ABC$, ta có
	\[BC^2=AC^2-AB^2=4R^2-\dfrac{16R^2}{5}=\dfrac{4R^2}{5}\Rightarrow BC=\dfrac{2R}{\sqrt 5}.\]
	Diện tích tam giác $ABC$ là
	\[S_{ABC}=\dfrac{1}{2}AB \cdot BC=\dfrac{1}{2} \cdot \dfrac{4R}{\sqrt 5} \cdot \dfrac{2R}{\sqrt 5}=\dfrac{4R^2}{\sqrt 5}.\]
\end{enumerate}

	(0,5 điểm) Cho $a$, $b$, $c$ là các số dương thỏa mãn điều kiện $a+b+c=1$. Tìm giá trị nhỏ nhất của biểu thức
	\[A=\dfrac{\left(1+a\right)\left(1+b\right)\left(1+c\right)}{\left(1-a\right)\left(1-b\right)\left(1-c\right)}.\]
	\loigiai{
		Vì $a$, $b$, $c>0$ và $a+b+c=1$ nên $\heva{& 1-a=b+c>0\\& 1-b=a+c>0\\& 1-c=a+b>0.}$\\
		Ta có $1+a=1+(1-b-c)=(1-b)+(1-c)$.\\
		Áp dụng bất đẳng thức Cô-si cho $2$ số dương $1-b$ và $1-c$ ta có
		\[(1-b)+(1-c)\ge 2\sqrt{\left(1-b\right)\left(1-c\right)} \Leftrightarrow 1+a \ge 2\sqrt{\left(1-b\right)\left(1-c\right)}. \tag{1}\]
		Tương tự ta có
		\[1+b \ge 2\sqrt{(1-a)(1-c)}. \tag{2}\]
		\[1+c \ge 2\sqrt{(1-b)(1-a)}. \tag{3}\]
		Nhân các vế của $(1)$, $(2)$ và $(3)$ ta có
		\[(1+a)(1+b)(1+c) \ge 8\sqrt{(1-a)^2(1-b)^2(1-c)^2}=8(1-a)(1-b)(1-c).\]
		Do đó $\dfrac{(1+a)(1+b)(1+c)}{(1-a)(1-b)(1-c)}\ge 8$.\\
		Dấu ''='' xảy ra $1-a=1-b=1-c \Leftrightarrow a=b=c$ mà $a+b+c=1 \Leftrightarrow a=b=c=\dfrac{1}{3}$.\\
		Vậy GTNN của $A$ là $8$ khi và chỉ khi $a=b=c=\dfrac{1}{3}$.
}
