\section{Số đo góc}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Đo góc}
\begin{listEX}
	\item Dụng cụ đo: Thước đo góc.
	\item Cách đo góc $xOy$.
\end{listEX}
\begin{enumerate}[Bước 1.]
	\item Đặt thước đo góc sao cho tâm của thước trùng với đỉnh $O$ của góc, một cạnh của góc đi qua vạch $0^\circ$.
	\item Xem cạnh thứ hai của góc đi qua vạch nào của thước. Giả sử đó là vạch $105^\circ$, ta viết $\widehat{xOy}=105^\circ$.
\end{enumerate}
\begin{note}
	Mỗi góc có một số đo dương. Số đo của góc bẹt là $180^\circ$. Số đo của mỗi góc không vượt quá $180^\circ$.
\end{note}

\subsubsection{So sánh hai góc}
\begin{dn}
	Nếu hai góc $\widehat{A}$ và $\widehat{B}$ có số đo bằng nhau thì hai góc đó bằng nhau. Ta viết $\widehat{A}=\widehat{B}$.
\end{dn}
\begin{dn}
	Nếu số đo góc $A$ nhỏ hơn số đo góc $B$ thì góc $A$ nhỏ hơn góc $B$. Ta viết $\widehat{A}<\widehat{B}$.
\end{dn}

\subsubsection{Góc vuông, góc nhọn, góc tù}
\begin{dn}
	Góc có số đo bằng $90^\circ$ là góc vuông.
\end{dn}
\begin{dn}
	Góc có số đo nhỏ hơn góc vuông là góc nhọn.
\end{dn}
\begin{dn}
	Góc có số đo lớn hơn $90^\circ$ nhưng nhỏ hơn $180^\circ$ là góc tù.
\end{dn}

\end{enumerate}
\end{dang}

\section{Khi nào thì $\widehat{xOy}+\widehat{yOz}=\widehat{xOz}$?}
\subsection{Tóm tắt lí thuyết}
\subsubsection{Tính chất cộng số đo hai góc}
\begin{dl}
	\immini{
		Nếu tia $Oy$ nằm giữa hai tia $Ox$ và $Oz$ thì $\widehat{xOy}+\widehat{yOz}=\widehat{xOz}$. Ngược lại, nếu $\widehat{xOy}+\widehat{yOz}=\widehat{xOz}$ thì tia $Oy$ nằm giữa hai tia $Ox$ và $Oz$.
	}{
		\begin{tikzpicture}
			\tkzDefPoints{0/0/O, 2/0/A, 2/0.7/B, 1/2/C}
			\tkzDrawLine[add=0 and .2](O,A)
			\tkzDrawLine[add=0 and .2](O,B)
			\tkzDrawLine[add=0 and .2](O,C)
			\tkzLabelLine[pos=1.2, below](O,A){$x$}
			\tkzLabelLine[pos=1.1, above](O,B){$y$}
			\tkzLabelLine[pos=1.2, left](O,C){$z$}
			\tkzLabelPoints[below left](O)
			\tkzMarkAngles(B,O,C)
			\tkzMarkAngles[mark=|, size=0.9](A,O,B)
		\end{tikzpicture}
	}
\end{dl}
\begin{dl}
	Nếu $\widehat{xOy}+\widehat{yOz} \ne \widehat{xOz}$ thì tia $Oy$ không nằm giữa hai tia $Ox$ và $Oz$.
\end{dl}

	}
\end{hq}
\subsubsection{Hai góc kề nhau, phụ nhau, bù nhau}
\begin{dn}
	Hai góc kề nhau là hai góc có một cạnh chung và hai cạnh còn lại nằm trên hai nửa mặt phẳng đối nhau bờ chứa cạnh chung.
\end{dn}
\begin{dn}
	Hai góc phụ nhau là hai góc có tổng số đo bằng $90^\circ$.
\end{dn}
\begin{dn}
	Hai góc bù nhau là hai góc có tổng số đo bằng $180^\circ$.
\end{dn}
\begin{note}
	\begin{listEX}
		\item Hai góc kề bù là hai góc vừa kề nhau, vừa bù nhau. Hai góc kề bù có tổng số đo bằng $180^\circ$.
		\item Hai góc cùng phụ (hoặc cùng bù) với một góc thứ ba thì bằng nhau.
	\end{listEX}
\end{note}
\subsection{Các dạng toán}
\begin{dang}{Tính số đo góc}
Nếu tia $Oy$ nằm giữa hai tia $Ox$ và $Oz$ thì $\widehat{xOy}+\widehat{yOz}=\widehat{xOz}$.
\end{dang}
\begin{vd}%[6H2B4]
Cho tia $Om$ nằm giữa hai tia $Ox$ và $Oy$. Biết $\widehat{xOm}=33^\circ$ và $\widehat{yOm}=67^\circ$. Tính $\widehat{xOy}$.
\loigiai{
	\immini{
		Vì tia $Om$ nằm giữa hai tia $Ox$ và $Oy$ nên\\ $\widehat{xOy}=\widehat{xOm}+\widehat{yOm}$\\
		$\widehat{xOy}=33^\circ+67^\circ$\\
		$\widehat{xOy}=100^\circ$.
	}{
		\begin{tikzpicture}
			\tkzDefPoints{0/0/O, 2/0/A}
			\tkzDefPointBy[rotation= center O angle 33](A) \tkzGetPoint{B}
			\tkzDefPointBy[rotation= center O angle 67](B) \tkzGetPoint{C}
			\tkzDrawLines[add=0 and .2](O,A O,B O,C)
			\tkzLabelLine[pos=1.1, below](O,A){$x$}
			\tkzLabelLine[pos=1.2, left](O,B){$m$}
			\tkzLabelLine[pos=1.2, below left](O,C){$y$}
			\tkzMarkAngle[size=.5](A,O,B)
			\tkzMarkAngle[arc=ll,size=.5 cm](B,O,C)
			\tkzLabelAngle(A,O,B){$33^\circ$}
			\tkzLabelAngle(B,O,C){$67^\circ$}
			\tkzLabelPoints[below left](O)
		\end{tikzpicture}
	}
}
\end{vd}
\begin{vd}%[6H2B4]
	\immini{
		Cho hình vẽ sau, biết tia $Ox$ và $Oy$ đối nhau, $\widehat{xOm}=55^\circ$, $\widehat{yOn}=65^\circ$, tia $On$ nằm giữa hai tia $Oy$ và $Om$. Tính $\widehat{mOn}$.
	}{
		\begin{tikzpicture}
		\tkzDefPoints{0/0/O, 2/0/A, -2/0/D}
		\tkzDefPointBy[rotation= center O angle 65](A) \tkzGetPoint{B}
		\tkzDefPointBy[rotation= center O angle -55](D) \tkzGetPoint{C}
		\tkzDrawLines[add=0 and .2](O,A O,B O,C O,D)
		\tkzLabelLine[pos=1.1, below](O,A){$y$}
		\tkzLabelLine[pos=1.2, left](O,B){$n$}
		\tkzLabelLine[pos=1.2, below left](O,C){$m$}
		\tkzLabelLine[pos=1.1, below](O,D){$x$}
		\tkzMarkAngle[size=.5](A,O,B)
		\tkzMarkAngle[arc=ll,size=.5 cm](C,O,D)
		\tkzLabelAngle(A,O,B){$65^\circ$}
		\tkzLabelAngle(C,O,D){$55^\circ$}
		\tkzLabelPoints[below](O)
		\end{tikzpicture}
	}
	\loigiai{
		Vì tia $Ox$, $Oy$ là hai tia đối nhau nên $\widehat{xOy}=180^\circ$.\\
		Mà $\widehat{xOm}+\widehat{mOn}+\widehat{nOy}=\widehat{xOy}$ nên $55^\circ+\widehat{mOn}+65^\circ=180^\circ$.\\
		Suy ra $\widehat{mOn}=60^\circ$.
	}
\end{vd}
\begin{vd}%[6H2B4]
	\immini{
		Trên hình bên, số đo góc $xOy$ nhỏ hơn số đo góc $zOy$ là $80^\circ$. Tính số đo của hai góc đó.
	}{
		\begin{tikzpicture}
		\tkzDefPoints{0/0/O, 2/0/A, -2/0/C}
		\tkzDefPointBy[rotation= center O angle 130](A) \tkzGetPoint{B}
		\tkzDrawLines[add=0 and .2](O,A O,B O,C)
		\tkzLabelLine[pos=1.1, below](O,A){$z$}
		\tkzLabelLine[pos=1.2, right](O,B){$y$}
		\tkzLabelLine[pos=1.1, below](O,C){$x$}
		\tkzLabelPoints[below](O)
		\end{tikzpicture}
	}
	\loigiai{
		$\widehat{xOy}$ và $\widehat{yOz}$ là hai góc kề bù nên $\widehat{xOy}+\widehat{yOz}=180^\circ$.\\
		Mà $\widehat{yOz}-\widehat{xOy}=80^\circ$ nên $\widehat{xOy}=(180^\circ - 80^\circ):2=50^\circ$ và $\widehat{yOz}=180^\circ - 50^\circ = 130^\circ$.
	}
\end{vd}
\begin{dang}{Xác định hai góc phụ nhau, bù nhau}
Phương pháp giải
\begin{itemize}
\item Hai góc có tổng số đo bằng $90^\circ$ là hai góc phụ nhau.
\item Hai góc có tổng số đo bằng $180^\circ$ là hai góc bù nhau.
\end{itemize}
\end{dang}
\begin{vd}%[6H2B4]
	Biết $\widehat{A}$ và $\widehat{B}$ phụ nhau và $\widehat{A}=65^\circ$. Tính số đo góc $B$.
	\loigiai{
		$\widehat{A}$ phụ với $\widehat{B}$ nên $\widehat{A}+\widehat{B}=90^\circ$.\\
		Mà $\widehat{A}=65^\circ$ nên $\widehat{B}=90^\circ - 65^\circ =25^\circ$.
	}
\end{vd}
\begin{vd}%[6H2B4]
	Cho hai góc $\widehat{A}$ và $\widehat{B}$ bù nhau. Biết $\widehat{A}=70^\circ$. Tính số đo góc $B$.
	\loigiai{
		$\widehat{A}$ bù với $\widehat{B}$ nên $\widehat{A}+\widehat{B}=180^\circ$.\\
		Mà $\widehat{A}=70^\circ$ nên $\widehat{B}=180^\circ - 70^\circ =110^\circ$.
	}
\end{vd}
\begin{vd}%[6H2B4]
	\immini{
		Cho hình vẽ bên, biết tia $Ox$ và $Oy$ đối nhau, $\widehat{xOa}=63^\circ$, $\widehat{yOb}=27^\circ$, tia $Ob$ nằm giữa hai tia $Oy$ và $Oa$.
		\begin{listEX}
			\item Viết tên các góc nhọn, góc vuông, góc tù.
			\item Viết tên cặp góc phụ nhau.
			\item Viết tên cặp góc bù nhau.
		\end{listEX}
	}{
		\begin{tikzpicture}
			\tkzDefPoints{0/0/O, 2/0/A, -2/0/D}
			\tkzDefPointBy[rotation= center O angle 27](A) \tkzGetPoint{B}
			\tkzDefPointBy[rotation= center O angle -63](D) \tkzGetPoint{C}
			\tkzDrawLines[add=0 and .2](O,A O,B O,C O,D)
			\tkzLabelLine[pos=1.1, below](O,A){$y$}
			\tkzLabelLine[pos=1.2, above](O,B){$b$}
			\tkzLabelLine[pos=1.2, below left](O,C){$a$}
			\tkzLabelLine[pos=1.1, below](O,D){$x$}
			\tkzMarkAngle[size=.5](A,O,B)
			\tkzMarkAngle[arc=ll,size=.5 cm](C,O,D)
			\tkzLabelAngle(A,O,B){$27^\circ$}
			\tkzLabelAngle(C,O,D){$63^\circ$}
			\tkzLabelPoints[below](O)
		\end{tikzpicture}
	}
	\loigiai{
		\begin{enumerate}
			\item
			\begin{itemize}
				\item Ta có $\widehat{xOa}+\widehat{aOy}=\widehat{xOy}$\\
				$63^\circ+\widehat{aOy}=180^\circ$\\
				$\widehat{aOy}=117^\circ$.
			 	\item Ta có $\widehat{xOb}+\widehat{bOy}=\widehat{xOy}$\\
				$\widehat{xOb}+27^\circ=180^\circ$\\
				$\widehat{xOb}=153^\circ$.
				\item Tia $Ob$ nằm giữa hai tia $Oy$ và $Oa$, ta có\\
				$\widehat{aOb}=\widehat{yOa}-\widehat{yOb}=117^\circ - 27^\circ=90^\circ$.
				\item Như vậy
				$\widehat{yOb}=27^\circ$; $\widehat{xOa}=63^\circ$ là các góc nhọn.\\
				$\widehat{xOb}=153^\circ$; $\widehat{yOa}=117^\circ$ là các góc tù.\\
				$\widehat{aOb}=90^\circ$ là góc vuông.
			\end{itemize}
			\item $\widehat{xOa}+\widehat{yOb}=63^\circ + 27^\circ = 90^\circ$ nên $\widehat{xOa}$ và $\widehat{yOb}$ là cặp góc phụ nhau.
			\item $\widehat{xOa}+\widehat{aOy}=180^\circ$ nên $\widehat{xOa}$ và $\widehat{aOy}$ là cặp góc bù nhau.\\
			$\widehat{xOb}+\widehat{yOb}=180^\circ$ nên $\widehat{xOb}$ và $\widehat{yOb}$ là cặp góc bù nhau.
		\end{enumerate}
	}
\end{vd}
\begin{dang}{Xác định một tia có nằm giữa hai tia còn lại hay không}
	Phương pháp giải
	\begin{itemize}
		\item Nếu $\widehat{xOy}+\widehat{yOz}=\widehat{xOz}$ thì tia $Oy$ nằm giữa hai tia $Ox$ và $Oz$.
		\item Nếu $\widehat{xOy}+\widehat{yOz}\ne \widehat{xOz}$ thì tia $Oy$ không nằm giữa hai tia $Ox$ và $Oz$.
	\end{itemize}
\end{dang}
\begin{vd}%[6H2B4]
	Cho $3$ tia chung gốc $Ox$, $Oy$, $Oz$. Biết $\widehat{xOy}=25^\circ$; $\widehat{yOz}=45^\circ$; $\widehat{zOx}=70^\circ$. Tia nào nằm giữa hai tia còn lại?
	\loigiai{
		Ta có $\widehat{xOy}+\widehat{yOz}=25^\circ + 45^\circ = 70^\circ$, suy ra $\widehat{xOy}+\widehat{yOz}=\widehat{zOx}$.\\
		Do đó, tia $Oy$ nằm giữa hai tia $Ox$ và $Oz$.
	}
\end{vd}
\begin{note}
	Bạn nên tính tổng hai góc có số đo nhỏ $\widehat{xOy}$, $\widehat{yOz}$. Sau đó so sánh kết quả với số đo góc lớn nhất $\widehat{xOz}$. Nếu kết quả bằng nhau thì tia $Oy$ nằm giữa hai tia $Ox$ và $Oz$.
\end{note}
\begin{vd}%[6H2B4]
	Cho ba tia chung gốc $Ox$, $Oy$, $Oz$ thỏa mãn $\widehat{xOy}=130^\circ$, $\widehat{yOz}=120^\circ$, $\widehat{zOx}=110^\circ$. Hỏi có tia nào nằm giữa hai tia còn lại không?
	\loigiai{
		\begin{itemize}
			\item $\widehat{xOy}+\widehat{yOz}>\widehat{xOz}$ (vì $130^\circ + 120^\circ > 110^\circ$) nên tia $Oy$ không nằm giữa hai tia $Ox$ và $Oz$.
			\item $\widehat{xOy}+\widehat{xOz}>\widehat{yOz}$ (vì $130^\circ + 110^\circ > 120^\circ$) nên tia $Ox$ không nằm giữa hai tia $Oy$ và $Oz$.
			\item $\widehat{yOz}+\widehat{xOz}>\widehat{xOy}$ (vì $120^\circ + 110^\circ > 130^\circ$) nên tia $Oz$ không nằm giữa hai tia $Ox$ và $Oy$.
		\end{itemize}
		Vậy không có tia nào nằm giữa hai tia còn lại.
	}
\end{vd}
\begin{note}
	Không phải cứ ba tia chung gốc thì có một tia nằm giữa hai tia còn lại.
\end{note}
