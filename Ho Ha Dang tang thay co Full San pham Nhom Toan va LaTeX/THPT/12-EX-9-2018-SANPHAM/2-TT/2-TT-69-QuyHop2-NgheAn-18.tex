\choice
{$w=5+2i$}

{\True $w=\dfrac{9}{2}+2i$}
{$w=\dfrac{9}{2}-2i$}
\loigiai{

\begin{align*}
(1)&\Leftrightarrow a-bi+i=(2-i)(a+bi-1)\\
&\Leftrightarrow a-bi+i=2a+2bi-2-ai-bi^2+i\\

&\Leftrightarrow \heva{&2-a-b=0\\&a-3b=0}\Leftrightarrow \heva{&a=\dfrac{3}{2}\\&b=\dfrac{1}{2}}
