\begin{bt}%[HK1 Trường Chinh, HCM]%[Phạm An Bình - DA 11HK1-18]%[1D1B3-1]
	Giải phương trình lượng giác sau $2\cos^2x-\sqrt{2} \cos x-2=0$.
	\loigiai{
			Đặt $t=\cos x$ $(-1\le t\le 1)$, phương trình trở thành
			$$2t^2-\sqrt{2} t -2=0 \Leftrightarrow \hoac{&t=\sqrt{2}&\text{(loại)}\\&t=-\dfrac{\sqrt{2}}{2}&\text{(nhận).}}$$\\
			Với $t=-\dfrac{\sqrt{2}}{2}$ thì
			$\cos x = -\dfrac{\sqrt{2}}{2}
			\Leftrightarrow \cos x = \cos \left( \dfrac{3}{4}\pi \right)
			\Leftrightarrow \hoac{&x=\dfrac{3}{4}\pi + k2\pi \\&x=-\dfrac{3}{4}\pi + k2\pi} (k\in \mathbb{Z})$.
	}
\end{bt}
\begin{bt}%[HK1 Trường Chinh, HCM]%[Phạm An Bình - DA 11HK1-18]%[1D1B3-3]
	Giải phương trình lượng giác sau $2 \cos x - 2\sin x = 1$.
	\loigiai{
		\begin{eqnarray*}
			2 \cos x - 2\sin x = 1
			&\Leftrightarrow &\dfrac{1}{\sqrt{2}} \cos x - \dfrac{1}{\sqrt{2}} \sin x = \dfrac{1}{2\sqrt{2}}\\
			&\Leftrightarrow &\cos \dfrac{\pi}{4} \cos x - \sin \dfrac{\pi}{4} \sin x = \dfrac{1}{2\sqrt{2}}\\
			&\Leftrightarrow &\cos \left( \dfrac{\pi}{4} + x \right)= \dfrac{1}{2\sqrt{2}}\\
			&\Leftrightarrow &\hoac{&\dfrac{\pi}{4} + x = \arccos \dfrac{1}{2\sqrt{2}} + k2\pi \\&\dfrac{\pi}{4} + x = -\arccos \dfrac{1}{2\sqrt{2}} + k2\pi} (k\in \mathbb{Z})\\
			&\Leftrightarrow &\hoac{&x = \arccos \dfrac{1}{2\sqrt{2}} -\dfrac{\pi}{4} + k2\pi \\&x = -\arccos \dfrac{1}{2\sqrt{2}} -\dfrac{\pi}{4} + k2\pi} (k\in \mathbb{Z}).
		\end{eqnarray*}
	}
\end{bt}
\begin{bt}%[HK1 Trường Chinh, HCM]%[Phạm An Bình - DA 11HK1-18]%[1D1K3-4]
	Giải phương trình lượng giác sau $\sin^6x + \cos^6 x = 2\left( \sin^8 x + \cos^8 x \right)$.
	\loigiai{
		\begin{eqnarray*}
			\sin^6x + \cos^6x = 2\left( \sin^8x + \cos^8x \right)
			&\Leftrightarrow &\sin^6x(1-2\sin^2x) + \cos^6x(1-2\cos^2x) = 0\\
			&\Leftrightarrow &\sin^6x\cos2x - \cos^6x\cos2x = 0\\
			&\Leftrightarrow &\cos2x (\sin^6x - \cos^6x) = 0\\
			&\Leftrightarrow &\cos2x (\sin^2x - \cos^2x) (\sin^4x + \sin^2x\cos^2x + \cos^4x) = 0\\
			&\Leftrightarrow &-\cos^22x (1 - \sin^2x\cos^2x) = 0\\
			&\Leftrightarrow &\cos^22x (4 - \sin^22x) = 0\\
			&\Leftrightarrow &\hoac{&\cos^22x=0\\&4 - \sin^22x = 0}\\
			&\Leftrightarrow &\hoac{&\cos2x=0\\&\sin^22x = 4\text{ (vô lý vì $\sin^22x\le 1$)}}\\
			&\Leftrightarrow &2x=\dfrac{\pi}{2}+k\pi \quad (k\in \mathbb{Z})\\
			&\Leftrightarrow &x=\dfrac{\pi}{4}+\dfrac{k\pi}{2} \quad (k\in \mathbb{Z}).
		\end{eqnarray*}
	}
\end{bt}

\begin{bt}%[HK1 Trường Chinh, HCM]%[Phạm An Bình - DA 11HK1-18]%[1D2B1-2]
		Từ các chữ số $1$; $2$; $5$; $6$; $7$ có thể lập được bao nhiêu số tự nhiên chẵn có $3$ chữ số khác nhau?
	\loigiai{
			Có $2$ cách chọn chữ số hàng đơn vị. Có $4$ cách chọn chữ số hàng chục. Có $3$ cách chọn chữ số hàng trăm. Vậy có $2\cdot 4\cdot 3=24$ số tự nhiên thỏa yêu cầu.	
	}
\end{bt}

\begin{bt}%[HK1 Trường Chinh, HCM]%[Phạm An Bình - DA 11HK1-18]%[1D2B5-2]
	Gieo ngẫu nhiên $1$ con súc sắc cân đối và đồng chất hai lần. Gọi $x$, $y$ lần lượt là số chấm của lần gieo thứ nhất và thứ hai. Tính xác suất để $\left| x-y \right| = 3$.
	\loigiai{
		Gọi $\Omega$ là không gian mẫu.\\
		$A$ là biến cố có được $\left| x-y \right| = 3$ suy ra $A=\left\lbrace (1;4); (2;5); (3;6); (4;1); (5;2); (6;3) \right\rbrace$.\\
		Xác suất cần tìm là $\mathrm{P}(A)=\dfrac{\left|A\right|}{\left| \Omega \right|}=\dfrac{6}{36}=\dfrac{1}{6}$.	
	}
\end{bt}

\begin{bt}%[HK1 Trường Chinh, HCM]%[Phạm An Bình - DA 11HK1-18]%[1D2K2-1]
	Cho tam giác $ABC$. Trên các cạnh $AB$, $BC$, $CA$ lần lượt lấy $3$ điểm phân biệt, $4$ điểm phân biệt và $2$ điểm phân biệt (không có điểm nào trùng $A$, $B$ và $C$). Có bao nhiêu tam giác có đỉnh là các điểm trên (khác $A$, $B$, $C$)?
	\loigiai{
		\begin{enumerate}[TH 1.]
			\item $3$ đỉnh nằm trên $3$ cạnh khác nhau, có $3\cdot 4\cdot 2=24$ tam giác.
			\item $2$ đỉnh nằm trên cùng $1$ cạnh, đỉnh còn lại nằm trên cạnh khác.
			\begin{itemize}
				\item Chọn 2 đỉnh nằm trên cạnh $AB$ có $\mathrm{C}_3^2$ cách, chọn đỉnh còn lại nằm trên cạnh khác có $4+2=6$ cách.
				\item Chọn 2 đỉnh nằm trên cạnh $BC$ có $\mathrm{C}_4^2$ cách, chọn đỉnh còn lại nằm trên cạnh khác có $3+2=5$ cách.
				\item Chọn 2 đỉnh nằm trên cạnh $CA$ có $\mathrm{C}_2^2$ cách, chọn đỉnh còn lại nằm trên cạnh khác có $3+4=7$ cách.
			\end{itemize}
			Vậy có $\mathrm{C}_3^2 6+\mathrm{C}_4^2 5+\mathrm{C}_2^2 7 = 55$ tam giác.
		\end{enumerate}
		Vậy có $24+55=79$ tam giác thỏa yêu cầu.	
	}
\end{bt}

\begin{bt}%[HK1 Trường Chinh, HCM]%[Phạm An Bình - DA 11HK1-18]%[1D2K2-6]
	Tìm số tự nhiên $n$ biết $\mathrm{C}_n^2 + \mathrm{A}_n^2 = 108$.
	\loigiai{
			Điều kiện: $n\ge 2$ và $n\in \mathbb{N}$.
			\begin{eqnarray*}
				\mathrm{C}_n^2 + \mathrm{A}_n^2 = 108
				&\Leftrightarrow &\dfrac{n!}{2!(n-2)!} + \dfrac{n!}{(n-2)!} = 108\\
				&\Leftrightarrow &\dfrac{1}{2}n(n-1) + n(n-1) = 108\\
				&\Leftrightarrow &n(n-1) = 72\\
				&\Leftrightarrow &n^2-n-72=0\\
				&\Leftrightarrow &\hoac{&n=9&\text{(nhận)}\\&n=-8&\text{(loại).}}
			\end{eqnarray*}
	}
\end{bt}
\begin{bt}%[HK1 Trường Chinh, HCM]%[Phạm An Bình - DA 11HK1-18]%[1D2B3-2]
	Tìm số hạng chứa $x^{10}$ trong khai triển của biểu thức $\left( x+x^2 \right)^8$.
	\loigiai{
		Khai triển nhị thức Newton ta có
		$$\left( x+x^2 \right)^8=\sum_{k=0}^{8} \mathrm{C}_8^k x^{8-k} (x^2)^k=\sum_{k=0}^{8} \mathrm{C}_8^k x^{8+k}.$$
		Số hạng chứa $x^{10}$ thỏa $8+k=10\Leftrightarrow k=2$.\\
		Vậy số hạng cần tìm là $\mathrm{C}_8^2 x^{10}=28x^{10}$.
	}
\end{bt}

\begin{bt}%[HK1 Trường Chinh, HCM]%[Phạm An Bình - DA 11HK1-18]%[1D3K1-1]
	Bằng phương pháp quy nạp, chứng minh rằng
	$$\dfrac{1}{1\cdot 3}+\dfrac{1}{3\cdot 5}+\dfrac{1}{5\cdot 7}+\cdots +\dfrac{1}{(2n-1)\cdot (2n+1)} = \dfrac{n}{(2n+1)}, \forall n\in \mathbb{N}^*.$$
	\loigiai{
		\begin{itemize}
			\item Với $n=1$ thì $\dfrac{1}{1\cdot 3}=\dfrac{1}{2\cdot 1+1}$ đúng.
			\item Giả sử đẳng thức đúng với $n=k$. Ta chứng minh đẳng thức cũng đúng với $n=k+1$. Có
			\begin{eqnarray*}
				& &\dfrac{1}{1\cdot 3}+\cdots +\dfrac{1}{(2k-1)\cdot (2k+1)} = \dfrac{k}{(2k+1)}\\
				&\Leftrightarrow &\dfrac{1}{1\cdot 3}+\cdots +\dfrac{1}{(2k-1)\cdot (2k+1)}+\dfrac{1}{(2k+1)\cdot (2k+3)} = \dfrac{k}{(2k+1)}+\dfrac{1}{(2k+1)\cdot (2k+3)}\\
				&\Leftrightarrow &\dfrac{1}{1\cdot 3}+\cdots +\dfrac{1}{(2k-1)\cdot (2k+1)}+\dfrac{1}{[2(k+1)-1]\cdot [2(k+1)+1]} = \dfrac{k(2k+3)+1}{(2k+1)\cdot (2k+3)}\\
				&\Leftrightarrow &\dfrac{1}{1\cdot 3}+\cdots +\dfrac{1}{(2k-1)\cdot (2k+1)}+\dfrac{1}{[2(k+1)-1]\cdot [2(k+1)+1]} = \dfrac{2k^2+3k+1}{(2k+1)\cdot (2k+3)}\\
				&\Leftrightarrow &\dfrac{1}{1\cdot 3}+\cdots +\dfrac{1}{(2k-1)\cdot (2k+1)}+\dfrac{1}{[2(k+1)-1]\cdot [2(k+1)+1]} = \dfrac{(2k+1)(k+1)}{(2k+1)\cdot (2k+3)}\\
				&\Leftrightarrow &\dfrac{1}{1\cdot 3}+\cdots +\dfrac{1}{(2k-1)\cdot (2k+1)}+\dfrac{1}{[2(k+1)-1]\cdot [2(k+1)+1]} = \dfrac{k+1}{[2(k+1)+1]}.
			\end{eqnarray*}
			Suy ra đẳng thức đúng với $n=k+1$.
			\item Áp dụng nguyên lý quy nạp, ta có điều phải chứng minh.
		\end{itemize}
	}
\end{bt}

	Cho cấp số cộng $(u_n)$ biết $u_1=\dfrac{1}{3}$ và công sai $d=-\dfrac{1}{3}$. Tính tổng $10$ số hạng đầu của $(u_n)$.
	\loigiai{
			$S_{10}=10u_1+\dfrac{10(10-1)}{2}d=10\left(\dfrac{1}{3}\right)+\dfrac{10(10-1)}{2}\left(-\dfrac{1}{3}\right)=-\dfrac{35}{3}$.
	}

\begin{bt}%[HK1 Trường Chinh, HCM]%[Phạm An Bình - DA 11HK1-18]%[1D3K3-4]
	Cho $4$ số $2x+1$, $3y$, $x-1$, $-y$ theo thứ tự là một cấp số cộng. Tìm $x$ và $y$.
	\loigiai{
		Do $4$ số $2x+1$, $3y$, $x-1$, $-y$ theo thứ tự là một cấp số cộng, nên ta có hệ phương trình\\
		$\heva{&3y=\dfrac{2x+1+x-1}{2} \\&x-1=\dfrac{3y-y}{2}}\Leftrightarrow \heva{&\dfrac{3}{2}x-3y=0 \\&x-y=1}\Leftrightarrow \heva{&x=2 \\&y=1.}$
	}
\end{bt}

\begin{bt}%[HK1 Trường Chinh, HCM]%[Phạm An Bình - DA 11HK1-18]%[1H2K1-2]%[1H2K2-3]%[1H2K3-2]%[1H2K3-2]
	Cho hình chóp $S.ABCD$ có đáy $ABCD$ là hình bình hành tâm $O$. Gọi $M$, $N$, $P$ lần lượt là trung điểm của các cạnh $SA$, $SB$ và $CD$.
	\begin{enumerate}
		\item Xác định giao tuyến của hai mặt phẳng $(SBP)$ và $(SAD)$.
		\item Tìm giao điểm của đường thẳng $BC$ và mặt phẳng $(OMN)$.
		\item Chứng minh rằng $PO$ song song $(SCB)$.
		\item Chứng minh rằng $MP$ song song $(SCB)$.
	\end{enumerate}
	\loigiai{
		\begin{center}
		\begin{tikzpicture}
		\tkzDefPoints{0/0/A, 4/0/B, -2/-2/D}
		\tkzDefPointBy[translation=from A to D](B)\tkzGetPoint{C}
		\tkzInterLL(A,C)(B,D)\tkzGetPoint{O}
		\tkzDefLine[perpendicular=through O](A,B)\tkzGetPoint{S}
		\tkzDefMidPoint(S,A)\tkzGetPoint{M}
		\tkzDefMidPoint(S,B)\tkzGetPoint{N}
		\tkzDefMidPoint(C,D)\tkzGetPoint{P}
		\tkzDefMidPoint(C,B)\tkzGetPoint{R}
		\tkzInterLL(B,P)(A,D)\tkzGetPoint{T}
		\tkzDrawSegments(B,C C,P S,B S,C S,T S,P P,T)
		\tkzDrawSegments[dashed](A,B S,A S,D A,C B,D A,T B,P D,P O,R P,O O,M O,N M,N)
		\tkzDrawPoints[size = 5pt](S,A,B,C,D,O,M,N,P,T,R)
		\tkzLabelPoints[left](A,D,T,O)
		\tkzLabelPoints[below left](M)
		\tkzLabelPoints[below](P)
		\tkzLabelPoints[right](C,B,N,R)
		\tkzLabelPoints[above](S)
		\tkzMarkSegments[mark=s||, size=2pt](S,M M,A)
		\tkzMarkSegments[mark=s|](S,N N,B)
		\tkzMarkSegments[mark=x](D,P P,C)
		\tkzMarkSegments[mark=|||](B,R R,C)
		\end{tikzpicture}
		\end{center}
		
		\begin{enumerate}
			\item Xác định giao tuyến của hai mặt phẳng $(SBP)$ và $(SAD)$.\\
			Trong mp $(ABCD)$, gọi T là giao điểm của $AD$ và $BP$.\\
			Có $T\in BP\subset (SBP)$ và $T\in AD\subset (SAD)$ nên $T\in (SBP)\cap (SAD)$\\Mặt khác có $S\in (SBP)\cap (SAD)$, suy ra $ST=(SBP)\cap (SAD)$.
				
			\item Tìm giao điểm của đường thẳng $BC$ và mặt phẳng $(OMN)$.\\
			Gọi $R$ là trung điểm của $BC$, suy ra $OR$ là đường trung bình của $\triangle ABC$.\\
			Mặt khác, có $MN$ là đường trung bình của $\triangle SAB$ suy ra $MN\parallel AB\parallel OR$.\\
			Vậy $R\in (OMN)$ và $R\in BC$, suy ra $R=BC\cap (OMN)$.				

			\item Chứng minh rằng $PO$ song song $(SCB)$.\\
			$PO$ là đường trung bình của $\triangle BCD$ nên $PO\parallel BC\subset (SCB)$, mặt khác có $P\notin (SCB)$ suy ra $PO\parallel (SCB)$.
			
			\item Chứng minh rằng $MP$ song song $(SCB)$.\\
			Do $P$ là trung điểm của $DC$ và $ABCD$ là hình bình hành nên $PC=\dfrac{1}{2}DC=\dfrac{1}{2}AB$ và $PC\parallel AB$.\\
			Mặt khác, do $MN$ là đường trung bình $\triangle SAB$ nên $MN=\dfrac{1}{2}AB$ và $MN\parallel AB$.\\
			Suy ra, $MN=PC$ và $MN\parallel PC$.\\
			Suy ra, $MNCP$ là hình bình hành.\\
			Suy ra, $MP\parallel NC\subset (SCB)$. Mà $M\notin (SCB)$ nên $MP\parallel (SCB)$.
		\end{enumerate}
	}
