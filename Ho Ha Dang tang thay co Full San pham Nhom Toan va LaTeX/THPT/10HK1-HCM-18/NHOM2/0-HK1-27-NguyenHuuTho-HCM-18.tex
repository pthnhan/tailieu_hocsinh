\begin{bt}%[Đề HK1, Nguyễn Hữu Thọ, Hồ Chí Minh 17-18]%[Trần Bá Huy]%[0D3B2-4]
Giải phương trình $\sqrt{x^2-x+1}=2x-1$.
\loigiai{
Ta có
$$\sqrt{x^2-x+1}=2x-1\Leftrightarrow \heva{& 2x-1\geq 0\\ & x^2-x+1=(2x-1)^2}\Leftrightarrow \heva{& x\geq \dfrac{1}{2}\\ & -3x^2+3x=0}
\Leftrightarrow \heva{& x\geq \dfrac{1}{2}\\ & \hoac{&x=0\ (\text{loại})\\ & x=1\ (\text{nhận}).}}$$
Vậy phương trình có một nghiệm $x=1$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Hữu Thọ, Hồ Chí Minh 17-18]%[Trần Bá Huy]%[0D3B2-4]
Giải phương trình $\sqrt{3x-2}=\sqrt{6-x}$.
\loigiai{
Ta có
$$\sqrt{3x-2}=\sqrt{6-x}\Leftrightarrow \heva{& 6-x\geq 0\\ & 3x-2=6-x}\Leftrightarrow \heva{& x\leq 6\\ & x=2\ (\text{nhận}).}$$
Vậy phương trình có một nghiệm $x=2$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Hữu Thọ, Hồ Chí Minh 17-18]%[Trần Bá Huy]%[0D3K2-4]
Giải phương trình $2x^2+\sqrt{x^2+6}=9$.
\loigiai{
Đặt $t=\sqrt{x^2+6},\ t>0$. Ta có $t^2=x^2+6\Rightarrow x^2=t^2-6$. Phương trình đã cho trở thành
$$2(t^2-6)+t=9\Leftrightarrow 2t^2+t-21=0\Leftrightarrow \hoac{& t=3\ (\text{nhận})\\ & t=-\dfrac{7}{2}\ (\text{loại}).}$$
Với $t=3$, suy ra $\sqrt{x^2+6}=3\Leftrightarrow x^2+6=9\Leftrightarrow x=\pm \sqrt{3}$.\\
Vậy phương trình có hai nghiệm $x=\pm \sqrt{3}$. 
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Hữu Thọ, Hồ Chí Minh 17-18]%[Trần Bá Huy]%[0D3B2-6]
Giải và biện luận theo tham số $m$ phương trình $$(m^2-3m+2)x=m-2.$$
\loigiai{
\begin{itemize}
\item {\bf TH1.} $m^2-3m+2\neq 0\Leftrightarrow \heva{&m\neq 1\\ &m\neq 2}$, phương trình có nghiệm duy nhất $$x=\dfrac{m-2}{m^2-3m+2}=\dfrac{m-2}{(m-2)(m-1)}=\dfrac{1}{m-1}.$$
\item {\bf TH2.} $m=1$. Thay vào phương trình đã cho, ta được $0x=-1$. Phương trình vô nghiệm.
\item {\bf TH3.} $m=2$. Thay vào phương trình đã cho, thu được $0x=0$. Phương trình có nghiệm với mọi $x\in\mathbb{R}$.
\end{itemize}
}
\end{bt}


\begin{bt}%[Đề HK1, Nguyễn Hữu Thọ, Hồ Chí Minh 17-18]%[Trần Bá Huy]%[0D3K2-5]
Tìm $m$ để phương trình $x^2-2(m+1)x+m^2+1=0$ có hai nghiệm phân biệt $x_1$, $x_2$ thỏa $x_1^2+x_2^2=26$.
\loigiai{
Ta có $\Delta'=(m+1)^2-(m^2+1)=2m$. Do đó, phương trình có hai nghiệm phân biệt $x_1$, $x_2$ khi và chỉ khi $\Delta'>0\Leftrightarrow 2m>0\Leftrightarrow m>0$.\\
Khi đó, theo định lí Vi-ét $\heva{& x_1+x_2=2(m+1)\\ & x_1x_2=m^2+1.}$\\
Theo giả thiết, ta có 
\begin{align*}
x_1^2+x_2^2=26\Leftrightarrow\ & (x_1+x_2)^2-2x_1x_2=26\Leftrightarrow 4(m+1)^2-2(m^2+1)=26\\
\Leftrightarrow\ & m^2+4m-12=0\Leftrightarrow \hoac{& m=2\ (\text{nhận})\\ & m=-6\ (\text{loại}).}
\end{align*}
Vậy $m=2$.
}
\end{bt}


\begin{bt}%[Đề HK1, Nguyễn Hữu Thọ, Hồ Chí Minh 17-18]%[Trần Bá Huy]%[0D3K3-4]
Giải hệ phương trình sau 
$$\heva{& x+y=xy+1\\ &  x^2+y^2=5.}$$
%$$\begin{cases} \end{cases}$$
\loigiai{
Hệ đã cho tương đương $\heva{& x+y=xy+1\\ & (x+y)^2-2xy=5.}$\\
Đặt $\heva{& S=x+y\\ & P=xy}$, điều kiện $S^2\geq 4P$. Hệ trở thành
$$\heva{& S=P+1\\ & S^2-2P=5}\Leftrightarrow \heva{& P=S-1\\ & S^2-2S-3=0}\Leftrightarrow \hoac{& S=-1,\ P=-2\ (\text{nhận})\\ & S=3,\ P=2\ (\text{nhận}).}$$
\begin{itemize}
\item Với $\heva{& S=-1\\ &P=-2}\Leftrightarrow \heva{& x+y=-1\\ & xy=-2}$. Khi đó $x,\ y$ là hai nghiệm của phương trình $$X^2+X-2=0\Leftrightarrow \hoac{& X=1\\ & X=-2.}\Rightarrow \hoac{& (x;y)=(1;-2)\\ & (x;y)=(-2;1).}$$
\item Với $\heva{& S=3\\ &P=2}\Leftrightarrow \heva{& x+y=3\\ & xy=2}$. Khi đó $x,\ y$ là hai nghiệm của phương trình $$X^2-3X+2=0\Leftrightarrow \hoac{& X=1\\ & X=2.}\Rightarrow \hoac{& (x;y)=(1;2)\\ & (x;y)=(2;1).}$$
\end{itemize}
Vậy hệ có $4$ nghiệm $(x;y)$ là $(1;-2)$, $(-2;1)$, $(1;2)$ và $(2;1)$.
}
\end{bt}


\begin{bt}%[Đề HK1, Nguyễn Hữu Thọ, Hồ Chí Minh 17-18]%[Trần Bá Huy]%[0H2B2-1]

Từ đó \\
$\overrightarrow{AB}\cdot\overrightarrow{AH}=AB\cdot AH\cdot \cos\left(\overrightarrow{AB};\overrightarrow{AH}\right)=6\cdot 3\sqrt{3}\cdot \cos 30^\circ=27$.
}{

\begin{bt}%[Đề HK1, Nguyễn Hữu Thọ, Hồ Chí Minh 17-18]%[Trần Bá Huy]%[0H1B4-3]
Trong mặt phẳng $Oxy$, cho $\triangle ABC$ với $A(1;0)$, $B(-2;3)$, $C(5;4)$. Tìm tọa độ điểm $D$ sao cho tứ giác $ABCD$ là hình bình hành.
\loigiai{
Gọi $D(x_D;y_D)$. Ta có $\overrightarrow{BC}=(7;1),\ \overrightarrow{AD}=(x_D-1;y_D)$.\\
Tứ giác $ABCD$ là hình bình hành khi
$$\overrightarrow{BC}=\overrightarrow{AD}\Leftrightarrow \heva{& x_D-1=7\\ & y_D=1}\Leftrightarrow \heva{& x_D=8\\ & y_D=1.}$$
Vậy $D(8;1)$.
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Hữu Thọ, Hồ Chí Minh 17-18]%[Trần Bá Huy]%[0H2K2-3]
Trong mặt phẳng $Oxy$, cho $\triangle ABC$ với $A(1;0)$, $B(-2;3)$, $C(5;4)$. Tìm tọa độ điểm $H$ là chân đường cao kẻ từ $A$ của tam giác $ABC$.
\loigiai{
Gọi $H(x;y)$. Ta có $\overrightarrow{BC}=(7;1)$, $\overrightarrow{AH}=(x-1;y)$, $\overrightarrow{BH}=(x+2;y-3)$.\\
Có $B,\ C,\ H$ thẳng hàng và $AH$ vuông góc với $BC$ nên
$$\heva{& (x+2)\cdot 1-(y-3)\cdot 7=0\\ & 7\cdot (x-1)+1\cdot y=0}\Leftrightarrow \heva{& x-y=-23\\ & 7x+y=7}\Leftrightarrow \heva{& x=\dfrac{13}{25}\\ & y=\dfrac{84}{25}}\Rightarrow H\left(\dfrac{13}{25};\dfrac{84}{25}\right).$$
}
\end{bt}

\begin{bt}%[Đề HK1, Nguyễn Hữu Thọ, Hồ Chí Minh 17-18]%[Trần Bá Huy]%[0D4K1-2]
Cho số thực $a$ tùy ý. Chứng minh rằng $\dfrac{a^2+6}{\sqrt{a^2+2}}\geq 4.$
\loigiai{
Ta có $a^2+2>0$ với mọi $a$ tùy ý.\\
Biến đổi vế trái và sử dụng bất đẳng thức Cô-si, ta được\\
$\text{VT}=\dfrac{(a^2+2)+4}{\sqrt{a^2+2}}=\sqrt{a^2+2}+\dfrac{4}{\sqrt{a^2+2}}\geq 2\sqrt{\sqrt{a^2+2}\cdot\dfrac{4}{\sqrt{a^2+2}}}=4$.\\
Vậy bất đẳng thức được chứng minh xong.
}
\end{bt}
