\begin{enumerate}
\item Tìm điều kiện xác định của biểu thức $\dfrac{1}{\sqrt{2-x}}-\dfrac{5}{x-1}$.
\item Tìm tọa độ giao điểm $M$ của đường thẳng $y=2x+3$ với trục $Oy$.
\item Với những giá trị nào của $m$ thì hàm số $y=(1-m^2)x+2017m$ đồng biến?
\item Tam giác đều $ABC$ có diện tích hình tròn ngoại tiếp bằng $3\pi$ cm$^2$. Tính độ dài cạnh.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Điều kiện xác định của biểu thức là $\heva{2-x > 0\\ x-1 \neq 0} \Leftrightarrow \heva{x < 2\\ x \neq 1}$.
\item $x=0 \Rightarrow y=3 \Rightarrow M(0;3)$.
\item Để hàm số đồng biến thì $1-m^2>0 \Leftrightarrow |m|<1 \Leftrightarrow -1<m<1$.
\item Diện tích hình tròn là $3\pi$ cm$^2$ nên bán kính đường tròn là $R= \sqrt{3}$ (cm).\\
Kẻ trung tuyến $AM$, gọi $G$ là trọng tâm của $\Delta ABC$. Vì $\Delta ABC$ đều nên $GA=\dfrac{2}{3} AM$ cũng là bán kính đường tròn ngoại tiếp $\Delta ABC$.\\
$\Rightarrow \dfrac{2}{3} AM=\sqrt{3}$ (cm) $\Rightarrow AM=\dfrac{3\sqrt{3}}{2}$ cm.\\
Mà $\Delta AMB$ vuông tại $M$ nên $\sin B=\dfrac{AM}{AB}\Leftrightarrow AB= \dfrac{AM}{\sin 60^{\circ}}=3$ cm.
\end{enumerate}
}

\begin{ex}%[Đề thi vào 10, Chuyên Lê Hồng Phong Nam Định (Vòng 1), 2017]%[9D1B8]
Cho biểu thức $A=\dfrac{x-1}{(x+ \sqrt{x})(x- \sqrt{x}+1)}:\dfrac{1}{x^2+ \sqrt{x}}$ (Với $x>0$).
\begin{enumerate}
\item Rút gọn biểu thức $A$.
\item Tìm các giá trị nguyên của $x$ để $\dfrac{1}{A}$ là một số nguyên.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Điều kiện: $x>0$.
$$A=\dfrac{(x-1}{\sqrt{x}(\sqrt{x} +1)(x- \sqrt{x} +1)} \cdot  \sqrt{x}(\sqrt{x} +1)(x- \sqrt{x}+1)=x-1.$$
\item Do $x\in\mathbb{Z}$ nên để $\dfrac{1}{A}=\dfrac{1}{x-1}$ là số nguyên thì $(x-1)$ là các ước nguyên của $1$.\\
$\Rightarrow (x-1)\in\{\pm1\} \Rightarrow x=2$ (Vì $x>0$).
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Lê Hồng Phong Nam Định (Vòng 1), 2017]%[9D4Y5]%[9D4K6]%[9D4G7]
\hfill
\begin{enumerate}[1)]
\item Cho phương trình $x^2-2mx+m^2-m+1=0$ (1) (Với $m$ là tham số).
\begin{enumerate}[a)]
\item Giải phương trình với $m=2$.
\item Tìm tất cả các giá trị của $m$ để phương trình (1) có hai nghiệm phân biệt $x_1$, $x_2$ thỏa mãn $x_1^2+2mx_2-3m^2+m-5\le 0$.
\end{enumerate}
\item Giải hệ phương trình $\heva{\sqrt{x} +2 \sqrt{x+3} =7- \sqrt{x^2+3} \\ \sqrt{x+y} + \sqrt{7-y} =y^2-6y+13}$.
\end{enumerate}
\loigiai{
\begin{enumerate}[1)]
\item 
\begin{enumerate}[a)]
\item Với $m=2$ phương trình trở thành $x^2-4x+3=0 \Leftrightarrow \hoac{x=1\\ x=3}$.
\item $\Delta ' = m-1$. Để phương trình có hai nghiệm phân biệt $x_1,x_2$ thì $\Delta ' >0 \Leftrightarrow m>1$.\\
Khi đó theo Vi-ét ta có $\heva{x_1+x_2=2m\\ x_1x_2=m^2-m+1}$.\\
Vì $x_1$ là nghiệm của phương trình nên $x_1^2-2mx_1+m^2-m+1=0 \Leftrightarrow x_1^2=2mx_1-m^2+m-1$.\\
Do đó
\begin{eqnarray*}
&& x_1^2+2mx_2-3m^2+m-5\le 0\\
&\Leftrightarrow & 2mx_1-m^2+m-1+2mx_2-3m^2+m-5\le 0\\
&\Leftrightarrow & 2m(x_1+x_2)-4m^2+2m-6\le 0\\
&\Leftrightarrow & 2m.2m-4m^2+2m-6\le 0\\
&\Leftrightarrow & m\le 3
\end{eqnarray*}
Vậy $1<m\le 3$.
\end{enumerate}
\item Điều kiện: $x\ge 0$, $y\le 7$, $x+y\ge 0$.\\
Giải phương trình thứ nhất
\begin{eqnarray*}
&&\sqrt{x} + 2\sqrt{x+3} =7- \sqrt{x^2+3}\\
&\Leftrightarrow & \sqrt{x} -1+ 2\sqrt{x+3} -4+ \sqrt{x^2+3} -2=0\\
&\Leftrightarrow & \dfrac{x-1}{\sqrt{x}+1} + \dfrac{2(x-1)}{\sqrt{x+3} +2} + \dfrac{x^2-1}{\sqrt{x^2+3} +2}=0\\
&\Leftrightarrow & (x-1)\left(\dfrac{1}{\sqrt{x}+1} + \dfrac{2}{\sqrt{x+3} +2} + \dfrac{x+1}{\sqrt{x^2+3} +2}\right)=0\\
&\Leftrightarrow & x=1 \qquad \text{(Thỏa mãn)}.
\end{eqnarray*}
Thế $x=1$ giải phương trình thứ hai
\begin{eqnarray*}
&&\sqrt{y+1} + \sqrt{7-y} =y^2-6y+13\\
&\Leftrightarrow &(y-3)^2+4- \sqrt{y+1} - \sqrt{7-y} =0\\
&\Leftrightarrow &(y-3)^2+ \dfrac{1}{2} \left(8- 2\sqrt{y+1} - 2\sqrt{7-y} \right) =0\\
&\Leftrightarrow &(y-3)^2 + \dfrac{1}{2} \left( \sqrt{y+1} - 2 \right)^2 + \dfrac{1}{2} \left( \sqrt{7-y} - 2\right)^2=0\\
&\Leftrightarrow & y=3 \qquad \text{(Thỏa mãn)}.
\end{eqnarray*}
Vậy nghiệm hệ là $(x;y)=(1;3)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Lê Hồng Phong Nam Định (Vòng 1), 2017]%[9H3K7]%[9H3K7]
Cho tam giác $ABC$ có ba góc nhòn nội tiếp đường tròn $(O)$, $AB<AC$. Các tiếp tuyến của đường tròn $(O)$ tại $B$, $C$ cắt nhau tại $M$. Đường thẳng qua $M$ song song với $AB$ cắt đường tròn $(O)$ tại $D$ và $E$ ($D$ thuộc cung nhỏ $BC$), cắt $BC$ tại $F$, cắt $AC$ tại $I$.
\begin{enumerate}
\item Chứng minh 5 điểm $M,B,O,I,C$ cùng thuộc một đường tròn.
\item Chứng minh $\dfrac{FI}{FE}=\dfrac{FD}{FM}$.
\item Đường thẳng $OI$ cắt $(O)$ tại $P$ và $Q$ ($P$ thuộc cung nhỏ $AB$), đường thẳng $QF$ cắt $(O)$ tại $T$ ($T$ khác $Q$). Tính tỉ số $\dfrac{TQ^2+TM^2}{MQ^2}$.
\end{enumerate}
\loigiai{
\begin{center}
\begin{tikzpicture}[scale=.5]
\tkzInit[xmin=-13,xmax=10,ymin=-9, ymax=9] \tkzClip[space=.2]
\tkzDefPoint(-5,0){O}
\tkzDefPoint(9,0){M}
\tkzDefPoint(-12,-4){A}
\tkzDrawCircle(O,A)
\tkzTangent[from=M](O,A) \tkzGetPoints{B}{C}
\tkzDrawSegments(M,B M,C A,C A,B B,C)
\tkzDefLine[parallel=through M](C,A)\tkzGetPoint{x}
\tkzInterLC(M,x)(O,A)
\tkzGetPoints{E}{D}
\tkzDrawSegment[](M,E)
\tkzInterLL(A,B)(E,M)
\tkzGetPoint{I}
\tkzInterLL(B,C)(E,M)
\tkzGetPoint{F}
\tkzInterLC(O,I)(O,A)
\tkzGetPoints{P}{Q}
\tkzDrawSegment[](P,Q)
\tkzInterLC(Q,F)(O,A)
\tkzGetPoints{Q}{T}
\tkzDrawSegments(Q,T T,M Q,M)
\tkzDrawPoints[fill=white](A,B,C,D,E,F,M,I,P,Q,T,O)
\tkzLabelPoints(M,C)
\tkzLabelPoints[above](Q, B)
\tkzLabelPoints[above left](I)
\tkzLabelPoints[left](E, A, T, O)
\tkzLabelPoints[below left](F, D, P)
\end{tikzpicture}
\end{center}
\begin{enumerate}
\item Ta có: $\widehat{OBM}=\widehat{OCM}=90^o$ (tính chất tiếp tuyến).\\
$\Rightarrow B,C,M,O$ cùng thuộc đường tròn đường kính $OM$\qquad (1).\\
Lại có: $ME\parallel AC\Rightarrow \widehat{BIM}=\widehat{BAC}$ (cặp góc đồng vị).\\
Mà $\widehat{BAC}=\widehat{BCM}$ (góc nội tiếp và góc tạo bởi tiếp tuyến và dây cùng chắn $\wideparen{BC}$).\\
$\Rightarrow \widehat{BIM}=\widehat{BCM}\Rightarrow$ tứ giác $BMCI$ nội tiếp.\\
$\Rightarrow B,C,I,M$ cùng thuộc một đường tròn\qquad (2).\\
Từ (1) và (2) $\Rightarrow B,M,O,I,C$ cùng thuộc đường tròn đường kính $OM$.
\item Theo a) thì tứ giác $BMCI$ nội tiếp nên $\Delta FBI\backsim\Delta FMC$ (g.g).\\
$\Rightarrow \dfrac{FI}{FC}=\dfrac{FB}{FM}\Leftrightarrow FI.FM=FB.FC$\qquad (3).\\
Lại dễ chứng minh được $\Delta FBE\backsim \Delta FDC$ (g.g)\\
$\Rightarrow \dfrac{FB}{FD}=\dfrac{FE}{FC}\Leftrightarrow FD.FE=FB.FC$\qquad (4).\\
Từ (3) và (4) $\Rightarrow FI.FM=FD.FE\Leftrightarrow \dfrac{FI}{FE}=\dfrac{FD}{FM}$.
\item Ta dễ chứng minh được $\Delta FBQ\backsim\Delta FTC$ (g.g)\\
$\Rightarrow \dfrac{FB}{FT}=\dfrac{FQ}{FC}\Leftrightarrow FT.FQ=FB.FC$\qquad (5).\\
Từ (3) và (5) $\Rightarrow FI.FM=FT.FQ\Leftrightarrow\dfrac{FI}{FT}=\dfrac{FQ}{FM}$.\\
Từ đó ta được $\Delta FIQ\backsim\Delta FTM$ (c.g.c).\\
$\Rightarrow \widehat{FIQ}=\widehat{FTM}=90^o\Rightarrow \Delta QTM$ vuông tại $T$.\\
$\Rightarrow MQ^2=TQ^2+TM^2\Leftrightarrow\dfrac{TQ^2+TM^2}{MQ^2}=1$. 
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Lê Hồng Phong Nam Định (Vòng 1), 2017]%[9D5G1]
Cho $a,b,c$ là các số thực thỏa mãn $a\ge -2$, $b\ge -2$ và $a+b+2c=6$. Chứng minh rằng
\begin{enumerate}
\item $a^2+b^2+4ab+16\ge4c^2-16c+20$.
\item $\dfrac{4-b^2}{4\left[(c-2)^2+1\right]}-\dfrac{a^2}{(a-b)^2+6ab+16}+5\ge 0$.
\end{enumerate}
\loigiai{ (Thực hiện bởi viet9a14124869 - VMF's Member)
\begin{enumerate}
\item Ta có $a+b=6-2c$ và $(a+2)(b+2)\geq 0$
\begin{eqnarray*}
&\Rightarrow & a^2+b^2+4ab+16=(a+b)^2+2(a+2)(b+2)-4(a+b)+8\\
&\geq & (a+b)^2-4(a+b)+8=(6-2c)^2-4(6-2c)+8=4c^2-16c+20.
\end{eqnarray*}
Đẳng thức xảy ra khi $a=-2$ hoặc $b=-2$.
\item Theo câu a) ta có $(a-b)^2+6ab+16\geq 4c^2-16c+20$. Khi đó
\begin{eqnarray*}
& A & =\dfrac{4-b^2}{4c^2-16c+20}-\dfrac{a^2}{(a-b)^2+6ab+16}+5\\
&& \geq \dfrac{4-b^2-a^2}{4c^2-16c+20}+5=\dfrac{4-(a+b)^2+2(a+2)(b+2)-4(a+b)-8}{4c^2-16c+20}+5\\
&& \geq \dfrac{4-(a+b)^2-4(a+b)-8}{4c^2-16c+20}+5=\dfrac{4-(6-2c)^2-4(6-2c)-8}{4c^2-16c+20}+5\\
&& =\dfrac{4(2c-3)^2}{4c^2-16c+20}\geq 0
\end{eqnarray*}
Dấu bằng xảy ra khi $(a,b,c)\in \left \{ \left(-2,5,\dfrac{3}{2}\right);\left(5,-2,\dfrac{3}{2}\right) \right \}$.
\end{enumerate}
}
\end{ex}
