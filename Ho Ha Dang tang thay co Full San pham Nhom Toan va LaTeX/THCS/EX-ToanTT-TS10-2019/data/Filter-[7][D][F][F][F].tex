		\tkzLabelPoints[above left](O)
		\tkzLabelPoints[above](A,B)
		\tkzDrawPoints[fill=black,size=1pt](A,B,O)
		\end{tikzpicture}}
	\dapso{$150$ mm}
	\loigiai{
	Khối lượng vật và khoảng cách đến $O$ là hai đại lượng tỉ lệ nghịch. Suy ra 
	\begin{eqnarray*}
		& & m_1\cdot OA = m_2 \cdot OB\\
		&\Leftrightarrow & OB = \dfrac{m_1\cdot OA}{m_2}\\
		&\Leftrightarrow & OB = \dfrac{V_1 d_1\cdot OA}{V_2 d_2}\\
		&\Leftrightarrow & OB = \dfrac{V_1}{V_2}\cdot \dfrac{d_1}{d_2} \cdot OA\\
		&\Leftrightarrow & OB = 3\cdot \dfrac{1}{2} \cdot 100 = 150.
	\end{eqnarray*}
	Vậy $OB=150$ mm.
