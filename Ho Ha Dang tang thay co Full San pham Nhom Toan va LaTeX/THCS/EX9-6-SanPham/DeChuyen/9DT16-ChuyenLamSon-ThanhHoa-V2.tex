\item Với $a \geq \dfrac{3}{8}$, chứng minh rằng $\sqrt[3]{3a-1+a\sqrt{8a-3}}+\sqrt[3]{3a-1-a\sqrt{8a-3}}=1$.

\begin{align*}
A^{3}&=6a-2+3A\sqrt[3]{(3a-1)^{2}-a^{2}(8a-3)}=6a-2+3A\sqrt[3]{-8a^{3}+12a^{2}-6a+1}\\

\Rightarrow &A^{3}-1=3(2a-1)(1-A)\Rightarrow (A-1)[A^{2}+A+1+3(2a-1)]=0
\end{align*}
Xét $A^{2}+A+1+3(2a-1)=A^{2}+A+6a-2\geq A^{2}+A+\dfrac{18}{8}-2>0\Rightarrow A=1$. 
\end{enumerate}

\begin{ex}%[Đề thi vào 10, Chuyên Lam Sơn - Vòng 2, 2016]%[9D4G7]%[9D5G5]
\hfill
\begin{enumerate}
\item Giải phương trình $\dfrac{4x}{x^{2}+x+3}+\dfrac{5x}{x^{2}-5x+3}=-\dfrac{3}{2}$.
\item Giải hệ phương trình $\heva{&x^{8}y^{8}+y^{4}=2x\\ &1+x=x(1+y)\sqrt{xy}}$
\end{enumerate}
\loigiai{
\begin{enumerate}
\item Nhận xét: $x=0$ không thỏa mãn phương trình. Ta có
\[
\dfrac{4x}{x^{2}+x+3}+\dfrac{5x}{x^{2}-5x+3}=-\dfrac{3}{2}\Leftrightarrow \dfrac{4}{x+\dfrac{3}{x}+1}+\dfrac{5}{x+\dfrac{3}{x}-5}=-\dfrac{3}{2}
\]
Đặt $t=x+\dfrac{3}{x}-2\Rightarrow x+\dfrac{3}{x}+1=t+3$ và $x+\dfrac{3}{x}-5=t-3$. Phương trình trở thành
\[
\dfrac{4}{t+3}+\dfrac{5}{t-3}=-\dfrac{3}{2}\Leftrightarrow t^{2}+6t-7=0\Leftrightarrow \hoac{&t=1\\ &t=-7}
\]
\begin{itemize}
\item Với $t=1\Rightarrow x^{2}-3x+3=0$ (vô nghiệm).
\item Với $t=-7\Rightarrow x^{2}+5x+3=0\Rightarrow x=\dfrac{-5\pm \sqrt{13}}{2}$.
\end{itemize}
Vậy phương trình có nghiệm $x=\dfrac{-5\pm \sqrt{13}}{2}$.
\item Điều kiện $xy\geq 0$.\\
$1+x=x(1+y)\sqrt{xy}\Rightarrow (1+x)^{2}=x^{2}(1+y)^{2}xy\Rightarrow (xy-1)(x^{2}y^{2}+2x^{2}y+xy+x^{2}+2x+1)=0$.\\
Vì $x^{8}y^{8}+x^{4}=2x\Rightarrow x\geq 0\Rightarrow y\geq 0\Rightarrow xy=1\Rightarrow x=\dfrac{1}{y}$. Ta có
\[
x^{8}y^{8}+x^{4}=2x\Leftrightarrow \dfrac{1}{y^{4}}+1=\dfrac{2}{y}\Leftrightarrow (y-1)(y^{4}+y^{3}+y^{2}+y+2)=0\Rightarrow y=1\Rightarrow x=1
\]
Vậy hệ phương trình có nghiệm $(x;y)=(1;1)$.
\end{enumerate}
}
\end{ex}
\begin{ex}%[Đề thi vào 10, Chuyên Lam Sơn - Vòng 2, 2016]%[9D5G3]%[9D5G2]
\hfill
\begin{enumerate}
\item Tìm nghiệm nguyên của phương trình $(x+1)(x+2)(x+8)(x+9)=y^{2}$.
\item Tìm các số tự nhiên $x$, $y$, $z$ thỏa mãn $2016^{x}+2017^{y}=2018^{z}$.
\end{enumerate}
\loigiai{
\begin{enumerate}
\item $(x+1)(x+2)(x+8)(x+9)=y^{2}\Leftrightarrow (x^{2}+10x+9)(x^{2}+10x+16)=y^{2}$.\\
Đặt $x^{2}+10x+\dfrac{25}{2}=a\Rightarrow a^{2}-\dfrac{49}{2}=y^{2}\Rightarrow (y-a)(y+a)=\dfrac{49}{4}\Rightarrow (2y-2a)(2y+2a)=49$\\
$\Rightarrow 2y-2a$ và $2y+2a$ thuộc ước của $49$. Ta xét các trường hợp:
\begin{itemize}
\item $\heva{&2y-2a=1\\ &2y+2a=49}\Leftrightarrow \heva{&2y=2a+1\\ &y=\dfrac{25}{2}}$ (vô lí).
\item $\heva{&2y-2a=-1\\ &2y+2a=-49}\Leftrightarrow \heva{&2y=2a-1\\ &y=-\dfrac{25}{2}}$ (vô lí).
\item $\heva{&2y-2a=49\\ &2y+2a=1}\Leftrightarrow \heva{&2y=2a+49\\ &y=\dfrac{25}{2}}$ (vô lí).
\item $\heva{&2y-2a=-49\\ &2y+2a=-1}\Leftrightarrow \heva{&2y=2a-49\\ &y=-\dfrac{25}{2}}$ (vô lí).
\item $\heva{&2y-2a=-7\\ &2y+2a=7}\Leftrightarrow \heva{&2y=2a-7\\ &y=0}$\\
Xét $2a-7=0\Leftrightarrow 2x^{2}+20x+18=0\Leftrightarrow x=-1$, $x=-9$ (thỏa mãn).
\item $\heva{&2y-2a=7\\ &2y+2a=-7}\Leftrightarrow \heva{&2y=2a+7\\ &y=0}$.\\
Xét $2a+7=0\Leftrightarrow 2x^{2}+20x+32=0\Leftrightarrow x=-2$, $x=-8$ (thỏa mãn).
\end{itemize}
Vậy các nghiệm nguyên của phương trình là $(x;y)\in\{(-1;0),(-9;0),(-2;0),(-8;0)\}$.
\item Xét $x\geq 1$: vì $2016^{x}\ \equiv\ 0(\mbox{mod }4)$ và $2017^{y}\ \equiv\ 1(\mbox{mod }4)$ nên $2018^{z}\ \equiv\ 1(\mbox{mod }4)$ (vô lí).\\
Xét $x=0$: vì $2016^{x}+2017^{y}=1+2017^{y}\ \equiv\ 2(\mbox{mod }4)\Rightarrow 2018^{z}\ \equiv\ 2(\mbox{mod }4)\Rightarrow z=1$.\\
Vậy các số tự nhiên cần tìm là $(x;y;z)=(0;1;1)$.
\end{enumerate}
}
\end{ex}

\begin{ex}%[Đề thi vào 10, Chuyên Lam Sơn - Vòng 2, 2016]%[9D5G2]
Một số tự nhiên gọi là \textit{số thú vị} nếu số đó có $10$ chữ số khác nhau và là bội của số $11111$. Hỏi có bao nhiêu số thú vị?
\loigiai{
Gọi $A$ là số thú vị $\Rightarrow$ tổng các chữ số của $A$ bằng $45\Rightarrow$ $A\ \vdots\ 99999$. Gọi $A=99999k$ (với $k$ là số tự nhiên) $\Rightarrow 10235\leq k\leq 98766$. Như vậy, ta đi tìm số $k$ sao cho khi nhân $k$ với $99999$ thì các chữ số từ $0$ đến $9$, mỗi số có mặt đúng một lần.\\
Mặt khác ta có kết quả: khi lấy $k=\overline{a_{1}a_{2}a_{3}a_{4}a_{5}}$ nhân với $99999$ thì ta thu được tích bằng giá trị của $k-1$ và viết thêm phần bù $9$ của $k-1$ vào sau số $k-1$.\\
Ví dụ: với $k=10235\Rightarrow k-1=10234\Rightarrow$ bù $9$ của $k-1$ là $89765\Rightarrow 10235.99999=1023489765$.\\
Chú ý rằng với $k=\overline{a_{1}a_{2}a_{3}a_{4}a_{5}}$ thì $k-1$ là số mà không tồn tại hai chữ số có tổng bằng $9$. Như vậy, bài toán được quy về việc đi tìm số các số tự nhiên $k$ thỏa mãn $10235\leq k\leq 98766$ và $k-1$ là số tự nhiên không tồn tại hai chữ số có tổng bằng $9$.\\
Vì $k-1$ là số tự nhiên không tồn tại hai chữ số có tổng bằng $9$ nên nếu chữ số $a_{i}$ xuất hiện trong $k-1$ thì chữ số $9-a_{i}$ không thể xuất hiện trong $k-1$, do vậy ta xét các cặp số $(0;9)$, $(1;8)$, $(2;7)$, $(3;6)$, $(4;5)$.\\
Ta lập số $k-1$ bằng cách duy nhất mỗi chữ số trong mỗi cặp (giả sử xét luôn chữ số $0$ là chữ số hàng chục nghìn) $\Rightarrow$ số lượng số $k-1$ là $2^{5}.5!=3840$. Vì số lượng các số có chữ số hàng chục nghìn tương ứng với $0$, $1$, $2$, .., $9$ là bằng nhau $\Rightarrow$ số lượng số có số $0$ đứng đầu là $384$ $\Rightarrow$ số lượng số $k-1$ cần tìm là $3840-384=3456$ số.\\
Vậy có $3456$ số thú vị.
}
\end{ex}
