\begin{bt}%[HK2 (2015-2016), THPT Chuyên Ams, Hà Nội]%[1D3K4-1]%[Phan Quốc Trí]
	Tìm số hạng đầu và công bội của cấp số nhân $(u_n)$ thỏa mãn
	$$\heva{&u_1 \cdot u_3 = \dfrac{1}{9}\\ & u_1+u_2+u_3 = \dfrac{1}{2}.}$$
	\loigiai{
		$\heva{&u_1 \cdot u_3 = \dfrac{1}{9}\\ & u_1+u_2+u_3 = \dfrac{1}{2}} \Leftrightarrow \heva{&u_2^2=\dfrac{1}{9}\\& u_1(1+q+q^2)=\dfrac{1}{2}}\Leftrightarrow \heva{&u_2= \pm \dfrac{1}{3}\\& u_1(1+q+q^2)=\dfrac{1}{2}}
		$
		\begin{itemize}
			\item[+)] Với $u_2=-\dfrac{1}{3}$, ta có $\heva{&u_1q= - \dfrac{1}{3}\\& u_1(1+q+q^2)=\dfrac{1}{2}} \Leftrightarrow \heva{&u_1q=-\dfrac{1}{3}\\& \hoac{&q=-2\\& q=-\dfrac{1}{2}}} \Leftrightarrow \hoac{&q=-2,u_1=\dfrac{1}{6}\\& q=-\dfrac{1}{2},u_1=\dfrac{2}{3}}$
			\item[+)] Với $u_2=\dfrac{1}{3}$, ta có $\heva{&u_1q=  \dfrac{1}{3}\\& u_1(1+q+q^2)=\dfrac{1}{2}}$\\
			Chia theo vế hai phương trình trên ta có: $\dfrac{1+q+q^2}{q}=\dfrac{3}{2} \Leftrightarrow 2q^2-q+2=0$ (vô nghiệm).
		\end{itemize}
		Vậy $q=-2,u_1=\dfrac{1}{6}$ hoặc $q=-\dfrac{1}{2},u_1=\dfrac{2}{3}$.
	}
\end{bt}
\begin{bt}%[HK2 (2015-2016), THPT Chuyên Ams, Hà Nội]%[1D4B2-7][1D4B2-5]%[Phan Quốc Trí]
	Tìm các giới hạn sau
	\begin{multicols}{2}
		\begin{description}
			\item[a)] $\lim \limits_{x \to +\infty} \dfrac{\sqrt{x^2+1}+x}{5-2x}$;
			\item[b)] $\lim \limits_{x \to 0^+} \dfrac{\sqrt{x^2+x+1}-\sqrt{x^2-x+1}}{x^2}$.
		\end{description}
	\end{multicols}
	\loigiai{
		\begin{enumerate}[a)]
			\item  $\lim \limits_{x \to +\infty} \dfrac{\sqrt{x^2+1}+x}{5-2x} =\lim \limits_{x \to +\infty} \dfrac{\sqrt{1+\frac{1}{x^2}}+1}{\frac{5}{x}-2}=\dfrac{1+1}{0-2}=-1 $.
			\item $\lim \limits_{x \to 0^+} \dfrac{\sqrt{x^2+x+1}-\sqrt{x^2-x+1}}{x^2} = \lim \limits_{x \to 0^+}\dfrac{2}{x \left(\sqrt{x^2+x+1}+\sqrt{x^2-x+1} \right) } $\\
			$$ = \lim \limits_{x \to 0^+} \dfrac{\dfrac{2}{\sqrt{x^2+x+1}+\sqrt{x^2-x+1}}}{x} $$
			Ta có 
			\begin{itemize}
				\item[+)] $\lim \limits_{x \to 0^+} \dfrac{2}{\sqrt{x^2+x+1}+\sqrt{x^2-x+1}} = 1 $.
				\item[+)] $\lim \limits_{x \to 0^+} x = 0 $
				\item[+)] $x>0$
			\end{itemize}
			nên $\lim \limits_{x \to 0^+} \dfrac{\dfrac{2}{\sqrt{x^2+x+1}+\sqrt{x^2-x+1}}}{x} = + \infty$.
		\end{enumerate}
		
	}
\end{bt}

		\item Cho hàm số $y=\dfrac{3x+5}{x+3}$ có đồ thị $(C)$. Viết phương trình tiếp tuyến của $(C)$, biết tiếp tuyến này vuông góc với đường thẳng $(d): 4x+y-3=0$.
	\end{enumerate}
	\loigiai{
		\begin{enumerate}[a)]
			\item 
			\begin{itemize}
				\item[+)] Khi $x \ne 1$ thì $f(x)=a$ là hàm số liên tục trên $\mathbb{R}$ nên liên tục trên $(-\infty; 1) \cup (1;+\infty )$.
				\item[+)] Tại $x =1$, ta có $f(1)=a$\\
				$\lim \limits_{x \to 1} f(x) = \lim \limits_{x \to 1} \left( \dfrac{1}{1-x}-\dfrac{3}{1-x^3} \right) =\lim \limits_{x \to 1}  \dfrac{x^2+x-2}{1-x^2} = \lim \limits_{x \to 1}  \dfrac{x+2}{-1-x}=-\dfrac{3}{2}   $
			\end{itemize}
			Hàm số liên tục trên $\mathbb{R}$ thì hàm số phải liên tục tại  $x =1 \Leftrightarrow \lim \limits_{x \to 1} f(x) = f(1) \Leftrightarrow a = -\dfrac{3}{2}$.
			\item  $y'=\dfrac{4}{(x+3)^2}$.\\ Tiếp tuyến vuông góc với đường thẳng $d$ nên có hệ số góc bằng $\dfrac{1}{4}$.\\

			+) $x=1 \Rightarrow y = 2$. Ta có tiếp tuyến $y=\dfrac{1}{4}(x-1)+2 \Leftrightarrow y = \dfrac{1}{4}x+\dfrac{7}{4}$.\\
			+) $x=-7 \Rightarrow y = 4$. Ta có tiếp tuyến $y=\dfrac{1}{4} (x+7)+4 \Leftrightarrow y = \dfrac{1}{4}x+\dfrac{23}{4}$.

\begin{bt}%[HK2 (2015-2016), THPT Chuyên Ams, Hà Nội]%[1H3K4-2][1H3K5-3]%[Phan Quốc Trí]
	Cho hình chóp tứ giác $S.ABCD$ có đáy $ABCD$ là hình thoi tâm $O$, cạnh $a$, $\widehat{ABC} = 60^{\circ}$. Cho biết $SO \perp (ABCD)$ và góc giữa đường thẳng $SC$ và mặt phẳng $(ABCD)$ bằng $60^{\circ}$.
	\begin{enumerate}[a)]
		\item Chứng minh: $(SAC) \perp (SBD)$ và $\Delta SAC$ là tam giác đều.
		\item Tính khoảng cách từ điểm $O$ đến mặt phẳng $(SCD)$ theo $a$.
		\item Lấy các điểm $E,F$ lần lượt trên các cạnh $SB$, $SD$ sao cho $\dfrac{SE}{SB}=\dfrac{SF}{SD}=\dfrac{2}{3}$. Chứng minh $EF\perp (SAC)$, $SC \perp (AEF)$.
		\item Gọi $I$ là giao điểm của $SC$ và mặt phẳng $(AEF)$. Tính diện tích tứ giác $AEIF$.
	\end{enumerate}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=1, line width=1.2pt]%1
			\tkzDefPoints{0/0/A, 7/0/D, -2.5/-1.5/B, 0/7/E}
			\tkzDefPointBy[translation=from A to D](B) \tkzGetPoint{C}
			\tkzDefMidPoint(A,C)\tkzGetPoint{O}
			\tkzDefPointBy[translation=from A to E](O) \tkzGetPoint{S}
			\tkzDefMidPoint(D,C)\tkzGetPoint{M}
			\tkzDefMidPoint(M,C)\tkzGetPoint{N}
			\tkzDefMidPoint(S,C)\tkzGetPoint{I}
			\tkzDefPointBy[homothety=center S ratio 0.7](N) \tkzGetPoint{H}
			\tkzInterLL(S,O)(A,I)\tkzGetPoint{G}
			\tkzDefLine[parallel=through G](B,D) \tkzGetPoint{g}
			\tkzInterLL(G,g)(S,D)\tkzGetPoint{F}
			\tkzInterLL(G,g)(S,B)\tkzGetPoint{E}
			\tkzDrawSegments(B,C S,B S,C S,D C,D S,N E,I I,F)
			\tkzDrawSegments[dashed](A,D A,C B,D  A,B S,A S,O O,N A,M O,H A,I E,F A,E A,F)
			\tkzLabelPoints(C,B,D,M,N,I,G)
			
			\tkzLabelPoints[above](S)
			\tkzLabelPoints[below](O,A)
			\tkzLabelPoints[right](F,H)
			\tkzLabelPoints[left](E)
			\tkzMarkRightAngle[opacity=0.5](A,M,C)
			\tkzMarkRightAngle[opacity=0.5](O,H,N)
			\tkzMarkAngle[fill=black,size=0.7, opacity=0.25](C,B,A)
			\tkzLabelAngle[pos=1](A,B,C){$60^{\circ}$}
			\tkzMarkAngle[fill=black,size=0.3, opacity=0.25](S,C,A)
			\tkzLabelAngle[pos=0.5](S,C,A){$60^{\circ}$}
			\end{tikzpicture}
		\end{center}
		\begin{enumerate}[a)]
			\item Ta có $\heva{&AC \perp SO\\&AC \perp BD} \Rightarrow AC \perp (SBD) \Rightarrow (SAC) \perp (SBD)$.\\
			Ta có $SO \perp (ABCD)$ nên $\left( SC,(ABCD) \right) =\widehat{SCO} = 60^{\circ}$ .\\
			$\heva{&\Delta SAC \text{ cân tại S}\\ & \widehat{SCA}=60^{\circ}} \Rightarrow SAC $ là tam giác đều.
			\item Gọi $M,N$ lần lượt là trung điểm $CD$ và $CM$. Kẻ $OH \perp SN$, ta có \\
			$\Delta ACD$ đều $\Rightarrow AM \perp CD \Rightarrow ON \perp CD$.\\
			$\heva{&CD \perp SO\\&CD \perp ON} \Rightarrow CD \perp (SON) \Rightarrow CD \perp OH$.\\
			$\heva{&OH\perp SN\\&OH \perp CD} \Rightarrow OH \perp (SCD) \Rightarrow d\left(O; (SCD) \right) = OH$.\\
			Ta có $ON=\dfrac{1}{2} AM = \dfrac{a\sqrt{3}}{4}$ và $SO= OC \cdot \tan 60^0 = \dfrac{a\sqrt{3}}{2}$.\\
			$\dfrac{1}{OH^2}=\dfrac{1}{OS^2}+\dfrac{1}{OC^2}\Rightarrow OH =  \dfrac{a\sqrt{15}}{10}$.
			\item Gọi $G=SO \cap EF, I=AG \cap SC$.\\
			$\heva{&BD\perp AC\\& BD \perp SO} \Rightarrow BD \perp (SAC)$.\\
			$\dfrac{SE}{SB}=\dfrac{SF}{SD}\Rightarrow EF \parallel BD \Rightarrow EF \perp (SAC) \Rightarrow EF \perp SC$.\\
			$\dfrac{SG}{SO}=\dfrac{SE}{SB}=\dfrac{2}{3}$ nên $G$ là trọng tâm $\Delta SAC \Rightarrow AI \perp SC\Rightarrow SC \perp (AEF)$.
			\item Ta có $AI=\dfrac{a\sqrt{3}}{2}, BD=2BO=a\sqrt{3}$ và $EF=\dfrac{2}{3} BD = \dfrac{2a\sqrt{3}}{3}$\\
			$EF \perp (SAC) \Rightarrow EF \perp AI \Rightarrow S_{AEIF}=\dfrac{1}{2}EF \cdot AI = \dfrac{1}{2}\cdot \dfrac{2a\sqrt{3}}{3}\cdot \dfrac{a\sqrt{3}}{2}=\dfrac{a^2}{2}$.
		\end{enumerate}
	}

\begin{bt}%[HK2 (2015-2016), THPT Chuyên Ams, Hà Nội]%[1D4G3]%[Phan Quốc Trí]
	Cho $2a+3b+6c=0$. Chứng minh rằng phương trình $a\cdot \tan^2 x + b \cdot \tan x + c = 0$ luôn có nghiệm thuộc khoảng $\left( 0; \dfrac{\pi}{4} \right)$.
	\loigiai{
		Đặt $t=\tan x \Rightarrow t \in (0;1)$. Phương trình trở thành: $a\cdot t^2 +b \cdot t +c=0 \quad (*)$.\\
		Ta cần chứng minh phương trình $(*)$ có nghiệm trên $(0;1)$.\\
		Xét hàm số $f(t)=at^2+bt+c$ trên $[0;1]$ ta có:
		\begin{itemize}
			\item[+] $f(x)$ liên tục trên $[0;1]$.
			\item[+] $f(0)=c, \; f\left( \dfrac{2}{3} \right) = \dfrac{4a}{9}+\dfrac{2b}{3}+c=\dfrac{2(2a+3b+6c)-3c}{9}=-\dfrac{c}{3}$.\\
			$\Rightarrow f(0)\cdot f \left( \dfrac{2}{3} \right)= -\dfrac{c^2}{3}$.
		\end{itemize}
		\begin{itemize}
			\item[-] Nếu $c=0$ thì $(*)$ có nghiệm $\hoac{&t=0\\&t=-\dfrac{b}{a}.}$ \\
			Vì $2a+3b=0 \Rightarrow -\dfrac{b}{a}=\dfrac{2}{3}\in (0;1)$.\\
			\item[-] Nếu $c \ne 0 $ thì $f(0) f\left( \dfrac{2}{3} \right) <0 \Rightarrow $ phương trình (*) có nghiệm $t \in \left( 0; \dfrac{2}{3} \right) \subset (0;1)$. 
		\end{itemize}
		Vậy phương trình (*) luôn có nghiệm thuộc $(0;1)$.\\
		Hay phương trình $a\cdot \tan^2 x + b \cdot \tan x + c = 0$ luôn có nghiệm thuộc khoảng $\left( 0; \dfrac{\pi}{4} \right)$.
	}
\end{bt}
